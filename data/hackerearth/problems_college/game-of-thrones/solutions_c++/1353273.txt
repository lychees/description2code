#include<bits/stdc++.h>
using namespace std;
inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}
#define mp make_pair
int main(){
    int t=scan();
    while(t--){
        unordered_map<int,bool> m;
        int n=scan();
        int ans=0;
        while(n--){
            int temp=scan();
            if(m[temp]!=1){
                m[temp]=1;
            }
            else{
                ans++;
            }
        }
        printf("%d\n",ans);
    }
}
