import math
n=input()
for i in range(0,n):
	values=[int(x) for x in raw_input().split()]
	a=values[0]
	b=values[1]
	c=values[2]-values[3]
	if b**2-4*a*c<0:
		print 0
		continue
	result1=max((-b+math.pow((b**2)-4*a*c,0.5))/(2*a),(-b-math.pow((b**2)-4*a*c,0.5))/(2*a))
	final=int(math.ceil(result1))
	if final<0:
		print 0
	else:
		print final
