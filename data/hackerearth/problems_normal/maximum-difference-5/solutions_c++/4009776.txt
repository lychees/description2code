#include <iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main()
{
   vector<int> va,vb,vc;
   int val,n,i,j;
   int diff=0,sum=0,diff1=0,sum1=0;
   cin>>n;
   for(i=0;i<n;i++)
   {
   	cin>>val;
   	va.push_back(val);
   }
   sort(va.begin(),va.end());
   for(i=0,j=n-1;i<=j;i++,j--)
   {
   	if(i==j){ vb.push_back(va[i]);  vc.push_back(va[i]);  }
   	else
   	{
     	vb.push_back(va[i]); vb.push_back(va[j]);	
    	vc.push_back(va[j]); vc.push_back(va[i]);	
   	}
   }
   for(i=1;i<n;i++)
   {
    diff=vb[i]-vb[i-1]; 
    if(diff<0) diff=diff*-1;
    sum+=diff;
    diff1=vc[i]-vc[i-1];
    if(diff1<0) diff1=diff1*-1;
    sum1+=diff1;
   }
   if(sum1<sum) cout<<sum<<endl;
   else cout<<sum1<<endl;
   
    return 0;
}
