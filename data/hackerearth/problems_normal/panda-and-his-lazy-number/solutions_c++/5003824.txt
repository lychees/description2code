#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--){
    	long long int x,y;
    	cin>>x>>y;
    	long long int i=sqrt(x);
    	long long int yy=i*i,sum=0;
    	if(yy<x){i++;yy=i*i;}
    	while(yy<=y){
    		sum++;
    		i++;
    		yy=i*i;
    		
    	}
    	cout<<sum<<endl;
    }
    
    return 0;
}
