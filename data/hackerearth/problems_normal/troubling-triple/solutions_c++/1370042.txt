#include<vector>
#include<algorithm>
#include <iostream>
using namespace std;

int main()
{
    int n,k;
    cin>>n>>k;
    vector<int> a;
    for(int i=0;i<n;i++)
   	{
   		int x;
   		cin>>x;
   		a.push_back(x);
   	}
   	sort(a.begin(),a.end());
   	long long int count=0;
   	for(int i=0;i<n;i++)
   	{
   		for(int j=i+1;j<n;j++)
   		{
   			int temp=a[i]*a[j];
   			int find=k/temp;
   			 std::vector<int>::iterator up;
   			 up=upper_bound(a.begin()+j+1,a.end(),find);
   			 int pos=up-a.begin()-j-1;
   			count+=max(pos,0);
   		}
   	}
   	cout<<count<<endl;
    return 0;
}
