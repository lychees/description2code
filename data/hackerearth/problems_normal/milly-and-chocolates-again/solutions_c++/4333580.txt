#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int arr[1000004];
    int ind = 1;
    for(int i = 1; i <= n; i++ ){
    	int val;
    	cin>>val;
    	for(int k = 1; k <= val; k++){
    		arr[ind] = i;
    		ind++;
    	}
    }
    int q;
    cin>>q;
    while(q--){
    	int x;
    	cin>>x;
    	cout<<arr[x]<<"\n";
    }
    return 0;
    
}
