#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
int main()
{
    vector<int> v;
    int n,q;
    unsigned long int m,key;
    cin>>n>>q;
    for(int i=0;i<n;i++)

    {
        cin>>m;
        v.push_back(m);
    }
    sort(v.begin(),v.end());
    while(q--)
    {
        cin>>key;
        if(binary_search(v.begin(),v.end(),key))
            cout<<"YES\n";
        else
            cout<<"NO\n";
    }
    return 0;
}
