#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int x;
    	scanf("%d",&x);
    	long int n=x+1;
    	int flag=0;
    	if(x==0)
    	{
    		cout<<4<<endl;
    	}
    	else if(x==1)
    	{
    		cout<<4<<endl;
    	}
    	else{
    	while(1)
    	{
    		for(int i=2;i<=sqrt(n);i++)
    		{
    			if(n%i==0)
    			{
    				flag=1;
    				break;
    			}
    		}
    		if(flag==1)
    		{
    			break;
    		}
    		else
    		{
    			n++;
    		}
    	}
    	printf("%lld\n",n);
    	}
    }
}
