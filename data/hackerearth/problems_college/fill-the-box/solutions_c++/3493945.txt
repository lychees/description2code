#include <iostream>
using namespace std;

int main()
{
    long int n,m,a,b,k,d[1000000]={0},r,s;
    cin>>n>>m;
    for(int i=0;i<m;i++)
    {
    	cin>>a>>b>>k;
        for(int j=a-1;j<=b-1;j++)
        d[j]+=k;
        
    }
    s=0;
    for(int i=0;i<n;i++)
    s+=d[i];
    r=s/n;
    cout<<r;
    return 0;
}
