#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    vector<int> v(100000);
    int i,t,n,flag;
    ll sum1,sum2,sum;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	sum=0;
    	flag=0;
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&(v[i]));
  			sum+=v[i];
    	}
    	if(n>2)
    	{
    		sum1=0;
    		for(i=1;i<n-1;i++)
    		{
    			sum1+=v[i-1];
    			sum2=sum-sum1-v[i];
    			if(sum1==sum2)
    			{
    				flag=1;
    				break;
    			}
    		}
    	}
    	if(flag)
    		printf("YES\n");
    	else
    		printf("NO\n");
    }
    return 0;
}
