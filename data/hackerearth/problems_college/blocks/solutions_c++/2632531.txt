using namespace std;
int power(long long int num,int p)
{
		  if(p == 0) return 1;
		  if(p == 1) return num;
		 
			long long int number = num;
		  for(int i=2;i<=p;i++) {
		    num *= number;
		    num %= MOD;
		  }
		  return num;
		}
int main()
{
 long long int t,N,M;
 long long int T[1001];
		  long long int S[1001];
		  long long int P[1001];
		 
		  T[0] = T[1] = 1;
		  T[2] = 2 ;
		  T[3] =  4;
		  T[4] = 8 ;
		 
		  P[0] = P[1] = 1;
		 
		  
		 
		  for(int i=5;i<=1000;i++)
		    T[i] = (T[i-1] + T[i-2] + T[i-3] + T[i-4])%MOD;
		 
		  S[0] = 1;
		  S[1] = 1;
		 
		  long long int sum;
		  
 sll(t);
 while(t--)
 {
 	sll2(N,M);
 	for(int i=0;i<=M;i++)
		      P[i] =  power(T[i],N);
		 
		    for(int i=2;i<=M;i++) {
		      sum = 0;
		      for(int j=1;j<i;j++) {
			sum += (S[j]*P[i-j])%MOD;
			sum %= MOD;
		      }
		      S[i] = (P[i] - sum);
		      S[i] = S[i]%MOD;
		    }
		    while(S[M] < 0)
		      S[M] += MOD;
		    pll(S[M]);
		    nl;
		  }
 return 0;
}
