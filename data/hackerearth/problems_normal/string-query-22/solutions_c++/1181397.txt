#include<bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,k,q;
    char str[200001],ch;
    
    scanf(" %s",str);
    
    map<char,vector<int>>mymap;
    map<char,vector<int>>::iterator it;
    
    i=0;
    while(str[i]!='\0')
    {
                       ch=str[i];
                       
                       if(mymap.count(ch))
                       {
                       mymap[ch].push_back(i);
                       }
                       else
                       {
                       	//it=mymap.find(ch);
                        //it->second.push_back(i);
                        //it->first=ch;
                       
                        mymap.insert(pair<char,vector<int>>(ch,{i}));
                        
                        //mymap.emplace(ch,push_back(i));
                        
                       }
                       
                       i++;
    }
    
    scanf("%d",&q);
    
    while(q--)
    {
              scanf("%d %c",&k,&ch);
              
              it=mymap.find(ch);
              j=it->second[k-1];
              
              it->second.erase(it->second.begin()+k-1);
              
              
              
              str[j]='`';
              
    }
    
    i=0;
    
    while(str[i]!='\0')
    {
                       if(str[i]!='`')
                       printf("%c",str[i]);
                       
                       i++;
    }
    
    
    return 0;
}
