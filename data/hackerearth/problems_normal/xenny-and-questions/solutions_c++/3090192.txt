#include<bits/stdc++.h>
using namespace std;
int arr[100002];
int main()
{
	ios_base::sync_with_stdio(false);
	int t,n,k,temp;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		memset(arr,0,sizeof(arr));
		int i=0;
		int max=0;
		while(i<n)
		{
			cin>>temp;arr[temp]++;
			temp>max ? max=temp:1;
			i++;
		}
		int count=0;
		unsigned long long total=0;
		i=max;
		while(count<k&&i>=1)
		{
			if(arr[i])
			{
				if(count+arr[i]<k)
				{
					count+=arr[i];total+=(arr[i]*i);
				}
				else
				{
					total+=((k-count)*i);count=k;
				}
			}
			i--;
		}
		cout<<total<<endl;
	}
	return 0;
}
