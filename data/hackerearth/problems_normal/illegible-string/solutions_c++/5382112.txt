#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    long long n,i,mn,mx;
    string s;
    cin>>n;
    cin>>s;i=mx=mn=0;
    while(i<n){
    	if(s[i]=='w')mx++;
    	i++;
    }
    mx+=n;i=0;
    while(i<n){
    	if(s[i]=='v'&&s[i+1]=='v'){mn++;
i+=2;}   
    	else if(s[i]=='v'&&s[i+1]=='w'){
    		while(s[i+1]=='w')i++;
    	if(s[i+1]=='v')	{
    		mn++;i+=2;
    	}
    	}
    	else i++;
    }
    cout<<n-mn<<" "<<mx<<endl;
    return 0;
}
