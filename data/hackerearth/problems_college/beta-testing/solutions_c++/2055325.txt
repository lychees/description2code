#include<iostream>
#include<vector>
#include<map>
#include<string.h>
#include<algorithm>
using namespace std;
int main()
{
                int n,i,c;
                string str;
                cin>>n;
                map<string,int>m;
                vector<pair<int,string> >v;

                while(n--)
                {
                   cin>>str;
                   m[str]++;
                }
                map<string,int>::iterator it;
                for(it=m.begin();it!=m.end();it++)
                {
                    v.push_back(make_pair(it->second,it->first));
                }
                sort(v.rbegin(),v.rend());
                for(i=0;i<v.size();i++)
                    cout<<v[i].second<<" "<<v[i].first<<endl;
                return 0;
}
