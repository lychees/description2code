#include <iostream>
using namespace std;

long fibo[1000000];
long sum[1000000];
int main() 
{
	int m;
	cin>>m;
	
	fibo[0]=2; fibo[1]=3;int k=2;
	
	sum[0]=2;sum[1]=5;
	for(int p=1;p<=m;p++)
	{
		int n;
		cin>>n;
		if(n>k)
		{
			for(int i=k;i<n;i++)
			{
				fibo[i]=fibo[i-1]+fibo[i-2];
				if(fibo[i]>1000000007)
				{
					fibo[i]=fibo[i]%1000000007;
				}
				sum[i]=sum[i-1]+fibo[i];
				if(sum[i]>1000000007)
				sum[i]=sum[i]%1000000007;
			}
			cout<< sum[n-1]<<endl; 
			k=n;
		}
		else
		{
			cout<< sum[n-1]<<endl;
		}
		
	}
	return 0;
}
