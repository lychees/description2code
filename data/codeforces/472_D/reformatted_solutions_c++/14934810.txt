#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <map>
#include <set>
#include <string>

using namespace std;
const int maxn = 2005;

int n, a[maxn][maxn];
set<int> ms;
set<int>::iterator it;

int main() {
  int i, j;
  scanf("%d", &n);
  for (i = 1; i <= n; i++)
    for (j = 1; j <= n; j++)
      scanf("%d", &a[i][j]);
  bool flag = 1;
  for (i = 1; i <= n; i++) {
    for (j = i + 1; j <= n; j++) {
      if (a[i][j] <= 0 || a[i][j] != a[j][i])
        flag = 0;
    }
    if (a[i][i] != 0)
      flag = 0;
  }
  for (i = 1; i <= n; i++)
    ms.insert(a[1][i]);
  for (i = 2; i <= n; i++) {
    for (j = 2; j < i; j++) {
      int t = (a[1][i] + a[1][j] - a[i][j]);
      if (t & 1 || ms.find(t / 2) == ms.end())
        flag = 0;
    }
  }
  if (flag)
    puts("YES");
  else
    puts("NO");
  return 0;
}
