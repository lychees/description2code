#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    int a[105][105];
    int dp[105][105][505];
    cin>>t;
    while(t--)
    {
        int n,m,p;
        cin>>n>>m>>p;
        memset(dp,0,sizeof(dp));
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                cin>>a[i][j];
            }
        }
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                for(int k=a[i][j];k<=p;k++)
                {
                    if(i==0&&j==0)
                        dp[i][j][a[i][j]]=1;
                    else if(i==0)
                        dp[i][j][k]=dp[i][j][k]||dp[i][j-1][k-a[i][j]];
                    else if(j==0)
                        dp[i][j][k]=dp[i][j][k]||dp[i-1][j][k-a[i][j]];
                    else 
                    {
                        dp[i][j][k]=dp[i][j][k]||dp[i][j-1][k-a[i][j]];
                        dp[i][j][k]=dp[i][j][k]||dp[i-1][j-1][k-a[i][j]];
                        dp[i][j][k]=dp[i][j][k]||dp[i-1][j][k-a[i][j]];
                    }
                }
            }
        }
        int ans=0;
        for(ans=p;ans>=0;ans--)
        {
            if(dp[n-1][m-1][ans])
                break;
        }
        cout<<ans<<endl;
    }
}
