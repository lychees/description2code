#include <stdio.h>
#include <vector>
#define MAX 10009
using namespace std;
int visit[MAX];
int ans, k;
vector <vector < int> > graph(MAX);
int dfs(int node)
{
	visit[node] = 1;
	int s = 1;	//present node
	for(int i = 0; i < (int) graph[node].size(); i++)
	{
		int present = graph[node][i];
		if(visit[present] == 0)
		{
			s += dfs(present);
		}
	}
	if(s >= k)
	{
		ans++;
		s = 0;
	}
	return s;
}
int main()
{
	int n, ip;
	scanf("%d%d", &k, &n);
	for(int i = 0; i <= n; i++)
		graph[i].clear(), visit[i] = 0;
	for(int i = 1; i < n; i++)
	{
		scanf("%d", &ip);
		ip--;
		graph[ip].push_back(i);
		graph[i].push_back(ip);
	}
	ans = 0;
	dfs(1);
	printf("%d\n", ans);
}
