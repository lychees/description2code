#include <iostream>
using namespace std;

int xorCount(int x){
	if(x%4==0){
		return x;
	}else if(x%4==1){
		return 1;
	}else if(x%4==2){
		return x+1;
	}
	return 0;
}

int main()
{
    int t;
    long long int l,r,c;
    scanf("%d",&t);
    while(t--){
    	scanf("%lld%lld",&l,&r);
    	c = xorCount(r) ^ xorCount(l-1);
    	printf("%lld\n",c);
    }
    return 0;
}
