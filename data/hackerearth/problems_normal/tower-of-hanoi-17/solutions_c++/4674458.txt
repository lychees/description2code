#include <iostream>
#include<bits/stdc++.h>
#define LL long long  
using namespace std;
int cmp(pair<LL int,LL int> x,pair<LL int,LL int> y){
	return x.first>y.first;
}


int main()
{
    int t;
    cin>>t;
    //cout<<t<<"ct"<<endl;
    while(t){
    	int n;
    	cin>>n;
    //	cout<<n<<"cc"<<endl;
    	LL int h[n];
    	vector<pair<LL int,LL int> > a;
    	for(int i=0;i<n;i++){
    		LL int x,y;
    		cin >> x >> y;
    		a.push_back(make_pair(x,y));
    	//	a.push_back(make_pair(y,x));
    	}
    	
    	sort(a.begin(),a.end(),cmp);
    	
    		
    	for(int i=0;i<n;i++){
    		h[i]=a[i].second;
    		//cout<<a[i].first<<"-> "<<a[i].second<<endl;
    	}
    	
    	for(int i=0;i<n;i++){
    		
    		for(int j=0;j<i;j++){
    			if(a[i].first<a[j].first && a[i].second<a[j].second){
    				h[i]=max(h[i],h[j]+a[i].second);
    				//cout<<a[i].first<<a[i].second<<"cng "<<a[j].first<<a[j].second<<endl;
    			}
    		}
    	}
    	
    		LL int ans=0;
    	for(int i=0;i<n;i++){
    		ans=max(ans,h[i]);
    		//cout<<h[i]<<endl;
    	}
    	
    	
    	cout<<ans<<endl;
    	t--;
    }
    
    return 0;
}
