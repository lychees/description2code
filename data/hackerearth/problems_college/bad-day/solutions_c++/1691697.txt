#include <iostream>
using namespace std;
#define maxi 1000001
long long arr[maxi]={0};
void fibo()
{
   arr[0]=0;
    arr[1]=1;
    arr[2]=2;
long long i ;
    for(i=3;i<=maxi;i++){
         arr[i]=(arr[i-1] + arr[i-2]);
    }
}
int main()
{
    //cout << "Hello World!" << endl;
    fibo();
    long long t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
          cout<<arr[n]<<endl;
    }
    return 0;
}
