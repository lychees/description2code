#include <bits/stdc++.h>

using namespace std;
void min_heapify(int *a,int i,int n)
{
    int j, temp;
    temp = a[i];
    j = 2 * i;
    while (j <= n)
    {
        if (j < n && a[j+1] < a[j])
            j = j + 1;
        if (temp < a[j])
            break;
        else if (temp >= a[j])
        {
            a[j/2] = a[j];
            j = 2 * j;
        }
    }
    a[j/2] = temp;
    return;
}
void build_minheap(int *a, int n)
{
    int i;
    for(i = n/2; i >= 1; i--)
    {
        min_heapify(a,i,n);
    }
}
int main()
{
    int n, i=-1, x;
    string str;
    int a[102];
    getline(cin,str);
    stringstream ss;
    ss.str(str);
    while(ss>>n)
    {
    	a[++i]=n;
    }
    build_minheap(a, i);
    cout<<a[1]<<endl;
    return 0;
}
