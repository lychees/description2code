#include<iostream>
using namespace std;
int dp[155][155][55];
bool visited[155][155][55];
int tot=0;
int func(int curr,int n,int five,int ten)
{
    if(curr==n)
        return 0;
    int ret=1e9;
    if(visited[curr][five][ten]==true)
        return dp[curr][five][ten];
    visited[curr][five][ten]=true;
    int one=tot-(five*5+ten*10)-curr*8;
    if(one>=8)
        ret=min(ret,8+func(curr+1,n,five,ten));
    if(five>=1 && one>=3)
        ret=min(ret,4+func(curr+1,n,five-1,ten));
    if(five>=2)
        ret=min(ret,2+func(curr+1,n,five-2,ten));
    if(ten>0)
        ret=min(ret,1+func(curr+1,n,five,ten-1));
    if(ten>0 && one>=3)
        ret=min(ret,4+func(curr+1,n,five+1,ten-1));

    dp[curr][five][ten]=ret;
    return dp[curr][five][ten];
}
int main(void)
{
    int t;
    cin>>t;
    while(t--)
    {
        for(int i=0;i<155;++i)
            for(int j=0;j<155;++j)
                for(int k=0;k<55;++k)
                    visited[i][j][k]=false;
        int n;
        cin>>n;
        int one,five,ten;
        cin>>one>>five>>ten;
        tot=one+five*5+ten*10;
        if(tot<n*8)
            cout<<"-1\n";
        else
            cout<<func(0,n,five,ten)<<endl;
    }
    return 0;
}
