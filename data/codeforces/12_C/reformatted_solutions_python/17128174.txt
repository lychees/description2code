def I():
    return map(int, raw_input().split())


n, m = I()
a = I()
f = {}
for _ in xrange(m):
    s = raw_input()
    f[s] = f.get(s, 0) + 1
a.sort()
b = sorted((f.values()))
i = sum([(a[x]*b[len(f)-x-1]) for x in xrange(len(f))])
j = sum([(a[len(a)-x-1]*b[len(f)-x-1]) for x in xrange(len(f))])
print i, j
