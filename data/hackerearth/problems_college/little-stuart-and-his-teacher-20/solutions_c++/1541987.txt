#include <iostream>
#include <bits/stdc++.h>
using namespace std;

const int MAX = 1000001;
long long int phi[MAX];

long long ans[MAX];
void euler_phi() {
    phi[1] = 1;
    for(int i=2; i<MAX; i++) {
        if(!phi[i]) {
            phi[i] = i-1;
            for(int j=(i<<1); j<MAX; j+=i) {
                if(!phi[j]) phi[j] = j;
                phi[j] = phi[j]/i*(i-1);
            }
        }
    }
}

void compute()
{
		ans[1] = ans[2] = 0;
		for ( int i = 3; i < MAX; i++) {
			ans[i] = ans[i-1] + ( (i - 1 - phi[i]) >= 0 ?i - 1 - phi[i] : 0);
		}
}
 

int main() 
{
	euler_phi();
	compute();
	int t;
	scanf("%d", &t);
	
	while ( t--) {
		long long n;
		scanf("%lld", &n);
		printf("%lld\n", ans[n]);
	}
	
	return 0;
}
