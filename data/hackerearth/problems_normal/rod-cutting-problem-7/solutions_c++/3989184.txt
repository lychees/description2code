#include <iostream>
using namespace std;

int main()
{
    int t,n,i,temp,count=0,a[100009];
    
    a[count]=3;
    i=0;
    
    while(i<=1000000000)
    {
    	i=a[count]*2+1;
    	count++;
    	a[count]=i;
    }
    
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	
    	int j=0;
    	
    	while(a[j]<=n)
    	{
    		j++;
    	}
    	
    	cout<<j<<endl;
    	
    }
    return 0;
}
