#include <bits/stdc++.h>
using namespace std;
int adm(long long n,long long z[])
{
	for(int i=1;i<=n;i++)
	{
		for(int j=i+1;j<=n;j++)
		{
			if((i*j)%(i+j)==0)
			{
				z[j]++;
			}
			
		}
		
	}
	
	
return 0;
}
	
int main()
{  long long c=0,n[100000],z[100000]={0},t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		cin>>n[i];
      c=max(c,n[i]);		
	}
	adm(c,z);
	for(int i=1;i<=c;i++)
	{
		z[i]=z[i]+z[i-1];
	}
	for(int i=0;i<t;i++)
	{
		cout<<z[n[i]]<<"\n";
	}
	
	
	return 0;
	
}
