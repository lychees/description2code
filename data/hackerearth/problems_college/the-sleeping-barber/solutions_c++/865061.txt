#include<iostream>
#include<algorithm>
using namespace std;
int arr[3000];
int main()
{
	int d, g, m;
	long long tot;
	cin >> d;
	while (d--)
	{
		int cnt = 0;
		tot = 0;
		cin >> g >> m;
		for (int i = 0; i < g; i++)
			cin >> arr[i];
		sort(arr, arr + g);
		for (int i = 0; i < g; i++)
		{
			tot += arr[i];
			if (tot>m)
				break;
			cnt++;
		}
		cout << cnt <<endl;

	}
	return 0;
}
