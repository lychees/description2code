#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int t,n,ans[20000]={0},i;
	ios::sync_with_stdio(0);
	
	for(i=1; i<=8; i++)
	{
	    if(i<=4 || i%2==0)
	    ans[i]=1;
	    else
	    ans[i]=0;
	}
	for(i=9; i<=10010; i++)
	{
	    if(ans[i-1]==1 && ans[i-4]==1)
	    ans[i]=0;
	    else 
	    ans[i]=1;
	}
	cin>>t;
	while(t--)
	{
	    cin>>n;
	    if(ans[n]==1)
	    cout<<"Yes"<<'\n';
	    else
	    cout<<"No"<<'\n';
	}
	return 0;
}
