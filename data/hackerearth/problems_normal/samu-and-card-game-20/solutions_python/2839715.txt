test_cases = int(raw_input())
def is_safe_move(x, y, row, col):
	if x >= 1 and row >= x and y >= 1 and col >= y:
		return True
	else:
		return False
		
def park_count(card_list, row, col):
	count = 0
	next_x = 1
	next_y = 1
	if row == 0 and col == 0:
		return 0
	for card in card_list:
		A = int(card[0])
		B = int(card[1])
		while (True):
			next_x = next_x + A
			next_y = next_y + B
			if is_safe_move(next_x, next_y, row, col):
				count = count + 1
			else:
				next_x = next_x - A
				next_y = next_y - B
				break
	
	return count 
	
	
for i in xrange(test_cases):
	count = 0
	row_and_col = raw_input().split(' ')
	row,col = int(row_and_col[0]), int(row_and_col[1])
	no_of_cards = int(raw_input())
	card_list = []

	for card in xrange(no_of_cards):
		card_list.append(raw_input().split(' '))
	
		
	print park_count(card_list, row, col)
