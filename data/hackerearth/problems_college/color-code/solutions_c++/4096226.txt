#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	int p=1;
	while(p<=t)
	{
		int a[3];
		char h[2];
		memset(h,'0',sizeof(h));
		for(int i=0;i<3;i++)
		{
			cin>>a[i];
		}
		cout<<"Case "<<p<<":";
		for(int i=0;i<3;i++)
		{
			memset(h,'0',sizeof(h));
			int k=a[i];
			int l=0;
			while(k>0)
			{
				int m;
				m=k%16;
				if(m>9)
				{
				    m=m+75;
				}
				else
				{
					m=m+48;
				}
				h[l++]=m;
				k=k/16;
			}
			cout<<" ";
			cout<<h[1]<<h[0];
		}
		cout<<endl;
		p++;
	}
	return 0;
}
