#include <iostream>
using namespace std;

int main()
{
    long long n;
    cin>>n;
    long long a[n];
    for(int i=0;i<n;i++)
    cin>>a[i];
    long long i=0,c=0,c1=0;
    while(i<n-1)
    {
    	if((abs(a[i+1]-a[i]))==1)
    	{
    		c++;
    		i++;
    	}
    	else
    	{
    		c=0;
    		i++;
    	}
    	if(c>c1)
    	{
    		c1=c;
    	}
    }
    cout<<c1+1;
    return 0;
}
