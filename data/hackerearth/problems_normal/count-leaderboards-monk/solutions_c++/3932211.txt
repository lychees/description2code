#include <cstdio>

#define MOD 1000000007

int main()
{
	long long choose[1001][1001], split[1001] = {1}, factorial[1001] = {1};
	for (int i = 0; i <= 1000; i++)
	{
		choose[i][0] = 1;
		choose[i][i] = 1;
		for (int j = 1; j < i; j++)
		{
			choose[i][j] = (choose[i-1][j-1] + choose[i-1][j]) % MOD;
		}
	}
	for (int i = 1; i <= 1000; i++)
	{
		factorial[i] = factorial[i-1] * i;
		factorial[i] %= MOD;
	}
	for (int i = 1; i <= 1000; i++)
	{
		for (int j = 0; j < i; j++)
		{
			long long temp = choose[i-1][j] * split[j];
			temp %= MOD;
			temp *= factorial[i-j];
			temp %= MOD;
			split[i] += temp;
			split[i] %= MOD;
		}
	}
	int n, k;
	scanf("%d %d", &n, &k);
	printf("%lld\n", (choose[n][k] * split[n]) % MOD);
}
