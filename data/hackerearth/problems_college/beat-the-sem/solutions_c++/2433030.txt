#include <bits/stdc++.h>
using namespace std;
int main() {
	int t;
	scanf("%d",&t);
	while(t--)
	{
		//cout<<1;
		long long a,b;
		scanf("%lld %lld",&a,&b);
	
	if(b>a) cout<<"impossible\n";
	else if((a+b)%2 ==1 ) cout<<"impossible\n";
	else if ((a-b)%2 == 1) cout<<"impossible\n";
	else
	{
		printf("%lld %lld\n",(a+b)/2,(a-b)/2);
	}
		
	}
	return 0;
}
