#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <string>
using namespace std;

int main(void) {
    int T = 0;
    cin >> T;
    while(T--) {
    	int arr[256] = {0};
    	int current = 1, maximum = 1, last = 0;
    	string str;
    	cin >> str;
    	int  i = 0;
    	arr[str[0]] = 1;
    	for ( i = 1; i < str.length(); ++i) {
    		last = arr[str[i]];
    		if(last == 0 || i - current > last)
    			++current;
    		else {
    			if(current > maximum)
    				maximum = current;
    			current = i - last;
    		}
    		arr[str[i]] = i;
    	}
    	if(current > maximum)
    		maximum = current;
    	cout << maximum << endl;
    }
    return 0;
}
