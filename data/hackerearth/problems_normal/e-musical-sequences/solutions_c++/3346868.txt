#include <iostream>
#include <cstdlib>
#include <cmath>
#include <algorithm>
#include <map>
#include <utility>
#include <vector>
using namespace std;

int main(){
	int t;
	long long int n, m, z;
	cin >> t;
	for(int i = 0; i < t; i++){
		cin >> n >> m >> z;
		long long int a[n+1];
		long long int zval = 0;
		for(int j = 0; j <= n; j++) a[j] = 0;
		for(int j = 0; j < n; j++){
			cin >> a[j];
		}
		for(int j = n-1; j >= 0; j--){
			a[n] += (pow(-1,n-1-j) * a[j]);
			a[n] = a[n]%m;
			if(a[n] < 0) a[n] = m + a[n];
		}
		//cout << a[n] << " = " << endl;
		if(a[n] < 0) a[n] = m + a[n];
		if(z < (n+1)){
			cout << a[z] << endl;
		}
		else{
			if(n%2 == 0){
				//cout << z/(n+1) << " " << floor(z/(n+1)) << " " << (long long int)(floor(z/(n+1))) << endl;
				if((z/(n+1)) %2 == 1) zval = (-1 * a[z%(n+1)]);
				else zval = a[z%(n+1)];
			}
			else{
				zval = a[z%(n+1)];
			}
			zval = zval%m;
			if(zval < 0) zval = m + zval;
			cout << zval << endl;
		}
	}
	return 0;
}
