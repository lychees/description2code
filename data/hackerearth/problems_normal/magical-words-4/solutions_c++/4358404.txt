#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int t,i,j,k;
    string str;
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	long long int n=str.length();
    	long long int dp[n][n];
    	memset(dp,0,sizeof(dp));
    	for(i=0;i<n;i++)
    		dp[i][i]=1;
    	for(k=2;k<=n;k++)
    	{
    		for(i=0;i+k<=n;i++)
    		{
    			j=i+k-1;
    			if(k==2)
    			{
    				if(str[i]==str[j])
    					dp[i][j]=k;
    			}	
    			else
    			{
    				if(dp[i+1][j-1]!=0 && str[i]==str[j])
    					dp[i][j]=k;
    			}
    		}
    	}
    	long long int sum=0;
    	for(i=0;i<n;i++)
    		for(j=0;j<n;j++)
    			sum = sum + (dp[i][j]*dp[i][j]) ;
    	cout<<sum<<endl;		
    }
    return 0;
}
