#include <iostream>
using namespace std;

int main()
{
	int a[100][100];
    int m,i,j,s1=0,s2=0;
    cin>>m;
    for(i=0;i<m;i++)
    {
    	for(j=0;j<m;j++)
    	{
    		cin>>a[i][j];
    	}
    }
    
    for(i=0;i<m;i++)
    {
    	s1+=a[i][i];
    }
    
    for(i=0,j=m-1;i<m,j>=0;i++,j--)
    {
    	s2+=a[i][j];
    }
    if(s1-s2>=0)
    {
    	cout<<s1-s2<<endl;
    }
    else
    {
    	cout<<s2-s1<<endl;
    }
    return 0;
}
