#include <iostream>
using namespace std;

int a1[32],a2[32],s1,s2;



int main()
{
    int T,i,cnt;
    cin>>T;
    long int n1,n2;
    
    while(T--)
    {
    	cin>>n1>>n2;
    	s1=s2=cnt=0;
    	while(n1)
    	{
    		a1[s1++]=n1%2;
    		n1/=2;
    	}
    	while(n2)
    	{
    		a2[s2++]=n2%2;
    		n2/=2;
    	}
    	for(i=0;i<(s1>s2?s1:s2);i++)
    	{
    		if(i<=s1&&i<=s2)
    		{
    			if(a1[i]!=a2[i])
    				cnt++;	
    		}
    		else if(i<=s1)
    		{
    			if(a1[i])
    				cnt++;
    		}
    		else if(i<=s2)
    		{
    			if(a2[i])
    				cnt++;
    		}
    	}
    	cout<<cnt<<endl;
    	
    }
    return 0;
}
