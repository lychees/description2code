#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,m;
    	cin>>n>>m;
    	long result=0, ac=0;
    	int a[500];
    	char b[1000];
    	while(n--)
    	{
    		for(int i=0;i<m;i++)
    			cin>>b[i];
    		for(int i=0;i<ac;i++)
    		{
    			if(b[a[i]]=='\\' && b[a[i]+1]=='/')
    				result++;
    		}
    		ac=0;
    		for(int i=0;i<m-1;i++)
    		{
    			if(b[i]=='/' && b[i+1]=='\\')
    				a[ac++]=i;
    		}
    	}
    	cout<<result<<endl;
    }
    return 0;
}
