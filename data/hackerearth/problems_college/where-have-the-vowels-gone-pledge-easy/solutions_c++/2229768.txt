#include<iostream>
#include<cmath>
using namespace std;
int main(){
	int n,sum=0;
	string s;
	cin>>n;
	cin>>s;
	int a=1,e=1,i=1,flag=1,o=1,u=1;
	for(int j=0;j<n;j++){
		if(sum==5){
			cout<<"YES"<<endl;
			flag=0;
			break;
		}
		if(s[j]=='a' && a==1){
			a=0;
			sum++;
			continue;
		}
		if(s[j]=='e' && e==1){
			e=0;
			sum++;
			continue;
		}
		if(s[j]=='i' && i==1){
			i=0;
			sum++;
			continue;
		}
		if(s[j]=='o' && o==1){
			o=0;
			sum++;
			continue;
		}
		if(s[j]=='u' && u==1){
			u=0;
			sum++;
			continue;
		}
	}
	if(flag==1)
	cout<<"NO"<<endl;
	return 0;
}
