#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t,n,k;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>n>>k;
    	if(k>n/2)
    		k=n-k;
    	for(int j=0;j<k-1;j++)
    		cout<<"9";;
    	cout<<"8";
    		
    	for(int j=0;j<n-2*k;j++)
    		cout<<"9";
    	for(int j=0;j<k-1;j++)
    		cout<<"0";
    	cout<<"1"<<endl;
    }
    return 0;
}
