#include<iostream>
#include<cmath>

using namespace std;

int main()

{
    int test;
    cin>>test;
    while(test--)
    {
        int a;
        cin>>a;
        int row,col;
        int mid = a/2;
        for(int i =0;i<a;i++)
        {
            for(int j =0;j<a;j++)
            {
                char temp;
                cin>>temp;
                if(temp == '*')
                {
                    row = i;
                    col = j;

                }
            }
        }

        int count = abs(row-mid)+abs(col-mid);
        cout<<count<<endl;
    }
}
