#include <iostream>
#include<vector>
using namespace std;

vector<int>byte(int a){
	vector<int>s;
	int i;
	while(a!=0){
		i=a%2;
		s.push_back(i);
		a=a/2;
	}
	return s;
}

int count_101(vector<int>vec){
	int ct=0;
	for(int i=0;i<vec.size()-2;i++){
		if(vec[i]==1 && vec[i+1]==0 && vec[i+2]==1){
			ct++;
		}
	}
	return ct;
}
int main()
{
    int t;
    cin>>t;
    for(int m=0;m<t;m++){
    	vector<int>w;
    	int r,k,count=0;
    	cin>>r>>k;
    	for(int i=5;i<r+1;i++){
    		w=byte(i);
    		if(count_101(w)>=k){
    			count++;
    		}
    	}
    	cout<<count << endl;
    }
}
