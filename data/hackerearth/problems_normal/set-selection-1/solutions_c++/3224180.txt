#include<bits/stdc++.h>
using namespace std;
const int MAXIM=1e6+3;
bool dp[2][MAXIM];
int ar[102];
int main() {
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
		scanf("%d",&ar[i]);
	for(int i=1;i<MAXIM;i++)
		dp[0][i]=false;
	dp[0][0]=true;
	for(int i=1;i<=n;i++) {
		for(int j=0;j<MAXIM;j++) {
			int previ;
			if(i&1)
				previ=0;
			else
				previ=1;
			dp[i&1][j]=dp[previ][j];
			if(j>=ar[i])
				dp[i&1][j]|=dp[previ][j-ar[i]];
		}
	}
	int t;
	cin>>t;
	while(t--) {
		int z;
		scanf("%d",&z);
		if(dp[n&1][z])
			printf("Yes\n");
		else
			printf("No\n");
	}
	return 0;
}
