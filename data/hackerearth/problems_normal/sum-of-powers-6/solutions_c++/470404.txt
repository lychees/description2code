#include <iostream>
using namespace std;

int main()
{
    int t,k;
    bool arr[1000001]={false};
    long long i,j,x;
    bool  flag;
    
    arr[1]=true;
    
    for(i=2;i<=1000000;i++)
    {
    	if(!arr[i])
    	for(j=i*i;j<=1000000;j=j*i)
    	{
    		arr[j]=true;
    	}
    }
    
    cin>>t;
    
    for(k=0;k<t;k++)
    {
    	cin>>x;
    	flag=false;
    	
    	for(i=1;i<=x;i++)
    	{
    		if(arr[i] && arr[x-i])
    		{
    			flag=true;
    			break;
    		}
    	}
    	
    	if(flag)
    	cout<<"Yes\n";
    	else
    	cout<<"No\n";
    }
    
    return 0;
}
