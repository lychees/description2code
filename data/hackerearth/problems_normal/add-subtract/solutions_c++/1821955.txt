typedef set<string> ss;

typedef pair<int,int> pa;
typedef long long int ll;

ll ans,n,k,i,a[104],f[104],cnt,j;
int main()
{
	//freopen("i.in", "r", stdin);
  	//freopen("o.out", "w", stdout);
  	int t;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		for(i=0;i<n;i++)
		cin>>a[i];
		ans=INT_MAX;
		for(i=1;i<=100;i++)
		{
			for(j=0;j<n;j++)
			f[j]=(i>a[j]?3*(i-a[j]):5*(a[j]-i));
			sort(f,f+n);
			cnt=0;
			for(j=0;j<k;j++)
			cnt+=f[j];
			ans=min(ans,cnt);
		}
		cout<<ans<<"\n";
	}
	return 0;
}
