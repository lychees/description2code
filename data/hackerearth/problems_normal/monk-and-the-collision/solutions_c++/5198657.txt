#include<bits/stdc++.h>
using namespace std;

int main()
{
    int test;
    cin>>test;
    while(test--)
    {
    	map<int,int>m;
    	int n,count=0;
    	cin>>n;
    	int arr[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>arr[i];
    		if(m[arr[i]%10]==0)
    		   m[arr[i]%10]++;
    		else
    		   count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
