#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;

int main()
{
	int t;
	cin>>t;
	for(int ii=0;ii<t;ii++)
	{
		string s;
		cin>>s;
		string rev = s;
		reverse(rev.rbegin(),rev.rend());
		int i=0, j=0, flag=0;
		while(i<s.length())
		{
			if((s[i]=='1' || s[i]=='0' || s[i]=='8') && (s[i]==rev[j]))
			{
				i++;
				j++;
			}
			else
			{
				flag=1;
				break;
			}
		}

		if(flag==1)
			cout<<"NO"<<endl;
		else
			cout<<"YES"<<endl;
	}
    return 0;
}
