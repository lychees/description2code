#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main()
{
    int N, temp;
    string s;
    cin >> N;
    
    for(int i=0; i<N; ++i){
    	cin >> temp;
    	if(temp%21 == 0){
    		cout << "The streak is broken!\n";
    		continue;
    	}
    	s = to_string(temp);
    	if(s.find("21") != string::npos){
    		cout << "The streak is broken!\n";
    		continue;
    	}
    	cout << "The streak lives still in our heart!\n";
    }
    return 0;
}
