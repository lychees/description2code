#include<iostream>
#include<cmath>
#define lld long long int
using namespace std;

int main()
{
    lld t,num;
    cin>>t;
    while(t--)
    {
        lld res,temp,temp1;
        cin>>num;
        res=(int)sqrt(num);
        if(res*res==num)
            cout<<"YES\n";
            else
             {
                temp=num-res*res;
                res+=1;
                temp1=res*res -num;
                if(temp<temp1)
                cout<<"-"<<temp<<endl;
                else
                cout<<"+"<<temp1<<endl;
             }
    }
    return 0;
}
