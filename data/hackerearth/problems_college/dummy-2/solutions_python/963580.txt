while t:
    t-=1
    num=0
    n,s=raw_input().split()
    n,s=(int)(n),(str)(s)
    l=len(s)
    k=1
    for i in range(l):

        if tmp<60:
            num+=(tmp-48)*k
        else:
            num+=(tmp-87)*k
        k*=n
    x=(str)(num)
