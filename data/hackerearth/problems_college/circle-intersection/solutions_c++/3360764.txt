#include <iostream>
# include <cstdio>
# include<cmath>
using namespace std;

int main() {
    float x1,x2,y1,y2,r1,r2;
    float dis,rdis;
	scanf("%d\t%d\t%d\n",&x1,&y1,&r1);
    scanf("%d\t%d\t%d\n",&x2,&y2,&r2);
    dis=sqrt(pow(x2-x1,2)+pow(y2-y1,2));
    rdis=r2+r1;
	if(dis==rdis)
	printf("T");
	if(dis>rdis)
	printf("N");
	if(dis<rdis)
	printf("Y");
	return 0;
}
