# a(n) = {a(n-1) + a(n-2), n even, a(n-1)*a(n-2), n odd}; a(0)=0; a(1)=1

t = int(input())
arr = [0]*25
arr[0] = 0
arr[1] = 1

for i in range(2, 25):
    if (i % 2 == 1):
        arr[i] = arr[i-1]*arr[i-2]
    else:
        arr[i] = arr[i-1]+arr[i-2]

while (t > 0):
    t -= 1
    line = raw_input()
    params = [int(x) for x in line.split()]

    n = params[0]
    # print(arr[5])

    print(arr[n+2])
