#include <iostream>
using namespace std;

int main() {
	int t, n, i, max, min;
	cin >> t;
	while(t--) {
		cin >> n;
		int a[n];
		for(i = 0; i < n; i++) cin >> a[i];
		min = a[0];
		max = a[0];
		for(i = 1; i < n; i++) {
			if(a[i] < min) min = a[i];
			if(a[i] > max) max = a[i];
		}
		cout << 2*(max-min) << '\n';
	}
    return 0;
}
