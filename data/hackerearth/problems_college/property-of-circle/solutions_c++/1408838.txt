#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int test;
    cin>>test;
    while(test--)
    {
    	double x1,y1,x2,y2,r1,r2;
    	cin>>x1>>y1>>r1>>x2>>y2>>r2;
    	double dist;
    	dist=((x2-x1)*(x2-x1))+((y2-y1)*(y2-y1));
    	dist=sqrt(dist);
    	if(dist==0&&r1==r2)
    	{
    		cout<<"C2~C1"<<endl;
    		
    	}
    	else if(r1==r2)
    	{
    		cout<<"NOT SATISFIED"<<endl;
    	}
    	else if(r1>r2&&dist<=(r1-r2))
    	{
    		cout<<"C1CC2"<<endl;
    	}
    	else if(r2>r1&&dist<=(r2-r1))
    	{
    		cout<<"C2CC1"<<endl;
    	}
    	else
    	{
    		cout<<"NOT SATISFIED"<<endl;
    	}
    }
    return 0;
}
