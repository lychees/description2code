#include <bits/stdc++.h>
using namespace std ;
int main()
{

    int n ;
    cin >> n ;
    pair <int , int > p[n] ;
    int i ;


    for ( i = 0; i< n ; i++)
    {
        int m ;
        int o ;
        cin >> m ;
        cin >> o ;
        p[i].first = m+o ;
        p[i].second = i+1 ;
    }
    sort ( p , p+n) ;
    for ( i = 0 ; i< n ; i++)
    {
        cout << p[i].second << " " ;
         // cout << p[i].first<< " " <<endl;  ;
    }
    cout << endl;

}
