#include <iostream>
using namespace std;

void part(int a[],int n)
{
    for(int i=0;i<n;i++)
    {
        for(int j=i+1;j<n;j++)
        {
            if(a[i]>a[j])
            {
                int temp=a[i];
                a[i]=a[j];
                a[j]=temp;
            }
        }
    }
}
 
int main() {
	int t,n;
	long int x;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		x=0;
		cin>>n;
		int m[n],w[n];
		for(int j=0;j<n;j++)
		{
			cin>>m[j];
		}
		for(int j=0;j<n;j++)
		{
			cin>>w[j];
		}
		part(m,n);
		part(w,n);
		for(int j=0;j<n;j++)
		{
			x+=m[j]*w[j];
		}
		cout<<x<<"\n";
	}
	return 0;
}
