#include<iostream>
#define MOD 1000000007
using namespace std;
int main(){
    long long a[100],n;
    string str;
    a[0]=0;a[1]=0;a[2]=1;
    for(long long i=3;i<=100;i++){
        a[i]=((i-1)*(a[i-1]+a[i-2]))%MOD;
    }    
    cin>>n>>str;
    if(n==1){cout<<"1\n";}else{
    cout<<a[n]<<endl;}
    return 0;
}
