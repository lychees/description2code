#include <iostream>
using namespace std;

int main()
{
    long T, N, P;
    cin >> T;

    long temp[1009][1009];
    for(long i = 0; i <= 1000; i++) {
    	temp[i][0] = 1;
    	temp[i][i] = 1;
    	for (long j = 1; j < i; j++) {
    		temp[i][j] = (temp[i - 1][j - 1] + temp[i - 1][j]) % 1000000007;
    	}
    }

    while(T--) {
    	cin >> N >> P;
    	long sum = 0;
    	for (long p = 0; p <= P; p++) {
    		sum += temp[N][p];
    		sum = sum % 1000000007;
    		//cout << temp[N][p] << endl;
    	}
    	cout << sum << endl;
    }
    return 0;
}
