#include <bits/stdc++.h>
#include<stack>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    int s1=0,a,n;
    stack<int>s;
    scanf("%d",&n);
    while(n--)
    {
    	scanf("%d",&a);
    	if(a==0)
    	s.pop();
    	else
    	s.push(a);
    	
    } 
    while(!s.empty())
    {
    s1=s1 + s.top();
    s.pop();
    }
    printf("%d",s1);
    
    return 0;
}
