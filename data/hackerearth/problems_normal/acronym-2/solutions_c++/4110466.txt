#include<bits/stdc++.h>
using namespace std;
set<string> st;
int main()
{
    //cout << "Hello World!" << endl;
    int d;
    string str;
    cin>>d;
    while(d--)
    {
    	cin>>str;
    	st.insert(str);
    }
    int m;
    cin>>m;
    int flag=1;
    while(m--)
    {
    	cin>>str;
    	if(st.find(str) == st.end())
    	{
    		char z = toupper(str[0]);
    		if(flag == 1) {
    			cout<<z;
    			flag++;
    		}
    		else cout<<"."<<z;
    	}
    }
    cout<<endl;
    return 0;
}
