#include <iostream>
#include <string>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	    string str;
	    cin>>str;
	    int i;
	    int count = 0;
	    //int j = 1;
	    int len = str.length();
	    for(i=0;i<len;i++)
	    {
	    	if(str[i] == str[i+1])
	    	{
	    		count++;
	    	}
	    }
	    cout<<count<<endl;
	}
}
