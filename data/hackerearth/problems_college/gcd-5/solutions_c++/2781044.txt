#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{   ll t;
    cin>>t;
    for(ll i=0;i<t;i++)
    {
            ll n;
            cin>>n;
            ll arr[n];
            for(ll j=0;j<n;j++)
                cin>>arr[j];
            ll j;
            for(j=1;j<n;j++)
            {
                if(arr[j]<arr[j-1])
                {
                    cout<<"NO"<<endl;
                    break;
                }
            }
            if(j==n)
                cout<<"YES"<<endl;
    }
    return 0;
}
