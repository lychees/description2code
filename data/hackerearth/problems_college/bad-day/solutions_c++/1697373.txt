#include <iostream>
using namespace std;

int main()
{
    int t,n,var1,var2,temp;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	var1=1;var2=2;temp=0;
    	if(n==1)
    	cout<<"1"<<"\n";
    	else if(n==2)
    	cout<<"2"<<"\n";
    	else
    	{
    	n-=2;
    	while(n--)
    	{
    		temp=var2;
    		var2=var1+var2;
    		var1=temp;
    	}
    	cout<<var2<<"\n";
    	}
    	
    }
    return 0;
}
