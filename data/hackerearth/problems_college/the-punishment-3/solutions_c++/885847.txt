#include<stdio.h>

void work(int n,int k)
{
    int a[100001]={},b[100001]={},c[100001]={},d[100001]={},e[100001],
    lena,lenb,
    lenc,lend,lene,i,j;
    a[0]=1;
    a[n]=1;
    lena=n+1;
    b[n-k]=1;
    lenb=n-k+1;


    d[k]=1;
    lend=k+1;
    //c[k]=1;
    //lenc=k+1;

    lenc=((lena<lenb)?lena:lenb);
    for(i=0;i<lenc;i++)
    {
        if(a[i]>=b[i])
            c[i]=a[i]-b[i];
        else
        {
            for(j=i;;j++)
                if(a[j]>0)
                break;
            a[j]--;
            j--;
            while(j>i)
                a[j]=9,j--;

            a[j]+=10;
            c[i]=a[i]-b[i];
        }
    }

    for(;i<lena;i++)
        c[i]=a[i],lenc++;

    for(;lenc>0;lenc--)
        if(c[lenc-1]!=0)
        break;

    lene=((lenc<lend)?lenc:lend);
    for(i=0;i<lene;i++)
    {
        if(c[i]>=d[i])
            e[i]=c[i]-d[i];
        else
        {
            for(j=i;;j++)
                if(c[j]>0)
                break;
            c[j]--;
            j--;
            while(j>i)
                c[j]=9,j--;

            c[j]+=10;
            e[i]=c[i]-d[i];
        }
    }

    for(;i<lenc;i++)
        e[i]=c[i],lene++;

    for(;lene>0;lene--)
        if(e[lene-1]!=0)
        break;

    for(i=lene-1;i>=0;i--)
        printf("%d",e[i]);
    printf("\n");
}



int main()
{

    int ntc,n,k;
    scanf("%d",&ntc);
    while(ntc--)
    {
        scanf("%d%d",&n,&k);
        work(n,k);





    }

}
