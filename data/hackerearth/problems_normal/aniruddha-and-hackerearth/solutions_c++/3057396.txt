#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
long long int end_with_H[1000005];
long long int end_with_E[1000005];
long long int dp[1000005];
int main()
{
	int t,n;
	scanf("%d",&t);
    end_with_H[1]=1;
    end_with_E[1]=1;
    dp[1]=2;
    for(int i=2;i<=1000000;i++)
    {
    	end_with_H[i]=(end_with_H[i-1]+end_with_E[i-1])%MOD;
    	end_with_E[i]=(end_with_H[i-1])%MOD;
    	dp[i]=(dp[i-1]+(end_with_H[i]+end_with_E[i])%MOD)%MOD;
    }
    while(t--)
    {
    	scanf("%d",&n);
    	cout<<dp[n]<<"\n";
    }
    return 0;
}
