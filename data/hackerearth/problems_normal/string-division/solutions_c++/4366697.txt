#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	long t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		long i;
		int flag[26];
		memset(flag,0,sizeof(flag));
		int count=0;
		for(i=0;i<s.length();i++)
		{
			if(flag[s[i]-'a']==0)
			{
				flag[s[i]-'a']++;
				count++;
			}
		}
		long prod=1;
		if(count==2)
		{
			for(i=0;i<26;i++)
			if(flag[i]!=0)
			prod=prod*flag[i];
		}
		if(count>=4)
		cout<<"YES\n";
		else if(count==3&&s.length()>=5)
		cout<<"YES\n";
		else if(count==2&&s.length()>=7||count==2&&s.length()==6&&prod>7)
		cout<<"YES\n";
		else if(count==1&&s.length()>=10)
		cout<<"YES\n";
		else
		cout<<"NO\n";
	}
    return 0;
}
