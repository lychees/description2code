#include <iostream>
#include <string>
using namespace std;

int main()
{
	string input;
	string output = "";
	
	getline (cin,input);
	for( unsigned int i = 0; i < input.length(); i++){
		char element = input[i];
		//cout << "Loop iteration " << i <<  " " << element << endl;
		
		if(islower(element)){
        	element = toupper(element);
    	} else {
    		element = tolower(element);
    	}
    	//cout << "Loop iteration " << i <<  " " << element << endl;
    	output += element;
	}
    //cout << "Hello World!" << endl;
    cout << output << endl;
    return 0;
}
