#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n,i,j,res,b[1050],limit,count;
	bool a[1050][1050];
    cin>>n;
    res=pow(2,n);
    memset(a,0,sizeof(a));
    for(i=1;i<=res;i++)
    b[i]=i;
    for(i=2;i<=res;i++)
    {
    	for(j=1;j<=(i-1);j++)
    	{
    		cin>>a[i][j];
    	}
    }
    j=res;
    while(j!=1)
    {count=0;
    	for(i=1;i<=j;i+=2)
    	{
    		if(a[b[i+1]][b[i]]==1)
    			b[++count]=b[i+1];
    		else
    			b[++count]=b[i];	
    	}
    	j=j/2;
    }
    cout<<b[1];
    
    return 0;
}
