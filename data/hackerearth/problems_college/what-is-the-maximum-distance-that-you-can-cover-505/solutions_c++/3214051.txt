#include <iostream>
#include <cmath>
#include <iomanip>
#include <stdio.h>
using namespace std;

int main()
{
	int cars,x;
	cin>>cars;
	double speeds[cars], fuel[cars], totfuel, temp;
	double distance=0.0;
	for(x=0; x<cars; ++x)
	{
		cin>>speeds[x];
	}
	for(x=0; x<cars; ++x)
	{
		cin>>fuel[x];
	}
	cin>>totfuel;
	for(x=0; x<cars; ++x)
	{
		temp = ( totfuel / fuel[x] ) * speeds[x];
		temp = floor(1000*temp)/1000;
		if (distance<temp)
		{
			distance = temp;
		}
	}
    cout <<fixed<<setprecision(3)<<distance<<endl;
    return 0;
}
