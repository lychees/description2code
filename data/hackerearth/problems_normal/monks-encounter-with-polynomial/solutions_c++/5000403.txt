#include <iostream>
using namespace std;

int main()
{
    int t,a,b,c,k,x;
    cin>>t;
    while(t--)
    {
    cin>>a>>b>>c>>k;
    if(c>=k)
    {
    	x=0;
    cout<<x<<endl;
   
    }
  else
  {
    	x=1;
    	while((a*x*x+b*x+c)<k)
    	{
    	x++;
    	}
    		cout<<x<<endl;
    }
    }
    
    
    
    
    return 0;
}
