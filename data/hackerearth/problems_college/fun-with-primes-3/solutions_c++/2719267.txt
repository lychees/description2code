#include <iostream>
using namespace std;

int main()
{
    int p[61],i,j,t,n,temp,c;
    
  	p[0]=p[1]=0;
    for(i=2;i<=60;i++){
    	p[i]=1;
    }
    
    for(i=2;i<=60;i++){
    	if(p[i]){
    		for(j=i*i;j<=60;j+=i){
    			p[j]=0;
    		}
    	}
    }
    
    cin>>t;
    while(t--){
    	cin>>n;
    	temp=n;
    	c=0;
    	while(temp){
    		c+=temp%10;
    		temp/=10;
    	}
    	
    	if(p[c]){
    		cout<<"YES"<<endl;
    	}else{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
