#define RFOR(i,a,b)     for(int i=b-1;i>=a;i--)

#define gcd(a,b)  __gcd(a,b)
#define MOD       1000000007

VI primes;
void sieve(int limit)
{
      int arr[limit];
      FOR(i,2,limit+1) arr[i] = 1;
      int i = 2;
      while(i<=limit)
      {
            primes.pb(i);
            int j = 2*i;
            while(j<=limit)
            {
                  arr[j] = 0;
                  j+=i;
            }
            i++;
            while(i<=limit && !arr[i])
                  i++;
      }
}

int main()
{
      sieve(100000);
      inp_s;
      LL arr[100000] = {0};
      FOR(i,1,100000) arr[i] = i;
      FOR(i,0,sz(primes))
      {
            int j = primes[i];
            while(j<100000)
            {
                  arr[j] = arr[j]*(primes[i]-1)/primes[i];
                  j+=primes[i];
            }
      }
      DRT()
      {
            LL n,x,k;
            cin>>n>>x>>k;
            LL ans = 0;
            FOR(i,x,n+1)
            {
                  if(arr[i]==k)
                  {
                        ans = i;
                  }
            }
            if(!ans) cout<<"-1"<<endl;
            else cout<<ans<<endl;
      }
      return 0;
}
