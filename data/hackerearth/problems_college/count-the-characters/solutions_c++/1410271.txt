#include <iostream>
#include <string.h>
using namespace std;
static int N,i,j,A[26]={0},ctr=0;
char S[105][105];
main()
{
	cin>>N;
	cin.ignore();
	for(i=0;i<N;i++)
	{
		cin>>S[i];
		for(j=0;j<strlen(S[i]);j++)
		{
			if(A[(S[i][j]-97)]==ctr)
			{
				A[(S[i][j]-97)]=i+1;
			}
		}
		ctr=i+1;
//		for(j=0;j<26;j++)
//		cout<<A[j]<<" ";
//		cout<<"\n";
	}
	j=0;
	//cout<<"Value of Counter = "<<ctr<<"\n";
	for(i=0;i<26;i++)
	{
		if(A[i]==ctr)
		j=j+1;
	}
	cout<<j;
}
