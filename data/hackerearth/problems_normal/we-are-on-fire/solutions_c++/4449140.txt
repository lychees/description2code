#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pii pair<int,int>
#define pll pair<ll,ll>
#define mp make_pair
#define ff first
#define ss second
#define pb push_back
#define pf pop_front
#define IOS ios::sync_with_stdio(false)
int n , m , q , a[1005][1005] ;
int vis[1005][1005];
int dx[] = {1,0,-1,0};
int dy[] = {0,1,0,-1};
int tot;
bool check(int x , int y){
    if((x >= 1 && x <= n) && (y <=m && y >= 1)  && (a[x][y]==1))
        return true;
    else return false;
}
void dfs(int x , int y){
    vis[x][y] = 1;
    for(int i = 0 ; i < 4 ; i++){
        int nx = x+dx[i];
        int ny = y+dy[i];
        if(check(nx,ny) &&!vis[nx][ny])dfs(nx,ny);
    }
    tot--;
    a[x][y] = 0;
}   
int main(){

    int i , j , k , x , y;
    scanf("%d %d %d",&n,&m,&q);
    for(i = 1 ; i <= n ; i++){
        for(j = 1 ; j <= m ; j++){
            scanf("%d",&a[i][j]);
            tot+=a[i][j];
        }
    }	
    while(q--){
        scanf("%d %d",&x,&y);
        if(!vis[x][y] && a[x][y]==1){
            vis[x][y] = 1;
            dfs(x,y);
        }
        printf("%d\n",tot );
    }

	return 0;
}
