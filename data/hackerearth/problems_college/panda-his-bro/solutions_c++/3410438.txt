inline LL power(LL x,LL y)
{
    LL ans=1;
    while(y>0){
        if(y&1)
            ans=(ans*x)%M;
        x=(x*x)%M;
        y/=2;
    }
    return ans;
}
inline int read()
{
	int ret = 0,temp=1;
	int c = getchar_unlocked();
	while(c<'0' || c>'9'){
		c = getchar_unlocked();
	}
	while(c>='0' && c<='9')
	{
		ret = (ret<<3) + (ret<<1) + c - '0';
		c = getchar_unlocked();
	}
        return ret;
}
LL mulmod(LL a,LL b) {
   long double res = a;
   res *= b;
   LL c = (LL)(res / M);
   a *= b;
   a -= c * M;
   a %= M;
   if (a < 0) a += M;
   return a;
}
const int N=100005;
vector<int> V[N];
int d[N];
bool mark[N];
int mnode,maxi=0;
void bfs(int x)
{
  memset(mark,0,sizeof(mark));
  memset(d,0,sizeof(d));
  queue<int> Q;
  Q.push(x);
  d[x]=0;
  while(!Q.empty()){
    int p=Q.front();
    mark[p]=1;
    for(int i=0;i<V[p].size();i++){
        if(mark[V[p][i]]==0){
          Q.push(V[p][i]);
          d[V[p][i]]=d[p]+1;
          if(maxi<d[V[p][i]]){
            maxi=d[V[p][i]];
            mnode=V[p][i];
          }
        }
    }
    Q.pop();
  }
}
int main()
{
    int n;
    scanf("%d",&n);
    int u,v;
    for(int i=0;i<n-1;i++){
      scanf("%d %d",&u,&v);
      V[u].pb(v);
      V[v].pb(u);
    }
    bfs(1);
    bfs(mnode);
    printf("%d\n",maxi);
}
