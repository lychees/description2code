d = dict()
count = 0
for i in "abcdefghijklmnopqrstuvwxyz":
    d[i] = count
    count += 1
length = int(raw_input())
strings = []
for i in range(0, length):
    newstring = raw_input()
    newstring = 'a' + newstring
    strings.append(newstring)

for s in strings:
    sol = []
    for i in range(0, len(s)-1):
        sl = s[i]
        el = s[i+1]
        start = d[sl]
        target = d[el]
        if start == target:
            clockwise = 0
            aclockwise = 0
        elif start > target:
            clockwise = 26 - start + target
            aclockwise = start - target
        else:
            clockwise = target - start
            aclockwise = 26 - target + start
        if clockwise <= aclockwise:
            sol.append(clockwise)
        else:
            sol.append(-aclockwise)
    for i in sol:
        print i,
    print ''
