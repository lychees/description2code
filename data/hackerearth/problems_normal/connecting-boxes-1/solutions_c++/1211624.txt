#include <iostream>
using namespace std;

int main()
{
    int n,q,i,x,y;
    cin>>n>>q;
    int connected[n+1];
    
    for(i=0;i<=n;i++)
    connected[i]=-1;
    
    while(q--)
    {
    	bool flag = false;
    	cin>>x>>y;
    
        if(x==y)
    	flag=true;
    	
    	if(connected[x]!=-1||connected[y]!=-1)
    	flag=true;
    	
    	if(x>y)
    	{
    	int temp = x;
    	x=y;
    	y=temp;
    	}
    	
    	for(i=x+1;i<y;i++)
    	{
    	if(connected[i]!=-1)
    	{
    	if(connected[i]>x&&connected[i]<y)
    	{continue;}
    	else
    	{flag=true;break;}
    	}
    	}
    	
    	if(flag)
    	cout<<"NO"<<endl;
    	else
    	{
    		cout<<"YES"<<endl;
    		connected[x]=y;connected[y]=x;
    	}
    }
    return 0;
}
