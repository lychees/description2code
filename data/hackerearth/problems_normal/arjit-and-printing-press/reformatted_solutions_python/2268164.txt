T = int(raw_input())
while (T != 0):
    W = raw_input()
    R = raw_input()
    # w_sort = ''.join(sorted(W))
    r_sort = ''.join(sorted(R))
    min_length = min(len(R), len(W))
    # result = min(W[0],r_sort[0])
    j = 0
    for i in xrange(min_length):
        #       print "W",W[i]
        #       print "R",r_sort[i]
        if W[i] > r_sort[j]:
            # print "Replacing"
            W = W.replace(W[i], r_sort[j], 1)
            j = j+1
    print W
    T = T-1
