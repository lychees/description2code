#include <iostream>
using namespace std;
#include<string.h>
 
int main()
{	
	int T;
    cin >> T;
    char street[10000];
    for(int i = 0; i < T; i++)
    {	
    	int p = 0;
    	cin >> street;
    	for(int j=0; street[j]!='\0';j++)
    	{
    		if(street[j]=='B'||street[j]=='S'||street[j]=='D')
    			continue;
    		if(street[j+2]!='\0'&&(street[j+1]=='B'||street[j+2]=='B'))
    			continue;
    		else if(street[j+1]!='\0'&&(street[j+1]=='B'))
    			continue;
    		if(street[j+1]=='S'||street[j-1]=='S')
    			continue;
    		p++;
    	}
    	cout<<p<<endl;
    }
    return 0;
    
}
