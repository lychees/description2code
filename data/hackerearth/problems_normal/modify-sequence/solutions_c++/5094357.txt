#include <iostream>
using namespace std;

int main()
{
	int N;
	cin>>N;
	int arr[N];
	//input
	for(int i=0; i<N; ++i)
		cin>>arr[i];
	int flag=0;
	for(int i=0; i<N-1; ++i)
	{
		if(arr[i]>arr[i+1])
		{
			flag = 1;
			break;
		}
		else
		{
			arr[i+1] = arr[i+1] - arr[i];
			arr[i] = 0;
		}
			
		
	}
	
	if(arr[N-1] > 0)
		flag = 1;
	
	
	
	if(flag == 1)
		cout<<"NO"<<endl;
	else
		cout<<"YES"<<endl;

    return 0;
}
