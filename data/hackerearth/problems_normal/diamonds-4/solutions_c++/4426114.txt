#include<iostream>
using namespace std;
main()
{ int i,j,m,n,t,c;
 char w[10000][10000];
 cin>>t;
 while(t--){c=0;
 cin>>n>>m;
  for(i=0;i<n;i++)
  for(j=0;j<m;j++)
 {cin>>w[i][j];}
for(i=0;i<n;i++)
 { for(j=0;j<m;j++)
  { if(w[i][j]=='/'&&w[i][j+1]=='\\'&&w[i+1][j]=='\\'&&w[i+1][j+1]=='/')
      { c++;
        j++;
      }
 }}cout<<c<<endl;
for(i=0;i<n;i++)
  for(j=0;j<m;j++)
 {w[i][j]=0;}
}}
