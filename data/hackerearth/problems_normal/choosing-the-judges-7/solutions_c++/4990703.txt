#include <iostream>
#include<bits/stdc++.h>
using namespace std;
long int ar[10010];

long int DP[10010],size;

long int ret( int n){
  if(n >= size ) return 0;
  if(DP[n] != -1) return DP[n];
  

  DP[n] = max(ret(n+1) , ar[n] + ret(n+2));
  return DP[n];

}







int main()
{
  long int t;
  cin >> t;
  int j = 1;
  while(j <= t){
    
    cin >> size;
    
    for(int i = 0; i < size; i++){
      cin >> ar[i];
    }
    for(int i = 0; i < 10010; i++){
      DP[i] = -1;
    }
    long int val = 0;
    
    val = ret(0);
    
    
    
    cout << "Case "<< j<<": "<< val <<endl;
    j++;
  }

    
    
    return 0;
}
