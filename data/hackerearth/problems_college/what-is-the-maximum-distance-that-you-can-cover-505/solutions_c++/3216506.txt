#include <iostream>
#include <iomanip>
#include <cmath>
#include <algorithm>
using namespace std;

int main()
{
	int n = 0;
	cin >> n;
	double speed[n], amount[n];
	for(int i = 0 ; i< n; i++)
		cin >> speed[i];
	for(int i = 0 ; i< n; i++)
		cin >> amount[i];
	double tot_amount  = 0;
	cin >> tot_amount;
	double max_dis = 0;
	for(int i = 0 ; i< n; i++)
	{
		double timme = tot_amount/amount[i];
		max_dis = max(max_dis, timme*speed[i]); 
	}
	double scale = 0.001;
	max_dis = floor(max_dis/scale) * scale;
	cout << fixed << setprecision(3) << max_dis << endl;
    return 0;
}
