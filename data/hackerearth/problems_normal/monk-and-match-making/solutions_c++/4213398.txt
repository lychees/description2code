#include <iostream>
#include<cstring>
#include<cstdio>
using namespace std;

int main()
{
	string str;
	cin >> str;
	int n;
	cin >> n;
	int l1, r1, l2, r2;
	while(n--) {
		cin >> l1 >> r1 >> l2 >> r2;
		if(r1-l1 != r2-l2) {
			cout << "No" << endl;
		} else if(!memcmp(&str[0]+l1-1, &str[0]+l2-1, r1-l1+1)) {
			cout << "Yes" << endl;
		} else {
			cout << "No" << endl;
		}
	}
    return 0;
}
