for i in range(input()):
    a, b = map(list, raw_input().split())
    c, test = 0, False
    for j in range(len(a)):
        if int(a[j]) + int(b[j]) >= 10:
            test, c = True, c+1
        elif (int(a[j]) + int(b[j]) == 9 and test):
            test, c = False, c+1
        else:
            test = False
    if c == 0:
        print "No carry operation"
    elif c == 1:
        print "1 carry operation"
    else:
        print c, "carry operations"
