#include <iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;

int main()
{   long long t,n,k,sum,diff,ans,a[1000000],i;
   scanf("%lld",&t);
   while(t--){
   	scanf("%lld %lld",&n,&k);
   	for(i=0;i<n;i++){
   		scanf("%lld",&a[i]);
   	}
   	std::sort(a,a+n);
   	sum=0;
   	for(i=0;i<n-1;i++){
   		diff=a[i]-a[i+1];
   		if(diff<0)diff=diff*(-1);
   		sum=sum+diff;
   	}
   	ans=sum*k;
   	printf("%lld\n",ans);
   }
    return 0;
}
