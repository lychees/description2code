#include <cstdio>

unsigned long long int z[1000001] = {};
int zero(int n) {
	int i, ct=0;
	for(i=5; n/i >= 1; i *= 5) {
		ct += n/i;
	}
	return ct;
}

void totalZero() {
	for(int i=5; i<1000001; ++i) {
		z[i] += z[i-1] + zero(i);
	}
}

int main() {
	//Code here
	totalZero();
	int tc;
	scanf("%d", &tc);
	while (tc--) {
		int a, b;
		scanf("%d%d", &a, &b);
		printf("%llu\n", (z[b]-z[a-1]));
	}
	return 0;
}
