#include<iostream>
#include<string>
using namespace std;

main()
{int t;
cin>>t;
string str,k;
while(t--)
{
	cin>>str;
	while(!str.empty())
	{int len=str.size();
		if(len%2==1)
	{
		k.push_back(str[len/2]);
		str.erase(str.begin()+(len/2));
	}
	else
	{
	int pos=len/2;
	pos=str[pos-1]<str[pos]?(pos-1):pos;
	k.push_back(str[pos]);
	str.erase(str.begin()+pos);
	}
	}
	cout<<k<<endl;
	k.clear();
}
	}
