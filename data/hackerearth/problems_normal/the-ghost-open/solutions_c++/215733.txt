#include<bits/stdc++.h>
using namespace std;



int main()
{
   int t;
   int d,x;
   int n;
   long long sum=0;
   cin>>t;
   while(t--)
   {
       cin>>d;
       n=ceil(log(d)/log(2.0));
       n=pow(2,n);
       x=n-d;
       sum=0;
       while(n>0)
       {
           sum+=(n/2);
           n=n/2;
       }
       cout<<sum-x<<"\n";
   }

}
