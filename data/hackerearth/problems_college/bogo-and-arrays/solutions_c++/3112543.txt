#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,s,c=0;
		cin>>n>>s;
		int i,a[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		long long total = (n*(n+1))/2;
		for(i=0;i<n;i++)
		{
	     if(a[i]<=s)
	     {
	     	c++;
		 }
		 else
		 {
		 	total -= (c*(c+1))/2;
		 	c=0;
		 }
		}
		total -= (c*(c+1))/2;
		cout<<total<<"\n";
	}
}
