#include <iostream>
#include <cstdio>
using namespace std;
#define mod 1000000007

int main()
{
	long long int t,n,ans,a[1000000],i;
    a[0]=3;
    for(i=1;i<=1000000;i++)
    {
    	a[i]=((a[i-1]%mod)*i)%mod;
    }
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld",&n);
    	ans=(a[n]+3)%mod;
    	printf("%lld\n",ans);
    }
   return 0;
}
