#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    ll n,a;
    int t;
    scanf("%lld",&t);
    while(t-->0)
    {
        scanf("%lld",&a);
        double sr=sqrt(a);
        ll sr2=(long long int)sr;
        //printf("root=%lld\n",sr2);
        ll b,ans;
        int flag=0;
        if(a>sr2*sr2)
        {
            if(a<=sr2*sr2+sr2)
            {
                b=sr2;
                flag=1;
            }

            else
            {
            b=sr2+1;
            flag=2;
            }
        }
        else if(a==sr2*sr2)
        {
            b=sr2;
            flag=2;
        }
        if(flag==1)
            ans=b*2;
        else
            ans=b*2-1;
        printf("%lld\n",ans);

    }
    return 0;
}
