t = input()
for i in range(t):
	a = '555'
	b = '33333'
	x = input()
	if x <= 5 :
		if x == 3:
			print a
		elif x == 5:
			print b
		else:
			print -1
		continue
	while (x > 0):
		n5 = 0
		n3 = 0
		i = 1
		x = x-(5**i)
		if x % 3 == 0:
			n5 = i
			n3 = x/3
			break
		i += 1
	if n5 == 0 and n3 == 0:
		print -1
	else:
		print a*n3 + b*n5
