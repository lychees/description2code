#include <iostream>
#include <string.h>
using namespace std;

int main()
{
	int n;
    cin>>n;
    char inputArray[30];
    
    while(n--){
    	
    	cin>>inputArray;
    
    	
    	for(int i = strlen(inputArray)-1; i >= 0; --i){
    		
    		cout<<inputArray[i];
    	}
    	cout<<endl;
    	
    }
    
    return 0;
}
