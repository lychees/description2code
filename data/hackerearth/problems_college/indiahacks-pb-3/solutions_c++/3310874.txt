#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,k,n,t,max,ans;
	cin>>t;
	while(t--)
	{
		ans=0;
		cin>>n>>k;
		int *arr=new int[n];
		int *dp=new int[n];
		for(i=0;i<n;i++)
		{
			cin>>arr[i];
		}
		for(i=0;i<=k;i++)
		{
			dp[i]=arr[i];
		}
		max=dp[0];
		for(i=k+1;i<n;i++)
		{
			if(max>0)
			{
				dp[i]=max+arr[i];
			}
			else
			{
				dp[i]=arr[i];
			}
			
			if(dp[i-k]>max)
			{
				max=dp[i-k];
			}
		}
		for(i=1;i<n;i++)
		{
			if(dp[i]>ans)
			{
				ans=dp[i];
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
