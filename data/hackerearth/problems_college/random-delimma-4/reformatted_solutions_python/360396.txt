t = input()
while t:
    k, n = map(int, raw_input().split())
    if (k > n):
        print '1.000000000'
    else:
        ans = 1.0
        tmp = n
        while k:
            ans = ans*(float(tmp)/n)
            tmp = tmp-1
            k = k-1
        ans = 1-ans
        print "%0.9f" % ans
    t = t-1
