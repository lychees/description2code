#include <bits/stdc++.h>
#define max 1000003
using namespace std;
int p[max] = { 0 };
void sieve()
{
	p[0] = p[1] = -1;
	int i, j;
	for (i = 2; i<(max); i++)
	{
		if (p[i] == 0)
		{
			for (j = i; j<max; j += i)
				p[j] = i;
		}
	}
}
int nd(int x)
{
	int count = 0;
	while (1)
	{
		if (x == 1)
			break;
		else
		{
			if (p[x] == 0)
				x = 1;
			else
			{
				x /= p[x];
				count++;
			}
		}
	}
	return count;
}

int main()
{
	sieve();
	long long s[max];
	s[0] = 0;
	s[1] = 0;
	s[2] = 1;
	for (int i = 3; i<max; i++)
		s[i] = s[i - 1] + nd(i);
	int t;
	scanf("%d", &t);
	while (t--)
	{
		int n;
		scanf("%d", &n);
		printf("%lld\n", s[n]);
	}
	return 0;
}
