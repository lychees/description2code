int digits(int a)
{	
	int c=0;
	while(a)
	{
	   c++;
	   a/=10;
	}
	return c;
}

int main()
{	
	int t,l,r,d;
	si(t);
	while(t--)
	{
		si(l);
		si(r);
		ll ans;
		d=digits(r);
		ll prod=1,val=1;
		for(int i=1;i<=d-1;i++)
		{
			prod*=10,val*=10;
		}
		val*=10;
		val--;
		prod*=5;
		prod--;
		if( r>=prod && l<=prod+1)
		ans=(ll)prod*(prod+1);
		else if(r < prod)
		ans=(ll)r*(val-r);
		else if( l> (prod+1))
		ans=(ll)l*(val-l);
		pl(ans);
		nl;
		
	}
}
