#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
   int n;cin>>n;
   map<int,int> m;
   for(int i=0;i<n;i++){
   	int t;cin>>t;
   	m[t]+=1;
   }
   int cnt=n;
   for(auto p:m){
   		cout<<cnt<<endl;
   		cnt-=p.second;
   }
   return 0;
}
