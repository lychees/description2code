#include <iostream>
using namespace std;
#include <vector>
#define ll long long

ll gcd(ll u,ll v)
{   if(u==v)
        return u;
    if(~u&1)
    {   if(v&1)
            return gcd(u>>1,v);
        else
            return gcd(u>>1,v>>1)<<1;
    }
    if(~v&1)
        return gcd(u,v>>1);
    if(u>v)
        return gcd((u-v)>>1,v);
    return gcd((v-u)>>1,u);
}

ll cop(ll a,ll b) 
{   if(gcd(a,b)==1)
        return 1;
    return 0;
}


int main()
{   ll n,q,i,j,s=0;
    cin>>n;
    vector<ll> z;
    for(i=0;i<n;++i)
    {   cin>>q;
        z.push_back(q);
    }
    for(i=0;i<n;++i)
        for(j=i+1;j<n;++j)
        {   if(z[i]>z[j])
            {   if(cop(z[i],z[j]))
                    s++;
            }
        }
    cout<<s<<"\n";
    return 0;
}
