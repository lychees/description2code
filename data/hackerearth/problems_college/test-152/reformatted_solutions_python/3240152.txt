t = int(raw_input())
for i in range(t):
    n = int(raw_input())
    s = raw_input()
    dict = {}
    for i in s:
        if i not in dict:
            dict.update({i: int(1)})
        else:
            dict[i] += 1
    m = 0
    for key in dict:
        if dict[key] > m:
            m = dict[key]
    if m % 2 == 0:
        print "Yes"
    else:
        print "No"
