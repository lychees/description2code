t=int(raw_input())
for j in range (0,t):
    sum=0
    num=int(raw_input())
    arr=list(map(int,raw_input().split()))
    arr.sort(reverse=True)
    for i in range(1,num):
        sum+=arr[0]-arr[i]
    temp1=sum
    for i in range(1,num-1):
        temp=temp1-(num-i)*(arr[i-1]-arr[i])
        sum=(sum+temp) 
        temp1=temp
    sum=sum%1000000007
    y=(sum*arr[0])%1000000007
    print y
