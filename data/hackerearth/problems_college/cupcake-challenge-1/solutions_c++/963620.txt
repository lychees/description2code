#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    int n = 3;
    int a[n];int nimsum = 0;int ans = 0;
    for(int i = 0 ; i < n ; ++i) { cin >> a[i] ; nimsum ^= a[i];}
    for(int i = 0 ; i < n ; ++i ) ans += (nimsum^a[i]) < a[i];
    if(ans > 0)
        cout << "NO BILL\n";
    else cout << "BILL\n";
    }
}
