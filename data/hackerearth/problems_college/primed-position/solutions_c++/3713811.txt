#include <bits/stdc++.h>
using namespace std;
#define MAX 1500005
#define MOD 1000000007
int a[MAX];
vector <int> prime;
vector <int> up;
void seive()
{
	a[1]=1;
	prime.emplace_back(1);
	for(int i=2;i<=MAX;++i)
	{
		if(a[i]==0)
		{
			prime.emplace_back(i);
			for(int j=2*i;j<=MAX;j+=i)
			{
				a[j]=1;
			}
		}
	}
	auto s =prime.size()+4;
	for(int i=1;i<=(s);++i)
	{
		if(a[i]==0)
		{
			up.emplace_back(prime[i]);
		}
	}
}
int main()
{
	seive();
	int t,n,m;
	long long x,y;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&m);
		x=up[n-1]%MOD;
		y=up[m-1]%MOD;
		x=(x*y)%MOD;
		printf("%lld\n",x);
	}
    return 0;
}
