#include <bits/stdc++.h>

using namespace std;

int arr[1000005];

int main()
{
	int num,sum;
	cin>>num>>sum;
	
	for(int i=0;i<num;i++)
		cin>>arr[i];
	
	sort(arr,arr+num);
	int flag=0;
	int low=0,high=num-1,temp;
	for(;low<high;)
	{
		temp=arr[low]+arr[high];
		//cout<<temp<<endl;
		if(sum==temp)
		{
			flag=1;
			break;
		}
		else if(sum>temp)
		{
			low++;
		}
		else
		{
			high--;
		}
		
	}
	if(flag==1)
		cout<<"YES"<<endl;
	else
		cout<<"NO"<<endl;

}
