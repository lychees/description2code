#include <iostream>
using namespace std;

string *words;
int k;
bool check(string s){
    int i;
    for(i=0;i<k;i++){
        if(s == words[i]) return true;
    }
    return false;
}
int main() {
	int i,l,c;
	cin>>k;
	words = new string[k];
	i=0;
	while(i < k){ // input words he dislikes
		cin>>words[i];
		i++;
	}
	cin>>l;
	string str[l];
	i=0;
	while(i < l){ // input sentence
		cin>>str[i];
		i++;
	}
	char ans[l];
	c=0,i=0;
	while(i < l){ //
        if(!check(str[i])){
            ans[c] = (char)toupper(str[i][0]);
            c++;
        }
        i++;
	}
	i=0;
	while(i<c-1){
        cout << ans[i] <<".";
        i++;
	}
    cout<<ans[i];
    return 0;
}
