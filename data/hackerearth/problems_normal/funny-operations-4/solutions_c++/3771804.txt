#include <iostream>
using namespace std;

int main()
{
    long long t,n,val[16],a,x,ans,j,i,len=0,num,l;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	cin>>val[i];
    	ans=-1;
    	for(i=0;i<1<<n;i++)
    	{
    		num=i;
    		a=-1;
    		x=0;
    		j=0;
    		l=0;
    		while(num>0)
    		{
    			if(num%2)
    			{
    				if(a<0)
    				a=val[j];
    				else
    				a=a&val[j];
    				x=x^val[j];
    				l++;
    			}
    			j++;
    			num=num/2;
    		}
    		if(ans<(a|x))
    		ans=(a|x),len=l;
    		else if(ans==(a|x) && l>len)
    		len=l;
    		
    	}
    	cout<<ans<<" "<<len<<"\n";
    }
    return 0;
}
