#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,m,i,d,D;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	i=n/m;
    	if(n==m*i)
    	cout<<n<<"\n";
    	else
    	{
    		d=n-(m*i);
    		D=(m*(i+1))-n;
    		if(d<D)
    		cout<<m*i<<"\n";
    		else 
    		cout<<m*(i+1)<<"\n";
    	}
    }
    return 0;
}
