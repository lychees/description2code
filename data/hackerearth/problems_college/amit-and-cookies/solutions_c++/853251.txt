#include <iostream>
using namespace std;

int main()
{
    int n,k,sum=0,count=0;
    cin>>n>>k;
    int rs[n];
    for(int i=0;i<n;i++){
    	
    	cin>>rs[i];
    }
    for(int j = 0;j<n;j++){
    	
    	for(int k = j+1;k<n;k++){
    		
    		if(rs[k]<rs[j]){
    			
    			rs[j] = rs[j]+rs[k];
    			rs[k] = rs[j] - rs[k];
    			rs[j] = rs[j] - rs[k];
    		}
    	}
    }
    
    for(int m = 0;m<n;m++){
    	
    	count++;
    	sum = rs[m] + sum;
    	if(sum>=k){
    		
    		break;
    	}
    }
    cout<<count-1;
}
