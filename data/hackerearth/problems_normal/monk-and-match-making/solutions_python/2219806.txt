s=raw_input()
l=len(s)
UPL=10**5
assert(1 <= l and l <= UPL)
t=int(input())
assert(1 <= t and t <= UPL)
for i in xrange(t):
	l1,r1,l2,r2=map(int, raw_input().split(' '))
	assert(1 <= l1 and l1 <= UPL)
	assert(1 <= r1 and r1 <= UPL)
	assert(1 <= l2 and l2 <= UPL)
	assert(1 <= r2 and r2 <= UPL)
	s1=s[l1-1:r1]
	s2=s[l2-1:r2]
	if s1==s2: print 'Yes'
	else: print 'No'
