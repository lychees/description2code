#include<bits/stdc++.h>
using namespace std;
int a[11],x[11];

int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		int i,j,n;
		scanf("%d",&n);
		for (i=0;i<n;i++)
			scanf("%d",a+i);
		for (i=0;i<n;i++)
			scanf("%d",x+i);
		int ans=0;
		for (i=0;i<n;i++)
		{
			ans=ans^(x[i]-1)^(a[i]-x[i]);
		}
		if (ans==0)
			printf("Second\n");
		else
			printf("First\n");
	}
	return 0;
}
