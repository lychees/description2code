#include <iostream>
#include <algorithm>
using namespace std;

int main()
{   int t;
   cin>>t;
    while(t--)
    {
    long long n,x,y;
    cin>>n>>x>>y;
    long long a[n];
    long long b[n];
    for(int i=0;i<n;i++)
    {
     cin>>a[i]>>b[i];	
    }
    sort(a,a+n);
    sort(b,b+n);
    long long mult=0;
for(int i=0;i<n;i++)
{
 
  if(a[i]>b[i])
  {
  	mult=mult+(y*(a[i]-b[i]));
  }
  else if(a[i]<b[i])
  {
   mult=mult+(x*(b[i]-a[i]));
  	
  }
	
}
  cout<<mult<<endl;  	
    	
    	
    }
}
