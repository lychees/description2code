#define MAXN 200005
typedef unsigned long long int uint64;
typedef long long int int64;
int64 dp[30000],dp1[30000];
int a[30000];
int main(){
	int t,n,i;
	cin>>t;
	while(t--){
		scanf("%d",&n);
		for(i=0;i<n;i++)
		scanf("%d",&a[i]);
		if(n==2){
			printf("%d\n",max(a[0],a[1]));
			continue;
		}
		dp[0]=a[0];
		dp[1]=max(a[0],a[1]);
		for(i=2;i<n-1;i++){
			dp[i]=max(dp[i-1],a[i]+dp[i-2]);
		}
		dp1[1]=a[1];
		dp1[2]=max(a[1],a[2]);
		for(i=3;i<n;i++){
			dp1[i]=max(dp1[i-1],dp1[i-2]+a[i]);
		}
		printf("%lld\n",max(dp1[n-1],dp[n-2]));
	}
	return 0;
}
