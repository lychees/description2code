#include <bits/stdc++.h>
using namespace std ;

#define inf 1111111

int N,E,T,e[111],t[111],d[111],dp[111][111][111] ;


int solve(int idx,int E,int T){
	
	if(E < 0 || T < 0)
		return -inf ; 
	if(idx <= 0 || E == 0 || T == 0)
		return 0 ; 
	int &ret = dp[idx][E][T] ;
	if(ret != -1)
		return ret ;
	ret = 0 ;
	ret = max(d[idx]+solve(idx-1,E-e[idx],T-t[idx]),solve(idx-1,E,T)) ;
	return ret ;
}

int main(){
	
	cin >> N >> E >> T ;
	int ans = 0 ;
	for(int i=1;i<=N;i++) cin >> e[i] ;
	for(int i=1;i<=N;i++) cin >> t[i] ;
	for(int i=1;i<=N;i++){ cin >> d[i] ; ans += d[i] ;}
	memset(dp,-1,sizeof dp) ;
	cout << ans - solve(N,E,T) << endl ;
	return 0 ; 
}
