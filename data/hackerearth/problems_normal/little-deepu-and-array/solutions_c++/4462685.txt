#include <bits/stdc++.h>
using namespace std;
int a[100005];
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)scanf("%d",&a[i]);
	int m,x;
	scanf("%d",&m);
	for(int i=0;i<m;i++)
	{
		scanf("%d",&x);
		for(int j=0;j<n;j++)
			if(a[j] > x)a[j]--;
	}
	for(int i=0;i<n;i++)printf("%d ",a[i]);
	return 0;
}
