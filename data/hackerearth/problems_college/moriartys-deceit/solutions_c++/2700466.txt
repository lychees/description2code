#include <iostream>
using namespace std;

long long int gcd(long long int a,long long int b){
	if(a==0)
		return b;
	return gcd(b%a,a);
}

int main()
{
    int t,i,n;
    long long int x;
    long long int *a;
    
    scanf("%d",&t);
    while(t--){
    	scanf("%d%lld",&n,&x);
    	a = (long long int *)malloc(sizeof(long long int)*n);
    	for(i=0;i<n;i++){
    		scanf("%lld",&a[i]);
    	}
    	
    	for(i=0;i<n;i++){
    		if(x%a[i]==0 || a[i]%x==0){
    			continue;
    		}else{
    			long long int g = gcd(x,a[i]);
    			if(g==1){
    				printf("%lld ",a[i]);
    			}else{
    				continue;
    			}
    		}
    	}
    	printf("\n");
    }
    
    return 0;
}
