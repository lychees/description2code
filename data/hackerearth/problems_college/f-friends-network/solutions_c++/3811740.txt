#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
    	int n;
    	cin  >> n;
    	vector< vector<int> > vec(n,vector<int> (n));
    	for(int i=0;i<n;i++){
    		for(int j=0;j<n;j++){
    			cin >> vec[i][j];
    		}
    	}
    	bool flag=0;
    	for(int i=0;i<n;i++){
    		for(int j=0;j<n;j++){
    			for(int k=0;k<n;k++){
    				if(vec[i][j] == 1 && vec[j][k]==1 && vec[k][i]==1){
    					flag=1;
    				}
    			}
    		}
    	}
    	if(flag){
    		cout << "Aye\n";
 
    	}
    	else cout << "Nae\n";
    }
    return 0;
}
