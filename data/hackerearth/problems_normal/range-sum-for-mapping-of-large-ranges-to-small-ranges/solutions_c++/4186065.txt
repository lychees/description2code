#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mset(m,v) memset(m,v,sizeof(m))
#define pb push_back
#define mp make_pair
#define fi first
#define se second
#define pi 3.1415926535897932384
#define mod 1000000007
#define MAXM 1000009
 
vector<int> v[MAXM];
bool visit[MAXM];
ll cnt=0;
 
void init()
{
    for(int i=0;i<MAXM;i++)
    {
        visit[i]=false;
        v[i].clear();
    }
}
 
void dfs(int x)
{
    visit[x]=true;
    cnt++;
    for(int i=0;i<v[x].size();i++)
    {
        if(!visit[v[x][i]])
        {
            dfs(v[x][i]);
        }
    }
}
 
int main()
{
    int t,n,k,i,j,x,y;
    cin>>t;
    while(t--)
    {
        ll ans=1;
        init();
        cin>>n>>k;
        for(i=0;i<k;i++)
        {
            cin>>x>>y;
            v[x].pb(y);
            v[y].pb(x);
        }
 
        for(i=1;i<=n;i++)
        {
            if(!visit[i])
            {
                cnt=0;
                dfs(i);
                if(cnt>1){
                    ans=(ans*((1LL*cnt*(cnt-1))/2)%mod)%mod;
                }
            }
        }
 
        cout<<ans<<endl;
    }
    return 0;
}
