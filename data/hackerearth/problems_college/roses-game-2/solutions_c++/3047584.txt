#include<iostream>
#include<math.h>

using namespace std;
#define ll long long int

int check(ll arr[],ll visited[],ll i,ll n)
{
	if(i>=n)
		return 0;

	if(visited[i]==1)
		return 0;
	if(arr[i]==-1)
		return 1;

	visited[i]=1;

	return check(arr,visited,arr[i],n);

}

int main() 
{
	ll i,j,n,t;

	cin>>t;
	while(t--)
	{
		cin>>n;

		ll a[n],visited[n];

		for(i=0;i<n;i++)
		{
			cin>>a[i];
			visited[i]=0;
		}

		if(check(a,visited,0,n))
			cout<<"Win\n";
		else cout<<"Lose\n";

	}
	return 0;
}
