#include <queue>
#include <vector>
#include <stdio.h>
#define SIZE 100010

using namespace std;

vector<int> edge[SIZE];
bool visited[SIZE];

void bfs(int node) {
	queue<int> q;
	q.push(node);
	
	while(q.size()>0) {
		
		int cur = q.front();
		q.pop();
		if(visited[cur]) continue;
		
		visited[cur] = true;
		
		for(int i=0; i<edge[cur].size(); i++) {
			if(!visited[edge[cur][i]]) q.push(edge[cur][i]);
		}
	}
	
	return;
}

int main()
{
    int test;
    scanf("%d",&test);
    
    while(test--) {
    	int n,m;
    	scanf("%d%d",&n,&m);
    	
    	for(int i=0; i<n; i++) {
    		visited[i] = false;
    		edge[i].clear();
    	}
    	
    	for(int i=0; i<m; i++) {
    		int a,b;
    		scanf("%d %d",&a,&b);
    		edge[a].push_back(b);
    		edge[b].push_back(a);
    	}
    	
    	int ans = 0;
    	for(int i=0; i<n; i++) {
    		if(visited[i]) continue;
    		ans++;
    		bfs(i);
    	}
    	printf("%d\n",ans);
    }
    return 0;
}
