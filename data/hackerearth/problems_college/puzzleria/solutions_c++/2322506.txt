#include <iostream>
using namespace std;
long long int fa(int a)
{
	long long int f=1;
	for(int i=1;i<=a;i++)
	{
		f=f*i;
	}
	return f;
}
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
  	int a,b;
  	long long int x;
  	cin>>a>>b;
  	if(a<=7&&b<=5)
  	{
  		x=(5040/(fa(a)*fa(7-a)))*(120/(fa(b)*fa(5-b)))*fa(a+b);
  		cout<<x<<endl;
  	}
  	
  }
    return 0;
}
