#include <iostream>
using namespace std;

int main()
{
	int t, n, p, i, pos;
	long long int r[1000], min, sum;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    		cin>>r[i];
    	min = r[0];
    	pos = 0;
    	for(i=0;i<n;i++)
    		if(r[i]<min)
    			{
    				min = r[i];
    				pos = i;
    			}
    	sum = 0;
    	for(i=0;i<n;i++)
    	{
    		if(i != pos)
    			sum += r[i];
    	}
    	sum += (n-1)*min;
    	cout<<sum<<endl;
    }
    return 0;
}
