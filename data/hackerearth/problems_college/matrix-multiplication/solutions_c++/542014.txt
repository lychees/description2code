#include <iostream>

using namespace std;

int main()
{
        int test_cases;
        cin>>test_cases;
        while(test_cases > 0)
        {		int n=0;
                int X1,X2,Y1,Y2;
                cin>>X1>>Y1>>X2>>Y2;
                if(Y1==X2)	n++;
                if(X1==Y2)	n++;
                cout<<n<<endl;
                test_cases --;
        }
        return 0;
}
