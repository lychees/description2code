#include<bits/stdc++.h>
using namespace std;

int main(){
    int t,n,i,j,flag = 0;
    string str;
    cin>>t;
    while(t--){
        cin>>str;
        cin>>n;
        char misd[n];
        for(i=0;i<n;i++){
            cin>>misd[i];
        }
        int sum = 0;
        for(i=0;i<str.length();i++){
            for(j=0;j<n;j++){
                if(str[i]==misd[j]){
                    flag = 1;
                    break;
                }
            }
            if(flag==0){
                sum += str[i]-'a'+1;
            }
            flag = 0;
        }
        cout<<sum<<endl;

    }
    return 0;
}
