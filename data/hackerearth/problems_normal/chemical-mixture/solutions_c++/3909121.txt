#define LL long long

const int N = 1e5+5;
LL a[N];
LL b[N];
std::vector< LL > v;
LL n1,n2,m;
vector<LL> solve(){
	vector<LL> ans;
	sort(a,a+n1);
	for(int i = 0 ; i < v.size();i++) {
		long long sum = v[i];
		long long req = m-sum;
		if(req<0) continue;
		ans.clear(); 
		for(int i = n1-1 ; i >=0 ; i--){
			if(req>=a[i]) {
				req = req-a[i];
				ans.pb(a[i]);
			}
			if(req == 0){
				return ans;
			}
		}
	}
	ans.clear();
	ans.pb(-1);
	return ans;
}
int main() {
	int t;
	cin >> t;
	while(t--){
		v.clear();
		scl(m);scl(n1);scl(n2);
		F(i,0,n1) scl(a[i]);
		F(i,0,n2) scl(b[i]);
		int p = (1<<n2);
		F(i,0,p){
			LL sum = 0;
			F(j,0,n2){
				if(i&(1<<j)) {
					sum+=b[j];
				}
			}
			v.pb(sum);
		}
		sort(v.begin(),v.end());
//		F(i,0,v.size()){
//			cout<<v[i]<<" ";
//		}
//		nl;
		std::vector< LL > ans = solve();
		if(ans[0] == -1){
			puts("NO");
			continue;
		}
		else {
			puts("YES");
			sort(ans.begin(),ans.end());
			F(i,0,ans.size()){
				cout<<ans[i]<<" ";
			}
		}
		nl;
	}
}
