#include <iostream>
#define ll long long
using namespace std;

int main()
{
    ll t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        ll a[n+1],sum=0;
        a[0]=0;
        for(ll i=1;i<=n;i++)
        {
         cin>>a[i];
         sum+=a[i];
        }
        ll dp[102][10002];
        for(ll i=0;i<=n;i++){
            for(ll j=0;j<=sum;j++){
                 dp[i][j]=0;
                if(j==0)dp[i][j]=1;
            }
           
        }
        for(ll i=1;i<=n;i++)
        {
            for(ll j=1;j<=sum;j++)
            {
                
                if(dp[i-1][j]==1)
                dp[i][j]=1;
                else if(j-a[i]>=0)
                {
                    if(dp[i-1][j-a[i]]==1)
                    dp[i][j]=1;
                }
            }
        }
        ll c=0;
         for(ll i=0;i<=sum;i++){
            
                if(dp[n][i]==1)c++;
        }
        cout<<c<<endl;
    }
    return 0;
}
