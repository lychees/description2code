#include<bits/stdc++.h>
#include<iostream>

using namespace std;

int lis(int* arr,int n)
{
	if(n<0)
	return 0;
	
	int ans=0,i;
	for(i=n;i>=0;i--)
	{
		if(arr[i]<=arr[n+1])
		{
			ans = max(ans,1+lis(arr,i-1));
		}
	}
	return ans;
}

int main()
{
	int n,i;
	vector<int> v;
	int arr[105];
	scanf("%d",&n);
	for(i=0;i<n;i++)
	scanf("%d",&arr[i]);
	int ans=0,dig;
	for(i=n-1;i>=0;i--)
	{
		ans = max(ans,1+lis(arr,i-1));
	}
	while(ans>0)
	{
		dig = ans%2;
		ans/=2;
		v.push_back(dig);
	}
	for(i=v.size()-1;i>=0;i--)
	printf("%d",v[i]);
	return 0;
}
