#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int t;
	cin>>t;
	for(int u=0;u<t;u++){
		long n;
		cin>>n;
		long a[n];
		for(long i=0;i<n;i++)cin>>a[i];
		sort(a,a+n);
		long long int s=0;long l=1;
		for(int i=1;i<n;i++){
			if(a[i]==a[i-1])l=l+1;
			else{s=s+l+l*(l-1)/2;l=1;
			}
		}
		s=s+l+l*(l-1)/2;
		cout<<s<<endl;
	}
}
