#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include<cstring>
using namespace std;

int main()
{
	int n,i,l,j;
	cin>>n;
	char a[101];
	int b[27]={0},count=0;
	for(i=0;i<n;i++)
	{
		cin>>a;
		l=strlen(a);
		
		for(j=0;j<l;j++)
		{
			if(b[a[j]-96]==i)
				b[a[j]-96]++;
		}
	}
	for(i=0;i<27;i++)
	{
		if(b[i]==n)
			count++;
	}
	cout <<count<<endl;
return 0;
}
