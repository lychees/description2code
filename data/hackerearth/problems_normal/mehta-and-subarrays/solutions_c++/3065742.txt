#include <bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

const int lmt = 1000005;
vector<pair<lli,int>> in;

int main(){
    int n;
    scanf("%d",&n);
    lli sum = 0,x;
    for(int i=1;i<=n;i++){
        scanf("%lld",&x);
        sum += x;
        in.pb({sum,i});
    }
    if(n==1){
        if(in[0].first >= 0)
            printf("1 1\n");
        else
            printf("-1\n");
        return 0;
    }
    in.pb({0,0});
    sort(in.begin(),in.end());
    int mn = in[0].second;
    int ans=0,cnt=0;
    if(in[0].first>=0){
        ans = 1;
        cnt = 1;
    }
//    DEBUG(mn);
//    DEBUG(ans);
//    DEBUG(cnt);
    for(int i=1;i<in.size();i++){
        int tmp = in[i].second - mn ;
        if(tmp>ans){
            ans = tmp;
            cnt = 1;
        }else if(tmp==ans && tmp>0){
            cnt++;
        }
        mn = min(mn,in[i].second);
//        DEBUG(ans);
//        DEBUG(mn);
//        DEBUG(cnt);
    }
    if(ans>0)
        printf("%d %d\n",ans,cnt);
    else
        printf("-1\n");
    return 0;
}
