#include <iostream>
#include <stack>
using namespace std;

int main()
{
	unsigned long q,cost = 0;
	stack<unsigned long> c;
	unsigned int query_type;
	
	cin>>q;
	for(unsigned long i=0;i<q;i++){
		cin>>query_type;
		if(query_type == 1){
			if(c.empty()){
				cout<<"No Food\n";
			}else{
				cout<<c.top()<<std::endl;
				c.pop();
			}
		}else{
			cin>>cost;
			c.push(cost);
		}
	}
    return 0;
}
