#include<bits/stdc++.h>
using namespace std;

#define ll long long

ll p[1010];

int main(){
	int t;
	cin>>t;
	while(t--){
        int n,k;
		cin>>n>>k;
		ll sum=0;
		for(int i=0;i<n;i+=1){
			scanf("%lld",&p[i]);
			sum+=p[i];
		}
//		sort(p,p+n);
		int dp[sum+5];
		memset(dp,0,sizeof(dp));
		dp[0]=1;
        //cout<<sum<<endl;
		for(int i=0;i<n;i+=1){
			for(int j=sum;j>=p[i];j-=1)
				dp[j]|=dp[j-p[i]];
		}
		if(dp[k]) cout<<"Yes"<<endl;
		else{
			//cout<<"No"<<endl;
			int i,j;
			i=j=k;
			while(dp[i]==0)i-=1;
			while(dp[j]==0)j+=1;
			if(j-k<=k-i)
    		cout<<"No"<<endl<<j<<endl;
    		else
    		cout<<"No"<<endl<<i<<endl;
		}

	}
	return 0;
}
