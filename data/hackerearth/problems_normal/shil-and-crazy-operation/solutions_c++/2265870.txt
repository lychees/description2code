#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios::sync_with_stdio(0);
    int n; long long t; cin>>n>>t;
    vector<int> v(n+1),p(n+1),hash(n+1),res(n+1);
    vector<bool> flag(n+1,0);
    for(int i = 1; i <= n; ++i)
    	cin>>v[i];
    for(int i = 1; i <= n; ++i)
    {
    	cin>>p[i];
    	hash[p[i]] = i;
    }
    int i = 1;
    while(i <= n)
    {
    	if(!flag[i])
    	{
    		int j = i;
    		vector<int> temp;
    		do
    		{
    			flag[j] = 1;
    			temp.push_back(j);
    			j = hash[j];
    		}while(p[j] != p[i]);
    		
    		for(int j = 0; j < temp.size(); ++j)
    		{
    			res[temp[((int)(t%(temp.size())) + j) % (temp.size())]] = v[temp[j]];
    		}
    	}
    	i++;
    }
    for(int i = 1; i <= n; ++i)
    cout<<res[i]<<" ";
    return 0;
}
