#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int i,j,k,n,t;
    scanf("%lld",&t);
    while(t--){
    scanf("%lld",&n);
    long long int a[n];
    for(i=0;i<n;i++)
    {
        scanf("%lld",&a[i]);
    }
    long long int c=0;
    long long int c1=0;
    for(i=0;i<n;i++)
    {
        if(a[i]%2!=0)
            c++;
            if(a[i]%2==0)
                c1++;
    }
    long long int ans;
    ans=c*c1;
    printf("%lld\n",ans);
    }
    return 0;
}
