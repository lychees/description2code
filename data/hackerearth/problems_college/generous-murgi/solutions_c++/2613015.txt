#include <iostream>
#include <cmath>
using namespace std;

int factorize(int n,int y){
    int sq=sqrt(n);
    int c=0;
    int i=1;
    while (i<=sq){
        if (n%i==0){
            if (i>y) c++;
            if (i!=(n/i))
            if ((n/i)>y) c++;
        }
        i++;
    }
    return c;
}

int main(){
    int t,x,y;
    cin >> t;
    while (t--){
        cin >> x >> y;
        if (x==y)
            cout << "-1\n";
        else
            cout << factorize(x-y,y) << endl;
    }
    return 0;
}
