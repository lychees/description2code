noT = int(raw_input())
while noT :
    noT-=1
    n,c,m = map(int,raw_input().split())
    bought = n//c    
    available = bought
    while available >= m:  
        reimb = available / m
        bought +=  reimb 
        ext   = available % m
        available = reimb + ext
    print bought
