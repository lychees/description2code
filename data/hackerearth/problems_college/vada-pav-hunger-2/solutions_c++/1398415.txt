#include<bits/stdc++.h>
using namespace std;
int v[1010][1010];
int visit[1010];
int sum[1010],ans=0;
void dfs(int p)
{
	//cout<<p<<endl;
	visit[p]=1;
	int i,c=0,mini=0,x=0;	
	for(i=1;i<=1000;i++)
	{
		if(!visit[i]&&v[p][i])
		{
			sum[i]+=v[p][i];
			//c++;
			//cout<<i<<" "<<sum[i]<<endl; 
			dfs(i);
		}
	}
	for(i=1;i<=1000;i++)
	{
		if(!visit[i]&&v[p][i])
		{
			x+=sum[i];
			mini=max(mini,sum[i]);
		}
		
		//cout<<p<<" "<<sum[p]<<endl;
	}
	sum[p]+=mini;
	visit[p]=0;
}
void dfss(int p)
{
	int i,x=0,mini=0;
	visit[p]=1;
	for(i=1;i<=1000;i++)
	{
		if(!visit[i]&&v[p][i])
		{
			//sum[i]+=v[p][i];
			//c++;
			//cout<<i<<" "<<sum[i]<<endl; 
			dfss(i);
		}
	}
	for(i=1;i<=1000;i++)
	{
		if(!visit[i]&&v[p][i])
		{
			x+=sum[i];
			mini=max(mini,sum[i]);
		}
		
		//cout<<p<<" "<<sum[p]<<endl;
	}	
	ans+=2*(x-mini);
	//cout<<p<<" "<<ans<<endl;
	visit[p]=0;
}
int main()
{
	int n,i,x,y,z;
	scanf("%d",&n);
	for(i=0;i<n-1;i++)
	{
		scanf("%d%d",&x,&y);
		v[x][y]=v[y][x]=1;
	}
	dfs(1);
	memset(visit,0,sizeof(visit));
	dfss(1);
	printf("%d\n",sum[1]+ans);	
	//for(i=1;i<=n;i++)
	//	cout<<sum[i]<<endl;
	return 0;
}
