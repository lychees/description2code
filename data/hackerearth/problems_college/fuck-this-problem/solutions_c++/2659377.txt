#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    //cout << "Hello World!" << endl;
    string str;
    int l;
    getline(cin,str);
    int len=str.length();
    cin >> l;
    for(int i=0; i < str.length(); i++)
    {
    	string s1;
    	s1=str.substr(i*l,l);
    	cout << s1 <<"\n";
    	string s2=str.substr(i*l+l,len-(i*l+l));
    	if(s2.length() < l)
    	{
    	cout << s2 <<"\n";	
    	break;
    	}
    }
    return 0;
}
