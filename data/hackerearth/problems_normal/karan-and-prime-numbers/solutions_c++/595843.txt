#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;
int main()
{
    int t,flag=0;
    unsigned long int a,b;
    unsigned long long res=0,no;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        res=0;
        cin>>a>>b;
        no=a;
        if(a==1)
        no=no+1;
        while(no<=b)
        {
            //if(2>sqrt(no))
              //  res=res+no;
            for(unsigned long j=2;j<=sqrt(no);j++)
            {
                if(no%j==0)
                {
                    flag=1;
                    break;
                }
            }
            if(flag==0)
                res+=no;
            flag=0;
            no++;
        }
        cout<<res<<endl;
    }
}
