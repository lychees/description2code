#include<bits/stdc++.h>
using namespace std;
 
#define fr(i,n) for(int i=0;i<n;i++)
#define fr1(i,n) for(int i=1;i<=n;i++)
#define ll long long
#define pii pair<int,int>
#define ff first
#define ss second
#define mp make_pair
#define pb push_back
#define MOD 1000000007
#define MAX 100002
#define INF 100005

int main(){
	int n;
	double p;
	cin>>n>>p;
	double dp[n+1];
	memset(dp,0,sizeof(dp));
	dp[0]=1.0;
	p=(1.0*p)/100.0;
	fr1(i,n){
		if(i-2>=0)dp[i]+=(p)*dp[i-2];
		if(i-3>=0)dp[i]+=(1-p)*dp[i-3];
	}
	printf("%.6f",dp[n]);
	return 0;
}
