#include<bits/stdc++.h>
#define MAX 100001
#define gc getchar_unlocked
using namespace std;

inline void inp(int *n )
{
*n=0;
int ch=gc();int sign=1;
while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=gc();}
while( ch >= '0' && ch <= '9' )
*n = (*n<<3)+(*n<<1) + ch-'0', ch=gc();
*n=*n*sign;
}

#define max 100000
bitset <max+1> p;
int size;
int prime[9600];
void sieve()
 {
      size=0;
      long long i,j;
      p.set(0,1);
      p.set(1,1);
      prime[size++]=2;
      for(i=3;i<max+1;i=i+2)
      {
         if(!p.test(i))
         {
            prime[size++]=i;
            for(j=i;j*i<max+1;j++)
               p.set(j*i,1);
         }
      }
 }
 
int fact(int n)
{
    int ans=0;
    map<int,int> m;
    for(int i=0;(prime[i]*prime[i])<=n;i++)
    {
        int c=0;
        if(n!=1)
        {
            int exp = 0;
            while(!(n % prime[i]))
            {
                n/= prime[i];
                exp++;
            }
            if(exp>c) c=exp;
        }
        if(c) m[prime[i]]=c;
    }
    if(n>1) m[n]=1;
    for(map<int,int> :: iterator it=m.begin(),f=m.end(); it!=f; it++)
    {
        ans+=((*it).second);
    }
    return ans;
}

int main()
{
    int t,i,a[MAX],ans=0;
    int res=0;
    sieve();
    inp(&t);
    for(i=0;i<t;i++) 
    {
    	inp(&a[i]);
    	res^=fact(a[i]);
    }	
    if(res) cout<<"ALICE";
    else cout<<"BOB";
 return 0;
}
