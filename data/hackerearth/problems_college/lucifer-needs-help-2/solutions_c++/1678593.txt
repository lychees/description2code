ll a[1000001],update[1000001]={0};
int main()
{
    int n,i,m,I,J;
    si(n);
//    ll a[n+1];
    for(i=1;i<=n;++i)
        sll(a[i]);
    si(m);
//    ll update[n+1],k;
    ll k;
//    memset(update,0,sizeof(update));
    for(i=0;i<m;++i)
    {
        si(I); si(J); sll(k);
        update[I]+=k;
        if(J+1 <= n)
        update[J+1]+= -k;
    }
    ll sum=0;
    for(i=1;i<=n;++i)
    {
        sum=sum+update[i];
        pll(a[i]+sum);
        nl;
    }
    return 0;
}
