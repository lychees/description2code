#include <cstdio>

#include <bits/stdc++.h>
using namespace std; 
int j(int n, int k)
{
  if (n == 1)
    return 1;
  else
    
    return (j(n - 1, k) + k-1) % n + 1;
}
 
// Driver Program to test above function
int main()
{
  long int t, n,m;
  cin >> t;
  while(t--) {
  	cin >> n;
  	m = j(n,2);
  	cout << m << endl;
  }
  return 0;
}
