int n,k;
int a[111111];

bool good(int r){
	int start = 0;
	REP(i,k){
		int end = start;
		while(end<n && a[end]<=a[start]+r)end++;
		if(end==n) return true;
		start=end-1;
	}
	return false;
}

			if(good(s))en=s-1,best=s;
