#include<iostream>
#include<bits/stdc++.h>
using namespace std;

bool comp_func(const pair<long long int,long long int>& a, const pair<long long int,long long int>& b){
    return a.second<b.second;
}

int main(){
    long long int t, n, m;
    long long int g, p;
    cin>>t;
    for (long long int i=1; i<=t; ++i){
    vector<pair<long long int, long long int> > x;
        cin>>n>>m;
        for (long long int j=0; j<n; ++j){
            cin>>g>>p;
            x.push_back(make_pair(g,p));
        }
        sort(x.begin(), x.end(), comp_func);
        long long int pow=m;
        int flag=0;
        typedef vector<pair<long long int,long long int> >::iterator itt;
        for(itt iterator=x.begin();iterator!=x.end();iterator++){
            if (pow > iterator->second)
                pow = pow+ iterator->first;
            else{
                flag = 1;
                break;
            }
        }
        if (flag == 1)
            cout<<"NO\n";
        else cout<<"YES\n";
    }
    return 0;
}
