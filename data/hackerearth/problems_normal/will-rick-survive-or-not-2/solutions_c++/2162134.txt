#include <iostream>
#include <vector>
#include <cstdio>
#include <cstdlib>
#include <algorithm>

#define gc getchar_unlocked
#define pc(x) putchar_unlocked(x)
using namespace std;

inline int readInt(){
	
	char c;
	int ans=0;
	do{
		c=gc();
	}while(c<'0' || c>'9');
	do{
		ans = (ans<<3) + (ans<<1) + (c-48);
		c=gc();
	}while(c>='0' && c<='9');
	return ans;
	
}


int main()
{
    std::ios::sync_with_stdio(false);
    int t;

    			flag=0;
    			break;
    		}
    		
    		if((i+1)%6 == 0)
    			round++;
    			
    	}
