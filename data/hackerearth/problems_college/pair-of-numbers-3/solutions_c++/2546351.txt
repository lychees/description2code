#include<bits/stdc++.h>

using namespace std;

int main()
{
	long long int t,n,i,j;
	cin>>t;
	while(t--)
	{
		int count=0;
		cin>>n;
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=n;j++)
			{
				if((i*j)%(i+j)==0  && i<j)
				{
					//cout<<i<<j<<endl;
					count++;
				}
			}
		}
		
		
		cout<<count<<endl;
		
	}
}
