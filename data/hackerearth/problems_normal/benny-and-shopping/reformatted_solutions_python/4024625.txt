t = int(raw_input())
for j in range(t):
    s = raw_input()
    # print(s)
    for i in range(len(s)):
        if (s[i] == '$'):
            index = i
            break
    s1 = ""
    for i in range(index+1, len(s)):
        if ((ord(s[i]) >= 48 and ord(s[i]) <= 57) or ord(s[i]) == 32):
            if (ord(s[i]) != 32):
                s1 += s[i]
        else:
            break
    print("$"+str(int(s1)))
