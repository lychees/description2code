//Game of Numbers
#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
	int test, start, end, i, index, temp, cnt;
    scanf("%d", &test);
    while(test--)
    {
    	scanf("%d %d", &start, &end);
    	int *arr;
		arr = new int[1000001];
    	for(i=2, index=i*i;index <= end;i++, index=i*i)
    	{
    		temp = start/index;
    		if(start % index != 0)
    			temp++;
    		while(index*temp <= end)
				arr[(index*temp++)-start] = 1;
    	}
    	for(i=0, cnt=0;i <= (end-start);i++)
			if(arr[i] == 0)
				cnt++;
    	printf("%d\n", cnt);
    }
    return 0;
}
