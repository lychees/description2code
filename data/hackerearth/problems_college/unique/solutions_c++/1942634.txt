# include<bits/stdc++.h>
using namespace std;

int main()
{
    set<int> S;
    int t,temp,n;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(int i=0;i<n;i++)
        {
           scanf("%d",&temp);
           S.insert(temp);
        }
        printf("%d\n",S.size());
        S.clear();
    }


}
