#include<iostream>

using namespace std;

long long cmb[1005][1005];

int main(){
	ios_base::sync_with_stdio(false);

    long long n, m = 0;

    cin >> n;

    cmb[2][0] = 1; cmb[2][1] = 2; cmb[2][2] = 1;
    for(int i = 3; i <= n; ++i){
        m += cmb[i-1][2] * 2;

        cmb[i][0] = 1;
        for(int j = 1; j <= 2; ++j){
            cmb[i][j] = cmb[i-1][j-1] + cmb[i-1][j];
        }
    }

    cout << m;

return 0;
}
