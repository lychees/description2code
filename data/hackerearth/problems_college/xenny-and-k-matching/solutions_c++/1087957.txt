#include <iostream>
using namespace std;

int main()
{
	int n,k;
	cin >> n >> k;
	int* l = new int[n];
	for (int i= 0; i<n; i++){
		cin >> l[i];
	}
	int t= 0;
	for (int i = 0; i < n; i++){
		if (i+k < n){
			if ((l[i+k] - l[i])% k == 0 || (l[i+k] + l[i])% k == 0)
				t++;
		}
	}
	cout << t << endl;
    return 0;
}
