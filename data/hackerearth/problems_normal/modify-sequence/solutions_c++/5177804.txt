#include <iostream>
using namespace std;

int main()
{
    int n,sub=0,i;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    if(n==1)
    {   if(a[0]==0)
    {
    	sub=0;
    }
    else
    {
    	sub=1;
    }
    	
    }
    else
    {
    sub=a[1]-a[0];
    for(i=2;i<n;i++)
    {
    	sub=a[i]-sub;
    }
    }
    if(sub==0)
    {
    	cout<<"YES";
    }
    else
    {
    	cout<<"NO";
    }
    
    return 0;
}
