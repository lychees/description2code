#include <iostream>
using namespace std;

int main()
{
    long int t,n;
    cin>>t;
    while(t--)
    {
    	long int f[107]={0};
    	cin>>n;
    	long int a[n],i;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	for(i=0;i<n;i++)
    	f[a[i]-0]++;
    	long int sum=0;
    	for(i=0;i<107;i++)
    	if(f[i]>1)
    	sum+=f[i]-1;
    	cout<<sum<<endl;
    }
    return 0;
}
