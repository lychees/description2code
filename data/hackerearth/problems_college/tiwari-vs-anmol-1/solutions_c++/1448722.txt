#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <vector>
#include <algorithm>
#define MAX 1000011
using namespace std;
int a[MAX],sieve[MAX],cumm[MAX];
void make_sieve(){
    sieve[1] = sieve[0] = 1;
    for(int i=2;i*i<MAX;i++){
        if(!sieve[i])
            for(int j=i*i;j<MAX;j+=i)
                sieve[j] = 1;
    }
}
void factorsieve()
{
    for(int i=1;i<MAX;i++)
        a[i]=i;

    for(int i=2;i*i<=MAX;i++)
    {
        if(a[i] == i)
        {
            for(int j=i*i;j<MAX;j+=i)
            {
                if(a[j] == j)
                {

                    a[j]=i;

                }
            }
        }

    }

    for(int i=1;i<MAX;i++){
        if(!sieve[i])//prime
            cumm[i] = cumm[i-1];
        else
            cumm[i] = cumm[i-1] + a[i];
    }
}
int main()
{
    int l,r;
    make_sieve();
    factorsieve();
    int t;
    cin>>t;
    while(t--){
        cin>>l>>r;
        cout<<cumm[r] - cumm[l-1]<<endl;
    }
    return 0;
}
