#define ll long long int

using namespace std;

int main()
{
	ll t;
	
	cin >> t;
	
	while (t--) {
		ll a;
		ll b;
		cin >> a >> b;
		
		ll ans = -1;
		
		for (ll i = a; i <= b; i++) {
			
			for (ll j = i + 1; j <= b; j++) {
				ans = max(ans, i ^ j);	
			}
			
		}
		
		cout << ans << endl;
	}
	return 0;
}
