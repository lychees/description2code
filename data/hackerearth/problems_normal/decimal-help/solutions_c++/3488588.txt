#include <iostream>
using namespace std;
 
int main()
{
    int t,digit,rem;
    cin>>t;
    while(t--)
    {
    	int n,d,r;
    	cin>>n>>d>>r;
    	rem=n%d;
    	while(r--)
    	{
    		rem=rem*10;
    		digit=rem/d;
    		rem=rem%d;
    	}
    	cout<<digit<<"\n";
    }
    return 0;
}
