#include <bits/stdc++.h>

using namespace std;

int DP[5005][5005];

bool isvowel(char c) {
	switch(c) {
		case 'a':
		case 'e':
		case 'i':
		case 'o':
		case 'u': return true;
		default: return false;
	}
}

int LCS(string harsh, string akshara) {
	int n = harsh.size();
	int m = akshara.size();
	for(int i = 1; i <= n; i ++) {
		for(int j = 1; j <= m; j ++) {
			if(harsh[i - 1] == akshara[j - 1] && !isvowel(harsh[i - 1])) {
				DP[i][j] = DP[i - 1][j - 1] + 1;
			} else {
				DP[i][j] = max(DP[i - 1][j], DP[i][j - 1]);
			}
		}
	}
	return DP[n][m];
}

int main() {
    string harsh, akshara;
    cin >> harsh >> akshara;
    cout << LCS(harsh, akshara);
    return 0;
}
