#include <iostream>
#include<vector>
using namespace std;
vector<int> v[10001];
int a[10001]={0};
int dp[10001];
int n;
int findpath(int i)
{
	int j,ans=0;
	if(i==n)
	return 1;
	if(a[i])
	return dp[i];
	a[i]=1;
	for(j=0;j<v[i].size();j++)
	{
		ans=ans+findpath(v[i][j]);
	}
	dp[i]=ans;
	return ans;
}

int main()
{
    //printf("Hello World!\n");
    int i,c,d;
    scanf("%d",&n);
   
    int count=0;
    
   // a[1]=1;
    do
    {
    	scanf("%d%d",&c,&d);
    	v[c].push_back(d);
    	
    }while(c!=0&&d!=0);
    if(n==1)
    printf("1\n");
    else
    {
    	count=findpath(1);
    
    printf("%d\n",count);
    }
    return 0;
}
