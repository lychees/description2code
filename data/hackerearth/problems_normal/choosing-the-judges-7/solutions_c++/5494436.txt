#include<iostream>
using namespace std;
int lookup[10005];
void init(int n)
{
    for(int i=0;i<n;i++)
    {
        lookup[i]=-1;
    }
}
long long int maxsum(int i,long long int a[],int n)
{
    if(i>=n)
        return 0;
    else if(lookup[i]!=-1)
        return lookup[i];
    else
        return lookup[i]=max(maxsum(i+1,a,n),a[i]+maxsum(i+2,a,n));
}
int main()
{
    int k=1;
    int t,n,i;
    cin>>t;
    while(t--)
    {
        cin>>n;
        init(n);
        long long int a[n];
        for(i=0;i<n;i++)
        {
            cin>>a[i];
        }
        long long int ans=maxsum(0,a,n);
        cout<<"Case "<<k<<": "<<ans<<endl;
        k++;

    }
}
