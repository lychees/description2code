#include<iostream>
#include <cassert>
using namespace std;

char str[11];
int palindrome ()
{
	int l,i;
    for(l = 0; str[l] != '\0'; l++);
    for(i = 0; (i < l/2) && (str[i] == str[l - i - 1]); i++);
    if(i == l/2)
        return 1;
    else
        return 0;
	
}
int main( )
{
    int t,flag,i;
    long long prod;
	cin >>t;
	assert(t>=1 && t<=100);
	while(t--)
	{
    	prod = 1;
    	flag = 2;
    	cin>>str;
		flag = palindrome();
		if(flag == 1)
			cout<<"Palindrome"<<endl;
		else
		{ 
		
			for(i = 0; str[i] != '\0'; i++)
			{
				prod = prod * ((int)str[i] - 96);
			}
			cout<<prod<<endl;
		}
	}
	
    
    return 0;
}
