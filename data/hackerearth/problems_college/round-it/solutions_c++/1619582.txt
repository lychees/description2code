#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n,b;
    	cin>>n>>b;
    	int x = n%b;
    	if(b&1){
    		if(x<=b/2)n=n-x;
    		else n=n+b-x;
    	}
    	else{
    		if(x<b/2)n=n-x;
    		else n=n+b-x;
    	}
    	cout<<n<<endl;
    }
    return 0;
}
