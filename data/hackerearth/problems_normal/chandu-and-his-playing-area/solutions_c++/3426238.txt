#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while (t>0)
	{
		t--;
		long long n;
		cin >> n;
		
		double side = 1.0*n/4;
		long long sideInt = (long long)side;
		double fraction = side - sideInt;
		long long area = sideInt*sideInt;
		
		if(fraction*4>=2)
			area = sideInt*(sideInt+1);
		
    	cout << area << endl;
	}
    return 0;
}
