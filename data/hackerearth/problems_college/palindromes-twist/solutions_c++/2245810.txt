#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char str[10000];
		int ac,bc;
		cin>>str;
		cin>>ac>>bc;
		int l=strlen(str)-1;
		int i=0,cost=0,flag=0;
		while(i<l)
		{
			if(str[i]=='/'&&str[l]=='a')
			{
				flag=1;
				cost+=ac;
			}
			if(str[i]=='/'&&str[l]=='b')
			{
				flag=1;
				cost+=bc;
			}
			if(str[l]=='/'&&str[i]=='a')
			{
				flag=1;
				cost+=ac;
			}
			if(str[l]=='/'&&str[i]=='b')
			{
				flag=1;
				cost+=bc;
			}
			if(str[i]=='/'&&str[l]=='/')
			{
				flag=1;
				cost=cost+2*(min(ac,bc));
			}
			if(str[i]!='/'&&str[l]!='/'&&str[i]!=str[l])
			{
				flag=0;
				break;
			}
			i++;
			l--;
		}
		if(flag==0)
			cout<<"-1"<<endl;
		else
			cout<<cost<<endl;
	}
}
