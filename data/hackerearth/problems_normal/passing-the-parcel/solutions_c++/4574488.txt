#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    string song;
    cin >> n >> song;
    set<int> remaining;
    for (int i = 1; i <= n; ++i)
    	remaining.insert(remaining.end(), i);
    int cur = 0;
    auto iter = remaining.begin();
    while (remaining.size() > 1) {
    	auto prev = iter++;
    	if (iter == remaining.end())
    		iter = remaining.begin();
    	if (song[cur] == 'b')
    		remaining.erase(prev);
    	cur++;
    	if (cur == song.size())
    		cur = 0;
    }
    cout << *remaining.begin() << endl;
    return 0;
}
