//-------------
int diff[100010];
int main()
{
	int n,q;
	sc2(n,q);
	int i,j;
	ll s=0;
	for (i=0;i<n;i++)
	{
		int l,r;
		sc2(l,r);
		if (l>r)swap(l,r);
		s+=(ll)l;
		diff[i]=r-l;
	}	
	sort(diff,diff+n);
	int f=1;
	for (i=0;i<n;i+=2)
	{
		if (diff[i]!=diff[i+1])
			f=0;
		s+=(ll)diff[i];
	}
	if (!f)
		s=-1;
	//printf("s made is :%lld\n",s);
	for (i=0;i<q;i++)
	{
		ll x;
		scanf("%lld",&x);
		if (x==s)
		{
			printf("YES\n");
		}
		else
			printf("NO\n");
	}

	return 0;

}
