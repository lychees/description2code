#include <iostream>
using namespace std;
int lis(int a[], int n) {
	int *longest = new int[n];
	int ans = 1;
	for(int i = 0 ; i< n ;i++) {
		longest[i] = 1;
	}
	for(int i = 0 ; i < n ; i++) {
		for(int j = 0 ; j < i ; j++) {
			if(a[i] > a[j]) {
				int max = longest[j] + 1;
				if(max > longest[i]) {
					longest[i] = max;
				}
			}
		}
		if(longest[i] > ans) {
			ans = longest[i];
		}
			
	}
	//int ans = longest[n-1];
	int binary = 1;
	int res = 0;
	while(ans > 0) {
		int b = ans%2;
		res = b*binary + res;
		binary = binary*10;
		ans = ans/2;
	}
	cout<<res<<endl;
	
}
int main()
{
    //cout << "Hello World!" << endl;
    //return 0;
    int n;
    cin>>n;
    int *a = new int[n];
    for(int i = 0 ; i< n ; i++) {
    	int j ;
    	cin >>j;
    	a[i] = j;
    }
    lis(a,n);
    return 0;
}
