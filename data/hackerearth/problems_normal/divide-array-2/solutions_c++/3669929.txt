#include <iostream>
using namespace std;

int main()
{
	//Read Size
     int n;
    cin>>n;
    
    //Read Elements
     int a[n];
    for(long int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    
    //Read Q Opeartions
     int q;
    cin>>q;
   
    int divider = 1;
   	for( int k=0;k<q;k++)
   	{
   		int no;
   		cin>>no;
   		divider *= no;
   	}
   	
   	for( int k=0;k<n;k++)
   	{
   		int quot = divider == 0?0:a[k]/divider;
   		cout<<quot<<" ";
   	}
   	return 0;
}
