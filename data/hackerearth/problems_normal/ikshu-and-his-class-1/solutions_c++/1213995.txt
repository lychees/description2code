typedef long long LL;
#define rep(it,s) for(__typeof((s).begin()) it=(s).begin();it!=(s).end();it++)

int n, k;
int par[100010];
int rk[100010];
int siz[100010];
long long f[100010];
int mod = 1000000007;

int get(int p) {
    if (par[p]==p) return p;
    par[p] = get(par[p]);
    return par[p];
}

void merge(int a, int b) {
    a = get(a);
    b = get(b);
    
    if (a==b) return;

    if (rk[a]>rk[b]) {
        par[b] = a;
        siz[a] += siz[b];
    }
    else if (rk[b]>rk[a]) {
        par[a] = b;
        siz[b] += siz[a];
    }
    else {
        par[b] = a;
        siz[a] += siz[b];
        rk[a]++;
    }
}

int main() {

    //freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);

    cin>>n>>k;
    for (int i=0; i<n; i++) {
        par[i] = i;
        rk[i] = 0;
        siz[i] = 1;
    }
    for (int i=0; i<k; i++) {
        int a, b;
        scanf("%d%d", &a, &b);
        merge(a, b);
    }
    f[0] = 1;
    for (int i=1; i<=n; i++) f[i] = (f[i-1]*i)%mod;

    long long res = 1;

    for (int i=0; i<n; i++) if (get(i)==i) {
        //cout<<i<<" "<<siz[i]<<" "<<f[siz[i]]<<endl;
        res = (res*f[siz[i]])%mod;
    }

    cout<<res<<endl;

    return 0;

}
