#include <iostream>
#include <string>
#define loop(n) for(int i=0;i<n;i++)
using namespace std;
long long div(long long m)
{
    long long d=5,r=0;
    for(int i=0;i<m;i++)
    {
       r=m-d;
       if((r%3)==0 && (r+d==m))
       {
           return r;
       }
       else
       {
           d=d+d;
           if(d>=m)
           {
               return -1;
               break;
           }
           continue;
       }
    }
}    
int main()
{
    string op="";
    int t=0,x=0;
    long long div_ans=0;
    int i=0,j=0,h=0;
    cin>>t;
    for(h=0;h<t;h++)
    {
        cin>>x;
        op="";
        if(x<3 && x<5)
        {
            op="-1";
        }
        else if(x%3==0)
        {
            loop(x)
            {
              op+="5";
            }  
        }
        else if(x%5==0)
        {
            loop(x)
            {
              op+="3";
            }  
        }
        else
        {
           div_ans=div((long long)x);
           if(div_ans==-1)
           {
               op="-1";
           }
           else
           {
               loop(div_ans)
               {
                   op+="5";
               }
               loop(5)
               {
                   op+="3";
               }
           }
        }
        cout<<op<<"\n";
    }
    return 0;
}
