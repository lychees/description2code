#include <iostream>
using namespace std;

int main()
{
    long long int j,i,n,k;
    cin>>n>>k;
    
    int a[100005];
    for(i=0;i<n;i++){
    	cin>>a[i];
    }
    
    long long int c=0,l;
    int temp,s=0,max;
    for(i=0;i<n-1 && c!=k;i++){
    	max=a[i];
    	
    	for(j=i+1;j<n;j++){
    		if(max<a[j]){
    			max=a[j];
    			l=j;
    		} 
    	}
    	if(max!=a[i]){
    		temp=a[i];
    		a[i]=a[l];
    		a[l]=temp;
    		c++;
    	}
    	
    }
    
    for(i=0;i<n;i++){
    	cout<<a[i]<<" ";
    }
    cout<<endl;
    
    return 0;
}
