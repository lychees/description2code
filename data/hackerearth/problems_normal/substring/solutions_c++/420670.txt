#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;

int main()
{
    int t;
    char str[100001];
    long int ans,cnt; 
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%s",str);
    	cnt = 0;
    	ans = 0;
    	for(int i=0;str[i]!='\0';i++)
    	{
    		if(str[i]=='A'||str[i]=='a'||str[i]=='B'||str[i]=='b'||str[i]=='C'||str[i]=='c')
    			cnt++;
    		else
    		{
    			ans = ans + (cnt*(cnt+1)/2);
    			cnt = 0;
    		}
    	}
    	ans = ans + (cnt*(cnt+1)/2);
    	printf("%ld\n",ans);
    }
    return 0;
}
