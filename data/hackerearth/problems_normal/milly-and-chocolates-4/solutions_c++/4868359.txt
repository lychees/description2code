#include <bits/stdc++.h>
using namespace std;

int main() {
	long long int t,n,sum,i,num;
	 double d;
	cin>>t;
	while(t--)
	{
	  cin>>n;
	  sum=0;
	  for(i=0;i<n;i++)
	   {
	      cin>>num;
	      sum+=num;
	   }
	   d=((double)(sum)/(double)(n));
	   printf("%.8f\n",d);
	}
	return 0;
}
