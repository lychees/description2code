#include<iostream>
#include<stdlib.h>
#include<string.h>
#include<algorithm>
using namespace std;
int prev1[26]={-1};
int maxlength(char s[],int n)
{
    int i;
    int len=0,maxlen=0;
    for(i=0;i<n;i++)
    {
        if(prev1[s[i]-'A']==-1)
        {
            prev1[s[i]-'A']=i;
            len++;
        }
        else
        {
            if((i-prev1[s[i]-'A'])<=len)
            {
                if(maxlen<len)
                {
                    maxlen=len;
                }

                len=(i-prev1[s[i]-'A']);
                prev1[s[i]-'A']=i;
            }
            else
                {
                    len++;
                    prev1[s[i]-'A']=i;
                }
        }

    }
    if(maxlen<len)
        maxlen=len;
    return maxlen;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		for(int i=0;i<26;i++)
		prev1[i]=-1;
		char s[10000];
		cin>>s;
		int n=strlen(s);
		cout<<maxlength(s,n)<<endl;
	}
    
}
