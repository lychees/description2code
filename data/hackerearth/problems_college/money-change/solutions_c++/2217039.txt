#include <bits/stdc++.h>

typedef  long long int ll;

using namespace std;

ll lol( ll S[], ll m, ll n )
{
    ll A[n + 1], i, j;
    memset(A, 0, sizeof(A));
    A[0] = 1;

    for(i = 0; i < m; i++)
        for(j = S[i]; j <= n; j++)
            A[j] += A[j - S[i]];

    return A[n];
}

int main()
{
    ll n, m, i;
    cin >> n >> m;
    ll a[m + 1];
    for (i = 0; i < m; i++) {
        cin >> a[i];
    }
    printf("%lld\n", lol(a, m, n));
    return 0;
}
