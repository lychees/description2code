#include<iostream>
#include<cmath>
#include<cstdio>
#define gc getchar
typedef long long int ll;
using namespace std;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
ll nod(ll x)
{
    ll nd=0,i;
    ll sq=sqrt(x);
    for(i=1;i<=sq;i++)
    {
        if(x%i==0)
            nd+=2;
    }
    if(sq*sq==x)
        nd--;
    return nd;
}
int main()
{
    ll t,n,sum;
    t=read_int();
    while(t--)
    {
        n=read_int();
        ll y=0,i=1;
        while(nod(y)<=n)
        {
            y+=i;
            i++;
        }
        cout<<y<<'\n';
    }
    return 0;
}
