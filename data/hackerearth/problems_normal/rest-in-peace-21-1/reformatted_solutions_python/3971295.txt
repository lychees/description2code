'''
# Read input from stdin and provide input before running code
'''
n = int(raw_input())
sub = str("21")
for i in range(0, n):
    num = str(raw_input())

    if int(num) % 21 == 0 or num.find(sub) != -1:
        print "The streak is broken!"

    else:
        print "The streak lives still in our heart!"
