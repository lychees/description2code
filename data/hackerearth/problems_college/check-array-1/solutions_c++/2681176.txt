#include<cstdio>
#include<iostream>
using namespace std;
typedef int w;
typedef long long l;
w a[100001];
w su[100002],pre[100005];
w n;

void solve(){
    cin>>n;
    for(int i=1;i<=n;++i){
        cin>>a[i];
        pre[i+1]=pre[i]+a[i];
    }
    for(int i=n;i>=1;--i){
        su[i-1]=su[i]+a[i];
    }
    for(int i=1;i<=n;++i){
        if(pre[i]==su[i]){
            cout<<"YES\n";
            return;
        }
    }
    cout<<"NO\n";

}

int main(){
    ios_base::sync_with_stdio(false);
    //freopen("in","r",stdin);
    w T;
    cin>>T;
    while(T--) solve();
}
