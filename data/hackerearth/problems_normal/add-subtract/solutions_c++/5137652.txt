#include <iostream>
#include <bits/stdc++.h>

using namespace std;

int main()
{
    int T;
    cin >> T;
    while(T--)
    {
        int arr[1001];
        int ans = 1e9;
        int n,k;
        cin >> n >> k;
        for (int i = 0; i < n ; ++i)
            cin >>arr[i];
        for (int i = 0; i < n ; ++i)
        {
            vector<int> cost;
            int price = 0;
            for (int j  = 0; j < n; ++j)
            {
                if (arr[i] - arr[j] > 0) cost.push_back( (arr[i] - arr[j])*3);
                else cost.push_back( (arr[j] - arr[i])*5 );
            }
            sort(cost.begin(),cost.end());
            for (int j = 0; j < k; ++j)
                price += cost[j];
            ans = min(ans, price);
        }
        cout << ans << endl;

    }
    return 0;
}
