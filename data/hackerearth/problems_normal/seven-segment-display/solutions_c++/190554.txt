#include<stdio.h>
char a[100005];
int main()
{
    int t,i;
    long long ans=0;
    scanf("%d",&t);
    while(t--)
    {
        i=0;
        ans=0;
        scanf("%s",a);
        while(a[i]!='\0')
        {
            if(a[i]=='0')
                ans+=6;
            else if(a[i]=='1')
                ans+=2;
            else if(a[i]=='2')
                ans+=5;
            else if(a[i]=='3')
                ans+=5;
            else if(a[i]=='4')
                ans+=4;
            else if(a[i]=='5')
                ans+=5;
            else if(a[i]=='6')
                ans+=6;
            else if(a[i]=='7')
                ans+=3;
            else if(a[i]=='8')
                ans+=7;
            else if(a[i]=='9')
                ans+=6;
            i++;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
