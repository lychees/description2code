#include<iostream>
using namespace std;
int main()
{
    int n, i, temp, total=0, flag1=0, flag2=0;
    cin>>n;
    int a[5]={0};
    for(i=1;i<=n;i++)
    {
    	cin>>temp;
    	a[temp]++;
    }
    
    total+=a[4];
    //cout<<"Total 4: "<<total<<endl;
    
   	if(a[2]!=0)
   	{
   		if(a[2]%2==0)
   		{
   			total+=a[2]/2;
    		//flag1=0;
    		//cout<<"Flag1 = 0 & Total 2: "<<total<<endl;
   		}
   		else
   		{
   			total+=a[2]/2;
   			flag1=1;
    		a[2]=1;
   			//cout<<"Flag1 = 1 & Total 2: "<<total<<endl;
    	}
    }
    
    if(a[1]!=0&&a[3]!=0){
   		if(a[1]==a[3])
   		{
   			total+=a[3];
   			//cout<<"Total 3: "<<total<<endl;
   		}
   		else if(a[1]<a[3])
   		{
   			total+=a[1];
   			a[3]-=a[1];
   			total+=a[3];
   			//cout<<"Total 1<3: "<<total<<endl;
   		}
   		else if(a[1]>a[3])
   		{
   			total+=a[3];
   			a[1]-=a[3];
   			flag2=1;
   			//cout<<"Total 1>3: "<<total<<endl;
   		}
    }
    	else if(a[1]!=0&&a[3]==0)
    	total+=a[1]/4;
   		
   		if(flag1==1&&flag2==1)
   		{
   			if(a[1]==(2*a[2]))
   			{
   				total+=a[2];
   				//cout<<"Flag1 = 1, flag2 = 1 & Total 1==2 : "<<total<<endl;
   			}
   			else if(a[1]>(2*a[2]))
   			{
   				total+=a[2];
   				a[1]-=a[2];
   				total+=a[1];
   				//cout<<"Flag1 = 1, flag2 = 1 & Total 1>2 : "<<total<<endl;
   			}
   			else if(a[1]<(2*a[2]))
   			{
   				total+=a[1];
   				a[2]-=a[1];
   				total+=a[2];
   				//cout<<"Flag1 = 1, flag2 = 1 & Total 1<2 : "<<total<<endl;
   			}
   		}
   		
   		else if(flag1==1&&flag2==0)
   		{
   			total+=a[2];
   			//cout<<"Flag1 = 1, flag 2 = 0 & Total : "<<total<<endl;
   		}
   		
   		else if(flag1==0&&flag2==1)
   		
   		{
   			total+=a[1];
   			//cout<<"Flag1 = 0, flag 2 = 1 & Total 1==2 : "<<total<<endl;
   		}
   		
   		cout<<total;
   		    
return 0;
}
