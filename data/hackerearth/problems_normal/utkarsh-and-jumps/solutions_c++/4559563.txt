#include<bits/stdc++.h>

#define forn(i,n) for(int i=0;i<n;i++)

using namespace std;
int main(){
	int n,p;
	cin>>n>>p;
	double p1 = p/(double)100;
	double dp[n+1] ;
	dp[0] = 1;
	for(int i=1;i<=n;i++){
		dp[i] = 0.00;
		if(i>=2) dp[i] += dp[i-2]*p1;
		if(i>=3) dp[i] += dp[i-3]*(1.00-p1);
		//cout<<i<<dp[i];;
	}
	cout<<setprecision(6)<<fixed<<dp[n];
}
