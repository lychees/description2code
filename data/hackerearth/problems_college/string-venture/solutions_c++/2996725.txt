#include <bits/stdc++.h>
using namespace std;

int main() {
	string input;
	getline(cin,input);
	//cout<<(input)<<endl;
	int l = input.size();
	int wt = 0;
	for (int i = 0; i < l; ++i)
	{
		/* code */wt += (int)(input[i]);
	}
	//cout<<wt<<endl;
	wt /= l;
	int hash[290] = {0};
	if(wt%2 == 0) {
		reverse(input.begin(),input.end());
		cout<<input<<endl;
	}
	
	else {
		for(int i = 0; i < l; i++) {
			hash[(int)input[i]]++;
		}
		int kk = 0,mm;
		for (int i = 0; i < 290; ++i)
		{
			/* code */
			if(hash[i] > kk) {
				kk = hash[i];
				mm = i;
			} 
		}
		cout<<(char)mm<<endl;
	}
	return 0;
}
