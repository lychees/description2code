#include <iostream>
using namespace std;

int main()
{
    int n;
    scanf("%d",&n);
    int*a=new int[n];
    int*rank=new int[n];
    for(int i=0;i<n;i++){
    	scanf("%d",&a[i]);
    	rank[i]=0;
    }
    int count;
    for(int i=0;i<n;i++){
    	count=0;
    	for(int j=0;j<n;j++){
    		if(a[j]>a[i]){
    			count++;
    		}
    	}
    	rank[i]=1+count;
    }
    for(int i=0;i<n;i++){
    	printf("%d ",rank[i]);
    }
    printf("\n");
    return 0;
}
