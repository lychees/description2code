#include<bits/stdc++.h>
using namespace std;
int main(){
	int t; cin >> t;
	vector<int> arr(1000001);
	fill(arr.begin(),arr.end(),1);
	arr[0]=arr[1]=0;
	for(int i=2;i<=1000000;i++){
		if(arr[i]==1){
			for(int j=i+i;j<=1000000;j=j+i)
				arr[j]=0;
		}
	}
	for(int i=2;i<=1000000;i++){ arr[i]+=arr[i-1];}
	while(t--){
		int l,r;
		cin >> l >> r;
	//	cout << arr[r] << arr[2];
		float h = (float)(arr[r] - arr[l-1])/(float)(r-l+1);
		printf("%.6f\n",h);
	}
	
	
	
	return 0;
}
