#include<bits/stdc++.h>
using namespace std;
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		char a[100001];
		long int n;
		cin>>n;
		
		for(long int i=1;i<=n;i++)
			cin>>a[i];
		long int l,r,a1,a2;
		
		a1=0;
		l=1;
		r=n;
		while(l<=r)
		{
			while((l<=n)&&(a[l]!='0'))
			{
				l++;
			}
			while((r>=1)&&(a[r]!='1'))
			{
				r--;
			}
			if(l<r)
			{
				a1=a1+1;
				l++;
				r--;
			}
		}
		a2=0;
		l=1;
		r=n;
		while(l<=r)
		{
			while((l<=n)&&(a[l]!='1'))
			{
				l++;
			}
			while((r>=1)&&(a[r]!='0'))
			{
				r--;
			}
			if(l<r)
			{
				a2=a2+1;
				l++;
				r--;
			}
		}
		if(a2>a1)
			cout<<a1<<endl;
		else
			cout<<a2<<endl;
	}
	return 0;
}
