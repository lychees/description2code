#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
#define ll long long int

ll pow_mod(ll base, ll exp)
{
    ll result = 1;
    while (exp)
    {
        if (exp & 1)
            result = (result*base)%MOD;
        exp >>= 1;
        base = (base*base)%MOD;
    }
    return result;
}

int main()
{
	int t;
	ll n, count;
	cin >> t;
	while(t--)
	{
		count = 0;
		cin >> n;
		if(n<3)
		{
			cout << count << endl;
			continue;
		}
		
		count = ((pow_mod(3,n)%MOD - (3*(pow_mod(2,n)%MOD))%MOD + 3)%MOD + MOD) % MOD;
		cout << count << endl;
	}
    return 0;
}
