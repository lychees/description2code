#include <iostream>
#include <string>
using namespace std;
int main(){

	int T;
	
	cin >> T;

	
	T--;
	string res;
	cin >> res;
	
	while(T--)
	{
	 int l=0;
	 string con;
	 cin >> l >> con;
	 string newStr;
	 
	 for(int i=0;i<l;i++)
	  newStr+=res[i];
	 
	 newStr+=con;
	 res=newStr;
	}
	
	cout << res << endl;
	
	return 0;
}
