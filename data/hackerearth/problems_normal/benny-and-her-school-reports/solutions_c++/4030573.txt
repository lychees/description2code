#define LL long long

int main()
{
    ios_base::sync_with_stdio(false);
    #ifdef _FILES
        freopen("in","r",stdin);

    #endif // _FILES
    LL T,n,m,x,sum,a,diff;
    cin>>T;
    while (T--)
    {
        cin>>n>>m>>x;
        sum = 0;
        for (int i=0;i<n;i++)
        {
            cin>>a;
            sum+=a;
        }
        diff = x*(n+1)-sum;
        if (diff>m) cout<<"Impossible"; else cout<<max((LL)1,diff);
        cout<<endl;
    }

    return 0;
}
