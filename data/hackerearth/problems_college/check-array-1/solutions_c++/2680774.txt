#include<bits/stdc++.h>
using namespace std;
int main()
        {
        long long t,n;
        cin>>t;
        while(t--)
                {
                cin>>n;
                long long arr[n];
                for(int i=0;i<n;i++) cin>>arr[i];
                if(n==1) {cout << "YES" <<endl; continue;}         
                if(n==2) {cout << "NO" <<endl; continue;}
                long long l=arr[0];
                long long r=0;
                for(int i=2;i<n;i++)
                        r=r+arr[i];
                int f=0;
                for(int i=2;i<n;i++)
                        {
                        if(l==r) 
                                {
                                f=1;
                                break;
                        }
                        l+=arr[i-1];
                        r-=arr[i];
                        
                }
                if(f==1) cout << "YES" <<endl;
                else cout << "NO" <<endl;
        }
        
}
