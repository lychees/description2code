//includes all headers
#include<bits/stdc++.h>

//change/set namespace
using namespace std;


char s[10004],p[10004];
//Main code
int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    cin.getline(s,sizeof(s));
    while(t--)
    {
        cin.getline(s,sizeof(s));
        cin.getline(p,sizeof(p));
        //cout<<s<<" "<<p<<endl;
        int ls=strlen(s);
        int lp=strlen(p);
        int i,j;
        for(i=0,j=0;(i<ls)&&(j<lp);)
        {
            if(s[i]==p[j])
            {
                j++;
            }
            i++;
        }
        if(j<lp)cout<<"NO"<<endl;
        else    cout<<"YES"<<endl;
    }

    return 0;
}
