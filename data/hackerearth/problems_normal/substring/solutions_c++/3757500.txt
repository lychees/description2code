#include <iostream>
using namespace std;

bool isInSet(char c){
	return ( c >= 'a' && c <='c' ) || ( c >= 'A' && c <='C' );
}

int main() {
    int t,len;
    long long ans,tmp;
    string str;
    cin>>t;
    while(t--) {
    	tmp=ans=0;
    	cin>>str;
    	len = str.length();
    	for(int i=0; i<len; ) {
    		while( isInSet(str[i]) ) {
    			tmp++;
    			i++;
    		}
    		ans+=(tmp*(tmp+1))/2;
    		tmp=0;
    		i++;
    	}
		ans+=(tmp*(tmp+1))/2;
    	cout<<ans<<endl;
    }
   
    return 0;
}
