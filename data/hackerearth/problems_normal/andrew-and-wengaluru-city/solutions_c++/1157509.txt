#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
inline int max ( int a, int b )
{
	return a > b ? a : b;
}

inline int min ( int a, int b )
{
	return a > b ? b : a;
}

int main()
{
    int t, n;
    cin >> t;
    while ( t-- )
    {
    	cin >> n;
    	int a[n], right[n];
    	for ( int i = 0; i < n ; ++i )
    	{
    		cin >> a[i];
    	}
    	right[n-1] = -1;
    	right[n-2] = a[n-1];
    	for ( int i = n - 3; i >= 0; --i )
    	{
    		right[i] = max(right[i+1], a[i+1]); 
    	}
    	long long total_rain = 0;
    	int left = a[0];
    	for ( int i = 1; i < n ; ++i )
    	{
    		total_rain = (total_rain + max(min(left, right[i]) - a[i] , 0))%MOD;
    		if ( a[i] > left )
    		{
    			left = a[i];
    		}
    	}
    	cout << total_rain << "\n";
    }
    return 0;
}
