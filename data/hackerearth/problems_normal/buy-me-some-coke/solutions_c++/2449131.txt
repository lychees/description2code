#include <bits/stdc++.h>
using namespace std;

int tot,n;
int dp[155][155][100];
int solve(int c,int f,int t)
{
	if(c==n)return 0;
	if(dp[c][f][t]!=0) return dp[c][f][t];
	int n1=tot-c*8;
	n1-= f*5 +t*10;
	int r=100000;
	if(n1>=8)	r= min(r,solve(c+1,f,t)+8);
	if(t>0 && n1>=3) r= min(r,solve(c+1,f+1,t-1)+4);
	if(t>0)r= min(r,solve(c+1,f,t-1)+1);
	if(f>=2) r= min(r,solve(c+1,f-2,t)+2);
	if(f>0 && n1>=3)r= min(r,solve(c+1,f-1,t)+4);
	dp[c][f][t]=r;
	//cout<<c<<" "<<f<<" "<<t<<" "<<dp[c][f][t]<<endl;
	return dp[c][f][t];
	
}


int main() {
	// your code goes here
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int o,f,t,i,j,s=0,x;
		scanf("%d%d%d%d",&n,&o,&f,&t);
		memset(dp,0,sizeof(dp));
		x=n;
	tot=o*1 +f*5 +t*10;
	
		if(n*8>tot) printf("-1\n");
		else
		{
		printf("%d\n",solve(0,f,t));
		}
		
	}
	return 0;
}
