struct A {
    A() {
    }

    A(ll z, ll o) {
        ze = z;
        on = o;
    }

    ll ze = 0, on = 0;

    A& operator+=(A other) {
        ze += other.ze;
        on += other.on;
        return *this;
    }

    bool operator !=(A other) {
        return !(ze == other.ze && on == other.on);
    }
};

vector<A> dp(42, A(-1, -1));


A calc(int n) { // cnt: ff = 0, ss = 1
    if (n == 0) {
        return A(1, 0);
    }
    else if (n == 1) {
        return A(0, 1);
    }

    if (dp[n] != A(-1, -1)) {
        return dp[n];
    }

    A ans;
    ans += calc(n - 1) += calc(n - 2);
    return dp[n] = ans;

}
