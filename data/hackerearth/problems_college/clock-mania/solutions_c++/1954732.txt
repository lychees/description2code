#include <stdio.h>
#include<iostream>
#include<algorithm>
#include<math.h>
int main()
{
    int t,h,m,hour,minute,result;
    scanf("%d",&t);
    while(t--)
    {  
    	scanf("%d%d",&h,&m);
    	//cin>>h>>m;
    	hour=0.5*(60*(h%12)+m);
    	minute=6*(m%60);
    	result=abs(hour-minute);
         if(360-result<result)
         	printf("%d\n",360-result);
         else
           printf("%d\n",result);
    }
    return 0;
}
