#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
using namespace std;
int main()
{
int t,n,m,x1,x2,y1,y2,m1,m2,p;
scanf("%d",&t);
while(t--)
{
        scanf("%d%d",&n,&m);
        scanf("%d%d",&x1,&y1);
        scanf("%d%d",&x2,&y2);
        scanf("%d",&p);

        int v1=abs(x1-x2);
        int v2=abs(n-abs(x1-x2));
        m1=min(v1,v2);

        int v3=abs(y1-y2);
        int v4=abs(m-abs(y1-y2));
        m2=min(v3,v4);

        printf("%d\n",p*(m1+m2));

        if(p*(m1+m2)<=1000)
        printf("You saved the group.\n");
        else
        printf("Let go of the group.\n");
}
return 0;
}
