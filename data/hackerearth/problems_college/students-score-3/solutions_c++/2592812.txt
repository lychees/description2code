#include <iostream>
using namespace std;
#define ll long long

int maxSubArraySum(int a[], int size)
{
   int max_so_far = 0, max_ending_here = 0;
   int i;
   for(i = 0; i < size; i++)
   {
     max_ending_here = max_ending_here + a[i];
     if (max_ending_here < 0)
        max_ending_here = 0;
     if (max_so_far < max_ending_here)
        max_so_far = max_ending_here;
    }
    return max_so_far;
}

int main()
{
   	int N;
   	cin>>N;
    
   	int MAX=-200000000;
   	int n;
   	int arr[100000];
   	for(int i=0;i<N;i++)
   	{
   		cin>>arr[i];
   	}
   	MAX=-2000000000;
   	MAX=max(maxSubArraySum(arr,N),MAX);
   	if(MAX<=0) cout<<0<<endl;
   	else cout<<MAX<<endl;
    return 0;
}
