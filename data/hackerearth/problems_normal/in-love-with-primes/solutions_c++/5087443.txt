int isprime(int,int);
 
int main()
{
	int n,i,j,count,number,flag=0;
	scanf("%d",&n);
	for(count=0;count<n;count++)
	{
		flag=0;
		scanf("%d",&number);
		for(i=1;i<number;i++)
		{
			for(j=1;j<number;j++)
			{
				if(i+j==number)
				{
					if(isprime(i,j)>0)
					{
						if(flag==0)
							printf("Deepa\n");flag++;break;
						
					}
				}
			}
			if(flag)break;
		}
		if(flag==0)
		printf("Arjit\n");
	}
	return 0;
    
    
}
int isprime(int a,int b)
{
	if(a==3||a==2&&b==3||b==2)
		return 3;
	if(a==1||b==1)
		return -1;
	int i,j,count1=0,count2=0;
	for(i=2;i<a/2;i++)
	{
		if(a%i==0)
		{
			count1++;
		}
		if(b%i==0)
			count2++;
		
	}
	
	
	if(count1==0&&count2==0)
		return 5;
	else
	return -1;
	
}
