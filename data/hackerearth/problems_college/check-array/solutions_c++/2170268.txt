#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
bool find(int arr[], int N)
{
    int sum=0,leftsum=0;
      for(int i=1;i<=N;i++)
          sum=sum+arr[i];
    

   for(int i=1;i<=N;i++)
   {
     if(leftsum==sum-arr[i])
         return true;
     else
      {
         leftsum=leftsum+arr[i];
         sum=sum-arr[i];
     }   
       
   } 
return false;    
}

int main() {
    int T,N;
 
    
    cin>>T;
    while(T--)
        {
            cin>>N;
            int arr[N+1];
        for(int i=1;i<=N;i++)
            cin>>arr[i];
        bool result=find(arr,N);
        if(result)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
        
        }
    return 0;
}
