#include<bits/stdc++.h>
using namespace std;
int main()
{
int n,i,j=0,flag=0;
cin>>n;
vector<int>v1;


for(i=0;i<n;i++)
{
    v1.push_back(i+1);

}
string s;
cin>>s;

int l=s.size();
for(i=0;i<s.size();i=((i+1)%l))
{

    if(j==v1.size())
            {
                j=0;
            }


    if(v1.size()==1)
    {
        flag=1;
        break;
        }
    else{
        if(s.at(i)=='a')
        {

            j=(j+1)%n;
        }
        else if(s.at(i)=='b')
        {

        v1.erase(v1.begin()+j);
            }
        }
}

if(flag==1)
{
    cout<<v1.at(0)<<endl;
}
return 0;
}
