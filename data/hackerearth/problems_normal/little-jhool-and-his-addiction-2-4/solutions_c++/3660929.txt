#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    int t,n,k,l,s,e,maxx,minn;
    vector<int > v;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	for(int i=0;i<n;i++)
    	{
    		cin>>l;
    		v.push_back(l);
    	}
    	sort(v.begin(),v.end());
    	s=0;
    	e=n-1;
    	maxx=INT_MIN;
    	minn=INT_MAX;
    	while(s<e)
    	{
    		maxx=max(maxx,v[e]+v[s]);
    		minn=min(minn,v[e]+v[s]);
    		e--;
    		s++;
    	//		cout<<maxx<<" "<<minn<<endl;
    	}
    	//cout<<maxx<<" "<<minn<<endl;
    	cout<<maxx-minn<<endl;
    	if((maxx-minn)>k)
    	cout<<"No more girlfriends!"<<endl;
    	else if((maxx-minn)==k)
    	cout<<"Lucky chap!"<<endl;
    	else cout<<"Chick magnet Jhool!"<<endl;
    	v.clear();
    	
    }
    return 0;
}
