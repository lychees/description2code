#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
int main()
{
    int t;
    cin>>t;
    int x=t+1;
    while(x--)
    {
        fflush(stdin);
        char str[105];
        gets(str);
        fflush(stdin);
        int i,j,l=strlen(str);
        i=0;
        j=l-1;
        for(i=0,j=l-1;i!=j && i<j; )
        {
            while(str[i]==str[j])
            {
                i++;
                j--;
            }
            while(str[i]==' ')
                i++;
            while(str[j]==' ')
                j--;
            while( (str[i]<'a' || str[i]>'z')  )
            {
                if(str[i]=='*' || str[i]==' ')
                    break;
                 i++;
            }
            while( str[j]<'a' || str[j]>'z')
            {
                if( str[j]=='*' || str[j]==' ')
                    break;
                 j--;
            }

           // cout<<i<<"    "<<j<<endl;
            if(str[i]=='*')
            {
                while(str[j]<'a' || str[j]>'z')
                    j--;
                cout<<str[j]<<endl;
                break;
            }
            else if(str[j]=='*')
            {
                while(str[i]<'a' || str[i]>'z')
                    i++;
                cout<<str[i]<<endl;
                break;
            }
        }
    }
    return 0;
}
