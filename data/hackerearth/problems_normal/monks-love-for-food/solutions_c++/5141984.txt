#include <iostream>
#include <stack>
#define QUERIES 100000
#define COST 1000000
using namespace std;


int main()
{
    int Q;
    char str[10];
    stack<int> cost;
    cin >> Q;

    for(int i =0 ; i <= Q; i++)
    {
    	cin.getline(str,10);
    	if(str[0] == '1')
    	{
    		if(cost.empty())
    		{
    			cout << "\nNo Food";
    		}
    		else
    		{
    			cout << "\n" << cost.top();
    			cost.pop();
    		}
    	}
    	else if(str[0] == '2')
    	{
    		cost.push(atoi(str+2));
    	}
    }
    
    return 0;
}
