#include<iostream>
using namespace std;
//using condom

int main()
{
	int sum=10001,n,t;
	cin>>t;
	
	while(t--)
	{
	
	cin>>n;
	
	int num[n],way=0;
	
	
	for(int i=0;i<n;i++)
	cin>>num[i];
	
	int arr[sum+1][n+1];
	
	for(int j=0;j<=n;j++)
	arr[0][j]=1;
	
	for(int j=1;j<=sum;j++)
	arr[j][0]=0;
	
	for(int i=1;i<=sum;i++)
	{
		for(int j=1;j<=n;j++)
		{
		
			
				arr[i][j]= arr[i][j-1];
				
				if((i-num[j-1])>=0)
				arr[i][j] = arr[i][j] || arr[i-num[j-1]][j-1] ;
			
		}
	}
	
	for(int i=0;i<=sum;i++)
	{
		if(arr[i][n]==1)
		way++;
	}
	
	cout<<way<<endl;
	}
return 0;
}
