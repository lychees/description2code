#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,i,j,c;
    cin>>n;
    for(i=1;i<=n;i++)
    {   c=0;
    	for(j=2;j*j<=i;j++)
    	{
    	  if((i%j)==0)
    	  {
    	  	c=1;break;
    	  }
    	}
    	if(c)cout<<i<<endl;
    }
    return 0;
}
