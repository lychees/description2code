#include <bits/stdc++.h>
using namespace std;

int binary_to_number(string &choice)
{
    int ret = 0;
    for(int q=0; q!=choice.size(); q++)
        if(choice[q]=='1')
            ret = ret + pow(2, q);
    return ret;
}

int main()
{
    int t;
    cin>>t;
    while(t!=0){
        t--;
        int n, k;
        cin>>n>>k;
        vector<int> friend_choice;
        for(int q=0; q!=n; q++){
            string curr_choice;
            cin>>curr_choice;
            int number = binary_to_number(curr_choice);
            friend_choice.push_back(number);
        }
        int limit = 1<<k;
        int min_count = k;
        for(int q=1; q!=limit; q++){
            int flag = 1;
            for(vector<int>::iterator iter = friend_choice.begin(); iter!=friend_choice.end(); iter++){
                if(!(*iter&q)){
                    flag = 0;
                    break;
                }
            }
            int counter = 0;
            if(flag){
                int w = 1;
                while(w<=q){
                    if(w&q)
                        counter++;
                    w<<=1;
                }
            }
            if(flag & (counter < min_count))
                min_count = counter;
        }
        cout<<min_count<<endl;
    }
}
