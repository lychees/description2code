ints = {}
for i in range(1, 31):
    ints[i] = 0
lines = input()
for i in range(lines):
    msg = raw_input()
    list = msg.split(' ')
    if msg.startswith('G:'):
        for each in list:
            try:
                ints[int(each)] += 2
            except BaseException:
                pass
    else:
        for each in list:
            try:
                ints[int(each)] += 1
            except BaseException:
                pass

vals = ints.values()
maxi = max(vals)
count = vals.count(maxi)
if count > 1:
    print "No Date"
else:
    for key in ints.keys():
        if ints[key] == maxi:
            if key == 19 or key == 20:
                print "Date"
            else:
                print "No Date"
            break
