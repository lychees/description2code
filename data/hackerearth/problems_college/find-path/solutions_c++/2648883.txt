#include <iostream>

using namespace std;
long long arr[100001];
int main()
{
    int t,n,change;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	change=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>arr[i];
    		
    	}
    	for(int i=1;i<n;i++)
    	{
    		
    		if(arr[i]<=arr[i-1])
    		{
    			change++;
    		}
    	}
    	if(change==n)
    	change--;
    	cout<<change<<endl;
    	
    }
    return 0;
}
