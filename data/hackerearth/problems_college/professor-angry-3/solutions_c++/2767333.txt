#include <iostream>
using namespace std;

int main()
{
    long long a,b,c,test;
    scanf("%lld",&test);
    while(test--)
    {
    	scanf("%lld %lld",&a,&b);
    	int cnt=0;
    	while(a--)
    	{
    		scanf("%lld",&c);
    		if(c<=0) cnt++;
    	}
    	if(cnt>=b) cout<<"NO"<<endl;
    	else	cout<<"YES"<<endl;
    }
    return 0;
}
