#include <bits/stdc++.h>
using namespace std;

int gcd(int a,int b) {
	if(b==0) return a;
	return gcd(b,a%b);
}

int main()
{
	ios::sync_with_stdio(false);
	bool isPrime[100001];
	for(int i = 0;i<=100000;i++) isPrime[i] = true;
	isPrime[1] = false;
	isPrime[0] = false;
	for(long long i = 2;i<=100000;i++) {
		if (isPrime[i]) {
			for(long long j = i*i;j<=100000;j+=i) {
				isPrime[j] = false;
			}
		}
	}
	
	vector<int> prime;
	for(int i = 2;i<=100000;i++) {
		if(isPrime[i]) prime.push_back(i);
	}
	
	
    int t; cin>>t;
    while (t--) {
    	int n; cin>>n;
    	int c = 1; //force 1 ((a,1)=1 but 1|a) , other coprime (n,i)=1 => n%i!=0 ; i!=1
    	for (int i = 2;i<=n;i++) {
    		if (n%i!=0 && gcd(n,i)==1) c++;
    	}
    	//cout<<c<<endl;
    	cout<<((isPrime[c])?"TRUE":"FALSE")<<endl;
    	
    	next: continue;
    }
    return 0;
}
