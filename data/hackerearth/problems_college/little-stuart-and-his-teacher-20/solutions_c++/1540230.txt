#include <bits/stdc++.h>
using namespace std;
#define int long long
int dp[1000005]; 
int isnotprime[1000005]; 
vector <int> divisors[1000005]; 
main()
{
	for (int g=2; g<=1000000; g++)
	{
		if (isnotprime[g]) continue;
		for (int y=g; y<=1000000; y+=g)
		{
			if (y!=g)
			isnotprime[y]=1; 
			divisors[y].push_back(g); 
		}
	}
	dp[1]=0; 
	for (int g=2; g<=1000000; g++)
	{
		dp[g]=dp[g-1]; 
		int r=g; 
		for (int y=0; y<divisors[g].size(); y++)
		{
			r/=divisors[g][y]; 
			r*=divisors[g][y]-1; 
		}
		dp[g]+=r; 
	}
	int q; cin >> q;
	for (int y=0; y<q; y++)
	{
		int d; cin >> d;
		int pair=(d*d-d)/2-dp[d]; 
		cout << pair << '\n';
	}
}
