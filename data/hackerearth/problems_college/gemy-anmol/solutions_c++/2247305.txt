#include <cstdio>
#include<iostream>
using namespace std;
#include<algorithm>
#include<cmath>
#include<cstdlib>
long long s;
void check(long long a[],long long n) {
	long long int pp,t,i,j,k,save,m,mm=100000,count,sum;
	pp=pow(2,n);
	//printf("%lld\n",pp);
	for(i=1;i<pp;i++) {
		count=0;
		save=i;j=0;
		int d[10]={0};
		while(save>0)  {
			d[j++]=save%2;
			save/=2;
		}
		sum=0;
		for(k=0;k<j;k++) {
			if(d[k]==1) {
				count++;
				sum+=a[k];
			}
		}
		//printf("\n%d\n",count);
		if(sum>=s) {
			mm=min(mm,count);
		}
	}
	if(mm==100000)
		printf("-1\n");
	else
		printf("%lld\n",mm);
}
int main()
{
	int t;
	long long a[100001],ans,n,i;
    scanf("%d",&t);
    while(t--) {
    	scanf("%lld %lld",&n,&s);
    	for(i=0;i<n;i++)  {
    		scanf("%lld",&a[i]);
    	}
    	check(a,n);
    }
    return 0;
}
