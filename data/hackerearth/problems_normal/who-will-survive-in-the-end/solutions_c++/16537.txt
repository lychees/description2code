#include<iostream>
#include<math.h>
using namespace std;
int main()
{
	int i=1,n,j,testcases,t;
	cin>>testcases;
	int arr[testcases];
	while(i<=testcases)
	{
		cin>>n;
		t=1;
		for(j=1;t<=n;j++)
		{
			t=2*t;
		}
		t=t/2;
		arr[i-1]=2*(n-t)+1;
		i++;
	}
	for(j=0;j<testcases;j++)
	{
		cout<<arr[j]<<endl;
	}
	return 0;
}
