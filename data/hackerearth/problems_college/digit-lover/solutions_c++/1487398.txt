#include <iostream>
using namespace std;
int sum(int n1)
{
	int s=0;
	while(n1>0)
    {
    	s+=n1%10;
    	n1=n1/10;
    }
    if(s>9)
    {
    	sum(s);
    }
    else
    return s;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	cout<<sum(n)<<endl;
    }
    return 0;
}
