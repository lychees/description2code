#include<bits/stdc++.h>
using namespace std;
bool fun(int i,int j){return i>j;}
int main()
{
    int t,N,A[10000],B[10000],cnt,i;
    cin>>t;
    while(t--)
    {
    	cin>>N;
    	cnt=0;
    	for(i=0;i<N;i++)
    	{
    		cin>>A[i];
    	}
    	for(i=0;i<N;i++)
    	{
    		cin>>B[i];
    	}
    	sort(A,A+N);
    	sort(B,B+N,fun);
    	for(i=0;i<N;i++)
    	{
    		if(A[i]%B[i]==0 || B[i]%A[i]==0)
    		cnt++;
    	}
    	cout<<cnt<<"\n";
    }
    return 0;
}
