#include<bits/stdc++.h>
#include<math.h>
#include<cmath>
using namespace std;
#define pb push_back
#define mp make_pair
#define sz(_vvv) (int)(_vvv.size())
#define all(_v) (_v).begin(),(_v).end()
#define Geti(x) scanf("%d",&x)
#define Getl(x) scanf("%ld",&x)
#define Getll(x) scanf("%64d",&x)
#define pii pair<int,int>
typedef long long ll;
const ll mod=1e9+7;
const int INF=0x7f7f7f7f;
const double eps = 1e-6;

bool primes[1000010];
void seive() {
	for (int i = 2; i <= sqrt(1000004); i ++) {
		if(!primes[i]) {
			for (int j = i * i; j <= 1000004; j += i) primes[j] = true;
		}
	}
}

int main() {
	seive();
    int n; cin >> n;
	int ans = 0, cnt = 0;
	vector<int> inp;
	
	for (int i = 0; i < n; i ++) {
		int x; cin >> x;
		inp.pb(x);
		if(!primes[x]) ans = max(ans,x);
	}
	for (int i = 0; i < n; i ++) if(inp[i] == ans) cnt ++;
	cout << ans << endl;
	cout << cnt << endl;
	return 0;
}
