#include <iostream>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    string s;
 
    while(t--)
    {
        cin>>s;
        long long a=0,b=0,c=0;
        long long len=s.size();
        long long ans=(len)*(len+1)/2;
        for(int i=0;i<s.size();i++)
        {
            if(s[i]=='a')
            {
                a=i+1;
                ans-=min(b,c);
            }
            else if(s[i]=='b')
            {
                b=i+1;
                ans-=min(a,c);
            }
            else
            {
                c=i+1;
                ans-=min(a,b);
            }
        }
        printf("%lld\n",ans);
    }
    return 0;
}
