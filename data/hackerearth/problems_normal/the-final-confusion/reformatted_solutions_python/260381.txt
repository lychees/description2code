

lst = []

a = int(raw_input())

for i in range(0, a):
    lst.append(int(raw_input()))


def fact(number):
    if number == 1:
        return 1
    else:
        return number * fact(number-1)


for num in lst:
    numerator = fact(2 * num)
    denominator = fact(num) * fact(num) * (num + 1)
    result = numerator/denominator
    print result
