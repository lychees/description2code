#include <iostream>
using namespace std;

int main()
{
    int n,t;
    cin>>t;
    while(t--){
    	cin>>n;
    	char**a=new char*[n];
    	for(int i=0;i<n;i++){
    		a[i]=new char[n];
    	}
    	for(int i=0;i<n;i++){
    		for(int j=0;j<n;j++){
    			cin>>a[i][j];
    		}
    	}
    	bool col=true,row=true;
    	for(int i=0;i<n;i++){
    		int start=0,end=n-1;
    		while(start<end){
    			if(a[i][start]!=a[i][end]){
    				row=false;
    			}
    			start++;
    			end--;
    		}
    		if(row==false){
    			break;
    		}
    	}
    	for(int i=0;i<n;i++){
    		int start=0,end=n-1;
    		while(start<end){
    			if(a[start][i]!=a[end][i]){
    				col=false;
    			}
    			start++;
    			end--;
    		}
    		if(col==false){
    			break;
    		}
    	}
    	if(col&&row){
    		cout<<"BOTH"<<endl;
    	}
    	else if(col){
    		cout<<"HORIZONTAL"<<endl;
    	}
    	else if(row){
    		cout<<"VERTICAL"<<endl;
    	}
    	else{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
