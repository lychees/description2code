#include <iostream>
#include <cstdio>
using namespace std;

using namespace std;

int main()
{
        int test_cases;
        scanf("%d",&test_cases);
        while(test_cases > 0)
        {
                int N,arr[10000],i=0;
                scanf("%d",&N);
                while(i<N)
                {
                        scanf("%d",&arr[i]);
                        i = i+1;
                }
                
                //your code here
                char c;
                
                for(i=0;i<N;i++)
                {
                	c=arr[i]^255;
                	cout<<c;
                }
                cout<<"\n";
                
                test_cases--;
        }
        return 0;
}
