#include <iostream>
using namespace std;

int main()
{
	long long int T,D,i,j=0,val;
	cin>>T;
	for(i=0;i<T;i++)
	{
		cin>>D;
		for(j=1;j<D;j++)
		{
			if(j<=D)
		 	{
		 		D=D-j;
		 	}
		 	else
		 	break;	
		}
		if(D==0)
		{
			//cout<<j-1<<"\n";
			val=j-1;
		}
		else
		{
			//cout<<j<<"\n";
			val=j;
		}
		if(val%3==0)
		{
			cout<<"SL\n";
		}
		else if(val%3==1)
		{
			cout<<"LB\n";
		}
		else
		{
			cout<<"BS\n";
		}
	}
    return 0;
}
