#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    int T,nG,nR,wealth[5000],mywealth,count=0;
    cin>>T;
    while(T--){
        count=0;
        cin>>nR>>nG;
        for(int i=1;i<=nR;i++){
            cin>>wealth[i];
        }
        mywealth=wealth[1];
        sort(wealth+1,wealth+nR+1, greater<int>());
        /*for(int i=1;i<=nR;i++){
            cout<<wealth[i] << " ";
        }*/
        int indx=find(wealth+1,wealth+nR+1,mywealth)-(wealth);
        //int x=nG/nR;
        
        int id;
        if(indx<=nG){
        	id=indx;
        }
        else {
        	id=indx%nG;
        	if(id==0){
        		id = nG;
        	}
        }
        //cout << indx << " " << id <<endl;
        for(int j=id;j<=nR;j+=nG){
            if(wealth[j]>mywealth){
                count++;
            }
        }
        cout<<count<<endl;
    }
 
}
