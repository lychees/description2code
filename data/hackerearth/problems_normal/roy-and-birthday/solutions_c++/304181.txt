long long int binom(long long int n,long long int p) { // binom(n, p) = C(n, p) % 1000000007
  long long int num;
  //num = fact(n);
  num=fa[n]%mod;
  long long int den;
  //den = (fact(p) * fact(n-p)) % 1000000007;
  den=((fa[p]%mod)*(fa[n-p]%mod))%mod;
  return (num * inv(den)) % 1000000007;
}

long long int binom2(long long int n,long long int p) { // binom(n, p) = C(n, p) % 1000000007
  long long int num;
  num = n%mod;
  long long int den;
  den = (p) % 1000000007;
  return (num * inv(den)) % 1000000007;
}


long long int fact(long long int n) { // fact(k) = k! % 1000000007
  long long int ans = 1;
  long long int i;
  for (i = 1; i <= n; i++) {
    ans *= i; ans %= 1000000007;
  }
  return ans;
}

long long int inv(long long int x) {
        long long int m;
        m=1000000007;
  long long int a = 1, b = x;
  while (b != 1) {
    long long int c;
    c= m / b;
    a *= c; a %= m;
    b *= c; b %= m;
    if (b > m/2) { a = m - a; b = m - b; }
  }
  return a;
}

int gcd(int a,int b)
{
 int r, i;
  while(b!=0){
    r = a % b;
    a = b;
    b = r;
  }
  return a;
}
