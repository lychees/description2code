#include <iostream>
#include <algorithm> 
using namespace std;

int main()
{
	int n;
	cin>>n;
	int i;
	for(i=0;i<n;i++){
	string s;
	cin>>s;
	int j;
	int arr[26];
       
         for(j=0;j<26;j++){
        arr[j]=0;
        }
//	sort(begin(s),end(s));
	for(j=0;j<s.size();j++){
		if(s[j]=='A'||s[j]=='a') arr[25]++;
		else if(s[j]=='B'||s[j]=='b') arr[24]++;
		else if(s[j]=='C'||s[j]=='c') arr[23]++;
		else if(s[j]=='D'||s[j]=='d') arr[22]++;
		else if(s[j]=='E'||s[j]=='e') arr[21]++;
		else if(s[j]=='F'||s[j]=='f') arr[20]++;
		else if(s[j]=='G'||s[j]=='g') arr[19]++;
		else if(s[j]=='H'||s[j]=='h') arr[18]++;
		else if(s[j]=='I'||s[j]=='i') arr[17]++;
		else if(s[j]=='J'||s[j]=='j') arr[16]++;
		else if(s[j]=='K'||s[j]=='k') arr[15]++;
		else if(s[j]=='L'||s[j]=='l') arr[14]++;
		else if(s[j]=='M'||s[j]=='m') arr[13]++;
		else if(s[j]=='N'||s[j]=='n') arr[12]++;
		else if(s[j]=='O'||s[j]=='o') arr[11]++;
		else if(s[j]=='P'||s[j]=='p') arr[10]++;
		else if(s[j]=='Q'||s[j]=='q') arr[9]++;
		else if(s[j]=='R'||s[j]=='r') arr[8]++;
		else if(s[j]=='S'||s[j]=='s') arr[7]++;
		else if(s[j]=='T'||s[j]=='t') arr[6]++;
		else if(s[j]=='U'||s[j]=='u') arr[5]++;
		else if(s[j]=='V'||s[j]=='v') arr[4]++;
		else if(s[j]=='W'||s[j]=='w') arr[3]++;
		else if(s[j]=='X'||s[j]=='x') arr[2]++;
		else if(s[j]=='Y'||s[j]=='y') arr[1]++;
		else if(s[j]=='Z'||s[j]=='z') arr[0]++;
	}
      
	string out;
	for(j=0;j<26;j++){
		if(arr[j]==j+1) out="Yes"; 
		else if(arr[j]!=j+1 && arr[j]!=0){out="No";
		break;
		}
	}
	 cout << out << endl;
	}
   
    return 0;
}
