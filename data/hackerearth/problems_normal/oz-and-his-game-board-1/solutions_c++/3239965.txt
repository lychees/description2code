#include <iostream>
using namespace std;


long minx(long a, long b){
	if(a<b) return a;
	else return b;
	
}
int main()
{
    long t,i,n,m,a,b,c;
    cin >> t;
    while(t--)
    {
    	long long s1=0,s2=0;
    	cin >> n >> m >> a >> b >> c;
    
    		s1+=n*a;
    		s1+=m*b;
    
    		long k=minx(n,m);
    		s2+=k*c*2;
    		n-=k;
    		m-=k;
    		s2+=n*a;
    		s2+=m*b;
    		if(s1>s2) cout << s1 <<'\n';
    		else 
    		cout << s2 << '\n';
    	
    }
    return 0;
}
