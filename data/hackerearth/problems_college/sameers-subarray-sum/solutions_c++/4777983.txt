#include <bits/stdc++.h>
using namespace std;
#define String string
long int m=9;
long int md[100005];
long int mexp(long int a,long int b)
{
	if(b==0)
	{
		return 1;
	}
	long int t=mexp(a,b/2);
	if(b%2==0)
	{
		return (t*t)%m;
	}
	else
	{
		return (t*t*a)%m;
	}
}
void pre(String s)
{
	long int i,l;
	l=s.size();
	md[1]=(s[0]-'0')%9;
	for(i=1;i<l;i++)
	{
		md[i+1]=(md[i]*1)%m+(s[i]-'0')%m;
		md[i+1]%=m;
	}
}
int main()
{
	String s;
    long int q,l,r,ans,c,i;
    cin>>l;
	cin>>s;
    scanf("%d",&q);
    pre(s);
    while(q--)
    {
    	scanf("%d%d",&l,&r);
    	if(l==r)
    	{
    		if(s[l-1]=='9'||s[l-1]=='0')
    		{
    			printf("SAMEER\n");
    		}
    		else
    		{
    			printf("NO\n");
    		}
    	}
    	else if(l==1)
    	{
    		if(md[r]==0)
    		{
    			printf("SAMEER\n");
    		}
    		else
    		{
    			printf("NO\n");
    		}
    	}
    	else
    	{
    		c=r-l+1;
    		ans=(md[r]-(md[l-1]*mexp(10,c))%m+m)%m;
    		if(ans==0)
    		{
    			printf("SAMEER\n");
    		}
    		else
    		{
    			printf("NO\n");
    		}    		
    	}
    }
    return 0;
}
