long long x, y;

long long solve()
{
	long long c = 0;
	while (x > 1 || y > 1)
	{
		if (x < y)
		{
			long long k = (y - x) / x;
			y -= k * x;
			if (x < y)
			{
				k++;
				y -= x;
			}
			c += k;
		}
		else
		{
			assert(x != y);
			long long k = (x - y) / y;
			x -= k * y;
			if (x > y)
			{
				k++;
				x -= y;
			}
			c += k;
		}
	}
	return c;
}
int main()
{
	int T;
	cin >> T;
	while (T--)
	{
		cin >> x >> y;
		cout << solve() << endl;
	}
}
