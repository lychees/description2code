#include<iostream>
#include<string>
#include<vector>
using namespace std;
int abs(int k)
{
	if(k<0)
	return -k;
	else
	return k;
}
int main()
{
	int room;
	cin>>room;
	int sit;
	cin>>sit;
	string s;
	cin>>s;
	while(sit-->0)
	{
		int n_pos,w_pos;
		cin>>n_pos;
		cin>>w_pos;
		if(s[n_pos-1]=='1' or s[w_pos-1]=='1')
		{
			cout<<"No"<<endl;
			continue;
		}
		int in_be=0;
		while(n_pos!=w_pos)
		{
			if((n_pos-w_pos)>0)
			{
				n_pos=n_pos/2;
				if(s[n_pos-1]=='1')
				{
					cout<<"No"<<endl;
					in_be=1;
					break;
				}
			}
			if((w_pos-n_pos)>0)
			{
				w_pos=w_pos/2;
				if(s[w_pos-1]=='1')
				{
					cout<<"No"<<endl;
					in_be=1;
					break;
				}
			}
		}
		if(in_be==1)
		{
			continue;
		}
		else
		{
			cout<<"Yes"<<endl;
		}
		
		
		
	}
