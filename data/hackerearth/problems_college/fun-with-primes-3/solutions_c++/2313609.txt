#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
    	int n, last, sum = 0, cnt = 0;
    	double s;
    	cin >> n;
    	
    	while(n>0){
    		last = n % 10;
    		sum+=last;
    		n/=10;
    	}
    	
    //	s = sqrt(sum);
    
    	for(int i=2; i<=sum; i++){
    		if(sum % i == 0)
    		cnt++;
    	}
    	if(cnt == 1)
    	cout << "YES" << endl;
    	else
    	cout << "NO" << endl;
     }
    return 0;
}
