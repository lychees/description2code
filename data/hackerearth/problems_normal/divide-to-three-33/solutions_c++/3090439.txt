#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int n,x;
    cin>>n;
    vector<int> v1;
    v1.resize(n);
    int sum = 0;
    for(int i=0;i<n;i++)
    {
    	cin>>x;
    	v1[i] = x;
    	sum += x;
    }
	if(n==3)
		cout<<*max_element(v1.begin(),v1.end());
	else
		cout<<(sum/3)+1;
    return 0;
}
