import itertools


def primes(n):
    prime_fac = []
    if n <= 2:
        prime_fac.append(n)
        return prime_fac

    d = 2
    while d * d <= n:
        while n % d == 0:
            prime_fac.append(d)
            n /= d

        d += 1

    if n > 2:
        prime_fac.append(n)

    return prime_fac


num = int(raw_input())
for n in range(num):
    item = int(raw_input())
    prime_lst = primes(item)
    comb = set()
    for i in range(len(prime_lst)):
        comb.update(itertools.combinations(prime_lst, i + 1))

    print len(comb)
