#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    int t;
    long long l,b,h;
    double ans;
    cin>>t;
    while(t--)
    {
    	cin>>l>>b>>h;
    	l= l>b?b:l;
    
    	l=h>l?l:h;
    	
    	ans=l/2.0;
    	cout<<std::fixed<<std::setprecision(1)<<ans<<endl;
    	
    }
    return 0;
}
