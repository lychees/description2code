#include<bits/stdc++.h>
#define mp make_pair
#define PII pair<int,int>
#define fi first
#define se second
using namespace std;

const int NMAX=15;

int t,n,a[NMAX],b[NMAX];

int main()
{
    int i,sol;

    cin.sync_with_stdio(false);
    cin>>t;
    while (t--)
        {
            cin>>n;sol=0;
            for (i=1;i<=n;i++) cin>>a[i];
            for (i=1;i<=n;i++)
                {
                    cin>>b[i];
                    sol^=b[i]-1;
                    sol^=a[i]-b[i];
                }
            if (sol>0) cout<<"First\n";
            else cout<<"Second\n";
        }
    return 0;
}
