'''

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
row = []
n, m = raw_input().split()
n, m = int(n), int(m)
col = [0 for x in range(m)]
nn = n
while n > 0:
    n -= 1
    t = [int(x) for x in raw_input().split()]
    col = [col[i] + t[i] for i in range(m)]
    row += [sum(t)]

maxe = 0
for i in range(nn):
    for j in range(m):
        t = row[i] ^ col[j]
        if t > maxe:
            maxe = t
print maxe
