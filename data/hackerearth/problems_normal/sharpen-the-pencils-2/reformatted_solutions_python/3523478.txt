t = int(raw_input())
for s in range(0, t):
    no = int(raw_input())
    l = map(int, raw_input().split())
    if no == 1:
        print "1 0"
    else:
        m = 0
        n = 0
        x = 0
        y = 0
        for i in xrange(0, no):
            if m <= n:
                m += l[x]/2.0
                x += 1
            else:
                n += l[no-1-y]
                y += 1
        print x, y
