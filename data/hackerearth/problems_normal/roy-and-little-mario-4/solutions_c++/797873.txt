#include<bits/stdc++.h>
using namespace std;

int count1=0;
long long int arr[100005]={1,1,2,4};
void docal()
{
		
		for(int j=4;j<=100005;j++)
		arr[j]=(arr[j-1]+arr[j-2]+arr[j-3])%1000000007;
		
}

int main()
{
	int t,no;
	cin>>t;
	docal();
	while(t--)
	{
		cin>>no;
	
		count1=0;
	
		cout<<arr[no]<<" "<<endl;
	}
	return 0;
}
