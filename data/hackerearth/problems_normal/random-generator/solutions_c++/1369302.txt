using namespace std;
int t;
#define MAX 100001
long long int a[MAX];
int main(){
	scanf("%d", &t);
	while (t--){
		long long int n;
		long long int k;
		long long int p;
		scanf("%lld%lld%d", &n,&k,&p);
		bool ok = 0;
		long long int maxt = 0;
		for (int i = 0; i < n; i++){
			scanf("%lld", &a[i]);
			maxt = max(maxt, a[i]);
		}
		sort(a, a + n);
		for (int i = 0; i < n; i++){
			int ii = lower_bound(a, a + n, (long long int)a[i] - (long long int)p)-a;
			int jj = upper_bound(a, a + n, (long long int)a[i] + (long long int)p)-a;
			if (jj - ii >= k){
				ok = true;
				break;
			}
			ii = lower_bound(a, a + n, a[i] - p * 2LL)-a;
			if (i - ii + 1 >= k){
				ok = true;
				break;
			}
			ii = upper_bound(a, a + n, a[i] +p*2LL) - a;
			if (ii - i>= k){
				ok = true;
				break;
			}
		}
		if (ok){
			puts("NO");
		}
		else{
			puts("YES");
		}
	}
	return 0;
}
