#include <iostream>
#define max(a,b) {a>b?a:b};
using namespace std;

int main()
{
    int G;
    cin>>G;
    while(G--)
    {
    	static int u=1;
    	long long int N;
    	cin>>N;
    	long long int arr[N];
    	long long int T[N];
    	
    	for(long long int i=0;i<N;i++)
    	{
    		cin>>arr[i];
    		T[i]=0;
		}
		if(N==0)
		{
			cout<<"Case "<<u<<": 0"<<endl;
			u++;
			//cout<<"0"<<endl;
			continue;
		}
		if(N==1)
		{
			cout<<"Case "<<u<<": "<<arr[N-1]<<endl;
			u++;
			//cout<<arr[N-1]<<endl;
			continue;
		}
		T[0]=arr[0];
		T[1]=max(arr[0],arr[1]);
		for(long long int i=2;i<N;i++)
		{
			int j=i-1;
			int m=T[j];
			int p=T[j-1]+arr[i];
			T[i]=max(m,p);
		}
		cout<<"Case "<<u<<": "<<T[N-1]<<endl;
		u++;
    }
    return 0;
}
