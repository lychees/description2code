#include <iostream>
using namespace std;
long long MOD = 1e9 + 7;
long long POW(long long a, long long b) {
	if (b == 0) return 1;
	if (b == 1) return a;
	long long ans = POW(a*a%MOD, b >> 1)%MOD;
	if (b%2) ans *= a;
	return ans%MOD;
}
int main()
{
    int T; 
    for (cin >> T; T; T--) {
    	long long N; cin >> N;
    	long long A = POW(3, N-1) + 1; 
    	long long B = POW(2, N);
    	if (A < B) A = A + MOD - B;
    	else A -= B;
    	A %=MOD;
    	A *=3;
    	cout << A%MOD <<"\n";
    	
    }
}
