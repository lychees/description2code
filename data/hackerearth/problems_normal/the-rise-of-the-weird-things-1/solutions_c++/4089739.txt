#include<iostream>
using namespace std;
int main()
{
	int v[1000],b[1000],y[1000],i,j,k,l,n[1000],count=0,x,z=0,coun=0,temp,sum=0,sum1=0;
	cin>>k;
	for(i=0;i<k;i++)
	{
		cin>>v[i];
	}
	for(i=0;i<k;i++)
	{
		for(l=i+1;l<k;l++)
		{
			if(v[i]>v[l])
			{
				temp=v[i];
				v[i]=v[l];
				v[l]=temp;
			}
		}
	}
	for(i=0;i<k;i++)
	{
		y[i]=v[i];
	}
	for(i=0;i<k;i++)
	{
		if(v[i]%2==0)
		{
			count++;
				b[i-(i-z)]=v[i];
				z++;
		}		
		if(v[i]%2!=0)
		{
			coun++;
			n[i-(i-x)]=v[i];
			x++;
		}
	}
	for(j=0;j<count;j++)
	{
		cout<<b[j]<<" ";
		sum=sum+b[j];
	}
	cout<<sum<<" ";
	for(j=0;j<coun;j++)
	{
		cout<<n[j]<<" ";
		sum1=sum1+n[j];
	}
	cout<<sum1<<" ";
}
