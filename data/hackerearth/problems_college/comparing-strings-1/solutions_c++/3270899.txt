#include<stdio.h>
#include<string.h>
int main(){
	int t, al, bl, i, f; scanf("%d", &t); char a[100001], b[100001];
	while(t-- && scanf("%s%s", a, b)){
		int aa[26] = {0}, bb[26] = {0}; al = strlen(a), bl = strlen(b), f = 0;
		for(i = 0; i < al; aa[a[i++] - 97] = 1);
		for(i = 0; i < bl; bb[b[i++] - 97] = 1);
		for(i = 0; i <= 25; i++)	if(aa[i] && bb[i]){f = 1; printf("YES\n"); break;}
		if(f == 0)	printf("NO\n");
	}		
	return 0;
}
