ll fun(ll num)
{
//	cout<<num<<"<--ele considered now and its dp currently--->"<<dp[num]<<endl;
	if(dp[num]!=-1)
	return dp[num];
	set <ll> ::iterator it=s.find(num);
	if(it!=s.end())
	{
		dp[num]=1;
	//	cout<<dp[num]<<"<--returning alredy calc dp of"<<num<<endl;
		return dp[num];
	}
	else
	{
		string str=to_string(num);
		bool found=false;
		for(ll x=0;x<str.size();x++)
		{
			ll temp=(num-((str[x]-'0')*(str[x]-'0')));
			if(temp>0 && str[x]!='0')
			found=fun(temp);
			if(found==1)
			break;
		}
		if(found==1)
		dp[num]=1;
		else
		dp[num]=0;
	//	cout<<dp[num]<<"<--dp calculated for::"<<num<<endl;
		return dp[num];
	}
	
	
}
