#include<bits/stdc++.h>
using namespace std;
 
const int N=110;
 
bool visited[N];
char adj[N][N];
int a[N];
int n;
vector<int>v;
 
void dfs(int u) {
    visited[u]=true;
    v.push_back(u);
    for(int i=0;i<n;i++) {
        if(adj[u][i]=='Y'&&visited[i]==false) {
            dfs(i);
        }
    }
    return;
}
int main() {
    scanf("%d",&n);
    for(int i=0;i<n;i++) {
        scanf("%d",a+i);
    }
    for(int i=0;i<n;i++) {
        scanf("%s",adj[i]);
    }
    vector<int>ans(n);
    vector<int>res;
    for(int i=0;i<n;i++) {
        v.clear();
        res.clear();
        if(visited[i]==false) {
            dfs(i);
            sort(v.begin(),v.end());
            for(int j=0;j<v.size();j++) {
                res.push_back(a[v[j]]);
            }
            sort(res.begin(),res.end());
            for(int j=0;j<v.size();j++) {
                ans[v[j]]=res[j];
            }
        }
    }
    for(int i=0;i<ans.size();i++) {
        printf("%d ",ans[i]);
    }
    return 0;
}
