#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define ll long long
#define mod 1000000007
#define bitcount    __builtin_popcountll
int b[12][30],ans;
char s[12];
void per(int a[],int l,int r)
{
    if(l==r)
    {
        int j=1;
        for(int i=0;i<=r;i++)
            j=(1ll*j*b[a[i]][s[i]-'a'])%mod;
        ans=(ans+j)%mod;
        //printf("%d\n",ans);
    }
    else
    {
        for(int i=l;i<=r;i++)
        {
            swap(a[l],a[i]);
            per(a,l+1,r);
            swap(a[l],a[i]);
        }
    }
}
int main()
{
   // freopen("in1CAs.txt","r",stdin);
   // freopen("out1CAs.txt","w",stdout);
    int n,k,i,j,p[12],d[12],x,y;
    p[0]=1;
    for(i=1;i<=10;i++)
        p[i]=2*p[i-1];
    char a[12][7];
    sd(n);
    sd(k);
    for(i=0;i<n;i++)
    {
        for(j=0;j<6;j++)
        {
            a[i][j]=getchar();
            a[i][j]=getchar();
            b[i][a[i][j]-'a']++;
        }
    }
    ss(s);
    i=1;
    j=p[n];
    while(i<j)
    {
        if(bitcount(i)==k)
        {
            y=0;
            for(x=0;x<n;x++)
            {
                if(i&p[x])
                    d[y++]=x;
            }
            per(d,0,y-1);
        }
        i++;
    }
    printf("%d\n",ans);
    return 0;
}
