#include <bits/stdc++.h>
using namespace std;
char s[100005];
int to[1000005][26], pool;
long long at[1000005];
long long cnt[100005];
void insert(char s[]) {
    int r = 0;
    for (int i = 0; s[i] != '\0'; ++i) {
        at[r] += 1;
        int c = s[i] - 'a';
        if (to[r][c] == 0)
            to[r][c] = ++pool;
        r = to[r][c];
    }
    at[r] += 1;
}
void dfs(int r, int d) {
    at[r] = (long long)at[r] * (at[r] - 1) / 2;
    for(int c = 0; c < 26; ++c) {
        if(to[r][c]) {
            dfs(to[r][c], d + 1);
        }
    }
    cnt[d] += at[r];
    for(int c = 0; c < 26; ++c)
        if(to[r][c])
            cnt[d] -= at[to[r][c]];
}
int main()
{
    int n, x = 0;
    scanf("%d", &n);

    for (int i = 0; i < n; ++i) {
        scanf("%s", s);
        insert(s);
        x = max(x, (int)strlen(s));
    }
    dfs(0, 0);
    for(int i = 0; i <= x; ++i)
        printf("%lld ", cnt[i]);
}
