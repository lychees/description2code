#include <iostream>
#include <vector>
#define MAX 100007
using namespace std;

vector <int> v[MAX];
bool visited[MAX];

void initialize()
{
	for(int i=0;i<MAX;++i)
	{
		v[i].clear();
		visited[i] = false;
	}
}

void dfs(int s)
{
	visited[s] = true;
	for(int i=0;i<v[s].size();++i)
	{
		if(!visited[v[s][i]])
		{
			dfs(v[s][i]);
		}
	}
}

int main()
{
    int T;
    cin >> T;
    while(T--)
    {
    	int N;
    	cin >> N;
    	initialize();
    	int e;
    	cin >> e;
    	while(e--)
    	{
    		int x, y;
    		cin >> x >> y;
    		v[x].push_back(y);
    		v[y].push_back(x);
    	}
    	int count = 0;
    	for(int i=0;i<N;++i)
    	{
    		if(!visited[i])
    		{
    			++count;
    			dfs(i);
    		}
    	}
    	cout << count << endl;
    }
    return 0;
}
