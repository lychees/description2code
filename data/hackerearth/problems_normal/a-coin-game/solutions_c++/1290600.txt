#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define ll long long int

int main()
{
ll T,n1,n2,x=1,i;
scanf("%lld",&T);
ll A[1000010]={0};
i=1;
while(i<=1000009 && (i+x)<=1000009)
{
if(A[i]==0)
{
A[i]=i+x;
A[i+x]=i;
x++;
}
i++;
}
while(T--)
{
scanf("%lld %lld",&n1,&n2);
if (n1==0 && n2==0){
printf("Don't Play\n");
continue;
}
if(A[n1]==n2 && (n1!=0 && n2!=0))
printf("Don't Play\n");
else printf("Play\n");
}
return 0;
}
