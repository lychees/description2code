
#include <math.h>
#include <iostream>
using namespace std;
int primeFactors(long int n)
{
	int count=0;
    while (n%2 == 0)
    {
    	count+=1;
        n = n/2;
    }
 
    for (long int i = 3; i <= sqrt(n); i = i+2)
    {
        while (n%i == 0)
        {
            count+=1;
            n = n/i;
        }
    }
    if (n > 2)
        count+=1;
    //cout<<count;
    return count;
}
int main()
{
	long int tc,x,i,temp=0,n = 1000000;
    long long arr[n+1];

    for(i=1;i<=n;i++){
    arr[i]=primeFactors(i)+temp;
    temp=arr[i];
    }

 	scanf("%ld",&tc);
 	while(tc--){
 		scanf("%ld",&x);
		printf("%ld\n",arr[x]);
	}
 
 
    return 0;
}
