using namespace std;

# define  mod 		 1000000007
# define  ll	   	 long long
# define  INF	     (ll)(1e18)
# define  mp         make_pair
# define  fill(a,v)  memset(a, v, sizeof a)

ll i,j;
map<pair <ll,pair <ll,ll> >,ll> map1;
vector <pair <ll,pair <ll,ll> > >v;
int main()
{
	ll N;
	cin>>N;
	ll minm,maxm,midm;
		ll a,b,c;
	for(i=0;i<N;i++)
	{
		cin>>a>>b>>c;
		 minm=min(a,min(b,c));
		 maxm=max(a,max(b,c));
		 midm=a+b+c-minm-maxm;
		v.push_back(mp(minm,mp(midm,maxm)));
		map1[mp(minm,mp(midm,maxm))]+=1;
	}
	ll count=0;
	for(i=0;i<N;i++)
	{
		minm=v[i].first;
		midm=v[i].second.first;
		maxm=v[i].second.second;
		if(	map1[mp(minm,mp(midm,maxm))]==1)
		count++;
		
	}
	cout<<count;

return 0;
}
