#include<bits/stdc++.h>
using namespace std;
 int a[1000001];
        int temp[1000001];
int main()
{

    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;

        for(int i=0;i<n;i++)
            cin>>a[i];
            int l;
            cin>>l;
            int length=0;
            temp[0]=a[0];
            length++; //the max length of LIS till now
        for(int i=1;i<n;i++)
        {

    int     b=(upper_bound(temp,temp+length,a[i]))-temp;
    if( b>0 && (a[i]==temp[b-1]))  //if element found upper bound gives the pos of the element +1
    {
        temp[b-1]=a[i];
        continue;
    }
        if(b>=length)
            length++;
        temp[b]=a[i];
        }
        if(length<l)
            cout<<-1<<"\n";
        else
        cout<<temp[l-1]<<"\n";
    }
}
