#include <iostream>
#include <stdio.h>
 
using namespace std;
long long int a[65],ans[65],sum,r,h,l;
int main()
{
    int n,k,i;
    scanf("%d%d",&n,&k);
    a[0]=1;
    for(i=1;i<=n;i++)
    	a[i]=2*a[i-1];
    for(i=0;i<k;i++)
    	ans[i]=0;
    ans[i]=1;
    for(i=k+1;i<=n;i++)
    {
    	sum=ans[i-1]*2;
    	sum+=a[i-k-1]-ans[i-k-1];
    	ans[i]=sum;
    }
    r=1;
    h=a[n];l=ans[n];
    while(r!=0)
    {
    	r=h%l;
    	h=l;
    	l=r;
    }
    ans[n]=ans[n]/h;a[n]=a[n]/h;
    printf("%lld/%lld",ans[n],a[n]);
    return 0;
}
