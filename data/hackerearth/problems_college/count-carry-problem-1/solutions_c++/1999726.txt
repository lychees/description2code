#include <iostream>
#include <string>
using namespace std;

int main()
{
    int tc;
    string  s,n1,n2;
    cin >> tc;
    while(tc--){
    	cin >> n1 >> n2 ;
    	int carry =0;
    	for(int i=n1.length()-1;i>=0;i--){
    		s[i] = n1[i] + n2[i] -96;
    		if(s[i]>9)
    		carry++;
    	}
    	if(carry == 0)
    	cout << "No carry operation" <<endl;
    	else if(carry == 1)
    	cout << "1 carry operation" << endl;
    	else
    	cout << carry << " carry operations" << endl;
    }
    return 0;
}
