#include <iostream>
using namespace std;

int main()
{
    int t; cin >> t;
    while(t--)
    {
    	int n; cin >> n;
    	int count3=0,count5=0;
    	if(n<3)
    	{
    		cout << -1 << endl; continue; 
    	}
    	while(n%3 && n>0)
    	{
    		count3+=5; n -= 5;
    	}
    	//cout << "n : " << n << endl;
    	if(n>=0 && n%3==0)
    		count5 = n;
    	else
    	{
    		cout << -1 <<endl; continue;
    	}
    	for(int i=0;i<count5;i++)
    		cout << 5;
    	for(int i=0;i<count3;i++)
    		cout << 3;
    	cout << endl;
    }
    return 0;
}
