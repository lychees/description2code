n = int(raw_input())
s = raw_input()

ar = s
j = 0

length = len(s)

new_ar = [s]

sp = True

while j < n:
    x, y = "", ""
    i = 0
    while i < length:
        if i % 2 == 0:
            x = x + ar[i]
        else:
            y = y + ar[i]
        i = i + 1
    ar = x + y[::-1]

    if ar in new_ar:
        print new_ar[n % len(new_ar)]
        sp = False
        break
    else:
        new_ar.append(ar)
    j = j + 1


if sp:
    print ar
