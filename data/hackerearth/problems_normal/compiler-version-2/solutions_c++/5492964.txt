#include <iostream>
using namespace std;

int main()
{
    char ch=getchar_unlocked();
    	while(ch>0)
    	{
    		if(ch=='/')
    		{
    			ch=getchar_unlocked();
    			if(ch=='/')
    			{
    				printf("//");
    				ch=getchar_unlocked();
    				for(;ch!='\n' && ch>0;ch=getchar_unlocked())
    				printf("%c",ch);
    				if(ch>0)
    				{
    					printf("\n");
    					ch=getchar_unlocked();
    				}
    			}
    			else
    			{
    				printf("/");
    			}
    		}
    		else
    		if(ch=='-')
    		{
    			ch=getchar_unlocked();
    			if(ch=='>')
    			{
    				printf(".");
    				ch=getchar_unlocked();
    			}
    			else
    			{
    				printf("-");
    			}
    		}
    		else
    		{
    			printf("%c",ch);
    			ch=getchar_unlocked();
    		}
    	}
    
    return 0;
}
