#include <iostream>
#include<cstring>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--)
	{
		char arr[32];
		cin>>arr;
		long long int res=arr[strlen(arr)-1]-'0';
		for(int j=strlen(arr)-1;j>=2;j-=2)
		{
			int a=arr[j-2]-'0';
			if(arr[j-1]=='+')
			{
				res+=a;
			}
			else if(arr[j-1]=='*')
			{
			res*=a;
			}
			else if(arr[j-1]=='-')
			{
			res-=a;
			}
		}
		cout<<res<<endl;
	}
	// your code goes here
	return 0;
}
