#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,k,n,t;
	cin>>t;
	while(t--){
	
    cin>>n;
    int count = 0;
    while (n > 0){
    
        count = count + 1;
        n = n & (n-1);
    }
    if(count % 2==0)
    cout<<"even"<<endl;
    else
    cout<<"odd"<<endl;
}
	return 0;
}
