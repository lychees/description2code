#include<bits/stdc++.h>

#define rep(i,a,b) for(i=a;i<b;i++)
using namespace std;
string ss[10000];
int main()
{
	int n,m;
	cin>>n>>m;
	string s1,s2;
	int i;
	map<string,string> mp;
	rep(i,0,m)
	{
		cin>>s1>>s2;
		if(s1.size()>s2.size())
		mp[s1]=s2;
		else
		mp[s1]=s1;
	}
	
	rep(i,0,n)
	{
		cin>>ss[i];
	}
	rep(i,0,n)
	{
		cout<<mp[ss[i]]<<" ";
	}
	return 0;
}
