#include <iostream>
#include <string>
using namespace std;
string reverse(string str){
	string temp=str;
	int j=str.size()-1;
	int i;
	for(i=0;i<str.size();i++){
		str[i]=temp[j];
		j--;
	}
	return str;
}
char mid_char(string str){
	return (str[str.size()/2]);
}
