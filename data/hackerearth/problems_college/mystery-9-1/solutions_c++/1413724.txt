#include <iostream>
#include <string>
using namespace std;

int main()
{
    string str;
    int t;
    cin>>t;
    while(t--) {
    	cin>>str;
    	int start = 0, end = str.length() - 1;
    	while(start <= end)	{
    		if(str[start] != str[end])
    			break;
    		start++;
    		end--;
    	}
    	if(start <= end)
    		cout<<-1<<endl;
    	else
    		cout<<str[start - 1] - 0<<endl;
    }
    return 0;
}
