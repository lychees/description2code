#include <iostream>
#include <cstdlib>
#include <string>
using namespace std;

int main(){
	long long i, j, l, T, f;
	string s;
	
	cin >> T;
	while(T--){
		cin >> s;
		l = s.size();
		f=1;
		for(i=1, j=l-2;i<l;i++, j--){
			if(abs((int)(s[i]-s[i-1]))!=abs((int)(s[j]-s[j+1]))){
				f=0;
				break;
			}
		}
		if(f){
			cout << "Funny" << '\n';
		}
		else{
			cout << "Not Funny" << '\n';
		}
	}
	return 0;
}
