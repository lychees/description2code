#include <iostream>
using namespace std;
int visited[100]={0};
int graph[101][101]={0};
int n, m;
int isCyclic(int start,int parent) {
	if(visited[start]==1) return 1;
	visited[start]=1;
	for(int i=1;i<=n;i++) {
		if(graph[start][i]==1 && i!=parent) {
			//visited[i]=1;
			if(isCyclic(i,start)) return 1;
		}
	}
	return 0;
}
int main()
{
    
    cin>>n>>m;
    int x, y;
    while(m--) {
    	cin>>x>>y;
    	graph[x][y]=graph[y][x]=1;
    	
    }
    //visited[1]=1;
    if(isCyclic(1,0)) cout<<"FHTAGN!";
    else cout<<"NO";
    
    return 0;
}
