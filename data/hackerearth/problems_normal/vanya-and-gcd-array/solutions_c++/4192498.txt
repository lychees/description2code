#include <iostream>
#include <cstring>

#define MOD 1000000007

using namespace std;

long long int cache[550][105];
int gcd(int a, int b);
long long int solve(long long int cache[][105], int *arr, int N);

int main()
{
	int N;
	int arr[550];
	cin>>N;
	for(int i=0; i<N; i++)
	{
		cin>>arr[i];
	}
	memset(cache, 0, sizeof(cache));
	
	cout<<solve(cache, arr, N);
    
    return 0;
}

long long int solve(long long int cache[][105], int *arr, int N)
{
	cache[N-1][arr[N-1]] = 1;
	int gd;
	long long int ans = cache[N-1][1];
	
	for(int i=N-2; i>=0; i--)
	{
		cache[i][arr[i]] = 1;
		for(int j=i+1; j<N; j++)
		{
			if(arr[j] > arr[i])
			{
				for(int k = 1; k<= 100; k++)
				{
					if(cache[j][k] > 0)
					{
						gd = gcd(arr[i], k);
						cache[i][gd] = (cache[i][gd] + cache[j][k])%MOD;
					}
				}
			}
		}
		ans = (ans + cache[i][1])%MOD;
	}
	
	return ans;
}

int gcd(int a, int b)
{
	if(b==0)
	{
		return a;
	}
	return gcd(b, a%b);
}
