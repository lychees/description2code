from collections import Counter
string_length = input()
input1 = raw_input()
freq_map = dict(Counter(input1))

query = input()
while query > 0:
    start, end = raw_input().split()
    if start in freq_map and end in freq_map:
        if start != end:
            print freq_map[start]*freq_map[end]
        else:
            print freq_map[start]*(freq_map[start]-1)/2
    else:
        print 0
    query = query - 1
