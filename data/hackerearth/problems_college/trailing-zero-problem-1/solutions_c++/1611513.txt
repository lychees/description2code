#include <iostream>
#include <stdio.h>
#include <math.h>

using namespace std;

typedef long long int Lint;

Lint Zeros( Lint n ) {
	
	if( n < 5 )
		return 0;
	Lint v = 5, val = n / v, ct = 0;
	
	while( val > 0 ) {
		ct += val;
		v = v*5;
		val = n / v;
	}
	
	return ct;
}

int main()
{
	
	Lint t, n, i, v4, z;
	double v, v1, v2, v3;
	
	scanf("%lld", &t );
	
	while( t-- ) {
		
		scanf("%lld", &n );
		v = log( static_cast<double> ( n ) );
		v1 = log( 2 * 3.14159625 * static_cast<double> ( n ) ) / 2.0;
		v2 = static_cast<double> ( n ) * v;
		v3 = ( v1 + v2 - static_cast<double> ( n ) ) / log( 10 );
		v4 = (Lint)(floor( v3 )) + 1;
		z = Zeros( n );
		printf("%lld\n", v4 - z);
	}
	return 0;
}
