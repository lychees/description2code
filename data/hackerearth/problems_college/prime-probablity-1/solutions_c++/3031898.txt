#include <iostream>
#include <stdio.h>
#define ll long long
using namespace std;
ll p[1000001]={0}, a[1000001]={0};
int main()
{
	ll i, j, t, l, r, cp;
    p[0]=0;p[1]=-1;
    for(i=2;i<=1000000;i++)
    {
    	if(p[i]==0)
    	{
    		p[i]=1;
    		for(j=i*2;j<=1000000;j+=i)
    			p[j]=-1;
    	}
    }
    for(i=1;i<=1000000;i++)
    {
    	if(p[i]==-1) a[i]=a[i-1];
    	else a[i]=1+a[i-1];
    }
    cin>>t;
    while(t--)
  	{
  		cin>>l>>r;
  		cp=0;
  		cp=a[r]-a[l];
  		if(p[l]==1) cp++;
  		float d=(r-l+1)*1.0;
  		float ans=(cp/d);
  		printf("%.6f\n",ans);
  	}
    return 0;
}
