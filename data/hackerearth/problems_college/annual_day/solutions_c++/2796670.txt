#include <iostream>
#include<algorithm>
using namespace std;
int countones(int num)
{   
	int count =0;
	while(num>0)
	{
		if(num%2!=0)
		count++;
		num=num/2;
	}
	return count;
}
int minarray(int array[],int size)
{
	int min=0;
	for(int i=1;i<size;i++)
	{
		if(array[i]<array[min])
		min=i;
	}
	return min;
}

int main()
{
    int t;
    cin>>t;
    int n,k;
    int x;
    for(int i=0;i<t;i++)
    {
    	cin>>n>>k;
    	int array[k];
    	for(int i=0;i<k;i++)
    	array[i]=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>x;
    	
    		if(countones(x)>array[minarray(array,k)])
    		array[minarray(array,k)]=countones(x);
    	}
    	int sum=0;
    	for(int i=0;i<k;i++)
    	{
    		sum=sum+array[i];
    		
    	}
    	cout<<sum<<"\n";
    }
    return 0;
}
