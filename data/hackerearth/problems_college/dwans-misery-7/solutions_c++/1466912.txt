#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(0);cin.tie(0);
    int t,n;cin>>t;
    long long a,b,c,a1,a2;
    while(t--){
    	cin>>n;
    	vector<int>v(n);
    	for(int i=0;i<n;i++)cin>>v[i];
    	a=v[0];b=0;
    	for(int i=1;i<n-1;i++){
    		c=max(a,b);
    		a=b+v[i];
    		b=c;
    	}
    	a1=max(a,b);
    	a=v[1];b=0;
    	for(int i=2;i<n;i++){
    		c=max(a,b);
    		a=b+v[i];
    		b=c;
    	}
    	a2=max(a,b);
    	cout<<max(a1,a2)<<endl;
    }
    return 0;
}
