#include<iostream>
using namespace std;
long NumberOfSetBits(long i)
{
    i = i - ((i >> 1) & 0x5555555555555555);
    i = (i & 0x3333333333333333) + ((i >> 2) & 0x3333333333333333);
    return (((i + (i >> 4)) & 0xF0F0F0F0F0F0F0F) * 0x101010101010101) >> 56;
}
int main()
{
	int t;
	cin>>t;
	while(t--) {
		long long int n;
		cin>>n;
		cout<<NumberOfSetBits(n)<<endl;	
	}
	return 0;
}
