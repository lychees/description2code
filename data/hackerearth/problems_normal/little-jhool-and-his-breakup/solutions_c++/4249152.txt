#include <iostream>
using namespace std;

int main()
{
    char ch[101];
	char msg[5];
	msg[0]='l';
	msg[1]='o';
	msg[2]='v';
	msg[3]='e';
	int i,j=0,count=0;
	gets(ch);
	for(i=0; ch[i]!='\0'; i++)
	{
		if(ch[i] == msg[j])
		{
			j++;
			count++;
		}
		if(count == 4)
			break;
	}
	if(count == 4)
		printf("I love you, too!");
	else 
		printf("Let us breakup!");
    return 0;
}
