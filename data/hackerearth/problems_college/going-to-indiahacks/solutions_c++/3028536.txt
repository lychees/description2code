#include <iostream>
using namespace std;

int main()
{
    long long int t,s[1005][1005];
    cin>>t;
    s[1][1]=1;
    for(int i=0;i<=1004;++i){
    	for(int j=0;j<=1004;++j){
    		if(i>1&& j>1){
    			s[i][j]=(s[i-1][j] + s[i][j-1] + s[i-1][j-1])%1000000007;
    		}
    		else{
    			if(i==1 && j>1){
    				s[i][j] = s[i][j-1];
    			}
    			else if(i>1 && j==1){
    				s[i][j] = s[i-1][j];
    			}
    		}
    	}
    }
 /*   
    for(int i=0;i<=5;i++){
    	for(int j=0;j<=5;j++){
    		cout<<s[i][j]<<" ";
    	}
    	cout<<endl;
    }
   */ 
    while(t--){
    	long long int a,b,c,d;
    	cin>>a>>b>>c>>d;
    	
    	cout<<s[c+1-a][d+1-b]<<endl;
    }
    
    
    return 0;
}
