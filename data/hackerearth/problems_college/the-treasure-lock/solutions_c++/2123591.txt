string add(ll n,ll m,string s)
{
    ll i;
    for(i=1;i<=n;i++)
        s=s+'5';
    for(i=1;i<=m;i++)
        s=s+'3';
    return s;
}
int main()
{
        ll t;
        ipl(t);
        while(t--)
        {
            ll n,m,i,j,a,b;
            ipl(n);
            string s;
            int f=0;
            if(n%5==0)
            {
                f=1;
                for(i=1;i<=n;i++)
                    s=s+'3';

            }
            else
            for(i=n;i>=0;i--)
            {
                if(i%3==0)
                {
                    if((n-i)%5==0)
                    {
                        f=1;
                        s=add(i,n-i,s);
                        break;
                    }
                }
            }
            if(f)
                {
                    //printf("in1\n");
                    cout<<s<<"\n";
                }
            else
                {
                    //printf("in2\n");
                    cout<<"-1\n";
                }

        }
        return 0;
}
