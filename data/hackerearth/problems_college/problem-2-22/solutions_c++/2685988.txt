#include <bits/stdc++.h>
using namespace std;

int main()
{

	int t;
	cin>>t;
	while(t--)
	{

		int r, x, y;
		cin>>r>>x>>y;
		int x1, x2, x3, x4;
		int y1, y2, y3, y4;
		cin>>x1>>y1;
		cin>>x2>>y2;
		cin>>x3>>y3;
		cin>>x4>>y4;
		x+=10000;y+=10000;
		x1+=10000;x2+=10000;x3+=10000;x4+=10000;
		y1+=10000;y2+=10000;y3+=10000;y4+=10000;
		int flag = 1;
		if(sqrt((x-x1)*(x-x1)+(y-y1)*(y-y1))>r)flag=0;
		if(sqrt((x-x2)*(x-x2)+(y-y2)*(y-y2))>r)flag=0;
		if(sqrt((x-x3)*(x-x3)+(y-y3)*(y-y3))>r)flag=0;
		if(sqrt((x-x4)*(x-x4)+(y-y4)*(y-y4))>r)flag=0;
		if(flag)cout<<"Yes"<<endl;
		else cout<<"No"<<endl;
	}
	return 0;
}
