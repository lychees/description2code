const int inf=1e9;
const int MOD=1e9+7;
const int N=0,LIM=1e6;

int bad[LIM],h[LIM];
int n,k;
void prepare(){
    int m,x;
    scanf("%d%d%d",&n,&k,&m);
    while (m--){
        scanf("%d",&x);
        if (x<LIM) h[x]=1;
    }
}
void solve(){
    fill(bad+1,bad+LIM,k+1);
    for(int cur=1,n1=0;n1<n;cur++){
        if (h[cur]||bad[cur]<=k) continue;
        printf("%d ",cur);
        n1++;
        bad[cur]=1;
        for(int i=1;i*cur<=LIM;i++) bad[i*cur]=min(bad[i*cur],bad[i]+1);
    }
}
int main(){
    prepare();
    solve();
}
