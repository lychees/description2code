#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,big,small,i;
    string s;
    cin>>t;
    if(t>10 || t<1)
    {
    	cout<<"Invalid Test"<<endl;
    	return 0;
    }
    while(t--)
    {
    	small=big=0;
    	cin>>s;
    	
    	if(s.size()>100 || s.size()<1)
    	{
    		goto a;
    	}
    	for(i=0;i<s.size();i++)
    	{
    		if(s[i]>=65 && s[i]<=90)
    		{
    			big++;
    		}
    		else if(s[i]>=97 && s[i]<=122)
    		{
    			small++;
    		}
    	}
    	a:
    	if(big==0 && small==0)
    	{
    		cout<<"Invalid Input"<<endl;
    	}
    	else
    	{
    		(big<small)?cout<<big<<endl:cout<<small<<endl;
    	}
    }
    return 0;
}
