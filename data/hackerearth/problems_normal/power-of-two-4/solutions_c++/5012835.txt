    #include <bits/stdc++.h>
     
    using namespace std;
     
    #define ll long long int
     
    #define sc1ll(x) cin>>x
     
    vector<ll> arr(100001);
     
    bool pow_2(ll num) //check for power of 2 or not
     
    { return (num&&!(num&(num-1))); }
     
    int main() {
     
    ll ans,test,i,j,num,maxx,minn,total=0;
    for(int i=0;i<32;i++)
        total+=(1<<i);
    sc1ll(test);
    while(test--)
    {
        sc1ll(num);
        for(i=0;i<num;i++)
            sc1ll(arr[i]);
        if(num==1) 
        {
            if(pow_2(arr[0])) 
                printf("YES\n");
            else
                printf("NO\n");
            continue;
        }
        bool flag=false;
        for(int i=0;i<32;i++) 
        {
            ans=total;
            for(int j=0;j<num;j++) 
            {
                if(arr[j]&(1<<i)) 
                    ans=ans&arr[j];
            }
            if(pow_2(ans))
            {
                flag=true;
                break;
            }
        }
        if(flag==true)
            printf("YES\n");
        else
            printf("NO\n");
    }
    }
