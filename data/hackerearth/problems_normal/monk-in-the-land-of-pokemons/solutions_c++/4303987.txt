#include <iostream>
#include<vector>
#include<algorithm>
#define len 11
using namespace std;

int fun(int x)
{ int p=x%len;
return p;}

int main()
{ int t, n, a, b, x, y;
cin>>t;
while(t--)
{ cin>>n;
int cnt=0;
vector <int> v[len];
   while(n--)
   { cin>>a>>b;
   x=fun(a);
   y=fun(b);
   v[x].push_back(a);
   if(find(v[y].begin(), v[y].end(), b)==v[y].end())
   ++cnt;
   else
   {  for(int i=0; i<v[y].size(); i++)
        {    if(v[y][i]==b)
             { v[y].erase(v[y].begin()+i);
             break;}
              }}
   }
   cout<<cnt<<endl;}
   
   
    return 0;
}
