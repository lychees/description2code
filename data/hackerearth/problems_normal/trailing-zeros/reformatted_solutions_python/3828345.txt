def factorial(n):
    if n == 0 or n == 1:
        return 1
    else:
        return n*factorial(n-1)


input = int(raw_input(""))

result = str(factorial(input))

count = 0
index = len(result)-1
while True:
    if result[index] == '0':
        count += 1
        index -= 1
    else:
        break
print count
