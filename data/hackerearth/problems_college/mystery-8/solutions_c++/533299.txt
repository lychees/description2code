#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	cin.ignore();
	while(t--)
	{
		int i,j=0,pos=-1;
		char str[101]={'\0'},temp[101]={'\0'};
		//cin.ignore();
		gets(str);
		//cin.ignore();
		for(i=0;str[i]!='\0';i++)
		{
			
			if((str[i]!='.')&&(str[i]!=',')&&(str[i]!='?')&&(str[i]!='!')&&(str[i]!=' '))
			{
				if(str[i]=='*')
				{
					pos=j;
				}
				temp[j++]=str[i];
			}
		}
		//cout<<str<<endl;
		//cout<<temp<<endl;
		//cout<<pos<<endl;
		cout<<temp[j-1-pos]<<endl;
		
	}
}
