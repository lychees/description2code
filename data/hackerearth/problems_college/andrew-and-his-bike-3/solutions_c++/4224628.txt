
#include<bits/stdc++.h>
using namespace std;
#define pb push_back
int s[100005];
vector<int> adj[100005];
void dfs(int u,int p) {
  s[u] = 1;
  for(int i = 0 ; i < adj[u].size() ;i++) {
    int v = adj[u][i];
    if(v == p ) continue;
    dfs(v,u);
    s[u] += s[v];
  }
}
int main() {
  int n;
  cin >> n ;
  for(int i = 0 ; i < n-1 ; i++) {
    int x,y;
    scanf("%d%d",&x,&y);
    adj[x].pb(y);
    adj[y].pb(x);
  }
  dfs(1,-1);
  s[1] = (n-1);
  for(int i = 1 ; i<=n ;i++) {
    cout<<s[i]<<" ";
  }
}
