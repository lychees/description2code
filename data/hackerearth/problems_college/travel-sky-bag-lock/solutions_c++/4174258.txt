#include <bits/stdc++.h>

using namespace std;
int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n,a;
		cin >> n >> a;
		int k=0;
		while(a!=0)
		{
			int i;
			i = a % 10;
			if(i==9)
			{
				k += 2;
			}
			if(i==1 || i==4 || i==6 || i==8)
			{
				k++;
			}
			a = a/10;
		}
		cout << k << endl;
	}
	return 0;
}
