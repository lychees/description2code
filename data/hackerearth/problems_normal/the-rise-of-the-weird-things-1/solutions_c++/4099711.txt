#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
    int n;
    cin>>n;
    vector<int> v1;
    vector<int> v2;
    vector<int>::iterator it;
    int i;
    for(i = 0; i < n; i++)
    {
        int x;
        cin>>x;
       if(x % 2 == 0)
            v1.push_back(x);
       else
            v2.push_back(x);
    }

    sort(v1.begin(), v1.end());
    sort(v2.begin(), v2.end());
    int sum = 0;
    for(it= v1.begin(); it != v1.end(); it++)
    {
        cout<<*it<<" ";
        sum += *it;
    }
    cout<<sum<<" ";
    sum = 0;
    for(it= v2.begin(); it != v2.end(); it++)
    {
        cout<<*it<<" ";
        sum += *it;
    }
    cout<<sum<<" ";


    return 0;
}
