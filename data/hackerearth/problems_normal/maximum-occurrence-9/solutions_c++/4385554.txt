#include <bits/stdc++.h>
using namespace std;

int main()
{
	string a;
	int i,m,j,n=0;
 	int hash[127]={0};
	getline(cin,a);
	for(i=0;i<a.length();i++)
	{
	    j=a[i];
		hash[j]++;

	}
	for(i=32;i<=127;i++)
	{
		if(hash[i]>n)
		{
			n=hash[i];
			m=i;
		}
	}
    cout<<char(m)<<" "<<n;
    return 0;
}
