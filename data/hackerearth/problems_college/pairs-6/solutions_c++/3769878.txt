#include<bits/stdc++.h>
#define MIN(a,b) ( (a) <= (b) ? (a) : (b))
using namespace std;
int main()
{
	int a,b,t,c,i,j,tmp,max; 
	cin>>t;
	while(t--)
	{
		cin>>a;
		cin>>b>>c;
		tmp=MIN(b,c);
		i=a/2;
		j=a%2;
		if(j==1)
			tmp++;
		if(b==c)
		{
			cout<<"Chotu"<<endl;
			continue;
		}
		{
			if(tmp>i)
			{
				cout<<"Chotu"<<endl;
			}
			else
			{
				cout<<"Motu"<<endl;
			}
		}
	}
	return 0;
}
