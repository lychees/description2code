#include<iostream>
using namespace std;
int main()
{
ios::sync_with_stdio(false);
long int t,i,k,z,s,j,x;
char a[100000];
cin>>t;
while(t--)
{s=0;
    cin>>a;
    for(i=0;a[i]!='\0';i++)
    {k=0;
    z=0;
        for(;a[i]!='*'&&a[i]!='\0';i++)
        {if(a[i]=='o')k++;
        if(a[i]=='x') z++;
        }
        if(z==0)
        {for (j=0;j<k;j++)
          cout<<"o";
        }
        else 
        { for(j=0;j<z;j++)
            cout<<"x";
        }
       
      if(a[i]=='\0') break;
        x=i;
        cout<<"*";
    }
    cout<<endl;
}
return 0;
}
