#include <iostream>

using namespace std;

int main()
{

	int tcCount = 0, tc;
	cin >> tcCount;

	for (tc = 0; tc < tcCount; ++tc) {
		string s;
		cin >> s;
		int count[26] = {0};
		int length  = s.length();

		for (int i = 0; i < length; ++i) {
			char c = s[i];
			count[c - 'a']++;
		}

		int minVal = 0, nextMinVal = 2147483647;
		for (int j = 26; j >= 0; --j) {
			for (int i = 25; i >= 0; --i) {
				if (count[i] == minVal) {
					cout << (char) (i + 'a') << " ";
				} else if (count[i] > minVal) {
					if (nextMinVal > count[i]) {
						nextMinVal = count[i];
					}
				}
				else {}
			}
			minVal = nextMinVal;
			nextMinVal = 2147483647;
		}
		cout << endl;
	}

	return 0;
}
