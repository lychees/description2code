    #include <bits/stdc++.h>
    using namespace std;
     
    int main()
    {
       long long int t;
       cin>>t;
       while(t--)
       {
       	long long int b,w;
       	cin>>b>>w;
       	long long int x,y,z;
       	cin>>x>>y>>z;
       	long long int ank[5];
       	ank[0]=b*x+w*y;
       	ank[1]=b*x+w*(y+z);
       	ank[2]=b*(x+z)+w*y;
       	ank[3]=b*(y+z)+w*y;
       	ank[4]=b*x+w*(x+z);
       	x=*min_element(ank,ank+5);
       	cout<<x<<endl;
       	
       	
       }
        return 0;
    }
