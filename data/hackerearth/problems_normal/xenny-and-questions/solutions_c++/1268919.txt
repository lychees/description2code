
#include <iostream>
#include <stdlib.h>
#include <stdio.h>
using namespace std;
#define gc getchar_unlocked
inline int read_int()
{
  char c = gc();
  while(c<'0' || c>'9') c = gc();
  int ret = 0;
  while(c>='0' && c<='9') {
    ret = 10 * ret + c - 48;
    c = gc();
  }
  return ret;
}


int cmpfunc (const void * a, const void * b)
{
   return ( *(int*)b - *(int*)a );
}

int main()
{
	int arr[100005],test,n,k;
	long long int sum;
	scanf("%d",&test);
	while(test--)
	{
		scanf("%d",&n);
		scanf("%d",&k);
		sum = 0;
		for(int i=0;i<n;i++)
		{
			arr[i] = read_int();
		}
		qsort(arr,n,sizeof(int), cmpfunc);

		if(k > n)
		{
			k = n;
		}
		//cout<<"k = "<<k<<endl;
		for(int i=0;i<k;i++)
		{
		//	cout<<"loop"<<endl;
			sum = sum + arr[i];
		}
		printf("%lld\n",sum);
	}

	return 0;
}
