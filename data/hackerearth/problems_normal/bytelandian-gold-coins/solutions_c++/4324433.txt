#include <stdio.h>
#include <stdlib.h>
long long dp[1000000] = {0};

long long max(long long a, long long b){
    return a>b?a:b;
}
long long sol(long long n)
{
    if(n<1000000)
        return dp[n];

    return sol(n/2) + sol(n/3) + sol(n/4);
}
int main()
{
    long long n,i;
    for(i = 1; i < 1000000; i++){
    dp[i] = max(i, dp[i/2] + dp[i/3] + dp[i/4]);
}
    while(scanf("%lld",&n) != EOF){
    printf("%lld\n", sol(n));
}

    return 0;
}
