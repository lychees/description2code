#include <iostream>
using namespace std;
long long reverse(long long a)
{
	long long rev=0;
	while(a!=0)
	{
		rev=(rev*10)+(a%10);
		a/=10;
	}
	return rev;
}

int main()
{
    long long t, x,y;
    cin>>t;
    for(long long i=0;i<t;i++)
    {
    	cin>>x>>y;
    	cout<<reverse(reverse(x)+reverse(y))<<endl;
    }
    return 0;
}
