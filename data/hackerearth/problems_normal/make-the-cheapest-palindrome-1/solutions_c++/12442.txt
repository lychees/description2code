#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;


char s[10099];

int main()
{
    int ac,bc;
    int t;
    scanf("%d",&t);
    while(t--)
    {
       scanf("%s",s);
       scanf("%d %d",&ac,&bc);
       int l=strlen(s);
       int mn=min(ac,bc);
       
       int ans=0; char t1='/';
       for(int i=0;i<l/2;i++)
       {
         char x,y;
         x=s[i]; y=s[l-i-1];
         if( x==y && x==t1) ans+=mn+mn;
         else
         if(x==t1 || y==t1)
         {
             if(x==t1) x=s[l-i-1];  
             else      x=s[i];
         if(x=='a') ans+=ac; else ans+=bc;
         }
         else
         if(x!=y) { ans=-1; break;}
       }
       printf("%d\n",ans);
    }
return 0;
}
