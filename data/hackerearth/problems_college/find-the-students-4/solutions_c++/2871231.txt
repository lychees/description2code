#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,a[1003];
    cin>>n;
    for(int i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    for(int i=0;i<n/2;i++)
    {
            int x=a[i];
    a[i]=a[n-1-i];
    a[n-1-i]=x;
    }
    int cnt=n-1;
    int x=a[cnt];
    int ans=n;
    while(ans!=0)
    {
          cout<<ans<<endl;
          for(int i=0;i<=cnt;i++)
          {
                  a[i]=a[i]-a[cnt];
          }
          for(int i=cnt;i>=0;i--)
          {
                  if(a[i]==0)
                  {
                      ans--; 
                      cnt--;                 
                  }
          }
    }       
    system("pause");
  return 0;
  
}
