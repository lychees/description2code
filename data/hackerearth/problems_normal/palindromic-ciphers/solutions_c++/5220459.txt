#include <iostream>
using namespace std;

int main()
{
    int n; cin>>n;
    for (int h=1;h<=n;h++)
    {
    	string s; cin>>s;
    	bool pal=true;
    	int i;
    	for (i=0;i<=11;i++)//for finding length of string
    	{
    		if (s[i]=='\0')break;
    	}
    	for (int j=0;j<i/2;j++)//for checking palindrome case
    	{
    		pal=pal&&(s[j]==s[i-j-1]);
    		if(!pal)break;
    	}
    	if (pal)cout<<"Palindrome\n";
    	else
    	{
    		long long int term=1;
    		for (int g=0;g<i;g++)
    		{
    			term*=(s[g]-96);
    		}
    		cout<<term<<endl;
    	}
    }
}
