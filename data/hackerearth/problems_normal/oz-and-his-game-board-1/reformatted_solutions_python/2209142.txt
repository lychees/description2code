t = raw_input()
p = []
# print 'Hi, %s.' % t
for i in range(int(t)):
    x = raw_input()
    p = x.split(" ")
    n = int(p[0])
    m = int(p[1])
    a = int(p[2])
    b = int(p[3])
    c = int(p[4])
    sum1 = n*a+m*b
    if (n > m):
        sum2 = 2*m*c+(n-m)*a
    else:
        sum2 = 2*n*c+(m-n)*b
    if (sum1 > sum2):
        print sum1
    else:
        print sum2
