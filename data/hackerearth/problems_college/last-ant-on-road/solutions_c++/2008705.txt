#include<iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;

    while(t--)
    {
        int n;
        cin>>n;

        if(n==2)
        {
            int a,b;
            cin>>a>>b;

            if(a==-1 && b==-1)
                cout<<2<<endl;
            else
                cout<<1<<endl;

            continue;

        }

        int a[n];


        for(int i=0;i<n;i++)
            cin>>a[i];

        int l=0,r=n-1;
        while(l!=r-1)
        {
                if(a[r]==1)
                r--;

                if(a[l]==-1)
                l++;

                if(a[r]!=1 && a[l]!=-1)
                {

                    for(int i=l;i<=r;i++)
                        if(a[i]==1 && a[i+1]==-1)
                        {
                            a[i]=-1;
                            a[i+1]=1;
                            i++;
                        }
                }
        }

        //cout<<endl<<l<<endl<<r<<endl;

        if( l > (n-r))
            cout<<r+1<<endl;
        else
            cout<<l+1<<endl;

    }




}
