#include <bits/stdc++.h>
using namespace std;
#define gc getchar_unlocked
int fin()
{
    int res=0;
    register char c=gc();

    while(c<'0' || c>'9')
    c=gc();

    while(c>='0' && c<='9')
    {
        res=res*10+(c-48);
        c=gc();
    }
    return res;
}

int index1[27];

int main()
{
    int t=fin();
    int n,q;
    char s[100010],tmp[20],c;
    int res,res_val,res_len,tmp_len,tmp_res;
    while(t--)
    {
	memset(index1,0,sizeof(index1));
	res=res_val=res_len=-1;
	n=fin();
	while((c=gc())!='\n')
	{
	    ++index1[c-'a'];
	}
	for(int i=0;i<n;++i)
	{
	    tmp_res=tmp_len=0;
	    while((c=gc())!='\n')
	    {
	    	if(c==EOF)
	    	break;
		tmp_res+=(index1[c-'a']);
		++tmp_len;
	    }
	    if(tmp_res>res_val)
	    {
		res_val=tmp_res;
		res=i;
		res_len=tmp_len;

		{
		    res=i;
		    res_len=tmp_len;

	    }

}
