#include <iostream>
using namespace std;

int main()
{
    int t,cnt=0,q;
    string str;
    char ch;
    cin>>t;
    while(t--)
    {
    	int arr[26];
    	for(int i=0;i<26;i++)
    		arr[i]=0;
    	cnt=0;	
    	cin>>str;
    	for(int i=0;i<str.size();i++)
    		arr[(int)(str[i]-'a')]++;
    	cin>>q;
    	while(q--)
    	{
    		cin>>ch;
    		arr[(int)(ch-'a')]=0;
    	}
    	for(int i=0;i<26;i++)
    		cnt=cnt+arr[i]*(i+1);
    	cout<<cnt<<endl;	
    		
    }
    return 0;
}
