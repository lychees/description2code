

N = int(raw_input())
sl = raw_input().split(" ")
A = []
B = []
for e in sl:
    A.append(int(e))
    B.append(0)
A.sort()
res = 0
B[0] = 1
for i in range(N):
    mx = 1
    for j in range(i):
        if A[i] % A[j] == 0:
            mx = max(mx, B[j]+1)
    B[i] = mx
    res = max(mx, res)
if res <= 1:
    print -1
else:
    print res
