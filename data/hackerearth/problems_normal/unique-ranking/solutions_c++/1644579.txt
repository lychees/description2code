inline void fs(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}

int main()
{
	//ios_base::sync_with_stdio(false);

	int n,k,i,j,marks[1001][100];
    fs(n);fs(k);
    for(i = 1; i<=n; ++i)
    {
        for(j=1;j<=k;j++)
            fs(marks[i][j]);
    }
    int q,x,o;
    vector< vector<int> > v;
    fs(q);
    while(q--)
    {
        fs(x);
        VI temp,p;
        while(x--)

            cin>>o;
            temp.PB(o);
        }
        for(i=1;i<=n;i++)
        {
         
            for(j=0;j<temp.size();j++)

                p.PB(marks[i][temp[j]]);

        	p.clear();
