int main(){
    int t;cin>>t;while(t--){
        int n,b;
        cin>>n>>b;
        int lo = 0 , hi = 1000005 , mid;
        while(lo <= hi){
            mid = (lo + hi)>>1;
            int val = mid*b;
            if(val == n) break;
            if(val > n) hi = mid-1;
            else lo = mid+1;
        }
        int val = mid*b;
        if(val == n) {
            cout<<val;el;continue;
        }
        if(val > n){
            int vl = val - b;
            int d1 = (val-n);
            int d2 = (n-vl);
            if(d1 == d2){
                cout<<val;el;continue;
            }
            if(d1 > d2){
                cout<<vl;el;continue;
            }
            else {
                cout<<val;el;continue;
            }
        }
        else {
            int vh = val + b;
            int d1 = (n - val);
            int d2 = (vh-n);
            if(d1 == d2){
                cout<<vh;el;continue;
            }
            if(d1 > d2){
                cout<<vh;el;continue;
            }
            else {
                cout<<val;el;continue;
            }
        }
    }
    return 0;
}
