#define in(a) scanf("%d",&a)

#define inll(a) scanf("%lld",&a)
#define ins(a) scanf("%s",a)
#define out(a) printf("%d\n",a)
#define fo(i,a,n) for(i=a;i<n;i++)
#define foo(i,a,n) for(i=a;i<=n;i++)
#define pb push_back
#define ff first
#define ss second

#define ll long long
#define vecp vector<pa>

ll a[1010];

int main()
{
	int n,i,l,r;
	ll ans;

	in(n);

	fo(i,0,n)		inll(a[i]);
	
	if(n==1)		{	outll(a[0]);		r0;	}

	sort(a,a+n);

	l=0 , r=n-1;		ans=0;

	while(l<=r)
	{
		if(r-l==1)		{	ans+=abs(a[r]-a[l]);	break;	}
		else if(r-l==0)							break;

		ans+=abs(a[r]-a[l]);
		ans+=abs(a[r]-a[l+1]);

		l++ , r--;
	}


	l=0 , r=n-1; 		ll ans2=0;

	while(l<=r)
	{
		if(r-l==1)		{	ans2+=abs(a[r]-a[l]);	break;	}
		else if(r-l==0)							break;

		ans2+=abs(a[r]-a[l]);
		ans2+=abs(a[r-1]-a[l]);

		l++ , r--;
	}

	ans=max(ans,ans2);


	outll(ans);
}
