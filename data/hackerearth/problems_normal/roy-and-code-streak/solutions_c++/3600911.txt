#include <iostream>
#include <map>
using namespace std;

int main() {
	int T, N;
	cin >> T;
	
	while (T--) {
		cin >> N;
		
		map<int, int> Solved;
		int s, r, highest = 0, current = 0;

		while (N--) {
			cin >> s >> r;

			if (r == 1) {
				// check the map to see problem has already been solved
				if (Solved.find(s) == Solved.end()) {
					current++;
					Solved[s] = 1;
				}
			} else if (r == 0) {
				current = 0;
			}

			if (current > highest) {
				highest = current;
			}
		}
		cout << highest << endl;
	}
	return 0;	
}
