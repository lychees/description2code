#define REP(i,n) for(int i=0;i<n;i++)
#define REPU(i,a,b) for(int i=a;i<b;i++)
#define sll(n) scanf("%lld",&n)
#define si(n) scanf("%d",&n)
#define pi(n) printf("%d",n)
#define sl(n) scanf("%ld",&n)
#define pin(n) printf("%d\n",n)
#define MP make_pair
#define PB push_back
#define all(a) a.begin(),a.end()
#define F first
#define S second
#define SET(a,b) memset(a,b,sizeof (a))
typedef long long LL;
#define PII pair<int,int>
using namespace std;
int main()
{
    int t; si(t);
    while(t--)
    {
        int n; si(n);
        int a[n];
        REP(i,n) si(a[i]);
        int f=0;
        LL lsum[n],rsum[n];
        lsum[0]=a[0];
        rsum[n-1]=a[n-1];
        for(int i=1;i<n;i++)
        {
            lsum[i]=lsum[i-1]+a[i];
        }
        for(int i=n-2;i>=0;i--)
        {
            rsum[i]=rsum[i+1]+a[i];
        }
        for(int i=0;i<n;i++)
        {
            if(lsum[i]==rsum[i]) f=1;
        }
        if(f) cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
    }
    return 0;
}
