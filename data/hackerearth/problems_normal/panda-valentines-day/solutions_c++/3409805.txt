#include <bits/stdc++.h>
using namespace std;

#define MAXN 1000000

int divisor[MAXN+5], ans[MAXN+5];

void sieve()
{
    long long int i, j;

    divisor[1] = 1;

    for (i = 2; i <= MAXN; i++)
    {
        if (!divisor[i])
        {
            divisor[i] = i;
            for (j = i*i; j <= MAXN; j += i)
            {
                if (!divisor[j])
                {
                    divisor[j] = i;
                }
            }
        }
    }
}

int main()
{
    int t, n, i, j, countDiv;
    
    sieve();
    
    ans[1] = 0;
    for (i = 2; i <= MAXN; i++)
    {
    	countDiv = 0;
    	j = i;
    	while (j != 1)
    	{
    		j /= divisor[j];
    		countDiv++;
    	}
    	ans[i] = ans[i-1] + countDiv;
    }
    
    cin >> t;
    
    while (t--)
    {
    	cin >> n;
    	cout << ans[n] << "\n";
    }
    
    return 0;
}
