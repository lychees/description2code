#include<bits/stdc++.h>
using namespace std;
long countBits (long n)
{
    long c=0;
    while(n)
    {
        n=n&(n-1);
        c++;
    }
    return c;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    vector<pair<long,long> > v;
    long t,n,k,m;
    cin>>t;
    assert(t>=1 && t<=10);
    while(t--)
    {
        cin>>n>>k;
        assert(n>=1 && n<=1000);
        long l;
        for(long i=0;i<n;i++)
        {
            cin>>m;
            l=countBits(m);
            v.push_back(make_pair(l,m));
        }
        sort(v.begin(),v.end());
        reverse(v.begin(),v.end());
        long s=0;
        for(long i=0;i<k;i++)
        {
            s+=v[i].first;
        }
        cout<<s<<endl;
        v.clear();
    }
    return 0;
}
