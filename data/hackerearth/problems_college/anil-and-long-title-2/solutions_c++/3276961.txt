#include <cstdio>
#include <cstring>

using namespace std;

int main()
{
	char str[105];
	scanf("%s", str);
	int size = strlen(str);

	for(int i=0; i<size; i++){
		if(i==0 && str[i] >=97 && str[i]<=122){
			printf("%c", str[i]-32);
		}
		else{
			printf("%c", str[i]);
		}
		
		if(str[i] == '_' && str[i+1] >=97 && str[i+1]<=122){
			printf("%c", str[i+1]-32);
			i++;
		}
	}
    return 0;
}
