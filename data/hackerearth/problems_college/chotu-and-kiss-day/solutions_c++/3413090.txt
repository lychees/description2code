#include <bits/stdc++.h>
#define ll long long
using namespace std;

bool Search(ll S[], int l, int r, int sum){
	while(l<r){
		if(S[l] + S[r] == sum)
			return true;
		else if(S[l] + S[r] > sum)
			r--;
		else
			l++;
	}
	return false;
}

int main()
{
	int N;
    cin>>N;
    ll K;
    ll *S = new ll[N];
    for(int i=0; i<N; i++) cin>>S[i];
    cin>>K;
    sort(S,S+N);
    
    for(int i=0;i<N;i++){
    	 if(Search(S, i, N-1, K - S[i]) == true){
    	 	cout<<"YES";
    	 	exit(0);
    	 }
    }
   
    cout<<"NO";
    
    return 0;
}
