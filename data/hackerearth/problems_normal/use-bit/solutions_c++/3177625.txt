#include <bits/stdc++.h>
using namespace std;

void swap(char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}

vector<string> sol;

void permute(char *a, int l, int r)
{
   int i;
   if (l == r)
     //printf("%s\n", a);
     sol.push_back(a);
   else
   {
       for (i = l; i <= r; i++)
       {
          swap((a+l), (a+i));
          permute(a, l+1, r);
          swap((a+l), (a+i)); //backtrack
       }
   }
}

int main()
{
    char str[10];
    cin >> str;
    int l = strlen(str);
    permute(str, 0, l-1);
    sort(sol.begin(), sol.end());
    cout << sol[0] << endl;
    for(int i = 1; i<sol.size(); ++i) {
    	if(sol[i] != sol[i-1])
    	cout << sol[i] << endl;
    }
    return 0;
}
