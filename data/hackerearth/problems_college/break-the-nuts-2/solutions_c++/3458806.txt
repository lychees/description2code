#include<bits/stdc++.h>
using namespace std;
#define ll long long
int a[1<<20];
int main()
{
	int n;
	cin>>n;
	ll  ans=1ULL;
	bool flag=false;
	for(int i=1;i<=n;++i)
	{
		cin>>a[i];
		if(a[i]==1)flag=true;
	}
	int first1=1;
	int last1=n;
	for(int i=1;i<=n;++i)
	{
		if(a[i]==1)
		{
			first1=i;
			break;
		}
	}
	for(int i=n;i>=1;--i)
	{
		if(a[i]==1)
		{
			last1=i;
			break;
		}
	}
	if(flag==false)puts("0");
	else
	{
		ll  zerocnt=0;
		for(int i=first1;i<=last1;++i)
		{
			if(a[i]==1)
			{
				ans*=(zerocnt+1);
				zerocnt=0;
			}
			else if(a[i]==0)
			{
				++zerocnt;
			}
		}
		cout<<ans<<endl;
	}
	return 0 ;
}
