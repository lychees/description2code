N,X = raw_input().split()
N = int(N)
X = int(X)
diff_level = [int(x) for x in raw_input().split()]
count_continuity = 0
max_score = 0
a = 0
while(count_continuity<=1):
    for i in range(a,len(diff_level),1):
        if(diff_level[i]>X):
            count_continuity+=1
            a = i+1
            break
        else:
            max_score+=1
    if(i==len(diff_level)-1):
        break
print max_score
