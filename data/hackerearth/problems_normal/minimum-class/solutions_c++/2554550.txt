#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	long long n, ans, cnt;
	cin >> t;
	while ( t-- ) {
		cin >> n;
		ans = cnt = 0;
		for ( long long i = 1;; i++ ) {
			cnt += (i/2) + (i%2);
			if ( cnt >= n ) {
				ans = i;
				break;
			}
		}
		cout << ans << endl;
	}
	return 0;
}
