#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int f[20];
int main()
{
	f[0]=0;//Just fo programming convenience
	f[1]=1;
	ll int i;
	for(i=2;i<=18;i++)
	{
		f[i]=f[i-1]*i;
	}
	//Calculate prefix sum
	for(i=1;i<=18;i++)
	{
		f[i]+=f[i-1];
	}
	//Answer all queries/test cases
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int a,b;
		scanf("%lld%lld",&a,&b);
		ll int ans=(f[b]-f[a-1])%10;
		printf("%lld\n",ans);
	}
	return 0;
}
