#include<bits/stdc++.h>
#define ll long long
#define sc(n) scanf("%d",&n)
#define prt(n) printf("%d ",n)
using namespace std;
int main()
{
//	freopen("in.txt","r",stdin);
	int n,t,i,ind,mem;
	ll temp,max;
	string s;
	int arr[125];
	sc(t);
	while(t--)
	{
		memset(arr,0,sizeof(arr));
		sc(n);
		cin>>s;
		int l=s.length();
		for(i=0;i<l;i++)
		{
			arr[s[i]]++;
		}
		max=0;
		ind=1;
		for(i=0;i<n;i++)
		{
			temp=0;
			cin>>s;
			l=s.length();
			for(int j=0;j<l;j++)
			{
				temp+=arr[s[j]];
			}
			if(temp>max)
			{
				max=temp;
				ind=i+1;
				mem=l;
			}			
			if(temp==max)
			{
				if(mem>l)
				{
					ind=i+1;
					mem=l;				
				}
			}
//			cout<<temp<<" ";
		}
		cout<<ind<<"\n";
	}


return 0;
}
