#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		string s;
		cin>>s;
		int cc=0,vc=0,cvc=0;
		for(int i=0;i<s.length();i++){
			switch(s[i]){
				case 'a':
				case 'e':
				case 'i':
				case 'o':
				case 'u':
				case 'y':vc++;
				cvc=0;
				break;
				default:cc++;
				cvc++;
			}
			if(cvc==3){
				cc=vc+1;
				break;
			}
		}
		//cout<<cc<<" "<<vc<<endl;
		if(cc>vc)
			cout<<"hard"<<endl;
		else
			cout<<"easy"<<endl;
	}
    return 0;
}
