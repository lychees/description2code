#include<iostream>
#include<vector>

using namespace std;

int main()
{
	int test_cases,count,t,i,flag;
	cin >> test_cases;
	string s;
	while(test_cases--){
		cin >> s;
		count = 0;
		flag = 0;
		vector<int> v;
		for(i=0;i<s.size();){
			if(s[i] != '|'){
				count++;
				i++;
			}
			else{
					if(flag == 0){
						flag = 1;
					}
					if(count != 0){
						v.push_back(count);
					}
					count = 0;
					while(s[i] == '|'){
						i++;
					}
			}
		}
		if(count != 0){
			v.push_back(count);
		}
		
		if(v.size() == 0){
			cout << "Bob" << endl;
		}
		else if(v.size() == 1){
			if(flag == 1){
				cout << "Alice" << endl;
			}
			else{
				cout << "Bob" << endl;
			}
		}
		else{
			t = v[0];
			for(i=1;i<v.size();i++){
				t = t ^ v[i];
			}
			if(t == 0){
				cout << "Bob" << endl;
			}
			else{
				cout << "Alice" << endl;
			}
		}
	}
	return 0;
}
