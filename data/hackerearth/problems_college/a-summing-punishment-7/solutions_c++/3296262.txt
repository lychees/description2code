#include <iostream>
using namespace std;

int main()
{
    int a[10000],i,q, n,l,r,sum=0;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    cin>>q;
    while(q--)
    {
    	sum=0;
    cin>>l>>r;
    for(i=l-1;i<=r-1;i++)
    {
    		sum+=a[i];
    }
    cout<<sum<<"\n";
    }
    
    return 0;
}
