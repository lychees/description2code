#include <iostream>
#define MOD 10000007
#include <bits/stdc++.h>
using namespace std;

long long dp[110][110];

long long recur(int n,int r) {
	if(r == 0) return (n == 0);
	if(dp[n][r] != -1) return dp[n][r];
	long long ret = 0;
	for(int i=1;i<=n-r+1;i++) {
		ret += recur(n-i,r-1);
	}
	ret %= MOD;
	return dp[n][r] = ret;
}
	

int main()
{
    int n,r;
    cin >> n >> r;
    memset(dp,-1,sizeof dp);
    if(recur(n,r)) cout << recur(n,r);
    else cout << -1;
    return 0;
}
