#include <iostream> 
#include <math.h>
using namespace std;
  
int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
              string a,b;
              cin>>a>>b;
              
              int lookup[26];
              
              for(int i=0;i<26;i++)
              lookup[i]=0;
              
              for(int i=0;i<a.length();i++)
              lookup[a[i]-97]++;
              
             
              int flag=0;
              
              for(int i=0;i<b.length();i++)
              {
                      if(lookup[b[i]-97]!=0)
                      {
                                            flag = 1;
                                            break;
                      }
              }
              
              if(flag)
                 cout<<"YES"<<endl;
              else
                  cout<<"NO"<<endl;
              
              
              
              
    }
    
    
//    system("pause");
    return 0;
}
