# print sorted array
def qsort(arr):
	if arr == []:
		return []
	else:
		pivot = arr[0]
		lesser = qsort([x for x in arr[1:] if x < pivot])
        greater = qsort([x for x in arr[1:] if x >= pivot])
        return greater + [pivot] + lesser
