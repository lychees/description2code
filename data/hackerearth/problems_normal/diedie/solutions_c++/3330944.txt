#include<bits/stdc++.h>
using namespace std;
#define lx long long
const int mod = 1e9+7;
lx int pw(lx int a, lx int b)
{
    if (b==0)
        return 1;
    if (b&1)
        return a*pw(a,b-1)%mod;
    return pw((a*a)%mod,b/2);
}
int main()
{
    ios::sync_with_stdio(0);
    //freopen("input.txt","r",stdin);
    lx int magic = pw(2,mod-2);
    int t;
    cin>>t;
    while(t--)
    {
        lx int n;
        cin>>n;
        cout<<pw(magic,n-1)<<endl;
    }
    return 0;
}
