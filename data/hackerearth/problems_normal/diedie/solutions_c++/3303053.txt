using namespace std;
ll mod=1000000007;
void euclid(ll a,ll b,ll& x,ll& y)
{
	if(a%b==0)
	{
		x=0;
		y=1;
		return;
	}
	euclid(b,a%b,x,y);
	ll temp=x;
	x=y;
	y=temp-y*(a/b);
}
ll inverse(ll a,ll m)
{
	ll x,y;
	euclid(a,m,x,y);
	if(x<0)x+=m;
	return x;
}
int main()
{
	ll t,n,res,ans,a;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		ans=1,a=2;
		while(n!=0)
		{
			if(n%2==1)
			{
				ans*=a;
				ans%=mod;
			}
			a=(a*a)%mod;
			n/=2;
		}
		res=(2*inverse(ans,mod))%mod;
		printf("%lld\n",res);
	}
	return 0;
}
