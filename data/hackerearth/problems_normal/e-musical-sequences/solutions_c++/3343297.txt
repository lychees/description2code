int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int t;
	cin >> t;
	while(t--){
		ll n,m,z;
		cin >> n >> m >> z;
		vector<int> b(n+1);
		for(int i=0;i<n;i++){
			cin >> b[i];
		}
		bool flag=0;
		for(int i=n-1;i>=0;i--){
			if(flag==1){
				b[n]=b[n]-b[i];
				flag=0;
			}
			else{
				b[n]=b[n]+b[i];
				flag=1;
			}
			b[n]%=m;
		}
		b[n]=(b[n]+m)%m;
		int idx=z%(n+1);
		ll ans=0;
		if(n&1) ans=b[idx];
		else if( (z/(n+1))%2==0 ){
			ans=b[idx];
		}
		else{
			ans=(m-b[idx])%m;
		}
		cout << ans << "\n";
	}
	return 0;
}
