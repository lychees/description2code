#include <iostream>
using namespace std;

long long int dp(int n)
{
	if(n==1)
		return 1;
	if(n%2==0)
		return 2*dp(n/2)-1;
	else
		return 2*dp(n/2)+1;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n;
    	cin>>n;
    	long long int result=dp(n);
    	cout<<result<<endl;
    }
    return 0;
}
