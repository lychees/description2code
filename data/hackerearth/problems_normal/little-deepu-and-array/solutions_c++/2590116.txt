#include <iostream>
using namespace std;

void HIT(int ar[], int n, int x)
{
	for(int i = 0; i < n; i++)
	{
		if(x < ar[i])
			ar[i] = ar[i] - 1;
	}
}

int main()
{
    int n;
    cin>>n;
    int ar[n];
    for(int i = 0; i < n; i++)
    	cin>>ar[i];
    int m;
    cin>>m;
    int hit[m];
    for(int i = 0; i < m; i++)
    	cin>>hit[i];
    for(int i = 0; i < m; i++)
    {
    	HIT(ar, n, hit[i]);
    }
    for(int i = 0; i < n; i++)
    	cout<<ar[i]<<" ";
    return 0;
}
