#include<cstdio>
#include<vector>
#include<iostream>
using namespace std;

int main()
{
	int t;
	long n;
	scanf("%d", &t);
	while (t--)
	{
		vector<long>inputs;
		scanf("%ld", &n);
		for (long i = 0; i < n;i++)
		{
			long t;
			scanf("%ld", &t);
			inputs.push_back(t);
		}
		long max = -1, maxp = -1;
		long long profit = 0;
		for (long i = 0; i < n; i++)
		{
			if (inputs[i] > max)
			{
				maxp = i;
				max = inputs[i];
			}
		}
		if (maxp != 0)
		{
			long long loss = 0;
			for (long i = 0; i < maxp; i++)
				loss += inputs[i];
			profit += max*maxp - loss;
		}
		
			while (maxp < n - 1)
			{

				maxp++;
				long pos = maxp;
				max = inputs[pos];
				for (long i = maxp + 1; i < n; i++)
				{
					if (inputs[i] > max)
					{
						maxp = i;
						max = inputs[i];
					}
				}
				//cout << max << endl;
				
				long long loss = 0;
				for (long i = pos; i < maxp; i++)
				{
					loss += inputs[i];
				}
				profit += (maxp - pos)*max - loss;
			}
		
		
		printf("%lld\n", profit);
		
	}
    return 0;
}
