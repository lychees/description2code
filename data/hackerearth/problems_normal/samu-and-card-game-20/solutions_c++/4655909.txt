#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    for (int i = 0; i < t; i++)
    {
    	int count = 0;
    	int startx = 1, starty = 1;
    	int n, m, k;
    	cin >> n >> m >> k;
    	for (int j = 0; j < k; j++)
    	{
    		int x, y;
    		cin >> x >> y;
    		while (startx + x >= 1 && starty + y >= 1 && startx + x <= n && starty + y <= m)
    		{
    			startx += x;
    			starty += y;
    			count++;
    		}
    	}
	cout << count << endl;
    	
    }
    return 0;
}
