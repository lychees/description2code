long long count_ans(long long num)
{
    long long ans = 0;
    long long dig = 1;
    long long now = 1;
    while(true)
    {
        if(now * 10 > num)
        {
            ans = ans + (num - now + 1) * dig ;
            break;
        }
        ans = ans + (now * 9 * dig);
        now = now * 10;
        dig++;
    }
    return ans ;
}


int main()
{
    long long t , n ;
    cin >> t;
    while(t--)
    {
        cin >> n;
        cout << count_ans(n) << endl;
    }
    return 0;
}
