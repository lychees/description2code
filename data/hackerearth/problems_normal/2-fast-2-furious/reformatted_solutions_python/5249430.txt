T = int(raw_input())

dom = [int(x) for x in raw_input().split()]
bra = [int(x) for x in raw_input().split()]
dmax = 0
bmax = 0
for i in range(0, T-1):
    domd = abs(dom[i] - dom[i+1])
    brad = abs(bra[i] - bra[i+1])

    if (domd > dmax):
        dmax = domd
    if (brad > bmax):
        bmax = brad

if (bmax > dmax):
    print "Brian"
    print bmax
else:
    print "Dom"
    print dmax
