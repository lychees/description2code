#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    int dp[101][10000];
    while(t--)
    {
        int n,w,x;
        cin>>n>>w;
        int pr[100],co[100];
        memset(dp,0,sizeof(dp));
        for(int i=0;i<n;i++)
        {
            cin>>x>>co[i];
            pr[i]=x-co[i];
        }
        // for(int i=0;i<n;i++)
        // {
        //     cout<<pr[i]<<" ";
        // }
        // cout<<endl;
        int op_cost=0,op_pr=0;
        for(int i=0;i<=n;i++)
        {
            for(int j=0;j<=w;j++)
            {
                if(i==0||j==0)
                    dp[i][j]=0;
                else if(co[i-1]<=j)
                {
                    dp[i][j]=max(dp[i-1][j],pr[i-1]+dp[i-1][j-co[i-1]]);//cost
                    if(dp[i][j]>op_pr||(dp[i][j]==op_pr&&j<op_cost))
                    {
                        op_pr=dp[i][j];
                        op_cost=j;
                    }
                }
                else
                {
                    dp[i][j]=dp[i-1][j];
                }
            }
        }
        cout<<op_cost<<" "<<op_pr+w<<endl;
        
    }
    return 0;
}
