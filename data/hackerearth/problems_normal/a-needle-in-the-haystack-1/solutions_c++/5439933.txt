#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	 cin.ignore();
	while(t--)
	{
		string s,text;
		getline(cin,s);
		getline(cin,text);
		int a[26]={0};
		int b[26]={0};
	//	memset(a,0,26);
	//	memset(b,0,26);
		int l1,l2;
		l1=s.length();
		l2=text.length();
	int flag=1;
	if(l1>l2)
	flag=0;
	else{
		int k;
			for(int i=0;i<l1;i++)
			{
			a[s[i]-'a']++;
			b[text[i]-'a']++;
			}	
	for(int i=l1;i<=l2;i++)
	{
		flag=1;
		for(k=0;k<26;k++)
		{
			if(a[k]!=b[k]){
			flag=0;
			break;			
			}
		}
		if(flag)
			break;
		b[text[i-l1]-'a']--;
		b[text[i]-'a']++;
	}
	}
	
	if(flag)
	cout<<"YES"<<endl;
	else
	cout<<"NO"<<endl;
	}
	return 0;
}
