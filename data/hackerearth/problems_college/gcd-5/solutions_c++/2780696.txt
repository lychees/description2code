#include <bits/stdc++.h>
using namespace std;
#define aa long long
int main()
{
    aa int tc;
    cin>>tc;
    while(tc--)
    {
        aa int n,coun=0;
        cin>>n;
        aa int a[n];
        aa int temp[n];
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
        }
        for(int i=0;i<n;i++)
        {
            temp[i] = a[i];
        }
        sort(a,a+n);
        for(int i=0;i<n;i++)
        {
            if(a[i] == temp[i])
                coun++;
        }
        if(coun == n) cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
    }
}
