nums = "0123456789"
for _ in xrange(input()):
    s = raw_input()
    words = s.split()
    text = ""
    for i in words:
        text += i
    ind = text.index('$')
    l = len(text)
    sol = ""
    for i in xrange(ind+1, l):
        if (text[i] in nums):
            sol += text[i]
        else:
            break
    print '$' + str(int(sol))
