#include <bits/stdc++.h>
using namespace std;

#define mp make_pair

int A[102], B[102], C[100005], dp[100005];
vector< vector< pair<int, int> > > X(100005);

int solve(int i)
{
	int ans = 0, N = X[i].size();
	
	for(int j=0; j<N; j++)
	{
		dp[j] = 1;
		
		for(int k=0; k<j; k++)
			if(X[i][j].first>X[i][k].second)
				dp[j] = max(dp[j], dp[k]+1);
		
		ans = max(dp[j], ans);
	}
	
	return ans;
}

int main()
{
    int N;
    scanf("%d", &N);
    
    for(int i=1; i<=N; i++)
    	scanf("%d", &A[i]);
    
    B[1] = A[1];
    
    for(int i=2; i<=N; i++)
    	B[i] = B[i-1] + A[i];
    
    for(int i=1; i<=N; i++)
    	for(int j=i; j<=N; j++)
    		X[B[j]-B[i-1]].push_back(mp(i, j));
    
    int ans = 0;
    
    for(int i=1; i<=100000; i++)
    {
    	C[i] = solve(i);
    	
    	ans = max(ans, C[i]);
    }
    
    printf("%d\n", ans);
    
    for(int i=1; i<=100000; i++)
    	if(C[i]==ans)
    		printf("%d ", i);
    
    printf("\n");
    
    return 0;
}
