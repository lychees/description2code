#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long A[102][102],n,T,i,j,k,l,d;
    string S;
    cin>>T;
    while(T--){
    	cin>>n;
    	cin>>S;
    	for(i=0;i<=n+1;i++) A[i][i]=0;
    	for(i=0;i<n-1;i++){
    		if(S[i]==S[i+1]) A[i][i+1]=0;
    		else A[i][i+1]=1;
    	}
    	for(i=2;i<n;i++){
    		for(j=0;i+j<n;j++){
    			if(S[j]==S[i+j]){
    				A[j][i+j]=A[j+1][i+j-1];
    			}
    			else A[j][i+j]=min(A[j+1][i+j],A[j][i+j-1])+1;
    	//	cout<<j<<" "<<i+j<<" "<<A[j][i+j]<<endl;
    		}
    		
    	}
    	cout<<A[0][n-1]<<endl;
    }
    return 0;
}
