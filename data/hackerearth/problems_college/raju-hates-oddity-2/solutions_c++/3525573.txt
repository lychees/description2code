#include<iostream>
using namespace std;
typedef long long int lld;
int main()
{
	lld t,n,a,b,i=0,ar[40];
	std::cin>>t;
	a=1;
	b=2;
	ar[0]=a;
	ar[1]=b;
	i=2;
	while(i<40)
	{
		ar[i]=a+b;
		a=b;
		b=ar[i];
		i++;
	}
	while(t--)
	{
		std::cin>>n;
		std::cout<<ar[n-1]<<endl;
	}
	return 0;
}
