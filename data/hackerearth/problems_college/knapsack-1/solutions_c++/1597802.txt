#include <bits/stdc++.h>
using namespace std ;
#define ll long long int
 
ll Knapsack(ll a[],ll n, ll c)		// Recursive solution O(n^2)
{
	if(n==0 || c==0)
		return 0;
	if(a[n-1]>c)
			return Knapsack(a,n-1,c);
	else
			return max(Knapsack(a,n-1,c),Knapsack(a,n,c-a[n-1])+a[n-1]);
			
}
 
int main()
{
	ll t,a[2005];
	cin>>t;
	while(t--)
	{
		ll n,c;
		cin>>n>>c;
		for(ll i=0;i<n;i++)
			cin>>a[i];
		cout<<Knapsack(a,n,c)<<endl;
		
	}
}
