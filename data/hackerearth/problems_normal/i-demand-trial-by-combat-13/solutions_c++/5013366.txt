#include <iostream>
using namespace std;

int main()
{
    int test_cases=0,num_lan=0,hours=0;
    cin>>test_cases;
    if(test_cases>=1 && test_cases<=1001)
    {
      while(test_cases)
      {
    	cin>>num_lan;
    	if(num_lan>1 && num_lan<=101)
    	{
    		int lan_array[num_lan];
    		cin>>hours;
    		hours= hours<=10000?hours:10000;
            for(int i=0;i<num_lan;i++)
            {
            	cin>>lan_array[i];
            }
            
            while(hours)
            {
            	int prev_ind_val=lan_array[0],temp;
            	
            	if(lan_array[1]==1)
            	   lan_array[0]=1;
            	else
            	   lan_array[0]=0;
            	for(int i=1;i<num_lan-1;i++)
            	{
            		temp=lan_array[i];
            		if(prev_ind_val && lan_array[i+1])
            		   lan_array[i]=1;
            		else
            		   lan_array[i]=0;
            		prev_ind_val=temp;
            	}
            	if(prev_ind_val)
            	  lan_array[num_lan-1]=1;
            	else
            	  lan_array[num_lan-1]=0;
            	hours--;
            }
            
            for(int i=0;i<num_lan;i++)
            {
            	cout<<lan_array[i]<<" ";
            }
            	
     	}
    	test_cases--;
    	cout<<endl;
      }
    }
    return 0;
}
