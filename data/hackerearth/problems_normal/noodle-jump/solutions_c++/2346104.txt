#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    	ll n,k;
	scanf("%lld %lld ",&n,&k);
	
	ll a[n];
	
	
	int i;
	
	for(i=0;i<n;i++)
	{
		scanf("%lld",&a[i]);
	}
	
	sort(a,a+n);
	ll prev=0;
	bool flag=false;
	for(i=0;i<n;i++)
	{
		if(abs(prev-a[i])>k)
		{
			flag=true;
			break;
		}
		prev=a[i];
		
	}
	
	if(flag)
	printf("%lld",prev);
	else
	printf("%lld",a[n-1]);
	
    return 0;
}
