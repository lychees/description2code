int main(){
	int t,n,k,i,j;
	scanf("%d",&t);
	while(t--){
		scanf("%d%d",&n,&k);
		int a[100005]={0};
		for(i=0;i<n;i++){
			scanf("%d",&j);
			a[j]++;
		}
		int64 ans=0;
		for(i=100000;i>=0;i--){
			if(!k)
			break;
			while(a[i]){
				if(!k)
				break;
				ans+=i;
				a[i]--;
				k--;
			}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
