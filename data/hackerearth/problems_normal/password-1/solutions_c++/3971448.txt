#include <iostream>
#include<string>
#include<algorithm>
using namespace std;

int main()
{
   int n;
   cin>>n;
   string s[n];
   for(int i=0;i<n;i++){
   	cin>>s[i];
   }
   for(int i=0;i<n;i++){
   	reverse(s[i].begin(),s[i].end());
   	for(int j=i+1;j<n;j++){
   	
   	if(s[i]==s[j]){
   		int length=s[i].size();
   		cout<<length<<" "<<s[i][length/2]<<endl;
   		
   	}
   }}
    return 0;
}
