#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	int n,m,p;
    	cin>>n>>m;
    	int x1,x2,y1,y2;
    	cin>>x1>>y1>>x2>>y2;
    	cin>>p;
    	int cost  = p*(min(abs(x1-x2),n-abs(x1-x2))+min(abs(y1-y2),m-abs(y1-y2)));
    	cout<<cost<<endl;
    	if(cost <= 1000)
    	{
    		cout<<"You saved the group."<<endl;
    	}
    	else
    		cout<<"Let go of the group."<<endl;
    	
    	t--;
    }
    return 0;
}
