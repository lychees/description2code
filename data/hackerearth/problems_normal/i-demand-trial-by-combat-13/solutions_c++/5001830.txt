#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int nT;
    cin >> nT;
    
    int nN, nM;
    int nCurrent;
    
    vector<int> vA(100);
    string s1;
    
    for(int i=0; i<nT; i++) {
    	cin >> nN >> nM;
    	
    	for(int j=0; j<nN; j++) {
    		scanf("%d",&vA[j]);
    	}
    	
    	nM = min(nM, 10000);
    	
    	for(int k=0; k<nM; k++) {
    		bool bSeated = true;
    		nCurrent = vA[0];
    		if(vA[1]) {
    			vA[0] = 1;
    		} else {
    			bSeated = false;
    			vA[0] = 0;
    		}
    		for(int j=1; j<nN-1; j++) {
    			if(nCurrent && vA[j+1]) {
    				nCurrent = vA[j];
	    			vA[j] = 1;
	    		} else {
	    			bSeated = false;
	    			nCurrent = vA[j];
	    			vA[j] = 0;
	    		}
    		}
			if(nCurrent) {
	    		vA[nN-1] = 1;
	    	} else {
	    		bSeated = false;
	    		vA[nN-1] = 0;
	    	}
	    	if(bSeated) break;
    	}
    	
    	for(int j=0; j<nN; j++) {
    		printf("%d ",vA[j]);
    	}
    	printf("\n");
    }
}
