#include<bits/stdc++.h>
using namespace std;
int main(){

  cin.sync_with_stdio(false);
  int t;
  cin >> t;
  while(t--){

     int n;
     cin >> n;
     int s = 0;
     int x;
     for( int i = 0 ; i < n; ++i ){
       cin >> x;
       s += x;
     }
     double ans = s/(n * 1.0);

     printf("%.8lf\n",ans);


  }

}
