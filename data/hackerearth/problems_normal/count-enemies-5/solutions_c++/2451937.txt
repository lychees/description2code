#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    
    while (t-- > 0) {
    	string s;
    	cin >> s;
    	
    	int survive = 0;
    	int enemies = 0;
    	int team = 0;
    	for (int i = 0; i < s.size(); i++) {
    		if (s[i] == 'O') enemies++;
    		else if (s[i] == 'X') team++;
    		else {
    			if (team == 0) survive += enemies;
    			enemies = 0;
    			team = 0;
    		}
    		if (i == s.size()-1) {
    			if (team == 0) survive += enemies;
    			enemies = 0;
    			team = 0;
    		}
    	}
    	
    	cout << survive << endl;
    }
    return 0;
}
