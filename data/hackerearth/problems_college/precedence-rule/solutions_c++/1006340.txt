#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t;
    //char s[31];
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	//int len=s.length();
    	int temp=0;
    	//int a=s[len-1]-'0',b=s[len-2],c=s[len-3]-'0';
    	long long result=s[s.length()-1]-'0';
   
    	for(int j=s.length()-3;j>=0;j=j-2)
    	{
    	temp=s[j]-'0';
    	if(s[j+1]=='*')
    	{
    		result *= (long long)temp;
    	}
    	else if(s[j+1]=='+')
    	{
    		result += (long long)temp;
    	}
    		else if(s[j+1]=='-')
    	{
    		result -=(long long)temp;
    	}
    
    		
    	}
    	cout<<result<<endl;
    	
    }
    return 0;
}
