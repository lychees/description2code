def t():
    n = int(input())
    g = False
    result = "Yes"
    for j in xrange(n):
        r,c,x = [int(x) for x in raw_input().split()]
        p = (r + c)%2
        if g == False:
            g = abs(p - x%2) + 1
        else:
            if(g != abs(p - x%2) + 1):
                result = "No"
    print(result)

N = int(input())
for i in xrange(N):
    t()
