#include <bits/stdc++.h>
using namespace std;

int fib(int n, int* o, int *z) {
  if(n == 0) {
    ++*z;
    return 0;
  }
  if(n == 1) {
    ++*o;
    return 1;
  }
  return fib(n-1, o, z) + fib(n-2, o, z);
}
int main() {
  int t;
  cin >> t;
  for(;t--;) {
    int n;
    cin >> n;
    int oc = 0, zc = 0;
    fib(n, &oc, &zc);
    cout << zc << ' ' << oc << endl;
  }
  return 0;
}
