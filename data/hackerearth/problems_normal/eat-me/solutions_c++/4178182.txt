#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll dp[1002][1002];
#define mod 1000000007
int main()
{
  ll int n,k;
  cin>>n>>k;
  ll int i,j;
  for(i=1;i<=n;i++)
  {
      dp[i][1]=i;
  }
  for(i=1;i<=n;i++)
  {
      for(j=2;j<=k;j++)
      {
          dp[i][j]=(dp[i-1][j]%mod+dp[i][j-1]%mod)%mod;
      }
  }
  ll sum=0;
      for(j=1;j<=k;j++)
      {
          sum=(sum%mod+dp[n][j]%mod)%mod;
      }

      cout<<sum%mod<<endl;


}
