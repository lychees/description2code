#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007LL
#define ll long long
#define sd(x) scanf("%d",&x)
#define sll(x) scanf("%lld",&x)

ll power(ll a,ll b)
{
	ll res = 1;
	while(b)
	{
		if(b&1)
		{
			res *= a;
			res %= MOD;
		}
		a *= a;
		a %= MOD;
		b /= 2;
	}
	return res;
}

#define MAXN 10010
ll DP[MAXN];

int main()
{ 
	for(int i=1;i<MAXN;++i)
		for(int j=i;j<MAXN;j+=i)
		{
			DP[j] += power(i,j);
			if(DP[j]>MOD)
				DP[j] -= MOD;
		}
	for(int i=1;i<MAXN;++i)
	{
		DP[i] += DP[i-1];
		if(DP[i]>MOD)
			DP[i] -= MOD;
	}
	int t;
	sd(t);
	while(t--)
	{
		int a,b;
		sd(a);
		sd(b);
		--a;
		ll res = DP[b]-DP[a];
		if(res<0)
			res += MOD;
		printf("%lld\n",res);
	}	
	return 0;
}
