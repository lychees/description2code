n = int(raw_input().strip())

speeds = list(map(int, raw_input().strip().split()))

con = list(map(int, raw_input().strip().split()))

fuel = int(raw_input().strip())

max = float(speeds[0])/float(con[0])
# print(3/2)
# print(speeds)
for i in range(1, n):
    temp = (float(speeds[i])/float(con[i]))
    # print temp
    if max < temp:
        max = temp


a = str(max * fuel).split(".")
a[1] = a[1][:3]
if (len(a[1]) < 3):
    a[1] = a[1]+"00"
a[1] = a[1][:3]
print ".".join(a)
