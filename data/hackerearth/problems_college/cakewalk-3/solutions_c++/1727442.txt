#include <iostream>
using namespace std;

int main()
{
    int n;
    cin >> n;
    string s;
    cin >> s;
    string ans = "";
    for (int i = 0; i < n-1; ++i)
    {    	    	
    	int x;
    	cin >> x;    	
    	for (int j = 0; j < x; ++j)
    	{
    		ans += s[j];
    	}
    	cin >> s;
    	for (char ch : s)
    		ans += ch;
    	s = ans;
    	ans = "";
    }
    
    cout << s;
    
    return 0;
}
