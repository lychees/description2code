#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,q,x,i;
    map<int,int>m;
    scanf("%d%d",&n,&q);
    for(i=0;i<n;i++){
    	scanf("%d",&x);
    	m[x]=1;
    }
    while(q--){
    	scanf("%d",&x);
    	if(m.find(x)!=m.end()){
    		printf("YES\n");
    	}
    	else{
    		printf("NO\n");
    	}
    }
    return 0;
}
