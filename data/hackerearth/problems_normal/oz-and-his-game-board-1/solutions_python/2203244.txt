t = input()
while t > 0:
	t -= 1
	n, m, nc, mc, oc = map(int, raw_input().split(" "))
	cost1 = (n*nc) + (m*mc)
	cost2 = -99999999999999
	cost3 = -99999999999999
	cost4 = -99999999999999
	if n > m:
		cost2 = (m*oc) + (n-m)*nc + (m*oc)
	elif n < m:
		cost3 = (n*oc) + (m-n)*mc + (n*oc)
	else:
		cost4 = (m+n)*oc
	print max(cost1, cost2, cost3, cost4)
