#include <bits/stdc++.h>

using namespace std;

int N, counter = 0, temp;
int Call[101], Correct[101];

int main()
{
    queue <int> Q;
    cin >> N;
    for(int i = 0; i < N; i++)
    {
        cin >> Call[i];
        Q.push(Call[i]);
    }
    for(int i = 0; i < N; i++)
        cin >> Correct[i];

    for(int i = 0; i < N; i++)
    {
        while(Q.front() != Correct[i])
        {
            temp = Q.front();
            Q.pop();
            Q.push(temp);
            counter++;
        }
        Q.pop();
        counter++;
    }
    cout << counter << endl;
}
