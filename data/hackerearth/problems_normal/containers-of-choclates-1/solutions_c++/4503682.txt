#include<bits/stdc++.h>
using namespace std;
vector<int> adj[20005];
int vis[20005]={0},pre[20005]={0},ct[2]={0};
int main()
{
    int t,i,n,j,sum,a,b,ele,side,f=1;
    ios::sync_with_stdio(0);
    cin.tie(0);
    cin>>t;
    while(t--)
    {
        cin>>n;
        sum=0;
        fill(vis,vis+20005,0);
        fill(pre,pre+20005,0);
        fill(ct,ct+2,0);
        for(i=0;i<20005;i++)
            adj[i].clear();
        for(i=0;i<n;i++)
        {
            cin>>a>>b;
            adj[a].push_back(b);
            adj[b].push_back(a);
            pre[a]=1;
            pre[b]=1;
        }
        for(i=1;i<=20000;i++)
        {
            if(pre[i]==1&&vis[i]==0)
            {
                queue<pair<int,int> > q;
                q.push(make_pair(i,0));
                vis[i]=1;
                while(!q.empty())
                {
                    ele=q.front().first;
                    side=q.front().second;
                    q.pop();
                    ct[side]++;
                    for(j=0;j<adj[ele].size();j++)
                    {
                        if(vis[adj[ele][j]]==0)
                        {
                            vis[adj[ele][j]]=1;
                            q.push(make_pair(adj[ele][j],(side+1)%2));
                        }
                    }
                }
                sum+=max(ct[0],ct[1]);
                ct[0]=0;
                ct[1]=0;
            }
        }
        cout<<"Case "<<f++<<": "<<sum<<"\n";
    }
    return 0;
}
