long long int max(long long int a, long long int b)
{
    if (a>b)
    {
        return a;
    }
    return b;
}

{

   	k++;

   	for(i=0;i<n;i++)
   	cin>>arr[i];
   	dp[0] = arr[0];
   	if(arr[1]>arr[0])
   	dp[1] = arr[1];
