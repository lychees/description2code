import sys


def main(file):
    ''' Main '''
    T = int(file.readline().strip())

    lines = file.readlines()

    for line in lines:
        a, b = map(long, line.strip().split(" "))

        while a > 1 and b > 1 and a != b:
            if b > a:
                b /= 2
            else:
                a /= 2

        print min(a, b)


if __name__ == "__main__":
    main(sys.stdin)
