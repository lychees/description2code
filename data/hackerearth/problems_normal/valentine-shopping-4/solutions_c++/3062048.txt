#include <iostream>
using namespace std;

int main()
{
        int t,d1,d2,d3,index,inc,min_index,n,m;
    double num,min;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d %d",&n,&m);
        for(index=0;index<n;index++)
        {
    	min_index=0;
    	min=1000;
    	for(inc=0;inc<m;inc++)
    	{
    		num=1000;
    		scanf("%d %d %d",&d1,&d2,&d3);
    		num=(100-d1)*num/100;
    		num=(100-d2)*num/100;
    		num=(100-d3)*num/100;
            if(num<min)
            {
            	min=num;
            	min_index=inc;
            }
    	}
    	printf("%d ",(min_index+1));
        }
        printf("\n");
    }
    return 0;
}
