#!/usr/bin/env python

import math
import sys

def gcd(a,b):

    while (b != 0):
        temp = a % b
        a = b
        b = temp
    return abs(a)


def solve(a, b, c):

    x = gcd(a,b)
    y = gcd(b,c)

    ret = gcd(x,y)

    num = ((a/ret) * (b/ret) * (c/ret)) % 1000000007

    return ret, num

testCount = eval(raw_input())

for test in xrange(1, testCount+1):

    inputs = raw_input().split()
    answer = solve(int(inputs[0]), int(inputs[1]), int(inputs[2]))
    print "%d %d" % (answer[0], answer[1])
