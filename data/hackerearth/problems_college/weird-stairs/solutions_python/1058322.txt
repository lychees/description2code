t=input()

for i in range(t):
    w,n=map(int,raw_input().split())
    l=map(int,raw_input().split())
    p=0
    s=0
    v=0
    for j in range(n):
        if l[j]==1:
            v=2*(j+1)           
        else:
            v=p+1
        s+=(v-p)
        p=v
    s+=n*w
    print s
