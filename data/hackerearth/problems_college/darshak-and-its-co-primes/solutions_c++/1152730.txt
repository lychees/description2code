#include<stdio.h>
#define ll long long
ll gcd ( ll a, ll b )
{
  ll c;
  while ( a != 0 ) {
     c = a; a = b%a;  b = c;
  }
  return b;
}
int main()
{
    ll t;
    scanf("%lld",&t);
    while(t--)
    {
        long long p,q;
        scanf("%lld%lld",&p,&q);
        if(gcd(p,q)==1)
            printf("Is a Co-Prime\n");
        else
            printf("Not a Co-Prime\n");

    }
    return 0;
}
