//this fn taken from geeksforgeeks
ll MaxSum(ll arr[], ll n)
{
  ll incl = arr[0];
  ll excl = 0;
  ll excl_new;
  ll i;
 
  for (i = 1; i < n; i++)
  {
     /* current max excluding i */
     excl_new = (incl > excl)? incl: excl;
 
     /* current max including i */
     incl = excl + arr[i];
     excl = excl_new;
  }
 
   return ((incl > excl)? incl : excl);
}
