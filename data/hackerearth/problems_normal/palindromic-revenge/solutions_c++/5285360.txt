#include <iostream>
using namespace std;

int main()
{
    int t; cin>>t;
    while(t--){
    	string s; cin>>s;
    	int i,j,ans=0;
    	if(s.length()&1) i=s.length()/2-1,j=s.length()/2+1;
    	else i=s.length()/2-1,j=s.length()/2;
    	while(i>=0){
    		int t1=s[i]-'a',t2=s[j]-'a';
    		ans+=abs(t1-t2);	
    		i--,j++;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
