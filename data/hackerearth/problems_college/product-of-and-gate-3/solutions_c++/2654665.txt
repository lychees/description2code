#include<iostream>
using namespace std;
long long int calc(long long int a,long long int b)
{
	
	long long int x=a^b;
	long long int s=x>>1;
	while(s)
	{
		x=x|s;
		s=s>>1;
	}
	return a&b&~x;
}
main()
{
	long long int t;
	cin>>t;
	while(t--)
	{
		long long int a,b;
		cin>>a>>b;
		long long int res=calc(a,b);
		cout<<res<<endl;
	}
}
