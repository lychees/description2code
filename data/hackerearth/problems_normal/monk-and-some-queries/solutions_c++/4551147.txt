#include <iostream>
#include <set>
using namespace std;

int main()
{
	//std::ios_base::sync_with_stdio(false);
	int T, condition, val;
    multiset<int> lst;
    cin >> T;
    while(T--){
    	cin >> condition;
    	switch(condition){
    		case 1 :{
    			cin >> val;
    			lst.insert(val);
    			break;
    		}
    		case 2 :{
    			cin >> val;
    			multiset<int>::iterator it = lst.find(val);
    			if(it!=lst.end())
    				lst.erase(it);
    			else
    				cout << "-1" << endl;
    			break;
    		}
    		case 3 :{
    			if(!lst.empty())
    			cout << *(lst.rbegin()) << endl;
    			else
    			cout << "-1" << endl;
    			break;
    		}
    		case 4:{
    			if(!lst.empty())
    			cout << *(lst.begin()) << endl;
    			else
    			cout << "-1" << endl;
    			break;
    		}
    	}
    }
    
    return 0;
}
