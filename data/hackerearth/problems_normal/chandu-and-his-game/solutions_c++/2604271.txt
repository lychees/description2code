#include <bits/stdc++.h>
using namespace std;
int x[100000],y[100000];
int main()
{
	int N;
    long int K,ans=0;
    cin>>N>>K;
    for(long int i=0;i<K;i++)
    {
    cin>>x[i];	
    }
    for(long int i=0;i<K;i++)
    {
    cin>>y[i];	
    }
    for(long int i=0;i<K;i++)
    {
    	ans=ans+min(x[i]-1,N-x[i])+min(y[i]-1,N-y[i]);
    }
    cout<<ans;
}
