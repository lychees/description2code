#include <iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;

long long C[2101][2101];
const int mod = 1000000007;
int main()
{
	for(int i = 0;i <= 2100;++i){
		C[i][i] = C[i][0] = 1;
		for(int j = 0;j < i;++j){
			C[i][j] = (C[i - 1][j - 1] + C[i - 1][j]) % mod;
		}
	}
	int test,n,p,x,y;
	scanf("%d",&test);
	while(test--){
		scanf("%d%d",&n,&p);
		vector<pair<int,int>> cord;
		while(p--){
			scanf("%d%d",&x,&y);		
			cord.push_back(make_pair(x,y));
		}
		cord.push_back(make_pair(0,0));
		cord.push_back(make_pair(n - 1,n - 1));
		sort(cord.begin(),cord.end());
		long long res = 1;
		for(int i = 1;i < (int)cord.size();++i){
			int x = cord[i].first - cord[i - 1].first,y = cord[i].second - cord[i - 1].second;
			res *= C[x + y][y];
			res %= mod;
		}
		printf("%lld\n",res);
	}
	
    return 0;
}
