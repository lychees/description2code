#include<bits/stdc++.h>
using namespace std;
vector<int> v[100001];

int main()
{
string c;
char ch;
cin>>c;
int q , n ,pos ,i;
n = c.size();
for(i=0;i<n;i++)
{
v[c[i]].push_back(i);
}
cin>>q;
while(q--)
{
cin>>pos>>ch;
c[v[ch][pos-1]]='0';
v[ch].erase(v[ch].begin()+pos-1);
}
for(i=0;i<n;i++)
{
if(c[i]!='0') cout<<c[i];
}
return 0;
}
