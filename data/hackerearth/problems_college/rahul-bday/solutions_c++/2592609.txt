#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    int t;
    cin >> t;
    while(t--)
        {
        long long int b,w;
        cin >> b >> w;
        long long int x,y,z;
        cin >> x >> y >> z;
        long long int ans;
        if(x>=y)
            {
                if(y+z<x)
                    {
                    ans = w*y + (y+z)*b;
                }
                else
                    ans = w*y + x*b;
            }
        else
            {
            if(x+z<y)
                {
                ans = b*x + (x+z)*w;
            }
            else
                ans = w*y + x*b;
        }
        cout << ans << endl;
    }
    return 0;
}
