#include<iostream>
using namespace std;
long long a[107];
int main()
{
	int t;
	cin >> t;
	for (int tt = 1; tt <= t; tt++)
	{
		int n;
		long long m;
		cin >> n >> m;
		for (int i = 1; i <= n; i++)
		{
			cin >> a[i];
		}
		int ans = 100000007;
		for (int mask = 0; mask < (1 << n); mask++)
		{
			int tmp = mask;
			long long sum = 0;
			int curans = 0;
			for (int i = 1; i <= n; i++)
			{
				if (tmp % 2)
				{
					sum += a[i];
					curans++;
				}
				tmp /= 2;
			}
			if (sum >= m && curans < ans) ans = curans;
		}
		if (ans>n) ans = -1;
		cout << ans << endl;
	}
}
