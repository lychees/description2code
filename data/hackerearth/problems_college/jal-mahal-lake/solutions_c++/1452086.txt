typedef long long LL;

#define MOD 2000000007
#define MAX 1000000

using namespace std;

int main() {
	int t;
	cin >> t;
	
	while (t--) {
		int n;
		cin >> n;
		pair<LL, LL> arr[n];
	//	int len[n];
	//	int x[n];
		for (int i=0; i<n; i++) cin>>arr[i].first;
		for (int i=0; i<n; i++) cin>>arr[i].second;
	//	int sum[n];
	//	for (int i=0; i<n; i++) sum[i]=sum[i-1]+arr[i].second;
		sort(arr,arr+n);
		LL ans=MOD;
		ans*=1000000;
		for (int j=0; j<n; j++) {
			LL sum=0;
			LL last=arr[j].first+arr[j].second;
			for (int i=j+1; i<n; i++) {
				sum += arr[i].first-last;
				last+=arr[i].second;
			}
			last=arr[j].first;
			for (int i=j-1; i>=0; i--) {
				sum += last-arr[i].first-arr[i].second;
				last -= arr[i].second;
			}
			if(ans>sum)ans=sum;
		}
		
		cout << ans << endl;
		
	}
	
	
	return 0;
}
