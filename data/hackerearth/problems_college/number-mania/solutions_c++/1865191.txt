#include<iostream>
using namespace std;
int main()
{
      int pr[101]={0};
      for(int i = 2; i <= 100 ; ++i ){
              if(pr[i]==0)
        for(int j = 2*i ; j <= 100 ; j=j+i)
         pr[j]=1;
      }

	int t,i,j,  n;
		cin>>t;
      for(int ii = 2 ; ii <= t ; ++ii ){
           if(pr[ii]==0)
            {
                n = ii;
		int arr[10000],r=0,carr,val;
		arr[0]=1;
		for(i=2;i<=n;i++)
		{
			  carr=0;
			for(j=0;j<=r;j++)
			{
				val=arr[j]*i+carr;
				arr[j]=val%10;
				carr=val/10;
			}
			while(carr)
			{
				r++;
				arr[r]=carr%10;
				carr=carr/10;
			}

		}
		for(j=r;j>=0;j--)
		{
			cout<<arr[j];
		}
		cout<<endl;
           }
      }
	return 0;
}
