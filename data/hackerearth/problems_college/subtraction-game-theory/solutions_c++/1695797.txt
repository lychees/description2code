#include <iostream>
#include <stdio.h>
using namespace std;
long long sod (long long no)
{
long long sum=0;
while (no!=0)
{
 sum+=no%10;
 no/=10;
}
return sum;
}
int main()
{
long long no,total=0,totalsod=0,i,x;
cin>>no;
for (i=0;i<no;i++)
{
	cin>>x;
	totalsod+=sod (x);
	total+=x;
}	
//cout<<totalsod%9-sod(total)%9;
cout<<"0";
}
