#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,i;
    scanf("%d",&t);

    int n,arr[100005];
    bool visited[100005];
    while(t--)
    {
        scanf("%d",&n);
        memset(visited, 0, sizeof(visited[0])*100005);

        for(i=0;i<n;i++)
            scanf("%d",&arr[i]);

        i=0;
        while(arr[i]!=-1 && visited[i]!=1)
        {
            visited[i]=1;
            i = arr[i];
        }
        if(arr[i]==-1)
            printf("Win\n");
        else
            printf("Lose\n");

    }
}
