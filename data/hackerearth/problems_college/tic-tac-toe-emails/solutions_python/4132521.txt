T=int(raw_input())
while T:
    T-=1
    s=raw_input().split()
    ans=""
    check=[]
    check=set(check)
    for i in range(len(s)):
        before=len(check)
        check.add(s[i])
        after=len(check)
        if(after != before):
            ans=ans+s[i]+" "
    print ans
