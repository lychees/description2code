#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	double s[t]; double p[t]; double cap,sol,max=0;
	for(int i=0;i<t;i++)cin>>s[i];
	for(int i=0;i<t;i++)cin>>p[i];
	cin>>cap;
	for(int i=0;i<t;i++){
		sol=(cap/p[i])*s[i];
		if (sol>max){max=sol;}
	}
	std::cout << std::fixed;
    std::cout << std::setprecision(3);
    cout<<(double)((int)(1000*max))/1000;
	return 0;
}
