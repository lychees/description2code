#include <bits/stdc++.h>
using namespace std;
 
typedef long long LL;
 
#define PII       pair<int,int>
#define all(c)    c.begin(),c.end()
#define sz(c)     (int)c.size()
#define clr(c)    c.clear()
#define pb        push_back
#define mp        make_pair
#define cin(x)    scanf("%d",&x)
#define MOD       1000000009
#define EPS       1E-10

int area(PII A , PII B , PII C)
{
	int f= A.first * (B.second - C.second) + B.first * (C.second - A.second) + C.first * (A.second - B.second);
	if(f < 0) f = -f;
	return f;
}

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		PII A , B , C , D;
		cin >> A.first >> A.second;
		cin >> B.first >> B.second;
		cin >> C.first >> C.second;
		cin >> D.first >> D.second;
		
		int a1 = area(A,B,C);
		int a2 = area(A,B,D);
		int a3 = area(C,B,D);
		int a4 = area(A,C,D);

		if((a2 + a3 + a4) == a1)
			cout << "INSIDE\n";
		else
			cout << "OUTSIDE\n";
	}
	return 0;
}
