#include<bits/stdc++.h>
using namespace std;
int main()
{
	int N;
	map< string,int >Mp;
	map<string,int >::iterator it;

	cin>>N;
	for(int i=0;i<N;i++)
	{
		string s;
		cin>>s;
		sort(s.begin(),s.end());
		Mp[s]+=1;
	}

	int max=-1;
	for(it=Mp.begin();it!=Mp.end();it++)
	{
		if(max<it->second)max=it->second;


	}

	cout<<max<<endl;


	return 0;
}
