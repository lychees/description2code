#include <iostream>
#include<vector>
#include<algorithm>
#include<math.h>

using namespace std;

int main()
{
    int t,n,m,i,j,k,l;
    cin>>t;
    while(t--){
    cin>>n;
    cin>>m;
    vector<int> v(n);
    for(i=0;i<n;i++){
    	cin>>j;
    	v[i]=j;
    }
    i=0;
    while(i<n){
    	k=1;
    	l=i-m;
    	if(l<0){
    		l=abs(l%2);
    	}
    	while(l<=m+i && k==1 && l<n){
    		k=k&&v[l];
    		l+=2;
    	}
    	i++;
    	cout<<k<<" ";
    }
    cout<<endl;
	}
return 0;
}
