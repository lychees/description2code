t = input()
for x in xrange(0, t):
    nm = raw_input().split()
    nm = map(int, nm)
    done = {}
    candies = raw_input().split()
    candies = map(int, candies)
    for y in xrange(0, nm[0]):
        done[candies[y]] = True
    for y in xrange(nm[0], len(candies)):
        try:
            if (done[candies[y]]):
                print("YES")
        except BaseException:
            done[candies[y]] = True
            print("NO")
