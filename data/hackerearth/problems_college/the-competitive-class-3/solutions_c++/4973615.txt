#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n],i,j,count=0,r;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    for(i=0;i<n;i++)
    {
    	count=0;
    	for(j=0;j<n;j++)
    	{
    		if(a[i]<a[j])
    		{
    			count++;
    			
    		}
    	}
    	cout<<(count+1)<<" ";
    }
    cout<<endl;
    return 0;
}
