#include <iostream>
using namespace std;

int main() {
	int t,ans,flag;
	string s1,s2,p;
	cin>>t;
	while(t--)
	{
		ans=0;
		cin>>s1>>s2;
		for(int i=0;i<=s1.length();i++)
		{
			p=s1.substr(0,i);
			p+=s2;
			p+=s1.substr(i,s1.length()-i);
			flag=0;
			for(int j=0;j<p.length()/2;j++)
			{
				if(p[j]!=p[p.length()-1-j]){flag=1;break;}
			}
			if(flag==0)ans++;
		}
		cout<<ans<<endl;
	}
	// your code goes here
	return 0;
}
