#include <iostream>
#include <stdio.h>

using namespace std;

int A[1001][1001],N,M;
long int counter=0;
bool visited[1001][1001];

void initialise()
    {
    for(int i=1;i<=N;i++)
      for(int j=1;j<=M;j++)
        {
            if(A[i][j]==1)
               {
                   ++counter;
                   visited[i][j]=false;
               }
            else
                visited[i][j]=true;
        }
    }
void dfs(int x,int y)
     {
         if(visited[x][y]==true)
            return;
         else
         {
             visited[x][y]=true;
             --counter;
             if(x>=2&&visited[x-1][y]==false)
             {
                 //--counter;
                 dfs(x-1,y);
             }
             if(x<=N-1&&visited[x+1][y]==false)
             {
                 //--counter;
                 dfs(x+1,y);
             }
             if(y>=2&&visited[x][y-1]==false)
             {
                 //--counter;
                 dfs(x,y-1);
             }
             if(y<=M-1&&visited[x][y+1]==false)
             {
                 //--counter;
                 dfs(x,y+1);
             }
         }
     }
int main()
{
    long int Q;
    int x,y;
    scanf("%d %d %ld",&N,&M,&Q);
    for(int i=1;i<=N;i++)
        for(int j=1;j<=M;j++)
           scanf("%d",&A[i][j]);
    initialise();
    while(Q--)
    {
        scanf("%d %d",&x,&y);
        dfs(x,y);
        printf("%ld\n",counter);
    }
}
