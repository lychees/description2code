typedef vector<int> vi; typedef pair<int,int> pii; typedef vector<pair<int,int> > vpii; typedef long long ll;
template<typename T, typename U> inline void amin(T &x, U y) { if(y < x) x = y; }
template<typename T, typename U> inline void amax(T &x, U y) { if(x < y) x = y; }

struct FenwickTree {
	typedef int T;
	vector<T> v;
	void init(int n) { v.assign(n, 0); }
	void add(int i, T x) {
		for(; i < (int)v.size(); i |= i+1) v[i] += x;
	}
	T sum(int i) const {	//[0, i)
		T r = 0;
		for(-- i; i >= 0; i = (i & (i+1)) - 1) r += v[i];
		return r;
	}
	T sum(int left, int right) const {	//[left, right)
		return sum(right) - sum(left);
	}
};

int main() {
	int N, M;
	while(~scanf("%d%d", &N, &M)) {
		vi lasttime(N);
		vector<char> lastdir(N);
		FenwickTree fts[2];
		rep(k, 2) fts[k].init(N + M + 1);
		int now = 0;
		rep(i, N) {
			int x;
			scanf("%d", &x), -- x;

			lasttime[x] = now;
			lastdir[x] = 'r';

			fts[1].add(now, 1);

			++ now;
		}
		ll anssum = 0;
		rep(i, M) {
			int x; char dir[2];
			scanf("%d%s", &x, dir), -- x;

			int pos;
			if(lastdir[x] == 'l')
				pos = fts[0].sum(lasttime[x]+1, now);
			else
				pos = N - 1 - fts[1].sum(lasttime[x]+1, now);
//			cerr << "pos = " << pos << endl;
			anssum += min(pos, N - 1 - pos);

			fts[lastdir[x] == 'r'].add(lasttime[x], -1);
			fts[*dir == 'r'].add(now, +1);

			lasttime[x] = now;
			lastdir[x] = *dir;

			++ now;
		}
		cout << anssum << endl;
	}
	return 0;
}
