#include <iostream>
using namespace std;
int reverse(int num){
      int new_num = 0;
    while(num > 0)
    {
            new_num = new_num*10 + (num % 10);
            num = num/10;
    }
    return new_num ;
}
int main()
{
    int t,n1,n2,p,q,r;
    cin>>t;
    while(t--)
    {
    	cin>>n1>>n2;
    	q=reverse(n1);
    	r=reverse(n2);
    	p=q+r;
    
    cout<<reverse(p)<<endl;
    }
    //cout << "Hello World!" << endl;
    return 0;
}
