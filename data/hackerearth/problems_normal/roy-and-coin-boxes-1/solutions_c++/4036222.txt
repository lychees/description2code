#include<bits/stdc++.h>
using namespace std;

const int N = 1000010;
int n, m, l, r, q, x, a[N], f[N];

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d%d", &n, &m);
	while(m--) {
		scanf("%d%d", &l, &r);
		++a[l], --a[r+1];
	}
	for(int i=1 ; i<=n ; ++i)
		a[i] += a[i-1];
	for(int i=1 ; i<=n ; ++i)
		a[i] = min(a[i], 1000000);
	for(int i=1 ; i<=n ; ++i)
		++f[a[i]];
	for(int i=1000000 ; i>=1 ; --i)
		f[i] += f[i+1];
	scanf("%d", &q);
	while(q--) {
		scanf("%d", &x);
		printf("%d\n", f[x]);
	}
	return 0;
}
