#include <bits/stdc++.h>
using namespace std;

const int maxn = 1e5+100;

int n;
map <char,int> mark;

int main(){
    string s;
    cin >> s;
    for(int i = 0;i < s.size();i++)
    	mark[s[i]]++;
    int odd = 0;
    for(int i = 0;i < 26;i++)
    	if(mark[i+'a'] % 2 == 1)
    		odd++;
    if(odd >= 2){
    	cout << -1 << endl;
    	return 0;
	}
	string t = "";
    for(int i = 0;i < 26;i++){
    	while(mark[i + 'a'] >= 2){
    		cout << char(i + 'a');
    		mark[i + 'a'] -= 2;
    		t += char(i+'a');
		}
	}
	for(int i = 0;i < 26;i++)
		if(mark[i + 'a'] == 1)
			cout << char(i + 'a');
	for(int i = t.size()-1;i >= 0;i--)
		cout << t[i];
}
