#include<bits/stdc++.h>

using namespace std;

long long int add (long long int n)
{    
	if(n==1)
        return 1;
	 return floor((n*n)/2)-1;
}

int main()
{
  int t;
  long long int n;
  scanf("%d",&t);
  while(t--)
  {
    scanf("%lld",&n);
    cout<<add(n)<<endl;
  }
  return 0;
}
