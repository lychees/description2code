#include <iostream>
using namespace std;
int main()
{
    int t_c;
    int c_stone,t_stone,stone1;
    int day=0,stone;
    cin>>t_c;
    for(int i=0; i<t_c; i++)
    {
        cin>>stone;
        day=0;
        t_stone=0;
        c_stone=0;
        stone1=stone;
        while(t_stone!=stone1)
        {
            day++;
            c_stone=1;
            while(c_stone<=stone)
                c_stone=c_stone*2;
            c_stone=c_stone/2;
            stone=stone-c_stone;
            t_stone=t_stone+c_stone;
        }
        cout<<day<<endl;
    }
}
