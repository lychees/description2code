#include<bits/stdc++.h>
 
int main()
{
 
    int n;
    scanf("%d\n",&n);
 
    char s[201];
    while(n--)
    {
        gets(s);
        int l=strlen(s),r=sqrt(l);
        int z,j,i;
        for(i=0;i<r;i++)
        {
            z=i;
            for(j=0;j<r;j++)
            {
                printf("%c",s[z]);
                z=(z+r);
            }
        }
         printf("\n");
    }
 
 
 
    return 0;
}
