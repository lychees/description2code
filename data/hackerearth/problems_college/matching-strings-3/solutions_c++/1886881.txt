#include <iostream>
using namespace std;
#include<string.h>
int main()
{
	int t;
	char str1[600],str2[600];
	scanf("%d",&t);
	int i,j,ct,l1,l2;
	while(t--)
	{
		scanf("%s",str1);
		scanf("%s",str2);
		l1=strlen(str1);
		l2=strlen(str2);
		i=0; j=0;
		ct=0;
		while(i<l1 && j<l2)
		{
			if(str1[i]==str2[j])
				ct++;
				i++;
				j++;
		}
		printf("%d\n",ct);
		strcpy(str1, "");
		strcpy(str2, "");
	}
}
