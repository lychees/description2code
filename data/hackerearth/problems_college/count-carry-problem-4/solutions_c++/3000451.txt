#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin>>t;
    while(t--){
    	long long int a,b,cnt=0;
    	cin>>a>>b;
    	while(a!=0&&b!=0)
    	{
    		int r,r1;
    		r = a % 10;
    		a = (a-r)/10;
    		r1 = b%10;
    		b = (b-r1)/10;
    		if(r+r1>=10) cnt++;
    	}
    	if(cnt == 0) {
    		cout<<"No carry operation"<<endl;
    	}
    	else if(cnt == 1)
    		cout<<cnt<<" carry operation"<<endl;
    	else
    		cout<<cnt<<" carry operations"<<endl;
    }
	return 0;
}
