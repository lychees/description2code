#include <iostream>
#include <stdio.h>
#include <stdlib.h>
using namespace std;
long long int n;
long long int arr[1000000];
int main(){
    cin >> n ;
    cin >> arr[0];
    long long mx = arr[0];
    long long mn = arr[0] ;
    for(int i=1;i<n;i++){
        cin >> arr[i];
        if(arr[i]  > mx){mx = arr[i];}
        if(arr[i] < mn){mn = arr[i];}
    }
    cout << mn*mx << endl ;
    return 0;
}
