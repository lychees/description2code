const int inf = (int) 1e9;
const int maxn = (int) 1e5 + 10;

//map<pair<int, int>, ll> MP;
map<int, ll> MP[55];

int gcd(int a, int b) {
  if (b == 0) return a;
  return gcd(b, a % b);
}

int L, R, n, a[maxn];

ll dp(int i, int g) {
  map<int, ll> :: iterator it = MP[i].find(g);
  if (it != MP[i].end()) return it->second;
  ll &res = MP[i][g];
  if (i == n) {
    if (g >= L && g <= R) res = 1;
  } else {
    res += dp(i + 1, g);
    res += dp(i + 1, gcd(g, a[i]));
  }
  return res;
}

int main() {
  int T;
  scanf("%d", &T);
  while (T--) {
    scanf("%d", &n);
    rep(i, n) scanf("%d", &a[i]);
    scanf("%d %d", &L, &R);
    rep(i, n + 1) MP[i].clear();
    ll ans = dp(0, 0);
    printf("%lld\n", ans);
  }

  return 0;
}
