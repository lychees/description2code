#include<iostream>
#include<vector>
#include<algorithm>
#include<string>

using namespace std;

int main()
{
	string s,query;
	cin>>s;
	int n,currposs;
	bool yes;
	vector<vector<int> > v;
	v.resize(26);
	for(int i=0;i<s.length();i++)
	{
		v[int(s[i])-97].push_back(i);
	}
	cin>>n;
	vector<int>::iterator it;
	for(int i=0;i<n;i++)
	{
		cin>>query;
		currposs=-1;
		yes=true;
		for(int j=0;j<query.length()&&yes;j++)
		{
			it=upper_bound(v[int(query[j])-97].begin(),v[int(query[j])-97].end(),currposs);
			if(it==v[int(query[j])-97].end()){yes=false;}
				else{currposs=*it;}
		}
		if(yes)
		{
			cout<<"Yes"<<endl;
		}
		else cout<<"No"<<endl;

	}
}
