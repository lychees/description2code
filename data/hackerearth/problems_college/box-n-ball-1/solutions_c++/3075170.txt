#include<iostream>
#include<iomanip>
using namespace std;
int main()
{
	long t,l,b,h;
	double result,min;
	cin>>t;
	while(t--)
	{
		cin>>l>>b>>h;
		if(l<b)
		{
			if(l<h)
				min=l;
			else
				min=h;
		}
		else
		{
			if(b<h)
				min=b;
			else
				min=h;
		}
		result=min/2;
		std::cout<<std::fixed;
		std::cout<<std::setprecision(1)<<result<<"\n";
	}
	return 0;
}
