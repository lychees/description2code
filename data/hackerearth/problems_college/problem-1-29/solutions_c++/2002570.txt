#include <bits/stdc++.h>

using namespace std;
#define ll long long int
#define s(x) scanf("%lld",&x)
ll a[100][100];
bool rec(ll a[][100], ll i, ll j) 
{
	if(i == 0) {
		if(j == 0) {
			if(a[0][0] == 1) return true;
			else return false;
		}
		else {
			if(a[i][j] == 1) return rec(a,0,j-1);
			else return false;
		}
	}
	if(j == 0) {
		if(a[i][j] == 1) return rec(a,i-1,0);
		else return false;
	}
	if(a[i][j] == 1){
		if(a[i-1][j] == 1) {
			if(a[i][j-1] == 1) {
				return (rec(a,i-1,j) || rec(a,i,j-1));
			}
			else {
				return rec(a,i-1,j);
			}
		}else{
			if(a[i][j-1]) return rec(a,i,j-1);
			else return false;
		}
	}	
	else return false;	
}

int main()
{
	
	ll t,n;
	s(t);
	while(t--) {
		s(n);
	bool b[100][100];
	for(int i = 0; i < n; i++) {
		for(int j = 0; j < n; j++) {
			cin >> a[i][j];
		}
	}
	bool x = rec(a,n-1,n-1);
	if(x == true) {
		cout << "POSSIBLE\n";
	} else {
		cout << "NOT POSSIBLE\n";
	}
	}
	return 0;
}
