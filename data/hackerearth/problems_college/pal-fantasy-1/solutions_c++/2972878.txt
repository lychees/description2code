#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s,s1;
        int x=0,i,ans=0,len=0,y=0,j;
        cin>>s;
        for(i=0;i<s.length();i++)
        {
            s1=s.substr(i,s.length()-i);
            len=s1.length();
            for(j=0;j<len/2;j++)
                if(s1[j]==s1[len-j-1])x++;
            if(x==len/2){y=s.length()-len;break;}
            x=0;
        }
        cout<<s.length()+y<<"\n";
    }
    return 0;
}
