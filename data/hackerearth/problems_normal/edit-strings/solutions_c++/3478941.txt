#include<iostream>

using namespace std;

int main(){

int l,n;
cin>>l>>n;
char s[l];
cin>>s;
int k;
char p;
int i;
int count;
while(n--){
	count=0;
	cin>>k>>p;
	
	s[k-1]=p;
	for(i=0;i<l;i++){
		
		if(s[i]=='.'&&s[i]==s[i+1])
			count++;
			
		
	}
	cout<<count<<endl;
	
	
}



return 0;
}
