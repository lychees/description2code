#include<bits/stdc++.h>
using namespace std;
int main()
{
	int x[1001];
	for(int q=1;q<1001;q++)
	x[q]=0;
	x[0]=1;
	int i,j,rem=0,sum=0,max=1,r,t,mul,m=0,n;
	cin>>r;
	while(r--)
	{
	cin>>n;
	for(i=1;i<=n;i++)
	{
		while(1)
		{
			mul=x[m]*i;
			mul=mul+rem;
			t=mul%10;
			rem=mul/10;
			x[m]=t;
			if(max<=m&&rem==0)
			{
				if(max<=m)
				max=m;
				m=0;
				break;
		}
		m++;
	 }
	 //cout<<max<<endl;
	}
	for(i=max;i>=0;i--)
	{
	sum=sum+x[i];
}
	cout<<sum<<endl;
	sum=0;
	for(int q=1;q<1001;q++)
	{
	x[q]=0;
	x[0]=1;
}
}
	
}
