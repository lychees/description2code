#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    long long int res,sum;
    scanf("%d",&t);
    while(t--)
    
    {
    	int n;
    	
    	scanf("%d",&n);
		long int a[n+1];
		for(int i=0;i<n;i++)
		{
			scanf("%ld",&a[i]);
		}
		sum=0;
		sort(a,a+n);
		for(int i=1;i<n;i++)
		{
			sum+=a[i];
		}
		sum+=a[0]*(n-1);
		printf("%lld\n",sum);
    }
    return 0;
}
