#include <bits/stdc++.h>
using namespace std;
string rev(string S){
    string R = "";
for(int i=S.length() - 1; i>=0; i--)
    R += S[i];

return R;
}
string getRem(int X){
    string R = "";
    int Y = X % 16;
    if(Y > 9){
        switch(Y){
            case 10: R += "U"; break;
            case 11: R += "V"; break;
            case 12: R += "W"; break;
            case 13: R += "X"; break;
            case 14: R += "Y"; break;
            case 15: R += "Z"; break;
        }
    }
    else{
        stringstream ss;
        ss << Y;
        R += ss.str();
    }
    return R;
}
string getHex(int X){
    string S = "";
    while(X > 15){
        S += getRem(X);
        X /= 16;
    }
    S += getRem(X);
    return rev(S);
}
int main(){
  //  FILE *out = freopen("sampleOutput.txt", "w", stdout);
    int T;
    cin>>T;
    assert(T>=1 && T<=100);
    for(int x=1; x<=T; x++){
        int R,G,B;
        cin>>R>>G>>B;
        assert(R>=0 && R<=255);
        assert(G>=0 && G<=255);
        assert(B>=0 && B<=255);
        string a = getHex(R), b = getHex(G), c = getHex(B);
        cout<<"Case "<<x<<": "<<setw(2)<<setfill('0')<<a<<" "<<setw(2)<<setfill('0')<<b<<" "<<setw(2)<<setfill('0')<<c<<endl;
    }
    return 0;
}
