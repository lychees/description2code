#include <iostream>
using namespace std;

int main()
{
    int t,n,b[1000];
    cin>>t;
    while(t--)
    {
   		cin>>n;
    for(int i=0;i<n;i++)
    {
    	cin>>b[i];
    }
    int small=b[0];
    	int total=0;
    	for(int i=0; i<n; i++)
    	{
    		if(small> b[i])
    		{
    			small=b[i];
    		}
    		total=total+b[i];
    	}
    	cout<<small-1<<" "<<total-n<<"\n";
    	
    }
    
    
    
    return 0;
}
