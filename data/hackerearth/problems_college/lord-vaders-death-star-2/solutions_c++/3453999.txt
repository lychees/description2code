#include <iostream>
#include<bits/stdc++.h>
#define scan(n) scanf("%lld",&n);
#define scan2(n,m) scanf("%d %d",&n,&m);
using namespace std;

int main()
{
    int n;
    scan(n);
    map<long long int,int>map1;
    map<long long int,int>::iterator it;
    int i;
    for(i=0;i<n;i++)
    {
    	long long num;
    	scan(num);
    	map1[num]++;
    }
    int flag=0;
    long long max = INT_MIN;
    for(it=map1.begin();it!=map1.end();++it)
    {
    	if(it->second==1)
    	{
    	    if(it->first>max)
    	    {
    	    	max = it->first;
    	    }
    		flag=1;
    	}
    }
    if(flag==0)
    {
    	cout<<"-1"<<"\n";
    }
    else
    {
    	cout<<max<<"\n";
    }
    return 0;
}
