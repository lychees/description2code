#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    string str;
    int flag=0;
    cin>>str;
    int l=str.length();
    for(int i=0;i<l;i++){
    	if(str[i]==str[l-i-1])
    		continue;
    	else{
    		cout<<"NO";
    		flag=1;
    		break;
    	}
    }
    if(!flag){
    	cout<<"YES";
    }
    
    return 0;
}
