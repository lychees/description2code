for _ in xrange(input()):
    n = input()
    d = []
    for i in xrange(n):
        d.append(list(map(int, raw_input().split())))
    dp = []
    for i in xrange(n):
        dp.append([0, 0, 0])
    for i in xrange(3):
        dp[0][i] = d[0][i]
    for i in xrange(1, n):
        dp[i][0] = min(dp[i-1][1], dp[i-1][2]) + d[i][0]
        dp[i][1] = min(dp[i-1][0], dp[i-1][2]) + d[i][1]
        dp[i][2] = min(dp[i-1][1], dp[i-1][0]) + d[i][2]

    print min(dp[n-1])
