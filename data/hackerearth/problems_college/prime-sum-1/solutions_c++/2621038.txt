#include<bits/stdc++.h>
#include<cmath>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
    	int i,k,z=1,f=2,o=0,g=0,h=1,u=0;
    	
    	cin>>n;
    	if(n==1){cout<<0;continue;}
    	if(n==2){cout<<1;continue;}
    	while(1){
    		o=0;
    		for(i=0;i<6;i++){
    			if(i%2==0){g=z;f+=1;u=0;}
    			else {h=z+1;f+=1;u=1;}
    			if(f==n){o=1;break;}
    		}if(o)break;
    		f+=1;
    		z+=1;
    		if(f==n){u=3;break;}
    		//cout<<z;
    		
    	}
    	//for(i=1;i<=100;i++)cout<<a[i];
    	
    	if(u==1)cout<<h<<endl;
    	else if(u==3)cout<<z<<endl;
    	else cout<<g<<endl;
    }
    return 0;
}
