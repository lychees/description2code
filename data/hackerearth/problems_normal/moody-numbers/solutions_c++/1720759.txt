#include <iostream>
#include <bits/stdc++.h>
#include <stdio.h>
using namespace std;
 
long SOD(long n)
{
	long s=0;
	while(n > 0)
	{
		s += n%10;
		n /= 10;
	}
 
	return s;
 
}
 
int main()
{
    int f=0,k=5;
    long n,t;
	scanf("%ld",&t);
	assert(t <= 1000000 && t >= 1) ;
	while(t--)
	{
		k=5;
		f=0;
		scanf("%ld",&n);
		assert(n <= 1000000000 && n >= 1) ;
		while(k--)
		{
			f = ( n == 4 || n == 1 );
			
			if(f)
			   break;
		    
		    n = SOD( (n*n) );
			
		}
		(f==1) ? printf("YES\n") : printf("NO\n");
	}
	
    return 0;
}
