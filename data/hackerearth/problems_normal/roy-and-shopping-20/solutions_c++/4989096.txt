#include <bits/stdc++.h>

using namespace std;
#define gc getchar
#define pc putchar

const int MAX = 1e6 + 1;
int ans[MAX];

void scanInt(int &x) {
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    // (x<<1) + (x<<3) makes the x ten times i.e. 10*x
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

 void writeInt(int x) {
 	register char buffer[35];
 	register int i=0;
 	do {
       	buffer[i++]=(x%10)+'0';
       	x/=10;
    }while(x);
   	i--;
    while(i>=0) pc(buffer[i--]);
	pc('\n');
}

void primeFactors(int n){
    if(ans[n]) return;
    int N=n;
    int c=0;
    while (n%2 == 0){
        c=1;
        n = n/2;
    }
    if(c){
        ans[N]=2;
        return;
    }
    int t;
    int sqroot = sqrt(n);
    for (int i = 3; i <= sqroot; i+=2){
        t=0;
        while (n%i == 0){
            t=1;
            n = n/i;
        }
        if(t){
            ans[N]=i;
            return;
        }
    }
    t=0;
    if (n > 2){
        t=1;
    }
    if(t){
        ans[N]=n;
        return;
    }
}
int main(){
	memset(ans,0,sizeof(ans));
    int n,r;
    scanf("%d",&n);
    while(n--){
        scanInt(r);
        primeFactors(r);
        writeInt(r-ans[r]);
    }
    return 0;
}
