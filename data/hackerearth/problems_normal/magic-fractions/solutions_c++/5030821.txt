#include<bits/stdc++.h>
using namespace std;

 int prime[501];
 void sieve()
 {
    int i;
    for(i=2;i<501;i++)
    {
        if(prime[i]!=-1)
        {   prime[i] = 1;
            for(int j = 2*i;j<501;j+=i)
                prime[j] = -1;
        }
    }
 }

sieve();
int n;

long long s = 1,temp = 1,i;
if(n==1)
{
    cout<<0;return 0;
}
else if(n==2)
{
    cout<<1;return 0;
}
for(i=3;i<=n;i++)
{
    if(prime[i]==1)
        temp*=2LL;

    s += temp;

}
cout<<s;

return 0;
}
