# include <stdio.h>
#include <iostream>
using namespace std;
void swap (char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}
void permute(char *a, int i, int n)
{
   int j;
   if (i == n)
{
     printf("%s\n", a);
}
   else
   {
        for (j = i; j <= n; j++)
       {
          swap((a+i), (a+j));
          permute(a, i+1, n);
          swap((a+i), (a+j)); //backtrack
       }
   }
} 
int main()
{
int no,S,E;
   string s;
   cin>>s;
   cin>>no;
for (int i=0;i<no;i++)
{  
   cin>>S>>E;
   permute(&s[0], S, E);
   cout<<"\n";
}
   getchar();
   return 0;
}
