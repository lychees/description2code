#include <bits/stdc++.h>
using namespace std;
int fi[1000000];
int prime(long long s)
{
	int f=1;
	for(int i=3;i<=sqrt(s)&&f==1;i+=2)
	{
		if(s%i==0)
		f=0;
	}
	return f;
}
int main()
{
    int t;
    cin>>t;
    int a=0,b=1;
    memset(fi,0,sizeof(int)*100000);
    fi[0]=fi[1]=1;
    while(b<=100000)
    {
    	int c=b;
    	b=a+b;
    	a=c;
    	fi[b]=1;
    }
    while(t--)
    {
    	int n;long long s=0;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		//cout<<i<<" "<<fi[i]<<endl;
    		if(fi[i]!=1)
    		s+=i;
    		
    	}
    	//cout<<s<<endl;
    	
    	if(s%2==0)
    	{
    		if(s==2)
    		cout<<"Tic"<<endl;
    		else
    		cout<<"Tac"<<endl;
    	}
    	else
    	{
    		if(prime(s)==1)
    		cout<<"Tic"<<endl;
    		else
    		cout<<"Toe"<<endl;
    	}
    }
    return 0;
}
