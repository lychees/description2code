#include<iostream>
using namespace std;
int main()
{
	int n,x,y,r,count=0;
	cin>>n;
	int m[1001][1001];
	for(int i=0;i<1001;i++)
	{
		for(int j=0;j<1001;j++)
		{
			m[i][j]=0;
		}
	}
	while(n>0)
	{
		cin>>x;
		cin>>y;
		cin>>r;
		r=r*r;
		for(int i=1;i<=1000;i++)
		{
			for(int j=1;j<=1000;j++)
			{
				if((i-x)*(i-x)+(j-y)*(j-y)<=r)
				{
					m[i][j]++;
				}
			}
		}
		n--;
	}
	for(int i=1;i<=1000;i++)
		{
			for(int j=1;j<=1000;j++)
			{
				if(m[i][j]>1)
				{
		          count++;
				}
			}
		}
		cout<<count;
	return 0;
}
