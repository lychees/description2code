#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    for(int f=0;f<t;f++)
    {
    	int n;
    	cin>>n;
    	pair<int,int> a[2*n];
    	for(int i=0;i<n;i++)
    	{
    		int start,end;
    		cin>>start>>end;
    		a[i].first=start;
    		a[i].second=1;
    		a[i+n].first=end-1;
    		a[i+n].second=2;
    	}
    	sort(a,a+2*n);
    	int b[2*n];
    	b[0]=1;
    	for(int i=1;i<2*n;i++)
    	{
    		if(a[i].second==1 )
    		  b[i]=b[i-1]+1;
    	    else if(a[i].second==2)
    	      b[i]=b[i-1]-1;
    	}
    	int max=b[0];
    	for(int i=1;i<2*n;i++)
    	{
    		if(b[i]>max)
    		 max=b[i];
    	}
    	cout<<max<<endl;
    }
    return 0;
}
