#include <iostream>
using namespace std;

int main (){
	int t; cin >> t;
	while(t--){
		unsigned long long n,mod;
		cin >> n >> mod;

		unsigned long long en = (mod*(mod-1))/2;

		unsigned long long total = en*(n/mod);

		unsigned long long sisa = n%mod;
		total = total + ((sisa*(sisa+1))/2);
		cout << total << endl;
	}

	return 0;
}
