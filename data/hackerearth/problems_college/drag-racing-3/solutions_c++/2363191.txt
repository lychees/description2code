#include <iostream>
#include<cstring>
#include<math.h>
#include<map>
using namespace std;

int main()
{
    
    long long num,win;
    string s1;
    string s2;
    string s3;
    map<string,int> mapping;
    map<string,int> ::iterator iter;
    cin>>num;
    num=pow(2,num)-1;
    while(num--)
    {
        cin>>s1>>s2;
        if(mapping[s1])
            mapping[s1]++;
        else
            mapping[s1]=1;
    }
    win=0;
    for(iter=mapping.begin();iter!=mapping.end();iter++)
    {
        if(iter->second>win)
        {
            win=iter->second;
            s3=iter->first;
        }
    }
    cout<<s3;
    return 0;
    
}
