#include <stdio.h>
#include <iostream>
using namespace std;
int gcd(int a,int b)
{
	if(b != 0)
	return gcd(b, a%b);
	else
	return a;
}
bool isPrime(int n)
{
	if(n == 1 || n == 0)
	return false;
	if(n == 2)
	return true;
	
	int i =2;
	for(i = 2;i*i<=n;i++)
	{
		if(n % i == 0)
		return false;
	}
	return true;
}
bool solve(int n)
{
	
	int i =1;int c =0;
	for(i =1;i<n;i++)
	{
		if(gcd(i,n) == 1)
		c++;
	}
	if(isPrime(c) == true)
	return true;
	else
	return false;
}
int main()
{
	int t;
	cin>>t;
    int n;
    while(t--)
    {
    scanf("%d",&n);
  if(solve(n) == true)
   cout<<"TRUE\n";
    else
    cout<<"FALSE\n";
    }
    return 0;
}
