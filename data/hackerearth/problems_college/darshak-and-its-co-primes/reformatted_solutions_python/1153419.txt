import string

t = int(raw_input())
mylist = []


def cop(pair):

    while pair[1] != 0:
        temp = pair[1]
        pair[1] = pair[0] % pair[1]
        pair[0] = temp
        res = pair[0]
    if res == 1:
        print('Is a Co-Prime')
    else:
        print('Not a Co-Prime')


for i in range(t):

    no = raw_input()
    mylist = (no.split())

    for i in range(len(mylist)):
        mylist[i] = string.atoi(mylist[i])

    res = cop(mylist)
