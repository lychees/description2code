#include <iostream>
#include<vector>
#include <algorithm>
using namespace std;


int main()
{ int i,j,k,n,num;
   cin>>n;
   for(j=0;j<n;j++)
   {
   	cin>>num;vector<int>arr;
   	for(i=0;i<num;i++)
   	{ 
   		cin>>k; arr.push_back(k);
   	}
   	sort(arr.begin(),arr.end());
  // 	vector<vector<int>> res(num+1);
   	vector<int>temp;
   	temp.push_back(0); int t;
   	for(i=0;i<num;i++)
   	{ t=temp.size();
   	auto it=find(temp.begin(),temp.end(),arr[i]);
   	if(it==temp.end())
   	temp.push_back(arr[i]);
   	 for(k=0;k<t;k++)
   	 { it=find(temp.begin(),temp.end(),arr[i]+temp[k]);
   	 	if(it==temp.end())
   	 	{
   	 		temp.push_back(arr[i]+temp[k]);
   	 	}
   	 }
   		
   		
   	}
   	cout<<temp.size()<<endl;
   }
    return 0;
}
