struct cmp
{
    bool operator()(pair < pair<int,int> , int> a,pair < pair<int,int> , int> b)
	{
		if( a.first.first < b.first.first)
	    	return true;
		else if ( a.first.first == b.first.first)
			return a.first.second < b.first.second;
		return false;
	}
};

vector < pair<int,int> >v[100005];
int vis[100005]={0},atime[100005] = {0};
int bfs(int node , int n)
{
	int ptime,i,u,ednum;
	queue< pair<int,int > >qu;
	pair<int,int>p;
	qu.push(mp(1,0));
	while(!qu.empty())
	{
		p = qu.front();
		qu.pop();
		//trace2(p.first , p.second);
		if(vis[p.first] == 1)
			continue;
		
		vis[p.first] = 1;
		node = p.first;
		ptime = p.second;
		//trace1(p.first);
		if(p.first == n)
			return ptime;
		
		//trace1(v[node].size());
		for(i=0;i<v[node].size();i++)
		{
			u = v[node][i].first;
			ednum = v[node][i].second;
			//trace3(node,u,ednum);
			if(atime[ednum] > ptime)
				qu.push(mp(u,ptime+1));
			else if(atime[ednum] == -1)
				qu.push(mp(u,ptime+1));
		}
	}
	return -1;
}
