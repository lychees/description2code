#include <iostream>
#include <string>

using namespace std;


int main(){
    string A,B;
    cin>>A>>B;
    int counter=0;
    for(int i=0;i<A.size();i++){
        if(A[i] != B[i])
            counter++;
    }
    if(counter > 1) cout<<"loses";
    else cout<<"wins";

    return 0;
}
