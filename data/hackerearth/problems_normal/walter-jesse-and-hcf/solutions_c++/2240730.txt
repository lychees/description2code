#include<iostream>
using namespace std;
long long gcd(long long x,long long y){
if(y==0)
    return x;
else
return gcd(y,x%y);
}
int main(){
long long d;
cin>>d;
while(d--){
    long long  x,y,k;
    cin>>x>>y>>k;
    if(x<y){
        long long temp=x;
        x=y;
        y=temp;
    }
int yy;
        yy=gcd(x,y);
        k--;
    //cout<<yy<<k;
if(k>0){
    for(int i=yy-1;i>=1;i--){
        if(yy%i==0){
            k--;
        }
    if(k==0){
    cout<<i<<"\n";
    break;
    }
    }
    if(k!=0)
        cout<<"No crymeth today\n";
}
else if(k==0)
    cout<<yy<<"\n";
else{
    cout<<"No crymeth today\n";
}
}
}
