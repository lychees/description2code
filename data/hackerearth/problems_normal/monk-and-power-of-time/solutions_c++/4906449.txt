#include<algorithm>
int d[100000];
using namespace std;
int main(){
	int n;
	cin>>n;
	int temp;
	queue<int> q;
	int i,j,k=0,l=0;
	for(i=0;i<n;i++){
		cin>>temp;
	//	cout<<"net"<<endl;
		q.push(temp);
	}
	for(i=0;i<n;i++){
		cin>>d[i];
		//cout<<"net"<<endl;
	}
	for(i=0;i<n;i++){
		if(q.front()!=d[i]){
			int v=q.front();
			q.pop();
			q.push(v);
			k++,i--;
		}
		if(q.front()==d[i]){
			l++;
			q.pop();
		}
	}
	cout<<(k+l)<<endl;
return 0;
}
