#include <cstdio>
#define sd(x) scanf("%d",&x)
#define pd(x) printf("%c",x)
#define newline printf("\n")
#define FOR(i,j,k) for(auto i=j;i<k;i++)
#define rep(i,n) FOR(i,0,n)
using namespace std;

int main()
{
    int t;
    sd(t);
    rep(i,t){
    	rep(j,t){
    		if(i==0 || i==t-1)
	    		pd('*');
	    	else{
	    		if(i==(t-j-1)){
	    			pd('*');
	    			break;
	    		}
	    		else
	    			pd(' ');
	    	}
    	}    	
    	newline;
    }    	
    return 0;
}
