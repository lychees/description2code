#include<bits/stdc++.h>
using namespace std;
#define ll long long
bitset<2010> b[2010];

int main(){
	int n;cin>>n;
	for(int i=1; i<=n; i++){
		for(int j=1; j<=n; j++){
			int x;cin>>x;
			b[i][j]=x;
		}
	}
	ll ans=0;
	for(int i=1; i<=n; i++)
	for(int j=i+1; j<=n; j++){
		int cnt = (b[i]&b[j]).count();
		ans+=(cnt*(cnt-1))/2;
	}
	cout<<(ans/2)<<endl;
	return 0;
}
