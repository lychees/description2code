for x in xrange(0, int(raw_input())):
    p = 0
    b = int(raw_input())
    a = map(int, raw_input().split())
    a[0] = a[0] % 2
    a[1] = a[1] % 2
    a[2] = a[2] % 2

    for y in xrange(0, b-1):
        d = map(int, raw_input().split())
        if p == 0:
            d[0] = d[0] % 2
            d[1] = d[1] % 2
            d[2] = d[2] % 2
            if a[0] == d[0] and a[1] == d[1]:
                if a[2] != d[2]:
                    p = 1
            elif a[0] != d[0] and a[1] != d[1]:
                if a[2] != d[2]:
                    p = 1
            else:
                if a[2] == d[2]:
                    p = 1
    if p == 0:
        print "Yes"
    else:
        print "No"
