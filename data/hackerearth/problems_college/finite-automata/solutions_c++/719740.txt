#include <iostream>
using namespace std;

int main() {
	int n,k=0;
	cin>>n;
	int a[n][3];
	for(int i=0;i<n;i++) a[i][0]=i;
	for(int i=0;i<n;i++)
	{
		for(int j=1;j<3;j++)
		{
			a[i][j]=k%n; k++;
		}
	}
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<3;j++)
		{
			cout<<a[i][j]<<" ";
		}cout<<endl;
	}
	return 0;
}
