#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin>>n;
	vector<int> a(n);
	int oc = 0,mino = 99999;
	int sum = 0;
	for(int i = 0;i<n;i++)
	{
		cin>>a[i];
		sum += a[i];
		if( a[i] % 2 != 0)
		{
			oc++;
			if(mino>a[i])
				mino = a[i];
		}
	}
	
	if(oc == 0)
	{
		cout<<":(\n";
		return 0;
	}
	
	if(oc%2==0)
	{
		cout<<sum-mino<<endl;
	}
	else cout<<sum<<endl;
	
	
    //cout << "Hello World!" << endl;
    return 0;
}
