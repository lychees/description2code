def factors(n):
    return set(reduce(list.__add__, ([i, n//i]
               for i in range(1, int(n**0.5) + 1) if n % i == 0)))


noc = int(raw_input())
while noc:
    noc -= 1
    curr = int(raw_input())
    if curr & 1:
        print 0
    else:
        outs = (factors(curr))
        cnt = 0
        for x in outs:
            if x & 1 == 1:
                continue
            else:
                cnt += 1
        print cnt
