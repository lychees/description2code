#include <iostream>
#include <vector>
#include <algorithm>
#include <string>

using namespace std;

struct word{
	string wrd;
	int freq = 0;
};

int main(){
	
	string sentence;
	string temp;
	bool found;
	vector<word> wordList;
	
	getline(cin, sentence);
	for(unsigned int i=0; i<sentence.length(); i++){
		if(sentence[i] != ' '){
			temp = temp + sentence[i];
		} 
		
		if(sentence[i] == ' ' || (i == sentence.length() - 1)){
			transform(temp.begin(), temp.end(), temp.begin(), ::toupper);
			for(vector<word>::iterator it = wordList.begin(); it != wordList.end(); it++){
				found = false;
				if(temp == (*it).wrd){
					found = true;
					(*it).freq++;
					break;
					
				}
			}
			if(!found){
				struct word newWord;
				newWord.wrd = temp;
				newWord.freq++;
				wordList.push_back(newWord);
			}
			temp.clear();
		}
	}
	for(vector<word>::iterator it = wordList.begin(); it != wordList.end(); it++){
		cout<<(*it).wrd<<" "<<(*it).freq;
		cout<<endl;
	}
	
	return 0;
}
