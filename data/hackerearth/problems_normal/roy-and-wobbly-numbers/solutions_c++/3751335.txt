#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	std::ios::sync_with_stdio(false);
	int t,e,o,k,n;
	cin>>t;
	while(t--)
	{
		e=1;o=0;
		cin>>n>>k;
		if(k>81)
			{ cout<<-1<<'\n'; continue;}
		for(int i=1; i<k; )
			{
				o++;
				if(e!=o && e >=1)
				{
					i++;
					if(o==10)
						e++;
					o = o % 10;
					e = e % 10;
				}
			}
		for(int i=0;i<n;i++)
			if (i%2==0) cout<<e;
			else cout<<o;
			cout<<'\n';	
	}
    return 0;
}
