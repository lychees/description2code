#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
    int T;
    scanf("%d", &T);
    while(T--) {
    	long long x, y;
    	scanf("%lld %lld", &x, &y);
    	long long result = x ^ y;
    	int count = 0;
    	while(result != 0) {
    		result = result & (result-1);
    		count++;
    	}
    	if(count%2 == 0)
    		printf("YES\n");
    	else
    		printf("NO\n");
    }
    return 0;
}
