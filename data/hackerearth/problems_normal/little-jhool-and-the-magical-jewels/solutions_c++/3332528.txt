#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char a[102];
		scanf("%s",a);  
		int len = strlen(a);
		int r,u,b,y,minn = 200;
		r = 0,b = 0,u = 0,y = 0;
		for(int i = 0; i < len; i++ )
		{
			if(a[i]=='r')
			{
				r++;
					}
					else if(a[i]=='u')
					{
						u++;
							}
							else if(a[i]=='b')
							{
									b++;
									}
									else if(a[i]=='y')
									{
										y++;
												}			
		}
		minn = min(min(min(r,u),b),y);
		cout<<minn<<endl;
	}
	return 0;
}
