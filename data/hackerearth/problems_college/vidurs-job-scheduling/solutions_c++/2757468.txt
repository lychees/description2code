#include <bits/stdc++.h>
using namespace std;


long long int n, A[2222], dp[2222][2222];

long long int solve(int l, int r) {
	if( l > r ) return 0;
	long long int &ret = dp[l][r];
	if( ret != 0 ) return ret;
	ret = 0;
	ret += A[r] - A[l-1];
	ret += max(solve(l+1,r), solve(l,r-1));
	return ret;
}

int main() {
	cin >> n;
	for(int i=1; i<=n; i++) {
		cin >> A[i];
		A[i] += A[i-1];
	}
	cout << solve(1, n) << "\n";
	return 0;
}
