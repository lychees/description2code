using namespace std;

int main()
{
	int T;
	scanf("%d",&T);
	for(int i=1; i<=T; i++){
		int x;
		scanf("%d",&x);
		int x0 = 7*x;
		int y0 = -4*x;
		//printf("x0=%d y0=%d\n",x0,y0);
		int n = int(floor(float(y0)/3));
		//printf("n=%d\n",n);
		if((x0 + 5*n >= 0 && y0 - 3*n > 0) || (x0 + 5*n > 0 && y0 - 3*n >= 0)){
			int five = 3*(x0 + 5*n);
			int three = 5*(y0 - 3*n);
			for(int i=1; i<=five; i++){
				printf("%c",'5');
			}
			for(int i=1; i<=three; i++){
				printf("%c",'3');
			}
			printf("\n");
		}
		else
			printf("-1\n");
	}
	return 0;
}
