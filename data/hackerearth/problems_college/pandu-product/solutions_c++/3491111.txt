#include <iostream>
using namespace std ;
#include <algorithm>
int main()
{
int n ;
cin >>n ;
long long int i ;
long long int a[100000] ;
long long p,l=0,m=0;
long long int b[100000] ;
for ( i = 0; i< n ; i++)
{
cin >> p ;
if ( p >= 0)
{
a[l] = p ;
l+=1;
}
else{
b[m] = p ;
m+=1;
}
}
sort ( a,a+l) ;
sort (b , b+m ) ;
long long y = a[l-1]*a[l-2] ;
long long z = b[0]*b[1] ;
if ( y >= z)
{
cout << y << endl;
}
else{
cout << z <<endl ;
}
return 0 ;
}
