#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>

using namespace std;

#define MAXN 111111
#define sd(x) scanf("%d",&x)
#define F first
#define S second
#define MOD 1000000007LL

int a[MAXN];
int plt[MAXN];
int n, m;
bool isc[MAXN];

void Solve(){
    long long int ans = 1, i, j;
    sd(n); sd(m);
    for(i = 0; i < n; i++){
        sd(a[i]);
    }
    sort(a, a + n);
    for(i = n - 1; i >= 0; i--){
    	j = (m - a[i] + 1) - (plt[m] - plt[a[i] - 1]) - (n - i - 1);
    	if(j < 0){
    		j = 0;
    	}
    	ans *= j;
    	ans %= MOD;
    }
    cout<<ans<<endl;
}

inline void Pre(){
	memset(isc, false, sizeof(isc));
	int i, j;
	for(i = 4; i < MAXN; i += 2){
		isc[i] = true;
	}
	for(i = 3; i * i < MAXN; i += 2){
		for(j = i * i; j < MAXN; j += i + i){
			isc[j] = true;
		}
	}
	plt[0] = 0;
	isc[1] = true;
	for(i = 1; i < MAXN; i++){
		plt[i] = plt[i - 1];
		if(isc[i] == false){
			plt[i]++;
		}
	}
	return;
}

int main(){
    Pre();
    int t;
    sd(t);
    while(t--){
        Solve();
    }
    return 0;
}
