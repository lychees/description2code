#include <bits/stdc++.h>
using namespace std;
int ans[1000005];
int fac[1000005];
int a[1000005];
int main()
{
    for(int i=2;i<=1000;i++)
        if(!a[i])
            for(int j=i*i;j<=1000000;j+=i)
                a[j]=i;
    for(int i=2;i<=1000000;i++)
    	if(!a[i])
    		fac[i]=1;
    	else
    		fac[i]=fac[i/a[i]]+fac[a[i]];
    for(int i=2;i<=1000000;i++)
    	ans[i]=ans[i-1]+fac[i];
    int n,t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        printf("%d\n",ans[n]);
    }
    return 0;
}
