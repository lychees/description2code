#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
    int t;
    cin>>t;
    getchar();
    	while(t--)
	{
		char str[101],ans;
		gets(str);
		int i=0,j=strlen(str)-1;
		while(i<j)
		{
			
			if(!isalpha(str[i]) && str[i]!='*')
			{
				i++;
				continue;
			}
			if(!isalpha(str[j]) && str[j]!='*')
			{
				j--;
				continue;
			}
			
			if(str[i]=='*')
			{
				ans=str[j];
				break;
			}
			if(str[j]=='*')
			{
				ans=str[i];
				break;
			}
			i++;j--;
		}
		printf("%c\n",ans);
	}
    return 0;
}
