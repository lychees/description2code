#include <iostream>
using namespace std;

int main()
{
    int p,q,arr[10000]={0},val;
    int max=0,min=10000;
    cin >> p;
    while(p--)
    {
    	cin >> val;
    	if(max<val)
    		max=val;
    	if(min>val)
    		min=val;
    	arr[val]++;
    }
    cin >> q;
    while(q--)
    {
    	cin >> val;
    	if(max<val)
    		max=val;
    	if(min>val)
    		min=val;
    	arr[val]--;
    }
    for(int i=min;i<=max;i++)
    {
    	if(arr[i]!=0)
    		cout << i << " ";
    }
    return 0;
}
