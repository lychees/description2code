from math import factorial as fact

t = input()

for i in range(t):
    m, n = (int(i) for i in raw_input().split())
    if m+1-n >= 0:
        mfact = fact(m)
        off = mfact * (m+1)
        bottom = fact((m+1-n))
        pnrpart = (off/bottom)
        ans = (mfact * pnrpart) % 1000000007
        print ans
    else:
        print -1
