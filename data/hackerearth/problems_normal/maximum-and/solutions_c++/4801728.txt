#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--)
   {
   	long long int a,b;
   	cin>>a>>b;
   	long long int temp;
   	if(b%2==0)
   		temp=b-2;
   	else
   		temp=b-1;
   	if(temp<a)
   		cout<<(a&b)<<endl;
   	else
   		cout<<(temp)<<endl;
   }
    return 0;
}
