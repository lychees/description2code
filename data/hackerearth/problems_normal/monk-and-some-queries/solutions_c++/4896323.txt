#include <bits/stdc++.h>
using namespace std;

multiset<int> s;

int main()
{
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	int q;
	cin >> q;
	while (q--) {
		int t, x;
		cin >> t;
		if (t == 1) {
			cin >> x;
			s.insert(x);
		} else if (t == 2) {
			cin >> x;
			auto it = s.find(x);
			if (it == s.end()) cout << "-1\n";
			else s.erase(it);
		} else if (t == 3) {
			if (s.empty()) cout << "-1\n";
			else cout << *s.rbegin() << '\n';
		} else {
			if (s.empty()) cout << "-1\n";
			else cout << *s.begin() << '\n';
		}
	}
    return 0;
}
