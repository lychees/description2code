#include <iostream>
using namespace std;

int main()
{
    long long n,len,i,j,m;
    bool notPrime[2000000]={false};
    long long count=0;
    
    cin>>n;
    m=n-1;
    
    len=2*n-2;
    
    for(i=2;i<=len;i++)
    {
    	if(!notPrime[i])
    	{
    		for(j=i*2;j<=len;j+=i)
    		{
    			notPrime[j]=true;
    		}
    		if(i<=m)
    		{
    			count=count+(i+1);
    			//cout<<i<<"\t"<<i+1<<"\n";
    		}	
    		else
    		{
    			count=count+m+1-(i-m);
    			//cout<<i<<"\t"<<(m+1-(i-m))<<"\n";
    		}	
    			
    	}
    }
    cout<<(count%1000000007);
    
    return 0;
}
