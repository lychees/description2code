#include <bits/stdc++.h>
#define _ ios_base::sync_with_stdio(false);cin.tie(0);
using namespace std;
#define pb push_back
#define pob pop_back
#define pf push_front
#define pof pop_front
#define mp make_pair
#define all(a) a.begin(),a.end()
#define bitcnt(x) __builtin_popcountll(x)
#define MOD 1000000000
#define MAXN 500005
typedef unsigned long long int uint64;
typedef long long int int64;
int n,s,d;
int64 k;
vector<int>v[100005];
bool root[100005];
int BIT[100005];
int64 ans=0;
void upd(int x,int inr){
	for(int i=x;i<=100002;i+=i&(-i))
	BIT[i]+=inr;
}
int64 query(int64 val){
	int64 ret=0;
	val=min(val,(int64)100002);
	while(val){
		ret+=BIT[val];
		val-=(val)&(-val);
	}
	return ret;
}
int cnt[100005];
void dfs(int nod){
	stack<int>s;
	s.push(nod);
	while(!s.empty()){
		nod=s.top();
		if(!cnt[nod]){
			ans+=query(k/nod);
			upd(nod,1);
		}
		if(cnt[nod]==v[nod].size()){
			upd(nod,-1);
			s.pop();
			continue;
		}
		int nxtnode=v[nod][cnt[nod]];
		s.push(nxtnode);
		cnt[nod]++;
	}
}
int main(){
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	cin>>n>>k;
	for(int i=1;i<n;i++){
		cin>>s>>d;
		v[s].pb(d);
		root[d]=true;
	}
	for(int i=1;i<=n;i++){
		if(root[i])
		continue;
		dfs(i);
		break;
	}
	cout<<ans;
	fclose(stdout);
	return 0;
}
