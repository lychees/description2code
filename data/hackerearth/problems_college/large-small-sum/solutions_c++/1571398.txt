#include<iostream>
#include<algorithm>
using namespace std;
int main(void)
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,i,j,arr[1001],x;
        cin>>n;
        for(x=1;x<=n;x++)
            cin>>arr[x];
        cin>>i>>j;
        sort(arr,arr+n+1);
        cout<<(arr[n-i+1]+arr[j])<<endl;
    }
    return 0;
}
