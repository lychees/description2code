#include<iostream>
#include<math.h>
using namespace std;
int main()
{
    int t,n,ans,j,a,s;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        ans=0;
        s=(int)(sqrt(n));
        for (j=1;j<=s;j++)
        {
            a=(int)(n/j);
            ans+=(a-j+1);
        }
        cout<<ans<<endl;
    }
    return 0;
}
