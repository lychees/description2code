#include <bits/stdc++.h>
using namespace std;

static int prms[10000];

long long hcf(long long a, long long b){
    if(!b)return a;
    return hcf(b, a%b);
}

void produce_prms(){
    prms[1]=2; prms[2]=3; int idx=2;
    for(int i=4; i<=40007; ++i){
        int chk=0;
        for(int j=1; prms[j] <= sqrt(i); ++j){
            if(i%prms[j]==0){chk=1; break;}
        }
        if(!chk)prms[++idx]=i;
    }
}
long long factorize(long long a){
    long long ret=1, ta=a;
    for(int i=1; prms[i]<=sqrt(a); ++i){
        long long tmp=1;
        while(ta%prms[i]==0){ta/=prms[i]; tmp++;}
        ret*=tmp;
    }
    if(ta>1 && ta<a)ret*=2;
    if(ta==a && ta>1)ret++;
    return ret;
}

int main(){
    int t; cin >> t;
    produce_prms();
    while(t--){
        long long n, r;
        cin >> n >> r;
        long long tmp=n-factorize(r-1);
        long long h=hcf(tmp, n);
        tmp/=h; n/=h;
        printf("%lld/%lld\n", tmp, n);
    }
    return 0;
}
