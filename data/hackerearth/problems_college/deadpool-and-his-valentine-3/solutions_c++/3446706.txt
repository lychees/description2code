#include<bits/stdc++.h>

using namespace std;

int main ()
{
	int k, m, x, y;
	scanf("%d%d", &k, &m);
	vector< int > hash(m + 1,0);
	for(int i = 0; i < k; i++)
	{
		scanf("%d", &x);
		for(int j = 0; j < x; j++)
		{
			scanf("%d", &y);
			hash[y] = 1;
		}
	}
	int i;
	for(i = 1; i <= m; i++)
	{
		if(hash[i] == 0)
		{
			break;
		}
	}
	if(i == m + 1)
	{
	 	cout << "YES";
	}
	else
	{
		cout << "NO";
	}
	return 0;
}
