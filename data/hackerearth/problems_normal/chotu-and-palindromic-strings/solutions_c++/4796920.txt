#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int compare(string s1,string s2)
{
	if(s1.compare(s2)==0)
	 return 1;
	else
	 return 0;
}
int main()
{
	string s1,s2,s3,s4,s5,s6;
	int t;
	cin>>t;
	while(t--)
	{
		cin>>s1;
		cin>>s2;
		s3= s1 + s2;
		s4= s2 + s1;
		s5= s3;
		s6= s4;
		reverse(s3.begin(),s3.end());
		reverse(s4.begin(),s4.end());
		if(compare(s3,s5)||compare(s4,s6))
		 cout<<"\nYES";
		else
		 cout<<"\nNO";		
	}
	return 0;
	
}
