#include<iostream>
#include<set>
using namespace std;
int main(){
    set<int> s;
    int n,cases;
    cin>>cases;
    while(cases--){
        s.clear();
    n =0;
    cin>>n;
    for(int i = 0; i <n ;i++){
        int val;
        cin>>val;
        s.insert(s.begin(),val%10);
    }
    cout<<(n-s.size())<<"\n";
}
return 0;
}
