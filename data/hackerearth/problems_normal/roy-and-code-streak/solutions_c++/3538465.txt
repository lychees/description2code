#include <iostream>
#include <stdlib.h>
#include <set>
#include <vector>
#include <algorithm>
 
using namespace std;



int main (void){

	int T;
	int N;
	int price;
	cin >> T;
	int cont = 0;
	set<int> quest;
	vector<int> vCont;
	for(int i = 0; i < T; i++){
		cin >> N;
		int S;
		int R;
		cont = 0;
		quest.clear();
		vCont.clear();
		for(int j = 0; j < N; j++){
			cin >> S; cin >> R;
			if(R == 1){
				if(quest.find(S) == quest.end()){
					quest.insert(S);
					cont++;
					
				}
			} else {
				cont = 0;
			}			
			vCont.push_back(cont);			
		}
				
		cout << *std::max_element(vCont.begin(), vCont.end())<< endl;
	}


}
