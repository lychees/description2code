#include <iostream>
using namespace std;
long long int find(long long int n, long long int M){
	if(n==1)
		return 2;
	long long int h= find(n/2,M)%M;
	if(n%2==0)
	  return (h*h)%M;
	else
	  return (((2*h)%M)*h)%M;

}
int main()
{
    //2^n-1
    long long int M = 1000000007;
    long long int t;
    cin>>t;
    while(t-- >0){
    	long long int n;
    	cin>>n;
    	cout<<find(n,M)-1<<endl;
    }
    return 0;
}
