#include<iostream>
#include<cstring>
using namespace std;

int main()
{
	int t;
	int ans,i,x;
	for(cin>>t;t;t--){
		char a[1000],b[1000];
		if(cin>>a&&cin>>b)
			x=min(strlen(a),strlen(b));
		else if(cin>>a)
			x=strlen(a);
		else if(cin>>b)
			x=strlen(b);
		ans=0;
		for(i=0;i<x;i++)
			if(a[i]==b[i])
				ans++;
		cout<<ans<<"\n";
	}
    return 0;
}
