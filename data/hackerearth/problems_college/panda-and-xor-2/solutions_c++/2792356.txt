#include <iostream>
using namespace std;

int count_one (int n)
{
 int count=0;
while( n )
        {
        n = n&(n-1);
           count++;
        }
        return count;
}

int main()
{
   int t;
   cin>>t;
   
   while(t--)
   {
   	 int x,y;
   	 cin>>x>>y;
   	 int z = x^y;
   	 int r = count_one(z);
   	 
   	 if(r&1) cout<<"NO\n";
   	 else cout<<"YES\n";
   	 
   	 
   }
    return 0;
}
