for _ in range(input()):
    x = input()
    if x == 1 or x == 2:
        print '-1'
        continue

    n = x/3
    r = x % 3

    while n:
        if not r % 5:
            break
        else:
            n -= 1
            r += 3

    n1 = r/5

    s = ""
    for _ in range(n*3):
        s += '5'
    for _ in range(n1*5):
        s += '3'
    print s
