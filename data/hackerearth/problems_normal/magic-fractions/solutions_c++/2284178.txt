#include <bits/stdc++.h>
using namespace std;
vector<int> prime(600,0);
void sieve()
{
	for(int i=2;i*i<=600;i++)
	{
		if(prime[i]==0)//not visited
		{
			for (int j = i*i; j<=600 ; j+=i)
			{
				prime[j]=1;//not prime	
			}
		}
	}
}
int main(int argc, char const *argv[])
{
	sieve();
	int n;cin>>n;
	int cnt=0;
	long long int sum=0;
	for (int i = 2; i<=n ; ++i)
	{
		if(prime[i]==0)//prime
		{
			cnt++;
		}
		sum=sum+pow(2,cnt-1);
	}
	cout<<sum<<endl;
	return 0;
}
