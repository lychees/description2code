#include <iostream>
#include <vector>
using namespace std;
int rever(int n)
{
	int temp = n;
	int k = 0;
	vector <int> v;
	while(temp)
	{
		v.push_back(temp%10);
		temp/=10;
		k++;
	}
	int prod = 1;
	k--;
	for(int i=1;i<=k;i++)
		prod*=10;
	int num = 0;
	for(int i=0;i<=k;i++)
	{
		num += prod*v[i];
	//	cout << prod*v[i] << endl;
		prod/=10;
	}
	return num;
}

int main()
{
	int test,n1,n2;
	cin >> test;
	while(test--)
	{
		cin >> n1 >> n2;
		cout << rever(rever(n1) + rever(n2)) << endl;
	}
	return 0;
}
