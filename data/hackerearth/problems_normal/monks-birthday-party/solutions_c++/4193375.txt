#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
    	int n;
    	vector<string> v;
    	cin >> n;
    	set<string> s;
    	for(int i = 0; i < n; i++){
    		string ele;
    		cin >> ele;
    		s.insert(ele);
    	}
    	v.assign(s.begin(), s.end());
    	sort(v.begin(), v.end());
    	for(int i = 0; i < v.size(); i++){
    		cout << v[i] << endl;
    	}
    }
    return 0;
}
