#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[1000010];
int main()
{
	int n;
	scanf("%d",&n);
	int i,j;
	memset(a,0,sizeof(a));
	for(i=0;i<n;i++)
	{
		scanf("%d",&j);
		a[j]=1;
	}
	ll int ans=0;
	ll int nn=0;
	for(i=0;i<1000010;i++)
	{
		if(a[i])
		nn++;
	}
	ans=(nn*(nn-1))/2;
	printf("%lld\n",ans);
	return 0;
}
