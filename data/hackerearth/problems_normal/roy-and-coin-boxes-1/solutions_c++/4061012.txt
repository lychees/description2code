#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define ll long long int
#define II pair<int,int>
#define faster std::ios::sync_with_stdio(false);cin.tie(NULL)
const int N=1e6+5;
ll BIT[N];int n,m;
ll dp[N];
void update(int x,int val){
  while(x<=n){
        BIT[x]+=val;
        x+=x&(-x);       
  }      
}
ll query(int x){
    ll ans=0;
    while(x>0){
        ans+=BIT[x];
        x-=x&(-x);    
    }   
    return ans;
}
void updateRange(int l,int r){
        update(l,1);
        update(r+1,-1);
}
int main(){
        scanf("%d%d",&n,&m);
        int m1=m,l,r,q;
        while(m--){
                scanf("%d%d",&l,&r);
                updateRange(l,r);
        }
        for(int i=1;i<=n;i++){
                dp[query(i)]++;
        }
        for(int i=m1-1;i>=1;i--){
                dp[i]+=dp[i+1];
        }
        scanf("%d",&q);
        while(q--){
                scanf("%d",&l);
                printf("%lld\n",dp[l]);
        }
	return 0;
}
