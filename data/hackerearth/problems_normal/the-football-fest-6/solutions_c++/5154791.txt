#include <iostream>
#include <stack>
using namespace std;

int main()
{
    int tcount; cin >> tcount;
    for (int t = 0; t < tcount; ++t) {
    	int ncount; cin >> ncount;
    	int start; cin >> start;
    	stack<int> players;
    	players.push(start);
    	for (int n = 0; n < ncount; ++n) {
    		char pass; cin >> pass;
    		if (pass == 'P') {
    			int next; cin >> next;
    			players.push(next);
    		} else {
    			int cur = players.top();
    			players.pop();
    			int prev = players.top();
    			players.push(cur);
    			players.push(prev);
    		}
    	}
    	cout << "Player " << players.top() << endl;
    }
    return 0;
}
