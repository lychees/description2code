#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
vector<char>v[21];
int n, k, z, i, j;
char c;

int main()
{
  scanf("%d", &z);
  
  while(z--)
  {
    scanf("%d%d", &n, &k);
    
    for(i = 0; i < n; i++)
    {
      for(j = 0; j < k; j++)
      {
	scanf(" %c", &c);
	v[j].push_back(c);
      }
    }
    for(j = 0; j < k; j++)
    {
      sort(v[j].begin(), v[j].end());
      if (v[j].size() % 2 == 1)
      {
	printf("%c", v[j][v[j].size() / 2]);
      }
      else
      {
	printf("%c", v[j][v[j].size() / 2 - 1]);
      }
      v[j].clear();
    }
    printf("\n");
  }
  
return 0;
}
