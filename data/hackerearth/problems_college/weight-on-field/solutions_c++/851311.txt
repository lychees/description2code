#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
const int nmax=55;
long long dp[nmax];
long long wt[nmax];
double res[nmax];
int main() {
    dp[0]=0ll;
    wt[0]=0ll;
    wt[1]=2ll;
    dp[1]=1ll;
    dp[2]=2ll;
    wt[2]=4ll;
    for(int i=3;i<nmax;i++) {
        dp[i]+=dp[i-1];
        dp[i]+=dp[i-2];
        wt[i]+=wt[i-1]+2*dp[i-1];
        wt[i]+=wt[i-2];
    }
    for(int i=1;i<nmax;i++) {
        res[i]=(double)(wt[i])/dp[i];
    }
    int t;
    cin>>t;
    while(t--) {
        int n;
        cin>>n;
        printf("%.6lf\012",res[n]);
    }
    return 0;
}
