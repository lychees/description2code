#include <iostream>
using namespace std;

long dokadum(long a)
{
	long b = 2;
	if(a<b)
		return 0;
	while(a>=b)
		b*=2;
	return b/2;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long m, n;
		int p;
	    cin>>m>>n;
	    cin>>p;
	    cin>>p;
	    long result;
	    if(n>m)
	    	result = m;
	    else
	    	result = n;
	    result = dokadum(result);
	    cout<<result*result<<endl;
	}
    return 0;
}
