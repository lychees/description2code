#include <iostream>
#include<algorithm>
using namespace std;
 
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int m,n;
    	scanf("%d%d",&m,&n);
    	int arr[n];
    	for(int i=0;i<n;i++)
    	scanf("%d",&arr[i]);
    	sort(arr,arr+n);
    	reverse(arr,arr+n);
    	int count=0;
    	for(int i=0;i<n;i++)
    	{
    		if(arr[i]<=m)
    		{
    			if(arr[i]==m)
    			{
    				count++;
    				continue;
    			}
    			if(arr[i]+arr[n-1]<=m)
    			{
    				count++;
    				n-=1;
    			}
    			else
    			count++;
    		}
    	}
    	printf("%d\n",count);
    }
    return 0;
}
