#include <iostream>
#include <cmath>
using namespace std;

int number=1;
int arr[1024][1024];
void fillup(int size,int i,int j)
{
    //cout<<"## Size: "<<size<<" I: "<<i<<" J: "<<j<<endl;
	if(size==1)
	{
		//cout<<"Setting ("<<i<<","<<j<<") to "<<number<<endl;
		arr[i][j]=number;
		number++;
		return;
	}
	fillup(size/2,i,j);
	fillup(size/2,i,size/2+j);
	fillup(size/2,size/2+i,j);
	fillup(size/2,size/2+i,size/2+j);
}
int main()
{
   int n;
   cin>>n;
   int size=pow(2,n);
   fillup(size,0,0);
   for(int i=0;i<size;i++)
   {
   	for(int j=0;j<size;j++)
   		cout<<arr[i][j]<<" ";
   	cout<<endl;
   }
}
