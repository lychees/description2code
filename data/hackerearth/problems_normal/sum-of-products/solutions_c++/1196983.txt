#include <bits/stdc++.h>

using namespace std;

int main() {
	// your code goes here
	
	int t; scanf("%d",&t);
	while(t-->0) {
		long long int n,i,j,res=0,prev=0,mod=(long long int)(1e9+7); scanf("%lld", &n);
		for(i=1;i*i<n;i++) {
			res+=i*(n/i); res%=mod;
		}
		//printf("%lld ",res);
		prev=n;
		for(i=2;i*i<=n;i++) {
			long long int num; num=n/i+1;
			res+=(prev-num+1)*(prev+num)/2%mod*(i-1)%mod; res%=mod;
			prev=num-1;
		}
		for(j=1;j*j<n;j++);
		res+=(prev-j+1)*(prev+j)/2%mod*(i-1)%mod; res%=mod;
		printf("%lld\n",res);
	}
	return 0;
}
