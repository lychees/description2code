#include <iostream>
using namespace std;

int main()
{
	int t,i,j,c,x,a[1000000],n,b,d,v;
    cin>>t;
    while(t--){c=0;v=0;
    	cin>>n;
    	for(i=1;i<=1000000;i++)
    	a[i]=0;
    	for(i=1;i<=n;i++){
    	cin>>x;
    	if(a[x]==0){
    		a[x]++;c++;
    	}
    	}d=1000000;b=1;
    	while(1){
    		for(i=d;i>=1;i--){
    			if(a[i]>0)
    			{cout<<i<<" ";d=i-1;v++;break;}
    		}if(v==c)
    		break;
    		for(j=b;j<=1000000;j++){
    			if(a[j]>0)
    			{cout<<j<<" ";b=j+1;v++;break;}
    		}
    		if(v==c)
    		break;
    	}
    	cout<<endl;
    }
    return 0;
}
