#include <iostream>
#include <stack>
using namespace std;

int main()
{
    
    int M;
    cin >> M;
    int i;
    int count =0;
    for(i=0;i<M;i++)
    {
    	string str;
    	cin >> str;
    	stack<char> s;
    	int j;
    	for(j=0;j<str.size();j++)
    	{
    		if(!s.empty() && s.top()==str[j])
    		{
    			s.pop();
    		}
    		else
    			s.push(str[j]);
    	}
    	if(s.empty())
    	count++;
    }
    cout << count << endl;
    return 0;
}
