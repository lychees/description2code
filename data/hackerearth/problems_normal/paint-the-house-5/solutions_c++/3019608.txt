#include<bits/stdc++.h>
#define int long long int
using namespace std;
const int N=1e5 +5;
const int inf= 1e6;
pair<int,int> a[N];
#undef int
int main(){
#define int long long int
int n, tot;
cin>>n>>tot;
while(n--){
int ind,cost,time;
cin>>ind>>cost>>time;
if(!a[ind].first){
a[ind].first= time;
a[ind].second= cost;
}
else{
if(a[ind].first==time){
a[ind].second=min(a[ind].second,cost);
}
else if(a[ind].first<time){
a[ind].first=time;
a[ind].second=cost;
}
}
}
int c=0;
int mini=0;
int f=0;
for(int i=1; i<N; i++){
if(a[i].first){
int time= a[i].first;
int cost= a[i].second;
if(time>mini){
mini=time;
c= c+cost;
f=1;
}
}
if(f) tot-=mini;
if(tot<=0) break;
}

cout<<c<<endl;
}
