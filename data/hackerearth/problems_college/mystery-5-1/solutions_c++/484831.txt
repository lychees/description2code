#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#define MAX 10000
using namespace std;

typedef long long int LL;
int main()
{
	LL T;
	scanf("%lld",&T);
	while(T--){
		char str[MAX];
		scanf("%s",str);
		LL ans = 1;
		LL cnt = 1;
		sort(str,str+strlen(str));
		for(LL i = 0; i < strlen(str)-1; i++){
			if(str[i]==str[i+1])
			{
				cnt++;
				if(ans < cnt)
					ans = cnt;
			}
			else
			{
				cnt = 1;
			}
			
		}
		printf("%lld\n",ans);
	}
    return 0;
}
