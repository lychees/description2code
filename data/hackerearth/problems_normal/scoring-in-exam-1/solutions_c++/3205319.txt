#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int N , Q;
    cin >> N >> Q;
    int *T = new int[N];
    int *S = new int[N];
    int *K = new int[Q];
    
    for(int i= 0; i < N; i++)
    {
    	cin >> T[i];
    }
    
    for(int i = 0; i < N; i++)
    {
    	cin >> S[i];
    }
    
    for(int i = 0; i < Q; i++)
    {
    	cin >> K[i];
    }
    
    vector<pair<int, int> > Result;
    
    for(int i=0; i<N; i++)
    {
    	Result.push_back(make_pair(S[i],T[i]));
    }
    
    sort(Result.begin(), Result.end());
    
    for(int i=0 ; i < Q ; i++)
    {
    	int result = 0;
    	for(int j = N - 1; j > N - K[i] -1; j--)
    	{
    		result += Result[j].second;
    	}
      cout << result << endl;	
    }
    
    
    return 0;
}
