#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>
#include <vector>

using namespace std;


int main()
{

    int t;
    cin >> t;
    while (t-- ) {
        string s;
        cin >> s;
        vector<int> v;
        int aa = 0, c = 0;
        while ( aa < s.size() ) {
            if ( s[aa] == '|' ) {
                if ( c ) v.push_back(c);
                c = 0;
            }
            else c++;
            aa++;
        }
        if ( c ) v.push_back(c);
        if ( v.size() == 0 || v[0] == s.size() ) {
            printf("Bob\n");
            continue;
        }
        int x = 0;
        for(int i = 0; i < v.size(); i++) x ^= v[i];
        if ( x == 0 ) printf("Bob\n");
        else printf("Alice\n"); }

    return 0; 
}
