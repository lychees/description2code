#include <iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main()
{
         int n,i,index,a;
         long long int result,max,temp;
         result=0;
         cin>>n;
         vector<int>v;
         for(i=0;i<n;i++)
         {
         	cin>>a;
         	v.push_back(a);
         	
         }
         while(1)
         {
         	if(v.size()==2)
         	break;
         	max=0;
         	index=0;
         	for(i=1;i<v.size()-1;i++)
         	{
         		temp=v.at(i-1)*v.at(i+1);
         		if(temp>max)
         		{
         		max=temp;
         		index=i;
         		}
         		
         	}
         	result+=max;
         	v.erase(v.begin()+index);
         
         }
         cout<<result<<endl;
          
    return 0;
}
