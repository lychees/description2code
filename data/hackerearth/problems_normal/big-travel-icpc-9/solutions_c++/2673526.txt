#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const ll MOD = 1000000007;
const int MAX = 200005;
ll A[MAX], B[MAX];

bool comp(const ll &a, const ll &b)
{
	return a > b;
}

void solve()
{
	int N;
	cin>>N;

	for(int i = 0 ; i < N ; i++)
		cin>>A[i]>>B[i];

	sort(A,A+N, comp);
	sort(B,B+N, comp);

	ll ans = 0;
	for(int i = 0 ; i < N ; i++)
	{
		ans = ans+(ll)(N-i-1)*A[i];
		ans = ans-(ll)i*(A[i]);
		if(ans < 0)
			ans = ans + MOD;
		else
			ans = ans%MOD;
		ans = ans+(ll)(N-i-1)*B[i];
		ans = ans-(ll)i*(B[i]);
		if(ans < 0)
			ans = ans + MOD;
		else
			ans = ans%MOD;
	}

	cout<<ans<<endl;
}

int main()
{
	ios_base::sync_with_stdio(false);
	int test = 1;
	while(test--)
	{
		solve();
	}
	return 0;
}
