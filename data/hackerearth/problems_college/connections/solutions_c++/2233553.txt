#include<cstdio>
using namespace std;
int main()
{
	int nl,n;
	scanf("%d %d",&nl,&n);
	int par[100000]={0};
	for(int i=1;i<=n;i++)
	{
		int x,y;
		scanf("%d %d",&x,&y);
		par[y]=x;
	}
	int q;
	scanf("%d",&q);
	while(q--)
	{
		int x,y;
		scanf("%d %d",&x,&y);
		if(par[x]==y || par[y]==x)
			printf("Direct Connection\n");
		else
		{
			while(par[x])
				x=par[x];
			while(par[y])
				y=par[y];
			if(x==y)
				printf("Indirect Connection\n");
			else
				printf("No Connection\n");
		}
	}
	return 0;
}
