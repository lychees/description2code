#include <bits/stdc++.h>
using namespace std;

int checkPrime(int n);
int main()
{
    long int n, i, flag = 0, t;
    
    cin>>t;

    while(t--)
    {
        flag=0;
        
        cin>>n;
    
        for(i=n/2; i>=2; i--)
        {
            
            if (checkPrime(i) == 1)
            {
            
                if (checkPrime(n-i) == 1)
                {
                    
                    if(i!=(n-i))
                    {
                        cout<<i<<" "<<n-i<<endl;
                        
                        flag = 1;
                        break;
                    }
                }
    
            }
        }
        if (flag==0)
            cout<<"-1 -1"<<endl;
    }  
        
    return 0;
}


int checkPrime(int n)
{
    long int i, flag = 1;

    for(i=2; i<=sqrt(n); ++i)
    {
        if(n%i==0)
            flag=0;
    }

    return flag;
}
