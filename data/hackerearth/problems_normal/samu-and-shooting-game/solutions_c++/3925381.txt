#include<iostream>
#include<cstdio>
#include<algorithm>
#include<iomanip>
#include<cstring>
#define MOD 1000000007
using namespace std;
int w,n;
int vis[1002][1002];
double dp[1002][1002];
double p1,p2;
int x,y;
double func(int ix,int p){
	if(p>w)
	p=w;
	if(ix==n){
		return p>=w;
	}
	if(vis[ix][p])
		return dp[ix][p];
	vis[ix][p]=1;
	double ans1=p1*func(ix+1,p+x)+(1.0-p1)*func(ix+1,p);
    double ans2=p2*func(ix+1,p+y)+(1.0-p2)*func(ix+1,p);
	dp[ix][p]=max(ans1,ans2);
	return dp[ix][p];
}

int main(){
	int t;
	cin>>t;
	while(t--){
		cin>>x>>y>>n>>w>>p1>>p2;
		memset(vis,0,sizeof(vis));
		memset(dp,0,sizeof(dp));
		p1=0.01*p1;
		p2=0.01*p2;
		double ans=func(0,0);
		ans*=100;
		cout<<setprecision(6)<<fixed<<ans<<endl;
	}
}
