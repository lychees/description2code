#include<iostream>
#include<string.h>
using namespace std;
#define max 100002
int main()
{
	int t,i,j,k,pos;
	cin>>t;
	char a[max][32];
	int temp[max];
	for(i=1; i<=t; i++)
	    cin>>a[i];
	//for(i=1; i<=t; i++)
	//    cout<<a[i]<<"  "<<strlen(a[i])<<endl;
	
	for(i=1; i<=t; i++)
	{
		if(a[i][0]!='0')
		{
			int count=1;
		    for(j=i+1; j<=t; j++)
		    {
			    if(strlen(a[i])==strlen(a[j]))
			    {
			        pos=0;
			        for(k=0; a[i][k]!='\0'; k++)
			        {
			            if(a[i][k]!=a[j][k])
			            {
			                pos=1; 
			                break;
			            }
			        }
			    
			        if(pos==0)
			        {
			            count++;
			            a[j][0]='0';
			        }
			    }
		    }
		    temp[i]=count;
	    }
	    else
	        temp[i]=0;
	}
    //for(i=1; i<=t; i++)
    //    cout<<temp[i]<<endl;
        
        
    int f=temp[1],v=1;
    for(i=2; i<=t; i++)
    {
    	if(f==temp[i])
    	{
    		v++;
		}
		if(temp[i]>f)
		{
			f=temp[i];
			v=1; 
		}
	}
	cout<<f*v;
	return 0;
}
