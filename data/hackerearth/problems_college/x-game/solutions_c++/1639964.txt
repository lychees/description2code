#include"bits/stdc++.h"
using namespace std;
#define sex 1000000007
long long int power(long long int n, long long int k)
{
  long long int result, pow;
  if(k==0)
    return 1;
  if(n==0)
    return 0;
  if(n==1)
    return 1;
  result=n%sex;
  pow=1;
  while(pow*2<=k)
  {
    pow=pow*2;
    result=(result*result);
    result=result%sex;
  }
  if(k-pow!=0)
  result=result*power(n, k-pow);
  result=result%sex;
  return result;
}
int main() {
  int t;
  cin>>t;
  while(t--)
    {
    int n;
    cin>>n;
    long long ans=0,temp;
    for(int i=0;i<n;i++)
      {
      cin>>temp;
      ans|=temp;
      }
    ans*=power(2,n-1);
    ans%=sex;
    cout<<ans<<"\n";
   }
    return 0;
}
