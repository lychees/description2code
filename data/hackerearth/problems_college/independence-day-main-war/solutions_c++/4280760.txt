#include <bits/stdc++.h>
using namespace std;

int main()
{
    int test;
    cin >> test;
    while(test--)
    {
    	double n;
    	cin >> n;
    	double x=0;
    	double y=0;
    	double val1;
    	double val2;
    	for(int i=0;i<(int)n;++i)
    	{
    			cin >> val1;
    			cin >> val2;
    			x+=val1;
    			y+=val2;
    		
    	}
    	double ans1=x/n;
    	double ans2=y/n;
    	printf("(");
    	printf("%.2lf ",ans1);
    	printf(", ");
    	printf("%.2lf",ans2);
    	printf(")\n");
    	
    	
    }
    return 0;
}
