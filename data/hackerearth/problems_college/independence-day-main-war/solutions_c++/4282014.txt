#include <iostream>
#include<bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    ll t,i,j,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	float sum1=0,sum2=0,x,y;
    	for(i=1;i<=n;i++)
    	{
    		cin>>x>>y;
    		sum1+=x;
    		sum2+=y;
    	}
    	printf("(%.2f , %.2f)",sum1/n,sum2/n);
    	cout<<endl;
    }
    return 0;
}
