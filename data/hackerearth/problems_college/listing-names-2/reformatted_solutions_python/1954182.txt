from operator import itemgetter
from collections import Counter
noc = int(raw_input())
names = {}
while noc:
    noc -= 1
    curr = raw_input()
    if curr in names:
        names[curr] += 1
    else:
        names[curr] = 1

inp = sorted(names.items(), key=itemgetter(0))
for a, b in inp:
    print a, b
