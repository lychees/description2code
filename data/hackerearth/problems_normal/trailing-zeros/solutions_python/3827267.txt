#!/usr/bin/python
num=int(raw_input())
fact=1
if num>=1 and num<=1000:
        for i in range(2,num+1):
                fact=fact*i

        facto=list(str(fact))
        cnt=0

        for i in range(1,len(facto)):
                if facto[-i]=='0':
                        cnt=cnt+1
                else:
                	break
        print cnt
