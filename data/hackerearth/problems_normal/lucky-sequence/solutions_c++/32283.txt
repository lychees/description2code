#include<cstdio>
#include<cmath>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
     while(t--)
    {
        long long k;
        scanf("%lld",&k);
        k+=1;
        int len=(int)(log(k)/log(2));

        k-=(long long)(pow(2.0,len));

        int ans[len];
        for(int i=0;i<len;i++)ans[i]=0;
        int counter=len-1;
        while(k)
        {
            ans[counter]=k%2;
            k/=2;
            counter--;




        }
        for(int i=0;i<len;i++)
        {
            if(ans[i]==0)printf("4");
            else printf("7");
        }
        printf("\n");
        }
}
