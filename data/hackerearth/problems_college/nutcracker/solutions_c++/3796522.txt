#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
    int T;
    vector <int> X;

    for (int i = 1; i <= 10000; i++)
    {
        X.push_back((5*i) / 2);
    }

    cin >> T;

    while (T--)
    {
        int N;

        cin >> N;

        if (find(X.begin(), X.end(), N) == X.end())
            cout << "Yes";
        else
            cout << "No";
        cout << endl;
    }

    
    return 0;
}
