#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n;
    int arr[1000000];
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>n;
    	for(int l=0;l<n;l++)
    	cin>>arr[l];
    	int len=1;
    	int max = arr[n-1];
    	for(int j=n-2;j>=0;j--)
    	{
    		if(arr[j] > max)
    		{
    			len++;
    			max = arr[j];
    		}
    	}
    	cout<<len<<endl;
    }
    return 0;
}
