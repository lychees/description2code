#include <iostream>
#include<algorithm>
using namespace std;

int count (int n)
    {
    	int count=0;
        while( n )
        {
        n = n&(n-1);
           count++;
        }
        return count;
}

int main() {
	int t,n,k;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		int arr[n+1],c[n+1];
		for(int i=0;i<n;i++)
		{
			cin>>arr[i];
			c[i]=count(arr[i]);
		}
		sort(c,c+n);
	int sum=0;
	for(int i=n-1;i>=n-k;i--)
	{
		sum=sum+c[i];
	}
	cout<<sum<<endl;
	}
	return 0;
}
