#include <iostream>
#include <string>
using namespace std;
 int gcd(int x,int y)
{
if(y==0)
return x ;
return gcd(y,x%y);
}

int main()
{
   // cout << "Hello World!" << endl;
   int t,x,y,a,c;
   string s;
   cin>>t;
   while(t--)
   {c=0;
   	cin>>x>>y>>s;
   	while(x!=0&&y!=0)
   	{
   	a=gcd(x,y);
   	//cout<<a<<" ";
   	x=x-a;
   	y=y-a;
   	c++;
   	}
   	if(c%2==0)
   	(s=="Chandu")?cout<<"Daspal":cout<<"Chandu";
   	else
   	cout<<s;
   	cout<<endl;
   }
    return 0;
}
