#include <iostream>
#include<map>
#include<algorithm>

using namespace std;

int main()
{
int x,t,n,r,i,a[105];
map<int,string>mp;
string s;
   cin>>t;
   
   while(t--)
   {
   		cin>>n;
   		for(i=1;i<=n;i++)
   		{
   			cin>>a[i]>>s;
   			mp[a[i]]=s;
   		}
   		sort(a+1,a+n+1);
   		cin>>r;
   		cout<<mp[a[r]]<<'\n';
   }
    return 0;
}
