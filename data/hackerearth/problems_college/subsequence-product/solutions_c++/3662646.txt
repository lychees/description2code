#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,h,pro;
    	cin>>n>>h;
    	int arr[n];
    	pro=1;
    	for(int i=0;i<n;++i)
    	{
    		cin>>arr[i];
    		pro=(pro*arr[i])%h;
    	}
    	if(pro==0) cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
    return 0;
}
