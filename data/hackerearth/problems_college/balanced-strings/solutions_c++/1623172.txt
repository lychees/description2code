#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s;
        cin>>s;
        int a[26]={0};
        int k=s.length();
        for(int i=0;i<k;i++)
            a[s[i]-97]++;
        bool f=1;
        k=a[0];
        for(int i=1;i<26;i++)
        {
            if(a[i]==0 || a[i]!=k)
            {
                f=0;break;
            }
        }
        if(f)
            cout<<"Yes\n";
        else
            cout<<"No\n";
    }
    return 0;
}
