#include<cstdio>
#include<iostream>
#include<string>
//uses C++11 features string.back()

using namespace std;

int main()
{
	int T, ai = 0, bi = 0;
	string a;
	unsigned long long b;
	const int pow[10][4]={ 0,0,0,0, //4 1 2 3
							1,1,1,1,
							6,2,4,8,
							1,3,9,7,
							6,4,6,4,
							5,5,5,5,
							6,6,6,6,
							1,7,9,3,
							6,8,4,2,
							1,9,1,9
						};
	cin>>T;
	while (T--)
	{
		cin>>a;
		ai = a.back() - '0';
		scanf("%llu",&b);
		bi = b % 4;
		
		cout<<pow[ai][bi]<<endl;
	}
	return 0;
}
