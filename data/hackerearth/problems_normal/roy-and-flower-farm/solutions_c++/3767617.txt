#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,amt;cin>>n>>amt;
		int wei[n],pro[n];
		for(int i=0;i<n;i++)
			cin>>pro[i]>>wei[i];
		
		int dp[n+1][amt+1];
		for(int i=0;i<=n;i++)
		{
			for(int j=0;j<=amt;j++)
			{
				if(i==0||j==0)
					dp[i][j]=0;
				else if(wei[i-1]>j)
					dp[i][j]=dp[i-1][j];
				else
					dp[i][j]=max(dp[i-1][j],pro[i-1]+dp[i-1][j-wei[i-1]]);
			}
		}
		int ans=0, res;
		for(int i=0;i<=amt;i++)
		{
			if(dp[n][i]+(amt-i) > ans)
			{
				res=i;
				ans=dp[n][i]+(amt-i);
			}
		}
		cout<<res<<" "<<ans<<endl;
	}
    return 0;
}
