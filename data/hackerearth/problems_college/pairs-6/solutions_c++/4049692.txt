#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int tc;
    cin >> tc;
    while(tc--)
    {
        int n, ct, t, ans=0;
        cin >> n;
        cin >> ct >> t;
		ans = abs(ct-t);
		if(ans > 1)
			cout << "Motu" << endl;
		else
			cout << "Chotu" << endl;
    }
    return 0;
}
