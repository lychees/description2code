#include <stdio.h>
#include<iostream>
#include<cstdlib>
#include<cstring>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	int dp[1000001];
	while(t--){
		int m,n;
		scanf("%d%d",&m,&n);
		string str;
		cin>>str;
		memset(dp,0,sizeof(dp));
		long long res=0;
		dp[0]=1;
		int count=0;
		for(int i=0;i<str.size();i++){
			if(str[i]=='R'){
				count++;
			}
			if(count>=m){
				res+=dp[count-m];
			}
			dp[count]++;
		}
		cout<<res<<" ";
		memset(dp,0,sizeof(dp));
		res=0;
		dp[0]=1;
		count=0;
		for(int i=0;i<str.size();i++){
			if(str[i]=='K'){
				count++;
			}
			if(count>=n){
				res+=dp[count-n];
			}
			dp[count]++;
		}
		cout<<res<<endl;
	}
}
