t = int(raw_input())

for i in range(t):
    n = int(raw_input())
    s = map(int, raw_input().strip().split(' '))

    ans = 999999999999
    for i in range(n):
        temp = s[i]
        cnt = 0
        while temp:
            temp = temp & (temp-1)
            cnt += 1
        ans = min(ans, cnt)
    print ans
