#include <iostream>
using namespace std;

int main()
{
   int t,m,n,d,flag,j,i,k,count;
   cin>>t;
   for(k=0;k<t;k++)
   {
   		flag=0;
   		cin>>m;
   		int *mx=new int[m];
   		int *my=new int[m];
   		cin>>n;
   		int *nx=new int[n];
   	 	int *ny=new int[n];
   		cin>>d;
   		for(i=0;i<m;i++)
   			cin>>mx[i]>>my[i];
   		for(i=0;i<n;i++)
   			cin>>nx[i]>>ny[i];
   		count=0;
   		for(i=0;i<m;i++)
   		{
   			for(j=0;j<n;j++)
   				if(d>=(abs(mx[i]-nx[j])+abs(my[i]-ny[j])))
   				{
   					count++;
   					break;
   				}
   			if(count>m/2)
   			{
   				flag=1;
   				break;
   			}
   		}
   	if(flag==1)
		cout<<"YES\n";
  	else
   		cout<<"NO\n";
   	delete[] mx;
   	delete[] my;
   	delete[] nx;
   	delete[] ny;
   }
    return 0;
}
