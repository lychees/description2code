#include <iostream>
#include <stdio.h>   
#include <math.h>
#include <stdlib.h>
#include <limits.h>
using namespace std;


int BalancedPartition ( int a[] , int n )
{

int sum = 0;
for( int i = 0 ; i < n ; i++)
    sum += a[i];

int *s = new int[sum+1];

s[0] = 1;
for(int i = 1 ; i < sum+1 ; i++)    s[i] = 0;

int diff = INT_MAX , ans;

for(int i = 0 ; i < n ; i++)
{
    for(int j = sum ; j >= a[i] ; j--)
    {
        s[j] = s[j] | s[j-a[i]];
        if( s[j] == 1 )
        {
            if( diff > abs( sum/2 - j) )
            {
                diff = abs( sum/2 - j );
                ans = j;
            }

        }
    }
}
return sum-ans-ans;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
    int n,result, arr[300];
    cin >>n;
    for(int i = 0; i < n; i++)
    {
        cin>>arr[i];
    }
    int min,pos;
    for(int i=0;i<n;++i)
    {
	    int k=0,j,brr[300];
	   for(j=0;j<n;++j)
	   if(i!=j)
	   brr[k++]=arr[j]; 
    result = BalancedPartition(brr,n-1);
    if(i==0)
    {
    min=abs(result);
    pos=i;
    }
    else if(min>abs(result))
    {
    min=abs(result);
    pos=i;
    }
    else if(min==abs(result))
    {
    if(arr[pos]>arr[i])
    pos=i;
    }
    }
    cout <<min<<" "<<arr[pos]<<endl;
    }
    return 0;
}
