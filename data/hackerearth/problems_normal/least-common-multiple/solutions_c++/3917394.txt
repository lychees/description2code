const int inf=1e9;
const int MOD=1e9+7;
const int N=32000000;

vector <int> pri;
char t[N];
bool isprime(ll cur){
    for(auto i:pri){
        if (1LL*i*i>cur) return 1;
        if (cur%i==0) return 0;
    }
}
void prepare(){
    ll n;
    cin>>n;
    ll ans=n;
    ll bar=min(ll(N),n+1);
    REP(i,2,bar) if (!t[i]){
    	pri.push_back(i);
        ll cur=1,need=n/i;
        while (cur<=need) cur*=i;
        ans=min(ans,n/cur*cur);
        for(ll j=1LL*i*i;j<bar;j+=i) t[j]=1;
    }
    ll cur=n/2+1;
    while (cur<n&&!isprime(cur)) cur++;
    //cout<<cur<<'\n';
    cout<<min(ans,cur);
}
