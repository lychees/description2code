#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t; cin>>t;
    int arr[10010]={0};
    while(t--){
    	memset(arr,0,sizeof(arr));
    	int e,ans=0; cin>>e;
    	for(int i=0;i<e;i++){
    		int a,b; cin>>a>>b;
    		if(!arr[a]) arr[a]=1,ans++;
    		if(!arr[b]) arr[b]=1,ans++;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
