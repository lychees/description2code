#include <bits/stdc++.h>
#define fore(i,a,n) for(int i=a;i<n;i++)
#define mp make_pair
#define EPS 1e-12
#define pb push_back
#define rev(i,n) for(int i=n-1;i>=0;i--)
#define endl '\n'
using namespace std;
int const N = 1005;

typedef long long ll;
typedef pair<int,int> ii;
ll const mod = 1000000007;
ll dp[N][N];

int main() {
	dp[0][0] = 1;
	fore(i,1,N) {
		dp[i][0] = 1;
		fore(j,1,i)
			dp[i][j] = (dp[i-1][j-1] + dp[i-1][j]) % mod;
		dp[i][i] = 1;
	}
	int test;
	scanf("%d",&test);
	fore(i,0,test) {
		int n,k;
		scanf("%d%d",&n,&k);
		ll ans = 0;
		fore(i,0,k+1)
			ans += dp[n][i];
		printf("%lld\n",ans % mod);
	}
}
