def tour():
    cities = {}
    n = int(raw_input())
    for i in xrange(n):
        c = raw_input()
        cities[c] = i

    dinput = [[0 for i in xrange(n)] for j in xrange(n)]
    for i in xrange(n):
        for j, val in enumerate(map(int, raw_input().split(" "))):
            dinput[i][j] = val

    numc = int(raw_input())
    sd, i = 0, 0
    for j in xrange(numc):
        c = raw_input()
        sd = sd + dinput[i][cities[c]]
        i = cities[c]
    print sd


tour()
