using namespace std;
ll comb[71][71];
void precomb()
{
	int i,j;
	comb[0][0]=1;
	for(i=1;i<65;++i)
	{
		for(j=0;j<=i;++j)
		{
			if(i==1||j==0)
			comb[i][j]=1;
			else
			comb[i][j]=comb[i-1][j]+comb[i-1][j-1];
		}
	}
}
int main()
{
	precomb();
	int t,n;
	ll ans;
	cin>>t;
	while(t--)
	{
		cin>>n;
		ans=comb[2*n][n];
		ans=ans/(n+1);
		cout<<ans<<endl;
	}
}
