#include <iostream>
#include <string.h>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char a[35];
    	cin>>a;
    
    	int len=strlen(a);
    		long sum=a[len-1]-'0';
    
    	
    	for(int i=len-1;i>=0;i-=2)
    	{
    		
    		switch(a[i-1])
    		{
 
    			case '+':
    				sum+=(a[i-2]-'0');
    				break;
    			case '-':
    				sum-=(a[i-2]-'0');
    				break;
    			case '*':
    				sum*=(a[i-2]-'0');
    				
    				break;
    		
    		}
    	
    	}
    		cout<<sum<<"\n";
    	
    }
    return 0;
}
