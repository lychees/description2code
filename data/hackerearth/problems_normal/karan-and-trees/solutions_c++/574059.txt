#include <bits/stdc++.h>
using namespace std;
#define int long long
void dfs (int a, int b, int & y)
{
    if (a==b) 
    {
        y=a; return; 
    }
    if (a>b) dfs (a/2, b, y);
    else dfs (a,b/2,y); 
}
main()
{
    int a; scanf("%lld", &a);
    for (int g=0; g<a; g++)
    {
        int b,c; scanf("%lld %lld", &b, &c);int z=0; 
        dfs(b,c,z); 
        printf("%lld\n", z);  
    }return 0; 
}
