#include<iostream>
#include<stdio.h>
//#include<conio.h>
#define m 100001
using namespace std;
int gcd(int a,int b)
{
if (b==0)
return a;
else
return gcd(b,a%b);
}
int main()
{
int num1[m],den1[m],num[m],den[m],posof[m];
int ans,u,v,i,j,t,n;
char str[50];
scanf("%d",&t);
while(t--)
{
scanf("%d",&n);
               for(i=0;i<n;i++)
               {
               scanf("%s",str);
               num[i]=den[i]=0;
                           for(j=0;str[j]!='/';j++)
                           num[i]=num[i]*10+str[j]-'0';
                           for(j++; str[j]!='\0'; j++)
                           den[i]=den[i]*10+str[j]-'0';
               }
               for(i=n-1;i>=0;i--)
               {
               num1[i]=num[i];
               den1[i]=den[i];
               posof[i]=i;
               j=i+1;
                              while(((float)num1[i]/den1[i])<((float)num1[j]/den1[j])&&j<n)
                              {
                              num1[i]=num1[i]+num1[j];
                              den1[i]=den1[i]+den1[j];
                              posof[i]=posof[j];
                              j=posof[i]+1;
                              }
               }
               for(i=0;i<n;i++)
               {
               u=num1[i];
               v=den1[i];
               ans=gcd(u,v);
               printf("%d/%d\n",u/ans,v/ans);
               }
printf("\n");
}
//getch();
return 0;
}
