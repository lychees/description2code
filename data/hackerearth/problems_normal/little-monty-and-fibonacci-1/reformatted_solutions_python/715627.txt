f = [0 for i in range(1001)]
s = [0 for i in range(1001)]
f[1] = f[2] = 1
s[1] = 1
s[2] = 4
for i in range(3, 1001):
    f[i] = f[i - 1] + f[i - 2]
    for j in range(1, i + 1):
        if f[i] % f[j] == 0:
            s[i] += 1
    s[i] += s[i - 1]
