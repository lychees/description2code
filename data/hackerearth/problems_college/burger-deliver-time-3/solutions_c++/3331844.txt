#include <iostream>
#include <utility>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int n;
	cin>>n;
	vector<pair<int,int>> vec;
	for(int i=1;i<=n;++i)
	{
		int x,y;
		cin>>x>>y;
		vec.push_back(pair<int,int>(x+y,i));
	}
	sort(vec.begin(),vec.end());
	for(int i=0;i<n;++i)
	{
		cout<<vec[i].second<<" ";
	}
    return 0;
}
