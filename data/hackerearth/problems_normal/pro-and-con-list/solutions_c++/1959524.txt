#include<bits/stdc++.h>
using namespace std;
long long int ans[100001];
 
int main()
{
 
	long long int t,n,i,j,k,a,b,rem;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		rem=0;
		for(i=0;i<n;i++)
		{
			scanf("%lld %lld",&a,&b);
			ans[i]=a+b;
			rem=rem+b;
		}
		
	
		sort(ans,ans+n);

		printf("%lld\n",ans[n-1]+ans[n-2]-rem);
	}	
	return 0;
}
