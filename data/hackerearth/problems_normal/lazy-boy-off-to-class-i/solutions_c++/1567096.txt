#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
    {
    	int64_t ans=0,a[20005];
    	float v=0;
    	int j=0;
    	for(int i=0;i<7;i++)
    	 {
    	 	int n;
    	 	cin>>n;
    	 	
    	 	for(int l=1;l<=2*n;l+=2)
    	 	 {
    	 	 	int64_t z,y;
    	 	 	cin>>z>>y;
    	 	 	a[j]=y-z;
    	 	 	j++;
    	 	 }
    	 	 v+=n;
    	 }
    	// cout<<v<<" ";
    	 v=0.75*v;
    	 
    	 int vx=v;
    	 float temp=vx;
    	 if(temp!=v) 
    	  {
    	  	vx++;
    	  }
    	 sort(a,a+j);
    	 int64_t x=0;
    	 for(int i=0;i<vx;i++)
    	   x+=a[i];
    	// cout<<x<<" "<<vx<<" ";
    	 ans=24*60*60*7;
    	 ans=ans-x;
    	 cout<<ans<<"\n";
    }
}
