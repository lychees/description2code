n, q = [int(i) for i in raw_input().strip().split()]
arr = [int(i) for i in raw_input().split()]
for i in range(q):
    flag = False
    query = [int(i) for i in raw_input().split()]
    if query[0] == 0:
        temp = query[1]-1
        arr[temp] = query[2]
    else:
        temp = query[1]
        for a_i in arr:
            if temp <= a_i:
                print(1+arr.index(a_i))
                flag = True
                break

        if not flag:
            print('-1')
