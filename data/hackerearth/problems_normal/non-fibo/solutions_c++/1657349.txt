#include <iostream>
#include<cstdio>
using namespace std;
long long int f[10000005];
int main()
{
    int t,i;
    long long int ans=0,temp,n;
    f[1]=1;
    f[2]=2;
    for(i=3;i<=1000;i++)
    {
    	f[i]=f[i-1]+f[i-2];
    }
    //printf("%lld",f[1000]);
    scanf("%d",&t);
    while(t--)
    {
    	ans=0;
    	scanf("%lld",&n);
    	for(i=1;i<=1000;i++)
    	{
    		if((f[i]-i)<n && ((f[i+1]-(i+1))>=n))
    		{
    			temp=f[i]-i;
    			temp=n-temp;
    			ans=f[i]+temp;
    			break;
    		}
    	}
    	printf("%lld\n",ans);
    }
    return 0;
}
