#include <iostream>
using namespace std;

int main()
{
	long long int t,x,k,c=1,j,d,z,i;
    scanf("%lld",&t);
    while(t--)
    {
    	j=0;i=1;
    	scanf("%lld%lld",&x,&k);
    	z=x;
    	while((z&1))
    	{
    		z=z>>1;
    		i=i*2;
    	}
    	
    	d=1;
    	while(j!=k)
    	{
    		if(x+d==(x^d))
    		j++;
    		d++;
    	}
    	printf("Case #%lld: %lld\n",c,d-1);
    	c++;
    }
    return 0;
}
