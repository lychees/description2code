#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,k,t;
    cin>>t;
    while(t--)
    {
        string s;
        cin>>s;
        char c ;
        int len = s.size();
        for(i=len-2;i>=0;i--)
       	{
       		if(s[i]<s[i+1])
        	break;
       	}
        if(i<0)
        {
        	cout<<"no answer\n";
   	  		continue;
  	  	}
        c= s[i];
      	for(j=i+1;j<len;j++)
      	if(s[j]>c)
    	k=j;
      	s[i]=s[k];
      	s[k]=c;
       	j=len-1;
       	k=i+1;
   	  	while(k<j)
   	  	{
  	  		c = s[j];
   	  		s[j]=s[k];
   	  		s[k]=c;
   	  		k++;
   	  		j--;
   	  	}
       	cout<<s<<endl;
    }
}
