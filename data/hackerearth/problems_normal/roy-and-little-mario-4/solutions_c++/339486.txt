#include <iostream>
using namespace std;

int main()
{
	int testcases;
	long long mario[100001];
	mario[1]=1;
	mario[2]=2;
	mario[3]=4;
	for(int i=4;i<100001;i++)
    {
        mario[i]=(mario[i-3]+mario[i-2]+mario[i-1])%1000000007;
    }
    cin>>testcases;
    while(testcases>0)
    {
        int N;
        cin>>N;
        cout<<(mario[N])<<"\n";
        testcases--;
    }
}
