#include <bits/stdc++.h>
using namespace std;
#define ll long long int

int main()
{	
	ll i,j,k,m,n,t,dp[1010][1010],w[1000+10],v[1000+10],wt;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		scanf("%lld",&wt);
		//memset(dp,0,sizeof(dp));
		for(i=0;i<n;i++)
		{
			cin>>w[i];
		}
		for(i=0;i<n;i++)
		{
			cin>>v[i];
		}
		for(i=0;i<=n;i++)
		{
			for(j=0;j<=wt;j++)
			{	
				if(i==0 || j==0)
				dp[i][j]=0;
				else if(j-w[i-1]>=0)
				{
					dp[i][j]=max(dp[i-1][j],(dp[i-1][j-w[i-1]]+v[i-1]));
				}
				else
				dp[i][j]=dp[i-1][j];
			}
		}
		cout<<dp[n][wt]<<endl;
		
	}
   // printf("Hello World!\n");
    return 0;
}
