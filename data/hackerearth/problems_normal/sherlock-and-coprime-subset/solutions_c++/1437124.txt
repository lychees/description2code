typedef unsigned long long int uint64;
typedef long long int int64;

vector<int>v;
int fact[55];
int a[55];
int n;
int dp[51][(1<<15)];

int go(int idx,int mask){
	if(idx==n){
		return 0;
	}
	if(dp[idx][mask]!=-1)
	return dp[idx][mask];
	int ret=0;
	for(int i=0;i<n;i++){
		if(((mask)&(fact[v[i]]))==0){
			ret=max(ret,1+go(idx+1,mask|fact[v[i]]));
		}
	}
	dp[idx][mask]=ret;
	return ret;
}

int main(){
int t,i,j;
for(i=2;i<=50;i++){
	int flag=0;
	for(j=2;j*j<=i;j++){
		if((i%j)==0){
			flag=1;
			break;
		}
	}
	if(flag==0){
		v.pb(i);
	}
}

for(i=1;i<=50;i++){
	int mask=0;
	for(j=0;j<v.size();j++){
		if(v[j]>i)
		break;
		if((i%v[j])==0){
			mask|=(1<<j);
		}
	}
	fact[i]=mask;
//	cout<<i<<" "<<mask<<endl;
}

scanf("%d",&t);

while(t--){
scanf("%d",&n);
v.clear();
int cnt=0;
for(i=0;i<n;i++){
scanf("%d",&a[i]);
if(a[i]!=1)
v.pb(a[i]);
else
cnt++;
}
n=v.size();
memset(dp,-1,sizeof(dp));
cout<<cnt+go(0,0)<<endl;}

	return 0;
}
