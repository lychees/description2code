#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
ll a,b,c,d;
ll dp[1005][1005];

int main()
{
    ll t,i,j;



    for(i=0;i<=1004;i++)
    {
    	dp[0][i]=0;
    	dp[i][0]=0;
    }

    for(i=1;i<=1004;i++)
    {
        for(j=1;j<=1004;j++)
        {
            if(i==1 && j==1)
                dp[i][j]=1;
            else
            dp[i][j]=((dp[i-1][j]+dp[i][j-1])%mod+dp[i-1][j-1])%mod;

        }

    }

       scanf("%lld",&t);
    while(t--)
    {

    	scanf("%lld %lld %lld %lld",&a,&b,&c,&d);

    	printf("%lld\n",dp[c-a+1][d-b+1]);
    }
    return 0;
}
