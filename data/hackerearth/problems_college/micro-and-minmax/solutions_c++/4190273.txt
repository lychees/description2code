#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int temp,n,max=0,min=1000000000;
		scanf("%d",&n);
		for(int i=0;i<n;i++)
		{
			scanf("%d",&temp);
			if(max<temp)max=temp;
			if(min>temp)min=temp;
		}
		temp=max-min;
		if(temp&1)printf("Yes\n");
		else printf("No\n");
	}
	return 0;
}
