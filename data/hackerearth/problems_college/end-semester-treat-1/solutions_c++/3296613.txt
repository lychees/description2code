#include <iostream>
using namespace std;

int main()
{
   long long t;
   cin>>t;
   while(t--)
   {
   		long long n;
   		cin>>n;long long tp[n];
   		for(long long i=0;i<n;i++)
   		cin>>tp[i];
   		long long m;
   		cin>>m;
   		m=2*m;long long r[m][n];
   		for(long long i=0;i<m;i++)
   		{
   			for(long long j=0;j<n;j++)
   			{
   				cin>>r[i][j];
   			}
   		}
   		long long ans=0;
  
   		for(long long i=m/2;i<m;i++)
   		{   
   			long long flag=0;long long zz[n];
   			for(long long j=0;j<n;j++)
   			{
   				if(r[i][j]-r[i-m/2][j]>tp[j])
   				{
   					flag=1;
   					break;
   				}
   				zz[j]=r[i][j];
   			
   				
   			}
   			
   			if(flag!=1)
   			{
   				
   				for(long long k=0;k<n;k++)
   				tp[k]+=zz[k];
   				
   				ans++;
   			}
   		
   		}
   	
   	
   		if(ans==m/2)
   		cout<<"YES"<<endl;
   		else
   		cout<<"NO"<<endl;
   		
   		
   	
   }
    return 0;
}
