#include <iostream>
#include <string>
using namespace std;

int main()
{
	int n;
	cin >> n;
	while(n--)
	{
    	string input;
    	cin >> input;
    	int l=0, count=0;
    	for(int i=0; input[i] != '\0'; i++)
        	l += 1;
    	for(int i=0; i<l-1; i++)
    	{
        	if(input[i] == 'X' && input[i+1] != 'Y')
            	count++;
        	else if(input[i] == 'Y' && input[i+1] != 'X')
            	count++;
    	}
    	cout << count << endl;
	}
}
