Q = int(raw_input())
food = []
results = []
for q in range(Q):
    query = map(int, raw_input().split())
    if query[0] == 1:
        if not food:
            results.append('No Food')
        else:
            results.append(food.pop())
    elif query[0] == 2:
        food.append(query[1])

for result in results:
    print result
