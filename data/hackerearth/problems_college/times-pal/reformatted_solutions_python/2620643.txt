def palin(s):
    if s == s[::-1]:
        return True
    else:
        return False


def solve(a, b):
    cnt = 0
    if palin(b+a):
        cnt += 1
    for i in range(len(a)):
        c = a[0:i+1]+b+a[i+1:len(a)]
        if palin(c):
            cnt += 1
    return cnt


t = int(raw_input())
while t:
    t -= 1
    a = raw_input()
    b = raw_input()
    print solve(a, b)
