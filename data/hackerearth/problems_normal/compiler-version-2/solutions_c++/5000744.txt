#include <bits/stdc++.h>

using namespace std;

int main()
{
    string s, c;

    while(getline(cin , s)){

        bool flag = false;
        c = "";

        for(int i = 0; i < s.size(); i++){

            if(flag){

                c += s[i];
                continue;
            }

            if(i < s.size() - 1 && s[i] == '/' && s[i + 1] == '/')  {

                c += "//";
                flag = true;
                i++;
            }

            else if(i < s.size() - 1 && s[i] == '-' && s[i + 1] == '>'){

                c += '.';
                i++;

            } else c += s[i];
        }

        cout << c << "\n";
    }

    return 0;
}
