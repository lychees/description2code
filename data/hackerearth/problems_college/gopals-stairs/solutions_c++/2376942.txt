#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
int main()
{
	ll i,t,n;
    vector<ll> dp(100001);
    dp[1]=1;
    dp[2]=2;
    dp[3]=4;
    for(i=4;i<=100000;i++)
    {
    	dp[i]=(dp[i-1]+dp[i-2])%mod;
    	dp[i]=(dp[i]+dp[i-3])%mod;
    }
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld",&n);
    	printf("%lld\n",dp[n]);
    }
    return 0;
}
