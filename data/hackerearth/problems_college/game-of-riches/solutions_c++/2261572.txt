#include<bits/stdc++.h>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
    	int n,m;
    	pair<int,int> a[1000];
    	cin>>n>>m;
    	
    	for(int i=0; i<n; i++){
    		cin>>a[i].first;
    		a[i].second=0;
    	}
    	a[0].second=1;
    	sort(a,a+n);
    	reverse(a,a+n);
    	int f=1;
    	int arr[n+1];
    	for(int i=0; i<n; i++){
    		if(f==m+1) f=1;
    		arr[i]=f;
    		f++;
    	}int ans=0;
    	int val=0;
    	for(int i=0; i<n; i++){
    		if(a[i].second==1){
    			ans=arr[i];
    			val=a[i].first;
    		}
    	}int k=0;
    	for(int i=0; i<n; i++){
    		if(arr[i]==ans && a[i].first>val) k++;
    	}
    	cout<<k<<endl;
    }
}
