t = int(raw_input())


def getStats(a):
    numNegatives = 0
    numPositives = 0
    maxNeg = 1
    isZero = False
    for i in range(len(a)):
        if a[i] < 0:
            numNegatives += 1
            maxNeg = a[i]
        elif a[i] > 0:
            numPositives += 1
        elif a[i] == 0:
            isZero = True
    return numNegatives, numPositives, maxNeg, isZero


def getProduct(a):
    productNeg = 1
    productPos = 1
    for i in range(len(a)):
        if a[i] < 0:
            productNeg *= a[i]
        elif a[i] > 0:
            productPos *= a[i]
    return productNeg, productPos
