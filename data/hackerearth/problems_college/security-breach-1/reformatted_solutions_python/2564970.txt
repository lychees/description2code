t = int(raw_input())
while t > 0:
    s = raw_input()
    r = ""
    for i in s:
        if i == "2":
            r += "cde"
        elif i == "6":
            r += "b3"
        elif i == "9":
            r += "c6a"
        elif i == "7":
            r += "acf"
    print(r)
    t = t-1
