#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t=0;
    scanf("%d",&t);
    while(t--)
    {
    	int r=0,k=0,c=0,i=0,j=0,ans=0;
    	scanf("%d%d",&r,&k);
    	bitset<16> m;
    	for(i=1;i<=r;i++)
    	{
    		m=i;
    		for(j=0;j<16;j++)
    		{
    			if(m[j]==1 && m[j+1]==0 && m[j+2]==1)
    			c++;
    			
    		}
    		if(c>=k)
    		ans++;
    		c=0;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
