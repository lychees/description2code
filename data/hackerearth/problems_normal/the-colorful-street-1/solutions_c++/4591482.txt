#include <bits/stdc++.h>
using namespace std;
int p[25],o[25],y[25];
 
int getmincost(int n,int idx){
	if (n==0)
		return 0;
	if(idx==1)
		return p[n] + min(getmincost(n-1,2),getmincost(n-1,3));
	else if(idx==2)
		return o[n] + min(getmincost(n-1,1),getmincost(n-1,3));
	else return y[n] + min(getmincost(n-1,1),getmincost(n-1,2));
 
}
int main(){
 
	int t,n;
	cin >> t;
	while(t--){
		cin >> n;
		for(int i=1;i<=n;i++)
			cin >> p[i] >> o[i] >> y[i];
			int ans = min(getmincost(n,1),min(getmincost(n,2),getmincost(n,3)));
			cout << ans << endl;
	}
	return 0;
 
}
