#include<bits/stdc++.h>
using namespace std;
#define MAX 100000
#define ll long long
vector<ll> v[MAX+3];
ll mx(ll a,ll b)
{
         return(a>b)?a:b;
}
void pre()
{
     v[1].push_back(1);
     for(ll i=2;i<=MAX+1;i++)
     {
             v[i].push_back(i);
             for(ll j=2;j*j<=i;j++)
             {
                     if(i%j==0)
                     {
                     v[i].push_back(j);
                     if(j*j!=i)
                     v[i].push_back(i/j);
                     }
             }
     }
     return;
}
int main()
{
    pre();
   ll n,t,a[MAX+3],dp[MAX+3];
   scanf("%lld",&n);
   for(int i=1;i<=n;i++)
   scanf("%lld",a+i);
   dp[1]=a[1];
   dp[2]=a[2]+a[1];
   dp[3]=a[3]+a[1];
   for(ll i=4;i<=MAX+1;i++)
   {dp[i]=-5555555;
        for(ll j=0;j<v[i].size();j++)
        {
              dp[i]= mx(dp[i],dp[i/v[i][j]]);                   
        }
        dp[i]+=a[i];  
   }
   for(int i=1;i<=n;i++)
   cout<<dp[i]<<" ";
   cout<<endl;
   return 0;
}
