#include<bits/stdc++.h>
using namespace std;
#define ll				long long int
#define MOD				1000000007
#define si(a)			scanf("%d", &a)
#define sl(a)			scanf("%lld", &a)
#define pi(a)			printf("%d", a)
#define pl(a)			printf("%lld", a)
#define pn 				printf("\n")
ll pow_mod(ll a, ll b) {
	ll res = 1;
	while(b) {
		if(b & 1)
			res = (res * a) % MOD;
		a = (a * a) % MOD;
		b >>= 1;
	}
	return res;
}
int main() {
	int t;
	si(t);
	assert(t >= 1 && t <= 3);
	while(t--) {
		int n, k, p;
		si(n); si(k);
		assert(n >= 1 && n <= 16);
		assert(k >= 1 && k <= 40);
		set<string> s[n];
		string str;
		for(int i = 0; i < n; ++i) {
			si(p);
			assert(p >= 1 && p <= 10);
			for(int j = 0; j < p; ++j) {
				cin >> str;
				assert(str.length() >= 1 && str.length() <= 10);
				s[i].insert(str);
			}
		}
		int min_res = INT_MAX;
		for(int i = 1; i < (1 << n); ++i) {
			set<string> tmp;
			for(int j = 0; j < n; ++j) {
				if(i & (1 << j)) {
					tmp.insert(s[j].begin(), s[j].end());
				}
			}
			if(tmp.size() >= k) {
				min_res = min(min_res, __builtin_popcount(i));
			}
		}
		if(min_res == INT_MAX) {
			printf("-1\n");
		} else {
			pl(min_res);
			pn;
		}
	}
	return 0;
}
