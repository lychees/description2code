#include <bits/stdc++.h>

using namespace std;

vector <int> primes;

void pre(){
	primes.push_back(2);
	for(int i = 3; i<3000; i++){
		int len = primes.size();
		bool isPrime = true;
		for(int j = 0; j<len; j++){
			if(i%primes[j]==0){
				isPrime = false;
				break;
			}
		}
		if(isPrime) primes.push_back(i);
	}
}

void solve(){
	int a,b;
	cin>>a>>b;
	int x = 1<<a, y = 1<<b;
	x--;
	y--;
	int count = 0;
	int len = primes.size();
	for(int i = 0; i<len; i++){
		if(x%primes[i]==0 && y%primes[i]==0){
			while(x%primes[i]==0 && y%primes[i]==0){
				x/=primes[i];
				y/=primes[i];
			}
			count++;
		}
	}
	if(x==y && x!=1) count++;
	cout<<count<<endl;
}

int main(){
	pre();
	int t;
	cin>>t;
	while(t--) solve();
	return 0;
}
