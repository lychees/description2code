#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
        int n,k;
        cin>>n>>k;
        
        int a[n];
        for(int i=0; i<n; i++){
            cin>>a[i];
            int count = 0;
            while(a[i]!=0){
                if(a[i]&1){
                    count++;
                }
                a[i] = a[i]/2;
            }
            a[i] = count;
        }
        
        for(int i=0; i<k; i++){
            for(int j=n-1; j>i; j--){
                if(a[j]>a[j-1]){
                    int temp = a[j];
                    a[j] = a[j-1];
                    a[j-1] = temp;
                }
            }
        }
        
        int sum = 0;
        for(int i=0; i<k; i++){
            sum += a[i]; 
        }
        cout<<sum<<endl;
    }
    return 0;
}
