#include <stdio.h>
#include <stdlib.h>

/*
 * 
 */
int main(int argc, char** argv) {
    int t;
    scanf("%d\n", &t);
    while(t--){
        long int x, y, n, c;
        int a, b;
        scanf("%ld", &n);
        scanf("%d %d %ld", &a, &b, &c);
        long int l_count = 0, r_count = 0;
        while(n--){
          scanf("%ld %ld", &x, &y);
          if((a*x + b*y + c) > 0) l_count++;
          else if((a*x + b*y + c) < 0) r_count++;
        }
        printf("%ld\n", (l_count > r_count) ? l_count : r_count);
    }
    return 0;
}
