#include <bits/stdc++.h>

using namespace std;

long long int sum[20000];

long long int solve(long long int x) {
	long long int ful = (x / 15000) % 1000000007;
	x %= 15000;
	long long int res = sum[14999] * ful % 1000000007;
	res += sum[x];
	res %= 1000000007;
	return res;
}

int main() {
	memset(sum, 0, sizeof(sum));
    sum[1] = 0;
    sum[2] = 1;
    for(int i = 3; i < 15005; i++) sum[i] = (sum[i - 2] + sum[i - 1]) % 10000;
    sum[0] = sum[15000];
    //for(int i = 0; i < 10; i++) cout << sum[i] << " ";
    //cout << endl;
    for(int i = 1; i < 15005; i++) sum[i] = (sum[i - 1] + sum[i]) % 1000000007;
    //for(int i = 0; i < 10; i++) cout << sum[i] << " ";
    //cout << endl;
    int t;
    cin >> t;
    while(t--) {
    	long long int l, r, ans;
    	cin >> l >> r;
    	ans = solve(r) - solve(l - 1);
    	ans %= 1000000007;
    	if(ans < 0) ans += 1000000007;
    	cout << ans << endl;
    }
    return 0;
}
