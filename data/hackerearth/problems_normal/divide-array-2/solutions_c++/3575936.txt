#include <iostream>

using namespace std;

int main()
{
    long long a[100001];
    long long n, Q, D, res;

    cin >> n;
    for (int i = 0; i < n; i++) cin >> a[i];

    cin >> Q;
    res = 1;
    for (int j = 0; j < Q; j++)
    {
        cin >> D;
        res *= D;
    }

    for (int i = 0; i < n; i++)
    {
        if (res != 0) a[i] /= res;
        else a[i] = 0;
        cout << a[i] << " ";
    }
    return 0;
}
