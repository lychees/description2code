#include <iostream>
using namespace std;

int main()
{
    long long int j,i,s,n,a;
    cin>>n;
    for( i=0;i<n;i++)
    {   s=0,j=0;
    	cin>>a;
    	int arr[1000];
    	while(a!=0)
    	{
    		arr[j]=a%2;
    		a=a/2;
    		s+=arr[j];
    		j++;
    	}
    	cout<<s<<"\n";
    }
    return 0;
}
