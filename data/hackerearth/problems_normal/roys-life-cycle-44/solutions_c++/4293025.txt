#include<iostream>

using namespace std;

int main(){
    int i, t, n, x = 0, y = 0, tempx = 0, tempy = 0;
    string logs;
    cin>>n;
    while(n--){
        cin>>logs;
        tempx = 0;
        for(i = 0; i < logs.length(); ++i){
            if(logs[i] == 'C'){
                tempx++;
                tempy++;
            }else{
                if(x < tempx){
                    x = tempx;
                }
                if(y < tempy){
                    y = tempy;
                }
                tempx = 0;
                tempy = 0;
            }
        }
        if(x < tempx){
            x = tempx;
        }
        if(y < tempy){
            y = tempy;
        }
        tempx = 0;
    }
    cout<<x<<" "<<y<<endl;
    return 0;
}
