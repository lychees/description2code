#include <iostream>
#include<algorithm>
using namespace std;

int main() {
	int t,n,count,i,j;
	int girls[10005],boys[10005];
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
		cin>>girls[i];
		for(i=0;i<n;i++)
		cin>>boys[i];
		sort(girls,girls+n);
		sort(boys,boys+n);
		count=0;
		for(i=0,j=n-1;i<n;i++,j--)
		{
			if((girls[i]%boys[j]==0)||(boys[j]%girls[i]==0))
			count++;
		}
		cout<<count<<"\n";
	}
	return 0;
}
