#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int t,i,k;
    char a[100];
    cin>>t;
    while(t--)
    {
    	cin>>a;
    	for(i=0;a[i]!=NULL;i++)
    	{
    		k=(int)a[i];
    		if(k<=99)
    		cout<<2;
    		else if(k>=100 && k<=102)
    		cout<<3;
    		else if(k>=103 && k<=105)
    		cout<<4;
    		else if(k>=106 && k<=108)
    		cout<<5;
    		else if(k>=109 && k<=111)
    		cout<<6;
    		else if(k>=112 && k<=115)
    		cout<<7;
    		else if(k>=116 && k<=118)
    		cout<<8;
    		else
    		cout<<9;
    		
    	}
    	cout<<endl;
    }
    return 0;
}
