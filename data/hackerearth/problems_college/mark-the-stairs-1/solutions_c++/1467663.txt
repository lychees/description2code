#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
	int t,n1,n2;
	cin>>t;
	while(t--)
	{
		cin>>n1>>n2;
		if(n1==0 && n2==0)
		{
			cout<<0<<endl;
			continue;
		}

		if(n1==1 && n2==1)
		{
			cout<<1<<endl;
			continue;
		}

		if(n2==n1 || n2==n1-2)
		{
			if(n1%2==0)
					cout<<(n1+n2)<<endl;
			else
					cout<<(n1+n2-1)<<endl;
			continue;
		}

		cout<<"No Number"<<endl;
	}
	return 0;
}
