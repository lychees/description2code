
#include<iostream>
using namespace std;
int main()
{
    int t,i,n,count,m,a,b,c[100000];
    cin>>t;
    while(t--)
    {
        count =0;
        cin>>n>>m;
        for(i=0;i<m;i++)
        {
            cin>>a>>b;
            c[a]++;
            c[b]++;
        }
        for(i=1;i<=n;i++)
        {
            if(c[i]%2!=0)count++;
            c[i] = 0;
        }
      
        if(count == 0 || count == 2 )cout<<"Yes";
        else cout<<"No";
        cout<<"\n";
    }
}
