#include <iostream>
using namespace std;

int gcd(int a,int b){
	if(b==0)return a;
	return gcd(b,a%b);
}

int main()
{
    int t,x,y,cnt,g;
    string s;
    cin>>t;
    while(t--){
    	cin>>x>>y>>s;
    	cnt=0;
    	while(x!=0&&y!=0){
    		g=gcd(x,y);
    		x-=g;y-=g;
    		cnt++;
    	}
    	if(cnt%2==0)
    		if(s=="Daspal")
    			s = "Chandu";
    		else
    			s = "Daspal";
    	cout<<s<<endl;
    	
    }
    return 0;
}
