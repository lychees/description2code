#include <iostream>
using namespace std;

int main()
{
	// cout << "Hello World!" << endl;
	int t;
	cin >> t;
	while (t--)
	{
		string str;
		cin >> str;
		int s = str.length();
		if (s >= 10)
		{
			cout << "YES" << endl;
			continue;
		}
		bool ok = 0;
		for (int i = 1; i<s; i++)
		{
			for (int j = i + 1; j<s; j++)
			{
				for (int k = j + 1; k<s; k++)
				{
					string s1 = str.substr(0, i);
					string s2 = str.substr(i, j - i);
					string s3 = str.substr(j, k - j);
					string s4 = str.substr(k, s - k);
					if (s1 != s2 && s1 != s3 && s1 != s4 && s2 != s3 && s2 != s4 && s3 != s4)
					{
						ok = 1;
					}
				}
			}
		}
		if (ok)
		{
			cout << "YES" << endl;
		}
		else
		{
			cout << "NO" << endl;
		}
	}
	return 0;
}
