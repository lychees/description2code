#include <cstdlib>
#include <iostream>
#include <cstdio>

using namespace std;

int N;
double opt[18][(1<<18)-1];
double Kn[20][20];
int Tmp[20];
int main()
{
	
	scanf("%d",&N);
	for (int i=1;i<=N;++i)
		for (int j=1;j<=N;++j)
			scanf("%lf",&Kn[i][j]);
	int K,Newi,MAX=(1<<N)-1;
	double New;
	opt[0][MAX]=1;
	for (int day=0;day<N-1;++day)
		for (int t=1;t<=MAX;++t)
			if (opt[day][t]>0)
			{
				K=t;
				for (int i=1;i<=N;++i)
				{
					Tmp[i]=K&1;
					K>>=1;
				}
				for (int i=1;i<=N;++i)
					if (Tmp[i])
						for (int j=1;j<=N;++j)
							if (Tmp[j]&&i!=j)
							{
								New=2.0/((N-day)*(N-day-1))*Kn[i][j]*opt[day][t];
								Tmp[j]=0;
								Newi=0;
								for (int q=N;q;--q)
									Newi=(Newi<<1)+Tmp[q];
								opt[day+1][Newi]+=New;
								Tmp[j]=1;
								//cerr<<"Day "<<day+1<<endl<<" "<<t<<" "<<opt[day][t]<<endl;
								//cerr<<j<<" eated by "<<i<<" ="<<New<<endl;
							}
			}
	for (int i=1;i<=N;++i)
	{
		printf("%.5lf",opt[N-1][1<<(i-1)]);
		if (N!=i)	printf(" ");
	}
	//printf("\n");
	return 0;
}
