#include <stdio.h>

int main()
{
long int n,a[1000000],i,j,count=0,lar=-1;
scanf("%ld",&n);
for(i=0;i<n;i++)
{
scanf("%ld",&a[i]);
}
for(i=0;i<n;i++)
{
count=0;
if(a[i]!=0)
{
for(j=i+1;j<n;j++)
{
if(a[j]==a[i])
{
a[j]=0;
count++;
}
}
if(count==0)
{
if(a[i]>lar)
{
lar=a[i];
}
}
}
}
printf("%ld\n",lar);
return 0;
}
