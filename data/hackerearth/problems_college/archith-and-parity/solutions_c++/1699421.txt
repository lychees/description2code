#include <iostream>
using namespace std;
long  countSetBits(long  n)
{
  long count = 0;
  while(n)
  {
    count += n & 1;
    n >>= 1;
  }
  return count;
}
 
/* Program to test function countSetBits */
int main()
{
    long t,n;
    cin >> t;
    
    while(t--)
    {
    	cin >> n;
    n = countSetBits(n);
    
    if(n%2)
    cout  << "odd";
    else 
    cout  << "even";
    cout << endl;
    }

    return 0;
}
