#define gc getchar
#define ii(x) scanf(" %d", &x)
#define ill(x) scanf(" %lld", &x)

#define per(i,a,b) for(i=a;i>=b;i--)

using namespace std;
 
void in(int &x){
    register int c=gc();
    x=0;
    for(;(c<48||c>57);c=gc());
    for(;c>47&&c<58;c=gc()){x=(x<<1)+(x<<3)+c-48;}
}

int n, k, a[100005];

bool solve(int jump){
	int cnt = 0, cur = 0, pre = 0, i;
	rep(i, 0, n){
		while(cur != n && a[cur] - a[pre] <= jump) cur++;
		cnt++;
		if(cur == n) break;
		pre = cur - 1;
	}
	if(cur != n || cnt > k) return false;
	return true;
}

int main()
{
	int t, i;
	in(t); while(t--){
		in(n); in(k);
		rep(i, 0, n) in(a[i]);
		int begin = 1, end = 1e9, shit = 0;
		while(begin < end){
			int mid = begin + end >> 1;
			if(solve(mid)) shit = mid, end = mid;
			else begin = mid + 1;
		}
		printf("%d\n", shit);
	}

	return 0;
}
