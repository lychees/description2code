#include<bits/stdc++.h>
#define ll long long int
#define mod 1000000007
#define MAX 10000006
using namespace std;
bitset<MAX>mark;
vector<int>primes;
ll pf,pfsum;
void sieve()
{
    for(int i=3; i<sqrt(MAX); i+=2)
    {
        if(mark[i]==0) for(int j=i*i; j<MAX; j+=i) mark[j]=1;
    }
    primes.push_back(2);
    for(int i=3; i<MAX; i+=2) if(mark[i]==0) primes.push_back(i);
}
ll divfact(ll n){
    ll ans,sum;
    pf=pfsum=0;
    int i;
    ans=1;
    for(i=0;primes[i]<=n;i++){
    int num=n;
    sum=0;
        pf+=1;
        pf=pf%mod;
        pfsum+=primes[i];
        pfsum%=mod;
    while(num>0){

    sum+=(num/primes[i]);
    num/=primes[i];
      }
    ans = (ans *(sum+1))%mod;
    }

return ans;
}
int main(){
    ll n,t,x;
    sieve();
    scanf("%lld",&t);
    while(t--){
        scanf("%lld",&n);
        ll x=divfact(n);
        printf("%lld %lld %lld\n",pf,pfsum,x);
        }

return 0;
}
