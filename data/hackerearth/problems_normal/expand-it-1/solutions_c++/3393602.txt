#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	char str[2000000],tmp;
	long long int arr[300] = {0},i,j,k,nu;
	int num,q;
	cin >> str;
	long long int l = strlen(str);
	for(i=0;i<l;)
	{
			j = 0;
			tmp = str[i];
			i++;
			nu = 0;
			while(str[i] >= '0' && str[i] <= '9' && i < l)
			{
				num = str[i]-'0';
				nu = (nu*10)+num;
				i++;
				j++;
			}
			arr[tmp-'a'] += nu;
	}
	/*for(i=0;i<26;i++)
	cout << arr[i] << " ";*/
	cin >> q;
	while(q--)
	{
		cin >> k;
		l = 0;
		for(i=0;i<26;i++)
		{
			if(arr[i]+l >= k)
			{
				cout << (char)(i+'a') << "\n";
				l += arr[i];
				break;
			}
			else
				l += arr[i];
		}
		if(l < k)
			cout << "-1\n";
	}
	return 0;
}
