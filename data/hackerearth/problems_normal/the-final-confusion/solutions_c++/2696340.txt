#include <iostream>
using namespace std;

long long int a[32][32]={0};
long long int ways (int full ,int half ){
	if(half==0&&full==0) return 1;
	if(a[half][full]==0){
	
	 if(half==0)
	a[half][full]=  ways(full-1,1);
	else if(full==0) a[half][full]=1;
	else a[half][full]=ways(full-1,half+1)+ways(full,half-1);
	}
	return a[half][full];
	}

int main()
{
  int t,n;
  cin>>t;
  while(t--){
  	
 
  	cin>>n;
  	cout<<ways(n,0)<<endl;
  	
  }
}
