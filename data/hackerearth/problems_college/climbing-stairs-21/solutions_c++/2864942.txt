#include <cstdio>
#include <iostream>
#include <vector>

using namespace std;
typedef long long int int32;

int32 Process(int N)
{
	vector<int32> dp;
	dp.resize(N+1);
	
	for(int i = 1; i <= N; ++i)
	{
		if(i == 1)
			dp[i] = 1;
		else if(i == 2)
			dp[i] = 2;
		else
			dp[i] = dp[i-1] + dp[i-2];
		
	}
	
	return dp[N];
}

int main()
{
	int T,N;
	cin >> T;
	while(T--)
	{
		cin >> N;
		cout << Process(N) << endl;
	}
	return 0;
}
