#include <iostream>
using namespace std;

int main()
{
    int m,n;
    cin>>n>>m;
    int a[n][m];
    int rowSum[n],colSum[m];
    for(int i=0;i<n;i++)
    {rowSum[i] = 0;
     for(int j=0;j<m;j++)
     {
     	cin>>a[i][j];
     	rowSum[i] = rowSum[i] + a[i][j];
     }
    }
    int max = 0;
    for(int i=0;i<m;i++)
    { 	colSum[i]=0;
    	for(int j=0;j<n;j++)
    	{
    		colSum[i] = colSum[i] + a[j][i];
    	}
    }
    
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<m;j++)
    	{
    		int x = rowSum[i]^colSum[j];
    		if(x > max)
    			max = x;
    	}
    }
    cout<<max;
    return 0;
}
