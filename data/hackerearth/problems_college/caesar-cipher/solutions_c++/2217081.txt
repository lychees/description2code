#include<iostream>
#include<stdio.h>
#include<cstring>
using namespace std;
char str[100001];
int main()
{
	int t,n,l,i,k,tmp;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&k);
		scanf("%s",&str);
		l=strlen(str);
		for(i=0;i<l;i++)
		{
			if(str[i]>90)
			{
				tmp=str[i]-'a';
				str[i]=(tmp+k)%26+'a';
			}
			else
			{
				tmp=str[i]-'A';
				str[i]=(tmp+k)%26+'A';
			}
		}
		cout<<str<<endl;
	}
	return 0;
}
