#include <iostream>
using namespace std;
int add(long long int);
int main()
{
	int t,m,i,j,temp2;
	unsigned long long int temp1,sum;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>m;
    	sum=0;
    	for(j=0;j<m;j++)
    	{
    		cin>>temp1>>temp2;
    		sum+=(temp1*temp2);
    		while(sum>=10)
    			sum=add(sum);
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
int add(long long int sum)
{
	unsigned long long int ans;
	while(sum)
	{
		ans+=(sum%10);
		sum/=10;
	}
	return ans;
}
