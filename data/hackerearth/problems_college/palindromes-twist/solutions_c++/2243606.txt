#include <iostream>
using namespace std;

int main()
{
    int test; cin>>test;
    while(test--)
    {
    	string s; cin>>s;
    	int a,b; cin>>a>>b;
    	int ans = 0,c=0;
    	for(int i=0;i<s.size()/2;i++)
    	{
    		if(s[i]=='a')
    		{
    			if(s[s.size()-1-i]=='a')ans=ans;
    			else if(s[s.size()-1-i]=='b')c=1;
    			else ans+=a;
    		}
    		else if(s[i]=='b')
    		{
    			if(s[s.size()-1-i]=='b')ans=ans;
    			else if(s[s.size()-1-i]=='a')c=1;
    			else ans+=b;
    		}
    		else
    		{
    			if(s[s.size()-1-i]=='b')ans+=b;
    			else if(s[s.size()-1-i]=='a')ans+=a;
    			else ans+=min(a,b)*2;
    		}
    	}
    	if(c)cout<<-1<<endl;
    	else cout<<ans<<endl;
    }
}
