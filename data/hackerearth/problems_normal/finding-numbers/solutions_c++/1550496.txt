#include<iostream>
#include<cstdio>
#define ll long long
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
	ll t,z,m,n,i,x,y,sum;
	scanf("%lld",&t);
	while(t--)
	{
		sum=0;
		scanf("%lld%lld%lld",&z,&m,&n);
		if(m<n)
		{
			x=m;
			y=n;
		}
		else
		{
			x=n;
			y=m;
		}
		//cout<<x<<" "<<y<<"\n";
		for(i=1;i<=sqrt(z);i++)
		{
			if((z%i==0)&&((z/i)<=y))
			{
				sum++;
				if(z/i<=x&&z/i!=i)
					sum++;
			}
			
		}
		printf("%lld\n",sum);
	}
	return 0;
}
