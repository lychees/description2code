#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

set<string> st;

int main(){
    int t,n;
    string s,tmp;
    cin>>t;
    while(t--){
        st.clear();
        cin>>n;
        for(int i=0;i<n;i++){
            cin>>s;
            st.insert(s);
        }
        bool ok = true;
        for(auto& it : st){
            s = it;
            //DEBUG(s);
            tmp = "";
            for(int i=0;i<s.size()-1;i++){
                tmp += s[i];
                if(st.find(tmp) != st.end()){
                    ok = false;
                    break;
                }
            }
            if(!ok) break;
        }
        if(ok)
            cout<<"YES\n";
        else
            cout<<"NO\n";
    }
    return 0;
}
