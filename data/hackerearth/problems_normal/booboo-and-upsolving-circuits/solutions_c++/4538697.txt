#include <bits/stdc++.h>
using namespace std;

long long f(long long a[], long long n, long long m, long long t) {
	long long sum = 0, cnt = 0, flag = 0;
	
	for(long long i = 0; i < n; ++i) {
		if(sum + a[i] <= t) {
			sum += a[i];
		} else {
			sum = a[i];
			cnt++;

			if(cnt >= m) {
				flag = 1;
				break;
			}
		}
	}

	if(flag == 1) return 0;
	return 1;
}

#define ll long long
long long UB(long long a[], long long m, long long n, long long low, long long up) {
	if(low <= up) {
		long long mid = low + (up - low)/2;
		long long ans = f(a, n, m, mid);
		long long ans2 = 0;
		if(mid - 1 >= low) ans2 = f(a, n, m, mid - 1);
		
		if(ans == 1 && ans2 == 1) {
			return UB(a, m, n, low, mid-1);
		} else if(ans == 1 && ans2 == 0) {
			return mid;
		} else if(ans == 0 && ans2 == 0) {
			return UB(a, m, n, mid + 1, up);
		}
	}
	return -1;
}

int main() {
	long long n, m;
	cin >> n >> m;

	long long a[n], mino= -1, sum = 0;
	for(long long i = 0; i < n; ++i) {cin >> a[i]; mino = max(mino, a[i]); sum += a[i];}

	cout << UB(a, m, n, mino, sum) << endl;
	return 0;
}
