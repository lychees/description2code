def happy(n):
    past = set()
    while n != 1:
        n = sum(int(i)**2 for i in str(n))
        if n in past:
            return False
        past.add(n)
    return True


def main():
    t = input()
    for i in xrange(t):
        t -= 1
        n = input()
        if happy(n):
            print "Case #"+str(i+1)+":YES"
        else:
            print "Case #"+str(i+1)+":NO"


if __name__ == '__main__':
    main()
