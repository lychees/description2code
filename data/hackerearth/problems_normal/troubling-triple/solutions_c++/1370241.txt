#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;


int main()
{
    int n;
    long long a;
    scanf("%d %lld",&n,&a);
    vector<long long> arr(n);
    for(int i=0;i<n;++i)
    {
    	scanf("%lld",&arr[i]);
    }
    sort(arr.begin(),arr.end());
    //arr.erase( unique( arr.begin(), arr.end() ), arr.end() );
    int ans=0;
    n=arr.size();
    for(int i=0;i<n-2;++i)
    {
    	int j=i+1,k=n-1;
    	while(j<k)
    	{
    		if(arr[i]*arr[j]*arr[k]<=a){
    				ans+=k-j;
    				j++;
    		}
    		else k--;
    	}
    }
    printf("%d",ans);
    return 0;
}
