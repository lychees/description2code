#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main(){
	int tt,n,i,j;
	char temp[100];
	unsigned long long int *arrIp,*arrReq,swap = 0,t;
	scanf("%d",&tt);
	gets(temp);
	while(tt--){
		scanf("%d",&n);
		
		arrIp = new unsigned long long int[n];
		arrReq  = new unsigned long long int[n];
		for(i = 0 ; i < n;i++){
			cin>>arrIp[i];
	
		}
		for(i = 0 ; i < n;i++){
			cin>>arrReq[i];
		}
		swap =0;
		for(i = 0 ; i < n;i++){
		
			if(arrIp[i] == arrReq[i]){
				for(j = i+1; j <n && arrIp[j] == arrReq[j];j++ );
				if(j == n){
					break;
				}
			}else{
				for(j = i+1; arrIp[j]!=arrReq[i];j++);
			//	printf("\n j %d",j);
				for(int k = j;k > i;k--){
					t = arrIp[k];
					arrIp[k] = arrIp[k-1];
					arrIp[k-1] = t;
					swap++;
				
				}
			}
		}
		
		printf("\n%llu",swap);
		
		
		
	}
	return 0;
	
}
