#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,i,p=0;
    	cin>>n;
    	double d[n],f[n],sum=0.0,temp,max=0.0;
    	for(i=0;i<n;i++)
    	cin>>d[i];
    	for(i=0;i<n;i++)
    	cin>>f[i];
    	for(i=0;i<n;i++)
    	{temp=d[i]*f[i];
    		if(max<temp){max=temp;p=i;}
    		sum+=temp;
    	}
    	printf("%d %0.3lf\n",p,max/sum);
    }
    return 0;
}
