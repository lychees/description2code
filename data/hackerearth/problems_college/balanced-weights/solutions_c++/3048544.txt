#include<bits/stdc++.h>
using namespace std;

long int arr[1000001];

long int gcd(long int a, long int b)
{
    while(true)
    {
        if(a>b)
            a %= b;
        else
            b %= a;
        if(a == 0)
            return b;
        if(b == 0)
            return a;
    }
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long int mini = INT_MAX;
        int n;
        cin>>n;
        for(int i=0; i<n; i++)
        {
            scanf("%ld", &arr[i]);
            if(arr[i] < mini)
                mini = arr[i];
        }

        long long int temp;
        for(int i=0; i<n; i++)
        {
            temp = gcd(mini, arr[i]);
            if(temp < mini)
                mini = temp;
        }
        mini *= n;
        cout<<mini<<endl;
    }
    return 0;
}
