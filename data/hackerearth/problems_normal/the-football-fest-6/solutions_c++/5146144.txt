#include <iostream>
using namespace std;

int main()
{
  	int T;
  	cin>>T;
  	while(T--)
  	{
  		int n,id;
  		cin>>n>>id;
  		int a[n];
  		int top=0,pointer=0;
  		a[0]=id;
  		while(n--)
  		{
  			char p;
  			cin>>p;
  			if(p=='P')
  			{
  				int num;
  				cin>>num;
  				pointer=a[top];
  				top++;
  				a[top]=num;
  				
  		}
  		else
  		{
  			int temp=a[top];
  			a[top-1]=temp;
  			a[top]=pointer;
  			pointer=temp;
  			
  		}}
  		cout<<"Player "<<a[top]<<endl;
  	}
    return 0;
}
