#include<bits/stdc++.h>

using namespace std;

int main()
	{
		int N;
		scanf("%d",&N);
		int freq[1000007]={0};
		long long int n;
		int x;
		for(int i=0;i<N;++i)
			{
				scanf("%d",&x);
				++freq[x];
				if(freq[x]==1)
					++n;
			}
		long long int ans=0;
		ans=(n*(n-1))/2;
		cout<<ans<<endl;
		return 0;
	}
