#include <iostream>
#include<string.h>
using namespace std;

int main()
{
char s[100000],r[100000];
int n,i,j,k,l,t,flag;
cin>>t;
while(t--)
{
	cin>>s;
	k=strlen(s);
	j=k-1;
	for(i=0;i<k;i++)
	{
		r[j]=s[i];
		j--;
	}
//	cout<<r;
	flag=0;
	for(i=1;i<=k/2+1;i++)
	{
		
		j=s[i]-s[i-1];
		l=r[i]-r[i-1];
	if(j<0)
	{
		j=(-1)*j;
	}
	if(l<0)
	{
		l=(-1)*l;
	}
//	cout<<j<<l<<endl;
	if(j!=l)
	{
	flag=1;
	break;
	}
	}
	if(flag==0)
	cout<<"Funny\n";
	else
	cout<<"Not Funny\n";
}
return 0;
}
