#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	char a[100005];
    	scanf("%s",a);
    	
    	int i,j,count=0,l=strlen(a);
    	for(i=0,j=0;i<l-1&&j<l;){
    		if(a[i]=='X' && a[j]!='Y'){
    			count++;
    			j++;
    		}
    		else if(a[i]=='Y' && a[j]!='X'){
    			count++;
    			j++;
    		}
    		else {
    			i=j;
    			j++;
    		}
    	}
    	
    	cout<<count-1<<endl;
    }
    return 0;
}
