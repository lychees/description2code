#include<stdio.h>
#include<iostream>
#include<algorithm>
#include <string.h>
#include <math.h>


using namespace std;


 
/* Function to get no of set bits in binary
   representation of passed binary no. */
int countSetBits(int n)
{
    unsigned int count = 0;
    while (n)
    {
      n &= (n-1) ;
      count++;
    }
    return count;
}
 
/* Program to test function countSetBits */
int main()
{
	int t;
	cin>>t;
	while (t--) {
	
   int n,index,max,i,j,k;
    max = 0;cin>>n;
   for(i = 1;i<=n ;i++) {
   	k = countSetBits(i);
   	if(k >= max) {
   		max = k;
   		index = i;
   	}
   	
   }
   cout<<index<<endl;
   
}
    return 0;
}
