#include <iostream>
using namespace std;

int min(int a,int b)
{
	if(a>b)
		return b;
	return a;
}

int max(int a,int b)
{
	if(a<b)
		return b;
	return a;
}

int main()
{
    int values[6]={0,1,2,6,6,3};
    int t;
    cin>>t;
    while(t--)
    {
    	int a, b;
    	cin>>a>>b;
    	int result=0;
    	for(int i=a;i<=min(b,5);i++)
    		result+=values[i];
    	if(b>5)
    		result+=(b-max(5, a-1))*9;
    	cout<<result<<endl;
    }
    return 0;
}
