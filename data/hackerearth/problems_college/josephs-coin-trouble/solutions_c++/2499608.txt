#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
	long n,a[4],i,j,m;
	char s[50],c ;
	cin>>n;
	getchar();
//	fflush(stdin);
    fgets(s,50,stdin);
   // printf("%s",s);
	sscanf(s,"%ld%c%ld%c%ld%c%ld",&a[0],&c,&a[1],&c,&a[2],&c,&a[3]);
//	cout<<a[0]<<" "<<a[1]<<" "<<a[2]<<" "<<a[3];
	m=1;
	for(i=0;s[i];i++)
	{
		if(s[i]==' ')
		m++;
	}
	long b[m][n+1];
	for(i=0;i<=n;i++)
	{
		if(i<a[0])
			b[0][i]=0;
		else if(i%a[0]==0)
		b[0][i]=1;
		else 
		b[0][i]=0;
	}
	for(i=0;i<m;i++)
	{
		b[i][0]=1;
	}
	for(i=1;i<m;i++)
	{
		for(j=1;j<=n;j++)
		{
			if(j<a[i])
			b[i][j]=b[i-1][j];
			else
			b[i][j]=b[i-1][j]+b[i][j-a[i]];
		}
	}
	
	cout<<b[m-1][n]<<endl;
	
	
}
