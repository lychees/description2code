#include <iostream>
#include<string.h>
using namespace std;

void answer(char* a,char* b,int n)
{
	int res=0;
	for(int i=0;a[i];i++)
	{
		if(a[i]!=b[i])
		res++;
	}
	if(res==n||(n>res&&(n-res)%2==0))
		cout<<"Yes\n";
	else
		cout<<"No\n";
}
int main()
{
    int n;
    cin>>n;
    char a[100];
    char b[100];
    int c;
    for(int i=0;i<n;i++)
    {
    	cin>>a>>b>>c;
    	if(strlen(a)!=strlen(b))
    		cout<<"No\n";
    	else
    		answer(a,b,c);
    }
    return 0;
}
