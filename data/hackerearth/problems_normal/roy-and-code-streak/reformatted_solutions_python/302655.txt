#!/usr/bin/env python
# -*- coding: utf-8 -*-

from collections import defaultdict
from math import factorial as f
from fractions import gcd as g

T = int(raw_input())
for t in xrange(T):
    N = int(raw_input())
    d = defaultdict(int)
    ret, cnt = 0, 0
    while N > 0:
        N -= 1
        a, b = [int(i) for i in raw_input().split()]
        if b == 0:
            cnt = 0
            continue
        if d[a]:
            continue
        d[a] = 1
        cnt += 1
        ret = max(ret, cnt)
    # print N, l
    print ret
