#include<bits/stdc++.h>
using namespace std;

int main()
{
	string s1, s2, s3, s4;
	int t;
	scanf("%d", &t);
	while(t--){
	cin >> s1 >> s2;
	s3 = s1;
	int ans = 0;
	for(int i = 0; i <= s1.length(); i++)
	{
		s1.insert(i, s2);
		//cout << s1 << endl;
		s4 = s1;
		reverse(s1.begin(), s1.end());
		if(s4 == s1)
			ans++;
		s1 = s3;
	}
	printf("%d\n", ans);}

	return 0;

}
