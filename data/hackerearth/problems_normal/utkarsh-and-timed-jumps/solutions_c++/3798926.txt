#include <iostream>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	
    int t;
    cin>>t;
    
    while(t--) {
    	long long int x;
    	cin>>x;
    	long long l = 0, h = 1500000000ll;
    	
    	while(l+1ll < h) {
    		long long mid = l + (h-l)/2ll;
    		if(mid*mid + mid < 2ll*x) 	l = mid;
    		else						h = mid;
    	}
    	
    	cout<<h<<"\n";
    	
    }
    
    return 0;
}
