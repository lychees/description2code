#include <iostream>
using namespace std;
#include<vector>
using std::vector;
#include<math.h>

int main()
{
    int n,m;
    cin>>n>>m;
    int max = -100;
    vector<vector<int> > mat(0);
    vector<int> chk(0);
    vector<int> tem(0);
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<m;j++)
    	{
    		int h;
    		cin>>h;
    		tem.push_back(h);
    	}
    	mat.push_back(tem);
    	tem.clear();
    }
    const int nn = n;
    const int mm = m;
    int arr[nn][mm];
    int sumr[nn];
    int sumc[mm];
    for(int i=0;i<nn;i++)
    {
    	int s=0;
    	for(int j=0;j<mm;j++)
    	s+= mat[i][j];
    	sumr[i]=s;
    }
    for(int i=0;i<mm;i++)
    {
    	int s=0;
    	for(int j=0;j<nn;j++)
    	s+= mat[j][i];
    	sumc[i]=s;
    }
    vector<int> bina(0);
    vector<int> binb(0);
    vector<int>::iterator it;
  
	for(int a=0;a<nn;a++)
	{
		
		for(int b=0;b<mm;b++)
		{
			int q=sumr[a]^sumc[b];
			if(q>max)
			max = q;
		}
	}
    cout<<max;
    return 0;
}
