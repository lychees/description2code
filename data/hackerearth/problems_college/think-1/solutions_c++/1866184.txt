#include<iostream>
#include<cmath>
using namespace std;
long long reverse(long long n){
	long long r=0;
	while(n){
		r=r*10+(n%10);
		n=n/10;
	}
	return r;
}
int main(){
	int t;
	cin>>t;
	while(t--){
		long long n,r;
		cin>>n;
		r=reverse(n);
		//cout<<"r="<<r<<" n*n="<<n*n<<" reverse="<<reverse(r*r)<<endl;
		if(n*n==reverse(r*r) && n!=r){
			cout<<r<<endl;
		}
		else
			cout<<"No"<<endl;
	}
}
