#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str1,str2;
		cin>>str1>>str2;
	    int count1[26]={0};
	    int i;
	    int count2[26]={0};
	    for(i=0;i<str1.length();i++)
	    {
	    	count1[str1[i]-'a']++;
		}
		 for(i=0;i<str2.length();i++)
	    {
	    	count2[str2[i]-'a']++;
		}
		int flag =0;
		for(i=0;i<26;i++)
		{
			if(count1[i]>0 && count2[i]>0)
			{
				flag =1;
			}
		}
		if(flag ==1)
		{
			cout<<"YES"<<"\n";
		}
		else
		{
			cout<<"NO"<<"\n";
		}
	}
}
