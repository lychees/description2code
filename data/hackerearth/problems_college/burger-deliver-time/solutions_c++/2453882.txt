#include <iostream>
using namespace std;

int main()
{
    int T,ti,di;
    cin>>T;
    int arr[T],order[T];
    for(int i=0;i<T;i++) order[i]=i+1;
    for(int t=0;t<T;t++)
    {
        cin>>ti>>di;
    	arr[t] = ti+di;
    }
		  int i, j, tmp;
      for (i = 1; i < T; i++) {
            j = i;
            while (j > 0 && arr[j - 1] > arr[j]) {
                  tmp = arr[j];
                  arr[j] = arr[j - 1];
                  arr[j - 1] = tmp;
                  tmp = order[j];
				  order[j] = order[j-1];
				  order[j-1] = tmp;
                  j--;
            }
      }
	for(int i=0;i<T;i++)
		cout<<order[i]<<" ";
    return 0;
}
