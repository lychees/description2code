#include <iostream>
using namespace std;
#define ll long long
#include <vector>

int main()
{	ll n,t,s=0;
    scanf("%lld",&n);
    vector<ll> z;
    while(n--)
    {	scanf("%lld",&t);
    	s+=t;
    	z.push_back(s);
    }
    scanf("%lld",&n);
    while(n--)
    {	ll a,b;
    	scanf("%lld%lld",&a,&b);
    	if(a==1)
    		printf("%lld\n",z[b-1]);
    	else
    		printf("%lld\n",z[b-1]-z[a-2]);
    }
    return 0;
}
