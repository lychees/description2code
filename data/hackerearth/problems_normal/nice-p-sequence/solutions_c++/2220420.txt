#include<bits/stdc++.h>
using namespace std;
void extendedEuclid(long long int, long long int);
long long int d,x,y;
int main(){
  	int t;
   	scanf("%d",&t);
   	while(t--)
   	{
   		long long a1,p,n;
   		scanf("%lld %lld %lld",&a1,&p,&n);
   		long long b;
   		extendedEuclid(a1,p);
   		if(d==1){
   		if(x<0)
   		x+=p;
   		b=x;
   		long long s=a1+b;
   		if(n%2==0)
   		s=s*(n/2);
   		else
   		s=s*(n/2)+a1;
   		printf("%lld\n",s);
   		}
   		else
   		printf("-1\n");
   	}
   	return 0;
}

void extendedEuclid(long long int A, long long int B) {
    if(B == 0) {
        d = A;
        x = 1;
        y = 0;
    }
    else {
        extendedEuclid(B, A%B);
        int temp = x;
        x = y;
        y = temp - (A/B)*y;
    }
 }
