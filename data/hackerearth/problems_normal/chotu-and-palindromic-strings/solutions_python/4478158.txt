total_cases=raw_input()
case=int(total_cases)
new_list=[]

for i in range(case):
	temp=[]
	first_string=raw_input()	
	second_string=raw_input()
	temp.append(first_string)
	temp.append(second_string)
	new_list.append(temp)


def check_palindrome(first_string,second_string):
	new_str=first_string+second_string
	cmp_str=reversed(list(first_string+second_string))
	cmp_str="".join(cmp_str)
	if new_str==cmp_str:
		return True
	else:
		return False
		
for data in new_list:
	first_string=data[0]
	second_string=data[1]
	case1=check_palindrome(first_string,second_string)
	case2=check_palindrome(second_string,first_string)
	if case1 or case2:
		print "YES"
	else:
		print "NO"
