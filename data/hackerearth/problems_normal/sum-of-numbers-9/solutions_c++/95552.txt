#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int a[22];
int x,n;

int f(int idx, int sum)
{
	if ( idx == n ) {
		if ( sum == x ) return true;
		return false;
	}
	return f(idx+1,sum+a[idx]) || f(idx+1,sum);
}

int main() {
	int t,i;
	cin >> t;
	while ( t-- ) {
		cin >> n;
		for ( i = 0; i < n; i++ ) cin >> a[i];
		cin >> x;
		if ( f(0,0) ) cout << "YES" << endl;
		else cout << "NO" << endl;
	}
	return 0;
}
