#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
       	int c=0;
    	while(n>0)
    	{
    		n -= pow(2,int(log2(n)));
    		c++;
    		if(n==1)
    		{
    			c++;
    			break;
    		}
    	}
    	printf("%d\n",c);
    }
    return 0;
}
