#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
    int n, tmp, c = 0;
    vector<int> v;
    cin >> n;
    int **a;
    a = new int*[n];
    for(int i=0; i<n; i++) {
        cin >> tmp;
        v.push_back(tmp);
    }
    sort(v.begin(), v.end());
    tmp = v[0]-1;
    for(int i=0; i<n; i++) {
    	if(v[i] > tmp) {
    		c++;
    		tmp = v[i]+4;
    	}
    }
    cout << c << endl;
    return 0;
}
