#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define ll long long

int dp[15][15];

int main()
{
    int t,n,m,i,j,k,x,y;
    cin>>t;
    while(t--)
    {
    	memset(dp,0,sizeof(dp));
    	cin>>n>>m;
    	while(m--)
    	{
    		cin>>x>>y;
    		x--;
    		y--;
    		dp[x][y]=1;
    		dp[y][x]=1;
    	}
    	ll sum=0;
    	for(i=1;i<(1<<n);i++)
    	{
    		int flag=1;
    		for(j=0;j<n;j++)
    		{
    			if(i & 1<<j)
    			{
    				for(k=0;k<n;k++)
    				{
    					if(i & 1<<k)
    					{
    						if(dp[j][k]==1)
    						{
    							flag=0;
    							break;
    						}
    					}
    				}
    				if(flag==0)
    					break;
    			}
    		}
    		if(flag)
    			sum+=flag;
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
