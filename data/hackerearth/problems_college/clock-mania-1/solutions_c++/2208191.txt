#include <iostream>
using namespace std;
int calcAngle(double h, double m)
{
 
    if (h == 12) h = 0;
    if (m == 60) m = 0;
    int hour_angle = 0.5 * (h*60 + m);
    int minute_angle = 6*m;
    int angle = abs(hour_angle - minute_angle);
 
    angle = min(360-angle, angle);
 
    return angle;
}
 
int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--)
   {
   	double h,m;
   	cin>>h>>m;
   cout<<calcAngle(h,m)<<endl;
   }
   
   
   
   
   return 0;
}
