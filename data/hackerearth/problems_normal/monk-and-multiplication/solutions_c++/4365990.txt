#include <iostream>
#include<queue>
#include<algorithm>
using namespace std;

int main()
{
	priority_queue<long long int> pq;
	long long int n,temp,i;
	cin>>n;
	for(i=0;i<n;i++){
		cin>>temp;
		pq.push(temp);
		if(pq.size()<3){
			cout<<-1<<"\n";
		}
		else{
			long long int x=pq.top();
			pq.pop();
			long long int y=pq.top();
			pq.pop();
			long long int z=pq.top();
			pq.pop();
			long long int ans=x*y*z;
			pq.push(x);
			pq.push(y);
			pq.push(z);
			cout<<ans<<"\n";
		}
	}

	return 0;
}
