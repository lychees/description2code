#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
#define p(x) cout<<x<<":"<<x<<"\n"
int cs,i,n,s,j,x;
int main()
{
  scanf("%d",&cs);
  while(cs--)
  {
    scanf("%d",&x);
    for(i=1;;i++)
    {
      n=i*(i+1)/2;
      s=0;
      for(j=1;j*j<n;j++)
        if(!(n%j))
          s+=2;
      if(j*j==n)
        s++;
      if(s>x)
        break;
    }
    printf("%d\n",n);
  }
	return 0;
}
