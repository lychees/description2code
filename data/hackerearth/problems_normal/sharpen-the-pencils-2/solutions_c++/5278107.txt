#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--)
   {
	   	int n;
	   	cin>>n;
	   	long long int sum=0;
	   	long long int a[n];
	   	for(int i=0;i<n;i++)
	   	{
	   		cin>>a[i];
	   	}
	   	int b[n];
	   	int c[n];
	   	b[0]=0;
	   	c[n-1]=0;
	   	int c1=0;
	   	int c2=0;
	   	for(int i=1;i<n;i++)
	   	{
	   		b[i]=b[i-1]+a[i-1];	
	   	}
	   	for(int i=n-2;i>=0;i--)
	   	{
	   		c[i]=c[i+1]+(a[i+1]*2);
	   	}
	   	for(int i=0;i<n;i++)
	   	{
	   		if(b[i]<=c[i])
	   			c1++;
	   		else
	   			c2++;
	   		
	   	}
	   	
   	cout<<c1<<" "<<c2<<endl;
   	 
   }
    return 0;
}
