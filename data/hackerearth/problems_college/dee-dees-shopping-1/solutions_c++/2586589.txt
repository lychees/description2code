#define DEBUG(x) cout << '>' << #x << ':' << x << endl;

inline bool EQ(double a, double b) { return fabs(a - b) < 1e-9; }

int main() {
	int t,n,i;
	sc(t);
	while(t--) {
		sc(n);
		int len,div=n/8;
		int ans=-1;
		for(i=div;i>=0;i--) {
			len=n;
			len-=i*8;
			if(len%6==0) {
				ans=i;
				ans+=len/6;
				break;
			}
		}
		print(ans);
	}	
	return 0;
}
