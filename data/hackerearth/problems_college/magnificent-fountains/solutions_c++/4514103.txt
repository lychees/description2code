#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
#define ll long long
#define mod 1000000007
using namespace std;
ll power(ll x,ll y)
{
    if(y==0)
        return 1;
    if(y==1)
        return x%mod;
    ll tmp=((power(x,y/2)%mod)*(power(x,y/2)%mod))%mod;
    if(y&1)
        return (tmp*x)%mod;
    return tmp%mod;
}    
int main(void)
{
     ll  i,j,k,l,n,m;
    cin>>l;
    while(l--)
    {
        int a[10000]={0};
        cin>>n;
        while(n--)
        {
            cin>>m;
            for(i=2;i*i<=m;i++)
            {
                if(m%i==0)
                {
                    int c=0;
                    while(m%i==0)
                    {
                        c++;
                        m/=i;
                    }   
                    a[i]=max(a[i],c); 
                } 
            }    
            if(m>1)
                 a[m]=max(1,a[m]);
        }
        ll ans=1;
        for(i=0;i<10000;i++)
        {
            ans=ans*power(i,a[i])%mod;
        }        
        cout<<ans<<"\n";
    }    
    getchar();
     getchar();
    return 0;
}
