#include <iostream>
using namespace std;

int game(int arr[],int n,int index){
	if(arr[index] == -2)
		return 0;
	else if(arr[index] == -1)
		return 1;
	else{
		int temp_index = arr[index];
		arr[index] = -2;
		game(arr,n,temp_index);
	}
}
int main()
{
	int arr[100000],n,t;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(int i =0 ;i<n;i++)
		 cin>>arr[i];
    	if(game(arr,n,0))
    	cout<<"Win"<<endl;
    	else 
    	cout<<"Lose"<<endl;
	}
	return 0;
}
