#include<iostream>
#include<string.h>
using namespace std;

int main(){
    int N,n,T,i,j;
    cin>>T;
    string p,s;
    int a[26],b[26];
    while(T--){
        cin>>p;
        cin>>s;
        int f=0,plen=p.length(),slen=s.length();
        memset(a,0,sizeof(a));
        memset(b,0,sizeof(b));
        for(j=0;j<plen;++j){
            a[p[j]-97]++;

        }

        for(j=0;j<plen;++j){
            b[s[j]-97]++;
        }

        int found=0;
        for(i=plen;i<slen;++i){
            // check if found
            int f=1;
            for(int k=0;k<26;k++){
                if(a[k] != b[k]){
                    f=0;
                    break;
                }
            }
            if(f){
                found=1;
                break;
            }
            b[s[i-plen]-97]--;
            b[s[i]-97]++;
        }
        f=1;
        for(int k=0;k<26 && !found;k++){
            if(a[k] != b[k]){
                f=0;
                break;
            }
        }

        if(f){
            cout<<"YES"<<endl;
        }
        else{
            cout<<"NO"<<endl;
        }
    }
    return 0;
}
