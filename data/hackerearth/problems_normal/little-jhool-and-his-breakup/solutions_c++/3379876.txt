#include <iostream>

#define _ ios_base::sync_with_stdio(0);cin.tie(0);cout.tie(0);

using namespace std;

int main() {
	char c;
	int l = 0, o = 0, v = 0, e = 0;
	while((cin >> c)) {
		if(c == 'l')
			l = 1;
		else if((c == 'o') && (l == 1))
			o = 1;
		else if((c == 'v') && (o == 1))
			v = 1;
		else if((c == 'e') && (v == 1))
			e = 1;
		else {}
	}
	if(l&&o&&v&&e)
		cout <<  "I love you, too!";
	else
		cout <<  "Let us breakup!";
	return 0;
}
