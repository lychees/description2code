ll gcd(ll a,ll b)
{
    if(b==0)
      return a;
    return gcd(b,a%b);
}

ll power(ll a,ll n)
{
    if(n==0)
      return 1;
    ll x=power(a,n/2);
    x=(x*x)%mod;
    if(n%2!=0)
    {
      x=x*a;
      x=x%mod;
    }
    return x;
}
