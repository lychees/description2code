#include <iostream>
#include <algorithm>
using namespace std;
#define ll long long  int
int main()
{
    int t ;
    cin >> t;
    while(t){
    	int n;
    	cin >> n;
    	ll ar[n];
    	for(int i=0; i < n ; i++)
    		cin >> ar[i];
    		
    	sort(ar,ar+n);
    	ll mean = 0;
    	
    	
    	
    	for(int i=0; i < n ; i++){
    		mean += ar[i];
    	}
    	
    	
    	
    	mean /= n;
    	//	cout << mean << endl;
    	ll median;
    	if(n%2 == 1){
    		median = ar[n/2];
    	}else{
    		median = (ar[n/2 - 1] + ar[n/2])/2;
    	}
    	//cout << median << endl;
    	ll ans = 0;
    	ll ans2 = 0;
    	for(int i=0; i < n ; i++){
    		ans += abs(mean - ar[i]);
    		ans2 += abs(median - ar[i]);
    	}
    	cout << min(ans,ans2 ) << endl;
    	t-=1;
    }
    return 0;
}
