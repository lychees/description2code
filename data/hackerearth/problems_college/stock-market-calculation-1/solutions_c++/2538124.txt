#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    int t;
    int n;
    cin>>t;
    while(t--){
        cin>>n;
        int c[n];
        for(int i = 0; i < n; i++)
            cin>>c[i];
        int max = 0;
        int j = n-1;
	while(c[j] < c[j-1] && j > 0)
		j--;
	long int ans = 0;
        for(int i = j; i >= 0; i--)
        {
            if(max < c[i])
                max = c[i];
            ans += max - c[i];
        }
        cout<<ans<<endl;
    }
    return 0;
}
