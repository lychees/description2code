#include<iostream>
#include<stdlib.h>
#include<algorithm>
#define MAX 100004
#define MOD 1000000007
using namespace std;
int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}
int main() {
	long long int  arr[MAX],cum[MAX],max;
	long long int ans= 0;
	int N,T;
	scanf("%d",&T);
	while(T--) {
	max=0;
		scanf("%d",&N);
		for(int i=0;i<N;i++) {
			scanf("%lld",&arr[i]);
			if(arr[i]>max) max=arr[i];
		}
		sort(arr,arr+N);
		cum[0]=arr[0];
		for(int i=1;i<N;i++) {
			cum[i]= (cum[i-1]+arr[i]);
		//	cout<<arr[i]<<endl;
		}
		ans = 0;
		for(int i=N-1;i>0;i--) {
			long long int tmp =(arr[i] * i - cum[i-1]);
			ans = ((tmp%MOD)+ans)%MOD;
		}
		cout<<(ans*max)%MOD<<endl;
	}
}
