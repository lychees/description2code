#include <cstdio>
using namespace std;
#define mod 1000000007
long long ar[60];
int main()
{
    long long n,ans=0;
   
    ar[0]=2;
    for(int i=1;i<60;i++) ar[i]=(ar[i-1]*ar[i-1])%mod;
    int ntc; scanf("%d",&ntc);
    while(ntc--)
    {
    	ans=1;
    	scanf("%lld",&n);
    	for(int i=0;i<60;i++)
    	{
    	if((n>>i) & 1)
    	{
    	ans*=ar[i]; ans%=mod;
    	}
    	}
    	ans=ans-1; if(ans<0) ans+=mod;
    	printf("%lld\n",ans);
    }
    return 0;
}
