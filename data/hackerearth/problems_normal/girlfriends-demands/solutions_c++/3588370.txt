#include<iostream>
#include<cstring>

using namespace std;

int main()
{
    char str[100000];
    cin>>str;
    long long len = strlen(str);
    int t;
    cin>>t;
    while(t--)
    {
        long long num1,num2;
        cin>>num1>>num2;
        num1 = (num1-1)%len;
        num2 = (num2-1)%len;
        if(str[num1]==str[num2])
            cout<<"Yes"<<endl;
        else
            cout<<"No"<<endl;
    }
    return 0;
}
