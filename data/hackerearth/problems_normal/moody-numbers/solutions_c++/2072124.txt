#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int dp[1000010];
ll int mysum(ll int n)
{
	ll int ans=0;
	ll int r;
	while(n!=0)
	{
		r=n%10;
		n=n/10;
		ans+=r;
	}
	return ans;
}
ll int mycheck(ll int n)
{
		map<ll int,int> m;
		int flag=0;
		int stage=1;
		while(1)
		{
			//~ cout<<n<<endl;
			if(m[n]==stage) //Same value repeated
			{
				break;
			}
			else
			{
				if(n<1000010)
				{
					if(dp[n]==1)
					{
						flag=1;
						break;
					}
					else if(dp[n]==-1)
					{
						flag=0;
						break;
					}
				}
				m[n]=stage;
			}
			if(n==1||n==4)
			{
				flag=1;
				break;
			}
			else
			{
				if(stage==1)
				{
					n=n*n;
				}
				else
				{
					n=mysum(n);
				}
				if(stage==1)
				{
					stage=2;
				}
				else
				{
					stage=1;
				}
			}
		}
		if(flag)
		return 1;
		return -1;
}
int main()
{

	ll int i;
	for(i=1;i<1000010;i++)

					{
						flag=0;
						break;
					}
				}
				m[n]=stage;
			}
			if(n==1||n==4)
			{
				flag=1;
				break;
			}
			else
			{
				if(stage==1)
				{
					n=n*n;
				}
				else
				{
					n=mysum(n);
				}
				if(stage==1)
				{
					stage=2;
				}
				else
				{
					stage=1;

			}
		}
		if(flag)
		{
			printf("YES\n");
