#include <iostream>
#include<string>
using namespace std;
int total_days(int days)
{
	int total=0;
	while(days!=0)
	{
	total+=days;
	days--;
	}
	return total;
}
int countSetBits(string a,int days)
{
	int count=0;
	for(int i=0;i<a.size();i++)
	{
		if(a[i]=='1')
		count++;
	}
	int total=total_days(days);
	if(total%2==0)
	return count;
	return a.size()-count;
}
int main()
{
    int t,days;
    string a;
    cin>>t;
    int count=0;
    while(t!=0)
    {
    	count=0;
    	cin>>a;
    	cin>>days;
    	count=countSetBits(a,days);
    	cout<<count<<endl;
    	t--;
    }
    return 0;
}
