#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    while(n--)
    {
    	char s[100001];
    	cin>>s;
    	int c=0;
    	int l=strlen(s);
    	int i=1,j=0;
    	while(i<l)
    	{
    		if(s[j]=='X')
    		{
    			if(s[i]=='Y')
    			{
    				j=i;
    				i++;
   		 		}
    			else
    			{
    				s[i]='0';
    				j=i;
    				i++;
    			}
    		}
    		else
    		if(s[j]=='Y')
    		{
    			if(s[i]=='X')
    			{
    				j=i;
    				i++;
   		 		}
    			else
    			{
    				s[i]='0';
    				j=i;
    				i++;
    			}
    		}
    		else
    		if(s[j]=='0')
    		{
    			j=j-1;
    		}
    	}
    	for(int k=0;k<l;k++)
    	{
    		if(s[k]=='0')
    		c++;
    	}
    	cout<<c<<endl;
    	
    }
    return 0;
}
