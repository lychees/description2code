#include <iostream>
#include<vector>
#include<utility>
#include<algorithm>
using namespace std;

int main()
{
    int n, q,i,k;
    scanf("%d %d",&n,&q);
    vector <pair<int,int>>v(100006);
    for(i=0;i<n;i++)
    {
    	scanf("%d",&v[i].second);
    }
    for(i=0;i<n;i++)
    {
    	scanf("%d",&v[i].first);
    }
    sort(v.begin(),v.begin()+n);
    reverse(v.begin(),v.begin()+n);
    while(q--)
    {
    	scanf("%d",&k);
    	int sum =0;
    	for(i=0;i<k;i++)
    	sum+=v[i].second;
    	printf("%d\n",sum);
    	
    }
    return 0;
}
