#include<bits/stdc++.h>
using namespace std;
int main()
{
unsigned int t,ctr,i;
long long int x;
int a[9]={1000,500,100,50,20,10,5,2,1};
cin>>t;
while(t--)
{
    i=0;
    cin>>x;
    ctr=0;
    while(x)
    {
         if(x/a[i])
         {
          ctr+=x/a[i];
          x%=a[i];
         }
         i++;
     }
      cout<<ctr<<endl;

}
return 0;
}
