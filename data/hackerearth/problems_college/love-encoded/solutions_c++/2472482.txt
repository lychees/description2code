#include<iostream>
#include<string>
using namespace std;
main(){
int t,s[10],tp;
char c[10][50];
cin>>t;

for(int i = 0;i < t ; i++){
    cin>>s[i];
    cin>>c[i];
}
for(int i = 0; i < t; i++){

    for(int j = 0; c[i][j] != '\0';j++){
        tp = int(c[i][j]);
        //cout<<tp<<"  ";
        if( (tp - s[i])  >= 65 ){
            tp -=s[i];
        }
        else{
           tp = 91 + ((tp - s[i]) - 65);
        }
        //cout<<tp<<" ";
        c[i][j] = char(tp);
    }
}

for(int i = 0;i < t;i++){
    cout<<c[i]<<endl;
}
}
