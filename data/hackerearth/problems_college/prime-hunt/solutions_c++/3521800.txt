#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long unsigned int m,n,q;bool prime[100000000];
	
    memset(prime, true, sizeof(prime));
    prime[1]=false;
 
    for (long long unsigned int p=2; p*p<=100000000; p++)
    {
        
        if (prime[p] == true)
        {
            // Update all multiples of p
            for (long long unsigned int i=p*2; i<=100000000; i += p)
                prime[i] = false;
        }
    }
    cin>>n;
    for(long int i=0;i<n;i++)
    {
    	cin>>m>>q;
    	long long int c=0;
    	for(long long unsigned int j=m;j<=q;j++)
    	{
    		if(prime[j]==true)
    		c++;
    	}
    	cout<<c<<endl;
    }
    return 0;
}
