#include <stdio.h>
#include <algorithm>

using namespace std;

int a [101];
int b [10001][101];

int main ()
{
	int i, j, sum, n, c, t;
	scanf ("%d", &t);
	
	while (t--){
	
	scanf ("%d", &n);
	sum = 0;
	for (i = 0; i < n; i++) {
		scanf ("%d", & a [i]);
		sum = sum + a [i];
	}
	for (i = 1; i <= n; i++) {
	//	printf ("%d ", sum);
	}
	
	//sort (a, a + n);
	for (i = 0; i <= n; i++) {
		b [0] [i] = 1;
	}
	for (i = 1; i <= sum; i++) {
		for (j = 1; j <= n; j++) {
			b [i][j] = b [i][j - 1];
			if (i >= a [j - 1]) {
				b [i][j] = b[i][j] || b [i - a [j - 1]][j - 1];
			}
			
		}
	}
	c = 0;
	for (i = 0; i <= sum; i++) {
		c = c + b [i] [n];
	}
	
	printf ("%d\n", c);
	}
	
	return 0;
}
