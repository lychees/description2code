#include <iostream>
using namespace std;
 
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	string a;string b;
    	cin >> a >> b;
    	int i;
    	int ax[26];int bx[26];
    	for(i=0;i<=25;i++){ax[i]=0;bx[i]=0;}
    	for(i=0;i<a.length();i++)
    	{
    		ax[a[i]-'a']++;
    	}
    	for(i=0;i<b.length();i++)
    	bx[b[i]-'a']++;
    	int flag=0;
    	for(i=0;i<26;i++)
    	{
    		if(ax[i]>0 && bx[i] >0)
    		{
    			flag=1;break;
    		}
    	}
    	if(flag==1)
    	cout << "Yes" << endl;
    	else
    	cout << "No" << endl;
    }
    return 0;
}
