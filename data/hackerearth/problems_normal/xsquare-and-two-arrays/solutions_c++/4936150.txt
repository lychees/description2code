#include<bits/stdc++.h>
using namespace std;
#define s(x) scanf("%d",&x);
#define sl(x) scanf("%lld",&x);
#define p(x) printf("%d",x);
#define pl(x) printf("%lld",x);
#define fr(i,a,n) for(i=a;i<n;i++)
#define ln() printf("\n");
typedef long long ll;
ll mod=1000000007;
int main()
{
	ll n,q,i,x,y,z;
	sl(n)sl(q);
	ll a[n+1],b[n+1];
	a[0]=0,b[0]=0;
	for(i=1;i<=n;i++)
		sl(a[i]);
	for(i=1;i<=n;i++)
		sl(b[i]);
	for(i=2;i<=n;i+=2)
        swap(a[i],b[i]);
    for(i=2;i<=n;i++)
		a[i]+=a[i-1];
    for(i=2;i<=n;i++)
		b[i]+=b[i-1];
	while(q--)
	{
		sl(x)sl(y)sl(z);
		if(x==1)
		{
			if(y%2==1)
			{
				pl(a[z]-a[y-1]);
				ln();
			}
			else
				{
					pl(b[z]-b[y-1]);
					ln();
				}
		}
		else
		{
			if(y%2==0)
			{
				pl(a[z]-a[y-1]);
				ln();
			}
			else
			{
				pl(b[z]-b[y-1]);
				ln();
			}
		}
	}
	return 0;
}
