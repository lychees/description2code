#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;


long long int mp[26];
int main()
{
    string s;
    cin>>s;
    int n = s.length();
    int i=0;
    while(i<n)
    {
    	
    		int j = i+1;
    		int res = 0;
    		while(j<n&&s[j]>='0'&&s[j]<='9')
    		{
    			res = res*10+s[j]-'0';
    			j++;
    		}
    		mp[s[i]-'a']+=res;
    		i = j;
    }
    vector<pair<long long int,long long int> >tmp;
    for(int i=0;i<26;i++)
    {
    	if(mp[i])
    	{
    		tmp.push_back(make_pair(i,mp[i]));
    	}
    }
    for(int i=1;i<tmp.size();i++)
    {
    	tmp[i].second += tmp[i-1].second;
    }
    long long int k,q;
    sort(tmp.begin(), tmp.end());
    cin>>q;
    while(q--)
    {
    	cin>>k;
    	int i=0;
    	for(i=0;i<tmp.size();i++)
	    {
	    	
	    	if(k<=tmp[i].second)
	    	{
	    		break;
	    	}
	    }
	    //2  7
	    if(i==tmp.size())
	    {
	    	cout<<"-1\n";
	    }
	    else
	    {
	    	cout<<(char)(tmp[i].first+'a')<<"\n";	
	    }
    }
	    
    
    return 0;
}
