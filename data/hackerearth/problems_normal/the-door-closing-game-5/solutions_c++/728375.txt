bool is_prime[MAX];
int prime[MAX];
int first;

int sieve()
{
    long i,j;
    prime[0]=2;
    int k=1;
    int sq=(sqrt(MAX));
    for(i=3; i<=sq; i+=2)
    {
        if(!is_prime[i])
        {
            for(j=i*i; j<=MAX; j+=(2*i))
                is_prime[j]=1;
        }
    }
    for(j=3; j<=MAX; j+=2)
    {
        if(!is_prime[j])
        {
            prime[k++]=j;
        }
    }
    return k;
}
int sz;



long long SOD(int n)
{
    int  i,j,k;
    long sq=sqrt(n);
    long res=1;
    for(i=0; prime[i]<=sq && i < sz; i++)
    {
        int cnt=1;
        if(n%prime[i]==0)
        {
            if(!first) first = prime[i];
            while(n%prime[i]==0)
            {
                cnt++;
                n=n/prime[i];
            }
            res*= (ceil(pow((double)prime[i], (double)cnt) - 1) / (prime[i] - 1));
            sq=sqrt(n);
        }
    }
    if(n>1)
    {
        res*= (ceil(pow((double)n, 2.0) - 1) / (n - 1));
        if(!first) first = n;
    }
    return res;
}
