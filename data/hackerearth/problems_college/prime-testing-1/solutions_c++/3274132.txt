#include<bits/stdc++.h>
bool prime[1001];
void sieve()
{
int i,j;
for(i=0;i<=1000;i++)
prime[i]=true;
prime[0]=false;
prime[1]=false;
for(i=2;i*i<1001;i++)
{
	if(prime[i]==true)
	{
		for(j=i*i;j<1001;j+=i)
		prime[j]=false;
	}
}
}
main()
{
	sieve();
	int n,i;
	scanf("%d",&n);
	while(n--)
	{
		int m,count,a;
		scanf("%d",&m);
		count=0;
		for(i=0;i<m;i++)
		{scanf("%d",&a);
		if(prime[a]==true)
		count++;}
		printf("%d\n",count);
	}
}
