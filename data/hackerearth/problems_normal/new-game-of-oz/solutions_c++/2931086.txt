#include<stdio.h>
#include<iostream>
using namespace std;
int compare (const void * a, const void * b)
{return *(int*)a-*(int*)b;}
 	
int main(){
	
	int n,t;
	int a[101];
	cin>>t;
	while(t--){
		
	cin>>n;
for(int i=0;i<n;i++)
	cin>>a[i];
	qsort(a,n,sizeof(int), compare);
	int ans = 0;
	for(int i =0 ;i<n;i++){
		++ans;
		if(i+1<n&&a[i+1]==a[i]+1)
		++i;
	}
	cout<<ans<<endl;
	}
return 0;	
}
