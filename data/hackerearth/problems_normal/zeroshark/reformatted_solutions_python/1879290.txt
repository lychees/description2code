T = int(raw_input())
M = 100001
F = [0]*M
P = 1000000007
F[0] = 0
F[1] = 2
F[2] = 3
for i in xrange(3, M):
    F[i] = (F[i-1] + F[i-2]) % P
for _ in xrange(T):
    N = int(raw_input())
    c = 0
    for i in xrange(N-1):
        d = 1
        if (i - 1 > 0):
            d *= F[i-1]
        if (N - i - 3 > 0):
            d *= F[N-i-3]
        if d == 0:
            d = 1
        c += d % P
    print c % P
