#include<bits/stdc++.h>
#define nax 1000005
#define ll long long
#define plll pair< pair<ll,ll> , ll >
#define  f first.first
#define  s first.second
#define  t second
using namespace std;
plll arr[nax];
ll ans[nax];
ll ftree[nax];

void update(ll x)
{
    while(x<nax)
    {
        ftree[x]++;
        x+=((-x)&(x));
    }
}
ll query(ll x)
{
  ll sum=0;
    while(x>0)
    {
        sum+=ftree[x];
        x-=((-x)&x);

    }
    return sum;
}


int main()
{
    for(ll i=0;i<nax;i++)
        ftree[i]=0;
    ll n;
    cin>>n;
    for(ll i=1;i<=n;i++)
    {
        cin>>arr[i].f>>arr[i].s;
        arr[i].t=i;
    }
    sort(arr+1,arr+n+1);
    arr[0].f=LONG_MAX;
    for(ll i=1;i<=n;i++)
    {
        if(arr[i-1].f==arr[i].f && arr[i-1].s==arr[i].s)
        {
            ans[arr[i].t]=ans[arr[i-1].t];
            update(arr[i].s);
        }
        else
        {


        ans[arr[i].t]=query(arr[i].s);
        update(arr[i].s);
        }


    }
    for(ll i=1;i<=n;i++)
        cout<<ans[i]<<"\n";

}
