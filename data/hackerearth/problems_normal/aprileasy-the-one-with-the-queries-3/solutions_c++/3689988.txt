#include<stdio.h>
#include<string.h>
#include<bits/stdc++.h>
using namespace std;
char a[1000000];
char b[1000000];
int main()
{
	vector<int>v[26];
	vector<int>:: iterator it[26];
	int i;
	scanf("%s",a);
	int lm=strlen(a);
	for(i=0;i<lm;i++)
	v[a[i]-97].push_back(i);
	for(i=0;i<26;i++)
	it[i]=v[i].begin();
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int tem=0;
		int f=0;
		for(i=0;i<26;i++)
		it[i]=v[i].begin();
		scanf("%s",b);
		int lem=strlen(b);
		for(i=0;i<lem;i++)
		{
			while(it[b[i]-97]!=v[b[i]-97].end()&&*it[b[i]-97]<tem)
				it[b[i]-97]++;
			if(it[b[i]-97]==v[b[i]-97].end())
			{
				f++;
				break;
			}
			tem=*(it[b[i]-97])+1;
		}
		if(f)
			printf("No\n");
		else
			printf("Yes\n");
	}
	return 0;
}
