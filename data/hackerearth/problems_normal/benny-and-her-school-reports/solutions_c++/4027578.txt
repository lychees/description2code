#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */  
    ll t;
    cin>>t;
    while(t--)
        {
        ll n,m,p,x;
        cin>>n>>m>>x;
        ll temp=n,s=0;
        while(n--)
        {
            cin>>p;
            s+=p;
        }
        ll q=x*(temp+1)-s;
        if(q>=1&&q<=m)cout<<q<<endl;
        else if(q>m)cout<<"Impossible\n";
        else cout<<"1\n";
        
        
    }
    return 0;
}
