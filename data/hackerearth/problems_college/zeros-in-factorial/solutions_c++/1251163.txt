#include <iostream>
using namespace std;
const int N=1e6+6;
long long arr[N];
int count_zero(){
   int cnt=0;
   for(int k=0;k<N;k++){
   	cnt=0;
   for(long long i=5;i<=k;i=i*5){
   	  int j=(k/i);
   	  cnt=cnt+j;
   }
   arr[k]=cnt;
   if(k)
   arr[k]=arr[k-1]+arr[k];
 //  cout<<arr[k]<<" ";
   }
 return cnt;
}
int main()
{
    long long t;
    long long a,b;
    count_zero();
    cin>>t;
    while(t--){
    	cin>>a>>b;
      
        cout<<arr[b]-arr[a-1]<<"\n";
    }
    return 0;
}
