#include <bits/stdc++.h>
#define ull unsigned long long
#define ll long long
#define ff first
#define ss second
#define pb push_back
#define ins insert
#define mp make_pair
#define all(x) x.begin(),x.end()
#define iOS ios::sync_with_stdio(false)
using namespace std;
int n,arr[1005],dp[1004];
int main()
{
    scanf("%d",&n);
    int ones=0;
    for(int i=1;i<=n;i++)
        scanf("%d",&arr[i]);
    sort(arr+1,arr+n+1);
    reverse(arr+1,arr+n+1);
    dp[1]=1;
    int ans=1;
    for(int i=2;i<=n;i++)
    {
        dp[i]=1;
        for(int j=i-1;j;j--)
            if(arr[j]%arr[i]==0)
                dp[i]=max(dp[i],dp[j]+1);
        ans=max(ans,dp[i]);
    }
    if(ans>1) printf("%d\n",ans);
    else printf("-1\n");
}
