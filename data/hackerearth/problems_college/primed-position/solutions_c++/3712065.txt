#include <stdio.h>
#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
	vector<long long int>v;
	long long int a[5000008];
void prime()
{   long long i,j,n,c=0;
	n=5000002;

	for(i=1;i<=n;i++)
		a[i]=i;
	for(i=2;i<=2240;i++)
	{
		if(a[i]==i)
		{
			
		
			
			          for(j=i*i;j<=n;j=i+j)
			           if(a[j]==j)
				a[j]=i;
		}
	}	
	
	for(i=2;i<=n;i++)
	{
		if(a[i]==i)
		{
			c++;
			if(i!=2&&a[c]==c){
				v.push_back(i);
			//	printf("%lld %lld##",c,i);
			}
		}
	}
	
}
int main()
{
	int t;
	scanf("%d",&t);
	 prime();
	while(t--){
		int n,m;
  
   scanf("%d%d",&n,&m);
  // printf("%d",v.size());
 // for(int i=0;i<6;i++)
 //printf("%lld ",v[i]);
   printf("%lld\n",(v[n-1]*v[m-1])%mod);
		
	}
    return 0;
}
