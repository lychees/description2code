#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++){
    	int px,py,ax,ay;
    	cin>>px>>py>>ax>>ay;
    	if((px>=ax && py>=ay)||px>=ay &&py>=ax){
    		cout<<"Possible"<<endl;
    	}
    	else{
    		cout<<"Not Possible"<<endl;
    	}
    }
    return 0;
}
