#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    unsigned short int testCases ;
    unsigned short int N ;
    unsigned short int i , j ;
    int matrixCost[100] ;
    string output[10] ;
    unsigned short int sum ;
    
    cin >> testCases ;
    for( i = 0 ; i < testCases ; i++ )
    {
    	sum = 0 ;
    	cin >> N ;
    	for ( j = 0 ; j < N ; j++ )
    	{
    		cin >> matrixCost[j] ;
    		sum += matrixCost[j] ;
    	}
    	if( ( sum % 2 ) == 0 )
    	{
    		output[i] = "YES" ;;
    	}
    	else
    	{
    		output[i] = "NO" ;
    	}
    	
    }
    for( i = 0 ; i < testCases ; i++ )
    {
    	cout << output[i] << endl ;
    }
    return 0;
}
