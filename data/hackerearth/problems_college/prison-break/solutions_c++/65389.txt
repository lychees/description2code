#include<iostream>
#include<cstdio>
#include<cmath>
#include<iomanip>
using namespace std;
int main()
{
    setprecision(25);
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long h,v,d,temp;
        scanf("%lld%lld%lld",&h,&d,&v);
        temp=d*d-h*h;
        double x=sqrt(temp);
        x/=h;
        x*=v;
        printf("%.9lf\n",x);
    }
}
