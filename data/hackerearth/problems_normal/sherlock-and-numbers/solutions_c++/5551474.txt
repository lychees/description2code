#include <bits/stdc++.h>
#define N 100001
typedef long long ll;
static ll a[N];
using namespace std;
int main()
{	ll n,k,p,t,i;
	cin>>t;
	while(t--){
		cin>>n>>k>>p;
		for(i=1;i<=k;i++){
			cin>>a[i];
			if(a[i]<=p) p++;
		}
		
		if(p<=n)cout <<p<< endl;
		else cout<<"-1"<<endl;
	}
    
    return 0;
}
