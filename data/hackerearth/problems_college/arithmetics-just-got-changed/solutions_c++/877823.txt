#include <iostream>
#include <vector>
#define ll long long
using namespace std;

ll rev(ll N) {
	vector<ll> num;
	ll ans=0;
	while(N) {
		ans=ans*10+(N%10);
		N/=10;
	}
	return ans;
}


int main()
{
    ll T,A,B;
    cin>>T;
    while(T--) {
    	cin>>A>>B;
    	cout<<rev(rev(A)+rev(B))<<endl;
    }
    return 0;
}
