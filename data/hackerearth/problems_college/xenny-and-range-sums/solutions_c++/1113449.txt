#include <iostream>
#include<cstdio>
using namespace std;
int a[1000005]={0},b[1000005]={0};
int main()
{
	int n,l,r,i,q;
	scanf("%d %d",&n,&q);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		b[i]=b[i-1]+a[i];
	}
	while(q--)
	{
		scanf("%d %d",&l,&r);
		printf("%d %d\n",b[r]-b[l-1],a[r]-(b[r-1]-b[l-1]));
	}
    return 0;
}
