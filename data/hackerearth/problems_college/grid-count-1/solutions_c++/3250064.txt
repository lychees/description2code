#include <iostream>
#include <map>
#include <string>

using namespace std;

const int N = 1003;
const long MAX = 1000000001;
long matrix[N][N] = {{0}};
long row[N][N] = {{0}}, col[N][N] = {{0}};

int main()
{
    ios::sync_with_stdio(0);
    
    int n, k, cnt = 0;
    map<long,int> mp;

    cin >> n >> k;

    for( int i = 1; i <= n; ++i )
        for( int j = 1; j <= n; ++j )
            cin >> matrix[i][j];

    for( int i = 1; i <= n; ++i )
        for( int j = 1; j <= n; ++j )
            row[i][j] = row[i][j-1] + matrix[i][j];

    for( long y = 1; y * y * y <= 3 * MAX; ++y )
        mp[y*y*y] = 1;

    for( int i = 1; i <= n - k + 1; ++i )
        for( int j = 1; j <= n - k +1; ++j ){
            long sum = 0;
            for( int u = 1; u <= k; ++u )
                sum += row[u+i-1][j+k-1] - row[u+i-1][j-1];

            for( long u = 1; u * u * u <= sum; ++u )
                if( u * u * u == sum ) {
                        cnt++;
                        break;
                }
    }

    cout << cnt;
    return 0;
}
