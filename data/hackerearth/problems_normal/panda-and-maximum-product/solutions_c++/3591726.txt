#include <bits/stdc++.h>
using namespace std;

int main()
{
   long int N,n,sum1,sum2;
   cin>>N;
   vector<long int> a;
   for(long int i=0;i<N;i++)
   {
   		cin>>n;
   		a.push_back(n);
   }
   if(N==2)
   	cout<<a[1]*a[0];
   else
   {
   	sort(a.begin(),a.end());
   	sum1=a[0]*a[1];
   	sum2=a[N-1]*a[N-2];
   	if(sum1>sum2)
   	cout<<sum1;
   	else if(sum1<sum2 && abs(sum1)>sum2)
   	{
   		for(long int j=2;j<N-2 && abs(sum1)>sum2;j++)	
   		{
   			sum1=a[j]*a[j-1];
   			if(sum1>sum2)
   				cout<<sum1;
   			else if(abs(sum1)<sum2)
   				cout<<sum1;
   		}
   	}
   	else
   		cout<<sum2;
   }
    return 0;
}
