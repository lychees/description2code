#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int i,n,k;
    	scanf("%d%d",&n,&k);
    	long long int p;
    	unordered_set<long long int> mys;
    	for(i=0;i<n;i++)
    	{	scanf("%lld",&p);
    		mys.insert(p);
    	}
    	while(k--)
    	{	
    		
    		scanf("%lld",&p);
    		
    		if(mys.find(p)!=mys.end())
    			printf("YES\n");
    		else
    			printf("NO\n");
    		mys.insert(p);
    	}
    	
    }
    return 0;
}
