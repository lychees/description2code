#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
#define pb push_back

using namespace std;

typedef long long ll;

int main()
{
    //freopen("test.txt", "r", stdin);
    int n, x, elem;
    cin >> n;
    elem = n;
    vector<int> num;
    for(int i=0; i<n; i++){
        cin >> x;
        num.pb(x);
    }

    int point = 0;
    for(int i=0; i<n-2; i++){
        int pos, cur = -1;
        for(int j=1; j<elem-1; j++){
            if(num[j-1]*num[j+1] > cur){
                cur = num[j-1]*num[j+1];
                pos = j;
            }
        }

        point += cur;
        num.erase(num.begin()+pos);
        elem--;
    }

    cout << point;
    return 0;
}
