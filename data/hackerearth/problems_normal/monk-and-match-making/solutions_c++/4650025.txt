#include<bits/stdc++.h>
#define B1 33
#define B2 166
#define M1 10000007
#define M2 10000009
using namespace std;

string s;
long long int p1[100005],p2[100005];
long long int h1[100005],h2[100005];

void prec()
{
	int n=s.size();
	p1[0]=1;
	p2[0]=1;
	for(int i=1;i<=n;i++)
	{
		p1[i]=(p1[i-1]*B1)%M1;
		p2[i]=(p2[i-1]*B2)%M2;
	}
	for(int i=1;i<=n;i++)
	{
		h1[i]=(h1[i-1]*B1+(s[i-1]-'a'+1))%M1;
		h2[i]=(h2[i-1]*B2+(s[i-1]-'a'+1))%M2;
	}
}

long long int hash1(int l,int r)
{
	long long int val;
	int len=(r-l+1);
	val=(h1[r]-(h1[l-1]*p1[len])%M1+M1)%M1;
	return val;
}

long long int hash2(int l,int r)
{
	long long int val;
	int len=(r-l+1);
	val=(h2[r]-(h2[l-1]*p2[len])%M2+M2)%M2;
	return val;
}

int main()
{
	cin >> s;
	prec();
	int n;
	cin >> n;
	int l1,l2,r1,r2;
	for(int i=0;i<n;i++)
	{
		cin >> l1 >> r1 >> l2 >> r2;
		if((r1-l1)!=(r2-l2))
			cout << "No" << endl;
		else
		{
			pair<long long int ,long long int> H1(hash1(l1,r1),hash2(l1,r1));
			pair<long long int ,long long int> H2(hash1(l2,r2),hash2(l2,r2));
			if(H1==H2)
			{
				cout << "Yes" << endl;
			}
			else
			{
				cout << "No" << endl;
			}
		}
	}
	return 0;
}
