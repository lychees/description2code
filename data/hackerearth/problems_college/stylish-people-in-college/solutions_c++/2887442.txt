#include<bits/stdc++.h>
using namespace std;
//#define max(a,b) a>b?a:b;

int main()
{
	int t,n;
	scanf("%d",&t);
	for(int j=1;j<=t;j++)
	{
		scanf("%d",&n);
		long long int a[n],i,dp[n];
		for(i=0;i<n;i++)
		scanf("%lld",a+i);
		dp[0]=a[0];
		if(n>=2)
		{
		dp[1]=a[1]>dp[0]?a[1]:dp[0];
		for(i=2;i<n;i++)
		{
			dp[i]=max(a[i]+dp[i-2],dp[i-1]);
		}
	    }
		sort(dp,dp+n);
		printf("Case %d: %lld\n",j,dp[n-1]);
	}
	return 0;
}
