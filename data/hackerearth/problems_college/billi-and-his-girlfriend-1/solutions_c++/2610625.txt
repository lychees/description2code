#define mset(x,v) memset(x, v, sizeof(x))

ll modexp(ll x,ll y,ll n){
ll ld,temp,ans=1;
ld =y%2;
y=y/2;
temp=x%n;
if(ld==1)
ans=temp;
while(y!=0){
ld =y%2;
y=y/2;
temp=(temp*temp)%n;
if(ld==1){
ans=(ans*temp)%n;
}
}
return ans;
}

ll sum(ll n){
ll ans=0;
while(n!=0){
ans=ans+n%10;
n=n/10;
}
return ans;
}


int main(){
ll t;
sl(t);
while(t--){
ll a,b,k;
sl(a);sl(b);sl(k);
ll temp;
temp=modexp(a,b,pow(10,k));
pl(sum(temp));
enter;
}
return 0;
}
