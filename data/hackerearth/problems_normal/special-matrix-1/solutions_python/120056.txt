import sys

t=int(input())

while t>0:
    n=int(input())
    i=1
    r=0
    c=0
    while i<=n:
        s=sys.stdin.readline().strip()
        if '*' in s:
            r=i
            c=s.index('*')+1
        i=i+1
    mi=(n+1)/2
    #print r,c
    print abs(mi-r)+abs(mi-c)
    t=t-1
