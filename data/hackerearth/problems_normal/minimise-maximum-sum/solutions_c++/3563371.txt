#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int N,Num, ans = INT_MIN;
		vector<int> V;
		cin>>N;
		for(int i=0;i<2*N;i++)
		{
			cin>>Num;
			V.push_back(Num);
		}
		sort(V.begin(), V.end());
		int end = 2*N-1;
		for(int i=0;i<N;i++)
		{
			int s = V[i]+V[end];
			end--;
			ans = max(s, ans);
		}
		cout<<ans<<endl;
	}
    return 0;
}
