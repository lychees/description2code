#include <bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		string s;
		cin>>s;
		int arr[30]={0};
		vector<pair<int,int> > vec;
		vec.clear();
		for(int i=0;i<s.length();i++){
			arr[s[i]-'a']++;
			if(arr[s[i]-'a']==1){
				vec.push_back(make_pair(i,s[i]));
			}
		}
		sort(vec.begin(),vec.end());
		for(int i=0;i<vec.size();i++){
			printf("%c",vec[i].second);
		}
		printf("\n");
	}
	return 0;
};
