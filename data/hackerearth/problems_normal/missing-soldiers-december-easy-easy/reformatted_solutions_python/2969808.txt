n = int(raw_input())
l = []
for _ in range(n):
    x, y, d = map(int, raw_input().split())
    l.append((x, d))
l.sort(key=lambda x: x[0])
start, d = l.pop(0)
sum = d+1
last = start + d
for start, d in l:
    if start > last:
        sum += d+1
        last = start + d
    elif start <= last and start+d > last:
        sum += start + d - last
        last = start + d
print sum
