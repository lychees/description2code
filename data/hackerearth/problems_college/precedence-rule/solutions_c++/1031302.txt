#include<iostream>
#include<cstring>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		string s;
		cin>>s;
		long int temp,ans=s[s.length()-1]-'0';
		for(int i=s.length()-3;i>=0;i-=2){
			temp=s[i]-'0';
			if(s[i+1]=='+')
			ans+=temp;
			else if(s[i+1]=='-')
			ans-=temp;
			else if(s[i+1]=='*')
			ans*=temp;
			
		}
		cout<<ans<<endl;
	}
	return 0;
}
