#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
long long expo(int x,int n)
{
	if(n==0)
		return 1;
	else if(n==1)
		return x;
	long long temp=expo(x,n/2);
	temp=((temp%mod)*(temp%mod))%mod;
	if(n%2)
		temp=((temp%mod)*(x%mod))%mod;
	return temp;
}
int main()
{
	int t;
    cin>>t;;
    while(t--)
    {
    	int n;
    	cin>>n;
    	map<int,int> mp;
    	int count=0;
    	for(int i=0;i<n;i++)
    	{
    		int x;
    		cin>>x;
    		if(!mp[x])
    		{
    			mp[x]=1;
    			count++;
    		}
    	}

    	cout<<expo(n,count)<<endl;
    }
	return 0;
}
