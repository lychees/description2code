#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int c1,d1,c2,d2,r1,r2;
     double ans;
    cin>>c1>>d1>>r1;
    cin>>c2>>d2>>r2;
    ans=sqrt((c2-c1)*(c2-c1)+(d2-d1)*(d2-d1));
    if(ans==r1+r2)
    {
    	printf("T");
    }
    else if(ans>r1+r2)
    	printf("N");
    else
    	printf("Y");
    return 0;
}
