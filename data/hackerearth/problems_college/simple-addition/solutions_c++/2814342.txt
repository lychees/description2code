#include <bits/stdc++.h>
#define INF 0x7fffffff
#define INFLL 1e17
using namespace std;

int solve( int x, int y){
	if( x<1 || y<1)
		return INF/2;
	//cout<<x<<" - "<<y<<"\n";
	if( x==y )
		return x==1?0:INF/2;
	if( x>y )
		return solve(y, x);

    if( y%x!=0 )
        return solve(y%x, x)+y/x;
    if( x==1 )
        return y/x-1;
    return INF/2;//min( solve(x-y, y), solve(x, y-x) )+1;
}

int main()
{
    int T, n, tam = 1000010, cas=0, idx;
    long long x, y, d, xx, yy;

    for( cin>>T; cas<T; cas++) {
    	cin>>n;
    	int res = n==1?0:INF/2;
    	for( int i=1; 2*i<=n; i++){
    		//cout<<"->"<<i<<" "<<n-i<<endl;
    		int aux = solve(i, n-i)+1;
    		//cout<<"*->"<<i<<" "<<n-i<<" - "<<aux<<endl;
    		res = min( res, aux );
    	}
    	//cout<<n<<"RES: ";
    	cout<<res<<"\n";
    }

    return 0;
}
