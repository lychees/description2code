#include<iostream>
#include<cstdio>
using namespace std;
int arr[51][51][51];

int main()
{
    int t, n;
    scanf("%d", &t);
    for(; t--;)
    {
        scanf("%d", &n);
        for(int i = 1; i<=n; i++)
        {
            for(int j = 1; j<=n; j++)
            {
                for(int k = 1; k<=n; k++)
                {
                    cin>>arr[i][j][k];
                }
            }
        }
        for(int i = 1; i<=n; i++)
        {
            for(int j = 1; j<=n; j++)
            {
                for(int k  = 1; k<=n; k++)
                    cout<<arr[i][j][k]-arr[i-1][j][k]-arr[i][j-1][k]+arr[i-1][j-1][k]-arr[i][j][k-1]+arr[i][j-1][k-1]+arr[i-1][j][k-1]-arr[i-1][j-1][k-1]<<" ";
                cout<<endl;
            }
        }

    }
}
