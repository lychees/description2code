#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int ar[10010];

long int DP[10010],size,k;

long int ret( int n){
  if(n >= size ) return 0;
  if(DP[n] != -1) return DP[n];
  

  DP[n] = max(ret(n+1) , ar[n] + ret(n+k+1));
  return DP[n];

}







int main()
{
  long int t;
  cin >> t;
  while(t--){
    
    cin >> size>>k;
    
    for(int i = 0; i < size; i++){
      cin >> ar[i];
    }
    for(int i = 0; i < 10010; i++){
      DP[i] = -1;
    }
    long int val = 0;
    long int max = -899859;
    for(int i = 0; i < size; i++){
      if(ar[i] > max) max = ar[i];
    }
    val = ret(0);
    if(val < 0)cout << 0 <<endl;
    else if(max > val) cout << max <<endl;
    
    else cout << val << endl;
  }

    
    
    return 0;
}
