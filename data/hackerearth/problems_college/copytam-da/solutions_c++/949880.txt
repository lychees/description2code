#include <iostream>
using namespace std;

int main()
{
    long int t;
    cin>>t;
    if(t<1 || t>10)
    {
    	return 0;
    }
    long int n;
    long int f[t];
    for(int i=0;i<t;i++)
    {
    	f[i]=0;
    	cin>>n;
    	if(n<1 || n>1000000)
    	{
    		return 0;
    	}
    	int a[n];
    	for(int j=0;j<n;j++)
    	{
    		cin>>a[j];
    		if(a[j]<1 || a[j]>10000)
    		{
    			return 0;
    		}
    		if(j!=0)
    		{
    			int fi=0;
    			for(int k=0;k<j;k++)
    			{
    				if(a[j]==a[k])
    				{
    					fi=1;
    					break;
    				}
    			}
    			if(fi==0)
    			{
    				f[i]++;
    			}
    		}
    		else
    		{
    			f[i]++;
    		}
    	}
    }
    for(int i=0;i<t;i++)
    {
    	cout<<f[i]<<endl;
    }
}
