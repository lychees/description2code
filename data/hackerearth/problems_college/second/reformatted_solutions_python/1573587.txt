n, m = map(int, raw_input().split())
lst = sorted(map(int, raw_input().split()))
ans = 0
if m in lst:
    pos = lst.index(m)
    ans = lst[pos]

    for i in xrange(pos+1, len(lst)):
        if lst[i] == (ans + m):
            ans = lst[i]
        elif lst[i] > (ans + m):
            break

print ans
