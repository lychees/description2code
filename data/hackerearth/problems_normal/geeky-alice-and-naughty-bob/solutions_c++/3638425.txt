int Rsum(long long int x)
{
	long long int sum=10,temp=10,y;
	while(sum%10!=sum)
	{	
		sum=0;
		temp=10;
		while(x>=temp/10)
		{
			y=x%temp;
			y=y/(temp/10);
			sum+=y;
			temp*=10;
		}
		x=sum;
	}
	return sum;	
}
