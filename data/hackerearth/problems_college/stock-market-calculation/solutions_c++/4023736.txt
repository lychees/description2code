#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define sc1ll(x) scanf("%lld",&x)
#define sc2ll(x,y) scanf("%lld%lld",&x,&y)
vector<ll> arr(100010);
int main()
{
    ios_base::sync_with_stdio(false);
    ll test,maxx,num;
    cin>>test;
    while(test--)
    {
        cin>>num;
        for(ll i=1;i<=num;i++)
            cin>>arr[i];
        maxx=-100;
        ll ans=0;
        for(ll i=num;i>=1;i--)
        {
            maxx=max(maxx,arr[i]);
            ans+=(maxx-arr[i]);
        }
        cout<<ans<<endl;
    }
}
