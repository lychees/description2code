int z[300005];
void z_calc(string s){
    int L = 0, R = 0;
    int n = s.size();
    for (int i = 1; i < n; i++) {
      if (i > R) {
        L = R = i;
        while (R < n && s[R-L] == s[R]) R++;
        z[i] = R-L; R--;
      } else {
        int k = i-L;
        if (z[k] < R-i+1) z[i] = z[k];
        else {
          L = i;
          while (R < n && s[R-L] == s[R]) R++;
          z[i] = R-L; R--;
        }
      }
    }

}
int main(){

	ll i , j , k , t;
    string a , b;
    cin >> t;
    while(t--){
        cin >> a >> b;
        memset(z,0,sizeof(z));
        a+=a;
        string temp;
        temp = b;
        temp+=a;
        z_calc(temp);
        int n = b.size();
        int cnt = 0;
        for(i = n ; i < temp.size()- n  ; i++){
            if(z[i]>=n)cnt++;
        }
        cout << cnt << endl;

    }
	return 0;
}
