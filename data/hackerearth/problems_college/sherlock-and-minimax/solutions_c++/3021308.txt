
#include <bits/stdc++.h>
using namespace std;
int n;
int arr[109];
int A, B;
int answer;

int check( int x )
{
   int ret = INT_MAX;
   for( int i = 0; i < n; ++i )
      ret =min(ret, abs( x- arr[i] ));
   return ret;
}

void limits( int x )
{
   if( x < A || x > B ) return;
   if(check(x) > check(answer)) answer = x;
}

int main()
{
   scanf( "%d", &n );
   for( int i = 0; i < n; ++i ) scanf( "%d", &arr[i] );
   scanf( "%d%d", &A, &B );

   answer = A;

   limits( A );
   limits( B );

   for( int i = 0; i < n; ++i )
      for( int j = i+1; j < n; ++j )
       {

         limits( (arr[i]+arr[j])/2 );
       }

   printf( "%d\n",answer );

   return 0;
}
