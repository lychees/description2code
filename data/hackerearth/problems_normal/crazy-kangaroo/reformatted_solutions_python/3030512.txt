#!/usr/bin/python
import math

'''
http://math.stackexchange.com/questions/211445/how-many-multiples-of-x-lie-in-the-arbitrary-range-y-z
'''

TestCases = int(raw_input())
if 1 <= TestCases <= 100000:
    for i in range(TestCases):
        # 1000000000000
        s = raw_input()
        n = map(float, s.split())
        print int(max(math.floor(n[1]/n[2])-math.ceil(n[0]/n[2])+1, 0))
        # print int((n[1]/n[2])-(n[0]-1)/n[2])
