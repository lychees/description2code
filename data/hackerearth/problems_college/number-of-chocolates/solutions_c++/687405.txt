#define PB push_back
#define MP make_pair
#define F first
#define S second
#define sc(x) scanf("%d",&x)
#define scl(x) scanf("%lld",&x)
#define MAXN 2000001
#define M 1000000007
#define LL long long
#define get getchar
#define pc(x) putchar(x)

using namespace std;

LL a[MAXN + 1], b[MAXN + 1];
vector<pair<int, int > > v;
int main()
{
    long long  n, m, ans, i, x, x1, x2, x3, x4,x5,x6,x7,x8;
    sc(n);
    sc(m);
    b[0] = 0;
    for(i = 1; i <= n; i++)
    {
        scl(a[i]);
        b[i] = b[i - 1] + a[i];
    }
    while(m--)
    {
        cin>>x1>>x3;
        cin>>x2>>x4;
        x1++;
        x2++;
        x3++;
        x4++;
        x5 = min(x1, x2);
        x6 = max(x1, x2);
        x7 = min(x3, x4);
        x8 = max(x3, x4);

        v.clear();
        v.PB(MP(x5, x6));
        v.PB(MP(x7, x8));
        sort(v.begin(), v.end());
        x1 = v[0].first;
        x2 = v[0].second;
        x3 = v[1].first;
        x4 = v[1].second;

        if(x3 > x2)
        {
            //cout<<"Yes";
            ans = b[x2] - b[x1 - 1] + b[x4] - b[x3 - 1];
        }
        else
        {
            ans = b[ max(x4, x2) ] - b[x1 - 1];
        }
        cout<<ans<<endl;
    }
    return 0;
}
