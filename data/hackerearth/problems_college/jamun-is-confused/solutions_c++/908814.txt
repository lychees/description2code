#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
#include<bits/stdc++.h>

#include<iostream>
using namespace std;
int gcd(int a,int b)
{

	if(b==0)
		return a;
	if(a==0)
		return b;
	return gcd(b,a%b);
}
int main(void)
{
	 int t,n,*ptr,i,ans;
	cin >> t;
	while(t--)
	{
		cin >> n;
		ptr=(int *)malloc(sizeof(int)*n);
		for(i=0;i<n;i++)
			scanf("%d",(ptr+i));
		sort(ptr,ptr+n);
		ans=ptr[0];
		for(i=1;i<n;i++)
			ans=gcd(ans,ptr[i]);
		cout << ans << endl;
	}
	return 0;
}
