using namespace std;
pair<int, int>b[100007];
int a[100007];
int main()
{
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);
	int t;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++)
	{
		int n;
		scanf("%d", &n);
		for (int i = 1; i <= n; i++)
		{
			scanf("%d", &a[i]);
			b[i].first = a[i];
			b[i].second = i;
		}
		sort(b + 1, b + 1 + n);
		reverse(b + 1, b + 1 + n);
		int j = 1;
		int have = 0;
		long long bal = 0;
		for (int i = 1; i <= n; i++)
		{
			if (b[j].second == i)
			{
				bal += 1LL * a[i] * have;
				have = 0;
				while (j <= n && b[j].second <= i) j++;
			}
			else
			{
				bal -= a[i];
				have++;
			}
		}
		printf("%lld\n", bal);
	}
}
