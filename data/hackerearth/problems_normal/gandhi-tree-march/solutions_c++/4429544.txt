#define ll long long int
#define maxN 20000
#define mid 10000
#define maxVal 10000
#define minVal -100000000
#define mod 1000000007LL

#define gcd(a,b) __gcd(a,b)

using namespace std;

char a[maxN+5];
vector<char> g[maxN+5];
int p,l;

void compute(int c)
{
    if(p<l&&a[p]!='.')
    {
        if(abs(c)<=maxVal)
            g[c+mid].push_back(a[p]);
        p+=2;
        compute(c-1);
        compute(c+1);
    }
    p++;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    #ifndef LOCAL_SYS
        //freopen("in.txt","r",stdin);
        //freopen("out.txt","w",stdout);
    #endif
    int t,i,c;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%s",&c,a);
        l=strlen(a);
        for(i=0;i<=maxN+2;i++)
            g[i].clear();
        p=0;
        compute(0);
        c=c+mid;
        if((int)g[c].size()==0)
            printf("Common Gandhijee!\n");
        else
        {
            sort(g[c].begin(),g[c].end());
            for(i=0;i<(int)g[c].size();i++)
                printf("%c",g[c][i]);
            printf("\n");
        }
    }
    return 0;
}
