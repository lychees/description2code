#include <bits/stdc++.h>
#define lli long long int
using namespace std;
struct Point{
	lli x;
	lli y;
	lli val;
};
int main(){
//	freopen("input.in","r",stdin);
//	freopen("output3.out","w",stdout);
	lli t=1;
	cin >> t;
	while(t--){
		int N;
		bool choice,res=true;
		cin >> N;
		Point A,B;
		cin >> A.x >> A.y >> A.val;
		if((A.x+A.y)%2==0 && A.val%2==0) choice=true;
		else if((A.x+A.y)%2==1 && A.val%2==1) choice=true;
 		else choice=false;
		for(int i=1;i<N;i++){
			cin >> B.x >> B.y >> B.val;
			if(choice){
				if((B.x+B.y)%2==0 && B.val%2==1 ) res=false;
				if((B.x+B.y)%2==1 && B.val%2==0 ) res=false;
//				if(res) cout << "Res is true And choice is true." << endl;
//				else cout << "Res is false And choice is true." << endl;
			} 
			else {
				if((B.x+B.y)%2==0 && B.val%2==0 ) res=false;
				if((B.x+B.y)%2==1 && B.val%2==1 ) res=false;
//				if(res) cout << "Res is true And choice is false." << endl;
//				else cout << "Res is false And choice is false." << endl;
			} 
		}
		if(res) cout << "Yes" << endl;
		else cout << "No" << endl;
	}
}
