#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long int t,i,j,a[100001];
    cin>>t;
    while(t--)
    {
     long long int n,count=0;
     cin>>n;
     for(i=0;i<n;i++)
     cin>>a[i];
     for(i=0;i<n;i++)
     {
      for(j=i;j<n;j++)	
      {
      	if(j>=i && a[i]==a[j])
      	{
      	 count++;	
      	}
      }
     }
     cout<<count<<endl;
    }
    return 0;
}
