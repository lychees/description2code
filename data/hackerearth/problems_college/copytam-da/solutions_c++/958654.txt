#include <iostream>
#include <algorithm>
using namespace std;
int main(void){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int a[n];
		int i;
		for(i=0;i<n;i++){
			cin>>a[i];
		}

		sort(a,a+n);
		int count=0;
		for(i=0;i<n;i++)
		{
			if(a[i]!=a[i+1]){
				count++;
			}
		}
		cout<<count<<endl;
	}
}
