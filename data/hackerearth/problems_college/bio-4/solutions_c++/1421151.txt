#include<bits/stdc++.h>
using namespace std;
char a[1005],b[1005];
int main() {
	int t,n,i,j,count,al,bl,r,k;
	scanf("%d",&t);
	getchar();
	while(t--) {
		scanf("%s%s",a,b);
		getchar();
		al=strlen(a);
		bl=strlen(b);
		if(al==bl) {
			count=0;
			for(i=0;i<al;i++) {
				if(a[i]!=b[i]) count++;
			}
			printf("%d\n",count);
		}
		else {
			count=0;
			int ans=INT_MAX;
			r=al-bl;
			for(i=0;i<r+1;i++) {
				count=0;
				k=i;
				for(j=0;j<bl;j++) {
					if(a[k]!=b[j]) count++;
					k++;
				}
				if(count==0) {
					ans=0;
					break;
				}
				else ans=min(ans,count);
			}
			printf("%d\n",ans);
		}
	}
	return 0;
}
