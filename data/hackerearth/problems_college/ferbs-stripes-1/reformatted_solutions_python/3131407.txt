n = input()
while (n):
    count = 1
    flag = 0
    m = input()
    a = sorted(map(int, raw_input().split()))
    if (m % 2 == 0):
        k = m/2
    else:
        k = m/2+1
    for i in range(len(a)-1):
        if a[i] == a[i+1]:
            count += 1
        else:
            count = 1
        if count > k:
            print("bad luck")
            flag = 1
            break
    if flag == 0:
        print("can do")
    n = n-1
