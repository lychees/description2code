import collections
n = int(input(""))
while n != 0:
    s = raw_input("")
    c = 0
    x = collections.Counter(s)

    k = list(x.values())

    for i in range(len(k)):
        k[i] = 1

    for i in k:
        c = c+i
    if c == 26:
        print("YES")
    else:
        print("NO")
    c = 0
    n = n-1
