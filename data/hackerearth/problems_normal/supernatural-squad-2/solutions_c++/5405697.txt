ll power(ll b, ll exp, ll m)
{
	ll ans = 1; b %= m;
	while (exp){
		if (exp & 1) ans = (ans*b) % m;
		exp >>= 1; b = (b*b) % m;
	}
	return ans;
}

ll dp[202][202];

ll cal(int n, int k)
{
	if (dp[n][k] >= 0)
		return dp[n][k];
	if (n<k)
		return 0;
	if (n<2 * k)
		return dp[n][k] = 1;
	ll res = 1;
	For(i, k, n)
		res += cal(n - i, i);
	return dp[n][k] = res;
}
int main()
{
	int t, n, k;
	memset(dp, -1, sizeof(ll) * 201 * 201);
	scanf("%d", &t);
	while (t--)
	{
		scanf("%d%d", &n, &k);
		cout << cal(n, k) << endl;;
	}
}
