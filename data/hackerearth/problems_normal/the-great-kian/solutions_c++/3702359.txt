#include <bits/stdc++.h>
using namespace std;
int main()
{
    long int t,sum1=0,sum2=0,sum3=0;
    cin>>t;
    long int arr[t];
    for(int i=1;i<=t;i++)
    cin>>arr[i];
    
    for(int i=1;i<=t;)
    {
    	if(i<=t )
     sum1+=arr[i];
     if(i+1 <=t)
     sum2 +=arr[i+1];
    	if(i+2 <=t)
     sum3 +=arr[i+2];
     i=i+3;
    }
    
    cout<<sum1<<" "<<sum2<<" "<<sum3;
    return 0;
}
