#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int a,b;
		scanf("%d%d",&a,&b);
		int p[99999] = {0};
		int k = sqrt(b);
		if(a==1)
		a=2;
		for(int i = 2; i <= k ; i++ )
		{
			for(int j = a; j<=b ; j++)
			{
				if(j==i)
				continue;
				else
				{
					if(j%i == 0)
					p[j] = 1;
				}
			}
		}
		int sum = 0;
		for(int j = a; j <=b; j++)
		{
			if(p[j]==0)
			{
			//	printf("%d\n",j);
				sum = sum  + j;
		}
	}
		printf("%d\n",sum);
}
	return 0;
}
