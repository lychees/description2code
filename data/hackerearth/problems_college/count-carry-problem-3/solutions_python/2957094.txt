testCases = input()

while testCases>0:
	a,b=raw_input().split(" ")
	aList=map(int,list(a))
	bList=map(int,list(b))
	arrLength=len(aList)
	aList.reverse()
	bList.reverse()
	carry=0
	carryCount=0
	for i in range(0,arrLength):
		carry=(carry+aList[i]+bList[i])/10
		if carry==1:
			carryCount+=1
	if carryCount==0:
		print "No carry operation"
	elif carryCount==1:
		print "1 carry operation"
	else:
		print "%s carry operations" %carryCount
		
	testCases-=1
