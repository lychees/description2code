#include<iostream>
#include<cstdio>
#include<map>
using namespace std;
map < long long int , long long int > mp;
long long int coin(long long int n)
{
    if(mp.find(n)==mp.end())
        {
            mp[n]=max(n,coin(n/2)+coin(n/3)+coin(n/4));
            return mp[n];
        }
    else
        return mp[n];
}
int main()
{
    mp[0]=0;
    mp[1]=1;
    mp[2]=2;
    long long int n;
    while(scanf("%lld",&n)==1)
    {
        long long int ans=coin(n);
        printf("%lld\n",ans);
    }
    return 0;
}
