int check(int a,int b)
{
	int s[20],w[20];
	int k=0;
	while(a>0)
	{
		s[k++]=a%10;
		a/=10;
	}
	int p=0;
	while(b>0)
	{
		w[p++]=b%10;
		b/=10;
	}
	for(int i=0;i<k;i++)
	{
		for(int j=0;j<p;j++)
		{
			if(s[i]==w[j])
			{
				return 1;
			}
		}
	}
	return 0;
}

int main()
{
	int t;
	scan(t);
	while(t--)
	{
		int x;
		scan(x);
		int cnt=0;
		
		for(int i=1;i*i<=x;i++)
		{
			if(x%i==0)
			{
				if(check(i,x))
				{
					cnt++;
				}
				if(i!=(x/i)&&check(x/i,x)==1)
				{
					cnt++;
				}
			}
		}
		cout<<cnt<<endl;
	}
}
