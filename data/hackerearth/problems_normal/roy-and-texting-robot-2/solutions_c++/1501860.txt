#include <bits/stdc++.h>

using namespace std;

int main() {
	int n_tests = 0;
	cin >> n_tests;

	while (n_tests--) {
		string msg;
		cin >> msg;

		int cur_btn = 1, nxt_btn = -1;
		int n_moves = 0;
		long long int n_steps = 0;

		for (int i = 0; i < msg.length(); i++) {
			nxt_btn = -1, n_moves = 0;
			
			switch(msg[i]) {
				case '.':
					nxt_btn = 1;
					n_moves = 1;
					break;
				case ',':
					nxt_btn = 1;
					n_moves = 2;
					break;
				case '?':
					nxt_btn = 1;
					n_moves = 3;
					break;
				case '!':
					nxt_btn = 1;
					n_moves = 4;
					break;
				case '_':
					nxt_btn = 0;
					n_moves = 1;
					break;
			}
			if (msg[i] >= '0' && msg[i] <= '9') {
				nxt_btn = msg[i] - '0';
				if (msg[i] == '1' || msg[i] == '7' || msg[i] == '9')
					n_moves = 5;
				else if (msg[i] == '0')
					n_moves = 2;
				else
					n_moves = 4;
			}

			if (msg[i] >= 'a' && msg[i] <= 'z') {
				if (msg[i] <= 'r') {
					nxt_btn = 2 + ((msg[i] - 'a') / 3);
					n_moves = ((msg[i] - 'a' + 1) % 3);
					if (n_moves == 0)
						n_moves = 3;
				}

				else if (msg[i] == 's')
					nxt_btn = 7, n_moves = 4;
				else if (msg[i] >= 't' && msg[i] <= 'v')
					nxt_btn = 8, n_moves = 1 + (msg[i] - 't');
				else
					nxt_btn = 9, n_moves = 1 + (msg[i] - 'w');
			}

			if (cur_btn != nxt_btn)
				cur_btn = nxt_btn, n_steps += 1;
			n_steps += n_moves;
		}
		cout << n_steps << endl;
	}
	return 0;
}
