#include <iostream>
using namespace std;

#define MAX 100000
int main()
{
	int N = 0;
	int k = 0;
	int arr[MAX];
	cin >> N >> k;
	
	for (int i = 0; i < N; i++) {
		cin >> arr[i];
	}
	
	for (int i = 0; i < N; i++) {
		if (k <= 0) break;
		int max = i;
		for (int j = i+1; j < N; j++) {
			if (arr[j] > arr[max])
			{
				max = j;
			}
		}
		if (max != i)
		{
			int temp = arr[i];
			arr[i] = arr[max];
			arr[max] = temp;
			k--;
		}
	}
	
	for (int i = 0; i < N; i++) {
		cout << arr[i] << " ";
	}
	
    return 0;
}
