#include <iostream>
using namespace std;

int main()
{
	int n;
    cin >> n;
    int input[n][2], output[n][2];
    for(int i=0; i<n; i++){
    	output[i][0]=0;
    	output[i][1]=0;
    }

    for(int i=0; i<n; i++){
    	for(int j=0; j<2; j++){
    		cin >> input[i][j];
    	}
    }

    for(int i=0;i<n;i++){
    	for(int j=0; j<n-i-1; j++){
    		if(input[i][0]==input[i+j+1][1]){
    			output[i][0]+=1;
    			output[i+j+1][0]+=1;
    		}
    		else{
    			output[i][0]+=1;
    			output[i+j+1][1]+=1;
    		}
    	}
    }

    for(int i=n-1;i>=0;i--){
        	for(int j=1; j<i+1; j++){
        		if(input[i][0]==input[i-j][1]){
        			output[i][0]+=1;
        			output[i-j][0]+=1;
        		}
        		else{
        			output[i][0]+=1;
        			output[i-j][1]+=1;
        		}
        	}
        }

    for(int i=0; i<n; i++){
    	cout<<output[i][0]<<" "<<output[i][1]<<endl;
    }
}
