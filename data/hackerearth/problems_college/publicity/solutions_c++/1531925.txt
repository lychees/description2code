typedef long long int ll;
typedef pair<int, int > pii;

#define MOD 1000000007
#define pb push_back

bool visited[100001];
list<int> adj[100001];

void dfs(int s)
{
	visited[s] = true;
	
	list<int>::iterator it = adj[s].begin();
	for(;it!=adj[s].end();it++)
	{
		if(!visited[*it])
			dfs(*it);
	}
}

void solve()
{
	for(int i=0;i<100001;i++)
	{
		visited[i] = false;
		adj[i].clear();
	}
	int n, e, u, v;
	cin>>n>>e;
	
	for(int i=0;i<e;i++)
	{
		cin>>u>>v;
		adj[u].push_back(v);
		adj[v].push_back(u);
	}
	int c = 0;
	for(int i=0;i<n;i++)
	{
		if(!visited[i])
		{
			c++;
			dfs(i);
		}
	}
	cout<<c<<endl;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		solve();
	}
	return 0;
}
