#include<iostream>
using namespace std;
#define ll long long 
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll int a,b,sum1=0,sum2=0,c=0;
		cin>>a>>b;
		while(a!=0&&b!=0)
		{
			sum1+=(a%10)+(b%10);
			if(sum1>10)
			{
				c++;
				sum1-=10;
			}
			else if(sum1==10)
			{
				c++;
				sum1=1;
			}
			else
			{
				sum1=0;
			}
			a/=10;b/=10;
		}
		if(c==0)
		{
			cout<<"No carry operation"<<endl;
		}
		else if(c==1)
		{
			cout<<"1 carry operation"<<endl;
		}
		else
		{
		cout<<c<<" carry operations"<<endl;
	}
	}
}
