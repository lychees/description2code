#include <bits/stdc++.h>
using namespace std ;
int rnd( int a , int b )
{
    int x ;
    b = abs( b ) ;
    if( a >= 0 )
    {
        x = a % b ;
        if( 2 * x < b ) return a - a % b ;
        else            return a - a % b + b ;
    }
    else
    {
        x = a % b ;
        x = -x ;
        if( 2 * x < b ) return a - a % b ;
        else            return a - a % b - b ;
    }
}

int main()
{
    int _t ;
    scanf( "%d" , &_t ) ;
    while( _t-- )
    {
        int n , b ;
        cin >> n >> b ;
        cout << rnd( n , b ) << endl ;
    }
    return 0 ;
}
