#include <iostream>
#define ll long long

using namespace std;
const int ma = 1e5+5;

int ar[ma];
bool vis[ma];
int n,k;
void init()
{
	for(int i=1;i<=n;i++)
		ar[i]=i;
}
int root(int x)
{
	while(x!=ar[x])
	{
		ar[x] = ar[ar[x]];
		x = ar[x];
	}
	return x;
}
void connect(int x,int y)
{
	int rx = root(x);
	int ry = root(y);
	ar[ry] = rx;
}
int main()
{

	cin>>n>>k;
	int x,y;
	init();
	for(int i=0;i<k;i++)
	{
		cin>>x>>y;

		connect(x,y);
	}
	int c=0;
	for(int i=1;i<=n;i++)
	{
		int tp = root(ar[i]);
		if(!vis[tp])
		{
			vis[tp]=true;
			c++;
		}
	}
	cout<<c<<endl;
}
