#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
long a[10000],n;
long max(long a,long b)
{
	if(a<b) return b;
	return a;
}
long rec(long s,long i,long c)
{
	long u;
	if(s==c)
	return 0;
	if(i==n)
	return 0;
	if(s+a[i]>c)
	return 0;
	u=max(1+rec(s+a[i],i+1,c),rec(s,i+1,c));
	return u;
	
}
int main()
{
	long i,j,k,t,x,y,p;
	scanf("%ld",&t);
	while(t--)
	{
		scanf("%ld %ld %ld",&x,&y,&n);
		for(i=0;i<n;i++)
		{
			scanf("%ld",&a[i]);
		}
		sort(a,a+n);
		j=0;i=0;p=0;
		while(p+a[i]<=x&&i<n)
		{
			p+=a[i];
			i++;
			j++;
		}
		i=0;p=0;k=0;
		while(p+a[i]<=y&&i<n)
		{
			p+=a[i];
			i++;
			k++;
		}
		if(j>k)
		printf("Chef Won\n");
		else if(k>j)
		printf("Jhool Won\n");
		else 
		printf("Tie\n");
	}
	return 0;
}
