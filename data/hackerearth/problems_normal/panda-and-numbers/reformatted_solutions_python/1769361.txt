from math import *


def digits(x):
    A = set()
    while (x > 0):
        A.add(x % 10)
        x /= 10
    return A


M = 1000001
A = [0]*M
i = 1
p = 1
while (p < M):
    A[int(p)] = 1
    i += 1
    p = pow(i, i)
A[13] = 1
for x in xrange(14, M):
    D = digits(x)
    for d in D:
        if (A[x-(d*d)] == 1):
            A[x] = 1
            break
T = int(raw_input())
for _ in xrange(T):
    N = int(raw_input())
    if (A[N] == 1):
        print "Yes"
    else:
        print "No"
