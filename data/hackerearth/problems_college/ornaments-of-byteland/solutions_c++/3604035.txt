#include <iostream>
#include<cstring>
using namespace std;

int main()
{	int t,i,j,flag=0;
	char ch[1000];
	int a[1000];
	for(i=0;i<1000;i++)
	a[i]=0;
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
		cin>>ch;
		for(j=0;j<strlen(ch);j++)
		{
			if(a[ch[j]-'a']==i)
			{
				a[ch[j]-'a']++;
				if(a[ch[j]-'a']==t)
				flag++;
			}
		}
	}
	cout<<flag;
    return 0;
}
