#include <cstdio>
#include <queue>
#include <algorithm>

#define MAX 1003
#define pi pair<int, int>

using namespace std;

int di[] = { 0, 1, 0, -1 };
int dj[] = { 1, 0, -1, 0 };


bool graph[MAX][MAX];
queue<pi> q;


int main()
{
	int n, m, queries, u, v;
	scanf("%d%d%d", &n, &m, &queries);

	int ans = 0;

	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			scanf("%d", &graph[i][j]);
			if (graph[i][j]) ans++;
		}
	}

	for (int i = 0; i < queries; i++)
	{
		scanf("%d%d", &u, &v);

		int des = 0;
		int x = u-1, y = v-1;

		if (graph[x][y])
        {
            q.push(pi(x, y));
            //visited[x][y] = true;
            //graph[x][y] = 0;

            while (!q.empty())
            {
                pi t = q.front();
                q.pop();
                u = t.first, v = t.second;

                if (graph[u][v])
                {
                    graph[u][v] = 0;
                    des++;
                    for (int k = 0; k < 4; k++)
                    {
                        int nx = u + di[k];
                        int ny = v + dj[k];

                        if (nx >= 0 && nx < n && ny >= 0 && ny < m && graph[nx][ny])
                            q.push(pi(nx, ny));
                    }
                }
            }
        }

		ans -= des;
		printf("%d\n", ans);
	}

	return 0;
}
