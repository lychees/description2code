using namespace std;
typedef long long int llt;
vector< vector<int> > v;
vector<int> reqStr;

void checkStr(int i,int n,bool &found)
{
	if(i==n)
		return ;
	bool allFound = true ,canInclude = true;
	if(!found)
	{
		for(int j = 0 ; j < 26 && canInclude; j++)
		{
			if(reqStr[j] - v[i][j] < 0)
				canInclude = false ;
		}

		if(canInclude)
		{
			for(int j = 0 ; j < 26; j++)
			{
				reqStr[j] -= v[i][j];
				if(reqStr[j] > 0)
					allFound = false ;
			}	
			if(allFound)
				found = true ;
			else
			{
				checkStr(i+1,n,found);
			}
		}	
		if(!found)
		{
			if(canInclude)
			{
				for(int j = 0 ; j < 26 ; j++)
					reqStr[j] += v[i][j];
			}
			checkStr(i+1,n,found);
		}
	}
	
}

int main()
{
	ios::sync_with_stdio(false);
//	f_in("input.txt");
	int t,n;
	bool found ;
	string s,rs;
	cin>>t;
	while(t--)
	{
		found = false ;
		reqStr.assign(26,0);
		cin>>n;
		v.assign(n,vector<int>(26,0));
		for(int i = 0 ; i < n ; i++)
		{
			cin>>s;
			for(int j = 0 ; j < s.length() ; j++)
				v[i][s[j]-'a']++;
			s.clear();
		}
		cin>>rs;
		for(int j = 0 ; j < rs.length() ; j++)
				reqStr[rs[j]-'a']++;

		checkStr(0,n,found);
		if(found)
			cout<<"YES";
		else
			cout<<"NO";
		cout<<"\n";
		v.clear();
		reqStr.clear();
		rs.clear();
	}

	return 0;
}
