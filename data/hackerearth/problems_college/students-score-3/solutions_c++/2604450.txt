#include <iostream>
using namespace std;

typedef long long int ll;

int main() {
	ll n;
	cin >> n;
	ll arr[n];
	for(ll i=0;i<n;i++) {
		cin >> arr[i];
	}
	ll temp = 0;
	ll curr = 0;
	for(ll i=0;i<n;i++) {
		temp += arr[i];
		if(temp < 0) {
			temp = 0;
		}
		if(temp > curr) {
			curr = temp;
		}
	}
	cout << curr << endl;
	return 0;
}
