#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,m,i;
        cin>>n>>m;
        string s;
        map<string,int>mp;
        for(i=0;i<n;i++)
        {
            cin>>s;
            mp[s]++;
        }
        int flag=0;
        for(auto it=mp.begin();it!=mp.end();it++)
            {
                if(it->second>1)
                   flag=1;
            }
            if(flag==1)
                cout<<"No"<<endl;
            else
                cout<<"Yes"<<endl;
    }
}
