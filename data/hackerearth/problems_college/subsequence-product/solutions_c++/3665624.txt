#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;
		cin>>n>>m;
		int a[n];
		long long int pro = 1;
		int k = 0;
		for( int i = 0; i< n ; i++ )
		{
			cin>>a[i];
		}
		for( int i = 0; i<n ; i++)
		{
			pro =( (pro%m) * (a[i] %m) )%m;
			if( pro == 0 )
			{
				k = 1;
				cout<<"YES"<<"\n";
				break;
			}	
		}
		if( k == 0)
		{
			cout<<"NO"<<"\n";
		}
	}
}
