#include<stdio.h>
#include<string>
#include<iostream>
#include<string.h>
#include<map>
#include<utility>
#include<vector>
#define pp pair<int,int>
#define mod 10000007

using namespace std;

vector<int> l,s;
int n,x,x1,x2,sz;
long long int dp[1005][2050];

long long int find(int i,int xr)
{
    if(i==n)
    {
        if(xr==x2)
            return 1;
        else
            return 0;
    }
    if(dp[i][xr]!=-1)
        return dp[i][xr];
    long long int a,b;
    a=find(i+1,xr);
    b=find(i+1,xr^s[i]);
    return dp[i][xr]=(a+b)%mod;
}

long long int solve(int i,int xr)
{
    if(i==sz)
    {
        if((xr>>10)==x1)
            return find(0,(x1<<10)^xr);
        return 0;
    }
    long long int a,b;
    a=solve(i+1,xr);
    b=solve(i+1,xr^l[i]);
    a=(a+b)%mod;
    return a;
}

int main()
{
    cin>>n>>x;
    x1=(x>>10);
    x2=(x1<<10)^x;
    //cout<<x1<<" "<<x2;
    memset(dp,-1,sizeof(dp));
    int i,k;
    i=0;
    while(i<n)
    {
        cin>>k;
        if(k<1024)
            s.push_back(k);
        else
            l.push_back(k);
        i++;
    }
    sz=l.size();
    long long int ans;
    n=s.size();
    //cout<<sz<<" "<<n;
    ans=solve(0,0);
    cout<<ans<<endl;
    return 0;
}
