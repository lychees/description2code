int main ()
{
  int n;

      cin >> numerator >> denominator >> r;
           result= answer(numerator,denominator,r);

int answer(int n,int d,int r)
{
    int rem=n;
    int digit=0;
    for(int i=0;i<r;i++)
    {
         if(rem==0) return rem;
         digit=((10*rem)/d);
         rem=((10*rem)%d);
    }
    return digit;
}
