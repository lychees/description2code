#include <iostream>
using namespace std;
typedef long long int ll;
ll arr[1000001];
ll a[1000000];
int main()
{
    ll max = 1000000;
    ll i, j;
    for(i = 2; i*i <= max; i++) {
    	if(arr[i] == 0) {
    		for(j = i * i; j <= max; j+=i) {
    			arr[j] = 1;
    		}
    	}
    }
    ll c = 0;
    for(i = 2; i <= max; i++) {
    	if(arr[i] == 0) {
    		a[c++] = i;
    	}
    }
    ll t;
    cin >> t;
    while(t--) {
    	ll n, c = 0;
    	cin >> n;
    	ll k = 0;
    	if(arr[n] == 0) {
    		cout << "2^0";
    	}
    	else {
    		while(n % a[c] == 0) {
    			k++;
    			n/=a[c];
    		}
    		cout << "2^" << k;
    		c++;
    		while(n > 1) {
    			if(arr[n] == 0) {
    				cout << "*" <<  n << "^1";
    				break;
    			}
    			else {
    				k = 0;
    				while(n % a[c] == 0) {
    					k++;
    					n/=a[c];
    				}
    				if(k != 0) {
    					cout << "*" << a[c] << "^" << k;
    				}
    				c++;
    			}
    		}
    	}
    	cout << endl;
    }
    return 0;
}
