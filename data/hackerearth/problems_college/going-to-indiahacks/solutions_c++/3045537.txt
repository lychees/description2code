#include <bits/stdc++.h>
#define M 1000000007

using namespace std;

long long dp[1005][1005];

int main() {
    long long a, b, c, d, t;

    long long nop1, nop2, nop3;

    for (int i = 1004; i >= 1; i--) {
        for (int j = 1004; j >= 1; j--) {
            if (i + 1 <= 1004) nop1 = dp[i + 1][j];
            else nop1 = 0;
            if (j + 1 <= 1004) nop2 = dp[i][j + 1];
            else nop2 = 0;
            if (i + 1 <= 1004 && j + 1 <= 1004) nop3 = dp[i + 1][j + 1];
            else nop3 = 0;
            dp[i][j] = (nop1 + nop2 + nop3) % M;
            if (i == 1004 && j == 1004) dp[i][j] = 1;
        }
    }
    
    cin >> t;

    while (t--) {
        cin >> a >> b >> c >> d;

        cout << dp[1004 + a - c][1004 + b - d] << endl;
    }

    return 0;
}
