#include<bits/stdc++.h>
using namespace std;
int main()
{
    int L,N;
    scanf("%d%d",&L,&N);
    string P,T;
    P="..";
    cin>>T;
    int F[2];
    F[0]=0;
    F[1]=1;
    int i=0,j=0,ans=0;
    while(i<T.size())
    {
        if(P[j]==T[i])
        {
            j++;
            i++;
        }
        else
        {
            if(j>0)
            j=F[j-1];
            else
                i++;
        }
        if(j==(2))
        {
            ans++;
            i=i-F[j-1];
            j=0;
        }
    }
    int k;
    char ch;
    while(N--)
    {
        scanf("%d",&k);
        cin>>ch;
        if(T[k-1]!='.')
        {
            if(ch=='.')
            {
                if(k>1&&T[k-2]=='.')
                    ans++;
                if(k<L&&T[k]=='.')
                    ans++;
            }
        }
        else if(T[k-1]=='.')
        {
            if(ch!='.')
            {
                if(k>1&&T[k-2]=='.')
                    ans--;
                if(k<L&&T[k]=='.')
                    ans--;
            }
        }
        T[k-1]=ch;
        printf("%d\n",ans);
    }
    return 0;
}
