#include <iostream>
#include<algorithm>
using namespace std;

int x[100003];
int y[100003];

int main()
{
    //cout << "Hello World!" << endl;
    int n,k,a,b,c,d;
    long long int ans = 0;
    cin>>n>>k;
    for(int i=0;i<k;i++)cin>>x[i];
    for(int i=0;i<k;i++)cin>>y[i];
    for(int i=0;i<k;i++)
    {
    	a = abs(x[i]-1) + abs(y[i]-1);
    	b = abs(x[i]-1) + abs(y[i]-n);
    	c = abs(x[i]-n) + abs(y[i]-1);
    	d = abs(x[i]-n) + abs(y[i]-n);
    	ans = ans + min(a,min(b,min(c,d))); 
    //	cout<<ans<<" "; 
    }
    //
    cout<<endl<<endl;
    cout<<ans;
    return 0;
}
