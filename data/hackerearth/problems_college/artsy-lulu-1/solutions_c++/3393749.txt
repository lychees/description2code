#include <iostream>
#include <sstream>
#include <ctype.h>
using namespace std;

bool symbCheck(char x, int y)
{
	bool flag;
	for(int i = 0; i < y; i++)
	{
		if(isalpha(static_cast<char>(x + i)))
		{
			flag = false;
		}
		else
		{
			return true;
		}
	}
	return flag;
}
int main()
{
	string tmp;
    while(getline(cin, tmp))
    {
    	char stChr;
    	int stInt;
    	istringstream fin(tmp);
    	fin >> stChr >> stInt;
    	if(!symbCheck(stChr, stInt))
    	{
	    	if(stInt >= 2)
	    	{
	    		for(int i = 0; i < stInt; i++)
	    		{
	    			int j = 0;
	    			while(j < i + 1)
	    			{
	    				cout << static_cast<char>(stChr + i) << " ";
	    				j++;
	    			}
	    			cout << endl;
	    		}
	    	}
	    	else
	    	{
	    		cout << "INVALID" << endl;
	    	}
    	}
    	else
    	{
    		cout << "INVALID" << endl;
    	}
    	cout << endl;
    }
}
