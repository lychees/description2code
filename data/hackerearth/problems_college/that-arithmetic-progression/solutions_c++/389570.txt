#include<stdio.h>
#include<iostream>
#include<stdlib.h>
#include<algorithm>
using namespace std;
main()
{
    int n, a[100009],i,d,e,j;
    cin>>n;
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    sort(a,a+n);
    if(n==1)
    cout<<"-1\n";
    else if(n==2)
    {
        d=a[1]-a[0];
        if(d==0)
        cout<<"1"<<"\n"<<a[0]<<"\n";
        else if(d%2)
        {
            cout<<"2\n";
            cout<<a[0]-d<<" "<<a[1]+d<<"\n" ;
        }
        else
        {
            cout<<"3\n";
            cout<<a[0]-d<<" "<<a[0]+(d/2)<<" "<<a[1]+d<<"\n" ;
        }
    }
    else
    {
        d=a[1]-a[0];
        e=a[n-1]-a[n-2];
        if(d>e)
        d=e;
        for(i=0;i<n-1;i++)
        {
            if((a[i+1]-a[i])!=d)
            break;
        }
        if(d==0)
        cout<<"1"<<"\n"<<a[0]<<"\n";
        else if(i!=n-1)
        {
            j=n-1;
            while(j>i)
            {
                a[j+1]=a[j];
                j--;
            }
            a[j+1]=a[i]+d;
            for(i=0;i<n;i++)
            {
                if((a[i+1]-a[i])!=d)
                break;
            }
            if(i!=n)
            cout<<"0\n";
            else
            cout<<"1"<<"\n"<<a[j+1]<<"\n";
            
        }
        else
        cout<<"2"<<"\n"<<a[0]-d<<" "<<a[n-1]+d<<"\n";
    }
    //system("pause");
}
