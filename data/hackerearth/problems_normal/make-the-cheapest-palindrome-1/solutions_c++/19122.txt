#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
inline bool is_palin(char str[]){
	int i=0,j=strlen(str)-1;
	while(i<j){
		if(str[i]!=str[j])return false;
		i++,j--;
	}
	return true;
}
int main()
{
	int t;
	cin>>t;
	while(t--){
		char str[10001];
		scanf("%s",str);
		int ca,cb,m=0;
		cin>>ca>>cb;
		int i=0,j=strlen(str)-1;
		while(i<j){
			if(str[i]=='/'&&str[j]=='/')m+=2*min(ca,cb);
			else{
				if(str[i]=='/'){
					if(str[j]=='a'){
						str[i]='a';
						m+=ca;
					}
					else {
						str[i]='b';
						m+=cb;
					}
				}
				if(str[j]=='/'){
					if(str[i]=='a'){
						str[j]='a';
						m+=ca;
					}
					else {
						str[j]='b';
						m+=cb;
					}
				}
				
			}
			i++,j--;
		}
		if(is_palin(str))
		cout<<m<<endl;
		else cout<<"-1"<<endl;
	}
	return 0;
}
