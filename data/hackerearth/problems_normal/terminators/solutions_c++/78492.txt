#include <iostream>
#include <vector>

#include <algorithm>

using namespace std;

int main()
{
    int t;

    cin >> t;

    while(t--) {
        int n;

        cin >> n;
        vector <pair<int, int> > v(n);
        for(int i = 0; i < n; i++) {
            cin >> v[i].first >> v[i].second;
        }

        sort(v.begin(), v.end());
    int ans = 1;
    int dp[n+1];
        for(int i = 0; i < n; i++) {
            dp[i] = 1;
            for(int j = 0; j < i; j++) {
                if(v[j].second < v[i].second) {
                    dp[i] = max(dp[i], 1+dp[j]);
                }
            }
            ans = max(ans, dp[i]);
        }

        cout << ans << endl;
    }
}
