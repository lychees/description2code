#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	long long int i;
    	i=0;
    	while(s[i]!=NULL)
    	{
    		if (s[i]=='2')
    		cout<<"cde";
    		else if(s[i]=='6')
    		cout<<"b3";
    		else if(s[i]=='7')
    		cout<<"acf";
    		else if (s[i]=='9')
    		cout<<"c6a";
    		i++;
    	}
    	cout<<endl;
    }
    return 0;
}
