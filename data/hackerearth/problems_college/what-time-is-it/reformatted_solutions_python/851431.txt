hours, minutes = map(float, raw_input().split(':'))
degree_per_min = 6.000000
hours += minutes/60.0
minutes %= 60.0
hour_pos_in_min = (hours % 12.0) * 5.0
out = abs(hour_pos_in_min - minutes) * degree_per_min
if out > 180:
    out = out - 180.0
print "%.6f" % (out)
