#include <iostream>
#include <cstdio>
using namespace std;
 
int main() {
	// your code goes here
	int c,t;
	long x,i,j,n;
	scanf("%d", &t);
	while(t--) 
	{
		int k=0;
		scanf("%ld", &n);
		if(n%3==0) 
		{
			for(i=0;i<n;i++)printf("5");
			k=1;
		}
		if(n%5==0) 
		{
			for(i=0;i<n;i++)printf("3");
			k=1;
		}
		x=n;
		if(!k)
		{
			c=0;
			while(x>0) 
			{ 
				x-=5; 
				if(x%3==0) 
				{ 
					for(i=0;i<x;i++)printf("5");
					for(j=0;j<n-i;j++)printf("3");
					c=1;
					break;
				} 
			}
			
			if(n%5!=0&&!c) printf("-1");
		}
		
		printf("\n");
	}
	return 0;
}
