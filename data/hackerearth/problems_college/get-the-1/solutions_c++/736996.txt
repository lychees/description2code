#include <iostream>
using namespace std;
long int calc(long int a)
{
	int k;
	if (a/10==0)
	 return a;
	long int sum = 0;
	while (a>0)
	{
		k = a%10;
		sum += (k*k);
		a/=10;
	}
	return calc(sum);
}
int main()
{
	int tc = 0;
	long int a = 0;
	cin>>tc;
	
	for(int i=0;i<tc;i++)
	{
		cin>>a;
		if(calc(a)==1)
		 cout<<"Case #"<<i+1<<":YES\n";
		else
		 cout<<"Case #"<<i+1<<":NO\n";
	}
    return 0;
}
