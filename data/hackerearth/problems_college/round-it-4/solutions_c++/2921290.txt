#include <iostream>
using namespace std;
 
int main()
{
   int t,n,b;
   cin>>t;
   while(t--)
   {
   	 cin>>n>>b;
   	 if(n==b)
   	 cout<<n;
   	 else if(n>b)
   	 {
   	 	int bcopy,ncopy,out,out1=0;
   	 	bcopy=b;
   	 	ncopy=n;
   	 	int i=1;
   	 	while(b<n)
   	 	{
   	 		b=bcopy*i;
   	 		i++;
   	 	}
   	 	if(b>n)
   	 	b=b-bcopy;
   	 	out=b;
   	 	
   	 	if(b==n)
   	 	 cout<<b<<endl;
   	 	else
   	    {
   	 		while(ncopy%bcopy!=0)
   	 		ncopy++;
   	 		out1=ncopy;
   	 	  if((n-out)<(out1-n))
   	 	   cout<<out<<endl;
   	 	   else if((n-out)>=(out1-n))
   	 	   cout<<out1<<endl;
   	 	   
   	 	   
   	 	   
   	 	   //cout<<out<<" "<<out1<<endl;
   	    }
   	 }
   	 else
   	 {
   	 	int rem=b/n;
   	 	int out=n*rem;
   	 	int out1=out+n;
   	 	if((n*rem)%b==0)
   	 	cout<<n*rem<<endl;
   	 	else
   	 	cout<<0<<endl;
   	 }
   	 
   }
}
