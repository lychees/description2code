#include <bits/stdc++.h>
#define ll long long int
#define ff fflush(stdout)
#define fb(i,n) for(i=0;i<n;i++)
#define sd(a) scanf("%d",&a)
#define sld(a) scanf("%lld",&a)
#define ss(a) scanf("%s",a)
#define pd(a) printf("%d",a)
#define pld(a) printf("%lld ",a)
#define ps(a) printf("%s",a)
#define pn() printf("\n")
#define w(t) while(t--)
#define mod 1000000007
#define PI 3.141592653589793

using namespace std;

int notprime[10000000];

int main()
{
    ll i,j;
    notprime[0]=1;
    notprime[1]=1;
    for(i=2;i<=sqrt(10000000);i++)
    {
        if(!notprime[i])
        {
            j=i*i;
            while(j<=1e7)
            {
                notprime[j]=1;
                j+=i;
            }
        }
    }
    //cout<<notprime[45553];
    ll t;sld(t);
    w(t)
    {
        ll n,i,j;
        sld(n);
        i=n/2;
        j=n-i;
        if(i==j)
        {
            i--;j++;
        }
        while(i!=1&&j<n)
        {
            if(!notprime[i]&&(!notprime[j]))
                break;
            i--;j++;
        }
        if(i!=1&&j<n)
            cout<<i<<" "<<j<<endl;
        else
        {
            cout<<"-1 -1"<<endl;
        }
    }
    return 0;
}
