#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define m 1000000

ll ma(ll a,ll b)
{
    if(a>=b)
        return a;
    return b;
}

ll dp[m];

void calc()
{
    dp[0]=0;
    for(ll i=0;i<m;i++)
       dp[i]=ma(i,dp[i/2]+dp[i/3]+dp[i/4]);
}

ll res(ll a)
{
    if(a<m)return dp[a];
    return ma(a,res(a/2)+res(a/3)+res(a/4));
}

int main()
{
  ll n,i,c;
  calc();

  while(cin>>n)
        printf("%lld\n",res(n));
return 0;
}
