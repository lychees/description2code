#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    int t,n,p,i;
    double ans;
    cin>>t;
    while(t--){
    	cin>>n;
    	ans=0;
    	for(i=0;i<n;i++){
    		cin>>p;
    		ans+=p;
    	}
    	cout<<setprecision (8)<<fixed<<ans/n<<'\n';
    }
    return 0;
}
