vector<long> v;
void isprime(long a)
{
	vector<long> v1;

	{
		int k=v.size();
		int j;
		for( j=0;j<k;j++)
		{
			if(i%v[j]==0)
			break;
		}
		if(j==k)
		v.push_back(i);
	}
	}
