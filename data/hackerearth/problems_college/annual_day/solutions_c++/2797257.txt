#include <bits/stdc++.h>

typedef long long int ll;

using namespace std;

ll b(ll n)
{
	ll i=0,s=0;
	
	while(n)
	{
		s+=pow(10,i)*(n%2);
		
		n/=2;
		
		i++;
	}
	
	return s;
}

ll cs(ll n)
{
	int c=0;
	
	while(n)
	{
		if(n%10)
		{
			c++;
		}
		
		n/=10;
	}
	
	return c;
}

main()
{
	int t,n,k,x,i,s;
	
	cin>>t;
	
	while(t--)
	{
		cin>>n>>k;
		
		int a[n];
		
		for(i=0;i<n;i++)
		{
			cin>>x;
			
			a[i]=cs(b(x));
		}
		
		sort(a,a+n);
		
		s=0;
		
		for(i=n-1;i>n-k-1;i--)
		{
			s+=a[i];
		}
		
		cout<<s<<endl;
	}
}
