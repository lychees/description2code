#include <bits/stdc++.h>

using namespace std;

#define rep(i, a, b) for(int i = (a); i < int(b); ++i)
#define rrep(i, a, b) for(int i = (a) - 1; i >= int(b); --i)
#define trav(it, v) for(typeof((v).begin()) it = (v).begin(); it != (v).end(); ++it)
#define all(v) (v).begin(), (v).end()

typedef double fl;
typedef long long ll;
typedef pair<int, int> pii;
typedef vector<int> vi;
typedef vector<pii> vpi;

char s[100010];

void solve(){
	scanf("%s", s);
	int len=strlen(s);
	int ind1=-1;
	int ind2=-1;
	bool ok =1;
	for(int i=0; 2*i < len; ++i){
		if(s[i+1] == s[len-1-i] && ok){
			ind1=i;
		}
		if(s[i] == s[len-2-i] && ok){
			ind2=len-1-i;
		}
		if(s[i] != s[len-1-i]){
			ok=0;
		}
		if(s[i+1] != s[len-1-i])
			ind1=-1;
		if(s[i] != s[len-2-i])
			ind2=-1;
	}
	if(ok)
		printf("-1\n");
	else if(ind1 != -1)
		printf("%d\n", ind1);
	else
		printf("%d\n", ind2);
}

int main(){
	int T;
	scanf("%d", &T);
	while(T--)
		solve();
}
