#include<bits/stdc++.h>
using namespace std;
int minn,cur;
int main()
{
    int t,l;
    scanf("%d",&t);
    while(t--)
    {
       int i,t1,a,b,q;
       cin>>l;
       minn=l;
       if(l==1) minn=0;
       for(i=1;i<=l/2;++i)
       {
        cur=1;
        a=i;
        b=l-i;
         while(1)
         {
          q=(b-1)/a;
          if(q==0)
            break;
           cur+=q;
           b=b-q*a;
           a^=b^=a^=b;
           if(cur>minn)
           break;
         }
         if(cur<minn && a==1 && b==1)
          minn=cur;
        }
        cout<<minn<<endl;
    }
return 0;
}
