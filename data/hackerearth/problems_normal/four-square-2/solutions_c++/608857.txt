#include<bits/stdc++.h>
using namespace std;


long long int sum1=0,sum2=0;


map < int , int > ma;
map < int , int > maa;



void main3(long long int n)
{
	long long int d= sqrt (n) +1;
	for(int i=1;i<=d;i++)
	{
		if(n%i==0)
		{
			if (i%2!=0)
			{
				if(ma.find(i) == ma.end())
				{
					sum1+=i;
					ma[i]=1;
				}
			}
			if ((n/i) %2 !=0)
			{
				if(ma.find(n/i) == ma.end())
				{
					sum1+=n/i;
					ma[n/i]=1;
				}
			}
			if(maa.find(i) == maa.end())
			{
				sum2+=i;
				maa[i]=1;
			}
			if(maa.find(n/i) == maa.end())
			{
				sum2+=n/i;
				maa[n/i]=1;
			}

		}
	}
}



int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		
		sum1=0;
		sum2=0;
		ma.clear();
		maa.clear();
		long long int x;
		cin>>x;
		long long int ans;
		if(x%2==0)
		{
			main3(x);	
			sum1*=24;
			ans=sum1;
		}
		else
		{
			main3(x);
			sum2*=8;
			ans=sum2;
		}
		if(x==0)
			ans=1;
		cout<<ans<<endl;
	}
	return 0;
}
