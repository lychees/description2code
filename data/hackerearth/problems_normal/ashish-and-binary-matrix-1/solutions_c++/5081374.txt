#include <bits/stdc++.h>
using namespace std;

void init()
{
    int n,m;
    cin>>n>>m;
    bool baby = false;
    map <string,long int> mp;
    for(int i=0;i<n;i++)
    {
        string s;
        cin>>s;
        if(++mp[s]==2)
            baby = true;
    }
    if(baby)
        cout<<"No\n";
    else
        cout<<"Yes\n";
}

int main()
{
    int t;
    cin>>t;
    while(t--)
        init();
    return 0;
}
