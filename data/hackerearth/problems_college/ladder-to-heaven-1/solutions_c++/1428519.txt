#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;
typedef long long int ll;

int main()
{
  int t;
  scanf("%d",&t);
  while(t--)
  {
      ll M;
      scanf("%lld",&M);
      ll fibo[] = {2,0};
      int i = 0;
      ll summed = 0;
      while(fibo[i] < M)
      {
          summed += fibo[i];
          i = (i + 1)&1;
          fibo[i] = 4*fibo[(i + 1)&1] + fibo[i];
      }
      printf("%lld\n",summed);
  }
  return 0;
}
