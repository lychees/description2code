# include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int a[5000],i,n,k=4999,j,r,z,y,rem=0,c1=0;
		for(i=0;i<4999;++i)
		a[i]=0;
		a[4999]=1;
		cin>>n;
		if(n==0)
		cout<<"0\n";
		else if(n==1)
		cout<<"0\n";
		else
		{
			for(i=n;i>1;--i)
			{
				rem=0;
				for(j=4999;j>=k;--j)
				{
					z=a[j];
					y=z*i+rem;
					a[j]=y%10;
					rem=y/10;
				}
				while(rem)
				{
					a[j--]=rem%10;
					rem/=10;
				}
				k=j;
			}
			while(a[j]==0)
			j++;
			for(i=j;i<5000;++i)
			if(a[i]==4||a[i]==7)
			c1++;
			cout<<c1<<endl;
		}
	}
	return 0;
}
