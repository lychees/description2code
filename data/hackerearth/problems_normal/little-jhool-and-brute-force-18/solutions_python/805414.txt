n = int(704800**(1/3.0))
lst = [ i**3 for i in xrange(n+1)]
hmap = [0]*704801
itr = 1;
while(itr<n):
    j=itr+1
    while(j<=n):
        a = lst[itr]+lst[j]
        if a<704801:
        	hmap[a] = hmap[a]+1
        j += 1
    itr+=1
for _ in xrange(input()):
    x = input()
    flag = 0
    for itr1 in xrange(x-1,9,-1):
        if hmap[itr1]>1:
            print itr1
            flag = 1
            break
    if not flag:
        print '-1'
