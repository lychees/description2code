#include <bits/stdc++.h>

using namespace std;
int gcd(int n1,int n2){

	while(n1!=n2)
    {
        if(n1 > n2)
            n1 -= n2;
        else
            n2 -= n1;
    }
    return n1;
}

	int n;

	vector<int> a(n);
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	int count=0;

	for(int i=0;i<n;i++){
		for(int j=i+1;j<n;j++){
			if(a[i]>a[j]){
				if(gcd(a[i],a[j])==1)
					count++;
			}
		}
	}

	cout<<count<<"\n";




	return(0);
