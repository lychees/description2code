#include <iostream>
#include<string.h>
using namespace std;
#define max 101
int main()
{
    char s[max];
    cin>>s;
    int len=strlen(s);
    for(int i=0;i<len;i++)
    {
    	if(s[i]>=65 && s[i]<=90)
    	s[i]+=32;
         else if(s[i]>=97 && s[i]<=122)
    	s[i]-=32;
    }
    cout<<s;
    return 0;
}
