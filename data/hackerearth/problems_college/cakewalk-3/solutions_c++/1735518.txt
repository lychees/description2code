#include <iostream>
#include <string>
using namespace std;
int main(){
	ios::sync_with_stdio(false);
	int T=0;
	
	cin >> T;
	cin.ignore();
	
	T--;
	string res="";
	cin >> res;
	
	while(T--)
	{
	 int lim=0;
	 string con="";
	 cin >> lim >> con;
	 string newStr="";
	 
	 for(int i=0;i<lim;i++)
	  newStr+=res[i];
	 
	 newStr+=con;
	 res=newStr;
	}
	
	cout << res << endl;
	
	return 0;
}
