#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--){
		string str;
		cin >> str;
		int cnt=0;
		int l=str.size();
		for(int i=0;i<l-1;){
			if(str[i]==str[i+1]){
				cnt++;
				l--;
				for(int j=i+1;j<l;j++)
				str[j]=str[j+1];
			}
			else 
			i++;

		}
for(int i=0;i<l;i++)
		cout << str[i];
		cout << "\n";
	}
    return 0;
}
