#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    ll n;
    cin >> n;
    if(!n)
    	cout << 0;
    else
    {
    ll a[n];
    for(ll i = 0; i < n; i++)
    {
    	cin >> a[i];
    }
    ll mxs = a[0], mxhere = a[0];
    for(ll i = 1; i < n; i++)
    {
    	if(a[i] + mxhere < a[i])
    	{
    	
    	
    			mxhere = a[i];
    	}
    	else
    	{
    		mxhere += a[i];
    			mxs = max(mxs, mxhere);
    	}
    }
    if(mxs > 0)
    	cout << mxs;
    else
    	cout << 0;
    }
    return 0;
}
