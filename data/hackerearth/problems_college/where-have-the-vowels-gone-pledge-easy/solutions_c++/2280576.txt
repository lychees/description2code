#include<cstdio>
#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int main()
{
	int Num,cnt;
	bool k;
	string inp;
	scanf("%d",&Num);
	cin>>inp;
	k=false;
	sort(inp.begin(),inp.end());
	if(binary_search(inp.begin(),inp.end(),'a'))
	{
		if(binary_search(inp.begin(),inp.end(),'e'))
		{
			if(binary_search(inp.begin(),inp.end(),'i'))
			{
				if(binary_search(inp.begin(),inp.end(),'o'))
				{
					if(binary_search(inp.begin(),inp.end(),'u'))
						k=true;
				}
			}
		}
	}
	if(k)
		printf("YES\n");
	else
		printf("NO\n");
	return 0;
}
