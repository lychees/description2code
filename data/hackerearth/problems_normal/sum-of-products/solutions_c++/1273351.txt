#include <cstdio>

#define ll long long
#define MOD 1000000007

ll f(ll b, ll a, ll d){
    return (((((b - a + 1) * (a + b)) / 2) % MOD) * d) % MOD;
}

int main(){
    int t, n;
    
	scanf("%d", &t);
	while (t--){
		scanf("%d",&n);
		
		int res = 0, d = 1, x = n, y = n;
        while(y > 0){
        	d = n / y;
            y = n / (d + 1); 
            res = (res + f(x, y + 1, d)) % MOD;
            x = y;
			d = n / x;
        }
        printf("%d\n",res);
    }
    
    return 0;
}
