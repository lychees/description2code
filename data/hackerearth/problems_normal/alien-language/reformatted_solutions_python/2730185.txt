'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
inp = int(raw_input())
for i in range(inp):
    jh = raw_input()
    ja = raw_input()
    y = 0
    for i in ja:
        if i in jh:
            y = y+1
    if y != 0:
        print("YES")
    else:
        print("NO")
