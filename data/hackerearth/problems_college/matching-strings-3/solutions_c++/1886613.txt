#include <string>
#include <iostream>
#include <cmath>
#include <math.h>

using namespace std;

int main()
{
	int testCases;
	scanf("%d", &testCases);
	for(int c=0;c<testCases;c++)
	{
		string a;
		string b;
		cin>>a;
		cin>>b;
		int count = 0;
		for(int i=0; i<min(a.length(), b.length()); i++)
			if(a[i] == b[i])
				count++;
		printf("%d\n", count);
	}
	return 0;
}
