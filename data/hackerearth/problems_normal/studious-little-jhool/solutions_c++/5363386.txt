#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
		map <ll, ll> m;
    ll t;
    ll p;
    	 
	for(ll i = 0; i <= 12; i++)
	{
		for(ll j = 0; j <= 12; j++)
		{
			p = 10 * i + 12 * j;
			m.insert(make_pair(p, i + j));
			
		}
	}
    cin >> t;
    while(t--)
    {
    	ll n;
    	cin >> n;
		if(m[n])
			cout << m[n] << endl;
		else
			cout << -1 << endl;
    	
    }
    return 0;
}
