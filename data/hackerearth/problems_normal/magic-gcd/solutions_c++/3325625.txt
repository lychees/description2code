#include<bits/stdc++.h>
using namespace std;

long long int gcd(long long int a,long long int b)
{
	if(a==0)
	return b;
	else 
	return gcd(b%a,a);	
}

long long int max(long long int a,long long int b)
{
	return a>b?a:b;
}

int main()
{
	int t;
	cin>>t;
	
	for(;t>0;t--)
	{
		long long int n1,n2,k;
		cin>>n1>>n2>>k;
			
		long long int maxi=0;	
		for(long long int i=n1-k;i<=n1+k;i++)		
		{
			for(long long int j=n2-k;j<=n2+k;j++)
			{
				int a1,a2;
				
				if(i<n1)
				a1=n1-i;
				else
				a1=i-n1;
				
				if(j<n2)
				a2=n2-j;
				else
				a2=j-n2;
				
				if(a1+a2>k)
				continue;
				
				if(maxi<gcd(i,j))
				maxi=gcd(i,j);	
			}
			
		}
		
		cout<<maxi<<endl;
	}
	
	
	return 0;
}
