#include <iostream>
#include<stdio.h>
#include<map>
using namespace std;

int main()
{
    // cout << "Hello World!" << endl;
    int t;
    string s;
    map<string,int>m;
    scanf("%d",&t);
    while(t--){
    	cin>>s;
    	m[s]+=1;
    }
    map<string,int>::iterator i;
    int maxi=0;
    int count=0;
    for(i=m.begin();i!=m.end();i++){
    	if((*i).second>maxi){
    		maxi=(*i).second;
    		count=(*i).second;
    	}else if((*i).second==maxi){
    		count+=(*i).second;
    	}
    }
    cout<<count<<endl;
    return 0;
}
