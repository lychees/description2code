#define MOD 1000000007
int t = 0;
long long int gcd(long long int a,long long int b){
	if(a>b){
		swap(a,b);
	}
	while(a){
		swap(a,b);
		a%=b;
	}
	return b;
}
long long int ppow(long long int i,long long int j){
	long long int res=1;
	while(j){
		if(j&1){
			res*=i;
			res%=MOD;
		}
		i*=i;
		i%=MOD;
		j>>=1;
	}
	return res%MOD;
}
int main(){
	cin >> t;
	while (t--){
		long long int a;
		long long int b;
		long long int c;
		scanf("%lld%lld%lld", &a, &b, &c);
		long long int mint = min(a, min(b, c));
		long long int ii = 1;
		int j=1;
		for (ii = mint; ii/j >= 0; ii--){
			ii=gcd(a,b);
			ii=gcd(ii,c);
			if (a%ii == 0 && b%ii == 0 && c%ii == 0){
				long long int ko = (((a*b)%MOD*c)%MOD )*ppow(((ii*ii)%MOD*ii)%MOD,MOD-2);
				ko %= MOD;
				ii %= MOD;
				printf("%lld %lld\n", ii, ko);
				break;
			}
		}
	}
	return 0;
}
