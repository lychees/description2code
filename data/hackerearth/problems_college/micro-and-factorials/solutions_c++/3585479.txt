void sieve()
{
	int i,j;
	for (i=2;i*i<=LIM;i++)
	{
		if (chk[i])
			continue;
		prime[p++]=i;
		for (j=i*i;j<=LIM;j+=i)
		{
			chk[j]=1;
		}
	}
	for (;i<=LIM;i++)
	{
		if (!chk[i])
			prime[p++]=i;
	}

}

ll doit(ll x,int y)
{

	return x/y+doit(x/y,y);
