#include<iostream>
#include<string>
using namespace std;


int main()
{
    string s;
    int t;
    cin>>t;

    while(t--)
    {
        cin>>s;
        int ans=0, count[100];

        for(int i=0; i<s.size(); i++)
        {
            count[i]=0;
        }

        for(int i=0; i<s.size(); i++)
        {
            count[s[i]-'a']++;
        if(count[s[i]-'a']==1)
            ans++;
        }

        if(ans==26)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }

    return 0;
}
