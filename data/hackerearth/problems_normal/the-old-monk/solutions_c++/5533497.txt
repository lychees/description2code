vll a, b;

int  main ()
{
	ios_base::sync_with_stdio(false);
    //freopen ("input.txt", "r", stdin);

	ll t, n, x;

	cin>>t;

	while(t--)
	{
		cin>>n;
		a.clear();
		b.clear();
		rep(i, 1, n)
		{
			cin>>x;
			a.pb(x);
		}
		rep(i, 1, n)
		{
			cin>>x;
			b.pb(x);
		}

		ll mx = 0;

		rep(i, 0, n-1)
		{
			ll key = a[i];
			ll lo = i;
			ll hi = n-1;
			ll ans=i;
			while(lo<=hi)
			{
				ll mid = (lo+hi)/2;
				if(b[mid]>=key)
				{
					ans = mid;
					lo = mid+1;
				}
				else
				{
					hi = mid-1;
				}
			}

			mx = max(mx, ans-i);
		}

		cout<<mx;nl;
	}

    return  0;
}
