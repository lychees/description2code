#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    int n;
    long long int m;
    vector<long long int> v;
    long long int sum,sum1;
    while(t--)
    {
    	sum=0;
    	sum1=0;
    	scanf("%d",&n);
    	v.resize(n);
    	for(int k=0;k<n;k++)
    	{
    		scanf("%lld",&v[k]);
    	}
    	scanf("%lld",&m);
    	if(m==0)
    	{
    		cout<<"0"<<endl;
    	}
    	else{
    	long long int i;
    	for(i=0;i<n;i++)
    	{
    		sum+=v[i];
    		if(sum>=m)
    		{
    			cout<<i+1<<endl;
    			break;
    		}
    	}
    	if(sum<m)
    	{
    		m=m%sum;
    		if(m==0)
    		{
    			m=sum;
    		}
    	for(i=0;i<n;i++)
    	{
    		sum1+=v[i];
    		if(sum1>=m)
    		{
    			cout<<i+1<<endl;
    			break;
    		}
    	}
    	}
    	}
    	v.clear();
    }
}
