import itertools
def prime_factors(n):
    i = 2
    factors = []
    while i * i <= n:
        if n % i:
            i += 1
        else:
            n //= i
            factors.append(i)
    if n > 1:
        factors.append(n)
    return factors
for i in range((int)(raw_input())):
    x=prime_factors((int)(raw_input()))
    s=set()
    for L in range(0, len(x)+1):
        for subset in itertools.combinations(x, L):
            s.add(subset)
