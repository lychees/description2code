#include <algorithm>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <map>
#include <queue>
#include <set>
#include <vector>
using namespace std;
bool k[1005];
bool used;
int a, t;
int main() {
  scanf("%d", &a);
  k[a] = true;
  while (scanf(",%d", &a))
    k[a] = true;
  for (int i = 0; i <= 1002; i++) {
    if (!k[i]) {
      if (!t)
        continue;
      if (i == t + 1)
        if (used)
          cout << "," << t;
        else {
          cout << t;
          used = true;
        }
      else if (used)
        cout << "," << t << "-" << i - 1;
      else {
        cout << t << "-" << i - 1;
        used = true;
      }
      t = 0;
    } else if (!t)
      t = i;
  }
  return 0;
}
