#include<bits/stdc++.h>
using namespace std;

#define LL long long

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		LL n;	
		LL ans = 0;
		scanf("%lld",&n);
		if(n == 1)
			printf("0\n");
		else if(n%2 == 0)
		{
			ans += n-1;
			ans += ((n-2)*n)/2;
			printf("%lld\n",ans); 
		}
		else
		{
			ans += n;
			ans += ((n-3)*(n+1))/2;
			printf("%lld\n",ans); 
		}
	}
	return 0;
}
