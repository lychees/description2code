#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int i,n;
    	cin>>i>>n;
    	if(i==2&&n==0)
        cout<<"2"<<endl;    		
    	else if((i==1&&n%2==1)||(i==2&&n%2==0)||(i==0&&n%2==0))
    		cout<<"0"<<endl;
    	else 
    	cout<<"1"<<endl;
    	
    }
    return 0;
}
