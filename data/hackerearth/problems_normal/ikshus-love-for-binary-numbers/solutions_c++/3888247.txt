using namespace std;
ll n,K;
ll dp[2][61][61];
ll func(ll prev,ll k,ll index)
{
    if(index==n && prev==1)
        return 1;
    else if(index==n)
        return 0;
    if( dp[prev][k][index]!=-1)
        return dp[prev][k][index];
    if(k>=K-1)
        return dp[prev][k][index]=func(prev,0,index+1)+func(1,k+1,index+1);
    else
        return dp[prev][k][index]=func(prev,0,index+1)+func(prev,k+1,index+1);
}
int main()
{
    ll arr[62]={0};
    arr[0]=1;
    for(int i=1;i<=61;i++)
    {
        arr[i]=arr[i-1]*2;
    }
    cin>>n>>K;
    ll sum=0;
    memset(dp,-1,sizeof(dp));
    sum=func(0,0,0);
    //cout<<func(0,0,0)<<endl;
    ll x=sum/(__gcd(sum,arr[n]));
    ll k=arr[n]/(__gcd(sum,arr[n]));
   // cout<<sum<<"/"<<arr[n]<<endl;
    cout<<x<<"/"<<k<<endl;
}
