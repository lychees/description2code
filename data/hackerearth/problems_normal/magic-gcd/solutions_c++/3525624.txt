#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll findGcd(ll a,ll b)
{
	if(a==0)return b;
	else return findGcd(b%a,a);
}
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		ll  n1,n2,k;
		cin>>n1>>n2>>k;
		ll ans=0;
		for(int i=-k;i<=k;++i)
		{
			for(int j=-k;j<=k;++j)
			{
				if( n1+i>=1 && n2+j>=1 && abs(i)+abs(j)<=k )
				{
					ans=max(ans,findGcd(n1+i,n2+j));
				}
			}
		}
		cout<<ans<<endl;	
	}
	return  0;
}
