str_arr = raw_input().split(' ')
arr = [int(num) for num in str_arr]
if len(arr) == 2:
    n = arr[0]
    x = arr[1]
count = 0
count1 = 0
str_arr1 = raw_input().split(' ')
arr1 = [int(num) for num in str_arr1]
if len(arr1) == n:
    for i in arr1:
        if i <= x:
            count += 1
        else:
            count1 += 1
        if count1 > 1:
            break
