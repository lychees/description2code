#include <iostream>
using namespace std;

int main()
{
	int n, i, j, k, l, m;
	scanf("%d", &n);
	for (i = 0; i < n; i++) {
		if (i == 0 || i == n - 1) {
			for (j = 0; j < n; j++)
				printf("*");
		} else {
			for (j = 0; j < n - i - 1; j++) {
				printf(" ");
			}
			printf("*");
		}
		printf("\n");
	}
    return 0;
}
