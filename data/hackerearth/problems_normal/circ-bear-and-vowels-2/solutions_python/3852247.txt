all_vowels = ['a', 'e', 'i', 'o', 'u', 'y']
total_cases = int(raw_input())
strings = list()
for case in range(total_cases):
    strings.append(str(raw_input()))

for string in strings:
    vowels = 0
    consecutive_consonants = 0
    is_hard = False
    string_length = len(string)
    for letter in list(string):
        if letter in all_vowels:
            vowels = vowels + 1
            consecutive_consonants = 0
        else:
            consecutive_consonants = consecutive_consonants + 1
        if consecutive_consonants == 3:
            is_hard = True
            break
    if is_hard:
        print 'hard'
        continue
    if string_length % 2 == 0:
        if vowels < int(string_length / 2):
            print 'hard'
            continue
    else:
        if vowels <= int(string_length / 2):
            print 'hard'
            continue
    print 'easy'
