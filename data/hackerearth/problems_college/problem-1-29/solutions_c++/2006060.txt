#include <iostream>
using namespace std;

int main()
{
    long long t;
    cin>>t;
    while(t--)
    {
    	int n,i,j;
    	cin>>n;
    	bool b[n+1][n+1];
    	for(i=1;i<=n;i++)
    	 for(j=1;j<=n;j++)
    	  cin>>b[i][j];
    	  b[1][1]=1;
    	  b[n][n]=1;
    	 for(i=n;i>0;i--)
    	 {
    	 	b[0][i]=0;
    	 	b[i][0]=0;
    	 	if(b[i][n]==0)
    	     b[i-1][n]=0;
    	    if(b[n][i]==0)
    	     b[n][i-1]=0;
    	 }    
    	for(i=n-1;i>=1;i--)
    	 for(j=n-1;j>=1;j--)
    	  if(b[i+1][j]==0&&b[i][j+1]==0)
    	   b[i][j]=0;
    	if(b[1][1]==0)
    	 printf("NOT POSSIBLE\n");
    	else
    	 printf("POSSIBLE\n");
    }
    return 0;
}
