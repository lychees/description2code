#include <bits/stdc++.h>
#define ll long long
#define MOD 1000000007
using namespace std;
ll fact[100005];
void sol(){
	fact[0]=fact[1]=1;
	int i;
	for(i=2;i<100000;i++){
		fact[i]=(fact[i-1]*i)%MOD;
	}
}
ll power(ll a,ll b){
	ll res=1;
	while(b>0){
		if(b%2)
		res=(res*a)%MOD;
		a=(a*a)%MOD;
		b/=2;
	}
	return res;
}
ll cat(ll n,ll k){
	if(n==0 || k==0)
	return 1;
	
	ll ans=((((fact[n]*power(fact[k],MOD-2))%MOD)*power(fact[n-k],MOD-2))%MOD);
	return ans%MOD;
}
int main()
{
    //cout << "Hello World!" << endl;
    sol();
    int t;
    scanf("%d",&t);
    while(t--){
    	ll n,i,k;
    	ll a[100005],ans=0;
    	scanf("%lld %lld",&n,&k);
    	for(i=0;i<n;i++){
    		scanf("%lld",&a[i]);
    	}
    	sort(a,a+n);
    	for(i=0;i<n-k+1;i++){
    		ll ans1=cat(n-i-1,k-1);
    		ans+=(ans1*a[i])%MOD;
    	}
    	ans%=MOD;
    	printf("%lld\n",ans);
    }
    return 0;
}
