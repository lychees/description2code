#include <string>
#include <iostream>

using namespace std;

bool find(char ch , string s){
	if(s.find(ch)==string::npos){
		return false;
	}
	return true;
}
string remove(string s){
	char ch = s[0];
	s.erase(0,1);
	while(find(ch,s)){
		ch =s[0];
		s.erase(0,1);
	}
	return s;
}
int main(int argc, char **argv) {
	int n;
	string s;
	cin>>n>>s;
	int len =s.size();
	int count=0;
	for (int k =0; k < len; ++k) {
		string s1="";
		int i=0,j=0;
		while(i!=n){
			if((j>=len-k)){
				cout<<count<<endl;
				return 0;
			}
			if(find(s[j],s1)){
				s1+=s[j];
			}else{
				s1+=s[j];
				i++;
			}
			j++;
			//cout<<i<<j<<k<<"   "<<s1<<endl;
		}
		count++;
		//cout<<i<<"1   "<<s1<<endl;
		for (i = j; i < s.size(); ++i) {
			if(find(s[i],s1)){
				s1+=s[i];
				count++;
				//cout<<"2    "<<s1<<endl;
			}else{
				break;
			}
		}
		s.erase(0,1);
	}
	cout<<count<<endl;
}
