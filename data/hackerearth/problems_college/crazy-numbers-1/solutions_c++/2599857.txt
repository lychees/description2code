#include <bits/stdc++.h>
#define INF 1000000000
#define MOD 1000000007
#define MAXN 1000005

using namespace std;

typedef long long ll;

ll dp[MAXN][10];
int main()
{
	ll t, n, sum;
	for (int i = 1; i < 10; i++)
		dp[1][i] = 1;
	for (int i = 2; i < MAXN; i++)
	{
		for (int j = 0; j < 10; j++)
		{
			if (j == 0)
				dp[i][j] = dp[i - 1][j + 1];
			else if (j == 9)
				dp[i][j] = dp[i - 1][j - 1];
			else
				dp[i][j] = (dp[i - 1][j - 1] + dp[i - 1][j + 1]) % MOD;
		}
	}
	scanf("%lld", &t);
	while (t--)
	{
		scanf("%lld", &n);
		if (n == 1)
			sum = 1;
		else
			sum = 0;
		for (int i = 0; i < 10; i++)
			sum = (sum + dp[n][i]) % MOD;
		printf("%lld\n", sum);
	}
	return 0;
}
