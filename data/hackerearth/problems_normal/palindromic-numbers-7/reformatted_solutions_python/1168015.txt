def is_palindrome(x):
    return str(x) == str(x)[::-1]


for i in range(int(raw_input())):
    a, b = str(raw_input()).split()
    count = 0
    for i in range(int(a), int(b)+1):
        if is_palindrome(i):
            count += 1
    print count
