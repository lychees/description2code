#include<iostream>
#include <string.h>
using namespace std;
#define MAX 100
int A[MAX][MAX], N, T, C;

int main()
{
	int i, j,t;
	cin>>T;
	for(t=0;t<T;++t)
	{	
		//memset(A, 0, sizeof(A));
		cin>>N>>C;
		for(i=0;i<N;++i)
			for(j=0;j<N;++j)
				cin>>A[i][j];
		
		for(i=1;i<N;++i)
			A[0][i]+=A[0][i-1];

		for(i=1;i<N;++i)
			A[i][0]+=A[i-1][0];

		for(i=1;i<N;++i)
			for(j=1;j<N;++j)
				if(A[i-1][j]>A[i][j-1])
					A[i][j]+=A[i-1][j];
				else
					A[i][j]+=A[i][j-1];

		if(A[N-1][N-1]>=C)
			cout<<A[N-1][N-1]-C<<endl;
		else
			cout<<"-1"<<endl;

	}

}
