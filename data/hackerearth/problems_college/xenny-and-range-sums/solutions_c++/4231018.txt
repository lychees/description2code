#include <iostream>
using namespace std;

int main()
{
	int n,q,temp;
	cin>>n>>q;
	int arrs[n+1],arr[n+1];
	arrs[0]=0;
	arr[0]=0;
	for (int i=1;i<=n;i++)
	{
		cin>>temp;
		arr[i]=temp;
		arrs[i]=temp+arrs[i-1];
		
	}
	
	
	long long sum=0,diff=0;
	while(q--)
	{
	int l,r;
	cin>>l>>r;
		sum=arrs[r]-arrs[l-1];
		diff=arr[r]-(arrs[r-1]-arrs[l-1]);
		
	
	cout<<sum<<" "<<diff<<endl;
	}
    return 0;
}
