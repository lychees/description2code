#include<bits/stdc++.h>
using namespace std;
#define mp make_pair
#define ft first
#define sd second
#define scan(x) scanf("%d",&x)
int main(){


	int n,q;
	scan(n);
	scan(q);	
	set<pair<int,int> > S;
	set<pair<int,int> > :: iterator it;	
	for(int i=0;i<q;i++){
		int u,v;
		scan(u),scan(v);
		bool f;
		u==v?f=0:f=1;	
		for(it=S.begin();it!=S.end();it++)
		{	int x=min(it->ft,it->sd);
			int y=max(it->ft,it->sd);
			if(!((x<u&&x<v&&y>u&&y>v)||(u<x&&v<x)||(u>y&&v>y)||(x>min(u,v)&&max(u,v)>y)))
			{	f=0;
				break;			
			}
		}
		if(f){
			S.insert(mp(u,v));
			printf("YES\n");
		}else{
			printf("NO\n");
		}	
	}
	return 0;
}
