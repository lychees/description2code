#include<bits/stdc++.h>
using namespace std;
#define ll long long 
ll mpow(ll a,ll b)
{
	ll ret=1;
	while(b>0)
	{
		if((b&1))
			ret=ret*a;
		a=a*a;
		b=b>>1;
	}
	return ret;
}
