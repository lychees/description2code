from itertools import permutations as perm
from sys import stdout


def solve(n):
    lst = [str(i) for i in xrange(1, n+1)]
    return [''.join(i) for i in perm(lst)]


def gen(x):
    ans = 0
    for element in x:
        ans *= 10
        ans += element
    return ans


t = int(raw_input())
for qq in xrange(t):
    n = int(raw_input())
    s = solve(n)
    for x in s:
        stdout.write(x+' ')
    stdout.write('\n')
