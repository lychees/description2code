#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
	int t;
	cin>>t;

	while(t--)
	{
		int n,a,sum[10001];

		for(int i=0;i<=10000;++i)
		sum[i]=0;
		sum[0]=1;
		cin>>n;

		for(int i=0;i<n;i++)
		{
			cin>>a;
			for(int j=10000;j>=0;j--)
			{
			    if(j+a<=10000)
				if(sum[j]>0)
				 sum[j+a]++;
			}
		}
		int ans=0;
		for(int i=0;i<10000;i++)
		{
			if(sum[i]>0)
			ans++;
		}
		printf("%d\n",ans);
	}

    return 0;
}
