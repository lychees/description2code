#include <bits/stdc++.h>
using namespace std;
int r[100009];
int gcd(int x,int y)
{
	if(y==0)
	return x;
	return gcd(y,x%y);
}
int main()
{
    int s,q;
    scanf("%d%d",&s,&q);
	while(q--)
	{
		int g;
		scanf("%d",&g);
		int y=gcd(s,g);
		if(r[y]==1)
		printf("-1\n");
		else
		{
			printf("%d\n",y);
			
		}
		r[y]=1;
	}
    return 0;
}
