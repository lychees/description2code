#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k,i,j,finans,sum;
    	cin>>n>>k;
    	int a[n];
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	int fin[n][n];
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<n;j++)
    		{
    			if(i!=j)
    			{
    				if(a[i]<a[j])
    				{
    					fin[i][j]=(a[j]-a[i])*5;
    				}
    				else if(a[i]>a[j])
    				{
    					fin[i][j]=(a[i]-a[j])*3;
    				}
    				else
    					fin[i][j]=0;
    			}
    			else
    				fin[i][j]=0;
  
    		}

    	}
    	finans=100000000;
    	for(i=0;i<n;i++)
    	{
    		sort(&fin[i][0],&fin[i][n]);
    		sum=0;
    		for(j=0;j<k;j++)
    		{
    			sum+=fin[i][j];
    		}
    		if(finans>sum)
    			finans=sum;
    	}
    	cout<<finans<<endl;
    }
    return 0;
}
