#include <iostream>
#include <stdio.h>
#include <vector>

using namespace std;

int main()
{
	int t, n, i, count, flag;
	
	cin >> t;
	
	while( t-- ) {
		
		string s;
		cin >> s;
		n = s.length();
		
		vector<int> A( 26, 0 );
		
		for( i = 0;i < n;i++ ) {
			A[ s[ i ] - 97 ]++;
		}
		
		count = A[ 0 ];
		flag = 1;
		
		for( i = 1;i < 26;i++ ) 
			if( A[ i ] != count ) {
				flag = 0;
				break;
			}
		
		if( flag )
			printf("Yes\n");
		else
			printf("No\n");
			
	}
	return 0;
}
