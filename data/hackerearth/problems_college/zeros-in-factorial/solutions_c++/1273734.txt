#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
const int MAXN = 1000005;
LL z[MAXN] = {0, 0, 0};
 
int main() {
ios_base::sync_with_stdio(false); cin.tie(0);
for (int i = 2; i < MAXN; ++i) {
int c = 0, x = i;
while (x > 0) c += (x /= 5);
z[i] = c;
}
for (int i = 1; i < MAXN; ++i)
z[i] += z[i - 1];
int T; cin >> T;
while (T--) {
int a, b;
cin >> a >> b;
cout << z[b] - z[a - 1] << "\n";
}
return 0;
}
