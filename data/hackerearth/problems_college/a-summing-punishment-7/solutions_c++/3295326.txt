#include <bits/stdc++.h>
using namespace std;

int main()
{
   int ts;
   cin>>ts;
   int a[ts];
  for(int i=1;i<=ts;i++){
  	cin>>a[i];
  }
  int x[100001];
  x[1]=a[1];
  for(int i=2;i<=ts;++i) {
  	x[i]=x[i-1]+a[i];
  	//cout<<x[i]<<" ";
  }
  
  int q;
  cin>>q;
  int l,k;
  while(q--){
  	cin>>l>>k;
  	cout<<x[k]-x[l-1]<<"\n";
  	
  }
  
    return 0;
}
