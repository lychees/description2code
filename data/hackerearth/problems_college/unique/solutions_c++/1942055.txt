#include <iostream>
#include <map>
#define ll long long
using namespace std;

int main()
{   
    ll t;
    cin>>t;
    while(t--){
        ll n,i,c=0;
        cin>>n;
        map<long long ,long long> m;
        ll ar[n];
        for(i=0;i<n;i++){
            cin>>ar[i];
            m[ar[i]]++;
        }
        map<long long,long long>::iterator it;
        for(it=m.begin();it!=m.end();it++){
            if((it->second)>1){
                c+=it->second-1;
            }
        }
        cout<<n-c<<endl;
    }

    return 0;
}
