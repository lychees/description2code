#include <bits/stdc++.h>

using namespace std;

int main()
{
	char s[1000][1000];
	int n,m,count,i,j;
	int t;
	scanf("%d",&t);
	while (t --) {
		scanf("%d %d",&n,&m);
		for (i = 0; i < n; i++) {
			scanf("%s",s[i]);
		}
		count = 0;
		for (i = 0; i < n; i++) {
			for (j = i; j < n; j++) {
				if (strcmp(s[i],s[j]) == 0) {
					count++;
				}
			}
		}
		if (count == n) {
			printf("Yes\n");
		} else {
			printf("No\n");
		}
	}
	
	return 0;
}
