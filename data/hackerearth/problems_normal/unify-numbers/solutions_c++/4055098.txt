#include<bits/stdc++.h>
using namespace std;
long long int min(long long int x,long long int y)
{
	return x<=y?x:y;
}
int main()
{
	int i;
    long long int t,n;
    long long int a[100005],ans,median;
    scanf("%lld",&t);
    while(t--)
    {
        ans = 0;
    	scanf("%lld",&n);
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&a[i]);
    		//sum+=a[i];
        }
        //ave1 = ceil(sum*1.0/n);
        //ave2 = floor(sum*1.0/n);
        //for(i=0;i<n;i++)
        //{
        //	ans1 += abs(a[i] - ave1);
        //	ans2 += abs(a[i] - ave2);
        //}
        sort(a,a+n);
        if ((n % 2) == 0) {
        median = (a[n/2] + a[(n/2) - 1])/2.0;
        } 
        else {
        median = a[n/2];
         }
        for(i=0;i<n;i++)
        {
        	ans += abs(a[i] - median);
        }
        printf("%lld\n",ans);
    }
    return 0;
}
