#include <stdio.h>
#include<stdlib.h>
#include<string.h>
 long int *a,ele;
 int n,flag=0;
void f(int i,long sum)
{
	if(i>n)
	{
		if(sum==ele&&flag==0)
		{
			flag=1;
			printf("YES\n");

		}
		return;

	}
f(i+1,sum);
f(i+1,sum+a[i]);
}
int main()
{

   int t,i;
   scanf("%d",&t);
   while(t--)
	{
		flag=0;
		scanf("%d",&n);
		a=(long *)malloc(n*sizeof(long));
		for(i=0;i<n;i++)
			scanf("%ld",&a[i]);
		scanf("%ld",&ele);
		f(0,0);
		if(flag==0)
		printf("NO\n");
		free(a);
	}
    return 0;
}
