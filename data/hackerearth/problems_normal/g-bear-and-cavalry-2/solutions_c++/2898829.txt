#include <iostream>
#include <algorithm>

using namespace std;

bool rev(int a, int b){
	return a > b;
}

bool asc(int a, int b){
	return a < b;
}

int main(){
	int nTest;
	cin>>nTest;
	while(nTest--){
		int n;
		cin>>n;
		int w[n], h[n];

		for(int i = 0; i < n; i++){
			cin>>w[i];
		}
		for(int i = 0; i < n; i++){
			cin>>h[i];
		}

		sort(h, h+n, rev);
		sort(w+1, w+n, asc);
		bool ok = true;
		for(int i = 1; i < n && ok; i++){
		//	cout<<w[0]<<" "<<h[0]<<" -> "<<w[i]<<" "<<h[i]<<endl;
			ok = (w[0]*h[0] > w[i]*h[i]);
		}

		if(ok)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
	return 0;
}
