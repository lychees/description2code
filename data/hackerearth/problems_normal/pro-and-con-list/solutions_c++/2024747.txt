#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int T,n,f,a;
    long long anger;
    long long *ans;
    
    cin>>T;
    
    while(T--)
    {
    	cin>>n;
    	ans=new long long[n];
    	
    	anger=0;
    	
    	for(int i=0;i<n;i++)
    	{
    		cin>>f;
    		cin>>a;
    		//cout<<"f= "<<f<<" and a = "<<a<<endl;
    		anger+=a;
    		
    		ans[i]=f+a;	
    		
    	}
    	//cout<<"anger = "<<anger<<endl;
    	//for(int i=0;i<n;i++)
    		//cout<<ans[i]<<"  ";
    	sort(ans,ans+n);
    	
    	cout<<ans[n-1]+ans[n-2]-anger<<endl;
    	
    }
    
    return 0;
}
