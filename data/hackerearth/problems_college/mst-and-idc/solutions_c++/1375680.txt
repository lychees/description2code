#include <stdio.h>
#include <algorithm>
using namespace std;
int A[1000000];
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int N;
		scanf("%d",&N);
		for(int i=0;i<N;i++)
		{
			scanf("%d",&A[i]);
		}
		sort(A,A+N);
		long long int sum=0;
		for(int i=1;i<N;i++)
		{
			sum+=A[i]-A[i-1];
		}
		sum=sum*2;
		printf("%lld\n",sum);
	}
}
