vector<list<ll> > adjlist(200000);
ll child[200000],visited[200000];
void dfs(int i)
{
	ll ans=0;
	
	for(list<ll> ::iterator it=adjlist[i].begin();it!=adjlist[i].end();it++)
	{
		if(!visited[*it])
		{
			visited[*it]=1;
			dfs(*it);
			ans+=child[*it];
//			cout<<ans<<endl;
		}
	}
	child[i]=ans+1;
}
