#include <iostream>
#define ll long long
#include<vector>
using namespace std;

int main()
{
    ll t,n;
    cin>>t;
    while(t--)
    {
    	ll v[100];
		v[0]=0;
		v[1]=1;
		v[2]=1;
    	cin>>n;
    	if(n==0)
    	{
    		cout<<"0 0"<<endl;
    	}
    	else{
    	for(ll i=3;i<=n;i++)
    	{
    		v[i]=v[i-1]+v[i-2];
    	}
    	cout<<v[n-1]<<" "<<v[n]<<endl;
    
    	}
    	}
    
    return 0;
}
