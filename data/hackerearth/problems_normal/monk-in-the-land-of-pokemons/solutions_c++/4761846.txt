#include <iostream>

using namespace std;
long int a,b,n,y=0;
int main()
{
    int t,i;
    cin>>t;
    while(t--)
    {long int count[1000005]={0};
    	y=0;
    	cin>>n;
     for(i=0;i<n;i++)
     {cin>>a>>b;
      count[a]++;
      if(count[b]>0)
      count[b]--;
      else
    {  y++;
    
    }
     }
     cout<<y<<endl;
    }
      
    return 0;
}
