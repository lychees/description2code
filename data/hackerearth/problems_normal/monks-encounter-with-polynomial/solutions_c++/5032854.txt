#include <iostream>

using namespace std;

long long int x=0,t,A,B,C,K,f=0,v;


int main()
{
    cin>>t;
    while(t--)
    {
        cin>>A;
        cin>>B;
        cin>>C;
        cin>>K;
        x=0;
        f=0;
        while(f==0)
        {
            v=A*x*x+B*x+C;
            if(v>=K){
                cout<<x;
                f=1;
            }
            else x++;

        }

        cout<<endl;
    }

    return 0;
}
