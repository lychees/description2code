#include <iostream>
#include <string>
using namespace std;

int main()
{
    int N;
    string s,p;
    cin>>N;
    for(int i=0;i<N;i++){
    	s.clear();
    	p.clear();
    	cin>>s>>p;
    	if(p.size()!=2*s.size()){
    		cout<<"Impossible\n";
    		continue;
    	}
    	if(p.compare(s+s)==0){
    		cout<<"Possible\n";
    		continue;
    	}
    	string tp(p),tpl;
  	    while(true){
	    	string tp1(tp);
	    	//cout<<tp1<<"121212121"<<tpl<<endl;
	    	std::size_t f=tp1.find(s);
	    	if(f==string::npos){
	    	cout<<"Impossible\n";
	    	goto fin;
	    	}
	    	tp1.erase(f,s.size());
	    	if(s.compare(tpl+tp1)==0){
	    		cout<<"Possible\n";
	    		goto fin;
	    	}else{
	    		tpl=tpl+tp.substr(0,f+1);
	    		tp.erase(0,f+1);
	    		//cout<<tp<<"0000"<<tpl<<"yyyy"<<tp1<<"xxxx"<<f<<endl;
	    	}
	    }
    	fin:;
    }
    return 0;
}
