#include<iostream>
using namespace std;

int main()  {
  int test;
  cin>>test;
  while(test--) {
      int N, K, flag = 0, counter=0;
      cin>>N>>K;
      for(int i = 10; i < 100 && N > 2; i++) {
        int j = i/10;
        if(j%10 == i%10)
        continue;
        counter++;
        if(counter == K)  {
          flag = 1;
          int num1 = j%10;
          int num2 = i%10;
          int Ntimes = 0;
          while(Ntimes < N) {
            if(Ntimes%2 == 0)
              cout<<num1;
            else
              cout<<num2;
            Ntimes++;
          }
          cout<<endl;
        }

      }
      if(!flag)
      cout<<-1<<endl;

  }
 return 0;
}
