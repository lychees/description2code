#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	string s;
    	int ct=0;
    	cin>>s;
    	for(int i=0;s[i]!='\0';i++){
    		if(s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u'||s[i]=='A'||s[i]=='E'||s[i]=='I'||s[i]=='O'||s[i]=='U'){
    			cout<<s[i];
    			ct=1;
    		}
    	}
    	if(ct==0)
    		cout<<"No";
    	cout<<endl;
    }
    return 0;
}
