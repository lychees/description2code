#include<iostream>
#include<cstdio>

using namespace std;

int gcd(int x, int y) {
   if(y == 0) return x;
   return gcd(y,x%y); 
}    

int solu(int a[], int n) {
    int sol = a[0];
    
    for(int i=1;i<n;i++) {
        sol = gcd(sol,a[i]);
    }   
    return sol;
}    

int main() {
    ios_base::sync_with_stdio(0);
    
    int t,n;
    cin>>t;
    
    while(t--) {
        cin>>n;
        int a[n];
        
        for(int i=0;i<n;i++) cin>>a[i];
        
        cout<<solu(a,n)<<endl;
    }    
    
    return 0;
}
