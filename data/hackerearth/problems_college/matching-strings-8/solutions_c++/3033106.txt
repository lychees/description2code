#include <iostream>

using namespace std;

int main()
{
    int t;
    char **a,**b;
    cin>>t;
    a = new char*[t];
    b = new char*[t];
    for(int i=0; i<t; i++){
        a[i] = new char[501];
        b[i] = new char[501];
        cin>>a[i];
        cin>>b[i];
    }

    for(int i=0; i<t; i++){
        int sum=0;
        for(int j=0; a[i][j]!='\0' && b[i][j]!='\0'; j++){
                    if(a[i][j] == b[i][j])
                        sum++;
        }
        cout<<sum<<endl;
    }


    return 0;
}
