#include<iostream>
using namespace std;

long MINIMIZE(long total) {
	if(total < 10)
		return total;
	else
		return (total % 10) + MINIMIZE(total/10);
}

int main() {
	int t, i, m, j;
	long len, k, total;
	
	cin>>t;
	for(i=0;i<t;i++) {
		cin>>m;
		total = 0;
		for(j=0;j<m;j++) {
			cin>>len>>k;
			total += len * k;
			total = MINIMIZE(total);
		}
		total = MINIMIZE(total);
		cout<<total<<endl;
	}
	
	return 0;
}
