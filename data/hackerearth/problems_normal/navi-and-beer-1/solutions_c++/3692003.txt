#include <iostream>
#include <cmath>
using namespace std;

// K * i^3
/*
 Sample Input(Plaintext Link)
 1
2 2 10
Sample Output(Plaintext Link)
 8
Explanation
Total Money required is : 2 * 1^3 + 2 * 2^3 = 18 but he already has 10 so ans is 8.
*/

long long int power(int a, int b)
{
  long long int pow1 = 1;
  while ( b ) 
  {
         if ( b & 1 ) 
         {
           pow1 = pow1 * a;
           --b;
         }
         a = a*a;
         b = b/2;
  }
  return pow1;
}

int main() {
	long long int t, n, k, i, m;
	cin>>t;
	while(t--){
		cin>>n>>k>>m;
		long long int bear_cost=0;
		for(i=1;i<=n;i++){
			bear_cost=bear_cost+(k*power(i,3));
		}
		long long int loan=bear_cost-m;
		if(loan>0)
			cout<<bear_cost-m<<"\n";
		else{
			cout<<0<<"\n";
		}
	}
	return 0;
}
