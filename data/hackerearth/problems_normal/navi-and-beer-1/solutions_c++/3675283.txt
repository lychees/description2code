#include <iostream>
using namespace std;

int main()
{
    int t=0;
    long long k = 0;
    long long m = 0;
    long long n = 0;
    
    cin >> t;
    while(t--){
    	cin >> n;
    	cin >> k;
    	cin >> m;
    	
    	long long totalAmt = 0;
    	for(long long i=1; i<=n; i++){
    		totalAmt += i*i*i;
    	}
    	
    	totalAmt *= k;
    	if(totalAmt-m <=0){
    		std::cout << 0 << std::endl;
    	}else{
    		std::cout << totalAmt-m << std::endl;
    	}
    }
}
