#include<bits/stdc++.h>
using namespace std;
int main(void)
{
	int t,end,start,curr_sum,k,len,C;
	string str;
	cin>>t;
	while(t--)
	{
		cin>>k;
		cin>>str;
		start=end=0;
		C=0;
		curr_sum=str[0]-'a'+1;
		len=str.length();
		while(end<len)
		{
			if(curr_sum==k)
			C++;
			if(curr_sum<=k)
			{
				end++;
				if(end<len)
				curr_sum+=(str[end]-'a'+1);
			}
			else
			{
				curr_sum-=(str[start]-'a'+1);
				start++;
			}
		}
		cout<<C<<endl;
	}
}
