b=int(raw_input())
a=map(int,raw_input().split(" "))
j=0
p=0
q=0
r=0
while(j<b):
	if(j%3==0):
		p+=int(a[j])
	elif((j-1)%3==0):
		q+=int(a[j])
	else:
		r+=int(a[j])
	j+=1
print p,q,r
