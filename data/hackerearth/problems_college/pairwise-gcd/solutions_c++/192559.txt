 inline void inp( int &n )
 {
    n=0;
    int ch=getchar_unlocked();int sign=1;
    while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getchar_unlocked();}

    while(  ch >= '0' && ch <= '9' )
            n = (n<<3)+(n<<1) + ch-'0', ch=getchar_unlocked();
    n=n*sign;
  }

 cin>>n;

ll arr[400000];
int calculate[400000];

int main()
{
   int t;
   cin>>t;
   while(t--)
   {int i, j, n;
    inp(n);
	
     memset(arr,0,sizeof(arr));
     memset(calculate,0,sizeof(calculate));
	for(i=2,calculate[1]=1; i<=n; i++)
	{
		if(!calculate[i])
		{
			calculate[i] = i-1;
			for(j=(i<<1); j<=n; j+=i)
			{
				if(!calculate[j]) calculate[j] = j;
					calculate[j] = calculate[j]/i*(i-1);
			}
		}
	}

	for(i=1; i<=n; i++)
		for(j=2; i*j<=n; j++)
			arr[i*j] += i*calculate[j];

	for(i=2; i<=n; i++)
		arr[i] += arr[i-1];
    
 	cout<<arr[n]<<endl;
   }
	return 0;
}
