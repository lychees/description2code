#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,x1,x2,y1,y2,x1a,x2a,y1a,y2a;
    cin>>t;
    while(t--)
    {
    	scanf("%d%d%d%d%d%d%d%d",&x1,&y1,&x2,&y2,&x1a,&y1a,&x2a,&y2a);
    	if(x1>x2)
    		swap(x1,x2);
    	if(y1>y2 )
    		swap(y1,y2);
    	if(x1a>x2a)
    		swap(x1a,x2a);
    	if(y1a>y2a)	
    		swap(y1a,y2a);
    	if(x2a<x1 || y2a<y1 || x1a>x2 || y1a>y2)
    		puts("Apart");
    	else if(x2a>x1 && y2a>y1 && x1a<x2 && y1a<y2)
    		puts("Rectangle");
    	else if((x1a==x2 && (y1a==y2||y2a==y1))||(x2a==x1 && (y2a==y1||y1a==y2)))
    		puts("Point");
    	else
    		puts("Line");
    }
    return 0;
}
