#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t)
    {
    	int n,q;
    	cin>>n>>q;
    	int arr[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>arr[i];
    	}
    	sort(arr,arr+n);
    	long sum=0;
    	if(n==0)
    	cout<<"0\n";
    	else
    	{
    		for(int i=0;i<n-1;i++)
    		{
    			long k=arr[i+1]-arr[i];
    			//cout<<k<<"\n";
    			k=k*q;
    			sum+=k;
    		}
    		cout<<sum<<"\n";
    	}
    	t--;
    }
    return 0;
}
