#include<bits/stdc++.h>
using namespace std;
int main()
{
    std::string s;
    std::getline(std::cin,s);
    int x=s.size();int i;
    int coun[1000]={0};
    for(i=0;i<x;i++)
        coun[s[i]]++;
    int mx=-1;char ch;
    for(i=0;i<x;i++)
    {
        if(mx<coun[s[i]])
        {
            mx=coun[s[i]];
            ch=s[i];
        }
        else
            if(mx==coun[s[i]] && ch>s[i])
                ch=s[i];
    }
    cout<<ch<<" "<<mx;
    return 0;
}
