#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	int a,b,ar[10000000],i,flag=0;
	cin>>a>>b;
	for(i=0;i<a;i++){
		cin>>ar[i];
	}
	sort(ar,ar+a);
	int low = 0;
	int high = a-1;
	while(low<high){
		if(ar[low]+ar[high]==b){
			flag = 1;
			break;
		}
		if(ar[low]+ar[high]>b){
			high--;
		}
		else{
			low++;
		}
		
	}
	if(flag == 1){
		cout<<"YES"<<endl;
	}
	else{
		cout<<"NO"<<endl;
	}
	
	return 0;
}
