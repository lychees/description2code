#include <iostream>
using namespace std;

int main()
{
    int n,c1=0,c2=0,m,i;
    int temp;
    int a[100],b[100];
    cin>>n;
    for(i=0;i<n;i++)
    {
    	 cin>>a[i];
    }
    for(i=0;i<n;i++)
    {
    	cin>>b[i];
    }
	i=0;
    while(i!=n)
    {
    	if(a[i]==b[i])
    	{
    		c2++;
    		i++;
    	}
    	else
    	{
    		temp=a[i];
    		for(int k=i;k<n;k++)
    		{
    		a[k]=a[k+1];
    		}
    		a[n-1]=temp;
    		c1++;
    		c2=0;
    		i=0;
    	}
    	
    }
    m=c1+c2;
    cout<<m;
    return 0;
}
