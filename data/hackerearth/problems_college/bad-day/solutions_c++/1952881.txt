#include<iostream>
#include<stdlib.h>
using namespace std;
long long int fib(long long int n){
	if(n==1){
		return 1;
	}
	else if(n==2){
		return 2;
	}
	else{
		return(fib(n-1) +fib(n-2));
	}
}
int main(){
	long long int t;
	cin>>t;
	while(t--){
		long long int n;
		cin>>n;
		long long int ans=fib(n);
		cout<<ans<<"\n";
	}
	return 0;
}
