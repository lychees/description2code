#include <iostream>
using namespace std;

int main()
{
 	int t;
 	cin>>t;
 	int n,c,m,choc,wrap;
 	while(t--){
 		choc = 0;
 		cin>>n>>c>>m;
 		choc += (int)n/c;
 		wrap = (int)n/c;
 		while(wrap >= m){
 			choc += (int)wrap / m;
 			wrap = (int)wrap/m + (wrap % m);
 		}
 		cout<<choc<<endl;
 	}
    return 0;
}
