#include <iostream>
#include <algorithm>
using namespace std;
bool check(long a[],long n,long mid,long k){
	long l=0,h=1;
	while(h<n&&k){
		while(a[h]-a[l]<=2*mid&&h<n)h++;
		k--;
		if(k&&h!=n){l=h;h=l+1;}
	}
	if(l==n-1&&k)return 1;
	if(h==n)return 1;
	return 0;
}
int main()
{
    long n,k,low,high,mid,min;cin>>n>>k;
    long a[n];for(low=0;low<n;low++)cin>>a[low];
    sort(a,a+n);low=0;high=a[n-1];
    while(low<high){
    	mid=(low+high)/2;
    	if(check(a,n,mid,k))min=high=mid;
    	else low=mid+1;
    }
    
    cout<<min;
    return 0;
}
