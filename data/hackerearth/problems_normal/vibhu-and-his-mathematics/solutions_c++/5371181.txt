//constants
const int mxn=1e5+1;
const int MOD=1e9+7;
using namespace std;
int DP[1000001];
int solve(int n)
{
	if(n==0|| n==1)
		return 1;
	if(DP[n]!=-1)
		return DP[n];
	return DP[n]=(solve(n-1)+1LL*(n-1)*solve(n-2))%MOD;

}
int main()
{
	int t=1;
	sc1(t);
	memset(DP,-1,sizeof(DP));
	while(t--)
	{
		int n;
		sc1(n);
		printf("%d\n",solve(n));
	}
	return 0;
}
