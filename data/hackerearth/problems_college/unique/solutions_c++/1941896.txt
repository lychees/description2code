#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	int t;
	ll n,i,j,cnt;
	cin>>t;
	while(t--)
	{
		cin>>n;
		ll a[n+1];
		for(i=0;i<n;i++)
		cin>>a[i];
		sort(a,a+n);
		cnt=1;
		for(i=0;i<n-1;i++)
		{
			if(a[i]!=a[i+1])
			cnt++;
		}
		cout<<cnt<<endl;
	}
	return 0;
}
