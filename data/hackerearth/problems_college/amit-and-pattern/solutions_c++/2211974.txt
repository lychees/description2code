#include <iostream>
#include <string>
using namespace std;

int main()
{
	int n,m;
	cin>>n>>m;
	int a[n][m];
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			cin>>a[i][j];
			
	string ans="";
	for(int j=0;j<m;j++){
		int i=0;
		int k=j;
		ans+=to_string(a[i][j])+" ";
			while(k>0 && i<=n-1){
				
				i++; k--;
				if(k>=0 && i<n)
					ans+=to_string(a[i][k])+" ";
			    //cout<<"i="<<i<<"k="<<k<<" "<<a[i][k]<<endl;
			}
		cout<<ans<<endl;
		ans="";
	}
	
	ans="";
	for(int i=1;i<n;i++){
		int j=m-1;
		int k=i;
		ans+=to_string(a[i][j])+" ";
		while(k<n-1 && j>=0){
			k++;j--;
			if(k<=n-1 && j>=0)
				ans+=to_string(a[k][j])+" ";
		}
		cout<<ans<<endl;
		ans="";
	}
	
	return 0;
}
