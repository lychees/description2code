#include <iostream>
using namespace std;

int main()
{
    size_t Skipped = 0;
	size_t score = 0;
	size_t nQuestions = 0 ; 
	size_t DifficultyLevel = 0;
	
	std::cin >> nQuestions; 
	std::cin >> DifficultyLevel; 
	size_t QuestionPaper; 
	size_t Loop = 0;
	for (Loop = 0; Loop < nQuestions; ++Loop)
	{
		std::cin >> QuestionPaper;
		if (QuestionPaper > DifficultyLevel)
		{
			if (++Skipped > 1)
			{
				break;
			}
		}
		else
		{
		++score;
		}
	}
	std::cout << score;
    return 0;
}
