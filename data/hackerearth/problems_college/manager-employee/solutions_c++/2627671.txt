#include <iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++)
		cin>>a[i];
		for(int i=0;i<n;i++)
		{
			int c=0;
			for(int j=0;j<n-(i+1);j++)
			{
				if(a[j]<a[j+1])
				{
					int temp=a[j];
					a[j]=a[j+1];
					a[j+1]=temp;
					c=1;
				}
			}
			if(c==0)
			break;
		}
		int min=a[0]+a[n-1],max=a[0]+a[n-1];
		for(int i=1;i<n/2;i++)
		{
			if((a[i]+a[n-(i+1)])>max)
			max=a[i]+a[n-(i+1)];
			if((a[i]+a[n-(i+1)])<min)
			min=a[i]+a[n-(i+1)];
		}
		cout<<max-min<<endl;
	}
	return 0;
}
