#include <iostream>

using namespace std;

int main() {
    int a, b, n;
    
    scanf ("%d %d %d", &a, &b, &n);
    
    int now = 0;
    int cnt = 0;
    
    while ( now < n ) {
        now += a;
        
        if (now >= n) {
            cnt++;
            break;
        }
        
        now -= b;
        cnt++;
    }
    
    printf ("%d\n", cnt);
    
    return 0;
}
