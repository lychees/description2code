#include <bits/stdc++.h>
using namespace std;
#define MAX 4000101
bool prime[4000101]={0};
long long int sum = 0;
void gen_primes() 
    { 
      int i,j;
      int id = 1;
      for(i=0;i<=MAX;i++) prime[i] = 1; 
      prime[0]=prime[1]=0;
       for(i=2;i<=(int)sqrt(MAX);i++) 
         if (prime[i]) 
           for(j=i;j*i<MAX;j++) prime[i*j] = 0; 
          
           
    }

int main()
{
	gen_primes();
	int n;
	cin >> n;
	if(n < 2) puts("0");
	else
	{
		for(int i=1;i<n;i++)if(prime[i])sum+=i;
	    cout<<sum<<endl;
	}
}
