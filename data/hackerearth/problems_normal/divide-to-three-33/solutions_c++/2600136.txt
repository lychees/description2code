#include <bits/stdc++.h>
using namespace std;
int a[13];
int ans=20000;
int n;
void fun(int f,int s,int t,int cnt)
{
	if(cnt==n)
	{
		int q=max(f,max(s,t));
		ans=min(ans,q);
		return ;
	}
	fun(f+a[cnt],s,t,cnt+1);
	fun(f,s+a[cnt],t,cnt+1);
	fun(f,s,t+a[cnt],cnt+1);
}
int main()
{
    
    cin>>n;
    
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    	
    }
    fun(0,0,0,0);
    cout<<ans<<endl;
    return 0;
}
