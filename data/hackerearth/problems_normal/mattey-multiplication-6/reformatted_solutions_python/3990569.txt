from math import log


def factSumTwo(n, m, acc=''):
    if m == 0:
        return acc
    else:
        r = int(log(m)/log(2))
        acc += '('+str(n)+'<<'+str(r)+') + '
    return factSumTwo(n, m-2**r, acc)


t = int(raw_input())
for _ in range(t):
    n, m = map(int, raw_input().split(' '))
    print(factSumTwo(n, m)[0:-2])
