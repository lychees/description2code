t=int(raw_input())
lst=[]
while t:
	t-=1
	a,b,n=map(int,raw_input().split())
	lst=map(int,raw_input().split())
	lst.sort()
	cnta=0
	cntb=0
	
	for i in xrange(n):
		if lst[i]<=a:
			a-=lst[i]
			cnta+=1
		if lst[i]<=b:
			b-=lst[i]
			cntb+=1
	
	if cnta>cntb:
		print "Raghu Won"
	elif cnta<cntb:
		print "Sayan Won"
	else:
		print "Tie"
