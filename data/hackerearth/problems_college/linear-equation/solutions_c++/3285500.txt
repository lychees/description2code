#include <cstdio>
#include <cassert>

const int MAXN = 50000;

int n;
long long a[MAXN];

void Init() {
  assert(scanf("%d", &n) == 1);
  for (int i = 0; i < n; ++i)
    assert(scanf("%lld", a + i) == 1);
}

void Work() {
  long long sum = 0;
  for (int i = 0; i < n; ++i)
    sum += a[i];
  sum /= n - 1;
  for (int i = 0; i < n; ++i)
    printf("%d%c", int(sum - a[i]), i == n - 1 ? '\n' : ' ');
}

int main() {
  int cases;
  assert(scanf("%d", &cases) == 1);
  while (cases--) {
    Init();
    Work();
  }
  return 0;
}
