#include<bits/stdc++.h>
#define ll long long
using namespace std;
vector<pair<ll,ll> > v , vc;
ll k[100005];
int main()
{
    ll t,n,q,a,b;
   //freopen("input.txt","r",stdin);
   //freopen("output.txt","w",stdout);
    cin>>t;
    while(t--)
    {
        cin>>n>>q;
       for(int i=0;i<n;i++)
       {
           cin>>a>>b;
           v.push_back(make_pair(a,b));
 
       }
       sort(v.begin(),v.end());
       ll l = v[0].first;
       ll r = v[0].second;
       for(int i=1;i<v.size();i++)
       {
         if(v[i].first<= r)
            r=max(r,v[i].second);
         else
         {
             vc.push_back(make_pair(l,r));
             l=v[i].first;
             r=v[i].second;
         }
 
 
       }
       ll kth;
 
       vc.push_back(make_pair(l,r));
       for(int i=0;i<vc.size();i++)
        k[i]=vc[i].second-vc[i].first+1;
       for(int i=1;i<vc.size();i++)
        k[i]+=k[i-1];
       while(q--)
       {
           cin>>kth;
           if(kth> k[vc.size()-1])
            cout<<"-1\n";
           else
           {
 
           for(int i=0;i<vc.size();i++)
           {
               if(k[i]>=kth)
               {
                   if(i>0)
                   kth-=k[i-1];
                   cout<<vc[i].first+kth-1<<"\n";
                   break;
               }
           }
       }
 
 
 
 
    }
    v.clear();
    vc.clear();
    }
}
