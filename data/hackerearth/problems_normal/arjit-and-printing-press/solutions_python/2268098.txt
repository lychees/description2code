test = input()
while test:
    test -= 1
    org = raw_input()
    org = list(org)
    res = raw_input()
    res = list(res)

    l = len(res)
    res.sort()
    i = 0
    l2 = len(org)
    while i<l2 and l>0:
        if org[i] > res[0]:
            org[i] = res[0]
            l -= 1
            i += 1
            res.remove(res[0])
        else:
            i += 1
    print ''.join(org)
