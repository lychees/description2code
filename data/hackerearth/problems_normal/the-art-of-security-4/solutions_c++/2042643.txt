void setup()
{
	cin.tie(0);
	ios::sync_with_stdio(false);
}
ll fact(int x)
{
ll ans = 1;
for(ll i = 1;i<=x;i++)
ans = ans*i;
return ans;
}

int arr[10];
int sample[10];
int mark[10] ={0},cc;
ll calculate(int x,int a,int n)
{
if(a==n){
int f = 0,k =0;
int sum1 = 0,sum2 = 0;
while(x)
{
if(x&1){
sum1+=sample[k];
}
else
sum2+=sample[k];
k++;
x = x/2;
if(sum1 > sum2)
f = 1;
}
if(f==0)
cc++;
return 0;
}
for(int i = 0; i < n;i++)
{
    if(mark[i] == 0)
    {
        sample[a] = arr[i];
        mark[i] = 1;
        calculate(x,a+1,n);
        mark[i] = 0;
    }
}

}
int main()
{
	setup();
	int n;
	cin>>n;
	if(n==0){
		cout<<"1 0"<<endl;
		cout<<"We will win!\n";
		return 0;
		
	}
	ll sum =0;

	rep(i,0,n){
	cin>>arr[i];
	sum+=arr[i];
	}
    long long ans = 0;
	for(int i = 0; i < pow(2,n)-1;i++)
	{
            cl(mark);
            cc= 0;
            calculate(i,0,n);
            ans = ans + cc;
	}
	cout<<ans<<" "<<sum<<endl;
	if(ans > sum)
	{
	cout<<"We will win!\n";
	}
	else
	{
    cout<<"Got no way out!\n";
	}



}
