#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
   int t,n;
   cin>>t;
   for(int l=0;l<t;l++)
   {
       int h=0,v=0;
       cin>>n;
       char a[n][n];
       for(int i=0;i<n;i++)
       {
           for(int j=0;j<n;j++)
            cin>>a[i][j];
       }
       for(int i=0;i<n/2;i++)
       {
           for(int j=0;j<n;j++)
            if(a[i][j]!=a[n-i-1][j]) h=1;
       }
       for(int j=0;j<n/2;j++)
       {
           for(int i=0;i<n;i++)
            if(a[i][j]!=a[i][n-j-1]) v=1;
       }
       if(h==0 && v==0) cout<<"BOTH"<<endl;
       else if(h!=0 && v==0) cout<<"VERTICAL"<<endl;
       else if(h==0 && v!=0) cout<<"HORIZONTAL"<<endl;
       else cout<<"NO"<<endl;
   }
   return 0;
}
