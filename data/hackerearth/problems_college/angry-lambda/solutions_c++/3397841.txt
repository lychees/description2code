#include <iostream>
using namespace std;

int main()
{
    int t,n,k,i,j,p,count=0;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	count=0;
    	cin>>n>>k;
    	for(j=0;j<n;j++)
    	{
    		cin>>p;
    		if(p<=0)
    		count++;
    	}
    //	cout<<count<<endl;
    	if(count>=k)
    	cout<<"NO"<<endl;
    	else
    	cout<<"YES"<<endl;
    }
    return 0;
}
