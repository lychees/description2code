#include<bits/stdc++.h>
using namespace std;

const int N=1e5+5;

int a[N];

int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int n;
        scanf("%d",&n);
        int cnt=0;
        for(int i=0;i<n;i++) {
            scanf("%d",a+i);
            if(a[i]%2==0) {
                cnt++;
            }
        }
        printf("%d\012",cnt);
    }
    return 0;
}
