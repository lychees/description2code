def calcAnge(h,m):
    if h == 12: h = 0    
    if m == 60: m = 0
    if h ==  0 : h = 12
    
    hour_angle = int(0.5 * ( h*60 + m))
    minu_angle = int(6 * m)
    
    angle = abs(hour_angle - minu_angle)
    
    return int(min( abs(360 - angle), angle))


noc = int(raw_input())
while noc:
    noc-=1
    h,m = map(int,raw_input().split())
    print calcAnge(h,m)
