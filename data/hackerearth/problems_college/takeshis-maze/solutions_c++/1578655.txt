//#include "stdafx.h"
#include <iostream>
#include <string.h>
using namespace std;

int matrix[101][101];
int N, C;


int main()
{
	//freopen("DPMaze.txt", "r", stdin);

	int test_case;
	cin >> test_case;

	for (int tc = 1; tc <= test_case; ++tc)
	{
		memset(matrix, 0, sizeof(matrix));

		cin >> N >> C;

		for (int i = 0; i < N; ++i)
		{
			for (int j = 0; j < N; ++j)
			{
				cin >> matrix[i][j];
			}
		}


		for(int col = 1; col < N; ++col)
		{
			matrix[0][col] = matrix[0][col] + matrix[0][col -1];
		}

		for(int row = 1; row < N; ++row)
		{
			matrix[row][0] = matrix[row][0] + matrix[row-1][0];
		}


		for(int row = 1; row < N; ++row)
		{
			for(int col = 1; col < N; ++col)
			{
				if(matrix[row-1][col] > matrix[row][col-1])
				{
					matrix[row][col] = matrix[row][col] + matrix[row-1][col] ;
				}
				else
				{
					matrix[row][col] = matrix[row][col] + matrix[row][col-1] ;
				}
			}
		}

		if(matrix[N-1][N-1] >= C)
		{
			cout << matrix[N-1][N-1] - C << endl;
		}
		else
		{
			cout << -1 << endl;
		}
	}
	//int result = CheckMatrixPattern();

	return 0;
}
