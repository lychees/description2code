def testCase():
    n = int(input())
    globalParity = False
    result = "Yes"
    for j in xrange(n):
        r, c, x = [int(x) for x in raw_input().split()]
        posParity = (r + c) % 2
        if not globalParity:
            globalParity = abs(posParity - x % 2) + 1
        else:
            if (globalParity != abs(posParity - x % 2) + 1):
                result = "No"
    print(result)


N = int(input())
for i in xrange(N):
    testCase()
