#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    long long n,sum=0;
    cin>>n;
    long long ar1[n],ar2[n];
    for(long long i=0;i<n;++i)
    	cin>>ar1[i];
    for(long long i=0;i<n;++i)
    	cin>>ar2[i];
    sort(ar1,ar1+n);
    sort(ar2,ar2+n);
    for(long long i=0;i<n;++i)
    	sum+=ar1[i]*ar2[i];
    cout<<sum;
    return 0;
}
