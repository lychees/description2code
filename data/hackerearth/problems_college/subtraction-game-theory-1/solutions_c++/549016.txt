#include<stdio.h>
#include <iostream>
 using namespace std;
int main() {
	int t;
	scanf("%d",&t);
	while(t--)
	{
	    int n,k,p,q;
	    scanf("%d %d",&n,&k);
	    p=(n-1)/(k+1);
	    q=((k+1)*p);
	    if(q==n-1)
	    printf("Nandu\n");
	    else
	    printf("Rani\n");
	}
	return 0;
}
