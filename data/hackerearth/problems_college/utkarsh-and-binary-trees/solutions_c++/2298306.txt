typedef vector<int> vi;

typedef long long int lld;
typedef unsigned long long int llu;

lld doit(lld n,lld h,vector<vector<long long int> > &dp,int root , long long int &sum)
{
	if(n==0)
		return 1;
	if(h==0)
		return 0;

	if(dp[n][h]!=-1)
		return dp[n][h];
	dp[n][h]=0;
	lld temp;
	for(int i =1;i<=n;i++)
	{
		temp = doit(i-1,h-1,dp,1,sum)*doit(n-i,h-1,dp,1,sum);
		dp[n][h] += temp;
		dp[n][h] %=1000000007;
		if(root==0)
		{
			temp%=1000000007;
			sum+=	temp*(long long int) i;
			sum%=1000000007;
		}
			
	}
	return dp[n][h];
}

int main()
{
	long long int n,h,sum=0;
	vector <vector<long long int> > dp(501,vector<long long int>(501,-1));
	slld(n);
	slld(h);
	doit(n,h,dp,0,sum);
	plld(sum);
	return 0;
}
