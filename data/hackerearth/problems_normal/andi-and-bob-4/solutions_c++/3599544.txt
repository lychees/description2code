#include<iostream>
using namespace std;

int
main ()
{
  int tc = 0;
  cin>>tc;
  while (tc--)
    {
      long long N;
      cin >> N;
      long long *inp = new long long[N];
      long long *arr = new long long[N];

      for (long long i = 0; i < N; i++)
	{
	  cin >> inp[i];
	}
      long long fodd = 0, todd = 0;
      long long feven = 0, teven = 0, temp = 0;

      if (inp[0] % 2 == 0)	//even
	arr[0] = 1;
      else
	arr[0] = 0;

      //cout << arr[0] << endl;
      for (long long i = 1; i < N; i++)
	{
	  if (inp[i] % 2 == 0)
	    arr[i] = arr[i - 1];
	  else
	    arr[i] = !arr[i - 1];
	  //cout << arr[i] << endl;
	}

      for (long long i = 0; i < N; i++)
	{
	  if (arr[i] == 0)
	    todd++;
	  else
	    teven++;
	}
      fodd = fodd + todd;
      feven = feven + teven;
      //cout << "Subsets starting from " << inp[0] << " have " << todd <<
      //  "odd sums and " << teven << " sums\n";
      for (long long i = 1; i < N; i++)
	{
	  if (inp[i - 1] % 2 != 0)	//last num is odd
	    {
	      todd--;
	      temp = todd;
	      todd = teven;
	      teven = temp;
	    }
	  else
	    {
	      teven--;
	    }
	  fodd = fodd + todd;
	  feven = feven + teven;
	  //cout << "Subsets starting from " << inp[i] << " have " << todd <<
	  //"odd sums and " << teven << " sums\n";
	}

      //cout << fodd << "\t" << feven << endl;

      if (fodd > feven)
	cout << "BOB\n";
      else if (feven > fodd)
	cout << "ANDI\n";
      else
	cout << "BOB\n";
    }
}
