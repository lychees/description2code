const int N=1e5+10;
ll cases,n,c,res;
char s[N];





int main()
{
	std::ios::sync_with_stdio(0);
	cin.tie(0);
	cin>>cases;
	for(int kase=1;kase<=cases;kase++)
	{
		cin>>s;
		n=strlen(s);
		res=0;
		for(int i=0;i<n;i++)
		{
			c=0;
			while((s[i]>='a'&&s[i]<='c')||(s[i]>='A'&&s[i]<='C')&&i<n)
			{
				c++;
				i++;
			}
			res+=(c*(c+1))/2;
		}
		cout<<res<<"\n";
	}
	return 0;
}
