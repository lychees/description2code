#include <iostream>
using namespace std;

int main()
{
    long int T, R, G, B;
    cin>>T>>R>>G>>B;
    long int redOff=1, greenOff=2, blueOff=4, bulbState;
    long int time_visible[8];

    for(int i=0;i<8;i++)
        time_visible[i] = 0;

    for(long int i=0; i<T; i++) {

        redOff=4; greenOff=2; blueOff=1;
        if((i/R)%2 == 0)
            redOff = 0;
        if((i/G)%2 == 0)
            greenOff = 0;
        if((i/B)%2 == 0)
            blueOff = 0;
        bulbState = redOff + greenOff + blueOff;
        switch(bulbState) {
            case 0 : time_visible[7]++;break;

            case 1 : time_visible[2]++;break;

            case 2 : time_visible[1]++;break;

            case 3 : time_visible[4]++;break;

            case 4 : time_visible[0]++;break;

            case 5 : time_visible[5]++;break;

            case 6 : time_visible[3]++;break;

            case 7 : time_visible[6]++;break;
        }
    }

    for(int i=0;i<8;i++)
        cout<<time_visible[i]<<" ";
    //cout<<time_visible[7];
    cout<<endl;
    return 0;
}
