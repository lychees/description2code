#include <iostream>
using namespace std;

int main()
{
    int t, n, k, x, z;
    for(cin >> t;t--;)
    {
    	cin >> n >> k;
    	z = 0;
    	x = 1;
    	for(int i = 2;i <= k;++i)
    	{
    		z++;
    		if(z == x)
    			z++;
    		if(z >= 10) z = 0, x++;
    	}
    	if(x > 9)
    	{
    		cout << -1 << endl;
    		continue;
    	}
    	for(int i = 0;i < n;++i)
    	{
    		if(i % 2 == 0) cout << x;
    		else cout << z;
    	}
    	cout << endl;
    }
    return 0;
}
