typedef unsigned long long int uint64;

int main(){
	int t,n,i,j;
	string s;
	int cnt[30][111];
	cin>>t;
	while(t--){
		cin>>n;
		for(i=0;i<30;i++){
			for(j=0;j<110;j++)
			cnt[i][j]=0;
		}
		for(i=0;i<n;i++){
		cin>>s;
		for(j=0;j<s.length();j++){
			cnt[int(s[j])-97][i]=1;
		}}
		for(j=0;j<30;j++){
			int val=0;
			for(i=0;i<n;i++)
			val+=cnt[j][i];
			cnt[j][n]=val;
		}
		int ans=0;
		char x;
		for(j=0;j<30;j++){
			if(cnt[j][n]>ans){
				ans=cnt[j][n];
				x=char(j+97);
			}
		}
		cout<<x<<endl;
	}
	
	return 0;
}
