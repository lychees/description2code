#include <bits/stdc++.h>
 
#define LLI long long int
#define LD long double
#define PB push_back
#define MP make_pair
#define FORi(i, a, b) for(int i = a; i < b ; ++i)
#define FORd(i, a, b) for(int i = a; i > b ; --i)
 
using namespace std;
 
const int mod = 1e9+7;

int main() {
	int n = 1500000;
	vector<char> prime (n+1, true);
	prime[0] = prime[1] = false;
	for (int i=2; i<=n; ++i)
		if (prime[i])
			if (i * 1ll * i <= n)
				for (int j=i*i; j<=n; j+=i)
					prime[j] = false;
	vector<LLI> v;
	n = 0;
	FORi(i,0,prime.size())
		if (prime[i]){
			n++;
			if (prime[n])
				v.PB(i);
		}
	int t,m;
	cin >> t;
	while(t--){
		scanf("%d%d", &n, &m);
		printf("%d\n", v[n-1]*v[m-1]%mod);
	}
	return 0;
}
