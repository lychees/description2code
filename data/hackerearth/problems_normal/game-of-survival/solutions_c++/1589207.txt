#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int n,T,i;
    cin >> T;
    while(T--)
    {
    	cin >> n;
    	int a[n],b[n];
    	
    	for(i=0 ; i<n ; i++)
    	cin >> a[i];
    	
    	for(i=0 ; i<n ; i++)
    	cin >> b[i];
    	
    	sort(a,a+n);
    	sort(b,b+n);
    	
    	int flag=0;
    	
    	for(i=0 ; i<n ; i++)
    	{
    		if(a[i]<b[i])
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==0)
    	cout <<"YES"<< endl;
    	else
    	cout <<"NO"<< endl;
    }
    return 0;
}
