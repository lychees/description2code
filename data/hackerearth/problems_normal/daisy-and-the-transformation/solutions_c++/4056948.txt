#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t>0){
    	string s1,s2;
    	int m;
    	cin>>s1;
    	cin>>s2;
    	cin>>m;
    	int n1 = s1.length();
    	int n2 = s2.length();
    	if(n1!=n2){
    		cout<<"No"<<endl;
    	}
    	else{
    		int count = 0;
    		for(int i = 0;	i<n2;	i++){
    			if(s1[i]!=s2[i]){
    				count++;
    			}
    		}
    		int temp = 1;
    		if(m>count)
    		 temp = (m-count)%2;
    		if(count==m || temp == 0){
    			cout<<"Yes"<<endl;
    		}
    		else{
    			cout<<"No"<<endl;
    		}
    	}
    	t--;
    }
    return 0;
}
