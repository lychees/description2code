#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,i,j,k;
	int ans,ans2;
	string str;
	cin>>t;
	while (t--)
	{
	cin>>n;
	cin>>str;
	ans=ans2=0;
	j=k=0;
	for (i=0;i<n;i++)
	if (str[i]=='0')
	j++;
	else
	k++;
	for (i=0;i<j;i++)
	if (str[i]!='0')
	ans++;
	for (i=j;i<n;i++)
	if (str[i]!='1')
	ans++;
	for (i=0;i<k;i++)
	if (str[i]!='1')
	ans2++;
	for (i=k;i<n;i++)
	if (str[i]!='0')
	ans2++;
	ans/=2;
	ans2/=2;
	cout<<min(ans,ans2)<<endl;
	}
	return 0;
	}
