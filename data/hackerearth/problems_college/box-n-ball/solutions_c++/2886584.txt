#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;
typedef unsigned long long int int32;

int main()
{
	int T;
	int32 L,B,H;
	cin >> T;
	while(T--)
	{
		cin >> L >> B >> H;
		if(L <= B && L <= H)
		{
			if(L % 2 == 1)
				cout << L/2 << ".5" << endl;
			else
				cout << L/2 << ".0" << endl;
		}
		else if(B <= L && B <= H)
		{
			if(B % 2 == 1)
				cout << B/2 << ".5" << endl;
			else
				cout << B/2 << ".0" << endl;
		}
		else
		{
			if(H % 2 == 1)
				cout << H/2 << ".5" << endl;
			else
				cout << H/2 << ".0" << endl;
		}
		
	}
    return 0;
}
