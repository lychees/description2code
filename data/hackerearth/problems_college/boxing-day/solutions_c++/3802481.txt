#include <iostream>
#include <set>
using namespace std;
using ULL = unsigned long long int;

int main()
{
	auto n = 0ULL, k = 0ULL;
	cin >> n >> k;
	auto islands = set<ULL> ();
	for (auto i = 1ULL; i <= n; ++i)	islands.insert(i);
	while (k--)
	{
		auto x = 0ULL, y = 0ULL;
		cin >> x >> y;
		islands.erase(y);
	}
	cout << islands.size() << endl;
    return 0;
}
