#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s;
    	cin>>s;
    	int n = s.size();
    	int j,k;
    	bool f;
    	for(int i=n/2;i<n;i++){
    		j=i+1,k=i-1;
    		if(s[i]==s[k])k--;
    		f = true;
    		while(j<n && k>=0){
    			if(s[j]!=s[k]){
    				f=false;
    				break;
    			}
    			j++;
    			k--;
    		}
    		if(f){
    			break;
    		}
    	}
    	int ans;
    	ans = k+1;
    	ans+=n;
    	cout<<ans<<endl;
    }
    return 0;
}
