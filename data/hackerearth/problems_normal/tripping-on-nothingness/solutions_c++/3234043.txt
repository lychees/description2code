#include <iostream>
using namespace std;

int find(int a,int b)
{  if(b==0)
    return a;
    else
    return find(b,a%b);}

int main()
{
    int i,m,q,s,a[100001],x;
    for(i=0;i<=100000;i++)
    a[i]=0;
    cin>>s>>q;
    for(i=0;i<q;i++)
    {cin>>x;
    m=find(s,x);
    if(a[m]==0)
    {cout<<m<<"\n";
    a[m]=1;}
    else
    cout<<"-1\n";}
    return 0;
}
