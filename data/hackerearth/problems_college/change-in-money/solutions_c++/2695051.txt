#include<iostream>
using namespace std;
int main()
{
    int n,m,i,j;
    cin>>n>>m;
    int coins[m];
    
    for(i=0;i<m;i++)
        cin>>coins[i];
    
    long int t[m+1][n+1];
    for(i=0;i<m+1;i++)
        t[i][0]=1;
    for(j=1;j<n+1;j++)
        t[0][j]=0;                  
    
    int k=m;
    while(k--)
    {
    	for(i=0;i<m-1;i++)
    	{
    		if(coins[i]>coins[i+1])
    		{
    			int temp=coins[i];
    			coins[i]=coins[i+1];
    			coins[i+1]=temp;
    		}
    	}
    }
    for(i=1;i<m+1;i++)
    {
        for(j=1;j<n+1;j++)
        {
            if(j>=coins[i-1])
               t[i][j]=t[i-1][j]+t[i][j-coins[i-1]];
            else
               t[i][j]=t[i-1][j];
        }
    }
    cout<<t[m][n];
}
