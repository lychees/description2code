#include<stdio.h>
#include<string.h>
int main()
{
    char ar[10000];
    int m,i,k,j;
    gets(ar);
    k=strlen(ar);
    scanf("%d",&m);
    j=0;
    for(i=0;i<k;i++)
    {
        printf("%c",ar[i]);
        j++;
        if(j==m)
        {
            printf("\n");
            j=0;
        }
    }
    return 0;
}
