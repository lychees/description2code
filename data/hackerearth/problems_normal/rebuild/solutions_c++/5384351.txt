struct Edge {
	int u,v;
	long long c;
	Edge() {}
	Edge(int u, int v, long long c):u(u),v(v),c(c) {}
	inline bool operator < (const Edge & rhs)const {return c<rhs.c;}
};
vector<vector<int>> V;
int root(int x) {
	while(id[x] != x){
		id[x] = id[id[x]];
		x = id[x];
	}
	return x;
}
map<int,int> Map;
long long kruskal(int M) {
	long long cost=0;
	Map.clear();
	for(auto E: V) {
		int a = root(-E[1]);
		int b = root(-E[2]);
		if(a!=b){
			id[a] = b;
			Map[-E[1]]++;
			Map[-E[2]]++;
			cost+=E[0];
		}
	}
	return cost;
}
