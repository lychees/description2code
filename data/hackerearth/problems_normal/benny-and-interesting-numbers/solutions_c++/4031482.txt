#include <bits/stdc++.h>
using namespace std;
#define ll long long
bool isp[1000009];
ll ps[1000009];
int pp = 0;
int main() 
{
	ios::sync_with_stdio(false);
	cin.tie(0);
	int t, i, j;
	ll n;
	bool z;
	isp[0] = true;
	isp[1] = true;
	for(i = 2; i <= 1000; i++)
	{
		if(!isp[i])
		{
			for(j = i+i; j <= 1000000; j+=i)
				isp[j] = true;
		}
	}
	ps[0] = 4;
	for(i = 3, pp = 1; i <= 316300; i+=2)
	{
		if(!isp[i])
			ps[pp++] = (ll)i*(ll)i;
	}
	cin >> t;
	while(t--)
	{
		cin >> n;
		z = true;
		for(i = 0; ps[i] <= n && i < pp; i++)
		{
			if(n % ps[i] == 0)
			{
				z = false;
				break;
			}
		}
		if(z)
			cout << "Yes\n";
		else
			cout << "No\n";
	}
	return 0;
}
