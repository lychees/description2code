dic = []
arr = []
length = []
word = raw_input()
n = int(raw_input())
for i in range(0, n):
    w = raw_input()
    dic.append(w)
for w in dic:
    if word in w:
        q = len(word)
        if w[0:q] == word:
            w = w.replace('word', '')
            arr.append(w)
            length.append(len(w))
if len(arr) != 0:
    min_len = min(length)
    index = length.index(min_len)
    print arr[index]
else:
    print word
