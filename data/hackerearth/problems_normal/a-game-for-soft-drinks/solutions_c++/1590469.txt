#include <bits/stdc++.h>
#define MAX 101
using namespace std;
char mat[MAX][MAX][MAX];
int ans = 0;
int notans = 0;

void pre()
{
	int  i , j , k;
	for(i=0;i<MAX;i++)
	{
		for(j=0;j<MAX;j++)
		{
			for(k=0;k<MAX;k++)
			{
				if((i>0 && mat[i-1][j][k] == 'L') ||(j>0 && mat[i][j-1][k] == 'L') ||(k>0 && mat[i][j][k-1] == 'L') ||
				(i>0&&j>0 && mat[i-1][j-1][k] == 'L') || (i>0 &&k>0&& mat[i-1][j][k-1] == 'L') ||(j>0&&k>0 && mat[i][j-1][k-1] == 'L') ||
				(i>0&&k>0&&j>0 && mat[i-1][j-1][k-1] == 'L') )
				{
					mat[i][j][k] = 'W';
					ans++;
				}else{
					mat[i][j][k] = 'L';
					notans++;
				}
			}
		}
	}
}
int main()
{
	int i , j , t;
	cin>>t;
	pre();
	while(t--)
	{
		
		int a, b , c;
		string inp;
		//cin>>inp;
		cin>>a>>b>>c;
		if(mat[a][b][c] == 'W'){
						cout << "Ashima" << endl;
				}
				else cout << "Aishwarya" << endl;
	}

	return 0;
}
