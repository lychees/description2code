#include <stdio.h>
int primes[100001];
int main()
{
    int i, j;
    for(i=0;i<100000;i++)
    primes[i]=0;
    primes[1]=1;
    for(i=2;i*i<=100000;i++)
    {
    	if(primes[i]==0)
    	for(j=i*i;j<=100000;j+=i)
    	primes[j]=1;
    }
    int t, n;
    scanf("%d", &t);
    while(t--)
    {
    	scanf("%d", &n);
    	if(n==3)
    	{printf("3\n");continue;}
    	if(primes[n])
    	printf("1\n");
    	else
    	printf("2\n");
    }
    return 0;
}
