using namespace std;
int pow(int a,int p,int n)
{
	int result=1;
	int d=a%n;
	while(p)
	{
		if(p&1)
		{
			result=(result%n*d%n)%n;
		}
		d=(d%n*d%n)%n;
		p=p>>1;
	}
	return result;
}
int main()
{
	//f_in("in6.txt");
	//f_out("in6out.txt");
	int a,n,t;
	cin>>t;
	assert(1<=t && t<=1000);
	while(t--)
	{
		cin>>a>>n;
		assert(1<=a && a<=1000000);
		assert(3<=n && n<=1000000);
		int d=pow(a,(n-1)/2,n);
		if(d==1)
		{
			cout<<"YES"<<endl;
		}
		else
		{
			cout<<"NO"<<endl;
		}
	}
	return 0;
}
