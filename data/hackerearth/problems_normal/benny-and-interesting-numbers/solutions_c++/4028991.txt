#include <bits/stdc++.h>
using namespace std;
bool a[320010];
vector<long long int> sqr;

void seive(){
	for(int i = 0; i<= 320008; i++)
	a[i] = false;
	
	
	for(int i = 4; i<=320000; i+=2)
	{
		a[i] = true;
		
	}
	
	
	
	for(int i = 3; i<=320000; i+=2)
	{
		if(!a[i])
		{
			for(int j = 2*i; j<=320000; j+=i)
			{
				a[j] = true;
				
			}
			
		}
		
	}
	
	
	for(long long int i = 2; i<=320000; i++)
	{
		if(a[i]==false)
		{
			sqr.push_back(i*i);
		}
		
		
	}
	
	
	
}



int main()
{
    seive();
    int t;
    cin>>t;
    while(t--)
    {
    	long long x;
    	cin>>x;
    	vector<long long>::iterator it;
    	int flag=1;
    	for(it=sqr.begin();it!=sqr.end();it++)
    	{
    		long long m = *it;
    		if(x%m==0)
    		{
    			cout<<"No\n";
    			flag=0;
    			break;
    		}
    		
    		if(m>x)
    		break;
    	}
    	
    	if(flag)
    	cout<<"Yes\n";
    }
    
    return 0;
}
