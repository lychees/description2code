#define ll long long int

#define sd(d)   scanf("%d", &d)

int main()
{
   int t;
   ll n, tmp, cnt;

    sd(t);

    w(t) {
   cin >> n;

        tmp = n;
        cnt = 0;

        while (tmp) {
            if (tmp % 2 == 1) {

        }

    }
