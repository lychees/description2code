#include<iostream>
using namespace std;
#include<string.h>
char s1[1000006],s2[1000006],r1[1000006],r2[1000006];
int a[1000006],b[1000006];
int main()
 {int t,i,j,l1,l2,p1,p2;
 cin>>t;
 while(t--) 
  { cin>>s1>>s2;
  p1=0;p2=0;
  l1=strlen(s1);
  l2=strlen(s2);
    for(i=0;i<l1;i++)
      {if(s1[i]!='#')
         {a[p1]=i;
         r1[p1++]=s1[i];
        }
        
      }
    r1[p1++]='\0';
    for(i=0;i<l2;i++)
      {if(s2[i]!='#')
         {b[p2]=i;
         r2[p2++]=s2[i];
        }
        
      }
    r2[p2++]='\0';
    if(strcmp(r1,r2))
        cout<<"No"<<endl;
    else
      {l1=strlen(r1);
	  for(i=0;i<l1;i++)
	     {if(r1[i]=='A' && a[i]<b[i])
	        break;
	      if(r1[i]=='B' && a[i]>b[i])
	        break;
	        
	     }
	 if(i<l1)
	    cout<<"No"<<endl;
	else
      cout<<"Yes"<<endl;
     }
   }
 }
