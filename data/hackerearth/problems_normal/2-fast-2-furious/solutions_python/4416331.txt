b = map(int , raw_input().split())
max1 = abs(d[0])
max2 = abs(b[0])
for i in range(1 , n):
        max1 = max( max1 , abs(d[i] - d[i-1] ) )
        max2 = max ( max2  , abs( b[i] - b[i-1] ) )
if max1 > max2:
