#include <iostream>
#include <algorithm>
#include <string.h>
using namespace std ;

int main(){

  int T ;
  char W[10005], R[10005] ;

  cin >> T ;

  while(T--){
    
    cin>>W>>R ;

    sort(R,R + strlen(R)) ;

    int pos = 0;
    int empty = 0 ;
    
    for(int i = 0;i<strlen(W);i++){
      if(W[i]>R[pos] && empty==0){
	cout<<R[pos] ;
	pos++ ;
	if(pos==strlen(R))
	  empty = 1 ;
      }
      else
	cout<<W[i] ;
    }

    cout<<endl ;
    
  }
  return 0 ;
}
