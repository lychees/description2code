#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

stack <ll> s;
ll h[100005], l[100005], r[100005], mod = 1e9 + 7;
int main()
{
    ll t, n, a, ans, ansi, i, j, x, p;
    cin>>t;
    while(t--){
    	cin>>n;
    	for(i = 1; i<=n; i++){
    		cin>>h[i];
    	}
    	
    	while(!s.empty()){
    		s.pop();
    	}
    	
    	for(i = n; i>0; i--){
    		while(!s.empty()&&h[s.top()]<h[i]){
    			s.pop();
    		}
    		if(s.empty()){
    			r[i] = n;
    		}
    		else{
    			r[i] = s.top();
    		}
    		s.push(i);
    	}
    	
    	while(!s.empty()){
    		s.pop();
    	}
    	for(i = 1; i<=n; i++){
    		while(!s.empty()&&h[s.top()]<h[i]){
    			s.pop();
    		}
    		if(s.empty()){
    			l[i] = 1;
    		}
    		else{
    			l[i] = s.top();
    		}
    		s.push(i);
    	}
    	ans = 0;
    	ansi = 1;
    	for(i = 1; i<=n; i++){
    		p = (long long)(r[i] - l[i]) * i;
    		//cout<<i<<" "<<l[i]<<" "<<r[i]<<endl;
    		p = p%mod;
    		if(p > ans){
    			ans = p;
    			ansi = i;
    		}
    	}
    	cout<<ansi<<endl;
    }
    return 0;
}
