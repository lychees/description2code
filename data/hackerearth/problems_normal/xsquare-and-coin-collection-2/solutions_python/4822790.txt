t = int(raw_input())

for i in range(t):
	(n, k) = map(int, raw_input().strip().split(' '))
	arr = map(int, raw_input().strip().split(' '))
	mask = [0]*n
	for i in range(n):
		if arr[i]<=k:
			mask[i]=1

	i=0
	ans=0
	while i<n:
		if mask[i]==0:
			i+=1
		else:
			temp=0
			while i<n and mask[i]!=0:
				temp+=arr[i]
				i+=1
			ans = max(ans, temp)
	print ans
