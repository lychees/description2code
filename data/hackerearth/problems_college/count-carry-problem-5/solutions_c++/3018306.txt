#include<iostream>
#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<vector>
#include<list>
#include<map>
#include<algorithm>
#include<math.h>
#include<limits.h>
#include<stack>
#include<queue>
#include<set>
#define ll long long
using namespace std;
char s[1000010],t[1000010];
int main()
{
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		scanf("%s%s",s,t);
		ll int ans=0,carry,temp;
		ll int ls,lt;
		ls=strlen(s);
		lt=strlen(t);
		ls--;
		lt--;
		carry=0;
		temp=0;
		while(1)
		{
			if(ls>=0 && lt>=0)
			{
				temp=(s[ls]-'0')+(t[lt]-'0')+carry;
				if(temp>=10)
				{
					ans++;
					carry=temp/10;
				}
				ls--;
				lt--;
			}
			else if(ls>=0)
			{
				temp=(s[ls]-'0')+carry;
				if(temp>=10)
				{
					ans++;
					carry=temp/10;
				}
				ls--;
			}
			else if(lt>=0)
			{
				temp=(s[lt]-'0')+carry;
				if(temp>=10)
				{
					ans++;
					carry=temp/10;
				}
				lt--;
			}
			else
			{
				break;
			}
		}
		if(ans==0)
		{
			printf("No carry operation");
		}
		else if(ans==1)
		{
			printf("%lld carry operation",ans);
		}
		else
		{
			printf("%lld carry operations",ans);
		}
		if(test!=0)
		cout<<endl;
	}
	//cout<<endl;
	return 0;
}
