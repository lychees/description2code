#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int T,t;
    long int N,n,num,i=0;
    vector<long int> x;
    cin>>T; t=T;
    while(t--){
    	x.clear();
    	cin>>N; n=N;
    	while(n--){
    		cin>>num;
    		x.push_back(num);
    	}
    	sort (x.begin(), x.end());
    	for(i=0;i<N;i++)
    	cout<<x[i]<<" ";
    	cout<<"\n";
    }
    return 0;
}
