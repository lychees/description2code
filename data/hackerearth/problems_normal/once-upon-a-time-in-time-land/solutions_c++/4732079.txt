#include <bits/stdc++.h>
using namespace std;

long long int n,arr[100000],k,dp[100000];

long long int solve(long long int i)
{
	if(i<0)
		return 0;
	if(dp[i]!=-1)
		return dp[i];
	long long int ans= max(solve(i-1),solve(i-k-1)+arr[i]);	
	dp[i]=ans;
	return dp[i];
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
    	cin>>n>>k;
    	for(int i=0;i<n;i++)
    		cin>>arr[i];
    	for(int i=0;i<=n;i++)
    		dp[i]=-1;
    	long long int ans=solve(n-1);	
    	cout<<ans<<endl;
	}	
    return 0;
}
