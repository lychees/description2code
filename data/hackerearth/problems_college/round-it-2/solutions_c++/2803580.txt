#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,b,r,ans;
    	cin>>n>>b;
    	if(n<b)
    	{
    	if(b&1) {if(n<=b/2) ans=0; else ans=b;}
    	else {if(n<b/2) ans=0; else ans=b;}
    	}
    	
    	else if(n>b) 
    	{
    	r=(n%b);
    	if(r<b-r) ans=n-r;
    	else ans=n+b-r;
    	}
    	
    	else ans=n;
    	cout<<ans<<endl;	
    }
    return 0;
}
