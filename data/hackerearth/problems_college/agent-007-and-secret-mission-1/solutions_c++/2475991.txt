#include <iostream>
using namespace std;

int main()
{
    int test;
    cin>>test;
    while(test--) {
    	string s;
    	cin>>s;
    	
		int set=0;
		for(int i=0; i<s.size(); i++) {
			if(s[i]=='1') set++;
		}
    	
    	long long n;
    	cin>>n;
    	
    	n = (n*(n+1))/2;
    	if(n&1) cout<<s.size()-set<<endl;
    	else cout<<set<<endl;
    }
    return 0;
}
