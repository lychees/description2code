from itertools import combinations


def combinations_with_sum(sequence, desired_sum):
    results = []
    for i in range(len(sequence)+1):
        results.extend([combination for combination in combinations(
            sequence, i) if sum(combination) == desired_sum])
    if results == []:
        return 'NO'
    else:
        return 'YES'


a = input()
while a > 0:
    input()
    m = map(int, raw_input().split(' '))
    n = input()
    print combinations_with_sum(m, n)
    a -= 1
