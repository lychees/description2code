#include <bits/stdc++.h>
using namespace std;
#define lli long long int
#define md 1000000007
vector<int>v[100005];
lli A[100005];
void dfs(int in,int p)
{
	lli ans=0;
	for(int i=0;i<v[in].size();i++)
	{
		if(v[in][i]==p) continue;
		dfs(v[in][i],in);
		ans%=md;A[v[in][i]]%=md;
		lli tmp=(ans*A[v[in][i]])%md;
		ans=(ans+A[v[in][i]])%md;
		ans=(ans+tmp)%md;
	}
	A[in]=(ans+1)%md;
}
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n-1;i++)
	{
		int x,y;
		cin>>x>>y;
		v[x].push_back(y);
		v[y].push_back(x);
	}
	dfs(0,-1);
	lli ans=0;
	for(int i=0;i<n;i++)
	{
		ans=(ans+A[i])%md;
		ans%=md;
	}

	cout<<(ans+1)%md<<"\n";
	return 0;
}
