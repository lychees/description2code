for i in range(int(raw_input())):
    a, b = map(int, raw_input().split())
    c = 0
    while a > 0 and b > 0:
        if (a % 10+b % 10) >= 10:
            c += 1
        a /= 10
        b /= 10
    if c == 0:
        print "No carry operation"
    elif c == 1:
        print str(c)+" carry operation"
    else:
        print str(c)+" carry operations"
