#include <iostream>
#include <vector>
#include <string.h>
#include <algorithm>
using namespace std;
typedef pair<int,int> pi;
typedef vector<pair<int,int> > vpi;
bool compare(const pi a,const pi b){
	if(a.second>b.second)
		return false;
	else if(a.second==b.second){
		if(a.first>b.first)
		return true;
		else
		return false;
	}
	return true;
}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s;
    	cin>>s;
    	int hasher[26];
    	memset(hasher,0,sizeof(hasher));
    	for(int i=0;i<s.length();i++){
    		hasher[s[i]-'a']++;
    	}
    	vpi fi(26);
    	for(int i=0;i<26;i++){
    		fi[i].first='a'+i;
    		fi[i].second=hasher[i];
    	}
    	sort(fi.begin(),fi.end(),compare);
    	for(int i=0;i<26;i++)
    		cout<<(char)fi[i].first<<" ";
    		cout<<"\n";
    }
}
