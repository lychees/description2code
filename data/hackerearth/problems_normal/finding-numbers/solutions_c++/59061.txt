#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t;
    long long int z,m,n,answer;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	answer=0;
    	cin>>z>>m>>n;
    	for(long long int j=1;j*j<=z && j<=min(n,m);j++)
    	{
    		if(!(z%j))
    		{
    			if(j<=n && (z/j)<=m) answer++;
    			if(j<=m && (z/j)<=n && (j!= z/j)) answer++;
    		}
    	}
    	cout<<answer<<endl;
    }
}
