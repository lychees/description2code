#include <iostream>
using namespace std;
#define ll long long

int main()
{
	ll t,i,j,k,l,m,n;
	cin>>t;
	int a[1000006];
	int b[1000006];
	while(t--){
		cin>>n;
		
		for(i=0;i<=n;i++){
			a[i]=i;
			b[i]=0;
		}
		for(i=0;i<=n;i++){
			if(i>=10){
				//a[i]=min(a[i],a[i-10]);
				if(a[i]>a[i-10]){
					a[i]=a[i-10];
					b[i]=b[i-10]+1;
				}
				
			}
			if(i>=12){
				//a[i]=min(a[i],a[i-12]);
				if(a[i]>=a[i-12]){
					a[i]=a[i-12];
					b[i]=b[i-12]+1;
				}
			}
		}
		if(a[n]==0)
		cout<<b[n]<<endl;
		else{
			cout<<-1<<endl;
		}
		
	}
    return 0;
}
