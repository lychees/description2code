#include<stdlib.h>
#include<iostream>
using namespace std;

int compare(const void *a,const void *b){
    return *((int *)b) - *((int *)a);
}

int main(){
    int T;
    cin >> T;
    while(T--){
        int nR,nG;
        cin >> nR >> nG;
        int arr[nR],i;
        for(i=0;i<nR;i++){
                int num;
            cin >> num;
            arr[i] = num;
        }
        int myValue = arr[0];
        int myIndex;
        qsort(arr,nR,sizeof(int),compare);
        for(i=0;i<nR;i++){
            if(myValue == arr[i]){
                myIndex = i;
                break;
            }
        }
        int ans = myIndex/nG;
        cout << ans << endl;
    }
    return 0;
}
