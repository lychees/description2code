#include <iostream>
#include<cmath>
using namespace std;
int arr[1000000];
long long table[1000000]={0};
int prime(int n);
int prime(int n)
{
	if(n==2||n==3||n==5||n==7||n==11)
	return 0;
	if(n%2==0)
	return 2;
	int lim=sqrt(n);
	for(int i=3;i<=lim;i+=2)
	{
		if(n%i==0)
		return i;
	}
	if(n>2)
	return 0;
}
int main()
{
    //cout << "Hello World!" << endl;
    table[2]=0;
    for(int i=3;i<=1000000;i++)
    {
    	arr[i]=prime(i);
    	table[i]=table[i-1]+arr[i];
    	table[i]%=1000000007;
    }
    /*for(int i=2;i<20;i++)
    cout<<table[i]<<' ';
    cout<<endl;*/
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int l,r;
    	scanf("%d%d",&l,&r);
    	if(l==2)
    	printf("%lld\n",table[r]%1000000007);
    	else
    	printf("%lld\n",((table[r]-table[l-1])+1000000007)%1000000007);
    }
    return 0;
}
