#include <iostream>
using namespace std;

int main()
{
   int n,flag=0;
   char s[10001],c[5]={0};
   cin>>n;
   cin>>s;
   for(int i=0;i<n;i++)
   {
   	if(s[i]=='a')
   	c[0]++;
   	else if(s[i]=='e')
   	c[1]++;
   	else if(s[i]=='i')
   	c[2]++;
   	else if(s[i]=='o')
   	c[3]++;
   	else if(s[i]=='u')
   	c[4]++;
   }
   for(int i=0;i<5;i++)
   {
   	if(c[i]==0)
   	{
   		flag=1;
   		break;
   	}
   }
   if(flag)
   cout<<"NO\n";
   else
   cout<<"YES\n";
}
