t = int(raw_input())
for tpa in range(0, t):
    n = int(raw_input())
    se = set()
    ans = 0
    count = 0
    for tpb in range(0, n):
        x, y = map(int, raw_input().split(' '))
        if y == 0:
            ans = max(ans, count)
            count = 0
        elif x in se:
            continue
        else:
            count += 1
            se.add(x)
    ans = max(ans, count)
    print(ans)
