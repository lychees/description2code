#include <iostream>
#include<string.h>
using namespace std;

int main()
{	int t;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>s;
		int k=s.length();
		//cout<<k;
		int flag=0,i,j;
		long long int pro=1;
		for(i=0,j=k-1;i<k/2;++i,--j)
		{
			if(s[i]!=s[j])
			{	flag=1;
				break;
			}
		}
		if(flag==1)
		{
			for(i=0;i<k;++i)
			{
				pro=pro*(s[i]-'a'+1);
			}
			cout<<pro<<"\n";
		}
		else
			cout<<"Palindrome\n";
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
