#include<bits/stdc++.h>
#include<algorithm>
using namespace std;
struct greedy
{
int starttime;
int journeytime;
int index;
};
struct greedy ar[100002];

bool comp(struct greedy a,struct greedy b)
{
return a.starttime<b.starttime;
}

main()
{
int n,m;
cin>>n>>m;
for(int i=0;i<n;i++)
  {cin>>ar[i].starttime>>ar[i].journeytime;
    ar[i].index=i;}
sort(ar,ar+n,comp);

int taxi[102];
int answer[100002];
for(int i=1;i<=m;i++)
  taxi[i]=0;
for(int i=0;i<n;i++)
{ bool check=false;
  for(int j=1;j<=m;j++)
     {
       if(taxi[j]<=ar[i].starttime)
       { check=true; answer[ar[i].index]=j;
         taxi[j]=ar[i].starttime+ar[i].journeytime;
         break;
       }

     }
     if(check==false)
      answer[ar[i].index]=-1;
}

for(int i=0;i<n;i++)
  cout<<answer[i]<<" ";


}
