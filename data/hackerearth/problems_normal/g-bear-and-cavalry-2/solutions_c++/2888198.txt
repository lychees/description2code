#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	int a[n],b[n],i;
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&a[i]);
    	}
    	int k = a[0];
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&b[i]);
    	}
    	int temp = a[n-1];
    	a[n-1] = a[0];
    	a[0] = temp;
    	sort(a,a+n-1);
    	sort(b,b+n);
    	int k1 = k*b[n-1];
    	int f = 0,j = 0;
    	for(i=n-2;i>=0;i--)
    	{
    		int k2 = b[i]*a[n-2-i];
    		if(k1>k2)
    		{
    			f = 0;
    		}
    		else
    		{
    			f = 1;
    			break;
    		}
    	}
    	if(f==0)
    	{
    		printf("YES\n");
    	}
    	else
    	{
    		printf("NO\n");
    	}
    }
    return 0;
}
