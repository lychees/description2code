#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
#include <ctime>
using namespace std;

int main() {
	ios_base::sync_with_stdio(0);
	int tt;
	cin >> tt;
	while(tt--) {
		int n;
		cin >> n;
		int w1;
		cin >> w1;
		vector <int> w(n - 1);
		vector <int> h(n);
		for(int i = 0; i < n - 1; ++i) cin >> w[i];
		for(int i = 0; i < n; ++i) cin >> h[i];
		sort(w.begin(), w.end(), greater <int> ());
		sort(h.begin(), h.end());
		int brave = w1 * h.back();
		int ok = 1;
		for(int i = 0; i < n - 1; ++i) if(w[i] * h[i] >= brave) ok = 0;
		if(ok) cout << "YES\n";
		else cout << "NO\n";
	}
	return 0;
}
