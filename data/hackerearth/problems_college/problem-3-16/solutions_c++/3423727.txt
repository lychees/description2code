#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n,k;
		scanf("%d %d",&n,&k);
		int a[15];
		for(int i=0;i<=k;i++){
			a[i]=0;
		}
		long long sum=0,ma=0;
		for(int i=0;i<n;i++){
			int x;
			scanf("%d",&x);
			sum++;
			a[x]++;
			ma=max(ma,(long long)a[x]);
		}
		printf("%lld\n",sum-ma);
	}
	return 0;
}
