n = input()
list1 = []
for i in range(0, n):
    list1.append(raw_input())


cost = [[0 for x in xrange(n)] for x in xrange(n)]
for i in range(0, n):
    line = raw_input()
    line = line.split()
    for j in range(0, n):
        cost[i][j] = int(line[j])
t = input()
i = 0
sum1 = 0
while (t > 0):
    temp = raw_input()
    j = list1.index(temp)
    sum1 = sum1+cost[i][j]
    i = j

    t = t-1
print sum1
