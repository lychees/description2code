#include<bits/stdc++.h>
#define mx 9000005
#define ll long long
#define pb push_back
using namespace std;
ll dp[mx];
vector<int>a;
int main()
{
    int n,x;
    scanf("%d",&n);
    while(scanf("%d",&x)!=-1)
    {
        a.pb(x);
    }
    dp[0]=1;

        for(int j=0;j<a.size();j++)
        {
            for(int i=0;i<=n;i++)
            {
                if(i-a[j]>=0)dp[i]+=dp[i-a[j]];
            }
        }

    printf("%d\n",dp[n]);



}
