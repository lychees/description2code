# include <iostream>
# include <string>
using namespace std;

string n,m;

int main()
{
    int t;  cin>>t;
    while (t--)
    {
        cin>>n>>m;
        if (n == m || (n == "2" && m == "4") || (n == "4" && m == "2") )
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    
    
    
    return 0;
}
