#include <iostream>
#include<algorithm>
using namespace std;

int arr[1003];

int main()
{
//    cout << "Hello World!" << endl;
	int n,diff;
	cin>>n;
	for(int i=0;i<n;i++)cin>>arr[i];
	sort(arr,arr+n);
	diff=arr[0];
	cout<<n<<endl;
	for(int i=1;i<n;i++)
	{
		int temp = arr[i]-diff;
		if(temp > 0)
		{
			cout<<(n-1)-i+1<<endl;
			diff = diff + temp;
		}
	}
    return 0;
}
