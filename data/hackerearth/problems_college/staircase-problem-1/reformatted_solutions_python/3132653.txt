noOfTestCases = int(raw_input())


def minSteps(steps):

    ans, curStep, incre = 1, 2, 2
    # print incre
    while (curStep < steps):
        incre = incre + 1
        # print incre
        curStep = curStep + incre
        ans = ans + 1

    return ans


for j in range(1, noOfTestCases + 1):
    print minSteps(int(raw_input()))
