#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
	long long int count,n,q,k,t,i,j,score[100001],time1[100001]={-1};
	cin>>n;
	cin>>q;
	for(i=0;i<n;i++)
	{
		cin>>time1[i];
	}
	for(j=0;j<n;j++)
	{
		cin>>score[j];
	}
	sort(time1,time1+n);
	
	while(q--)
	{
		cin>>k;
		count=0;
	for(i=n-1;i>=n-k;i--)
	{
		count=count+time1[i];
	}
	cout<<count<<endl;
	}
	return 0;
}
