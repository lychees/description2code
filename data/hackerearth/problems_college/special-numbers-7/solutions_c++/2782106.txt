#include <bits/stdc++.h>
using namespace std;

int main()
{
	int a[100001]={0},b[1001],k;
	for(int i=2;i<=100000;i++)
	{
		if(a[i]==0)
		{
			for(int j=i;j<=100000;j+=i)
			a[j]++;
		}
	}
	k=0;
	
	/*for(int i=0;i<101;i++)
	cout<<a[i]<<" ";
	cout<<"\n";*/
	for(int i=0;i<=100000;i++)
	{
		if(a[i]>=3)
		b[k++]=i;
	}
/*	for(int i=0;i<101;i++)
	cout<<b[i]<<" ";*/
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<b[n-1]<<"\n";
	}
    return 0;
}
