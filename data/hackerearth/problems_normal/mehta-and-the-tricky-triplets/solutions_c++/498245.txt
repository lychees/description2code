#include<iostream>
#include<cstdio>
#include<cstring>
#define ll long long
using namespace std;
ll primes[10];
bool marked[10];
ll counts[10][10];
int main()
{
	ll t,i,l,tri4=0,j;
	ll tri235=0,tri237=0,tri357=0,tri257=0;
	char ch,str[20];
	scanf("%lld",&t);
	scanf("%c",&ch);
	for(i=0;i<10;i++)
	{
		primes[i]=0;
	}
		for(i=0;i<10;i++)
		{
			for(j=0;j<10;j++)
			{
				counts[i][j]=0;
			}
		}
	while(t--)
	{
		for(i=0;i<10;i++)
		{
			marked[i]=0;
		}
		
		scanf("%s",&str);
		l=strlen(str);
		for(i=0;i<l;i++)
		{
			if(!marked[str[i]-48])
			{
				marked[str[i]-48]=1;
				primes[str[i]-48]++;
			}
		}
		for(i=0;i<10;i++)
		{
			for(j=i+1;j<10;j++)
			{
				if(marked[i] && marked[j])
				{
					counts[i][j]+=1;
				}
			}
		}
		if(marked[2]&& marked[3] &&marked[5])
		{
			tri235++;
		}
		if(marked[2]&& marked[3] &&marked[7])
		{
		tri237++;
		}
		if(marked[2]&& marked[5] &&marked[7])
		{
		tri257++;
		}
		if(marked[3]&& marked[5] &&marked[7])
		{
		 tri357++;
		 }
		if(marked[2]&& marked[3]&& marked[5] &&marked[7])
		{
			tri4++;
		}
	}
	ll no=0,k;
	if(primes[2]>=3)
	{
		k=primes[2];
		no+=(k)*(k-1)*(k-2)/6;
	}
	if(primes[3]>=3)
	{
		k=primes[3];
		no+=(k)*(k-1)*(k-2)/6;
		k=counts[2][3];
		no-=(k)*(k-1)*(k-2)/6;
	}
	if(primes[5]>=3)
	{
		k=primes[5];
		no+=(k)*(k-1)*(k-2)/6;
		k=counts[2][5];
		no-=(k)*(k-1)*(k-2)/6;
		k=counts[3][5];
		no-=(k)*(k-1)*(k-2)/6;
		k=tri235;
		no+=(k)*(k-1)*(k-2)/6;
	}
	if(primes[7]>=3)
	{
		k=primes[7];
		no+=(k)*(k-1)*(k-2)/6;
		k=counts[2][7];
		no-=(k)*(k-1)*(k-2)/6;
		k=counts[3][7];
		no-=(k)*(k-1)*(k-2)/6;
		k=counts[5][7];
		no-=(k)*(k-1)*(k-2)/6;
		k=tri237;
		no+=(k)*(k-1)*(k-2)/6;
		k=tri257;
		no+=(k)*(k-1)*(k-2)/6;
		k=tri357;
		no+=(k)*(k-1)*(k-2)/6;
	}
	
	k=tri4;

	no-=(k)*(k-1)*(k-2)/6;
	printf("%lld\n",no);
	return 0;
}
