#include <cstdio>
#include <iostream>
#include <string>
using namespace std;
int main() {
  freopen("input.txt", "r", stdin);
  freopen("output.txt", "w", stdout);

  int n, k, a[20000];
  cin >> n >> k;
  for (int i = 1; i <= n; i++) {
    cin >> a[i];
  }

  while (a[k] == 0) {
    k++;
    if (k > n) {
      k = 1;
    }
  }
  cout << k;

  return 0;
}
