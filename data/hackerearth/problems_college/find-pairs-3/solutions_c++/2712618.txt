#include <iostream>
#include<map>
using namespace std;
map <int,int> mp;
int main()
{
    int t; cin >> t;
    while(t--)
    {
    	int n; cin >> n;
    	int a[n];
    	for(int i=0;i<n;i++) cin >> a[i],mp[a[i]]++;
    	long long ans = n;
    	for(int i=0;i<n;i++)
    	{
    		long long y = mp[a[i]];
    		if(y)
    			ans += y*(y-1)/2,mp[a[i]]=0;
    	}
    	cout << ans << endl;
    }
    return 0;
}
