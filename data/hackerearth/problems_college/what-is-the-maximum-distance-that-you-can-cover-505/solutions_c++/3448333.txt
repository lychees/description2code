#include <bits/stdc++.h>

using namespace std;

double speed [1000000 + 10];
double fuel  [1000000 + 10];

double truncate_to_places(double d, int n) {

    d = (int)(d * pow(10 , n) ) ;
    return d /= pow(10 , n);
}


int main()
{
    int n;
    double given;
    double res = -1;

    cin >> n;

    for(int i = 0; i < n; i++) cin >> speed[i];
    for(int i = 0; i < n; i++) cin >> fuel[i];

    cin >> given;

    for(int i = 0; i < n; i++){

        res = max(res , given * speed[i] / fuel[i]);
    }


    printf("%.3lf", truncate_to_places(res , 3));

    return 0;
}
