#include <iostream>
#include<string.h>
using namespace std;
char getcolor( int p,int q ,int r )
{
	if(p==0 && q==0 && r==0)
	return 'B';

	if(p==0 && q==0 && r==1)
	return 'b';

		if(p==1 && q==0 && r==0)
	return 'r';
		if(p==0 && q==1 && r==0)
	return 'g';
		if(p==1 && q==1 && r==0)
	return 'y';
		if(p==0 && q==1 && r==1)
	return 'c';
		if(p==1 && q==0 && r==1)
	return 'm';
		if(p==1 && q==1 && r==1)
	return 'w';
}
int main()
{
	long n,r,g,blue;
	cin>>n;
	int a[n],b[n],c[n];
	int color[8]={0};
	bool flag1=0,flag2=0,flag3=0;
	int f1=0,f2=0,f3=0;

	cin>>r>>g>>blue;

	for( int i=0;i<n;i++ )
	{
		// red
		if(f1<r)
		{
			a[i]=flag1;
			f1++;
		}
		if(f1==r)
		{
			f1=0;
			flag1=flag1^1;
		}

		//green
		if(f2<g)
		{
			b[i]=flag2;
			f2++;
		}
		if(f2==g)
		{
			f2=0;
			flag2=flag2^1;
		}

		// blue


		if(f3<blue)
		{
			c[i]=flag3;
			f3++;
		}
		if(f3==blue)
		{
			f3=0;
			flag3=flag3^1;
		}
		char T=getcolor(a[i],b[i],c[i]);

		switch(T)
		{
			case 'r': color[0]++;break;
			case 'g':color[1]++;break;
			case 'b':color[2]++;break;//blue
			case 'y':color[3]++;break;
			case 'c':color[4]++;break;
			case 'm':color[5]++;break;
			case 'w':color[6]++;break;
			case 'B':color[7]++;break;// Black

		}


	}



	for( int i=0;i<8;i++ )
	cout<<color[i]<<" ";
	return 0;
}
