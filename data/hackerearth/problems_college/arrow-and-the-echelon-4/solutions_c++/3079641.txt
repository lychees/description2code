#include<bits/stdc++.h>
using namespace std;

int main()
{
	int a[16]={0,-9,-3,3,-6,0,6,-3,3,9,0,6,12,3,9};
	int b[16]={0,10,5,0,10,5,0,10,5,0,10,5,0,10,5};
	int i,j,t,c5,c3;
	long long int x;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld",&x);
		c3=c5=0;
		j= x%15;
		c3= x-j;
		c3+=a[j];
		c5=b[j];
		if(c3>=0)
		{
			while(c3--)
		printf("5");
		while(c5--)
		printf("3");
		printf("\n");
		}
		else
		printf("-1\n");
	}
}
