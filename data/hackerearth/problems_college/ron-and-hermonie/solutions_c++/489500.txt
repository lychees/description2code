#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<algorithm>
#include<limits.h>
#define gc getchar

long long int read() {
 long long int ret=0;
 char c=gc();
 while(c<'0'||c>'9') c=gc();
 while(c>='0'&&c<='9') {
   ret=10*ret+c-48;
   c=gc();
 }
 return ret;
}

using namespace std;

int main(){
   long long int no_of_test_cases = read();

   int arr[10][4] = {
                     {0,0,0,0},
                     {1,1,1,1},
                     {6,2,4,8},
                     {1,3,9,7},
                     {6,4,6,4},
                     {5,5,5,5},
                     {6,6,6,6},
                     {1,7,9,3},
                     {6,8,4,2},
                     {1,9,1,9}
                    };

   while(no_of_test_cases--){
       long long int base = read();
       long long int exponent = read();
       int res;

       int lastDigit = base%10;

       res = arr[lastDigit][exponent%4 ];

       printf("%d\n",res);
   }
}
