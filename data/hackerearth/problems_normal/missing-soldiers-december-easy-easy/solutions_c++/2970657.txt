#include <bits/stdc++.h>


using namespace std;





int main (int argc, char**argv){
  //  freopen("/Users/Arseniy/All/input.txt", "r", stdin);
   // freopen("/Users/Arseniy/All/output.txt", "w", stdout);
    int n;
    cin >> n;
    vector <pair <int, int> > segments;
    for (int i = 0; i < n; i++){
        int x, y, d;
        cin >> x >> y >> d;
        segments.push_back(make_pair(x, 0));
        segments.push_back(make_pair(x+d, 1));
    }
    sort(segments.begin(), segments.end());
    int ans = 0;
    int c = 0;
    int cur = segments[0].first;
    for (int i = 0; i < 2*n; i++){
        if (segments[i].second == 0){
            if ((c == 0) && (cur < segments[i].first))
                cur = segments[i].first;
            c++;
        }
        else{
            ans += segments[i].first - cur + 1;
            cur = segments[i].first + 1;
            c--;
        }
    }
    cout  << ans << endl;
    return 0;
}
