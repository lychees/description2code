#include<bits/stdc++.h>
using namespace std;

long long int p[1000009];

void init() {
   long long int i,j,k;
   for(i=1;i<1000009;i++) {
   	  p[i]=i*i;
   }
}


int main() {
    long long int i,j,k,t,x,y;
    cin >> t;
    init();
    while(t--) {
    	cin >> k;
        i=sqrt(k);
        if(p[i]==k) {
        	cout << "YES" << endl;
        }
        else {
        	x=(int)sqrt(k);
        	y=x+1;
        	x=min(k-p[x],p[y]-k);
            if((p[y]-k)==x) {
            	cout << "+" << x << endl;
            }
            else {
            	cout << "-"  << x << endl;
            }
        }
    }

	return 0;
}
