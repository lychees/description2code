#include<stdio.h>
#include<iostream>
using namespace std;
long long int point[1005];
int main()
{
    point[0]=0;
    point[1]=1;
    point[2]=2;
    int test,i,j,n;
    for(i=3;i<=1000;i++)
    {
        point[i]=point[i-1]+point[i-2];
    }
    scanf("%d",&test);
    while(test--)
    {
        scanf("%d",&n);
        cout<<point[n]<<endl;
    }
    return 0;
}
