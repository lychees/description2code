#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int main()
{
	int idea;
	int num;
	vector<int> arr;
	cin>>idea>>num;
	for(int i=1;i<=idea;i++)
	{
		arr.push_back(i);
	}

	int pos=0;
	while(arr.size()>1)
	{
		int sz=arr.size();
		pos=(pos+num-1)%sz;
		//cout<<arr[pos]<<endl;
		arr.erase(arr.begin()+pos);
	}
	cout<<arr[0]<<endl;
}
