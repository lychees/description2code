def comp(a,number):
    for i in a:
        if i>=number:
            return a.index(i)+1
    return -1

result=[]
NQ=map(int,raw_input("").split())
n,q=NQ[0],NQ[1]
a= map(int,raw_input("").split())
for i in xrange(q):
    xy=map(int,raw_input("").split())
    if len(xy)==3:
        x,y=xy[1],xy[2]
        a[x-1] = y
    else:
        number = xy[1]
        result.append(comp(a,number))
for c in result:
    print c
