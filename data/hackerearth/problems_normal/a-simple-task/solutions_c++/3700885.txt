#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#define ll long 
using namespace std;
ll gcd(ll a,ll b){
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}
 
int main() {
    ll arr[100],diff[100],m,i,x=0,fact[5000],j;
    cin>>m;
    for(i=0;i<m;i++){
        cin>>arr[i];
    }
    for(i=0;i<m;i++){
        diff[i]=abs(arr[(i+1)%m]-arr[i]);
        x=gcd(x,diff[i]);
    }
    i=1;
    j=0;
    while(i*i<=x){
        if(x%i==0){
            fact[j]=i;
            j++;
            if(i*i!=x){
                fact[j]=x/i;
                j++;
            }
        }
        i++;
    }
    sort(fact,fact+j);
    for(i=0;i<j;i++){
        if(fact[i]>1)
            cout<<fact[i]<<" ";
    }
    return 0;
}
