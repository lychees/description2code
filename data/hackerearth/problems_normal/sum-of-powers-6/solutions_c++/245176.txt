#include<cstdio>
#include<vector>
#include<cstring>
#include<iostream>
using namespace std;
typedef long long int i64;
vector<int> p;
#define high 1000000
bool value[high+10];
void preprocess(){
    p.push_back(1);
    for(int k=2;k<=1000;k++){
        i64 m = k*k;
        while(m<=high){
            p.push_back(m);
            m*=k;
        }
    }
    memset(value,0,sizeof(value));
    int sz = p.size();
    for(int i=0;i<sz;i++){
        for(int j=0;j<sz;j++){
                int temp = p[i]+p[j];
            if(temp<=high)value[temp]=1;

        }
    }

//    cout<<p.size()<<endl;
}
int main(){
    int t,x;
    scanf("%d",&t);
 //   t=1;
    preprocess();
    while(t--){
        scanf("%d",&x);
        if(value[x])printf("Yes\n");
        else printf("No\n");
    }
return 0;
}
