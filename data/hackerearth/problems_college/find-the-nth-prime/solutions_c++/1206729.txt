#include <iostream>
using namespace std;
inline int prime(int N)
{
	int f=0;
	int i=2;
	while(i<=(N/2))
	{
		if(N%i==0)
		{
			f=1;
			break;
		}
		i++;
	}
	return f;
}
int main()
{
    int n,i=2,p=0;
    cin>>n;
    while(p!=n)
    {
    	if(prime(i)==0)	
    	{
    		p++;
    		i++;
    	}
    	else
    	i++;
    }
    cout<<(i-1);
    return 0;
}
