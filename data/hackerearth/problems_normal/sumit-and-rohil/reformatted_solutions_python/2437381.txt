def main():
    n = raw_input()
    li = []
    for i in xrange(0, int(n)):
        stri = raw_input()
        li.append(stri)
    li2 = []
    for i in li:
        out = (sorted(i[1:-1]))
        stri = ''.join(i for i in out)
        stri = i[0]+stri+i[-1]
        li2.append(stri)
    print len(list(set(li2)))


if __name__ == '__main__':
    main()
