#include<bits/stdc++.h>
using namespace std;
long long int arr[2002];
long long int dp[2002][2002];
long long int find(int i,int sum) {
	if(i>2000||sum<=0) {
		return 0;
	}
	if(dp[i][sum]!=-1) {
		return dp[i][sum];
	}
	long long int j,tot=0;
	if(arr[i]>0) {
	
	for(j=0;;j++) {
		if((sum-j*i)>=0) {
		
		tot=max(tot,j*i+find(i+1,sum-j*i));
	}
	else {
		break;
	}
	}
}
else {
	tot = find(i+1,sum);
}
	return dp[i][sum]=tot;
}
int main()
{
	long long int t,n,k,i,a;
	cin>>t;
	while(t--) {
		memset(dp,-1,sizeof(dp));
		memset(arr,0,sizeof(arr));
		cin>>n>>k;
		for(i=0;i<n;i++) {
			scanf("%lld",&a);
			arr[a]++;
		}
		cout<<find(1,k)<<endl;
	}
	return 0;
}
