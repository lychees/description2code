#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int n;
   	cin>>n;
   	double nu=0,de=0,temp;
   	for(int i=0;i<n;i++)
   	{
   		cin>>temp;
   		nu+=temp;
   		de+=100;
   	}
   	cout<<fixed;
   	cout<<setprecision(8)<<(nu/de)*100<<endl;
   }
    return 0;
}
