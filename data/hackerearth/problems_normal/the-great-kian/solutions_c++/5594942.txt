#include<iostream>
using namespace std;
int main()
{
    int n; long long unsigned int sum1=0, sum2=0, sum3=0 ;
    cin>>n; long long unsigned int a[n];
    for(int i=0; i<n; i++) {cin>>a[i];}
    for(int j=1; j<n; j++)
    {if(3*j-3>=n) {break;}sum1+=a[3*j-3];
    if(3*j-2>=n) {break;}
    
    sum2+=a[3*j-2]; 
    if(3*j-1>=n) {break;}
    sum3+=a[3*j-1]; 
        
    }cout<<sum1<<" "<<sum2<<" "<<sum3<<" ";
}
