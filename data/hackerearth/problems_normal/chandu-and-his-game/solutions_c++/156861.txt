int main(){
    int n,k,a,b;
    cin>>n>>k;
    ll ans=0;
    while(k--){
        scanf("%d%d",&a,&b);
        int x=abs(a-1)+abs(b-1);
        int y=abs(a-1)+abs(b-n);
        int z=abs(a-n)+abs(b-1);
        int w=abs(a-n)+abs(b-n);
        ans+=min(x,min(y,min(z,w)));
    }
    cout<<ans<<endl;
    return 0;
}
