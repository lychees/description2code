#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int size,amount,total,counter;
	cin>>size;
	cin>>amount;
    unsigned long int a[size];
    for(int i=0;i<size;i++){
    	cin>>a[i];
    }
    sort(a,a+size);
    total=0;
    for(int i=0;i<size;i++){
    	total=total+a[i];
    	if(total<=amount){
    		counter++;
    	}
    }
   // cout<<"size"<<size;
    cout<<counter;
    return 0;
}
