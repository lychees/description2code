#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	cin >> n;
	int a[n];
	for (int i = 0; i < n; i++)
		cin >> a[i];
	
	int sum = 0, max = 0;
	
	for (int i = 0; i < n; i++)
	{
		max = max + a[i];
		if (max > sum)
			sum = max;
		if (max < 0)
			max = 0;
	}
	cout << sum << endl;
	return 0;
}
