#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
		scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int k;
			scanf("%d",&k);
		int c=0;
		if(n%k==0)
		{
			int l=INT_MIN;
		for(int i=k;i<=n;i=i+k)
		{
			if(i%l ==0 && l!= INT_MIN)
			continue;
		
			if(__gcd(i,n)==k)
			{
				c++;
			}
			else
			if(l==INT_MIN)
			l =i;	
			
		}
	}
		printf("%d\n",c);
	}
}
