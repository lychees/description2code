#include<bits/stdc++.h>
using namespace std;
long long int func(long long int value,char c,long long int num)
{
	if(c=='N')return -value;
	else if(c=='+')return value+num;
	else if(c=='-')return value-num;
	else if(c=='*')return value*num;
	else if(c=='/')return value/num;
}
 int main()
{
	long long int test,i,j;
	cin>>test;

	for(i=0;i<test;i++)
	{
		long long int n;
		cin>>n;
		long long int small=1;
		long long int large=1,num;
		vector <long long int >vect;
		for(j=0;j<n;j++)
		{
			char c;
			cin>>c;
			if(c!='N')
			{
				cin>>num;
			}
			vect.push_back(small);
			vect.push_back(large);
			vect.push_back(func(small,c,num));
			vect.push_back(func(large,c,num));

			sort(vect.begin(),vect.end());

			small=vect[0];
			large=vect[vect.size()-1];
		}
		cout<<vect[vect.size()-1]<<endl;
	}
	return 0;
}
