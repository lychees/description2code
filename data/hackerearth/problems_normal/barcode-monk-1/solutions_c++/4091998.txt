	#include<bits/stdc++.h>
	using namespace std;
	 
	#define fr(i,n) for(int i=0;i<n;i++)
	#define fr1(i,n) for(int i=1;i<=n;i++)
	#define ll long long
	#define pii pair<int,int>
	#define ff first
	#define ss second
	#define mp make_pair
	#define pb push_back
	#define MOD 1000000007
	#define MAX 10000002
	#define INF 100005
	int n,k;
	int sum[1002],p[1002],up[1002],arr[1002];
	vector<int> v;
	int main(){
		cin>>n>>k;
		fr(i,k){
			cin>>arr[i];

		}
		sum[0]=1;
		fr1(i,n){
			fr(j,n){
				if(i-arr[j]>=0){
					sum[i]|=sum[i-arr[j]];
				}
			}
			if(sum[i])v.pb(i);
		}
		up[0]=1;
		fr1(i,n){
			fr(j,v.size()){
				p[i]=(((i-v[j]>=0)?up[i-v[j]]:0)+p[i])%MOD;
			}
			up[i]=(up[i-1]+p[i-1])%MOD;
		}
		cout<<(p[n]+up[n])%MOD<<endl;
		
		return 0;
	}
