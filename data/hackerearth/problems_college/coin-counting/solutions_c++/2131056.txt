#include<bits/stdc++.h>
using namespace std;
int S[3]={1,3,5};
int table[10000];
int count(int n )
{
    memset(table, 0, sizeof(table));
    table[0] = 1;
    for(int i=0; i<3; i++)
        for(int j=S[i]; j<=n; j++)
            table[j] += table[j-S[i]];
    return table[n];
}
int main()
{
    int num,n,i,sum,test;
    scanf("%d",&test);
    while(test--)
    {
        scanf("%d",&sum);
        printf("%d\n",count(sum));
    }
}
