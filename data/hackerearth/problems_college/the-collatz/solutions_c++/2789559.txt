#include <bits/stdc++.h>
using namespace std;
 
int main()
{
	long t,n;
	cin>>t;
	while(t--){
		cin>>n;
		long steps=0,even=0,odd=1;
		while(n!=1){
			if(n%2==0){
				even+=n;
				n/=2;
			}else{
				odd+=n;
				n=3*n+1;
			}
			steps++;
		}
		
		if(even%steps< odd%steps)
			cout<<"Odd Rules\n";
		else if(even%steps > odd%steps)
			cout<<"Even Rules\n";
		else
			cout<<"Tie\n";
	}
	return 0;
}
