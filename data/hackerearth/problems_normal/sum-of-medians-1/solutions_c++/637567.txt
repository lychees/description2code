#include<iostream>
#include<cstdio>
#include<set>
using namespace std;

int main()
{
int n,i,j,k,s=0,x,med;
std::multiset<int> m;
std::multiset<int>::iterator it;

scanf("%d",&n);

scanf("%d",&x);
m.insert(x);
it=m.begin();
s+=*it;
med=*it;

for(i=0;i<n-1;i++)
{
scanf("%d",&x);
m.insert(x);
if(i%2==0 && x<med)
it--;
else
if(i%2!=0 && x>=med)
it++;

med=*it;
s=(s+*it)%100000;
}

printf("%d\n",s);
return 0;
}
