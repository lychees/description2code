import sys
T = int(sys.stdin.readline())
for i in xrange(T):
	personalities = {}
	N = int(sys.stdin.readline())
	line = sys.stdin.readline().split()
	total = 0
	for j in xrange(N):
		p = line[j]
		if p not in personalities:
			total = total+1
			personalities[p] = True
	print total
