#include <iostream>
using namespace std;

int prime[1000005];

int main()
{
	prime[1] = 0;
	for(int i = 2; i <= 1000000; i++) prime[i] = 1;
	for(int p = 2; p <= 1000; p++){
		if(prime[p] == 0) continue;
		for(int q = 2*p; q <= 1000000; q += p) prime[q] = 0;
	}
	int num_of_primePrimes[1000005] = {0};
	int prime_count = 0, primePrime_count = 0;
	for(int i = 1; i <= 1000000; i++){
		if(prime[i] == 1){
			prime_count++;
		}
		//Found a primePrime
		if(prime[prime_count]){
			primePrime_count++;
		}
		num_of_primePrimes[i] = primePrime_count;
	}
	
	int t = 0, l, r;
	cin>>t;
	while(t--){
		cin>>l>>r;
		cout<<num_of_primePrimes[r] - num_of_primePrimes[l-1]<<endl;
	}
	
    return 0;
}
