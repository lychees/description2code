#include <iostream>
using namespace std;

int main()
{
    string s;
    cin >> s;
    
    int len=s.length();
    int i=0;
    if(s[i]>='a'&&s[i]<='z')
    {s[i]=s[i]-'a'+'A';i++;}
    while(i<len)
    {
    	if(s[i]=='_')
    	{while(s[i]=='_')
    	i++;
    	s[i]=s[i]-'a'+'A';
    	}
    	i++;
    	
    }
    cout << s << endl;
    return 0;
}
