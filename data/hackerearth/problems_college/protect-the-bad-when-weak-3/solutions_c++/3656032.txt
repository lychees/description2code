#include <iostream>
#include<stdio.h>
using namespace std;
int n;
int a[102][102];
int vis[102][102];
int count=0;
long long find(int i,int j)
{
	if(i==n-1&&j==n-1)
	{
		if(a[i][j]==0)
			count++;
	}
vis[i][j]=1;
	long long ans=0;
	if(i-1>=0&&a[i-1][j]==0&&vis[i-1][j]!=1)
	find(i-1,j);
	if(j-1>=0&&a[i][j-1]==0&&vis[i][j-1]!=1)
	find(i,j-1);
	if(i+1<n&&a[i+1][j]==0&&vis[i+1][j]!=1)
	find(i+1,j);
	if(j+1<n&&a[i][j+1]==0&&vis[i][j+1]!=1)
	find(i,j+1);
	vis[i][j]=2;
	return ans;
	
}
int main()
{
	int i,j;
   scanf("%d",&n);
   for(i=0;i<n;i++)
   	for(j=0;j<n;j++)
   		scanf("%d",&a[i][j]);
   find(0,0);
   printf("%d\n",count);
}
