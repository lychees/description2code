#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

int main()
{
    char a[1000];
    int i, k = 0, length;
    scanf("%s", &a);
    
    length = strlen(a);
    for(i = 0; i < length; i++)
    {
    	switch(a[i]){
    		case '0': k += 6; break;
    		case '1': k += 2; break;
    		case '2': k += 5; break;
    		case '3': k += 5; break;
    		case '4': k += 4; break;
    		case '5': k += 5; break;
    		case '6': k += 6; break;
    		case '7': k += 3; break;
    		case '8': k += 7; break;
    		case '9': k += 6; break;
    	}
    }
    printf("%d\n", k);
    return 0;
}
