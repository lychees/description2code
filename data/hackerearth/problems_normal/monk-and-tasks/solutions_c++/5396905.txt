#include <bits/stdc++.h>
using namespace std;
bool comparator(pair<long long int,long long int > p1,pair<long long int,long long int >p2 ){
	return (p1.second<p2.second);
}
int main()
{
    int t;
    cin>>t;
    long int n;
    long long int m,i,y;
    while(t--){
    	
    	scanf("%ld",&n);
    	pair<long long int ,long long int>a[n];
    	for(i=0;i<n;i++){
    		y=0;
    		scanf("%lld",&a[i].first);
    		m=a[i].first;
    		while(m){
    			m=m&(m-1);
    			y++;
    		}
    		a[i].second=y;
    	}
    	stable_sort(a,a+n,comparator);
    	for(i=0;i<n;i++)
    		printf("%lld ",a[i].first);
    	cout<<endl;
    }
    return 0;
}
