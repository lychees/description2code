#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int arr[n];
    	for(int i=0;i<n;i++)
    	cin>>arr[i];
    	int arr1[n];
    	arr1[n-1]=arr[n-1];
    	for(int i=n-2;i>=0;i--)
    	arr1[i]=max(arr1[i+1],arr[i]);
    	int count=0;
    	for(int i=0;i<n;i++)
    	{
    		if(arr1[i]>arr[i])
    		count+=(arr1[i]-arr[i]);
    	}
    	
    	 cout<<count<<endl;
    }
   
    return 0;
}
