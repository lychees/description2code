#include<iostream>

using namespace std;

int main()
{
	bool names[26];
	bool flag;
	int t, n;
	char city[12];
	cin >> t;
	while (t--)
	{
		cin >> n;
		flag = true;
		for (int i = 0; i < 26; i++)
			names[i] = false;
		while (n--)
		{
			cin>>city;
			if (flag)
			{
				if (names[city[0] - 97])
					flag = false;
				else
					names[city[0] - 97] = true;
			}
		}

		if (flag)
			cout << "YES\n";
		else
			cout << "NO\n";
	}
	return 0;
}
