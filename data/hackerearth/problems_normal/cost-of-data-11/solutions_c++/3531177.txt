#include <bits/stdc++.h>

using namespace std;

int ans = 0;

struct trie {

	trie* child[26 + 2];
	bool isLeaf;

	trie() {
		memset(child, 0, sizeof(child));
		isLeaf = 0;
	}

	void insert(const string &str , int index) {
		if(str.length() == index)
			isLeaf = 1;
		else {

			int cur = str[index] - 'a';

			if(child[cur] == 0 ){
				child[cur] = new trie();
                ans++;
			}

			child[cur]->insert(str , index + 1);
		}
	}
};


int main()
{

    int n;
    string s;
    trie tree;

    cin >> n;

    for(int i = 0; i < n; i++){

        cin >> s;
        tree.insert(s , 0);
    }

    cout << ans + 1 << "\n";

    return 0;
}
