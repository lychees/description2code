#include <bits/stdc++.h>
using namespace std;
int a[400];
int dp[400][500];
int fun(int index, int m,int n)
{
	if(index==n&&m==0)
	return 0;
	if(m==0&&index!=n)
	return INT_MAX;
	if(index==n)
	return 0;
	if(dp[index][m]!=-1)
	return dp[index][m];
	int res=INT_MAX;;
	for(int i=0;i+index<n;i++)
	{
		int sum=0;
		for(int j=0;j<=i;j++)
		sum=sum+a[index+j];
		res=min(res,max(sum,fun(index+i+1,m-1,n)));
	}
	dp[index][m]=res;
	return dp[index][m];
	
}
int main()
{
	int m,n;
	scanf("%d%d",&m,&n);
	for(int i=0;i<n;i++)
	scanf("%d",&a[i]);
	memset(dp,-1,sizeof dp);
	int ans=fun(0,m,n);
	cout<<ans;
	return 0;
}
