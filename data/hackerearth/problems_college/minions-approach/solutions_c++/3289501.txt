    for(int i = 0; i < n; i++) {
        cout << A[i] << " " ;

    cout << endl;
}

bool prime[100020];

void se(int n)
{
    memset(prime, true, sizeof(prime));

    for (int p=2; p*p<=n; p++)
    {
        if (prime[p] == true)
        {
            for (int i=p*2; i<=n; i += p)
                prime[i] = false;
        }
    }

}

int main()
{
    int t, i, j, n, cnt;
    cin >> t;
    se(100010);
    prime[1] = false;
    while(t--) {
        cnt = 0;
        cin >> n;
        for(i = 1; i <= n; i++) {
            if(__gcd(i, n) == 1) {
                cnt++;
            }
        }
        //cout << n << " " << cnt << endl;
        if(prime[cnt]) {
            cout << "Correct\n";
        }
        else {
            cout << "Incorrect\n";
        }
    }


    return 0;
}
