#include <bits/stdc++.h>
using namespace std;


string a,b;


void change(int n)
{
	string c;
	c=a;
	for(int i=0;i<n/2;i++)
	{
		c[2*i]=a[i];
		c[2*i+1]=a[n-i-1];
	}
	if(n%2==1)
	{
		c[n-1]=a[n/2];
	}
	a=c;
}


int main()
{
    int k; cin>>k;
    cin>>a;
    int n=a.length();
    int p=1;
    b=a;
    change(n);
    while(b!=a)
    {
    	p++;
    	change(n);
    }
    k=p-k%p;
    for(int i=0;i<k;i++)
    {
    	change(n);
    }
    cout<<a<<endl;
}
