#include <iostream>
using namespace std;

int main()
{
    long long int n,i,ctr=0,main_count;
    cin>>n;
    char str[100001];
    while(n--)
    {
    	cin>>str;
    	ctr=0;
    	main_count=0;
    	for(i=0;str[i]!='\0';++i)
    	{
    		if(str[i]=='A' || str[i]=='a' || str[i]=='B' || str[i]=='b' || str[i]=='c' || str[i]=='C')
    		{
    			++ctr;
    		}
    		else 
    		{
    			main_count+=(ctr*(ctr+1))/2;
    			ctr=0;
    		}
    	}
    
    		main_count+=(ctr*(ctr+1))/2;
    	
    	cout<<main_count<<"\n";
    }
    return 0;
}
