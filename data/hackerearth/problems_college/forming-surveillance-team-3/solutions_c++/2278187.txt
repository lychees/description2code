#include <iostream>
#include<map>
   using namespace std;
    int main()
    {
	int t;
	cin>>t;
	
	while(t--)
	{
	map<int,int>m; 
	int n;
	cin>>n;
	int arr[n];
	int count=n;
	for(int i=0;i<n;i++)
	{
	cin>>arr[i];
	   if(m[arr[i]]!=0)
	   {
	   	
	   	count--;
	   }
	   m[arr[i]]=1;
	
		}
		cout<<count<<endl;
	}
	  return 0;
   }
