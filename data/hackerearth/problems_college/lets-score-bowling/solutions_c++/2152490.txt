#include <iostream>
using namespace std;
int call(int f)
{
	if(f>0&&f<7)
	return 1;
	if(f>6&&f<9)
	return 2;
	if(f==0)
	return 0;
	if(f==9)
	return 10;
}



int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	int a,b,c;
	cin>>a>>b>>c;
	int total=0;
	
	total+=call(a);
	if(a==9)
	{
		cout<<total<<endl;
		continue;
	}
	total+=call(b);
	if(b==9)
	{total=total-3;
	cout<<total<<endl;
	continue;
	}
	total+=call(c);
	if(c==9)
	{
		total=total-6;
		cout<<total<<endl;
		continue;
	}
	
	if(a==9||b==9||c==9)
	cout<<total<<endl;
	else
	cout<<"0"<<endl;
	}
	
	
   // cout << "Hello World!" << endl;
    return 0;
}
