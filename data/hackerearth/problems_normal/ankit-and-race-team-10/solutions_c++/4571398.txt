lli fact[1000001],infact[1000001];
void prefact(lli N){
	int i;
	lli MOD = 1000000007ll;
	fact[0] = 1;
	for (i = 1; i < N; i++){
		fact[i] = (i * fact[i - 1]) % MOD;
	}
	infact[N - 1] = powermod(fact[N - 1], MOD - 2,MOD);
	for (i = N - 2; i >= 0; i--){
		infact[i] = ((i + 1) * infact[i + 1]) % MOD;
	}
}
lli f(lli n,lli r){
	lli ret = fact[n];
	ret *= infact[r];
	ret %= mod;
	ret *= infact[n - r];
	ret %= mod;
	return ret;
}
