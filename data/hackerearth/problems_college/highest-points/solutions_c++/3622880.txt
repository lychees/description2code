#include <iostream>
using namespace std;

int main()
{
    int n,i,j,t=0;
    cin>>n;
    int a[n][3],b[n];
    for(i=0;i<n;i++)
    {
      for(j=0;j<3;j++)
      {
      	cin>>a[i][j];
      }
    }
    for(i=0;i<n;i++)
    {
    	b[i]=(a[i][0]*3)+(a[i][1]*1)+(a[i][2]*-1);
    	if(t<b[i]&&n>1)
    	  t=b[i];
    	else if(n==1)
    	  t=b[i];
    }
    cout<<t;
    return 0;
}
