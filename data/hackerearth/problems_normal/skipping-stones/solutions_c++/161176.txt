#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{

   // cout<<"hi";

        result[i]=0;

    probability[0]=0;
    probability[n+1]=1;
    //cout<<"hi";
     for(int i=1;i<=n;i++)
       cin>>probability[i];
    int distance[n+2];
    distance[0]=0;
     distance[n+1]=D;
    for(int i=1;i<=n+1;i++)
        cin>>distance[i];
    for(int i=1;i<=n+1;i++)
        for(int j=0;j<i;j++)
        {
            if(distance[i]-distance[j]<=l )
            {
                if(result[i]<(result[j]*probability[i]))
                     result[i]=result[j]*probability[i];
            }


        }
