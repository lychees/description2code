#include<bits/stdc++.h>
using namespace std;

int main()
{
    //freopen("in.txt", "r", stdin);
    string str[2000], word;
    int indx=0, i, j;

    while(cin >> word){
        for(i=0; i<word.size(); i++)
            if(word[i]>='a' and word[i]<='z')
                word[i] -= 32;

        str[indx++] = word;
    }

    for(i=0; i<indx; i++){
        int cnt=1;

        for(j=0; j<indx; j++){
            if(j!=i and str[j]!="" and str[i]!="" and str[i]==str[j]){
                cnt++;
                str[j] = "";
            }
        }

        if(str[i]!="")
            cout << str[i] << " " << cnt << '\n';
    }

    return 0;
}
