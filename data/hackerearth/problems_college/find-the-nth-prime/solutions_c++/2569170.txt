#include <iostream>
using namespace std;

int main()
{
   int a[1000],n,i,j,flag=0,count=0;
   cin>>n;
   for(i=2;i<=7919;i++)
   {
   	for(j=2;j<=i/2;++j)
  {
  	flag=0;
      if(i%j==0)
      {
          flag=1;
          break;
      }
  }
  if (flag==0){a[count]=i;
  				count++;}				

   }
   cout<<a[n-1];
}
