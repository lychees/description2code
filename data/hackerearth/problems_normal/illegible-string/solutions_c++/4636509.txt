#include <iostream>
using namespace std;
#define ll long long
#include <vector>
#include <algorithm>

int main()
{	ll n,k=0,i,m=0;
	vector<ll> z;
	string s;
	cin>>n>>s;
	for(i=0;i<s.length();i++)
	{	if(s[i]=='v')
			k++;
		else if(s[i]=='w')
			k+=2;
		else
		{	if(k)
				z.push_back(k);
			k=0;
			m++;
		}
	}
	if(k)
		z.push_back(k);
	ll p=0,q=0;
	for(i=0;i<z.size();i++)
	{	p+=z[i];
		q+=(z[i]+1)/2;
	}
	cout<<m+q<<" "<<m+p<<"\n";
	return 0;
}
