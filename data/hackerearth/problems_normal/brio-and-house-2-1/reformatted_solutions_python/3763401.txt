import math
t = int(raw_input())
while t > 0:
    ans = ""
    str = raw_input().split(' ')
    x1 = int(str[0])
    y1 = int(str[1])
    x2 = int(str[2])
    y2 = int(str[3])
    r = int(raw_input())
    a = y2-y1
    b = x1-x2
    c = a*x1+b*y1
    if c < 0:
        c *= -1
    dis = c/(math.sqrt(a*a+b*b))
    if dis > r:
        ans = "SAFE"
    elif dis < r:
        ans = "REPLANNING"
    else:
        ans = "JUST MISSED"
    print ans
    t -= 1
