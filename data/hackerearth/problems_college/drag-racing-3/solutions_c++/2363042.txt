#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main()
{

    string winner[100000];
    string loser[100000];

    int n1,n=1;
    cin>>n1;

    for(int i=0;i<n1;i++)
    n*=2;
    n--;

    //cout<<n;
    for(int i=0;i<n;i++){
        cin>>winner[i];
        cin>>loser[i];
    }

    sort(winner,winner+n);

    int count=1;
    if(n>1)
    for(int i= 0;i<n-1;i++){
        if(winner[i]==winner[i+1]){
            count++;
        if(count>=n1)
            cout<<winner[i]<<endl;
        }
            else
            {
            count=1;
            }
    }
    else
    cout<<winner[0]<<endl;

    //cout << "Hello world!" << endl;
    return 0;
}
