#include<iostream>
using namespace std;

long callcm(long a,long b){
	long i=2,lcm;
	if(a==b)
		return a;
	if(a<b){
		lcm=b;
		while(lcm%a != 0){
			lcm=b*i;
			i++;
		}
		return lcm;
	}
	else{
		lcm=a;
		while(lcm%b != 0){
			lcm=a*i;
			i++;
		}
		return lcm;
	} 
}
int main(){
	long t,a,b,c,d,x,y,count,lcm;
	cin>>t;
	while(t--){
		cin>>a>>b>>c>>d;
		count=0;
		lcm=callcm(a,b);
		x=c/lcm;
		y=d/lcm;
		count=y-x;
		if(c%lcm==0)
			++count;
		cout<<count<<endl;
	}
}
