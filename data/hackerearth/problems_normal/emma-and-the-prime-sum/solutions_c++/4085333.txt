#include<cstdio>
#include<cstring>
#define MAXN 1000000
bool isPrime[MAXN+10];
long long ans[MAXN+10];
void init(){
    memset(isPrime,true,sizeof(isPrime));
    isPrime[1]=false;
    for(int i=4;i<=MAXN;i+=2){
        isPrime[i]=false;
    }
    for(int i=3;i<=MAXN;i+=2){
        if(isPrime[i]){
            for(int j=i+i;j<=MAXN;j+=i)
            isPrime[j]=false;
        }
    }
    ans[1]=0;
    ans[2]=2;
    for(int i=3;i<=MAXN;i++){
        if(isPrime[i]){
            ans[i]=ans[i-1]+i;
        }
        else
            ans[i]=ans[i-1];
    }
    return ;
}
void out(){
    for(int i=1;i<=10;i++){
        printf("%d\n",ans[i]);
    }
    return ;
}
int main(){
    init();
   // out();
    int cs;
    scanf("%d",&cs);
    while(cs--){
        int x,y;
        scanf("%d%d",&x,&y);
        printf("%lld\n",ans[y]-ans[x-1]);
    }
    return 0;
}
