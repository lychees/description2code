#include<bits/stdc++.h>
using namespace std;

int hsh[26][100005]={0};
int main()
{
    char str[100005];
    char ch;
    int Q;
    scanf("%s",str);
    scanf("%c",&ch);
    scanf("%d",&Q);
    int l=strlen(str);
    int i=0;
    int j,t;
    while(str[i]!='\0')
    {
        hsh[str[i]-'a'][i]=1;
        i++;
    }
    for(i=0;i<26;i++)
    {
        t=-1;
        for(j=l;j>=0;j--)
        {
            if(hsh[i][j]==1)
                t=j;
            hsh[i][j]=t;
        }
    }

    while(Q--)
    {
        char q[100005];
        scanf("%s",q);
        int flag=0;
        t=-1;
        i=0;
        while(q[i]!='\0')
        {
            t=hsh[q[i]-'a'][t+1];
            if(t==-1)
            {
                flag=1;
                break;
            }
            i++;
        }
        if(flag==0)
            printf("Yes\n");
        else
            printf("No\n");

    }
}
