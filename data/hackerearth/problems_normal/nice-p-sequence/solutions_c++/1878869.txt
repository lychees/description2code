long long gcd(long long a, long long b) {
	long long r;
	while(b) {
		r=a%b;
		a=b;
		b=r;
	}
	return a;
}

long long ee(long long a, long long b, long long &x1, long long &y1) {
	long long t,q,r,x2,y2;

	x1=1;x2=0;y1=0;y2=1;
	while(b) {
		q=a/b;
		r=a%b;
		a=b;
		b=r;

		t=x2;
		x2=x1-q*x2;
		x1=t;

		t=y2;
		y2=y1-q*y2;
		y1=t;
	}
}

long long find(long long curr) {
	long long a,b,d,x,y;
	a=curr;
	b=p;
	d=gcd(abs(a),abs(b));
	//cout<<d<<endl;
	if(d>1) return -1;
	ee(a,b,x,y);
	while(x<0) {
		x+=b;
		y-=a;
	}
	while(x-b>=0) {
		x-=b;
		y+=a;
	}
	return x;
}
