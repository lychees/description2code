t = int(raw_input())
for _ in range(t):
    n = int(raw_input())
    a = map(int, raw_input().split())
    cnt = 0
    for i in range(n):
        for j in range(i, n):
            if (a[i] == a[j]):
                cnt += 1
    print cnt
