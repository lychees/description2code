#include <iostream>
#include <algorithm>
using namespace std;
struct entry {
    int t;
    int p;
    int in;
};
int comparision(struct entry a,struct entry b) {
    if((a.p/a.t)==(b.p/b.t)) {
		if(a.in<b.in) {
			return 0;
		}
		else {
			return 1;
		}
    }
    else {
		if((a.p/a.t)<(b.p/b.t)) {
			return 1;
		}
		else {
			return 0;
		}
    }
}
int main() {
	int t;
	cin>>t;
	for(int i=0;i<t;++i) {
		int n;
		cin>>n;
		struct entry levels[n];
		for(int j=0;j<n;++j) {
			cin>>levels[j].t;
		}
		for(int j=0;j<n;++j) {
			cin>>levels[j].p;
		}
		for(int j=0;j<n;++j) {
			levels[j].in=j;
		}
		sort(levels,levels+n,comparision);
		cout<<"Case #"<<i+1<<": ";
		for(int j=n-1;j>=0;--j) {
			cout<<levels[j].in<<" ";
		}
		cout<<endl;
	}
    return 0;
}
