using namespace std;
int isprime(long long int s)
{
	 long long int i,flag=1;
	 for(i=2;i<=(sqrt(s));i++)
	 {
	 	if(s%i==0)
	 	{
	 		flag=0;
	 		break;
	 	}
	 }
	 return(flag);
}

{
	long long int t,i,k;
