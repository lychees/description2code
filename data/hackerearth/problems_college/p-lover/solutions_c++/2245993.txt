#include<iostream>
#include<vector>
using namespace std;
bool isPalin(long n){
	long x=0,y=n;
	while(y>=1){
		x=x*10+y%10;
		y/=10;
	}
	return (x==n);

}
bool isPrime(long n){
	if(n%2==0) return false;
	for(int i=3;i*i<=n;i+=2){
		if(n%i==0) return false;
	}
	return true;
}
int main(){
	ios::sync_with_stdio(false);
	vector<long> v;
	v.push_back(2);
	for(long i=3;i<1000000;i++){
		if(isPalin(i)){
			if(isPrime(i)){
				v.push_back(i);
			}
		}
	}
	v.push_back(1003001);
	int x = v.size();

	int t,n;
	cin>>t;
	while(t--){
		cin>>n;
		int h=0;
		while(v[h]<n ) h++;
		cout<<v[h]<<endl;
	}
}
