#include<bits/stdc++.h>
using namespace std;

typedef int (*compfn)(const void*, const void*);
    
struct student
{
    int t,s;
};

inline int comp(struct student *elem1,struct student *elem2) {
    if(elem1->s<elem2->s)
    return 1;
    else if(elem1->s>elem2->s)
    return -1;
    else
    return 0;
}

int main()
{
	int n,q,k,p;
	student arr[100005];
	cin>>n>>q;
	for(int i=0;i<n;i++) {
		cin>>arr[i].t;
	}
	for(int i=0;i<n;i++) {
		cin>>arr[i].s;
	}
	qsort(&arr,n,sizeof(struct student),(compfn)comp);
	int dp[100005];
	dp[0]=arr[0].t;
	for(int i=1;i<n;i++) {
		dp[i]=dp[i-1]+arr[i].t;
	}
	while(q--) {
		cin>>p;
		cout<<dp[p-1]<<endl;
	}
    return 0;
}
