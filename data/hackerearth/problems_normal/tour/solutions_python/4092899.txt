n=int(raw_input())
cities={}
for i in range(n):
    city=raw_input()
    cities[city]=i

#print cities
cost=[]
row=[]
for i in range(n):
    row=[int(i) for i in raw_input().split()]
    cost.append(row)

#print cost

l_v_id=0
tot=0
p=int(raw_input())
for k in range(p):
    d=raw_input()
    v=cities[d]
    tot+=cost[l_v_id][v]
    l_v_id=v

print tot
