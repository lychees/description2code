from __future__ import print_function


def main():
    fib = [2, 3]
    a = 2
    b = 3
    count = 2
    while (count <= 10000):
        c = a+b
        fib = fib+[c]
        count = count+1
        a, b = b, c
    t = input()
    for _ in range(t):
        n = input()
        print(fib[n-1])


main()
