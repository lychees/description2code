#include <iostream>
using namespace std;

int max(int a[], int size)
{	int m=0,h=0;
   	int i;
   	for(i=0;i<size;i++)
   	{	h=h+a[i];
    	if(h<0)
        	h=0;
     	if(m<h)
        	m=h;
    }
    return m;
} 

int main()
{	int x;
	cin>>x;
	int a[x];
	for(int i=0;i<x;++i)
		cin>>a[i];
   	int m=max(a,x);
   	cout<<m;
   return 0;
}
