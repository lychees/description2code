#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    long long int n,i,j,k,r,c,x,flag=0,lag=0;
    for(i=0;i<t;i++){
    	cin>>n;
    	lag=0; flag=0;
    	for(j=0;j<n;j++){
    		cin>>r;
    		cin>>c;
    		cin>>x;
    		if(((r+c)%2==0 && x%2==0) ||((r+c)%2!=0 && x%2!=0)){
    			lag++;
    		}
    		else if(((r+c)%2!=0 && x%2==0) ||((r+c)%2==0 && x%2!=0)){
    	flag++;
    		}
    		
    	}
    	if(lag==n || flag==n){
    		cout<<"Yes"<<endl;
    	}
    	else{
    		cout<<"No"<<endl;
    	}
    }
    return 0;
}
