#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main() {
	// your code cgoes here
	int n,count=0,i,flag=0;
	long long int a[1001],b[1001],d,max=-1,min=1000;
	unordered_map<int,int> check;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%lld",&a[i]);
		if(max<a[i])
		max=a[i];
		check[a[i]]=1;
	}
	sort(a,a+n);
	if(n==2)
	{
		if(check[abs(a[0]-a[1])]==1)
		{
			printf("Bob");
			exit(0);
		}
		else
		{
			count++;
			n++;
			a[n-1]=abs(a[1]-a[0]);
			check[a[n-1]]=1;
			sort(a,a+n);
		}
		
	}
	if(n>=3)
	{
		for(i=0;i<n-1;i++)
		{
			b[i]=a[i+1]-a[i];
			if(min>b[i])
			min=b[i];
		}
		for(i=0;i<n;i++)
		{
			if(b[i]%min!=0)
			{
				flag=1;
				break;
			}
		}
		if(flag==0)
		{
			int temp=max;
			temp=temp-min;
			while(temp>0)
			{
				if(check[temp]==1)
				{}
				else
				{
					count++;
				}
				temp=temp-min;
			}
		}
		else
		{
			count=count+max-n;
		}
	}
	if(count%2==1)
	printf("Alice");
	else
	printf("Bob");
	return 0;
}
