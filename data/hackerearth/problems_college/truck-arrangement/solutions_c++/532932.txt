#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        vector<pair<int,int> > vec;
        for(int i=0;i<n;i++)
        {
            int temp;
            cin>>temp;
            vec.push_back(make_pair(temp,i));
        }
        sort(vec.begin(),vec.end());
        int maxi=INT_MIN;
        for(int i=0;i<n;i++)
        {
            if(abs(vec.at(i).second-i)>maxi)
                maxi=abs(vec.at(i).second-i);
        }
        cout<<maxi<<endl;
    }
}
