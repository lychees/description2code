#include <iostream>
#include<cstdio>
#include<map>
#include<math.h>
using namespace std;

int main()
{
   int t;
   scanf("%d",&t);
  long int a1;

   while(t--)
   {
map<int,int>mp;
       scanf("%ld",&a1);
       int a=a1;
       while(a)
       {
           int i=a%10;

            mp[i]++;
           a=a/10;

       }
int count=0;
       for(long int i=1;i<=sqrt(a1);i++)
       {

           if(a1%i==0)
           {
               int a=i;
               while(a)
       {
           int i=a%10;
           if(mp[i]!=0)
            {count++;
            break;
            }
           a=a/10;

       }
           }


           int j=a1/i;

           if(a1%j==0&&j>sqrt(a1))
           {
               int a=j;
               while(a)
       {
           int i=a%10;
           if(mp[i]!=0)
            {count++;
            break;
            }
           a=a/10;

       }
           }
       }
printf("%ld\n",count);
   }
    return 0;
}
