#include<bits/stdc++.h>
using namespace std;
int a[100000],b[100000];
int main(){
  int N;
  scanf("%d",&N);
  for(int i=0;i<N;i++)
    scanf("%d",&a[i]);
  sort(a,a+N);
  int i=0,j=N-1;
  int k=0;
  while(i<j){
    b[k++]=a[i];
    b[k++]=a[j];
    i++;
    j--;
  }
  if(i==j)
    b[k++]=a[i];
  long long sum=0;
  for(int i=1;i<k;i++)
    sum+=abs(b[i]-b[i-1]);
  sum+=abs(b[0]-b[k-1]);
  printf("%lld\n",sum);
  return 0;
}
