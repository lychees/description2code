#include<iostream>
#include<map>
#include<algorithm>
#define pii pair< int, int >
using namespace std;
string ans[]={"YES","NO"};
int main()
{
	int n,q,a,b;
	cin>>n>>q;
	int color[n+1],status[n+1];
	map<int, pii > rnd;
	for(int i=0;i<=n;i++)
	{
		color[i]=-1;
		status[i]=0;
	}
	for(int i=0;i<q;i++)
	{
		cin>>a>>b;
		if(a>b)
		{
			a=a^b;
			b=a^b;
			a=a^b;
		}
		if(a==b)
		{
			cout<<ans[1]<<endl;
		
		}
		
		else if(color[a]==color[b]&&status[a]==0&&status[b]==0)
		{
			
			int flag=0;
			for(int j=a+1;j<b;j++)
			{
				if(color[j]!=color[a])
				{
					if(rnd[color[j]].first==j&&!(rnd[color[j]].second>=a&&rnd[color[j]].second<=b))
					{
					flag=1;
					break;
					}
					if(rnd[color[j]].second==j&&!(rnd[color[j]].first>=a&&rnd[color[j]].first<=b))
					{
					flag=1;
					break;
					}
				}
			}
			if(flag==0)
			{
			cout<<ans[0]<<endl;
			for(int j=a;j<=b;j++)
			{
				if(color[j]==-1)
				{
				color[j]=i;
				}
				else
				{
					if(rnd[color[j]].first<a&&rnd[color[j]].second>b)
					{
						color[j]=i;
					}
					else if(rnd[color[j]].first>b&&rnd[color[j]].second<a)
					{
						color[j]=i;
					}
				}
			}
			status[a]=1;
			status[b]=1;
			rnd[i]=pii(a,b);
			}
			else
			{
				cout<<ans[1]<<endl;
			}
		}
		else
		{
			cout<<ans[1]<<endl;
		}
		
		
	}
	return 0;
}
