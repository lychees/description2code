#include <cstdio>
#include <vector>
#include <algorithm>
#define MAX 1000010
using namespace std;

int i, n, m, l, r, q, x, sum;
int cnt[MAX] = {0}, pre[MAX] = {0}, coins[MAX] = {0};

int main(void)
{
  scanf("%d %d",&n,&m);
  for(i = 0; i < m; i++)
  {
    scanf("%d %d",&l,&r);
    cnt[l]++; cnt[r+1]--;
  }

  sum = 0;
  for(i = 1; i <= n; i++)
  {
    sum += cnt[i];
    pre[i] = sum;
    coins[sum]++;
  }

  for(i = 1; i < MAX; i++)
    coins[i] += coins[i-1];

  scanf("%d",&q);
  for(i = 0; i < q; i++)
  {
    scanf("%d",&x);
    if (x == 0)
      printf("%d\n",n);
    else
      printf("%d\n",n - coins[x-1]);
  }
  return 0;
}
