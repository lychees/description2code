#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

int main()
{
    long long t,n;
    scanf("%d\n", &t);
    while(t--){
    	scanf("%ld\n", &n);
    	int maxBit = log2l(n)+1;
    	unsigned long long sum=0;
    	for(int i=0; i<=maxBit; i++){
    		for(int j=i+1; j<=maxBit; j++){
    			long long v = (1LL<<i) + (1LL<<j);
    			if (v<=n) {
    				sum += v%1000000007;
    				sum %= 1000000007;
    			}
    		}
    	}
    	printf("%d\n", sum);
    }
    return 0;
}
