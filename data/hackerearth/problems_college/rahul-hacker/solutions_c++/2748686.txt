#include <bits/stdc++.h>
#include <algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;
		cin>>n>>m;
		int rev1,rev2,rev3,dig;
		rev1=rev2=rev3=0;
		while(n)
		{
			dig=n%10;
			rev1=rev1*10+dig;
			n=n/10;
		}
		while(m)
		{
			dig=m%10;
			rev2=rev2*10+dig;
			m=m/10;
		}
		int tempn=rev1+rev2;
		while(tempn)
		{
			dig=tempn%10;
			rev3=rev3*10+dig;
			tempn=tempn/10;
		}
		cout<<rev3<<endl;
	}
    return 0;
}
