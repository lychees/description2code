#include<bits/stdc++.h>
using namespace std;
int ar[11111];
int main()
{
    int t,n,i,j,ans;
    cin>>t;
    while(t--)
    {
        ans=0;
        cin>>n;
        for(i=0;i<n;i++)
            cin>>ar[i];
        sort(ar,ar+n);
        for(i=1;i<n;i++)
            ar[i]-=ar[0];
        for(i=1;i<n;i++){
			if(ar[i]/5 >0){
				ans+=(ar[i]/5);
				ar[i]%=5;
			}
			if(ar[i]/2 >0){
				ans+=(ar[i]/2);
				ar[i]%=2;
			}
			if(ar[i]!=0) ans++;
		}
		cout<<ans<<endl;
    }
}
