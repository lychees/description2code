#include <bits/stdc++.h>
#define ll long long
#define pb push_back
#define mk make_pair
#define pi pair<int, int>
#define fi first
#define se second
using namespace std;
const int N = 1e5 + 5;
const int M = 1e9 + 7;
bool pr[N];
vector < int > v;
int main() {
	int t;
	cin >> t;
	pr[1] = 1;
	for(int i = 2; i < N; i++) {
		if(!pr[i]) {
			for(int j = 2 * i; j < N; j += i) {
				pr[j] = 1;
			}
		}
	}
	for(int i = 2; i < N; i++) {
		if(!pr[i]) {
			v.pb(i);
		}
	}
	while(t--) {
		int n, m;
		scanf("%d%d", &n, &m);
		if(n <= m) {
			printf("0\n");
			continue;
		}
		ll ans1 = 1, ans2 = 1;
		for(int i = 2; i <= n; i++) {
			if(pr[i]) continue; 
			ll ct = 0;
			ll ko = n, l = i;
			while(ko >= l) {
				ct += ko / l;
				l *= i;
			}
			ans1 *= (ct + 1);
			if(ans1 >= M) ans1 %= M;
		}
	//	cout << ans1 << endl;
		for(int i = 2; i <= m; i++) {
			if(pr[i]) continue;
			ll ct = 0;
			ll ko = m, l = i;
			while(ko >= l) {
				ct += ko / l;
				l *= i;
			}
			ans2 *= (ct + 1);
			if(ans2 >= M) ans2 %= M;
		}
		ans1 = (ans1 - ans2 + M) % M;
		printf("%lld\n", ans1);
	}
	return 0;
}
