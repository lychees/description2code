#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T,N;
    cin>>T;
    while(T--)
    {
    	cin>>N;
    	int arr[N],max=0;
    	for(int i=0;i<N;i++)
    	{
    		cin>>arr[i];
    	}
    	max=arr[N-1];
    	vector<int> vec;
    	vec.push_back(max);
    	for(int i=N-2;i>=0;i--)
    	{
    		if(arr[i]>max)
    		{
    			max=arr[i];
    			vec.push_back(arr[i]);
    		}
    	}
    	
    	cout<<vec.size()<<endl;
    }
    return 0;
}
