#include<bits/stdc++.h>
using namespace std;
int prime[4000001]={0};
int nu=0;
int main()
{
	int n;
    int i,j,k;
    long long unsigned int sum=0;
    scanf("%d",&n);
    for(i=2;i*i<=n;i++)
	{
		if(!prime[i])
		{
			for(j=i*i;j<=n;j=j+i)
			{
				prime[j]=1;
			}
		}
		 
	}
	for(i=2;i<n;i++)
	{
		if(!prime[i])
		sum=sum+i;
		
	}
	printf("%llu",sum);
}
