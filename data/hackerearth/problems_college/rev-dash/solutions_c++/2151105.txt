#include <iostream>
#include <cstring>
using namespace std;

int main()
{ string str;
  cin>>str;
  
	  if(str.size()>=3){
	  
	  	  cout<<str[0];
		  for(int i=1;i<str.size();i++){
		  	
		  	cout<<"-"<<str[i];
		  	
		  }
		  cout<<"\n";
		  cout<<str[str.size()-1];
		  for(int i=str.size()-2;i>=0;i--){
		  	
		  	cout<<"-"<<str[i];
		  	
		  }
  
      }
  
  
  
    return 0;
}
