#include <iostream>
using namespace std;

int main()
{
    int test,hr,min,n,a1,a2;
    cin>>test;
    while(test--){
    	cin>>hr>>min;
    	if(hr==12)
    	hr=0;
    	a1=hr*30;
    	a1=a1+(min*0.5);
    	a2=min*6;
    	n=a1-a2;
    	if(n<0)
    	n*=-1;
    	if(n>180){
    		n=360-n;
    	}
    	cout<<n<<endl;
    }
    return 0;
}
