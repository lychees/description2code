#include<bits/stdc++.h>

using namespace std;

#define ff first
#define ss second
#define pb push_back
#define mp make_pair
#define all(x) x.begin(),x.end()
#define sz(x) ((int)x.size())
#define eps 1e-9

const int MOD = 1e9+7;

typedef long long ll;
typedef pair<int,int> pii;

ll POWER[65];
ll power(ll a, ll b) {ll ret=1;while(b) {if(b&1) ret*=a;a*=a;if(ret>=MOD) ret%=MOD;if(a>=MOD) a%=MOD;b>>=1;}return ret;}
ll inv(ll x) {return power(x,MOD-2);}

void precompute() {
	POWER[0]=1;
	for(int i=1;i<63;i++) POWER[i]=POWER[i-1]<<1LL;
}
int n;
string str[35];
int main() {
	//freopen("TASK.in","r",stdin);	
	//freopen("TASK.out","w",stdout);
	precompute();
	int t;
	cin>>t;
	while(t--) {
		int n;
		cin>>n;
		for(int i=0;i<n;i++) cin>>str[i];
		string r;
		cin>>r;
		vector<int> needed(26,0);
		for(int i=0;i<sz(r);i++) needed[r[i]-'a']++;
		int f=n/2,s=n-f;
		bool ans=false;
		set<vector<int> > st;
		for(int i=0;i<POWER[f];i++) {
			vector<int> v(26,0);
			for(int j=0;j<f;j++) {
				if(POWER[j]&i) {
					for(int k=0;k<sz(str[j]);k++) v[str[j][k]-'a']++;
				}
			}
			st.insert(v);
		}
		for(int i=0;i<POWER[s];i++) {
			vector<int> v(26,0);
			for(int j=0;j<s;j++) {
				if(POWER[j]&i) {
					for(int k=0;k<sz(str[f+j]);k++) v[str[f+j][k]-'a']++;
				}
			}
			for(int j=0;j<26;j++) v[j]=needed[j]-v[j];
			if(st.find(v)!=st.end()) ans=true;
		}
		if(ans) cout<<"YES\n";
		else cout<<"NO\n";
	}
	return 0;
}
