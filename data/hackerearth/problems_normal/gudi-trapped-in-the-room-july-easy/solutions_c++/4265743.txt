#include<bits/stdc++.h>
#define ll long long int
using namespace std;
set<string> s;
int add,shift;

void dfs(string tmp)
{
    s.insert(tmp);
    string next=tmp;
    int i,n=tmp.size();
    for(i=0;i<n;i++)
    {
        next[(i+shift)%n]=tmp[i];
    }
    if(s.find(next)==s.end()) dfs(next);
    next=tmp;
    for(i=0;i<n;i++)
    {
        if(i%2) next[i]='0'+(next[i]-'0'+add)%10;
    }
    if(s.find(next)==s.end()) dfs(next);
    return;
}

int main()
{
	ios_base::sync_with_stdio(false); cin.tie(0);
	int t; cin>>t;
	while(t--)
	{
	    string inp;
	    cin>>inp>>add>>shift;
	    s.clear();
	    dfs(inp);
	    cout<<(*s.begin())<<endl;
	}
	return 0;
}
