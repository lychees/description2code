int recurse(string str,int i,int index,vector<char> vec[],vector<char> vec1[])
{

    if(str[i]>=97&&str[i]<=132)
    {
        if(index>=0)
            vec[index].push_back(str[i]);
        else
            vec1[-1*index].push_back(str[i]);
        //arr[str[i]]=index;
    }
    else if(str[i]=='.')
    {
        return i;
    }
    if(str[i+1]=='(')
    {
        i=recurse(str,i+2,index-1,vec,vec1);
    }
    if(str[i+1]==')')
        return i+1;
    i=recurse(str,i+1,index+1,vec,vec1);
    if(str[i+1]==')')
        return i+1;
    return i;
}
