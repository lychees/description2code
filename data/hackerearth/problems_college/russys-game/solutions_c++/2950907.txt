#include <bits/stdc++.h>
using namespace std;

int main() {
	long long int t,n,i,j,k,l,g;
	cin>>t;
	while(t--){
		cin>>n;
		g=0;
		long long int a[n];
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		
		j=a[0];
		
		if(j==-1)
		g=-1;
		else{
			while(n--){
				if(a[j]==-1)
				{
					g=1;
					break;
				}
				else
				j=a[j];
			}
		}
		
		if(g)
		cout<<"Win\n";
		else
		cout<<"Lose\n";
		
	}
}
