t=input()
while t:
    (n,k)=map(int,raw_input().split())
    a=map(int,raw_input().split())
    flag=1
    for i in range(n):
        if a[i]>0:
            flag=0
    if flag:
        print 0
    else:
        dp=[0]*n
        dp[0]=max(0,a[0])
        for i in range(1,k+1):
                dp[i]=max(a[i],dp[i-1])
        for i in range(k+1,n):
            dp[i]=max(dp[i-1],dp[i-(k+1)]+a[i])
        print dp[n-1]
    t-=1
