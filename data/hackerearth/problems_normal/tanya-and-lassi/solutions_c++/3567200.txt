#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int l;
    	cin>>l;
    	vector<int> val(l);
    	for(int i=0;i<l;i++) cin>>val[i];
    	long long dp[10000];
    	dp[0] = val[0];
    	for(int i=1;i<l;i++){
    		dp[i] = val[i];
    		for(int j=i-1;j>=(i-1)/2;j--){
    			if(dp[i]<(dp[j] + dp[i-j-1])) {
    				dp[i] = dp[j] + dp[i-j-1];
    			}
    		}
    	}
    	cout<<dp[l-1]<<endl;
    }
    return 0;
}
