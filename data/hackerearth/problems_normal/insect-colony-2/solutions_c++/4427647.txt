#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int t,a,b,n,c;
    
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	c=0;
    	if(n==1)
    	{
    		cin>>a;
    		(a%2==0)?cout<<"Yes":cout<<"No";
    		cout<<endl;
    	}
    	
    	else
    	{
    		cin>>a;
    		int ctr=0;
    		while(--n)
    		{
    			ctr++;
    			cin>>b;
    			c+=b;
    		}
    		//cout<<ctr<<" ";
    		((c+a)%2==0)?cout<<"Yes":cout<<"No";
    		cout<<endl;
    		
    	}
    }
    
    return 0;
}
