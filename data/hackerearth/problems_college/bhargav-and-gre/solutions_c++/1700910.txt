#include <iostream>

using namespace std;

int main()
{
	string w;
	int n;
	int a, s;
	int sz;

	cin >> w;
	cin >> n;

	a = 0;
	s = 0;
	sz = w.length();

	while (n--) {
		string str;

		cin >> str;

		if ((str.length() == sz) && (str[0] == w[0])) {
			//cout << str[0] << " " << w[0] << endl;
			s++;
		} else if ((str.length() == sz) && (str[0] == w[0] + 3)) {
			//cout << str[0] << " " << w[0] << " " << w[0] + 3 << endl;
			a++;
		}
	}
	cout << s << " " << a << endl;
	
	return 0;
}
