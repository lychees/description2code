#include <iostream>
#include<cstring>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	long int x=s.length(),i,count=0;
    	for(i=0;i<x-1;i++)
    	{
    		if(s[i]==s[i+1])
    		count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
