#include <iostream>
#include <iomanip>
using namespace std;
 
#define MAXMASK 1000000
#define N 20
 
int n, k, m;
int a[N];
double p[N];
double dp[MAXMASK][N];
int tp[N];
 
inline void solve(){
	int mask, i, j, l, tm;
	cin>>n>>k>>m;
	for(i = 0; i < n; i++){
		cin>>a[i]>>p[i];
		p[i] /= 100;
	}
	for(mask = 0; mask < tp[n]; mask++){
		dp[mask][0] = 1;
		l = 0;
		tm = m;
		for(i = 0; i < n; i++){
			if(mask & tp[i]){
				l++;
			}
			else{
				tm -= a[i];
			}
			dp[mask][i + 1] = 0;
		}
		if(tm < 0){
			dp[mask][0] = 0;
			continue;
		}
		for(j = 1; j <= l; j++){
			for(i = 0; i < n; i++){
				if(mask & tp[i]){
					if(tm >= a[i])
					dp[mask][j] = max(dp[mask][j], p[i] * dp[mask - tp[i]][j - 1] + (1 - p[i]) * dp[mask - tp[i]][j]);
				}
			}
			//cout<<mask<<" "<<j<<" "<<dp[mask][j]<<endl;
		}
	}
	printf("%0.6f\n", dp[tp[n] - 1][k]);
}
 
inline void pre(){
	int i;
	tp[0] = 1;
	for(i = 1; i < N; i++){
		tp[i] = tp[i - 1] + tp[i - 1];
	}
}
 
int main()
{
	pre();
 	int t;
    cin>>t;
    while(t--){
    	solve();
    }
    return 0;
}
