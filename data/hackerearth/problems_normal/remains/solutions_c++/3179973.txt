#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	long long int x, y, n, ans, temp;
    	cin>>x>>y>>n;
    	ans = x+y;
    	n-=2;
    	while(n--)
    	{
    		temp = (x-y);
    		if(temp<0)
    			temp *= -1;
    		//cout<<temp<<endl;
    		x = y;
    		y = temp;
    		ans += temp;
    		if(x == y)
    		{
    			long long int t = n/3;
    			y+=x;
    			ans += t*y;
    			long long int left = n%3;
    			if(left == 2)
    				ans += x;
    			break;
    		}
    	}
    	cout<<ans<<endl;
    }
    
    return 0;
}
