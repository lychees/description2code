#include <iostream>
#include <algorithm>

using namespace std;

main()
{
	int testcases;
	cin>>testcases;
	while(testcases--){
		long A,B,N;
		cin>>A>>B>>N;
		long L,R;
		L=1;
		R= max(A,B)*N;
		while(L<=R){
			long M = (L+R)/2;
			long factors = M/A+M/B-(M/((A*B)/__gcd(A,B)));
			if(factors==N)
			 {
			  cout<<max((M/A)*A, (M/B)*B)<<"\n";
			  break;
			}
			else if(factors<N)
			 L = M+1;
			else
			 R = M-1; 
			 	 
		}
	}
}
