#include<bits/stdc++.h>
using namespace std;
#define ll long long 
int main()
{	
    int t;
    cin>>t;
    while(t--)
    {
    int n,q;
    vector< pair< pair<int,int> , int> > vi(100001); 
    cin>>n>>q;
    int l,r,c;
    for(int i=1;i<=q;i++)
    {
        cin>>l>>r>>c;
        vi[i]={{l,r},c};
    }
    int qu;
    cin>>qu;
    int qe[100001]={0};
    while(qu--)
    {
       int l,r;
       cin>>l>>r;
       qe[l]++;
       qe[r+1]--;
    }
    for(int i=1;i<=q;i++)
        qe[i]+=qe[i-1];
    ll val[100001]={0};
    for(int i=1;i<=q;i++)
    {
        int l=vi[i].first.first;
        int r=vi[i].first.second;
        int c=vi[i].second;
        val[l]+=1LL*c*qe[i];
        val[r+1]-=1LL*c*qe[i];
    }
    for(int i=1;i<=n;i++)
        val[i]+=val[i-1];
    for(int i=1;i<=n;i++)
        cout<<val[i]<<" ";
    cout<<endl;
    }
    return 0;
}
