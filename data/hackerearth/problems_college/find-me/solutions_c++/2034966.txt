#include <iostream>
using namespace std;

int main()
{
    int t;
    bool flag = false;
    long i,n,p,count;
    cin >> t;
    while(t--){
    	cin>>n>>p;
    	count = 0;
    	flag = false;
    	long arr[n];
    	for(i=0;i<n;i++){
    		cin >> arr[i];
    		if(arr[i] == p)
    		flag = true;
    	}
    	if(flag){
    		for(i=0;i<n;i++){
    			if(arr[i]>p){
    				count++;
    			}
    		}
    		cout<<"YES "<<count<<endl;
    	} else {
    		cout<<"NO "<<count<<endl;
    	}
    }
    return 0;
}
