long long  binarry(long long  m,long long  n)
{
    long long  i=0,j=0;
    long long  a[100],b[100];

    do
    {
        a[i++]=m%2;
        m=m/2;
    }
    while(m!=0);

    do
    {
        b[j++]=n%2;
        n=n/2;
    }
    while(n!=0);

    if(i==j)
    {
        long long  ans=0;
        for(i=j-1;i>=0;i--)
        {
            if(a[i]==b[i])
            ans=ans+a[i]*pow(2,i);
            else
            break;
        }
        return ans;
    }
    else
    return 0;


}
