#include <iostream>
#include<vector>
#include<algorithm>
#include<stdio.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	int n;
    	cin>>n;
    	vector<int> vec;
    	vec.resize(n);
    	
    	for(int i=0;i<n;i++)
    		scanf("%d",&vec[i]);
    	sort(vec.begin(),vec.end());
    	int tt=0;
    	int i;
    	int flag=1;
    	for(i=0;i<n;i++)
    	{
    		if(vec[i]<=tt)
    		{
    			flag=0;
    			break;
    		}
    		tt++;
    		
    		if((i+1)%6==0)
    			tt++;
    	}
    	if(flag==0)
    	{
    		cout<<"Goodbye Rick\n";
    		cout<<i<<endl;
    	}
    	else
    		cout<<"Rick now go and save Carl and Judas\n";
    	t--;
    }
    
}
