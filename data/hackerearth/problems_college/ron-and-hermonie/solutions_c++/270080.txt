#include <iostream>
#include <stdio.h>
using namespace std;
int main(int argc, char const *argv[])
{
	int t,i,flag,f,rep;
	scanf("%d",&t);
	while(t--){
		unsigned long long int b,e;
		scanf("%llu%llu",&b,&e);
		if(b==0)printf("0\n");
		else if(b==1)printf("1\n");
		else{
		f=1;flag=1;rep=0;
		int a[10];
		for(i=0;i<10;i++)
			a[i]=-1;
		b%=10;
		while(1){
			f=(f*b)%10;
			if(a[f]!=-1)
				break;
			rep++;
			a[f]=rep;
			
			if(rep==e){
				printf("%d\n",f);
				flag=0;
				break;
				
			}
		}
		
		if(flag==1){
			e=e%rep;
			for(i=0;i<10;i++)
				if(a[i]!=-1)
				a[i]%=rep;
		
			for(i=0;i<10;i++){
				if(a[i]==e){
					printf("%d\n",i);
				break;
			}
			}
		}

	}
}
	return 0;
}
