#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i,j,k;
		cin>>n;
		char a[n][n+2];
		for(i=0;i<n;i++)
		{
			scanf("%s",&a[i]);
		}
		int f1=1,f2=1;
		for(i=0;i<=n/2;i++)
		{
			for(j=0;j<n;j++)
			{
				if(a[i][j] != a[n-1-i][j])
				{
					f1=0;
					break;
				}
			}
		}
		
		for(i=0;i<=n/2;i++)
		{
			for(j=0;j<n;j++)
			{
				if(a[j][i] != a[j][n-1-i])
				{
					f2=0;
					break;
				}
			}
		}
		
		if(f1&&f2)
			cout<<"BOTH"<<endl;
		else if(f1)
			cout<<"HORIZONTAL"<<endl;
		else if(f2)
			cout<<"VERTICAL"<<endl;
		else
			cout<<"NO"<<endl;
	}
	return 0;
}
