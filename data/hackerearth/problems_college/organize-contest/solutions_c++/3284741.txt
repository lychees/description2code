#include <bits/stdc++.h>
using namespace std;
typedef struct player
{
    char name[33];
    int points;
}P;


int main()
{
  
   int i,j,k,m=0,N,temp,max,pos;
P *p,arr[1001];     
  char name[33];
   scanf("%d",&N);
   for(i=0;i<N;i++)
   {
       
       //printf("%s %d \n",arr[i].name,arr[i].points);
     if(i==0)
     {  
        cin>>arr[m].name>>arr[m].points;  
        m++;    
        max=arr[0].points;
     }
     else{
          cin>>name>>temp;  
         for(j=0;j<m;j++)
         {
             if(strcmp(arr[j].name,name)==0)
             break;
         }
         if(j==m)
         {
             strcpy(arr[m].name,name);
             arr[m].points=temp;
             m++;
         }
         else
         {
             arr[j].points+=temp;
         }
        for(k=0;k<m;k++)
        {   
              if(max<arr[k].points)
              {
                  max=arr[k].points;
                  pos=k;
              }
        }
        
     }
   }
  
   cout<<arr[pos].name;
    return 0;
}
