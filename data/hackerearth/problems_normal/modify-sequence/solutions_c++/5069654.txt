#include <iostream>
using namespace std;

bool can_reduce_cur_to_zero(int prev, int cur, int next) {
	return prev + next >= cur;
}

int main()
{
    int ncount; cin >> ncount;
    
	int lsb_count = 0;
	int last; cin >> last;
	for (int n = 1; n < ncount; ++n) {
		int cur; cin >> cur;
		//cout << last << "," << cur << endl;
		int diff = cur - last;
		if (diff < 0) {
			cout << "NO" << endl;
			return 0;
		}
		last = cur - last;
	}
	if (last != 0) {
		cout << "NO" << endl;
	} else {
		cout << "YES" << endl;
	}
    return 0;
}
