#define MOD 1000000007

    cout << e1 << " " << e2 << endl;
}

vector<int> G[MAX],R[MAX],temp;
stack<int> s;
bool visited[MAX]={false},a[MAX]={false};
int n,m,f=1,c;

void dfs(int i) {
    visited[i]=true;
    if(!f)
        temp.pb(i);
    for(v(it)=(f?G[i].begin():R[i].begin());it!=(f?G[i].end():R[i].end());it++)
        if(!visited[*it]) {
            c++;
            dfs(*it);
        }
    if(f)
        s.push(i);
}

int main() {
    int u,v;
    ios_base::sync_with_stdio(false);
    cin>>n>>m;
    rep(i,m) {
        cin>>u>>v;
        G[u].pb(v);
        R[v].pb(u);
    }
    REP(1,n)
        if(!visited[i])
            dfs(i);
    f=0;
    mem(visited,false);
    while(em(s)) {
        if(!visited[s.top()]) {
            c=1,dfs(s.top());
            if(c>1)
                while(em(temp))
                    a[temp.back()]=true,temp.pop_back();
        }
        temp.clear();
        s.pop();
    }
    REP(1,n)
        cout<<a[i]<<" ";
    return 0;
}
