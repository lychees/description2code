#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int a[1000];
    a[0] = 0;
    a[1] = 1;
    long long int sum = 0;
    long long int i,b;
    for(i=2;i<100;i++) {
    	a[i] = a[i-1] + a[i-2];
    }
    int n;
    scanf("%d",&n);
    while(n--) {
    	scanf("%lld",&b);
    	sum = 0;
    	for(i=0;a[i]<=b;i++) {
    		if(a[i]%2==0)
    		sum = sum + a[i];
    	}
    	printf("%lld\n",sum);
    }
    return 0;
}
