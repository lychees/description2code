n = int(raw_input())
c = 0
op = [0]*n
l = []
for i in range(n):
    l.append(raw_input())
for i in range(n):
    if op[i] == 0:
        op[i] = 1
        c = c+1
        a = list(l[i])
        l1 = len(a)
        for j in range(i+1, n):
            b = list(l[j])
            l2 = len(b)
            if l1 == l2 and a[0] == b[0] and a[l1 -
                                               1] == b[l2-1] and sorted(a) == sorted(b):
                op[j] = 1

print c
