#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int T;
    long long N;
    cin>>T;
    
    while(T--)
    {
    	cin>>N;
    	long long m=sqrt(N);
    	if(m*m==N)
    	cout<<"YES\n";
    	else if(N-m*m<(m+1)*(m+1)-N)
    	{
    		cout<<"-"<<N-m*m<<"\n";
    	}
    	else cout<<"+"<<(m+1)*(m+1)-N<<"\n";
    }
    return 0;
}
