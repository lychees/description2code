#include <bits/stdc++.h>
//zone
using namespace std;

int main()
{
    int t;
    cin >> t;
    string s;
    string r;
    long long int ans;
    while(t--)
    {
    	cin >> s;
    	r = s;
    	ans = 1;
    	reverse(s.begin(),s.end());
    	if(r == s)
    	{
    		cout<<"Palindrome\n";
    	}
    	else
    	{
    		for(int i = 0; i < s.length(); i++ )
    		{
    			ans = ans * (s[i]-96);
    		}
    		cout<<ans<<endl;
    	}
    }
    return 0;
}
