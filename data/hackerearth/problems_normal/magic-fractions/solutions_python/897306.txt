def fun():
	n = int(input())
	if(n<2):
		return 0;
	if(n==2):
		return 1;
	ans = 1
	prev = 1
	for i in range(3,n+1):
		isprime = True
		num =i
		j = 2
		while j*j<=i:
			if(num%j==0):
				isprime=False
				break
			j = j+1
		if(isprime):
			prev = prev*2
		ans = ans + prev
	return ans
print(fun())
