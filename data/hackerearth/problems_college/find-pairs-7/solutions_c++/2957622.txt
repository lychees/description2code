#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		int n,c=0;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++){cin>>a[i];}
		for(int i=0;i<n;i++){
			for(int j=0;j<n;j++){
				if(j>=i && a[i]==a[j]){c++;}
			}
		} 
		cout<<c<<endl;
	}
   // cout << "Hello World!" << endl;
    return 0;
}
