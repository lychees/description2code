#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
ll n,h;
ll d[100010];

int check(ll m){
    ll height=(m/n)*d[n-1];
    ll nh=h- m*(m+1)/2;
    if(m%n)
        height+=d[m%n-1];
    if(height>=nh){
        return 1;
    }
    return 0;
}

int main()
{
    #ifdef LOCAL_MACHINE
        freopen("in.txt", "r", stdin);
    #endif // LOCAL_MACHINE
    cin >> n >> h;
    memset(d,0,sizeof d);
    cin >> d[0];
    for(int i=1;i<n;++i){
        cin >> d[i];
        d[i]+=d[i-1];
    }
    for(int i=1;i<=n;++i){
        if(check(i)){
            cout << i << endl;
            return 0;
        }
    }
    ll l=1, r=2e9+100, ans(0);
    for(int i=0;i<50;++i){

        ll m=l+(r-l)/2;
    // cout << m << endl;
        if(check(m)){
            r=m-1;
            ans=m;
        }else
            l=m+1;
    }
    cout << ans << endl;

    return 0;
}
