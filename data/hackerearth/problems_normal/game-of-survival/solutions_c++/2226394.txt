#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	int t,n,flag;
	cin>>t;
while(t--)
{
	cin>>n;
	flag=1;
	int sword[n];
	int animal[n];
	for(int i=0;i<n;i++)
		cin>>sword[i];
	for(int i=0;i<n;i++)
		cin>>animal[i];
	sort(sword,sword+n);
	sort(animal,animal+n);
	for(int i=0;i<n;i++)
	{
		if(animal[i]>=sword[i])
		{	flag=0;
			break;
		}
	}
	if(flag==0)
		cout<<"NO"<<endl;
	else
		cout<<"YES"<<endl;
}
return 0;
}
