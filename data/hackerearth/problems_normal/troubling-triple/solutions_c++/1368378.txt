#include <bits/stdc++.h>
using namespace std;
#define int long long
vector <int> num; map <int, int> is; 
int lessthan[1000005]; 
main()
{
    int a,b; cin >> a >> b;
    for (int g=0; g<a; g++)
    {
    	int d; cin >> d; num.push_back(d); 
    	is[d]++; 
    }
    for (int g=1; g<=1000001; g++)
    {
    	lessthan[g]=lessthan[g-1]+is[g]; 
    }
    int totans=0; 
    for (int g=0; g<num.size(); g++)
    {
    	for (int y=0; y<num.size(); y++)
    	{
    		if (g==y) continue;
    		int l=b/(num[g]*num[y]); 
    		int r=lessthan[l]; 
    		if (num[g]<=l) r--; 
    		if (num[y]<=l) r--; 
    		totans+=r; 
    	}
    }
    cout << totans/6; 
    return 0;
}
