#include <bits/stdc++.h>
using namespace std;
#define maxi 1000005
#define ll long long int
#define f first
#define s second
pair <ll,ll> p[maxi];
ll val[maxi];

int main()
{
	ll i,n,d,a,b,q,k,sum;
	cin >> n;
	for(i=0;i<n;i++){
		cin >> a >> b;
		p[i]={a,b};
	}
	sort(p,p+n);
	k=n-1;
	sum=0;
	d=p[n-1].f;
	for(i=d;i>=0;i--){
		if(i<p[k].f && i==p[k-1].f){
			k--;i++;
		}
		if(i==p[k].f){
			sum=max( sum,p[k].s );
			val[i]=sum;
		}
		else{
			val[i]=sum;
		}
	}
	cin >> q;
	sum=0;
	while(q--){
		cin >> k;
		k=(val[k]/k)*100;
		sum+=k;
	}
	cout << sum << endl;
    return 0;
}
