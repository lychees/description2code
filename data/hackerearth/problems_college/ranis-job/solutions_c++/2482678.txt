#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,sum,c,j;
    cin>>t;
    while(t--)
    {
    	c=0;
    	sum=0;
    	cin>>n;
    	int a[n],b[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		sum+=a[i];
    	}
    	for(int i=0;i<n;i++)
    	cin>>b[i];
    	sort(b,b+n);
    	j=n-1;
    	while(sum>0&&j>=0)
    	{
    		sum-=b[j--];
    		c++;
    	}
    	if(sum>0&&j==-1)
    	cout<<"Impossible\n";
    	else
    	cout<<c<<"\n";
    }
    return 0;
}
