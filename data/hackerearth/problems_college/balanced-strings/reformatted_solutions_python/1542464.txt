t = input()
for x in range(0, t):
    a = raw_input()
    a = list(a)
    lis = []
    for x in range(0, 26):
        lis.append(0)
    for x in a:
        i = ord(x)-ord('a')
        lis[i] = lis[i]+1
    test = lis[0]
    flag = 0
    for x in lis:
        if x != test:
            print("No")
            flag = 1
            break
    if flag == 0:
        print("Yes")
