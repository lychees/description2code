#include <iostream>
using namespace std;

int main()
{
    char str[100];
    cin>>str;
    for(int i=0;str[i]!=0;i++)
    {
    	if(isupper(str[i]))
    	{
    	str[i]=tolower(str[i]);
    	}
    	else if(islower(str[i]))
    	{
    	str[i]=toupper(str[i]);
    	}
    }
    for(int i=0;str[i]!=0;i++)
    cout<<str[i];
    return 0;
}
