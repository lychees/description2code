VL prime;
bool is_prime[10000];
void sieve(ll N)  // O(sqrt(N) * N)
{
    MEM(is_prime,N,1);
    is_prime[1]=is_prime[0]=0;
    l i,j;
    for(i=2;i*i<=N;++i)
    {
        if(is_prime[i])
        {
            for(j=2*i;j<=N;j+=i)
                is_prime[j]=0;
        }
    }
    for(i=2;i<=N;++i)
        if(is_prime[i])
            prime.pb(i);
}
bool primecheck(ll N)
{
    l i;
    FORN(i,prime.size())
    {
        if(prime[i]>sqrt(N)+1)
            break;
        if(N%prime[i]==0)
            return 0;
    }
    return 1;
}
map<l,long> dp;
void pre(l N)
{
    l i,lol,num1,num2,num3;
    FORAB(i,3,N)
    {
        num1=(i*i)-(i-1);
        num2=num1-(i-1);
        num3=num2-(i-1);
        lol=0;
        lol+=primecheck(num1)+primecheck(num2)+primecheck(num3);
        if(i==3)
            dp[i]=lol;
        else
            dp[i]=dp[i-2]+lol;

    }
}

      }

 }
