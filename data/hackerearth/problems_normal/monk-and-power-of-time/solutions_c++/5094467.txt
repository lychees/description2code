#include<iostream>
#include<vector>
#include<queue>
using namespace std;
int main(){
    int n;
    cin>>n;
    vector<int> calling(n);
    vector<int> ideal(n);
    queue<int> q;
    for(int i=0;i<n;i++){
        cin>>calling[i];
        q.push(calling[i]);
    }
    int time=0;
    for(int i=0;i<n;i++){
        cin>>ideal[i];
        while(q.front() != ideal[i]){
            int temp = q.front();
            q.pop();
            q.push(temp);
            time++;
            //cout<<"\n "<<temp<<" popped and pushed "<<time;
        }
        q.pop();
        time++;
    }
    cout<<time;
    return 0;
}
