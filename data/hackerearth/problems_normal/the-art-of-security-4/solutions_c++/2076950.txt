#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
#include<numeric>
using namespace std;
int n,i,res,s,j;
int A[10];
int main()
{
  scanf("%d",&n);
  for(i=0;i<n;i++)
    scanf("%d",&A[i]);
  sort(A,A+n);
  do
  {
    for(i=0;i<1<<n;i++)
    {
      for(j=s=0;j<n;j++)
      {
        if(i&1<<j)
          s+=A[j];
        else
          s-=A[j];
        if(s<0)
          break;
      }
      res+=j==n;
    }
  }while(next_permutation(A,A+n));
  printf("%d %d\n%s\n",res,accumulate(A,A+n,0),accumulate(A,A+n,0)>=res?"Got no way out!":"We will win!");
  return 0;
}
