#include <iostream>
using namespace std;
#include<string.h>
int main()
{
   // cout << "Hello World!" << endl;
   int n;
   cin>>n;
   while(n--)
   {
   	char s[3000];
   	int i,a,b=0,l;
   	cin>>s;
   	
   	l=strlen(s);
   	for(i=0;s[i];i++)
   	if(s[i]!='0')
   	break;
   	
   	if(i==l)
   	{cout<<0<<'\n';
   	 continue;
   	}
   	a=s[l-1]-'0';
   	if(l-2>=0)b=s[l-2]-'0';
   	//if(a%9==0)a=9;
   //	if(3*b%9==0)b=9;
   	int ans=(a+b*3)%9;
   	if(ans==0)ans=9;
   	cout<<ans<<'\n';
   	
   }
   
   return 0;
}
