#include<iostream>
using namespace std;
int main()
{
    int hsh[100001]={0};
       int n , m;
       cin >> n >> m;
         int x ;
         for(int i = 1 ; i <= n ; ++i ){
            cin >> x;
              hsh[x] = 1;
              
         
         }
         int ans = 0;
         int i = 1;
         while(true)
         {
           int val = m*i;
           if(hsh[val]==0)
            break;
            else
            ans = val;
            i++;
         }
      cout << ans << endl;
}
