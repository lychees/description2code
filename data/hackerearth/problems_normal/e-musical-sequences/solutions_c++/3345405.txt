#include <iostream>
#define ll long long
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        ll n,m,z;
        cin>>n>>m>>z;
        ll a[n+1];
        for(ll i=0;i<n;i++)
            cin>>a[i];
        ll sign=1;a[n]=0;
        for(ll i=1;i<=n;i++)
        {
            a[n]=a[n]+sign*a[n-i];
            sign=sign*-1;
        }
        a[n]=a[n]%m;
        if(a[n]<0)a[n]=a[n]+m;
        
        if(z<=n)cout<<a[z]%m<<endl;
        else
        if(n%2==0)
        {
           ll h=z-n-1;
           ll d=h/(n+1);
           ll rem=h%(n+1);
           if(d%2==0)
           {if(a[rem]==0)cout<<a[rem]%m<<endl;
           else
             cout<<m-(a[rem]%m)<<endl;  
           }
           else
           {
           	 cout<<a[rem]%m<<endl;
           }


        }
        else
        {
           ll h=z-n-1;
           ll rem=h%(n+1);
           
           	 cout<<a[rem]%m<<endl;
           
        }
    }
    return 0;
}
