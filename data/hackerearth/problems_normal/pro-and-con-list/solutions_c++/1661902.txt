#include <stdio.h>
#include <bits/stdc++.h>

using namespace std;

long long int a [1000001];

int main ()
{
	long long int i, j, n, sum, x, y, t;
	
	scanf ("%lld", &t);
	
	while (t--) {
	
	
	scanf ("%lld", &n);
	sum = 0;
	for (i = 0; i < n; i++) {
		scanf ("%lld%lld", &x, &y);
		a [i] = x + y;
		sum = sum + y;
	}
	
	sort (a, a + n);
	for (i = 0; i < n; i++) {
		//printf ("%d ", a [i]);
	}
	printf ("%lld\n", a [n - 1] + a [n - 2] - sum);
	}
	return 0;
}
