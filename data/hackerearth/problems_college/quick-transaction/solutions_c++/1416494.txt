#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,i,j,ans;
	int n;
	scanf("%d",&t);
	while (t --) {
		scanf("%d",&n);
		ans = 0;
		ans += n / 1000;
		n = n % 1000;
		ans += n/500;
		n = n % 500;
		ans += n/100;
		n = n % 100;
		ans += n / 50;
		n = n % 50;
		ans += n/20;
		n = n % 20;
		ans += n/10;
		n = n % 10;
		ans += n/5;
		n = n % 5;
		ans += n/2;
		n = n % 2;
		ans += n/1;
		n = n % 1;
		printf("%d\n",ans);
	}
	
	return 0;
}
