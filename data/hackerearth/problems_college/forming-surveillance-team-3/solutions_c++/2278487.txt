#include <iostream>
#include <set>
using namespace std;

int main()
{
    long long int t,m,i,ip;
    set<long long int> g;
    cin>>t;
    while(t--)
    {
    	cin>>m;
    	for(i=0;i<m;i++)
    	{
    		cin>>ip;
    		g.insert(ip);
    	}
    	
    	cout<<g.size()<<endl;
    	g.clear();
    	
    }
    
    return 0;
}
