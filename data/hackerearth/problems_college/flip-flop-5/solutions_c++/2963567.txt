#include<iostream>
#include<cstdlib>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char str[100000];
		cin>>str;
		int l=strlen(str);
		int c=0;
		for(int i=0;i<l-1;i++)
		{
			if(str[i]==str[i+1])
			{
				c++;
			}
		}
		cout<<c<<"\n";
	}
}
