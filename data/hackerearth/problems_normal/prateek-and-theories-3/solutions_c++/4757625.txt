#include <bits/stdc++.h>

using namespace std;

int main()
{
	map <int, int> m;
	int t, n;
	scanf("%d", &t);
	while (t--) {
		scanf("%d", &n);
		int x, y;
		m.clear();
		for (int i = 0; i < n; i++) {
			scanf("%d %d", &x, &y);
			m[x]++;
			m[y]--;
		}
		
		int mx = INT_MIN;
		map <int, int> :: iterator i;
		int ans = 0;
		for (i = m.begin(); i != m.end(); i++) {
			ans += i->second;
			mx = max(mx, ans);
		}
		printf("%d\n", mx);
	}
	
	return 0;
}
