#include <bits/stdc++.h>
using namespace std;
unsigned long long int fib[102];
void init()
{
	fib[0]=0;
	fib[1]=1;
	int i;
	for(i=2;i<=101;i++)
	{
		fib[i]=fib[i-1]+fib[i-2];
	}
}
int main()
{
	init();
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
       	cout<<fib[n+1]<<endl;
    }
}
