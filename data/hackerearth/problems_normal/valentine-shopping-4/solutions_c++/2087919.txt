#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,m;
    	double x,y,z,price=100.0;
    	double d1,after_dis,max_dis,pos;
    	scanf("%d %d",&n,&m);
    	for(int i=1;i<=n;i++)
    	{
    		max_dis=INT_MAX;
    		pos=-1;
    		for(int j=1;j<=m;j++)
    		{
    			scanf("%lf %lf %lf",&x,&y,&z);
    			//d1=(x+y-(x*y)/100);
    			//cout<<x<<" "<<y<<" "<<z<<endl;
    			double p=(1-(x/100));
    			double q=(1-(y/100));
    			double r=(1-(z/100));
    			after_dis=p*q*r*price;
    			//cout<<after_dis<<endl;
    			if(max_dis>after_dis)
    			{
    				max_dis=after_dis;
    				pos=j;
    			}
    			//cout<<j<<endl;
    		}
    		//printf("%d ",pos);
    		cout<<pos<<" ";
    	}
    	printf("\n");
    }
}
