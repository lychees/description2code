char grid[2009][2009];
int DP[2009][2009];
int DPV[2009][2009];
bool check(int i,int j){
    if(grid[i][j]!='_') return false;
    bool turn=true;
    for(int x=0 ; x<32 ; x++){
        if(turn){
            if(grid[i+x][j]!='_' || grid[i][j+x]!='_') return false;
        }else {
            if(grid[i+x][j]!='#' || grid[i][j+x]!='#') return false;
        }
        if(DP[i+x][j+31]<32) return false;
        if(DPV[i+31][j+x]<32) return false;
        turn=!turn;
    }
    return true;
}
int main(){
    #ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
    #endif
    int n,m;
    sc2i(n,m);
    for(int i=1 ; i<=n ;i++){
        scanf("%s",&grid[i][1]);
    }
    memset(DP,0,sizeof(DP));
    for(int i=1 ; i<=n ; i++){
        for(int j=1 ; j<=m ; j++){
            if(j==1) DP[i][j]=1;
            else {
                if(grid[i][j-1]=='_' && grid[i][j]=='#')
                    DP[i][j]=DP[i][j-1]+1;
                else if(grid[i][j-1]=='#' && grid[i][j]=='_')
                    DP[i][j]=DP[i][j-1]+1;
                else DP[i][j]=1;
            }
        }
    }
    memset(DPV,0,sizeof(DPV));
    for(int j=1 ; j<=m ; j++){
        for(int i=1 ; i<=n ; i++){
            if(i==1) DPV[i][j]=1;
            else {
                if(grid[i-1][j]=='_' && grid[i][j]=='#')
                    DPV[i][j]=DPV[i-1][j]+1;
                else if(grid[i-1][j]=='#' && grid[i][j]=='_')
                    DPV[i][j]=DPV[i-1][j]+1;
                else DPV[i][j]=1;
            }
        }
    }
    ll ans=0;
    for(int i=1 ; i<=n-31 ; i++){
        for(int j=1 ; j<=m-31 ; j++){
            if(check(i,j)) ans++;
        }
    }
    cout<<ans;
    return 0;
}
