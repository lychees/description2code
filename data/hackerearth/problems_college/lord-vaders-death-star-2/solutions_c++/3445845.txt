int main(void)
{
  int i,n,ans=0;

  cin >> n;
  long key[n];
  map<long,int> mp;
  for(i=0;i<n;i++)
  {
      scan(key[i]);
      if(mp.find(key[i])==mp.end())
         mp[key[i]]=1;
      else
         mp[key[i]]++;
  }

  int st=0;
  for(i=0;i<n;i++)
  {
      if(mp[key[i]]==1)
      {
          st=1;
          if(key[i]>ans)
            ans = key[i];
      }
  }
  if(st)
    cout << ans;
  else
    cout << "-1";
  return 0;
}
