#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll a,b;
ll giv(ll n)
{
		if(a%n == 0)
		{
				return b/n - a/n + 1;		
		}
		else return b/n - a/n;

}
int main()
{
		int k,t;
		cin>>k>>t;
		int i;
		ll karr[11];
		for(i=0;i<k;++i)
				cin>>karr[i];
		ll lim = 1<<k;
		while(t--)
		{
				cin>>a>>b;
				int j;
				int cnt;
				ll pro;
				ll tmp;
				ll ans[12];
				for(i=0;i<12;++i) ans[i]=0;
				//memset(ans,0,sizeof(ans));
				for(i=1;i<lim;++i)
				{
						cnt = 0;
						pro = 1;
						for(j=0;j<k;++j)
						{
								tmp = 1<<j;
								if(tmp&i)
								{
										cnt++;
										pro *= karr[j];
								}
						}
						ans[cnt]+=giv(pro);
						//cout<<"ans of pro="<<pro<<" is "<<giv(pro)<<endl;
				}
				ll p1,p2;
				p1=p2=0;
				for(i=2;i<=k;i+=2)
						p1+=ans[i];
				for(i=1;i<=k;i+=2)
						p2+=ans[i];
				cout<<p2-p1<<endl;
		}
}
