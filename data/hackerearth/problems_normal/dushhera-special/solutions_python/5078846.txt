n = int(raw_input())
for i in range(n):
    l = map(int, raw_input().split())
    p = l[0]
    K = l[1] + 1
    k = K
    lst = []
    for x in range(p):
        lst.append(k)
        k = k+K
    lst = map(str, lst)    
    print ' '.join(lst)
