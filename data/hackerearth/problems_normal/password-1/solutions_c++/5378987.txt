#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
   	int t;
   	cin>>t;
   	string *s=new string[t];
   	for(int i=0;i<t;i++)
   		cin>>s[i];
   	for(int i=0;i<t;i++)
   	{
   		string r=s[i];
   		reverse(r.begin(),r.end());
   		for(int j=i+1;j<t;j++)
   		{
   			if(r==s[j])
   			{
   				cout<<r.length()<<" "<<r[r.length()/2];
   			}
   		}
   	}
    return 0;
}
