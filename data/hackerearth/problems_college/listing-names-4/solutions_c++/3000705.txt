#include <iostream>
#include <vector>
#include <algorithm>
#include <stack>
#include <queue>
#include <cstring>
#include <string>

using namespace std;
#define MIN(a,b) (((a)<(b))?(a):(b))
#define MAX(a,b) (((a)>(b))?(a):(b))
#define FORI(i,a,b) for (int i=(int)a; i<=(int)b; i++)
#define ull unsigned long long

int main()
{

	int N;
	string name;
	vector<string> vv;
	cin >> N;
	while (N--)
	{
		cin >> name;
		vv.push_back(name);
	}
	sort(vv.begin(), vv.end());
	name = *vv.begin();
	string next;
	int c = 1;
	vector<string>::iterator itr = vv.begin();
	++itr;
	while (itr != vv.end())
	{
		next = *itr;
		if (name.compare(next) == 0)
		{
			c++;
			++itr;
		}
		else
		{
			cout << name << " " << c << endl;
			name = next;
			c = 0;
		}
	}
	cout << name << " " << c << endl;
	return 0;
}
