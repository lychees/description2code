    bool g = 0;
    FOR(i, 0, 4) {
      if (a[i] % 2 == 0 && a[(i + 1) % 4] % 2 == 0) {

        a[i] /= 2;
        a[(i + 1) % 4] /= 2;
        g = 1;
        break;

    }

    }
  }
