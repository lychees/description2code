from fractions import gcd
test_case = input()
numbers=[]
for i in range(0,test_case):
    case = input()
    numbers.append(case)
#print numbers
    

def myCounter(num):
    count = 0
    for i in range(1,num+1):
        res = gcd(i,num)
        if (res==1):
            count += 1
    #print count

    if count > 1:
        for j in range(2,count):
            if(count%j)==0:
                print "FALSE"
                break
        else:
            print "TRUE"
    else:
        print "FALSE"


for i in range(0,len(numbers)):
    myCounter(numbers[i])
