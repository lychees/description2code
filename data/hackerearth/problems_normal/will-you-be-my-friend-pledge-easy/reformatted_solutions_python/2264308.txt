A = int(raw_input())
N = int(raw_input())
rawList = raw_input()
arr = rawList.split(' ')

for i in range(0, N):
    arr[i] = int(arr[i])

count = 0
# print arr


def hcf_of_num(num1, num2):
    if num1 > num2:
        small = num2
    else:
        small = num1

    for i in range(1, small + 1):
        if num1 % i == 0 and num2 % i == 0:
            hcf = i
    return hcf


for i in range(0, N):
    if hcf_of_num(A, arr[i]) > 1:
        count += 1

print count
