#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#define ll long long
#include<math.h>
#define N 10000
#define MAXN ( 32000*32000 )
int A[ N ] , B[ N ];
int n , k ;
bool s[ N ];
using namespace std;
 
bool solve( int mi ){
	for( int i=0 ; i<n ; ++i ){
		int a = A[ i ] , b = mi / A[ i ] ;
		for(int i=0;i<sizeof( s );i++)
		s[i]=0;
		for( int j=0 ; j<n ; ++j ){
			if( A[ j ] <=a && B[ j ] <=b ) s[ j ] =1;
		}
		int maxi = 0 ,ct =0;
		for( int j=0 ; j<n ; ++j ){
			if( s[ j ] ) ct =0;
			else{
				ct++;
				maxi = max( maxi , ct );
			}
		}
		if( maxi <= k ) return 1;
	}
	return 0;
}
 
int main(){
	scanf( "%d%d" , &n, &k );
	k--;
	for( int i = 0 ; i<n ; ++i )
		scanf( "%d%d" ,&A[ i ] ,&B[ i ]);
	ll lo=0 , hi = MAXN;
	while( hi -lo >1 ){
		ll m = ( hi+lo )/2;
		if( solve( m ) ) hi = m;
		else lo = m;
	}
	printf("%d\n" , hi );
}
