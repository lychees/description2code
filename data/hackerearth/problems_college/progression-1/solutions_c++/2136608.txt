#include <iostream>
using namespace std;

int main()
{
    int tc,n;
    cin>>tc;
    for(int i=0;i<tc;i++)
   	{
   		cin>>n;
   		if(n<44)
   			cout<<"INCORRECT"<<"\n";
   		else
    	{
   			if(((n%45)%3)==0)
   				cout<<"CORRECT"<<"\n";
   			else
   				cout<<"INCORRECT"<<"\n";
   		}
   		
   	} 	
    return 0;
}
