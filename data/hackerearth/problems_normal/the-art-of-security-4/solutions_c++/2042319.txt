#include<bits/stdc++.h>
using namespace std;
int power[10];
int n,x;
long long int arr[2<<9][4001];
int rec(int val, int first, int second){
	if(val==x){
		return 1;
	} 
	if(arr[val][first]!=-1) return arr[val][first];
	else{
		long long int ans=0;
		for(int i=0; i<n; i++){
			int k=(1<<i);
			if((val&k)==0){
				if(second>=power[i]+first) ans=ans+rec(val+k,first+power[i],second);
				ans=ans+rec(val+k,first,second+power[i]);
				
			}
		}
		return arr[val][first]=ans;
	}
}
int main(){
	cin>>n;
	memset(arr,-1,sizeof arr);
	x=1<<(n); x--;
	long long int tot=0;
	for(int i=0; i<n; i++) cin>>power[i], tot+=power[i];
	long long int y=rec(0,0,0);
	cout<<y<<" "<<tot<<endl;
	if(y>tot) cout<<"We will win!";
	else cout<<"Got no way out!";
}
