ll step( int n , initializer_list<int> p ) {
	vector<ll> dp( n + 1 , 0 );
	dp[ 0 ] = 1;
	for (int i = 1; i <= n; ++i) {
		for (int st : p) {
			if (i - st >= 0)
				dp[ i ] += dp[ i - st ];
		}
	}
	return dp[ n ];
}
