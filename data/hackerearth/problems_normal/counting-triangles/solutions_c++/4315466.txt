#include <bits/stdc++.h>

using namespace std;
#define MAXN 100005
int n;
long long niz[3];
map<long long, int> cnt;

int main()
{
    cin >> n;
    for (int i = 0; i < n; i++){
        cin >> niz[0] >> niz[1] >> niz[2];
        sort(niz, niz + 3);
        long long val = niz[0] + niz[1] * 13 + niz[2] * 169;
        cnt[val]++;
    }
    int res = 0;
    for (auto it : cnt){
        if (it.second == 1)
            res++;
    }
    cout << res << "\n";
}
