#include <iostream>
#include <cstring>
#include<cmath>
using namespace std;

int primes[1000001];
long long int sum[1000001];
void seive(){
	memset(primes,0,sizeof(primes));
	memset(sum,0,sizeof(sum));
	primes[0] = primes[1] = 1;
	for(int i = 2; i <= sqrt(1000000); i++){
		if(primes[i]==0)
		{
			for(int j = i*i; j <= 1000000; j = j+i){
				primes[j] = 1;
			}
		}

	}
	for(int i = 0; i <= 1000000;i++){
		if(primes[i] == 0){
			sum[i] = (sum[i-1]+1);
		}
		else{
			sum[i] = sum[i-1];
		}
	}
}
int main()
{
	int t;
	cin >>t;
	seive();
	while(t--){
		int l,r,s;
		cin >>l>>r;
		s=sum[r]-sum[l-1];
	//	cout <<s<<endl;
		printf("%f\n",(float)s/(r-l+1)); 
	}
   // cout << "Hello World!" << endl;
    return 0;
}
