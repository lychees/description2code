#include <iostream>
#include <stdio.h>
#include <vector>
#include <set>
#include <utility>

using namespace std;

string s[100];
bool B[26];
int val;
vector<int> v[26];
struct node
{
	int a,b;
	node(int _a,int _b)
	{
		a=_a,b=_b;
	}
	bool operator < (const node &Z) const
	{
		if(a==Z.a) return b<Z.b;
		return a<Z.a;
	}
};
set<node>st;

void dfs(int pos,int n)
{
	val=max(val,n);
	for(int i=0;i<v[pos].size();i++) dfs(v[pos][i],n+1);
}
int main()
{
	#ifndef ONLINE_JUDGE
		freopen("I.in","r",stdin);
	#endif
		
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>s[i];
		for(int j=0;s[i][j];j++) B[s[i][j]-'a']=1;
	}
	for(int i=0;i<n-1;i++)
	{
		for(int j=0;s[i][j] and s[i+1][j];j++) if(s[i][j]!=s[i+1][j])
		{
			v[s[i+1][j]-'a'].push_back(s[i][j]-'a');
			break;
		}
	}
	for(int i=0;i<26;i++)
	{
		if(!B[i]) continue;
		val=0;
		dfs(i,0);
		st.insert(node(val,i));
	}
	int last=-1;
	for(set<node>::iterator it = st.begin();it!=st.end();it++)
	{
		node pr= *it;
		// printf("%d %c\n", pr.a,'a'+pr.b);
		if(last!=pr.a and it!=st.begin()) putchar('\n');
		printf("%c",'a'+pr.b );
		last=pr.a;
	}
	puts("");
	return 0;
}
