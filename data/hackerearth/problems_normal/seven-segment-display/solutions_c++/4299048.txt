#include <bits/stdc++.h>
#define SI set<int> 
#define VI vector<int> 
#define ull unsigned long long 
#define inint(a) scanf("%d",&a)
#define input(a) cin>>a
#define ll long long 
#define print(a) cout<<a
#define printn(a) cout<<a<<"\n"
#define f first
#define s second
#define len(s) s.length()
#define range(a,b,n) for(int i=a;i<b;i+=n)
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    inint(t);
    while(t--){
    	string s;
    	input(s);
    	ull ans=0;
    	for(int i=0;i<len(s);i++){
    		switch(s[i]){
    			case '0':ans+=6;break;
    			case '1':ans+=2;break;
    			case '3':ans+=5;break;
    			case '4':ans+=4;break;
    			case '5':ans+=5;break;
    			case '6':ans+=6;break;
    			case '7':ans+=3;break;
    			case '8':ans+=7;break;
    			case '9':ans+=6;break;
    			case '2':ans+=5;break;
    		}
    	}
    	cout<<ans<<"\n";
    }
    return 0;
}
