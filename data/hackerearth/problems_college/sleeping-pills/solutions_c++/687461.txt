#include <iostream>
#include <cstdio>
#include <cmath>
#define MOD 1000000007
using namespace std;
typedef unsigned long long int ull;
ull power(ull a,ull b)
{
    if(b==0)
	return 1;
	else
	{
		ull tmp=power(a,b/2)%MOD;
		if(b%2==0)
		return (tmp*tmp)%MOD;
		else
		return ((a%MOD)*((tmp*tmp)%MOD))%MOD;
	}
}

ull modInv(ull a)
{
	return power(a,MOD-2);
}

ull ncr(ull n,ull k)
{
	ull num=1;
	for(ull i=0;i<k;i++)
	num=(num*(n-i))%MOD;
	
	ull den=1;
	for(ull i=1;i<=k;i++)
	den=(i*den)%MOD;
	
	return (num*modInv(den))%MOD;
}


int main() 
{
	ull a,b,tmp;
	int t;
    ull m,n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%llu %llu",&n,&m);
		ull total=0;
		for (int i = 0; i < n; ++i)
		{
			ull x;
			scanf("%llu",&x);
			total+=x;
		}

		if(total>m)
		{
			cout<<"0"<<endl;
		}
		else
		{
			ull binCoef=ncr(m-total+n-1,n-1);
			cout<<binCoef<<endl;
		}
	}
	return 0;
}
