#include <iostream>
#include <vector>
#include <cmath>
#include <climits>

using namespace std;
int main()
{
    int t;
   // long long mintime=LONG_LONG_MAX;
    //long long n,k,m;
    int index=-1;
    cin>>t;
    while(t--)
    {
    	long long mintime=LONG_LONG_MAX;
    	long long n,k,m;
    	index=-1;
        cin>>n>>k>>m;
        vector<long long > vec;
        for(int i=0 ; i < n ;i++)
        {
            int abc;
            cin>>abc;
            vec.push_back(abc);
        }
		long long i;
        for( i=0; i < vec.size();i++)
        {
            long long sum=0;
            long long j;
            for(j=0; j < vec[i];j++)
            {
                long long  val;
                cin>>val;
                sum+=val;
            }

            sum=(k*vec[i])+sum+(m)*(vec[i]-1);
            if (mintime> sum)
            {
                mintime=sum;
                index=i+1; 
            }
            
        }        
        cout<<index<<" "<<mintime<<endl;
        
    }
    return 0;
      
  }
