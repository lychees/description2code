#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    int t,n;
    unsigned long long k=0;
    scanf("%d",&t);
    while(t--)
    {
    	k=0;
    	scanf("%d",&n);
    	int hash[n];
    	int arr1[n];
    	int arr2[n];
    	
    	for(int i=0;i<n;i++)
    	{
    	scanf("%d",&arr1[i]);
    	arr1[i]--;
    	hash[arr1[i]]=i;
    	}
    	
    	for(int i=0;i<n;i++)
    	{scanf("%d",&arr2[i]);
    	arr2[i]--;
    	}
    	
    	for(int i=0;i<n;i++)
    	{
    		k += hash[arr2[i]] - i;
    		for(int j=0;j<n;j++)
    		{
    			if(hash[j]<hash[arr2[i]]&&hash[j]>=i)
    			hash[j]++;
    		}
    		hash[arr2[i]]=i;
    	}
    	printf("%llu\n",k);
    }
    return 0;
}
