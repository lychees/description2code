#include <bits/stdc++.h>
#define ll long long unsigned int
#define pb push_back
#define Max 1000000
using namespace std;
typedef vector <ll> V;
typedef pair <int,int> P;
ll A[Max];	V v(1000004);
void sieve()
{
	for(int i=3;i<=Max;i++)
	{
		A[i]=0;	v[i]=0;
	}
	int s=sqrt(Max);
	for(int i=2;i<=Max;i+=2)
		A[i]+=1;
	v[2]=500000;
	for(int i=3;i<=s;i+=2)
	{
		int ct=1;
		if(!A[i])
		{
			for(int j=i*i;j<=Max;j+=2*i)
			{
				if(j<=1000000 && !A[j])
				{	ct+=1;	A[j]+=1;	}
			}
			v[i]=ct;
		}
	}
}
 
int main()
{
	sieve();	ll t,n;
	ios::sync_with_stdio(false);
	cin>>t;
	while(t--)
	{
		cin>>n;
		if(n>1000)
			cout<<"1"<<'\n';
		else
			cout<<v[n]<<'\n';
	}
	return 0;
}
