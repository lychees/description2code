#include<bits/stdc++.h>
using namespace std;
#define LL long int


int main()
{
	int t;
    LL  element, size, result;
    ios_base::sync_with_stdio(false);
    
    cin >> t;
   // assert(t>=1 && t<=10);
    
    for(int i=1; i<=t; i++){
    	cin >> size;
    //	assert(size>=1 && size<=100000);
   		
   		cin >> element;
   		result = element%5;
    
   		for(int j=1; j<size; j++){
    		cin >> element;
    	//	assert(element>=1 && element<=1000000000L);
    		element=element%5;
    		result=result^element;
    	} 
    	
    	if(result > 0){
    		cout << "Aniruddha" << endl;
    	} else{
    		cout << "Andrew" << endl;
    	}
    }
    return 0;
}
