#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(0);cin.tie(0);
    int t,n,pos;cin>>t;
    while(t--){
    	cin>>n;
    	vector<string>s(n);
    	for(int i=0;i<n;i++){
    		cin>>s[i];
    	}
    	sort(s.begin(),s.end());
    	string ans="YES";
    	for(int i=0;i<n-1;i++){
			if(s[i].size()<=s[i+1].size()){
				pos=0;
				while(pos<s[i].size()){
					if(s[i][pos]!=s[i+1][pos])break;
					pos++;
				}
				if(pos==s[i].size())ans="NO";
			}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
