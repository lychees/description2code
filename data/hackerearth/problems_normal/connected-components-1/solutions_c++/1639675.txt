#include <bits/stdc++.h>
#define M 1000000007
#define lli long long

using namespace std;

lli fact[300005];

lli fast_pow(lli a, lli b)
{
	lli res = 1;
	while ( b > 0 ) {
		if ( b&1 ) res = (res*a)%M;
		a = (a*a)%M;
		b >>= 1;
	}
	return res;
}

lli C(lli n, lli k)
{
	if ( k > n ) return 0;
	if ( k == n ) return 1;
	lli num = fact[n];
	lli den = (fact[k]*fact[n-k])%M;
	num = (num*fast_pow(den,M-2))%M;
	return num;
}

lli cal(lli n, lli k)
{
	return C(n+k-1,k-1);
}

int main()
{
	fact[0] = 1;
	for ( lli i = 1; i <= 300000; i++ ) fact[i] = (fact[i-1]*i)%M;	
	int t,n,m,c;
	cin >> t;
	while ( t-- ) {
		cin >> n >> m >> c;
		lli ans = cal(n-m-c,c);
		ans = (ans*cal(m-c+1,c+1))%M;
		cout << ans << endl;
	}
	return 0;
}
