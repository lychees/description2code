#include <iostream>
using namespace std;

int main()
{   long t,m=0;
    cin>>t;
    while(m<t){
    	long n;
    	cin>>n;
    	long sum=1;
    	for(long i=1;i<=n;i++){
    		for(long k=1;k<i;k++){
    			for(long b=k;b<=i;b++)
    		     {if(k*b>i){
    		     	break;
    		     }
    		     	if(i==(k*b))
    		     	sum++;
    		     	
    		     }
    		     
    		}}
    		
    	
    	cout<<sum<<"\n";
    	m++;
    }
    return 0;
}
