#include<stdio.h>
#include<iostream>
#include<algorithm>
#define ll long long
using namespace std;
ll a[10004][3];

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<3;j++)
			{
				scanf("%lld",&a[i][j]);
			}
		}
		int flag=0;
		if(a[0][2]%2)
		{
		
			for(ll i=1;i<n;i++)
			{
				if((abs(a[0][0]-a[i][0])+abs(a[0][1]-a[i][1]))%2 && a[i][2]%2!=0)
				{
					flag=1;
					break;
				}
				if((abs(a[0][0]-a[i][0])+abs(a[0][1]-a[i][1]))%2==0 && a[i][2]%2==0)
				{
					flag=1;
					break;
				}
			}
		}
		else
		{
			for(ll i=1;i<n;i++)
			{
				if((abs(a[0][0]-a[i][0])+abs(a[0][1]-a[i][1]))%2 && a[i][2]%2==0)
				{
					flag=1;
					break;
				}
				if((abs(a[0][0]-a[i][0])+abs(a[0][1]-a[i][1]))%2==0 && a[i][2]%2!=0)
				{
					flag=1;
					break;
				}
			}
		}
		if(flag==0)
		printf("Yes\n");
		else
		printf("No\n");
	}
	return 0;
}
