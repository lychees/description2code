#include<iostream>
#include<string.h>
#include<stdio.h>
#include<cmath>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        char a[1000001];
        cin>>a;
        int f=0,l=strlen(a);l--;
        for(int i=0;i<strlen(a)-1;i++)
        {
            //cout<<abs('a[i]'-'a[i+1]')<<endl;
            if(abs(a[i]-a[i+1])!=abs(a[l-i]-a[l-i-1]))
            {
                f=1;break;
            }
            i++;
        }
        if(f==1)
            cout<<"Not Funny\n";
        else
            cout<<"Funny\n";
    }
    return 0;
}
