#include <iostream>
using namespace std;

int main()
{
    int dp[102]={0};
    dp[0]=1;
    int s[3]={1,3,5};
    for(int i=0; i<3; i++){
    	for(int j=s[i];j<102;j++)
    	dp[j]+=dp[j-s[i]];
    	
    }
    int T,n;
    cin>>T;
    while(T--){
    	cin>>n;
    	cout<<dp[n]<<endl;
    }
    return 0;
}
