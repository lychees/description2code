#include <bits/stdc++.h>
using namespace std;

bool cmp(char a , char b)
{
	return a<b;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	
    	if(s.size()<=2)
    	{
    		cout<<"Not possible!"<<endl;
    		continue;
    	}
    	
    	sort(s.begin(),s.end());
    	next_permutation(s.begin(),s.end());
    	next_permutation(s.begin(),s.end());
    	cout<<s<<" ";
    	sort(s.begin(),s.end(),cmp);
    	prev_permutation(s.begin(),s.end());
    	prev_permutation(s.begin(),s.end());
    	prev_permutation(s.begin(),s.end());
    	cout<<s<<endl;
    	
    }
