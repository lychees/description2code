#include <iostream>
using namespace std;

int main()
{
    int i,j,n,m,k,a[100][100],x,y,s;
    cin>>n>>m;
    for(i=1;i<=n;i++)
    {
    	for(j=1;j<=m;j++)
    	{
    		cin>>a[i][j];
    	}
    }
    k=2;
    while(k<=(m+n))
    {
    	
    for(i=1;i<=n;i++)
    {
    	for(j=1;j<=m;j++)
    	{
    		if(i+j==k)cout<<a[i][j]<<" ";
    	}
    }
    k+=1;
    cout<<endl;
    }
    return 0;
}
