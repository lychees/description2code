ll power(ll a,ll b)
{
	if(!b)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2);
	temp=(temp*temp);
	if(b&1)
		temp=(temp*a);
	return temp;
}
class graph
{
	vi * adj;
	int time;
	public:
		graph( int v)
		{
			adj = new vi[v];
			time = 0;
		}
		void add_edge(int u, int v)
		{
			adj[u].pb( v);
			adj[v].pb( u);
		}
};

class comp
{
	public:
		bool operator() (const pp& a,const pp& b)
		{
			return( a.first > b.first);
		}
};
int main()
{
	ios_base::sync_with_stdio(false);
	int i, n, q, k;
	cin >> n >> q;
	vector< pp > v(n);
	for( i=0;i<n;i++ )
	{
		cin >> v[i].second;
	}
	for( i=0;i<n;i++ )
	{
		cin >> v[i].first;
	}
	sort( v.begin(), v.end(), comp());
	vl st(n);
	ll sum = 0;
	for( i=0;i<n;i++ )
	{
		sum += v[i].second;
		st[i] = sum;
	}
	while( q--)
	{
		cin >> k;
		cout << st[k-1] << "\n";
	}
    return 0;
}
