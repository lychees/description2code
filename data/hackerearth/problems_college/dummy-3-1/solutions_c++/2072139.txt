#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,flag=0;
    	cin>>n;
    	int a[n+1],b[n+1];
    	for(int i=1;i<=n;i++)
    	{
    		cin>>a[i];
    	}
    	for(int i=1;i<=n;i++)
    	{
    		b[i]=a[a[i]];
    	}
    	for(int i=1;i<=n;i++)
    	{
    		if(b[i]!=i)
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==0)
    	{
    		cout<<"inverse"<<endl;
    	}
    	else
    	{
    		cout<<"not inverse"<<endl;
    	}
    	
    }
}
