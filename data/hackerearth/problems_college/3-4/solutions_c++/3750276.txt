#include <iostream>
#include<algorithm>

#include<cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
       int n,k,flag=0;
       cin>>n>>k;
       int a[k];
       for(int i=0;i<k;i++)
       {
           cin>>a[i];
       }
       int b[10002]={0};
       for(int i=0;i<k;i++)
       {
           b[a[i]]++;
       }
       for(int i=0;i<10002;i++)
       {
           if(b[i]>0)
           {
               if(b[i]%2!=0)
               {
                   cout<<i<<endl;
                   flag=1;
               }
           }
           if(flag==1)
            break;
       }
    }
    return 0;
}
