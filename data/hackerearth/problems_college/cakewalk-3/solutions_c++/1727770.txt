#include <iostream>
#include<string.h>
using namespace std;
int main()
{
    int n,j;
    cin>>n;
    char s[1000],s1[1000];
    cin>>s;
    for(int i=0;i<n-1;i++){
    	int m;
    	cin>>m;
    	cin>>s1;
    	for(j=0;j<m;j++);
    	s[j]='\0';
    	strcat(s,s1);
    }
    cout<<s<<endl;
    return 0;
}
