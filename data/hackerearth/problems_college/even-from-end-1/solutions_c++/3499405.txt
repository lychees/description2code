#include<stdio.h>

int main(){
	int n, i, c;
	while(scanf("%d", &n) == 1){
		int a[n]; c = 0;
		for(i = 0; i < n; scanf("%d", &a[i++]));
		for(i = n - 1; i >= 0; i--)	if(!(a[i] % 2))	c++;
		if(c){
			for(i = n - 1; i >= 0; i--)	if(!(a[i] % 2))	printf("%d ", a[i]);
		}
			else	printf("None to display");
		printf("\n");
	}
	return 0;
}
