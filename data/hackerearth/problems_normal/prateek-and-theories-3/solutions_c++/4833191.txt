#include<bits/stdc++.h>
using namespace std;
int main(){
    int tc;cin>>tc;
    while(tc--){
        int n;cin>>n;
        map<int,int>mpp;
        for(int i=1;i<=n;++i){
            int st,en;cin>>st>>en;
            mpp[st]+=1;
            mpp[en]-=1;
        }
        int X=0;
        int ans=-9;
        for(auto it=mpp.begin();it!=mpp.end();++it){
            X+=(*it).second;
            ans=max(ans,X);
        }
        cout<<ans<<endl;
    }
    return 0;
}
