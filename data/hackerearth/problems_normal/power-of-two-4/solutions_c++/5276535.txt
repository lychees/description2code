#include <bits/stdc++.h>
using namespace std;
#define int long long
bool IsPowerOfTwo(int num)
{ return (num&&!(num&(num-1))); }
main()
{
	int total;	
   	for(int i=0;i<32;i++)
    total+=(1<<i);
   int t;
   cin>>t;
   while(t--)
   {
   	int n;
   	cin>>n;
   	
   	
   	int a[n];
   	for(int i=0;i<n;i++)
   	{
   		cin>>a[i];
   	}
   	
   	if(n==1)
   	{
   	//	cout<<"a";
   		if(IsPowerOfTwo(a[0]))
   		cout<<"YES"<<endl;
   		else
   		cout<<"NO"<<endl;
   		continue;
   		
   	}
   	bool f=false;
  // 	int total;
   
    
  	for(int i=0;i<32;i++)
  	{
  		int ans = total;
  		
  		for(int j=0;j<n;j++)
  		{
  			if(a[j]&(1<<i))
  				ans=ans&a[j];
  		}
  		
  		if(IsPowerOfTwo(ans)){
  		f=true;
  		break;}
  		
  	}
  	
  	if(f)
  	cout<<"YES"<<endl;
  	else
  	
  cout<<"NO"<<endl;
   	
   }
    return 0;
}
