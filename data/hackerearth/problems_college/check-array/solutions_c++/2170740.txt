#include<bits/stdc++.h>
using namespace std;

long long int a[100005];
long long int b[100005];

int main() {
    long long int i,j,k,t,n;
    cin >> t;
    while(t--) {
    	scanf("%lld",&n);
 
    	for(i=1;i<=n;i++) {
    		scanf("%lld",&a[i]);
    	}
    	b[0]=0;

    	for(i=1;i<=n;i++) {
           b[i]=b[i-1]+a[i];
    	}
        
     
       
    	for(i=2;i<n;i++) {
    		if(b[i-1]==b[n]-b[i]) {
    			printf("YES\n");
    			break;
    		}

    	}

    	if(i==n)
    		printf("NO\n");

    }
	return 0;
}
