typedef long long ll;

using namespace std;
ll mod = 1000000007;

inline ll mpow(ll b, ll ex){
    if (b==1)return 1;
    ll r = 1;
    while (ex ){
        if (ex&1)r=(r * b)%mod;
        ex = ex >> 1;
        b = (b * b)%mod;}
    return r;
}

ll t,a[100],n;

ll comp(vector<ll> v){
	int l = v.size();
	if(l&1)return 0;

	ll res=1;
	for (int i = 0; i < l/2; ++i)
	{
		res=res*mpow(v[i],mod-2)%mod;
	}

	for (int i = l/2; i < l; ++i)
	{
		res=res*v[i]%mod;
	}

	return res;
}


int main() {
        
    std::ios_base::sync_with_stdio(0);

    cin >> t;

    while(t--){
    	cin >> n;

    	for (int i = 0; i < n; ++i)
    	{
    		cin >> a[i];
    	}

    	sort(a,a+n);

    	ll ans =1;

    	for (int i = 0; i < (1<<n); ++i)
    	{
    		std::vector<ll> v;
    		for (int j = 0; j < n; ++j)
    		{
    			if (i&(1<<j))
    			{
    				v.push_back(a[j]);
    			}
    		}
    		ans = max(ans,comp(v));
    	}

    	cout << ans << endl;

    }



    return 0;
}
