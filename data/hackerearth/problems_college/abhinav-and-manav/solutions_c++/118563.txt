int tens(long int n)
{
	int i,count=0;
	while(n>=10)
	{
		count++;
		n=n/10;
	}
	return count;
}
long long int pow(int i,int j)
{
	long long int temp=1;
	int x;
	for(x=j;x>0;x--)
	{
		temp=temp*i;
	}
	return temp;
}
