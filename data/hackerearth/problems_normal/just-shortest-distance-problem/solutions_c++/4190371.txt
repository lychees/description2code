const int MAX = 1e5 + 10;
using namespace std;
vector<int>v[MAX];
int dis[MAX];
void update_bitch(int x)
{
	for (int i = 0; i < v[x].size(); i++)
	{
		int cur = v[x][i];
		if (dis[cur] > dis[x] + 1)
		{
			dis[cur] = dis[x] + 1;
			update_bitch(cur);
		}
	}
}
