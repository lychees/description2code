#include <stdio.h>
#include<algorithm>
using namespace std;
int main()
{
    int n,i,flag=0,j,k;
    long long int m;
    scanf("%d%lld",&n,&m);
    long long int a[n];
    for(i=0;i<n;i++)
    	scanf("%lld",&a[i]);
    sort(a, a+n);
    for(i=0;i<n;i++){
    	for(j=i+1;j<n;j++){
    		for(k=j+1;k<n;k++){
    			if(a[i]+a[j]+a[k]==m){
    				flag=1;
    				break;
    			}
    		}
    		if(flag==1)
    				break;
    	}
    	if(flag==1)
    				break;
    }
    if(flag==1)
    	printf("YES\n");
    else
    printf("NO\n");
    return 0;
}
