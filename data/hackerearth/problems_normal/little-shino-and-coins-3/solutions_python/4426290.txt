K=int(raw_input().strip())
arr=list(raw_input())
count=0
c=[0 for i in xrange(26)]
for i in xrange(0,len(arr)):
	x,countK=c[:],0
	for j in xrange(i,len(arr)):
		if x[ord(arr[j])-ord('a')]==0:
			countK+=1
			x[ord(arr[j])-ord('a')]=1
		if countK==K:
			k=j
			while k<len(arr) and x[ord(arr[k])-ord('a')]==1:
				count+=1
				k+=1
			break
print count
