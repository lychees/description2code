T = int(raw_input())

for _ in range(T):
    (N, K) = map(int, str(raw_input()).split())
    A = map(int, str(raw_input()).split())

    min_cost = 1000000
    for i in range(min(A), max(A)+1):
        s = sum(sorted([5*(num-i) if i < num else 3*(i-num) for num in A])[:K])
        if s < min_cost:
            min_cost = s

    print min_cost
