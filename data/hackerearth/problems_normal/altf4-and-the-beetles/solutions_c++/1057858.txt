int main() {
	int T; cin >> T;
	while (T--) {
		int N; ll A, B, C;
		cin >> N >> A >> B >> C;
		vector<i_i> v;
		for (int i = 0; i < N; i++) {
			int P, Q; cin >> P >> Q;
			v.push_back(i_i(P, -1));
			v.push_back(i_i(Q, 1));
		}
		sort(v.begin(), v.end());
		int a = N, b = 0, c = 0;
		ll maxi = 0;
		for (int i = 0; i < N * 2; i++) {
			maxi = max(maxi, a * A + b * B + c * C);
			int t = v[i].first, x = v[i].second;
			if (x == -1) { a--; b++; }
			else if (x == 1) { b--; c++; }
			maxi = max(maxi, a * A + b * B + c * C);
		}
		cout << maxi << endl;
	}
}
