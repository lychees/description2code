#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int a[1000001];

inline int gcd(int a, int b)
{
	if(b == 0)
		return a;
	return gcd(b, a%b);
}

int main()
{
    int t, n, g;
    scanf("%d", &t);
    while(t--)
    {
    	scanf("%d", &n);
    	for(int i = 0; i < n; i++)
    		scanf("%d", &a[i]);
    	g = a[0];
    	for(int i = 1; i < n; i++)
    		g = gcd(g, a[i]);
    	printf("%lld\n", (ll)g*n);
    }
    return 0;
}
