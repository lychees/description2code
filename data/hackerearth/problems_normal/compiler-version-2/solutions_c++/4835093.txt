#include <iostream>
#include<string.h>
#include<stdlib.h>

using namespace std;

int main()
{
	char s[100];
    while(gets(s)!=NULL){
    	int length=strlen(s);
    	for(int i=0;i<length;i++){
    		int j;
    		if(s[i]=='/' && s[i+1]=='/'){
    			break;
    		}
    		if(s[i]=='-' && s[i+1]=='>'){
    			s[i]='.';
    			for(j=i+1;j<length;j++){
    				s[j]=s[j+1];
    			}
    			s[++j]='\0';
    		}
    	}
    	puts(s);
    }
    return 0;
}
