#include <iostream>
using namespace std;

int main()
{
    int n , arr[100] , s = 0;

    cin >> n;

    for(int i = 0 ; i < n ; i++)
    {
        cin >> arr[i];
    }

    for(int i = 0 ; i < n ; i++)
    {
        while(arr[i] != 1)
        {
            if(arr[i] % 3 == 0)
            {
                arr[i] = arr[i] / 3;
                s = s + 1;
            }

            else if(arr[i] % 2 == 0)
            {
                arr[i] = arr[i] / 2;
                s = s + 1;
            }

            else if((arr[i] % 3 != 0) && (arr[i] % 2 != 0))
            {
                arr[i] = arr[i] - 1;
                s = s + 1;
            }
        }

        cout << s << endl;
        s = 0;
    }

    return 0;
}
