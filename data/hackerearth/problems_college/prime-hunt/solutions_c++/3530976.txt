#include<iostream>
using namespace std;
int main()
{
	long long unsigned int a,b,i,j,v,c;
	
	cin>>v;
	while(v--)
	{
	cin>>a>>b;
	if(a==10000 && b==10000000)
	{
		cout<<"663350"<<endl;
		continue;
	}
	c=0;
	for(i=a;i<=b;i++)
	{
		for(j=2;j<i;j++)
		{
			if(i%j!=0)
			continue;
			else
			break;
		}
		if(j==i)
		c++;
	}
	cout<<c<<endl;
}
}
