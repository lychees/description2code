#include<iostream>
#include<string.h>
using namespace std;
int main()
{
  int t,i;
  cin>>t;
  while(t--)
  {
       string n1;
       string n2;
       cin>>n1;
       cin>>n2;
       int l1,l2;
       l1=n1.length();
       l2=n2.length();
       if(l1==l2)
       {
     for(i=0;i<l1;i++)
     {
         if(n1[i]==n2[i])
            cout<<"0";
         else
            cout<<"1";
     }
       }
       cout<<"\n";
  }
}
