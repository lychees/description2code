#include<iostream>
using namespace std;
 
int main() {
 
	int t;
	cin>>t;
	while(t--)
	{
 
		int flag = 0;
		int n;
		cin>>n;
		int a = n/3, b = 0;
		while(a>=0)
		{
 
			int sum = a*3 + b*5;
			if(sum==n)
			{flag = 1; break;}
			else if(sum<n)b++;
			else a--;
		}
		if(flag)
		{
 
			for(int i=1; i<=a*3; i++)cout<<"5";
			for(int i=1; i<=b*5; i++)cout<<"3";
			cout<<endl;
		}
		else cout<<"-1\n";
	}
	return 0;
}
