t = int(raw_input())
# inital amnt
while (t > 0):
    # orig n & m
    orign, origm = list(map(int, raw_input().split()))
    # n*m list
    itmlist = [None]*orign
    for n in range(0, orign):
        # shop prices
        shoplist = [None]*origm
        for m in range(0, origm):

            x = x - x*i

            x = x - x*k
            shoplist[m] = x

        itmlist[n] = shoplist[:]
