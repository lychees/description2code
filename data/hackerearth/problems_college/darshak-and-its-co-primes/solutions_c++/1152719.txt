#include<iostream>
#include<cstring>
using namespace std;

unsigned long long int gcd(unsigned long long int a,unsigned long long int b){
if(a%b==0)
return b;
return gcd(b,a%b);
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
unsigned long long int a,b,c;
cin>>a>>b;
c=gcd(a,b);
if(c==1)
cout<<"Is a Co-Prime"<<endl;
else
cout<<"Not a Co-Prime"<<endl;
}
return 0;
}
