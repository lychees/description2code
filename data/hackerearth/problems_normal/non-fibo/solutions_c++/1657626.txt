#include <iostream>
#define lli long long int
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--)
    {
        lli n;
    	cin>>n;
    	lli a=1,b=1;
    	lli count=0;
    	while(count<n)
    	{
    		if(a!=b)count+=b-a-1;
    		long long int c=a+b;
    		a=b;
    		b=c;
    	}
    	lli d=a-1;
    	d-=count-n;
    	cout<<d<<endl;
    }
    return 0;
}
