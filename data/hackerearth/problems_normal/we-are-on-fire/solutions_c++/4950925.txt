#include<iostream>

using namespace std;
//stack <int> B;
int n,m,count,A[1005][1005];
long int q;
void func(int x,int y){
		//	cout<<count<<endl;
			A[x][y]=0;
			count--;		
			//B.push(A[x][y]);
			if(x!=1 && A[x-1][y]==1){
				func(x-1,y);
			}
			if(x!=n && A[x+1][y]==1){
				func(x+1,y);
			}
			if(y!=1&&A[x][y-1]==1){
				func(x,y-1);
			}
			if(y!=m&&A[x][y+1]==1){
				func(x,y+1);
			}	
}
int main(){
	int x,y;
	//int A[n][m];
	scanf("%d %d %ld",&n,&m,&q);
	//cin>>n>>m>>q;
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++){
			scanf("%d",&A[i][j]);
			if(A[i][j]==1){
				count++;
			}		
		}
	}
	for(long int i=0;i<q;i++){
		
	//	cin>>x>>y;
		scanf("%d %d",&x,&y);
		if(A[x][y]==1){
		func(x,y);
		}
		printf("%d\n",count);
	}
	
	
	
	return 0;
}
