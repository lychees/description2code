#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int tc;
	cin >> tc;
	while(tc--)
	{
		long int n;
		cin >> n;
		
		long int k;
		cin >> k;

		long int arr[n];
		for(long int i=0;i<n;i++)
			cin >> arr[i];
		
		long int index;
		for(long int i=0;i<n && k>0;i++)
		{
			index=i;
			long int mini=arr[i];
			for(long int j=i+1;j<=i+k && j<n;j++)
			{
				if(arr[j] < mini)
				{
					mini = arr[j];
					index = j;	
				}	
			}
			k = k - (index - i);
			long int alpha = arr[index];
			for(long int k1=index;k1 > i ;k1--)
			{
				arr[k1] = arr[k1-1];
			}
			arr[i] = alpha; 
		}
		for(long int i=0;i<n;i++)
			cout << arr[i] << " ";
		cout << endl;
	}
	return 0;
}
