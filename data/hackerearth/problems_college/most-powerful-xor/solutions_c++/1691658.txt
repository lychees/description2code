#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long t;
    cin>>t;
    while(t--)
    {
    	long long a,b,i,j,x,max=0;
    	cin>>a>>b;
    	for(i=a;i<=b;i++)
    	{
    		for(j=i;j<=b;j++)
    		{
    			x=i^j;
    			if(x>max)
    			max=x;
    		}
    	}
    	cout<<max<<endl;
    }
    return 0;
}
