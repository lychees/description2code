#include <iostream>
using namespace std;

int main()
{
	int T; cin >> T;
	while (T--) {
		int count = 0;
		string s; cin >> s;
		for (int i = 4; i < s.size() - 4; i++) {
			char c = s[i];
			if (!(c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u')) {
				count++;
			}
		}
		cout << count + 4 << "/" << s.size() << endl;
	}
	return 0;
}
