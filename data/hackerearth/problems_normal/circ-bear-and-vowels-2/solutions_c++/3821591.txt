#include <iostream>
#include <string>
#include <stdio.h>


using namespace std;

int t;
string s,vo="aeiouy";

int main(){

    //freopen("data.txt","r",stdin);
    int v,c,x;
    cin>>t;
    while(t--)
    {
        cin>>s;
        v=0;
        c=0;
        x=0;
        bool ok, okk=false;
        for(int i=0; i<s.size(); i++)
        {
            ok=false;
            for(int j=0; j<vo.size(); j++)
            {
                if(s[i]==vo[j])
                {
                    ok=true;
                    x=0;
                    v++;
                    break;
                }
            }
            if(!ok)
            {
                x++;
                c++;
            }
            if(x>2)
                {
                    okk=true;
                    break;
                }
        }
        if(c>v || okk)
        {
            cout<<"hard\n";
        }
        else
            cout<<"easy\n";
    }
}
