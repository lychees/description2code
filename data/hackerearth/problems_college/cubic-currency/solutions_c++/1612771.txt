typedef long long ll;
ll ff[30061];
ll f[22];

{
    int t;
    cin>>t;
    f[1] = 1;
    for(int i=2;i<=21;i++) f[i] = i*i*i;
    ff[0] = 1;
    for(int i=1;i<=21;i++)
    {
    	for(int j=0;j<=10000;j++)
    	ff[j+f[i]] += ff[j];
    }
   // for(int i=1;i<=100;i++) cout << i << ": " << ff[i] << endl;
  //  cout << endl;
    while(t--){
    	cin>>n;

    }

}
