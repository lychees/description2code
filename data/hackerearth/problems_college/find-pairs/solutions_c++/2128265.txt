#include <iostream>
using namespace std;
int main()
 {
    int t,i,j,k,*o;
	cin>>t;
	o=new int[1000009];
	while(t--)
	 {
	 	int n;
	 	cin>>n;
	 	
	 	for(i=1;i<=n;i++)
	 	 {
	 	 	cin>>o[i];
		 }
		 int count=0;
		 for(i=1;i<=n;i++)
		  {
		  	for(j=i;j<=n;j++)
		  	 {
		  	 	if(o[i]==o[j])
		  	 	 {
		  	 	 	count++;
					}
			   }
		  }
	cout<<count<<endl;	
	 }
	return 0;
 	
 }
