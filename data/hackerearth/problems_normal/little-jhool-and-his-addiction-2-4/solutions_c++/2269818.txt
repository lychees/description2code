#include <iostream>
#include <algorithm>
#include <string>
#include <string.h>
using namespace std;

int main()
{
    int t;

    cin>>t;

    while(t--){
        long int n,k;
        cin>>n>>k;

        long int A[n];

        for(int i = 0;i < n;i++){
            cin>>A[i];
        }

        sort(A,A + n);
        int i = 0;
        int j = n - 1;
        long long int B[n/2];
        while(i < j){
            B[i] = A[i] + A[j];
            i++;
            j--;
        }
        sort(B,B + n/2);

        long int ans = B[n/2 - 1] - B[0];

        cout<<ans<<"\n";
        if(ans < k){
            cout<<"Chick magnet Jhool!\n";
        }else if(ans == k){
            cout<<"Lucky chap!\n";
        }else{
            cout<<"No more girlfriends!\n";
        }


    }

    return 0;

}
