#include <iostream>
using namespace std;

int main()
{
    long long int T,N,K,M,sum,i,ans,total;
    cin>>T;
    
    while(T--)
    {
    	cin>>N>>K>>M;
    	sum=0;
    	total=0;
    	for(i=1;i<=N;i++)
    	{
    		sum+=i*i*i;
    	}
    	total=K*sum;
    	ans=total-M;
    	
    	if(ans>0)
    	cout<<ans<<endl;
    	
    	else
    	cout<<0<<endl;
    }
    return 0;
}
