#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main(){
int t;
ll a,b;
scanf("%d",&t);
while(t--){
    scanf("%lld %lld",&a,&b);
    ll x = __gcd(a,b);
    if(x==1)
        printf("Is a Co-Prime\n");
    else
        printf("Not a Co-Prime\n");
}
return 0;
}
