#include <bits/stdc++.h>
#define ll long long int
#define MOD 1000000007
using namespace std;
 
int main()
{
    int t,l,up,lo,i;
    string s;
    cin>>t;
    if(t<1 || t>10)
    	cout<<"Invalid Test\n";
    else
    {
    	while(t--)
    	{
    		cin>>s;
    		l=s.length();
    		if(!(l>=1 && l<=100))
    			cout<<"Invalid Input\n";
    		else
    		{
    			up=0;
    			lo=0;
    			for(i=0;i<l;i++)
    			{
    				if(s[i]>=65 && s[i]<=90)
    					up++;
    				else if(s[i]>=97 && s[i]<=122)
    					lo++;
    			}
    			if(up==0 && lo==0)
    				cout<<"Invalid Input\n";
    			else
    			cout<<min(lo,up)<<endl;
    		}
    	}
    }
    return 0;
}
