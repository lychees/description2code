int main(){
	int t;
	cin>>t;
	while(t--){
		int a;
		cin>>a;
		a%=360;
		int s=180;
		int gc=__gcd(a,s);
		a/=gc;
		s/=gc;
		if(s==1){
			if(a==1){
				cout<<"$"<<endl;
			}
			else{
				cout<<a<<"$"<<endl;
			}
		}
		else{
			if(a==1){
				cout<<"$/"<<s<<endl;
			}
			else{
				cout<<a<<"$"<<"/"<<s<<endl;
			}
		}
	}
	return 0;
}
