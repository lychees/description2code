#include <iostream>
#include <vector>

using namespace std;

int main()
{
    int test ;
    scanf("%d",&test) ;
    
    while(test--)
    {
    	int low , high ;
    	scanf("%d%d",&low,&high) ;
    	
    	// -1 means not visited 
    	vector<bool> mark(high-low+1,true) ;
    	
    	for(int i = 2 ; i*i <= high ; i++)
    	{
    		int temp ;
    		
    		if(low%(i*i) != 0)
    			temp = low + (i*i) - (low%(i*i)) ;
    		else
    			temp = low ; 
    		
    		while(temp <= high)
    		{
    			if(temp >= low && temp <= high)
    			{
    				mark[temp-low] = false ; 	
    			}
    		
    			temp += i*i ;
    		}
    	}
    	
    	int count = 0 ;
    	for(int i = low , j = 0 ; i <= high ; i++ , j++)
    	{
    		if(mark[j] == true)
    			count ++ ; 
    	}
    	printf("%d\n",count) ;
    }
    
    return 0;
}
