#include <iostream>
#include <math.h>

using namespace std;


int operacion(int* arreglo, int size){
	int c = arreglo[0];
	//cout << "numero" << c << '\n';
	for(int i = 1; i < size; i++){
		c = c & arreglo[i];
	}
	return c;
}

int creadorSubArreglos(int arreglo[], int ini, int final, int count){
	int* arr = new int[count + 1];

	for(int i  = 0; i < count + 1; i++){
		if(i != count){
			arr[i] = arreglo[ini + i];
		}else{
			arr[i] = arreglo[final];
		}
	}

	/*for(int i = 0; i < count + 1; i++){
		cout << arr[i] << '\t';
	}cout << '\n';*/

	//cout << arr[0] << '\t' << arr[1] << '\n';

	int c = operacion(arr, count + 1);
	return c;
}



bool comprobador(int* arreglo, int size){
	int count = 0;
	while(count < size){
		int i = 0;
		while(i < size - count + 1){
			for(int j = i; j < size; j++){
				int num = creadorSubArreglos(arreglo, i, j, count);
				if(num != 0){
					double numero = log2 (num);
					if(numero - (int)numero < 10e-3 && numero - (int)numero > -10e-3){
						return true;
					}
				}
			}
			i++;
		}
		count++;
	}
	return false;
}

int main(){
	int n;
	cin >> n;

	for(int i = 0; i < n; i++){
		int a;
		cin >> a;
		int arr[a];
		for(int j = 0; j < a; j++){
			cin >> arr[j];
		}

		if(comprobador(arr, a)){
			cout << "YES";
		}else{
			cout << "NO";
		}
		if(i != n - 1){
			cout << '\n';
		}
	}
}
