#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int arr1[n];
	int arr2[n];
	for(int i=0;i<n;i++){
		cin>>arr1[i];
	}
	for(int i=0;i<n;i++){
		cin>>arr2[i];
	}
	int max1=0;
	for(int i=0;i<n-1;i++){
		int a = abs((arr1[i])-(arr1[i+1]));
		if(a>max1){
			max1=a;
		}
	}
	int max2=0;
	for(int i=0;i<n-1;i++){
		int a = abs((arr2[i])-(arr2[i+1]));
		if(a>max2){
			max2=a;
		}
	}
	if(max1>max2){
		cout << "Dom\n" << max1;
	}else{
		cout << "Brian\n" << max2;
	}
    return 0;
}
