#include <iostream>
#include <queue>
using namespace std;

int main()
{
    int N;
    cin >> N;
    queue<int> callingQ;
    queue<int> idealQ;
    for(int i = 0; i < N; i++){
    	int val;
    	cin >> val;
    	callingQ.push(val);
    }
    for(int i = 0; i < N; i++){
    	int val;
    	cin >> val;
    	idealQ.push(val);
    }
    int totalTime = 0;
    while(N){
    	int co, io;
    	co = callingQ.front();
    	io = idealQ.front();
    	if(co == io){
    		callingQ.pop();
    		idealQ.pop();
    		totalTime++;
    		N--;
    	}
    	else{
    		callingQ.pop();
    		callingQ.push(co);
    		totalTime++;
    	}
    }
    cout << totalTime << endl;
    return 0;
}
