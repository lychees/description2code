#include <iostream>
using namespace std;

int main()
{
    int N;
    cin>>N;
    int flag=0;
    int a[100],b[100],c[100];
    for (int i=0;i<N;i++)
    {
    	cin>>a[i]>>b[i]>>c[i];
    }
    for (int i=0;i<N;i++)
    {
    	int d=(b[i]*b[i])-4*a[i]*c[i];
    	if (d>=0)
    	flag++;
    }
    cout<<flag;
    return 0;
}
