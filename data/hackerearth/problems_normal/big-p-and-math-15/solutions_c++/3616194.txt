#include <bits/stdc++.h>
using namespace std;
int main()
{
    long int a,b;
    cin>>a>>b;
    long int mx = 0,mn = 0;
    long int f = 1;
    while(a!=0||b!=0)
    {
       long int x,y;
       x = a%10;
       y = b%10;
       if(x==5||x==6)
       {
           mx += 6*f;
           mn += 5*f;
       }
       else
       {
           mx += x*f;
           mn += x*f;
       }

       if(y==5||y==6)
       {
           mx += 6*f;
           mn += 5*f;
       }
       else
       {
           mx += y*f;
           mn += y*f;
       }
       a /= 10;
       b /= 10;
       f *= 10;
    }
    cout<<mn<<" "<<mx<<endl;
    return 0;
}
