#include<bits/stdc++.h>
using namespace std;
int reve(long int N)
{
	long int n=N;
	long int reverse=0;
	while(n)
	{
		reverse=reverse*10+n%10;
		n/=10;
	}
	if(reverse==N)
	return 1;
	else
	return 0;
}
int PORNOT(long int N)
{
	if(N==2) return 1;
	long int i;
	if(N%2==0) return 0;
	for(i=3;i<=sqrt(N);i+=2)
	{
		if(N%i==0)
		return 0;
	}
	return 1;
}
int main()
{
	long int T,i,j,X;
	cin>>T;
	while(T--)
	{
		cin>>X;
		j=X;
		while(1)
		{
			if(reve(j) && PORNOT(j))
			{
				printf("%ld\n",j);
				break;
			}
			else
			j++;
		}
	}
}
