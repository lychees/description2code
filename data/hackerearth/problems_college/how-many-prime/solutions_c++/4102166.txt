#include <iostream>
using namespace std;

#define MAX 1000007


int not_primes[MAX];

int main()
{
	not_primes[1] = 1;
	not_primes[1] = 1;
	for(int i=2; i<MAX; i++)
 	{
 		if(not_primes[i]==0)
 		{
 			for(int j=i+i; j<MAX; j+=i)
 			{
 				not_primes[j] = 1;
 			}
 		}
 	}
 	int T;
 	cin>>T;
 	while(T--)
 	{
 		int n;
 		cin>>n;
 		int start;
 		int flag = 0;
 		int end;
 		if(n%2==0)
 		{
 			start = n/2-1;
 		}
 		else
 		{
 			start = n/2;
 		}
 		for(; start>1; start--)
 		{
 			if(not_primes[start]==0 && not_primes[n-start]==0)
 			{
 				end = n-start;
 				flag = 1;
 				break;
 			}
 		}
 		if(flag==1)
 		{
 			cout<<start<<" "<<end<<endl;
 		}
 		else
 		{
 			cout<<"-1"<<" "<<"-1"<<endl;
 		}
 	}
    return 0;
}
