r=input()
N=input()

for i in range(N):
        a=[]
        st=raw_input().split(' ')
        for k in st:
                a.append(int(k))
        
        j=0
        if a[j] < r or a[j+1] < r:
                print("UPLOAD ANOTHER")
        elif a[j] >= r and a[j+1] >= r:
                if a[j] == a[j+1]:
                        print("ACCEPTED")
                else:
                        print("CROP IT")
