#include <iostream>
using namespace std;

int main()
{
	int tc,size,count,i;
	long long int tmp;
    cin >>tc;
    while(tc--)
    {
    	count = 0;
    	cin >> size;
    	for(i=0;i<size;i++)
    	{
    		cin >> tmp;
    		if(tmp%2 != 0 )
    		{
    			count++;
    		}
    	}
    	if(count%2==0)	
    		cout << "Yes"<<endl;
    	else
    		cout << "No" << endl;
    	
    }
    
    return 0;
}
