using namespace std;
 
int T[300][300];
 
int main()
{
    ios::sync_with_stdio(false);
    int t,i,j,k,n,a,b,c,m;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        for(i=0;i<=n;++i)
        for(j=0;j<=n;++j)
        {
            if(i==j)
                T[i][j]=0;
            else
                T[i][j]=1000000;
        }
        for(i=0;i<m;++i)
        {
            cin>>a>>b>>c;
            for(j=a;j<b;++j)
            {
                for(k=j+1;k<=b;++k)
                {
                    T[j][k]=min(T[j][k],c);
                }
            }
        }
        for(k=0;k<=n;++k)
        for(i=0;i<=n;++i)
        for(j=0;j<=n;++j)
            T[i][j]=min(T[i][j],T[i][k]+T[k][j]);
        if(T[0][n]>=1000000)
            cout<<"-1\n";
        else 
            cout<<T[0][n]<<endl;
    }
    return 0;
}
