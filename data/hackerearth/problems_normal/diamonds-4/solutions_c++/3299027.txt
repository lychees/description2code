#include <iostream>
using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	int T,i,j,N,M,count;
    cin>>T;
    while(T--)
    {
    	cin>>N>>M;
    	char x[N][M];
    	for(i=0;i<N;++i)
    	for(j=0;j<M;++j)
    	cin>>x[i][j];
    	count=0;
    	for(i=0;i<N-1;++i)
    	for(j=0;j<M-1;++j)
    	{
    		if(x[i][j]=='/'&&x[i][j+1]=='\\'&&x[i+1][j]=='\\'&&x[i+1][j+1]=='/')
    		++count;
    	}
    	cout<<count<<endl;
    }
}
