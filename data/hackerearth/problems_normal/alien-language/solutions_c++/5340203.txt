#include <iostream>
using namespace std;
string s,w;
int count[26],count1[26],t,flag;
int main()
{
   cin>>t;
   while(t--)
   {cin>>s;
   cin>>w;
   for(int i=0;i<26;i++)
   {count1[i]=0;
   count[i]=0;
   }
   flag=0;
   for(int i=0;i<s.length();i++)
   count1[s[i]-'a']++;
   for(int i=0;i<w.length();i++)
   count[w[i]-'a']++;
   for(int i=0;i<26;i++)
   if(count1[i]>0&&count[i]>0)
   flag=1;
   if(flag)
   cout<<"YES"<<endl;
   else
   cout<<"NO"<<endl;
   }
   
    return 0;
}
