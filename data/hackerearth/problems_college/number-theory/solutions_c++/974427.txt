#include<bits/stdc++.h>
int main() {
	int t,i,j,k,n,remainder,f,l,p;
	long long c;
	char a[15];
	scanf("%d",&t);
	while(t--) {
		c=0;
		scanf("%s",a);
		n=atoi(a);
		for(i=0;a[i]!='\0';i++) {
			if(a[i]=='1') {
				c++;
				break;
			}
		}
		for(i=2;i*i<=n;i++) {
			if(n%i==0) {
				l=k=i;
				while(k != 0) {
					f=1;
                    remainder = k % 10;
                    for(j=0;a[j]!='\0';j++) {
		         	    if(a[j]-48==remainder){
				            f=0;c++;break;
			             }
		             }
		             if(f==0)break;
		             k=k/10;
                }
                //printf("aa %d %d\n",n,c);
                p=n/i;
                if(l==p)continue;
                while(p != 0) {
                	f=1;
                    remainder = p % 10;
                    for(j=0;a[j]!='\0';j++) {
		         	    if(a[j]-48==remainder){
				            f=0;c++;break;
			             }
		             }
		             if(f==0)break;
		             p=p/10;
                }
			}
		}
		if(n!=1)c+=1;
		printf("%lld\n",c);
	}
	return 0;
}
