#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    string n;
    long int t;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	long int l=n.size();
    	if(l<=2)
    	{
    		cout<<"Not possible!"<<endl;
    		continue;
    	}
    	sort(n.begin(),n.end());
    	next_permutation(n.begin(),n.end());
    	next_permutation(n.begin(),n.end());
    	cout<<n<<" ";
    	sort(n.begin(),n.end());
    	reverse(n.begin(),n.end());
    	prev_permutation(n.begin(),n.end());
    	prev_permutation(n.begin(),n.end());
    	cout<<n<<" ";
    	cout<<endl;
    }
    return 0;
}
