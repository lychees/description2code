#include <bits/stdc++.h>

using namespace std;

#define ll long long

int main() {
  ll t;
  cin >> t;
  while (t--) {
    ll m, n;
    cin >> m >> n;
    if (n == 0)
      cout << "0" << endl;
    else if (m >= n && n != 0) {
      cout << "10" << endl;
    } else if (m < n && n != 0) {
      ll r = n % m;
      ll r1 = m - r;
      ll r2 = n - r1;
      ll t = (5 * (n / m));
      // cout<<r2<<endl;
      if (r1 != 0)
        t += 5;
      if (r2 % m == 0)
        t += 5 * (r2 / m);
      else
        t += 5 * (r2 / m) + 5;
      // t+=5*()
      cout << t << endl;
    }
  }
  return 0;
}
