t=int(raw_input())
while t:
    m=map(int,raw_input().split())
    x1=m[0]
    y1=m[1]
    x2=m[2]
    y2=m[3]
    b=(x1-x2)
    a=(y1-y2)
    c=(y1*x2-x1*y2)
    #print a,b,c
    r=int(raw_input())
    if r*r*(a*a+b*b)>c*c:
	print "REPLANNING"
    elif r*r*(a*a+b*b)<c*c:
	print "SAFE"
    else:
	print "JUST MISSED"
    t-=1
