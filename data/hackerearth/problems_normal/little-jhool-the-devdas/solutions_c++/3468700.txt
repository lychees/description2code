#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int n,i,j,k,sum;
	cin >> n;
	vector<int> a;
	
	sum =0;
	for(i=0;i<n;i++)
	{
		cin >>k;
		a.push_back(k);
		sum += k;
	}
	
	sort(a.begin(), a.end());
	
	if(sum %2 == 1)
	{
		cout << sum ;
		return 0;
	}	
	else
	{
		for(i=0;i<a.size();i++)
		{
			sum -= a[i];
			if(sum % 2 == 1)
			{
				cout << sum;
				return 0;
			}	
		}
			
	}
	
	cout << ":(" ;
	
	
	//for(int i : a)
	//cout << i << " ";
    return 0;
}
