#include <bits/stdc++.h>
using namespace std;

int ans;

void dide(int a, int b, int c, int arr[], int n, int curr)
{
	if(curr==n){
		ans=min(ans,max(a,max(b,c)));
		return;
	}
	dide(a+arr[curr],b,c,arr,n,curr+1);
	dide(a,b+arr[curr],c,arr,n,curr+1);
	dide(a,b,c+arr[curr],arr,n,curr+1);
}

int main()
{
    
    int n,arr[12];
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",&arr[i]);
    ans=INT_MAX;
    dide(0,0,0,arr,n,0);
    cout<<ans<<endl;
    return 0;
}
