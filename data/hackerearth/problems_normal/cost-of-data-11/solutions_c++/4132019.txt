#include <iostream>
#include <string>
using namespace std;

int count;

//trie Node
class trieNode
{
	public:
		trieNode* edges[26];
};

//Create Node
trieNode* createNode()
{
	trieNode* newNode = new trieNode;
	
	for(int i = 0; i < 26; i++)
		newNode->edges[i] = NULL;
		
	return newNode;	
}

//Insert
void insertTrie(trieNode* root, string key)
{
	trieNode* iter = root;
	for(int i = 0; key[i] != '\0'; i++)
	{
		if(iter->edges[key[i] - 'a'] == NULL)
		{
			iter->edges[key[i] - 'a'] = createNode();
			count++;
		}
			
		iter = iter->edges[key[i] - 'a'];
	}
}

int main()
{
	count = 1;
    int n;
    cin>>n;
    string s;
    trieNode* root = createNode();
    
    while(n--)
    {
    	cin>>s;
    	insertTrie(root, s);
    }
    
    cout<<count;
    return 0;
}
