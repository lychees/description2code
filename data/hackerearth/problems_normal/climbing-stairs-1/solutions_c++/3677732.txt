#include <iostream>
using namespace std;
 
int main()
{
	int A;
	int B;
	int N;
	cin >>A>>B>>N;
	if(A<1 || N<1 || B<1)
	{
		return 0;
	}
	int days=0;
	int climbsteps = 0;
	while(true)
	{
		climbsteps = climbsteps+A;
		days++;
		if(climbsteps>=N)
		{
			break;
		}else
		{
			climbsteps = climbsteps - B;
		}
	}
    cout << days << endl;
    return 0;
}
