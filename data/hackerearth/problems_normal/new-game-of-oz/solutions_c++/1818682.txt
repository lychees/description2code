#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,i,j,count;
    cin>>t;
    for(j=0;j<t;j++)
    {
    	cin>>n;
    	int a[n];
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	count=0;
    	sort(a,a+n);
    	i=0;
    	while(i<n-1)
    	{
    		if(a[i]==a[i+1]-1)
    			i=i+2;
    		else
    			i=i+1;
    		count++;
    	}
    	if(i==n-1)
    		count++;
    	cout<<count<<endl;
    }
    
    return 0;
}
