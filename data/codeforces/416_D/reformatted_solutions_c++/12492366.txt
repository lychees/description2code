using namespace std;
long long point, n, a[200005], ans;
int main() {
  cin >> n;
  for (int i = 1; i <= n; i++)
    cin >> a[i];
  point = 1;
  while (point <= n) {
    ans++;
    int fi = point;
    while (fi <= n && a[fi] == -1)
      fi++;
    int se = fi + 1;
    while (se <= n + 1 && a[se] == -1)
      se++;
    if (se == n + 1)
      break;
    if (abs(a[se] - a[fi]) % (se - fi)) {
      point = se;
      continue;
    }
    int inc = (a[se] - a[fi]) / (se - fi);
    if (a[se] - inc * (se - point) <= 0) {
      point = se;
      continue;
    }
    point = se + 1;
    while (point <= n && a[se] + inc * (point - se) > 0 &&
           (a[point] == -1 || a[point] == a[se] + inc * (point - se)))
      point++;
  }
  cout << ans;
  return 0;
}
