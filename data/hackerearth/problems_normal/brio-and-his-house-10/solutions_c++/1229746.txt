#include<iostream>
#include<cmath>
#include<iomanip>
#define pi 3.1415
using namespace std;
int main(){
	int T;
	cin>>T;
	while(T--){
		double A,B,C,ans;
		cin>>A>>B>>C;
		ans=(A*B*C)/sqrt((A+B+C)*(A+B-C)*(A+C-B)*(B+C-A));
		ans=ans*ans*pi;
		cout<<fixed<<setprecision(4);
		cout<<ans<<endl;
	}
return 0;
}
