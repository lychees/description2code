#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char s[1001];
    	cin>>s;
    	int l = strlen(s);
    	if(l<3)
    	{
    		cout<<"NO"<<endl;
    		continue;
    	}
    	
    	int hash[26]={0};
    	for(int i=0;i<l;i++)
    		hash[s[i]-97]++;
    		
    	int final[26],k=0;
    	for(int i=0;i<26;i++)
    	{
    		if(hash[i]!=0)
    			final[k++] = hash[i];
    	}
    	
    	
    	if(k>=4)
    		cout<<"YES"<<endl;
    	else if(k==1)
        {
            if(l>=10)
    		  cout<<"YES"<<endl;
            else
                cout<<"NO"<<endl;
        }
    	else if(k==2)
        {
    		if(l<7)
                cout<<"NO"<<endl;
            else
                cout<<"YES"<<endl;
        }
    	else if(k==3)
        {
    		if(l>=5)
                cout<<"YES"<<endl;
            else
                cout<<"NO"<<endl;
        }
    }
    return 0;
}
