{
   // cout<<"dj"<<endl;
   //if(shots==0 && score!=0)return 0;
   //if(score==w || shots>n)
   //if(shots>n && score<w)return 0;

   if(shots>n || score>=w)
    {
        if(score>=w) return 1; else  return 0;
    }
   else if(visited[shots][score]!=0)return dp[shots][score];
    //cout<<dp[shots][score]<<endl;
    else {visited[shots][score]=true;return dp[shots][score]=max(prob(shots+1,score+x)*p1+(1-p1)*prob(shots+1,score),prob(shots+1,score+y)*p2+(1-p2)*prob(shots+1,score));}
}
