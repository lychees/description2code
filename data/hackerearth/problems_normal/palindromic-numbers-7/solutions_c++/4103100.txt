#include <iostream>
#include<string>
#include<cstdio>
#include <algorithm>
using namespace std;

int main()
{
    int t,a,b,c;
    cin>>t;
    while(t--)
    {
    	c=0;
    	cin>>a>>b;
    	for(int i=a;i<=b;i++)
    	{
    		string s = to_string(i);
		string t=s;
		reverse(s.begin(),s.end());
		if(s.compare(t)==0)
	      c++;
    		
    	}
    	cout<<c<<endl;
    }
    return 0;
}
