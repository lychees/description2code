#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {   int i,k=0,j;
    	string s,s1;
    	int a[27]={0};
    		cin>>s;
    		cin>>s1;
    		char s2[100002];
    		for(i=0;i<s1.length();i++)
    			a[int(s1[i])-97]++;
    	
    		for(i=0;i<s.length();i++)
    		{
    			if(a[int(s[i])-97]!=0)
    			{
    				for(j=0;j<a[int(s[i])-97];j++)
    				{s2[k++]=s[i];
    				}
    				a[int(s[i])-97]=0;
    			}
    		}
    		for(i=0;i<s1.length();i++)
    		{
    			cout<<s2[i];
    		}
    		cout<<"\n";
    }
    return 0;
}
