#include <bits/stdc++.h>
#define ll int 
using namespace std;
ll tar,m1,m2;
short dp[1000010][2];
ll solve(ll i,ll s)
{
	if(i<0)
	return 0;
	
	if((i<m1) && (i<m2) && (s==1))
	return 1;
	if((i<m1) && (i<m2)  && (s==0))
	return 0;
	
	if(dp[i][s]!=-1)
	return dp[i][s];
	
	ll res=0;
	
	if(i>=m1 && i>=m2 && s==0)
	{
		res= solve(i-m1,abs(s-1)) || solve(i-m2,abs(s-1));
	}
	
	else if(i>=m1 && i>=m2 && s==1)
	{
		res= solve(i-m1,abs(s-1)) && solve(i-m2,abs(s-1));
	}
	
	else if(i>=m1)
		res= solve(i-m1,abs(s-1));
		
		else
		res=  solve(i-m2,abs(s-1));
		
		
		
		dp[i][s]=res;
		return res;
	
	
}
ll solve1(ll i,ll s)
{
	
	
	if((i<m1)  && (s==1))
	return 1;
	
	if((i<m1) && (s==0))
	return 0;
	
	if(dp[i][s]!=-1)
	return dp[i][s];
	
	ll res=0;
	if(s==0)
	res= solve(i-m1,abs(s-1))|| solve(i,abs(s-1));
	else
		res= solve(i-m1,abs(s-1)) && solve(i,abs(s-1));
	
		dp[i][s]=res;
		return res;	
}

int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	ll i;
    
    	cin>>tar>>m1>>m2;
    	
    	if(tar==0)
    	{
    		cout<<"0"<<endl;
    		continue;
    	}
 
    	if( m1 ==0 && m2==0)
    	{
    		cout<<"0"<<endl;
    		continue;
    	}
    
    	
    	for(i=0;i<=tar;i++)
    	{
    		dp[i][0]=-1;
    		dp[i][1]=-1;
    	}
    	if((m1==0) || (m2==0))
    	{
    		if(m1==0)
    		m1=m2;
    		
    		ll ans=solve1(tar,0);
    	if(ans==1)
    	cout<<"1"<<endl;
    	else
    	cout<<"0"<<endl;
    	}
    	
    	else
    	
    	{
    
   		 ll ans=solve(tar,0);
    	if(ans==1)
    	cout<<"1"<<endl;
    	else
    	cout<<"0"<<endl;
    	}
    }
    return 0;
}
