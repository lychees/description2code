#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,q;
	cin>>n;
	int i,a[n+1];
	
	for(i=1;i<=n;i++)
	{
		cin>>a[i];
	}
long long int k;
cin>>k;
sort(a,a+n);
int flag=0;
for(i=0;i<n-2;i++)
{
	int l = i+1,r=n-1;
	while(r>l)
	{
		 if( a[i] +a[l] + a[r] == k)
            {
            	flag=1;
            	break;
            }
            else if (a[i] + a[l] + a[r] < k)
                l++;
            else 
                r--;
	}
	if(flag==1)
	{
		break;
	}
}
if(flag==1)
{
	cout<<"YES"<<"\n";
}
else
{
	cout<<"NO"<<"\n";
}
}
