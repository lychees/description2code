#include <iostream>
using namespace std;

int main()
{
    int t,n,i,j,max,hash[100][130],b[130],k,flag;
    char s[100][100];
    scanf("%d",&t);
   
    while(t--)
    {
    	
  for(j=0;j<=99;j++)
  {
   for(i=97;i<=122;i++)
    {
     	hash[j][i]=0;
    	b[i]=0;
    }
  }
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    	{
    	scanf("%s",s[i]);
    	
    
    		for(j=0;s[i][j];j++)
    		{
    			
    			if(!hash[i][s[i][j]])
    			{
    			hash[i][s[i][j]]=1;
                 b[s[i][j]]++;
    			}
    	    	}
             
    			

    		
    		
    	}
    	max=0;
    	for(i=97;i<=122;i++)
    	{
    	  if(b[i]>max)
    	  {
    	  max=b[i];
    	  flag=i;
    	  }
    	}
 
    	
    	printf("%c\n",flag);
    	
    }
    return 0;
}
