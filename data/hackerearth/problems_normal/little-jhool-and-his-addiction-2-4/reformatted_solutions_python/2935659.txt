noOfTestCases = int(raw_input())


def minimumPossDiff(items, k):
    citems = []
    for i in items:
        citems.append(int(i))
    citems.sort()

    times = []
    traversalPointer = 0
    traversalEnd = len(citems) / 2 - 1
    listLen = len(citems)

    while (traversalPointer <= traversalEnd):
        times.append(citems[traversalPointer] +
                     citems[listLen - traversalPointer - 1])
        traversalPointer += 1

    mx = max(times)
    mn = min(times)
    diff = mx - mn
    print diff

    if (diff > k):
        print 'No more girlfriends!'
    elif (diff == k):
        print 'Lucky chap!'
    else:
        print 'Chick magnet Jhool!'


for i in range(1, noOfTestCases + 1):
    count, k = raw_input().split(" ")
    k = int(k)
    items = raw_input().split(" ")
    minimumPossDiff(items, k)
