#include <iostream>
using namespace std;

int main()
{
    int n,k,count=0,i,j;
    cin>>n>>k;
    int a[n];
 
    for(int i=0;i<n;i++){
    	cin>>a[i];
    
    }
    for(i=0;i<n;i++){
       if(i+k<n){
       int sum=a[i]+a[i+k];
       if(sum%k==0) count++;
       else{
       	int diff=a[i]-a[i+k];
       	if(diff<0) diff=diff*(-1);
       	if(diff%k==0) count++;
       }
       }
    }
    cout<<count;
}
