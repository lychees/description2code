#include <iostream>
#include<climits>
using namespace std;

int main()
{
    int a[10001]={0};
    int p,q,n,max=INT_MIN,min=INT_MAX;
    cin>>p;
    while(p--)
    {
    	cin>>n;
    	if(n>max)
    	 max=n;
    	else if(n<min)
    	  min=n;
    	a[n]++;
    }
    cin>>q;
    while(q--)
    {
    	cin>>n;
    	if(n>max)
    	 max=n;
    	else if(n<min)
    	  min=n;
    	a[n]--;
    }
    while(min<=max)
    {
    	if(a[min])
    	  cout<<min<<" ";
    	min++;  
    }
    
    return 0;
}
