#include<bits/stdc++.h>
using namespace std;
 string all[1048576];
 int done[1048576]={0};
int main()
{
    ios::sync_with_stdio(false);
    int N,Q,K,L;
   
    cin>>N>>K>>L>>Q;
    assert(1<=N && N<=100000 && 1<=K && K<=5 && 1<=L && L<=15 && 1<=Q && Q<=100000);
    int i,j,id,attr;
    string name;
    map<string,int> allnames;
    for(i=0;i<N;i++)
    {
        id=0;
        cin>>name;
        allnames.insert(pair<string,int>(name,1));
        for(j=0;j<K;j++)
        {
            cin>>attr;
            id=(id*(L+1))+attr;
        }
        done[id]=i;
        all[id]=name;
    }
    while(Q--)
    {
        id=0;
        for(i=0;i<K;i++)
        {
            cin>>attr;
            id=(id*(L+1))+attr;
        }
        if(all[id].length()==0)
        {
            cout<<"You cant fool me :P";
        }
        else
        {
            cout<<all[id];
        }
        cout<<endl;
    }
}
