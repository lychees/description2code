#include <cstdio>
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
const int MAX=1000005;
int arr[256];
int main()
{
	int T;
	cin >> T;
	while(T--){
		int N;
		
		cin >> N;
		string s;
		cin >> s;
		int i;

		for(i=0;i<256;i++) arr[i]=0;

		for(i=0;s[i];i++) arr[s[i]-'a']++;

		
		
		vector<pair<int,pair<int,int> > > V;
		for( i=0;i<N;i++){
			string tmp;
			cin >> tmp;
			int len=0,j;
			int score=0;
			for(j=0;tmp[j];j++) {
				score+=arr[tmp[j]-'a'];
			}
			len=j;
		
			V.push_back(make_pair(score,make_pair(tmp.size(),i+1)));
			
			
		}
		sort(V.begin(),V.end());
		int ms=V.back().first;
		
		for(i=V.size()-1;i>=0;i--){
			if(V[i].first!=ms) break;
		}
		
		printf("%d\n",V[i+1].second.second);
	}
    return 0;
}
