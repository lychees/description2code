#include<bits/stdc++.h>
using namespace std;
static int a[101];
static long long dp1[71][1000];
int n;
long long dp(int cur,int sum,int prsum){
	if(cur==n){
		if(sum>=prsum){
		return 1;}
		else
			return 0;
	}
	//if(dp1[cur][sum]!=-1)
	//	return dp1[cur][sum];
	long long ans=0;
	if(sum>=prsum&&prsum!=-1)
		ans+=dp(cur+1,a[cur],sum);
		if(prsum==-1)
		   prsum=0;
	ans+=dp(cur+1,sum+a[cur],prsum);
	return (dp1[cur][sum]=ans);
}
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		for(int i=0;i<71;i++)
			for(int j=0;j<1000;j++)
				dp1[i][j]=-1;
				n=0;
		char m[100];
		scanf("%s",m);
		for(n=0;m[n]!='\0';n++)
			a[n]=m[n]-'0';
		printf("%lld\n",dp(0,0,-1));
	}
	return 0;
}
