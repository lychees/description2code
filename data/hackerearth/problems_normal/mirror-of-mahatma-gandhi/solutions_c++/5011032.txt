#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long test,len,i,flag=0;
	cin >> test;
	string input,output;
	while(test--)
	{
		flag=0;
		cin >> input;
		len=input.size();
		for(i=0;i<len;i++)
		{
			if(int(input[i])==48 || int(input[i])==49 || int(input[i])==56)
				flag=0;
			else
			{
				flag=1;break;
			}
		}
		if(flag==1)
			cout << "NO" << endl;
		else
		{
			output=input;
			reverse(output.begin(),output.end());
			if(output==input)
			cout << "YES" << endl;
			else
				cout << "NO" << endl;
		}
	}
	return 0;
}
