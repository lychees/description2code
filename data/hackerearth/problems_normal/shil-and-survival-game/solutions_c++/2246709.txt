/////////////////////////////////////////////////////////////
int S[MAX_N];
int leftMax[MAX_N];
int rightMax[MAX_N];
/////////////////////////////////////////////////////////////
int main(){
	RF("input.txt");
	//	WF("output.txt");

	int N;
	scanf("%d", &N);
	for (int i = 1; i <= N; i++) scanf("%d", S + i);
	for (int i = 1; i <= N; i++)
		leftMax[i] = max(leftMax[i - 1], S[i]);
	for (int i = N; i > 0; i--)
		rightMax[i] = max(rightMax[i + 1], S[i]);

	vector <int> V;
	for (int i = 1; i <= N; i++){
		if (i > 1 && i < N && S[i] < leftMax[i - 1] && S[i] < rightMax[i + 1])
			continue;
		V.emplace_back(i);
	}
	for (int k = 0; k < SZ(V); k++)
		printf("%d%c", V[k], " \n"[k == SZ(V) - 1]);
	return 0;
}
