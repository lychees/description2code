#include<iostream>
#include <algorithm>
#include<cstring>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int m;
		cin>>m;
		bool a[m+1];
		memset(a,0,sizeof(a));
		int n;
		cin>>n;
		int v[n];
		for(int i=0;i<n;i++)
		{
			cin>>v[i];
			a[v[i]]=1;
		}
		sort(v,v+n);
		for(int i=v[0]+1;i<=m;i++)
		{
			if(!(a[i]))
			{
				for(int j=0;j<n;j++)
				{
					if((i-v[j])<0)
					break;
					a[i]=(!(a[i-v[j]]));
					if(a[i]==1)
					break;
				}
			}
		}
		if(a[m])
		cout<<"Little Deepu"<<endl;
		else
		cout<<"Kate"<<endl;
	}
	return 0;
}
