#include<bits/stdc++.h>
using namespace std;
const int MAXN = 103;
int dp[MAXN][MAXN];
int ar[MAXN][MAXN];

int main() {
	int t;
	cin>>t;
	while(t--) {
		int maxi=-1;
		int n,m;
		cin>>n>>m;
		for(int i=1;i<=n;i++)
			for(int j=1;j<=m;j++)
				dp[i][j]=0,scanf("%d",&ar[i][j]);
		dp[1][1]=1;
		for(int i=1;i<=n;i++) {
			for(int j=1;j<=m;j++) {
				if(dp[i][j]!=0&&ar[i][j+1]>ar[i][j])
					dp[i][j+1]=max(dp[i][j+1],1+dp[i][j]);
				if(dp[i][j]!=0&&ar[i+1][j]>ar[i][j])
					dp[i+1][j]=max(dp[i+1][j],1+dp[i][j]);
				maxi=max(maxi,dp[i][j]);
			}
		}
		cout<<maxi<<endl;
	}
	return 0;
}
