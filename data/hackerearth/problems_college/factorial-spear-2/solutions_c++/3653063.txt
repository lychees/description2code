#include <cstdio>

#define SIZE 10000001

int arr[SIZE], ans[SIZE];

int main()
{
	int t, n, i, j, k, count;
	
	for (i = 2; i*i <= SIZE; ++i) {
		if (arr[i] == 0) {
			for (j = i*2; j < SIZE; j += i) {
				arr[j] = i;
			}	
		}
	}

	for (i = 2; i < SIZE; ++i) {
		count = 0;
		j = i;
		if (arr[i] == 0) {
			ans[i] = ans[i-1] + 1;
		} else {
			while (arr[j] != 0) {
				++count;
				j /= arr[j];
			}
			++count;
			ans[i] = ans[i-1] + count;
		}	
	}

	scanf("%d", &t);

	for (; t; --t) {
		scanf("%d", &n);
		printf("%d\n", ans[n]);
	}

	return 0;
}
