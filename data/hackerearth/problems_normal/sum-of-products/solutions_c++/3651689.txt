#include <stdio.h>
#include <math.h>
 
int main()
{
 
	long long int sum, i, a, s, e, tmp;
	int t;
	scanf("%d", &t);
	while (t--) {
		scanf("%lld", &a);
		sum = 0;
		for ( i = 1 ; i <= a; ) {
			e = a/i;
			s = (a/(i+1)) + 1;
			
			sum += (((s + e) * ( e - s + 1)) / 2 ) * i;
			if( s == 1) break;
			i = a/(s-1);
		}
		
		printf("%lld\n", sum % 1000000007);
	}
    return 0;
}
