#include <iostream>
using namespace std;

int main()
{
    int n,q;
    cin>>n>>q;
    int arr[n];
    for(int i=0;i<n;i++)
    cin>>arr[i];
    int l,r;
    while(q--)
    {
    	cin>>l>>r;
    	l--;
    	r--;
    	long long sum=arr[r];
    	long long count=arr[r];
    	for(int i=l;i<r;i++)
    	{
    		sum+=arr[i];
    		count-=arr[i];
    	}
    	cout<<sum<<" "<<count<<endl;
    }
}
