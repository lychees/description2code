#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,c,mat[101][101];
	cin >> t;
	while(t--)
	{
		cin >> n >> c;
		for(int i=0; i < n; i++)
		{
			for(int j=0; j < n; j++)
			cin >> mat[i][j];
		}
		for(int i=1; i < n; i++)
		{
			mat[0][i]=mat[0][i]+mat[0][i-1];
			mat[i][0]=mat[i][0]+mat[i-1][0];
		}
		for(int i=1; i < n; i++)
		{
			for(int j=1; j < n; j++)
			{
				mat[i][j]=max(mat[i-1][j],mat[i][j-1])+mat[i][j];
			}
		}
		int val=mat[n-1][n-1];
		if(val>=c)
		cout << val-c <<"\n";
		else
		cout << "-1\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
