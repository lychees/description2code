#include <iostream>

using namespace std;

const int MOD = 1000000007;

typedef long long int ll;

int fib[15001];
int fibSum[15001];

void pre() {
	
	fib[1] = 0;
	fib[2] = 1;
	
	for (int i = 3; i <= 15000; i++) {
		fib[i] = fib[i - 1] + fib[i - 2];
		fib[i] %= 10000;
	}
	
	fibSum[0] = 0;
	
	for (int i = 1; i <= 15000; i++) {
		fibSum[i] = (fibSum[i - 1] + fib[i]) % MOD;
	}
}

ll get(ll n) {
	ll times = (n / 15000) % MOD;
	n %= 15000;
	ll res = (times * fibSum[15000]) % MOD;
	res += fibSum[n];
	res %= MOD;
	return res;
}

int main() {

	pre();

	int t, rem;
	cin >> t;
	ll l, r, ans;
	while (t--) {
		cin >> l >> r;
		
		ans = get(r) - get(l - 1);
		
		ans %= MOD;
		
		if (ans < 0) ans += MOD;
		
		cout << ans << endl;
	}
	return 0;
}
