#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{

    int t;
    scanf("%d",&t);

    while(t--)
    {
      int n,k;
      scanf("%d",&n);
      scanf("%d",&k);
      int *a = new int[n];
      int *d = new int[k];
      for(int i=0 ; i< n;i++)
       scanf("%d",&a[i]);
      for(int i=0 ; i< k;i++)
      scanf("%d",&d[i]);

      int id=0,max=0;
      int g,x;
      for(g=0;g<n;g++)
       {
         for(x=max;x<k;x++)
         {
            if(a[g]<d[x])
              break;
         }

         if(x>max)
         {
           max=x;
           id=g;
         }

       }
       printf("%d\n",id);
    }
    return 0;
}
