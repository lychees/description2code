int n,m,ar[1<<20];

		si(ar[j]);
	
	}

		s[j]=s[j-1]+ar[j];

	reu(j,1,sz+1)

		for(int q=1;q<=m&&q+j-1<=sz;q++)

			bst[q]=max(bst[q],s[j+q-1]-s[j-1]);
		
		}

	rep(j,m+1)
