#include <bits/stdc++.h>
using namespace std;
map<string ,int> mp;
int main()
{
   // cout << "Hello World!" << endl;
   int n;
   cin>>n;
   string a,b;
   for(int i=0;i<((1<<n)-1);i++)
   {
   		cin>>a>>b;
   		if(mp.find(a)==mp.end())
   		mp[a]=1;
   		else
   		mp[a]++;
   }
   int mx=0;
   string p;
   for(auto &e:mp)
   {
   	if(e.second>mx)
   	{
   		mx=e.second;
   		p=e.first;
   	}
   }
   cout<<p;
    return 0;
}
