#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i,q;
        scanf("%d%d",&n,&q);
        long long int h[n],sum=0;
        for(i=0;i<n;i++)
            scanf("%lld",&h[i]);
        sort(h,h+n);
        for(i=0;i<n-1;i++)
            sum+= abs(h[i]-h[i+1]);
        sum=(sum)*q;
        printf("%lld\n",sum);
        sum=0;
    }
}
