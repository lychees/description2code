n = input()
l = list()
while(n>0):
	s = raw_input()
	l.append(s)
	n-=1
c=[0]*31
for line in l:
	if line[0] == "G":
		li = [int(s) for s in line.split() if s.isdigit()]
		for a in li:
			c[a] = c[a] + (li.count(a))*2
	if line[0] == "M":
		li1 = [int(s) for s in line.split() if s.isdigit()]
		for a in li1:
			c[a] = c[a] + (li1.count(a))
max = c.index(max(c))
if max==19 or max==20:
	print "Date"
else:
	print "No Date"
