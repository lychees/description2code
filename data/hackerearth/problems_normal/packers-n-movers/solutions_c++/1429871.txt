#include <bits/stdc++.h>
using namespace std;

int main()
{
   int n,m;
   cin>>m>>n;

   int port[n+1],tot[n+1];

   for(int i=0; i<n; i++)
   {
    cin>>port[i];
    tot[i]=(i>0?tot[i-1]:0)+port[i];
   }

   int dp[n][m];

   for(int i=0; i<n; ++i)
    for(int j=0; j<m; j++)
     dp[i][j]=1e9;

   for(int i=0; i<n; ++i)
    dp[i][0]=tot[i];

   for(int i=1; i<n; i++)
     for(int j=1; j<m; j++)
        for(int start=i; start>=1 ; start--)
             dp[i][j]=min(dp[i][j],max(dp[start-1][j-1],tot[i]-tot[start-1])); 
   cout<<dp[n-1][m-1];
   return 0;
}
