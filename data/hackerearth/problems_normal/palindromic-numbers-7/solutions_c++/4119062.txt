#include <iostream>
using namespace std;
bool pal(int n)
{
	if(n<10) return true;
	int val = 0;
	int tmp;
	int m;
	m=n;
	while(m)
	{
		tmp=m%10;
		val = val*10 + tmp;
		m/=10;
	}
	//cout<<"val="<<val<<" n="<<n<<endl;
	return val == n;
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int a;
    	int b;
    	int i;
    	cin>>a>>b;
    	int cnt = 0;
    	for(i=a;i<=b;++i)
    		if(pal(i)) cnt++;
    	cout<<cnt<<endl;
    }
    //cout<<pal(111)<<endl;
    //cout<<pal(121)<<endl;
    return 0;
}
