#include <bits/stdc++.h>
#define INF 999999999
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    
    int n,t,i,x,ctr2;
    
    double a[1001],ctr=0,max,p;
    
    cin>>t;
    
    while(t--)
    {
    	max=-INF;
    	ctr=0;
    	ctr2=1;
    	cin>>n;
    	
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	
    	for(i=0;i<n;i++)
    	{
    		cin>>p;
    		a[i]=a[i]*p;
    		//cout<<max<<endl;
    		if(a[i]>=max)
    		{
    			max=a[i];
    			x=i;
    		}
    		ctr+=a[i];
    	}
    	
    	sort(a,a+n);
    	
    	//for(i=0;i<n;i++)
    	//cout<<a[i]<<" ";
    	/*
    	for(i=n-1;i>=0;i--)
    	{
    		if(a[i-1]==a[i])
    		ctr2++;
    	}
    	
    	*/
    	
    	cout<<x<<" ";
    	cout<<fixed<<setprecision(3)<<(a[n-1]*ctr2)/ctr;
    	cout<<endl;
    	memset(a,0,sizeof(a));
    	//a.clear();
    	
    }
    
    
    return 0;
}
