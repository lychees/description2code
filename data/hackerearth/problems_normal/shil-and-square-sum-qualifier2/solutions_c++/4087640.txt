#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
 
long long int n,i,num,k,j,sum0[1000010],sum1[1000010],sum2[1000010];
int main(){
//	freopen("input.txt","r",stdin);
	ios::sync_with_stdio(0);
	
	cin>>n>>k;
	cin>>num;
	sum0[0]=sum1[0]=sum2[0] = 0;
	sum0[1]=sum1[1]=sum2[1] = num;
	for(i=2;i<=n;i++){
		cin>>num;
		sum0[i] = (sum0[i-1]+num)%mod;
		sum1[i] = (sum1[i-1]+i*num)%mod;
		sum2[i] = (sum2[i-1]+(((i*i)%mod)*num)%mod);
	}
	for(i=1;i<=n-k+1;i++){
		num = (sum2[k+i-1]-sum2[i-1])%mod;
		num = (num+((((i-1)*(i-1))%mod)*(sum0[k+i-1]-sum0[i-1]))%mod)%mod;
		num = (num-(2*(i-1)*(sum1[k+i-1]-sum1[i-1]))%mod)%mod;
		if(num < 0)
			num =mod+num;
		cout<<num<<" ";
	}
	
	return 0;
}
