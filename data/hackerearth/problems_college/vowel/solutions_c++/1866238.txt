#include <bits/stdc++.h>
using namespace std;
char vowel[]={'a','e','i','o','u'};
int check_vowel(char s){
	for(int i=0;i<5;i++)
	if(tolower(s)==vowel[i])
	return 1;
	return 0;
	
}
int main()
{
    int t,i,n,f;
    
    string s;
    cin>>t;
    while(t--){
    	cin>>s;
    	f=0;
    	n=s.length();
    	for(i=0;i<n;i++){
    		if(check_vowel(s[i]))
    		{cout<<s[i];
    			f=1;}
    	}
    	if(f==0)
    	cout<<"No";
    	cout<<endl;
    	
    	
    }
    return 0;
}
