#include<iostream>
#include<cstring>
#include<string>
using namespace std;

int main()
{
    string s;
    int t,l,i,d;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	l=s.size();
    	
    	for(d=9;d>=1;d--)
    	{
    		for(i=0;i<l;i++)
    		{
    			if((s[i]-'0')%d!=0) break;
    		}
    		if(i==l)
    		{
    			for(i=0;i<l;i++) cout<<(s[i]-'0')/d;
    			cout<<endl;
    			break;
    		}
    	}
    }
    return 0;
}
