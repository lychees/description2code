#include <bits/stdc++.h>
using namespace std;

int main()
{
   	int t, i,  n, k;
   	char s[1000000];
   	cin>>t;
   	while(t--) {
   		scanf(" %[^\n]s",s);
   		k = 0.0;
   		n = strlen(s);
   		for(i = 0 ; i < n; i++)
   		k += s[i];
   		k = k/n;
   		if(k%2 == 0)
   		cout<<s<<"\n";
   		else {
   			for(i = n-1; i >=0;i--)
   			cout<<s[i];
   			cout<<"\n";
   		}
   	}
    return 0;
}
