#!usr/bin/python

t = input()

while (t > 0):
    r = u = b = y = 0
    string = raw_input()

    for ch in string:
        if (ch == 'r'):
            r += 1
        elif (ch == 'u'):
            u += 1
        elif (ch == 'b'):
            b += 1
        elif (ch == 'y'):
            y += 1

    print min(r, u, b, y)
    t -= 1
