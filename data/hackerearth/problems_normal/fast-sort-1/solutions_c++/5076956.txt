#include <iostream>
using namespace std;
#include <algorithm>

struct data{
	char alpha;
	int val;
};

bool comp(data x, data y){
	if(x.val < y.val ) return true;
	else return false;
}

int main()
{
	int t;
	cin >> t;
	while(t--){
		string ref;
		cin >> ref;
		string st;
		cin >> st;
		int l = st.length();
		data ar[l];
		for(int i = 0; i < l; i++){
			char a = st[i];
			int index;
			for(int k = 0; k < 26; k++){
				if(ref[k] == a){ index = k; break;}
			}
			ar[i].alpha = a;
			ar[i].val = index;
		}
		sort(ar,ar+l, comp);
		for(int i = 0; i < l; i++){
			char a = ar[i].alpha;
			cout << a;
		}
		cout << endl;
		
	}
    
    return 0;
}
