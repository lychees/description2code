#include<bits/stdc++.h>
using namespace std;
#define fast cin.sync_with_stdio(0);cin.tie(0)
#define pii pair<int,int>
#define ff first
#define ss second
#define mp make_pair
#define pb push_back
#define INF 99999999
#define N 1100000
#define ll long long
#define llu unsigned long long 
#define MOD 1000000007
int ans[N];
int arr[N];
int dp[N][20];
int str[N];
int main()
{
	fast;
	int t;
	cin>>t;
	while(t--)
	{
	int n=20,l,i,j;
	cin>>l;
	for(i=0;i<N;i++)
	{
	  ans[i]=str[i]=arr[i]=0;
	  for(j=0;j<n;j++)
	    dp[i][j]=0;
	}   
	for(i=0;i<l;i++)
	{
		cin>>arr[i];
		str[arr[i]]++;
	}
	for(i = 0; i<(1<<n);i++)
	{
		dp[i][-1]=str[i];
		for(j=0;j<n;j++)
		{
			if(i&(1<<j))
			  dp[i][j]=dp[i][j-1]+dp[i^(1<<j)][j-1];
			else
			  dp[i][j]=dp[i][j-1];
		}
		ans[i]+=dp[i][n-1];
	}
	int fin=0;
	for(i=0;i<l;i++)
	{
		int cnt=0,temp=arr[i],mask=0;
		mask=(1<<n)-1-arr[i];
		
		 fin+=ans[mask];
	}
	cout<<fin<<"\n";
	}
	return  0;
}
