#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string s;
    while(t--){
    	cin>>s;
    	int v = 0, c = 0;
    	int n = s.length();
    	int acc = 0;
    	bool ok = true;
    	for(int i=0;i<n;i++){
    		if (s[i]=='a' || s[i] == 'e' || s[i] == 'i' || s[i] =='o' || s[i] =='u' || s[i] == 'y'){
    			v++;
    			acc = 0;
    			continue;
    		}
    		c++;
    		if (acc<2) acc++;
    		else {
    			ok = false;
    			break;
    		}
    	}
    	if (!ok) cout << "hard\n";
    	else if (c>v) cout << "hard\n";
    	else cout << "easy\n";
    }
    return 0;
}
