using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int m;
        long long sum=0;
        scanf("%d",&m);
        while(m--)
        {
            int a;
            long long int b;

            scanf("%lld",&b);
            scanf("%d",&a);
            sum=sum+(a*b);
            if(sum%9 != 0)
            sum=sum%9;

        }


if(sum!=0 && sum%9==0)
sum=9;

       if(sum>9)
        sum=sum%9;
        printf("%lld\n",sum);
    }
    return 0;
}
