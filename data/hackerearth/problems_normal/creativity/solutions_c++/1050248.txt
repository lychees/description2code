#include <iostream>
#include <string>

using namespace std;

int main()
{
	int testCount = 0;
	string s1 = "";
	string s2 = "";
	
	cin >> testCount;
	cin.clear();
	cin.ignore();
	
	while (testCount > 0)
	{
		getline(cin, s1);
		getline(cin, s2);

		if (s2.find(s1) != string::npos)
		{
			cout << "possible" << endl;
		}
		else
		{
			cout << "not possible" << endl;
		}
		
		testCount--;
	}
    return 0;
}
