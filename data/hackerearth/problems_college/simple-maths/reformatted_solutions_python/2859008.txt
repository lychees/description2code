def hcf(a, b):
    if (b == 0):
        return a
    return hcf(b, a % b)


def lcm(a, b):
    return (a*b)/hcf(a, b)


t = int(raw_input())
while (t > 0):
    a, b, c = map(int, raw_input().split())
    if (b == 0 or c == 0):
        print -1
    else:
        count = a/c
        count += a/b
        count -= a/lcm(b, c)
        print count
    t -= 1
