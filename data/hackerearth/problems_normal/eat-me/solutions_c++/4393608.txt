typedef long long ll;
ll MOD=1000000007;
ll pw(ll a,ll b){
  ll r=1;
  while(b>0){
     if(b&1){
       r=(r*a)%MOD;
        }
     b=b>>1;
     a=(a*a)%MOD;
  }
return (r)%MOD;
}
ll fact[1000005];
void fun(){
  fact[0]=1;
  fr(i,1,1000005){
     fact[i]=(fact[i-1]*i)%MOD;
  }
}
 
int main(){
    fun();
   int n,k;
   si(n),si(k);
   ll ans=0;
   fr(i,1,k+1){
      ans=(ans+ ((fact[i+n-1]*pw(fact[n-1],MOD-2)%MOD)%MOD *pw(fact[i],MOD-2)%MOD)%MOD  )%MOD;
   }
  pll(ans);pln;
 
return 0;
}
