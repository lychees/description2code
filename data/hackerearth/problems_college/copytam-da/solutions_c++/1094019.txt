#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int n;
   	cin>>n;
   	int a,hash[10001]={0};
   	for(int i=0;i<n;i++)
   	cin>>a,hash[a]=1;
   	int count=0;
   	for(int i=0;i<10001;i++)
   	if(hash[i])
   	count++;
   	
   	cout<<count<<endl;
   }
    return 0;
}
