#include<iostream>

using namespace std;
int answers[10000009];
int main(){


    int t,sq,ans,count,loop,i;
    answers[1]=1 , answers[2]=1;
    cin>>t;
    while(t--){
        cin>>sq;
        ans=0;
        for(i=1;i<=sq;i++){
            if(answers[i])
                ans+=answers[i];
            else{

                count=0;
                for(loop=1;loop*loop<=i;loop++){
                    if(i%loop==0){
                        count++;
                    }
                }
                answers[i]=count;
                ans+=count;
            }
        }
        cout<<ans<<endl;



    }
    return 0;
}
