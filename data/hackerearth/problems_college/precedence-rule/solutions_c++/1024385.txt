#include <iostream>
#include <string>
using namespace std;

int main()
{
    int T;
    cin >> T;
    for (int it = 0; it < T; it++) {
    	string st;
    	cin >> st;
    	int n = st.length() - 1;
    	long long result = st[n] - '0';
    	for (int i = n - 1; i >= 1; i -= 2)
    		switch (st[i]) {
    			case '+':
    				result = ((long long)st[i-1] - '0') + result;
    				break;
    			case '*':
    				result = ((long long)st[i-1] - '0') * result;
    				break;

    			case '-':
    				result = result - ((long long)st[i-1] - '0');
    				break;
    		}
    	cout << result << endl;	
    }
    return 0;
}
