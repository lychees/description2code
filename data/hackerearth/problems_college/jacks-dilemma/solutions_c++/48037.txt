#include<string>

using namespace std;

int main()
{
    int t,q,b,j,l,temp1,temp,n;

    for(q=0; q<t; q++)
        {
            cin>>n;
            cin>>j>>l;
            cin>>c;
            temp=j*10;
            if(c=='U')
            {
                temp1=((n-l)*5)+10+(n*5)+10;
                if(n!=j)
                     {
                        temp1=temp1+10;
                     }
            }
            if(c=='D')
            {
                if(l<j)
                {
                    temp1=(l*5)+(n*10)+10+10+10;
                    if(n!=j)
                     {
                        temp1=temp1+10;
                     }
                }
                else
                {
                    temp1=(l*5)+10;

                        temp1=temp1+10;

                }
            }
            if(temp1>temp)
            {
                cout<<"Stairs"<<" "<<temp<<endl;
            }
            else
            {
                cout<<"Lift"<<" "<<temp1<<endl;
            }
        }
        return 0;
}
