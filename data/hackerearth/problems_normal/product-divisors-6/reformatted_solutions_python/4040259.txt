from sys import stdin
x = 6
# print x
pr = [0]*(10**x+5)
hi = 10**x
# print pr
for n in xrange(2, hi):
    if pr[n] == 0:
        cur = n
        while cur <= hi:
            pr[cur] = n
            cur += n
# print pr
a = [0]*(10**x+5)
n = int(stdin.readline())
aa = map(int, stdin.readline().split())
# print a
for m in aa:
    while m != 1:
        a[pr[m]] += 1
        m /= pr[m]
# print aa
ans = 1
mo = 10**9 + 7
for i in a:
    ans = (ans*(i+1)) % mo
print ans
