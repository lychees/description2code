#include<bits/stdc++.h>
using namespace std;
int n,x,sum;
const int MAXN = 1e5+3;
bool dp[102][MAXN];
int ar[102];
void subsetSum() {
	for(int i=0;i<=x;i++)
		dp[0][i]=false;
	for(int i=0;i<=n;i++)
		dp[i][0]=true;
	for(int i=1;i<=n;i++) {
		for(int j=1;j<=x;j++) {
			dp[i][j]=dp[i-1][j];
			if(j>=ar[i])
				dp[i][j]=dp[i][j] | dp[i-1][j-ar[i]];
		}
	}
}
int main() {
	int t;
	cin>>t;
	while(t--) {
		cin>>n>>x;
		sum=0;
		for(int i=1;i<=n;i++) {
			scanf("%d",&ar[i]);
			sum+=ar[i];
		}
		if(sum<=2*x) {
			subsetSum();
			int dif=2*x-sum;
			bool f=false;
			for(int i=x;!f and i>=x-dif;i--) {
				if(dp[n][i])
					f=true;
			}
			if(f)
				cout<<"YES\n";
			else
				cout<<"NO\n";
		}
		else
			cout<<"NO\n";
	}
	return 0;
}
