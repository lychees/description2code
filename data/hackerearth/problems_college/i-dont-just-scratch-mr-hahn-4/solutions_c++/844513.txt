#include <stdio.h>
#include <map>
using namespace std;
map<int,int> nssd(int k) {
	if (k == 1) {
		map<int,int> ans;
		for (int i = 0; i < 10; i++)
		  ans[i*i] = 1;
		  return ans;
	}
	map<int,int> bns = nssd(k-1), ans;
	for (int i = 0; i < 10; i++)
	  for (map<int,int>::iterator it = bns.begin(); it != bns.end(); it++)
	    ans[i*i + it->first] += it->second;
	 return ans;
}
int cyc(int k) {
	if (!k) return 1;
while (k != 1 && k != 89) {
	char buf[42];
	sprintf(buf, "%i", k);
	int kk=0;
	for (int i =0; buf[i];i++) kk += (buf[i]-'0') * (buf[i]-'0');
	k=kk;
}
return k;
}

{
