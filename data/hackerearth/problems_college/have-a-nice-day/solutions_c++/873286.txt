#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	vector<int> men(n), women(n);
    	for(int i=0;i<n;i++) cin>>men[i];
    	for(int i=0;i<n;i++) cin>>women[i];
    	sort(men.begin(),men.end());
    	sort(women.begin(),women.end());
    	long long ans = 0;
    	for(int i=0;i<n;i++) ans += (long long)men[i]*women[i];
    	cout<<ans<<endl;
    }
    return 0;
}
