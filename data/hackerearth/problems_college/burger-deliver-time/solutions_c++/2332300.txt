#include <bits/stdc++.h>
using namespace std;
map<pair<int,int>,bool>m;
map<pair<int,int>,bool>::iterator it;

{
	int t;
	cin>>t;
	int a,b,i=1;
	pair<int,int>o;
	while(t--){
		cin>>a>>b;
	o=make_pair(a+b,i);
	m[o]=true;

	}

		o=it->first;
		cout<<o.second<<" ";
	}
	
	
    return 0;
}
