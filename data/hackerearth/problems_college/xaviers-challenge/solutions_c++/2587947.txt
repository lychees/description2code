#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    int n;
    char s[102];
    scanf(" %[^\t\n]s",s);
    n=strlen(s);
    int ans=0;
    for(int i=0;i<n;i++) ans+=(int)s[i];
    ans/=n;
    //cout<<ans<<" "<<n<<endl;
    if(ans%2==0) 
    {
    	for(int i=0;i<n;i++) cout<<s[i];
    	cout<<endl;
    }
    else 
    {
    	for(int i=0;i<n;i++) cout<<s[n-i-1];
    	cout<<endl;
    }
    }
    return 0;
}
