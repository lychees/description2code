def modpow(base, exp, mod):
    res = 1
    while exp > 0:
        if exp & 1:
            res = res * base % mod
        exp >>= 1
    return res


def pow(base, exp):
    res = 1
    while exp:
        res = res * base
        exp -= 1
    return res


fact = [1]
for i in range(1, 150):
    fact.append(i*fact[i-1])

    n, m = int(x[0]), int(x[1])
