#include <iostream>
using namespace std;

int main()
{
    long int t,n,k,i;
    cin>>t;
    while(t--)
    {    long int count=0;
    	cin>>n>>k;
    	long int a[n];
    	for(i=0;i<n;i++)
    	{
       	 cin>>a[i];
    	 if(a[i]<=0)
    	 count++;
    	}
    	if(count>=k)
    	cout<<"NO"<<endl;
    	else
    	cout<<"YES"<<endl;
    }
    return 0;
}
