#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
ll int f[1055000];
ll int a[1030];
ll int mysqr(ll int x)
{
	return (x*x)%mod;
}
ll int mypow(ll int x,ll int n)
{
	if(n==1)
	return x;
	if((n%2)==0)
	return mysqr(mypow(x,n/2))%mod;
	return (x * mysqr(mypow(x,n/2)))%mod;
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	f[0]=1;
	ll int i,j;
	for(i=1;i<1055000;i++)
	{
		f[i]=(f[i-1]*i)%mod;
	}
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n,s=0,num,den,ans;
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		ans=1;
		s=a[0];
		for(i=1;i<n;i++)
		{
			s+=a[i];
			//~ cout<<s<<" C "<<a[i]-1<<endl;
			num=f[s-1];
			den=mypow((f[s-1-(a[i]-1)]*f[a[i]-1])%mod,mod-2);
			num=(num*den)%mod;
			ans=(ans*num)%mod;
		}
		cout<<ans<<endl;
	}
	return 0;
}
