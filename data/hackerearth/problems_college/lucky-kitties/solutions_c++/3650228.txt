#include <iostream>
using namespace std;

int main()
{
	int k;
	cin>>k;
    long long n;
    cin>>n;
    int luck,count=0;
    luck=0;
    for(int i=1;i<=n;++i)
    {
    	for(int j=i;j!=0;j/=10)
    	{
    		if(j%10==4||j%10==7){++luck;}
    	}
    	if(luck>k){++count;luck=0;}
    }
    cout<<count;
}
