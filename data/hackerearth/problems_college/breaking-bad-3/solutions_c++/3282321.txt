#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		scanf("%d",&n);
		long long int a[n];
		for(int i=0;i<n;i++)
		scanf("%lld",&a[i]);
		sort(a,a+n);
		long long ans=0;
		for(int i=n-1;i>0;i-=2)
		   ans+=a[i]*a[i-1];
		  printf("%lld\n",ans);
	}
	return 0;
}
