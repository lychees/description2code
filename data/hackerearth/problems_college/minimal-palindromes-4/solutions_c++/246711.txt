void Tconfigure(vector<int> &v,int sz)
{
    int i;
    vector<int> res(MAX);
    //for(i=0;i<sz;i++)
    //printf("%d ",v[i]);

    //  Max value is 9, if number is divided by k(>9)
    //  it's number of digits will be reduced
    // Ex- (9999/10 is 999.9 which is not of same number of digits

    int k=9,flag=0;
    while(k>0)
    {
        for(i=0;i<sz;i++)
        {
            if(v[i]%k==0)
            res[i]=v[i]/k;
            else
            {
                k--;
                break;
            }
        }
        //All the digits are divisible by k, print the new constructed palindrome
        if(i==sz)
        {
            for(i=0;i<sz;i++)
            printf("%d",res[i]);
            flag=1;
            break;
        }
    }
    //if there is no possible such palindrome, then print the given palindrome
    if(!flag)
    for(i=0;i<sz;i++)
    printf("%d",v[i]);

}

int main()
{
    int t,i,j;
    vector<int> v(MAX);
    char str[MAX];
    sd(t);
    while(t--)

        scanf("%s",str);
        for(i=0;str[i]!='\0';i++)
        v[i]=str[i]-'0';
        Tconfigure(v,i);
        printf("\n");
    }
