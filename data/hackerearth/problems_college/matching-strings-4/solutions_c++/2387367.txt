#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int a,b,c,l=0;
		string s1,s2;
		cin>>s1>>s2;
		a=s1.size();
		b=s2.size();
		if(a>b)
		c=b;
		else
		c=a;
		for(int i=0;i<c;i++)
		{
			if(s1[i]==s2[i])
			l++;
		}
		cout<<l<<endl;
	}
	return 0;
}
