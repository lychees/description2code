#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
	cout<<setprecision(6)<<fixed;
    int n,L,D;
    cin>>n>>L>>D;
    double a[n+2];
    a[0]=0;
    a[n+1]=D;
    double p[n+2];
    double P[n+2];P[0]=1;
    p[0]=p[n+1] = 1;
    for(int i=1;i<=n;i++)cin>>p[i];
    for(int i=1;i<=n;i++)cin>>a[i];
    
	for(int i=1;i<=n+1;i++){
    	double max = -1;
    	for(int j=0;j<i;j++){
    		if(a[i]-a[j] <= L)
    			P[i] = p[i]*P[j];
    		if(max <= P[i])max = P[i];
    	}
    	P[i]=max;
    }
    cout<<setprecision(6)<<P[n+1]<<endl;
    return 0;
}
