#include<bits/stdc++.h>
using namespace std;
int A[100001][3];
int B[100001][3];
int main()
{
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<3;j++)
                scanf("%d",&A[i][j]);
        }
        for(int i=0;i<3;i++)
            B[0][i]=A[0][i];
        for(int i=1;i<n;i++)
        {
            B[i][0]=min(B[i-1][1],B[i-1][2])+A[i][0];
            B[i][1]=min(B[i-1][0],B[i-1][2])+A[i][1];
            B[i][2]=min(B[i-1][0],B[i-1][1])+A[i][2];
        }
        int res;
        res=min(min(B[n-1][0],B[n-1][1]),B[n-1][2]);
        printf("%d\n",res);
    }
    return 0;
}
