#include <bits/stdc++.h>
using namespace std;
 
int main()
{
	int t;
	cin>>t;
	string s,s1;
	//int count[27];
	while(t--){
	s.clear();s1.clear();
	int count=0,slen,slen1;
	  cin>>s>>s1;
	  slen=s.length();
	  slen1=s1.length();
	  if(slen>slen1){
	  	for(int i=0;i<s1.length();i++){
	  		if(s[i]==s1[i])
	  		count++;
	  	}
	  }else{
	  	for(int i=0;i<s.length();i++){
	  		if(s1[i]==s[i])
	  		count++;
	  	}
	  }
	  cout<<count<<"\n";
	}
    return 0;
}
