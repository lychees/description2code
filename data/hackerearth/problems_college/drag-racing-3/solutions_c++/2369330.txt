#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,lt;
	cin>>n;
	lt = pow(2,n)-1;
	string a,b;
	map<string,bool> hash;
	while(lt--)
	{
		cin>>a>>b;
		if(hash.find(a)==hash.end() || hash[a]==true)
		hash[a]=true;
		else
		hash[a]=false;
		
		hash[b]=false;
	}
	map<string,bool> :: iterator it;
	for(it = hash.begin();it!=hash.end();it++)
	{
		if(it->second==true)
		{
			cout<<it->first<<endl;
			break;
		}
	}
	return 0;
}
