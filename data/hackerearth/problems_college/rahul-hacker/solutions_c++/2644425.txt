#include <iostream>
using namespace std;

int main()
{
	long long int t,a,b,rem,rev,rem1,rev1,rem2,rev2,s;
    cin>>t;
    while(t--)
    {
    	rev1=0;
    	rev2=0;
    	rev=0;
    	cin>>a>>b;
    	while(a!=0)
    	{
    	rem=a%10;
    	rev=(rev*10)+rem;
    	a=a/10;
    	}
    	while(b!=0)
    	{
    	rem1=b%10;
    	rev1=(rev1*10)+rem1;
    	b=b/10;
    	}
    	s=rev+rev1;
    	while(s!=0)
    	{
    	rem2=s%10;
    	rev2=(rev2*10)+rem2;
    	s=s/10;
    	}
    	cout<<rev2<<endl;
    }
    return 0;
}
