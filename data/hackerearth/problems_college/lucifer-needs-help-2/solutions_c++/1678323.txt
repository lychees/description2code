#include <bits/stdc++.h>
#define DEBUG 0

using namespace std;
#define limt 1000010

long long a[limt];
long long add[limt];

int main()
{
	//freopen("in.txt", "r", stdin);
	int n, m;
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i) {
	    scanf("%lld", &a[i]);
	}

	scanf("%d", &m);
	int l, r;
	long long k;
	while (m--) {
		scanf("%d%d%lld", &l, &r, &k);
		add[l] += k;
		add[r + 1] -= k;
	}

	for (int i = 1; i <= n; ++i) {
	    add[i] += add[i - 1];
	}

	for (int i = 1; i <= n; ++i) {
	    printf("%lld\n", a[i] + add[i]);
	}

    return 0;
}
