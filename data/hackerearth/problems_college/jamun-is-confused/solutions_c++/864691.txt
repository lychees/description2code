#include <iostream>
using namespace std;
#define ll long long

ll gcd(ll a,ll b)
{
	if(!a)
	{
		return b;
	}
	else if(!b)
	{
		return a;
	}
	else if(a>b)
	{
		return gcd(a%b,b);
	}
	else
	{
		return gcd(a,b%a);
	}
}

int main()
{
    int t;
    cin>>t;
    int n,i;
    ll g;
    while(t--)
    {
    	cin>>n;
    	ll a[n];
    	for(i=0;i<n;i++)cin>>a[i];
    	g = a[0];
    	for(i=1;i<n;i++)g = gcd(g,a[i]);
    	cout<<g<<"\n";
    }
    return 0;
}
