#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string a;
 	while(t--)
 	{
 		int count=0,flag=1,tempcount=0;
 		cin>>a;
 		for(int i=0;i<a.size();i++)
 		{
 			// cout<<(int)(a.at(i));
 			switch((int)(a.at(i)))
 			{
 				case 88:
 				flag=0;
 				break;
 				case 79:
 				tempcount++;
 				break;
 				case 42:
 				// cout<<tempcount;
 				if(flag==1)
 				count+=tempcount;
 				flag=1;
 				tempcount=0;
 				break;
 			}
 		}
 		if(flag==1)
 		count+=tempcount;
 		cout<<count<<endl;
 	}
    return 0;
}
