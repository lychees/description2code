testCases = int(raw_input())


def closest(a, b, c):
    a, b, c = int(a), int(b), int(c)
    xo = int(pow(a, b))
    num1 = xo + c - (xo % c)
    num2 = xo - (xo % c)
    if (xo < 0):
        return 0
    if (c - (xo % c) > (xo % c)):
        return num2
    elif (c - (xo % c) < (xo % c)):
        return num1
    else:
        return num2


for j in range(1, testCases + 1):
    a, b, c = raw_input().split(" ")
    print closest(a, b, c)
