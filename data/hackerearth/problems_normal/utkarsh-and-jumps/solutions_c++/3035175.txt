#include <iostream>
#include<iomanip>
using namespace std;
double d[1000001];
int main()
{
   	double p;
   	int n;
    cin>>n>>p;
    d[0]=d[1]=0;
    d[2]=p/100;
    d[3]=1-p/100;
    for(int i=4;i<=n;i++){
    	d[i]=(d[i-2]*p/100)+(d[i-3]*(100-p)/100);
    }
    cout<<fixed<<setprecision(6)<<d[n];
    return 0;
}
