inline void inp( int &n ) {
        n=0; int ch = getcx(); int sign = 1;
        while(ch < '0' || ch > '9') { if(ch == '-') sign=-1; ch = getcx(); }
        while(ch >= '0' && ch <= '9') { n = (n << 3) + (n << 1) + ch - '0', ch = getcx(); }
		n = n * sign;
}


vector<int> P;
bool mark[100009];
void init() {
	P.PB(2);
	for(int i=3;i<=400;i+=2){
		if(!mark[i]){
			for(int j=i*i;j<=100000;j+=(i<<1)){
				mark[j] = 1;
			}
		}
	}
	for(int i=3;i<=100000;i+=2){
		if(!mark[i])
			P.PB(i);
	}
}

int factor[100];
int freq[100];
int nfactors;
void factorize(int x) {
	MS(freq);
	nfactors = 0;
	for (int i=0;i<P.size();i++) {
		if(x == 1) break;
		if (x%P[i] == 0) {
			int c=0;
			while (x % P[i] == 0) {
				x/=P[i];
				c++;
			}
			factor[nfactors] = P[i];
			freq[nfactors++] = c;
		}
	}
	if (x > 1) {
		factor[nfactors] = x;
		freq[nfactors++] = 1;
	}
}
void findDivisors(VI &divs, int index, int product) {
	if(index == nfactors){
		divs.PB(product);
		return;
	}
	findDivisors(divs, index+1, product);
	for (int i=0;i<freq[index];i++) {
		product = product * factor[index];
		findDivisors(divs, index+1, product);
	}
}


int main() {
	init();
	std::ios_base::sync_with_stdio(false);
	int t;
	cin >> t;
	while(t--){
		int n, k;
		cin >> n >> k;
		factorize(k-1);
		VI a;
		findDivisors(a, 0, 1);
		int num = n - SZ(a);
		int den = n;
		int g = __gcd(num, den);
		num /= g, den /= g;
		printf("%d/%d\n", num, den);
	}
	return 0;
}
