#include <iostream>
using namespace std;
#define ll long long
#include <vector>
#include <algorithm>

int main()
{	ll t,x,i,p,q,r=0;
	cin>>t;
	vector<ll> z;
	for(i=0;i<t;++i)
	{	cin>>x;
		z.push_back(x);
	}
	sort(z.begin(),z.end());
	for(i=1;i<t;++i)
	{	if(z[i]>0 && z[i-1]<=0)
		{	r=i;
			break;
		}
	}
	p=z[0]*z[1];
	q=z[t-2]*z[t-1];
	if(r!=0)
		r=z[r-1]*z[r];
	if(q>p)
		p=q;
	if(r>p)
		p=r;
	cout<<p;
    return 0;
}
