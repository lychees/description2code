#include<bits/stdc++.h>
using namespace std;
int main()

{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        int a,b;long long int s=1,m=0;
        cin>>a>>b;
        for(int j=a;j<=b;j++)
        {
            for(int k=j;k>1;k--)
            {
                s=s*k;
            }
            m=m+s;s=1;
        }
        cout<<m%10<<endl;

    }
    return 0;
}
