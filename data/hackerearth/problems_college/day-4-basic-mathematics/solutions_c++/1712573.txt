#include<iostream>
using namespace std;
int main()
{
    long long i,n;
    cin>>n;
    for(i=0;i<n;i++)
    {
        long long a,b=0,j=0;
        cin>>a;
        do
        {
            b=b+j;
            j++;
        }while(a>b);
        j--;
        if((j%3)==0)
            cout<<"SL"<<endl;
        else if((j%3)==1)
            cout<<"LB"<<endl;
        else if((j%3)==2)
            cout<<"BS"<<endl;
    }
    return 0;
}
