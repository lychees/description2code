'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
t=int(raw_input(''))
while t != 0:
	t-=1
	n,tc=map(int,raw_input().split())
	x = [int(raw_input('')) for i in xrange(n)]	
	i=1
	j=0
	f=0
	vc=x[0]
	while i < n:
		while vc < tc and i < n:
			vc +=x[i]
			i+=1
		while vc > tc and j<=i:
			vc-=x[j]
			j+=1
		if vc==tc:
			f=1
			break;
	if f==1 or vc == tc:
		print 'YES'
	else:
		print 'NO'
