#include<iostream>
#include<stdlib.h>
using namespace std;

int cmp(const void*x,const void*y)
{
	return *(int*)x - *(int*)y;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int arr[3];
		cin>>arr[0]>>arr[1]>>arr[2];
		qsort(arr,3,sizeof(int),cmp);
		int ans=0;
		int d1=arr[1]-arr[0];
		int d2=arr[2]-arr[1];
		if(d1>d2)
			ans=d1-1;
		else
			ans=d2-1;
		cout<<ans<<endl;
	}
	return 0;
}
