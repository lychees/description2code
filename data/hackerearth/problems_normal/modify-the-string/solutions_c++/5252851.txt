#include <iostream>
#include <string>
using namespace std;

int main()
{
    
    char buf[140];
    cin >> buf; 
    std::string s(buf); 
    int len = s.length(); 
    for ( int i=0; i < len; i++ )
       {
       	if (isupper(buf[i]))
       	     buf[i]=tolower(buf[i]); 
       	else if (islower(buf[i]))
       		buf[i]=toupper(buf[i]);  
       	}
    
    
    cout << buf  << endl;
    return 0;
}
