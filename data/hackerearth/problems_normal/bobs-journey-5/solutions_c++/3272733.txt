#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t, n, i, j, k, ans, a[27];

    string s;

    cin>>t;

    while(t--) {
        cin>>n;
        for (i = 0; i < 26; i++) {
            a[i] = 0;
        }

        while(n--) {
            s.clear();
            cin>>s;

            a[s[0] - 'a']++;
        }
        int flag = 0;
        for (i = 0; i < 26; i++) {
            if (a[i] > 1) {
                cout<<"NO"<<endl;
                flag = 1;
                break;
            }
        }
        if (flag == 0) {
            cout<<"YES"<<endl;
        }
    }
    return 0;
}
