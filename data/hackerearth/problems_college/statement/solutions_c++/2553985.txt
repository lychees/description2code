#include <bits/stdc++.h>
using namespace std;

int hsh[1000002]={0};


int max(int a,int b)
{
	if(a>b)return a;
	return b;
}

int main()
{
    int T,N;
    cin>>T;
    while(T--)
    {
    	for(int i=0;i<1000002;i++)hsh[i]=0;
    	int N;
    	cin>>N;
    	int arr[N];
    	for(int i=0;i<N;i++)
    	{
    		cin>>arr[i];
    	}
    	sort(arr,arr+N);
    	for(int i=0;i<N;i++)
    	{
    		
    	
    		
    		for(int j=1;j<=sqrt(arr[i]);j++)
    		{
    			if(arr[i]%j==0)
    			{
    				int a=hsh[arr[i]],b=hsh[arr[i]];
    				if(hsh[j]>0)
    				{
    					a=max(hsh[arr[i]],hsh[j]+1);
    				}
    				if((j!=arr[i]/j) && hsh[arr[i]/j]>0)
    				{
    					b=max(hsh[arr[i]],hsh[arr[i]/j]+1);
    					
    				}
    				hsh[arr[i]]=max(a,b);
    				//cout<<j<<" "<<hsh[j]<<endl;
    			}
    		}
    		if(hsh[arr[i]]==0)hsh[arr[i]]=1;
    		//cout<<arr[i]<<" "<<hsh[arr[i]]<<"----"<<endl;
    	}
    	int maxe=0;
    	for(int i=0;i<1000001;i++)
    	{
    		if(hsh[i]>maxe)
    		{
    			maxe=hsh[i];
    		}
    	}
    	cout<<maxe<<endl;
    }
    return 0;
}
