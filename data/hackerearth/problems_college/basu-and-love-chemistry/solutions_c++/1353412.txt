#include<bits/stdc++.h>
using namespace std;
int main()
{
	string s;
	cin>>s;
	long long dp[200002]={0},p;
	int l=s.size();
	dp[0]=(int)s[0]-(int)('0');
	p=dp[0];
	for(int i=1;i<l;i++)
	{
		dp[i]= ((dp[i-1]*10) + ((int)s[i]-(int)('0'))*(i+1))%(1000000007);
		p=(p+dp[i])%(1000000007);
	}
	printf("%lld",p);
	}
