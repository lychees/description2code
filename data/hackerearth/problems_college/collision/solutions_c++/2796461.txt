#include<bits/stdc++.h>
using namespace std;
#define LL long long
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
    
    int n;
    cin>>n;
    int has[100]={0};
    int cu=0;
    for(int i=0;i<n;i++)
    {
        LL ax;
        cin>>ax;
        LL bx = ax%10;
        //cout<<bx<<endl;
       has[bx]++;
    }
    
    for(int i=0;i<10;i++)
    {
    //	cout<<has[i]<<endl;
        if(has[i]>1)cu+=(has[i]-1);
        
    }
    cout<<cu<<endl;
	}
    
}
