#include <iostream>
#include <math.h>
#define ll long long int
using namespace std;
struct points{
	ll x,y;
};
ll gcdHelper(ll a,ll b);
ll gcd(ll a,ll b){
	if(a>b)
		return gcdHelper(a,b);
	return gcdHelper(b,a);
}
ll gcdHelper(ll a,ll b){
	if(b==0)
		return a;
	return gcd(b,a%b);
}
int main()
{
   // cout << "Hello World!" << endl;
   ll n,area,bp,ans;
   	cin >> n;
   	points a[n];
	for(ll i =0;i<n;i++){
		cin >> a[i].x >> a[i].y;
	}
	area=0,bp=0;
	for(ll i=0;i<n;i++){
		area+=(a[i].x*a[(i+1)%n].y-a[(i+1)%n].x*a[i].y);
		bp+=gcd( abs(a[(i+1)%n].x-a[i].x) , abs(a[(i+1)%n].y-a[i].y) )+1;
	}
	bp-=n;
	area=abs(area)/2;
	ans=area-(bp/2)+1;
	cout << ans << endl;
    return 0;
}
