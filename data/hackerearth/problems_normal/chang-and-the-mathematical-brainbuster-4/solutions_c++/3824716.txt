#include <bits/stdc++.h>
using namespace std;
vector <int> a;
vector <int> b;
int main()
{
    int n,x,idx;
    cin>>n;
    for(int i=0;i<n;i++) {
    		cin>>x;
    		a.push_back(x);
    		b.push_back(3*x);
    }
    vector <int> ::iterator it;
    int maxm = 0;
    sort(a.begin(),a.end());
    sort(b.begin(),b.end());
    for(int i=0;i<n;i++){
    	idx = lower_bound(b.begin(),b.end(),a[i])-b.begin();
    //	cout<<idx<<endl;
    	maxm=max(abs(idx-i)+1,maxm);
    }
    
  	cout<<n-maxm<<endl;
    return 0;
}
