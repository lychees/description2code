#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    int t; cin >> t;
    while(t--)
    {
        ll n; cin >> n;
        ll a=1,b=2,c=3;
        while(n>0)
        {
            a=b;b=c;c=a+b;
            n-=c-b-1;
        }
        n+=c-b-1;
        cout << n+b << endl;
    }
    return 0;
}
