#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	vector<pair<int,int>> v;
    	int t1,t2;
    	for(int i=0;i<n;i++)
    	{
    		cin>>t1>>t2;
    		v.push_back(make_pair(t1,t2));
    	}
    	long long int dp[n],ans=-1,flag=0;
    	memset(dp,0,sizeof(dp));
    	sort(v.begin(),v.end());
    	dp[0]=v[0].second;
    	for(int i=1;i<n;i++)
    	{
    		long long int r=v[i].first;
    		long long int h=v[i].second;
//    		cout<<"r="<<r<<" "<<"h="<<h<<"\n";
    		flag=0;
    		for(int j=i-1;j>=0;j--)
    		{
    			if(r>v[j].first && h>v[j].second)
    			{
    				flag=1;
    				dp[i]=max(h+dp[j],dp[i]);
    			}
    		}
    		if(flag==0) dp[i]=h;
    		ans=max(ans,dp[i]);
    	}
    	cout<<ans<<"\n";
    }
    return 0;
}
