#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	string s;
    	cin >> s;
    	int n = s.size();
    	int i=n-1,j,flag = 1;
    	while(flag && i>=0)
    	{
    		for(j=i-1;j>=0;j--)
    		{
    			if(s[j]<s[i])
    			{
    				flag=0;
    				char c;
    				c = s[j];
    				s[j]=s[i];
    				s[i]=c;
    				j++;
    				sort(s.begin()+j, s.end());
    				break;
    			}
    		}
    		if(flag)
    		i--;
    		else
    		break;
    	}
    	if(flag)
    	cout << "no answer" << endl;
    	else
    	cout << s << endl;
    }
    return 0;
}
