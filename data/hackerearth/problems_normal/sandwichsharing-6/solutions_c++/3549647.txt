#include <bits/stdc++.h>
using namespace std;
#define gc getchar_unlocked
void SD(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}

int main()
{
    int t,n,p;
    SD(t);
    while(t--)
    {
     SD(n);SD(p);
     while(p--)
     {
       if(pow(2,p) >= n)
       {
                     n = 3;break;
       }
      if(n & 1)
      n = n/2 + 2;
      else
      n = n/2 + 1;           
      if(n == 3)
      break;
     }
     
     printf("%d\n",n);
    }
    return 0;
}
