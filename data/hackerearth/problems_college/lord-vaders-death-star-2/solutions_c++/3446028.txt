#include<bits/stdc++.h>
#define ll long long
using namespace std;
int arr[1000006];
map<int,int> m;
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>arr[i];
		m[arr[i]]++;
	}
	
	map<int,int>::iterator it;
	int ans=-1;
	for(it=m.begin();it!=m.end();it++)
	{
		if(m[it->first]==1)
		{
			ans=max(ans,it->first);
		}
	}
	
	
	cout<<ans<<endl;
	return 0;
}
