#include <bits/stdc++.h>
using namespace std;

const int NN = 112;
int val[NN],n, k;

int f(int num){
	priority_queue<int,vector<int>,greater<int> > pq;
	for(int i = 0; i < n; ++i){
		if (val[i] > num)
			pq.push((val[i]-num)*5);
		else 
			pq.push((num-val[i])*3);
	}
	
	int ans = 0;
	for(int i = 0; i < k; ++i){
		int add = pq.top();
		pq.pop();
		ans += add;
	}
	return ans;
}

void solve() {
	cin >> n >> k;
	for(int i = 0; i < n; ++i)
		cin >> val[i];
	sort(val,val+n);
	int ans = numeric_limits<int> :: max();
	for(int i = 1; i <= 100; ++i)
		ans = min(ans,f(i));
	cout << ans << "\n";
}

int main()
{
	int t;
	cin >> t;
	while(t--)
		solve();
    return 0;
}
