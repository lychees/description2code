#include <iostream>
using namespace std;

int main()
{
    int size, count;
    cin>>size>>count;
    int num[100000];
    for(int i=0;i<size;i++)
    	cin>>num[i];
   	while(count--)
   	{
   		int val;
   		cin>>val;
   		if(val==0)
   		{
   			int a, b;
   			cin>>a>>b;
   			num[a-1]=b;
   		}
   		else
   		{
   			int a;
   			cin>>a;
   			bool found=false;
   			for(int i=0;i<size;i++)
   			{
   				if(num[i]>=a)
   				{
   					cout<<(i+1)<<endl;
   					found=true;
   					break;
   				}
   			}
   			if(!found)
   				cout<<-1<<endl;
   		}
   	}
    return 0;
}
