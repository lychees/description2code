__author__ = 'Prasad'
import re
import math
import sys


def get_distribution(bananas, lemurs):
    div = float(bananas)/float(lemurs)
    mod = math.modf(div)
    if mod[0] < 0.5:
        return int(math.floor(div))
    else:
        return int(math.ceil(div))


if __name__ == "__main__":
    ans = 1000000
    pattern = "(.*) (.*)"
    regex = re.compile(pattern)
    fd = sys.stdin
    line = fd.readline()
    mt = regex.match(line)
    bananas = int(mt.group(1))
    lemurs = int(mt.group(2))
    dist = get_distribution(bananas, lemurs)
    dist_comp = xrange(0, dist+20)
    lemur_comp = xrange(1, lemurs*200/100)
    for no_of_lemurs in lemur_comp:
        for d in dist_comp:
            no_of_bananas = d * no_of_lemurs
            ans_calculated = abs(lemurs - no_of_lemurs) + \
                abs(bananas - no_of_bananas)
            ans = min(ans, ans_calculated)
    print ans
