#include <iostream>
#include <cstring>
using namespace std;
typedef long long ll;
ll m,w;
ll worth[2001],weight[2001];
ll ans[2001][2001];

int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	scanf("%lld",&m);
    	scanf("%lld",&w);
    	memset(ans,0,sizeof(ans[0][0])*2001*2001);
    	
    	for(int i=0;i<m;i++) scanf("%lld",&weight[i]);
    	for(int i=0;i<m;i++) scanf("%lld",&worth[i]);
    	for(int j=0;j<=w;j++){
    		for(int i=1;i<=m;i++){
    			if (j<weight[i-1]) {
    				ans[j][i] = ans[j][i-1];
    				continue;
    			}
    		//	if (ans[j-weight[i]]+worth[i]>ans[j] && !vis[i]){
    				ans[j][i] = max(ans[j][i-1],ans[j-weight[i-1]][i-1]+worth[i-1]);
    		//		vis[i] = true;
    	//		}
    		}
    	}
    	ll res = 0;
    	//for(int i=0;i<=w;i++) res = max(res,ans[i][m]);
    	res = ans[w][m];
    	printf("%lld\n",res);
    }
    return 0;
}
