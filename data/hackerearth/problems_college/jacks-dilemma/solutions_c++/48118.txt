#include<algorithm>
#include<iostream>
using namespace std;

int main()
     {
          long long t,n,l,sum=0,top,j,stairs;
          char ch;
     cin>>t;
     while(t--)
     {
          sum=0;
        cin>>top>>j>>l;
        cin>>ch;
        stairs = j*10;
        if(ch=='U'){
        sum=(top-l)*5 + 10 + top*5 + 10 + 10;
        if(top==j){sum-=10;}
        }
        else
        {
          if(j<=l){
          sum=l*5+10+10;
          }
          else
          {
               sum= l*5 + 10 + top*5 + top*5 +10+10+10;
               if(top==j){sum-=10;}
          }
        }
        //cout<<sum<<" ";
        if(sum==stairs){cout<<"Lift "<<sum<<endl;continue;}
        if(min(sum,stairs)==sum){ cout<<"Lift "<<min(sum,stairs)<<endl;}
      else if(min(sum,stairs)==stairs){ cout<<"Stairs "<<min(sum,stairs)<<endl;}
       
     }

     return 0;
 }
