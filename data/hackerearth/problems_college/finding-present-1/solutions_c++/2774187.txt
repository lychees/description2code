#include <iostream>
#define MAX 100001
using namespace std;

int main()
{
    int p,q,k,i;
    int h[MAX]={0};
    
    cin>>p;
    for(i=0;i<p;i++){
    	cin>>k;
    	h[k]++;
    }
    
    cin>>q;
    for(i=0;i<q;i++){
    	cin>>k;
    	h[k]--;
    }
    
    for(i=0;i<MAX;i++){
    	if(h[i]!=0){
    		cout<<i<<" ";
    	}
    }
    cout<<endl;
    return 0;
}
