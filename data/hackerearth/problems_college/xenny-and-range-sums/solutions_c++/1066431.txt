#include <iostream>
using namespace std;

int main()
{
   	int N,Q,L,R,num,i;
   	cin>>N>>Q;
   	int arr[N+1],sum[N+1],diff[N+1];
    for(i=0;i<N;i++)
    {
    	cin>>num;
    	if(i==0)
    		sum[0]=diff[0]=num;
    	else
    	{
    		sum[i] = num + sum[i-1];
    		diff[i] = num - sum[i-1];
    	}	
    }
    for(i=0;i<Q;i++)
    {
    	cin>>L>>R;
    	L--;
    	R--;
    	if(L==0)
    	{
    		cout<<sum[R]<<" "<<diff[R]<<endl;
    	}
    	else
    		cout<<sum[R]-sum[L-1]<<" "<<diff[R]+sum[L-1]<<endl;
    }
    return 0;
}
