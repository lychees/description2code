
#include <bits/stdc++.h>
using namespace std;
int x[1000003],y[1000003];
int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		memset(x,0,sizeof(x));memset(y,0,sizeof(y));
		long n,x1,y1,x2,y2;
		long long q;
		cin>>n>>q;
		for(int i=0;i<n;i++)
		{
			cin>>x1>>y1>>x2>>y2;
			long a=min(x1,x2);
			long b=max(x1,x2);
			long c=min(y1,y2);
			long d=max(y1,y2);
			x[a]++;
			x[b+1]--;
			y[c]++;
			y[d+1]--;
		}
		for(int i=1;i<1000001;i++)
		{
			x[i]+=x[i-1];
			y[i]+=y[i-1];
		}
		long i,j;
		while(q--)
		{
			cin>>i>>j;
			if(!i)
				cout<<y[j]<<endl;
			else
				cout<<x[i]<<endl;
		}
	}
	return 0;
}
