#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
    {
    	int64_t g,n,a[3000];
    	cin>>g>>n;
    	int i=0;
    	for(i=0;i<g;i++)
    	 cin>>a[i];
    	 sort(a,a+g);
    	 int64_t s=0;
    	 for(i=0;i<g;i++)
    	   {
    	   	s+=a[i];
    	   	if(s>n)
    	   	  break;
    	   }
    	   cout<<i<<"\n";
    	 
    }
    return 0;
}
