#include<iostream>
#include<string>
using namespace std;
main(){
    string s;
    int n,l,c = 0;
    getline(cin,s);
    cin>>n;
    c = n;
    l = s.length();
    for(int i = 0;i <= l;){
        c = 0;
        for(int j = i;c < n;c++,j++,i++){
            if(i <= l)
            cout<<s[j];
        }
        cout<<endl;
    }
}
