#include<bits/stdc++.h>
using namespace std;
int main(){
	ios::sync_with_stdio(false);
	int i,j,n,t;
	cin>>t;
	while(t--)
	{
		string s1,s2;
		cin>>s1>>s2;
		sort(s2.begin(),s2.end());
		j=0;
		for(i=0;i<s1.size();i++)
		{
			if(s1[i]>s2[j]&&j<s2.size())
			{
				cout<<s2[j];
				j++;
			}
			else
			cout<<s1[i];
		}
		cout<<endl;
	}
	return 0;
}
