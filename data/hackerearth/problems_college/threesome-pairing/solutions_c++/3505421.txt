#include <iostream>
using namespace std;

int main()
{
    int t,n,i,j;
    cin>>t;
    while(t--)
    {
    	int ans=0;
    	cin>>n;
    	int a[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	
    	for(i=0;i<n;i++)
    	{
    		for(j=i+1;j<n;j++)
    		{
    			if((a[i]+a[j])%3==0)
    			{
    			    ans++;
    			}
    		}
    	}
    	cout<<ans<<endl;
    	
    }
    
    return 0;
}
