#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

bool sortOrder(int i, int j) { return i > j; }

int main ()
{
   int x, y, z, max, stalls, cows, used[200], gap[199], unused;

   cin >> max >> stalls >> cows;

   for (x = 0;x < 199;x++)
       gap[x] = 1;

   for (x = 0;x < cows;x++)
       cin >> used[x];

   sort(used, used + cows);

   for (x = 0;x < cows - 1;x++)
       gap[x] = used[x + 1] - used[x];

   sort(gap, gap + cows, sortOrder);

   unused = used[0] + stalls - used[cows - 1] - 1;

   for (x = 0;x < max - 1;x++)
       unused = unused + gap[x] - 1;

   cout << stalls - unused << "\n";

   return 0;
}
