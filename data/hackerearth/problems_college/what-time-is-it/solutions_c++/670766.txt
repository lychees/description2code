#include <iostream>
#include<cmath>
#include<cstdio>
using namespace std;

int main()
{
    string s;
    cin>>s;
    
   double m=(s[3]-'0')*10+s[4]-'0';
   double h=(s[0]-'0')*10+s[1]-'0';
   		
   	//cout<<h<<" "<<m<<endl;
   	if(h==12)
   		h=0;
   	if(m==60)
   		m=0;
   	
   	if(abs(h*30+0.5*m-m*6)>180)
   	printf("%.6lf\n",abs(360-(h*30+0.5*m-m*6)));
   	else
   	printf("%.6lf\n",abs(h*30+0.5*m-m*6));
   	
   	return 0;
}
