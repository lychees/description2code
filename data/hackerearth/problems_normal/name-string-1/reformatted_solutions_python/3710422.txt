l = []

for x in range(int(raw_input())):
    l.append(tuple(sorted(map(int, raw_input().split()))))

for case in l:
    if any(value >= 1 for value in case) and case[1] < case[2]:
        print((case[1]*2)+1)
    elif any(value >= 1 for value in case):
        print(case[1]*2)
    else:
        print(0)
