#include <iostream>
using namespace std;
int giveparity(long int dev)
{
	long int rem,cnt=0;
	while(dev>0)
	{
		if(dev%2)
			cnt++;
		dev=dev/2;
	}
	return cnt;
}
int main()
{
	int t,i,parity;
	long int n;
    cin>>t;
    for(i=1;i<=t;i++)
    {
    	cin>>n;
    	parity=giveparity(n);
    	if(parity%2)
			cout<<"odd"<<endl;
		else
			cout<<"even"<<endl;
    }
    return 0;
}
