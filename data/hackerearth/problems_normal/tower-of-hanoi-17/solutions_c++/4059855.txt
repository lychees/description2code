#include <iostream>
#include <algorithm>
#include <string.h>
#include <stdio.h>
#include <vector>

using namespace std;

int main()
{
    int t;

    cin >> t;

    while(t--)
    {
        int n;

        cin >> n;

        pair <long int, long int> discs[n];

        long int x, y;

        for(int i=0;i<n;i++)
        {
            cin >> x >> y;

            discs[i] = make_pair(x, y);
        }

        sort(discs, discs+n);

        long long int dp[n], ans=0;

        for(int i=0;i<n;i++)
        {
            dp[i]=discs[i].second;

            for(int j=i-1;j>=0;j--)
            {
                if(discs[j].first<discs[i].first && discs[j].second<discs[i].second && dp[j]+discs[i].second>dp[i])
                {
                    dp[i]=discs[i].second+dp[j];
                }
            }

            ans=max(ans, dp[i]);
        }

        cout << ans << endl;
    }

    return 0;
}
