#include <bits/stdc++.h>
using namespace std;
const int maxn =1e5+5 ;
int a[maxn];
int n , k , x  ;
typedef long long ll ;
typedef pair<ll,ll> pi ;
const ll inf = 2000000000 ;
int main()
{
    int t ;
    cin >> t; 
    while( t-- )
    {
    	
    	cin >> n >> k ;
    	priority_queue<pi , vector<pi> , greater<pi> > q;
    	for( int i=0 ; i<n ; ++i )
    	{
    	cin >> x ;
    	q.push(make_pair(x*2,x));
    	}
    	ll ans = 0LL;
        for( int i=0 ; i<k; ++i )
        {
        	int xxx = q.top().first ;
        	int yyy = q.top().second ;
        	q.pop();
        	//cout <<xxx<<" "<< yyy << endl ;
        	ans = xxx+yyy ;
        	q.push( make_pair( xxx+yyy , yyy ) ) ;
        }
        //cout << ans << endl ;
        ll res = 0LL;
        while(!q.empty())
			{
    		pi m=q.top();
    		q.pop();
    		res=max( res , m.first-m.second );
 
			}
		cout << res << endl ;	
    }
    return 0;
}
