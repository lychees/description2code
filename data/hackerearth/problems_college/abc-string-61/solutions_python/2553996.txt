MOD=10**9+7
def pmod(num,exp):
	if exp == 0:
		return 1
	elif exp == 1:
		return num%MOD
	elif exp % 2:
		return num*pmod(num*num%MOD, (exp-1)/2) % MOD
	else:
		return pmod(num*num%MOD, (exp / 2)) % MOD
