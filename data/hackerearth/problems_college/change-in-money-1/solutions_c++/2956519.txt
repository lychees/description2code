#include <bits/stdc++.h>
using namespace std;
long int count(long int S[], int m, int n )
{
    long int table[n+1];
    memset(table, 0, sizeof(table));
    table[0] = 1;
    for(int i=0; i<m; i++)
    {
        for(int j=S[i]; j<=n; j++)
        {
            table[j] += table[j-S[i]];
        }
    }
    return table[n];
}
int main()
{
    int n,m;
    scanf("%d %d",&n,&m);
    int i;
    long int c[m];
    for(i=0;i<m;i++)
    {
    	scanf("%ld",&c[i]);
    }
    printf("%ld\n",count(c,m,n));
    return 0;
}
