#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
   	int test;
   	cin>>test;
   	while(test--) {
   		int r,c;
   		cin>>r>>c;
   		long long cnt_even=0,cnt_odd=0;
   		for(int i=1; i<=r; i++) {
   			if(i&1) {
   				cnt_odd += min(i,c);
   			}
   			else cnt_even += min(i,c);
   		}
   		
   		cout<<cnt_even<<" "<<cnt_odd<<endl;
   	}
    return 0;
}
