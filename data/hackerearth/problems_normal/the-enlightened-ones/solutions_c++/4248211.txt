#include<bits/stdc++.h>

using namespace std;
vector<int>v;
int check(int mid,int K)
	{
		long long int temp=v[0]+mid;
				
		K--;
		int i;
		for(i=1;i<v.size();i++)
			{
				if(temp+mid>=v[i])
					continue;
				if(K==0)
					{
						return 0;
					}
				K--;
				temp=v[i]+mid;
			}
		return 1;
	}
				
int main()
	{
		int N,K;
		cin>>N>>K;
		
		int x;
		for(int i=0;i<N;i++)
			{
				cin>>x;
				v.push_back(x);
			}
		sort(v.begin(),v.end());
		int low=0;
		int high=100000001;
		int mid;
		while(low<=high)
			{
				mid=(low+high)/2;
				if(check(mid,K) && (check(mid-1,K))==0)
					break;
				if(check(mid,K)==1)
					{
						high=mid-1;
					}
				else 
					low=mid+1;
			}
		cout<<mid<<endl;
		return 0;
	}
