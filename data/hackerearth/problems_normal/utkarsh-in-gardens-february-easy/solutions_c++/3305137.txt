#include<bits/stdc++.h>
using namespace std;
#define FOR(i,a,b) for(int i = (a); i <= (b); ++i)
#define FORD(i,a,b) for(int i = (a); i >= (b); --i)
#define RI(i,n) FOR(i,1,(n))
#define REP(i,n) FOR(i,0,(n)-1)
#define mini(a,b) a=min(a,b)
#define maxi(a,b) a=max(a,b)
#define mp make_pair
#define pb push_back
#define st first
#define nd second
#define sz(w) (int) w.size()
typedef vector<int> vi;
typedef long long ll;
typedef long double ld;
typedef pair<int,int> pii;
const int inf = 1e9 + 5;
const int nax = 2005;

int t[nax][nax];
bitset<nax> b[nax];

int main() {
	int n;
	scanf("%d", &n);
	REP(i, n) REP(j, n) {
		scanf("%d", &t[i][j]);
		if(t[i][j]) b[i].set(j, true);
	}
	ll ans = 0;
	REP(i, n) FOR(j, i+1, n-1) {
		bitset<nax> r = b[i] & b[j];
		int x = r.count();
		ans += x * (x - 1) / 2;
	}
	printf("%lld\n", ans / 2);
	return 0;
}
