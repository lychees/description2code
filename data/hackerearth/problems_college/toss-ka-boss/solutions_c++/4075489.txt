#include <iostream>
using namespace std;

int main()
{
    int t;
    long long i,d,sum=0;
    string s;
    cin>>t;
    while(t--){
    	cin>>s;
    	sum=0;
    	d =0;
    	for(i=0;i<s.length();i++)
    	{
    		if(s[i]=='H')
    		{
    			d++;
    			sum+=d;
    		}
    		else
    		{
    			d=0;
    		}
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
