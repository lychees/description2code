#include <iostream>
#include <stdio.h>
using namespace std;
int check_21(int a){
	long b,c;
	while(a>0){
		c = a%10;
		if(c == 1){
			b = a/10;
			b = b%10;
			if(b == 2){
				return 0;
			}
		}
		a = a/10;
	}
	return 1;
}
int main()
{
    int t;
    long n;
    scanf("%d",&t);
    while(t--){
    	scanf("%ld",&n);
    	if(n%21 == 0 || check_21(n) == 0){
    		printf("The streak is broken!\n");
    	}
    	else{
    		printf("The streak lives still in our heart!\n");
    	}
    }
    return 0;
}
