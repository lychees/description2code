const int mod = 1000000007;
const int N = 20005;

int L, C;
ll len[N];
int mx;
int letter[N];

int code(char x){
    return x;  
}

void upd(ll x, ll n, int val){
	x = min(x, 2000ll);
	while(x > L){
		for(int i = x - L;i < x;++ i){
			if(len[i] >= n){
				x = i;
				break;
			}
			n -= len[i];
		}
	}
	if(letter[x] == val)
		return;
    letter[x] = (letter[x] == 0 ? val : -1);
}

int main (){
    ios_base::sync_with_stdio(0);
    cin.tie();
    cin >> L >> C;
    for(int i = 1;i <= L;++ i)
        len[i] = 1;
    for(int i = L + 1;;++ i){
        for(int j = i - L;j < i;++ j)
            len[i] += len[j];
        if(len[i] > 1e15){
            mx = i;
            break;
        }
    }
    for (int i = mx + 1; i <= 5000; i++)
    	len[i] = (ll) 1e18;
    for(int i = 1;i <= C;++ i){
        ll x, y;
        char c;
        cin >> x >> y >> c;
        upd(y, x, code(c));
    }
    int res = 1;
    for(int i = 1;i <= L;++ i){
        if(letter[i] == -1){
            cout << 0 << endl;
            exit(0);
        }
        if(letter[i] == 0)
            res = res * 1ll * 62 % mod;
    }
    cout << res << endl;
    return 0;
}
