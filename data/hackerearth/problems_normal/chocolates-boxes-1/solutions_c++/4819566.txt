#include <iostream>
#define ll long long
using namespace std;
int arr[100001];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		int k;
		cin>>n;
		for(int i=0;i<n;i++)
			cin>>arr[i];
		cin>>k;
		
		int hash[101];
		for(int i=0;i<101;i++)
			hash[i]=0;
		
		ll count;
		count=0;
		for(int i=0;i<n;i++)
		{
			if(k-arr[i]>0 && k-arr[i]<=100)
				count += hash[ k-arr[i] ];
			hash[ arr[i] ]++;
		}
		cout<<count<<"\n";
	}
}
