#include<bits/stdc++.h>
using namespace std;
# define ll long long
bool  check(string s)
{
    char ch=s[0];
    for(int i=1;i<s.size();i++)
        if(ch!=s[i])
        return false;
    return true;
}
int main()
{
    string a,temp;
    cin>>a;
    int i,j,n=a.size(),tt,ans=0;
    for(i=0;i<n;i++)
    {
        for(j=i;j<n;j++)
        {
            tt=j-i+1;
            temp=a.substr(i,tt);
            if(check(temp))
                ans++;
        }
    }
    cout<<ans;
}
