#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		string a,b;
		cin>>a>>b;
		int sum=0;
		if(a[0]>'Z'||a[0]<'A'){
			cout<<0;
			continue;
		}
		if(b[0]>'Z'||b[0]<'A'){
			cout<<0;
			continue;
		}
		for(int i=0;i<min(a.length(),b.length());i++)
		  sum+=(a[i]==b[i]);
	 cout<<sum<<endl;  
	}
    return 0;
}
