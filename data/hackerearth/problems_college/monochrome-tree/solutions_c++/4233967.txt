#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define ll long long
#define mod 1000000007
#define bitcount    __builtin_popcountll
#define pb push_back
vector<int> q[100005];
int a[100005],b[100005],t[100005],c[100005],d[100005];
int dfs(int cur,int par)
{
	int j=0,k=0,ans=0,ans1=0;
	for(int i=0;i<q[cur].size();i++)
	{
		if(q[cur][i]==par)
			continue;
		int x=dfs(q[cur][i],cur);
		ans+=x;
		if(d[q[cur][i]]<c[q[cur][i]]-1)
		{
			ans1+=d[q[cur][i]];
			j++;
		}
		else
		{
			ans1+=c[q[cur][i]];
			k++;
		}
	}
	d[cur]=ans+(a[cur]^b[cur]);
	c[cur]=ans1+j-k+2-(a[cur]^b[cur]);
	//printf("%d %d %d\n",cur,c[cur],d[cur]);
	return min(c[cur],d[cur]);
}
int main()
{
    //freopen("in.txt","r",stdin);
    //freopen("out.txt","w",stdout);
    int n,i,j,k;
    sd(n);
    for(i=0;i<n-1;i++)
    {
    	sd(j);
    	sd(k);
    	q[j].pb(k);
    	q[k].pb(j);
    }
    for(i=1;i<=n;i++)
    	sd(a[i]);
    for(i=1;i<=n;i++)
    	sd(b[i]);
    j=dfs(1,0);
    printf("%d\n",j);
    return 0;
}
