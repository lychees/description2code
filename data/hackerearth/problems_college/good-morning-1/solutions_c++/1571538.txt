#include <iostream>
#include<cstdio>
//#include<ncurses>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n,d,sd=0,count;
    
    cin>>t;
    for(int j=0;j<t;j++){
    	sd=0;
    	count=0;
   		cin>>n>>d;
   		for(;n!=0;n--){
   			sd=sd+(2);
   			count++;
   			if(count==3){
   				n+=2;
   				count=0;
   			}
   			//cout<<sd<<" ";
   		}
   		if(sd>=d)
   			cout<<"Yes\n";
   		else
   			cout<<"No\n";
    }
    return 0;
}
