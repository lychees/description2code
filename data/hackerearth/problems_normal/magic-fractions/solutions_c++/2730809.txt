#include <iostream>
#include <math.h>
using namespace std;
 
int main()
{
    int n;
	cin>>n;
	if(n==1)
	{
		cout<<0;
		return 0;
	}
	int isprime[501];
	//isprime[1] = 0;
	for (int i = 1; i <= 500 ; i++) isprime[i] = 1;
		for (int i = 2; i <= 500; i++) {
			if (isprime[i] == 0)
				continue;
			for (int j = i * 2; j <= 500; j += i)
				isprime[j] = 0;
		}
 
	long long sum[n]; sum[1] = 0; sum[0] = 0;
	long long tot = 0;
	for(int i = 2; i<=n; i++){
		sum[i] = sum[i-1];
		if(isprime[i])
			 sum[i]++;
		tot += pow(2,(sum[i]-1));
 
	}
	cout<<tot;
    return 0;
}
