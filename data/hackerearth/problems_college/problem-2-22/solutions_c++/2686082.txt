#include<bits/stdc++.h>
using namespace std;
typedef struct my
{

    long a,b;
}node;

double foo(long x,long y,long p,long q)
{
    double ax = ((x-p) * (x-p));
    double bx = ((y-q) * (y-q));
    double cx = ax + bx;
    double ex = sqrt(cx);
    return ex;
}

int main()
{

    int t;
    cin>>t;
    while(t--)
    {
        long xc,yc;
        double r;
        cin>>r>>xc>>yc;
        node x[4];
        for(int i=0;i<4;i++)cin>>x[i].a>>x[i].b;
        int f=0;

        for(int i=0;i<4;i++)
        {
            if(foo(x[i].a,x[i].b,xc,yc)>r)
            {
                f=1;
                break;
            }
        }
        if(f)cout<<"No"<<endl;
        else cout<<"Yes"<<endl;
    }

    return 0;
}
