#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <algorithm>
using namespace std;
long long int a,b,c;
long long moves=0;
int main(){
	int t;cin >> t;
	while(t--){
		moves=0;
		cin >>a >> b >> c ; 
		int arr[3];
		arr[0]=a;
		arr[1]=b;
		arr[2]=c;
		sort(arr,arr+3);
		a=arr[0];
		b=arr[1];
		c=arr[2];
		while(1){
			if(c==b+1&&b==a+1){break;}
			if(c-b>b-a){
				moves++;
				long long temp=b;
				b++;
				a=temp;
			}
			else{
				moves++;
				long long temp=b;
				c=b;
				b--;
			}
		}
		cout << moves << endl ; 
	}
	return 0;
}
