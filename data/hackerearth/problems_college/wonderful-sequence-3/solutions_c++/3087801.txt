#include<iostream>
#include<algorithm>
using namespace std;

int a[100009];

int lis(int n)
{
	int len=0;
	for(int i=0;i<n;i++)
    {
        if(a[i]>0)
        {
            if(len==0)
            {
                a[0]=a[i];
                len++;
            }
            else
            {
                int l=0,r=len;
                while(l<r-1)
                {
                    int mid=(l+r)>>1;
                    if(a[mid]<=a[i])
                        l=mid;
                    else
                        r=mid;
                }
                if(a[l]>a[i])
                    a[l]=a[i];
                else
                {
                    a[r]=a[i];
                    if(r==len)
                        len++;
                }
            }
        }
    }
    return len;
}

int main()
{
    int t,n;
    cin>> t;
    while(t--)
    {
        cin>> n;
        for(int i=0;i<n;i++)
        {
            cin>> a[i];
            a[i]=a[i]-i;
        }
        cout<<n-lis(n)<<endl;
    }
    return 0;
}
