def hasLessThanFourDivisors(n):
    count = 0
    i = 1
    while (i*i <= n):
        if (n % i == 0):
            if (i*i == n):
                count += 1
            else:
                count += 2
        i += 1
        if (count == 4):
            return False

    return True


count = int(raw_input())
for i in range(0, count):
    n = int(raw_input())
    if (hasLessThanFourDivisors(n)):
        print 'NO'
    else:
        print 'YES'
