#include <iostream>
#include<cmath>
using namespace std;


int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int arr[n/2], k=1, f=0;
    	arr[0]=2;arr[1]=3;
    	for(int i=4;i<=n;i++)
    	{ 
    		int count=0;
    		for(int j=2;j<=sqrt(i);j++)
    		{
    			if(i%j==0)
    			{
    				count++;
    				break;
    			}
    		
    		}
    		if(count==0)
    		{
    			k++;
    			arr[k]=i;
    		
    		}
    	}
    	int c=0;
    	for(int i=0,j=k ; i<=k && j>=0 ; )
    	{
    		if(n==(arr[i]+arr[j]))
    		{
    			c++;
    			cout<<"Deepa\n";
    			break;
    		}
    		else if(n>(arr[i]+arr[j]))
    			i++;
    		else if(n<(arr[i]+arr[j]))
    			j--;
    	}
    	if(c==0)
    	cout<<"Arjit\n";
    	
    	
    }
    
    
    
    return 0;
}
