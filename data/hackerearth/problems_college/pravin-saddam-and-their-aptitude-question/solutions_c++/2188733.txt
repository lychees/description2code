#include <iostream>
using namespace std;
unsigned long long int a[20];
unsigned long long int fact(int n)
{
	if(a[n]==0)
	{
		a[n] = n*fact(n-1);
		return a[n];
	}
	else
	{
		return a[n];
	}
}
int main()
{
    int t,i;
    for(i=0;i<20;i++)
    {
    	a[i] = 0;
    }
    a[1] = 1;
    scanf("%d",&t);
    while(t--)
    {
    	int a,b;
    	scanf("%d %d",&a,&b);
    	i = a;
    	unsigned long long int s = 0;
    	while(i<=b)
    	{
    		s = (s + fact(i))%10;
    		i++;
    	}
    	printf("%llu\n",s);
    }
    return 0;
}
