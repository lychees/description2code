n = input()
for i in range(0, n):
    k = []
    l = raw_input()
    length = len(l)
    for m in range(0, length):
        if l[m] == '$':
            t = l[m+1:]
            for r in t:
                if ((r.isdigit() == 1)):
                    k = k+[r]
                if (r.isdigit() == 0) and (r != ' '):
                    break
    k = ''.join(k)
    n = str(int(k))
    p = ['$']
    for e in n:
        p = p+[e]
    p = ''.join(p)
    print p
