#include <iostream>
using namespace std;

int main()
{
	int N,Q;
    cin >> N >> Q;
    int arr[N+1];
    for(int i=1; i<=N;++i){
    	cin >> arr[i];
    }
    while(Q--){
    	int type;
    	cin >> type;
    	if (type==0){
    		int x,y;
    		cin >> x >> y;
    		arr[x]=y;
    	} else if(type == 1){
    		int v;
    		cin >> v;
    		for(int i=1;i<=N;++i){
    			if(arr[i]>=v){
    				cout << i;
    				break;
    			}
    			if(i==N){
    				cout << "-1";
    			}
    		}
    		cout << endl;
    	}
    }
    return 0;
}
