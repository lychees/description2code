#include <bits/stdc++.h>

using namespace std;

int main () {

long n;
cin >> n;
vector<long>a;

for (long i = 0;i < n;i++){
    long k;
    cin >> k;
    a.push_back(k);
}

long max = 0;
for (long i = 0;i < n;i++){
    for (long j = i+1;j < n;j++){
        long p = a[i]*a[j];
        if (p > max){
            max = p;
        }
    }
}

cout << max << endl;

return 0;
}
