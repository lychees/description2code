#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	
	cin>>t;
	string s1,s2;
	while(t--)
	{
		s1.clear();
		s2.clear();
		cin>>s1>>s2;
		int i=0;
		int l1=s1.length();
		int l2=s2.length();
		int c=0;
		if(l2<l1)
		l1=l2;
		while(i<l1)
		{
			if(s1[i]==s2[i])
			c++;
			i++;
		}
		printf("%d\n",c);
	}
	return 0;
	
}
