#include<bits/stdc++.h>
#include<algorithm>


using namespace std;
int main() {
     int t;
     cin>>t;
     long ans=0;
     while(t--)
     {
         ans=0;
         int n;long b;
         cin>>n;
         for(int i=0;i<n;i++)
          {
              cin>>b;
             ans=ans^b;
          }
         cout<<ans<<endl;
     }
     return 0;
     }
