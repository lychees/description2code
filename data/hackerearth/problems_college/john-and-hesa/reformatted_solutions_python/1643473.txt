t = int(input())
for i in range(t):
    x, p = raw_input().split()
    x = int(x)
    p = int(p)
    n = 0
    while x % p == 0:
        x = x/p
        n = n+1
    if x == 1:
        x = 0
    print("%d %d" % (n, x))
