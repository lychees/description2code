#include <bits/stdc++.h>
using namespace std;

typedef long long int ll;

int main()
{
    ios_base::sync_with_stdio(false);
    ll t,i,ans,l;
    string str;
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	l = str.length();
    	ans = str[l-1]-'0';
    	for(i=l-2;i>=0;i--)
    	{
    		if(str[i]=='*')
    			ans*=str[--i]-'0';
    		else if(str[i]=='+')
    			ans+=str[--i]-'0';
    		else
    			ans-=str[--i]-'0';
    	}
    	cout<<ans<<endl;
    }
    //cout << "Hello World!" << endl;
    return 0;
}
