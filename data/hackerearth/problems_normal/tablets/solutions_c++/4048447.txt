#include<bits/stdc++.h>
using namespace std;
int main(){
    long long n;
    cin>>n;
    long long a[n];
    for(long long i=0;i<n;i++){
        cin>>a[i];
    }
    long long dp[n],sum,flag=0;
    dp[0] = 1,sum=0;
    for(long long i=1;i<n;i++){
            if(a[i]>a[i-1]){
                dp[i] = dp[i-1] + 1;
            }
            else{
                dp[i] = 1;
            }
    }
    for(long long i=n-2;i>=0;i--){
        if(a[i]>a[i+1]&&dp[i]<=dp[i+1])
        {
            dp[i]=dp[i+1]+1;
        }
    }
    for(long long i=0;i<n;i++){
        sum = sum + dp[i];
    }
    cout<<sum<<endl;
    return 0;
}
