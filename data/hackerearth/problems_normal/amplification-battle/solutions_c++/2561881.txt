int main()
{
	int t;
	ll a,b,c,lo,hi,lo1,hi1,curr;
	sc("%d",&t);
	while(t--){
		sc("%lld %lld %lld",&a,&b,&c);
		curr=1;
		lo=c/b,hi=c;
		if((c%b)!=0){
			lo++;
		}
		while(lo>1){
			if(curr==1){
				lo1=lo/a;
				if((lo%a)!=0){
					lo1++;
				}
				hi1=lo-1;
				curr=0;
			}
			else{
				lo1=lo/b;
				if((lo%b)!=0){
					lo1++;
				}
				hi1=lo-1;
				curr=1;
			}
			lo=lo1;
			hi=hi1;
		}
		
		if(curr==0){
			pr("Bob\n");
		}
		else{
			pr("Adam\n");
		}
	}
	return 0;
}
