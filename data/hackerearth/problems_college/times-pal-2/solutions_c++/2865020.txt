#include <cstring>
#include <cstdio>
#include <iostream>
#define DEBUG if(0)
using namespace std;
int const maxRange = 1e5 + 1;
typedef long long int int32;
char A[	maxRange], B[maxRange],C[maxRange];
string input;

bool Check(char input[])
{
	int len = strlen(input);
	for(int i = 0; i < len/2; ++i)
	{
		if(input[i] != input[len - 1 - i])
		return false;
	}
	return true;
}

int32 Process()
{
	int32 ans = 0;
	DEBUG cout << "HERE" << endl;
	for(int i = 0; input[i] != '\0' ; ++i)
	{
		if(i == 0)
			sprintf(C,"%s%s",B,input.c_str());
		//else if( input[i+1] == '\0' )
		//	sprintf(C,"%s%s",input.c_str(),B);
		else
		{
			sprintf(C,"%s%s%s",input.substr(0,i).c_str(),B,input.substr(i).c_str());
		}
		DEBUG cout << C << endl;
		if(Check(C)) ++ans;
	}
	sprintf(C,"%s%s",input.c_str(),B);
	DEBUG cout << C << endl;
	if(Check(C)) ++ans;
	return ans;
}

int main()
{
	int T;
	cin >> T;
	
	while(T--)
	{
		cin >> input;
		cin >> B;
		cout << Process() << endl;
	}
	return 0;
}
