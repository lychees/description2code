#include <iostream>
#define max 1000000
using namespace std;

int main()
{
    long long n,t;
    int dp[max];
    dp[1]=0;
    for(int i=2;i<=max;i++)
    	dp[i]=1;
    for(int i=2;i<=max;i++){
    	for(int j=i+i;j<=max;j=j+i)
    	dp[j]=dp[j]+1;
    }
    for(int i=2;i<=max;i++)
     dp[i]=dp[i-1]+dp[i];
    cin>>t;
    while(t--){
    	cin>>n;
    	cout<<dp[n]<<"\n";
    }
    
    return 0;
}
