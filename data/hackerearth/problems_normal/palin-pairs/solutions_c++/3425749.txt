#include <bits/stdc++.h>

using namespace std;

map < string, int > p;

int main()
{
    ios::sync_with_stdio(false);cin.tie(0);

    long long n, ans = 0;

    string s;


    cin >>      n          ;

    for(int i = 0; i < n; i++){

        cin >> s;
        ans += p[s];
        reverse(s.begin(), s.end());
        p[s]++;
    }

    cout << ans;

    return 0;
}
