#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	string str[100000];
	int i;
	for(i=0;i<t;i++)
	{
		cin>>str[i];
	}
	sort(str,str+t);
	int c=1;
	for(i=0;i<t;i++)
	{
		if(str[i] == str[i+1])
		{
			c++;
		}
		else
		{
			cout<<str[i]<<" "<<c<<"\n";
			c=1;
		}
	}
}
