inline void in(int &n)
{
    n=0; int ch = getcx(); int sign = 1;
    while(ch < '0' || ch > '9') { if(ch == '-') sign=-1; ch = getcx(); }
    while(ch >= '0' && ch <= '9') { n = (n << 3) + (n << 1) + ch - '0', ch = getcx(); }
    n = n * sign;
}
int main()
{
    int t,n,s,i,j,k,x,y,diff,monk,sum,z[55][5500],a[55],min;
    in(t);
    while(t--)
    {
        //in(n);
        scanf("%d",&n);
        min=INF;
        s=0;
        for(i=1; i<=n; i++)
        {
            //in(a[i]);
            scanf("%d",&a[i]);
            s=s+a[i];
        }
        for(k=1; k<=n; k++)
        {
            s=s-a[k];
            sum=s/2;
            for(i=0; i<=n; i++)
            {
                if(i!=k)
                {
                    for(j=0; j<=sum; j++)
                    {
                        if(i==0 || j==0)
                            z[i][j]=0;
                        else if(j<a[i])
                            z[i][j]=z[i-1][j];
                        else
                            z[i][j]=max(z[i-1][j],a[i]+z[i-1][j-a[i]]);
                    }
                }
                else
                {
                    for(j=0; j<=sum; j++)
                    {
                        z[i][j]=z[i-1][j];
                    }
                }
            }
            x=z[n][sum];
            y=s-x;
            diff=abs(x-y);
            if(diff<min)
            {
                min=diff;
                monk=a[k];
            }
            else if(diff==min)
            {
                if(a[k]<monk)
                {
                    monk=a[k];
                }
            }
            s=s+a[k];
        }
        printf("%d %d\n",min,monk);
    }
    return 0;
}
