#include <iostream>
#define ll long long
#define MAX 1000000007
using namespace std;
void func()
{
	string a,b;
	cin>>a>>b;
	ll myarray[26]={0},myarray2[26]={0};
	ll duttar=1;
	for(ll i=b.length()-1;i>=0;i--)
	myarray2[b[i]-'a']++;
	for(ll i=a.length()-1;i>=0;i--)
	myarray[a[i]-'a']++;
	for(ll i=0;i<26;i++)
	{
		if(myarray[i]-myarray2[i]>=0)
			for(ll j=myarray[i];myarray2[i]>0;j--,myarray2[i]--)
				duttar=(duttar*j)%MAX;
		else if(myarray[i]-myarray2[i]<0)
		{
			cout<<0;
			return ;
		}
	}
	cout<<duttar;
}
int main()
{
	ll n;
	cin>>n;
	while(n--)
	{
		func();
		cout<<endl;
	}
	return 0;
}
