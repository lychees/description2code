#include<bits/stdc++.h>
#define ll long long
#define max(a, b) (a>b?a:b)
using namespace std;

map<int,ll> mp;

ll foo(int n)
{
	if(n==0)
	return 0;
	ll j = mp[n];
	
	if(j==0)
	{
		j=max(n,foo(n/2)+foo(n/3)+foo(n/4));
		mp[n] = j;
	}
	return j;
}
int main()
{
	int n;
	while(scanf("%d",&n)!=EOF)
	{
		mp.clear();
		printf("%lld\n", foo(n));
	}

 	return 0;
}
