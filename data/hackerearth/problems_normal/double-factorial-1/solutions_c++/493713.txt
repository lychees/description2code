#include<iostream>
using namespace std;

static int a[25];

void fact(int n,int m)
{
 int i=1,j,k,l;
  a[0]=1;
while(i<=n)
{     k=24;
    while(a[k]==0)
    {
        k--;
    }
    j=0;
    l=0;
    while(j<=k||l>0)
    {

        a[j]=a[j]*i+l;
        l=a[j]/10;
        a[j]=a[j]%10;
        j++;
    }
    i++;
}
 k=m-1;
    while(a[k]==0)
    {
        k--;
    }
    j=k;
    if(j<0)
        cout<<0;
while(j>=0)
{
    cout<<a[j];
    j--;
}
j=24;
while(j>=0)
    a[j--]=0;
cout<<"\n";
}


int main()
{  int t,n,m,i;
    cin>>t;
     for(i=0;i<t;i++)
     {
         cin>>n>>m;
         if(m==0)
            cout<<0<<"\n";
         else if(n<=1)
            cout<<1<<"\n";
         else if(n>=5)
            cout<<0<<"\n";
         else
         {
             if(n==2)
                fact(2,m);
            else if(n==3)
                fact(6,m);
            else if(n==4)
                fact(24,m);
         }

     }
}
