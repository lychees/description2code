#include <bits/stdc++.h>
#define mod 1000000007
#define MAX 100005
#define ll long long
#define sc(a) scanf("%d",&a)
#define pf(a) printf("%d", a)
using namespace std;
bool vis[100005]={};
vector <int> gr[100005];
	int a[100005]={};
int main()
{

	
	int n,m;
	cin>>n>>m;
	for(int i=0;i<=n;i++)
	{vis[i]=false;
		a[i]=0;
	}
	for(int i=0;i<m;i++)
	{
		int u,v;
		cin>>u>>v;
		gr[u].push_back(v);
		gr[v].push_back(u);
	}
	a[0]=0;
	queue <int> q;
	q.push(0);
	vis[0]=true;
	while(!q.empty())
	{
		int u=q.front();
		q.pop();
		//vis[u]=true;
		for(int i=0;i<gr[u].size();i++)
		{
			if(vis[gr[u][i]]==false)
			{
			q.push(gr[u][i]);
			a[gr[u][i]]=max(a[gr[u][i]],1+a[u]);
			vis[gr[u][i]]=true;
			}
		
		}
	}
	for(int i=1;i<n;i++)
	{
		if(a[i]==0)
		a[i]=-1;
	cout<<a[i]<<"\n";
	}
	
}
