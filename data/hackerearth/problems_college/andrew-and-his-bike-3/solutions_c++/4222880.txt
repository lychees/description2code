#include<bits/stdc++.h>
using namespace std;
vector<int> adj[100001];
static int dp[1000001];
static int ans[100001];
int dfs(int cur,int pr){
	dp[cur]=1;
	for(int i=0;i<adj[cur].size();i++){
		if(adj[cur][i]==pr)
			continue;
	    dp[cur]+=dfs(adj[cur][i],cur);
	}
	return dp[cur];
}
