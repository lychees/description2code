for _ in range(input()):
    treeNos= input()
    a, b, c = map(int, raw_input().split())
    max1, max2 = 0, 0
    for _ in range(treeNos):
        x, y = map(int, raw_input().split())
        if a*x + b*y + c > 0:
            max1 +=1
        if  a*x + b*y + c < 0:
            max2 += 1        
    print max(max1, max2)
