#include <iostream>
using namespace std;

void func(int *score,int x)
{
	if(x>6 && x<9)
    			*score+=2;
    		else if(x>0)
    			*score+=1;
}
int main()
{
    int T,x,y,z;
    cin >> T;
    while(T--){
    	cin >> x >> y >> z;
    	int score = 0;
    	if(x==9)
    		score+=10;
    	else
    	{
    		func(&score, x);
    		if(y==9)
    		{
    			score+=10-3;
    			
    		}
    		else
    		{
    			func(&score,y);
    			if(z==9)
    			{
    				score+=10-6;
    			}
    			else
    			{
    				score = 0;
    			}
    		}
    	}
    	cout << score << endl;
    	
    	
    }
    return 0;
}
