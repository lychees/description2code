#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
#include<algorithm>
using namespace std;

#define gc getchar_unlocked

void sc(int &a)
{
	register int b;
	a = 0;

	b = gc();

	for (; ((b < 48) || (b > 57)); b = gc());
	for (; ((b > 47) && (b < 58)); b = gc()) {
		a = (a << 1) + (a << 3) + b-48;
	}
}

bool comp(const int &a,const int &b)
{
   return (a>b?1:0);
}
int main()
{
   int T;
   sc(T);
   while(T--)
   {
   	int N,i;
   	int mn[1005]={0},wmn[1005]={0};
   	int ans=0;
   	sc(N);
   	for(i=0;i<N;i++)
   	  sc(mn[i]);
   	for(i=0;i<N;i++)
   	  sc(wmn[i]);
   	sort(mn,mn+N,comp);
   	sort(wmn,wmn+N,comp);
   	for(i=0;i<N;i++)
   	{
   		ans+=(mn[i]*wmn[i]);
   	}
   	cout<<ans<<"\n";
   }
    return 0;
}
