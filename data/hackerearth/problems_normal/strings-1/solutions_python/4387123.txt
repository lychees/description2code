T = int(raw_input())
N = []
M = []
for i in range(T):
	x,y = map(int,raw_input().split())
	N.append(x)
	M.append(y)

for i in range(T):
	if N[i] == M[i]:
		print "YES"
	elif (N[i] == 2 and M[i] == 4) or (N[i] == 4 and M[i] == 2):
		print "YES"
	else:
		print "NO"
