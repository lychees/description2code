#include <iostream>
using namespace std;
int main()
{
int N;
cin>>N;
for(int i=0;i<N;i++)
{
int n,m;
cin>>n;
cin>>m;
for(int j=0;j<n;j++)
{
double small=1;
int si=0;
for(int k=0;k<m;k++)
{
double d1,d2,d3,p;
cin>>d1;
cin>>d2;
cin>>d3;
p=(1-(d1/100))*(1-(d2/100))*(1-(d3/100));

if(p<small)
{
small=p;
si=k+1;
}
}//kkk
if(j<(n-1))
cout<<si<<" ";
else
cout<<si<<"\n";
}//jjjj
}//iii


return 0;
}
