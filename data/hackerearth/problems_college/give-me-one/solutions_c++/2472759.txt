#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

const int lmt = 20000005;
int dp[lmt];
const int oo = 1000000000;

int main(){
    int t,n;
    for(int i=1;i<lmt;i++)
        dp[i]=oo;
    dp[1]=0;
    for(int i=1;i<lmt;i++){
        if(i+1<lmt)
            dp[i+1]=min(dp[i+1],dp[i]+1);
        if(2*i<lmt)
            dp[2*i]=min(dp[2*i],dp[i]+1);
        if(3*i<lmt)
            dp[3*i]=min(dp[3*i],dp[i]+1);
    }
    scanf("%d",&t);
    while(t--){
        scanf("%d",&n);
        printf("%d\n",dp[n]);
    }
    return 0;
}
