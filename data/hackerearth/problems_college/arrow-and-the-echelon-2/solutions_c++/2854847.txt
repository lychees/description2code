#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t,n,i,j,k;
    scanf ("%d",&t);
    while (t--)
   	{
   		scanf ("%d",&n);
   		int k=0,l=0;
   		if (n%3==0)
   			k=n;
   		else
   		{
   			l=0;
   			while ((n-l)%3)
   			{
   				if (n>=(l+5))
   					l+=5;
   				else
   					break;
   			}
   			k=n-l;
   			if (k%3)
   			{
   				k=0;
   				l=0;
   			}
   		}
   		if (k!=0 || l!=0)
   		{
   			for (i=0;i<k;++i)
   				printf ("5");
   			for (i=0;i<l;++i)
   				printf ("3");
   			printf ("\n");
   		}
   		else
   			puts ("-1");
   	}
    return 0;
}
