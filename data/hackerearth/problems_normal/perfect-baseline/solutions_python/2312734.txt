T=int(raw_input())
for _ in xrange(T):
	N,K=(int(e) for e in raw_input().split())
	lis=[]
	for i in xrange(N):
		lis.append(raw_input())
	stringg=''
	for i in xrange(K):
		l=[]
		for e in lis:
			l.append(e[i])
		sl=sorted(l)
		mediann=len(sl)/2
		if len(sl)%2==0:
			stringg+=sl[mediann-1]
		else:
			stringg+=sl[mediann]
	print stringg
