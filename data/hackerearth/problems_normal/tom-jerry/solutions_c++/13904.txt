#include<cstdio>
#include<iostream>
#include<algorithm>
#include<utility>
#include<cstring>
#include<cmath>

using namespace std;

int dx[] = {1,-1,0,0};
int dy[] = {0,0,1,-1};
int cnt =0;

int a[100][100];
bool visited[105][105];

void dfs(int x , int y, int n){

    if((x == n-1) && (y == n-1))cnt++;
    if(x >= n || x <0 )return;
    if(y >= n || y< 0)return ;
    if(visited[x][y]) return;
    if(a[x][y]==1)return;
	
    	
	visited[x][y] = true;
    for(int i=0; i< 4; i++){
    	dfs(x+dx[i],y+dy[i],n);
  	}
    visited[x][y] = false;

}
int main(){

    int n,i,j;
    scanf("%d",&n);

    for( i = 0 ; i < n ; i++){
		for( j = 0 ; j < n ; j++){
			scanf("%d",&a[i][j]);
        }
    }
    dfs(0,0,n);
    cout<<(int)ceil(cnt/2.0)<<endl;
    return 0;
}
