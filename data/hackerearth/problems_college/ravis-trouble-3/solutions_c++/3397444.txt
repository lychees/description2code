#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		char a[10010],b[10010];
		scanf("%s",a);
		scanf("%s",b);
		ll cnt=0,j=0;
		for(int i=0;i<strlen(b);i++)
		{
			while(j<strlen(a)){
				if(b[i]==a[j])
				 {cnt++;j++;break;}
				 else{j++;}
			}
		}
		if(cnt==strlen(b))
		printf("YES\n");
		else printf("NO\n");
	}
	
}
