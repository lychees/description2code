#include<iostream>
#include<cstdio>
inline int readInt(){
	int number = 0;
	char c = getchar_unlocked();
	while(c<'0')
		c = getchar_unlocked();
	while(c>='0' && c<='9'){
		number = (number<<3)+(number<<1)+c-'0';
		c = getchar_unlocked();
	}
return number;
}

long long int gcd( long long int a, long long int b )
{
  if ( a==0 ) return b;
  return gcd(b%a,a);
}
using namespace std;
int main(){
long long int t,n,k,i,g,c,tmp,num,den;
cin>>t;
while(t--){
    n=readInt();
    k=readInt();
    c=0;
    num=0;
    for(i=0;i<n;i++){
        tmp=readInt();
        c+=(tmp==k);
    }
    if(c>2)
        num=(c*(c-1)*(c-2));
    den=n*(n-1)*(n-2);
    g=gcd(num,den);
    cout <<num/g<<"/"<<den/g<<endl;
}
return 0;
}
