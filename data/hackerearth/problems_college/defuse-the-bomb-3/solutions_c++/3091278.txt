#include<bits/stdc++.h>
#define gc getchar
using namespace std;
typedef long long ll;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
int main()
{
    int t;
    ll b,r,l,i,j,num;
    string a;
    cin>>t;
    while(t--)
    {
        cin>>a>>b;
        ll cd=0,x=b;
        l=a.length();
        while(x!=0)
        {
            r=x%10;
            cd++;
            x/=10;
        }
        for(i=0;i<=l-cd;i++)
        {
            num=0;
            for(j=0,x=i;j<cd;j++,x++)
            {
                num=num*10+(a[x]-48);
            }
            cout<<num*b<<'\n';
        }
    }
    return 0;
}
