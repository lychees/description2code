#include <bits/stdc++.h>
using namespace std;
long long ar[1000001];
int main()
{
    int t;
    cin>>t;
    while(t--){
    	long long n,m,sum=0,cnt=0,f=0;
    	cin>>n>>m;
    	for(int i=0;i<n;i++) cin>>ar[i];
    	sort(ar,ar+n);
    	for(int i=n-1;i>=0;i--){
    		sum+=ar[i];
    		cnt++;
    		if(sum>=m){
    			f=1;
    		
    		break;}
    	}
    	if(f)
    	cout<<cnt<<endl;
    	else cout<<"-1"<<endl;
    }
}
