def fast_power(a, n):
    result = 1
    value = a
    power = n
    while power > 0:
        if power % 2 == 1:
            result = (result * value) % 1000000007
        value = (value * value) % 1000000007
        power = power//2
    return result


t = int(input())
for _ in range(t):
    n = int(input())
    print(fast_power(2, n)-1)
