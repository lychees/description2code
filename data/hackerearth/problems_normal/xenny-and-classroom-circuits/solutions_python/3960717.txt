test_cases_left = int(raw_input())
while test_cases_left:
	test_cases_left -= 1
	n = int(raw_input())
	sequence = raw_input()
	boys_count = 0
	girls_count = 0

	for i in range(0, 2*n, 2):
		if sequence[i] == 'G':
			girls_count += 1
		elif sequence[i] == 'B':
			boys_count += 1
	
	print min(girls_count, boys_count)
