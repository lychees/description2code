#include <bits/stdc++.h>
#define lli long long int
using namespace std;
lli minify(int n, bool max){
	lli a=0;
	lli r;
	lli i=1;
	while(n>0){
		r = n%10;
		if(!max){
			if(r==6)
				r=5;
		}
		else{
			if(r==5)
				r=6;
		}
		a = a+r*i;
		n = n/10;
		i*=10;
	}
	return a;
}
int main()
{
	lli a,b;
	cin>>a>>b;
	cout<<(minify(a,false)+minify(b,false));
	cout<<" "<<(minify(a,true)+minify(b,true))<<"\n";
    return 0;
}
