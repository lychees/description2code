#include <iostream>
#include <stdio.h>
using namespace std;

int main() {
	// your code goes here
	int t;
	scanf("%d", &t);
	while(t--){
		int n;
		scanf("%d", &n);
		int c=0;
		int a[n], b[n];
		for(int i=0;i<n;i++){
			scanf("%d", &a[i]);
		}
		for(int i=0;i<n;i++){
			b[a[i]-1]=i+1;
		}
		for(int i=0;i<n;i++){
			if(a[i]!=b[i]){
				c=1;
			}
		}
		if(c==1){
			printf("not inverse\n");
		}
		else{
			printf("inverse\n");
		}
	}

}
