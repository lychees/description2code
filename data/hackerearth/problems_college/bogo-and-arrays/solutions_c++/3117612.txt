#include<bits/stdc++.h>
using namespace std;
#define i64 int64_t

int main(){
   int tt;
   cin>>tt;
   while(tt--){

   	int n,k;
   	cin>>n>>k;
   	vector <int> arr(n);
   	for(int i=0;i<n;i++){
   		cin>>arr[i];
   	}
   	i64 ans=0;
   	int i=0;
   	while(i<n){
   		if(arr[i]>k){
   			i++;continue;
   		}
   		int count=0;
   		while(i<n and arr[i]<=k){
   			count++;
   			i++;
   		}
   		ans+=(count*(count+1))/2;
   		
   	}
   	//print(arr);
   	cout<<(i64)n*(n+1)/2-ans<<endl;
   }


}
