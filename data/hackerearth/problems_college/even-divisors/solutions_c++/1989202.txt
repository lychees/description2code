#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        unsigned long long int i,n;
        cin>>n;
        int z=0;

            for(i=1;i<=sqrt(n);i++)
            {
                if(n%i==0)
                {
                    if(i%2==0)
                    {
                        z++;
                    }
                    if((n/i)%2==0 && (n/i)*(n/i)!=n)
                    {
                        z++;
                    }
                    
                }
            }

        cout<<z<<endl;
    }


    return 0;
}
