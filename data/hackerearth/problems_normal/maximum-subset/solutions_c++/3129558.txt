#include<bits/stdc++.h>
using namespace std;
#define ll long long
const int mod = 1000000007 ;
long long int  bigmod ( long long a, int p, int m ){
    if ( p == 0 )   return 1 ;
    if ( p % 2 ){
        return ( ( a % m ) * ( bigmod ( a, p - 1, m ) ) ) % m;
    }
    else{
        long long c = bigmod ( a, p / 2, m );
        return ( (c%m) * (c%m) ) % m;
    }
}
int n , m ;
ll arr[20] ;
ll ret ;
void f(int idx1 , int c1 , int idx2 , int c2 , ll m1 , ll m2){
    if(c1 > m || c2 > m || idx2 < 0 || idx1 >= n)    return ;
//    cout << "HELLO" << ' ' << idx1 << ' ' << idx2 << ' ' << c1 << ' ' << c2 << endl;
    if(idx2 < idx1){
        if(c1 != c2)    return ;
        ll u = bigmod(m1 , mod-2 , mod) ;
        ret = max( ret , (m2*u)%mod) ;
//        cout << m1 << ' ' << m2 << endl ;
        return ;
    }
    if(c1 == c2){
        f(idx1 + 1 , c1 + 1 , idx2 , c2 , (m1*arr[idx1])%mod , m2) ;
        f(idx1 + 1 , c1 , idx2 , c2 , m1 , m2) ;
    }
    else if(c1 > c2){
        f(idx1 , c1 , idx2 - 1 , c2 + 1 , m1 , (m2*arr[idx2])%mod) ;
        f(idx1 , c1 , idx2 - 1 , c2 , m1 , m2 ) ;
    }
    return ;
}

int main(){
    int T , i ;
    cin>>T ;
    while(T--){
        cin >> n ;
        for(i = 0 ; i < n ; i++)    cin >> arr[i] ;
        sort(arr , arr + n) ;
        ret = 0 ;
        m = n/2 ;
        f(0,0,n-1,0,1,1) ;
        cout << ret << endl ;
    }
}
