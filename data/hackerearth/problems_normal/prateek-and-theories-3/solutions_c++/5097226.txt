#include<bits/stdc++.h>
using namespace std;
map<int,int> mymap;
map<int,int>:: iterator it;
int main()
{
    ios::sync_with_stdio(false);
    int i,n,t;
    cin>>t;
    while(t--)
    {
        cin>>n;
        int start[n+3],end[n+3];
        for(i=0;i<n;i++){
            cin>>start[i]>>end[i];
        }
        for(i=0;i<n;i++)
        {
            mymap[end[i]]--;
            mymap[start[i]]++;
        }
        int ans=0,temp=0;
        for(it=mymap.begin();it!=mymap.end();it++)
            {
                temp+=(it->second);
                if(temp>ans)
                    ans=temp;
            }
        cout<<ans<<"\n";
        mymap.clear();
    }
    return 0;
}
