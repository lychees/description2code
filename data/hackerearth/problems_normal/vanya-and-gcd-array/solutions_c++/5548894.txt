#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
int n;
#define M 1000000007
int a[502];
int dp[502][102][102];
ll func(int i, int gc,int last)
{
	int &ans=dp[i][gc][last];
	if(i==n)
	if(gc==1)
		return 1;
	else
		return 0;
	else
	{
		if(ans!=-1)
			return ans;
		if(a[i]>last)
		{
			return ans=(func(i+1,gc,last)+func(i+1,__gcd(gc,a[i]),a[i]))%M;
		}
		else
		{
			return ans=func(i+1,gc,last)%M;
		}
	}
	
}
int main()
{
	scanf("%d",&n);
	int i;
	memset(dp,-1,sizeof(dp));
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	ll ans=func(0,0,0);
	cout<<ans<<endl;
	return 0;
}
