string X;
int aCost, bCost, L;

int solve(int curr,  int i ){
	if( i>(L-1-i) )
		return curr;

	if( X[i]=='a' && X[L-1-i]=='a' )
		return solve( curr, i+1 );

	if( X[i]=='b' && X[L-1-i]=='b' )
		return solve( curr, i+1 );

	if( (X[i]=='a' && X[L-1-i]=='b') || (X[i]=='b' && X[L-1-i]=='a') )
		return -1;

	if( X[i]=='/' && X[L-1-i]=='/' )
		return solve( curr+2*(min(aCost, bCost)), i+1);

	if( (X[i]=='/' && X[L-1-i]=='a') || (X[i]=='a' && X[L-1-i]=='/') )
		return solve( curr+aCost, i+1 );
	
	if( (X[i]=='/' && X[L-1-i]=='b') || (X[i]=='b' && X[L-1-i]=='/') )
		return solve( curr+bCost, i+1 );
}

int main(int argc, char** argv) {
	int T=SI;
	while( T-- ){
		cin>>X;
		L=X.size();
		cin>>aCost>>bCost;
		cout<<solve(0,0)<<endl;
	}
	return 0;
}
