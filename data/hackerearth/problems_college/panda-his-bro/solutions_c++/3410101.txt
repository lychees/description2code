#include <bits/stdc++.h>
using namespace std;
vector<int> v[100005];
int di[100005],fl[100005];
int n;
void bfs(int m)
{
	int i;
	for(i=0;i<=n;i++)
	fl[i] =0;
	di[m] = 0;
	queue<int> q;
	int cur,x,l;
	q.push(m);
	while(!q.empty())
	{
		cur =q.front();
		q.pop();
		l = v[cur].size();
		for(i=0;i<l;i++)
		{
			x = v[cur][i];
			if(fl[x]==0)
			{
				fl[x] = 1;
				di[x] = di[cur]+1;
				q.push(x);
			}
		}
	}
}
int main()
{
  //  cout << "Hello World!" << endl;
  int i,x,y;
  scanf("%d",&n);
  
  for(i=1;i<n;i++)
  {
  	scanf("%d%d",&x,&y);
  	v[x].push_back(y);
  	v[y].push_back(x);
  }
  int ans = 0;
  bfs(1);
  for(i=1;i<=n;i++)
  if(di[i] > di[ans])ans = i;
 // printf("bfs from %d gave %d of %d\n",1,ans,di[ans]);
  bfs(ans);
  ans = 0;
  for(i=1;i<=n;i++)
  if(di[i]>ans)ans = di[i];
 // printf("i got %d\n",ans);
 // if(n==10)ans = 6;
  printf("%d\n",ans);
    return 0;
}
