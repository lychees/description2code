    int BIT[27][MAXN];
    void update(int loc,int x,int vl){
    for(int i=x;i<=MAXN;i+=(i&-i))
    BIT[loc][i]+=vl;
    }
    int query(int loc,int x){
    int ans=0;
    for(int i=x;i>=1;i-=(i&-i))
    ans+=BIT[loc][i];
    return ans;
    }
    int remocu(int loc,int x){
    int mid,lb=1,ub=MAXN,ans;
    while(lb<=ub){
    mid=((lb+ub)>>1);
    if(query(loc,mid)>=x){
    ans=mid;
    ub=mid-1;
    }
    else
    lb=mid+1;
    }
    update(loc,ans,-1);
    return ans;
    }
