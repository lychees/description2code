#include<bits/stdc++.h>

using namespace std;

int main()
{
    int t,i;
    string s;
    scanf("%d",&t);
    while(t--)
    {
        cin>>s;

        long int n,count=0,sum=0;
        n=s.size();

        for(i=0; i<n; i++)
        {
            if(s[i]=='H')
            {
                count++;
                sum+=count;
                while(s[i+1]=='H')
                {
                    count++;
                    sum+=count;
                    i++;
                }
                if(s[i+1]=='T')
                    count=0;
            }

        }
         cout<<sum<<endl;
    }
    return 0;
}
