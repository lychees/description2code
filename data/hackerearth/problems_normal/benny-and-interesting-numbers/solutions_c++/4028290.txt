#include <bits/stdc++.h>

#define REP(i, n)	for(ll i=0;i<n;i++)
#define REPN(i, n)	for(ll i=1;i<=n;i++)
#define SET(i, n)	memset(i, n, sizeof(i))

#define MAX			319227
#define INF			(1 << 29)
#define EPS 		1e-9
#define MOD			10000007
#define pb			push_back

using namespace std;

typedef long long ll;
typedef pair<ll, ll>	pii;

ll n, P[MAX];
vector<ll>primes;

void init(){
	ll sq = sqrt(MAX);
	for(ll i=2;i<=sq;i++){
		if(!P[i]){
			for(ll j=i*i;j<MAX;j+=i){
				P[j] = true;
			}
		}
	}
	primes.pb(2);
	for(ll i=3;i<MAX;i+=2) if(!P[i]) primes.pb(i);
}

ll solve(ll x){
	ll sq = sqrt(x);
	for(ll i=0;i<primes.size()&&primes[i] <= sq;i++){
		if(x % primes[i] == 0){
			ll cnt = 0;
			while(x % primes[i] == 0) cnt++, x /= primes[i];
			if(cnt != 1) return false;
			sq = sqrt(x);
		}
	}
	return true;
}

int main(){
	init();
	ll tc, x;
	scanf("%lld", &tc);
	while(tc--){
		scanf("%lld", &x);
		ll ret = solve(x);
		if(ret) printf("Yes\n");
		else printf("No\n");
	}
	return 0;
}
