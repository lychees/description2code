#include <iostream>
//#include<conio.h>
using namespace std;
int find(long long int n)
{
	long long int sumeven=0,sumodd=0,step=0;
	while(n!=1)
	{
		//cout<<n<<endl;
		if(n%2==0)
		{
			 sumeven+=n;
		     n=n/2;
		}
		else
		{
			sumodd+=n;
		    n=n*3+1;
		}
		step++;
	}
	if((sumeven%step == (sumodd+1)%step ))
	       return 2;
	else
	      return (sumeven%step < (sumodd+1)%step ) ;
}
int main()
{
    long long int n,test,k;
    cin>>test;
    while(test--)
    {
        cin>>n;
        k=find(n);
        if(k==2)
            cout<<"Tie"<<endl;
        else if(k)
           cout<<"Odd Rules"<<endl;
        else
            cout<<"Even Rules"<<endl;
            
    }
    //getch();
    return 0;                                                
}
