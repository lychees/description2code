#include<iostream>
using namespace std;
int main()
{
	int a[1000],n;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int j=1;j<n;j++)
	{
		int key = a[j];
		int i=j-1;
		while(i>=0 and a[i]>key)
		{
			a[i+1]=a[i];
			i--;
		}
		a[i+1]=key;
	}
	int m=0;
	while(m<n)
	{
		int op = a[m];
		int j=m,k=m;
	while(j<n)
	{
		a[j]=a[j]-op;
		if(a[j]==0)
			m++;
		j++;		
	}
	cout<<(n-k)<<endl;
}
	return 0;
}
