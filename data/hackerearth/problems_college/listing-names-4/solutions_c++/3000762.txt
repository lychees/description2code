#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
        ll i, t;
        map <string, ll> mp;
        map <string, ll>:: iterator it;
        string str;
        scanf("%lld", &t);
        
        for (i = 0; i < t; i++) {
                cin >> str;
                it = mp.find(str);
                if(it != mp.end()) {
                        mp.find(str)->second += 1;
                } else {
                        mp[str] = 1;
                }
        }
        
        for (it = mp.begin(); it != mp.end(); it++) {
                cout << it->first << " " << it->second << endl;
        }
        return 0;
}
