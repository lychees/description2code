#include <iostream>
using namespace std;
void sum()
{
	int n;
	cin>>n;
	int i,arr[n],s[n],p[n],flag=0;
	for(i=0;i<n;i++)
	{
		cin>>arr[i];
	}
	s[0]=arr[0];
	for(i=1;i<n;i++)
	{
		s[i]=s[i-1]+arr[i];
	}
	p[0]=s[n-1];
	for(i=1;i<n;i++)
	
		
		{
			p[i]=p[i-1]-arr[i-1];
		}
		/*	for(i=0;i<n;i++)
	{
		cout<<s[i]<<" "<<p[i]<<" ";
	}*/
	
	for(i=1;i<n-1;i++)
	{
		if(s[i-1]==p[i+1])
		flag=1;
		
	}
		if(flag==0)
		cout<<"NO\n";
	else
	cout<<"YES\n";
	
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    sum();
    return 0;
}
