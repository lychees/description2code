def main():
	test_cases = int(raw_input())
	for _ in xrange(test_cases):
		n = int(raw_input())
		list1 = raw_input().split()
		list2 = []
		for i in xrange(n):
			list2.append(int(list1[i])%5)
		
		XOR = list2[0]
		for i in xrange(1, n):
			XOR = XOR ^ list2[i]

		if XOR == 0:
			print "Andrew"
		else:
			print "Aniruddha"
	
	return 0

if __name__ == "__main__":
	main()
