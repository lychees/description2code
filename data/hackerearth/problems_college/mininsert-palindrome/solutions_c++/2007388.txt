#include <cstdio>
#include <cstring>
#include<iostream>
using namespace std;

int min(int a, int b)
{   return a < b ? a : b;  }

int findMinInsertionsDP(string str, int n)
{

    int table[n][n], l, h, gap;


    memset(table, 0, sizeof(table));


    for (gap = 1; gap < n; ++gap)
        for (l = 0, h = gap; h < n; ++l, ++h)
            table[l][h] = (str[l] == str[h])? table[l+1][h-1] :
                          (min(table[l][h-1], table[l+1][h]) + 1);


    return table[0][n-1];
}


int main()
{
  
 string S;
  int t,l;
    scanf("%d",&t);
    while(t--)
    {
    	
    	scanf("%d",&l);
    	cin>>S;
    	printf("%d\n", findMinInsertionsDP(S,l));
    }
    return 0;
}
