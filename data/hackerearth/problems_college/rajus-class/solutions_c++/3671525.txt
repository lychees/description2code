
#include<iostream>
#include<string>
using namespace std;
int main()
{
    long long i,tmp,j,r,t,n,s,small,pos;
    
      
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	long long ar[n];
    		string s[n],tmps;
        for(i=0;i<n;i++)
        {
        	cin>>ar[i];cin>>s[i];
		}
		cin>>r;
		for(i=0;i<n;i++)
		{   
		    small=ar[i]; pos=i;
			for(j=i+1;j<n;j++)
			{
				if(ar[j]<small)
				 {
				   small=ar[j];
				  pos=j;}
			}
			tmp=ar[pos];
			ar[pos]=ar[i];
			ar[i]=tmp;
			tmps=s[pos];
			s[pos]=s[i];
			s[i]=tmps;
			
			
			
		}
	
	     cout<<s[r-1]<<endl;    	
	}
	return 0;
}
