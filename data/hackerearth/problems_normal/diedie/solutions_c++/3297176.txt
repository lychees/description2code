const long long MOD = 1E9 + 7;
long long bin(long long x, long long y) {
    if (y == 0) {
        return 1;
    }
    long long u = bin(x, y/2);
    if (y % 2 == 0) {
        return u*u % MOD;
    }
    else {
        return u*u % MOD * x % MOD;
    }
}
void solve() {
    long long n;
    cin>>n;
    long long ans = bin(2, n - 1);
    //cerr<<ans<<endl;
    ans = bin(ans, MOD - 2);
    cout<<ans<<endl;
}

        solve();
