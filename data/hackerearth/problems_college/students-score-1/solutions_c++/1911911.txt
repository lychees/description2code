#include <iostream>
using namespace std;

int arr[100001];

int calculate(int n){
	int maxSum=0,sumEnd=0;
	for(int i=0;i<n;i++){
		sumEnd=sumEnd+arr[i];
		if(sumEnd<0){
			sumEnd=0;
			continue;
		}
		if(maxSum<sumEnd)
			maxSum=sumEnd;
	}
	return maxSum;
}

int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    	cin>>arr[i];
    cout<<calculate(n);
    
    return 0;
}
