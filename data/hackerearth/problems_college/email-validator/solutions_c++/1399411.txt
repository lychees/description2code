#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    cin >> t;
    string s;
    while(t--) {
    	cin >> s;
    	int length = s.length();
    	s = " " + s;
    	int index = s.find_first_of("@");
    	bool mainflag = true;
    	if (index > 17 || index <= 1)
    		mainflag = false;
    	else {
    		bool flag = true;
    		for (int i=1;i<index;i++) {
    			if (!(s[i] >= 'a' && s[i]<='z')) {
    				if (!(s[i] >= 'A' && s[i]<='Z')) {
    					if (!(s[i] >= '0' && s[i]<='9')) {
    						if (!(s[i]=='_'))
    							flag = false;
	    				}
	    			}
	    		}
	    		if (flag == false) {
	    			mainflag = false;
	    			break;
	    		}
	    	}
	    	if (mainflag) {
	    		int index1 = s.find_first_of(".",index);
	    		if (index1==-1 || (index1-index-1)>16)
	    			mainflag = false;
	    		else {
	    			bool letter = false, flag = true;
	    			for(int i=index+1;i<index1;i++) {
	    				if (!(s[i] >= 'a' && s[i]<='z')) {
		    				if (!(s[i] >= 'A' && s[i]<='Z')) {
		    					if (!(s[i] >= '0' && s[i]<='9')) {
		    						if (!(s[i]=='-'))
		    							flag = false;
			    				}
			    			}
			    			else
			    			    letter = true;
			    		}
			    		else 
			    		    letter = true;
			    		if (!flag) {
			    			mainflag = false;
	    					break;
			    		}
	    			}
	    			if (mainflag && letter) {
	    				if ((s.length()-1-index1)<=0 || (s.length()-1-index1)>10) {
	    					mainflag = false;
	    				}
	    				for (int i = index1+1;i<s.length() && mainflag;i++) {
	    					if (!(s[i] >= 'a' && s[i]<='z')) {
	    						mainflag = false;
	    						break;
	    					}
	    				}
	    			}
	    			else 
	    			    mainflag = false;
	    		}
	    	}
    	}
    	if (mainflag)
    		cout << "YES\n";
    	else cout << "NO\n";
    }
    return 0;
}
