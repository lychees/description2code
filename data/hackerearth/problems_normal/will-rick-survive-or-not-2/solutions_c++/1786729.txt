#include<iostream>
#include<algorithm>
using namespace std;

inline void ip(int &n)
  {
     char c=getchar_unlocked();
     n=0;
     while(c>'9'||c<'0')
      c=getchar_unlocked();
     while(c<='9'&&c>='0')
      {n=(n<<3)+(n<<1)+c-'0';
      c=getchar_unlocked();}
  }
  
  inline void ip(long int &n)
  {
     char c=getchar_unlocked();
     n=0;
     while(c>'9'||c<'0')
      c=getchar_unlocked();
     while(c<='9'&&c>='0')
      {n=(n<<3)+(n<<1)+c-'0';
      c=getchar_unlocked();}
  }

int main()
 {
  int t;
  ip(t);
  while(t--)
  {  long int n;
     ip(n);
      long int a[n];
      for(long int i=0;i<n;i++)
        ip(a[i]);
      sort(a,a+n);
      int f=0,j=0;
      long int i;
      for(i=0;i<n;i++)
      {
      	if(i%6==0 && i!=0)
      	  j++;
      	if((i+j+1)>a[i])
      	  {f=1;break;}
      }
      if(f)
       printf("Goodbye Rick\n%ld\n",i);
      else printf("Rick now go and save Carl and Judas\n");
  }
 return 0;
 }
