#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str,temp,temp1;
		cin>>str;
		long long int ans=0;
		/*	for odd	*/
		for(int i=0;i<str.length();++i)
		{
			for(int j=i,k=i;j<str.length()&&k>=0;++j,--k)
			{
				if(str[j]==str[k])
					ans=ans+(j-k+1)*(j-k+1);
				else
					break;
			}
		}
		/*	for even	*/
		for(int i=0;i<str.length()-1;++i)
		{
			for(int j=i+1,k=i;j<str.length()&&k>=0;++j,--k)
			{
				if(str[j]==str[k])
					ans=ans+(j-k+1)*(j-k+1);
				else
					break;
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
