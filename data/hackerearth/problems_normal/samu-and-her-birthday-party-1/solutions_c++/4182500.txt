#include <iostream>
#include <string>
using namespace std;
int a[505];
int main()
{
	//freopen("input.txt", "r", stdin);
	int t;
	cin >> t;
	while (t--){
		int n, k;
		cin >> n >> k;
		string s;
		for (int i = 0; i < n; i++){
			cin >> s;
			int t1=0;
			for (int j = 0; j <k; j++){
				t1 *= 2;
				t1 += (s[j] - '0');				
			}
			a[i] = t1;
		}
		
		int cntMin = 15;
		for (int i = 0; i < (1 << k); i++){
			int flag = 0;
			int cnt = 0;
			int ii = i;
			while (ii != 0){
				if (ii % 2 == 1)cnt++;
				ii /= 2;
			}
			//cout << "i " << i << " cnt " << cnt << endl;
			if (cnt >= cntMin)continue;

			for (int j = 0; j < n; j++){
				if ((a[j] & i) == 0){
					flag = 1; break;
				}
			}
			if (flag == 0)cntMin = cnt;
		}
		cout << cntMin << endl;

	}
}
