#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int s=0;
        int tp;
        while(n--)
        {
            cin>>tp;
            if(tp%2==1)
            s++;
        }
        if(s%2==0)
        cout<<"Yes"<<endl;
        else
        cout<<"No"<<endl;
    }
    return 0;
}
