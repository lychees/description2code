	#include<iostream>
	using namespace std;
	#define MAX 101
	int Arr[MAX][MAX];

	int Ans=0;
	 
	

	int main()
	{
		int i,j,t;	
		int temp;
		int c;
		int n;
		cin>>t;
		
		for(int k=0;k<t;k++)
		{
			n=0;
			c=0;
			Ans=0;
			temp=0;
			cin>>n;
			cin>>c;
			std::fill(Arr[0] + 0, Arr[99] + 100, 0);
			for(int i=1;i<=n;++i)
				for(j=1;j<=n;++j)
					cin>>Arr[i][j];

			//int result=maxValue(n,n);

			for(int i=n; i>0;i--)
			{
				for(int j=n;j>0;j--)
					{			
						temp = Arr[i+1][j]>Arr[i][j+1] ? Arr[i+1][j]: Arr[i][j+1];
						Arr[i][j] = Arr[i][j] + temp;
					}
			}
			Ans = Arr[1][1] - c;
		
			if(Ans>=0)
			{
			//	Ans=result-c;
				cout<<Ans<<endl;
			}
			else
			{
				//Ans=-1;
				cout<<"-1"<<endl;
			}
		}

		return 0;
	}
