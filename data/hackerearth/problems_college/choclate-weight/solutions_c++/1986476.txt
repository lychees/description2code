#include <iostream>
#include<bits/stdc++.h>
#define ll long long int 
using namespace std;
ll A[100000];
int binarySearch(ll arr[], ll l, ll r, ll x)
{
   if (r >= l)
   {
        ll mid = l+(r-l)/2;
        if (arr[mid] == x)  return mid;
        if (arr[mid] > x) return binarySearch(arr, l, mid-1, x);
        return binarySearch(arr, mid+1, r, x);
   }
   return -1;
}

int search(ll A[] , ll N , ll p)
{
	int i=0,j=N-1;
	while(i<=j){
	if(A[i]+A[j]==p && i!=j){
		return 1;
	}
	else if(A[i]+A[j]>p){
		j--;
	}
	else i++;
	}
	return -1;
}

int main()
{
    //cout << "Hello World!" << endl;
    ll N;
    ll t;
    cin>>N;
    for(ll i=0;i<N;i++){
    	cin>>A[i];
    }
    sort(A,A+N);
    cin>>t;
    ll p;
    while(t--){
    	cin>>p;
    	if(binarySearch(A,0,N-1,p)!=-1 || search(A,N,p)!=-1){
    		cout<<"I Like You\n";
    	}
    	else cout<<"I Hate You\n";
    }
    return 0;
}
