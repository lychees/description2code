#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    string s;
    vector<string> v;
    while(t--)
    {
        v.clear();
        scanf("%d");
        getline(cin,s);
        //cout<<s<<"\n";
        stringstream ss(s);
        string buf,temp;
        while(ss>>buf)
            v.push_back(buf);
        int i=0,j=v.size()-1;
        while(i<j)
        {
            temp=v[i];
            v[i]=v[j];
            v[j]=temp;
            i++;j--;
        }
        for(int i=0;i<v.size();i++)
            cout<<v[i]<<" ";
        printf("\n");
    }
    return 0;
}
