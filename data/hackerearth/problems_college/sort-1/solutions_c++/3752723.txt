#include <iostream>
#include<algorithm>
using namespace std;

int main()
{    long long int t,n;
     cin>>t;
     while(t--){
     	cin>>n;
     	long long int a[n];
     	for(int i=0;i<n;i++)
     	    cin>>a[i];
     	    
     	   sort(a,a+n);
     	   for(int i=0;i<n;i++)
     	    cout<<a[i]<<" ";
     	    
     	   cout<<"\n";
     }
    
    return 0;
}
