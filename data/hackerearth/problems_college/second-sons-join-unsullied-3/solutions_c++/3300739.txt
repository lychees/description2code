#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    while(n--)
    {
    	string str;
    	cin>>str;
    	int u=0,s=0;
    	for(int i=0;i<str.length();i++)
    	{
    		if(str[i]=='U')
    			u++;
    		if(str[i]=='S')
    			s++;
    	}
    	int m = str.length();
    	if(abs(u-s)>1)
    		cout<<"-1"<<endl;
    	else
    	{
    		if(u==s)
    		{
    			int swaps1=0, swaps2=0;
    			for(int i=0;i<m-1;i+=2)
    			{
    				if(str[i]=='S')
    					swaps1++;
    				else
    					swaps2++;
    				if(str[i+1]=='U')
    					swaps1++;
    				else
    					swaps2++;
    			}
    			
    			if(swaps1<swaps2)
    				cout<<swaps1/2<<endl;
    			else
    				cout<<swaps2/2<<endl;
    		}
    		else if(u-s==1)
    		{
    			int swaps1=0;
    			for(int i=0;i<m;i+=2)
    			{
    				if(i==m-1)
    				{
    					if(str[i]=='S')
    					swaps1++;
    				}
    				else {
    					
    				if( str[i]=='S')
    					swaps1++;
    				if(str[i+1]=='U')
    					swaps1++;
    				}
    				
    			}
    			cout<<swaps1/2<<endl;
    		}
    		else
    		{
    			int swaps1=0;
    			for(int i=0;i<m;i+=2)
    			{
    				if(i==m-1)
    				{
    					if(str[i]=='U')
    						swaps1++;
    				}
    				else 
    				{
    					if( str[i]=='U')
    					swaps1++;
    					if(str[i+1]=='S')
    					swaps1++;
    				}
    				
    			}
    			cout<<swaps1/2<<endl;
    		}
    	}
    }
    return 0;
}
