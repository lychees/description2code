from collections import Counter

kases = int(raw_input())

for _k in xrange(kases):
    d = Counter('abcdefghijklmnopqrstuvwxyz')
    d.update(raw_input())
    sort_list = sorted(d.items(), key=lambda (k, v): (v, k), reverse=True)
    d.clear()
    temp = sort_list[1][1]
    outstr = tempstr = ''

    for ele in sort_list:
        if ele[1] != temp:
            outstr = tempstr + outstr
            temp = ele[1]
            tempstr = ''
        tempstr += ele[0] + ' '
    outstr = tempstr + outstr

    print outstr
