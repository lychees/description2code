#include <iostream>
#include <string.h>
#include <stdio.h>
using namespace std;

int main()
{
    int T;
    scanf("%d", &T);
    while(T--) {
    	string str;
    	cin >> str;
    	int n = str.length();
    	for(int i=0; i<n; i++) {
    		if(str[i] == 'B') {
    			if(i-1 >= 0 && str[i-1] == '-')
    				str[i-1] = 'X';
    			if(i-2 <n && str[i-2] == '-')
    				str[i-2] = 'X';
    		}
    		else if(str[i] == 'S') {
    			if(i-1 >= 0 && str[i-1] == '-')
    				str[i-1] = 'X';
    			if(i+1 <n && str[i+1] == '-')
    				str[i+1] = 'X';
    		}
    	}
    	
    	//cout << str << endl;
    	
    	int park = 0;
    	for(int i=0; i<n; i++)
    		if(str[i] == '-')
    			park++;
    	printf("%d\n", park);
    }
    return 0;
}
