#include<cstdio>
#include<cstdio>
#include<cmath>
#include<iostream>
using namespace std;
int reverse(int number)
{
	int temp = number;
	int reversed = 0;
	while(temp)
	{
		reversed = reversed*10;
		reversed = reversed + temp%10;
		temp = temp/10;
		
	}
	return reversed;
}
int eliminate(int num)
{
	int tempo = num;
	while(tempo%10 == 0)
	{
		tempo = tempo/10;
	}
	return tempo;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int a,b;
		cin>>a>>b;
		int temp1 = eliminate(a);
		int temp2 = eliminate(b);
		int sum = reverse(temp1)+reverse(temp2);
		int p = eliminate(sum);
		 p = reverse(p);
		int q = eliminate(p);
		cout<<q<<"\n";
		
	}
}
