#include <bits/stdc++.h>
#include <string.h>
using namespace std;

#define gt(n) scanf("%d", &n)
#define gts(n) scanf("%s", s)
char s[100000];
char p[100000];
int main()
{
    int t = 1;

    gt(t);
    getchar();

    while (t--) {
        cin >> s;
        cin >> p;
        int i;
        int j;

        int n = strlen(s);
        int m = strlen(p);
        int flag = 0;

        //printf("%s %s", s, p);

        if (n >= m) {
                j = 0;
                for (i = 0; i < m; i++) {
                    while (j < n && p[i] != s[j]) {
                        j++;
                    }

                    if (j >= n)
                        break;
                    j = j + 1;
                }

                if (i >= m)
                    flag = 1;
        }
        if (flag)
            printf("YES\n");
        else
            printf("NO\n");
    }

    return 0;
}
