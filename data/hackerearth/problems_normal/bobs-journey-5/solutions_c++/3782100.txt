#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
    	int a[123]={0},flag=0;
    	cin>>n;
    	string s;
    	for(int i=0;i<n;i++)
    	{
    		cin>>s;
    		a[s[0]]++;
    	}
    	for(int i=96;i<123;i++)
    	{
    		if(a[i]>1)
    		{
    			flag=1;
    			cout<<"NO\n";
    			break;
    		}
    	}
    	if(!flag)
    	cout<<"YES\n";
    }
    return 0;
}
