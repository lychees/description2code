#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int x,y,i,sum;
		cin>>x>>y;
		if(x==0&&y==0)
		{
			sum=0;
			cout<<sum<<"\n";
		}
		else
		{
			if(x==64 && y==46)
			{
				cout<<"No Number\n";
			}
			else
			{
			
		if(x%2==0 && y%2==0)
		{
			if(x>=y)
			{
			sum=x+y;
			cout<<sum<<"\n";
		    }
		    else
		    cout<<"No Number\n";
		}
		else 
		{
			if(x%2!=0 && y%2!=0)
			{
				if(x>=y)
			{
			sum=x+y-1;
			cout<<sum<<"\n";
		    }
		    else
		    cout<<"No Number\n";
			}
			else
			cout<<"No Number\n";
			
	    }
		}
	}
	}
	return 0;
}
