#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;

#define MAX 10000010
#define ull unsigned long long

ull f[2][MAX], res;
char s[MAX];
int n;

int main(){
	scanf("%s", s);
	n = strlen(s);
	if (s[0] == '1')
		f[1][0] = 1;
	for (int i = 1; i < n; i++){
		if (s[i] == '0'){
			f[0][i] = f[0][i - 1] + f[1][i - 1];
			f[1][i] = f[1][i - 1];
		}
		else{
			res += f[0][i - 1];
			f[0][i] = f[0][i - 1];
			f[1][i] = f[1][i - 1] + 1;
		}
	}
	printf("%llu", res);
	return 0;
}
