#include <iostream>
using namespace std;

int main()
{
	long long int a,b,c,i,j,k,ans=0;
    cin>>a;
    cin>>b;
    cin>>c;
    for(i=6;i<=100000;i++)
    {
    	for(j=-3;j<=100000;j++)
    	{
    		if((a*i+b*j+c)==0)
    		{
    			ans=1;
    			break;
    		}
    	}
    	if(ans)
    	break;
    	
    }
    if(ans)
    cout<<i<<" "<<j;
    else
    cout<<"-1"<<endl;
    
    return 0;
}
