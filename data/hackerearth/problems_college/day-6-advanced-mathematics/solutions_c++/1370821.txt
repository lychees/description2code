#include <iostream>
using namespace std;

int main()
{
    long long t,k,i,flag;
    string s;
    cin>>t;
    while(t--){
    	cin>>s;
    	flag=0;
    	k=s.length();
    	for(i=0;i<k-1;i++){
    		if(s[i]==s[i+1]){
    			flag=1;
    			break;
    		}
    	}
    	
    	if(flag==0)
    		cout<<"KISS\n";
    	else
    		cout<<"SLAP\n";
    }
    return 0;
}
