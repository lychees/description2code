#include <iostream>
using namespace std;

int main()
{
	int t, i, l, len, temp, j, f=0;
	string s;
    cin>>t;
    while(t--)
    {
    	f=0;
    	cin>>s;
    	for(i=0;i<s.length()/2;i++)
    	{
    		if(s[i]!=s[s.length()-i-1])
    		{
    			f=1;break;
    		}
    	}
    	if(f==0)
    		cout<<s.length()<<endl;
    	else
    	{
    		l=s.length();len=s.length();j=0;temp=-1;
    		for(i=0;i<s.length();i++)
    		{
    			if(s[i]!=s[len-j-1])
    			{
    				l=l+i-temp;
    				temp=i;
    				j=0;
    			}
    			else j++;
    		}
    		cout<<l<<endl;
    	}
    }
    return 0;
}
