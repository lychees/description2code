#include <bits/stdc++.h>

typedef long long int li;

#define min(a,b) a<=b?a:b
#define max(a,b) a>=b?a:b
#define Fi(i, st, ft) for(i = st;i <= ft; ++i)
#define Fd(i, st, ft) for(i = st;i >= ft; --i)
#define pb(a, b) a.push_back(b)
#define mod 1000000007
#define MAXN 1500000
#define MX 1450000
#define inf 1000000000000000000ll
const int maxBufSize = (10000000);

using namespace std;

li t , n , s , l;
li arr[100005] , sum[100005];

int main()
{
	li i , j , a , b , c , ans;
	scanf("%lld" , &t);
	while(t--) {
		scanf("%lld %lld %lld" , &s , &n , &l);
		memset(arr , 0 , sizeof(arr));
		//memset(sum , 0 , sizeof(sum));
		Fi(i , 1 , n) {
			scanf("%lld" , &a);
			arr[a] = 1;
		}
		Fi(i , 1 , s) {
			if(arr[i] == 1) {
				sum[i] = sum[i-1]+1;
			} else {
				sum[i] = sum[i-1];
			}
		}
		ans = 0;
		Fi(i , 1 , s) {
			if(arr[i] == 1) {
				b = 0;
				if(s-i > l) {
					b += sum[i-1];
				} else {
					if(i > l) {
						b += sum[i-1]-sum[l];
						a = min((i-(l+1)) , l);
						b += sum[a];
					}
				}
				//cout << i << " " << b << "\n";
				ans += b;
			} 
		}
		printf("%lld\n" , ans);
	}
	return 0;
}
