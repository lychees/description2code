#include<iostream>
using namespace std;
#define MAX 1000000007
long long int arr[100007];
void calci()
{
	arr[1]=1;
	arr[2]=2;
	arr[3]=4;
	for(int i=4;i<100001;i++)
	{
		arr[i]=(arr[i-1]%MAX+arr[i-2]%MAX+arr[i-3]%MAX)%MAX;
	}
}
int main()
{
	long int test;
	calci();
	cin>>test;
	while(test--)
	{
		int num;
		cin>>num;
		cout<<arr[num]<<endl;
	}
    return 0;
}
