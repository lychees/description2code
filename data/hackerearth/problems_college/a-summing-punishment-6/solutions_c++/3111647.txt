#include <iostream>
using namespace std;

int main()
{
   int n,q,l,r,i,j,sum=0;
   cin>>n;
   int a[n];
   for(i=1;i<=n;i++)
   {
   		cin>>a[i];
   }
   cin>>q;
   for(j=1;j<=q;j++)
   {
   		cin>>l;
   		cin>>r;
   		sum=0;
   		for(i=l;i<=r;i++)
   		{
   		   
   			sum=sum+a[i];
   			
   		}
   		cout<<sum<<endl;
   		
   }
    return 0;
}
