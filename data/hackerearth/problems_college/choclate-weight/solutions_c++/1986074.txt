#include<bits/stdc++.h>
using namespace std;

int binary(int a[],int l,int h,int x)
{
	if(l<=h)
	{
			int mid=(l+h)/2;
			if(a[mid]==x)
			{
				return mid;
			}
			else if(a[mid]>x)
			{
				return binary(a,l,mid-1,x);
			}
			else
			{
				return binary(a,mid+1,h,x);
			}
	}
	return -1;
}
int main()
{
	int n,i;
	cin>>n;
	int a[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	
	sort(a,a+n);
	int q,x,j;
	cin>>q;
	for(j=0;j<q;j++)
	{
		cin>>x;
		int flag=0;
		if(binary(a,0,n-1,x)!=-1)
		{
			
			cout<<"I Like You\n";
			//cout<<binary(a,0,n-1,x)<<endl;
			continue;
		}
		
			int start=0,end=n-1,sum;
			while(start<end)
			{
				sum=a[start]+a[end];
				if(sum>x)
				{
					end--;
				}
				else if(sum<x)
				{
					start++;
				}
				else
				{
					printf("I Like You\n");
					flag=1;
					break;
				}
			}

			if(flag)
				continue;
					
					printf("I Hate You\n");
		
	}
		return 0;
}
