//chotu and palindromic strings
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int test,flag1,flag2,i;
	string a,b,c,d;
	cin >> test;
	while(test--)
	{
		cin >> a;
		cin >> b;
		flag1 = 0;flag2 = 0;
		c = a + b;
		d = b + a;
		for(i = 0;i < (c.size()/2);i++)
		{
			if(c[i] != c[c.size() -1 -i])
			{
				flag1 = 1;
				break;
			}
		}
		for(i = 0;i < (d.size()/2);i++)
		{
			if(d[i] != d[d.size()-1-i])
			{
				flag2 = 1;
				break;
			}
		}
		if(flag1 == 0 || flag2 == 0)
		{
			cout << "YES" << endl;
		}
		else 
		{
			cout << "NO" << endl;
		}		
	}
	return 0;
}
