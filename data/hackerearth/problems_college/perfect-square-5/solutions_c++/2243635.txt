#include<iostream>
#include<cmath>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        unsigned long long n,f=0,c=0,diff=0,d=0;
        cin>>n;
        f=floor(sqrt(n));
        c=ceil(sqrt(n));
        //cout<<c<<endl;
        diff=n-(f*f);
        d=c*c-n;
        //cout<<diff<<" "<<d<<endl;
        if(diff==0)
            cout<<"YES"<<endl;
        else if(diff<d)
            cout<<"-"<<diff<<endl;
        else
            cout<<"+"<<d<<endl;
    }
    return 0;
}
