#include <bits/stdc++.h>
#define pii pair<long long, int>
#define iq first
#define ind second
using namespace std;
 
int main() {
	
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	
	int i, c, p, n;
	cin>>c>>p>>n;
	int n_iq[n], p_iq[p];
	stack<int> s[c];
	for(i=0; i<n; i++) {
		cin>>n_iq[i];
		s[i].push(n_iq[i]);
 
	}	
	for(i =0 ; i<p; i++) {
		cin>>p_iq[i];
	}
 
	priority_queue<pii, vector<pii>, greater<pii> > pq;
	for( i=0; i<c; i++) {
		if(i<n) { 
			pq.push(pii(n_iq[i],i));
		} else {
			pq.push(pii(0, i));
		}
	}
 
	for(i=0; i<p; i++) {
		pii t = pq.top();
		pq.pop();
		cout<<t.ind+1<<" ";
		if(s[t.ind].size()==0) {
			pq.push(pii(p_iq[i], t.ind));
		} else {
			pq.push(pii((s[t.ind].size()+1)*(p_iq[i]+s[t.ind].top()), t.ind));
		} 
		s[t.ind].push(p_iq[i]);
	}
 
	cout<<endl;
 
	return 0;
}
