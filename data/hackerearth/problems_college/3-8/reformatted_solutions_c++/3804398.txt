#include <iostream>
using namespace std;
string str, nstr;
int main() {
  int t;
  cin >> t;
  while (t--) {

    cin >> str;
    int i;
    nstr = str[0];
    for (i = 1; i < str.size(); ++i) {
      if (*nstr.begin() <= str[i])
        nstr = str[i] + nstr;
      else
        nstr += str[i];
    }
    cout << nstr << endl;
  }
}
