#include<bits/stdc++.h>
using namespace std;
int codeStreak(string s) {
	int c , ans = 0;
	for(int i = 0; s[i]; i++) {
		if(s[i] == 'C') {
			c = 0;
			while(s[i] == 'C' && s[i]) {
				c++;
				i++;
			}
			ans = max(ans, c);
		}
	}
	ans = max(ans, c);
	return ans;
}
int main()
{
	int n, tans = 0, fans = 0;
	cin >> n;
	vector<string> v(n);
	string full;
	for(int i = 0; i < n; i++) {
		cin >> v[i];
		full += v[i];
		tans = max(tans, codeStreak(v[i]));
	}
	fans = max(fans, codeStreak(full));
	cout << tans << " " << fans << "\n";
}
