#include <iostream>
using namespace std;

int main()
{
    int n, t ; 
    cin>>t ;
    while( t-- ){
      
        cin>> n ;
        if( n % 12 == 0 )
           cout<<n/12<<endl ;
        else if( n % 10 == 0  ){
        	if( n > 60 )
        	   cout<<(n/60)*5+ (n-(n/60)*60)/10<<endl ; 
        	else
        	    cout<<n/10<<endl ;
        }
        else{
        	int least = 9999999 ;
            for( int i = 1 ; i <= n/12; i ++ ){
            	
            	for( int j = 1 ; j <= n/10 ; j ++ ){
            		
            		
            		if( i * 12 + j* 10 == n  )
            		      if(  i+j < least  )
            		         least = i+j ;
            		if( i * 12 + j* 10 > n )
            		    break ;
            	}
            }
            
            if( least != 9999999 )
               cout<<least<<endl ;
            else
               cout<<-1<<endl ;
          	
        }
    }
    return 0;
}
