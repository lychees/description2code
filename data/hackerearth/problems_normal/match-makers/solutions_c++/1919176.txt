#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t,n,j,B[10000],G[10000],i,count;
    scanf("%d",&t);
    while(t-->0)
    	{
    		count=0;
    		scanf("%d",&n);
    		for(i=0;i<n;i++)
    			{
    				scanf("%d",&B[i]);
    			}
    		sort(B,B+n);
    		for(i=0;i<n;i++)
    			{
    				scanf("%d",&G[i]);
    			}
    		sort(G,G+n);
    		j = n-1;
    		for(i=0;i<n;i++)
    			{
    				if(B[j]%G[i]==0 || G[i]%B[j]==0)
    					{
    						count++;
    					}
    				j--;
    			}
    		printf("%d\n",count);
    		
    	}
    return 0;
}
