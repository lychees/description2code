#include <iostream>
#include <algorithm>
#include <string>
#include <math.h>
using namespace std;
double x[1001],y[1001];
bool isgood(int i, int j)
{double slope = (y[j]-y[i])/(x[j]-x[i]);
	for(int k = i+1;k<j;k++)
	{if(y[k] > y[i]+(x[k]-x[i])*slope)
	{return false;
		
	}}
	return true;}
	
int main()
{int N,ans=0;
cin>>N;

for(int i = 0;i<N;i++)
{cin>>x[i]>>y[i];}

for(int i = 0;i<N;i++)
{for(int j = i+1;j<N;j++)
{if(isgood(i,j))
{ans++;}}}
cout<<ans;
    return 0;
}
