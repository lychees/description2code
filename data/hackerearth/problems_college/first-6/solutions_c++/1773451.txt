#include <bits/stdc++.h>
using namespace std;

void hack(string s) {
    int a[11000];
    for(int i = s.length()-1; i >= 0; i--) {
        a[i] = s[i] ^ (0xf0fffff0 | (0xffffffff & 0x0f00000f));
    }
    for(int i = 0; i < s.length(); i++) {
        a[i] = a[i] ^ (0xf0ff0ff0 | (0xffffffff & 0x0f00f00f));
    }
    for(int i = s.length()-1; i >= 0; i--) {
        a[i] = a[i] ^ (0xf0ff00f0 | (0xffffffff & 0x0f00ff0f));
    }
    for (int i = 0; i < s.size(); ++i) cout << a[i] << " ";
}

int main() {
    string s;
    cin >> s;

    hack(s);
    return 0;
}
