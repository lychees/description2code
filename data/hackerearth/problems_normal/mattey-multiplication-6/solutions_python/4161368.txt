test=input('')
cnt=0

def pos_set_bit(num):
	ll=[]
	cnte=0
	while num:
		if num&1==1:
			ll.append(cnte)
		num>>=1
		cnte+=1
	return ll[::-1]

while test>cnt:
	n,m=map(int,raw_input('').split(' '))
	ll=pos_set_bit(m)
	
	ans=''
	for i in range(len(ll)):
		
		if i!=len(ll)-1:
			ans+=("("+str(n)+"<<"+str(ll[i])+")"+" "+"+"+" ")
		else:
			ans+=("("+str(n)+"<<"+str(ll[i])+")")

	print ans
	test-=1
