#include <stdio.h>
#define MOD 10000007
long long a[1000006];
int main()
{
    int t,n,i;
    a[1]=a[2]=1;
    a[3]=2,a[4]=4;
    for(i=5;i<1000006;i++)
        a[i]=(a[i]+a[i-1]+a[i-3]+a[i-4])%MOD;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        printf("%lld\n",a[n]);
    }
    return 0;
}
