#include <iostream>
#include<stdio.h>
#include<math.h>
#include<algorithm>
#include<string.h>
#include<stack>
#include<vector>
#include<set>
#include<queue>
#include<list>
#define ll long long int
using namespace std;
int main()
{
    ll n,m;
    cin>>n>>m;
    ll num,i;
    priority_queue <ll> que;
    for(i=0;i<n;i++)
    {
        cin>>num;
        que.push(num);
    }
    ll ans=0;
    for(i=0;i<m;i++)
    {
        num=que.top();
        ans=ans+num;
        que.pop();
        num=num-1;
        que.push(num);
    }
    cout<<ans<<endl;
}
