#include <iostream>
using namespace std;

int main()
{
    int n,i;
    cin>>n;
    
    long long int b[101],sum=0,a[101]={0};
    
    for(i=0;i<n;i++){
    	cin>>b[i];
    }
    
    a[0]=b[0];
    for(i=0;i<n;i++){
    	sum=b[i]*(i+1);
    
    	for(int j=0;j<i;j++){

    		sum=sum-a[j];
    	}
    	a[i]=sum;
    	cout<<sum<<" ";
    }
    
    return 0;
}
