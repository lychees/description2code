fact = [0]*251
count = [0]*251


def pre():
    fact[0] = 1
    fact[1] = 1
    for i in range(249):
        fact[i+2] = (i+2)*fact[i+1]
    for i in range(251):
        n = fact[i]
        while (n > 0):
            if (n % 10 == 4 or n % 10 == 7):
                count[i] += 1
            n /= 10


pre()
