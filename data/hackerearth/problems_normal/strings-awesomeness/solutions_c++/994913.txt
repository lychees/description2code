int main() {

	int m = strlen(a);
	char *S = new char[5000001];
	scanf("%s", S);

	char t[12];

	rer(i, 0, n-m) {
		rep(j, m)

		sort(t, t + m);
		bool ok = true;
		rep(j, m)
			ok &= a[j] == t[j];
		sum[i+1] = sum[i] + ok;
	}

			ans = 0;
		else
			ans = sum[R - m + 1] - sum[L];
		printf("%d\n", ans);
	}
