#include <iostream>
using namespace std;
#define int long long
const int mod=1e9+7; 
int dp[200001]; 
main()
{
    string a; cin >> a; int ans=0; 
    for (int g=0; g<a.length(); g++)
    {
    	if (g==0) {dp[g]=a[g]-'0';ans+=dp[g];ans%=mod; continue;}
    	dp[g]=(10*dp[g-1])+(g+1)*(a[g]-'0');
    	dp[g]%=mod; 
    	ans+=dp[g]; ans%=mod; 
    }
    cout << ans; 
    return 0;
}
