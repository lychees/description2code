N=int(raw_input())
for i in range(N):
    l=raw_input().split()
    n=int(l[0])
    l=l[1:]
    li=[]
    for x in l:
        li.append(int(x))
    
    sl=[]
    
    li=sorted(li)
    
    f=0
   
    k=len(li)-1
    
    for j in range(n/2):
        sl.append(li[f]+li[k])
        f=f+1
        k=k-1
        
    sl=sorted(sl)
    print sl[len(sl)-1]-sl[0]
