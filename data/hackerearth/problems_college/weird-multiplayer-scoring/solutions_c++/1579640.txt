#include<iostream>
#include<algorithm>
#include<map>
#include<string>
using namespace std;
map <string, long long> Mymap;
map<string, long long> :: iterator it;
int main()
{

	int n, m;
	cin >> n >> m;
	string name, winner;
	long long points, tot;
	tot = 0;
	while (m--)
	{
		cin >> name >> points;
		if (Mymap.find(name) == Mymap.end())
		{
			Mymap.insert(make_pair(name, points));
			if (points > tot)
			{
				tot = points;
				winner = name;
			}
		}
		else
		{
			it = Mymap.find(name);
			it->second = it->second + points;
			if (it->second > tot)
			{
				tot = it->second;
				winner = name;
			}
		}
	}

	cout << winner;
	return 0;
}
