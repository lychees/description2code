#include<bits/stdc++.h>
using namespace std;
int main()
{
  int t;cin>>t;
  while(t--)
  {
    unsigned int n1, n2;cin>>n1>>n2;
    if(n1 == n2) 
      printf("%u\n",((n1%2 == 0)?2*n1:(2*n1 - 1)));
    else if(n1 == n2 + 2)
      printf("%u\n",((n1%2 == 0)?n1 + n2:(n1 + n2 - 1)));
    else
      printf("No Number\n");
  }
  return 0;
}
