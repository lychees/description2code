#include <iostream>
using namespace std;

string tap[] = {"_0", ".,?!1", "abc2", "def3", "ghi4", "jkl5", "mno6", "pqrs7", "tuv8", "wxyz9"};

pair<int, int> next( char c ){
	for( int i=0 ; i<10; i++){
		for(int j=0; j < tap[i].size(); j++){
			if( tap[i][j]==c )
				return make_pair(i, j);
		}
	}
	return make_pair(-1, -1);
}

int main()
{
	int row[] = {4,1,1,1,2,2,2,3,3,3};
	int col[] = {2,1,2,3,1,2,3,1,2,3};
	
    int cas;
    string str;
    pair<int, int>pi;
    for(cin>>cas; cas; --cas){
    	cin>>str;
    	int prev = 1, nx, res = 0, cost;
    	for( int i=0; i<str.size(); i++){
    		pi = next( str[i] );
    		nx = pi.first;
    		cost = pi.second+1;
    		res += cost+(nx!=prev);//max( abs(row[nx]-row[prev]), abs(col[nx]-col[prev]) )+cost;
    		prev = nx;
    	}
    	cout<<res<<"\n";
    }
    return 0;
}
