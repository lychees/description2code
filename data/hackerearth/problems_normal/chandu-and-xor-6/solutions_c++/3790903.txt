#include <iostream>
using namespace std;

int main()
{
    int matrix[1001][1001];
    int rowSum=0,colSum=0;
   	int N, M, num;
   	cin>>N>>M;
   	
   	for(int i=0; i<M; i++){
   		cin>>matrix[0][i];
   		colSum += matrix[0][i];
   	}
   	matrix[0][M] = colSum; 
   	for(int i=1; i<N; i++){
   			colSum = 0;
   			for(int j=0; j<M; j++){
   				cin>>num;
   				colSum += num;
   				matrix[i][j] = matrix[i-1][j] + num;
   			}
   			matrix[i][M] = colSum;
   	}
   	
   	int myxor = 0;
   	for(int i=0; i<N; i++){
   		for(int j=0; j<M; j++){
   			
   			if( (matrix[i][M] ^ matrix[N-1][j]) > myxor){
   				myxor = matrix[i][M] ^ matrix[N-1][j];
   			
   			//	cout<<matrix[i][M]<<"  "<<matrix[N-1][j]<<"  "<<myxor<<endl; 
   			}
   		}
   	
   	}
   	cout<<myxor<<endl;
    return 0;
}
