#include <iostream>
#include <string>
#include <map>
#include <vector>
#include <algorithm>

using namespace std;

bool compare(pair<int,string> pair1,pair<int,string> pair2){
	return pair1.first>=pair2.first;
}

int main(){
	int n;
	cin>>n;
	map<string,int> mymap;
	for(int i=0;i<n;i++){
		string s;
		cin>>s;
		mymap[s]++;
	}
	vector<pair<int,string> > myvector;

	for(map<string, int>::iterator it=mymap.begin(); it!=mymap.end();++it){
	//	cout<<it->second<<" "<<it->first<<endl;
		pair<int,string> mypair;
		mypair.first=it->second;
		mypair.second=it->first;
		myvector.push_back(mypair);
	}
	sort(myvector.begin(),myvector.end(),compare);
	for(vector<pair<int,string> >::iterator it=myvector.begin();it!=myvector.end();it++){
		cout<<(*it).second<<" "<<(*it).first<<endl;
	}
	
}
