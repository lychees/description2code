#include<iostream>
#include<string>
using namespace std;
int i,j,k,f;
int main(){
	string s;
	string word[501];
	int t;
	cin>>t;
	char l;
	for(i=0;i<t;i++)
	{
		cin>>word[i];
	}
	for(i=0;i<t;i++)
	{
		s=word[i];
		for(j=0;s[j]!='\0';j++)
		{
			l=s[j];
			for(k=j;s[k]!='\0';k++)
			{
				if(s[k]<l)
				l=s[k];
			}
			for(f=j;s[f]!='\0';f++)
			{
				if(l==s[f])
				{
					s[f]=s[j];
					s[j]=l;
				}
			}
		}
		cout<<s<<endl;
	}
	return 0;
}
