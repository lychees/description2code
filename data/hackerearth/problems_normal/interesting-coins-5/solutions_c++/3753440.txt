long long arr[1005];
long long n;
long long isValid(long long x);
long long bin_search(long long l,long long r);
int main()
{
    int t;
    std::ios::sync_with_stdio(false);
    cin>>t;
    while(t--)
    {
        cin>>n;
        long long i,j;
        long long minm=LONG_LONG_MAX,maxm=LONG_LONG_MIN;
        for(i=0;i<n;i++)
        {
            cin>>arr[i];
            if(arr[i]>maxm)
                maxm=arr[i];
            if(arr[i]<minm)
                minm=arr[i];
        }
        long long k=max(n-minm,maxm-1);
        /*for(i=1;i<=k;i++)
        {
            cout<<"isValid("<<i<<"):"<<isValid(i)<<"\n";
        }*/
        long long ans=bin_search(1,k);
        if(isValid(ans)==0)
            ans++;
        else if(isValid(ans-1)==1)
            ans--;
        cout<<ans<<"\n";
    }
    return 0;

}
long long bin_search(long long l,long long r)
{
    if(l==r)
        return l;
    else
    {
        long long mid=l+(r-l)/2;
        if(isValid(mid)==1)
        {
            if(l<mid)
                return bin_search(l,mid-1);
            else
                return mid;
        }
        else
        {
            if(mid<r)
                return bin_search(mid+1,r);
            else
                return mid;
        }

    }
}
long long isValid(long long x)
{
    long long i;
    long long a[n];
    for(i=0;i<n;i++)
        a[i]=arr[i];
    if(a[0]-1<=x)
        a[0]=1;
    else
        a[0]-=x;
    for(i=1;i<n;i++)
    {
        long long minm=a[i-1]+1;
       // cout<<minm<<"\n";
        if(a[i]<minm)
        {
            if(minm-a[i]<=x)
            {
                a[i]=minm;
            }
            else
                return 0;
        }
        else if(a[i]>minm)
        {
            if(a[i]-minm<=x)
                a[i]=minm;
            else
                a[i]-=x;
        }
    }
    return 1;
}
