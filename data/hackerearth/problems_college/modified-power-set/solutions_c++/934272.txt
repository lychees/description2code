#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    char str[1000009];
    long long len;
    while(t--)
    {
    	cin>>str;
    	int alpha1[26],alpha2[26],num[10];
    	for(int i=0;i<26;i++)
    	{
    		alpha1[i]=0;
    		alpha2[i]=0;
    	}
    	for(int i=0;i<10;i++)
    		num[i]=0;
    	len=0;
    	for(int i=0;str[i]!=NULL;i++)
    	{
    		char a=str[i];
    		if(a>='0'&&a<='9')
    		{
    			num[a-'0']++;
    			if(num[a-'0']<2)
    				len++;
    		}
    		else if(a>='a'&&a<='z')
    		{
    			alpha1[a-'a']++;
    			if(alpha1[a-'a']<2)
    				len++;
    		}
    		else
    		{
    			alpha2[a-'A']++;
    			if(alpha2[a-'A']<2)
    				len++;
    		}
    	}
    	len=((len+1)*len)/2;
    	cout<<len<<endl;
    	
    }
    return 0;
}
