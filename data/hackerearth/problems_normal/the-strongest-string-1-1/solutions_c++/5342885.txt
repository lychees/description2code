#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n; cin>>n;
    string a; cin>>a;
    bool check[26]={0};
    int left=0,right=n-1;
    string ans="";
    for(int i=25;i>=0;i--)
    {
    	for(int j=left;j<=right;j++)
    	{
    		if(a[j]==i+'a')
    		{
    			ans+=a[j];
    			left=j+1;
    			break;
    		}
    	}
    }
    cout<<ans<<endl;
}
