#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int bit[100005][26];
void add(int ind,char c,int n)
{
	while(ind<=n)
	{
		bit[ind][c-'a']++;
		ind+=(ind&(-ind));
	}
}
void upd(int ind,char c,char pr,int n)
{
	while(ind<=n)
	{
		bit[ind][c-'a']++;
		bit[ind][pr-'a']--;
		ind+=(ind&(-ind));
	}
}
int que(int ind,int w)
{
	int s=0;
	while(ind>0)
	{
		s+=bit[ind][w];
		ind-=(ind&(-ind));
	}
	return s;
}
int main()
{
    string s;
    int a[26],len,i,q,l,r,x,w,ce,co;
    cin>>len>>q;
    cin>>s;
    int ind;
    char c;
    for(i=0;i<len;i++)
    {
    	add(i+1,s[i],len);
    }
    while(q--)
    {
    	cin>>x;
    	if(x==2)
    	{
    		cin>>l>>r;
    		for(i=0;i<26;i++)
    		{
    			a[i]=0;
    		}
    		if(l==r)
    		{
    			cout<<"yes"<<endl;
    		}
    		else
    		{
    			w=r-l+1;
    			ce=0;co=0;
    			for(i=0;i<26;i++)
    			{
    				a[i]=que(r,i)-que(l-1,i);
    			}
    			for(i=0;i<26;i++)
    			{
    				if(a[i]%2==0)
    				{
    					ce++;
    				}
    				else
    				{
    					co++;
    				}
    			}
    			if(w%2==0&&co==0)
    			{
    				cout<<"yes"<<endl;
    			}
    			else if(w%2!=0&&co==1)
    			{
    				cout<<"yes"<<endl;
    			}
    			else
    			{
    				cout<<"no"<<endl;
    			}
    		}
    	}
    	else
    	{
    		cin>>ind>>c;
    		upd(ind,c,s[ind-1],len);
    		s[ind-1]=c;
    	}
    }
    return 0;
}
