#include <iostream>
using namespace std;

int main(){
	
	char ch;
	int h, k;

	while(cin >> ch >> h)
	{
		k = int(ch) + (h - 1);
		if((( k > 90) && (k < 97)) || (k > 122) || (h == 1)){
    		cout << "INVALID" << endl << endl;
    		continue;
    	}
		else
		{
			for(int i = 1; i <= h; ++i)
    		{	
        		for(int j = 1; j <= i; ++j)
        		{
        	  		cout << ch << " ";
        				
        		}
        		ch++;
        		cout << endl;
    		}
    		cout << endl;
		}
	}
	
	return 0;
}
