#include<bits/stdc++.h>
#define ll long long
#define f(a,b,c) for(ll int i=a;i<b;i+=c)
using namespace std;

int main() {
	cin.tie(0);
	ios_base::sync_with_stdio(false);
	int T;
	scanf("%d",&T);
	while(T--) {
		ll int n,p;
		scanf("%d%d",&n,&p);
		char st[100005];
		ll int leftt[100005],rightt[100005];
		string str;
		scanf("%s",st);
		str = string(st);
		for(int i=0;i<n;i++)
			if(i-1>=0)
				leftt[i] = leftt[i - 1] + (str[i] == '1');
			else
				leftt[i] = (str[i]=='1');
		
		for(int i=n-1;i>=0;i--)
			if(i+1 <= n-1)
				rightt[i] = rightt[i+1] + (str[i]=='1');
			else
				rightt[i] = (str[i]=='1');
		
		ll ans=0;
		for(int i=0;i<n;i++) {
			ll count=0;
			if(str[i]=='1') {
				count += (i - p - 1 >= 0) ? (leftt[i - 1] - leftt[i - p - 1]) : ((i - 1 >= 0) ? leftt[i - 1] : 0);
				count += (i + p + 1 <= n - 1) ? (rightt[i + 1] - rightt[i + p + 1]) : ((i + 1 <= n - 1) ? rightt[ i + 1] : 0);
				count += 1;
			}
			ans+=count;
		}
		
		if(ans==0) {
			puts("0/1");
		}
		else {
			ll gc = __gcd(ans, (n * n));
			ll ad = ans / gc;
			ll df = (n * n) / gc;
			printf("%lld",ad);
			printf("/");
			printf("%lld",df);
			printf("\n");
		}
	}
	return 0;
}
