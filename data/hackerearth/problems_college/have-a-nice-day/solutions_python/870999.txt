import sys

t = int(sys.stdin.readline())

while t:
	n = int(sys.stdin.readline())
	ans = 0
	x = map(int,sys.stdin.readline().split())
	y = map(int,sys.stdin.readline().split())
	
	x.sort()
	y.sort()
	for i in range(n):
		ans = ans + x[i]*y[i]
	print ans
	t = t-1
