#include<bits/stdc++.h>
using namespace std;
#define MAX 100005
#define MO 1000000000
long long a[MAX];
long long wood ( long long mid , long long n ) {
    long long i,s=0;
    for ( i=0 ; i<n ; i++ ) {
        if( a[i] >= mid )
            s = s+(a[i]-mid);
    }
    return s;
}
int main () {
    long long l = 0,pre_l , h = MO;
    long long n,m,i,mid,s=0;
    cin>>n>>m;
    for ( i=0 ; i<n ; i++ ) {
        cin>>a[i];
        s = s + a[i];
    }
    if ( s < m )
        cout<<"-1"<<endl;
    else {
        while ( l <= h ) {
            mid = (l+h)/2;
            if ( wood(mid,n) == m ) {
                cout<<mid<<endl;
                break;
            }
            else if ( wood(mid,n) > m ) {
                pre_l = l;
                l = mid;
                if ( l == pre_l ) {
                    cout<<mid<<endl;
                    break;
                }
            }
            else if ( wood(mid,n) < m ) {
                h = mid;
            }
        }
    }
    return 0;
}
