#include <iostream>
using namespace std;
void quickSort(long long arr[], long left, long right) {
      long i = left, j = right;
      long tmp;
      long pivot = arr[(left + right) / 2];
 
      /* partition */
      while (i <= j) {
            while (arr[i] < pivot)
                  i++;
            while (arr[j] > pivot)
                  j--;
            if (i <= j) {
                  tmp = arr[i];
                  arr[i] = arr[j];
                  arr[j] = tmp;
                  i++;
                  j--;
            }
      };
 
      /* recursion */
      if (left < j)
            quickSort(arr, left, j);
      if (i < right)
            quickSort(arr, i, right);
}
int main()
{
    long n,i,z,x=0;
    long long k;
    cin>>n>>k;
    long long a[n];
    z=1;
    for(i=1;i<=n;i++)
    {
    	cin>>a[i]; 	
    }
    quickSort(a,1,n);
    do
    {
    	k=k-a[z++];
    	if(k>0)
    		x++;
    	else
    		break;
    }while(1);
    cout<<x;
    return 0;
}
