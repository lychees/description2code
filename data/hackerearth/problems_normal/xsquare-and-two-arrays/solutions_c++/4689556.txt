#include <bits/stdc++.h>
using namespace std;

int main()
{
    long i,j,k,n,index,l,r,q;
    cin>>n>>q;
    long a[n],b[n];
    for(i=1;i<=n;++i)
    	cin>>a[i];
    for(i=1;i<=n;++i)
    	cin>>b[i];
    for(;q--;)
    {
    	cin>>index>>l>>r;
    	long c=0;
    	if(index==1)
    	{
    		for(i=l;i<=r;i+=2)
    		{
    			c+=a[i];
    		}
    		for(i=l+1;i<=r;i+=2)
    		{
    			c+=b[i];
    		}
    	}
    	else if(index==2)
    	{
    		for(i=l;i<=r;i+=2)
    		{
    			c+=b[i];
    		}
    		for(i=l+1;i<=r;i+=2)
    		{
    			c+=a[i];
    		}
    	}
    	cout<<c<<endl;
    }
    return 0;
}
