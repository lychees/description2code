#include <iostream>
using namespace std;
long long fact(int a)
{
	if(a==0) return 1;
	else return (a*fact(a-1));
}	

int main()
{ int t,c,v;
  long long a;
  cin>>t;
  while(t--)
  {
  	cin>>c>>v;
  	a=(fact(7)*fact(5)*fact(c+v))/(fact(c)*fact(7-c)*fact(v)*fact(5-v));
  	cout<<a<<endl;
  }
    
    return 0;
}
