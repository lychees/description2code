#include <iostream>
using namespace std;

int main()
{
	int T;
	cin >> T;
	while(T--){
		long N,sum,sum2,no;
		cin >> N;
		if(N==1){
			cout << "1" << endl;
			continue;
		}
		else if(N%2==0){
			sum = N-1;
			no = (N-2)/2;
			sum2 = no*(no+1);
			sum+=2*sum2;
			cout << sum << endl;
		}
		else{
			sum = N-1;
			no = (N-1)/2;
			sum2 = no*(no);
			sum = sum+ 2*sum2;
			sum--;
			cout << sum << endl;
			
		}
	}
    return 0;
}
