def prime(n):
    ps, sieve = [], [True] * (n + 1)
    for p in range(2, n + 1):
        if sieve[p]:
           ps.append(p)
           for i in range(p * p, n + 1, p):
               sieve[i] = False
    return ps

a=prime(2000009)
dic={}
for i in a:
	dic[i]=True
t=input()
for j in xrange(t):
	x=input()
	if x==3:
		print "3"
	elif x not in dic:
		print "1"
	else:
		print "2"
