#include <bits/stdc++.h>
#define ll long long
#define MOD 1000000007
using namespace std;

int f(int n) {
    if (n == 0) return 0;
    else {
        int tmp = sqrt(n);
        //cout << tmp * tmp << " ";
        return 1 + f(n - tmp * tmp);
    }
}

int main()
{
    int t, n;
    cin >> t;

    while (t--) {
        cin >> n;
        int mn = INT_MAX;
        int tmp = sqrt(n);
        while (tmp >= 1) {
            if (f(n - tmp * tmp) + 1 < mn) {
                mn = f(n - tmp * tmp) + 1;
            }
            tmp--;
        }
        cout << mn << endl;
    }
    return 0;
}
