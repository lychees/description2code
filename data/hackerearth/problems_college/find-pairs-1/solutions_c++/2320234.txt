#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
#define os 1000000
long long cs, n, x, s, i;
long long C[2*os+1];
int main()
{
  scanf("%lld",&cs);
  while (cs--)
  {
    memset(C,0,sizeof C);
    scanf("%lld",&n);
    while (n--)
    {
      scanf("%lld",&x);
      C[x + os]++;
    }
    s = 0;
    for (i = 0; i <= 2 * os; i++)
      s += C[i] * (C[i] + 1);
    printf("%lld\n",s/2);
  }
  return 0;
}
