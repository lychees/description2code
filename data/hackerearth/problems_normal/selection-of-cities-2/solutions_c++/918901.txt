#include <bits/stdc++.h>
using namespace std;

int power(long long a, long long b, long long  MOD) {
long long x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>MOD) x%=MOD;
        }
        y = (y*y);
        if(y>MOD) y%=MOD;
        b /= 2;
    }
    return x;
}


int main(int argc, char const *argv[])
{
	long long testcases;
	cin >> testcases;
	for (int i = 0; i < testcases; ++i)
	{
		long long N;
		cin >> N;

		cout << (power(2, N, 1000000007)-1)%1000000007 << endl;
	}
	
	return 0;
}
