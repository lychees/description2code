def reverser(n) :
    list1 = list()
    for digit in str(n) :
        list1.append(digit)
    list1.reverse()
    number = ""
    for num in list1 :
        number += num
    number = int(number)
    return number

test_cases = int(raw_input())
for a in range(1 , test_cases + 1) :
    inp = raw_input().split()
    num1 = inp[0]
    num2 = inp[1]
    sum = reverser(int(num1)) + reverser(int(num2))
    print reverser(sum)
