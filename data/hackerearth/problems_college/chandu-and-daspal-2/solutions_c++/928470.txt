#include <bits/stdc++.h>
using namespace std;
int gcd(int a, int b)
{
	if(b == 0)
		return a;
	else
		gcd(b, a%b);
}
int main()
{
	int t;
	int x,y,val;
	char a[10];
	scanf("%d",&t);
	while(t--)
	{
		int count = 0;
		scanf("%d%d",&x,&y);
		cin>>a;
		while(x>0 && y>0)
		{
			val = gcd(x,y);
			x -= val;
			y -= val;
			count ++;
		}
		if(count%2 == 0)
		{
			if(a[0] == 'C')
				printf("Daspal\n");
			else
				printf("Chandu\n");
		}
		else
		{
			if(a[0] == 'C')
				printf("Chandu\n");
			else
				printf("Daspal\n");
		}
	}
	return 0;
}
