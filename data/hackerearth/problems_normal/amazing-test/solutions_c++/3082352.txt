#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
vector<int> v1;
vector<vector<bool> > v2;
int main()
{
    int t,n,m,p1,p2,l,r,sum;
    cin>>t;
    while(t--)
    {
    	v1.clear();
    	cin>>n>>m;
    	sum=0;
    	v1.resize(n);
    	v2.resize(n+1);
    	for(p1=0;p1<n;p1++)
    	{
    		cin>>v1[p1];
    		sum+=v1[p1];
    	}
    	for(p1=0;p1<=n;p1++)
    	{
    		v2[p1].resize(m+1,false);
    		v2[p1][0]=true;
    	}
    	for(p1=1;p1<=n;p1++)
    	{
    		for(p2=1;p2<=m;p2++)
    		{
    			if(p2<v1[p1-1])
    				v2[p1][p2]=v2[p1-1][p2];
    			else
    				v2[p1][p2]=v2[p1-1][p2]||v2[p1-1][p2-v1[p1-1]];
    		}
    	}
    	if(m>=sum)
    	{
    		cout<<"YES\n";
    		continue;
    	}
    	for(p1=0;p1<n;p1++)
    		if(v1[p1]>m)
    		{
    			cout<<"NO\n";
    			break;
    		}
    	if(p1<n)
    		continue;
    	if(sum>2*m)
    	{
    		cout<<"NO\n";
    		continue;
    	}
    	if(v2[n][m])
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    	
    }
    return 0;
}
