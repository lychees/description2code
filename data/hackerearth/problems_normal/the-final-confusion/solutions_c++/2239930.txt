    #include <iostream>
    using namespace std;
     
    typedef unsigned long long ull;
     
    ull dp[35];
    ull gcd( ull a, ull b ) {  if( !b ) return a;  return gcd( b, a % b ); }
    void init() {
    	dp[1] = 1;
    	for(int i=2; i<=30; ++i) {
    		ull num = dp[i-1]*2*(2*i-1);
    		ull den = i+1;
    		//ull hcf = gcd(num, den);
    		dp[i] = num / den;
    	}
    }
     
    int main()
    {
    	init();
    	int tc;
    	cin >> tc;
    	while(tc--) {
    		int n;
    		cin >> n;
    		cout << dp[n] << endl;
    	}
        return 0;
    }
