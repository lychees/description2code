N = int(raw_input())
for i in range(0, N):
    count = raw_input()
    li = map(int, raw_input().split())

    isInverse = True
    for i in range(0,len(li)):
        ix = li[i]
        if i+1 != li[ix-1]:
            isInverse = False
            break

    if (isInverse):
        print "inverse"
    else:
        print "not inverse"
