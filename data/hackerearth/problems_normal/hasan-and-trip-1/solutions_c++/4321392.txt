#include <bits/stdc++.h>
#define ll long long int
using namespace std;
double dp[10000];
pair<pair<ll,ll>,ll> p;
vector<pair<pair<ll,ll> ,ll> > v;
ll n;
double distance(ll x1,ll y1,ll x2,ll y2)
{
	ll temp1,temp2;
	double ans;
	temp1=pow(abs(x1-x2),2);
	temp2=pow(abs(y1-y2),2);
	ans = sqrt(temp1+temp2);
	return ans;
}
double solve(ll i,ll value)
{
	if ( i > n) {
		return 0;
	}
	if ( i == n-1) return v[i].second;
	if ( dp[i] != -1e9 ) {
		return dp[i];
	}
	ll x,y;
	ll k1,k2;
	double ans = -1e9;
	x=v[i].first.first;
	y= v[i].first.second;
	ll k  = v[i].second;
	double temp;
	for (ll j = i+1; j < n; j++) {
		k = v[i].second;
		temp=distance(x,y,v[j].first.first,v[j].first.second);
		double ans1 = (k-(temp))+(solve(j,k));
		ans = max(ans,ans1);
	}
	dp[i] = ans;
	return ans;
}
	
	
int main()
{
	ll t,i,j,k,x,y,f;	
	v.clear();
	scanf("%lld",&n);
	
	for ( i = 0; i < n; i++) {
		scanf("%lld%lld%lld",&x,&y,&f);
		v.push_back(make_pair(make_pair(x,y),f));
	}
	for ( i = 0; i < n+15; i++) {
		dp[i] = -1e9;
	}
	double ans =  solve(0,0);
	printf("%.6lf\n",ans);

	
	return 0;
}
