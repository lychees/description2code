#include<cstdio>
#include<iostream>
#define ll long long
using namespace std;
#include<algorithm>
int main()
{
  ll t,n,ans,k,i;

      ll moves=0;
      cin>>n>>k;
      ll x[k+1],y[k+1];
      for(i=0;i<k;i++)
      {
          cin>>x[i];
      }
      for(i=0;i<k;i++)
      {
          cin>>y[i];
      }
      for(i=0;i<k;i++)
      {
             ll a=(x[i]-1)+(y[i]-1);
             ll b=(n-x[i])+(n-y[i]);
             ll c=(n-x[i])+(y[i]-1);
             ll d=(x[i]-1)+(n-y[i]);
            moves+=min(a,min(b,min(c,d)));
      }
        cout<<moves<<endl;

}
