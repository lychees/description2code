    #include <bits/stdc++.h>
    using namespace std;
    int dp[500005];
    int cnt[500005][13];
    int val[] = {1,2,2,1,1,2,2,1};
    int main()
    {
	    string def = "ahackert";
	    int m = (int)def.size();
	    string s;
	    cin >> s;
	    s = "X" + s;
	    int n = (int)s.size();
	    for ( int i = 1; i <= n-1; i++ ) {
		    if ( s[i] >= 'A' && s[i] <= 'Z' ) s[i] = (char)(s[i]+32);
		    for ( int j = 1; j <= m-1; j++ ) {
			    cnt[i][j] = cnt[i-1][j];
			    if ( s[i] == def[j] ) cnt[i][j]++;
		    }
	    }
	    for ( int i = 11; i <= n-1; i++ ) {
		    for ( int j = 1; j <= m-1; j++ ) {
		    	if ( cnt[i][j]-cnt[i-11][j] != val[j] ) goto p1;
		    }
		    dp[i] = 1;
		    p1: { }
	    }
	    for ( int i = 1; i <= n-1; i++ ) dp[i] += dp[i-1];
	    int q,a,b;
	    scanf("%d", &q);
	    while ( q-- ) {
		    scanf("%d%d", &a, &b);
		    a += 10;
		    if ( b < a ) {
			    printf("0\n");
			    continue;
	    	}
	    	else printf("%d\n", dp[b]-dp[a-1]);
	    }
    	return 0;
    }
