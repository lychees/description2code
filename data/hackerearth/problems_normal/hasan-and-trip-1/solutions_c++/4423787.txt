#include <bits/stdc++.h>
using namespace std;
double dp[3002];
int main()
{
   int n,i,j;
   scanf("%d",&n);
   double x[n],y[n],f[n];
   for(i=0;i<n;i++)
   {
   	scanf("%lf%lf%lf\n",&x[i],&y[i],&f[i]);
   }
   for(i=0;i<n;i++)
   dp[i]=INT_MIN;
   double ans=INT_MIN;
   dp[0]=f[0];
   for(i=0;i<n;i++)
   {
   	for(j=i+1;j<n;j++)
   	{
   		double dist=sqrt((y[j]-y[i])*(y[j]-y[i])+(x[j]-x[i])*(x[j]-x[i]));
   		dp[j]=max(dp[j],dp[i]+f[j]-dist);
   	}
   }
   printf("%lf\n",dp[n-1]);
    return 0;
}
