#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,cnt=0;
    	cin>>n;
    	long int arr[n];
    	for(int i=0;i<n;i++) 
    	 cin>>arr[i];
    	for(int i=0;i<n;i++)
    	{
    		for(int j=i+1;j<n;j++)
    		{
    			if(arr[i]==arr[j]) continue;
    			if((arr[i]+arr[j])%2==0) cnt++;
      		}
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
