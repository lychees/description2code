#include<bits/stdc++.h>
using namespace std;
int main(void) {
	// your code goes here
	int n,i,j,count[101][26]={0}, ans=0;
	string s;
	 cin>>n;             //scanf("%d",&n);
	for(i=0;i<n;i++)
	{
	  cin>>s;
	  for(j=0;j<s.length();j++)
	  {
	      count[i][s[j]-'a']++;
	  }
	}
	for(i=0;i<26;i++)
	{
	    int ok=1;
	    for(j=0;j<n;j++)
	    {
	        if(count[j][i]==0)
	        ok=0;
	    }
	    if(ok)
	    ans++;
	}
           cout<<ans<<endl;            //printf("%d\n",ans);
	
	return 0;
}
