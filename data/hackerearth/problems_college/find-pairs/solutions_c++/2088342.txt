#include<iostream>
#define ll long long int 
using namespace std;
int main()
{
	int t;
	ll n,i,j;
	cin>>t;
	while(t--)
	{
		cin>>n;
		ll arr[n];
		for(i=0;i<n;i++)
		{
			cin>>arr[i];
		}
		ll pair=0;
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				if((j>=i)&&(arr[i]==arr[j]))
				{
					pair+=1;
				}
			}
		}
		cout<<pair<<endl;
		
		
	}
	return 0;
}
