#include <iostream>
using namespace std;

int main()
{
    long int n,i;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	long int a,b,ans=0,c,d;
    	cin>>a>>b;
    	
    	while(a>0 || b>0)
    	{
    		c=a%10;
    		d=b%10;
    		if(c+d>=10)
    		{
    			ans++;
    		}
    		a=a/10;
    		b=b/10;
    	}
    	if(ans>1)
    	{
    	cout<<ans<<" carry operations\n";	
    	}
    	else if(ans==1)
    	{
    	cout<<ans<<" carry operation\n";	
    	}
    	else
    	cout<<"No carry operation\n";
    	
    }
    return 0;
}
