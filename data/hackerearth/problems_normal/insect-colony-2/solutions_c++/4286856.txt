int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	
    	int x,c=0;
    	
    	rep(x,1,n,1)
    	{
    		int p;
    		cin>>p;
    		if(p&1) c++;
    	}
    	
    	if(c%2 == 0) cout<<"Yes\n";
    	else cout<<"No\n";
    	
    }
    
    return 0;
}
