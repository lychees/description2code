const long long INF = LONG_MAX;

typedef long long LL;

map<int,LL> m;

LL gcd(LL a,LL b)
{
        if(b==0) return a;
        else return gcd(b,a%b);
}

int a[55];

int main()
{

int test; INT(test);

while(test>0)
{
        --test;
        int n,L,R;

        INT(n);

        FOR(i,0,n)
        INT(a[i]);

        INT(L); INT(R);

        LL ans=0;

        m.clear();

        FOR(i,0,n)
        {
            for(map<int,LL>::iterator it=m.begin();it!=m.end();++it)
            {
                int g = gcd(a[i],it->first);
                m[g] += (LL)m[it->first];
             }

                m[a[i]] += 1;
        }

     FOR(i,L,R+1)
        ans += m[i];

    PLLD(ans);

}
return 0;
}
