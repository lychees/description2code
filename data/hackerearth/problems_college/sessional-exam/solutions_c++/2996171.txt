#include<iostream>
using namespace std;
int main()
{
    int n,m,i,j,k,l=0;
    cin>>n>>m;
    char a[n][m];
    for(i=0;i<n;i++)
    {
        cin>>a[i];
    }
    for(i=0;i<n;i++)
    {
        for(j=0;j<m;j++)
        {
            for(k=0;k<n;k++)
            {
                if(a[k][j]>a[i][j])
                    break;
            }
            if(k>=n)
            {
                l++;
                break;
            }
        }
    }
    cout<<l<<endl;
    return 0;
}
