#include <iostream>
#include<string>
#include <cctype>
using namespace std;

int main()
{
    std::string str;
    getline(cin,str);
    int maxc=0,count;
    int n=str.length();
    char apna;
    for(int i=0;i<n;i++){
    	char c=str[i];
    	count=1;
    	for(int j=i+1;j<n;j++){
    	if(str[j]==c)
    	{count++;
    	 if(count>=maxc)
        	{  
        		if(maxc==count && apna!=str[j]){
        			apna=(apna<=str[j]?apna:str[j]);
        		}
        		else{
        		maxc=count;
        		apna=str[j];
        		}
        	}
    	}
    	}
    	
    }
    cout<<apna<<" "<<maxc;
    return 0;
}
