#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int n;
    cin>>n;int i,j;
    int hash[n][26];
    for(i=0;i<n;i++)
    for(j=0;j<26;j++)
    hash[i][j]=0;
    string s[n];
    
    for(i=0;i<n;i++)
    cin>>s[i];
    
    for(i=0;i<n;i++)
    {
    	int k=s[i].length();
    	for(j=0;j<k;j++)
    	hash[i][s[i][j]-'a']++;
    }
    int ans=0,m;
    for(i=0;i<26;i++)
    {m=0;
    	for(j=0;j<n;j++)
    	{
    		if(hash[j][i]!=0)
    		m=1;
    		else
    		{
    			m=0;
    			break;
    		}
    	}
    	if(m==1)
    	ans++;
    }
    cout<<ans<<"\n";
    return 0;
}
