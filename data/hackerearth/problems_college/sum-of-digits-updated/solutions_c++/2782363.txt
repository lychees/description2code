#include <bits/stdc++.h>
using namespace std;
char s[10000001];
int main()
{
    long long int n, sum ,i, t, a;
    cin>>t;
    while(t--) {
    	cin>>s;
    	sum = 0;
    	n = strlen(s);
    	for(i=0;i<n;i++) {
    		a = (long long int)(s[i]) - 48;
    		sum+=a;
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
