#include <iostream>
#include <cmath>
#include <cstdio>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int k,n;
   	cin>>k>>n;
   	double ans;
   	if(n<k)
   	 ans=1;
    else if(k==1)
        ans=0;
   	 else
   	 {
   	 	ans=1;
   	 	int temp=n-1;
   	 	while(1)
   	 	{
   	 		ans=ans*(double(temp)/n);
   	 		temp--;
   	 		if(temp==n-k || temp<=0)
                break;
   	 	}

   	 	ans=double(1)-ans;
   	 }
   	 printf("%.9lf\n",ans);
   }
    return 0;
}
