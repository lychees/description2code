#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,i,k=0,j;
    cin>>n;
    vector <int> a(n);
    for(i=0;i<n;i++)
    cin>>a[i];
    sort(a.begin(),a.end());
    vector <int> dp(n,1);
    
    for(i=1;i<n;i++)
    {
    	for(j=i-1;j>=0;j--)
    	{
    		if(a[i]%a[j]==0)
    		dp[i]=max(dp[i],dp[j]+1);
    	}
    }
    for(i=0;i<n;i++)
    k=max(k,dp[i]);
    if(k>1)
    cout<<k;
    else cout<<-1;
    
    
    return 0;
}
