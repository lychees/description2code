#include <iostream>
#include <vector>
using namespace std;

int main()
{
    long long int T,r,g,b,i,j,len,total,answer[8]={};
     vector<long long int>R,G,B;
     cin>>T>>r>>g>>b;
     len=total=0;
     for(i=0;i<T;i++)
     {
         if(len==r)
         {
            R.push_back(1);
            total++;
            if(total==r)
                len=total=0;
         }
         else
         {
            R.push_back(0);
            len++;
         }
     }
     len=total=0;
     for(i=0;i<T;i++)
     {
         if(len==g)
         {
            G.push_back(1);
            total++;
            if(total==g)
                len=total=0;
         }
         else
         {
            G.push_back(0);
            len++;
         }
     }
     len=total=0;
     for(i=0;i<T;i++)
     {
         if(len==b)
         {
            B.push_back(1);
            total++;
            if(total==b)
                len=total=0;
         }
         else
         {
            B.push_back(0);
            len++;
         }
     }
     for(i=0;i<T;i++)
     {
         if(R[i]==1&&G[i]==0&&B[i]==0)
            answer[0]++;
         if(R[i]==0&&G[i]==1&&B[i]==0)
            answer[1]++;
         if(R[i]==0&&G[i]==0&&B[i]==1)
            answer[2]++;
         if(R[i]==1&&G[i]==1&&B[i]==0)
            answer[3]++;
         if(R[i]==0&&G[i]==1&&B[i]==1)
            answer[4]++;
         if(R[i]==1&&G[i]==0&&B[i]==1)
            answer[5]++;
         if(R[i]==1&&G[i]==1&&B[i]==1)
            answer[6]++;
         if(R[i]==0&&G[i]==0&&B[i]==0)
            answer[7]++;
     }
     for(i=0;i<8;i++)
        cout<<answer[i]<<" ";
    return 0;
}
