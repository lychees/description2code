#include<stdio.h>
#include<string.h>
#define min(a, b) (((a) <= (b)) ? (a) : (b))
int main(){
	int t, ac, bc, i, l, s, f; scanf("%d", &t); char a[100000];
	while(t--){
		scanf("%s%d%d", a, &ac, &bc), l = strlen(a), s = f = 0;
		for(i = 0; i < l / 2; i++){
			if(a[i] == '/' && a[l - i - 1] == '/')	s += (min(ac, bc) * 2);
			else if((a[i] == 'a' && a[l - i - 1] == '/') || (a[i] == '/' && a[l - i - 1] == 'a'))	s += ac;
			else if((a[i] == 'b' && a[l - i - 1] == '/') || (a[i] == '/' && a[l - i - 1] == 'b'))	s += bc;
			else if(a[i] != a[l - i - 1])	f = 1;
		}
		s == 0 || f == 1? printf("-1\n") : printf("%d\n", s);
	}

	return 0;
}
