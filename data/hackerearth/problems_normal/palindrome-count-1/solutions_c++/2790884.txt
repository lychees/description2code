#include <iostream>
#include<cstring>
using namespace std;

int main()
{   string s;
	while(cin>>s)
	{
		int a[26]={0};
		int l=s.length();
		 int count=l;
		
		for(int i=1;i<l;i++)
		{ int low=i-1;
		  int high=i;
		  while(low>=0&&high<l&&s[low]==s[high])
		    { low--;
		      high++;
		    	count++;
		    }
		   low=i-1;
		   high=i+1;
		  while(low>=0&&high<l&&s[low]==s[high])
		    { low--;
		      high++;
		    	count++;
		    }
			
		}
		cout<<count<<endl;
	}
    return 0;
}
