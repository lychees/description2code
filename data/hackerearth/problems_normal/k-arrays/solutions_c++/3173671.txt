#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
ll fast(ll base , ll exp){
  ll  res = 1;
  while(exp){
    if(exp & 1)
    res = ( res * base)%mod;
    exp = exp >> 1;
    base = ( base * base )%mod;
  }
  return res;
}
