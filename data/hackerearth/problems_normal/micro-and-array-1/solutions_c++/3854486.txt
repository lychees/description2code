#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int n, q, t, x, y, v, pos = -1 ;
    vector<int> a ;
    vector<int> :: iterator it ;
    cin>>n>>q ;
    a.resize(n) ;
    
    for(int i = 0; i < n; i++){
    	cin>>a[i] ;
    }
    
    for(int i = 0; i < q; i++){
    	cin>>t ;
    	if(t == 0){
    		cin>>x>>y ;
    		a[x-1] = y ;
    		
    	}else if(t == 1){
    		cin>>v ;
    		pos = -1 ;
			for(int i = 0; i < n; i++){
				if(a[i] >= v){
					pos = i+1 ;
					break ;
				}
			}
			cout<<pos<<endl ;
    	}
    }
    
    return 0;
}
