#include <iostream>
using namespace std;

int main()
{
    int t, i, n, ans, j;
    cin>>t;
    while(t--){
    	cin>>n;
    	ans=1;
    	int a[n][2];
    	for(i=0;i<n;i++){
    		cin>>a[i][0];
    		a[i][1]=0;
    	}
    	a[n-1][1] = 1;
    	for(i=n-2;i>=0;i--){
    		for(j=i+1;j<n;j++){
    			if(a[i][0] > a[j][0]){
    				if(j==0){
    					ans++;
    					a[i][1] = 1;
    				}
    				else if(a[j][1] == 1){
    					ans++;
    					a[i][1] = 1;
    					break;
    				}
    			}
    			else break;
    		}
    	}
    	cout<<ans<<endl;	
    }
    return 0;
}
