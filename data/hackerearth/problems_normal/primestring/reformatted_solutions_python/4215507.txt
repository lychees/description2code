from math import sqrt


def sieve(n):
    primes = range(3, n+1, 2)
    for base in xrange(len(primes)):
        if primes[base] is None:
            continue
        if primes[base] > sqrt(n):
            break

        for i in range(base+(base+1)*primes[base], len(primes), primes[base]):
            primes[i] = None

    primes.insert(0, 2)
    return filter(None, primes)


primes = sieve(100000)


def prime(n):
    if n in primes:
        return True
    return False
