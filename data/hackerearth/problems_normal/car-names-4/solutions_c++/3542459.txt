#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	bool flag = true;
    	if (str.length()%3!=0) {cout<<"Not OK"<<endl;continue;};
    	int len = str.length()/3;
    	if (str[0]==str[len] || str[len] == str[2*len] || str[0]==str[2*len]) {cout<<"Not OK"<<endl;continue;};
    	string b="";
    	for(int i=0;i<len;i++) b+=str[0];
    	if (str.substr(0,len)!=b) {cout<<"Not OK"<<endl;continue;};
    	b="";
    	for(int i=0;i<len;i++) b+=str[len];
    	if (str.substr(len,len)!=b) {cout<<"Not OK"<<endl;continue;};
    	b="";
    	for(int i=0;i<len;i++) b+=str[2*len];
    	if (str.substr(2*len,len)!=b) {cout<<"Not OK"<<endl;continue;};
    	cout<<"OK"<<endl;
    }
}
