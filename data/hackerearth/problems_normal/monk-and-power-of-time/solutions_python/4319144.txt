import sys
n = int(sys.stdin.readline().strip())
pca = map(int,sys.stdin.readline().strip().split(' '))
pco = map(int,sys.stdin.readline().strip().split(' '))
t=0 ; i=0 ; j=0 ; k=0;
while i<n:
    if pca[j]==-1:
        j=(j+1)%n
        continue
    if pco[i] !=pca[j]:
        t+=1
    else:
        pca[j]=-1;
        i+=1
