t=input()
while t:
	t-=1
	n , s = raw_input().split()
	n = int(n)
	ch = [0]*26
	for i in xrange(len(s)):
		ch[ord(s[i])-97] += 1
	
	max, pos, team_len, index = 0 , 1, 11, 1
	while n:
		n -= 1
		s = raw_input()
		points = 0
		l = len(s)
		for i in xrange(l):
			points  = points + ch[ord(s[i])-97]
		

		if (points == max) and (team_len > l):
			pos = index
			team_len = l
			
		if points > max:
			pos = index
			max = points
			team_len = l
			
		
		index += 1
	
	print pos
