//Question  on dynamic programming from hacker earth
#include<bits/stdc++.h>
using namespace std;
long int a[100002];
long int sum[100002];
int main()
{
    int t,n,k,i;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&k);
        for(i=1;i<=n;i++)
        {
            scanf("%ld",&a[i]);
            sum[i]=0;
        }
        sum[0]=0;
        for(i=1;i<=n;i++)
        {
            if(k+1>=i)
            {
                if(a[i]>sum[i-1])
                    sum[i]=a[i];
                else
                    sum[i]=sum[i-1];
            }
            else
            {
                if((a[i]+sum[i-k-1])>sum[i-1])
                    sum[i]=a[i]+sum[i-k-1];
                else
                    sum[i]=sum[i-1];
            }
        }
        printf("%ld\n",sum[n]);
    }
    return 0;
}
