#include <iostream>
#include <cmath>

using namespace std;

inline int sum_digits(int x)
{
    int result=0;
    while(x)
    {
        result += x%10;
        x /= 10;
    }
    return result;
}

int T,len,i,ans;
string s;

int main()
{
    cin >> T;
    while(T--)
    {
        ans = 0;
        cin >> s;
        len = s.length();
        ans = sum_digits(s[len-1]-'0');
        if(len > 1)
            ans += sum_digits(3*(s[len-2]-'0'));
        for(i=len-3;i>=0;--i)
            ans += sum_digits((s[i]-'0')*9);
        while(ans/10)
            ans = sum_digits(ans);
        cout << ans << endl;
    }
    return 0;
}
