int v [ 1009999 ] ;
const int MOD = 1000000000 + 7 ;
int fact [ 1009999 ] ;
inline void prep ( int n )
{
    fact [ 0 ] = 1 ;
    FORN ( i , 1 , n )
        fact [ i ] = ( 1LL * fact [ i - 1 ] * i ) % MOD ;
}
inline int ridic ( int x , int y )
{
    int ret = 1 ;
    for ( ; y ; y >>= 1 )
    {
        if ( y & 1 )
        {
            ret = ( 1LL * ret * x ) % MOD ;
        }
        x = ( 1LL * x * x ) % MOD ;
    }
    return ret ;
}
inline int comb ( int x , int y )
{
    int kkt1 = ridic ( fact [ x - y ] , MOD - 2 ) % MOD ;
    int kkt2 = ridic ( fact [ y ] , MOD - 2 ) % MOD ;
    int S = 1LL * kkt1 * kkt2 % MOD ;
    return 1LL * fact [ x ] * S % MOD ;
}
