#include <iostream>
using namespace std;
#include <cstring>
int main()
{
	int t,i,count,n;
	char s[10000];
   cin>>t;
   while(t--)
   {
   	count=0;
   	cin>>s;
   	
   	for(i=0;i<=(strlen(s)-1)/2;i++)
   	{
   		
   	
   		if(s[i]>=s[strlen(s)-i-1])
   		{
   			count=count+s[i]-s[strlen(s)-i-1];
   		}
   		else if(s[i]<s[strlen(s)-i-1])
   		{
   			count=count+s[strlen(s)-i-1]-s[i];
   		}
   	
   	}
   	cout<<count<<endl;
   }
    return 0;
}
