#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pii pair<int,int>
#define pll pair<ll,ll>
#define mp make_pair
#define ff first
#define ss second
#define pb push_back
#define pf pop_front
#define IOS ios::sync_with_stdio(false)
int n ;
string s;
int m[10000006];
char rep(char ch){
    if(ch=='W')return 'B';
    else return 'W';
}
int vis[10000005];
int res;
int main(){
 
	int i , j , k ;
    cin >> n >> s;
    for(i = 0 ; i < s.size() ; i++){
        if(s[i]=='B')res+=(1<<i);
    }
    ll black = (1LL<<n)-1;
    ll white = 0;
    m[res] = 0;
    if(res==black || res==white){
        cout << 0 ;
        return 0;
    }
    queue <int> q;
    vis[res] = 1;
    q.push(res);
    while(!q.empty()){
        int el = q.front();
        int temp = el;
        q.pop();
        for(i = 0 ; i < n ; i++){
            temp = el;
            if(i >= 1){
                temp^=(1<<i)^(1<<(i-1));
                if(vis[temp]==0){
                    m[temp] = m[el] + 1;
                    if(temp==black || temp==white){
                        cout << m[temp] << endl;
                        return 0;
                    }
                    q.push(temp);
                    vis[temp] = 1;
                }
            }
            if(i >= 2){
                temp = el;
                temp^=(1<<i)^(1<<(i-1))^(1<<(i-2));
                if(vis[temp]==0){
                    m[temp] = m[el] + 1;
                    if(temp==black || temp==white){
                        cout << m[temp] << endl;
                        return 0;
                    }
                    q.push(temp);
                    vis[temp] = 1;
                }
            }
        }
    }
    cout << min(m[black],m[white]);
 
	return 0;
}
