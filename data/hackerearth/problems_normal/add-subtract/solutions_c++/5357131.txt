#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	int min=INT_MAX;
    	int n,k;
    	cin>>n>>k;
    	assert(n<=100&&n>=0);
    	assert(k<=100&&k>=0);
    	int i;
    	int value,sum=0;
    	vector<int> a(n);
    	vector<int> x(n,0);
    	for(i=0;i<n;i++){
    		
    		cin>>a[i];
    		assert(a[i]>=0&&a[i]<=100);
    		
    	}
    	
    	for(value=0;value<=100;value++){
    		sum=0;
    		for(i=0;i<n;i++){
    			if(a[i]>value)
    			x[i]=(a[i]-value)*5;
    			else if(a[i]<value)
    			x[i]=(value-a[i])*3;
    			else
    			x[i]=0;
    			
    		}
    		sort(x.begin(),x.end());
    		for(int j=0;j<k;j++){
    			sum=sum+x[j];
    		}
    		
    		
    		if(min>sum)
    		min=sum;
    		
    		  	}
    	
    	
    	cout<<min;
    	cout<<endl;
    	
    	
    }
    return 0;
}
