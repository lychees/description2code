#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,i,swaps,p,q,x,y;
    char s[100005],s1[100005],s2[100005];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        scanf("%s",s);
        p=q=x=y=0;
        for(i=0;i<n;i++)
        {
            if(s[i]=='0')
                p++;
            else
                q++;
        }
        for(i=0;i<n;i++)
        {
            if(i>p-1)
                s1[i]='1';
            else
                s1[i]='0';
                //cout<<"a";
        }
        for(i=0;i<n;i++)
        {
            if(i>q-1)
                s2[i]='0';
            else
                s2[i]='1';

        }
        for(i=0;i<n;i++)
        {
            if(s[i]!=s1[i])
                x++;
            if(s[i]!=s2[i])
                y++;
        }
        //cout<<x<<"\t"<<y<<"\n";
        if(x>y)
            swaps=y;
        else
            swaps=x;
        //cout<<swaps<"\n";
        swaps=ceil(1.0*swaps/2);
       // cout<<s1<<"\t"<<s2<<"\n";
        cout<<swaps<<"\n";
        //cout<<p<<"\t"<<q<<"\n";
    }
    //cout<<s1<<"\n"<<s2;

    return 0;
}
