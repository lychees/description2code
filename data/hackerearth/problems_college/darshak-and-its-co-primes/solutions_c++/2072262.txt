#include <bits/stdc++.h>
using namespace std;
long long int gcd(long long int m,long long int n)
{
	if(m==0)
	{
		return n;
	}
	else
	{
		return (gcd(n%m,m));
	}
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long int a,b;
    	long long int flag=0;
    	scanf("%lld %lld",&a,&b);
    	flag=gcd(a,b);
    	if(flag==1)
    	{
    		printf("Is a Co-Prime\n");
    	}
    	else
    	{
    		printf("Not a Co-Prime\n");
    	}
    }
    return 0;
}
