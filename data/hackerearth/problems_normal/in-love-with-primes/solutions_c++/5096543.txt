#include<bits/stdc++.h>
using namespace std;
#define N 10000001
bool prime[N];

void sieve();

main()
{
  int t,n,k;
  cin>>t;
  sieve();
  while(t--)
  {
    cin>>n;
    for(int i=2;i<=n;i++)
    {
      if(prime[i] && prime[n-i])
      {
        k=1;
        cout<<"Deepa"<<endl;
        break;
      }
    }
    if(k!=1)
    cout<<"Arjit"<<endl;
    k=0;
  }
}



void sieve()
{
  memset(prime,true,sizeof(prime));
  prime[0]=false;
  prime[1]=false;

  for(int p=2;p*p<=N;p++)
  {
    if(prime[p]==true)
    {
      for(int i=p*2;i<=N;i+=p)
      prime[i]=false;
    }
  }
}
