T = int(raw_input())
while T>0:
	line = map(int, raw_input().strip().split(" "))
	A = line[0]
	B = line[1]
	count = 0
	while A>0 or B>0:
		if((A%10)+(B%10))>=10:
			count = count + 1
		A = (int)(A/10)
		B = (int)(B/10)
		
	if count == 0:
		print "No carry operation"
	if count == 1:
		print "1 carry operation"
	if count > 1:
		print "%d carry operations" % count
	T = T-1
