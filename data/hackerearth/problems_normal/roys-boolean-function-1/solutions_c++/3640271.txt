#include <iostream>
using namespace std;

long gcd ( long a, long b );
int main() {
	long t,a,b,c,z,i,q;
	cin>>t;
	while(t--)
	{
		cin>>a;
		if(a==1)
		cout<<"FALSE"<<endl;
		else
		{
		z=0;
		q=0;
		for(b=1;b<a;b++)
		{  c=gcd(a,b);
			if(c==1)
			z++;
		}
		if(z==1)
		cout<<"FALSE"<<endl;
		else
		{
		for(i=2;i<z;i++)
		{
			if(z%i==0)
			{
				q++;
				
			}
		}
		if(q>0)
		cout<<"FALSE"<<endl;
		else
		cout<<"TRUE"<<endl;
	}
	}}
}
long gcd ( long a, long b )
{
  long c;
  while ( a != 0 ) 
  {
     c = a; a = b%a;  b = c;
  }
  return b;
}
