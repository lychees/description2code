from sys import stdin
t = int(stdin.readline())


def com(a, b):
    l1 = len(a)
    l2 = len(b)
    l = l1

    if l2 < l:
        l = l2
    while l >= 1:
        # print "comparing ",a[-l:],b[:l]
        if a[-l:] == b[:l]:
            return l
        l -= 1
    return 0


for tt in xrange(t):
    a = stdin.readline().strip()
    b = stdin.readline().strip()
    fir = com(a, b)

    print a + b[fir:]
