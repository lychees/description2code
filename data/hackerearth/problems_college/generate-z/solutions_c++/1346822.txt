#include <iostream>
#include <stdio.h>
#include <stdlib.h>
using namespace std;

int main(void) {
    
    int N = 0, i = 0, j = 0, k = 0;
    
    scanf("%d",&N);
    
    for(i = 0; i < N; ++i)
    	printf("*");
    for(i = 1; i < N - 1; ++i) {
    	printf("\n");
    	for(j = N - i - 1; j > 0; --j)
    		printf(" ");
    	printf("*");
    }
    printf("\n");
    for(i = 0; i < N; ++i)
    	printf("*");
    return 0;
}
