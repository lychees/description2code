#include<bits/stdc++.h>

#define slong(a) scanf("%lld",&a)
#define all(c) c.begin(),c.end()
#define r_all(c) (c).rbegin(),(c).rend()
#define iter(c,i) for(typeof((c).begin()) it=(c).begin();it!=(c).end();it++)
#define pb(c) push_back(c)
typedef long long int ll;
using namespace std;
vector<long long int> vec;
int main()
{
    ll t,n,m,tmp,i,a,b,anss,l,r,sz;
    string str;
    cin>>t;
    while(t--)
    {
        cin>>str;
        cin>>a>>b;
        anss=0;
        sz=str.length();
        l=0;
        r=sz-1;
        while(l<r)
        {
            if(str[l]==str[r])
            {
                if(str[l]=='/')
                    anss+=2*min(a,b);
            }
            else
            {
                if(str[l]=='/')
                {
                    if(str[r]=='a')
                    {
                        anss+=a;
                    }
                    else
                        anss+=b;
                }
                else if(str[r]=='/')
                {
                    if(str[l]=='a')
                    {
                        anss+=a;
                    }
                    else
                        anss+=b;
                }
                else
                {
                    anss=-1;
                    break;
                }
            }
            l++;
            r--;
        }
        cout<<anss<<endl;
    }
    return 0;
}
