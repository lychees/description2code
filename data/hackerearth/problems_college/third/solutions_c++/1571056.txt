#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define f first
#define s second
#define MOD 1000000007
#define MAX 1000000
using namespace std;
int A[1000000+10],dp[1000000+10];
int main()
{
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(int i=0;i<n;i++)
            {
                scanf("%d",&A[i]);
                dp[i]=0;
            }
            dp[n]=0;
            dp[1]=A[0];
            for(int i=2;i<=n;i++)
                dp[i]=max(dp[i-1],A[i-1]+dp[i-2]);
            printf("%d\n",dp[n]);
    }
    return 0;
}
