{
	int t;

		int a,b,ans=0;

		if(a%2) aodd=1; else aeven=1; 
		if(b%2) bodd=1; else beven=1;
		
		if(a==b || a-b==2) {
		if(aeven&&beven) {ans=a+b; cout<<ans<<endl;}
