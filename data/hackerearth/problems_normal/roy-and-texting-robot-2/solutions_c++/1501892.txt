LL power(LL b,LL e)
{
	if(b==0 && e==0) return 1;
	if(b==0 && e>1)  return 0;	
	if(b>0 && e==0) return 1;
	LL r=1;
	while(e)
	{
		if(e&1) { r = r*b;  r %= MOD; }
		e >>= 1; b = b*b; b %= MOD;
		
	}
return r;
}

int keynum(char c)
{

		return c - 48;


	if(c>='a' && c<='c') return 2;

	if(c>='d' && c<='f') return 3;

	if(c>='g' && c<='i') return 4;

	if(c>='j' && c<='l') return 5;

	if(c>='m' && c<='o') return 6;

	if(c>='p' && c<='s') return 7;

	if(c>='t' && c<='v') return 8;

	if(c>='w' && c<='z') return 9;
	
	return 0;

}
