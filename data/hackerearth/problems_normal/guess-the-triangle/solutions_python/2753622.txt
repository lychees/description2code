def solve(cord):
	res = []
	x = cord[0][0] + cord[1][0] + cord[2][0]
	y = cord[0][1] + cord[1][1] + cord[2][1]
	res.append((x-2*cord[2][0], y-2*cord[2][1]))
	res.append((x-2*cord[1][0], y-2*cord[1][1]))
	res.append((x-2*cord[0][0], y-2*cord[0][1]))
	res.sort(key= lambda x : (x[0], x[1]))
	for x in res:
		print '%1.4f %1.4f'%(x[0], x[1])
		
if __name__ == '__main__':
	cords = []
	for i in range(3):
		cords.append(map(float, raw_input().split(' ')))
	solve(cords)
