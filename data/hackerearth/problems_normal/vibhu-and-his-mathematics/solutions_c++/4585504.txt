#include <iostream>
#include <cstdio>
#define ll long long int
#define di (1000000000+7)
using namespace std;
ll dp[1000007];



int main()
{
    ll t;
    scanf("%lld",&t);
    dp[0]=1;
    dp[1]=1;
    for(int i=2;i<1000004;i++)
    	dp[i]=dp[i-1]%di+(i-1)*dp[i-2]%di;

    while(t--)
    {
        
    	ll n;
    	scanf("%lld",&n);
        cout<<dp[n]%di<<endl;
    }
    
    return 0;
}
