#include <iostream>
#include <string>
using namespace std;

int main()
{
	int n;
    string a;
    scanf("%d",&n);
    while(n--)
    {
    cin>>a;
    for(int i=0;i<a.size();i++)
    {
    	if(a[i]+i<='z')
    printf("%c",a[i]+i);
        else
    printf("%c",a[i]+i-26);
    }
    printf("\n");
    }
    return 0;
}
