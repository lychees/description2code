#include <iostream>
using namespace std;
#define MAX 1000000007

long long int gcd(long long int a,long long int b) {
    return b == 0 ? a : gcd(b, a % b);
}

long long int exp(long long int a,long long int b){
	if(b==0)
	return 1;
	else if(b%2==0){
		return exp((a*a)%MAX,b/2)%MAX;
	}else{
		return ((a%MAX)*(exp((a*a)%MAX,(b-1)/2)%MAX))%MAX;
	}
}

int main()
{
   long long int n,i,gx=0,fx=1,tmp;
   cin>>n;
   for(i=0;i<n;++i){
   	cin>>tmp;
   	gx = gcd(tmp,gx);
   fx = ((fx%MAX)*(tmp%MAX))%MAX;
   	
   }
   
   cout<<exp(fx,gx)%MAX;
    return 0;
}
