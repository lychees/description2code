#include <iostream>
using namespace std;

int n;
string s;
int mx()
{
	int c=0;
	for(int i=0;i<s.size();++i)
	{
		if(s[i]=='w')
			++c;
	}
	c += s.size();
	return c;
}

int mn()
{
	int c=s.size();
	for(int i=0;i<s.size();++i)
	{
		if(s[i]=='v')
		{
			int j=i+1;
			while(j<n&&s[j]=='w')
				++j;
			if(j==n)
				break;
			if(s[j]=='v')
			{
				c -= 1;
				i=j;
			}
		}
	}
	return c;
}
int main()
{
    cin>>n>>s;
    cout<<mn()<<' '<<mx();
    return 0;
}
