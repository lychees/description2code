#include <bits/stdc++.h>
using namespace std;

int v[101][101];
int a[101][101];
int b, c, q;
int x, y, z;

void dfs(int i, int j) {
	v[i][j] = true;
	if(i + 1 < b && v[i + 1][j] == false && fabs(a[i][j] - a[i + 1][j]) <= z) 
	{
		dfs(i + 1, j);
	}
	if(i - 1 >= 0 && v[i - 1][j] == false && fabs(a[i][j] - a[i - 1][j]) <= z) 
	{
		dfs(i - 1, j);
	}
	if(j + 1 < c && v[i][j + 1] == false && fabs(a[i][j] - a[i][j + 1]) <= z) 
	{
		dfs(i, j + 1);
	}
	if(j - 1 >= 0 && v[i][j - 1] == false && fabs(a[i][j] - a[i][j - 1]) <= z) 
	{
		dfs(i, j - 1);
	}
}

int main() {
	
	scanf("%d%d%d", &b, &c, &q);
	for(int i = 0; i < b; i++) 
	{
		for(int j = 0; j < c; j++) 
			scanf("%d", &a[i][j]);
	}
	while(q--) 
	{
		scanf("%d%d%d", &x, &y, &z);
		for(int i = 0; i <= 100; i++) 
		{
			for(int j = 0; j <= 100; j++) 
				v[i][j] = false;
		}
		
		x--; 
		y--;
		dfs(x, y);
		int count = 0;
		for(int i = 0; i <= b; i++) {
			for(int j = 0; j <= c; j++) 
				if(v[i][j]) 
					count++;
		}
		printf("%d\n", count);
	}
}
