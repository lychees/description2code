T = int(raw_input())
count = T
res = []
while count != 0:
    # input next line containing N K M
    N, K, M = map(int, raw_input().split(' '))
    i = N
    string_list = []
    while i > 0:
        string_list.append(raw_input())
        i = i-1
    sub_str = []

    for element in string_list:
        sub_str.append(element[:M])

    sub_str.sort()

    for element in string_list:
        if element.startswith(sub_str[K-1]):
            res.append(element)
            break
    count = count-1

counter = 0
while counter < T:
    print res[counter]
    counter = counter+1
