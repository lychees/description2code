#include<bits/stdc++.h>
#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	for(int j=1;j<=t;j++)
	{
		unsigned long long int x,k;
		cin>>x>>k;
		long long int c=0;
		unsigned long long int y=1;
		while(c<=k)
		{
			long long int z = x+y;
			long long int b = x^y;
			if(z == b)
			{
				c++;
				if(c==k)
				{
					cout<<"Case #"<<j<<": "<<y<<"\n";
					break;
				}
				y++;
			}
			else
			{
				y++;
			}
		}
	}
}
