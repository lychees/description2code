#include <iostream>
using namespace std;
#define lim 510
#define mod 1000000007
long long a[lim][lim],n,h, sol[lim][lim];

int main() {
	for(h=1; h<lim; h++) {
		a[0][h]=1;
		a[1][h]=1;
	}
	for(int n=1; n<lim; n++) {
		a[n][0]=0;
	}
	
	for(int n=2; n<lim; n++) {
		for(int h=2;h<lim;h++) {
			if(h>n) {
				a[n][h]=a[n][h-1];
				continue;
			}
			a[n][h]=0;
			for(int k=1;k<=n;k++) {
				long long variable = (a[k-1][h-1]*a[n-k][h-1])%mod;
				a[n][h] = (a[n][h]+variable)%mod;
			}
		}
	}
	
	for(int n=2; n<lim; n++) {
		for(int h=2;h<lim;h++) {
			if(h>n) {
				sol[n][h]=sol[n][h-1];
				continue;
			}
			sol[n][h]=0;
			for(int k=1;k<=n;k++) {
				long long variable = (a[k-1][h-1]*a[n-k][h-1])%mod;
				variable = variable*k;
				sol[n][h] = (sol[n][h]+variable)%mod;
			}
		}
	}
	cin>>n>>h;
	cout<<sol[n][h]%mod<<endl;
	return 0;
}
