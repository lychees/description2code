    #include<iostream>
    using namespace std;
    int dp[1000005];
    int main() {
        
        int t;
        cin>>t;
        dp[1]=-1;
        dp[2]=1;
        dp[3]=1;
        dp[4]=2;
        dp[5]=1;
        dp[6]=2;
        dp[7]=1;
        dp[8]=2;
        for(int i = 9; i<=1000000; i++)
        {
        	dp[i]=min(dp[i-2],dp[i-3]);
        	dp[i]=min(dp[i],dp[i-5]);
        	dp[i]=min(dp[i],dp[i-7]);
        	dp[i]++;
        }
        
        
        
        while(t--)
        {
        	int x;
        	cin>>x;
        	cout<<dp[x]<<endl;
        	
        }
    	return 0;
    }
