#include <iostream>
using namespace std;

int main()
{
    string str;
    getline(cin, str);
    int len = str.length();
    for(int i=0; i<len; i++)
    {
    	char chr = str[i];
    	int charcode = int(chr);
    	if(charcode >= 65 && charcode <= 90 )
    	{
    		chr = tolower(chr);
    	}
    	else
    	{
    		chr = toupper(chr);
    	}
    	cout << chr;
    }
    return 0;
}
