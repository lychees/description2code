def prime(a):
    if (a == 2 or a == 3 or a == 5 or a == 7):
        return True
    if (a % 2 == 0 or a % 3 == 0 or a % 5 == 0 or a % 7 == 0):
        return False
    for g in range(2, a):
        if (a % g == 0):
            return False
    return True
