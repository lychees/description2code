#include <bits/stdc++.h>
using namespace std;

char s[1000][1000];
int a[100000];

int main()
{
	int i,j,c,n,m;
    cin>>n>>m;
    for(i=0;i<n;i++)
    	scanf("%s",s[i]);
    
    for(i=0;i<m;i++){
    	int z=0;
    	for(j=0;j<n;j++){
    		z=max(z,s[j][i]-'0');
    	}
    	for(j=0;j<n;j++){
    		if(s[j][i]-'0'==z)
    			a[j]=1;
    	}
    }
    c=0;
    for(i=0;i<n;i++)
    	if(a[i]==1)
    		c++;
    cout<<c<<endl;
    return 0;
}
