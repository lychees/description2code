#include<bits/stdc++.h>
using namespace std;
#define debug(x) cerr<<#x<<" "<<x<<endl;
#define debug1(x,y) cerr<<#x<<" "<<x<<#y<<" "<<y<<endl;
#define MAX1 100000+10
#define MAX2 1000000+10
string s1,s2;
int main()
{
    int t,count,mini;
    scanf("%d",&t);
    while(t--)
    { count=0; mini=1000000000;
        cin>>s1>>s2;
        int len1=s1.length();
        int len2=s2.length();
        for(int i=0;i<=len1-len2;i++)
        {  count=0;
            for(int j=0;j<len2;j++)
            { // cout<<s1[i+j]<<" " <<s2[j]<<endl;
                if(s1[i+j]!=s2[j])
                   { count++; }
            }
      //      cout<<s1[i]<<" ";
            mini=min(mini,count);
        }
        printf("%d\n",mini);
    }
return 0;
}
