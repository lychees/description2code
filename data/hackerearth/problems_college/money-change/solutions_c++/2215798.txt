#include <cmath>
#include <cstdio>
#include <vector>
#include <string.h>
#include <iostream>
#include <algorithm>
using namespace std;
unsigned long long DP[300];
int main() {
    int n,m,num;
    memset(DP,0,sizeof(DP));
    scanf("%d %d",&n,&m);
    DP[0]=1;
    vector<int> coins;
    for(int i=1 ; i<=m ; i++){
        scanf("%d",&num);
        coins.push_back(num);
    }
    sort(coins.begin(),coins.end());
    for(int i=0 ; i<m; i++){
        int idx=0;
        while(idx+coins[i]<300){
            DP[idx+coins[i]]+=DP[idx];
            idx++;
        }
    }
    cout<<DP[n];
    return 0;
}
