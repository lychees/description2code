#include <iostream>
using namespace std;


void rev(string s, int i){
	
	if(s[i]=='\0')
		return;
	rev(s, i+1);
	
	cout<<s[i];
}

void reverse(string s){
	rev(s,0);
}
int main()
{
    int tc;
    cin>>tc;
    string s;
    while(tc--){
    	cin>>s;
    	reverse(s);
    	cout<<endl;
    }
    
    return 0;
}
