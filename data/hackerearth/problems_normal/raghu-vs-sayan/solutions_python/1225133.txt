for t in xrange(int(raw_input())):
	r, s, N = map(int,raw_input().split())
	vals = map(int,raw_input().split())
	vals.sort()
	rcount, scount = 0,0
	rsum, ssum = 0,0
	for i in xrange(len(vals)):
		rsum += vals[i]
		if rsum > r:
			break
		rcount += 1
	for i in xrange(len(vals)):
		ssum += vals[i]
		if ssum > s:
			break
		scount += 1
	if rcount == scount:
		print 'Tie'
		continue
	print 'Sayan Won' if scount > rcount else 'Raghu Won'
