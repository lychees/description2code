#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    //cout << "Hello World!" << endl;
    ll t;
    cin>>t;
    while(t--){
    	ll n;
    	cin>>n;
    	ll t=n,count2=0,count5=0;;
    	ll power2=1,power5=1;
    	for(ll i=1;;i++){
    		power2*=2;
    		count2+=n/power2;
    		if(n/power2==0)break;
    	}
    	for(ll i=1;;i++){
    		power5*=5;
    		//t/=power5;
    		count5+=t/power5;
    		if(t/power5==0)break;
    	}
    	cout<<min(count2,count5)<<"\n";
    }
    return 0;
}
