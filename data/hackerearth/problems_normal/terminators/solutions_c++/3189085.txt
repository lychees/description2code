#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1e3+2;
int dp[MAXN][MAXN];
int seq1[MAXN];
int seq2[MAXN];

int main() {
	int t;
	cin>>t;
	while(t--) {
		int n;
		scanf("%d",&n);
		for(int i=1;i<=n;i++) {
			int a,b;
			scanf("%d%d",&a,&b);
			seq1[a]=i;
			seq2[b]=i;
		}
		for(int i=0;i<=n;i++)
			dp[0][i]=dp[i][0]=0;
		for(int i=1;i<=n;i++) {
			for(int j=1;j<=n;j++) {
				if(seq1[i]==seq2[j])
					dp[i][j]=1+dp[i-1][j-1];
				else
					dp[i][j]=0;
				dp[i][j]=max(dp[i-1][j],dp[i][j]);
				dp[i][j]=max(dp[i][j-1],dp[i][j]);
			}
		}
		printf("%d\n",dp[n][n]);
	}
	return 0;
}
