#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

ll arr[1003];
int main()
{
    int cases;
    scanf("%d",&cases);
    while(cases--)
    {
        int n;
        ll a,b,sum =0;
        scanf("%d",&n);
        for(int i=0; i<n; i++)
        {
            scanf("%lld",&a);
            sum += a;
        }
        a = sum/n;
        b = sum%n;
        if(b == 0)
            printf("YES\n%lld %lld\n",a,b);
        else
            printf("NO\n%lld %lld\n",a,b);
    }

return 0;
}
