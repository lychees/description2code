#include <bits/stdc++.h>
using namespace std;

#define sd(a) scanf("%d", &a)
#define ss(a) scanf("%s", &a)
#define sl(a) scanf("%lld", &a)
#define clr(a) memset(a, 0, sizeof(a))
#define debug(a) printf("check%d\n", a)
#define F first
#define S second
#define PB push_back
#define ll long long

map<int, map<int, int> > mp;

pair<int, int> subs[100000];

int main()
{
	// ios_base::sync_with_stdio(false);
	// freopen("input.in", "r", stdin);
	int t;
	pair<int, int> p;
	sd(t);
	while ( t-- )
	{
		int m, n, great = 0, good = 0, G, H;
		sd(m); sd(n);
		
		for ( int j = 0 ; j < m ; j++ )
		{
			sd(p.first); sd(p.second);
			subs[j] = p;
		}
		// for ( int j = 0 ; j < m ; j++ )
		// {
		// 	p = subs[j];
		// 	printf("%d %d\n", p.first, p.second);
		// }
		for ( int i = 0 ; i < n ; i++ )
		{	
			sd(p.first); sd(p.second);
			mp[p.first][p.second] += 1;
		}
		sd(G); sd(H);
		for ( int i = 0 ; i < m ; i++ )
		{
			p.first = subs[i].first; p.second = subs[i].second;
			if ( mp.count(p.first) )
			{
				if ( mp[p.first].count(p.second) )
				{
					if ( mp[p.first][p.second] )
					{
						great++;
						mp[p.first][p.second]--;
						if ( mp[p.first][p.second] == 0 )
							mp[p.first].erase(p.second);
					}
					else
					{
						good++;
					}
				}
			}
		}
		if ( great >= H )
		{
			printf("Great\n");
		}
		else if ( good >= G )
		{
			printf("Good\n");
		}
		else
			printf(":(\n");
	}
}
