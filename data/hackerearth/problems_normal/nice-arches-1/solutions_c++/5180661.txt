#include <bits/stdc++.h>
using namespace std;

int main()
{
 
   int t;cin>>t;
   int ans=0;
   while(t--){
   stack<char>st;
       	string s;cin>>s;
       	int n=s.size();
       	for(int i=0;i<n;i++){
       		if(!st.empty() && s[i]==st.top())
       		st.pop();
       		else
       		st.push(s[i]);
       	}
       	ans=ans+(st.size()==0);
   }  
   cout<<ans<<endl;
   return 0;
}
