#include <bits/stdc++.h>
using namespace std;
int is(int m)
{
	int n;
	set<int> ss;
	for(n=2;n*n<=m;n++)
	{
		if(m%n==0)
		{
			ss.insert(n);
			while(m%n==0)m/=n;
		}
	
	}
	if(m!=1)ss.insert(m);
	return ss.size()>2;
}
int main()
{
   // cout << "Hello World!" << endl;
   vector<int> sol;
   int n = 6;
   while(sol.size()<=1001)
   {
   	if(is(n))sol.push_back(n);
   	n++;
   }
   int t;
   scanf("%d",&t);
   while(t--)
   {
   	scanf("%d",&n);
   	printf("%d\n",sol[n-1]);
   }
    return 0;
}
