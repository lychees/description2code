#include <iostream>
using namespace std;
typedef long long ll;
int main()
{
   // cout << "Hello World!" << endl;
   ll t;
   cin>>t;
   while(t--)
   {
   		int n;
   		cin>>n;
   		ll sum=0;
   		while(n>9)
   		{
   			sum=0;
   			while(n!=0)
   			{
   				sum+=(n%10);
   				n/=10;
   			}
   			n=sum;
   		}
   		cout<<n<<endl;
   }
    return 0;
}
