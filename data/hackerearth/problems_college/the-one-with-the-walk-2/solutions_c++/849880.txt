#include<iostream>

using namespace std;

int main()
{
	long int T;
	cin>>T;
	int a[T], b[T], c[T], d[T];
	int maxx=0,maxy=0;

	for(int i=0;i<T;i++)
	{
		cin>>a[i];
		cin>>b[i];
		
		cin>>c[i];
		c[i]-=a[i];
		if(c[i]>maxx)
			maxx=c[i];

		cin>>d[i];
		d[i]-=b[i];
		if(d[i]>maxy)
			maxy=d[i];
		//cout<<c[i]<<":"<<d[i]<<":"<<endl;
	}
   	long int table[maxx+1][maxy+1];
		
	for(int x=0;x<=maxx;x++)
	{
		for(int y=0;y<=maxy;y++)
		{
			if(x==0 || y==0)
				table[x][y]=1;
			else
			{
				table[x][y] = table[x-1][y] + table[x][y-1] + table[x-1][y-1];
				
				if(table[x][y] > 1000000007)
					table[x][y] = table[x][y] % 1000000007;
			}
		}
	}
	for(int j=0;j<T;j++)
	{	
		cout<<table[c[j]][d[j]]<<endl;

	}
}
