#include <iostream>
using namespace std;

int main()
{
    long long t;
    long long n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(n==1)
    	cout<<1<<endl;
    	else if(n==2)
    	cout<<1<<endl;
    	else if(n%3==0)
    	cout<<n/3<<endl;
    	else
    	cout<<n<<endl;
    }
    return 0;
}
