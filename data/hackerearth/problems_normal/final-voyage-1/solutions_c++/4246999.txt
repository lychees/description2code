#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long t,n,w,i,j;
	cin>>t;
	while(t--)
	{
		cin>>n>>w;
		long long dp[n+1][w+1],wt[n],val[n];
		for(i=0;i<n;i++)
		cin>>wt[i];
		for(i=0;i<n;i++)
		cin>>val[i];
		for(i=0;i<=n;i++)
		{
			for(j=0;j<=w;j++)
			{
				if(j==0||i==0)
				dp[i][j]=0;
				else if(wt[i-1]>j)
				dp[i][j]=dp[i-1][j];
				else
				dp[i][j]=max(dp[i-1][j],val[i-1]+dp[i-1][j-wt[i-1]]);
			}
		}
		cout<<dp[n][w]<<endl;
	}
    return 0;
}
