
#include <iostream>
#include<math.h>
#include<stdlib.h>
using namespace std;

int main()
{
                 long long int n,x,l,ans;
                 cin>>n;
                 while(n--)
                 {
                 	 cin>>x;
                 	 l=floor(log2(x));
                 	 ans=2*(x-pow(2,l))+1;
                 	 cout<<ans<<endl;
                 }
    return 0;
}
