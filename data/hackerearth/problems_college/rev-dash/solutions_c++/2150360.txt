#include<iostream>
using namespace std;
int main()
{
    string st;
    cin>>st;
    int sz;
    sz=st.length();
    for(int i=0;i<sz;i++)
    {
        cout<<st[i];
        if(i!=sz-1)
            cout<<"-";
    }
    cout<<endl;

    for(int i=sz-1;i>=0;i--)
    {
        cout<<st[i];
        if(i!=0)
            cout<<"-";
    }
    return 0;
}
