#include<bits/stdc++.h>
#define M 1000000007
#define ll long long
using namespace std;
ll mypow(int a, int b, int MOD)
{
ll x=1,y=a;
while(b > 0)
{
if(b%2 == 1)
{
x=(x*y);
if(x>MOD) x%=MOD;
}
y = (y*y);
if(y>MOD) y%=MOD;
b /= 2;
}
return x;
}
ll myEuler(int n, int MOD)
{
return mypow(n,MOD-2,MOD);
}
int myFact(int n, int p)
{
int k=0;
while (n>=p)
{
k+=n/p;
n/=p;
}
return k;
}


ll factMOD(int n, int MOD)
{
ll res = 1;
while (n > 0)
{
for (int i=2, m=n%MOD; i<=m; i++)
res = (res * i) % MOD;
if ((n/=MOD)%2 > 0)
res = MOD - res;
}
return res;
}

ll C(int n, int r, int MOD)
{
if (myFact(n, MOD) > myFact(r, MOD) + myFact(n-r, MOD))
return 0;

return (factMOD(n, MOD) *
((myEuler(factMOD(r, MOD), MOD) *
myEuler(factMOD(n-r, MOD), MOD)) % MOD)) % MOD;
}

ll int fact[1000010];
int main(){
int t;

fact[0]=1;
for(int i=1;i<=1000000;i++)
{
fact[i]=(i*fact[i-1])%M;
}
cin>>t;
while(t--){
        string str1,str2;
ll int a1[26]={0},b1[26]={0},ans=1;
cin>>str1>>str2;
int l=str1.length();
for(int i=0;i<l;i++){
a1[str1[i]-'a']++;
}
l=str2.length();
for(int i=0;i<l;i++){
b1[str2[i]-'a']++;
}
for(int i=0;i<26;i++){
if(a1[i]>=b1[i]&&b1[i]!=0){
ans=(((ans*(C(a1[i],b1[i],M)))%M) * fact[b1[i]])%M;
}
else if(a1[i]<b1[i])
{
ans=0;break;
}

}
cout<<ans<<endl;
}
}
