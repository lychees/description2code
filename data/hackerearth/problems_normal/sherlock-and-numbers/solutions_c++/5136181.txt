#include <iostream>
#include <vector>
#include <algorithm>

#define ll long long int

using namespace std;

int main()
{
    int T;
    scanf("%d", &T);
    
    ll N, P, temp;
    int K;
    
    for(int t = 0; t < T; t++) {
    	ll count = 0;
    	scanf("%lld %d %lld", &N, &K, &P);
   		ll A[K];
   		for(int i = 0; i < K; i++) {
   			scanf("%lld", &A[i]);
   			if(A[i] <= P)
   				count++;
   		}
   		
   		if(P > N - K) {
   			cout << "-1" << endl;
   		}
   		
   		else {
   			while(1) {
		   		temp = P + count;
		   		count = 0;
		   		for(int i = P + 1; i <= temp; i++) {
		   			if(binary_search(A, A + K, i)) {
		   				count++;
		   			}
		   		}
		   		if(count == 0) {
		   			printf("%lld\n", temp);
		   			break;
		   		}
		   		
		   		else {
		   			P = temp;
		   		}
   			}
   		}
   		
    }
    return 0;
}
