#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
vector<int> arr;
int n;
long long prod(int in)
{
	long long p=1; int z=0;
	for(int i=in;i<n;i++)
	{
		if(arr[i]==0)
		{
			z++;
			continue;
		}
		p=p*arr[i];
	}
	if(z!=n-in)
	return p;
	else
	return 0LL;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	
    	cin>>n;
    	int k;
    	long long minp, maxp;
    	
    	for(int i=0;i<n;i++)
    	{
    		cin>>k;
    		arr.push_back(k);
    	}
    	sort(arr.begin(),arr.end());
    	int i=0,j=0,z=0;
    	if(arr.size()==1)
    	{
    		cout<<arr[0]<<" "<<arr[0]<<"\n";
    		arr.clear();
    		continue;
    	}
    	for(int p=0;p<n;p++)
    	{
    		if(arr[p]<0)
    		i++;
    		else
    		if(arr[p]>0)
    		j++;
    		else
    		z++;
    	}
    	if(i%2==0)
    	{
    		maxp=prod(0);
    		if(i!=0)
    		minp=maxp/arr[i-1];
    		else
    		minp=arr[0];
    	}
    	else
    	if(i%2!=0)
    	{
    		minp=prod(0);
    		if(z!=0 && j==0)
    		maxp=0;
    		else
    		maxp=minp/arr[i-1];
    	}
    	
    	cout<<maxp<<" "<<minp<<"\n";
    	arr.clear();
    }
    return 0;
}
