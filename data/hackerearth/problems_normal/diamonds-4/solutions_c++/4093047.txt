#include <iostream>
using namespace std;

int main()
{
    int t,n,m;
    cin >> t;
    while(t--){
    	cin >> n >> m;
    	char a[n+1][m+1];
    	for(int i = 0; i < n; i++){
    		for(int j = 0; j < m; j++){
    			cin >> a[i][j];
    		}
    	}
    	int count = 0;
    	for(int i = 0; i < n; i++){
    		for(int j = 0; j < m; j++){
    			if(a[i][j] == '/' && i < n-1 && j < m-1){
    				if(a[i][j+1] != '/' && a[i+1][j] != '/' && a[i+1][j+1] == '/'){
    					count++;
    				}
    			}
    		}
    	}
    	cout << count << endl;
    }
    return 0;
}
