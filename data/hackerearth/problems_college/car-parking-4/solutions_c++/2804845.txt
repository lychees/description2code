#include <iostream>
using namespace std;

int main() {
	int t;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>s;
		int n=s.length();
		int a[n],ans=0;
		for(int i=0;i<n;i++) a[i]=0;
		/*if(n>0)
		{
			if(s[0]=='D'||s[0]=='S'||s[0]=='B') a[0]=-1;
		}
		if(n>1)
		{
			if(s[1]=='D') a[1]=-1;
			if(s[1]=='B') { a[1]=-1; a[0]=-1;  }
			if(s[1]=='S') { a[1]=-1; a[0]=-1; if(n>2) a[2]=-1; }
			
		}*/
		for(int i=0;i<n;i++)
		{
			if(s[i]=='D') a[i]=-1;
			if(s[i]=='B') { a[i]=-1; a[i-1]=-1;  a[i-2]=-1; }
			if(s[i]=='S') { a[i]=-1; a[i-1]=-1;  a[i+1]=-1; }
		}
		for(int i=0;i<n;i++)
		{
			//cout<<a[i];
			if(a[i]==0) ans++;
		}
		cout<<ans<<endl;
	}
	return 0;
}
