from sys import stdin
n,m = map(int,stdin.readline().split())
a = map(int,stdin.readline().split())
def solve(x):
	tot = 0
	cur = 0
	for i in a:
		if i+cur<=x:
			cur+= i
		else:
			tot+=1
			cur = i
	if cur>0:
		tot+=1
	return tot<=m
lo = max(a)
hi = 10**18
while hi-lo > 1:
	mid = (lo+hi)/2
	if solve(mid):
		hi = mid
	else:
		lo = mid
ans = hi
if solve(lo):
	ans = lo
print ans
