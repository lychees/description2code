	[p,x,y,u,v] = str(raw_input()).split()
	[p,x,y,u,v] = [int(p),int(x),int(y),int(u),int(v)]
	a = abs(u-x)
	b = abs(v-y)
	if(a==b):
		if a==0 and p>=1:
			print "Counter-Terrorists Win !"
		elif a != 0 and p>1:
			print "Counter-Terrorists Win !"
		else:
			print "Terrorists Win !"
	else:
		if (a%2 == 1 and b%2 == 0) or (a%2 == 0 and b%2 == 1):
			print "Terrorists Win !"
		elif p<=2:
			print "Terrorists Win !"
		else:
			print "Counter-Terrorists Win !"
