#include <iostream>
using namespace std;

bool check(const int n)
{
	int no = 0;
	for (long i = 2; i*i <= n; ++i)
	{
		if (n == i*i) ++no;
		else if (n % i == 0) return true;
		if (no == 2) return true;
	}
		
	return false;
}

int main()
{
	long n, x;
	
	for (cin >> n; n--;)
	{
		cin >> x;
		
		if (check(x)) cout << "YES\n";
		else cout << "NO\n";
	}
	
	return 0;
}
