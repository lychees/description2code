struct node

};
int comparex(const void* a,const void* b)
{
	node* n1=(node*)a;
	node* n2=(node*)b;
	if(n1->x!=n2->x)
	  return (n1->x - n2->x);
	if(n1->x==n2->x) 
	  return (n1->y-n2->y);
}
int comparey(const void* a,const void* b)
{
	node* n1=(node*)a;
	node* n2=(node*)b;
	return (n1->y - n2->y);
}
