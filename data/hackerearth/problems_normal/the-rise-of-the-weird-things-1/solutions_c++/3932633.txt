#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,a[100005],b[100005],c=0,d=0,sum_a=0,sum_b=0,p;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	cin>>p;
    	if(p%2==0)
    		a[c++]=p,sum_a+=p;
    	else
    		b[d++]=p,sum_b+=p;
    }
    sort(a,a+c);
    sort(b,b+d);
    for(int i=0;i<c;i++)
    cout<<a[i]<<" ";
    cout<<sum_a<<" ";
    
    for(int i=0;i<d;i++)
    cout<<b[i]<<" ";
    cout<<sum_b<<endl;
    
    return 0;
}
