#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,k,c0,c1,t,i,ar[100005];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        scanf("%d",&k);
        c0=0;c1=0;
        for(i=0;i<n;i++)
            scanf("%d",&ar[i]);

        for(i=0;i<n;i++)
        {
            if(ar[i]==0)
                c0++;

        }
        for(i=0;i<n;i++)
        {
            if(ar[i]==1)
                c1++;
        }
        for(i=1;i<=k;i++)
        {
            if(c0>0)
            {
                c0--;c1++;
            }
            else
            {c1--;c0++;}
        }
        printf("%d\n",c1);
    }
    return 0;


}
