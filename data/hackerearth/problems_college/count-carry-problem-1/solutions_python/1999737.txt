for _ in range(input()):
    a, b = raw_input().split()
    a = map(int, list(a))
    b = map(int, list(b))
    i = len(a)-1
    j = len(b)-1
    flg = 0
    while(i>=0 and j>=0):
        if b[i]+a[j]>=10:
            flg+=1
        i-=1
        j-=1
    if flg==1:
        print flg,"carry operation"
    elif flg>1:
        print flg,"carry operations"
    else:
        print "No carry operation"
