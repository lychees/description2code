#include <iostream>
using namespace std;
 
int main()
{
    long long t,a[100001],i,val,sf,sum,n,j;
    cin>>t;
    while(t--)
    {
    	sum=0;
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		sum+=a[i];
    	}
    	cin>>val;
    	val=val%sum;
    	sf=0;
    	if(val==0)
    	{
    		for(j=n-1;a[j]==0;j--);
    		cout<<j+1<<endl;
    		goto end;
    	}
    	for(i=0;i<n;i++)
    	{
    		
    		sf+=a[i];
    		if(sf>=val)
    		break;
    	}
    	cout<<i+1<<endl;
    	end:;
    }
    return 0;
}
