#include<iostream>
#include<string.h>
using namespace std;
int main()
{
long long int n,q,l,r,sum,diff;
cin>>n>>q;
long long int a[n];
for(long long int i=0;i<n;i++)
cin>>a[i];
	for(long long int j=1;j<=q;j++)
	{
	cin>>l>>r;
	sum=0;
	diff=a[r-1];
	for(long long int i=l-1;i<r;i++)
	{
		sum+=a[i];
	}
	for(long long int i=r-2;i>=l-1;i--)
	{
		diff-=a[i];
	}
	cout<<sum<<" "<<diff<<endl;
}
	
	
}
