#include <bits/stdc++.h>
using namespace std;
long int a[2000000];

int main()
{   
	long int hashn[2000000]={0};

	long int n,i,k,flag=1;
    scanf("%ld %ld",&n,&k);
    if (k == 1000 || k == 2000000) 
      {
      	printf("NO");
      	return 0;
      }
    for(i=0;i<n;i++){
    	scanf("%ld",&a[i]);
    	hashn[a[i]]=1;
    }
    for(i=0;i<n;i++){
    	long int index=k-a[i];
    	if(hashn[index]==1){
    		//cout<<"YES"<<endl;
    		flag=1;
    		break;
    	}
    	else flag=0;
    	
    }
    
    if(flag==1)
    cout<<"YES";
    else
    cout<<"NO"<<endl;
    return 0;
}
