#include <iostream>
using namespace std;

long long count1(int n)
{
	if(n==0)
	return 0;
	if(n==1)
	return 1;
	
	long long dp[n+1];
	
	dp[0]=0;
	dp[1]=1;
	
	for(int i=2;i<=n;i++)
	{
		dp[i]=dp[i-1]+dp[i-2];
	}
	
	return dp[n];
}

long long count0(int n)
{
	if(n==0)
	return 1;
	if(n==1)
	return 0;
	
	long long dp[n+1];
	
	dp[0]=1;
	dp[1]=0;
	
	for(int i=2;i<=n;i++)
	dp[i]=dp[i-1]+dp[i-2];
	
	return dp[n];
}

int main()
{  int T;
cin>>T;
while(T--)
{  int N;
	cin>>N;
	
	cout<<count0(N)<<" "<<count1(N)<<endl;
}
    
    return 0;
}
