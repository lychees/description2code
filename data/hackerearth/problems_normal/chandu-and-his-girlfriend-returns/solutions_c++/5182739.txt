#include <bits/stdc++.h>
using namespace std;

int main(){

    int t;
    cin >> t;

    while(t--){
        int n, m, j = 0, k = 0, l = 0;
        cin >> n >> m;
        int a[n], b[m], c[n + m];

        for(int i = 0; i < n; i++){
            cin >> a[i];
        }

        for(int i = 0; i < m; i++){
            cin >> b[i];
        }

        while(j < n && k < m){
            if(a[j] >= b[k] ){
                c[l++] = a[j++];
            }
            else{
                c[l++] = b[k++];
            }
        }

        while(j < n){
            c[l++] = a[j++]; 
        }
        while(k < m){
            c[l++] = b[k++];
        }

        for(int i = 0; i < n + m; i++){
            cout << c[i] << " ";
        }
        
        cout << endl;

    }

    return 0;
}
