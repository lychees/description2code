#include <bits/stdc++.h>

using namespace std;

int pre[1110111];
int z[1110111];
int main()
{
	ios_base::sync_with_stdio(0);
	cin.tie (0);	cout.tie (0);
	int t;	cin >> t;
	while (t --){
		memset (z , 0 , sizeof (z));
		string a , b;	cin >> a >> b;
		string c = b;
		int n = a.size() , m = b.size();
		b += '&';	b += a;
		int l = 0,r = 0;
		for (int i = 1;i < b.size();i ++) {
			if (i <= r)	z[i] = min (r - i + 1 , z[i - l]);
			while (i + z[i] < b.size() && b[z[i]] == b[i + z[i]])
				++ z[i];
			if (r < i + z[i])	l = i , r = i + z[i] - 1;
		}
		int pre = 0;    
		for (int i = n + 1;i < b.size();i ++) {
			if (z[i] == b.size() - i){
				pre = z[i];
				break;
			}
		}
		for (int i = 0;i < n - pre;i ++)
			cout << a[i];
		cout << c;
		cout << endl;                               
	}
}
