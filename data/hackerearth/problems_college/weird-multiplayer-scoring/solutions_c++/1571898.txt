#include <iostream>
using namespace std;
#include <cstring>
#include <map>
int main()
{
 	long n, m;
 	cin>>n>>m;
 	map<string, long> board;
 	string winner = "dj";
 	long win_pts=-10000000;
 	while(m--)
 	{
		 string name;
		 long pts;
		 cin>>name>>pts;
		 if(board.find(name) == board.end())
		 {
		 	board[name] = pts;
		 	
		 	if(board[name]>win_pts)
		 	{
		 		win_pts = board[name];
		 		winner.assign(name);
		 	}
		 	//cout<<"naem and pts "<<name<<" "<<board[name]<<endl;
		 	//cout<<"winner till now "<<winner<<" "<<win_pts<<endl;
		 }
		 else
		 {
		 	board[name] += pts;
		 	
		 	if(board[name]>win_pts)
		 	{
		 		win_pts = board[name];
		 		winner.assign(name);
		 	}
		 	//cout<<"naem and pts "<<name<<" "<<board[name]<<endl;
		 	//cout<<"winner till now "<<winner<<" "<<win_pts<<endl;
		 }
		 
 	}
 	cout<<winner<<endl;
    return 0;
}
