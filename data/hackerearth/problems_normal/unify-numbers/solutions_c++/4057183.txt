#include <iostream>
#include<math.h>
#include<algorithm>
using namespace std;
long long mymod(long long num)
{
	if(num< 0)
	num=-num;
	return num;
}
int main()
{
	long long t,n,i,m,res;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	long long arr[n];
    	res=0;
    	for(i=0;i<n;i++)
    	cin>>arr[i];
    	sort(arr,arr+n);
    	
    	if(n&1)
    	m=arr[n/2];
    	else
    	m=(arr[n/2]+arr[n/2-1])/2;
    
    	for(i=0;i<n;i++)
    	res+=mymod(arr[i]-m);
    	cout<<res<<endl;
    }
    return 0;
}
