#include <bits/stdc++.h>
using namespace std;

#define ll long long int
#define mp(i,j) make_pair(i,j)
#define pb(x) push_back(x)
#define MAXN 1000010
#define MOD 1000000007

char str[MAXN];

int main()
{
	ll i,n,t,flag,mis1,mis2;

	scanf("%lld", &t);

	while(t--) {
		scanf("%lld%s", &n,str);
		
		n*=2;

		flag = 0;
		mis1 = 0;
		for(i=0;i<n;i++) {
			if(flag==0) {
				if(str[i]=='B') mis1++;
				flag = 1;
			} else {
				if(str[i]=='G') mis1++;
				flag = 0;
			}
		}

		flag = 0;
		mis2 = 0;
		for(i=0;i<n;i++) {
			if(flag==0) {
				if(str[i]=='G') mis2++;
				flag = 1;
			} else {
				if(str[i]=='B') mis2++;
				flag = 0;
			}
		}
		printf("%lld\n", (min(mis1,mis2))/2);
	}

	return 0;
}
