#include <iostream>
using namespace std;



int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int len,count=0,val=1;
    	cin>>len;
    if(len>1)
    	while(val<len)
    	{
    		val=val*2+1;
    		count++;
    	//	cout<<val<<endl;
    	}
    	if(val!=len) count--;
    	cout<<count<<endl;
    }
    return 0;
}
