#include<cstdio>
#define MAX 500
#define read(x) scanf("%d",&x)
#define write(x) printf("%d\n",x)

using namespace std;

int multiply(int x, int res[], int res_size)
{
    int i,carry = 0;
    for (i=0; i<res_size; i++)
    {
        int prod = res[i] * x + carry;
        res[i] = prod % 10;
        carry  = prod/10;
    }
    while (carry)
    {
        res[res_size] = carry%10;
        carry = carry/10;
        res_size++;
    }
    return res_size;
}

void power(int a,int b)
{
    int c,x,i,res[MAX];
    res[0] = 1;
    c=0;
    int res_size = 1;
    for (x=1; x<=b; x++)
        res_size = multiply(a, res, res_size);
    for (i=res_size-1; i>=0; i--)
        if(res[i]%2==1)
            c++;
    write(c);
}


int main()
{
    int t,a,b;
    read(t);
    while(t--)
    {
        read(a);
        read(b);
        power(a,b);
    }
    return 0;
}
