#include <bits/stdc++.h>
using namespace std;
#define lli long long int
int main()
{
    vector<string>  v;
    char s[100000];
    lli t;
    cin>>t;
    getchar();
    while(t--)
    {
    	cin.getline(s,100000);
    	//cout<<s<<endl;
    	char *token = strtok(s, " ");
		    while (token != NULL)
		    {
		    	//cout<<token<<endl;
		    	string inp(token);
		        v.push_back(inp);
		        token =strtok(NULL, " ");
		    }
		for(int i=v.size()-1;i>=0;i--)
		cout<<v[i]<<" ";
		cout<<endl;
		v.clear();
    }
		    
}
