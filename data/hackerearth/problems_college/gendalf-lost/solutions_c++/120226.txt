#include <bits/stdc++.h>
#include<string>
#define limit 1000001
using namespace std;
string L[limit];
 
inline string word_input(){
        char str[100];
        int i;
        string w;
        scanf("%s",str);
        int l=strlen(str);
        w=str;
        assert(l>=1 && l<=10);
        for(i=0;i<l;i++) {
                        assert((str[i]>='0' && str[i]<='9') || (str[i]>='A'  &&  str[i]<='Z') );
                }
        assert(w.size()==l);
        return w;
       
}
 
int main(void){
        string word,W;
        char str[11];
        int n,q,i,ans,j,l;
        int left,right,mid;
        scanf("%d",&n);
        assert(n>=1 && n<limit);
        for(i=1;i<=n;i++)
                L[i]=word_input();
        sort(L+1,L+n+1);
        scanf("%d",&q);
        for(i=0;i<q;i++){
                W=word_input();
                left=1;
                right=n;
                ans=-1;
                while(left<=right){
                        mid=(left+right)/2;
                        if(L[mid]==W){
                                ans=mid;
                                break;
                        }
                        else if(W<L[mid])
                                right=mid-1;
                        else
                                left=mid+1;
                }
                printf("%d\n",ans);
                //cout<<ans;
        }
        return 0;
}
