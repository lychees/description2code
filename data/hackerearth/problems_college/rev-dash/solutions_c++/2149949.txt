#include<bits/stdc++.h>
using namespace std;
int main()
{
        char s[1000];
        cin>>s;
        int i,l;
        l=strlen(s);
        for(i=0;i<l;i++)
        {
        	                cout<<s[i];if(i!=(l-1)){cout<<'-';}
        }
        cout<<'\n';
        for(i=l-1;i>=0;i--)
        {
                cout<<s[i];if(i!=0){cout<<'-';}
        }

        return 0;
}
