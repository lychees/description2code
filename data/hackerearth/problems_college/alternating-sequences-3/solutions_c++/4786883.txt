#include <iostream>
#include<vector>
#include<cstring>
#include<climits>
using namespace std;
bool sign(long long int v)
{
	if(v<0)
		return true;
	else
		return false;
}
int main()
{
    int n;
    cin>>n;
    long long int a[n];
    for(int i=0;i<n;i++)
    	cin>>a[i];
    int lis[n];
    //memset(lis,1,sizeof(lis));
    for(int i=0;i<n;i++)
    	lis[i]=1;
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<i;j++)
    	{
    		if(abs(a[j])<abs(a[i]) && sign(a[j])!=sign(a[i]) && lis[i]<lis[j]+1)
    		{
    			lis[i]=lis[j]+1;
    		}
    	}
    }
    int mx=INT_MIN;
    for(int i=0;i<n;i++)
    {
    	if(mx<lis[i])
    		mx=lis[i];
    }
    cout<<mx;
    return 0;
}
