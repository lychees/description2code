#include <bits/stdc++.h>
using namespace std;
#include <iomanip>
int main()
{
    int n;
    cin>>n;
    double a[n],b[n];
    for(int i=0;i<n;i++){
    	cin>>a[i];
    }
    for(int i=0;i<n;i++){
    	cin>>b[i];
    }
    double x;
    cin>>x;
    double ans=0;
    for(int i=0;i<n;i++){
    	double y=x/b[i];
    	y=y*a[i];
    	if(y>ans){
    		ans=y;
    	}
    }
    long long z=ans*1000;
    ans=z*1.0/1000;
    cout<<std::fixed<<std::setprecision(3)<<ans<<endl;
}
