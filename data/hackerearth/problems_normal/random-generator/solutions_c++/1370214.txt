#include<bits/stdc++.h>
using namespace std;

const int N=1e5+10;

int a[N];

int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int n,k,p;
        scanf("%d%d%d",&n,&k,&p);
        for(int i=0;i<n;i++) {
            scanf("%d",a+i);
        }
        sort(a,a+n);
        for(int i=0;i<n;i++) {
            if(i+k-1<n&&a[i+k-1]<=a[i]+2*p) {
                puts("NO");
                goto hell;
            }
        }
        puts("YES");
        hell:;
    }
    return 0;
}
