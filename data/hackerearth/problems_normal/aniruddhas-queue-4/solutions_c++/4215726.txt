#include <iostream>
using namespace std;
int main(){
	int t;cin>>t;
	for (int j=0;j<t;j++){	long long sum=0;
		int n;cin>>n;int a[n];long long x;
		for(int i=0;i<n;i++){cin>>a[i];sum+=a[i];
		}cin>>x;
		int d;int c=0;
		d=x%sum;
		if(d==0){int r=0;
		for(int l=0;l<n;l++)
		{r+=a[l];if(r==sum){ cout<<l+1<<endl;break;}
		}
		}
	else{
		for(int k=0;k<n;k++){c+=a[k];
		if(c>=d){cout<<k+1<<endl;break;
		}}
		}
	}
}
