#include<cstdio>
#include<algorithm>
#define mod 1000000007
long long arr[100005],b[100005];
int main()
{
	long long t,n,i,j;
	long long ans=0,temp,s=0,s1=0,k;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		s=0,s1=0,k=1;
		for(i=1;i<=n;i++)
		{
			scanf("%lld",&arr[i]);
			if(arr[i]>0)
			s=(s+arr[i])%mod;
			else
			s=(s-arr[i])%mod;
		}
	//	printf("s=%lld\n",s);
		std::sort(arr+1,arr+n+1);
		for(i=1;i<=n;i++)
		{
				if(k<=n)
				b[k]=-arr[i];
				else
				break;
				k=k+2;
		}
		k=2;
		for(j=i;j<=n;j++)
		{
			b[k]=arr[j];
			k=k+2;
		}
		for(i=1;i<=n;i++)
		{
	//	printf("%lld ",b[i]);
		    s1=(s1+b[i]+mod)%mod;	
		}
	//	printf("s1=%lld\n",s1);
		ans=(s1*s)%mod;
		printf("%lld\n",ans);
	}
	return 0;
}
