template<typename P> struct Cmp
{
    bool operator()(const P &p1, const P &p2)
    {
        if(p1.first < p2.first){
        	return true;
        }else if(p1.first == p2.first){
        	if(p1.second > p2.second){
        		return true;
        	}else{
        		return false;
        	}
        }
        else return false;
    }
};
