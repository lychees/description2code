#include <algorithm>
#include <cmath>
#include <cstdio>
#include <iomanip>
#include <iostream>
#include <map>
#include <queue>
#include <set>
#include <string>
#include <vector>

using namespace std;

int main() {

  string s, t;
  cin >> s >> t;
  int i1 = 0, i2 = t.size() - 1;

  for (int j = 0; i1 < t.size() && j < s.size(); ++i1)
    if (t[i1] == s[j])
      ++j;

  for (int j = s.size() - 1; i2 >= 0 && j >= 0; --i2)
    if (t[i2] == s[j])
      --j;

  if (i1 > i2)
    cout << 0 << endl;
  else
    cout << i2 - i1 + 2 << endl;

  return 0;
}
