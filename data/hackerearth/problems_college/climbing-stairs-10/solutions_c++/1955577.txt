#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#include<algorithm>
using namespace std;
int main()
{	
	long long t,a,b,i,n;
	long long int sum;
	cin>>t;
	while(t--)
	{
		a=0;
		b=1;
		sum=0;
		cin>>n;
		for(i=0;i<=n;i++)
		{
			a=b;
			b=sum;
			sum=a+b;	
		}
		cout<<sum<<endl;
	}
	return 0;
}
