#include <bits/stdc++.h>

using namespace std;


int main()
{
    int i, j, n, k, q, l, r;
    cin >> n;
    int A[n+1], B[n+1];
    for(i = 1; i <= n; i++) {
        cin >> A[i];
    }
    B[1] = A[1];
    for(i = 2; i <= n; i++) {
        B[i] = B[i-1] + A[i];
    }
    /*for(i = 1; i <= n; i++) {

        cout << B[i] << " ";
    }
    cout << endl;*/
    cin >> q;
    while(q--) {
        cin >> l >> r;
        l--;
        if(l < 1) {
        	cout << B[r] << endl;
        } else
        	cout << B[r] - B[l] << endl;
    }
    return 0;
}
