#include <iostream>
#include<string>
using namespace std;

class stack1{
	private:
	char a[100000];
	int top;
	public:
	void push(char d){if (top<=99999)a[top++]=d; else cout<<"error";}
	void pop(){if(top>0)top--;else cout<<"error";}
	int get_top(){return top;}
	char upper(){if(top>0)return a[top-1];else cout<<"stack empty"<<endl;}
	stack1(){top=0;}
	void empty(){top=0;}
	
};


int main(){

    int m;
    string s;
    int c=0;
    int bu=0;
    int ch=0;
    stack1 st;
    cin>>m;
    for(int i=1;i<=m;i++)
    {
    	
    	cin>>s;
        st.push(s[0]);
        
        
     for(unsigned int i=1;i<s.length();i++)
     {  
       	if(st.get_top()>0&&st.upper()==s[i])
     	    st.pop();
     	   else 
     	    st.push(s[i]);
     }
     if (st.get_top()==0)
     
     bu++; else st.empty();
     	
     	
     
    }
    
    cout<<bu;
    
    
    
    return 0;
}
