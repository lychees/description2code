#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s;
        cin>>s;
        for(int i=0;i<s.length();i++)
        {
            if(s[i]=='B')
            {
                if(s[i-1]=='-')
                    s[i-1]='0';
                if(s[i-2]=='-')
                    s[i-2]='0';
            }
            if(s[i]=='S')
            {
                if(s[i-1]=='-')
                    s[i-1]='0';
                if(s[i+1]=='-')
                    s[i+1]='0';
            }
        }
        int c=0;
        for(int i=0;i<s.length();i++)
            if(s[i]=='-')
                c++;
        cout<<c<<endl;
    }
    return 0;
}
