  #include <iostream>
  #define ll long long
    using namespace std;

    int main()
    {
        //taking input for number of test cases, T.
        int t;
        cin>>t;
		while(t--)
		{
        	ll int n,k,m;
        	cin>>n>>k>>m;

   
        	ll int ans=n*n;
        	ans*=(n+1)*(n+1);
        	ans/=4;
        	ans*=k;
        	ans-=m;
        	if(ans<0)
        	cout<<"0\n";
        	else
        	cout<<ans<<endl;
		}
        return 0;
    }
