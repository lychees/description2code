#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long int sum = 0;
	int k;
	long long int m;
	
	int n1,x;
	cin>>k>>m;
	m = m*(m+1)/2;
	map<int,int>mp;
    for(int i = 0; i < m; i++){
    	cin>>n1;
    	for(int j = 0; j < n1; j++){
    		cin>>x;
    		if(mp[x] < 1){
    			sum = sum + x;
    			mp[x]++;
    		}
    		
    	}
    }
    if(sum == m){
    	cout<<"YES"<<endl;
    }
    else{
    	cout<<"NO"<<endl;
    }
    return 0;
}
