#include <iostream>
#include <math.h>
using namespace std;
typedef long long int ll;

ll min1(ll a, ll b)
{
	if(a < b){
		return a;
	}
	else
		return b;
}
ll max1(ll a, ll b)
{
	if(a > b){
		return a;
	}
	else
		return b;
}	

int main() {
	ll t;
	cin >> t;
	ll a, b, a1, b1, x, y;
	ll c, i;
	while(t--){
		c = 0;
		cin >> a1 >> b1;
		a = min1(a1, b1);
		b = max1(a1, b1);
		x = (ll) sqrt(a);
		y = (ll) sqrt(b);
		c = y - x;
		if(x * x == a){
			c++;
		}
		if(y == 1000000000 &&(y * y != b)){
			c--;
		}
		cout << c << endl;
	}
	return 0;
}
