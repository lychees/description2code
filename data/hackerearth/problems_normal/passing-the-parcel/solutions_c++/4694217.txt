#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    string a;
    cin>>n>>a;
    list <int> l;
    list <int>::iterator it;int j=0;
    for(int i=1;i<n+1;i++)l.push_back(i);
    it=l.begin();
    while(l.size()!=1){
    	if(a[j++]=='b')
    	it=l.erase(it);
    	else{
    		it++;
    	}
    	if(it==l.end())
    	it=l.begin();
    	if(j==a.size())
    	j=0;
    	
    }
    cout<<*it;
    return 0;
}
