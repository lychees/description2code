#include <iostream>
using namespace std;

int main()
{
    int t,i,k;
    string s;
    cin>>t;
    while(t--){
    	cin>>s;
    	int c[26] = {0};
    	k = 0;
    	for(i=0;i<s.size();i++){
    		c[s[i]-'a']++;
    	}
    	
    	for(i=0;i<26;i++){
    		if(c[i]%2==1){
    			k++;
    		}
    	}
	   	if(k>1){
	   		cout<<"No"<<endl;
	   	}else{
	   		cout<<"Yes"<<endl;
	   	}
    }
    return 0;
}
