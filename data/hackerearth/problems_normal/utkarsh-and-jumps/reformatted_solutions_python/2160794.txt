# print 'Hello World!'

N, P = map(int, raw_input().split())
p2 = float(P)/100
p3 = 1-(float(P)/100)
dp = []
dp.append(1)
dp.append(0)
dp.append(p2)
dp.append(p3)

# print dp

for i in range(4, N+1):
    dp.append(p2 * dp[i-2] + p3 * dp[i-3])

print "%.6f" % dp[N]
