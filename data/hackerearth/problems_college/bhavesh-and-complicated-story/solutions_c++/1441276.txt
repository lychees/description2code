using namespace std;
#define MAX_N 1000002
long long int bit[MAX_N];
void add(int i, long long int x){
	i++;
	while (i < MAX_N){
		bit[i] += x;
		i += i&-i;
	}
}
long long int sum(long long int i){
	i++;
	long long int r = 0;
	while (i){
		r += bit[i];
		i -= i&-i;
	}
	return r;
}
int main(){
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; i++){
		int a;
		scanf("%d", &a);
		add(a, 1);
	}
	int q;
	scanf("%d", &q);
	while (q--){
		int ty;
		scanf("%d", &ty);
		if (ty == 0){
			int a, b;
			scanf("%d%d", &a, &b);
			long long int ans = sum(b) - sum(a - 1);
			printf("%lld\n", ans);
		}
		else{
			if (ty == 1){
				int a;
				scanf("%d", &a);
				add(a, -1);
			}
			else{
				int a;
				scanf("%d", &a);
				add(a, 1);
			}
		}
	}
	return 0;
}
