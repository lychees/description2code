#include<bits/stdc++.h>
using namespace std;
int main()
{
  int ts;
  cin>>ts;
  while(ts--)
  {
    int n,a;
    cin>>n;
    vector<int> pos(n+1,0);
    for(int i=1;i<=n;i++)
    {
      cin>>a;
      pos[i]=a;
    }
    sort(pos.begin()+1,pos.end());
    int i=1,j=n;
    while(i<j)
    {
       if(pos[i]!=1)
           i++;
        if(pos[j]!=-1)
            j--;
    }
   // i--;
    //j++;
    a=min(i,j);
    cout<<a<<endl;
  }
}
