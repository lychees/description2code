#include <iostream>
using namespace std;

int main()
{
    int even[1001],odd[1001];
    int e,o,i,j,k,test,n;
    int ecnt,ocnt;
    scanf("%d",&test);
    while(test--)
    {
    	e=0;o=0;
    	ecnt=ocnt=0;
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&j);
    		if(j%2==0)
    			even[e++]=j;
    		else
    			odd[o++]=j;
    	}
    	for(i=0;i<e;i++)
    	{
    		k=even[i];
    		for(j=i+1;j<e;j++)
    		{
    			if(k!=even[j])
    				ecnt++;
    		}
    	}
    	for(i=0;i<o;i++)
    	{
    		k=odd[i];
    		for(j=i+1;j<o;j++)
    		{
    			if(k!=odd[j])
    				ocnt++;
    		}
    	}
    	printf("%d\n",ecnt+ocnt);
    }
    return 0;
}
