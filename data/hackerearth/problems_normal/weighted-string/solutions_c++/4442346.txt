#include <bits/stdc++.h>
using namespace std;
int main()
{
	long long t,k,len,i,j,sum,c;
	string s;
    cin>>t;
    while(t--){
    	cin>>k;
    	cin>>s;
    	len=s.size();
    	c=0;
    	for(i=0;i<len;i++){sum=0;
    		for(j=i;j<len;j++){
    			sum=sum+(s[j]-'a'+1);
    			if(sum==k)c++;
    			if(sum>k)break;
    		}
    	}
    	cout<<c<<endl;
    }
    return 0;
}
