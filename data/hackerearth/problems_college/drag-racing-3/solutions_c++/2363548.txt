#include <bits/stdc++.h>
using namespace std;

int main()
{
    map <string,int> racers;
    map <string,int>::iterator it;
    int n,races;
    cin>>n;
    races = (int)pow(2,n) - 1;
    while(races--){
    	string w,l;
    	cin>>w>>l;
    	if(racers.find(w) == racers.end()){
    		racers[w] = 1;
    	}else{
    		int temp = racers[w];
    		racers[w] = temp+1;
    	}
    }

	int max = -1;
	string winner="";
    for(it = racers.begin();it!=racers.end();it++){
    	if(it->second>max){
    		winner = it->first;
    		max = it->second;
    	}
    }
    
    cout<<winner<<endl;
    
    return 0;
}
