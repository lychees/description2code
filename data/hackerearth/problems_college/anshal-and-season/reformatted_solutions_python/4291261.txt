A = raw_input()
b = raw_input()

A = [x for x in A.split(' ')]
a = A[0]

if len(a) != len(b):
    print "WRONG"
else:
    aa = [0]*26
    bb = [0]*26

    for i in range(0, len(a)):
        aa[ord(a[i])-65] = aa[ord(a[i])-65]+1

    for i in range(0, len(b)):
        bb[ord(b[i])-65] = bb[ord(b[i])-65]+1

    f = 0
    for i in range(0, 26):
        if aa[i] != bb[i]:
            f = 1

    if f == 0:
        print "CORRECT"
    else:
        print "WRONG"
