#include<iostream>
#include<cstring>
using namespace std;

int main()
{
	int test;
	string site;
	int count;
	cin>>test;
	for(int i=0; i<test; i++)
	{
		count=0;
		cin>>site;
		for(int j=4; j<site.length()-4; j++)
		{
			if( (site[j]=='a') || (site[j]=='e') || (site[j]=='i') || (site[j]=='o') || (site[j]=='u') || (site[j]=='A') || (site[j]=='E') || (site[j]=='I') || (site[j]=='O') || (site[j]=='U'))
				count++;	
		}
		cout<<site.length()-count-4<<"/"<<site.length()<<endl;
		//cout<<count;
	}
	return 0;
}
