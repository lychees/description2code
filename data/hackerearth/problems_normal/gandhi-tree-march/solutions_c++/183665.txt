#include <iostream>
#include <stdio.h>
#include <string.h>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
    int test;
    scanf("%d",&test); 
    for(int i=0;i<test;i++)
      {
         char A[100001];
         pair < char , pair <int , int> > P , P1;
         vector < pair <char, pair <int , int > > > V, V1;
         vector < pair <char, pair <int , int > > > :: iterator it;
           
         int n;
         
         scanf("%d %s",&n,&A);
         int len=strlen(A);
         int top=0;
         
         for(int j=0;j<len;j++)
           {
              if(A[j]>='a'&&A[j]<='z')
                {
                   P1.first = A[j];
                   P1.second.first=0;
                   P1.second.second = 0;
                                      
                   if(!V.empty())
                     {
                        P = V.back();         
                        V.pop_back();
                        if(P.second.second == 0)
                          {
                             P1.second.first = P.second.first - 1;
                             P.second.second = 1;
                             V.push_back(P);
                             V.push_back(P1);     
                          }
                        else
                          {
                             P1.second.first = P.second.first + 1;
                             V1.push_back(P);
                             V.push_back(P1); 
                          }
                      }
                    else
                      {
                         V.push_back(P1);
                      }
                } 
                
               if(A[j]=='.')
                 {
                     if(!V.empty())
                       {
                          P = V.back();         
                          V.pop_back();
                          if(P.second.second == 0)
                          {
                             P.second.second = 1;
                             V.push_back(P);  
                          }
                         else
                          {
                             V1.push_back(P);
                          }
                       }
                 }
             }
             
         sort(V1.begin(),V1.end());    
        
         int flag=0;
         
         for(it = V1.begin(); it!=V1.end();it++)
           {
              if(it->second.first==n)
                {
                  printf("%c",it->first);
                  flag=1;
                }
           }
           
         if(flag==0)
           printf("Common Gandhijee!\n");
           
         else
           printf("\n");  
              
    }
             
return 0;
}
