#include<iostream>
#include<queue>
using namespace std;

int main(){
	int n;
	cin >> n;
	queue<int> s;
	for(int i = 1; i <= n; i++)
		s.push(i);
	queue<char> l;
	string str;
	cin >> str;
	for(int i = 0; i < str.length(); i++)
		l.push(str[i]);
		
	int num;
	char c;
	while(s.size() != 1){
		if(l.front() == 'b'){
			s.pop();
			c = l.front();
			l.push(c);
			l.pop();
		}
		else{
			num = s.front();
			s.push(num);
			s.pop();
			c = l.front();
			l.push(c);
			l.pop();
		}
	}
	cout << s.front();
}
