#include <bits/stdc++.h>

using namespace std;

int a[20],x[20];

int main (){

	int t;
	scanf("%d",&t);

	while(t--){
		int n;
		scanf("%d",&n);
		for(int i = 0;i<n;i++)
			scanf("%d",&a[i]);
		for(int i = 0;i<n;i++)
			scanf("%d",&x[i]);
		int ae = 0;
		for(int i = 0;i<n;i++){
			ae^=(x[i]-1);
			ae^=(a[i]-x[i]);
		}

		if(ae)
			printf("First\n");
		else
			printf("Second\n");

	}

	return 0;
}
