#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long n,i,x=0,y=0,p1=0,p2=0;
        cin>>n;

        for(i=0;i<=n;i++)
        {
            x=i;
            y=n-i;
            if(x%3==0 && y%5==0){p1=x;p2=y;}
        }
        if(p1==0 && p2==0){cout<<"-1\n";continue;}
        else
        {
            while(p1--)cout<<5;
            while(p2--)cout<<3;
        }
        cout<<endl;
    }
    return 0;
}
