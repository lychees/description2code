#include<iostream>
#include<cstdio>
using namespace std;
int pr[1000001]={0};
int hsh[1000001]={0};
int main()
{
     pr[0] = 1;
     pr[1] = 1;
      for(int i = 2 ; i <= 1000 ; ++i )
      {
          if(pr[i] == 0 )
            for(int j = 2*i ; j <= 1000000 ; j=j+i )
             pr[j] = 1;
      }
      for(int i = 1 ; i <= 1000000 ; ++i)
      {
            if(pr[i] == 0)
                hsh[i] = 1 + hsh[i-1];
            else
                 hsh[i]= hsh[i-1];
      }
        int t;
        cin >> t;
         while(t--)
         {
             int l , r;
             cin >> l >> r;
             float f = (hsh[r]-hsh[l-1])/((float)(r-l+1));
             printf("%6f\n",f);
         }
    return 0;
 
}
