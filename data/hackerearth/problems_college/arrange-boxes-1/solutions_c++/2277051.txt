#include<bits/stdc++.h>
using namespace std;
long long gcd(long long a,long long b) {
	if(b == 0)
	return a;
	else
	return gcd(b,a%b);
}
int main() {
	long long int n; 
	cin >> n;
	while(n--) {
		long long int j,k,i,m,x;
		cin >> m;
		long long int a[m],b[m];
		for(i=0;i<m;i++) {
			cin >> x;
			a[i] = x;
		} 
		for(i=0;i<m;i++)
		cin >> b[i];
		long long int cc = 0;
		for(i=0;i<m;i++) {
			if(a[i] == b[i])
			continue;
			for(j=i+1;j<m;j++) {
				if(a[j] == b[i])
				break;
			}
//			cout << j <<"\n";
			cc += (j-i);
			for(k=j;k>i;k--) {
				a[k] = a[k-1];
			}
			a[i] = b[i];
		}
		cout << cc <<"\n"; 
	}
	return 0;
}
