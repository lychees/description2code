#include<iostream>
using namespace std;
main()
{
    int t;
    cin>>t;
    while(t--)
    {
        unsigned int n,a,b;
        int counter=0;
        cin>>n;
        for(int i=1;i<n;i++)
        {
            for(int j=1;j<=n;j++)
            {
                if(i>=j)
                {
                    continue;
                }
                int x,y;
                x=i+j;
                y=i*j;
                if(y%x==0)
                {
                    counter++;
                }
            }
        }
        cout<<counter<<endl;
    }
}
