#include <iostream>
using namespace std;
 
int main()
{
    int k;
    cin>>k;
    int a[100005];
    a[0]=0;
    int i;
    for( i=1;i<=k;i++){
    	cin>>a[i];
    }
    for(i=1;i<=k;i++)
    {
    	a[i]=a[i]+a[i-1];
    }
    int n;
    cin>>n;
    while(n--)
    {
    	int b,c;
    	cin>>b>>c;
    	cout<<a[c]-a[b-1]<<endl;
    }
    return 0;
}
