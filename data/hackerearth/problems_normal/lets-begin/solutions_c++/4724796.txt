#include <bits/stdc++.h>
#define ll long int
using namespace std;

int main()
{
    int t;
    ll n,r,cnt;
    cin>>t;
    while(t--)
    {
    	cnt=0;
    	cin>>n;
    	if(n==1)
    	{
    		cout<<"-1"<<endl;
    		continue;
    	}
    	r=n%7;
    	cnt+=(n/7);
    	switch(r)
    	{
    		case 1:	cnt+=1;
    				break;
    		case 2: cnt+=1;
    				break;
    		case 3: cnt+=1;
    				break;
    		case 4: cnt+=2;
    				break;
    		case 5: cnt+=1;
    				break;
    		case 6: cnt+=2;
    				break;
    		default: break;
    			
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
