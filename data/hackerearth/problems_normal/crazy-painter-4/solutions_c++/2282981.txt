#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#define LL long long
LL c[30];
int main()
{
	LL t,i,n,sum,pts,full,rem,count;
	scanf("%lld",&t);
	while(t--)
   {
   	sum=0;
   	count=0;
   	scanf("%lld",&n);
   	for(i=0;i<=25;i++)
   	{
   	scanf("%lld",&c[i]);
   	sum+=c[i];
      }
   	pts=12*n;
   	full=pts/26;
   	rem=pts%26;
   	count=count+ full*(sum+(full-1)*13);
   	for(i=0;i<rem;i++)
   	{
   		count+=(c[i]+full);
   	}
   	printf("%lld\n",count);
   }
}
