#include <iostream>
using namespace std;

int main()
{
   	int t;
   	cin>>t;
   	
   	for(int i=0;i<t;i++)
   	{
   		int count=0;
   		int n;
   		cin>>n;
   		int temp;
   		for(int j=0;j<n;j++)
   		{
   			cin>>temp;
   			if(temp%2==0)
   				count++;
   		}
   		cout<<count<<endl;
   	}
    return 0;
}
