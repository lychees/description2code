#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define d1(a)cout<<#a<<": "<<a<<"\n";
#define d2(a,b)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<"\n";
#define d3(a,b,c)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<" , "<<#c<<": "<<c<<"\n";
#define d4(a,b,c,d)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<" , "<<#c<<": "<<c<<" , "<<#d<<": "<<d<<"\n";

int main() {
	int t;
	cin>>t;
	while(t--){
		ll n,k;
		cin>>n>>k;
		ll ans = 0, x,curr = 0;
		for(int i=0;i<n;i++){
			cin>>x;
			if(x<=k){
				curr += x;
				ans = max(ans, curr);
			}
			else{
				ans = max(ans, curr);
				curr = 0;
			}
		}
		ans = max(ans, curr);
		cout<<ans<<"\n";
	}
	return 0;
}
