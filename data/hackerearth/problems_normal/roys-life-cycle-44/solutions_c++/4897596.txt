#include <iostream>
#include <string>

int main()
{
int days;
std::cin >> days;

int mStreak(0);
int dStreak(0);
int mContinuous(0);
int continuous(0);
while(days--)
{
std::string log;
std::cin >> log;
for (char check: log)
if (check == 'C')
{
++dStreak;
++continuous;
}
else
{
	mStreak = (dStreak>mStreak?dStreak:mStreak);
	dStreak = 0 ;
	mContinuous = (continuous>mContinuous?continuous:mContinuous);
	continuous = 0;
}
mStreak = (dStreak>mStreak?dStreak:mStreak);
dStreak = 0 ;
}
mContinuous = (continuous>mContinuous?continuous:mContinuous);
	continuous = 0;

std::cout << mStreak << ' ' << mContinuous;
return 0;
}
