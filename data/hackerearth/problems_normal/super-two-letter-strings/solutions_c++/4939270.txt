#include<bits/stdc++.h>
using namespace std;
#define ull long long int
#define mod 1000000007
ull dp[10004][11][2];
ull n,p;
ull rec(ull l,ull k,ull bit)
{
    if(k==0 && bit==0)
        return 0;
    if(k==0 && bit==1 && l==n)
        return 1;
    if(l==n)
        return 1;
    if(dp[l][k][bit]!=-1)
        return dp[l][k][bit];
    dp[l][k][bit]=((rec(l+1,p,1)%mod)+(rec(l+1,k-1,0)%mod))%mod;;
    return dp[l][k][bit];
}
int main()
{
    ios::sync_with_stdio(false);
    ull t;
    cin>>t;
    while(t--)
    {
        cin>>n>>p;
        memset(dp,-1,sizeof(dp));
        cout<<rec(1,p,1)<<"\n";
    }
    return 0;
}
