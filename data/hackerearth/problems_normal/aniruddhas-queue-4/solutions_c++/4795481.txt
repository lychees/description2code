#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
	int t, n, i;
	ll x, sum;
	cin >> t;
	while(t--) {
		sum = 0;
		cin >> n;
		vector<ll> v(n);
		for(i = 0; i < n; i++) {
			cin >> v[i];
			sum += v[i];
		}
		cin >> x;
		if(x == 0) {
			cout << "0\n"; continue;
		}
		
		x %= sum;
		if(x == 0) x = sum;
		int ans = 0;
		ll rsum = 0;
		//i = 0;
		while(rsum < x) {
			rsum += v[ans++];
		}
		cout << ans << "\n";
	}
}
