#include <bits/stdc++.h>
using namespace std;

int main(){
 	int t;
    scanf("%d",&t);
    while(t--){
    	long long n;
    	long long ar[100004];
    	scanf("%lld",&n);
    	for(int i = 0; i < n; i++){
    		scanf("%lld",&ar[i]);
    	}
    	sort(ar,ar+n);
    	long long y = ar[n/2], c = 0;
    	for(int i = 0; i < n; i++){
    		c += abs(ar[i]-y);
    	}
    	printf("%lld\n",c);
    }
    return 0;
}
