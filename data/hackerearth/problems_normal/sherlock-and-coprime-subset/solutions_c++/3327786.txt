int rec(int pos,int mask) {
  if(pos==n) {
    return 0;
  }
  if(dp[pos][mask]!=-1)
    return dp[pos][mask];
//  cout<<mask<<" ";
  int ans = 0;
  if(!(mask&(val[pos]))) {
    ans = max(ans,max(1+rec(pos+1,mask|val[pos]),rec(pos+1,mask)));
  }
  else {
    ans = max(ans,rec(pos+1,mask));
  }
  return dp[pos][mask]=ans;
}
