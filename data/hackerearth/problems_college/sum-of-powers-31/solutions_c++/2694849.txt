 #include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t;
    int f;
    int a[1000000] = {0};
    a[1] = 1;
    for(int i = 2; i <= 1000;i++){
        for(int j = 2;pow(i,j) <= 1000000;j++){
            int pos = pow(i,j);
            a[pos] = 1;
        }
    }
    cin>>t;
    while(t--){
        cin>>f;
        bool chk = false;
        for(int i = 1;i <= f/2 ; i++){
            if(a[i] && a[f-i]) {
                chk = true;
                break;
            }
        }
        if (chk) cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
        }
return 0;       
}
