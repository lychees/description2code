t = int(raw_input())
for i in range(t):
    ab = raw_input().split()
    a = int(ab[0])
    b = int(ab[1])
    count = 0
    while a != 1 and b != 1:
        if a > b:
            count += a/b
            a = a % b
            if a == 0:
                break
        else:
            count += b/a
            b = b % a
            if b == 0:
                break

    print count+abs(a-b)
