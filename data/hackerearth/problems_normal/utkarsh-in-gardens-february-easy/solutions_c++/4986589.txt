int main()
{
	int n;
	cin>>n;
	bitset<2002> a[2002];
	for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++)
		{
			int t;
			cin>>t;
			a[i][j]=t;
		}

	{
		for(int j=i+1;j<=n;j++)

			ans+=(cnt*(cnt-1))/2;    //select any two pairs
		}
	}
	cout<<ans/2;
