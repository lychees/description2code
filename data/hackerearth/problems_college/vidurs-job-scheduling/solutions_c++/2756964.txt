#include <bits/stdc++.h>
using namespace std;
int b[2005][2005];

int a[2005];
 int n;
int check(int start,int end){
    if(b[start][end]!=-1)return b[start][end];
    if(start==end)return n*a[start];
    int t1=check(start+1,end);
    int t2=check(start,end-1);
 int t=n-(end-start+1)+1;
 int ans=max(t1+t*a[start],t2+t*a[end]);
 b[start][end]=ans;
 return ans;
 
    
}
int main()
{ 
    
   for(int i=0;i<2005;i++)
   for(int j=0;j<2005;j++)b[i][j]=-1;
    cin>>n;
    
    
    for(int i=0;i<n;i++)cin>>a[i];
   cout<< check(0,n-1);
    
    
    return 0;
}
