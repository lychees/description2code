#include<bits/stdc++.h>
using namespace std;
int arr[1000001];
int main()
{
int t,n,f;

arr[1]=1;

for(long long int i=2;i<1000001;i++)

 {
    if(arr[i]!=0)
        continue;
    for(long long int j=i*i;j<1000001;j*=i)
        arr[j]=1;
 }
scanf("%d",&t);
 while(t--)
   {
    scanf("%d",&n);
    f=1;
    for(int i=1;i<n;i++)
     {
        if(arr[n-i] && arr[i])
         {
            f=0;
         printf("Yes\n");
         break;
         }
     }
        if(f)
        printf("No\n");
   }
}
