#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <vector>
#include <string>
#include <map>
#include <algorithm>
#include <set>
using namespace std;


const int max_n = 100000;

const int max_k = 10;
long long f[max_n * 2+ 1][2][2];

const int mod = 1000000007;
int main()
{

    int test;
    cin >> test;


    f[0][0][0] = 1;
    f[0][1][0] = 1;
    for (int i = 1; i < 100000; i++)
    {
        f[i][0][0] = f[i - 1][1][0];
        f[i][0][1] = (f[i - 1][1][1] + f[i - 1][0][0]) % mod;
        f[i][1][0] = (f[i - 1][0][0] + f[i - 1][1][0]) % mod;
        f[i][1][1] = (f[i - 1][0][1] + f[i - 1][1][1]) % mod;
    }

    for (int i = 0; i < test; i++)
    {
        int k;
        cin >> k;
        cout << (f[k - 1][1][1] + f[k - 1][0][1]) % mod << endl;
    }

    return 0;
}
