#include <iostream>
#include <sstream>
#include <iomanip>
#include <cmath>
using namespace std;


static double angle(int h, int m) {
    double hAngle = 0.5d * (h * 60 + m);
    double mAngle = 6 * m;
    double angle = abs(hAngle - mAngle);
    angle = angle < 360 - angle? angle: 360 - angle;;
    return angle;
}

int main()
{
    int H, M;
    char c;
    string s;
    cin >> s;
    
    istringstream iss(s);
	iss >> H >> c >> M;
	cout << fixed << setprecision(6)<<angle(H,M)<<endl;
    return 0;
}
