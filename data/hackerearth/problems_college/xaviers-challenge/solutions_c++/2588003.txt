#include <iostream>
#include <cstring>
#include <string>
#include <cstdio>
#include <algorithm>
using namespace std;


int main()
{
    char ch;
    int t;
    scanf("%d",&t);
    scanf("%c",&ch);
    while(t--)
    {
    	string s;
    	getline(cin,s);
    	int len = s.size();
    	int sum = 0;
    	for(int i=0;i<len;i++)
    	{
    		sum += s[i];
    	}
    	int f = sum/len;
    	if(f%2)
    	{
    		reverse(s.begin(),s.end());
    		printf("%s",s.c_str());
    	}
    	else
    	{
    		printf("%s",s.c_str());
    	}

    }
    return 0;
}
