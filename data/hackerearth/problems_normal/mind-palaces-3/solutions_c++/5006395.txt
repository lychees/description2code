#include <iostream>
using namespace std;

int main()
{
    int n,m;
    cin>>n>>m;
    int arr[n][m];
    for(int i=0; i<n;i++){
    	for(int j=0;j<m;j++){
    		cin>>arr[i][j];
    	}
    }
    int t;
    cin>>t;
    while(t--){
    	int j=m-1, i=0;
    	int x=-1,y=-1;
    	int Y;
    	cin>>Y;
    	while(i<n && j>=0){
    		if(arr[i][j] == Y) {x=j; y=i; break;}
    		if(arr[i][j] > Y) j--;
    		if(arr[i][j] < Y) i++;
    	}
    	cout<<y<<" "<<x<<endl;
    }
    return 0;
}
