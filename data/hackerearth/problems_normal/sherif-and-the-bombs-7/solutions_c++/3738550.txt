#include<bits/stdc++.h>
using namespace std;
int C[1001][1001];
int main()
{
    int t; cin>>t; while(t--)
    {
        long long  x,y,r; 
        cin>>x>>y>>r;
        r*=r;
        for(int i=1; i<=1000;i++)
            for(int j=1; j<1001;j++)
                if(  (x-i)*(x-i) + (y-j)*(y-j) <= r   )
                    C[i][j]++;
    }
    int ans=0;
    for(int i=0; i<=1000;i++)
    for(int j=0; j<=1000;j++)
        if(C[i][j]>1)
            ans++;
    cout<<ans<<endl;
    return 0;
}
