#include<bits/stdc++.h>
using namespace std;
int main()
{
       int t;
       cin>>t;

       while(t--)
       {
            int n;
            cin>>n;
             map<long long ,int>mp;
            long long a[n];
            long long b[n];
            for(int i=0;i<n;i++)
            {
                 cin>>a[i];
                 b[i]=a[i];
            }
            sort(a,a+n);
            int k=n;
            for(int i=0;i<n;i++)
            {
                    mp[a[i]]=k;
                      k--;
                }


          for(int i=0;i<n;i++)
          {
                 cout<<mp[b[i]]<<" ";
          }
          printf("\n");
       }
       return 0;

}
