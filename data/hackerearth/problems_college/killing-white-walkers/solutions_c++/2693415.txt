#include <iostream>
using namespace std;
long long int fun(long long int n)
{
	if(n<=1)
	  return n;
	if(n%2==0)
	  return 2*fun(n/2)-1;
	return 2*fun(n/2)+1;
	  
}
int main()
{
    long long int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cout<<fun(n)<<endl;
    	//cout
    }
    return 0;
}
