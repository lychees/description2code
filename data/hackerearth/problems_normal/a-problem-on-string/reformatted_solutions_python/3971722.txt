def main():
    n, s = raw_input(), raw_input()
    q = int(raw_input())

    count = {}
    for c in s:
        if c in count:
            count[c] += 1
        else:
            count[c] = 1

    for i in xrange(q):
        x, y = raw_input().split()
        nx, ny = 0, 0
        if x in count:
            nx = count[x]
        if y in count:
            ny = count[y]
        if x == y:
            print(nx * (nx-1)) // 2
        else:
            print nx * ny


main()
