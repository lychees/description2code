    #include <iostream>
    #include <cstring>
    using namespace std;
     
    int main()
    {
        int t;
        cin>>t;
        while(t--)
        {
        	string s;
        	cin>>s;
        	for(int i=0;i<s.size();i++)
        	{
        		if(i==0)
        		{
        			if((s[i]-'a')<=13)
        		cout<<s[i]-'a'<<" ";
        		else
        		cout<<(s[i]-'a')-26<<" ";
        		}
        		else if(-13<(s[i]-s[i-1])&&(s[i]-s[i-1])<=13)
        		cout<<(s[i]-s[i-1])<<" ";
        		else if((s[i]-s[i-1])==-13)
        		cout<<"13"<<" ";
        		else if((s[i]-s[i-1])>13)
        		cout<<(s[i]-s[i-1])-26<<" ";
        		else
        		cout<<26-abs(s[i]-s[i-1])<<" ";
        	}
        	cout<<endl;
        }
        return 0;
    }
