#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d", &t);

    while (t--) {
        int n, m;
        scanf("%d %d", &n, &m);

        vector<int> v(n);
        for (int i=0; i<n; i++) {
			double x, y, z, minn = 1000000000;
			int idx = 0;
            for (int j=0; j<m; j++) {
				double cost = 100;
                cin >> x >> y >> z;
				cost = cost - (cost * (x/100));
				cost = cost - (cost * (y/100));
				cost = cost - (cost * (z/100));
				if (cost < minn) {
					minn = cost;
					idx = j;
				}
            }
            v[i] = idx + 1;
        }

        for (int i=0; i<n; i++) {
			printf("%d ", v[i]);
        }
        printf("\n");
    }

    return 0;
}
