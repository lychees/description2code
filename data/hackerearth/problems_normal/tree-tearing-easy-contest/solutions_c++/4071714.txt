#include<bits/stdc++.h>
using namespace std;
vector<vector<int> > graph;
bool visit[100005]={false};
int ans=0,k;
int dfs(int i)
{
    visit[i]=true;
    int c=0;
    for(int j=0;j<graph[i].size();j++)
    {
        if(!visit[graph[i][j]])
            c+=dfs(graph[i][j]);
    }
    c++;
    if(c>=k)
    {
        ans++;
        c=0;
    }
    return c;
}
int main()
{
    ios::sync_with_stdio(false);
    int x,n,i;
    cin>>k>>n;
    graph.resize(n+1);
    for(i=2;i<=n;i++)
    {
        cin>>x;
        graph[i].push_back(x);
        graph[x].push_back(i);
    }
    dfs(1);
    cout<<ans;
    return 0;
}
