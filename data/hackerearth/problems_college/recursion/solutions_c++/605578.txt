int main()
{
    vector<long long int> v;
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
        int s=1;
        scanf("%d",&n);
        for(int i=0;i<n;i++)
        {
            v.push_back(i+1);
            s=s*(i+1);
        }
        for(int i=1;i<=s;i++)
        {
            for(int i=0;i<n;i++) cout<<v[i];
            next_permutation(v.begin(),v.end());
            cout<<" ";
        }
        cout<<endl;
        v.clear();
    }
    return 0;
}
