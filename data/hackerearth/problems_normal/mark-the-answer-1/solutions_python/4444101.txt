
if __name__ == '__main__':
    N, X = map(int, raw_input().split())
    A = map(int, raw_input().split())
    i = 0
    once = 0
    if A == [101]:
        print 0
    else:
        while i < N:
            if A[i] > X and once:
                i -= 1
                break
            elif A[i] > X and not once:
                once = 1
            i += 1
        print i
