def tc(s):
	n = len(s)
	for i in range(1, n - 2):
		for j in range(i + 1, n - 1):
			s1, s2 = s[:i], s[i:j]
			if s1 == s2:
				continue
			for k in range(j + 1, n):
				s3, s4 = s[j:k], s[k:]
				if s1 != s3 and s2 != s3 and s1 != s4 and s2 != s4 and s3 != s4:
					return 'YES'
	return 'NO'
for t in xrange(int(raw_input())):
	print tc(raw_input())
