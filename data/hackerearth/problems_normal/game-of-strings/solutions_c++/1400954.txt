#include<iostream>
using namespace std;
#include<cstring>
int arr[5001][5001];
int main()
{
    char str1[5001],str2[5001];
    cin>>str1>>str2;
    int len1=strlen(str1),len2=strlen(str2);

for(int i=1;i<=len1;i++)
{
    for(int j=1;j<=len2;j++)
    {
        char r=str1[i-1];
        if(r==str2[j-1]&&r!='a'&&r!='e'&&r!='i'&&r!='o'&&r!='u')
            arr[i][j]=arr[i-1][j-1]+1;
            
        else
        arr[i][j]=max(arr[i][j-1],arr[i-1][j]);
           
    }
 
}
    cout<<arr[len1][len2]<<endl;
    return 0;
}
