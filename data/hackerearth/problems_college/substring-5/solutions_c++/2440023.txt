#include <iostream>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int l=s.length();
    s[l]='$';
    int ans=0;
    int k=1;
    for(int i=0;i<l;i++)
    {
    	if(s[i]==s[i+1]) k++;
    	else 
    	{
    		ans+=(k*(k+1))/2;
    		k=1;
    	}
    }
    cout<<ans;
    return 0;
}
