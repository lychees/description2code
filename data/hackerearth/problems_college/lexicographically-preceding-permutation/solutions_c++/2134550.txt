#include <iostream>
#include<algorithm>
#include<cstring>

using namespace std;
int arr[1000009];

bool cmp(int a,int b)
{
    return b<a;
}
int main()
{
    int t,c;
   cin>>t;
   while(t--)
   {
       int n;
    memset(arr,0,sizeof arr);
   	cin>>n;
   	for(int i=0;i<n;i++)
   	cin>>arr[i];
   	int i;
   	for(i=n-1;i>0;i--)
   	{
   		if(arr[i]<arr[i-1])
   		{
   			break;
   		}
   	}
   	if(i!=0){
   		i--;
   	int l=i+1;
   	for(int k=i+2;k<n;k++)
   	{
   		if(arr[k]<arr[i] && arr[l]<arr[k])
   		{
   			l=k;
   		}
   	}

   	int temp=arr[i];
   	arr[i]=arr[l];
   	arr[l]=temp;
   	sort(arr+i+1,arr+n,cmp);
   	}
   	for(int i=0;i<n;i++)
   	cout<<arr[i]<<" ";
   	cout<<"\n";
   }
    return 0;
}
