#include <iostream>
#include <string.h>
using namespace std;
char s[2001][2001];
int ans[2001][2001];
   
int main()
{
   int a,b;
   cin>>a>>b;
   for(int i=1 ; i<=a ; i++){
   		scanf("%s",&s[i][1]);
   }
   memset(ans,0,sizeof(ans));
   for(int i=1 ; i<=a ; i++){
   		for(int j=1 ; j<=b ; j++){
   			ans[i][j]=1;
   			if(i==1 || j==1) ans[i][j]=1;
   			else {
   				if(s[i][j]=='#' && s[i-1][j]=='_' && s[i][j-1]=='_' && s[i-1][j-1]=='#')
   					ans[i][j]=min(ans[i-1][j],min(ans[i][j-1],ans[i-1][j-1]))+1;
   				else if(s[i][j]=='_' && s[i-1][j]=='#' && s[i][j-1]=='#' && s[i-1][j-1]=='_')
   					ans[i][j]=min(ans[i-1][j],min(ans[i][j-1],ans[i-1][j-1]))+1;
   			}
   		}
   }
   long long A=0;
   for(int i=1 ; i<=a ; i++){
   		for(int j=1 ; j<=b ; j++){
   			if(ans[i][j]>=32 && s[i][j]=='_') A++;
   		}
   }
   cout<<A;
   return 0;
}
