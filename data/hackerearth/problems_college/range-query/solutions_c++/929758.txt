#include<stdio.h>
int main()
{

    int nq,n,l,u,i,k,j,val;
    scanf("%d",&n);
    int arr[101][n];
    for(i=0;i<n;i++)
        scanf("%d",&arr[0][i]);

    for(i=1;i<=100;i++)
    {
        k=0;
        for(j=0;j<n;j++)
        {
            if(arr[0][j]==i)
                k++;
            arr[i][j]=k;
        }
    }
    scanf("%d",&nq);
    while(nq--)
    {
        scanf("%d%d%d",&l,&u,&val);
        if(l<0||l>=n||u<0||u>=n||val<=0||val>100)
            k=0;
        else if(l==0)
            k=arr[val][u];
        else
            k=arr[val][u]-arr[val][l-1];

        printf("%d\n",k);
    }

}
