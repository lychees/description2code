#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll arr[1000005];
int main()
{
	int t,n,i,j;
	ll count;
	scanf("%d",&t);
	while(t--)
	{
		count=0;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		scanf("%lld",&arr[i]);
		for(i=0;i<n;i++)
		{
			for(j=i;j<n;j++)
			{
				if(arr[i]==arr[j])
				count++;
			}
		}
		printf("%lld\n",count);
	}


return 0;
}
