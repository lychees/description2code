#include <bits/stdc++.h>
using namespace std;
#define LL long int

int main()
{
    LL t,n,x,y,i;
    cin>>t;
    while(t--)
    {
    	cin>>x>>y>>n;
    	long a[n+1],sum;
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	sort(a,a+n);
    	i=0;sum=0;
    	while(i<n)
    	{
    		sum+=a[i];
    		if(sum>x||sum>y)
    			break;
    		i++;
    	}
    	if(i<n)
    	{
    		if(sum>x&&sum>y)
    			printf("Tie\n");
    		else if(sum>x)
    			printf("Jhool Won\n");
    		else
    			printf("Chef Won\n");
    	}
    	else
    		printf("Tie\n");
    }
    return 0;
}
