#include <iostream>
using namespace std;
bool vis[27];
int main()
{
    //cout << "Hello World!" << endl;
    int n;
    string s;
    cin>>n>>s;
    int si=0,ei=s.length();
    while(si<ei)
    {
    	int maxidx=-1;
    	char max='A';
    	for(int i=si;i<ei;i++)
    	{
    		if(s[i]>max && !vis[s[i]-'a'])
    		{
    			max=s[i];
    			maxidx=i;
    		}
    	}
    	if(maxidx==-1)break;
    	cout<<max;
    	si=maxidx+1;
    	vis[max-'a']=true;
    }
    return 0;
}
