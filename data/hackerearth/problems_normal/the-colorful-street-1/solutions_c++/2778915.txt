#include <iostream>
using namespace std;

int min(int a, int b)
{
	return(a<b?a:b);
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int **arr=new int *[n];
    	for(int i=0;i<n;i++)
    		arr[i]=new int[3];
    	for(int i=0;i<n;i++)
    	{
    		cin>>arr[i][0]>>arr[i][1]>>arr[i][2];
    	}
    	int r=0;
    	int b=0;
    	int g=0;
    	int t=0;
    	int r1,b1;
    	for(int i=0;i<n;i++)
    	{
    		r1=r;
    		r=min(b,g)+arr[i][0];
    		b1=b;
    		b=min(r1,g)+arr[i][1];
    		g=min(r1,b1)+arr[i][2];
    		t=min(r,min(b,g));
    	}
    	cout<<t<<"\n";   	
    }
    return 0;
}
