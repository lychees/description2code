#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
	int n,i;
	scanf("%d",&n);
	long long  a[n],b[n],sum;
	for(i=0;i<n;i++)
	scanf("%lld",&b[i]);
	sum=0;
	a[0]=b[0];
	a[-1]=0;
	for(i=0;i<n;i++)
	{
		sum+=a[i-1];
		a[i]=b[i]*(i+1)-sum;
		printf("%lld ",a[i]);
	}
	printf("\n");
	return 0;
}
