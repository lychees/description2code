#include<iostream>
#include<string>
#include<cstdlib>
#include<cinttypes>
#include<cmath>
#include<stdio.h>
#include<limits.h>
#include<stdint.h>
using namespace std;

void func(int *x, int *a, int *b, int large, int small)
{
	int i,n = large + small;
	x[n - 1] = 1;
	i = n - 2;
	int temp = small - 1;
	while (temp > 0)
	{
		x[i] = 0; i--; temp--;
	}
	temp = large - small;
	while (temp > 0)
	{
		x[i] = 9; i--; temp--;
	}
	x[i] = 8;
	i--;
	while (i >= 0)
	{
		x[i] = 9; i--;
	}
}

int main()
{
	int *x, *a, *b, i;
	int t, k, n;
	cin >> t;
	while (t>0)
	{
		cin >> n >> k;
		x = new int[n];
		a = new int[n - k];
		b = new int[k];
		if (k > n - k) func(x, b, a, k, n-k);
		else func(x, a, b, n-k, k);
		for (i = 0; i < n; i++)
			cout << x[i];
		cout << endl;
		t--;
	}
}
