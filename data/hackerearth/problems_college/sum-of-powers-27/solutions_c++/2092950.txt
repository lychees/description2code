#include <iostream>
using namespace std;
#define ll long long int
 
int main()
{
	int t;
	ll j,n;
	bool isPerfect[1000001];
	int perfect[1000001];
	ll count=0;
	
	isPerfect[1]=true;
	for(ll i=2;i<=1000000;i++){
		if(isPerfect[i])continue;
		for(j=i*i;j<=1000000;j*=i){
			isPerfect[j]=true;
		}
	}
	
		for(int i=1;i<=1000000;i++){
			if(isPerfect[i]) perfect[count++]=i;
		}
		
	
		
	cin>>t;
	while(t--){
		cin>>n;
	
		bool flag=true;
		for(int i=0;i<count && perfect[i]<n;i++){
			if(isPerfect[n-perfect[i]]){
				flag=false;
			
				cout<<"Yes"<<"\n";
				break;
			}
		}
		if(flag){
			cout<<"No"<<"\n";
		}
		
	}
	
	return 0;
}
