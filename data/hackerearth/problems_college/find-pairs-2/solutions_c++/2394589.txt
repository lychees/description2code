#include<bits/stdc++.h>
using namespace std;
long int a[1000000];
int main(void)
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long int n;
		scanf("%ld",&n);
		for(long int i=0;i<n;i++)
		scanf("%ld",&a[i]);
		long long int res=n;
		for(long int i=0;i<n;i++)
		for(long int j=i+1;j<n;j++)
		if(a[i]==a[j])
		res++;
		printf("%lld\n",res);
	}
	return 0;
}
