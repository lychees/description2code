#include <iostream>
#include<algorithm>
#include<vector>
#include<stdio.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    long long int n1,n2,n3,ans;
    vector<long long int >v;
    
    while(t--)
    {
    	 vector<long long int >v;
    	scanf("%lld %lld %lld",&n1,&n2,&n3);
    	v.push_back(n1);
    		v.push_back(n2);
    			v.push_back(n3);
    			sort(v.begin(),v.end());

    			ans =(ans *v[0])%(1000000007);
    			ans=(ans *(v[1]-1) )%(1000000007);
    			ans=(ans *(v[2]-2) )%(1000000007);
    			
    	printf("%lld",ans);cout<<endl;
    
    }

}
