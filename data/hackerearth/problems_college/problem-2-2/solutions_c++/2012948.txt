#include<bits/stdc++.h>

using namespace std;

int main()
{
	int red,white;
	cin>>red>>white;
	int max,min;
	max=(red>=white?red:white);
	min=(red<white)?red:white;
	if(red&&white)
	{
		min++;
		if(max%min==0){
			cout<<(max/min)<<endl;
		}
		else{
			cout<<(max/min)+1<<endl;
		}
	}
	else
	 cout<<max<<endl;
	return 0; 
}
