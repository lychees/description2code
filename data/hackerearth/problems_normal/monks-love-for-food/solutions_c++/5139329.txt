#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int q,r=0,n,c;
    cin>>q;
    int a[q];
    while(q--)
    {
    	cin>>n;
    	if(n==2)
    	{
    		cin>>c;
    		a[r]=c;
    		r++;
    	}
    	else
    	if(n==1)
    	{
    		if(r<0)
    		cout<<"No Food\n";
    		else
    		if(r==0)
    		cout<<"No Food\n";
    		else
    		{
    			r--;
    			cout<<a[r]<<"\n";
    		}
    	}
    	
    }
   
    return 0;
}
