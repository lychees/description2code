#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	char c[1000000];
	while(t--)
	{
		cin>>c;
		int i,j,flag,a[30]={0};
		for(i=0;c[i]!='\0';i++)a[c[i]-'A']++;
		flag=1;
		for(i=0;i<26;i++)
		if(a[i]!=0 && a[i]!=(26-i))flag=0;
		if(flag)cout<<"Yes\n";
		else cout<<"No\n";
	} return 0;
}
