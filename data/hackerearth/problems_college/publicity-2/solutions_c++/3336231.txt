ll gcd(ll n1,ll n2){
	if(n1%n2==0)return n2;
	return gcd(n2,n1%n2);
}
ll powmod(ll base,ll exponent)
{
	ll ans=1;
	while(exponent){
		if(exponent&1)ans=(ans*base)%mod;
		base=(base*base)%mod;
		exponent/=2;
	}
	return ans;
}
vector<int> adj[upperlimit+1];
bool visited[upperlimit+1];
int answer;
void dfs(int node){
	visited[node]=true;
	for(int i=0;i<adj[node].size();i++){
		if(!visited[adj[node][i]])dfs(adj[node][i]);
	}
}

			adj[a].pb(b);

		}
		answer=0;
		for(i=0;i<n;i++)if(!visited[i]){
			dfs(i);
			answer++;
		}
