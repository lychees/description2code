#include<iostream>
#include<cstdio>
#define ll long long
using namespace std;
int f[20000009];
int main()
{
	f[0]=0;
    f[1]=0;
    f[2]=1;
    f[3]=1;
    for(int i=4;i<=20000009;i++)
    {
        int m=f[i-1]+1;
        if(i%2==0 && (i/2)>0)
            m=min(m,f[i/2]+1);
        if(i%3==0 && i/3>0)
            m=min(m,f[i/3]+1);
        f[i]=m;
    }
    int t;
    scanf("%d",&t);
    ll n;
    while(t--)
    {
        scanf("%d",&n);
        printf("%d\n",f[n]);
    }
    return 0;
}
