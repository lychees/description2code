#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
int arr[5005];
ll sum[5005];
ll dp[5005][5005];
int main(){
	int n,m,k;
	cin >> n >> m >> k;
	for(int i=1;i<=n;i++) cin >> arr[i];
	for(int i=1;i<=n;i++) sum[i] = sum[i-1]+arr[i];
	ll ans = 0;
	for(int i=1;i<=n;i++){
		for(int j=0;j<=k;j++){
			dp[i][j] = max(dp[i-1][j],dp[i][j]);
			if(i>=m && j>0){
				dp[i][j] = max(dp[i][j],dp[i-m][j-1]+sum[i]-sum[i-m]);
			}
		}
	}
	for(int i=1;i<=n;i++) ans = max(ans,dp[i][k]);
	cout << ans << endl;
	return 0;
}
