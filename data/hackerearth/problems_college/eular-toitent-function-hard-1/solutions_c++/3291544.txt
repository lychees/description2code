#include <bits/stdc++.h>

using namespace std;

int isP[10000005];
int e[10000005];

void sieve()
{
    int i, j;
    for (i = 2; i <= 10000000; ++i) {
        if (!isP[i]) {
            for (j = 1; i*j <= 10000000; ++j)
                isP[i*j] = i;
        }
    }
}

void eular()
{
    int i;
    e[1] = 1;
    for (i = 2; i <= 10000000; ++i) {
        if (isP[i] == isP[i/isP[i]])
            e[i] = e[i/isP[i]]*isP[i];
        else
            e[i] = e[i/isP[i]]*(isP[i]-1);
    }
}

int main()
{
    sieve();
    eular();
    int t, n, last, x, ans;
    scanf("%d", &t);
    while (t--) {
        scanf("%d", &n);
        printf("%d\n", e[n]);
    }
    return 0;
}
