#include <iostream>
#include<bits/stdc++.h>
using namespace std;

vector<pair<long long , long long> > vec;

int main()
{
    long long n,s,e,i,j,x,d,left,right,ans ;
    cin>>n>>s>>e;
    for(i=0;i<n;i++)
    {
    	cin>>x>>d;
    	left = x - d;
    	right = x + d;
    	if( s >= right || e<=left ) // No need to add the interval which is out of range
    		continue;
    	left = max( left , s );
    	right = min(right , e);
    	vec.push_back(make_pair(left,right));
    }
    if(vec.size()==0)
    {
    	cout<<e-s<<endl;
    	return 0;
    }
    sort(vec.begin() , vec.end());
    
    left =  vec[0].first;
    right = vec[0].second;
    
    ans = max(0LL , left-s); // ans can be zero or left - s as always s will be smaller than left.. 
    
    for(i=1;i<vec.size();i++)
    {
    	if(vec[i].second > right ){
    		ans = ans + max( 0LL , vec[i].first - right );
    		right = vec[i].second;
    	}
    }
    
    ans += max( 0LL , e - right );
    cout<<ans<<endl;
    return 0;
}
