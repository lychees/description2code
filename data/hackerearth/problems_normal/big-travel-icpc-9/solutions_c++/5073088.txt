#include <iostream>
using namespace std;
#include <algorithm>



int main()
{
	long int M = 1000000007;
	int n;
	cin >> n;
	long int arx[n];
	long int ary[n];
	
	for(int i = 0; i < n; i++){
		cin >> arx[i];
		cin >> ary[i];
	}
	sort(arx,arx+n);
	sort(ary,ary+n);
	
	
	long int sum = 0;
	long int dummy = 0;
	for(int i = 0; i < n; i++){
		sum = (sum + (((long int)arx[i]*i)%M - dummy)%M)%M;
		dummy = (dummy + arx[i])%M;
	}
	
	dummy = 0;
	for(int i = 0; i < n; i++){
		sum = (sum + (((long int)ary[i]*i)%M - dummy))%M;
		dummy = (dummy + ary[i])%M;
	}
	if(sum < 0) sum = sum%M + M;
	
	cout << sum << endl;
	
	
    
    return 0;
}
