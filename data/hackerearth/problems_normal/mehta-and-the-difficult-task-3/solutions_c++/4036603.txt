void multiply(long long int M[16][16],long long int F[16][16])
{
	long long int x[16][16];
	long long d=0,s=0;
	long long int i,j,k;
	for(i=0;i<16;i++)
	{
		for(j=0;j<16;j++)
		{
			s=0;
			for(k=0;k<16;k++)
			{
				d=((long long)M[i][k])*(long long)F[k][j];
				if(d>=MOD)
					d=d%MOD;
				s=s+d;
				if(s>=MOD)
					s=s%MOD;
			}
			x[i][j]=s;
		}
	}
	for(i=0;i<16;i++)
	{
		for(j=0;j<16;j++)
		M[i][j]=x[i][j];
	}
}
void power(long long int M[16][16],long long int l)
{
	if(l==0||l==1)
	return ;
	power(M,l/2);
	multiply(M,M);
	if(l%2!=0)
	{
		multiply(M,arr);
	}
}
