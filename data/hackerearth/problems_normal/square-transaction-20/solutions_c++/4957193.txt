#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10;

long long a[N];

int main() {
	int n, t, q;
	scanf("%d", &n);
	a[0] = 0;
	for (int i = 1; i <= n; i++) {
		scanf("%lld", &a[i]);
		a[i] += a[i - 1];
	}
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++) {
		scanf("%d", &q);
		int res = lower_bound(a, a + n + 1, q) - a;
		printf("%d\n", (res == 0 || res == n + 1) ? -1 : res);
	}
	return 0;
}
