#include <iostream>
using namespace std;

long long C[50][50];

void init(){
	for(int i=0;i<50;i++){
		C[i][0]=1;
		C[i][i]=1;
	}
	for(int i=1;i<50;i++){
		for(int j=1;j<i;j++){
			C[i][j]=((i*C[i-1][j-1])/j);
		}
	}
}

void solve(){
	int n,k;
	cin>>n>>k;
	cout<<C[n+k-1][k]<<"\n";
}

int main(){
	init();
	int t;
	cin>>t;
	while(t--){
		solve();
	}
}
