from math import sqrt
from itertools import count, islice


def isPrime(n):  # assuming n > 1
    if n % 2 == 0 or n % 3 == 0:
        return False

    root = int(sqrt(n))

    for i in range(5, root+1, 6):
        if (n % i == 0):
            return False

    for i in range(7, root+1, 6):
        if (n % i == 0):
            return False

    return True


def isPalindrome(num):
    return str(num) == str(num)[::-1]
