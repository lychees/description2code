#include <bits/stdc++.h>
using namespace std;
#define mp make_pair
#define f first
int n,q,i,p[1000000],x,y,a,b,c[1000000],ans,no[1000000];
set<pair<int,int> > s;
int root(int n){
	while(p[n]!=n){
		p[n]=p[p[n]];
		n=p[n];
	}
	return n;
}
int main()
{
   scanf("%d %d",&n,&q);
   for(i=1;i<=n;i++)
   {p[i]=i;c[i]=0;no[i]=1;s.insert(mp(no[i],i));}
   while(q--){
   	scanf("%d %d",&x,&y);
   	a=root(x);
   	b=root(y);
   	
   	if(a!=b)
   {	p[a]=b;
  s.erase(mp(no[a],a));
   s.erase (mp(no[b],b));
   no[b]=no[b]+no[a];
   
   s.insert(mp(no[b],b));
   }
   ans=(*(s.begin())).f-(*(--s.end())).f;
 printf("%d\n",-1*ans);
  }  return 0;
}
