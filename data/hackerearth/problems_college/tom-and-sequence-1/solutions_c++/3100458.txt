#include<bits/stdc++.h>
#define mod 1000000007;
using namespace std;
int  a[5000010];
long long b[5000010];
int main()
{
    long long   i,j;
     for(i=0;i<5000010;i++)
        a[i]=1;
        a[0]=0;
        a[1]=0;
        for(i=2;i<=sqrt(5000008);i++)
        {
            if(a[i]==1)
            {
                for(j=2;i*j<=5000008;j++)
                    a[i*j]=0;
            }
               
        }
        long long k=0;
        for(i=0;i<5000010;i++)
        {
            if(a[i])
            {
                b[k++]=i;
            }
        }
        
    long long  t;
    cin >>t;
    
   while(t--)
   {
       long long n,temp;
       cin>>n;
       temp=1;
       for(i=0;i<n;i++)
       {
           temp=(temp*b[i])%mod;
       }
       temp=temp%mod;
       cout<<temp<<endl;
   }
    return 0;
}
