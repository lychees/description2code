#include<bits/stdc++.h>

using namespace std;

#define sci(x) scanf("%d",&x)
#define scl(x) scanf("%lld",&x)
#define pri(x) printf("%d",x)
#define pll(x) printf("%lld",x)
#define nl printf("\n")
#define MOD 1000000007
typedef long long ll;
int search(ll a[], ll beg, ll end, ll item)
{
        ll mid;
        while(beg<=end)
        {
                mid=(beg+end)/2;
                if(a[mid]==item)
                {
                        return 1;
                }
                else if(a[mid]<item)
                {
                        end=mid-1;
                }
                else
                        beg=mid+1;
        }
        return 0;
}
int main()
{
        ll n,m;
        scl(n); scl(m);
        ll i;
        ll a[n];
        for(i=0;i<n;++i)
                scl(a[i]);
        sort(a,a+n);
        ll check,j,k;
        int flag;
        for(i=0;i<n-2;++i)
        {
                flag=0;
                for(j=i+1;j<n-1;++j)
                {
                        check=m-a[i]-a[j];
                        flag=search(a,j+1,n,check);
                        if(flag==1)
                                break;
                }
                if(flag==1)
                        break;
        }
        if(flag)
                printf("YES\n");
        else
                printf("NO\n");
    return 0;
}
