#include <bits/stdc++.h>
using namespace std;
int adda (char a,char b){
int i=(int)a;
int j=(int)b;
i-=48;j-=48;
if(i+j>9)
return 1;
else
return 0;
}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	char a[1000],b[1000];int count=0;
    	cin>>a>>b;
    	for(int i=0;i<strlen(a);i++){
    		if(adda(a[i],b[i]))
    		count++;
    	}
    	if(count==1)
    	cout<<count<<" carry operation"<<endl;
    	else if(count>1)
    	cout<<count<<" carry operations"<<endl;
    	else
    	cout<<"No carry operation"<<endl;
    }
    return 0;
}
