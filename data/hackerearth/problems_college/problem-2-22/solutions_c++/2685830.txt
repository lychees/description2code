#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		int r, cx, cy, flag = 0;
		cin >> r >> cx >> cy;
		int px[4], py[4];
		for (int i = 0; i < 4; i++)
			cin >> px[i] >> py[i];
			
		for (int i = 0; i < 4; i++)
		{
			if (pow(cx - px[i], 2) + pow(cy - py[i], 2) > pow(r, 2))
			{
				flag = 1;
				cout << "No" << endl;
				break;
			}
		}
		if (flag != 1)
			cout << "Yes" << endl;
	}
	return 0;
}
