#include <bits/stdc++.h>
using namespace std;
#define ll long  int
ll arr[100005],pos[100005],brr[100005];
bool comp(ll a,ll b)
{
	return a>b;
}
int main()
{
	ll n,k,i,temp,kl;
	cin>>n>>k;
	arr[0]=0;
	brr[0]=0;
	for(i=1;i<=n;i++)
	{
		cin>>arr[i];
		pos[arr[i]]=i;
		brr[i]=arr[i];
	}
	sort(brr+1,brr+n+1,comp);
	for(i=1;i<=n&&k;i++)
	{
		if(brr[i]!=arr[i])
		{
		//	cout<<brr[i]<<" "<<arr[i]<<endl;
			kl=pos[brr[i]];
			arr[kl]=arr[i];
		//	cout<<kl<<" "<<arr[kl]<<" "<<arr[i]<<endl;
			pos[arr[i]]=kl;
			pos[brr[i]]=i;
			arr[i]=brr[i];
		//	cout<<pos[arr[i]]<<" "<<pos[brr[i]]<<" "<<arr[i]<<endl;
			k--;
		}
	}
	for(i=1;i<=n;i++)
	cout<<arr[i]<<" ";
    
    return 0;
}
