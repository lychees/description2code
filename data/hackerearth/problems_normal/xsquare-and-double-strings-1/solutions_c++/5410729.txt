#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int hashs[26]={0},i,f=1;
    	for(i=0;i<s.size();i++)
    	{
    		hashs[s[i]-'a']++;
    	}
    	for(i=0;i<26;i++)
    	{
    		if(hashs[i]>=2)
    		{f=0;break;}
    	}
    	if(f==0)
    	   cout<<"Yes\n";
    	else
    	   cout<<"No\n";
     }
    return 0;
}
