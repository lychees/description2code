int has1[26];
int has2[26];

int main()
{
    int t,i,ans;
    s(t);
    while(t--)
    {
        ans=0;
        memset(has1,0,sizeof(has1));
        memset(has2,0,sizeof(has2));
        cin>>str1;
        cin>>str2;
        for(i=0;i<str2.length();i++)
            has2[str2[i]-'a']++;
        for(i=0;i<str1.length();i++)
            has1[str1[i]-'a']++;
        for(i=0;i<26;i++)

            if(has1[i]>has2[i])

                ans=-1;
                break;
            }
            if(has1[i]<has2[i]&&has2[i]!=0)
                ans++;
        }
            printf("%d\n",ans);
    }

}
