#include<bits/stdc++.h>
#define mx 100005
#define mod 1000000007
#define ll long long
#define pb push_back
using namespace std;
int a[mx];
ll sum[mx];
int main()
{
    int t;
    scanf("%d",&t);
    for(int c=1;c<=t;c++)
    {
        memset(a,-1,sizeof a);
        ll ans=0;
        int n=50000;
        scanf("%d",&n);
        sum[0]=0;
        for(int i=1;i<=n;i++)
        {
            ll x=100000;
            scanf("%lld",&x);
            //if(i!=n)x=1;
            sum[i]=sum[i-1]+x;
            a[x]=i;
        }
        //printf("%lld\n",sum[n]);
        ll pre=0;
        for(ll i=mx-1;i>=1;i--)
        {
            if(a[i]!=-1 && pre<a[i])
            {
                ll j=a[i]-pre-1;
               ans+= (j*i-(sum[a[i]-1]-sum[pre]));
               pre=a[i];
            }
        }
        printf("%lld\n",ans);

    }


}
