#include <iostream>
#include<set>
//#include<algorithm>
using namespace std;

int main()
{
    set<long int> st;
    int t;
    cin>>t;
    while(t--)
    {
    	int n,m;
    	long int x;
    	cin>>n >>m;
    	for(int i=0;i<n;i++)
    	{
    		cin>>x;
    		st.insert(x);
    	}
    	for(int i=0;i<m;i++)
    	{
    		cin>>x;
    		set<long int>::iterator it;
    		it=st.find(x);
    		if(it!=st.end())
    		cout<<"YES"<<endl;
    		else
    		{
    		cout<<"NO"<<endl;
    		st.insert(x);
    		}
    	}
    	st.clear();
    }
    return 0;
}
