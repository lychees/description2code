#define REP(i,a) for(i=0;i<a;i++)
#define FOR(i,a,b) for(i=a;i<b;i++)

double evaluate(int k, int n){
	int i;
	double res = 1.0;
	REP(i,k){
		res *= (n-i)*1.0 /n;
	}
	return res;
}
int main()
{
	int t;
	int k,n;
	double res;
	scanf("%d",&t);
	while(t-->0){
		scanf("%d %d",&k,&n);
		if(k>n)res = 1.0;
		else{
			res = 1 -evaluate(k,n);
		}
		printf("%.9lf\n",res);
	}	
	return 0;

}
