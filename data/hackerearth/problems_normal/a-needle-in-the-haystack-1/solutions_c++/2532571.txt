#include <iostream>
#include <cstring>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    char c;
    cin.get(c);
    while(t--)
    {
        string p,hay;
        cin>>p>>hay;
        int lp = p.length();
        int lh = hay.length();
        int p_abc[26] = {0}; 
        int found = 0;
        for(int i=0;i<lp;i++)
            p_abc[p[i]-'a']++;
        int checker_abc[26] = {0}; 
        for(int i=0;i<lp;i++)
        {
            checker_abc[hay[i]-'a']++;
        }
        for(int i=lp;i<lh;i++)
        {
            found = 1;
            for(int k=0;k<26;k++)
            {
                if(p_abc[k] != checker_abc[k])
                {
                    found = 0;break;
                }
            }
            if(found)
            {
                break;
            }
            checker_abc[hay[i-lp]-'a']--;
            checker_abc[hay[i]-'a']++;
        }
        found =1;
        for(int k=0;k<26;k++)
            {
                if(p_abc[k] != checker_abc[k])
                {
                    found = 0;break;
                }
            }
        if(found)
        	cout<<"YES"<<endl;
        else
        	cout<<"NO"<<endl;
    }
    return 0;
}
