#include<iostream>
#include<string>
#include<cstring>
#include<climits>
using namespace std;
long long int counter[200];
//int arr[]
bool isbalanced(string str,long long int i,long long int j)
{
    for(long long int k=i;k<=j;k++)
    {
        counter[str[k]]++;
    }

    long long int flag = 1;
    for(long long int k=97;k<=122;k++)
    {
        //cout<<counter[k]<<' ';
        if(counter[k]%2!=0)
            flag = 0;
    }
    //cout<<endl;
    memset(counter,0,sizeof counter);
    return flag;
}
long long int calc(string str,long long int i,long long int j)
{
   // cout<<i<<' '<<j<<' '<<isbalanced(str,i,j)<<endl;
    if(j-i<=2&&str[i]==str[j])
        return 1;
    else if(isbalanced(str,i,j))
    {
        //cout<<'b'<<endl;
        return 1;
    }
    else
    {
        return -1;
    }
}
int main()
{
    long long int t;
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin>>t;
    string str;
    long long int l;
    while(t--)
    {
        cin>>str;
        l=str.length();
        if(l%2!=0)
            cout<<-1<<endl;
        else
        {
            cout<<calc(str,0,l-1)<<endl;
        }
           // cout<<<<endl;
    }
}
