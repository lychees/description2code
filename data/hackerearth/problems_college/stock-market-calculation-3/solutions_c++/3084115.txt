#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,j;
	scanf("%lld",&t);
	for(j=0;j<t;j++)
	{
		long long int i,n,max=0,sum=0,x=0,c=0,k,p=0;
		scanf("%lld",&n);
		long long int a[n];
		for(i=0;i<n;i++)
		{
			scanf("%lld",&a[i]);
		}
		
		while(p<n)
		{
			c=0;
			max=0;
			sum=0;
			for(i=p;i<n;i++)
			{
				if(a[i]>=max)
			{
				k=i;
			max=a[i];}
			}
			i=p;
			while(i<k)
			{
				sum=sum+a[i];
				i++;
				c++;
			}
			if(sum!=0)
			x=x+c*a[k]-sum;
			p=k;
			p++;
		}
		printf("%lld\n",x);
	}
	return 0;
}
