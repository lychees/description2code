    #include<iostream>
    #include<algorithm>
    #include<vector>
    using namespace std;
    vector<int>v;
    int main(){
    	int t,n;
    	cin>>t;
    	while(t--){
    		v.clear();
    		cin>>n;
    		while(n--){
    			int x; 
    			cin>>x;
    			v.push_back(x);
    		}
    		sort(v.begin(), v.end());
    		int ans=0;
    		for(int i=1;i<v.size();i++)
    		v[i]=v[i]-v[0];
    		for(int i=1; i<v.size(); i++){
    			if(v[i]/5 >0){
    				ans+=(v[i]/5);
    				v[i]%=5;
    			}
    			if(v[i]/2 >0){
    				ans+=(v[i]/2);
    				v[i]%=2;
    			}
    			if(v[i]) ans++;
    		}
    		cout<<ans<<endl;
    	}
    	return 0;
    }
