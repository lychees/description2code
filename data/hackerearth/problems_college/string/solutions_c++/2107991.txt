#include <bits/stdc++.h>
#define ll long long
using namespace std;
typedef struct node
{
	char c;
	ll int mc;
}node;
bool my(node a,node b)
{
	if(a.mc<b.mc)
	return true;
	return false;
}
int main()
{
	ll int k;
	char str[100010];
	scanf("%s%lld",str,&k);
	ll int i;
	node h[26];
	for(i=0;i<26;i++)
	{
		h[i].c=(char)('a'+i);
		h[i].mc=0;
	}
	for(i=0;str[i]!='\0';i++)
	{
		(h[str[i]-'a'].mc)++;
	}
	//~ cout<<"here "<<endl;
	sort(h,h+26,my);
	for(i=0;i<26;i++)
	{
		if(h[i].mc<=k)
		{
			k-=h[i].mc;
			h[i].mc=0;
		}
		else
		break;
	}
	ll int ans=0;
	//~ cout<<"here "<<endl;
	for(i=0;i<26;i++)
	{
		if(h[i].mc!=0)
		{
			ans++;
		}
	}
	ll int h2[26];
	memset(h2,0,sizeof(h2));
	for(i=0;i<26;i++)
	{
		h2[h[i].c-'a']=h[i].mc;
	}
	cout<<ans<<endl;
	string s="";
	for(i=0;str[i]!='\0';i++)
	{
		if(h2[str[i]-'a'] >0 )
		{
			s+=(char)str[i];
		}
	}
	cout<<s<<endl;
	return 0;
}
