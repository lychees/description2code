#include<iostream>
using namespace std;
int main()
{
string str1,str2;
int x,t,length,i;
cin>>t;
while(t--)
{
    x=0;
cin>>str1;
cin>>str2;
length=str1.size();
int c[length];
for(i=0;i<length;i++)
{
    x=(int)str1[i]-str2[i];

    if((x==1)||(x==-1))
    {
        c[i]=1;
    }
        else
        {
            c[i]=0;
        }
    }
for(i=0;i<length;i++)
{
    cout<<c[i];
}
cout<<endl;
}
return 0;
}
