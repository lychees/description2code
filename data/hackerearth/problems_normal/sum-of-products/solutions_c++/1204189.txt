#include<stdio.h>
#include<iostream>
#include<algorithm>
#define MOD 1000000007
using namespace std;
int main()
{
	long long int t,n;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		long long int s=0;
		long long int d=1;
		long long int q=n;
		while(d<q)
		{
			s+=q*(q+1+2*d)/2;
			d++; 
			q=n/d; 
		}
		cout<<((s-d*(d-1)/2*d+q*(q+1)/2))%MOD<<endl;
	}
	return 0;
}
