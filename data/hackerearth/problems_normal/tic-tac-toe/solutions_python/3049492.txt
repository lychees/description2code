t = input()
mod = 1000000007
while t!=0:
	n = input()
	n %= mod
	maxx = ((2*n-1)*(n-1)*n)/6
	maxx %= mod
	m = ((n-1)*n)/2
	minwins = m/n
	carry = m%n;
	minn = (n-carry)*minwins*minwins
	minn %= mod
	minwins += 1
	minn += carry*minwins*minwins
	minn %= mod
	print str(minn) + " " + str(maxx)
	t -= 1
