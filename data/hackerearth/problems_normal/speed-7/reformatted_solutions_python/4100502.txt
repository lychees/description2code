n = int(raw_input().strip())
for i in xrange(n):
    number = int(raw_input().strip())
    cars = map(int, raw_input().strip().split(" "))
    count = 1
    minimum = cars[0]
    if number == 1:
        print 1
    else:
        for j in cars[1:]:
            if j < minimum:
                count += 1
                minimum = j
        print count
