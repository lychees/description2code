#include <iostream>
using namespace std;
 

int trailingcount(int  n)
{
    int count = 0;
    for (int i=5; n/i>=1; i *= 5)
 	{
 		count += n/i;
 	}
  	return count;
}
 int main()
{
    int t,num;
    cin >>t;
    for(int test=0;test<t;test++)
    {
    	cin >> num;
    	cout << trailingcount(num)<<endl;
    }
}
