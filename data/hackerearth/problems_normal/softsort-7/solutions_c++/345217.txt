#include <iostream>
using namespace std;
#define mod 1000000007
long long fac[1000000];
void fac_fun()
{
	fac[0]=1;
	fac[1]=1;
	fac[2]=2;
	for(int i=3;i<1000000;i++)
	{
		fac[i]=fac[i-1]*i;
		fac[i]=fac[i]%mod;
	}
}
int main()
{
	fac_fun();
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	long long ans=((3*fac[n])%mod)+3;
    	cout<<ans<<endl;
    }
    return 0;
}
