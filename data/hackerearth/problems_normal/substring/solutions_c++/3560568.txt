#include <iostream>
#include <string>
using namespace std;

bool check(char a)
{
	if(a=='A' || a=='a' || a=='B' || a=='b' || a=='C' || a=='c')
		return true;
	return false;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a;
    	cin>>a;
    	int result=0;
    	int current=0;
    	for(int i=0;i<a.size();i++)
    	{
    		if(check(a.at(i)))
    			current++;
    		else
    		{
    			result+=(current*(current+1))/2;
    			current=0;
    		}
    	}
    	result+=(current*(current+1))/2;
    	cout<<result<<endl;
    }
    return 0;
}
