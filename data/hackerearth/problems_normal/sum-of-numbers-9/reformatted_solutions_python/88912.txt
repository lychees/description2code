from itertools import combinations


def get_case():
    N = int(raw_input())
    I = map(int, raw_input().split(' '))
    S = int(raw_input())
    return (N, I, S)


T = int(raw_input())

for i in range(T):
    N, I, S = get_case()
    ans = 'NO'
    for i in range(N):
        for c in combinations(I, i+1):
            if sum(c) == S:
                ans = 'YES'
    print ans
