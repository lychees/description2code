#include<stdio.h>
int a[1010][5]={0};
int main()
{
        int n,q,i,j,u,v;
        scanf("%d %d",&n,&q);
        for(i=0;i<q;i++)
        {
        		int f=1;
                scanf("%d %d",&a[i][1],&a[i][2]);
                if(a[i][1]==a[i][2])
                {
                	f=0;
                	goto A;
                }
                if(a[i][1]>a[i][2])
                {
                	int temp=a[i][1];
                	a[i][1]=a[i][2];
                	a[i][2]=temp;
                }
                for(j=0;j<i;j++)
                {
                		if(a[j][0]==1)
                		{
                		if(a[j][1]==a[i][1] || a[j][1]==a[i][2] || a[j][2]==a[i][1] || a[j][2]==a[i][2])
                		{
                			f=0;
                			break;
                		}
                        else if(a[j][1]<a[i][1])
                        {
                        	if(a[j][2]>a[i][1] && a[j][2]<a[i][2])
                        	{
                        		f=0;
                        		break;
                        	}
                        }
                        else if(a[j][1]>a[i][1] && a[j][1]<a[i][2])
                        {
                        	if(a[j][2]>a[i][2])
                        	{
                        		f=0;
                        		break;
                        	}
                        }
                		}
                }
      			A:a[i][0]=f;
                if(f)
                {
                	printf("YES\n");
                }
                else
                {
                	printf("NO\n");
                }
        }
        return(0);
}
