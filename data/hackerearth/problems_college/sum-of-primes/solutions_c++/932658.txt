#include <bits/stdc++.h>

#define LL long long
#define MAX 1000050
#define MD 1000000007
#define N 1000007
#define check_bit(a,b) (a & (1<<b))
#define PB push_back
#define FOR(aa,nn) for(aa=0; aa<nn; aa++)


using namespace std;

bool is_prime[MAX];
int prime[MAX];
int sz;
int sieve()
{
    long i,j;
    prime[1]=2;
    prime[0]=0;
    int k=2;
    int sq=(sqrt(MAX));
    for(i=3; i<=sq; i+=2)
    {
        if(!is_prime[i])
        {
            for(j=i*i; j<=MAX; j+=(2*i))
                is_prime[j]=1;
        }
    }
    for(j=3; j<=MAX; j+=2)
    {
        if(!is_prime[j])
        {
            prime[k++]=j;
        }
    }
    return k;
}


int BS1(int num)
{
    int st = 0, ed = sz, mid;
    int ans = sz;

    while(st <= ed)
    {
        mid = (st + ed)/2;
        if(prime[mid] == num) return mid;
        if(num < prime[mid])
        {
            ans = min(ans, mid);
            ed = mid - 1;
        }
        else
            st = mid + 1;
    }
    return ans - 1;
}


int BS2(int num)
{
    int st = 0, ed = sz, mid;
    int ans = 0;

    while(st <= ed)
    {
        mid = (st + ed)/2;
        if(prime[mid] == num) return mid - 1;
        if(num < prime[mid])
        {
            ed = mid - 1;
        }
        else
        {
            st = mid + 1;
            ans = max(ans, mid);
        }
    }
    return ans;
}

LL sum[MAX];

bool check(LL num)
{
    if(num == 0) return false;
    LL sq = sqrt(num);
    for(int i = 1; i < sz && prime[i] <= sq; i++)
        if(num % prime[i] == 0) return false;
    return true;
}

int main()
{
    //   FILE * fin, * fout, *fp;
//     fp=fopen("out.txt", "w");

    int i,j;
    string str, tmp;
    int t, x;
    int n;
    sz = sieve();

    for(i = 0; i < sz; i++)
    {
        sum[i] = sum[i - 1] + prime[i];
    }
    int a, b;
    cin>>t;
    while(t--)
    {
        cin>>a>>b;
        if(a > b)
            swap(a, b);
        int c = BS2(a);
        int d = BS1(b);
        LL tot = sum[d] - sum[c];


        if(check(tot))
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;

    }




    return 0;
}
