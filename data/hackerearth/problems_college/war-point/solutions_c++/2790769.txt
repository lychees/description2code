#include<bits/stdc++.h>
using namespace std;



int main()
{

	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{

		int n;
		long long int sum = 0, left=0, right=0, diff=INT_MAX;
		cin>>n;
		int *arr = new int[n];
		for(int i=0; i<n; i++)
		{
            cin>>arr[i];
			sum+=arr[i];
			
		}
		left=arr[0];
		right=sum-left;
		diff = abs(right-left);
		for(int i=1; i<n; i++)
		{

			left+=arr[i];
			right=sum-left;
			if(abs(right-left)<diff)diff=abs(right-left);
			
		}
		cout<<diff<<endl;
	}
	return 0;
}
