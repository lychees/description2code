#include <cstdio>
#include <utility>

using namespace std;

typedef long long ll;
pair<ll,ll> niz[1000001];

const int MOD=1e9+7;

int main()
{
    niz[1].first=1;
    niz[1].second=1;
    for(int i=2;i<1000001;i++){
        niz[i].second=(i*niz[i-1].first)%MOD;
        niz[i].first=(niz[i-1].first+niz[i-1].second)%MOD;
    }

    int t;
    scanf("%d",&t);
    for(int loo=0;loo<t;loo++){
        int n;
        scanf("%d", &n);
        printf("%d\n",niz[n].first);
    }
    return 0;
}
