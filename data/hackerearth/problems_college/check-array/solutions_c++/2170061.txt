#include <iostream>
using namespace std;

int main()
{
    int T,N,Ar[100005];
    cin>>T;
    
    while(T--)
    {
    	cin>>N;
    	for(int i=0;i<N;i++)cin>>Ar[i];
    	
    	int i=0,j=N-1;
    	long long sum1=Ar[0],sum2=Ar[N-1];
    	while(i<j-2)
    	{
    		if(sum1<=sum2)
    			i++,sum1+=Ar[i];
    		else
    			j--,sum2+=Ar[j];
    	}
    	if(sum1==sum2)
    	cout<<"YES\n";
    	else cout<<"NO\n";
    }
    return 0;
    
}
