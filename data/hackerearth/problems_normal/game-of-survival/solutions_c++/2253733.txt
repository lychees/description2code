#include<bits/stdc++.h>

using namespace std;

int main()
{
	long int t;
	cin>> t;
	
	while(t--)
	{
		long int n;
		cin>> n;
		
		long int i,s[n],a[n],count = 0;
		
		for( i = 0; i < n; i++)
		cin >> s[i];
		
		for( i = 0; i < n; i++)
		cin >> a[i];
		
		sort(&s[0],&s[n]);
		sort(&a[0],&a[n]);
		
		for( i = 0; i < n; i++)
		{
			if(s[i] >= a[i])
			count++;
		}
		
		if( count == n)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
	}
	
	return 0;
	
}
