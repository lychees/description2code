#include <iostream>
#include <algorithm>
#define endl '\n'

using namespace std;

int main(int argc, char **argv){
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	int k, max, count = 0;
	int a[1000000];
	cin >> k;
	for(int i = 0; i < k; i++){
		cin >> a[i];
	}
	sort(a,a+k);
	max = 0;
	for(int i = k-1; i >= 0; i--){
		count++;
		a[i] = a[i] + count + 1;
		if(a[i] > max){
			max = a[i];
		}
	}
	//max+=1;
	cout << max << endl;

	return 0;
}
