void sieve()
{
	for(int i=2;i<=Max;i++)
	{
		A[i]=1;
	}
	int s=sqrt(Max);
	for(int i=2;i<=s;i++)
	{
		if(A[i])
		{
			for(int j=i*i;j<=Max;j+=i)
				A[j]=0;
		}
	}
	V.pb(2);
	for(int i=3;i<=Max;i+=2)
	{
		if(A[i])
			V.pb(i);
	}
}

int fast_gcd(int a, int b)
{
    while(b) b ^= a ^= b ^= a %= b;
    return a;
}

			C[j]+=1;
	}
}
