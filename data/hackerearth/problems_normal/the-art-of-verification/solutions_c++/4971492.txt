#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    
    string ch;
    int c=1,j;
    cin>>ch;
    
    for(int i=0;i<ch.size();i++)
    {
    	if(c==1 && ch[i]=='=')
    	{
    		cout<<"username: ";
    		
    		for( j=i+1;(ch[j]!='&' || ch[j+1]!='p');j++)
    			cout<<ch[j];
    		i=j;
    		cout<<endl;
    			
    		c++;
    		continue;
    	}
    	if(c==2 && ch[i]=='=')
    	{
    		cout<<"pwd: ";
    		
    		for(j=i+1;(ch[j]!='&' || ch[j+1]!='p');j++)
    			cout<<ch[j];
    			i=j;
    		
    		cout<<endl;
    			
    		c++;
    		continue;
    	}
    	if(c==3 && ch[i]=='=')
    	{
    		cout<<"profile: ";
    		
    		for(j=i+1;(ch[j]!='&' || ch[j+1]!='r' );j++)
    			cout<<ch[j];
    			i=j;
    			
    			cout<<endl;
    		c++;
    		continue;
    	}
    	if(c==4 && ch[i]=='=')
    	{
    		cout<<"role: ";
    		
    		for(j=i+1;(ch[j]!='&' || ch[j+1]!='k');j++)
    			cout<<ch[j];
    			i=j;
    			cout<<endl;
    		c++;
    		continue;
    	}
    	if(c==5 && ch[i]=='=')
    	{
    		cout<<"key: ";
    		
    		for(j=i+1;j<ch.size();j++)
    			cout<<ch[j];
    			i=j;
    			cout<<endl;
    		c++;
    		continue;
    	}
    }
    return 0;
}
