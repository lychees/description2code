#include<bits/stdc++.h>
using  namespace std;
int dp[1<<20];
int solve(int num)
{
	if(dp[num]!=0)
		return dp[num];
	for(int i=4;i<=num;++i)
	{
		dp[i]=i;
		for(int j=1;j<=i;++j)
		{
			int temp=j*j;
			if(temp>i)
				break;
			else 
			{
				dp[i]=min(dp[i],1+dp[i-temp]);
			}
		}
	}
	return dp[num];
}
int main()
{
	memset(dp,0,sizeof dp);
	dp[1]=1;
	dp[2]=2;
	dp[3]=3;
	int tc;
	cin>>tc;
	while(tc--)
	{
		int n;
		cin>>n;
		int ans=solve(n);
		cout<<ans<<endl;
	}
	return 0;
}
