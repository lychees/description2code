#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    int n,m;
    while(t--){
    cin>>n>>m;
    int x,l;
    priority_queue<int> pq;
    for(int i=0;i<n;i++)
    {
        cin>>x;
        if(i==0)
           l=x;
        pq.push(x);
    }
    int count=0;
    while(!pq.empty())
    {
        x=pq.top();
        if(x==l)
          break;
        else
        count++;
        pq.pop();
    }
    if(!pq.empty()){
    	pq.pop();
    count++;
    }
    /*while(!pq.empty()&&l==pq.top())
    {
    	count++;
    	pq.pop();
    }*/
    int count1=0;
    //cout<<count<<endl;
    while(count>m)
    {
        count=count-m;
        count1++;
    }
    cout<<count1<<endl;
    }
    return 0;
}
