#include <bits/stdc++.h>
using namespace std;

int main()
{
	long int t,i=0;
	cin>>t;
	string s[t];
	while(i<t)
	{
		cin>>s[i];
		i++;
	}
	i=0;
	map<string,long> m;
	while(i<t)
	{
		m[s[i]]=0;
		i++;
	}
	i=0;
	while(i<t)
	{
		m[s[i]]++;
		i++;
	}
	long int max=0;
	i=0;
	while(i<t)
	{
		if(m[s[i]]>max)
		{
			max=m[s[i]];
		}
	
		i++;
		
	}
	i=0;
	long int count=0;
	map<string,long>::iterator it;
	it=m.begin();
	while(it!=m.end())
	{
//			cout<<m[it->first]<<"  "<<it->first<<" \n";
		if(max==m[it->first])
		{
			count=count+m[it->first];
		}
		it++;
	}
	cout<<count;
}
