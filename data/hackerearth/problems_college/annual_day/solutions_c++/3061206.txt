#include <iostream>
#include <algorithm>
#include <functional> 
using namespace std;

int main()
{
    int t,i;
    cin>>t;
    while(t--)
    {
    	int n,k,ans=0;
    	cin>>n>>k;
    	int a[n],b[n];
    	
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	for(i=0;i<n;i++)
    	{
    		b[i]=__builtin_popcount(a[i]);
    		
    	}
    	sort(b, b+n,greater<int>());
    	for(i=0;i<k;i++)
    	{
    		ans+=b[i];
    	}
    	
    	cout<<ans<<"\n";
    }
    return 0;
}
