#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll t,n,k,dp[205][205];
ll calc(ll people,ll smallest)
{
    if(dp[people][smallest]!=-1)
        return dp[people][smallest];

    if(people<smallest)
        return 0;

    if(people<2*smallest)
        return 1;

    ll ans=1;
    for(ll i=smallest;i<people;i++)
        ans+=calc(people-i,i);
    dp[people][smallest]=ans;
    return ans;
}
int main()
{
    cin>>t;
    while(t--)
    {
        memset(dp,-1,sizeof(dp));
        cin>>n>>k;
        cout<<calc(n,k)<<endl;
    }
    return 0;
}
