#include <iostream>
#include<string.h>
using namespace std;
int search(char a,char b[])
{
	int l=strlen(b);
	for(int i=0;i<l;i++)
{	   if(b[i]==a)
	   return 1;
}
	   return 0;
}

int main()
{
    int n,i,c,l,j,f,d[130]={0},g;
    char s[101][101];
    cin>>n;
    for(i=0;i<n;i++)
    cin>>s[i];
    l=strlen(s[0]);
    for(i=0;i<l;i++)
    {   c=0;f=0;
    	for(j=1;j<n;j++)
    	{   
    		f=search(s[0][i],s[j]);
    		
    		if(!f)
    		break;
    		else 
    		c++;
    		
       	}
       	if(c==n-1)
       	{if(d[s[0][i]]==0)
       	{d[s[0][i]]+=1;
        g++;}}
        
    }
    cout<<g<<'\n';
    return 0;
}
