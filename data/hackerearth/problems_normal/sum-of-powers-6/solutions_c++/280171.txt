void precomputation()
{
	int nums[10000];
	int i,n=1000000,ind=0,temp,j;
	
	//FOR(i,0,1000001)
	//	status[i]=false;
	nums[ind++]=1;
	FOR(i,2,1000)
	{
		temp=i;
		while(1)
		{
			temp*=i;
			if(temp<n)
				nums[ind++]=temp;
			else
				break;
		}
	}
	
	//printf("%d",ind);	
	FOR(i,0,ind)
		FOR(j,i,ind)
			status[nums[i]+nums[j]]=true;
		
}
int main()
{
	precomputation();
	int t,a;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&a);
		if(status[a])
			printf("Yes\n");
		else
			printf("No\n");
	}
	return 0;
}
