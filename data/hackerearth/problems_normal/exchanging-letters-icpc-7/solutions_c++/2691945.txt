typedef long long ll;

const int N = 2e3 + 7;
const int mod = 1e9 + 7;

int ncr[N][N];

void init()
{
	ncr[0][0] = 1;
	for(int i=1;i<N;i++)
	{
		for(int j=0;j<=i;j++)
		{
			if(j == 0 or j==i) ncr[i][j] = 1;
			else if(j>i) ncr[i][j] = 0;
			else ncr[i][j] = (ncr[i-1][j-1] + ncr[i-1][j]) % mod;
		}
	}
}

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("I.in","r",stdin);
	#endif
	init();
	int t,n,k,x;
	int sm = 0;
	scanf("%d%d%d",&n,&k,&t);
	for(int i=0;i<k;i++)
	{
		scanf("%d",&x);
		sm += x;
	}
	int tp = t-sm+k;
	if(tp&1) puts("0");
	else 
	{
		tp=tp/2;
		if(tp<0 or tp>k) puts("0");
		else printf("%d\n", ncr[k][tp]);
	}

	return 0;
}
