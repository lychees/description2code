#include <iostream>

using namespace std;

int main() {
    int t;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        string outputs[61];
        for(int i=1;i<61;i++){
            outputs[i]="";
        }
        while(n--){
            long int x;
            scanf("%ld",&x);
            int no1s=__builtin_popcount(x);
            if(no1s==1){
                printf("%ld",x);
                printf(" ");
            }
            else{
                outputs[no1s].append(to_string(x));
                outputs[no1s].append(" ");
            }
        }
        for(int i=2;i<61;i++){
                cout<<outputs[i];
        }
        printf("\n");
    }
    return 0;
}
