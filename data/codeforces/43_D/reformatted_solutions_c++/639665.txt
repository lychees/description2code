#define li long long int
#define rep(i, to) for (li i = 0; i < ((li)to); i++)
#define pb push_back
#define sz(v) ((li)v.size())

vector<pair<li, li>> ans;

void cal(li n, li m) {
  for (li i = 1; i <= m; i++) {
    if (i % 2)
      for (li j = 2; j <= n; j++)
        ans.pb(make_pair(i, j));
    else
      for (li j = n; 2 <= j; j--)
        ans.pb(make_pair(i, j));
  }
  for (li i = m; 1 <= i; i--)
    ans.pb(make_pair(i, 1));
}

void cal0(li n, li m) {
  cout << "0\n1 1" << endl;
  if (m % 2) {
    cal(m, n);
    rep(i, sz(ans)) cout << ans[i].second << " " << ans[i].first << endl;
    return;
  }
  cal(n, m);
  rep(i, sz(ans)) cout << ans[i].first << " " << ans[i].second << endl;
}

int main() {
  li n, m;
  cin >> m >> n;
  if ((n % 2) * (m % 2) || ((n == 1 || m == 1) && 2 < n * m)) {
    cout << "1" << endl << m << " " << n << " 1 1" << endl;
    for (li i = 1; i <= m; i++) {
      if (i % 2)
        for (li j = 1; j <= n; j++)
          cout << i << " " << j << endl;
      else
        for (li j = n; 1 <= j; j--)
          cout << i << " " << j << endl;
    }
    cout << "1 1" << endl;
  } else {
    cal0(n, m);
  }
}
