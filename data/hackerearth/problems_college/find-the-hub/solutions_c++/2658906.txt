#include <iostream>
#include <map>
#include <climits>
#define X 100000
#include <string.h>
using namespace std;
bool arr1[X+1];

void markMultiples(int a)
{
    int i = 2, num;
    while ( (num = i*a) <= X )
    {
        arr1[num] = 1; 
        ++i;
    }
}
 
void prime()
{
        memset(arr1, 0, sizeof(arr1));
        for (int i=2; i<=X; ++i)
        {
            if ( arr1[i] == 0 )
            {
                markMultiples(i);
            }
        }
    
}
int main()
{
    int n;
    cin>>n;
    map<int, int> num;
    int arr[n];
    for(int i=0; i<n; i++)
    {
    	cin>>arr[i];
    	num[arr[i]]++;
    }
    prime();
    int max = INT_MIN;
    for(int i=0; i<n; i++)
    {
    	if(arr1[arr[i]]==0)
    	{
    		if(arr[i]>max)
    		max = arr[i];
    	}
    }
    cout<<max<<"\n";
    cout<<num[max]<<"\n";
    return 0;
}
