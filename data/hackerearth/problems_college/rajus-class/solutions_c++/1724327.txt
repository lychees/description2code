using namespace std;

struct node {
	ll val;
	string str;
};

bool cmp(struct node a, struct node b)
{
	if (a.val == b.val) {
		return a.str < b.str;
	}
	return a.val < b.val;
}

int main()
{
	ll t;
	
	cin >> t;
	
	while (t--) {
		ll n;
		cin >> n;
		struct node arr[10005];
		
		for (ll i = 0; i < n; i++) {
			cin >> arr[i].val;
			cin >> arr[i].str;	
		}
		
		sort(arr, arr + n, cmp);
		
		ll q;
		cin >> q;
		q--;
		cout << arr[q].str << endl;
		
	}
	return 0;
}
