#include<bits/stdc++.h>
using namespace std;
int check(string str)
{
    int C[26]={0},flag=1,i;
    for(int j=0;j<str.length();j++)
    {
        C[str.at(j)-97]++;
    }

    for(i=0;i<26;i++)
    {
        if(C[i]%2!=0)
        {
            flag=0;
            break;
        }
    }
    return flag;
}
int main()
{
    int T,i,j;
    string S,str;
    cin>>T;
    while(T--)
    {
        cin>>S;
        int flag=0;
        flag=check(S);
        if(flag==1)
            cout<<"1\n";
        else
            cout<<"-1\n";
    }
}
