#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a=0,b=0;
		for(int i=n/3;i>=0;i--)
		{
			for(int j=0;j<=n/5;j++)
			{
				if(i*3+j*5==n)
				{
					a=i*3;
					b=j*5;
					break;
				}
				else if (i*3+j*5>n)
				{
					j=n/5;
				}
				
			}
		}
		if(a==0&&b==0)
		{
			cout<<-1<<endl;
		}
		else 
		{
			while(a--)
			{
				cout<<5;
			}
			while(b--)
			{
				cout<<3;
			}
			cout<<endl;
		}
	}
    
    return 0;
}
