#include<bits/stdc++.h>
using namespace std;
long int mod = 1e9+7;
int arr[501];
int memo[501][101];
long long dp(int n)
{
	int i,j,k;
	memset(memo,0,sizeof(0));
	for(i=0;i<n;i++)
		memo[i][arr[i]] = 1;
	for(i=0;i<n;i++)
	{
		for(j=0;j<i;j++)
		{
			if(arr[i]>arr[j])
			{
				for(k=1;k<=100;k++)
				{
					int gcd = __gcd(arr[i],k);
					memo[i][gcd] = (memo[i][gcd]%mod + memo[j][k]%mod)%mod;
				}
			}
		}
	}
	long long res = 0;
	for(i=0;i<n;i++)
		res = (res%mod + memo[i][1]%mod)%mod;
	return res;
}
int main()
{
	int i,n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>arr[i];
	}
	cout<<dp(n);
}
