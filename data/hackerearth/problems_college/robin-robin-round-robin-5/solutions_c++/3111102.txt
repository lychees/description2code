#include <bits/stdc++.h>
using namespace std;

const int G = 100005;
int N, C, A[G], Q, first[G], order[G];
struct p { int r, s; } rep[G];
bool cmp (p a, p b) { return a.r == b.r ? a.s < b.s : a.r < b.r; }

int main () {
	scanf("%d%d", &N, &C);
	for (int i = 1; i <= N; i++) {
		scanf("%d", &A[i]);
		rep[i].r = (A[i]-1)/C;
		rep[i].s = i;
	}
	sort(rep+1, rep+N+1, cmp);
	int t = 0;
	for (int i = 1; i <= N; i++) {
		first[i] = t;
		if (A[i] < C) t += A[i];
		else t += C;
		order[rep[i].s] = i;
	}
	scanf("%d", &Q);
	for (int i = 0; i < Q; i++) {
		int M, D;
		scanf("%d%d", &M, &D);
		if (M == 1) printf("%d\n", first[D]);
		if (M == 2) printf("%d\n", order[D]);
	}
	return 0;
}
