#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
int main()
{
    long long int a,b,c,t,i,j,k;
    double d;
    slld(t);
    while(t--)
    {
        slld(a);
        slld(b);
        slld(c);
        j=1;
        d=c;
        while(d>b)
        {
            if(j==1)
            {
                d/=b;
                j=1-j;
            }
            else
            {
                d/=a;
                j=1-j;
            }
        }
        if(d>a)
            printf("Adam\n");
        else
        {
            if(j==1)
                printf("Adam\n");
            else
                printf("Bob\n");
        }
    }
    return 0;
}
