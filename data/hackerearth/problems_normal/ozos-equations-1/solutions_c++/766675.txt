#include <iostream>
#include <cmath>
using namespace std;

int msum (long long n) {
  int ans = 0;
  while (n > 0) {
    ans += n%10;
    n /= 10;
  }
  return ans;
}

int main(int argc, char *argv[])
{
  int t, i;
  long long n, s, tmp;
  cin>>t;
  while (t--) {
    cin>>n;
    s = sqrt(n);
    for (i = 82; i > -1 ; i--) {
      tmp = s - i;
      if (tmp < 1) continue;
      if ((n % tmp) == 0) {
	if (tmp + msum(tmp) == (n/tmp)) {
	  std::cout << tmp << std::endl;
	  break;
	}
      }
    }
    if (i < 0) {
      std::cout << -1 << std::endl;
    }
  }
  return 0;
}
