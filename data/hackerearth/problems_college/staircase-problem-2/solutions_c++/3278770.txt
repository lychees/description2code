#include <iostream>
using namespace std;

int main()
{
    int t;
    long  n;
    
    cin>>t;
    
    while(t--){
    	cin>>n;
    	
    	long result=0;
    	
    	if(n==1 || n==2)
    	{	cout<<"1"<<endl;continue;}
    	
    	int value=2;
    	result=2;
    	int step=1;
    	
    	while(result<n){
    		value++;
    		result=result+value;
    		
    		step++;
    	}
    	
    	cout<<step<<endl;
    	
    	
    }
    return 0;
}
