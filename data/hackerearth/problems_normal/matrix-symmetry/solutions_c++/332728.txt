#include<cstdio>
#include<string.h>
#include<iostream>
using namespace std;

int main(){
	int t,n,i,j;
	char a[60][60];
	scanf("%d",&t);
	while(t--){
		int hori=0;
		int ver=0;
		scanf("%d",&n);
		for (i=1;i<=n;i++){
			for(int j=1; j<=n; j++){
				cin >> a[i][j];
				}
				}	
		
		//horizontal
		for(i=1;i<=n/2;i++){
			for(j=1;j<=n;j++){
				if(int (a[i][j]) != int(a[n+1-i][j])){
					hori++;
				}	
			}
			if(hori>=1){
					break;
				}
		}
		
		
		//vertical
		for(i=1;i<=n/2;i++){
			for(j=1;j<=n;j++){
				if(int (a[j][i]) != int (a[j][n+1-i])){
					ver++;
				}	
			}
			if(ver>=1){
					break;
				}
		}
		
		if(hori>=1 && ver>=1){
			printf("NO\n");
		}
		
		if(hori==0 && ver==0){
			printf("BOTH\n");
		}
		
		if(hori>=1 && ver==0){
			printf("VERTICAL\n");
		}
		
		if(hori==0 && ver>=1){
			printf("HORIZONTAL\n");
		}
	}
	return 0;
}
