#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int n,k;
    scanf("%lld%lld",&n,&k);
    long long int temp,temp1,temp2,ans=1e9;
    for(int i=1;i<=2*k;i++)
    {
    	temp=n%i;
    	temp1=abs(i-k);
    	temp2=i-temp;
    	
    	temp=min(temp,temp2);
    	ans=min(ans,temp+temp1);
    }
    printf("%lld",ans);
    return 0;
}
