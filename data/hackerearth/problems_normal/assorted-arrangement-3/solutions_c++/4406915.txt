using namespace std;
int n, m, c[55], x;
ll mx = 0, tmp;
vector<int> v[55];

int gcd(int a, int b)
{
	if(b == 0)
		return a;

	return gcd(b, a % b);
}

bool check(int id, ll x)
{
	for(int i : v[id])
		if(x % i == 0)
			return 0;
	return 1;
}

	int g;

	ind(n);

	for(int i = 1; i <= m; i++)
		ind(c[i]);

	for(int i = 1; i <= m; i++)
	{
		for(int j = i + 1; j <= m; j++)
		{
			g = gcd(c[j], c[i]);
			v[i].push_back(c[j] / g);
		}
	}
	for(int i = 0; i < n; i++)
	{
		ind(x);
		g = mx / c[x];
		g++;
		while(!check(x, g))
			g++;
		mx = g * c[x];
	}
	pr(mx);
