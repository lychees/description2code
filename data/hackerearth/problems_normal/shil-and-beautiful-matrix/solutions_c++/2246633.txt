#include<bits/stdc++.h>
#define MOD 1000000007
using namespace std;
long long int fact(long long int n)
{
	long long int ans=1;
	for(int i=n;i>0;i--)
	{
		ans*=i;
	}
	return ans;
}
int main() 
{
	int n,m,k=0;
	long long int tmp[100005];
	cin>>n>>m;
	for(int i=0;i<100005;i++)
	{
		tmp[i]=0;
	}
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			tmp[(i^j)]++;
		}
		//cout<<endl;
	}
	long long int cnt=1;
	for(int i=0;i<100005;i++)
	{
		while(tmp[i]>0)
		{
			cnt=((cnt%MOD) * (tmp[i]%MOD) )%MOD;
			tmp[i]--;
		}
	}
	cout<<cnt<<endl;
	return 0;
}
