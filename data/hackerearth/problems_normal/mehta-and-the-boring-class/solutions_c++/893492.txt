#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MAX 1000000
#define max 1000000

ll dp[max+2];
void pre()
{
    dp[0] = dp[1] = 0;
    for ( int i = 2; i <= 1000000; i++ ) {
        if ( dp[i] > 0 ) continue;
        for ( int j = i; j <= 1000000; j += i ) {
            int k = j;
            while ( k%i == 0 )
            {
                dp[j]++,
                k /= i;
             }
        }
    }
    for ( int i = 2; i <= 1000000; i++ ) dp[i] += dp[i-1];
    return;
}
int main()
{
    ll t,a,b;
    //prime();
    //fun();
    pre();
    scanf("%lld",&t);
    //cin>>t;
    while(t)
    {
        scanf("%lld",&a);
        //cin>>a;
        //cin>>b;
        scanf("%lld",&b);
        printf("%lld",dp[b]-dp[a-1]);
        printf("\n");
        t--;
    }
    return 0;
}
