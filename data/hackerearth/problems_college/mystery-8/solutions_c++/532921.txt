using namespace std;
bool isvalid(char c)
{
	if(c >='a' && c<='z')
		return true;
	if(c >='A' && c<='Z')
		return true;
	if(c=='*')
		return true;
	return false;
}
int main()
{
//	freopen("in.txt","r",stdin);
	int T;
	cin>>T;
	getchar();
	while(T--)
	{
		char s[1000];
		char str[1000];
		gets(s);
		int positionOfMissing = -1;
		int len = 0;
		for(int i=0;i<strlen(s);i++)
		{
			if(isvalid(s[i]))
			{
				str[len++]=s[i];
				if(s[i]=='*')
					positionOfMissing=len;

			}
		}
		int pos = len - positionOfMissing;
		cout<<str[pos]<<endl;

	}
	return 0;
}
