#include<bits/stdc++.h>
using namespace std;
int solution(int n)
{
	int sum=0;
	while(n)
	{
		int r=n%10;
		sum+=r*r;
		n=n/10;
	}
return sum;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,m,i=1;
    	scanf("%d%d",&n,&m);
    	int mark[1000]={0};
    	n=solution(n);
    	m--;
    	int indx[1000];
    	vector<int> arr;
    	arr.push_back(n);
    	mark[n]=1;
    	indx[n]=i++;
    	while(m--)
    	{
    		n=solution(n);
    		if(!mark[n])
    		{
    			arr.push_back(n);
    			mark[n]=1;
    			indx[n]=i++;
    		}
    		else break;
    	}
    	if(m>0)
    	{
    		int d=indx[n]-1;
    		int k=arr.size()-d;
    		m=m%k;
    		n=arr[d+m];
    	}
    	printf("%d\n",n);
    }
    return 0;
}
