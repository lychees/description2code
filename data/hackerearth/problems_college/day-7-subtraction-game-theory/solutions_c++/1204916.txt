#include<bits/stdc++.h>
#define MOD 1000000007
using namespace std;
int main()
{
	int x,y,q,n;
	cin>>x>>y;
		//pre-computation
    long long a[1000002];
   int i;
   a[0]=1;
   for(i=1;i<=1000000;i++)
   a[i]=(a[i-1]*x-y+MOD)%MOD;
   //answering queries
	scanf("%d",&q);
	for(;q>0;q--)
	{ scanf("%d",&n);
	 printf("%lld\n",a[n]);
		
	}
	return 0;
}
