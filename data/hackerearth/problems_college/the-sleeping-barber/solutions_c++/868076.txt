#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
typedef long long int lld;
int main()
{
    int t;
    cin>>t;
    
    lld n , count , sum , i , m;
    while(t--){
    	cin>>n>>m;
    	vector<int> g(n);
    	for(int i=0;i<n;++i){
    		cin>>g[i];
    	}
    	sort(g.begin() , g.end());
    	
    	count=0;
    	sum=0;
    	i=0;
    	while(sum+g[i]<=m){
    		sum += g[i++];
    		++count;
    	}
    	
    	cout<<count<<endl;
    }
    return 0;
}
