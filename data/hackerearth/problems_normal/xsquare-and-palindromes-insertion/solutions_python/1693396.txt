def answer(mystr):
	arr = [0 for x in range(1,27)]
	for c in mystr:
		arr[ord(c)-ord('a')] += 1
	numodd = 0
	for i in arr:
		if i%2==1:
			numodd +=1
	
	if numodd<=1:
		return 0

	return numodd-1

n = int(raw_input())
for _ in xrange(0,n):
	str1 = raw_input()
	print answer(str1)
