#include<bits/stdc++.h>
using namespace std;
#define ll	long long int
int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	assert(t>=1 && t<=100000);
	while(t--)
	{
		ll n,q;
		cin>>n>>q;
		assert(n>=3 && n<=1000000);
		assert(q>=3 && q<=n);
		q--;
		ll ans=n*n-(n%q)*((n+q-1)/q)*((n+q-1)/q);
		ans=ans-(q-n%q)*(n/q)*(n/q);
		ans/=2;
		//cout<<ans<<" ";
		ans=(n*(n-1))/2-ans;
		assert(ans>0);
		cout<<ans<<endl;
	}
	return 0;
}
