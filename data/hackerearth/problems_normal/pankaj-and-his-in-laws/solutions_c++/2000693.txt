#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
int a[n];
for(int i=0;i<n;i++){
	cin>>a[i];
}
int b[n];
for(int i=0;i<n;i++){
	b[i]=1;
}
for(int i=1;i<n;i++)
{
	for(int j=0;j<i;j++){
		if(a[i]>a[j]){
			b[i]=max(b[i],b[j]+1);
		}
	}
}
int max1=b[0];
for(int i=1;i<n;i++){
	if(max1<b[i]){
		max1=b[i];
	}
}
int n1=max1;
stack<int>s;
//cout<<n1<<" ";
while(n1){
s.push(n1%2);
n1/=2;
} 
while(!s.empty()){
	cout<<s.top();
	s.pop();
}
return 0;
}
