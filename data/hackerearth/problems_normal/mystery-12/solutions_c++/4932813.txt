ll modcheck(ll n,ll mod)
{
	while(n<0)
		n+=mod;
	if(n>=mod)
		n%=mod;
	return n;
}

ll fast_exp(ll a, ll b,ll mod) //O(log(b))
{
    ll x=1,y=a;
    while(b)
    {
        if(b&1)
            x=(x*y)%mod;
        y = (y*y)%mod;
        b>>=1;
    }
    return x;
}

ll modinverse(ll N,ll mod)  //O(log(mod))
 {
     return fast_exp(N,mod-2,mod);
 }

}
