#include <bits/stdc++.h>
using namespace std;

long long int sign(long long int a)
{
	if(a>0)
		return 1;
	return 0;
}

int main()
{
    long long int n;
    cin>>n;
    long long int a[n];
    for(long long int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    long long int max[n];
    max[n-1]=1;
    for(long long int k=n-2;k>=0;k--)
    {
    	long long int s=sign(a[k]),m=0;
    	for(long long int i=k+1;i<n;i++)
    	{
    		if(abs(a[k])<abs(a[i]) && sign(a[i])!=s)
    		{
    			if(m<max[i])
    				m=max[i];
    		}
    	}
    	max[k]=1+m;
    	//cout<<k<<" : "<<max[k]<<endl;
    }
    sort(max,max+n);
    cout<<max[n-1]<<endl;
    return 0;
}
