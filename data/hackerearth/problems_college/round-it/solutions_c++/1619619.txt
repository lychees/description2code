#include<iostream>
using namespace std;

class Rounder                                
{
		public:
			int round(int n,int b);
};

int Rounder::round(int n,int b) 
		{
			int temp = n%b;
			if(temp >= (float)b/2)
			{
				n = n + (b - temp);		                                
           	}
			else
			{
				n = n - temp;
			}

		   return n;                    
		}

int main()
{
	  	int t;
	  	int a,b;
	  	Rounder rounder;
	  	cin>>t;
	  	while(t--){
	  		cin>>a>>b;
	  		cout<<rounder.round(a,b)<<endl; 	  		
	  	}	   

	return 0;
}
