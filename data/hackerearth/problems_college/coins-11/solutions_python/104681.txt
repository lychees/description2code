lim=100010

primes =[1]*lim

primes[1]=0;

i=2;
while(i<lim):
    if(primes[i]==1):
        j=i
        while((i*j)<lim):
            primes[i*j]=0
            j+=1

    i+=1
        

t=int( raw_input() )

while(t!=0):
    t-=1;

    n=int( raw_input())

    if(primes[n]==0):
        print '1'
    elif(n==3):
        print '3'
    else:
        print '2'
