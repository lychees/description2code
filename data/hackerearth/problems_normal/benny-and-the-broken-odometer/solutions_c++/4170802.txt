#define pb push_back

void solve(int n) {
	vector<int> v;
	while (n > 0) {
		int c = (n % 10);
		if (c > 3) {
			--c;
		}
		v.pb(c);
		n /= 10;
	}
	long long st = 1LL;
	long long ret = 0LL;
	for (int i = 0; i < (int)v.size(); i++) {
		ret = ret + 1LL * v[i] * st;
		st = st * 9LL;
	}
	cout << ret << endl;
}
int main() {
	int tc;
	scanf("%d", &tc);
	while (tc--) {
		int n;
		scanf("%d", &n);
		solve(n);
	}
	return 0;
}
