#include<iostream>
#include<math.h>
#include<iomanip>
#define pi 2*acos(0.0)
using namespace std;
int main()
{
	int t;
	double v,r,h,s;
	cin>>t;
	while(t--)
	{
		cin>>v;
		r=cbrt((2*v)/(4*pi));
		h=v/(pi*r*r);
		s=2*pi*r*(r+h);
		cout.setf(ios::fixed, ios::floatfield);
		cout.precision(9);
		cout<<s<<endl;
	}
	return 0;
}
