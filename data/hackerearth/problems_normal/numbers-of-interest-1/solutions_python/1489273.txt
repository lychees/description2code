def findmagicnum(invar):
	listvar=invar.split(" ")
	finlist=[]
	x=int(listvar[0])
	y=int(listvar[1])
	n=int(listvar[2])
	for i in range(x):
		finlist.append(y)
	for i in range(len(finlist),n):
		varsum=sum(finlist[i-x:i])
		finlist.append(varsum)
	return finlist[len(finlist)-1]
