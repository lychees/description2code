#include <iostream>
using namespace std;

int main()
{
    int T;
    cin >> T;
    while(T--){
    	int n;
    	cin >> n;
    	long long ans = 0;
    	for(int i=1;i*i<=n;i++){
    		int l = n/(i+1)+1;
    		int r = n/i;
    		if(i < l)
    			ans += 1LL*i*(r-l+1)*(r+l)/2;
			ans += n/i*i;
			ans %= 1000000007;
    	}
    	cout << ans<<endl;
    }
    return 0;
}
