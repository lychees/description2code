#include<bits/stdc++.h>
using namespace std;
#define ll long long 

int main()
{
    std::ios::sync_with_stdio(false);
    int n; cin>>n;
    int arr[n];
    for(int i=0; i<n; i++)
        cin>>arr[i];
    
    sort(arr, arr+n);    
    int i=0, cur=n;
    while(i<n)
    {
        int temp=arr[i], count=1; i++;
        while(i<n && temp==arr[i]) count++, i++;
        cout<<cur<<"\n";
        cur-=count;
    }
    
    return 0;    
}
