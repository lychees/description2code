#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    char vowel[10]={'a','e','i','o','u','A','E','I','O','U'};
    while(t--){
    	char s[10000];
    	cin>>s;
    	int n=strlen(s);int g=0;
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<10;j++) if(s[i]==vowel[j]) {cout<<vowel[j];g++;}
    	}
    	if(g==0) cout<<"No";
    	cout<<endl;
    }
    return 0;
}
