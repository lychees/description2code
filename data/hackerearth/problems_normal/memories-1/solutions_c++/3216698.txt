#include <iostream>
#include<vector>
#include<bits/stdc++.h>
using namespace std;

int main()
{
     int t;
     cin>>t;
     while(t--)
     {
     	int n,p;
     	cin>>n>>p;
     	int a,b,j,k,cnt=0,count=0;
     	cnt=pow(2,n);
     	vector<int> v(cnt,0);
     		for(j=1; j<=p; j++)
     		{
			 cin>>a>>b;
			for(k=1; k<cnt; k++)
			{
			  if( (k&(1<<(a-1))) && (k&(1<<(b-1)))){
		         v[k] = v[k]+1;
					}
					else{
						if(v[k]==0)
							count++;
					}
				}
			}
			count=0;
			for(k=1; k<cnt; k++)
				if(v[k]==0) count++;
        cout<<count<<endl;
     }
    return 0;
}
