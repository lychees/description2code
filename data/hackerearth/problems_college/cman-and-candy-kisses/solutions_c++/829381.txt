#include <bits/stdc++.h>

using namespace std;

long long MOD = 1000000007;
long long fact[400001];

void pre(){
	fact[0] = 1;
	for(int i = 1; i<400001; i++){
		fact[i] = (fact[i-1]*i)%MOD;
	}
}

long long powMod(long long a, long long b){
	if(b==0) return 1;
	if(b==1) return a%MOD;
	if(b%2) return (a*powMod(a*a%MOD, b/2))%MOD;
	else return powMod(a*a%MOD, b/2);
}

long long nCr(long long a, long long b){
	return (((fact[a]*powMod(fact[b],MOD-2))%MOD)*powMod(fact[a-b],MOD-2))%MOD;
}

void solve(){
	long long n,r,x;
	cin>>n>>r;
	long long sum = 0;
	for(int i = 0; i<n; i++){
		cin>>x;
		sum += x;
	}
	cout<<nCr(r-sum+n-1, r-sum)<<endl;
}

int main(){
	pre();
	int t;
	cin>>t;
	while(t--) solve();
	return 0;
}
