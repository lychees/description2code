#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
    string W, R;
    int t;
    cin >> t;
    while(t--){
    	cin >> W >> R;
    	sort(R.begin(), R.end());
    	auto itW = W.begin();
    	auto itR = R.begin();
    	while(itW != W.end()){
    		if(itR < R.end() && *itW > * itR){
    			cout << *itR;
    			itR++;
    		}
    		else{
    			cout << *itW;
    		}
    		itW++;
    	}
    	cout << endl;
    }
    return 0;
}
