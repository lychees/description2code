t = input()
lookup = []
lookup2 = []
for i in xrange(1, 90):
    for j in xrange(i, 90):
        s = (i*i*i)+(j*j*j)
        if s not in lookup:
            lookup.append(s)
        else:
            lookup2.append(s)
while t > 0:
    t -= 1
    num = input()
    got = False
    num -= 1
    while num > 0:
        if num in lookup2:
            got = True
            print num
            break
        num -= 1
    if not got:
        print "-1"
