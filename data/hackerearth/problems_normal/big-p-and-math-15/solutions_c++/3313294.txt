#include <iostream>
using namespace std;
string s1,s2;
int main()
{
    cin>>s1>>s2;
    int i,n;
    int min1=0,min2=0,max1=0,max2=0;
    n=s1.size();
    for(i=0;i<n;i++)
    {
    	if(s1[i]=='5'||s1[i]=='6')
    	{
    		min1=min1*10+5;
    		max1=max1*10+6;
    	}
    	else
    	{
    		min1=min1*10+s1[i]-'0';
    		max1=max1*10+s1[i]-'0';
    	}
    }
    n=s2.size();
    for(i=0;i<n;i++)
    {
    	if(s2[i]=='5'||s2[i]=='6')
    	{
    		min2=min2*10+5;
    		max2=max2*10+6;
    	}
    	else
    	{
    		min2=min2*10+s2[i]-'0';
    		max2=max2*10+s2[i]-'0';
    	}
    }
    cout<<min1+min2<<' '<<max1+max2<<endl;
    return 0;
}
