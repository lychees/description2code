#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	int n;
    	cin>>n;
    	long ctr=0;
    	int min=1000;
    	for(int i=0;i<n;i++)
    	{
    		int m;
    		cin>>m;
    		ctr=ctr+m;
    		if(m<min)
    			min=m;
    	}
    	ctr=ctr-n;
    	min=min-1;
    	cout<<min<<" "<<ctr<<endl;
    }
    return 0;
}
