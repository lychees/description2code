#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int a,b,s=0,s1=0,s2,s3=0;
        cin>>a>>b;
        while(a)
        {
            s=s*10+(a%10);
            a/=10;
        }
        while(b)
        {
            s1=s1*10+(b%10);
            b/=10;
        }
        s2=s+s1;
        while(s2)
        {
            s3=s3*10+(s2%10);
            s2/=10;
        }
        cout<<s3<<"\n";
    }
    return 0;
}
