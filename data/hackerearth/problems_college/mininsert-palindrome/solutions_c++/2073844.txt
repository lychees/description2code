#include <iostream>
#include <algorithm>
using namespace std;
string s;
int n;
int solve(int f, int l)
{
	int a=100, b=100;
	if(f==l)
	return 0;
	if(f+1==l&&s[f]==s[l])
	return 0;
	if(f+1==l-1&&s[f]==s[l])
	return 0;
	if(s[f]==s[l])
		return solve(f+1, l-1);
	else
	{
		if(f+1<n)
		a=solve(f+1, l);
		if(l-1>=0)
		b=solve(f, l-1);
		return min(a,b)+1;
	}
}
int main()
{
    int t, ans;
    scanf("%d", &t);
    while(t--)
    {
    	scanf("%d", &n);
    	cin>>s;
    	ans=solve(0, n-1);
    	printf("%d\n", ans);
    }
    return 0;
}
