        map<int,int>mymap;

        for(int u = 1 ; u<=j ; u++)

            for(int v = u+1;v <=j ;v++)

                int val = u*u*u + v*v*v;

                if(val<n && mymap[val] >=2)
                    result  = max(result,val);
            }
        }
