#include <iostream>
using namespace std;

int arr[1000001], ans[100000], k;

void set()
{
	int i, j;
	for (i = 1; i*i*i <= 1000000; i++) {
		for (j = i; ((i*i*i) + (j*j*j)) <= 1000000; j++) {
			arr[(i*i*i) + (j*j*j)]++;
		}
	}

	for (i = 0; i < 1000001; i++) {
		if (arr[i] > 1) ans[k++] = i;
	}
}

int main()
{
	set();
	int t, n, start, mid, end, index;
	cin >> t;

	for (; t; t--) {
		cin >> n;
		index = -1; start = 0; end = k-1; 
		while (start <= end) {
			mid = start + (end - start) / 2;
			if (ans[mid] == n) {
				index = mid;
				break;
			} else if (ans[mid] > n) {
				end = mid - 1;
			} else {
				index = mid;
				start = mid + 1;
			}
		}

		if (index == -1) cout << -1 << endl;
		else cout << ans[index] << endl;
	}

	return 0;
}
