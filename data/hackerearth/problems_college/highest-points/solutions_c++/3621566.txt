#include <iostream>
using namespace std;

int main()
{
    int n,i;
    cin>>n;
    int w[n],l[n],t[n],a,b,c,p[i];
    for(i=0;i<n;++i)
    {cin>>a>>b>>c;
    w[i]=a;
    t[i]=b;
    l[i]=c;
    p[i]=w[i]*3+t[i]*1-l[i];
    
}
int large =p[0];
for(i=0;i<n;++i)
{if(large<p[i])
large=p[i];
}
cout<<large;
}
