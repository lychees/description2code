#include <iostream>
#include <string.h>
#include <math.h>
using namespace std;

int main()
{
    int t,l,i,j;
    char s[201];
    cin>>t;
    gets(s);
    while(t--){
    	gets(s);
    	l=strlen(s);
    	for(i=0;i<sqrt(l);i++){
    		for(j=i;j<l;j=j+sqrt(l)){
    			cout<<s[j];
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
