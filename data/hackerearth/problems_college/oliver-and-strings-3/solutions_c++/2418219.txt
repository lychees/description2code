#include <bits/stdc++.h>

using namespace std;

int score(char ch) {
	if (ch == '$')	return 100;
	return int(tolower(ch)) - 96;
}

int main()
{
	ios_base::sync_with_stdio(0);
	long long t;
	cin >> t;
	while (t--) {
		string s;
		cin >> s;
		long long n = s.length();
		long long inside_score = 0;
		for (long long i=1;i<n-1;i++)	inside_score += score(s[i]);
		long long total = 100, index = 1;
		while (s[index] != '$') {
			total += inside_score + 100;
			char note = s[index];
			inside_score -= score(note);
			if (note == s[n-index-1]) {
				total += inside_score + 100;
				inside_score -= score(s[n-index-1]);
				s[n-index-1] = '$';
			} else {
				total += score(s[n-index-1]);
				break;
			}
			index += 1;
		}
		cout << total << endl;
	}
	return 0;
}
