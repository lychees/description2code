#include<bits/stdc++.h>
using namespace std;
long long int f[1000006];
long long int g[1000006];

int main()
{

memset(f,-1,sizeof(f));

memset(g,0,sizeof(g));
f[0] = f[1] = g[0] = g[1] = 0;
for(int i=2;i<1000006;i++)
{ for ( int j = i;j < 1000006; j+=i )
{ if ( f[j] == -1 )
f[j] = i;
g[j] += i;
}
g[i] -= i;
g[i] += 1;
}
for(int j=1;j<1000006;j++)
{
f[j] = f[j-1]+f[j];
g[j] = g[j-1]+g[j];
}
int t;
cin>>t;
while ( t-- )
{ int n;
cin>>n;
cout<<(f[n]+g[n])%n <<endl;
}
return 0;
}
