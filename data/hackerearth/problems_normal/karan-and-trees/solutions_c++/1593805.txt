#include <iostream>
#include<cstring>
using namespace std;
 
int main()
{
   int t;
   
   
   cin >> t;
   while(t--){
   	
   	int x;
   	int y;
   	
   	cin >> x >> y;
   	while(x != y)
   	{
   		if((x - y) >= 1)
   		x = x/2;
   		else if((y - x) >= 1)
   		y = y/2;
   	}
   	cout << x << endl;;
   }
    return 0;
}
