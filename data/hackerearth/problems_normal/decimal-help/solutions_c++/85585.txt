#include <iostream>
#include <cmath>
using namespace std;

int findxmody(long int x, int power, long int y)
{
long int modulus=1;
for(int i=1;i<=power;i++)
{
	modulus=(x*modulus)%y;
}
return modulus;
}

int main()
{
	int T;
	long int N, D, R;
	cin>>T;
	while(T--)
	{
		cin>>N>>D>>R;
		long int mod=findxmody(10,R-1,D);
		cout<<((int)(floor(10*mod*N/D)))%10<<endl;;
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
