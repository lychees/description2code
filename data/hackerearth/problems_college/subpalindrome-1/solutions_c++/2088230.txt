#include <bits/stdc++.h>
using namespace std;
long long  dp[101][101];
int main()
{
    //cout << "Hello World!" << endl;
    long long t;
    cin>>t;
    while(t--)
    {
    	long long i,j;
    	string s;
    	cin>>s;
    	long long l=s.size();
		for(i=0;i<l;i++)
		{
			for(j=0;j<l;j++)
			dp[i][j]=0;
		}
		for(int i=l-1;i>=0;i--)
		{
			int c=1;
			dp[i][i]=1;
			if(i+1!=l)
			{
				if(s[i]==s[i+1])
				{
					c++;
					dp[i][i+1]=3;
				}
				else
					dp[i][i+1]=2;
			}
			for(int j=i+2;j<l;j++)
			{
				if(s[i]==s[j])
				{
					c++;
					c+=dp[i+1][j-1];
				}
				dp[i][j]=dp[i+1][j]+c;
			}
		}
		cout << dp[0][l-1] << endl;
	}
	return 0;
}
