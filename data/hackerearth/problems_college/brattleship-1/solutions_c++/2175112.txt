#include <iostream>

using namespace std;

int r,c,w;

void solve()
{
	int ans=0;
	if(c==w)
		ans=r+w-1;
	else if(r==1)
		ans=w-1+c/w;
	else
		ans=r*(c/w)+w-1;
	cout<<ans<<"\n";
}
	
int main()
{
    int tc,k=1;
    cin>>tc;
    while(tc--)
    {
    	cin>>r>>c>>w;
    	cout<<"Case #"<<k++<<": ";
    	solve();
    }
    return 0;
}
