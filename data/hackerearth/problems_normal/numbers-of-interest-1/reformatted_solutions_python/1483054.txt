for _ in xrange(input()):
    x, y, n = map(int, raw_input().split())
    if x >= n:
        print y
    else:
        series = [y]*x + [x*y]
        n -= x
        n -= 1
        i = 0
        while n > 0:
            series.append(series[-1]*2-series[i])
            i += 1
            n -= 1
        print series[-1]
