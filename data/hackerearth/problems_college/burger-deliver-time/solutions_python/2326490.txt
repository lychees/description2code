orderNumber = raw_input()
order = {}
for i in range(int(orderNumber)):
	temp = raw_input()
	order[i+1] = int(temp.split(' ')[0])+int(temp.split(' ')[1])
orderSorted = sorted(order.items(), key=lambda x: (x[1],x[0]))
orderSortedKey = [x[0] for x in orderSorted]
print ' '.join(str(x) for x in orderSortedKey)
