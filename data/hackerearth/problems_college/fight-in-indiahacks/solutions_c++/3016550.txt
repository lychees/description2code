#include <iostream>
using namespace std;
unsigned long long int m1,m2;

bool win(unsigned long long int n,bool move)
{
	//if((m1>n && m2>n)||(m1==0 && m2>n)) return (!move);
	if(m1>n && m2>n) return (!move);
		
		//if((m1<=n)&&(m1!=0))
		if(m1<=n)
		{
			if(m2<=n)
			{
				if(move) return ( (win(n-m1,!move)) || (win(n-m2,!move)) );
				else
				return ( (win(n-m1,!move)) && (win(n-m2,!move)) );
			}
			else
				return (win(n-m1,!move));
		}
		else
			return (win(n-m2,!move));
}

int main()
{
	int t;
	unsigned long long int n;
	cin>>t;
	while(t--)

		cin>>n>>m1>>m2;
		if((n==0)||(m1==0)||(m2==0))
			cout<<"1\n";
		else
		{
			//if(m1!=0 && m2!=0) n=n%(m1+m2);
			n=n%(m1+m2);
			//if((m1==0 && m2==0)||(n==0))
			//cout<<"0\n";
			//else
			//{
				/*if(m2==0)
				{

				}*/
				if(win(n,true))
				cout<<"1\n";
				else
				cout<<"0\n";
			//}
		}
	}
	return 0;
}
