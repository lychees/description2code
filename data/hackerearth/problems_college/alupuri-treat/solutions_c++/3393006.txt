#include <iostream>
using namespace std;

int main()
{
    long long int n;
    cin>>n;
    long long int sum=0,sum1=0,flag=0,a=0,b=n-1;
    long int m[n][n];
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<n;j++)
    	{
    		cin>>m[i][j];
    		if(i == j)
    			sum += m[i][j];
    		if(i==a && j==b)
    		{
    			sum1 += m[i][j];
    			a++;
    			b--;
    		}
    		
    	}
    }
    sum -= sum1;
    sum=(sum<0)?(-1)*sum:sum;
    cout<<sum;
    return 0;
}
