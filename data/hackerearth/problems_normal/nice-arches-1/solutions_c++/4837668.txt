#include<bits/stdc++.h>
#include<iostream>
#include<list>

int main()
{
int t,cnt=0;
cin>>t;
while(t--)
{
list <char> mylist;
char str[100005];
int i,j,k,p;
cin>>str;
p=strlen(str);
for(i=0;i<p;i++)
{
if(str[i]=='A')
{
if(!mylist.empty())
{
if(mylist.back()=='A')
mylist.pop_back();
else
mylist.push_back('A');

}
else

}
else
{

{
if(mylist.back()=='B')
mylist.pop_back();
else
mylist.push_back('B');

else

}

}

}
