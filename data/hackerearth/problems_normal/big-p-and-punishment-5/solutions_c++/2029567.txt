#include <iostream>
#include <cstring>
#include <vector>
using namespace std;
vector<int> v[10000];
int sit[10001]={0};

void follow(int a){
	sit[a]=1;
	for(int i=0;i<v[a].size();i++){
		if(sit[v[a][i]]==0){
			follow(v[a][i]);
		}
	}
}

int main()
{
    int t,n,f,s,a,b;
    cin>>t;
    while(t--){
    	int count=0;
    	memset(sit,0,sizeof(sit));
    	cin>>n>>f>>s;
    	for(int i=0;i<f;i++){
    		cin>>a>>b;
    		v[a].push_back(b);
    	}
    	for(int i=0;i<s;i++){
    		cin>>a;
    		follow(a);
    	}
    	for(int i=1;i<=n;i++){
    		if(sit[i]==1)
    			count++;
    		v[i].clear();
    	}
    	cout<<count<<endl;
    }
    return 0;
}
