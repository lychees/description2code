#include<iostream>
#include<stdio.h>
using namespace std;
void fact(int x);
int multiply(int x,int res_size,int *a);
int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        int x;
        cin>>x;
        fact(x);
    }
    return 0;
}

void fact(int x)
{
    int arr[200];
    arr[0]=1;
    int res_size=1;
    for(int i=2;i<=x;i++)
    {
        res_size=multiply(i,res_size,arr);
    }
    for(int i=res_size-1;i>=0;i--)
    {
        cout<<arr[i];
    }
    cout<<endl;
}

int multiply(int x,int res_size,int *a)
{
    int carry=0;
    int prod=1;
    for(int i=0;i<res_size;i++)
    {
        prod=a[i]*x+carry;
        a[i]=prod%10;
        carry=prod/10;
    }
    while(carry>0)
    {
        a[res_size]=carry%10;
        carry = carry/10;
        res_size++;
    }
    return res_size;


}
