#include<iostream>
using namespace std;
#include<string.h>
#include<stdio.h>
    int main()
    {
        int t;
        scanf("%d",&t);
        while(t--)
        {

        	string s;
        	cin >> s;
        	int i, l = s.size();
        	if(l>=2 && s[0]=='S' && s[1]=='-')
        	{
        	  s[1]='A';
        	  //cout << l << endl;
        	}
        	for(i=1;i<l;i++)
        	{
        		if(i+1<l && s[i-1]=='-' && s[i]=='S' && s[i+1]=='-')
        		{
        			s[i-1]=s[i+1]='A';
        		}
        		else if(i-2>=0 &&  s[i-2]=='-' && s[i-1]=='-' && s[i]=='B')
        		{
        			s[i-1]=s[i-2]='A';
        		}
        			else if(i-2>=0 &&  s[i-2]=='-' &&  s[i]=='B')
        		{
        		            s[i-2]='A';
        		}
        		else if(i+1<l && s[i]=='S' && s[i+1]=='-')
        		{
        			s[i+1]='A';
        		}
        		else if((s[i-1]=='-' && s[i]=='S') || (s[i-1]=='-' && s[i]=='B'))
        		{
        			s[i-1]='A';
        		}
        	}

        //	cout << s << endl;
        	int count = 0;
        	for(i=0;i<l;i++)
        	{
        		if(s[i]=='-')
        		count++;
        	}

        	printf("%d\n",count);
        }
        return 0;
    }
