    #include <iostream>
    #include <vector>
    using namespace std;
     
    int main()
    {
    	int n;
    	cin >> n;
    	cout << n <<"\n"; 
    	//n = 3;
    	vector<int> num(10,-1);
    	while(n) {
    		n--;
    		for(int i = 0; i < 10; i++) {
    			int no;
    			cin >> no;
    		//	cout << no << "\n";
    			num[no] = no;
    		}
    	for(int i = 1; i < 10; i++){
    		if(num[i] == -1){
    			cout << i << endl;
    			
    			return 0;
    		}
    		
    	}
    	if(num[0] == -1){
    		cout << "10\n"; 
    		return 0;
    	}
    }
        return 0;
    }
