#include <cstdio>
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <set>
#include <map>

#define MAX 1000006
using namespace std;

int main()
{
    int t,n,sum,i,ans[MAX]={0};
    for(i=1;i<MAX;i++)
    {
    	t=i*2;
    	while(t<MAX)
    	{
    		ans[t]++;
    		t+=i;
    	}
    }
    sum=0;
    for(i=1;i<MAX;i++)
    {
    	sum+=ans[i];
    	ans[i]=sum;
    }
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	cout<<ans[n]<<endl;
    }
    return 0;
}
