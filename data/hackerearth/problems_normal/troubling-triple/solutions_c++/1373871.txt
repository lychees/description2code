#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define LL long long 
int N , K , A[2005] , DP[2005] ;

int solve(int idx,int val){
	int left , right , cnt ;
	left = idx ;
	right = N ;
	cnt = 0 ;
	while(left < right){
		while(left < right && (1LL * A[left] * A[right]) > val){
			right -- ;
		}
		if(left < right){
			cnt += (right - left) ;
		}
		left ++ ;
	}
	return cnt ; 
}
int main()
{
    cin >> N >> K ;
    for(int i=1;i<=N;i++) cin >> A[i] ;
    sort(A+1,A+1+N) ; 
    LL ans = 0 ;
    for(int i=1;i<N;i++){
		ans += solve(i+1,K/A[i]) ;    	
    }
    cout << ans << endl ;
    return 0;
}
