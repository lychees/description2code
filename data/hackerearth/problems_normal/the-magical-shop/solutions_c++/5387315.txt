#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long i,j,a,b,sum=0;
    cin>>a>>b;

    string s;
    cin>>s;

    long long p[s.size()];
    p[0]=a;

    for(i=1; i<s.size(); i++)
        p[i]=((p[i-1]%b)*(p[i-1]%b))%b;

    for(i=0; i<s.size(); i++)
    {
        if(s[i]=='1')
        {
            sum=((sum%b)+(p[i]%b))%b;
        }

    }

    cout<<sum<<"\n";


    return 0;
}
