#include <bits/stdc++.h>
using namespace std;

const int MAX = 1e5 + 5;
const int LIM = 1<<20;

int a[MAX];
int freq[LIM];
long long dp[LIM][21];

int main() {
    #ifndef ONLINE_JUDGE
        freopen("inp.txt", "r", stdin);
    #endif
    int t, n;
    scanf("%d", &t);
    while(t--) {
        scanf("%d", &n);
        for(int i=1; i<=n; ++i) {
            scanf("%d", &a[i]);
            freq[a[i]] += 1;
        }
        for(int mask=0; mask<LIM; ++mask) {
            dp[mask][0] = freq[mask];
            if (mask & 1) {
                dp[mask][0] += freq[mask ^ 1]; 
            }
            for(int i=1; i<=20; ++i) {
                dp[mask][i] = dp[mask][i-1];
                if (mask & (1<<i)) {
                    dp[mask][i] += dp[mask ^ (1<<i)][i-1];
                }
            }
        }
        long long ans = 0;
		for(int i=1; i<=n; ++i) {
            ans += dp[LIM - a[i] - 1][20];
            freq[a[i]] = 0;
        }
        printf("%lld\n", ans);
    }
}
