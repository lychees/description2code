using namespace std;

lli a[1000001];
int main()
{
    int t,n,i;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        lli maxm=-1,minm=1e9;
        minm*=minm;
        for(i=0;i<n;i++)
        {
            scanf("%lld",&a[i]);
            minm=min(minm,a[i]);
            maxm=max(maxm,a[i]);
        }
        printf("%lld\n",(maxm-minm)*2);
    }
    return 0;
}
