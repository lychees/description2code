    #include <iostream>
    #include <bits/stdc++.h>
    #define MOD 1000000007
    #define ll unsigned long long
    using namespace std;
     
    int main()
    {
        //cout << "Hello World!" << endl;
        ll t,n,ans;
        cin>>t;
        while(t--){
        	cin>>n;
        	ans=((n%MOD)*(((2*n)-1)%MOD))%MOD;
        	cout<<ans<<endl;
        }
        return 0;
    }
