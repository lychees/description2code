#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long int n,k;
    	scanf("%ld %ld",&n,&k);
    	long int a[n],b[n],i;
    	for(i=0;i<n;i++)
    	{
    		scanf("%ld",&a[i]);
    	}
    	for(i=0;i<n;i++)
    	{
    		scanf("%ld",&b[i]);
    	}
    	int max = 0;
    	for(i=0;i<n;i++)
    	{
    		long int d = k/a[i];
    		long int temp = d*b[i];
    		if(temp>max)
    		{
    			max = temp;
    		}
    	}
    	printf("%ld\n",max);
    }
    return 0;
}
