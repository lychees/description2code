noOfTestCases = int(raw_input())


def order(list1):
	countHash = {}
	for index, no in enumerate(list1):
		no = int(no)
		countOf1 = bin(no).count('1', 0, len(bin(no)))
		if(countHash.has_key(countOf1)):
			countHash[countOf1][index] = no
		else:
			countHash[countOf1] = {}
			countHash[countOf1][index] = no
	
	order = ""
	
	sortedOuterKeys = sorted(countHash.keys())
	
	for i in sortedOuterKeys:
		sortedInnerKeys = sorted(countHash[i].keys())
		for j in  sortedInnerKeys:
			order += str(countHash[i][j]) + " "
			
	return order
