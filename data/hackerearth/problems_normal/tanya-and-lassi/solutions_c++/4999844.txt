#include<bits/stdc++.h>
using namespace std;
 
typedef long long lld;
lld dp[100005];
lld dp_soln(lld price[],lld n)
{
	for(lld i=0;i<=n;i++)
	dp[i]=0;
	
	for(lld i=1;i<=n;i++)
	{   dp[i]=INT_MIN;
		for(lld j=1;j<=i;j++)
		{
			 dp[i]=max(dp[i],price[j]+dp[i-j]);
		}
		
	}
	return dp[n];
}
 
int main()
{
	lld t;
	cin>>t;
	
	while(t--)
	{
		lld n;
		cin>>n;
		lld price[100005];
		for(lld i=1;i<=n;i++)
		cin>>price[i];
		
		cout<<dp_soln(price,n)<<"\n";
		
		
	}
	
	
	
}
