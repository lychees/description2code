#include <bits/stdc++.h>
using namespace std;

int main()
{
  int t;
  t = 1;
  while(t--)
  {
      int n,m,a[100004],b[1000004];
      cin>>n>>m;
      vector < pair< int,int > > v,v1;
      for (int i = 0; i < n; ++i)
         {
           cin>>a[i]>>b[i];
           
           v.push_back(make_pair(a[i],i));
           v1.push_back(make_pair(a[i]+b[i],i));
         }  
         sort(v.begin(),v.end());
         int d[100005];
         vector < pair<int,int> > v2;
         for (int i = 0; i < m; ++i)
         {
            v2.push_back(make_pair(0,0));
         }
         for (int i = 0; i < n; ++i)
         {
            for (int j = 0; j < m; ++j)
            {
                if((v2[j].first == 0)||(v2[j].second<=v[i].first))
                {
                  d[v[i].second] = j+1;
                  v2[j].first = 1;
                  v2[j].second = v1[v[i].second].first;
                  break;
                }
                d[v[i].second] = -1;
            }

         }
         for (int i = 0; i < n; ++i)
         {
           cout<<d[i]<<" ";
         }
         cout<<endl;

      
  }
}
