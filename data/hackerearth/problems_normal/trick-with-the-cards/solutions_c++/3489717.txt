#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cmath>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    long long int num;
    cin>>t;
    while(t--)
    {
    	cin>>num;
    	if(num<4)cout<<ceil(num/3.0)<<endl;
    	else
    	{
    		if(num%3==0)
    		{
    			num = num / 3;
    			cout<<num<<endl;
    		}
    		else cout<<num<<endl;
    	}
    }
    return 0;
}
