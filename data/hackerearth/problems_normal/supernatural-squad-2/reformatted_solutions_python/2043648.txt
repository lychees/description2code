a = [[0 for x in range(213)] for x in range(213)]


def sol(n, k):
    if a[n][k] > 0:
        return a[n][k]
    if n < k:
        return 0
    if n < 2 * k:
        return 1
    res = 1
    for i in range(k, n):
        res = res + sol(n-i, i)
    a[n][k] = res
    return res


for t in range(int(raw_input())):
    n, k = map(int, raw_input().split())
    for i in range(213):
        for j in range(213):
            a[i][j] = 0
    print(sol(n, k))
