#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int d,wt,sum=0;
   	cin>>d>>wt;
   	if(d>60)
   	{
   		cout<<"NA\n";
   	}
   	else if(d<=60 & d>=2)
   	{
   		sum+= 50;
   		d-=2;
   		sum+= (d*12);
   		sum += (wt*2);
   		cout<<sum<<"\n";
   	}
   	else if(d<2)
   	{
   		sum+=50;
   		sum+= (wt*2);
   		cout<<sum<<"\n";
   	}
   }
    return 0;
}
