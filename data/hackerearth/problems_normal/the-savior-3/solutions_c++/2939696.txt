#include <iostream>

using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int c,sum=0,a[1000],x=0;
		cin>>c;
		for(int i=0;i<c;i++)	
			cin>>a[i];
		
    	for(int i=0;i<c-1;i++)
    	{
    		for(int j=i+1;j<c;j++)
    		{
    			if(a[i]!=a[j]){
    			
    				sum = a[i]+a[j];
    			if(sum%2==0) 
    				x++;    			    			
    			}
    		}
    	}
    	cout<<x<<endl;
	}
	
    return 0;
}
