    #include <bits/stdc++.h>
    using namespace std;
     
    int main()
    {
    int n;
    int a[100009];
    while(cin>>n){
    for(int i = 0; i < n; i++){
    cin>>a[i];
    }
    int cnt = 0;
    for(int i = n-1; i >= 0; i--){
    if(a[i]%2 == 0){
    cout<<a[i]<<" ";
    cnt++;
    }
    }
    if(cnt == 0){
    cout<<"None to display";
    }
    cout<<endl;
    }
    }
