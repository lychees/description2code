#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
int main()
{
    int T;
    scanf("%d", &T);
    while (T--) {
    	string s1, s2;
    	cin >> s1 >> s2;
    	reverse(s1.begin(), s1.end());
    	reverse(s2.begin(), s2.end());
    	int ans = 0, p = 0;
    	for (int i = 0; i < s1.size() || i < s2.size(); ++i) {
    		if (i < s1.size()) p += s1[i] - '0';
    		if (i < s2.size()) p += s2[i] - '0';
    		if (p > 9) ++ans;
    		p /= 10;
    	}
    	if (ans == 0) puts("No carry operation");
    	else if (ans == 1) puts("1 carry operation");
    	else printf("%d carry operations\n", ans);
    }
    return 0;
}
