#include<bits/stdc++.h>
using namespace std;
int main ()
{
int n,j,i;
vector<int> vec;
cin>>n;
for(j=1;j<=n;j++)
vec.push_back(j);
string str;
cin>>str;
i=0;
j=0;
while(vec.size()>1)
	{
	if(i==str.length())
	i=0;		
	if(j==vec.size())
	j=0;	
	if(str[i]=='a')
	j++;
	if(str[i]=='b')
	vec.erase(std::remove(vec.begin(), vec.end(), vec[j]), vec.end());
	i++;
	}
cout<<vec[0];
return 0;
}
