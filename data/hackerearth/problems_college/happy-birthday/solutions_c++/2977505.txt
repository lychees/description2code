#include <bits/stdc++.h>

using namespace std;

int main()
{
    int  s , flag = 0;
    int  i  , s1 = 0 , s2 = 0 , s3 = 0 , s4 = 0;//use in program as counter
    long int n , no = 0 ;//number of minimum taxi

    scanf("%ld",&n);

    for (i = 0 ; i < n ; i++)
    {
        scanf("%d",&s);

        if (s % 4 == 0)
            s4++;
        else if (s % 4 == 3)
            s3++;
        else if (s % 4 == 2)
            s2++;
        else
            s1++;
    }

    if (s3 >= s1 )
    {
        //do nothing
    }
    else
    {
        flag  = s1 - s3 ;
    }

    if ( s2 % 2 == 0)
    {
        no = s4 + s3 + (s2/2) + ( ceil ( (double) flag / 4 ) ) ;
    }

    else
    {
        if ( flag == 0 && flag == 1)
        {
            no = s4 + s3 + s2/2 + 1 ;
        }
        else
        {
            no = s4 + s3 + s2/2 + 1 + ( ceil ( (double) (flag-2) / 4 ) );
        }
    }
    printf("%ld",no);

    return 0;
}
