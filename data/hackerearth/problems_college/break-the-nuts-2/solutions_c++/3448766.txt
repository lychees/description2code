#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,x;
	cin >> n;
	vector<int> vec;
	for(int i=1;i<=n;i++)
	{
		cin >> x;
		vec.push_back(x);
	}
	int i=0;
	while(i<n)
	{
		if(vec[i]==1)
		{
			break;
		}
		++i;
	}
	if(i==n)
		cout << "0\n";
	else
	{
		long long int counter=0,ans = 1;
		for(int j=i+1;j<n;j++)
		{
			if(vec[j]==1)
			{
				ans *= (counter+1);
				counter=0;
			}
			else
				++counter;
		}
		cout << ans << endl;
	}

	return 0;
}
