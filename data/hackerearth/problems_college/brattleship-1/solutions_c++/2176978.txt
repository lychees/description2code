#include<bits/stdc++.h>
using namespace std;
int main()
{
	
	int test;
	cin>>test;
	int arr[21][21];
	
	arr[1][1]=1;
	for(int i=2;i<=20;i++)
	arr[1][i]=0;
	arr[2][1]=2;
	arr[2][2]=2;
	for(int i=1;i<21;i++)
	arr[i][i]=i;
	for(int i=1;i<21;i++)
	arr[i][1]=i;
	
	for(int i=3;i<21;i++)
	{
		for(int j=2;j<=i-1;j++)
		{
			if(i-j<=j)
			{
				arr[i][j]=1+j;
			}
			else
			{
				arr[i][j]=1+arr[i-j][j];
			}
		}
	}
	for(int pk=1;pk<=test;pk++)
		{
		int r,c,w;
		cin>>r>>c>>w;
		printf("Case #%d: %d\n",pk,arr[c][w]*r);
		
	}
	return 0;
}
