#include <iostream>
#include <string.h>
using namespace std;

int main()
{
	unsigned int t,v,check;
	cin>>t;
	while(t--){
			char str[100];
		    cin>>str;
		    v=0;
		    check=0;
		    for(unsigned int i=0;i<strlen(str);i++){
		    
		    	if(str[i]=='a' || str[i]=='e' || str[i]=='i' || str[i]=='o' || str[i]=='u' || str[i] == 'y'){
		    		v++;
		    		check=0;
		    	}else{
		    		check++;
		    		if(check==3)
		    		break;
		    	}
		    }
		    	//cout<<v<<" "<<check<<" "<<endl;
		    if(check==3 || strlen(str) > 2*v){
		    	cout<<"hard"<<endl;
		    }else{
		    	cout<<"easy"<<endl;
		    }
	}

    return 0;
}
