#include <iostream>
using namespace std;

int main()
{
   int t;int p=0;
   cin>>t;
   string s;
   for(int j=0;j<t;j++)
   {cin>>s;p=0;
   for(int i=0;i<s.length()/2;i++)
   {if(abs(s[i]-s[i+1])!=abs(s[s.length()-1-i]-s[s.length()-2-i]))
   {p=1;cout<<"Not Funny"<<endl;break;}
   }
   if(p==0)cout<<"Funny"<<endl;
   }
    return 0;
}
