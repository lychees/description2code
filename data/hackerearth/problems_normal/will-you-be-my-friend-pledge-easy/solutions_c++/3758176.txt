#include <iostream>
using namespace std;


bool check(int m,int f)
{
    while(m != f)
    {
        if(m > f)
            m-= f;
        else
            f -= m;
    }

       /*while(n1 != n2) {
        if(n1 > n2)
            n1 -= n2;
        else
            n2 -= n1;
    }*/

    if(m>1)
        return true;

	return false;

}

int main()
{
   int max;
   int sum=0;
   cin>>max;

   int n;
   cin>>n;

   int a[n];

   for(int i=0;i<n;i++)
   cin>>a[i];

   bool arr[n];

   for(int i=0;i<n;i++)
   {
  		arr[i] = check(max,a[i]);
   }
   for(int i=0;i<n;i++)
   {
   		if(arr[i]==true)
   		sum++;
   }

    cout<<sum;
}
