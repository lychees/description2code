#include<iostream>
#include<algorithm>
using namespace std;

int main(){
	int a[101], A[50*99];
	int s = 0;
	for(int i = 0; i <= 100; i++){
		a[i] = i;
	}
	
	for(int i = 1; i < 100; i++){
		for(int j = i + 1; j <= 100; j++){
			A[s] = a[i]*a[i]*a[i] + a[j]*a[j]*a[j];
			s++;
		}
	}
	sort(A, A + 50*99);
	int t;
	cin >> t;
	while(t--){
		int FLAG = 1;
		int n;
		cin >> n;
		for(int i = 50*99 - 1; i > 0 ; i--){
			if(A[i] == A[i - 1] && A[i] < n){
				cout << A[i] << "\n";
				FLAG = 0;
				break;
			}
		}
		if(FLAG){
			cout << "-1" << "\n";
		}
	}
}
