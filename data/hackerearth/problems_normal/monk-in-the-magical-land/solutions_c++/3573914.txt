#include <cstring>
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;
const int inf=109;
vector<int>adj[inf];
int key[inf];
int chest[inf];
int cost[inf];
bool vis[inf];
int gcd(int a,int b){return b==0?a:gcd(b,a%b);}
int main()
{
	int t,n,m,k;
	scanf("%d",&t);
	vector<int>::iterator it;
	while(t--)
	{
		scanf("%d%d%d",&n,&m,&k);
		for(int i=0;i<n;++i)
			adj[i].clear();
		for (int i=0;i<n;++i)
			scanf("%d",&key[i]);
		for (int i=0;i<m;++i)
			scanf("%d",&chest[i]);
		for (int i=0;i<m;++i)
			scanf("%d",&cost[i]);
		for (int i=0;i<n;++i)
			for (int j=0;j<m;++j)
				if (gcd(key[i],chest[j])!=1)
					adj[i].push_back(j);
		int mx=0;
		int s=1<<n;
		for (int i=0;i<s;++i)
		{
			int t=i,r=0;
			while(t)
			{
				r++;
				t-=t&-t;
			}
			if (r!=k)continue;
			int sel=0;
			t=i;
			for (int j=0;j<n;++j)
				if (t&(1<<j))
				{
					for (it=adj[j].begin();it!=adj[j].end();++it)
						if (!vis[*it])
							vis[*it]=true,sel+=cost[*it];
				}
			memset(vis,0,sizeof(vis));
			mx=max(mx,sel);
		}
		printf("%d\n",mx);
	}
}
