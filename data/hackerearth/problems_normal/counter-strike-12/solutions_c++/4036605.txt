#include <iostream>
using namespace std;

int main()
{	int t,n,m,d,nx[1000],ny[1000],mx[1000],my[1000],x,y;
	cin>>t;
	while(t--)
	{    int count=0,flag=0;
		cin>>n>>m>>d;
		int k=m/2;
		for(int i=0;i<n;++i)
			cin>>nx[i]>>ny[i];
		for(int i=0;i<m;++i)
			cin>>mx[i]>>my[i];	
		for(int i=0;i<m;++i)
		{
			for(int j=0;j<n;++j)
			{
				if(nx[j]>mx[i])
					x=nx[j]-mx[i];
				else
					x=mx[i]-nx[j];	
				if(ny[j]>my[i])
					y=ny[j]-my[i];
				else
					y=my[i]-ny[j];		
				if((x+y)<=d)
				{
					++count;
					break;
				}
				
			}
			if(count>=k)
			{	flag=1;
				break;
			}
		}
		if(flag==1)
		cout<<"YES\n";
		else
		cout<<"NO\n";
	}
   // cout << "Hello World!" << endl;
    return 0;
}
