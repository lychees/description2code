tc = int(raw_input())
if 1 <= tc <= 100:
    for i in range(tc):
        n = input()
        if 1 <= n <= 100000:
            l = map(int, raw_input().split())
            # l=t[:n]
            for i in range(len(l)):
                if (l[i] > len(l)):
                    print('not inverse')
                    break

                if (l[l[i]-1] != i+1):
                    print('not inverse')
                    break

            else:
                print('inverse')
