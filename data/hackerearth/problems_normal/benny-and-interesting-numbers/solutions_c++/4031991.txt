vector <ll> primes;
bool mark[2000006];
void sieve()
{
	for(ll i=2;i<=1000000;i++)
	{
		if(!mark[i])
		{
			primes.pb(i);
			for(ll j=i;j<=1000000;j=j+i)
			{
				mark[j]=1;
			}
		}
	}
}
