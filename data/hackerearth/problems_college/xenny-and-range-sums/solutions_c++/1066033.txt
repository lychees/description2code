#include <cstdio>

long long v[1000005];

int main()
{
    int n, q, l, r;
    scanf("%d%d", &n, &q);
    v[0]=0LL;
    for(int i=1; i<=n; i++)
    {
    	scanf("%lld", &v[i]);
    	v[i] += v[i-1];
    }
    while(q--)
    {
    	scanf("%d%d", &l, &r);
    	printf("%lld %lld\n", v[r]-v[l-1], v[r]-v[r-1]-(v[r-1]-v[l-1]));
    }
    return 0;
}
