#include <bits/stdc++.h>
using namespace std;
#define rdl(a) scanf("%lld",&a)
#define ll long long int
ll a[1010][1010], dp[1010][1010];
map<ll,int> mmap;
int main()
{
    int n,k,ans;
    ll dff, val=1000000000001;
    cin>>n>>k;
    for(int i=0;i<n;++i) for(int j=0;j<n;++j) rdl(a[i][j]);
    ans=0;
    for(ll i=1;(i*i*i)<=val;++i) {
    	dff=i*i*i;
    	mmap[dff]=1;
    }
    ans=0;
    for(int i=1;i<=n;++i) for(int j=1;j<=n;++j) {
    	dp[i][j]=a[i-1][j-1]+dp[i-1][j]+dp[i][j-1]-dp[i-1][j-1];
    	if(i>=k and j>=k) {
    		dff=dp[i][j]-dp[i-k][j]-dp[i][j-k]+dp[i-k][j-k];
    		if(mmap[dff]>0) ++ans;
    	}
    }
    cout<<ans<<endl;
}
