#include<bits/stdc++.h>
using namespace std;
int main()
{
    string s,t;
    int freq[256]={0};
    getline(cin,s);
    int l=s.length();
    int tot=0;
    for(int i=0;i<l;i++)
    {
        tot+=(int)s[i];
        freq[s[i]]++;
        t=s[i]+t;
    }
    tot/=l;
    if(tot%2==0)
    {
        cout<<t<<endl;
    }
    else
    {
        int k=0;
        int fr=0;
        for(int i=0;i<256;i++)
            {
                if(fr<freq[i])
                {
                    k=i;
                    fr=freq[i];
                }
            }
            cout<<((char)k)<<endl;
    }
}
