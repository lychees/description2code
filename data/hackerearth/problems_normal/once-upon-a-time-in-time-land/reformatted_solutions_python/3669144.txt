test = int(raw_input())
while test > 0:
    dp = []
    test -= 1
    N, K = map(int, raw_input().split())
    A = map(int, raw_input().split())
    for i in range(0, min(1 + K, N)):
        if len(dp) == 0:
            dp.append(A[i])
        else:
            dp.append(max(dp[-1], A[i]))

    for i in range(1 + K, N):
        dp.append(max(dp[i - K - 1] + A[i], A[i], dp[i - K - 1], dp[-1]))
    print max(max(dp), 0)
