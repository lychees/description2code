#include<bits/stdc++.h>
using namespace std;
const int N=1024;
vector<int>adj[N];
int dist[N];
#define MAX1 1000000
void update(int s)
{
	for(int i=0;i<adj[s].size();i++)
	{
		if(dist[adj[s][i]]>dist[s]+1)
		{
			dist[adj[s][i]]=dist[s]+1;
			update(adj[s][i]);
		}
	}
}
int main()
{
	ios::sync_with_stdio(false);
	int i,j,n,q,a,b,k;
	cin>>n>>q;
	for(i=2;i<=n;i++)
	dist[i]=MAX1;
	while(q--)
	{
		cin>>k;
		if(k==1)
		{
			cin>>a;
			if(dist[a]!=MAX1)
			cout<<dist[a]<<endl;
			else
			cout<<"-1"<<endl;
		}
		else
		{
			cin>>a>>b;
			adj[a].push_back(b);
			if(dist[b]>dist[a]+1)
			{
				dist[b]=dist[a]+1;
				update(b);
			}
			
		}
	}
	
	return 0;
}
