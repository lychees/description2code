#include <cstdio>
using namespace std;

#define lli long long int
int main()
{
    int t;
    lli n,divd = 3;
    scanf("%d",&t);
    while(t--){
    	scanf("%lld",&n);
    	//n = 10e15;
		if(n < 1)
			printf("0\n");    	
		else if(n < divd)
			printf("1\n");
		else if (n % divd == 0)
			printf("%lld\n",n/divd);
		else 
			printf("%lld\n",n);
    }
    return 0;
}
