#include<stdio.h>
#include<math.h>
#include<iostream>
using namespace std;
long long int divisors_odd(long long int n){
	long long int sum=0;
	for(long long int i=1; i<=sqrt(n); i++){
		if(n%i==0){
			if(i!=(n/i))
				sum=sum+i+(n/i);
			else
				sum+=i;
		}
	}
	return 8*sum;
}
long long int divisors_even(long long int n){
	long long int sum=0;
	for(long long int i=1; i<=sqrt(n); i++){
		if(n%i==0){
			if(i%2!=0)
				sum+=i;
			if(n/i!=i && (n/i)%2!=0)
				sum+=n/i;
		}
	}
	return 24*sum;
}
int main(){
	long long int t, num;
	scanf("%lld", &t);
	while(t--){
		scanf("%lld", &num);
		if(num%2==0)
			printf("%lld\n", divisors_even(num));
		else
			printf("%lld\n", divisors_odd(num));
		
	}	
	return 0;
}
