#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--){
   	int n;
   	cin>>n;
   	int a[n+1],s[n+1];
   	int i;
   	for(i=0;i<n;i++)
   		cin>>s[i];
   	for(i=0;i<n;i++)
   		cin>>a[i];
   	sort(s,s+n);
   	sort(a,a+n);
   	for(i=0;i<n;i++)
   		if(s[i]<=a[i]){
   				cout<<"NO\n";
   			break;
   			
   		}
   	if(i==n)
   		cout<<"YES\n";
   }
    return 0;
}
