#include <iostream>
#include<math.h>
using namespace std;
int main()
{long int t,n,ans,i,flag,q,rev,rem;
	cin>>t;
	while(t--)
	{
		cin>>n;
		ans=4;
		flag=1;
		if(n==2||n==3||n==5||n==7||n==11)
		cout<<n<<endl;
		else
		{
		while(ans!=5)
		{rev=0;
		if(n%2==0)
		n++;
		else
		{
			for(i=3;i<=sqrt(n);i+=2)
		{
			if(n%i==0)
			{flag=0;
			break;
			}
			else flag=1;
		}
		if(flag==1)
		{
			q=n;
			while(q!=0)
			{
				rem=q%10;
				rev=(rev*10)+rem;
			q=q/10;
			}
			if(rev==n)
			{
			cout<<n<<endl;
			ans=5;
			break;
			}
			else
			n++;
		}
		else if(flag==0)
		n++;
		}
		}
	}
	}
    return 0;
}
