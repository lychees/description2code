#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,i,j,n,c;
	  cin>>t;
	  while(t--)
	  {  c=0;
	  	cin>>n;
	  	int a[n];
	  	for(i=0;i<n;i++)
	  	  cin>>a[i];
	  	  for(i=0;i<n;i++)
	  	  {
	  	  	for(j=i;j<n;j++)
	  	  	 if(a[i]==a[j])
	  	  	 {
	  	  	 	c++;
	  	  	 }
	  	  }
	  	  cout<<c<<endl;
	  }
	return 0;
}
