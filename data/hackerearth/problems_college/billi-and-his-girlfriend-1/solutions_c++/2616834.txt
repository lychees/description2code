#include <iostream>
#include<bits/stdc++.h>
using namespace std;
long long fast_expo(long long int num,long long int po,int k)
{
	long long res=1;
	long long mod;
	mod=pow(10,k);
	 while(po>0) {
       if(po%2==1) res=(res*num)%mod;
       num=(num*num)%mod;
       po/=2;
    }
    return res%mod;
}

int main()
{
    long long int t,a,b,k,sum;
    long long ans;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>k;
    	ans=fast_expo(a,b,k);
    	sum=0;
   		//cout<<ans<<endl;
    	while(ans>0)
    	{
    		sum+=ans%10;
    		ans=ans/10;
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
