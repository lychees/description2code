#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	string s1,s2,s3,s4,s5,s6;
	int t;
	cin>>t;
	while(t--)
	{
		cin>>s1;
		cin>>s2;
		s3= s1 + s2;
		s4= s2 + s1;
		s5= s3;
		s6= s4;
		reverse(s3.begin(),s3.end());
		reverse(s4.begin(),s4.end());
		if(s3.compare(s5)==0||s4.compare(s6)==0)
		 cout<<"\nYES";
		else
		 cout<<"\nNO";		
	}
	return 0;
	
}
