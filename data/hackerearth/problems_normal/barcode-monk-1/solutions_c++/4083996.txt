#include <bits/stdc++.h>

using namespace std;

#define MAX 1005
#define INF 100000000
#define MOD 1000000007
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define ll long long

int dp[MAX][2], a[MAX];
bool sum[MAX];

int main(){
	int n, k;
	cin>>n>>k;

	sum[0] = true, dp[0][0] = 1;

	for(int i=0 ; i<k ; i++)
		cin>>a[i];

	for(int i=1 ; i<=n ; i++){
		for(int j=0 ; j<k ; j++){
			int w = a[j];
			if(i - w < 0)
				continue;
			sum[i] = sum[i] | sum[i - w];
		}
	}

	for(int i=1 ; i<=n ; i++){
		dp[i][0] = (dp[i-1][0] + dp[i-1][1]) % MOD;
		for(int j=1 ; j<=i ; j++){
			if(sum[j])
				dp[i][1] = (dp[i][1] + dp[i-j][0]) % MOD;
		}
	}	

	cout<<(dp[n][0] + dp[n][1]) % MOD<"\n";

	return 0;
}
