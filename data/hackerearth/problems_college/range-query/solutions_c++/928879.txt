#include<iostream>
#include<bits/stdc++.h>

using namespace std;

int occur[101][10001]={0};

int main(){
int n;
cin>>n;
int a[n],i,j;
for(i=0;i<n;i++){
        cin>>a[i];
        if(i>=1){
            for(j=0;j<=100;j++){
            if(j==a[i]) occur[j][i]=occur[j][i-1]+1;
            else occur[j][i]=occur[j][i-1];
        }
        }
        else occur[a[i]][0]=1;
}
/*for(i=0;i<=100;i++){
    for(j=0;j<n;j++){
        cout<<i<<":"<<occur[i][j]<<" ";
    }
    cout<<"\n";
}*/
int q,temp,x,y;
cin>>q;
while(q--){
        cin>>x>>y>>temp;
        //cout<<count(a+x,a+y+1,temp)<<"\n";
        if(x==0) cout<<occur[temp][y]<<"\n";
        else cout<<(occur[temp][y]-occur[temp][x])<<"\n";
}
return 0;
}
