#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--){
		string s;
		cin>>s;
		int odd = 0;
		vector<int> m(26,0);
		for(int i=0;i<s.length();i++){
			m[s[i]-'a']++;
			if(m[s[i]-'a']%2==0){
				odd--;
			}
			else{
				odd++;
			}
		}
		if(odd==0){
			cout<<1<<endl;
		}
		else{
			cout<<-1<<endl;
		}
	}
	return 0;
}
