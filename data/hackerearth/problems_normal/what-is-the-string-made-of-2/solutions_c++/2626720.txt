#include<bits/stdc++.h>
#define ll long long int
#define inps(x) scanf("%s",x)
using namespace std;
ll a[]={6,2,5,5,4,5,6,3,7,6};
int main(){
	ll c = 0;
	char s[109];
	inps(s);
	//for(ll i=0;i<=9;++i){
	//	cout<<a[i]<<" ";
	//}cout<<endl;
	for(ll i=0;i<strlen(s);++i){
		c+=a[s[i]-'0'];
	}
	printf("%lld",c);
	return 0;
}
