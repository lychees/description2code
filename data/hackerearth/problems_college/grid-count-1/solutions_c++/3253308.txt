#include <iostream>
#include <vector>
#include <cstdio>
#include <cmath>
using namespace std;

bool is_perfect_cube(long long n) {
    long long root(round(cbrt(n)));
    return n == root * root * root;
}

int main()
{
    int n,k; scanf("%d%d",&n,&k);
    vector<vector<long long> > mat;
    
    long long tmp, ans = 0; 
    vector<long long> row;
    
    for(int i=0; i<n; i++) {
    	row.clear();
    	for(int j=0; j<n; j++) {
    		scanf("%lld", &tmp);
    		row.push_back(tmp);
    	}
    	mat.push_back(row);
    }
    
    for(int c=1; c<n; c++) mat[0][c] += mat[0][c-1];
    for(int r=1; r<n; r++) mat[r][0] += mat[r-1][0];
    
    for(int i=1; i<n; i++) {
    	for(int j=1; j<n; j++) {
    		mat[i][j] += mat[i-1][j] + mat[i][j-1] - mat[i-1][j-1];
    	}
    }
    
    for(int i=k-1; i<n; i++) {
    	for(int j=k-1; j<n; j++) {
    		tmp = mat[i][j];
    		if(i >= k) tmp -= mat[i-k][j];
    		if(j >= k) tmp -= mat[i][j-k];
    		if(i >= k && j >= k) tmp += mat[i-k][j-k];
    		
    		if(is_perfect_cube(tmp)) ans++;
    	}
    }
    
    cout << ans << endl;
    
    return 0;
}
