#include<bits/stdc++.h>

using namespace std;

int main()
{
	
	int k;
	string s;
	int t;
	cin>>t;
	while(t--)
	{
		cin>>k>>s;
		for(int i=0;s[i]!='\0';i++)
		{
			
			if(s[i]>64&&s[i]<91)
			{
				
				s[i]=(((s[i]-65)+k)%26)+65;
			}
			if(s[i]>=97&&s[i]<=122)
			{
				
				s[i]=(((s[i]-97)+k)%26)+97;
			}
			
			
		}
		cout<<s<<endl;
	}
}
