# include <bits/stdc++.h>
using namespace std;
int main(void)
{
	long t;
	cin>>t;
	while(t--)
	{
		long long n;
		cin>>n;
		long long a[n];
		for(int i=0;i<n;i++)
			cin>>a[i];
		long long price=0;
		sort(a,a+n);
		long long i;
		for(i=0;i<n-1;i=i+2)
		{
			price=price+a[i]*a[i+1];
		}
		cout<<price<<endl;
	}
}
