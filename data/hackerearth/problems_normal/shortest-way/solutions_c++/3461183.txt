#include <bits/stdc++.h>
using namespace std;



long long solve(long long a, long long b)
{
	
	long long ret = 0;
	while(a > 1 or b > 1) {
		if(a > b) {
			ret += a / b;
			a = a % b;
		}else {
			ret += b / a;
			b = b % a;
		}
	}
	return ret > 0 ? ret -1 : ret;

}

int main()
{
	int t;
	cin >> t;
	while(t--) {
		long long a, b;
		scanf("%lld %lld", &a, &b);
		cout << solve(a, b) << endl;
	}
}
