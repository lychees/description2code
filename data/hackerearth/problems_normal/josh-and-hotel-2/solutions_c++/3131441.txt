#include<bits/stdc++.h>
using namespace std;
int main(){
	string s;
	int n;
	for( ;; ){
		cin>>s;
		if(s=="0") break;
		scanf("%d",&n);
		int cur=1;
		bool p[32];
		bool nost[32];
		memset (p,false,sizeof(p));
		memset (nost,false,sizeof(nost));
		int ans=0;
		for(int i=0;i<s.size();i++){
			if(nost[s[i]-'A']==true){
				nost[s[i]-'A']=false;
			}
			else if(p[s[i]-'A']==false){
				if(cur==n+1){
					ans++;
					nost[s[i]-'A']=true;
				}
				else{
					p[s[i]-'A']=true;
					cur++;
				}
			}
			else{
				p[s[i]-'A']=false;
				cur--;
			}
		}
		printf("%d\n",ans);
		}
		return 0;
}
