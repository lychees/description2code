#include <bits/stdc++.h>

using namespace std;

const int mod=1000000007;
long long int dp[1007][1007];

void func()
{
	long long int i,j,k;
	
	for (j=1;j<=1004;++j)
		dp[1][j]=1;
	
	for (j=1;j<=1004;++j)
		dp[j][1]=1;
		
	for (i=2;i<=1004;++i)
       for (j=2;j<=1004;++j)
       {
            dp[j][i]=(dp[j-1][i]%mod)+(dp[j][i-1]%mod)+(dp[j-1][i-1]%mod);
            dp[j][i]%=mod;
       }
}

int main()
{
    long long int t;
    func();
    scanf ("%lld",&t);
    while (t--)
    {
        long long int a,b,c,d,i,j;
        scanf ("%lld %lld %lld %lld",&a,&b,&c,&d);

        i=a-1;
        j=b-1;
		c=c-i;
		d=d-j;
        printf ("%lld\n",dp[c][d]);
    }
    return 0;
}
