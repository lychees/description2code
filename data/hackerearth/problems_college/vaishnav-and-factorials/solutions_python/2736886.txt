def factorial(num):
	prod=1
	while(num>0):
		prod=prod*num
		num=num-1
	return prod
def lucky_numbers(string):
	count=0
	for i in range(len(string)):
		if '4'== string[i]:
			count+=1
		if '7'==string[i]:
			count+=1
	return count
