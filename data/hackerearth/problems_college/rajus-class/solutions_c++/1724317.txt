#include <bits/stdc++.h>
using namespace std;

int main()
{
    int ite;
    cin>>ite;
    while(ite--){
    	int n;
    	cin>>n;
    	vector<pair<int, string>> vtr;
    	string s;
    	int p;
    	while(n--){
    		cin>>p>>s;
    		vtr.push_back(make_pair(p,s));
    	}
    	sort(vtr.begin(), vtr.end());
    	cin>>p;
    	cout<<vtr[p-1].second<<endl;
    }
    return 0;
}
