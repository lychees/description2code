#include <iostream>
using namespace std;

int coprime(int max,int n)
{
	for(int i =2;i<=n;i++)
	{
		if(max%i==0 && n % i==0)
		return 1;
		
	}
		return 0;
}

int main()
{
   int max,n,c=0;
    cin>>max>>n;
    int a[n];
    for(int i =0 ;i<n;i++)
    {
    	cin>>a[i];
    	if(coprime(max,a[i]))
    	c++;
    }
    cout<<c;
}
