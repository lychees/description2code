#include <iostream>
#include <vector>
#include <cmath> 
#include <algorithm>
typedef long long ll; 
using namespace std; ll bad[12]; ll ans [100050]; int MOD=1000000007;
int main()
{
   int a; cin >> a; int k=0;
   for (int g=0; g<a; g++)
   {
       cin >> bad[g]; 
       k=(bad[g]>k) ? bad[g] : k;
   }ans[0]=1; 
   ans[1]=1; ans[2]=2; 
   for (int g=3; g<=k; ++g)
   {
       ans[g]=((ans[g-3])%MOD+(ans[g-2])%MOD+(ans[g-1])%MOD)%MOD; 
   }
   for (int g=0; g<a; g++)
   {
       cout << ans[bad[g]]%MOD << '\n';
   }
    
    
    
}
