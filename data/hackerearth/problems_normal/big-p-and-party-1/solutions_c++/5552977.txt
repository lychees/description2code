#include <bits/stdc++.h>
using namespace std;
int main()
{
	queue<int> q;
	int a,b;
	scanf("%d %d",&a,&b);
	int graph[a][a];
	int i,j;
	int ans[1005]={0};
	int visited[1005]={0};
	for(i=0;i<a;i++)
	{
		for(j=0;j<a;j++)
		graph[i][j]=0;
	}
	while(b--)
	{
		int c1,c2;
		scanf("%d %d",&c1,&c2);
		graph[c1][c2]=1;
		graph[c2][c1]=1;
	}
	q.push(0);
	int temp;
	visited[0]=1;
	ans[0]=0;
	while(!q.empty())
	{
		temp=q.front();
		q.pop();
		for(i=0;i<a;i++)
		{
			if(graph[temp][i]==1 && visited[i]==0)
			{
			q.push(i);
			visited[i]=1;
			if(ans[i]==0)
			ans[i]=ans[temp]+1;
		}
	
		
	}}
	for(i=1;i<a;i++)
	{
		if(ans[i]==0)
		printf("-1\n");
		else
		printf("%d\n",ans[i]);
	}
    return 0;
}
