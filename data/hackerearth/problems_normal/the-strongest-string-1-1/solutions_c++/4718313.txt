#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int N,i,ind;
    string s2;
    cin>>N;
    string s;
    cin>>s;
    int hash[26]={0};
    for(i=0;i<N;i++)
    	hash[s[i]-'a']++;
    int visited[26]={0};
    char c;
    int prev_ind;
    for(i=25;i>=0;i--)
    {
    	if(hash[i]>0 && visited[i]==0)
    	{
    		c=i+'a';
    		string s1(1,c);
    		if(s2.length()>0)
    		{
    			ind=s.find_first_of(s1,prev_ind+1);
    			if(ind>prev_ind)
    			{
    				s2.push_back(s[ind]);
    				prev_ind=ind;
    			}	
    		}
    		else if(s2.length()==0)
    		{
    			ind=s.find_first_of(s1);
    			s2.push_back(s[ind]);
    			prev_ind=ind;
    		}
    		
    		visited[i]=1;
    		s1.clear();
    	}
    }
    cout<<s2<<"\n";
    return 0;
    
}
