string s;

int chck(string a){
	int l=a.length();
	for(int i=0;i<a.length();i++){
		if(a[i]!=a[l-1-i])
		return i;
	}
	return -1;
}

int main(){
	int t;
	cin>>t;
	while(t--){
	//	memset(dp,-1,sizeof(dp));
		cin>>s;
		string x,nx;
		nx=s;
		while(1){
			int ret=chck(s);
			if(ret==-1)
			break;
			s=nx;
			x="";
			for(int j=ret;j>=0;j--)
			x+=s[j];
			x=s+x;
			s=x;
		}
		cout<<s.length()<<endl;
	}
	return 0;
}
