tc = int(raw_input())
c = []
for i in range(0, tc):
    count = 0
    dc = 1
    dp = 0
    str = raw_input()
    N, M = map(int, str.split())
    dnc = N - 1
    dnp = M
    while dnc and dnp:
        count = count + 1
        x = dp + min(dc, dnp)
        y = dc + min(dp, dnc)
        dnp = dnp - min(dnp, dc)
        dnc = dnc - min(dnc, dp)
        dp = x
        dc = y
    if dnc:
        count = count + (1 + (dnc - 1)/dp)
    print count
