#include <iostream>
#include<cstdio>
#include<cstring>
using namespace std;
 
int main() {
int t,test;
 
scanf("%d",&t);
test=0;
char str[100100];
while(t--){
	scanf("%s",str);
	int b[100100];
	b[0]=-1;
	int i=0,j=-1;
	
	while(i<strlen(str)){
		
		while(j>=0 && (str[i]^str[j])) j=b[j];
		i++;
		j++;
		b[i]=j;
	}
	printf("Case %d: %d\n",++test,b[strlen(str)]>0?b[strlen(str)]:-1);
}
 
	return 0;
}
