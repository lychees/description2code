#include<bits/stdc++.h>
using namespace std;
#define si(n) scanf("%lld",&(n))
#define pi(n) printf("%lld\n",(n));
typedef long long ll;
int main()
{
	int a[26];
	ll t;
	cin>>t;
	while(t--)
	{
		memset(a,0,sizeof(int)*26);
		ll n;
		cin>>n;
		while(n--)
		{
			string s;cin>>s;
			for(ll i=0;i<26;i++)
			{
				if(s.find('a'+i)!=string::npos)
				a[i]++;
			}
		}
		ll m=0;
		for(int i=0;i<26;i++)
		{
			if(a[i]>a[m])
			m=i;
		}
		cout<<char('a'+m)<<endl;
	}
	return 0;
}
