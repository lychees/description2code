string mix(string a,string b){
    int i,j;
    bool valid =true;
    for(i=0;i<a.length();i++){
        valid=true;
            for(j=i;j<a.length();j++){
                    if((j-i)>=b.length()||b[j-i]!=a[j])
                    {
                        valid=false;
                        break;
                    }

            }
              if(valid==true)
                break;
    }

    for(j=a.length()-i;j<b.length();j++)
        {
            a+=b[j];
        }

return a;
}
