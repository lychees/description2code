using namespace std;
bool prime[2000005]={0};
ll pr1[2000005];
ll pr2[2000005];
ll x=1;
ll y=1;
void sieve()
{
    prime[1]=true;
    prime[0]=true;
	for(ll i=2;i<=sqrt(2000004);i++)
	{
		if(prime[i]==false)
		{
			for(ll j=i*i;j<=2000004;j+=i)
			{
			prime[j]=true;
			}
		}
	}
	for(ll i=2;i<=2000004;i++)
	{
		if(prime[i]==false)
		{
			pr1[x]=i;
			if(prime[x]==false)
			{
				pr2[y]=i;
				y++;
			}
			x++;
		}
	}
}
int main()
{
	sieve();

	ll t;
	sf(t);
	while(t--)
    {
        ll n,m;
        sf(n);sf(m);
        ll prod =pr2[n];
        prod=(prod%mod*pr2[m]%mod)%mod;
        pf(prod);
    }
}
