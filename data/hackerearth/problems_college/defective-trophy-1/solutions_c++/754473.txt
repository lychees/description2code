#include <iostream>
#include <cmath>

using namespace std;

int
solve (int n) {
	if (n == 0 || n == 1) return 0;
	return 1 + solve ((int) ceil (n / 3.0));
}

int main()
{
   int T, N;
   scanf ("%d", &T);
   while (T--) {
       scanf ("%d", &N);
       int ret = solve (N);
       printf ("%d\n", ret);
   }
}
