#include <bits/stdc++.h>
using namespace std;
int xmove[8]={2,1,-1,-2,-2,-1,1,2};
int ymove[8]={1,2,2,1,-1,-2,-2,-1};
vector<pair<int,int> >v;
vector<long long int>res;
int visit[1005][1005];
int a[1005][1005];
int n,m;
long long int cnt;
long long int fact[100005];
#define MOD 1000000007
void dfs(int x,int y)
{
	visit[x][y]=1;
	cnt++;
	for(int i=0;i<8;i++)
	{
		int xx=x+xmove[i];
		int yy=y+ymove[i];
		if(xx>=1&&xx<=n&&yy>=1&&yy<=m&&a[xx][yy]==1&&visit[xx][yy]==0)
		{
			dfs(xx,yy);
		}
	}
}
int main()
{
	int t,x,y,q;
	scanf("%d",&t);
	fact[1]=1;
	for(int i=2;i<=1000000;i++)
	fact[i]=(fact[i-1]*i)%MOD;
	while(t--)
	{
		memset(a,0,sizeof a);
		scanf("%d%d%d",&n,&m,&q);
		v.clear();
		for(int i=1;i<=q;i++)
		{
			scanf("%d%d",&x,&y);
			a[x][y]=1;
			v.push_back(make_pair(x,y));
		}
		memset(visit,0,sizeof visit);
		res.clear();
		for(int i=0;i<v.size();i++)
		{
			pair<int,int>p=v[i];
			x=p.first;
			y=p.second;
			if(visit[x][y]==1)
			continue;
			cnt=0;
			dfs(x,y);
			res.push_back(cnt);
		}
		long long int ans=1;
		for(int i=0;i<res.size();i++)
		{
			
			ans=(ans*fact[res[i]])%MOD;
		}
		cout<<ans<<"\n";
	}
	return 0;
	
	
}
