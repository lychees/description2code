n, p = map(int, raw_input().split())
prob2 = p/100.0
prob3 = 1 - prob2
if n <= 1:
    print("%.6f" % 0.0)
elif n == 2:
    print("%.6f" % prob2)
elif n == 3:
    print("%.6f" % prob3)
else:
    pre = prob3
    pre2 = prob2
    pre3 = 0.0
    for i in xrange(4, n+1):
        cur = (pre2 * prob2) + (pre3 * prob3)
        pre3 = pre2
        pre2 = pre
        pre = cur
    print("%.6f" % cur)
