#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
#define gc getchar
inline long long int read()
{
	register char c = gc();
	while(c<'0' || c>'9') c = gc();
	long long int ret = 0;
	while(c>='0' && c<='9')
	{
		ret = 10 * ret + c - 48;
		c = gc();
	}
	return ret;
}
long long unsigned a[1000003],k=1;
int main()
{
    int t,i,j;
    
    for(i=1;i<1000002;i++)
    {
    	k=(k*i)%MOD;
    	a[i]=(3*k+3)%MOD;
    }
    t=read();
    while(t--)
    {
    	long long unsigned n;
    	//scanf("%llu",&n);
    	n=read();
    	printf("%llu\n",a[n]);	
    }
	return 0;
}
