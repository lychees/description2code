#include<iostream>
#include<cstdio>
#include<string.h>
using namespace std;
int main()
{
    int test_cases;
    cin>>test_cases;
    getchar();
    while(test_cases>0)
    {
        char arr[101];
        gets(arr);
        int i,j,n;
        n=strlen(arr);
        i=0;
        j=n-1;
        
        while(i<j)
        {
        	while( (i<j) && arr[i]!='*'&& (arr[i]==','||arr[i]=='?'||arr[i]=='.'||arr[i]=='!' ||
        	arr[i]==' '))
        		i++;
        		
        	while( (i<j) && arr[j]!='*' && (arr[j]==','||arr[j]=='?'||arr[j]=='.'||arr[j]=='!' ||
        	arr[j]==' '))
        		j--;
        		
        	if(arr[i]=='*')
        	{
        		cout<<arr[j]<<"\n";
        		break;
        	}
        	
        	if(arr[j]=='*')
        	{
        		cout<<arr[i]<<"\n";
        		break;
        	}
        	
        	if(arr[i]==arr[j])
        	{
        		i++;
        		j--;
        	}
        	
        }
        
        test_cases--;
    }
    return 0;
}
