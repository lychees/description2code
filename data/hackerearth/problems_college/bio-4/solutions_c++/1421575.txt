#include<iostream>
//#include<conio.h>
#include<string.h>

using namespace std;

int match[1000];
char act[10002];
char mod[10002];

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
              cin>>act;
              cin>>mod;
              int ctr=0,nctr=0;
              int al=strlen(act);
              int ml=strlen(mod);
              
              for(int i=0;i+ml<=al;i++)
              {
                      nctr=0;
                      for(int j=0,r=i;j<ml;j++,r++)
                      {
                              if(act[r]!=mod[j])
                                nctr++;
                              
                              }
                      match[ctr++]=nctr;
                      }
              
              int min=match[0];
              for(int i=1;i<ctr;i++)
              {
                      if(match[i]<min)min=match[i];
                      }
              
              cout<<min<<"\n";
              
    
    
    
    
    }
  //  getch();
    return 0;
}
