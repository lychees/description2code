from fractions import gcd


def isPrime(num):
    count = 0
    for i in range(1, num+1):
        if gcd(num, i) == 1:
            count = count + 1

    c = 0
    for k in range(2, count+1):
        if count % k == 0:
            c += 1

    if c == 1:
        print "TRUE"
    else:
        print "FALSE"

for i in range(1, t+1):
    n = int(raw_input())
    count = isPrime(n)
