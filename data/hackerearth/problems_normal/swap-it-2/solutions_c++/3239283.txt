#include<bits/stdc++.h>
using namespace std;

int n,k;
int A[1010];

int main()
{
    int t,cas=1;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        for(int i=0;i<n;i++)    cin>>A[i];

        int pos=-1;

        for(int i=0;i<n;i++)
        {
            pos=i;
            for(int j=i;j<n;j++)
            {
                if(j-i>k)   break;
                if(A[pos]>=A[j])    pos=j;
            }
            for(int j=pos; j-1>=i ; j--)
                swap(A[j],A[j-1]);
            k=k-(pos-i);

            if(k<=0)    break;
        }

        for(int i=0;i<n-1;i++)  cout<<A[i]<<" ";
        cout<<A[n-1]<<endl;
    }
    return 0;
}
