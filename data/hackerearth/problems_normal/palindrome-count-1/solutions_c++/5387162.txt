#include <iostream>
#include <algorithm>

using namespace std;


bool isPalind(string str, int left, int right) {

     int gap = right - left;

     if (gap < 0) return false;
     if (gap == 0) return true;
     if (gap == 1 || gap == 2) return  (str[left] == str[right] ? true:false) ;

     bool ret = false;

     do {
          if (str[left++] != str[right--]) {
               break;
          }

          if (left>=right) ret = true;
     } while(left<right);

     return ret;
}

int palindCnt(string str) {

     int totCnt = 0;
     for (int i=0; i<str.size(); i++){
          for (int j=i; j<str.size(); j++){

               if (isPalind(str,i,j)) {
                    //cout << str.substr(i,j-i+1) << " is palind.\n";
                    totCnt ++;
               }
          }
     }

     return totCnt;
}

int main(int argc, char **argv) {

string str ; //= argv[1];
cin >> str;

//cout << "isPalind :" << isPalind(str,0, str.size() - 1) << endl;
cout <<  palindCnt(str) << endl;

return 0;

}
