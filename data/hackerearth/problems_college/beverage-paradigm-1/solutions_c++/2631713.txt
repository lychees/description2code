#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	scanf("%d",&n);
	int a[n];
	int i,j;
	for (i = 0; i < n; i++) {
		scanf("%d",&a[i]);
	}
	int k;
	int wt;
	scanf("%d",&wt);
	int f = 0;
	for (i = 0; i < n; i++) {
		for (j = i + 1; j < n; j++) {
			for (k = j + 1; k < n; k++) {
				if (a[i] + a[j] + a[k] == wt) {
					f = 1;
					break;
				}
			}
			if (f == 1) {
				break;
			}
		}
		if (f == 1) {
			break;
		}
	}
	if (f == 1) {
		printf("True\n");
	} else {
		printf("False\n");
	}
	
	return 0;
}
