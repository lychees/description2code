#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>

using namespace std;

void solve(int n,long long int e)
{
	
	int ans,m;
	if(n==0||n==1||n==5||n==6)
		ans=n;
	else if(n==2)
	{
		m=e%4;
		if(m==0)
			ans=6;
		else if(m==1)
			ans=2;
		else if(m==2)
			ans=4;
		else
			ans=8;
	}
	else if(n==3)
	{
		m=e%4;
		if(m==0)
			ans=1;
		else if(m==1)
			ans=3;
		else if(m==2)
			ans=9;
		else
			ans=7;
	}
	else if(n==4)
	{
		m=e%2;
		if(m==0)
			ans=6;
		else
			ans=4;
	}
	else if(n==7)
	{
		m=e%4;
		if(m==0)
			ans=1;
		else if(m==1)
			ans=7;
		else if(m==2)
			ans=9;
		else
			ans=3;
	}
	else if(n==8)
	{
		m=e%4;
		if(m==0)
			ans=6;
		else if(m==1)
			ans=8;
		else if(m==2)
			ans=4;
		else
			ans=2;
	}
	else if(n==9)
	{
		m=e%2;
		if(m==0)
			ans=1;
		else
			ans=9;
	}
	printf("%d\n",ans);
}

int main()
{
    char b[19];
    int l1,m;
    long int t;
    long long int e;
    scanf("%ld",&t);
    while(t--)
    {
    	scanf("%s %lld",b,&e);
    	l1=strlen(b);
    	//cout<<l1;
    	m=b[l1-1];
    	solve(m-48,e);
    }
    return 0;
}
