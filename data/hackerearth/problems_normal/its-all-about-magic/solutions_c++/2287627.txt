#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    long long int i,fib[91];
    fib[0]=0,fib[1]=1;
    for(i=2;i<91;i++)
    {
    	fib[i]=fib[i-1]+fib[i-2];
    }
    while(t--)
    {
    	int n;
    	cin>>n;
    	cout<<fib[n-1]<<" "<<fib[n]<<endl;
    }
}
