#include<iostream>
using namespace std;
void fastscan(int &x)
{
bool neg=false;
register int c;
x =0;
c=getchar();
if(c=='-')
{
neg = true;
c=getchar();
}
for(;(c>47 && c<58);c=getchar())
x = (x<<1) + (x<<3) +c -48;
if(neg)
x *=-1;
}
int main(){
	ios_base::sync_with_stdio(false);
	 int x,p;
	fastscan(x);
	while(x--){
		int n,c,m,res;
		fastscan(n);
		fastscan(c);
		fastscan(m);
		res=n/c;
		p=res;
		while(p>0){
			res+=p/m;
			p=p%m+p/m;
			if(p<m)
			break;
		}
		cout<<res<<endl;
	}

}
