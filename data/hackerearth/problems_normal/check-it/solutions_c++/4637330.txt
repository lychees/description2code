#include<bits/stdc++.h>
#define ll long long int
using namespace std;
ll n,m,arr[10000];
pair<ll,ll> v[10000];
ll root(ll i)
{
    while(i!=arr[i])
        i=arr[arr[i]];
    return i;
}
main()
{
    ll t,i,x,y,flag;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        for(i=1;i<n+1;i++)
            arr[i]=i;
        pair<ll,ll> v[m+1];
        for(i=1;i<m+1;i++)
        {
            cin>>x>>y;
            v[i]=make_pair(x,y);
        }
        flag=0;
        for(i=0;i<n-1;i++)
        {
            cin>>x;
            if(root(v[x].first)==root(v[x].second))
                flag=1;
            else
                arr[root(v[x].first)]=root(v[x].second);
        }
        if(flag==0)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    return 0;
}
