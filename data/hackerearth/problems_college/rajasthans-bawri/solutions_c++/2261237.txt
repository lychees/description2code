#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	int a, b, c, d;
    	cin>>a>>b>>c>>d;
    	if((a>=c && b>=d) || (a>=d && b>= c))
    	{
    		cout<<"Possible"<<endl;
    	}
    	else
    	{
    		cout<<"Not Possible"<<endl;
    	}
    }
    return 0;
}
