#include<bits/stdc++.h>
#define pb push_back
using namespace std;

int main()
{
	int n,val;
	queue<int>q;
	vector<int>ideal;
	cin >> n;
	
	for(int i=0; i<n; i++)
	{
		cin >> val;
		q.push(val);
	}
	
	for(int i=0; i<n; i++)
	{
		cin >> val;
		ideal.pb(val);
		
	}

	int time=0, i=0;
	while(!q.empty())
	{
		if(ideal[i]==q.front())
			i++;
		

		else
		{
			val = q.front();
			q.push(val);
		}
		q.pop();
		time++;
	}
	
	cout<<time<<endl;


}
