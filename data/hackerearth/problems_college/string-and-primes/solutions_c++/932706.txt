#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define MAX 10005
int a[MAX];
int primes[MAX];
long long arr[MAX];
void gen_primes()
     {
       int i,j;
       for(i=0;i<MAX;i++) primes[i] = 1;
       primes[0] = 0;
       primes[1] = 0;
       for(i=2;i<=(int)sqrt(MAX);i++)
         if (primes[i])
           for(j=i;j*i<MAX;j++) primes[i*j] = 0;
     }

int main() {
	int t;
	cin>>t;
	gen_primes();
	string s;
	while(t--) {
		cin>>s;
		int f = 0;
		for(int i = 0 ; i < s.length(); i++) {
			if(!primes[s[i]]) {
				f = 1;
				cout<<s[i];
			}
		}
		if(f == 0) {
			cout<<"NULL";
		}
		cout<<endl;
	}
	return 0;
}
