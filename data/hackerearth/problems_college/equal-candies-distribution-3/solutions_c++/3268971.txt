#include <iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,i,c=0,s;
		scanf("%d",&n);
		int a[n],b[n];
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		sort(a,a+n);
		for(i=0;i<n;i++)
			b[i]=a[i]-a[0];
			for(i=0;i<n;i++)
			{
				s=b[i];
				c=c+s/5;
				s=s%5;
				c=c+s/2;
				s=s%2;
				c=c+s/1;
				s=s%1;
			}
			printf("%d\n",c);
	}
	return 0;
}
