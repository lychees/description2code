#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
int main(){
	long long t,a,b,c,d,e;
	scanf("%lld",&t);
	while(t--){
		long long int ar[3];
		 scanf("%lld%lld%lld",&ar[0],&ar[1],&ar[2]);
		sort(ar,ar+3);
		a = ar[0] % mod;
		b = (ar[1]-1) % mod ;
		c = (ar[2]-2) % mod;
		if(c<0) c+=mod;
		d = (a*b)%mod;
		e = (d*c)%mod;
		printf("%lld\n",e);
	}
	return 0;
}
