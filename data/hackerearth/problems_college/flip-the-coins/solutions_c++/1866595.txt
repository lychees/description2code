#include<bits/stdc++.h>
using namespace std;
int main(void)
{
	int t,q,n,arr[101],cnt=0,i,j,a,b;
	for(i=0;i<101;i++)
	{
		arr[i]=0;//indicating they are tails
	}
	cin>>t>>n;
	for(i=0;i<n;i++)
	{
		cin>>q>>a>>b;
		if(q==0)
		{
			for(j=a;j<=b;j++)
			{
				if(arr[j]==0)arr[j]=1;
				else arr[j]=0;
			}
		}
		if(q==1)
		{
			for(j=a;j<=b;j++)
			{
				if(arr[j]==1)cnt++;
			}
			cout<<cnt<<endl;cnt=0;
		}
		
	}
}
