#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long n,m,a,b,c,p,q,ans;
		cin>>n>>m;
		cin>>a>>b>>c;
	
		if(c>a||c>b)
		{
				p=n*a+m*b;
			    if(m>=n)
			    {
			    	q=n*c+n*c+(m-n)*b;
			    }
			    if(m<n)
			    {
			    	q=m*c+m*c+(n-m)*a;
			    }
			    if(p>=q)
			    {
			    	ans=p;
			    }
			     if(p<q)
			    {
			    	ans=q;
			    }
			    
			
		}
		else
		{
		  ans= n*a+m*b;
		}
		cout<<ans<<"\n";
		
	}
	return 0;
}
