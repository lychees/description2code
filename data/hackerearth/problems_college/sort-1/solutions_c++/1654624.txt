#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int t;
	cin>>t;
	unsigned long int x,i,n;
	vector<unsigned long int> v;
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
        {
            cin>>x;
            v.push_back(x);
        }
        sort(v.begin(),v.end());
        for(i=0;i<n;i++)
        {
            cout<<v[i]<<" ";
        }
        cout<<endl;
        v.clear();
	}
    return 0;
}
