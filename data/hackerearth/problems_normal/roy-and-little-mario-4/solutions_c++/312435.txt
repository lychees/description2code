#include<iostream>

#define MOD 1000000007

using namespace std;

int dp[100001];

int main()
{
    int t, i;
    cin>>t;
    while (t--)
    {
        int n;
        cin>>n;

        dp[1] = 1;
        dp[2] = 2;
        dp[3] = 4;

        for (i = 4; i <= n; i++){
            if (dp[i]!=0)
                continue;
            dp[i] = (((dp[i-1] + dp[i-2]) % MOD) + dp[i-3])%1000000007;
        }
        cout<<dp[n]<<endl;
    }
}
