#include <iostream>
#include<bits/stdc++.h>
using namespace std;
long long X(long long a[],long long b[],long long count,long long l,long long R,long long result)
{
	if(l==R)
		return 0;	
	 if(count==1){
		return result+=a[l]+X(a,b,2,++l,R,result);
	}
	else if(count==2)
	{
		return result+=b[l]+X(a,b,1,++l,R,result);	
	}	
	
}


int main()
{
long long q,a[100000],b[100000],n,result=0;
cin>>n>>q;
long long p[q][3],r[q];
for(long long i=0;i<n;i++)
	cin>>a[i];
for(long long i=0;i<n;i++)
	cin>>b[i];
for(long long i=0;i<q;i++)
	for(long long j=0;j<3;j++)
		cin>>p[i][j];
for(long long int i=0;i<q;i++){

	r[i]=X(a,b,p[i][0],p[i][1]-1,p[i][2],result);	
}
for(long long int i=0;i<q;i++)
	cout<<r[i]<<endl;		
		
    return 0;
}
