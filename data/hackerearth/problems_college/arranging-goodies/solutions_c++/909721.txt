#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>

#define ll long long
using namespace std;

int main()
{
ios::sync_with_stdio(false);
   int n,m,mid = 0,index = 0;
   cin >>n >> m;
   int arr[m];
   if(!(m%2)) mid = m/2;
   else mid = (m+1)/2;
   arr[index++] = mid;
   int i_l,i_r;
   i_l = i_r = mid;
   bool flag = true;
   if(!(m%2)) flag = false;
   for(int i=1;i<m;i++)
   {
       if(flag)
       {
           int temp = --i_l;
           arr[index++] = temp;
           flag = false;
       }
       else
       {
           int temp = ++i_r;
           arr[index++] = temp;
           flag = true;
       }
   }
   for(int i =0 ;i<n;i++)
   {
       cout << arr[i%m] << endl;
   }
   return 0;
}
