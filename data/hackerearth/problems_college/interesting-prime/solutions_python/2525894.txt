import sys,math
import itertools
def prime_fact(number):
	prime=[]
	while(number%2==0):
		prime.append(2)
		number=number/2
	for i in range(3,int(math.sqrt(number))+1,2):
		while(number%i==0):
			prime.append(i)
			number=number/i
	if(number>2):
		prime.append(number)
	return prime
tcases=input()
for i in range(0,tcases):
	count=0
	n=input()
	l1=prime_fact(n)
	for j in range(1,len(l1)+1):
		l2=list(itertools.combinations(l1, j))
		count=count+len(set(l2))
	print count
