#include<bits/stdc++.h>
using namespace std;
#define ll long long int

ll parrent[1009],mark,city,road;

ll visit[1005][1005];
bool take[1005];
vector<ll>graph[1005];

void dfs(ll u,ll p)
{
    ll v,l,i;
    take[u]=true;
    l=graph[u].size();
    for(i=0;i<l;i++)
    {
        v=graph[u][i];
        if(!take[v] && visit[u][v]<p)
            dfs(v,p);
    }
}

int main()
{
    ll t,q,a,b,c,i;

    //freopen("input.txt","r",stdin);
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld %lld",&city,&road);
        for(i=0;i<=city;i++)
            graph[i].clear();
        memset(visit,0,sizeof visit);
        for(i=1; i<=road; i++)
        {
            scanf("%lld %lld %lld",&a,&b,&c);
            graph[a].push_back(b);
            graph[b].push_back(a);
            visit[a][b]=c;
            visit[b][a]=c;
        }

        scanf("%lld",&q);
        c=0;
        for(i=1;i<=q;i++)
        {
            scanf("%lld %lld",&a,&b);
            memset(take,false,sizeof take);
            dfs(a,visit[a][b]);
            if(take[a] && take[b])
               continue;
            c++;
        }
        a=__gcd(q,c);
        q=ceil(q/a);
        c=ceil(c/a);
        printf("%lld/%lld\n",c,q);
    }
    return 0;
}
