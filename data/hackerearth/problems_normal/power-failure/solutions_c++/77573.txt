#include<iostream>
#include<algorithm>

using namespace std;

int prime[100001];
int counter[100001];
int arr[100001];

int main()
{
	int n,m,t,i,j;
	long long int ways;
	long long int small_count;
	cin>>t;
	for(i=0;i<100001;i++)
	{
		prime[i]=1;
	}
	prime[0]=prime[1]=0;
	for(i=2;i<100001;i++)
	{
		if(prime[i]==1)
		{
			for(j=2*i;j<100001;j+=i)
			{
				prime[j]=0;
			}
		}
	}
	counter[100000]=1;
	for(i=100000-1;i>-1;i--)
	{
		if(prime[i]==0)
		{
			counter[i]=counter[i+1]+1;
		}
		else
		{
			counter[i]=counter[i+1];
		}
	}
	
	for(i=0;i<t;i++)
	{
		cin>>n;
		cin>>m;
		for(j=0;j<n;j++)
		{
			cin>>arr[j];
		}
		sort(arr,arr+n);
		ways=1;small_count=0;
		for(j=n-1;j>-1;j--)
		{
			ways*=(counter[arr[j]]-counter[m+1]-small_count);
			if(ways>100000000){ways%=1000000007;}
			small_count++;
		}
		cout<<ways<<endl;
	}
	return 0;
}
