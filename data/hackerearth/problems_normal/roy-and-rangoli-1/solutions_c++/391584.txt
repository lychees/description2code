int  a[2000001]={0};

int prime[2000005];
int p=1;
void generate(int y)
    {int i,j;
     prime[0]=2;
	 for(i=3;i*i<=y;i+=2)
        {if(a[i])
          continue;
         prime[p++]=i;
		for(j=i*i;j<=y;j+=2*i)
           {a[j]=1;
           }
        }
        for(i;i<=y;i+=2)
          if(!a[i])
            prime[p++]=i;
        
    }
int main()
 {
