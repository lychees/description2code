t=int(raw_input())
while t:
	s=raw_input()
	n=len(s)
	s=s.strip().split(".")[1]
	su=4
	for i in s:
		if i!='a' and i!='e' and i!='i' and i!='o' and i!='u':
			su+=1
	print str(su)+'/'+str(n)		
	t-=1
