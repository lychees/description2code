#include<iostream>
#include<string>
#include<cstring>
using namespace std;
int main()
{
  string str;
  getline(cin,str);
  int a[128];
  int i,l=0;
  for(i=0;i<128;i++)
  a[i]=0;
  for(i=0;i<str.size();i++)
  {
     int d=(int)str[i];
      a[d]++;
  }
  char x;
  for(i=0;i<128;i++)
  {
     if(a[i]>l)
     {
        l=a[i];
        x=(char)i;
     }
  }
 
 
  cout<<x<<" "<<l<<endl;
}
