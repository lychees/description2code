#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int find(int n,int arr[])
{
	if(arr[n]==n)
		return n;
	int x = find(arr[n],arr);
	arr[n] = x;
	return x;
}
void uni(int x,int y,int arr[])
{
	x = find(x,arr);
	y = find(y,arr);

	arr[x]=y;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;
		cin>>n>>m;
		int arr[n+1];
		for(int i=0;i<=n;i++)
			arr[i]=0;
		pair<int ,int > gp[m];
		for(int i=0;i<m;i++)
		{
			int x,y;
			cin>>x>>y;
			gp[i] = make_pair(x,y);
		
		}
		for(int i=0;i<n-1;i++)
		{
			int x;
			cin>>x;
			pair<int,int> p = gp[x-1];
			arr[p.first]++;
			arr[p.second]++;
		}
		int f=1;
		for(int i=1;i<n;i++)
		{
			if(arr[i]==0)
			{
				f=0;
				break;
			}
		}
		if(f==1)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}

}
