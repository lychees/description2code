#include<iostream>
using namespace std;
typedef long long ll;
int main()
{
    int w;
    cin>>w;
    while(w--)
    {
        ll n,t,i,a;
        cin>>n>>t;
        for(i=1;i<=t;i++)
        {
         if(n%2==0)
         a=n/2;
         else
         a=(n+1)/2;
         n=n-a;
         n=n-(n/4);
        }
        cout<<n<<endl;
    }
}
