def compute_max_perm_sums(n):
	ret, diff, last = [0, 0, 1, 3], 4, 3
	for i in xrange(4, n + 1, 2):
		ret.append(last + diff)
		ret.append(last + diff + diff)
		last, diff = last + diff + diff, diff + 2
	return ret

def main():
	max_perm_sums = compute_max_perm_sums(100000)
	max_perm_sums[1] = 1
	num_cases = int(raw_input())
	for cas in xrange(num_cases):
		n = int(raw_input())
		print max_perm_sums[n]
		

if __name__ == '__main__':
	main()
