#include <bits/stdc++.h>
using namespace std;
typedef long int li;
const li SIZE =1000001;
bool primes[SIZE];
li cum_sum[SIZE];

void mark_Primes()
{
	for(li i=2;i<SIZE;i++)
	{
		if(!primes[i])
		{
			for(li j=2;j<SIZE && (i*j)<SIZE;j++)
			{
				primes[i*j]=true;
			}
		}
		
	}
}


int main()
{
	mark_Primes();
	for(li i=2;i<SIZE;i++)
    {
    	cum_sum[i]=primes[i]?cum_sum[i-1]:cum_sum[i-1]+i;
    }
    li N;
    cin>>N;
    while(N--)
    {
    	li l,r;
    	cin>>l>>r;
    	cout<<cum_sum[r]-cum_sum[l-1]<<endl;
    }
    return 0;
}
