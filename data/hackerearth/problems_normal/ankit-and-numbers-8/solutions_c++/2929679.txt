#include <iostream>
#include <math.h>
using namespace std;
 
int main()
{
	int t;
	cin >> t;
	for (int i=0; i<t; i++) {
		int n;
		cin >> n;
		
		long int res = (pow(2,n-1)*((n*(n+1))/2));
		cout <<res<<endl;
	}
    return 0;
}
