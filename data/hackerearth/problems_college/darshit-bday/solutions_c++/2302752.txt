#include <iostream>
using namespace std;
typedef long long ll;
int main()
{
    ll t,b,w,x,y,z,s1,s2;
    scanf("%lld",&t);
    while(t--){
    	scanf("%lld%lld%lld%lld%lld",&b,&w,&x,&y,&z);
    	s1=b*x+w*y;
    	ll q=min(x,y);
    	ll y1=(max(x,y)==x)?b:w;
    	s2=(b+w)*q+y1*z;
    	
    	printf("%lld\n",min(s1,s2));
    	
    }
    return 0;
}
