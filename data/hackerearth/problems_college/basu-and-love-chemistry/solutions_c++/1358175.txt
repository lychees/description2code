#include <iostream>
# include<bits/stdc++.h>
using namespace std;

long long SumSubstring(char s[])
{
    long long sum = 0, mult = 1;
    for (int i = strlen(s); i > 0; i--, mult = (mult * 10)%1000000007 + 1)
        sum += (((s[i - 1] - '0') * mult)%1000000007 * i) % 1000000007;
    return sum % 1000000007;
}

int main()
{
    char s[2000000];
    cin>>s;
    cout<<SumSubstring(s)<<endl;
    return (0);
}
