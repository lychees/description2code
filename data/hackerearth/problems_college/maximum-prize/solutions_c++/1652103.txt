#include <iostream>
#include<algorithm>
#define ll long long
using namespace std;

int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		ll n,k,sum=0;
		cin>>n>>k;
		ll a[n];
		for(ll i=0;i<n;i++) cin>>a[i];
		sort(a,a+n);
		for(ll i=n-1;i>=n-k && i>=0;i--) sum+=a[i];
		cout<<sum<<endl;
	}
	
	
	
    return 0;
}
