#include<bits/stdc++.h>
using namespace std;
int A[1001][1001];
int main()
{
    int T=0,N,M,ans,i,j,k;
    char c;

    scanf("%d",&T);
    while(T--)
    {
        scanf("%d %d",&N,&M);
        c=getchar();
        for(i=0;i<N;i++)
        {
            for(j=0;j<M;j++)
            {
                c=getchar();
                //printf("%c\n",c);
                if(c=='\\')
                    A[i][j]=1;
                else if(c=='/')
                    A[i][j]=0;
                c=getchar();
            }
        }
        /*for(i=0;i<N;i++)
        {
            for(j=0;j<M;j++)
                printf("%d ",A[i][j]);
            printf("\n");
        }*/
        ans=0;
        for(i=0;i<N-1;i++)
        {
            for(j=0;j<M-1;j++)
            {
                if(A[i][j]==0&&A[i][j+1]==1&&A[i+1][j]==1&&A[i+1][j+1]==0)
                    ans++;
            }
        }
        printf("%d\n",ans);
    }
}
