#include <iostream>
using namespace std;
// resubmitting
int main()
{
    int t,n,a,b,c;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int ans=0;
    	for(int i=0;i<n;++i)
    	{
    		cin>>a>>b>>c;
    		ans=ans^a^b^c;
    	}
    	if(ans!=0) cout<<"Stark"<<endl;
    	else cout<<"Lannister"<<endl;
    }
    return 0;
}
