#include <bits/stdc++.h>
#define ll long long
using namespace std;
int dp[100010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	string s;
	cin>>s;
	int q;
	cin>>q;
	int i,x,y,l=s.length();
	dp[0]=0;
	for(i=1;i<l;i++)
	{
		if(s.at(i)==s.at(i-1))
		{
			dp[i]=dp[i-1]+1;
		}
		else
		dp[i]=dp[i-1];
	}
	for(i=0;i<q;i++)
	{
		cin>>x>>y;
		x--;y--;//zero-based index
		cout<<dp[y]-dp[x]<<endl;
	}
	return 0;
}
