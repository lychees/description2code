#include<bits/stdc++.h>
using namespace std;
static int fl[30][31];
int n;
bool top_down(int cur,int rem[26]){
	bool f1 = true;
	for(int i=0;i<26;i++){
		if(rem[i]!=0)
		    f1 = false;
	}
	if(f1)
	   return true;
	if(cur==n){
		for(int i=0;i<26;i++)
		 if(rem[i]>0)
		 	return false;
		return true;
	}
	bool f = true;
	for(int i=0;i<26;i++){
		if(fl[cur][i]>rem[i])
			f = false;
	}
	bool ans=false;
	if(f){
		for(int i=0;i<26;i++)
		    rem[i]-=fl[cur][i];
		ans |= top_down(cur+1,rem);
		for(int i=0;i<26;i++)
			rem[i]+=fl[cur][i];
	    if(ans)
	    	return true;
	}
	return (top_down(cur+1,rem));
}
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
	//	int n;
		scanf("%d",&n);
		for(int i=0;i<n;i++){
			char mohib[31];
			scanf("%s",mohib);
			for(int j=0;j<26;j++)
				fl[i][j]=0;
			for(int j=0;mohib[j];j++)
				fl[i][mohib[j]-'a']++;
		}
		char name[901];
		scanf("%s",name);
		int a[26];
		for(int i=0;i<26;i++)
			a[i]=0;
		for(int i=0;name[i]!='\0';i++)
			a[name[i]-'a']++;
		puts(top_down(0,a)?"YES":"NO");
	}
	return 0;
}
