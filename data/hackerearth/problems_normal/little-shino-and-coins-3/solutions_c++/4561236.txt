#include <iostream>
#include <cstdlib>
#include <cstdio>
using namespace std;

int main()
{
	string str;
	int k;
	cin>>k;
	cin>>str;
	int counter = 0;
	
	for(int i=0;i<str.length();i++){
		int uniq = 0;
		int hash[27] = {0};
		for(int j=i;j<str.length();j++){
			if(i == j){
				uniq++;
				hash[int(str[j])-97] = 1;
			}else{
				
				if(hash[int(str[j])-97] == 0){
					//its unique
					uniq++;
					hash[int(str[j])-97] = 1;
				}
			}
			// cout<<i<<" i "<<j<<" j "<<" un-> "<<uniq<<endl;
			if(uniq == k){
				counter++;
			}
			
		}
	}
	cout<<counter<<endl;
    return 0;
}
