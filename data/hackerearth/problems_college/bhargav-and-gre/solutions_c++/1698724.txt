#include <iostream>
#include <cstring>
using namespace std;

unsigned long SIZE=1000005;
int main() {
	char a[SIZE],b[SIZE];
	cin>>a;
	int d=(int)a[0];
	long testkases,syn=0,ant=0,len1;
	len1=strlen(a);
	cin>>testkases;
	while(testkases--)
	{
		cin>>b;
		int c=(int)b[0];
		if(c==d && len1==strlen(b))
			syn++;
		if(d==(c-3) && len1==strlen(b))
			ant++;
	}
	cout<<syn<<" "<<ant;
}
