#include <iostream>

using namespace std;

int main(int argc, char const *argv[])
{
	int n, m;
	cin>>n>>m;
	char ch;
	if (n & 1)
		ch = '.';
	else
		ch = '#';

	for (int j = 0; j < n; ++j)
	{
		if (n == 1 && m == 1) {
			cout<<"not exists";
			break;
		}
		for (int k = 1; k <= m; ++k)
		{
			cout<<ch;
			if (!(m & 1) && k == m)
				break;
			if (ch == '.')
				ch = '#';
			else
				ch = '.';
		}
		cout<<endl;
	}
	return 0;
}
