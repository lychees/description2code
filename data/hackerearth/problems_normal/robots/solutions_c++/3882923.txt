#include<cstdio>
#include<iostream>
#include<algorithm>
#include<bitset>

using namespace std;

#define MAX 1007
#define MOD 1000000000+7

int mx[MAX+7];

int main(){
	int t,n;
	scanf("%d",&t);
	for(int cas=1;cas<=t;cas++)
	{
		scanf("%d",&n);
		for(int i=0;i<n;i++)
			scanf("%d",&mx[i]);
		sort(mx,mx+n);

		long res = 1LL;
		for(int i=0;i<n;i++){
			res *= (mx[i]-i);
			res %= MOD;
		}
		if(res<0) res = 0;
		cout<<res<<endl;
	}
}
