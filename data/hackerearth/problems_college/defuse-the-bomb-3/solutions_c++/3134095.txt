#include<iostream>
#include<string>
using namespace std;
int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		string a,b,s;
		int i,j,k,la,lb;
		cin >> a >> b;
		la=a.size();
		lb=b.size();
		for(i=0;i<la-lb+1;i++)
		{
			for(j=0;j<lb;j++)
			{
				s=s+a[j+i];
			}
			k=(stoi(s)*stoi(b));
			cout << k << endl;
			s="";
		}
	}
}
