#include <bits/stdc++.h>
#define lli long long int
#define M 1000000007

using namespace std;

void swap(int *a, int *b) {
    int t = *a;
    *a = *b;
    *b = t;
}


lli f(lli n) {
    lli l = 0;
    lli h = 2 * 1000000000;

    while (l < h) {
        lli m = (l + h) / 2;
        lli tmp = (m * (m + 1)) / 2;
        if (tmp >= n) {
            h = m;
        } else {
            l = m + 1;
        }
    }
    return l;
}

int main()
{
    lli n, t, t1, t2, x, y, z;

    cin >> t;
    while (t--) {
        cin >> n;
        cout << f(n) << endl;
    }

    return 0;
}
