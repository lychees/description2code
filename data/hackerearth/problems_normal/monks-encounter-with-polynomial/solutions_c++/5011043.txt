#include <iostream>
#include<cmath>
using namespace std;
#define ll long long int


int main()
{
    int t;
    cin>>t;
        while(t--){
    	ll a,b,c,k,r,i;
    	ll ans=-1;
    	ll lb=0,ub=100000;
    	cin>>a>>b>>c>>k;
        while(lb<=ub){
        	ll mid=(lb+ub)/2;
            r=(a*mid*mid)+(b*mid)+c;
            if(r>=k){
            	ans=mid;
            	ub=mid-1;
            	    }
            	    else{
            	    	lb=mid+1;
            	    }
        } 
       cout<<ans<<endl;
        	  	}
        return 0;
}
