long long int dp[201][201];

long long int part(int n, int k, int low) {


    if ((low <= n && n < 2*low && n <= k) || n == 0)
        return 1;
    if (n < low)
        return 0;
    if(dp[n][k]!=-1)return dp[n][k];
    long long int sum = 0;
    for (int i = low;i <= k; i++) {
        sum += part(n-i,i,low);
    }
//    d(n);d(k);d(sum);en();
    return dp[n][k] = sum;

}
