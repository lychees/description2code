#include<bits/stdc++.h>
#define ull unsigned long long
using namespace std;
ull fib[90];
ull query[90];
ull sum1,sum2;
int main()
{
	fib[1]=1;
	fib[2]=1;
	for(int i=3;i<=80;++i)
	{
		fib[i]=fib[i-1]+fib[i-2];
	}
	int n,m;
	cin>>n>>m;
	for(int i=1;i<=m;++i)	
	{
		cin>>query[i];
		sum2+=query[i];
	}
	for(int i=1;i<=n;++i)
		sum1+=fib[i];
	cout<<sum1-sum2<<endl;

	return 0;
}
