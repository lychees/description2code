#include <bits/stdc++.h>
using namespace std;
vector<int> fa[105];
int dp[505][505];
int hi = 1000000007;
int a[505];
int i,j;
int gcd(int a,int b)
{
	int t;
	while(a%b)
	{
		t=a%b;
		a=b;
		b=t;
	
	}
	return b;
}
int main()
{
    //cout << "Hello World!" << endl;
    int n,m;
    for(n=1;n<=100;n++)
    {
    	for(m=n;m<=100;m+=n)
    	fa[m].push_back(n);
    }
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    scanf("%d",&a[i]);
    
    for(i=1;i<=n;i++)
    {
    	
    	dp[i][a[i]] += 1;
    	
    	for(j=1;j<i;j++)
    	{
    		if(a[j] < a[i])
    		{
    			
    			int x = a[j];
    			int k;
    			for(k=0;k<fa[x].size();k++)
    			{
    				int cur = fa[x][k];
    				int g = gcd(cur,a[i]);
    				dp[i][g] += dp[j][cur];
    				if(dp[i][g]>=hi)dp[i][g] -=hi;
    			}
    		}
    	}
    }
    
    int ans = 0;
    for(i=1;i<=n;i++)
    {
    	ans += dp[i][1];
    	if(ans>=hi)ans-=hi;
    }
    printf("%d\n",ans);
    return 0;
}
