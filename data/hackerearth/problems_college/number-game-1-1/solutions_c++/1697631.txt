#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	int n;
	int c;
	cin>>t;
	while(t--)
	{
		cin>>n;
		c=0;
		while(n > 1)
		{
		if(n%3==0)
		{
			n=n/3;
		}
		else if(n%2==0)
		{
			n=n/2;
		}
		else
		{
			n=n-1;
		}
		c++;
		}
		cout<<c<<'\n';
	}
	return 0;
}
