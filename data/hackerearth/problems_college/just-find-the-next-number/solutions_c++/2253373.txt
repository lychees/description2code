#include<bits/stdc++.h>

typedef long long int LL;

#define scll(x) scanf("%lld",&x);
#define sci(x) scanf("%d",&x);
#define prll(x) printf("%lld\n",x);
#define pri(x) printf("%d\n",x);


/*
	n & (n>>1) = if n would contain any *11* n>>1 would be *01* type & of which will never be 0 
	therefore if its result is 0 th number contains no consecutive 11
*/

int main()
{
	int t,n,x;
	cin>>t;
	while(t--)
	{
		cin>>n;
		x=n;
		while(1)
		{
			x=n;
			if(((x&(x>>1))==0)){
				cout<<n<<endl;
				break;
			}
			n++;		
		}				
	}
	return 0;
}
