#include<bits/stdc++.h>
using namespace std;
#define ll long long unsigned int
int main()
{
	ios_base::sync_with_stdio(false);
    ll t,n,i;
    cin>>t;
    while(t--)
    {
    	
    	cin>>n;vector<ll>a(n);
    	for(i=0;i<n;i++)
    	{
    		
    		cin>>a[i];
    		
    	}
    	sort(a.begin(),a.end());
    	
    	a.erase( unique( a.begin(), a.end() ), a.end() );

    	cout<<a.size()<<"\n";
    	
    }
    return 0;
}
