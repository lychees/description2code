#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	cin >> t;
	for (int i = 0; i < t; i++)
	{
		string reserve, s;
		cin >> s >> reserve;
		string ::iterator it;
		int pos = 0;
		sort(reserve.begin(), reserve.end());
		for (it = s.begin(); it != s.end(); ++it)
		{
			string ::iterator is;
			for (is = reserve.begin(); is != reserve.end(); ++is)
			{
				if ((int)*it > (int)*is)
				{
					s.at(pos) = *is;
					reserve.erase(is);
					break;
				}
			}
			pos++;
		}
		cout << s << endl;
	}
}
