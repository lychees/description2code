from collections import defaultdict
for T in range(input()):
    n = input()
    fi = defaultdict(int)
    item = 0
    for i in range(n):
        x, y = map(int, raw_input().split(' '))
        fi[x] += 1
        if fi[y] != 0:
            fi[y] -= 1
        else:
            item += 1
    print item
