#include<bits/stdc++.h>
#define N 999999
using namespace std;
typedef long long int ll;
void swapping(ll *a ,ll *b)
{
    ll c=*a;
    *a=*b;
    *b=c;
}
ll partion(ll a[],ll s,ll e)
{

    ll pivot=a[e],i=s-1,j=0;
    for(j=s;j<=e-1;j++)
    {
        if(a[j]<=pivot)
        {
            i++;
            swapping(&a[i],&a[j]);
        }
    }
    swapping(&a[i+1],&a[e]);
    return (i+1);
}
void quicksort(ll a[],ll s,ll e)
{
    if(s<e){
    ll p=partion(a,s,e);
    quicksort(a,s,p-1);
    quicksort(a,p+1,e);
    }
}
bool binarysearch(ll a[],ll l,ll r,ll w)
{
	if(l<=r)
	{
		ll  m=(l+r)/2;
		if(a[m]==w)  return true;
		else if(w<a[m])   binarysearch(a,l,m-1,w);
		else  binarysearch(a,m+1,r,w);
	}
	else  return false;
}
int main()
{
	ll n,a[N],i,j,m,w; cin>>n>>m; for(i=0;i<n;i++) cin>>a[i];
	quicksort(a,0,n-1);
	while(m--)
	{
		cin>>w;
		if(binarysearch(a,0,n-1,w)==true)  printf("YES\n");  else printf("NO\n");
	}
	
}
