#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
int main()
{
    //cout << "Hello World!" << endl;
    int t,n,i,a[1000001];
    a[0]=1;
    a[1]=2;
    for(i=2;i<1000001;i++){
    	a[i]=(a[i-1]+a[i-2])%mod;
    }
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cout<<a[n-1]%mod<<endl;
    }
    
    return 0;
}
