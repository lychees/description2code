#include <bits/stdc++.h>
using namespace std;

struct st{
	int x,y;
	bool operator<(const st &p) const{
		if(x==p.x) return y>p.y;
		return x<p.x;
	}
}a[1000010];
int main()
{
	int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
    	scanf("%d%d",&a[i].x,&a[i].y);
    }
    sort(a,a+n);
    for(int i=0;i<n;i++)
    {
    	printf("%d %d\n",a[i].x,a[i].y);
    }
    
    return 0;
}
