#include<bits/stdc++.h>
using namespace std;
#define in(x) scanf("%lld",&x)
#define mod 1000000007
typedef long long int ll;
int main()
{
    ll n,i,d,ans,t,x,y,a,b,c;
    in(n),in(d);
    vector<pair<ll,pair<ll,ll> > > v;
    for(i=0;i<n;i++)
    {
        in(t),in(x),in(y);
        v.push_back(make_pair(t,make_pair(x,y)));
    }
    sort(v.begin(),v.end());
    t=v[0].first,i=0,y=0;
    while(i<n && t==v[i].first)
    {
        x=v[i].second.second;
        if(x>y)
            y=x,ans=v[i].second.first;
        i++;
    }
    while(i<n && d>0)
    {
        a=v[i].first;
        b=v[i].second.first;
        c=v[i].second.second;
        if(c>y)
        {
            d=d-y*(a-t);
            t=a,y=c;
            if(d<=0)
                break;
            else
                ans=ans+b;
        }
        i++;
    }
    cout << ans << endl;
    return 0;
}
