#include<bits/stdc++.h>
using namespace std;

int arr[5];
bool fun(int index, int no)
{
    if(index==5)
    {
        //cout<<no<<endl;
        if(no==23)
            return true;
        return false;
    }
    else
    {
        if(fun(index+1, no+arr[index])==true)
            return true;
        if(fun(index+1, no-arr[index])==true)
            return true;
        if(fun(index+1, no*arr[index])==true)
            return true;
        return false;
    }
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        for(int i=0; i<5; i++)
            cin>>arr[i];
        sort(&arr[0], &arr[5]);
        int flag=0;
        while(next_permutation(&arr[0], &arr[5]))
        {
            if(fun(1,arr[0])==true)
            {
                flag=1;
                break;
            }
        }

        if(flag==1)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    return 0;
}
