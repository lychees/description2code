#include <bits/stdc++.h>
using namespace std;


int main()
{
  
        long long n;
        cin >> n;
        long long a[n];
        cin >> a[0];
        long long cur_sum = a[0],max_sum = a[0];
        for(long long i=1 ; i<n ; i++)
        {
			cin >> a[i];
			cur_sum = max(cur_sum+a[i],a[i]);
			max_sum = max(cur_sum,max_sum);
		}
		cout << max_sum;
}
