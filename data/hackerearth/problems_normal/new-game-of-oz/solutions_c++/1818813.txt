#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int t,n,i,b,count=0,elem;
    vector<int> V;
    cin>>t;
    while(t--)
    {
    	count=0;
    	elem=1;
    	cin>>n;
	    for(i=0;i<n;++i)
	    {
	       cin>>b;
	       V.push_back(b);	
	    }
	    sort(V.begin(),V.end());
	    for(i=0;i<n-1;++i)
	    {
	       if(V[i]+1==V[i+1])
	       {
	       	 
	       	 elem++;
	       }
	       else
	       {
	       	  //cout<<"elem no. "<<elem<<endl;
	       	  count=count+((elem+1)/2);
	       	  elem=1;
	       }
	       
	    }
	    count=count+((elem+1)/2);
	    cout<<count<<endl;
	    V.clear();
	    
    }
    return 0;
}
