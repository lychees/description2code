#include <iostream>
using namespace std;
#define pi(x) printf("%d ",x)
#define si(x) scanf("%d",&x)
void hit(int A[],int n,int x)
{
    int i;
    for(i=0;i<n;i++)
    {
        if(A[i]>x)
            A[i]--;
    }
}
int main(int argc, const char * argv[])
{

    int n,m,x,A[100000],i;
    si(n);
    for(i=0;i<n;i++)
    {
        si(A[i]);
    }
    si(m);
    while(m--)
    {
        si(x);
        hit(A,n,x);
    }
    for(i=0;i<n;i++)
    {
        pi(A[i]);
    }
    return 0;
}
