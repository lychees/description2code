#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define lli long long int
long int md=1000000007;
int arr[1000005];
lli f[1000005];
void pre()
{
	lli i;
	f[0]=1;
	f[1]=1;
	for(i=2;i<=1000000;i++)
	{
		f[i]=f[i-1]*i;
		f[i]%=md;
	}
}
int main()
{
    int t,n,m,i,j,k,fm;
    lli ans;
    cin>>t;
    pre();
    while(t--)
    {
    	cin>>n>>m;
    	fm=0;
    	for(i=1;i<=n;i++)
    	{
    		for(j=1;j<=m;j++)
    		{
    			k=i|j;
    			fm=max(fm,k);
    			arr[k]++;
    		}
    	}
    	//cout<<fm<<endl;
    	ans=1;
    	for(i=1;i<=fm;i++)
    	{
    		ans=ans*f[arr[i]];
    		ans%=md;
    		arr[i]=0;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
