#include <string.h>
#include <iostream>
#define SIZE 100010
using namespace std;

int cnt[SIZE];

int main()
{
    int test;
    scanf("%d",&test);
    while(test--) {
    	int n;
    	scanf("%d",&n);
    	memset(cnt,0,sizeof(cnt));
    	for(int i=0; i<n; i++) {
    		int num;
    		scanf("%d",&num);
    		cnt[num]++;
    	}
    	
    	int ans= 0 ;
    	for(int i=0; i<SIZE; i++)
    		if(cnt[i]) ans++;
    	cout<<ans<<endl;
    }
    return 0;
}
