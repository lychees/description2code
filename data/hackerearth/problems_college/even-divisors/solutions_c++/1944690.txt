#include <iostream>
using namespace std;

long long factorise(long long &n, long long q) {
    long long c;
    
    n /= q;
    
    for (c = 1; n % q == 0; ++c, n /= q);
    
    return c;
}

long long count_factors(long long n) {
    long long count = 1;
    
    if (n < 2) {
        return 1;
    }
    
    if ((n & 1) == 0) {
        count *= (factorise(n, 2) + 1);
    }
    
    if ((n % 3) == 0) {
        count *= (factorise(n, 3) + 1);
    }
    
    for (int q = 5; q * q <= n; q += 6) {
        if ((n % q) == 0) {
            count *= (factorise(n, q) + 1);
        }
        
        if ((n % (q + 2)) == 0) {
            count *= (factorise(n, q + 2) + 1);
        }
    }
    
    if (n > 1) {
        count *= 2;
    }
    
    return count;
}
