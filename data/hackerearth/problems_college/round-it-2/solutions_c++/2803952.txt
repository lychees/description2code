int solve(){
    int n, b, x, ans;
    cin>>n>>b;
    x = n / b;
    if((n - x * b) < ((x + 1) * b - n) ){
        ans = x * b;
    }
    else{
        ans = (x + 1) * b;
    }
    cout<<ans<<endl;
    return 0;
}
