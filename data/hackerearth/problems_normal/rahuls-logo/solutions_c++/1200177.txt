ll f(ll n, ll st, ll en)
{
	
	
	
	
	ll mid; 
	
	while(st<en)
	{
		mid = (st+en)/2;
		
		if( (4*(mid+1)*(mid+2)) <= n )
		{
			st = mid + 1;
		}
		else
		{
			en = mid;
		}
			
	}
	
	
	return st-1;
	
	
}

ll g(ll n, ll st, ll en)
{
	
	
	
	
	ll mid; 
	
	while(st<en)
	{
		mid = (st+en)/2;
		
		if( (4*(mid+1)*(mid+3)) <= n )
		{
			st = mid + 1;
		}
		else
		{
			en = mid;
		}
			
	}
	
	
	return st-1;
	
	
}
