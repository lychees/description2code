    #include <stdio.h>
    int main()
    {
    	long long int n,i,j,temp;
    	scanf("%lld",&n);
    	long long int a[n],b[n],c[n];
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld %lld",&a[i],&b[i]);
    	}
    	for(i=0;i<n;i++)
    	{
    		c[i]=a[i]+b[i];
    	}
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<n-1-i;j++)
    		{
    			if(c[j]>c[j+1])
    			{
    				temp=c[j];
    				c[j]=c[j+1];
    				c[j+1]=temp;
    			}
    		}
    	}
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<n;j++)
    		{
    			if(a[j]+b[j]==c[i])
    			{
    				a[j]=0;
    				b[j]=0;
    				printf("%lld ",j+1);
    				break;
    			}
    		}
    	}
    	return 0;
    }
