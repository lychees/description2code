#include <iostream>
using namespace std;

int main()
{
    long int t;
    cin>>t;
    int x,y,n;
    long int arr[52];
    long sum=0;
    for(long int i=0;i<t;i++)
    {
    	cin>>x>>y>>n;
    	for(int j=0;j<n;j++)
    	{
    		if(j==0)
	    	{
	    		for(int k=0;k<x;k++)
	    		{
	    			arr[k]=y;
	    			if(k==x-1)
	    				j=k;
	    		}
	    		continue;
	    	}
    		sum=0;
    		for(int p=1;p<=x;p++)
    			sum=sum+arr[j-p];
    		arr[j]=sum;
    	}
    	cout<<arr[n-1]<<endl;
    }
    return 0;
}
