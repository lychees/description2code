#include <bits/stdc++.h>
using namespace std;

int main() {
	int t, n, m, k, x, y, a, b;
	size_t cnt;
	cin >> t;
	while(t--){
		x=1;y=1;cnt=0;
		cin >> n >> m;
		cin >> k;
		while(k--) {
			cin >> a >> b;
			while(1) {
				x+=a;
				y+=b;
				if (x>=1 && x<=n && y>=1 && y<=m) {
					cnt++;
				}
				else {
					x-=a;
					y-=b;
					break;
				}
			}
		}
		cout << cnt << endl;
	}
	return 0;
}
