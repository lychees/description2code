typedef long long int ll;

int n, k, v[mx];
int main(){
//    ios_base::sync_with_stdio(0);
//    cin.tie(0);
//    freopen("data.in","r",stdin);
    scanf("%d%d", &n, &k);
    for(int i = 0; i < n; i++){
        scanf("%d", &v[i]);
    }
    sort(v, v + n);
    int ans = 1e9 + 10;
    for(int i = 0; i < n - k; i++){
        ans = min (ans, v[i + k - 1] - v[i]);
    }
    printf("%d", ans);
    return 0;
}
