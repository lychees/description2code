#include<iostream>

using namespace std;

 unsigned long long int fac(int n)
{
        if(n == 1)
                return 1;
        return (n * fac(n - 1))%10;
}
int main()
{
        int tc, n, m, i, final;
        unsigned long long int val;
        cin >> tc;
        while(tc--)
        {
                cin >> n >> m;
                if( n >= 5)
                        final = 0;
                else{
                val = fac(n);
               // cout << val << endl;
                final = val%10;
                for(i = n + 1; i <= m; i++)
                {
                        val = val * i;
                        final += val % 10;
                        final %= 10;
                }
        }
                cout << final%10 << endl;
        }
        return 0;
}
