#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int t,c,c1,max,max1,i,j;
	string s;
	cin>>t;
	int a[t];
	c1=0,c=0,max=0,max1=0;
	for(j=0;j<t;j++){
		cin>>s;
		c=0,max=0;
		for(i=0;i<s.length();i++){
			if(s[i]=='C'){
				++c;
				if(c1>0){
					++c1;
					if(max1<c1)
					max1=c1;
				}
			}
			else
			c=0,c1=0;
			if(max<c)
			max=c;
		}
		a[j]=max;
		if(c1==0&&c>0)
		c1=c;
	}
	sort(a,a+t);
	cout<<a[t-1]<<" ";
	if(max1<a[t-1])
	max1=a[t-1];
	cout<<max1;
	return 0;
}
