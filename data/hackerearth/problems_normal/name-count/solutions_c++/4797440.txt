
#include<bits/stdc++.h>
using namespace std;
typedef unsigned long long int lu;
#define  od 1000000007
int main()
{
	lu t,n,k,ans,i;
	scanf("%llu",&t);
	while(t--)
	{
		scanf("%llu%llu",&n,&k);
		ans=1;
		for(i=k;i>k-n;i--)
		ans=(ans*i)%od;
		printf("%llu\n",ans);
	}
  return 0;
}
