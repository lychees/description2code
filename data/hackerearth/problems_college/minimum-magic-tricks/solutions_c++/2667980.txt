#include <bits/stdc++.h>
using namespace std;

#define mod 1000000007

int main() {
    long long dp[1001];
    dp[1] = 1;
    dp[2] = 1;
    for (int i = 3; i <= 1000; i++) {        
        dp[i] = (dp[i-1] + (2*dp[i-2]%mod))%mod;
        
    }
	int t;
	cin >> t;
	while (t--) {
	    int n;
	    cin >> n;
	    cout << dp[n] << endl;
	}
	return 0;
}
