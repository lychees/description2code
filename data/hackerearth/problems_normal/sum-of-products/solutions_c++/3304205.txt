#include<iostream>
#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;

ll sod(ll n)
{
	ll y=0;
	ll res=n,t,tp;
	for(ll i=2;i<=sqrt(n);i++)
	{
		y=n/i;
		res+=y*i;
		res%=mod;
	
	}
	for(ll i=sqrt(n)+1;i<=n/2;){
		y=n/i;
		t=n/y;
		tp=(t*(t+1)/2-(i-1)*i/2)%mod;
		t=(tp*y)%mod;
		res+=t;
		res%=mod;
		i=(n/y)+1;
	}
	ll p=(n/2);
	ll sum=(n*(n+1)/2-p*(p+1)/2)%mod;
	//cout<<sum;
	res+=sum;
	
 return res%mod;
}
int main()
{
	ll t;
	ll n;
	cin>>t;
	while(t--)
	{
		cin>>n;
	cout<<sod(n)<<endl;
	}
  return 0;
}
