#include <bits/stdc++.h>
#define ll long long int
#define mod (int)(1e9 + 7)

using namespace std;

int t, n, q, l, r;
int a[100005], sum[100005];

int main()
{
	cin >> n;	
	memset(sum, 0, sizeof(sum));
	
	for (int i = 1; i <= n; i++) {
		cin >> a[i];
		sum[i] += sum[i - 1] + a[i];
	}
	
	cin >> q;
	while (q--) {
		cin >> l >> r;
		cout << sum[r] - sum[l - 1] << endl;	
	}
	return 0;
}
