#include <bits/stdc++.h>
using namespace std;
char s[1000003];
vector<long long> v1,v2;
long long f ( char c, int x){
	int l = strlen(s);
	int i=0, k;
	v1.clear();
	v2.clear();
	long long cnt=0;
	long long p=0,q=0;
	for ( i = 0; i < l; i++ ){
		cnt++;
		if ( s[i] == c ){
			p++;
		v1.push_back(cnt);
		cnt = 0;
		}
		else q++;
	}
	cnt=0;
	for ( i = l - 1; i >= 0; i-- ){
		cnt++;
		if ( s[i] == c ){
			v2.push_back(cnt);
			cnt=0;
		}
	}
	reverse(v2.begin(),v2.end());
	int sz = v1.size();
	i=0;
	long long ans = 0;
	if (x>=1){
	while ( i + x <= sz ){
		ans = ans + v1[i]*v2[i+x-1];
		i++;
	} 
	return ans;
	}
	else {
		long long h=0;
		for(i=0;i<l;i++)
		{
			if(s[i]==c)
			{
				ans+=(h*(h+1))/2;
				h=0;
			}
			else h++;
		}
		ans+=(h*(h+1))/2;
		return ans;
	}
}
int main()
{
	int t,m,n, i, k;
	cin >> t;
	while ( t-- ){
		scanf("%d%d",&m,&n);
		scanf("%s",s);
		printf("%lld %lld\n",f('R',m), f('K',n));
	}
    //cout << "Hello World!" << endl;
    return 0;
}
