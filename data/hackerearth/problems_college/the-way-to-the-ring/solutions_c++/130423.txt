#include<iostream>

using namespace std;

long long int test,n,m,i,j,mat[1000][1000],arr[1000],x,flag;

int main()
{

	cin>>test;
	while(test--)
	{
		cin>>n>>m;
		
		mat[0][0]=0;
		for(i=1;i<=m;i++)
		{
			cin>>arr[i];
			mat[0][i]=0;
		}

		for(i=1;i<=n;i++)
		{
			mat[i][0]=0;
			for(j=1;j<=m;j++)
			{
				if(i>arr[j])
				{
					flag=1;

					if(mat[i-arr[j]][m])
					{
						mat[i][j]=mat[i-arr[j]][m]+1;
				
						if((j-1)&&(mat[i][j-1])&&(mat[i][j-1]<mat[i][j]))
							mat[i][j]=mat[i][j-1];
				
					}
					else
						mat[i][j] = mat[i][j-1];

				
				}
				else if(i==arr[j])
					mat[i][j]=1;
				else
					mat[i][j]=mat[i][j-1];

			}
		}

		
		if(mat[n][m])
			cout<<mat[n][m]<<endl;
		else
			cout<<-1<<endl;

	}
	return 0;

}
