#include <iostream>
#include<cstdio>
using namespace std;
int a[100005];
int main()
{
	int t,n,i,ans;
	scanf("%d",&t);
	while(t--)
	{
		ans=0;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		scanf("%d",&a[i]);
		for(i=0;i<n;i++)
		a[i]=a[i]%5;
		for(i=0;i<n;i++)
		ans=ans^a[i];
		if(ans)
		printf("Aniruddha\n");
		else
		printf("Andrew\n");
	}
    return 0;
}
