#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,i,j,n,k,l,m,p;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	l=p=n;
    	bool ch=false;
    	for(i=0;i<=n;i++)
    	{
    		for(j=0;j<=n;j++)
    		{
    			k=i*6+j*8;
    			if(k==n)
    			{
    				l=i+j;
    				ch = true;
    				break;
    			}
    		}
    	}
    	for(i=0;i<=n;i++)
    	{
    		for(j=0;j<=n;j++)
    		{
    			m=i*8+j*6;
    			if(m==n)
    			{
    				p=i+j;
    				ch = true;
    				break;
    			}
    		}
    	}
    	if(ch)
    		cout<<min(l,p)<<"\n";
    	else
    		cout<<"-1\n";
    }
    return 0;
}
