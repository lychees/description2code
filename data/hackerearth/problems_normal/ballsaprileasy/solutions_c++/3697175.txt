#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long n=0;
    cin>>n;
    long sum=0;
    long a[n];
    for(long i=0; i<n; i++)
    {
    	cin>>a[i];
    	
    }
    sort(a,a+n);
   	long x=0;
   	long square=0;
    for(long i=1; i<=n; i++)
    {
    	square+=2*x +1;
	    
    	while(a[i-1]>square)
    	{
    		sum+=square;
    		x++;
    		square+=2*x+1;
    	};
    	if(a[i-1] <= square)
    	{
    		sum+=(square-a[i-1]);
    		x++;
    	}
    }
    cout<<sum<<endl;
    return 0;
}
