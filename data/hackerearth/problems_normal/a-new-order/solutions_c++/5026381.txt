#include<bits/stdc++.h>
using namespace std;
#define forn(i, n)  for(int i = 0;i<n;i++)
#define for1(i, n)  for(int i = 1;i<=n;i++)
#define FOR(i, a, b) for(int i = a;i <= b;i ++)
#define last __last
int g[26][26];///do thi nho hon
vector<int> ad[26];

string a[105];
bool exist[26];

void xep(int l,int r,int id)
{
    int start[26], last[26];
    memset(last, 0, sizeof last);
    memset(start,0,sizeof start);

    forn(i, 26)
    {
        FOR(row, l, r)
        {
            if(id > a[row].size() - 1) continue;
            {
                if(a[row][id] == char(i + 'a'))
                {
                    exist[i] = true;
                    if(start[i] == 0) start[i] = row;
                    last[i] = row;
                }
            }
        }
    }
    forn(i, 26)
     forn(j, 26)
     {
         if(i == j) continue;
         if(start[i] == 0 || start[j] == 0) continue;
         if(start[i] > last[j]) g[i][j] = 1;
     }

    for(int i = 0; i < 26;i ++)
    {
      //  if(i == 3 && id == 3) cout<< start[i] << " ss " <<last[i] <<endl;
        if(start[i]==0) continue;

        xep(start[i], last[i], id + 1);
    }
}


int col[26];
vector<int> topo;

int stt[26];
vector<vector<int> > ans;
int counter[26];
void sort_rank()
{
     ///
     vector<int> cur;
     forn(i, 26)
     {
         forn(j, 26) if(g[i][j] == 1) counter[i] ++;
         if(counter[i] == 0 && exist[i] == true)
                cur.push_back(i);
     }
     while(1)
     {
         if(cur.size() == 0) break;
         sort(cur.begin(),cur.end());
         ans.push_back(cur);

         vector<int> ncur;

         forn(i, cur.size())
         {
             forn(j, 26)
             {
                 if(g[j][cur[i]] == 1)
                 {
                     g[j][cur[i]] = 0;
                     counter[j]--;
                     if(counter[j] == 0)
                     {
                         ncur.push_back(j);
                     }
                 }
             }
         }
         cur = ncur;
     }
}

int main()
{
   //freopen("test.inp","r",stdin);
   ios_base:: sync_with_stdio(false);
    int n;
    cin >> n;
    for1(i, n) cin >> a[i];
    memset(exist, false, sizeof exist);
    xep(1, n, 0);
    sort_rank();

    forn(i, ans.size())
    {
        forn(j, ans[i].size()) cout<< char(ans[i][j] + 'a');
        cout<< endl;
    }
}
