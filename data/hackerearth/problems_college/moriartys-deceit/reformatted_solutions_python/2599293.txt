def cop(a, b):
    if (a < b):
        a, b = b, a
    while (b != 0):
        t = a
        a = b
        b = t % b
    if a == 1:
        return 1
    else:
        return 0


for _ in xrange(input()):
    n, x = map(int, raw_input().split())
    p = map(int, raw_input().split())
    q = []
    for i in p:
        if (cop(i, x) == 1):
            q.append(i)
    print ' '.join([str(i) for i in q])
