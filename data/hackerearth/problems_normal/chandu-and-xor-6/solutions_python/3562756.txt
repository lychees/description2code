import itertools

N,M = map(int, raw_input().split())
# print N,M
Matrix = [[0 for x in range(M)] for x in range(N)]
for x in range(N):
    Matrix[x] =  map(int, raw_input().split())

row_sum = []
col_sum = []
for x in range(N):
    sum=0
    for y in range(M):
        # print Matrix[x][y]
        sum = sum + Matrix[x][y]
    col_sum.append(sum)
col_sum = set(col_sum)

for x in range(M):
    sum=0
    for y in range(N):
        # print Matrix[x][y]
        sum = sum +Matrix[y][x]
    row_sum.append(sum)
row_sum = set(row_sum)

# print row_sum,col_sum
c = list(itertools.product(row_sum,col_sum))
# print c
output = 0
for item in c:
    xor = item[0] ^ item[1]
    if  output < xor:
        output = xor
print output
