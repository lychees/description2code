#include<iostream>
#include<math.h>
using namespace std;
#define max 1000010
long long a[max];
void sieve(long long x,long long y)
{
	a[0]=0;
	a[1]=0;
	long sum=0;
	for(int i=2;i<=max;i++)
	{
	a[i]=1;
	}
	
	for(int i=2;i<=sqrt(max);i++)
	{
		if(a[i]==1)
		{
			for(int j=i;i*j<=max;j++)
			{
				a[i*j]=0;	
			}
		}
	}
	for(int i=x;i<=y;i++)
	{
		if(a[i]==1)
		{
		sum=sum+i;
		}
	}
	cout<<sum<<endl;
}
int main()
{
	int t;
	long long x,y;
	cin>>t;
	while(t--)
	{
		cin>>x>>y;
		{
		sieve(x,y);
		}
	}
	
}
