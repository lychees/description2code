#include<bits/stdc++.h>
using namespace std;


int A[10000];
int main(){
	int M2=0;
	string s;
	getline(cin, s);

	for(int i=0;i<s.size();i++){
		if(s[i]==' ') M2++;
		A[s[i]-'!']++;
	}
	

	int a=0,M=0;
		for(int i=0;i<1000;i++){
			if(M<A[i]){
				M=A[i];
				a=i;
			}
		}
	char c=a+'!';
	if(M<=M2) cout << "  " << M2 << endl; 
	else cout << c << " " << M << endl;
	
	
	return 0;
}
