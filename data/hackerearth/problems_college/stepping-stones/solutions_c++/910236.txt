#include <iostream>

using namespace std;

#define MOD 1000000007;
int n,a,b,k,s[5005][5005];

int abs(int a)
{
  if(a<0)
    return -a;
  else
    return a;
}
int mod(int a)
{
  while(a>0)
    a-=MOD;
  while(a<0)
    a+=MOD;
  return a;
}
int main()
{
  cin>>n>>a>>b>>k;
  for(int i=1;i<=n;i++)
    s[i][k]=i;
  for(int j=0;j<=k;j++)
    s[0][j]=0;
  
  for(int j=k-1;j>=0;j--)
    for(int x=1;x<=n;x++)
    {
      int d=abs(x-b)-1;
      s[x][j]=s[x-1][j]+mod((s[min(x+d,n)][j+1]-s[max(x-d-1,0)][j+1]));
      s[x][j]=mod(s[x][j]);
      s[x][j]=s[x][j]-mod((s[x][j+1]-s[x-1][j+1]));
      s[x][j]=mod(s[x][j]);
    }
  cout<<mod(s[a][0]-s[a-1][0])<<endl;
}
