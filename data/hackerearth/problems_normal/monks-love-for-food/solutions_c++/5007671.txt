#include <bits/stdc++.h>
using namespace std;


class Solution{
private:
	int Q;
	stack <int> MyStack;
public:
	
	void set_value(){
		cin>>Q;
	}
	void ans(){
		for(int i=0;i<Q;++i){
			int a;cin>>a;
			if(a==1){
				if(MyStack.size()==0) cout<<"No Food"<<endl;
				else{
					cout<<MyStack.top()<<endl;
					MyStack.pop();
				}
			}
			if(a==2){
				int c;
				cin>>c;
				MyStack.push(c);
			}
		}
	}
};

int main(){
	Solution S;
	S.set_value();
	S.ans();
}
