#include<iostream>
#include<algorithm>
#include<vector>
#include<string.h>
#include<stdio.h>
#include<math.h>

using namespace std;

int main()
{
    long long k,n,t;
    cin>>t;

    while(t--) {

    cin>>k>>n;

    if(k>n) {
        printf("%.9lf\n",1.0);
        continue;
    }
    long long num=n;
    double x=1,p=1.0;

    for(int i=1;i<=k;i++) {
        p=p*num;
        p  = p/n;
        num--;
        x *= 10;
    }

    double ans = 1.0-p;
    printf("%0.9lf\n",ans);
    }

    return 0;
}
