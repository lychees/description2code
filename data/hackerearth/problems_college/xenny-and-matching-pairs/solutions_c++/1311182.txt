#include <bits/stdc++.h>
#include <set>
using namespace std;
int main() {
    int t,n,i;
    
    scanf("%d",&t);
    while(t--)
    {
    	set<int> s;
    set<int>:: iterator it;
    	scanf("%d",&n);
    	while(n--)
    	{
    		scanf("%d",&i);
    		s.insert(i);
    		it=s.find(i);
    		++it;
    		if(it!=s.end())
    		  s.erase(it);	
    	}
    	printf("%d\n",s.size());
    }
             
    return 0;
}
