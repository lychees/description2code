#include <iostream>
#include<vector>

using namespace std;
int main()
{
    int test,n,arr[100000];
    cin>>test;
    while(test--){
    int flag[100000]={0};
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    int i=0;
    while(true)
    {
        if(i==-1){
            cout<<"Win\n";
            break;
        }
        else if(arr[i]==-1){
            cout<<"Win\n";
            break;
        }
        flag[i]=1;
        i=arr[i];
        if(flag[i]==1){
            cout<<"Lose\n";
            break;
        }
    }
   }
}
