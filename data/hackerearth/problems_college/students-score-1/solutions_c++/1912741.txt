#include <bits/stdc++.h>
using namespace std;
int ms(int a[],int n)
{
	int maxsofar=0,maxend=0;
	for(int i=0;i<n;i++)
	{
		maxend=maxend+a[i];
		if(maxend<0)
		maxend=0;
		if(maxsofar<maxend)
		maxsofar=maxend;
	}
	return maxsofar;
}
int main() {
	int n;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)
	cin>>a[i];
	int p=ms(a,n);
	if(p>0)
	cout<<p;
	else
	cout<<"0";
	return 0;
}
