n = raw_input()
sum = 0

for i in n:
    if int(i) == 0 or int(i) == 6 or int(i)==9:
        sum += 6
    elif int(i)==1:
        sum += 2
    elif int(i)==2 or int(i)==3 or int(i)==5:
        sum += 5
    elif int(i)==4:
        sum += 4
    elif int(i) ==7:
        sum += 3
    elif int(i) == 8:
        sum += 7

print sum
