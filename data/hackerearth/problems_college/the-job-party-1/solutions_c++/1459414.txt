    #include <stdio.h>
     
    int main()
    {
    int t;
    scanf("%d",&t);
    while(t--)
    {
    long long int n,k;
    scanf("%lld%lld",&n,&k);
    if(k%2)
    {
    long long int ans=((k+1)/2)*k;
    printf("%lld\n",ans);
    }
    else
    {
    long long int ans=(k/2)*(k+1);
    printf("%lld\n",ans);
    }
    }
    return 0;
    }
