#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin >> n;
    int arr[n];
    int dp[n];
    memset(dp, 0, sizeof dp);
    for(int i = 0; i < n; i++){
    	scanf("%d", &arr[i]);
    }
    dp[0] = arr[0];
    for(int i = 0; i < n; i++){
    	int c = i+1;
    	for (int j = c+c; j <= n; j += c){
    		dp[j-1] = max(dp[j-1], arr[j-1] + dp[i]);
    	}
    	printf("%d ", dp[i]);
    }
    
}
