#include <iostream>
using namespace std;
int min(int x,int y) {
	if(x<=y)
		return x;
	else
		return y;
}

int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    cin>>a[i];
    int ct=0;
    for(int i=0;i<n-1;i++) {
    	int m = min(a[i],a[i+1]);
    	a[i]-=m;
    	a[i+1]-=m;
    }
    for(ct=0;ct<n;ct++) {
    	if(a[ct]!=0)
    	break;
    }
    if(ct==n)
    cout<<"YES"<<endl;
    else
    cout<<"NO"<<endl;
    return 0;
}
