#include <iostream>
using namespace std;


int main()
{
	int n, score;
	int suma;

	suma = 0;
	
	cin >> n;

	int scores[n];
	
	int cont 	= 1;
	int pos 	= 1;
	scores[0] 	= 0;
	while(cont <= n ){
		cin >> score;
		if(score == 0){
			pos--;
		}else{
			scores[pos] = scores[pos-1] + score;
			suma = scores[pos];
			pos++;
		}
		cont++;	
	}
	
	cout << suma << endl;
    return 0;
}
