using namespace std;
#define MIN(a,b) (((a)<(b))?(a):(b))
#define MAX(a,b) (((a)>(b))?(a):(b))
#define FORI(i,a,b) for (int i=(int)a; i<=(int)b; i++)
#define ull unsigned long long
 
int main()
{
	int N;
	ull a, b, da, db, ra, rb, c, cc;
	cin >> N;
	while (N--)
	{
		cin >> a >> b;
		cc = c = 0;
		while (a / 10 != 0 || b / 10 != 0)
		{
			ra = a % 10;
			rb = b % 10;
			if (c + ra + rb > 9)
			{
				cc++;
				c = 1;
			}
			else
				c = 0;
			a /= 10;
			b /= 10;
		}
		ra = a % 10;
		rb = b % 10;
		if (c + ra + rb > 9)
		{
			cc++;
			c = 1;
		}
		else
			c = 0;
		if (cc)
		{
			if(cc == 1)
				cout << cc <<" carry operation"<< endl;
			else
				cout << cc <<" carry operations"<< endl;
		}
		else
			cout << "No carry operation" << endl;
	}
	return 0;
}
