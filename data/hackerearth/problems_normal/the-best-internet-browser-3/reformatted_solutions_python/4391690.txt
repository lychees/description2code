vowels = ['a', 'e', 'u', 'o', 'i']
t = raw_input()
t = int(t)  # nr of test cases
for i in range(t):
    site = raw_input()
    l = len(site)  # total necessary characters
    n = 4  # for '.com'
    for s in list(site[4:-4]):
        if vowels.count(s) == 0:
            n += 1
    print str(n)+'/'+str(l)
