#include <iostream>
using namespace std;
int main(){
	int t,flag,i,count1,count2,smallcount;
	string s;
	cin>>t;
	while(t--){
		count1=0;count2=0;smallcount=0;
		cin>>s;
		for(i=0;i<s.size();i++){
			if(s[i]=='a'|| s[i]=='e' || s[i]=='i' || s[i]=='o' || s[i]=='u' || s[i]=='y'){
				smallcount=0;
				count1++;
			}
			else{
				count2++;
				smallcount++;
				if(smallcount==3)
					break;
			}
		}
		if(smallcount==3 || count2>count1)
			cout<<"hard"<<endl;
		else
			cout<<"easy"<<endl;
	}
	return 0;
}
