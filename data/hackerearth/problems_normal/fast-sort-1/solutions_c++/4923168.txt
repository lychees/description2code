#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;
typedef struct pp
{
	char c;
	int p;
}node;
bool func(node a,node b)
{
	return a.p<b.p;
}
int main()
{
    int t;
    cin >> t;
    char temp;
    node tt;
    string s;
    for(int x=0;x<t;++x)
    {
    	int a[26];
    	for(int j=0;j<26;++j)
    	{
    		cin >> temp;
    		a[temp-'a']=j;
    	}
    	cin >> s;
    	//node *n=new node[s.size()];
    	vector<node> v;
    	for(int i=0;i<s.size();++i)
    	{
    		tt.c=s[i];
    		tt.p=a[s[i]-'a'];
    		v.push_back(tt);
    	}
    	sort(v.begin(),v.end(),func);
    	for(int i=0;i<v.size();++i)
    	{
    		cout << v[i].c;
    	}
    	cout << "\n";
    }
    return 0;
}
