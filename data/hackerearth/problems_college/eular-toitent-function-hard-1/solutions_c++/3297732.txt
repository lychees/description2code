#include<iostream>
#include <cstdio>
using namespace std;

int phi[10000005];

int main()
{
  int n = 10000000;
    for (int i=1; i<=n; i++)
        phi[i] = i;

    int p, i;

    for (p=2; p<=n; p++)
    {
        if (phi[p] == p)
        {
            phi[p] = p-1;

            for (i = 2*p; i<=n; i += p)
            {
               phi[i] = (phi[i]/p) * (p-1);
            }
        }
    }
    int cases, query;
  scanf("%d", &cases);
    while(cases--)
    {
        scanf("%d", &query);
        printf("%d\n", phi[query]);
  }
    return 0;
}
