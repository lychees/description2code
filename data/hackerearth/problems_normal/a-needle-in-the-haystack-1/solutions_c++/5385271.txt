#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string s1,s2;
    while(t--)
    {
    	cin>>s1>>s2;
    	int flag=0,k=0,h=0;
    	int ls1=s1.length();
    	int ls2=s2.length();
    	for(int i=0;i<ls2;i++)
    	{
    		if(s2[i]==s1[0])
    		{
    			if(s2[i+1]==s1[1])
    			{
    				k=i+2; h=2;
    				while(s2[k]==s1[h] && h<ls1 && k<ls2)
    				{
    					k++; h++;
    				}
    				if(h==ls1)
    				{
    					flag=1;
    					break;
    				}
    			}
    			else
    			if(s2[i-1]==s1[1])
    			{
    				k=i-2; h=2;
    				while(s2[k]==s1[h] && h<ls1 && k>=0)
    				{
    					k--; h++;
    				}
    				if(h==ls1)
    				{
    					flag=1;
    					break;
    				}
    			}
    		}
    		if(flag==1)
    		break;
    	}
    	if(flag==1)
    	cout<<"YES\n";
    	else
    	cout<<"NO\n";
    }
    return 0;
}
