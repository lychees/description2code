#include <bits/stdc++.h>
using namespace std;

vector<pair<int, int>> v;
vector<pair<int, int>>v_d;

int main()
{
	int n, t, i, j, a, b, l, count;
    cin>>t;
    
    while(t--)
    {
    	v.clear();
    	v_d.clear();
    	cin>>n;
    	for(i=1;i<=n;i++)
    	{
    		cin>>a>>b;
    		v.push_back(make_pair(a,1));
    		v.push_back(make_pair(b,-1));
    	}
    	sort(v.begin(), v.end());
    	
    	l = (int)v.size();
    	for(i=0;i<l;i++)
    	{
    		j = i, count = v[j].second;
    		while(j<l-1 && v[j].first==v[j+1].first)
    		{
    			count += v[j+1].second;
    			j++, i++;
    		}
    		v_d.push_back(make_pair(v[j].first, count));
    	}
    	
    	l = (int)v_d.size();
    	int maxm = 0;
    	count = 0;
    	for(i=0;i<l;i++)
    	{
    		count += v_d[i].second;
    		maxm = max(maxm, count);
    	}
    	cout<<maxm<<endl;
    }
    return 0;
}
