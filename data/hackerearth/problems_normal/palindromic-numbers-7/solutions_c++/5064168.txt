#include "bits/stdc++.h"
using namespace std;
bool isPal(long int n)
{
	long int arr[10];
	long int arrC=0;
	while(n!=0)
	{
		arr[arrC++]=(n%10);
		n=n/10;
	}
	for(long int i=0,j=arrC-1;i<j;i++,j--)
		if(arr[i] != arr[j])
			return 0;
	return 1;
}
int main()
{
	long int tc;
	cin >> tc;
	while(tc--)
	{
		long int a,b;
		cin >> a >> b;
		long int count=0;
		for(long int i=a;i<=b;i++)
			if(isPal(i))
				count++;
		cout << count << endl;
	}
	return 0;
}
