#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	vector <bool> ele(10001, true);
    	int i, k, c=0;
    	for(i=0; i<n; i++){
    		cin>>k;
    		if(ele[k]){
    			c++;
    			ele[k]=false;
    		}
    	}
    	cout<<c<<endl;

    }
    return 0;
}
