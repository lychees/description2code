typedef long long ll;

ll z[100], o[100];

void precalc(){
    z[0] = 0;
    o[0] = 0;
    z[1] = 1;
    o[1] = 0;
    for(int i=2;i<=91;i++){
        z[i] = z[i-1] + o[i-1];
        o[i] = z[i-1];
    }
}

int main(){
    //ios::sync_with_stdio(false);
    #ifdef DEBUG
    freopen("input.txt", "r", stdin);
    #endif // DEBUG

    int T,M;
    precalc();
    scanf("%d", &T);
    while(T--){
        scanf("%d", &M);
        printf("%lld %lld\n", o[M], z[M]);
    }

    return 0;
}
