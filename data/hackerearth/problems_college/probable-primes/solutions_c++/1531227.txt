using namespace std;
long long int ar[1000002]={0},cum[1000002] = {0};
void markMultiples(bool arr[], int a, int n)
{
    int i = 2, num;
    while ( (num = i*a) <= n )
    {
        arr[ num-1 ] = 1;
        ++i;
    }
}
void SieveOfEratosthenes(int n)
{
    if (n >= 2)
    {
        bool arr[n];
        memset(arr, 0, sizeof(arr));
 
        for (int i=1; i<n; ++i)
        {
            if ( arr[i] == 0 )
            {
                ar[i+1] = i+1;
                markMultiples(arr, i+1, n);
            }
        }
    }
}

int main()
{
	SieveOfEratosthenes(1000000);
	for(int i=1;i<1000001;i++)
	{
		if(ar[i]) cum[i] = cum[i-1]+1;
		else cum[i] = cum[i-1];
	}
	//for(int i=0;i<10;i++) cout<<cum[i]<<" ";
	int t;
	cin>>t;
	while(t--)
	{
		int l,r;
		cin>>l>>r;
		int total = 0;
		int req = 0;
		total = r-l+1;
		req = cum[r]-cum[l-1];
		float ans = (float)req/total;
		printf("%0.6f\n",ans);
	}
	return 0;
	
}
