using namespace std;
#define MOD 1000000007
long long a[55][55];
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int T;
	string s;
	int i,n,j;
	cin>>T;
	while(T--)
	{
		cin>>s;
		n=s.size();
		for(i=0;i<n;i++)
			a[i][i]=1;
		for(j=1;j<n;j++)
		{
			for(i=0;i+j<n;i++)
			{
				a[i][i+j]=a[i+1][i+j]+a[i][i+j-1]-a[i+1][i+j-1];
				if(s[i]==s[i+j])
				{
					a[i][i+j]++;
					a[i][i+j]+=a[i+1][i+j-1];
				}
			}
		}
		cout<<a[0][n-1]<<endl;
	}
	//system("pause");
	return 0;
}
