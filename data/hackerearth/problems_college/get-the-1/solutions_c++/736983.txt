#include <iostream>
using namespace std;

int pow (int x)
{
	int a=1;
	for(int i=0; i<x; i++)
		a=a*10;
	return a;
}
	
int sq(int a)
{
	int x=a;
	int sum=0;
	while (x!=0)
	{
		int c=x%10;
		sum=sum+(c*c);
		x=x/10;		
	}
	return sum;
}

int main()
{
    int T;
    cin>>T;
    int cases[1000];
	for (int i=0; i<T; i++)
	{
		cin>>cases[i];
	}
	for (int i=0; i<T; i++)
	{
		while (cases[i]/10)
		{
			cases[i]=sq(cases[i]);
		}
		if (cases[i]==1)
			cout<<"Case #"<<i+1<<":YES\n";
		else cout<<"Case #"<<i+1<<":NO\n";
	}
    return 0;
}
