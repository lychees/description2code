#include<bits/stdc++.h>
#define md 1000000007
using namespace std;
char a[200010];
long long int dp[200010];
int main(){
	cin>>a;
	int x=strlen(a);
	dp[0]=a[0]-'0';
	for(int i=1;i<x;i++){
		dp[i]=(dp[i-1]*10+i*(a[i]-'0')+(a[i]-'0'))%md;
	}
	long long sum=0;
	for(int i=0;i<x;i++){
		sum=(sum+dp[i])%md;
	}
	cout<<sum<<endl;
	return 0;
}
