typedef pair<int,int> pii;
const int maxn=1000;
int c[maxn][maxn];
int n;
int xs[]={1,-1,0,0},ys[]={0,0,1,-1};
bool dfs(int x,int y){
  if(x>=n || x<0 || y<0 || y>=n || !c[x][y])
    return 0;
  if(x==n-1 && y==n-1)
    return 1;
  c[x][y]=0;
  for(int i=0;i<4;i++)
    if(dfs(xs[i]+x,ys[i]+y))
      return 1;
  return 0;
}
