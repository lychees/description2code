#include<bits/stdc++.h>
using namespace std;
main()
{
	long long int t;
	cin>>t;
	while(t--)
	{
		string a,b;
		cin>>a>>b;
		long long int ct=0;
		
		long long int l1=a.size();
		long long int l2=b.size(),i,j;
		for(i=0;i<l1+1;i++)
		{
			string x="";
			if(i==0)
			{
			   x+=b;
			   x+=a;
			}
			else if(i==l1)
			{
				x+=a;
				x+=b;
			}  
			else
			{
				
				for(j=0;j<i;j++)
				  x+=a[j];
				x+=b;
				for(j=i;j<l1;j++)
				 x+=a[j];
			}
			string temp=x;
			reverse(x.begin(),x.end());
			if(temp==x)
			 ct++;
						
		}
        cout<<ct<<endl;		
	}
	
	
	
	
}
