#include<bits/stdc++.h>

#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000000007
#define bitcount __builtin_popcountll
#define ll long long
int main()
{
	int i,j,k;
	ll sum=0;
	int t,n,arr[100005];

	while(t--)
    {
        sum=0;
