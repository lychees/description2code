#include<bits/stdc++.h>
using namespace std;
char arr[2001][2001]={NULL};
char a[2001][2001]={NULL};
char s1[33]="_#_#_#_#_#_#_#_#_#_#_#_#_#_#_#_#";
char s2[33]="#_#_#_#_#_#_#_#_#_#_#_#_#_#_#_#_";
char temp[33]={NULL};
int main()
{
int n,m,flag=0,count=0,i,j,k;
scanf("%d %d",&n,&m);
for(i=0;i<n;i++)
{
scanf("%s",arr[i]);
}
for(i=0;i<n;i++)
for(j=0;j<m;j++)
{
k=0;
for(k=0;k<32;k++)
{
if(j+k>=m)
	break;
temp[k]=arr[i][j+k];
}
temp[k]='\0';
if(strcmp(temp,s1)==0)
	{a[i][j]='1';
	}
if(strcmp(temp,s2)==0)
		{a[i][j]='2';
		}
if(strcmp(temp,s1)!=0&&strcmp(temp,s2)!=0)
	a[i][j]='0';
}
for(i=0;i<n-31;i++)
for(j=0;j<m-31;j++)
{
if(a[i][j]=='1')
{
flag=0;
for(k=i;k<(i+32);k=k+2)
{
if((a[k][j]!='1')||(a[k+1][j]!='2'))
    {
	flag=1;
     break;
    }
}
if(flag!=1)
	count++;
}
}
printf("%d\n",count);
}
