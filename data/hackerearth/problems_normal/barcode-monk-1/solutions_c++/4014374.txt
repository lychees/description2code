#include <bits/stdc++.h>
using namespace std;
#define ll long long
const ll mod=1e9+7;
bool superBrush[1005];
ll n,k,b,dp[1005][2];
int main()
{
    cin>>n>>k;
    superBrush[0]=true;
    for(ll i=0;i<k;i++)
    {
        cin>>b;
        for(ll j=b;j<=n;j++)
            superBrush[j]|=superBrush[j-b];
    }
    dp[0][0]=1;
    for(ll i=1;i<=n;i++)
    {
        dp[i][0]=(dp[i-1][0]+dp[i-1][1])%mod;
        dp[i][1]=0;
        for(ll j=1;j<=i;j++)
        {
            if(superBrush[j])
            {
                dp[i][1]=(dp[i][1]+dp[i-j][0])%mod;
            }
        }
       // cout<<dp[i][0]<<" "<<dp[i][1]<<endl;
    }
    ll res=(dp[n][1]+dp[n][0])%mod;
    cout<<res<<endl;
    return 0;
}
