
#include <iostream>
#include <stdio.h>
#include <math.h>

using namespace std;

#define RES (1000000000+9)

int main(){
        int T;
        cin >> T;
        while(T--) {
                long long int num, a=26, ans=1, m;
                cin >> num;
                m = (num+1)/2;
                while(m) {
                	if(m%2==1) {
                		ans *= a;
                		ans %= RES;
                	}
                	m /= 2;
                	a *= a;
                	a %= RES;
                }
                
                printf("%lld\n", ans);
        }
}
