#include<iostream>
#include<cstdio>
#include<cmath>
typedef long long ll;
 
using namespace std;
 
int main()
{
register ll i;
ll arr[2001],cnt=0,su,hrs,n;
 
 
scanf("%lld %lld",&n,&su);
for(i=0;i<n;i++)
{
scanf("%lld",&arr[i]);
}
for(i=1;i<n;i++)
{
if(arr[i]<=arr[i-1])
{
 
hrs=int(ceil(((arr[i-1]-arr[i]+1)/double(su))));
//printf("hrs=%lld",hrs);
 
arr[i]+=(su*hrs);
//printf("\n arr %lld \n",arr[i]);
cnt+=hrs;
}
}
printf("%lld\n",cnt);
}
