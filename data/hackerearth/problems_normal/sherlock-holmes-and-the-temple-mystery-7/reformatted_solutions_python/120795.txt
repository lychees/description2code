import sys

t = int(input())

while t > 0:
    n = int(input())
    s = sys.stdin.readline().strip().split()
    s = map(int, s)
    l = [0]
    d = {0: 0}
    for i in s:
        k = []
        for j in l:
            if (j+i) not in d:
                k.append(j+i)
                d[j+i] = 0
        l.extend(k)
    if (sum(s)/2.0) in d:
        print "YES"
    else:
        print "NO"
    t = t-1
