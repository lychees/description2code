def check_hard(s):
	res = "hard"
	vowels = "aeiouy"
	vcount = 0
	cont_const = 0
	for v in vowels:
		for l in s:
			if v == l:
				vcount+=1
	for l in s:
		if l not in vowels:
			cont_const += 1
		else:
			cont_const = 0
		if cont_const >=3 :
			return "hard"
			
	if len(s)-vcount <= vcount and vcount > 0:
		return "easy"
	return res

rounds = int(raw_input())
for i in range(rounds):
	s = raw_input()
	print check_hard(s)
