#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
using namespace std;
#define MAX 158
int main()
{	
	int n,a,i,j,carry,temp,x;
	int arr[MAX];
	cin>>n;
	while(n--)
	{
		i=0;					//initialization for the while loop
		while(i<MAX)
		{
			arr[i]=0;
			i++;
		}
		cin>>a;
		arr[MAX-1]=1;			//initialising the last element 1
		carry=0;				//initialising the carry to 0
		for(i=2;i<=a;i++)
		{
			j=MAX-1;			//initialisng the value of j to the last index
			/*while(arr[j]==0)
			j--;*/
			while(j>=0)
			{
				temp=arr[j]*i+carry;
				if(temp>=10)
				{
					arr[j]=temp%10;
					temp=temp/10;
					carry=temp;
				}
				else
				{
					arr[j]=temp;
					carry=0;
				}
				j--;
			}
		}
		x=0;
		for(i=0;i<MAX;i++)
		{
			if(arr[i]!=0)
			x=1;
			if(x==1)
			cout<<arr[i];
		}
		cout<<endl;
	}
	return 0;
}
