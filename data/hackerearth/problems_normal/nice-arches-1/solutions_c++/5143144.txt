#include <iostream>
#include <stack>
#include<cstring>

using namespace std;

int main()
{
	// get count
	int tcCount = 0;
	cin >> tcCount;
	char arr[1000000];
	int bublyCount = 0;
	string s;
	stack<char> bubblyStack;
	char c;
	
	for (int j = 0; j < tcCount; j++) {
		cin >> s;
		
		strcpy(arr, s.c_str());
		
		int length = s.length();
		
		bubblyStack.push(arr[0]);
		// cout << "pushed : " << arr[0] << endl;
		for (int i = 1; i < length; i++) {
			c = arr[i];
			// cout << "char = " << c << endl;
			
			if (bubblyStack.size() > 0 && c == bubblyStack.top()) {
				// cout << "poped : " << c << endl;
				bubblyStack.pop();
				continue;
			}
			else {
				bubblyStack.push(c);
				// cout << "pushed : " << c << endl;
			}
			
			
			if (bubblyStack.size() >= 3) {
				// cout << bubblyStack.size() << endl;
			//	break;
			}
		}
		
		if (bubblyStack.size() == 0) {
			bublyCount++;
			// cout << "bubbly" << endl;
		}
		else {
			// cout << "not bubbly" << endl;
			while (bubblyStack.size() > 0) {
				bubblyStack.pop();
			}
		}
		
		
	}
	
	cout << bublyCount << endl; 
	
    return 0;
}
