#include <iostream>
using namespace std;

long long int m[1000005] = {0};

int main()
{
    long long int n;
    long long int s=0,l,b,c;
    
    scanf("%lld", &n);
    
    long long int a[n];
    
    for(int i=0;i<n;i++) scanf("%lld", &a[i]);
    
    long long int q;
    
    scanf("%lld", &q);
    
    for(int i=0;i<q;i++) {
    	scanf("%lld %lld %lld", &b, &c, &l);
    	b--;
    	m[b] += l;
    	m[c] += -l;
    }
    
    for(int i=0;i<n;i++) {
    	s += m[i];
    	a[i] += s;
    	printf("%lld\n", a[i]);
    }
    
    
    return 0;
}
