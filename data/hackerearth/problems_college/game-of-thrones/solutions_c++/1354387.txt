#include <cmath>
#include <cstdio>
#include <vector>
#include <set>
#include <iostream>
#include <algorithm>
using namespace std;


int main() 
{
	int T,N,x;
	cin>>T;
	while(T--)
	{
		cin>>N;
		set<int>s;
		for(int i=0;i<N;i++)
			{
				cin>>x;
				s.insert(x);
			}
		cout<<N-s.size()<<endl;
	}
    return 0;
}
