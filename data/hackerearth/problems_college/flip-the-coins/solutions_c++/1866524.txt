#include <iostream>
using namespace std;

int main()
{
   int n,q,i,j,a,b,k,l,m;
   cin>>n>>q;
   int p[1000]={0};
   while(q--)
   {
   	cin>>l>>a>>b;m=0;
   	if(l==0)
   	{
   		for(i=a;i<=b;i++){p[i]=1;}
   	}
   	else
   	{
   		for(i=a;i<=b;i++){if(p[i])m+=1;}
   	cout<<m<<endl;
   	}
   	
   }
    return 0;
}
