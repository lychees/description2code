#include <iostream>
using namespace std;

#define MAX 10001

bool checkIfPossible(int a[], int n)
{
	if(n < 0)
		return true;
	
	if(n == 0)
	{	if(a[n] == 0)
			return true;
		else
			return false;
	}
		
	if(a[n] > a[n-1])
		return false;
	else if(a[n] == a[n-1])
		return checkIfPossible(a, n-2);
	else
	{
		a[n-1] -= a[n];
		return checkIfPossible(a, n-1);
	}
}

int main()
{
	int n, a[MAX];
	
    cin >> n;
    for(int i=0; i<n; ++i) cin >> a[i];
    
    if(checkIfPossible(a, n-1))
    	cout << "YES" << endl;
    else
    	cout << "NO" << endl;
    
    return 0;
}
