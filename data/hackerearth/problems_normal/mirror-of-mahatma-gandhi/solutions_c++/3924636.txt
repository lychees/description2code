#include <iostream>
using namespace std;
bool ispalin(string n, int l)
{
	for(int i=0,j=l-1;i<l;i++,j--)
	{
		if(n[j]!=n[i])
		return false;
	}
	return true;
}
int main()
{
    int t,l;
    string n;
    cin>>t;
    
    while(t--)
    {	cin>>n;
    	bool f=true;
    	l=n.length();
    	//cout<<n<<endl;
    	for(int i=0;i<l;i++)
    	{
    		if(n[i]=='0' || n[i]=='1' || n[i]=='8')
    			f=true;
    		else
    		{
    			f=false;
    			break;
    		}
    	}
    	if(f==false)
    	cout<<"NO"<<endl;
    	else
    	{
    		f=ispalin(n,l);
    		//cout<<f;
    		if(f==true)
    		cout<<"YES"<<endl;
    		else
    		cout<<"NO"<<endl;
    	}
    	
    }
    return 0;
}
