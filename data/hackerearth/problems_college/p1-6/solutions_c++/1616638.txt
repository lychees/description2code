#include <stdio.h>
#include<string.h>

char s[1000005],s2[]="101";
unsigned long long int M[5][1000005]={0};
int main()
{
	int i,len;
    
    scanf("%s",s);
    len=strlen(s);
    
    int j;
    for (j=1;j<=1000005;j++)
        M[0][j]=1;
    for (j=1;j<=4;j++)
        M[j][0]=0;
    
for (i=1;i<=3;i++)
        {
            M[i][0]=0;
            for (j=1;j<=len;j++)
                if (s2[i-1]==s[j-1])
                M[i][j]=(M[i-1][j]+M[i][j-1]);
                else M[i][j]= M[i][j-1];
        }
        printf("%llu\n",M[3][len]);
        
    return 0 ;
}
