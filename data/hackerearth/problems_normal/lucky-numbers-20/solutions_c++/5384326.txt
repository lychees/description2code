#include <iostream>
#include <cmath>
#include <vector>
#include <stdio.h>
#define ll long long int
#define M 1000000007
using namespace std;
ll getValue(int lim,int start){
	return ((ll)pow(2,lim)+(ll)pow(2,start));
}
void update(int* lim,int* start){
	if((*lim-*start)==1){
		(*lim)++;
		*start=0;
	}
	else{
		(*start)++;
	}
}
vector<ll> getData(){
	int lim=1,start=0;
	vector<ll> data; 
	ll val = getValue(lim,start),n=pow(10,18);
	while(val<=n){
		data.push_back(val);
		update(&lim,&start);
		val=getValue(lim,start);
	}
	return data;
}
int main()
{
	vector<ll> data = getData();
	int t;
	scanf("%d",&t);
	while(t--){
		ll n,sum=0;
		scanf("%lld",&n);
		for(int i=0;data[i]<=n;i++)
		sum=(sum+(data[i]%M))%M;
		printf("%lld\n",sum);
	}
    return 0;
}
