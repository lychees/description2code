int w[MOD+5];
vector<int>primes;


ll check(int r)
{
    int i=0; ll ans=0;
    while(primes[i]*primes[i]<r)
    {
        ans+=(ll)(upper_bound(all(primes),r/primes[i])-primes.begin()-i-1);
        i++;
    }
    return ans;
}
