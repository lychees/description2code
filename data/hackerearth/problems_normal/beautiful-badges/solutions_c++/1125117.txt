#include <iostream>
#include <algorithm>
#include <vector>
#include <cstring>
using namespace std;

#define SQR(_x) ((_x)*(_x))
#define MAX 100100
char f[MAX];

int main()
{
	int n;
	vector<pair<int, int> >v;
	scanf("%d", &n);
	for(int i = 0; i < n; i++) {
		int ci, di;
		scanf("%d %d", &ci, &di);
		v.push_back(make_pair(di, ci));
	}
	sort(v.rbegin(), v.rend());
	int resp = 0;
	memset(f, 0, sizeof(f));
	
	int a = -1;
	int b = -1;
	for(int i = 0; i < n && (a < 0 || b < 0); i++) {
		if (a < 0 && v[i].second == 1) a = i;
		if (b < 0 && v[i].second == 2) b = i;
	}

	while(a < n && b < n) {
		if (a < b) {
			if (2*SQR(v[b].first) > SQR(v[a].first)) {
				resp++; a++; b++;
				while(a < n && v[a].second != 1) a++;
				while(b < n && v[b].second != 2) b++;
			} else {
				a++;
				while(a < n && v[a].second != 1) a++;
			}
		} else {
			if (2*SQR(v[a].first) > SQR(v[b].first)) {
				resp++; a++; b++;
				while(a < n && v[a].second != 1) a++;
				while(b < n && v[b].second != 2) b++;
			} else {
				b++;
				while(b < n && v[b].second != 2) b++;
			}			
		}
	}
	printf("%d\n", resp);
	
    return 0;
}
