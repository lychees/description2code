#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--){
    int n;
    cin>>n;
    vector<int> v;
    int cnt=0,cnt1=0;
    for (int i=0;i<n;i++) {
    	int x;
    cin>>x;
    if (x<0)cnt++;
    else cnt1++;
    v.push_back(x);
    }
    if (cnt<(cnt1)) cout<<cnt+1<<"\n";
    else if (cnt>(cnt1)) cout<<cnt<<"\n";
    else if (cnt==(cnt1)) cout<<cnt<<"\n";
    }
    return 0;
}
