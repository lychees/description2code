ll arr[1000005];

int main(){

   int n;
   scanf("%d",&n);

   for(int i=0;i<n;++i)
     scanf("%lld",&arr[i]);

   sort(arr,arr+n);

   ll layer=1;

   ll sum=0;

   for(int i=0;i<n;){

      if(arr[i]<=layer*layer){

         sum+=layer*layer-arr[i];
         ++i;
         layer++;

      }

      else{

        sum+=layer*layer;
        layer++;
      }

   }
   cout<<sum<<"\n";

   return 0;
}
