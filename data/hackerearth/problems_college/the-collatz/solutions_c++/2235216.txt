#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,x,step,even,odd;
    cin>>n;
    while(n--) {
    	cin>>x;
    	step=0;
    	even=0;
    	odd=1;
    	while(x!=1) {
    		if(x%2==0) {
    			even+=x;
    			x/=2;
    		}
    		else {
    			odd+=x;
    			x=(3*x)+1;
    		}
    		step++;
    	}
    	//cout<<step<<" "<<even<<" "<<odd<<"\n";
    	int a=even%step;
    	int b=odd%step;
    	if(a>b)
    		cout<<"Even Rules\n";
    	else if(a<b)
    		cout<<"Odd Rules\n";
    	else
    		cout<<"Tie\n";
    }
    
    return 0;
}
