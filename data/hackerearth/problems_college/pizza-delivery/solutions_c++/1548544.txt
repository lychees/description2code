#include <iostream>
using namespace std;

int main()
{


		int n;
		cin>>n;

		long long int dst[n+1][n+1];
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
			{
				cin>>dst[i][j];
			}
		}
		for(int k=0;k<n;k++)
            {
                for(int i=0;i<n;i++)
                {

                    for(int j=0;j<n;j++)
                    {
                       dst[i][j]=min(dst[i][j],dst[i][k]+dst[k][j]);
                    }
                }

            }
		int t;
	    cin>>t;
	    while(t--)
	    {
	    	int src,gas,dest;
	    	cin>>src>>gas>>dest;	
	    	long long int max_price=dst[src][gas]+dst[gas][dest];
	    	long long int min_price=dst[src][dest];
	    	cout<<max_price<<" "<<max_price-min_price<<endl;
	    }

   // cout << "Hello World!" << endl;
    return 0;
}
