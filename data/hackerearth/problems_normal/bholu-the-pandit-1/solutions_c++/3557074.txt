#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;

ll fun(ll p,ll q)
{
	ll count=1;
	while(q)
	{
		if(q&1)
		count=(count*p)%mod;
		p=(p*p)%mod;
		q>>=1;
	}
	return count;
}

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	ll a,b,c;
    	cin>>a>>b>>c;
    	ll temp=__gcd(a,__gcd(b,c));
    	ll ans1=temp;
    	temp=(temp*((temp*temp)%mod))%mod;
    //	cout<<temp<<endl;
    	ll z;
    	z=(a*((b*c)%mod))%mod;
    	ll ans;
    	ans=(z*fun(temp,mod-2))%mod;
    	cout<<ans1<<" "<<ans<<endl;
    }
    return 0;
}
