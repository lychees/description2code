import math
t = input()
for i in range(t):
    l, p, c = map(int, raw_input().split())
    val = l
    res = 0
    while val < p:
        val *= c
        res += 1
    cnt = 0
    res -= 1
    for j in range(31, -1, -1):
        if res & (1 << j) == 0:
            cnt += 1
        else:
            break
    print "Case #%d:" % (i + 1), 32 - cnt
