#include <iostream>
#include <stdio.h>
#include <vector>
#include <utility>
#include <algorithm>
using namespace std;

vector <pair <int, int>> v;
vector <pair <int, int>> :: iterator it;

int main()
{
    int N;
    scanf("%d", &N);
    int i = 1;
    while(i <= N)
    {
    	int t, d;
    	scanf("%d %d", &t, &d);
    	v.push_back(make_pair(t + d, i));
    	++i;
    }
    sort(v.begin(), v.end());
    it = v.begin();
    while(it != v.end())
    {
    	cout << it ->second << " ";
    	++it;
    }
    return 0;
}
