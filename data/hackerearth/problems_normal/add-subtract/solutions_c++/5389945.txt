#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,k,i,j,ans,temp;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&k);
		int a[n];
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		ans=INT_MAX;
		for(i=1;i<=100;i++)
		{
			vector<int> v;
			for(j=0;j<n;j++)
			{
				if(i>=a[j])
				{
					v.push_back((i-a[j])*3);
				}
				else
				{
					v.push_back((a[j]-i)*5);
				}
			}
			sort(v.begin(),v.end());
			temp=0;
			for(j=0;j<k;j++)
			{
				temp+=v[j];
			}
			ans=min(ans,temp);
		}
		printf("%d\n",ans);
	}
    return 0;
}
