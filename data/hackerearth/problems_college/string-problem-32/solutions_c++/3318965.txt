int readint() {
    int n = 0;
    char c = getchar_unlocked();
    while ( !( '0' <= c && c <= '9' ) ) {
        c = getchar_unlocked();
    }
    while ( '0' <= c && c <= '9' ) {
        n = (n<<3) + (n<<1) + c - '0';
        c = getchar_unlocked();
    }
    return n;
}
bool ispal(string s)
{
	int i=0, j=s.length()-1;
	while(i<=j)
	{
		if(s[i]!=s[j]) return false;
		i++;
		j--;
	}
	return true;
}
