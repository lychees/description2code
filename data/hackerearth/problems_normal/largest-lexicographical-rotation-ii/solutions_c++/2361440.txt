 int minm(string s)
{
	int i=0,k=0,j=1;
	s=s+s;
	while((i+k)<s.size() && (j+k)<s.size())
	{
	if(s[i+k]==s[j+k])
	k++;
	else
	if(s[i+k]<s[j+k])
	{
		i=i+k+1;
		if(i<=j)
		i=j+1;
		k=0;
	}
	else
	if(s[j+k]<s[i+k])
	{
		j=j+k+1;
		if(j<=i)
		j=i+1;
		k=0;
	}
	}
	return min(i,j);
}
