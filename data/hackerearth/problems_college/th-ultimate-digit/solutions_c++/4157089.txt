#include<bits/stdc++.h>
using namespace std;


typedef long long int LL;

#define scll(Q) scanf("%lld",&Q);
#define sci(Q) scanf("%d",&Q);
#define prll(Q) printf("%lld\n",Q);
#define pri(Q) printf("%d\n",Q);


int compute(unsigned long long a, unsigned long long b){
	
	int bit=b%2, result=1;
	
	while(b){
		bit=b%2;
		if(bit==1){
			result=(result*a)%100000;			
		}
		a=(a*a)%100000;
		b=b/2;
	}
	return result;	
}


int main()
{
	int t;
	cin>>t;
	while(t--){
		char str[10009];
		unsigned long long b;
		cin>>str>>b;
		int a = str[strlen(str)-1] - 48;
		
		cout<<compute(a,b)%10<<endl;	
	}
	return 0;
}
