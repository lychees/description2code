bool israt(int a[])
{
    int dis1=(a[2]-a[0])*(a[2]-a[0])+(a[3]-a[1])*(a[3]-a[1]);
    int dis2=(a[4]-a[0])*(a[4]-a[0])+(a[5]-a[1])*(a[5]-a[1]);
    int dis3=(a[4]-a[2])*(a[4]-a[2])+(a[5]-a[3])*(a[5]-a[3]);
    if (dis1==0 || dis2==0 || dis3==0) return false;

    if (dis1+dis2==dis3)
    {
        return true;
    }
    else if (dis1+dis3==dis2)
    {
        return true;
    }
    else if (dis2+dis3==dis1)
    {
        return true;
    }
    return false;
}
