/*
If P is inside triangle so three area part will equal to big area

		-x2y1
|x1 y1| -x3y2
|x2 y2| -x1y3
|x3 y3| x1y2
|x1 y1| x2y3
		x3y1
		
area = abs(x1*y2+x2*y3+x3*y1-x2*y1-x3*y2-x1*y3)/2
area*2 = abs(x1*y2+x2*y3+x3*y1-x2*y1-x3*y2-x1*y3)

let X,Y,Z be three small area
then 	area = X+Y+Z
		2*area = 2*X+2*Y+2*Z
so we can store any area in variable with area*2 and don't lose significance
(area*2 is int)

*/

double d(double x1,double y1,double x2,double y2) {
	return sqrt((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1));
}

int main()
{
    ios::sync_with_stdio(false);
    int t; cin>>t;
    while (t--) {
    	long long x1,y1,x2,y2,x3,y3,x,y; cin>>x1>>y1>>x2>>y2>>x3>>y3>>x>>y;
		long long A = abs(x1*y2+x2*y3+x3*y1-x2*y1-x3*y2-x1*y3);
		long long X = abs(x*y2+x2*y3+x3*y-x2*y-x3*y2-x*y3);
		long long Y = abs(x1*y+x*y3+x3*y1-x*y1-x3*y-x1*y3);
		long long Z = abs(x1*y2+x2*y+x*y1-x2*y1-x*y2-x1*y);
		//cout<<A<<' '<<X<<' '<<Y<<' '<<Z<<endl;
		if (A==X+Y+Z) {
			cout<<"INSIDE"<<endl;
		} else {
			cout<<"OUTSIDE"<<endl;
		}
    }
    return 0;
}
