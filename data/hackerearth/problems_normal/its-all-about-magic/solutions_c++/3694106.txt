#include <bits/stdc++.h>
using namespace std;
int main() {
    int T, M;
    cin >> T;
    while (T--) {
        cin >> M;
        long long ux, uy;
        ux = 0, uy = 1;
        for (int i = 0; i < M; ++i) {
            long long vx, vy;
            vx = 0, vy = 0;
            vx += uy;
            vx += ux;
            vy += ux;
            ux = vx;
            uy = vy;
        }
        cout << uy << " " << ux << '\n';
    }
    return 0;
}
