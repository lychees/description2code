#include <iostream>

using namespace std;
int pali(int n)
{    
	int n1,s,d;s=d=0;
	n1=n;
    		while(n1!=0)
    		{
    			d=n1%10;
    			s=s*10+d;
    			n1=n1/10;
    		}
    		
    		if(s==n)
    		return 1;
    		else 
    		return 0;
    			
   }

int main()
{
    int t,d,c,a,b,n;

    {
