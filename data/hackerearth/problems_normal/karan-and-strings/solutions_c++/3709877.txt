#include <iostream>
#include <cassert>
#include <string.h>


using namespace std;

int main()
{
	char input[100000],minput[100000];
	int len,i,j=0,t,k;
	char a;
	cin >> t;
	assert(1 <= t & t <= 100);
	for ( k = 0; k < t; k++)
	{
		fflush(stdin);
		cin >> input;
		len = strlen(input);
		assert(1 <= len < 100000);
		a = input[0];
		j = 0;
		minput[j++] = a;
		i = 1;
		while(i < len)
		{
			if (input[i] == a)
				i++;
			else
			{
				a = input[i];
				minput[j++] = input[i];
			}
		}
			
		minput[j] = '\0';
		cout << minput << endl;
	}
	
	return 0;
}
