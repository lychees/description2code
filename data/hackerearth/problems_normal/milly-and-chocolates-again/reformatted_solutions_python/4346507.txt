def main():
    raw_input()
    c = [int(i) for i in raw_input().split()]
    i = int(raw_input())
    q = []
    for j in range(i):
        q.append(int(raw_input()))

    index = []
    a = 0
    for i in c:

        a += 1
    for i in q:
