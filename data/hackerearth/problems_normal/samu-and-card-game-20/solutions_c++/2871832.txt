#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,m,x=1,y=1,a,b,k,jump=0;
    	cin>>n>>m>>k;
    	for(int i=0;i<k;i++)
    	{
    		cin>>a>>b;
    		while( (x+a<=n&& x+a>0) && (y+b<=m&& y+b>0)  )
    		{
    			x=x+a;
    			y=y+b;
    			jump++;    		
    		}    		
    	}   	    	
    	cout<<jump<<endl;
    }
    return 0;
}
