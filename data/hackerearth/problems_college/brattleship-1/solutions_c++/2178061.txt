#include<iostream>
#include<cstdio>
using namespace std;

int main()
{
	      int t,r,c,w,i,j=1;
	      cin>>t;
	      while(t--)
	      {
	      	  cin>>r>>c>>w;
	      	  if(c%w==0)
	      	    {
				  i=(r*(c/w))+w-1;
			    }
			  else
			  {
			      i=(r*(c/w))+w;	  
			  }
	      	    printf("Case #%d: %d\n",j,i);
				  j++; 
		  }
		    return 0;
	        
}
