#include <iostream>
#include <set>
#include<stdio.h>
using namespace std;
int main()
{
	long long t;
	scanf("%lld",&t);
	while(t--)
	{
		long long n,m,i, x;
		scanf("%lld",&n);
		scanf("%lld",&m);
		set<long long > s;
		for(i=0;i<n;i++)
		{
			scanf("%lld",&x);
			s.insert(x);
		}
		while(m--)
		{
			scanf("%lld",&x);
			if(s.find(x)!=s.end())
				printf("YES\n");
			else
			{
				printf("NO\n");
				s.insert(x);
			}
		}
	}
	return 0;
}
