

    #include<bits/stdc++.h>
    using namespace std;
    set<long int> s;
    void init()
    {
        long int i,j,k;
        s.insert(1);
        for(i=2;i<10000;i++)
        {
            j=i;
     
            j=j*j;
            while(j<10000000)
            {
                s.insert(j);
                j=j*i;
            }
        }
    }
    int main()
    {
        init();
        int t;
        cin>>t;
        while(t--)
        {
          long  int x;
            scanf("%ld",&x);
            int flag = 0;
            for(set<long int>::iterator it = s.begin(); it!=s.end();it++)
            {
                if(*it >= x)
                    break;
                long int y = *it;
                if(s.find(x - y) != s.end())
                {
                    
                    flag = 1;
                    break;
                }
            }
            if(flag)
                printf("Yes\n");
            else
                printf("No\n");
     
        }
        return 0;
    }
