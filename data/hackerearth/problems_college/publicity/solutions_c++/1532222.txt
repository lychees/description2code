void visit(list <int> *l, bool *vis, int i)
{
	queue<int> q;
	q.push(i); vis[i]=true;
	list <int>::iterator it;
	while(!q.empty())
	{
		int t= q.front();
		q.pop();
		it = l[t].begin();
		while(it != l[t].end())
		{
			if(!vis[*it])
			{
				vis[*it]=true;
				q.push(*it);
			}
			it++;
		}
	}
}
int main()
{
	int t;
	si(t);
	while(t--)
	{
		int n, e;
		si(n); si(e);
		list<int> lis[n];
		bool vis[n];
		for(int i=0; i<n;i++)
			vis[i]=false;
		for(int i=0; i<e;i++)
		{
			int a,b;
			si(a); si(b);
			lis[a].push_back(b);
			lis[b].push_back(a);
		}
		int count=0;
		for(int i=0;i<n;i++)
		{
			if(!vis[i])
			{
				count+=1;
				visit(lis, vis, i);
			}
		}
		printf("%d\n", count);
	}
	return 0;
}
