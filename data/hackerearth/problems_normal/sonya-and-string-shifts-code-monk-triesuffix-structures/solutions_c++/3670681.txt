#include <iostream>
#define MAXN 1000069

using namespace std;
int N;
string lexi;
char car[2*MAXN];
int ans[MAXN];

int check(int arxi , int min_arxi){
    int flag = 1;
    for(int i=0; i<N; i++){
        if(car[i+arxi] < car[min_arxi+i]){
            flag = 1;
            break;
        }
        if(car[i+arxi] > car[min_arxi+i]){
            flag = 0;
            break;
        }
    }

    if(flag == 1) return arxi;
    return min_arxi;
}


int main(){
    cin >> N;
    cin >> lexi;

    for(int i=0; i<N; i++)
        car[i] = lexi[i];
    for(int i=0; i<N; i++){
        car[N+i] = lexi[i];
    }


    int arxi;
    int min_arxi = 0;

    for(int i=1; i<N; i++){
        arxi = i;
        ans[i] = check(arxi , min_arxi);

        if(ans[i] == i) min_arxi = i;
    }
    int Q;
    cin >> Q;

    while(Q--){
        int num;
        cin >> num;
        cout << ans[num] << endl;
    }

    return 0;
}
