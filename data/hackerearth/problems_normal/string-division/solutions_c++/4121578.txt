#include<bits/stdc++.h>
#include<stdio.h>
#define pb(a) push_back(a)
#define ll long long int
using namespace std;
int main()
{
    int t;
    cin>>t;
    cin.ignore();
    while(t--)
    {
        string ch;
        getline(cin,ch);
        vector <char>::iterator it;
        vector <char> a,b;
        int i=0;
        while(ch[i]!='\0')
        {
            a.pb(ch[i]);
            b.pb(ch[i++]);
        }
        sort(a.begin(),a.end());
        it=unique(a.begin(),a.end());
        a.resize(distance(a.begin(),it));
        if(a.size()>=4){cout<<"YES\n";}
        else if(a.size()==3 && b.size()>4){cout<<"YES\n";}
        else if(a.size()==1 && b.size()>9){cout<<"YES\n";}
        else if(a.size()==2 && b.size()>6){cout<<"YES\n";}
        else {cout<<"NO\n";}
    }
    return 0;
}
