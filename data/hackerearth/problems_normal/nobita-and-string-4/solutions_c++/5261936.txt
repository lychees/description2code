#include <iostream>
#include<stack>
using namespace std;

int main()
{
    int t; cin>>t;
    cin.ignore();
    while(t--){
    	string s,temp="";
    	stack<string>st;
    	getline(cin,s);
    	for(int i=0;i<s.length();i++){
    		while(s[i]!=' '&&i<s.length()){
    			temp+=s[i];
    			i++;
    		}
    		//cout<<temp<<endl;
    		st.push(temp);
    		temp="";
    		//i++;
    	}
    	while(!st.empty()){
    		string ans=st.top();
    		cout<<ans<<" ";
    		st.pop();
    	}
    	cout<<endl;
    }
    return 0;
}
