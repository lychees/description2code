#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	vector<ll>v;
	ll n,i,b;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>b;
		v.push_back(b);
	}
	ll max1=v[0];
	ll max_curr=v[0];
	for(i=1;i<n;i++)
	{
		max_curr=max(v[i],max_curr+v[i]);
		max1=max(max1,max_curr);
	
		
	}
	cout<<max1;
	return 0;
}
