#include <iostream>
#include <stdio.h>
#include <cmath>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int k,n;
        cin>>k>>n;
        double ans=1.0,tmp=1.0;


        for(int i=0;i<k;i++)
        {
            tmp=(double)tmp*(n-i);
            tmp=(double)tmp/n;
        }
        ans=1-tmp;

        printf("%0.9lf\n",ans);
        //cout<<tmp;

    }
    return 0;
}
