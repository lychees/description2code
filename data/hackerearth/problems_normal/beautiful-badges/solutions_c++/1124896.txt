#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main(){
    ios::sync_with_stdio(false);

    int n;
    cin >> n;

    vector<int> a, b;
    a.reserve(n);
    b.reserve(n);

    for(int i = 0; i < n; i++){
        int ci, di;
        cin >> ci >> di;
        if(ci == 1){
            a.push_back(di * di);
        }else{
            b.push_back(di * di);
        }
    }

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    int pa = 0;
    int pb = 0;
    int sol = 0;
    while(pa < a.size() and pb < b.size()){
        if(b[pb] >= 2 * a[pa]){
            pa++;
        }else if(a[pa] >= 2 * b[pb]){
            pb++;
        }else{
            pa++;
            pb++;
            sol++;
        }
    }

    cout << sol << '\n';
}
