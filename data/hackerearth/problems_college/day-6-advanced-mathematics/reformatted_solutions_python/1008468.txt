n = int(raw_input())


def comp_string(word):
    if word[0] == word[1]:
        print "SLAP"
        return
    elif len(word) > 2:
        comp_string(word[1:])
    else:
        print "KISS"
        return


for i in range(n):
    input_string = raw_input()
    comp_string(input_string)
