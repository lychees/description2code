#include <iostream>
#include <algorithm>

using namespace std;

bool isPrefix( string s1, string s2 ) {
	
	int n1, n2, i;
	n1 = s1.length();
	n2 = s2.length();
	
	if( n2 < n1 )
		return false;
	
	for( i = 0;i < n1;i++ ) {
		if( s1[ i ] != s2[ i ] )
			return false;
	}
	
	return true;
}

int main()
{
	ios_base::sync_with_stdio( false );
	int t, n, i, flag;
	
	cin >> t;
	
	while( t-- ) {
		
		cin >> n;
		vector<string> A( n );
		
		for( i = 0;i < n;i++ )
			cin >> A[ i ];
		
		sort( A.begin(), A.end() );
		
		flag = 1;
		for( i = 0;i < n - 1;i++ ) {
			if( isPrefix( A[ i ], A[ i + 1 ] ) ) {
				flag = 0;
				break;
			}
		}
		
		if( flag )
			cout << "YES\n";
		else
			cout << "NO\n";
	}
	return 0;
}
