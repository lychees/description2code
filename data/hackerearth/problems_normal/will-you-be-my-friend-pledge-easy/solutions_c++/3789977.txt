#include<iostream>
using namespace std;
int get_gcd(int a,int b)
	{
	int no;
	no=(a>b?a:b)/2;
	for(int i=2;i<=no;i++)
		{
		if(a%i==0 && b%i==0)
			return 1;
		}
	return 0;
	}
int main()
	{
	int i,n,no,a[1001],cnt=0;
	cin>>no;
	cin>>n;
	for(i=0;i<n;i++)
		{
		cin>>a[i];
		}	
	for(i=0;i<n;i++)
		{
		cnt=cnt+get_gcd(no,a[i]);
		}
	cout<<cnt;
	return 0;
	}
