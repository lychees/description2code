#include <iostream>
#include <stdio.h>
#include <string>
#include <algorithm>
#include <vector>
#include <math.h>
using namespace std;
typedef long long ll;
int main()
{
	int t;
	scanf("%d",&t);
	while(t-->0)
	{
		string s1,s2;
		cin>>s1;
		cin>>s2;
		int count=0;
		int l=min(s1.length(),s2.length());
		for(int i=0;i<l;i++)
		{
			if(s1[i]==s2[i])
				count++;
		}
		printf("%d\n",count );	
	}
	return 0;
}
