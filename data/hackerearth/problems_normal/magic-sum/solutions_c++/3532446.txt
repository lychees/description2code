#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000000007
#define bitcount __builtin_popcountll
#define ll long long
int main()
{
    int i,j,k;
    int t,n,arr[1000],ans;
    sd(t);
    while(t--)
    {
        sd(n);
        for(i=1;i<=n;i++)
            sd(arr[i]);
        ans=-10000000;
        for(i=n/2+1;i<=n;i++)
            ans=max(ans,arr[i]);
        for(i=n/2;i>0;i--)
        {
            ans=max(ans,arr[2*i]+arr[2*i+1]+arr[i]);
            arr[i]=max(arr[i*2],arr[2*i+1])+arr[i];
        }
        printf("%d\n",ans);
    }
    return 0;
}
