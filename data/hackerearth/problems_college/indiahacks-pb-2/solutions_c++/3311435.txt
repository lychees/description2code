#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n; string str,ans;
	cin>>n;
	map<string,int>mp;
	mp.clear();
	for(int i=0;i<n;i++)
	{
	cin>>str;
	mp[str]++;
	reverse(str.begin(),str.end());
	if(mp[str])
		ans = str;
	}
	
	cout<<ans.length()<<" "<<ans[(ans.length()-1)/2]<<endl;
    return 0;
}
