#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char s[1000005],s1[1000005];
    	cin>>s;
    	cin>>s1;
    	int l=strlen(s);
    	int l1=strlen(s1);
    	sort(s,s+l);
    	sort(s1,s1+l1);
    	if(strcmp(s,s1)==0)
    	cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
    }
    return 0;
}
