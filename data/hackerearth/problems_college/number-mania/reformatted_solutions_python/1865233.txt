from math import sqrt
from itertools import count, islice


def isPrime(n):
    if n < 2:
        return False
    return all(n % i for i in islice(count(2), int(sqrt(n)-1)))


def factorial(n):
    if n == 0:
        return 1
    else:
        return n * factorial(n-1)


valids = {}
for x in range(2, 101):
    if isPrime(x):
        valids[x] = factorial(x)
