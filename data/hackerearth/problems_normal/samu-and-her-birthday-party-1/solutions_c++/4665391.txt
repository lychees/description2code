#include <iostream>
#include<math.h>
#include<vector>
using namespace std;

int func(int ans)
{
	int cnt=0;
	while(ans>0)
	{
		if(ans%2==1)
			cnt++;
		ans/=2;
	}
	return cnt;
}
int main()
{
	int t;
	cin>>t;
	while(t>0)
	{
		int n,k;
		cin>>n>>k;
		
		vector<int> final;
		final.resize(n);	
		int tempnum;
		
		for(int i=0;i<n;i++)
		{
			int tempsum=0;
			int base=1;
			string s;
			cin>>s;
			
			for(int j=0;j<k;j++)
			{
			
				if(s.at(j)=='1')
					tempsum+=base;
				base=base*2;
			}
			final[i]=tempsum;
		}
		
		int maximum=pow(2,k+1)-1;
		int maximumans=k;
		for(int i=1;i<=maximum;i++)
		{
			// generate binary number
			int temp=i;
			int cnt=0;
			for(int j=0;j<n;j++)
			{
				if(final[j]&i)
				{
					cnt++;
				}
			}
			if(cnt==n)
			{
				// count number of bits
				int bitscount=func(i);
				maximumans=min(maximumans,bitscount);
			}
		}
		cout<<maximumans<<endl;
		t--;
	}
}
