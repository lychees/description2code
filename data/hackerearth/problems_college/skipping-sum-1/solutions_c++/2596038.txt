#include<bits/stdc++.h>
#define ll long long


using namespace std;

ll arr[11][100005];
ll a[100001];

void init(ll n)
{
    for(ll i=1;i<11;i++)
    {

        for(ll j=1;j<n+1;j++)
        {

            if(j-i>0)
                arr[i][j]=arr[i][j-i]+a[j];
            else
                arr[i][j]=a[j];

               // cout<<arr[i][j]<<" ";

        }
       // cout<<"\n";
    }
}

int main()
{
    ll n,q,b1,b2,k;
  cin>>n>>q;
  for(int i=1;i<n+1;i++)
    cin>>a[i];
  init(n);
  while(q--)
  {
      cin>>b1>>b2>>k;
      if(b1-k<=0)
      {
          ll x=(b2-b1)/k;
          b2=b1+k*x;
         // cout<<b1<<" "<<b2;
          cout<<arr[k][b2]<<"\n";
      }
      else
      {

          ll x=(b2-b1)/k;
         // cout<<b1<<" "<<b2;
          b2=b1+k*x;
          cout<<arr[k][b2]-arr[k][b1-k]<<"\n";
      }
  }

}
