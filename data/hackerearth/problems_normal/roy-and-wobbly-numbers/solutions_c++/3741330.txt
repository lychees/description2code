#include <iostream>
using namespace std;

int main()
{
	int T,N,K,i,j,coun,tens,digi;
	char flag='n';
	cin>>T;
	if(0<T && T<=100)
	{
		while(T>0)
		{
            coun=0;
		    flag='n';
			cin>>N;
			cin>>K;
			if(2<N && N<=1000 && 0<K && K<=100)
			{
				for(i=1;i<10;i++)
				{
					for(j=0;j<10;j++)
					{
						if(i!=j)
						{
							coun++;
							if(coun==K)
                            {
                                tens=i;
                                digi=j;
                                flag='y';
                                break;
                            }
						}
					}
					if(flag=='y')
                    {
                        break;
                    }
				}
				if(flag=='y')
                {
                    for(i=1;i<N+1;i++)
                    {
                        if(i%2==0)
                        {
                            cout<<digi;
                        }
                        else
                        {
                            cout<<tens;
                        }
                    }
                }
                else
                {
                    cout<<-1;
                }
			}
			cout<<endl;
			T--;
		}
	}
    return 0;
}
