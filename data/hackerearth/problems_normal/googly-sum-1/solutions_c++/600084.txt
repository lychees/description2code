#include <iostream>
#include <stdio.h>
using namespace std;

#define N 5005
#define M 1000000
#define K 400002
#define ll long long

int main()
{
    int n;
    scanf("%d", &n);
    int A[N];
    for(int i = 0; i < n; ++i)
    {
    	scanf("%d", &A[i]);
    }
    int B[M] = {0};
    int count = 0;
    B[A[0]*2 + K] = 1;
    for(int i = 1; i < n; ++i)
    {
    	for(int j = 0; j < i; ++j)
    	{
    		if(B[A[i] - A[j] + K] == 1)
    		{
    			count++;
    			break;
    		}
    	}
    	for(int j = 0; j <= i; ++j)
    	{
    		B[A[i] + A[j] + K] = 1;
    	}
    }
    printf("%d\n", count);
    return 0;
}
