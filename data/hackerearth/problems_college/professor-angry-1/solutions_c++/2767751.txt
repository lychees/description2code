    #include<iostream>
    #include<algorithm>
    using namespace std;
    int main()
    {
    int t;
    cin>>t;
    while(t--)
    {
    int n,k;
    cin>>n;
    cin>>k;
    int a[n],i;
    for(i=0;i<n;i++)
    {
    cin>>a[i];
    }
    //sort(a,a+n);
    int c=0;
    for(i=0;i<n;i++)
    {
    if(a[i]<=0)
    {
    c++;
    }
    }
    if(c>=k)
    {
    cout<<"NO"<<"\n";
    }
    else
    {
    cout<<"YES"<<"\n";
    }
    }
    }
