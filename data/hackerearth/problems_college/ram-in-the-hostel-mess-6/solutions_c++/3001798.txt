#include<iostream>
#include<stdio.h>
using namespace std;
int a[1000001];
void abc()
{
    a[1]=2;
    a[2]=3;
    for(int i=3;i<1000000;i++)
        a[i]=a[i-1]+a[i-2];
}
int main()
{
    abc();
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        cout<<a[n]*a[n]<<endl;
    }
    return 0;
}
