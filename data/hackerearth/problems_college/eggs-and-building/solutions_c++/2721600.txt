#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int dp[15][1010];//n floors & kbuildings
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int q;//queries
	cin>>q;
	while(q--)
	{
		ll int n,k;//number of floors & no of eggs
		cin>>n>>k;
		ll int i,j,x,l;
		for(i=0;i<=k;i++)
		dp[i][0]=0;
		for(i=1;i<=n;i++)
		dp[0][i]=INT_MAX;
		for(i=0;i<=n;i++)
		{
			dp[1][i]=i; //one egg
		}
		for(i=1;i<=k;i++)
		{
			dp[i][1]=1; //k eggs but 1 building
		}
		for(i=2;i<=k;i++)
		{
			for(j=2;j<=n;j++)
			{
				x=INT_MAX;
				for(l=1;l<=j;l++)
				{
					x=min(x,1+max(dp[i-1][l-1],dp[i][j-l]));
				}
				dp[i][j]=x;
			}
		}
		cout<<dp[k][n]<<endl;
		//~ for(i=0;i<=2;i++)
		//~ {
			//~ for(j=0;j<=6;j++)
			//~ cout<<dp[i][j]<<" ";
			//~ cout<<endl;
		//~ }
	}
	return 0;
}
