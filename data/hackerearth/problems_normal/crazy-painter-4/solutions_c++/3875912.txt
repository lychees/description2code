#include <iostream>
#include <vector>
using namespace std;

int main()
{
    
    int t;
    cin>>t;
    while(t--)
    {
    	unsigned long long int n,tp,f,ad,freq[27]={0};
    	cin>>n;
    	vector<unsigned long long int>dp;
    	dp.push_back(0);
    	unsigned long long int sum=0;
    	for(int i=0;i<26;i++)
    	{unsigned long long int x;
    		cin>>x;
    		dp.push_back(x);
    	}
    	tp=12*n;
    	f=tp/26;
    	ad=tp-(f*26);
    	
    	for(int i=1;i<=27;i++)
    	{freq[i]=f;}
    	
    	for(int i=1;i<=ad;i++)
    	{
    		freq[i]++;
    	}
    	for(int i=1;i<=26;i++)
    	{
    		sum+=freq[i]*dp[i]+(freq[i]*(freq[i]-1))/2;
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
