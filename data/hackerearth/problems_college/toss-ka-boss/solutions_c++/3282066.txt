#include <bits/stdc++.h>
using namespace std;
char s[2000001];
int main()
{
    long long int  k, t, m;
    cin>>t;
    int i;
    while(t--) {
    	
    	cin>>s;
    	m = 0;
    	k = 0;
    	for(i = 0 ; i < strlen(s); i++) {
    		if(s[i] == 'H')
    			k++;
    		else
    			k = 0;
    		m += k;
    	}
    	cout<<m<<endl;
    }
    return 0;
}
