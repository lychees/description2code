#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	string str;
    	cin>>str;
    	if(str.length()<=3) { cout<<str<<endl; continue; }
    	string value;
    	string rev;
    	for(int i=1;i<=str.length();i++) {
    		rev+=str[str.length()-i];
    	}
    	value+=rev.substr(0,3)+',';
    	int index = 3;
    	while(rev.substr(index).length()>2) {
    		value+=rev.substr(index,2)+',';
    		index +=2;
    	}
    	value+=rev.substr(index);
    	string rev_val;
    	for(int i=1;i<=value.length();i++) {
    		rev_val+=value[value.length()-i];
    	}
    	cout<<rev_val<<endl;
    }
    return 0;
}
