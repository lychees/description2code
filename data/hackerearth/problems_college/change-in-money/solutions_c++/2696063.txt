#include <bits/stdc++.h>
using namespace std;

long long int coins(int arr[], int n, int m)
{
   long long int dp[m+1];
   memset(dp,0,sizeof(dp));
	dp[0]=1;    
	for(int i=0; i<n; i++)
	{

		for(int j=arr[i]; j<=m; j++)
			dp[j]+=dp[j-arr[i]];
		
	}
    return dp[m];
}

int main()
{
  int n, m;
	cin>>m>>n;
	int *arr = new int[n];
	for(int i=0; i<n; i++)cin>>arr[i];
    cout<<coins(arr, n, m)<<endl;
	delete arr;
	
	return 0;
    }
