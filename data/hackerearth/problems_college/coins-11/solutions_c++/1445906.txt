#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,i,f=0;
    	cin>>n;
    	if(n==3)
    	cout<<"3"<<endl;
        else if(n==1)
        cout<<"1"<<endl;
        else
        {
    	
    	for(i=2;i<=(n-1)/2;i++)
    	{
    		if(n%i==0)
    		{
    			f=1;
    			break;
    		}
    	}
    	if(f)
    	cout<<"1"<<endl;
    	else
    	cout<<"2"<<endl;
    }}
    return 0;
}
