noc = input()
while noc:
    noc-=1
    ins = raw_input()
    rins= ins[::-1]
    invalid = False
    for x in range(1,len(ins)):
        if abs(ord(ins[x]) - ord(ins[x-1])) ==  abs(ord(rins[x]) - ord(rins[x-1])):
            continue
        else:
            invalid = True
    print "Not Funny" if invalid else "Funny"
