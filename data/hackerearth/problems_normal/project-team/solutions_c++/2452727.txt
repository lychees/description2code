#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,it,n,i,j;
    cin>>t;
    for(it=1;it<=t;it++){
    	int mx=-1,mn=999999;
    	cin>>n;
    	int arr[n];
    	for(i=0;i<n;i++)
    	cin>>arr[i];
    	sort(arr,arr+n);
    	i=0;
    	j=n-1;
    	while(i<j){
    		if(arr[i]+arr[j]>mx)
    		mx=arr[i]+arr[j];
    		if(arr[i]+arr[j]<mn)
    		mn=arr[i]+arr[j];
    		i++;
    		j--;
    	}
    	cout<<mx-mn<<endl;
    }
}
