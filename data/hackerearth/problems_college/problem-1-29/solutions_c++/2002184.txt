#include<iostream>
using namespace std;
int f=0;
void path(int a[][50],int n,int i,int j)
{
//    cout<<i<<" "<<j<<endl;
    if(a[i][j]==1&&(i==n-1&&j==n-1))
    {
        //cout<<"here";
        f=1;
        return;
    }
    if(a[i][j]==0)
        return ;
    if(i+1<n&&a[i+1][j]==1)
    {
       // cout<<"down:"<<endl;
        path(a,n,i+1,j);
    }
    if(j+1<n&&a[i][j+1]==1)
    {
        //cout<<"right"<<endl;
        path(a,n,i,j+1);
    }
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
      int i=0,j=0,n;
    int a[50][50];
    cin>>n;
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
        cin>>a[i][j];
    path(a,n,0,0);
    if(f==0)
        cout<<"NOT POSSIBLE "<<endl;
    else
        cout<<"POSSIBLE"<<endl;
        f=0;
    }


}
