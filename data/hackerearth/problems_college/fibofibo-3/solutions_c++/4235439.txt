#include <iostream>
using namespace std;

long long int zero,one;

long long int fibo(int n){
	if(n==0){
		zero++;
		return 0;
	}
	if(n==1){
		one++;
		return 1;
	}
	return fibo(n-1)+fibo(n-2);
}

int main()
{
	int t;
	cin>>t;
	
	while(t--){
		int n;
		cin>>n;
		zero=0;
		one=0;
		
		long long int k = fibo(n);
		cout<<zero<<" "<<one<<endl;
	}
	
    return 0;
}
