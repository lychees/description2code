import math


def primes(n):
    sieve = [True] * n
    for i in xrange(3, int(n**0.5)+1, 2):
        if sieve[i]:
            sieve[i*i::2*i] = [False]*((n-i*i-1)/(2*i)+1)
    return [2] + [i for i in xrange(3, n, 2) if sieve[i]]


new = primes(10000000)


n = int(raw_input())
for i in range(1, n):
    if i in new:
        print math.factorial(i)
    else:
        pass
