#include <bits/stdc++.h>
#define ll long long
using namespace std;
const int M = 1e9 + 7;
ll ex(ll a, int n) {
	if(n == 0) return 1;
	if(n == 1) return a;
	if(n % 2 == 0) return ex((a * a) % M, n / 2);
	if(n % 2 == 1) return (a * ex((a * a) % M, (n - 1) / 2)) % M;
}
int main()
{
    int t;
    scanf("%d", &t);
    while(t--) {
    	int n, k;
    	scanf("%d%d", &n, &k);
    	ll a[k][n];
    	for(int i = 0; i < k; i++) {
    		for(int j = 0; j < n; j++) {
    			cin >> a[i][j];
    		}
    	}
    	ll ma = -1, mi = 1e9, l1 = 1e9, l2 = -1;
    	for(int i = 1; i < (1 << n); i++) {
    		ll mid = 1;
    		ll ct = 0;
    		for(int j = 0; j < n; j++) {
    			if(i & (1 << j)) {
    				ll val = 0;
    				for(int l = 0; l < k; l++) {
    					val += a[l][j];
    					val %= M;
    				}
    				mid *= val;
    				mid %= M;
    				ct++;
    			}
    		}
    		ll in = ex(ct, M - 2);
    		mid *= in;
    		mid %= M;
    		if(mid > ma) {
    			ma = mid;
    			l1 = ct;
    		} else if(mid == ma) {
    			l1 = min(l1, ct);
    		}
    		if(mid < mi) {
    			mi = mid;
    			l2 = ct;
    		} else if(mid == mi) {
    			l2 = max(l2, ct);
    		}
    	}
    //	cout << ma << " " << mi << endl;
    //	cout << l1 << " " << l2 << endl;
    	cout << (ma ^ mi) << " " << (l1 ^ l2) << endl;
    }
    return 0;
}
