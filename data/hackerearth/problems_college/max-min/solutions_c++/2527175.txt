#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ll i,t,n,k,ans=1000000000,mini,maxi;
    scanf("%lld%lld",&n,&k);
    vector<ll> v(n);
    for(i=0;i<n;i++)
    	scanf("%lld",&(v[i]));
    sort(v.begin(),v.end());
    for(i=0;i<=n-k;i++)
    {
    	mini=v[i];
    	maxi=v[i+k-1];
    	if(ans>maxi-mini)
    		ans=maxi-mini;
    }
    printf("%d",ans);
    return 0;
}
