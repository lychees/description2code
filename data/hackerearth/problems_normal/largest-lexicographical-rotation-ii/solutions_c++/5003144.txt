#include <bits/stdc++.h>
using namespace std;
const int MAXN = 4e6 + 3;
char s[MAXN];
char t[MAXN * 2];
vector<int> vec;
int mark[MAXN];
void solve() {
	gets(s);
	int n = strlen(s);
	strcpy(t, s), strcpy(t + n, s);
	int i = 0, j = 1, l = 0;
	while (i + l < n) {
		if (t[i + l] == t[j + l]) l++;
		else {
			if (t[i + l] < t[j + l]) i = max(i + l, j), j = i + 1, l = 0;
			else j = j + l + 1, l = 0;
		}
	}
	printf("%s", (s + i)); s[i] = '\0'; puts(s);
}

int main() {
	solve();
	return 0;
}
