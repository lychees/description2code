cache = None
def solve(n, k):
	if k > n:
		return 0
	if k > n/2:
		return 1
	#print "n, k ", n,k
	if cache[n][k] != -1 :
		return cache[n][k]
	_sum = solve(n-k, k) + solve(n, k+1)
	cache[n][k] = _sum
	return _sum
	
T = int(raw_input())
while T>0:
	T -= 1
	N, K = map(int, raw_input().strip().split(" "))
	cache = [[-1]*(N+1) for i in range(N+1)]
	#print "cache ", cache
	print solve(N, K)
