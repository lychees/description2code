k = 1
while (k == 1):
	try:
		line = raw_input()
		if line=='return 0;':
			print line
			break
		if '->' not in line:
			print line
			continue;
		else:
			i = -1
			while (i+2<len(line)):
				i+=1
				if line[i]=='/' and line[i+1]=='/':
					break
				elif line[i]=='-' and line[i+1]=='>':
					if [i-1]=='-':
						continue;
					line = line.replace('->', '.', 1)
		print line
	except EOFError:
		k = 2
