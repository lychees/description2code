#include <algorithm>

#include <string.h>
#include <string>
#include <vector>

using namespace std;

int a[1000];
int b[1000];

int main() {
  int n, c;
  while (cin >> n >> c) {
    int i;
    int x = 1, ans = 0;
    int sum = 0;
    for (i = 0; i < n; i++) {
      cin >> a[i] >> b[i];
      sum += a[i] * x / b[i] + 1;
    }
    while (sum < c) {
      x++;
      sum = 0;
      for (i = 0; i < n; i++)
        sum += a[i] * x / b[i] + 1;
    }
    while (sum == c) {
      ans++;
      x++;
      sum = 0;
      for (i = 0; i < n; i++)
        sum += a[i] * x / b[i] + 1;
    }
    cout << ans << endl;
  }
  return 0;
}
