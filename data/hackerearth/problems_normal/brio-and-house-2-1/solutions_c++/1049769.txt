#include <iostream>
#include<cmath>
using namespace std;

int main() {
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		int x1,y1,x2,y2,r,x,y;
		cin>>x1>>y1>>x2>>y2>>r;
		x=x1*(y2-y1)-y1*(x2-x1);
		y=sqrt((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1));
		long p;
		p=x/y;
		if(p<0) p*=-1;
		if(p<r) cout<<"REPLANNING"<<endl;
		if(p==r) cout<<"JUST MISSED"<<endl;
		if(p>r) cout<<"SAFE"<<endl;
	}
}
