#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t,i,sum;
    char s[100];
    cin>>t;
    while(t--){
    	
    	scanf(" %[^\n]s",&s);
    	sum = 0;
    	for(i=0;i<strlen(s);i++){
    		sum+=s[i];
    	}
    	sum/=strlen(s);
    	
    	if(sum%2==0){
    		cout<<s;
    	}else{
    		for(i=strlen(s)-1;i>=0;i--){
    			cout<<s[i];
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
