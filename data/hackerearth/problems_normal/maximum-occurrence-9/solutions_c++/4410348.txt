#include <iostream>
#include <string>
#include<utility>
#include <algorithm>
#include <cstddef>
using namespace std;

int main()
{
    string s;
    pair<char, int> cur_max('\0', 0);
    size_t i;
    int t;
    getline(cin, s);
    //cout<<s<<endl;
    sort(s.begin(), s.end());
    //cout<<s<<endl;
    i = s.find_last_of(s[0]);
    if(cur_max.second<i+1)
    {
    	cur_max.first=s[0];
    	cur_max.second=i+1;
    	//cout<<cur_max.second<<endl;
    }
    //cout<<cur_max.first<<endl;
    while(i != string::npos && i<s.length()-1)
    {
    	//cout<<"s["<<i+1<<"]="<<s[i+1]<<endl;
    	
    	s=s.substr(i+1);
    	//cout<<s<<endl;
    	i = s.find_last_of(s[0]);
    	//cout<<s[i]<<" "<<i<<endl;
    	if((i+1)>cur_max.second)
    	{
    		//cout<<s[i]<<endl;
    		cur_max.first=s[i];
    		cur_max.second=i+1;
    	}
    }
    cout<<cur_max.first<< " "<<cur_max.second;
    return 0;
}
