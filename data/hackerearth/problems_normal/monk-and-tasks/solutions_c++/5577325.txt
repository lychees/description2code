#include <iostream>
#include <map>
#include <vector>

using namespace std;

int main()
{
	long int n,t,a[100000],b,temp,i;
	cin >> t;
	while(t-->0){
		cin >> n;
		map <int,vector<int> > m;
		map <int,vector<int> > :: iterator it;
		for(i=0;i<n;i++){
			cin >> a[i];
		}
		for(i=0;i<n;i++){
			b=a[i];
			temp=0;
			while(b>0){
				if(b%2==1){
					temp++;
				}
				b=b/2;
			}
			m[temp].push_back(a[i]);
		}
		for(it=m.begin();it!=m.end();it++){
			for(i=0;i<it->second.size();i++)
				cout << it->second[i] << " ";
		}
		cout << endl;
	}
    return 0;
}
