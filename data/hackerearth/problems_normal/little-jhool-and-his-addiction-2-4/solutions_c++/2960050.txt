#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n,k;
		cin >> n >> k;
		int arr[n];
		for(int i=0;i<n;i++)
		{
			cin>> arr[i];
		}
		sort(arr,arr+n);
		int i=0,j=n-1;
		int maxi=INT_MIN,mini=INT_MAX;
		while(i<j)
		{
			int temp=arr[i]+arr[j];
			if(temp>maxi)
			maxi=temp;
			if(mini>temp)
			mini=temp;
			i++;
			j--;
		}
		int ans=maxi-mini;
		cout << ans << endl;
		if(ans>k)
		{
			cout << "No more girlfriends!\n";
		}
		else if(ans==k)
		cout << "Lucky chap!\n";
		else
		cout << "Chick magnet Jhool!\n";
	
	}
		return 0;
}
