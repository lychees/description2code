 #include <iostream>  
 using namespace std;  
 int main()  
 {  
   int t;  
   cin>>t;  
   while(t--){  
        char arr[10000];  
        int a[500]={0},mx=0;  
        cin>>arr;  
        for(int i=0;arr[i]!='\0';i++){  
             int temp = arr[i]-97;  
             a[temp]++;  
             if(a[temp]>mx){  
                  mx = a[temp];  
             }  
        }  
        int flag = 1;  
        for(int i=0;i<26;i++){  
             if(a[i]!=mx){  
                  flag = 0;  
             }  
        }  
        if(flag)  
        {  
             cout<<"Yes"<<endl;  
        }  
        else{  
             cout<<"No"<<endl;  
        }  
   }  
   return 0;  
 }
