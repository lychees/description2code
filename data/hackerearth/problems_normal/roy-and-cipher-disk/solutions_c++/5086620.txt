#include <iostream>
using namespace std;

int main()
{
	unsigned int t;
	string s;
	cin>>t;
	for(unsigned int i=0;i<t;i++){
		cin>>s;
		char current_pos = 'a';
		for(unsigned int j=0;j<s.length();j++){
			int clockwise = s[j] - current_pos;
			int anticlockwise = current_pos - s[j];
			if(clockwise < 0) {
				clockwise = ('z' - current_pos) + (s[j] - 'a') + 1;
			}
			if(anticlockwise < 0){
				anticlockwise = ('z' - s[j]) + (current_pos - 'a') + 1;
			}
			if(clockwise > anticlockwise){
				cout<<-anticlockwise<<" ";
			}else{
				cout<<clockwise<<" ";
			}
			current_pos  = s[j];
		}
		cout<<endl;
	}
	return 0;
}
