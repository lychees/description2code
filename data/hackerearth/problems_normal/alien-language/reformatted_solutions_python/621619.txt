t = input()
for _ in xrange(t):
    t = raw_input().strip()
    p = raw_input().strip()
    d = {}
    for c in p:
        if d.has_key(c):
            continue
        if t.find(c) != -1:
            print "YES"
            d[c] = True
            break
        d[c] = False
    else:
        print "NO"
