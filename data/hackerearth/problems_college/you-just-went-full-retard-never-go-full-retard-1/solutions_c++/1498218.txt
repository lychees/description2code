#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long n;
    	cin>>n;
    	int f=0;
    	
    	if((n%2==0&&n!=2))
    	f=1;
    	for(int i=3;i<=sqrt(n)&&f==0;i+=2)
    	{
    		if(n%i==0)
    		f=1;
    	}
    	if(n==1)
    	f=1;
    	if(f==0)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
    return 0;
}
