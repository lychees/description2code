#include <iostream>
using namespace std;

int main()
{
    int a,b,c,t;
    
    
    cin >> t;
    while(t--)
    {
    	cin >> a >> b >> c;
    	int d = a/b;
    	int arr[c];
    	int dv;
    	int rem = a % b;
    	int i,cnt;
    	cnt = 0;
    	
    	
    	if(c != 0)
    	{
    		cout << d << ".";
    		while(c--)
        	{
        		//cout << "rem " << rem << endl;
        		rem = rem * 10;
        		dv = rem / b;
        		//cout << "dv " << dv << endl;
        		rem = rem % b;
        		//cout << "rem after " << rem << endl;
        		arr[cnt] = dv;
        		cnt++;
        	}
    		for(i = 0;i < cnt;i++)
    		{
    			cout << arr[i];
			}
			cout << endl;
		}
		else
		{
			cout << d << endl;
		}
    }
    return 0;
}
