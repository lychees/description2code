#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll dp[10004], a[10004], n ;

ll solve(ll i){
	if(i > n)
	return 0;
	
	if(dp[i] != -1)
	return dp[i];
	
	return 
	dp[i] = max(solve(i+2) + a[i], solve(i+1)) ;
}

int main()
{
    ll t, k = 1;
    cin >> t;
    while(t--){
    	
    	cin >> n;
    	
    	for(ll i = 1; i <= n; i++){
    		cin >> a[i] ;
    	}
    	
    	for(ll i = 1; i <= n; i++){
    		dp[i] = -1 ;
    	}
    	
    	cout << "Case " << k++ << ": " << solve(1) << endl ;
    	
    }
    return 0;
}
