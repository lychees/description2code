#include<bits/stdc++.h>
#define ull unsigned long long int
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        ull a,b,c;
        cin>>a>>b>>c;
        if((a+b)>c && (a+c)>b && (b+c)>a && (a!=0) && b!=0 & c!=0)
        {
            if(a==b && b==c && c==a)
                cout<<"YES. It is a equilateral triangle."<<endl;
            else if(a!=b && b!=c && c!=a)
                cout<<"YES. It is a scalene triangle."<<endl;
            else
                cout<<"YES. It is a isosceles triangle."<<endl;
        }
        else
            cout<<"NO"<<endl;
    }
}
