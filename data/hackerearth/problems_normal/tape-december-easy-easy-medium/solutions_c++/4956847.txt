#include <iostream>
using namespace std;

int add_sum(int a[], int index, int k){
	int sum = 0;
	for(int j=index; j<k; j++){
		sum += a[j];
	}
	return sum;
}


int main()
{
	int N,K,D;
	cin>>N>>K>>D;
	int a[N];
	for(int i=0; i<N; i++){
		cin>>a[i];
	}
	int opt = 0;
	bool flag = false;
	int i = 0;
	int sum = add_sum(a, 0, K);
	//for(int j=K; j<N; j++){
		i=1;
		int j=K;
		if(N-3*K-1 <= 0){
			flag = true;
			cout<<1<<endl;
		}
			
		while(j <= N && !flag){
			opt++;
			while(j <= i*(2*K-1) && j<=N){
				if (sum <= 0){
					cout<<(-1)<<endl;
					flag = true;
					break;
				}	
				sum = sum - a[j-K] + a[j];
				j++;
			}
			i++;
		}
		if(!flag) cout<<opt<<endl;
	//}
	
    return 0;
}
