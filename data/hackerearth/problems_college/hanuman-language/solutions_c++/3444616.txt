#include <iostream>
#include <sstream>
#include <string>
#include <vector>
using namespace::std;

int main()
{
    // Our test string
    std::string str;
    int t;
    cin>>t;
    t++;
    while(t--)
    {
     getline(std::cin, str);
     std::vector<std::string> words;
     std::istringstream iss(str);
     for(std::string word; iss >> word; words.push_back(word));
     std::string rwstr;
     for(int i = words.size() - 1; i >= 0; rwstr += words[i--] + " ");
     std::cout <<rwstr<<"\n";
    }
    return 0;
}
