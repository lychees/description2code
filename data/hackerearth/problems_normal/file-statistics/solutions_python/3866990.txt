import os

def fileman():
	path = raw_input()

	f = open(path)

	lc = 0
	wc = 0
	cc = 0

	for line in f:
		lc = lc + 1
		wc = wc + (len(line.split(" ")))
		cc = cc + (len(line))
	
	f.close()
	
	print str(lc)+"\n"+str(wc)+"\n"+str(cc)

	info = os.stat(path)

	print str(info[4])+"\n"+str(info[5])+"\n"+str(info[8])
