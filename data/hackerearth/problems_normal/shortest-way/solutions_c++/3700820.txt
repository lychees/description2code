#include <iostream>
#include <cassert>
using namespace std;
typedef long long int lli;
void minOperation(lli a, lli b){
	lli i=0;
   while(a>0 && b>0){
			if(a>b){

				i+=a/b;
				a%=b;

			} else {

				i+=b/a;
				b%=a;

			}
		}
		i--;
		cout << i << endl;
}
int main(){
    lli t;
    cin >> t;
    assert(t>=1 and t<=10);
    lli a,b;
    for(int i=0;i<t;i++){
        cin >> a >> b;
        assert(a>=1 and b>=1);
        minOperation(a, b);
    }

    return 0;
}
