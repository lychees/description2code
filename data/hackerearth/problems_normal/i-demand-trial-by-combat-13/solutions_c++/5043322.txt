#include <iostream>
using namespace std;

int main()
{
    int t,a[102],m,n,i,j,prev;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	for(i=0;i<m;i++)
    	{
    		for(j=0;j<n;j++)
    		if(a[j]!=0)
    		 break;
    		 if(j==n)
    		 goto end;
    		prev=-1;
    		for(j=0;j<n;j++)
    		{
    			if(j<n-1)
    			{	if(prev==-1 && a[j+1]==1)
    				{
    				prev=a[j];
    				a[j]=1;
    				}
    				else if(prev==1 && a[j+1]==1)
    				{
    				prev=a[j];
    				a[j]=1;
    				}
    				else 
    				{
    					prev=a[j];
    					a[j]=0;
    					
    				}
    			
    			}
    			else if(prev==1)
    			a[j]=1;
    			else a[j]=0;
    		}
    		
    	}
    	end:
    	for(i=0;i<n;i++)
    	cout<<a[i]<<" ";
    	cout<<endl;
    	
    }
    return 0;
}
