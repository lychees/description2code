#include <bits/stdc++.h>
using namespace std;
int a[1000001];
int main()
{
    int t,n,i;
    cin>>t;
    while(t--)
    	{
    		cin>>n;
    		for(i=1;i<=n;i++)
    			cin>>a[i];
    		sort(a+1,a+n+1);
    		if((a[n]-a[1])%2==1)
    			cout<<"Yes\n";
    		else
    			cout<<"No\n";
    	}
    return 0;
}
