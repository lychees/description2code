#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long int a[10000],sum1,n,m,t,i,c=0;
    cin>>t;
    while(t--)
    {sum1=0;c=0;
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>m;
    		sum1=sum1+m;
    	}
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	sort(a,a+n);
    	for(i=n-1;i>=0;i--)
    	{
    		if(sum1>0)
    		{
    			sum1=sum1-a[i];
    			c++;
    		}
    		else
    		break;
    	}
    	cout<<c<<endl;
    }

return 0;
}
