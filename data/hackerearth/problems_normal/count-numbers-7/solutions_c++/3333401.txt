#include <iostream>
#include <cmath>
using namespace std;

long int multiple(long int s, long int f, long int n)
{
	long int fn, sn;
	sn = ceil(s / (double)n) * n;
	fn = floor(f / (double)n) * n;
	
	//cout<<sn<<" "<<fn<<" "<<(fn - sn) / n + 1<<endl;
	
	if(s <= sn && fn <= f)
		return (fn - sn) / n + 1;
	else
		return 0;
}

int main()
{
	long long int k, t, from, to, c = 0, m = 1, count = 0;
	cin>>k>>t;
	
	int a[k];
	
	for(int i = 0; i < k; i++)
		cin>>a[i];
	
		
	while(t--)
	{
		cin>>from>>to;
		
		c = 0;
		
		for(int i = 1; i < (1<<k); i++)
		{
			count = 0;
			m = 1;
			for(int j = 0; j < k; j++)
				if(i & (1<<j))
				{
					//cout<<a[j]<<" ";
					m *= a[j];
					count++;
				}
				
			if(count & 1)
				c += multiple(from, to, m);
			else
				c -= multiple(from, to, m);
			//cout<<" "<<m<<endl;
		}
		
		cout<<c<<endl;
	}
    return 0;
}
