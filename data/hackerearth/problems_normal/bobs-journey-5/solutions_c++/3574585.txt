#include <iostream>
#include <cassert>

using namespace std;


#define assn(n,a,b) assert(n>=a && n<=b)


int main()
{
    int t;
    cin >> t;
    assn(t,1,100);
    while(t--)
    {
        int n,i,flag=0,arr[26]={};
        string s;
        cin >> n;
        assn(n,1,1000);
        for(i=0; i<n; i++)
        {
            cin >> s;
            assn(int(s.size()),1,10);
            arr[s[0]-'a']++;
        }
        for(i=0; i<26; i++)
            if(arr[i]>1)flag=1;
        if(flag==0)
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
