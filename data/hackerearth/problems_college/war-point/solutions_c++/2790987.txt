#include <iostream>
#include<cmath>
using namespace std;

int main()
{
	long long int t;
	cin>>t;
	while(t--)
	{
		long long int n;
		cin>>n;
		long long int a[n],b[n],c[n],i,sum[n];
		cin>>a[0];
		sum[0]=a[0];
		for(i=1;i<n;i++)
		{
			cin>>a[i];
			sum[i]=sum[i-1]+a[i];
		}
		for(i=0;i<n;i++)
		{   
			b[i]=sum[i];
			c[i]=sum[n-1]-b[i];
		}
		long long int min;
		min=abs(c[0]-b[0]);
		for(i=1;i<n;i++)
		{
			if(abs(c[i]-b[i])<min)
			min=abs(c[i]-b[i]);
			
		}
		cout<<min<<endl;
	}
	
	return 0;
}
