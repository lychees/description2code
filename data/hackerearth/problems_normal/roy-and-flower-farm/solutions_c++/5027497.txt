#include <bits/stdc++.h>
#define mp make_pair
using namespace std;
pair <int,int> dp[105][10005];
void init()
{
    int n,p;
    int cost[105],prof[105];
    cin>>n>>p;
    for(int i=1;i<=n;i++)
    {
        int x,y;
        cin>>x>>y;
        prof[i] = x - y;
        cost[i] = y;
    }


    for(int i=0;i<=p;i++)
        dp[0][i] = mp(0,0);

    for(int i=0;i<=n;i++)
        dp[n][0] = mp(0,0);

    for(int i=1;i<=n;i++)
        for(int inv=1;inv<=p;inv++)
        {
            if(cost[i]>inv)
                dp[i][inv] = dp[i-1][inv];
            else
            {
                int pro1 = prof[i] + dp[i-1][inv-cost[i]].first;
                int pro2 = dp[i-1][inv].first;
                int c1 = cost[i] + dp[i-1][inv-cost[i]].second;
                int c2 = dp[i-1][inv].second;
                if(pro1>pro2)
                    dp[i][inv] = mp(pro1,c1);
                else if(pro2>pro1)
                    dp[i][inv] = mp(pro2,c2);
                else
                    dp[i][inv] = mp(pro1,min(c1,c2));
            }
        }
     cout<<dp[n][p].second<<" "<<dp[n][p].first + p<<endl;
}

int main()
{
    int t;
    cin>>t;

    for(int i=0;i<t;i++)
        init();
    return 0;
}
