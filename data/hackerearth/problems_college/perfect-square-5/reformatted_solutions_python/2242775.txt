for _ in range(int(raw_input())):
    sqval = int(raw_input())
    val = sqval ** 0.5
    if int(val) ** 2 == sqval:
        print "YES"
    else:
        low = int(val) ** 2
        high = (int(val) + 1) ** 2
        lowdiff = sqval - low
        highdiff = high - sqval
        if lowdiff < highdiff:
            print "-%i" % lowdiff
        else:
            print "+%i" % highdiff
