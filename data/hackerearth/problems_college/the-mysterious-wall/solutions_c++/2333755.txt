#include <iostream>
#include <string.h>

using namespace std;

int main()
{
    int t;
    cin>>t;
    bool flag;
    string a,b;
    int co[26],temp[26];
    while(t--)
    {
        cin>>a;
        cin>>b;
        int len1 = b.length();
        memset(co,0,sizeof(co));
        for(int i=0;i<len1;i++)
        {
            co[b[i]-'a']++;
        }
        memset(temp,0,sizeof(temp));
        int coo = 0,st=0;
        flag = false;
        int len = a.length();
        for(int i=0;i<len;i++)
        {
                temp[a[i]-'a']++;
                coo++;
            if(temp[a[i]-'a']  <= co[a[i]-'a'])
            {


            }
            else
            {
                while(temp[a[i]-'a'] > co[a[i]-'a'])
                {
                    coo--;
                    temp[a[st]-'a']--;
                    st++;
                }
            }
            if(coo == len1)
            {
                flag = true;
            }
        //    cout<<coo<<" "<<st<<endl;
        }
        if(flag)
        {
            cout<<"YES"<<endl;
        }
        else
        {
            cout<<"NO"<<endl;
        }
    }
    return 0;
}
