#include<bits/stdc++.h>

#define ll long long

typedef struct p{
	double first;
	double second;
}p;
 
	bool cmp(p s1,p s2)
	{
		if(s1.first!=s2.first)
		return (s1.first<s2.first);
		else
			return s1.second<s2.second;
	}
int main()
{
	ll n,i;
	cin>>n;
	string s;
	p a[n];
 
	for(i=0;i<n;i++)
	{
		cin>>s;
		double a1=(double)(s[0]-48)*10+(double)(s[1]-48)+(double)(s[3]-48)/10+(double)(s[4]-48)/100;
		double b1=(double)(s[6]-48)*10+(double)(s[7]-48)+(double)(s[9]-48)/10+(double)(s[10]-48)/100;
		a[i].first=a1;
		a[i].second=b1;
		
	}
	sort(a,a+n,cmp);
	ll flag=0;
	for(i=1;i<n;i++)
	{
		if(a[i].first<a[i-1].second)
		{
			flag=1;
			break;	
		}	
	}
	if(flag==1)
	cout<<"Will need a moderator!"<<endl;
	else 
	cout<<"Who needs a moderator?"<<endl;
	return 0;
}
