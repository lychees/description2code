#define N 200000
#define ln 15

ll i,j,k,n,t,x[N],y[N],p,q,s,r;
int main(){
	inpll(t);
	while(t--){
		inpll(n);
		rep(i,0,n) inpll(x[i]);
		rep(i,0,n) inpll(y[i]);
		inpll(s); inpll(r);
		inpll(p); inpll(q);
		ll mini=LONG_MAX;
		rep(i,0,n){
			ll t1=(abs(x[i])+abs(y[i]))*p;
			ll t2=(abs(s-x[i])+abs(r-y[i]))*q;
			mini=min(t1+t2,mini);
		}
		ll tt=((abs(s)+abs(r))*p);
		mini=min(tt,mini);
		cout<<mini<<'\n';
	}
	return 0;
}
