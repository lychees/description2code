#include <iostream>
#include <stdio.h>
#include <string.h>

using namespace std;

int main () {
	
char s[110];
gets(s);	
	
int max=0,sum,n;
int i,j,k,t;
char mch;
n = strlen(s);
sum=0;

for(i=0;i<n;i++) {
     sum += (int)s[i];
     k=0;
     for(j=0;j<n;j++)  {
         if(s[i]==s[j]) k++;	
     	
       }
       if(k>max) { max = k; mch = s[i]; }
     
 }

sum /= n;
	
if(sum%2==1) {
	cout << mch;	
  }
	
else 
  for(i=n-1;i>=0;i--)
  cout << s[i];
	
}
