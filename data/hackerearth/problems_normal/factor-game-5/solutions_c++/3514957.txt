#include <bits/stdc++.h>

using namespace std;

int vis [100000 + 10];

int N = 100000 + 10;

void sieve(){

    for(long long i = 2; i <= N; i++){

        int num = i;

        for(long long j = 2; j * j <= N && num != 1; j++){

            while(num % j == 0){

                vis[i]++;
                num /= j;
            }
        }

        if(num != 1) vis[i]++;
    }
}


int main()
{

    int n, sumOfPrimes = 0, even = 0, odd = 0, tem1, ans = 0;

    sieve();

    cin >> n;

    for(int i = 0; i < n; i++){

        cin >> tem1;
        ans ^= vis[tem1];
    }

    if (ans == 0) cout << "BOB\n";
    else cout << "ALICE\n";

    return 0;
}
