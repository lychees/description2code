#include<iostream>
#include<math.h>
#include<string.h>

using namespace std;

int main()
{
    int k;
    cin>>k;
    char ch[k][100];
    for(int i=0;i<k;i++)
    {
        cin>>ch[i];
    }
    for(int i=0;i<k;i++)
    {
        for(int j=i+1;j<k;j++)
           {
               if(strcmp(ch[i],ch[j])>0)
                swap(ch[i],ch[j]);
    }
}
for(int i=0;i<k;i++)
    {
        for(int j=i+1;j<k;j++)
           {
               if(strcmp(ch[i],ch[j])==0)
                strcpy(ch[j]," ");
    }
}
for(int i=0;i<k;i++)
    {
        if(strcmp(ch[i]," ")!=0)
            cout<<ch[i]<<endl;
    }
}
