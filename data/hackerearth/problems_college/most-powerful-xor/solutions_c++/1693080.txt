#include <iostream>
using namespace std;

int main()
{
    long t, l, r, i, maxor;
    cin >> t;
    
    while (t--) {
    	cin >> l >> r;
    	maxor = 0;
    	for(; l <= r; l++)
    		for(i = l; i <= r; i++)
    			if((l ^ i) > maxor)
    				maxor = l ^ i;
    				
    	cout << maxor << endl;
    			
    }
    
    return 0;
}
