#include<bits/stdc++.h>
using namespace std;


bool bs(long long a[],long long l,long long r,long long key)
{
    if(l<=r)
    {
        int mid=l+(r-l)/2;

        if(a[mid]==key)
            return true;

        else if(a[mid]<key)
            return bs(a,mid+1,r,key);

        else
            return bs(a,l,mid-1,key);

    }

    else
        return false;

}

int main()
{

    long long n,q;
    cin>>n>>q;
    long long a[n],i,j;

    for(i=0; i<n; i++)
        cin>>a[i];
        
     sort(a,a+n);   

    while(q--)
    {
        long long temp;
        cin>>temp;

        if(bs(a,0,n-1,temp)==true)
        {
            cout<<"YES\n";
        }
        else
            cout<<"NO\n";

    }
    return 0;
}
