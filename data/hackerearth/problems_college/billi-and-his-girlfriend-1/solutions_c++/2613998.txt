#include <bits/stdc++.h>
using namespace std;
#define ll long long

ll power(ll a,ll b,ll mod)
{
	if(b==0)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2,mod);
	temp=(temp*temp)%mod;
	if(b&1)
		temp=(temp*a)%mod;
	return temp;
}
int main()
{
    ll ans,a,b,k,t,r,c,temp,cal,i,val,mod;
    scanf("%lld",&t);
    while(t--)
    {
    	cal=0;
    	c=0;
    	scanf("%lld%lld%lld",&a,&b,&k);
    	val=10;
    	mod=1;
    	for(i=1;i<=k;i++)
    	{
    		mod=mod*val;
    	}
    	ans=power(a,b,mod);
    	temp=ans;
    	while(temp>0)
    	{
    		r=temp%10;
    		cal+=r;
    		c++;
    		
    		temp=temp/10;
    	}
    	printf("%lld\n",cal);
    }
    return 0;
}
