#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>
using namespace std;
int convert(int n)
{
	int sum =0;
	int i=0;
	while(n!=0)
	{
		sum = sum + (n%10)*pow(2,i);
		n = n/10;
		i++;
	}
	return sum ;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int m,n;
		cin>>m>>n;
		int A = convert(m);
		int B = convert(n);
		if(A>2 && B>2)
		{
			if(A < B)
			{
				cout<<"The Mountain"<<"\n";
			}
			else
			{
				cout<<"The Hound"<<"\n";
			}
		}
		else
		{
			if(pow(A,B)>pow(B,A))
			{
				cout<<"The Mountain"<<"\n";
			}
			else
			{
				cout<<"The Hound"<<"\n";
			}
		}
	}
}
