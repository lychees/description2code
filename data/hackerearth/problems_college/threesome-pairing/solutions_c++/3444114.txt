#include<bits/stdc++.h>
using namespace std;
int main()
{
	//freopen("cook.txt","r",stdin);
	long long int i,j,t,n,ar[101],co;
	scanf("%lld",&t);
	while(t--)
	{
		co=0;
		scanf("%lld",&n);
		for(i=0;i<n;i++)
		{
			scanf("%lld",(ar+i));
		}
		for(i=0;i<n;i++)
		{
			for(j=i+1;j<n;j++)
			{
				if((ar[i]+ar[j])%3==0)
				++co;
			}
		}
		printf("%lld\n",co);
	}
	return 0;
}
