#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define ll long long 
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s1,s2;
		cin>>s1>>s2;
		reverse(s2.begin(),s2.end());
		reverse(s1.begin(),s1.end());
		
		ll  ctr=0;
		ll carry=0;
		for(int i=0;i<min(s1.size(),s2.size());i++)
		{
			if((s1[i]-'0')+(s2[i]-'0')+carry>9)
			{
				ctr++;
				carry=1;
			}
			else
			{
				carry=0;
			}
		}
		if(ctr==0)
		{
			cout<<"No carry operation\n";
		}
		else if(ctr==1)
		cout<<ctr<<" "<<"carry operation\n";
		else
		cout<<ctr<<" "<<"carry operations\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
