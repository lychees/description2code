#include <bits/stdc++.h>
using namespace std;
long long int gcd(long long int a,long long int b) {
    return b == 0 ? a : gcd(b, a % b);
}

int main() {
	long long int t,i,j,k,l,m,n,o,p;
	vector <bool> a(100000,true);
	a[0]=a[1]=false;
	for(i=2;i<100000;i++)
	{
		if(a[i])
		{
			for(j=2;i*j<100000;j++)
			a[i*j]=false;
		}
	}
	cin>>t;
	while(t--){
		cin>>n;
		k=1;
		for(i=2;i<n;i++)
		{
			j=gcd(n,i);
			if(j==1)
			k++;
		}
			//cout<<k;
			if(a[k])
			cout<<"Correct\n";
			else
			cout<<"Incorrect\n";
	}
}
