# rounds
n = int(raw_input())

# total matchers
total_matches = pow(2, n) - 1

winners = set()
losers = set()

for _ in xrange(total_matches):
	winner, loser = [x for x in raw_input().split()]
	winners.add(winner)
	losers.add(loser)


ans = None

for winner in winners:
	if winner not in losers:
		ans = winner
		break

print(ans)
