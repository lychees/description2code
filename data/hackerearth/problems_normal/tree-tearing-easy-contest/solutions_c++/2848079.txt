#include <bits/stdc++.h>

using namespace std;

int K, N;
vector<int> adj[10001];
int ans;

int solve(int u)
{
    int x=1;
    for(auto& v: adj[u])
        x+=solve(v);
    if(x>=K)
    {
        ans++;
        x=0;
    }
    return x;
}

int main()
{
    scanf("%d%d", &K, &N);
    int a;
    for(int i=2; i<=N; i++)
    {
        scanf("%d", &a);
        adj[a].push_back(i);
    }
    solve(1);
    printf("%d\n", ans);
    return 0;
}
