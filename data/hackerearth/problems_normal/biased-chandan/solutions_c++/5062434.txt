#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int n;
	cin>>n;
	
	long int arrC=0;
	long int arr[n];
	for(long int i=0;i<n;i++)
	{
		long int a;
		cin>>a;
		if(a==0)
		{
			if(arrC!=0)
				arrC--;
		}
		else
		{
			arr[arrC++]=a;
		}
	}
	cout<<accumulate(arr,arr+arrC,0)<<endl;
	return 0;
}
