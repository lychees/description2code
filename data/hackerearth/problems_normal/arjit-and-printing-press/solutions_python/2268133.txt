for x in xrange(int(raw_input())):
    word = list(raw_input())
    reserve = list(raw_input())
    reserve.sort()
    result = ""
    i = 0
    j = 0
    while len(result) < len(word):
        if j < len(reserve) and word[i] < reserve[j]:
            result += word[i]
            i += 1
        elif j < len(reserve) and word[i] > reserve[j]:
            result += reserve[j]
            i += 1
            j += 1
        else:
            result += word[i]
            i += 1
    print result
