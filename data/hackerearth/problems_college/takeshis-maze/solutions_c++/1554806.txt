#include <iostream>
using namespace std;

int checkMaxcoins(int arr[101][101],int N,int C)
{
	for (int i=N-1;i>=0;--i)
		for(int j=N-1;j>=0;--j)
			arr[i][j] += (arr[i+1][j] > arr[i][j+1]?arr[i+1][j]:arr[i][j+1]);
	return (C > arr[0][0])?-1:(arr[0][0] - C);
}
int main()
{
	int N,C, test_case,Answer;
    cin>>test_case;
	while(test_case--)
	{
		Answer = -1;
		int arr[101][101] = {0};
		cin >> N >> C;
		if (N > 1 && N <= 100)
		{
			for (int i=0;i<N;++i)
				for(int j=0;j<N;++j)
					cin>>arr[i][j];
			Answer = checkMaxcoins(arr,N,C);
		}
		cout<<Answer<<endl;
	}
    return 0;
}
