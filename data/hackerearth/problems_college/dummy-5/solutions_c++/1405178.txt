#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int fn( int a[][3],int n , int x , int y , int z)
{
	if(x==0 && y == 0 && z==0)
	return 1;
	if(n==0 && (x!=0 || y!=0 || z!=0 ) )
	     return 0;
      if(a[n-1][0] > x  || a[n-1][1] > y || a[n-1][2] > z)
	  		 return fn(a, n-1,x,y,z);
  return fn(a,n-1,x,y,z)|| fn(a,n-1,x-a[n-1][0],y-a[n-1][1],z-a[n-1][2]);			   	
}
int main()
{   
      int t,tt=1,i,j;
      scanf("%d",&t);
      while(t--)
         {
         	int n,x,y,z;
          scanf("%d%d%d%d",&x,&y,&z,&n);
          int a[n+1][3];
		  for(i = 0 ; i < n ; ++i )
		  	{
  			for(j = 0 ; j < 3 ; ++j )
  			   scanf("%d",&a[i][j]);
		  }
  			if(fn(a,n,x,y,z)==1)
			  printf("Case #%d: yes\n",tt++);
			  else
			  printf("Case #%d: no\n",tt++);   
  		}       	
      return 0;
}
