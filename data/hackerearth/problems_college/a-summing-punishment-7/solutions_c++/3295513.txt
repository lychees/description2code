#include <bits/stdc++.h>
using namespace std;
int main() {
    int n,i;
    int a[100005] = {0};
    cin >> n;
    for(i = 0; i < n; i++) {
        cin >> a[i];
    }
    int q;
    long long int sum;
    cin >> q;
    while(q--) {
        sum = 0;
        int l,r;
        cin >> l >> r;
        for(i = l-1; i < r; i++) {
            sum  = sum + a[i];
        }
        cout<<sum<<endl;
    }

    return 0;
}
