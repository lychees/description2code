#include <iostream>
using namespace std;

int Count( int n ) {
	
	int count = 0;
	
	while( n ) {
		n = n & ( n - 1 );
		count++;
	}
	return count;
}

int main()
{
	ios::sync_with_stdio( false );
	int t, n;
	
	cin >> t;
	
	while( t-- ) {
		
		cin >> n;
		
		if( Count( n ) & 0x1 )
			cout << "odd\n";
		else
			cout << "even\n";
	}
	
	return 0;
}
