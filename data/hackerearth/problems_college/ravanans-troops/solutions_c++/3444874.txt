#include <iostream>
using namespace std;

int main()
{
    int t=0;
    scanf("%d",&t);
    while(t--)
    {
    	int n=0;
    	scanf("%d",&n);
    	int dp[n],x=0,temp=0,i=0;
    	
    	dp[0]=0;
    	dp[1]=1;
    	dp[2]=2;
    	dp[3]=3;
    	
    	for(i=4;i<=n;i++)
    	{
    		dp[i]=i;
    		
    		for(x=1;x<=i;x++)
    		{
    			temp = x*x;
    			if(temp > i)break;
    			
    			else
    			dp[i] = min(dp[i],1+dp[i-temp]);
    		}
    	}
    	cout<<dp[n]<<endl;
    }
    return 0;
}
