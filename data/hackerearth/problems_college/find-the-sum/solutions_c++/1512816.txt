#include<bits/stdc++.h>
using namespace std;

inline int scan(){
char c = getchar_unlocked();
int x = 0;
while(c<'0'||c>'9'){
c=getchar_unlocked();
}
while(c>='0'&&c<='9'){
x=(x<<1)+(x<<3)+c-'0';
c=getchar_unlocked();
}
return x;
}
long long solve (long long n)
{
    long long i,sum;
    for(i=1;n!=0;i++)
    {
        if(n-i>=0){
                n-=i;
        if(i%2!=0)
        {
            sum=sum+i;
        }
        }
        else
        {
          if(n%2!=0)
          {
              sum=sum+i;
              
          }
          n=0;
        }
    }
    return sum;
}
int main()
{
    int t=scan();
    while(t--)
    {
       long long l,r;
       scanf("%Ld %Ld",&l,&r);
       long long s1=solve(r);
        long long s2=solve(l-1);
        printf("%Ld\n",s1-s2);
    }
    return 0;
}
