n = input()
a = map(int, raw_input().split())
m = input()
b = [
    a[x] +
    a[y] +
    a[z] for x in xrange(
        n -
        2) for y in xrange(
            x +
            1,
            n -
            1) for z in xrange(
                y +
                1,
        n)]
print "True" if m in b else "False"
