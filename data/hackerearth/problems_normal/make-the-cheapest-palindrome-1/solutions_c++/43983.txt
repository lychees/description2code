#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cstring>

using namespace std;

int main(){
	int t;
	scanf ("%d", &t);
	while (t--){
		char input[10001];
		scanf ("%s", input);
		int a, b;
		scanf ("%d %d", &a, &b);
		int A[] = {a, b};
		int min = a;
		if (a > b)
			min = b;
		int len = strlen (input);
		int mid = len/2;
		int total = 0;
		for (int i = 0; i < mid; i++){
			if (input[i] == input[len -1 -i]){
				if (input[i] == '/')
					total += 2 * min;
			}
			else {
				if (input[i] == '/' && (input[len-1-i] == 'a' || input[len-1-i] == 'b')){
					total += A[input[len-1-i]-'a'];	
				}
				else if (input[len -1 -i] == '/' && (input[i] == 'a' || input[i] == 'b')){
					total += A[input[i] - 'a'];
				}
				else {
					total = -1;
					break;
				}
			}
		}
		printf ("%d\n", total);
	}
}
