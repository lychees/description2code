#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int T;
    cin>>T;
    int Ar[T];
    for(int i=0;i<T;i++)
    	cin>>Ar[i];
    long long maxi=0,sum=0;
    for(int i=0;i<T;i++)
    {
    	if(sum+Ar[i]>=0)
    	{
    		sum+=Ar[i];
    		maxi=max(maxi,sum);
    	}
    	else
    	{
    		sum=0;
    	}
    }
    cout<<maxi<<"\n";
    return 0;
}
