const ll mod = 1000000007L;

using namespace std;
typedef pair<ll, ll>pll;
int ar[2][1000005];
string s;
int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt", "r", stdin);
#endif
	ios_base::sync_with_stdio(0); cin.tie(0);
	int t;
	int cur = 0, prev = 1;
	cin >> t;
	while (t--)
	{
		int n, m, p, x;
		cin >> n >> m >> p >> x;
		cin >> s;
		n--;
		for (int i = 0; i < m; i++)
		{
			ar[cur][i] = s[i] - '0';
		}
		int k = 0;
		while (n)
		{
			if (n & 1)
			{
				ll temp = (1ll << k) % m;
				temp = (temp * p) % m;
				cur = 1 - cur;
				prev = 1 - prev;
				for (int i = 0; i < m; i++)
				{
					ar[cur][i] = ((ar[prev][i] ^ ar[prev][(i - temp + m) % m]) ^ x);
				}
			}
			n >>= 1;
			k++;
		}
		for (int i = 0; i < m; i++)
		{
			cout << ar[cur][i];
		}
		cout << endl;
	}
	return 0;
}
