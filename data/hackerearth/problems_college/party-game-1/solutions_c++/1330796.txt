#include <bits/stdc++.h>
using namespace std;
typedef int ll;
const ll N = 1<<18;
const ll INF = 1000000000;
const ll md =1000*1000*1000;
ll x,y,n,c,j,a[100001],t;
int main()
{
  //freopen("input.txt","r",stdin);
 //freopen("output.txt","w",stdout);
 cin>>t;
    while (t--)
    {
        cin>>x>>y>>n;
        c=j=0;
        for (int i=1;i<=n;i++)
            cin>>a[i];
        sort (a+1,a+n+1);
        for (int i=1;(x>=a[i] && i<=n);i++)
            x-=a[i],c++;
        for (int i=1;(y>=a[i] && i<=n);i++)
            y-=a[i],j++;
            if (c>j)
                puts("Chef Won");
            if (c<j)
                puts("Jhool Won");
            if (c==j)
                puts("Tie");
    }
}
