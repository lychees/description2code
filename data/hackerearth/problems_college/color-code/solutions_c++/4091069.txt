#include<stdio.h>
#include<string.h>

char x[3];

void solve(int c)
{
    sprintf(x,"%X",c);
    int len=strlen(x);
    if(x[0]>='A'&&x[0]<='F')
        x[0]+=20;
    if(x[1]>='A'&&x[1]<='F')
        x[1]+=20;
    if(len==1)
        x[1]=x[0],x[0]='0';
    printf("%s ",x);
}

int main()
{
    int t,r,g,b,i=1;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d%d",&r,&g,&b);
        printf("Case %d: ",i);
        solve(r);
        solve(g);
        solve(b);
        printf("\n");
        i++;
    }
    return 0;
}
