#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define MAX 1000000003
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    ll P,Q,A,B,C,N;
    while(t--)
    {
    	scanf("%lld %lld %lld %lld",&N,&A,&B,&C);
    	ll P[N+1],Q[N+1];
    	for(int i=0;i<N;i+=1)
    	{
    		scanf("%lld %lld",&P[i],&Q[i]);
    	}
    	P[N]=MAX;
    	Q[N]=MAX;
    	sort(P,P+N);
    	sort(Q,Q+N);
    	ll curr=N*A,ans=N*A;
    	int i=0,j=0;
    	while(i<N || j<N)
    	{
    		if(P[i]<=Q[j]) 
    		{
    			curr+=(B-A);
    			i+=1;
    		}
    		else 
    		{
    			curr+=(C-B);
    			j+=1;
    		}
    		
    		ans=max(ans,curr);
    	}
    	printf("%lld\n",ans);    	
    }
    
    return 0;
}
