#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    
    int a[n];
    
    for(int i=0;i<n;i++)
    	cin>>a[i];
    	
    int m1 = 0;
    int m2 = 0;
    
    for(int i=0;i<n;i++)
    {
    	m1 = m1+a[i];
    	
    	if(m1<0)
    	{
    		m1 = 0;
    	}
    	
    	if(m2<m1)
    	{
    		m2 = m1;
    	}
    	
    }
    
    if(m2>0)
      cout<<m2<<"\n";
    else
      cout<<0<<"\n";
    
    return 0;
}
