#include<bits/stdc++.h>

#define ll long long
#define pll pair<ll,ll>
#define vl vector<ll>
#define mod 1000000007
#define fori(z,n) for(int i=z;i<n;i++)

using namespace std;

ll NCR(int n, int r) {
    if(r > n / 2) r = n - r; // because C(n, r) == C(n, n - r)
    ll ans = 1;
    int i;
    for(i = 1; i <= r; i++) {
        ans *= n - r + i;
        ans /= i;
    }
    return ans;
}

int main() {
    ios_base::sync_with_stdio(0);

    ll t, n;
    cin >> t;
    while(t--) {
        cin >> n;
        if (n == 1) {
            cout << "0\n";
            continue;
        }
        cout << NCR(n, 2) << endl;
    }

	return 0;
}
