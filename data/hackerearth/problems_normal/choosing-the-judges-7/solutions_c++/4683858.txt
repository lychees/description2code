#include <iostream>
using namespace std;

int main()
{
    int tc,i=1;
    cin>>tc;
    while(tc--)
    {
        int n;
        cin>>n;
        long long a[n];
        for(int i=0;i<n;i++)
            cin>>a[i];
        long long dp[n+1];
        if(n==0) cout<<"Case "<<i++<<": "<<0<<endl;
        else if(n==1) cout<<"Case "<<i++<<": "<<a[0]<<endl;
        else 
        {
            dp[0] = 0;
            dp[1] = a[0];
            dp[2] = max(a[0],a[1]);
            for(int i=3;i<=n;i++)
                dp[i] = max(a[i-1]+dp[i-2],dp[i-1]);
            cout<<"Case "<<i++<<": "<<dp[n]<<endl;
        }
    }
}
