#include <iostream>
#include <cstring>
#define ASCII_SIZE 256
using namespace std;

char getMaxOccuringChar(char* str)
{
    int count[ASCII_SIZE];
    for (int j=0; j < ASCII_SIZE; j++)
        count[j] = 0;
     int len = strlen(str);
    for(int j=0;j < len;j++)
    	count[str[j]]+=1;
    int max = -1;
    char c;
    for (int i = 0; i < len; i++) {
        if (max < count[str[i]]) {
            max = count[str[i]];
            c = str[i];
        }
    }
    return c;
}
