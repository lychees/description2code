#include <iostream>
#include <string>
#include <cmath>
using namespace std;

int main()
{
    int t,i,dist,cdist;
    string str;
    cin>>t;
  
    while(t--)
    {
    	cin>>str;
    	char cur='a';
    	for(i=0;i<str.size();i++)
    	{
    		
    		dist=((((int)str[i]-(int)cur)%26)+26)%26;
    		cdist=((((int)str[i]-(int)cur)%26)-26)%26;
    		if(dist>abs(cdist))
    		cout<<cdist<<" ";
    		else
    		cout<<dist<<" ";
    		cur=str[i];
    	}
    	cout<<endl;
    }
    return 0;
}
