#include<bits/stdc++.h>
using namespace std;
int isprime(int n)
{
	for(int i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
		{
			return 1;
		}
	}
	return 0;
}
int main()
{
    int n;
    cin>>n;
    for(int i=4;i<=n;i++)
    {
    	int flag=isprime(i);
    	if(flag==1)
    	{
    		cout<<i<<endl;
    	}
    }
}
