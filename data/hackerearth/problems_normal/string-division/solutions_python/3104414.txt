def brt4c(S):
	n = len(S)
	for i in xrange(n):
		for j in xrange(i + 1, n): 
			for k in xrange(j + 1, n - 1): 
				st = set()
				st.add(S[:i + 1])
				st.add(S[i + 1 : j + 1])
				st.add(S[j + 1 : k + 1])
				st.add(S[k + 1 :])
				if len(st) == 4:
					return "YES"
	return "NO"
	
for tc in xrange(input()):
	S = raw_input()
	if len(S) < 4 : print "NO"
	elif len(S) > 9 : print "YES"
	else : print brt4c(S)
