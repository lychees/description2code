#include <iostream>
#define uint unsigned int
using namespace std;

void merger(uint arr[], uint start, uint mid, uint end) {
	uint left = start;
	uint left_end = mid;
	uint right = mid+1;
	uint right_end = end;
	uint total = end-start+1;
	uint merged[total];
	uint cur = 0;
	while(left<=left_end && right<=right_end) {
		if(arr[left] <= arr[right]) {
			merged[cur++] = arr[left++];
		} else {
			merged[cur++] = arr[right++];
		}
	}
	while(left<=left_end) {
		merged[cur++] = arr[left++];
	}
	while(right<=right_end) {
		merged[cur++] = arr[right++];
	}
	for(cur=0; cur<total; cur++) {
		arr[start+cur] = merged[cur];
	}
}
void mergeSort(uint arr[], uint start, uint end) {
	if(start<end) {
		uint mid = (start+end)/2;
		mergeSort(arr, start, mid);
		mergeSort(arr, mid+1, end);
		merger(arr, start, mid, end);
	}
}

void sort(uint arr[], uint length) {
	mergeSort(arr, 0, length-1);	
}

int main()
{
	uint arr[10001], N, t, i, Ai, Bi;
	unsigned long int sum[10001], A, B;
	arr[0] = 0;
	sum[0] = 0;
	cin>>t;
	while(t--) {
		cin>>A>>B>>N;
		for(i=1; i<=N; i++) {
			cin>>arr[i];
		}
		sort(arr, N+1);
		for(i=1; i<=N; i++) {
			sum[i] = sum[i-1] + arr[i];
		}
		Ai = 0;
		Bi = 0;
		i=0;
		while(i<=N && (A>=sum[i] || B>=sum[i])) {
			if(A>=sum[i]) {
				Ai = i;
			}
			if(B>=sum[i]) {
				Bi =i;
			}
			i++;
		}
		if(Ai == Bi) {
			cout<<"Tie"<<endl;
		} else if(Ai < Bi) {
			cout<<"Sayan Won"<<endl;
		} else {
			cout<<"Raghu Won"<<endl;
		}
	}	
	return 0;
}
