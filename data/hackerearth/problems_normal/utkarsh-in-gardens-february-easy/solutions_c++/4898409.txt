#include<bits/stdc++.h>
using namespace std;
int arr[2005][2005];
long long ans;
int n;
vector<int> g[2005];

void fun(int a,int b){
	long long c=0;
	for(int i:g[a]){
		if(arr[b][i]) c++;
	}
	//cout << a << ' ' << b << ' ' << c << endl;
	ans += c*(c-1)/2;
}

int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			cin>>arr[i][j];
			if(arr[i][j]){
				g[i].push_back(j);
				//g[j].push_back(i);
			}
		}
	}
	for(int i=0;i<n;i++){
		for(int j=i+1;j<n;j++){
			fun(i,j);
		}
	}
	printf("%lld\n",ans/2 );
	return 0;
}
