#include <iostream>
using namespace std;

int main()
{
    int r, c, i=0, j=0, max=0, k=0, count=0;
    cin>>r>>c;
    int arr[r][c];
    for(i=0; i<r; i++)
    {
    	for(j=0; j<c; j++)
    	  cin>>arr[r][c];
    }
    
    for(j=0; j<c; j++)
    {
    	for(i=0; i<r; i++)
    	{
    		if(arr[i][j]>max)
    		  max=arr[i][j];
    	}
    	for(i=0; i<r; i++)
    	{
    		if(arr[i][j]==max)
    		{  
    			count++;
    			for(k=0; k<c; k++)
    			  arr[i][k]=0;
    		}       		  
    	}
    }
    cout<<count;
    return 0;
}
