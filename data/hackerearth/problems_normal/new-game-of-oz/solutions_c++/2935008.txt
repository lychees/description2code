#include <iostream>

int main()
{

   int test;
   cin>>test;
   while(test--){
   	
   	int num;
   	cin>>num;
   	int a[num];
   	int count=0;
   	for(int i=0;i<num;i++)
   	   cin>>a[i];
   	   
   	   sort(a,a+num);
   	   int i=0;
    while(i<num){
    	
    	if(i+1<num && a[i+1]-a[i]==1)
    		i+=2;
    	else

    		count++;  	 
    	
    	
    }
     cout<<count<<endl;
   }

}
