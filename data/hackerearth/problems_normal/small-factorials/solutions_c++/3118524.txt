#include <iostream>
using namespace std;
#define MAX 200
void printFactorial(int pNum);
int multiply(int pIndex, int pResArr[], int pResArrSize);

int main()
{
    int tc;
	int num;
	cin >> tc;

	while( tc > 0) {

		cin >> num;
		printFactorial(num);

		tc--;
	}
}

void printFactorial(int pNum) {

	int resArr[MAX];
 
    // Initialize result
    resArr[0] = 1;
    int resArrSize = 1;
 
    // Apply simple factorial formula n! = 1 * 2 * 3 * 4...*n
    for (int i = 2 ; i <= pNum ; i++)
        resArrSize = multiply(i, resArr, resArrSize);
 
    //cout << "Factorial of given number is \n";
    for (int i = resArrSize - 1 ; i >= 0 ; i--)
        cout << resArr[i];

	cout << "\n";
}

int multiply(int pIndex, int pResArr[], int pResArrSize) {

	int carry = 0;  // Initialize carry
 
    // One by one multiply n with individual digits of res[]
    for (int i = 0 ; i < pResArrSize ; i++)
    {
        int prod = pResArr[i] * pIndex + carry;
        pResArr[i] = prod % 10;  // Store last digit of 'prod' in res[]
        carry  = prod/10;    // Put rest in carry
    }
 
    // Put carry in res and increase result size
    while (carry)
    {
        pResArr[pResArrSize] = carry%10;
        carry = carry/10;
        pResArrSize++;
    }
    return pResArrSize;
}
