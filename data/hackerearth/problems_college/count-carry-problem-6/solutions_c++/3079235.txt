#include <iostream>
using namespace std;

int main()
{
	long long int n,a,b,count;
	int sum;
	cin>>n;
	while(n--)
	{
		cin>>a>>b;
		count=0;
		while(a>0 && b>0)
		{
			sum=(a%10)+(b%10);
			a=a/10;
			b=b/10;
			if(sum>9)
				count++;
		}
		if(count==0)
			cout<<"No carry operation"<<endl;
		else if(count == 1)
			cout<<"1 carry operation"<<endl;
		else
			cout<<count<<" carry operations"<<endl;
	}
	return 0;
}
