#include <iostream>
#define MOD 1000000007
using namespace std;
long long int fact[1000100];
long long int ifact[1000100];
using namespace std;
long long pow1(long long int a, long long int b){
	long long ans=1;
	while(b){
		if(b%2==1) ans= (ans*a)%MOD;
		b/=2;
		a=(a*a)%MOD;
	}
	return ans%MOD;
}
long long inv(long long int a){
	return pow1(a,MOD-2);	
}
void precompute(){
	fact[0]=fact[1]=1;
	for(long long i=2;i<=1000010;i++)	
		fact[i]=(fact[i-1]*i)%MOD;
	for(long long j=0;j<=1000010;j++ )
		ifact[j]= inv(fact[j])%MOD;
}
long long int a[100010];
int main()
{
    precompute();
    int t,n,r;
    long long sum;
    cin>>t;
    while(t--){
    	cin>>n>>r;
    	sum=0;
    	for(int i=0;i<n;i++){
    		cin>>a[i];
    		sum+=a[i];
    	}
    	sum = r-sum;
    	
    	long long ncr = (((fact[n+sum-1]*ifact[n-1])%MOD) * ifact[sum]) %MOD;
    	cout<<ncr<<endl;
    	
    }
    return 0;
}
