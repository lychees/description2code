#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int i,len,t,k,l,m;
    char start[1000000],end[1000000];
    int s[100000],e[100000];
    
    scanf("%d",&t);
    
    for(k=0;k<t;k++)
    {
		scanf("%s",start);
		scanf("%s",end);
		
		len=strlen(start);
		
		l=0;m=0;
		
		for(i=0;i<len;i++)
		{
			if(start[i]!='#')
			s[l++]=i;
			
			if(end[i]!='#')
			e[m++]=i;
		}
		
		if(l!=m)
		cout<<"No\n";
		else
		{
			for(i=0;i<l;i++)
			{
				if(start[s[i]]!=end[e[i]])
				{
					cout<<"No\n";
					break;
				}
				
				else if(start[s[i]]=='A' && s[i]<e[i])
				{
					cout<<"No\n";
					break;
				}
				
				else if(start[s[i]]=='B' && s[i]>e[i])
				{
					cout<<"No\n";
					break;
				}
			}
		}
		
		if(i==l)
		cout<<"Yes\n";
    }
    
    return 0;
}
