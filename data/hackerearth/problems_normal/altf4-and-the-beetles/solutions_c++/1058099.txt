#include <iostream>
#include <algorithm>
#include <map>
#define ll long long
using namespace std;





int main() {
	int t;
	ll n,a,b,c;
	//cin.tie(0);
	cin>>t;
	
	while(t--)
	{
		map<ll,vector<ll> > my;
		cin>>n>>a>>b>>c;
		
		for(int i=0;i<n;i++)
		{
			ll v1,v2;
			cin>>v1>>v2;
			
			my[v1].push_back(0);
			my[v2+1].push_back(1);
		}
		
		ll ans= a*n;
		
		ll fans=-1;
		
		for(map<ll,vector<ll>>::iterator it=my.begin();it!=my.end();++it)
		{
		//cout<<it->first<<" ";
		
		//if(it->first==4)cout<<it->second[0]<<" ";
		
		  for(int i=0;i<(it->second.size());i++)
		  {
		   if((it->second)[i]==0){ans+=b; ans-=a;}
		   else { ans+=c; ans-=b; }
		  }
		  
		  if(it==my.begin() && it->first==0)fans=ans;
		  else fans=max(ans,fans);
		}
		
		cout<<fans<<endl;
		
	
	}
	return 0;
}
