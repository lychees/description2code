#include<iostream>
#include<cmath>
#include<string>
inline long long int power(long long int a, long long int b)
{
if ((a==0)&&(b==0))
	return 0;
	long long count= 1;
	while(b--)
	count=count*a;
	return count;
}
using namespace std;
int main ()
{
	std::ios::sync_with_stdio(false);
	string a;
	long long int t, b, x;
	cin>>t;
	while(t--)
	{
		cin>>a;
		cin>>b;
		x=a[a.length()-1]-'0';
		if(b%4)
		b=b%4;
		else
		b=4;
		cout<<power(x, b)%10<<"\n";
	}
	return 0;
}
