#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
   
    
    for(int i=1;i<=t;i++)
    	{
    		long long  n,k;
    		long long sum=0;
    		cin>>n>>k;
    		
    		int arr[100000];
    		for(int i=0;i<n;i++)
    			cin>>arr[i];
    			
    		sort(arr,arr+n);
    		int p=n-1;
    			while(k!=0&&p>-1)
    			{
    				sum=sum+arr[p];
    				p--;
    				k--;
    			}
    	cout<<sum<<endl;
    	
    	
    	}
    return 0;
}
