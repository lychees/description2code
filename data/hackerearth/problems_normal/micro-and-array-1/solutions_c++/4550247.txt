#include <iostream>
using namespace std;

int main()
{
	int n,q,x,y,v;
	int q_type;
	cin>>n>>q;
	int a[n];
	int i=1;
	while(i<=n)
		cin>>a[i++];
    while(q--){
    	cin>>q_type;	
    	if(q_type == 0){
    		cin>>x>>y;
    		a[x] = y;
    	}
    	else{
    		cin>>v;
    		int i=1;
    		while(i<=n && a[i]<v)
    			i++;
    		if(i>n)
    			cout<<-1<<endl;
    		else
    			cout<<i<<endl;
    	}
    }
    return 0;
}
