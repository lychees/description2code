#include <iostream>
#include<math.h>
using namespace std;

int main()
{
	int t,n,ans;
    cin>>t;
    while(t--)
    {
    	
    	cin>>n;
    	if(n==1 || n==0)
    	{cout<<"0"<<endl; continue;}
    	
    	
    	ans=int(ceil(log(n)/(log(3))));
    	cout<<ans<<endl;
    	
    	
    }
    
    return 0;
}
