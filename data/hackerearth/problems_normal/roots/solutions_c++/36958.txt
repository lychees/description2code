long long power(LL a,LL b)
{
    long long r=1,x=a;
    if(a==0)
    return 0;
    while(b)
    {
        if(b&1)r=(r*x)%MOD;
        x=(x*x)%MOD;
        b>>=1;
    }
    return r%MOD;
}
int gcd(int a,int b)
{
    if(b==0)
    return a;
    gcd(b,a%b);
}
