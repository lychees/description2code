#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int myfind(vector<ll int> &a,ll int l,ll int h,ll int v,ll int k)
{
	ll int m;
	ll int ans=h+1;
	while(l<=h)
	{
		m=((h-l)>>1)+l;
		if(a[m]*v >k)
		{
			ans=m;
			h=m-1;
		}
		else
		{
			l=m+1;
		}
	}
	return ans;
}
int main()
{
	ll int h[1000010];
	memset(h,0,sizeof(h));
	vector<ll int> a;
	ll int n,k;
	scanf("%lld%lld",&n,&k);
	ll int i,j,mc=0;
	map<ll int,ll int> m;
	for(i=0;i<n;i++)
	{
		scanf("%lld",&j);
		if(m[j]==1)//Removing duplicates
		continue;
		else
		{
			a.push_back(j);
			mc++;
		}
	}
	n=mc;
	ll int ans=0;
	sort(a.begin(),a.end(),less<ll int>());
	for(i=0;i<(n-2);i++)
	{
		for(j=(i+1);j<(n-1);j++)
		{
			mc=myfind(a,j+1,n-1,a[i]*a[j],k);
			ans+=mc-(j+1);
		}
	}
	printf("%lld\n",ans);
	return 0;
}
