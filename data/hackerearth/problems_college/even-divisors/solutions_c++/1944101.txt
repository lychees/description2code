#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll prime[100001];
ll ind;
bool p[100001];
void seive(){
  ll i,j;
  for(i=0;i<100001;i++)
     p[i]=false;
  for(i=2;i<=sqrt(100000);i++){
  	 if(!p[i]){
  	 	for(j=2*i;j<=100000;j+=i){
  	 		p[j]=true;
  	 	}
  	 }
  }   
  ind=0;
  for(i=2;i<=100000;i++){
  	 if(!p[i])
  	   prime[ind++]=i;
  	   
  }
}
void calc(ll n,map<ll,ll> &pr){
	for(ll i=0;prime[i]*prime[i]<=n ;i++){
		 if(n%prime[i]==0){
		 	ll count=0;
		 	while(n%prime[i]==0){
		 		count++;
		 		n/=prime[i];
		 	}
		 //	cout<<prime[i]<<" "<<count<<endl;
		 	if(prime[i]==2)
		     	pr[prime[i]]=count;
		    else
		        pr[prime[i]]=count+1;
		 }
	}
	if(n>1){
		pr[n]=2;
	   //	cout<<"n="<<n<<endl; 
	}
}
int main()
{
   ll t,n,p,i;
   seive();
   scanf("%lld",&t);
   while(t--){
   	  map<ll,ll> pr;
   	  map<ll,ll> :: iterator it;
   	  scanf("%lld",&n);
   	  calc(n,pr);
   	  p=0;
   	  if(n>2 && (pr.begin())->first==2){
   	  	  p=1;
   	      for(it=pr.begin();it!=pr.end();it++){
   	  	      
   	  	         p*=(it->second);
   	      }
   	  }
   	  else if(n==2)
   	     p=1;
   	  else
   	     p=0;
   	  printf("%lld\n",p);
   } 
   return 0;
}
