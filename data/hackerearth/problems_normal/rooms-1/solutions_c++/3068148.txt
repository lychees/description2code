    vector<pair<int,int> >v;
    priority_queue<int,vector<int>,greater<int> >p;
    int main(){
    	int i,j,t,n,s,d;

    		scanf("%d",&a[i]);

    		}
    		sort(all(v));
    		int cnt=1;
    		p.push(v[0].second);
    		for(i=1;i<v.size();i++){
    			d=v[i].first;
    			j=p.top();
    			if(d>=j){
    				p.pop();
    				p.push(v[i].second);
    			}
    			else{
    				cnt++;
    				p.push(v[i].second);
    			}
    		}
    		printf("%d\n",cnt);
    		v.clear();
    		p=priority_queue<int,vector<int>,greater<int> >();
    	}
