#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long t,n,i,k,rem,start=1;
	scanf("%lld",&t);
	while(t--)
	{
		k=0;
		scanf("%lld",&n);
		while(n!=0)
		{
			rem=n%2;
			if(rem)
				k++;
			n=n/2;
		}
		if(k%2==0)
			start=(start==1)?0:1;
		else
			start=start;
		if(start==1)
		{
			printf("Sherlock\n");
			start=1;		
		}
		else
		{
			printf("Watson\n");
			start=0;
		}
	}
	return 0;
}
