#include <iostream>
using namespace std;

int main()
{
	unsigned int T;
	cin>>T;
	long long int *ans = new long long int[T];
	for(unsigned int j=0; j<T; ++j) {
		int x, y, n;
		cin>>x>>y>>n;
		long long int *arr = new long long int[n];
		if( n <= x)
			ans[j] = y;
		else
		{
			for(int i=0 ;i<x; ++i)
				arr[i] = y;
			arr[x] = y*x;
			int i=x+1;
			for(i=x+1; i<n; ++i)
			{
				arr[i] = 2*arr[i-1] - arr[i-x-1]; 
			}
			ans[j] = arr[i-1];
		}
	
		delete[] arr;
		
	}
	
	for(unsigned int i=0 ;i<T; ++i )
	{
		cout<<ans[i]<<"\n";
	}
    return 0;
}
