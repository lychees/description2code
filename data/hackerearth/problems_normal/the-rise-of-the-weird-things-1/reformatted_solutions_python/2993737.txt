x = int(raw_input())
y = [int(n) for n in raw_input().split()]
z = []
v = []
for i in y:
    if i % 2 == 0:
        z.append(i)
    else:
        v.append(i)
z.sort()
v.sort()
zx = ""
vx = ""
zs = 0
vs = 0
for i in z:
    zx += str(i)
    zx += " "
    zs += i
zx = zx[:-1]
for i in v:
    vx += str(i)
    vx += " "
    vs += i
vx = vx[:-1]
print zx, zs, vx, vs
