#include<iostream>
#include<string>
using namespace std;

int main(){
	string s;
    while(getline(cin,s)){
    int flag=0;
    for(int i=0;i<s.length()-1;i++){
    	if(s[i]=='/' && s[i+1]=='/')flag=1;
    	if(s[i]=='-' && s[i+1]=='>' && flag==0){
    		s[i]='.';
    		s.erase(i+1,1);
    	}
    }
    cout<<s<<endl;
    }
    return 0;
}
