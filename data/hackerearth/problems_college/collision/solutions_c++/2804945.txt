#include<bits/stdc++.h>
using namespace std;
bool arr[110];
int main()
{
  	int t,n;
	cin>>t;
	long long c=0;
	while(t--)
	{
		c=0;
		memset(arr,0,sizeof(arr));
		cin>>n;
		for(int i=0;i<n;i++)
		{
			int temp;
			cin>>temp;
			if(arr[temp%10]==1)
			 c++;
			else
			arr[temp%10]=1;
		}
		cout<<c<<"\n";
	}
	
	
	
	return 0;
}
