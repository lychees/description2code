#include <bits/stdc++.h>

using namespace std;


vector < pair < double, double > > p;
int main() {
	ios_base::sync_with_stdio(false);	
	double x1, x2, x3, y1, y2, y3;
	cin >> x1 >> y1 >> x2 >> y2 >> x3 >> y3;
	double X1, X2, X3, Y1, Y2, Y3;
	X3 = x2 + x3 - x1;
	Y3 = y2 + y3 - y1;
	X1 = 2 * x3 - X3;
	Y1 = 2 * y3 - Y3;
	X2 = 2 * x1 - X1;
	Y2 = 2 * y1 - Y1;
	p.push_back(make_pair(X1, Y1));
	
	p.push_back(make_pair(X2, Y2));
	p.push_back(make_pair(X3, Y3));
	sort(p.begin(), p.end());
	for (int i = 0; i < p.size(); i++) {
		printf("%.4lf %.4lf\n", p[i].first, p[i].second);
	}
	
	return 0;
}
