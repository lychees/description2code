#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    for(int i=1;i<=n;i++)
    	cout<<"*";
    cout<<endl;
    for(int j=1;j<=(n-2);j++)
    {
	    for(int i=(n-j-1);i>=1;i--)
	    {
	    	cout<<" ";
	    }
    cout<<"*";
    cout<<endl;
    }
    for(int i=1;i<=n;i++)
    	cout<<"*";
    return 0;
}
