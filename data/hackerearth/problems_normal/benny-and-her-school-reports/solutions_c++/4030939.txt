#include <iostream>
#include <cmath>
#define ll long long int
using namespace std;

int main()
{
	int t;
	ll i,j,n,m,x,sum,avg,temp;
	cin >> t;
	while(t--){
		cin >> n >> m >> x;
		ll arr[n];
		sum = 0;
		temp = 0;
		for(i=0 ; i<n ; i++){
			cin >> arr[i];
			sum = sum + arr[i];
		}
		temp = abs(x*(n+1) - sum);
		avg = (sum+temp) / (n+1);
		if(x*(n+1) <= sum){
			cout << 1 << "\n";
		}
		else{
			if((temp >= 1 && temp <= m) && avg >= x){
			   cout << temp << endl;
		     }
		    else{
			   cout << "Impossible" << endl;
		    }
		}
	}
    return 0;
}
