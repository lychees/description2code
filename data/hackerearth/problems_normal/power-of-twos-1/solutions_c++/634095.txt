#include <iostream>
using namespace std;

int divcnt[1000010];
long long divsum[1000010];
int main()
{
    for(int i=0; i<=1000000; i++)
    	divcnt[i] = 0;
    divsum[1] = 0;
    for(int i=2; i<=1000000; i++)
    {
    	for(int j=i; j<=1000000; j+=i)
    		divcnt[j]++;
    }
    for(int i=2; i<=1000000; i++)
    	divsum[i] = divsum[i-1] + 1ll*divcnt[i];
    	
    int T;
    cin>>T;
    while(T--)
    {
    	int N;
    	cin>>N;
    	cout<<divsum[N]<<endl;
    }
    return 0;
}
