#include <iostream>
using namespace std;
int a[30];
int opt(int n,int m)
{ int max=0,j=0;
	for(int i=0;i<n;i++)
	{
		if(a[i]>max){max=a[i];j=i;}
	}
	if(max==0)return m;
	else{
		a[j]=0;
		m=m+1;
		m%=2;
		opt(n,m);
	}
	return 0;
}
int main()
{
    int n,t;
    cin>>t;
     while(t--)
     {
     	 cin>>n;
     	 for(int j=0;j<30;j++)
     	 a[j]=0;
     	 for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=i;j++)
		{
			if(i%j==0)a[i+1]++;
		}
	}
     	 n=opt(n,0);
     	 cout<<n+1<<endl;
     }
}
