#include<iostream>
#include<string.h>
using namespace std;

typedef long long ll;

int main(){


    int t;

    cin>>t;
    while(t--){
        char c[55], k[55];
        int b,e;
        int i=0;
        int is_odd=0;
        cin>>c;
        int l = strlen(c);
        //cout<<l;
        if(l%2){

            k[i]= c[l/2];
            l=l-1;
            is_odd=1;
            i++;
        }
        b = (l/2)-1;
        e=b+1+is_odd;
        while(l){

                k[i]=c[b]<c[e]?c[b]:c[e];
                i++;
                k[i]=c[b]>c[e]?c[b]:c[e];
                i++;
                b--;e++;
                l-=2;
        }
        k[i] = '\0';
        cout<<k<<"\n";
    }
    return 0;

}
