ll gcd(ll a,ll b)
{
    if(b==0)
      return a;
    return gcd(b,a%b);
}
 
ll power(ll a,ll n)
{
    if(n==0)
      return 1;
    ll x=power(a,n/2);
    x=(x*x)%mod;
    if(n%2!=0)
    {
      x=x*a;
      x=x%mod;
    }
    return x;
}

bool visited[100001];
int dp[100001];

int dfs(int i,vvi &v)
{
    int j,k=1;
    visited[i]=1;
    for(j=0;j<v[i].size();j++)
    {
        if(visited[v[i][j]]==0)
            k+=dfs(v[i][j],v);
    }
    dp[i]=k;
    return k;
}

   
int main()
{
    int i,j,k,t,l,r,q,query,n,m;
   
    cin>>n;
    vvi v(n);
    for(i=0;i<n-1;i++)
    {
        cin>>j>>k;
        j--,k--;
        v[j].pb(k);
        v[k].pb(j);
    }

    dfs(0,v);
    dp[0]--;
    for(i=0;i<n;i++)
        cout<<dp[i]<<" ";
    cout<<endl;
    
    return 0;

}
