#include<bits/stdc++.h>
using namespace std;

#define mod 1000000007
#define W while
#define LL long long int
#define pb push_back

int main()
{
	LL  t,n,i,temp,flag;
	cin>>t;
	W(t--)
	{
		flag=0;
		string str,str1;
		cin>>str;
		n=str.length();
		LL hash[26];
		for(i=0;i<26;i++)
			hash[i]=0;
		for(i=0;i<n;i++)
		{
			hash[str[i]-'a']++;
		}
		cin>>str1;
		n=str1.length();
		for(i=0;i<n;i++)
		{
			if(hash[str1[i]-'a']!=0)
			{
				flag=1;
				break;
			}
		}
		if(flag==1)
			cout<<"Yes"<<endl;
		else
			cout<<"No"<<endl;
	}
	return 0;
}
