#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin>>n;
    int R = 0;
    map< pair< pair<long long int,long long int> ,long long int>,long long int > M;
    for(int i=0;i<n;i++) {
    	vector<long long int> V(3);
    	cin>>V[0]>>V[1]>>V[2];
    	sort(V.begin(),V.end());
    	long long int val = M[make_pair(make_pair(V[0],V[1]),V[2])];
    	if(val == 0) {
    		R += 1;
    	}
    	else if(val == 1) {
    		
    		R -= 1;
    	}
    	M[make_pair(make_pair(V[0],V[1]),V[2])] += 1;
    }
    cout<<R<<endl;
    return 0;
}
