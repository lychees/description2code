void fs(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}

int n;
int a[100];
int dp[100][100];
map<int,int> m;

int main()
{
    cin>>n;
    rep(i,n) cin>>a[i];
    dp[0][0]=a[0];
    repn(i,n-1) rep(j,i+1) dp[i][j]=dp[i-1][j]+a[i];
    //rep(i,n) rep(j,n) cout<<"dp["<<i<<"]["<<j<<"]="<<dp[i][j]<<endl;
    rep(i,n)
    {
        rep(j,i+1)
        {
            if(m.find(dp[i][j])!=m.end()) continue;
            {
                m[dp[i][j]]++;
                int last=i+1;
                for(int k=i+1; k<n; k++)
                {
                    for(int l=last; l<=k; l++)
                    {
                        if(dp[k][l]==dp[i][j])
                        {
                            m[dp[k][l]]++; last=k+1; break;
                        }
                    }
                    
                }
            }
        }
    }
    int mx=0;
    si s;
    foreach(it,m)
    {
        if(it->se>mx)
        {
            mx=it->se; s.clear(); s.insert(it->fi);
        }
        else if(it->se==mx) s.insert(it->fi);
    }
    cout<<mx<<endl;
    foreach(it,s) cout<<*it<<" ";
    cout<<endl;
    return 0;
}
