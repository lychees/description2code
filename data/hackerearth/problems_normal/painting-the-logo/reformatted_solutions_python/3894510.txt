def a(n):
    return n**2 - (2*n - (n % 2))


for _ in range(int(raw_input())):
    n = int(raw_input())
    x = int((n/2) ** .5) + 2
    while a(x) + a(x-1) > n:
        x -= 1
    if x < 4:
        print 0, 0
    else:
        f, s = x - 2, x - 3
        if s % 2:
            f, s = s, f
        print f, s
