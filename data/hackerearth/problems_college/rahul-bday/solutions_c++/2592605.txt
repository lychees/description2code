#include <bits/stdc++.h>

#define ll long long int

using namespace std;

int main() {   
    int t;
    cin>>t;
    while(t--) {
        ll b, w, x, y, z, mini=0, amt1, amt2, amt3;
        cin>>b>>w;
        cin>>x>>y>>z;
        amt1=(b*x)+(w*y);
        amt2=((b+w)*x)+(w*z);
        amt3=(y*(b+w))+(b*z);
        mini=min(amt1, amt2);
        mini=min(mini, amt3);
        cout<<mini<<endl;
    }
    return 0;
}
