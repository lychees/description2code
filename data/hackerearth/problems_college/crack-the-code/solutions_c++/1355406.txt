#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
      
      int x=0,y=1,p=1,fib=0;
      int a[100000];
      
      for(int i=0;i<100000;i++) 
       {
        if(i<=1)
         fib=i;
         else
         {
         	
         	fib=x+y;
         	x=y;
         	y=fib;	
         }
         a[p]=fib;
         
         p++;
     }
	
    
    int t,r,s;
    cin>>t;
    while(t--)
    {  int k=0;
    	cin>>r>>s;
    	
    	for(int i=1;i<=10000;i++)
    	{
    		if(a[i]>=r && a[i]<=s)
    		 k++;
    		 if(a[i]>s)
    		 break;
    		
    	}
    	
    	if(k%2==0)
    	cout<<"VALID"<<endl;
    	else
    	cout<<"INVALID"<<endl;
    	
    	
    	
    }
    return 0;
}
