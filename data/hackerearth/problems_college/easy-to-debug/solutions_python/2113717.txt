from sys import stdin
t = int(stdin.readline())
fa = [1]
total = 0
j=1
while j*j<t:
	if t%j==0:
		total+=j;total+=t/j
	j+=1
if j*j==t:
	total+=j

if total!=2*t:
	print "error"
else:
	for m in xrange(1,100):
		fa.append(m*fa[m-1])
	for m in xrange(t):
		ans = ''
		for j in xrange(m+1):
			cur = fa[m]/(fa[j]*fa[m-j])
			ans+=str(cur)
		print ans
