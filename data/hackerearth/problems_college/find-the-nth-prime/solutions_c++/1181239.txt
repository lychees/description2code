#include <iostream>
#include <algorithm>
#include <cstring>
#include <string>
#include <cstdio>
#include <stdlib.h>
#include <math.h>
#include <vector>

typedef long long int ll;
typedef long int l;
using namespace std;

int main()
{
    l arr[1000];
    int prime[1000000]={0};
    prime[0]=1;
    prime[1]=1;
    l j=0;
    for(ll i=2;i<1000000;i++)
    {
        if(prime[i]==0)
        {
            arr[j++]=i;
            for(ll k=i*i;k<1000000;k+=i)
            {
                prime[k]=1;
            }
        }
    }
    int n;
    cin>>n;
  cout<<arr[n-1];

    return 0;
}
