#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
long long int t,b,e;
scanf("%lld",&t);
int mod[10][4]={{0},{1},{2,4,8,6},{3,9,7,1},{4,6},{5},{6},{7,9,3,1},{8,4,2,6},{9,1}};
int rep[10]={1,1,4,4,2,1,1,4,4,2};
while(t--)
{
	scanf("%lld %lld",&b,&e);
	b=b%10;
	if(b<=1||b==5||b==6)
	{
		printf("%d\n",b);
	}
	else
	{
		e=(e%rep[b])-1;
		if(e==-1)
		{
			//printf("xcv");
			e=rep[b]-1;
		}
		printf("%d\n",mod[b][e]);
	}
}
return 0;
}
