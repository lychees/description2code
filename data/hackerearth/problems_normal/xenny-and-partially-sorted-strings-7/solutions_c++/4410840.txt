#include <iostream>
#include <string>
#include <cstddef>
#include <utility>
#include <vector>
#include <algorithm>

using namespace std;
int m;

bool comp(string s1, string s2)
{
	string s3 = s1.substr(0, m);
	string s4 = s2.substr(0, m);
	return s3<s4;
}
int main()
{
    int t, n, k, i, j;
    string x;
    cin>>t;
    for(i=0; i<t; i++)
    {
    	cin>>n;
	    cin>>k;
	    cin>>m;
	    vector<string> s;
	    for(j=0; j<n; j++)
	    {
	    	
	    	cin>>x;
	    	s.push_back(x);
	    }
	    sort(s.begin(), s.end(), comp);
	    cout<<s[k-1]<<endl;
    }
    
    return 0;
}
