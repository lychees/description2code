#include<iostream>
using namespace std;
int main()
{
    int t,i,j,k;
    double n,x,c,a[10000];
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>n;
        c=0;
        for(j=0;j<n;j++)
        {
            cin>>a[j];
        }
        for(j=0;j<n;j++)
        {
            for(k=0;k<n-1;k++)
            {
                if(a[k]>a[k+1])
                {
                    x=a[k];
                    a[k]=a[k+1];
                    a[k+1]=x;
                }
            }
        }
        for(j=0;j<n;j++)
        {
            if(a[j]!=a[j+1]) c++;
        }
        cout<<c<<endl;
    }
}
