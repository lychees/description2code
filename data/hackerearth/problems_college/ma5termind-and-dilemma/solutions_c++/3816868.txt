#include<bits/stdc++.h>
using namespace std;
char str[100005];
int main()
{
  int t ,n,i,ans,j,count1;
  cin>>t;
  while(t--)
  {
      cin>>n;
      cin>>str;
      ans=0;
      count1=0;
      for(i=0,j=n-1;i<j;)
      {
         while(str[i]-'0'!=0)
         i++;
         while(str[j]-'0'!=1)
         j--;
         if(i<j)
         {
         ans++;
         i++;j--;
         }
      }
      for(i=0,j=n-1; i<j;)
      {
         while(str[i]-'0'!=1)
         i++;
         while(str[j]-'0'!=0)
         j--;
         if(i<j)
         {
         count1++;
         i++;j--;
         }

      }
      cout<<min(ans,count1)<<endl;

  }

}
