#define __USE_MINGW_ANSI_STDIO 0
#include <bits/stdc++.h>

#define F first
#define S second
#define pb push_back
#define mp make_pair
#define forn(i, n) for(int i = 0 ; (i) < (n) ; ++i)
#define eprintf(...) fprintf(stderr, __VA_ARGS__),fflush(stderr)
#define sz(a) ((int)(a).size())
#define all(a) (a).begin(),a.end()
#define pw(x) (1LL<<(x))

using namespace std;

typedef long long ll;
typedef double dbl;
typedef vector<int> vi;
typedef pair<int, int> pi;

const int inf = (int)1.01e9;
const dbl eps = 1e-9;

/* --- main part --- */

#define TASK "1"

const int N = 21;

int a[N][N];


int main()
{
    #ifdef home
        assert(freopen(TASK".in", "r", stdin));
        assert(freopen(TASK".out", "w", stdout));
    #endif
    int n, m, C;
    scanf("%d%d%d", &n, &m, &C);
    forn(i, n) forn(j, m) scanf("%d", &a[i][j]);
    int res = -inf;
    forn(mm, pw(m))
    {
        int val = -C * (m - __builtin_popcount(mm));
        forn(i, n)
        {
            int s = 0;
            forn(j, m) if (mm & pw(j)) s += a[i][j];
            val += max(s, -C);
        }
        res = max(res, val);
    }
    printf("%d\n", res);

    #ifdef home
        eprintf("time = %d ms\n", (int)(clock() * 1000. / CLOCKS_PER_SEC));
    #endif
    return 0;
}
