#include <bits/stdc++.h>
 
using namespace std;
#define ll long long
 
template<class T> T nextInt() {
    T x = 0, p = 1;
    char c;
    do {
        c = getchar();
    } while (c <= 32);
    if (c == '-') {
        c = getchar();
        p = -1;
    }
    while (c >= '0' && c <= '9') {
        x = x * 10 + c - '0';
        c = getchar();
    }
    return x * p;
}
int main()
{
	int n;
	scanf("%d",&n);
	string str;
	cin>>str;
	for(int i = 1;i<n;i++){
		int x;
		string temp;
		scanf("%d",&x);
		cin>>temp;
		str = str.substr(0,x) + temp;
	}
	cout<<str<<endl;
}
