int solve(long long int l, int ans, long long int k) {
    if(k == l / 2 + 1)
		return ans;
    if(k <= l/2)
		return solve(l / 2, ans, k);
    return solve(l / 2, 1 - ans, l - k + 1);
}

    int t;

    while(t--) {
        long long int n, k, m, l = 0;
        scanf("%lld %lld", &n, &k);
        m = n;
        while(m--) l = 2 * l + 1;
        if(l < k) {
            printf("-1\n");
            continue;
        }
        if(solve(l, 0, k)) printf("C\n");
        else printf("G\n");
    }
}
