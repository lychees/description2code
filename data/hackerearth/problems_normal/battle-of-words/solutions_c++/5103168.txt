#include <iostream>
#include <string>
#include<vector>
using namespace std;

int main()
{
    int t;
    
    cin>>t;
    getchar();
    while(t--)
    {
    	string str,str2;
    	vector<int> v(26);
    	vector<int> v2(26);
    	getline(cin,str);
    	getline(cin,str2);
    	//cout<<str<<endl;
    	//cout<<str2<<endl;
    	
    	for(int i=0;i<str.size();i++)
    	{
    		if(str[i]!=' ')
    		v[str[i]-'a']++;
    	}
    	for(int i=0;i<str2.size();i++)
    	{
    		if(str2[i]!=' ')
    		v2[str2[i]-'a']++;
    	}
    	for(int i=0;i<26;i++)
    	{
    		//cout<<v[i]<<" "<<v2[i]<<endl;
    		if(v[i]==v2[i])
    		{
    		v[i]=0;v2[i]=0;
    		}
    		else if(v[i]>v2[i])
    		{
    			v[i]=v[i]-v2[i];
    			v2[i]=0;
    			
    		}
    		else
    		{
    			v2[i]=v2[i]-v[i];
    			v[i]=0;
    		}
    	}
    	int sum1=0;
    	int sum2=0;
    	for(int i=0;i<26;i++)
    	{
    		
    		sum1=sum1+v[i];
    		sum2=sum2+v2[i];
    	}
    	if(sum1==0 && sum2!=0)
    	cout<<"You lose some."<<endl;
    	else if(sum1!=0 && sum2==0)
    	{
    		cout<<"You win some."<<endl;
    	}
    	else
    	cout<<"You draw some."<<endl;
    	
    	
    	
    }
    return 0;
}
