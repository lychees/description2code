#include <stdio.h>
#include<iostream>
using namespace std;
long long int a[100009][11];
int arr[100009];
int main()
{
   int n,q;
   scanf("%d",&n);
   scanf("%d",&q);
   
   for(int i=0;i<n;i++)
   {
   	scanf("%d",&arr[i]);
   	
   }
  
   for(int i=1;i<=10;i++)
   {
   		for(int k=0;k<i;k++)
   		{for(int j=k;j<n;j=j+i)
   		{
   			
   			if(j==k)
   			{
   				a[j][i]=arr[j];
   				continue;
   			}
   			a[j][i]=arr[j]+a[j-i][i];
   		}}
   }
   int l,r,k;
   for(int i=0;i<q;i++)
   {
   		cin>>l>>r>>k;
   
   l--;r--;
   /*for(int i=1;i<=10;i++)
   {
   		for(int j=0;j<n;j++)
   		cout<<a[j][i]<<" ";
   		cout<<"\n";
   }*/
  	long long int ans=0;
   if(l-k>=0)
   {
   	ans=ans-a[l-k][k];
   }
   int temp=r-l;
   temp=temp%k;
   r=r-temp;
   ans=ans+a[r][k];
   cout<<ans<<"\n";
   }
    return 0;
}
