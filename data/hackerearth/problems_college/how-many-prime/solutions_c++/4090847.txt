#include<bits/stdc++.h>
using namespace std;
bool prime[1000005];
void markprime()
{
    memset(prime,true,sizeof(prime));
    prime[1]=false;
    for(int i=2;i*i<=1000005;i++)
        for(int j=2*i;j<=1000005;j=j+i)
            prime[j]=false;
}
int main()
{
    int t,i,a;
    bool flag;
    markprime();
    cin>>t;
    while(t--)
    {
        cin>>a;
        flag=false;
        a++;
        for(i=(a/2)-1;i>=1;i--)
        {
            if(prime[i] && prime[a-1-i])
            {
                flag=true;
                cout<<i<<" "<<(a-i-1)<<endl;
                break;
            }
        }
        if(!flag)
            cout<<"-1 -1"<<endl;
    }
}
