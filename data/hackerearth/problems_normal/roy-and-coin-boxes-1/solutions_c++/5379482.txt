#include<stdio.h>
#include<iostream>
typedef long long ll;
using namespace std;
ll ar[1000003],tree[1000004],mm[1000003];
int main()
{
	ll n,q;
	cin>>n;
	cin>>q;
	ll  i;
	while(q--)
	{
		ll l,r;
		cin>>l>>r;
		ar[l]++;
		ar[r+1]--;
	}
	ll sum=0;
	for(i=1;i<=n;i++){
		sum=sum+ar[i];
		mm[i]=sum;
		tree[sum]++;
	}
	
	for(i=1;i<=1000000;i++)
	tree[i]=tree[i]+tree[i-1];
	ll qq;
	cin>>qq;
	while(qq--)
	{
		ll x;
		cin>>x;
		cout<<n-tree[x-1]<<endl;
	}
	return 0;
}
