k, m = [int(i) for i in raw_input('').split()]

states = [False for i in xrange(m)]


def main():
    for i in xrange(k):
        # import ipdb; ipdb.set_trace()
        for j in raw_input('').split()[1:]:
            states[int(j)-1] = True
        if all(states):
            print 'YES'
            return None
    print 'NO'


if __name__ == '__main__':
    main()
