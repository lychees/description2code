using namespace std;

const int nm=100002;

int f[nm];
bool fr[nm];

void init() {
	int i,j,n;
	f[1]=0;
	for(i=2;i<nm;++i) {
		memset(fr,1,sizeof(fr));
		for(j=2;j*j<=i;++j) {
			if (i%j==0) {
				fr[f[i/j]]=0;
				fr[f[j]]=0;
			}
		}
		for(j=1;;++j) if (fr[j]) break;
		f[i]=j;
	}
}
