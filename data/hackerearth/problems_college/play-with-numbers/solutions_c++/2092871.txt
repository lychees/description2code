#include <iostream>
#include<stdio.h>
#include<math.h>
using namespace std;

int main()
{
   long long int t;
   cin>>t;
  
   while(t--)
 {long long int n,a,b,count=0;
   cin>>n>>a>>b;
   if(n<=a)
   count++;
   if(n<=b)
   count++;
   int temp=2;
   while(sqrt(n)>=temp)
   {
    if(n%temp==0)
    {
      if(n%(n/temp)==0)
     {
       if(temp<=a&&n/temp<=b)
       count++;
       if(temp<=b&&n/temp<=a&&(n/temp!=temp))
       count++;

     }
      
    }
    temp++;  
   }
   cout<<count<<endl;
 }
}
