#include <iostream>
using namespace std;

int main()
{
    int t,flag;
    long n,b;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	flag=0;;
    	cin>>n;
    	cin>>b;
    	int k=n+1;
    	while(k%b!=0)
    	{
    		k++;	
    	}
    	
    	if((k-n)<=(b/2))
    	cout<<k<<endl;
    	else
    	{
    		k=k-b;
    		cout<<k<<endl;
    	}
    }
    return 0;
}
