int main(){
	int K;
	cin >> K;
	string s;
	cin >> s;
	int n = s.length();
	char *ans = new char[n];
	for (int i = 1; i <= n; ++i){
		vi vv;
		int index = i;
		vv.push_back(index);
		if (i != 1){			
			while (true){
				int newIndex = index + index - 1;
				if (newIndex > n){
					newIndex = (n - index + 1) << 1;
				}
				if (newIndex == i)
					break;
				vv.push_back(newIndex);
				index = newIndex;
			}
		}
		int period = vv.size();
		int kk = K % period;
		kk = (period - kk) % period;
		int curIndex = vv[kk];
		ans[curIndex - 1] = s[i - 1];
	}
	for (int i = 0; i < n; ++i)
		cout << ans[i]; el;
	return 0;
}
