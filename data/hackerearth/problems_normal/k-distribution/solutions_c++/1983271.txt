#include <iostream>
#include <cstring>
#define MOD 1000000007
using namespace std;
long long int power(long long int b,long long int e)
{
	long long ans=1,temp;
	while(e>0)
	{
		if(e%2)
			ans=(ans*b)%MOD;
		b=(b*b)%MOD;
		e/=2;
	}
	return ans;
}
int n,K;
long long ar[105];
int dp[105][100005];
int fun(int idx,int k)
{
	if(idx==n)
		return 1;
	if(dp[idx][k]!=-1)
		return dp[idx][k];
	long long ans=fun(idx+1,k);
	if(k+ar[idx]<K)
		ans+=fun(idx+1,k+ar[idx]);
	return dp[idx][k]=ans%MOD;
}
int main()
{
	scanf("%d%d",&n,&K);
	memset(dp,-1,sizeof(dp));
	long long sum=0;
	for(int i=0;i<n;i++)
	{
		scanf("%lld",&ar[i]);
		sum+=ar[i];
	}
	if(sum<2*K)
	{
		printf("0\n");
		return 0;
	}
	long long ans=power(2,n);
	ans-=((2LL*fun(0,0))%MOD);		// 2 ways to choose that subset
	ans%=MOD;
	ans=(ans+MOD)%MOD;
	printf("%lld\n",ans);
    return 0;
}
