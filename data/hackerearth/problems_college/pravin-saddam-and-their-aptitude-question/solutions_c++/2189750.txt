#include <iostream>
using namespace std;
int fact(long long n)
{
	if(n==0)
	   return 1;
	 return (n%10)*fact(n-1);  
	
}

int main()
{
    long long m,n,t,k;
    cin>>t;
    while(t--)
    {k=0;
    	cin>>n>>m;
    	for(long long i=n;i<=m;i++)
    	k=(k+fact(i)%10)%10;
    	cout<<k<<"\n";
    }
    return 0;
}
