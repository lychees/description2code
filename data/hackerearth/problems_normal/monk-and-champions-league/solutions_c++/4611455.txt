#include<iostream>
using namespace std;
#include<queue>
int main()
{
    priority_queue<int> pq;
    int row,person,i,ele,amount=0;
    cin>>row>>person;
    for(i=0;i<row;i++)
    {
        cin>>ele;
        pq.push(ele);
    }
    //cout<<"AAaa";
    while(person>0)
    {
        amount=amount+pq.top();
        ele=pq.top();
        ele--;
        pq.pop();
        if(ele>0)
        {
            pq.push(ele);
        }
        person--;
    }
    cout<<amount<<endl;
}
