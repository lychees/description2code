#include<stdio.h>
using namespace std;
int lucky(int x)
{
	int count=0;
	while(x!=0)
	{
		if(x%10==4)
		count++;
		if(x%10==7)
		count++;
		x/=10;
	}
	return count;
}
int main()
{
	int a,l,i,j,found=0;
	scanf("%d%d",&a,&l);
	for(i=a+1;i<10000000007;i++)
	{
		for(j=0;j<l;j++)
		{
			if(lucky(a+j)!=lucky(i+j))
			break;
		}
		if(j==l)
		{
			found=1;
			printf("%d\n",i);
			break;
		}
	}
	if(found!=1)
	printf("-1");
}
