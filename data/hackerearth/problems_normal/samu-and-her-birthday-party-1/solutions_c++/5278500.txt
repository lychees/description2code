#include <iostream>
#include<algorithm>
#include<climits>
#include<cstring>

using namespace std;

#define llu long long int

char str[505][15];

int help[505];

int set_bits( llu n)
{
	int res = 0;
	
	while( n )
	 {
	 	 res++;
	 	 
	 	 n = n&(n-1); 
	 }
	 
	 return res;
}

int check( llu n)
{
	for( int i=0; i<n; i++ )  
	 {
	 	if( help[i] == 0 )
	 	 return 0;
	 }
	 
	 return 1;
}

int main()
{
	llu t;

	cin >> t;
	
	for(llu test_case=1 ; test_case<=t; test_case++ )
	 {
	 	 llu n , k;
		  
		 cin >> n >> k;
		  
		 for( int i=0; i<n; i++ )
		 {
		 	cin >> str[i];
		 } 
		 
		 int answer = k+1 ;
		 
		 for(llu i=1; i<( 1<<k ) ; i++ )
		  { 
		      int x = set_bits(i);
		    
		  	  if( x < answer )
			   {
			   	   memset(help,0,sizeof(help) ) ;
					   
			       for(int j=0; j<k; j++)
			        {
			        	if( (i&(1<<j)) > 0 )
			        	 {
			        	 	 for( int z = 0; z<n ; z++ )
			        	 	  {
			        	 	  	 if( str[z][j] == '1' )
			        	 	  	  {
			        	 	  	  	 help[z] = 1;
			        	 	  	  }
			        	 	  }
			        	 }
			        }
			       
				   if( check(n) ==1 )
				   {
				   	 answer = x;
				   }
				   
			   } 
		  }
		  
		  cout << answer << "\n" ;
		  	 
	 }
	 
	 return 0;
}
