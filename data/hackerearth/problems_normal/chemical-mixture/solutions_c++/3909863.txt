#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000000007

#define ll long long

#define pii pair<pi,int>

ll int m,arr1[75],arr2[75],ans,maxcap;
ll int a[75];
ll int n1,n2,t;
bool done;
void recurse(ll int val,ll int pos)
{
	if(val<0)
		return;
	if(pos==n2)
	{
		ll int x=val;
		ll int temp=0,i,temp2=0;
		if(x>0)
		{
			for(i=n1-1;i>=0;i--)
			{
				if(arr1[i]<=x)
				{
					x-=arr1[i];
					temp++;
					temp2+=arr1[i];
				}
			}
			if(x==0)
			{
				done=true;
				if(temp2>maxcap){
					maxcap=temp2;
					ans=temp;
					ll int k=0;
					for(i=n1-1;i>=0;i--)
					{
						if(arr1[i]<=val)
						{
							val-=arr1[i];
							a[k++]=arr1[i];
						}
					}
				}
			}
		}
		else if(x==0)
			done=true;
		return;
	}
	recurse(val-arr2[pos],pos+1);
	recurse(val,pos+1);
}
int main()
{
	ll int i,j,k;
	
	slld(t);
	while(t--)
	{
		done=false;
		maxcap=0;
		ans=0;
		slld(m);
		slld(n1);
		slld(n2);
		for(i=0;i<n1;i++)
			slld(arr1[i]);
		for(i=0;i<n2;i++)
			slld(arr2[i]);
		sort(arr1,arr1+n1);
		//sort(arr2,arr2+n2);
		recurse(m,0);
		if(!done)
		{
			printf("NO\n");
		}
		else
		{
			sort(a,a+ans);
			printf("YES\n");
			for(i=0;i<ans;i++)
			{
				printf("%lld ",a[i] );
			}
			printf("\n");
		}
	}
	return 0;
}
