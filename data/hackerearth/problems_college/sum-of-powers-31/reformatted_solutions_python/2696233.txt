from math import sqrt, ceil, log
for _ in range(int(raw_input())):
    s = set([1])
    x = int(raw_input())
    if x == 2:
        print "Yes"
        continue
    ans = "No"
    for a in range(2, int(sqrt(x)) + 1):
        for m in range(2, int(ceil(log(x, a))) + 1):
            y = a ** m
            if (x - y) in s or x == 2 * y:
                ans = "Yes"
                break
            s.add(y)
        if ans == "Yes":
            break
    print ans
