#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--){
    	string a,b;
    	cin>>a>>b;
    	int f=0;
    	for(int i=0;i<a.size();i++){
    		if(b.find(a[i])!=-1){
    			f=1;
    			break;
    		}
    	}
    	if(f)
    	cout<<"YES\n";
    	else
    	cout<<"NO\n";
    }
    
    return 0;
}
