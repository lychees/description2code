#include <iostream>
#include <vector>
#include <algorithm>

const uint64_t one_bit = 1;

int main(int argc, char* argv[]) {
  std::vector<uint64_t> numbers;
  for(int i = 0; i < 64; ++i) {
    for(int j = i+1; j < 64; ++j) {
      uint64_t newNumber = (one_bit << i) | (one_bit << j);
      numbers.push_back(newNumber);
    }
  }

  std::sort(numbers.begin(), numbers.end(), [](uint64_t a, uint64_t b) {
     return a < b; 
  });

  uint64_t N, T;
  std::cin >> T;
  for(int i = 0; i < T; ++i) {
    uint64_t sum = 0;
    std::cin >> N;
    for(auto number : numbers) {
      if(number > N)
        break;
      sum += number % 1000000007;
    }
    std::cout << sum % 1000000007 << std::endl;
  }

  return 0;
}
