#include <iostream>
#include<string>
#include<cstring>

using namespace std;

int main()
{
    int t,m=1;
    cin>>t;
    while(t--)
    {
    	cout<<"Case "<<m++<<": ";
    	char s1[100],s2[100];
    	cin>>s1>>s2;
    	int i=0,one0=0,zero1=0,que0=0,que1=0,d,l1=strlen(s1),l2=strlen(s2);
    	if(l1!=l2)cout<<-1<<endl;
    	else{
    	while(s1[i]!='\0')
    	  {
    			if(s1[i]=='0'&&s2[i]=='1')zero1++;
    			else if(s1[i]=='1'&&s2[i]=='0')one0++;
    			else if(s1[i]=='?'&&s2[i]=='0')que0++;
    			else if(s1[i]=='?'&&s2[i]=='1')que1++;
    			i++;
    		
    		}
    		d=zero1-one0;
    		if(d>=0)cout<<zero1+que0+que1<<endl;
    		else
    		{
    			d=abs(d);
    			if(que1<d)cout<<-1<<endl;
    			else
    			cout<<one0+que1+que0<<endl;
    		}
    	}
    }
    return 0;
}
