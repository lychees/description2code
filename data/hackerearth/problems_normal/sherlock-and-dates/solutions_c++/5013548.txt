    #include <iostream>
    #include <string>
    using namespace std;
     
    int main()
    {
        int t,d1,d2,m1,m2,y1,y2;
        string s1,s2;
        cin>>t;
        while(t--){
        	cin>>s1>>s2;
        	d1=10*(s1[0]-'0')+s1[1]-'0';
        	d2=10*(s2[0]-'0')+s2[1]-'0';
        	m1=10*(s1[3]-'0')+s1[4]-'0';
        	m2=10*(s2[3]-'0')+s2[4]-'0';
        	y1=1000*(s1[6]-'0')+100*(s1[7]-'0')+10*(s1[8]-'0')+s1[9]-'0';
        	y2=1000*(s2[6]-'0')+100*(s2[7]-'0')+10*(s2[8]-'0')+s2[9]-'0';
        	int count=0;
        	if(y1==y2)
    		{
    			y1%=100;
    			if(y1>2 && y1<=13)
    			{
    				if(m1>y1-1 || m2<y1-1);
    				else{
    					if(m1==m2)
    					{
    						if(d1<=m1-1 && d2>=m2-1) count++;
    					}
    					else if(m1==y1-1)
    					{
    						if(d1<=m1-1) count++;
    					}
    					else if(m2==y2-1)
    					{
    						if(d2>=m2-1) count++;
    					}
    					else count++;
    				}
    			}
    		}
    		else
        	for(int i=y1;i<=y2;i++ ){
        		
        		if(i==y1){
        			if(y1%100<=13 && (y1%100)>2)
        			{
        				if(m1 < y1%100-1) count++;
    					else if(y1%100==m1+1)
    						{
    							if(d1 <= m1-1) count++;
    						}
        			}
        		}
        		else if(i==y2){
        			if(y2%100<=13 && (y2%100)>2)
        			{
        				if(m2 > y2%100-1) count++;
    					else if(y2%100==m2+1)
    						{
    							if(d2 >= m2-1) count++;
    						}
        			}
        		}
        		else{
        			if(i%100<=13 && i%100>2)
        			count++;
        		}
        	}
        	cout<<count<<endl;
        	
        }
        return 0;
    }
