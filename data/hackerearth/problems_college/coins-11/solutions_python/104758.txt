def primes_sieve(limit):
    limitn = limit+1
    not_prime = set()
    primes = []

    for i in range(2, limitn):
        if i in not_prime:
            continue

        for f in range(i*2, limitn, i):
            not_prime.add(f)

        primes.append(i)

    return primes

prime = primes_sieve(100001)

import fileinput
a=[]
for line in fileinput.input():
	a.append(int(line.strip()))
for i in range(1,a[0]+1):
	if a[i] == 3:
		print 3
	elif a[i] in prime:
		print 2
	else:
		print 1
