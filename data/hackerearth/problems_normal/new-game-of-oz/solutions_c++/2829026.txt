#include<iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,i,a[100000],c=0;
    cin>>t;
    while(t--)
    {c=0;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    sort(a,a+n);
    i=0;
    while(i<n)
    {
    	if((a[i]+1)!=a[i+1])
    	{
    		c++;
    		i++;
    	}
    	else
    	{
    	   i=i+2;
    	   c++;
    	}
    }
    cout<<c<<"\n";
    }
    return 0;
}
