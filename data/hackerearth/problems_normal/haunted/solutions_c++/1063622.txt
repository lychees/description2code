#include<stdio.h>
#include<map>
#include<set>
#include<algorithm>
#include<stdlib.h>
#include<iostream>
#include<vector>


using namespace std ;


map<int,int> m ;
int main(){
	vector<int> in ;
	
	int age, freq ;
	int n,m1 ;
	cin>>n>>m1 ;
	age =0 ;
	freq = 0 ;
	for(int i=1;i<=n;i++){
		int temp ;
		cin>>temp ;
		in.push_back(temp) ;
		m[temp] = m[temp]+1 ;
		if(m[temp]>freq){
			
			freq= m[temp] ;
			age = temp ;
		}
		else if(m[temp]==freq && temp>age){
			age = temp ;
			
		}
		cout<<age<<" "<<freq<<endl ;
	}
}
