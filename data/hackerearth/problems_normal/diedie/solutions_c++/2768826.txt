#define modd 1000000007

typedef long long ll;
ll fast_expo(ll a,ll b)
{
	ll res=1;
	while(b)	
	{
		if(b&1) res=(res*a)%modd;
		a=(a*a)%modd;
		b=b>>1;
	}
	return res;
}

ll n,res;

	{
		ci(n);
		res=fast_expo(2,n-1);
