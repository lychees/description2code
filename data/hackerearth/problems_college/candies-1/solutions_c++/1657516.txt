#include<bits/stdc++.h>
using namespace std;
int main(){ int tc,i,x;
long long int can[1000];
cin>>tc;
while(tc--){
    cin>>x;
    long long int total=0;
    for(i=0;i<x;i++){
        cin>>can[i];
        total=total+can[i];
    }
    if(total%x==0){
        cout<<"YES"<<endl;
        cout<<total/x<<" "<<0<<endl;
    }
    else{
        cout<<"NO"<<endl;
        cout<<total/x<<" "<<total-x*(total/x)<<endl;
    }
}
return 0;}
