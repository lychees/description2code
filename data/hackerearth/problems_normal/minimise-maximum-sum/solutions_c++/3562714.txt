#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;long long A[100000],n,i,j,sum[10000];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<2*n;i++)
    	cin>>A[i];
    	sort(A,A+2*n);
    	for(i=0;i<n;i++)
    	sum[i]=A[i]+A[2*n-i-1];
    	sort(sum,sum+n);
    	cout<<sum[n-1]<<endl;
    }
    return 0;
}
