test = input()
for _ in xrange(test):
    m, k = map(int, raw_input().split(" "))
    if m == 1:
        print 8*k
    elif m == 2:
        if k % 2 == 0:
            print 8*(k/2) + 1
        else:
            print 8*(k/2 + 1) - 1
    elif m == 3:
        print 2 + 4*k
    elif m == 4:
        if k % 2 == 0:
            print 4*(k+1) - 1
        else:
            print 4*k + 1
    elif m == 5:
        print 4 + 8*k
