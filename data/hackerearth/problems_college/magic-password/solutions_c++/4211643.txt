#include<bits/stdc++.h>
using namespace std;
static char str1[1001];
int solve(int ind,char *str){
	if(ind==strlen(str1)){
		if(strcmp(str,str1)!=0)
		    return 100000;
		return 0;
	}
	if(strcmp(str,str1)==0)
	     return 0;
	int ans = 100000;
	ans = min(ans,solve(ind+1,str));
	for(int i=ind+1;i<strlen(str);i++){
		swap(str[ind],str[i]);
		ans = min(ans,1+solve(ind+1,str));
		swap(str[ind],str[i]);
	}
	return ans;
}
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		char str[101];//str1[101];
		scanf("%s",str);
		strcpy(str1,str);
		str1[strlen(str)] = '\0';
		int n1;
		scanf("%d",&n1);
		int n = strlen(str);
		for(int i=0;i<n1;i++){
			char ch;
			scanf("%c",&ch);
			while(!(ch=='L'||ch=='R'))
			    scanf("%c",&ch);
			if(ch=='R'){
				char temp = str[0];
				for(int j=0;j<n-1;j++)
				 str[j] = str[j+1];
				str[n-1] = temp;
			} else {
				char temp = str[n-1];
				for(int j=n-1;j>0;j--){
					str[j] = str[j-1];
				}
				str[0] = temp;
			}
		}
		int ans = solve(0,str);
		printf("%d\n",ans);
	}
	return 0;
}
