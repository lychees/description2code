#include <iostream>
using namespace std;

int main()
{
    int n,i,j;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	for(j=0;j<n;j++)
    	{
    		if(i==0 || i==n-1)
    			cout<<"*";
    		else if(j==n-i-1)
    			cout<<"*";
    		else if(j<n-i-1)
    			cout<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
