#include <iostream>
using namespace std;

int main()
{
  long long int t,n,j;
  cin>>t;
  while(t--)
  { cin>>n;
  	long long int rev=0,d;
   
   while(n)
   {
   	d=n%10;
   	rev=rev*10+d;
   	n/=10;
   }
    
    while(rev)
    {   j=rev%10;
         rev/=10;
    	if(j==2)
    	cout<<"cde";
    	else if(j==6)
    	cout<<"b3";
    	else if(j==7)
    	cout<<"acf";
    	else if(j==9)
    	cout<<"c6a";
    	
    	
    }
    cout<<endl;
  }
  return 0;
}
