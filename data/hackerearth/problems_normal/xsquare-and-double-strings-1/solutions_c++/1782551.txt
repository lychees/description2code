#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int arr[26];
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    char str[101];
    cin >> t;
    while(t--)
    {
    	scanf("%s",str);
    	int i;
    	bool chek=false;
    	for(i=0;i<26;i++)
    	arr[i]=0;
    	for(i=0;str[i];i++)
    	{
    	     int tmp=++arr[str[i]-'a'];
    	     if(tmp>1)
    	     chek=true;
    	}
    	if(chek)
    	cout <<"Yes\n";
    	else
    	cout << "No\n";
    }
    return 0;
}
