    #include <iostream>
    using namespace std;
     
    int main()
    {
        int t;
        cin>>t;
        while(t--)
        {
        	int n,k,count=0;
        	cin>>n>>k;
        	while(n--)
        	{int temp;
        	cin>>temp;
        	if(temp<=0)
        	count++;	
        	}
        	if(count>=k)
        	cout<<"NO";
        	else cout<<"YES";
        	
        	cout<<endl;
        }
        return 0;
    }
