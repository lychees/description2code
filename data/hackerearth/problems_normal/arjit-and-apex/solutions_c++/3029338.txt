#include <bits/stdc++.h>
using namespace std;

#define pb push_back
#define mp make_pair
#define fs first
#define sc second

#define MAX_P 10007


int main()
{
  int T,N,M,i,j,k,A,B,G,H;
  scanf("%d",&T);
  while(T--)
    {
      scanf("%d%d",&M,&N);
      vector< vector<int> > m(MAX_P),n(MAX_P);
      for(i=0;i<M;i++)
	{
	  scanf("%d%d",&A,&B);
	  m[A].pb(B);
	};
      for(i=0;i<N;i++)
	{
	  scanf("%d%d",&A,&B);
	  n[A].pb(B);
	};
      scanf("%d%d",&G,&H);
      A=B=0;
      for(k=1;k<=10000;k++)
	{
	  sort(m[k].begin(),m[k].end());
	  sort(n[k].begin(),n[k].end());
	  A+=min(m[k].size(),n[k].size());
	  for(i=0,j=0;i<m[k].size()&&j<n[k].size();j++)
	    {
	      for(;m[k][i]<n[k][j]&&i<m[k].size();i++);
	      if(m[k][i]==n[k][j])
		B++;
	    };
	};
      if(A>=G)
	{
	  printf("%s\n",(B>=H)?"Great":"Good");
	}
      else
	{
	  printf(":(\n");
	};
    };
  return 0;
};
