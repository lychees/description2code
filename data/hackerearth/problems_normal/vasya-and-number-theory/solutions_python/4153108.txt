
def size_good_subset(arr,n):
	c= [1 for i in range(n)]
	for i in range(n):
		j = n - 1 -i 
		maxCount = 1
		for k in range(j+1,n):
			if (arr[k] % arr[j] == 0) and (maxCount <= c[k]):
				maxCount = c[k] + 1
		c[j] = maxCount
	print max(c) if max(c)!=1 else -1
	
#Array size
n = input()
array = map(int,raw_input().split())
array.sort()
size_good_subset(array,n)
