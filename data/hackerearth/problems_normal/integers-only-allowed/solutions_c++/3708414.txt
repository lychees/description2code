#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main(){
	ll n,x[1005],y[1005],a,b,i;
	cin >> n;
	for(i=0; i<n; i++)
	 cin >> x[i] >> y[i];
	 
	 x[n] = x[0];
	 y[n] = y[0];
	 ll bo = 0;
	for(i=0; i<n; i++){
		a = abs(x[i+1]-x[i]);
		b = abs(y[i+1]-y[i]);
		bo+=__gcd(a,b);
	}
	
	ll s = 0;
	for(i=0; i<n; i++){
		s+=x[i+1]*y[i]-x[i]*y[i+1];
	//	s = abs(s);
	} 
	
	 i = s-bo+2;
	i = i/2;
//	cout << s << " " << bo << endl;
	cout << i << endl;
	return 0;
}
