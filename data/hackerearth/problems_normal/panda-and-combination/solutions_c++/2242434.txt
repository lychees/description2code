#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main()
{
	int t;
	scanf("%d", &t);
	while(t--){
		ll n, m;
		scanf("%lld%lld", &n, &m);
		ll ans = 0;
		if(m == 1){
			printf("%lld\n", n);
			return 0;
		}
		while(n){
			ans += n%m;
			n /= m;
		}
		printf("%lld\n", ans);
	}
}
