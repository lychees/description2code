#include <iostream>
#include <string>
using namespace std;
 
int main()
{
	int t;
	cin>>t;
	while(t--){
		string s1,s2;
		cin>>s1>>s2;
		int a[26],b[26];
		for(int i=0;i<26;i++){
			a[i]=0;
			b[i]=0;
		}
		for(int i=0;i<s1.length();i++)
		a[s1[i]-97]=1;
		for(int i=0;i<s2.length();i++)
		b[s2[i]-97]=1;
		bool isFound=false;
		for(int i=0;i<26;i++){
			if((a[i]==b[i])&&a[i]!=0){
				isFound=true;
				break;
			}
		}
		if(isFound)
		cout<<"Yes"<<endl;
		else
		cout<<"No"<<endl;
	}
    return 0;
}
