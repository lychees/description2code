#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int t;
	cin >> t;
	for(int j=0;j<t;j++)
    {
    	string s;
    	vector<int> v(26, 0);
    	cin >> s;
    	int ans=0;
    	for(int i=0;i<s.length();i++)
    	{
        	char x = s.at(i);
        	v[int(x)-97]++;
    	}
    	for(int i=1;i<26;i++)
    	{
    		if(v[i]!=v[0]){ans=1;break;}
    	}
    	if(ans==0){cout << "Yes"<<endl;}
    	else{cout << "No"<<endl;}
    }
    return 0;
}
