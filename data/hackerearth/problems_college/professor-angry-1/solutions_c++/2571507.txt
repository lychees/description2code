#include <iostream>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,k,count = 0,_time;
		scanf("%d%d",&n,&k);
		for(int i=0;i<n;i++)
		{
			scanf("%d",&_time);
			if(_time<=0)
				count++;
		}
		if(count>=k)
			printf("NO\n");
		else
			printf("YES\n");
	}
	return 0;
}
