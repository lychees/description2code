import math

def minEffect(m):
	return int(m/2)

for T in range(input()):
	h, m = raw_input().rstrip(' ').split(' ')
	if h[0] == '0':
		h = h[1:]
	if m == '00':
		m = 0
	elif m[0] == '0':
		m = m[1:]
	degM = 6 * int(m)
	degH = 30 * int(h) + minEffect(int(m))
	ans = abs(degH - degM)
	if ans > 180:
		print 360 - ans
	else:
		print ans
