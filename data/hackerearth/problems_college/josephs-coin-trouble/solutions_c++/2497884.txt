#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int count( vector<int> myvec,int m, int n )
{
    if (n == 0)
        return 1;
    if (n < 0)
        return 0;
    if (m <=0 && n >= 1)
        return 0;
    return count( myvec, m - 1, n ) + count( myvec, m, n-myvec[m-1] );
}
int main()
{
   int n,count1;
   cin>>n;
   vector<int>myvec;
   int temp;
   while(cin>>temp)
   {
   	myvec.push_back(temp);
   }
   sort(myvec.begin(),myvec.end());
   count1=count(myvec,myvec.size(),n);
   cout<<count1<<endl;
    return 0;
}
