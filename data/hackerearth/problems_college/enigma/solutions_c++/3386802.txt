#include <iostream>
#include<cmath>
using namespace std;
int sum(int n)
{
		int sum=0;
    		while(n!=0)
    		{
    			sum+=pow(n%10,2);
    			n/=10;
    		}
    		return sum;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int m,n;
    	cin>>n>>m;
         n=sum(n);
         m=m-1;
    	while(m>0)
    	{	
    		if(n==1)
    		{
    			break;
    		}
    		if(m>=8)
    		{
    			if(n==4||n==16||n==20||n==37||n==42||n==58||n==89||n==145)
    			{
    				m=m%8;
    			}
    			else
    			{
    			   n=sum(n);
    			   m--;
    			}
 
    		}
    		else
    		{
    			n=sum(n);
    			m--;
    		}
    	}
    cout<<n<<endl;
    }
    return 0;
}
