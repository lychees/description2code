#include <iostream>
using namespace std;

int main()
{
    long long t,n,p,u,r,s,temp;
    cin>>t;
    for(int i=1;i<=t;i++){
    	cin>>n>>p>>u>>r>>s;
    	while(n){
    		if((p+r)<=u){
    			n-=1;
    			p+=r;
    		}
    		else if((p-s)>=0){
    			n-=1;
    			p-=s;
    		}
    		else
    			break;
    		//cout<<n<<"  "<<p<<endl;
    	}
    	cout<<"Case #"<<i<<": "<<p<<"\n";
    }
    return 0;
}
