#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--){
    	int n,i,x,count=0;
    	unordered_set<int>unset;
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>x;
    		if(unset.find(x%10)!=unset.end())
    		count++;
    		else
    		unset.insert(x%10);
    	}
    	cout<<count<<endl;
    }
    return 0;
}
