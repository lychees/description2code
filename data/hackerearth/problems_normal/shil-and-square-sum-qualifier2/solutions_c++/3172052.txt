#include <bits/stdc++.h>
using namespace std;
#define ll long long 
#define mod 1000000007
#define sz 1000010
ll dp[sz];
ll ans[sz];
ll lw[sz];
ll arr[sz];
ll sum[sz];
int main()
{
    ll n,k,i,j;
    cin>>n>>k;
    
    
    for(i=1;i<=n;i++)
    {
    cin>>arr[i];
    sum[i]=(arr[i]+sum[i-1])%mod;
    }
       
	ll v1,v2,v3;
	for(i=1;i<=k;i++)
	{
		lw[1]=(lw[1]+(i*arr[i])%mod)%mod;
	}
	for(i=2;i<=n-k+1;i++)
	{
		v1=(sum[k+i-2]-sum[i-2]);
		if(v1<0)
		v1+=mod;
		
		v2=(lw[i-1]-v1);
		if(v2<0)
		v2+=mod;
		
	lw[i]=(v2+(k*arr[k+i-1])%mod)%mod;
	}
	
    for(i=1;i<=k;i++)
    {
    	dp[1]=(dp[1]+((2*i-1)*arr[i])%mod)%mod;
    	
    }
   
    
    for(i=2;i<=n-k+1;i++)
    {
    	v1=(sum[k+i-1]-sum[i-1]);
    	if(v1<0)
    	v1+=mod;
		    	dp[i]=(2*lw[i]-v1)%mod;
		    	if(dp[i]<0)
		    	dp[i]=dp[i]+mod;
    
    }
    
    
    for(i=1;i<=k;i++)
    {
    	ans[1]=(ans[1]+((i*i)%mod*arr[i])%mod)%mod;
    	
    }
    cout<<ans[1]<<" ";
    
    for(i=2;i<=n-k+1;i++)
    {
    	ans[i]=(ans[i-1]+((((k)*(k))%mod)*arr[i+k-1])%mod)%mod;
    	
    	ans[i]=ans[i]-dp[i-1];
    	if(ans[i]<0)
    	ans[i]+=mod;
    	
    	cout<<ans[i]<<" ";
    }
    cout<<endl;
    
    
    
    
    
    return 0;
}
