#include<iostream>
#include<cstdio>
#include<cmath>					  
using namespace std;

int main()
{                                                  
	int t,n,arr[10000]={0},i,num,c=7,j,p=0,newnum=0;

scanf("%d",&t);
while(t--)
{
  scanf("%d",&n);
   for(i=0;i<n;i++)
   scanf("%d",&arr[i]);
   
	  
   	for(i=0;i<n;i++)
	{  int arr1[10]={0};
		c=7;
	 num=arr[i]; 	
	 
	 while(num!=0)
	 {
	 arr1[c]=num%2;
	 num=num/2;
	 c--;
	 }
	 
		c=7;
		
	 for(j=0;j<=c;j++)
	 {
	 if(arr1[j]==0)
	 arr1[j]=1;

	 }

     }

	}
	printf("\n");
   
}

}
