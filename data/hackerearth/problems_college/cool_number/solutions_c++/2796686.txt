#include<iostream>
using namespace std;
bool f(int a,int k)
{
	int i=99;
	int c=0;
	bool arr[100]={false};
	while(a>0)
	{
		arr[i]=a%2;
		a=a/2;
		i--;
	}
	for(i=0;i<=97;i++)
	{
		if(arr[i]==true&&arr[i+1]==false&&arr[i+2]==true)
		c++;
		if(c>=k)
		return true;
	}
	return false;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int r,k,i;
		cin>>r>>k;
		int c=0;
		
		
			for(i=1;i<=r;i++)
			{if(f(i,k)==true)c++;}
			
		
		cout<<c<<endl;
	}
}
