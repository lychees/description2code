#include <cstdio>
#include <iostream>
#define llu long long unsigned

using namespace std;

llu gcd(llu a, llu b)
{
	llu smaller = a < b ? a : b;
	llu larger = a < b ? b : a;

	llu tmp;
	while(smaller != 0)
	{
		tmp = smaller;
		smaller = larger % smaller;
		larger = tmp;
	}
	return larger;
}

llu lcm(llu a, llu b)
{
	return a*(b/gcd(a,b));
}

llu process(llu a,llu b,llu n)
{
	llu lcm_ab = lcm(a,b), count = 1, smaller_count = 1, larger_count =0;
	llu num_terms = lcm_ab/a + lcm_ab/b -1;
	llu rem = n % num_terms;
	llu smaller = a < b ? a : b;
	llu larger = a < b ? b : a;
	llu smaller_sum = smaller, larger_sum = 0;
	bool is_smaller_to_be_used = true;

	while(count < rem)
	{
		if(smaller_sum+smaller < larger_sum+larger)
		{
			smaller_sum += smaller;
			is_smaller_to_be_used = true;
			smaller_count++;
		}
		else
		{
			larger_sum += larger;
			is_smaller_to_be_used = false;
			larger_count++;
		}
		count++;
	}
	llu ret = lcm_ab * (n/num_terms);
	if(rem != 0)
	{
		if(is_smaller_to_be_used)
			ret += smaller * smaller_count;
		else
			ret += larger * larger_count;
	}
	return ret;
}

int main()
{
	int t;
	llu a, b, n;
	scanf("%d",&t);
	while(t > 0)
	{
		cin >> a >> b >> n;
		llu res = process(a,b,n);
		cout << res << endl;
		t--;
	}

	return 0;
}
