#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,i,j;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	int sum=0,n;
    	cin>>n;
    	int a[n],b[n];
    	for(j=0;j<n;j++)
    		cin>>a[j];
    	for(j=0;j<n;j++)
    		cin>>b[j];
    	sort(a,a+n);
    	sort(b,b+n);
    	for(j=0;j<n;j++)
    		sum+=(a[j]*b[j]);
    	cout<<sum<<endl;
    }
    return 0;
}
