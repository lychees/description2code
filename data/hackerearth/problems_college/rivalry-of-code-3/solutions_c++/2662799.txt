#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int test;
	cin>>test;
	string d,s;
	while(test--)
	{
		cin>>d>>s;
		int i,j,ld,ls,t,ans=INT_MAX;
		ld=d.length();
		ls=s.length();
		if(ld<=ls)
		{
			for(i=0;i<(ls-ld+1);i++)
			{
				t=0;
				for(j=0;j<ld;j++)
				{
					if(s.at(i+j)!=d.at(j))
					{
						t++;
					}
				}
				ans=min(ans,t);
			}
			cout<<ans<<endl;
		}
		else
		{
			//~ cout<<"here "<<endl;
			for(i=0;i<ld-ls+1;i++)
			{
				t=0;
				for(j=0;j<ls;j++)
				{
					if(s.at(j)!=d.at(i+j))
					{
						t++;
					}
				}
				ans=min(ans,t);
			}
			cout<<ans<<endl;
		}
	}
	return 0;
}
