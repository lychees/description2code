#include<bits/stdc++.h>
#define bs 1000000007
using namespace std;
const int N = 20000;
int tests;
long long l, r;
int fib[N];
long long ans;
long long full_block;
long long solve(long long x)
{
	long long ful = (x / 15000)%bs;
	x %= 15000;
	long long res = fib[14999] * ful % bs;
	res += fib[x];
	res %= bs;
	return res;
}
int main(){
	ios_base::sync_with_stdio(0);
	cin >> tests;
	fib[1] = 0;
	fib[2] = 1;
	for (int i = 3; i <= 15000; i++)
	{
		fib[i] = fib[i - 1] + fib[i - 2];
		fib[i] %= 10000;
	}
	fib[0] = fib[15000];
	for (int i = 1; i <= 15000; i++)
	{
		fib[i] += fib[i - 1];
		fib[i] %= bs;
	}
	for (int i = 0; i < 15000; i++)
	{
		full_block = full_block + fib[i];
		full_block %= bs;
	}
	for (; tests; --tests)
	{
		cin >> l >> r;
		long long res = solve(r) - solve(l - 1);
		res %= bs;
		if (res < 0)
			res += bs;
		cout << res << "\n";
	}
	return 0;
}
