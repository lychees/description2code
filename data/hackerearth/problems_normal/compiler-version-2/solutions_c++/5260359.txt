#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s,ans;
    while(getline(cin,s)){
    	bool flag=0;
    	ans="";
    	for(int i=0;i<s.length();i++){
    		if(flag){
    			ans+=s[i];
    			//continue;
    		} 
    		else if(i<s.length()-1&&s[i]=='/'&&s[i+1]=='/'){
    			ans+="//";
    			flag=1;
    			i++;
    		}
    		else if(i<s.length()-1&&s[i]=='-'&&s[i+1]=='>'){
    			ans+=".";
    			i++;
    		}
    		else{
    			ans+=s[i];
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
