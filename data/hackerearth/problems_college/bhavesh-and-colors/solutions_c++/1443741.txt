#include <bits/stdc++.h>
using namespace std;
#define MAX 1000006
int failurefunction[MAX];
int arr[MAX];
void computefailurefunction(int *arr, int n)
{
    int size = n;
    failurefunction[0] = failurefunction[1] = 0;
    for(int i = 2; i <= size; i++) {
        int j = failurefunction[i - 1];
        while(true) {
            if(arr[j] == arr[i - 1]) {
                failurefunction[i] = j + 1;
                break;
            }
            if(j == 0) {
                failurefunction[i] = 0;
                break;
            }
            j = failurefunction[j];
        }
    }
}

int main()
{
    int t;
    scanf("%d", &t);
    for(int i = 1; i <= t; i++) {
    	int n;
        scanf("%d", &n);
        for(int i = 0; i < n; i++) {
        	scanf("%d", &arr[i]);
        }
        computefailurefunction(arr, n);
        int ans = INT_MIN;
        for(int i = 0; i <= n; i++) {
        	ans = max(ans, failurefunction[i]);
        }
        printf("%d\n", ans);
    }
    return 0;
}
