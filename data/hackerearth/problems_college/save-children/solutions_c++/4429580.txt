#include<bits/stdc++.h>
using namespace  std;
#define ll long long
    int child[100005];
    int tree[100005];
int arr[500];

int main()
{
    int t,n,i,j;
    cin>>t;
    while(t--)
     {
         cin>>n;
         for(i=0;i<=200;i++)
            arr[i]=0;
         for(i=1;i<=n;i++)
         {
             cin>>child[i];
             arr[child[i]+100]++;
         }
         for(i=1;i<=n;i++)
         {
             cin>>tree[i];
         }
         sort(child+1,child+n+1);
         sort(tree+1,tree+n+1);
         int ans1=INT_MIN;
         for(i=1;i<=n;i++)
         {
             ans1=max(ans1,abs(child[i]-tree[i]));
         }
         int ans2=INT_MAX;
         int minco;
         for(i=0;i<=200;i++)
         {
             int sum=0;
            for(j=0;j<=200;j++)
            {
                sum=sum+arr[j]*(abs(j-i));
            }
             if(sum<ans2)
                {
                   // cout<<sum<<" "<<i-100<<endl;
                    minco=i-100;
                    ans2=sum;
                }
         }
         cout<<ans1<<endl;
         cout<<minco<<endl;

     }
return 0;
}
