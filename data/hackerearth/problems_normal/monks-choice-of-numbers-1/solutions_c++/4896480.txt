#include <iostream>
#include<algorithm>
#include<vector>
using namespace std;

int cnt1(int n)
    {   int count=0;
        while( n )
        {
        n = n&(n-1);
           count++;
        }
        return count;
}

struct myclass {
  bool operator() (int i,int j) { return (i>j);}
} myobject;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	vector<int> arra;
    	int n,k,i,a,ans;
    	cin>>n>>k;
    	for(i=0;i<n;i++)
    	{
    	    cin>>a;
    	    ans=cnt1(a);
    	    arra.push_back(ans);
    	}
    	sort(arra.begin(), arra.end(), myobject); 
    	ans=0;
    	for(i=0;i<k;i++)
    	{
    		ans+=arra[i];
    		//cout<<arra[i]<<" "<<i<<" here is \n";
    	}
    	cout<<ans<<"\n";
    }
    return 0;
}
