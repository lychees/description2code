#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int ar[1000005]= {0};
int main()
{
    int T,i,j,k,N;
    cin >> T;
    while(T--)
    {
    	cin >> N;
    	vector<int> arr(ar,ar+N);
    	for( i=0; i < N;i++)
    	{
    		cin >> arr[i];
    	}
    	sort(arr.begin(),arr.end());
    	for(i=N-1;i >0;i--) 
    	{
    		cout << arr[i] << " ";
    	}
    	cout << arr[0] << endl;
    }
    return 0;
}
