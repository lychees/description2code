#include <bits/stdc++.h>
using namespace std;
map < int , int > m;
int main()
{
	int t,n,r,s;
	cin>>t;
	while(t--)
	{
		scanf("%d",&n);
		int ans = 0;
		int k = 0;
		m.clear();
		for(int i=0;i<n;i++)
		{
			scanf("%d%d",&s,&r);
			if(r==1)
			{
				if(m[s]==0) k++;
				ans = max(ans , k);
				m[s]++;
			}
			else k = 0;		
		}
		cout<<ans<<"\n";
	}
}
