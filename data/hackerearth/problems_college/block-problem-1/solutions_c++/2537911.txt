#include <iostream>
#include <stdint.h>
 
using namespace std;
#define MOD 1000000007
 
uint64_t pow(uint64_t x, uint64_t y, uint64_t p)
{
	if (y == 0)
		return 1;
 
	if (y == 1)
		return (x % p);
 
	uint64_t t = pow(x, y/2, p);
	t = (t * t) % p;
 
	if (y % 2)
		t = (t * x) % p;
 
	return t;
}
 
main()
{
	uint64_t *P = new uint64_t[1001];
	P[1] = 1;
	P[2] = 2;
	P[3] = 4;
	P[4] = 8;
 
	for (int i = 5; i <= 1000; i++)
		P[i] = (P[i - 1] + P[i - 2] + P[i - 3] + P[i - 4]) % MOD;
 
	int T;
	cin >> T;
 
	for (int t = 0; t < T; t++)
	{
		int N, M;
		cin >> N >> M;
 
		uint64_t *Q1 = new uint64_t[1 + M];
		uint64_t *Q2 = new uint64_t[1 + M];
 
		for (int i = 1; i <= M; i++)
			Q1[i] = pow(P[i], N, MOD);
 
		Q2[1] = Q1[1];
		for (int i = 2; i <= M; i++)
		{
			uint64_t s = 0;
			for (int j = 1; j < i; j++)
				s = (s + Q2[j] * Q1[i - j]) % MOD;
 
			Q2[i] = (Q1[i] + MOD - s) % MOD;
		}
 
		std::cout << Q2[M] << std::endl;
 
		delete[] Q1;
		delete[] Q2;
	}
 
	delete[] P;
}
