#include <iostream>
#include <string>
using namespace std;

int main()
{
	int n;
	cin>>n;
	
	string s;
	int len;
	while(n--)
	{
		cin>>s;
		len = s.size();
		int count=0;
		for(int i=0 ; i<len-1 ;i++)
		{
			if(s[i+1]==s[i])
			{
				cout<<"SLAP"<<endl;
				break;
			}
			count++;
		}
		
		if(count==len-1)
		cout<<"KISS"<<endl;
	}
    return 0;
}
