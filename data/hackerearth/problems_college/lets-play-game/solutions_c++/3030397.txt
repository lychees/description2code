#include<iostream>
#include<algorithm>
using namespace std;

int main(){
	int t;
	cin>>t;
	while(t--){
		long n,k,i;
		cin>>n>>k;
		long time[n], profit[n], res[n];
		for(i=0;i<n;++i){
			cin>>time[i];
		}
		for(i=0;i<n;++i){
			cin>>profit[i];
		}
		for(i=0;i<n;++i){
			res[i]=(k/time[i])*profit[i];
		}
		sort(res, res+n);
		cout<<res[n-1]<<endl;
	}
	return 0;
}
