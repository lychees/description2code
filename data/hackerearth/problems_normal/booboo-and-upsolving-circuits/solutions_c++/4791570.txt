#include <iostream>
#include <vector>
#include <math.h>
using namespace std;
vector <long long int> A(1000001);
long long int days_using_time(long long int t,vector <long long int> &A)
{
    long long int i,sum=0,count=1;
    for(i=0;i<A.size();i++)
    {
        // sum = sum + A[i];
        if(sum + A[i] > t)
        {
            count++;
            sum = A[i];
            // sum = 0;
        }
        else
            sum += A[i];
    }
    return count;
}

int main()
{
    long long int t,k,n,a,b,c,d,i,val,z,low,q,num_sum=0,m,max=0,high,mid,target,no_of_days;
    cin >> n >> m;
    for(i=0;i<n;i++)
    {
        cin >> A[i];
        num_sum = num_sum + A[i];
        if(A[i]>max)
            max = A[i];        
    }
    low = max;
    high = num_sum;    
    target = m;
    a = days_using_time(low,A);
    // cout << "a = " << a << endl;
    if(a <= m)
    {
        cout << low << endl;
        return 0;
    }
    if(m == 1)
    {
        cout << high << endl;
        return 0;
    }
    else
    {
        low++;
        // cout << "I am in else" << endl;
        while(low < high)
        {
            mid = (low+high)/2;
            no_of_days = days_using_time(mid,A);
            // cout << "no_of_days taken by :" << mid << " is : " << no_of_days << endl;
            if(no_of_days <= target)
                high = mid;
           
            else
                low = mid+1;
        }
        cout << high << endl;
    }
    return 0;
}
