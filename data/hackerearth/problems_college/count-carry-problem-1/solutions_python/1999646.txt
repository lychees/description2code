t = input()
for i in range(t):
	a,b = raw_input().split(' ')
	ln = len(a)
	cnt = 0
	crry = 0
	for i in range(ln):
		c = int(a[ln-i-1])+int(b[ln-i-1])+crry
		#print a,b,' ',c
		if c >= 10:
			crry = c%10
			cnt +=1
		else:
			crry = 0
	s = str(cnt) if cnt else 'No'
	g = s+ ' carry operation'+('s' if cnt>1 else '')
	print g
