void solve()
{
	string s;
	int k;
	cin>>s>>k;
	int a[26] = {0};
	for(int i=0;i<s.size();i++)
		a[s[i] - 'a']++;
	vector<int> cnt;
	for(int i=0;i<26;i++)
	{
		if(a[i] > 0)
			cnt.push_back(a[i]);
	}
	sort(cnt.begin(), cnt.end());
	int c = 0;
	string res = "";
	for(int i=0;i<cnt.size();i++)
	{
		if(cnt[i] <= k)
		{
//			for(int z=0;z<cnt[i];z++)
//				res += (char)(i + 'a');
			
			k -= cnt[i];
			cnt[i] = 0;
		}
		else
		{
//			for(int z=0;z<k;z++)
//				res += (char)(i + 'a');
			break;
		}
	}
	for(int i=0;i<cnt.size();i++)
	{
		if(cnt[i] > 0)
		{
			c++;
			for(int z=0;z<cnt[i];z++)
				res += (char)(i + 'a');
		}
	}
	cout<<c<<"\n";
	cout<<res<<"\n";
}
