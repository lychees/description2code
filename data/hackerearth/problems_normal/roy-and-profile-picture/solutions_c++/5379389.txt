#include <iostream>

using namespace std;

int main()
{
    int arr[2][1000],l,no;
    cin>>l>>no;
        for(int i=0;i<no;i++)
        {
            cin>>arr[0][i]>>arr[1][i];
        }
        for(int i=0;i<no;i++)
        {
            if(arr[0][i]<l || arr[1][i]<l)
                cout<<"UPLOAD ANOTHER"<<"\n";
            else if(arr[0][i]==arr[1][i])
                cout<<"ACCEPTED"<<"\n";
            else
                cout<<"CROP IT"<<"\n";
        }
    return 0;

}
