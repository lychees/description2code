#include<iostream>
using namespace std;
int main()
{
	int t;
	long int n,p,u,r,s;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		cin>>n>>p>>u>>r>>s;
		for(int j=0;j<n;j++)
		{
			if((p+r)<=u)
			p=p+r;
			else if((p+r)>u && (p-s)>0)
			p=p-s;
			
			//cout<<p;
		}
		cout<<"Case #"<<i+1<<": "<<p<<"\n";
	}
	return 0;
}
