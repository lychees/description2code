#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

int main() {
    long long int mod = 1000000007;
    int t;
    cin>>t;
    while(t--) {
	int n;
	cin>>n;
	vector<long long int> ar(n);

	for(int i=0; i<n; i++)
	    cin>>ar[i];
	sort(ar.begin(), ar.end());

	vector<long long int> br(n);
	br[0] = ar[0];
	for(int i=1; i<n; i++)
	    br[i] = (br[i-1] + ar[i]) % mod;
	long long int ans = 0;
	for(int i=1;i<n; i++)
	    ans = (ar[i]*i - br[i-1] + mod + ans) % mod;
	cout<<(ans*ar[n-1])%mod<<endl;
    }
    return 0;
}
