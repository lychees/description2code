def gcd(a, b):
    while b:
        a, b = b, a % b
    return a


t = input()
for i in range(0, t):
    s = raw_input().split()
    r = ''
    m, n = int(s[0]), int(s[1])
    a = raw_input().split()
    for i in range(0, len(a)):
        a[i] = int(a[i])
    for i in a:
        x = gcd(i, n)

        if x == 1:
            r = r+str(i)+" "
    print r
