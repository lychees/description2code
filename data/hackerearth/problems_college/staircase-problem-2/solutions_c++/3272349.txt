#include <bits/stdc++.h>

using namespace std;

int dp[100005];

int stair(int n, int x1, int x2)
{
	if(n == 0) {
		return 0;
	}	
	if(n < 0) {
		return 9999999;
	}
	if(dp[n] != -1) {
		return dp[n];
	}
	return dp[n] = min(1+stair(n-x1, x1, x1+1), 1+stair(n-x2, x2, x2+1));
}

int main()
{
	int t,n;
	scanf("%d",&t);
	while(t--) {
		scanf("%d",&n);
		for(int i = 0; i <= n; i++) {
			dp[i] = -1;
		}
		printf("%d\n",stair(n,1,2));
	}
	return 0;
}
