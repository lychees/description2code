#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int i,n,a[1000],dp[1000];
    scanf("%d",&n);
    for(i=0;i<n;i++)
    	scanf("%d",&a[i]);
    sort(a,a+n);
    for(i=0;i<n-1;i++)
    	dp[i]=a[i+1]-a[i];
    int ans=0;
    for(i=0;i<n-1;i++)
    	ans=__gcd(ans,dp[i]);
    vector<int> ve;
    for(i=2;i<=sqrt(ans);i++){
    	if(i*i==ans)
    		ve.push_back(i);
    	else if(ans%i==0){
    		ve.push_back(i);	ve.push_back(ans/i);
    	}
    }
    if(ans!=1)
    	ve.push_back(ans);
    sort(ve.begin(),ve.end());
    n=ve.size();
    for(i=0;i<n;i++)
    	cout<<ve[i]<<" ";
    return 0;
}
