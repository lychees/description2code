#include <iostream>
using namespace std;
#include<vector>
#include<algorithm>
typedef long long int ll;




int canGayleHit(vector<ll> &gi,vector<ll> &pi,ll M)
{
	ll curr_pow = M;
	std::sort(pi.begin(),pi.end());
	//cout<<"curr_pow"<<curr_pow<<"\n";
	for(ll i=0;i<pi.size();i++)
	{
		//cout<<gi[i]<<" "<<pi[i]<<"\n";
		if(pi[i]>curr_pow)
			return false;
		else
		{
			curr_pow = curr_pow + gi[i];
		}
	
	}
	
	return true;
	
	
}

int main()
{
  //  cout << "Hello World!" << endl;
    ll T;
    cin>>T;
    
    for(ll i=0;i<T;i++)
    {
    	ll N,M;
    	cin>>N>>M;
    	vector<ll>Gi;
    	vector<ll>Pi;
    	for(ll j=0;j<N;j++) 
    	{
    		ll g, p;
    		cin>>g>>p;
    		Gi.push_back(g);
    		Pi.push_back(p);
    		
    		
    	}
    	if(canGayleHit(Gi,Pi,M))
    		cout<<"YES"<<"\n";
    	else
    		cout<<"NO"<<"\n";
    	
    	
    }
    
    
    return 0;
}
