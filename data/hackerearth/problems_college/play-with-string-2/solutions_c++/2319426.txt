#include <bits/stdc++.h>
using namespace std;
int main()
{
    string S;
    int T,i,n,flag;
    cin>>T;
    while(T--)
    {
    	flag=0;
    	cin>>S;
    	n=S.length();
    	for(i=1;i<n;i++)
    	{
    		if(abs(S[i]-S[i-1])!=abs(S[n-i-1]-S[n-i]))
    		{
    			flag=1;
    			cout<<"Not Funny\n";
    			break;
    		}
    	}
    	if(!flag)
    	cout<<"Funny\n";
    }
    return 0;
}
