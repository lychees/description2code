#include <iostream>
#define ull unsigned long long int

using namespace std;

ull odd = 0 , even = 0 ; 

void util(ull x)
{
	int ex = 0 ; 
	if(x%2 == 0)
	{
		ex = x ;
		x-- ; 
	}
	
	ull ans = x*(x+1) ; 
	ans /= 2 ; 
	
	ull temp = (x+1)*(x-1) ; 
	temp /= 4 ;
	
	odd += (ans - temp) ; 
	even += (temp + ex) ;
}


int main()
{
    int test ;
    scanf("%d",&test) ;
    
    while(test--)
    {
    	ull r , c ; 
    	scanf("%lld%lld",&r,&c) ; 
    	
    	odd = even = 0 ;
    	
    	if(c >= r)
    	{
    		util(r) ; 
    	}
    	else
    	{
    		util(c) ;
    		long long int i = c+1 ; 
    		
    		for(;i<=r;i++)
    		{
    			if(i%2 == 0)
    				even += c ;
    			else
    				odd += c ; 
    		}
    		
    		
    		
    	}
    	
    	printf("%lld %lld\n",even,odd) ;
    }
    
    return 0;
}
