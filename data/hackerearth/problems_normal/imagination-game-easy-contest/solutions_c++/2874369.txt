#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <set>
#include <string>

using namespace std;

int main () {

    int t, n, res;
    int a[10];
    int x[10];

    scanf("%d", &t);

    while ( t-- ) {
        scanf("%d", &n);
		res = 0;
        for ( int i = 0; i < n; i++ ) {
            scanf("%d", &a[i]);
        }

        for ( int i = 0; i < n; i++ ) {
            scanf("%d", &x[i]);
            res = res ^ ( x[i] - 1 );
            res = res ^ ( a[i] - x[i]);
        }

        if ( res == 0 ) {
            printf("Second\n");
        }
        else {
            printf("First\n");
        }
    }
    return 0;
}
