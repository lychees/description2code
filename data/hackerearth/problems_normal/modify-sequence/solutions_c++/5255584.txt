#include<iostream>
#include<vector>
using namespace std;
int main()
{
	int a,b,c=0;
	vector<int> seq;
	cin>>a;
	for(int x=0; x<a; x++)
	{
		cin>>b;
		seq.push_back(b);
	}
	for(int x=0; x<a-1; x++)
	{
		while(seq[x]>0 && seq[x+1]>0)
		{
			if(seq[x]>seq[x+1])
			{
				seq[x] -= seq[x+1];
				seq[x+1] = 0;
			}
			else
			{
				seq[x+1] -= seq[x];
				seq[x] = 0;
			}
		}
	}
	for(int x=0; x<a; x++)
	{
		if(seq[x]>0)
		{
			c = 1;
			break;
		}
	}
	if(c==0) cout<<"YES";
	else cout<<"NO";
	return 0;
}
