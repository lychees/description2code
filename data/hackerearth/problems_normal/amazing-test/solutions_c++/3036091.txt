#include<iostream>
using namespace std;


int main(){
    long long M,t,n,x,j,i,sum;
    cin>>t;
    while(t--){
    cin>>n>>x;
    sum=0;
    long long arr[n];
    for(i=0;i<n;i++){
        cin>>arr[i];
        sum+=arr[i];
    }
    M=2*x;
    long long m[M+10];

    for(i=0; i<M+10; i++)
        m[i]=0;
    m[0]=1;
    for(i=0; i<n; i++)
        for(j=M; j>=arr[i]; j--){
            m[j] |= m[j-arr[i]];
        }
    for(i = x; i > 0; i--){
        if(m[i])
            break;
    }
    if(x>=(sum-i))
        cout<<"YES\n";
    else
        cout<<"NO\n";
    }
    return 0;

}
