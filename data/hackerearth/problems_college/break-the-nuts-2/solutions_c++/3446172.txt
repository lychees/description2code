#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
	int t,a;
	scanf("%d",&t);
	vector<int> v;
	
	for(int i=0;i<t;i++){
		scanf("%d",&a);
		if(a==1) v.push_back(i);
	}
	ll ans = 1;
//	for(int i=0;i<v.size();i++)
	//	printf("%d ",v.size());
	if(v.empty()) printf("0\n");
	else{
		for(int i=0;i<v.size()-1;i++){
		//	printf("%d %d\n",v[i+1],v[i]);
			ans*=(v[i+1]-v[i]);
		}
		printf("%lld\n",ans);
	}
    return 0;
}
