#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,q;
	cin>>n>>q;
	int i,j=0,total=0;
	vector<long long int>a;
	for(i=1;i<=sqrt(n);i++)
	{
		if(n%i==0)
		{
		   if(i==n/i)
		   {
		   	 a.push_back(i);
		   	 total++;
		   }
		   else
		   {
		   	 a.push_back(i);
		   	 total++;
		   	 a.push_back(n/i);
		   	 total++;
		   }
		}
	}
	while(q--)
	{
		long long int type ,k,c=0;
		cin>>type;
		if(type==1)
		{
			cin>>k;
			for(i=0;i<a.size();i++)
			{
				if(k%a[i]==0)
				{
					c++;
				}
			}
		}
		else
		if(type==2)
		{
			cin>>k;
			for(i=0;i<a.size();i++)
			{
				if(a[i]%k==0)
				{
					c++;
				}
			}
		}
		else
		if(type==3)
		{
			cin>>k;
			for(i=0;i<a.size();i++)
			{
				if(a[i]%k!=0)
				{
					c++;
				}
			}
		}
		cout<<c<<"\n";
	}
}
