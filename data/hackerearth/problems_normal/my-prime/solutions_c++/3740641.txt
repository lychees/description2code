#include <iostream>
using namespace std;

int main()
{
	int MAX = 1000000;
	int prime[1000002] = {0};
    int N;
    cin>>N;
    
    int data[N];
    
    int max = -1;
    
    for(int i=0; i<N; i++) {
    	cin>>data[i];
    	if(max < data[i])
    		max = data[i];
    }
    
    for(int i=0; i<N; i++) {
    	int val = data[i] * 2;
    	int j = 2;
    	while(val <= max) {
    		prime[val] = 1;
    		val = data[i] * ++j;
    	}
    }
    for(int i=0; i<N; i++) {
    	if(!prime[data[i]]) {
    		cout<< data[i] << " ";
    	}
    }
    return 0;
}
