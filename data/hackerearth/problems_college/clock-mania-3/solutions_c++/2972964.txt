#include <iostream>
using namespace std;
#define MIN(a,b) (((a)<(b))?(a):(b))

 
int calcAngle(double h, double m)
{
    if (h <0 || m < 0 || h >12 || m > 60)
        return 0;
 
    if (h == 12) h = 0;
    if (m == 60) m = 0;
 
    int hour_angle = 0.5 * (h*60 + m);
    int minute_angle = 6*m;
 
    int angle = abs(hour_angle - minute_angle);
 
    angle = MIN(360-angle, angle);
 
    return angle;
}

int main()
{
	int t, h, m,a;
	cin>>t;
	while(t--)
	{
		cin>>h>>m;
		a = calcAngle(h,m);
		cout<<a<<endl;
	}
    
    return 0;
}
