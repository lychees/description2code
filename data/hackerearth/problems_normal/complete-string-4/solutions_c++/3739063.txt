#include <iostream>
#include <string>
#include <vector>
#include <set>
using namespace std;


bool test(string& ch){
    bool ver=false;
    string alphabet = "abcdefghijklmnopqrstuvwxyz";
    for(size_t i=0;i<ch.size();i++){
        for(size_t j=0;j<alphabet.size();j++){
            if(ch.at(i)==alphabet[j]){
                ver =true;
                break;
            }else{
                ver=false;
            }
        }
    }
    return ver;
};

void verif(string ch){
    set<char>s;
    if(ch.size()<26){
        cout<<"NO"<<endl;
    }else{
        for(size_t j = 0; j < ch.size();j++)
        s.insert(ch[j]);
        if(s.size()==26){
            cout<<"YES"<<endl;
        }else{
            cout<<"NO"<<endl;;
        }
    }
}
