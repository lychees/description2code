
#include <iostream>
#include <stdlib.h>
#include <algorithm>

using namespace std;

int main()
{
    int test;

    long long a[100000],s1,s2,s3,prod;

    cin >> test;

    while(test--)
    {
        int n;

        s1=s2=s3=0 ;

        cin >> n;

        for(int i=0;i<n;i++)
        {
            cin >> a[i];

            s1=(s1+abs(a[i]))%1000000007;

        }

        sort(a,a+n);

        for(int i=0;i<n/2;i++)
        {
            s2+=-a[i];
        }

        for(int i=n/2;i<n;i++)
        {
            s3+=a[i];
        }

        s2=abs(s2+s3)%1000000007;

        prod=(s1*s2)%1000000007;

        cout << prod << "\n";

    }

}
