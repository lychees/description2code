#include <iostream>
using namespace std;

int gcd(int a, int b) {
	if(b==0) return a;
	return gcd(b,a%b);
}

int main()
{
    int T,N;
    cin>>T;
    while(T--) {
    	cin>>N;
    	int a,b;
    	N--;
    	cin>>a;
    	while(N--) {
    		cin>>b;
    		a=gcd(a,b);
    	}
    	cout<<a<<endl;
    }
    return 0;
}
