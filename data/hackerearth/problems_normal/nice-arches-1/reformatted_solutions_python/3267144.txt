def is_bubbly(s):
    stk = []
    for i in s:
        if len(stk) == 0:
            stk.append(i)
        elif i == stk[-1]:
            stk.pop()
        else:
            stk.append(i)
    if len(stk) == 0:
        return True
    return False


t = int(raw_input())
count = 0
while t > 0:
    t -= 1
    s = raw_input()
    if len(s) % 2 != 0:
        continue
    if is_bubbly(s):
        count += 1
print count
