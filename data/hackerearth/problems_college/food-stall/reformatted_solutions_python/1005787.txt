import sys
t = int(raw_input())
while t > 0:
    t -= 1
    n, d = map(int, sys.stdin.readline().split())
    a = []
    i = 0

    while i < n:
        x = int(raw_input())
        a.append(x)
        i += 1

    i = 0
    count = 0
    while i < n:
        j = i+1
        while j < n:
            if abs(a[i]-a[j]) == d:
                count += 1
            j += 1
        i += 1
    print count
