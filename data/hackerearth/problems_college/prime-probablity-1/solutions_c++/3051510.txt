#include<bits/stdc++.h>
using namespace std;
#define MAX 1000000
#define ll long long
int cnt[MAX+3],isprime[MAX+3];
void seive()
{
	for(ll i=2;i<=MAX+2;i++)
	isprime[i]=1;
	for(ll i=2;i*i<=MAX;i++)
	{
		for(ll j=i;i*j<=MAX;j++)
		{
			isprime[i*j]=0;
		}
	}
	for(ll i=2;i<=MAX;i++)
	cnt[i]=cnt[i-1]+isprime[i];
}
int main()
{
	seive();
	ll t,a,b;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&a,&b);
		ll num,den;
		den=b-a+1;
		num=cnt[b]-cnt[a-1];
		float ans;
		ans=num/(float)den;
		printf("%.6f\n",ans);
	}
	return 0;
}
