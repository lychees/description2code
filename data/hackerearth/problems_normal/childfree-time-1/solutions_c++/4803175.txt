#include <cstdio>
#include <iostream>
#include <algorithm>
#include <string.h>
#include <vector>
#include <set>
#include <map>
#include <stack>
#include <list>
#include <math.h>

#define ll long long int
#define maxN 1000000
#define maxVal (ll)1000000000000000000
#define minVal -100000000
#define mod 1000000007LL

#define gcd(a,b) __gcd(a,b)

using namespace std;

ll n,m;
vector<pair<ll,ll> > g[maxN+5];
ll d[maxN+5];
ll out[maxN+5]={0};
ll visit[maxN+5]={0};
stack<ll> s;

void topoSort(ll u)
{
	ll i,v;
	visit[u]=1;
	for(i=0;i<(ll)g[u].size();i++)
	{
		v=g[u][i].first;
		if(visit[v]==0)
			topoSort(v);
	}
	s.push(u);
}

int main()
{
    ll i,u,v,w,p,k;
    cin>>n>>m;
    while(m--)
    {
    	cin>>u>>v>>w;
    	out[u]++;
    	g[v].push_back(make_pair(u,w));
    }
    for(i=1;i<=n;i++)
    {
    	d[i]=-1;
    	if(out[i]==0)
    		d[i]=0;
    }
    for(i=1;i<=n;i++)
    	if(visit[i]==0)
    		topoSort(i);
    while(!s.empty())
    {
    	u=s.top();
    	s.pop();
    	if(d[u]!=maxVal)
    	{
    		for(i=0;i<(ll)g[u].size();i++)
    		{
    			v=g[u][i].first;
    			w=g[u][i].second;
    			if(d[v]==0)
    				continue;
    			d[v]=max(d[v],d[u]+w+2);
    		}
    	}
    }
    p=0;
    k=0;
    for(i=1;i<=n;i++)
    {
    	if(d[i]!=-1)
    	{
    		if(d[i]>p)
    		{
    			k=i;
    			p=d[i];
    		}
    	}
    }
    cout<<p<<"\n";
    return 0;
}
