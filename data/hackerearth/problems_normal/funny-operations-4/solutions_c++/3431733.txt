#include <iostream>
#include<vector>
using namespace std;

int main()
{
   	int t;
   	cin >> t;
   	while(t--){
   		int n,i,s; int a[20];
   		cin >> n;
   		for(i=0;i<n;i++)
   		cin >>a[i];
   		int mx = 0,len =0;
   		for(s = 1;s<(1<<n);s++)
   		{
   			int fx,fa,l=0;
   			bool f = false;
   			for(int e = 0;e< n;e++)
   			{
   				if(s & (1<<e))
   				{
   					if(!f)
   					{
   						fa = fx = a[e];
   						f = true;
   					}
   					else 
   					{
   						fa = fa & a[e];
   						fx = fx ^ a[e];
   					}
   					l++;
   				}
   			}
   			if((fa|fx)>mx){
   				mx = fa|fx;
   				len = l;
   			}
   			else if((fa | fx)==mx && l >len)len = l;
   		}
   		cout << mx << " "<< len <<endl;
   	}
      return 0;
}
