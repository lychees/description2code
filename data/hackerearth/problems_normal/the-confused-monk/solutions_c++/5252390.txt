#include <iostream>
using namespace std;

int gcd(int a,int b)
{
	if(b==0)
	{
		return a;
	}
	else
	{
		return gcd(b,a%b);
	}
}


int main()
{
    long long num=0,module=1000000007;
    cin>>num;
    long long dataArray[num];
    long long f=1;
    long long g=0;
    for(int i=0;i<num;i++)
    {
    	cin>>dataArray[i];
    	f=f*dataArray[i]%module;
    }
    if(num<=1)
    {
    	g=dataArray[0];
    }
    else
    {
    for(int i=1;i<num;i++)
    {
    	g=gcd(dataArray[i-1],dataArray[i]);
    	dataArray[i]=g;
    }
    }
    // cout<<"f is "<<f<<endl;
    // cout<<"g is "<<g<<endl;
    long long result=1;
    while(g!=0)
    {
    	if(g%2==1)
    	{
    		result=result*f%module;
    	}
    	f=f*f%module;
    	g/=2;
    }
    cout<<result<<endl;
    return 0;
}
