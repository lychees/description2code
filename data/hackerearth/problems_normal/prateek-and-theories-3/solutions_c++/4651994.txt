#include <bits/stdc++.h>
using namespace std;

typedef long long int ll;

int main()
{
	ll t,n,l,a,b,x,y,st=-1,en=-1,dp[10001],i,max,tp,tp2;
	map<ll,ll> m;
	
	cin>>t;
	
	while(t--)
	{
		cin>>n;
		m.clear();
		while(n--)
		{
			cin>>a>>b;
			
			m[a]++;
			m[b]--;
		}
		max=0;
		tp2=0;
		for(map<ll,ll>::iterator k=m.begin();k!=m.end();k++)
		{
			pair<ll,ll> temp=*k;
			tp=temp.second;
			tp2+=tp;
			if(tp2>max)
			max=tp2;
		}
		
		cout<<max<<endl;
	}
	
    return 0;
}
