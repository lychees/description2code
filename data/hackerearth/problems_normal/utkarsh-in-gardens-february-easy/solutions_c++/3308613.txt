#include <bits/stdc++.h>
using namespace std;

const int tam = 2005;
bitset<tam> Gmat[tam];
int visit[tam];
int cnt, act;

int main()
{
    for( int N; cin>>N; ){
    	//Gmat.assign(N, vector<int>(N, 0) );
    	for( int aux, i=0; i<N; i++){
    		for( int j=0; j<N; j++){
    			cin>>aux ;
    			Gmat[i][j] = aux;
    		}
    	}
    	
    	long long res= 0, cnt;
    	for( int i=0; i< N ; i++){
    		for( int j=i+1; j<N; j++ ){
    			cnt = (Gmat[i] & Gmat[j]).count();
    			res += cnt*(cnt-1)/2;
    		}
    	}
    	cout<<res/2<<"\n";
    }
    return 0;
}
