#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
		{
			int n;
			vector<int>food;
			vector<int>pok(1000000,0);
			cin>>n;
			int temp=0;
			while(n--)
					{
					int x,y;
					cin>>x>>y;
					food.push_back(x);
					pok[x]++;
					if(pok[y])
						{
							pok[y]--;
						}
					else
						{
						temp++;	
						}
					}
					cout<<temp<<endl;
		}
	
}
