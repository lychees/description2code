#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	int i,n,a[100000],s;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	scanf("%d",&s);
	sort(a,a+n);
	int min=0;
	for(i=n-1;i>=0;i--)
	{
		if(s>=a[i])
		{
			s-=a[i];
			min++;
		}
		else
			break;
	}
	if(s>0)
		min++;
	printf("%d\n",min);
	return 0;
}
