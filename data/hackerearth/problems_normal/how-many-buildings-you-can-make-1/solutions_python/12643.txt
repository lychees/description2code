from math import log

n= int(raw_input())
test_cases= []
for i in xrange(n):
	s= int(raw_input())
	test_cases.append(s)

def count_rectangles(s):
	count= s
	max_height= int(s**0.5)
	for j, i in enumerate(xrange(2, max_height+1)):
		count+= (s/i) - (j+1)
	return count

for test_case in test_cases:
	print count_rectangles(test_case)
