#include <iostream>
#include<string>
using namespace std;

int main()
{
    long t,i,j;
    cin>>t;
    string s;
    while(t--)
    {
    	long count=1,p=0,ans=0;
    	 cin>>s;
    	 for(i=1;i<s.length();i++)
    	 {
    	 	if(s[p]!=s[i])
    	 	{
    	 		count++;
    	 		p=i;
			  }
    	 }
    	 ans=s.length()-count;
    	 cout<<ans<<"\n";
       
    }
   
    return 0;
}
