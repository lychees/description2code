#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int lcs[1001][1001]={0};
        int i,j,n;
        cin>>n;
        int firstday[1001]={0},secondday[1001]={0},pos;
        i=1;
        while(i<=n)
        {
            cin>>pos;
            firstday[pos]=i;
            cin>>pos;
            secondday[pos]=i;
            i++;
        }
        for(i=1;i<=n;i++)
        {
            for(j=1;j<=n;j++)
            {
                if(firstday[i]==secondday[j])
                {
                    lcs[i][j]=lcs[i-1][j-1]+1;
                }
                else if(lcs[i-1][j]>lcs[i][j-1])
                {
                    lcs[i][j]=lcs[i-1][j];
                }
                else
                {
                    lcs[i][j]=lcs[i][j-1];
                }
            }
        }
        cout<<lcs[n][n]<<endl;
    }
    return 0;
}
