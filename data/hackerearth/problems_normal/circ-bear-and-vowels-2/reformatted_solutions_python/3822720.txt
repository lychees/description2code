import re

reg1 = re.compile('[^aeiouy]{3}')
vowels = ['a', 'e', 'i', 'o', 'u', 'y']


def checkHard(a):
    if reg1.findall(a):
        return 'hard'
    vowelCount = 0
    for i in vowels:
        vowelCount = vowelCount + a.count(i)
    conCount = len(a) - vowelCount
    if conCount > vowelCount:
        return 'hard'
    return 'easy'


for i in range(int(raw_input())):
    print(checkHard(raw_input()))
