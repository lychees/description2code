#include<bits/stdc++.h>
using namespace std;
int main(){


     int t;
     cin >> t;
     while(t--){

            long long  b , w , x , y , z;
            cin >>  b >> w >> x >> y >> z;
            
            long long black = b*x;
            long long white = w*y;
          long long ans = black + white;
        ans = min( ans , w*y +  b*(y+z));
        ans = min( ans , b*x + w*(x+z));
          cout << ans << endl;
              
     }
       return 0;
}
