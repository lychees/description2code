#include <iostream>
#include<string.h>
using namespace std;


int main()
{
    int t,cnt;
    cin>>t;
    
    while(t--)
    {
    	string c,s,ans;
        cin>>c;
       // cout<<c[0];
        int len=c.length();
        for(int i=0;i<len;i++)
        {
            if( c[i]>=48 && c[i]<=57 )
            {
                cnt=(int)(c[i]-'0');
                s=ans;
                for(int j=0;j < cnt-1;j++)
                {
                    //int l1=s.length();




                    ans.append(s);
                   //strcat(ans,s);
                }

            }
            else
            {
                ans.append(1,c[i]);
               //strcat(s,c[i]);
            }

        }
        cout<<ans<<endl;
    }


    return 0;
}
