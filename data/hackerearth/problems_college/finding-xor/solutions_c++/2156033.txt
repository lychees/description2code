#include<stdio.h>
#include<string.h>
int main()
{
    int t,i;
    char a[10000],b[10000];
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%s%s",a,b);
    	for(i=0;a[i]!='\0';i++)
    	{
    		if(a[i]==b[i])
    			printf("0");
    		else 
    			printf("1");
    	}
    	printf("\n");
    }
    return 0;
}
