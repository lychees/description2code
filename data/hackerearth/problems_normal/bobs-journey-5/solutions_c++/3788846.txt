#include<bits/stdc++.h>
using namespace std;

int ind[100000];
string ans[10000];

int hashf(string s)
{
 return s[0];
}

int main()
{
 int t;
 cin>>t;
 for(int i=0;i<t;i++)
 {
  int n,flag=0;
  memset (ind,0,100000);
  string str;
  
  cin>>n;
  for(int j=0;j<n;j++)
  {
   cin>>str;
   int s=hashf(str);
   ind[s]++;
   if(ind[s]>1)
   {
   	flag=1;
	
	
   }
  }
  if(flag==1)
  {
  	ans[i]="NO";
	
  }
  else
  {
   ans[i]="YES";
  }
 }
 for(int k=0;k<t;k++)
 	cout<<ans[k]<<endl;
}
