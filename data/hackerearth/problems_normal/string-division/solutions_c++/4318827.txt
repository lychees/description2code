#include <iostream>

using namespace std;

int main(){
int n;
cin>>n;
while(n--){
int i;
string a;
cin>>a;
int l=a.size();
int count =0 ;
if(l<4){
        cout<<"NO"<<endl;
        continue;
}
else if(l==4){
    int arr[26]={0};
    for(i=0;i<l;i++){
        if(arr[a[i]-'a']!=1){
            count = count+1;
            arr[a[i]-'a']=1;
        }
    }
    if(count==4) cout<<"YES"<<endl;
    else cout<<"NO"<<endl;
    continue;
}
else if(l==5||l==6){
int arr[26]={0};
    for(i=0;i<l;i++){
        if(arr[a[i]-'a']!=1){
            count = count+1;
            arr[a[i]-'a']=1;
        }
    }
    if(count>=3) cout<<"YES"<<endl;
    else cout<<"NO"<<endl;
    continue;
}
else if(l>6 && l<10){
int arr[26]={0};
    for(i=0;i<l;i++){
        if(arr[a[i]-'a']!=1){
            count = count+1;
            arr[a[i]-'a']=1;
        }
    }
    if(count>=2) cout<<"YES"<<endl;
    else cout<<"NO"<<endl;
    continue;
}
else cout<<"YES"<<endl;

}
return 0;
}
