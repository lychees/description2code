#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int n,l1,l2,i,j,score;
	cin>>n;
	for(i=0;i<n;i++)
	{
		string s1,s2;
		cin>>s1>>s2;
		l1=s1.length();l2=s2.length();score=0;
		for(j=0;j<min(l1,l2);j++)
		{
			if(s1.at(j)==s2.at(j))
			score++;
		}
		cout<<score<<endl;
	}
	return 0;
}
