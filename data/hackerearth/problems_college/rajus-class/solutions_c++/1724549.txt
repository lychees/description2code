#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
	ios::sync_with_stdio( false );
	
	int t, n, r, i, age;
		
	cin >> t;
	
	while( t-- ) {
		
		cin >> n;
		vector< pair<int, string> > A( n );
		
		for( i = 0;i < n;i++ ) {
			string s;
			cin >> age;
			cin >> s;
			pair<int, string> P;
			P.first = age;
			P.second = s;
			A[ i ] = P;
		}
		
		sort( A.begin(), A.end() );
		
		cin >> r;
		cout << A[ r - 1 ].second << endl;
	}
	return 0;
}
