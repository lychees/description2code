# print 'Hello World!'
t = input()
while t > 0:
    t -= 1
    s = raw_input()
    r = s[::-1]
    c = 0
    for i in range(1, len(s)):
        c += 1
        if abs(ord(s[i])-ord(s[i-1])) != abs(ord(r[i])-ord(r[i-1])):
            print"Not Funny"
            break
    if c == len(s)-1:
        print "Funny"
