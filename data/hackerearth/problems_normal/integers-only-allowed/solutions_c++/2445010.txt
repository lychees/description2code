#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;
typedef long long ll;

const int MAXN = 1005;
int N;
ll X[MAXN], Y[MAXN];

ll gcd(ll m, ll n)
{
	if (n == 0)
		return m;
	if (m % n == 0)
		return n;
	return gcd(n, m % n);
}

int main()
{
	ios::sync_with_stdio(0);
	
	cin >> N;
	for (int i = 0; i < N; i++)
		cin >> X[i] >> Y[i];
	
	ll A = 0;
	for (int i = 0; i < N; i++)
		A += X[i]*Y[(i+1)%N] - X[(i+1)%N]*Y[i];	
	A = abs(A);
	
	ll B = 0;
	for (int i = 0; i < N; i++)
	{
		ll x = abs(X[(i+1)%N] - X[i]);
		ll y = abs(Y[(i+1)%N] - Y[i]);
		B += gcd(x, y);
	}
	
	cout << (A - B + 2)/2 << "\n";
	
	return 0;
}
