#include<cstdio>
#include<algorithm>
using namespace std;

int main()
{
int tc;
unsigned long long int n,list[100005],diff[100005];
unsigned long long int result;
unsigned long long int i;

scanf("%d",&tc);
while(tc-->0)
{
	scanf("%llu",&n);	
	for(i=0;i<n;i++)
	{
		scanf("%llu",&list[i]);
	}	
	sort(list,list+n);	
	
	result=0;
	
	for(i=1;i<n;i++)
	{
		diff[i-1]=list[i]-list[i-1];
		result=(result+(diff[i-1]*i*(n-i)%1000000007))%1000000007;
	}
	
	result=(result*list[n-1])%1000000007;
	printf("%llu\n",result);
}
return 0;
}
