
#include <bits/stdc++.h>
#define ll long long int
#define pb push_back
using namespace std;
int main()
{
    int n,i,j;vector<int> f,m;
    cin>>n;
    for(i=0;i<n;i++){
        cin>>j;
        f.pb(j);
    }
    for(i=0;i<n;i++) {
        cin>>j;
        m.pb(j);
    }
    sort(f.begin(),f.end());
    sort(m.begin(),m.end());
    int a[25]={0},b[25]={0},ans=0;
    for(i=0;i<21;i++) a[i]=b[i]=0;
    for(i=0;i<f.size();i++){
        int x=25,y=25;
        for(j=0;j<m.size();j++){
            if(m[j]<f[i] and !b[j] and !a[i]){
                x=j;
            }
        }
        for(j=m.size()-1;j>=0;j--){
            if(m[j]<f[i] and !b[j] and !a[i]){
                y=j;
            }
        }
        if(x!=25 || y!=25 ) {a[i]=1;b[max(x,y)]=1;ans+=100;}
    }
    for(i=0;i<f.size();i++){
        for(j=0;j<m.size();j++){
            if(a[i]==0 and b[j]==0 and f[i]==m[j]){//cout<<j<<" "<<i<<endl;
                ans+=50;a[i]=1;b[j]=1;
            }
        }
    }
    cout<<ans;
}
