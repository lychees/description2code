#include <iostream>
using namespace std;

int main()
{
	int t;
	long long m,x,n,s,k,i;
    cin>>t;
    while(t--){s=0;
    	cin>>n>>m>>x;
    	int a[n];
    	for(i=0;i<n;i++)
    	{cin>>a[i];s=s+a[i];}
    	k=(x*(n+1))-s;
    	if(k<=0)
    	cout<<1<<endl;
    	else
    	if(k>m)
    	cout<<"Impossible\n";
    	else
    	cout<<k<<endl;
    }
    return 0;
}
