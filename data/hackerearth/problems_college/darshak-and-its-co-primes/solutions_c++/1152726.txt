#include <bits/stdc++.h>
#define ll long long int

using namespace std;

bool cp(int a, int b)
{
  for( ; ; ) {
    if(!(a %= b)) return b == 1 ;
    if(!(b %= a)) return a == 1 ;
  }
}

int main()
{
	ll t,p,q;
	bool ans;
	cin>>t;
	while(t--)
	{
		scanf("%lld %lld",&p,&q);
		ans=cp(p,q);
		if(ans)
			printf("Is a Co-Prime\n");
		else
			printf("Not a Co-Prime\n");
	}
	return 0;
}
