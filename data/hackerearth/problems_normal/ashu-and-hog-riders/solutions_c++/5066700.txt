#include<bits/stdc++.h>
using namespace std;
int a[1000][1000];
bool visited[1000000];
int main()
{
    int n;
    scanf("%d",&n);
    map<pair<int,int>,int > mp;
    mp.clear();
    map<pair<int,int>,int >::iterator it,it1;
    int k=0;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            scanf("%d",&a[i][j]);
            if(a[i][j]==1)
            {
                mp.insert(make_pair(make_pair(i,j),k));
                k++;
            }
        }
    }
    queue<pair<int,int> > Q;
    while(!Q.empty())
        Q.pop();
    memset(visited,false,sizeof(visited));
    int ans=0,temp;
    int dx[]={-1,1,0,0};
    int dy[]={0,0,-1,1};
    int x,y,x1,y1;
    for(it=mp.begin();it!=mp.end();it++)
    {
        if(visited[it->second]==false)
        {
            temp=1;
            Q.push(it->first);
            visited[it->second]=true;
            while(!Q.empty())
            {
                x=Q.front().first;
                y=Q.front().second;
                //cout<<"starting at:"<<x<<"  "<<y<<endl;
                Q.pop();
                for(int j=0;j<4;j++)
                {
                    x1=x+dx[j];
                    y1=y+dy[j];
                    if((x1<0)||(y1<0)||(x1>(n-1))||(y1>(n-1)))
                        continue;
                    it1=mp.find(make_pair(x1,y1));
                    if(it1==mp.end())
                        continue;
                    if(visited[it1->second]==false)
                    {
                        //cout<<"counted:"<<(it1->first).first<<"  "<<(it1->first).second<<endl;
                        Q.push(it1->first);
                        visited[it1->second]=true;
                        temp++;
                    }
                }
                //cout<<"temp="<<temp<<endl;
            }
            if(temp>ans)
                ans=temp;
        }
        //cout<<endl<<endl;
    }
    printf("%d\n",ans);
    return 0;
}
