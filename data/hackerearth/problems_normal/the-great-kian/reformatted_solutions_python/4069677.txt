n = input()
n = n+1
arr = [0]+list(map(int, raw_input().split()))
i, j, k = 1, 2, 3
sumi, sumj, sumk = 0, 0, 0
while i < n or j < n or k < n:
    if i < n:
        sumi = sumi+arr[i]
        i = i+3
    if j < n:
        sumj = sumj+arr[j]
        j = j+3
    if k < n:
        sumk = sumk+arr[k]
        k = k+3
print sumi,
print sumj,
print sumk
