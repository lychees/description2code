#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
	int t,n;
	int arr[100000];
	int brr[100000];
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		for(int i=0;i<n;i++)scanf("%d",arr+i);
		brr[0] = arr[0];
		for(int i=1;i<n;i++)brr[i] = brr[i-1]+arr[i];
		bool flag = false;
		for(int i=0;i<n && !flag;i++){
			if(i==0){
				if(brr[n-1]-brr[0]==0)flag = true;
			}else if(i==n-1){
				if(brr[n-2]==0)flag = true;
			}else{
				if(brr[i-1]==(brr[n-1]-brr[i]))flag = true;
			}
		}
		if(flag)cout << "YES" << endl;
		else cout << "NO" << endl;
	}
    return 0;
}
