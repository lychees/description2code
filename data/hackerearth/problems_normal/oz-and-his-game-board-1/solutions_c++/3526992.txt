#include<cstdio>

int max_value(int a, int b){
    if(a > b) return a;
    return b;
}

int main(){
    int t;
    scanf("%d", &t);
    while(t--){
        long long n, m, a, b ,c;
        scanf("%lld %lld", &n, &m);
        scanf("%lld %lld %lld", &a, &b, &c);
        if(n > m){
            if(c*2*m > a*m + b*m){
                printf("%lld\n", c*2*m + a*(n-m));
            }
            else{
                printf("%lld\n", a*n + b*m);
            }
        }
        else{
            if(c*2*n > a*n + b*n){
                printf("%lld\n", c*2*n + b*(m-n));
            }
            else{
                printf("%lld\n", a*n + b*m);
            }
        }
    }
}
