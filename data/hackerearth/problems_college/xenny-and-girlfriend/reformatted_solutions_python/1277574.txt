for _ in range(input()):
    a, b = map(int, raw_input().split())
    c = [0]*(a+1)
    counter = 0
    for _ in range(b):
        u, v = map(int, raw_input().split())
        c[u] += 1
        c[v] += 1
    for u in range(1, a+1):
        if c[u] & 1:
            counter += 1
    if counter == 0 or counter == 2:
        print "Yes"
    else:
        print "No"
