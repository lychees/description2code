#include <iostream>
#include <vector>
 
using namespace std;
 
int main()
{
    int t;
    long int m;
    vector<long int> n;
    cin>>t;
    for(int i=0;i<t;i++)
        {
            cin>>m;
            n.push_back(m);
        }
    for(int j=0;j<t;j++)
        {
        if(n[j]%2==0)
        cout<<n[j]<<endl;
        else
        cout<<n[j]-1<<endl;
        }
    return 0;
}
