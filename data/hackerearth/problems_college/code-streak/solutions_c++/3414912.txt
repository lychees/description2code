#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   int n;
   cin>>t;
   int s,r;
   while(t--){
   	cin>>n;
   	int count1 = 0;
   	int ans = INT_MIN;
   	map<int,int>mp1;
   	for(int i = 0; i < n; i++){
   		cin>>s>>r;
   		if(r == 1){
   			if(mp1[s] == 0){
   				count1++;
   				mp1[s] = 1;
   			}
   		}
   		
   		else{
   			ans = max(ans,count1);
   			count1 = 0;
   		}
   		
   	}
   	ans = max(ans,count1);
   	cout<<ans<<endl;
   }
    return 0;
}
