#include <bits/stdc++.h>
using namespace std;

void solve()
{
    int p, x, y, u, v;
    int step = 1;
    scanf("%d%d%d%d%d", &p, &x, &y, &u, &v);
    int d1 = u - x;
    int d2 = v - y;
    if((d1 != d2) && (abs(d1 - d2) % 2 == 0)) {
        step++;
    }
    if(x == u && y == v) {
        printf("Counter-Terrorists Win !\n");
    }
    else if(d1 == d2 && step < p) {
        printf("Counter-Terrorists Win !\n");
    }
    else {
        (step < p && (abs(d1 - d2) % 2 == 0)) ? printf("Counter-Terrorists Win !\n") : printf("Terrorists Win !\n");
    }
}

int main()
{
    int t;
    scanf("%d", &t);
    while(t--) {
        solve();
    }
    return 0;
}
