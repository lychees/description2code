#include <iostream>
#include <algorithm>
using namespace std;
#define N 100005
#define ll long long

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	ll A[N];
    	ll favour[N];
    	ll anger[N];
    	ll sum = 0;
    	for(int i = 0; i < n; ++i)
    	{
    		cin>>favour[i]>>anger[i];
    		sum -= anger[i];
    		A[i] = favour[i] + anger[i];
    	}
    	sort(A, A+n);
    	sum += A[n-1] + A[n-2];
    	cout<<sum<<endl;
    }
    return 0;
}
