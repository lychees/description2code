#include<bits/stdc++.h>
using namespace std;
int main(){
	long long s,x,q,n;
	bool flag;
	s=0;
	cin>>n;
	long long b[n];
	for(long long i=0;i<n;i++){
		cin>>x;
		s+=x;
		b[i]=s;
	}
	cin>>q;
	while(q--){
		cin>>x;
		flag=false;
		for(long long i=0;i<n;i++){
			if(b[i]>=x){
				cout<<i+1<<endl;
				flag=true;
				break;
			}
		}
		if(!flag){
			cout<<-1<<endl;
		}
	}
	return 0;
}
