#include <iostream>
#include <cmath>
#include <iomanip>

#define ll long long

{
    ll n = 0;
	cin >> n;
	for (ll i = 1; i <= n; i++)
		cout << fixed << setprecision(0) << fmod(((pow(1.6180339,i)-(pow(-0.6180339,i))) / 2.236067977) , pow(10,9)+7)  << endl;
    
    return 0;
}
