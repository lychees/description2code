#include<stdio.h>
#include<iostream>
using namespace std;
long long cal(long long int i) {
	return (i*(i+1))/2;
}
long long int find_m(long long int p,long long int q,long long int r) {
	long long int mid;
	while(p<=q) {
		mid = (p+q)/2;
		if(cal(mid)<=r) {
			if(mid+1<=q&&cal(mid+1)<=r) {
				p=mid+1;
			}
			else {
				return mid;
			}
		}
    	else {
    		q=mid-1;
    	}
	}
	return -1;
}

int main()
{
	long long int t,l,r,a,b,p,q,an,pp;
	cin>>t;
	while(t--) {
		an=pp=0;
		scanf("%lld%lld",&l,&r);
		a = find_m(1,2000000,l-1);
		//cout<<a<<endl;
		if(a!=-1) {
		
		if(a%2==0) {
			an+=((a/2)*(1+a-1))/2;
		}
		else {
			an+=(((a+1)/2)*(1+a))/2;
		}
		p = (l-1)-cal(a);
		if(p%2!=0) {
			an+=a+1;
		}
	}
		a = find_m(1,2000000,r);
		//cout<<a<<endl;
		if(a!=-1) {
		
		if(a%2==0) {
			pp+=((a/2)*(1+a-1))/2;
		}
		else {
			pp+=(((a+1)/2)*(1+a))/2;
		}
		p = (r)-cal(a);
		if(p%2!=0) {
			pp+=a+1;
		}
	}
		cout<<pp-an<<endl;
	}
	return 0;
}
