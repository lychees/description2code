s = list(raw_input().rstrip())
cords = {}
x, y = 100000, 100000
cords.update({(x,y):0})
for d in s:
    if d == 'R':
        y += 1
    elif d == 'L':
        y -= 1
    elif d == 'U':
        x -= 1
    else:
        x += 1
    if cords.has_key((x,y)):
        cords[(x,y)]+=1
    else:
        cords.update({(x, y): 0})
print sum(cords.values())
