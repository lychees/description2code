#include<bits/stdc++.h>
using namespace std;
const int MOD = 1000000007;
int main(){
    int t;
    long long trifib[100001];
    trifib[0]=1;
    trifib[1]=1;
    trifib[2]=2;
    for(int i=3;i<=100000;i++){
        trifib[i] = (trifib[i-1]+trifib[i-2]+trifib[i-3])%MOD;
    }
    cin>>t;
    int n;
    while(t--){
        cin>>n;
        cout<<trifib[n]<<endl;
    }
    return 0;
}
