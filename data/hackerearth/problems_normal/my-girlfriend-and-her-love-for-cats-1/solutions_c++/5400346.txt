#include<bits/stdc++.h>
using namespace std;
int main()
{
		long long int n,a,b;
		cin>>n;
		vector<long long int> v1;
		vector<long long int> v2;
		for(int i=0;i<n;i++)
		{
			cin>>a;
			v1.push_back(a);
		}
		for(int i=0;i<n;i++)
		{
			cin>>b;
			v2.push_back(b);
		}
		sort(v1.begin(),v1.end());
		sort(v2.begin(),v2.end());
		long long int s=0;
		for(int i=0;i<n;i++)
		{
			s=s+(v1[i]*v2[i]);
		}
		cout<<s;
		return 0;
	}
