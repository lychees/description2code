#include <iostream>
#include <algorithm>

using namespace std;

bool compare(int i,int j)
{
return i>j;
}
int main(void)
{
unsigned int k;
cin>>k;
unsigned int days=0;
unsigned int arr[k];
for(int i=0;i<k;i++)
arr[i]=0;

for(int i=0;i<k;i++)
cin>>arr[i];
unsigned int max=0;
unsigned int count=0;
sort(arr,arr+k,compare);
for(int i=0;i<k;i++)
{
count++;
days=count+arr[i];
if(days>max)
max=days;
}
cout<<max+1<<endl;
return 0;
}
