#include<stdio.h>
#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

vector<int> axis;
int check(int pos, int m){
	int prev = axis[0]+pos;
	m--;
	for(int i=1;i<axis.size();i++){
		if(prev+pos >= axis[i]) continue;
		else{
			if(m == 0) return 0;
			prev = axis[i]+pos;
			m--;
		}
	}
	return 1;
}
int main(){
	int N,M;
	cin >> N >> M;
	axis.resize(N);
	for(int i=0;i<N;i++)cin >> axis[i];
	sort(axis.begin(),axis.end());
	int e = 1000000;
	int st = 0;
	int mid;
	int ans = 10000000;
	while(st<=e){
		mid = (st+e)/2;
		int v = check(mid,M);
		if(v){
			ans = ans>mid?mid:ans;
			e = mid-1;
		}else st = mid+1;
	}
	cout << ans << endl;
	return 0;
}
