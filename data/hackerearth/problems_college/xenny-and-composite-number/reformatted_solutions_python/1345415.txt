def isPrime(n):
    for i in range(3, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True


for i in xrange(input()):
    num = input()
    if num < 4:
        print 4
    elif num % 2 == 1:
        print num+1
    else:
        if isPrime(num+1):
            print num+2
        else:
            print num+1
