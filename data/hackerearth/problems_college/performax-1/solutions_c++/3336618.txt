#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int per;
	cin>>per;
	int a[n];
	int i,j;
	for(i=0;i<n;i++)
	cin>>a[i];
	int sum=0;
	for(i=1;i<n;i++)
	{
		int c=0;
		if(a[i]<a[i-1])
		{
			while(1)
			{
				a[i] = a[i]+per;
				c++;
				if(a[i]>a[i-1])
				{
					break;
				}
			}
			
			sum = sum +c;
		}
	}
	cout<<sum<<"\n";
}
