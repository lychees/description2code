#include <stdio.h>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

typedef vector<int> vi;

bool cmp(vi x,vi y)
{
	return (x[0] < y[0]);
}

int main()
{
	int t,i,j,n,max;
	scanf("%d",&t);
	vi a2(2);	
	while(t--)
	{
		scanf("%d",&n);
		vector <vi >a(n,a2);
		for(i=0;i<n;i++)
			scanf("%d%d",&a[i][0],&a[i][1]);
		sort(a.begin(),a.end(),cmp);
		int b[1005]={0},ans=0;
		for(i=n-1;i>=0;i--)
		{
			max=0;
			for(j=i+1;j<n;j++)
			{
				if(a[j][1] > a[i][1] && b[j] > max)
					max=b[j];
			}
			b[i]=max+1; 
			if(b[i] > ans)		ans=b[i];
		}
		printf("%d\n",ans);
	}
	return 0;
}
