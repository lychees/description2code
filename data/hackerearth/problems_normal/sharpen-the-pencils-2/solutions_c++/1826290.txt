#include <iostream>

using namespace std;

int main()
{
	
	int t;
	cin>> t;
	while(t--){
		long long int n;
		cin>> n;
		double a[n];
		for(int i = 0; i < n; i++){
			cin >> a[i];
		}
		
		long long int i = 0;
		long long int j = n-1;
		
		double x;
		int flag = 0;
		int boy = 0;
		int girl = 1;
		if (n== 1){
			boy = 1;
			girl = 0;
		} else {
		while(i <= j){
			x = a[i]/2;
			boy++;
			while( x > 0){
				if( i == j ) {
					j--;
					break;
				} else if(a[j] == x) {
					a[j] = 0;
					j--;
					if(flag){
						girl++;
						flag = 0;
					}
					flag = 1;
					x = 0;
				} else if( a[j] > x){
					a[j] = a[j] -x;
					x = 0;
					if (flag){
						girl++;
						flag = 0;
					}
					
				} else {
					x -= a[j];
					a[j] = 0;
					j--;
					if(flag){
						girl++;
						flag = 0;
					}
					flag = 1;
				}
			}
			i++;
			if(i == j){
				
				break;
			}
			
			}
			
		}
	
		
		cout<<boy<<" "<<girl<<endl;
	}
	
   
    return 0;
}
