#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[501];
    bool b[501];
    for(int i=0;i<=500;i++){
    	a[i]=0;
    	b[i]=false;
    }
    for(int i=1;i<=n;i++){
    	int m;
    	cin>>m;
    	int num;
    	for(int j=1;j<=m;j++){
    		cin>>num;
    		if(!b[num])
    		b[num]=true;
    	}
    	for(int j=0;j<=500;j++){
    		if(b[j]){
    			a[j]++;
    		}
    		b[j]=false;
    	}
    }
    long long int sumall=0;
    long long int inall=0; 
    long long int mod=1000000000;
    for(int i=0;i<=500;i++){
    	if(a[i]!=0){
    		if(a[i]==(n)){
    			inall=(inall+i)%mod;
    			//cout<<i<<endl;
    		}
    		sumall=(sumall+i)%mod;
    	}
    }
    cout<<(-1*inall)<<endl;
    return 0;
}
