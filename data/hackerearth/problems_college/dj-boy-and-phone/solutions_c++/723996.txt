#include<bits/stdc++.h>
using namespace std;
long long int num=1e9+9;
main()
{
	long long int tc,i,j,n;
	scanf("%lld",&tc);
	long long int dp[10][10005];
	for(i=0;i<=9;i++)
	dp[i][1]=1;
	for(j=2;j<=10003;j++)
	{
		dp[0][j]=(dp[0][j-1]+dp[8][j-1])%num;
		dp[1][j]=(dp[1][j-1]+dp[2][j-1]+dp[4][j-1])%num;
		dp[2][j]=(dp[2][j-1]+dp[1][j-1]+dp[3][j-1]+dp[5][j-1])%num;
		dp[3][j]=(dp[3][j-1]+dp[2][j-1]+dp[6][j-1])%num;
		dp[4][j]=(dp[1][j-1]+dp[4][j-1]+dp[7][j-1]+dp[5][j-1])%num;
		dp[5][j]=(dp[5][j-1]+dp[2][j-1]+dp[4][j-1]+dp[6][j-1]+dp[8][j-1])%num;
		dp[6][j]=(dp[6][j-1]+dp[3][j-1]+dp[5][j-1]+dp[9][j-1])%num;
		dp[7][j]=(dp[4][j-1]+dp[7][j-1]+dp[8][j-1])%num;
		dp[8][j]=(dp[5][j-1]+dp[8][j-1]+dp[0][j-1]+dp[7][j-1]+dp[9][j-1])%num;
		dp[9][j]=(dp[9][j-1]+dp[6][j-1]+dp[8][j-1])%num;
	}
	while(tc--)
	{
		scanf("%lld",&n);
		long long int sum=0;
		for(i=0;i<10;i++)
		sum=(sum+dp[i][n])%num;
		printf("%lld\n",sum);
	}
}
