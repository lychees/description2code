#include <iostream>
using namespace std;

void merge(int *A,int start,int mid,int end)
{
    int p=start,q=mid+1;
    int j,k=0;
    int SA[end-start+1];
    for(j=start;j<=end;j++)
    {
        if(p>mid)
        {
            SA[k++]=A[q++];
        }
        else if(q>end)
        {
            SA[k++]=A[p++];
        }
        else if(A[p]>=A[q])
        {
            SA[k++]=A[p++];
        }
        else
        {
            SA[k++]=A[q++];
        }
    }
    int i;
    for(i=0;i<k;i++)
    {
    	A[start++]=SA[i];
    }
}

void mergesort(int *A,int start,int end)
{
    //int p=start,q=end;
    int mid=(start+end)/2;
    if(start<end)
    {
        mergesort(A,start,mid);
        mergesort(A,mid+1,end);
        merge(A,start,mid,end);
    }
    //return SA;
    //merge(A,start,mid,end);
}
