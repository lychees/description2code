#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define lli long long int
long int m=1e9+9;
void mul(lli a[][2],lli b[][2],lli c[][2])
{
	lli i,j,k;
	for(i=0;i<2;i++)
	{
		for(j=0;j<2;j++)
		{
			c[i][j]=0;
			for(k=0;k<2;k++)
			{
				c[i][j]+=a[i][k]*b[k][j];
				c[i][j]%=m;
			}
		}
	}
}
void mexp(lli a[][2],lli b,lli ans[][2])
{
	lli i,j;
	if(b==0)
	{
		for(i=0;i<2;i++)
		{
			for(j=0;j<2;j++)
			{
				if(i==j)
				{
					ans[i][j]=1;
				}
				else
				{
					ans[i][j]=0;
				}
			}
		}
		return;
	}
	lli tmp[2][2];
	mexp(a,b/2,tmp);
	if(b%2==0)
	{
		mul(tmp,tmp,ans);
	}
	else
	{
		mul(tmp,tmp,ans);
		for(i=0;i<2;i++)
		{
			for(j=0;j<2;j++)
			{
				tmp[i][j]=ans[i][j];
			}
		}
		mul(tmp,a,ans);
	}
}
int main()
{
    lli i,j,k,t,n,a[2][2]={{0,26},{1,0}},b[2][1]={26,26},tmp[2][2],fa[2][1];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	mexp(a,n-2,tmp);
    	for(i=0;i<2;i++)
    	{
    		for(j=0;j<1;j++)
    		{
    			fa[i][j]=0;
    			for(k=0;k<2;k++)
    			{
    				fa[i][j]+=tmp[i][k]*b[k][j];
    				fa[i][j]%=m;
    			}
    		}
    	}
    	if(n==1||n==2)
    	{
    		fa[0][0]=26;
    	}
    	cout<<fa[0][0]<<endl;
    }
    return 0;
}
