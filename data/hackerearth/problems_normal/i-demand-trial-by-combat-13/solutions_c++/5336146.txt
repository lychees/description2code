#include <bits/stdc++.h>
#define MAX 102
#define MOD 1000000007
using namespace std;
typedef long long int lli;
typedef unsigned long long ulli;
int Lannister, Hours;
bool Arr[MAX];
bool Check_1(bool Temp[])
{
    for(int i = 0; i < Lannister; i++)
        if(!Temp[i])
            return false;
    return true;
}
bool Check_0(bool Temp[])
{
    for(int i = 0; i < Lannister; i++)
        if(Temp[i])
            return false;
    return true;
}
void Solve()
{
    while(Hours--)
    {
        int Temp[MAX];
        for(int i = 0; i < Lannister; i++)
        {
            if(!i)
                Temp[i] = Arr[i + 1];
            else if(i == Lannister - 1)
                Temp[i] = Arr[i - 1];
            else
            {
                if(Arr[i - 1] && Arr[i + 1])
                    Temp[i] = Arr[i - 1];
                else
                    Temp[i] = 0;
            }
        }
        for(int i = 0; i < Lannister; i++)
            Arr[i] = Temp[i];
        if(Check_1(Arr))
            break;
        if(Check_0(Arr))
            break;

    }
    for(int i = 0; i < Lannister; i++)
        cout << Arr[i] << ' ';
    cout << endl;
}
int main()
{
    int test;
    cin >> test;
    while(test--)
    {
        cin >> Lannister >> Hours;
        for(int i = 0; i < Lannister; i++)
            cin >> Arr[i];
        Solve();
    }
}
