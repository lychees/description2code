#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    std::ios::sync_with_stdio(false);
    int t;
    long int n,x,y;
    cin>>t;
    while(t--){
    	cin>>n>>x>>y;
    	long long int a[n],b[n],i,sum=0;
    	for(i=0;i<n;i++){
    		cin>>a[i];
    		cin>>b[i];
    	}
    	sort(a,a+n);
    	sort(b,b+n);
    	for(i=0;i<n;i++){
    		if(a[i]==b[i])
    		continue;
    		else
    		{
    			if(a[i]<b[i])
    				sum+=(x*(b[i]-a[i]));
    			else
    				sum+=(y*(a[i]-b[i]));
    		}
    	}
    	cout << sum << endl;
    }
    
    return 0;
}
