t = int(raw_input())
dc = {}
for i in range(0,t):
    a = raw_input()
    b = a.split(' ')
    x1 = int(b[0])
    x2 = int(b[1])
    x3 = int(b[2])
    x4 = "%s"%(x1+x2+x3)
    x5 = "%s"%(x1*x2*x3)
    x6 = [x4,x5]
    x7 = "-".join(x6)
    if (dc.has_key(x7)):
        dc[x7] = dc[x7] + 1
    else:
        dc[x7] = 1
ans = 0
for a,b in dc.items():
    if(b==1):
        ans = ans+1
print ans
