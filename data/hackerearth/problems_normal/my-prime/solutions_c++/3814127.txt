#include <iostream>
using namespace std;

int main()
{
        int N;
    cin>>N;
    int numbers[N];
    int correctNumbers[N];
    for(int i=1;i<=N;i++)
    {
        cin>>numbers[i];
        correctNumbers[i]=numbers[i];
        for(int j=1;j<i;j++)
        {
          //  cout<<"Checking Number "<<numbers[j]<<" with "<<numbers[i]<<endl;
            if(numbers[j]%numbers[i]==0)
            {
                correctNumbers[j]=-1;
            }
            if(numbers[i]%numbers[j]==0)
            {
                correctNumbers[i]=-1;
            }
        }
    }
    for(int i=1;i<=N;i++)
    {
    if(correctNumbers[i]!=-1)
    {
    cout<<correctNumbers[i]<<" ";
    }
    }



    return 0;
}
