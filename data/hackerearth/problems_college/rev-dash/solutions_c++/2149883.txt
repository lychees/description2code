#include <bits/stdc++.h>

using namespace std;

int main()
{
	string a;
	cin >> a;
	for(int i = 0; i < a.size()-1; i++) {
		cout << a[i] << "-";
	}
	cout << a[a.size()-1] << endl;
	for(int i = a.size()-1; i > 0; i--) {
		cout << a[i] << "-";
	}
	cout << a[0] << endl;
	return 0;
	
}
