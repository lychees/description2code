#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
	int t;
	scanf("%d ",&t);
	while(t--)
	{
		int x;
		scanf("%d ",&x);
		x=x%26;
		char s[1000000];
		gets(s);
		int l=strlen(s);
		for(int i=0;i<l;i++)
		{
			if(s[i]>=65 && s[i]<=90)
			{
				if(s[i]+x>90)
					s[i]=64+(s[i]+x)%90;
				else
					s[i]=s[i]+x;
			}
			else
			{
				if(s[i]+x>122)
					s[i]=96+(s[i]+x)%122;
				else
					s[i]=s[i]+x;
			}
		}
		printf("%s\n",s);
	}
	return 0;
}
