#include<iostream>
using namespace std;
long long dp[57][257];
int c[57];
int main()
{
	int n, m;
	cin >> n >> m;
	for (int i = 1; i <= m; i++)
	{
		cin >> c[i];
	}
	dp[0][0] = 1;
	for (int i = 1; i <= m; i++)
	{
		for (int j = 0; j <= n; j++)
		{
			dp[i][j] = dp[i - 1][j];
			if (j >= c[i])
			{
				dp[i][j] += dp[i][j - c[i]];
			}
		}
	}
	cout << dp[m][n] << endl;
}
