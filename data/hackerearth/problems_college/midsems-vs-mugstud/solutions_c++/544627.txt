#include <bits/stdc++.h>
using namespace std;
int arr[102][102]; 
main()
{   
    int a; scanf("%d", &a);
    for (int g=0; g<a; g++)
    {
        int b; scanf("%d", &b);int dp[102][102]; memset(dp,0,sizeof(dp[0][0])*102*102); 
        memset(arr,0,sizeof(arr[0][0])*102*102); 
        for (int g=1; g<=b; g++)
        {
            for (int y=1; y<=b; y++)
            {
                for (int z=1; z<=b; z++)
                {
                    int r; scanf("%d", &r); 
                    int k=dp[y][z-1]+dp[y-1][z]-dp[y-1][z-1]+dp[y][z]; 
                    printf("%d ", r-k); 
                    arr[y][z]+=r-k; 
                    dp[y][z]=r; 
                }
                printf("\n"); 
            }
            for (int y=1; y<=b; y++)
            {
                for (int z=1; z<=b; z++)
                {
                    dp[y][z]=arr[y][z]; 
                }
                
            }
            
                    
        }
    }

}
