#include <iostream>
#include <algorithm>
#define mod 1000000007
using namespace std;

long long max(long long a, long long b)
{
	if (a <= b) return b;
	return a;
}
long long gcd(long long a, long long b)
{
	if (a % b == 0) return b;
	return gcd(b, a % b);
}

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin >> t;
    while (t--) {
    	long long a[3];
    	cin >> a[0] >> a[1] >> a[2];
    	sort(a, a + 2);
    	long long x = gcd(a[2], a[1]);
    	long long y = gcd(max(x, a[0]), min(x, a[0]));
    	//cout << "H" << endl;
    	cout << y << " ";
    	long long cnt = ((((a[0] / y) * (a[1] / y)) % mod) * (a[2] / y)) % mod;
    	cout << cnt << endl;
    }
    return 0;
}
