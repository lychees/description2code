#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;
long long int my_abs(long long int b)
{
	if(b<0)
		return b*-1;
	return b;
}
int main()
{
	int n;
	cin>>n;
	vector<long long int> arr;
	for(int i=0;i<n;i++)
	{
			long long int temp;
			cin>>temp;
			arr.push_back(temp);
	}
	sort(arr.begin(), arr.end());
	vector<long long int> farr;
	int start=0, end = n-1;
	while(start<end)
	{
		farr.push_back(arr[start]);
		farr.push_back(arr[end]);
		start++;
		end--;
	}
	if(start==end)
		farr.push_back(arr[start]);
	
	//ret
	long long int out = 0;
	for(int i=1;i<farr.size();i++)
	{
		out=out+my_abs(farr[i]-farr[i-1]);
	}
	out=out+my_abs(farr[n-1]-farr[0]);
	cout<<out;
	
	
	return 0;
}
