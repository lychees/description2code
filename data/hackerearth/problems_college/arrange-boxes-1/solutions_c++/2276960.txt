#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll a[100005],b[100005];

int main()
{

    int t,n;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	map<ll,ll> gr;
    	int i,j;
    	for(i=1;i<=n;i++)
    	scanf("%lld",&a[i]);
    	
    	for(i=1;i<=n;i++)
    	{
    		scanf("%lld",&b[i]);
    		gr[b[i]] = i;
    	}

    	for(i=1;i<=n;i++)
    	{
    		a[i] = gr[a[i]];    		
    	}

   ll total  = 0;
   
    for(i=1;i<=n;i++)
    for(j=i+1;j<=n;j++)
    if(a[j] < a[i])total++;
    printf("%lld\n",total);
    }
  
  
  
    return 0;
}
