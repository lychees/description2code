#include <bits/stdc++.h>
 
#define fi first
#define se second
#define pb push_back
#define mp make_pair
#define pii pair<int,int>
#define pll pair<ll,ll>
#define MOD 1000000007
#define ll long long int
#define MAXINT 2000000000
#define SET(x,y) memset(x,y,sizeof(x));
 
using namespace std;

int D;
ll DP[10][100005][2];

ll work(int i,int S,int k) {
	ll &ans = DP[i][S][k];
	int x,j;
	if(ans!=-1)	return ans;
	ans = 0;
	if(S==i) {
		if(k && D!=0)	ans = 0;
		else	ans = 1;
	}
	else if(S<i)	ans = 0;
	else {
		if(k) {
			for(j=i;j<=min(9,i+D-1);j++) {
				ans += work(j,S-i,1);
				ans%=MOD;
			}
			if(i+D<=9) {
				ans += work(i+D,S-i,0);
				ans%=MOD;
			}
		}
		else {
			for(j=i;j<=min(9,i+D);j++) {
				ans += work(j,S-i,0);
				ans%=MOD;
			}
		}
	}
	//cout<<i<<" "<<S<<" "<<k<<" "<<ans<<endl;
	return ans;
}

ll solve(void) {
	int S,i,j;
	cin>>S>>D;
	SET(DP,-1);
	ll ans = 0;
	for(i=1;i<=9;i++) {
		ans += work(i,S,1);
		ans%=MOD;
	}
	return ans;
}

int main () {
	int t,i;
	cin>>t;
	for(i=1;i<=t;i++) {
		cout<<"Case #"<<i<<": ";
		cout<<solve()<<endl;
	}
	return 0;
}
