#include <bits/stdc++.h>
#define m 1000000007
using namespace std;

long long int power(long long int a,long long int b)
{
	long long int ans=1;
	
	while(b>0)
	{
	if(b%2!=0)
	ans=(ans*a)%m;
	
	a=(a*a)%m;
	b=b/2;
	}
	return ans;
	
}

long long int inverse(long long int a)
{
	return power(a,m-2);
	
}

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	long long int n,k;
   	cin>>n>>k;
   	vector<long long>f(k+1,1);
   	
   	for(int i=2;i<=k;i++)
   	   f[i]=(f[i-1]*i)%m;
   	   
   	   long long int ans=(f[k]*(inverse((f[k-n]))%m))%m;
   	   cout<<ans<<endl;
  
   	
   	
   	
   	
   	
   }
}
