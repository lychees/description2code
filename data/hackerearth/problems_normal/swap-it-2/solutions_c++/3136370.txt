const int N = 1005;
const int MAX = 1000000000;
//cout<<fixed<<setprecision(10);
int n,k,ind;
int a[N];

void make_crash(int l, int r) {
	for(int i = r;i>l;i--)
		swap(a[i],a[i-1]);
}

int main() {
	sync;
	int t;
	cin >> t;
	for(int g = 0;g<t;g++) {
		cin >> n >> k;
		for(int i = 0;i<n;i++)
			cin >> a[i];
		for(int i = 0;i<n;i++) {
			ind = i;
			for(int j = i+1;j-i<=k && j<n;j++)
			 if (a[j] < a[ind])
				 ind = j;
			k-=(ind-i);
			make_crash(i,ind);
		}
		for(int i = 0;i<n;i++)
			cout << a[i] << " ";
		cout << endl;
	}
}
