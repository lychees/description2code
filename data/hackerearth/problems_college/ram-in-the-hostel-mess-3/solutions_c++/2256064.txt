 #include <iostream>
    using namespace std;
    int fun(int n)
    {
    	if(n==1)
    	 return 2;
    	else if(n==2)
    	 return 3;
    	else
    	 return (fun(n-1)+fun(n-2));
    }
    
    int main()
    {
        int t,n,i;
        cin>>t;
       for(i=0;i<t;i++)
        {
        	cin>>n;
        	int x=fun(n);
        	int ans=x*x;
        	cout<<ans<<endl;
        }
        return 0;
    }
