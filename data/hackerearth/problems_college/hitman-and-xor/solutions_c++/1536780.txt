#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,i,eve,odd,ctr,*a;
    cin>>t;
    while(t--)
    {
        odd=0;
        eve=0;
        cin>>n;
        a=new int [n];
        for(i=0;i<n;i++)
        {
            cin>>a[i];
            if(a[i]&1)
                odd++;
            else
                eve++;
        }
        cout<<1LL*odd*eve<<endl;
    }
return 0;
}
