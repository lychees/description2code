#include <iostream>
using namespace std;
int main()
{
    int i,t,n,k,a[101][2];
    long long int cnt;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	for(i=0; i<101; i++){
    		a[i][0]=0;
    		a[i][1]=0;
    	}
    	while(n--)
    	{
    		scanf("%d",&i);
    		a[i][0]++;
    	}
    	scanf("%d",&k);
    	cnt=0;
    	for(i=1; i<101; i++)
    	{
    		if(k-i<101 && k-i>=1)
    		{
    			if(a[i][1]==0 && a[k-i][1]==0){
    			if(a[i][0]>0 && a[k-i][0]>0)
    			{
    				if(i==(k-i))
    					cnt+=(a[i][0]*(a[i][0]-1)/2);
    				else
    					cnt+=(a[i][0]*a[k-i][0]);
    				a[i][1]=1;
    				a[k-i][1]=1;
    			}
    			}
    		}
    	}
    	printf("%lld\n",cnt);
    }
    return 0;
}
