#include<iostream>
#include<vector>
using namespace std;
void sol()
{
    int a,a1,c;
    int n,i,temp=0,arr[50001];
    cin>>a>>a1>>c>>n;
    arr[0]=a%10;arr[1]=a1%10;
     temp=(10*a+a1)%41;
    if(n==1)
    {
        if(a==0) cout<<"YES"<<endl;
        else
        cout<<"NO"<<endl;
        return ;
    }
    if(n==2)
    {
        if(temp==0) cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
        return ;
    }
    for(i=2;i<n;i++)
    {
        arr[i]=(arr[i-1]*c+arr[i-2])%10;
        temp=(temp*10+arr[i])%41;
    }
   if(temp==0) cout<<"YES"<<endl;
   else cout<<"NO"<<endl;
}
int main()
{
    long int t;
    cin>>t;
    while(t--) sol();
    return 0;
}
