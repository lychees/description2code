#include <stdio.h>
#include <iostream>

using namespace std;
# define MAX 10000007
int dp[1000010];
bool done[1000010];
int now(int n){
	if(done[n])
		return dp[n];
	int x=0,y=0,z=0;
	if(n<=0)
		return 0;
	if(n==1 ||n==2)
		return 1;
	if(n==3)
		return 2;
	if(n==4)
		return 4;
	
	if(n>=4)
		x=now(n-4);
	if(n>=3)
		y=now(n-3);
	z=now(n-1);
	if(x>MAX)
		x=x%MAX;
	if(y>MAX)
		y=y%MAX;
	if(z>MAX)
		z=z%MAX;
	done[n]=true;
	dp[n]= (x+y+z);
	return dp[n];
}
int main()
{
    //printf("Hello World!\n");
    int t;
    cin>>t;
    dp[0]=0;
    dp[1]=1;
    dp[2]=1;
    dp[3]=2;
    dp[4]=4;
    for(int i=5;i<=1000000;i++)
   	{
    	dp[i]=dp[i-1]+dp[i-3]+dp[i-4];
    	if(dp[i]>MAX)
    		dp[i]%=MAX;
   	}
    
    while(t--){
    	int n;
    	cin>>n;
    	cout<<dp[n]%MAX<<"\n";
    }
    return 0;
}
