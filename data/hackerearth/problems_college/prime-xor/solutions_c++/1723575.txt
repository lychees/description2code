#include<bits/stdc++.h>
using namespace std;
int a[10000000];
int main()
{
int i,j,t,m,n;
for(i=0;i<1000000;i++)
    a[i]=1;
    a[0]=0;
    a[1]=0;
for(i=2;i<1000000;i++)
{

    if(a[i]==1)
    {
        for(j=2*i;j<10000000;j+=i)
            a[j]=0;
    }
}
cin>>t;
while(t--)
{
    long int su=0;
    cin>>m>>n;
    if(m==2 || m== 1)
    {
        for(i=3;i<=n;i+=2)
          if(a[i]==1)
            su++;
            //if(m==1)
             //   su++;
            cout<<su<<endl;
    }
    else
        cout<<"0"<<endl;


}
return 0;
}
