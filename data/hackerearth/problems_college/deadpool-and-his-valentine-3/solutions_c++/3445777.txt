#include<bits/stdc++.h>
using namespace std;
int main ()
{
    std::ios::sync_with_stdio(false);
    cin.tie(0);
    int n,m;
    cin>>n>>m;
    set<int>s;
    for(int i=0;i<n;i++)
    {
        int x;
        cin>>x;
        for(int j=0;j<x;j++)
        {
        int y;
        cin>>y;
        s.insert(y);
        }
    }
    if(s.size()==m)
    cout<<"YES\n";
    else
    cout<<"NO\n";
}
