long long pcount(long long  n, long long d) {
  long long  r = 0;
  while (n != 0) {
    r += ((n%10) == d);
    n /= 10;
  }
  return r;
}
 
unsigned long long count(unsigned long long n, unsigned long long d) {
  if (n < 10)
    return (d > 0 && n >= d);
  if ((n % 10) != 9)
  return pcount(n, d) + count(n-1, d);
return 10*count(n/10, d) + (n/10) + (d > 0);
}
