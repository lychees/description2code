#include "bits/stdc++.h"

#define limit 1000001
#define ll long long
ll dp[1000001];
// vector<int> v[1000001];
ll a[1000001];


ll recurse(int n){
    
    if(dp[n] != -1)
        return dp[n];
    ll ans = 0;
    for(int j = 2; j  <= sqrt(n)+1; j++){
        if(n%j==0){
            ans = max(ans, recurse(n/j)+a[n]); 
            int fact = n/j;
            if(fact != j && fact != 1)
            ans = max(ans, recurse(n/fact)+a[n]);
        }
    }
    ans = max(ans, recurse(1)+a[n]);
    return dp[n] = ans;
}
