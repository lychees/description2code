#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    while(n--)
    {
    	char str[100];
    	int fre[26]={0},flag=0;
    	cin>>str;
    	for(int i=0;str[i]!='\0';i++)
    	 fre[str[i]-'a']++;
    	for(int i=0;i<26;i++) 
    	 if(fre[i]==0)
    	  flag=1;
    	if(!flag)
    	 cout<<"YES"<<endl;
    	else
    	 cout<<"NO"<<endl;    
    }
    return 0;
}
