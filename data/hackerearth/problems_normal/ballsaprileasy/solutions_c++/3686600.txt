using namespace std;
typedef long long int LL;
LL arr[1000010];
int main()
{
int N;
scanf("%d",&N);
for(int i=1;i<=N;i++)
	scanf("%lld",&arr[i]);

sort(arr+1 , arr+N + 1);

LL ans = 0 , l = 1;
int at = 1;
while(at<=N)
{
	if((l*l)>=arr[at])
		ans = ans + (l*l) - arr[at++];
	else
		ans = ans + (l*l);
	l++;
}
printf("%lld\n", ans);


return 0;
}
