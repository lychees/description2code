t=int(input())
while t>0:
    n=int(input())
    i=10
    tens=0
    while i<=n:
		tens=tens+int(n/i)
		i=i*10
    i=5
    fives=0
    while i<=n:
		fives=fives+int(n/i)
		i=i*5
    fives=fives-tens
    i=2
    twos=0
    while i<=n:
		twos=twos+int(n/i)
		i=i*2
    twos=twos-tens
    mini=min(twos,fives)
    print (mini+tens)
    t=t-1
