#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int count[26]; //frequency count
    	for(int j=0;j<26;j++)
    	{
    		count[j]=0; //initialize with zero
    	}
    	string city;
    	for(int i=0;i<n;i++)
    	{
    		cin>>city;
    		count[city.at(0)-'a']++; //check the first alphabet of city and increase count
    	}
    	int flag=0;
    	for(int j=0;j<26;j++)
    	{
    		if(count[j]>1)
    		{
    			flag=1;
    			break;
    		}
    	}
    	
    	if(flag==1)
    		cout<<"NO"<<endl;
    	else
    		cout<<"YES"<<endl;
    }
    return 0;
}
