#include<bits/stdc++.h>
using namespace std;
 
#define mod 1000000007
typedef long int ll;
 
int n,i,a[205],j,b[205],l,r,q,sum,sum1;
ll ans,val,dp1[2][400002],dp2[2][400002];
int main()
{
  	sum=sum1=0;
  	
  	cin>>n;
  	for(i=0;i<n;i++)
  		cin>>a[i],sum1+=a[i];
  	for(j=0;j<n;j++)
  		cin>>b[j],sum+=b[j];
  	sum=max(sum,sum1);
  	
  	for(i=0;i<=n;i++)
  	{
  		for(j=0;j<=sum;j++)
  		{
  			if(i==0)
  			{
  				if(j==0)
  				dp1[i&1][j]=dp2[i&1][j]=1;
  				else
  				dp1[i&1][j]=dp2[i&1][j]=0;
  				continue;
  			}
  			dp1[i&1][j]=dp1[(i-1)&1][j];
  			if(j>=a[i-1])
  				dp1[i&1][j]=(dp1[i&1][j]+dp1[(i-1)&1][j-a[i-1]])%mod;
  			dp2[i&1][j]=dp2[(i-1)&1][j];
  			if(j>=b[i-1])
  				dp2[i&1][j]=(dp2[i&1][j]+dp2[(i-1)&1][j-b[i-1]])%mod;
  		}
  	}
  	cin>>q;
  	for(i=1;i<=sum;i++)
  		dp2[n&1][i]=(dp2[n&1][i]+dp2[n&1][i-1])%mod;
  	ans=0;
  	for(i=0;i<=sum;i++)
  	{
  		l=max(0,i-q); r=min(sum,i+q);
  		val=dp2[n&1][r];
  		if(l!=0)
  			val-=dp2[n&1][l-1];
  		val=(val+mod)%mod;
  		ans=(ans+(dp1[n&1][i]*val)%mod)%mod;
  	}
  	cout<<ans<<"\n";
	return 0;
}
