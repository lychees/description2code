#include<iostream>
#include<vector>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cstdio>
#include<stack>
#include<queue>
using namespace std;
#define max 1000010
long long store[max];
int main()
{
   int t;
   cin>>t;
   for(int i=1;i<=max;i++)
      store[i]=i;
   for(int i=2;i<=max;i++)
   {
       if(store[i]==i)
       {
           for(int j=i*2;j<=max;j+=i)
           {
                store[j]-=store[j]/i;
           }
       }
   }
   store[1]=1;         
   for(int i=2;i<=max;i++)
   {
       if(store[i]==i)
         store[i]=i-1;
       store[i]+=store[i-1];
   }    
   while(t--)
   {
       int n,m;
       scanf("%d%d",&n,&m);
       n=n/m;
       printf("%lld\n",store[n]);
   }
   return 0;
}
