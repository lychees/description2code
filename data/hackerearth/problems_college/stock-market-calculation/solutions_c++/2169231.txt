#include <iostream>
using namespace std;

int main()
{
    long long int t,n,a[50000];
    cin>>t;
    while(t--){
    	cin>>n;
    	for(long long int i=0;i<n;i++){
    		
    		cin>>a[i];
    	}
    	
    	
    	long long int max=0,answer=0;
    	for(long long int i=n-1;i>=0;i--){
    		
    		if(a[i]>max)
    		{
    			max=a[i];
    		}
    		else{
    			
    			answer+=max-a[i];
    		}
    	}
    	
    	cout<<answer<<endl;
    }
    return 0;
}
