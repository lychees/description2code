#include<iostream>
#define MOD 1000000007
using namespace std;
 
int N, K ,A[101],ans=1;
int dp[100001];
long long int S, R;
 
void add(int &a,int b){
	a+=b;
	if (a>=MOD)
      a-=MOD;
}
 
int main(){
 
 
cin >> N >> K;
K--;
 
for (int i=0;i<N;i++){
   cin>>A[i];
   S += A[i];
   ans*=2;
   ans%=MOD;
}
 
dp[0]=1;
 
for (int i=0;i<N;i++)
	for (int j=K-A[i];j>=0;--j)
		if (dp[j]>0)
            add(dp[j+A[i]],dp[j]);
 
int ans2 = 0;
 
for (int i=0;i<=K;i++){
	R=S-i;
	if (R<=K)
		add(ans2,dp[i]);
	else
		add(ans2,2*dp[i]%MOD);
}
 
ans=(ans-ans2+MOD)%MOD;
 
cout<<ans<<endl;
 
return 0;
}
