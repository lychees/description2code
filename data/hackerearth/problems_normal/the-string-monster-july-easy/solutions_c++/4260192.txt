#include <bits/stdc++.h>
using namespace std;

int t,n,a[35][30],ans; string s;

void rec(int pos, vector<int> v){
	if (pos>n){
		int ok=1;
		for (int i=0;i<26;i++) ok&=!v[i];
		if (ok) ans++; return;
	}
	rec(pos+1,v);
	int ok=1;
	for (int i=0;i<26;i++) ok&=v[i]>=a[pos][i];
	if (ok){
		for (int i=0;i<26;i++) v[i]-=a[pos][i];
		rec(pos+1,v);
	}
}

int main(){
	ios::sync_with_stdio(0); cin.tie(0);
	cin >> t;
	while (t--){
		cin >> n;
		memset(a,0,sizeof(a));
		for (int i=1;i<=n;i++){
			cin >> s;
			for (int j=0;j<(int)s.length();j++)
			a[i][s[j]-'a']++;
		}
		cin >> s;
		vector<int> v(26);
		for (int i=0;i<(int)s.length();i++)
		v[s[i]-'a']++;
		ans=0; rec(1,v);
		cout << (ans?"YES":"NO") << "\n";
	}
}
