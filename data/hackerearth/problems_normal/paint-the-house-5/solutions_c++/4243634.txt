#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
typedef long long ll;
struct worker
{
    ll t,x,y;
};
bool way(worker a,worker b)
{
    if(a.t==b.t)
    {
        if(a.y==b.y)
        {
            return a.x<b.x;
        }
        else{
            return a.y>b.y;
        }
    }
    else
    {
        return a.t<b.t;
    }
}

int main()
{
    ll n,b;
    cin>>n>>b;
    worker a[n];
    for(ll i=0;i<n;i++)
    {
        ll s,f,g;
        cin>>s>>f>>g;
        a[i].t=s;
        a[i].x=f;
        a[i].y=g;
    }
    sort(a,a+n,way);
    ll best=0,cost=0,time=0,area=0;
    for(ll i=0;i<n;i++)
    {
        ll r=best*(a[i].t-time);
        area+=r;
        if(area>=b)
        {
            break ;
        }
        if(a[i].y>best)
        {
            cost+=a[i].x;
            best=a[i].y;
        }
        time=a[i].t;
        
    }
    cout<<cost;
}
