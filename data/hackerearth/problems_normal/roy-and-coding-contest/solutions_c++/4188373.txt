#include <iostream>
#include<vector>
#include<cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,m;
    	cin>>n>>m;
    	int pen=1;
    	int pc=1;
    	int count=1;
    	if(n==1)
    	count=0;
    	
    	while(pen<m && pc<n)
    	{
    		pen*=2;
    		pc*=2;
    		count++;
    	}
     if(pen>=m && pc<n)
     {
     	pen=m;
     	pc=n-pc;
     //	float x=pc/m;
     	int y=floor(pc/m);
     	if(pc%m==0)
     	count+=y;
     	else
     	count+=y+1;
     	
     }
     cout<<count<<endl;
    	
    }
    return 0;
}
