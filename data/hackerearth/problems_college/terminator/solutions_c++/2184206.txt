#include <bits/stdc++.h>
using namespace std;
//int arr[26];
int main()
{
	int test;
	cin>>test;
	while(test--)
	{
	string s; cin>>s;
	int arr[s.size()];
	int c=0;
	memset(arr,0,sizeof(arr));
	for(int i=0;i<s.size();i++)
	{
		if(s[i]=='x')c=1;
		if(s[i]=='*')c=0;
		if(s[i]=='o'&&c==1)arr[i]=1;
	}
	c=0;
	for(int i=s.size()-1;i>=0;i--)
	{
		if(s[i]=='x')c=1;
		if(s[i]=='*')c=0;
		if(s[i]=='o'&&c==1)arr[i]=1;
	}
	for(int i=0;i<s.size();i++)if(arr[i]==0)cout<<s[i];
	cout<<endl;
	}
}
