s = raw_input()
points = []

points.append([0, 0])

x = 0
y = 0
f = 0
for i in s:
    if i == 'L':
        y = y-1
    elif i == 'R':
        y = y+1
    elif i == 'U':
        x = x-1
    elif i == 'D':
        x = x+1

    points.append([x, y])

points.sort()
# print points
pre = points[0]
for i in range(1, len(points)):
    # print "com[aring ",points[i]," with ",pre
    if points[i] == pre:
        f += 1
    pre = points[i]

print f
