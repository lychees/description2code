#include <iostream>
#include <math.h>
using namespace std;
int main()
{
	long T, A[100000], B[100000], index[100000], no, count, x;
	cin >> T;
	while (T--)
	{
		cin >> no;
		count = 0;
		for (long i = 0; i<no; i++)
		{
			cin >> A[i];
			index[A[i]] = i;
		}
		for (long i = 0; i<no; i++)
		{
			cin >> B[i];
		}
		for (long i = no - 1; i >= 0; i--)
		{
			x = B[i];
			long temp = index[x];
			count += abs(temp - i);
			for (long j = temp; j<i; j++)
			{
				long temp1 = A[j];
				A[j] = A[j + 1];
				A[j + 1] = temp1;
				index[A[j]]--;
			}
		}

		cout << count << "\n";
	}
}
