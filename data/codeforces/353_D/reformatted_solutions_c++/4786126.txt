#include <algorithm>
#include <cmath>
#include <cstdio>
#include <iostream>

using namespace std;

string s;
int ans = 0, man = 0, n, i;

int main() {
  cin >> s;
  n = s.length();
  for (i = 0; i < n; i++) {
    if (s[i] == 'M')
      man++;
    else if (man > 0)
      ans = max(ans + 1, man);
  }
  cout << ans;
}
