#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n,x,q,ans;
	cin>>n;
	vector <int> a(n);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
	for(int i=1;i<n;i++)a[i]+=a[i-1];
	cin>>q;
	while(q--)
	{
		scanf("%d",&x);
		ans = lower_bound(a.begin(),a.end(),x) - a.begin();
		if(ans == n)puts("-1");
		else printf("%d\n",ans+1);
	}
	return 0;
}
