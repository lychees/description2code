const double eps = 1e-7;
const int MAXN=1000+100;
const int mod=1000*1000*1000 + 7;
int l[MAXN];
bool can[MAXN];
int dp[MAXN];

int main()
{
	int n,k;
	cin>>n>>k;
	for(int i=1;i<=k;i++)
		cin>>l[i];
	sort( l+1 , l+k+1 );
	can[0]=1;
	for(int i=1;i<MAXN;i++)
		for(int j=1;j<=k;j++)
			if( i-l[j]>=0 && can[ i-l[j] ] )
				can[i]=1;
	dp[0]=1;
	for(int i=1;i<=n;i++)
	{
		dp[i]=dp[i-1];
		for(int j=1;j<=i;j++)
			if( can[j] && i-j>0 )
				dp[i]=( dp[i] + dp[i-j-1] )%mod;
		if( can[i] )
			dp[i]++;
	}
	cout<< dp[n] << endl;
}
