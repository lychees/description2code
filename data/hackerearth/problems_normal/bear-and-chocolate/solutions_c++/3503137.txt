#include<iostream>
using namespace std;
int main()
{
int T,N,row[1000],col[1000];
char mat[1000][1000];
int sumofrow=0, sumofcol=0,total=0,part=0;
int flag=0;
cin>>T;
while(T--)
{
	
	sumofrow=0; sumofcol=0;total=0;part=0;flag=0;
	cin>>N;
	
	for(int i=0;i<N;i++)
	{
		for(int j=0;j<N;j++)
		{
			cin>>mat[i][j];
		}
	}
	
	for(int i=0;i<N;i++)
	{
		for(int j=0;j<N;j++)
		{
			if(mat[i][j]=='#')
			{
				++row[i];
			}
		}
	}
	for(int i=0;i<N;i++)
	{
		for(int j=0;j<N;j++)
		{
			if(mat[j][i]=='#')
			{
				++col[i];
			}
		}
	}
	
	for(int i=0;i<N;i++)
	{
		total=total+row[i];
		
	}
	if(total%2==0)
	{
		part=total/2;
	for(int i=0;i<N;i++)
	{
		if(part==sumofrow)
		{
			flag=1;
			break;
		}
		else{
		sumofrow=sumofrow+row[i];
		}
	}
	if(!flag)
	{
		for(int i=0;i<N;i++)
		{
		if(part==sumofcol)
		{
			flag=1;
			break;
		}
		else
			{
			sumofcol=sumofcol+col[i];
			}
		}
	}
	}
	for(int i=0;i<N;i++)
	{
		for(int j=0;j<N;j++)
		{
			row[i]=0;
			col[i]=0;
		}
	}
	if(flag)
	{
		cout<<"\nYES";
	}
	else
	{
		cout<<"\nNO";
	}
	
}
return 0;
}
