#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
pair<int,int> bees[40010];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        int a,b,c;
        scanf("%d%d%d%d",&n,&a,&b,&c);
        for(int i=0;i<n;i++)
        {
            int x,y;
            scanf("%d%d",&x,&y);
            bees[2*i]=make_pair(x,-1);
            bees[2*i+1]=make_pair(y,1);
        }
        sort(bees,bees+2*n);
        ll maxx=-1;
        int inside=0;
        int passed=0;
        for(int i=0;i<2*n;i++)
        {
            if(bees[i].second==1) passed++;
            inside-=bees[i].second;
            int left=n-passed-inside;
            maxx=max(maxx,1ll*c*passed+1ll*b*inside+1ll*a*(n-inside-passed));
        }
        printf("%lld\n",maxx);
    }
    return 0;
}
