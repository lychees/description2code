#include<bits/stdc++.h>
using namespace std;
int a[110],n; 
int gcd(int x ,int y)
{
    if ( y==0 ) return x;
    return gcd( y , x % y );
} 
int main()
{
    scanf("%d",&n);
    for (int i = 1; i <=n; ++i)scanf("%d",&a[i]);
    sort(a+1,a+n+1);
    int ans=a[1];
    for (int i =2; i <=n; ++i)ans = gcd( ans , a[i] );
    if (( ( a[n] / ans - n ) & 1 )==1) printf("Alice");
    else printf("Bob");
    return 0;
}
