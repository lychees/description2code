#include <iostream>
using namespace std;
 
int main()
{
	long long a[]={1729, 4104, 13832, 20683, 32832, 39312, 40033, 46683, 64232, 65728, 110656, 110808, 134379, 149389, 165464, 171288, 195841, 216027, 216125, 262656, 314496, 320264, 327763, 373464, 402597, 439101, 443889, 513000, 513856, 515375, 525824, 558441, 593047, 684019, 704977};
	long long i,n,t,ans;
	cin>>t;
	while(t--)
	{
		cin>>n;
		if(n<a[0])
			ans=-1;
		else
		{
			for(i=0;i<34;i++)
			{
				if(n>=a[i]&&n<a[i+1])
				{
					ans=a[i];
					break;
				}
			}
			if(n==a[34])
				ans=n;
		}
		cout<<ans<<endl;
	}
    return 0;
}
