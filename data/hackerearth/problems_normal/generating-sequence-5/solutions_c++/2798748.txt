#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    for(int i=0;i<t;++i)
    {
    	long long int g,n,val;
    	scanf("%lld%lld",&g,&n);
    	val = (g*2);
    	for(int j=0;j<=(n-2);++j)
    	{
    		printf("%lld ",val);
    	}
    	val = (g*3);
    	printf("%lld\n",val);
    }
    return 0;
}
