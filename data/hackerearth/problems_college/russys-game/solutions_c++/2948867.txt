// CP template
#include <bits/stdc++.h>

using namespace std;

long gcd(long a, long b){
	if(b==0){
		return a;
	}
	else
		return gcd(b, a%b);
}

int main()
{
	int t;
	cin>>t;
	for(int i=1; i<=t; i++){
		int n;
		cin>>n;
		vector<int> mem;
		for(int j=0; j<n; j++){
			int x;
			cin>>x;
			mem.push_back(x);
		}
		set<int> visit;
		visit.insert(0);
		int k=0; bool loop = false, finish =false;
		while(!finish && !loop){
			set<int>::iterator it= visit.find(mem[k]);
			if(it != visit.end())
				loop = true;
			else if(mem[k] == -1)
				finish = true;
			else
				visit.insert(mem[k]);
			k = mem[k];
		}
		if(finish)
			printf("Win\n");
		else
			printf("Lose\n");
	}
	return 0;
}
