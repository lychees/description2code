#include <bits/stdc++.h>
using namespace std;

int gcd(long long a,long long b)
{
	long long temp;
	while(b!=0)
	{
		temp=a%b;
		a=b;
		b=temp;
	}
	return a;
}
int main()
{
    long long T,ar[3],first,ans,cubes;
    cin >> T;
    while(T--)
    {
    	cubes=1;
    	cin >> ar[0] >> ar[1] >> ar[2];
    	first=gcd(ar[0],ar[1]);
    	ans=gcd(first,ar[2]);
    	for(int i=0;i<3;i++)
    		ar[i]/=ans;
    	
    	for(int i=0;i<3;i++)
    	{
    		cubes*=ar[i];
    		cubes%=1000000007;
    	}
    	cout << ans << " " << cubes << endl;
    }
}
