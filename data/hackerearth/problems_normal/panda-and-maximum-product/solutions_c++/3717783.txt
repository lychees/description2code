#include <bits/stdc++.h>
using namespace std;
long long int a[1000069];
int main()
{
    long long int ans,n,i,j,ans1;
    scanf("%lld",&n);
    for(i=0;i<n;i++)
    {
    	scanf("%lld ",&a[i]);
    	//printf("%lld %lld\n",i,a[i]);
    }
    sort(a,a+n);
    ans=a[n-1]*a[n-2];
    ans1=a[0]*a[1];
    //printf("%lld %lld\n",a[0],a[1]);
    if(ans>ans1)
    	printf("%lld\n",ans);
    else
   		printf("%lld\n",ans1);
    return 0;
}
