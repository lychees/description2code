#include <stdio.h>
 int main() 
{
	int t,a,b,c;
	int max1, max2;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&a);
		scanf("%d",&b);
		scanf("%d",&c);
		if(a>=b)
		{
			max1=a;
			max2=b;
		}
		else
		{
			max1 = b;
			max2 = a;
		}
 		if(max2<=c)
		max2=c;
		if(max2!=0&&max1!=0)
		{
			if(max1>max2)
			{
				printf("%d\n",2*max2+1);
			}
			else
			{
				if(max1!=max2)
					printf("%d\n",2*max1+1);
				else
					printf("%d\n",2*max1);
			}
	
		}
		else if(max1==0&&max2==0)
			printf("%d\n",0);
		else 
			printf("%d\n",1);
	}
	return 0;
}
