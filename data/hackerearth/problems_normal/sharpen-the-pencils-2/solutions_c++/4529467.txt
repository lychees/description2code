#include<iostream>
using namespace std;
long long int arr[1000001];
int main()
{
    long long int t,n;
    ios::sync_with_stdio(false);
    cin>>t;
    while(t--)
    {
        cin>>n;
        long double sum = 0;
        for(long long int i=0;i<n;i++)
        {
            cin>>arr[i];
            sum+=(long double)arr[i];
        }
        long double mid = (sum*2.0)/3.0;
        long double start = 0;
        long long int j=0;long long int count1=0;long long int count2=0;
        //cout<<sum<<' '<<mid<<endl;
        while(start+arr[j]<mid)
        {
            start+=arr[j];
            count1++;
            j++;
        }
        if(start+arr[j]==mid)
        {
          //  cout<<"S"<<endl;
            count1++;
            count2=n-count1;
            cout<<count1<<' '<<count2<<endl;
        }
        else{
        count2=n-1-count1;
        if(start/2.0<=(sum-start-arr[j]))
            count1++;
        else if(start/2.0>(sum-start-arr[j]))
            count2++;
        cout<<count1<<' '<<count2<<endl;
        }
    }
}
