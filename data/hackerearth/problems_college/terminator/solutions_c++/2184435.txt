#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,i,j,c,l;
    char s[101];
    scanf("%d",&t);
    while(t--){
        int x[101]={0},ans[101]={0};
        c = 0;
        scanf("%s",s);
        l=strlen(s);
        for(i=0;i<l;i++){
            if(s[i] == 'x')
                x[c++] = i;
        }
        for(i=0;i<c;i++){
            for(j=x[i];j<l;j++){
                if(s[j] == '*')
                    {
                        ans[j] = 1;
                        break;
                    }
                else if(s[j] == 'x')
                    ans[j] = 1;
                else
                    ans[j] = -1;
            }
            for(j=x[i];j>=0;j--){
                if(s[j] == '*')
                    {
                        ans[j] = 1;
                        break;
                    }
                else if(s[j] == 'x')
                    ans[j] = 1;
                else
                    ans[j] = -1;
            }
        }
        for(i=0;i<l;i++){
            if(ans[i] != -1)
                printf("%c",s[i]);
        }
        printf("\n");
    }
    return 0;
}
