#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    vector<ll> v(100000);
    ll sum1,sum2,i,t,n,sum,flag;
    scanf("%lld",&t);
    while(t--)
    {
    	flag=0;
    	scanf("%lld",&n);
    	sum=sum1=sum2=0;
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&(v[i]));
    		sum+=v[i];
    	}
    	sum1=v[0];
    		for(i=1;i<n-1;i++)
    		{
    			sum2=sum-sum1-v[i];
    			if(sum1==sum2)
    			{
    				flag=1;
    				break;
    			}
    			sum1+=v[i];
    		}
    	if(flag)
    		printf("YES\n");
    	else
    		printf("NO\n");
    }
    return 0;
}
