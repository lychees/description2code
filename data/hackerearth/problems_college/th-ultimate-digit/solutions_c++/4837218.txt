#include <iostream>
#include <string>
#include <cmath>
using namespace std;

int main() {
	int t, l, m, v, ld;
	string a;
	long long b;
	cin >> t;
	while(t--) {
		cin >> a >> b;
		l = a.size();
		ld = a[l-1] - 0x30;
		if(!ld) {
			cout << "0\n";
			continue;
		} else if((ld & 1) && ld % 5) {
			m = a[l-1] - 0x30;
			v = ((int) pow(m, b % 4)) % 10;
		} else {
//			dprintf(2, "%i, %li\n", ld, b);
			switch(ld) {
				case 2:
					switch((b-1)%4) {
						case 0: v=2; break;
						case 1: v=4; break;
						case 2: v=8; break;
						case 3: v=6; break;
					}
					break;
				case 4:
					v = (b&1) ? 4 : 6;
					break;
				case 8:
					switch((b-1)%4) {
						case 0: v=8; break;
						case 1: v=4; break;
						case 2: v=2; break;
						case 3: v=6; break;
					}
					break;
				case 5: v = 5; break;
				case 6: v = 6; break;
			}
		}
		cout << v << "\n";
	}
	return 0;
}
