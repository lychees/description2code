#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000000007
int main()
{
    int x,S,Q,A,sel[100005];
    for(int i=0;i<=100000;i++)
        sel[i]=0;
    sd(x);
    sd(Q);
    while(Q--)
    {
        S=x;
        sd(A);
        if(A<S)
                swap(A,S);
        while(A%S!=0)
        {
            A%=S;
            if(A<S)
                swap(A,S);
        }
        if(sel[S]==0)
        {
            printf("%d\n",S);
            sel[S]=1;
        }
        else
            printf("-1\n");
    }
	return 0;
}
