#include <iostream>
#include <cstdio>
using namespace std;
int calcLift( int n, int j, int L, char direction )
{
	int time = 20;
	if (('U' == direction ) || ('u' == direction))
	{
		time+= ( n - L ) * 5 + n * 5;
		if ( n != j )
			time += 10;
	}
	else
	{
		if ( L >= j )
			time+= (L * 5 ) ;
		else
			time+= (2*n + L ) * 5;
	}
	return time;
}

int main()
{
	int t,T;
	scanf(" %d",&T);
	t=1;
	int n,j,L;
	char U;
	while(t++<=T)
	{
		scanf(" %d %d %d %c",&n, &j, &L, &U);
		//if( n>100 || n<1 ) return -1;
		//if(j>n || j<1) return -1;
		//if( L<0 || L<n+1)return -1
		
		int liftTimeTaken = calcLift( n, j, L, U );
		int stairsTime = j*10;
		if ( liftTimeTaken > stairsTime ) 
		{
			cout << "Stairs " << stairsTime << endl;
		}
		else
		{
			cout << "Lift " << liftTimeTaken << endl;
		}
	} 
	
    	return 0;
}
