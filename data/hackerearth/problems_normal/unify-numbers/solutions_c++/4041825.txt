ll arr[1000005];

ll cal(ll p)
{
    ll ans=0;
    int i;
        f(i,0,n)

            ans+=abs(p-arr[i]);

     return ans;
}
int main()
{
    int t,i;
    ll sm;
    sfd(t);
    while(t--)
    {
        sfd(n);

        ll mx=-1;
        f(i,0,n)
        {
            sflld(arr[i]);
            mx=max(mx,arr[i]);
            sm+=arr[i];
        }
        ll ans=0;
        ll p;
        ll l=1,r=sm;
        while(l<=r)
        {
            ll mid=(l+r)/2;
            ll lm=cal(mid-1),rm=cal(mid+1),m=cal(mid);
            //cout<<lm<<" "<<m<<" "<<rm<<endl;
            if(lm>=m&&m<=rm)
            {
                p=mid;
                break;
            }

            if(lm>m)
                l=mid+1;
            else
                r=mid-1;
