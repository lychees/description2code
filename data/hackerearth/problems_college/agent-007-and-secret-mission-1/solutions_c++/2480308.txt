#include <iostream>
#include <string>
using namespace std;

int main()
{
    int T=0,nDays=0,nCount =0,nActTimeCode=0;
    string S;
    cin>>T;
    for (int i=0; i<T; i++)
    {
    	S.clear();
    	cin>>S;
    	cin>>nDays;
    	nCount =0;
    	nActTimeCode=0;
    	for ( int j=0; j< S.size();j++)
    	{
    		if(S[j] != '0')
    		{
    			nCount++;
    		}
    	}
    	nActTimeCode = (nDays * (nDays +1))/2;
    	if(0 == nActTimeCode%2)
    	{
    		cout<<nCount<<endl;
    	}
    	else
    		cout<<(S.size() - nCount)<<endl;
    }
    
    return 0;
}
