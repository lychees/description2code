#include <iostream>
using namespace std;

int main()
{
	int p,q,x;
	int a[1000050]={0};
	cin>>p;
	for(int i = 0; i < p; i++) {
		cin>>x;
		a[x]++;
	}
	cin>>q;
	for(int i = 0; i < q; i++) {
		cin>>x;
		a[x]--;
	}
	for(int i = 0; i < 1000050; i++) {
		while(a[i] < 0) {
			cout<<i<<" ";
			break;
			a[i]++;
		}
	}
	cout<<endl;
    return 0;
}
