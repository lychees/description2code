 #include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;

int main() {
	int t;
	cin>>t;

int arr[50001];
int count,ans,dis,n;

	while(t--){

            scanf("%d",&n);

            ans=0;count=0;
            memset(arr,0,sizeof arr);

            for(long int i=0;i<n;i++){
                scanf("%d",&dis);

                arr[dis]++;
            }

            int j=0;
            int flag=1;

            for(long int i=1;i<=n;i++){

                while(arr[j]==0)
                    j++;

                arr[j]--;

                if(j-count<=0){
                    flag=0;
                    break;
                }

                if(i%6==0 ){
                    count++;
                }

                ans++;
                count++;

            }

            if(flag) cout<<"Rick now go and save Carl and Judas\n";
            else cout<<"Goodbye Rick\n"<<ans<<endl;
}
	return 0;
}
