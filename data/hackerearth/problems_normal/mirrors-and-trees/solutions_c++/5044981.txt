#include <bits/stdc++.h>
using namespace std;
int arr[200005];
int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	vector<int> lr;
    	cin>>n;
    	for(int i=0;i<n;++i)
    		cin>>arr[i];
    	int m=1,k=0;
    	while(true){
    		bool flag=true;
    		int l=k,r=k;
    		for(int i=0;i<m;++i){
    			if(flag && arr[i+k]!=0){
    				l=i+k;
    				flag=false;
    			}
    			if(arr[i+k]!=0)
    				r=i+k;
    		}	
    		if(arr[l]==0 &&  arr[r]==0)
    			break;
    		if(l!=r)
    			lr.push_back(arr[l]);
    		cout<<arr[r]<<endl;
    		k+=m;
    		m*=2;
    	}
    	for(int i=0;i<lr.size();++i)
    		cout<<lr[i]<<endl;
    	cout<<endl;
    }
    return 0;
}
