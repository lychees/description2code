#include<iostream>
#define MOD 1000000007
using namespace std;
long long BIT[100003];
long long modulo(long long a,long long n)
{
	 if(n==0)
		return 1;
	else if(n%2==0)
		return modulo((a*a)%MOD,n/2)%MOD;
	else return (a*modulo((a*a)%MOD,(n-1)/2))%MOD;
}
long long inv(long long val)
{
	return modulo(val,MOD-2);
}
int main()
{
	long n,a[100004],q,id,v,t;
	cin>>n;
	long long mul=1,zeroes=0;
	for(long i=1;i<=n;i++)
	{
		cin>>a[i];
		if(a[i]>0)
			mul=(mul*a[i])%MOD;
		else 
			zeroes++;
	}

	cin>>q;
	while(q--)
	{
		cin>>t;
		if(t==0)
		{
			cin>>id>>v;
			if(a[id]==0)
				zeroes--;
			else
				mul=(mul*inv(a[id]))%MOD;
			a[id]=v;
			if(v==0)
				zeroes++;
			else
			mul=(mul*v)%MOD;
		}
		else
		{
			cin>>id;
			if(zeroes>1 ||(zeroes==1 && a[id]!=0))
				cout<<"0"<<endl;
			else if(zeroes==1 && a[id]==0)
				cout<<mul<<endl;
			else
			cout<<(mul*inv(a[id]))%MOD<<endl;
		}
	}
	return 0;
}
