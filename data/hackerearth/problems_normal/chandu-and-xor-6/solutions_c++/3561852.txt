#include<iostream>
#include<stdio.h>
#include<map>
#include<string>
#include<set>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
	int n,m,c_sum=0,r_sum=0,max=0,num;
	vector<int> v[1005];
	set<int> g1;
	set<int> g2;
	cin>>n;
	cin>>m;
	for(int i=0;i<n;i++)
	{
		r_sum=0;
		for(int j=0;j<m;j++)
		{
			cin>>num;
			v[i].push_back(num);
			r_sum=r_sum+num;
		}
		
		g1.insert(r_sum);
	}
	for(int i=0;i<m;i++)
	{
		c_sum=0;
		for(int j=0;j<n;j++)
		{
			c_sum=c_sum+v[j][i];
			
		}
		g2.insert(c_sum);
	}
	
	set<int>::const_iterator it1;
	set<int>::const_iterator it2;
	max=0;
	
	for(it1=g1.begin();it1!=g1.end();it1++)
	{
		for(it2=g2.begin();it2!=g2.end();it2++)
		{
			num=*it1 ^ *it2;
			if(num>max)
			{
				max=num;
			}
		}
	}
        cout<<max<<endl;
    
	return 0;
}
