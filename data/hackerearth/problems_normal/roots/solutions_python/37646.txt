def discriminant(a,b,c,d):
	d = (b * b) - (4 * a * c)
	if d >= 0:
		return True
	else:
		return False

import sys
n = sys.stdin.readline()
i = 0
dis = 1
count = 0

while i < int(n):
	coeffs = sys.stdin.readline()
	coeffslist = coeffs.split()
	a,b,c = coeffslist[0],coeffslist[1],coeffslist[2]
	inta, intb, intc = int(a), int(b), int(c)
	if(discriminant(inta,intb,intc,dis)):	
		count += 1
	i += 1	

print count
