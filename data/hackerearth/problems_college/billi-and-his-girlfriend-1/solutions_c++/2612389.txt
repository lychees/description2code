#include<iostream>
#include<math.h>
using namespace std;
long long int power(long long int a,long long int b,long long int k)
{
	if(b==0)return 1;
	else if(b==1)return a;
	else if(b%2==0)
	{
		return power((a*a)%k,b/2,k);
	}
	else
	{
		return((power((a*a)%k,b/2,k)*a)%k);
	}
}
		
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		long long int a,b,k,i,ans=1;
		 
		cin>>a>>b>>k;
		k=pow(10,k);
		a=a%k;
		ans=power(a,b,k);
		long long int m=0;
		while(ans>0)
		{
			m+=ans%10;
			ans/=10;
		}
		cout<<m<<endl;
	}
}
