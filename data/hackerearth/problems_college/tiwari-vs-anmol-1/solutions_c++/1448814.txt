#include<bits/stdc++.h>
using namespace std;
#define ll long long 
#define mod 1000000007
ll a[1000005]={0};
ll b[1000005]={0};
void solve() {
	ll i,j,k;
	for(i=2;i<1000002;i++) {
		if(i%2==0) {
			if(i!=2)
			a[i]=2;
			continue;
		}
		for(j=2;j*i<1000002;j++) {
			if(!a[i*j]) a[i*j]=i;
		}
	}
	for(i=2;i<1000001;i++) b[i]=(b[i-1]+a[i])%mod;
}
int main() {
	memset(a,0,sizeof(a));
	memset(b,0,sizeof(b));
	solve();
	int t,i,l,r;
	scanf("%d",&t);
	while(t--) {
		scanf("%d%d",&l,&r);
		ll ans=b[r]-b[l-1];
		printf("%lld\n",ans);
	}
	return 0;
}
