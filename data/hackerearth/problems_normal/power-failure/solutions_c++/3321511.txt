using namespace std;
#define MOD 1000000007
int isprime[111111];
int numprime[111111];
int a[111111];
int main()
{
	std::ios::sync_with_stdio(false);cin.tie(0);
	int T;
	int i,n,m;
	int t=1,num,tt;
	long long j,re;
	for(i=1;i<=100000;i++)
		isprime[i]=1;
	for(i=2;i<=100000;i++)
	{
		if(isprime[i])
		{
			for(j=(long long)i*i;j<=100000;j+=i)
				isprime[j]=0;
		}
		else
			t++;
		numprime[i]=t;
	}
	numprime[1]=1;
	cin>>T;
	while(T--)
	{
		cin>>n>>m;
		for(i=0;i<n;i++)
			cin>>a[i];
		sort(a,a+n);
		re=1;
		t=numprime[m];
		for(i=n-1;i>=0;i--)
		{
			tt=t-numprime[a[i]-1];
			//cout<<a[i]<<' '<<tt<<endl;
			if(tt<=0)
			{
				re=0;
				break;
			}
			else
			{
				re=re*tt%MOD;
				t--;
			}
		}
		cout<<re<<endl;
	}
	//system("pause");
	return 0;
}
