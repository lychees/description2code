#include <bits/stdc++.h>
using namespace std;

vector< string > all(66000);
 
void cal(int n){
    string x = "";
    int val = n;
    char test;
    while(n>0){
        test = n%2 + '0';
        x = test+x;
        n/=2;
    }
    while(x.size()<16)
        x ='0'+x;
    
    all[val] = x;
}

int main(){
	for(int i = 0;i<65536;i++)
		cal(i);	//generated
	#define LL long long int
	int tt;
	cin>>tt;
	while(tt--){
		int n, x;
		cin>>n;
		std::vector<int> v;
		for(int i = 0;i<n;i++) cin>>x, v.push_back(x);
		priority_queue< pair<LL, int> > pq;
		int pow = 1;
		for(int i = 0;i<n;i++)
			pow*=2;
		for(int ii = 1;ii<pow;ii++){
			int jj = 15, count = 0;
			LL arith2 = 0, AND;
			bool genady = false;
			while(jj>=0){
				if(all[ii][jj]=='1'){
					count++;
					arith2 ^= v[15-jj];
					if(genady==false)
						AND = v[15-jj], genady = true;
					else
						AND&=v[15-jj];
				}
				jj--;
			}
			pq.push(make_pair((arith2|AND), count));
		}
		pair<LL, int> XXX = pq.top();
		cout<<XXX.first<<" "<<XXX.second<<endl;
	}
}
