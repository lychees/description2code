from sys import stdin
import decimal as m
t = int(stdin.readline())
m.getcontext().prec = 20
for i in range(t):
    l, r = map(int, stdin.readline().split())
    lt = m.Decimal(l).sqrt()
    lsq, rsq = int(m.Decimal(l).sqrt()), int(m.Decimal(r).sqrt())
    if lsq**2 < l:
        lsq += 1
    if lsq <= rsq:
        lsq = rsq-lsq+1
    else:
        lsq = 0
    print lsq
