#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
using namespace std;

int main()
{
    vector<string> v;
    vector<string>::iterator it;
    int i;
    cin>>i;
    string str;
    while(i--){
    	cin>>str;
    	v.push_back(str);
    }
    sort(v.begin(), v.end());
    it = unique(v.begin(), v.end());
    v.resize(distance(v.begin(), it));
    for(it = v.begin(); it!= v.end(); it++){
    	cout<<*it<<endl;
    }
//    cout<<*it;
    return 0;
}
