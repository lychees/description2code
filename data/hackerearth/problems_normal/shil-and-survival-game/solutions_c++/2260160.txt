#include <iostream>
#include<algorithm>
using namespace std;

int main()
{	
	int N;
	cin>>N;
	pair<int,int> Ar[100005];
	for(int i=0;i<N;i++)
	{
		cin>>Ar[i].first;
		Ar[i].second=i;
	}
	sort(Ar,Ar+N);
	int st,ed;
	int result[100005],c=0;
	result[c++]=Ar[N-1].second;
	st=Ar[N-1].second,ed=Ar[N-1].second;
	for(int i=N-2;i>=0;i--)	
	{
		if(Ar[i].second<st)result[c++]=st=Ar[i].second;
		if(Ar[i].second>ed)result[c++]=ed=Ar[i].second;
	}
	sort(result,result+c);
	for(int i=0;i<c;i++)
		cout<<result[i]+1<<" ";
	cout<<"\n";
	return 0;
	
}
