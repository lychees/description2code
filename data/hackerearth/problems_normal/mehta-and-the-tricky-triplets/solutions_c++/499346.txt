int bin_search( long long int a[] , long long int num , int len)
{
    int start = 0;
    int end = len-1;
    int mid = (start+end)/2;
    while(start<=end)
    {

        if(num>a[mid])
            start = mid+1;
        else if(num< a[mid])
            end = mid - 1;
        else if(a[mid] == num)
            return mid;
        mid = (start+end)/2;
    }
}
int nbit(ll a, int ctr)
{
    int req;
    while(ctr--)
    {
        req = a%2;
        a/=2;
    }
    return req;
}
ll cal(ll num)
{
    ll tmp = num * (num-1);
    tmp *= (num -2);
    tmp /= 2;
    tmp /=3;
    return tmp;
}
