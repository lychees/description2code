#include <bits/stdc++.h>
using namespace std;

int t, len; string s;

bool check(int x) {
	int j = len-1;
	int p = j-x+1;
	for(int i = x; i <= j; ++i) {
		//cout << s[i] << " " << s[j] << endl;
		if(s[i] != s[j])
			return false;
		--j;	
	}
	return true;
}	

int main()
{
    cin >> t;
    while(t--) {
    	cin >> s;
    	len = s.size();
    	//cout << len << endl;
    	if(check(0))
    		cout << len << '\n';
    	else {
    		for(int i = 1; i < len; ++i) {
    			if(check(i)) {
    				cout << len+i << '\n';
    				//cout << "ggg: " << i << endl;
    				break;
    			}
    		}
    	}	
    }
    return 0;
}
