#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int has[26],has1[26],coun=0,c=0;
        memset(has,0,sizeof(has));
        memset(has1,0,sizeof(has1));
        char s[1010],t[1010];
        cin>>s;
        cin>>t;
        for(int i=0;i<strlen(s);i++)
        {
            has[s[i]-'a']++;
        }
        for(int i=0;i<strlen(t);i++)
        {
            has1[t[i]-'a']++;
        }
        for(int i=0;i<26;i++)
        {
            if(has[i]<has1[i])
                coun++;
            else if(has[i]>has1[i])
                c=1;
        }
        if(c==1)
            cout<<"-1"<<endl;
        else
            cout<<coun<<endl;


    }

    return 0;
}
