#include <iostream>
using namespace std;

int main()
{
    string s;
    int flag = 0;
    cin >> s;
    int l = s.size();
    for(int i=0, j=l-1; i<s.size(); i++, j--)
    {
    	if(s[i] != s[j])
    	{
    		flag = 1;
    		break;
    	}
    }
    if(flag)
    	cout << "NO";
    else
    	cout << "YES";
    return 0;
}
