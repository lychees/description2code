#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	long long int mod = 1e9+7;
	
	while(t--) {
		long long int a[1001];
		int n;
		cin>>n;
		for(int i = 0; i < n; ++i) {
			cin>>a[i];
		}
		sort(a, a+n);
		long long ans = 1;
		
		for(int i = 0; i < n; ++i) {
			ans = (ans * (max(0ll, a[i]-i)))%mod;
		}
		cout<<ans<<"\n";
		
	}
    return 0;
}
