#include <iostream>
#include<string>
#include<algorithm>
using namespace std;
int main() {
	// your code goes here
	string code;
	string domain;
	cin>>domain;
	while(1)
	{
		string code_line;
		cin>>code_line;
		if(code_line.size()==4)
		{
			//find exit 
			if(code_line[0]=='e' && code_line[1]=='x' && code_line[2]=='i' && code_line[3]=='t')
				break;
		}
		code+=code_line;
	}
	//remove space from string
	int i;
	for(i=0;i<code.size();i++)
		if(code[i]==' ')
			code.erase(i,i);
	//convertng into lowercase
	//transform(code.begin(), code.end(), code.begin(), ::tolower);
	//cout<<code;
	//find links
	int pos=0,poss=0;
	string href ("<ahref");
	string End (">");
	while(pos<code.size() && pos>=0)
	{
		pos=code.find(href,pos);
		//cout<<pos<<" ";
		
		if(pos<0)
			break;
		poss=code.find(End,pos+4);
		string new_link;
		bool flag=0;
		for(i=pos+8;i<=poss-2;i++){
			new_link+=code[i];
			flag=1;
		}
		if(flag)
		{
			int j=0;
			bool t=1;
			if(domain.size()<=new_link.size()){
			for(j=0;j<domain.size();j++)
			{
				if(domain[j]!=new_link[j])
				{
					t=0;
					break;
				}
			}
			}
			if(t)
				cout<<new_link<<endl;
		}
		pos=pos+8;
		poss=poss+1;
	}
	return 0;
}
