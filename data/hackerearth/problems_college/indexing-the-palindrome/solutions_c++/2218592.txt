# include <iostream>
# include<string>
using namespace std;
bool palin(string s)
{
    int n=s.length();
    for (int i=0;i<n;i++)
        if (s[i]!=s[n-i-1])
            return 0;
    return 1;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
   {
    string s;
    cin>>s;
    int n=s.length();
    int idx1=0;
    int idx2=0;
    bool p=1;
    for (int i=0;i<n;i++)
    {
        if(s[i]!=s[n-i-1])
        {
            p=0;
            idx1=i;
            idx2=n-i-1;
        }
    }
    if (p==1)
        cout<<-1<<endl;
    else if (idx1==idx2)
        cout<<idx1<<endl;
    else
    {
        string s1=s.substr(0,idx1)+s.substr(idx1+1,n-idx1-1);
        if (palin(s1))
            cout<<idx1<<endl;
        else
            cout<<idx2<<endl;
    }    
    }
}
