#include <bits/stdc++.h>
using namespace std;

typedef long long LL;

const int MAXN = 1e7 + 10;
const LL MOD = 1e9 + 7;
const int LM = 10;

int sm[MAXN];
int dcnt[MAXN];
int ppow[MAXN];

vector<int> pr;
void go(){
	memset(sm, -1, sizeof sm);
	for(int i = 2;i < MAXN;++i){
		if(sm[i] == -1){
			sm[i] = i;
			pr.push_back(i);
		}
		for(int j = 0;j < pr.size() && pr[j] <= sm[i] && i * pr[j] < MAXN;++j){
			sm[i * pr[j]] = pr[j];
		}
	}
	dcnt[1] = 1;
	ppow[1] = 1;
	for(int i = 2;i < MAXN;++i){
		if(sm[i] == sm[i / sm[i]]){
			ppow[i] = sm[i] * ppow[i / sm[i]];
		} else {
			ppow[i] = sm[i];
		}
		if(i == ppow[i]){
			dcnt[i] = dcnt[i / sm[i]] + 1;
		} else {
			dcnt[i] = dcnt[ppow[i]] * dcnt[i / ppow[i]];
		}
	}
	// for(int i = 1;i <= 20;++i){
	// 	cout << i << " " << dcnt[i] << "\n";
	// }
}

int main(){
	go();
	int t;
	scanf("%d", &t);
	while(t--){
		int N;
		scanf("%d", &N);
		if(dcnt[N] < 4){
			printf("NO\n");
		} else {
			printf("YES\n");
		}
	}
	return 0;
}
