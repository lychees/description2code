#include <iostream>
#include<cmath>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    long n;
    cin>>n;
    long ans = 0;
    long j = 0;
    vector<long> balls;
    for(long i =0; i<n; i++)
    {
    	long temp;
    	cin>>temp;
    	balls.push_back(temp);
    }
   	sort(balls.begin(), balls.end());
    for(long i =0; i<n; i++)
    {
    	long temp = balls[i];
    	j++;
    	long sq = pow(j,2);
    	if(temp <= sq)
    		ans += sq - temp;
    	else
    		while(1)
    		{
    			ans += sq;
    			j++;
    			sq = pow(j,2);
    			if(temp <= sq)
    			{
    				ans += sq - temp;
    				break;
    			}
    		}
    }
    cout<<ans;
    return 0;
}
