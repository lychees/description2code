t = int(input())
for i in range(t):
    n = int(input())
    a = list(map(int, raw_input().split()))
    m = int(input())
    t = m % sum(a)
    if t == 0:
        if m == 0:
            print(0)
        else:
            for j in range(len(a)-1, 0, -1):
                if a[j] > 0:
                    print(j+1)
                    break
    else:
        x = 0
        for j in range(len(a)):
            x += a[j]
            if x >= t:
                print(j+1)
                break
