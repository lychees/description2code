#include<iostream>
#include<algorithm>
using namespace std;
int a[100001];
void solve()
{
    int n,m,i,o=0,j,l,r;
    o=0;
    cin>>m>>n;
    for(i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    l=0;
    r=n-1;
    while(l<=r)
    {
        if(a[r]>m)
        {
            r--;
            continue;
        }
        o++;
        if(l==r)break;
        if(a[l]+a[r]<=m)
        {
            l++;
            r--;
        }
        else r--;
    }
    cout<<o<<'\n';
}
int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int t;
    cin>>t;
    while(t--)
        solve();
}
