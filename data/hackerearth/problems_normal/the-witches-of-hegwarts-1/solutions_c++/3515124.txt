#include <iostream>
using namespace std;
int mini[1000]={0};

int Win(int x){
	int c1,c2 = 0,temp;
	if(x<1000 && mini[x]!=0){
		return mini[x];
	}
	if(x==1)
		return 0;
	if(x==2 || x==3)
	    return 1;
	c1 = c1 + x%2;
	c1 = c1 + 1 + Win((x-x%2)/2);
	
	c2 = c2 + x%3;
	c2 = c2 + 1 + Win((x-x%3)/3);
	temp = c1<c2?c1:c2;
	if(x<1000){
		mini[x] = temp;
	}
	
return temp;
	//cout<<count<<"\n";
}

int main()
{
	int *a,n,i;
	string s;
	getline(cin,s);
	n = atoi( s.c_str() );
	a = (int*)malloc(sizeof(int)*n);
	for(i = 0;i<n;i++){
		getline(cin,s);
		a[i] = atoi( s.c_str() );
		cout << Win(a[i])<<"\n";
		//cout <<" "<<a[i]<<" ";
	}
	
	
	
    //cout << "Hello World!" << endl;
    return 0;
}
