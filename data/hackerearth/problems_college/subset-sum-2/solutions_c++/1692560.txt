#include <iostream>
using namespace std;
#include<math.h>

int main()
{
    int t,n;long long int sum;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>n;
    	 sum=0;
    	for(int j=1;j<=n;j++)
    	{
    		sum=sum+(j*pow(2,n-1));
    	}
    	sum=(sum%((int)pow(10,9)+7));
    	cout<<sum<<endl;
    }
    return 0;
}
