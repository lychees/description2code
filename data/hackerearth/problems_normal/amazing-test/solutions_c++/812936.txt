using namespace std;
int arr[100010];
int n;
 bool subset[100010][110];
bool isSubsetSum(int sum)
{
 
      for (int i = 0; i <= n; i++)
      subset[0][i] = true;
 
    for (int i = 1; i <= sum; i++)
      subset[i][0] = false;
 
     for (int i = 1; i <= sum; i++)
     {
       for (int j = 1; j <= n; j++)
       {
         subset[i][j] = subset[i][j-1];
         if (i >= arr[j-1])
           subset[i][j] = subset[i][j] || subset[i - arr[j-1]][j-1];
       }
     }
 
     return subset[sum][n];
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int x,i,j;
		cin>>n>>x;
		long long sum=0;
		for(i=0;i<n;i++)
		{
			cin>>arr[i];
			sum+=arr[i];
		}
		if(x>=sum/2)
		{
			bool flag=isSubsetSum(x);
			if(flag)
			{
				cout<<"YES\n";
			}
			else
			{
				for(i=x-1;i>=sum/2;i--)
					{
						if(subset[i][n]){
						flag=true;break;
						}
					
					}
					if(flag)
					cout<<"YES\n";
					else
				cout<<"NO\n";
			}	
		}
		else
		{
			 cout<<"NO\n";	
		}
	}	
	return 0;
}
