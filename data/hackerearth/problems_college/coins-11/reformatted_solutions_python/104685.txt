import sys
import math

n = input()


def is_prime(n):
    if n == 1:
        return False
    if n % 2 == 0 and n > 2:
        return False
    for i in range(3, int(math.sqrt(n)) + 1, 2):
        if n % i == 0:
            return False
    return True


for i in range(n):
    temp = input()
    if temp == 3:
        print(3)
    elif is_prime(temp):
        print(2)
    else:
        print(1)
