#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int solve(int);
int main() {
   int n,m;vector<int> v;vector<int> v2;
   cin>>n;
   for(int i=0;i<n;i++){
   	cin>>m;
   	v.push_back(m);
   }
   int j = 0;
   while(j<n){
    int s = solve(v[j]);
     v2.push_back(s);
     j++;  
    }
    for(int i=0;i<v2.size();i++){
    	cout<<v2[i]*v2[i]<<'\n';
	}
   }
int solve(int v){
	if(v==0)return 1;
	if(v==1)return 2;
	else{
		return solve(v-1)+solve(v-2);
	}
}
