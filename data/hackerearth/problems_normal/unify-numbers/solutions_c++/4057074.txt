#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	long long a[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	sort(a,a+n);
    	long long cost=0;
    	long long val;
    	if(n%2==0)
    	{
    		val=(a[n/2]+a[n/2-1])/2;
    	}
    	else
    	{
    		val=a[n/2];
    	}
    	
    	for(int i=0;i<n;i++)
    	{
    		cost+=abs(val-a[i]);
    	}
    	cout<<cost<<endl;
    }
    return 0;
}
