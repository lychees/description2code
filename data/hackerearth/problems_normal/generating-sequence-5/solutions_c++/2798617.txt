#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n;
    long long g;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>g>>n;
    	for(int j=0;j<n-1;j++)
    	{
    		cout<<g*2<<" ";
    	}
    	cout<<g*3;
    	cout<<endl;
    }
    return 0;
}
