#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    long long n,res,ct=0;
    cin>>n;
     vector <int> v;
     for(long long i=0;i<n;i++)
     {
     	cin>>res;
     	v.push_back(res);
     }
     std::sort(v.begin(),v.end());
     for(long long i=0;i<n;)
     {
     	long long res=v[i];
     	while(v[i]<=res+4&&i<n)
     	  i++;
     	  ct++;
     	  
     }
     
     cout<<"\n"<<ct;
     
}
