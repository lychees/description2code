#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int c;
    cin>>c;
    int a[n][n],k=0,p[(n*n)];
    memset(a,0,n*(sizeof(int)));
    while(c--)
    {
    	int x,y;
    	cin>>x;
    	p[k++]=x;
    	cin>>y;
    	a[x][y]=1;
    	for(int i=0;i<k;i++)
    	{
    		if(a[i][y]!=1)
    		a[i][y]=-1;
    	}
    }
    int q;
    cin>>q;
    while(q--)
    {
    	int xx,yy;
    	cin>>xx>>yy;
    	if(a[xx][yy]==1)
    	cout<<"Direct Connection"<<endl;
    	else
    	if(a[xx][yy]==-1)
    	cout<<"Indirect Connection"<<endl;
    	else
    	cout<<"No Connection"<<endl;
    }
    return 0;
}
