#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   string a,b;
   for(;t--;)
   {
   	cin>>a>>b;
   	int i,count=0,l;
   	l=a.length();
   	for(i=l-1;i>=0;i--)
   	{
   		if((a[i]-48)+(b[i]-48)>=10)
   		{
   			count++;
   			a[i+1]+=1;
   		}
   	}
   	if(count==0)
   	cout<<("No carry operation\n");
   	else if(count==1)
   	cout<<("1 carry operation\n");
   	else 
   	cout<<count<<(" carry operations\n");
   }
    return 0;
}
