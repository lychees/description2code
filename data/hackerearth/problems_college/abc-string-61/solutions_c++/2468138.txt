#include <iostream>
using namespace std;
long long int mp(long long int a, long long int b, long long int m)
{
	long long int ans=1,cur=a;
	while(b>0)
	{
		if(b%2==1)
			ans=(ans*cur) % m;
		cur=(cur*cur) % m;
		b/=2;
	}
	return ans%m;

}
int main()
{
    int t;
    cin >> t;
    long long int mod=1000000007;
    while(t--)
	{
		long long int n,ans;
		cin >> n;
		if(n<3)
		{
			cout << "0" << endl;
			continue;
		}
		long long int val = mp(3,n-1,mod) - mp(2,n,mod);
		if (val<0)
		{
			val+=mod;
			val%=mod;
		}
		ans=(3*((val) %mod +1 ) % mod) %mod;
		cout << ans <<endl;
		
	}

	return 0;
}
