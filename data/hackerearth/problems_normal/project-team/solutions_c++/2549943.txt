#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,a1;
	vector<int> a;
	cin >> t;
	while(t--){
		cin >> n;
		a.clear();
		for(int i=0;i<n;i++){
			cin >> a1;
			a.push_back(a1);
		}
		sort(a.begin(),a.end());
		int i=0,j=a.size()-1,m=INT_MAX,M=INT_MIN,s;
		while(i<j){
			s=a[i]+a[j];
			m=min(m,s);
			M=max(M,s);
			i++;j--;
		}
		cout << M-m << endl;
	}
    return 0;
}
