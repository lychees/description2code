#include <iostream>
using namespace std;
int arr[1001];
int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    for(int i=0;i<n;i++){
    	cin>>arr[i];
    }
    int fr=0;
    
    while(fr<n&&arr[fr]!=1)
    	fr++;
    if(fr==n)
    {
    	cout<<"0"<<endl;
    	return 0;
    }
    long long sm=1;
    int prev=fr;
    fr++;
    while(fr<n){
    	if(arr[fr]==1){
    		sm*=(fr-prev);
    		prev=fr;
    	}
    	fr++;
    }
    cout<<sm<<endl;
    return 0;
}
