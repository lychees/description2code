#include <iostream>
using namespace std;

int main()
{
   int n,home=0,away=0;
   int *a=NULL,*b=NULL;
   cin>>n;
   a=new int[n];
   b=new int[n];
   for(int i=0;i<n;i++)
   {
   	 cin>>a[i]>>b[i];
   }
   for(int i=0;i<n;i++)
   {
   	home=n-1;
   	for(int j=0;j<n;j++)
   	{ if(i!=j){
   		 if(b[i]!=a[j])away++;
   		 else home++;
   	}
   
   	}
   		cout<<home<<' '<<away<<'\n';
   		home=away=0;
   }
   return 0;
}
