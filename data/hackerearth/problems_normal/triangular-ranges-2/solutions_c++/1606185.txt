#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

vector <long long>arr;
int main()
{
    int t;
    cin>>t;
    	long long  i,p,x;
    	p=1;
    	x=0;
    	
    	while(1)
    	{
    		x=x+p;
    		if(x>1000000000000LL)break;
    		arr.push_back(x);
    		p++;
    	}
    	
    while(t--)
    {
    	long long l,r,lb,rb;
    	cin>>l>>r;
    	
    	long long ans=0;
    	
    	for(i=0;i<arr.size();i++)
    	{
    		lb=lower_bound(arr.begin()+i,arr.end(),l-arr[i])-arr.begin();
    		rb=lower_bound(arr.begin()+i,arr.end(),r-arr[i])-arr.begin();
    		
    		if(arr[rb]!=r-arr[i])rb--;
    		
    		if((arr[lb]+arr[i])>=l && arr[rb]+arr[i]<=r && lb<=rb)
    		    ans=ans+(rb-lb+1);
    	}
    	cout<<ans<<endl;
    }
}
