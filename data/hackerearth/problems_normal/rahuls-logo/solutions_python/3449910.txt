def logo(n):
    start = 1
    logo = []
    for i in range(n):
        logo.append( (bin(start)[2:].replace('1','/')+
                  bin(start)[2:][::-1].replace('1','\\')).center(n*2).replace('0',' '))
        start = (start << 1) + i % 2
    logo =  logo+map('  {}'.format, map(''.join,zip(*zip(*logo)[::-1]))[::-1])
    return '\n'.join(map(str.rstrip, logo))
N = int(raw_input())
print logo(N)
print logo(N+1)
