#include <bits/stdc++.h>
using namespace std;
long func(long x)
{
 long sum=0,rem;
 while(x>0)
 {
  rem=x%10;
   sum=sum+pow(rem,2);
   x=x/10;
 }
if(sum>=10)
func(sum);
else
return sum;
}
int main() {
	long a;int t,i=0,b[1005],j;
	cin>>t;
   	while(cin>>a)
  {	if(func(a)==1){b[i]=1;i++;}
     else {b[i]=0;i++;}
  }
  i=1;
  for(j=0;j<t;j++)
  {if(b[j]==1){cout<<"Case #"<<i<<":YES"<<"\n";i++;}
	  else {cout<<"Case #"<<i<<":NO"<<"\n";i++;}
  }
  
	return 0;
}
