#include<iostream>
#include<algorithm>
using namespace std;
int hcf(int a, int h) 
{
    int temp; 
    while(1)
    {
        temp = a%h;
        if(temp==0)
        return h;
        a = h;
        h = temp;
    }
} 
int main()
{
	long long int t,a[3];
	cin>>t;
	while(t--)
	{
		int bol=1;
		cin>>a[0]>>a[1]>>a[2];
		sort(a,a+3);
		if(a[0]*a[0]+a[1]*a[1]==a[2]*a[2])
		{
			if(hcf(a[2],a[1])==1 && hcf(a[2],a[0])==1 && hcf(a[1],a[0])==1 )
			{
				bol=0;
				cout<<"YES"<<endl;
			}
		}
		if(bol)
			cout<<"NO"<<endl;
	}
}
