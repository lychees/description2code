#include <iostream>
#include<math.h>
using namespace std;
int main()
{
    long long int a,b,m,n,x,test,flag;
    cin>>test;
    while(test--)
    {	
    	cin>>x;
    	flag=0;
    	for(m=2;m<200;m++)
    	{
    	 for(a=1;a<1500;a++)
    	  {
    		if(pow(a,m)>x)
    			break;
    		for(n=2;n<200;n++)
        	   {
    			for(b=1;b<1500;b++)
    			 {
    				if(pow(b,n)>x)
    					break;
    				if(x==(pow(a,m)+pow(b,n)))
   						{
   							flag=1;
   							break;
   						}
    			  }
    				if(flag==1)
    					break;
    			}
    	            if(flag==1)
    					break;
    		}
    		   	    if(flag==1)
    					break;
	   	 }
	   	  if(flag==1)
	   	  	cout<<"Yes\n";
	   	  else
	   	  	cout<<"No\n";
    }
}
