T = int(raw_input())
W = []
R = []
for i in xrange(2*T):
	if i%2:
		R.append(''.join(sorted(raw_input())))
	else:
		W.append(raw_input())
for i in xrange(T):
	s = list(W[i])
	k=0
	for j in xrange(len(W[i])):
		if W[i][j]>R[i][k]:
			s[j] = R[i][k]
			k = k+1
		if k==len(R[i]):
			break
	W[i] = ''.join(s)
	print W[i]
