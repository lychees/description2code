#include<stdlib.h>
#include<iostream>

using namespace std;

int main()
{
	unsigned int n,l,r,q;
	int a[1000005],sum[1000005];
	cin>>n>>q;
	a[0]=sum[0]=0;
	for(l=1;l<=n;l++)
	{
		cin>>a[l];
		sum[l]=a[l]+sum[l-1];
	}
	while(q--)
	{
		cin>>l>>r;
		cout<<sum[r]-sum[l-1]<<" "<<(a[r]-sum[r-1]+sum[l-1])<<endl;
	}
	return 0;
}
