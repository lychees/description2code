#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,a,s=0;
	cin>>n;
	vector<int> v;
	for(int i=0;i<n;i++)
	{
		cin>>a;
		v.push_back(a);
	}
	int dp[n];
//	for(int i=0;i<n;i++)
	dp[0]=1;
	for(int i=1;i<n;i++)
	{
		if(v[i]>v[i-1])
		{
			dp[i]=dp[i-1]+1;
		}
		else
		dp[i]=1;
	}
	for(int i=n-2;i>=0;i--)
	{
		if(v[i]>v[i+1]&&dp[i]<=dp[i+1])
		{
			dp[i]=dp[i+1]+1;
		}
	}
//	int s=0;
	for(int i=0;i<n;i++)
	{
		s=s+dp[i];
	}
	cout<<s<<"\n";
	
	return 0;
}
