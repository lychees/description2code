#include <bits/stdc++.h>

using namespace std;

int arr[100][100];
int DP[100][100];

int main(int argc, char const *argv[]) {
	int t;
	int n, m;
	cin >> t;	
	while(t --) {
		cin >> n >> m;
		for(int i = 0; i < n; i ++) {
			for(int j = 0; j < m; j ++) {
				cin >> arr[i][j];
				DP[i][j] = 0;
			}
		}
		DP[0][0] = 1;
		for(int i = 1; i < m; i ++) {
			if(arr[0][i] > arr[0][i - 1]) {
				DP[0][i] = DP[0][i - 1] + 1;	
			} else {
				DP[0][i] = 0;
				break;
			}
		}
		for(int i = 1; i < n; i ++) {
			if(arr[i][0] > arr[i - 1][0]) {
				DP[i][0] = DP[i - 1][0] + 1;	
			} else {
				DP[i][0] = 0;
				break;
			}
		}
		for(int i = 1; i < n; i ++) {
			for(int j = 1; j < m; j ++) {
				if(DP[i - 1][j] != 0 && arr[i][j] > arr[i - 1][j]) {
					DP[i][j] = DP[i - 1][j] + 1;
				} else {
					DP[i][j] = 0;
				}
				if(DP[i][j - 1] != 0 && arr[i][j] > arr[i][j - 1] && DP[i][j - 1] + 1 > DP[i][j]) {
					DP[i][j] = DP[i][j - 1] + 1;
				}
			}
		}/*
		for(int i = 0; i < n; i ++) {
			for(int j = 0; j < m; j ++) {
				cout << DP[i][j];
			}
			cout << endl;
		}*/
		int max = INT_MIN;
		for(int i = 0; i < n; i ++) {
			for(int j = 0; j < m; j ++) {
				if(DP[i][j] > max) {
					max = DP[i][j];
				}
			}
		}
		cout << max << endl;
	}
	return 0;
}
