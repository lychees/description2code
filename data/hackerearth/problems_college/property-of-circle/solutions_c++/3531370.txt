#include<iostream>
#include<math.h>
using namespace std;

int main()
{
    int t,x1,x2,y1,y2,r1,r2;float c;
    cin>>t;
    while(t--)
    {
    	cin>>x1>>y1>>r1;
        cin>>x2>>y2>>r2;
    	c=sqrt((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1));
    	
    	
    	if(r1==r2&&c==0)
    	cout<<"C2~C1\n";
    	
       else	if(c<=r2-r1)
    		    cout<<"C2CC1\n";
      	
    
    	else if(c<=r1-r2)
    	 	     cout<<"C1CC2\n";
       	
        
        else
        cout<<"NOT SATISFIED\n";
    }
    return 0;
}
