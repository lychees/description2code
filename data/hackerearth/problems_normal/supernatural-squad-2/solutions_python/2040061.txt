dp = [[0 for c in range(201)] for r in range(201)]
def recursion(start, N):
	if N == 0: return 1
	if N < 0: return 0
	if dp[start][N]: return dp[start][N]
	
	count = 0
	for i in range(start, N+1):
		dp[i][N - i] = recursion(i, N - i)
		count += dp[i][N - i]
		
	return count

for tc in range(input()):
	N, start = map(int, raw_input().split())
	print recursion(start, N)
