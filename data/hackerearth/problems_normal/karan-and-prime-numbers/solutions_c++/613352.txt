#include <bits/stdc++.h>
int seive[1000001]= {0},i,j,T,a,b;
    long long int sum=0;
int main()
{

    seive[1]=1;
    seive[0]=1;
    scanf("%d",&T);
    while(T--)
    {
        sum=0;
        scanf("%d %d",&a,&b);
        b++;
        for(i=2; i<b; i++)
        {
            if((i*i)>b)
                break;
            for(j=i*i; j<b; j+=i)
                seive[j]=1;
        }
        for(j=a; j<b; j++)
            if(!seive[j])
                sum+=j;

        printf("%lld\n",sum);
    }

        //printf("%d\n",j);
        return 0;

}
