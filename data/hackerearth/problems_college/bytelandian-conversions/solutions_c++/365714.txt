#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;

int main()
{
	char bin[32],tri[20];
	int pow2[32],pow3[20],l2,l3,n1,n2,t2,t3;
	pow2[0]=1;
	pow3[0]=1;
	
	scanf("%s",bin);
	scanf("%s",tri);
	l2 = strlen(bin);
	l3 = strlen(tri);
	
	n1 = 0;n2=0;
	for(int i=l2-1;i>=0;i--)
	{
		bin[i] -=48;
		n1 += pow2[l2-i-1]*(bin[i]);
		pow2[l2-i] = 2*pow2[l2-i-1];
	}
	for(int i=l3-1;i>=0;i--)
	{
		tri[i] -=48;
		n2 += pow3[l3-i-1]*(tri[i]);
		pow3[l3-i] = 3*pow3[l3-i-1];
	}
	for(int i=l2-1;i>=0;i--)
	{
		t2 = n1-pow2[l2-i-1]*bin[i] + pow2[l2-i-1]*((bin[i]+1)%2);
		for(int j = l3-1;j>=0;j--)
		{
			t3 = n2-pow3[l3-j-1]*tri[j] + pow3[l3-j-1]*((tri[j]+1)%3);
			if(t2==t3)
			{
				printf("%d",t3);
				return 0;
			}
			
			t3 = n2-pow3[l3-j-1]*tri[j] + pow3[l3-j-1]*((tri[j]+2)%3);
			if(t2==t3)
			{
				printf("%d",t3);
				return 0;
			}
		}
	}
    return 0;
}
