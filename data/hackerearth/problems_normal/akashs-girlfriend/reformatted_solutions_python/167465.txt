def func(a, b, c):
    x = 1
    y = a
    while (b > 0):
        if (b % 2 == 1):
            x = (x * y) % c
        y = (y * y) % c
        b = b/2
    return x % c


MOD = 1000000009
t = input()
for _ in range(t):
    n = input()
    n -= 1
    ans = ((5*func(10, n, MOD)) % MOD-(4*func(8, n, MOD)) % MOD) % MOD
    print ans
