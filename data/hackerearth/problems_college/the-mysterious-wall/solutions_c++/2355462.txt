#include<bits/stdc++.h>

#define ll long long int

using namespace std;

char s1[100005],s2[100005];
int l1,l2;
int freq1[26] , freq2[26];


int main() {

    int t;
    scanf("%d",&t);

    while(t--) {

        scanf("%s",s1);
        scanf("%s",s2);

        l1=strlen(s1);
        l2=strlen(s2);

        if(l2>l1) {
            printf("NO\n");
            continue;
        }


        /* RESET FREQUENCY TO ZERO */
        for(int i=0;i<26;i++) {
            freq1[i]=0;
        }

        for(int i=0;i<26;i++) {
            freq2[i]=0;
        }

        /* count character frequency for S1 and S2 */
        for(int i=0;i<l2;i++) {
            freq1[ s1[i]-'a' ] ++;
        }

        for(int i=0;i<l2;i++) {
            freq2[ s2[i]- 'a' ] ++;
        }

        bool matched=1;
        for(int i=0;i + l2 <= l1;i++) {
            matched=1;
            for(int j=0;j<26;j++) {
                if(freq1[j]!=freq2[j]) {
                    matched=0;
                    break;
                }
            }

    		if(i+l2<l1)
            	freq1[ s1[i+l2] - 'a'] ++;
            freq1[ s1[i] - 'a' ] --;

            if(matched) {
                printf("YES\n");
                break;
            }
        }

        if(!matched)
            printf("NO\n");

    }

    return 0;
}
