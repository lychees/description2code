from operator import add
for i in xrange(input()):
	a,k = map(int,raw_input().split())
	store  = map(int,raw_input().split())
	store.sort()
	a=a/2
	mini,miny =store[0:a],store[a:2*a]
	miny.reverse()
	final = map(add,mini,miny)
	check = max(final)-min(final)
	print check
	if check>k:
		print "No more girlfriends!"
	elif check==k:
		print "Lucky chap!"
	else:
		print "Chick magnet Jhool!"
