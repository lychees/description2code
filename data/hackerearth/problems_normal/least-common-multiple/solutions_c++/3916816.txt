vector<int> _prime_sieve(int n)
{
   vector<int> res;
   vector<char> prime(n+1, true);
   prime[0] = prime[1] = false;
   res.push_back(2);
   for(int i = 3; i * 1ll * i <= n; i += 2)
   {
      if(prime[i])
      {
         for(int j = i * i; j <= n; j += i) prime[j] = false;
      }
   }
   for(int i = 3; i <= n; i += 2)
   {
      if(prime[i]) res.push_back(i);
   }
   return res;
}
int main()
{
   //freopen("in.txt", "r", stdin);
   ll n;
   scanf("%lld", &n);
   vector<int> p = _prime_sieve(sqrt(double(n)));
   ll a = n / 2 + 1;
   while(true)
   {
	  ll a2 = a;
	  int b2 = -1;
	  bool b3 = false;
	  for(int i = 0; i < p.size(); i++)
	  {
		  while((a2 % p[i]) == 0)
		  {
			  if(b2 != -1 && b2 != p[i])
			  {
				  b3 = true;
				  break;
			  }
			  a2 /= p[i];
			  b2 = p[i];
		  }
		  if(b3) break;
	  }
	  if(!b3 && (b2 == -1 || a2 == 1))
	  {
		  printf("%lld\n", a);
		  break;
	  }
	  a++;
   }
   return 0;
}
