#include <bits/stdc++.h>
using namespace std;
char ans[25][25];
int main()
{
	string s;
	int n,i,t,j;
	scanf("%d\n",&t);
	while(t--)
	{
		getline(cin,s);
		n=s.size();
		int r,c,tmp=0;
		r=sqrt(n);
		c=-1;
		for(i=0;i<n;i++)
		{
			c++;
			ans[tmp][c]=s[i];
			if(c==r-1)
			c=-1,tmp++;
		}
		for(i=0;i<r;i++)
		for(j=0;j<r;j++)
		cout<<ans[j][i];
		cout<<endl;
	}
    return 0;
}
