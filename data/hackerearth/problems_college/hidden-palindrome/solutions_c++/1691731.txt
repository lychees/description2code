#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long t;
    cin>>t;
    while(t--)
    {
    	string str,str2,str3;
    	int max=0,i,j;
    	cin>>str;
    	for(i=0;i<str.size();i++)
    	{
    		for(j=str.size()-i;j>=0;j--)
    		{
    		str2=str.substr(i,j);
    		str3=str2;
    		std::reverse(str2.begin(),str2.end());
    		if(str3.compare(str2)==NULL)
    		{
    		if(max<str2.size())
    		max=str2.size();
    		}
    	}
    }
    cout<<max<<endl;
    }
    return 0;
}
