#include <iostream>
#include <vector>
using namespace std;

int main()
{	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		vector <int> count;
		count.push_back(-1);
		int n=s.length();
		for(int i=0;i<s.length();i++)
		{
			if(s[i]=='*')
			{
				count.push_back(i);
			}
		}
		count.push_back(n);
		int j=0;
		int c=0;
		int check;
		while(count[j]!=n)
		{
			check=0;
		for(int i=(count[j]+1);i<count[j+1];i++)
		{	
				if(s[i]=='X')
				{
					check=1;
				}
		}
		if(check==0)
		{
			c=c+count[j+1]-count[j]-1;
		}
			j++;
		}
		cout<<c<<endl;
		count.clear();
	}
    
    return 0;
}
