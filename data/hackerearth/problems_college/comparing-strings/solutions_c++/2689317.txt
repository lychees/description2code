#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char a[100000] , b[100000];
    	cin>>a>>b;
    	int flag = 0 , arr[200] = {0};
    	for(int i = 0 ; a[i] != '\0' ; i++)
    	{
    		arr[(int)a[i]] = 1;
    	}
    	for(int i = 0 ; b[i] != '\0' ; i++)
    	{
    		if(arr[(int)b[i]] == 1)
    		{
    			flag = 1;
    			break;
    		}
    	}
    	if(flag == 1)
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    return 0;
}
