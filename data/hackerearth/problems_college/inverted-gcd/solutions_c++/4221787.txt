#include <iostream>
using namespace std;
#include <vector>
#define ll long long

ll cop(ll u,ll v)
{   if(((u|v)&1)==0) 
        return 0;
    while((u&1)==0)
        u>>=1;
    if(u==1) 
        return 1;
    do
    {   while((v&1)==0) 
            v>>=1;
        if(v==1) 
            return 1;
        if(u>v) 
        {   ll t=v; 
            v=u; 
            u=t; 
        }
        v-=u;
    }while(v!=0);
    return 0;
}

int main()
{   ll n,q,i,j,s=0;
    scanf("%lld",&n);
    ll z[n];
    for(i=0;i<n;++i)
    	scanf("%lld",&z[i]);
    for(i=0;i<n;++i)
        for(j=i+1;j<n;++j)
        {   if(z[i]>z[j])
            {   if(cop(z[i],z[j]))
                    s++;
            }
        }
    printf("%lld\n",s);
    return 0;
}
