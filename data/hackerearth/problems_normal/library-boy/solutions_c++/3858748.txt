#include <iostream>
#include <string>
using namespace std;

bool is_lower_alphanumeric(const string& txt)
{
  for(char c : txt)
  {
    if (!(c >= 'a' and c <= 'z'))
    return false;
  }
  return true;
}

int main()
{
    int n;
    cin>>n;
    
    int i,ans=0,temp,temp1;
    char ch;
    int count[26]={0};
    
    string str;
    
    for(i=0;i<n;i++)
    {
    	cin>>str;
    	if(str.length()>=3 && str.length()<=20 && is_lower_alphanumeric(str))
    	{
    	ch=str.at(0);
    	count[ch-97]=count[ch-97]+1;
    	}
    
    	
    }
    
    for(i=0;i<26;i++)
    {
    	if(count[i]!=0)
    	{
         	if(count[i]>10)
         	{
    		temp=count[i]/10;
    		temp1=count[i]%10;
    		
    		if(temp1!=0)
    		ans=ans+temp+1;
    		else
    		ans=ans+temp;
         	}
    	    else
    	    ans=ans+1;
    	}
   
    }
     	cout<<ans;
    return 0;
}
