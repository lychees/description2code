#include<iostream>
using namespace std;


long day(long n , int l , int r)
{
	
	long mid,limit1,limit2;mid=(l+r)/2;
	
	limit1=mid*(mid+1)/2 -mid+1;
	limit2=mid*(mid+1)/2;
	
	if(limit1<=n && limit2>=n)
	return mid;
	
	if(limit1<n && limit2<n)
	return day(n,mid,r);
	
	if(limit1>n && limit2>n)
	return day(n,l,mid);
	
	
}


int main()
{
	
	int t;
	cin>>t;
	
	long n,temp;
	while(t--)
	{
	cin>>n;
	if(n==2)
	cout<<"BS"<<endl;
	else if(n==1)
	cout<<"LB"<<endl;
	else
	{
		temp=day(n,0,n);
		
		if(temp%3==0)
		cout<<"SL"<<endl;
		else if(temp%3==1)
		cout<<"LB"<<endl;
		else
		cout<<"BS"<<endl;
	}
	}
}
