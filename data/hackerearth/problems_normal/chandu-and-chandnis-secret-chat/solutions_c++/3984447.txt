#include <bits/stdc++.h>
using namespace std;

string s;

bool comp(int i,int j){
	return (s[i] != s[j]) ? (s[i] > s[j]) : (i < j);
}

int main()
{
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>s>>n;
		n--;
		int l = s.length();
		int arr[l];
		for(int i=0;i<l;i++)
			arr[i] = i;
		sort(arr,arr+l,comp);
		for(int i=0;i<l;i++,n = arr[n]){
			 cout<<s[n];
		}
		cout<<"\n";
	}
    return 0;
}
