#include<bits/stdc++.h>
using namespace std;

int main() {
  //freopen("a.cpp","r",stdin);
  int tt;
  cin>>tt;
  int cases=1;
  while(tt--) {
    cout<<"Case #"<<cases++<<": ";
    int n,p,u,r,s;
    cin>>n>>p>>u>>r>>s;
    while(true) {
      if(n<=0) {
        break;
      }
      if(p+r<=u) {
        p+=r;
      } else if(p-s>=0) {
        p-=s;
      } else {
        break;
      }
      n--;
    }
    cout<<p<<endl;
  }
  return 0;
}
