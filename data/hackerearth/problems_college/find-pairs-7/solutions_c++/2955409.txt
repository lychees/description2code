#include <iostream>
using namespace std;
#define ll long long
int h[2000002];
int main(){
	//freopen("in","r",stdin);
	int t; cin>>t;
	while(t--){
		int n; cin>>n;
		int x;
		ll res=0;
		for(int i=0;i<2000002;i++){
			h[i]=0;
		}
		for(int i=0;i<n;i++){
			scanf("%d",&x);
			h[x+1000000]++;
		}
		for(int i=0;i<2000002;i++){
			res += h[i]*1ll*(h[i]+1)/2ll;
		}
		cout<<res<<endl;
	}
	return 0;
}
