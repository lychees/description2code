#include <iostream>
#include <string>
using namespace std;

int main()
{
    string s;
    int l;
    cin >> s;
    l=s.size();
    for(int i=0;i<l;i++) {
    	if(isupper(s[i])) {
    		s[i]=s[i]+32;
    	}
    	else {
    		s[i] = s[i]-32;
    	}
    }
    for(int j=0;j<l;j++) {
    	cout << s[j];
    }
}
