T = int(raw_input())
if T in range(1, 101):
    for cases in range(T):
        string = raw_input()
        counts = {}
        for letter in string:
            counts[letter] = counts.get(letter, 0) + 1
        new_dict = {}
        for k, v in counts.iteritems():
            new_dict.setdefault(v, []).append(k)
        newList = []
        for key in sorted(new_dict):
            for val in sorted(new_dict[key]):
                for i in range(key):
                    newList.append(val)
        print ''.join(val for val in newList)
