def main():
    string = raw_input()
    dict_count = input()
    user_dict = list()
    res = None
    for each in xrange(dict_count):
        user_dict.append(raw_input())
    for each in sorted(user_dict):
        if each.startswith(string):
            res = each
            break
    print res if res else string


if __name__ == '__main__':
    main()
