#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
    	int n;
    	cin>>n;
    	
    	pair<int,int> pr[n];
    	int a,b;
    	int dp[n];
    	
    	for(int i=0;i<n;i++)
    	{
    		cin>>a>>b;
    		pr[i]=make_pair(a,b);
    		dp[i]=1;
    	}
    	sort(pr,pr+n);
    	int maxi=1;
    	for(int i=1;i<n;i++)
    	{
    		for(int j=i-1;j>=0;j--)
    		{
    			if(pr[i].second>pr[j].second)
    			{
    				dp[i]=max(dp[i],dp[j]+1);
    			}
    		    maxi=max(dp[i],maxi);
    		}
    	}
    	cout<<maxi<<endl;
    	
    }
}
