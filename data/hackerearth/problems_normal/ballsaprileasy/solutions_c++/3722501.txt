#include <bits/stdc++.h>
using namespace std;
const int MAXN = 1000006;
int A[MAXN];
int main()
{
	int n;
	cin>>n;
	for (int i = 0; i < n; ++i)
		cin>>A[i];
	sort(A, A+n);
	long long int ans = 0, layer = 1, pos = 0;
	while(pos < n)
	{
		if(layer*layer < A[pos])
			ans+=layer*layer;
		else
		{
			ans+=(layer*layer)-A[pos];
			pos++;
		}
		layer++;
	}
	cout<<ans<<"\n";
	return 0;
}
