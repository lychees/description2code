#include<bits/stdc++.h>
using namespace std;
int main() {
	int t;
	cin>>t;
	while(t--) {
		int a,b;
		long long c;
		scanf("%d%d%lld",&a,&b,&c);
		int cc=0;
		while(c>1) {
			if(cc&1)
				c=ceil(1.0*c/a);
			else
				c=ceil(1.0*c/b);
			cc++;
		}
		if(!(cc&1))
			printf("%s\n","Bob");
		else
			printf("%s\n","Adam");
	}
	return 0;
}
