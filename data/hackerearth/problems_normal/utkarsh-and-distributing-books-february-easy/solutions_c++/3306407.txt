#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int testCase;
    cin>>testCase;
    while(testCase--) {
    	int numberOfPiles;
    	cin>>numberOfPiles;
    	int ans1=1010,ans2=0;
    	int books[numberOfPiles];
    	for(int iterator = 0 ; iterator < numberOfPiles; iterator++) {
    		cin>>books[iterator];
    		ans2 = ans2 + books[iterator] - 1;
    		ans1 = min(ans1,books[iterator]);
    	}
    	cout<<ans1 - 1<<" "<<ans2<<endl;
    	
    	
    }
    return 0;
}
