#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;

void func(char a){
	switch(a){
		case '2': cout<<"cde";break;
		case '7': cout<<"acf";break;
		case '6': cout<<"b3";break;
		case '9': cout<<"c6a";break;
	}
}

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s;
    	cin>>s;
    	for(int i=0;i<s.length();i++){
    		func(s[i]);
    		
    	}
    	cout<<endl;
    }
    return 0;
}
