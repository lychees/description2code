# print t
while t > 0:
    t = t - 1
    s = raw_input()
    n, k = s.split(" ")
    # print "n is %r and k is %r" % (n,k)
    n = int(n)
    k = int(k)
    a = []
    for i in range(0, n):
        a.append(i)
    i = 0
    j = n
    while (j != 1):
        i = (i+k) % j
        a.remove(a[i])
        j = j - 1
        if (i == j):
            i = 0
    print a[0]+1
