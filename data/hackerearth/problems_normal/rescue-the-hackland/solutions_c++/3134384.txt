#include <iostream>
using namespace std;

int main()
{
	int test;
		long long result = 1;
	cin>>test;
	for(int i=0;i<test;i++) {
            result=1;
		long long left,right,k;
		cin>>left>>right>>k;
		while((left*k + 1) <=right) {
      
		result=result+1;
			left = (left*k)+1;
        }
		
		cout<<result<<endl;
	}
}
