#include <iostream>
#include <cstdio>
using namespace std;

int main() {
	int t,n;
	long long int arr[50000];
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		for(int i=0;i<n;i++){
			scanf("%lld",arr+i);
		}
		long long int profit = 0;
		int sno = 0;
		long long int spent = 0;
		for(int i=0;i<n-1;i++){
			if(arr[i] <= arr[i+1]){
				sno++;
				spent+=arr[i];
			}else{
				int j = i+1;
				while(j<n){
					if(arr[j]>=arr[i])break;
					j++;
				}
				if(j==n){
					profit+=(sno*arr[i])-spent;
					sno=0;
					spent=0;
				}else{
					sno++;
					spent+=arr[i];
				}
			}
		}
		if(sno>0){
			profit+=(sno*arr[n-1])-spent;
		}
		cout << profit << endl;
	}
	
	// your code goes here
	return 0;
}
