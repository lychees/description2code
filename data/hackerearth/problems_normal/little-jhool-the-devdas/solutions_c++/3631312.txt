#include<iostream>
using namespace std;
int main()
{
	int n,p,j=0,k=0,ref;
	unsigned long long int s=0;
	cin>>n;
	int q[n],r[n];
	while(n--)
	{
		cin>>p;
		s=s+p;
		if(p%2==0)
		{
			
			q[j]=p;
			j++;
		}
		else
		{
			r[k]=p;
			k++;
		}
	}
	if(s%2==0)
	{
		if(k==0)
		cout<<":(";
		else
		{
			ref=r[0];
			for(j=0;j<k;j++)
			{
				if(ref>r[j])
				ref=r[j];
			}
		cout<<s-ref;	
		}
	}
	else
	cout<<s;
}
