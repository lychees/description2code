    #include<stdio.h>
    #include<iostream>
    #include<vector>
    using namespace std;
    long ans=0;
    void fct(long long int n1,long long int n,vector<int> v,int t,int *l )
    {
    int a[1000];
    while(n1){
    v.push_back(n1%10);
    n1/=10;
    }
    n -=1;
    while(n>=1){
    int temp = 0;
    for(int i = 0;i<v.size();i++){
    v[i] = v[i]*n + temp;
    temp = v[i]/10;
    v[i] %=10;
    if(i==v.size()-1 && temp!=0)
    v.push_back(0);
    }
    --n;
    }
    for(int i=v.size()-1;i>=0;i--){
    a[i]=v[i];
    //cout<<a[i];
    if(a[i]==7 || a[i]==4)
        ans++;
    }
    if(*l!=t)
    //cout<<"\n";
    v.clear();
    ++*l;
    return ;
    }
    int main(){
    vector<int> v;
    int t,l;
    long long n,n1;
    scanf("%d",&t);
    l=1;
    while(l<=t){
        ans=0;
    scanf("%lli",&n);
    n1 =n;
    fct(n1,n,v,t,&l);
    cout<<ans<<"\n";
    }
    return 0;
    }
