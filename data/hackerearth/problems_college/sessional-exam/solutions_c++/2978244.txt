#include <iostream>
#include <vector>

using namespace std;

int main()
{
	ios_base::sync_with_stdio( false );
	int n, m, i, j, ct, maxVal;
	
	cin >> n >> m;
	vector< vector<int> > A( n, vector<int> ( m ) );
	vector<bool> B( n, false );
	
	for( i = 0;i < n;i++ ) {
		string s;
		cin >> s;		
		for( j = 0;j < m;j++ ) {
			A[ i ][ j ] = s[ j ] - '0';
		}
	}
	
	for( j = 0;j < m;j++ ) {
		
		maxVal = A[ 0 ][ j ];
		for( i = 1;i < n;i++ ) {
			maxVal = max( maxVal, A[ i ][ j ] );
		}
		
		for( i = 0;i < n;i++ ) {
			if( A[ i ][ j ] == maxVal ) {
				B[ i ] = true;
			}
		}
	}
	
	ct = 0;
	for( i = 0;i < n;i++ ) {
		
		if( B[ i ] ) {
			ct++;
		}
	}
	
	cout << ct << "\n";
	
	return 0;
}
