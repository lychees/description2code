N = int(raw_input())
x = []
y = []
distx = [0]*N
disty = [0]*N
mod =  1000000007
for i in xrange(0,N):
	a,b = map(int,raw_input().split(" "))
	x.append(a)
	y.append(b)
x = sorted(x)
y = sorted(y)

ans = 0

for i in xrange(1,N):
	distx[i] = (distx[i-1]+((x[i]-x[i-1])*i)%mod)%mod
	ans = (ans+distx[i]) %mod
	
for i in xrange(1,N):
	disty[i] = (disty[i-1]+((y[i]-y[i-1])*i)%mod)%mod
	ans = (ans+disty[i])%mod
print ans
