#include <iostream>
using namespace std;

//amit_singh

long long int A[100000],B[100000];

#define InpA() {for(i=0;i<=N-1;i++) cin>>A[i]; }
#define InpB() {for(i=0;i<=N-1;i++) cin>>B[i]; }

int main()
{
    int N,Q,i,ch,l,r;
    
    long long int sum,count;
    
    cin>>N>>Q;
    
    InpA()
    InpB()
    
    while(Q--)
    {
     cin>>ch>>l>>r;
     
     sum=0;
     count=0;
     
     if(ch==1)
     {
     	for(i=l-1;i<=r-1;i++)
     	{
     		if(count%2==0)
     		 sum=sum+A[i];
     		else
     		 sum=sum+B[i];
     		 
     		count++; 
     	}
     	
     }
     else
     {
     	for(i=l-1;i<=r-1;i++)
     	 {
     	 	if(count%2==0)
     		 sum=sum+B[i];
     		else
     		 sum=sum+A[i];
     		 
     		count++; 
     	 }
     }
     
     
     cout<<sum<<endl;
    }
    
    return 0;
}
