#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long m,n;
    	scanf("%ld%ld",&m,&n);
    	long arr[n];
    	int win[m+1];
    	long i,j;
    	for(i=0;i<n;i++)
    	  scanf("%ld",&arr[i]);
    	win[0]=0;
    	for(i=1;i<=m;i++)
    	{
    		win[i]=0;
    		for(j=0;j<n;j++)
    		{
    			if(i>=arr[j])
    			{
    				if(win[i-arr[j]]==0)
    				{
    					win[i]=1;
    					break;
    				}
    			}
    		}
    	}
    	if(win[m]==1)
    	  cout<<"Little Deepu\n";
    	else
    	  cout<<"Kate\n";
    }
    return 0;
}
