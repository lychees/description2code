typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
typedef pair<ll,ll> ii;


const ll N=1e15;
ll cases,n,p,t,idx,s[100],sz;


int main()
{
	std::ios::sync_with_stdio(0);
	cin.tie(0);
	s[0]=0;
	p=1;
	for(int i=1;;i++)
	{
		p=p*2;
		s[i]=s[i-1]+i*p;
		if(s[i]>N)
		{
			sz=i;
			break;
		}
	}
	cin>>cases;
	while(cases--)
	{
		cin>>n;
		idx=lower_bound(s,s+sz+1,n)-s;
		n-=s[idx-1];
		t=n/idx;
		if(n%idx==0)t--;
		n%=idx;
		if(n==0)n=idx;
		n=idx-n;
		if(t&(1LL<<n))cout<<"Earth\n";
		else cout<<"Hacker\n";
	}
	return 0;
}
