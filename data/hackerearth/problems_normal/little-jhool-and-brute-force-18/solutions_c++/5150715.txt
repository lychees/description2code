#include <iostream>
using namespace std;

int a[704980];
int main()
{
	
	for(int i=0;i<704980;i++)
	{
		a[i]=0;
	}
	for(int i=1;i<90;i++)
	{
		for(int j=i;j<90;j++)
		{
			int x=i*i*i;
			int y=j*j*j;
			if((x+y)<704980)
			   a[x+y]++;
		}
	}
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int flag=0;
		for(int i=n;i>0;i--)
		{
			if(a[i]>1)
			{
				flag=1;
				cout<<i<<"\n";
				break;
			}
		}
		if(flag==0)
		{
			cout<<"-1"<<"\n";
		}
	}
    return 0;
}
