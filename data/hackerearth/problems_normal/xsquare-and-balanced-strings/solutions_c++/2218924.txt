#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;

    while(t--)
    {
        string s;
        cin>>s;

        map<char,int> m;
        map<char,int>::iterator it;

        for(int i=0;i<s.size();i++)
            m[s[i]]++;

        bool flag=true;
        for(it=m.begin() ; it!=m.end() ; it++)
        {
            if((*it).second % 2 != 0)
                {
                    flag=false;
                    break;
                }
        }

        if(flag)
            cout<<1<<endl;
        else
            cout<<-1<<endl;
    }


}
