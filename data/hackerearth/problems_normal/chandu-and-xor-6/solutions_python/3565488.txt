a=raw_input().split()
n=int(a[0])
m=int(a[1])
mat=[[0]*m]*n
for i in range(n):
	mat[i]=raw_input().split()

rw=[0]*n
cm = [0]*m
for i in range(n):
	for j in range(m):
		mat[i][j]=int(mat[i][j])
		rw[i]=rw[i]+mat[i][j]
		cm[j]=cm[j]+mat[i][j]

mxm = -1
for i in range(n):
	for j in range(m):
		tem = rw[i] ^ cm[j]
		if(mxm<tem):
			mxm = tem

print mxm
