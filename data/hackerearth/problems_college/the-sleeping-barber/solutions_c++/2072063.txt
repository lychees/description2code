#include <bits/stdc++.h>
using namespace std;

int main()
{
    int d;
    cin>>d;
    long long g,m,n,count;
    while(d--){
    	n=0;
    	count=0;
    	cin>>g>>m;
    	int a[g];
    	
    	for(int i=0;i<g;i++){
    		cin>>a[i];
    		
    	}
    	sort(a,a+g);
    	for(int i=0;i<g;i++){
    		n+=a[i];
    		if(n<=m){
    			count++;
    		}
    	}
    	cout<<count<<"\n";
    }
    return 0;
}
