#include <bits/stdc++.h>
using namespace std;
#define lli long long int
#define MAX 1000005
#define MOD 1000003
vector<lli>f(MAX);
lli powM(lli a,lli b,lli mod){
	if(b==0)return 1;
	lli ans=powM(a,b/2,mod);
	ans=(ans*ans)%mod;
	if(b&1)ans=(ans*a)%mod;
	return ans%mod;
}
void pre(){
	f[0]=f[1]=1;
	for(int i=2;i<MAX;i++){
		f[i]=i*f[i-1]%MOD;
	}
}
int main()
{
   int t;
   pre();
  // cout<<powM(2,3,1000003)<<endl;
   lli n,x;
   cin>>t;
   while(t--){
   	cin>>n>>x;
   	if(n<MAX){
   		cout<<(x*f[n]%MOD)<<endl;
   	}
   	else{
   		cout<<"0"<<endl;
   	}
   }
    return 0;
}
