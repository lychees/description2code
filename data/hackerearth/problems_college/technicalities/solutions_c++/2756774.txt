
#include<bits/stdc++.h>

using namespace std;

#define si(x) scanf("%d",&x)
#define pri(x) printf("%d",x)
#define sll(x) scanf("%lld",&x)
#define pll(x) printf("%lld",x)
#define sstr(s) scanf("%s",s)
#define pstr(s) printf("%s",s)
#define nl printf("\n")
#define ll long long int
#define mod 1000000007

int main()
{
    int t;
    si(t);
    while(t--) {
	int n;
	si(n);
	char s[n+1];
	sstr(s);
	int a[n+1];
	for(int i=0;i<n;++i) {
	    a[i]=s[i]-'0';
	}
	ll num=0;
	if(n>10) {
	    for(int i=0;i<10;++i) {
		num = num*10 + a[i];
	    }
	    num = num%mod;
	    for(int i=10;i<n;++i) {
		num = num*10 + a[i];
		num = num%mod;
	    }
	    pll(num); nl;
	}
	else {
	    for(int i=0;i<min(10,n);++i) {
		num = num*10 + a[i];
	    }
	    pll(num%mod); nl;
	}
    }
    return 0;
}
