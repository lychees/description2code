int readInt () {
	bool minus = false;
	int result = 0;
	char ch;
	ch = getchar();
	while (true) {
		if (ch == '-') break;
		if (ch >= '0' && ch <= '9') break;
		ch = getchar();
	}
	if (ch == '-') minus = true; else result = ch-'0';
	while (true) {
		ch = getchar();
		if (ch < '0' || ch > '9') break;
		result = result*10 + (ch - '0');
	}
	if (minus)
		return -result;
	else
		return result;
}

bool RelativelyPrime (int check1, int check2) { // Assumes a, b > 0
   return (check1<check2) ? RelativelyPrime(check2,check1) : !(check1%check2) ? (check2==1) : RelativelyPrime (check2, check1%check2);
}
