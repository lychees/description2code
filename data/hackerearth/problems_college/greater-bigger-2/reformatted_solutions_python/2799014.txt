def nextLexo(data, n):
    i = n-2
    while not (i < 0 or data[i] < data[i+1]):
        i -= 1
    if i < 0:
        return False
    j = n-1
    while not (data[j] > data[i]):
        j -= 1
    data[i], data[j] = data[j], data[i]
    data[i+1:] = reversed(data[i+1:])
    return True


for T in range(input()):
    s = list(raw_input().rstrip())
    print "".join(s) if nextLexo(s, len(s)) else 'no answer'
