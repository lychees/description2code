#include<bits/stdc++.h>
using namespace std;

string a,b;

void change(int n) {
    string str;
    str.resize(n);
    for(int i=0;i<n/2;i++) {
        str[(i<<1)]=a[i];
        str[(i<<1)+1]=a[n-i-1];
    }
    if(n%2==1) {
        str[n-1]=a[n/2];
    }
    a=str;
    return;
}
int main() {
    int k;
    scanf("%d",&k);
    cin>>a;
    b=a;
    int n=a.size();
    change(n);
    int p=1;
    while(a!=b) {
        p++;
        change(n);
    }
    k=(p-k%p);
    a=b;
    while(k--) {
        change(n);
    }
    cout<<a<<endl;
    return 0;
}
