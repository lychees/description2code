#include <iostream>
using namespace std;
long long gcd(long long a, long long b)
{
	if (b==0)
		return a;
	else
		return gcd(b,a%b);
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	long long arr[n],g;
    	for(int i=0;i<n;i++)
    	cin>>arr[i];
     g=arr[0];
    for(int i=1;i<n;i++)
    {
    	g=gcd(arr[i],g);
    }
    cout<<g<<endl;
    	
    }
    return 0;
}
