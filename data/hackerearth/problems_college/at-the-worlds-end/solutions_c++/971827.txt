#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
using namespace std;
int a[505][505];
int dp[505][505];
int minVal[505][505];
int main() {
    int t;
    scanf("%d",&t);
    while(t--) {
        int n,m;
        scanf("%d%d",&m,&n);
        for(int i=1;i<=m;i++) for(int j=1;j<=n;j++) scanf("%d",&a[i][j]);
        for(int i=1;i<=m;i++) {
            dp[i][1]=dp[i-1][1]+a[i][1];
            minVal[i][1]=min(minVal[i-1][1],dp[i][1]);
        }
        for(int i=1;i<=n;i++) {
            dp[1][i]=dp[1][i-1]+a[1][i];
            minVal[1][i]=min(minVal[1][i-1],dp[1][i]);
        }
        for(int i=2;i<=m;i++) {
            for(int j=2;j<=n;j++) {
                int temp1=dp[i-1][j]+a[i][j];
                int temp2=dp[i][j-1]+a[i][j];
                int minVal1=min(minVal[i-1][j],temp1);
                int minVal2=min(minVal[i][j-1],temp2);
                if(minVal1>minVal2) {
                    dp[i][j]=temp1;
                    minVal[i][j]=minVal1;
                }
                else if(minVal1<minVal2) {
                    dp[i][j]=temp2;
                    minVal[i][j]=minVal2;
                }
                else {
                    dp[i][j]=max(temp1,temp2);
                    minVal[i][j]=minVal1;
                }
            }
        }
        cout<<(-minVal[m][n])+1<<endl;
    }
    return 0;
}
