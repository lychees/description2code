#include <iostream>
#include <cmath>
#include <string.h>
using namespace std;

int main()
{
    int t,z,a[1003][1003],c=0;
    float x,y;
    //string s;
    char input[203];
    cin>>t;
    for(int i=0;i<=t;i++)
    {
    	c=0;
    	//cin>>s;
    	cin.getline(input,sizeof(input));
    	z=strlen(input);
    	x=y=sqrt(z);
    	//cout<<z<<"\t"<<x;
    	for(int j=0;j<x;j++)
    	{
    		for(int k=0;k<y;k++)
    		{
    			a[j][k]=input[c];
    			c++;
    		}
    	}
    	for(int j=0;j<x;j++)
    	{
    		for(int k=0;k<y;k++)
    		{
    			cout<<char(a[k][j]);
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
