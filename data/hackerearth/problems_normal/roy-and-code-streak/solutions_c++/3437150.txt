#include<stdio.h>
#define max(a, b) (((a) >= (b)) ? (a) : (b))
int main(){
	int t, n, s, r; scanf("%d", &t);
	while(t-- && scanf("%d", &n)){
		int in[1000001] = {0}, c = 0, m = 0;
		while(n-- && scanf("%d%d", &s, &r)){
			if(r == 0){
				if(c > m)	m = c;
				c = 0;
			}
				else{	
					if(!in[s])	c++;
					in[s] = 1;
				}
		}
		printf("%d\n", max(m, c));
	}		
	return 0;
}
