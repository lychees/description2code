#include <bits/stdc++.h>
#define LL long long int
using namespace std;

LL a[1000011];

LL gen(LL n){

LL ans=-1,fib1 = 1,fib2 = 1;

LL total = 0,add = 0;

while(total<n){
LL next = fib1+fib2;

add = next-fib2-1;
if(total+add<n)
    total+=add;
    else {

    //cout << "add ,total = " << add  << ',' <<  total << endl;
    ans = fib2+(n-total);
    break;
    }

fib1=fib2;
fib2=next;

//cout << fib1 << " " << fib2 << endl;
}

return ans;
}



int main()
{
    ios_base::sync_with_stdio(false);

    int t;
    cin >> t;
    while(t--){
        LL n;
        cin >> n;
        cout << gen(n) << endl;
    }

   return 0;
}
