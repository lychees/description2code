#include <iostream>
#include <string>
#include <cmath>
using namespace std;

int main()
{
	int t,len,cost;
	string str;
	cin >> t;
	for(int i=0;i<t;i++){
		cin >> str;
		len=str.length();
		cost=0;
		for(int j=0;j<len/2;j++)
			cost += abs(str[j]-str[len-1-j]);
		cout << cost << endl;
	}




    return 0;
}
