#include<bits/stdc++.h>
using namespace std ;
int gcd(int a,int  b)
   {
     if (b == 0)
       return a ;
    else
        return gcd(b, a % b) ;
   }

int main()
{   int test ;
    cin>>test;
    while(test--)
    {   int flag = 0;
       int n ;
       cin>>n;
       int s[n];
       for(int i = 0 ; i < n ; i++)
        cin>>s[i];
       int g = gcd(s[0],s[1]);
          for(int i = 2 ; i<n;i++)
        {
            int temp = gcd(s[i],g);
            if(  temp != 1)
                g=temp;

            else
                {   cout<<1<<endl;
                    flag = 1;
                    break;
                }
        }
        if(!flag)
            cout<<g<<endl;
    }
    return 0 ;
}
