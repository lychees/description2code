#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    cin >> n;

    char arr[n+1][200];

    vector <string> :: iterator it;
    for (int i = 0;i <= n;i++){
        gets(arr[i]);
    //cout << strlen(arr[i]);
    }

    for (int i = 1;i <=n;i++)
    {
        vector <string> pr;
        int sum = 0;
        for (int j = 0;j < strlen(arr[i]);j++)
        {
            if (arr[i][j] > 48 && arr[i][j] <= 57)
                sum = sum + (arr[i][j] - 48);
        }

       // cout << sum << endl;
       string name = "";
       for (int j = 0;j < strlen(arr[i]);j++)
       {
           if (((arr[i][j] >= 'a' && arr[i][j] <= 'z') || (arr[i][j] >= 'A' && arr[i][j] <= 'Z')) && arr[i][j] != ' ')
                name = name + arr[i][j];
           if (arr[i][j] == ' ' || arr[i][j+1] == '\0')
           {
               pr.push_back(name);
               name = "";
           }
       }
        //cout << endl;
       sort(pr.begin(),pr.end());

       for (int i = 0;i < pr.size();i++){
            sum = sum - pr[i].size();
        cout << pr[i] << " " << abs(sum) << " ";
       }

       cout << endl;
    }

    return 0;
}
