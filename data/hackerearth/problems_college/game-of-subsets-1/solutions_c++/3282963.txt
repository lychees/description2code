#include <iostream>
using namespace std;
long long int power(int x,int n,int m)
{
	if(n==0)
		return 1;
	
	long long int z = power(x,n/2,m)%m;
	z = (z * z)%m;
	if(n%2)
		z = (x*z)%m;
	return z;
}

int main()
{
    int test,m=1000000007,n;
    cin >> test;
    while(test--)
    {
    	cin >> n;
    	cout<<power(3,n,m)<<endl;;
    }
}
