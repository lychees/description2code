import itertools


def getFactors(n):
    i = 2
    allFactors = []
    while i * i <= n:
        if n % i:
            i += 1
        else:
            n //= i
            allFactors.append(i)
    if n > 1:
        allFactors.append(n)
    return allFactors


def getCombination(n):
    allFactors = getFactors(n)
    nOfComb = 0
    for k in range(1, len(allFactors) + 1):
        nOfComb += len(set(itertools.combinations(allFactors, k)))
    return nOfComb


for _ in range(input()):
    print getCombination(input())
