#include <stdio.h>
#include<cmath>
#include<iostream>
#define LL long long int
LL nx[1000009],ny[1000009];
using namespace std;
int main()
{
    LL t;
    cin>>t;
    while(t--)
    {
        LL i,n,noT,niT,fx,fy,ans;
        scanf("%lld",&n);
        for(i=0;i<n;i++)scanf("%lld",&nx[i]);
        for(i=0;i<n;i++)scanf("%lld",&ny[i]);
        scanf("%lld %lld %lld %lld",&fx,&fy,&noT,&niT);
        ans=(llabs(fx)+llabs(fy))*noT;
        for(i=0;i<n;i++)
        {
        	
            LL curr1=(llabs(nx[i])+llabs(ny[i]))*noT;
            LL curr2=(llabs(fy-ny[i])+llabs(fx-nx[i]))*niT;
        //    cout<<curr1<<"  "<<curr2<<endl;
            if(curr1 + curr2<ans)ans=curr1+curr2;
        }
        cout<<ans<<endl;
    }
    return 0;
}
