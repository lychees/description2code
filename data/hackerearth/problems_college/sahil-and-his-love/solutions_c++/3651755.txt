#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int
#define s(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)
#define sd(x) scanf("%lf", &x)
#define mod 1000000007
#define get getchar_unlocked

int dp[105][105];

int func2(int l, int r, string &s)
{
    if (l > r)
        return 0;
    if (dp[l][r] != -1)
        return dp[l][r];
    if (l == r)
        return dp[l][r] = 1;
    dp[l][r] = 2*(s[l] == s[r]) + func2(l+1, r-1, s);
    dp[l][r] = max(dp[l][r], max(func2(l, r-1, s), func2(l+1, r, s)));
    return dp[l][r];
}

int func(string s)
{
    int i, j;
    for (i = 0; i < s.size(); ++i) {
        for (j = i; j < s.size(); ++j)
            dp[i][j] = -1;
    }
    return func2(0, s.size()-1, s);
}

int main()
{
    int t, i, ans, n;
    cin >> t;
    string s, h;
    while (t--) {
        cin >> s;
        h.clear();
        ans = 0;
        n = s.size();
        for (i = 0; i < n; ++i) {
            h.pb(s[s.size()-1]);
            s.pop_back();
            ans = max(ans, func(s+h));
        }
        printf("%d\n", ans);
    }
    return 0;
}
