#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	map<string,int>vec;
	while(t--)
	{
		string str;
		cin>>str;
		sort(str.begin(),str.end());
		vec[str]++;
	}
	map<string,int>::iterator it;
	int max=0;
	for(it = vec.begin();it!=vec.end();++it)
	{
		if(max<it->second)
		{
			max = it->second;
		}
	}
	cout<<max<<"\n";
}
