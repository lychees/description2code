#include <cstdio>
int main() {
  long long n;
  scanf("%lld", &n);
  long long m = (n + 2) / 2;
  while (m <= n) 
  {
    long long x = m;
    bool ok = false;
    for (long long p = 2; p * p <= x; p++) 
    {
      long long pp = 1;
      while (x % p == 0) 
      {
        pp *= p;
        x /= p;
      }
      if (pp == 1) 
      continue;
      if (2 * pp > n) 
      {
        ok = true;
        break;
      }
      if (2 * x <= n) 
        break;
    }
    if (ok || 2 * x > n) 
      break;
    m++;
  }
  printf("%lld\n", m);
}
