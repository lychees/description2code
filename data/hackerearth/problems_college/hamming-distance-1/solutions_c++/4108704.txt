#include<cstdio>
#include<vector>

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,m;
		scanf("%d%d",&n,&m);
		int ans=n^m;
		int count=0;
		while(ans)

	}
