#include <iostream>
using namespace std;
int gcd(int a, int b){
	if(b == 0) return a;
	return gcd(b,a%b);
}
 
int main()
{
    int A, B, C;
    int num,den,DIV;
    cin >> A >> B >> C;
    if( C <= A && C <= B ){
    	num = C*C;
    	den = 2*A*B;
    }
    else if(A+B <= C){
    	num = A*B;
    	den = A*B;
    }
    else if( C > A && C <= B){
    	num = C*C - (C-A)*(C-A);
    	den = 2*A*B;
    }
    else if(C <= A && C > B){
    	num = C*C - (C-B)*(C-B);
    	den = 2*A*B;
    }
    else if(C > A && C > B){
    	num = C*C - (C-A)*(C-A) - (C-B)*(C-B);
    	den = 2*A*B;
    }
    
    DIV = gcd(num,den);
    num = num/DIV;
    den = den/DIV;
    cout << num << "/" << den << endl;
    return 0;
}
