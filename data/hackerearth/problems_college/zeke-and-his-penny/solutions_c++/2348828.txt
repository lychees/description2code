#include<iostream>
#include<algorithm>
#define Q 1000000007
 
using namespace std;
 
int main()
{
	long long t,sum,a[3];
	cin>>t;
	while(t--){
		cin>>a[0]>>a[1]>>a[2];
		sort(a,a+3);
	//	cout<<a[0]<<" "<<a[1]<<" "<<a[2]<<endl;;
		sum=(a[0]%Q);
		//cout<<sum<<" ";
		sum=sum*(a[1]-1)%Q;
		//cout<<sum<<" ";
		sum=sum*(a[2]-2)%Q;
		//cout<<sum<<" ";
		cout<<sum%Q<<endl;
	}
	return 0;
}
