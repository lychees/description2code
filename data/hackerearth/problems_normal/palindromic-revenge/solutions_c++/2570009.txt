#include <iostream>
#include <string.h>
using namespace std;

int main()
{
	int t,i=0,j,count=0,temp;
	char string[100000];
	cin>>t;
	while(t--)
	{
		cin>>string;
		count=0;
		for(i=0,j=(strlen(string)-1);(i-1!=j && i!=j);i++,j--) //i-1!=j <- this so that the middle 2 characters are also compared in even strlen
		{
			//cout<<endl<<string[i]<<" and "<<string[j];
			if(string[i]!=string[j])
			{
				temp=(int)string[i]-(int)string[j];
				(temp<0)?count=count+(-temp):count+=temp;
			}
			//cout<<" count= "<<count;
			
		}
		cout<<endl<<count;
	}
}
