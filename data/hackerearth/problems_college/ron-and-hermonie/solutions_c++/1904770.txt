#include<bits/stdc++.h>
using namespace std;
int LastDigit[][4]={ {0,0,0,0},{1,1,1,1},{6,2,4,8},{1,3,9,7},{6,4,6,4},{5,5,5,5},{6,6,6,6},{1,7,9,3},{6,8,4,2},{1,9,1,9}};
int main()
{
    long long int T,a,b,A;
    scanf("%lld",&T);
    //cin>>T;
    while(T--){
        scanf("%lld%lld",&a,&b);
        //cin>>a>>b;
        if(b==0) A=1;
        else A=LastDigit[a%10][b%4];
        printf("%lld\n",A);
    }
  return 0;
}
