int a[100005];
int visit[100005]={0};
int main(){
	int t,n,i;
	scanf("%d",&t);
	while(t--){
		int l=0,r=0,diff;
		scanf("%d",&n);
		diff=n+1;
		for(i=1;i<=n;i++){
			scanf("%d",&a[i]);
			visit[i]=0;
		}
		int64 s=0;
		for(i=1;i<=n;i++){
			s+=a[i];
			if(s>=n)
			s%=n;
			if(s==0){
				if(i<diff){
					diff=i;
					l=1;
					r=i;
				}
				continue;
			}
			if(visit[s]==0){
				visit[s]=i;
				continue;
			}
			if(visit[s]){
				if(i-visit[s]<diff){
					diff=i-visit[s];
					l=visit[s]+1;
					r=i;
				}
				visit[s]=i;
			}
			
		}
		printf("%d %d\n",l,r);
	}
	return 0;
}
