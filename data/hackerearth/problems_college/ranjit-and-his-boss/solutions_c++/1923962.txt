map<pair<int,int>,int> mp;
set<int> st[100005];
int main()
{
    int n,x,y,i,j;
    sc(n);
    for(i=0;i<n;i++)
    {
        sc(x,y);
        st[x].insert(y);
        mp[make_pair(x,y)]++;
    }
    map<pair<int,int>,int> :: iterator it,it2;
    lli ans=0;
    int temp;
    for(it=mp.begin();it!=mp.end();it++)
    {
        int a=it->first.first;
        if(st[a].size()<=1) continue;
        for(it2=mp.lower_bound(make_pair(a,it->first.second+1));it2!=mp.end();it2++)
        {
            if(it2->first.first != it->first.first) break;
            temp=it2->first.second-it->first.second;
            if(mp.find(make_pair(it->first.first+temp,it->first.second))!=mp.end() and mp.find(make_pair(it2->first.first+temp,it2->first.second))!=mp.end())
                ans++;
        }
    }
    printf("%lld\n",ans);
}
