#include <bits/stdc++.h>
using namespace std;

int maximum(int x,int y)
{
	if(x>y)return x;return y;
}
int minimum(int x,int y)
{if(x>y)return y;
return x;
}
int main()
{
    int t,n,x,y,i,j,min,max,a[1000000];
   	cin>>t;  
   	while(t--)
   	{ 
   		cin>>n>>x>>y;
   		min=minimum(x,y);
   		max=maximum(x,y);
   		a[x]=a[y]=1;
   		for(i=0;i<min;i++)
   		{
   			a[i]=0;
   		}
   		for(i=min+1;i<max;i++)
   		{
   			if(a[i-min]==0)
   				a[i]=1;
   			else a[i]=0;
   		}
   		for(i=max+1;i<=n;i++)
   		{
   			if(a[i-min]==0 || a[i-max]==0)
   				a[i]=1;
   			else a[i]=0;
   		}
   		if(a[n]==0)cout<<"0\n";
   		else cout<<"1\n";
   	}
    return 0;
}
