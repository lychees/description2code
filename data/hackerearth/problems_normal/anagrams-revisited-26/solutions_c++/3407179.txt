#include<iostream>
#include<cstdio>
#include<map>
#include<algorithm>
typedef long long ll;
using namespace std;
int main(){
	int n;
	cin>>n;
	int max=0;
	map<string,int> mp;
	while(n--){
		string str;
		cin>>str;
		sort(str.begin(),str.end());
		//cout<<str<<endl;
		mp[str]++;
		if(max<mp[str]){
			max=mp[str];
		}
	}
	cout<<max<<endl;
}
