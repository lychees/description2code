#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

ll max(ll a, ll b)
{
	if(a > b) return a;
	else return b;
}

int main()
{
	ios::sync_with_stdio(false);
	
    ll t;
    cin >> t;
    
    while(t--)
    {
    	ll n, q;
    	cin >> n >> q;
    	
    	ll arr[n];
    	for(ll i=0;i<n;i++) cin >> arr[i]; 
    	
    	sort(arr,arr+n);
    	ll ret = 0;
    	
    	for(ll i=n-1;i>=max(0,n-q);i--) ret += arr[i];
    	
    	cout << ret << endl;
    }
    return 0;
}
