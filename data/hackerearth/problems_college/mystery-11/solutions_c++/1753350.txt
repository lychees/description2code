#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	char ch[1000000];
	int t;
	cin>>t;
	while(t--)
	{
	long long s=0;
	cin>>ch;
	for(int i=0;i<strlen(ch);i++)
    {
        if(ch[i] == '0')
           s += 6;
        else if(ch[i] == '1')
           s += 2;
        else if(ch[i] == '2')
           s += 5;
        else if(ch[i] == '3')
           s += 5;
        else  if(ch[i] == '4')
           s += 4;
        else if(ch[i] == '5')
           s += 5;
        else if(ch[i] == '6')
           s += 6;
        else if(ch[i] == '7')
           s += 3;
        else if(ch[i] == '8')
           s += 7;
        else if(ch[i] == '9')
           s += 6;
    }
    cout<<s<<endl;
	}
}
