#include <bits/stdc++.h>

using namespace std;

#define mod 1000000007
long long a[10000001];
long long sum[10000001];

int main()
{
	int t,n,i,j;
	a[0] = 0;
	a[1] = 1;
	a[2] = 1;
	sum[0] = 0;
	sum[1] = 1;
	sum[2] = 2;
	for (i = 3; i <= 1000001; i++) {
		a[i] = a[i-2] + a[i-1];
		a[i] %= mod;
		sum[i] = (a[i]-1) % mod;
	}
	scanf("%d",&t);
	while (t --) {
		scanf("%d",&n);
		printf("%lld\n",sum[n+2]);
	}
	
	return 0;
}
