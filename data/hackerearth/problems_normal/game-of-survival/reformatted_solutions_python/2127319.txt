t = int(raw_input())
while t != 0:
    t -= 1
    n = int(raw_input())
    a = []
    s = []
    for i in range(n):
        a.append(int(raw_input()))
    for i in range(n):
        s.append(int(raw_input()))
    a.sort()
    s.sort()
    for i, j in zip(a, s):
        if i < j:
            print "NO"
            break
    else:
        print "YES"
