#include<bits/stdc++.h>
using namespace std;

int root(int a[],int i)
{
    while(a[i]!=i)
    {
        i=a[i];
    }
    return i;
}
void init(int a[],int n)
{
    for(int i=1;i<=n;i++)
    {
        a[i]=i;
    }
}
void uni(int a[],int A,int B)
{
    int roota = root(a,A);
    int rootb  =root(a,B);
    a[rootb] = roota;
}

int main()
{
    int n;
    cin>>n;
    int a[n+1];
    int k;
    cin>>k;
    int p,q;
    init(a,n);
    while(k--)
    {
        cin>>p>>q;
        uni(a,p,q);
    }
    int flag=0;
    for(int i=1;i<=n;i++)
    {
        if(root(a,i)==i)
            flag++;
    }
    cout<<flag;
    return 0;
}
