#include<iostream>
using namespace std;
int main()
{
    int n,cnt;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cnt=0;
        long int m;
        cin>>m;
        long int a[m];
        for(int j=0;j<m;j++)
        {
            cin>>a[j];
        }
        for(int k=0;k<m;k++)
        {
            for(int l=0;l<m;l++)
            {
                if(l>=k&&a[k]==a[l])
                {
                   cnt++;
                }
            }
        }
        cout<<cnt<<"\n";
    }
    return 0;
}
