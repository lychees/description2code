#include<bits/stdc++.h>
using namespace std;

int p[1000001];

int fun(int i) {
    
    long long int j=i;
    while(j>0) {
        long long int temp=j%10;
        if(temp==0) {
            j=j/10;
            continue;
        }
        else  {
            temp=temp*temp;
            if(i>=temp&&p[i-temp]) return 1;
            j=j/10;
        }
    }
    return 0;
    
}
int main () {
    
    long long int i;
    
    for(i=1;i<=1000000;i++) {
        
        if(pow(i,i)<=1000000) p[int(pow(i,i))]=1;
    }
    
    
    for(i=1;i<=1000000;i++) p[i]=p[i]||fun(i);
    
    int t;
    scanf("%d",&t);
    while(t--) {
        int n;
        scanf("%d",&n);
        if(p[n]) printf("Yes\n");
        else printf("No\n");
    }
        
     return 0;   
    }
