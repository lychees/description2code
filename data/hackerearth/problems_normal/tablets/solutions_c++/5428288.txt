#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    long long a[n];
    for(int i=0;i<n;i++)
     cin>>a[i];
    long long ans[n];
    for(int i=0;i<n;i++)ans[i] = 1;
    for(int i=1;i<n;i++)
     {
     	if(a[i]>a[i-1])
     	 ans[i]=max(ans[i],1+ans[i-1]);
     }
    for(int i=n-2;i>=0;i--)
     {
     	if(a[i]>a[i+1])
     	 ans[i]=max(ans[i],1+ans[i+1]);
     }
    for(int i=1;i<n-1;i++)
     if(a[i]>a[i-1] && a[i]>a[i+1])
      ans[i] = min(1+max(ans[i-1],ans[i+1]),ans[i]);
    long long c=0;
    for(int i=0;i<n;i++)c+=ans[i];
    cout<<c;
    return 0;
}
