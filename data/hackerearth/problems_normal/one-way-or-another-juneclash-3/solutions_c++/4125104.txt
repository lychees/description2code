#include <iostream>
#include <string>
using namespace std;
#define ull long long int
int main()
{
    int N,t;
    ull r,c,x;
    bool rcpar,xpar;
    bool rcpar1,xpar1;
    cin>>N;
    for(int i=0;i<N;i++){
    	cin>>t;
    	cin>>r>>c>>x;
    	rcpar=(r+c)&1;
    	xpar=x&1;
    	while(--t){
    		cin>>r>>c>>x;
    		rcpar1=(r+c)&1;
    		xpar1=x&1;
    		if(rcpar==rcpar1&&xpar!=xpar1)
    		break;
    		if(rcpar!=rcpar1&&xpar==xpar1)
    		break;
    	}
    	if(t){
	    	cout<<"No\n";
	    	if(N-1-i)
	    	while(--t){
	    	cin>>r>>c>>x;
	    	}
	    }
    	else
    	cout<<"Yes\n";
    }
    return 0;
}
