#include <bits/stdc++.h>
using namespace std;
int main()
{
	long long t,x,y,n,res=0;
	cin>>t;
	while(t--)
	{
		res=0;
		cin>>n>>x>>y;
		long long a[n];
		long long b[n];
		for(long long i=0;i<n;i++)
		{
			cin>>a[i]>>b[i];
		}
		sort(a,a+n);
		reverse(a,a+n);
		sort(b,b+n);
		reverse(b,b+n);
		for(long long i=0;i<n;i++)
		{
			if(a[i]>b[i])
			{
				res+=((a[i]-b[i])*y);
			}
			else
			{
				res+=((b[i]-a[i])*x);
			}
		}
		cout<<res<<endl;
	}
    return 0;
}
