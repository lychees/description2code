#include <iostream>
using namespace std;
#include <stdio.h>

int main()
{
    int t;
    scanf("%d",&t);
    while (t--)
    {
    	long long int a[4],x,y,d,b,r;
    	for (int i=0;i<4;i++)
    	{
    		scanf("%lld ",&a[i]);
    	}
    	x= a[0]-a[2];
    	x=abs(x);
    	y=a[1]-a[3];
    	y=abs(y);
    	d=x;b=y;
    	if(x==0 && y!=0)
    	{
    		b=y;
    	}
    	else if (y==0 && x!=0)
    	{
    		b=x;
    	}
    	else if (x==0 && y==0)
    	{
    		b=1;
    	}
    	else
    	{
    		while (d>0)
   	 		{
    			r=b%d;
    			b=d;
    			d=r;
    		}
    	}
    	printf("%lld\n",b-1);
    }
}
