#include <iostream>
using namespace std;

int main()
{
    long long t,n,i,j,k,l,m,sum=0,a[1000000];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=1;i<=n;++i)
    	a[i]=i;
    	for(i=1;i<=n/2;++i)
    	{
    		if(i==n/2)
    		{
    		if(n%2==0)
    		break;
    		else
    		sum=sum+(a[n-i+1]-a[i]);	
    		}
    		else
    		sum=sum+(a[n-i+1]-a[i])+(a[n-i+1]-a[i+1]);
    	}
    	l=n/2;
    	//if(n%2!=0)
    	sum=sum+(a[l+1]-1);
    	cout<<sum<<endl;
    	sum=0;
    }
    return 0;
}
