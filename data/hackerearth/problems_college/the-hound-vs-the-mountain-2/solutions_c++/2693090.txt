#include<bits/stdc++.h>

using namespace std;
#define ll long long
#define ul unsigned long
#define DEBUG(x) cout<<__LINE__<<">"<<x<<endl;
ll ipow(ul b,ul e){
    ll res = 1;
    for(;e>0;e >>=1,b *=b) 
        if(e&0x1)res *= b;
    return res;
}

int main(){
    int N; string s1,s2;
    string m1="The Hound",m2 = "The Mountain";
    bitset<8> b1,b2;
    cin>>N;
    while(N--){
        cin>>s1>>s2;
        b1 = bitset<8>(s1);
        b2 = bitset<8>(s2);
        ul base1 = b1.to_ulong(),base2 = b2.to_ulong();
        if(base1 > 2 && base2 > 2) {
            if(base1 >= base2) cout<<m1<<endl;
            else cout<<m2<<endl;
        }
        else{
            ll val1 = ipow(base1,base2);
            ll val2 = ipow(base2,base1);
            if(val1 >= val2) cout<<m1<<endl;
            else cout<<m2<<endl;
        }
    }
    return 0;
}
