t = input()
for _ in xrange(t):
    n, m = map(int, raw_input().split())
    l = map(int, raw_input().split())
    already = set(l[:n])
    new = l[n:]
    ans = [0]*m
    for index, element in enumerate(new):
        if element in already:
            ans[index] = 'YES'
        else:
            ans[index] = 'NO'
            already.add(element)
    print '\n'.join(x for x in ans)
