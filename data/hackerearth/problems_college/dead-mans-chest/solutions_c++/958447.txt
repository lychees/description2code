#include<bits/stdc++.h>
using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		char str[100001];
		cin>>str;
		int len=strlen(str);
		int i=0,j=len-1;
		int freq[26]={0};
		int largest=0;
		while(str[i]!='\0')
		{
		    freq[str[i]-'a']++;
		    i++;
		}
		int odd=0,even=0;
		for(i=0;i<26;i++)
		{
		    if(freq[i]%2==1)
		        odd++;
		    else
		        even++;
		    if(freq[i]>largest)
		        largest=freq[i];
		}
		if(odd<=1)
		    cout<<"JACK";
	
		else
		{
		    if(odd%2==1)
		    {
		        cout<<"JACK";
		    }
		    else
		        cout<<"DAVY JONES";
		}
		cout<<endl;
	}
	return 0;
}
