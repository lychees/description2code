#include <bits/stdc++.h>
using namespace std;
int dp[110][100005];
int a[110];
int fun(int index,int sum,int n)
{
		if(sum==0)
	return true;
	if(index==n)
	return false;

	if(dp[index][sum]!=-1)
	return dp[index][sum];
	if(a[index]<=sum)
	dp[index][sum]=fun(index+1,sum-a[index],n)||fun(index+1,sum,n);
	else
	dp[index][sum]=fun(index+1,sum,n);
	return dp[index][sum];
}
int main()
{
	int n,t,x;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	scanf("%d",&a[i]);
	memset(dp,-1,sizeof dp);
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&x);
		if(x>100000)
		{
			cout<<"No\n";
			continue;
		}
		int res=fun(0,x,n);
		if(res)
		cout<<"Yes\n";
		else
		cout<<"No\n";
	}
	return 0;
	
}
