t = int(raw_input());
for k in range(1,t+1):
	n = int(input());
	a = [int(x) for x in raw_input().split()]
	count = 0
	for i in range(n):
		for j in range(i+1,n):
			if a[i] != a[j]:
				if ((a[i]%2)==0 and (a[j]%2)==0):
					count += 1;
				elif ((a[i]%2)!=0 and (a[j]%2)!=0):
					count += 1;
				else :
					pass
	print count
