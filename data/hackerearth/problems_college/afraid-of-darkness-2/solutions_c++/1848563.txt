#define ll long long int

#define get getchar

using namespace std;

bool cmp(ll a, ll b)
{
	return b < a;
}

int main()
{
	ll t;
	
	
	scanf("%lld", &t);
	
	while (t--) {
		ll arr[1005];	
		ll n;
		scanf("%lld", &n);
		
		for (ll i = 0; i < n; i++) {
			scanf("%lld", &arr[i]);	
		}
		
		ll ct = 0;
		ll k = 1;
		
		while (1) {
			
			sort(arr, arr + n, cmp);
			
			
			ll flag = 0;
			
			for (ll i = 0; i < k; i++) {
				
				if (arr[i] == 0) {
					flag = 1;
					break;
				}
				else {
					arr[i]--;
				}
				
			}
			
			k++;
			
			if (flag == 1) {
				break;
			}
			else {
				ct++;
			}
			
			
		}
		
		cout << ct << endl;
		
	}
	
	return 0;
}
