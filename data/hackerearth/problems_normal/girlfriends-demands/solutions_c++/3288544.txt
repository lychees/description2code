#include <iostream>
#include<cstring>
#include<cstdio>
#define intt long long int
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    char s[1000000];
    scanf("%s",s);
    intt l=strlen(s);
    intt n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	intt a,b;
    	scanf("%lld%lld",&a,&b);
    	a=a-1;
    	b=b-1;
    	if(a>=l)
    	a=a%(l);
    	if(b>=l)
    	b=b%(l);
        if(s[a]==s[b])
    	cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
    }
    return 0;
}
