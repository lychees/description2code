#include <string.h>
#define ll long long int

using namespace std;

int main()
{
    ll t;

    cin >> t;

    getchar();

    while (t--) {
        string str;
        cin >> str;

        ll count1 = 0;
        ll arr[100];
        ll i;

        memset(arr, -1, sizeof(arr));
        ll st = 0;
        ll maxi = -1;

        for (i = 0; i < str.length(); i++) {

            if (arr[str[i] - 'A'] == -1) {
                arr[str[i] - 'A'] = i;
                count1++;
            }
            else {

                if (arr[str[i] - 'A'] < st) {
                    arr[str[i] - 'A'] = i;
                    count1++;
                }
                else {
                    maxi = max(maxi, count1);
                    ll temp = st;
                    st = arr[str[i] - 'A'] + 1;
                    count1 = i - st + 1;
                    arr[str[i] - 'A'] = i;
                }

            }
        }

        cout << max(maxi, count1) << endl;


    }

    return 0;

}
