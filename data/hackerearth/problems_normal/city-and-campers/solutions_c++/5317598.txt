#include<bits/stdc++.h>
#define N 100005
using namespace std;
long long arr[N],n,minn,maxx,q,a,b,size[N];
set<pair<long long, long long> > st;
long long root(long long x)
{
	while(x!=arr[x])
	{
		arr[x]=arr[arr[x]];
		x=arr[x];
	}
	return x;
}
long long _find(long long x,long long y)
{
	long long root_x=root(x);
	long long root_y=root(y);
	if(root_x!=root_y)
	{
        arr[root_y]=arr[root_x];
		st.erase(make_pair(size[root_x],root_x));
		st.erase(make_pair(size[root_y],root_y));
		size[root_x]+=size[root_y];
		st.insert(make_pair(size[root_x],root_x));
	}
    set<pair<long long, long long> >::iterator it=st.begin();
    set<pair<long long, long long> >::iterator itt=st.end();
    itt--;
    return (*itt).first-(*it).first; 
}
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>n>>q;
	for(int i=1;i<=n;i++)
	{
		arr[i]=i;
        size[i]=1;
		st.insert(make_pair(size[i],i));
	}
	for(int i=0;i<q;i++)
	{
		cin>>a>>b;
		cout<<_find(a,b)<<endl;
	}
	return 0;
}
