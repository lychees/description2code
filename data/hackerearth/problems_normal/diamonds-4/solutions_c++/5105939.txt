#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int y,x;
    	cin>>y>>x;
    	char diamond[y][x];
    	for(int i=0; i<y; i++)
    	{
    		for(int j=0; j<x; j++)
    		{
    			cin>>diamond[i][j];
    		}
    	}
    	int diamondCount=0;
    	for(int i=0; i<y; i++)
    	{
    		for(int j=0; j<x; j++)
    		{
    			if(diamond[i][j]=='/')
    			{
    				if(diamond[i][j+1]=='\\' && j<x-1)
    				{
    					if(diamond[i+1][j]=='\\' && i<y)
    					{
    						if(diamond[i+1][j+1]=='/')
    						{
    							diamondCount++;
    						}
    					}
    				}
    			}
    			
    		}
    	}
    	cout<<diamondCount<<endl;
    }
}
