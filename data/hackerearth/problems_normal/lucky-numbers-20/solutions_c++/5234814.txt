#include<cstdio>
#include<iostream>
#include<cstdlib>
using namespace std;
#define size 65536
#define MODULO_NUM 1000000007

int main()
{
    int no;
    scanf("%d\n",&no);
	while (no--) {
		long long int n;
		scanf("%lld", &n);
		long long int result = 0;
		for (long long int i=1; i<=n; i=i*2) {
			for (long long int j=i*2; j<=n-i; j=j*2) {
				if (i==j)
					continue;
				if (i+j <= n) {
					result = (result+i+j)%(MODULO_NUM);
				} else {
					break;
				}
			}
		}
		cout<<result<<"\n";
	}    
    return 0;
}
