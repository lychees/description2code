string s;
struct data
{
	char c;
	int x;
}d[111];
bool cmp(data d1,data d2)
{
	return d1.x<d2.x||((d2.x==d1.x)&&d1.c<d2.c);
}
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int i,n;
	int T;
	string re;
	cin>>T;
	while(T--)
	{
		cin>>s;
		n=s.size();
		for(i=0;i<26;i++)
			d[i].c='a'+i;
		for(i=0;i<26;i++)
			d[i].x=0;
		for(i=0;i<n;i++)
			d[s[i]-'a'].x++;
		sort(d,d+26,cmp);
		re="";
		for(i=0;i<26;i++)
			while(d[i].x--)
				re+=d[i].c;
		cout<<re<<endl;
	}
//	system("pause");
	return 0;
}
