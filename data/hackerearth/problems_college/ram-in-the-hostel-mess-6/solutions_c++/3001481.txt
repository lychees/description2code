#include<bits/stdc++.h>
#include<stdlib.h>
#include<math.h>
using namespace std;
int fact (int n, int r) {
	if(n==0 || n==1 || n==r)
		return 1;
	else
		return n*fact(n-1,r);	
}
 
 
int main (){
	int t;
	cin>>t;
	while(t--) {
	 	int n,i;
		cin>>n;
		long long int sol = 1+n;
		for(i=2;i<=(n+1)/2;i++) {
			int N = n - 2*i + 1;
			int R = i+1;
			if(N > R-1) 
				sol += fact(N+R-1,N)/fact(R-1,1);
			else 
				sol += fact(N+R-1,R-1)/fact(N,1);
		}
		cout<<sol*sol<<endl;
	}
	return 0;
}
