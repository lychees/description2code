#include<stdio.h>

int main(){
	int t, n, c, no, i, j; scanf("%d", &t); int a[1001];
	for(i = 2; i <= 1001; i++)	a[i] = i;
	for(i = 2; i <= 1001 / 2; i++)	for(j = 2; i * j <= 1001; j++)	a[i * j] = 0;
	while(t-- && scanf("%d", &n)){
		c = 0;
		while(n-- && scanf("%d", &no))	if(no >= 2 && a[no])	c++;
		printf("%d\n", c);
	} 		
	return 0;
}
