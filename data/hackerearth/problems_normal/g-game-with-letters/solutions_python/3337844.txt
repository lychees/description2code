for _ in range(int(raw_input())):
	n,s=map(str,raw_input().split())
	n=int(n)
	h=[0 for x in range(26)]
	
	for i in range(len(s)):
		h[ord(s[i])-ord('a')]+=1
	m=-1
	a=0
	ml=0
	for i in range(n):
		s=raw_input()
		l=len(s)
		ans=0
		hs=[0 for x in range(26)]
		for j in range(len(s)):
			hs[ord(s[j])-ord('a')]+=1
		for j in range(26):
			ans+=h[j]*hs[j]
		if m<ans:
			m=ans
			a=i
			ml=l
		elif m==ans and l<ml:
			a=i
			ml=l
	print a+1
