for _ in range(input()):
    s = list(raw_input())
    x, y = 0, 0
    for i in range(len(s)):
        if s[i] == "X":
            x+=1
        elif s[i] == "Y":
            y += 1
    if x == len(s) or y == len(s):
        print len(s)-1
    else:
        cnt = 0
        for i in range(len(s)-1):
            if s[i] == s[i+1]:
                cnt+=1
        print cnt
