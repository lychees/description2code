#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,n,i,j;
	bool isprime[100009];
	for (i = 2; i <= 100009; i++) {
			isprime[i] = true;
	}
	for (i = 2; i*i <= 100009; i++) {
		if (isprime[i]) {
			for (j = i*i; j <= 100009; j+=i) {
				isprime[j] = false;
			}
		}
	}
	scanf("%d",&t);
	while (t --) {
		scanf("%d",&n);
		if (n == 0) {
			printf("4\n");
			continue;
		}
		for (i = n+1; i < 1000000; i++) {
			if (isprime[i]) {
				continue;
			} else {
				printf("%d\n",i);
				break;
			}
		}
	}
	
	return 0;
}
