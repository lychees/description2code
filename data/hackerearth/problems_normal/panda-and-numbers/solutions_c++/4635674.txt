#define IOS ios::sync_with_stdio(false)
int panda[1000005];
int dp[1000005];
ll pow1[10];
void calc_pow1(){
    for(int i = 1 ; i <= 9 ; i++){
        ll val = 1;
        for(int j = 1 ; j <= i ; j++){
            val = val*i;
            if(val>=1000002){
                val = -1;
                break;
            }
        }
        pow1[i] = val;
        //cout << pow1[i] << endl;
    }
}
void calc_panda(){
    int i , j , k;
    panda[1] = dp[1] = 1;
    i = 2 ;
    while(true){
        ll val = pow1[i];
        if(val == -1)break;
        panda[val] =1;
        dp[val] = 1;
        i++;
    }
}
void solve(){
    int i , j , k , dig;
    for(i = 2 ; i <= 1000002 ; i++){
        int temp = i;
        while(temp > 0){
            dig = temp%10;
            temp/=10;
            ll val  = dig*dig; 
            if(val <= i ){
                dp[i] = dp[i]||dp[i-val];
            }
        }
    }
}
int main(){

    IOS;
    cin.tie(0);
    ll i , j , k  , t , n;
    calc_pow1();
	calc_panda();
    solve();
    cin >> t;
    while(t--){
        cin >> n;
        if(dp[n])cout << "Yes\n";
        else cout << "No\n";
    }
    //for(i = 1 ; i <= 13 ; i++)cout << dp[i] << endl;
	return 0;
}
