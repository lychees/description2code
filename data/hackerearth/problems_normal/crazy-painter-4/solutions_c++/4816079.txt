#include <iostream>
#include <vector>
using namespace std;

int main() {
    int T;
    cin >> T;
    for(int i = 0; i < T; i++) {
        long long N;
        cin >> N;
        long long nodes = 10*(N-1) + 10 + 2*N;
        vector<long long> sums;
        long long cur_sum = 0;
        for(int i = 0; i < 26; i++) {
            long long tmp;
            cin >> tmp;
            cur_sum += tmp;
            sums.push_back(cur_sum);
        }
        long long cycles = nodes / 26;
        long long remainder = nodes % 26;
        long long soln_cost = cur_sum * cycles;
        if(remainder > 0) {
            soln_cost += sums[remainder - 1];
            if(cycles > 0) {
                soln_cost += cycles * remainder;
            }
        }
        soln_cost += 13*(cycles * (cycles - 1));
        cout << soln_cost << endl;
    }
}
