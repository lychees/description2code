template < typename T > T abs(T x)
{
	return x > 0 ? x : -x;
}

template < typename T > T sqr(T x)
{
	return x * x;
}

vector <int> ans;
int a[100005], maxl[100005], maxr[100005];

int main()
{	
	int n;
	
	cin >> n;
	
	for (int i = 1; i <= n; i++)
		scanf("%d", a + i);
		
	for (int i = 1; i <= n; i++)
		maxl[i] = max(maxl[i - 1], a[i]);
		
	for (int j = n; j >= 1; j--) {
		maxr[j] = max(maxr[j + 1], a[j]);
	}
	
	for (int i = 1; i <= n; i++) {
		if (maxl[i - 1] < a[i] || maxr[i + 1] < a[i])
			ans.pb(i);
	}	
	
	for (int i = 0; i < ans.size(); i++) {
		printf("%d ", ans[i]);
	}
	
	return 0;	
}
