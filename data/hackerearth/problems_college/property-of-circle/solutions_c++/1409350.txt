#include<iostream>
#include<string.h>
#include<cstdio>
#include<string>
#include<math.h>
using namespace std;

int main()
{
    int task,p=0,n;
    scanf("%d",&task);
    while(task--)
    {
        float x,x1,y,y1;int r,r1;
        cin>>x>>y>>r;
        cin>>x1>>y1>>r1;
        float d=(sqrt((x-x1)*(x-x1)+(y-y1)*(y-y1)));
        if(x==x1&&y==y1&&r==r1)
            cout<<"C2~C1\n";
        else if(d<=r-r1)
            cout<<"C1CC2\n";
            else
            if(d<=r1-r)
            cout<<"C2CC1\n";
        else
            cout<<"NOT SATISFIED\n";

    }
}
