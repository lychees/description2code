#include<iostream>
using namespace std;
int main(){
	int q;
	cin>>q;
	int one=1;
	int three=3;
	int five=5;
	int a;
	while(q--){
	cin>>a;
	int temp=0;
	for(int i=0;i<=a;i++){
		for(int j=0;j<=a;j++){
			for(int k=0;k<=a;k++){
				if((one*k + three *j +five *i)==a){
				    temp++;
				    break;
				}
			}
		}
	}
	cout<<temp<<endl;		
	}
}
