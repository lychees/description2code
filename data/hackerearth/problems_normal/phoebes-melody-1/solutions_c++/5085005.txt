#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    //cout << "Hello World!" << endl;
    long long t;
    cin>>t;
    while(t--)
    {
    	long long n,k,i,j,flag=0,x,y,m;
    	cin>>n>>k;
    	long long a[n+1];
    	for(i=1;i<=n;i++)
    	cin>>a[i];
    	long long ans=0;
    	for(i=1;i<=n;i++)
    	{
    		x=k;
    	   while(i+x<=n || i-x>=1)
    		{
    			if(i+x<=n && a[i]>a[i+x])
    			{
    				ans +=x;
    				break;
    			}
    			if(i-x>=1 && a[i]>a[i-x])
    			{
    				ans+=x;
    				break;
    			}
    			++x;
    	}
    	}
    	cout<<ans<<endl;
 
    }
    return 0;
}
