#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t>0){
        t--;
        int n;
        cin>>n;
        int arr[n];
        int count=0;
        for(int i=0;i<n;i++)
            cin>>arr[i];
        int hash[10]={0};
        for(int i=0;i<n;i++){
            hash[arr[i]%10]++;
        }
        for(int i=0;i<10;i++){
            if(hash[i]>1)
                count+=hash[i]-1;
        }
        cout<<count<<'\n';
    }
    return 0;
}
