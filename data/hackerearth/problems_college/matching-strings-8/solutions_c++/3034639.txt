#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	string s, t;
    	cin >> s >> t;
    	int ans = 0;
    	int n = s.size(), m = t.size(), i = 0;
    	while(i < n && i < m) {
    		if(s[i] == t[i]) {
    			ans++;
    		}
    		i++;
    	}
    	cout << ans << endl;
    }
    return 0;
}
