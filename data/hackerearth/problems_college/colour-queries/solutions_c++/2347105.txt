#include <iostream>
#include <cstdio>
#include <vector>
#include <cmath>
using namespace std;
int n,m,total;
int out(int a,int b)
{
	if(a<0||b<0||a>=n||b>=m)
	{
		return 0;
	}
	else{
		return 1;
	}
}

int count(int x, int y, vector< vector<int> >&matrix,vector< vector<int> >&duplicate,int d)
{
	if(duplicate[x][y]==0)
	{
		total++;
		duplicate[x][y]=1;
	}
	if(out(x-1,y)&&abs(matrix[x][y]-matrix[x-1][y])<=d&&duplicate[x-1][y]==0)
	{
		total++;
		duplicate[x-1][y]=1;
		count(x-1,y,matrix,duplicate,d);
	}
	if(out(x,y-1)&&abs(matrix[x][y]-matrix[x][y-1])<=d&&duplicate[x][y-1]==0)
	{
		total++;
		duplicate[x][y-1]=1;
		count(x,y-1,matrix,duplicate,d);
	}
	if(out(x,y+1)&&abs(matrix[x][y]-matrix[x][y+1])<=d&&duplicate[x][y+1]==0)
	{
		total++;
		duplicate[x][y+1]=1;
		count(x,y+1,matrix,duplicate,d);
	}
	if(out(x+1,y)&&abs(matrix[x][y]-matrix[x+1][y])<=d&&duplicate[x+1][y]==0)
	{
		total++;
		duplicate[x+1][y]=1;
	count(x+1,y,matrix,duplicate,d);
	}
	return 0;

}

					duplicate[i][j]=0;
				}
			}	
		cin>>x>>y>>d;
