#include <bits/stdc++.h>
using namespace std;
int hs[704978];
vector<int> cube;

int main()
{
	for(int i = 1; i*i*i <= 704977; ++i){
		cube.push_back(i*i*i);
	}
	for(int i = 0; i < cube.size(); ++i){
		for(int j = i+1; j < cube.size(); ++j){
			int tmp = cube[i] + cube[j];
			if(tmp <= 704977){
				hs[tmp]++;
			}
		}
	}
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int ans = -1;
		for(int i = n-1; i >= 0; --i){
			if(hs[i] > 1){
                ans = i;
				break;
			}
		}
		cout<<ans<<endl;
	}
    return 0;
}
