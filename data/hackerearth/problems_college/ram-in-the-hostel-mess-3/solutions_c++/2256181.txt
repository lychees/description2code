#include<bits/stdc++.h>
using namespace std;
long long int f[10000];
int main()
{
    int t;
    cin>>t;
    f[1]=2;
    f[2]=3;
    while(t--)
    {
        long long int n;
        cin>>n;
     for(int i=3;i<=n;++i)
     {
         f[i]=f[i-1]+f[i-2];
     }
     n=f[n]*f[n];
     cout<<n<<endl;
    }
    return 0;
}
