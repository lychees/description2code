#include <iostream>
using namespace std;

int vis[1001][1001],a[1001][1001],n;

int valid(int x,int y)
{
	if(x>=1 && x<=n && y>=1 && y<=n)
	return 1;
	return 0;
}	

int func(int x,int y)
{
	if(x==n && y==n && a[n][n]==0)
	return 1;
	int ans=0;
	vis[x][y]=1;
	if(valid(x-1,y) && vis[x-1][y]!=1 && a[x-1][y]==0)
	ans+=func(x-1,y);
	if(valid(x+1,y) && vis[x+1][y]!=1 && a[x+1][y]==0)
	ans+=func(x+1,y);
	if(valid(x,y+1) && vis[x][y+1]!=1 && a[x][y+1]==0)
	ans+=func(x,y+1);
	if(valid(x,y-1) && vis[x][y-1]!=1 && a[x][y-1]==0)
	ans+=func(x,y-1);
	vis[x][y]=-1;
	return ans;
}

int main() 
{
	int i,j;
	cin>>n;
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		cin>>a[i][j];
	}
	if(a[n][n]==1)
	{
		cout<<"0";
		return 0;
	}
	cout<<func(1,1);
	return 0;
}
