}

const int MAXN = 10000005;
int p[MAXN], pr[MAXN], pn, cnt[MAXN];
LL sum[MAXN];

int main() {
	
	for (int i = 2; i < MAXN; i++) p[i] = i;
	for (int i = 2; i < MAXN; i++) {
		if (p[i] == i) pr[pn++] = i;
		int t;
		for (int j = 0; j < pn && (t = pr[j] * i) < MAXN; j++){
			p[t] = pr[j];
			if (i % pr[j] == 0) break;
		}
	}
	for (int i = 2; i < MAXN; i++) {
		int x = i;
		sum[i] = sum[i - 1];
		cnt[i] = 1 + cnt[i / p[i]];
		sum[i] += cnt[i];
	}
	int ncase;
	for (scanf("%d", &ncase); ncase--;) {
		int n;scanf("%d", &n);
		printf("%lld\n", sum[n]);
	}
	return 0;
}
