#define FORI(i,n) REP(i,1,n)

ll len(ll a) {
	for (int i = 2; i<a; i++) if (a%i!=0) return 1+len(i);
	return 2;
}

ll f(ll X) {
	if (X<2) return 0;
	if (X==2) return 1;
	ll done = 0, excl = 1, res = 0;
	for (ll a = 2; excl <= X; a++) {
		// # divisible by excl
		ll cur = X/excl;
		// every bth is divisible by a
		ll b = a/__gcd(excl, a);
		// there are k not divisible by a
		ll k = cur-cur/b;
		done += k;
		res += k * len(a);
		excl *= b;
		
		//printf("a=%lld, cur = %lld, k = %lld, done = %lld, res = %lld, len = %lld\n", a, cur, k, done, res, len(a));
	}
	return res-4;
}

int main() {
	ll A,B;
	scanf("%lld%lld", &A, &B);
	printf("%lld\n", f(B)-f(A-1));
}
