	long long m,n,x,y,z,ans,i,b[27],temp,mod=1000000007;
long long factorial(long long n);
long long power(long long x,long long y){
	long long ans=1;
	while(y>0){
		if(y%2==1){
			ans=(ans*x)%mod;
		}
		x=(x*x)%mod;
		y=y/2;
	}
	return ans;
}

int main()
{   long long t;
scanf("%lld",&t);
while(t--){

	char a[1000009];
	for(i=1;i<=26;i++)b[i]=0;
    scanf("%s",&a);
    n=strlen(a);
    m=0;
    for(i=0;i<n;i++){
    	if(a[i]=='a'||a[i]=='e'||a[i]=='i'||a[i]=='o'||a[i]=='u')
    		m++;
    	b[a[i]-96]++;
    }
    if(m > (n-m+1))
    	printf("-1\n");
    else {
    	z=factorial(n-m);
    	z=z%mod;
    	x=(z*z)%mod;
    	y=factorial(n-(2*m)+1);
    	y=y%mod;
    	
    	ans=(x*(n-m+1));
    	ans=ans%mod;
    	ans=(ans*power(y,mod-2))%mod;
    	temp=1;
    		for(i=1;i<=26;i++){
    		temp *= factorial(b[i]);
    		temp=temp%mod;
    		ans=ans*power(temp,mod-2);
    		ans=ans%mod;
    		temp=1;
    	}
    	ans=ans%mod;
    	printf("%lld\n",ans);
    }
}
    return 0;
}
long long factorial(long long n)
{
  long long c;
  long  long result = 1,mod=1000000007;
 
  for (c = 1; c <= n; c++)
    result = (result * c)%mod;
 
  return result;
}
