#include <bits/stdc++.h>
using namespace std;
#define MAX 10000
int main()
{
    int tc,N;
    long long a[MAX];
    
    cin >> tc;
    while(tc--) {
    	cin >> N;
    	for(int i = 0 ;i < N;i++) 
    		cin >> a[i];
    		
    	int asort = 1;
    	for(int i = 1; i < N ;i++) {
    		if(a[i] < a[i-1])	{
    			asort = 0;
    			break;
    		}
    	}
    
    	int start = 0, end = 0, pass = 0;
    	for(int i = 1; i < N; i++) {
    		if(a[i] < a[i-1]) {
    		end++;
    		pass++;
    		}
    		else{
    			if(pass > 0 ) break;
    			start = end = i;
    		}
    	}
    	int temp;
    	int first = start,second = end;
    	
    		while(first < second) {
    		temp = a[first];
    		a[first] = a[second];
    		a[second] = temp;
    		first++;
    		second--;
    	}
    	int flag = 1;
    		
    	for(int i = 1; i < N ;i++) {
    		if(a[i] < a[i-1])	{
    			flag = 0;
    			break;
    		}
    	}
    	if(flag && !asort)
    		cout << start+1 << " " << end+1 << endl;
    	else
    		cout <<"-1 -1" << endl; 
    }	
    return 0;
}
