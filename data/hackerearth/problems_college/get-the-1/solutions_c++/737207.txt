#include <iostream>
using namespace std;


    int main()
{
    int T,i;
    cin>> T;
    int TC[T];
    int sum = 0;
  
  for(i=0; i<T; i++)
  cin >> TC[i];
  
  for (i=0; i<T; i++)
  {
 
  	here:
  	while(TC[i]>0)
  	{  		
  		int z = TC[i]%10;
  		sum = sum + ((z)*(z));
  		TC[i] = TC[i]/10;  		
  		}
  		
  		TC[i] = sum;
  		sum=0;
  	
  	if ((TC[i]!= 16) && (TC[i] != 1)) goto here;
  		
  	
  	if (TC[i] == 1 ) cout << "Case #" << i+1 << ":YES\n";
  	else if (TC[i] == 16) cout << "Case #"<< i+1<< ":NO\n";
  	
  	
  }
  
    return 0;
}
