#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k;
    	cin>>n>>k;
    	int a, b;
    	if(k>81)
    		cout<<-1<<endl;
    	else
    	{
    		k--;
			a = k/9;
			a++;
			b = k%9;
    		if(a<=b)
    			b++;
	    	for(int i=0;i<n;i++)
	    	{
	    		if(i%2==0)
	    			cout<<a;
	    		else
	    			cout<<b;
	    	}
	    	cout<<endl;
    	}
    }
    return 0;
}
