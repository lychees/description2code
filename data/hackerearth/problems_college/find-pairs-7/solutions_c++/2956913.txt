#include <iostream>
using namespace std;

int main()
{
    int test;
    cin>>test;
    for(int q=0;q<test;q++)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	int count=n;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<n;j++)
    		{
    			if(i!=j&&j>=i&&a[i]==a[j])
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
