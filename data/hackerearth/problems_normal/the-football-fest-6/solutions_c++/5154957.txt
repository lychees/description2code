#include <iostream>
#include <stack>
#include <climits>
#include <cassert>
#define MAXID 1000000
using namespace std;
stack<int> past;
stack<int> present;
void output(int n, int id, char pass, int passId){
	for(int i=0;i<n;++i){
		if(present.empty() and pass=='P'){
			present.push(passId);
			//cout << "instant ball : " << present.top() << endl;
			return;
		}
		if(pass=='P'){
			past.push(present.top());
			present.push(passId);
			//cout << "instant ball : " << present.top() << endl;
			return;
		}
		else if(pass=='B'){
			int t1=present.top();
			int t2=past.top();
			past.push(t1);
			present.push(t2);
			//cout << "instant ball : " << present.top() << endl;
			return;
		}
	}
}
int main(){
	int t;
	cin >> t;
	int n;
	int id;
	char pass;
	int passId;
	for(int i=0;i<t;++i){
		cin >> n >> id;
		past.push(id);
		for(int j=0;j<n;++j){
			cin >> pass;
			if(pass=='P'){
				cin >> passId;
				output(n, id, pass, passId);
			}
			else if(pass=='B'){
				passId=INT_MAX;
				output(n, id, pass, passId);
			}

		}
		cout << "Player " << present.top() << endl;
	}
	return 0;
}
