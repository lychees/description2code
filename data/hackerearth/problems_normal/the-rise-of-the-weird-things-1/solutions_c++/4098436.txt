#include <iostream>
using namespace std;
void insertion_sort(int a[],int n)
{
  int i,j,key;
  for(j=1;j<n;j++)
  {
    key=a[j];
    i=j-1;
    while ((i>=0)&&(a[i]>key)) {
      a[i+1]=a[i];
      i--;
    }
    a[i+1]=key;
  }
}
int main()
{ int n;
  cin>>n;
  int A[n];
  int ctr_even=0;
  int sum_even=0;
  int ctr_odd=0;
  int sum_odd=0;
  for(int i=0;i<n;i++)
  {
    cin>>A[i];
    if(A[i]%2==0)
    {
      ctr_even++;
      sum_even=sum_even+A[i];
    }
    else{
      ctr_odd++;
      sum_odd=sum_odd+A[i];
    }
  }
  int Even[ctr_even];
  int Odd[ctr_odd];
  int j=0;
  for(int i=0;i<n;i++)
  { if(A[i]%2==0)
    {
      Even[j]=A[i];
      j++;
    }
  }
  j=0;
  for(int i=0;i<n;i++)
  {
    if(A[i]%2!=0)
    {
      Odd[j]=A[i];
      j++;
    }
  }
  insertion_sort(Even,ctr_even);
  insertion_sort(Odd,ctr_odd);
  int B[n+2];
  int k=0;
  for(k=0;k<ctr_even;k++)
  {
    B[k]=Even[k];
  }
  B[k]=sum_even;
  j=0;
  for(int z=k+1;z<n+1;z++)
  {
    B[z]=Odd[j];
    j++;
  }
  B[n+1]=sum_odd;
  for(int i=0;i<n+2;i++)
  {
    cout<<B[i]<<" ";
  }
  return 0;
}
