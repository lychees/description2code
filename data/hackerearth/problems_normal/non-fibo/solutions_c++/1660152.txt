#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    int t; scanf("%d",&t);
    while(t--)
    {
        ll n; scanf("%lld",&n);
        ll a=1,b=2,c=3;
        while(n>0)
        {
            a=b;b=c;c=a+b;
            n-=c-b-1;
        }
        n+=c-b-1;
        printf("%lld\n",n+b);
    }
    return 0;
}
