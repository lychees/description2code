#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define MAX 1000005

int primes[MAX];
long long arr[MAX];
void gen_primes()
     {
       int i,j;
       for(i=0;i<MAX;i++) primes[i] = 1;
       primes[0] = 0;
       primes[1] = 0;
       for(i=2;i<=(int)sqrt(MAX);i++)
         if (primes[i])
           for(j=i;j*i<MAX;j++) primes[i*j] = 0;
     }

bool isPrime (long long n)
{
   if (n<=1) return false;
   if (n==2) return true;
   if (n%2==0) return false;
   for (long long i = 3; i * i <= n; i+=2)
      if (n%i==0)
         return false;

   return true;
}
int main()
{
    gen_primes();
    arr[0] = 0;
    for(int i = 1; i < MAX; i++) {
        if(primes[i]) {
            arr[i] = arr[i - 1] + (long long)i;
        }
        else {
            arr[i] = arr[i - 1];
        }
    }
    int t;
    scanf("%d", &t);
    while(t--) {
        int a, b;
        scanf("%d%d", &a, &b);
        long long ans = arr[b] - arr[a - 1];
        if(ans == 0) {
            puts("NO");
        }
        else if(isPrime(ans)) {
            puts("YES");
        }
        else {
            puts("NO");
        }
    }
    return 0;
}
