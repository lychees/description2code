#include<iostream>
using namespace std;
int main(){
	
	int t;
	cin>>t;long x;
	for(int i=0;i<t;i++){
		cin>>x;
		int y=x%3;
		if(y==0){
			for(long j=0;j<x;j++)cout<<"5";
			cout<<endl;
		}
		if(y==1){
			if(x<10)cout<<"-1"<<endl;
			else{
				long f=x-10;
				for(long j=0;j<f;j++){
					cout<<"5";
				}
				cout<<"3333333333"<<endl;
				
			}
		}
		if(y==2){
			if(x==2)cout<<"-1"<<endl;
			else{
				long f=x-5;
				for(long j=0;j<f;j++)cout<<"5";
				cout<<"33333"<<endl;
			}
		}
	}
}
