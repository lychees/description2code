#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007 
#define ll long long
int main()
{
  //  cout << "Hello World!" << endl;
    ll t;
    cin>>t;
    vector<ll> dp(100005,0);
    dp[0]=1;
    dp[1]=1;
    dp[2]=2;
    dp[3]=4;
    for(ll i=4;i<100005;i++)
    {
    	dp[i]=dp[i-1]+dp[i-2];
    	dp[i]%=mod;
    	dp[i]+=dp[i-3];
    	dp[i]%=mod;
    	//dp[i]=(dp[i-1]+dp[i-2]+dp[i-3])%mod;
    }
    while(t--)
    {
    	ll query;
    	cin>>query;
    	cout<<dp[query]<<endl;
    }
    return 0;
}
