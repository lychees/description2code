#include<bits/stdc++.h>
using namespace std;
#define pdn(n) printf("%d\n",n)
#define pln(n) printf("%ld\n",n)
#define plln(n) printf("%lld\n",n)
#define pn(n) printf("\n")
#define psn(n) printf("%s\n",n)

#define s2ll(n1,n2) scanf("%lld%lld",&n1,&n2)
#define sll(n) scanf("%lld",&n)
#define sl(n) scanf("%ld",&n)
#define sd(n) scanf("%d",&n)
#define ss(n) scanf("%s",n)

typedef long long LL;

int main()
{
    LL t,a[1001],b[1001];
    cin>>t;

    a[0]=b[0]=0;
    a[1]=b[1]=1;
    a[2]=2,b[2]=0;
    a[3]=3,b[3]=1;

    for(LL i=4 ; i<1001 ; i++)
    {
        if(i%4==0)
        {
            a[i]=a[i-1]-1;
            b[i]=b[i-1]+1;
        }

        if(i%4==1 || i%4==3)
        {
            a[i]=a[i-1]+1;
            b[i]=b[i-1]+1;
        }

        if(i%4==2)
        {
            a[i]=a[i-1]+1;
            b[i]=b[i-1]-1;
        }

    }

    while(t--)
    {
        int x,y,flag=1;
         cin>>x>>y;

         for(LL i=0 ; i<3000 ; i++)
         {
             if(a[i]>(x+10))
             {
                 flag=0;
                 break;

             }

             if(a[i]==x && b[i]==y)
                 {
                     cout<<i<<endl;
                     break;
                 }
         }

         if(flag==0)
            cout<<"No Number"<<endl;
    }

return 0;
}
