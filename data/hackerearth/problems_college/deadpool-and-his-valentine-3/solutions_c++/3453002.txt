#include <iostream>
using namespace std;
#define ll long long
#include <vector>
#include <algorithm>

int main()
{	ll a,b,z[150]={0},x,q,k=1,i;
    cin>>a>>b;
    while(cin>>x)
    {	while(x--)
    	{	cin>>q;
    		z[q]++;
    	}
    }
    for(i=1;i<=b;++i)
    {	if(z[i]==0)
    	{	k=0;
    		break;
    	}
    }
    if(k)
    	cout<<"YES\n";
    else
    	cout<<"NO\n";
    return 0;
}
