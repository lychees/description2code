t = int(raw_input())
while t > 0:
    n = int(raw_input())
    count = 0
    ans = [0]*1000001
    while n > 0:
        a, b = map(int, raw_input().split(" "))
        ans[a] += 1
        n -= 1
        if (ans[b] == 0):
            count += 1
        else:
            ans[b] -= 1
    print count
    t -= 1
