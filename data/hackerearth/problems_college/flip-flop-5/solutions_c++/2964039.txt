#include<bits/stdc++.h>
#include<algorithm>
#include<cstring>
using namespace std;
main()
{
	int t,i;
	long long count,l;
	char a[100001];
	scanf("%d",&t);
	while(t--)
	{
		count=0;
		scanf("%s",a);
		l=strlen(a);
	//	sort(a,a+l);
		for(i=0;i<l-1;i++)
		{
			if(a[i]==a[i+1])
			
			count++;
		}
		printf("%d\n",count);
	}
}
