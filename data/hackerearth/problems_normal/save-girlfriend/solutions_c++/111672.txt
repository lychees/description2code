#include <stdio.h>
#include <string.h>1
#include <vector>
using namespace std;
int sieve[10005];
vector<int> primes;
int sol[10005];
int main()
{
int t,a,b,max,i,j,count;
scanf("%d",&t);
while(t--)
{
max=0;
count=0;
memset(sol,0,sizeof(sol));
scanf("%d %d",&a,&b);//inputting a and b
for(i=1;i*i<=b;i+=1)
{
for(j=(i-a%i)%i+a;j<=b;j+=i)
{
if(j/i==i)
sol[j-a]+=1;
else if(j/i>i)
sol[j-a]+=2;
}
}
max=0;
for(i=0;i<=(b-a);i++)
{
if(max<sol[i])
{
max=sol[i];
count=1;
}
else if(max==sol[i])
count++;
}
printf("%d\n",count+1);
}
//system("pause");
return 0;
}
