ip = raw_input()

m = ip.split(' ')
p = int(m[0])
q = int(m[1])

matrix = []

for i in range(p):
    x = raw_input().split(' ')
    k = []
    for j in range(q):
        k.append(int(x[j]))
    matrix.append(k)

rowsum = []
colsum = []

for i in range(p):
    rowsum.append(sum(matrix[i]))

for i in range(q):
    colsum.append(sum(row[i] for row in matrix))

xor = []

for i in range(len(rowsum)):
    for j in range(len(colsum)):
        xor.append(rowsum[i] ^ colsum[j])

print max(xor)
