#include <iostream>
using namespace std;
 bool coprime(int u, int v)
{
    if (((u | v) & 1) == 0) return false;

    while ((u & 1) == 0) u >>= 1;
    if (u == 1) return true;

    do
    {
        while ((v & 1) == 0) v >>= 1;
        if (v == 1) return true;

        if (u > v) { long t = v; v = u; u = t; }
        v -= u;
    } while (v != 0);

    return false;
}
int main()
{
    int n,i,j,count=0,a,b,t;
    cin>>n;
    int array[n];
    for(i=0;i<n;i++){
    	cin>>array[i];
    }
    for(i=0;i<n;i++){
    	for(j=i+1;j<n;j++){
    		if(array[j]<array[i]){
    			//cout<<"in if loop\n";
    			a=array[i];
    			b=array[j];
    			if(coprime(a,b))
    			count+=1;
    		}
    	}
    }
    cout<<count<<endl;
    return 0;
}
