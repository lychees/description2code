ll powmod(ll a,ll b)
{
	if(b==0)return 1;
	ll x=powmod(a,b/2);
	ll y=(x*x)%MOD;
	if(b%2)
		return (a*y)%MOD;
	return y%MOD;
}

vi g[N];
ll sub[N];
bool vis[N];
void dfs(int u)
{
	vis[u]=true;
	sub[u]++;
	REP(i,0,sz(g[u]))
	{
		int v=g[u][i];
		if(!vis[v])
		{
			dfs(v);
			sub[u]+=sub[v];
		}
	}
}
