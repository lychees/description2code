#include<bits/stdc++.h>
using namespace std;

map< long long, long long > h;
long long coins(long long n)
{
  if(n==0){
    return 0;
  }

  if(h[n]){

    return h[n];
  }
  else{
    h[n] = max(n, coins(n/2) + coins(n/3) + coins(n/4));
  }
  return h[n];
}

int main(){

  long long n;
  while(cin>>n){
   // cout<<n<<endl;
    cout<<coins(n)<<endl;
  }
return 0;
}
