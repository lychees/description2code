#include <iostream>
using namespace std;
typedef long long ll;
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	ll c,sf,bf,f,sfs,bfs,fs,sum,sp,bfp,fp;int flag=0;
    	cin>>c>>sf>>bf>>f;
    	sfs=sf*100;
    	bfs=bf*50;
    	fs=f*25;
    	sum=bfs+sfs+fs;
    	sp=sfs;
    	bfp=bfs;
    	fp=fs;
    	while(sum<c)
    	{
    		ll left=c-sum;
    	//	cout<<left<<endl;
    		if(left>=sfs)
    		{
    			left-=sfs;
    			sum+=sfs;
    			sp+=sfs;
    			flag=1;
    		}
    		if(left>=bfs)
    		{
    		left-=bfs;
    		sum+=bfs;
    		bfp+=bfs;
    		flag=1;
    		}
    		if(left>=fs)
    		{
    			left-=fs;
    			sum+=fs;
    			fp+=fs;
    			flag=1;
    			//cout<<fp;
    		}
    		else
    		{
    			flag=0;
    			goto a;
    		}
    	}
    	a:
    	if(flag==0)
    	cout<<"No"<<endl;
    	else
    	cout<<sp<<" "<<bfp<<" "<<fp<<endl;

    }
    return 0;
}
