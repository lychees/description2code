#include <bits/stdc++.h>
using namespace std;

int gcd(int a, int b)
{
	int t;
	while(b)
	{
		t = a%b;
		a = b;
		b = t;
	}
	return a;
}

int main()
{
    int T, x, y;
    cin>>T;
    while(T--)
    {
    	cin>>x;
    	cin>>y;
    	y += x;
    	int g = gcd(x, y);
    	cout<<y/g<<endl;
    }
    return 0;
}
