#include <iostream>
#include <cstdio>
#include <map>

using namespace std;

#define M 1000000007

long long factorial[3500 * 3500];

void findFac()
{
	factorial[0] = 1;

	for(int i = 1; i < (3500 * 3500); i++) {
		factorial[i] = ((factorial[i - 1] % M) * (i % M)) % M;
	}
}

int main()
{
	findFac();
	int t;
	scanf("%d", &t);
	while(t--) {
		int n, m;
		scanf("%d %d", &n, &m);
		
		std::map <int, int> count;
		
		for(int i = 1; i <= n; i++) {
			for(int j = 1; j <= m; j++) {
				count[i | j]++;
			}
		}
		
		long long ans = 1;
		
		std::map <int, int> :: iterator it;
		
		for(it = count.begin(); it != count.end(); it++) {
			ans = ((ans % M) * (factorial[it->second] % M)) % M;
		}
		
		printf("%lld\n", ans);
	}
    return 0;
}
