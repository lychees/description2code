    #include <iostream>
    #include <algorithm>
     
    using namespace std;
     
    int main() {
        int n;
        cin >> n;
        while (n--) {
            string first, second;
            cin >> first >> second;
            if (first.size() > second.size()) swap(first, second);
            int ans = first.size();
            for (int i = 0; i < second.size() - first.size() + 1; i++) {
    			int d = 0;
    			for (int j = 0; j < first.size(); j++) {
    				if (first[j] != second[j + i]) d++;
    			}
    			ans = min(ans, d);
    		}
    		cout << ans<< endl;
        }
        return 0;
    }
