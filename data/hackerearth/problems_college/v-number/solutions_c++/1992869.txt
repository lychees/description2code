#include<bits/stdc++.h>
using namespace std;
int main()
{
	int T;
	long int N,i,j,k,sum1,sum2;
	string S;
	cin>>T;
	while(T--)
	{
		cin>>S;
		sum1=sum2=0;

        j=S.length()-1;
		for(i=0;i<(S.length()/2);i++)
        {
            sum1+=S[i]-48;
            sum2+=S[j--]-48;
        }
        if(sum1==sum2)
		cout<<"YES\n";
		else
		cout<<"NO\n";

	}
}
