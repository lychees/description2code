def digits_sum(number):
    number = str(number)
    while len(number)>1:
        partial = 0
        for el in number:
            partial = partial + int(el)
        number = str(partial)
    return int(number)

from sys import stdin
numbers = stdin.read().split()
for i in range(1,len(numbers)):
    print digits_sum(int(numbers[i],3))
