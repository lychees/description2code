#include <iostream>
#include <bits/stdc++.h>
using namespace std;


map<int, int> visited;
int main()
{
	int S, Q, A;
	cin>>S>>Q;

	while(Q--)
	{
		cin>>A;
		
		int gc = __gcd(S, A);
		if (visited.count(gc)==0) {
			visited[gc] = 1;
			cout<<gc<<endl;
		}
		else {
			cout<<-1<<endl;
		}
		
	}
    return 0;
}
