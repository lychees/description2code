#include <iostream>
using namespace std;
int power(int x, unsigned int y)
{
    int temp;
    if( y == 0)
        return 1;
    temp = power(x, y/2);
    if (y%2 == 0)
        return temp*temp;
    else
        return x*temp*temp;
}
int main()
{
   int t,k;
   
   cin>>t;
   while(t--)
   {
string a[8]={ "Rohit", "Dhawan", "Kohli", "Yuvraj", "Raina" , "Dhoni", "Sir Jadeja"};
   	int sum=0;
   	cin>>k;
   	sum= 7;
   	string ans="";
   	if(k<=sum)
   	{
   		ans=a[k-1];
   		cout<<ans<<endl;
   	}
   	else
   	{
   		int x;
   		
   	for(int i=1;i<100000;i++)
   	{
   		int temp=sum;
   		sum=sum+power(2,i)*7;
   		if(k<=sum)
   		{
   			x=k-temp;
            int z=power(2,i); 
   			if(x%z==0)
   			{
                x=(x/z)-1;
   				ans=a[x];
                break;
   			}
   			else
   			{
                 x=x/z;
   				ans=a[x];
               break;
   			}
   		}
   	}
   	cout<<ans<<endl;
   	}
   }
    return 0;
}
