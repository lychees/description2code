#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,ctr1=0,ctr=0;
    cin>>n;
    int arr[n];
    ctr=0;
    for (int i=0; i<n; i++)
    {
        cin>>arr[i];
        if(arr[i]==i)
        {
            ctr1++;
        }
    }
    if (ctr1==n)
    {
        cout << n-2 << endl;
        ctr=-1;
        return 0;
    }
    ctr1=0;
    for(int i=0; i<n; i++)
    {
        if (arr[i]!=i && arr[i]==arr[arr[i]])
        {
            int temp = arr[arr[i]];
            arr[arr[i]]=arr[i];
            arr[i]=temp;
            ctr1=-1;
            break;
        }
    }
    if (ctr1==-1)
    {
        for (int i=0; i<n; i++)
        {
            if(arr[i]==i)
                ctr++;
        }

    }
    else
    {
        for(int i=0; i<n; i++)
        {
            if (arr[i]!=i)
            {
                int temp = arr[arr[i]];
                arr[arr[i]]=arr[i];
                arr[i]=temp;
                ctr1=-1;
                break;
            }
        }

        for (int i=0; i<n; i++)
        {
            if(arr[i]==i) ctr++;
        }

    }
    cout<<ctr<<endl;
}
