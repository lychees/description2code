#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,n,k,ans=0,cnt,temp;
    scanf("%d%d",&k,&n);
    for(i=1;i<=n;i++)
    {
    	temp=i;
    	while(temp)
    	{
    		if(((temp%10)==4)||((temp%10)==7))
    			cnt++;
    		temp/=10;
    	}
    	if(cnt>k)
    	{
    		ans++;
    		cnt=0;
    	}
    }
    printf("%d\n",ans);
    return 0;
}
