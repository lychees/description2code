#include <iostream>
#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
   
  ll t;
  cin>>t;
  while(t--){
      string s;
      ll b=0,r=0,y=0,u=0;
      cin>>s;
      for(ll i=0;i<s.size();i++){
          if(s[i]=='r')r++;
          else if(s[i]=='b')b++;
          else if(s[i]=='u')u++;
          else if(s[i]=='y')y++;
      }
     // b=b/2;
      ll ans=min(min(min(b,u),y),r);
      cout<<ans<<endl;
  }
    return 0;
}
