#include <bits/stdc++.h>
 
using namespace std;
 
int main()
{
    int T;
    scanf("%d",&T);
    while (T>0){
        int n,minn=2e9,maxn=0;
 
        scanf("%d",&n);
 
        int niz[n];
 
        for (int i=0;i<n;i++) {
            scanf("%d",&niz[i]);
        }
 
        sort(niz,niz+n);
 
        for (int i=0;i<n/2;i++) {
            int zbirPara = niz[i]+niz[n-1-i];
            maxn=max(maxn,zbirPara);
            minn=min(minn,zbirPara);
        }
 
        printf("%d\n",maxn-minn);
        T--;
    }
}
