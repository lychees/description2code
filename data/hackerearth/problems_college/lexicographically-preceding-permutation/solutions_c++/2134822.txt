int main()
{
   // printf("Hello World!\n");
   int t;
   cin>>t;
   while(t--)
   {
   	int n;
   	cin>>n;
   	vector<int>vec;
   	vector<int>::iterator itr;
   	for(int i=1;i<=n;i++)
   	{
   		int x;
   		cin>>x;
   		vec.push_back(x);
   	}
   	if(!is_sorted(vec.begin(),vec.end()))
   	prev_permutation(vec.begin(),vec.end());

   	cout<<endl;
   }
    return 0;
}
