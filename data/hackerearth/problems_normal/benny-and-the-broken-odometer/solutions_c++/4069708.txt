#include <iostream>
#include <cmath>
#include <vector>
using namespace std;

int main()
{
    long long int p1,p2,t,n,result,digit,m;
    cin>>t;
    vector<long long int> extra(10,0);
    extra[0]=1;
    for(p1=1;p1<10;p1++)
    {
    	extra[p1]=extra[p1-1]*9+pow(10,p1);
    	//cout<<extra[p1]<<"- >";
    }
    //cout<<endl;
    while(t--)
    {
    	cin>>n;
    	m=n;
    	p1=1;
    	p2=0;
    	result=0;
    	while(n>0)
    	{
    		digit=n%10;
    		n/=10;
    		if(digit<3)
    		{
    			if(p2)
    				result=result+digit*extra[p2-1];
    		}
    		else
    		{
    			result+=p1;
    			//cout<<result<<"-";
    			digit--;
    			if(p2)
    				result=result+digit*extra[p2-1];
    		}
    		p1*=10;
    		p2++;
    		//cout<<result<<"-";
    	}
    	cout<<m-result<<endl;
    }
    return 0;
}
