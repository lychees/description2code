#include <iostream>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(n<3) cout<<"1\n";
    	else
    	{
    		int i=2,num=0;
    		while(n>=i)
    		{
    			n-=i;
    			i++;
    			num++;
    		}
    		if(n>0) num++;
    		cout<<num<<endl;
    	}
    }
    return 0;
}
