#include <bits/stdc++.h>
using namespace std;

long long n,k,top,topa,i,ans,t,fact[1000000],mod=1000000007;
long long facto(long long n){
	 fact[0]=1;
	for(i=1;i<=n;i++){
		fact[i]=fact[i-1]*i;
		fact[i]=fact[i]%mod;
	}
}
long long power(long long x,long long y){
	long long ans=1;
	while(y>0){
		if(y%2==1){
			ans=ans*x;
			ans=ans%mod;
		}
		x=x*x;
		x=x%mod;
		y=y/2;
	}
	return ans;
}
int main()
{
	facto(100000);
	scanf("%lld",&t);
	while(t--){
 scanf("%lld %lld",&n,&k);
 top=fact[k]%mod;
 topa=fact[k-n]%mod;
// cout<<top<<"  "<<topa<<endl; 
 ans=top;
 ans=ans*power(topa,mod-2);
 ans=ans%mod;
 cout<<ans<<endl;}
    return 0;
}
