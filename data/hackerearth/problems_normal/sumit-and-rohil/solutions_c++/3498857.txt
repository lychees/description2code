#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main()
{
	int i,j,k,t,l,m,n;
	
	cin >> n;
	t = n;
	string arr[n];
	string arr_sort[n];
	
	int a[n];
//	memset(a,0.sizeof(a));
	for(i=0;i<n;i++)
	a[i] = 0;
	
	i=0;
	while(t--)
	{
		cin >> arr[i++];
	}
	
	for(i=0;i<n;i++)
	{
		arr_sort[i] = arr[i];
		sort(arr_sort[i].begin(), arr_sort[i].end());
	}
	
	for(i=0;i<n;i++)
	{
		if(a[i] != 1)
		{
			a[i] = 2;
		
		for(j=i+1;j<n;j++)
		{
			if(a[j] != 1)
				if(arr[i][0] == arr[j][0] && arr[i][arr[i].size()-1] == arr[j][arr[j].size()-1])
					if(arr_sort[i] == arr_sort[j])
						a[j] = 1;
		}
		}//end if
	}
	l=0;
	
	for(i=0;i<n;i++)
	if(a[i] == 2)
		l++;
		
	cout << l;
	
    return 0;
}
