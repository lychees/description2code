#include <iostream>
using namespace std;

     int gcd(int number1, int number2) {
        if(number2 == 0){
            return number1;
        }
        return gcd(number2, number1%number2);
    }

int main()
{
    int t,n;
    cin>>t;
    
    for(int i=0;i<t;i++)
    {
    	cin>>n;
    	int a[n];
   
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	
    	int d=a[0];
    	for(int i=1;i<n;i++)
    	{
    		d=gcd(d,a[i]);
    		if(d==1)
    			break;
    	}
    	
    	cout<<d<<endl;
    }
}
