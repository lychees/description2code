#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--){
		string s;
		cin >> s;
		int i,count,sumsofar=0,contisum=0;
		for(int i=0 ; i<s.length() ; i++){
			count=0;
			if(s[i] == 'A' || s[i] == 'a' || s[i] == 'B' || s[i] == 'b' || s[i] == 'C' || s[i] == 'c'){
				count++;
				contisum = contisum + count;
				sumsofar = sumsofar + contisum;
			}
			else{
				contisum = 0;
			}
		}
		cout << sumsofar << "\n";
	}
    return 0;
}
