#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,x;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int a[n];
    	set <int> s;
    	for(int i=0;i<n;i++)
    	{
    		cin>>x;
    		s.insert(x);
    	}
    	int j=0;
    	while(!s.empty())
    	{
    		a[j]=*s.begin();
    		s.erase(a[j]);
    		j++;
    	}
    	int i;
    	sort(a,a+j);
    	if(j%2==0)
    		for(i=0;i<j/2;i++)
    		{
    			cout<<a[j-i-1]<<" "<<a[i]<<" ";
    		}
    	else
    	{
    		for(i=0;i<j/2;i++)
    		{
    			cout<<a[j-i-1]<<" "<<a[i]<<" ";
    		}
    		cout<<a[i];
    	}
    	cout<<endl;
    }
    return 0;
}
