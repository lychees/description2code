#include <iostream>
using namespace std;
#include <math.h>

int main ()
{
	
	int t; 
	
		
		
		long steps[105] ;
		steps[1]=1;
		steps[2]=2;
		
		for(int i=3;i<=100;i++)
		{
			steps[i] = steps[i-1]+steps[i-2];
			
		}
		
		
	cin>>t;
	
	
	while(t--)
	{	
	
		
		int s ;
		cin>>s;
		cout<<steps[s]<<endl;
	
		
	}
}
