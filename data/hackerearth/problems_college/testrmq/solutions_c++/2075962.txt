#include <bits/stdc++.h>
#define MAX 1000000
using namespace std;

int *arr;
int tree[MAX];

void build(int node,int a,int b)
{
	if(a > b)
		return;
	if(a==b)
	{
		tree[node]=arr[a];
		return ;
	}
	build(node*2,a,(a+b)/2);
	build(node*2+1,(a+b)/2+1,b);
	tree[node]=min(tree[node*2],tree[node*2+1]);
}

int query(int node,int a,int b,int i,int j)
{
	if(a > b || a>j || b<i)
		return (MAX+1);
	if(a>=i && b<= j)
		return tree[node];
	int q1=query(node*2,a,(a+b)/2,i,j);
	int q2=query(node*2+1,(a+b)/2+1,b,i,j);
	int res=min(q1,q2);
	return res;
}
int main()
{
    int q,n,i,j;
    scanf("%d",&n);
    arr=new int [n];
    for(int i=0;i<n;i++)
    	scanf("%d",&arr[i]);
    build(1,0,n-1);
    scanf("%d",&q);
    while(q--)
    {
    	scanf("%d %d",&i,&j);
    	i--;
    	j--;
    	printf("%d\n",query(1,0,n-1,i,j));
    }
    return 0;
}
