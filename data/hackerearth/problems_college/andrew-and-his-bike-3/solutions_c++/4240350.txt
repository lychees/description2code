#include<bits/stdc++.h>
using namespace std;

typedef pair<int,int> ii;
typedef vector<int> vi;
typedef vector<ii> vii;
typedef long long int ll;

#define rep(i,a,b) for(i=a;i<b;i++)
#define repv(i,a,b) for(i=a;i>b;i--)
#define SET(a,b) memset(a,b,sizeof(a))
#define inf INT_MAX
#define pb push_back

vector<vi> ad(100010, vi(0));

int visited[100010],nodes[100010];

int dfs(int x)
{
	int i,sum=0;
	vi next=ad[x];
	visited[x]=1;
	rep(i,0,next.size())
	{
		if(visited[next[i]]==0)
		{
			sum+=dfs(next[i]);
			sum+=1;
		}
	}
	return nodes[x]=sum;
}

int main()
{
	int n,i,x,y;
	cin>>n;
	rep(i,0,n-1)
	{
		cin>>x>>y;
		ad[x].pb(y);
		ad[y].pb(x);
	}
	dfs(1);
	rep(i,1,n+1)
	{
		if(i!=1)
			nodes[i]++;
		cout<<nodes[i]<<" ";
	}
	cout<<"\n";
	return 0;
}
