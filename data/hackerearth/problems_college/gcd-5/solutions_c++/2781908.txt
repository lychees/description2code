#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long x,i;
    	cin>>x;
    	long a[x];
    	for(i=0;i<x;i++)
    	{
    	 cin>>a[i];	
    	}
    	int k=0;
    	for(i=0;i<x-1;i++)
    	{
    		if(a[i]>a[i+1])
    		{
    			k=1;
    			break;
    		}
    	}
    	if(k==1)
    	cout<<"NO\n";
    	else
    	cout<<"YES\n";
    }
    return 0;
}
