#include <iostream>
#include <bits/stdc++.h>

#define mod 1000000007
#define ll long long int

using namespace std;

int main()
{
    char T[200007];
    cin>>T;
    ll ctr=0;
    ll seq=0;
    for(int i = 0 ; i < strlen(T) ; i++){
    	ctr = ctr*10;
		seq += (i+1)*(T[i]-'0');	
		ctr += seq;
		ctr = ctr%mod;
    }
    cout<<ctr<<endl;
    return 0;
}
