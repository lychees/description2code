#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s;
    	cin>>s;
    	int n;
    	cin>>n;
    	int count[2] = {0 , 0};
    	for(int i = 0 ; i < s.size() ; i++)
    		count[s[i]-48]++;
    	swap(count[0] , count[1]);
    	n = (n+1)/2;
    	cout<<count[n%2]<<endl;
    }
    return 0;
}
