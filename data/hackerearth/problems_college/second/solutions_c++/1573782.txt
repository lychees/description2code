#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

struct myclass
{
	bool operator() (int i,int j) { return (i<j);}
}myobject;

int main()
{
	int n,m,num,outp=0,i;
	long long int s=0;
    vector<int> a;
    cin>>n>>m;
    while(n--)
    {
    	cin>>num;
    	a.push_back(num);
    }
    sort(a.begin(),a.end(),myobject);
    for(i=0;i<a.size();i++)
    {
    	if(m+s == a[i])
    	  {
    	  	outp++;
    	  	s+=m;
    	  }
    }
    cout<<s;
    return 0;
}
