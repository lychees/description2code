#include <iostream>
using namespace std;

int main()
{
    int T;
    cin >> T;
    int n, b, g, temp;
    while (T--)
    {
    	cin >> n >> b >> g;
    	if (b > g)
    	{
    		temp = g;
    		g = b;
    		b = temp;
    	}
    	if (g-1 > b)
    		cout << "Little Jhool wins!" << endl;
    	else
    		cout << "The teacher wins!" << endl;
    }
    return 0;
}
