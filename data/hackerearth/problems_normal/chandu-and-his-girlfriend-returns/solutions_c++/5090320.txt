#include <bits/stdc++.h>
using namespace std;
vector<long long> v;
void merge(long long a[],long long b[],long long n,long long m){
	long long i=0,j=0;
	while(i<n&&j<m){
		if(a[i]>=b[j])v.push_back(a[i++]);
		else v.push_back(b[j++]);
	}
	if(i!=n){
		while(i!=n)v.push_back(a[i++]);
	}
	if(j!=m){
		while(j!=m)v.push_back(b[j++]);
	}
	j=v.size();
	for(i=0;i<j;i++)cout<<v[i]<<" ";
	cout<<"\n";
}
int main()
{ios_base::sync_with_stdio(false);
cin.tie(0);
   // cout << "Hello World!" << endl;
    int t;
    long long n,m,i;
    
    cin>>t;
    while(t--){
    cin>>n>>m;
    long long a[n],b[m];
    for(i=0;i<n;i++)cin>>a[i];
    for(i=0;i<m;i++)cin>>b[i];
    merge(a,b,n,m);
    v.clear();
    }
    return 0;
}
