#include <bits/stdc++.h>

using namespace std;

bool is_perfect_cube(int N) {
    int root(round(cbrt(N)));
    return N == root * root * root;
}

int x[1005][1005];

int main() {
	long long int n,k,a[1005][1005],sum=0,flag=0,h;
	scanf("%lld%lld",&n,&k);
	for(int i=1;i<=n;i++)
	{
	    for(int j=1;j<=n;j++)
	    {
	        scanf("%lld",&a[i][j]);
	        x[i][j]=x[i][j-1]+a[i][j];
	    }
	}
	h=n-k;
	for(int i=1;i<=h+1;i++)
	{
	    for(int j=1;j<=h+1;j++)
	    {
	    	sum=0;
	        for(int q=j;q<j+k;q++)
	        {
	            int w = i+k-1;
	            sum+=x[q][w]-x[q][i-1];
	        }

	        if(is_perfect_cube(sum))
	            flag++;
	    }
	}
	
	
	
	printf("%lld\n",flag);
	return 0;
}
