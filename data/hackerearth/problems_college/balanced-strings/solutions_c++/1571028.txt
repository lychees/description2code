#include <iostream>
#include <string.h>
using namespace std;

int main()
{
    int tc,r[26],flag;
    char s[1000],ch[26];
    ch[0]='a';
    cin>>tc;
    for(int i=0;i<tc;i++)
    {
    	flag=0;
    	cin>>s;
    for(int k=0;k<26;k++)	
    	{
    		r[k]=0;
    		char z=ch[0]+k;
    		for(int j=0;j<strlen(s);j++)
    		{
    			if(z==s[j])
    			r[k]++;
    		
    		}
    	}
    	for(int x=0;x<26;x++)
    	{
    		if(r[0]!=r[x])
    		flag++;
    	}
    	if(flag==0)
    	cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
    }
    return 0;
}
