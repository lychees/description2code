#include <bits/stdc++.h>
#define MAX 1000001
using namespace std;

int main()
{
	int i,j,k;
	int prime[MAX];
	int p[MAX]={0};
	p[0]=p[1]=-1;
	for(i=2;i<sqrt(MAX);i++)
	{
		for(j=i*i;j<MAX;j+=i)
		{
			p[j]=-1;
		}
	}
	k=0;
	for(i=0;i<MAX;i++)
	{
		if(p[i]==0)
		{
			prime[k++]=i;
		}
	}
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,count;
		scanf("%d",&n);
		if(p[n]==0)
		cout<<"2^0"<<endl;
		else{
		i=0;
		count=0;
		while(n)
		{
	
			if(n%prime[i]==0)
			{
				count++;
				n/=prime[i];
			}
			else
			{
				if(prime[i]==2)
				cout<<prime[i]<<"^"<<count;
				else if(prime[i]!=2 && count>0)
				cout<<"*"<<prime[i]<<"^"<<count;
				i++;
				count=0;
				if(n==1)
				n=0;
			}
		}
		cout<<endl;}
	}
	return 0;
}
