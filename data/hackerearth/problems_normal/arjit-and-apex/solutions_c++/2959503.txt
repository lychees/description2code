#include<iostream>
#include<algorithm>
#include<utility>
#include<set>
using namespace std;
int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n,m;
		cin >> m >> n;
		multiset<int> bar[10002],foo[10002];
		for(int i=0;i<m;i++)
		{
			int u,v;
			cin >> u >> v;
			bar[u].insert(v);	
		} 
		for(int i=0;i<n;i++)
		{
			int w,x;
			cin >> w >> x;
			foo[w].insert(x);
		}
		int good=0;int great=0;
		for(int i=1;i<=10002;i++)
		{
			int temp=0;
			//multiset<int>::iterator j;
			for( auto j:bar[i])
				if(foo[i].count(j))
				temp++,foo[i].erase(foo[i].find(j));
				
			int t1 = bar[i].size();
			int t2 = foo[i].size();
			good += temp, great += temp;
			t1 -= temp;
			good += min(t1, t2);
		}
		int g, h;
		cin >> g >> h;
		if (good >= g and great >= h)
			cout << "Great\n";
		else if (good >= g)
			cout << "Good\n";
		else
			cout << ":(\n";
			
	}
	return 0;
}
