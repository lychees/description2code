

for n in range(0, input()):
    s = raw_input()
    l = len(s)
   # print l
    h = l/2
    a = s[0:h]
    b = s[h:]
    s1 = 0
    s2 = 0
    for x in range(0, len(a)):
        s1 = s1+int(a[x])
    for x in range(0, len(b)):
        s2 = s2+int(b[x])
    if (s1 == s2):
        print'YES'
    else:
        print 'NO'
