#include <iostream>
using namespace std;
#define MOD 1000000007
long long dp[1000003];
int main()
{
    dp[1] = 2;
    long long nE=1,nH=1;
    
    for(int i=2;i<1000001;i++)
    {
    	dp[i] = ((2*nH) + (nE))%MOD;
    	dp[i] = (dp[i]+dp[i-1])%MOD;
    	long long tnE = nE, tnH = nH;
    	
    	nE = tnH;
        nH = (tnE+tnH)%MOD;
    }
    
    int t;
    
    cin >> t;
    
    while(t--)
    {
    	int n;
    	cin >> n;
    	cout << dp[n] << "\n";
    }
    return 0;
}
