import math
for i in range(input()):
    l = [int(x) for x in raw_input().split()]
    if (l[0] > l[1] and l[0] > l[2]):
        print(-1)
    else:
        a = math.ceil((l[5]/(l[4]*1.0))*60)
        b = math.ceil((l[5]/(l[3]*1.0))*60)
        if (l[0] > l[1]):
            print(int(a+l[2]))
        elif (l[0] > l[2]):
            print(int(b+l[1]))
        else:
            c = min(a+l[2], b+l[1])
            print(int(c))
