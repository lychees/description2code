#include <bits/stdc++.h>
using namespace std;

map<char, int> mp; 

bool comp(char a, char b)
{
	return (mp[a]<mp[b]);
}
int main()
{
	int t; cin >> t;
	while(t--)
	{
		string p, f; cin >> p >> f;
		int j = 1;
		for(int i=0;i<p.length();i++)
		{
			mp[p[i]]  = j;
			j++;
		}
		
		sort(f.begin(), f.end(), comp);
		cout << f << endl;
	}
}
