#include<iostream>
#include<string>

using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);cin.tie(NULL);
    int t;
    string s;
    cin>>t;
    while(t-- > 0)
    {
        int counter = 0,len = 0;
        cin>>s;
        len = s.length();
        for(int i=0;i<len;i++)
        {
            if(s[i] == 'A'||s[i] == 'a')
            {
                cout<<s[i];
                counter++;
            }
            if(s[i] == 'E'||s[i] == 'e')
            {
                cout<<s[i];
                counter++;
            }
            if(s[i] == 'I'||s[i] == 'i')
            {
                cout<<s[i];
                counter++;
            }
            if(s[i] == 'O'||s[i] == 'o')
            {
                cout<<s[i];
                counter++;
            }
            if(s[i] == 'U'||s[i] == 'u')
            {
                cout<<s[i];
                counter++;
            }
        }
        if(!counter)
        {
            cout<<"No";
        }
        cout<<"\n";
    }

    return 0;
}
