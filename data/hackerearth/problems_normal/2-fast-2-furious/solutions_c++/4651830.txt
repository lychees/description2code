#include <iostream>
#include<vector>
#include<math.h>
using namespace std;

int main()
{
    
    int n;
    cin>>n;
    vector<int> vec;
    vec.resize(n);
    for(int i=0;i<n;i++)
    	cin>>vec[i];
    int max1=0;
    for(int i=1;i<n;i++)
    {
    	max1=max(max1,abs(vec[i]-vec[i-1]));
    }
    
    
    for(int i=0;i<n;i++)
    	cin>>vec[i];
    int max2=0;
    for(int i=1;i<n;i++)
    {
    	max2=max(max2,abs(vec[i]-vec[i-1]));
    }
    if(max1>max2)
    {
    	cout<<"Dom\n";
    	cout<<max1<<endl;
    }
    else if(max1<max2)
    {
    	cout<<"Brian\n";
    	cout<<max2<<endl;
    	
    }
    else
    {
    	cout<<"Tie\n";
    	cout<<max1<<endl;
    	
    }

	
}
