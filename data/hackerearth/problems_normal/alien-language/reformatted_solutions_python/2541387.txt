def CheckSubstring(v, k):
    if v.find(k) != -1:
        return "YES"
    for j in xrange(len(k)):
        if v.find(k[j:j+1]) != -1:
            return "YES"
    return "NO"


if __name__ == "__main__":
    n = int(raw_input())
    for i in xrange(n):
        v = raw_input()
        k = raw_input()
        print CheckSubstring(v, k)
