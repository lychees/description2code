ull ans, L, R, p[111];
int T, n;

void dzen(int z, ull x, int k)
{
	if (x > R)
		return;
	if (z == n + 1)
	{
		if (k == 0)
			return;
		if (k & 1)
			ans += R / x - (L - (ll)1) / x;
		else
			ans -= R / x - (L - (ull)1) / x;
		return;
	}
	dzen(z + 1, x, k);
	if ((ull)10 * x > R && p[z] > 10)
		return;
	if ((ull)100 * x > R && p[z] > 100)
		return;
	dzen(z + 1, x * (ull)p[z], k + 1);
	return;
}
