t=int(raw_input())
while t>0:
	a=int(raw_input())
	r=0
	for b in range((a/5)+1):
		for c in range((a/3)+1):
			if b*5+c*3==a:
				r=1
				break
			if b*5+c*3>a:
				break
		if r==1:
			break
	
				
				
	s="5"*(c*3)
	s+="3"*(b*5)
	if r==1:
	     print(s)
	else:
		print(-1)
	t-=1
