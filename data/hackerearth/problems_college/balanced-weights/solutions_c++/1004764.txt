#include<iostream>
#include<algorithm>
using namespace std;

long long int gcd(long long int a,long long int b) {
	if (a % b == 0)
		return b;
	a=a%b;
	return gcd(b, a);
}

int main(){
	int t;
	cin>>t;
	while(t--){
		long int n;
		scanf("%ld",&n);
		long long int a,b;
//		cin>>a;
//		cin>>b;
		scanf("%lld",&a);
		scanf("%lld",&b);
		long long int result=0;
		if(a>b){
			result=gcd(a,b);
		}
		else{
			result=gcd(b,a);
		}
		for(int i=2;i<n;i++){
//			cin>>a;
			scanf("%lld",&a);
			if(result>a)
				result=gcd(result,a);
			else
				result=gcd(a,result);
		}
		cout<<result*n<<endl;
	}
	return 0;
}
