#include <bits/stdc++.h>
using namespace std;
long long int h[26];
#define m  1000000007

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string a,b;
    	cin>>a>>b;
		memset(h,0,sizeof(h));    	
    	for(int i=0;i<a.length();i++){
    		h[a[i]-'a']++;
    	}
    	long long ans = 1;
    	int t = 1;
    	for(int i=0;i<b.length();i++){
    		ans *= h[b[i]-'a'];
    		ans %= m;
    		h[b[i]-'a']--;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
