#include <bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d", &x)

#define MOD 1000000009


int dp[2][100100];

int solve(){
    int n, f, i, j, x;
    cin>>n;
    f = 0;
    for(i = 0; i <= 100000; i++){
    	dp[0][i] = 0;
    }
    dp[0][0] = 1;
    
    for(i = 1; i <= n; i++){
        cin>>x;
        for(j = 0; j <= 100000; j++){
            dp[1-f][j] = dp[f][j];
        }
        for(j = 0; j < 100000 - x; j++){
            if(dp[f][j] == 1){
                dp[1-f][j + x] = 1;
            }
        }
        f = 1-f;
    }
    int ans = 0;
    for(j = 0; j <= 100000; j++){
        if(dp[f][j] == 1) ans++;
    }
    cout<<ans<<endl;

}

int main(){
    int t;
    cin>>t;
    while(t--){
        solve();
    }
    return 0;
}
