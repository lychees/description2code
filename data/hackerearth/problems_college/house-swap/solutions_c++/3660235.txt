#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
	int n;
	scanf("%d", &n);
	int a[1000001];
	for(int i = 0; i < n; i++) {
		scanf("%d", &a[i]);
	}
	int swap = 0, count = 0;
	for(int i = 0; i < n; i++) {
		if(a[i] == i)
			count++;
		else if(a[a[i]] == i && swap == 0) {
			swap = 1;
			count += 2;
		}
	}
	if((!swap) && count != n) count++;
	else if(!swap) count -= 2;
    cout << count << endl;
    return 0;
}
