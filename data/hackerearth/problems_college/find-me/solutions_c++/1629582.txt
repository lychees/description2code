#include <iostream>
using namespace std;

int main()
{
    int t ,i ,j , n , p , k , ans = 0;
    cin>>t;
    while(t--)
    {
    	cin>>n>>p;
    	k = 0;
    	ans = 0;
    	//int arr[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>j;
    		if(j==p)
    		{
    			k = 1;
    		}
    		if(j>p)
    		{
    			ans++;
    		}
    	}
    	if(k==1)
    	{
    		cout<<"YES "<<ans<<endl;
    	}else{
    		cout<<"NO 0\n";
    	}
    }
    return 0;
}
