#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    char ch;
    cin.get(ch);
    while(t--){
    	char pat[10005];
    	char text[105005];
    	cin>>pat>>text;
    	int lp=strlen(pat);
    	int lt=strlen(text);
    	int fpat[26]={0};
    	for(int i=0;i<lp;i++){
    		fpat[pat[i]-'a']++;
    	}
    	int ft[26]={0};
    	for(int i=0;i<lp;i++){
    		ft[text[i]-'a']++;
    	}
    	int f=0;
    	for(int i=lp;i<lt;i++){
    		int found=1;
    		for(int j=0;j<26;j++){
    			if(fpat[j]!=ft[j]){
    				found=0;
    				break;
    			}
    		}
    		if(found){
    			f=1;
    			break;
    		}
    		ft[text[i-lp]-'a']--;
    		ft[text[i]-'a']++;
    	}
    	int found=1;
    	for(int j=0;j<26 && !f;j++){
    		if(fpat[j] != ft[j]){
    			found=0;
    			break;
    		}
    	}
    	if(found){
    		f=1;
    	}
    	if(!f){
    		cout<<"NO"<<endl;
    	}
    	else if(f){
    		cout<<"YES"<<endl;
    	}
    }
    
    return 0;
}
