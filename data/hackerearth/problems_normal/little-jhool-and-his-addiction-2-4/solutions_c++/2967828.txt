#include <bits/stdc++.h>

using namespace std;
long arr[1000006];
int main() {
	int t;
	long i,n,k,maxs,mins;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%ld%ld",&n,&k);
		for(i=0;i<n;i++)
			scanf("%ld",&arr[i]);
		sort(arr,arr+n);
		for(i=0;i<=n/2;i++)
		{
			if(i==0)
			{ 	maxs=arr[n-1-i]+arr[i]; mins=maxs; }
			else 
			{
				if(maxs<arr[n-1-i]+arr[i])
					maxs=arr[n-1-i]+arr[i];	
				if(mins>arr[n-1-i]+arr[i])
					mins=arr[n-1-i]+arr[i];
			}
		}
		if(k>maxs-mins)
			printf("%ld\nChick magnet Jhool!\n",maxs-mins);	
		else if(k==maxs-mins)
			printf("%ld\nLucky chap!\n",maxs-mins);
		else
			printf("%ld\nNo more girlfriends!\n",maxs-mins);		
	}
	return 0;
}
