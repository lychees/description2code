#include <iostream>
#include<algorithm>
using namespace std;
bool comp(int i,int j)
{
	return i>j;
}
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int n,sum=0,sum2=0,cnt=0,x;
   	cin>>n;
   	int *a=new int[n];
   	for(int i=0;i<n;i++)
   	{
   		cin>>x;
   		sum+=x;
   	}
   	for(int i=0;i<n;i++)
   	cin>>a[i];
   	sort(a,a+n,comp);
   	for(int i=0;i<n;i++)
   	{
   		if(sum2>=sum)
   		break;
   		sum2+=a[i];
   		cnt++;
   		
   	}
   	cout<<cnt<<endl;
   	cnt=0;
   }
    return 0;
}
