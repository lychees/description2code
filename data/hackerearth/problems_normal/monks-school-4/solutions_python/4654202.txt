teachers = {}

N,M = map(int, raw_input().split(' '))

for i in xrange(N):
	teachers[raw_input()] = []
	

for i in xrange(M):
	(t,s,a) = raw_input().split(' ')
	a = int(a)
	teachers[t].append((s, a))

steachers = sorted(teachers.keys())
for st in steachers:
	print st
	for stt in sorted(teachers[st], key=lambda x:x[1]):
		print stt[0] + ' ' + str(stt[1])
