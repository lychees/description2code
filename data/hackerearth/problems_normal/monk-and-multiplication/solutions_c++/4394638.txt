#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long a,n,i,a1,a2,a3,ans;
    priority_queue<long long > h;
    cin>>n;
    for(i=0;i<n;i++){
    	cin>>a;
    	if(i<2){
    		cout<< -1 <<endl;
    		h.push(a);
    	}
    	else{
    		h.push(a);
    		a1=h.top();
    		h.pop();
    		a2=h.top();
    		h.pop();
    		a3=h.top();
    		h.pop();
    		ans=a1*a2*a3;
    		cout<<ans<<endl;
    		h.push(a1);
    		h.push(a2);
    		h.push(a3);
    	}
    }
    return 0;
}
