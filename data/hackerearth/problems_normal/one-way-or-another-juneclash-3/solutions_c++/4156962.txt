#include <iostream>
#include <cstdio>

using namespace std;
const int MAX = 1e4 + 5;
long long r[MAX], c[MAX], x[MAX];

long long llabs(long long a)
{
    return (a >= 0 ? a : -a);
}

int main()
{
    int t, n;
    long long dis;
    bool flag;
    for(scanf("%d", &t);t--;)
    {
        scanf("%d", &n);
        for(int i = 0;i < n;++i)
            scanf("%lld %lld %lld", &r[i], &c[i], &x[i]);
        flag = true;
        for(int i = 1;i < n;++i)
        {
            dis = llabs(r[i] - r[i-1]) + llabs(c[i] - c[i-1]);
            if((dis & 1) and ((x[i] & 1) == (x[i-1] & 1)))
            {
                flag = false;
                break;
            }
            if(((dis & 1) == 0) and ((x[i] & 1) != (x[i-1] & 1)))
            {
                flag = false;
                break;
            }
        }
        if(flag) printf("Yes\n");
        else printf("No\n");
    }
    return 0;
}
