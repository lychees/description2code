#include <bits/stdc++.h>
using namespace std;

int k,ada[30],jenis,ans; string s;

int main(){
	ios::sync_with_stdio(0); cin.tie(0);
	cin >> k >> s;
	for (int i=0;i<(int)s.size();i++){
		memset(ada,0,sizeof(ada));
		jenis=0;
		for (int j=i;j<(int)s.size();j++){
			if (!ada[s[j]-'a']) jenis++;
			ada[s[j]-'a']++;
			if (jenis==k) ans++;
		}
	}
	cout << ans << "\n";
}
