#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main() {
	int t;
	cin >>t;
	while(t--){
	    ll n,m,x,i,sum=0;
	    cin >>n>>m>>x;
	    ll num[n];
	    for(i=0;i<n;i++){
	        cin>>num[i];
	        sum=sum+num[i];
	    }
	    ll ans=(n+1)*x-sum;
	    if(ans<=1){
	        cout <<"1"<<endl;
	    }
	    else if(ans<=m){
	        cout <<ans<<endl;
	    }
	    else{
	        cout <<"Impossible"<<endl;
	    }
	        
	}
	return 0;
}
