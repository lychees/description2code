int a[100005];
char b[100005];
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    scanf("%d",&t);
    a[1] = 1;
    int n;
    for(n=2;n<=100000;n++)
    {
    	a[n] = (a[n-1] + a[n-2])%26;
    }

    while(t--)
    {
    map<int,int> g;
    int l,r;
    scanf("%d%d%s",&l,&r,&b);
    int co[26],v[26];
    int i;
    for(i=0;i<26;i++)co[i] =v[i] =  0;
    for(i=l;i<=r;i++)co[a[i]]++;
    for(i=0;i<26;i++)
    {
    	int j;
    	int ma = -1;
    	int an = -1;
    	for(j=0;j<26;j++)
    	{
    		if(co[j] > ma && v[j]==0)
    		{
    			an = j;
    			ma = co[j];
    		}
    	}
    	g['a' + i] = 'a' + an;
    	v[an] = 1;
    }
    for(i=0;b[i];i++)
    b[i] = g[b[i]];
    printf("%s\n",b);
    }

}
