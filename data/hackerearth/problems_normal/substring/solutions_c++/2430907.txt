#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main(){
 int tc;
 cin>>tc;
 assert(tc<=10);
 while(tc--){
  string str;
  cin>>str;
  int j=0;
  int len=str.length();
  assert(len<=100000);
  ll sub_str=0;
  while(j<len){
   ll count=0;
   while(j<len&&(str[j]=='A'||str[j]=='B'||str[j]=='C'||str[j]=='a'||str[j]=='b'||str[j]=='c')){
     count++;
     j++;
   }
   sub_str+=((count*(count+1))/2);
   j++;
  }
  cout<<sub_str<<endl;
 }
return 0;
}
