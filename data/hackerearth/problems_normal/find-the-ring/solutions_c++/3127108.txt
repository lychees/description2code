#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,index,n;
    scanf("%d",&t);
    while(t--){
    	scanf("%d%d",&index,&n);
    	if(index==0){
    		if(n%2==0) printf("0\n");
    		else printf("1\n");
    	}
    	else if(index==1){
    		if(n%2==0) printf("1\n");
    		else printf("0\n");
    	}
    	else{
    		if(n==0) printf("2\n");
    		else if(n%2==0) printf("0\n");
    		else if(n%2!=0) printf("1\n");
    	}
    }
    return 0;
}
