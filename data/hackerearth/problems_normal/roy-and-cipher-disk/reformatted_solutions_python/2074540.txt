test = int(raw_input())
for i in range(test):
    st = raw_input()
    x = 0
    a = ord('a')
    for j in range(len(st)):
        z = ord(st[j])-a
        cw = z-x
        if cw < 0:
            cw += 26
        acw = z-x
        if acw > 0:
            acw -= 26
        if abs(acw) < abs(cw):
            print acw,
        else:
            print cw,
        x = z
    print ''
