#include <iostream>
#include <map>
using namespace std;

int parse(map<string,int> &mp,string s)
{
	
	s=s+'/';
	int l=s.size();
	int res=0;
	for(int i=1;i<s.size();i++)
	{
		if(s[i]=='/')
		{
			string str = s.substr(0,i);
		//	cout<<str<<endl;
			if(!mp[str])
			{
				mp[str]=1;
				res++;
			}
		}
	}
	return res;
}

int main()
{
    int t;
    cin>>t;
    int tn=1;
    while(t--)
    {
    	map<string,int> mp;
    	int n,m;
    	cin>>n>>m;
    	for(int i=0;i<n;i++)
    	{
    		string s;
    		cin>>s;
    		parse(mp,s);
    	}
    	int res=0;
    	for(int i=0;i<m;i++)
    	{
    		string s;
    		cin>>s;
    		res=res+parse(mp,s);
    	}
    	cout<<res<<endl;	
    	tn++;
    }
    //
    return 0;
}
