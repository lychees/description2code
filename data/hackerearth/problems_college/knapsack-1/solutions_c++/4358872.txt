#include<bits/stdc++.h>
using namespace std;

void fn(int a[2100],int sum, int currindex, int n,int k,int& maxsum,int& maxdiff)
{
	if(sum==k)
	  {
	  	maxsum=k;
	  	maxdiff=0;
	  	return;
	  	
	  }
	 if(sum>k)return;
	 
	 if(k-sum<maxdiff)
	   {
	   	maxdiff=k-sum;
	   	maxsum=sum;
	   	
	   }
	   
	   for(int i=currindex;i<n;i++)
	     {
	     	sum+=a[i];
	     	fn(a,sum,i,n,k,maxsum,maxdiff);
	     } 
}
int main()
{  int t;
   cin>>t;
   while(t--)
   {
   	int i,n,k,a[2100];
   	cin>>n>>k;
   	for(i=0;i<n;i++)cin>>a[i]; 
   	int maxdiff=INT_MAX,maxsum=INT_MIN;
   	fn(a,0,0,n,k,maxsum,maxdiff);
   	cout<<maxsum<<endl; 
   }
	return 0;
}
