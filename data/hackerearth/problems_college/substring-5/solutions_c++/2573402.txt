#include <iostream>
using namespace std;

int main()
{
   int i,j,k,n,m,flag=1,count=0;
   string str,str1;
   cin>>str;
   n=str.size();
   for(i=2;i<=n;i++)
   {
   		for(j=0;j<n-i+1;j++)
   		{
   			str1=str.substr(j,i);
   			m=str1.size();
   			flag=1;
   			for(k=0;k<m-1;k++)
   			{
   				if(str1[k]==str1[k+1])
   					continue; 
   				else
   					{
   						flag=0;
   						break;
   					}
   			}
   			if(flag==1)
   				count++;
   		}
   }
			cout<<count+n;
}
