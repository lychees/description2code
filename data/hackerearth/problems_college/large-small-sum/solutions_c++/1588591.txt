#include <iostream>
using namespace std;
void swap(int* a, int* b)
{	int t;
    t = *a;
    *a = *b;
    *b = t;
}

int partition (int arr[], int l, int h)
{
    int x = arr[h];    
    int i = (l - 1);  
 
    for (int j = l; j <= h- 1; j++)
    {         
        if (arr[j]<=x)
        {
            i++;   
            swap(&arr[i], &arr[j]);  
        }
    }
    swap(&arr[i + 1], &arr[h]);  
    return (i + 1);
}
 

void quickSort(int arr[], int l, int h)
{
    if (l < h)
    {
        int p = partition(arr, l, h); 
        quickSort(arr, l, p - 1);
        quickSort(arr, p + 1, h);
    }
}


int main()
{
   int t,n,a[1005],i,j,k;
   cin>>t;
   while(t--)
   {	cin>>n;
   		for(i=0;i<n;i++)
   		{
   			cin>>a[i];
   		}
   		quickSort(a,0,n-1);
   		cin>>j>>k;
   		cout<<a[k-1]+a[n-j]<<endl;
   		
   		
   	
   }
    return 0;
}
