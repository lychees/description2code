#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	int a,b;
    	cin>>a>>b;
    	int x,y;
    	x=b/9;
    	y=b%9;
    	if(y!=0)
    	{	
		x++;
    	y--;
    	if(y>=x)
    	y++;
    	}
    	else
    	{
    		if(x==9)
    		y=8;
    		else
    		y=9;
		}
    	if(x>9)
    	cout<<"-1";
    	else
    	{
    		for(int j=0;j<a;j++)
    		{
    			if(j%2==0)
    			cout<<x;
    			else
    			{
    				cout<<y;
    			}
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
