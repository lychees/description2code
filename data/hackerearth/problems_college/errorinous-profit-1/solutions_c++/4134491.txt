#include<bits/stdc++.h>
using namespace std;
int reverse(int n)
{
    int r,num=0;
    while(n!=0)
    {
        r = n % 10;
        num = num*10 + r;
        n = n/10;
    }
    return num;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        float p,e;
        scanf("%f %f",&p,&e);
        float errprice= p + (p * e)/100;

        float n = reverse((int)errprice);

        float err = 1.0 - (n/p);

        float ans = errprice + errprice*err;

        cout<<(int)ans<<endl;
    }
    return 0;
}
