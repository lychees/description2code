#include <iostream>
using namespace std;

int main()
{
   	int t;
	cin>>t;
	while(t>0){
		char pattern[26];
		string finalString;
		cin>>pattern>>finalString;
		int temp[26] = {0};
		for (int i=0 ;i < finalString.length(); i++){
	            int index = finalString[i]-97;
	            temp[index]++;
	    }
	    for (int i = 0; i < 26; i++){
	            if (temp[pattern[i]-97]>0){
	                for (int j = 0 ; j < temp[pattern[i]-97]; j++){
	                    cout<<pattern[i];
	                }
	            }
	        }
	    cout<<endl;
		t--;
	}
    return 0;
}
