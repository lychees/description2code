#include<bits/stdc++.h>
using namespace std;

long long a[100][100];
long long fib[60];
int fn()
{   fib[1] = 1;
    fib[2] = 1;

    for(int i = 3 ; i<=55 ; i++)
    {
        fib[i] = fib[i-1]+fib[i-2];
    }

    for(int i = 1 ; i<=52; i++)
    {
        long long fact = fib[i];
        int idx = 1;
        for(int j = i ; j<= 52; j++)
        {
            a[j][i] = fact * fib[idx++];
        }
        fact++;
    }

}




int main()
{   std::ios::sync_with_stdio(false);
    fn();

    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;


        long long sum = 0;

        for(int i = 1; i<=n ; i++)
        {
            sum+=a[n][i];
        }
        double x = 2*sum;
       // cout<<2*sum<<"   "<<fib[n+1]<<endl;
        double y = fib[n+1];



        printf("%.6lf\n",x/y);

    }

    return 0;
}
