#include <iostream>
#include <math.h>
using namespace std;
const int MAX = 1000000007;
long long int ex(long long int a,int b){
	if(b==0)
	return 1;
	else if(b%2==0){
		return ex((a*a)%MAX,b/2);
	}else{
		return ((a%MAX)*ex((a*a)%MAX,(b-1)/2))%MAX;
	}
}


int main()
{
    long long int i,j,t,it,n,tmp,sum,product,ma;
    cin>>t;
    for(it=1;it<=t;++it){
    	sum = 0;
    	ma =-1;
    	product = 1;
    	cin>>n;
    	int *A = new int[n];
    	for(i=0;i<n;++i)
    	cin>>A[i];
    	
    	for(i=1;i<pow(2,n);++i){
    		sum=0;product=1;
    	for(j=0;j<n;++j){
    		if((i&(1<<j))>0){
    			
    			sum= (sum+(A[j]));
    			product = (product*(A[j]%MAX))%MAX;
    		}	
    	}
    	
    	sum = ex(sum,MAX-2);
    	
    	ma = max(ma,((product%MAX)*(sum%MAX))%MAX);
    	
    	}
    	cout<<"Case #"<<it<<": "<<ma<<"\n";
    }
    return 0;
}
