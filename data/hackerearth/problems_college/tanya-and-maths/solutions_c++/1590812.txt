#include <bits/stdc++.h>
using namespace std;
#define ll long long

#define MAXN 1000000
ll phi[MAXN + 1], prime[MAXN/10], sz;
bitset <MAXN + 1> mark;
 
 

int main() {
  ios_base::sync_with_stdio(0);
  for (int i = 2; i <= MAXN; i++ ){
	if(!mark[i]){
		phi[i] = i-1;
		prime[sz++]= i;
	}
	for (int j=0; j<sz && prime[j]*i <= MAXN; j++ ){
		mark[prime[j]*i]=1;
		if(i%prime[j]==0){
			phi[i*prime[j]] = phi[i]*prime[j];
			break;
		}
		else phi[i*prime[j]] = phi[i]*(prime[j]-1 );
	}
  }
  int t;
  cin>>t;
  
  phi[1]  =1;
  while(t-->0){ 
    int a,b;
    cin>>a>>b;
    int x = __gcd(a,b);
    ll res = 0;
    for(int i=1 ;i<=x; i++) {
      if(x%i==0) res = res+phi[i];
    }
    cout<<res<<endl;
  }
  
  
  return 0;
}
