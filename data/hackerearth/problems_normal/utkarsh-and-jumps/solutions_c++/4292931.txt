#include<iostream>

using namespace std;

#define rep0(i, n) for(int i=0;i<n;i++)
#define rep1(i, n) for(int i=1;i<=n;i++)

double dp[1000001];

double solve(int n, int p){
    if(n < 0) return 0;
    double p1 = (double)p/100;
    double p2 = 1 - (double)p/100;
    if(dp[n] != -1) return dp[n];
    dp[n] = p1*solve(n-2, p) + p2*solve(n-3, p);
    return dp[n];
}
void init(){
    rep0(i, 1000001){
        dp[i] = -1;
    }
}

int main(){
    init();
    dp[0] = 1;
    int n, p;
    cin >> n >> p;
    printf("%.6lf\n", solve(n, p));
}
