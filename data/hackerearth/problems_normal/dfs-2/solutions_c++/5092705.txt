#include <bits/stdc++.h>
using namespace std;

int n;
vector<int> a;
int main() {
    cin >> n;
    a.resize(n);
    for (int i = 0; i < n; ++i) cin >> a[i];
    
    sort(a.begin(), a.end());
    int q; cin >> q;
    while (q--) {
    	int l, r; cin >> l >> r;
    	cout << upper_bound(a.begin(), a.end(), r) - lower_bound(a.begin(), a.end(), l) << '\n';
    }
    return 0;
}
