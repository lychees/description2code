using namespace std;
typedef unsigned long long int ull;
typedef long long int ll;
typedef vector<vector<ll> > mat;
#define N 100005

ll a[N],n,BIT[N][2];

void upd(int idx,int s){
	int i = idx;
	while(idx <= n){
		BIT[idx][0] += 1ll * i * s;
		BIT[idx][1] += s;
		idx += idx&-idx;
	}
}

ll query(int idx, int t){
	if(idx < 1 or idx > n) return 0;
	ll sum = 0;
	while(idx > 0){
		sum += BIT[idx][t];
		idx -= idx&-idx;
	}
  return sum;
}

			cin >> k >> l >> r;
			if(k <= l){
			   ll tmp = query(r, 0) - query(l-1, 0);
			   ll tmp1 = query(r, 1) - query(l-1, 1);
			   ans = 1ll*tmp - 1ll*k*tmp1;
			} else if(k >= r){
				ll tmp = query(r, 0) - query(l-1, 0);
				ll tmp1 = query(r, 1) - query(l-1, 1);
				ans = 1ll*k*tmp1 - 1ll*tmp;
			} else {
				ll tmp1 = query(k-1,1) - query(l-1, 1);
				ll tmp2 = query(k-1, 0) - query(l-1, 0);
				ll tmp3 = query(r, 1) - query(k, 1);

			}

			upd(k, s);
