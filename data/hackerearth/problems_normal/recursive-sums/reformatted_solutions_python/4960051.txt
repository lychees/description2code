t = input()
for _ in range(t):
    n = input()
    sum = 0
    for i in range(n):
        ip = raw_input().split(' ')
        p = int(ip[0])
        q = int(ip[1])
        sum += p*q
    while sum >= 10:
        s = str(sum)
        sum = 0
        for c in s:
            sum += int(c)
    print sum
