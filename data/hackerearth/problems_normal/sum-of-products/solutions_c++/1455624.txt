#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
ll a[1000005];
inline int inp( )
{
    int n = 0 , s = 1 ;
    char p = getchar_unlocked( ) ;
    if( p == '-' )
        s = -1 ;
    while( ( p < '0' || p > '9' ) && p != EOF && p != '-' )
        p = getchar_unlocked( ) ;
    if( p == '-' )
        s = -1 , p = getchar_unlocked( ) ;
    while( p >= '0' && p <= '9' )
    {
        n = ( n << 3 ) + ( n << 1 ) + ( p - '0' ) ;
        p = getchar_unlocked( ) ;
    }
    return n * s ;
}
ll sam(ll p) {
	ll sn;
	sn=(p*(p+1)/2);
	return sn%mod;
}
int main() {
	int t;
	long long n,sum,res,ans,i,j,k,tmp;
	scanf("%d",&t);
	while(t--) {
		scanf("%lld",&n);
		sum=res=ans=k=tmp=0;
		for(i=1;i*i<=n;i++) {
			res=n/i;
			sum=(sum+(i*res)%mod)%mod;
			if(k) {
				j=n/tmp;
				tmp=sam(tmp);
				tmp=(tmp-(sam(res)))%mod;
				sum=(sum+(j*tmp)%mod)%mod;
			}
			k=1;
			tmp=res;
		}
		j=n/tmp;
		res=sam(tmp);
	    ans=sam(j);
	    ans=res-ans;
	    sum=(sum+(j*ans)%mod)%mod;
		printf("%lld\n",sum%mod);
	}
	return 0;
}
