#include<iostream>
#include<cstdlib>
using namespace std;
int cmpfnc(const void *a,const void *b)
{
return(*(unsigned long int *)a-*(unsigned long int *)b);

}
int main()
{
int n,Q,l,r;
int i,j;
cin>>n;
cin>>l;
cin>>r;
cin>>Q;

unsigned long int a[n],qr[Q],num;

for(i=0;i<n;i++)
cin>>a[i];

for(i=0;i<Q;i++)
cin>>qr[i];

qsort(a+l,r-l+1,sizeof(unsigned long int),cmpfnc);

int flag =0;
for(i=0;i<Q;i++)
{
flag=0;
num=qr[i];
for(j=l;j<=r;j++)
{
if(a[j]>num)
{
cout<<a[j]<<"\n";
flag=1;
break;

}

}
if(flag==0)
cout<<-1<<"\n";
}

//for(i=0;i<n;i++)
//cout<<a[i]<<"\n";
return 0;
}
