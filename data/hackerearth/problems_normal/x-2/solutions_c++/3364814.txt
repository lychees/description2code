#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int
#define MX 1000005

ll ans[MX];

int main()
{
    ll t, i, j, x, sum, k, n;
    cin >> t;
    vector<ll> v;
    while(t--) {
        v.clear();
        cin >> n >> k;
        memset(ans, 0, sizeof(ans));
        for(i = 0; i < n; i++) {
            cin >> x;
            v.pb(x);
        }
        j = i = sum = 0;
        while(j < n || i < n) {
            if(j < n && sum + v[j] <= k) {
                sum += v[j];
                j++;
            } else {
                sum -= v[i];
                if(sum <= k && i < n) {
                    ans[1]++;
                    ans[j - i + 1]--;
                }
                i++;
            }
        }
        sum = 0;
        for(i = 1; i <= n; i++) {
            sum += ans[i];
            cout << sum << " ";
        }
        cout << endl;
    }


    return 0;
}
