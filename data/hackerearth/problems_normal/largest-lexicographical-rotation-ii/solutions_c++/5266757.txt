#include <iostream>
#include<algorithm>
#include<vector>
#include<string>
#include<cstring>
using namespace std;

int main()
{string str;
    unsigned i,j,k,num,n,l;
    cin>>str; string res,curr;
    n=str.size();
    str+=str;
    for(i=0,j=1;j<n;)
    {
    	for(k=0;k<n &&str[i+k]==str[j+k];++k);
    	if(k>=n)
    	break;
    	if(str[i+k]>str[j+k])
    	{
    		j+=(k+1);
    	}
    	else{
    	      l=i+k;
    	      i=j;
    	      j=max(l+1,j+1);
    	}
    }
    cout<<str.substr(i,n)<<endl;
    return 0;
}
