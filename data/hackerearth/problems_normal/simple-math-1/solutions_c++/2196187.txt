#include<iostream>
#include<cstdio>
using namespace std;
#define MOD 10000007
int main()
{

    long long dp[103][103];
    int i=0,j=0;
    for(i=0;i<102;i++){
    dp[i][0]=1;
    dp[i][i]=1;
    }
    for(i=1;i<102;i++)
        for(j=1;j<102;j++)
            //if (i!=j)
                dp[i][j] = ( (dp[i-1][j]) % MOD + (dp[i-1][j-1]) %  MOD ) % MOD;
    int n,r;
    scanf("%d%d",&n,&r);
    if(n<r)printf("-1\n");
    else{
    n=n-r;
    printf("%lld",dp[n+r-1][r-1]%MOD);
    }
    return 0;
}
