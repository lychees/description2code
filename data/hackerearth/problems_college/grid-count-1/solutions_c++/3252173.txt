# include<bits/stdc++.h>
using namespace std;
int dp[1000][1000];
int main()
{
    for(int i=0;i<1000;i++)
        for(int j=0;j<1000;j++)
            dp[i][j]=0;
    int n,k;
    long long ans,sum,counter=0,cuber;
    cin>>n>>k;
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
        {
            cin>>((dp[i][j]));
        }
    for(int i=0;i<=n-k;i++)
    {
        sum=0;
        for(int a=i;a<i+k;a++)
        {
            for(int b=0;b<k;b++)
                sum+=dp[a][b];

        }
        cuber=cbrt(sum);
        if(cuber*cuber*cuber==sum||(cuber+1)*(cuber+1)*(cuber+1)==sum)
            counter++;
        for(int j=k;j<n;j++)
        {
            for(int l=i;l<i+k;l++)
            {
                sum+=dp[l][j];
                sum-=dp[l][j-k];
            }
           cuber=cbrt(sum);
       if(cuber*cuber*cuber==sum||(cuber+1)*(cuber+1)*(cuber+1)==sum)
            counter++;

        }

    }
    cout<<counter<<"\n";
}
