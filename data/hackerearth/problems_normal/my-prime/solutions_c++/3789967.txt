#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    
    long long int arr[n];
    for(int i=0;i<n;i++) {
    	cin>>arr[i];
    }
    int x,count=0;
    for(int j=0;j<n;j++) {
    	x=arr[j];
	    for(int i=0;i<n;i++) {
	    	if(arr[i]==x)
	    		continue;
	    	
	    	if(arr[i]<x) {
	    		if((x%arr[i])==0)
	    			count++;
	    	}
	    }
	   	if(count==0)
	    	cout<<x<<" ";
	    count=0;
    }
}
