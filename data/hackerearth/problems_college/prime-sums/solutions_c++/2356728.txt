#include<vector>

#include<assert.h>
#define ll long long
#define l long
#define rep(i,a,b)  for(int i=a;i<b;i++)
bool a[4000001];
 unsigned long long int sum;
int main()
{
	l j,i,n;
    rep(i,2,4000001)
    a[i]=true;
    
    int k=sqrt(4000001);
    rep(i,2,k+1)
    {
    	if(a[i])
    	{
    		for(j=2;i*j<=4000000;j++)
    		a[j*i]=false;
    	}
    }
sum=0;
cin >> n;
for(i=2;i<n;i++)
{
	if(a[i])
	sum=sum+i;
}
cout << sum ;
return 0;
}
