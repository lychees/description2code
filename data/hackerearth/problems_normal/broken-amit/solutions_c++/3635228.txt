#include<bits/stdc++.h>

using namespace std;

long arr[20],value[20];
long sum=0;
int main()
{
    long n;
    cin>>n;
    for(long i=0;i<n;i++)
    {


        cin>>arr[i];
        sum+=arr[i];
    }
    sum*=n;
    for(long i=0;i<n;i++)
    {
        cin>>value[i];
        value[i]*=i;
    }
    sort(value,value+n,greater<long>());
    long k=0;
    long tmp;
    for(long i=0;i<n;i++)
    {

        tmp=0;
        for(long j=1;j<=k;j++)
        {
            sum-=(value[i]*j);
            tmp=value[i]*j;
        }
        sum-=(tmp*(n-i-1));

        ++k;
    }
    cout<<sum;
}
