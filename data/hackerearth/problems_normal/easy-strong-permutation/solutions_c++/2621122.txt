#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long N;
    scanf("%lld",&N);
    long long a[N];
    for(long long i=0;i<N;i++){
    	scanf("%lld",&a[i]);
    }
    sort(a,a+N);
    long long i=0,j=N-1;
    long long max_result=0;
    long long last = j;
    while(i<N/2 || j>N/2){
    	max_result += (a[j]-a[i]);
    	//cout<<i<<" "<<j<<endl;
    	if(last==j){
    		i++;
    		last = i;
    	}else{
    		j--;
    		last = j;
    	}
    }
    max_result += (a[j]-a[0]);
    printf("%lld\n",max_result);
    
    return 0;
}
