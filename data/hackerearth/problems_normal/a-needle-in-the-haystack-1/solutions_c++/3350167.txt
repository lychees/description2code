#include<bits/stdc++.h>
using namespace std;


int main(){
int T;
cin>>T;
while(T--){
    string p,t;
    cin>>p;
    cin>>t;
    int l1=p.length();
    int l2=t.length();
    int matchlen=0;
    bool flag=false;
    map<char,int> m;
    map<char,int> tempm;
    for(int i=0;i<l1;i++){
        if(m.find(p[i])==m.end()){
            m[p[i]]=1;
        }
        else
            m[p[i]]+=1;
    }
    for(int i=0;i<l2;i++){

        if(m.find(t[i])!=m.end()&&m[t[i]]>tempm[t[i]]){
           if(tempm.find(t[i])!=tempm.end()){
              tempm[t[i]]+=1;

           }
           else
            tempm[t[i]]=1;

          matchlen+=1;
        }
        else if(m.find(t[i])!=m.end()&&m[t[i]]==tempm[t[i]]){
            if(matchlen==l1){
                flag=true;
                break;
            }
            else{
               int temp=i-matchlen;
               while(temp<i){
                    if(t[temp]!=t[i]){
                        tempm[t[temp]]-=1;
                        matchlen--;
                        if(tempm[t[temp]]==0){
                            tempm.erase(t[temp]);
                        }
                    }
                    else{
                        tempm[t[temp]]-=1;
                        matchlen--;
                        break;
                    }
                    temp++;

               }
               tempm[t[i]]+=1;
               matchlen++;

            }
        }
        else{
                if(matchlen==l1){
                    flag=true;
                    break;
                }
                else{
                    matchlen=0;
                    if(!tempm.empty())tempm.clear();
                }

        }
    }
    if(flag||matchlen==l1){
        cout<<"YES"<<endl;
    }
    else{
        cout<<"NO"<<endl;
    }



}






}
