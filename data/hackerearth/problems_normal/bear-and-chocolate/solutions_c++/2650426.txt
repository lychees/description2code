#include <iostream>
using namespace std;

int main()
{
	int i,j,k,t,n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
			int row[1000]={0},col[1000]={0};
		char ch;
			getchar();
		for(i=0;i<n;i++)
		{	for(j=0;j<n;j++)
				{
					ch=getchar();
					if(ch=='#')
						{row[i]++;
						col[j]++;
						}
				}
				getchar();	
				}
		for(i=1;i<n;i++)
			{row[i]+=row[i-1];
			col[i]+=col[i-1];
			}
		int f=0;
		for(i=0;i<n-1;i++)
		{
			if(row[i]==row[n-1]-row[i])
			{
				f=1;
				break;
			}
			else if(col[i]==col[n-1]-col[i])
				{
					f=1;
					break;
				}
		}
		if(f==1)
			printf("YES\n");
		else
			printf("NO\n");
		
	}
    return 0;
}
