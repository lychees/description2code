#include<bits/stdc++.h>
using namespace std;
int main()
{
    long int T,i,j,k,aC,bC,cost;
    string str;
    cin>>T;
    while(T--)
    {
        cin>>str;
        cin>>aC;
        cin>>bC;
        i=0;
        j=str.length()-1;
        cost=0;
        while(i<j)
        {
            if(str[i]!=str[j] && (str[i]!='/' && str[j]!='/'))
            {
                cost=-1;
                break;
            }
            else if(str[i]==str[j] && str[i]!='/')
            {
                i++;
                j--;
            }
            else if(str[i]==str[j] && str[i]=='/')
            {
                k=min(aC,bC);
                cost+=2*k;
                i++;j--;
            }
            else if(str[i]=='/' || str[j]=='/')
            {
                if(str[i]=='/' && str[j]=='a')
                    cost+=aC;
                else if(str[i]=='/' && str[j]=='b')
                    cost+=bC;
                else if(str[i]=='a' && str[j]=='/')
                    cost+=aC;
                else if(str[i]=='b' && str[j]=='/')
                    cost+=bC;
                i++;j--;
            }
        }
        cout<<cost<<"\n";
    }
}
