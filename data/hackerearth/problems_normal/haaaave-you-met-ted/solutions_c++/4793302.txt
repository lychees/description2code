#include<iostream>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        long arr[n];
        for(int i=0;i<n;i++){
            cin>>arr[i];
        }
        int min=99999;
        for(int i=0;i<n;i++){
            int count=0;
            long a=arr[i];
            while(a){
                a=a&(a-1);
                    count++;
                //a=a>>1;
            }
            if(count<min)
                min=count;
        }
        cout<<min<<"\n";
    }
}
