#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		int index,n;
		cin>>index>>n;
		if(n==0)
		{
			cout<<index<<endl;
		}
		else if(index == 0 || index==1)
		{
			if(n%2==0)
				cout<<index<<endl;
			else
				cout<<(!index)<<endl;
		}
		else if(index==2)
		{
			n-=1;
			if(n%2==0)
				cout<<"1"<<endl;
			else
				cout<<"0"<<endl;

		}
	}
	return 0;
}
