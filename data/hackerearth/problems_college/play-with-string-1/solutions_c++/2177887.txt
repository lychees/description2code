#include <bits/stdc++.h>
using namespace std;

int main()
{
    char a[100000];
    int flag,t,i,j,l;
    cin>>t;
    while(t--)
    {
    	flag=1;
    	cin>>a;
    	l=strlen(a);
    	for(i=1,j=l-2;i<l && j>=0;i++,j--)
    	{
    		if(!(abs(a[i]-a[i-1])==abs(a[j]-a[j+1])))
    		{
    			flag=0;
    			break;
    		}
    	}
    	if(flag==1)
    		cout<<"Funny\n";
    		else
    		cout<<"Not Funny\n";
    }
    return 0;
}
