#include<bits/stdc++.h>
#define lo long long
using namespace std;
int main()
{
 int t;
 scanf("%d",&t);
 while(t--)
 {
  lo a,b;
  scanf("%lld %lld",&a,&b);
  lo temp1=a;
  lo temp2=b;
  lo rem1,rem2,c=0;
  while(temp1&&temp2)
  {
   rem1=temp1%10;
   rem2=temp2%10;
   if(rem1+rem2+c>=10)
    c++;
   temp1/=10;
   temp2/=10;


  }
  if(c==0)
  printf("No carry operation\n");
  else if(c==1)
  printf("%d carry operation\n",c);
  else
  printf("%d carry operations\n",c);
 }
 return 0;
}
