#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n,b,tmp,tmp2,d1,d2;
		cin>>n>>b;
		tmp = n;
		while(tmp%b!=0)
		tmp--;
		d1 = abs(tmp-n);
		tmp2 = n;
		while(tmp2%b!=0)
		tmp2++;
		d2 = abs(tmp2-n);
		
		if(d1>=d2)
		cout<<tmp2<<endl;
		else
		cout<<tmp<<endl;
		
	}
	return 0;
}
