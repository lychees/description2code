#include<bits/stdc++.h>
using namespace std;

const int N=1e7+10;

bool isPrime[N];

int main() {
  for(int i=2;i<N;i++) {
    if(isPrime[i]==false) {
      for(int j=i+i;j<N;j+=i) {
        isPrime[j]=true;
      }
    }
  }
  vector<int>res;
  for(int i=3;i<=1e7;i++) {
    if(isPrime[i]==false) {
      res.push_back(i);
    }
  }
  set<int>s;
  for(int i=1;i<res.size();i++) {
    s.insert(res[i]-res[i-1]);
  }
  vector<int>fn;
  set<int> :: iterator it;
  for(it=s.begin();it!=s.end();it++) {
    fn.push_back(* it);
  }
  int sz=fn.size();
  int tt;
  cin>>tt;
  while(tt--) {
    int n;
    cin>>n;
    if(n>308||n<4) {
      puts("NO");
      goto hell;
    } else {
      for(int i=0;i<sz;i++) {
        if(2*fn[i]==n) {
          puts("YES");
          goto hell;
        }
      }
      for(int i=0,j=sz-1;i<j;) {
        if(fn[i]+fn[j]==n) {
          puts("YES");
          goto hell;
        } else if(fn[i]+fn[j]<n) {
          i++;
        } else {
          j--;
        }
      }
    }
    puts("NO");
    hell:;
  }
  return 0;
}
