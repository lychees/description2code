#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,i,j;
    long long arr[10000],d,dp[10000];
    cin>>t;
    for(j=1;j<=t;j++)
    {
    	d=-1;
    	
    	cin>>n;
    	fill_n(dp,n,0);
    	for(i=0;i<n;i++)
    	  cin>>arr[i];
    	  if(n==1)
    	  {cout<<"Case "<<j<<": "<<arr[0]<<endl;}
    	  else{
    	  dp[0]=arr[0];dp[1]=max(arr[0],arr[1]);
    	for(i=2;i<n;i++)
          dp[i]=max(arr[i]+dp[i-2],dp[i-1]);
    	for(i=0;i<n;i++)
    	d=max(d,dp[i]);
    	
    	cout<<"Case "<<j<<": "<<d<<endl;
    }
    }
    return 0;
    
    
    
}
