#include <stdio.h>
#include <string>
#include <iostream>
#include <algorithm>

using namespace std;

int compare(string s1, string s2)
{
	if(s1+s2>s2+s1)
	  return 1;
	else
	  return 0;  
}

int main()
{
	int n,i;
	scanf("%d",&n);
    string s[n];
    
	for(i=0;i<n;i++)
	{
	   cin>>s[i];   	
	}
	
	sort(s,s+n,compare);	
	 
	for(i=0;i<n;i++)
	    cout<<s[i];
	    
	return 0;
}
