#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int sum[200005] ;
int main()
{
   int i, j , k, t ,len , count   , ans , n ;
   string str ;
   cin>>t ;
   while(t--)
   {
   	 cin>>n ;
   	 cin>>str ;
   	 len =str.length() ;
   	 memset(sum , 0 , sizeof(sum)) ;
   	 sum[0] = str[0] + 1 - ('a') ;
   	 for(i = 1  ; i < len ; i++)
   	 {
   	 	sum[i] = sum[i-1] +  (str[i] - 'a' + 1) ;
   	 	//cout<<sum[i]<<" ";
   	 }
   //	 cout<<"\n";
   i = 0 , j = 0 ;
   count = 0 ;
   int next = 0 ;
   int prev = 0 ;
   while(j <= len && i <= len)
   {
   	 ans = next - prev ;
   	 
   	 if(ans < n )
   	  {
   	    next = sum[j] ;
   	    j++;
   	  }
   	 else if(ans > n)
	   {
	   	 prev = sum[i] ;
	   	 i++;
	   }
	 else
   	   {
   	 	 next = sum[j] ;
   	 	 prev = sum[i] ;
   	 	 i++ ;
   	 	 j++;
   	 	 count++;
   	   }
   }
   
   cout<<count<<endl;
   
   
   }
   
    return 0;
}
