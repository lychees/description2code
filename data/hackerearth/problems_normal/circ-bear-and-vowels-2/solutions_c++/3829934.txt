#include <iostream>
#include<cstring>
#include<string>
using namespace std;

bool check(char a)
{
	if(a=='a' || a=='e' || a=='i' || a=='o' || a=='u' || a=='y')
	return false;
	return true;
}


int main()
{
    int t,len,i;
    string s;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	len=s.length();
    	int con=0,vow=0,flag=0;
    	for(i=0;i<len;i++)
    	{
    		if(check(s[i]))
    		{
    			if(i+1<len && check(s[i+1]))
    			{
    				if(i+2<len && check(s[i+2]))
    				{	con+=3;
    					flag=1;
    					i+=2;
    					break;
    				}
    				else
    				con+=2;
    				i+=1;
    			}
    			else
    			con+=1;
    		}
    		else
    		vow+=1;
    	}
    	//cout<<vow<<" "<<con<<endl;
    	if(flag==1)
    	{
    		cout<<"hard\n";
    	}
    	else if(con>vow)
    		cout<<"hard\n";
    	else
    		cout<<"easy\n";
    }
    return 0;
}
