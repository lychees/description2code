#include <bits/stdc++.h>
using namespace std;

int up[30][1000];

int go(int x) {
	string s = to_string(x);
	int res = 0;
	for (char c: s) {
		res+=(c-'0')*(c-'0');
	}
	return res;
}

void make() {
	for (int i=0;i<1000;i++) up[0][i] = go(i);
	for (int i=1;i<30;i++) for (int j=0;j<1000;j++) {
		up[i][j] = up[i-1][up[i-1][j]];
	}
}

int N,M;

int main() {
	make();
	int T;
	scanf("%d",&T);
	for (int t=0;t<T;t++) {
		scanf("%d%d",&N,&M);
		N = go(N);
		M-=1;
		for (int i=0;i<30;i++) if (M&(1<<i)) N = up[i][N];
		printf("%d\n",N);
	}
	
    return 0;
}
