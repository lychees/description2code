#include <bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int A[n];
		vector<pair<int,int> > vec(n);
		for(int i=0;i<n;i++){
			cin>>vec[i].first;
			vec[i].second=i;
			A[i]=vec[i].first;
		}
		sort(vec.rbegin(),vec.rend());
		long long int profit=0;
		for(int i=0;i<n;i++){
			int id=vec[i].second;
			if(A[id]>=0){				
				for(int j=id-1;j>=0 && A[j]>=0;j--){
					profit+=A[id]-A[j];
					A[j]=-A[j];
				}
				A[id]=-A[id];
			}			
		}
		cout<<profit<<endl;
	}
	return 0;
}
