using namespace std;

int bits(int n)
{
	int c1=0;
	while(n)
	{
		n=n&(n-1);
		c1++;
	}
	return c1;
}

int main()
{
	int t,n,k,j,sum,x,y;
	scanf("%d",&t);
	while(t--)
	{
		sum=0;
		scanf("%d%d",&n,&k);
		int arr[n+1];
		for(int i=0;i<n;i++)
		{
			scanf("%d",&x);
			y=bits(x);
			arr[i]=y;
		}
		sort(arr,arr+n);
		j=n-1;
		for(int i=0;i<k&&j>=0;i++)
		{
			sum+=arr[j];
			j--;
		}
		cout<<sum<<endl;
	}
	return 0;
}
