#include<bits/stdc++.h>
using namespace std;
inline void inp(long long int &n) 
{
    	n=0;
	register long long int ch=getchar_unlocked();
	long long int sign=1;
    	while(ch<'0'||ch>'9')
	{
		if(ch=='-')
			sign=-1; 
		ch=getchar_unlocked();
	}
    	while(ch>='0'&&ch<='9')
            n=(n<<3)+(n<<1)+ch-'0',ch=getchar_unlocked();
    	n=n*sign;
}
long long int ar[1000001];
int main()
{
	long long int t;
	inp(t);
	while(t--)
	{
		long long int n;
		inp(n);
		for(int i=0;i<n;++i)
			inp(ar[i]);
		long long int max_cur=ar[0];
		long long int max_far=ar[0];
		long long int temp=0;
		int	start_index=0;
		int	end_index=0;
		for(int i=1;i<n;++i)
		{
			if(max_cur+ar[i]>ar[i])
				max_cur=max_cur+ar[i];
			else
			{
				max_cur=ar[i];
				temp=i;
				}
			if(max_cur>max_far)
			{
				max_far=max_cur;
				start_index=temp;
				end_index=i;
			}
		}
		printf("%lld\n",max_far);
	}
	return 0;
}
