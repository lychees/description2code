T = int(raw_input())

passHistory = []

for i in range(T):
	nid = map(int, raw_input().strip().split(' '))
	N = nid[0]
	ID = nid[1]
	passHistory.append(ID)
	
	for j in range(N):
		currPass = raw_input().strip().split(' ')
		passType = currPass[0]
		
		if passType=='P':
			passedToPlayer = int(currPass[1])
			passHistory.append(passedToPlayer)
		else:
			currPlayer = passHistory.pop()
			prevPlayer = passHistory.pop()
			passHistory.append(prevPlayer)
			passHistory.append(currPlayer)
			passHistory.append(prevPlayer)
			
	print 'Player ' + str(passHistory.pop())
