#include <iostream>
using namespace std;

int main()
{
	int T,N,i,j,k,flag;;
    cin>>T;
    while(T--){
    	cin>>N;
    	i=1,k=1,flag=0;
    	while(true){
    		for(j=1;j<=i+1;j++){
    			if(k==N){
    				flag=1;
    				break;
    			}
    			k++;
    		}
    		if(flag)
    		{
    			cout<<i<<endl;
    			break;
    		}
    		i++;
    	}
    }
}
