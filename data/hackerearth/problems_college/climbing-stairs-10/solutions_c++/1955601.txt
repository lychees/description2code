#include<bits/stdc++.h>
using namespace std;

unsigned long long int util(unsigned long long int dp[],int n)
{
	if(n==1)
	{
		dp[n]=1;
		return dp[n];
	}
	if(n==2)
	{
		dp[n]=2;
		return dp[n];
	}
	if(dp[n]!=0)
	return dp[n];
	
	unsigned long long int one = util(dp,n-1);
	unsigned long long int two = util(dp,n-2);
	dp[n]=one+two;
	return dp[n];
}

unsigned long long int ways(int n)
{
	unsigned long long int dp[n+1];
	for(int i=0;i<=n;i++)
	dp[i]=0;
	unsigned long long int ans = util(dp,n);
	return ans;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		cout<<ways(n)<<endl;
	}
	return 0;
}
