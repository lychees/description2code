#include <iostream>
#include<cstdio>
#include<cstring>
using namespace std;
bool pos[1000000];
int main()
{
    int t,i,sz,k;
    long long int sum;
    string str;
    cin>>t;
    while(t--)
   	{
   		
   		k=0;
   		sum=0;
   		cin>>str;
   		sz=str.size();
   		for(i=0;i<sz;i++)
   			if(str[i]=='a'||str[i]=='z')
   				pos[i]=true;
   		
   		for(i=0;i<sz;i++)
   		{
   			if(pos[i])
   				sum+=sz-i;	
   			else
   			{
   				while(k<sz&&pos[k]!=true)
   					k++;	
   				sum+=sz-k;
   			}
   			pos[i]=false;
   		}
   		cout<<sum<<endl;
   	}
    return 0;
}
