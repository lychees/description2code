#include<bits/stdc++.h>

using namespace std;

long long int ans[1000005];
void pre_compute(){
	ans[0]=0;
	ans[1]=0;
	ans[2]=0;
	ans[3]=0;
	ans[4]=0;
	for(long long int i=5;i<=1000000;++i){
		long long int tmp=i;
		long long int cnt=0;
		long long int pw=5;
		while(pw<=tmp){
			cnt+=(tmp/pw);
			pw*=5;
		}
		ans[i]=cnt;
	}
	for(int i=1;i<=1000000;++i)
		ans[i]+=ans[i-1];
}
int main(){
	pre_compute();
	int t;
	scanf("%d",&t);
	while(t--){
		int a,b;
		scanf("%d %d",&a,&b);
		printf("%lld\n",ans[b]-ans[a-1]);
	}
	return 0;
}
