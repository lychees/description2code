#include <iostream>
#define m 100000007
using namespace std;
string s;
long long int p=31;
long long int pw[100001];
long long int h[100001];
void process(){
        pw[0]=1;
        for(int i=1;i<s.length();i++){
                pw[i]=pw[i-1]*p;
                //cout<<pw[i]<<endl;
        }
        h[0]=s[0]-'a'+1;
        for(int i=1;i<s.length();i++){
                h[i]=h[i-1]*p+(s[i]-'a'+1);
                //cout<<h[i]<<endl;
        }
}
long long int hsh(int x,int y){
        if(x==0)return (h[y]);
        return (h[y]-h[x-1]*pw[y-x+1]);
}
int main()
{
        cin>>s;
        process();
        int q;
        cin>>q;
        while(q--){
                int l1,r1,l2,r2;
                cin>>l1>>r1>>l2>>r2;
                //cout<<hsh(l1-1,r1-1)<<endl;
                //cout<<hsh(l2-1,r2-1)<<endl;
                if(hsh(l1-1,r1-1)==hsh(l2-1,r2-1))cout<<"Yes"<<endl;
                else cout<<"No"<<endl;

        }
}
