#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t=0;
    scanf("%d",&t);
    while(t--){
    	int n=0;
    scanf("%d",&n);
    int a[n];
    for(int i=0;i<n;i++)
     scanf("%d",&a[i]);
     int count=0;
     for(int i=0;i<n;i++)
       {
       	  for(int j=i;j<n;j++)
       	  {
       	  	 if(a[i]==a[j])
       	  	   count++;
       	  }
       }
      printf("%d\n",count);
    }
 return 0;
}
