#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
    int t, n, p, x, s;
    bool found;
    scanf("%d", &t);
    while(t--)
    {
    	s=0;
    	scanf("%d%d",&n, &p);
    	found=0;
    	while(n--)
    	{
    		scanf("%d", &x);
    		if(p==x)
    		found=1;
    		if(x>p)
    		s++;
    	}
    	if(found)
    	printf("YES %d\n", s);
    	else
    	printf("NO 0\n");
    }
    return 0;
}
