def pro_con(a):
    hap = 0
    for i in range(len(a)):
        hap += (-a[i][1])
    c = [0]*len(a)

    for i in range(len(a)):
        c[i] = a[i][0]+a[i][1]
    c.sort()
    return hap+c[-1]+c[-2]


for i in range(input()):
    n = input()
    a = [0]*n
    for i in range(n):
        a[i] = map(int, raw_input().split())
    print pro_con(a)
