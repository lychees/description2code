w = raw_input()
alphabet = 'abcdefghijklmnompqrstuvwxyz'
N = int(raw_input())
antonyms = 0
synonyms = 0
for i in range(0, N) :
    current_word = raw_input()
    if alphabet.index(current_word[0]) == alphabet.index(w[0]) + 3 and len(current_word)==len(w) :
        antonyms +=1
    elif current_word[0] ==  w[0] and len(current_word)==len(w):
        synonyms += 1
print( str(synonyms) + " " + str(antonyms))
