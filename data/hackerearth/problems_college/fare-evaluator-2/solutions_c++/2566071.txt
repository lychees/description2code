#include<bits/stdc++.h>
using namespace std;
int main()
{
    int test;
    cin>>test;
    while(test--)
    {
    	int d,w,ans;
    	cin>>d>>w;
    	if(d>60)
    	{
			cout<<"NA"<<endl;
			continue;
		}
    	else if(d<=2)
    	{
    		ans=50;
    	}
    	else
    	{
    		ans=50+(d-2)*12;
    	}
    	ans+=2*w;
    	cout<<ans<<endl;
    }
    return 0;
}
