#include<iostream>

using namespace std;

long long count_one(long long n)
{
    long long count =0;
    while(n)
    {
        n=n&(n-1);
        count++;
    }
    return count;
}


int main()
{
    int test;
    cin>>test;
    while(test--)
    {
        long long a,b;
        cin>>a>>b;
        long long tmp = a^b;
        long long ans = count_one(tmp);
        cout<<ans<<endl;
    }
}
