#include<stdio.h>
#include<algorithm>

using namespace std;
int main()
{
    long long int n,k,i,c=1,p,flag=0,s=0;
    scanf("%lld %lld",&n,&k);
    int a[n];
    for(i=0;i<n;i++)
    {
                    scanf("%d",&a[i]);
    }
    sort(a,a+n);
    i=n-1;
    while(flag==0)
    {
                       p=k;
                    while(p--)
                    {
                              if(i<0)
                              {flag=1;
                              break;
                              }
                              s=s+(a[i]*c);
                              // printf("%lld",i);
                              i--;
                             
                    }
                    if(flag==1)
                    break;
                    c++;
    }
    printf("%lld\n",s);
    s=0;
    c=0;
    flag=0;
    
  //  getch();
    return 0;
}
