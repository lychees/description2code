#include<iostream>
#include<string>
#include<cstdio>
#define gc getchar
using namespace std;

typedef long long int ll;

long long int read() {
	char c = gc();
	while(c<'0' || c>'9') c = gc();
		long long int ret = 0;
	
	while(c>='0' && c<='9') {
	ret = 10 * ret + c - 48;
	c = gc();
	}

	return ret;

}
int main()
{
    long long int i,n,x,count=0,m;
    m=3*100000000;
    n=read();

    for(i=0;i<n;i++)
    {
        x=read();
        if(x<m)
            count++;
    }
    cout<<count;
    return 0;
}
