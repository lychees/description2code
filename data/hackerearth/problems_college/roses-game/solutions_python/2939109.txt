for case in xrange(cases):
	n = int(raw_input())
	a = [int(i) for i in raw_input().split(' ')]
	
	visited = [0]
	win = False
	
	loc = a[0]
	
	while loc not in visited:
		visited.append(loc)
		if loc==-1:
			win = True
			break
		else:
			loc = a[loc]
	
	if win:
		print "Win"
	else:
		print "Lose"
