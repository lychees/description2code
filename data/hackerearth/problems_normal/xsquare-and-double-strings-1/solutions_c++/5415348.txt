#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		int f[26]={0};
		int i=0;
		while(s[i]!='\0')
		f[s[i]-'a']++,i++;
		int flag=0;
		for(int i=0;i<26;i++)
		{

			if( f[i]>=2)
			{
				flag=1;
				break;
			}
		}
		if(s.length()==1)
		cout<<"No"<<endl;
	else	if(flag==1)
		cout<<"Yes"<<endl;
	else if(flag==0)
		cout<<"No"<<endl;
		s.clear();

	}
}
