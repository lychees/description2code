#include <iostream>
#include <algorithm>
using namespace std;
struct point{
	int x,y;
};
struct slides{
	point end1,end2;
};

int comp(slides a,slides b){
	if(max(a.end2.y,a.end1.y)==max(b.end2.y,b.end1.y)){
		if(a.end2.x==b.end2.x)
		 	return a.end1.x>b.end1.x;
		 return a.end2.x>b.end2.x;
	}
	
	return max(a.end2.y,a.end1.y)>max(b.end2.y,b.end1.y);
}
int main()
{
	int x,n,xn;
    cin >> x >> n;
    slides a[n];
    for(int i=0;i<n;i++){
    	cin >> a[i].end1.x >> a[i].end1.y >> a[i].end2.x >> a[i].end2.y;
    }
    sort(a,a+n,comp);
    int i=0;
    xn=x;
    while(1){
    	if(i>(n-1))
    		break;
	    while((i<n)&&(!(xn>=a[i].end1.x&&xn<=a[i].end2.x))){
			i++;
		//	cout << i <<" dff" <<endl;
	    }
	   // cout << i <<" sad" <<endl;
	    if(i>(n-1))
    		break;
	    if(a[i].end2.y>a[i].end1.y)
	    	xn=a[i].end1.x;
	    else
	    	xn=a[i].end2.x;
	    i++;
    }
    cout << xn << endl;
    // for(int i =0;i<n;i++){
    // 	cout << a[i].end1.x <<" "<< a[i].end1.y <<" "<< a[i].end2.x <<" "<< a[i].end2.y <<endl;
    // }
    return 0;
}
