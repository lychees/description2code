def swaps(arng,reqarng):
	count = 0 
	for i in range(len(arng)):
		if arng[i]!=reqarng[i]:
			count = count + 1
	return count/2


t = int(input())
for i in range(t):
	n = int(input())
	arng = raw_input()
	for j in range(2*n):
		if j==0:
			reqarng = "B"
		else:
			if j%2==0:
				reqarng = reqarng + "B"
			else:
				reqarng = reqarng + "G"
	s1 = swaps(arng,reqarng)
	s2 = swaps(arng,reqarng[::-1])
	if s1>s2:
		print s2
	else:
		print s1
