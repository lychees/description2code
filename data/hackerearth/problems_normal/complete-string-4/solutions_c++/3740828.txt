
#include <vector>
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
bool complete(vector<bool> t)
{
    bool b=false;
    unsigned int i=0;
    while(t[i]!=false) i++;
    if (i>=t.size()) b=true;
    return b;

}
int main()
{
    int n;
    cin>>n;
    vector<string> s(n);
    vector<bool> b(26);
    for(int i=0;i<n;i++)
    {
        cin>>s[i];
        for(int j=0;j<26;j++) b[j]=false;
        for(int c=0;c<s[i].size();c++)
            b[s[i][c]-97]=true;
        if(complete(b)) cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
    }
    return 0;

}
