#include <iostream>
#include <cstdio>
#include <vector>
#include <string>
#include <algorithm>
#include <cstring>

#define MAX(x,y) ((x)>(y)?(x):(y))
#define MIN(x,y) ((x)<(y)?(x):(y))
#define pb push_back
#define mp make_pair
#define rep(i,n) for(long long i=0;i<n;i++)
#define repr(i,j,n) for(long long i=j;i<=n;i++)
#define endl '\n'
using namespace std;
typedef long long ll;
typedef vector<ll> vll;
const ll maxn = (ll) 1e6+9;
//const ll mod = (ll) 1e9+7;
ll a[maxn];
//ll dp[1024][1024];

int main()
{
	std::ios::sync_with_stdio(0);
	ll i,j,k,t,x,y,q,m,l,r,n,mx=0,mod;
	cin>>t;
	while(t--)
	{
		mx=0;
		cin>>x>>y>>n>>mod;
		a[x]++;a[y]++;
		mx=MAX(mx,MAX(x,y));
		for(i=2;i<n;i++)
		{
			j=(x+y)%mod;
			a[j]++;x=y;y=j;
			mx=MAX(mx,j);
		}
		ll ans=0;
		for(i=0;i<=mx;i++)
		{
			ans+=a[i]*a[i];
		}
		cout<<ans<<endl;
		memset(a,0,sizeof(ll)*(mx+1));
	}
}
