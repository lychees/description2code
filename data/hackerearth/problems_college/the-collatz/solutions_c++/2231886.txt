#include <iostream>
using namespace std;

long long ce,co,st;

void rec(long long n)
{
	if(n==1)
	{
		st++;
		co+=n;
		return;
	}
	if(n%2==0){
		st++;ce+=n;rec(n/2);
	}
	else{
		st++;co+=n;rec(3*n+1);}
}
	
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	long long N;
    	ce=co=st=0;
    	cin>>N;
    	rec(N);
    	//cout<<ce<<" "<<co<<" "<<st<<" ";
    	st--;
    	if(ce%st>co%st)
    		cout<<"Even Rules\n";
    	else if(ce%st<co%st)cout<<"Odd Rules\n";
    	else cout<<"Tie\n";
    }
    return 0;
}
