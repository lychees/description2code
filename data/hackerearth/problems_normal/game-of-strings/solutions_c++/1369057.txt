#include<iostream>
#include<cstring>
#include<cstdlib>
#include<stdio.h>
using namespace std;
long long int L[5001][5001];
void lcs( char *X, char *Y, int m, int n )
{
   
   for (int i=0; i<=m; i++)
   {
     for (int j=0; j<=n; j++)
     {
       if (i == 0 || j == 0)
         L[i][j] = 0;
       else if (X[i-1] == Y[j-1])
         L[i][j] = L[i-1][j-1] + 1;
       else
         L[i][j] = max(L[i-1][j], L[i][j-1]);
     }
   }
   cout<<L[m][n]<<endl;
}
 
/* Driver program to test above function */
int main()
{
  char X[5001],Y[5001];
  scanf("%s %s",X,Y);
  char a[5001],b[5001];
  int pos=0;
  for(int i=0;i<strlen(X);i++)
  {
  	if(X[i]=='a'||X[i]=='e'||X[i]=='i'||X[i]=='o'||X[i]=='u')
  		continue;
  	else
  	{
  		a[pos]=X[i];
  		pos++;
  	}
  }
  pos=0;
  for(int i=0;i<strlen(Y);i++)
  {
  	if(Y[i]=='a'||Y[i]=='e'||Y[i]=='i'||Y[i]=='o'||Y[i]=='u')
  		continue;
  	else
  	{
  		b[pos]=Y[i];
  		pos++;
  	}
  }
  //cout<<m<<" "<<n<<endl;
  lcs(a, b,strlen(a),strlen(b));
  return 0;
}
