using namespace std;
int n, t;

int main(){
    scanf("%d", &t);
    while(t--){
        scanf("%d", &n);
        int x , y;
        bool b = false;
        for(int i = n / 3; i >= 0 ; i--){
            int c = n - 3 * i;
            if(c % 5 == 0 && c >= 0){
                x = i * 3;
                y = c;
                b = true;
                break;
            }
        }
        if(b){
            while(x--)putchar('5');
            while(y--)putchar('3');
            puts("");
        }
        else puts("-1");
    }
    return 0;
}
