#include <bits/stdc++.h>
#define ll long long
using namespace std;
int a[10010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	vector<int> v[110];
	int n;
	cin>>n;
	int i;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		v[a[i]].push_back(i);
	}
	int q;//queries
	cin>>q;
	int l,r,val,ans;
	vector<int>::iterator it1,it2;
	for(i=0;i<q;i++)
	{
		cin>>l>>r>>val; //left & right index & value
		it1=lower_bound(v[val].begin(),v[val].end(),l);
		it2=lower_bound(v[val].begin(),v[val].end(),r);
		if(it1==v[val].end())
		{
			ans=0;
		}
		else
		{
			if(*it1>r)
			{
				ans=0;
			}
			else if(*it2==r)
			{
				ans=(it2-it1)+1;
			}
			else
			{
				ans=it2-it1;
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
