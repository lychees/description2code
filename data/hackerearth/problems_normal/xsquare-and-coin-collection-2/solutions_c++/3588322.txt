#include<bits/stdc++.h>
using namespace std;
#define LL long long
int main()
{
         int t;
         cin>>t;
         while(t--)
         {
                LL  n,m;
                cin>>n>>m;
                LL  a[n];
                for(int i=0;i<n;i++)
                {
                              cin>>a[i];

                }

                LL  mx=0;
                LL  sum=0;
                for(int i=0;i<n;i++)
                {
                       if(a[i]<=m)
                       {
                             sum+=a[i];
                             mx=max(mx,sum);
                       }
                       else
                       {
                              sum=0;
                       }
                }
                cout<<mx<<endl;
         }
         return 0;
}
