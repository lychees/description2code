#include <iostream>
using namespace std;

int main()
{	int f=0;
	cin>>f;
	while(f>0)
	{	
		int n;
		cin>>n;
		float a[n];
		float b[n+1];
		for(int i=0;i<n;i++)
		{	
			cin>>a[i];
		}
		int i=0;
		for(i=0;i<n;i++)
		{	int t=a[i];
			b[t]=i+1;
		}
		for(i=0;i<n;i++)
		{
			if(a[i]!=b[i+1])
				break;
		}
		if(i!=n)
		{	cout<<"not inverse";
		}
		else
		{	cout<<"inverse";
		}
		if((f-1)!=0)
		{
			cout<<endl;
		}
    	f--;
	}
    return 0;
}
