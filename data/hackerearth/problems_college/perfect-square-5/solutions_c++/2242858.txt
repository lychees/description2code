#include<iostream>
#include<math.h>
using namespace std;
int main(){
long long T,num,temp1,prev,nxt;
double temp2;
cin>>T;
while(T--){
cin>>num;
temp1=sqrt(num);
temp2=sqrt(num);
if(temp1==temp2)
	cout<<"YES"<<endl;
else{
temp1=sqrt(num);
prev=pow(temp1,2);
nxt=pow(temp1+1,2);
	if((num-prev)<(nxt-num))
		cout<<"-"<<(num-prev)<<endl;
	else
		cout<<"+"<<(nxt-num)<<endl;
}
}
return 0;
}
