#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
    	int a,b;
    	cin>>a;
    	cin>>b;
    	int flag=0;
    	int count=0;
    	for(int i=a;i<=b;i++)
    	{
    		for(int j=2;j<=sqrt(i);j++)
    		{
    			if(i%j==0)
    			{
    				flag=1;
    				 break;
    			}
    		}
    		if(flag==0&&i!=1)
    		{
    			count++;
    		}
    		flag=0;
    	}
    	cout<<count<<endl;
    }
}
