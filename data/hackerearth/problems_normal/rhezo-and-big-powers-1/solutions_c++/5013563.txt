#include<bits/stdc++.h>
#define mod 1000000007
#define phi 1000000006
#define ll long long
using namespace std;
ll power(ll a,ll b)
{
	ll ans=1;
	a=a%mod;
	while(b)
	{
		if(b&1)ans=(ans*a)%mod;
		a=(a*a)%mod;
		b>>=1;
	}
	return ans;
}
int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	ll a;
	cin>>a;
	string b;
	cin>>b;
	int len=b.length();
	ll ans=0;
	for(int i=0;i<len;i++)
	{
		 ans=(ans*10+(ll)(b[i]-'0'))%phi;
	}
	//cout<<a<<" "<<b<<endl;
	cout<<power(a,ans);
	return 0;
}
