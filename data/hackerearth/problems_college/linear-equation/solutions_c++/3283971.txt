#include <iostream>
#include<cstdio>
using namespace std;

int main()
{	int t,n,i;
	long arr[50001];
	long long sum=0;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		sum=0;
		for(i=0;i<n;i++)
		{	scanf("%ld",&arr[i]);
			sum+=arr[i];
		}
		sum/=(n-1);
		for(i=0;i<n;i++)
		{
			printf("%lld ",(sum-arr[i]));
		}
		printf("\n");
	}
    return 0;
}
