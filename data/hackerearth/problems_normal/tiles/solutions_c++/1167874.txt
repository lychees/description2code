long long mod =1000000007;

struct mat {
	vector< vector<ull> > C;
	int N;
	mat(int n) {
		N =n;
		C.resize(N,vector<ull>(N,0));}
	mat() {}
};

mat prod(mat &A, mat &B) {
	mat ret(A.N);
	for(int i =0; i < A.N; i++) for(int j =0; j < A.N; j++)
		for(int k =0; k < A.N; k++) ret.C[i][j] =(ret.C[i][j]+A.C[i][k]*B.C[k][j])%mod;
	return ret;}

mat pw(mat &A, int e) {
	if(e <= 0) {
		mat ret(A.N);
		for(int i =0; i < A.N; i++) ret.C[i][i] =1;
		return ret;}
	mat B =pw(A,e/2);
	mat C =prod(B,B);
	if(e%2 == 0) return C;
	else return prod(A,C);}
