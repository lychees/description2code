#include <iostream>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long int n,k,c = 0;
    	scanf("%lld",&n);
    	k = n;
    	while(k>0)
    	{
    		if(k%2==1)
    		{
    			c++;
    		}
    		k = k/2;
    	}
    	printf("%lld\n",c);
    }
    return 0;
}
