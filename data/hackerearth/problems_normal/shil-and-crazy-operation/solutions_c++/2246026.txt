const int N = int(1e5)+10;
int A[N];
int P[N];
VI g[N];
int comp[N];
int sub[N];
int cno;
int vis[N];
int ans[N];
VI chain[N];
int pos[N];
void dfs(int u)
{
	vis[u]=1;
	comp[u]=cno;
	sub[cno]++;
	pos[u]=SZ(chain[cno]);
	chain[cno].PB(u);
	for(int i=0;i<SZ(g[u]);i++)
		if(!vis[g[u][i]])
			dfs(g[u][i]);
}
int main()
{
	int n;
	LL t;
	si(n);sll(t);
	for(int i=1;i<=n;i++)
		si(A[i]);
	for(int i=1;i<=n;i++)
	{
		int x;si(x);
		g[i].PB(x);
	}
	for(int i=1;i<=n;i++)
		if(!vis[i])
			dfs(i),cno++;
	for(int i=1;i<=n;i++)
	{
		int id = comp[i];
		int frwd = t % sub[id];
		int nxt = chain[id][(pos[i]-frwd + SZ(chain[id]))%SZ(chain[id])];
		ans[nxt]=A[i];
	}
	for(int i=1;i<=n;i++)
		printf("%d ",ans[i]);
	printf("\n");
	return 0;
}
