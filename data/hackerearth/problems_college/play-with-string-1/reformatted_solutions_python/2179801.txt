tc = int(input())
while tc > 0:
    s = raw_input()
    t = s[::-1]
    flag = 0
    for i in range(1, len(s)):
        if abs(ord(s[i])-ord(s[i-1])) != abs(ord(t[i])-ord(t[i-1])):
            flag = 1
            break

    if flag == 1:
        print 'Not Funny'
    else:
        print 'Funny'
    tc = tc-1
