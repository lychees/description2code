#include <iostream>
using namespace std;
string toBinary(int n)
{
	string s;
	while(n!=0) {s=(n%2==0 ? '0':'1')+s; n/=2;}
	return s;
}
bool isSparse(string s)
{
	int size=s.size();
	for(int i=0; i<size-1; ++i)
	{
		if(s[i]=='1' && s[i+1]=='1')
			return false;
	}
	return true;
}
int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	
    	while(1)
    	{
	    	if(isSparse(toBinary(n)))
	    	{
	    		cout<<n<<endl;
	    		break;
	    	}
	    	else
	    		n++;
    	}
    	
    }
    return 0;
}
