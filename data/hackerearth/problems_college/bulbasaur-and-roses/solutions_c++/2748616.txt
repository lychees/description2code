#include <iostream>
using namespace std;

void add(char a[],char b[],char c[])
{
	int temp=0;
	int i;
	int t;
	for(i=0;i<200;i++)
	{
		t=(b[i]-'0')+(c[i]-'0')+temp;
		temp=t/10;
		a[i]=char('0'+t%10);
	}
}

void multiply(char a[],char b[],char c[])
{
	int temp=0;
	int ctr=0;
	int t;
	for(int i=0;i<200;i++)
	{
		for(int j=0;j<200;j++)
		{
			t=(a[j+ctr]-'0')+(b[j]-'0')*(c[i]-'0')+temp;
			temp=t/10;
			a[j+ctr]=char('0'+t%10);
		}
		ctr++;
	}
}
void disp(char a[])
{
	int chk=0;
	for(int i=199;i>=0;i--)
	{
		if(a[i]!='0' || chk==1)
		{
			chk=1;
			cout<<a[i];
		}
		
	}
	cout<<endl;
}
int main()
{
    int t;
    cin>>t;
    char a[22][200];
    for(int i=0;i<21;i++)
    {
    	for(int j=0;j<200;j++)
    	a[i][j]='0';
    }
    a[0][0]='1';
    a[1][0]='2';
    int chk=0;
    for(int i=2;i<21;i++)
    {
    	if(chk==1)
    	{
    		add(a[i],a[i-1],a[i-2]);
    		chk=0;
    	}
    	else if(chk==0)
    	{
    		multiply(a[i],a[i-1],a[i-2]);
    		chk=1;
    	}
    }
    
   while(t--)
    {
	    int n;
	    cin>>n;
	    disp(a[n-1]);
    }
  	
    return 0;
}
