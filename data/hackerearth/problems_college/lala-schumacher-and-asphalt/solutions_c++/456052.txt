#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long int t,n,i,fx[100009],fy[100009];
	long long int wnit,nit,dis,x,y;
	long long int finx,finy,disfrom;
    scanf("%lld",&t);
    //printf("t=%lld\n",t);
    while(t--)
    {
    	
    	scanf("%lld",&n);
    //	printf("t=%lld n=%lld\n",t,n);
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&fx[i]);
    	}
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&fy[i]);
    	}
    	scanf("%lld%lld",&finx,&finy);	
    	scanf("%lld%lld",&wnit,&nit);
    //	printf("finx=%lld finy=%lld wnit=%lld nit=%lld\n",finx,finy,wnit,nit);
    	long long int mini=(abs(finx)+abs(finy))*wnit;
    //	printf("mini=%lld\n",mini);
    	for(i=0;i<n;i++)
    	{
    		long long int x=fx[i];
    		long long int y=fy[i];
    		dis=abs(x)+abs(y);
    		disfrom=abs(finx-x)+abs(finy-y);
    		disfrom=disfrom*nit+dis*wnit;
    	//	printf("disfrom=%d\n",disfrom);
    		if(disfrom<mini)
    		mini=disfrom;
    	}
    	printf("%lld\n",mini);
    }
    return 0;
}
