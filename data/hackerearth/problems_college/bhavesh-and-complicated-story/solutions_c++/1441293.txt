struct FenwickTree {
	typedef long long T;
	vector<T> v;
	void init(int n) { v.assign(n, 0); }
	void add(int i, T x) {
		for(; i < (int)v.size(); i |= i+1) v[i] += x;
	}
	T sum(int i) const {	//[0, i)
		T r = 0;
		for(-- i; i >= 0; i = (i & (i+1)) - 1) r += v[i];
		return r;
	}
	T sum(int left, int right) const {	//[left, right)
		return sum(right) - sum(left);
	}
};

int main() {
	int N;
	scanf("%d", &N);
	FenwickTree ft; ft.init(100003);
	rep(i, N) {
		int x;
		scanf("%d", &x);
		ft.add(x, +1);
	}
	int Q;
	scanf("%d", &Q);
	rep(ii, Q) {
		int ty;
		scanf("%d", &ty);
		if(ty == 0) {
			int X, Y;
			scanf("%d%d", &X, &Y);
			long long ans = ft.sum(X, Y+1);
			printf("%lld\n", ans);
		}else if(ty == 1 || ty == 2) {
			int Z;
			scanf("%d", &Z);
			ft.add(Z, ty == 1 ? -1 : +1);
		}
	}
	return 0;
}
