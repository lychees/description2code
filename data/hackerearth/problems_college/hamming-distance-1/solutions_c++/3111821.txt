#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int a,b;
		cin>>a>>b;
		int c = 0,p=0;;
		int rem1,rem2;
		while(p<=32)
		{
			rem1 = a%2;
			rem2 = b%2;
			p++;
			if(rem1!=rem2)
			{
				c++;
			}
			a = a/2;
			b = b/2;
		}
		cout<<c<<"\n";
	}
}
