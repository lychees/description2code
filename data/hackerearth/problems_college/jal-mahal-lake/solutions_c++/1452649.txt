#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int main() {
	int t;
	scanf("%d",&t);
	while (t--) {
		int n;
		scanf("%d",&n);
		long long int a[n];
		long long int l[n];
		for (int i = 0; i < n; i++) {
			scanf("%lld",&a[i]);
		}
		for (int i = 0; i < n; i++) {
			scanf("%lld",&l[i]);
		}
		for (int i = 0; i < n-1; i++) {
			for (int j = i+1; j < n; j++) {
				if (a[j] < a[i]) {
					long long int temp = a[i];
					a[i] = a[j];
					a[j] = temp;
					temp = l[i];
					l[i] = l[j];
					l[j] = temp;
				}
			}
		}
		int f = 1;
		long long int energy = 0LL;
		int i = 0;
		while (i < n) {
			long long int cc = 1LL;
			long long int m1 = 0LL,m2 = 0LL;
			for (int j = n-1; j > i; j--) {
				long long int temp = a[j-1]+l[j-1];
				temp = a[j]-temp;
				m1 += cc*temp;
				cc++;
			}
			cc = 1LL;
			for (int k = 0; k < i; k++) {
				long long int temp = a[k]+l[k];
				temp = a[k+1]-temp;
				m2 += cc*temp;
				cc++;
			}
			i++;
			if (f) {
				energy = m1+m2;
				f = 0;
			}
			else
			energy = min(energy,m1+m2);
		}
		printf("%lld\n",energy);
	}
	return 0;
}
