    set<string>mp;
    int add,shift;
    void dfs( string tmp)
    {
      //cout<<tmp<<endl;
      mp.insert(tmp);
      string next=tmp;
      int i,n=tmp.size();
      for(i=0;i<n;i++)
      {
        next[(i+shift)%n]=tmp[i];
      }
      if(mp.find(next)==mp.end())
        dfs(next);
      next=tmp;
      for(i=0;i<n;i++)
      {
        if(i%2)
        next[i]= '0'+ (next[i]-'0'+add)%10;
      }
      if(mp.find(next)==mp.end())
        dfs(next);
      return;
    }

        cin>>inp>>add>>shift;
        mp.clear();

      }
     return 0;
