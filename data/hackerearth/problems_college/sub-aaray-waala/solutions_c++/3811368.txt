#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
long long big_power(long long n,long long p)
{
    long long tmp;
    if(p==0)
        return 1;
    tmp=big_power(n,p/2);
    tmp=((tmp%mod)*(tmp%mod))%mod;
    if(p%2!=0)
        return ((tmp%mod)*(n%mod))%mod;
    else
        return tmp%mod;
}
int main()
{
	ll t,n,a,d,ans;
	cin>>t;
	while(t--)
	{
		set<ll> s;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>a;
			s.insert(a);
		}
		d=s.size();
	
	ans=big_power(n,d);
	cout<<ans<<"\n";
	s.clear();
	}
}
