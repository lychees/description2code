#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    if(t>10 || t<1)
    	cout<<"Invalid Test"<<endl;
    else{
    	while(t--){
    		string s;
    		cin>>s;
    		if(s.length()<1 || s.length()>100)
    			cout<<"Invalid Input"<<endl;
    		else{
    			int u=0,l=0;
    			for(int i=0;i<s.length();i++){
    				int j = s[i];
    				if(j>=65 && j<=90)
    					u++;
    				else if(j>=97 && j<=122)
    					l++;
    			}
    			if(u==0 && l==0)
    				cout<<"Invalid Input"<<endl;
    			else if(u<l)
    				cout<<u<<endl;
    			else
    				cout<<l<<endl;
    		}
    	}
    }
}
