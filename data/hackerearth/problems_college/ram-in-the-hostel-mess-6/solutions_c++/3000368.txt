#include <iostream>
using namespace std;
typedef long long ll;
int main()
{
 
 	ll t,n;
 	cin>>t;
 	
 	ll ar[4][100001];
 	
 	while(t--)
 	{
 		cin>>n;
 	
 		ar[0][0]=ar[1][0]=ar[2][0]=ar[3][0]=1;
 		//3 implies both emp.
 		
 		for(int i=1;i<n;i++)
 		{
 			ar[0][i]=ar[1][i-1]+ar[3][i-1];
 			ar[1][i]=ar[0][i-1]+ar[3][i-1];
 			ar[2][i]=ar[3][i-1];
 			ar[3][i]=ar[1][i-1]+ar[2][i-1]+ar[0][i-1]+ar[3][i-1];
 		}
 	
 		cout<<ar[1][n-1]+ar[0][n-1]+ar[2][n-1]+ar[3][n-1]<<endl;
 	
 	
 	}
 
 
 	return 0;
}
