#include<iostream>
#include<math.h>
#include<stdio.h>
using namespace std;
int f(long long base,long long n,long long m)
{
	if(n==0)
	 return 1;
	if(n==1)
	  return base;
	long long k=f(base,n/2,m);
	if(n%2==0)
	  return (k%m*k%m)%m;
	else
	  return (((k%m*k%m)%m)*base%m)%m;
}
int main()
{
	int k;
	long a,b,t;
	scanf("%ld",&t);
	while(t--)
	{
		int sum=0;
		scanf("%ld %ld %d",&a,&b,&k);
		long j=pow(10,k);
		a=f(a,b,j);
		//cout<<a<<" ";
		//a=a%j;
		while(k--)
		{
			sum+=a%10;
			a/=10;
		}
		printf("%d\n",sum);
	}

return 0;
}
