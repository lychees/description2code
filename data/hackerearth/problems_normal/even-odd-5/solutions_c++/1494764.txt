#include <stdio.h>
 
int main()
{
	unsigned  long long i,j,t,r,c,co,ce,k;
	scanf("%llu",&t);
	while(t--)
	{
	co=ce=0;
	scanf("%llu %llu",&r,&c);
	if(r%2==0){i=r/2;j=i;}
	else{
		i=r/2;
	    j=i+1;
	}
	if(r==c || c>r)
	{
		ce=i*(i+1);
		co=j*j;
	}
	else
	{
		k=(c/2);
		ce=k*(k+1);
		co=k*k;
		i=i-k;
		j=j-k;
		ce+=i*c;
        co+=j*c;		
	}
	i=j=k=0;
    printf("%llu %llu\n",ce,co);
	}
	
    return 0;
}
