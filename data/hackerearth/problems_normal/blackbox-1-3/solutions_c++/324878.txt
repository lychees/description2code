long long int max(long long int a,long long int b)
{
	if(a>b)
	return a;
	else
	return b;
}
long long int gcd(long long int m,long long int n)
{
    long long int r;

    /* Check For Proper Input */
    if((m == 0) || (n == 0))
        return 0;
    

    do
    {
        r = m % n;
        if(r == 0)
            break;
        m = n;
        n = r;
    }
    while(true);

    return n;
}
