#include<iostream>
using namespace std;
void qsort(int a[], int first, int last)
{
    int pivot, j, temp, i;

     if(first<last){
         pivot=first;
         i=first;
         j=last;

         while(i<j){
             while(a[i]<=a[pivot]&&i<last)
                 i++;
             while(a[j]>a[pivot])
                 j--;
             if(i<j){
                 temp=a[i];
                  a[i]=a[j];
                  a[j]=temp;
             }
         }
         temp=a[pivot];
         a[pivot]=a[j];
         a[j]=temp;
         qsort(a,first,j-1);
         qsort(a,j+1,last);
    }
}
int main()
{
	int n, i, j;
	cin>>n;
	int a[n], b[n], c[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		b[i]=a[i];
	}
	qsort(a,0,n-1);

    for(i=0;i<n;i++)
        c[n-1-i]=a[i];

    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
       {
           if(b[i]==c[j])
           {
               cout<<j+1<<" ";
               break;
           }
        }
    }
return 0;
}
