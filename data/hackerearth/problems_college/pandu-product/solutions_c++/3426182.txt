#include <iostream>
using namespace std;

int main()
{
    long long int n,t,m,x,i,m1=-1000000000,m2=-1000000000,m3=0,m4=0,fm;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>t;
    	if(t>=0)
    	{
    		if(t>=m1)
    		{
    			m2=m1;
    			m1=t;
    		}
    		else  if(t>m2)
    			m2=t;
    	}
    	else if(t<0)
    	{
    		if(t<=m3)
    		{
    			m4=m3;
    			m3=t;
    		}
    		else if(t<m4)
    			m4=t;
    	}
    	t=0;
    }
    m=m1*m2;
    x=m3*m4;
    fm=(m>=x)?m:x;
    cout<<fm;
    return 0;
}
