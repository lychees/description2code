#include <iostream>
#include <stdlib.h>
using namespace std;
int compare (const void * a, const void * b)
{
  return ( *(long long int*)a - *(long long int*)b );
}
int main() {
	int t;
	long long int n,d,ar[100000],count;
	cin>>t;
	while(t--)
	{
		count=0;
		cin>>n>>d;
		for(long long int i=0;i<n;i++)
		cin>>ar[i];
		qsort (ar, n, sizeof(long long int), compare);
		for(long long int i=0;i<n-1;i++)
		{
			if((ar[i+1]-ar[i])<=d)
			{
				count++;
				i++;
			}
			
			
		}
		cout<<count<<endl;
		
	}
	return 0;
}
