void sort(long long a[], long long low, long long  high)
    {
       long long  N = high - low;
        if (N <= 1)
            return;
        long long mid = low + N/2;
        // recursively sort
        sort(a, low, mid);
        sort(a, mid, high);
        // merge two sorted subarrays
        long long  temp[1005];
        long long  i = low, j = mid;
        for (long long  k = 0; k < N; k++)
        {
            if (i == mid)
                temp[k] = a[j++];
            else if (j == high)
                temp[k] = a[i++];
            else if (a[j]<a[i])
                temp[k] = a[j++];
            else
                temp[k] = a[i++];
        }
        for (long long  k = 0; k < N; k++)
            a[low + k] = temp[k];
    }
