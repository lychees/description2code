#include<cstdio>
#include<cmath>
#include<cstdlib>
#include<climits>
#include<algorithm>
using namespace std;
int cmp(const void *a,const void *b)
{
   if(((int *)a)[0]-((int *)b)[0]==0)
      return ((int *)a)[1]-((int *)b)[1];
   else
      return((int *)a)[0]-((int *)b)[0];
}
int main()
{
   int test;
   scanf("%d",&test);
   while(test--)
   {
   int n;
   scanf("%d",&n);
   int arr[n][2];
   for(int i=0;i<n;i++)
   {
      scanf("%d",&arr[i][0]);
      arr[i][1]=i;
   }
   qsort(arr,n,2*sizeof(int),cmp);
   int ans=INT_MIN;
   for(int i=0;i<n;i++)
      ans=max(ans,(int)fabs(arr[i][1]-i));
   printf("%d\n",ans);
   }
   return 0;
}
