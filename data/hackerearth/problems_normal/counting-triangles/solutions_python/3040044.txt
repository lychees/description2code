def count_triangles(D):
    count = 0
    for key in D.keys():
        if D[key] == 1:
            count += 1
    print(count)
    
def test():
    N= int(raw_input())
    counter =0
    D={}
    while counter <N:
        x =[int(x) for x in raw_input().split(' ')]
        x.sort()
        key = ''.join([str(m) for m in x])
        if key in D:
            D[key] +=1
        else:
            D[key]=1
            
        counter += 1

    count_triangles(D)

test()
