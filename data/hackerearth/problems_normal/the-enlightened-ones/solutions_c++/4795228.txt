#define _CRT_SECURE_NO_WARNINGS
#include <iostream>

using namespace std;

int A[10000000] = { 0, };

int max_pos = 10000000;

int compare(const void * a, const void * b)
{
	return (*(int*)a - *(int*)b);
}
bool check(int arr[], int N, int range, int monks)
{
	int tmp_val = 0;
	
	tmp_val = arr[0] + 2*range;
	monks--;

	for (int i = 1; i < N; i++)
	{

		if ((tmp_val) >= arr[i])
		{
			continue;
		}

		if (monks == 0) return false;

		monks--;
		tmp_val = arr[i] + 2*range;
	}

	return true;
}

int main()
{
	int K, N, i, start, end, mid;

#ifdef _MSC_VER
	freopen("Text.txt", "r", stdin);
#endif

	cin >> N;
	cin >> K;

	for (i = 0; i < N; i++)
	{
		cin >> A[i];
	}

	qsort(A, N, sizeof(int), compare);

	start = 1;
	end = max_pos;

	while (start <= end)
	{
		mid = start + (end - start) / 2;

		if (check(A, N, mid, K))
		{
			if (check(A, N, mid - 1, K))
			{
				end = mid - 1;
				continue;
			}
			else
			{
				break;
			}

		}
		else
		{
			start = mid + 1;
		}


	}
	cout << mid << endl;

	return 0;
}
