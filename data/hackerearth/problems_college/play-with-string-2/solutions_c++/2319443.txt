#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	
	for(;t>0;t--)
	{
		string s;
		cin>>s;
		
		int i=1,j=s.size()-2;
		
		int f=0;
		while(i<=s.size()-1)
		{
			if(abs(s[i]-s[i-1])!=abs(s[j]-s[j+1]))
			{
				f=1;
				break;
			}
			i++;
			j--;
		}
		
		if(f==0)
		cout<<"Funny\n";
		else
		cout<<"Not Funny\n";
	}
	
	return 0;
}
