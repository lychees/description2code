#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int i,j,k,n,t;
	cin>>n;
	long long int a[n+1],l=1,r=n,ans,mid;
	cin>>j;
	a[1]=j;
	for(i=2;i<=n;i++)
	{
		cin>>j;
		a[i]=a[i-1]+j;
	}
	sort(a+1,a+n+1);
/*	for(i=1;i<=n;i++)
		cout<<a[i]<<" ";*/
ans=-1;
	cin>>t;
	while(t--)
	{
		cin>>j;
		l=1;r=n;
		while(l<=r)
		{
			mid=(l+r)/2;
			if(j>a[n])
			{
				ans=-1;
				break;
			}
			else if(j<a[1])
			{
			ans=1;
			break;
			}
			else if(j==a[mid])
			{
				ans=mid;
				break;
			}
			else if(j>a[mid] && j<a[mid+1])
			{
				ans=mid+1;
				break;
			}
			else if(j>a[mid])
			{
				l=mid+1;
			}
			else if(j<a[mid] && j>a[mid-1])
			{
				ans=mid;
				break;
			}
			else
				r=mid-1;
		}
		cout<<ans<<endl;
	}

	return 0;
}
