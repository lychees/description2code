records = int(raw_input())
timeslots = []
for recordno in xrange(records):
    timeslot = raw_input()
    (start, end) = timeslot.split('-')
    timeslots.append((start, end))

timeslots.sort(key=lambda x:x[0])

bol_need = False
index = 0
while index < len(timeslots)-1:
    if timeslots[index+1][0] < timeslots[index][1]:
        bol_need = True
        break
    index += 1
if bol_need:
    print "Will need a moderator!"
else:
    print "Who needs a moderator?"
