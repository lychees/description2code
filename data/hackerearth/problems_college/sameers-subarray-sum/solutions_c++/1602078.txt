#include<algorithm>
#include<iostream>
using namespace std;
int main()
{
ios_base::sync_with_stdio(false);
int n,q;
cin>>n;
char a[n+2];
long long int sum[n+2];
fill(sum,sum+n+2,0);
cin>>a;
cin>>q;
for(int i=1;i<=n;i++)
{
sum[i]=sum[i-1]+a[i-1]-'0';
//cout<<sum[i]<<" ";
}
while(q--)
{
int l,r;
cin>>l>>r;
int temp=sum[r]-sum[l-1];
if(temp%9==0)
cout<<"SAMEER"<<endl;
else
cout<<"NO"<<endl;
}
return 0;
}
