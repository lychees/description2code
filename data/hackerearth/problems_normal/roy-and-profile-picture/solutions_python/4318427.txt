L = int(raw_input())
T = int(raw_input())

for i in range(T):
	(W,H) = map(int,raw_input().split())
	if(W<L or H<L):
		print "UPLOAD ANOTHER"
	elif(W==H):
		print "ACCEPTED"
	else:
		print "CROP IT"
