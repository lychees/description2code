#include <iostream>
using namespace std;
#define lld long long int
lld gcd(lld a,lld b)
{
	lld k;
	while(b>0)
	{
		k=b;
		b=a%b;
		a=k;
	}
	return a;
}
int main()
{
   lld t,n,res,a,b;
   scanf("%lld",&t);
   while(t--)
   {
   	res=1;
   	scanf("%lld%lld",&n,&a);
   	res=a;
   	for(int i=1;i<n;i++)
   	{
   		scanf("%lld",&b);
   		res=gcd(res,b);
   	}
   	printf("%lld\n",(n*res));
   }
    return 0;
}
