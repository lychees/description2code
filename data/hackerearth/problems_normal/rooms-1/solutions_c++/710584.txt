#include<stdio.h>
#include<algorithm>
#include<utility>
#include<vector>

using namespace std;

int main()
{
	int t;
	
	scanf("%d",&t);
	
	while(t--)
	{
		int n,i,count=0,k=0,max=0;
		
		scanf("%d",&n);
		
		long int a[n],x[n];
		vector< pair<long int, long int> > times(2*n);
		
		for(i=0;i<n;i++)
		scanf("%ld",&a[i]);
		
		for(i=0;i<n;i++)
		scanf("%ld",&x[i]);
		
		for(i=0;i<n;i++)
		{
			times[k].first=a[i];
			times[k++].second=1;
			
			times[k].first=a[i]+x[i];
			times[k++].second=-1;
		}
		
		sort(times.begin(),times.end());
		
		for(i=0;i<2*n;i++)
		{
			//printf("%d\n",times[i].first);
			count+=times[i].second;
			
			if(count > max)
			max=count;
		}
		
		printf("%d\n",max);
	}
	
	return 0;
}
