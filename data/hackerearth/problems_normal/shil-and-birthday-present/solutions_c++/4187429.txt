#include <iostream>
using namespace std;

int main()
{
    long int n,i,c,max,a[1000000],aux[1000000];
    cin>>n;
    c=n;
    for(i=0;i<n;i++)
    cin>>a[i];
    max=a[0];
    for(i=0;i<n;i++)
    if(max<a[i])
    max=a[i];
    for(i=0;i<=max;i++)
    aux[i]=0;
    for(i=0;i<n;i++)
    aux[a[i]]++;
    c=0;
    for(i=0;i<=max;i++)
    if(aux[i]!=0)
    c++;
    c=(c*(c-1))/2;
    cout<<c;
}
