#include<iostream>
using namespace std;

int gcd(int m,int n)
{
if(m==0) return n;
if(n==0) return m;
if(m>n) return gcd(m-n,n);
else  return gcd(m,n-m);
}
int main(){
int A,n,*x;
register int k,c=0;
cin>>A;
cin>>n;
x=new int[n];
for(int i=0;i<n;i++)
{
cin>>x[i];
 k=gcd(A,x[i]);
if(k>1) c++;

}
cout<<c<<endl;
}
