#include <bits/stdc++.h>

using namespace std;

#define ll long long int
#define sl(x) scanf("%lld", &x)

ll a[100005];

bool check(ll m, ll n, ll M)
{
    ll sum = 0, i;
    for (i = 1; i <= n; ++i) {
        if (a[i] > m)
            sum += (a[i]-m);
    }
    return (sum >= M);
}

ll b_search(ll n, ll m)
{
    ll l = 0, r = 1000000000000000000LL, mid;
    while (r-l > 1) {
        mid = (l+r)/2;
        if (check(mid, n, m))
            l = mid;
        else
            r = mid-1;
    }
    if (check(r, n, m))
        return r;
    if (check(l, n, m))
        return l;
    return -1;
}

int main()
{
    ll n, m, i;
    sl(n); sl(m);
    for (i = 1; i <= n; ++i)
        sl(a[i]);
    printf("%lld\n", b_search(n, m));
    return 0;
}
