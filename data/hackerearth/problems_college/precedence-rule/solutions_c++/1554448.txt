    #include<bits/stdc++.h>
    using namespace std;
     
    int main() {
    int t;
    cin>>t;
    while(t--)
    {
    char s[32];
    cin>>s;
    int l=strlen(s);
    long long ans=s[l-1]-48;
    for(int i=l-3;i>=0;i-=2)
    {
    int tmp=s[i]-48;
    if(s[i+1]=='+')
    ans+=tmp;
    else
    if(s[i+1]=='*')
    ans*=tmp;
    else
    if(s[i+1]=='-')
    ans-=tmp;
    }
    cout<<ans<<endl;
    }
    return 0;
    }
