#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t, n, a[5];
	cin >> t;
	a[0] = 1;
	a[1] = 2;
	a[2] = 3;
	a[3] = 5;
	a[4] = 7;

	while (t--)
	{
	    int flag = 0;
		cin >> n;
		for (int i = 0; i <= 4; i++)
        {
            if (a[i] == n)
            {
                flag = 1;
                cout << 1 << endl;
            }
        }

		if (flag != 1)
		{
			int c = 0;
			for (int i = 4; i >= 0; i--)
			{
				while (a[i] <= n)
				{
					c++;
					n = n - a[i];
				}
			}
			cout << c << endl;
		}
	}
	return 0;
}
