#include<stdio.h>
int hashtable[20];
int hashsize=20;
int hashfunc(int x){
	return x%10;
}
int main()
{
  int t;	
  scanf("%d",&t);
  for(int i=0;i<t;i++){
  	int n=0;
  	scanf("%d",&n);
  	int a[n];
  	for(int i=0;i<n;i++){
  		scanf("%d",&a[i]);
  	}
  	int index=0;
  	int collision=0;
  	for(int i=0;i<20;i++){
	    hashtable[i]=-1;
	  }
	  for(int i=0;i<n;i++){
	      index=hashfunc(a[i]);
	      if(hashtable[index]!=-1){
	          collision++;
	      }
	      else
	        hashtable[index]=a[i];
	  }
  	printf("%d \n",collision);
  }
  return 0;
}
