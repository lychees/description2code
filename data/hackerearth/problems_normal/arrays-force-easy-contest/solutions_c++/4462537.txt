#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    long long t;
    cin>>t;
    while(t--){
    	long long c,d,n,m,i;
    	cin>>c>>d>>n>>m;
    	long long b[m],a[n];
    	for(i=0;i<m;i++)
    		b[i]=0;
    	a[0]=c;
    	b[a[0]%m]++;
    	a[1]=d;
    	b[a[1]%m]++;
    	for(i=2;i<n;i++){
    		a[i] = (a[i-1]+a[i-2])%m;
    		b[a[i]]++;
    	}
    	long long x =0;
    	for(i=0;i<m;i++)
    		x += b[i]*b[i];
    	cout<<x<<endl;
    }
    return 0;
}
