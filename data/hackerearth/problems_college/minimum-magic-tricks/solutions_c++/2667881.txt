#include <iostream>
using namespace std;

long long int ans1(int h,int n){
	long long int sum=0;
	long long int temp=1;
	
	for(int i=0;i<=n-1;i=i+2){
		sum = (sum + temp)%1000000007;
		temp = (temp*4)%1000000007;
	}
	
	return sum;
}

long long int ans2(int h,int n){
	long long int sum=0;
	long long int temp=2;
	
	for(int i=2;i<=n-1;i=i+2){
		sum = (sum + temp)%1000000007;
		temp = (temp*4)%1000000007;
	}
	return sum;
}

int main()
{
