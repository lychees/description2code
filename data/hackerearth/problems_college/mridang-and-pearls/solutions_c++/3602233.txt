#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N = 1000001;

vector<pair<int,int > > ans;
map<int ,int > mp;
set<int > s;
int main() {
    int x,n,l;
    scanf("%d",&n);
    l = 1;
    for(int i = 1; i <= n; i++) {
        scanf("%d",&x);
        if(s.count(x)) {
            ans.push_back(make_pair(l,i));
            s.clear();mp.clear();
            l = i+1;
        }
        else {
            mp[x] = i;
            s.insert(x);
        }
    }
    if(!ans.size()) puts("-1");
    else {
          printf("%d\n",ans.size());
          if(ans[ans.size()-1].second != n) ans[ans.size()-1].second = n;
    for(int i = 0; i < ans.size(); i++) printf("%d %d\n",ans[i].first,ans[i].second);
    }

    return 0;
}
