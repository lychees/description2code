#include <iostream>
#include <algorithm>

using namespace std;

int check(long int A[],int mid,int N,int K)
     {
         int position=A[0]+mid;
         K--;
         for(int i=1;i<N;i++)
         {
             if((position+mid)>=A[i])
                continue;
             if(K==0)
                return 0;
             position=A[i]+mid;
             K--;
         }
     return 1;
     }
int main()
{
    int N,K,mid,l,r,check_mid,check_mid_before;
    long int A[100000];
    cin>>N>>K;
    for(int i=0;i<N;i++)
        cin>>A[i];
    sort(A,A+N);
    l=0;
    r=10000000;
    while(l<=r)
    {
        mid=(l+r)/2;
        check_mid=check(A,mid,N,K);
        check_mid_before=check(A,mid-1,N,K);
        if((check_mid==1)&&(check_mid_before==0))                // Deciding the region
            break;
        else if((check_mid==0)&&(check_mid_before==0))
            l=mid+1;
        else
            r=mid-1;
    }
    cout<<mid<<endl;
}
