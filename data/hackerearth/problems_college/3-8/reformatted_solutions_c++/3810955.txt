#include <algorithm>
#include <deque>
#include <iostream>
using namespace std;

int main() {
  int t;
  cin >> t;
  while (t--) {
    string s;
    cin >> s;
    deque<char> q;
    deque<char>::iterator it;
    int i;
    for (i = 0; s[i] != '\0'; i++) {
      if (q[0] <= s[i])
        q.push_front(s[i]);
      else
        q.push_back(s[i]);
    }
    for (it = q.begin(); it < q.end(); it++)
      cout << *it;
    cout << endl;
  }
  return 0;
}
