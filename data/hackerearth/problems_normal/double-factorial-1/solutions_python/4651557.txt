one="720"
two="620448401733239439360000"
 
for i in range(int(raw_input())):
	n,m=map(int,raw_input().split())
	if m==0:
		print 0
 
	else:
	
		if n==0:
			print 1
		elif n==1:
			print 1
		elif n==2:
			print 2
		elif n==3:
			if m>=3:
				print 720
			else:
				print one[3-m:]
		elif n==4:
			if m<=4:
				print 0
			else:
				if m==17:
					m-=1
				if m==22:
					m-=1
				print two[24-m:]	
		
		else:
			print 0
