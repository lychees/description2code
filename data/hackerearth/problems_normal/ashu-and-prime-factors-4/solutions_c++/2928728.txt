#include<bits/stdc++.h>
using namespace std;

int main()
{
    int sieve[1000005]={0},ans[1000005]={0},T,x;
    for(int i=2;i<=1000000;i++)
    {
    	if(sieve[i]==0)
    	{
    	for(int j=i;j<=1000000;j+=i)
    	{
    		if(sieve[j]==0)
    		{
    		ans[i]++;
    		sieve[j]=1;
    		}
    	}
    	}
    	
   }
  /* for(int i=2;i<=1000000;i++)
   {

   }*/
    cin>>T;
    while(T--)
    {
    	cin>>x;

    }
    return 0;
}
