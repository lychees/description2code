#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int m,n;
	cin>>n>>m;
	int a[n][m],row[n],col[m],mx=0;
	map <int, int> hm1;
	for(int i=0;i<n;i++)
	row[i]=0;
	for(int j=0;j<m;j++)
	col[j]=0;
	for(int i=0;i<n;i++)
	for(int j=0;j<m;j++)
	cin>>a[i][j];
	
	for(int i=0;i<n;i++)
	for(int j=0;j<m;j++)
	{
	
		row[i]+=a[i][j];
	}
	
	for(int j=0;j<m;j++)
	for(int i=0;i<n;i++)
	{
		col[j]+=a[i][j];
	}
	
	for(int i=0;i<n;i++)
	for(int j=0;j<m;j++)
	{
		int t =row[i]^col[j];
		hm1[t]=t;
		if(t>mx)
		mx=t;
	}
	cout<<mx<<endl;
    return 0;
}
