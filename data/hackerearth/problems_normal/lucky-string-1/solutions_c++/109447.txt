#include <iostream>
using namespace std;

typedef long long int LL;

int main()
{
    int t; cin >> t;
	while(t--)
	{
		LL n; cin >> n;
		int digits = 1;LL y = 2;
		LL tmp;
		for(tmp = n-digits*y;tmp>0;tmp -= digits*y) {y *= 2; digits++;}
		
		n = tmp+digits*y;
		LL number;
		if(n%digits==0) number = n/digits-1;
		else number = n/digits;
		
		int idx = (digits-n%digits)%digits;
		int arr[digits];
		for(int i=0;i<digits;i++)	arr[i]=4;
		
		int i=0;
		while(number>0)
		{
			arr[i++] = (number%2==0)?4:5;
			number /= 2;
		}
		
		if(arr[idx] == 4){
			cout << "Hacker\n";
		}
		else 
			cout << "Earth\n";
	}
}
