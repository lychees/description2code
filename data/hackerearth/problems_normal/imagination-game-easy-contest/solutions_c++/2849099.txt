#include <cstdio>

using namespace std;

const int N = 100010;
int n, a[N];

void solve()
{
	scanf("%d", &n);
	for (int i = 1; i <= n; i++) scanf("%d", &a[i]);
	int s = 0;
	for (int i = 1; i <= n; i++)
	{
		int x;
		scanf("%d", &x);
		s ^= (x - 1) ^ (a[i] - x);
	}
	if (!s) printf("Second\n"); else
	printf("First\n");
}

int main()
{
	int t;
    scanf("%d", &t);
    while (t--) solve();
    return 0;
}
