inline long long readInt(){

n = (n<<3) + (n<<1) + c - '0', c = gc();

}

        snum[i++]=a%10+48;

    pc(snum[i--]);

}

int calSum(long long n){
	int sum = 0;
		while(n>0){
			sum += (n%10);
			n /= 10;
		}
	return sum;
}
