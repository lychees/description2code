#include<iostream>
#include<bits/stdc++.h>

using namespace std;

void tribonacci(int n)
	{	int a[n+1];
		a[0]=1;
		a[1]=1;
		a[2]=2;
		for(int i=3;i<=n;++i)
			a[i]=a[i-1]+a[i-2]+a[i-3];
		cout<<"Jack-"<<a[n]<<endl;
	}
void fibonacci(int n)
	{
		int a[n+1];
		a[0]=1;
		a[1]=1;
		for(int i=2;i<=n;++i)
			a[i]=a[i-1]+a[i-2];
		cout<<"Jill-"<<a[n]<<endl;
	}

int main()
	{
		int n;
		cin>>n;
		if(n<0)
			cout<<"ERROR"<<endl;
		else{
			tribonacci(n);
			fibonacci(n);
		}
		return 0;
	}
