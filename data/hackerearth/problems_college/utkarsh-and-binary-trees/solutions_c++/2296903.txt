#include <iostream>
using namespace std;

long long MOD = 1000000007;
long long mat[510][510];

long long solve(int nm, int ht)
{    
    for(int j=0; j<=ht; j++)
        mat[0][j] = 1;
    for(int i=1; i<=nm; i++)
        for(int j=1; j<=ht; j++)
            for(int k=1; k<=i; k++)
                mat[i][j] = (mat[i][j] + mat[k-1][j-1]*mat[i-k][j-1])%MOD;

    return mat[nm-1][ht-1];
}
int main()
{
    int nm,ht;
    cin>>nm>>ht;
    long long val=0;
    solve(nm,ht);
    for(int p=1;p<=nm;p++)
        val = (val + ((p*mat[p-1][ht-1])%MOD*mat[nm-p][ht-1])%MOD)%MOD;
    cout<<val<<endl;
    return 0;
}
