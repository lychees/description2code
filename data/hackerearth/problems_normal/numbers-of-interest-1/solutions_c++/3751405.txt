#include <bits/stdc++.h>
using namespace std;

static int loop,sum;
vector<long int>::iterator iter;

int main()
{
	int t,n,x,y;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d %d",&x,&y,&n);
		vector< long int > v(x,y);
		for(loop=1;loop<=(n-x);loop++)
		{
			iter=v.end();
			v.push_back( accumulate( v.end()-x , v.end() , (long int)0 ) );
		}
		printf("%ld\n",*(v.begin()+n-1));
	}

	return 0;
}
