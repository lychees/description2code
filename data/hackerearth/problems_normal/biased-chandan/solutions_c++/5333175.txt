#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int T;
    int c;
    vector<int>v;
      vector<int>::iterator itr;
      vector<int>::iterator itr1;
      vector<int>::iterator itr2;
      vector<int>::iterator itr3;
      vector<int>::iterator itr4;
      vector<int>::iterator itr5;
      int sum=0;
      cin >> T;
    while(T--) {
    	cin >> c;
    	v.push_back(c);
    }
    itr=v.begin();
    while(itr!=v.end()) {
    	if((*itr)==0 &&  itr!=v.begin()) {
    		itr3 = itr;
    		itr2=itr-1;
    		itr=itr-2;
    		v.erase(itr3);
    		v.erase(itr2);
    	}
    	if((*itr)==0 && itr==v.begin()+1) {
    		itr4=itr-1;
    		v.erase(itr4);
		}
    	itr=itr+1;
    }
    for(itr1=v.begin();itr1!=v.end();itr1++) {
    	sum = sum+(*itr1);
    }
    cout << sum << endl;
    return 0;
}
