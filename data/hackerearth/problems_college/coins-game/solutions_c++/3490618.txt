#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>

using namespace std;

int a[100010];
long long dp[100010];

int main()
{
    int z;
    cin>>z;
    while(z--)
    {
        int n;
        cin>>n;;
        for (int i = 0; i < n; ++i)
        {
            cin>>a[i];
        }
        long long sum = 0;
        dp[n] = dp[n + 1] = dp[n + 2] = 0;
        for (int i = n - 1; i >= 0; --i)
        {
              sum += a[i];
              dp[i] = sum - min(min(dp[i + 1], dp[i + 2]), dp[i + 3]);
         }
        cout<<"\n"<<dp[0];
    }
    return 0;
}
