#include<algorithm>
#include<cstdio>
#include<iostream>
#define MOD 1000000007
#define ll long long
using namespace std;
  
  bool comp(int x,int y){
      return x>y;
  }
  
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		ll arr[n];
		for(int i=0;i<n;i++){
			scanf("%lld",&arr[i]);
		}
		sort(arr,arr+n,[](ll x,ll y){return x>y;});
		ll fact=n-1;
		ll sum=0;
		for(int i=0;i<n;i++){
		    sum=sum+(fact*arr[i]);
		    fact-=2;
		}
		ll maxi=arr[0]%MOD;
		sum%=MOD;
		sum=(sum*maxi)%MOD;
		cout<<sum<<endl;
	}
}
