#include<stdio.h>
#include<algorithm>
#include<math.h>
#include<string.h>
#include<iostream>
using namespace std;

int main()
{
	
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	int i=0,j=0,t,len,l=0,k=0;
	char arr[2][1000000];

	cin>>t;
	while(t--)
	{
	
	i=0;
	j=0;
	scanf("%s",arr[0]);
	scanf("%s",arr[1]);
	
	len=strlen(arr[0]);
	
	int flag=0;
	
	while(i!=len&&j!=len)
	{
		while(arr[0][i]=='#' && i<len-1)
		i++;
		
	    while(arr[1][j]=='#' && j<len-1)
		j++;
		
		
		if(arr[0][i]!=arr[1][j])
		{
			flag=0;
			break;
		}
		
		
		if(arr[0][i]=='A'&&(i-j)>=0)
		{
		 // printf("A matches\n");
		  flag=1;
		  i++;
		  j++;
		  continue;
		}
		
		if(arr[0][i]=='B'&& (i-j)<=0)
		{
			
		// printf("B matches\n");
		  flag=1;
		  i++;
		  j++;
		  continue;
		}
		
		if(arr[0][i]=='#')
		{
			i++;
			j++;
			flag=1;
		}
		
		else
		{
			flag=0;
			break;
		}
		
		
	}
	
			if(flag==0)
			printf("No\n");
			else
			printf("Yes\n");
	
 
    }
	return 0;
}
