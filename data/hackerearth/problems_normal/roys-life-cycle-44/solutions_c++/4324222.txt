#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	int n,maxX=0,maxY=0,temp1=0,temp2=0,j;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		string str;
		cin>>str;
		for(j=0;j<str.length();j++)
		{
			if(str[j]=='C')
			{
				temp1++;
				temp2++;
			}
			maxX=max(temp1,maxX);
			maxY=max(temp2,maxY);
			if(str[j]!='C')
			{
				temp1=0;
				temp2=0;
			}
		}
		temp1=0;
	}
	cout<<maxX<<" "<<maxY<<endl;
	return 0;
}
