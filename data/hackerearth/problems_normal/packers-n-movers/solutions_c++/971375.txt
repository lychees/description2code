#include<bits/stdc++.h>
using namespace std;
#define maxn 101
#define F(i,p,n) for(i=p;i<n;i++)
#define I(i,p,q) for(i=p;i>=q;i--)
#define Ss(x) scanf("%s",x)
#define S(x) scanf("%d",&x)
#define P(x) printf("%d\n",x)
#define Ps(x) printf("%d ",x)
#define LL long long int
#define inf 0x7fffffff
int a[maxn];
main()
{
    int i,j,m,n,lo,hi,mid,mm=-1,sum=0,mx=0;
    S(m);
    S(n);
    F(i,0,n)
    S(a[i]);
    F(i,0,n)
    {
        sum+=a[i];
        if(a[i]>mm)
            mm=a[i];
    }
    lo=mm;
    hi=sum;
   // printf("lo=%d hi=%d\n",lo,hi);
    int ans;
    while(lo<hi)
    {
        mid=(hi+lo)/2;
        int req=0,cap=0;
        for(j=0;j<n;j++)
        {
            if(cap+a[j]>mid)
            {
                req++;
                cap=a[j];
            }
            else
                cap+=a[j];
            //printf("cap=%d\n",cap);
        }
        if(cap)
            req++;
           // printf("mid=%d req=%d\n",mid,req);
        if(req>m)
            lo=mid+1;
        else
            hi=mid;
           // printf("lo=%d hi=%d\n",lo,hi);
    }
    //(lo);
    sum=0;
    F(j,0,n)
    {
        if(sum+a[j]>lo)
            sum=a[j];
        else
            sum+=a[j];
        if(sum>mx)
            mx=sum;
    }
    P(mx);
}
