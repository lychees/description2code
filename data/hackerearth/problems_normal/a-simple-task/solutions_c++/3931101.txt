#include <iostream>
#include <algorithm>

using namespace std;

int gcd(int a, int b)
{
	if (b == 0) return a;
	return gcd(b, a % b);
}

int main()
{
	int i, j, k, n;
	cin >> n;
	int arr[n];
	
	for (i = 0; i < n; ++i) {
		cin >> arr[i];
	}
	
	sort(arr, arr + n);
	
	for (i = 0; i < (n - 1); ++i) {
		arr[i] = arr[i+1] - arr[i];
	}
	
	k = arr[0];
	for (i = 1; i < (n - 1); ++i) {
		k = gcd(k, arr[i]);
	}
	
	for (i = 2; i <= k; ++i) {
		if ((k % i) == 0) {
			cout << i << " ";
		}
	}
	cout << endl;
	
	return 0;
}
