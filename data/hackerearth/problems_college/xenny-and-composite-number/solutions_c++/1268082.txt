#include <bits/stdc++.h>

using namespace std;

int check(int n);

int main()
{
	int t;
	
	cin >> t;
	
	while(t--)
	{
	int n;
	
	cin >> n;
	
	check(n);
	}
	return 0;
}

int check(int n)
{
	int i;
	int j;
	
	for(i = n+1;i <= 100000;i++){
          int flag = 0;
          for(j = 2;j <= sqrt(i);j++){
                if((i%j) == 0){
                         flag = 1;
                         break;
                }
          }
          if(flag == 1 && i != 1){
              printf("%d\n",i);
              break;
          }
       }
}
