using namespace std;
#define pb push_back
#define sd(x) scanf("%d",&x)
#define sld(x) scanf("%lld",&x)
#define ll long long
#define mod 1000000007
long long func(ll l)
{
ll i=1;
while(1)
{
    if((i*(i+1))/2>=l)
        break;
        i++;
}
return i;
}
int main()
{
int t;
sd(t);
while(t--)
{
ll sum=0;
ll l,r;
sld(l);
sld(r);
ll pos_l,pos_r;
ll pos_ll,pos_rr;
pos_l=func(l);
pos_r=func(r);
pos_ll=((pos_l)*(pos_l-1))/2;
pos_rr=((pos_r)*(pos_r-1))/2;
if(pos_l==pos_r)
{
if(((r-pos_rr)&1)&&((l-pos_ll)&1))
{
  sum+=pos_l;
}
else if((!((r-pos_rr)&1))&&(!((l-pos_ll)&1)))
{
sum-=pos_l;
}
}
else
{
if((l-pos_ll-1)&1)
    sum-=pos_l;
if((r-pos_rr)&1)
    sum+=pos_r;
pos_r/=2;
pos_l/=2;
sum+=((pos_r*pos_r)-(pos_l*pos_l));
}
cout<<sum<<endl;
}
return 0;
}
