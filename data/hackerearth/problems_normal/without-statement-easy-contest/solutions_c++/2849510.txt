#include<bits/stdc++.h>
using namespace std;
int solve(int n)
{
	int sum=0;
	while(n)
	{
		int x=n%10;
		sum+=x*x;
		n=n/10;
	}
return sum;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,m,i=1;
    	scanf("%d%d",&n,&m);
    	int vis[1000]={0};
    	n=solve(n);
    	m--;
    	int indx[1000];
    	vector<int> dp;
    	dp.push_back(n);
    	vis[n]=1;
    	indx[n]=i++;
    	while(m--)
    	{
    		n=solve(n);
    		if(!vis[n])
    		{
    			dp.push_back(n);
    			vis[n]=1;
    			indx[n]=i++;
    		}
    		else break;
    	}
    	if(m>0)
    	{
    		int d=indx[n]-1;
    		int k=dp.size()-d;
    		m=m%k;
    		n=dp[d+m];
    	}
    	printf("%d\n",n);
    }
    return 0;
}
