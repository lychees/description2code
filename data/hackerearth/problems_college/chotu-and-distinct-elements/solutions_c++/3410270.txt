#include<bits/stdc++.h>
using namespace std;
int a[1000005] ;
int main()
{
	int t ;
	scanf("%d",&t);
	while(t--)
	{
		int n,i ,count=0;
		scanf("%d",&n);
		set<int> s ;
		for(i =0 ;i<n;i++)
		{
			scanf("%d",&a[i]) ;
			s.insert(a[i]) ;
		}
		
		set<int> :: iterator it1= s.begin() ;
		set<int> :: const_iterator it2 = s.end();
		--it2 ;
		int k = s.size() ;
		
		if(!(k&1))
		{
			while(it1!=it2&&count<k)
			{
				printf("%d %d ",*it2,*it1) ;
				++it1 ;
				--it2 ;
				count+=2 ; 
			}
		}
		else
		{
			while(it1!=it2)
			{
				printf("%d %d ",*it2,*it1) ;
				++it1 ;
				--it2 ;
			}
			printf("%d",*it2) ;
		}
		printf("\n") ;
	}
}
