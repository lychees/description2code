#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int hk[n+1],ak[n+1];
	int color[100001]={0};
	for(int i=1;i<=n;i++)
	{
		cin>>hk[i]>>ak[i];
		color[hk[i]]++;
	}
	int h,a,temp;
	for(int i=1;i<=n;i++)
	{
		h=n-1;
		a=n-1;
		temp=color[ak[i]];
		h=h+temp;
		a=a-temp;
		cout<<h<<" "<<a<<endl;
	}
}
