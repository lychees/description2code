#include <bits/stdc++.h>
using namespace std;
string s,t;
int slength;
int tlength;
bool isSubsequence(int mid)
{
	int i = 0;
//	cout<<tlength<<endl;
	for(int j=0;j<tlength;j++)
	{
//	    cout<<"i="<<i<<" "<<"j="<<j<<endl;
		if(t[j]==s[i])
		{
//            cout<<"+"<<endl;
			i++;
			if(i==mid)
			{
				return true;
			}
		}
		else
		{
//		    cout<<"-"<<endl;
//			if( (mid - i) > (tlength-j-1) )
//			{
//				return false;
//			}
		}
	}
//	cout<<"I am here"<<endl;
	return false;
}
int main()
{
	s.clear();
	t.clear();
	cin>>s>>t;

	slength = s.length();
	tlength = t.length();
//	cout<<s<<""<<slength<<endl;
//	cout<<t<<""<<tlength<<endl;
	int l =1,r = slength;
	int answer = 0;
	while(l<=r)
	{
		int mid = (l+r)/2;
//		cout<<"mid="<<mid<<endl;
		if(isSubsequence(mid))
		{
//            cout<<"+"<<endl;
			answer = mid;
			l = mid + 1;
		}
		else
		{
//		    cout<<"-"<<endl;
			r = mid - 1;
		}
	}
	cout<<answer<<endl;
	return 0;
}
