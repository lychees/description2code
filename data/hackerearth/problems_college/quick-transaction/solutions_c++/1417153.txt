long long int countCoins(long long int* coin,long long int size,long long int value)
{

        for(i=size-1;i>=0;i--)
        {
                count=value/coin[i];
                if(count)
                {
                        value-=count*coin[i];
                        total+=count;
                        //sum+=count;
                        //printf("%d number of Rs %d coins\n",count,coin[i]);
                }
        }
        return total;
}
