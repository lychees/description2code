#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	unsigned long long int n;
    	cin >> n;
    	unsigned long long int second=0, third=0;
    	if(n == 4)  cout << "1\n";
    	else if (n==5) cout << "5\n";
    	else if (n==6) cout << "15\n";
    	else if (n < 4) cout << "0\n";
    	else {
    		second = n*(n-4);
    		third = 2*n*(n-5);
    		n = (n + second + third)/2;
    		cout << n << "\n";
    	}
    }
    return 0;
}
