#include <iostream>
#include <string>

using namespace std;

int c[27];
int main()
{
    int t;
    char st[101];
    scanf("%d",&t);
    while(t--) {
        for(int i=0;i<=26;i++) c[i]=0;
        scanf("%s",st);
        string s(st);

        bool flag = false;
        for(int i=0; s[i]!='\0'; i++) {
            int temp = (++c[s[i]-'a']);
            if(temp>1)
              flag=true;
        }
        if(flag) {
            printf("Yes\n");
        } else {
            printf("No\n");
        }
    }
    
    return 0;
}
