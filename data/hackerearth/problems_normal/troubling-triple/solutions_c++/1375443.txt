#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
#define INF 1000000007
#define min(a,b) (a<b?a:b)
#define max(a,b) (a>b?a:b)
int main()
{
    long long int n,a[2005]={0},i,k,j,ans=0;
    scanf("%lld%lld",&n,&k);
    for(i=0;i<n;i++)
    {
        scanf("%lld",&a[i]);
    }
    sort(a,a+n);
    for(i=0;i<n;i++)
    {
        for(j=i+1;j<n;j++)
        {
            long long int s=a[i]*a[j];
            long long int idx=upper_bound(a+j+1,a+n,k/s)-a;
            idx--;
            ans+=(idx-j);
            //printf("i=%lld j=%lld idx=%lld\n",i,j,idx);
        }
    }
    printf("%lld\n",ans);
    return 0;
}
