#include <stdio.h>
#include <algorithm>
using namespace std;
int main()
{
    int cases;
    scanf("%d",&cases);
    while(cases--)
    {
    	int n;
    	scanf("%d",&n);
    	int men[n],women[n];
    	for(int i=0;i<n;i++)
    	{
    		scanf("%d",&men[i]);
    	}
    	for(int i=0;i<n;i++)
    	{
    		scanf("%d",&women[i]);
    	}
    	sort(men,men+n);
    	sort(women,women+n);
    	int sum=0;
    	for(int i=0;i<n;i++)
    	{
    		sum+=men[i]*women[i];
    	}
    	printf("%d\n",sum);
    }
    return 0;
}
