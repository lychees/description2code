#include<stdio.h>
#include<stdlib.h>
int main()
{
	int n,c,count,i;
	int k,l,rank;
	scanf("%d %d",&n,&c);
	int *A = (int *) malloc(sizeof(int)*n);
	for(i=1;i<=n;i++)
		scanf("%d",&A[i]);
	int q;
	scanf("%d",&q);
	while(q--)
	{
		rank=1;
		count = 0;
		int m,d;
		scanf("%d %d",&m,&d);
		if(m==1)
		{
			for(i=1;i<d;i++)
			{
				if(A[i]>c)
					count = count+ c;
				else if(A[i]<c)
					count = count + A[i];
			}
			printf("%d\n",count);
		}
		else if(m==2)
		{
			k = A[d]/c;
			l = A[d]%c;
			if(l!=0)
				k = k+1;
			for(i=1;i<=d-1;i++)
			{
				if(A[i]<=c*k)
					rank++;
			}
			for(i=d;i<=n;i++)
			{
				if(A[i]<=c*(k-1))
					rank++;
			}
			printf("%d\n",rank);
		}
	}
}
