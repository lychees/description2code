#include <iostream>
#include<string.h>
using namespace std;
long long a[1010][1010],d[1010][1010]={0},sum=0;

int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
     for(int j=0;j<n;j++)
     { cin>>a[i][j];
      d[i][j]=1;
     }
      for(int k=100;k>=1;k--)
      for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
          {if(a[i][j]==k)
          	{if(i-1>=0&&i-1<n)
            if(a[i-1][j]<a[i][j])
             {d[i-1][j]+=d[i][j];
              d[i-1][j]= d[i-1][j]%1000000007;}
   
           if(i+1<n&&i+1>=0)
            if(a[i+1][j]<a[i][j])
             {d[i+1][j]+=d[i][j];
               d[i+1][j]=d[i+1][j]%1000000007;}
            
           if(j-1>=0&&j-1<n)
             if(a[i][j-1]<a[i][j])
               {d[i][j-1]+=d[i][j];
               d[i][j-1]=d[i][j-1]%1000000007;}
           if(j+1<n&&j+1>=1)
             if(a[i][j+1]<a[i][j])
             { d[i][j+1]+=d[i][j];
               d[i][j+1]=d[i][j+1]%1000000007;}
          	}
          }
          sum=0;
           for(int i=0;i<n;i++)
         for(int j=0;j<n;j++)
           sum=(sum+(d[i][j])%1000000007)%1000000007;
           cout<<sum;
           
    return 0;
}
