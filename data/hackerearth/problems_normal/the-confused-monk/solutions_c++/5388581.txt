#include<bits/stdc++.h>

using namespace std;
int arr[55];
#define M 1000000007

int gcd(int a, int b) {
   if(b == 0)
     return a;
   else
    return gcd(b, a % b);
 }

long long int power(int c, int d) {
    if(d == 0)
      return 1;
    long long int k = power(c, d/2);
    if(d % 2 == 1)
      return (((k % M) * (k % M)) % M * (c % M)) % M;
    else
      return ((k % M) * (k % M)) % M;
}


int main() {
  int n, z = 0, i;
  long long int mul = 1;

  scanf("%d",&n);

  for(i = 0; i < n; i++) {
     scanf("%d",&arr[i]);
     mul = (mul % M * arr[i] % M) % M;
     z = gcd(z , arr[i]);
  }
  long long int p = power(mul, z);
  
  printf("%lld\n",p);

return 0;
}
