#include <iostream>
using namespace std;

int main()
{
    int t,n,c;
    cin>>t;
    while(t--){
    	cin>>n;
    	c=0;
    	while(n>1){
    		if(n%3==0){
    			n/=3;
    			c++;
    		}else if(n%2==0){
    			n/=2;
    			c++;
    		}else{
    			n-=1;
    			c++;
    		}
    	}
    	cout<<c<<endl;
    }
    return 0;
}
