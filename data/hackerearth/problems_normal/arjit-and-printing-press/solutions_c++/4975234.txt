#include <bits/stdc++.h>

using namespace std;

int main() {
    int T;
    cin >> T;
    for (int t = 0; t < T; t++) {
    	string W, R;
    	cin >> W >> R;
    	sort(R.begin(), R.end());
    	int j = 0;
    	for (int i = 0; i < W.length(); i++) {
    		if (j >= R.length())
    			break;
    		if (R[j] < W[i]) {
    			W[i] = R[j];
    			j++;
    		}
    	}
    	cout << W << endl;
    }
    return 0;
}
