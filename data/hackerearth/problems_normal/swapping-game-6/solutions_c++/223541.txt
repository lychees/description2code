#include <iostream>
#include <string.h>
#include <string>
#include <sstream>
#include <stdio.h>
#include <math.h>
#define ll long long int
using namespace std;

char s[1005];
stringstream so,sc;
void copy(char t[],char r[])
{

	int i;
	int l = strlen(r)-1;
	for(i=0;t[i]!='\0';i++)
		s[i]=t[i];
	for(;l>=0;l--,i++)
		s[i]=r[l];

	s[i]='\0';

}
void change(char s[])
{


	char temp[550];
	char rev[550];
	int rc=0,tc=0,i;
	for(i=0;s[i]!='\0';i++)
		if(i%2)
			rev[rc++]=s[i];
		else
			temp[tc++]=s[i];
	rev[rc]=temp[tc]='\0';

	copy(temp,rev);

}

int main()
{

	ll k,i;
	cin>>k;

	scanf("%s",s);
	so<<s;
	int found=0;
	for(i=0;i<k;i++)
	{

		change(s);
		//cout<<"after change:"<<s<<endl;
		sc<<s;
		if(so.str()==sc.str() && found==0)
		{
			// repeat length will be i+1
			//cout<<"k:"<<k<<" i:"<<i<<endl;
			//cout<<"surpassing "<<floor((k/(i+1)))<<" times\n";
			
			i  = i+((i+1)*((k / (i+1))-1));
			found=1;
			
			//cout<<"For :"<<so.str()<<": repeat length is "<<i<<endl;

		}
		sc.str( std::string() );
		sc.clear();
	}
	printf("%s",s);



	return 0;
}
