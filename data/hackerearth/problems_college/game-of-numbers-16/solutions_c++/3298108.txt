#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long int i,j,c=0,x,y,n;
        cin>>n;
        for(i=2;i<=n;i++)
        {

            for(j=i+1;j<=n;j++)
            {
                x=i*j;
                y=i+j;
                if(x%y==0)
                   {
                       //cout<<i<<" "<<j<<endl;
                       c++;
                   }
            }
        }
        cout<<c<<endl;
    }
}
