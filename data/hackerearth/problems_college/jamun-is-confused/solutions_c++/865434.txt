#include <iostream>
#include<cmath>
using namespace std;

int64_t gcd(int64_t a,int64_t b)
{
  if(b==0)
    return a;
  if(a%b==0)
   return b;
  else
    return gcd(b,a%b);
}
int main()
{
    int tst;
    cin>>tst;
    while(tst--)
     {  int n,arr[1001],i;
     	cin>>n;
     	for(i=0;i<n;i++)
     	 cin>>arr[i];
     	 int re=arr[0];
     	for(i=1;i<n;i++)
     	  re=gcd(re,arr[i]);
     	  cout<<re<<"\n";
     }
     	   
}
