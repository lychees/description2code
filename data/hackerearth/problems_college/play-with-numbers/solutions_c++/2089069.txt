#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long n,m,z,ans=0,i;
        cin>>z>>n>>m;
        for(i=1;i*i<=z;i++)
        {
            if(z%i==0){if(i<=n && z/i<=m)ans++;if(i<=m && z/i<=n && z/i!=i)ans++;}
        }
        cout<<ans<<endl;
    }
    return 0;
}
