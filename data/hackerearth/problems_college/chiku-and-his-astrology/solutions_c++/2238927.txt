

#include <iostream>
#include <vector>


int main()
{
	int t;
	std::cin>>t;
	while(t--)
	{
		std::string s;
		std::cin>>s;
		int a[2]={0};
		for(int i=0 ; i<s.size() ; ++i){
			if( s[i] == '0')
				a[0]++;
			else
				a[1]++;
		}

		if(!a[0])
			std::cout<<a[1]<<"\n";
		else if(!a[1])
			std::cout<<a[0]<<"\n";
		else
		{
			if( a[0] < a[1] )
				std::cout<<a[1]-a[0]<<"\n";
			else
				std::cout<<a[0]-a[1]<<"\n";
		}
	}
}
