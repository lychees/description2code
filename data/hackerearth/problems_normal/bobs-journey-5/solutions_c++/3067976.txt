#include <iostream>
using namespace std;

int main()
{
    long long int n,t,i,z,flag,j,k;
    cin>>t;
    for(z=0;z<t;z++)
    {
    	cin>>n;
    	flag=0;
    	long long int b[n];
    	for(i=1;i<=n;i++)
    	{
    		char a[20];
    		cin>>a;
    		b[i-1]=a[0];
    	}
    	long long int ctr[26]={0};
    	for(j=0;j<n;j++)
    	{
    		ctr[b[j]-97]++;
    	}
    	for(k=0;k<26;k++)
    	{
    		if(ctr[k]>1)
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag)
    		cout<<"NO"<<endl;
    	else
    		cout<<"YES"<<endl;
    }
    return 0;
}
