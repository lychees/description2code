using namespace std;
ll a[1000007];
int main()
{
    ll n,i,j,flag,res;
    scanf("%lld",&n);
    for(i=0;i<n;i++)
    {
    	scanf("%lld",&a[i]);
    }
    sort(a,a+n);
    flag=0;
    res=999999999;
    for(i=n-1;i>=0;i--)
    {
    	for(j=0;j<=i;j++)
    	{
    		if(a[i]-3*a[j]<=0)
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag)
    	res=min(res,n-(i-j+1));
    }
    printf("%lld\n",res);
    
	return 0;
}
