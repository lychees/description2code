#include <bits/stdc++.h>
using namespace std;
typedef map<string,int> mp;
int main()
{
    int i,n;
    mp m;
    cin>>n;
    while(n--)
    {
        string s;
        cin>>s;
        string t=s;
        std::reverse(s.begin(),s.end());
        m[s]++;
        m[t]++;

    }

    mp::iterator y;
    for(y=m.begin();y!=m.end();y++)
    {
        if((*y).second==2)
        {
            string t1=(*y).first;
            int t2=t1.length();
            cout<<t2<<" "<<t1.at(t2/2);
            break;
        }
    }
    return 0;
}
