#define ll long long
 
template<class T> T nextInt() {
    T x = 0, p = 1;
    char c;
    do {
        c = getchar();
    } while (c <= 32);
    if (c == '-') {
        c = getchar();
        p = -1;
    }
    while (c >= '0' && c <= '9') {
        x = x * 10 + c - '0';
        c = getchar();
    }
    return x * p;
}
int main()
{
	int test;
	test = nextInt<int>();
	while(test--){
		ll n, mod;
		n = nextInt<ll>();
		mod = nextInt<ll>();
		ll k = (n+1)/mod;
		ll p = n%mod;
		if(p == mod-1) p = 0;
		ll ans = (mod*(mod-1))/2;
		ans = ans*k;
		//cout<<ans<<endl;
		ans += (p*(p+1))/2;
		printf("%lld\n",ans);
	}
}
