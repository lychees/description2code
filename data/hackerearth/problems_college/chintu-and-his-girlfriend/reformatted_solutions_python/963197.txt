str = raw_input()
a, b = str.split(' ')
a, b = int(a), int(b)
flag = 0

arr = []

for i in range(0, a):
    x = raw_input()
    arr.append(x)

for j in range(0, 26):
    cnt = 0
    for i in range(0, a):
        if arr[i].find(chr(97+j)) != -1:
            cnt += 1
    if cnt == a:
        flag += 1


if flag < b:
    print ":-("
else:
    print ":-)"
