from collections import defaultdict


temp = raw_input()
T = int(temp)
i = 0
while i < T:
    hm = defaultdict(lambda: 0)
    count = 0
    temp2 = raw_input()
    N = int(temp2)
    j = 0
    while j < N:
        s = raw_input()
        a = s.split()
        hm[a[0]] = hm[a[0]]+1
        if hm[a[1]] >= 1:
            hm[a[1]] = hm[a[1]]-1
        else:
            count = count+1
        j = j+1
    i = i+1
    print count
