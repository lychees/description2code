#include <iostream>
#include <vector>
using namespace std;

int main()
{
  string S;
  cin >> S;
  int str_size = S.length();
    vector<vector<bool>> word_j_to_i_palindrome(str_size, vector<bool>(str_size, false));
    int total_number = 0;
    
    for (int i = 0; i < str_size; ++i) {
        for (int j = 0; j < str_size - i; ++j) {
            if(i == 0) {
                word_j_to_i_palindrome[j][j + i] = true; // a char must a palindrome
            } else if (i == 1) {
                word_j_to_i_palindrome[j][j + i] = (S[j] == S[j + i]) ? true : false; //check if word from j to j+1 is a palindrome, ex: aa, bb
            } else {
                if ((S[j] == S[j + i]) && word_j_to_i_palindrome[j + 1][j + i - 1]) {
                    word_j_to_i_palindrome[j][j + i] = true;
                }
            }
            
            if (word_j_to_i_palindrome[j][j + i]) {
                total_number++;
            }
        }
    }
  cout << total_number <<endl;
}
