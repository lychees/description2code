#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
    	int n,load;
    	int weight[1001],value[1001];
    	cin>>n;
    	cin>>load;
    	for(int i=1;i<=n;i++)
    	{
    		cin>>weight[i];
    	}
    	for(int i=1;i<=n;i++)
    	{
    		cin>>value[i];
    	}
    	int dp[1001][1001]={0};
    	for(int i=1;i<=n;i++)
    	{
    		for(int j=0;j<=load;j++)
    		{
    			if(j-weight[i]>=0)
    			{
    				dp[i][j]=max(dp[i-1][j],dp[i-1][j-weight[i]]+value[i]);
    			}
    			else
    				dp[i][j]=dp[i-1][j];
    		}
    	}
    	cout<<dp[n][load]<<endl;
    }
    return 0;
}
