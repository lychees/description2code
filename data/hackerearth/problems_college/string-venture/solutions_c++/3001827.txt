#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int main()
{
    char a[100];
    gets(a);
    //cin>>a;
    //cout<<a<<endl;
    int n=0;
    for(int i=0;i<strlen(a);i++)
        n+=a[i];
    n=n/strlen(a);
    if(n%2!=0)
    {
        char max1=a[0];
        int t=0;
        for(int i=0;i<strlen(a);i++)
        {
            t=count(a,a+strlen(a),a[i]);
            if(t>count(a,a+strlen(a),max1))
                max1=a[i];
        }
        cout<<max1<<endl;
    }
    else
    {
        for(int i=strlen(a)-1;i>=0;i--)
            cout<<a[i];
    }
    return 0;
}
