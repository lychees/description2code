#include <bits/stdc++.h>
using namespace std;

const int LIM = 500000; // primes>=4e4

int N,K,M;
bool bad[21][LIM];

int main() {
	scanf("%d%d%d",&N,&K,&M);
	K = min(K,20);
	for (int i=0;i<M;i++) {
		int b;
		scanf("%d",&b);
		if (b<LIM) bad[K][b] = 1;
	}
	bad[0][1] = 1;
	int on = 1;
	for (int i=0;i<N;i++) {
		while (bad[K][on]) on+=1;
		printf("%d ",on);
		for (int j=1;j<=K;j++) {
			for (int k=0;k<LIM;k+=on) {
				bad[j][k]|=bad[j-1][k/on];
			}
		}
	}
	printf("\n");
}
