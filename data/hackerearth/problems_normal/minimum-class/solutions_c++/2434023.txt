#include<bits/stdc++.h>
#define lli long long int
using namespace std;
lli arr[1000007];
int binsearch(int x)
{
    int lbound=0,ubound=1000006,mid=(lbound+ubound)/2;
    while(lbound<=ubound)
    {
        mid=(lbound+ubound)/2;
        if(arr[mid]>=x)
        {
            if(arr[mid-1]>=x)
                ubound=mid-1;
            else
                return mid;
        }
        else if(arr[mid]<x)
            lbound=mid+1;
    }
    return -1;
}
int main()
{
    ios::sync_with_stdio(false);
    arr[0]=0;
    arr[1]=1;
    lli c=1;
    for(int i=2;i<1000007;i++)
    {
        if(i%2==0)
        {
            arr[i]+=arr[i-1]+c;
        }
        else
        {
            c++;
            arr[i]+=arr[i-1]+c;
        }
    }
    int t;
    cin>>t;
    while(t--)
    {
        lli n;
        cin>>n;
        lli ans=binsearch(n);
        cout<<ans<<endl;
    }
    return 0;
}
