#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;

int main() {

  int i,j,k,t,cou=0;
  string s;
  cin>>s;
  for(i=0;i<s.size()-1;i++) {
     for(j=i+1;j<s.size();j++) {
         if(s[i] == s[j]) {
            t=(i+j)/2;
            for(k=0;k+i<=t;k++) {
                if(s[k+i] != s[j-k]) break;
            }
            if(k>(t-i)) cou++;
         }
     }
  }
  cou += s.size();
  printf("%d",cou);
  return 0;
}
