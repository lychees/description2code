#include <iostream>
#include <vector>
using namespace std;

int main()
{
	long long int T,i,j,N,M,rem,div,total,num;
    vector<long long int>V;
    cin>>T;
    for(i=0;i<T;i++)
    {
        cin>>N;
        total=0;
        vector<long long int>T;
        for(j=0;j<N;j++)
        {
            cin>>num;
            T.push_back(num);
            total+=num;
        }
        cin>>M;
        div=M/total;
        if(M%total==0)
        rem=M-((div-1)*total);
        else
        rem=M-div*total;
        total=0;
        for(j=0;j<N;j++)
        {
            total+=T[j];
            if(total>=rem)
            {
                V.push_back(j+1);
                break;
            }
        }
    }
    for(i=0;i<T;i++)
        cout<<V[i]<<endl;
  return 0;
}
