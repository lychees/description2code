#include <iostream>
#include<stdio.h>
using namespace std;
#define ll long long
ll ex_gcd(ll a,ll b,ll &x,ll &y)
{
	if(b==0){x=1,y=0;return a;}
	ll x1,y1,d=ex_gcd(b,a%b,x1,y1);
	x=y1;
	y=x1-a/b*x;
	return d;
}
int main()
{
    int T,a,b,c;
    scanf("%d",&T);
    while(T--)
    {
    	scanf("%d%d%d",&c,&a,&b);
    	ll x,y,d;
    	d=ex_gcd(a,b,x,y);
    	if(c%d){puts("-1");continue;}
    	y*=(c/d);ll aa=a/d,bb=b/d;y=-y;
    	y%=aa;
    	if(y<0)y+=aa;
    	x=(b*y+c)/a;
    	cout<<x+y<<"\n";
    }
    return 0;
}
