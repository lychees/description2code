#include <iostream>
using namespace std;

bool check(string s1,string s2)
{
	long int i,size;
	s1+=s2;
	size = s1.size();
	for(i=0;i<size/2;i++)
	{
		if(s1[i]!=s1[size-i-1])
			return false;
	}
	return true;
}
int main()
{
    int tc;
    string str1,str2;
    cin >> tc;
    while(tc--)
    {
    	cin >> str1 >> str2;
    	if(check(str1,str2) || check(str2,str1))
    		cout << "YES"<<endl;
    	else
    		cout << "NO"<<endl;
    }
    return 0;
}
