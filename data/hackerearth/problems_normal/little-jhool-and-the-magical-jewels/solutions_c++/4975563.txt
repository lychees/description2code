#include<string>
#include <iostream>
#include<ctype.h>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);cin.tie(0);
    int tc,size,i;
    cin>>tc;
    string s;
    while(tc--)
    {
    	cin>>s;
    	size=s.size();
    	i=0;
    	int r,u,b,y;
    	r=0;
    	u=0;
    	b=0;
    	y=0;
    	while(i<size)
    	{
    		if(tolower(s[i])=='r')
    		r++;
    		else if(tolower(s[i])=='u')
    		u++;
    		
    		else if(tolower(s[i])=='b')
    		b++;
    		else if(tolower(s[i])=='y')
    		y++;
    		i++;
    	}
    	int min=r;
    	if(u<min)
    	min=u;
    	if(b<min)
    	min=b;
    	if(y<min)
    	min=y;
    	cout<<min<<"\n";
    	
    	
    	
    }
}
