#include<iostream>
using namespace std;

int main()
{
    long long int N,A[100000],Q,D,i,j,divi=1;
    cin>>N;
    for(i=0;i<N;i++)
    	{
    		cin>>A[i];
    	}
    cin>>Q;
    for(i=0;i<Q;i++)
    	{
    		cin>>D;
    		divi*=D;
    	}
    if(divi==0)
    	{
    		for(i=0;i<N;i++)
    			{
    				cout<<"0 ";
    			}
    	}
    else
    	{
    		for(i=0;i<N;i++)
    			{
    				A[i]/=divi;
    				cout<<A[i]<<" ";
    			}
    	}
}
