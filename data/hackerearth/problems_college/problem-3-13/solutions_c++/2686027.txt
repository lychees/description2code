#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
int a[1007][1007];
int dpl[1007][1007];
int dpu[1007][1007];
int main()
{
	int n, m;
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= m; j++)
		{
			scanf("%d", &a[i][j]);
		}
	}
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= m; j++)
		{
			dpu[i][j] = 1;
			if (a[i][j] == a[i - 1][j])
			{
				dpu[i][j] = dpu[i - 1][j] + 1;
			}
		}
	}
	for (int i = 1; i <= n; i++)
	{
		for (int j = m;j>=1;j--)
		{
			dpl[i][j] = 1;
			if (a[i][j] == a[i][j + 1])
			{
				dpl[i][j] = dpl[i][j + 1] + 1;
			}
		}
	}
	int ans = 0;
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= m; j++)
		{
			ans = max(ans, dpu[i][j] + dpl[i][j] - 1);
		}
	}
	printf("%d\n", ans);
}
