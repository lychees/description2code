#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    long int n;
    long long int sum;
    while(t--)
    {
        cin>>n;
        sum=0;
        for(int i=n-1;i>0;i--)
        {
            sum+=i;
        }
        cout<<sum<<"\n";
    }
    return 0;
}
