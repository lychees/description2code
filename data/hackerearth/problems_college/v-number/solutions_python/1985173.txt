for _ in xrange(input()):
	arr=list(raw_input())
	arr=map(int,arr)
	
	s=sum(arr)
	if s==0:
		print "YES"
		continue
	
	if len(arr)%2!=0:
		print "NO"
		continue
	
	cur=arr[0]
	flag=0
	for i in xrange(len(arr)-1):
		s-=arr[i]
		if cur==s:
			flag=1
			break
		else:
			cur+=arr[i+1]
			
	if flag:
		print "YES"
	else:
		print "NO"
