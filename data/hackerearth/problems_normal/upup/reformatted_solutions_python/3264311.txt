import string
s = raw_input()
s1 = []
convert = -1
for i in range(0, len(s)):
    if i == 0 and s[i].islower():
        s1.append(s[i].upper())
    elif s[i] != ' ' and convert == 1:
        s1.append(s[i].upper())
        convert = -1
    elif s[i] == ' ':
        convert = 1
        s1.append(s[i])
    else:
        s1.append(s[i])
print ''.join(s1)
