#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int T, N, i, u, j;
    cin>>T;
    while(T--)
    {
    	cin>>N;
    	if(N%2)
    	{
    		cout<<0<<endl;
    		continue;
    	}
    	u=sqrt(N);
    	int count=1;
    	for(i=2; i<=u; i++)
    	{
    		if(i%2==0 && N%i==0)
    			count++;
    		if(N%i==0 && (N/i)%2==0)
    			count++;
    	}
    	if(u*u==N)
    		count--;
    	cout<<count<<endl;
    }
    return 0;
}
