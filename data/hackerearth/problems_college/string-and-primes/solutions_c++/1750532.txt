#include<iostream>
#include<cstring>
#include<cstdio>
#include<vector>
#include<set>
using namespace std;
vector<int> v(123);
set<int> res;
int main()
{
    int t,i,len,j,num;
    char str[100001];
    for(i=2;i<=122;i++)
    {
        if(v[i]==0)
        {
            if((i>=65)&&(i<=122))
            {
                res.insert(i);
            }
            for(j=2*i;j<=122;j=j+i)
            {
                v[j]=1;
            }
        }
    }
    scanf("%d",&t);
    while(t--)
    {
        vector<char> ans;
        scanf("%s",str);
        len=strlen(str);
        for(i=0;i<len;i++)
        {
           num=str[i];
           if(res.find(num)==res.end())
           {
                ans.push_back(str[i]);
            }
        }
        if(ans.size()!=0)
        {
            for(i=0;i<ans.size();i++)
            {
                printf("%c",ans[i]);
            }
        }
        else
            printf("NULL");
        printf("\n");
    }
    return 0;
}
