#include<bits/stdc++.h>
typedef long long int ll;
using namespace std;
int main()
{
    ll t,a,b,ans;
    cin>>t;
    while(t--)
    {
        cin>>a>>b;
        if(b-a==1)
            ans=a&b;
        else
        {
            ll temp=(ll)log2(b),temp2,temp3;
            if(floor(pow(2,temp))==b)
                {
                    temp2=(pow(2,temp)-1);
                    temp3=(pow(2,temp)-2);
                    ans=temp2&temp3;
                }
            else if(b%2==0)
                ans=b-2;
            else
                ans=b-1;
        }
        cout<<ans<<endl;
    }
}
