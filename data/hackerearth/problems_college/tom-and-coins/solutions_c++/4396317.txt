#include <iostream>
using namespace std;

#define ll long long 
#define rep(i,n) for(int i=0;i<n;i++)
#define repT(T) while(T--)

int main(){
    //cout << "Hello World!" << endl;
    ll T;
    cin>>T;
    repT(T){
    	ll a,k,n;
    	int i;
    	cin>>a>>k>>n;
    	i=0;
    	n-=1;
    	rep(i,n){
    		a+=k;
    	}
    	cout<<a<<endl;
    }
    return 0;
}
