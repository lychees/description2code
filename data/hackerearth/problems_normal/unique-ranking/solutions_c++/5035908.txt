#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

long marks[1000][20];

int main()
{
    long N, M, Q, q, num;
    vector <vector <long> > v;

    cin >> N >> M;
    
    for (long n = 0; n < N; n++)
    {
        for (long m = 0; m < M; m++)
        {
            cin >> marks[n][m];
        }
    }
    cin >> Q;
    while(Q--)
    {
        vector <long> query, tmp;
        cin >> q;
         for (long i = 0; i < q; i++)
        {
            cin >> num;  
            query.push_back(num);      
        }
        for (long n = 1; n <= N; n++)
        {
            tmp.clear();
            for (long m = 0; m < q; m++)
            {
                tmp.push_back(marks[n - 1][query[m] - 1]);
            }
            tmp.push_back(n);
            v.push_back(tmp);
        }
        sort(v.begin(), v.end());
        cout << v[0][q] << endl;

        v.clear();
    }
    return 0;
}
