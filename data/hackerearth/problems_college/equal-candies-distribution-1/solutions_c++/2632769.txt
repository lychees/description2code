int main() {
    int t,i,j,choc[100010],minchoc[100010];
    int op[]={1,2,5};
    minchoc[0]=0;
    for(i=1;i<100010;i++)
        minchoc[i]=9999999;
    for(i=1;i<100010;i++)
        for(j=0;j<3;j++)
            if(i-op[j]>=0 && minchoc[i-op[j]]+1<minchoc[i])
                minchoc[i]=minchoc[i-op[j]]+1;
    cin>>t;
    while(t--)
    {
        int cnt=9999999,n,x,y,temp,ans=9999999;
        map<int,int> m;
        map<int,int>::iterator it;
        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>choc[i];
            cnt=min(cnt,choc[i]);
            if(!m.count(choc[i]))
                m[choc[i]]=1;
            else
                m[choc[i]]++;
        }
        for(i=0;i<=cnt;i++)
        {
            temp=0;
            for(it=m.begin();it!=m.end();it++)
            {
                x=it->first;
                y=it->second;
                x-=i;
                temp+=(minchoc[x]*y);
            }
            ans=min(ans,temp);
        }
      //  if(choc[0]==0 && choc[n-1]==0)
       //     ans=n;
        cout<<ans<<endl;
    }
    return 0;
