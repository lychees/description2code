{

    int T;

   // binary res;

    while(T){

    	cin>>B;
    	
    //	cout<<"log2(B) , "<<log2(B);
        res = log2(B);
       // cout<<"B,"<<B<<"\n";
       if((B-A)==1){
       	cout<<(B&A)<<"\n";
       	T--;
       	continue;
       }
       else if(pow(2,res)==B)
               if(A==(B-1)){
              	cout<<"0\n";
             	T--;
            	continue;
               }
               else
               Q = B-1;
            else if(!(B&1))
                    Q = B-1;
                 else

    	  P = Q-1;
     // cout<<"Q,"<<Q<<"\n";

    	if(P>=A)
    	cout<<(Q&P)<<"\n";

    }

}
