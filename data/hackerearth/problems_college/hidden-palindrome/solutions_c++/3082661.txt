#include <iostream>
#include <algorithm>

using namespace std;

bool is_palin(string str) {
	for(int i = 0, j = str.length()-1;i < j;i++, j--) {
		if(str[i] != str[j]) return false;
	}	
	return true;
}
int main()
{
	int t;
	cin >> t;
	while(t--) {
		string str;
		cin >> str;
		int ans = 1;
		for(int i = 0;i < str.length();i++) {
			for(int j = ans;i + j-1 < str.length();j++) {
				if(is_palin(str.substr(i, j))) {
					ans = max(ans, j);
				}
			}
		}
		cout << ans << endl;
	}
    return 0;
}
