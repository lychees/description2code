#include<bits/stdc++.h>
#define scn(n) scanf("%d",&n)
#define scnll(n) scanf("%lld",&n)
#define scnch(n) scanf("%c",&n)
#define scns(n) scanf("%s",&n)
#define loop(i,n,m) for(i=n;i<=m;i++)
#define revloop(i,n,m) for(i=n;i>=m;i--)
#define print(n) printf("%d",n)
#define printc(n) printf("%c",n)
#define printll(n) printf("%lld",n)
#define prints(n) printf("%s",n)
#define next printf("\n")
#define ll long long
#define MAX(x,y) ((x)>(y)?(x):(y))
#define mod 1000000007
int data[1001][1001],dp[1001][1001];

using namespace std;
int main()
{
   	int t;int n,j,k,i;
    scn(t);
   	data[0][0]=1;
   	loop(i,1,1000)
   	{
   	    data[i][0]=1;
   	}
   	loop(i,1,1000)
   	{
   	    loop(j,1,i)
   	    {
   	        if(i==j)
            {
                data[i][j]=1;
            }
            else
            {
                data[i][j]=(1ll*data[i-1][j]+1ll*data[i-1][j-1])%mod;
            }

   	    }
   	}

   	loop(i,1,1000)
   	{
   	    dp[i][0]=1;
   	    loop(j,1,i)
   	    {
   	        dp[i][j]=(1ll*data[i][j]+1ll*dp[i][j-1])%mod;
   	    }
   	}
   	while(t--)
   	{
   	    scn(n);scn(k);
   	    print(dp[n][k]);
   	    next;

    }
}
