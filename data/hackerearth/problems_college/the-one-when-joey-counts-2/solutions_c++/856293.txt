#include <iostream>
#define ll long long
using namespace std;
ll dp[33][3]={0};
void print(ll k,int len){
	if(len==0)cout<<endl;
	for(int i=0;i<3;i++){
		if(dp[len][i]<k)k-=dp[len][i];
		else if(k>=0)  {
			cout<<(i+2);
			print(k,len-1);
			break;
		}
	}
}
int main() {
	// your code goes here
	dp[1][0]=dp[1][1]=dp[1][2]=1;
	for(int i=2;i<33;i++)
	{
		for(int j=0;j<3;j++)
		dp[i][j]+=dp[i-1][0]+dp[i-1][1]+dp[i-1][2];
	}
	int t;
	cin>>t;
	while(t--){
		ll k;
		cin>>k;
		int flag=0;
		for(int i=1;i<33;i++)
		{
			for(int j=0;j<3;j++)
			{
				if(dp[i][j]<k)k-=dp[i][j];
				else if(k>=0) {
					cout<<(j+2);
					print(k,i-1);
					flag=1;
					break;
				}
			}
			if(flag)break;
		}
	}
	return 0;
}
