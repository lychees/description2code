#include <iostream>
#include<algorithm>
using namespace std;
int main()
{
    long int a[10001],t,n,c,d;
    long long int x,y,s;
    cin>>t;
    while(t--)
    {   s=0;c=0;d=0;
    	cin>>x>>y>>n;
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	sort(a,a+n);
    	for(int i=0;i<n;i++)
    	{
    		s+=a[i];
    	if(s>x)
    	break;
    	else
    	c++;}
    	s=0;
    	for(int i=0;i<n;i++)
    	{
    		s+=a[i];
    	if(s>y)
    	 	break;
    	 	else
    	 	d++;
    	}
    	if(c<d)
    	cout<<"Jhool Won\n";
    	if(c>d)
    	cout<<"Chef Won\n";
    	else if(c==d)
    	cout<<"Tie\n";
    }
    return 0;
}
