#include<bits/stdc++.h>
using namespace std;

int fib( int n)
{
    int dp[n],i;
    for( i=0;i<=n;i++)
    {
        dp[i]=0;
    }
    dp[0] =0;
    dp[1] =1;

    for( i=2;i<=n;i++)
    {
        dp[i] = dp[i-1]+dp[i-2];
    }
    return dp[n];
}

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        int n;
        cin >> n;
        if( n ==0)
        {
        	
        	cout << "0\n";
        }
        else
        	cout << fib(n+2)*(fib(n+2)) << endl;
    }
}
