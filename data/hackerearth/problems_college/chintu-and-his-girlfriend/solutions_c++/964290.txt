#include <iostream>
using namespace std;

int main()
{
    int n,k,l,a[123]={0},i,t,c=0;
    string s;
    cin>>n>>k;
    t=n;
    while(n--)
    {
    	cin>>s;
    	l=s.length();
    	for(i=0;i<l;i++)
    	{    		
    	if(a[(int)s[i]]==c)
    	{
    	a[(int)s[i]]++;
    	}
    	}
    	
    c++;	
    }
   c=0;
    for(i=97;i<123;i++)
     if(a[i]==t)
    {
     c++;
    }
  
    if(c>=k)
    cout<<":-)";
    else
    cout<<":-(";
    
    return 0;
}
