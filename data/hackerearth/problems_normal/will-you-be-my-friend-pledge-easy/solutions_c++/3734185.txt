#include <iostream>
using namespace std;

int hcf(int a, int b)	//a bada
{
	if(a%b==0)
		return b;
	return hcf(b, a%b);
}

int max(int a, int b)
{
	if(a>b)
		return a;
	return b;
}

int min(int a, int b)
{
	if(a<b)
		return a;
	return b;
}

int main()
{
    int a, n;
    cin>>a>>n;
    int count=0;
    while(n--)
    {
    	int k;
    	cin>>k;
    	if(hcf(max(a,k),min(a,k))>1)
    		count++;
    }
    cout<<count;
    return 0;
}
