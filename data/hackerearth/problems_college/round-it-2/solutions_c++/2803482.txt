#include<iostream>
#include<stdlib.h>
using namespace std;
typedef unsigned  long long int ulli;
int main()
{
	int lower,upper,divisor,remaind,Test_cases,N,B;
	cin >> Test_cases;
	while(Test_cases--)
	{
		cin >> N >> B;
		divisor=int (N/B);
		lower= divisor * B;
		upper = lower + B;
		remaind = N % B;
		//cout << divisor << lower << upper << remaind << endl;
		//cout << B/2 << int(B/2);
		//if(remaind==0)
		//	cout << N;
		if(B%2 ==0)
		{
		if(remaind >=  B/2)
			cout << upper << endl;
		else
			cout << lower << endl;
		}
		else
		{
		if(remaind >=  B/2 + 0.5)
			cout << upper << endl;
		else
			cout << lower << endl;
		}
		}
		return 0;
}
