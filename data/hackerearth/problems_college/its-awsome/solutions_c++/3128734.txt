#include<bits/stdc++.h>
using namespace std;
main()
{
	int t,j;
	cin>>t;
	for(j=0;j<t;j++)
	{
		string num;
		int s=0,i;
		cin>>num;
		for(i=0;i<num.length();i++)
		{
			if(num[i]=='4'||num[i]=='7')
			{
				s++;
			}
			
		}
		if(s==4||s==7)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
	}
	return 0;
}
