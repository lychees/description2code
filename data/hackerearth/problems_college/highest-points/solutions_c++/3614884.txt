#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define loop(i,a,b) for(i=a;i<b;i++)

int main(){
    ll n,w,t,l,i;
    cin>>n;
    ll a[n];
    loop(i,0,n){
        cin>>w>>t>>l;
        a[i]=3*w+1*t-1*l;
    }
    sort(a,a+n);
    cout<<a[n-1];
   
    return 0;
}
