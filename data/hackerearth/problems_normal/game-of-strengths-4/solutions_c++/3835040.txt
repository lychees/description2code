#include<bits/stdc++.h>

#define MOD 1000000007

using namespace std;

long long arr[100000];

int main()
{
    int N;
    int test,i;
    cin>>test;
    while(test--)
    {
        cin>>N;
        long long sum=0;
        for(int i=0;i<N;i++)
            cin>>arr[i];
        sort(arr,arr+N);
        for(i=0;i<N-1;i++)
            sum = (sum + (N-1-i)*arr[N-1-i])%MOD;
        long long temp = MOD*(long long)100000;
        for(i=0;i<N-1;i++)
            sum = (temp + sum - (N-1-i)*arr[i])%MOD;
        sum = (sum*arr[N-1])%MOD;
        cout<<sum<<endl;
    }
    return 0;
}
