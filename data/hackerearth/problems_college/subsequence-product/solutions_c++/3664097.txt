#include<bits/stdc++.h>
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		long long int n,h;
		scanf("%lld%lld",&n,&h);
		long long int m,pro=1;
		for( long long int i=0;i<n;i++){
			scanf("%lld",&m);
			pro=(pro*m)%h;
		}
		if(pro==0)
			printf("YES\n");
		else
			printf("NO\n");
	}
}
