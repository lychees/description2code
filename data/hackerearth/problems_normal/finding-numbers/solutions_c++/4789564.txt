#include<cmath>
#include <iostream>
using namespace std;
unsigned long long findNoOfSeq(unsigned long long z,unsigned long long m,unsigned long long n)
{
	unsigned long long cnt=0,n1,n2;
	long long sqt=sqrt(z);
	for( n1=1;n1<=sqt;n1++)
	{
		n2=z/n1;
		if(n2*n1==z)
		{
			if(n2<=m)
			{
				cnt++;
			}
			if(n2<=n&&n2!=n1)
			{
				cnt++;
			}
		}
	}
	return cnt;
}
int main()
{
	long t;
	cin>>t;
	unsigned long long z,m,n;
	while(t--)
	{
		cin>>z>>m>>n;
		unsigned long long ans=findNoOfSeq(z,m,n);
		cout<<ans<<"\n";
	}
	return 0;
}
