#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define d1(a)cout<<#a<<": "<<a<<"\n";
#define d2(a,b)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<"\n";
#define d3(a,b,c)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<" , "<<#c<<": "<<c<<"\n";
#define d4(a,b,c,d)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<" , "<<#c<<": "<<c<<" , "<<#d<<": "<<d<<"\n";

ll mod = 1000000007;
ll dp[1005][1005];

void precomp(){
	for(int i=1;i<1005;i++){
		for(int j=0;j<1005;j++){
			if(j == 0 || j == i){
				dp[i][j] = 1;
			}
			else{
				dp[i][j] = (dp[i-1][j-1] + dp[i-1][j])%mod;
				if(dp[i][j]<0){
					dp[i][j] += mod;
				}
			}
		}
	}
}

int main() {
	int t;
	cin>>t;
	precomp();
	while(t--){
		ll n,p;
		cin>>n>>p;
		ll ans = 0;
		for(int i=0;i<=p;i++){
			ans = (ans + dp[n][i])%mod;
			if(ans<0){
				ans += mod;
			}
		}
		cout<<ans<<"\n";
	}
	return 0;
}
