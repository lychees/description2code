#include<stdio.h>
#include<math.h>
int main(){
	
	int test;
	scanf("%d",&test);
	
	while(test--){
		int size;
		scanf("%d",&size);
		double d[size],f[size];
		
		for(int i=0;i<size;i++) scanf("%lf",&d[i]);
		
		for(int i=0;i<size;i++) scanf("%lf",&f[i]);
		
		double total=0;
		for(int i=0;i<size;i++){
			total += (d[i]*f[i]);
		}
		double max = 0;
		int index = 0;
		for(int i=0;i<size;i++){
			double result = d[i]*f[i]/total;
			if(result>max){
				max = result;
				index = i;
			}
		}
		printf("%d %.3lf\n",index,max);
	}
	return 0;
}
