#include<bits/stdc++.h>
using namespace std;
main()
{
int n;
cin>>n;
int student[10002];
for(int i=0;i<n;i++)
  student[i]=true;
char ar[10002];
cin>>ar;
int count=n; int position=0;
int i=0;
while(1)
{
  if(ar[i]=='b')
  {  student[position]=false;
     count--;
  }
  if(position==n-1)
    position=0;
  else
    position++;
  
  if(i==strlen(ar)-1)
    i=0;
  else i++;
  while(student[position]!=true)
     {
       if(position==n-1)
       position=0;
       else
       position++;
     }
     //cout<<position+1<<endl;
     if(count==1)
     {cout<<position+1<<endl;
      break;}

}

}
