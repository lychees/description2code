#include <iostream>
using namespace std;
#include<vector>
#include<algorithm>
#define MOD 1000000007
int main() 
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		vector<int> ma(n);
		for(int i=0;i<n;i++)cin>>ma[i];
		std::sort(ma.begin(),ma.end());
		long long ans=ma[0];
		for(int i=1;i<n;i++)
		{
			ans =(ans*(ma[i]-i)%MOD)%MOD;
		}
		cout<<ans<<endl;
	}
	// your code goes here
	return 0;
}
