val = [int(x) for x in raw_input().split()]
sr=[]
sc=[]
matrix=[]
for i in range(val[0]):
	inp = [int(x) for x in raw_input().split()]
	matrix.append(inp)
		
for x in matrix:
	s = sum(x)
	sr.append(s)
for i in range(val[1]):
	j=0
	s=0
	while(j<val[0]):
		s = s + matrix[j][i]
		j=j+1
	sc.append(s)

ma = sr[0]^sc[0]
for x in sr:
	for y in sc:
		if x^y>ma:
			ma = x^y
print ma
