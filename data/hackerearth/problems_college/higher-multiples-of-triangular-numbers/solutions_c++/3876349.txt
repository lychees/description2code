#include <bits/stdc++.h>
typedef unsigned long long ll;
using namespace std;

int main()
{
ll  i ,t;
ll  sum,n,d;
cin>>t;
while(t--){
	cin>>n;
	sum=0;
	i=1;
while(1) {
	d =0;
	sum += i;
		for(ll x = 1; x < sqrt(sum); x++)
			if(sum % x == 0)
				d+=2;

		if(d >n) {
			cout << sum << endl;
			break;
			}

		++i;

		}

}

		return 0;
}
