#include <bits/stdc++.h>
using namespace std;
unsigned long long cost[26];
int main()
{
    int t,r;
    long long n,x,d;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(int i=0;i<26;i++)
    	cin>>cost[i];
    	n*=12;
    	d=n/26;
    	r=n%26;
    	unsigned long long sum=0;
    	x=(d*(d-1))/2;
    	for(int i=0;i<26;i++)
    	sum+=d*cost[i]+x;
    	for(int i=0;i<r;i++)
    	sum+=cost[i]+d;
    	cout<<sum<<"\n";
    }
    return 0;
}
