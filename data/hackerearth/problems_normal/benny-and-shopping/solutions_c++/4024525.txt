#include<bits/stdc++.h>
using namespace std;

char str[1009];
vector<int>v;

int main(){
	int n,i;
	cin>>n;
	char q=getchar();
	while(n--){
		cin.getline(str,sizeof(str));
		int len=strlen(str);
		for(i=0;i<len;++i){
			if(str[i]=='$'){
				int j=i+1;
				while(j<len){
					if(str[j]>=97&&str[j]<=122) break;
					if(str[j]>=65&&str[j]<=90) break;
					int temp=str[j]-48;
					if(v.empty()){
						if(temp>=1&&temp<=9){
							v.push_back(temp);
						}
					}
					else{
						if(temp>=0&&temp<=9){
							v.push_back(temp);
						}
					}
					++j;
				}
				break;
			}
		}
		if(v.empty()) v.push_back(0);
		printf("$");
		for(i=0;i<v.size();++i) cout<<v[i];
		cout<<endl;
		v.clear();
	}
	return 0;
}
