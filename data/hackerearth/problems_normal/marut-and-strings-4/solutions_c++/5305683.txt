#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    if(t<1 || t>10)
    cout<<"Invalid Test"<<endl;
    else
    {
    	while(t--)
    	{
    		string s;
    		cin>>s;
    		int i,j,a=0,b=0,c=0;
    		j = s.length();
    		for(i=0;i<j;i++)
    		{
    			if((int)s[i]>=97 && (int)s[i]<=122)
    			a++;
    			else if((int)s[i]>=65 && (int)s[i]<=90)
    			b++;
    			else c++;
    		}
    		if(j>100)
    		cout<<"Invalid Input"<<endl;
    		
    		else if(a>b)
    		cout<<b<<endl;
    		
    		else if(b>a)
    		cout<<a<<endl;
    		
    		else if(c==j)
    		cout<<"Invalid Input"<<endl;
    		
    		else cout<<a<<endl;
    	}
    }
    return 0;
}
