#include <iostream>
#include <stdlib.h>
using namespace std;
int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}
int main() {
	int t;
	long int n,i,count;
	int ar[100000],temp;
	cin>>t;
	while(t--)
	{
		count=0;
		for(i=0;i<100000;i++)
		ar[i]=0;
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>temp;
			ar[temp]=1;
		}
		for(i=0;i<100000;i++)
		{
			if(ar[i]==1)
			count++;
		}
		cout<<count<<endl;
	}
	return 0;
}
