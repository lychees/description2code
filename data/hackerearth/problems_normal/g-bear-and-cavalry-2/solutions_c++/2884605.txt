#include<bits/stdc++.h>
using namespace std;
int main(){

   int t;
   cin >> t;
   while(t--){

      int n;
      cin >> n;
      int a[n+1];
      int b[n+1];

      for(int i = 1 ; i <= n ; ++i )
       cin >> a[i];
      for(int i = 1 ; i <= n ; ++i )
       cin >> b[i];
       int mx = 0;

       for(int i = 1 ; i <= n ; ++i )
             mx = max( mx , a[1]*b[i]);

       int id = 0;
       for(int i = 1 ; i <= n ; ++i  ){

          if(mx == (a[1]*b[i]))
           {
           id = i;
           break;
           }

       }
       vector<int> aa;
       vector<int> bb;
       for(int i = 2 ; i <= n ; ++i )
       aa.push_back(a[i]);
       for(int i = 1 ; i <= n;  ++i ){
          if(id != i )
          bb.push_back(b[i]);
       }
       sort(aa.begin(),aa.end());
       sort(bb.begin(),bb.end(),greater<int>());
             int flag = 1;
             for(int i = 0 ; i < aa.size() ; ++i ){

               if(aa[i]*bb[i] >= mx )
               flag = 0;

             }
             if(flag)
             cout<<"YES"<<endl;
             else
             cout<<"NO"<<endl;



   }

}
