t = int(input())
while t > 0:
	count = 1
	t = t - 1
	n = int(input())
	speed_str = raw_input().strip()
	speed = speed_str.split(" ")
	speed = map(int, speed)
	mini = speed[0]
	for i in range(1, n):
		if speed[i] <= mini:
			count = count + 1
			mini = speed[i]
	print(count)
