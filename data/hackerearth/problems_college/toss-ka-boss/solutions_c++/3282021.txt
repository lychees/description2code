#include <iostream>
using namespace std;
# define ll long long
ll ar[1111111];
int main()
{
	int t,i;
	string n;
	char ch;
	cin>>t;
	while(t--)
	{
		cin>>n;
		ar[0]=0;
		for(i=0;i<n.size();i++)
		{
			ch=n[i];
			if(ch=='H')
			ar[i+1]=ar[i]+1;
			else
			ar[i+1]=0;
		}
		ll ans=0;
		for(i=1;i<=n.size();i++)
		ans+=ar[i];
		cout<<ans<<'\n';
	}
    return 0;
}
