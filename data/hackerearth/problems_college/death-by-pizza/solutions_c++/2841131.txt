    int n, q;

    long long int a[n];
    for(int i = 0; i < n; i++)
        scanf("%lld", &a[i]);
    while(q--) {
        long long int type, from, to;
        scanf("%lld %lld %lld", &type, &from, &to);
        from--;
        to--;
        if(type) {
            long long int s = 0;
            for(int i = from; i <= to; i++) s += a[i];
            printf("%lld/%lld\n", s, to - from + 1);
        }
        else {
            for(int i = from; i <= to; i += 5) {
                a[i] += 16;
                if(i + 1 <= to) a[i + 1] += 9;
                if(i + 2 <= to) a[i + 2] += 26;
                if(i + 3 <= to) a[i + 3] += 26;
                if(i + 4 <= to) a[i + 4] += 1;
            }
        }
    }
