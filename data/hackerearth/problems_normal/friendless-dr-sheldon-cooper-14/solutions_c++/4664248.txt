#include <iostream>
#include<queue>
#include<cstdio>
#include<vector>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int a,b,i,x,y;
        cin>>a>>b;
        int visited[a+1];
        fill(visited,visited+a+1,0);
        vector < int > adj[a+1];
        for(i=1;i<=b;i++)
        {
            cin>>x>>y;
            adj[x].push_back(y);
            adj[y].push_back(x);
        }
        queue < int > q;
        q.push(1);
        visited[1]=1;
        int c=0;
        while(!q.empty())
        {
            int e1=q.front();
            q.pop();
            visited[e1]=1;
            for(i=0;i<adj[e1].size();i++)
            {
                if(visited[adj[e1][i]]==0)
                {
                    q.push(adj[e1][i]);
                    c++;
                    visited[adj[e1][i]]=1;
                }
            }
        }
        cout<<c<<"\n";
    }
    return 0;
}
