#include<bits/stdc++.h>
using namespace std;
long long cal_totient(long long n){
	float result=n;
	for(long long p=2;p*p<=n;++p){
		if(n%p == 0){
			while(n%p == 0){
				n/=p;
			}
				result*=(1.0-(1.0/(float)p));
			
		}
	}
	if(n>1) result *=(1.0-(1.0/(float)n));
	
	return (long long)result ;
}
int main(){
	long long t,n;
	cin>>t;
	while(t--){
		cin>>n;
		cout<<cal_totient(n)<<endl;
	}
}
