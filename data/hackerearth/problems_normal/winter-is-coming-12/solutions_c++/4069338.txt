#include <bits/stdc++.h>
using namespace std;
int arr[100001];
int dp[100001];
int main()
{
    int n,t,l,r,temp;
    ios_base::sync_with_stdio(0);
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(int i=1;i<=n;i++)
    	cin>>arr[i];
    	memset(dp,0,sizeof(dp));
    	long long sum=0;
    	int ans,temp;
    	ans=1e9;
    	for(int i=1;i<=n;i++)
    	{
    		sum+=arr[i];
    		sum%=n;
    		if(dp[sum]==0)
    		{
    			if(sum==0)
    			temp=i;
    			else
    			temp=1e9;
    		}
    		else
    		temp=i-dp[sum];
    		if(temp<ans)
    		{
    			ans=temp;
    			l=dp[sum]+1;
    			r=i;
    		}
    		dp[sum]=i;
    	}
    	cout<<l<<" "<<r<<"\n";
    }
    return 0;
}
