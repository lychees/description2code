ll a[MAXN];
int main()
{
	ios_base::sync_with_stdio(false);

	ll t, n, x, ans, k;
	
    cin >> t;

	while (t--)
	{
		cin >> n >> x;

		for (int i = 0; i < n; i++)
			cin >> a[i];

		ans = 0;
		for (int i = 1; i < n; i++)
		{
			if (a[i - 1] >= a[i])
			{
				k = (a[i - 1] - a[i]) / x + 1;
				ans += k;
				a[i] += k * x;
 			}
		}

		cout << ans << endl;
	}
  	return 0;
}
