T = int(raw_input())
Result = [0] * T

for i in range(T):
    N = int(raw_input())
    res = [] * N
    in_str = raw_input()
    res = in_str.split()
    temp = 0.0
    for j in range(N):
        temp += int(res[j])
    Result[i] = temp/N

for k in range(T):
    print("%.8f" % Result[k])
