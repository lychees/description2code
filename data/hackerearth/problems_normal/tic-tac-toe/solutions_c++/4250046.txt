#define md 1000000007
#define ll unsigned long long

ll fn1(ll n) {
	ll v1 = (n * (n - 1));
	ll v2 = ((2 * n) - 1);
	if (v1 % 3 == 0) v1 = v1 / 3;
	else v2 = v2 / 3;
	if (v1 % 2 == 0) v1 = v1 / 2;
	else v2 = v2 / 2;
	return ((v1%md) * (v2%md)) % md;
}
