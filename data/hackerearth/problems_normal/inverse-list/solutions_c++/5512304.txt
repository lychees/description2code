#include<iostream>
using namespace std;
int main(){
	int t,n,i,count,a;
	cin>>t;
	while(t--){
		cin>>n;
		int a[n];
		for(i=1,count=0;i<=n;i++)
			cin>>a[i];
		for(i=1;i<=n;i++){
			if(i!=a[a[i]]){
				count=10;
				break;
			}
		}
		if(count==10)
		cout<<"not inverse";
		else
		cout<<"inverse";
		cout<<endl;
	}
	return 0;
}
