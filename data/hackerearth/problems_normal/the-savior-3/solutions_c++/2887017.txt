#include <iostream>
using namespace std;

int main()
{
	int T, i=0; cin >> T;
	for (i=0; i<T; i++) {
		int N, j=0, k=0; cin >> N;
		int nos[N];
		int even_cnt = 0;
		for (j=0; j<N ; j++) {
			cin >> nos[j];
		}
		for (j=0; j<N; j++) {
			int even_1 = nos[j]%2;
			for (k = j+1; k<N; k++) {
				int even_2 =  nos[k]%2;
				if (nos[j] != nos[k] && even_1 == even_2)
					even_cnt++;
			}
		}
		cout <<even_cnt<<endl;
	}
    return 0;
}
