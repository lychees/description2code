#include<iostream>
using namespace std;
struct node
  {int data;
   node* next;
  };
node* create(int n)
{
	node *front=NULL, *rear=NULL;
	for(int i=0;i<n;i++)
	{
		node* temp=new node;
		temp->data=i+1;
		temp->next=NULL;
		if(rear==NULL)
		{
		front=temp;
		rear=temp;}
		else
		{
			rear->next=temp;
			rear=temp;
		}	
	}
	rear->next=front;
	return front;
}
node* del(node *head,int k,int &n)
{
	node* temp1=new node;
	node* temp2=new node;
	temp1=temp2=head;
	k=k%n;
	for(int i=0;i<k;i++)
	{
		temp1=temp1->next;
		if(i>0)
		temp2=temp2->next;
	}
	if(k==0)
	{
		for(;temp2->next!=temp1;temp2=temp2->next);
	}
	temp2->next=temp1->next;
	delete temp1;
	n--;
	return temp2->next;
}
