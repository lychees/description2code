class Solution(object):
    def solve(self, r, c):
        m = min(r, c)
        return (m*(4+(m-1)*(r+c+2)))/2


if __name__ == "__main__":
    tc = int(raw_input())
    while (tc > 0):
        r, c = map(int, raw_input().split())
        ans = Solution()
        print ans.solve(r, c)
        tc -= 1
