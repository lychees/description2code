#include<bits/stdc++.h>
using namespace std;
int main()
{
	string str;
	long long i,x,y,z,p,q,r;
	map<long long,long long>mp;
    cin>>str;
    mp[0] = 0;
    for(i = 0; i < str.length();i++)
    	mp[i+1] = (mp[i] + (str[i] - 'a')+ str.length());
    cin>>q;
    while(q-- )
    {
    	cin>>x>>y>>p>>r;
    	if(y-x != r-p  || str[x-1] != str[p-1] || str[y-1] !=  str[r-1])
            cout<<"No"<<endl;
    	else if((mp[y] - mp[x-1]) == (mp[r] - mp[p-1]))
    		cout<<"Yes"<<endl;
    	else
    		cout<<"No"<<endl;
    }
}
