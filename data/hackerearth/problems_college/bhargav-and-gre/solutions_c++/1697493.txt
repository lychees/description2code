#include <iostream>
using namespace std;

int main()
{
	string w,str;
	long t,s,a;
	
	cin >> w;
	cin >> t;
	s=0;a=0;
	while(t--)
	{
	cin >> str;
	if(str[0] == w[0] && str.length()==w.length())
	s++;
	if(w[0]+3 == str[0] && str.length()==w.length())
	a++;	
    
	}
	cout << s << " " << a<< endl;
    return 0;
}
