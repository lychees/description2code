T = int(raw_input())
for _ in range(T):
    N = int(raw_input())
    if (N <= 2):
        print 1
    elif (N % 2 == 1):
        no1 = (N+1)/2
        no2 = no1+1
        repeat = (N-1)/2
        maxdiff = no1*repeat+no2*(repeat-1) + 1
        print maxdiff
    else:
        no1 = N/2
        no2 = no1+1
        maxdiff = no1*no1+no2*(no1-1)
        print maxdiff
