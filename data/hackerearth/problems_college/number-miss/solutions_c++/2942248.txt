#include <bits/stdc++.h>
using namespace std;
int main()
{
	int a1[10001];
	int a2[10001];
	for(int i=0; i<=10000; i++)
	{
		a1[i]=0;
		a2[i]=0;
	}
	int n,m;
	scanf("%d",&n);
	for(int i=1; i<=n; i++)
	{
		int temp;
		scanf("%d",&temp);
		a1[temp]=a1[temp]+1;
	}
	scanf("%d",&m);
	for(int i=1; i<=m; i++)
	{
		int temp;
		scanf("%d",&temp);
		a2[temp]=a2[temp]+1;
	}
	for(int i=0; i<=10000; i++)
	{
		if(a1[i]==a2[i])
		continue;
		else
		cout<<i<<" ";
	}
	return 0;
}
