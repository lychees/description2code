#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    for(int i=0; i<t; i++)
    {
        int x,y;
       long long n,temp;
        cin>>n;
        if((n%2)!=0)
            n--;
        if(n%4==0)
        {
            temp=n/4;
            long long result=temp*temp;
            cout<<result<<"\n";

        }
        else
        {
            temp=n/2;
            long long x=temp/2,y=temp-x;
            long long max_area=x*y;
            cout<<max_area<<"\n";
        }
    }
    return 0;
}
