#include <bits/stdc++.h>

using namespace std;

char stri[101];int chk[26];
main()
{
   int n,i,count=0;;scanf("%d",&n);for(i=0;i<n;i++)
   {
   	scanf("%s",stri);
   chk[stri[0]-97]++;
   
   }for(i=0;i<26;i++)
   {
   	if(chk[i]!=0)
   	count++;
   }for(i=0;i<26;i++)
   {
   	if(chk[i]>10)
   	{while(chk[i]>10)
   		{count++;
   		chk[i]=chk[i]-10;
   		}
   	}
   }printf("%d",count);
}
