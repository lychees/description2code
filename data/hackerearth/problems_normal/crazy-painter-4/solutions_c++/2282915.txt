using namespace std;

int main()
{
    int test_Cases = 0;
    cin >> test_Cases;
    
    while (test_Cases --)
    {
        long long int Painting_id;
        cin >> Painting_id;
        
        long long int paintCost[30];
        long long int totalSum = 0;
        for( int i = 0;i< 26;i++)
        {
            cin >> paintCost[i];
            totalSum = totalSum + paintCost[i];
        }
        
        long long int totalLettersUsed =  10 * Painting_id + 2 * Painting_id;
        
        long long int painting_cost = 0;
        
        long long int whole_used = totalLettersUsed / 26;
        
        int partial_used = totalLettersUsed % 26 ;
        
        //cout << "total sum --> " <<totalSum<<endl;
        //cout << "TotalLetterUsed -> " << totalLettersUsed << " whole_used --> " <<whole_used << " partial used --> " <<partial_used<<endl;
        
        if ( whole_used >= 1)
            painting_cost = whole_used * totalSum + 13 * (whole_used -1) * (whole_used);
        
        //cout << "painting cost "<<painting_cost<<endl;
        
        for( int i = 0 ;i < partial_used ; i++)
            painting_cost = painting_cost + paintCost[i] + whole_used;
        
        cout << painting_cost << endl;
        
        
    }
}
