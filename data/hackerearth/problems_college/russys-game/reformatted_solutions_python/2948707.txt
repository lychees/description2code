n = int(raw_input())
for j in range(n):
    n = int(raw_input())
    a = map(int, raw_input().split())
    count = [0]*n
    i = 0
    count[0] = 1
    while a[i] != -1:
        i = a[i]
        count[i] += 1
        if count[i] > 1 or i >= n:
            print "Lose"
            break
    else:
        print "Win"
