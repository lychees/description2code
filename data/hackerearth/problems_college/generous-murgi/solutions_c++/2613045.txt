#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,x,y;
    cin >> t;
    while(t--)
    {
        cin >> x >> y;
        
        vector<int> x_ke_fact;
        vector<int> n_ke_fact;
        for(int i = 1; i<=sqrt(x) +1; i++)
        {
            if(x%i==0)
            {
                x_ke_fact.push_back(i);
                x_ke_fact.push_back(x/i);
            }
        }
        for(int i = 1; i<=sqrt(x-y) +1; i++)
        {
            if((x-y)%i==0)
            {
                n_ke_fact.push_back(i);
                n_ke_fact.push_back((x-y)/i);
            }
        }
        
        sort(x_ke_fact.begin(),x_ke_fact.end());
        sort(n_ke_fact.begin(),n_ke_fact.end());
        x_ke_fact.erase( unique( x_ke_fact.begin(), x_ke_fact.end() ), x_ke_fact.end() );
        n_ke_fact.erase( unique( n_ke_fact.begin(), n_ke_fact.end() ), n_ke_fact.end() );
        int ans = 0,pos;
        vector<int>::iterator iter;
        for(int i  = 0; i<n_ke_fact.size(); i++)
        {
            iter = upper_bound(x_ke_fact.begin(),x_ke_fact.end(),n_ke_fact[i]);
            pos = iter - x_ke_fact.begin() -1;
            if(iter==x_ke_fact.end() || x_ke_fact[pos]!=n_ke_fact[i])
            {
               if(x%n_ke_fact[i]==y)
                ans++;
            }

        }
        if(y == 0)
           ans = x_ke_fact.size();

        if (x == y)
           ans = -1;

        cout << (ans) << "\n";
    }

}
