#include<bits/stdc++.h>
using namespace std;
long long n,a[60],m,g,MOD=1000000007,ans=1;
int main()
{
    cin>>n;
    cin>>a[0];
    m=a[0];g=a[0];
    for(int i=1;i<n;i++)
    {
        cin>>a[i];
        m*=a[i];
        m%=MOD;
        g=__gcd(g,a[i]);
    }
    while(g!=0)
    {
        if(g%2!=0)ans=(ans*m)%MOD;
        m=(m*m)%MOD;
        g/=2;
    }
    cout<<ans<<endl;
    return 0;
}
