LL ss[10000090]={0};
void sei(){
	
	for(LL i=2;i<10000090;i++){
		if(!ss[i]){
			for(LL j=2*i;j<10000090;j+=i){
				ss[j]=1;
			}
		}
	}
}

int main(){
    //ios_base::sync_with_stdio(0);
	vector<LL> vec;
	sei();
	for(LL i=2;i<10000090;i++) if(!ss[i]) vec.PB(i); //cout << vec.size();

	LL tt=1;
	FL(0,i,vec.size()){
		tt=((vec[i]%MOD) * (tt%MOD))%MOD;
		vec[i]=tt%MOD;		
	}
	int t,n;SI(t); W(t){
		SI(n);
		cout << vec[n-1]%MOD << endl;
	}
    
    return 0;
}
