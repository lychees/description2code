#include <iostream>
using namespace std;
 
int arr[1000002];
int m1,m2;
 
bool fun(int n)
{
	if(arr[n]!=-1)
		return arr[n];
	else
	{
		arr[n]=0;
		int a,b;
		a=n-m1;
		b=n-m2;
		if(a<0 && b<0)
		{
			arr[n]=0;
			return 0;
		}
		if(a>=0)
		{
		if(arr[a]!=-1)
		{
			if(arr[a]==0)
			{
				arr[n]=1;
				return 1;
			}
		}
		else
		{
			arr[a]=fun(a);
			if(arr[a]==0)
			{
				arr[n]=1;
				return 1;
			}
		}
		}
		if(b>=0)
		{
		if(arr[b]!=-1)
		{
			if(arr[b]==0)
			{
				arr[n]=1;
				return 1;
			}
		}
		else
		{
			arr[b]=fun(b);
			if(arr[b]==0)
			{
				arr[n]=1;
				return 1;
			}
		}
		}
	}
	return arr[n];
}
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n>>m1>>m2;
    	for(int i=0;i<1000001;i++)
    	{
    		arr[i]=-1;
    	}
    	arr[0]=0;
    	int ans=fun(n);
    	if(ans==1)
    		cout<<"1"<<endl;
    	else
    		cout<<"0"<<endl;
    }
    return 0;
}
