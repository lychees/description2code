#include <bits/stdc++.h>
using namespace std;

int main()
{
	const int m=2010;int n;cin>>n;
	bitset<m> am[m];
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			int x;cin>>x;
			am[i][j]=x;
		}
	}
	long long int ans=0;
	for(int i=0;i<n;i++){
		for(int j=i+1;j<n;j++){
			long long int cnt = (am[i] & am[j]).count();
			//cout<<"r="<<i+1<<" c="<<j+1<<" cnt="<<cnt<<endl;
			ans+=(cnt*(cnt-1))/2;
		}
	}
	cout<<ans/2;
    return 0;
}
