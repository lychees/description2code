#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int test,x,y,n;
	scanf("%d",&test);
	while(test--){
		int c1 = 0,c2 = 0;
		vector< int> arr;
		scanf("%d%d%d",&x,&y,&n);
		for(int i = 0;i < n; ++i){
			int g;
			scanf("%d",&g);
			arr.push_back(g);
		}
		sort(arr.begin(),arr.end());
		int sum = 0;
		for(int i = 1;i <= n;++i){
			sum += arr[i - 1];
			if(sum <= x)c1 = i;
			if(sum <= y)c2 = i;
		}
		if(c1 > c2){
			printf("Chef Won\n");
		}
		else if(c1 < c2){
			printf("Jhool Won\n");
		}
		else printf("Tie\n");
	}
    return 0;
}
