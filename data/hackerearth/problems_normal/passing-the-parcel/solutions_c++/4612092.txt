#include <iostream>
#include<list>
using namespace std;

int main()
{
    int n;
    cin>>n;
   	string s;
   	cin>>s;
   	list<int> mylist;
   	for(int i=1;i<=n;i++)
   	{
   		mylist.push_back(i);
   	}
   	list<int>::iterator it,temp;
   	int i=0;
   	int number=s.size();
   	it=mylist.begin();
   	while(mylist.size()>1)
   	{
   		if(s.at(i)=='b')
   		{
   			temp=it;
   			it++;
   			mylist.erase(temp);
   		}
   		else
   		{
   			it++;
   		}
   		if(it==mylist.end())
   			it=mylist.begin();
   		i++;
   		if(i==number)
   			i=0;
   	}
   	it=mylist.begin();
   	cout<<*it<<endl;
}
