using namespace std;

#define mod 1000000007
#define ll long long
#define INF (ll)(1e18)
#define mp make_pair
#define fill(a,v) memset(a, v, sizeof a)


bool vis[1000006];
ll dp[1000006];
vector < pair<ll,ll> > v[1000006]; 
ll dfs(ll now)
{
	ll i,j;
	if(vis[now]==false)
	return 0;
	
	if(dp[now]!=-1)
	return dp[now];
	
	ll temp=0;
	for(i=0;i<v[now].size();i++)
	temp=max(temp,v[now][i].second+dfs(v[now][i].first));
	
	return dp[now]=temp+2;
}
int main()
{
	ll N,M;
	cin>>N>>M;
	fill(vis,0);
	fill(dp,-1);
	while(M--)
	{
		ll X,Y,W;
		cin>>X>>Y>>W;
		v[X].push_back(mp(Y,W));
		vis[X]=true;
	}
	ll ans=0;
	ll i,j;
	for(i=1;i<=N;i++)
	ans=max(ans,dfs(i));
	cout<<ans;

return 0;
}
