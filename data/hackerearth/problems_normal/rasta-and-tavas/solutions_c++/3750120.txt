#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,k;
	scanf("%d%d",&n,&k);
	int ans = 0;
	for(int i=1;i<=n;i++)
	{
		int t = i;
		int sum = 0;
		
		if(t%2 == 0)
		{
			while(t%2 == 0)
				t /= 2;
			sum += 2;
		}

		for(int j=3;j*j<=t;j+=2)
		{
			if(t%j == 0)
			{
				while(t%j == 0)
					t /= j;
				sum += j;
			}
		}

		if(t > 1)
			sum += t;

		if(sum == k)
			ans++;
	}

	printf("%d\n",ans);

	return 0;
}
