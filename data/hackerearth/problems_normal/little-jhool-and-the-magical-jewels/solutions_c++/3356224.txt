#include <iostream>
#include<algorithm>
using namespace std;


int main()
{
    int t,i,j,arr1[1000];
    char arr[1000];
    cin>>t;
    for(i=0;i<t;i++)
    {
    	for(j=0;j<4;j++)
    	{
    		arr1[j]=0;
    	}
    	cin>>arr;
    	for(j=0;arr[j]!='\0';j++)
    	{
    		if (arr[j]=='r')
    	   		arr1[0]++;
    	   	else if(arr[j]=='u')
    	   		arr1[1]++;
    	   	else if(arr[j]=='b')
    	   		arr1[2]++;
    	   	else if(arr[j]=='y')
    	   		arr1[3]++;
    	   	else
    			continue;
    	}
    	sort(arr1,arr1+4);
    	cout<<arr1[0];
    	cout<<endl;
    }
    
    return 0;
}
