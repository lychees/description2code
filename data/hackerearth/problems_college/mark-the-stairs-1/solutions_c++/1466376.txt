#include<bits/stdc++.h>
 
using namespace std;
 
int main()
{
    long long int t;
    scanf("%lld",&t);
  while(t--)
  {
       long long int x,y;
       scanf("%lld%lld",&x,&y);
       
       if(x == y)
       {
            if(x%2 == 0)
            cout<<2*x<<"\n";
            else
            cout<<2*x-1<<"\n";
       }
       else if(x-2 == y)
       {
          if(x%2 == 0)
          cout<<abs(2*x-2)<<"\n";
          else
          cout<<abs(2*x-3)<<"\n";
 
      }
       else
         cout<<"No Number\n";
  }
  return 0;
}
