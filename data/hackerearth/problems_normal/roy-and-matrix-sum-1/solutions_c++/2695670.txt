#include <iostream>

#define ull unsigned long long int 

using namespace std;

ull util (ull x)
{
	ull re = x*(x+1) ;
	return re ; 
}

int main()
{
	int test ; 
	cin >> test ;
	
	while(test --)
	{
		int n ;
		cin >> n ;
		
		ull sum = 0 ;
		
		for(int i = 1 ; i < n ; i++)
		{
			sum += util(i) ;
		}
		
		cout << sum << endl ; 
	}
	
	return 0;
}
