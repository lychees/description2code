#include<bits/stdc++.h>
using namespace std;
int a[102],dp[10004];
int main()
{
int t;
scanf("%d",&t);
while(t--)
{
 int n;
 scanf("%d",&n);
 for(int i=0;i<n;i++)
  {
  scanf("%d",&a[i]);
  }
  memset(dp,0,sizeof(dp)); 
   dp[0]=1;
   for(int j=0;j<n;j++)
    {
      for(int i=10000;i;i--)
       {
          if(a[j]<=i && dp[i-a[j]])
           dp[i]=1;
       } 
     }
     int cnt=0;
     for(int i=0;i<=10000;i++)
     {
      if(dp[i])
      cnt++;
     }
      printf("%d\n",cnt);
    }
return 0;
}
