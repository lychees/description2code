def gcd(a, b):
    if b > a:
        return gcd(b, a)

    if a % b == 0:
        return b

    return gcd(b, a % b) 


n=input()

mod=(10**9)+7
temp=arr[0]

for item in arr:
	temp=gcd(item,temp)
	temp=temp%mod
	
	
prod=1
for it in arr:
	prod=prod*it
	prod=prod%mod
