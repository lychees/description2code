
#include<bits/stdc++.h>
using namespace std;

#define max(a,b) (a)>(b)?(a):(b)
#define min(a,b) (a)>(b)?(b):(a)

#define rep(i,a,b) for(i=a;i<b;i++)
#define f first
#define s second
#define sqr(a) (a)*(a)
#define Pi 3.141592653589793
#define MOD 1000000007

// For i/o
#define sd1(a) scanf("%d",&a)
#define sd2(a,b) scanf("%d %d",&a,&b)
#define sd3(a,b,c) scanf("%d%d%d",&a,&b,&c)
#define sf scanf
#define pf printf

// For debugging
#define deb(a) printf("deb%d\n",a)

int i,j;

int main()
{
	int t;
	sd1(t);
	while(t--)
	{
		int n;
		double arr[100];
		sd1(n);
		rep(i,0,n)
			sf("%lf",&arr[i]);
		rep(i,0,n)
		{
			double temp;
			sf("%lf",&temp);
			arr[i] *= temp;
		}
		double max=-1,den=0;
		int index;
		rep(i,0,n)
		{
			if(arr[i] > max)
			{
				max = arr[i];
				index = i;
			}
			den = den + arr[i];
		}
		printf("%d %0.3lf\n",index,max/den);
	}
	return 0;
}
