s = set()
s1 = set()

for x in xrange(1, 90):
    for y in xrange(x, 90):
        n = x*x*x + y*y*y
        if n not in s:
            s.add(n)
        else:
            s1.add(n)

for _ in xrange(int(raw_input())):
    n = int(raw_input())
    done = False
    while n:
        if n in s1:
            done = True
            print n
            break
        n -= 1
    if not done:
        print -1
