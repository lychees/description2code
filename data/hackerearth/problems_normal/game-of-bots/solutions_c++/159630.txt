#include <iostream>
#include<stdio.h>
#include<stack>
#include<string.h>
using namespace std;


char c1[1000001],c2[1000001];

int main()
{
	int t,j;
	cin>>t;
	
	 
	
	stack< pair<char,int> > st;
	
	while(t--)
	{
		scanf("%s",c1);
		scanf("%s",c2);
		int flag = 0;
		int l1 = strlen(c1),l2 = strlen(c2);
		
		if(l1!=l2)
		{	
			flag = 1;
			
		}
		
		else
		{	
			while(st.size()!=0)
				st.pop();
				
			pair<char,int> x;
			
			for(int i = 0; i<l1; i++)
			{
				if(c1[i] == 'A' || c1[i] == 'B')
				{
					x.first = c1[i];
					x.second = i;
					st.push(x);
				}
			}
			
			for(int i = l1-1; i >= 0; i--)
			{
				if( c2[i] == '#' )
					continue;
				else
				{
					if(c2[i] == 'A' && c2[i] == st.top().first)
					{	if(i > st.top().second)
						{
							flag = 1;
							break;
						}
						else
						{
							st.pop();
							continue;
						}
					}
					else if(c2[i] == 'B' && c2[i] == st.top().first)
					{	if(i < st.top().second)
						{
							flag = 1;
							break;
						}
						else
						{
							st.pop();
							continue;
						}
					}
					
					
					else
						flag = 1;
					
				}
				
				
				
			}
			
			
			
			
		}
		
		if(!flag )
			cout<<"Yes\n";
		else cout<<"No\n";
		
	}


	return 0;
}
