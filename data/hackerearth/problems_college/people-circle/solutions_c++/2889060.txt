#include<bits/stdc++.h>
using namespace std;

#define fastScan ios_base::sync_with_stdio(0); cin.tie(NULL);


int main()
{
    fastScan;
    
    int t;
    cin>>t;
    while(t--){
        int m,f,k,i,j,s,x,count=0,ite=0;
        cin>>m>>f>>k;
        s=m+f;
        int arr[s];
        vector<int> v;
        for(i=0;i<s;i++)
          v.push_back(i);
        for(i=0;i<s;i++)
         arr[i]=0;
        while(f--){
            x=v[(ite+k-1)%v.size()];
            arr[x]=1;
            v.erase(v.begin()+((ite+k-1)%v.size()));
            ite=(ite+k-1)%(v.size()+1);
            
        }
        for(i=0;i<s;i++){
            if(arr[i])
             cout<<"F";
            else
             cout<<"M"; 
        }
        cout<<endl;
    }
    return 0;
}
