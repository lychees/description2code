from __future__ import print_function
tc = input()
while tc:
    shift = input()
    codeword = raw_input()
    lovedecoded = []
    for i in codeword:
        ind = ord(i)-shift
        if ind >= 65:
            lovedecoded.append(chr(ind))
        else:
            tmp = 65 - ind  # ord('A') = 65
            lovedecoded.append(chr(91 - tmp))  # ord('Z') = 90
    for i in lovedecoded:
        print(i, end='')
    print()
    tc -= 1
