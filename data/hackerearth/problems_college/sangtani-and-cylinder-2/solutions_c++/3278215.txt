#include <bits/stdc++.h>
using namespace std;
#define pi (2*acos(0.0))
int main(){
	int t;
    double v,r,h,area;
	scanf("%d",&t);
	while(t--){
		scanf("%lf",&v);
		r = v/(2*pi);
		r = pow(r,(double)1/3);
		h = v/(pi*r*r);
		area = 2*r*pi*(h+r);
		printf("%0.9lf\n",area);
	}
	return 0;
}
