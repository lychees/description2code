#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main() 
{
    int i,l,t;
    string str;
    vector<string>s;
    cin>>t;
    cin>>str;
    s.push_back(str);
    t--;
    int j = 0;
    while(t--) {
        string str1="";
        cin>>l>>str;
        for(i = 0; i < l; i++) {
            str1 += s[j][i];
        }
        str1 += str;
        s.push_back(str1);
        j++;
    }
    cout<<s[s.size()-1]<<endl;
	return 0;
}
