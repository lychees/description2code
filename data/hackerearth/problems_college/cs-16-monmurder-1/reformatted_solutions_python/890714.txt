t = input()
for i in range(t):
    n = input()
    pile = []
    j = 2

    while (n != 1):
        if (n % j == 0):
            c = 0
            while (n % j == 0):
                c = c+1
                n = n/j

            pile.append(c)

        j = j+1

    ans = 0
    for j in pile:
        ans = (ans ^ j)

    if (ans == 0):
        print "Pappu Bhujia"
    else:
        print "Printf{}"
