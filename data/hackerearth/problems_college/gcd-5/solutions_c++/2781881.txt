#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T;
    unsigned long long int N,a,b,i;
    bool flag;
    cin>>T;
    
    while( T-- ){
    	
    	cin>>N;
    	flag = false;
    	cin>>a;
    	
    	for( i = 1 ; i < N ; i++  ){
    		cin>>b;
    		if( a <= b ){
    			a = b;
    		}
    		else{
    			//cout<<"NO\n";
    			flag = true;
    			//break;
    		}
    	}
    	
    	if( !flag )
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    
    return 0;
}
