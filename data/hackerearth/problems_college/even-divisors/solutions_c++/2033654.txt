#include <iostream>
#include <cmath>

using namespace std;

int main(void)
{
    int cases;
    cin >> cases;
    
    while (cases--)
    {
        long long num, div = 0, i = 1;
        cin >> num;
        
        if (num % 2 == 0)
        {
            long x = int(sqrt(num));
            while (i <= x)
            {	
            	if (num % i == 0)
            	{
	                if (i % 2 == 0)
	                {
	                    div++;
	                }
	                
	                long p = num / i;
	                
	                if (p % 2 == 0 && i * i != num)
	                {
	                    div++;
	                }
            	}
            	i++;
            }
        }
        
        cout << div << endl;
    }
}
