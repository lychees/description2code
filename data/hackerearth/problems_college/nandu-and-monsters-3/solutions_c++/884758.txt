#include <iostream>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
    long n,q,a,b;
    cin>>n>>q;
    string s;
    cin>>s;
    bool flag;
    while(q--)
    {
    	flag = true;
    	cin>>a>>b;
    	
    	while(a!=b)
    	{
    		if(s[a-1]=='1' || s[b-1]=='1')
    		{
    			flag = false;
    			break;
    		}
    		if(a>b)
    		{
    			a/=2;
    		}
    		else
    		{
    			b/=2;
    		}
    	}
    	
    	if(s[a-1]=='1' || s[b-1]=='1')
    	{
    		flag = false;
    	}
    	
    	if(flag)
    	{
    		cout<<"Yes\n";
    	}
    	else
    	{
    		cout<<"No\n";
    	}
    }
    return 0;
}
