#include<iostream>
using namespace std;
int main(){
 long int n,k;
 int top_left,top_right,bottom_right,bottom_left;
 int x[100000],y[100000];
 long int count=0; 
 cin>>n;
 cin>>k;
for(int i=0;i<k;i++) cin>>x[i];
for(int j=0;j<k;j++) cin>>y[j];

 for(int i=0;i<k;i++){ 
    top_left=x[i]+y[i]-2;
    top_right=x[i]+(n+1-y[i])-2;
    bottom_left=(n+1-x[i])+y[i]-2;
    bottom_right=(n+1-x[i])+(n+1-y[i])-2;   
    
     int small=top_right;

     int small1=bottom_right;

 if(top_left<small)
    small=top_left;

 if(bottom_left<small1)
    small1=bottom_left;
 
 if(small<small1) count+=small;

 else count+=small1; 

}

cout<<count<<endl;

return 0;
}
