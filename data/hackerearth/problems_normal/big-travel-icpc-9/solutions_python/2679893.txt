N = int(raw_input().strip())
x = []
y = []
sum_x = sum_y = sum = 0
for n in range(N):
	[x1,y1] = [int(i) for i in raw_input().strip().split()]
	x.append(x1)
	y.append(y1)
x.sort();y.sort()	
for i in range(N):
    sum_x += x[i]
    sum_y += y[i]
for i in range(N):
    sum_x -= x[i]
    sum_y -= y[i]
    sum += sum_x - (N-i-1)*x[i] + sum_y - (N-i-1)*y[i]
print sum%1000000007;
