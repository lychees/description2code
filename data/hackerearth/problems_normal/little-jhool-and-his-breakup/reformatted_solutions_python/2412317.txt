s = raw_input()
l = len(s)
flag = []
for i in range(4):
    flag.append(0)
for i in range(0, l):
    ch = s[i]
    if ch == 'l':
        flag[0] = 1
    elif (ch == 'o' and flag[0] == 1):
        flag[1] = 1
    elif (ch == 'v' and flag[1] == 1):
        flag[2] = 1
    elif (ch == 'e' and flag[2] == 1):
        flag[3] = 1
check = 1
for i in range(4):
    if flag[i] == 0:
        check = 0
        break
if check == 1:
    print "I love you, too!"
else:
    print "Let us breakup!"
