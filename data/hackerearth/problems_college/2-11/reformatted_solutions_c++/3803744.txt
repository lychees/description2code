#include <bits/stdc++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(false);
  int n;
  cin >> n;
  int adj[26][26];
  for (int i = 0; i < 26; i++) {
    for (int j = 0; j < 26; j++) {
      cin >> adj[i][j];
    }
  }

  for (int k = 0; k < 26; k++) {
    for (int i = 0; i < 26; i++) {
      for (int j = 0; j < 26; j++) {
        if (adj[i][k] + adj[k][j] < adj[i][j]) {
          adj[i][j] = adj[i][k] + adj[k][j];
        }
      }
    }
  }

  string a, b;
  while (n--) {
    cin >> a;
    cin >> b;
    int l = a.length();
    long sum = 0;
    for (int i = 0; i < l; i++) {
      if (a[i] != b[i])
        sum += adj[(a[i] - 'a')][(b[i] - 'a')];
    }
    cout << sum << endl;
  }
  return 0;
}
