#include<bits/stdc++.h>
#define ll long long int
using namespace std;
vector<pair<ll,ll> > v;
vector<pair<ll,ll> > snew;
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output8.txt","w",stdout);
	ll t,n,q,a,b,i,j,cnt,k;
	scanf("%lld",&t);
	while(t--)
	{
		v.clear();
		snew.clear();
		scanf("%lld%lld",&n,&q);
		for(i=0;i<n;i++)
		{
			scanf("%lld%lld",&a,&b);
			v.push_back(make_pair(a,b));
		}
		sort(v.begin(),v.end());
		snew.push_back(make_pair(v[0].first,v[0].second));
		cnt=0;
		for(i=1;i<v.size();i++)
		{
			if(v[i].first>snew[cnt].second && v[i].second>snew[cnt].second)
			{
				cnt++;
				snew.push_back(make_pair(v[i].first,v[i].second));
			}
			else if(v[i].first<=snew[cnt].second && v[i].second>snew[cnt].second)
			{
				cnt++;
				snew.push_back(make_pair((snew[cnt-1].second)+1,v[i].second));
			}
			else if(v[i].first<snew[cnt].second && v[i].second<snew[cnt].second)
			{
				
				continue;
			}
		}
		/*for(i=0;i<snew.size();i++)
		{
			printf("%lld %lld\n",snew[i].first,snew[i].second);
		}*/
		ll ans=-1;
		for(i=0;i<q;i++)
		{
			ans=-1;
			scanf("%lld",&k);
			ll temp=k;
			for(j=0;j<snew.size();j++)
			{
				temp=(temp+snew[j].first)-1;
				if(temp>=snew[j].first && temp<=snew[j].second)
				{
					ans=temp;
					break;
				}
				else
				{
					temp=(temp-snew[j].second);
				}
			}
			printf("%lld\n",ans);
		}
	}
	//fclose(stdout);
	return 0;
}
