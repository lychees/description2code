#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
    	int n;
    	cin >> n;
    	int a[n+1];
    	for(int i=1;i<=n;++i)
    		cin >> a[i];
    	int x;
    	int ans=0;
    	for(int i=1;i<=n;++i){
    		cin >> x;
    		ans=ans^(x-1)^(a[i]-x);
    	}
    	if(ans==0)
    		cout << "Second" << endl;
    	else
    		cout << "First" << endl;
    	
    }
    return 0;
}
