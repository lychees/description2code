#include <iostream>
#include<vector>
#include<cstdio>
#include<cstring>
using namespace std;

long long LIM = 1e12;
long long getNum(char * str, int low,int end){
	
	long long num = 0;
	for(int i=low;i<end;i++)
		num = num*10 + str[i]-'0';
	return num;
}


bool isValid(int x,int y,int z, char *str,int len){
	
	long long num = getNum(str,0,x);
	if((x-0>1 && str[0]=='0') || num>LIM)
		return false;
	num = getNum(str,x,y);
	if((y-x>1 && str[x]=='0')|| num>LIM)
		return false;
	num = getNum(str,y,z);
	if((z-y>1 && str[y]=='0') ||  num>LIM)
		return false;
	num = getNum(str,z,len);
	if((len-z>1 && str[z]=='0') || num>LIM)
		return false;
	return true;
}

long long getSum(int x,int y,int z,int len,char * str){
	return getNum(str,0,x) + getNum(str,x,y) + getNum(str,y,z) + getNum(str,z,len);
}

int main()
{
    char str[25];
    int t;
    long long ans=0;
    long long tmp=0;
    scanf("%d",&t);
    while(t--){
    	scanf("%s",str);
    //	cout<<str<<"\n";
    	int n=strlen(str);
    	ans=tmp=0;
    	for(int i=1;i<n-2;i++){
    		for(int j=i+1;j<n-1;j++){
    			for(int k=j+1;k<n;k++){
    				if(isValid(i,j,k,str,n)){
    					tmp = getSum(i,j,k,n,str);
    					if(tmp>ans)
    						ans=tmp;
    				}
    			}
    		}
    	}
    	if(ans)
    		cout<<ans<<"\n";
   	 	else{
   	 		if(strlen(str)==4 && str[0]=='0' && str[1]=='0' && str[2]=='0' && str[3]=='0')
   	 			cout<<"0\n";
   	 		else
   	 		 cout<<"unlucky\n";
   	 	}
    		
    }    
    return 0;
}
