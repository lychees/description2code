#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int a[t],temp;
    int c=0;
    unsigned long long int n;
    for(int i=0;i<t;i++)
    {
    	cin>>n;
    	a[i]=0;
    	c=0;
    	while(n!=0)
    	{
    		temp=n%10;
    		if(temp==4 || temp==7)
    		{
    			c++;
    		}
    		n=n/10;
    	}
    	if(c==0)
    	a[i]=1;
    	else
    	{
    	while(c!=0)
    	{
    		temp=c%10;
    		if(temp!=4 && temp!=7)
    		{
    			a[i]=1;
    			break;
    		}
    		c=c/10;
    	}
    	}
    }
    for(int i=0;i<t;i++)
    {
    	if(a[i]==1)
    	cout<<"NO"<<endl;
    	else
    	cout<<"YES"<<endl;
    }
    return 0;
}
