#include <iostream>
using namespace std;

int main()
{
   int t,i,n,c,count,flag,s;
   cin>>t;
   while(t--)
   {
   	c=0,count=0,flag=0;
   	cin>>n;
   	int a[n];
   	for(i=0;i<n;i++)
   	{
   		cin>>a[i];
   	}
    while(count<=n)
    {
    	s=a[c];
    	if(s==-1)
    	{
    	cout<<"Win\n";
    	flag=1;
    	break;
    	}
    	c=s;
    	count++;
    }
    if(flag==0)
    {
    	cout<<"Lose\n";
    }
   }
}
