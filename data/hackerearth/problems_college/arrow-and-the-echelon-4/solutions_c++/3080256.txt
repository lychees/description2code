#include<iostream>
using namespace std;
int main()
{
    int t,n,a,i,b,k;
    cin>>t;
    while(t>0)
    {
        a=-1;
        b=-1;
        cin>>n;
        k=n/3;
        for(i=k;i>=0;i--)
        {
            if((n-(3*i))%5==0)
            {
               a=3*i;
               b=n-(3*i);
               break;
            }
        }
        if(a==-1)
            cout<<"-1";
        else
        {
            for(i=1;i<=a;i++)
            {
                cout<<"5";
            }
            for(i=1;i<=b;i++)
            {
                cout<<"3";
            }
        }
        t--;
        cout<<"\n";
    }
    return 0;
}
