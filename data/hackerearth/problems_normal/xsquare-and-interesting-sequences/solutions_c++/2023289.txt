#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000009
long int pow(int a, long long int num)
{
	if(num==0)
		return 1;
	long long int temp;
	temp=pow(a,num/2);
	if(num%2==0)
	{
		temp=((temp%MOD)*(temp%MOD))%MOD;
		return temp;
	}
	else
	{
		temp=((a)*((((temp%MOD)*(temp%MOD))%MOD)%MOD))%MOD;
		return temp;
	}
}
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   		long long int n;
   		cin>>n;
   		long int res=pow(26,(n+1)/2);
   		cout<<res<<endl;
   }
}
