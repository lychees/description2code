#include <iostream>
using namespace std;
 
void initi(int asci[])
{
    for(int i = 0; i < 256; i++)
        asci[i] = 0;
}

 int main(void)
{
    int T, N, j;
    char str[11];
    int asci[256];
    cin >> T;
    while(T--)
    {
        initi(asci);
        cin >> N;
        for(j = 0; j < N; j++)
        {
            cin >> str;
            asci[str[0]]++;
        }
        for(j = 0; j < 256; j++)
        {
            if(asci[j] > 1)
                break;
        }
        if(j >= 256)
            cout << "YES\n";
        else
            cout << "NO\n";
    }
    return 0;
}
