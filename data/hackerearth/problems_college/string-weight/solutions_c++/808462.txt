int main(int argc,char* argv[])
{

        while(T--)
        {
                int weight=0,i,k;
                char str[10001]="";
                int arr[26]={0};
                scanf("\n%s",str);
                //printf("%s",str);
                for(i=0;str[i]!='\0';i++)
                {
                        k=str[i]-'a';
                        arr[k]++;
                }
                scanf("%d\n",&N);

                        scanf("%c ",&c);
                        //cout<<c;

                        arr[k]=0;
                }
                for(i=0;i<26;i++)
                {
                        weight+=(i+1)*arr[i];
                }
                printf("%d\n",weight);
        }

        return 0;
}
