#include <iostream>
using namespace std;
int n,t;
long long int a;
int main()
{
    //cout << "Hello World!" << endl;
    ios_base::sync_with_stdio(false);
    cin>>n;
    for(int i=0;i<n;i++){
    	for (int j=0;j<n;j++){
    		cin>>t;
    		if (i==j) a+=t;//cout << "added " << t <<endl;} 
    		if ((i+j) == (n-1)) a-=t;//cout << "substrscted " << t << endl; }
    	}
    }
    cout << abs(a) << endl;
    return 0;
}
