#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    cin>>s;
    bool hash[26];
    memset(hash,0,sizeof(hash));
    for(int i=0;i<s.length();++i)
    	hash[s[i]-'a']=true;
    int count=0;
    for(int i=0;i<26;i++)
    	if(hash[i])
    		count++;
    if(count & 1) {
    	cout<<"IGNORE HIM!";
    }else
    	cout<<"CHAT WITH HER!";
    return 0;
}
