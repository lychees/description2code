#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
const int M = 1e3 + 5;
#define MOD 1000000007
#define ll long long int
ll U[M];
ll P[M];
int super[M];
int wid[M];
void supercal(int n, int k)
{
	memset(super, 0, sizeof(super));
	super[0] = 1;
	for(int i = 0;i<k;i++)
	{
		int u = wid[i];
		if(u>n)continue;
		for(int j = wid[i];j<=n;j++)
		{
			super[j] = super[j] | super[j-u];
		}
	}
}
void cal(int N, int K)
{
	memset(P, 0, sizeof(P));
	memset(U, 0, sizeof(U));
	supercal(N, K);
	U[0] = 1;P[0] = 0;
	for(int i = 1;i<=N;i++)
	{
		U[i] = (U[i-1] + P[i-1])%MOD;
		for(int j = 1;j<=N;j++)
		{
			if(super[j])
			{
				P[i] += U[i-j];
				P[i] %= MOD;
			}
		}
	}
}
int main()
{
    int n, k;
    scanf("%d %d", &n, &k);
    for(int i = 0;i<k;i++)
    {
    	scanf("%d", &wid[i]);
    }
    cal(n, k);
    ll ans = (P[n] + U[n])%MOD;
    printf("%lld", ans);
    return 0;
}
