#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t,flag;
    long long a,b,i,j,sum;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	sum=0;
    	for(i=a;i<=b;i++)
    	{
    		flag=0;
    		if(i==1)
    		flag=1;
    		for(j=2;j<=sqrt(i);j++)
    		{
    			if(i%j==0)
    			{
    			flag=1;
    			break;
    			}
    		}
    		if(flag==0)
    		sum=sum+i;
    		
    	}
    cout<<sum<<endl;	
    	
    }
    
    
    
    return 0;
}
