#include <iostream>
#include<algorithm>
using namespace std;

int main() {
	
	
		int n;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++) cin>>a[i];
		int max=0,min=0;
		for(int i=0;i<n;i++)
		{
			max+=a[i];
			if(max>min) min=max;
			if(max<0) max=0;
		}
		if(min>0)
		cout<<min;
	
	return 0;
}
