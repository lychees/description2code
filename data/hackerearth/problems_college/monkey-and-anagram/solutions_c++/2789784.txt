#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s,t;
    	cin>>s>>t;
    	map<char ,int> f1,f2;
    	for(int i=0;i<t.length();i++)
    	{
    		f1[t[i]]++;
    	}
    	int f=0;
    	for(int i=0;i<s.length();i++)
    	{
    	//	cout<<s[i]<<endl;
    		if(f1[s[i]]==0)
    		{
    			f=1;
    			cout<<"-1\n";
    			break;
    		}
    		else
    		{
    			f1[s[i]]--;
    		}
    	}
    	if(f)
    	{
    		continue;
    	}
    	int ans=0;
    	for(int i=0;i<t.length();i++)
    	{
    		if(f1[t[i]]>0)
    		{
    			ans++;
    			f1[t[i]]=0;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
