#include <bits/stdc++.h>
using namespace std;
int count(int n)
{
	int r,c=0;
	while(n)
	{
		r=n%2;
		if(r)
			c++;
		n/=2;
	}
	return c;
}
int main()
{
    int t,a,b,ans;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d%d",&a,&b);
    	a=a^b;
    	ans=count(a);
    	if(ans%2)
    		printf("NO\n");
    	else
    		printf("YES\n");
    }
    return 0;
}
