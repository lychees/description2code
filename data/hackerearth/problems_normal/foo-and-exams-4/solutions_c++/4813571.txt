#include<iostream>
#include<cmath>
int main()
{
	unsigned long k,a,b,c,d,t,T;
	std::cin>>T;
	while(T--)
	{
		std::cin>>a>>b>>c>>d>>k;
		t=(int)floor(cbrt((float)((k)/a)))+1; 	//get the upper bound by cuberoot(k/a)
		while(t--)								//work your way down
		if((d+(t)*(c+(t)*(b+a*(t))))<=k||t==0) 	//when the polynomial goes below k
		{
 		std::cout<<t<<"\n";						//print the value of t
		break;									//break the loop
 		}
	}
    return 0;
}
