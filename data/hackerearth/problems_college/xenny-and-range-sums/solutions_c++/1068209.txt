#include <iostream>
using namespace std;
long long a[1000007], f[1000007];
int main(){
    int n,q,l,r;
    cin>>n>>q;
    a[0]=0,f[0]=0;
    for(int i=1;i<=n;i++) 
    	cin>>a[i];
    for(int i=1;i<=n;i++) 
    	f[i]=f[i-1] + a[i];
    while(q--){
        cin>>l>>r;
        cout<<f[r]-f[l-1]<<" "<<a[r]-(f[r-1]-f[l-1])<<"\n";
    }
}
