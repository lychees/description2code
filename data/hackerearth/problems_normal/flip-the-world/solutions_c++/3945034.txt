#include <iostream>
using namespace std;
int n,m;
int zr,zc;

int main()
{
	int t;
	cin >> t;
	int i,j;
	while(t--){
		cin >> n >> m;
		string str;
		int a[n][m];
		for(i=0;i<n;i++){
			cin >> str;
			for(j=0;j<m;j++){
				a[i][j] = str[j]-'0';
			}
		}

		// cout << endl << endl;
		int count =0;
		//find zero
		bool flag = false;
		for(i=n-1;i>=0;i--){
			for(j=m-1;j>=0;j--){
				if(a[i][j] == 0){
					flag = true;
					break;
				}
			}
			if(flag == true){
				break;
			}
		}
		zr = i;
		zc = j;
		// cout << zr<< " " << zc << endl;
		while(i >=0 && j>=0){
			for(i=zr;i>=0;i--){
				for(j=zc;j>=0;j--){
					a[i][j] ^=1;
					// cout << a[i][j] ;
				}
				// cout << endl;
			}
			// for(int k=0;k<n;k++){
			// 	for (int l=0;l<m;l++){
			// 		cout << a[k][l];
			// 	}
			// 	cout << endl;
			// }
			// cout << "------------------------------------------" <<endl;
			count +=1;
			flag = false;
			for(i=n-1;i>=0;i--){
				for(j=m-1;j>=0;j--){
					if(a[i][j] == 0){
						flag = true;
						break;
					}
				}
				if(flag == true){
					break;
				}
			}
			zr = i;
			zc = j;
		}
		cout << count << endl;
	}
    return 0;
}
