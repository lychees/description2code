#include <iostream>
#include<vector>
#include<string>
#include<algorithm>
#include<sstream>
using namespace std;

int main()
{
    int T; cin >> T; T++;
    while(T--)
    {
    	vector<string> vec_string;
    	string str;
    	getline(cin, str);
    	stringstream ss(str);
    	string local_string;
    	while(ss >> local_string)
    	{
    		if( std::find(vec_string.begin(), vec_string.end(), local_string) == vec_string.end())
    		{
    			vec_string.push_back(local_string);
    		}
    	}
    	for(int i=0; i<vec_string.size(); i++)
    		cout << vec_string[i] <<" ";
    	cout << endl;
    }
    return 0;
}
