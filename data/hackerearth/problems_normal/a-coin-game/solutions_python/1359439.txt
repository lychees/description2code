maxn = 10**6
cold_pos = {}

def gen_cold():
	d = 0
	i = 0
	while (i < maxn):
		if not (cold_pos.has_key(i)):
			cold_pos[i] = i + d
			cold_pos[i + d] = i
			d += 1
		i += 1

if __name__ == "__main__":
	gen_cold()
	t = int(raw_input())
	for i in range(t):
		n = map(int, raw_input().split(" "))
		if n[1] == cold_pos[n[0]]:
			print "Don't Play"
		else:
			print "Play"
