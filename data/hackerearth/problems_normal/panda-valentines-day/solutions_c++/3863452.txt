ll power(ll a,ll b)
{
	if(!b)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2);
	temp=(temp*temp);
	if(b&1)
		temp=(temp*a);
	return temp;
}
ll ncr(int n,int r)
{
	ll ans=1;
	int i;
	r=min(r,n-r);
	for( i=0;i<r;i++ )
	{
		ans*=(n-i);
		ans/=(i+1);
	}
	return ans;
}
