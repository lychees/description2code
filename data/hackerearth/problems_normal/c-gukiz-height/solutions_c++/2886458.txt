#include "bits/stdc++.h"

using namespace std;

const int MAXN = 100007;
const long long INF = 1LL*1000*1000*1000*1000+7;
static int a[MAXN];
static long long pref[MAXN];

long double podwa(long long p)
{
  return (long double)p*(long double)(p+1)/2;
}

int main()
{
  int n, H;
  cin >> n >> H;
  long long sum = 0;
  for(int i=0; i<n; i++)
  {
    cin >> a[i];
    sum += a[i];
    pref[i+1] = sum;
    if(sum >= H - podwa(i+1))
    {
      cout << i+1 << "\n";
      return 0;
    }
  }


  long long res = INF;
  for(int i=0; i<n; i++)
  {
    long long p = 0, k = INF;
    while(k-p > 1)
    {
      long long m = (p+k)/2;
      long double tmp = (long double)pref[i] + (long double)sum*(long double)m;
    //  assert(tmp >= 0);
      //cerr << tmp << "\n";
      if(tmp > (long double)H - podwa(i + (long double)m*(long double)n))
        k = m;
      else
        p = m;
    }
    //cerr << 1LL*p*n << "\n";
    for(int e=max(0LL,p-50); e<p+50; e++)
    {
      long double tmp = (long double)pref[i] + (long double)sum*(long double)e;
      //cerr << tmp << " " << H-  - podwa(i + 1LL*e*n)
        if(tmp >=  H - podwa(i + (long double)e*n) - 1e-3)
          res = min(res, (long long)(i + (long double)e*n));
    }
  }
  cout << res << "\n";
}
