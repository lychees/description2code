n=int(raw_input())
speed=map(float,raw_input().split())
con=map(float,raw_input().split())
i=0
l=[]
mi=0
while i<n:
    l.append(round(con[i]/speed[i],3))
    if l[mi]>l[i]:
        mi=i
    i+=1
  
tot=float(raw_input())
dis=0.0
temp=int(tot/con[mi])
temp2=int(tot)%int(con[mi])
t2=(speed[mi]/con[mi])*temp2
dis+=(speed[mi]*temp)+t2
a=str(dis)+'0000'
i=a.find('.')+4
print a[:i]
