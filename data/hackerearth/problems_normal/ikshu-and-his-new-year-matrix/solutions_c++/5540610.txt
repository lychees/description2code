void sieve(int n)
{
    for(int i=2;i<=n;i++)prime[i]=1;
    for(int i=2;i<=sqrt(n);i++)
    {
        if(prime[i])
            for(int j=i*i;j<=n;j+=i)prime[j]=0;
    }
    for(int i=2;i<=n;++i)if(prime[i])p.push_back(i);

}
