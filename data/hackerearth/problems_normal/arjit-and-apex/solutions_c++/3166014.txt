#include <iostream>
#include <utility>
#include <algorithm>

using namespace std;

pair<int,int> m_array[100000];
pair<int,int> n_array[100000];

int main()
{
    int t;
    scanf("%d", &t);
    while(t--){
    	long m, n, g, h, i, j;
    	scanf("%ld %ld", &m, &n);
    	for(i = 0;i < m;i++)
    		scanf("%d %d", &m_array[i]);
    	for(i = 0;i < n;i++)
    		scanf("%d %d", &n_array[i]);	
    	scanf("%ld %ld", &g, &h);
    	sort(m_array,m_array + m);
    	sort(n_array,n_array + n);
    	i = 0; j = 0;
    	while(i < n && j < m){
    		if(n_array[i].first == m_array[j].first){
    			g--;
    			if(n_array[i].second == m_array[j].second)
    				h--;
    			i++; j++;
    		}
    		else if(n_array[i].first < m_array[j].first)
    			i++;
    		else
    			j++;
    	}
    	if(g <= 0){
    		if(h <= 0)
    			printf("Great\n");
    		else
    			printf("Good\n");
    	}
    	else
    		printf(":(\n");
    }
    return 0;
}
