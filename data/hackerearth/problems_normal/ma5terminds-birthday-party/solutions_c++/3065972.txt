using namespace std;
int prime[1000005];
void sieve()
{
	prime[1]=1;
    for(int i=2;i<1000001;i++)
    {
    	if(!prime[i])
    	for(int j=i*2;j<1000001;j+=i)
    	prime[j]=1;
	}
	prime[4]=0;
	for(int i=2;i<1000001;i++)
	prime[i]+=prime[i-1];
}
int main()
{
	//ios_base::sync_with_stdio(false);
	//clock_t tStart = clock();
	sieve();
	int t,l,r;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&l,&r);
		printf("%d\n",prime[r]-prime[l-1]);
	}
	//cout<<fixed<<setprecision(10)<<(double)((clock() - tStart)/CLOCKS_PER_SEC);
    return 0;
}
