#include <iostream>
#include<vector>

#include<math.h>
using namespace std;

vector<int >primes;
const int maxi=10000;
void calPrimes(){
	int a[maxi];
	fill(a,a+maxi,1);
	for(int i=2;i<maxi;++i){
		if(a[i]==1){
			primes.push_back(i);
			int j=i+i;
			while(j<maxi){
				a[j]=0;
				j+=i;
			}
		}
	}
}

long long int solve(int n){
	calPrimes();
	int res=n;
	int ss=sqrt(n);
	for(int i=0;i<primes.size() && primes[i]<=ss;++i){
		if(n%primes[i] == 0 ) res-= res/primes[i];
		while(n%primes[i] == 0)n/=primes[i];
		
	}
	if(n>1)res-= res/n;
	return res;

}

int main()
{
	int n,t;
	cin>>t;
	while(t--){
		cin>>n;;
		cout<<solve(n)<<endl;
    
    }
    //cout << "Hello World!" << endl;
    return 0;
}
