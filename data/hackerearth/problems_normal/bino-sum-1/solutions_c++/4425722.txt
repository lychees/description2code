#include <iostream>
#define mo 1000000007
using namespace std;
typedef long long int nt;
nt a[1005][1005];
int main()
{   int t,i,j,n,p;
 nt sum=0;
    cin>>t;
     //for(i=0;i<=200;i++)
     //fac[i]=-1;
while(t--)
{sum=0;
	cin>>n>>p;
	for(i=1;i<=n;i++)
	a[i][0]=1;
	a[1][1]=1;
	for(i=2;i<=n;i++)
	for(j=1;j<=i;j++)
	{if(i==j)a[i][j]=1;
	else a[i][j]=(a[i-1][j-1]+a[i-1][j])%mo;
	}//sum=(sum+a[n][i])%mo;
	for(i=0;i<=p;i++)
	sum=(sum+a[n][i])%mo;
	cout<<sum<<endl;
}
    //out << "Hello World!" << endl;
    return 0;
}
