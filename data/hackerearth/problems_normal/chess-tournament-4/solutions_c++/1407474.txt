#include <iostream>
#include <algorithm>
#include<cmath>
using namespace std;

int cal(int *arr,int i,int j,int m,int n)
{
    
    j=j-2;
    i=i-1;
    if((*(arr+(j)*n+i))==1)
    {
    
     		return j+2;
    } else {
      	
      
        return i+1;
    }


    }
int main() {
    
    int N,m,n,tempM;
    cin>>N;
    m=n=pow(2,N);
    tempM=m=m-1;
    n=n-1;
    int a[m][n];
    
    for(int i=0;i<m;i++)
    {
        for(int j=0;j<n;j++)
        {
            if(i>=j)
                cin>>a[i][j];
            else
               a[i][j]=0;
        }
    }
     
    
    int res[m/2];
    int temp=0;int count=0;
    while(1)
    {    
        temp=0;
        
        for(int i=1;i<=tempM+1;i=i+2)
        {
            
            if(!count){
                
                res[temp]=cal((int *)a,i,i+1,m,n);
                temp++;
                
            } else {
                
                res[temp]=cal((int *)a,res[i-1],res[i],m,n);
                temp++;
            }
        }
       
        if(temp==1)
            break;
        tempM=temp-1;
        count=1;
        
        
    }
    
    cout<<res[0];
   
    return 0;
}
