#include<bits/stdc++.h>
using namespace std;
bool memo[100001];
int arr[100001];
void dp(int m,int n)
{
	sort(arr,arr+n);
	memo[0] = 1;
	for(int i=1;i<=m;i++)
	{
		if(memo[i]==0)
		{
			for(int j=0;j<n;j++)
			{
				if(arr[j]<=i && !memo[i-arr[j]])
				{
					memo[i] = 1;
					break;
				}
			}
		}
	}
	if(memo[m])
		cout<<"Little Deepu"<<endl;
	else
		cout<<"Kate"<<endl;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		memset(memo,0,sizeof(memo));
		int m,n;
		cin>>m>>n;
		for(int i=0;i<n;i++)
		{
			cin>>arr[i];
			memo[arr[i]] = 1;
		}
		dp(m,n);
	}
}
