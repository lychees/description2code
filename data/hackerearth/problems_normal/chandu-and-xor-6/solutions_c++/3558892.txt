#include <iostream>
using namespace std;
int mat[1001][1001];
int main()
{
    int n,m,i,j,row[1000],col[1000];
    cin>>n>>m;
    for(i=0;i<n;i++)
    {
    	for(j=0;j<m;j++)
    	cin>>mat[i][j];
    }
    int sum=0;
    for(i=0;i<n;i++)
    {
    	sum=0;
    	for(j=0;j<m;j++)
    	sum+=mat[i][j];
    	row[i]=sum;
    }
    for(i=0;i<m;i++)
    {
    	sum=0;
    	for(j=0;j<n;j++)
    	sum+=mat[j][i];
    	col[i]=sum;
    }
    int max=0;
    int temp;
    for(i=0;i<n;i++)
    {
    	for(j=0;j<m;j++)
    	{
    		temp=row[i]^col[j];
    		if(temp>max)
    		max=temp;
    	}
    }
    cout<<max<<endl;
    
}
