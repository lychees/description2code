from string import rjust


def drawZ(n):
    s = ''
    for i in range(n):
        s = s+'*'
    print(s),


def printZ(n):
    for i in range(n):
        if i in [0, n-1]:
            drawZ(n)
            print('')
        else:
            print '*'.rjust(n-i)


n = input()
printZ(n)
