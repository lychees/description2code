#include <iostream>
using namespace std;

int main(){
long int a,b,i,k,t,arr[10000],sum;
cin >> t;
while(t--){
	k=0;
	cin >> a >> b;
	for(i=0;i<a;i++)
		cin >> arr[i];
	for(i=0;i<a;i++){
		if(arr[i]==b){
			k=1;break;
		}
	}
	if(k==1){
		sum=0;
		cout << "YES ";
		for(i=0;i<a;i++){
			if(arr[i]>b){
				sum++;
			}
		}
		cout << sum << endl;
	}
	else {
		cout << "NO 0" << endl; 
	}
}

return 0;
}
