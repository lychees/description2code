#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		char s[100001];
		cin>>s;
		long int totalCount=0,count=0;
		bool check=true;
		for(int i=0;s[i]!='\0';i++){
			if(s[i]=='O' && check){
				count++;
			}
			else if(s[i]=='X'){
				count=0;
				check=false;
			}
			else if(s[i]=='*'){
				totalCount+=count;
				count=0;
				check=true;
			}
		}
		cout<<totalCount+count<<endl;
	}
    return 0;
}
