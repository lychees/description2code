#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int i=n,j=0,flag=0;
    	while(i%8!=0 || j%6!=0)
    	{
    		if(j==n)
    		{
    			flag=1;
    			break;
    		}
    		else
    		{
    			i--;
    			j++;
    		}
    	}
    	//cout<<i<<" "<<j<<endl;
    	if(flag==1)
    	{
    		cout<<"-1"<<endl;
    	}
    	else
    	{
    		int temp=((i/8)+(j/6));
    		cout<<temp<<endl;
    	}
    }
}
