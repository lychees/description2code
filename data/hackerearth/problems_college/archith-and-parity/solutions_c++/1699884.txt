#include <iostream>
using namespace std;

int countBits(int N)
{
	if(N==0)
		return 0;
	if(N==1)
		return 1;
	if(N%2==0)
		return countBits(N/2);
	else
		return 1+countBits(N/2);
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int N;
		cin>>N;
		int m=countBits(N);
		if(m%2==0)
			cout<<"even"<<endl;
		else
			cout<<"odd"<<endl;
	}
	return 0;
}
