#include <iostream>
using namespace std;

int main()
{
	int n,x,sum=0;
    cin>>n;
    int flag=1;
    for(int i=0;i<n;i++){
    	cin>>x;
    	if(x==0)
    		flag=0;
    	sum+=x;
    }
    if(flag==0)
    	cout<<"No";
    else if(sum%2==1)
    	cout<<"No";
    else if(sum/2!=n-1)
    	cout<<"No";
    else cout<<"Yes";
    return 0;
}
