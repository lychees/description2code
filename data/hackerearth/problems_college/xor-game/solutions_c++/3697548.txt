#include <iostream>
using namespace std;

int main()
{
   int a,b,c=0,k=0;
   cin>>a;
   cin>>b;
   for(int i=a;i<=b;i++)
   {
   	for(int j=i;j<=b;j++)
   	{
   		int g=i^j;
   	   if(g>c)
   	   {
   	   	c=g;
   	   	//cout<<c<<endl;
   	   	
   	   k++;
   	   }
   	}
   }
   cout<<c<<endl;
    return 0;
}
