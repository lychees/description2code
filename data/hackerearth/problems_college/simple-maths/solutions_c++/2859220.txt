#include<bits/stdc++.h>
using namespace std;
long long int gcd(long long int a, long long int b)
{
	    // base case
	    if (a == b)
		            return a;
	     
	        // a is greater
	        if (a > b)
			        return gcd(a-b, b);
		    return gcd(a, b-a);
}
long long int lcm(long long int a, long long int b)
{
	    return (a*b)/gcd(a, b);
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	long long n,a,b;
	cin>>n>>a>>b;
	if(a==0 and b==0)
		cout<<"-1"<<endl;
	else if(a==0)
	{
		cout<<"-1"<<endl;
	}
	else if(b==0)
		cout<<"-1"<<endl;
	else
	{//cout<<"RR";
	long long ans=n/a+n/b-n/lcm(a,b);
	cout<<ans<<endl;
	}
	}
	return 0;
}
