#include <bits/stdc++.h>
using namespace std;

int main()
{
	
    long long n,k,a[100001],i,j,temp,max,c=0;
    cin>>n>>k;
    
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    	
    }
    
if(k>=n)
{
for(i=n;i>0;--i)
cout<<i<<" ";
}
else
    {for(i=0;i<n-1;i++)
    {
    	max=i;
    	for(j=i+1;j<n;j++)
    	{
    		if(a[j]>a[max])
    		max=j;
    	}
    	
    	if(i!=max)
    	{
    		//swap
    		temp=a[i];
    		a[i]=a[max];
    		a[max]=temp;
    		c++;
    		if(c==k)
    		break;
    		
    	}
    	
    }
    for(i=0;i<n;i++)
    cout<<a[i]<<" ";
    cout<<"\n";}
    return 0;
}
