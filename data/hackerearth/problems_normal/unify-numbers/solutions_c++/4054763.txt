#include <stdio.h>
#include <algorithm>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	long int a[100001],dif,j=0;
    	long long int sum=0,avg;
    	scanf("%d",&n);
    	
    	for(int i=0;i<n;i++)
    		scanf("%ld",&a[i]);
    	
        sort(a, a+n);
     
     avg = a[n/2];
    		
    	for(int i=0;i<n;i++)
    	{
    		dif = avg - a[i];
    		if(dif<0)
    		dif = dif*(-1);
    		sum+=dif;
    	}
    		
    	printf("%lld\n",sum);
    }
    return 0;
}
