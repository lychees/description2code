#include <iostream>

using namespace std;

int main()
{
    int n;

    cin>>n;

    int v[n];

    for(int i=0;i<n;i++)
    {
        cin>>v[i];
    }

    int sum=0,k=0,t[n];

    for(int i=0;i<n;i++)
    {
        if(v[i]!=0)
        {
            sum+=v[i];
            t[k++]=v[i];
        }else
        {
            sum-=t[k-1];
            k--;
        }
    }

    cout<<sum;

    return 0;
}
