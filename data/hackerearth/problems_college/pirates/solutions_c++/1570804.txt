#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
   int n,a[100011],i,ans=0,l=1;
   cin>>n;
   for(i=0;i<n;i++)
   cin>>a[i];
   for(i=1;i<n;i++)
   {
   	
   	if(abs(a[i]-a[i-1])==1)
   	l++;
   	else
   	l=1;
   	if(l>=ans)
   	{
   	  ans=l;
   	}
   	
   }
   	if(l>=ans)
   	ans=l;
   if(ans==1)
   cout<<"0\n";
   else
   cout<<ans<<'\n';
   return 0;
}
