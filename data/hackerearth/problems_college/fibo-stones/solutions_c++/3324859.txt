#include<iostream>
using namespace std;
int main()
{
    int n,m;
    cin>>n>>m;
    if(n>=3&&n<=80&&m>=1&&m<=n)
    {
    long int a[n+1],b[m+1];
    long long int sum=0,sum2=0,sum3;
    for(int j=1;j<m+1;j++)
    {
        cin>>b[j];
        sum2=sum2+b[j];
    }
    for(int i=1;i<n+1;i++)
    {
        if(i==1)
            a[i]=1;
        else if(i==2)
            a[i]=1;
        else
            a[i]=a[i-1]+a[i-2];
        sum=sum+a[i];
    }
   if(sum>=sum2)
    {
    sum3=sum-sum2;
    cout<<sum3;
    }
    if(sum2>sum)
    {
    sum3=sum2-sum;
    cout<<sum3;
    }

    }
    return 0;
}
