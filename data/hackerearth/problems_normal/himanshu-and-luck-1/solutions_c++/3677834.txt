#include<iostream>

#include<algorithm>
#include<string>
#include<vector>

using namespace std;
#define big_num 1000000000000000000
vector<long long int>def;
void luck(long long int n)
{
	if(n>big_num)return ;
	long long int temp = n;
	temp = temp * 10;
	temp = temp + 4;
	def.push_back(temp);
	luck(temp);
	temp = temp + 3;
	def.push_back(temp);
	luck(temp);
}
int main()
{
	int i;
    luck(0L);
	sort(def.begin(),def.end());
	int j;
	long long x,y;
	vector<long long int>::iterator c,d;
	cin>>j;
	while(j--)
	{
		cin>>x>>y;
		c = lower_bound(def.begin(),def.end(),x);
		d = upper_bound(def.begin(),def.end(),y);
		x = c - def.begin();
		y = d - def.begin();
		x = abs(x-y);
		cout<<x<<endl;
	}
	return 0;
}
