t=int(raw_input())
while (t>0):
	str=raw_input()
	l=len(str)
	count=ans=0
	for i in range(0,l):
		if(str[i]=='a' or str[i]=='A' or str[i]=='b' or str[i]=='B' or str[i]=='c' or str[i]=='C'):
			count+=1
		else:
			ans+=(count*(count+1))/2
			count=0
	ans+=(count*(count+1))/2
	t-=1
	ans=int(ans)
	print (ans)
