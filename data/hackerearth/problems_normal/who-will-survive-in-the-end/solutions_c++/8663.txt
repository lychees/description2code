#include <iostream>
#include <queue>
using namespace std;

int log(int num)
{
  int count = 0;
  while (num != 1)
    {
      num = num/2;
      ++count;
    }
  return count;
}
int getNum(int num)
{
  queue<int>circle;
  for(int i = 1; i <= num; i+=2)
    {
      circle.push(i);
    }
  while(circle.size() > 1)
    //int end = circle.size();
    //for(int i = 0; i < end; ++i)
    {
      circle.pop();
      circle.push(circle.front());
      circle.pop();
    }
  return circle.front();
}
int power(int number, int toPower)
{
  int power = number;
  while(toPower > 1)
    {
      power *= 2;
      --toPower;
    }
  return power;
}
int main()
{
  int num = 0;
  cin >> num;
  int current = 0;
  for(int i = 0; i < num; ++i)
    {
      cin >> current;
      int base = power(2, log(current));
      int difference = current - base;
      cout << difference*2+1 << endl;
    }
  return 0;
}
