#include <iostream>
using namespace std;

int main()
{
    long t,n,count;
    cin>>t;
    while(t--)
    {	count=0;
    	cin>>n;
    	const long int N=n;
    	long arr[N];
    	for(int i=0;i<N;i++)
    	{
    		cin>>arr[i];
    		if(arr[i]%2==0)count++;
    	}
    cout<<count<<endl;
    }
    
    return 0;
}
