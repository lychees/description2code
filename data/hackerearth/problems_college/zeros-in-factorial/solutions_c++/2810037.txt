#include <iostream>
#include <vector>
using namespace std;
typedef unsigned long long Int;
const int MAX = 1000000;
vector<Int> table;
inline Int min(Int a, Int b) { return a < b ? a : b; }

void preprocess() {
	Int fullSum = 0;
	Int twos = 0, fives = 0;
	table.assign(MAX, 0);
	for (int i = 1; i <= MAX; ++i) {
		int tmp = i;
		while (tmp%2 == 0) {twos++; tmp/=2;}
		while (tmp%5 == 0) {fives++; tmp/=5;}
		fullSum += min(twos, fives);
		table[i] = fullSum;
	}
}

int main() {
	int TC; cin >> TC;
	preprocess();
	while (TC--) {
		int a, b; cin >> a >> b;
		cout << table[b] - table[a-1] << endl;
	}
}
