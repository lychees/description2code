#include<bits/stdc++.h>
using namespace std;
long long temp[1000006];
int main(){
	long long n;
	scanf("%lld",&n);
	long long a[n+5];
	memset(a,0,sizeof(a));
	long long count[n+2];
	memset(count,0,sizeof(count));
	long long m;
	scanf("%lld",&m);
	for(int i=0;i<m;i++){
		int l,r;
		scanf("%d%d",&l,&r);
		count[l]++;
		count[r+1]--;
	}
	long long max=-1;
	for(int i=1;i<=n;i++){
		a[i]=a[i-1]+count[i];
		if(a[i]>max)
		max=a[i];
	}
	memset(temp,0,sizeof(temp));
	for(int i=0;i<=n;i++)
	temp[a[i]]++;
	for(long long i=max;i>=0;i--){
		temp[i]=temp[i]+temp[i+1];
	}
	long long q;
	scanf("%lld",&q);
	while(q--){
		long long t;
		scanf("%lld",&t);
		printf("%lld\n",temp[t]);
	}
}
