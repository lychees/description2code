//vector<int> g[2*100000+5];int visited[2*100000+5];
struct s
{
	int price,tasks;
}data[2005];
lld pre[2005],dp[2005];
bool way(s a,s b)
{
	return a.price<b.price;
}
int main()
{
	int n,t,k;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		pre[0]=0;
		for(int i=1;i<=n;++i)
		{
			cin>>data[i].price;
		}
		for(int i=1;i<=n;++i)
		{
			cin>>data[i].tasks;
		}
		sort(data+1,data+n+1,way);
		for(int i=1;i<=n;++i)
		{
			pre[i]=pre[i-1]+data[i].tasks;
		}
		for(int i=1;i<=n;++i)
		{
			dp[i]=100000000000000;
			for(int j=0;j<i;++j)
			{
				dp[i]=min(dp[i],dp[j]+(pre[i]-pre[j]+k)*data[i].price);
			}
		}
		cout<<dp[n]<<endl;
	}
}
