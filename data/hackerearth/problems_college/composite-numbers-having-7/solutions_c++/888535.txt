

#include<cstdio>
#include<cmath>
#include<iostream>
#define Max 1000000
using namespace std;
long int x[Max+7];
void sieve(void)
{
    int i,j,s;
    x[0]=x[1]=1;
    for(i=4; i<=Max; i+=2)
    {
        x[i]=1;
    }
    //s=sqrt(Max);
    for(i=3; i*i<=Max; i+=2)
    {
        if(x[i]==0)
        {
            for(j=i*i; j<=Max; j+=i+i)
            {
                x[j]=1;
            }
        }
    }
}
bool Find(long int n)
{
    while(n)
    {
        if(n%10==7)
            return true;
        n=n/10;
    }
    return false;
}
int main()
{
    long int t,cnt,m,n,i;
    sieve();
    cin>>t;
    while(t--)
    {
        scanf("%ld %ld",&m,&n);
        cnt=0;
        for(i=m; i<=n; i++)
        {
            if(x[i]==1)
            {
                if(Find(i))
                    cnt++;
            }
        }
        if(cnt==0)
            cout<<"-1\n";
        else
            cout<<cnt<<endl;
        //if(t)
        // printf("\n");
    }
    return 0;
}
