#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		vector<int> v(2*n);
		for(int i=0;i<2*n;i++) cin>>v[i];
		sort(v.begin(),v.end());
		//for(int i=0;i<n;i++) cout<<v[i]<<" ";
		int ms,i=0,j=2*n-1;
		ms=v[i]+v[j];
		
		i++,j--;
		while(i<j)
		{
			
			if(v[i]+v[j]>ms)
			ms=v[i]+v[j];
			i++; j--;
		}
		cout<<ms<<"\n";
	}
}
