using namespace std;
ll f[1000007],inv[1000007],mod=1000000007;
void euclid(ll a,ll b,ll& x,ll& y)
{
	if(a%b==0)
	{
		x=0;
		y=1;
		return;
	}
	euclid(b,a%b,x,y);
	ll temp=x;
	x=y;
	y=temp-(a/b)*y;
}
ll inverse(ll a,ll b)
{
	ll x,y;
	euclid(a,b,x,y);
	if(x<0)x+=b;
	return x;
}
ll compute(ll a,ll b)
{
	ll res;
	if(a<b || a==0)
	return 0;
	if(b==0)
	return 1;
	res=(f[a]*inv[b])%mod;
	res=(res*inv[a-b])%mod;
	return res;
}
int main()
{
	ll t,i,maxi=1000000,c1,c2,l,res,fr[27],p,q;
	string s;
	scanf("%lld",&t);
	f[0]=f[1]=inv[0]=inv[1]=1;
	for(i=2;i<=maxi;i++)
	{
	f[i]=(i*f[i-1])%mod;
	inv[i]=(inverse(f[i],mod))%mod;
	}
	while(t--)
	{
		cin>>s;
		l=s.length();
		c1=c2=0;
		memset(fr,0,sizeof(fr));
		for(i=0;i<l;i++)
		{
			if(s[i]=='a' || s[i]=='e' || s[i]=='o' || s[i]=='i' || s[i]=='u')
			c1++;
			else
			c2++;
			fr[s[i]-97]++;
		}
		if(c2+1<c1)
		printf("-1\n");
		else
		{
		q=compute(c2+1,c1);
		p=(f[c1]*f[c2])%mod;
		for(i=0;i<26;i++)
		{
			if(fr[i]>0)
			{
				p=(p*inv[fr[i]])%mod;
			}
		}
		res=(p*q)%mod;
		printf("%lld\n",res);
		}
	}
	return 0;
}
