#print 'Hello World!'
arr = []
no = int(raw_input())
for i in range(no):
	strInput = raw_input()
	firstChar = strInput[0]
	lastChar = strInput[-1]
	sortedStrInput = "".join(sorted(strInput))
	elem = [sortedStrInput, firstChar, lastChar]
	if elem not in arr:
		arr.append(elem)
		
print len(arr)
