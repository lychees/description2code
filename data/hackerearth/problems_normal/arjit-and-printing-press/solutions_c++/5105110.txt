#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
 
int main()
{
    int t;
    cin >> t;
    for(int tt=0; tt<t; tt++)
    {
    	char *s,*r;
     	s = (char*)malloc(sizeof(char)*10000);
     	r = (char*)malloc(sizeof(char)*10000);
     	cin >> s >> r;
     	sort(r,r+strlen(r));
     	for(int i=0,j=0;s[i]&&r[j];i++)
     	{
      		if(r[j]<s[i])
       		s[i] = r[j++];
     	}
     	cout<<s<<endl;
    }
}
