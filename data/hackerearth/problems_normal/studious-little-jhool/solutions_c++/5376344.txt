#include <bits/stdc++.h>
using namespace std;

int main()
{
    int a[114];
    for(int i=0; i<12; i++)
    	a[i]=0;
    a[10]=a[12]=1;
    for(int i=13; i<=113; i++)
    {
    	a[i]=0;
    	int x=0, y=0;
    	if(a[i-10]!=0)
    		x=a[i-10]+1;
    	if(a[i-12]!=0)
    		y=a[i-12]+1;
    	if(x==0)
    		a[i]=y;
    	else if(y==0)
    		a[i]=x;
    	else
    		a[i]=min(x, y);
    }
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	if(a[n]==0)
    		cout<<"-1\n";
    	else
    		cout<<a[n]<<"\n";
    }
    return 0;
}
