#include<iostream>
#include<stdio.h>
using namespace   std;
int hasher[1029];
long long int fact[1029];
long long int mod=1000000007;
void pre()
{
	fact[0]=1;
	  for(int i=1;i<=1029;i++)
	  {
	  	   fact[i]=(i*fact[i-1])%mod;
	  }
}
int main()
{
	pre();
	int n,m;
	cin>>n>>m;
	for(int i=1;i<=n;i++)
	{
		  for(int j=1;j<=m;j++)
		  {
		  	  hasher[i^j]++;
		  	 // if(i^j)
		  }
	}
	long long int ans=1;
	
	for(int i=0;i<=1024;i++)
	{
		 if(hasher[i])
		 ans=(ans*fact[hasher[i]])%mod;
	}
	cout<<ans<<endl;
	return 0;
}
