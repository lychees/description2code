def sieve():
    for i in xrange(3, 1001, 2):
        if not A[i]:
            for j in xrange(i*i, 1000001, 2*i):
                if not A[j]:
                    A[j] = i


A = [0]*1000001
sieve()
for _ in range(input()):
    n = input()
    if n % 2 == 0:
        print n-2
    else:
        if A[n] == 0:
            print 0
        else:
            print n-A[n]
