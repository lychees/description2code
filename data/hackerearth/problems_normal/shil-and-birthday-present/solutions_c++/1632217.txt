#include <iostream>
#include <vector>
#include <unordered_map>

#define NC2(x) ((x)*( (x)-1 ))/2

using namespace std;

int main() {
	
	unsigned long long N;
	unsigned long long no_of_unique_elem = 0;
	scanf("%llu", &N);
	
	std::unordered_map<long long, bool> map;
	
	while( N > 0 )
	{
		N--;
		long long elem;
		
		scanf("%lld", &elem);
		
		if( map.find( elem )==map.end() )
		{
			map[elem] = true;
			no_of_unique_elem++;
		}
		
	}
	
	unsigned long long result = NC2( no_of_unique_elem );
	
	printf("%llu\n", result);
	
	return 0;
}
