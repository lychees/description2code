#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long int n,i,j,tmp,pos;
    scanf("%ld",&n);
    long int a[n];
    for(i=0;i<n;i++)
    {
    	scanf("%ld",&a[i]);
    }
    for(i=0;i<n-1;i++)
    {
    	if(a[i]>0)
    	{
    		a[i]-=1;
    		a[i+1]+=1;
    		tmp=a[i];
    		a[i]=a[i+1];
    		a[i+1]=tmp;
    	}
    }
    int flag=1;
    for(i=0;i<n;i++)
    {
    	for(j=i+1;j<n;j++)
    	{
    		if(a[i]>a[j])
    		{
    			flag=0;
    			break;
    		}
    	}
    	if(flag==0)
    	break;
    }
    if(flag==0)
    printf(":(");
    else
    {
    	for(i=0;i<n;i++)
    	printf("%d ",a[i]);
    }
    
    
    return 0;
}
