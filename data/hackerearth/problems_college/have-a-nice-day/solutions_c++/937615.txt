# include <iostream>
# include <stdio.h>
# include <algorithm>
# include <vector>
# define MAX 1000
using namespace std;

vector <int> male(MAX);
vector <int> female(MAX);
int main()
{
    int t,N;
    scanf("%d",&t);
    while(t--)
    {
        int c=0;
       scanf("%d",&N);
       for(int i=0;i<N;i++)
       {
           cin>>male[i];
       }
       for(int i=0;i<N;i++)
       {
           cin>>female[i];
       }
       sort(male.begin(),male.begin()+N);
       sort(female.begin(),female.begin()+N);
       for(int i=0;i<N;i++)
       {
           c+=male[i]*female[i];
       }
       cout<<c<<endl;
    }

    return 0;
}
