#include <iostream>
#include <string>
#include<algorithm>
using namespace std;

int main()
{
	int testcase;
	cin >> testcase;

	while (testcase--)
	{
		int n;
		cin >> n;
		char myMatrix;

		int row=0, column=0, i, j;

		for (i = 0; i<n; i++)
		{
			for (j = 0; j<n; j++)
			{
				cin >> myMatrix;

				if (myMatrix == 42)
				{
					row = i;
					column = j;
				}
			}
		}

		int result = abs(ceil(n/2)-row)+abs(ceil(n/2)-column);
		cout << result << endl;
	}

	return 0;
}
