#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll logf(ll a)
{
    ll r=0;
    while(a>1)
    {
        r++;
        a/=2;
    }
    return r;
}
ll pow(ll a,ll b)
{
    ll r=1;
    while(b>0)
    {
        if(b&1)
            r*=a;
        a*=a;
        b/=2;
    }
    return r;
}
int main()
{
    ll t,n;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&n);
        ll d=logf(n);
        ll x=pow(2,d);
        printf("%lld\n",2*(n-x)+1);
    }
}
