k = int(raw_input())
s = raw_input()
l = len(s)


def swap(s):
    j = l - 1
    while j:
        if not j % 2 == 0:
            e = s[j]
            s = s[:j] + s[j + 1:] + e
        j -= 1
    return s


c = s
count = 0
itr = ""
while itr != s:
    itr = swap(c)
    c = itr
    count += 1

count = k % count

for i in range(count):
    s = swap(s)

print s
