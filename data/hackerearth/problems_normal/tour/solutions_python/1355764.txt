ans = 0
n = input()
arr = [raw_input() for _ in range(n)]
ar = [[int(_) for _ in raw_input().split()] for i in range(n)]
prev=0
p = input()
for i in range(p):
    temp = raw_input()
    for j in range(n):
        if(temp==arr[j]):
            ans+=(ar[j][prev])
            prev=j
            break
print ans
