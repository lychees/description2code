#include <iostream>
using namespace std;

int main()
{
  long long int t,r1,r2,x,y;
   int temp;
   cin>>t;
   while(t--)
   {
   	temp=0;
   	cin>>x>>y;
   	while(x>0 && y>0)
   	{
   		r1=x%10;
   		r2=y%10;
   		if((r1+r2)>9)
   		temp++;
   		x=x/10;
   		y=y/10;
   	}
   	if(temp==0)
   	cout<<"No carry operation"<<endl;
   	else if(temp==1)
   		cout<<"1 carry operation"<<endl;
   		else
   			cout<<temp<<" carry operations"<<endl;
   }
}
