#include <bits/stdc++.h>
using namespace std;
int dp[155][155][155];
int visit[155][155][155];
int tot,c;
int fun(int coke,int n5,int n10)
{
	if(coke==c)
	return 0;
	if(visit[coke][n5][n10]!=0)
	return dp[coke][n5][n10];
	visit[coke][n5][n10]=1;
	int res=1e9;
	int n1=tot-8*coke;
	n1=n1-(5*n5+10*n10);
	if(n1>=8)
	res=min(res,8+fun(coke+1,n5,n10));
	if(n5>=1&&n1>=3)
	res=min(res,4+fun(coke+1,n5-1,n10));
	if(n10>=1)
	res=min(res,1+fun(coke+1,n5,n10-1));
	if(n5>=2)
	res=min(res,2+fun(coke+1,n5-2,n10));
	if(n10>=1&&n1>=3)
	res=min(res,4+fun(coke+1,n5+1,n10-1));
	dp[coke][n5][n10]=res;
	return res;
}
int main()
{
	int t,n1,n5,n10;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d%d%d",&c,&n1,&n5,&n10);
		tot=0;
		tot=n1+5*n5+10*n10;
		if(8*c>tot)
		{
			printf("-1\n");
			continue;
		}
		else
		{
			memset(visit,0,sizeof visit);
			int ans=fun(0,n5,n10);
			cout<<ans<<"\n";
		}
		
	}
	return 0;
}
