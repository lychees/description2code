#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int n,i,flag=0,j;
    cin>>n;
    for(i=4;i<=n;i++)
    { 
    	for(j=2;j<i;j++)
    	{
          	if(i%j==0)
    	      { flag=1;
    		     break;
            	}
    	}
    	if(flag==1)
    	cout<<i<<endl;
    	flag=0;
    	
    }
    return 0;
}
