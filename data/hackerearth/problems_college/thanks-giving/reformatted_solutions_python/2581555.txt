a = raw_input()
count = [0]*26
minn = ord(a[0])
for i in range(len(a)):
    count[ord(a[i])-97] += 1

freq = 0
for i in range(26):
    if count[i] > freq:
        freq = count[i]
        minn = i+97


if chr(minn) in "headrush" and chr(minn) in "cubing club":
    print "Thanks both"
    exit(0)

if chr(minn) in "head rush":
    print "Thanks head rush"
    exit(0)

if chr(minn) in "cubing club":
    print "Thanks cubing club"
    exit(0)

else:
    print "none"
