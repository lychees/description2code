def main():
    import sys
    input()
    inp = sys.stdin.read().split()
    for i in inp:
        if i[0]!='R':
            print "NO"
            continue
        c = 0
        ans = "YES"
        for j in i:
            if j=='R': c=0
            elif j=='Y': c+=1
            else:
                ans = "NO"
                break
            if c==3:
                ans = "NO"
                break
        print ans
main()
