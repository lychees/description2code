#include <iostream>
using namespace std;

int main()
{
    int n,k,i,firstnonzero;
    char ch;
    cin>>n;
    cin>>ch;
    string s;
    while(n--){
    	getline(cin,s);
    	k = 0;
    	cout<<"$";
    	firstnonzero = 0;
    	for(i=0;i<s.length();i++){
    		if(s[i] == '$'){
    			i++;
    			while(s[i] == ' ' || s[i] == '0'){
    				i++;
    			}
    			while(s[i] == ' ' || (s[i] >= '0' && s[i] <= '9')){
    				firstnonzero = 1;
    				if(s[i] == ' '){
    					i++;
    					continue;
    				}
    				cout<<s[i++];
    			}
    			break;
    		}
    	}
    	if(!firstnonzero) cout<<'0';
    	cout<<endl;
    }
    return 0;
}
