#print 'Hello World!'
s = raw_input()
n=''
i=0
if 'a'<=s[0] and 'z'>=s[0]:
	n+=chr(ord(s[0])-32)
else:
	n+=s[0]
i+=1
while i < len(s):
	if s[i]==' ' and i+1<len(s) and 'a'<=s[i+1] and 'z'>=s[i+1]:
		n=n+s[i]+ chr(ord(s[i+1])-32)
		i+=2
	else:
		n+=s[i]
		i+=1
		
print n
