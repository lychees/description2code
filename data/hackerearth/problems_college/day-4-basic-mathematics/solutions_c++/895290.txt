#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
  //  cout << "Hello World!" << endl;
  	int test;
  	cin>>test;
  	for(int i=0;i<test;i++)
  	{
  		int n;
  		cin>>n;
  		int c=0,d=1;
  		while(n>c)
  		{
  			c=d*(d+1)/2;
  			d++;
  		}
  		d--;
  		if(d%3==0)
  		cout<<"SL"<<endl;
  		else if(d%3==1)
  		cout<<"LB"<<endl;
  		else if(d%3==2)
  		cout<<"BS"<<endl;
  	}
    return 0;
}
