#include <bits/stdc++.h>
 
using namespace std;
 
#define sz(a) int((a).size()) 
#define sl(a) int((a).length()) 
#define pb push_back 
#define all(c) (c).begin(),(c).end() 
#define rall(c) (c).rbegin(),(c).rend() 
#define tr(con,it) for (__typeof((con).begin()) it = con.begin(); it != con.end(); it++) 
#define wl(x) (cout << (x) << endl)
#define w(x) (cout << (x))
#define r(x) (cin >> (x))
#define FOR(i,a,b) for (int i = (a); i < (b); i++)
#define ROF(i,a,b) for (int i = (b-1); i >= (a); i--)
#define PP(x) tr(x,it) cout << it->first << " " << it->second << endl
#define mp(x,y) make_pair((x),(y))
#define reset(x) memset((x),0,sizeof((x)))
#define log(x) cout << #x << " = " << x << endl
#define log2(x,y) cout << #x << " = " << x << " " << #y << " = " << y << endl
#define log3(x,y,z) cout << #x << " = " << x << " " << #y << " = " << y << " " << #z << " = " << z << endl
#define MAXN 100010
#define MDL 1000000007
 
int main() {
int T;
//r(T);
T=1;
 
 int m,n,temp,lb;
 long long s,q,z;
 
 FOR(tt,0,T)
 {
   r(n);r(m);
   vector < long long > x(n),y(n),inc;
   FOR(i,0,n) { r(x[i]); if (i>0) y[i]=y[i-1]+x[i]; else y[i]=x[i]; }
   s=max(*max_element(all(x)),y[n-1]/m);
   while(true)
   {
     q=0;
     temp=0;
     lb=0;
     inc.clear();
     while(true)
     {
       temp++;
       z=q+s;
       if (z==y[n-1]) break;
       lb=lower_bound(y.begin()+lb,y.end(),z)-y.begin();
       if (lb==n) break;
       if (z!=y[lb]) inc.pb(y[lb]-z);
       if (y[lb]==z) q=y[lb];
       else q=y[lb-1];
     }
     sort(all(inc));
     if (temp<=m) break;
     s+=inc[temp-m-1];
   }
   wl(s);
 
   cerr << "case " << (tt+1) << " " << clock() << " ms" << endl;
 }
 
 return 0;
}
