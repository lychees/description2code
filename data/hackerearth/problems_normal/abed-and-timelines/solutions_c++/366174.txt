using namespace std;
#include<iostream>
int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
        int n,m;
        cin>>n; cin>>m;
        int x1,y1,x2,y2,p;
        cin>>x1; cin>>y1; cin>>x2; cin>>y2; cin>>p;
        long long x,y;
        if(x1>x2)
            x=x1-x2;
        else x=x2-x1;
        x=min(x,n-x);
        if(y1>y2)
            y=y1-y2;
        else y=y2-y1;
        y=min(y,m-y);
        long long sum=(x*p)+(y*p);
        cout<<sum<<endl;
        if(sum>1000)
            cout<<"Let go of the group."<<endl;
        else cout<<"You saved the group."<<endl;
        t--;
    }
}
