#include <bits/stdc++.h>
using namespace std;
int ans = INT_MAX;
int arr[100];
int n;

int max(int a,int b)
{
	return a>b?a:b;
}

void sol(int s1,int s2,int s3,int ind)
{
	if(ind==n)
	{
		int m = max(s1,max(s2,s3));
		ans = ans>m?m:ans;
		return;
	}
	sol(s1+arr[ind],s2,s3,ind+1);
	sol(s1,s2+arr[ind],s3,ind+1);
	sol(s1,s2,s3+arr[ind],ind+1);
}

int main()
{
	cin>>n;
	
	for(int i=0;i<n;i++)
	{
		cin>>arr[i];
	}
	sol(0,0,0,0);
	cout<<ans<<endl;
    return 0;
}
