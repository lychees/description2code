#include <iostream>
using namespace std;
#define LIMIT 510
#define MOD 1000000007
long long arr[LIMIT][LIMIT],n,h,ans[LIMIT][LIMIT];

int main() {
	for(h=1; h<LIMIT; h++) {
		arr[0][h]=1;
		arr[1][h]=1;
	}
	for(int n=1; n<LIMIT; n++) {
		arr[n][0]=0;
	}
	
	for(int n=2; n<LIMIT; n++) {
		for(int h=2;h<LIMIT;h++) {
			if(h>n) {
				arr[n][h]=arr[n][h-1];
				continue;
			}
			arr[n][h]=0;
			for(int i=1;i<=n;i++) {
				long long temp = (arr[i-1][h-1]*arr[n-i][h-1])%MOD;
				arr[n][h] = (arr[n][h]+temp)%MOD;
			}
		}
	}
	
	for(int n=2; n<LIMIT; n++) {
		for(int h=2;h<LIMIT;h++) {
			if(h>n) {
				ans[n][h]=ans[n][h-1];
				continue;
			}
			ans[n][h]=0;
			for(int i=1;i<=n;i++) {
				long long temp = (arr[i-1][h-1]*arr[n-i][h-1])%MOD;
				temp = temp*i;
				ans[n][h] = (ans[n][h]+temp)%MOD;
			}
		}
	}
cin>>n>>h;
cout<<ans[n][h]%MOD<<"\n";
return 0;
}
