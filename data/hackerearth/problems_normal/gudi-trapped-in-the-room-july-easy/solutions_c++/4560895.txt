#include<bits/stdc++.h>
using namespace std;
set<string> myset;
queue<string> q;
set<string>:: iterator it;
int main()
{
    ios::sync_with_stdio(false);
    int t,a,h,i,n;
    string str,temp,next;
    cin>>t;
    while(t--)
    {
        cin>>str>>a>>h;
        n=str.length();
        q.push(str);
        while(!q.empty())
        {
            temp=q.front();
            next=temp;
            q.pop();
            if(myset.find(temp)==myset.end()){
                myset.insert(temp);
            }
            for(i=0;i<n;i++)
            {
                next[(i+h)%n]=temp[i];
            }
            if(myset.find(next)==myset.end()){
                myset.insert(next);
                q.push(next);
            }
            for(i=0;i<n;i++)
            {
                if(i%2==1)
                    temp[i]=char((temp[i]-'0'+a)%10+'0');
            }
            if(myset.find(temp)==myset.end()){
                myset.insert(temp);
                q.push(temp);
            }
        }
        it=myset.begin();
        cout<<*it<<"\n";
        myset.clear();
    }
    return 0;
}
