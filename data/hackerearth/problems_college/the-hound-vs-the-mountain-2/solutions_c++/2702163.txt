#include <bits/stdc++.h>
using namespace std;

double convert(string a)
{
	double number = 0;
	int idx=0;
	while(idx < a.length())
	{
		number = number * 2 + a[idx] - '0';
		idx++;
	}
	return number;
}

int main()
{
    int t;  cin >> t;
    while(t--)
    {
    	string a, b;
    	cin >> a >> b;    	
   		double A = convert(a);
   		double B = convert(b);
   		if( B * log(A) <= A * log(B))
   			cout<<"The Hound\n";
   		else
   			cout<<"The Mountain\n";
    }
    
    return 0;
}
