#include <iostream>
using namespace std;

int main()
{
    long long int t,n,d,count=0;
    cin >> t;
    while(t > 0)
    {
    	cin >> n;
    	count=0;
    	while(n > 0)
    	{	
    	  d=n % 10;
    	  if(d == 4 || d == 7)
    	  count++;
    	  n=n/10;
    	}
    	if(count == 4 || count == 7)
    	cout << "YES" << endl;
    	else 
    	cout << "NO" << endl;
    	t--;
    }
    return 0;
}
