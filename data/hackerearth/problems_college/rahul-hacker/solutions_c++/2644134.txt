#include <iostream>
using namespace std;

int main()
{
   int num1,num2,n,rev1,rev2,a,b,sum,rev3,c;
    cin>>n;
    while(n--)
    {
        rev1=0;
        rev2=0;
        a=0;
        b=0;
        sum=0;
        rev3=0;
        c=0;
        cin>>num1;
        cin>>num2;
        while(num1!=0)
        {
            a=num1%10;
            rev1=(rev1*10)+a;
            num1=(num1-a)/10;
        }
        while(num2!=0)
        {
            b=num2%10;
            rev2=(rev2*10)+b;
            num2=(num2-b)/10;
        }
        sum=rev1+rev2;
        while(sum!=0)
        {
            c=sum%10;
            rev3=(rev3*10)+c;
            sum=(sum-c)/10;
        }
        cout<<rev3<<endl;
    }
}
