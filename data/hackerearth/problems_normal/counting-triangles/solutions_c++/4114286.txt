#include<bits/stdc++.h>
#define LL long long
using namespace std;

int main()
{
       int n;

       vector<LL>vc(3);
       map<vector<LL>,int>mp;
       map<vector<LL>,int>::iterator it;
       cin>>n;
       for(int i=0;i<n;i++)
       {
              cin>>vc[0]>>vc[1]>>vc[2];
              sort(vc.begin(),vc.end());
              mp[vc]++;


       }


       int count=0;


       for(   it=mp.begin();it!=mp.end();it++)
       {
             if(it->second==1)
             {
                  count++;
             }
       }

       cout<<count<<endl;

       return 0;


}
