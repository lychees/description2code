#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <iterator>
using namespace std;

int main()
{
    unsigned int T, i, j, Arr1[26], Arr2[26], sum[100000];
    char s1[100000], s2[100000], *p1, *p2;
    scanf("%d", &T);
    
    for(i = 1; i <= T; i++)
    {
        cin >> s1;
        cin >> s2;
        for (j = 97; j <= 122; j++)
        {
            Arr1[j-97] = 0;
            Arr2[j-97] = 0;
            p1 = strchr(s1, char(j));
            p2 = strchr(s2, char(j));
            
            if (p1 != NULL && p2 != NULL)
            {
                Arr1[j-97] = 1;
                Arr2[j-97] = 1;
            }
        }
        sum[i-1] = 0;
        for(auto& num : Arr1)
            sum[i-1] += num;
    }
    for(i = 1; i <= T; i++)
    {
        if (sum[i-1] >= 1)
        {
            printf("Yes\n");
        }
        else
        {
            printf("No\n");
        }
    }
    
    return 0;
}
