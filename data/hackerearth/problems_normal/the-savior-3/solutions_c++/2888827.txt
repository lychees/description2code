#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    
    while(tc--)
    {
    	
    	int n;
    	cin>>n;
    	int i,j;
    	int a[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	int count = 0;
    	for(i=0;i<n-1;i++)
    	{
    		for(j=i+1;j<n;j++)
    		{
    			if(((a[i] + a[j]) % 2 == 0) && a[i] != a[j])
    			count++;
    		}	
    	}
    cout<<count<<endl;
    }
    return 0;
}
