#include <iostream>
#include<bits/stdc++.h>
using namespace std;


bool cmp(pair<long long int,long long int> a, pair<long long int,long long int> b)
{
	return (a.second<b.second);
}



void func(vector<pair <long long int,long long int>> R, int N, long long int M)
{
	sort(R.begin(),R.end(),cmp);
	int i;
	for(i=0;i<N;i++)
	{
		if(M<R[i].second)
		{
			printf("NO\n");
			return;
		}
		else
		{
			M+=R[i].first;
		}
	}
	printf("YES\n");
}


int main()
{
    int T,i,N,j;
    long long int M;
    vector<pair <long long int,long long int>> R;
    cin>>T;
    for(i=0;i<T;i++)
    {
    	cin>>N>>M;
    	R.clear();
    	for(j=0;j<N;j++)
    	{
    		long long int g,p;
    		cin>>g>>p;
    		R.push_back(make_pair(g,p));
    	}	
       func(R,N,M);
    }
    
    return 0;
}
