#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    
    long long int distinct_no=0;
    bool no[1000000]={false};
    
    while(n--)
    {
    	long long int x;
    	cin>>x;
    	
    	if(no[x]==false){
    		no[x]=true;
    		distinct_no++;
    	}
    }
    
  	
	cout<<(distinct_no)*(distinct_no-1)/2<<endl;
    
    return 0;
}
