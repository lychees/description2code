#include <iostream>
#include <stdlib.h>
#include <string>

using namespace std;

struct node{
	int val;
	node* children[26];
};

struct trie{
	node* root;
};

node* getNode(){
	node* temp=(node *)malloc(sizeof(node));
	temp->val=0;
	for(int i=0;i<26;i++)
		temp->children[i]=NULL;
	return temp;
}

void init(trie* t){
	t->root=getNode();
}

void insert(trie* t,string s,int rank){
	node* pCrawl = t->root;
	int len=s.length();
	for(int i=0;i<len;i++){
		if(!pCrawl->children[s[i]-'a'])
			pCrawl->children[s[i]-'a']=getNode();
		pCrawl=pCrawl->children[s[i]-'a'];
		pCrawl->val=max(rank,pCrawl->val);
	}
	pCrawl->val=rank;
}

int search(trie* t,string s){
	int len=s.length();
	node* pCrawl=t->root;
	for(int i=0;i<len;i++){
		if(!pCrawl->children[s[i]-'a'])
			return -1;
		pCrawl=pCrawl->children[s[i]-'a'];
	}
	return pCrawl->val;
	
}


int main()
{
//    cout << "Hello World!" << endl;
	int n,q,rank;
	string s;
	trie t;
	
	cin >> n >> q;
	
	init(&t);
	
	for(int i=0;i<n;i++){
		cin >> s >> rank;
		insert(&t,s,rank);
	}
	
	for(int i=0;i<q;i++){
		cin >> s;
		rank = search(&t,s) ;
		cout << rank << endl;
	}
	
    return 0;
}
