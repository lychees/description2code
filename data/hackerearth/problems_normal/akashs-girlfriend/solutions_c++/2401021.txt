#include <iostream>
using namespace std;

int const MOD = 1e9 + 9;

int test;
long long n;

int bp(long long x, long long d){
	if(!d){
		return 1;
	}
	if(d&1){
		return x * bp(x, d-1) % MOD;
	}
	x = bp(x, d/2);
	return x * x % MOD;
}

int solve(long long n){
	if(n == 1){
		return 1;
	}
	if(n == 2){
		return 18;
	}
	if(n&1){
		long long a = solve(n >> 1);
		long long notA = (bp(10, n >> 1) - a + MOD) % MOD;
		long long b = (notA + a * 9) % MOD;
		long long notB = (notA * 9 + a) % MOD;
		return(a * notB + b * notA) % MOD;
	}else{
		long long a = solve(n >> 1);
		long long notA = (bp(10, n >> 1) - a + MOD) % MOD;
		return (a * 2 * notA) % MOD;
	}
	
}

int main()
{
    scanf("%d",&test);
    for(int t=0;t<test;++t){
    	scanf("%lld",&n);
    	printf("%d\n",solve(n));
    }
    return 0;
}
