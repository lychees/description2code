N, K = [int(i) for i in raw_input().split()]
KnownAngle = [int(x) for x in raw_input().split()]

Todraw = [int(x) for x in raw_input().split()]


def Tell(toknow):
    for j in KnownAngle:
        for i in range(1, 360):
            if (j*i) % 360 == toknow:
                return True


def Telll(toknow):
    for i in KnownAngle:
        for k in KnownAngle:
            if (i+k) == toknow or (i-k) == toknow or (k-i) == toknow:
                return True


for j in Todraw:
    if Tell(j) or Telll(j):
        print "YES"
    else:
        print "NO"
