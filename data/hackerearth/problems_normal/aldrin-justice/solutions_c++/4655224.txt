#include <stdio.h>
#define MAX(x, y) (((x) > (y)) ? (x) : (y))
#define MIN(x, y) (((x) < (y)) ? (x) : (y))
int t,b1,b2,m1,m2,l,r;
int main()
{
scanf("%d",&t);
while(t)
{
scanf("%d%d%d%d",&b1,&b2,&m1,&m2);
r= MAX ( MIN(b1,b2),MIN(m1,m2 ));
l = MIN ( MAX(b1,b2),MAX(m1,m2 ));

if((l>r)) printf("Nothing\n");
else if(l==r) printf("Point\n");
else printf("Line\n");
t--;
}
return 0;
}
