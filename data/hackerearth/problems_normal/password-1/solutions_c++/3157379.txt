#include <bits/stdc++.h>
using namespace std;
int palindrome(string a,string b){
	int al,bl,j,k=0,count=0;
	al=a.length();
	bl=b.length();
	if(al!=bl)
		return 0;
	else {
		for(j=al-1;j>=0;j--){
			if(a.at(j)==b.at(k)){
				count++;
			}
			k++;
		}
		if(count==al)
			return 1;
	}
	return 0;
	
}
int main()
{
	int n,x,i,j;
    cin>>n;
    string s[n];
    for(i=0;i<n;i++){
    	cin>>s[i];
    }
    for(i=0;i<n-1;i++){
    	for(j=i+1;j<n;j++){
    		
    		x=palindrome(s[i],s[j]);
    		if(x){
    			break;
    		}
    	}
    	if(x){
    		cout<<s[i].length()<<" "<<s[i].at(s[i].length()/2);
    		break;
    	}
    }
    
    
    return 0;
}
