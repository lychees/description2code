#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		long long int n;
		scanf("%lld",&n);
		bool f=false;
		for(int i=2;i<=sqrt(n);i++){
			if(n%i==0){
				int c=0;
				while(n%i==0){
					n/=i;c++;
				}
				if(c>1)
				   f=true;
			}
		}
		if(f)
			printf("No\n");
			else
			printf("Yes\n");
	}
	return 0;
}
