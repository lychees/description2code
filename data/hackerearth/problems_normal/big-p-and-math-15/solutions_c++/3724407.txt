#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int a,b;
    
    cin>>a>>b;
    
    int a1=0,a2=0,power=0;
    while(a){
    	
    	int temp = a%10;
    	if(temp == 5)//if digit is 5
    		a1 = a1 + 6*pow(10, power);
    	else
			a1 = a1 + temp*pow(10, power);
    		
    	if(temp == 6)//if digit is 6
    		a2 = a2 + 5*pow(10, power);
    	else
    		a2 = a2 + temp*pow(10, power);
    		
    		power++;
    		a = a/10;
    }
    
    int b1=0,b2=0;
    power = 0;
    while(b){
    	
    	int temp = b%10;
    	if(temp == 5)//if digit is 5
    		b1 = b1 + 6*pow(10, power);
    	else
			b1 = b1 + temp*pow(10, power);
    		
    	if(temp == 6)//if digit is 6
    		b2 = b2 + 5*pow(10, power);
    	else
    		b2 = b2 + temp*pow(10, power);
    		
    		power++;
    		b = b/10;
    }
    
    
    cout<<a2+b2<<" ";
    cout<<a1+b1<<endl;
    
    return 0;
}
