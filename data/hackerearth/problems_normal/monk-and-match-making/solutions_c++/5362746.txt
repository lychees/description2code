#include<bits/stdc++.h>
#define Base1 29
#define Base2 255
#define MOD1 1000000007
#define MOD2 1000000009
using namespace std;
long long arr[100005],mas[100005],p[100005],k[100005];
long long q,n,a,b,x,y;
string s;
void precompute()
{
	p[0]=1;
	k[0]=1;
	for(int i=1;i<=n;i++)
	{
		p[i]=(p[i-1]*Base1)%MOD1;
		k[i]=(k[i-1]*Base2)%MOD2;
		arr[i]=(arr[i-1]*Base1+(s[i-1]-'a'+1))%MOD1;
		mas[i]=(mas[i-1]*Base2+(s[i-1]-'a'+1))%MOD2;
		//cout<<"---------------"<<endl;
		//cout<<p[i]<<" "<<k[i]<<endl;
		//cout<<arr[i]<<" "<<mas[i]<<endl;
	}
}

long long gethash(int l,int r)
{
	int len=r-l+1;
	return 	(arr[r]-(arr[l-1]*p[len])%MOD1+MOD1)%MOD1;
}
long long _gethash(int l,int r)
{
	int len=r-l+1;
	return 	(mas[r]-(mas[l-1]*k[len])%MOD2+MOD2)%MOD2;
}
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>s;
	n=s.size();
	cin>>q;
	precompute();
	for(int i=0;i<q;i++)
	{
		cin>>a>>b>>x>>y;
		if(y-x==b-a)
		{
			pair<long long,long long> X(gethash(a,b),_gethash(a,b));
			pair<long long,long long> Y(gethash(x,y),_gethash(x,y));
			//cout<<X.first<<" "<<X.second<<" "<<Y.first<<" "<<Y.second<<endl;
			if(X==Y)
				cout<<"Yes"<<endl;
			else 
				cout<<"No"<<endl;
		}
		else
		cout<<"No"<<endl;
	}
	return 0;
}
