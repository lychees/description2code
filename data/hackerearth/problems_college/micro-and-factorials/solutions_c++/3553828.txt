#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
ll int mysqr(ll int x)
{
	return (x*x)%mod;
}
ll int mypow(ll int x,ll int nn)
{
	ll int ans=1;
	if(nn==0)
	return 1;
	if(nn==1)
	return x;
	if(!(nn&1))     //n is even
	return mysqr(mypow(x,nn>>1));
	return (x * mysqr(mypow(x,(nn-1)>>1)))%mod;
}
bool p[100010];
ll int a[10000];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int i,j;
	memset(p,0,sizeof(p));
	p[0]=1;p[1]=1;
	for(i=2;i<100010;i++)
	{
		if(p[i])
		continue;
		for(j=2;(j*i)<100010;j++)
		{
			p[j*i]=1;
		}
	}
	j=0;
	for(i=0;i<100010;i++)
	{
		if(!p[i])
		{
			a[j++]=i;
			//~ cout<<i<<endl;break;
		}
	}
	//~ cout<<j<<endl;
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int x,y;
		cin>>y>>x;
		ll int ans,ans1=1,ans2=1,t1,t2,t3,t4;
		if(y<=x)
		ans=0;
		else
		{
			for(j=0;a[j]<=y;j++)
			{
				t1=1;t3=0,t4=0;
				while(1)
				{
					t1=t1*a[j];
					t2=x/t1;
					t3=(t3+t2)%mod;
					if(t2==0)
					break;
					//~ cout<<t1<<" "<<t2<<" "<<a[j]<<"here "<<endl;
				}
				t1=1;
				while(1)
				{
					t1=t1*a[j];
					t2=y/t1;
					t4=(t4+t2)%mod;
					if(t2==0)
					break;
					//~ cout<<t1<<" "<<t2<<" "<<a[j]<<"here2 "<<endl;
				}
				//~ cout<<ans1<<" "<<ans2<<endl;
				ans1=(ans1*(t3+1))%mod;
				ans2=(ans2*(t4+1))%mod;
			}
			ans=(ans2-ans1+mod)%mod;
		}
		cout<<ans<<endl;
	}
	return 0;
}
