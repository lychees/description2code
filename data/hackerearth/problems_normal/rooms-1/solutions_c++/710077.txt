#include<cstdio>
#include<utility>
#include<algorithm>
using namespace std;
int n,i,j,k,l,m,cnt,z=0,t,u,a,ans;
pair< int , int >p[200009];
int ar[200009],br[200009];





int main()
{
scanf("%d",&t);
z=1;
while(t--)
{
scanf("%d",&n);
for(i=0;i<n;i++)
{
    scanf("%d",&ar[i]);

}
for(i=0;i<n;i++)
{
    scanf("%d",&br[i]);

}
k=0;
for(i=0;i<2*n;i=i+2)
{


p[i].first=ar[k];
p[i].second=1;
p[i+1].first=br[k]+ar[k];
p[i+1].second=-1;




k++;
}
ans=0;
sort(p,p+2*n);
for(i=0;i<2*n;i++)
{
   // printf("%d\n",p[i].first);
    if(p[i].second==1)
    cnt++;
    else
    cnt--;
    ans=max(ans,cnt);


}
printf("%d\n",ans);
ans=0;
cnt=0;

}







return 0;


}
