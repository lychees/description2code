#include <iostream>
using namespace std;

int main()
{
    int t,n;
    cin >> t;
    for(int c = 1; c <= t; c++){
    	cin >> n;
    	long long a[n];
    	pair<long long, long long>dp[n];
    	for(int i = 0; i < n; i++){
    		cin >> a[i];
    	}
    	dp[0].first = a[0];
    	dp[0].second = a[0];
    	dp[1].first = a[1];
    	dp[1].second = max(a[0], a[1]);
    	for(int i = 2; i < n; i++){
    		dp[i].first = a[i];
    		dp[i].first += dp[i-2].second;
    		dp[i].second = max(dp[i].first, dp[i-1].second);
    	}
    	cout << "Case " << c << ": " << dp[n-1].second << endl;
    }
    return 0;
}
