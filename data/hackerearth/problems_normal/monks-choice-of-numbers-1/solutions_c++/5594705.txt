#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,k;
		cin>>n>>k;
		int b[n];
		int a;
		int k1=0;
		for(int i=0;i<n;i++)
		{
			cin>>a;
			int j=a;
			int x=0;
			while(j)
			{
			x+=(1&j);
			j>>=1;	
			}
			b[k1++]=x;
			
		}
		sort(b,b+n);
		long long s=0;
		for(int i=n-1;i>=n-k;i--)
		{
			s+=b[i];
		}
		cout<<s<<endl;
		
	}
}
