#include<bits/stdc++.h>
using namespace std;
int main()
{
   int t,i;
	cin>>t;
	cin.ignore();
	string p,f;
	while(t--){
		cin>>p>>f;
		int a[27];
		for(i=0;i<26;i++)
		a[i]=0;
		for(i=0;i<f.length();i++)
		a[f[i]-'a']++;
		for(i=0;i<p.length();i++){
			while(a[p[i]-'a']--){
				printf("%c",p[i]);
			}
		}
		printf("\n");
	}
}
