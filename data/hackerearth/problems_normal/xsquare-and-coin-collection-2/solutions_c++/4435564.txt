#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	ll k,ans=0,count=0;
    	scanf("%d%lld",&n,&k);
    	ll a[n];
    	for(int i=0;i<n;i++)
    	scanf("%lld",&a[i]);
    	
    	for(int i=0;i<n;i++)
    	{
    		if(a[i]>k)
    		{
    			ans=max(ans,count);
    			count=0;
    		}
    		else
    			count=count+a[i];
    	}
    		ans=max(ans,count);
    	printf("%lld\n",ans);
    }
    return 0;
}
