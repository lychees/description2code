#include<bits/stdc++.h>
using namespace std;

char s1[5007];
char s2[5007];
int dp[5007][5007];
int solve()
{
    int n,m;
    n=strlen(s1);
    m=strlen(s2);
    int i,j;
    for(i=0;i<=m;i++)dp[0][i]=0;
    for(j=0;j<=n;j++)dp[i][0]=0;
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=m;j++)
        {
           if(s1[i-1]==s2[j-1]&&!(s1[i-1]=='i'||s1[i-1]=='o'||s1[i-1]=='u'||s1[i-1]=='a'||s1[i-1]=='e'))
           {
               dp[i][j]=1+dp[i-1][j-1];
           }

           else if((s1[i]=='i'||s1[i]=='o'||s1[i]=='u'||s1[i]=='a'||s1[i]=='e'))
           {
               dp[i][j]=dp[i-1][j];
           }

           else
           {
               dp[i][j]=max(dp[i-1][j],dp[i][j-1]);
           }
        }
    }
    cout<<dp[n][m]<<endl;
}
int main()
{
    cin>>s1>>s2;
    solve();
}
