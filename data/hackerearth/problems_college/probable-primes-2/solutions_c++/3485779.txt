using namespace std;

int primes[1000002];


bool isPrime(int n) {
    if (n == 1) {
        return false;
    }
    if (n == 2) {
        return true;
    }
    for (int i = 2; i <= sqrt(n); i++) {
        if (n % i == 0) {
            return false;
        }
    }
    return true;
}


int main()
{
    primes[0] = primes[1] = 0;
    for (int i = 2; i <= 1000000; i++) {
        if (isPrime(i)) {
            primes[i] = 1;
        }
    }
    for (int i = 1; i <= 1000000; i++) {
        primes[i] += primes[i - 1];
    }

    int t, a, b;

    cin >> t;

    while (t--) {
        cin >> a >> b;
        float x = primes[b] - primes[a - 1];
        float y = (b - a + 1);
        printf("%.6f\n", (x / y));
    }

    return 0;
}
