#include <iostream>
using namespace std;

bool pr[1000001];
int primes[100000];
int cnt;
bool check[100000001];
int cumm[10001];
void sieve()
{
	for(int i=2;i<=1000;++i)
	{
		if(pr[i]==0)
		{
			for(int j=i*i;j<=1000000;j+=i)
				pr[j]=1;
		}
	}
	cnt=0;
	for(int i=2;i<=1000000;++i)
	{
		if(pr[i]==0)
		{
			primes[cnt++]=i;
		}
	}
	int ans=0;
	for(int i=0;i<cnt;++i)
	{
		for(int j=i+1;j<cnt;++j)
		{
			if(1LL*primes[i]*primes[j]>100000000) break;
			check[primes[i]*primes[j]]=1;
		}
	}
	for(int i=0;i<10000;++i)
	{
		for(int j=i*10000+1;j<=(i+1)*10000;++j)
		{
			if(check[j]) cumm[i]++;
		}
	}
}
int main()
{
    sieve();
    int t,l,r;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d %d",&l,&r);
    	int b1=(l-1)/10000,b2=(r-1)/10000;
    	int ans=0;
    	if(b1==b2)
    	{
    		for(int i=l;i<=r;++i)
    			if(check[i]) ans++;
    		printf("%d\n",ans);
    		continue;
    	}
    	for(int i=l;i<=(b1+1)*10000;++i)
    		if(check[i]) ans++;
    	for(int i=b1+1;i<b2;++i) ans+=cumm[i];
    	for(int i=b2*10000+1;i<=r;++i)
    		if(check[i]) ans++;
    	printf("%d\n",ans);
    }
    return 0;
}
