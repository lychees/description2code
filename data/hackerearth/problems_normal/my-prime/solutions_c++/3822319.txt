#include <iostream>
using namespace std;

int main()
{
    int n,count=0;
    cin>>n;
    int s[n];
    for(int i=0;i<n;i++)
    cin>>s[i];
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<n;j++)
    	{
    		if(s[i]>s[j])
    		{
    			if(s[i]%s[j]==0)
    				count++;
    		}
    	}
    	if(count==0)
    	cout<<s[i]<<" ";
    	count=0;
    }
    return 0;
}
