#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++) cin>>arr[i];
    sort(arr,arr+n);
    int ans=1;
    int temp=arr[0];
    for(int i=1;i<n;i++)
    {
    	if(arr[i]>temp+4)
    	{
    		temp=arr[i];
    		ans++;
    	}
    }
    cout<<ans;
    return 0;
}
