#include<iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int a,b,c;
        cin>>a>>b>>c;
        //line is ax + by + c=0
        int tr1=0,tr2=0;
        while(n--)
        {
            int x,y;
            cin>>x>>y;
            int value=a*x+b*y+c;
            if(value<0) tr1++;
            else if(value>0) tr2++;
        }
        if(tr1<= tr2)  cout<<tr2<<endl;
        else cout<<tr1<<endl;
    }
}
