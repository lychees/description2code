def solve(lst, pos, num, n):
    if pos < n:
        for i in range(1, n+1):
            if i not in num[:pos]:
                num[pos] = i
                solve(lst, pos+1, num, n)
    else:
        lst.append("".join(map(str, num[:n])))


t = int(raw_input())
for T in range(t):
    n = int(raw_input())
    lst = []
    num = [0 for i in range(n)]
    solve(lst, 0, num, n)
    print(" ".join(lst))
