#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	int x;
    	cin>>x;
    	
    	int th=0,f=0,i=1,k=1;
    	
    		if((x-5)%3==0){
    			f=5;
    			x=x-5;
    			th=3;
    			while((x-th*k)!=0){
    				k++;
    			}
    			x=x-th*k;
    		}
    		if(x%3==0 && x!=0){
    			th=3;
    			while((x-th*k)!=0){
    				k++;
    			}
    			x=x-th*k;
    		}
    		
    	if(f==0 && th==0)cout<<-1<<endl;
    	else{
    		for(i=0;i<th*k;i++)cout<<5;
    		for(i=0;i<f;i++) cout<<3;
    		cout<<endl;
    	}
    	
    }
    return 0;
}
