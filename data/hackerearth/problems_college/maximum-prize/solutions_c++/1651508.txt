#include <iostream>
#include<algorithm>
using namespace std;
 
int main()
{
     int n,t,k;int *arr,i;
       cin>>t;long int sum=0;
       
       while(t-->0)
       {
       	cin>>n>>k;
       	arr=new int[n];
       	for(i=0;i<n;i++)
       	cin>>arr[i];
       	
       sort(arr,arr+n);
       	i=n-1;
       	sum=0;
       while(k>0 && i>=0)
       {
       	sum+=arr[i];
       	k--;i--;
       }
        cout<<sum<<endl;
    
}
return 0;
}
