#define mod 1000000007
struct mat
{
	long long int a,b,c,d;
};
mat mul(mat x,mat y)
{
	mat z;
	z.a=(x.a*y.a+x.b*y.c)%mod;
	z.b=(x.a*y.b+x.b*y.d)%mod;
	z.c=(x.c*y.a+x.d*y.c)%mod;
	z.d=(x.c*y.b+x.d*y.d)%mod;
	return z;
}
mat pmod(mat a,int b)
{
    if(b==1)
        return a;
    mat ans;
    if(b%2==0)
    {
        ans=pmod(a,b/2);
        ans=mul(ans,ans);
    }
    else
    {
        ans=pmod(a,(b-1)/2);
        ans=mul(ans,ans);
        ans=mul(ans,a);
    }
    return ans;
}
