#include <bits/stdc++.h>
using namespace std;

typedef long long LL;

#define PII       pair<int,int>
#define all(c)    c.begin(),c.end()
#define sz(c)     (int)c.size()
#define clr(c)    c.clear()
#define pb        push_back
#define mp        make_pair
#define cin(x)    scanf("%d",&x)
#define MOD		1000000007
#define EPS		1E-10

LL F(PII A, PII B, PII C)
{
	LL ret = 1LL * A.first * (B.second - C.second);
	ret += 1LL * B.first * (C.second - A.second);
	ret += 1LL * C.first * (A.second - B.second);
	return ret;
}

LL area(vector<PII> &A)
{
	LL ret = 0;
	for(int i = 2; i < sz(A); i++)
		ret += F(A[0] , A[i - 1] , A[i]);
	if(ret < 0)
	ret *= -1;
	return ret;
}

LL boundary(vector<PII> &A)
{
	LL ret = 0;
	for(int i = 0; i < sz(A); i++)
	{
		int a = i;
		int b = (i + 1) % (sz(A));
		int dx = A[a].first - A[b].first;
		int dy = A[a].second - A[b].second;
		dx = abs(dx);
		dy = abs(dy);
		ret += __gcd(dx,dy);
	}
	return ret;
}

int main()
{
	int n;
	cin >> n;
	vector<PII> arr(n);
	for(int i = 0; i < n; i++)
		cin >> arr[i].first >> arr[i].second;
	LL ans = area(arr);
	ans += 2;
	ans -= boundary(arr);
	ans /= 2;
	cout << ans << "\n";
	return 0;
}
