#include <iostream>
using namespace std;
#include <string>
#include <algorithm>
 
 
int main()
{
	int n;
	cin >> n;
	string st;
	char test = 'C';
	int max =0;
	int maxmax = 0;
	for(int i =0; i < n;i++){
		string temp;
		cin >> temp;
		st.append(temp);
		int l = temp.length();
		
		int count = 0;
		for(int q = 0; q < l;q++){
			if(temp[q] == test){
				
				count++;
				if(max < count) max = count;
			}else{
				if(max < count) max = count;
				count = 0;
			}
		}
 
 
	}
	
	int m = st.length();
	
	int count = 0;
		for(int q = 0; q < m;q++){
			if(st[q] == test){
				count++;
				if(maxmax < count) maxmax = count;
			}else{
				if(maxmax < count) maxmax = count;
				count = 0;
			}
		}
 
		cout << max << " " << maxmax << " "<< endl;
 
    return 0;
}
