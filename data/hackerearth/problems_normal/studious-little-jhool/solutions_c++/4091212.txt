#include <bits/stdc++.h>
using namespace std;

int solve(vector<int>& dp, int N) {
	if(N==0) return 0;
	if(N<0) return 10000;
	if(dp[N] != -1) return dp[N];
	dp[N] = 1 + min(solve(dp,N-10),solve(dp,N-12));
	return dp[N];
}

int main()
{
    //cout << "Hello World!" << endl;
    int T;
    cin>>T;
    while(T--) {
    	int N;
    	cin>>N;
    	vector<int> dp(N+1,-1);
    	int res=solve(dp,N);
    	if(res > N) {
    		cout<<"-1"<<endl;
    	} else {
    		cout<<res<<endl;
    	}
    }
    return 0;
}
