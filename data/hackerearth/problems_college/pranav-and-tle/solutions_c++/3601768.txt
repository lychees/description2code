#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t,i;
    long m;
    long long p=0;
    char a[500010];
    scanf("%d",&t);
    while(t--)
    {	cin>>a;
    p=0;
    scanf("%ld",&m);
    
    for(i=0;i<strlen(a);i++)
    {
    	p=(p*10)+(a[i]-'0');
    	if(p>=m)
    		p=p%m;
    }
    p=p%m;
    printf("%lld\n",p);
    }
    return 0;
}
