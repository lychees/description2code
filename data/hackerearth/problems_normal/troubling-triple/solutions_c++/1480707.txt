#include <bits/stdc++.h>
using namespace std;
#define ll  long long
vector<ll> v;
vector<ll>::iterator up;
inline ll scan(){
unsigned ll t=0;char c=getchar_unlocked();
while(c<'0' || c>'9')c=getchar_unlocked();
while(c>='0' && c<='9'){t=(t<<3)+(t<<1)+c-'0';c=getchar_unlocked();}
return(t);}
int main() 
{
	ll n,m,i,j,kl,km,res=0;
	n=scan(),m=scan();
	for(i=0;i<n;i++)
	{
		kl=scan();
		v.push_back(kl);
	}
	sort(v.begin(),v.end());
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			km=(ll)m/v[i];
			km=(ll)km/v[j];
			up=upper_bound(v.begin(),v.end(),km);
			if(up-v.begin()>j)
			res=res+(up-(v.begin()+j)-1);
		}
	}
	printf("%lld\n",res);
	return 0;
}
