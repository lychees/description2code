#include <iostream>
#include <vector>
#include <algorithm>
#include <utility>

#define LL long long

using namespace std;



bool compare(const pair<LL int,LL int>&i, const pair<LL int,LL int>&j)
{
	
	return i.second < j.second;
	
	
}


int main()
{
    int T;
    cin>>T;
    
    
    for(int i=0;i<T;i++)
    
    {
    	
    LL int N,M;
    
    cin>>N>>M;
    
    LL int gi,pi;
    
    vector< pair<LL int,LL int> > vp;
    
    for(LL int i=0;i<N;i++)
    {
    	cin>>gi>>pi;
    	
    	vp.push_back(make_pair(gi,pi));
    	
    	
    	
    }
    
    
    sort(vp.begin(),vp.end(),compare);
    
    bool flag = true;
    
    for(LL int i=0;i<N;i++)
    { 
    	
    	//cout<<vp[i].first<<"   "<<vp[i].second<<endl;
    	
    	if(vp[i].second<M) { M+=vp[i].first; }
    	
    	else {flag=false; break;}
    	
    	//
    	
    }
    
    vp.clear();
    
    
    if(flag) cout<<"YES"<<endl;
    
    else cout<<"NO"<<endl;
    
    }
    
    return 0;
}
