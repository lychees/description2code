#include <iostream>
using namespace std;

int coprime(int m,int n) {
	for(int j=2;j<=n;j++) {
		if(n%j ==0 && m%j==0) return 1;
	}
	return 0;
}


int main()
{
	
	
   	int m;
   	int n;
   	cin >>m>>n;
   	int a[n];
   	int c=0;
   	for(int i=0;i<n;i++) {
   		cin >>a[i];
   		if(coprime(m,a[i])) c++;
   	}
   	cout <<c;
    return 0;
}
