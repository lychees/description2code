for _ in range(input()):
    a = list(raw_input())
    b = list(set(a))
    if len(b) >= 4:
        print "YES"
    elif len(b) == 1 and len(a) >= 10:
        print "YES"
    elif len(b) == 2 and len(a) >= 7:
        print "YES"
    elif len(b) == 3 and len(a) >= 5:
        print "YES"
    else:
        print "NO"
