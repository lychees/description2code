#include <iostream>
#include<stdio.h>
#include<string.h>
using namespace std;

int main()
{
    int t,p,q;
    int count[26],count2[26];
    string a,b;
     scanf("%d \n",&t); 
        while(t--)
          {
          	for(int i=0;i<26;i++)
          	  count[i]=count2[i]=0;
          	  p=q=0;
          	  getline(cin,a);
          	  getline(cin,b);
          	    int n=a.length();
          	    for(int i=0;i<n;i++)
          	       count[(int)(a[i]-'a')]++;
          	     n=b.length();
          	    for(int i=0;i<n;i++)
          	       count2[(int)(b[i]-'a')]++;
 
          	       for(int i=0;i<26;i++)
          	         {
          	         	if(count[i]>count2[i])
          	         	      p=1;
          	         	      else if(count2[i]>count[i])
          	         	        q=1;
          	         	        if(p==1&&q==1)
          	         	          break;
          	         }
          	         if(p==q)
          	           printf("You draw some. \n");
          	           else if(p==1)
          	           printf("You win some. \n");
          	           else
          	           printf("You lose some. \n");
          	
          }
}
