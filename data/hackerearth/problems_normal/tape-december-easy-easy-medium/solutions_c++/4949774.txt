#include <iostream>
using namespace std;

int main()
{
	long sum, count = 0, index = -1;
	long N, K, D, lastzeroindex;
	cin >> N >> K >> D;

	long * list = new long[N];

	for (long n = 0; n < N; n++)
	{
			cin >> list[n];
	}

	
	for (long n = index; n <= N - K; n++)
	{
		sum = 0;
		lastzeroindex = -1;
		
		for (long k = n; k < min(n + K, N); k++)
		{
			sum += list[k];
			if (list[k] != 0)
				lastzeroindex = k;
		}
		if (sum == 0)
		{
			count = -1;
			break;
		}
		if (sum < D)
		{
			count++;
			n = lastzeroindex;
			//for (long m = n; m < min(n + K, N); m++)
			//{
			//	if (list[m] != 0)
			//		lastzeroindex = m;
			//}
			n = lastzeroindex + K - 1; 
		}
	}
	cout << count << endl;
	return 0;
}
