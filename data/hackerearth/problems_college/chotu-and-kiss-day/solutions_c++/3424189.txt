#include<bits/stdc++.h>
using namespace std;
main()
{
	long long int n,f,l,s,m,flag=0;
	scanf("%lld",&n);
	long long int a[n],i,j,k;
	for(i=0;i<n;i++)
	{
		scanf("%lld",&a[i]);
	}
	scanf("%lld",&k);
	sort(a,a+n);
	for(i=0;i<n;i++)
	{
		for(j=n-1;j>=0;j--)
		{
		    if(i<j){
				s=k-(a[i]+a[j]);
				if(binary_search(a,a+n,s))
				{
					printf("YES\n") ;
					flag=1 ;
					break ;
				}			
     	  	}
		}
		if(flag==1)
			break ;
	}
	if(flag!=1)
		printf("NO");
}
