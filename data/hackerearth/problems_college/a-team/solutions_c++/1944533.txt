#include <iostream>

#include <cstdio>

int main()
{

    	        double dist_min=p1_dist;
    	        if(p2_dist<p1_dist)
    	        {
    	          dist_min= p2_dist;
    	        }
    	        if(c_dist>dist_min)
    	        {
    	          printf("NO\n");	
    	        }
    	        else
    	        {
    	        	printf("YES\n");	
    	        }
    	    }
    	    else
    	    {
    	    	printf("YES\n");
    	    }
    	}
    	else
    	{
    		//printf(" m is %f\n",m);
    		double c= -(m*(double)X1)+(double)Y1;
    		
    		double POI_x= c/(mc-m);
    		double POI_y=mc*POI_x;
    		double x_big=(double)X1,x_small=(double)X2,y_big=(double)Y1,y_small=(double)Y2;
    		//printf("%f %f \n",POI_x,POI_y);
    		if(X2>X1)
    		{
    			 x_big=(double)X2,x_small=(double)X1;
    		}
    		if(Y2>Y1)
    		{
    			 y_big=(double)Y2,y_small=(double)Y1;
    		}
    		int counter=0;
    		if(POI_x > x_small && POI_x < x_big )
    		{
    		 counter++;

    		if(POI_y > y_small && POI_y < y_big )

    		}
    		
    		if(counter!=2)

    			 double sign_1=(POI_x/(double)Xc),sign_2=(POI_y/(double)Yc);
    			 if( POI_dist<c_dist && sign_1 >0 && sign_2>0)

    	}
    }

}
