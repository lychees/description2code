#include <iostream>
using namespace std;

#include<cstring>
int main()
{
   string s;
   //scanf("%s",s);
   cin>>s;
   int cnt=0, l=s.length();
   if(s[0]>=97&&s[0]<=122)
   	 	     s[0]-=32;
   for(int i=0;i<l;i++)
   {
   	if(s[i]=='_')
   	    {cnt=1;	
   	    }
   	 if(cnt)
   	 {
   	 	if(s[i+1]>=97&&s[i+1]<=122)
   	 	     s[i+1]-=32;
   	 	cnt=0;     
   	 }
   	
   }
   //printf("%d",cnt);
   cout<<s;
    return 0;
}
