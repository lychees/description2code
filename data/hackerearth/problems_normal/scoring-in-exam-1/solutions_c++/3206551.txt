#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define F first
#define S second
bool cmp(pair<ll,ll> a,pair<ll,ll> b)
{
    if(a.S == b.S )return a.F<b.F;
    else return a.S>b.S;
}
int main()
{

        int n,q;
        cin>>n>>q;
        pair<ll,ll> p[n];
        for(int i=0;i<n;i++)
        {
            cin>>p[i].F;
        }
        for(int i=0;i<n;i++)
        {
            cin>>p[i].S;
        }
        sort(p,p+n,cmp);
        for(int i=1;i<n;i++)
        {p[i].F+=p[i-1].F;p[i].S+=p[i-1].S;}
        while(q--)
        {
            int k;
            cin>>k;
            cout<<p[k-1].F<<endl;
        }
}
