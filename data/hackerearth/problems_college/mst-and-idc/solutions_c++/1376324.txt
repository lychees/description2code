#include<bits/stdc++.h>
using namespace std;
int main()
{
    long int tc,i,n;
    scanf("%ld",&tc);
    while(tc--)
    {
        long int sum=0;
        scanf("%ld",&n);
        long int a[n];
        for(i=0;i<n;i++)
            scanf("%ld",&a[i]);
        sort(a,a+n);
        for(i=0;i<n-1;i++)
            sum+=(a[i+1]-a[i]);
        printf("%ld\n",sum*2);
    }
    return 0;
}
