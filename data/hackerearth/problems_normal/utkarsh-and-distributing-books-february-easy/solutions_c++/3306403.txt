#include<math.h>
#include<stdlib.h>
#include <iostream>
using namespace std;


	int main()
{
		int t,n,b[1000],temp,sum;
		cin>>t;
		while(t>0)
		{
			cin>>n;
			temp=1001;
			sum=0;
			for (int i=0;i<n;i++)
			{
				cin>>b[i];
				if (b[i]<temp)
					temp=b[i];
				sum=sum+b[i]-1;

			}
			cout<<temp-1<<" "<<sum<<endl;

			t--;
		}
	return 0;
}
