#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
	int t,a,b;
	long long n,x,pos,low,high,mid;
	scanf("%d",&t);
    while(t--){
    	cin>>a>>b>>n;
    	x = (a*b)/__gcd(a,b);
    	low = 1;
    	high = max(a,b)*n;
    	while (low<=high){
    		mid = low + (high-low)/2;
    		pos = mid/a + mid/b - mid/x;
    		if ( pos == n ) {
                cout<<max((mid/a)*a, (mid/b)*b)<<endl;
				break;
			}
			else if ( pos < n ) {
				low = mid+1;
			}
			else {
				high = mid-1;
			}
    	}
    }
	return 0;
}
