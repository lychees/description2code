void FastIO()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);
	cout.precision(15);
}

//global definitions


int main() 
{
	FastIO();
	// code
	int t,N,sw1,sw2;
	string s;
	cin >> t;
	while(t--)
	{
		cin >> N;
		cin >> s;
		sw1=0;
		sw2=0;
		for(int i=0;i<s.length();i++)
		{
			if( ((i%2==0) && (s[i]!='B')) || ((i%2==1)&&(s[i]!='G')) )
				sw1++;
			if( ((i%2==0)&&(s[i]!='G')) || ((i%2==1)&&(s[i]!='B')) )
				sw2++;
		}
		cout << min(sw1,sw2)/2 << '\n';
	}
	return 0;
}
