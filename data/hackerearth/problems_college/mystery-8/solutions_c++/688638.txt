#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    int test_cases;
    cin>>test_cases;
    getchar();
    while(test_cases>0)
    {
        char str[101];
        gets(str);
        string s="";
        for(int i=0;str[i];i++)
         {
         	if((str[i]>='a'&&str[i]<='z')||str[i]=='*')
         	 {
         	 	s=s+str[i];
         	 }
         	
         }
        int i=0,j=s.size()-1; 
        for(i=0;i<j;i++,j--)
         {
         	if(s[i]=='*')
         	 {
         	 	cout<<s[j]<<endl;
         	 }
         	else if(s[j]=='*')
         	 {
         	 	cout<<s[i]<<endl;
         	 }
         }
        
        test_cases--;
    }
    return 0;
}
