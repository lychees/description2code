#include <iostream>
using namespace std;

int main()
{
  int t;
  cin>>t;
  while(t--)
  {
  	int h[26]={0};
  	string s;
  	cin>>s;
  	int i=0;int m=0;
  	while(s[i])
  	{
  		h[s[i]-'a']++;
  		if(m<h[s[i]-'a'])
  		m=h[s[i]-'a'];
  		i++;
  	}
  //	cout<<m<<endl;
  int l=1;
  while(l<=m)
 {
 	for(char k='a';k<='z';k++)
  {
  	if(h[k-'a']==l)
  	for(int n=0;n<l;n++)
  	{cout<<k;}
  }
  l++;
 }
  	cout<<endl;
  }
    return 0;
}
