#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int a,b,c,t;
    cin>>t;
    while(t--)
    {
    	long long int arr[109]={0};
    	cin>>a>>b>>c;
    	arr[0]=a;
    	arr[1]=b;
    	if(c==0) cout<<a<<endl;
    	else if(c==1) cout<<b<<endl;
    	else
    	{
    		for(int i=2;i<=c;i++)
    		{
    			arr[i]=arr[i-1]+arr[i-2];
    		}
    		cout<<arr[c]<<endl;
    	}  	
    	
    }
    return 0;
}
