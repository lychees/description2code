ll gcd(ll a,ll b) {return b==0?a:gcd(b,a%b);}
ll lcm(ll a,ll b) {return a*(b/gcd(a,b));}

ll dp[MAXN];

ll solve(int n)
{
  if (dp[n]!=-1) return dp[n];
  dp[n]=(solve(n-1)+(1ll*(n-1)*solve(n-2))%MDL)%MDL;
  return dp[n];
}
