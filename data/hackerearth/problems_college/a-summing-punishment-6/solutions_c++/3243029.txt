#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int arr[n],q,l,r,j,sum,i;
    for(i=1;i<=n;i++)
    {
    	cin>>arr[i];
    }
    cin>>q;
    for(i=0;i<q;i++)
    {    
    	cin>>l>>r;
    	sum=0;
    	for(j=l;j<=r;j++)
    	{
    		sum=sum+arr[j];    	}
    		cout<<sum<<endl;
    	
    }
    
    
    return 0;
}
