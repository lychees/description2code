#include <iostream>
using namespace std;
#include<math.h>

int main()
{
    string str;
    cin>>str;
    uint64_t len=1,ans=0;
    for(int i=1;i<str.size();i++)
    {
 		   	if(str[i]==str[i-1])len++;
 		   	else
 		   	{
 		   		ans+=((len*(len+1))/2);
 		   		len=1;
 		   	}
    }
    ans+=((len*(len+1))/2);
    cout<<ans;
    return 0;
}
