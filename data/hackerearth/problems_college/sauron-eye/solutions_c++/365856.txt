#include<iostream>
#include <cstdio>
#include<algorithm>
using namespace std;

#define ll long long

#define X first
#define Y second

#define pll pair<ll,ll>
#define ppll pair< pll,pll >

typedef ppll matrix;

#define M 1000000007

matrix fm(matrix a,matrix b)
{
	matrix c;
	c.X.X=((a.X.X*b.X.X)%M+(a.X.Y*b.Y.X)%M+M)%M;
	c.X.Y=((a.X.X*b.X.Y)%M+(a.X.Y*b.Y.Y)%M+M)%M;
	c.Y.X=((a.Y.X*b.X.X)%M+(a.Y.Y*b.Y.X)%M+M)%M;
	c.Y.Y=((a.Y.X*b.X.Y)%M+(a.Y.Y*b.Y.Y)%M+M)%M;
	return c;
}

ll fp(ll e)
{
	if(e==1)return 1;
	if(e==2)return 3;
	e-=2;
	matrix a,c;
	a.X.X=3;
	a.X.Y=-1;
	a.Y.X=1;
	a.Y.Y=0;
	c=a;
	while(e)
	{
		if(e&1)c=fm(c,a);
		a=fm(a,a);
		e>>=1;
	}
	return (3*c.Y.X+c.Y.Y)%M;
}

int main()
{
	ll t,n;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		printf("%lld\n",fp(n));
	}
	return 0;
}
