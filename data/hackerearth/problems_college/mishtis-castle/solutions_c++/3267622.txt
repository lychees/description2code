#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n,c;
		scanf("%d %d",&n,&c);
		int a[n][n],dp[n][n];
		for(int i=0;i<n;i++)
		   for(int j=0;j<n;j++)
		      scanf("%d",&a[i][j]);
		      dp[0][0]=0;
	    for(int i=0;i<n;i++)
	        for(int j=0;j<n;j++){
	        	if(i==0&&j==0)
	        	   continue;
	        	dp[i][j]=-100000;
	        	if(i>0)
	        	dp[i][j]=max(dp[i][j],dp[i-1][j]);
	        	if(j>0)
	        	dp[i][j]=max(dp[i][j],dp[i][j-1]);
	        	dp[i][j]+=a[i][j];
	        	//printf("%d\n",dp[i][j]);
	        }
	   if(dp[n-1][n-1]<c)
	       printf("-1\n");
	       else
	       printf("%d\n",dp[n-1][n-1]-c);
	}
	return 0;
}
