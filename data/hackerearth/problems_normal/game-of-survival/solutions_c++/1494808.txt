#include <iostream>
#include<algorithm>



using namespace std;

int main()
{
		int t,i,j,k,s[100000],a[100000],n;
    cin>>t;
    
    
   	for(i=0;i<t;i++)
	{
			
		cin>>n;
		k=0;
		for(j=0;j<n;j++)
		{
			cin>>s[j];
		}
		for(j=0;j<n;j++)
		{
			cin>>a[j];
		}
		
		sort(a,a+n);
		sort(s,s+n);
		
		
		for(j=0;j<n;j++)
		{
			if(s[j]<a[j])
			{  k=1; break;}
			
			
			
		}
		
		if(k==0)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
	
	
	
    
	}    
    
    return 0;
}
