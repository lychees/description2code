#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    
    char s[n];
    cin>>s;
    
    int hash[26]={0};
    
    int i = 1,j = strlen(s);
    // char max = s[0];
    // int maxindex = 0;
    while(i<j)
    {
        char max = 'A';
        int maxindex = -1;
    	for(int k=i;k<j;k++)
    	{
    		if(s[k]>max && hash[s[k]-97]==0)
    		{
    			max = s[k];
    			maxindex = k;
    		}	
    	}
        if(max=='A')
            break;
        hash[s[maxindex]-97]=1;
        cout<<max;
        i = maxindex+1;
    }
    return 0;
}
