#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int main()
{
    int n, k, i;
    scanf("%d%d",&n, &k);
    long long int A[n+1], d;
    for(i=0;i<n;i++)
    {
    	scanf("%lld",&A[i]);
    }
    sort(A, A+n);
    for(i=0;i<n-1;i++)
    {
    	d=A[i+1]-A[i];
    	if(d>k)
    	{
    		printf("%lld", A[i]);
    		break;
    	}
    	else if(i==n-2)
    	{
    		printf("%lld", A[i+1]);
    	}
    }
    return 0;
}
