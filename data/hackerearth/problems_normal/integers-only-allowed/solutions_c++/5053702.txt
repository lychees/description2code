#include <bits/stdc++.h>

using namespace std;

#define Point pair<long long, long long>
#define x first
#define y second

int main() {
    int N;
    cin >> N;
    Point arr[N];
    for (int i = 0; i < N; i++) {
    	cin >> arr[i].x >> arr[i].y;
    }
    
    long long area = 0;
    for (int i = 0; i < N; i++) {
    	if (i != N-1)
    		area += (arr[i].x*arr[i+1].y - arr[i+1].x*arr[i].y);
    	else 
    		area += (arr[N-1].x*arr[0].y - arr[0].x*arr[N-1].y);
    }
    area = 0.5 * abs(area);

    long long boundary = 0;
    for (int i = 0; i < N; i++) {
    	if (i != N-1)
    		boundary += __gcd(abs(arr[i].x-arr[i+1].x), abs(arr[i].y-arr[i+1].y));
    	else
    		boundary += __gcd(abs(arr[N-1].x-arr[0].x), abs(arr[N-1].y-arr[0].y));
    }

	long long ans = area + 1 - boundary/2;
    cout << ans << endl;
    
    return 0;
}
