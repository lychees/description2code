#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll arr[105];
bool dp[100005][105];
bool vis[1000005];
int main()
{
    //cout << "Hello World!" << endl;
    ll n,t,z;
    cin>>n;
    ll s = 0;
    for(ll i=0;i<n;++i)
    {
    	cin>>arr[i];
    	s = s+arr[i];
    }
    memset(dp,false,sizeof(dp));
    for(ll i=0;i<=n;++i)
    	dp[0][i] = true;
    for(ll i=1;i<=s;++i)
    {
    	for(ll j=1;j<=n;++j)
    	{
    		dp[i][j] = dp[i][j-1];
    		if(i-arr[j-1]>=0)
    			dp[i][j] = dp[i][j] || dp[i-arr[j-1]][j-1];
    		if(dp[i][j])
    			vis[i] = true;
    	}
    }
    cin>>t;
    while(t--)
    {
    	cin>>z;
    	if(vis[z])
    		cout<<"Yes"<<endl;
    	else
    		cout<<"No"<<endl;
    }
    return 0;
}
