#include<bits/stdc++.h>
using namespace std;
int main()
{
		int n;
		cin>>n;
		int count[26], ans;
		//memset(count, 0, 26);
		for(int i=0;i<26;i++)
		count[i]=0;
		char s[20];
		for(int i=0;i<n;i++)
		{
			cin>>s;
			count[s[0]-97]++;
		}
		for(int i=0;i<26;i++)
		{
			if(count[i]>0)
			{
				if(count[i]%10!=0)
					ans= ans + count[i]/10 + 1;
					else
					ans+= count[i]/10;
			}
		}
		cout<<ans<<endl;
}
