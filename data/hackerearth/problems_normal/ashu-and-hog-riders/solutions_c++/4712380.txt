#include <bits/stdc++.h>

#define MAX 1001
#define QMAX 2000

using namespace std;

typedef struct _point { int _x; int _y;} point;

point Queue[QMAX];

int G[MAX][MAX];
bool vis[MAX][MAX];

int N;
int ans = 0;
int c =0;

void bfs(int x, int y)
{
    int f =0;
    int r =0;
    point p; p._x = x; p._y = y;
    Queue[r++] = p;

    while(r!=f)
    {
        point u = Queue[f];
        f =  (f+1)%QMAX;
        point v;

        if((u._x<N) && (G[u._x+1][u._y] == 1) && (vis[u._x+1][u._y] == false))
        {
            v._x = u._x+1; v._y = u._y;
            Queue[r] = v;
            r = (r+1)%QMAX;
            vis[u._x+1][u._y] = true;
                c++;
        }

        if((u._x>1) && (G[u._x-1][u._y] == 1) && (vis[u._x-1][u._y] == false))
        {
            v._x = u._x-1; v._y = u._y;
            Queue[r] = v;
            r = (r+1)%QMAX;
            vis[u._x-1][u._y] = true;
                c++;
        }

        if((u._y<N) && (G[u._x][u._y+1] == 1) && (vis[u._x][u._y+1] == false))
        {
            v._x = u._x; v._y = u._y+1;
            Queue[r] = v;
            r = (r+1)%QMAX;
            vis[u._x][u._y+1] = true;
                c++;
        }

        if((u._y>1) && (G[u._x][u._y-1] == 1) && (vis[u._x][u._y-1] == false))
        {
            v._x = u._x; v._y = u._y-1;
            Queue[r] = v;
            r = (r+1)%QMAX;
            vis[u._x][u._y-1] = true;
                c++;
        }
    }

}

int main()
{
    scanf("%d", &N);
    for(int i=1; i<=N; i++)
    {
        for(int j=1; j<=N; j++)
            scanf("%d",&G[i][j]);
    }

    for(int i=1; i<=N; i++)
    {
        for(int j=1; j<=N; j++)
        {
            if(!vis[i][j] && G[i][j]==1)
            {
                c=1;
                vis[i][j] = true;
                //dfs(i,j);
                bfs(i,j);
                if(ans<c)
                    ans = c;
            }
        }

    }

    printf("%d",ans);

    return 0;
}
