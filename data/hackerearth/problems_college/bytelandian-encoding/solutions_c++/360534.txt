#include <iostream>
#include <string.h>
using namespace std;

int main()
{
    int t,n, temp, loc, p;
    char s[50];
    scanf("%d",&t);
    while(t--)
    {
    	temp = 0;
    	scanf("%s",&s);
    	n = strlen(s);
    	p = n;
    	while(temp != p)
    	{
    		if(n % 2 == 0)
    		{
    			if(s[(n/2)-1] <= s[(n/2)])
    			{
    				printf("%c",s[(n/2)-1]);
    				loc = (n/2)-1;
    			}
    			else
    			{
    				printf("%c",s[(n/2)]);
    				loc = (n/2);
    			}
    		}
    		else
    		{
    			printf("%c",s[(n/2)]);
    			loc = (n/2);
    		}
    		++temp;
    		//cout<<temp<<endl;
    		for(int i = loc; i<n; ++i)
    			s[i] = s[i+1];
    		--n;
    	}
    	printf("\n");
    }
    return 0;
}
