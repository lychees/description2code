from fractions import gcd
n=raw_input("")
n=int(n)
for y in range(0,n):
	li=raw_input("")
	li=map(str,li.split())
	a=int(li[0])
	b=int(li[1])
	name=li[2]
	if(name=='Chandu'):
		flag=1
	else:
		flag=0
	while(a!=0 and b!=0):
		x=gcd(a,b)
		a=a-x
		b=b-x
		flag=flag^1
	

		
	if(flag==0):
		print 'Chandu'
	else:
		print 'Daspal'
