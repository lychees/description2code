#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin >> t;
   
    while(t--){
    	
    	int n;
    	cin >> n;
    	
    	vector<int> store (10, 0);
    	
    	for(int i=0; i<n; i++){
    		string s;
    		cin >> s;
    		int m = s[0]-'0';
    		store[m]++;
    	}
    	
    	int threshold;
    	cin >> threshold;
    	
    	vector<int> res;
    	
    	for(int i=1;i<10;i++){
    		if(((n * log10(1 + 1/(float)i)) * threshold < store[i] || (n * log10(1 + 1/(float)i)/threshold) > store[i]))
    			res.push_back(i);
    	}
    	
    	if(res.size() > 0){
    		cout << res[0] << endl;
    	}
    	else cout << -1 << endl;
    }
    return 0;
}
