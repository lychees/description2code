#include <iostream>
#include <cstring>
using namespace std;
typedef long long int ll;
int main()
{
    int t;
    cin >> t;
    while(t--) {
    	string s;
    	cin >> s;
    	ll l = s.size();
    	ll i;
    	string s1 = "";
    	s1 = s1 + s[0];
    	for(i = 1; i < l; i++) {
    		if(s[i] < s1[0]) {
    			s1 = s1 + s[i];
    		}
    		else {
    			s1 = s[i] + s1;
    		}
    	}
    	cout << s1 << endl;
    }
    return 0;
}
