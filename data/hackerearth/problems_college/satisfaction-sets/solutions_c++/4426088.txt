    #include<bits/stdc++.h>
    using namespace std;
    const int mod = 1e9;
    set<int> s;
    bool binarySearch(vector<int> v,int x)
    {
        for(int i=0;i<v.size();i++)
        {
            if(v[i]==x)
                return true;
        }
        return  false;
    }
    int main()
    {
        //freopen("in.txt","r",stdin);
        ios_base::sync_with_stdio(false);
        int n,j,p;
        long t;
        cin>>n;
        vector<int> v[n];
        for(int i=0;i<n;i++)
        {
            cin>>j;
            for(int k=0;k<j;k++)
            {
                cin>>p;
                v[i].push_back(p);
                s.insert(v[i][k]);
            }
        }
        long long sum=0,sum1=0;
        int flag;
        set<int> :: iterator it;
        for(it=s.begin();it!=s.end();it++)
        {
            sum=sum+(*it);
            sum %= mod;
            flag=1;
            for(int i=0;i<n;i++)
            {
                if(!binarySearch(v[i],*it)){
                    flag=0;
                    break;
                }
            }
            if(flag){
                sum1+= (*it);
                sum1 %= mod;
            }
        }
        cout<<(-1)*sum1<<endl;
        return 0;
    }
