#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	string s;
	cin >> t;
	while(t--)
	{
		int count[26]={0},fg=0;
		cin >> s;
		for(int i=0;i<s.length();i++)
		{
			count[ s[i]-97 ]++;
			
		}
		for(int i=0;i<26;i++)
		{
			if(count[i]%2)
			{
			fg=1;
			break;
			}
		
		}
		if(fg)
		cout << -1 << endl;
		else
		cout << 1 << endl;
	}
}
