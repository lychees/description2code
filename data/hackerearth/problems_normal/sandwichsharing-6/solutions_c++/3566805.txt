#include <iostream>
#include <cassert>
using namespace std;

int main()
{
    int	T; //Number of test-cases
    
    //Read count of test-cases
    cin >> T;
    
    assert((T >= 1) && (T <= 100000));
    
    while (T--)
    {
    	long	N; //Number of sides
    	long	P; //Number of friends
    	
    	//Read N and P
    	cin >> N >> P;
    	
    	assert((N >= 3) && (N <= 1000000000L));
    	assert((P >= 0) && (P <= 1000000000L));
    	
    	long	result = 0; //Contains count of sides left after all friends have cut the sandwich as per given rule
    	
    	if ((P == 0) || (N == 3))
    		result = N;
    	
    	while ((P > 0) && (N > 3))
    	{
    		N = ((N % 2) == 0) ? ((N >> 1) + 1) : ((N >> 1) + 2);
    		result = N;
    		--P;
    	}
    	
    	cout << result << endl;
    }
    
    return 0;
}
