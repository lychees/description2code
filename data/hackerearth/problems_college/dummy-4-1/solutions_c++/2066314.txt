#include <bits/stdc++.h>
using namespace std;

int a[100][100];
int n;
long long cnt;
bool visit[100][100];
int dx[] = {1,0,-1,0}, dy[]= {0,1,0,-1};

void f(int i,int j) {
    if(i==n-1&&j==n-1) {
        if(!a[n-1][n-1]) cnt++;
        return ;
    }
    if(a[i][j]) return ;
    visit[i][j] = true;
    for(int k=0;k<4;k++) {
        int x = i + dx[k], y = j+ dy[k];
        if(x>-1&&x<n&&y>-1&&y<n&&!visit[x][y]&&!a[x][y]) f(x,y);
    }
    visit[i][j] = false;
}
int main() {
    int j,i;
    scanf("%d",&n);
    for(i=0;i<n;i++)
            for(j=0;j<n;j++)
                  scanf("%d",&a[i][j]);

    f(0,0);
    printf("%lld",cnt);
    return 0;
}
