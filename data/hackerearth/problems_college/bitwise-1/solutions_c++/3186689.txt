#include <iostream>

using namespace std;

int main()
{
	long long t,m;
	cin >> t;
	for(m=1;m<=t;m++)
	{
		long long x,k;
		cin >> x >> k;
		long long i=0,j=0;
		while(j<=k)
		{
			long long a1=x^i;
			long long a2=x+i;
			if(a1==a2)
			{
				j++;
			}
			i++;
		}
		i--;
		cout << "Case #" << m << ": " << i << endl;
	}
}
