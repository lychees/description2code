noofstudents = int(raw_input())
records = []


def processRecord(record):
    (name, marks) = record.split()
    marks = int(marks)
    records.append((name, marks))


count = 0
while count < noofstudents:
    record = raw_input()
    processRecord(record)
    count += 1

# Note: Starting with Python 2.2, sorts are guaranteed to be stable. That
# means that when multiple records have the same key, their original order
# is preserved.
a = sorted(records, key=lambda record: record[0])
b = sorted(a, key=lambda record: record[1], reverse=True)

for record in b:
    print("{} {}".format(record[0], record[1]))
