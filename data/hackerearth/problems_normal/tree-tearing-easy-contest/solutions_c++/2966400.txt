#include "bits/stdc++.h"
using namespace std;
const int N = 1e4 + 4;
int k;
int n;
int x;
int ans = 0;
vector < int > v[N];
int dfs(int node){
	int res = 1;
	for(int next : v[node]){
		res += dfs(next);
	}
		if(res >= k){
			++ans;
			return 0;
		}
	return res;
}
int main(){
	cin >> k;
	cin >> n;
	for(int i = 2 ; i <= n ; ++i){
		cin >> x;
		v[x].emplace_back(i);
	}
	dfs(1);
	cout << ans;
}
