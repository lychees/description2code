#include <iostream>
#include <string>
using namespace std;

string getReverseStr(string str)
{
	string ans = str;
	int length = str.length();
	for (int i = 0; i < length/2; i++)
	{
		char temp = ans[i];
		ans[i] = ans[length - i - 1];
		ans[length - i - 1] = temp;
	}
	return ans;
}

void havePassword(string *list, int n)
{
	for (int i = 0; i < n - 1; i++)
	{
		string reverseStr = getReverseStr(list[i]);
		for (int j = i + 1; j < n; j++)
			if (reverseStr == list[j])
			{
				int length = list[i].length();
				cout << length << " " << list[i][length/2];
			}
	}
}

int main()
{
	int n;
	cin >> n;
	string *list = new string[n];
	for (int i = 0; i < n; i++)
		cin >> list[i];
	havePassword(list, n);
	return 0;
}
