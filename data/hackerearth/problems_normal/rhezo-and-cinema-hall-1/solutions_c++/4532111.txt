#include<bits/stdc++.h>
using namespace std;
#define ff first
#define ss second
#define pb push_back
#define mp make_pair
#define all(x) x.begin(),x.end()
#define sz(x) ((int)x.size())
const int MOD = 1e9+7;
typedef long long ll;
typedef pair<int,int> pii;
vector<int> v;
vector<int> siz;
int POWER[20];
int k;
const int MAXN = 503;
ll dp[2][MAXN][MAXN];
ll go(int lft, int up, bool f) {
	if(lft==0 and f) return 1;
	else if(lft==0) return 0;
	ll ret=0;
	if(dp[f][lft][up]!=-1) return dp[f][lft][up];
	for(int i=0;i<sz(v);i++) {
		ret+=go(lft-1,min(k,up+siz[i]),(f|(up+siz[i]>=k)));
		if(ret>=MOD) ret-=MOD;
	}
	return dp[f][lft][up]=ret;
}
int main() {
	// freopen("TASK.in","r",stdin);	
	// freopen("TASK.out","w",stdout);
	int n,m;
	cin>>n>>m>>k;
	POWER[0]=1;
	for(int i=1;i<15;i++) POWER[i]=POWER[i-1]<<1;
	for(int i=0;i<POWER[m];i++) {
		bool f=true;
		int previ=-2;
		int c=0;
		for(int j=0;j<m;j++) {
			if(POWER[j]&i) {
				c++;
				if(previ==j-1) f=false;
				else previ=j;
			}
		}
		if(f and c>=2) v.pb(i),siz.pb(c);
	}
	memset(dp,-1,sizeof(dp));
	cout<<go(n,0,false)<<endl;
	return 0;
}
