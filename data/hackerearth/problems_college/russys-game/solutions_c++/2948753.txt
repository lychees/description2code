#include <iostream>
using namespace std;
long long a[100000],b[100000];
int main()
{
    long long t,n,i;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;++i)
    	{b[i]=0;
    		cin>>a[i];
    	}
    	for(i=0;i<n;)
    	{
    		if(a[i]==-1)
    		{
    			cout<<"Win\n";
    			break;
    		}
    		if(b[i]==0)
    		{b[i]=1;
    			i=a[i];
    		}
    		else
    		{cout<<"Lose\n";
    			break;
    		}
    	}
    }
    return 0;
}
