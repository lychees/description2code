#include <iostream>
#include <sstream>
#include <string>

std::string convert(int num) {
	std::stringstream ss;
	ss << num;
	return ss.str();
}

int main(void) {
    
    int T = 0;
    std::cin >> T;
    while(T--) {
    	int a = 0, b = 0, c = 0, d = 0, s = 0;
    	std::cin >> a >> b >> c >> d >> s;
    	int res = c - s;
    	if (res > b) {
    		std::cout << b << std::endl;
    	} else {
    		std::cout << ((res >= a && res <= c) ? convert(res) : "STAY IN LONDON") << std::endl;
    	}
    }
    return 0;
}
