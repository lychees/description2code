#include <bits/stdc++.h>
using namespace std;
int main()
{
    int N, M;
    cin >> N >> M;
    int a[N], b[M];
    long long int sum=0, sum1=0; 
    for(int i=0; i<N; i++){
    	cin >> a[i];
    	sum += a[i];
    }
	for(int i=0; i<M; i++){
    	cin >> b[i];
    	sum1 += b[i];
    }
    cout << sum*sum1 << endl;
	return 0;
}
