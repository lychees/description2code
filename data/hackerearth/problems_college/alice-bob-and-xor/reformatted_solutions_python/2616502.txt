k = input()


def funxor(n):
    r = n % 4
    if (r == 0):
        return n
    if (r == 1):
        return 1
    if (r == 2):
        return (n+1)
    if (r == 3):
        return 0


while (k > 0):
    l, r = map(int, raw_input().split())
    xl = funxor(l-1)
    xr = funxor(r)
    res = xl ^ xr
    print(res)
    k = k-1
