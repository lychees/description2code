#include <iostream>
#include <cstdio>
using namespace std;
long long n,v[100];
long long arr[100][3];
bool dp(long long a,long long b,long long c)
{
	long long i;

	if(a==0 && b==0 && c==0)
	return true;
	if(a<0||b<0||c<0)
	return false;
	for(i=0;i<n;i++)
	{
		if(v[i]==0 && arr[i][0]<=a && arr[i][1]<=b && arr[i][2]<=c)
		{
			v[i]=1;
			if(dp(a-arr[i][0],b-arr[i][1],c-arr[i][2])==true)
			return true;
			v[i]=0;
		}
	}
	return false;
}

int main()
{
    long long i,j,k,m,l,t,a,b,c;
    scanf("%lld",&t);
    
    for(l=1;l<=t;l++)
    {
    	for(i=0;i<25;i++)
    	v[i]=0;
    	scanf("%lld %lld %lld %lld",&a,&b,&c,&n);
    	
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld %lld %lld",&arr[i][0],&arr[i][1],&arr[i][2]);
    	}
    	if(dp(a,b,c)==true)
    	printf("Case #%lld: yes\n",l);
    	else
    	printf("Case #%lld: no\n",l);
    }
    return 0;
}
