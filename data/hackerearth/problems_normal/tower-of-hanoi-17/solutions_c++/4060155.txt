#include<bits/stdc++.h>
using namespace std ;
vector < pair < long long , long long > > v ;
int main () {
    int t , n ;
    long long ri , hi;
    scanf ( "%d" , &t) ;
    while ( t-- ) {
        int i ;
        long long dp[202] = {0} ;
        scanf ( "%d" ,&n ) ;
        for ( i = 0 ; i < n ; i++ ) {
            scanf ( "%lld %lld" , &ri , &hi ) ;
            v.push_back(make_pair ( hi ,ri ) ) ;
        }
        int j,l = v.size() ;
        sort ( v.begin() , v.end() ) ;
        dp[0] = v[0].first ;
        long long ans = 0 ;
        for ( i = 1 ; i < l ; i++ ) {
            for ( j = 0 ; j < i ; j++ ) {
                if ( v[i].first > v[j].first && v[i].second > v[j].second ) {
                    dp[i] = max ( dp[i] , dp[j] + v[i].first ) ;
                }
                else {
                    dp[i] = max ( dp[i] , v[i].first ) ;
                }
            }
            ans = max ( ans , dp[i] ) ;
        }
        cout << ans << endl ;
        v.clear() ;
    }
    return 0 ;
}
