#include<iostream>
#include<string>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {   string s;
        cin >> s;

        int cnt[26]={0};
        int i=0;
        while(i < s.length())
        {
            cnt[s[i]-65]++;
            i++;
        }

        /*for(int k=0;k<26;k++)
            cout<<cnt[k]<<"\n";*/

        for(int k=0;k<26;k++)
        {
            for(int j= 0 ; j < cnt[k] ; j++)
            {
                cout<<(char)(k+65);
            }
        }

        cout<<"\n";

    }
}
