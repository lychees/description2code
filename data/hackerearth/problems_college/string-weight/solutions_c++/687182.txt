#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	
	long long int T;
	cin>>T;
	while(T--)
	{
		int table[300]={0};
		char w[15000],miss;
		cin>>w;
		int n;
		cin>>n;
		long long int weight=0;
		for(int i=0;i<n;i++)
		{
			cin>>miss;
			table[miss]++;
		}
		for(int i=0;i<strlen(w);i++)
		{
			if(table[w[i]]==0)
			weight+=(w[i]-96);
		}
		cout<<weight<<endl;
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
