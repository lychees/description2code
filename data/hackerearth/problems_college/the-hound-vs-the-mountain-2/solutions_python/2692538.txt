def stringToNum(s):
	ans = 0
	idx = 7
	for bit in s:
		if bit == '1':
			ans = ans + 2**idx
		idx = idx - 1
	
	return ans
	
t = int(raw_input())

while t > 0:
	t = t-1
	args = raw_input().split()
	
	num1 = 0
	num2 = 0
	
	num1 = stringToNum(args[0])
	num2 = stringToNum(args[1])
	
	if num1**num2 <= num2**num1:
		print "The Hound"
	else:
		print "The Mountain"
