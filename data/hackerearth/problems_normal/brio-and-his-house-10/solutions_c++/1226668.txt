#include <iostream>
using namespace std;

int main()
{
	int tc;
	double a,b,c,mul,base,ans;
    scanf("%d",&tc);
    while(tc--)
    {
    	scanf("%lf%lf%lf",&a,&b,&c);
    	mul=a*b*c;
    	base=(a+b+c)*(b+c-a)*(c+a-b)*(a+b-c);
    	ans=(mul*mul)/base;
    	ans=3.1415*ans;
    	printf("%.4lf\n",ans);
    }
    return 0;
}
