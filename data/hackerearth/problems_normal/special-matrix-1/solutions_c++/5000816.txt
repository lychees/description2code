#include <iostream>
#include <cmath>
using namespace std;

int main()
{
   int t;

   cin>>t;

   while(t--)
   {
       int n,l,c,x,y,rez,t=0;

       cin>>n;
       
       char a[n][n];

       for(int i=0;i<n;i++)
       {
           for(int j=0;j<n;j++)
           {
               cin>>a[i][j];
           }
       }

       l=n/2;
       c=l;

       for(int i=0;i<n && t==0;i++)
       {
           for(int j=0;j<n && t==0;j++)
           {
               if(a[i][j]=='*')
               {
                   x=abs(l-i);
                   y=abs(c-j);
               }
           }
       }

       rez=abs(x+y);

       cout<<rez<<"\n";

   }

    return 0;
}
