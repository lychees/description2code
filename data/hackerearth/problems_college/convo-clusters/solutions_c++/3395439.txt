bool isvowel(char t) {

	

	t = tolower(t);
	if(t == 'a' ||
		t == 'e' ||
		t == 'i' ||
		t == 'o' ||
		t == 'u') {
		return true;
	}
	return false;
}

bool iscons(char t)
{
	if(isvowel(t))
		return false;
	return true;
}
