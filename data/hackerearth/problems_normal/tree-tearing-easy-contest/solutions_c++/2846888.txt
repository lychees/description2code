using namespace std;

int k,n;
vector<int> g[N];
int dp[N];
int ans;

void dfs(int v)
{
	int cnt=1;
	
	for (int i=0;i<g[v].size();i++)
	{
		int to=g[v][i];
		dfs(to);
		cnt+=dp[to];
	}
	if (cnt>=k)
	{
		dp[v]=0;
		++ans;
	}
	else
	dp[v]=cnt;
}
