#include <iostream>
#include <vector>

using namespace std;

int main()
{
	int N;
	cin >> N;
	vector<long long int> numbers(0, N);
	int num;
	for(int i = 0; i < N; ++i) {
		cin >> num;
		numbers.push_back(num);
	}
	// for(int i = 0; i < N; ++i) {
	// 	cout << numbers[i] << " ";
	// }
	long long int result = 0;
	for (int i = 0; i < N; ++i) {
		result |= numbers[i];
	}
	
	cout << result;
	
}
