#include <iostream>
#include<math.h>
using namespace std;

int prime(int n)
{
	for(int i=2;i<(int)sqrt(n)+1;i++)
		if(n%i == 0)
			return i;
	return 0;
}

int main()
{
    int a[1000001]={0};
    for(int i=3;i<1000001;i++)
    	a[i] = prime(i) + a[i-1];
    int t;
    cin>>t;
    while(t--)
    {
    	int l,r;
    	cin>>l>>r;
    	cout<<a[r]-a[l-1]<<endl;
    }
    return 0;
}
