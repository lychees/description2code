#include <cstdio>
#include <iostream>
#include <algorithm>

using namespace std;

#define s(a) scanf("%d",&a)
#define p(a) printf("%d",a)
#define sp   printf(" ")
#define nl   printf("\n")

typedef long long ll;

int main()
{
	int T;
	scanf("%d",&T);
	while(T--) {
		int n;
		s(n);
		int arr[n];
		for(int i=0;i<n;i++) s(arr[i]);
		ll cnt1=0ll,cnt2=0ll,fcnt=0ll;
		for(int i=n-1;i>=0;i--) {
			if(arr[i]%2) {fcnt+=cnt2;cnt1++;}
			else {fcnt+=cnt1;cnt2++;}
		}
		cout<<fcnt;nl;
	}
	return 0;
}
