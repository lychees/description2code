#include <bits/stdc++.h>
#define ll long long int
#define mod (int)(1e9 + 7)
using namespace std;

int t, n;
int a[100005];

int main()
{
	cin >> t;
	while (t--) {
		cin >> n;
		
		map<int, int> mp;
		
		for (int i = 0; i < n; i++) {
			cin >> a[i];
		}
		int l = 0, r = 1;
		int li = 0, ri = 1;
		int mx = 1, la = 0, ra = 0;
		mp[a[0]]++;
		while (r < n) {
			if (mp.count(a[r]) > 0) {
				if (r - l > mx) {
					mx = (r - l);
					li = l;
					ri = r;
				}
				mp[a[r]]--;
				for (int i = l; i < r; i++) {
					if (a[i] == a[r]) {
						l = i + 1;
						break;
					}
				}
				mp[a[r]]++;
				r++;
			} else {
				mp[a[r]]++;
				r++;
				
			}
		}
		if (r - l > mx) {
			li = l;
			ri = r;
		}
		cout << li + 1 << " " << ri << endl;
	} 
	
	return 0;
}
