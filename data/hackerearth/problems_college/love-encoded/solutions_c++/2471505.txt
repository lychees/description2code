#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    int t , n , val;
    char a[100];
    scanf("%d" , &t);
    while(t--)
    {
    	scanf("%d" , &n);
    	scanf("%s" , &a);
    	for(int i = 0 ; a[i] != '\0' ; i++)
    	{
    		if( ((int)a[i]-n) < 65  )
    			val = (int)a[i] - n + 26;
    		else
    			val = ((int)a[i]-n);
    		printf("%c" , val);
    	}
    	printf("\n");
    }
    return 0;
}
