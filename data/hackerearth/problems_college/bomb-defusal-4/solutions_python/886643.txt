T = int(raw_input())
f = dict()
MAXVAL = 10000
f[(0,0)] = 0
f[(0,1)] = MAXVAL
f[(1,0)] = MAXVAL
def solve(a, b):
	if (a,b) in f:
		return f[(a,b)]
	else:
		step = (a + b) / 2
		f[(a,b)] = solve(abs(a-step), abs(b-step)) + 1
		f[(b,a)] = f[(a,b)]
		return f[(a,b)]
		
for i in range(T):
	P, x, y, u, v = [int(_) for _ in raw_input().split()]
	a = abs(x - u)
	b = abs(y - v)
	if solve(a, b) < P:
		print 'Counter-Terrorists Win !'
	else:
		print 'Terrorists Win !'
