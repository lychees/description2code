#include <iostream>
using namespace std;

int main()
{
    int arr[8],r=1,g=1,b=1,rt,gt,bt,t;
    for(int i=0;i<8;i++)
    arr[i]=0;
    cin>>t>>rt>>gt>>bt;
    for(int i=0;i<t;i++)
    {
    	if(i%rt==0)
    	{
    		if(r==1)
    		r=0;
    		else
    		r=1;
    	}
    	
    	if(i%gt==0)
    	{
    		if(g==1)
    		g=0;
    		else
    		g=1;
    	}
    	
    	if(i%bt==0)
    	{
    		if(b==1)
    		b=0;
    		else
    		b=1;
    	}
    	if(r==0 && g==0 && b==0)
    	arr[7]++;
    	if(r==0 && g==0 && b==1)
    	arr[2]++;
    	if(r==0 && g==1 && b==0)
    	arr[1]++;
    	if(r==0 && g==1 && b==1)
    	arr[4]++;
    	if(r==1 && g==0 && b==0)
    	arr[0]++;
    	if(r==1 && g==0 && b==1)
    	arr[5]++;
    	if(r==1 && g==1 && b==0)
    	arr[3]++;
    	if(r==1 && g==1 && b==1)
    	arr[6]++;
    }
    for(int m=0;m<8;m++)
    cout<<arr[m]<<" ";
    return 0;
}
