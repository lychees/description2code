#include<iostream>
#include<math.h>
#include<algorithm>
#include<vector>
#define max 100001

using namespace std;

struct data1{
	int diff;
	int num;
};


bool wayToSort(const data1 &i,const data1 &j){
	return i.diff<j.diff;
}

bool wayToSort2(const data1 &i,const data1 &j){
	return i.num<j.num;
}

int gcd(long int a,long int b){
	long int t;
	while(b!=0){
		t=b;
		b=a%b;
		a=t;
	}
	return a;
}



int main(){
    int test;
    cin>>test;
    while(test--){
    	long int n,i;
    	cin>>n;
    	struct data1 data[n];
    	long int arr[n];
    	for(i=0;i<n;i++){
    		cin>>data[i].num;
    	    data[i].diff=max;
        }
    		
    	long int p;
    	int q;
    	cin>>p;cin>>q;
    	
		
    		
    	long int count=0;	
    	for(i=0;i<n;i++){
    	
    		if(gcd(data[i].num,p)>1){
    			data[i].diff=abs(data[i].num-p);
    			count++;
    	    }
        }
        
        if(count<q){
        	cout<<-1<<endl;
        	continue;
        }
        
        else{
			sort(data,data+n,wayToSort);
			
			sort(data,data+q,wayToSort2);
			
	   
	    
	    for(i=0;i<q;i++){
	    	cout<<data[i].num<<" ";
	    	
	    }
	    
	    cout<<endl;
		}
	}
    
  return 0; 
}
