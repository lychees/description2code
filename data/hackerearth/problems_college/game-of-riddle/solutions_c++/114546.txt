#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
	int a[10],n;
	//freopen("in.txt","r",stdin);
    int t;
    scanf("%d",&t);
    while(t--)
    {
    		scanf("%d",&n);
    		for(int i=1;i<=9;i++)
    			scanf("%d",&a[i]);
    		
    		//find min
    		int min=a[9],ind=9,flag=0,dig;
    		for(int i=8;i>=1;i--)
    			if(a[i]<min)
    			{
    				min=a[i];
    				ind=i;
    			}
    		
    		if(n<min)
    		{
    			printf("-1\n");
    		}
    		else
    		{
    			int temp=n%min;
    			int dig=n/min;
    			int f=0;
    			while(temp>0 && f<dig && n>0)
    			{
    				for(int i=9;i>0;i--)
    				{
    					if(a[i]-min<=temp)
    					{
    						n-=a[i];
    						printf("%d",i);
    						break;
    					}
    				}
    				temp=n%min;
    				f++;
    			}
    			while(n>0 && f<dig)
    			{
    				printf("%d",ind);
    				n-=min;
    			}
    			printf("\n");
    		}
    		
    		
    }
    return 0;
    
}
