#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

int main()
{
    int t,x,y,count;
    
    scanf("%d", &t);
    
    for (int i = 0; i < t; i++) {
    	scanf("%d%d", &x, &y);
    	
    	bool a[1000001] = {0};
    	
    	for (int j = 2; j <= sqrt(1000000000); j++) {
    		int c = j * j;
    		int s = x + (c - x % c) % c;
    		
    		for (int k = s; k <= y; k += c) {
    			a[k-x] = 1;
    		}
    	}
    	
    	count = 0;
    	
    	for (int j = x; j <= y; j++) {
    		if (a[j-x] == 0) {
    			count++;
    		}
    	}
    	
    	printf("%d\n", count);
    }
    
    
    return 0;
}
