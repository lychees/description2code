#include<iostream>
#include<cmath>
using namespace std;

int f(int s,int f1,int c){
    if(s*c>=f1)
        return 0;
    
    int max1=s*sqrt((f1/s));
    if(max1<s*c)
        max1=s*c;
    return (1+f(max1,f1,c));
}

int main(){
    int t;
    cin>>t;
    for(int i=0;i<t;i++){
        int s,f1,c;
        cin>>s>>f1>>c;
        cout<<"Case #"<<i+1<<": "<<f(s,f1,c)<<endl;
    }
}
