#include <bits/stdc++.h>
using namespace std;
#define gc getchar
int input() {
  char c = gc();
  while(c<'0' || c>'9') c = gc();
  int ret = 0;
  while(c>='0' && c<='9') {
    ret = 10 * ret + c - 48;
    c = gc();
  }
  return ret;
}
int main()
{
    int t,n,m;
    t=input();
    while(t--){
    	n=input();
    	m=input();
    	for(int i=0;i<n;i++){
    		double min=1,sprice;
    		int pos=-1,dis1,dis2,dis3;
    		for(int j=1;j<=m;j++){
    			dis1=input(); dis2=input(); dis3=input();
    			sprice=(double)(100-dis1)*(double)(100-dis2)*(double)(100-dis3)/(double)1000000;
    			if(min>sprice){
    				min=sprice;
    				pos=j;
    			}
    		}
    		printf("%d ",pos);
    	}
    	printf("\n");
    }
    return 0;
}
