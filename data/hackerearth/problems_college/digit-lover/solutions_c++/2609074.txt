#include <iostream>
using namespace std;

int digitSum(int x){
	int s=0;
	while(x>0){
		s+=(x%10);
		x/=10;
	}	
	
	if(s>=10){
		return digitSum(s);
	}else{
		return s;	
	}
}

int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	int sum = 0;
    	
    	sum = digitSum(n);
    	cout<<sum<<endl;
    }
    return 0;
}
