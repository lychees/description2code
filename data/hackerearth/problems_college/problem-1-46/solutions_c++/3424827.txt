#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const ll N = 100100;

ll cc[N][26];

ll solve(ll x, ll l, ll r) {
	if (l == 0)
	    return cc[r][x];
	return cc[r][x] - cc[l - 1][x];
}

int main()
{
   ios::sync_with_stdio(false);
   ll n;
   cin >> n;
   string s;
   cin >> s;
   for (ll i = 0; i < s.length(); i++) {
   	    cc[i][ s[i] - 'a' ] = 1;
   }
   for (ll i = 1; i < s.length(); i++) {
   	    for (ll j = 0; j < 26; j++) {
   	    	 cc[i][j] += cc[i - 1][j];
   	    }
   }

   ll q;
   cin >> q;
   while (q--) {
   	      ll l, r;
   	      cin >> l >> r;
   	      ll odd = 0,  len = r - l + 1;
   	      for (ll i = 0; i < 26; i++) {
   	      	   ll C = solve(i, l - 1, r - 1);
   	      	   if (C % 2) odd++;
   	      	   
   	      }
   	      if (len % 2) {
   	      	  if (odd == 1 && len > 1)
   	      	      cout << "YES" << endl;
   	      	  else
   	      	      cout << "NO" << endl;
   	      } else {
   	      	  if (odd <= 2 && len > 0)
   	      	      cout << "YES" << endl;
   	      	  else
   	      	      cout << "NO" << endl;
   	      }
   }
   return 0;
}
