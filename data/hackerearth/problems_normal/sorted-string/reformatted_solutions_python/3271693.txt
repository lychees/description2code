t = input()

while t:

    s = raw_input()
    a = {}
    for i in range(len(s)):
        a[s[i]] = a.get(s[i], 0) + 1
    # print a
    l = []
    for k, v in a.items():
        l.append((k, v))

    l = sorted(l, key=lambda x: x[0])
    l = sorted(l, key=lambda x: x[1])
    str = ""
    for i in l:
        str += i[0]*i[1]
    print str
    t -= 1
