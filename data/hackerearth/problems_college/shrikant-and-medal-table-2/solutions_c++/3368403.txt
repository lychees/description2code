#include<bits/stdc++.h>
using namespace std;
struct medals{
	int gold;
	int bronze;
	int silver;
	string name;
};
int cmpfunc(struct medals  a,struct medals b)
{
	if(a.gold > b.gold)
	{
		return a.gold;
	}
	else if(a.gold == b.gold && a.silver > b.silver)
	{
		return a.silver;
	}
	else if(a.gold == b.gold && a.silver == b.silver && a.bronze > b.bronze)
	{
		return a.bronze;
	}
	else if(a.gold == b.gold && a.silver == b.silver && a.bronze == b.bronze)
	{
		if(a.name > b.name)
		{
			return 0;
		}
		else
		{
			return 1;
		}
	}
	return 0;
}
int main()
{
	map<string , medals>map1;
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		string  str1,str2,str3;
		cin>>str1>>str2>>str3;
		map1[str1].gold++;
		map1[str2].silver++;
		map1[str3].bronze++;
	}
	struct medals player[101];
	map<string , medals>::iterator it;
	int i=0;
	for(it = map1.begin() ; it != map1.end() ; it++)
	{
		player[i].name = it->first;
		player[i].gold = it->second.gold;
		player[i].silver = it->second.silver;
		player[i].bronze = it->second.bronze;
		i++;
	}
	sort(player,player+i,cmpfunc);
	int j;
	for(j=0;j<i;j++)
	{
		cout<<player[j].name <<" "<<player[j].gold<<" "<<player[j].silver<<" "<<player[j].bronze<<"\n";
	}
}
