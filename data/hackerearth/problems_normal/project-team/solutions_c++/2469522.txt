#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++)cin>>a[i];
    	sort(a,a+n);
    	int mn = INT_MAX;
    	int mx = INT_MIN;
    	for(int i=0;i<n/2;i++)
    	{
    		int temp = a[i]+a[n-i-1];
    		mn = min(mn,temp);
    		mx = max(mx,temp);
    	}
    	cout<<mx-mn<<endl;
    }
    return 0;
}
