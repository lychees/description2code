#include <iostream>
#include <string>
using namespace std;

int main()
{
	int T;
	bool alphabet[26];
	string temp, out;
	cin >> T;
	//T++;
	while(T--)
	{
		cin >> temp;
		out = "";
	
		for (int i = 0; i < 26; i++) 
		{
			alphabet[i] = 0;
		}
	
		for (int i = 0; i < temp.length(); i++)
		{
			if(!alphabet[temp[i]-97])
			{
				alphabet[temp[i]-97] = 1;
				out.push_back(temp[i]);
			}
		}
		cout << out << endl;
	}
    return 0;
}
