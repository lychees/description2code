#include<bits/stdc++.h>

using namespace std;
long long arr[10005],dp[10005];
int n;

long long solve()
{


    long long low=0,high=10e9;
    long long mid;
    long long mini=LONG_MAX;
    while(low<high)
    {
        for(int i=0;i<n;i++)
            dp[i]=arr[i];
        mid=(low+high)/2;
        dp[0]=max(1LL,dp[0]-mid);
        int flag=0;
        for(int i=1;i<n;i++)
        {
            if(dp[i]+mid >=dp[i-1]+1 && dp[i]-mid <= dp[i-1]+1)
            dp[i]=dp[i-1]+1;
            else
            {

               if(dp[i]-mid>=dp[i-1]+1)
                    dp[i]=dp[i]-mid;
               else
                dp[i]+=mid;




            }



        }
        flag=0;
          for(int i=1;i<n;i++)
            {
                if(dp[i]<=dp[i-1])
                    flag=1;
            }
            if(flag!=1)
            {


                high=mid;
                mini=min(mini,mid);

            }
            else
            {


                low=mid+1;
                //cout<<high<<" "<<low<<"\n";
               // mini=min(mini,mid);
            }
    }
    return  mini;

}
int main()
{
   // freopen("input.txt","r",stdin);

    int t;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(int i=0;i<n;++i)
            cin>>arr[i];
        cout<<solve()<<"\n";
    }
}
