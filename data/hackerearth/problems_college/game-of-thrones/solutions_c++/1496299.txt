#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long n;
    	scanf("%lld",&n);
    	long long a[n],c=0;
    	for(int i=0;i<n;i++)
    	scanf("%lld",&a[i]);
    	sort(a,a+n);
    	for(int i=1;i<n;i++)
    	{
    		if(a[i]==a[i-1])
    		c++; 
    	}
    	
    	printf("%lld\n",c);
    }
    return 0;
}
