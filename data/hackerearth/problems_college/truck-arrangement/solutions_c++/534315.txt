int main()
{
        int test_cases;
        scanf("%d",&test_cases);
        while(test_cases > 0)
        {
                int N,arr[10000],i=0,ans;
                vector<pair<int,int> >v;
                scanf("%d",&N);
                while(i<N)
                {
                        scanf("%d",&arr[i]);
                        v.push_back(make_pair(arr[i],i));
                        i = i+1;
                }
                
                //your code here
                
                sort(v.begin(),v.end());
                int max=0;
                for(i=0;i<N;i++)
                {

                	if(diff<0)

                	if(max<abs(diff))
                		max=diff;
                }
                cout<<max<<endl;
                test_cases--;
        }

}
