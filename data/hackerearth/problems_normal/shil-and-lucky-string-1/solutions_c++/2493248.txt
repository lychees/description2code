typedef unsigned long long int uint64;

string s;
int cnt1[30];
int cnt2[30];
int n;

void create(){
	int i;
	memset(cnt1,0,sizeof(cnt1));
	memset(cnt2,0,sizeof(cnt2));
	for(i=0;i<(n/2);i++)
	cnt1[s[i]-'a']++;
	
	for(i=n/2;i<n;i++)
	cnt2[s[i]-'a']++;
	
}

int getequ(){
	int i;
	create();
	int match=0;
	for(i=0;i<26;i++){
		match+=min(cnt1[i],cnt2[i]);
	}
	//cout<<"equ "<<match<<endl;
	return (n/2)-match;
}

int getmore(){
	create();
	int i,j,ret=0;
	ret+=cnt2[0];
	cnt2[25]+=cnt2[0];
	cnt2[0]=0;
	ret+=cnt1[25];
	cnt1[0]+=cnt1[25];
	cnt1[25]=0;
	for(i=0;i<26;i++){
		while(cnt1[i]){
			
			for(j=i+1;j<26;j++){
				if(cnt2[j]){
					cnt2[j]--;
					cnt1[i]--;
					break;
				}
			}
			
			if(j==26)
			break;
		}
	}
	
	for(i=0;i<26;i++)
	ret+=cnt1[i];
//	cout<<"Getmore "<<ret<<endl;
	return ret;
}

int getless(){
	create();
	int i,j,ret=0;
	ret+=cnt2[25];
	cnt2[0]+=cnt2[25];
	cnt2[25]=0;
	ret+=cnt1[0];
	cnt1[25]+=cnt1[0];
	cnt1[0]=0;
	for(i=0;i<26;i++){
		while(cnt1[i]){
			
			for(j=i-1;j>=0;j--){
				if(cnt2[j]){
					cnt2[j]--;
					cnt1[i]--;
					break;
				}
			}
			
			if(j==-1)
			break;
		}
	}
	
	for(i=0;i<26;i++)
	ret+=cnt1[i];
//	cout<<"Getless "<<ret<<endl;
	return ret;
}

int main(){
	int i;
	cin>>n>>s;
	
	
	int ans=1e9;
	ans=min(ans,getless());
	ans=min(ans,getmore());
	ans=min(ans,getequ());
	cout<<ans;
	return 0;
}
