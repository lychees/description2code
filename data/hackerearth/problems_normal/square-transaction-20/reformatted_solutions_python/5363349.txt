from bisect import bisect_left
t = input()
l = list()
for i in range(t):
    l.append(0)
a1 = 0
a2 = 0
for i in map(int, raw_input().split()):
    a1 = a1+i
    l[a2] = a1
    a2 += 1
q = input()
for i in range(q):
    query = input()
    z = bisect_left(l, query)+1
    if z > t:
        print -1
    else:
        print z
