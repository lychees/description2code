#include <bits/stdc++.h>
using namespace std;
const int N = 1200;
int tab[N];
int horse[N];

int main()
{
	int t;
	scanf("%d",&t);
	while( t-- )
	{
		int n;
		scanf("%d",&n);
		for (int i=0;i<n;i++) scanf("%d",tab+i);
		for (int i=0;i<n;i++) scanf("%d",horse+i);
		sort( tab+1, tab+n);
		sort( horse, horse+n);
		for (int i=0;i<n;i++)
			tab[i] *= horse[n-i-1];
		bool OK = 0;
		for (int i=1;i<n;i++)
			if ( tab[i] >= tab[0] ) OK = 1;
		if ( OK ) printf("NO\n");
		else printf("YES\n");
	}
}
