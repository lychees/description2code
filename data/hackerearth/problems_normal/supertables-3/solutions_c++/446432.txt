#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
using namespace std;
long long gcd(long long a , long long b)
{
	long long rem;
	while (b != 0)
	{
		rem = a % b;
		a = b;
		b = rem;
	}
	return a;
}
int main()
{
	long long t,a,b,n;
	cin >> t;
	while ( t-- ) {
		cin >> a >> b >> n;
		long long p = gcd(a,b);
		p = (a*b)/p;
		long long low,high,mid;
		low = 1;
		high = 100000000000000000LL;
		long long ans;
		while ( low <= high ) {
			mid = (low+high)/2;
			long long val = mid/a + mid/b - mid/p;
			if ( val == n ) {
                ans = max((mid/a)*a, (mid/b)*b);
				cout << ans << endl;
				break;
			}
			else if ( val < n ) {
				low = mid+1;
			}
			else {
				high = mid-1;
			}
		}
	}
    return 0;
}
