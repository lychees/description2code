#include <iostream>
using namespace std;

int main()
{
    int k,n;
    cin >> k;
    string ignore[k];
    for(int i = 0; i < k; i++){
    	cin >> ignore[i];
    }
    cin >> n;
    string ans = "";
    char word[100];
    for(int i = 0; i < n; i++){
    	cin >> word;
    	for(int j = 0; j < k; j++){
    		if(word == ignore[j]){
    			word[0] = '0';
    			break;
    		}
    	}
    	if(ans == "" && word[0] != '0'){
    		ans += word[0] - 'a' + 'A';
    	}else if(word[0] != '0'){
    		ans += ".";
    		ans += word[0] - 'a' + 'A';
    	}
    }
    cout << ans << endl;
    return 0;
}
