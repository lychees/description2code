#include <iostream>
#include <vector>
using namespace std;

int main()
{
	long long int T,i,j,k,N;
    vector<long long int>answer;
    cin>>T;
    for(i=0;i<T;i++)
    {
        long long int A[1001],counter=0;
        cin>>N;
        for(j=0;j<N;j++)
        {
            cin>>A[j];
        }
        for(j=0;j<N;j++)
            for(k=j+1;k<N;k++)
            if(A[j]!=A[k])
            if((A[j]+A[k])%2==0)
            counter++;
        answer.push_back(counter);
    }
    for(i=0;i<T;i++)
        cout<<answer[i]<<endl;
        return 0;
}
