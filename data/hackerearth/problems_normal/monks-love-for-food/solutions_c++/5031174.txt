#include <iostream>
#include <stack>

using namespace std;

int main(){
	stack <int> foodpack;
	int q=0;
	
	cin>> q;
	
	for(int x=0; x<q; x++){
		int q1, q2;
		cin>> q1;
		if(q1 == 2){
			cin>> q2;
			foodpack.push(q2);
		}
		else{
			if(q1 ==1 && !foodpack.empty()){
				cout<<foodpack.top()<<endl;
				foodpack.pop();
			}
			else
				cout<<"No Food\n";
		}
	}
	return 0;
}
