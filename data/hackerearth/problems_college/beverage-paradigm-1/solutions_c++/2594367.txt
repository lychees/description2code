int findOut(int arr[],int n,long long int a)
{
	if(a==0) return 1;
	if(a>0 && n<0) return 0;
	if(arr[n]>a) return findOut(arr,n-1,a);
	return (findOut(arr,n-1,a) || findOut(arr,n-1,a-arr[n]));
}

{
	int n;long long int a; cin>>n;
	int arr[n];
	for(int i=0;i<n;i++) cin>>arr[i];
	cin>>a;
	if(findOut(arr,n-1,a)) cout<<"True"; else cout<<"False";
}
