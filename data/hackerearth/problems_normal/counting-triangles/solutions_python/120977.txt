t = input();

sidesPresent = {};

count = 0;
while t > 0:
	sides = raw_input();
	x = sides.split();
	[a,b,c] = sorted(x);
	hashval = a + "|" + b + "|" + c;
	if hashval not in sidesPresent:
		count += 1;
		sidesPresent[hashval] = True;
	elif hashval in sidesPresent:
		if sidesPresent[hashval] == True:
			sidesPresent[hashval] = False;
			count -= 1;
	t -= 1;
print count;
