#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
int cs,n,a,b,c,t;
int M[701][151][51];
int solve(int a,int b,int c)
{
  //int &res=M[a][b][c];
  if(M[a][b][c]==-1)
    if(t-8*n==a+5*b+10*c)
      M[a][b][c]=0;
    else
    {
      M[a][b][c]=1000000000;
      if(a>=8)
      M[a][b][c]=min(M[a][b][c],solve(a-8,b,c)+8);
      if(b>=1 && a>=3)
        M[a][b][c]=min(M[a][b][c],solve(a-3,b-1,c)+4);
      if(b>=2)
        M[a][b][c]=min(M[a][b][c],solve(a+2,b-2,c)+2);
      if(c)
        M[a][b][c]=min(M[a][b][c],solve(a+2,b,c-1)+1);
      if(a>=3 && c>=1)
        M[a][b][c]=min(M[a][b][c],solve(a-3,b+1,c-1)+4);
    }
  return M[a][b][c];
}
int main()
{
  scanf("%d",&cs);
  while(cs--)
  {
    scanf("%d%d%d%d",&n,&a,&b,&c);
    t=a+5*b+10*c;
    if(t<8*n)
    {
      printf("-1\n");
      continue;
    }
    memset(M,-1,sizeof M);
    printf("%d\n",solve(a,b,c));
  }
  return 0;
}
