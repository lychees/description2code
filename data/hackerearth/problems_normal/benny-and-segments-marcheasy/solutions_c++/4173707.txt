#include<bits/stdc++.h>
using namespace std;
struct seg
{
    int xl;
    int xr;
};
struct comp
{
    bool operator()(const seg &s1,const seg &s2)
    {
        if(s1.xr==s2.xr)
            return s1.xl<s2.xl;
        return s1.xr<s2.xr;
    }
};
int main()
{
    int t;
    int n,l;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&l);
        seg line[n];
        for(int i=0;i<n;i++)
        {
            scanf("%d%d",&line[i].xl,&line[i].xr);
        }
        sort(line,line+n,comp());
        /*for(int i=0;i<n;i++)
        {
            printf("%d %d \n",line[i].xl,line[i].xr);
        }*/
        //printf("\n\n");
        int c;
        bool possible=false;
        int currleft,currright;
        for(int i=0;i<n;i++)
        {
            c=0;
 
            for(int j=i;j<n;j++)
            {
                if(c==0)
                {
                    currleft=line[j].xl;
                    currright=line[j].xr;
                    c=line[j].xr-line[j].xl;
                }
                else if(line[j].xl<=currright)
                {
                    //printf("%d %d\n",line[j].xl,currright);
                    if(line[j].xl>currleft)
                    {
                        currright=line[j].xr;
                    }
                        c=currright-currleft;
                }
                else if(line[j].xl>line[j-1].xr)
                    break;
                //printf("xl=%d xr=%d c=%d\n",currleft,currright,c);
                if(c==l)
                {
                    possible=true;
                    break;
                }
 
            }
            if(possible)
                break;
        }
        if(possible)
            printf("Yes\n");
        else
            printf("No\n");
    }
    return 0;
}
