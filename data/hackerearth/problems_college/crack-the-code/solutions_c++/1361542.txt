#include<bits/stdc++.h>
#define ll long long
#define ld double
#define f first
#define s second
#define pb push_back
#define M 1000000007
#define MX 1000003
#define INF 10000000000
#define PI 3.141593
#define mmx(a,b) a>b?a:b
#define mmn(a,b) a<b?a:b
using namespace std;
typedef pair<ll,ll> ii;
typedef vector<ll> vv;
ll arr[100];
ll pre()
{
    arr[0]=1;
    arr[1]=1;
    for(ll i=2;;i++)
    {
        arr[i]=arr[i-1]+arr[i-2];
        if(arr[i]>1000000)
            return i;
    }
}
int main()
{
   ll mx=pre();
   ll t;
   cin>>t;
   while(t--)
   {
       ll x,y,cnt=0;
       cin>>x>>y;
       for(ll i=0;i<mx;i++)
       {
           if(arr[i]>=x&&arr[i]<=y)
            cnt++;
       }
       if(cnt&1)
        cout<<"INVALID\n";
       else cout<<"VALID\n";
   }

}
