#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#define MOD 1000000007
#define LL long long
using namespace std;

LL power(LL base, LL exp)
{
    LL result=1;
    base%=MOD;
    while(exp>0)
    {
        if(exp&1) result=(result*base)%MOD;
        base=(base*base)%MOD;
        exp>>=1;
    }
    return result;
}
int R[15];
LL func(int n)
{
    if(n<=10) return R[n];
    LL ans=power(2,ceil(n*1.0/4));
    ans=(ans*func(ceil(n*1.0/5)))%MOD;
    ans=(ans*func(ceil(n*1.0/10)))%MOD;
    return ans;
}

int main() {
    int t,n;
    R[0]=1;
    R[1]=1;
    R[2]=1;
    R[3]=3;
    R[4]=3;
    R[5]=15;
    R[6]=15;
    R[7]=105;
    R[8]=105;
    R[9]=105;
    R[10]=105;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        LL val=func(n);
        //printf("%lld\n",val);
        LL x,y,ans=0;
        for(int i=1;i<5;++i)
        {
            x=i*val;y=5*val;
            ans=max(ans,power(x,y));
            ans=max(ans,power(y,x));
        }
        for(int i=1;i<4;++i)
        {
            if(i==2) continue;
            x=i*val;y=4*val;
            ans=max(ans,power(x,y));
            ans=max(ans,power(y,x));
        }
        for(int i=1;i<3;++i)
        {
            x=i*val;y=3*val;
            ans=max(ans,power(x,y));
            ans=max(ans,power(y,x));
        }
        x=val,y=2*val;
        ans=max(ans,power(x,y));
        ans=max(ans,power(y,x));
        x=val,y=val;
        ans=max(ans,power(x,y));
        printf("%lld\n",ans);
    }
    return 0;
}
