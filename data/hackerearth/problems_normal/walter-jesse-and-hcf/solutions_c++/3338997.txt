#include<bits/stdc++.h>
#define ll long long int
#define MAX 10000009
#define MOD 1000000007
using namespace std;

vector<ll>v;

void factor(ll N)
{
    ll i;
    v.clear();
    for(i=1; i*i<=N; i++)
    {
        if(N%i==0)
        {
            v.push_back(i);
            if(N/i!=i)
                v.push_back(N/i);
        }
    }
}

int main()
{
    ll n,a,i,j,b,c;
    while(scanf("%lld",&n)!=EOF)
    {
        for(i=1; i<=n; i++)
        {
            scanf("%lld %lld %lld",&a,&b,&c);
            a=__gcd(a,b);

            factor(a);
            sort(v.rbegin(),v.rend());
            if(c>v.size())
            {
                printf("No crymeth today\n");
                continue;
            }

            printf("%lld\n",v[c-1]);
        }
        return 0;
    }
}
