#include<iostream>
using namespace std;
int main()
{ int t,j,z;
  cin>>t;
int *b=new int[t];
for(int i=0;i<t;i++)
 { cin>>b[i];
 }
int *a=new int[t];
for(int i=0;i<t;i++)
 {z=0; 
  if(i==0)
    { a[i]=b[i]*1;}
     else
      {  for(j=(i-1);j>=0;j--)
         { z=a[j]+z;}
        a[i]=((i+1)*b[i])-z;
     
       }
}
for(int i=0;i<t;i++)
 { cout<<a[i]<<" ";}
return 0;
}
