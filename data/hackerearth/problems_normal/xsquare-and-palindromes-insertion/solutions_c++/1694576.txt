#include <iostream>
#include<string.h>
using namespace std;
int numberChar(char *str){

int count[26]={0},c=0;

 while(*str)
 {
   count[(*str)-'a']++;
   str++;
 }
  for(int i=0;i<26;i++)
     if(count[i]%2!=0 && count[i]!=0)
          c++;
  return c;

}

int main()
{
    int t,c=0;
    cin>>t;

    char  str[1005];
    while(t--)
    {
      cin>>str;

     c= numberChar(str);
     if(c==1 || c==0) cout<<"0"<<endl;
     else cout<<c-1<<endl;

    }

    return 0;
}
