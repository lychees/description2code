#include<iostream>
#include<stdlib.h>
#include<string.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t)
	{
	char s[10002];
	int hash[26]={0};
	cin>>s;
	for(int i=0;s[i]!=0;i++)
	{
		int x=s[i]-'a';
		hash[x]++;
	}
	int max=0;
	int i=0;
	for(i=0;i<26;i++)
	{
		if(hash[i]>hash[max])
			max=i;
	}
	cout<<hash[max]<<endl;
	--t;
	}
return 0;
}
