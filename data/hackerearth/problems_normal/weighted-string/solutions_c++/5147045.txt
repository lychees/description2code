#include <bits/stdc++.h>
using namespace std;
char a[1000005];
int main()
{
   // cout << "Hello World!" << endl;
   int t,n;
   scanf("%d",&t);
   while(t--)
   {
   	scanf("%d",&n);
   	scanf("%s",&a);
   	set<int> s;
   	s.insert(0);
   	int ans = 0;
   	int sum = 0;
   	int i;
   	for(i=0;a[i];i++)
   	{
   		sum += a[i]-'a' + 1;
   		s.insert(sum);
   		if(s.find(sum-n)!=s.end())
   		ans++;
   	}
   	printf("%d\n",ans);
   }
    return 0;
}
