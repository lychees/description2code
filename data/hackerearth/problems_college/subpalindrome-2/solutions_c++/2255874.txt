inline void scanint(int &x)
{
    register int c = 0;
    x = 0;
    int flag = 0;
    for (; ((c != 45) && (c < 48 || c > 57)); c = gc());
    for (; ((c == 45) || (c > 47 && c < 58)); c = gc()) {
        if (c == 45)
            flag = 1;

        else
            x = (x<<1) + (x<<3) + c - 48;
    }

    if (flag)
        x = -x;
}

inline void writeint (int n)
{
    int N = n, rev, count = 0;
    rev = N;
    if (N == 0) { pc('0'); pc('\n'); return ;}
    while ((rev % 10) == 0) { count++; rev /= 10;} //obtain the count of the number of 0s
    rev = 0;
    while (N != 0) { rev = (rev<<3) + (rev<<1) + N % 10; N /= 10;}  //store reverse of N in rev
    while (rev != 0) { pc(rev % 10 + '0'); rev /= 10;}
    while (count--) pc('0');
}

ll dp[1005][1005];
string s;
ll rec(int i, int j)
{
    if(i > j)
        return 0;
    if(i == j)
        return 1;
    if(dp[i][j] != -1)
        return dp[i][j];

    if(s[i] != s[j])
        return dp[i][j] = rec(i + 1, j) + rec(i , j - 1) - rec(i + 1, j - 1);

    else
        return dp[i][j] = rec(i + 1, j) + rec(i, j - 1) + 1;
}
