#include<stdio.h>
int visited[1<<24]={0};
int gcd(int a,int b)
{
    if(b==0)
        return a;
    return gcd(b,a%b);
}
int main()
{
    long long int i,j,mx=(1<<23);
    for(i=2;i<=mx;i++)
    {
        if(visited[i]==0)
        {

            for(j=1;i*j<=mx;j++)
            {
                visited[i*j]++;
            }
        }
    }
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int a,b;
        scanf("%d%d",&a,&b);
        int x=gcd((1<<a)-1,(1<<b)-1);
        printf("%d\n",visited[x]);
    }
    return 0;
}
