m = input()
count = 0
while m:
    l = []
    s = raw_input()
    c = ''
    for i in s:
        if i == c:
            l.pop()
            try:
                c = l[-1]
            except BaseException:
                c = ''
        else:
            l.append(i)
            c = i

    if not l:
        count += 1
    m -= 1

print count
