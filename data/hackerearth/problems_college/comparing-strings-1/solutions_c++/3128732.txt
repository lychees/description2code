#include<bits/stdc++.h>
using namespace std;
main()
{
	int t,j;
	cin>>t;
	for(j=0;j<t;j++)
	{
		string a,b;
		int hash[200]={0};
		int i;
		cin>>a>>b;
		for(i=0;i<a.length();i++)
		{
			if(hash[a[i]]==0)
			hash[a[i]]=1;
		}
		for(i=0;i<b.length();i++)
		{
			if(hash[b[i]]==1)
			{
				cout<<"YES"<<endl;
				break;
			}
			
		}
		if(i==b.length())
		cout<<"NO"<<endl;
		
	}
	return 0;
}
