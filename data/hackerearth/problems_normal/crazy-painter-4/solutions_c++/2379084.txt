#include <iostream>
#include <cstdio>
#include <cassert>
using namespace std;
int main()
{
    long long t;
    scanf("%lld", &t);
    while (t--) {
        unsigned long long n;
        scanf("%lld", &n);
        const unsigned long long total_points = 12 * n;
        const unsigned long long freq = total_points / 26;
        const unsigned long long additional = total_points - 26 * freq;
        unsigned long long *c = new unsigned long long[27];
        for (int i = 1; i <= 26; i++) {
            scanf("%lld", &c[i]);
        }
        unsigned long long *f = new unsigned long long[27];
        for (int i = 1; i <= 26; i++) {
            f[i] = freq;
        }
        for (int i = 1; i <= additional; i++) {
            f[i]++;
        }
        unsigned long long cost = 0;
        for (int i = 1; i <= 26; i++) {
            cost += f[i] * c[i];
            cost += f[i] * (f[i] - 1) / 2;
        }
        printf("%llu\n", cost);
        delete c;
        delete f;
    }
    return 0;
}
