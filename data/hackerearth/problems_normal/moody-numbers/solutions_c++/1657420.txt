#include<cstdio>
#include<cstring>
bool reachable[1005];
int digits_sum(int N)
{
  if(N<10)return N;
  return N%10+digits_sum(N/10);
}
int main()
{
  int T,N,state;
  bool possible;
  scanf("%d",&T);
  while(T--)
    {
      scanf("%d",&N);
      memset(reachable, 0, sizeof(bool) * 1005);
      if(N==1 || N==4)
	printf("YES\n");
      else
	{
	  state=digits_sum(N);
	  reachable[state]=true;
	  possible=false;
	  int limit=20;
	  int iter=0;
	  while(true)
	    {
	      N=state*state;
	      iter++;
	      if(iter>limit)break;
	      if(N==1 || N==4)
		{
		  possible=true;
		  break;
		}
	      state=digits_sum(N);
	      if(reachable[state])break;
	      reachable[state]=true;
	    }
	  if(possible)
	    printf("YES\n");
	  else
	    printf("NO\n");
	}
    }
}
