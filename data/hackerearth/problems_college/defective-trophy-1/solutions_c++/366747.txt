using namespace std;
#include<iostream>
#include<math.h>
int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
        int n;
        cin>>n;
        if(n==0)
            cout<<"0"<<endl;
        else
        {
        double k=log(n)/log(3);
        int j=log(n)/log(3);
        if(k==j)
            cout<<k<<endl;
        else cout<<j+1<<endl;
        }
        t--;
    }
}
