#include <iostream>

using namespace std;

int main()
{
    long long int n,a,temp;
    long long int sum_even=0,sum_odd=0,count1=0;
    cin>>n;
    while(n--)
    {
        sum_even=0;sum_odd=0;count1=0;
        cin>>a;
        temp=a;
        while(a!=1)
        {
            if(a%2==0)
            {

                a=a/2;
                count1++;
            }
            else
            {

                a=3*a+1;count1++;
            }

        }
    a=temp;
        while(a!=1)
        {

            if(a%2==0)
            {
                
                sum_even = sum_even+a%count1;
                sum_even = sum_even%count1;
                a=a/2;

            }
            else
            {
                sum_odd = sum_odd+a%count1;
                sum_odd = sum_odd%count1;
                a=3*a+1;
            }

        }
        sum_odd = sum_odd+1%count1;
        sum_odd = sum_odd%count1;
        if(sum_even>sum_odd)
                cout<<"Even Rules\n";
        else
            if(sum_odd>sum_even)
                cout<<"Odd Rules\n";
        else
            cout<<"Tie\n";
    }
    return 0;
}
