#include <iostream>
#include <string>
using namespace std;

int main()
{
    int n, t;
    cin >> n;
    string s;
    for(int j=0; j<n; j++)
    {
    	cin >> s;
    	int c=0;
    	t=s.length();
    	if(t%3!=0){c=1;}
    	else
    	{
    		for(int i=0; i<s.length(); i++)
    		{
    			//cout << s[0] << "	" << s[t/3]<< "	" <<s[(2*t)/3]<< "	" <<s[i]<<endl;
    			if(i<t/3 && s[0]!=s[i]){c=1;break;}
    			if(i>=t/3 && i<(2*t)/3 && s[t/3]!=s[i]){c=1;break;}
    			if(i>=(2*t)/3 && s[(2*t)/3]!=s[i]){c=1;break;}
    		}
    	}
    	if(s[0]==s[t/2] || s[(2*t)/3]==s[0] || s[(2*t)/3]==s[t/3]){c=1;}
    	if(c==0){cout <<"OK"<<endl;}
    	else{cout <<"Not OK"<<endl;}
    }
    return 0;
}
