#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
    ll n,i,x;
    vector<ll>a;
    cin >> n;
    for(i=0; i<n; i++){
     cin >> x;
      a.push_back(x);
  }
      
    sort(a.begin(),a.begin()+n);
    ll ans = a[n-1]*a[n-2];
    ll k = a[0]*a[1];
     ans = max(ans,k);
     cout << ans << endl;
    return 0;
}
