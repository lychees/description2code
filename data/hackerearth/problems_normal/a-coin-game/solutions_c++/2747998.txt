#include <bits/stdc++.h>
using namespace std;
#define MAXN 1000006
int loser[MAXN];
void solve() {
	int d = 0;
	for (int i = 0; i < MAXN; ++i) {
		if(loser[i]) continue;		
		loser[i] = i + d;
                if(i + d < MAXN) loser[i + d] = i;
		d++;
	}
}
int main()
{
    //cout << "Hello World!" << endl;
    solve();
    int t;
    cin>>t;
    while(t--)
    {
    	int flag=0;
    	int n,m;
    	cin>>n>>m;
    if(n==loser[m])
   	 cout<<"Don't Play"<<endl;
    	 else
    	 cout<<"Play"<<endl;
    	 
    }
    return 0;
}
