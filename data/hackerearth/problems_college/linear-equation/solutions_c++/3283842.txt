#include <bits/stdc++.h>

using namespace std;

#define ll long long int

int main()
{
    ll t, i, j, k, n, sum, V;
    cin >> t;
    while(t--) {
        sum = 0;
        cin >> n;
        ll A[n+1];
        for(i =  0; i < n; i++) {
            cin >> A[i];
            sum += A[i];
        }
        V = sum / (n-1);
        for(i = 0 ;i < n; i++) {
            cout << V - A[i] << " ";
        }
        cout << endl;
    }

    return 0;
}
