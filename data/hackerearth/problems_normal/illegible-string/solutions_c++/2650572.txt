#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    string s;
    cin>>s;
    int max_len=0;
    int min_len=0;
    int i=0;
    while(i<n)
    {
    	if(s[i]!='v' && s[i]!='w')
    	{
    		
    		min_len=min_len+1;
    		max_len=max_len+1;
    		i=i+1;
    	}
    	else 
    	{
    		int n_w=0;
    		int n_v=0;
    		while(i<n && (s[i]=='v' || s[i]=='w'))
    		{
    			if(s[i]=='w') n_w=n_w+1;
    			else n_v=n_v+1;
    			i=i+1;
    			
    		}
    		max_len=max_len+n_v+(2*n_w);
    		min_len=min_len+ ((n_v+(2*n_w)+1)/2);
    	}
    }
    
    cout<<min_len<<" "<<max_len<<endl;
    return 0;
}
