#include <iostream>
#include <utility>
#include <algorithm>
#include <cstdio>
using namespace std;

int main() 
{
    
    int test;
    scanf("%d", &test);
    int m, n, x, y, goodCount=0;
    int g, h, greatCount = 0;
    
    for(int i=0; i < test; i++)
    {
        
        pair< int, int> mPair[100000], nPair[100000];
        
        scanf("%d%d", &m, &n);
        
        for(int j=0; j < m; j++)
        {
            scanf("%d%d", &x, &y);
            
            mPair[j] = make_pair(x, y);
        }
        
        for(int j=0; j < n; j++)
        {
            scanf("%d%d", &x, &y);
            nPair[j] = make_pair(x, y);
        }
        
        scanf("%d%d", &g, &h);
        
        
        sort(mPair, mPair+m);
        sort(nPair, nPair+n);
        
        int j=0;
        for(int i =0; i < m && j < n; i++)
        {
            while( mPair[i].first > nPair[j].first)
                j++;
            
            if(mPair[i].first == nPair[j].first)
            {
                if(mPair[i].second == nPair[j].second)
                    greatCount++;
                goodCount++;
            }
        }
        
        if(g == goodCount && h <= greatCount)
            printf("Great\n");
        else if( g == goodCount)
            printf("Good\n");
        else
            printf(":(\n");
    }
	// your code goes here
	return 0;
}
