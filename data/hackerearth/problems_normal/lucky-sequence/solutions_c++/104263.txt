#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int i=0;
		long long int n;
		char c[100000];
		cin>>n;
		n++;
		while(n>1)
		{
			if(n%2==0)
			c[i++]='4';

			c[i++]='7';
			n=n/2;
		}
		i--;
		while(i>=0)
	    cout<<c[i--];
		
		cout<<endl;
	}
return 0;
}
