for _ in range(input()):
    a, b = map(int, raw_input().split())
    p = []
    q = []
    for i in range(32):
        p.append(a % 2)
        a /= 2
    for i in range(32):
        q.append(b % 2)
        b /= 2

    c = 0
    for i in range(32):
        if (p[i] != q[i]):
            c += 1
    print c
