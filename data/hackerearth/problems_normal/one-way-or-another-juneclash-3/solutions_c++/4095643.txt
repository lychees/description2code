#include <bits/stdc++.h>
using namespace std;

#define DEBUG(x) 	cerr << #x << " = " << x << endl;
#define INPUT		freopen("Data.inp", "r", stdin);
#define OUTPUT		freopen("Data.out", "w", stdout);

typedef long long LL;
typedef pair<int, int> II;

int main() {
	int TC; scanf("%d", &TC);
	while (TC--) {
		int n; scanf("%d", &n);
		int c[2][2]; memset(c, 0, sizeof c);
		while (n--) {
			LL x, y, v; scanf("%lld%lld%lld", &x, &y, &v);
			c[v % 2][(x + y) % 2] = 1;
		}
		if (c[0][0] + c[0][1] == 2 || c[1][0] + c[1][1] == 2) puts("No"); else puts("Yes");
	}	
	return 0;
}
