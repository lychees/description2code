#include <bits/stdc++.h>
using namespace std;

#define mod 1000000007
#define ll long long int
vector<int> primes;
ll p[10005];
int prime()
{
	int i,j;
	p[0]=1;
	p[1]=1;
	
	for(i=2;i*i<10005;i++)
	{
		if(!p[i])
		for(j=i*2;j<10005;j=j+i)
		{
			p[j]=1;
		}
	}
	
	for(i=1;i<10005;i++)
	{
		if(!p[i])
		  primes.push_back(i);
	}
}
int main()
{
     ll t;
     prime();
    cin >> t;
    while(t--)
    {
    	ll n,i,j;
    	cin >> n;
    	ll a[n];
    	
    	for(i=0;i<n;i++)
    	{
    		cin >> a[i];
    	}
    	ll lcm = 1;
    	for(i=0;i<primes.size();i++)
    	{
    		ll ct = 0;
    		for(j=0;j<n;j++)
    		{
    			int temp=0;
    			while(a[j]%primes[i]==0)
    			{
    				a[j]=a[j]/primes[i];
    				temp++;
    				if(temp > ct)
    				{
    					ct=temp;
    					lcm=(lcm*primes[i])%mod;
    				}
    			}
    		}
    	}
    	cout << lcm << "\n";
    }
    return 0;
}
