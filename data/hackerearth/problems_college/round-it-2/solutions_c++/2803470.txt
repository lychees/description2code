#include <bits/stdc++.h>
#define ll long long int
#define size 100005
using namespace std;

int main() 
{
	ios::sync_with_stdio(false);
	ll t,n,d,b,first,second;
	cin>>t;
	while(t--)
	{
		cin>>n>>b;
		d=n/b;
		first=n-d*b;
		second=(d+1)*b-n;
		if(first>=second)
		{
			cout<<(d+1)*b<<endl;
		}
		else if(first<second)
		{
			cout<<d*b<<endl;
		}
		
	}
	return 0;
}
