#include<iostream>
using namespace std;
int main(){
	ios::sync_with_stdio(false);
	int t,n;
	long long m,min,minj;
	cin>>t;
	while(t--){
		cin>>n>>m;
		long long  a[n];
		for(int i=0;i<n;i++) cin>>a[i];
		for(int i=0;i<n;i++){
			min=a[i];
			minj=i;
			for(int j=i;j<n;j++){
				if(a[j]<min){
					  min = a[j];
					  minj = j;
				}
			}
			a[minj]=a[i];
			a[i]=min;
		}
		int k=0,l=n-1;
		while(m>0 && l>=0){
			m-=a[l];
			l--;
			k++;
		}
		if(m>0) cout<<-1<<endl;
		else cout<<k<<endl;
	}
}
