#include <iostream>
using namespace std;
int a[100009];
int main()
{
    int t,i,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int xo=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		xo=a[i]^xo;
    	}
    	int c=0;
    	for(i=0;i<n;i++)
    	{
    		if(a[i]^xo==0)
    		{
    			c++;
    		}
    	}
    	cout<<c<<endl;
    }
    return 0;
}
