#include <iostream>
#include <vector>
using namespace std;
int a[100007];
int x[100007];
int main()
{
	int n,k;
	cin>>n;
	vector<int> arr;
	int cnt=0,b=0,c=0;
	for(int i=0;i<n;i++)
	{
		cin>>k;
		arr.push_back(k);
		a[arr[i]]++;
		b=max(b,a[arr[i]]);
		if(i && arr[i]==arr[i-1])
		{
		    cnt++;
		    x[arr[i]]++;
		    c=max(c,x[arr[i]]);
		}
	}
	if((b>(n+1)/2))
	{
		cout<<-1;
		return 0;
	}
	cout<<max(c,(cnt+1)/2);
	return 0;
}
