#include<iostream>
#define max 120
using namespace std;
    int a[max][max];
void display(int n)
{ int i,j;
       for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            cout<<a[i][j]<<" ";
        }
        cout<<endl;
    }
}
void rotate(int n)
{ int i,j,first,last;
        for(i=0;i<(n/2);i++)
    {
        first=i;
        last=n-i-1;
        for(j=first;j<last;j++)
        {
            int t=a[first][j];
            a[first][j]=a[last-j+first][first];
            a[last-j+first][first]=a[last][last-j+first];
            a[last][last-j+first]=a[j][last];
            a[j][last]=t;
        }
    }
}
int main()
{   int first,last,i,j;
    int n,q;
    int angle;
    cin>>n>>q;
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            cin>>a[i][j];
        }
    }
    while(q--){

        cin>>angle;


    int d=angle/90;
    d=d%4;
    for(i=0;i<d;i++)
    {
        rotate(n);
    }
    display(n);
    cout<<endl;
    d=4-d;
    for(i=0;i<d;i++)
    {
        rotate(n);
    }
    }
    return 0;


}
