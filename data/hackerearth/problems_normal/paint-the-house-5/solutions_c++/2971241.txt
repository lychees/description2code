#include <bits/stdc++.h>
using namespace std;
#define rep(i,a,b) for(int i=a;i<=b;++i)
#define ll long long 
ll int n,d;
pair<int,pair<int,int> > p[100005];
bool mera_taur_function(pair<int,pair<int,int> >a,pair<int,pair<int,int> >b)
{
	if(a.first==b.first)
	{
		if(a.second.second==b.second.second)
			return a.second.first<b.second.first;
		return a.second.second>b.second.second;
	}
	return a.first<b.first;
}
int main()
{
    cin>>n>>d;
    rep(i,0,n-1)
    {
    	cin>>p[i].first>>p[i].second.first>>p[i].second.second;
    }
    sort(p,p+n,mera_taur_function);
    ll ltime=p[0].first,speed=0,cost=0,area=0;
    rep(i,0,n-1)
    {
    	area += 1LL*speed*(p[i].first-ltime);
    	if(area>=d)break;
    	if(p[i].second.second > speed)
    	{
    		speed = p[i].second.second;
    		cost = cost + p[i].second.first;
    	}
    	ltime = p[i].first;
    }
    cout<<cost;
    return 0;
}
