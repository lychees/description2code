#include <iostream>
using namespace std;

long long specialSum(long long );
long long foo(long long );
long long gcd(long long, long long);
int main()
{
	int tests;
	cin >> tests;

	for (int i = 0; i < tests; i++)
	{
		long long num;
		cin >> num;
		cout << num<<endl;
	}
}

long long gcd(long long a, long long b)
{
	return b == 0 ? a : gcd(b, a % b);
}

long long foo(long long n)
{
	long long ret = 0;
	for (long long i = 1; i <= n; i++)
	{
		if (gcd(n, i) == 1)
		{
			ret += 1;
		}
	}
	return ret;
}

long long specialSum(long long N)
{
	int ret = 0;
	for (long long i = 1; i <= N; i++)
	{
		if (N % i == 0)
		{
			ret += foo(i);
		}
	}
	return ret;
}
