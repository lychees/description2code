#include <bits/stdc++.h>
using namespace std;

map<int, long long> m;

long long f(int n) {
	if (m[n]) {
		return m[n];
	}
	if (n <= 2) {
		return n;
	}
	m[n] = max(n*1LL, f(n/2) + f(n/3) + f(n/4));
	return m[n];
}

int main()
{
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	int n;
	while (cin >> n) cout << f(n) << '\n';
    return 0;
}
