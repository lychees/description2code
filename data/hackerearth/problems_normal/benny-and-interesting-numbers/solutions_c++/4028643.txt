#include <bits/stdc++.h>
using namespace std;
const int lim = 20000;
int intr(long long x) {
   for (long long p = 2; p * p <= x; ++p) {
      if (x % p == 0) {
         int cnt = 0;
         while (x % p == 0) {
            x /= p;
            ++cnt;
         }
         if (cnt > 1) {
            return false;
         }
      }
   }  
   return true;
}
int main() {
   int t;
   scanf("%d", &t);
   while (t--) {
      long long x;
      scanf("%lld", &x);
      printf("%s\n", intr(x) ? "Yes" : "No");
   }
}
