#include <iostream>
using namespace std;

int main()
{
    int n;
    cin >> n;
    int a[n], psum[300005] = {0}, nsum[300005] = {0};
    for(int i = 0 ; i < n ; i++){
    	cin >> a[i];
    }
    int count = 0;
    for(int i = 0 ; i < n ; i++){
    	for(int j = 0 ; j < i ; j++){
    		int x = a[i]-a[j];
    		if(x >= 0){ if(psum[x] == 1){
    			count++;
    			break;
    		}
    		}
    		else if(nsum[-x] == 1){
    			count++;
    			break;
    		}
    	}
    	for(int j = 0 ; j <= i ; j++){
    		int x = a[i]+a[j];
    		if(x >= 0) psum[x] = 1;
    		else nsum[-x] = 1;
    	}
    }
    cout << count << endl;
    return 0;
}
