#include<iostream>
#include<cstring>
using namespace std;
#define ll long long
#define mod 1000000007
string a,b;
long long n;
void sol()
{
	cin>>a>>b;
	ll ar[26]={0},br[26]={0};
	ll ans=1;
	for(ll i=a.length()-1;i>=0;i--)
	ar[a[i]-'a']++;
	for(ll i=b.length()-1;i>=0;i--)
	br[b[i]-'a']++;
	for(ll i=0;i<26;i++)
	{
		if(ar[i]-br[i]>=0)
		{
			for(ll j=ar[i];br[i]>0;j--,br[i]--)	ans=(ans*j)%mod;
		}
		else if(ar[i]-br[i]<0)
		{
			cout<<0<<endl;
			return ;
		}
	}
	cout<<ans<<endl;
}
int main()
{
	cin>>n;
	while(n--)
	{
		sol();
	}
	return 0;
}
