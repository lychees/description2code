#include <iostream>
#include <cstring>
#include <algorithm>
#include <map>
#include <unordered_map>
using namespace std;

int n;
long long w[41];


int main()
{
    ios::sync_with_stdio(0);
	int t;
	cin>>t;
	while(t--){		
		cin>>n;
		for(int i=0; i<n; i++)
        cin >> w[i]; 
        int n1 = n/2,n2 = n-n1;
        map<long long,int> mp;
        for(int i=0;i<(1<<n1);i++){
        	long long sum = 0;
        	for(int j=0;j<n1;j++){
        		if ((i>>j)&1) sum += w[j];
        	}
        	if (mp.count(sum)) mp[sum]++;
        	else mp[sum] = 1;
        }
        long long ans = 0;
        for(int i=0;i<(1<<n2);i++){
        	long long sum = 0;
        	for(int j=0;j<n2;j++){
        		if ((i>>j)&1) sum += w[j+n1];
        	}
        	if (mp.count(-sum)) {
        		//cout << -sum << endl;
        		ans += mp[-sum];
        	}
        	
        }
        cout << ans-1 << "\n";
	}
    
	

}
