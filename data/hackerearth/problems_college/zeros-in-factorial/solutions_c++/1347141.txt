#include <iostream>
#include<map>
using namespace std;

int main()
{
   	long long m[1000001]={0};
   	int i,j;
   	int t,a,b;
   	
   	for(i=1;i<=1000000;i++)
   	{
   		j=i;
   		m[i]=m[i-1];
   		
   		for(;j>0;j=j/5)
   		{
   			m[i]+=j/5;
   		}
   	}
   	
   	cin>>t;
   	
   	for(i=0;i<t;i++)
   	{
   		cin>>a>>b;
   		cout<<m[b]-m[a-1]<<"\n";
   	}
   	
    return 0;
}
