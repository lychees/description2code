#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n, i, k;
    cin>>n;
    while(n--) {
    	char s[50000];
    	cin>>s;
    	k=1;
 		for(i=1;i<(strlen(s)/2)+1;i++) {
 			if(abs(s[i]-s[i-1])!=abs(s[strlen(s)-i-1]-s[strlen(s)-i])) {
 				k=0;
 				break;
 			}
 		}
 		if(k==0)
 			cout<<"Not Funny\n";
 		else
 			cout<<"Funny\n";
    }
    return 0;
}
