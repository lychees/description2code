long long count_carry(long long n1,long long  n2)
{
	int s=0,c=0,count=0;
	while(n1>0||n2>0)
	{
		int unit_digit1=n1%10;
		int unit_digit2=n2%10;
		s=unit_digit1+unit_digit2+c;
		c=s/10;
		if(c>0)
		count++;
		n1=n1/10;
		n2=n2/10;
	}
	return count;
}
int main()
{
	long long t,n1,n2,n;
	int i,j;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
	
		for(i=n;i>=0;i--)
		{
		j=n-i;
		
			if((i%3==0)&&(j%5==0))
			break;
	
		
	}
	if(i>=0)
	{
		for(int u=0;u<i;u++)
		cout<<"5";
		for(int u=0;u<j;u++)
		cout<<"3";
	}
	else
	cout<<"-1";
	cout<<endl;
	
	}
}
