#include <cstdio>
#define MOD 1000000007
using namespace std;

long long h[1000005], e[1000005], sum[1000005];

int main()
{
    h[1] = e[1] = 1LL;
    sum[1] = 2LL;
    for(int i=2; i<1000001; i++)
    {
    	h[i] = (h[i-1]+e[i-1])%MOD;
    	e[i] = h[i-1];
    	sum[i] = (sum[i-1] + h[i] + e[i])%MOD;
    }
    
    int t, n;
    scanf("%d", &t);
    while(t--)
    {
    	scanf("%d", &n);
    	printf("%lld\n", sum[n]);
    }
    
    return 0;
}
