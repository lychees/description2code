#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    
    float x1,x2,x3,y1,y2,y3,a,b,c,d,e,f;
    cin>>x1>>y1>>x2>>y2>>x3>>y3;
    a=x2+x1-x3;b=y2+y1-y3;
    c=x1+x3-x2;d=y1+y3-y2;
    e=x2+x3-x1;f=y2+y3-y1;
   
    multimap<float,float> m;
    m.insert(make_pair(a,b));
    m.insert(make_pair(c,d));
    m.insert(make_pair(e,f));
   
    for(auto it =m.begin(); it!=m.end(); it++)
	printf("%.4f %.4f\n",it->first,it->second);
    return 0;
}
