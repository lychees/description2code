#include <iostream>
using namespace std;

int main()
{
	long long int t, i, j, m = 0, k, l;
	cin >> t;
	while (t--)
	{
		m = 0;
		cin >> i >> j;
		if (j - i == 1)cout << (j&i) << endl;
		else{
			l = 1;
			while (l <= j)
				l = l << 1;
			l = l >> 1;
			if (l == j && ((l - 2) >= i))cout << (l - 2) << endl;
			else if (l == j && ((l - 2)<i)) cout << "0" << endl;
			else
			{

				if (((j - l) % 2 == 0) && ((j - 1) >= i))cout << j - 2 << endl;
				else if (((j - l) % 2 == 1) && ((j - 1) >= i)){ cout << j - 1 << endl; }
				else cout << "0" << endl;
			}
		}
	}
	return 0;
}
