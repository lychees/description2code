#include <stdio.h>
#include <string>
#include <iostream>
#include <utility>
#include <algorithm>
using namespace std;
int main()
{
    int i;

    	int n;
    	scanf("%d",&n);
    	string str[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>str[i];
    	}
    	sort(str,str+n);
    	string prev=str[0];
    	int prevCount=1;
    //	cout<<str[0]<<" "<<str[1]<<" "<<str[2]<<endl;
    	//cout<<"Hello\n";
    	for(i=1;i<n;i++)
    	{
    		if(str[i]==prev)
    		prevCount++;
    		else
    		{
    			cout<<prev<<" "<<prevCount<<endl;
    			prev=str[i];
    			prevCount=1;
    		}
    	}
    	cout<<prev<<" "<<prevCount<<endl;


    return 0;
}
