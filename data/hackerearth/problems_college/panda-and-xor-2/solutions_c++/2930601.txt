#include <iostream>
using namespace std;
#define ll  unsigned long long  

bool compute(ll n){
	ll count =0;
	while(n){
		bool c=n&1;
		if(c)
		count++;
		n=n>>1;
		
	}
	if(count%2 == 0)
	 return true;
	
	else
	 return false; 
}

int main()
{
	int T;
	ll a,b;
	cin>>T;
	while(T--){
		cin>>a>>b;
		if(compute(a^b))
		 cout<<"YES"<<"\n";
		else
		 cout<<"NO"<<"\n";
		
	}
    return 0;
}
