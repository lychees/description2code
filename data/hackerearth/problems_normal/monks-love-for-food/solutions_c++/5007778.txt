#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,q,val;
    stack<int> st;
    cin>>n;
    while(n--)
    {
    	cin>>q;
    	if(q==1)
    	{
    		if(st.empty())
    		cout<<"No Food"<<endl;
    		else
    		{
    			cout<<st.top()<<endl;
    			st.pop();
    		}
    	}
    	else
    	{
    		cin>>val;
    		st.push(val);
    	}
    }
    return 0;
}
