#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<map>
#include<vector>
#include<algorithm>
#include<cmath>
#include<cstdlib>
#include<set>
#include<stack>
#include<deque>
#include<queue>
#include <limits.h>
#define pb(x) push_back(x)
#define getcx getchar_unlocked
using namespace std;
vector<int> g[26];
int l1,l2;
string s1,s2;
int main()
{
    int r[26];
    int in[26];
    bool present[26];
    
    for(int i=0;i<26;i++)
      {r[i]=26;in[i]=0;present[i]=0;}
      
    int t;
    scanf("%d",&t);
    cin>>s1;
    l1=s1.length();
    for(int j=0;j<l1;j++)
                  present[s1[j]-'a']=1;
    
    for(int m=0;m<t-1;m++){
            cin>>s2;
            l2=s2.length();
            for(int j=0;j<l2;j++)
                  present[s2[j]-'a']=1;
            for(int i=0;i<l1;i++){
                    
                    if(s1[i]==s2[i])continue;
                    g[s1[i]-'a'].pb(s2[i]-'a');
                    in[s2[i]-'a']++;
                    break;
            }
            s1=s2;
            l1=l2;
    }
    
    queue<int> q;
    
    for(int i=0;i<26;i++)
       if(present[i] && in[i]==0)
          q.push(i);
     
    while(!q.empty()){
       int x=q.front();q.pop();
       for(int i=0;i<g[x].size();i++){
               in[g[x][i]]--;
               r[g[x][i]]=min(r[x]-1,r[g[x][i]]);
               if(in[g[x][i]]==0)
                 q.push(g[x][i]);
       } 
    }
    
   vector<int> final[26];
   for(int i=0;i<26;i++) 
     if(present[i])
       final[r[i]-1].pb(i);
    
   for(int i=0;i<26;i++)   
      sort(final[i].begin(),final[i].end());
      
   for(int i=25;i>=0;i--){
      for(int j=0;j<final[i].size();j++)
        cout<<char('a'+final[i][j]);
        if(final[i].size()!=0)cout<<endl;
  } 
  
   return 0;
}
