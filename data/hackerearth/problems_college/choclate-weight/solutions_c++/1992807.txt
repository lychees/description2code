#include<bits/stdc++.h>
using namespace std;
int main()
{
  	int n;
  	cin>>n;
  	long int a[n];
  	for(int i=0;i<n;i++)
  	{
  		scanf("%ld",&a[i]);
  	}
  	int q;
  	cin>>q;
  	while(q--)
  	{
  		long int inp;
  		cin>>inp;
  		int res=-1;
  		long int sum=0;
  		for(int i=0;i<n-1;i++)
  		{
  			sum=a[i];
  			if(sum==inp)
  			{
  				res=0;
  				break;
  			}
  			for(int j=i+1;j<n;j++)
  			{
  				sum+=a[j];
  				if(sum==inp)
  				{
  					res=0;
  					break;
  				}
  				sum=a[i];
  			}
  			if(res==0)
  			{
  				break;
  			}
  		}
  		if(a[n-1]==inp)
  		{
  			res=0;
  		}
  		if(res==-1)
  		{
  			cout<<"I Hate You"<<endl;
  		}
  		else
  		{
  			cout<<"I Like You"<<endl;
  		}
  	}
}
