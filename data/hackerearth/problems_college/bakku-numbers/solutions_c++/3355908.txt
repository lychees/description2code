#include <bits/stdc++.h>
using namespace std;
#define ll long long int
#define mod 1000000007
int main()
{
    int a, b, q, l, r, s,p;
    cin>>a>>b>>q;
    while (q--)
    {
        cin>>l>>r;
        --l;
        s = r/a + r/b - r/(a*b);
        p = l/a + l/b - l/(a*b);
        printf("%.6lf\n", double(s-p)/(r-l));
    }
}
