#include<bits/stdc++.h>
using namespace std;
int P[10000],A[10000],B[10000];
int main(){

	int n,a,b,c,t;
	cin >> t;
	while(t--){

		cin >> n >> a >> b >> c;
		memset(P,0,sizeof(P));
		memset(B,0,sizeof(B));
		memset(A,0,sizeof(A));
		for(int i=0;i<=a;i++)
			A[i]=1;
		for(int i=0;i<=b;i++)
			B[i]=1;
		
		for(int i=0;i<=a;i++){
			for(int j=0;j<=b;j++){
				P[i+j]+=(A[i]*B[j]);
			}
		}
		for(int i=0;i<=a+b;i++)
			A[i]=P[i];
		for(int i=0;i<=b;i++)
			B[i]=0;
		for(int i=1;i<=c;i++)
			B[i]=1;
		
		for(int i=0;i<=a+b;i++)
			for(int j=0;j<=c;j++)
				P[i+j]+=(A[i]*B[j]);
		long long int sum=0;
		for(int i=0;i<=n;i++)
			sum+=P[i];
		cout << sum << endl;
	}
	return 0;
}
