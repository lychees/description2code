t = int(input())
if(t>=1 and t <= 50):
	for i in range(t):
		s = list(raw_input())
		v = list('aeiouy')
		c = list('bcdfghjklmnpqrstvwxz')
		t1 = 0
		t2 = 0
		flag = 0
		x = 0
		l = len(s)
		if(l >= 1 and l <= 50):
			for p in range(l):
				if(s[p] in v):
					t1 += 1
					flag = 0
				if(s[p] in c):
					t2 += 1
					flag += 1
					if (flag == 3):
						print "hard"
						flag = 0
						x = 1
						break
		if(t1 >= t2 and x==0):
			print "easy"
		if( t2 > t1 and x == 0 ):
			print "hard"
