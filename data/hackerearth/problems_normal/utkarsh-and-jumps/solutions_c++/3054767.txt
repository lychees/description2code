#include<bits/stdc++.h>
using namespace std;
const int MOD = 1e9 + 7;
int main()
{
//	ios_base::sync_with_stdio(0);
	double dp[1000002],p2 = 0.00,p3 = 0.00;
	int P,N;
	
	cin >> N >> P;
	p2 = P/100.00;
	p3 = 1.00 - p2;

	dp[0] = 0;
	dp[1] = 0;
	dp[2] = p2;
	dp[3] = p3;
	for(int i = 4;i <= N;i++)
		dp[i] = (p2 * dp[i -2]) + (dp[i -3]*p3);

	printf("%.6f\n",dp[N]);
	return 0;
}
