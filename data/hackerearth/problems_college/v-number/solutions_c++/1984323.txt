#include <bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
   long long int tc;
   cin>>tc;
   while(tc--)
   {
       string s;
       cin>>s;
       long long int len,sum1=0,sum2=0;
       len = s.length();
        if(len%2==0)
        {
              for(int i=0;i<(len/2);i++)
              {
                  sum1 += s[i]-'0';
              }
              for(int i=(len/2);i<len;i++)
              {
                  sum2 += s[i]-'0';
              }
              if(sum1==sum2) cout<<"YES"<<endl; else cout<<"NO"<<endl;
        }
      else
      {
          for(int i=0;i<(len/2);i++)
          {
              sum1 += s[i]-'0';
          }
          for(int i=(len/2)+1;i<len;i++)
          {
              sum2 += s[i]-'0';
          }
          if(sum1==sum2) cout<<"YES"<<endl; else cout<<"NO"<<endl;
      }
   }
}
