a,b=map(str,raw_input().split())
fnd=[]
f=0
for i in range(int(b)):
    l=list(map(str,raw_input().split()))
    for i in range(len(a)):
        if i not in fnd:
            if a[i] in l:
                fnd.append(i)
                break
    if len(fnd)==len(a):
        f=1
        break
if(f):
    print 'YES'
else:
    print 'NO'
