#include <iostream>
using namespace std;

int main()
{
	int n,k,x,y,count=0;
	cin>>n;
    int iland[n];
    for(int i=0;i<=n;i++){
    	iland[i]=1;
    }
    cin>>k;
    for(int i=1;i<=k;i++){
    	cin>>x>>y;
    	iland[y]=0;
    	iland[x]++;
    }
    for(int i=1;i<=n;i++){
    	if(iland[i]){
    		count++;
    	}
    }
    cout<<count;
    return 0;
}
