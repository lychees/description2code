#include<bits/stdc++.h>
using namespace std;
int main()
{
vector <int> map1;
vector <int>::iterator t;
int T,M,x;
cin>>T;
while(T)
{
    map1.clear();
    cin>>M;
    while(M)
    {
    cin>>x;
    t=find(map1.begin(),map1.end(),x);
    if(t==map1.end())
        map1.push_back(x);

    M--;
    }
    cout<<map1.size()<<endl;


T--;
}
return 0;
}
