#include<iostream>
int t,cs=0;
int main(){
	int n;
	long int arr[10000],hsh[1000];
	std::cin>>t;
	while(cs<t){
		int m,collison=0;
		std::cin>>n;
		for(int i=0;i<n;i++)
			std::cin>>arr[i];
			int j=0;
			int h=arr[0]%100;
			hsh[j]=h;
		for(int i=1;i<n;i++){
			h=arr[i]%100;
			m=collison;
			for(int k=0;k<=j;k++){
				if(h==arr[k])
					collison++;
			}
			if(collison==m){
				hsh[j]=h;
				j++;
			}
		}
		std::cout<<collison<<"\n";
		cs++;
	}
	return 0;
}
