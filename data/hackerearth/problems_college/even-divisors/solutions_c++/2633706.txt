#include<bits/stdc++.h>
using namespace std;

int main()
{
    long long i,t,n,cal;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld",&n);
    	if(n%2)
    		printf("0\n");
    	else
    	{
    	cal=0;
    	for(i=1;i<=sqrt(n);i++)
    	{
    		if(n%i==0)
    		{
    			if(i*i!=n)
    			{
    				if(i%2==0)
    					cal++;
    				if((n/i)%2==0)
    					cal++;
    			}
    			else
    			{
    				if(i%2==0)
    					cal++;
    			}
    				
    		}
    	}
    	printf("%lld\n",cal);
    	}
    	
    }
    return 0;
}
