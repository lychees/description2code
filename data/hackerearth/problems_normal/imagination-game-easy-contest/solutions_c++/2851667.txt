#include <iostream>
using namespace std;
int a[11];
int main()
{
    int t;
    cin>>t;
    
    while(t--) {
    	int n;
    	cin>>n;
    	int xorr = 0;
    	for(int i = 0; i < n; ++i) {
    		cin>>a[i];
    	}
    	for(int i = 0; i < n; ++i) {
    		int x;
    		cin>>x;
    		xorr ^= a[i]-x;
    		xorr ^= x-1;
    	}
    	if(xorr != 0) {
    		cout<<"First"<<endl;
    	}
    	else {
    		cout<<"Second"<<endl;
    	}
    }
    
    return 0;
}
