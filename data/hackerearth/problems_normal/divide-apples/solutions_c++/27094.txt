#include <iostream>
#include<algorithm>
#include<cstdio>
#include<vector>

using namespace std;

#define F first
#define S second
#define LL long long

inline LL pos(LL a){
    if(a<0){
        return -a;
    }
    return a;
}

LL a[10001],b,d[10001];

int main(){
    LL n;
    cin>>n;
    LL i,j=2,k,l,fg,ans=1000000000,ans2;
    int in;
    ans*=ans;
    ans2=ans;
    for(i=0;i<n;++i){
        scanf("%d",&in);
        a[i]=in;
        j+=a[i];
    }
    j/=n;
    for(i=0;i<n;i++){
        d[i]=a[i]-j;
    }
    for(i=0;i<n;++i){
        if(a[i]!=j){
            l=0;
            b=0;
            for(k=i;k<n;k++){
                b+=d[k];
                if(b>0){
                    l=l+b;
                }
                else{
                    l=l-b;
                }
            }
            for(k=0;k<i;k++){
                b+=d[k];
                if(b>0){
                    l=l+b;
                }
                else{
                    l=l-b;
                }
            }
            if(l<ans){
                ans=l;
            }
        }
    }
    if(ans==ans2){
        printf("0\n");
    }
    else{
        printf("%lld\n",ans);
    }
    return 0;
}
