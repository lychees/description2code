#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	scanf("%d",&n);
	long long int dp[n+2],a[n+2];
	for(int i=1;i<=n;i++)
	{
		scanf("%lld",&a[i]);
	}
	printf("%lld ",a[1]);
	dp[1]=a[1];
	for(int i=2;i<=n;i++)
	{
		long long int sq=((long long int)sqrt(i))+1;
		long long int ma=0;
		dp[i]=0;
		for(int j=1;j<=sq;j++)
		{
			if(i%j==0)
			{
				if(ma<dp[j])ma=dp[j];
				if(ma<dp[i/j])ma=dp[i/j];
			}
		}
		dp[i]=ma+a[i];
		printf("%lld ",dp[i]);
	}
	printf("\n");
    return 0;
}
