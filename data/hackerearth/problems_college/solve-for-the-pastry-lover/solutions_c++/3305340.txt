#include <iostream>
#include<vector>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
    int A[10];
    for(int i=0;i<10;i++)
    A[i]=-1;
    int k,min=100000;
    cin>>k;
    
    string s;
    cin>>s;
    if(k==1)
    cout<<"-1"<<endl;
    else{
    for(int i=0;i<k;i++)
    {
    	
    	if(A[s[i]-'0']==-1)
    	A[s[i]-'0']=i;
    	else{
    		if(min>i-A[s[i]-'0'])
    		min=i-A[s[i]-'0'];
    	A[s[i]-'0']=i;
    	}
    }
    if(min!=100000)
    cout<<min<<endl;
    else
    cout<<"-1"<<endl;
	}
	}
    return 0;
}
