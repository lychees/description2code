#include <iostream>
using namespace std;
char a[100];
int i,count,n;
string s;
int main()
{
    cin>>s;
    n=s.length();
 	count=0;
    
    for(i=0;i<n;i++){
    	a[i]=s[i];
    switch(a[i]){
    	case '1':
    		count+=2;
    		break;
    	case '2':
    		count+=5;
    		break;
    	case '3':
    		count+=5;
    		break;
    	case '4':
    		count+=4;
    		break;
    	case '5':
    		count+=5;
    		break;
    	case '6':
    		count+=6;
    		break;
    	case '7':
    		count+=3;
    		break;
    	case '8':
    		count+=7;
    		break;
    	case '9':
    		count+=6;
    		break;
    	case '0':
    		count+=6;
    		break;
    	default:
    		count+=0;
    		break;
    		
    }
    }
    cout<<count;
}
