#include<bits/stdc++.h>
using namespace std ;
int main()
{
	int T;scanf("%d",&T);
	while(T--)
	{
		int N; scanf("%d",&N);
		vector<int> IN;
		int temp ;
		for(int i=0;i<N;++i){ scanf("%d",&temp); IN.push_back(temp);} 
		vector<bool> visited(N,0) ;
		bool flag=false ;
		int current=0 ;
		while(!flag)
		{
			if(visited[current]) break ;
			if(IN[current]==-1) flag=true ;
			visited[current]=true ;
			current=IN[current] ;			
		}
		if(flag) cout<<"Win\n" ;
		else cout<<"Lose\n" ;
	}
	return 0 ;
}
