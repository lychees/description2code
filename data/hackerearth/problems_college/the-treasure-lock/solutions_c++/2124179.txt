#include <iostream>
using namespace std;
void print(int five,int three)
{
	while(five--)
		cout<<"5";
	while(three--)
		cout<<"3";
	cout<<endl;
}
int main()
{
	int t,i,n,flag;
	cin>>t;
	while(t--)
	{
		i=flag=0;
		cin>>n;
		//if(n==1 || n==2 || n==4)
		//	cout<<"-1\n";
		if(n%3==0)
		{
			print(n,0);
			flag=1;
		}
		else if(n%5==0)
		{
			print(0,n);
			flag=1;
		}
		else
		{
			i=i+5;
			while(i<n)
			{
				if((n-i)%3==0)
				{
					print(n-i,i);
					flag=1;
					break;
				}
				i=i+5;
			}
		}
		if(flag==0)
			cout<<"-1\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
