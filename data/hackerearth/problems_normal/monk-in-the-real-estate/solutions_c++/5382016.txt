#include <iostream>
using namespace std;

int main()
{
    long long t,e,x,y;
    cin>>t;
    long long a[10001];
    
    
    
    
    while(t--)
    {
    	for(long long i=0; i<10001; i++)
    {
    	a[i]=0;
    }
    	cin>>e;
    	while(e--)
    	{
    		cin>>x>>y;
    		a[x]=1;
    		a[y]=1;
    	}
    
    	int count=0;
    	for(int k=0;k<10001;k++)
    	{
    		if(a[k]==1) 
    		{
    		 count +=1;
                 		 
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
