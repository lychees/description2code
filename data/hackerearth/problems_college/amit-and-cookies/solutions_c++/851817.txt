#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int n,k,i,sum=0,c=0;
    cin>>n>>k;
    int a[n];
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    sort(a,a+n);
    i=0;
    while(1)
    {
      if((sum+a[i])<=k)
      {sum+=a[i];c++;}
       else
       break;
       i++;
    }
    cout<<c;
}
