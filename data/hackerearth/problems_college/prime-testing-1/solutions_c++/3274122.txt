#include <iostream>
using namespace std;
typedef long long int lli;
int isprime(int n)
{
	int i;
	if(n==0||n==1)
		return 0;
	for(i=2;i<=n/2;i++)
	{
		if(n%i==0)
		{
				return 0;
		}
	}
	return 1;
}
int main()
{
    int t,a,ctr,i;
    lli n;
    cin>>t;
    while(t--)
    {
    	ctr=0;
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>a;
    		if(isprime(a)==1)
    			ctr++;
    	}
    	cout<<ctr<<endl;
    }
    return 0;
}
