#include<bits/stdc++.h>
using namespace std;
int r[100005];
int root(int x){while(x!=r[x])x=r[x];return x;}
void uni(int x,int y)
{	
	r[root(x)]=root(y);
}
int main()
{
	int n,q;cin>>n>>q;
	for(int i=1;i<=n;i++)r[i]=i;
	while(q--)
	{
		int t;
		cin>>t;
		if(t==1)
		{
			int a,b;cin>>a>>b;
			uni(a,b);
		}
		else if(t==2)
		{
				int a;cin>>a;
			r[root(a)]=a;
			r[a]=a;
		}
		else if(t==3)
		{
			int a;cin>>a;
			cout<<root(a)<<endl;
		}
	}
	return 0;
}
