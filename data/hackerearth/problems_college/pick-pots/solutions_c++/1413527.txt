using namespace std;

lli mod=1e9+7;
lli dp[1000001][2];

void precompute()
{
    dp[1][0]=0;
    dp[1][1]=1;
    for(int i=2;i<1000001;i++)
    {
        dp[i][0]=(dp[i-1][1]+dp[i-1][0])%mod;
        dp[i][1]=(dp[i-1][0]+1)%mod;
    }
}

int main()
{
    int t,n;
    precompute();
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        printf("%lld\n",(dp[n][0]+dp[n][1])%mod);
    }
    return 0;
}
