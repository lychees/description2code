#include <iostream>
using namespace std;
#include <math.h>
#define ll long long
int main()
{   ll t;
    cin>>t;
    while(t--)
    {   ll n,c=0,i,j;
        cin>>n;
        for(i=1;i<n;++i)
            for(j=i+1;j<=n;++j)
                if((i*j)%(i+j)==0)
                    ++c;
        cout<<c<<"\n";
    }
    return 0;
}
