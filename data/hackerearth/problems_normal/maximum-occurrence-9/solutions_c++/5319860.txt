#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	string s;
	getline(cin,s);
	int len=s.length();
	map<int,int>m;
	for(int i=0;i<len;i++)
	{
		m[(int)s[i]]+=1;
	}
	int ans = (m.begin())->first;
	for(auto it=m.begin();it!=m.end();it++)
	{
		//cout<<(char)it->first<<it->second;
		if(it->second > m[ans])
			ans=it->first;
	}
	cout<<(char)ans<<" "<<m[ans];
    
    return 0;
}
