#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,i;
	char ch;
	string s,ss="";
    cin >> t;
	while(t--)
	{
		cin >> s;
		for(i=0;i<s.size();i++)
		{
			ch=s[i];
			if(i==0)
				ss=ss+ch;
			else
			{
				if((int)ch>=(int)ss[0])
					ss=ch+ss;
				else
					ss=ss+ch;
			}
		}
		cout << ss << endl;
		ss="";
	}
    return 0;
}
