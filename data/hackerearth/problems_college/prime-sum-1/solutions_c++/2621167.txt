#include <bits/stdc++.h>
using namespace std;
int isprime(int n)
{
	int i;
	for(i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
		return 0;
	}
	return 1;
}
int main()
{
    //cout << "Hello World!" << endl;
    int a[]={2,3,5,7};
    int t,y,cnt,n;
    cin>>t;
    while(t--)
    {
    	cnt=0;
    	cin>>y;
    	n=y;
    	if(y==2 || y==3 || y==5 || y==7)
    	cout<<"1"<<endl;
    	else if(n==1)
    	cout<<"-1"<<endl;
    	else
    	{
    		while(n>10)
    		{
    		n=n-7;
            cnt++;
    		}
    		if(isprime(n))
    		cnt++;
    		else if(n==4 || n==6 || n==8 || n==10 || n==9)
    		cnt=cnt+2;
    		cout<<cnt<<endl;
    	}
    	
    }
    return 0;
}
