
#include<bits/stdc++.h>


using namespace std;

bool dp[100005];
int arr[105];
int main()
{
 //freopen("input.in", "r", stdin);
 int n,t;

     for(int i=0;i<100001;i++)
     dp[i]=false;


     cin>>n;
     for(int i=0;i<n;i++)
     cin>>arr[i];

 dp[0]=true;
 for(int i=0;i<n;i++)
 {

     for(int j=100000;j>=0;j--)
     {
         if((j-arr[i])>=0)
            dp[j]|=dp[j-arr[i]];
     }
 }
cin>>t;
while(t--)
{
  cin>>n;
  if(n>100000)
        cout<<"No\n";
  else if(dp[n])
    cout<<"Yes\n";
  else
    cout<<"No\n";

}
}
