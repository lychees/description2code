#include<bits/stdc++.h>
using namespace std;

#define MAX 9000005
vector<int> primes;
bool v[MAX];

void Sieve()
{	
	memset(v,0,sizeof v);
	for (int i = 2; i < MAX; i += 2)	v[i] = true;//even numbers have smallest prime factor 2
	v[2]=false;
	v[1]=true;
	primes.push_back(2);
	
	for (long long i = 3; i < MAX; i += 2)
	{
		if (!v[i])
		{
			primes.push_back(i);
			for (long long j = i; (j*i) < MAX; j += 2)
			{
				if (!v[j*i])
					v[j*i] = true;
			}
		}
	}
}
	long long arr[100009]={0};
		
int main()
{ arr[1]=2;
	Sieve();
	long long mod=1000000007;
//	cout<<primes.size()<<endl;
	for(int i=2;i<=100008;i++)
	{
		arr[i]=arr[i-1]*primes[i-1];
		arr[i]%=mod;
	}
	int n,t;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<arr[n]<<endl;
	}
	
	
}
