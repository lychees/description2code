#include <bits/stdc++.h>
typedef long long int ll;
using namespace std;
void dfs(vector <ll> v[],ll root)
{
	stack <ll> s;ll k,i;
	s.push(root);
	while(!s.empty())
	{
		k=s.top();
		s.pop();
		for(i=v[k].size()-1;i>=0;i--)
		{
			s.push(v[k][i]);
		}
		cout<<k<<" ";
	}
}

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	ll n,X,Y,Z;
    	cin>>n;
    	vector <ll> v[100002];
    	while(n--)
    	{
    		cin>>X>>Y>>Z;
    		if(Y!=0)
    		v[X].push_back(Y);
    		if(Z!=0)
    		v[X].push_back(Z);
    	}
    	dfs(v,1);
    	cout<<endl;
    }
    return 0;
}
