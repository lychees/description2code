#include <bits/stdc++.h>
using namespace std;

bool prime(int n){
	for(int i=2;i<=(int)sqrt(n);i++){
		if(n%i==0)
			return false;
	}
return true;
}

int mul(int ans[],int n,int s){
	int cary=0;
	for(int i=0;i<s;i++){
		int p=ans[i]*n+cary;
		ans[i]=p%10;
		cary=p/10;
	}
	while(cary){
		ans[s]=cary%10;
		cary=cary/10;
		s++;
	}
	return s;
}

void fact(int n){
	int ans[500];
	ans[0]=1;
	int s=1;
	for(int i=2;i<=n;i++){
		s=mul(ans,i,s);
	}
	for(int i=s-1;i>=0;i--)
		printf("%d",ans[i]);
}

int main()
{
    int N;
    scanf("%d",&N);
    for(int i=2;i<=N;i++){
    	if(prime(i)){
    		fact(i);
    		printf("\n");
    	}
    }
    return 0;
}
