#include<iostream>
#include<algorithm>
#include<math.h>
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<limits.h>
#include<vector>
#include<set>
#include<stack>
#include<queue>
#define MOD 1000000007
#define SZ 100005
using namespace std;

int main()
{
    long int n,count[SZ],i,num;
    memset(count,0,sizeof(SZ));
	cin>>n;
	
	//
	count[0] = 0;
	count[1] = 1;
	count[2] = 2;
	count[3] = 4;
	count[4] = 7;
	for(i=5;i <= SZ;i++)
	{
		count[i] = (count[i-1]+count[i-2]+count[i-3])%MOD;
	}
	//
	while(n--)
	{
		cin>>num;
		cout<<count[num]<<endl;
	}
	return 0;
}
