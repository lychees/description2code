#include <bits/stdc++.h>
using namespace std;

#define MX 9000001
#define MOD 1000000007

bool Pri[MX + 2];

int main()
{
	vector<int> Primes;
    for(int i = 2; i <= MX; i += 2) {
    	Pri[i] = true;
    }
    Primes.push_back(2);
    for(int i = 3; i <= MX; i += 2) {
    	if(Pri[i])
    		continue;
    	long long j = i;
    	for(j = j * j; j <= MX; j += i) {
    		Pri[j] = true;
    	}
    	Primes.push_back(i);
    }
	
	int sz = Primes.size();   
//	cout<<"Sz => " << sz << "\n";
    long long Dp[sz + 2];
    Dp[0] = 1;
    for(int i = 1; i <= sz; ++i) {
    	Dp[i] = Dp[i - 1] * Primes[i - 1];
    	if(Dp[i] >= MOD)
    		Dp[i] %= MOD;
    }
    int T;
    scanf("%d", &T);
    while(T--) {
    	int N;
    	scanf("%d", &N);
    	printf("%lld\n", Dp[N]);
    }
    
    return 0;
}
