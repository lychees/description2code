int gcd(int a, int b) { return b == 0 ? a : gcd(b, a % b);}

int main(){
	fastscan	
	ll mat[1009][1009];
	for(int j=1;j<1009;j++) mat[1][j] = 1;
	for(int i=2;i<1009;i++){
		for(int j=1;j<1009;j++){
			ll x = 0;
			if(i>=2 && j>=2) x = (x+mat[i-1][j-1])%mod;
			if(i>=2) x = (x+mat[i-1][j])%mod;
			if(j>=2) x = (x+mat[i][j-1])%mod;
			mat[i][j] = x;
		}
	}
	int t;
	cin>>t;
	while(t--){
		int a,b,c,d;
		cin>>a>>b>>c>>d;
		cout<<mat[c-a+1][d-b+1]<<"\n";
	}
	return 0;
