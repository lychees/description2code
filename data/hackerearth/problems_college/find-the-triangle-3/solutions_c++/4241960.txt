#include <bits/stdc++.h>
using namespace std;
int input[100001];

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	
    	for(int i=0;i<n;i++)
    	cin>>input[i];
    	
    	sort(input,input+n);
    	
    	bool flag=false;
    	for(int i=n-1;i>=2;i--)
    	{
    		if(input[i]<input[i-1]+input[i-2])
    		{
    			flag=true;
    			cout<<input[i-2]<<" "<<input[i-1]<<" "<<input[i]<<endl;
    			break;
    		}
    	}
    	if(!flag)
    	cout<<"-1"<<endl;
    }
    return 0;
}
