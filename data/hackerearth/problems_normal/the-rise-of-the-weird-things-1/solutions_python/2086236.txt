s = raw_input()
ar = map(int,raw_input().split())
e = []
o = []
for i in ar:
    if i%2 == 0:
        e.append(i)
    else:
        o.append(i)
e = sorted(e)
o = sorted(o)
se = str(sum(e))
so = str(sum(o))
e = map(str,e)
o = map(str,o)
print ' '.join(e)+' '+se+' '+' '.join(o)+' '+so
