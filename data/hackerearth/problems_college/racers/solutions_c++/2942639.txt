#define si(x) scanf("%d",&x)

using namespace std;
int n,m,k,vis[10000]={},linked[10000];
ll arrx[100069],arry[100069],brrx[100069],brry[100069];
bool dfs(int i,ll yo)
{
    if(vis[i])
        return false;
    vis[i]=1;
    FOR(j,0,m)
    {
        if((linked[j]==-1||dfs(linked[j],yo))&&(((brrx[j]-arrx[i])*(brrx[j]-arrx[i])+(brry[j]-arry[i])*(brry[j]-arry[i]))<=yo))
        {
          //  cout<<u<<" p "<<yo<<endl;
            linked[j]=i;
            return true;
        }
    }
    return false;
}
bool check(ll yo)
{
    int c=0;
    FOR(i,0,n)
    {
        memset(vis,0,sizeof(vis));
        if(dfs(i,yo))
            c++;
    }
    if(c>=k)
        return true;
    else
        return false;
}
int main()
{
    cin>>n>>m>>k;
    FOR(i,0,n)
        sll(arrx[i]),sll(arry[i]);
    FOR(i,0,m)
        sll(brrx[i]),sll(brry[i]);
    ll l=0,h=1e15;
    while(l<h)
    {
        memset(linked,-1,sizeof(linked));
        ll m=(l+h)/2;
        if(check(m))
            h=m;
        else
            l=m+1;
        //cout<<l<<" "<<h<<"\n";
    }
    cout<<h<<endl;
    return 0;
}
