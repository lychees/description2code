#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    
    int t;
    cin>>t;
    getchar();
    for(int i=0;i<t;i++)
    {
    	
    	string s;
    	getline(cin,s);
    	
    if(s.length()>=10)
    {
    	cout<<"YES"<<endl;
    	continue;
    }
    int flag=0;
    for(int i=1;i<s.length();i++)
    {
    	for(int j=i+1;j<s.length();j++)
    	{
    		for(int k=j+1;k<s.length();k++)
    		{
    		    	string s1 = s.substr(0,i);
    		    //	cout<<s1<<endl;
					string s2 = s.substr(i, j - i);
				//	cout<<s2<<endl;
					string s3 = s.substr(j, k - j);
				//	cout<<s3<<endl;
					string s4 = s.substr(k, s.size() - k);
				//	cout<<s4<<endl;
					if (s1 != s2 && s1 != s3 && s1 != s4 && s2 != s3 && s2 != s4 && s3 != s4)
					{
						flag = 1;
					}
    		}
    	}
    }
    
    
    if (flag) cout << "YES" << endl;
		else cout << "NO" << endl;
    	
    }
    return 0;
}
