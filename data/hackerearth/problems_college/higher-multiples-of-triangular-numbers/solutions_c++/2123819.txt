#include <iostream>
#include<math.h>
using namespace std;
int fact( long long int n)
{
	int f=0,i,sq=ceil(sqrt(n));
	for(i=1;i<sq;i++)
	{
		if(i==sq)
			f++;
		else if(n%i==0)
			f=f+2;
	}
	//cout<<f<<" ";
	return f;
}
	
int main()
{
	long long int i,j,t,n,a[1000000]={};
	for(i=1;i<1000000;i++)
		a[i]=a[i-1]+i;
	cin>>t;
	//i=1;
	while(t--)
	{
		cin>>n;
		i=1;
		while(1)
		{
			if(fact(a[i])>n)
			{
				cout<<a[i]<<endl;
				break;
			}
			i++;
		}
	}
    return 0;
}
