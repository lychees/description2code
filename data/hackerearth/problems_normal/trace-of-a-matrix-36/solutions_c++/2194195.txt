#include <iostream>
#include <cmath>
#include <cstdlib>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
    	int r,c;
    	cin >> r >> c;
    	long int x = min(r,c);
    	long int diff=r+1;
    	long int l = 1 + (x-1)*diff;
    	long int sum = x*(1+l);
    	diff=c+1;
    	l = 1+(x-1)*diff;
    	sum+=x*(l+1);
    	sum/=2;
    	cout << sum << endl;
    }
    return 0;
}
