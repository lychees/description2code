#include <bits/stdc++.h>
using namespace std;

int main()
{
   	int n;
   	cin >>n;
   	int a[n], a1[n];
   	for ( int j = 0; j < n ; j++ ) {
   		cin >>a[j];
   	}
   	sort(a, a+n);
   	int cnt1= 0, cnt2 = n-1;
   	long long int diff1 =0, diff2 = 0;
   	int i = 0;
   	while( cnt1 <cnt2){
   		a1[i] = a[cnt1];
   		a1[i+1] = a[cnt2];
   		cnt1++;
   		cnt2--;
   		i+=2;
   	}
   	if (cnt1==cnt2){
   		a1[i] = a[cnt1];
   	}
   	diff1 =0;
   	for ( int i =1 ; i < n ; i++) {
   		long long int x;
   		x =a1[i] -a1[i-1];
   		if (x < 0) {
   			x = -x;
   		}
   		diff1 +=x;
   	}
   	cnt1 = 0, cnt2=n-1;
   	i = 0;
   	diff2 = 0;
   	while ( cnt1 <cnt2) {
   		a1[i] = a[cnt2];
  		a1[i+1] = a[cnt1];
  		cnt1++;
  		cnt2--;
  		i = i+2;
   	}
   	if(cnt1 == cnt2) {
   		a1[i] = a[cnt1];

   	}
   	for ( int i = 1; i < n ; i++ ){
   		long long int x;
   		 x = a1[i]-a1[i-1];
   		 if (x <0){
   		 	x = -x;
   		 }
   		 diff2 = diff2+x;
   	}
   	cout<<max(diff1,diff2)<<endl;
    return 0;
}
