n = input()
a = map(int, raw_input().split())
l, b = map(int, raw_input().split())
area = l*b
amount = 0
for i in range(n):
    if (a[i] >= 1 and a[i] <= 10):
        amount += int(area*a[i]*a[i]*0.75)
    elif (a[i] >= 11 and a[i] <= 20):
        s = 0
        s = (a[i] % 10)+(a[i]/10)
        amount += int(area*a[i]*s*0.7)
    else:
        k = (a[i] % 10)+(a[i]/10)
        amount += int(area*a[i]*k*0.8)
print amount
