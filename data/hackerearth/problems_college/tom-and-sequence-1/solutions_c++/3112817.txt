#include<bits/stdc++.h>
#define maxi 10000001
#define mod 1000000007
long long int ar[maxi];
long long int vec[maxi+1];

using namespace std;
int main()
{

vector<long long> prime;	
	long long int i,j;
	for(i=0;i<=maxi;i++)
	{
		vec[i]=1;
	}
	for(i=2;i<=maxi;i++)
	{
		if(vec[i]==1)
		{
			prime.push_back(i);
			for(j=i;i*j<=maxi;j++)
			{
				vec[i*j]=0;
			}
		}
	}
	
    ar[0]=2;
	for(i=1;i<prime.size();i++)
	{
		ar[i] = (((ar[i-1])%mod)*(prime[i]%mod))%mod;
	}
	int t;
	cin>>t;
	while(t--)
	{
		long long int n;
		cin>>n;
		cout<<ar[n-1]<<"\n";
	}
}
