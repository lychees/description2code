#include<iostream>
#include<stdio.h>
using namespace std;


double bSS(int n1, int m1, int x1){
	double dice[51][2501];
	int i,j,k;
	for(i=1;i<m1+1;i++)
	{
  	for(j=1;j<x1+1;j++)
      	{
	       dice[i][j]=(double)0;
      	}
	}

	for(i=1;i<=n1;i++)
 	{
      	dice[1][i]=(double)1/n1;
    }
 	for(i=2;i<=m1;i++)
   	{
	    for(j=1;j<=x1&&j<=n1*m1 ;j++)
     	{
       		for(k=1;k<=n1 && k<j;k++)
         	{
	            dice[i][j]=dice[i][j]+dice[i-1][j-k];

         	}
        	dice[i][j]*=(double)1/n1;
     	}
  	}
/*
  	for(i=0;i<=m1;i++){
            for(j=0;j<=k;j++){
                cout<<dice[i][j]<<"---";
            }
            cout<<endl;
		}*/
	return dice[m1][x1];
}


int main(){
	int t,n,m,k,i,tens;
    double result;
    cin>>t;
	while(t--){
		cin>>m>>n>>k;
		if(k>2500 & k<=10000)
            cout<<"0.000 0"<<endl;
		else {
			result=bSS(n,m,k);
            tens=0;
            if(result == 0){
                cout<<"0.000 0"<<endl;
            }
            else{
                while(result<1){
                    result=result*10;
                    tens++;
                }
                printf("%.3lf",result,tens);
                cout<<" "<<tens<<endl;
            }
        }
    }
}
