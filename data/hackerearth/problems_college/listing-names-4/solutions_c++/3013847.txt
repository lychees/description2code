#include <iostream>
#include <map>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string s;
    map<string,int> m;
    while(t--)
    {
    		cin>>s;
    		if(m.find(s)==m.end())
    		m[s]=1;
    		else
    		m[s]++;
    		
    	
    }
    map<string,int>::iterator i;
    for(i=m.begin();i!=m.end();i++)
    cout<<i->first<<" "<<i->second<<endl;
    
    return 0;
}
