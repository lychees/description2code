#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int MOD = 1e9+7;
const double ESP = 1e-9;

int main(void)
{
    // double timeStart = clock();
    //freopen("input.txt","r",stdin);
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    ll testCases,n,i,j;
    cin >> testCases;
    while(testCases--)
    {
        cin >> n;
        if(n == 1)
        {
            cout << '0' << '\n';
            continue;
        }

        ll ans = 0,sum = n*(n-1)/2,r = n-2,res = 0;
        ll k = n/2-1;
        ans = 2*sum;
        res = r;
        while(k--)
        {
            ans += 2*(sum-res);
            r -= 2;
            res += r;
        }
        if(n%2)
        {
            ans += (sum-res);
        }
        cout << ans << '\n';
    }
    
    // double timeEnd = clock();
    // cout << '\n' << (timeEnd - timeStart)/(CLOCKS_PER_SEC) << " seconds" << '\n';
    return 0;
}
