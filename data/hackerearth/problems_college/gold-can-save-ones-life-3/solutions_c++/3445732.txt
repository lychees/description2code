using namespace std;
long long int maxx(long long int a, long long int b)
{
	if(a>b)
		return a;
	else
		return b;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,k;
        cin>>n>>k;
        long long int arr[n];
        for(int i=0;i<n;i++)
            cin>>arr[i];
        long long int dp[n];
        long long int ma[n];
        dp[0]=maxx(arr[0] , 0);
        ma[0]=dp[0];
        for(int i=1;i<n;i++)
        {
            int j = i-k-1;
            if(j<0)
            {
                dp[i]=maxx(arr[i] , 0);
                ma[i]=maxx(ma[i-1],dp[i]);
            }
            else
            {
                dp[i] =maxx(ma[j]+arr[i] , ma[j]);
                ma[i] = maxx(ma[i-1] , dp[i]);
            }
        }
 
        cout<<ma[n-1]<<endl;
    }
    return 0;
}
