#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int len,i,j,val,index;
    scanf("%d",&len);
    char ch[100001];
    scanf("%s",ch);
    val=122;
    index=-1;
    for(i=0;i<len;i++)
    {
    	if(val==96)
    	break;
    	else if(i==len-1 && ch[i]!=val)
    	{
    		i=index;
    		val--;
    	}
    	if(ch[i]==val)
    	{
    		printf("%c",ch[i]);
    		val--;
    		index=i;
    	}
    }
    return 0;
}
