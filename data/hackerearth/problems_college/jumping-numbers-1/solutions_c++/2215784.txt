#include<iostream>
using namespace std;
const long long MOD = 4294967143;
long long dp[279][13][13];
int main()
{
	int t;
	cin >> t;
	for (int tt = 1; tt <= t; tt++)
	{
		int n;
		cin >> n;
		for (int i = 0; i < 279; i++)
		{
			for (int j = 0; j < 13; j++)
			{
				for (int k = 0; k < 13; k++)
				{
					dp[i][j][k] = 0;
				}
			}
		}
		for (int i = 1; i < 11; i++)
		{
			dp[1][i][i] = 1;
		}
		for (int i = 2; i <= n; i++)
		{
			for (int j = 0; j < 11; j++)
			{
				int pr = j - 1;
				int nx = j + 1;
				for (int k = 0; k < 11; k++)
				{
					if (pr>=0)
					dp[i][k][j] += dp[i - 1][k][pr];
					if (nx<=10)
					dp[i][k][j] += dp[i - 1][k][nx];
					dp[i][k][j] %= MOD;
				}
			}
		}
		long long ans = 0;
		for (int j = 1; j < 11; j++)
		{
			int pr = j - 1;			
			int nx = j + 1;
			if (nx <= 10)
			{
				ans += dp[n][j][nx];
			}
			if (pr >= 0)
			{
				ans += dp[n][j][pr];
			}
			ans %= MOD;
		}
		ans %= MOD;
		cout << ans << endl;
	}
}
