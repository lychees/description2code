#include <iostream>
using namespace std;
#include<algorithm>
using namespace std;
inline void scan(int &a){
	char ch=a=0;
	while(ch<33)	ch=getc(stdin);
	while(ch>33)	a=a*10+ch-'0',ch=getc(stdin);
}
long long max(int a,int b){
	return a>b?a:b;	
}
int main()
{
    int n,a,b,c,t,s,e,i,j,P[20010],Q[20010];
    long long res,current;
    scan(t);
    while(t--){
    	res=0;
    	scan(n),scan(a),scan(b),scan(c);
    	for(i=0;i<n;i++){
    		scan(P[i]),scan(Q[i]);
    	}
    	P[n]=Q[n]=2000000000;
    	sort(P,P+n+1);
    	sort(Q,Q+n+1);
    	current=n;
    	current = current*a;
    	res = current;
    	i=0,j=0;
    	while(i<n || j<n){
    		if(P[i]<=Q[j])	current+=(b-a),i++;
    		else	current+=(c-b),j++;
    		res = max(current,res);
    	}
    	printf("%lld\n",res);
    }
    return 0;
}
