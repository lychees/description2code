    #include <iostream>
    using namespace std;
     
    int main()
    {
       int t;
       cin>>t;
       while(t--)
       {
       	long long int a,b,reva=0,revb=0,sum,revs=0;
       	cin>>a>>b;
       	while(a!=0)
       	{ reva=reva*10+a%10;
       		a=a/10;	
       	}
       	while(b!=0)
       	{ revb=revb*10+b%10;
       		b=b/10;	
       	}
       	
       sum=reva+revb;
      
       while(sum!=0)
       	{ revs=revs*10+sum%10;
       		sum=sum/10;	
       	}
       	cout<<revs<<endl;
       
       }
        return 0;
    }
