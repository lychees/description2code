#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,di,ww;
	 int i,j;
	scanf("%d",&n);
	int a[n];
	vector<int>v;
	vector<int>y;
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	sort(a,a+n);
	di=a[1]-a[0];
	for(i=2;i<=di;i++)
	{
		if(a[1]%i==a[0]%i)
		{
			ww=a[1]%i;
		v.push_back(i);
		y.push_back(ww);
		}
	}
	for(j=2;j<n;j++)
	for(i=0;i<v.size();i++)
	{
		//cout<<v[i]<<" ";
		if(v[i]==-1)
		continue;
		if(a[j]%v[i]!=y[i])
		v[i]=-1;
	}
	for(i=0;i<v.size();i++)
	{
		if(v[i]!=-1)
		cout<<v[i]<<" ";
	}
	return 0;
}
