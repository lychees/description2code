#include<bits/stdc++.h>
using namespace std;

#define REP(i,a,b) for(int i=a;i<b;i++)
#define rep(i,n) REP(i,0,n)

#define ll long long
#define ull unsigned ll

char str[1000099];
bool checkPal(int i,int j)
{
    while(i<=j)
    {
        if(str[i]==str[j])
        {
            i++;
            j--;
        }
        else
            return false;

    }
    return true;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        scanf("%s",str);
        int n = strlen(str);
        int i=0;
        int j=n-1;
        int ans=-1;
        while(i<=j)
        {
            if(str[i]==str[j])
            {
                i++;
                j--;
            }
            else
            {
                if(str[i+1]==str[j] && checkPal(i+2,j-1))
                {
                    ans = i;
                    i += 2;
                    j--;
                    break;
                }
                else if(str[i]==str[j-1] && checkPal(i+1,j-2))
                {
                    ans = j;
                    i++;
                    j -= 2;
                    break;
                }

            }
        }
        printf("%d\n",ans);

    }
    return 0;
}
