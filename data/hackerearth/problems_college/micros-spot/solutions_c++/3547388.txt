#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
int dpx[1005][1005],dpy[1005][1005];

int main()
{
	ios_base::sync_with_stdio(false);
	int test;
	cin>>test;
	while(test--)
	{
		int n,m;
		cin>>n>>m;
		
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<m;j++)
			cin>>dpy[i][j];	
		}
		
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<m;j++)
			cin>>dpx[i][j];
		}
		
		for(int j=0;j<m;j++)
		{
			int max=-1;
			for(int i=0;i<n;i++)
			{
				if(dpy[i][j]>max)
				max=dpy[i][j];
			}
			
			for(int i=0;i<n;i++)
			{
				if(dpy[i][j]==max)
				dpy[i][j]=1;
				else
				dpy[i][j]=0;
			}
		}
		
		for(int i=0;i<n;i++)
		{
			int max=-1;
			for(int j=0;j<m;j++)
			{
				if(dpx[i][j]>max)
				max=dpx[i][j];
			}
			
			for(int j=0;j<m;j++)
			{
				if(dpx[i][j]==max)
				dpx[i][j]=1;
				else
				dpx[i][j]=0;
			}
		}
		
		int ans=0;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<m;j++)
			{
				if(dpx[i][j]&&dpy[i][j])
				ans++;
			}
		}
		cout<<ans<<endl;
	}
	
	return 0;
}
