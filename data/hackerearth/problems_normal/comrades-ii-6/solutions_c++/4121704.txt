#include <bits/stdc++.h>
using namespace std;

vector<int> immediateSoldiers[100010];
int dp[100010];
int dfs(int idx) {
	int &res = dp[idx];
	if (res == -1) {
		res = 0;
		for (int soldier = 0; soldier < immediateSoldiers[idx].size(); soldier++) {
			res += dfs(immediateSoldiers[idx][soldier]) + 1;
		}
	}
	return res;
}

int main(){
	int testcases;
	cin>>testcases;
	while(testcases--) {
		int soldiers;
		cin >> soldiers;
		int superior;
		memset(dp, -1, sizeof(dp));
		for(int soldier = 0; soldier <= soldiers; soldier++) {
			immediateSoldiers[soldier].clear();
		}
		for (int soldier = 1; soldier <= soldiers; soldier++) {
			cin>>superior;
			immediateSoldiers[superior].push_back(soldier);
		}
		dfs(0);
		long long fistBumps = soldiers * 1LL * (soldiers - 1) / 2, handShakes = 0;
		for (int i = 1; i <= soldiers; i++) {
			// cout << dp[i] << " ";
			handShakes += dp[i];
			// fistBumps += soldiers - 1 - dp[i];
		}
		// cout << "\n";
		cout << handShakes << " " << fistBumps - handShakes << "\n";
	}
}
