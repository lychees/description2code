#include<stdio.h>
#include<math.h>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
int n,i,j,c,i1,j1,count=0;
scanf("%d",&n);
int arr1[n];
int arr2[n];
i=0;
j=0;
for(i1=0;i1<n;i1++)
{
scanf("%d",&c);
if(c==1)
{

    scanf("%d",&arr1[i++]);
}
else
{
     scanf("%d",&arr2[j++]);
}
}
sort(arr1,arr1+i);
sort(arr2,arr2+j);
for(int i1=0,j1=0;i1<i&&j1<j;)
    {
     if(pow(arr1[i1],2)*2<=pow(arr2[j1],2))       
        {
            i1++;
        }
        else if(pow(arr1[i1],2)>=pow(arr2[j1],2)*2)
       {
           j1++;
       }
       else
       {
           count++;
           i1++;
           j1++;
       }
    }
    printf("%d\n",count);
return 0;
}
