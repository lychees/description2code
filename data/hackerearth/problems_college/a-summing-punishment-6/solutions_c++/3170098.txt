{
    //cout << "Hello World!" << endl;
    ll n,q,x,y,i,j;

    ll a[n+1],b[n+1];
    for(i=1;i<=n;i++){
    cin>>a[i];
    b[i]=0;
    }
    b[1]=a[1];
    b[0]=0;

    cin>>q;

    	cout<<b[y]-b[x-1]<<endl;
    }
