#include <iostream>
#include<cmath>
using namespace std;
bool isprime(long long int a)
{
	
	for(int i=2;i<=sqrt(a);i++)
	{
		if(a%i==0)
			return false;
	}
	return true;
}

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    long long int sum[1000001];
    sum[0]=0;
    sum[1]=0;
    sum[2]=2;
    for(long long int i=3;i<=1000000;i++)
    {
    	if(isprime(i))
    	{
    		sum[i]=sum[i-1]+i;
    	}
    	else
    	sum[i]=sum[i-1];
    	//cout<<sum[i]<<" ";
    }
   /* for(int i=0;i<30;i++)
    	cout<<sum[i]<<" ";*/
    while(t--)
    {
    	long long int a;
    	long long int b;
    	cin>>a;
    	cin>>b;
    	cout<<sum[b]-sum[a-1]<<endl;
    	
    }
    return 0;
}
