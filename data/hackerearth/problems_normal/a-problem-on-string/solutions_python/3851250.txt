for c in word:

        query = raw_input().split()

                count_1 = container[ord(query[0])-97]
                print (count_1*(count_1-1))/2
        else:
                count_1 = container[ord(query[0])-97]
                count_2 = container[ord(query[1])-97]
                print count_1*count_2
