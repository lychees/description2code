#include<bits/stdc++.h>
using namespace std;
long long int IsPrime(long long int number)
{
    int i;
    for (i=2; i*i<=number; i++) {
        if (number % i == 0) return 0;
    }
    return 1;
}

int main()
{
    ios::sync_with_stdio(false);
    long long int i,j,n,t,k,sum;
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>n;
        if(n==1 || n==0)
            cout<<4;
        else if(n%2!=0)
            cout<<(n+1);
        else
        {
            k=IsPrime(n+1);
            if(k==0)
                cout<<n+1;
            else
                cout<<n+2;
        }
        if(i!=t-1)
            cout<<endl;
    }
    return 0;
}
