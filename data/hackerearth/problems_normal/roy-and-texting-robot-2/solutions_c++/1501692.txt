#include <iostream>
using namespace std;

int main()
{
    string s[10];
    s[0]="_0";
    s[1]=".,?!1";
    s[2]="abc2";
    s[3]="def3";
    s[4]="ghi4";
    s[5]="jkl5";
    s[6]="mno6";
    s[7]="pqrs7";
    s[8]="tuv8";
    s[9]="wxyz9";
    int t,i,cl,j;
    cin>>t;
    string str;
    while(t--)
    {
    	cin>>str;
    	i=0,j=1;
    	cl=0;
    	while(i<str.length())
    	{
    		if(s[j].find(str[i])!=string::npos)
    			cl+=s[j].find(str[i])+1;
    		else
    		{
    			cl++;
    			for(j=0;j<10;j++)
    			{
    				if(s[j].find(str[i])!=string::npos)
    				{
    					cl+=s[j].find(str[i])+1;
    					break;
    				}
    			}
    		}
    		i++;
    	}
    	cout<<cl<<endl;
    }
    return 0;
}
