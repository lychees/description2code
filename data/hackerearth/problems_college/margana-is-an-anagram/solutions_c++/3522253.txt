#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	
	char x[1004],y[1004];
	int t,i;
	scanf("%d",&t);
	while(t--){
		int a[26]={0},b[26]={0};
		scanf("%s%s",x,y);
		//cout<<x<<y<<endl;
		
		
		for(i=0;i<strlen(x);i++)
			a[x[i]-'a']++;
			
			for(i=0;i<strlen(y);i++)
			b[y[i]-'a']++;
			
			int flag = 1;
			for(int k = 0;k<26;k++){
				//cout<<a[k]<<" "<<b[k]<<endl;
				if(a[k]!=b[k])
				flag = 0;
			}
			if(flag==1)
			printf("Yes\n");
			else
			printf("No\n");
		
	}
   //cout << "Hello World!" << endl;
    return 0;
}
