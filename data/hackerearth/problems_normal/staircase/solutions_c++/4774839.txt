#include<bits/stdc++.h>
using namespace std;
int n,dp1[10000],dp2[100005];
int jack(int i)
{
   // cout<<i<<endl;
    if(dp1[i]!=0)
        return dp1[i];
        if(i>n)
            return 0;
    if(i==n)
        return 1;
     dp1[i]=jack(i+1)+jack(i+2)+jack(i+3);
     return dp1[i];
}
int jill(int i)
{
    if(dp2[i]!=0)
        return dp2[i];
        if(i>n)
            return 0;
    if(i==n)
        return 1;
    return dp2[i]=jill(i+1)+jill(i+2);
}
int main()
{

    cin>>n;
    if(n<0)
        cout<<"ERROR"<<endl;
    else{
    fill(dp1,dp1+n,0);
    fill(dp2,dp2+n,0);

    int c=jack(0);
    int d=jill(0);
   // for(int i=0;i<=n;i++)
   //     cout<<dp1[i];
    cout<<"Jack-"<<c<<endl;
    cout<<"Jill-"<<d<<endl;}
    return 0;
}
