#include <bits/stdc++.h>
using namespace std;

int t, n, k, l;
int cnt[35][255];
int cur[255];
int tar[255];
char s[35][255], diu[1055];

bool dfs(int id) {
	if (id == n) {
		for (int i =0 ;i < 26; i++) {
			if (cur[i] != tar[i]) return false;
		}
		return true;
	}
	
	if (dfs(id + 1)) return true;
	
	bool flag = false;
	
	for (int i = 0; i < 26; i++) {
		cur[i] += cnt[id][i];
		if (cur[i] > tar[i]) flag = true;
	}
	
	if (!flag) {
		if (dfs(id + 1)) {
			return true;
		}
	}
	
	for (int i = 0; i < 26; i++) cur[i] -= cnt[id][i];
	
	return false;
}
int main()
{
	scanf("%d", &t);
	
	while (t--) {
		scanf("%d", &n);
		memset(cnt, 0, sizeof(cnt));
		memset(tar, 0, sizeof(tar));
		for (int i = 0; i < n; i++) {
			scanf("\n%s", s[i]);
			int m = strlen(s[i]);
			for (int j = 0; j < m; j++) cnt[i][s[i][j] - 'a']++;
		}
		scanf("\n%s", diu);
		int m = strlen(diu);
		for (int i = 0; i < m; i++) tar[diu[i] - 'a']++;
		
		memset(cur, 0, sizeof(cur));
		
		if (dfs(0)) puts("YES");
		else puts("NO");
	}
	
    return 0;
}
