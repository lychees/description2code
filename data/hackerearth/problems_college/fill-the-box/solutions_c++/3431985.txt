#include<bits/stdc++.h>
using namespace std;
int main() 
{
    long long  N,M;
    long  i,a,b,k;
    cin>>N>>M;
    long long  sum = 0;
    for(i=0 ; i< M ; i++)
    {
            cin>>a;
            cin>>b;
            cin>>k;
            sum =sum + (b-a+1)*k;

    }
    cout<<sum/N<<endl;
    return 0;
}
