#include <stdio.h>
int arr[1000001];
int f(int n, int a, int b){
 
if(arr[n]<2) return arr[n];
   if(n<a)
            return arr[n]=0;
 
   else {
   arr[n]=((n>=a) && !f(n-a, a, b)) || ((n>=b) && !f(n-b, a, b));
            return  arr[n];}
 
}
 
int main()
{
   int n,m1,m2,t,i;
   scanf("%d",&t);
   while(t--)
   {
   scanf("%d%d%d",&n,&m1,&m2);
    for(i=0;i<=n;i++)
    arr[i]=2;
    printf("%d\n",f(n,m1,m2));
    }
    return 0;
}
