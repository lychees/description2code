#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int t,a[100010],count=0,max=0;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		cin>>a[i];
	}
	for(int i=1;i<t;i++)
	{
		if(a[i]-a[i-1]==1||a[i-1]-a[i]==1)
		{
			while(i<t && a[i]-a[i-1]==1 || a[i-1]-a[i]==1)
			{
				count++;
				//cout<<a[i]<<" "<<a[i-1];
				i++;
			}
		}
		if(max<count)
		max=count;
		count=0;
	}
	cout<<++max<<endl;
	return 0;
}
