#include <iostream>
#include <bits/stdc++.h>
#include <stack>
#include <stdio.h>
#define mod 1000000007

long long int fstore[100009];
long long int bstore[100009];
long long int nums[100009];
std::stack <long long  int> sf,sb;
using namespace std;
int main()
{
	
    long long int n;
    scanf("%lld",&n);
    while(n--){
    	while(!sf.empty())sf.pop();
    	while(!sb.empty())sb.pop();
    	long long  int ans=-1,index=-1;
    	long long  int z;
    	scanf("%lld",&z);
    	for(long long int i=1;i<=z;i++){
    		scanf("%lld",&nums[i]);
    	}//scanning
    	
    	sb.push(1);
    	bstore[1]=0;
    	for(long long int i=2;i<=z;i++){
    		while((!sb.empty())&&nums[sb.top()]<nums[i])sb.pop();
    		if(sb.empty()){bstore[i]=i-1;sb.push(i);}
    		else{bstore[i]=i-sb.top();sb.push(i);}
    	}
    	sf.push(z);
    	fstore[z]=0;
    	for(long long int i=z-1;i>=1;i--){
    		while((!sf.empty())&&nums[sf.top()]<nums[i])sf.pop();
    		if(sf.empty()){fstore[i]=z-i;sf.push(i);}
    		else{fstore[i]=sf.top()-i;sf.push(i);}
    	}
    	
    	for(long long int i = 1; i <= z; i++){

            long long sum = bstore[i] + fstore[i];

            sum = ( (sum % mod) *  (i) ) % mod;

            if(sum > ans){                

                ans= sum;
                index = i;
            }
        }
    	
    	
    	
    
    	
    	
    	
    	printf("%lld\n",index);
   	
    }
    return 0;
}
