#include<iostream>
#include<cmath>
#include<cstdio>
using namespace std;

typedef long long int ll;
int main(){
int n,q,x,y,sum ;
scanf("%d",&n);
int A[n];
for(int i = 0 ; i < n ; i++){
    cin>>A[i];
}
scanf("%d",&q);
while(q--){
    scanf("%d%d",&x,&y);
    sum = 0 ;
    for(int i = x ; i <= y ; i++){
            sum += A[i-1];
    }printf("%d\n",sum);
}
//std::partial_sum(A,A+n,temp);

  return 0;
}
