#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<string>
using namespace std;
string tobin(int n)
{
    string r;
    if(n==0)
        r="0";
    while(n!=0)
    {
        r=(n%2?"1":"0")+r;
        n/=2;
    }
    return r;
}
bool cons(string a)
{
    bool r=true;
    for(int i=0;i<a.size();i++)
        if(a[i]=='1'&&a[i+1]=='1')
            {r=false;break;}
    return r;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int a;
        cin>>a;
        ll:
        string s=tobin(a);
        //cout<<s<<endl;
        if(cons(s))
            cout<<a<<endl;
        else
            {
                a++;goto ll;
            }
    }
    return 0;
}
