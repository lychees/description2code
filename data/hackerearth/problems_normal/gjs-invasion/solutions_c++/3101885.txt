#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int n,dp[65540][20],ways[65540][20],f[20];
ll int myfind(ll int x,ll int r)//student & room no
{
	
	ll int i,j,t=0,ms=0,ans=0;
	if(r==0)
	{
		dp[x][r]=1;
	}
	if(dp[x][r]==-1)
	{
		for(i=(n-1);i>=0;i--)
		{
			if(!(x&(1<<i)))
			{
				ms++;
				if((i+1)==r)
				continue;
				else
				{
					t=myfind(x|(1<<i),r-1);//possible correct ways
					ans+=t;
					ways[x][r]+=ways[(x|(1<<i))][r-1]+t*(ms-1);
				}
			}
		}
		dp[x][r]=ans;
	}
	//~ cout<<x<<" "<<r<<" "<<dp[x][r]<<endl;
	return dp[x][r];
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int i;
	f[0]=1;
	for(i=1;i<18;i++)
	f[i]=f[i-1]*i;
	ll int test;
	cin>>test;
	while(test--)
	{
		cin>>n;
		ll int j,ans;
		for(i=0;i<=(1<<n);i++)
		{
			for(j=0;j<=n;j++)
			{
				dp[i][j]=-1;
				ways[i][j]=0;
			}
		}
		myfind(0ll,n);//student & room no
		ans=ways[0][n];
		cout<<ans<<endl;
	}
	return 0;
}
