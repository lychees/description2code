t = input()
for xx in range(0, t):
    a, b, c = map(int, raw_input().split())
    ans = a*b*c
    x = min(a, b, c)
    ans -= x
    x = min(a, b)
    y = min(x, c)
    ans -= (y*(c-1))
    ans -= ((x-y)*c)
    x = min(a, c)
    y = min(x, b)
    ans -= (y*(b-1))
    ans -= ((x-y)*b)
    x = min(b, c)
    y = min(x, a)
    ans -= (y*(a-1))
    ans -= ((x-y)*a)
    print ans % 1000000007
