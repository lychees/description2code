#include <bits/stdc++.h>
using namespace std;

int ar[(int)1e6];
map<int, int> m;

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	int t; cin >> t;
	while (t--) {
		int n; cin >> n;
		m.clear();
		for (int i = 0; i < n; ++i)
			cin >> ar[i];
		int l = 0, r = 0;
		for (int i = 0, j = 0; j < n; ++j) {
			while (m[ar[j]]) {
				m[ar[i]]--;
				++i;
			}
			m[ar[j]]++;
			if (r - l < j - i) {
				l = i; r = j;
			}
		}
		cout << l + 1 << " " << r + 1 << endl;
	}	

	return 0;
}
