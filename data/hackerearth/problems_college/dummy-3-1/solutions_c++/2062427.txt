#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll i,j;

int main()
{
    int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        int a[n+1],b[n+1];
        for(i=1;i<=n;i++)
            scanf("%d",&a[i]);
        int t=0;
        for(i=1;i<=n;i++){
            //cout<<i<<" "<<a[i]<<endl;
            b[a[i]]=i;
        }
        for(i=1;i<=n;i++){
            if(a[i]!=b[i]){
                t=1;
                break;
            }
        }
        if(t==1)
            cout<<"not inverse"<<endl;
        else
            cout<<"inverse"<<endl;
    }
    return 0;
}
