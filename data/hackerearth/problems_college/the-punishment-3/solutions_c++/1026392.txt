#include<stdio.h>
int max(int a,int b)
{
    if(a<b)
    {
        return(b);
    }
    return(a);
}
int main()
{
    int t,n,k,i;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d",&n,&k);
        int a=max(k,n-k);
        int b=n-a;
        for(i=0;i<b-1;i++)
        {
            printf("9");
        }
        printf("8");
        for(i=0;i<a-b;i++)
        {
            printf("9");
        }
        for(i=0;i<b-1;i++)
        {
            printf("0");
        }
        printf("1");
        printf("\n");
    }
    return(0);
}
