#include <iostream>
using namespace std;

int main()
{
    int t,n,k,i,j;
    cin>>t;
    long long ans[t];
    for(i=0;i<t;i++)
    {
    	cin>>n>>k;
    	ans[i]=0;
    	long long ht[n],dp[n];
    	for(j=0;j<n;j++) cin>>ht[j];
    	if(ht[0]<=k) dp[0]=ht[0]; else dp[0]=0;
    	ans[i]=dp[0];
    	for(j=1;j<n;j++)
    	{
    		if(ht[j]<=k) dp[j]=dp[j-1]+ht[j];
    		else dp[j]=0;
    		if(dp[j]>ans[i]) ans[i]=dp[j];
    	}
    }
    for(i=0;i<t;i++) cout<<ans[i]<<endl;
    return 0;
}
