#include <cstdio>
#include <cstring>
#include <algorithm>
#define MX 200005
using namespace std;
typedef long long LL;

int a[MX], N;
int val[1000005];
LL ans;

void add(int x) {for (; x < 1000005; x += x & -x) val[x]++;}
int sum(int x) {int ret = 0; for (; x; x -= x & -x) ret += val[x]; return ret;}
int main() {
	
	int i, k;
	scanf("%d", &N);
	for (i = 0; i < N; i++) scanf("%d", a + i);
	for (i = 0; i < N; i++) {
		scanf("%d", &k);
		ans += i - sum(k);
		add(a[i]);
	}
	int bf = 1;
	for (i = 2; 1LL * i * i <= ans; i++) {
		if (ans % i == 0) bf = 0;
	}
	if (ans == 1) bf = 0;
	puts(bf ? "MAGIC INVERSION" : "SIMPLE INVERSION");
	return 0;
}
