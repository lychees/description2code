#include<bits/stdc++.h>
using namespace std;
int main(){
	int t,n;
	scanf("%d",&t);
	while(t--) {
		scanf("%d",&n);
		long arr[n];
		for(int i=0;i<n;i++) {
			scanf("%ld",&arr[i]);
		}
		sort(arr,arr+n);
		int i,j;
		scanf("%d %d" ,&i,&j);
		printf("%ld\n",arr[j-1]+arr[n-i]);
	}
	return 0;
}
