spd = map(float, raw_input().split())
mlg = map(float, raw_input().split())
fl = float(raw_input())

d = []

for i, j in enumerate(mlg):
    d.append((fl/j)*spd[i])

b = str(max(d)).split('.')
if len(b[1]) < 3:
    while len(b[1]) < 3:
        b[1] += '0'

print "%s.%s" % (b[0], b[1][:3])
