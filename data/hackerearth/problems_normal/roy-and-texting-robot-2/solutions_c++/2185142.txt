#include <iostream>
#include <string>
using namespace std;
 
int getVal(char x) {
	int val = -1;
	switch(x) {
		case '.':
		case 'a':
		case 'd':
		case 'g':
		case 'j':
		case 'm':
		case 'p':
		case 't':
		case 'w':
		case '_':
			val = 1;
			break;
		case ',':
		case 'b':
		case 'e':
		case 'h':
		case 'k':
		case 'n':
		case 'q':
		case 'u':
		case 'x':
		case '0':
			val = 2;
			break;
		case '?':
		case 'c':
		case 'f':
		case 'i':
		case 'l':
		case 'o':
		case 'r':
		case 'v':
		case 'y':
			val = 3;
			break;
		case '!':
		case '2':
		case '3':
		case '4':
		case '5':
		case '6':
		case 's':
		case '8':
		case 'z':
			val = 4;
			break;
		case '1':
		case '7':
		case '9':
			val = 5;
			break;
	}
	return val;
}
 
int getKey(char x) {
	int key = -1;
	switch(x) {
		case '.':
		case ',':
		case '?':
		case '!':
			key = 1;
			break;
		case 'a':
		case 'b':
		case 'c':
			key = 2;
			break;
		case 'd':
		case 'e':
		case 'f':
			key = 3;
			break;
		case 'g':
		case 'h':
		case 'i':
			key = 4;
			break;
		case 'j':
		case 'k':
		case 'l':
			key = 5;
			break;
		case 'm':
		case 'n':
		case 'o':
			key = 6;
			break;
		case 'p':
		case 'q':
		case 'r':
		case 's':
			key = 7;
			break;
		case 't':
		case 'u':
		case 'v':
			key = 8;
			break;
		case 'w':
		case 'x':
		case 'y':
		case 'z':
			key = 9;
			break;
		case '_':
			key = 0;
			break;
		default:
			key = x - '0';
	}
	return key;
}
 
int main()
{
	int t, length, steps, i, last, current;
	cin>>t;
	string s;
	while(t--) {
		cin>>s;
		length = s.size();
		last = 1;
		steps = 0;
		for(i=0; i<length; i++) {
			steps += getVal(s[i]);
			current = getKey(s[i]);
			if(current != last) {
				steps++;
				last = current;
			}
		}
		cout<<steps<<endl;
	}
    return 0;
}
