#include<stdio.h>
#include<iostream>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<stdlib.h>
#include<vector>
#include<map>
#include<set>

using namespace std;

main()
{
	map<int,string>m;
	m[2]="cde";
	m[7]="acf";
	m[6]="b3";
	m[9]="c6a";
	
	int t;
	cin>>t;
	
	while(t--)
	{
		char a[10000];
		cin>>a;
		
		
		int l=strlen(a);
		
		for(int i=0;i<l;i++)
		{
			cout<<m[a[i]-'0'];
		}
		cout<<endl;
	}
	
}
