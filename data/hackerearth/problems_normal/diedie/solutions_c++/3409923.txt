const long long MOD = 1E9 + 7;
long long bin(long long x, long long y) {
    if (y == 0) {
        return 1;
    }
    long long u = bin(x, y/2);
    if (y % 2 == 0) {
        return u*u % MOD;
    }
    else {
        return u*u % MOD * x % MOD;
    }
}


ll pow(ll a,ll b){
	ll res  = 1;
	while(b){
		if(b%2==1)
			res = (res * a)%MOD;
		a = (a*a)%MOD;
		b = b / 2;
	}
	return res;
}
