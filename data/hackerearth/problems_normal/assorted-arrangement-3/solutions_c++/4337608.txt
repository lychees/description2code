#include<iostream>
using namespace std;
#define ull unsigned long long
ull arr[1000000];
ull mm[10000];
int tm[1000000];
int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=0;i<m;i++) cin>>mm[i];
	for(int i=0;i<n;i++) cin>>arr[i];
	for(int i=0;i<n;i++) tm[i]=arr[i];
	for(int i=0;i<n;i++) arr[i]=mm[arr[i]-1];
	for(int i=1;i<n;i++)
	{
		ull k=arr[i-1]/arr[i];
		while(1)
		{
			int f=(arr[i]*k>arr[i-1]);
			for(int ii=tm[i];ii<m;ii++)
			{
				if(!f) break;
			f=(arr[i]*k)%mm[ii];
			if(!f) break;
			}
			if(f) {arr[i]=arr[i]*k;break;}
			k++;
		}
	}
	ull ans=arr[0];
	for(int i=0;i<n;i++) ans=max(ans,arr[i]);
	cout<<ans;
	return 0;
}
