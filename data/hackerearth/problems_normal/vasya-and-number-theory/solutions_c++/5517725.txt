#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int max(int a,int b)
{return a>b?a:b;
}

int main()
{	vector<int> v;
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{int tmp;
	cin>>tmp;
	v.push_back(tmp);}
	sort(v.begin(),v.end());
	vector<int> count(n,1);
	for(int i=1;i<n;i++)
	for(int j=0;j<i;j++)
	if (v[i]%v[j]==0)
	count[i]=max(count[i],count[j]+1);
	int m=1;
	for(int i=0;i<n;i++)
	m=max(m,count[i]);
	if (m==1)
	cout<<-1<<endl;
	else cout<<m<<endl;
    
    return 0;
}
