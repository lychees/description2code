#include<cstdio>
long long int a[1000007]={0},t,i,x,ans,j;
int main()
{
	scanf("%lld",&t);
	a[1]=1;
	for(i=2;i<=1000002;i++)
	{
		for(j=i*i;j<=1000002;j*=i)
		{
			a[j]=1;
		}
	}
	while(t--)
	{
		scanf("%lld",&x);
		ans=0;
		for(i=1;i<x;i++)
		{
			if(a[i]==1&&a[x-i]==1)
			{
				ans=1;
				break;
			}
		}	
		if(ans)
		printf("Yes\n");
		else
		printf("No\n");	
	}
	return 0;
}
