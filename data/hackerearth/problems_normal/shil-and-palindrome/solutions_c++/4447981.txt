#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int a[26]={0};
    string s;
    cin>>s;
    for(int i=0;i<s.size();i++){
    	a[s[i]-97]++;
    }
    int p=0,i;
   	if(s.size()%2==0){
   		for(i=0;i<26;i++){
   			if(a[i]%2 != 0){
   				cout<<"-1"<<endl;
   				break;
   			}
   			for(int j=1;j<=a[i]/2;j++){
   				
   				s[p]=(char)(97 + i);
   				s[s.size()-1-p]=(char)(97 + i);
   				p++;
   			}
   		}
   	}
   	else{
   		int flag=0;
   		for(i=0;i<26;i++){
   			
   			if(a[i]%2 != 0){
   				flag++;
   				if(flag==2){
   					cout<<"-1"; 
   					break;
   				}
   				else if(flag==1) {
   					s[s.size()/2]=(char)(97+i);
   					a[i]--;
   				}
   			}
   			for(int j=1;j<=a[i]/2;j++){
   				
   				s[p]=(char)(97 + i);
   				s[s.size()-1-p]=(char)(97 + i);
   				p++;
   				}
   			}
   		}
   	
   	if(i==26) cout<<s;
    return 0;
}
