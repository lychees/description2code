#include <bits/stdc++.h>
using namespace std;

int phi[123456];
bool primes[123456];

int main()
{
    for(int i=1; i<=100000; i++) phi[i]=i;
    
    for(int i=2; i<=100000; i++)
    {
    	if(primes[i]==0)
    	{
    		for(int j=i; j<=100000; j+=i)
    		{
    			if(i!=j)
    				primes[j]=1;
    			phi[j]/=(i);
    			phi[j]*=(i-1);
    		}
    	}
    }
    
    int t;
    scanf("%d",&t);
    
    while(t--)
    {
    	int a,b;
    	scanf("%d%d",&a,&b);
    	
    	int x=min(a,b);
    	long long ans=0;
    	
    	for(int i=1; i<=x; i++) if(a%i==0 && b%i==0) ans+=phi[i];
    	printf("%lld\n",ans);
    }
    return 0;
}
