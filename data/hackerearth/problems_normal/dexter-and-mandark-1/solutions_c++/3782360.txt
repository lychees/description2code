#include <bits/stdc++.h>
using namespace std;
long long int dp[1005][1005];
#define mod 1000000007
int n,m;
long long int fun(int a,int b)
{   
	if(a<m*b)
	return 0;
	
	if(a+b==n)
	{
		if(a > b)
		return 1;
		return 0;
	}
	
	long long int &ret=dp[a][b];
	if(ret!=-1)
	return ret;
	
	ret=0;
//	if(a+1 >=b*m)
	ret+=fun(a+1,b);
	ret%=mod;
	
//	if(a >=(b+1)*m)
	ret+=fun(a,b+1);
	
	ret%=mod;
	return ret;
	
}
int main()
{
    int i,j,t;
    
   cin>>t;
   while(t--)
   {
   cin>>n>>m;
   memset(dp,-1,sizeof(dp));
   //cout<<dp[4][4][6];
   cout<<fun(0,0)<<endl;
    }
    
    
}
