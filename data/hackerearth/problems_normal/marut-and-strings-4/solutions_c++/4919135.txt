#include<bits/stdc++.h>
using namespace std;
int main()
{
int  t;
cin>>t;
if(t<1 || t>10)
cout<<"Invalid Test"<<endl;
else
{
while(t--)
{
char s[10000];
cin>>s;
int p=strlen(s);
int i,j,f=0,cnt1=0,cnt2=0;
for(i=0;i<p;i++)
{
if(s[i]>=97 && s[i]<=122)
{
cnt1++;
f=1;
}
else if(s[i]>=65 && s[i]<=90)
{
cnt2++;
f=1;
}
}
if(f==0 || p>100)
cout<<"Invalid Input"<<endl;
else
cout<<min(cnt1,cnt2)<<endl;
}
}
return 0;
}
