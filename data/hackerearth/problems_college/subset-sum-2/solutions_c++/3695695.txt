#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
/* find sum of all subset of 1 to n */
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		long long ans = 1;
		for(int i=0 ; i<n-1 ; i++)
			ans = ( ans << 1 )%mod;
		ans = ( ans * ((n*(n+1))/2)%mod)%mod;
		printf("%lld\n",ans);
	}
	
	return 0;
}
