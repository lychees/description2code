#include <bits/stdc++.h>

using namespace std;

//my orders are absolute
int main()
{
    int t;
    cin >> t;
    int n;
    int i;
    int flag = 0;
    while(t--)
    {
    	cin >> n;
    	flag = 0;
    	int a[n + 1];
    	int b[n + 1];
    	for(i = 1; i <= n ; i++)
    	{cin >> a[i];
    	b[a[i]] = i ; 
    	}
    	for(i = 1 ; i <= n ; i++)
    	{
    		if(a[i] != b[i])
    		{
    			flag = 1;
    			break;
    		}
    	}
    	if(flag == 1)
    	{
    		cout << "not inverse" <<endl;
    	}
    	else
    	  cout << "inverse"<<endl;
    	
    }
    return 0;
}
