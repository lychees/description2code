t=input()
while t>0:
    sum=0
    n=input()
    a=map(int,raw_input().split())
    a.sort(reverse=True)
    for i in range(1,n):
        sum+=a[0]-a[i]
    temp1=sum
    for i in range (1,n-1):
        temp=temp1-(n-i)*(a[i-1]-a[i])
        sum=(sum+temp) 
        temp1=temp
    sum=sum%1000000007
    y=(sum*a[0])%1000000007
    print y
    t-=1
