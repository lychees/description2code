#include<bits/stdc++.h>
using namespace std;

int main()
	{
		long long int N,k;
		scanf("%lld %lld",&N,&k);
		long long int a[N];
		for(int i=0;i<N;i++)
			scanf("%lld",&a[i]);
		sort(a,a+N);
		int not_seen=0;
		for(int i=0;i<N-1;i++)
			{
				if(a[i+1]-a[i]<=k)
					continue;
				else
					{
						not_seen=1;
						cout<<a[i]<<endl;
						break;
					}
			}
		if(not_seen==0)
			cout<<a[N-1]<<endl;
		return 0;
	}
