#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    cin>>s;
     int ans=0;
    for(int i=0;i<s.length();i++)
    { 
    	ans++;
    	for(int j=i+1;j<s.length();j++)
    	{
    		if(s[j]==s[j-1])
    			ans++;
    		else
    			break;
    	}
    }
   
    cout<<ans<<endl;
    return 0;
}
