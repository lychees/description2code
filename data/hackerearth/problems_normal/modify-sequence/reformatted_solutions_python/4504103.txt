t = input()
a = map(int, raw_input().split())

i = 0
f = 1
while i < t-1:
    if a[i] < a[i+1]:

        a[i+1] = a[i+1]-a[i]
        a[i] = 0

    elif a[i] > a[i+1]:
        f = 0
        break
    else:
        a[i] = 0
        a[i+1] = 0
        i = i+1
    i = i+1
if f == 0:
    print 'NO'
elif a.count(0) == t:
    print 'YES'
else:
    print 'NO'
