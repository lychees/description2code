#include<iostream>
using namespace std;
int main()
{
register long int r;
register long int c;
register int tc;
register long int e;
register long int o;
register int i;
cin>>tc;
while(tc--)
{
cin>>r>>c;
e=o=0;
i=1;
if(r<=c)
{
while(i<=r)
    if(i%2==0)
    e=e+i++;
else
    o=o+i++;
}
else
{
while(i<=r)
{
	if(i<c)
	if(i%2==0)
	e=e+i++;
	else
	o=o+i++;
	else
	{
		if(i%2==0)
		e=e+c;
		else
		o=o+c;
		i++;
	}
}
}
cout<<e<<" "<<o<<"\n";
}


return 0;
}
