#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int i,j,c=0,n,ar[100000],x=0,y=0,f=0,sum=0;
        cin>>n;
        for(i=0;i<n;i++)
            cin>>ar[i];
        sort(ar,ar+n);
        for(i=1;i<n;i++)
        {
            f=0;
            if(ar[i]>ar[i-1])
            {
                x=ar[i]-ar[i-1];
                if(x>=5)
                {
                    if(x%5==0)
                    {
                        c=c+ x/5;
                    }
                    else
                    {
                        f=1;
                        y=x%5;
                        if(y==4 || y==3)
                            c=c+2;
                        if(y==2 || y==1)
                            c=c+1;
                    }
                    if(f==1)
                    {
                        c=c+(x-y)/5;
                    }
                }
                else{
                    y=x%5;
                    if(y==4 || y==3)
                        c=c+2;
                    if(y==2 || y==1)
                        c=c+1;
                }
                sum=sum+x;
                if(i<n-1){
                    ar[i+1]=ar[i+1]+sum;
            }
        }
        }
        cout<<c<<endl;
    }
    return 0;
}
