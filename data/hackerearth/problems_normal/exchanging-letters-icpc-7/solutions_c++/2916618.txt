using namespace std;
#define MOD 1000000007
long long C[2222][2222];
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int i,j,k,sum,n;
	int x;
	for(i=1;i<=2000;i++)
	{
		C[i][0]=C[i][i]=1;
		for(j=1;j<i;j++)
		{
			C[i][j]=(C[i-1][j]+C[i-1][j-1])%MOD;
		}
	}
	cin>>n>>k>>sum;
	for(i=0;i<k;i++)
	{
		cin>>x;
		sum-=x;
	}
	sum=abs(sum);
	if((k-sum)%2==1||sum>k)
	{
		cout<<0<<endl;
	}
	else
	{
		j=(k-sum)/2;
		cout<<C[k][j]<<endl;
	}
//	system("pause");
	return 0;
}
