#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int a,b,c,f,m,l,ans;
		cin>>a>>b>>c;
		f=max(a,max(b,c));
		l=min(a,min(b,c));
		if(a>l&&a<f)
		m=a;
		else if(b>l&&b<f)
		m=b;
		else m=c;
		
		ans=max(f-m,m-l);
		cout<<ans-1<<endl;
	}
	return 0;
}
