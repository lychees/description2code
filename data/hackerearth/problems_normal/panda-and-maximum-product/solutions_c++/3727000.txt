#include<bits/stdc++.h>
using namespace std;
int main(){

	long long int max2,max1,min2,min1,a;
    int n;
   // max2=max1=-INT_MAX;
   // min2=min1=INT_MAX;
    cin>>n;
    for(int i=0;i<n;i++)
       {
       	cin>>a;
       	if(a>max1)
       	   {max2=max1;max1=a;}
       	else if (a>max2)
       	   max2=a;
       	else if(a<min1)
       	   {min2=min1;min1=a;}
       	else if(a<min2)
       	   min2=a;
       }
      // cout<<max1<<" "<<max2<<" "<<min1<<" "<<min2<<endl;
       cout<<max(max2*max1,min2*min1);
       return 0;
}
