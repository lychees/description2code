#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <math.h>
using namespace std;

int main()
{
    int n;
    scanf("%d", &n);
    int a[n];
    for(int i=0;i<n;++i)
    {
    	scanf("%d", &a[i]);
    }
    sort(a,a+n);
    int i, j, temp, b[n+1], k = 0;
    i = 0;
    j = n-1;
    long long sum = 0;
    while(k < n)
    {
    	b[k] = a[i];
    	++k;
    	b[k] = a[j];
    	++i;
    	--j;
    	++k;
    }
    /*
    printf("\n");
    for(int i=0;i<n;++i)
    {
    	printf("%d ", b[i]);
    }
    */
    for(i = 0;i<n - 1;++i)
    {
    	sum += abs(b[i] - b[i+1]);
    }
    sum += abs(b[n-1] - b[0]);
    printf("%lld\n", sum);
    return 0;
}
