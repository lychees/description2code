#include <iostream>
using namespace std;

int main()
{
	bool is_prime;

	int count = 1;
	int my_prime = 2;
	int n;
	cin>>n;

	for(int i = 3; count <n; i += 2)
	{ 
		is_prime = true;
		for(int j = 3; j * j <= i && is_prime; j += 2)
			if(i % j == 0) is_prime = false;
		if(is_prime) {
			++count;
			my_prime = i;
		}
	}

	cout << my_prime;
	return 0;
}
