#include <cstdio>

using namespace std;

int main()
{
	int t;
	
	scanf("%d", &t);
	
	long long int n, b, q;
	
	while (t--) {
		scanf("%lld%lld", &n, &b);
		q = n / b;
		
		if ((double) (n % b) >= (double) b / (double) 2) {
			printf("%lld\n", (q + 1) * b);
		} else {
			printf("%lld\n", q * b);
		}
	}
	
	return 0;
}
