#include<iostream>
 
using namespace std;
 
long long a[50002];
long long b[50002];
long long profit[11];
 
int f(long long c[],long long x,long long y);
 
int main()
{
    long long t,i,n,j,k,start,cp,sellingday;
    long long sp;
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>n;
        for(j=1;j<=n;j++)
        {
            cin>>a[j];
            b[j]=a[j]+b[j-1];
        }
        j=0;
        start=1;
        //profit=0;
        while(start<n)
        {
            sellingday=f(a,start,n);
           // cout<<sellingday<<"\t";
 
            cp=b[sellingday]-b[start-1]-a[sellingday];
            sp=(sellingday-start)*a[sellingday];
            profit[i]=profit[i]+sp-cp;
 
            start=sellingday+1;
 
        }
 
    }
    for(i=0;i<t;i++)
        cout<<profit[i]<<"\n";
    return 0;
}
int f(long long c[],long long x,long long y)
{
    long long k;
    long long maxi=0;
    long long sellingday;
    for(k=x;k<=y;k++)
    {
        if(a[k]>maxi)
        {
            maxi=a[k];
            sellingday=k;
 
        }
 
    }
    return sellingday;
}
