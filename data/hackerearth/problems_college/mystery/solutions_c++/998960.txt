#include <iostream>
using namespace std;

int main()
{
    long long int a,b,c,sum=0;
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%Ld%Ld%Ld",&a,&b,&c);
    	sum=0;
    	if(c==1)
    	{
    		printf("%Ld\n",b);
    		continue;
    	}
    	else if(c==0)
    	{
    		printf("%Ld\n",a);
    		continue;
    	}
    	else
    	{
    		for(int i=0;i<c-1;i++)
    		{
    			sum=a+b;
    			a=b;
    			b=sum;
    		}
    		printf("%Ld\n",sum);
    	}
    }
    return 0;
}
