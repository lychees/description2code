#include<bits/stdc++.h>
using namespace std;
string str;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int aa,bb,ab,ba;
    	cin>>aa>>bb>>ab>>ba;
    	int m = min(ab,ba);
    	ab=ba=m;
    	
    	int i,j;
    	for(i=0;i<aa/2;++i)
    		str+="aa";
    	for(i=0;i<ab;++i)
    		str+="ab";
    	for(i=0;i<bb/2;++i)
    		str+="bb";
    	if(aa&1) str+="a";
    	else if(bb&1) str+="b";
    	cout<<str;
    	reverse(str.begin(),str.end());
    	cout<<str<<endl;
    	str.clear();
    	
    }
}
