while True:
    try:
        line=raw_input()
        if line[0:2]=='//':
            print line
            
        else:
            x=line.split('//')
            if len(x)==1:
                print line.replace('->','.')
                
            else:
                x[0]=x[0].replace('->','.')
                print "//".join(x)

    except EOFError:
        break
