#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        int store = 0,ans = 0;
        for(int i=n;i<=1000001;i++){
            int take = i;
            store = 0;
            ans = 0;
            while(take!=0){
                int keep = take%2;
                take = take/2;
                if(keep == store && store == 1){
                    ans = 1;
                }
                store = keep;
            }
            if(ans == 1){
                continue;
            }else {
                cout<<i<<"\n";
                break;
            }
        }
    }
    return 0;
}
