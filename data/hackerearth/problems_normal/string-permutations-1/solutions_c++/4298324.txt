#include <iostream>
#include <string>
#include <map>
using namespace std;

int main()
{
    int n,i;cin>>n;
    string s[n];
    map<string,int> ma;
    for(i=0;i<n;i++)	cin>>s[i],ma[s[i]]++;
    auto it=ma.begin();
    
    int maxm=0,dup=0;
    for(;it!=ma.end();it++){
    	if(it->second>maxm)
    		maxm=it->second;
    }
    it=ma.begin();
    for(;it!=ma.end();it++){
    	if(it->second==maxm)
    		dup++;
    }
    cout<<maxm*dup<<endl;
    return 0;
}
