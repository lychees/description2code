def palindrome(a, b):
    count = 0
    for i in range(a, b+1):
        i = str(i)
        if i == i[::-1]:
            count = count+1
    return count


tc = int(raw_input())
for i in range(tc):
    a, b = raw_input().split(' ')
    a = int(a)
    b = int(b)
    print palindrome(a, b)
