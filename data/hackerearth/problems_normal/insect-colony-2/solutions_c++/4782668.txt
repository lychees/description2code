#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,i,even=0,odd=0;
    	cin>>n;
    	long long int a[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		if(a[i]%2==0)
    		{
    			even++;
    		}
    		else
    		{
    			odd++;
    		}
    	}
    	if(odd%2==0)
    	{
    		cout<<"Yes\n";
    	}
    	else
    	{
    		cout<<"No\n";
    	}
    }
}
