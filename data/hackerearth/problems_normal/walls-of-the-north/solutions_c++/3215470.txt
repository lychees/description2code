#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	vector<long long>a(n);
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	pair<long,long>u(a[n-1],1);
    	pair<long,long>l(a[n-1],1);
    	for(int i=n-2;i>=0;i--)
		{
			pair<long,long>nu(0,0);
    			pair<long,long>nl(0,0);
			if(a[i]>=u.first)
			{
				nl.first=a[i];
				nl.second=u.second+1;//cout<<nl.second;
			}
			if(a[i]<=l.first)
			{
				nu.first=a[i];
				nu.second=l.second+1;//cout<<nu.second;
			}
			if(nl.second>=l.second)
			l=nl;
			if(nu.second>=u.second)
			u=nu;
		}   
		cout<<(u.second>l.second?u.second:l.second)<<endl;
    }
    return 0;
}
