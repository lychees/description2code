#include <bits/stdc++.h>
using namespace std;
int search(int a[],int n,int x)
{
	for(int i=0;i<n;i++)
	if(a[i]==x) return i;
	return -1;
}
int main() {
	// your code goes here
	int n,k,x;
	cin>>n>>k;
	int c[n];
	for(int i=0;i<n;i++)
	{
		cin>>c[i];
	}

	int i=0,s=n;
	while(i<k&&s>0)
	{
		if(search(c,n,s)!=n-s)
		{
			swap(c[search(c,n,s)],c[n-s]);
			i++;
			s--;
		}
		else s--;
	}
	for(int i=0;i<n;i++)
	cout<<c[i]<<" ";
	return 0;
}
