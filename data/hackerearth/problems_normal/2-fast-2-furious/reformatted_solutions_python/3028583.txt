n = int(raw_input())


def speed_variation(l):
    highest = 0
    for i in range(n-1):
        high = abs(l[i+1]-l[i])
        if (high >= highest):
            highest = high

    return (highest)


d = map(int, raw_input().split())
b = map(int, raw_input().split())
d_high = speed_variation(d)
b_high = speed_variation(b)
if (d_high > b_high):
    print "Dom\n", d_high
elif (b_high > d_high):
    print "Brian\n", b_high
else:
    print "Tie"
