for _ in range(input()):
	n=input()
	r,c,x=map(int,raw_input().split())
	s=r+c
	if s%2:
		if x%2:
			f=1
		else:
			f=2
	else:
		if x%2==0:
			f=1
		else:
			f=2
	ans=1
	for i in range(n-1):
		r,c,x=map(int,raw_input().split())
		s=r+c
		if s%2:
			if x%2 and f==1:
				continue
			if x%2==0 and f==2:
				continue
			ans=0
			
		else:
			if x%2==0 and f==1:
				continue
			if x%2 and f==2:
				continue
			ans=0
			
	if ans:
		print "Yes"
	else:
		print "No"
