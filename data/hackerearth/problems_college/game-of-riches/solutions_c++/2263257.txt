#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,g,mywealth,wealth[100],sum=0,i,temp,k,j,mygroup;
    	cin>>n>>g;
    	cin>>wealth[1];
    	mywealth=wealth[1];
    	for(i=2;i<=n;i++)
    	{
    		cin>>wealth[i];
    		
    	}
    	for(i=1;i<=n;i++)
    	for(j=i+1;j<=n;j++)
    	{
    		if(wealth[i]<wealth[j])
    		{
    			temp=wealth[i];
    			wealth[i]=wealth[j];
    			wealth[j]=temp;
    		}
    	}
    	for(i=1;i<=n;i++)
    	{
    		if(wealth[i]==mywealth)
    		{
    			k=i;
    			break;
    		}
    	}
    	if(k%g==0)
    	mygroup=g;
    	else
    	mygroup=k%g;
    	
    	for(i=1;i<=n;i++)
    	{
    		if(wealth[i]>mywealth && i%g==0)
    		{
    			sum++;
    		
    		}
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
