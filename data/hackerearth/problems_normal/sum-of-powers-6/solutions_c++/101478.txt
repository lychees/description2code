#include<stdio.h>
#include<iostream>
#include<math.h>
using namespace std;
#define LL long long int
int main()
{
    int t,i,j,n;
    scanf("%d",&t);
    int number[1000005]={0};
    int limit=log(1000005)+1;
    for(i=1;i*i<=1000005;i++)
    {
        int power=i;
        for(j=2;j<=limit;j++)
        {
            power=power*i;
            if(power<=1000004)
                number[power]=1;
            else
                break;
        }
    }
    //printf("yes");
    int num[1000005];
    int pos=0;
    for(i=0;i<1000005;i++)
    {
        if(number[i])
            num[pos++]=i;
    }
    //printf("%d\n",pos);
    while(t--)
    {
        scanf("%d",&n);
        bool flag=false;
        for(i=0;i<pos;i++)
        {
            int NUM=num[i];
            int x=n-NUM;
            int lb=0,ub=pos-1;
            while(lb<=ub){
            int mid=(lb+ub)/2;
            if(num[mid]==x)
            {
                flag=true;
                goto End;
                //break;
            }
            else if(num[mid]>x)
                ub=mid-1;
            else
                lb=mid+1;
            }
        }
        End:
        if(flag)
            printf("Yes\n");
        else
            printf("No\n");
    }
    return 0;
}
