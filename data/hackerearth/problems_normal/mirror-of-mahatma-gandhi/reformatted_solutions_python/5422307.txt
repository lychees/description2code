N = input()
for i in range(N):
    k = raw_input().split()
    m = list()
    for j in k[0]:
        m.append(j)
    h = ['2', '3', '4', '5', '6', '7', '9']
    f = 0
    for q in m:
        for y in h:
            if q == y:
                f = 1
    if f == 1:
        print 'NO'
        continue
    l = list(m)
    m.reverse()
    if l == m:
        print 'YES'
    else:
        print 'NO'
