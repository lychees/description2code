			/*
			      *** PATIENCE ABOVE PERFECTION ***
			        "When in doubt, use brute force. :D"

			        -Founder of alloj.wordpress.com
			*/
			#include<bits/stdc++.h>
			using namespace std;
			#define ll long long int
			int main()
			{
		      int t;
		      cin>>t;
		      while(t--)
		      {
		      	ll w,b;
		      	cin>>b>>w;
  				   ll x,y,z;
  				   cin>>x>>y>>z;
  				   ll ans=0;
  				   ll T1=z+x,T2=z+y;
			      ans=min(y,T1)*w+min(x,T2)*b;
			      cout<<ans<<endl;
		      }
			   return 0;
			}
