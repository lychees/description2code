#include <iostream>
using namespace std;

int main()
{
	int n,t,i,j,flag;
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	flag=0;
   	for(i=0;i<=n;i++)
   	{
   		if(i%5==0&&(n-i)%3==0) {flag=1;break;}
   	}
   	if(flag) {
   		for(j=0;j<n-i;j++) cout<<5;
   		for(j=0;j<i;j++) cout<<3;
   		cout<<endl;
   	}
   	else cout<<"-1\n";
   }
}
