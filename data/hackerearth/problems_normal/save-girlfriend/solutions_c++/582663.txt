#include<iostream>
#include<cstring>
#include<cstdio>
#include<cstdlib>
#include<math.h> 
using namespace std;
int main()
{
     int t,i,j,s[10005];
     scanf("%d",&t);
     while (t--)
     {
     	int a,b;
     	scanf("%d%d",&a,&b);
     	for(i=0;i<=b-a+1;i++)
     		s[i] = 0;
     	for (i=1;i*i<=b;i++)
     	{
     		int sq = i*i;
     		for (j = ( (a-1) / i + 1) * i; j <= b; j += i)
     		{
     			if (j<sq)
     				continue;
     			else if (j==sq)
     			{
     				s[j-a]++;
     			}
     			else s[j-a] += 2;
     		}
     	}

     	int mx,ans=0;
     	for(i=a;i<=b;i++)
     	{
     		//cout<<s[i-a]<<endl;
     		if (i==a)
     			mx = s[i-a];
     		else if (mx<s[i-a])
     			mx = s[i-a];
     	}
     //	cout<<"max="<<mx<<endl;
     	for(i=a;i<=b;i++)
     	if (s[i-a]==mx)
     	ans++;
     	printf("%d\n",ans+1);
     }
return 0;
}
