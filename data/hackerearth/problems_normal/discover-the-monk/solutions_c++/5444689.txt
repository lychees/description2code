ll bS(ll arr[], ll value, ll min, ll max){
  ll pos = -1;

  while (max >= min && pos == -1) {
    ll mid = min+(max-min)/2;

    if(arr[mid] ==  value){
      pos = mid;
    }else if(arr[mid] < value){
      min = mid +1;
    }else if(arr[mid] > value){
      max = mid -1;
    }

  }
  return pos;
}
