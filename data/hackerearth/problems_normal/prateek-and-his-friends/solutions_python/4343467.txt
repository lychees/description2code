t= int(raw_input())

for i in range(t):
	n, x = map(int,raw_input().split())
	a= []
	for j in range(n):
		a.append(int(raw_input()))
	st=0
	flag = 0
	s =0
	for j in range(n):
		s+= a[j]
		while s>x:
			s-=a[st]
			st+=1
		if(s==x):
			flag=1
			break;
	if (flag==1):
		print('YES')
	else:
		print('NO')
