#include <iostream>
#include<vector>
#include<cmath>
#include<algorithm>
using namespace std;

int main()
{
	int N;
	cin>>N;
	int len=pow(2.0,N);
	int input[len+1][len+1];
	for(int i=0;i<=len;i++)
	{
		for(int j=0;j<=len;j++)
		{
			input[i][j]=0;
		}
	}
	for(int i=2;i<=len;i++)
	{
		for(int j=1;j<i;j++)
		{
			cin>>input[i][j];
		}
	}
    vector<int> oldd;
    vector<int> neww;
    for(int i=len;i>0;i--)
    {
    	oldd.push_back(i);
    }
    int check=oldd.size();
    //for(int k=0;k<check;k++)
    //{
    //	cout<<oldd[k]<<" ";
    //}
    cout<<endl;
    while(check>1)
    {
    	int i=0;
    	int j=1;
    	while(j<check)
    	{
    		if(input[oldd[i]][oldd[j]]==1)
    		{
    			neww.push_back(oldd[i]);
    		}
    		else neww.push_back(oldd[j]);
    		i+=2;
    		j+=2;
    	}
    	check=neww.size();
    	oldd.clear();
    	copy(neww.begin(),neww.end(),oldd.begin());
    	neww.clear();
    	//check=oldd.size();
    //	for(int k=0;k<check;k++)
    //	{
    //		cout<<oldd[k]<<" ";
	//	 }
		 //cout<<oldd[0]<<" ";
		 //cout<<oldd[1]<<" ";
    //	cout<<endl;
	 }
    cout<<oldd[0]<<endl;
    return 0;
}
