t = input()
while (t > 0):
    t = t-1
    n = input()
    primefac = {}
    d = 2
    while d*d <= n:
        while (n % d) == 0:
            if d not in primefac:
                primefac[d] = 0
            primefac[d] += 1
            n //= d
        d += 1
    if n > 1:
        primefac[n] = 1
    if len(primefac) == 0:
        print 1
    else:
        sols = 1
        for w, v in primefac.items():
            sols = sols*(primefac[w]+1)
        print sols-1
