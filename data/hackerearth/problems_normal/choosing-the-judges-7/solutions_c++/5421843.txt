#include <iostream>
using namespace std;

int a[10005];
int dp[10005];
int main() {

	int n, t;
	cin >> t;
	for(int l = 0 ; l < t ; ++l) {
		cin >> n;
		for(int i = 0 ; i < n ; ++i) {
			cin >> a[i];
			dp[i] = 0;
		}
		dp[0] = a[0];
		dp[1] = max(a[0], a[1]);
		for(int i = 2 ; i < n ; ++i) {
			dp[i] = max(dp[i - 2] + a[i], dp[i - 1]);
		}
		cout << "Case " << l + 1 << ": " <<  dp[n - 1] << endl;
	}
    return 0;
}
