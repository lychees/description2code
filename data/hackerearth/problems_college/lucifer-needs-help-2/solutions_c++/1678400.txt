#include <cstdio>
using namespace std;

int N, M, i, j, k;

long long arr[1000000 + 1], p = 0;

int main()
{
    scanf("%d", &N);
    for(int i = 1; i <= N; ++i)
    	scanf("%lld", arr + i);
    for(int i = N; i > 1; --i)
    	arr[i] -= arr[i - 1];
    
    scanf("%d", &M);
    while(M--) {
    	scanf("%d%d%d", &i, &j, &k);
    	arr[i] += k;
    	arr[j + 1] -= k;
    }
    p = 0;
    for(int i = 1; i <= N; ++i) {
    	p += arr[i];
    	printf("%lld\n", p);
    }
    return 0;
}
