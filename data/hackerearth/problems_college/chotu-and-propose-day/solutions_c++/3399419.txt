#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t,aa,bb,ab,ba,n,m,i,j,k,l,p;
    cin >> t;
    while(t--)
    {
    	cin >> aa >> bb >> ab >> ba;
    	string s="";
		if(aa%2!=0&&aa!=0)
		{
			s=s+"aa";
			aa--;
		}
		else if(bb!=0)
		{
			if(bb%2==0)
			{
				s=s+"bbbb";
				bb=bb-1;
			}
			else
			{
				s=s+"bb";
				bb--;
			}
		}
		while(bb>1)
		{
			s="bb"+s+"bb";
			bb=bb-2;
		}
		if(ab>=ba)	n=ba;
		else	n=ab;
		for(i=1;i<=n;i++)
		{
			s="ab"+s+"ba";
		}
		while(aa>1)
		{
			s="aa"+s+"aa";
			aa=aa-2;
		}
		cout << s << endl;
    }
    return 0;
}
