#include<bits/stdc++.h>
using namespace std;
#define N 1000001
int main()
{
  //const int N=ie5+10;
  int p[N];
  int n,l,r,q;
  cin>>n;
  //int p[n];
  for(int i=0;i<n;i++)
  cin>>p[i];
  sort(p,p+n);
  cin>>q;
  int ub,lb;
  while(q--)
  {
    cin>>l>>r;
    ub=upper_bound(p,p+n,r)-p;
    lb=lower_bound(p,p+n,l)-p;
    cout<<ub-lb<<endl;
  }
  return 0;
}
