#include <iostream>
using namespace std;

int main()
{



{
   int test,i,j=0,k;
   int n[100],p[100],po[100][10000];
   cin>>test;
   for(i=0;i<test;i++)
   {
       cin>>n[i]>>p[i];
       for(j=0;j<n[i];j++)
        {
            cin>>po[i][j];
        }
   }
   
   for(k=0;k<test;k++)
   {
       int temp=0;
       for(i=0;i<n[k];i++)
       {
           if(po[k][i]==p[k])
                temp=1;
       }
       if(temp==1)
            cout<<"YES ";
       else
            cout<<"NO 0"<<endl;
            
       int count=0;
       for(i=0;i<n[k];i++)
       {
           if(po[k][i]>p[k])
                count++;
       }
       if(temp==1)
            cout<<count<<endl;
       
   }
   return 0;
}
}
