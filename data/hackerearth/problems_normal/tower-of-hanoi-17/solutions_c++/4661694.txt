#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    int t;
    cin>>t;
    for(int j = 0 ; j < t ; j++)
    {
        pair<long long,long long> p[205];
        cin>>n;
        for(int i = 0 ; i < n ; i++)
        {
            cin>>p[i].first>>p[i].second;
        }
        sort(p,p+n);
        reverse(p,p+n);
       
        long long dp[205];
        memset(dp,0,sizeof(dp));
        dp[0] = p[0].second;
        for(int i = 1 ; i < n ; i++)
        {
            dp[i] = p[i].second;
            for(int k = 0 ; k < i ; k++)
            {
                if( p[i].second < p[k].second && p[i].first < p[k].first)
                {
                    dp[i] = max(dp[i], dp[k]+p[i].second);
                }
            }
        }
        long long max = INT_MIN;
        for(int k = 0 ; k < n ;k++)
        {
            if(dp[k] > max)
            {
                max = dp[k];
            }
        }
        cout<<max<<endl;

    }
    return(0);
}
