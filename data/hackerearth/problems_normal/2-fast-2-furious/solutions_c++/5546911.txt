#include<iostream>
using namespace std;
int main(){
	int n,i,max,max1,x;
	cin>>n;
	int a[n];
	cin>>a[0];
	for(i=1,max=0;i<n;i++){
		cin>>a[i];
		x=a[i]-a[i-1];
		if(x<0)
		x=x*-1;
		if(max<x)
		max=x;
	}
	cin>>a[0];
	for(i=1,max1=0;i<n;i++){
		cin>>a[i];
		x=a[i]-a[i-1];
		if(x<0)
		x=x*-1;
		if(max1<x)
		max1=x;
	}
	if(max==max1)
	cout<<"Tie\n"<<max;
	else if(max1>max)
	cout<<"Brian\n"<<max1;
	else
	cout<<"Dom\n"<<max;
	return 0;
}
