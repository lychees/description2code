int main()
{
    ll n, i, ones = 0, cnt, f, k, ans, j;
    cin >> n;
    ll A[n+1];
    for(i = 0 ;i < n; i++) {
        cin >> A[i];
    }
    cnt = 0, f = 0, ans = 1;
    for(j = 0;j < n ; j++) {
        if(A[j] == 1) {
            break;
        }
    }
    if(j == n) {
        cout << "0\n";
        return 0;
    }
    for(i = j + 1 ; i < n; i++ ) {
        if(A[i] == 1) {
            ans *= (1 + cnt);
            f  = 0;
            cnt = 0;
        } else {
            cnt++;
        }
    }
    cout << ans << endl;

    return 0;
}
