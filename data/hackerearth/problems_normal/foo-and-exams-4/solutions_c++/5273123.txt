#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int tc;
	cin>>tc;
	while(tc--)
	{
		long long a,b,c,d,k;
		cin >> a >> b >> c >> d >> k;
		
		long long ans=0;
		long long left=0,right=ceil(double(pow(double(1000000000000000000)/a,0.3333333333334)));
		while(left<=right)
		{
			long long mid=(left+right)/2;
			if(a*mid*mid*mid + b*mid*mid +c*mid +d <=k)
			{
				ans = mid;
				left=mid+1;
			}
			else
			{
				right=mid-1;
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
