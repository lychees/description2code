#define MOD 1000000007
typedef long long LL;
typedef pair<int, int> ii;
int main(){
    ios_base::sync_with_stdio(false);
    int N;
    cin >> N;
    vector <int> A(N);
    for (int i = 0; i < N; i++)
    	cin >> A[i];
    priority_queue <int> Q;
    for (int i = 0; i < N; i++) {
    	Q.push(A[i]);
    	if (Q.size() < 3) {
    		cout << "-1\n";
    		continue;
    	}
    	int largest = Q.top();
    	//cout << largest << " ";
    	Q.pop();
    	int Sec_largest = Q.top();
    	//cout << Sec_largest << " ";
    	Q.pop();
    	int Thi_largest = Q.top();
    	//cout << Thi_largest << " ";
    	Q.push(largest);
    	Q.push(Sec_largest);
    	LL product = static_cast<LL>(largest) * Sec_largest * Thi_largest;
    	cout << product << "\n";
    }
    return 0;
}
