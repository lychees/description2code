#include<bits/stdc++.h>
using namespace std;
int main()
{
	int tc,N,M,x1,x2,x3;
	long int discount[1001];
	cin>>tc;
	while(tc--)
	{		
		cin>>N>>M;	
		while(N--)
		{
			for(int i=0;i<M+1;i++)
			{
			discount[i]=0;
			}
			for(int i=1;i<M+1;i++)
			{
				cin>>x1>>x2>>x3;
			    discount[i]=(100-x1)*(100-x2)*(100-x3);
			}
			int min=1;
			for(int i=1;i<M+1;i++)
			{
			if(discount[min]>discount[i])
			{
				min=i;
			}
			}
			cout<<min<<" ";
		}
		cout<<endl;
	}
}
