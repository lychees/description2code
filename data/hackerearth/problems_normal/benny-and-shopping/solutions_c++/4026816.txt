#include <iostream>
#include <string>

using namespace std;

int main()
{
    int n,i,j,l;
    cin>>n;
    string s[n+1],a[n+1];
    for(i=0;i<=n;i++)
    	getline(cin,s[i]);
	
    	
    for(i=1;i<=n;i++){
    	j=0;l=s[i].length();
    	while(s[i][j] != '$' && j<l)
    		j++;
    	j++;
    	while(s[i][j]==' ' || s[i][j]=='0')
    	 j++;
    	while(j < l){
    		
    		if(isalpha(s[i][j]))
    			break;
    		else if (s[i][j]>='0'&&s[i][j]<='9'){
    		  a[i].push_back(s[i][j]);
    		}
    		j++;
    			
    	}
    }
    
    for(i=1;i<=n;i++){
    	    if(a[i].length()==0)
    	    cout<<"$0"<<endl;
    	    else
    		cout<<'$'<<a[i]<<endl;
    }
    	
    return 0;
}
