#include<bits/stdc++.h>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    int i,t;
    cin>>t;
    string str;
    while(t--)
    {
        cin>>str;
        if(str.find('|')==str.npos)
        {
            cout<<"Bob"<<endl;
            continue;
        }
        str = str + "|";
        int a = 0,b = 0;
        for(i=0;i<str.length();i++)
        {
            if(str[i]=='_')
            {
                b++;
            }
            else
            {
                //cout<<b<<" ";
                a = a^b;
                //cout<<a<<endl;
                b=0;
            }
        }
        if(a!=0)
            cout<<"Alice"<<endl;
        else cout<<"Bob"<<endl;
    }
    return 0;
}
