#include <iostream>

using namespace std;

int main()
{
    string s;
    long q;
    long long a,b;
    cin>>s;
    cin>>q;
    for(long i=0;i<q;i++)
    {
        cin>>a>>b;
        a=a%s.size();
        b=b%s.size();
        if(a==0||b==0)
        {
            if(a==0 && b==0)
                cout<<"Yes"<<endl;
            else if(a==0)
            {
                if(s[s.size()-1]==s[b-1])
                    cout<<"Yes"<<endl;
                else
                    cout<<"No"<<endl;
            }
            else if(b==0)
            {
                if(s[s.size()-1]==s[a-1])
                    cout<<"Yes"<<endl;
                else
                    cout<<"No"<<endl;
            }


        }
       else  if(s[a-1]==s[b-1])
            cout<<"Yes"<<endl;
       else
            cout<<"No"<<endl;
    }
    return 0;
}
