arr = map(int, raw_input().split())
q = input()
for i in xrange(1, len(arr)):
    arr[i] = arr[i]+arr[i-1]
for z in xrange(q):
    qr = input()
    l = 0
    r = len(arr)
    ans = -1
    while l < r:
        mid = (l+r)/2
        if arr[mid] == qr:
            ans = mid+1
            break
        elif arr[mid] > qr:
            r = mid
            ans = mid+1
        else:
            l = mid+1
    print ans
