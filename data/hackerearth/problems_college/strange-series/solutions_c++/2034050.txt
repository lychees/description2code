{
	for(int i=0;i<=50;i++)dp[i][0]=1;

		{
			if(dp[i][j]==0)
			{
				if(i==j)dp[i][j]=1;
				else if(j==1)dp[i][j]=i;
				else dp[i][j]=dp[i-1][j]+dp[i-1][j-1];
			}
		}
}
