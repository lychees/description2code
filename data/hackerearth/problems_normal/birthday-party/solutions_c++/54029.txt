#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;
#define ni(x) scanf("%lld",&x)
int N,M;
int valid[1001],cnt[1001];
void solve(){
	ni(M),ni(N); int x;
	memset(valid,0,4004);
	memset(cnt,0,4004);
	valid[0]=1;
	for(int i=0;i<N;i++){
		ni(x);
		for(int j=(1000-x);j>=0 ;j--){
			if(valid[j])valid[j+x]=1;
		}
	}
	valid[0]=0;
	
	
	
	for(int i=0;i<M;i++){
		ni(x); cnt[x]++;
	}
	int ans=0;
	for(int i=0;i<=1000;i++){
		if(valid[cnt[i]]) ans++;
	}
	printf("%d\n",ans);
	
	
}
int main()
{
    int tc; cin>>tc;
    while(tc--){
    	solve();
    }
    return 0;
}
