#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[1000010]={0};
int main()
{
	ll int i,j,t;
	for(i=2;i<1000010;i++)
	{
		if(a[i]!=0)
		continue;
		for(j=1;j*i<1000010;j++)
		{
			t=j*i;
			while((t%i)==0)
			{
				a[j*i]++;
				t/=i;
			}
		}
	}
	for(i=1;i<1000010;i++)
	{
		a[i]+=a[i-1]; //Prefix sum
	}
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int x,y,ans;
		scanf("%lld%lld",&x,&y);
		ans=a[y]-a[x-1];
		printf("%lld\n",ans);
	}
	return 0;
}
