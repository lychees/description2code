#include <iostream>
using namespace std;

int main()
{
	int N,level;
	cin >> N >>level;
	int arr[N];
	int score=0,leave=0;
	for(int i=0;i<N;i++)
	   cin >> arr[i];
	for(int i=0;i<N;i++){
		if(arr[i]<=level && leave<=1)
		   score++;
		else if(leave == 1)
		  break;
		else
		  leave++;
	}
	cout <<score << endl;
    return 0;
}
