#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,m;
    scanf("%d %d",&m,&n);
    int *a,i,sum=0;
    a=(int*)malloc(sizeof(int)*m);
    priority_queue <int> q;
    for(i=0;i<m;i++)
    {
        scanf("%d",&a[i]);
        q.push(a[i]);
    }
    for(i=0;i<n;i++)
    {
        sum+=q.top();
        int x=q.top();
        q.pop();
        x--;
        if(x)
            q.push(x);
    }
    cout<<sum;
    return 0;
}
