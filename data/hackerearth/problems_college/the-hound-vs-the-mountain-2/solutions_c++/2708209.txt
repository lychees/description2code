#include <iostream>
#include <cmath>
using namespace std;

long int dec(long int b)
{
	long int m=1,d=0,r;
	while(b!=0)
	{
		r=b%10;
		d=d+r*m;
		m=m*2;
		b=b/10;
	}
	return d;
}

int main()
{
	int t;
	cin >>t;
	while(t--)
	{
		long int a,b;
		cin>>a>>b;
		a=dec(a); b=dec(b);
		if(b*log(a)<=a*log(b))
			cout<<"The Hound"<<endl;
		else
			cout<<"The Mountain"<<endl;
	}

    return 0;
}
