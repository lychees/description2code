#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
    int t,n,j;
    scanf("%d",&t);
    while(t--){
         scanf("%d",&n);
         long long a[n],b[n],maxm;
         for(int i=0;i<n;i++){
                  scanf("%lld",&a[i]);
                 // cout<<"frst "<<i<<endl;
         }
          for(int i=0;i<n;i++){
                  scanf("%lld",&b[i]);
                  // cout<<"second "<<i<<endl;
         }
         maxm=0;
         for(int i=0;i<n;i++){
           j=i;
            while(j>=i && b[j]>=a[i]&&j<n){
                if((j-i)>maxm){
                    maxm = j-i;
                   // cout<<"hee"<<maxm<<endl;
                }
                j++;
            }
         }
        cout<<maxm<<endl;
    }
    return 0;
}
