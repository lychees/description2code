#include<bits/stdc++.h>
#define mp make_pair
#define endl '\n'
#define pi pair<int,int>
#define int long long
#define mod (1000*1000*1000 +7)
using namespace std;

int n,t=1,a[250],b[250],dpp[(int)1e6],dpn[(int)1e6],spp[(int)1e6],spn[(int)1e6];

int f(int i,int j)
{
    if(j<0)return spn[-j]%mod;
    return spp[j]%mod;
}

 main()
{
    ios_base::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
//    sieve(10000);


  // cin>>t;
   while(t--)
   {
       cin>>n;
       int s=0,l=0;
       for(int i=0;i<n;i++)cin>>a[i],s+=a[i];
       for(int i=0;i<n;i++)cin>>b[i],l+=b[i];
       int k;
       cin>>k;
       s=max(l,s);
       for(int i=0;i<=k;i++)spp[i]=1,spn[i]=1;

       for(int i=n-1;i>=0;i--)
       {
         for(int j=0;j<=s;++j)
         {
             dpp[j]=(f(i+1,j+a[i])+f(i+1,j-b[i])+f(i+1,j)+f(i+1,j+a[i]-b[i]))%mod;

             dpn[j]=(f(i+1,-j+a[i])+f(i+1,-j-b[i])+f(i+1,-j)+f(i+1,-j+a[i]-b[i]))%mod;
         }

         for(int j=0;j<=s;++j)
         {
             spp[j]=dpp[j];

             spn[j]=dpn[j];
        }


       }

       cout<<dpp[0];

   }
   return 0;
}
