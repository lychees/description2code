#include<bits/stdc++.h>
using namespace std;
string str(int x){
	char m[10],p=0;
	while(x!=0){
		m[p++]=(x%10+'0');
		x/=10;
	}
	if(p==0)
	   m[p++]='0';
	string ans="";
	for(int i=p-1;i>=0;i--)
	   ans+=m[i];
	return ans;
}
unsigned long long conc(string b,string a){
	unsigned long long f=1,num=0;
	for(int i=a.size()-1;i>=0;i--){
		num+=(a.at(i)-'0')*f;
		f*=10;
	}
	for(int i=b.size()-1;i>=0;i--){
		num+=(b.at(i)-'0')*f;
		f*=10;
	}
	return num;
}
bool cmp(string a,string b){
	 unsigned long long num1=conc(a,b),num2=conc(b,a);
	 if(num1>num2)
	    return false;
	  else
	    return true;
}
int main(){
	int n;
	scanf("%d",&n);
	string a[n];
	int a1[n];
	int cnt=n;
           for(int i=0;i<cnt;i++)
              scanf("%d",&a1[i]);
	for(int i=0;i<cnt;i++){
		a[i]=str(a1[i]);
	}
	sort(a,a+n,cmp);
	string ans="";
	for(int j=n-1;j>=0;j--){
		ans+=a[j];
	}
	cout<<ans;
	return 0;
}
