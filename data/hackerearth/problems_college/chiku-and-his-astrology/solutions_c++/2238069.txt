#include <iostream>
using namespace std;

int main()
{
    int i,j,k,t;
    char str[200001];
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	i=0;
    	k=0;
    	while(str[i])
    	{
    		k=k+(str[i]-48);
    		i++;
    	}
    	j=i-k;
    	if(k>j)
    	cout<<k-j<<endl;
    	else
    	cout<<j-k<<endl;
    }
    return 0;
}
