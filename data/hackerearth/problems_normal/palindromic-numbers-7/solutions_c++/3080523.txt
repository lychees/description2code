#include<iostream>
using namespace std;
int main(){
    int a,b,test,temp,count,rev;

    cin>>test;
    while(test>0)
    {
        test--;
        count=0;
        cin>>a>>b;
        for(a;a<=b;a++)
        {

            temp=a;
            rev=0;
            while(temp!=0)
            {
                rev*=10;
                rev += temp%10;

                temp/=10;
            }
            if(rev==a)
            {
                count++;
                
            }
        }
        cout<<count<<endl;
    }
    return 0;
}
