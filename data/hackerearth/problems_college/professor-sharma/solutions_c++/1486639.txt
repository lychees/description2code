#include <iostream>
using namespace std;
int t,x,y;
int s[1000011];
int main()
{
	scanf("%d",&t);
	for (int test=1;test<=t;test++){
		scanf("%d%d",&x,&y);
		s[0]=1; s[1]=1;
		for (int e=1;e<=100000;e++){
			int nho=0;
			for (int i=1;i<=s[0];i++){
				int sum=s[i]*x+nho;
				s[i]=sum%10;
				nho=sum/10;
			}
			while(nho){
				s[0]++;
				s[s[0]]=nho%10;
				nho/=10;
			}
			int ss=0;
			for (int i=s[0];i>=1;i--){
				ss=ss*10+s[i];
				if (ss>=y) break;
			}
			if (ss==y){
				printf("Case %d: %d\n",test,e);
				break;
			}
		}
	}
    return 0;
}
