#include<bits/stdc++.h>
using namespace std;

vector<pair<int,int> >a;


vector<pair<int,int> > merge(const vector<pair<int,int> >&x)
{
	int n=x.size();
	if(n==1)
		return x;
	vector<pair<int,int> >res;
	res.push_back(x[0]);
	for(int i=1;i<n;i++)
	{
		if(x[i].first<=res.back().second)
		{
			res.back().second=max(res.back().second,x[i].second);
		}
		else
			res.push_back(x[i]);
	}
	return res;
}
int main()
{
	int n;
	cin>>n;
	while(n--)
	{
		int x,y,d;
		cin>>x>>y>>d;
		a.push_back(make_pair(x,x+d));
	}
	sort(a.begin(),a.end());
	a=merge(a);
	
	int ans=0;
	for(int i=0;i<a.size();i++)
	{
		ans+=(a[i].second-a[i].first+1);
	}
	cout<<ans<<endl;
	
	
	
	
	
	
	
	
	
	
	
	
	
}
