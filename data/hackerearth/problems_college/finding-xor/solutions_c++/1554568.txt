#include <iostream>
#include <string>
using std::cout;
using std::cin;
using std::string;

int main()
{
	string s1,s2;
	int T;
	cin>>T;
	while(T)
	{
		  string s3;
		  cin>>s1>>s2;
		  int n,m;
		  n=s1.length();
		  m=s2.length();
		  while(n>0 || m>0)
		  {
		  	if(n>0 && m>0 )
		  	{
		  	 s3+=((s1[n-1]-'0')^(s2[n-1]-'0'))+'0';
		  	 n--;
		  	 m--;
		  	}
		  	else if(n>0 && m<0)
		  	{
		  		s3+=(s1[n-1]-'0');
		  		n--;
		  	}
		  	else if(m>0 && n<0)
		  	{
		  		s3+=(s2[m-1]-'0');
		  		m--;
		  	}
		  }
		  string::reverse_iterator rit=s3.rbegin();
		  for(;rit!=s3.rend();rit++)
		  {
		  	  cout<<*rit;
		  }
		  cout<<"\n";
		  s3.clear();
		  T--;
	}
}
