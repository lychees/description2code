#include <iostream>
using namespace std;

int main()
{
	int i;
	int n,m;
	cin>>n>>m;
	long long int a[n];
	long int b[m];
	for(i=0;i<m;i++)
		cin>>b[i];
	a[0]=1;
	
	a[1]=1;
	for(i=2;i<n;i++)
		a[i]=a[i-1]+a[i-2];
	for(i=0;i<m;i++)
		{
			for(int j=0;j<n;j++)
			{
				if(a[j]==b[i])
				{
					a[j]=0;
					break;
				}
			}
		}
		long long int s=0;
		for(i=0;i<n;i++)
			s+=a[i];
		cout<<s;
	
	return 0;
}
