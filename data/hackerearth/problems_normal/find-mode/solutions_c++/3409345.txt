#include <iostream>
using namespace std;

int main()
{
    int t;
   
    cin>>t;
    while(t--)
    {
    	 int a[1000001]={0};
    	int n;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		int t;
    		cin>>t;
    		a[t]++;
    	}
    	int max=0;
    	for(int i=0;i<1000001;i++)
    	{
    		if(a[i]>max)
    		max=a[i];
    	}
    	for(int i=1000001;i>0;i--)
    	{
    		if(a[i]==max)
    		cout<<i<<" ";
    	}
    	cout<<"\n";
    }
    return 0;
}
