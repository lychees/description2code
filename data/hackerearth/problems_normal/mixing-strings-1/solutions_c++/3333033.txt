#include <bits/stdc++.h>
using namespace std;

void unir( string &nw, string o ){
	for( int i=max(0, (int)(nw.size()-o.size()) ), j, k; i<=nw.size(); i++){
		for( j=0, k=i;j<o.size() && k<nw.size() && nw[k]==o[j] ; j++, k++);
		if( k==nw.size() ){
			for( ; j<o.size() ; j++)
				nw+=o[j];
			break;
		}
	}
}

int main()
{
    for( int n; cin>>n; ){
    	vector<string> pal(n);
    	vector<int> per(n);
    	
    	for( int i=0; i<n; i++){
    		per[i]=i;
    		cin>>pal[i];
    	}
    	
    	int res = 200;
    	do{
    		string nuevo = pal[ per[0] ];
    		
    		for( int i=1; i<n; i++){
    			unir(nuevo, pal[ per[i] ] );
    		}
    		res = min(res, (int)nuevo.size());
    		
    	}while( next_permutation(per.begin(), per.end()) );
    	
    	cout<<res<<"\n";
    }
    return 0;
}
