def maxsum(c, i, w):
    if (i <= 0 or w <= 0):
        return 0

    if (c[i] > w):
        return maxsum(c, i-1, w)
    if (d[i][w] != -1):
        return d[i][w]
    d[i][w] = max(maxsum(c, i-1, w), c[i] + maxsum(c, i-1, w-c[i]))
    return d[i][w]
