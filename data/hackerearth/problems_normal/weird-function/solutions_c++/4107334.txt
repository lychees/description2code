#include<bits/stdc++.h>
using namespace std;
#define ll long long int 
#define mod 1000000007
vector<int> v[10005];
ll ans[10005];
ll po(ll b,ll e)
{  ll res=1;
   while(e)
   {  if(e%2)
      res=(res*b)%mod;
      b=(b*b)%mod;
      e/=2;
   }
   return res;
}
ll megh[10005];
void pre()
{  int i,j; 
   for(i=1;i<=10000;i++)
    {  for(j=i;j<=10000;j+=i)
      {  v[j].push_back(i);
	  }
	}
	vector<int>:: iterator it;
	for(i=1;i<=10000;i++)
	{    for(it=v[i].begin();it!=v[i].end();it++)
	   {   ans[i]+=(po((long long)*it,(long long)i));
	      ans[i]%=mod;
	   }
	}
	megh[0]=0;
	megh[1]=ans[1];
	for(i=2;i<=10000;i++)
	{   megh[i]=megh[i-1]+ans[i];
	   megh[i]%=mod;
	}
}

int main()
{    


    pre();
    int t;
    scanf("%d",&t);
    while(t--)
    {    int a,b;
        scanf("%d %d",&a,&b);
        cout<<(megh[b]-megh[a-1]+mod)%mod<<endl;
        
	}
	return 0;
}
