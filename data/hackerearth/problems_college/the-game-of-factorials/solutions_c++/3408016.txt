#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int arr[2000];
		arr[0]=1;
		int m=1;
		int carry=0;
		int x=0;
		for(int i=1;i<=n;i++){
			for(int j=0;j<m;j++){
				x=arr[j]*i+carry;
				arr[j]=x%10;
				carry=x/10;
			}
			while(carry>0){
				arr[m]=carry%10;
				carry/=10;
				m++;
			}
		}
		long long int sum=0;
		for(int i=m-1;i>=0;i--){
			sum+=arr[i];
		}
		cout<<sum<<endl;
	}
	return 0;
}
