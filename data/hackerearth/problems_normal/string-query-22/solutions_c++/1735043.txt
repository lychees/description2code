#include<bits/stdc++.h>
using namespace std;

int main()
{
      char a[200000];
      map <char ,vector<int> > s;
      char c=getchar_unlocked();
      int i=0;
      while(c!='\n')
      {
            a[i]=c;
            s[c].push_back(i);
            i++;
            c=getchar_unlocked();
            if(c=='\n')
                  break;
      }
      int q,k;
      cin>>q;
      while(q--)
      {
            cin>>k>>c;
            k--;
            a[s[c][k]]='*';
            s[c].erase(s[c].begin()+k);
      }
      for(int j=0;j<i;j++)
      {
            if(a[j]!='*')
                  cout<<a[j];
      }
      cout<<"\n";
      return 0;
}
