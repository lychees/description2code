test_str = ('a', 'e', 'i', 'o', 'u')
test_cases = int(raw_input())

while test_cases:
    input_string = raw_input()
    input_string_stripped = input_string.split('www.')[1]
    input_string_stripped = input_string_stripped.split('.com')[0]
    count = 0
    for i in input_string_stripped:
        if i not in test_str:
            count = count + 1
    test_cases -= 1
    print "{}/{}".format(count+int(4), len(input_string))
