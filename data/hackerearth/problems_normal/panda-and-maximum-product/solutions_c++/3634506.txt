#include <iostream>
using namespace std;
int main()
{
   int n;
   cin>>n;
   long long int a[n],b[n];
   long long int m1=0,m2=0,n1=0,n2=0,x=0,y=0,i,r;
   for(i=0;i<n;i++){
   	cin>>r;
   	if(r>0){
   		x++;
   		a[x-1]=r;
   	}
   	else{
   		y++;
   		b[y-1]=r;
   	}
   	   }
   	   if(x>1){
   	   	if(a[0]>=a[1])
   	   	{
   	   		m1=a[0];
   	   		m2=a[1];
   	   	}
   	   	else{
   	   		m2=a[0];
   	   		m1=a[1];
   	   	}
   	   	for(i=2;i<x;i++){
   	   		if(a[i]>=m1){
   	   			m2=m1;
   	   			m1=a[i];
   	   		}
   	   		else if(a[i]>m2)
   	   		m2=a[i];
   	   	}
   	   }
   if(y>1){
   	   	if(b[0]<=b[1])
   	   	{
   	   		n1=b[0];
   	   		n2=b[1];
   	   	}
   	   	else{
  	   		n2=b[0];
   	   		n1=b[1];
   	   	}
   	   	for(i=2;i<y;i++){
   	   		if(b[i]<=n1){
   	   			n2=n1;
   	   			n1=b[i];
   	   		}
   	   		else if(b[i]<n2)
   	   		n2=b[i];
   	   	}
   	   }
   	 long long int p,q;
   	 p=m1*m2;
   	 q=n1*n2;
   	 if(p>q)
 	 cout<<p;
   	 else
   	 cout<<q;
 
    return 0;
}
