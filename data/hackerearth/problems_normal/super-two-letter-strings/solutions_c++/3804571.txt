#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		ll n, i, j, dp[10007], p;
		scanf("%lld%lld",&n,&p);
		dp[0] = 0;
		dp[1] = 1;
		for( i = 2 ; i <= p ; i++ )
		{
			dp[i] = 2 * dp[i - 1];
			if(dp[i] >= mod)
				dp[i] %= mod;
		}
		for( ; i <= n ; i++ )
		{
			dp[i] = 0;
			for( j = i - 1 ; j >= i - p && j >= 0 ; j-- )
			{
				dp[i] += dp[j];
			}
			if(dp[i] >= mod)
				dp[i] %= mod;
		}
		/*for( i = 0 ; i <= n ; i++ )
			printf("%lld ",dp[i]);*/
		printf("%lld\n",dp[n]);
	}
	return 0;
}
