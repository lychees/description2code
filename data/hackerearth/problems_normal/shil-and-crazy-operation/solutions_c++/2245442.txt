typedef unsigned long long int uint64;

int a[100005];
int b[100005];
vector<int>comp,idx;
int vis[100005];
int perm[100005];

void bfs(int node){
	queue<int>q;
	q.push(node);
	while(!q.empty()){
		node=q.front();
		q.pop();
		vis[node]=1;
		comp.pb(node);
		if(vis[perm[node]])
		continue;
		q.push(perm[node]);
	}
}

int main(){
	int n,i;
	int64 t;
	scanf("%d%lld",&n,&t);
	for(i=1;i<=n;i++)
	scanf("%d",&a[i]);
	for(i=1;i<=n;i++)
	scanf("%d",&perm[i]);
	
	for(i=1;i<=n;i++){
		if(vis[i])
		continue;
		comp.clear();
		bfs(i);
		int compsz=comp.size();
		idx=comp;
		int st=t%compsz;
		int sz=0;
		while(sz!=compsz){
			b[comp[sz++]]=a[comp[st++]];
			st%=compsz;
		}
	}
	for(i=1;i<=n;i++)
	printf("%d ",b[i]);
	return 0;
}
