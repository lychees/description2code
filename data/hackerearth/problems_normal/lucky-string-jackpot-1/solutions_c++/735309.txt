

#include <iostream>
using namespace std;

int main()
{
	long long t, n;
    cin >> t;
    while(t-- > 0) {
    	cin >> n;
    	
    	long long d=1, k=2, f=n;
    	while(n-k*d > 0) {
    	
    		n=n-k*d;
    		k=k*2;
    		d=d+1;
    	}
    	d = d-1;
    	
    	if(n >= 2*d+1) {
    		cout << 2*d+1 << endl;
    	} else if (n >= d+1 && d+1 > 2*d-1 ){
    		cout<<d+1<<endl;
    	} else {
    		cout << 2*d-1 << endl;
    	}
    }
}
