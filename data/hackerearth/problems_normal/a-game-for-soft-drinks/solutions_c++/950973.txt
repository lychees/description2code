#include <bits/stdc++.h>
using namespace std;
#define sc(n) scanf("%d",&(n))
#define scll(n) scanf("%lld",&(n))
int dp[101][101][101] = {0};
void pre(){
	dp[0][0][0] = -1;
	for(int i=1;i<=100;i++){
		if(i%2==0){
			dp[i][i][0] = -1;
			dp[0][i][i] = -1;
			dp[i][0][i] = -1;
			dp[i][0][0] = -1;
			dp[0][i][0] = -1;
			dp[0][0][i] = -1;
		}
		else{
			dp[i][i][0] = 1;
			dp[0][i][i] = 1;
			dp[i][0][i] = 1;
			dp[i][0][0] = 1;
			dp[0][i][0] = 1;
			dp[0][0][i] = 1;
		}
	}
}
int recur(int n1,int n2,int n3,int turn){
	if(dp[n1][n2][n3]!=0){
		return turn*dp[n1][n2][n3];
	}
	if(n1==0&&n2==0&&n3==0){
		return turn*(-1);
	}
	if((n1==n2&&n3==0)||(n2==n3&&n1==0)||(n1==n3&&n2==0)){
		int p = max(n1,max(n2,n3));
		if(p%2==0){
			return turn*(-1);
		}
		else return turn;
	}
	if((n1==0&&n2==0&&n3!=0)||(n1==0&&n3==0&&n2!=0)||(n2==0&&n3==0&&n1!=0)){
		if((n1+n2+n3)%2==0){
			return turn*(-1);
		}
		else return turn;
	}
	bool flag = false;
	int x,y,z;
	for(int i=1;i<=7;i++){
		x=n1;y=n2;z=n3;
		if(i&1)x--;
		if(i&2)y--;
		if(i&4)z--;
		//cout<<x<<" "<<y<< " "<<z<<endl;
		if(x>=0&&y>=0&&z>=0){
			int temp = recur(x,y,z,turn*(-1));
			if(temp==turn){
				flag = true;
				break;
			}
			//cout<<turn<<endl;
		}
	}
	x=n1;y=n2;z=n3;
	if(flag){
		dp[x][y][z] = 1;
		dp[x][z][y] = dp[x][y][z];
		dp[y][x][z] = dp[x][y][z];
		dp[y][z][x] = dp[x][y][z];
		dp[z][y][x] = dp[x][y][z];
		dp[z][x][y] = dp[x][y][z];
		return turn;
	}
	else{
		dp[x][y][z] = -1;
		dp[x][z][y] = dp[x][y][z];
		dp[y][x][z] = dp[x][y][z];
		dp[y][z][x] = dp[x][y][z];
		dp[z][y][x] = dp[x][y][z];
		dp[z][x][y] = dp[x][y][z];
		return turn*(-1);
	}
}
int main()
{
	pre();
	int test;
	scanf("%d",&test);
	while(test--){
		int n1,n2,n3;
		sc(n1);sc(n2);sc(n3);
		if(recur(n1,n2,n3,1)==1){
			printf("Ashima\n");
		}
		else printf("Aishwarya\n");
	}
	return 0;
}
