#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <list>
#include <stack>
#include <vector>
#include <queue>
#include <cstring>
#include <algorithm>

using namespace std;

int main()
{
	string w;
	cin>>w;
	long long n,a=0,s=0;
	scanf("%lld",&n);
	for(long long i=0;i<n;i++)
	{
		string x;
		cin>>x;
		if(x.length()==w.length())
		{
			if(w[0]==x[0])
				s++;
			else if(x[0]-w[0]<=3)
				a++;
		}
	}
	printf("%lld %lld",s,a);
	return 0;
}
