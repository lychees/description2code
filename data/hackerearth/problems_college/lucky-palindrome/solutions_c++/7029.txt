char read_char()
{

    up=fread(buff,sizeof(char),size,stdin);

    lp=0;
  }

}
int read_int()
{
  char c;
  do{
    c=read_char();
  }while(c<'0' || c>'9');
  
  int val=0;
  for(;c>='0' && c<='9';c=read_char())
    val=(val<<3) +(val<<1) + c-'0';
  
  return val;
}
string new_str(string str,int n)
{
  string temp="^";
  int i;
  for(i=0;i<n;i++)
    temp+= "#"+str.substr(i,1);
  temp+="#$";
//  cout<<temp<<endl;
  return temp;
}

int manacher(string str,int len,int &count,int &max_len)
{
  string s=new_str(str,len);
  int n=s.length(),i,j,r=0,c=0;
 // cout<<s<<endl;
  vector<int>ch(n);
  
  for(i=1;i<n-1;i++)
  {
    int i_mir=2*c-i;
    ch[i]=r>i?min(r-i,ch[i_mir]):0;
    
    while(s[i+1+ch[i]]==s[i-1-ch[i]])
      ch[i]++;
    
    if(i+ch[i]>r)
    {
      c=i;
      r=i+ch[i];
    }
  }
  
  for(i=1;i<n-1;i++)
  {
    if(ch[i]>max_len)
    {
      max_len=ch[i];
      count=1;
    }
    else if(ch[i]==max_len)
      count++;
  }
  
  return max_len;
 // printf("%d %d\n",max_len,count);
}
