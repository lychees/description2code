#include <iostream>
using namespace std;
int remaining(int n, int k) {
    int r = 0;
    for (int i = 2; i <=n; i++)
        r = (r + k) % i;

    return r;
}
int main()
{
    int t;
    cin >>t;
    while(t--)
    {
    	int n;
    	cin >>n;
    	cout << remaining(n,2)+1<<endl;
    }
    return 0;
}
