#include <iostream>
using namespace std;

int main() {
    int n,a[100001],q,l,r;
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>a[i];
    }
    
    cin>>q;
    while(q--){
        int sum=0;
        cin>>l>>r;
        for(int i=l;i<=r;i++){
            sum+=a[i];
            
        }
        cout<<sum<<endl;
    }
	// your code goes here
	return 0;
}
