nooftestcases = int(raw_input())

for _ in range(nooftestcases):
	target = [char for char in raw_input()]
	reserve = [char for char in raw_input()]
	
	#print target
	#print reserve
	
	reserve.sort()
	current = 0
	
	for index in range(len(target)):
		if reserve[current] < target[index]:
			target[index] = reserve[current]
			current += 1
		if current >= len(reserve):
			break
			
	#print target
	#print reserve
	print "".join(target)
