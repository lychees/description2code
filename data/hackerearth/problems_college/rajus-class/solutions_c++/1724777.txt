#include <iostream>
#include<algorithm>
#include<vector>
using namespace std;
inline int Read()
{
	register int c=getchar();
	int x=0;
	for(;(c<48 || c>57);c=getchar());
	for(;c>47 && c<58;c=getchar())
		x=(x<<1)+(x<<3)+c-48;
	return x;
}
int main()
{
    int t=Read();
    while(t--)
    {
    	int n=Read();
    	pair<int,string> a[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i].first>>a[i].second;
    	}
    	sort(a,a+n);
    	int r=Read();
    	cout<<a[r-1].second<<endl;
    }
    return 0;
}
