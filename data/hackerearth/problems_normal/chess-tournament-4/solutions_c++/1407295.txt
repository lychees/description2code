#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int N;
    cin >> N;
    int nRow =  pow(2, N);;
    int nCol =  nRow;
    int A[nRow][nCol];
    for(int i = 0; i < nRow; i++)
    {
    	for(int j = 0; j < i; j++)
    		cin >>  A[i][j];
    }
    int *arr = new int[nRow];
    for(int i = 0; i < nRow; i++)
    	arr[i] = i;
    
    while(nRow != 1)
    {
    	int *temp = new int[nRow / 2];
    	int index = 0;
    	for(int i = 0; i < nRow; i = i + 2)
    	{
    		if(A[arr[i + 1]][arr[i]] == 0)
    			temp[index++] = arr[i];
    		else
    			temp[index++] = arr[i + 1];
    	}
    	arr = temp;
    	nRow = nRow / 2;
    }
    cout << arr[0] + 1 << endl;
    return 0;
}
