#include <iostream>
#include<algorithm>
using namespace std;

int m[1234], w[1234];

int main()
{
    int t, n, ans;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	ans = 0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>m[i];
    	}
    	for(int i=0;i<n;i++)
    	{
    		cin>>w[i];
    	}
    	sort(m,m+n);
    	sort(w,w+n);
    	for(int i=0;i<n;i++)
    	{
    		ans += m[i] * w[i];
    	}
    	cout<<ans<<endl;
    }
}
