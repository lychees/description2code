#include <bits/stdc++.h>
using namespace std;
 
#define MOD 1000000007LL
#define ll long long
#define MAXN 50010
#define sd(t) scanf("%d",&t)
int A[MAXN];
 
int main()
{
	cin.tie(0);
	ios::sync_with_stdio(false);
	int t,c,n;
	sd(t);
	while(t--)
	{
		sd(A[0]);
		sd(A[1]);
		sd(c);
		sd(n);
		int res = 0;
		if(n==1)
			res = A[0]%41;
		else
			res = (A[0]*10+A[1])%41;
		for(int i=2;i<n;++i)
		{
			res *= 10;
			A[i] = (A[i-1]*c+A[i-2])%10;
			res += A[i];
			res %= 41;
		}
		if(res)
			printf("NO\n");
		else
			printf("YES\n");
	}
	return 0;
}
