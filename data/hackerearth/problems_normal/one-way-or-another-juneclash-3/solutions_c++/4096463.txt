#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
	ll t,n;
	ll aa,bb,xx,a,b,x;
	bool flag;
	cin>>t;
	while(t--)
	{
		flag=true;
		cin>>n;
		cin>>aa>>bb>>xx;
		for(int i=1;i<n;i++)
		{
			cin>>a>>b>>x;
			if((abs(a-aa)+abs(b-bb))%2==0)
			{
				if(xx%2!=x%2)
				{
					flag=false;
				}
			}
			else
			{
				if(xx%2==x%2)
				{
					flag=false;
				}
			}
		}
		if(flag==false)
		{
			cout<<"No"<<endl;
		}
		else
		{
			cout<<"Yes"<<endl;
		}
	}
	return 0;
}
