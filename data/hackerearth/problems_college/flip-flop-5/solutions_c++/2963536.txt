#include <bits/stdc++.h>

using namespace std;

string s;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        cin >> s;
        int i;
        int ans=20;
        for(i=1;i<s.size();i++)
        {
            if(s[i]==s[i-1]) ans++;
        }
        cout << ans-20 << endl;
    }
}
