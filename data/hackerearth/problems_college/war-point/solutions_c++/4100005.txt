#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n;
		cin>>n;
		long long int a[n],s1=0;
		for(int i=0;i<n;i++)
		{
			cin>>a[i];	
			s1 = s1+a[i];
		}
		long long int s2 = 0,min = 99999999;
		for(int i=0;i<n;i++)
		{
			s1 = s1 - a[i];
			s2 = s2 + a[i];
			
			if(abs(s2-s1)<min)
			{
				min = abs(s2-s1);
			}
		}
		cout<<min<<"\n";
	
	}
}
