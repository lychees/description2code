for _ in range(int(raw_input())):
    x = int(raw_input())
    if x == 1 or x == 2:
        print '-1'
        continue

    n1 = x/3
    r = x % 3
    while n1:
        if not r % 5:
            break
        else:
            n1 -= 1
            r += 3

    n2 = r/5
    if n1 == n2 == 0:
        print '-1'
    s = ""
    for _ in range(n1*3):
        s += '5'
    for _ in range(n2*5):
        s += '3'
    print s
