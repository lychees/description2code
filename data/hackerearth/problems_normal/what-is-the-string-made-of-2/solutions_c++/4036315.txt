#include <iostream>
using namespace std;

int main() {
	
	long int l,sm=0;
	string s;
	
	cin>>s;
	
	l=s.size();
	
	for(long int x=0;x<l;x++)
	{
		if(s[x] == '1')
		sm+=2;
		
		else if(s[x] == '0' || s[x] == '6' || s[x] == '9')
		sm+=6;
		
		else if(s[x] == '2' || s[x] == '3' || s[x] == '5')
		sm+=5;
		
		else if(s[x] == '8')
		sm+=7;
		
		else if(s[x] == '7')
		sm+=3;
		
		else
		sm+=4;
		
	}
	
	cout<<sm;
	
	return 0;
	
}
