#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,n,i,j,k,c;
	cin>>t;
	for(i=0;i<t;i++)
	{
		cin>>n;
		long long int a[n];
		for(j=0;j<n;j++)
		cin>>a[j];
		c=0;
		for(j=0;j<n;j++)
		{
			for(k=j+1;k<n;k++)
			{
				if((a[j]+a[k])%3==0)
				c+=1;
			}
		}
		cout<<c<<"\n";
	}
	return 0;
}
