int getFib(int L)
{
    std::vector<int> v(L+1,0);
    v[0] = 1;
    v[1] = 1;
    
    if (L == 1)
    {
        return 1;
    }
    else
    {
        for (int i=2; i<=L; ++i)
        {
            v[i] = v[i-1] + v[i-2];
        }
    }
    
    return v[L];
}

int main(void)
{
    short T;
    short L;
    
    std::cin>>T;
    while (T--)
    {
        std::cin>>L;
        std::cout<<getFib(L)<<std::endl;
        
    }
    return 0;
}
