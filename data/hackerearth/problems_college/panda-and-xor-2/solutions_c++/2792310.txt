#include<bits/stdc++.h>
using namespace std ;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m,c=0;
		cin>>n>>m;
		int p = n^m;
		while(p!=0)
		{
			if(p%2==1)
			{
				c++;
			}
			p = p/2;
		}
		if(c%2 == 0)
		{
			cout<<"YES"<<"\n";
		}
		else
		{
			cout<<"NO"<<"\n";
		}
	}
}
