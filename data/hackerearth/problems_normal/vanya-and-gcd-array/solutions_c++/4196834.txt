#define MOD 1000000007
using namespace std;

long long gcd(long long x, long long y) {
	return y == 0 ? x : gcd(y, x % y);
}

int main() {
	long long n;
	cin>>n;
	long long arr[n];
	for(int i=0;i<n;i++)
		cin>>arr[i];
	long long dp[n][101];
	for(int i=0;i<n;i++)
		for(int j=0;j<101;j++)
			dp[i][j] = 0;
	
	dp[0][arr[0]] = 1;
	long long ans = dp[0][1];
	for(int i=1; i<n; i++) {
		for(int j=i-1;j>=0;j--) {
			if(arr[i]>arr[j]) {
				for(int k=1;k<=100;k++) {
					dp[i][gcd(arr[i],k)] += dp[j][k];
					dp[i][gcd(arr[i],k)] %= MOD;
				}	
			}
		}
		dp[i][arr[i]]++;
		ans += dp[i][1];
		ans %= MOD;
	}
	
	cout<<ans%MOD<<endl;
}
