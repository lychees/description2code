import sys

t = int(sys.stdin.readline())
while t > 0:
    n = int(sys.stdin.readline())
    r, c, x = map(int, sys.stdin.readline().split())
    n -= 1
    while n > 0:
        n -= 1
        r1, c1, x1 = map(int, sys.stdin.readline().split())
        tmp = r1-r+c1-c
        flag = 0
        if tmp % 2 == 0:
            if (x % 2 == 0 and x1 % 2 != 0) or (x % 2 != 0 and x1 % 2 == 0):
                flag = 1
                break
        else:
            if (x % 2 == 0 and x1 % 2 == 0) or (x % 2 != 0 and x1 % 2 != 0):
                flag = 1
                break
    if flag == 1:
        print "No"
        while n > 0:
            n -= 1
            r, c, x = map(int, sys.stdin.readline().split())
    elif n == 0:
        print "Yes"

    t -= 1
