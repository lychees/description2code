#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    map<string,int> hash;
    long long int count=0;
    while(n--){
    	string temp;
    	cin>>temp;
    	count=count+hash[temp];
    	reverse(temp.begin(),temp.end());
    	hash[temp]++;
    }
    cout<<count;
    return 0;
}
