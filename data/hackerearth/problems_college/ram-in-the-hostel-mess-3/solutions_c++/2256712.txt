#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
   long long int n,a[10000]={0};
    while(t--)
    {
        cin>>n;
        if(n==1)
        {
            cout<<"4\n";
            continue;
        }
       a[0]=0;
		a[1]=2;
		a[2]=3;
		for(int i=3;i<10000;i++)
		a[i]=a[i-1]+a[i-2];
		cout<<a[n]*a[n]<<"\n";
    }
    return 0;
}
