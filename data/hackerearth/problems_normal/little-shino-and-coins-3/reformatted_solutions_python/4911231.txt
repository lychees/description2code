num = int(raw_input())
arr = list(raw_input())
n = len(arr)
count = 0

for i in xrange(n):
    tmp = []
    tmp = tmp + [0]*26
    diff = 0
    for j in xrange(i, n):
        ind = ord(arr[j]) - ord('a')
        # ch = tmp[ind]
        tmp[ind] = tmp[ind] + 1
        if tmp[ind] == 1:
            diff += 1
        if diff == num:
            count += 1
# print n
print count
