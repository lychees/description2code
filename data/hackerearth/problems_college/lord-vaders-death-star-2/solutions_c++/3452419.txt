#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,flag=0;
	long long a;
	map<long long,int> m;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a;
		m[a]++;
	}
	map<long long,int>::reverse_iterator it;
	for(it=m.rbegin();it!=m.rend();it++)
	{
		//cout<<it->first<<" "<<it->second<<"\n";
		if(it->second==1)
		{
			flag=1;
			cout<<it->first<<"\n";
			break;
		}
	}
	if(!flag)
	cout<<"-1\n";
}
