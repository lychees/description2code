#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll mybin[100][100];
void cal()
{
	ll i,j;
	mybin[0][0]=0;
	for(i=1; i<=60; i++)
	{
		for(j=0; j<=i; j++)
		{
			if(j==0 || i==j)
			{
				mybin[i][j]=1;
				continue;
			}
			mybin[i][j]=(mybin[i-1][j-1]+mybin[i-1][j]);
		}
	}
}
int main()
{
	cal();
	ll t;
	scanf("%lld",&t);
	while(t--)
	{
		ll n;
		scanf("%lld",&n);
		printf("%lld\n",mybin[2*n][n]/(n+1));
	}
	return 0;
}
