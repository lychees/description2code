#include<map>
#include<string>
#include<iostream>
using namespace std;

int main()
{
    int t;
    string str;
    map<string,int> mp;
    cin>>t;
    while(t--)
    {
       cin>>str;
       if(mp.find(str)==mp.end())
          mp.insert(pair<string,int>(str,1));
       else
          mp[str]+=1;
     }
     map<string,int>::iterator it;
     for(std::map<string,int>::iterator it=mp.begin(); it!=mp.end(); ++it)
     cout<<it->first<<" "<<it->second <<'\n';
     return 0;
 }
