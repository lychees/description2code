#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	char a[100001];
	while(t--) {
	    scanf("%s",a);
	    int k=0,x=strlen(a);
	    if(x==1) cout<<"Funny\n";
	    else {
	        for(int i=1;i<=x/2;i++) {
	            if(abs(a[i]-a[i-1])!=abs(a[x-i-1]-a[x-i])) {
	                k=1;break;
	            } else k=0;
	        }
	        if(k==1) cout<<"Not Funny\n";
	        else cout<<"Funny\n";
	    }
	}
	return 0;
}
