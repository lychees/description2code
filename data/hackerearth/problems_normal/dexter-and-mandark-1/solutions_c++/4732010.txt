#include <bits/stdc++.h>
 
using namespace std;

 
int main(){
	ios_base::sync_with_stdio(false);
	int t;	
	cin>>t;
 
	while(t--){
		int n, m;
		cin>>n>>m;
        int dp[n+1][n+1];
		memset(dp, 0, sizeof(dp));
	
		dp[1][1] = 1;
		
		for(int i=2 ; i<=n ; i++){
			for(int j=1 ; j<=i ; j++){
				if(j < m * (i-j))
					continue;
				dp[i][j] = (dp[i-1][j-1] + dp[i-1][j]) %1000000007;		
			}
 
		}
        
		int ans = 0;
 
		for(int i=1 ; i<=n ; i++){
			if(i==n-i)
				continue;
			ans = (ans + dp[n][i]) %1000000007;
		}
 
		cout<<ans<<"\n";
	}
 
	return 0;
}
