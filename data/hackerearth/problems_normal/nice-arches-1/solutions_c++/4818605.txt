#include <iostream>
#include <stack>
#include <string>

using namespace std;

int main()
{
	
    int TC=0;
    cin>>TC;
    int count=0;
    while(TC--)
    {
    	string s;
    	cin>>s;
    	stack<char> stk;
    	auto it = s.begin();
    	while(it != s.end())
    	{
    		if(stk.empty() || stk.top()!= *it)
    			stk.push(*it);
    		else
    			if(stk.top() == *it)
    				stk.pop();
    		it++;
    	}
    	if(stk.empty())
    		count++;
    }
    cout<<count;
    
    return 0;
}
