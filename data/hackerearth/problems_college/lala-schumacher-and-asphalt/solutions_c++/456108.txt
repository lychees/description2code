#include <bits/stdc++.h>
using namespace std;
long long abso(long long a)
{
	if(a<0)
		return a*(-1);
	return a;
}
int main()
{
	int t,i,j,m;
	long long k,l,des1,des2;
	long long ans,mini;
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
		scanf("%d",&j);
		long long A[j],B[j];
		mini=0;
		for(k=0;k<j;k++)
			scanf("%lld",&A[k]);
		for(k=0;k<j;k++)
			scanf("%lld",&B[k]);
		scanf("%lld%lld",&des1,&des2);
		scanf("%lld%lld",&k,&l);
		if(k<=l)
		{
			mini=k*(abso(des2)+abso(des1));
		}
		else
		{
			mini=k*(abso(des2)+abso(des1));
			for(m=0;m<j;m++)
			{
				ans=k*(abso(A[m])+abso(B[m]));
				ans=ans+l*(abso(A[m]-des1)+abso(B[m]-des2));
				mini=min(mini,ans);
			}
		}
		printf("%lld\n",mini);
	}
	return 0;
}
