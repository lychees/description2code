#include <iostream>
using namespace std;


int freq[1000001];
int main()
{
    int n;
    cin>>n;
    int x;
    long long int t =0;
    for(int i =1; i<=n; i++){
    	cin>>x;
    	freq[x]++;
    	if(freq[x] == 1)
    		t++;
    }
    long long int ans = t*(t-1);
    ans = ans/2;
    cout<<ans;
    
    return 0;
}
