ll power(ll a,ll b)
{
	if(!b)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2);
	temp=(temp*temp);
	if(b&1)
		temp=(temp*a);
	return temp;
}
int lcm(int a,int b)
{
	int val=__gcd(a,b);
	return((a*b)/val);
}
int main()
    {
		//ios_base::sync_with_stdio(false);
		ll ans;
		int i, t, n, sset, c, sign, j, val;
		ll p, q, cal;
		//cin >> n;
		sc("%d",&n);
		vi v(n);
		for( i=0;i<n;i++ )
			cin >> v[i];
		sset = 1<<n;
		//cin >> t;
		sc("%d",&t);
		while( t-- )
		{
			//cin >> p >> q;
			sc("%lld %lld",&p,&q);
			ans = 0;
			for( i=1;i<sset;i++ )
			{
				c=0;
				val = 1;
				//vi take(n);
				for( j=0;j<n;j++ )
				{
					if( (i)&(1<<j) )
					{
						c++;
						val = lcm(val,v[j]);
					}
				}
				if(c&1)
					sign=1;
				else
					sign=-1;
				cal = (q/val) - (p/val) ;
				if((p%val)==0)
					cal++;
				ans += sign*cal;
			}
			pf("%lld\n",ans);
		}
    return 0;
}
