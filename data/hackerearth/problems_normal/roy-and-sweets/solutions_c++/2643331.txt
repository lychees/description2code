#include <bits/stdc++.h>
using namespace std;


int main()
{
    int n,f,m;
    long long sum=0;
    int MAX=1e6+5;
    cin>>n;
    vector<int> sweet(MAX,0);
    int s,q;
    for(int i=0;i<n;i++)
    	{
    		cin>>s>>q;
    		sweet[s]=q;
    		
    	}
    
    for(int i=MAX-1;i>=0;i--)
    {
    	sweet[i]=max(sweet[i],sweet[i+1]);
    }
    cin>>f;
    while(f--)
    {
    	cin>>m;

		sum+=sweet[m]/m;
    	
    }
    cout<<sum*100;
    return 0;
}
