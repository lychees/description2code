#include<bits/stdc++.h>
using namespace std;
int hcf(int , int);
int main(){
  int A,N,arr[1001];
  cin>>A>>N;
  int count=0,num;
  for(int i=0; i<N; i++){
        cin>>num;
        if(hcf(max(A,num),min(A,num))>1){
                count+=1;
        }
  }
  cout<<count<<endl;


return 0;
}
int hcf(int a, int b){
  if(a%b==0){
        return b;
  }
  int temp = a%b;
  a = b;
  b = temp;
  hcf(a,b);
}
