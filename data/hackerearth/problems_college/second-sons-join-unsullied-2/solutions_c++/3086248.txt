#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	char str[50000];
    	scanf("%s",str);
    	int n=strlen(str);
    	int cu=0,cs=0,i;
    	for(i=0;i<n;i++)
    	{
    		if(str[i]=='U')
    		cu++;
    		else
    		cs++;
    	}
    	if(abs(cu-cs)<=1)
    	{
    		int c1=0,c2=0;
    		for(i=0;i<n;i++)
			{
				if(i%2==0)
				{
					if(str[i]=='U')
					c1++;
				}
				else
				{
					if(str[i]=='S')
					c1++;
				}
			}
			for(i=0;i<n;i++)
			{
				if(i%2==0)
				{
					if(str[i]=='S')
					c2++;
				}
				else
				{
					if(str[i]=='U')
					c2++;
				}
			}
			if(c1%2==0&&c2%2==0)
			c1=min(c1/2,c2/2);
			else if(c1%2==0)
			c1=c1/2;
			else
			c1=c2/2;
			
			printf("%d\n",c1);
    	}
    	else
    	printf("-1\n");
    }
    return 0;
}
