#include<bits/stdc++.h>
using namespace std;
int arr[2000];
int fateh(int sum,int i,int j)
{
    if(i==0)
    {
        return 0;
    }
    else if(j<0)
    {
        return 0;
    }
    if(sum+arr[j]>i)
    {
        return sum;
    }
    else
    {
        return max(fateh(sum+arr[j],i,j),fateh(sum,i,j-1));
    }
}
int main()
{
    int a,b,c,d,e,f,g,h,i,j,k,l;
    cin>>a;
    while(a--)
    {
        cin>>b>>c;
        for(i=0;i<b;i++)
            cin>>arr[i];
        sort(arr,arr+b);
        cout<<fateh(0,c,b-1)<<"\n";
    }
    return 0;
}
