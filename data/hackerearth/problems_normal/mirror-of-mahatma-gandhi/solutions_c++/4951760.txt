#include <iostream>
#include <string>
using namespace std;

int main()
{
	int t;
	string str;
	string nonMirrorSet("2345679");
	bool found=false;
	
	cin>>t;
	for (int i=1; i<=t; i++)
	{
		cin>>str;
		if (str.find_first_of(nonMirrorSet) == string::npos)
		{
			found = true;
		} else {
			found = false;
		}
		if (found && (str == string(str.rbegin(), str.rend())))
		{
			cout<<"YES" << endl;
		} else {
			cout<<"NO"<<endl;
		}
	}
    return 0;
}
