#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
    int t;
    int a[500]; //array will have the capacity to store 200 digits.
   long int n,i,j,temp,m,x;

    scanf("%d",&t);
    while(t--)
    {
       scanf("%d",&n);
       a[0]=1;  
       m=1;    
      long int sum=0;  
       temp = 0; 
       for(i=1;i<=n;i++)
       {
            for(j=0;j<m;j++)
            {
               x = a[j]*i+temp; 
               a[j]=x%10; 
              
               temp = x/10; 
            }
             while(temp>0) 
             { 
               a[m]=temp%10;
              
               temp = temp/10;
               m++; 
             }
      }
     for(i=m-1;i>=0;i--)
       {
            if(a[i]==4 || a[i]==7)
             sum+=1;
        }
               
      cout<<sum<<"\n";        
    }
    return 0;
}
