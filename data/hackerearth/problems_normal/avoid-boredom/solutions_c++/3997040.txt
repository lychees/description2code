#include <iostream>
#include <vector>
using namespace std;
int a[101];
long long ans[102][102];
int main()
{
    int t;
    cin >> t;
    while(t--){
    	int n, m;
    	cin >> n >> m;
    	for(int i = 0;i < n;++i){
    		cin >> a[i];
    	}
    	for(int i = 0;i <= n;++i){
    		for(int j = 0;j <= n;++j){
    			ans[i][j] = 0;
    		}
    	}
    	for(int i = 0;i < n;++i){
    		for(int j = 0;j < i;++j){
    			for(int k = 0;k <= j;++k){
    				long long temp = (k > 0?ans[j][k]:0) + abs(a[j] - a[i]);
    				if(temp > ans[i][k + 1])ans[i][k + 1] = temp;
    			//	cout << ans[i][k + 1] << " ";
    			}
    		}
    	}
    	long long ret = 0;
    	for(int i = m;i <= n;++i){
    		if(ret < ans[i - 1][m - 1])ret = ans[i - 1][m - 1];
    	}
    	cout << ret << "\n";
    }
    return 0;
}
