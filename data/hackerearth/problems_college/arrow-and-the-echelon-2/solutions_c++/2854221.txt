#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		int num;
		cin>>num;
		int x = num/15;
		int y = num%15;
		int flag = 0;
		int val1 =0;
		int val2 =0;
		int val = 0;
		for(int i=0;val<=y;i++){
			val = i*3;
			if((y-val)%5 == 0){
				val2 = i;
				val1 = (y-val)/5;
				flag = 1;
			}
		}
		if(flag != 0)
			for(int i=0;i<x;i++) cout<<"33333";
		//cout<<num<<":";
		switch(flag){
			case 1: for(int i=0;i<val2;i++) cout<<"555";
					for(int i=0;i<val1;i++) cout<<"33333";
					break;
			case 0: cout<<"-1";
					break;
		}
		cout<<endl;
	}
    return 0;
}
