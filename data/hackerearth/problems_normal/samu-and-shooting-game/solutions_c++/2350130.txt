int X;

int W;
double P1;
double P2;

bool vis[1001][1001];
double dp[1001][1001];

double rec(int pos, int w)
{
	if(w >= W)
		return 1.00;
	if(pos == N)
		return 0.00;
	
	if(vis[pos][w])
		return dp[pos][w];
	vis[pos][w] = true;
	double ans = 0.00;
	ans =  max(ans, (rec(pos + 1, w + X) * P1) + (rec(pos + 1, w) *(1.00 - P1)));
	ans =  max(ans,  (rec(pos + 1, w + Y) * P2) + rec(pos + 1, w) * (1.00 - P2));
	return dp[pos][w] = ans;
}

}
