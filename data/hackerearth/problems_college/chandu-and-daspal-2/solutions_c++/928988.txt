#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
using namespace std;
int gcd(int a,int b) {
    return b==0?a:gcd(b,a%b);
}
int main() {
    string str[2];
    str[0]="Chandu";
    str[1]="Daspal";
    int t;
    scanf("%d",&t);
    while(t--) {
        string st;
        int a,b;
        scanf("%d%d",&a,&b);
        cin>>st;
        bool flag;
        if(st.compare(str[0])==0) flag=0;
        else flag=1;
        while(true) {
            int temp=gcd(a,b);
            a-=temp;
            b-=temp;
            if(a==0||b==0) break;
            flag=!flag;
        }
        cout<<str[flag]<<endl;
    }
    return 0;
}
