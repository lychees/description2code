#include <bits/stdc++.h>
using namespace std;

int main()
{long long int a[100000],product,product1;
 long int n,i;
    cin>>n;
    for(i=0;i<n;i++)
    	cin>>a[i];
    sort(a,a+n);
    product1=(a[0]*a[1]);
    product=(a[n-1]*a[n-2]);
    if(product > product1)
    cout<<product;
    else
    cout<<product1;
    return 0;
}
