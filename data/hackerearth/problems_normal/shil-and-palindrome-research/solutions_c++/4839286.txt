#include<bits/stdc++.h>
using namespace std;
int bit[100100][26];
char a[100100];
int a1[26],b[26],n;
void update(int x,char c)
{
    int p=(int)c-97;
    for(;x<=n;x+=x&-x)
    {
        bit[x][p]++;
    }
}
void update1(int x,char c)
{
    int p=(int)c-97;
    int h=(int)a[x]-97;
    a[x]=c;
    for(;x<=n;x+=x&-x)
    {
        bit[x][h]--;
        bit[x][p]++;
    }
}
void query(int x)
{
    int i;
    for(;x>0;x-=x&-x)
    {
        for(i=0;i<26;i++)
        {
            b[i]+=bit[x][i];
        }
    }
}
void query1(int x)
{
    int i;
    int j=x;
    for(;x>0;x-=x&-x)
    {
        for(i=0;i<26;i++)
        {
            a1[i]+=bit[x][i];
        }
    }
}
int main()
{
    int t,q,i,k,l,r,ct,p;
    char c;
    ios::sync_with_stdio(0);
    cin.tie(0);
    cin>>n>>q;
    memset(bit,0,sizeof(bit));
    for(i=1;i<=n;i++)
    {
        cin>>a[i];
        update(i,a[i]);
    }
    while(q--)
    {
        cin>>k;
        if(k==1)
        {
            cin>>l>>c;
            update1(l,c);
        }
        else
        {
            cin>>l>>r;
            memset(a1,0,sizeof(a1));
            memset(b,0,sizeof(b));
            query(l-1);
            query1(r);
            ct=0;
            for(i=0;i<26;i++)
            {
                p=abs(a1[i]-b[i]);
                if(p%2==1)
                ct++;
            }
            if(ct>1)
            cout<<"no\n";
            else
            cout<<"yes\n";
        }
    }
    return 0;
}
