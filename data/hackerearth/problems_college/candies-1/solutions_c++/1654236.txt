#include <iostream>
using namespace std;
#define LL long long int 
int main()
{
    int t , n ;
    cin >> t ;
    while(t--){
    	cin >> n ;
    	LL sum =0 ; 
    	for(int i=0;i<n;i++){
    		LL v ;
    		cin >> v ;
    		sum += v ;
    	}
    	puts(sum%n ? "NO" : "YES") ;
    	cout << sum/n << " " << sum%n << endl ;
    }
    return 0;
}
