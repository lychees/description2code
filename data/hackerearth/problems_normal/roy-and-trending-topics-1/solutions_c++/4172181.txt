#include<iostream>
#include<stdio.h>
#include<queue>
#include<map>
using namespace std;
int main()
{ priority_queue<pair<long long int,long long int> >prior;
  map<long long int,long long int> mymap;
  int n;
  scanf("%d",&n);
  while(n--)
  { long long int oldz,p,l,c,s,id;
    cin>>id>>oldz>>p>>l>>c>>s;
    long long int newz=(p*50)+(l*5)+(c*10)+(s*20);
    mymap.insert ( std::pair<long long int,long long int>(id,newz) );
    newz=newz-oldz;
    prior.push(make_pair(newz,id));
  }
  int a=5;
  std::map<long long int,long long int>::iterator it;
  while(a--)
  { pair<long long int,long long int> pi=prior.top();
    prior.pop();
    long long int id=pi.second;
    it = mymap.find(id);
    cout<<it->first<<" "<<it->second<<"\n";
  }
  return 0;
}
