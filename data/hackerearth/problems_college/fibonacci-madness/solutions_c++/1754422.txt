#include <iostream>

using namespace std;

int main()
{
    int n=0;
    int a[1000],i,j;
    unsigned long int D[1000][1000];
    cin>>n;
    
	for(i=0;i<n;i++)
        cin>>a[i];
	
    for(i=0;i<n;i++)
    {
    	D[i][0]=0;
        D[i][1]=1;
        for(j=2;j<a[i];j++)
        	D[i][j]=D[i][j-1]+D[i][j-2];        
    }
        for(i=0;i<n;i++)
        {
        	cout<<"\n";
        	for(j=a[i]-1;j>=0;j--)
            cout<<D[i][j]<<" ";
        }
        
        
    return 0;
}
