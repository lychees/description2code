# include <bits/stdc++.h>
using namespace std;
void swap (char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}
void permute(char *a, int i, int n) 
{
   int j; 
   if (i == n)
   {
     printf("%s\n", a);
   }
   else
   {
        for (j = i; j <= n; j++)
       {
          swap((a+i), (a+j));
          permute(a, i+1, n);
          swap((a+i), (a+j));
       }
   }
} 
 
int main()
{
  	char a[100000];
  	cin>>a;
  	int n;
  	cin>>n;
  	while(n--)
  	{
  		int l,r;
  		cin>>l>>r;
  		permute(a,l,r);
  		cout<<"\n";
  	}
}
