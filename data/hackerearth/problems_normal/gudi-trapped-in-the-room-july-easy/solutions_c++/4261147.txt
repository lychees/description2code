#include <bits/stdc++.h>
using namespace std;

#define MAX 1000002

char s[10];
int dp[MAX];
int a, h, n, mini, dig, mul;

int countdig(int x) {
    if (x == 0) return 1;
    int ans = 0;
    while (x > 0) {
        x/=10;
        ans++;
    }
    return ans;
}

int rotates(int x) {
    int i, temp;
    for (i=1; i<=h; i++) {
        temp = x%10;
        x/=10;
        x+=(temp*mul);
    }
    return x;
}

int rotate(int xx[], int val) {
    int i, ans = rotates(val);
    for (i=dig-1; i>=0; i--) {
        xx[i] = ans%10;
        ans/=10;
    }
}

void incr(int xx[]) {
    for (int i=1; i<dig; i+=2) {
        xx[i] += a;
        if (xx[i] >= 10) xx[i] -= 10;
    }
}

void decr(int xx[]) {
    for (int i=1; i<dig; i+=2) {
        xx[i] -= a;
        if (xx[i] < 0) xx[i] += 10;
    }
}

int getVal(int xx[]) {
    int i, x = 0;
    for (i=0; i<dig; i++) x = (10*x) + xx[i];
    return x;
}

void dfs(int xx[]) {
    int i, x = getVal(xx);
    if (dp[x] == 1) return;
    // cout<<x<<endl;
    if (x < mini) mini = x;
    dp[x] = 1;
    int temp[10];
    for (i=0; i<dig; i++) temp[i] = xx[i];
    incr(temp);
    dfs(temp);
    decr(temp);
    rotate(temp, x);
    dfs(temp);
}

int main() {
	int t, i, j;
	scanf("%d", &t);
	while (t--) {
	    int x[10];
	    for (i=0; i<MAX; i++) dp[i] = 0;
	    scanf("%s", s);
	    scanf("%d %d", &a, &h);
	    dig = strlen(s), n = 0;
	    for (i=0; i<dig; i++) {
	        x[i] = s[i]-'0';
	        n = (n*10) + x[i];
	    }
	    mini = n, h = h%dig, mul = pow(10, dig-1);
	    dfs(x);
	    int ansdig = countdig(mini);
	    for (i=1; i<=(dig-ansdig); i++) cout<<"0";
	    cout<<mini<<endl;
	}
	return 0;
}
