int main()
{
    int n,m,i,j,k,summ=0,ind,tmp,maxx=0;
    cin>>n;
    vector<int> vec(n);
    for(i=0;i<n;i++)
    {
        cin>>vec[i];
    }
     vector<int> vec1(vec);

    while(vec1.size()!=2){
     //   cout<<i;
     maxx=0;
        for(j=0;j<vec1.size()-2;j++)
        {
         //   cout<<vec1[j];
            tmp=vec1[j]*vec1[j+2];
            if(tmp>maxx)
            {
                ind=j+1;
                maxx=tmp;
            }
        }
        summ+=maxx;

        vec1.erase(vec1.begin()+ind);
    }

    cout<<summ<<endl;
    return 0;
}
