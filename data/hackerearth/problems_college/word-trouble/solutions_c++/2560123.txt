#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    string s1,s2;
    int len1,len2,diff,count,i;
    while(n--)
    {
      	cin>>s1>>s2;
    	len1 = s1.size();
    	len2 = s2.size();
    	count = 0;
    	if(len1 > len2)
    	{
    		diff = len1 - len2;
    		for(i=0;i<len2;i++)
    		{
    			if(s1[i] != s2[i])
    				count++;
    		}
    		count += diff;
    		cout<<"Non Exact "<<count<<endl;
    	}
    	else if(len2 > len1)
    	{
    		diff = len2 - len1;
    		for(i=0;i<len1;i++)
    		{
    			if(s1[i] != s2[i])
    				count++;
    		}
    		count += diff;
    		cout<<"Non Exact "<<count<<endl;
    	}
    	else
    	{
    		for(i=0;i<len1;i++)
    		{
    			if(s1[i] != s2[i])
    				count++;
    		}
    		if(count == 0)
    			cout<<"Exact"<<endl;
    		else
    			cout<<"Non Exact "<<count<<endl;
    	}
   }
}
