#include<bits/stdc++.h>
#define ll long long
#define MAX 50000
#define rep(i,a,b) for(ll i=a;i<b;i++)
#define IOS ios::sync_with_stdio(false)
#define vi vector < ll >
#define pb push_back
#define mod 1000000007
using namespace std;

vi a;
ll dp[1005][1005];

ll gcd(ll a,ll b)
{
    while(b!=0)
    {
        ll r=b;
        b=a%b;
        a=r;
    }
    return a;
}

int main()
{
    IOS;
   ll n;
    cin>>n;
    rep(i,0,n)
    {
        ll k;
        cin>>k;
        a.pb(k);
    }
    rep(i,1,n+1)
    {
        dp[a[i-1]][i]=1;
    }

    rep(i,2,n+1)
    {
        rep(j,1,i)
        {
            if(a[j-1]>=a[i-1]) continue;
            rep(k,1,105)
            {
                if(dp[k][j]>=1)
                {
                    ll s=gcd(k,a[i-1]);
                    dp[s][i]=(dp[s][i]+dp[k][j])%mod;
                }
            }
        }
    }
    ll l=0;
    rep(i,1,n+1)
    {
        l=(l+dp[1][i])%mod;
    }
    cout<<l<<endl;
    return 0;
}
