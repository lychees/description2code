#include <iostream>
    using namespace std;
    int main()
    {
    unsigned long long int i,s=0,t,n,j;
    cin>>t;
    while(t--)
    {
    cin>>n;
    s=0;
    for(i=2;i<n;i=i*2)
    {
    for(j=1;j<i;j=j*2)
    {
    if(i+j<=n)
    s=(s%1000000007+i%1000000007+j%1000000007)%1000000007;
    }
    }
    cout<<s<<endl;
    }
    return 0;
    }
