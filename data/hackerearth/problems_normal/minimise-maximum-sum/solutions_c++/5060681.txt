#include "bits/stdc++.h"
using namespace std;
long long findMax(long long *arr,long long n)
{
	long long ans=INT_MIN;
	for(long long i=0;i<n;i++)
	{
		if(arr[i]>ans)
			ans=arr[i];
	}
	return ans;
}
int main()
{
	long int tc;
	cin>>tc;
	while(tc--)
	{
		long long  n;
		cin>>n;
		long long arr[2*n];
		long long brr[n];
		for(long long i=0;i<2*n;i++)
			cin>>arr[i];
		sort(arr,arr+2*n);
		for(long long i=0,j=2*n-1;i<n;i++,j--)
		{
			brr[i]=arr[i]+arr[j];
		}
		cout<<findMax(brr,n)<<endl;
	}
	return 0;
}
