#include <bits/stdc++.h>
using namespace std;
# define mod 1000000007
# define ll long long int
ll n,t,arr[100005],prime[100005],ans;
void seive()
{
    prime[0]=1;prime[1]=1;
    for(int i=2;i*i<100000;i++)
    {
        if(!prime[i])
        {
            for(int j=i*i;j<=100000;j+=i)
            {
                prime[j]=1;
            }
        }
    }
}
ll pwr(ll a,ll b)
{
	if(b==0)
		return 1;
	if(b==1)
		return a%mod;
    ll p=pwr(a,b/2)%mod;
    p=(p*p)%mod;
    if(b%2!=0)
        p=(p*a)%mod;
    return p%mod;
}
int main()
{
    seive();
    scanf("%lld",&t);
    while(t--)
    {
    	ans=1;
        scanf("%lld",&n);
        for(int i=0;i<n;i++)
            scanf("%lld",&arr[i]);
        for(int i=2;i<=10000;i++)
        {
            if(!prime[i])
            {
                ll counti=0;
                for(int j=0;j<n;j++)
                {
                    ll temp=arr[j],cnt=0;
                    while(temp%i==0)
                    {
                        cnt++;
                        temp=temp/i;
                    }
                    counti=max(counti,cnt);
                }
                ans=(ans*pwr(i,counti))%mod;
            }
        }
        printf("%lld\n",ans%mod);
    }
    return 0;
}
