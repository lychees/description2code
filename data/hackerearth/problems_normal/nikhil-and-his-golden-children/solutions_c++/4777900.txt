#include <bits/stdc++.h>

#define m 1000000007
#define ll long long
using namespace std;

ll power(ll a,ll b)
{
	ll ans=1;

	while(b>0)
	{
		if(b%2!=0)
		ans=(ans*a)%m;


		a=(a*a)%m;
		b=b/2;

	}
	return ans;


}
ll inverse(int a)
{
	return power(a,m-2);

}
ll comb(ll n,ll k)
{
	vector<ll>v(n+1,1);

	for(ll i=2;i<=n;i++)
	{
		v[i]=(v[i-1]*i)%m;
	}

	return (v[n]*((inverse(v[n-k])*inverse(v[k]))%m))%m;


}
