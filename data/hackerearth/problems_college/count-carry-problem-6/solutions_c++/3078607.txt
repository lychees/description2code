#include <iostream>
#include <string>
#include <cstring>
#include <math.h>
using namespace std;
int main()
{
    long long int t,i,l,j,k,c=0,carry=0;string s1,s2;int s[1000000];
    for(i=0;i<1000000;i++)
        s[i]=0;
    cin>>t;
    while(t--)
    {
        cin>>s1;cin>>s2;carry=0;
        l=max(s1.length(),s2.length());
        for(i=0,j=s1.length()-1,k=s2.length()-1;i<l;i++,j--,k--)
        {
            s[i]=(s1[j]-'0')+(s2[k]-'0')+c;
            if(s[i]>9)
            {
                c=1;carry++;
                s[i]-=10;
            }
            else
                c=0;
 
        }
        if(carry==0)
            cout<<"No carry operation\n";
        else if(carry==1)
            cout<<"1 carry operation\n";
        else
            cout<<carry<<" carry operations\n";
 
 
    }
return 0;
}
