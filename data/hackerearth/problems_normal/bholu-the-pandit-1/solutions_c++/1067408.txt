#include <iostream>
#include<algorithm>
#define MOD 1000000007
using namespace std;

int main()
{
    long long int a,b,c,d,ans,t;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>c;
    	d=__gcd(a,b);
    	d=__gcd(d,c);
    	a=a/d;
    	b=b/d;
    	c=c/d;
    	ans=(a*b)%MOD;
    	ans=(ans*c)%MOD;
    	cout<<d<<" "<<ans<<endl;
    }
    return 0;
}
