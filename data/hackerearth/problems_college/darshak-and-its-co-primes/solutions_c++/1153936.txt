using namespace std;
#include<stdio.h>
#include<iostream>
int hcf(int a, int h) 
{
    int temp; 
    while(1)
    {
        temp = a%h;
        if(temp==0)
        return h;
        a = h;
        h = temp;
    }
} 

int main() 
{
    int c,d,gcd,n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	scanf("%d%d",&c,&d);
    	gcd=hcf(c,d);
 	  
    	if(gcd==1) 
    	{
       	 printf("Is a Co-Prime\n");
    	}
    	else
   		{
   	     printf("Not a Co-Prime\n");
   	 	}
    }
}
