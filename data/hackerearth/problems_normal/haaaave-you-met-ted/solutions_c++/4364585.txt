#include <bits/stdc++.h>
using namespace std;
int main (void)
{
	int test;
	scanf ("%d", &test);
	
	while (test--) {
		int n;
		scanf ("%d", &n);
		int i, arr[n];
		
		for (i = 0; i < n; i++) {
			scanf ("%d", &arr[i]);
		}
		int min = 32, temp, count;
		for (i = 0; i < n; i++) {
			temp = arr[i];
			count = 0;
			while (temp) {
				if(temp&1){
					count++;
				}
				temp = temp>>1;
			}
			if (count< min) {
				min = count;
			}
		}
		printf ("%d\n",min);
	}
	return 0;
}
