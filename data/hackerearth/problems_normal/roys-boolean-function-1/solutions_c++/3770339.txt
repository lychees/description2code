#include <iostream>
#include <math.h>
using namespace std;

bool isPrime(int n)
{
    if(n <= 1) return false;
    for(int i=2; i<=sqrt(n); i++)
    {
        if(n%i == 0) return false;
    }
    return true;
}

bool isRelativelyPrime(int a, int b)
{
    if(a == 1 || b == 1) return true;
    if(a%b == 0 || b%a == 0) return false;
    return a > b ? isRelativelyPrime(a%b, b) : isRelativelyPrime(a, b%a);
}

int numberRelativelyPrime(int n)
{
    int res=0;
    for(int i=1; i<n; i++) if(isRelativelyPrime(i, n)) res++;
    return res;
}

bool isRoy(int n)
{
    return isPrime(numberRelativelyPrime(n));
}

int main()
{
    int t;
    cin >>t;
    for(int i=0; i<t; i++)
    {
    	int tem;
    	cin >>tem;
    	if(isRoy(tem)) cout << "TRUE" << endl;
    	else cout << "FALSE" << endl;
    }
    return 0;
}
