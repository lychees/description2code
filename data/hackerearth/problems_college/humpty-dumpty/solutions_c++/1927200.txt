#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long i,j,k,n,d,p,flag,sum,T;
    long long A[10001];
    double x,y,r,dis,sumD;
    cin>>T;
    while(T--){
    	cin>>n;
    	cin>>x>>y;
    	sum=0;
    	for(i=0;i<n;i++){
    		cin>>A[i];
    		sum+=A[i];
    	}
    	sort(A,A+n);
    	flag=1;
    	sumD=sum*1.0;
    	r=A[n-1]*1.0;
    	dis=sqrt(pow(x,2)+pow(y,2));
    	if(dis>r){
    		if(dis>sumD) flag=0;
    	}
    	else{
    		sumD=sumD-r+dis;
    		if(r>sumD) flag=0;
    	}
    	if(flag==1) cout<<"Able\n";
    	else cout<<"Not able\n";
    }
    return 0;
}
