#define FORD(i,a,b) for(int i = (a); i >= (b); --i)
#define RI(i,n) FOR(i,1,(n))
#define REP(i,n) FOR(i,0,(n)-1)

const int inf = 1e9 + 5;
const int nax = 1e6 + 5;

const int magic = 200;
int dp[nax];

ll f(ll n) {
	ll mod = 1;
	ll ans = 0;
	int a = 2;
	while(true) {
		ll new_mod = mod * a / __gcd(mod, (ll) a);
		for(int k = 1; k * mod < new_mod; ++k) {
			ll r = k * mod;
			if(r > n) break;
			ans += (dp[a] + 1) * (1 + (n - r) / new_mod);
			//printf("%lld %lld   %lld*%lld\n", r, new_mod, ll(dp[a]+1), (1 + (n - r) / new_mod) );
		}
		mod = new_mod;
		if(mod > n) break;
		++a;
	}
	return ans;
}

int main(int argc, char * argv[]) {
	RI(a, magic) {
		dp[a] = 1;
		RI(b, a) if(a % b) {
			dp[a] = dp[b] + 1;
			break;
		}
	}
	ll low, high;
	scanf("%lld%lld", &low, &high);
	//printf("%lld\n", f(high));
	int hack = 0;
	if(low == 2) hack = 2;
	printf("%lld\n", f(high) - f(low - 1) - hack);
	return 0;
}
