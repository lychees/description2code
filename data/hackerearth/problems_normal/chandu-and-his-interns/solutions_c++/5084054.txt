#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <cmath>

using namespace std;

#define PR_RANGE 10000001

bool isprime[PR_RANGE+1];

void sieve()
{
	long long int i, j;

	for (i = 2; i <= PR_RANGE; i++)
	{
		if (isprime[i])
		{
			for (j = i*i; j <= PR_RANGE; j += i)
			{
				isprime[j] = false;
			}
		}
	}
}

int main()
{
	int i, N, B, sq;

#ifdef _MSC_VER
	freopen("Text.txt", "r", stdin);
#endif
	std::fill_n(isprime, PR_RANGE+1, true);
	sieve();

	cin >> N;

	for (i = 0; i < N; i++)
	{
		cin >> B;
		
		sq = int(sqrt(B));
		
		if (isprime[B])
		{
			cout << "NO" << endl;
		}
		else if (sq*sq == B && isprime[sq])
		{
			cout << "NO" << endl;
		}
		else
		{
			cout << "YES" << endl;
		}
	}

	return 0;
}
