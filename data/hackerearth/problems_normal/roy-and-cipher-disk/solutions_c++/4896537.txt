#include <iostream>
using namespace std;
 
int main()
{
int t;
cin>>t;
while(t--)
{
char s[100];
cin>>s;
int cost,start=97;
for(int i=0;s[i]!=NULL;i++)
{
cost=s[i]-start;
if(cost>-13 && cost<=13)
cout<<cost;
else if (cost>13)
cout<<cost-26;
else if (cost<=-13)
cout<<26+cost;
cout<<" ";
start=s[i];
}
cout<<endl;
}
}
