from collections import defaultdict
N=int(raw_input())
di= defaultdict(list)
li=[]
l2=[]
for i in range(N):
    c,r=map(int,raw_input().split(" "))
    li.append(c)
    l2.append(r)
for j in range(len(li)):
            di[li[j]].append(l2[j])
for c1 in di:
    di[c1].sort(reverse=True)

for i in di:
    for j in di[i] :  
        print "%d %d"%(i,j)
