#include<cstdio>
using namespace std;
char S[52];
int is_v(char c)
{

 if(c=='a' || c== 'e' || c=='i' || c=='o' || c=='u' || c=='y') return 1;
return 0;
}
int main()
{
 int ntc; scanf("%d",&ntc);
 while(ntc--)
 {
    scanf("%s",S);
	
	int v=0,c=0;
	int ans= 0; int cv = 0;
	for(int i=0;S[i]!='\0';i++)
	{
	 if(is_v(S[i]))
	 {
	   v++; cv=0;
	 }
	 else
	 {
	 c++; cv++;
	 }
	 
	 if(cv == 3)
	 {
	  ans = 1;  break;
	 }
	 
	}
	
	if(c>v) ans = 1;
	if(ans)  printf("hard\n");else printf("easy\n");
 }

}
