#include <iostream>
using namespace std;

int main() {
	long long int A[10005],val[10005];
	int i,j,t,n;
	cin>>t;
	for(j=0;j<t;j++)
	{
	cin>>n;
	for(i=1;i<=n;i++)
	{
		cin>>A[i];
	}
	A[0]=0;val[0]=0;
	for(i=1;i<=n;i++)
	{
		if (i==1 || i==2)
		{val[i]=A[i];}
		else if(i==3)
		{val[i]=val[1]+A[3];}
		else
		{
			val[i]=max(val[i-2],val[i-3])+A[i];
		}
	}
	cout<<max(val[n],val[n-1])<<endl;
	}
	return 0;
}
