#include<bits/stdc++.h>
using namespace std;

inline void fast_and_furious() {
	ios_base::sync_with_stdio(0);
	cin.tie(NULL);cout.tie(NULL);
}

typedef long long ll;
const ll N=5000000;
#define M 1000000007
ll t,n,m;
vector<bool> isprime(N,1);
vector<ll> f;

int main() {
	fast_and_furious();
	cin>>t;
    isprime[0]=isprime[1]=0;
    for (ll i=2;i*i<=N;i++){
        if(isprime[i]) {
            for (ll j=i*i;j<=N;j+=i){
                isprime[j]=0;
            }
        }
    }
    int cntr=0;
    for(int i=2;i<N;i++){
    	if(isprime[i]){
    		cntr++;
    		if(isprime[cntr])
	    		f.push_back(i);
    	}
    }
	while(t--){
		cin>>n>>m;
		cout<<(f[n-1]*f[m-1])%M<<"\n";	
	}
	return 0;
}
