#include<bits/stdc++.h>
using namespace std;
int main()
{
   int i,t,*a,n,ctr,max1;
   
   ctr=0;
   max1=0;
    cin>>n;
   	a=new int [n];
   	for(i=0;i<n;i++)
   	  cin>>a[i];
   	  for(i=0;i<n-1;i++)
   	 {
   	  	
     if(abs(a[i]-a[i+1])==1)
               ctr++;
            else 
            ctr=0;
            if(max1< ctr)
               max1=ctr;
   	  }
     cout<<max1+1<<endl; 	
	
	return 0;
}
