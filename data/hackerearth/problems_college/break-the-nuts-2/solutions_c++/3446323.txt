#include<iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;

    bool arr[110];
    for(int i=0; i<110; i++)
        arr[i] = false;

    for(int i=0; i<n; i++)
    {
        int temp;
        cin>>temp;
        if(temp == 1)
            arr[i] = true;
    }

    int first=0, last=0;

    for(int i=0; i<n; i++)
    {
        if(arr[i] == true)
        {
            first = i;
            break;
        }
    }

    for(int i=n-1; i>=0; i--)
        if(arr[i] == true)
    {
        last = i;
        break;
    }

    if(first == last)
    {
    	if(first==0 && arr[0] == false)
    		cout<<"0\n";
    	else
        	cout<<"1\n";
    }
    else
    {
        int ctr = 1;
        long long int ans = 1;
        for(int i=first+1; i<=last; i++)
        {
            if(arr[i] == false)
                ctr++;
            else
            {
                ans *= ctr;
                ctr = 1;
            }
        }
        cout<<ans<<endl;
    }

    return 0;
}
