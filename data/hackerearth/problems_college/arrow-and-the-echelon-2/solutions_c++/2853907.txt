#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
	int t=0,n=0,i=0,j=0,k=0,m=0;;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		if(n%3==0)
		{
			for(i=0;i<n;i++)
				printf("5");
			printf("\n");
		}
		else if(n%5==0)
		{
			for(i=0;i<n;i++)
				printf("3");
			printf("\n");
		}
		else
		{
			m=n;
			i=1,j=0;
			while(m>3)
			{
				m=m-5*i;
				if(m%3==0)
				{
					for(k=0;k<m;k++)
						printf("5");
					for(k=0;k<n-m;k++)
						printf("3");
					printf("\n");
					j=1;
					break;
				}
				else	i++;
			}
			if(j==0)
				printf("-1\n");
		}
	}
	return 0;
}
