#include<bits/stdc++.h>
#define lli long long int
using namespace std;

int main()
{
    int t;
    lli a,b,sqa,sqb;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld%lld",&a,&b);
        sqa=sqrt(a);
        sqb=sqrt(b);
        if(sqa*sqa>a) sqa--;
        if(sqb*sqb>b) sqb--;
        printf("%lld\n",sqb-sqa+((sqa*sqa)==a));
    }
    return 0;
}
