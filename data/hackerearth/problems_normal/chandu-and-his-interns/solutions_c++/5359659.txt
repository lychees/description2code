#include <iostream>
#include <math.h>
using namespace std;


int main()
{
	int n;
	cin>>n;
	
    while(n--){
    	int y=0;
    	int x;
    	cin>>x;
    	for(int i=1;i<=sqrt(x);i++){
    		double m = x/i;
    		if(x%i==0){
    			if(m == i){
    				y++;
    				
    			}else{
    				y=y+2;
    				
    			}
    		}if(y>=4){
    			break;
    		}
    	}
    	
    	if(y>=4){
    		cout<<"YES"<<endl;
    	}else{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
