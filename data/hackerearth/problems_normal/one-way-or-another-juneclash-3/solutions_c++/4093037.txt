#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define pb push_back
#define mp make_pair
int main()
{
  //ios::sync_with_stdio(false);
  ll i,j,k,m,n,t,x,y,z;

   scanf("%lld",&t);
   ll cnt=0;
   while(t--)
   {
     vector<pair<pair<ll,ll>,ll> >v;
     scanf("%lld",&n);
     while(n--)
     {
       scanf("%lld%lld%lld",&x,&y,&z);
       v.pb(mp(mp(x,y),z));
     }
     ll f=0;
     for(i=0;i<v.size();i++)
     {
       for(j=i+1;j*j<=v.size();j++)
       {
          ll dis=abs(v[i].first.first-v[j].first.first)+abs(v[i].first.second-v[j].first.second)-1;
          if(v[i].second%2==0 && v[j].second%2==0 || v[i].second%2==1 && v[j].second%2==1)
          {
            if(dis%2==0)
            {
              f=1;
              break;
            }
          }
          else
          {
            if(dis%2==1)
            {
              f=1;
              break;
            }
          }
       }
       if(f)
        break;
     }
     if(f)
      printf("No\n");
      else
      printf("Yes\n");
   }
 return 0;
}
