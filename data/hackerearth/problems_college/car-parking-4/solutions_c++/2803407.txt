#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;string a;int n;int counter;
    cin>>t;
    for(int i =0;i<t;i++)
    {
    	cin>>a;counter =0;
    	n =a.length();
    	for(int j=0;j<n;j++)
    	{
    		if(a[j]=='-')
    		{
    			if(!((a[j+1]=='B')||(a[j+2]=='B')))
    			{
    				if(!((a[j-1]=='S')||(a[j+1]=='S')))
    				{
    					counter++;
    				}
    			}
    		}
    	}
    	cout<<counter<<endl;
    }
    return 0;
}
