prime = [0]*1000001
for i in range(3 , 1001 , 2):
	if not prime[i]:
		for j in range(i*i , 1000001 , 2*i):
			if not prime[j]:
				prime[j] = i
		

t = int( raw_input() )
for i in range(t):
	n = int( raw_input() )
	if n % 2 == 0:
		print n-2
	elif prime[n] == 0:
		print 0
	else:
		print n - prime[n]
