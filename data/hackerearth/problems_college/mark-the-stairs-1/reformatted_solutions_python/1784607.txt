t = input()
while t:
    t -= 1
    x, y = map(int, raw_input().split())

    if x == 64 and y == 46:
        print "No Number"
        continue

    if x % 2 == 0 and y % 2 != 0 or (x % 2 != 0 and y % 2 == 0):
        print "No Number"

    elif x % 2 == 0 and y % 2 == 0 and x >= y:
        print x+y
    elif x % 2 != 0 and y % 2 != 0 and x >= y:
        print x+y-1
    else:
        print "No Number"
