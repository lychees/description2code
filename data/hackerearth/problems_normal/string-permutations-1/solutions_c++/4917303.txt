#include<iostream>
#include<string>
#include<queue>
#include<unordered_map>
#include<utility>
using namespace std;

int main() {
  int t;
  cin >> t;
  string name;
  unordered_map<string, int> nameList;
  while (t--) {
    name.clear();
    cin >> name;
    nameList[name]++;
  }
  int max = -1;
  int count = 0;
  for (unordered_map<string, int>::iterator it = nameList.begin(); it != nameList.end(); it++) {
    if (it->second > max) {
      max = it->second;
      count = max;
    }
    else if (it->second == max) {
      count += max;
    }
  }
  cout << count << endl;

  return 0;
}
