#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	{
		string S,P;
		cin>>S>>P;
		int len_S = S.size();
		int len_P = P.size();
		if(len_S<len_P)
			cout<<"NO"<<endl;
		else
		{
			int j = 0;
			for(int i=0;i<len_S;i++)
			{
				if(S[i]==P[j])
					j++;
			}
			if(j==len_P)
				cout<<"YES"<<endl;
			else
				cout<<"NO"<<endl;
		}
	}
	return 0;
}
