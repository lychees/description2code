#include <iostream>
using namespace std;

int main()
{
    int t,x;
    int amt[]={1000,500,100,50,20,10,5,2,1};
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&x);
    	int ans=0;
    	for(int i=0;i<9;++i)
    	{
    		ans+=x/amt[i];
    		x%=amt[i];
    		if(x==0) break;
    	}
    	printf("%d\n",ans);
    }
    return 0;
}
