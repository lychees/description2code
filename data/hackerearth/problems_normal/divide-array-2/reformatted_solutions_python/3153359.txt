n = int(raw_input())
a = [int(i) for i in raw_input().split(" ")]
q = int(raw_input())
p = 1
while q > 0:
    q -= 1
    x = int(raw_input())
    p = p*x
for i in range(n):
    a[i] /= p
# print (a)
print " ".join(map(str, a))
