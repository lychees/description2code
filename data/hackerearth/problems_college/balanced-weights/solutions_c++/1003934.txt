#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <string.h>
#include <math.h>
#include <limits.h>
#include <sstream>
#include <vector>
using namespace std;
#define rep(i,a,N) for(int i=a;i<N;++i)

unsigned long long GCD(unsigned long long u, unsigned long long v) {
    while ( v != 0) {
        unsigned long long r = u % v;
        u = v;
        v = r;
    }
    return u;
}

int main(){
	unsigned long long Ans,T,N,gcd;
	scanf("%llu",&T);
	while(T--){
		scanf("%llu",&N);
		int A[N];
		rep(i,0,N)
			scanf("%llu",&A[i]);
		gcd=GCD(A[0],A[1]);
		rep(i,2,N)
			gcd=GCD(gcd,A[i]);
		Ans=gcd*N;
		printf("%llu\n",Ans );
	}
	return 0;
}
