#include <iostream>
#include <functional>  
#include <algorithm> 
using namespace std;



int main()
{
  int t,i,top;long n,sum;
  cin>>t;
  while(t--)
  {	cin>>n;
  	top=i=0;
  	sum=0;
  	long a[n];
  	for(int i=0;i<n;i++)	cin>>a[i];
  	std::sort(a, a+n);
  	top=n-1;
  	i=top-1;
  	while(top>=0){
  		if(a[top]==-1)	top--;
  		else
  		{	i=top-1;
  			while(i>=0){
  				if(a[i]<a[top]&&a[i]!=-1){
	  			a[top]=a[i];
	  			a[i]=-1;
	  			}
	  		i--;
  			}
  		top--;
  		}
  	}
  	for(int i=0;i<n;++i){
  		if(a[i]!=-1){
  			sum++;
  		}
  	}
  	cout<<sum<<endl;
  }
    return 0;
}
