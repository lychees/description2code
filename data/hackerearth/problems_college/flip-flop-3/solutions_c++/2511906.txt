#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    while(n--){
    	string str;
    	cin>>str;
    	int l = str.length();
    	int count=1;
    	int flag=0;
    	if(str[0]=='X')
    		flag=0;
    	else
    		flag=1;
    		
    	for(int i=1;i<l;i++){
    		if(str[i]=='X'){
    			if(flag==1){
    				count++;
    				flag=0;
    			}
    		}
    		else if(str[i]=='Y'){
    			if(flag==0){
    				count++;
    				flag=1;
    			}
    		}
    	}
    	cout<<(l-count)<<endl;
    }
    return 0;
}
