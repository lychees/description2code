#include <iostream>
using namespace std;

int main()
{
	int N;
	cin>>N;
	int E,O;
	for(int i=0;i<N;i++)
	{
		int a,b;
		cin>>a>>b;
		E=0;O=0;
		for(int j=1;j<=a;j++)
		{
			if(j%2==0)
			{
			if(j>b)
			{
				E+=b;
			}
			else
			{
				E+=j;
			}
			
			}
			
			else if(j%2!=0)
			{
				if(j>b)
				{
					O+=b;
				}
				
				else
				{
					O+=j;
				}
			}
			
	
		//	cout<<E<<" "<<O<<endl;
		}
		cout<<E<<" "<<O<<endl;
	
	}
	
	
}
