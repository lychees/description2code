#include <stdio.h>
using namespace std;

  //#include<bits/stdc++.h>
  #include<iostream>
int main()
{
     int n;
      cin>>n;
      long long int ans=0;
       for(int i=1;i<=n;i++)
        {
        	int j=i-1;
        	if(i>=3)
        	ans+=j*(j-1);
        }
         cout<<ans<<endl;
    return 0;
}
