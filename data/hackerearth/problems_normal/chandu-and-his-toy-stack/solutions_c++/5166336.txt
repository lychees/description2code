#include <iostream>
#include <algorithm>
#include <vector>
#define ll long long int
using namespace std;
vector <ll> a,b;
int main()
{
    ll i,t,n,x,y,sum,p,q;
    cin>>t;
    while(t--)
    {
    	sum=0;
    	cin>>n>>x>>y;
    	for(i=0; i<n; i++)
    	{
    		cin>>p>>q;
    		a.push_back(p);
    		b.push_back(q);
    	}
    	sort(a.begin(),a.end());
    	sort(b.begin(),b.end());
    	for(i=0; i<n; i++)
    	{
    		if(a[i] < b[i])
    		{
    			sum = sum + abs(a[i]-b[i])*x;		
    		}
    		else if(a[i] > b[i])
    		{
    			sum = sum + abs(a[i]-b[i])*y;
    		}
    	}
    	cout<<sum<<"\n";
    	a.clear();
    	b.clear();
    }
    return 0;
}
