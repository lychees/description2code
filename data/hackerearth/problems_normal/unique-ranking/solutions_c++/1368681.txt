double EPS = 1e-10;

int main() {
	int n, k; cin >> n >> k;
	vector< vector<int> > a(n, vector<int>(k));
	for (int i = 0; i < n; i++)
		for (int j = 0; j < k; j++)
			scanf("%d", &a[i][j]);
	int q; cin >> q;
	while (q--) {
		vector<int> v(n);
		for (int i = 0; i < n; i++)
			v[i] = i;
		int x; cin >> x;
		while (x--) {
			int j; scanf("%d", &j); j--;
			int mini = 100;
			for (int l = 0; l < v.size(); l++)
				mini = min(mini, a[v[l]][j]);
			vector<int> w;
			for (int l = 0; l < v.size(); l++)
				if (a[v[l]][j] == mini)
					w.push_back(v[l]);
			v = w;
		}
		cout << v[0] + 1 << endl;
	}
}
