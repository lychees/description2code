#include <iostream>
using namespace std;

int main()
{
   int t;
   cin >> t;
   while(t>0)
   {
   	t=t-1;
   	int n,d,dist=0,m;
   	cin >> n >> d;
   	while(d>0 && n>0)
   	{
   		dist = dist+n*2;
   		m=n*2;
   		d=d-m;
   		n=(dist/6)*2;
   		dist = dist%6 ;
   	}
   	if(d<=0)
   	cout << "Yes" << "\n";
   	else
   	cout << "No" << "\n";
   }
    return 0;
}
