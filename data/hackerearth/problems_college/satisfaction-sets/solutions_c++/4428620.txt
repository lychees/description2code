#include <bits/stdc++.h>
using namespace std;

long long A[1000005], B[1000005], C[501];

int main()
{
	long long i,j,k,l,m,n, un = 0, inter = 0;
	cin>>n;
	k = -1;
	for(i = 0; i < n; i++) {
		cin>>m;
		memset(C,0,sizeof(C));
		for(j = 0; j < m; j++) {
			cin>>l;
			if(C[l] == 0) {
				B[l]++;
				C[l] = 1;
			}
		}
	}
	for(i = 0; i < 501; i++) {
		if(B[i] == n) {
			inter += i;
		}
	}
	cout<<-1*inter<<endl;
	
    return 0;
}
