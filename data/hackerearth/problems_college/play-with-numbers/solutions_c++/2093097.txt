        #include<bits/stdc++.h>
        using namespace std;
        int main(){
        	int t,count,temp;
        	long long int z,m,n;
        	cin>>t;
        	while(t--){
        		cin>>z>>n>>m;
        		temp = sqrt(z)+1;
        		count = 0;
        		for(int a=1;a<temp;a++){
        			if(z%a==0){
        				if(a<=n and z/a<=m){
        					count++;
        				}
        				if(a<=m and z/a<=n and a!=z/a){
        					count++;
        				}
        			}
        		}
        		cout<<count<<endl;
        	}
        	return 0;
        }
