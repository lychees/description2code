#include <iostream>
using namespace std;

int main()
{
   int t;
   scanf("%d\n",&t);
   while(t--)
   {
   	 string str1,str2;
   	 cin>>str1>>str2;
   	 int len1,len2,len,i,count=0;
   	 for(len1=0;str1[len1]!='\0';len1++);
   	 for(len2=0;str2[len2]!='\0';len2++);
   	 len=len1;
   	 if(len2<len1)
   	 len=len2;
   	 for(i=0;i<len;i++)
   	 {
   	 	if(str1[i]==str2[i])
   	 	count++;
   	 }
   	 cout<<count;
   	 cout<<"\n";
   }
}
