#include <iostream>
using namespace std;
long  power(long x,long y)
{
    if( y == 0)
        return 1;
    else if (y%2 == 0)
        return power(x, y/2)*power(x, y/2);
    else
        return x*power(x, y/2)*power(x, y/2);

}
int main()
{
    long a,n;
    cin>>a>>n;
    if(a==1)
     cout<<n;
    else 
      cout<<(a*((power(a,n)-1)/(a-1)));
    return 0;
}
