#include <iostream>
#include <stdio.h>
#include <string.h>
#include <math.h>
#include <vector>
#include <string>
#include <map>
#include <set>
#include <queue>
#include <algorithm>
#define lli long long int
#define pb push_back
#define mod 1000000007
#define pii pair<int,int>

using namespace std;
lli arr[500009],col[500009];
lli dp[500009][12] , pow2[500009];
vector<int> vec;
int main()
{
    lli n,m,k,i,j,a,b,c;
    memset(col,0,sizeof(col));
    cin>>n>>k>>m;
    for(i=0;i<n;i++){
        scanf("%lld",&arr[i]);
        col[arr[i]]++;
    }
    for(i=1;i<=k;i++){
        if(col[i]!=0)vec.pb(col[i]);
    }
    n=vec.size();
    memset(dp,0,sizeof(dp));
    lli ans=0;
    pow2[0]=1;
    for(i=1;i<=500000;i++)pow2[i]=(pow2[i-1]*2)%mod;
    for(i=0;i<=n;i++)dp[i][0]=1;
    for(i=1;i<=n;i++){
        for(j=1;j<=m;j++){
            lli num =(dp[i-1][j-1]%mod * (pow2[vec[i-1]]-1)%mod)%mod;
            dp[i][j]=(dp[i-1][j] + num )%mod;
        }
    }
    for(i=0;i<=m;i++)ans=(ans + dp[n][i])%mod;
    cout<<ans;


    return 0;
}
