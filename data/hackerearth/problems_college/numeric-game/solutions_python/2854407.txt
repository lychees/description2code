from math import ceil, floor
r= (1+5**0.5)/2
def game(a1, a2, b1, b2):
    c = 0
    for a in range(a1, a2+1):
        l=ceil(a*r)
        if b2>=l:
            c += b2 - max(l, b1) + 1
    return c

def startgame():
    t = int(input())    
    for i in range(t):
        a1, a2, b1, b2 = map(int, raw_input().split())
        c = 0
        c += game(a1,a2,b1,b2)
        c += game(b1,b2,a1,a2)     
        print('Case #%d: %d'%(i+1, c))    
startgame()
