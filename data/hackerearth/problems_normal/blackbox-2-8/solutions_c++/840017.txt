#include <iostream>
using namespace std;
int a[3000]={0};
long long ans[3000]={0};

void fun(int n)
{

    a[0]=1;
    int c,x,l=1;
    for(int j=1;j<=n;j++)
    {
        c=0;
        for(int i=0;i<l;i++)
    {

        x=(a[i]*(j))+c;
        c=x/10;
        a[i]=x%10;
    }

    while(c>0)
    {
        a[l++]=c%10;
        c=c/10;
    }

    for(int k=l-1;k>=0;k--)
    {
        ans[j]+=((long long )(a[k]));
    }
    }

    /*for(int k=l-1;k>=0;k--)
    cout<<a[j];*/

}

int funn(int n)
{

    a[0]=1;
    int c,x,l=1;
    for(int j=1;j<=n;j++)
    {
        c=0;
        for(int i=0;i<l;i++)
    {

        x=(a[i]*(j))+c;
        c=x/10;
        a[i]=x%10;
    }

    while(c>0)
    {
        a[l++]=c%10;
        c=c/10;
    }
    }
    int sum=0;
    for(int k=l-1;k>=0;k--)
    {
       sum+=a[k];
    }
   return sum;

}

int main()
{
 fun(1001);
