ll power(ll x,ll y)
{
    if( y == 0)
        return 1;
        ll temp=power(x,y/2)%mod;
        temp=(temp*temp)%mod;
        if(y&1)
        temp=(temp*x)%mod;
        return temp;
  /*  else if (y%2 == 0)

}

{
