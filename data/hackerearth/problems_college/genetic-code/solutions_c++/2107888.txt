#include <iostream>
using namespace std;

int n, sum, res;
string s;
int main () {


 
    cin >> s;
    for(int i = 0; i < s.size(); i++){
        sum++;
        if(s[i] != s[i+1]){
            if(sum % 2 == 0){
                res++;
            }
            sum = 0;
        }
    }
    cout<<res;  
    return 0;
}
