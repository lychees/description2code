#include <iostream>
    #include<bits/stdc++.h>
    using namespace std;

    int main()
    {
    int n;
    cin>>n;
    double x[n];
    double y[n];
    for(int i=0;i<n;i++)cin>>x[i]>>y[i];
    int ans=0;
    for(int i=0;i<n;i++){
    double s=-INFINITY;
    for(int j=i+1;j<n;j++){
    double t=(y[j]-y[i])/(x[j]-x[i]);
    if(t>=s){
    s=t;
    ans++;
    }
    }
    }
    cout<<ans<<endl;
    return 0;
    }
