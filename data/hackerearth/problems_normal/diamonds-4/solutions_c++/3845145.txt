#include <iostream>
using namespace std;

char b[1007][1007];

int main()
{
	int t; cin >> t;
	while (t--) {
		int n, m; cin >> n >> m;
		for (int i = 0; i < n; ++i) {
			for (int j = 0; j < m; ++j) {
				cin >> b[i][j];
			}
		}
		
		int cnt = 0;
		for (int i = 0; i < n - 1; ++i) {
			for (int j = 0; j < m - 1; ++j) {
				if (b[i][j] == '/' && b[i][j+1] == '\\' &&
					b[i+1][j] == '\\' && b[i+1][j+1] == '/')
					++cnt;
			}
		}
		cout << cnt << endl;
	}
    return 0;
}
