num = int(raw_input())
for i in range(num):
    string = raw_input()
    k = ''
    x = len(string)
    while x > 1:
        if x % 2 == 0:
            temp1 = string[x/2-1]
            temp2 = string[x/2]
            if temp1 < temp2:
                pos = x/2-1

            else:
                pos = x/2

        else:
            pos = x/2

        k += string[pos]
        # print k
        string = string[:pos]+string[pos+1:]
        x = len(string)

    k += string
    print k
