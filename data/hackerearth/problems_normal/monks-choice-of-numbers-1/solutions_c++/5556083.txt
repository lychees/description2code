#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		int n,k,sum=0;
		unsigned int num;
		cin>>n>>k;
		vector<int> ans;
		for(int i=0;i<n;i++)
			{
				cin>>num;
				ans.push_back(__builtin_popcount(num));
			}
			sort(ans.begin(),ans.end(),greater<int>());
			for(int i=0;i<k;i++)
			sum+=ans[i];
			cout<<sum<<endl;
	}
	return 0;
}
