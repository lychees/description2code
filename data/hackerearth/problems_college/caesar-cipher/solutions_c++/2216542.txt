#include<iostream>
using namespace std;
int k, i, z;
string s;

int main()
{
  cin >> z;
  
  while(z--)
  {
    cin >> k >> s;
    k %= 26;
    
    for(i = 0; i < s.size(); i++)
    {
      if ('a' <= s[i] && s[i] <= 'z')
      {
	if (int(s[i] + k) > 'z')
	{
	  s[i] += k - 26;
	}
	else
	{
	  s[i] += k;
	}
      }
      else
      {
	s[i] += k;
	if (s[i] > 'Z'){s[i] -= 26;}
      }
    }
    cout << s << endl;
  }
  
return 0;
}
