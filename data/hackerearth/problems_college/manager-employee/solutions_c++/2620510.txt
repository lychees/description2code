#include <iostream>
using namespace std;
#define ll long long

int main()
{	ll t;
    cin>>t;
    while(t--)
    {	ll a;
    	cin>>a;
    	ll z[a];
    	for(ll i=0;i<a;++i)
    		cin>>z[i];
    	for (ll i=0;i<a-1;i++)
  			for(ll j=0;j<a-i-1;j++)
    			if(z[j]>z[j+1])
      			{	ll x=z[j];
        			z[j]=z[j+1];
        			z[j+1]=x;
      			}
      	for(ll i=0;i<=a/2;++i)
      		z[i]+=z[a-i-1];
      	for (ll i=0;i<a/2;i++)
  			for(ll j=0;j<a/2-i;j++)
    			if(z[j]>z[j+1])
      			{	ll x=z[j];
        			z[j]=z[j+1];
        			z[j+1]=x;
      			}
      	cout<<z[a/2-1]-z[0]<<"\n";
    }
    return 0;
}
