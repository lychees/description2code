#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main(){

    int t;
    cin >> t;
    while(t--){

       int n , k ;
       scanf("%d%d",&n,&k);

     int a[n+1];
     for(int i = 1; i <= n ; ++i ){
      cin >> a[i];
     }
     ll ans = 0;
     for(int i = 1 ; i <= n ; ){
       int id = i;
       while(id<=n && a[id] <= k )
          ++id;
     if(id <= n){
     int times = (n - id + 1);
     int tt = ( id - i + 1);
     ans = ans + ( 1LL * tt * times);
     }
     i = id + 1;

    }
    cout << ans << endl;

    }

}
