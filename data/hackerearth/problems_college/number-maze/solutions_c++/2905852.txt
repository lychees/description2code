#include <bits/stdc++.h>
using namespace std;

int main()
{
   int l , h , i;
   cin >> l >> h;
   for (i= l+1 ; ; i++){
   	if (i % 3 == 0 && i % 5 == 0){cout << i;break;}
   	if (i % 3 != 0 && i % 5 != 0){cout << i ; break;}
   }
   for (int j = i +1 ; j < h ; j++){
   		if (j % 3 == 0 && j % 5 == 0){cout <<','<<j;}
   	if (j % 3 != 0 && j % 5 != 0){cout <<','<< j;}
   }
    return 0;
}
