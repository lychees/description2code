total = input()
x = []
x = map(int, raw_input().split())
evenList = []
oddList = []
sumE = 0
sumO = 0
for i in x:
    if i % 2 == 0:
        evenList.append(i)
        evenList.sort()
        sumE = sumE + i
    else:
        oddList.append(i)
        oddList.sort()
        sumO = sumO + i
evenList.append(sumE)
oddList.append(sumO)
evenList.extend(oddList)
print ' '.join(map(str, evenList))
