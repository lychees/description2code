#include <bits/stdc++.h>
#define oo 10005
#define N 100005

using namespace std;

int a[N];

bool check(int dist, int n, int k) {
    int d = 1, c = 0, tmp = 0;
    for (int i = 1; i <= n; i++) {
        while (c != n+1 && a[c]-a[d] <= dist)
        c++;
        tmp++;
        if (c == n+1) break;
        d = c-1;
    }
    if (c != n+1) return false;
    if (tmp <= k) return true;
    return false;
}

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0); cout.tie(0);
   
    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        for (int i = 1; i <= n; i++)
            cin >> a[i];

        int d = 1, c = 1e9, res = -1;

        while (d <= c) {
            int mid = (d+c)/2;
            if (check(mid, n, k))
            res = mid,c = mid-1;
            else d = mid+1;
        }
        cout << res << "\n";
    }
    return 0;
}
