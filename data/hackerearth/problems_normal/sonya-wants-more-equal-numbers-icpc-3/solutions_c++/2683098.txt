#include<bits/stdc++.h>
using namespace std;
const int N=111;
const int maxn=111111;
int a[N],dp[maxn][N];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        scanf("%d",&a[i]);
        a[i]+=a[i-1];
    }
    for(int i=1;i<=n;i++){
        for(int j=i;j>=1;j--){
            int s=a[i]-a[j-1];
            dp[s][i]=max(dp[s][i],dp[s][j-1]+1);
        }
        for(int j=1;j<maxn;j++) dp[j][i]=max(dp[j][i],dp[j][i-1]);
    }
    int cnt=1;
    for(int i=1;i<maxn;i++) cnt=max(cnt,dp[i][n]);
    printf("%d\n",cnt);
    for(int i=1;i<maxn;++i)
    if(dp[i][n]==cnt) printf("%d ",i);
    return 0;
}
