#define ll long long unsigned int

#define Max 1000050
using namespace std;
typedef vector <int> V;
typedef pair <int,int> P;
ll A[Max],B[Max];	V v;
void sieve()
{
	for(int i=3;i<=Max;i++)
	{
		A[i]=1;	B[i]=0;
	}
	int s=sqrt(Max);	B[2]=2;	B[1]=B[0]=0;
	for(int i=3;i<=s;i+=2)
	{
		if(A[i])
		{
			for(int j=i*i;j<=Max;j+=2*i)
				A[j]=0;
		}
	}
	for(int i=3;i<=Max;i++)
	{
		if(A[i]&& (i&1))
			B[i]=B[i-1]+i;
		else
			B[i]=B[i-1];
	}
}

	sieve();	ll sum;	int t,a,b;

	}

}
