set<int> s;
int m, k, idx;

void Pro(int x, int num) {
  if ((int)s.size() >= m)
    return;
  if (!num) {
    int dif = abs(abs(x) - k);
    if (dif < 10000)
      s.insert(idx * 10000 + dif), s.insert(dif * 10000 + idx);
    return;
  }
  Pro(x + num % 10, num / 10);
  Pro(x - num % 10, num / 10);
  Pro(x * (num % 10), num / 10);
}
int main() {
#ifndef ONLINE_JUDGE

#endif
  cin >> k >> m;
  for (int i = 0; i < 10000; ++i)
    idx = i, Pro(i % 10, i / 10);
  int cnt = 0;
  for (set<int>::iterator it = s.begin(); it != s.end(); ++it, ++cnt) {
    if (cnt == m)
      break;
    printf("%08d\n", *it);
  }
  fclose(stdin);
  fclose(stdout);
  return 0;
}
