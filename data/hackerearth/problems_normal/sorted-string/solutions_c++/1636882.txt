#include<bits/stdc++.h>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    int t;
    cin>>t;

    while(t--)
    {
        string s;
        cin>>s;
        pair<int, char> arr[26];
        for(int i = 0; i < 26; ++i)
        {
            arr[i].second = char('a' + i);
            arr[i].first = 0;
        }

        for(char c : s)
        {
            arr[c - 'a'].first ++;
        }
        sort(arr, arr+26);
        for(int i = 0; i < 26; i++)
        {
            for(int j = 0; j < arr[i].first; j++)
                cout<<arr[i].second;
        }
        cout<<endl;
    }

    return 0;
}
