#include <iostream>
using namespace std;
int max(int a, int b){
	return(((a*b)+1)/2);
}

int solve(int w, int h){
	int ans = max(w/2, h/2)+ max((w+1)/2, h/2)+ max(w/2, (h+1)/2)+ max((w+1)/2, (h+1)/2);
	return ans;
	
}
int main()
{
    int t;
    scanf("%d", &t);
  while(t--)    
    {
    	int w, h;
    scanf("%d %d", &w, &h);
    int ans = solve(w, h);
    printf("%d\n", ans);
	}
    
    
    return 0;
}
