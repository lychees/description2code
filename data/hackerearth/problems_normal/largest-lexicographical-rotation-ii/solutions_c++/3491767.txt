#include<bits/stdc++.h>
using namespace std;

int main()
{
	std::ios::sync_with_stdio(false);
	long i = 0,curr=0,len=0;
	char str[4000001];
	cin>>str;
	for(len = 0;str[len]!='\0';len++);
	
	for(i=1;i<len;i++)
	{
		if(str[i]>str[curr])
			curr = i;
		else if((str[i]==str[curr])&&(str[i]!=str[i-1]))
		{
			long tmp_curr = curr;
			long tmp_i = i;
			curr++;
			i++;
			bool flag = false;
			bool flag1 = false;
			bool flag2 = false;
			while(i<len)
			{
				if(str[i]>str[curr])
				{
					flag = true;
					flag1 = true;
					break;
				}
				else if(str[i]<str[curr])
				{
					flag2 = true;
					break;	
				}
				curr++;
				i++;
			}
			if((flag==false)&&(i==len))
			{
				i=0;
				while(i<tmp_curr&&curr<tmp_i)
				{
					if(str[i]>str[curr])
					{
						flag = true;
						flag1 = false;
						break;
					}
					else if(str[i]<str[curr])
					{
						flag2 = false;
						break;
					}
					curr++;
					i++;
				}
			}
			if(flag)
			{
				if(flag1)
				{
					if(str[i]>str[tmp_i])
					{
						curr = i;
					}
					else
					{
						curr = tmp_i;
					}
				}
				else
				{
					curr = tmp_i;
				}
				i = curr;
			}
			else
			{
				if(flag2)
				{
					if(str[curr]<=str[tmp_curr])
						curr = tmp_curr;
				}
				else
				{
					curr = tmp_curr;
					i = tmp_i;
				}
			}
			
		}
	}
	for(i=curr;i<len;i++)
		cout<<str[i];
	for(long j=0;j<curr;j++)
		cout<<str[j];
    return 0;
}
