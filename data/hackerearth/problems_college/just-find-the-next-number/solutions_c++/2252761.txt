#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    long long int n,a,b,x;
    cin>>n;
    x=n;
    do
    {
    	a=x%2;
    	b=(x/2)%2;    	
		
    	if(b==1&&a==1)
    	{	
    		n++; 
    		x=n;
    	}
    	x/=2;
    	
    }while(x!=0);
    
    cout<<n<<"\n";
    }
    return 0;
}
