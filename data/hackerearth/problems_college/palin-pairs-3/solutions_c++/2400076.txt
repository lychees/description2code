#include<bits/stdc++.h>
#include<string>
using namespace std;
#define llu unsigned long long

int main()
{
    int t;
    cin>>t;
    map<string,int> M;
    llu count=0;
    string s;
    while(t--)
    {
        cin>>s;
        count+=M[s];
        reverse(s.begin(),s.end());
        M[s]++;
    }
    cout<<count<<endl;
    return 0;
}
