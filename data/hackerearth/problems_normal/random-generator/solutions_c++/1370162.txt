#include<bits/stdc++.h>
#define ll long long
#define ld double
#define f first
#define s second
#define pb push_back
#define M 1000000007
#define MX 1000001
#define INF 10000000000
#define PI 3.141593
#define mmx(a,b) a>b?a:b
#define mmn(a,b) a<b?a:b
using namespace std;
typedef pair<ll,ll> ii;
typedef vector<ll> vv;
ll arr[100005];
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
        ll n,k,p,fl=0;
        cin>>n>>k>>p;
        for(ll i=0;i<n;i++)
            cin>>arr[i];
        sort(arr,arr+n);
        for(ll i=0;i<n;i++)
        {
            ll x=i+k-1;
            if(x>=n)
                break;
            ll diff=arr[x]-arr[i];
            if(diff<=2*p)
            {
                fl=1;
                break;
            }
        }
        if(fl)
            cout<<"NO\n";
        else cout<<"YES\n";
    }
}
