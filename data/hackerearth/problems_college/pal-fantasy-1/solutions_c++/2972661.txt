    #include<stdio.h>
    #include<string.h>
    int main()
    {
        int t,i,j,l,flg;
        char s[100],r[100];
        scanf("%d",&t);
        while(t--)
        {
            scanf("%s",s);
            l=strlen(s);
            for(i=0;i<l;i++)
            {
                r[l-i-1]=s[i];
            }
            for(i=0;i<l;i++)
            {
                    flg=0;
                    for(j=0;j<l-i;j++)
                    {
                            if(s[i+j]==r[j])
                                flg=1;
                            else
                            {
                                flg=0;
                                break;
                            }
                    }
                    if(flg==1)
                        break;
            }
     
            printf("%d\n",i+l);
        }
        return 0;
    }
