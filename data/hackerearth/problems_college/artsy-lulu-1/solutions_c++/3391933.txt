#include <iostream>
using namespace std;
void triangle(char sym, int h)
{
	for(int i = 0 ; i < h; i++ )
	{
		for(int j = 0; j <=i; j++)
			cout << char(i+sym) << " ";
		cout << endl;
	}
}
int main()
{
	char sym = ' ';
	int h;
	while(cin>> sym >> h)
	{
		
		if(h >=2 and h<=26 and
			((sym>= 'a' and h+sym-1 <= 'z') or (sym >='A' and h+sym-1 <='Z')))
		{
		
			triangle(sym, h);
			//cout<< "VALID"<<endl;
		}	
		else
			cout << "INVALID" << endl;
			//cout << "INVALID" << " " << int(sym) << " " << h  << " " <<char(h+sym) <<endl;
			
		cout << endl;
	}
    return 0;
}
