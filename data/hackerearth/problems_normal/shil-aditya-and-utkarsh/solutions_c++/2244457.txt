#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
 
int main()
{
     int n;
     cin>>n;
     int j=2;
     int i=1;
     long long ans=0;
     while(j<=n-1)
         {
             ans=ans+(n-i)*(n-j);
             i++;
             j++;
     }
      cout<<ans<<endl;
     
     
      
     
     
}
