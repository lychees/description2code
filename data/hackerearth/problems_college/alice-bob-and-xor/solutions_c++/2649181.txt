#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int k,ans;
	cin>>k;
	while(k--)
	{
		int l,r,x,y,xl;
		cin>>l>>r;
		if(l==1)
		{
			x=1;
			if(r==1)
			y=1;
			else
			{
				if(r&1)
				y=((r-2+1)/2)%2;
				else
				y=((r-1-2+1)/2)%2 ^ r;
			}
			ans=x^y;
		}
		else
		{
			if(l&1)
			x=((l-2+1)/2)%2;
			else
			x=((l-1-2+1)/2)%2 ^ l;
			x^=l;
			if(r&1)
			y=((r-2+1)/2)%2;
			else
			y=((r-1-2+1)/2)%2 ^ r;
			ans=x^y;
		}
		cout<<ans<<endl;
	}
	return 0;
}
