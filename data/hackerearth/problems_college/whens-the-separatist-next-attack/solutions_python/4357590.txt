testcases = int(raw_input())
for test in range(1,testcases+1):
	separatistnumber = raw_input()
	length = len(set(separatistnumber))
	if length == 1:
		length = 2
	base_values = range(length)
	base_values[0],base_values[1] = base_values[1],base_values[0]
	base_system_assign = {}
	for char in separatistnumber:
		if not char in base_system_assign:
			base_system_assign[char] = base_values.pop(0)

	value = 0
	separatistnumber = separatistnumber[::-1]
	for index, char in enumerate(separatistnumber):
		value += base_system_assign[char]*pow(length,index)
	print(value)
