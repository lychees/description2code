#include <bits/stdc++.h>

using namespace std;
typedef long long ll;

int main() {
	int tt;
	scanf("%d", &tt);
	while (tt--) {
		int n;
		scanf("%d", &n);
		ll r, c, x;
		ll val = -1;
		bool ans = true;
		for (int i = 0; i < n; i++) {
			scanf("%lld %lld %lld", &r, &c, &x);
			ll v = (r ^ c ^ x) % 2;
			if (i) {
				if (val != v) {
					ans = false;
				}
			} else {
				val = v;
			}
		}
		puts(ans ? "Yes" : "No");
	}
}
