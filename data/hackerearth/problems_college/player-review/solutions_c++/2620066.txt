#include <bits/stdc++.h>

using namespace std;

int main()
{
        int n;
        long long sum=0,ans=0;
        scanf("%d",&n);
        int a[n];
        for(int i=0;i<n;i++){
            scanf("%d",&a[i]);
            sum+=a[i];
            if(sum<0)sum=0;
            ans=max(sum,ans);
        }
    printf("%lld\n",ans);
    return 0;
}
