#include <iostream>
typedef long long int lli;
using namespace std;

lli GCD(lli a, lli b)
{
	return a%b==0?b:GCD(b,a%b);
}

int main()
{
    int t;
    lli a,b,r;
    cin>>t;
    while(t--){
    	scanf("%lld%lld",&a,&b);
    	if(a>b)
    		r=GCD(a,b);
    	else 
    		r=GCD(b,a);
    		
    	if(r==1)
    		cout<<"Is a Co-Prime"<<endl;
    	else 
    		cout<<"Not a Co-Prime"<<endl;
    		
    }
}
