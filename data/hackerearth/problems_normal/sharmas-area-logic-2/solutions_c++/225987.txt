#include<iostream>
#include<string>
#include<cstdio>
#include<algorithm>
#define N 1001
#define ll long long int
#define FOR(i,a) for(i=0;i<a;i++)
using namespace std;

ll AreaOfRectrangle(int a[],int b[],int n)
{
    //ll area;
    int x1=a[0],x2=a[0],y1=b[0],y2=b[0];
    int i=0,j=0;
    for(i=0;i<n;i++)
    if(x1<a[i])//rightmost x coordinate
    x1=a[i];

    for(i=0;i<n;i++)
    if(x2>a[i])//leftmost x coordinate
    x2=a[i];

    for(i=0;i<n;i++)
    if(y1<b[i])//rightmost y coordinate
    y1=b[i];

    for(i=0;i<n;i++)
    if(y2>b[i])//leftmost y coordinate
    y2=b[i];

    //printf("%d %d %d %d %d\n",x1,x2,y1,y2,abs((x1-x2)*(y1-y2)));
    ll area = abs((x1-x2)*(y1-y2));
    if(area!=0)
    return area;
    else
    {
        return max(abs(x1-x2),abs(y1-y2));
    }
}

int main()
{
    int i,n,t;
    int a[N],b[N];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(i=0;i<n;i++)
        scanf("%d%d",&a[i],&b[i]);
        printf("%d\n",AreaOfRectrangle(a,b,n));
    }
    return 0;
}
