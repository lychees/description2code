#include <iostream>
using namespace std;

int main()
{
	int n,i,x,y,t;
   	cin>>n;
   	int arr[n+1];
   	for(int i=0;i<n+1;i++)
   	{
   		arr[i]=1;
   	}
   	cin>>t;
   	cout<<"\n";
   	while(t--)
   	{
   		cin>>x;
   		cin>>y;
   		arr[y]=0;
   	}
   	
   	int tot=0;
   	for(i=1;i<n+1;i++)
   	{	if(arr[i]>0)
   			tot++;
   	}
   	cout<<tot;
    return 0;
}
