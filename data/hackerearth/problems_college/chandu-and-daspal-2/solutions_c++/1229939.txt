#include <iostream>
#include<string.h>
#include<iomanip>
using namespace std;

 int gcd(int a,int b)
	{
		if(b==0)
			return a;
		if(a==0)
			return b;
		return(gcd(b,a%b)); 
	}
int main()
{
    int t,a,b,i,ans;
    string s1("");
    string s2("");
    cin >> t;
    while(t--)
    {
    	i=0;
    	cin >> a >> b >> s1;
    	if(s1=="Chandu")
    		s2="Daspal";
    	else
    		s2="Chandu";
    	while(1)
    	{
 			ans=gcd(a,b);
 			a-=ans;
 			b-=ans;
 			if(a==0||b==0)
 				break;
 			i++;
    	}
    	if(i%2==0)
    		cout << s1 << endl;
    	else
    		cout << s2 << endl;
    }
    return 0;
}
