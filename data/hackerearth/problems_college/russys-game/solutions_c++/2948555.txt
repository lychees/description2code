    #include<iostream>
    using namespace std;
     
    int main()
    {
    	int arr[100],a,t,i,flag=1,n;
    	cin>>t;
    	while(t--)
        {
            flag=0;
            cin>>n;
            for(i=0;i<n;++i)
                cin>>arr[i];
            i=0;
            while(1)
            {
                a=arr[i];
                if(a==-1)
                {
                    cout<<"Win"<<endl;
                    flag=1;
                    break;
                }
                if(a==-2)
                {
                    flag=2;break;
                }
                else
                {
                    arr[i]=-2;
                    i=a;
                }
            }
            if(flag!=1)
                cout<<"Lose"<<endl;
        }
    	return 0;
    }
