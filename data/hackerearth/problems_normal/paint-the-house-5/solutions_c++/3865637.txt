#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef unsigned long long ull;
typedef pair<int, int> ii;

struct data
{ ll t, x, y; };

const int N = 1e5+5;
int n; ll d;
data a[N];

bool cmp(data x, data y) { return (x.t < y.t) || ((x.t == y.t && x.y > y.y) || (x.t == y.t && x.y == y.y && x.x < y.x)); }

void Solve()
{
    sort(a + 1, a + n + 1, cmp);
    ll res = 0, py = -1, pt = -1;
    for (int i = 1; i <= n;)
    {
        if (py == -1 || a[i].y > py)
        {
            if (py != -1) d -= (a[i].t - pt) * py;
            if (d <= 0) break;
            res += a[i].x, pt = a[i].t, py = a[i].y;
        }
        int t = a[i].t;
        for (; i <= n && a[i].t == t; ++i);
    }
    cout << res;
}

int main()
{
    //freopen("inp.txt", "rt", stdin);
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    cin >> n >> d;
    for (int i = 1; i <= n; ++i) cin >> a[i].t >> a[i].x >> a[i].y;
    Solve();
    return 0;
}
