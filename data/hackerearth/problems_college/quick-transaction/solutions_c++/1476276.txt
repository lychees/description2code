#include<bits/stdc++.h>
using namespace std;
int main()
{
        int t,ans,c,i,n;
        cin>>t;
        int a[]={1,2,5,10,20,50,100,500,1000};
        while(t--)
        {
                c=0;
                cin>>n;
                for(i=8;i>=0;i--)
                {

                        if(a[i]<=n)
                        {
                                ans=n/a[i];
                                c=c+ans;
                                n=n-ans*a[i];
                        }
                }
                cout<<c<<'\n';
        }
        return 0;
}
