#include <iostream>
using namespace std;

int main()
{
    int n,a[100002],i,j,k,count;
    cin>>n;
    for(i=0;i<n;i++)
     cin>>a[i];
   count=0;  
   for(i=0;i<n-2;i++)
   {
   	  for(j=i+1;j<n-1;j++)
   	   {
   	   	  for(k=j+1;k<n;k++)
   	   	  {
   	   	  	  if(a[j]-a[i]==a[k]-a[j])
   	   	  	    count++;
   	   	  }
   	   }
   }
   cout<<count<<endl;
    return 0;
}
