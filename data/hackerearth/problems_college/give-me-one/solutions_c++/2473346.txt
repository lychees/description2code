#include <iostream>
using namespace std;
int a[20000005];
int SolveIt(int N)
{
	int x;
	if(N<=1)
	{
		return 0;
	}
	if(a[N]!=-1)
	{
		return a[N];
	}
	x=1+SolveIt(N-1);
	if(N%2==0)
	{
		x=min(x,1+SolveIt(N/2));
	}
	if(N%3==0)
	{
		x=min(x,1+SolveIt(N/3));
	}
	a[N]=x;
	return x;
}
int main()
{
	int T,i,N;
    for(i=0;i<20000005;i++)
    {
    	a[i]=-1;
    }
    a[0]=a[1]=0;
    for(i=1;i<20000005;i+=1000)
    {
        SolveIt(i);
    }
    SolveIt(20000004);
    cin>>T;
    while(T--)
    {
    	cin>>N;
    	cout<<a[N]<<endl;
    }
}
