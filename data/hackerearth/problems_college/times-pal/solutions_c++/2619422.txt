#include <bits/stdc++.h>
using namespace std;
int is_palin(string str)
{
	int len=str.length(),l1;l1=len/2;
	
	for(int i=0;i<l1;i++)
	{
		if(str[i]!=str[len-1-i]) return 0;
	}
	return 1;
}
int main()
{
    int t,i;
    scanf("%d",&t);
    while(t--)
    {
    	string str1,str2,str3;
    	cin>>str1;
    	cin>>str2;
    	int count=0,le1=str1.length(),le2=str2.length();
    		str3=str1;
    		for(i=0;i<=le1;i++)
    		{
    			str3.insert(i,str2);
    		//	cout<<str3<<endl;
    			if(is_palin(str3)==1) count++;
    			str3=str1;
    		}
    	cout<<count<<endl;
    	
    }
    return 0;
}
