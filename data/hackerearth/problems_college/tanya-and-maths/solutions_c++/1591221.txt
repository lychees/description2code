#include <iostream>
#define MAX 100001
using namespace std;

int phi[MAX] = {0};
 
void euler_phi() {
    phi[1] = 1;
    for(int i=2; i<MAX; i++) {
        if(!phi[i]) {
            phi[i] = i-1;
            for(int j=(i<<1); j<MAX; j+=i) {
                if(!phi[j]) phi[j] = j;
                phi[j] = phi[j]/i*(i-1);
            }
        }
    }
}

int main()
{
    int t;
    cin>>t;
    euler_phi();
    while(t--){
    	int a, b, sum=0, i;
    	cin>>a>>b;
    	for(i=1; i<=a && i<=b; i++){
    		if(a%i==0 && b%i==0)
    			sum+=phi[i];
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
