    #include <iostream>
    using namespace std;
     
    int main()
    {
        int T,N;
        scanf("%d",&T);
        while(T--)
        {
        	scanf("%d",&N);
        	int A[N],B[N],i,flag=1;
        	for(i=0;i<N;i++)
        	 scanf("%d",&A[i]);
        	for(i=0;i<N;i++) 
        	 B[A[i]-1]=i+1;
        	for(i=0;i<N;i++)
        	 if(A[i]!=B[i])
        	  {flag=0;break;}
        	if(flag)
        	 printf("inverse\n");
        	else
        	 printf("not inverse\n");
        }
        return 0;
    }
