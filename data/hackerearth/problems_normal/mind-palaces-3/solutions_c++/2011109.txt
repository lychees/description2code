#include <stdio.h>

int main()
{
    int n,m,i,j,t,k,flag;
    scanf("%d%d",&n,&m);
    long int a[n][m];
    for(i=0;i<n;i++)
    {
    	for(j=0;j<m;j++)
    	{
    		scanf("%ld",&a[i][j]);
    	}
    }
   scanf("%d",&t);
   while(t--)
   {
   		flag=0;
   		scanf("%d",&k);
   		for(i=0;i<n;i++)
   		{
    		for(j=0;j<m;j++)
    		{
    			if(a[i][j]==k)
    			{
    				flag=1;
    				goto A;
    			}
    		}
   		 }
   		 A:
   		 if(flag==1)
   		 	printf("%d %d\n",i,j);
   		 else
   		 	printf("%d %d\n",-1,-1);
   }
   return 0;    
}
