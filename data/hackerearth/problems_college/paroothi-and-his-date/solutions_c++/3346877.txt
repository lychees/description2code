#include <iostream>
using namespace std;

int main() {
	// your code goes here
	long long int a[1000000]={0};
	long long int n;
	cin>>n;
	long long int max=0,max2=a[0],p,i;
	
	for(i=0;i<n;i++)
		{
		cin>>p;
		a[p]++;
		if(a[p]>max2)
			{
			max=p;
			max2=a[p];
			}
		if(max2==a[p] && p<max)
			{
			max=p;
			max2=a[p];
			}
		
		}
	cout<<max;
	
	return 0;
}
