#include <bits/stdc++.h>
using namespace std;

int main()
{
    int test;
    scanf("%d",&test);
    while(test--){
    	int n,c;
    	scanf("%d %d",&n,&c);
    	int a[n][n];
    	for(int i = 0;i<n;i++){
    		for(int j = 0;j<n;j++){
    			scanf("%d",&a[i][j]);
    		}
    	}
    	for(int i=1;i<n;i++){
    		a[i][0] += a[i-1][0];
    	}
    	for(int i=1;i<n;i++){
    		a[0][i] += a[0][i-1];
    	}
    	for(int i=1;i<n;i++){
    		for(int j=1;j<n;j++){
    			a[i][j] += max(a[i-1][j],a[i][j-1]);
    		} 
    	}
    	if(a[n-1][n-1]<c)cout<<"-1\n";
    	else cout<<a[n-1][n-1]-c<<endl;
    }
    return 0;
}
