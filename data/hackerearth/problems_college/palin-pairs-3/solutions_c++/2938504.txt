#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    map<string ,int > m;
    int n;
    long long count=0;
    cin>>n;
    while(n--){
    	string str;
    	cin>>str;
    	count+=m[str];
    	reverse(str.begin(),str.end());
    	m[str]++;
    }   
    cout<<count;
    return 0;
}
