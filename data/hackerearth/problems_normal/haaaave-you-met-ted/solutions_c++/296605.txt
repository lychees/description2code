#include <iostream>
#include <algorithm>
using namespace std;
int bitcount(int n){
    int count = 0;
    while(n > 0){
        count = count + 1;
        n = n & (n-1);
    }
    return count;
}
int main()
{
    int t;
	cin>>t;
	int n;

	while(t--)
	{
		cin>>n;
		int a[n];
		int ans=1000000;
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
			ans=min(bitcount(a[i]),ans);
		}
			
		
		cout<<ans<<"\n";
	}
	return 0;
}
