#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int a[n],i=0;
	for(int k=0;k<n;k++)
	{
		char s[20];
		cin>>s;
		a[i++]=s[0];
	}
	 sort(a,a+n);
	int cnt=1,count=1;
	for(i=1;i<n;i++)
	{
		if(a[i]==a[i-1])
		{
			cnt++;
			if(cnt>10)
			{
				count++;
				cnt=1;
				
			}
			
		}
		else
		{
			count++;
			cnt=1;
		}
	}
	cout<<count<<endl;
	return 0;
}
