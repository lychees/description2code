#include <iostream>
#include <string>
using namespace std;

int main()
{
    int n, i, k;
    cin>>n;
    string s[n];
    for(i=0;i<n;i++)
    	cin>>s[i];
    for(i=0;i<n-1;i++){
    	for(k=i+1;k<n;k++){
    		if(s[i].compare(s[k])>0){
    			string temp=s[i];
    			s[i]=s[k];
    			s[k]=temp;
    		}
    	}
    }
    k=1;
    for(i=0;i<n;i++){
    	if(s[i]!=s[i+1]){
    		cout<<s[i]<<" "<<k<<endl;
    		k=1;
    	}
    	else k++;
    }
}
