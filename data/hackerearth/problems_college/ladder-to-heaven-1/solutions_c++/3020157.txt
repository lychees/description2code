#include <iostream>
using namespace std;
int main()
{
    long long a,b,c,t,n,sum;
    cin>>t;
    while(t!=0)
    {
     cin>>n;
     sum=2;
     a=1,b=2,c=3;
     while(c<=n)
     {
      c=a+b;
      a=b;
      b=c;
      if(c%2==0 && c<=n)
         sum=sum+c;
     }
     cout<<sum<<"\n";
     t--;
    }
    
    return 0;
}
