#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    double a,b,p,mid;
    cin>>b>>a>>p;
    p/=100;
    double l,r,t;
    l=0.00;
    r=a;
    while(l<=r)
    {
    	mid=(l+r)/2;
    	t=a-mid;
    	//cout<<mid<<"-"<<(t*t)/(a*a)<<endl;
    	if(((t*t)/(a*a))==p)
    		break;
    	else if(((t*t)/(a*a))<p)
    		r=mid-0.000001;
    	else
    		l=mid+0.000001;
    }
    printf("%.2lf",mid);
    return 0;
}
