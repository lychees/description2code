#define ll long long

#define mx 100005

using namespace std;

int main()
{
	ll N,x;
	cin>>N;
	
	vector <ll> V;
	for(int i=0;i<N;i++)cin>>x , V.push_back(x);
	
	ll sm  = 0;
//	vector<ll> :: iterator it , jt;
	for(int i=0;i<N;i++)
	{
		if(V[i] == 0)
		{
			int j = i-1;
			while(V[j] <= 0)j--;
			
			if(j >= 0)
			V[j] = -1;
		}
	}
	
	for(int i=0;i<V.size();i++)
		if(V[i] != -1)
		sm += V[i];

	cout<<sm<<endl;
   return 0;
}
