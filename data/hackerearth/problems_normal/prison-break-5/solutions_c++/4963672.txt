#include <iostream>

using namespace std;

int N,Answer;
bool a[25][25],visited[25][25];

bool isSafe(int i,int j){
	return (i>=0 && i<N && j>=0 && j<N && a[i][j]==0 && visited[i][j]==false);
}

void DFS(int x,int y){

	if(x==N-1 && y==N-1){
		Answer++;
		return;
	}
	visited[x][y]=true;

	int x1[]={-1,0,1,0};
	int y1[]={0,1,0,-1};

	for(int i=0;i<4;i++){
		if(isSafe(x+x1[i],y+y1[i])){
			DFS(x+x1[i],y+y1[i]);
		}
	}

	visited[x][y]=false;
}

int main(){

	int T;
	cin>>T;
	for(int t=1;t<=T;t++){
		cin>>N;

		Answer=0;

		for(int i=0;i<N;i++){
			for(int j=0;j<N;j++){
				cin>>a[i][j];
				visited[i][j]=false;
			}
		}

		DFS(0,0);

		cout<<Answer<<endl;
	}
	return 0;
}
