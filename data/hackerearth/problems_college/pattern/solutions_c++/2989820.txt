#include <iostream>
using namespace std;

int main()
{
    int n,max;
    cin>>n;
    max=n-1;
    for(int i=0;i<n;i++)
    {
     if(i==n-1)
      max=n-1;	 
     for(int j=0;j<=max;j++)
     {
     	if((i==0)||(i==n-1))
     	 cout<<"*";
     	if(((i!=0)&&(i!=n-1))&&(j==max))
     	 cout<<"*";
     	if(((i!=0)&&(i!=n-1))&&(j!=max))
     	 cout<<" ";
     }
     max--;
     cout<<"\n";
    }
    return 0;
}
