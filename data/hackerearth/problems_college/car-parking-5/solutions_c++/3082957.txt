#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
 
int main()
{
	ios_base::sync_with_stdio(0);
    int testCases;
    cin>>testCases;
    for(int c=0;c<testCases;c++)
    {
    	int count=0;
    	string s;
    	cin>>s;
    	int len=s.size();
    	for(int i=0;i<len;i++)
    	{
    		if(s[i]=='S')
    		{
    			if((i-1)>=0)
    				s[i-1]=0;
    			s[i]=0;
    			if((i+1)<len && s[i+1]!='S')
    				s[i+1]=0;
    		}
    		if(s[i]=='B')
    		{
    			if((i-1)>=0)
    				s[i-1]=0;
    			s[i]=0;
    			if((i-2)>=0)
    				s[i-2]=0;
    		}
    		if(s[i]=='D')
    			s[i]=0;
    	}
    	for(int i=0;i<len;i++)
    		if(s[i]=='-')
    			++count;
    	cout<<count<<"\n";
	}
    return 0;
}
