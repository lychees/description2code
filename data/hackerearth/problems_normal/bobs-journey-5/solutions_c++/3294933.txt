#include <iostream>
#include <map>
#include <string>
using namespace std;

int main()
{
	int T;
	cin>>T;
	int str[26];
	string s;
	while(T) {
		int N,flag=0;
		cin>>N;
		int i=0;
		for(i=0;i<26;++i) str[i] = 0;
		for(i=0;i<N;++i) {
			cin>>s;
			if(str[s[0]-'a']){
				flag=1;
			}else{
				str[s[0]-'a']=1;
			}
		}
		if(flag)cout<<"NO"<<endl;
		else cout<<"YES"<<endl;
		T--;
	}
   
    return 0;
}
