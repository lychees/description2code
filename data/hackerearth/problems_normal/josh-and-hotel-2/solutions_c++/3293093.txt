#include <bits/stdc++.h>
using namespace std;

int main()
{
    char str[53];
    bool letters[26],taken[26];
    int x;
    cin>>str;
    while(str[0]!= '0')
    {
    	cin>>x;
    	int count=0;
    	for(int i=0;i<26;i++)
    		{
    			letters[i]=false;
    			taken[i]=false;
    		}
    	for(int i=0;i<strlen(str);i++)
    	{
    		if((letters[str[i]-'A'])==false)	//first occurence in pair
    			{
    				if(x>0)	//room availabe
    					{
    						x--;
    						taken[str[i]-'A']=true;
    					}
    				else
    					count++;//must leave without staying...
    				letters[str[i]-'A']=true;	
    			}
    		else	//second occurence in pair
    			{
    				if(taken[str[i]-'A']==true)
    					{
    						x++;
    						taken[str[i]-'A']=false;
    						
    					}
    				letters[str[i]-'A']=false;
    				
    			}
    	}
    	cout<<count<<endl;
    	cin>>str;
    	
    }
    
    return 0;
}
