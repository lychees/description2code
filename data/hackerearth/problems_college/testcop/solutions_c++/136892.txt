#include <iostream>
using namespace std;

int gcd(int a, int b)
{
	return (b==0) ? a : gcd(b,a%b);
}

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n;
    	cin >> n;
    	int ans = 0;
    	int num;
    	for(int i=0;i<n;i++)
    	{
    		cin >> num;
    		ans = gcd(ans,num);
    	}
    	if(ans==1)
    	{
    		cout << "YES" << endl;
    	}
    	else
    		cout << "NO" << endl;
    }
    return 0;
}
