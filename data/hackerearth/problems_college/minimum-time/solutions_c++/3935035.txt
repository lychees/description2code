#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

int main()
{
    double l, a,v, d, w,t=0.0;
    double s1,s2,s3, r,v1;
    cin>>a>>v;
    cin>>l>>d>>w;
    
    s1=w*w/(2.0*a);
    if(s1>=d){
    	t+=sqrt(2.0*d/a);
    	w=a*t;
    }
    else {
        t+=w/a;
    	s2=((v*v)-(w*w))/(2.0*a);
    	r=d-s1;
    	if(r/2.0<=s2){
    		v1=sqrt((a*r)+(w*w));
    		t+=((v1-w)/a)*2.0;
    	}
    	
    	else {
    		t+=2.0*((v-w)/a);
    		r-=2.0*s2;
    		t+=r/v;
    	}
    }
    
    r=l-d;
    s1=(v*v-w*w)/(2.0*a);
    
    if(s1>=r){
    	v1=sqrt((w*w)+(2.0*a*r));
    	t+=(v1-w)/a;
    }
    
    else{
    	t+=(v-w)/a;
    	r-=((v*v)-(w*w))/(2.0*a);
    	t+=r/v;
    }
    
    cout<<std::fixed<<std::setprecision(12)<<t;
    return 0;
}
