#include <bits/stdc++.h>
using namespace std ;
#define MAXN 200005
#define MOD 1000000007

int T , N , ans , SN[MAXN] , DP[MAXN] , SUM[MAXN] , base ;
char str[MAXN] ;

int main(){
	scanf("%s",str+1) ;
	N = strlen(str+1) ;
	base = 1 ;
	for(int i=N;i>=1;i--){
		SN[i] = ( (1LL * (str[i]-'0') * base) % MOD + SN[i+1] );
		SN[i] %= MOD ;
		base = (1LL * base * 10) % MOD;		
	}
	for(int i=1;i<=N;i++){
		SUM[i] = SUM[i-1] + (str[i]-'0' );
	}
	ans = 0 ;
	DP[1] = SUM[N] ;
	ans += DP[1] ;
	ans %= MOD ;		
	for(int i=2;i<=N;i++){
		DP[i] = ( 1LL * ( ( DP[i-1] - SN[N-i+2] ) % MOD + MOD) % MOD * 10 ) % MOD + (SUM[N] - SUM[i-1]) ;
		DP[i] %= MOD ; 		
		ans += DP[i] ;
		ans %= MOD ;	
	}
	printf("%d\n",ans) ;
	return 0 ;
}
