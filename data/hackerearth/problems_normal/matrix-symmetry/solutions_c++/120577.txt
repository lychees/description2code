#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
	    int n,i;
	    char a[50][50];
	    cin>>n;
	    i=0;
	    while(i<n)
	    {
	        cin>>a[i];
	        i++;
	    }
	    int flagh=1,flagv=1;
	    for(i=0;i<n;i++)
	    {
	        for(int j=0;j<n;j++)
	        {
	            if(a[i][j]!=a[n-i-1][j])
	            {
                    flagh=0;
	            }
	            if(a[i][j]!=a[i][n-j-1])
	            {
	                flagv=0;
	            }
	        }
	    }
	    end:
	    if(flagh==1 && flagv==1)
	    {
	        cout<<"BOTH\n";
	    }
	    else if(flagh==1 && flagv==0)
	    {
	        cout<<"HORIZONTAL\n";
	    }
	    else if(flagh==0 && flagv==1)
	    {
	        cout<<"VERTICAL\n";
	    }
	    else
	    {
	        cout<<"NO\n";
	    }
	}
	return 0;
}
