LL dp[110];

int main()
{
    int n; Ri(n); int i;
    for(i = 1; i <= n; i++)
        Ri(a[i]);
    for(i = 1; i <= n; i++)
        if(a[i]) break;
    int cnt = 0; dp[i] = 1LL;
    i++;
    for(; i <= n; i++)
    {
        if(a[i] == 0)
        {
            cnt++;
            dp[i] = dp[i-1];
        }
        else
            dp[i] = dp[i-1] * (cnt+1), cnt = 0;
    }
    Pl(dp[n]);

}
