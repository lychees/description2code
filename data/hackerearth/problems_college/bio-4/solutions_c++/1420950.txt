#include <iostream>
#include <fstream>
#include <vector>
#include <memory.h>
#include <string>
#include <cstring>
#include <algorithm>
#include <map>
#include <assert.h>
#include <queue>
#include <stack>
using namespace std;
typedef long long ll;
const int mod = 1e9 + 7;
const int N = 1005;
string s;
string t;
int main(){
    int T;
    cin >> T;
    while(T --> 0){
        cin >> s >> t;
        int n = (int)s.size();
        int m = (int)t.size();
        int ans = 1e9;
        for(int i = 0; i + m - 1 < n; ++i){
            int cur_ans = 0;
            for(int j = 0; j < m; ++j){
                if(s[i + j] != t[j])
                    cur_ans++;
            }
            ans = min(ans, cur_ans);
        }
        cout << ans << "\n";
    }
    return 0;
}
