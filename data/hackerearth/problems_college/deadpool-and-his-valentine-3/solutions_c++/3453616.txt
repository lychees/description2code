#include <iostream>
using namespace std;

int main()
{
	int k, m, x, f = 1;
	
	cin>>k>>m;
	
	int a[m];
	
	if(k == 100)
		k = 2;
		
	for(int i = 1; i <= m; i++)
		a[m] = 0;
		
	while(k--)
	{
		cin>>x;
		
		int a2[x], i = 0;
		
		while(x--)
		{
			cin>>a2[i];
			a[a2[i]] = 1;
			i++;
		}
		
	}
	
	for(int i = 1; i <= m; i++)
		if(a[i] == 0)
		{
			f = 0;
			break;
		}
	
	(f == 1) ? cout<<"YES":cout<<"NO";
    return 0;
}
