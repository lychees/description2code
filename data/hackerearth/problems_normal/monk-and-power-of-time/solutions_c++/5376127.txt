#include <iostream>
#include <cstdlib>
#include <queue>
using namespace std;

int main()
{
    int N, x;
    cin >> N;
    
    queue<int> q, call;
    for (int i = 0; i < N; i++)
    	{
    		cin >> x;
    		q.push(x);
    	}
    	
    
    for (int i = 0; i < N; i++)
    	{
    		cin >> x;
    		call.push(x);
    	}
	
	int count = 0;
    while(!q.empty()){
    	if (q.front() == call.front())
    		{
    			q.pop();
    			call.pop();
    			
    		}
    	else{
    		q.push(q.front());
    		q.pop();
    		
    	}
    	
    	count++;
    }	
    cout << count << endl;
    return 0;
}
