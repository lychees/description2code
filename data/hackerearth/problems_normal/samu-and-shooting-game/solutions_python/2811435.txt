def solve():
	dp = [(w+1)*[0] for _ in xrange(n+1)]
	for i in xrange(n+1):
		dp[i][0] = 1
	for i in xrange(1,n+1):
		for j in xrange(1,w+1):
			dp[i][j] = max(dp[i-1][max(j-x, 0)]*p1 + dp[i-1][j]*(1-p1), dp[i-1][max(j-y, 0)]*p2 + dp[i-1][j]*(1-p2))
	return dp[n][w]
	

for _ in xrange(int(raw_input())):
	x, y, n, w, p1, p2 = map(int, raw_input().split())
	p1 /= 100.0
	p2 /= 100.0
	global x,y,n,w,p1,p2
	print '%.6f' % (100*solve())
