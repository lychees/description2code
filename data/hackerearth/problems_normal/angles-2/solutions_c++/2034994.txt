#include <bits/stdc++.h>
using namespace std;
int getgcd(int a, int b)
{
	if(b==0)
		return a;
	else
		return getgcd(b,a%b);
}
int main()
{
    int n,k;
    cin>>n>>k;
    int i,j;
    int a[10];
    for(i=0;i<n;i++)
    {
    	cin>>a[i];	
    }
    int gcd[10];
    for(i=0;i<n;i++)
    {
    	gcd[i]=getgcd(a[i],360);
    }
   	while(k--)
   	{
   		int flag=0;
   		int b;
   		cin>>b;
   		for(j=0;j<n;j++)
	    {
	    	if(b%gcd[j]==0)
	    	{
	    		flag=1;
	    		break;
	    	}
	    }
	    if(flag==1)
	    cout<<"YES"<<endl;
	    else
	    cout<<"NO"<<endl;
   	}
    
}
