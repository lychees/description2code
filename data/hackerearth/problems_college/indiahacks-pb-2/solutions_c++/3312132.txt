#include <bits/stdc++.h>

using namespace std;

#define ll long long
#define S(x) scanf("%d", &x)
#define Sl(x) scanf("%lld", &x)
#define mem(a,x) memset(a,x,sizeof(a))

int main()
{
    int n,ans;
    string s,tmp;
    char ch;

    map <string, int> m;

    cin >> n;

    while(n--){
        cin >> s;

        ++m[s];

        tmp = s;

        reverse(tmp.begin(), tmp.end());

        ++m[tmp];

        if(m[tmp] == 2){
            ans = tmp.size();
            ch = tmp[ans/2];
        }
    }
    cout << ans << " " << ch;
    return 0;
}
