#include <iostream>
#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    ll t,i,j,k,l,c;
    cin>>t;
    while(t--){
    	ll n,m;
    	cin>>n>>m>>c;
    	cout<<n/m;
    	if(c>0)
    	cout<<".";
    	n=n%m;
    	while(c){
    		cout<<(n*10)/m;
    		n=(n*10)%m;
    		c--;
    	}
    	cout<<endl;
    	
    }
    return 0;
}
