#include <iostream>
#include <set>
#include <map>
#include <string>
#include <cmath>
#include <utility>
using namespace std;


map <string,bool> win;

map <string,bool> loss;


int main()
{
    int N,no_of_drivers;	
    scanf("%d",&N);
    
    no_of_drivers = pow(2,N) -1;
    
    for(int i=0;i<no_of_drivers;i++)
    {
    	char s1[20],s2[20];
    	
    	scanf("%s %s",s1,s2);
    	
    	string a(s1);
    	string b(s2);
    	
    	if(!loss[a])
    		win[a]=true;
    	
    	loss[b] = true;
    	win.erase(b);
		
		
    }
    
    
    for (map <string,bool>::iterator it=win.begin(); it!=win.end(); ++it)
    {
    	if(it->second)
    	{
    		cout <<it->first;
    		break;
    	}
    }	
    	
    
    
}
