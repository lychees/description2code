#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		vector<long long int> x;
		long int n,i,ip;
		cin>>n;
		for(i=0;i<n;++i)
		{
			cin>>ip;
			x.push_back(ip);
		}
		sort(x.begin(),x.end());
		for(i=0;i<n;++i)
		cout<<x[i]<<" ";
		
		cout<<"\n";
		x.clear();
		
	}
    
    return 0;
}
