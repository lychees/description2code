#include <iostream>
using namespace std;

int main()
{
     int total, length;
     long array[100000], index;
     long long sum, even, odd;
     cin >> total;
     while(total--) {
     	cin >> length;
     	sum  = 0;
     	odd = even = 0;
     	for (index=0;index < length; index++) {
     		cin >> array[index];
     		if (array[index]%2) {
     			even++;
     		} else {
     			odd++;
     		}
     	}
     	sum = even*odd;
     	cout << sum << "\n";
     }
}
