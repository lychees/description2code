#include<stdio.h>
#include<string.h>
int main(){
    int t;
    long long int ans,tmp;
    char s[10000];
    scanf("%d\n",&t);
    while(t--)
    {
        gets(s);
        ans=s[strlen(s)-1]-'0';
        for(int i=strlen(s)-3;i>=0;i-=2)
        {
            tmp=s[i]-'0';
            if(s[i+1]=='+')ans+=tmp;
            else if(s[i+1]=='-')ans-=tmp;
            else if(s[i+1]=='*')ans*=tmp;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
