while (1):
    try:
        n = raw_input()
        l = len(n)
        b = 0
        c = 0
        for i in range(0, l-1):
            if (n[i] == ' '):
                b = int(n[0:i])
                c = int(n[i+1:l])
                break

        a = b*c
        if (a != 0):
            print a
    except EOFError:
        break
