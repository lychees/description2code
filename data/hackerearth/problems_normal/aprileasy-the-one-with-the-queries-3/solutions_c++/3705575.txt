#include <bits/stdc++.h>

using namespace std;

int main()
{
	int i,j,k,l=0,m,n,t,q;
	
	string s;
	cin >> s;
	vector<int> v[26];
	vector<int>::iterator it;
	for(i=0;i<s.length();i++)
		v[s[i] - 'a'].push_back(i);
		
	cin >> q;
	while (q--)
    {
    	string str;
        cin >> str;
        bool possible = true;

        int last = -1;

        for (i = 0; i < str.size(); i++)
        {
            it = upper_bound(v[str[i] - 'a'].begin(), v[str[i] - 'a'].end(), last);
            
           // cout << it - v[str[i]-'a'].begin()<<endl;

            if (it == v[str[i] - 'a'].end())
            {
                possible = false;
                break;
            }
            last = *it;
            //cout<<last<<endl;
        }

        if (possible)
            cout << "Yes\n";
        else
            cout << "No\n";
}}
