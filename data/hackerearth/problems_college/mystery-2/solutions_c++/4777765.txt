#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl
    int t;
    cin>>t;
    for(;t--;)
    {
    	int a,b,i=0,j,arr[999];
    	cin>>a>>b;
    	while(a!=0)
    	{
    		arr[i]=a%b;
    		i++;
    		a=a/b;
    	}
    	for(j=i-1;j>=0;j--)
    	cout<<arr[j];
    	
    	cout<<("\n");
    }
    return 0;
}
