#include <iostream>
using namespace std;

int max(int a,int b)
{
		if(a>b)
		return a;


		else
		return b;
}

int hcf(int a,int b)
{
	if(b==0)
	return a;
	
	else if(a==0)
	return b;

	else
	return hcf(b,(a%b));
}

int main()
{
    long t,i;
    int a,b;
    long long n;

    cin>>t;

    for(i=0;i<t;i++)
    {
    	long long l,h,mid;
    	long long values,comm_values;

		cin>>a>>b>>n;

		l=1;
		h=max(a,b)*n;

		while(l<=h)
		{
			mid=(l+h)/2;

			values=(mid/a)+(mid/b);
			comm_values=mid*hcf(a,b)/(a*b);
			values=values-comm_values;

			if(values==n)
			break;

			else if(values>n)
			h=mid-1;

			else
			l=mid+1;
		}
		cout<<max((mid/a)*a,(mid/b)*b)<<"\n";
    }
    return 0;
}
