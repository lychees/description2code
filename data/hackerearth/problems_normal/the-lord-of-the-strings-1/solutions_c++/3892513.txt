#include <iostream>
#include<string>
struct node {
	struct node *link;
	char value;
}*head,*temp;

void create();
int trace(char);
void push(char);
using namespace std;

int main()
{	
	create();
    int t;
    cin>>t;
    
    for(int i=1;i<=t+1;i++) {
    	create();
    	string s;
    	cin>>s;
    	for(int j=0;j<s.length();j++) {
    		if(trace(s[j]) == 1) {
    			continue;
    		}
    		else
    			cout<<s[j];
    	}
    	cout<<endl;
    }
}

void create() {
	head = NULL;
};

int trace(char a) {
	if(head==NULL) {
		push(a);
	}
	else {
		temp = head;
		while(temp != NULL) {
			if(temp->value == a) {
				return 1;
			}
			else {
				temp = temp->link;
			}
		}
		push(a);
	}
};

void push(char a) {
	if(head==NULL) {
		head = new node;
		head->link=NULL;
		head->value=a;
	}
	else {
		temp = new node;
		temp->link=head;
		temp->value=a;
		head=temp;
	}
};
