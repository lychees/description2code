#include <bits/stdc++.h>
using namespace std;

//ifstream in("test.txt");

class circle
{
    int x, y, r;
public:
    void read_circle() { cin>>x>>y>>r; }
    bool point_in_circle(int, int);
};

bool circle::point_in_circle(int a, int b)
{
    double check = pow(a-x, 2) + pow(y-b, 2);
    if(check <= r*r)
        return true;
    return false;
}

int main()
{
    int n;
    cin>>n;
    circle circles[n];
    for(int q=0; q<n; q++)
        circles[q].read_circle();
    int overall = 0;
    for(int q=1; q<=1000; q++)
        for(int w=1; w<=1000; w++){
            int counter = 0;
            for(int e=0; e<n; e++)
                if(circles[e].point_in_circle(q, w)){
                    counter++;
                    if(counter>1){
                        overall++;
                        break;
                    }
                }
        }
    cout<<overall<<endl;
}
