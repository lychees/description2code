#include <iostream>
using namespace std;
char hex[16];
void tohex(int a)
{
	char c,b;
	b=::hex[a%16];
	a=a/16;
	c=::hex[a];
	cout<<c<<b;
}

int main()
{
    int t;
    unsigned int r,g,b;
    for(int i=0;i<16;i++)
    {
    	if(i<10)
    	::hex[i]=i+'0';
    	switch(i){
    	case 15:
    	::hex[i]='Z';
    	break;
    	case 14:
    	::hex[i]='Y';
    	break;
    	case 13:
    	::hex[i]='X';
    	break;
    	case 12:
    	::hex[i]='W';
    	break;
    	case 11:
    	::hex[i]='V';
    	break;
    	case 10:
    	::hex[i]='U';
    	break;
    	}
    	      
    }
    cin>>t;
    int i=1;
    while(t--)
    {
    	cin>>r>>g>>b;
    	printf("Case %d: ",i++);
        tohex(r);
        cout<<" ";
        tohex(g);
        cout<<" ";
        tohex(b);
        cout<<endl;
    }
    return 0;
}
