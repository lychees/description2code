using namespace std;
int a[1000001];
int main(){
	a[1] = 1;
	for (int j = 2; j <= 1000000; j++){
		if (!a[j]){
			for (int k = j * 2; k <= 1000000; k += j){
				a[k] = 1;
			}
		}
	}
	a[4] = 0;
	for (int i = 1; i <= 1000000; i++){
		a[i] += a[i - 1];
	}
	int q;
	scanf("%d", &q);
	while (q--){
		int r, l;
		scanf("%d%d", &r, &l);
		printf("%d\n", a[l] - a[r - 1]);
	}
	return 0;
}
