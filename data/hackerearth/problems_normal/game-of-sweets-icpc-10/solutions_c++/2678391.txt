using namespace std;
typedef long long LL;
typedef vector<int> VI;
typedef vector< vector<int> > VVI;
typedef pair< LL, LL > PII;

const int N = 100010;
const int M = 10010;
bitset<N> mask,ans;
int a[N];
inline void Set(int i)
{
	mask[i]=1;
}
inline void shift(int l)
{
	mask = mask>>l;
}
inline void Or()
{
	ans|=mask;
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int n; cin>>n;
	for(int i=0; i<n; i++) cin>>a[i], Set(a[i]);
	sort(a,a+n);
	int cnt=0;
	for(int i=0; i<n-1; i++)
	{
		shift(a[i] - cnt);
		Or();
		cnt = a[i];
	}
	LL ret = -1;
	ret = ans.count();
	ret--;
	for(int i=0; i<n-1; i++) if(a[i]==a[i+1]) {ret++; break;}
	if(ret<0) ret = 0;
	cout<<ret<<endl;
	return 0;
}
