const int MAX_N = 1e5 + 10;

int T, N, K;
int P;
int A[MAX_N];
int main(){
  scanf("%d", &T);
  while(T--){
    scanf("%d%d%d", &N, &K, &P);
    for(int i = 0; i < N; i++){
      scanf("%d", A + i);
    }
    sort(A, A + N);
    int i = 0, j = 0, bad = 0;
    while(i < N){
      while(j < N && A[j] <= 1LL * A[i] + 2 * P)
	j ++;
      if(j - i >= K)
	bad = true;
      if(j == N)
	break;
      while(i < j && 1LL * A[i] + 2 * P < A[j])
	i ++;
    }
    puts(bad ? "NO" : "YES");
  }
  return 0;
}
