#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int n;
	cin>>n;
	int a[1010]={0};
	for(int i=0;i<n;i++){
		int x;
		cin>>x;
		a[x]++;
	}
	cout<<n<<endl;
	for(int i=0;i<=1000;i++){
		if(a[i]>0){
			n=n-a[i];
			if(n!=0)
			cout<<n<<endl;
		}
	}
	return 0;
}
