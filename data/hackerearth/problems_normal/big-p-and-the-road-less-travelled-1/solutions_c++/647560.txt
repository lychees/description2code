#include <bits/stdc++.h>
using namespace std;
vector<int> x[100005];
long long dp[100005];
int pre(int p)
{
	int ans=0;
	if(p==1)	ans = 1;
	else if(dp[p]==-1)
	{
		for(int i=0;i<x[p].size();i++)
			ans+=pre(x[p][i]);
		dp[p]=ans;
	}
	else
		ans=dp[p];
	return ans;
}
int main()
{
	int n;
	scanf("%d",&n);
	int a=1,b=1;
	for(int i=0;i<100005;i++)
		dp[i]=-1;
	while(1)
	{
		scanf("%d%d",&a,&b);
		if(a==0 &&b==0)
			break;
		x[b].push_back(a);
	}
	long long ans=pre(n);
	printf("%lld\n",ans);
    return 0;
}
