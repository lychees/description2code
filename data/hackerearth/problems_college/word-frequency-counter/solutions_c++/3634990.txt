#include<bits/stdc++.h>
using namespace std;

int main()
 {
     string s;

     vector < pair<string,int> > words;

     while (cin>>s)
     {
          //cin>>s;
          int x = s.length();

           int found = 0;

            for(int i = 0; i<x; i++)
              s[i] = toupper(s[i]);

                for(int i = 0; i<words.size(); i++)
                   {
                       if(words[i].first == s)
                       {
                           words[i].second++;
                           found = 1;
                             break;
                       }
                   }

                if(!found)
                     words.push_back(make_pair(s,1));

     }

     for(int i = 0; i<words.size(); i++)
        cout<<words[i].first<<" "<<words[i].second<<endl;



 return 0;
 }
