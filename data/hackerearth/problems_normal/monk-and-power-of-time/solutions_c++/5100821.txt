#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	cin>>n;
	queue<int> calling,ideal;
	int time=0;
	
	for(int i=0;i<n;i++)
	{
		int a;
		cin>>a;
		calling.push(a);
	}
	
	for(int i=0;i<n;i++)
	{
		int a;
		cin>>a;
		ideal.push(a);
	}
	
	while(!calling.empty())
	{
		int temp=calling.front();
		if(temp==ideal.front())
		{
			calling.pop();
			ideal.pop();
			time++;
		}
		else
		{
			calling.pop();
			calling.push(temp);
			time++;
		}
		//cout<<*it1<<" "<<*it<<endl;
	}
	cout<<time<<endl;
	
}
