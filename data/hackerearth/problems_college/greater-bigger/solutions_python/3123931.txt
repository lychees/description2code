for _ in xrange(input()):
	a=list(raw_input())
	b=[]
	flag=0
	for x in xrange(len(a)-1,0,-1):
		if a[x]>a[x-1]:
			b=a[:x-1]
			for y in xrange(len(a)-1,x-1,-1):
				if a[x-1]<a[y]:
					b.append(a[y])
					c=a[x-1:]
					del(c[y-x+1])
					c.sort()
					b=b+c
					flag=1
				if flag==1:
					break
		if flag==1:
			break
	if flag==0:
		print "no answer"
	else:
		print ''.join(x for x in b)
