#include<bits/stdc++.h>
using namespace std;
int getMinimum(string s, int aCost, int bCost)
{

    int i=0;
    int j=s.size()-1;
    int count = 0;

    for(int i=0; i<s.size(); i++, j--)
    {
        if(s[i]=='/')
        {
            if(s[i]==s[j])
            {
                count+=2*min(aCost, bCost);
                s[i]=s[j] = 'a';
            }
            else
            {
                count += s[j]=='a' ? aCost : bCost;
                s[i] = s[j];
            }
        }
    }
    if(s!=string(s.rbegin(), s.rend()))
        return -1;

    return count;

}
int main()
{
    long int t,a,b;
    string s;
    cin>>t;
    while(t--)
    {
        cin>>s>>a>>b;
        cout<<getMinimum(s,a,b)<<"\n";
    }
    return 0;
}
