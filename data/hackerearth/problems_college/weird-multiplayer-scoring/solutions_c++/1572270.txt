#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

map<string,ll> mp;
map<string,ll> :: iterator it;

int main() {
  int n,m;
  scanf("%d%d",&n,&m);
  int maxVal=-1e9;
  string res;
  for(int i=1;i<=m;i++) {
    string str;
    cin>>str;
    int points;
    scanf("%d",&points);
    if(mp.find(str)==mp.end()) {
      mp[str]=points;
    } else {
      mp[str]+=points;
    }
    for(it=mp.begin();it!=mp.end();it++) {
      if(it->second>maxVal) {
        res=it->first;
        maxVal=it->second;
      }
    }
  }
  //cout<<maxVal<<endl;
  cout<<res<<endl;
  return 0;
}
