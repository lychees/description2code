#include <iostream>
using namespace std;

typedef long long ll;

ll A,B,C;


ll gcd(ll a,ll b)
{
	if(b==0)
	{
		return a;
	}
	return gcd(b,a%b);
}

int main()
{
    cin >> A >> B >>C;
    if(C>=A+B)
    {
    	cout<<"1/1"<<endl;
    }
    else if(C<=A && C<=B)
    {
    	ll ans = C*C;
    	ll ans1=2*A*B;
    	
    	ll ans2=gcd(ans,ans1);
    	
    	ans/=ans2;
    	ans1/=ans2;
    	
    	cout<<ans<<"/"<<ans1<<endl;
    }
    else if(C>=B && C<=A)
    {
    	ll ans = C*C;
    	ans-=((C-B)*(C-B));
    	
    	ll ans1=2*A*B;
    	
    	ll ans2=gcd(ans,ans1);
    	
    	ans/=ans2;
    	ans1/=ans2;
    	
    	cout<<ans<<"/"<<ans1<<endl;
    }
    else if(C>=A && C<=B)
    {
    	ll ans = C*C;
    	ans-=((C-A)*(C-A));
    	
    	ll ans1=2*A*B;
    	
    	ll ans2=gcd(ans,ans1);
    	
    	ans/=ans2;
    	ans1/=ans2;
    	
    	cout<<ans<<"/"<<ans1<<endl;
    }
    else if(C>=B && C>=A)
    {
    	ll ans = A*B;
    	ans-=((A+B)-C)*((A+B)-C);
    		
    	ll ans1=2*A*B;
    	
       // cout<<ans<<" "<<ans1<<endl;
    	ll ans2=gcd(ans,ans1);
    	
    	ans/=ans2;
    	ans1/=ans2;
    	
    	cout<<ans+A*B<<"/"<<ans1<<endl;
    }
   
    
    return 0;
}
