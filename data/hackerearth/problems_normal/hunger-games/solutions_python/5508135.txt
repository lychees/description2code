	n = int(raw_input().strip())
	line = raw_input().split()
	arr = map(int, line)
	stk1 = []
	stk2 = []
	arr.sort()
	i = 0
	while i < n:
		stk1.append(arr[i])
		if i+1 < n:
			stk2.append(arr[i+1])
		i += 2
	while len(stk2) != 0:
		stk1.append(stk2.pop())
	i = 0
	cost = 0
	while i != n:
		if i+1 < n:
			temp = abs(stk1[i+1] - stk1[i])
			cost = max(cost,temp)
		i += 1
	temp = stk1[-1]-stk1[0]
	cost = max(cost, temp)
