#include <iostream>
using namespace std;
int getsum(long long int a)
{
	a=a*a;
	int ans=0;
	while (a)
	{
		ans+=a%10;
		a/=10;
	}
	return ans;
}
bool init(long long int x)
{
	if (x==1 or x==4)
		return true;
	else
	{
		int curr[200];
		int k=0,i;
		long long int y;
		curr[k++]=x;
		y=getsum(x);
		while (true)
		{
			curr[k++]=y;
			if (y==1 or y==4)
			{
				return true;
				break;
			}
			y=getsum(y);
			for (i=0;i<k;i++) if (curr[i]==y) break;
			if (i!=k) break;
		}
		return false;
	}
}				
	inline void readint(int &a)
{
    register int c;
    a = 0;
    do c = getchar_unlocked(); while(c < '0');
    do{
    a = (a << 1) + (a << 3) + c - '0';
    c = getchar_unlocked();
    }while(c >= '0');
}
