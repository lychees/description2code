#include<iostream>
#include<math.h>
int sum(int a)
{
	int p,s=0;
    while(a>0)
	{
		p=a%10;
		s+=pow(p,2);
		a=a/10;
	}
		if(s>9)
		{
			a=s;
	     	sum(a);
		}
		else
		if(s==1)
		return 1;
}
using namespace std;
int main()
{
	int t,n,a;
	cin>>t;
	for(int i=1;i<=t;i++)
	{
		cin>>a;
		n=sum(a);
		if(n==1)
		cout<<"Case #"<<i<<":YES"<<endl;
		else
		cout<<"Case #"<<i<<":NO"<<endl;
	}
	return 0;
}
