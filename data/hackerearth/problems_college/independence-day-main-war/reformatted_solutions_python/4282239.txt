t = int(raw_input().strip())

for i in range(0, t):

    no = int(raw_input().strip())

    points = []

    for j in range(0, no):

        points.append(map(float, raw_input().strip().split(" ")))

    x = [l[0] for l in points]

    y = [k[1] for k in points]
    x = sum(x)/len(points)
    y = sum(y)/len(points)
    print "(%.2f , %.2f)" % (round(x, 2), round(y, 2))
