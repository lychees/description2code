import math


def logs(l):
    mins = l[0]
    for i in range(1, len(l)):
        if l[i] is 1:
            continue
        ans = math.log(1.0*l[i]/mins, 2)
        if int(ans) != ans:
            return False

    return True


num = int(raw_input())
l = raw_input().split()
for i in range(num):
    l[i] = int(l[i])

l.sort()
if logs(l):
    print 'YES'
else:
    print 'NO'
