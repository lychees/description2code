#include <iostream>
#include <cstdio>

#define ll long long int

using namespace std;

ll max(ll a,ll b)
{
	return ((a>b)?a:b);
}

int main()
{
	ll i,t,j,c,n;
	ll a[105][105];
	for(i=0;i<105;i++) {
		a[i][0]=0;
		a[0][i]=0;
	}
	cin >> t;
	while(t--) {
		cin >> n >> c;
		for(i=1;i<=n;i++) {
			for(j=1;j<=n;j++) {
				cin >> a[i][j];
				a[i][j]=max(a[i-1][j],a[i][j-1])+a[i][j];
			}
		}
		if(a[n][n]<c) cout << "-1\n";
		else cout << a[n][n]-c << endl;
	}
    return 0;
}
