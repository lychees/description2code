#include<stdio.h>
#include<bits/stdc++.h>
#include<iostream>
#include<math.h>
#include<stdlib.h>
using namespace std;
long long int factors[1000001],c[1000001]={0},coun[10000001]={0};

void mark(bool arr[], int a, int n)
{
    int i = 2, num;
    while ( (num = i*a) <= n )
    {
        arr[ num-1 ] = 1;
        ++i;
    }
}

int fact(long long int n)
{
    long long int j=0;
    if (n >= 2)
    {
        bool arr[n];
        memset(arr, 0, sizeof(arr));
        for (long long int i=1; i<n; ++i)
        {
            if ( arr[i] == 0 )
            {
                factors[j]=i+1;
                j++;
                mark(arr, i+1, n);
            }
        }
    }
    return j;
}

int main()
{
	long long int t,n,ans,i,j,num,k,k1;
	cin>>n;
	num=fact(n);
	for(i=0;i<num;i++)
    {
        j=1;
        k1=0;
        ans=factors[i];
        while(k1<=n)
        {
            k1=j*ans;
            c[k1]++;
            j++;
        }
    }
    for(i=2;i<=n;i++)
    {
        coun[c[i]]++;
    }
	cin>>t;
	while(t--)
	{
		cin>>k;
		if(k>n)
            cout<<"0"<<endl;
        else
            cout<<coun[k]<<endl;
	}
}
