#include <iostream>
#include <vector>
#include <algorithm>
#define DIVI 1000000007

using namespace std;

typedef long long int Lint;

int main()
{
	ios_base::sync_with_stdio( false );
	Lint t, n, i, sum1, sum2, sum3, sum4, j;
	
	cin >> t;
	
	while( t-- ) {
		
		cin >> n;
		vector<Lint> A( n );
		
		for( i = 0;i < n;i++ ) {
			cin >> A[ i ];
		}
		
		sort( A.rbegin(), A.rend() );
		
		sum1 =0;
		sum2 = 0;
		
		for( i = 0;i < n / 2;i++ )
			sum1 = ( sum1 + A[ i ] + DIVI ) % DIVI ;
		for( j = i;j < n;j++ )
			sum2 = ( sum2 + A[ j ] + DIVI ) % DIVI;
			
		sum2 = ( ( sum2 * -1 ) + DIVI ) % DIVI;
		sum3 = ( ( sum1 % DIVI ) + ( sum2 % DIVI ) ) % DIVI;

		sum4 = 0;
		for( i = 0;i < n;i++ ) {

			if( A[ i ] < 0 )
				sum4 = ( sum4 + ( -1 * A[ i ] ) ) % DIVI;
			else
				sum4 = ( sum4 + A[ i ] ) % DIVI;
		}

		cout << ( sum3 * sum4 ) % DIVI<< "\n";
	}
	
	return 0;
}
