#include <iostream>
using namespace std;

int main()
{
    int t,n,a[11],m[11],val;
    cin>>t;
    while(t--){
    	val=0;
    	cin>>n;
    	for(int i=0;i<n;i++)cin>>a[i];
    	for(int i=0;i<n;i++)cin>>m[i];
    	for(int i=0;i<n;i++){
    		val=val^(m[i]-1)^(a[i]-m[i]);
    	}
    	if(!val)cout<<"Second\n";
    	else cout<<"First\n";
    }
    return 0;
}
