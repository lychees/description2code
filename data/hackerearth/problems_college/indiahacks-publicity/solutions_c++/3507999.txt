#include <bits/stdc++.h>
#define ll long long int

using namespace std;

vector<int> G[100005];
bool mark[100005];

void dfs(int j) {
    mark[j] = true;
    for (int i = 0; i < G[j].size(); i++) {
        if (!mark[G[j][i]]) {
            dfs(G[j][i]);
        }
    }
}

int main()
{
    int t, n, e, a, b;
    cin >> t;

    while (t--) {
        cin >> n >> e;
        for (int i = 0; i <= n; i++) {
            G[i].clear();
        }
        for (int i = 0; i < e; i++) {
            cin >> a >> b;
            G[a].push_back(b);
            G[b].push_back(a);
        }

        memset(mark, false, sizeof(mark));

        int cnt = 0;
        for (int i = 0; i < n; i++) {
            if (!mark[i]) {
                dfs(i);
                cnt++;
            }
        }
        cout << cnt << endl;
    }

    return 0;
}
