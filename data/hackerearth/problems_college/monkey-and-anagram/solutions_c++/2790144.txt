#include<bits/stdc++.h>
using namespace std;

int main()
{
	
	int e;
	cin>>e;
	string s,t;
	while(e--)
	{
		
		int first[200]={0};
		int second[200]={0};
		cin>>s>>t;
		int n1=s.length();
		int n2=t.length();
		
		if(n1>n2)
		{cout<<"-1";}
	else
	{
		for(int i=0;i<n1;i++)
		{
			first[int(s[i])]++;
		}
			for(int i=0;i<n2;i++)
		{
			second[int(t[i])]++;
		}
		int count=0;
		for(int i=97;i<123;i++)
		{
		if(second[i]>first[i])
		++count;
		else
		if(first[i]>second[i])
		{count=-1;break;}
		}
		cout<<count<<"\n";
	}
		
		
	}
	return 0;
}
