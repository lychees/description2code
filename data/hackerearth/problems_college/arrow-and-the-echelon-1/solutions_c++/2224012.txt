#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int x,c=0,i,flag=0;

    	{
    		x=x-3;

    	cout<<"-1";

    	}

    }
    	
    return 0;
}
