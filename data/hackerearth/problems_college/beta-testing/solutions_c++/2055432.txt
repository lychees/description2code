#include<iostream>
#include<string>
#include<map>
#include<iterator>
using namespace std;

struct winner
{
	string name;
	int count;
};

int main()
{
	int t,cnt=0,size;
	cin >> t;
	string s;
	map<string,int>::iterator it;
	map <string, int> mymap;
	for (int i=0; i<t; i++)
	{
		cin >> s;
		it = mymap.find(s);
		if (it != mymap.end())
			mymap[s]++;
		else
		{
			mymap[s] = 1;
			cnt++;
		}
	}
	size = cnt;
	winner w[cnt];
	for (std::map<string,int>::iterator it=mymap.begin(); it!=mymap.end(); ++it)
    	{
		cnt--;
		w[cnt].name = it->first;
		w[cnt].count = it->second;
	}
	for (int i=0; i<size; i++)
		cout << w[i].name << " " << w[i].count << endl;
	
}
