#include <bits/stdc++.h>
using namespace std;
int main()
{
  int t,q,n;
    string str;
    cin>>str;
    map<char,int> map;
   
  for(int j=0;j<str.length();j++)
    {
        map[str.at(j)]++;
    }

  int max=INT_MIN;
  for (std::map<char,int>::iterator it=map.begin(); it!=map.end(); ++it)
    {
        
        if(it->second>max)
        {
          max=it->second;
        }     

    }
    char minchar;
    int min=INT_MAX;
    for (std::map<char,int>::iterator it=map.begin(); it!=map.end(); ++it)
    {
        if(it->second==max && it->first-'a'<min)
        {
          minchar=it->first;
          min=it->first-'a';
        }     
    }

    if(minchar=='u')
    {
      printf("Thanks Both\n");
    }

    else if(minchar=='h'||minchar=='e'||minchar=='a'||minchar=='d'||minchar=='r'||minchar=='s'||minchar=='h')
    {
      printf("Thanks head rush\n");
    }
    else if(minchar=='c'||minchar=='l'||minchar=='b'||minchar=='i'||minchar=='n'||minchar=='g')
    {
      printf("Thanks cubing club\n");
    }
    else
    {
      printf("none\n");
    }


}
