#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	long long r,s,r_sum=0,s_sum=0;
    	cin>>r>>s>>n;
    	vector<int>cal;
    	for(int i=0;i<n;i++)
    {
    	int inp;
    	cin>>inp;
    	cal.push_back(inp);
    }
    sort(cal.begin(),cal.end());
    int i=0;
    while(i<n)
    {
    	if(r_sum>r || s_sum>s)
    	break;
    	else
    	{r_sum+=cal[i];
    		s_sum+=cal[i];
    		i++;
    	}
    }
    if(r_sum>r && s_sum<s)cout<<"Sayan Won"<<endl;
    else if(s_sum>s && r_sum<r)cout<<"Raghu Won"<<endl;
    else cout<<"Tie"<<endl;
    }
    return 0;
}
