void createHash(){

    hash2.pb(0);
    for(long i=0;i<s.size();i++){

      h2=add(mul(power2[i+1],s[i]-'a',MOD),hash2[i],MOD);

      hash1.pb(h1);
      hash2.pb(h2);
    }
}
