    #include<bits/stdc++.h>
    using namespace std;
    int main()
    {
    	string str;
    	cin>>str;
    	int n=str.length();
    	int dp[n+10];
    	memset(dp,0,sizeof(dp));
    	if(str[1]==str[0])
    	dp[1]=1;
    	for(int i=2;i<=n;i++)
    	{
    		if(str[i-1]==str[i-2])
    		dp[i]=dp[i-1]+1;
    		else
    		dp[i]=dp[i-1];
    	}
    	int m;
    	cin>>m;
    	while(m--)
    	{
    		int a,b;
    		cin>>a>>b;
    		cout<<dp[b]-dp[a]<<endl;
    	}
    }
