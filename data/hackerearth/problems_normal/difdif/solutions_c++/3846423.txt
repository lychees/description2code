#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(0);
	
	set<string>set;
	
	string s,c;
	long int l,x,y;
	
	cin>>s;
	l=s.size();
	
	for(x=0;x<l;x++)
	{
		
		for(y=x;y<l;y++)
		{
		c=c+s[y];
		set.insert(c);
		}
		c="";
		
		
	}
	cout<<set.size();
	return 0;
		
		
		
		
	}
