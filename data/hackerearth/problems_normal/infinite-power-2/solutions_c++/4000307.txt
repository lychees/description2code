#include <iostream>
#include<stdio.h>
using namespace std;

long long gcd(long long a, long long b){
   if(b==0)
    return a;
else
    return gcd(b,a%b);
}

int main()
{
    long long t;
    scanf("%lld",&t);
    while(t--){
    	long long num;
    	 scanf("%lld",&num);

    	long long x=num;
    	bool flag=false;
    	while(x!=0){
    		if(x%2!=0){
    			flag=true;
    			break;
    		}
    		x/=10;
    	}

    	if(!flag)
    		cout<<"Unlimited Power\n";
    	else{
    		//cout<<"FGFgf";
    		long long arr[11];
    		long long mn=0,mx=0;
    		long long x=num,ind=0;
    		while(x!=0){
    			arr[ind]=x%10;
    			x/=10;
    			ind++;
    		}
    		//cout<<ind<<"\n";
    		//cout<<"fggfh";
    		long long mxx[11],mnn[11];
    		for(long long i=ind-1;i>=0;i--){
    			if(arr[i]%2==1){
    			   mxx[i]=arr[i]+1;
    			   long long k=i;
    			   while(mxx[k]==10){
                    if(k==ind-1){
                    mxx[k]=20;
                    break;
    			   }
                   
    			   	mxx[k]=0;
    			   	mxx[k+1]+=2;
    			   	k++;
    			   }
    			   for(long long j=i-1;j>=0;j--)
    			       mxx[j]=0;
    			   break;
    			}
    			else
    				mxx[i]=arr[i];
    		}
    		//for(long long i=0;i<ind;i++)
              //  cout<<mxx[i]<<" ";
    		//cout<<"weeew";

    			for(long long i=ind-1;i>=0;i--){
    			if(arr[i]%2==1){
    			   mnn[i]=arr[i]-1;
    			   for(long long j=i-1;j>=0;j--)
    			       mnn[j]=8;
    			   break;
    			}
    			else
    				mnn[i]=arr[i];
    		}

    	//	for(long long i=0;i<ind;i++)
          //      cout<<mnn[i]<<" ";

    		for(long long i=ind-1;i>=0;i--){
    		    mn=mn*10+mnn[i];
    		    mx=mx*10+mxx[i];
    		}

    		mx=mx-num;

    		//cout<<mn<<" "<<mx<<"\n";
    		long long v=mn,b=mx;

    	}
    }
