#include <iostream>
#define LL long long 
using namespace std;


 LL int ans;

void see(int x,int y,string s){
	int n=s.size();
	while(x>=0 && y<n && s[x]==s[y]){
		ans+=(LL int)(y-x+1)*(y-x+1);
		x--;
		y++;
	}
}


int main()
{
    int t;
    cin>>t;
    
    while(t){
    	string s;
    	cin>>s;
    	int n=s.size();
    	ans=0;
    	for(int i=0;i<n;i++){
    		see(i,i,s);
    		if(i+1<n && s[i]==s[i+1]){
    			see(i,i+1,s);
    		}
    	}
    	
    	
    	cout<<ans<<endl;
    	t--;
    }
    
    return 0;
}
