#include <iostream>
#include<vector>

using namespace std;

int main()
{	int maxa=0,maxb=0,n;
	cin>>n;
	vector<int> a(n);
	for(int i=0;i<n;i++)
	{cin>>a[i];
	if(i>0)
	{int x=abs(a[i]-a[i-1]);
		if(x>maxa)
		maxa=x;
	}}
	vector<int> b(n);
	for(int i=0;i<n;i++)
	{cin>>b[i];
	if(i>0)
	{int x=abs(b[i]-b[i-1]);
		if(x>maxb)
		maxb=x;
	}}
	if(maxa>maxb)
		cout<<"Dom\n"<<maxa;
	else
		cout<<"Brian\n"<<maxb;
		
	
    return 0;
}
