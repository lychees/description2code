#include<cstdio>
#include<algorithm>
using namespace std;
int gcd(int a,int b)
{
	if(a%b==0)
	 return b;
	gcd(b,a%b); 
}
int main()
{
	int i,n,a[200000];
	scanf("%d",&n);
	int g,maxm;
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
		if(i==0)
		{
		 g=a[i];
	     maxm=a[i];
		}
		if(i)
	     g=gcd(a[i],g);
		maxm=max(a[i],maxm);
	}
	int ans=(maxm/g-n);
	if(ans&1)
	 printf("Alice\n");
	else
	 printf("Bob\n");
