#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		long long int n;
		cin >> n;
		long long int evenSum = 0, oddSum = 0, count = 0, Orig = n;
		while(n != 1)
		{
			if(n%2 == 0)
			{
				evenSum = evenSum + n;
				n=n/2;
				count++;
			}
			else
			{
				oddSum = oddSum + n;
				n = 3*n+1;
				count++;
			}
		}
		/*count--;

		oddSum = oddSum + 1;	
		/*cout << evenSum << endl;

		int oddRemainder = oddSum % count;
		if(evenRemainder > oddRemainder)
			cout << "Even Rules\n";
		else if(evenRemainder < oddRemainder)
			cout << "Odd Rules\n";
		else
			cout << "Tie\n";
	}
}
