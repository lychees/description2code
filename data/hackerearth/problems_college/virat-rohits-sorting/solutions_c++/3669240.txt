#include<bits/stdc++.h>
using namespace std;
bool compare(pair<int,int>p1,pair<int,int>p2)
{
	if(p1.first<p2.first)
	{
		return 1;
	}
	else
	if(p1.first==p2.first && p1.second>p2.second)
	{
		return 1;
	}
	return 0;
}
int main()
{
	int n;
	cin>>n;
	int i,j;
	pair<int,int>p[n];
	for(i=0;i<n;i++)
	{
		int a,b;
		cin>>a>>b;
		p[i] = make_pair(a,b);
	}
	sort(p,p+n,compare);
	for(i=0;i<n;i++)
	{
		cout<<p[i].first<<" "<<p[i].second<<"\n";
	}
}
