test = int(raw_input())

k = 3
arr = [0]*(100001)
arr[0] = 1
arr[1] = 1
arr[2] = 2
arr[3] = 4
for i in range(0, test):
    num = int(raw_input())
    if num > k:
        for j in range(k, num+1):
            arr[j] = arr[j-1]+arr[j-2]+arr[j-3]
        print arr[num] % 1000000007
    else:
        print arr[num] % 1000000007
    k = max(num, k)
