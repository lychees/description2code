#include<iostream>
#include<algorithm>
using namespace std;
#define ll long long
ll a[123456],b[123456];
int main()
{
	ll n,m,i,j,k,p,q,re=0;
	cin>>n>>m;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	for(i=0;i<n;i++)
	{
		cin>>b[i];
	}
	sort(a,a+n);
	sort(b,b+n);
	p=0,q=0;int f=0;
	while(p<n && q<n)
	{
		if(a[p]<=b[q] && m>=a[p])
		{
			re+=1;
			m-=a[p];
			p+=1;
		}
		else if(b[q]<a[p] && m>=b[q])
		{
			re+=1;
			m-=b[q];
			q+=1;
		}
	
		else if(a[p]>m && b[q]>m)
		break;
	}
	
	if(p<n && m>0)
	{
		while(p<n)
		{
			if(m>=a[p])
			{
				re+=1;
				m-=a[p];
			}
			else break;
		}
	}

		if(q<n && m>0)
	{
		while(q<n)
		{
			if(m>=b[q])
			{
				re+=1;
				m-=b[q];
			}
			else break;
		}
	}
	if(re>n)
	cout<<n;
	else cout<<re;
	return 0;
}
