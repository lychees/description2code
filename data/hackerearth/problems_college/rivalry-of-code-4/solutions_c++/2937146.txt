#include <iostream>
using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    while (t--) {
    	string d, s;
    	cin >> d >> s;
    	int n = d.size();
    	int m = s.size();
    	if (n > m) {
    		swap(d, s);
    		swap(n, m);
    	}
    	
    	int mn = n;
    	for (int i = 0; i < m - n + 1; ++i) {
    		int cnt = 0;
    		for (int j = 0; j < n; ++j) {
    			if (d[j] != s[i + j]) ++cnt;
    		}
    		if (cnt < mn) mn = cnt;
    	}
    	printf("%d\n", mn);
    }
    return 0;
}
