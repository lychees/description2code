M = 1000000007
a = [input() for _ in range(input())]
p = [1]*1024
for i in range(1, 1024):
    p[i] = p[i-1]*i % M


def d(x, y):
    return x*pow(y, M-2, M) % M


def c(n, m):
    return d(d(p[n], p[m]), p[n-m])


s, t = sum(a)-1, 1
while a:
    x = a.pop()
    t = t*c(s, x-1) % M
    s -= x
print t
