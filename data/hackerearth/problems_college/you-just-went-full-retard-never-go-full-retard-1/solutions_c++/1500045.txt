#include<iostream>
#include<math.h>
using namespace std;
bool isPrime(long x)
{
	if(x==2)
		return true;
	if(x%2==0 || x==1 || x==0)
		return false;
	for(long i=3;i<=sqrt(x);i++)
		if(x%i==0)
			return false;
	return true;
}
int main()
{
    long t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(isPrime(n))
    		cout<<"YES"<<endl;
    	else
    		cout<<"NO"<<endl;
    }
    	
}
