t = int(raw_input(''))
while t:
    ans = int(0)
    m = int(raw_input(''))
    while m:
        a, b = map(int, raw_input().split())
        m -= 1
        ans += a*b
    while ans >= 10:
        temp = ans
        s = 0
        while temp:
            s += temp % 10
            temp /= 10
        ans = s
    print ans
    t -= 1
