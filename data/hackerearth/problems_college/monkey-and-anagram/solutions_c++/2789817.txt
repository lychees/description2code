#include<bits/stdc++.h>
using namespace std;



int main()
{

	int t;
	cin>>t;
	while(t--)
	{

		string s , t;
		int c1[26]={0}, c2[26]={0};
		cin>>s>>t;
		int flag = 1, count=0;
		for(int i=0; i<s.size(); i++)c1[s[i]-'a']++;
		for(int i=0; i<t.size(); i++)c2[t[i]-'a']++;
		for(int i=0; i<26; i++)
		{

			if(c1[i]>c2[i]){flag=0; break;}
			else if(c2[i]>c1[i])
				count++;
		}
		if(flag)cout<<count<<endl;
		else cout<<"-1\n";
	}
	return 0;
}
