#include<iostream>
#include<string.h>
int maxscore,maxscoreteamlen=1000;
using namespace std;
	
void score(int count[],string b,int &maxscoreteam,int &maxscoreteamlen,int team){
	int len=b.length();
	int score=0;
	for(int i=0;i<len;i++){
		score=score+count[b[i]-'a'];
	}
//	cout<<"score for "<<b<<" is : "<<score<<endl;
	if(score>maxscore){
		maxscore=score;
		maxscoreteam=team;
		maxscoreteamlen=len;
	}
	else if(score==maxscore){
		if(maxscoreteamlen>len)
		{
//			cout<<"inn here for team:"<<b<<endl;
			maxscoreteamlen=len;
			maxscoreteam=team;
		}
		else if(maxscoreteamlen==len){
//			cout<<"inhere string is :"<<b<<endl;
			if(maxscoreteam>team)
			{
//				cout<<"HI\n";
				maxscoreteam=team;
			}
		}
	}
	
}
int main() {
	int T,N,maxscoreteam=0,teamlen;
	string s;
	cin>>T;
	while(T--){
		cin>>N>>s;
		maxscore=0;
		int count[26]={0};
		for(int i=0;i<s.length();i++){
			count[s[i]-'a']++;
		}
		string team[N];
		for(int i=0;i<N;i++)
		{
			cin>>team[i];
			score(count,team[i],maxscoreteam,maxscoreteamlen,i);
		}
		cout<<maxscoreteam+1<<endl;
	}
	return 0;
}
