#include<stdio.h>
#define max(a, b) (((a) >= (b)) ? (a) : (b))
int main(){
	int t, n, b, c1, c2, cp; scanf("%d", &t);
	while(t-- && scanf("%d%d", &n, &b)){
		cp = n, c1 = c2 = 0;
		if(!(n % b))	printf("%d\n", n);
		else{
			while(--n && n % b)	c1++;
			while(++cp && cp % b)	c2++;
			if(c1 < c2)	printf("%d\n", n);
				else if(c2 < c1)	printf("%d\n", cp);
					else	printf("%d\n", max(n, cp));
		}
	}		
	return 0;
}
