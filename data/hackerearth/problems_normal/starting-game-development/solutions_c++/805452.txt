#include <iostream>
#include <algorithm>
#include<climits>
using namespace std;

int main()
{
	
	int level[100][5000], n, m, q, val;
	
	scanf("%d %d %d", &n, &m, &q);
	
	for(int i=0; i<n; i++) {
		for(int j=0; j<m; j++) {
			scanf("%d", &level[i][j]);
		}
	}
		
	for(int i=0; i<q; i++) {
		int minm = INT_MAX;
		
		for(int j=0; j<n; j++) {
			scanf("%d", &val);
			int* f = std::upper_bound (level[j], level[j]+m, val);
			
			int a = f-level[j];
			
		//	cout<<f-level[j]<<" "<<a;
			
			minm = min(minm, a);
		}
		
		//cout<<"\n";
		
		printf("%d\n", minm);
	}
	
	return 0;
}
