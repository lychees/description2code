#include <bits/stdc++.h>
#define mp make_pair
#define pb push_back
#define ll long long
#define debug(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define fr(i,beg,end) for(i=beg;i<end;i++)
#define itfr(it,stl) for(it=stl.begin();it!=stl.end();it++)
#define PII pair<int,int>
#define init(x,val) memset(x,val,sizeof(x))
#define fst first
#define snd second
using namespace std;
set<string> st1;
set<string> st2;
set<string> ::iterator it;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    //freopen("input.txt","r",stdin);
    int i,j,t,n;
    cin>>t;
    while(t--)
    {
        string s;
        st1.clear();
        st2.clear();
        cin>>n;
        st1.insert("");
        st2.insert("");
        fr(i,0,n/2)
        {
            cin>>s;
            set<string> temp;
            string val;
            itfr(it,st1)
            {
                val=*it;
                val=val+s;
                sort(val.begin(),val.end());
                temp.insert(val);
            }
            itfr(it,temp)
                st1.insert(*it);
        }
        fr(i,n/2,n)
        {
            cin>>s;
            set<string> temp;
            string val;
            itfr(it,st2)
            {
                val=*it;
                val=val+s;
                sort(val.begin(),val.end());
                temp.insert(val);
            }
            itfr(it,temp)
                st2.insert(*it);
        }
        cin>>s;
        string ans;
        sort(s.begin(),s.end());
        bool flag=false;
        if(st1.find(s)!=st1.end())
            flag=true;
        if(st2.find(s)!=st2.end())
            flag=true;
        int cnt[26];
        if(!flag)
        {
            itfr(it,st1)
            {
                ans=*it;
                fr(i,0,26)
                    cnt[i]=0;
                fr(i,0,s.length())
                    cnt[s[i]-'a']++;
                fr(i,0,ans.length())
                {
                    cnt[ans[i]-'a']--;
                    if(cnt[ans[i]-'a']<0)
                        break;
                }
                if(i<ans.length())
                    continue;
                ans="";
                fr(i,0,26)
                {
                    while(cnt[i])
                    {
                        ans=ans+char(i+'a');
                        cnt[i]--;
                    }
                }
                if(st2.find(ans)!=st2.end())
                {
                    flag=true;
                    break;
                }
            }
        }
        if(flag)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    return 0;
}
