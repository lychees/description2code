#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    int n,L,D;
    cin>>n>>L>>D;
    int d[n+1];
    float p[n+1],res[n+1];
    d[0]=0;
    d[n+1]=D;
    p[0]=1.0;
    p[n+1]=1.0;
    for(int i=1;i<=n;i++){
    	cin>>p[i];
    }
    for(int i=1;i<=n;i++){
    	cin>>d[i];
    }
    res[0]=1.0;
    for(int i=1;i<=n+1;i++){
    	res[i]=0;
    	for(int j=i-1;j>=0;j--){
    		if(d[i]-d[j]<=L)
    			res[i]=max(res[j]*p[i],res[i]);
    	}
    }
    cout<<fixed;
    cout<<setprecision(6)<<res[n+1];
}
