#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,i,a;
    cin>>t;
    long long int freq[26]={0};
    string strr;
    cin>>strr;
    for(i=0;i<t;i++)
    freq[strr[i]-'a']++;
    char st,en;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>st>>en;
    	// cout<<st<<"---------"<<en<<endl;
    	if(st!=en)
    	cout<<freq[st-'a']*freq[en-'a']<<endl;
    	else
    	{
    		if(freq[st-'a']&1)
    		cout<<(freq[st-'a']*(freq[en-'a']-1)/2)<<endl;
    		else
    		cout<<((freq[st-'a']/2)*(freq[en-'a']-1))<<endl;
    	}
    }
    return 0;
}
