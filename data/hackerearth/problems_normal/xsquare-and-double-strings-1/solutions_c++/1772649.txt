#include <bits/stdc++.h>
using namespace std;

int main()
{
   	int t;
   	string s;
   	cin >> t;
   	while(t--)
   	{
   		cin >> s;
   		vector<int> freq(26,0);
   		for(int i=0;i<s.size();++i)
   			++freq[s[i]-'a'];
   		bool flag = false;
   		for(int i=0;i<26;++i)
   			if(freq[i]>=2)
   				flag = true;
   		if(flag)
   			cout << "Yes\n";
   		else
   			cout << "No\n";
   	}
    return 0;
}
