#include <iostream>
#include <math.h>
using namespace std;
int main()
{
    int T;
    unsigned long long int x, y;
    cin >> T;
    for (int i = 0; i < T; i++)
    {
    	cin >> x >> y;
    	while(x != y)
    	{
    		while(x > y)
    		{
    			x = floor(x/2);
    		}
    		while(y > x)
    		{
    			y = floor(y/2);	
    		}
    	}
    	cout << x << endl;
    }
    return 0;
}
