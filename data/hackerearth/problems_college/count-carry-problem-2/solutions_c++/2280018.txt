int main() {
    int n;
    cin >> n;
    char a[50], b[50];
    rep(i, 0, n) {
        cin >> a >> b;
        int n = strlen(a);
        int m = strlen(b);
        int j = n - 1, k = m - 1, ans = 0;
        while(j >= 0 || k >= 0) {
            if(((a[j] - '0') + (b[k] - '0')) > 9)
                ans++;
            j--;
            k--;
        }
        if(ans == 0)
            cout << "No carry operation" << endl;
        else if(ans == 1)
            cout << ans << " carry operation"<< endl;
        else
            cout << ans << " carry operations"<< endl;
    }
}
