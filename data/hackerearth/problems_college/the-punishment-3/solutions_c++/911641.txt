#include<bits/stdc++.h>
#define si(n)scanf("%d",&n)
#define pi(n)printf("%d\n",n)
#define sll(n)scanf("%lld",&n)
#define pll(n)printf("%lld\n",n)
using ll=long long;using namespace std;
int main(){
    int t;si(t);
    while(t--){
        int n,k;si(n);si(k);vector<char>v(n,'9');
        k=min(k,n-k);
        v[k-1]='8';v[n-1]='1';
        for(int i=0;i<k-1;i++){
            v[n-2-i]='0';
        }
        for(int i=0;i<n;i++)putchar(v[i]);
        putchar('\n');
    }
    return 0;
}
