#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    //cout << "Hello World!" << endl;
    char ch;
    ll i,j,k,l,m,n,h;
    while((cin>>ch>>h) >0){
    	if(h<2){ cout<<"INVALID\n\n";continue;}
    	else if(ch>='a' && ch<='z' &&  h>'z'-ch+1){ cout<<"INVALID\n\n";continue;}
    	else if(ch>='A' && ch<='Z' &&  h>'Z'-ch+1){ cout<<"INVALID\n\n";continue;}
    	else{
    		for(i=0;i<h;i++){
    			for(j=0;j<i+1;j++){
    				cout<<char(ch+i)<<" ";
    			}
    			cout<<endl;
          }
          cout<<endl;
        }
  }
    
    return 0;
}
