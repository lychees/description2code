#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    int k, m, x, y;
    int A[105];
    memset(A, 0, sizeof(A));

    cin >> k >> m;
    for (int i = 0; i < k; i++) {
        cin >> x;
        for (int i = 0; i < x; i++) {
            cin >> y;
            A[y]++;
        }
    }
    for (int i = 1; i <= m; i++) {
        if (A[i] == 0) {
            cout << "NO\n";
            return 0;
        }
    }
    cout << "YES\n";
    return 0;
}
