#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
int a[10001];
static int dp[501][101][101];
int gcd(int a,int b){
	if(b==0)
		return a;
	return gcd(b,a%b);
}
int n;
int solve(int ind,int _gcd,int last){
	if(ind==n){
		return (_gcd==1);
	}
	for(int i=last;i>=last;i--){
		if(dp[ind][_gcd][i]!=-1){
			return dp[ind][_gcd][i];
		}
	}
	int ans = 0;
	if(a[ind]>last)
		ans += solve(ind+1,gcd(_gcd,a[ind]),a[ind]);
    ans += solve(ind+1,_gcd,last);
    if(ans>=mod)
        ans-=mod;
    dp[ind][_gcd][last]=ans;
    return ans;
}
int main(){
	//int n;
	for(int i=0;i<501;i++)
		for(int j=0;j<101;j++)
			for(int k=0;k<101;k++)
				dp[i][j][k]=-1;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
    int ans = solve(0,0,0);
    printf("%d\n",ans);
	return 0;
}
