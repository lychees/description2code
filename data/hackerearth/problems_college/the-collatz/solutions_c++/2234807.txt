#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	vector<int>v;
    	cin>>n;
    	long long int sum=0;
    	while(n>1)
    	{
    		sum+=n;
    		if(n%2==0)
    		{
    			v.push_back(n);
    			n=n/2;
    		}
    		else
    		{
    			v.push_back(n);
    			n=3*n+1;
    		}
    	}
    	sum++;
    	v.push_back(1);//for 1
    	int evensum=0,oddsum=0,evenrem,oddrem;
    	vector<int>::iterator it;
    	for(it=v.begin();it!=v.end();++it)
    	{
    		if(*it%2==0)
    		{
    			evensum+=*it;
    		}
    		else
    		{
    			oddsum+=*it;
    		}
    	}
    	//cout<<sum<<"  "<<oddsum<<"  "<<evensum<<endl;
    	evenrem=evensum%(v.size()-1);
    	oddrem=oddsum%(v.size()-1);
    	//cout<<v.size()<<"  "<<evenrem<<"  "<<oddrem<<endl;
    	if(oddrem==evenrem)
    	{
    		cout<<"Tie"<<endl;
    	}
    	else
    	{
    		if(oddrem>evenrem)
    		{
    			cout<<"Odd Rules"<<endl;
    		}
    		else
    		{
    			cout<<"Even Rules"<<endl;
    		}
    	}
    	
    	
    }
}
