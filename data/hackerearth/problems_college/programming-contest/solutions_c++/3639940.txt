#include <bits/stdc++.h>
using namespace std;

#define si(c) ((int)(c).size())

bool eq(const string &a, const string &b, int len) {
    int la = si(a), lb = si(b);
    for (int i = 0; i < len; i++) if (b[i] != a[la-1-i]) return false;
    return true;
}

int go(const string &a, const string &b) {
    int mx = 0;
    for (int len = 1; len <= min(si(a), si(b)); len++) if (eq(a,b,len))
        mx = len;
    return si(b) - mx;
}

int main() {
    string s[4]; s[0] = "";
    cin >> s[1] >> s[2] >> s[3];
    sort(s, s+4);
    int res = 1000; 
    do {
        int cand = 0;
        for (int i = 1; i <= 3; i++) cand += go(s[i-1], s[i]);
//        cerr << cand << endl;
        res = min(res, cand);
    } while (next_permutation(s+1,s+4));
    cout << res << endl;
}
