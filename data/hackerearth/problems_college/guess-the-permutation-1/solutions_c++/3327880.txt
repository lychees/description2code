#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int a[n][n],b[n];
		
		int i,j,temp,k=0;
		for(i=0;i<n;i++)
	    {
	    	b[i] = 0;
	    	
	    }
		
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				scanf("%d",&a[i][j]);
				if(a[i][j]==0)
				{
					a[i][j] = n;
				}
			}
		}
		
		
	    set<int>m[n];
		for(i=0;i<n;i++)
		{
			m[i].insert(a[i],a[i]+n);
			if(m[i].size() == n)
			{
			b[i] = 1;
		    }
	    }
	    for(i=0;i<n;i++)
	    {
	    	if(b[i])
	    	{
	    		for(j=0;j<n;j++)
	    		{
	    			if(a[i][j]==n-1)
	    			{
	    				if(j>i)
	    				{
	    					b[i]=0;
	    				}
	    				else
	    				{
	    					k = 1;
	    					temp = i;
	    					break;
	    				}
	    			}
	    			if ( k==1)
	    			break;
	    		}
	    		
	    	}
	    	if ( k==1)
	    			break;
	    		
	    }
	    for(i=0;i<n;i++)
	    {
	    	printf("%d ",a[temp][i]);
	    }
	    printf("\n");
}
}
