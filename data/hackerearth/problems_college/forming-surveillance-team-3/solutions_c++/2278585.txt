#include<bits/stdc++.h>

using namespace std;

int srch(int ar[],int m,int c)
{
    for(int i=0; i<c; i++)
    {
        if(ar[i]==m)
        {
            return 0;
        }
        
    }
    return 1;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        int c=0;
        cin>>n;
        int ar[n];
        
        while(n--)
        {
            int m;
            cin>>m;
            if(srch(ar,m,c))
            {
                ar[c]=m;
                c++;
            }
            
            
        }
       cout<<c<<endl;
    }
    return 0;
}
