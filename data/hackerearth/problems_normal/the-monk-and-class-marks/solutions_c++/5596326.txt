#include <iostream>
#include <map>
#include <algorithm>
#include <vector>

using namespace std;

int main()
{
    int n;
    cin>>n;
    vector <pair<int,string>> x;
    for(int i=0;i<n;i++){
    	string s;
    	cin>>s;
    	int m;
    	cin>>m;
    	x.push_back(pair<int,string>(-m,s));
    }
    sort(x.begin(),x.end());
    for(int i=0;i<n;i++){
    	cout<<x[i].second<<" ";
    	cout<<-x[i].first<<endl;
    	
    }
    
    
}
