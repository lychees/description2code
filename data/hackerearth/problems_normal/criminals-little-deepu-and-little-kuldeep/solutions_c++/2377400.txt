#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,ans,p_ans,i,j;
    cin>>t;
    
    while(t--)
    {
    long long a[100000],n;
     p_ans = ans =1;
     cin>>n;
   
    	 for( i=0;i<n;i++)
    	 {
    	 	cin>>a[i];
    	 	
    	 }
    	 
    	 sort(a,a+n);
    	 
    	 for( i=0;i<n-1;i++)
    	 {
    	 	if(a[i]==a[i+1])
    	 	{
    	 		ans++;
    	 	}
    	 	
    	 	else    	          
        {
          if (ans > p_ans)
        p_ans = ans; 
          ans = 1;      
        }

    }
      if (ans > p_ans)   
    p_ans = ans;
      cout << p_ans << endl;
    }
  return 0;
}
