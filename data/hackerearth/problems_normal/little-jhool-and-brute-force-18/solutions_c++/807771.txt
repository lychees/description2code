#include <iostream>
#include <cstdio>

using namespace std;
#define max 704977

long long int x[100];
long long int xy[max]= {0};

long long int jhool(long long int n){
      long long int i, t=-1;
      for(i=n;i>=0;i--){
        if(xy[i] > 2){
            t=i;
            break;
        }
      }
  return t;
}

int main(){
    long long int i,t,n,j,temp=0;

    for(i=0;i<=90;i++){
        x[i] = i*i*i;
    }

   for(i=0;i<=90;i++){
        for(j=0;j<=90;j++){
            temp = x[i]+x[j];
            if(temp < max)
                xy[temp]++;
        }
    }


cin>>t;

while(t){
    t--;
    cin>>n;
    cout<<jhool(n)<<"\n";
}
return 0;
}
