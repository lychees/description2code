// next_permutation example
#include <iostream>     // std::cout
#include <algorithm>    // std::next_permutation, std::sort
#include<bits/stdc++.h>
using namespace std;
int dx[4]= {-1,1,0,0};
int dy[4]= {0,0,1,-1};
int ans=0;int a[111][111];
int n,m,q,x,y,d;
int visited[111][111];

void dfs(int x,int y , int d)
{
    //cout<<x<<" "<<y<<" "<<d<<endl;
    visited[x][y]=1;
    for(int i=0; i<4; i++)
    {
        int newx=x+dx[i];
        int newy=y+dy[i];
        if(newx<0 || newx>=n || newy>=m || newy<0 || visited[newx][newy]==1)
        {
            continue;
        }
        if(abs(a[x][y]-a[newx][newy])<=d)
        {
            ans+=1;
            dfs(newx,newy,d);
        }
    }
    return ;
}
int main ()
{
    //freopen("in.txt","r",stdin);
    cin>>n>>m>>q;
    for(int i= 0; i<n; i++)
    {
        for(int j= 0; j<m; j++)
        {
            cin>>a[i][j];
        }
    }
    while(q--)
    {
        for(int i= 0; i<n; i++)
        {
            for(int j= 0; j<m; j++)
            {
                visited[i][j]=0;
            }
        }
        cin>>x>>y>>d;
        x-=1;y-=1;
        ans=1;
        dfs(x,y,d);
        cout<<ans<<endl;
    }
    return 0;
}
