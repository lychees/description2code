t=int(raw_input())
for i in range(0,t):
	res=""
	n=int(raw_input())
	l=list(map(int,raw_input().split(" ")))
	for j in range(0,n):
		bi=bin(l[j])
		bi=str(bi[2:])
		out=""
		for h in range(0,len(bi)):
			if(bi[h]=='1'):
				out=out+'0'
			if(bi[h]=='0'):
				out=out+'1'
		res=res+chr(int(out,2))
	print res
