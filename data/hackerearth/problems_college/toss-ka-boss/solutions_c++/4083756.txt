#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   cin >> t;
   while(t--)
   {
   	string s;
   	cin >> s;
   	long int dhoni = 0;
   	int i;
   	int cnt = 0;
   	for(i = 0; i < s.length(); i++)
   	{
   		if(s[i] == 'H')
   		{
   			cnt++;
   			dhoni += cnt;
   		}
   		else
   		{
   			cnt = 0;
   		}
   	}
   	cout<<dhoni<<endl;
   }
   return 0;
   }
