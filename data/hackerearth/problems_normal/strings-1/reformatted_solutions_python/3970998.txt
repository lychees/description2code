from math import log
t = int(raw_input())
for i in range(t):
    m, n = [int(x) for x in raw_input().split()]
    if m == 1 or n == 1:
        if m == n:
            print("YES")
        else:
            print("NO")
    elif log(m) - log(n) == log(log(m)) - log(log(n)):
        print("YES")
    else:
        print("NO")
