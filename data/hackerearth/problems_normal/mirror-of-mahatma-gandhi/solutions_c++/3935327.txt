#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   string s,r;int check;
   while(t--)
   {
   	cin>>s;
   	check=1;
   	int n=s.length();
   	
   	for(int i=0;i<n;i++)
   	{
   		if(s[i]=='1'||s[i]=='0'||s[i]=='8')
   		{
   			r=s;
   			reverse(r.begin(),r.end());
   			
   			if(s==r)
   			{
   				check=1;
   			}
   			else
   			check=0;
   		}
   		else
   		{
   			check=0;break;
   		}
   	}
   	if(check)
   	cout<<"YES\n";
   	else
   	cout<<"NO\n";
   }
    return 0;
}
