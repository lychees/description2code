#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pii pair<int,int>
#define pll pair<ll,ll>
#define mp make_pair
#define ff first
#define ss second
#define pb push_back
#define pf pop_front
#define IOS ios::sync_with_stdio(false)
int t , n , f, s;
vector <int> g[10005];
int vis[10005] , slap[10005] , ans;
void dfs(int v){
    vis[v] = 1;
    ans++;
    for(int i = 0 ; i < g[v].size() ; i++){
        int el = g[v][i];
        if(!vis[el])dfs(el);
    }
}
int main(){

    IOS;
    cin.tie(NULL);
	int i , j , k , a, b ;
    cin >> t;
    while(t--){
        cin >> n >> f >> s;
        memset(vis,0,sizeof(vis));
        for(i = 1 ; i <= 10000 ; i++){
            g[i].clear();
        }
        for(i = 1 ; i <= f ; i++){
            cin >> a >> b;
            g[a].pb(b);
        }
        ans = 0;
        for(i = 1 ; i <= s ; i++){
            cin >> slap[i];
            if(!vis[slap[i]])dfs(slap[i]);
        }
        cout << ans << endl;
    }
	return 0;
}
