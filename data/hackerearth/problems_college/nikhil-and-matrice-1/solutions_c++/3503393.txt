ll gcd(ll a,ll b){return (!a)?b:(!b)?a:gcd(b,a%b);}

ll pow(ll a,ll b,ll MOD) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>MOD) x%=MOD;
        }
        y = (y*y);
        if(y>MOD) y%=MOD;
        b /= 2;
    }
    return x;
}

int mat[100][100],t,n;
int arr1[]={0,1,0,-1};
int arr2[]={1,0,-1,0};

bool out(int c)
{
    if(c<0||c>=n)
        return true;
    return false;
}

void func(int x,int y,int dir,int num)
{
    if(num>n*n)
        return;

    if((mat[x+arr1[dir]][y+arr2[dir]]!=-1)||(out(x+arr1[dir])||(out(y+arr2[dir]))))
    {
        mat[x][y]=num;
        dir=(dir+1)%4;
        func(x+arr1[dir],y+arr2[dir],dir,num+1);
        return;
    }
    
    mat[x][y]=num;
    func(x+arr1[dir],y+arr2[dir],dir,num+1);
}

int main()
{
    //    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

    cin>>t;
    while(t--)
    {
        memset(mat,-1,sizeof(mat));
        cin>>n;
        func(0,0,0,1);

        for (int i = 0; i < n; ++i)
        {
            for (int j = 0;j < n; ++j)
                cout<<mat[i][j];
            cout<<endl;
        }
        cout<<endl;
    }    

    return 0;
}
