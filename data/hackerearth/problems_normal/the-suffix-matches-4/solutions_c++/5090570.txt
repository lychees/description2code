#include <bits/stdc++.h>
using namespace std;
char a[200005];
int z[200005];
int main()
{
   // cout << "Hello World!" << endl;
   scanf("%s",&a);
   int n;
   for(n=0;a[n];n++);
   reverse(a,a+n);
   int i,l,r;
   for(i=1,l=r=0;i<n;i++)
   {
   	if(i<=r)
   	z[i] = min(z[i-l],r-i+1);
   	while(i+z[i] < n && a[z[i]] == a[i+z[i]])
   	z[i]++;
   	if(i+z[i]-1 > r)
   	{
   		l = i;
   		r = i + z[i] - 1;
   	}
   }
   z[0] = n;
   int q,x;
   scanf("%d",&q);
   while(q--)
   {
   	scanf("%d",&x);
   	printf("%d\n",z[n-x]);
   }
    return 0;
}
