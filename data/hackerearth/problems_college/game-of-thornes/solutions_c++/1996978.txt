#include <bits/stdc++.h>
#define ll long long
using namespace std;
int a[26]={0};
int main()
{
    string s;
    cin>>s;
    int n=s.length();
    for(int i=0;i<n;i++)
        a[s[i]-'a']+=1;
    int c=0;
    for(int i=0;i<26;i++)
    {
        if(a[i]%2!=0)
            c++;
    }
    if(c>1)
        cout<<"NO"<<endl;
    else
        cout<<"YES"<<endl;
    return 0;
}
