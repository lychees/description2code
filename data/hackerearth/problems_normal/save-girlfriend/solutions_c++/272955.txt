#include <iostream>
#include <vector>
#include <cmath>
#include <unordered_set>
using namespace std;

namespace
{
    typedef vector<unsigned int> ValCol;

    unsigned int maxDelims(unsigned int a, unsigned int b)
    {
        ValCol counts(b - a + 1, 1);
        if(a == 1)
            counts[0] = 0;
        const unsigned int last = sqrt(b);
        for(unsigned int i = 2; i <= last; ++i)
        {
            for(unsigned int j = a+(i - a%i)%i; j <= b; j += i)
            {
                if(j/i > i)
                    counts[j-a] += 2;
                else if(j/i == 1)
                    ++counts[j-a];
            }
        }
        unsigned int max = 0;
        unsigned int count = 0;
        for(auto c : counts)
        {
            if(c > max)
            {
                max = c;
                count = 1;
            }
            else if(c == max)
            {
                ++count;
            }
        }
        return count;
    }
}

int main()
{
    int t = 0;
    cin>>t;
    while(t-- > 0)
    {
        unsigned int a = 0;
        unsigned int b = 0;
        cin>>a>>b;
        cout<<maxDelims(a, b)+1<<endl;
    }
    return 0;
}
