#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;

string a[105];
ll n,ctr;

int main()
{
    scanf("%lld",&n);
    for(ll i=0;i<n;i++)
        cin>>a[i];
    for(ll i=0;i<n;i++)
    {
        string str = a[i];
        ll len = str.size();
        if(len%2==0)
        {
            stack<char> st;
            for(ll j=0;j<len;j++)
            {
                if(st.empty())
                    st.push(str[j]);
                else
                {
                    if(str[j]==st.top())
                        st.pop();
                    else
                        st.push(str[j]);
                }
            }
            if(st.empty())
                ctr++;
        }
    }
    cout<<ctr<<endl;
    return 0;
}
