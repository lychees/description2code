#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    string s;
    set<string> u;
    int n,m;
    bool flag;
    while(t--)
    {
        flag=true;
        scanf("%d%d",&n,&m);
        u.clear();
        for(int i=1;i<=n;i++)
        {
            cin>>s;
            u.insert(s);
            if(u.size()<i)
            {
                flag=false;
            }
        }
        if(flag==true)
            printf("Yes\n");
        else
            printf("No\n");
    }
    return 0;
}
