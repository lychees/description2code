#include<bits/stdc++.h>
using namespace std;
long long sum;
void primeFactors(int n)
{
	bool flag=0;
    	while(n%2==0)
    	{
    	    	if(!flag)sum+=2;//printf("%d ", 2);
        	n = n/2;flag=1;
    	}
	flag=0;
    	for(int i = 3; i <= sqrt(n); i = i+2)
    	{
        	while (n%i == 0)
        	{
            		if(!flag)sum+=i;//printf("%d ", i);
            		n = n/i;flag=1;
        	}
		flag=0;
    	}
    	if(n>2)
        	sum+=n;//printf ("%d ", n);
}
int main()
{
	long n,k,count=0;
	scanf("%ld%ld",&n,&k);
	if(k==1||n==1)printf("0\n");
	else
	{
		for(long i=2;i<=n;i++)
		{
			sum=0;
			if(!(i&(i-1)))sum+=2;
			else	primeFactors(i);
			if(sum==k)
				count++;
		}
		printf("%ld\n",count);
	}
	return 0;
}
