#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int i,j,k,m,n,t,x,s;
	cin>>t;
	while(t--)
	{
           cin>>n>>m>>x;s=0;
           for(i=0;i<n;i++)
	   {
	   	scanf("%lld",&k);
	   	s=s+k;
	   }
	   if(((n+1)*x)<=s+1) cout<<"1\n";
	   else if(((((n+1)*x)-s))>m) cout<<"Impossible\n";
	   else cout<<(((n+1)*x)-s)<<endl;
	}
	return 0;
}
