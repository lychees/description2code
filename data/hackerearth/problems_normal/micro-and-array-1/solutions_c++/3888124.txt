#include<bits/stdc++.h>

using namespace std;

int main(){
    int n,q;
    cin>>n>>q;
    vector<int>vec(n);
    for(int i=0;i<n;i++){
        cin>>vec[i];
    }
/*    for(int i=0;i<n;i++){
        cout<<vec[i]<<" ";
    }*/
    while(q--){
        int qt,v,x,y;
        cin>>qt;
        //cout<<qt<<" ";
        if(qt==1){
            cin>>v;
            int f=0,i;
            for(i=0;i<n;i++){
                if(vec[i]>=v){
                    f=1;
                    break;
                }
            }
             	if(f==1)
                	cout<<i+1<<endl;
                else
                    cout<<"-1"<<endl;
        }
        else{
            cin>>x>>y;
            vec[x-1]=y;
        }
    }
}
