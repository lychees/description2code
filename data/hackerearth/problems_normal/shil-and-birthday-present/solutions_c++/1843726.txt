#include <bits/stdc++.h>
using namespace std;
int compare(const void*a,const void*b){
	return(*(int*)a-*(int*)b);
}
int main(){
	int n,i,j;
	unsigned long long sum=0;
	scanf("%d",&n);
	int a[n];
	int ans[n];
	for(i=0;i<n;i++)	scanf("%d",&a[i]);
	qsort(a,n,sizeof(int),compare);
	ans[0]=0;
	for(i=1;i<n;i++){
		if(a[i]==a[i-1])	ans[i]=ans[i-1];
		else {
			ans[i]=1+ans[i-1];
			sum+=ans[i];
		}	
	}
	printf("%llu\n",sum);
	return 0;
}
