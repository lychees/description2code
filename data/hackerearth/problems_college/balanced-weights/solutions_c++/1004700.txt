#include <iostream>
#include<cstdio>
using namespace std;

long long gcd ( long long a, long long b )
{
  if ( a==0 ) return b;
  return gcd ( b%a, a );
}

int main()
{
   long long t,n,i,ans,temp;
   scanf("%lld",&t);
   while(t--)
   {
   	scanf("%lld",&n);
   	temp=n;
   	scanf("%lld",&ans);
   	n--;
   	while(n--) {scanf("%lld",&i);ans=gcd(ans,i);}
   	printf("%lld\n",ans*temp);
   }
}
