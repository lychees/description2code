#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,l,r,q,maxi,mini;
	cin>>n>>l>>r>>q;
	int a[n],b[n],j,i;
	j = 0;
	maxi = -1;
	mini = 10000000;
	for(i = 0; i < n; i++) cin>>a[i];
	for(i = l; i <= r; i++) {
		b[j++] = a[i];
		maxi = max(a[i],maxi);
		mini = min(a[i],mini);
	}
	sort(b,b+j);
	int x;
	while(q--) {
		cin>>x;
		if(x < mini) {
			cout<<b[0]<<endl;
		}
		else if(x >= maxi) {
			cout<<-1<<endl;
		}
		else {
			int y = lower_bound(b,b+j,x)-b;
			if(b[y] > x) {
				cout<<b[y]<<endl;
			}
			else {
				cout<<b[y+1]<<endl;
			}
			//cout<<b[y]<<endl;
		}
	}
    return 0;
}
