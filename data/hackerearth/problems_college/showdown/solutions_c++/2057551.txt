#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<vector>
#include<queue>
#define mod 1000000007ll
using namespace std;
typedef vector<int> vi;
typedef vector<vi> vvi;
int n,res,A[55],B[55];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)scanf("%d",&A[i]);
	for(int i=1;i<=n;i++)scanf("%d",&B[i]);
		
	sort(A+1,A+n+1);
	sort(B+1,B+n+1);
	for(int i=1;i<=n;i++)
	{
		for(int j=n;j>=1;j--)
		if(A[i]>B[j]&&B[j]!=-1)
		{
			res+=100;
			A[i]=B[j]=-1;
			break;
		}
	}
	for(int i=1;i<=n;i++)
	{
		for(int j=n;j>=1;j--)
		if(A[i]==B[j]&&B[j]!=-1)
		{
			res+=50;
			A[i]=B[j]=-1;
			break;
		}
	}
	cout<<res<<"\n";
	return 0;
}
