#include <bits/stdc++.h>
using namespace std;
const int MAX = 1000 + 7;
int y[MAX];
int countx[MAX];
int main()
{
   int N;
   cin >> N ;
   for(int i = 1; i <= N ; i++)
   {
     cin >> y[i];
	 countx[y[i]]++;
   }
   
   int ans = 0;
   for(int i = 0; i <= 1000 ; i++)
   {
   	
            if(!countx[i])
			continue;
			
            for(int j = i ; j <= 1000 ; j++)
			{
	             		
			        if(!countx[j])
					continue;
			 int temp_ans ;
			 if(i != j)
			temp_ans = countx[i] + countx[j];
			else
			temp_ans = countx[i];
			
			int k = 1;
			int m =  N;
			
			while (k <= N && (y[k] <  i || y[k] > j ) )
			  k++;
			
			while ( m >= 0 && (y[m]  < i || y[m] > j  ))
			  m--;
			
			
			
			if( y[k] != i &&  y[k] != j )
			 temp_ans++;
			
			if(y[m] != i  && y[m] != j)
			temp_ans++;
			
			ans = max(ans , temp_ans);
			 
			  
			
			
			}
   
  
   
   
   }
   
   
   cout << ans << endl;
   return 0;

}
