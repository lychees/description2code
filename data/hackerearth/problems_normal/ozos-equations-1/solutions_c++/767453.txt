int sod(long long v){
	int s=0;
	while(v){
		s+=v%10;
		v/=10;
	}
	return s;
}
int main(){
	int t;
	long double sq,n,p;
	long long v,va,ans;
	cin>>t;
	for(int k=0;k<t;k++){
		cin>>n;
		ans = -1;
		for(int i=1;i<100;i++){
			p = (long double)(i*i) + 4*n;
			sq = sqrtl(p);
			if(sq == floorl(sq)){
				v = -(long long)i + (long long)sq;
				if(v>0 && v%2==0){
					va = v/2;
					if(sod(va) == i && (ans==-1 || va<ans))
						ans = va;
				}
			}
		}
		cout<<ans<<"\n";
	}
	return 0;
}
