#include<bits/stdc++.h>
using namespace std;
 
int a[10002];
 
 
int main()
{
    int t,pos;
    cin>>t;
    while(t--)
    {
        int n,k,i=0;
        cin>>n>>k;
        for(i=1; i<=n; i++)
            cin>>a[i];
 
        for(int i=1; i<=n; i++)
        {
            pos = i ;
            for(int j=i+1; j<=n; j++)
            {
                if((j-i)>k)
                    break;
 
                if(a[j]<a[pos])
                    pos = j ;
 
            }
 
            for(int j=pos; j>i; j--)
            {
                swap(a[j],a[j-1]);
            }
 
            k = k - ( pos - i );
 
        }
 
        for(int i=1; i<=n; i++)
        {
            cout<<a[i]<<" ";
        }
        cout<<endl;
    }
    return 0;
}
