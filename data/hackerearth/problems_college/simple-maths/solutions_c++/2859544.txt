#include <iostream>
using namespace std;

int gcd(int a, int b)
{
    for (;;)
    {
        if (a == 0) return b;
        b %= a;
        if (b == 0) return a;
        a %= b;
    }
}

int lcm(int a, int b)
{
    int temp = gcd(a, b);

    return temp ? (a / temp * b) : 0;
}

int main()
{
    int t,a,b;
    long long n,m1,m2,m3;
    cin>>t;
    while(t--)
    {
    	cin>>n>>a>>b;
    	if(a==0||b==0)	cout<<"-1"<<endl;
    	else {
    	m1=n/a;
    	m2=n/b;
    	int lc = lcm(a,b);
    	m3=n/lc;
    	cout<<(m1+m2-m3)<<endl;
    	}
    }
    return 0;
}
