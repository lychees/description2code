#include <bits/stdc++.h>
using namespace std;

#define MP make_pair
int main()
{
    for( long long n, k; cin>>n>>k; ){
    	vector<long long>X(k);
    	vector<long long>Y(k);
    	long long res = 0, mn;
    	pair<long long, long long>corner[] = { MP(1, 1), MP(1, n), MP(n, 1), MP(n,n)};
    	
    	for( int i=0 ; i<k; i++){
    		cin>>X[i];
    	}
    	for( int i=0 ; i<k; i++){
    		cin>>Y[i];
    		mn = n*n;
    		for( auto pii : corner){
    			mn = min(mn, abs(X[i]-pii.first)+abs(Y[i]-pii.second) );
    		}
    		res+=mn;
    	}
    	
    	
    	cout<<res<<"\n";
    }
    return 0;
}
