for i in range(input()):
    n, x = map(int, raw_input().split())
    costs = []
    for j in range(n):
        costs.append(input())

    left = 0
    right = 0
    ans = 'NO'
    cur_sum = costs[0]

    while left < n and right < n:
        if cur_sum < x:
            right += 1
            if right >= n:
                break
            cur_sum += costs[right]
        elif cur_sum > x:
            if left == right:
                right += 1
                if right >= n:
                    break
                cur_sum += costs[right]
            cur_sum -= costs[left]
            left += 1
        else:
            ans = 'YES'
            break

    print ans
