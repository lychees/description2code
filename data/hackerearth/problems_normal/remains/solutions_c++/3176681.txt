#include<iostream>
#include<algorithm>
#include<string.h>
#include<vector>
#include<cmath>
using namespace std;

long long a[1005],x,y,n,ans;

int main()
{
    int t;
    cin>>t;
    while(t--) {
        long long i,j,m,temp,t1,t2;
        cin>>a[1]>>a[2]>>n;
        ans=a[1]+a[2]; t1=a[1]; t2=a[2];
        for(i=3;i<=n;i++) {
            if(t1==t2) {
                temp=n-i;
                ans+=((temp/3)*2*t1);
                if(temp%3==1) ans+=t1;
                else if(temp%3==2) ans+=t1+t1;
                break;
            }
            else if(t1!=t2) temp=abs(t1-t2),ans+=temp,t1=t2,t2=temp;
        }
        cout<<ans<<"\n";
    }
    return 0;
}
