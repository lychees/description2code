#include <bits/stdc++.h>
using namespace std;

int main(void) {
	
	int T = 0;
	scanf("%d", &T);
	while(T--) {
		int N = 0;
		scanf("%d", &N);
		vector<int> arr(N + 1,0), suffix(N + 1, 0);
		for(int i = 1; i <= N; ++i) {
			scanf("%d", &arr[i]);
			suffix[i] =  suffix[i - 1] + arr[i];
		}
		
		int flag = 0;
		for(int j = 2; j < N && !flag; ++j) {
			if(suffix[j - 1] == (suffix[N] - suffix[j]))
				flag = 1;
		}
		if(flag)
			puts("YES");
		else
			puts("NO");
		
	}
	
	return 0;
}
