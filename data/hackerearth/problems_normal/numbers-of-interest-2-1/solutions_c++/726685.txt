#include <cstring>
#include <cmath>
#include <iostream>
#include <algorithm>

#define N 1000005

using namespace std;

long long x[N];
long long y[N];
long long f[N];
long long g[N];
int main()
{
	for (int i = 0; i < N; i++) {
		x[i] = i;
		y[i] = 1;
	}
	x[0] = x[1] = 0;
	y[0] = y[1] = 0;
	f[0] = g[0] = 0;
	
	for (int i = 2; i < N; i++) {
		if (x[i] == i)
			for (int j = 2; i*j <= N; j++) x[i*j] = min((long long)i, x[i*j]);
	}

	for (int i = 2; i < N; i++)
            for (int j = 2; i*j <= N; j++) y[i*j] += i;


	for (int i = 0; i < N; i++) {
        f[i] = f[i-1] + x[i];
        g[i] = g[i-1] + y[i];
	}

    int t;
    int n;

    cin >> t;

    while (t--) {
        cin >> n;
        cout << (f[n]+g[n])%n << endl;
    }
    return 0;
}
