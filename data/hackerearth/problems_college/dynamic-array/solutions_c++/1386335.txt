int gcd(int a,int b)
{
	if(b==0) return a;
	else return gcd(b,a%b);
}



int arr[MAX+5];

struct data
{
	int sum;          //no of elements in the interval
	int gcdresult;    //gcd result of the numbers in the interval
	void assignleaf()
	{	
		// if no element is present , then sum=0 , gcd=0;
		sum=0;      
		gcdresult=0;
	}

	void assignleaf(int idx ,int ins)
	{
		if(ins==1) //insertion
		{
			if(arr[idx]==0)
			{
				///first occurrence of the number
				arr[idx] = 1 ;
				sum = 1;
				gcdresult = idx;
						
			}
			else if(arr[idx]>0)
			{
				///multiple occurrence of the number
				arr[idx] += 1 ;
				sum += 1;
				gcdresult  = idx;		
			}
		}
		else if(ins==0) //deletion
		{
			--arr[idx];
			--sum;

			if(arr[idx]==0)
				 sum=0 , gcdresult=0;

		}
	}

	void combine(data &l, data &r)
	{
		sum = r.sum + l.sum ;
		gcdresult = gcd(l.gcdresult,r.gcdresult);				
	}



};
		

data tree[4*MAX+5];

void build_tree(int node,int s,int e)
{
	if(s>e) return;
	
	if(s==e)
	{
		tree[node].assignleaf();
		return;
	}
	
	int mid=(s+e)/2;

	build_tree(2*node,s,mid);
	build_tree(2*node+1,mid+1,e);

	tree[node].combine(tree[2*node],tree[2*node+1]);

}

void point_update(int node,int segs,int sege,int index,int ins)
{
	if(segs==sege && segs==index)
	{
		tree[node].assignleaf(index,ins);
		 return;
	}

	int mid = (segs+sege)/2;

	if(index<=mid)
		point_update(2*node,segs,mid,index,ins);
	else 
		point_update(2*node+1,mid+1,sege,index,ins);


	tree[node].combine(tree[2*node],tree[2*node+1]);

}

int Xth_smallest(int node,int segs,int sege,int x)
{

	if(segs==sege && arr[segs]>0)
		return segs;		

	int mid = (segs+sege)/2;
	
	
	if(tree[2*node].sum<x)
	{
		x = x -  tree[2*node].sum;
		return Xth_smallest(2*node+1,mid+1,sege,x);
	}
	else
	{
		return Xth_smallest(2*node,segs,mid,x);
	}
	

}

int gcd_query(int node,int segs,int sege,int qs,int qe)
{

	if(segs>sege || sege<qs || segs>qe) return 0;

	if(segs>=qs && sege<= qe) 
	{
		return tree[node].gcdresult;		
	}

	
	int mid = (segs+sege)/2;

	int q1 = gcd_query(2*node, segs,mid, qs, qe);
	int q2 = gcd_query(2*node+1, 1+mid, sege, qs, qe);
 
	int res = gcd(q1, q2);
 
	return res;
}
