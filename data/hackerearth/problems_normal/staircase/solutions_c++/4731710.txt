#include <bits/stdc++.h>

using namespace std;

int jack(int n)
{
	if (n < 0) {
		return 0;
	}
	
	if (n == 0) {
		return 1;
	}
	
	return jack(n - 1) + jack(n - 2) + jack(n - 3); 
}

int jill(int n)
{
	if (n < 0) {
		return 0;
	}
	
	if (n == 0) {
		return 1;
	}
	
	return jill(n - 1) + jill(n - 2);
}

int main()
{
	int n;
	scanf("%d", &n);
	if (n < 0) {
		printf("ERROR\n");
		return 0;
	}
	printf("Jack-%d\n", jack(n));
	printf("Jill-%d\n", jill(n));
	
	return 0;
}
