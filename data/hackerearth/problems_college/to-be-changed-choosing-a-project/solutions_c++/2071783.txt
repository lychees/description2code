//given n,r,each cut at r point, last element to survive.
//n=5,r=3,12345->1245->245->24->4

#include<iostream>
using namespace std;
int main()
{
int n,r,i,j,m;
cin>>n>>r;

int a[n];
int q=n;

for(i=0;i<n;i++)
a[i]=i+1;

int k=0;

for(j=0;j<q-1;j++)
{
m=(k+r-1)%n;
for(i=m;i<n;i++)
{
a[i]=a[i+1];
}
k=m;
n--;

}

cout<<a[0];

}
