#include<stdio.h>
int main()
{
    int n,m,i,a[100000],b[100000];
    long long A,B;
    A=B=0;
    scanf("%d%d",&n,&m);
    for(i=0;i<n;i++)
    {
    	scanf("%d",&a[i]);
    	A=A+a[i];
    }
    for(i=0;i<m;i++)
    {
    	scanf("%d",&b[i]);
    	B=B+b[i];
    }
    printf("%lld",A*B);
    return 0;
}
