#include <iostream>
#include <string>
using namespace std;

int main()
{
    int T,flag=0;
    string str;
    cin>>T;
    
    while(T>0) {
    	
    	cin>>str;
		flag=0;
		for(int i=0; str[i]!='\0';i++) {
			
			
			if(str[i]=='A'||str[i]=='E'||str[i]=='I'||str[i]=='O'||str[i]=='U'||str[i]=='a'||str[i]=='e'||str[i]=='i'||str[i]=='o'||str[i]=='u') {
				cout<<str[i];
				flag++;
			}
		}
		if(flag==0) cout<<"No";
    	cout<<"\n";
    	
    	T--;
    }
    return 0;
}
