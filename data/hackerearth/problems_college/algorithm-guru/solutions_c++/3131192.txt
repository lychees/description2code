#include <bits/stdc++.h>

using namespace std;

int main()
{
    int tc;
    cin >> tc;

    while(tc--){
    int n,k;
    cin >> n >> k;
    int a[n];

    for (int i = 0;i <n;i++)
        cin >> a[i];

    int m = 0;
    int min1 = INFINITY;
    int ans  =0;
    for(int i = 0;i < n;i++)
    {
        if ((k % a[i]) < min1)
        {
            min1 = k % a[i];
            ans = k - min1;
        }
    }

    if (min1 == 0)
    {
        cout << k << endl;
    } else
    {

        for (int i = 0;i < n;i++)
        {
            for (int j = i + 1;j < n;j++)
            {
                if ((k % (a[i] + a[j])) < min1)
                    {
                        min1 = k % (a[i]+a[j]);
                        ans = k - min1;

                       // cout << "min is : " << min1 << "   " << ans << endl;
                    }
            }
        }

        cout << ans << endl;
    }
    }

    return 0;
}
