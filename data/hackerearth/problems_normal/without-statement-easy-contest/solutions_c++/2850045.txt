int convert( int n ) {
    int num = 0;
    while( n ) {
        int d = n % 10;
        num += d * d;
        n /= 10;
    }
    return num;
}

int main() {
    int t; cin >> t;
    while( t-- ) {
        int n, m; cin >> n >> m;
        if( m <= 1000 ) {
            while( m -- ) n = convert( n );
        } else {
            map<int, int> M, R;
            int i;
            fr(i, 1, m) {
                n = convert( n );
                if( M.count(n) ) {
                    int p = i - M[n];
                    m -= i;
                    m %= p;
                    n = R[M[n] + m];
                    break;
                } else {
                    M[n] = i;
                    R[i] = n;
                }
            }
        }
        cout << n << "\n";
    }
    return 0;
}
