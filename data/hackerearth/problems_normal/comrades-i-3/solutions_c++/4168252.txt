#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mset(m,v) memset(m,v,sizeof(m))
#define pb push_back
#define mp make_pair
#define fi first
#define se second
#define pi 3.1415926535897932384
#define mod 1000000009
#define MAXM 100001
 
int a[MAXM],h[MAXM],in[MAXM],out[MAXM];
vector<int> v[MAXM];
int ans=0,p=0,TIME=0;
 
int dfs(int x,int ans)
{
    h[x]=ans;
    in[x]=++TIME;
    for(int i=0;i<v[x].size();i++)
    {
        dfs(v[x][i],ans+1);
    }
    out[x]=++TIME;
}
 
void init()
{
    for(int i=0;i<MAXM;i++){
        v[i].clear();
    }
}
 
int main()
{
    int t,n,i,j,k,m,x,y,q;
    cin>>t;
    while(t--)
    {
        init();
        cin>>n;
        for(i=1;i<=n;i++){
            cin>>x;
            v[x].pb(i);
        }
 
        TIME=1;
        dfs(0,1);
        cin>>q;
        while(q--)
        {
            ans=0;p=0;
            cin>>x>>y;
            if(in[y]<in[x] && out[y]>out[x]){
                cout<<h[x]-h[y]-1<<endl;
            }
            else{
                cout<<-1<<endl;
            }
        }
    }
    return 0;
}
