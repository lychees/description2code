#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char s1[100000];
    	char s2[100000];
    	cin>>s1>>s2;
    	int len1=strlen(s1);
    	int len2=strlen(s2);
    	int flag=1;
    	int count=0;
    	if(len1>len2)
    	{
    		flag=0;
    	for(int i=0;i<len2;i++)
    	{
    		if(s1[i]!=s2[i])
    	     {
    	     	flag=0;
    	     	count++;
    	     }
    	     
    	}
    	count=count+(len1-len2);
    	}
    	else 
    	{
    		if(len2>len1)
    		flag=0;
    		for(int i=0;i<len1;i++)
    	    {
    		if(s1[i]!=s2[i])
    	     {
    	     	flag=0;
    	     	count++;
    	     }
    	    }
    	count=count+(len2-len1);
    	}
    	if(flag==1)
    	cout<<"Exact"<<endl;
    	else
    	cout<<"Non Exact"<<" "<<count<<endl;
    	
    	
    }
    return 0;
}
