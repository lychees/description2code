#include <iostream>
#include <algorithm>
using namespace std;

#define MAX 1000000007
int main()
{
    int t;
    cin >> t;
    while(t--){
    	long long a[3];
    	scanf("%lld%lld%lld",&a[0],&a[1],&a[2]);
//    	cin >> a[0] >> a[1] >> a[2];
    	sort(a, a+3);
    	
    	a[0] = a[0]%MAX;
    	a[1] = a[1]%MAX;
    	a[2] = a[2]%MAX;
    	
    	long long count = 1;
    	count = (a[0] * (a[1]-1))%MAX;
    	count = (count * (a[2]-2))%MAX;
    	
    	cout << count << endl;
    }
    return 0;
}
