#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int h,m,i,x,y,ans=0,temp;
        cin>>h>>m;
        if(h>12)h-=12;
        x=(h*30)+((m*30)/60);
        y=m*6;
        temp=abs(x-y);
        ans=360-temp;
        ans=min(ans,temp);
        cout<<ans<<"\n";
    }
    return 0;
}
