#include<bits/stdc++.h>
using namespace std;
class interval{
public:
   int x;
   int y;
   int d;

};
bool compareInterval(interval i1,interval i2)
{
 return (i1.x<i2.x);
}
interval arr[10000000];
int main()
{

    stack<interval> st;
    int n,i=0;
    cin>>n;
    while(i<n)
    {
        cin>>arr[i].x;
        cin>>arr[i].y;
        cin>>arr[i].d;
        i++;
    }
        sort(arr,arr+n,compareInterval);

        st.push(arr[0]);

        for(int j=1;j<n;j++)
        {
            interval tmp = st.top();


            if((tmp.x+tmp.d)< arr[j].x)
            {
                st.push(arr[j]);
            }
            else if((tmp.x+tmp.d) < (arr[j].x+arr[j].d))
            {
                    st.pop();
                    tmp.d = (arr[j].x+arr[j].d)-tmp.x;
                    st.push(tmp);
            }
        }
        int res = 0;
        while(!st.empty())
        {
           // cout<<st.top().x<<" "<<st.top().d+1<<" ";
            res += st.top().d+1;
            st.pop();
        }
        cout<<res;
      return 0;
    }
