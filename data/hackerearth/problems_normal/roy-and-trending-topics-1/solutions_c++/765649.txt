#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
struct pr{
	ll id,old,ne,ch;
};
typedef struct pr P;
P a[1000005];
int cmp(P a, P b){
	if(a.ch >b.ch)return 1;
	if(a.ch<b.ch)return 0;
if(a.id>b.id)return 1;
return 0;
}
int main()
{
  int n;
  scanf("%d",&n);
  int i;
  P cur;
  for(i=0;i<n;i++)
  {
  	ll u,v,w,x,y,z;
  	scanf("%lld%lld%lld%lld%lld%ldd",&u,&v,&w,&x,&y,&z);
  	cur.id = u;
  	cur.old = v;
  	cur.ne = w*50 + x*5 + y*10 + z*20;
  
  	cur.ch = cur.ne - cur.old;
  //		printf("%lld has %lld old and %lld new and change is %lld\n",cur.id,cur.old,cur.ne,cur.ch);
  	a[i] = cur;
  }
  sort(a,a+n,cmp);
  //reverse(a,a+n);
  for(i=0;i<5;i++)
  printf("%lld %lld\n",a[i].id,a[i].ne);
    return 0;
}
