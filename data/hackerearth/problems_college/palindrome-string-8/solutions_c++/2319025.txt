#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	char str[102];
    	cin>>str;
    	int count[256]={0};
    	int i,odd=0;
    	for(i=0;i<strlen(str);i++)
    	{
    		count[str[i]]++;
    	}
    	for(i=0;i<256;i++)
    	{
    		if(count[i]&1)
    		odd++;
    	}
    	if(odd<=1)
    	cout<<"Yes\n";
    	else
    	cout<<"No\n";
    }
    return 0;
}
