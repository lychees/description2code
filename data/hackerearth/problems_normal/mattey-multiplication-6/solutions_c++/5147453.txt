#define MAX 105
using namespace std;
typedef long long int lli;
typedef unsigned long long int ulli;
void Binary(lli n, vector<int> &Arr)
{
    if(!n)
        return;
    int m = n % 2;
    Binary(n / 2, Arr);
    Arr.push_back(m);
}
int main()
{
    int test;
    cin >> test;
    lli N, M;
    while(test--)
    {
        cin >> N >> M;
        vector<int> Arr;
        Binary(M, Arr);
        int Rec = 0;
        for(int i = 0; i < Arr.size(); i++)
        {
            if(Arr[i] && !Rec)
            {
                cout << '(' << N << "<<" << Arr.size() - 1 - i << ") ";
                ++Rec;
            }
            else if(Arr[i] && Rec)
                cout << "+ (" << N << "<<" << Arr.size() - 1 - i << ") ";
        }
        cout << endl;
    }
    return 0;
}
