#include <iostream>
#include<string.h>
#include <stdio.h>
#include<stdlib.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,i,j,sum,f;
    char s[100];
    scanf("%d",&t);
    while(t--)
    {
		scanf(" %[^\n]s",&s);
		sum=0;
    	j=0;
    	j=strlen(s);
    	for(i=0;s[i]!='\0';i++)
    	{
    		
    		sum+=s[i];
    	}
    	//printf("%d",j);
    	if((sum/j)%2==0)
    	printf("%s",s);
    	else
    	{
    		for(i=j-1;i>=0;i--)
    		printf("%c",s[i]);
    	}
    	
    }
    return 0;
}
