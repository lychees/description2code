#include<iostream>
#include<cstring>
using namespace std;
#define ll long long
char str[100005];
ll one(ll l)
{
	ll i,ans=0;
	for(i=0;i<l;i++)
	{
		if(str[i]=='1')
			ans++;
	}
	return ans;
}
ll zero(ll l)
{
	ll i,ans=0;
	for(i=0;i<l;i++)
	{
		if(str[i]=='0')
			ans++;
	}
	return ans;
}
int main()
{
	
	int ans,l,i,t,n,tmp;
	cin>>t;
	while(t--)
	{
		cin>>str;
		cin>>n;
		l=strlen(str);
		if(n==0)
		{
			cout<<one(l)<<endl;
			continue;
		}
		if(n&1)
			tmp=n;
		else
			tmp=n-1;
		for(i=1;i<=tmp;i+=4)
		{
			if(i==tmp)
				break;
		}
		if(i==tmp)
			cout<<zero(l)<<endl;
		else
			cout<<one(l)<<endl;
	}
	return 0;
}
