#include<bits/stdc++.h>
#define mod 1000000007
#define ll long long int
using namespace std;
int a[1000001];
vector<int> p;
void seive(){
	memset(a,0,sizeof a);
    a[1]=1;
	for(int i=2; i<1000001; i++){
		if(!a[i]){
			p.push_back(i);
			for(int j=2; i*j<1000001; j++){
				a[i*j]=1;
			}
		}	
	}
}
int main(){
    seive();
	int sum[1000001];
	int count[1000001];
	sum[0]=0; count[0]=0;
	for(int i=1; i<1000001; i++){
		sum[i]=sum[i-1];
        //cout<<a[i]<<" "<<i<<endl;
		count[i]=count[i-1];
		if(a[i]==0) sum[i]+=i;
		if(a[i]==0) count[i]++;
		sum[i]%=mod;
		count[i]%=mod;
	}
		
	int t; cin>>t;
	while(t--){
		int n; cin>>n;
		cout<<count[n]<<" "<<sum[n]<<" ";
        int sz=p.size();
        //cout<<sz<<endl;
        int i=n;
		ll fact=1;
		for(int j=0; j<sz; j++){
			ll prime=p[j];
			int val=prime;
			ll count=0;
			if(prime>i) break;
			else{
				while(i>=prime){
					count = count + i/prime;
					prime*=val;
				}
			} 
			count++; count--; count++;
			fact*=count;
			fact%=mod;
        }
        cout<<fact<<endl;
    }
}
