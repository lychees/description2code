LL getpow(LL a,LL b,LL mod){
    LL ret=1LL;
    LL inter=a;
    while(b){
        if(b&1){
            ret *= inter;
            ret %= mod;
        }
        inter *= inter;
        inter %= mod;
        b = b>>1;
    }
    return ret;
}
void solve()
{
    LL n,p;
    scanf("%lld %lld",&n,&p);
    if(3*n-p>=0){
        cout<<0<<endl;
        return;
    }
    LL ans1 = getpow(6,n,p);
    LL ans2 = getpow(ans1,p-2,p);
    LL ans = 1;
    for(LL i = 3*n+1;i<p;i++)
    {
        ans *= i;
        if(ans>=p)ans%=p;
    }
    ans = getpow(ans,p-2,p);
    ans *= p-1;
    ans%=p;
    ans*=ans2;
    ans%=p;
    printf("%lld\n",ans);

}

{

    solve();
