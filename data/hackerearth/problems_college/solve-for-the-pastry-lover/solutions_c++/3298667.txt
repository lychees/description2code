#include <iostream>
#include <cmath>
using namespace std;
int main()
{
	 long long int n,t,i,j,m,min;
	char *arr,key;
	cin >> t;
	while(t--)
	{
		min = 1000000000000000;
		cin >> n;
		arr = new char[n];
		for(i=0;i<n;i++)
			cin >> arr[i];
		for(i=0;i<n;i++)
		{
			m = 1000000000000000;
			key = arr[i];
			for(j=i+1;j<n;j++)
			{
				if(key == arr[j])
				{
					if((j-i) < m)
						m = j-i;
				}					
			}
			if(min > m)
				min = m;
		}
	if(min == 1000000000000000)
		min = -1;
	cout << min << "\n";
	}
	
}
