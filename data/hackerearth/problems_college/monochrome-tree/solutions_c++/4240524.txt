ll modx(ll Base, ll exponent)
{
	ll ans = 1;
	if(Base == 1)
		return Base;
	while(exponent)
	{
		if(exponent & 1)
			ans = (ans * Base)%mod;
		Base = (Base * Base)%mod;
		exponent = exponent >> 1;
	}
	return ans;
}
	
ll inmodx(ll num)
{
	return (modx(num, mod-2LL));
}

int init[N], final[N];
bool visited[N];
int s[N], antis[N];
vi v[N];

void dfs(int curr)
{
	visited[curr] = 1;
	int i, xtr = 0, xfa = 0;
	fo(i,0,v[curr].size())
	{
		if(!visited[v[curr][i]])
		{
			dfs(v[curr][i]);
			xtr += s[v[curr][i]];
			xfa += antis[v[curr][i]];
		}
	}
	if(init[curr] == final[curr])
	{
		s[curr] = min(xtr, 2+xfa);
		antis[curr] = min(1+xtr, 1+xfa);
	}
	else
	{
		antis[curr] = min(xfa, 2+xtr);
		s[curr] = min(1+xtr, 1+xfa);
	}
}
