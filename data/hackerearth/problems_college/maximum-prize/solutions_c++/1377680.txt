#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    unsigned long long int t,n,k,a[1000000],sum,i,x;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	for(i=0;i<n;i++)
    	{cin>>a[i];}
    	sort(a,a+n,greater<int>());
    	x = min(n,k);
    	sum=0;
    	for(i=0;i<x;i++)
    	{sum=sum+a[i];}
    	cout<<sum<<endl;
    }
    return 0;
}
