#include<iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    string book[2000];
    int a[26];
    memset(a,0,sizeof(a));
    for(int i=0;i<n;i++)
    {
        cin>>book[i];
        a[book[i][0]-97]++;
    }
    int c=0,temp=0,pow,val;
    for(int i=0;i<26;i++)
    {
        temp=a[i];
        if(temp<=10 && temp != 0)
        {
            c++;
        }
        else
        {
            val=temp/10;
            pow = val*10;
            if(a[i]>pow)
            {
                c += val+1;
            }
            else
            {
                c += val;
            }
        }
    }
    cout<<c<<endl;

    return 0;
}
