#include<bits/stdc++.h>

using namespace std;
bool p[510];
long long q[510];
int main()
{
	int n;cin>>n;
	p[1]=true;
	for(int i=2;i<=22;i++) {
		if(p[i]) continue;
		for(int j=i;i*j<=500;j++) p[j*i]=true;
	}
	//for(int i=2;i<=8;i++) cout<<p[i];
	long long aa = 1;q[2]=1;
	for(int i=3;i<=500;i++) {
		if(!p[i]) aa*=2;
		q[i]=q[i-1]+aa;
	}
	//if(!p[i]) q[i]=q[i-1]+1;else q[i]=q[i-1];
	cout<<q[n];
}
