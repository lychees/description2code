#include<iostream>
#include<math.h>
#include<stdlib.h>
#include<stdio.h>
#define MOD 1000000007
using namespace std;
int main()
{
	 int t;
	 cin>>t;
	 while(t--)
	 {
	 	int n,flag=0;
	 	cin>>n;
	 	if(n==1)
	 	{
	 		cout<<"1"<<endl;
	 		continue;
	 	}
	 	if(n==3)
	 	{
	 		cout<<"3"<<endl;
	 		continue;
	 	}
	 	for(int i=2;i<=sqrt(n);i++)
	 	{
	 		if(n%i==0)
	 		{
	 			flag=1;
	 			break;
	 		}
	 	}
	 	if(flag==0)
	 	cout<<"2"<<endl;
	 	else
	 	cout<<"1"<<endl;
	 }
}
