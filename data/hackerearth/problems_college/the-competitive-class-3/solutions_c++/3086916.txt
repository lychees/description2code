#include<iostream>
using namespace std;
int main()
{
    int N,i,j,count;
    cin>>N;
    int a[N];
    for(i=0;i<N;i++)
        cin>>a[i];
    for(i=0;i<N;i++)
    {
        count=1;
        for(j=0;j<N;j++)
        {
            if(a[i]<a[j])
                count++;
        }
        cout<<count<<" ";
    }
}
