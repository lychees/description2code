#include <iostream>
#include<cstring>
using namespace std;

int main()
{
   char str[101];
    int t,len,i,flag;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%s",str);
    	len=strlen(str);
    	flag=0;
    	for(i=0;i<len;i++)
    	{
    		if(str[i]==str[i+1])
    		flag=1;
    	}
    	if(flag==0)
    	printf("KISS\n");
    	else
    	printf("SLAP\n");
    }
    return 0;
}
