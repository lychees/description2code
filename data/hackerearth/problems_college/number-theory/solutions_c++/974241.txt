#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,x,j;
    cin>>t;
    while(t--){
 		int a[10] = {0};
    	cin>>n;
    	x = n;
    	while(x > 0){
    		a[x%10]++;
    		x = x/10;
    	}
    	int c = 0;
    	for(int i = 2; i <= sqrt(n); i++){
    		if(n % i == 0){
    			j = i;
    			int flagi = 0;
    			while(j > 0){
    				if(a[j % 10] > 0){
    					flagi = 1;
    					break;
    				}
    				j = j / 10;
    			}
    			j = n/i;
    			int flagni = 0;
    			while(j > 0){
    				if(a[j % 10] > 0){
    					flagni = 1;
    					break;
    				}
    				j = j / 10;
    			}
    			if(flagi == 1)
    				c++;
    			if(flagni == 1 && n != i * i)
    				c++;
    		}
    	}
    	if(n == 1)
    		cout<<1<<endl;
    	else if(a[1] > 0)
    		cout<<c + 2<<endl;
    	else
    		cout<<c + 1<<endl;
    }
    return 0;
}
