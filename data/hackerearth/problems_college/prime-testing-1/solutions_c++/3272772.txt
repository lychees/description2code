#include <iostream>
using namespace std;

int main() {
	int t,n,count,x,ar[1001]={0},i,j;
	ar[0]=ar[1]=1;
	for(i=2;i<=1000;i++){
		if(ar[i]==0){
			for(j=2;i*j<=1000;j++){
				ar[i*j]=1;
			}
		}
	}
	cin>>t;
	while(t--){
		cin>>n;
		count=0;
		while(n--){
			cin>>x;
			if(ar[x]==0)
			count++;
		}
		cout<<count<<endl;
	}
	// your code goes here
	return 0;
}
