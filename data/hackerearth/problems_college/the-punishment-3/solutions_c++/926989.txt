#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int t,N,K;
	int D;
	int A,B,x;
    cin >> t;
    vector<int> ans;
    
    while(t--){
    	cin >> N >> K;
    	D = N - K;
    	ans.clear();
    	if(K == D){
    		ans.push_back(1);
    		for(int i = N - 1;i > K;i--)
    			ans.push_back(0);
    		ans.push_back(8);
    		for(int i = K-1;i > 0;i--)
    			ans.push_back(9);
    	}else{
    		A = K;
    		B = N - K;
    		if(A > B){
    			x = A;
    			A = B;
    			B = x;
    		}
    		ans.push_back(1);
    		for(int i = N - 1;i > B;i--)
    			ans.push_back(0);
    		ans.push_back(9);
    		for(int i = B-1;i > A;i--)
    			ans.push_back(9);
    		ans.push_back(8);
    		for(int i = A-1;i > 0;i--)
    			ans.push_back(9);
    	}
    	for(vector<int>::reverse_iterator itr = ans.rbegin();itr != ans.rend();itr++)
    		cout << *itr;
    	cout << endl;
    }
    return 0;
}
