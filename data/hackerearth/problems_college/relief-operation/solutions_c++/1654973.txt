#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <string>

#define PI  3.14159
using namespace std;
long long int knapSack(int W, int wt[], int val[], int n)
{
   int i, w;
   int K[n+1][W+1];
 
   // Build table K[][] in bottom up manner
   for (i = 0; i <= n; i++)
   {
       for (w = 0; w <= W; w++)
       {
           if (i==0 || w==0)
               K[i][w] = 0;
           else if (wt[i-1] <= w)
                 K[i][w] = max(val[i-1] + K[i-1][w-wt[i-1]],  K[i-1][w]);
           else
                 K[i][w] = K[i-1][w];
       }
   }
 
   return K[n][W];
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t-->0)
	{
		int n,w;
		scanf("%d%d",&n,&w);
		int wt[n];int val[n];
		for(int i=0;i<n;i++)
			scanf("%d",&wt[i]);
		for(int i=0;i<n;i++)
			scanf("%d",&val[i]);
		long long int ans=knapSack(w,wt,val,n);
		printf("%lld\n",ans );
	}
	return 0;
}
