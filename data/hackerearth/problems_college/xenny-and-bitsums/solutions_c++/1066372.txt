#include <iostream>
using namespace std;

long long f(long long a){
	long long r=a;
	long long pow[40];
	int tp[40];
	int q=0;
	pow[0]=1;
	for(int i=1;i<40;i++){
		pow[i]=2*pow[i-1];
	}
	while(r!=0){
		tp[q]=(r&1);
		q+=1;
		r/=2;
	}
	long long ans=0;
	int len=q;
	r=a;
	for(int i=(len-1);i>=0;i--){
		if(tp[i]==1){
			ans+=((r-pow[i])+1);
			ans+=(((i*(pow[i]))/2));
			r-=pow[i];
		}
	}
	return ans;
}

int main(){
	int t;
	long long x,y;
	cin>>t;
	while(t--){
		cin>>x>>y;
		if(x==1){
			cout<<f(y)<<"\n";
		}else{
			cout<<(f(y)-f(x-1))<<"\n";
		}
	}
	return 0;
}
