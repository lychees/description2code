#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int t;
    cin>>t;
    while(t--)
    {
    	int l,k;
    	cin>>l>>k;
    	if(l<n||k<n)
    		cout<<"UPLOAD ANOTHER"<<endl;
    	else
    	{
    		if(l==k)cout<<"ACCEPTED"<<endl;
    		else cout<<"CROP IT"<<endl;
    	}
    }
    return 0;
}
