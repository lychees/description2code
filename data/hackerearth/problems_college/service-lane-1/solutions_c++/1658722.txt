#include<iostream>

using namespace std;

int main (){
	int n,t,min=4;
	int entry,ex;
	cin>>n;
	cin>>t;
	int w[n];
	for(int i=0;i<n;i++)
		cin>>w[i];
	while(t)
	{
		min=4;
		cin>>entry;
		cin>>ex;
		for(int j=entry;j<=ex;j++)
		{
			if(w[j]<min)
				min=w[j];
		}	
		if(min==1)
			cout<<"1"<<endl;
		else if(min==2)
			cout<<"2"<<endl;
		else if(min==3)
			cout<<"3"<<endl;
		else
			cout<<"null";
        t--;
     }	    
}
