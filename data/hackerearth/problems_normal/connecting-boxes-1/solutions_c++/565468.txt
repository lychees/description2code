#include<iostream>
#include<algorithm>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
    int n,q,arr[1005][2],k=0;
    scanf("%d%d",&n,&q);
    while(q--)
    {
        int a,b,flag=0;
        scanf("%d%d",&a,&b);
        for(int i=0;i<k;i++)
        {
            int ss1=0,ss2=10;
            if(a<arr[i][0]&&a>arr[i][1])
                ss1=1;
            if(a>arr[i][0]||a<arr[i][1])
                ss1=-1;
        //----------------------------------------
            if(b<arr[i][0]&&b>arr[i][1])
                ss2=1;
            if(b>arr[i][0]||b<arr[i][1])
                ss2=-1;
        //--------------------------------------
        if(ss1!=ss2)
        {
                flag=1;
                break;
        }
    }
     if(a==b||flag)
     {
         printf("NO\n");
     }
     else
     {
        arr[k][0]=max(a,b),arr[k++][1]=min(a,b);
        printf("YES\n");
     }
    }
}
