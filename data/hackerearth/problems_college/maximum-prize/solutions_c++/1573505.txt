#include<iostream>
#include<cstdio>
#define ll long long
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
ll arr[100061];
bool cmp(const ll &a,const ll &b)
{
	return a>b;
}
int main()
{
	ll t,n,k,res,i;
	scanf("%lld",&t);
	while(t--)
	{
		res=0;
		scanf("%lld%lld",&n,&k);
		for(i=0;i<n;i++)
			scanf("%lld",&arr[i]);
		sort(arr,arr+n,cmp);
		for(i=0;i<min(n,k);i++)
			res+=arr[i];
		printf("%lld\n",res);
	}
	return 0;
}
