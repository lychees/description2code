#include <iostream>
using namespace std;

int power(int a,int n)
{
    int temp;
    if(n==0)
    return 1;
    if(n==1)
    return a;
    temp = power(a,n/2);
    if(n%2==0)
    return temp*temp;
    else
    return temp*temp*a;
}

int main()
{
	int N,i,j,a,b,result,c;
	cin >> N;
	a = power(2,N);
	int arr[a-1][a-1];
	for(i=0;i<a-1;i++)
		for(j=0;j<=i;j++)
			cin >> arr[i][j];
	b = a;
	int arr1[b];
	for(j=0;j<b;j++)
		arr1[j] = j+1;
	for(i=0;i<N;i++)
		arr1[j]=j+1;
	for(i=0;i<N;i++)
	{
		int RESULT[b];
		for(j=0;j<b/2;j++)
		{
			c = arr[arr1[2*j+1]-2][arr1[2*j]-1];
			if(c==0)
				RESULT[j] = arr1[2*j];
			else
				RESULT[j] = arr1[2*j+1];
		}
		for(j=0;j<b/2;j++)
			arr1[j] = RESULT[j];
		if(b/2==1)
			result = RESULT[0];
		b = b/2;
	}
	cout << result <<endl;
	return 0;
}
