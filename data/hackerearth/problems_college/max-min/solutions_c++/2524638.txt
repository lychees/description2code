#include <bits/stdc++.h>
using namespace std;


int main() {
	int n,k;
	cin >> n >> k ;
	int arr[n];
	for (int i = 0; i < n; ++i)
	{
		cin >> arr[i];
	}
	sort(arr, arr+n);
	int min = 1000000004;
	for (int i = k-1; i < n; ++i)
	{
		int c = arr[i] - arr[i-k+1];
		if( c< min){
			min = c;
		}
	}
	cout << min << endl;
 	return 0;
}
