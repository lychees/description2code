#include<bits/stdc++.h>
using namespace std;
#define ll long long
vector<ll>v,ar;
int main()
{
    ll i,k=0,x,j,num,n,n1=2300000,z;
    for(i=0;i<n1;i++)
    v.push_back(0);
    for(i=2;i<n1;i++)
    {
        if(v[i-1]==0)
        {
            ar.push_back(i);
            x=2;
            while((z=i*x)<=n1)
            {
                v.at(z-1)=1;
                x++;
            }
            //cout<<ar[k]<<" ";
            //k++;
        }
    }
    cin>>num;
    cout<<ar[num-1]<<endl;
    return 0;
}
