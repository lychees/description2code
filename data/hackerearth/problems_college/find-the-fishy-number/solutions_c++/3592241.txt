int f(vector <int> t, int s) {
		vector<int> a(10);
		for(int i = 0; i < t.size(); i++) {
			int p = t[i], r = 0;
			while(p) {
				r = p % 10;
				p /= 10;
			}
			a[r]++;
		}
		int ans = -1;
		for(int i = 1; ans == -1 && i < 10; i++) {
			double e = log(1 + 1.0 / i) / log(10.0) * t.size();
			if(e * s < a[i] || e > a[i] * s)
				ans = i;
		}
		return ans;
	}
