t = input()
l = [[1, 0]]
o = []
for i in range(0, t):
    n = input()
    if n >= len(l):
        for j in range(len(l), n+1):
            zero = l[len(l)-1][0]+l[len(l)-1][1]
            one = l[len(l)-1][1]
            l.append([one, zero])
    o.append(l[n])

for i in o:
    s = str(i[0])+" "+str(i[1])
    print s
