#include <iostream>
#include<string.h>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,k;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	int arr[n];
    	int sum=0;
    	for(int i=0;i<n;i++)
    	{
    	cin>>arr[i];
    	sum+=arr[i];
    	}
    	
    	sort(arr,arr+n);
    	
    	
    	
    	int dp[sum+2];
    	memset(dp,0,sizeof(dp));
    	dp[0]=1;
     	for(int i=0; i<n; i++)
  		for(int j=sum; j>=arr[i]; j--)
		dp[j] |= dp[j-arr[i]];
		
		
		
    	if(dp[k])
    	cout<<"Yes"<<endl;
    	else
    	{
    		int i,j;
    		i=j=k;
    		while(dp[i]==0)
    		i--;
    		while(dp[j]==0)
    		j++;
    		if(j-k<=k-i)
    		cout<<"No"<<endl<<j<<endl;
    		else
    		cout<<"No"<<endl<<i<<endl;
    	}
    	
    }
    return 0;
}
