#include <iostream>
#include<algorithm>
using namespace std;



int main()
{
	int n,m;
	cin>>n;
	int a[n];
	int i=0;
	while(i<n)
{	cin>>a[i];
i++;
}
cin>>m;
int b;i=0;
while (i<m){
	cin>>b;i++;
	for(int j=0;j<n;j++){
		if(a[j]>b)
		a[j]--;
	}
}


for(i=0;i<n;i++){
	
cout<<	a[i]<<" ";
}


	
}
