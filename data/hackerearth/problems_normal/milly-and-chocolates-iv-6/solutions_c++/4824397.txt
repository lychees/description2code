#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,k,m;
    	cin>>n>>k>>m;
    	long long int cost[n],arr[n];
    	long long int temp;
    	for(int i=0;i<n;i++)
    	{
    		cin>>temp;
    		arr[i]=temp;
    		cost[i]=(temp)*k+(temp-1)*m;
    	}
    	long long int ans=LLONG_MAX,ind=-1;
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<arr[i];j++)
    		{
    			cin>>temp;
    			cost[i]+=temp;
    		}
    		
    		if(ans>cost[i])
    		{
    			ind=i+1;
    			ans=cost[i];
    		}
    	}
    	cout<<ind<<" "<<ans<<"\n";
    }
    return 0;
}
