#include<bits/stdc++.h>
using namespace std;
int n,coo,ceo,cto,a[111];
int dp[1<<21][21],dp1[1<<21][121];
int rec1(int mask,int id,int c)
{

    if(mask==(1<<(n))-1)
    {
        dp1[mask][id]=abs(cto-a[id]);
        return dp1[mask][id];
    }
    int j=0;
    for(int i=0;i<n;i++)
        if(mask&(1<<i))
            j++;
    if(j==n/2+1&&n%2==0&&c==0)
    {
        dp1[mask][id]=abs(a[id]-a[n])+rec1(mask,n,1);
        return dp1[mask][id];
    }
    if(dp1[mask][id]!=-1)
        return dp1[mask][id];
    dp1[mask][id]=1000000000;
    for(int i=0;i<n;i++)
    {
        if(mask&(1<<i))
            continue;
        if(mask==0)
            dp1[mask][id]=min(dp1[mask][id],abs(a[i]-coo)+rec1(mask|(1<<i),i,c));
        else
            dp1[mask][id]=min(dp1[mask][id],abs(a[i]-a[id])+rec1(mask|(1<<i),i,c));
    }
    //printf("%d %d %d\n",mask,id,dp1[mask][id]);
    return dp1[mask][id];
}
int rec(int mask,int id,int c)
{

    if(mask==(1<<(n))-1)
    {
        dp[mask][id]=abs(cto-a[id]);
        return dp[mask][id];
    }
    int j=0;
    for(int i=0;i<n;i++)
        if(mask&(1<<i))
            j++;
    if(j==n/2&&c==0)
    {
        dp[mask][id]=abs(a[id]-a[n])+rec(mask,n,1);
        return dp[mask][id];
    }
    if(dp[mask][id]!=-1)
        return dp[mask][id];
    dp[mask][id]=1000000000;
    for(int i=0;i<n;i++)
    {
        if(mask&(1<<i))
            continue;
        if(mask==0)
            dp[mask][id]=min(dp[mask][id],abs(a[i]-coo)+rec(mask|(1<<i),i,c));
        else
            dp[mask][id]=min(dp[mask][id],abs(a[i]-a[id])+rec(mask|(1<<i),i,c));
    }
    //printf("%d %d %d\n",mask,id,dp[mask][id]);
    return dp[mask][id];
}
int main()
{
    int i;
    scanf("%d %d %d %d",&ceo,&cto,&coo,&n);
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    a[n]=ceo;
    for(i=0;i<(1<<n);i++)
        for(int j=0;j<=n;j++)
            dp[i][j]=dp1[i][j]=-1;
    if(n%2==1)
        printf("%d\n",min(rec(0,0,0),rec1(0,0,0)));
    else
        printf("%d\n",rec(0,0,0));
    return 0;
}
