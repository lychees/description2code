#include <bits/stdc++.h>
using namespace std;
long int a[15],s;
int n;

bool ans()
{
    long long int sum[1<<16];
    sum[0] = 0;
    for(int m=1;m<(1<<n);m++)
    {
        // for each mask we calculate sum
        for(int i=0;i<n;i++)
        {
            if(m&(1<<i))
            {
                sum[m] = a[i] + sum[m^(1<<i)];
                if(sum[m]==s)
                    return true;
            }
        }
    }
    return false;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(int i=0;i<n;i++) cin>>a[i];
        cin>>s;
        if(ans())
            cout<<"YES\n";
        else
            cout<<"NO\n";
    }
    return 0;
}
