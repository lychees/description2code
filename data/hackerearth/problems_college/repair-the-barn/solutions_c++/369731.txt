#include <iostream>
#include <cstdio>
#include <algorithm>
#include <utility>
using namespace std;

struct sort_pred {
    bool operator()(const std::pair<int,int> &left, const std::pair<int,int> &right) {
        return left.first > right.first;
    }
};

int main()
{
    int m,s,c,num,A[201]={0};
    int i,j,k,ans;
    pair<int,int> B[200];
    scanf("%d%d%d",&m,&s,&c);
    for(i=0;i<c;i++)
    {
    	scanf("%d",&num);
    	A[num]=1;
    }
    
    ans=201;
    
    i=0;
    while(A[i]!=1)
    	i++,ans--;
    
    j=200;
    while(A[j]!=1)
    	j--,ans--;
    
    k=0;
    for(;i<j;i++)
    {
    	if(A[i]==0)
    	{
    		if(A[i-1])
    		{
    			B[k].first=1;
    			B[k].second=i;
    			k++;
    		}
    		else
    			B[k-1].first++;
    	}
    }
    sort(B,B+k,sort_pred());
    
    m=min(m-1,k);
    i=0;
    while(i<m)
    {
    	ans -=B[i].first;
    	//printf("%d %d\n",B[i].first,B[i].second);
    	i++;
    }
    printf("%d",ans);
    return 0;
}
