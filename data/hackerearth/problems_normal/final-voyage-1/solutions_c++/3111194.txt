#include "bits/stdc++.h"

typedef long long ll;

#define FF first

#define s(n) scanf("%d",&n)
#define rep(i,start,end) for(int i = start;i<end;i++)

int DP[1005][1005];
int wt[1005], val[1005];

void build(int n, int m){
    rep(i,0,n+3) DP[0][i] = DP[i][0] =0;
    rep(i,1,m+1){
        int j = 1;
        while(j<n+1 and j<wt[i-1]) DP[i][j] = DP[i-1][j], j++;
        while(j<n+1){
            DP[i][j] = max(DP[i-1][j-wt[i-1]]+val[i-1], DP[i-1][j]);
            j++;
        }
    }
}

int main(){
    int tt;s(tt);
    int load, n;
    while(tt--){
        s(n);s(load);
        rep(i,0,n) cin>>wt[i];
        rep(i,0,n) cin>>val[i];
        build(load, n);
        cout<<DP[n][load]<<endl;
    }
}
