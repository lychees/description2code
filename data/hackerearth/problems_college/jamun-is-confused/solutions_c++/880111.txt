#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,n;
    int a[1001];
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	for(int i =0; i<n; i++)
    	{
    		scanf("%d",&a[i]);
    	}
    	if(n==1){
    		cout<<a[0]<<endl;
    		continue;
    	}
    	int min = __gcd(a[0],a[1]);
    	for(int i = 2; i<n; i++)
    	{
    		min=__gcd(min,a[i]);
    	}
    	printf("%d\n",min);
    }
    return 0;
}
