#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int main(){
    int t;
    scanf("%d",&t);
    while(t--)
    {

	long long int ans,lalit;
	long long int n,a[100];
	scanf("%lld",&n);
	for(long long int i=0;i<n;i++) scanf("%lld",a+i);
	lalit=a[0];
	long long int s=a[0],t=a[0];
	for(long long int i=1;i<n;i++){
	    lalit+=a[i];
	    s=__gcd(s,a[i]);
	    t=max(t,a[i]);
	}
	long long cnt=t/s  - n;
	ans=(   (cnt+n)* ( 2*s + (cnt+n-1)*s   )  )/2;
	if(cnt%2) printf("Joey "); else printf("Chandler ");
	printf("%lld\n",ans-lalit);
    }
    return 0;
}
