#include <bits/stdc++.h>

using namespace std;

#define fi first
#define se second
#define pb push_back
#define mp make_pair


typedef long long ll;
typedef double dbl;
typedef vector<int> vi;
typedef pair<int, int> pi;

#define MAXV 1000000005

class maxbpm {
  int nl,nr;
  int *pr, *nx, *seen;
  const int *f, *a;
  public:
  maxbpm(int nl, int nr, const int *f, const int *a) {
    this->nl=nl;
    this->nr=nr;
    this->f=f;
    this->a=a;

    nx=new int[nl];
    pr=new int[nr];
    seen=new int[nl];
  }

  bool dfs(int u, const int &mx){
    if(u==-1) return true;
    if(seen[u]) return false;
    seen[u]=true;
    for(int v=0;v<nr;v++){
      if(f[v+a[u]]<=mx && dfs(pr[v], mx)){
        pr[v]=u;
        nx[u]=v;
        return true;
      }
    }
    return false;
  }

  bool calc(const int &mx){
    int max=0;
    memset(pr,-1,sizeof(int)*nr);
    memset(nx,-1,sizeof(int)*nl);
    for(int u=0;u<nl;u++){
      memset(seen,0,sizeof(int)*nl);
      if(dfs(u,mx)) max++;
    }
    return max==nl;
  }
};

int main() {
  int n, m;
  int *a, *f;
  scanf("%d %d",&n,&m);
  a=new int[n];
  f=new int[n+m];
  for(auto i=0;i<n;i++){
    scanf("%d",a+i);
  }
  for(auto i=0;i<n+m-1;i++){
    scanf("%d",f+i);
  }
  maxbpm bpm(n,n,f,a);
  int lo=0, hi=MAXV, md;
  while(lo+1<hi){
    md=(lo+hi)/2;
    if(bpm.calc(md))hi=md;
    else lo=md;
  }
  cout<<hi<<endl;
  return 0;
}
