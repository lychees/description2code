#include<iostream>
#include<bits/stdc++.h>
using namespace std;
main()
{
	long long n;
	cin>>n;
	long long a[n];
	for(int i=0;i<n;i++)
		cin>>a[i];
	long long curs=a[0],maxs=INT_MIN;
	for(int i=1;i<n;i++)
	{
		curs=max(a[i],curs+a[i]);
		maxs=max(maxs,curs);
	}
	if(maxs)
		cout<<maxs<<endl;
	else
		cout<<"0\n";
}
