#include<bits/stdc++.h>
#define pii pair<long long,long long>
#define fi first
#define se second
#define mp make_pair
#define vpi vector< pii >
#define pb push_back
#define ll long long int
#define mod 1000000007
#define gcd(a,b) __gcd(a,b)
#define sf(n) scanf("%lld",&(n))
#define pf(n) printf("%lld\n",(n))
#define setbits(n) __builtin_popcount((n))
#define pq priority_queue
using namespace std;
int dist[10000005]={0};
bool visited[10000005]={0};
int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n;
    cin>>n;
    string s;
    cin>>s;
    int mask=0;
    for(int i=0;i<n;i++)
    {
        if(s[i]=='B')
            mask=mask|(1<<i);
    }
    queue<int>q;
    q.push(mask);
    while(!q.empty())
    {
        int k=q.front();
        //cout<<k<<endl;
        if((k==(1<<n)-1)||(k==0))
        {
            cout<<dist[k]<<endl;
            return 0;
        }
        q.pop();
        for(int i=0;i<n-1;i++)
        {
            int t=k^(1<<i)^(1<<(i+1));
            if(visited[t]==false)
            {
                visited[t]=true;
                dist[t]=dist[k]+1;
                q.push(t);
            }
        }
        for(int i=0;i<n-2;i++)
        {
            int t=k^(1<<i)^(1<<(i+1))^(1<<(i+2));
            if(visited[t]==false)
            {
                visited[t]=true;
                dist[t]=dist[k]+1;
                q.push(t);
            }
        }
    }


}
