#include <bits/stdc++.h>

using namespace std;

int arr[1001];

int main()
{
	int t, n, k, c;
	cin >> t;
	while (t--)
	{
		c = 0;
		cin >> n >> k;
		for (int i = 0; i < n; i++)
		{
			cin >> arr[i];
			if (arr[i] <= 0)
				c++;
		}
		if (c >= k)
			cout << "NO" << endl;
		else
			cout << "YES" << endl;
	}
	return 0;
}
