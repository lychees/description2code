#include <iostream>
#include<cstdio>
#include<utility>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
#define sd(x) scanf("%d",&x)
#define sld(x) scanf("%lld",&x)
#define fill(a,v) memset(a,v,sizeof(a))
#define ll long long
int main()
{
int t;
sd(t);
while(t--)
{
int n,m;
sd(n);
sd(m);
int a[105][105];
int count=0;
int flag=0;
char str[105][105];
for(int i=0;i<n;i++)
{cin>>str[i];
//cout<<str[i]<<endl;
}
for(int i=0;i<n;i++)
{
for(int j=0;j<m;j++)
{
a[i][j]=(int)(str[i][j]-'0');
//cout<<a[i][j]<<" ";
}
//cout<<endl;
}
for(int i=0;i<n;i++)
{
for(int j=0;j<m;j++)
{
if(i+1<n)
{if(a[i][j]&&a[i+1][j])
{flag=1;
//cout<<"hello"<<endl;
break;
}
}
if(i-1>=0)
{if(a[i][j]&&a[i-1][j])
{flag=1;
break;
}
}
if(j+1<m)
{if(a[i][j]&&a[i][j+1])
{flag=1;
break;
}
}
if(j-1>=0)
{if(a[i][j]&&a[i][j-1])
{flag=1;
break;
}
}
if(a[i][j]==0)
count++;
}
if(flag)
break;
}
//cout<<"flag="<<flag<<endl;
if(flag)
{cout<<"yes"<<" ";
cout<<"0"<<endl;
}
else
{
if(count==m*n&&(m*n!=1))
{
cout<<"no"<<" ";
cout<<"2"<<endl;
}
else
{
cout<<"no"<<" ";
cout<<"1"<<endl;
}
}
}
return 0;
}
