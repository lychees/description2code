#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

int main()
{
   string ip;
   cin>>ip;
   int l = ip.length();
   int n;
   cin>>n;
   string dict[n];
   int i,flag;
   for(i=0;i<n;i++)cin>>dict[i];
   sort(dict,dict+n);
   flag = 0;
   for(i=0;i<n;i++)
   {
   		if(ip==dict[i].substr(0,l))
   		{
   			cout<<dict[i];
   			flag = 1;
   			break;
   		}
   }
   
   if(flag==0)
   {
   		cout<<ip;
   }
   return 0;
}
