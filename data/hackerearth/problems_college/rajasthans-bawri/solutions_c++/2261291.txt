#include<bits/stdc++.h>
using namespace std;
int main()
{
    int px,py,sx,sy,t;
    cin>>t;
    while(t--)
    {
        cin>>px>>py>>sx>>sy;
        if(min(px,py)>=min(sx,sy)&&max(px,py)>=max(sx,sy))
            cout<<"Possible"<<endl;
        else
            cout<<"Not Possible"<<endl;
    }
}
