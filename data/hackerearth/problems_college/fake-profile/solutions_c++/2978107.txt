#include<bits/stdc++.h>
using namespace std;
int main()
{
	string str;
	getline(cin,str);
	int i,j;
	int len = str.length();
	int count[26]={0};
	for(i=0;i<len;i++)
	{
		if(str[i]!=' ')
		count[str[i]-'a']=1;
	}
	int c=0;
	for(i=0;i<26;i++)
	{
		if(count[i]==1)
		{
			c++;
		}
	}
	if(c%2==0)
	{
		cout<<"CHAT WITH HER!"<<"\n";
	}
	else
	{
		cout<<"IGNORE HIM!"<<"\n";
	}
}
