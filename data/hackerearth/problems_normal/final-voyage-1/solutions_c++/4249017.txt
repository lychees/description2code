#define fr(i,a,b) for(int i=a;i<b;i++)
#define frn(i,a,b) for(int i=a;i>b;i--)
#define si(x) scanf("%d",&x)
#define sll(x)  scanf("%lld",&x)
#define pln printf("\n")
#define pi(x) printf("%d",x)
#define pll(x) printf("%lld",x)
#define mp make_pair

#define x first
#define y second
typedef long long ll;
ll MOD=1000000007;
ll pw(ll a,ll b){
  ll r=1;
  while(b>0){
     if(b&1){
       r=(r*a)%MOD;
        }
     b=b>>1;
     a=(a*a)%MOD;
  }
return r;
}


int main(){
    int t;
    si(t);

    while(t--){
       int n,w;
       si(n);
       si(w);
       int a[1001][3];
       fr(i,0,n){
          si(a[i][0]);
       }
       fr(i,0,n){
          si(a[i][1]);
       }
       ll dp[1002][10101]={{}};
     fr(i,1,n+1){
         fr(j,1,w+1){

              if(a[i-1][0]<=j){
                 dp[i][j]=max(dp[i-1][j],dp[i-1][j-a[i-1][0]]+a[i-1][1]);
              }
              else{
                dp[i][j]=dp[i-1][j];
              }
         }
     }
     pll(dp[n][w]);pln;
   }

 return 0;
}
