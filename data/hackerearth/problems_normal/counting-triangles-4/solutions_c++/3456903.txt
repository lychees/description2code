#include<iostream>
using namespace std;
int main()
{ long int n,t,sum1,sum2,sum;
   cin>>t;
   while(t>0)
 {
   cin>>n;
   sum1=0; sum2=0;
   for(long int i=1;i<=n;i++)
     sum2=sum2+8;
    for(long int j=1;j<n;j++)
        sum1=sum1+6;
	   
   sum=sum1+sum2;
    cout<<sum<<endl;
 --t;
}
}
