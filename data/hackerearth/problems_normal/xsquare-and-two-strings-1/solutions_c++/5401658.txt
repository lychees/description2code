#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin >> t;
    while(t--){
    	string s1,s2;
    	cin>>s1 >> s2;
    	map<char,int> mp;
    	int l = s1.length();
    	for(int i = 0; i < l;i++){
    		mp[s1[i]]++;
    	}
    	int f = 0;
    	for(int i = 0;i < s2.length(); i++){
    		if(mp[s2[i]] > 0){
    			f = 1;
    			break;
    		}
    	}
    	if(f) cout<< "Yes\n";
    	else cout <<"No\n";
    }
    
    return 0;
}
