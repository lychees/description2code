#include <bits/stdc++.h>
#define ll long long 
using namespace std;
ll n,e;
vector <ll> *v;
int vis[100001];

void dfs(ll x)
{
	vis[x]=1;
	
	for(ll i=0;i<v[x].size();i++)
	{
		if(vis[v[x][i]]==0)
		dfs(v[x][i]);
	}
}
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	cin>>n>>e;
    	ll c=0;
    	v=new vector<ll>[n];
    	memset(vis,0,sizeof(vis));
    	ll i;
    	for(i=0;i<e;i++)
    	{
    		ll x,y;
    		cin>>x>>y;
    		v[x].push_back(y);
    		v[y].push_back(x);
    		
    	}
    	for(i=0;i<n;i++)
    	{
    		if(vis[i]==0)
    		{
    			c++;
    			dfs(i);
    		}
    	}
    	cout<<c<<endl;
    
    }
    
    return 0;
}
