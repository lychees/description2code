#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    unsigned long long int a,b;
    while(t--)
    {
    	cin>>a>>b;
    	if(b%2==0)
    	{
    		if(b-2>=a)
    		cout<<b-2<<endl;
    		else
    		cout<<(a&b)<<endl;
    	}
    	else
    	cout<<b-1<<endl;
    }
    return 0;
}
