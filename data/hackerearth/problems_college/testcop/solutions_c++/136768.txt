#include <iostream>

int gcd(int a, int b)
{
	if ( b == 0 ) return a;
	return gcd(b,a%b);
}

using namespace std;

int main()
{
	int t,n,x;
	cin >> t;
	while ( t-- ) {
		cin >> n;
		int val;
		for ( int i = 0; i < n; i++ ) {
			cin >> x;
			if ( i == 0 ) val = x;
			else val = gcd(val,x);
		}
		if ( val == 1 ) cout << "YES" << endl;
		else cout << "NO" << endl;
	}
    return 0;
}
