#include <bits/stdc++.h>
using namespace std;

const int MaxN = (int) 5e3 + 10;
const long long inf = 1e17;
int n, m, k, a[MaxN];
long long s[MaxN], dp[MaxN][MaxN];

int main() {
    scanf("%d%d%d", &n, &m, &k);
    for (int i = 1; i <= n; ++i) scanf("%d", &a[i]);

    s[0] = 0;
    for (int i = 1; i <= n; ++i) s[i] = s[i - 1] + a[i];

    for (int i = m; i <= n; ++i)
        for (int j = 1; j <= k; ++j)
            dp[i][j] = max(dp[i - 1][j], dp[i - m][j - 1] + s[i] - s[i - m]);

    cout << dp[n][k];
    return 0;
}
