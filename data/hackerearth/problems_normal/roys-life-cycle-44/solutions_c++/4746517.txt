#include<iostream>
#include<cstdio>
#include<limits.h>
#include<algorithm>
using namespace std;
int main(){
	int n;
	cin >> n;
	int streak=0, p =INT_MIN, x = INT_MIN;
	while(n--){
		string str;
		cin >> str;
		int m = INT_MIN;
		int count = 0;
		for(int i=0; str[i] != '\0'; i++){
			if(str[i] == 'C'){
				count++;
				streak++;
			}else{
				count = 0;
				streak = 0;
			}
			m = max(count, m);
			p = max(p, streak);
		}
		x = max(m, x);
	}
	cout << x << " " << p << endl;
	return 0;
}
