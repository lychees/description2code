#include <bits/stdc++.h>
using namespace std;
int Win[1100][1100];
vector<int>Nig,temp;
int main()
{
    int n,sz,val;
    cin>>n;
    sz=1;
    for(int i=1;i<=n;i++)
    sz*=2;
    for(int i=2;i<=sz;i++)
    {
    	for(int j=1;j<i;j++)
    	{
    		cin>>val;
    		//cout<<val<<" ";
    		if(val==1)
    		Win[i][j]=Win[j][i]=i;
    		else
    		Win[i][j]=Win[j][i]=j;
    	}
    	cout<<endl;
    }
    for(int i=1;i<=sz;i++)
    Nig.push_back(i);
    while(Nig.size()!=1)
    {
    	//cout<<Nig.size()<<endl;
    	temp.clear();
    	for(int j=0;j<Nig.size()-1;j+=2)
    	{
    		//cout<<Nig[j]<<"*"<<Nig[j+1]<<"--"<<Win[Nig[j]][Nig[j+1]]<<"\n";
    		if(Win[Nig[j]][Nig[j+1]])
    		temp.push_back(Win[Nig[j]][Nig[j+1]]);
    	}
    	Nig.clear();
    	for(int j=0;j<temp.size();j++)
    	{
    		Nig.push_back(temp[j]);
    	}
    }
    cout<<Nig[0]<<"\n";
    return 0;
}
