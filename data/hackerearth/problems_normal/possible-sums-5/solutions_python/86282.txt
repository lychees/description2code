def possibleSums(n,elems):
		sums = sum(elems)
		dp = [[False] * (sums+1) for _ in xrange(n)]
		dp[0][0] = True
		dp[0][elems[0]] = True
		for i in xrange(1,n):
				for j in xrange(sums+1):
						dp[i][j] = dp[i-1][j] or (dp[i-1][j-elems[i]] if j - elems[i] >= 0 else False)
		return sum(dp[n-1])

def main():
		ntests = int(raw_input())
		for _ in xrange(ntests):
				nelems = int(raw_input())
				elems = map(int,raw_input().split())
				print possibleSums(nelems,elems)

main()
