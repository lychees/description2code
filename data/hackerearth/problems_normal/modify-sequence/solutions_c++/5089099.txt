#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,n;
	cin>>n;
	int a[n];
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0;i<n-1;i++)
	{
		int x=min(a[i],a[i+1]);
		
		a[i]-=x;
		a[i+1]-=x;
		
		
	}
	int flag=0;
	for(i=0;i<n;i++)
	{
		if(a[i]!=0)
		{
			flag=1;
			break;
		}
	}
	if(flag==1)
		cout<<"NO";
	else 
		cout<<"YES";
	return 0;
}
