#include <bits/stdc++.h>

using namespace std;

int n;
int ans = 1e9;
int arr [12 + 10];

void rec(int sum1, int sum2, int sum3, int index){

    if(index == n){

        int maxi = max(sum1, max(sum2, sum3) );
        ans = min(maxi, ans);
        return;
    }

    rec(sum1 + arr[index], sum2, sum3, index + 1);
    rec(sum1, sum2 + arr[index], sum3, index + 1);
    rec(sum1, sum2, sum3 + arr[index], index + 1);

}


int main()
{
    //ios::sync_with_stdio(false);cin.tie(0);


    cin >>     n         ;

    for(int i = 0; i < n; i++) cin >> arr[i];

    rec(0, 0, 0, 0);

    cout << ans << "\n";

    return 0;
}
