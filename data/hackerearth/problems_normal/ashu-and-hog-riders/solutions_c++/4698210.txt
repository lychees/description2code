#include<bits/stdc++.h>
using namespace std;

int n;
int mat[1000][1000],dp[1000][1000],ans = 0 ;
bool visited[1000][1000];

int dfs(int x,int y) {
    if(mat[x][y] == 0 || visited[x][y] == true)
        return 0;
    if(dp[x][y] != -1)
        return dp[x][y];

    //cout<<x<<" "<<y<<endl;
    visited[x][y] = true;
    int t1,t2,t3,t4;
    t1 = t2 = t3 = t4 = 0;
    if(x > 0)
        t1 = dfs(x-1,y);
    if(x != n-1)
        t2 = dfs(x+1,y);
    if(y > 0)
        t3 = dfs(x,y-1);
    if(y != n-1)
        t4 = dfs(x,y+1);

    if(t1 == 0 && t2 == 0 && t3 == 0 && t4 == 0) {
        return dp[x][y] = 1;
    }
        

    return dp[x][y] = 1 + t1 + t2 + t3 + t4;
}

int main() {
    //ios_base :: sync_with_stdio(false);
    cin>>n;
    for(int i=0;i<n;i++) {
        for(int j=0;j<n;j++) {
            cin>>mat[i][j];
            visited[i][j] = false;
            dp[i][j] = -1;
        }
    }
    int ans_temp;
    for(int i=0;i<n;i++) {
        for(int j=0;j<n;j++) {
            if(mat[i][j] == 1 && visited[i][j] == false)
                ans_temp = dfs(i,j);
            if(ans_temp > ans)
                ans = ans_temp;
        }
    }
    /*for(int i=0;i<n;i++) {
        for(int j=0;j<n;j++) {
            cout<<dp[i][j]<< " ";
        }
        cout<<endl;
    }*/
    cout<<ans<<endl;
    return 0;
}
