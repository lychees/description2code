#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t>0)
	{
		t--;
		long long b,w;
		cin>>b>>w;
		long long x,y,z;
		cin>>x>>y>>z;
		long long poss1,poss2,poss3,poss4;
		if(w==0 || b==0)
		{
			if(w==0)
			cout<<b*x<<'\n';
			else
			cout<<w*y<<'\n';
		}
		poss1=b*x + w*y;
		poss2=b*x + w*(x+z);
		poss3=b*(y+z) + w*y;
		if(poss1<poss2 && poss1<poss3)
		cout<<poss1<<'\n';
		else if(poss2<poss3)
		cout<<poss2<<'\n';
		else
		cout<<poss3<<'\n';
	}
	return 0;
}
