#include<iostream>
#include<cstdio>
using namespace std;
int st[1000000];
int top=-1;
int bsearch(int low,int High,int temp){
while ( low <= High )                       // start a binary search for the new number position
        {
                   int mid =  ( low + High )  /  2 ;
                    if  ( temp > st [ mid ] )
                    {
                        low = mid +  1 ;
                    }
                    else
                    {
                        High = mid -  1 ;
                    }

        }
        return low;
}
int main(){
    int t;
    cin>>t;
    while(t--){
int n;
cin>>n;
int a[n];
top=-1;
for(int i=0;i<n;i++){
    cin>>a[i];
}top++;
st[top]=a[0];
for(int i=1;i<n;i++){
if(a[i]>=st[top]){
    top++;
    st[top]=a[i];
}else{int g;
    g=bsearch(0,top,a[i]);
    st[g]=a[i];
}



}
int l;
cin>>l;
l--;
if(top<l){
    cout<<-1<<endl;
}
else{
    cout<<st[l]<<endl;
}


}



}
