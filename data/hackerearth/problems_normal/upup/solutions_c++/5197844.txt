#include <bits/stdc++.h>
using namespace std;

int main()
{
   	string str;
   	getline(cin,str);
   	int l = str.length();
   	int flag=0;
   	for(int i=0;i<l;i++)
   	{
   		if(i==0)
   		{
   			cout<<(char)toupper(str[i]);
   		}
   		else if(str[i]==' ')
   		{
   			flag=1;
   			cout<<str[i];
   		}
   		else if(flag==1)
   		{
   			cout<<(char)toupper(str[i]);
   			flag=0;
   		}
   		else cout<<str[i];
   	}
    return 0;
}
