#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int test,i,j,k,len,ans;
    cin>>test;
    char a[1000007];
    while(test--)
    {
    	ans=0;
    	cin>>a;
    	len=strlen(a);
    	for(i=0;i<len;i++)
    	{
    		ans=ans+(a[i]-a[len-i-1]>=0?a[i]-a[len-i-1]:0);
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
