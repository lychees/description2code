#include <iostream>
#include<map>
using namespace std;

int main() {
	int t, n;
	cin>>t;
	map<int, int> mp;
	while(t--)
	{
		int max = 0, i, j, k = 0;
		cin>>n;
		while(n--)
		{
			cin>>i>>j;
			if(j == 0)
			{
				k = 0;
			}
			else
			{
				if(mp.find(i) == mp.end())
				{
					mp[i] = j;
					k++;
					if(k > max)
					{
						max = k;
					}
				}
			}
		}
		cout << max << endl;
		mp.erase(mp.begin(), mp.end());
		
	}
	// your code goes here
	return 0;
}
