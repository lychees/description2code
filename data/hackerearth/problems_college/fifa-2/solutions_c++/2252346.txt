#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int a[t+2],b[t+2];
    int h[111111],aw[111111];
    for(int i=0;i<111110;i++)
    	h[i]=aw[i]=0;
	for(int i=0;i<t;i++)
    {
    	scanf("%d %d",&a[i],&b[i]);
    	h[a[i]]++;
    	aw[b[i]]++;
    }
    int a1,a2;
    for(int i=0;i<t;i++)
    {
    	a1 = t-1+h[b[i]];
    	//if(h[b[i]]==a[i])
    	//a1--;
    	a2=((t-1)*2)-a1;
    
    	printf("%d %d\n",a1,a2);
    }
    return 0;
}
