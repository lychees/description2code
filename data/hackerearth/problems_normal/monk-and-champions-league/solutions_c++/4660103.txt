#include <iostream>
using namespace std;
#include <queue>
int main()
{
    int N,M;
    int prof = 0;
    priority_queue<int> Q;
    cin >> M>>N;
    for(int i = 1; i <= M;i++){
    	int a;
    	cin >> a;
    	Q.push(a);
    	
    }
    for(int k = 1; k <= N;k++){
    	int dummy = Q.top();
    	prof = prof + Q.top();
    	Q.pop();
    	dummy -- ;
    	Q.push(dummy);
    	
    }
    cout << prof << endl;
    return 0;
}
