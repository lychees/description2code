#include<bits/stdc++.h>
using namespace std;
int a[1002][1002], v[1002][1002];
int count_hogs(int x,int y)
{
	queue<pair<int,int> > q;
	v[x][y] = 1;
	int hogs = 1;
	
	q.push(make_pair(x,y));
	
	while(!q.empty())
	{
		x = q.front().first;
		y = q.front().second;
		q.pop();
		
		for(int i=-1;i<=1;i+=2)
		{
			if(!v[x+i][y] && a[x+i][y])
			{
				q.push(make_pair(x+i,y));
				hogs++;
				v[x+i][y]=1;
			}	
		}
		
		for(int i=-1;i<=1;i+=2)
		{
			if(!v[x][y+i] && a[x][y+i])
			{
				q.push(make_pair(x,y+i));
				hogs++;
				v[x][y+i]=1;
			}	
		}
	}
	return hogs;
}
int main()
{
	int n,i,j;
	cin>>n;
	memset(a,0,sizeof(a));
	memset(v,0,sizeof(a));
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			cin>>a[i][j];
	int maxhogs = 0;
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			if(!v[i][j] && a[i][j])
			{
				maxhogs = max(count_hogs(i,j),maxhogs);
			}
		}
	}
	cout<<maxhogs;
	return 0;
}
