#include <iostream>
#include <algorithm>
using namespace std;
bool comp(int64_t a,int64_t b)
{
	return (a<b);
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int64_t n,d,re=0,i;
    	cin>>n>>d;
    	int64_t a[n];
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	sort(a,a+n,comp);
    	for(i=0;i<n;i++)
    	{
    		if(a[i+1]-a[i]<=d)
    		{
    			a[i+1]=0;a[i]=0;
    			re++;
    		}
    	}
    	cout<<re<<"\n";
    }
    return 0;
}
