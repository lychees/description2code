#include<bits/stdc++.h>
using namespace std;

const long long MOD=1000000007;
long long int dp[2][500001][2];

main()
{
	int n;
	scanf("%d",&n);
	
	int a[n+1],b[n+1];
	
	for(int i=1;i<=n;i++)
	  scanf("%d",&a[i]);
	  
	for(int i=1;i<=n;i++)
	  scanf("%d",&b[i]);
	
	dp[0][0][0]=1;
	
	for(int i=1;i<=n;i++)
	{
		memset(dp[i%2],0,sizeof(dp[i%2]));
		
		for(int j=0;j<=400000;j++)
		{
			for(int k=0;k<2;k++)
			{
		        long long temp_sum=j,sum;
		        
				if(k==1)
				   temp_sum=-j;
				
				sum = temp_sum + a[i];
                dp[i%2][abs(sum)][sum < 0] += dp[(i-1)%2][j][k];
                dp[i%2][abs(sum)][sum < 0] %= MOD;
 
                sum = temp_sum - b[i];
                dp[i%2][abs(sum)][sum < 0] += dp[(i-1)%2][j][k];
                dp[i%2][abs(sum)][sum < 0] %= MOD;
 
                sum = temp_sum + a[i] - b[i];
                dp[i%2][abs(sum)][sum < 0] += dp[(i-1)%2][j][k];
                dp[i%2][abs(sum)][sum < 0] %= MOD;
 
                sum = temp_sum;
                dp[i%2][abs(sum)][sum < 0] += dp[(i-1)%2][j][k];
                dp[i%2][abs(sum)][sum < 0] %= MOD;	
					
			}
		}
	}
	
	int q;
	long long ans=0;
	
	scanf("%d",&q);
	
	for(int i=0;i<=q;i++)
	{
		for(int k=0;k<2;k++ )
		{
			ans=(ans + dp[n%2][i][k]) % MOD;
		}
	}
	
	printf("%lld",ans);
}
