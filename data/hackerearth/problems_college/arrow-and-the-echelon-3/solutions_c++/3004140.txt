#include<bits/stdc++.h>
using namespace std;
int a[100001];
void compare()
{
	int x,j,y,flag;
	for(int i=1;i<=100000;i++)
	{
		x=i/3;
		flag=0;
		for(j=x;j>=0;j--)
		{
			y=i-j*3;
			if(y%5==0)
			{
				flag=1;
				break;
			}
		}
		if(flag==1)
			a[i]=j*3;
		else
			a[i]=-1;
	}
}
int main()
{
	compare();
	int t;
	cin>>t;
	while(t--)
	{
		int n;
	    cin>>n;
		int x=a[n];
		if(x==-1)
			printf("-1\n");
		else
		{
			for(int i=1;i<=x;i++)
				printf("5");
			for(int i=1;i<=n-x;i++)
				printf("3");
			printf("\n");
		}
	}
	return 0;
}
