#include <bits/stdc++.h>
using namespace std;

long power(long b, long a)
{
	long p=1,i;
	for(i=1;i<=a;i++)
	{
		p=p*b;
	}
	return p;
}

int main()
{
    long n,a,b,c,d;
    
    cin>>n;
    
    while(n--)
    {
    	cin>>a;
    	
    	cin>>b>>c>>d;
    	
    	b+=c+d;
    	
    	cout<<power(b,a)<<endl;;
    }
    
    return 0;
}
