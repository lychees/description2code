def find_number_of_divisor(n, arr, modulo):
    counter = {}
    for i in range(n):
        number = arr[i]
        factor = 2
        while factor*factor <= number:
            while number % factor == 0:
                if factor in counter:
                    counter[factor] += 1
                else:
                    counter[factor] = 1
                number = number/factor
            factor += 1
        if number > 1:
            if number in counter:
                counter[number] += 1
            else:
                counter[number] = 1
    answer = 1
    for i in counter:
        answer *= (counter[i]+1)
    print answer % modulo


n = input()
arr = map(int, raw_input().split())
modulo = 10**9 + 7
find_number_of_divisor(n, arr, modulo)
