#include<iostream>
#include<cmath>
using namespace std;
double s[10007];
int main()
{
	int t;
	cin >> t;
	for (int tt = 1; tt <= t; tt++)
	{
		int n;
		cin >> n;
		int x, y;
		cin >> x >> y;
		double dest = sqrt(x*x + y*y);
		for (int i = 1; i <= n; i++)
		{
			cin >> s[i];
		}
		s[n + 1] = dest;
		n++;
		bool ok = true;
		for (int i = 1; i <= n; i++)
		{
			double sum = 0;
			for (int j = 1; j <= n; j++)
			{
				if (i != j) sum += s[j];
			}
			if (sum < s[i]) ok = false;
		}
		if (ok)
		{
			cout << "Able" << endl;
		}
		else
		{
			cout << "Not able" << endl;
		}
	}
}
