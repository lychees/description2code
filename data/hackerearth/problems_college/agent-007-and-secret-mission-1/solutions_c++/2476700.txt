#include <string.h>
#include <stdlib.h>
#include <stdio.h>
#include <inttypes.h>

typedef unsigned long long ull;
#define MAX 100000

int main()
{
   ull t=0, in=0, n=0;

   scanf ("%llu", &t) ;
   while (t--)
   {
        ull num_toggles = 0;
        char in_s[MAX], in_s_comple[MAX];
        ull i = 0, len=0, comple_1 = 0,  in_s_1 =0;

        scanf ("%s", in_s) ;
        scanf ("%llu", &n) ;

        len = strnlen (in_s, MAX);
        for (i=0; i<len; ++i)
        {
            if (in_s[i] == '0')
            {
                in_s_comple[i] = '1';
                comple_1 ++;
            }
            else if (in_s[i] == '1')
            {
                in_s_comple[i] = '0';
                in_s_1 ++;
            }
        }

        num_toggles = n * (n + 1) / 2;
        if (num_toggles % 2 == 0) //even
            printf ("%llu\n", in_s_1);
        else // odd
            printf ("%llu\n", comple_1);
   }

   return 0;
}
