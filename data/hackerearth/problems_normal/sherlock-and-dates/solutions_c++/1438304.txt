#include <iostream>
#include <cstdio>
using namespace std;

bool valid(int d,int m,int y)
{
	if (d>31) return false;
	if (m>12) return false;
	if (m==1||m==3||m==5||m==7||m==8||m==10||m==12)
	{
		return true;
	}
	else if (m!=2)
	{
		if (d==31) return false;
		else return true;
	}
	else
	{
		if (d>29) return false;
		else if (d<=28) return true;
		if (y%4!=0)
		{
			if (d==29) return false;
			else return true; 
		} 
		if (y%4==0 && y%100!=0) return true;
		if (y%4==0 && y%100==0 && y%400!=0) return false;
		else return true;
	}
}
int ad[5000],am[5000],ay[5000],sd,sm,sy,ed,em,ey;
bool bet(int i)
{
	if (ay[i]>sy && ay[i]<ey) return true;
	if (ay[i]<sy || ay[i]>ey) return false;
	if (sy==ey)
	{
		if (am[i]<sm || am[i]>em) return false;
		if (am[i]>sm && am[i]<em) return true;
		if (sm==em)
		{
			if (ad[i]<sd || ad[i]>ed) return false;
			return true;
		}
		else if (am[i]==sm)
		{
			if (ad[i]<sd) return false;
			return true;
		}
		else
		{
			if (ad[i]>ed) return false;
			return true;
		}
	}
	else if (ay[i]==sy)
	{
		if (am[i]>sm) return true;
		if (am[i]<sm) return false;
		else if (am[i]==sm)
		{
			if (ad[i]<sd) return false;
			else return true;
		}
	}
	else if (ay[i]==ey)
	{
		if (am[i]>em) return false;
		if (am[i]<em) return true;
		else if (am[i]==em)
		{
			if (ad[i]>ed) return false;
			else return true;
		}
	}
}
int main()
{
	int d,m,y,t,k=0,ans,i;
	for (d=1;d<=31;d++)
	{
		m=d+1;
		for (y=1;y<=9999;y++)
		{
			if (m+1==(y%100))
			{
				if (valid(d,m,y))
				{
					ad[k]=d;
					am[k]=m;
					ay[k]=y;
					k++;
				}
			}
		}
	}
	cin>>t;
	while (t--)
	{
		scanf("%d:%d:%d %d:%d:%d",&sd,&sm,&sy,&ed,&em,&ey);
		ans=0;
		for (i=0;i<k;i++)
		{
			if (bet(i)) ans++;
		}
		cout<<ans<<endl;
	}
	return 0;
}
