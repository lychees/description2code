inline void getstring(char *s)
{
     char ch;
     ch=gc();
     int i=0;
     while((ch>=32)&&(ch<=110))
     {
          s[i++]=ch;
          ch=gc();
     }

     s[i]='\0';
}

inline void putstring(char *s)
{
     int i=0;
     while(s[i]>=32 && s[i]<=127)
     {
          pc(s[i]);

     }

}
