#include<bits/stdc++.h>
using namespace std;
#define l1 long long int
l1 k;
l1 bi(l1 in,l1 en,l1 sum[],l1 l)
{
	l1 mid;
	mid=(in+en+1)/2;
	while(in<=en)
	{
		if(l==sum[mid]||(l>sum[mid]&&l<sum[mid+1]))
		return sum[mid];
		else if(l>sum[mid])
		in=mid+1;
		else
		en=mid-1;
		mid=(in+en)/2;	
		
	}
	if(l>sum[k])
	return sum[k];
	return -1;
	
}
int main()
{
	l1 i,j=0,q,n,l,m,ind1,ind2;
	cin>>n;
	l1 num[n],sum[n];
	map<l1,l1>pum;
	for(i=0;i<n;i++)
	cin>>num[i];
	j=0;
	sort(num,num+n);
	for(i=0;i<n;i++)
	pum[num[i]]=++j;
	sum[0]=num[0];
	j=0;
	for(i=1;i<n;i++)
	{
		if(num[i]!=num[i-1])
		sum[++j]=num[i];
	}
	k=j;
	cin>>q;
	while(q--)
	{
		cin>>l>>m;
		if(l>m)
		swap(l,m);
		ind1=bi(0,k,sum,l);
		ind2=bi(0,k,sum,m);
		//cout<<ind1<< " "<<ind2<<endl;
		if(ind1==-1&&ind2==-1)
		cout<<"0"<<endl;
		else if(ind1==-1)
		cout<<pum[ind2]<<endl;
		else
		{
			ind1=bi(0,k,sum,l-1);
			if(ind1==-1)
			cout<<pum[ind2]<<endl;
			else
			cout<<pum[ind2]-pum[ind1]<<endl;
		}
		
		
	}
	return 0;
}
