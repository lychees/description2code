	listtt=raw_input()
	acost=input()
	bcost=input()
	if acost<bcost:
		lcost=acost
	else:
		lcost=bcost	
	last=len(listtt)-1
	first=0
	flag=0
	totalcost=0
	while first<last:
		if listtt[first]=='/' and listtt[last]=='/':
			totalcost+=2*lcost
		elif listtt[first]=='/' and listtt[last]=='a':
			totalcost+=acost
		elif listtt[first]=='/' and listtt[last]=='b':
			totalcost+=bcost
		elif listtt[last]=='/' and listtt[first]=='a':
			totalcost+=acost
		elif listtt[last]=='/' and listtt[first]=='b':
			totalcost+=bcost
		elif listtt[first]!=listtt[last]:
			flag=1
			break
		first=first+1
		last=last-1
	if flag==1:
		print'-1'
