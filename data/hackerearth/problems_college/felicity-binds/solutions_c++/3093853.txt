long long probability(int ,int);

int main()
{ int no_testcases;

int n,p=0;
while(p<no_testcases){
cin>>n;
int a[n],b[n],i=0;

	cin>>a[i];i++;
}
i=0;
while(i<n){
	cin>>b[i];i++;
}

sort(b,b+n);

long long result=0;i=0;
while(i<n){
	
    result=probability(a[i],b[i])+result;i++;
}
printf("%d\n",result);
p++;}
    return 0;
} 

long long probability(int x,int y){
    
  long long rest=0;
  int d=x,l=y;
  while(!(l<=0)){
  if(l&1 !=0)
  rest=rest+d;
  
  d=d<<1;
  l=l>>1;
  
  }
   
return rest;
}
