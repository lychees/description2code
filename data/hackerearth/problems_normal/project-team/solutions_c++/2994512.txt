#include <bits/stdc++.h>

using namespace std;

int main()
{
    int T;
    cin >> T;
    while (T>0){
        int n,minn=2e9,maxn=0;

        cin >> n;

        int niz[n];

        for (int i=0;i<n;i++) {
            cin >> niz[i];
        }

        sort(niz,niz+n);

        for (int i=0;i<n/2;i++) {
            int zbirPara = niz[i]+niz[n-1-i];
            maxn=max(maxn,zbirPara);
            minn=min(minn,zbirPara);
        }

        cout << maxn-minn << endl;
        T--;
    }
}
