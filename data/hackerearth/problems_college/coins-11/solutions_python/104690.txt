import math
def isPrime(n):
    if(n==1):
        return False
    if(n==2 or n==3):
        return True
    if(n%2==0):
        return False
    s=int(math.sqrt(n))+1
    for i in xrange(3,s,2):
        if(n%i==0):
            return False
    return True
