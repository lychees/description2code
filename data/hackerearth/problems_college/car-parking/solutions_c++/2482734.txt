#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    char a[1000000];
    int t,i,j,k,l,n;
    cin>>t;
    while(t--)
    {l=0;
   
    	cin>>a;
    	 n=strlen(a);
    	 if(n<3)
    	 {
    	 	if(n==1)
    	 	{
    	 		if(a[0]=='-')
    	 		l=1;
    	 	}
    	 	else if(n==2)
    	 	{
    	 		if(a[0]=='-' && a[1]!='S' && a[1]!='B')
    	 		l++;
    	 		if(a[1]=='-' && a[0]!='S')
    	 		l++;
    	 		
    	 	}
    	 	
    	 	
    	 }
    	 else{
    	for(i=1;i<strlen(a)-2;i++)
    	{
    		if(a[i]=='-')
    		{
    			if((a[i+1]!='S'&&a[i-1]!='S')&&(a[i+1]!='B'&& a[i+2]!='B'))
    			l++;
    		}
    	}
    	if(a[0]=='-')
    	{
    		if(a[1]!='S' && (a[1]!='B' && a[2]!='B'))
    		l++;
    	}
    	 if(a[n-2]=='-')
    	{
    		if((a[n-1]!='S'&& a[n-3]!='S' ) && a[n-1]!='B')
    		l++;
    	}
    	 if(a[n-1]=='-')
    	{
    		if(a[n-2]!='S')
    		l++;
    	}
    	 }
    	cout<<l<<endl;
    	
    }
    return 0;
}
