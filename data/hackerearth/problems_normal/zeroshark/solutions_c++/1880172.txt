#include <iostream>
#include <cstring>
#define prim 1000000007
using namespace std;

int T,dp[100010][2][2],N;

int main(){
    cin >> T;

    int i;
    while (T--){
        cin >> N;
        memset(dp,0,sizeof(dp));
        dp[0][0][1]=dp[0][1][1]=1;
        for (i=1; i<=N; i++){
            dp[i][1][0]=(dp[i-1][1][0]+dp[i-1][0][0])%prim;
            dp[i][1][1]=(dp[i-1][1][1]+dp[i-1][0][1])%prim;
            dp[i][0][0]=(dp[i-1][0][1]+dp[i-1][1][0])%prim;
            dp[i][0][1]=dp[i-1][1][1];
        }

        cout << dp[N][1][0] << "\n";
    }

    return 0;
}
