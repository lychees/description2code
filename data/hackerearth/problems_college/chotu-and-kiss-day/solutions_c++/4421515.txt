#include <iostream>
#include <bits/stdc++.h>
#define s(x) scanf("%d",&x)
using namespace std;
int arr[3005];
int main()
{
    int n,k,i,j,flag,x;
    s(n);
    for(i=0;i<n;i++)
    {
    	s(arr[i]);
    }
    s(k);
    flag=0;
    sort(arr,arr+n);
    for(i=0;i<n;i++)
    {
    	for(j=i+1;j<n;j++)
    	{
    		x=k-arr[i]-arr[j];
    		if(binary_search(arr,arr+n,x))
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag&&x!=arr[i]&&x!=arr[j])
    	{
    		break;
    	}
    }
    if(flag)
    {
    	printf("YES\n");
    }
    else
    {
    	printf("NO\n");
    }
    return 0;
}
