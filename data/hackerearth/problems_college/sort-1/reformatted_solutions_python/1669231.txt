# print 'Hello World!'
n = int(raw_input())
i = 0
a = []
while i < n:
    tc = int(raw_input())
    while len(a) < tc:
        tcc = map(int, raw_input().split())
        a += tcc
    print " ".join(str(x) for x in sorted(a))
    a = []
    i += 1
