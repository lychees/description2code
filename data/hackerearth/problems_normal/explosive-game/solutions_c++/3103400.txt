#define fors(i,s) for(auto i=(s).begin();i!=(s).end();++i)
#define all(v) (v).begin(),(v).end()
using namespace std;
typedef long long in;
typedef vector<in> VI;
typedef vector<VI> VVI;
VI dfs;
in sm=0;
int main(){
  ios::sync_with_stdio(0);
  cin.tie(0);
  in n,q;
  cin>>n>>q;
  in a,b;
  forn(i,n){
    cin>>a>>b;
    if(a>b)
      swap(a,b);
    dfs.PB(b-a);
    sm+=a+b;
  }
  sort(all(dfs));
  sm/=2;
  for(in i=0;i<n;i+=2){
    if(dfs[i]!=dfs[i+1]){
      forn(j,q){
	cout<<"NO\n";
      }
      return 0;
    }
  }
  forn(i,q){
    cin>>a;
    if(a==sm)
      cout<<"YES\n";
    else
      cout<<"NO\n";
  }
  return 0;
}
