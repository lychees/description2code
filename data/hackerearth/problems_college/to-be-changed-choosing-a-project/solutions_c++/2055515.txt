#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair <int,int> pii;
typedef vector <int> vi;

#define rep(i, n) for(int i = 0; i < (n); ++i)

#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define all(c) c.begin(), c.end()
#define mset(a, v) memset(a, v, sizeof(a))
#define sz(a) ((int)a.size())

#define dbn cerr << "\n"
#define dbg(x) cerr << #x << " : " << (x) << " "
#define dbs(x) cerr << (x) << " "

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n, r;
    cin >> n >> r;

    vi a(n);
    rep (i, n) a[i] = i + 1;

    int pos = 0;
    while (sz(a) > 1) {
        int delta = (r - 1) % sz(a);
        pos += delta;
        pos %= sz(a);
        a.erase(a.begin() + pos);
        pos = pos % sz(a);
    }

    cout << a[0] << "\n";
    return 0;
}
