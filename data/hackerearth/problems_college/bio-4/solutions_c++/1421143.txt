#include <iostream>
#include <cstring>
#include <algorithm>
#include <cstdio>

using namespace std;
const int MAX = 1005;
int dp[MAX][MAX];

int main()
{
    //freopen("input.txt", "r" , stdin);
    int t, result, ans, n, m, ans1;
    string s, s1;
    for(cin >> t;t--;)
    {
        cin >> s >> s1;
        n = s.length();
        m = s1.length();
        result = 0;
        for(int i = 0;i <= n;++i)
        {
            for(int j = 0;j <= m;++j)
            {
                if(i == 0 or j == 0)
                    dp[i][j] = 0;
                else if(s[i-1] == s1[j-1])
                {
                    dp[i][j] = dp[i-1][j-1] + 1;
                    if(result < dp[i][j])
                        result = dp[i][j], ans = i, ans1 = j; 
                }
                else dp[i][j] = 0;
            }
        }
        ans -= ans1;
        result = 0;
        for(int i = 0;i < m;++i)
        {
            if(s[ans+i] != s1[i])
                result++;
        }
        cout << result << endl;
    }
    return 0;
}
