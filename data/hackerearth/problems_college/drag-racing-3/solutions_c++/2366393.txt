#include <iostream>
#include <string>
#include <algorithm>
#include <vector>
#include <map>
#include <cmath>

using namespace std;


string match[32770][2];

int main()
{
    map<string, int> mp;
    int N;
    cin >> N;
    
    int lim = (int) pow(2,N);
    lim--;
    
    string a,b;
    map<string, int>::iterator it;
    for(int i=0; i<lim; i++)
    {
    	cin >> a >> b;
    	//cout << a << " " << b << endl;
    	match[i][0] = a;
    	match[i][1] = b;
    	
    	it = mp.find(a);
    	if(it!=mp.end())
    	{	mp[a] = it->second + 1;	}
    	else
    	{	mp[a] = 1; }
    	
    	it = mp.find(b);
    	if(it!=mp.end())
    	{	mp[b] = it->second + 1;	}
    	else
    	{	mp[b] = 1; }
    }
    
    int mcount=0;
    
    for(it=mp.begin(); it!=mp.end(); it++)
    {	//cout << it->first << " " << it->second << endl;
    	mcount = max(mcount, it->second);	
    }
    
    //cout << mcount << endl;
 	a = "";
 	b = "";
 	for(it=mp.begin(); it!=mp.end(); it++)
    {	if(mcount==it->second)
    	{	if(a=="")
    		{	a = it->first;	}
    		else if(a!="" && b=="")
    		{	b = it->first;	}
    	}
    }
    //cout << a << " " << b << endl;
    
    string ans;
    for(int i=0; i<lim; i++)
    {	if(match[i][0]==a && match[i][1]==b)
    	{	ans = a;	break;	}
    	else if(match[i][0]==b && match[i][1]==a)
    	{	ans = b;	break;	}
    }
    	
    
    cout << ans << endl;
    return 0;
}
