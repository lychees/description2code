#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,k;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	int a[n],num=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		if(a[i]<=0)
    			num++;
    	}
    	if(num>=k)
    		cout<<"NO"<<endl;
    	else
    		cout<<"YES"<<endl;
    }
    return 0;
}
