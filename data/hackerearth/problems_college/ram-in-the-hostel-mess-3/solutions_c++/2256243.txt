#include <iostream>
using namespace std;

typedef long long int ll;

ll fib(int x)
{
	ll a=1,b=1;
	ll c;
	while(x--)
	{
		c=a+b;
		a=b;b=c;
	}
	return c;
	
}

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n;
    	cin >> n;
    	n=fib(n);
    	cout << n*n << endl;
    }
    return 0;
}
