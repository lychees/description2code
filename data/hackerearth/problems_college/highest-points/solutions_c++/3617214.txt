#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;

    long long max_score =INT_MIN;

    for(int i=0;i<n;i++)
    {
        int a,b,c;
        cin>>a>>b>>c;

        long long int score = 3*a+b-c;
        if(score > max_score)
            max_score = score;
    }
    cout<<max_score;
}
