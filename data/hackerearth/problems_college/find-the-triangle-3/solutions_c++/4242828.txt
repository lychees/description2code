#define iter(i,a,b) for(int i=a; i<b; i++)
#define ited(i,a,b) for(int i=a;i>=b;i--)
#define mp make_pair 
#define pb push_back

using namespace std;
int main()
{
	int t;
	si(t);
	while(t--)
	{
		int n;
		si(n);
		int a[n];
		iter(i,0,n)
			si(a[i]);
		sort(a,a+n);
		int mx1=a[n-1],mx2=a[n-2],mx3,f=0;
		ited(i,n-3,0)
		{
			mx3=a[i];
			if(mx3+mx2>mx1)
			{
				f=1;
				break;
			}
			else
			{
				mx1=mx2;
				mx2=mx3;
			}
		}
		if(f)
			printf("%d %d %d\n",mx3,mx2,mx1);
		else
			printf("-1\n");
	}
	return 0;
}
