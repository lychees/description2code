#include <bits/stdc++.h>
using namespace std;

int main()
{
   int n;
   cin >> n;
   int arr[n];
   for(int i = 0;i < n;i++){
   	cin >> arr[i];
   }
   sort(arr,arr+n);
   int q;
   cin >> q;
   int num;
   while(q--){
   	cin >> num;
   	if(binary_search(arr,arr+n,num)){
   		cout<<"I Like You\n";
   	}else {
   		int l = 0;
   		int r = n-1;
   		bool fl = false;
   		while(l < r){
   			int sum = arr[l]+arr[r];
   			if(sum == num){
   				fl = true;
   				break;
   			}else if(sum > num){
   				r--;
   			}else{
   				l++;
   			}
   		}
   		
   		if(fl){
   			cout<<"I Like You\n";
   		}else{
   			cout<<"I Hate You\n";
   		}
   	}
   }
    return 0;
}
