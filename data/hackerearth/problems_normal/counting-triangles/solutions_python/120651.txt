triangles = {}
n = input()
for _ in range(n):
    a,b,c = map(int, raw_input().split())
    t_list = [a, b, c]
    t_list.sort()
    t_list = tuple(t_list)
    if t_list not in triangles:
        triangles[t_list] = 1
    else:
        triangles[t_list] += 1
ans_list = triangles.values()
ans = 0
for t in ans_list:
    if t==1: ans += 1
print ans
