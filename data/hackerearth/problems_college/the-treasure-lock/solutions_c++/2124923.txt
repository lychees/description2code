#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int flag=0;
    	int n;
    	cin>>n;
    	int n1,n2;
    	if(n%3==0)
    	{
    		flag=1;
    		while(n--)
    		{
    			cout<<5;
    		}
    		cout<<endl;
    	}
    	else if(n%5==0)
    	{
    		flag=1;
    		while(n--)
    		{
    			cout<<3;
    		}
    		cout<<endl;
    	}
    	else{
    	n1=n;n2=0;
    	
    	for(int i=0;i<=n;i++)
    	{
    		if(n1%3==0 && n2%5==0)
    		{
    			flag=1;
    			break;
    		}
    		n1--;
    		n2++;
    	}
    	string s;
    	if(flag==1)
    	{
    			while(n1--)
    			{
    				s=s+'5';
    			}
    			while(n2--)
    			{
    				s=s+'3';
    			}
    			cout<<s;
    			cout<<endl;
    	}
    	}
    	if(flag==0)
    	{
    		cout<<-1<<endl;
    	}
    }
}
