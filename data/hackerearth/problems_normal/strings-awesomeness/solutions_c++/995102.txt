#include <iostream>
#include <algorithm>
#include <cstdio>
#include <string>
using namespace std;


int ar[10000007]={0};
int main()
{
    char p[1000007];
    scanf("%s",p);
    string s(p);
    //cin >> s;
    //transform(s.begin(),s.end(), s.begin(), ::tolower);
    int t;

    int n,i;
    int count=0;
    string check="hackerearth";
    int c[26]={0};
    for(i=0;i<check.size();i++)
    {
        c[check[i]-'a']++;
    }

    int tmp[26]={0};
    for(i=0;i<s.size();i++)
    {
        if(s[i]>='A' && s[i]<='Z')
            s[i]=(char)(s[i]+32);
        tmp[s[i]-'a']++;
        if((i-check.size())>=0)
            tmp[s[i-check.size()]-'a']--;
        int flag=1;
        for(int j=0;j<26;j++)
        {
            if(c[j]!=tmp[j])
                flag=0;
        }
        if(flag)
            count++;
        ar[i]=count;

    }


    //cin >> n;
    scanf("%d",&n);
    while(n--)
    {
        int l,r;
        scanf("%d %d",&l,&r);
        if((l-1+9)<r-1)
            printf("%d\n",ar[r-1]-ar[l-1+9]);
        else
            printf("0\n");

    }


    return 0;
}
