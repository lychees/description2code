#include<bits/stdc++.h>
using namespace std;
int bit[1000007];
int query(int id)
{
	int ans=0;
	while(id!=0) {
		ans+=bit[id];
		id = id - (id&(-id));
	}
	return ans;
}
void update(int id, int n)
{
	while(id<=n) {
		bit[id]=bit[id]+1;
		id= id + (id&(-id));
	}
}
