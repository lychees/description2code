#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	long long q;
    	cin>>q;
    	int count = 0;
    	for(int j=1;j<q;j++)
    	{
    		for(int k=j+1;k<=q;k++)
    		{
    			if((j*k)%(j+k)==0)
    				count++;
    		}
    	}
    	cout<<count<<"\n";
    }
    return 0;
}
