#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   	int n, str,sum = 0;
   	int j;
   	cin >> n;
   	int a[n];
   	for(int i = 0; i < n; i++){
   		cin >> a[i];
   	}
   	cin >> str;
   	sort(a, a+n);
   	for(j = n-1; j > 0; j--){
   		sum = sum + a[j];
   		if(sum >= str){
   			break;
   		}
   	}
   	cout << n-j;
    return 0;
}
