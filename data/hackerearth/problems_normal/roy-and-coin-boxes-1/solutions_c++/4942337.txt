#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
   // cout << "Hello World!" << endl;
    ll n;
    cin>>n;
    ll cnt[n+1];
    memset(cnt,0,sizeof(cnt));
    
    
    ll m,a,b;
    cin>>m;
    while(m--)
    {
    	cin>>a>>b;
    	cnt[a]++;
    	cnt[b+1]--;
    	
    
    }
    	
    
    
    ll temp=0,t[1000001];
    memset(t,0,sizeof(t) );
    	
    
    for(ll i=1;i<=n;i++)
    {
    	temp+=cnt[i];
    	t[temp]++;
    }t[0]=0;
    	
    for(ll i=1000000-1;i>0;i--)
    {
    	t[i]+=t[i+1];
    }
    ll q,z;
    cin>>q;
    while(q--)
    {
    	cin>>z;
    	if(z==0)
    	cout<<n<<endl;
    	else
    	cout<<t[z]<<endl;
    	
    }
    return 0;
}
