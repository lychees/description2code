#include <iostream>
#include<string>
using namespace std;

int main()
{
    int n,size=0;
    cin>>n;
    int a[n],f=0;
    string s[n];
    cin>>s[size];
    a[size]=1;
    size++;
    string tem;
    for(int i=1;i<n;i++)
    {
    	f=0;
    	cin>>tem;
    	for(int j=0;j<size;j++)
    	{
    		if(tem.compare(s[j])==0)
    		{
    			a[j]=a[j]+1;
    			f=1;
    		}
    	}
    	if(f==0)
    	{
    		s[size]=tem;
    		a[size]=1;
    		size++;
    	}
    }
    for(int i=0;i<size;i++)
    {
    	cout<<s[i]<<" "<<a[i]<<endl;
    }
    return 0;
}
