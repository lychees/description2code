#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,i;
    cin>>n;
    map<string,int>mp;
    string s[n+1];
    int ans=0;
    char c;
    for(i=0;i<n;i++)
    {
    	cin>>s[i];
    	int len=s[i].length();
    	if(mp[s[i]])
    	{
    	ans=len;
    	c=s[i][len/2];
    	}
    	else
    	{
    		mp[s[i]]=1;
    	string temp;
    	for(int j=len-1;j>=0;j--)
    	temp+=s[i][j];
    	mp[temp]=1;
    	}
    }
    cout<<ans<<" "<<c<<endl;
    return 0;
}
