#include <bits/stdc++.h>
using namespace std;

#define debug(args...) fprintf(stderr,args)

typedef long long lint;
typedef pair<int,int> pii;
typedef pair<lint,lint> pll;

const int INF = 0x3f3f3f3f;
const lint LINF = 0x3f3f3f3f3f3f3f3fll;

int main() {
    int t;
    for(scanf("%d",&t);t--;) {
        int n,k;
        scanf("%d%d",&n,&k);
        stack<int> parities;
        int parity = 1;
        int ans;
        while(1) {
            if(max(1, (n + parity)/2) >= k) {
                if(n == 1) ans = 0;
                else ans = 2*(k-1) + (parity^1);
                break;
            }
            k -= (n + parity)/2;
            int n_par = n%2;
            n -= (n + parity)/2;            
            parities.push(parity);
            parity ^= n_par;
        }
        while(!parities.empty()) {
            ans = 2*ans + parities.top();
            parities.pop();
        }            
        printf("%d\n",ans+1);
    }        
    return 0;
}
