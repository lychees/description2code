int dfn[N],low[N],in[N],s[N],cnt,top,p,be[N],sz[N];
vi g[N];
void tarjan(int u){
    dfn[u]=low[u]=++cnt;s[++top]=u;in[u]=1;
    for(int i=0;i<g[u].size();i++){
        int j=g[u][i];
        if(!dfn[j])tarjan(j),low[u]=min(low[u],low[j]);
        else if(in[j])low[u]=min(low[u],dfn[j]);
    }
    if(dfn[u]==low[u]){
        int j;++p;
        do{
            j=s[top--];in[j]=0;
            be[j]=p;sz[p]++;
        }while(j!=u);
    }
}
