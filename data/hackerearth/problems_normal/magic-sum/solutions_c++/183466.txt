long long T,N,maxSum;
vector<long long> data;

long long setMaximum(long long i)
{
	if(i>=N)
	{
		return 0;
	}
	else
	{
		long long val1=setMaximum(2*i);
		long long val2=setMaximum(2*i+1);
		long long temp=val1+val2+data[i];
		if(temp>maxSum)
		{
			maxSum=temp;
		}
		return (data[i]+max(val1,val2));
	}
}
