#include <bits/stdc++.h>
typedef long long ll;
using namespace std;

int main()
{
	ll t,n,k;

	cin>>t;

	while(t--)
	{
		cin>>n>>k;
		ll a[n];

		for(ll i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a,a+n);

		vector <ll> v;

		ll x,y;
		x=0;y=n-1;

		while(x<y)
		{
			v.push_back(a[x]+a[y]);
			x++;
			y--;
		}
		sort(v.begin(),v.end());
		ll min=v[v.size()-1]-v[0];
		cout<<min<<"\n";

		if(min>k)
		{
			cout<<"No more girlfriends!\n";
		}
		else
			if(min==k)
			{
				cout<<"Lucky chap!\n";
			}
			else
				if(min<k)
				{
					cout<<"Chick magnet Jhool!\n";
				}
	}



	return 0;
}
