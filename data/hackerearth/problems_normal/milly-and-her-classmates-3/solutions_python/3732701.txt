'''

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
for x in xrange(int(raw_input())):
	raw_input()
	minHt = 0
	maxHt = 0
	minHtPos = 0
	maxHtPos = 0
	maxFound = False
	minFound = False
	turnPossible = True
	heights = map(int, raw_input().split())
	for y in xrange(1, len(heights)):
		if maxFound:
			if minFound:
				if heights[y] < heights[y - 1]:
					turnPossible = False
			else:
				if y == len(heights) - 1:
					minHt = heights[y]
					minHtPos = y
					minFound = True
				if heights[y] > heights[y - 1]:
					minHt = heights[y - 1]
					minHtPos = y - 1
					minFound = True
		else:
			if heights[y] < heights[y - 1]:
				maxHt = heights[y - 1]
				maxHtPos = y - 1
				maxFound = True
	if turnPossible:
		if maxHtPos > 0 and minHtPos < len(heights) - 1:
			if minHt > heights[maxHtPos - 1] and maxHt < heights[minHtPos + 1]:
				print str(maxHtPos + 1) + " " + str(minHtPos + 1)
			else:
				print "-1 -1"
		else:
			if minHtPos == 0:
				print "-1 -1"
			elif minHtPos < len(heights) - 1 and maxHt < heights[minHtPos + 1]:
				print str(maxHtPos + 1) + " " + str(minHtPos + 1)
			elif minHtPos == len(heights) - 1:
				print str(maxHtPos + 1) + " " + str(minHtPos + 1)
			else:
				print "-1 -1"
	else:
		print "-1 -1"
