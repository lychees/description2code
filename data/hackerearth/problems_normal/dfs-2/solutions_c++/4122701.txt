#include <algorithm>
#include <cstdio>
#include <map>
#include <queue>
#include <set>
#include <stack>
#include <vector>

typedef long long ll;

using namespace std;

ll query(const vector<ll> &v, int n, ll l, ll r) {
    int lo, hi, mid, lower_index, upper_index;

    if (l > v[n - 1])
        return 0;

    lo = 0;
    hi = n - 1;
    while (lo < hi) {
        mid = lo + (hi - lo) / 2;
        if (v[mid] >= l)
            hi = mid;
        else
            lo = mid + 1;
    }
    lower_index = lo;

    if (r > v[n - 1])
        upper_index = n - 1;
    else {
        lo = 0;
        hi = n - 1;
        while (lo < hi) {
            mid = lo + (hi - lo + 1) / 2;
            if (v[mid] <= r)
                lo = mid;
            else
                hi = mid - 1;
        }
        upper_index = lo;
    }

    return upper_index - lower_index + 1;
}

int main() {
    int n, q;
    ll l, r;

    scanf("%d", &n);
    vector<ll> v(n);
    for (int i = 0; i < n; ++i)
        scanf("%lld", &v[i]);
    sort(v.begin(), v.end());

    scanf("%d", &q);
    for (int i = 0; i < q; ++i) {
        scanf("%lld %lld", &l, &r);
        printf("%lld\n", query(v, n, l, r));
    }
    return 0;
}
