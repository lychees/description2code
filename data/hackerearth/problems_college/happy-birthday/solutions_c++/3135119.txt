#include<iostream>
using namespace std;
int main()
{
  int n,i,f=0,th=0,t=0,o=0,p=0,q=0,r=0,l=0,w=0;
  cin>>n;
  int a[n];
  for(i=0;i<n;i++)
  {
  cin>>a[i];
  if(a[i]==4)
   f++;
  else  if(a[i]==3)
   th++;
   else if(a[i]==2)
   t++;
   else
    o++;
  }
  int count=0;
  count=f;
  if(th==o)
  {
    count=count+th;
  }
  else if(th>o)
  {
      p=th-o;
      count=count+o+p;
  }
  else if(th<o)
  {
      count=count+th;
      q=o-th;
  }
  if(t%2==0)
 {
     count=count+(t/2);
 }
 else{
    count=count+(t/2);
    r=t%2;

 }
 if(q>=4)
 {
  l=q/4;
    count=count+l;
    w=q%4;
 }
if(r==1&&(w==0||w==1||w==2))
{
    count=count+1;
}
if(r==0&&(w==1||w==2||w==4))
{
    count=count+1;
}
if(r==1&&w==3)
{
count=count+2;
}

cout<<count;
}
