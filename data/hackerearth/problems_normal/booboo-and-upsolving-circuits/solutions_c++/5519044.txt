#include<bits/stdc++.h>

#define ll long long

using namespace std;

ll n,m,temp;
vector<ll> v;

bool check(ll t)
{
    ll ctr = 0;
    ll sum = 0;

    ll len = v.size();

    for(ll i=0;i<len;i++)
    {
        if(v[i] > t)
            return false;

        if(sum + v[i] > t)
        {
            ctr++;
            sum = 0;
            sum = v[i];
        }
        else
            sum += v[i];
    }
    ctr++;


    if(ctr > m)
        return false;

    else
        return true;


}


int main()
{

    cin>>n>>m;

    ll l,r,mid;

    for(ll i=1;i<=n;i++)
    {
        cin>>temp;
        v.push_back(temp);
    }


    l = 1;
    r = 1e18;

    while(l<=r)
    {
        mid = (l+r)/2;
        //cout<<mid<<endl;

        if(check(mid) && !check(mid-1))
          break;

        else if(check(mid))
            r = mid - 1;

        else
            l = mid + 1;
    }



    cout<<mid;

    return 0;
}
