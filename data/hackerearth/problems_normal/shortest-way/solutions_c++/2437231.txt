#include<bits/stdc++.h>
#define ll unsigned long long int
using namespace std;
ll ans=0ll;
void gcd(ll x, ll y){
	if(!y) return;
	ans+=x/y;
	gcd(y,x%y);
}	
int main(){
    int t;
    cin>>t;
    while(t--){
    	ans=0;
    	ll a,b; cin>>a>>b;
    	gcd(a,b);
    	cout<<ans-1<<endl;
    }
}
