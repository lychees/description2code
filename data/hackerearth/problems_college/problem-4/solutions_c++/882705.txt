#include<stdio.h>
#define Sameer_Gupta

using namespace std;

int main()
{
	int hash[27]={0},i;
	char s[5000];
	
	scanf("%s",&s);

	   for(i=0;s[i]!='\0';i++)
	   {
	   	hash[s[i]-97]++;
	   }

	   
	  for(i=0;i<26;i++)
	   {
	   	
	   	if(hash[i]!=0)
	   	{
	   		
	   		while(hash[i]--)
         	{
	   	        printf("%c",i+97);
	        }
	   		
	   	}
	   
       }
	   


	return 0;
}
