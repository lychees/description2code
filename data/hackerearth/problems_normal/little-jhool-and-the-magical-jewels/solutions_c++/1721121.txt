#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	string s;
    	cin >> s;
    	int a[4] = {0};
    	for(int i =0; i < s.length();i++)
    	{
    		if(s[i] == 'r')
    			a[0]++;
    		if(s[i] == 'u')
    			a[1]++;
    		if(s[i] == 'b')
    			a[2]++;
    		if(s[i] == 'y')
    			a[3]++;
    	}
    	int min = a[0];
    	for(int i =0; i < 4; i++)
    	{
    		if(a[i]<min)
    			min = a[i];
    	}
    	cout << min << endl;
    }
    return 0;
}
