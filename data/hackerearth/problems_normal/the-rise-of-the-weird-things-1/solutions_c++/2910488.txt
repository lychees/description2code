#include<iostream>
using namespace std;
int main()
{
	long temp,j,i,t,n,zom[1000],vam[1000],ctrz=0,ctrv=0,s1=0,s2=0;
	cin>>t;
	for(i=0;i<t;i++)
	{
		cin>>n;
		if(n%2==0)
			zom[ctrz++]=n;
		else
			vam[ctrv++]=n;	
	}
	for(i=0;i<ctrz;i++)
	{
		for(j=0;j<(ctrz-1)-i;j++)
		{
			if(zom[j]>zom[j+1])
			{
				temp=zom[j];
				zom[j]=zom[j+1];
				zom[j+1]=temp;
			}
		}
	}
	for(i=0;i<ctrv;i++)
	{
		for(j=0;j<(ctrv-1)-i;j++)
		{
			if(vam[j]>vam[j+1])
			{
				temp=vam[j];
				vam[j]=vam[j+1];
				vam[j+1]=temp;
			}
		}
	}
	for(i=0;i<ctrz;i++)
	{
		s1+=zom[i];
		cout<<zom[i]<<" ";
	}
	cout<<s1<<" ";
	for(i=0;i<ctrv;i++)
	{
		s2+=vam[i];
		cout<<vam[i]<<" ";
	}
	cout<<s2;
}
