#include <bits/stdc++.h>
#include <iostream>

using namespace std;

int main()
{
	int n;
	cin >> n;
	int a[n];
	
	for(int i = 0; i < n; i++) {
		cin >> a[i];
	}
	
	sort(a, a+n);
	
	int s;
	
	cin >> s;
	
	int ss = 0;
	int c = 0;
	
	for(int i = n-1; i >= 0; i--) {
		ss = ss + a[i];
		c++;
		
		if(ss >= s) {
			break;
		}
	}
	
	cout << c << endl;
	
    //cout << "Hello World!" << endl;
    return 0;
}
