#include <iostream>
using namespace std;

int main()
{
   int n,t,a,b,c,X,Y;
   cin>>t;
   for(int i=1;i<=t;i++)
   {
   	int side1=0,side2=0;
   		cin>>n;
   		cin>>a>>b>>c;
   		for(int j=1;j<=(n);j++)
   		{
   				cin>>X;
   				cin>>Y;
   				if(((a*X)+(b*Y) + c)>0)
   				{
   						side1++;
   				}
   				else if(((a*X)+(b*Y)+c)<0)
   				{
   					side2++;
   				}
   		}
   if(side1>side2)
   {
   	cout<<side1<<endl;
   }
   else
   {
   	cout<<side2<<endl;
    
	}
	
	}
	return 0;
}
