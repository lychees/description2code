#include "iostream"
#include "cstdio"
#include "string"

using namespace std;

int main(int argc, char const *argv[])
{
	int t;
	scanf("%d",&t);
	getchar();
	string s1,s2;
	while(t>0)
	{
		s1.clear();
		s2.clear();
		getline(cin,s1);
		getline(cin,s2);
		
		// cout<<s1<<"\n"<<s2;
		if (s2.find(s1)!=std::string::npos)
		{
			printf("possible\n");
		}
		else
			printf("not possible\n");
		t--;
	}
	return 0;
}
