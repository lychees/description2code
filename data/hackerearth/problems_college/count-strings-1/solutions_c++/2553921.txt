#include <bits/stdc++.h>
using namespace std;

int main()
{
	int T,i,j,a[27];
	string A,B;
	cin>>T;
	while(T--){
		cin>>A;
		cin>>B;
		j=A.size();
		memset(a,0,sizeof(a));
		for(i=0; i<j; ++i)
			++a[A[i]-'a'];
		j=B.size();
		long long res=a[B[0]-'a'];
		--a[B[0]-'a'];
		for(i=1; i<j; ++i){
			res *= (long long)a[B[i]-'a'];
			res %=1000000007;
			--a[B[i]-'a'];
		}
		cout<<res<<endl;
	}
    return 0;
}
