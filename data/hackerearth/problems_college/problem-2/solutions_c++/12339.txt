#include<stdio.h>
#define N_MAX 45
#define NIL 0
bool emp[N_MAX][N_MAX];
int salary[N_MAX];
char str[N_MAX];
int n;
void dp(int k)
{bool flg=0;
    if(salary[k]==NIL)
    {
        for(int i=1;i<=n;i++)
        {
            if(emp[k][i])
            {flg=1;
                if(salary[i]==NIL)
                {
                    dp(i);
                }
                salary[k]+=salary[i];
            }
        }
        if(flg==0)
        {
            salary[k]=1;
            return;
        }
    }
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(int i=1;i<=n;i++)salary[i]=NIL;
        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=n;j++)emp[i][j]=0;
        }
        for(int i=1;i<=n;i++)
        {
            scanf("%s",str);
            for(int j=0;j<n;j++)
            {
                if(str[j]=='Y')emp[i][j+1]=1;
            }
        }
        for(int i=1;i<=n;i++)dp(i);
        /*for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=n;j++)printf("%d",emp[i][j]);
            printf("\n");
        }*/
        int sum=0;
        //for(int i=1;i<=n;i++)printf("%d ",salary[i]);printf("\n");
        for(int i=1;i<=n;i++)
        {
            sum+=salary[i];
        }
        printf("%d\n",sum);
    }
    return 0;
}
