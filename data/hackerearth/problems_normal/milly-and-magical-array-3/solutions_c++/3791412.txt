#include<stdio.h>
#include<math.h>
#define ll long long int
ll a[100005];
int main()
{
	ll t;
	scanf("%lld",&t);
	while(t--)
	{
		ll n,ans,k,sum=0,i;
		scanf("%lld%lld",&n,&k);
		for(i=1;i<=n;i++) scanf("%lld",&a[i]);
		for(i=2;i<=n;i++)
		{
			ll temp=a[i]-a[i-1];
			//printf("%lld\n",temp);
			if(temp<=0)
				{
					temp=-temp;
					a[i]+=(((temp/k)+1)*k);
					sum+=((temp/k)+1);
				}
		}
		printf("%lld\n",sum);
	}
	return 0;
}
