#include<iostream>
#include<vector>
#include<queue>
#include<string.h>
using namespace std;
int main(){
	int v,e,i,a,b,n;
	cin>>v>>e;
	vector<int>vec[e];
	for(i=0;i<e;i++){
		cin>>a>>b;
		vec[a].push_back(b);
		vec[b].push_back(a);
	}
	queue<int>q;
	bool visited[v];
	int level[v];
	memset(visited,false,sizeof(visited));
	memset(level,0,sizeof(level));
	q.push(0);
	level[0]=0;
	while(!q.empty()){
		n=q.front();
		q.pop();
		visited[n]=true;
		for(i=0;i<vec[n].size();i++){
			if(visited[vec[n][i]]==false){
				level[vec[n][i]]=level[n]+1;
				q.push(vec[n][i]);
				visited[vec[n][i]]=true;
			}
		}
	}
	for(i=1;i<v;i++){
		if(level[i]>0)
		cout<<level[i];	
		else
		cout<<-1;
		cout<<endl;
	}
	return 0;
}
