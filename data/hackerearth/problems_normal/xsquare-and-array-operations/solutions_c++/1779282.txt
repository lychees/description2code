#include<iostream>
using namespace std;
int max(int a , int b)
{
    return a>b?a:b;
}
int main()
{
long long int t;
    cin>>t;
    while(t--)
        {
            long long int cost=0;
            long long int n;
            long long int arr[100005];

            cin>>n;
            for(long long int i=0;i<n;i++)
            {
                cin>>arr[i];
            }
            for(long long int i=0;i<n-1;i++)
            {
                cost+=max(arr[i],arr[i+1]);
            }
            cout<<cost<<endl;
        }
}
