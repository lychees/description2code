for i in range(int(raw_input())):
    x, y = map(int, raw_input().split())
    c = 1
    yy = len(str(y))
    flg = True
    while flg:
        a = pow(x, c)
        # print a
        if str(a)[:yy] == str(y):
            flg = False
            break
        c = c+1
    print 'Case '+str(i+1)+':'+' '+str(c)
