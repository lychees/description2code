#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    long int *a = new long int[n], sum1 = 0, sum2 = 0, sum3 = 0;
    for(int i = 0; i < n; i++)
    	cin>>a[i];
    for(int i = 0; i < n; i++)
    {
    	if(i%3==0)
    		sum1 += a[i];
    	else if(i%3 == 1)
    		sum2 += a[i];
    	else
    		sum3 += a[i];
    }
    cout<<sum1<<" "<<sum2<<" "<<sum3;
    return 0;
}
