# include <iostream>
# include <cmath>
using namespace std;

int main () {
	long long int N, *A, answer = 1;

	cin >> N;
	A = new long long int [N];

	int z = (pow (10, 9)) + 7;
	for (int i = 0; i < N; i++) {
		cin >> A[i];
		answer = (answer*A[i]) % z;
	}
	cout << answer << endl;
	return 0;
}
