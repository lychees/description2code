N = int(raw_input())
file = ""

grt = 0

h = 0


for i in range(0,N):
	
	raw = raw_input()
	
	count = 0
	
	for i in range(1, len(raw)):
		if(raw[i] == "C" and raw[i] == raw[i-1]):
			count +=1
			if count+1 > grt:
				grt = count+1
		else:
			count = 0
		
	
	file = file + raw
	
count = 0
for i in range(1,len(file)):
	if(file[i] == "C" and file[i] == file[i-1]):
		count +=1
		if(count+1 > h):
			h = count+1
	else:
		count = 0
	
		
print grt, h
