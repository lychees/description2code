#include <bits/stdc++.h>
using namespace std;
#define ll long long int
ll a[100010];
void sieve()
{
	ll i,j;
	for(i=0;i<=100005;i++)
		a[i]=1;
	a[0]=0;
	a[1]=0;
	for(i=2;i<=100005/2;i++)
	{
		if(a[i]==1)
		{
			for(j=i+i;j<=100005;j=j+i)
				a[j]=0;
		}
	}
}
int main()
{
	sieve();
	ll t;
	scanf("%lld",&t);
	while(t--)
	{	
		char str1[100010],*b,*e;
		ll g,i,flag=0;
		scanf("%s",str1);
		g=strlen(str1);
		for(b=str1,e=&str1[g-1];b<=e;b++)
		{
			if(a[(int)*b]==0)
			{
			//	printf("%c\n",(int)*b);
				flag++;
			}
		}
		if(flag==g)
			printf("NULL");
		else
		{
			for(b=str1,e=&str1[g-1];b<=e;b++)
			{
				if(a[(int)*b]==0)
				{
					printf("%c",(int)*b);
				}
			}
		}
		printf("\n");
	}
	return 0;
}
