#include<bits/stdc++.h>
using namespace std;
long long gcd(long long a,long long b)
{
	long long mod ;
	while(a!=0)
	{
		mod = b%a;
		b = a;
		a = mod;
	}
	return b;
	
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int l,b,h;
		scanf("%lld%lld%lld",&l,&b,&h);
		long long int min = gcd(l,gcd(b,h)) % 1000000007; 
		long long int no = (l /min)%1000000007;
		no = (no * b/min) % 1000000007;
		no = (no * h/min)% 1000000007;
		printf("%lld %lld\n",min,no);
	}
	return 0;
 }
