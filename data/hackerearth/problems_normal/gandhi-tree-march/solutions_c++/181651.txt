#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;


struct stack{

	int l;
	int id;
	char a;

}s[1000000];

int top=-1;


void push(struct stack t){

	s[++top]=t;

}

struct stack pop(){

	return s[top--];

}

int main(){


	int T;
	string str;
	int id;
	cin>>T;
	int idt;
	while(T--){
		
	vector<char> list;
		cin>>id>>str;
		for(long i=0;i<str.size();++i){
			if(str[i]>='a' && str[i] <='z'){
				if(top!=-1){
					idt=s[top].id+s[top].l;			
					if(idt==id) 
						list.push_back(str[i]);
				}
				else{
					idt=0;
					if(idt==id) 
						list.push_back(str[i]);
				}
			
			}
			else if(str[i]=='('){
				struct stack temp;
				temp.id=idt;
				temp.l=-1;
				if(top!=-1)
					s[top].l=1;
				push(temp);

			}
			else if(str[i]=='.'){
				s[top].l=1;
			}
			else if(str[i]==')'){
				pop();
			}

		}

	sort(list.begin(),list.end());
      if(list.size()>0){
	for(long i=0;i<list.size();++i){
		cout<<list[i];
	}
	cout<<"\n";
     }
	else cout<<"Common Gandhijee!\n";
	}
}
