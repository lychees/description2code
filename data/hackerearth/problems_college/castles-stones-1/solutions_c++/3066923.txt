#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int m, n, i;
    	cin>>m>>n;
    	int a[n];
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	sort(a,a+n);
    	reverse(a,a+n);
    	int count=0;
    	for(int i=0;i<n;i++)
    	{
    		if(a[i]<=m)
    		{
    			if(a[i]==m)
    			{
    				count++;
    				continue;
    			}
    			if(a[i]+a[n-1]<=m)
    			{
    				count++;
    				n-=1;
    			}
    			else
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
