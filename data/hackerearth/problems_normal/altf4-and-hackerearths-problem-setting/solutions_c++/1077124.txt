#include<stdio.h>
#include<cstring>
#include<string>
#include<iostream>
#include<cstdlib>
using namespace std;
#define scant() int tc;scanf("%d",&tc)
typedef long long int ll;
struct node
{
	ll price,task;
};
struct node a[2002];
int compare(const void *a, const void *b)
{
	return ((struct node *)a)->price - ((struct node *)b)->price;
}
ll suffix_sum[2002];
ll min(ll a, ll b)
{
	return a<b?a:b;
}
ll ans[2002];
#define INF 1000000000000LL
int main()
{
	scant();
	ll n,k;
	for(int i=0;i<tc;i++)
	{
		scanf("%lld%lld",&n,&k);
		for(int i=0;i<n;i++)
		{
			scanf("%lld",&(a[i].price));
		}
		for(int i=0;i<n;i++)
		{
			scanf("%lld",&(a[i].task));
		}
		qsort(a,n,sizeof(struct node),compare);
		suffix_sum[n-1]=a[n-1].task;
		for(int i=n-2;i>=0;i--)
			suffix_sum[i]=suffix_sum[i+1]+a[i].task;
		ans[0]=0;
		for(int i=1;i<=n;i++)
		{
			ans[i]=INF;
			for(int j=0;j<i;j++)
			{
				ans[i]=min(ans[i],ans[j]+k*a[i-1].price+a[i-1].price*(a[i-1].task+suffix_sum[j]-suffix_sum[i-1]));
			}
		}
		cout<<ans[n]<<endl;
	}
}
