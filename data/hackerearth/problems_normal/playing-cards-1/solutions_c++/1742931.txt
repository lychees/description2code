using namespace std;
#define MAX 10000
#define MOD 1000000007
long long int k[MAX];
long long int p(long long int a, long long int b){
	long long int res = 1LL;
	while (b){
		if (b & 1LL){
			res *= a;
			res %= MOD;
		}
		a *= a;
		a %= MOD;
		b >>= 1LL;
	}
	return res;
}
long long int c(long long int a, long long int b){
	if(a<b){
		return 0;
	}
	long long int u = k[b] * k[a - b] % MOD;
	u %= MOD;
	return (k[a] * p(u, MOD - 2)) % MOD;
}
int main(){
	k[0] = 1LL;
	for (int i = 1; i < MAX; i++){
		k[i] = k[i - 1];
		k[i] *= (long long int)(i);
		k[i] %= MOD;
	}
	int t;
	scanf("%d", &t);
	while (t--){
		int n, m;
		scanf("%d%d", &n, &m);
		long long int a = 50 * n;
		long long int f = 6 * n;
		long long int r = 18 * n;
		long long int way = 0;
		if (m % 2){
			puts("0");
			continue;
		}
		long long int mu = m/2;
		for (int i = 0; i <= mu; i++){  //number of face card on black
			long long int en = c(f, i);
			en %= MOD;
			en *= c(f, mu - i);
			en %= MOD;
			en *= c(r, mu - i);
			en %= MOD;
			en *= c(r, i);
			en %= MOD;
			way += en;
			way%=MOD;
		}
		printf("%lld\n", way);
	}
	return 0;
}
