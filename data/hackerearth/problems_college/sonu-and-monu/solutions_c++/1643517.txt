#include <bits/stdc++.h>
#define l r
using namespace std;
char s[1000006];
int main()
{
    long long i,j,k,n,m,p,t;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld %lld %s",&n,&p,s);
    	
    	long long l[100005]={0},cnt=0;
    	j=0;
    	long long f=0;
    	for(i=0;i<n;i++)
    	{
    		if(s[i]=='1')
    		cnt++;
    		if(f==0&&(i==p||i+1>=n))
    		{
    			f=1;
    			k=cnt;
    		}
    	}
    	if(s[j]=='1')
    	l[j]=k;
    	for(j=j+1;j<n;j++)
    	{
    		if(j+p<n&&s[j+p]=='1')
	   		k++;
	   		if(s[j-1]=='1')
	   		k--;
    		if(s[j]=='0')
    		{
    			continue;
    		}
    		l[j]=k;
    	}
    	j=n-1;
    	while(j>=0&&s[j]=='0')
    	j--;
    	k=0;
    	for(i=0;j-i>=0&&i<=p;i++)
    	{
    		if(s[j-i]=='1')
    		k++;
    	}
    	if(s[j]=='1')
    	r[j]+=k;
    	for(j=j-1;j>=0;j--)
    	{
	   		if(j-p>=0&&s[j-p]=='1')
	   		k++;
	   		if(s[j+1]=='1')
	   		k--;
    		if(s[j]=='0')
    		{
    			continue;
    		}
    		r[j]+=k;
    	}
    	//for(i=0;i<n;i++)
    	//printf("%lld %lld \n",l[i],r[i]);
    	k=0;
    	for(i=0;i<n;i++)
    	{
    		k+=r[i];
    	}
    	m=n*n;
    	n=k-cnt;
    	
    	if(n==0)
    	printf("0/1\n");
    	else
    	{
    		k=__gcd(n,m);
    		printf("%lld/%lld\n",n/k,m/k);
    	}
    	
    }
    return 0;
}
