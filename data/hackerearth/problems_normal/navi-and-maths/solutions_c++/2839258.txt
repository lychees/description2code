#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define mod 1000000007
#define si(i) scanf("%lld",&i);
#define pi(i) printf("%lld",i);
#define rep(i,st,end) for(i=st;i<end;i++) 

ll power(ll a,ll b)
{
    ll x=1,y=a;

    while(b>0)
    {
        if(b%2)
        {
            x=(x*y)%mod;
        }
        y=(y*y)%mod;
        b/=2;
    }
    return x;
}

int main(){
    ll t,l;
    si(t);
    for(l=1;l<=t;l++){
    	ll n,i,j,k,p,sum,cnt,tmp,mx;
    	si(n);
    	ll arr[n];
    	rep(i,0,n) si(arr[i]);
    	mx=0;
    	for(i=0;i<(1<<n);i++){
    		cnt=0;
    		tmp=1;
    		sum=0;
    		for(j=0;j<n;j++){
    			if(i&(1<<j)){
    				cnt++;
    				tmp=(((tmp*arr[j])+mod)%mod);
    				sum+=arr[j];
    			}
    		}
    		
    		if(cnt>=2){
    			sum%=mod;
    			sum=power(sum,mod-2);
    			tmp*=sum;
    			tmp=(tmp+mod)%mod;
    			if(tmp>mx) mx=tmp;
    		}
    	}
    	printf("Case #%lld: %lld\n",l,mx);
    }
    return 0;
}
