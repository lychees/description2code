#include <bits/stdc++.h>
using namespace std;
struct point
{
	long x;
	long y;
}p[1005];

long gcd(long a,long b)
{
	long t;
	a=abs(a);
	b=abs(b);		
	if(a<b)
	{
		a+=b; b=a-b; a-=b;
	}
	while (b)
	{
		t=a;
		a=b;
		b=t%a;
	}
	return a;
}

int main()
{
	int n,i;
	cin>>n;
	long b=0l,a=0l,res;
	for(i=0;i<n;i++)
	{
		cin >> p[i].x >> p[i].y;
		if(i!=0)
		{
			a += ((p[i-1].x * p[i].y) - (p[i-1].y * p[i].x));
			b += (gcd(p[i].x - p[i-1].x,p[i].y - p[i-1].y));
		}
	}
	b+= (gcd(p[0].x - p[n-1].x,p[0].y - p[n-1].y));
	a += ((p[n-1].x * p[0].y) - (p[n-1].y * p[0].x));
	res = (abs(a)+2 - b)/2;
	if(res<0)
		res=0;
//	cout <<a <<"\n" <<b <<"\n";
	cout << res <<"\n";
	return 0;
}
