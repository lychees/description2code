def main():
    s = raw_input()
    ans = len(s)
    l = len(s)
    c = 1
    i = 1
    while i < l:
        if s[i] == s[i-1]:
            c = c+1
        else:
            if c > 0:
                ans = ans + ((c-1)*c)/2
            c = 1
        i = i+1
    if c > 0:
        ans = ans + ((c-1)*c)/2
    print ans


if __name__ == "__main__":
    main()
