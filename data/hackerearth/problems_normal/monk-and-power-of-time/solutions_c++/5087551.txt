#include <iostream>
using namespace std;

void rotate(int arr[],int n,int low) {
	int temp = arr[low];
	int i;
	for(i=low;i<n-1;i++) {
		arr[i]=arr[i+1];
	}
	arr[n-1]=temp;
}

int findRes(int callArr[],int idArr[],int n,int low) {
	if(low==n){
		return 0;
	}
	else {
		if(callArr[low]!=idArr[low]) {
			rotate(callArr,n,low);
			return 1+findRes(callArr,idArr,n,low);
		}
		else {
			return 1+findRes(callArr,idArr,n,low+1);
		}
	}
}
int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    int callArr[n],idArr[n];
    for(int i=0;i<n;i++){
    	cin>>callArr[i];
    }
    for(int i=0;i<n;i++){
    	cin>>idArr[i];
    }
    int res=findRes(callArr,idArr,n,0);
    cout<<res;
    return 0;
}
