#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

long long int coin[55];
long long int dp[255][55];
long long int m;
long long int cool(long long int n,long long int j)
    {
     if(dp[n][j]!=-1)
        return dp[n][j];
    
     if(n==0)
         return 1;
    
       long long int res=0;
    
        for(long long int i=j;i<m;i++)
            if(n-coin[i]>=0)
           res+=cool(n-coin[i],i);
                       
    dp[n][j]=res;
    return res;
    
}

 int main() {
    
    for(long long int i=0;i<255;i++)
        for(long long int j=0;j<55;j++)
        dp[i][j]=-1;
    
      long long int cost;
      cin>>cost>>m;
      for(long long int i=0;i<m;i++)
          cin>>coin[i];
      cout<<cool(cost,0)<<endl;
    
    return 0;
}
