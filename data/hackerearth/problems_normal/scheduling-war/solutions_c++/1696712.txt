int main()
{
    int nmbr;
    scanf("%d",&nmbr);
    int array[nmbr+1];
    for(int i=1;i<=nmbr;i++)
      scanf("%d",&array[i]);
      
      int b_c;
      scanf("%d",&b_c);
      int ss,v[110]={0},d1=0,d2=0;
      
      while(b_c--)
      { 
      	scanf("%d",&ss);
        int array2[ss+1],p,l_s1=0,l_s2=0;
        for(int j=1;j<=ss;j++)
      	   {scanf("%d",&array2[j]);
      	    
      	    
      	    if(v[array2[j]]==0) 
      	    {   
      	        l_s1+=array[array2[j]];
      	    	v[array2[j]]++;
      	        
      	    }
      	    l_s2+=array[array2[j]];
      	   
      	   }
      	  
      	   
      	  scanf("%d",&p);
          if(l_s1>=p) d1-=l_s1-p;
            else d1+=(p-l_s1);
            
            if(p>=l_s2) d2+=(p-l_s2); 
           
      }
    if(d2>d1) cout<<d2;
    else
    cout<<d1;
    
    //cout << "Hello World!" << endl;
    return 0;
}
