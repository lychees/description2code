#include<bits/stdc++.h>
#define ll long long int
#define MAX 10000009
#define MOD 1000000007
using namespace std;

vector<ll>v,v1;

void factor(ll N)
{
    ll i,ans=0;
    v.clear();
    for(i=2; i*i<=N; i++)
    {
        if(N%i==0)
        {
            while(N%i==0)
            {
                N=N/i;
                v.push_back(i);
            }
        }
    }
    if(N>1)
        v.push_back(N);
}

int main()
{
    ll i,t,T,n,k,a,b,ans;
    scanf("%lld",&T);
    for(t=1; t<=T; t++)
    {
        scanf("%lld %lld",&n,&k);
        factor(k);
        ans=0;
        v1.clear();
        while(v1.size()<n)
            v1.push_back(1);

        sort(v.rbegin(),v.rend());

        for(i=0;i<v.size();i++)
        {
            sort(v1.begin(),v1.end());

            a=v1[0];
            b=v[i];

            v1.erase(v1.begin());
            v1.push_back(a*b);
        }
        for(i=0; i<n; i++)
            ans+=v1[i];
        printf("%lld\n",ans);
    }
    return 0;
}
