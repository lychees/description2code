#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,a,b,c;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>c;
    	int d,e,x;
    	d=c*c;
		e=2*a*b;
		if((a+b)<=c)
		{	d=1;
			e=1;}
		else
		{
			if(a<c)
				d-=(c-a)*(c-a);
			if(b<c)
				d-=(c-b)*(c-b);
		}
		x=__gcd(d,e);
		d/=x;
		e/=x;
		cout<<d<<'/'<<e<<endl;
    	
    }
    return 0;
}
