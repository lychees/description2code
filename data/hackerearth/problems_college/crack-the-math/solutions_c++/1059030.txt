#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    string A,B;
    cin>>t;
    while(t--)
    {
    	cin>>A>>B;
    	for(int i=0;i<A.length();i+=1)
    	{
    		if(A[i]!=B[i])
    		printf("1");
    		else printf("0");
    	}
    	cout<<endl;
    }
    
    return 0;
}
