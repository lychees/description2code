#include <bits/stdc++.h>
using namespace std;


int main()
{
	int t;
	cin >> t;
	while (t > 0){
		int n;
		cin >> n;
		int val;
		int total = 0;
		map<int,int> mymap;
		for (int i = 0; i < n; i++){
			cin >> val;
			if (mymap.find(val) == mymap.end()){
				total++;
				mymap[val] = val;
			}
		}
		cout << total << endl;
		t--;
	}
    return 0;
}
