#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long count0,count1,i,count_prev0,count_prev1;
    int t,m;
    count0=0;
    count1=0;
    cin>>t;
    while(t--)
    {
    	count1=0,count_prev1=0;
    	count0=0,count_prev0=0;
    	cin>>m;
    	for(i=1;i<=m;i++)
    	{
    		if(i==1)
    		{
    			count1=0;
    			count0=1;
    			count_prev0=1;
    		}
    		else
    		{
    			count1+=count_prev0-count_prev1;
    			count0+=count_prev1;
    			count_prev0=count0;
    			count_prev1=count1;
    		}
    	}
    	cout<<count1<<" "<<count0<<"\n";
    }
    return 0;
}
