n, m = map(int, raw_input().split())
mat = []
for i in range(n):
    mat.append(map(int, raw_input().split()))
rowsum = [sum(x) for x in mat]
# print rowsum
colsum = [0] * m
for i in range(m):
    for x in mat:
        colsum[i] += x[i]
# print colsum
maxxor = 0
for i in rowsum:
    for j in colsum:
        if maxxor < (i ^ j):
            maxxor = (i ^ j)
print maxxor
