#include <iostream>
#include<vector>
#include <map>

using namespace std;

int main()
{
    std::map<unsigned long long int,unsigned long long int> hash;
    unsigned long long int N,M;
    cin>>N>>M;
    unsigned long long int max =0;
    unsigned long long int max_age;
    for(int i=0;i<N;i++)
    {
    unsigned long long int temp; 	
    scanf("%ull",&temp);
    hash[temp]++;
    if(hash[temp]>max)
    {
    	max = hash[temp];
    	max_age = temp;
    }
    else if(hash[temp] == max) {
    	if(temp>max_age)
    	max_age = temp;
    	
    }
    printf("%llu %llu\n",max_age,hash[max_age]);
    }
    
    
    return 0;
}
