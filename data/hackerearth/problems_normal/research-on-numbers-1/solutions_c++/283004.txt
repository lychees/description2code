int64 c[10];
int64 a[1000006];
vector<int>v;
int main(){
	int t,i,n,q,k,j;
	cin>>t;
	while(t--){
		cin>>q>>k;
		for(i=1;i<=k;i++)
		cin>>a[i];
		for(i=1;i<=k;i++)
		cin>>c[i];
		for(i=k+1;i<=1000000;i++){
			int64 val=0;
			for(j=1;j<=k;j++){
				val+=c[j]*a[i-j];
				if(val>=M)
				val%=M;
			}
			a[i]=val;
		}
		int l,r;
		while(q--){
			cin>>l>>r;
			int cnt=0;
			//vector<int>v;
			for(i=l;i<=r;i++){
				v.pb(a[i]);
				cnt++;
				if(cnt==100)
				break;
			}
			i++;
			make_heap(all(v));
			for(j=i;j<=r;j++){
				if(a[j]<v.front()){
					pop_heap(all(v));
					v.pob();
					v.pb(a[j]);
					push_heap(all(v));
				}
			}
			sort_heap(all(v));

			cout<<v[i]<<" ";
