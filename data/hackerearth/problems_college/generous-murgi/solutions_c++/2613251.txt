#include<bits/stdc++.h>
using namespace std;
int main()
{
int t,cnt,x,y,i;
scanf("%d",&t);
for(;t>0;t--)
{
scanf("%d%d",&x,&y);
if(x==y)
{printf("-1\n");continue;}
cnt=0;
set<int> st;
set<int>::iterator it;
int tt=x-y;
for(i=1;i<=sqrt(tt)+1;i++)
{
    if(tt%i==0)
        {
            st.insert(i);
            st.insert( (tt/i) );
            //printf("%d %d\n",i,(tt/i) );
        }
}
for(it=st.begin();it!=st.end();it++)
{
    //printf("%d %d %d\n",x,*it,x%(*it));
      if(x%(*it)==y)
       cnt++;
}
printf("%d\n",cnt);
}
return 0;
}
