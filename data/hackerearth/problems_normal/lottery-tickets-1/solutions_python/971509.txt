def noncollide(a,b):
	if a[1] <= b[0]:
		return 1
	return 0
t=int(raw_input())
for sdgsdg in xrange(t):
	n=int(raw_input())
	a=[]
	for i in xrange(n):
		a.append([int(e) for e in raw_input().split(' ')])
	a.sort(key= lambda x:x[1])
	pr=[e[2] for e in a]
	for i in xrange(n):
		for j in xrange(i+1,n):
			if noncollide(a[i],a[j])==1:
				if pr[j] < pr[i]+a[j][2]:
					pr[j]=pr[i]+a[j][2]
	print "%.2f"%(max(pr)/100.0)
