#include <bits/stdc++.h>
using namespace std;
#define BIG 1000000
int dp[1000][1000], a[1000];
int m,p;
inline int foo(int index, int packs)
{
    if(index == p)
		return 0;
	if(packs == m)
		return BIG;
	if(dp[index][packs] != -1)
		return dp[index][packs];
	int i,sum=0,ans=INT_MAX;

	for(i = index;i<p;i++)
    {
		sum += a[i];
		ans = min(ans,max(sum,foo(i+1,packs+1)));
	}
	dp[index][packs] = ans;
    //printf("%d %d %d\n",index,packs,ans);
	return ans;
}
int main()
{
	int i;
	 scanf("%d%d", &m, &p);
	memset(dp, -1, sizeof(dp));
	for(i = 0; i < p; i++)
		scanf("%d", a + i);
	printf("%d\n",foo(0,0));
	/*for(i=0;i<p;i++)
    {
        for(int j=0;j<p;j++)
            printf("%d ",dp[i][j]);
        printf("\n");
    }*/
	return 0;
}
