const int N = 1e5 + 5;
const int MOD = 1e9 + 7;

int dp[N][30];

void form(string s) {
	int n = s.size();
	for(int i = 0;i < 26;i++)
		dp[n - 1][i] = -1;
	dp[n - 1][s[n - 1] - 'a'] = n - 1;

	for(int i = n - 2;i >= 0;i--) {
		for(int j = 0;j < 26;j++) {
			dp[i][j] = dp[i + 1][j];
		}
		dp[i][s[i] - 'a'] = i;
	}
}


bool check(string s , int n) {
	int ind = 0;
	for(int i = 0;i < s.size();i++) {
		if (ind >= n)
			return false;
		ind = dp[ind][s[i] - 'a'];
		if (ind == -1)
			return false;
		ind++;
	}
	return true;
}

int main(int argc, char const *argv[])
{
	fast;

	// FILE;

	string s;
	cin >> s;

	form(s);
	

	int q;
	cin >> q;

	while (q--) {
		string x;
		cin >> x;
		if (check(x , s.size()))
			cout << "Yes\n";
		else
			cout << "No\n";
	}

	return 0;
}
