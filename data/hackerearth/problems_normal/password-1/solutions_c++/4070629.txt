#include <iostream>
#include<set>
using namespace std;
string rev(string s)
{
	string t="";
	for(int i=s.length()-1;i>=0;i--)
	{
		t+=s[i];
	}
	return t;
}
int main()
{
    set<string>s;int i,n;string p[200];
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>p[i];
    	s.insert(p[i]);
    }
    for(i=0;i<n;i++)
    {
    	string t=rev(p[i]);
    	if((s.find(t)!=s.end()))
    	{
    		cout<<p[i].length()<<" "<<p[i][(p[i].length())/2];
    		return 0;
    	}
    }
    return 0;
}
