#include<iostream>
#include<stdlib.h>
#include<string.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t)
	{
	int a[10000]={0},n,i=0,carry=0,cnt=0;
	cin>>n;
	int num=n;
	while(n!=0)
	{
		a[i++]=n%10;
		n/=10;
	}
	int k=i;
	for(int j=num;j>1;j--,i=0)
	{
		do
		{
			carry=(j*a[i])+carry;
			a[i++]=carry%10;
			carry/=10;
		}while(i<=k);
		if(carry!=0)
		{
			do
			{
				a[i++]=carry%10;
				k++;
				carry/=10;
			}while(carry!=0);
		}
	}
	for(int j=k;j>-1;j--)
	{
		if(a[j]==4||a[j]==7)
			cnt++;
	}
	cout<<cnt<<endl;
	--t;
	}
}
