int arr[10], N , cnt=0;
bool take[10] ;
void way(int pos,int A,int B)
{
    if(B>A) return  ;
    if(pos>N)
    {
        cnt++;
        return ;
    }
    for(int i=1;i<=N;i++)
    {
        if(take[i]) continue ;
        take[i]=true;
        way(pos+1,A+arr[i],B);
        way(pos+1,A,B+arr[i]);
        take[i] = false;
    }
}
int main()
{
    //freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
    int total = 0 ;
    cin >> N ;
    for(int i=1;i<=N;i++)
    {
        cin >> arr[i] ;
        total+=arr[i];
    }
    sort(arr+1,arr+N+1);
    reverse(arr+1,arr+N+1);
    way(1,0,0);
    cout << cnt << ' '<< total << endl ;
    if(cnt<=total) puts("Got no way out!");
    else puts("We will win!");
    return 0;
}
