#include<bits/stdc++.h>
using namespace std;
main()
{
	int c=1,n,i,k=0;
	scanf("%d",&n);
	int a[n];
//	l=strlen(a);
	for(i=0;i<n;i++)
	{
	scanf("%d",&a[i]);	
    }
    sort(a,a+n);
    for(i=0;i<n;i++)
    {
    if(a[i]+k>4){
    c++;
    k=a[i]+k-4;
    }
    else
    k=a[i]+k;
    }
    cout<<c;
}
