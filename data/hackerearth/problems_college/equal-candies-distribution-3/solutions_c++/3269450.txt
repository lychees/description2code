    #include <iostream>
    using namespace std;
    
    long long int cal(long long int t)
    {
        long long int num=0;
        if(t>=5)
        {
            num = t/5;
            t = t%5;
        }
        if(t>=2)
        {
            num += t/2;
            t = t%2;
        }
        num += t;
        return num;
    }
    
    int arr_sm(long long int array[],int array_len)
    {
        long long int smallest = 200000000;
        long long int i;
        for (i = 0; i < array_len; i++)
        {
            if (array[i] < smallest) {
                smallest = array[i];
            }
        }
        return smallest;
    }
    long long int mod(long long int num)
    {
        if(num>0)
            return num;
        else
            return (-1)*num;
    }
    int main()
    {
        long long int T,N,i,j,min,sum,t;
        cin>>T;
        while(T--)
        {
            min = 1000000;
            cin>>N;
            int A[N];
            for(i=0 ; i< N ; i++)
            {
                cin>>A[i];
                if(A[i]<min)
                    min = A[i];
            }
            long long int sum[6];
            for(j=0 ; j<=5 ; j++)
            {
                sum[j]=0;
                for(i=0 ; i< N ; i++)
                {
                    t = mod(A[i] - (min-j));
                    sum[j] = sum[j] + cal(t);
                }
            }
            cout<<arr_sm(sum,6)<<endl;
        }
        return 0;
    }
