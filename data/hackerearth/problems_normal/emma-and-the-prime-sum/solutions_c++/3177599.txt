//emma and the prime numbers
#include<cstdio>
#include<math.h>
using namespace std;

#define max 1000005

long long int temp[max];

int main()
{
	int t,i,j;
	temp[0]=0;
	temp[1]=0; 
	temp[2]=2;
	temp[3]=3+2;
	temp[4]=5; 
	temp[5]=5+5;
	temp[6]=10; 
	temp[7]=17;
	//cin>>t;
	
	for(i=8; i<=max; i++)
	{
		if(i%2==0 || i%3==0 || i%5==0 || i%7==0)
		    temp[i]=temp[i-1];
		else
		{   int pos=0;
			for(j=2; j<=sqrt(i); j++)
			{
			    if(i%j==0)
			    {
			    	pos=1;
			    	break;
				}
		    }
		    if(pos==0)
		        temp[i]=i+temp[i-1]; 
		    else
		        temp[i]=temp[i-1];
		}
	}	
  
	scanf("%d",&t);
	
	while(t--)
	{
		int x,y;
		scanf("%d%d",&x,&y);
		printf("%lld\n",temp[y]-temp[x-1]);  
	}
	//for(i=1; i<=max; i++)
	//   printf("%d\n",temp[i]);
	
	return 0;
}
