#include<iostream>
#include<vector>
#include<cstdio>
#include<cstring>
#include <sstream>
#include <algorithm>
#include <iterator>
#include<set>
using namespace std;
int main()
{
    int n;
    scanf("%d", &n);
    int s1[n];
    bool hh[500000];
    int ans = 0;


    for(int i = 0; i < n; i++) {cin>>s1[i];
        for(int j = 0; j < i; j++) {
            if(hh[200000 + s1[i] - s1[j]] == 1) {ans++; break;}
        }
        for(int j = 0; j <= i; j++) hh[200000 + s1[j] + s1[i]] = 1;
    }
    cout<<ans<<endl;
    return 0;
}
