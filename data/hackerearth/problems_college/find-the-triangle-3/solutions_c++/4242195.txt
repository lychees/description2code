#include <bits/stdc++.h>
using namespace std;
int rcomp(int a,int b){
    return a>b;
}
int a[100005];
int main() {
	int t;
	cin>>t;
	while(t--){
	    int n,i;
	    cin>>n;
	    for(i=0;i<n;i++){
	        cin>>a[i];
	    }
	    sort(a,a+n,rcomp);
	    for(i=0;i<=n-3;i++){
	        if(a[i+1]+a[i+2]>a[i]){
	            printf("%d %d %d\n",a[i+2],a[i+1],a[i]);
	            break;
	        }
	    }
	    if(i==n-2)
	        cout<<"-1\n";
	}
	return 0;
}
