lis = dict()  # gives the index in list to givver letter
for i in range(input()):
    name = raw_input()
    first_letter = name[0]

    # checking if letter in list
    if first_letter in lis:
        lis[first_letter] += 1
    else:
        lis[first_letter] = 1
count = len(lis)
for j in lis:
    cout = lis[j]
    if cout % 10 == 0:
        count += (lis[j]/10)-1
    else:
        count += lis[j]/10

print count
