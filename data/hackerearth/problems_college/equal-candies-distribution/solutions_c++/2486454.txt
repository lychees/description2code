#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;

typedef long long ll;

int main()
{
    int T;
    scanf("%d", &T);
    while(T--) {
    	int N;
    	scanf("%d", &N);
    	int a[N], b[N], s, c=0;
    	
    	for(int i=0; i<N; i++)
    		scanf("%d", &a[i]);
    	sort(a, a+N);
    	for(int i=0; i<N; i++)
    		b[i] = a[i] - a[0];
    	
    	for(int i=0; i<N; i++) {
    		s = b[i];
    		c = c + s/5;
    		s = s%5;
    		c = c + s/2;
    		s = s%2;
    		c = c + s/1;
    	}
    	printf("%d\n", c);
    }
    return 0;
}
