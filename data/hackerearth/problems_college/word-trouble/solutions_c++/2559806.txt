#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int T,i,j,k,c;
	string s1,s2;
	cin>>T;
	while(T--)
	{
		cin>>s1;
		cin>>s2;
		if(s1.length()!=s2.length())
		{
			i=0;
			c=0;
			while(i<min(s1.length(),s2.length()))
			{
				if(s1[i]!=s2[i])
				c++;
				i++;
			}
			c+=max(s1.length(),s2.length())-min(s1.length(),s2.length());
			cout<<"Non Exact "<<c<<"\n";
		}
		else
		{
			i=0;
			c=0;
			while(i<s1.length())
			{
				if(s1[i]!=s2[i])
				c++;
				i++;
			}
			if(c==0)
			cout<<"Exact "<<"\n";
			else
			cout<<"Non Exact "<<c<<"\n";
		}
	}
}
