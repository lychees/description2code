#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    char s[100],tmp[100];
    cin>>s;
    int len=strlen(s),flag=1;
    for(int i=0;i<len;i++)
    tmp[len-i-1]=s[i];
   // cout<<s<<"\n";
    //cout<<tmp;
    for(int j=0;j<len;j++)
    {
    	if(tmp[j]!=s[j])
    	{
    		flag=0;
    		break;
    	}
    }
    if(flag)
    cout<<"YES";
    else
    cout<<"NO";
    
    return 0;
}
