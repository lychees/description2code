using namespace std;

{

    l=s[j];
   // cout<<l<<" ";
    	
    	if(64<l&&l<91)

    			l=l-32;
    //	else
    	//	l=l;
    //cout<<l<<endl;
    s[j]=l;
    //cout<<s[j];
    	j--;
    }

}
