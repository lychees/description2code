#include <iostream>
using namespace std;
int arr[100000]={0};
int ln =0;
int main()
{
	int t,n,i,j,x;
	scanf("%d",&t);
	while(t--)
	{
		cin>>n;
		ln=0;
		for(j=0;j<n;++j)
		{
			scanf("%d",&x);
			i=x-j;
			if(i>0)
			{
				 int l=0;
				 int r= ln-1;
				 int y=-1;
				 while(l<=r)
				 {
				 	int middile = (l+r)/2;
				 	if(arr[middile]<=i)
				 		l=middile+1;
				 	else
				 	{
				 		y=middile;
				 		r=middile-1;
				 	}
				 }
				 int b;
				 if(y>=0)b=y;
				 else
				 {
				 	
				 	
				 	b=ln;
				 	++ln;
				 }
				 arr[b]=i;
			}
		}
		cout<<n-ln<<endl;
		
	}
}
