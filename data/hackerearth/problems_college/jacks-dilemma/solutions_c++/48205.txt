#define ll long long

vector< vector<long long> > C(1000+1,vector<long long> (1000+1,0));

long long Co(int n, int r)
{

    for (int i=0; i<=n; i++)
    {
        for (int k=0; k<=r && k<=i; k++)
            if (k==0 || k==i)
                C[i][k] = 1;
            else
                C[i][k] = (C[i-1][k-1] + C[i-1][k]);
    }
    return C[n][r];
}
