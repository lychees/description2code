#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,m; cin>>n>>m;
	int a[n+1];
	int arr[n+1];
	memset(arr,0,sizeof arr);
	for(int i=0; i<n; i++) cin>>a[i];
	for(int i=0; i<n; i++){
		if(!(a[i]%m)) arr[a[i]/m]++;
	}
	int ans=0;
	for(int i=1; i<n; i++){
		if(!arr[i]) break;
		ans++;
	}
	cout<<ans*m<<endl;
}
