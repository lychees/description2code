import math

no_of_test = int(raw_input())
while no_of_test > 0:
    t_range = raw_input()
    t = long(t_range)
    n = long(math.ceil((-1+(math.ceil(math.sqrt(8*t-1))))/2))
    if n % 3 == 0:
        print 'SL'
    elif n % 3 == 1:
        print 'LB'
    else:
        print 'BS'
    no_of_test -= 1
