#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int c=0;
    	for(int i=1;i<=n-1;i++)
    	{
    		for(int j=i+1;j<=n;j++)
    	{
    			if((i^j)<=n)
    		c++;
    	}
    		
    	}
    	cout<<c<<endl;
    }
    return 0;
}
