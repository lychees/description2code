#include <iostream>
#include<map>

using namespace std;

int main()
{
    int t;
    cin >> t;
    map<char, int> Weight;
    Weight['a']=1;
	Weight['b']=2;
	Weight['c']=3;
	Weight['d']=4;
	Weight['e']=5;
	Weight['f']=6;
	Weight['g']=7;
	Weight['h']=8;
	Weight['i']=9;
	Weight['j']=10;
	Weight['k']=11;
	Weight['l']=12;
	Weight['m']=13;
	Weight['n']=14;
	Weight['o']=15;
	Weight['p']=16;
	Weight['q']=17;
	Weight['r']=18;
	Weight['s']=19;
	Weight['t']=20;
	Weight['u']=21;
	Weight['v']=22;
	Weight['w']=23;
	Weight['x']=24;
	Weight['y']=25;
	Weight['z']=26;
    while(t--)
    {
    	long long int k, curr_sum, count = 0, m = 0;
    	string s;
    	
    	cin >> k;
    	cin >> s;
    	
    	long long int len = s.size();
    	
    	int arr[len];
    	
    	for(int i = 0; i < len; i++)
    		arr[i] = Weight[s[i]];
    	
    	curr_sum = arr[0];
    	
    	for(int i = 1; i <= len; i++)
    	{
    		
    		while(curr_sum > k && m < len-1)
    		{
    			curr_sum = curr_sum - arr[m];
    			m++;
    		}
    		if(curr_sum == k)
    			count++;
    		if(i < len)
    		curr_sum += arr[i];
    	}
    	cout<<count<<endl;
    }
    return 0;
}
