const int N = 1e5 + 10;

void ZMatching(char *P, char*T, int *zz)
{
    int n = strlen(T + 1);
    int m = strlen(P + 1);
    int z[n + m + 10];
    ///Nhap T @ P
    char S[n + m + 10];
    strcpy(S + 1, P + 1);
    S[m + 1] = '#';///1 cai phan ko match voi cai gi
    strcpy(S + m + 2, T + 1);
  ////cout<<"size S " << strlen(S + 1) << endl;
 //   printf("%s\n", S + 1);
    int len = z[1] = n + m + 1;

    int L = 0, R = 0;

    for(int i = 2;i <= len;i ++)
    {
        if(i == 6)
        {
            //cout<< S[i] << endl;
            //cout<< L << " " << R << endl;
        }
        if(R < i)
        {
            L = R = i;
            while(R <= len && S[R] == S[R- L + 1]) R ++;
            z[i] = R- L;
            R--;
        }
        else
        {
            int k = i - L + 1;
            if(z[k] < R - i + 1)
            {
                z[i] = z[k];
            }
            else
            {
                L = i;
                while(R <= len && S[R] == S[R - L + 1]) R ++;
                z[i] = R - L;
                R --;
            }
        }
    }
    for(int i = m + 2; i <= len;i ++)
    {
        zz[i - m - 1] = z[i];
    }
}
char P[N], T[N];
char cT[N * 2];

int z[N];

int main()

        strcpy(cT + 1, T + 1);

        ZMatching(P, cT, z);
