#include <bits/stdc++.h>
#include<stdio.h>
using namespace std;

int main()
{
    int t, n, i, j, k=0;
    int a[1000009];
    int arr[1000009];
    cin >> t;
    while ( t-- )
    {
    	cin >> n;
    	for( i = 0; i < n; i++) {
    	scanf("%d",&a[i]);
    	}
    	sort( a, a+n );
    	for ( i = 0, k = 0; i <n; i++) {
    		if(a[i] != a[i+1]){
    			arr[k] = a[i];
    			k++;
    		}
    		else{
    			for ( j = i; a[j] == a[j+1]; j++) {
    				if(arr[k-1] != a[i]) {
    				arr[k] = a[i];
    				k++;
    				}
    				i = j;
    				i++;
    			}
    		}
    	}
    	/*for(i=0; i<n;i++)
    		printf("%d",a[i]);
    		printf("\n");
    	for(i=0; i<k;i++)
    		printf("%d",arr[i]);
    		printf("\n");*/
    	//printf("%d",k);
    	for ( i = 0, j = k-1; i < (k+1)/2; i++, j-- ){
    		if( i == j)
    			printf("%d",arr[i]);
    		else
    			printf("%d %d ",arr[j],arr[i]);
    	}
    	printf("\n");
    }
    return 0;
}
