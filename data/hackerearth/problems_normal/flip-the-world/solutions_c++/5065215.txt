#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
    	int n,m;
    	cin >> n >> m;
    	char ar[n][m];
    	for(int i = 0; i < n; i++){
    		for(int j = 0; j < m ; j++){
    			cin >> ar[i][j];
    		}
    	}
    	int count = 0;
    	for(int i = n-1; i >= 0; i--){
    		for(int j = m-1; j >= 0; j--){
    			if(ar[i][j] != '1'){
    				count++;
    				// flip the elements from 00 to i,j
    				for(int q = 0; q <= i; q++){
    					for(int w = 0; w <= j; w++){
    						if(ar[q][w] == '0') ar[q][w] = '1';
    						else ar[q][w] = '0';
    					}
    				}
    			}
    		}
    	}
    	cout << count << endl;
    }
    return 0;
}
