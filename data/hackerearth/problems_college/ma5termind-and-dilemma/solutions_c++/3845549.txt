#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
    int t,co,ci,a,b,i,n;
    string s;
    scanf("%d",&t);
    while(t--)
    {
        co=0;
        ci=0,a=0,b=0;
        scanf("%d",&n);
        cin>>s;
        for(i=0;i<n;i++)
        {
            if(s[i]=='0')
                co++;
            else if(s[i]=='1')
                ci++;
        }
        for(i=0;i<co;i++)
        {
            if(s[i]!='0')
                a++;

        }
        for(i=0;i<ci;i++)
            if(s[i]!='1')
                b++;
        printf("%d\n",min(a,b));
    }
}
