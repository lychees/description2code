#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
bool isPrime(ll x) {
	for(ll a = 2; a * a <= x; ++a)
		if(x % a == 0)
			return false;
	return true;
}
ll mypow(ll a, int b) {
	ll r = 1;
    while(b--) {
		r *= a;
    	if(r > 1e16) return r;
    }
	return r;
}
int main() {
    ll n;
    scanf("%lld", &n);
    	
    ll ans = n / 2 + 1;
    while(ans < n && !isPrime(ans)) ++ans;
    	
    for(int k = 2; mypow(2, k) <= n; ++k) {
		ll m = 1;
		while(mypow(m, k) <= n / 2)
			++m;
		while(m < n && !isPrime(m) && mypow(m, k) < ans) ++m;
    		
		ans = min(ans, mypow(m, k));
	}
	printf("%lld\n", ans);
	return 0;
}
