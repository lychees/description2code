using namespace std;
int id[100005];
int root(int x)
{
	while(x!=(id[x]))
	{
		id[x] = id[id[x]];
		x = id[x];
	}
	return x;
}
void union1(int x,int y)
{
	int a,b;
	a = root(x);
	b = root(y);
	id[b] = a;
}
int main()
{
	int i,j,x,y,n,k,ans=0;
    ios::sync_with_stdio(false);
    cin >> n >> k;
    for(int i=0;i<=n;i++)
    id[i] = i;
    for(i=1;i<=k;i++)
    {
    	cin >> x >> y;
    	union1(x,y);


    }
    for(i=1;i<=n;i++)
       if(id[i]==i)
         ans++;
    cout << ans << endl;
}
