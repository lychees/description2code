#include <bits/stdc++.h>
using namespace std;

int a[100005][2];

void preorder(int x){
	cout<<x<<" ";
	if(a[x][0]) preorder(a[x][0]);
	if(a[x][1]) preorder(a[x][1]);
}

int main(){
	int t,n,x,y,z;
	cin>>t;
	while(t--){
		cin>>n;
		for(int i=0;i<100005;i++){
			a[i][0]=a[i][1]=0;
		}
		for(int i=1;i<=n;i++){
			cin>>x>>y>>z;
			a[x][0]=y;
			a[x][1]=z;
		}
		preorder(1);
		cout<<endl;
	}
	return 0;
}
