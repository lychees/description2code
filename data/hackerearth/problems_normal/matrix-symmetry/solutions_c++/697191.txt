#include <iostream>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {	
    	int v=1,h=1;
    	cin>>n;
    	char a[n][n];
    	for(int i=0;i<n;i++)
    		for(int j=0;j<n;j++)
    			cin>>a[i][j];
    	for(int i=0;i<n;i++)
    	
    		for(int j=0;j<n;j++)
    		{
    			if(a[i][j]!=a[n-1-i][j])
    				h=0;
    			if(a[j][i]!=a[j][n-1-i])
    				v=0;
    		}
    	if(h==1 && v==0)
    		cout<<"HORIZONTAL"<<endl;
    	else if(h==0 && v==1)
    		cout<<"VERTICAL"<<endl;
    	else if(h==0 && v==0)
    		cout<<"NO"<<endl;
    	else
    		cout<<"BOTH"<<endl;
    	
    }
    return 0;
}
