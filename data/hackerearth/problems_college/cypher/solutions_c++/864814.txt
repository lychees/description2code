#include <iostream>
#include <cstring>
#include <bits/stdc++.h>

using namespace std;
map<char,int> alph;
int main()
{
	for(int i=0;i<26;i++)
	{
	 	alph[97+i]=i;
	}
	int tc;
	cin>>tc;
	while(tc--)
	{
		string s;
		cin>>s;
		int i=1;
		while(s[i])
		{
			char val=s[i];
			int v=alph[val];
			v=(v+i)%26;
			s[i]=97+v;
			i++;
		}
		cout<<s<<endl;
	}

}
