const int oo = 1000000009;
const double eps = 1e-6;
const int mod = 1000000007;

ll p(ll cur, int n) {
    ll ret = 1;
    for(; n; n /= 2) {
        if(n & 1) {
            ret = ret * cur % mod;
        }
        cur = cur * cur % mod;
    }

    return ret;
}

ll R(int n) {
    if(n <= 1) return 1;
    if(n == 2) return 1;
    if(n <= 10) {
        ll ans = 1;
        for(int i = 3; i <= n && i < 9; i += 2) {
            ans *= i;
        }
        return ans;
    }
    return p(2, (n + 3) / 4) * R((n + 4) / 5) % mod * R((n + 9) / 10) % mod;
}

int main() {
    int t;
    scanf("%d", &t);
    while(t--) {
        int n;
        scanf("%d", &n);
        ll r = R(n);
        ll ans = r;
        //trace1(r);
        rep(i, 1, 5) rep(j, 1, 5) {
            ll a = i * r % mod, b = j * r % mod;
            if(__gcd(a, b) == r) ans = max(ans, p(a, b));
        }
        printf("%lld\n", ans);
    }
    
	return 0;
}
