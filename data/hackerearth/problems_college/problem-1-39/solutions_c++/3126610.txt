#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<math.h>
using namespace std;
typedef unsigned long long ll;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        ll a,b,c;
        cin>>a>>b>>c;
        ll d;
        d=pow(a,b);
        ll e=d%c;
        if(e<(c/2))
        {
            if((d+e)%c==0)
                cout<<d+e<<endl;
            else
                cout<<d-e<<endl;
        }
        else if(e==(c/2))
        {
            cout<<d-e<<endl;
        }
        else
        {
            e=c-e;
            if((d+e)%c==0)
                cout<<d+e<<endl;
            else
                cout<<d-e<<endl;
        }
    }
    return 0;
}
