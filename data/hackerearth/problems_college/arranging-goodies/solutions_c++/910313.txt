#include <iostream>
using namespace std;

int arr[100000];
int main()
{
    int n,m;
    cin>>n>>m;
    if(m&1)
    {
    	int start=(m+1)/2,p,i=0,inc=1;
    	p=start;
    	while(p>0)
    	{
    		arr[i++]=p;
    		if(p>=start) p-=inc;
    		else p+=inc;
    		inc++;
    	}
    }
    else
    {
    	int start=(m)/2,p,i=0,inc=1;
    	p=start;
    	while(p>0)
    	{
    		arr[i++]=p;
    		if(p>start) p-=inc;
    		else p+=inc;
    		inc++;
    	}
    }
    int j=0;
    for(int i=0;i<n;++i)
    {
    	if(j==m) j=0;
    	cout<<arr[j]<<endl;
    	++j;
    }
    
    return 0;
}
