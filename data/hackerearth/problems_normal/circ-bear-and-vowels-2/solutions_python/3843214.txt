from sys import stdin
class Solution:
	def solve(self, ls, string):
		vovs = 0
		cons = 0
		last = ''
		conts = 0
		for c in string:
			if c in ls:
				vovs += 1
				conts = 0
			else:
				cons += 1
				conts += 1
				if conts == 3:
					return False
		if vovs < cons:
			return False
		return True

T = int(stdin.readline().strip())
s = Solution()
ls = ['a', 'e', 'i', 'o', 'u', 'y']
for _ in range(T):
	string = stdin.readline().strip()
	if s.solve(ls, string):
		print "easy"
	else:
		print "hard"
