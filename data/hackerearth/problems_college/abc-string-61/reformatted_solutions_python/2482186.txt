import sys
if sys.version_info[0] == 2:
    input = raw_input
t = int(input())
mod = 1000000007


def modpow(x, n, m):
    if n == 0:
        return 1
    elif n == 1:
        return x
    elif n % 2 == 0:
        return modpow(x*(x % m), n/2, m) % m
    elif n % 2 == 1:
        return (x*modpow(x*(x % m), (n-1)/2, m) % m) % m


while t > 0:
    n = int(input())
    if n < 3:
        print(0)
    else:
        print((modpow(3, n, mod)-3*modpow(2, n, mod)+3) % mod)
    t -= 1
