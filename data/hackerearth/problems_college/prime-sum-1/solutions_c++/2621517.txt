#include<bits/stdc++.h>

using namespace std;

const int N=1234567;
const int INF=2e9;

int f[N];

void prepare() {
    f[2]=1;
    f[1]=INF;
    for(int i=3;i<N;i++) {
        f[i]=INF;
        if(i>=2 && f[i-2]!=INF) {
            f[i]=min(f[i], f[i-2]+1);
        }
        if(i>=3 && f[i-3]!=INF) {
            f[i]=min(f[i], f[i-3]+1);
        }
        if(i>=5 && f[i-5]!=INF) {
            f[i]=min(f[i], f[i-5]+1);
        }
        if(i>=7 && f[i-7]!=INF) {
            f[i]=min(f[i], f[i-7]+1);
        }
    }
}

int main() {
    int test;
    scanf("%d", &test);
    prepare();
    while(test--) {
        int n;
        scanf("%d", &n);
        printf("%d\n", f[n]);
    }
}
