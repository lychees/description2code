#include<iostream>
#include<math.h>
using namespace std;
int ar[1000002];
void prime()
{
ar[1]=-1;
for(int i=2;i<=sqrt(1000002);i++)
{
   if(ar[i]==0)
   {
      //ar[i]=1;
       for(int j=2*i;j<=1000002;j+=i)
         ar[j]=-1;
   }
 }
}

main()
{
prime();
long long int sum[1000002];
sum[0]=0;
for(int i=1;i<1000002;i++)
{
  if(ar[i]==0)
    sum[i]=i+sum[i-1];
  else
    sum[i]=sum[i-1];
}
int t;
cin>>t;
for(int z=0;z<t;z++)
{
  int a,b;
  cin>>a>>b;
  cout<<sum[b]-sum[a-1]<<endl;
}

}
