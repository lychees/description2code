#include <iostream>
#include <string>
using namespace std;

int main()
{
	int t; 
	cin >> t;
	for (int i = 0; i < t; i++) {
		string str;
		cin >> str;
		
		bool delMode = false;
		string result;
		
		for (int j = 0; j < str.length(); j++) {
			char ch = str[j];
			
			switch (ch) {
				case 'o' :
					if (!delMode)  result += ch;
					break;
				
				case 'x' :
					//delete all the previously appended o's until  * is found 
					for (int k = result.length() - 1; k >= 0; k--) {
						if (result[k] == '*' || result[k] == 'x') break;
						if (result[k] == 'o') {
							result.erase(k);
						}
					}
					result += ch;
					delMode = true;
					break;
				case '*' :
					result += ch;
					delMode = false;
					break;
			}
		}
		cout << result << endl;
	}
    return 0;
}
