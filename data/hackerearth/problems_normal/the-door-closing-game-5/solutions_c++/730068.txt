#include<stdio.h>
#include<string>
#include<iostream>
using namespace std;
int a[1000];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string str;
        cin>>str;
        int j=0,count;
        for(int i=0; str[i]!='\0';)
        {
            count=0;
            while((str[i]!='\0')&&(str[i]=='_'))
            {
                count++;
                i++;
            }
            if(count!=0)
            {
                a[j]=count;
                j++;
            }
            if(str[i]=='\0')
                break;
            i++;
        }
        if(j==1&&a[j-1]==str.length())
        {

            cout<<"Bob"<<endl;
            continue;
        }
        int x=0;
        for(int i=0; i<j; i++)
        {

            x=x^a[i];
        }
        if(x==0)
            cout<<"Bob"<<endl;
        else
            cout<<"Alice"<<endl;

    }

}
