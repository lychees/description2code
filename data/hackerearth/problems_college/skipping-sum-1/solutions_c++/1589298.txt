#include <iostream>
#include<vector>
#include<cstring>
#include<algorithm>

using namespace std;

long long ans[100005][12];
vector<int>v(100005);
long long brute(int k,int a,int b){
	long long sum=0;
	while(a<=b){
		sum+=v[a];
		a+=k;
	}
	return sum;
}
int main()
{
     int N,Q;
     cin>>N>>Q;
     for(int i=0;i<N;i++)cin>>v[i];
     
     memset(ans,0,sizeof(ans));
     for(int i=N-1;i>=0;i--){
     	for(int j=1;j<=10;j++){
     		ans[i][j]=i+j<=N-1?ans[i+j][j]+v[i]:v[i];
     	}
     }
      
     int a,b,k;
     
     while(Q--){
     	cin>>a>>b>>k;
     	int x=(b-a)/k;
     	x=a+x*k;
     	
     	if(a==b){
     		cout<<v[a-1]<<endl;
     		continue;
     	}
     	cout<<ans[a-1][k]+((b-a)%k==0?v[b-1]-ans[b-1][k]:v[x-1]-ans[x-1][k])<<endl;
     
     
     }
    return 0;
}
