#include <iostream>
#include <math.h>
using namespace std;

int main() {
	long long int q,n,i,j,k,x;
	long long int b[50];
	b[1]=2;
	for(i=2;i<=30;i++){b[i]=2*b[i-1];}
	cin>>q;
	while(q--)
	{
		cin>>x;
		if(x==1)
		{
			cin>>n>>k;
			n=b[n];
			long long int p,y=1;
			i=1;j=1;
			while(y!=k)
			{
				p=(n*n)/4;
				if((k-y)/p==0){}
				else if((k-y)/p==1)
				{
					y=y+p;
					j=j+n/2;
				}
				else if((k-y)/p==2)
				{
					y=y+2*p;
					i=i+n/2;
				}
				else if((k-y)/p==3)
				{
					y=y+3*p;
					i=i+n/2;
					j=j+n/2;
				}
				n=n/2;
			}
			cout<<i<<" "<<j<<endl;
		}
		else if(x==2)
		{
			cin>>n>>i>>j;
			n=b[n];
			k=1;
			while(n!=1)
			{
				if(i<=n/2 && j<=n/2){}
				else if(i<=n/2 && j>n/2)
				{
					k=k+(n*n)/4;
					j=j-n/2;
				}
				else if(i>n/2 && j<=n/2)
				{
					k=k+(n*n)/2;
					i=i-n/2;
				}
				else if(i>n/2 && j>n/2)
				{
					k=k+3*(n*n)/4;
					i=i-n/2;
					j=j-n/2;
				}
				n=n/2;
			}
			cout<<k<<endl;
		}
	}
	return 0;
}
