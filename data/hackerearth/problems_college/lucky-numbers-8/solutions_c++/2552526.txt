#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,i,j,k,a[1000000];
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&a[i]);
    	}
    	k=0;
    	for(i=0;i<n-1;i++)
    	{
    		for(j=i+1;j<n;j++)
    		{
    			if(a[i]<=a[j])
    			{
    				break;
    			}
    		}
    		if(j==n)
    		k++;
    	}
    	printf("%d\n",k+1);
    }
    return 0;
}
