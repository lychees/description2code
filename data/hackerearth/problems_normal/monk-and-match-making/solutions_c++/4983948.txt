#include <cmath>
#include <cstdio>
#include <iostream>
#include <vector>

using namespace std;

typedef long long ll;

const ll p = 31;

int main() {
    int Q, s_size;
    string S;

    cin >> S >> Q;

    s_size = S.size();
    vector<ll> p_pow(s_size);
    p_pow[0] = 1;
    for (int i = 1; i < s_size; ++i)
        p_pow[i] = p_pow[i - 1] * p;

    vector<ll> hash(s_size);
    for (int i = 0; i < s_size; ++i) {
        hash[i] = (S[i] - 'a' + 1) * p_pow[i];
        if (i)
            hash[i] += hash[i - 1];
    }

    int L1, R1, L2, R2;
    ll H1, H2;
    for (int i = 0; i < Q; ++i) {
        cin >> L1 >> R1 >> L2 >> R2;
        L1--;
        R1--;
        L2--;
        R2--;

        H1 = hash[R1];
        if (L1)
            H1 -= hash[L1 - 1];
        H2 = hash[R2];
        if (L2)
            H2 -= hash[L2 - 1];

        if ((L1 < L2 && H1 * p_pow[L2 - L1] == H2) ||
            (L1 > L2 && H1 == H2 * p_pow[L1 - L2]) || (L1 == L2 && R1 == R2))
            cout << "Yes\n";
        else
            cout << "No\n";
    }

    return 0;
}
