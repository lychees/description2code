#include <iostream>
using namespace std;

int main()
{
    size_t Skipped = 0;
	size_t score = 0;
	size_t nQuestions = 0 ; 
	size_t DifficultyLevel = 0;
	
	std::cin >> nQuestions; 
	std::cin >> DifficultyLevel; 
	size_t* QuestionPaper; 
	QuestionPaper = new size_t[nQuestions];
	for ( size_t Loop = 0; Loop < nQuestions; ++Loop)
	{
		std::cin >> QuestionPaper[Loop];
		if (QuestionPaper[Loop] > DifficultyLevel)
		{
			if (++Skipped > 1)
			{
				break;
			}
		}
		else
		{
			++score;
		}
	}
	std::cout << score;
	delete[] QuestionPaper; 
    return 0;
}
