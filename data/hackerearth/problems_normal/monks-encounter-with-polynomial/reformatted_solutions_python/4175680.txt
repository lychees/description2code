case = int(raw_input())
while case > 0:
    case = case - 1
    a, b, c, k = map(int, raw_input().split())
    m = 0
    n = 100000
    ans = 0
    while m <= n:
        mid = (m+n)/2
        val = a*(mid*mid) + b*mid + c
        if val >= k:
            ans = mid
            n = mid - 1
        else:
            m = mid + 1
    print ans
