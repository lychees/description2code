#include <cstdio>
#define MOD 1000003
#define MAXN 100000
#define MAXX 1000000000000000000LL
long long ar[MOD+1];
void init(){
	ar[0]=1;
	for(int i=1;i<=MOD;i++)
		ar[i] = (ar[i-1]*(long long)i)%MOD;
}
int main()
{
	int T;
	long long N, A;
	init();
	scanf("%lld",&T);
	while(T--){
		scanf("%lld %lld",&N,&A);
		A = A % MOD;
		if(N>MOD)
			N = MOD;
		printf("%lld\n",(ar[N]*A)%MOD);
	}
    return 0;
}
