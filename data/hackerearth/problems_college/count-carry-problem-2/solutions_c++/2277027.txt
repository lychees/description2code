#include <iostream>
using namespace std;
#include<cstring>
int main()
{
	long long int t,cnt;
	char s[1000001];
	char str[1000001];
	long long int i,j;
    cin>>t;
    while(t--)  {
    	cnt=0;
    	scanf("\n%s %s",s,str);
    	i=strlen(s);
    	i-=1;
    	j=strlen(str);
    	j-=1;
    	while(i>=0 && j>=0) {
    		if(((str[j]-'0')+(s[i]-'0'))>9)
    			cnt++;
    		i--;
    		j--;
    	}
    	if(cnt==0)
    		printf("No carry operation\n");
    	else if(cnt==1)
    		printf("%lld carry operation\n",cnt);
    	else
    		printf("%lld carry operations\n",cnt);
    }
    return 0;
}
