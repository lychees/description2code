#include <iostream>
using namespace std;

int main()
{
   long long int a[100000];
   int n,max=0,sum=0;
   cin>>n;
   for(int i=0;i<n;i++)
   {
   	cin>>a[i];
   	sum=sum+a[i];;
   	 if(sum<0)
   	 {
   	 	sum=0;
   	 }
   if(sum>max)
   {
   	max=sum;
   	
   }
   }
   
 if(max>0)	
   cout<<max<<endl;
   else 
    cout<<"0"<<endl;
    return 0;
}
