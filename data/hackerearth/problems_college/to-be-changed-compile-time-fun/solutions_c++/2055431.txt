#include <stdio.h>
int main()
{
	int n;
	scanf ("%d",&n);
	int a[1000],i;
	for(i=0;i<n;i++)
	{
		scanf ("%d",&a[i]);
	}
	long long ans=0;
	while (n>=3)
	{
		int max=-1,index;
		for(i=1;i<n-1;i++)
		{
			if(a[i-1]*a[i+1]>max)
			{
				max=a[i-1]*a[i+1];
				index=i;
			}
		}
		for(i=index;i<n-1;i++)
			a[i]=a[i+1];
		ans+=max;
		n--;
	}
	printf ("%lld",ans);
	return 0;	
}
