void sieve()
{
	mset(prime,true);
	prime[0]=prime[1]=false;
	for(int i=4;i<1000001;i+=2)
		prime[i]=false;

	for(int i=3;i*i<1000001;i+=2)
	{
		if(prime[i])
		{
			for(int j=i*i;j<1000001;j+=i)
			{
				if(prime[j])
					prime[j]=false;
			}
		}
	}
