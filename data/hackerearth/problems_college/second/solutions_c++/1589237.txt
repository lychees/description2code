using namespace std;

int main() {
	// your code goes here
	int c,n,m,a[100010]={0};
	cin>>n>>m;
	for(int i=0;i<n;i++)
	{
		cin>>c;
		a[c]++;
	}
	if(a[m]==0)
	cout<<"0";
	else
	{
		c=m;
		while(a[c]!=0)
		{
			c=m+c;
		}
	//	cout<<m<<" ";
		//m=m/2;
		c=c-m;
		cout<<c;
	}
	cout<<endl;
	return 0;
}
