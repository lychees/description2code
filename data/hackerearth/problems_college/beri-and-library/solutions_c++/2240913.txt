#include<bits/stdc++.h>
#define lli long long int
using namespace std;
int main()
{
        lli a,count,i,j,k,m,n,t;
        scanf("%lld",&t);
        while(t--)
        {
                scanf("%lld",&n);
                count=0;
                a=1;
                while(n!=0)
                {
                        a=a*2;
                        if(a>n)
                        {
                                count++;
                                n=n-a/2;
                                a=1;
                        }
                }
                printf("%lld\n",count);
        }
}
