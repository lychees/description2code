#include "iostream" 
#include "bits/stdc++.h"

using namespace std;

bool isSubstr(string a,string b){

	if(b.length()>a.length()) return false;

	int l = b.length();
		for(int i=0;i+l<=a.length();i++){
			string s = a.substr(i,l);
			if(s==b) return true;
		}
	return false;
	
}

{
