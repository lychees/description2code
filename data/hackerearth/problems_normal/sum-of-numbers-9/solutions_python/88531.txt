from sys import stdin

def can_make_sum(numbers, index, target):
  if index == len(numbers):
    return False
  number = numbers[index]
  if number == target:
    return True
  return can_make_sum(numbers, index+1, target) or can_make_sum(numbers, index+1, target - number)

for case in range(int(stdin.readline())):
  stdin.readline()
  numbers = [int(number) for number in stdin.readline().split()]
  target = int(stdin.readline())
  print 'YES' if can_make_sum(numbers, 0, target) else 'NO'
