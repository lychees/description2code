#include <iostream>
using namespace std;


int main()
{
    int t, n, a[100005];
    cin >> t;
    while(t--) {
    	int b[100005] = {0};
    	cin >> n;
    	for (int i = 0; i < n; i++) {
    		scanf("%d", &a[i]);
    		b[a[i]] = i + 1;
    	}
    	int j = 0;
    	for(j = 1; j <= n; j++) {
    		if (b[j] != a[j - 1]) {
    			break;
    		}
    	}
    	if (j == n + 1)
    		cout << "inverse" <<endl;
    	else 
    		cout << "not inverse" <<endl;
     		
    }
    return 0;
}
