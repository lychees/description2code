#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    
    
    int t=0;
    cin>>t;
    long ans=0;
    long cur[t+1];
    cur[0]=-1;
    long temp = 0;
    for(long i=1;i<=t;++i)
    {
    	cin>>cur[i];
    }
   sort(cur,cur+t+1);
    long k=0;
     for(long i=1;i<=t;)
    {
    
    	if(k*k >= cur[i])
    	{
    		temp = k*k - cur[i];
    		++i;
    		++k;
    	}
    	else
    	{
    			temp = k*k;
    			++k;
    		
    		
    	}
    	if(temp>0)
    		ans+=temp;
    }
    cout<<ans<<endl;
    return 0;
}
