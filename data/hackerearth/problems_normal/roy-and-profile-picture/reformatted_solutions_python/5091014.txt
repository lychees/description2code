L = int(raw_input())
N = int(raw_input())

for i in range(N):
    W, H = [int(x) for x in raw_input().split(" ")]
    if W >= L and H >= L:
        if W != H:
            print "CROP IT"
        elif W == H:
            print "ACCEPTED"
    else:
        print "UPLOAD ANOTHER"
