#include <cstdio>
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	int n,k;
	    cin>>n;
	    int a[1000001];
	    for(int i=0;i<n;i++)
	    {
	    	cin>>a[i];
	    }
	    cin>>k;
	    vector<int> st;
	    for(int i=0;i<n;i++)
	    {
	    	if(st.size()==0)
	    	{
	    		st.push_back(a[i]); 
	    	}
	    	else
	    	{
	    		int x = st.back();
	    		if(a[i]<x)
	    		{
	    			int idx = lower_bound(st.begin(), st.end(), a[i]) - st.begin();
	    			//cout<<a[i]<<" "<<idx<<" "<<st[idx]<<"\n";
	    			st[idx] = a[i];
	    		}
	    		else
	    		{
	    			st.push_back(a[i]);
	    		}
	    	}
	    }
	    if(k>st.size())
	    {
	    	cout<<"-1\n";
	    }
	    else
	    {
	    	cout<<st[k-1]<<"\n";
	    }
    }
    
    return 0;
}
