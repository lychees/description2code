#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    long long int n,c,m,x;
    cin>>n>>c>>m;
    x=n/c;
    long long int y=x;
    long long int z=0;
    while(y>=m)
    {
        z=z+y/m;
        y=(y/m)+(y%m);

    }
    x=x+z;
    cout<<x<<endl;
    }
    return 0;

}
