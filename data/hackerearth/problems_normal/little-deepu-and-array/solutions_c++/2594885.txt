#include <iostream>
using namespace std;

int main()
{
   int Tc;
   long long int *Arr;
   cin>>Tc;
   Arr =(long long int *)malloc(Tc*sizeof(long long int ));
   for(int i=0;i<Tc;i++)
   cin>>Arr[i];
   int M;
   cin>>M;
   long long int X;
   while(M--)
   {
   	cin>>X;
   	for(int i=0;i<Tc;i++)
   	{
   		if(Arr[i]>X)
   		{
   			Arr[i]=Arr[i]-1;
   		}
   	}
   }
   for(int i=0;i<Tc;i++)
   cout<<Arr[i]<<" ";
    return 0;
}
