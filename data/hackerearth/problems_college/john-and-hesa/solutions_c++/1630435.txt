#include <iostream>
#include <cmath>
#include <algorithm>
#define in long long
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int x,p;
        cin>>x>>p;
        if(p==1)
            cout<<x<<" 0"<<endl;
        else
        {
            int k=0;
            if(x%p==0)
            {
                while(x%p==0)
                {
                    x/=p;
                    k+=1;
                }
                if(x==1)
                    cout<<k<<" 0"<<endl;
                else
                    cout<<k<<" "<<x<<endl;

            }
            else
                cout<<"0 "<<x<<endl;
        }
    }
    return 0;
}
