#include<bits/stdc++.h>
using namespace std;
int main()
{
ios_base::sync_with_stdio(false);

{
    long long p,q,i,k=0;
    cin>>p>>q;
    for(i=p;i<=q;i++)
    {
        long long tmp=i,sq=tmp*tmp;
        long long z=sq,c=0,num1=0,num2=0;
        int ct=0,t=0;
        while(z>0)
        {
            z/=10;
        c++;
        }
        if(c&1)
        {
            while(ct!=(c+1)/2)
            {
                ct++;
                num1=num1+pow(10,ct-1)*(sq%10);
                sq/=10;
            }
            while(ct!=c)
            {
                ct++;
                num2=num2+pow(10,t)*(sq%10);
                t++;
                sq/=10;
            }
        }
        else
        {
            while(ct!=(c)/2)
            {
                ct++;
                num1=num1+pow(10,ct-1)*(sq%10);
                sq/=10;
            }
            while(ct!=c)
            {
                ct++;
                num2=num2+pow(10,t)*(sq%10);
                t++;
                sq/=10;
            }
        }
     //   cout<<num1<<" "<<num2<<endl;
        if(num1+num2==i)
        {
        k++;
            cout<<i<<" ";
        }
    }
    if(k==0)
        cout<<"INVALID RANGE";
    cout<<endl;
}
return 0;
}
