#include <iostream>
using namespace std;

int main()
{
    string s1,s2,c1,c2;
    int t,i,j,l,k;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>s1;
    	cin>>s2;
    	c1=s1+s2;
    	c2=s2+s1;
    	l=c1.length();
    	for(j=0;j<l/2;j++)
    	{
    		if(c1[j]==c1[l-j-1])
    		continue;
    		else
    		break;
    	}
    	if(j==(l/2))
    	{
    		cout<<"YES"<<endl;
    		continue;
    	}
    	else
    	{
    	for(k=0;k<l/2;k++)
    	{
    		if(c2[k]==c2[l-k-1])
    		continue;
    		else
    		break;
    	}
    	if(k==((l/2)))
    		cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    	
    }}
    return 0;
}
