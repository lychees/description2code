#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{	
	int t;
	cin>>t;
	while(t--)
	{
		string a,b;
		cin>>a>>b;
		int n;
		int temp=0;
		cin>>n;
		
		if(a==b)
		{
			if(2*a.size()==n)
			{
				cout<<"Yes\n";
				continue;
			}
		}
		for(int i=0;i<a.size();i++)
		{
			if(a[i]!=b[i])
				temp++;
			
		}
		if(temp<n)
		{
			if((n-temp)%2==0)
			{
				cout<<"Yes\n";
				continue;
			}
		}
		if(temp==n)
		cout<<"Yes\n";
		else cout <<"No\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
