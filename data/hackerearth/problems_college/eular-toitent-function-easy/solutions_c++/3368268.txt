#include<bits/stdc++.h>
using namespace std;
#define max 10000002
long long int a[max];

void seive(){

	for(int i=2;i*i<max;i++)
		if(a[i]==0){
			//a[i]=i-1;
			for(int j=i*2;j<max;j+=i){
               if(a[j]==0)
                    a[j]=j;

				a[j]*=(i-1);
				a[j]/=i;
			}

		}
}
int main(){
	seive();
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		long long int x=a[n];
		if(x==0)
			x=n-1;
		printf("%lld\n",x);
	}
}
