#include <iostream>
using namespace std;
 
int main()
{
	char str[100];
	scanf("%s",str);
	int c=0;
	int i=0;
	while(str[i]!='\0')
	{
		if(str[i]=='0' || str[i] =='9' || str[i] =='6')
		   c+=6;
		else if(str[i] == '1')
		  c+=2;
		else if(str[i] =='2' || str[i] =='3' || str[i] =='5')
		  c+=5;
		else if(str[i] =='4') 
		  c+=4;
		else if(str[i] == '7')
		 c+=3;
		else
		c+=7;
		
		i++;
	}
    cout << c << endl;
    return 0;
}
