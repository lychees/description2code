#include <iostream>
#include <math.h>
using namespace std;

int N, A, B;

int DoWork()
{
	int Sum=0;
	int Days=0;
	while(true)
	{
		Sum+=A;
		// cout<<"Sum= "<<Sum<<endl;
		if(Sum>=N)
		{
			return Days+1;
		}
		
		Sum-=B;
		
		Days++;
	}
	
	return -1;
}
