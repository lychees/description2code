#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
	
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	if(n%3==0){
    		for(int i=0;i<n;i++)
    			cout<<"5";
    		cout<<endl;
    		continue;
    	}
    	if(n%5==0){
    		for(int i=0;i<n;i++)
    			cout<<"3";
    		cout<<endl;
    		continue;
    	}
    	int x=0,y=0;
    	bool flag=false;
    	while(1){
    		y=n-x;
    		if(y<0){
    			flag=false;
    			break;
    		}
    		if(y%3==0){
    			flag=true;
    			break;
    		}
    		x+=5;
    	}
    	if(!flag)
    		cout<<-1<<endl;
    	else{
    		for(int i=1;i<=y;i++)
    			cout<<5;
    		for(int i=1;i<=x;i++)
    			cout<<3;
    		cout<<endl;
    	}
   
    }
    return 0;
}
