def chkans(rowORcol,tot):
	prev = 0
	for i in rowORcol:
		prev += i
		tot -= i
		if(prev==tot):
			return True
	return False
	
t = int(input())
while(t>0):
	t-=1
	N = int(input())
	col = [0 for i in xrange(N)]
	tot = 0
	row = []
	for i in xrange(N):
		ri = raw_input()
		lc = 0
		for j in xrange(N):
			if(ri[j]=='#'):
				col[j]+=1
				tot+=1
				lc+=1
		row.append(lc)
	if(chkans(row,tot) or chkans(col,tot)):
		print "YES"
	else:
		print "NO"
