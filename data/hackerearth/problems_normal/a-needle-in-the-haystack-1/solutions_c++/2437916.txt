#include <iostream>
using namespace std;

bool isscrambled(int f[], string s, int r, int n) {
	int e = 0;
	for (int i = 0; i < s.size(); i++) {
		f[s[i]-'a']--;
		if (f[s[i]-'a'] == 0) e++;
		
		if (i >= n) {
			if (f[s[i-n]-'a'] == 0) e--;
			f[s[i-n]-'a']++;
		}
		
		//cout << e << endl;
		
		if (r == e) return true;
	}
	return false;
}

int main()
{
    int n;
    cin >> n;
    
    while (n-- > 0) {
    	string a, b;
    	cin >> a >> b;
    	
    	int f[26] = {0};
    	int r = 0;
    	for (int i = 0; i < a.size(); i++) {
    		if (f[a[i] - 'a'] == 0) r++;
    		f[a[i] - 'a']++;
    	}
    	
    	if (isscrambled(f, b, r, a.size())) cout << "YES" << endl;
    	else                             cout << "NO" << endl;
    }
    return 0;
}
