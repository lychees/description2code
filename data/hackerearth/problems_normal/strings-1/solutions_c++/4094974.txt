#include <iostream>
using namespace std;

int main()
{
    int t;
    string m,n;
    cin>>t;
    while(t--)
    {
    	cin>>m>>n;
    	if(m==n)
    		cout<<"YES"<<endl;
    	else if(m.size()!=1 && n.size()!=1)
    		cout<<"NO"<<endl;
    	else if(m>n)
    	{
    		swap(m,n);
    		if(m[0]=='2'&&n[0]=='4')
    			cout<<"YES"<<endl;
    		else
    			cout<<"NO"<<endl;
    	}
    	else if(n>m)
    	{
    		swap(n,m);
    		if(n[0]=='2'&&m[0]=='4')
    			cout<<"YES"<<endl;
    		else
    			cout<<"NO"<<endl;
    	}
    	else
    		cout<<"NO"<<endl;
    }
    return 0;
}
