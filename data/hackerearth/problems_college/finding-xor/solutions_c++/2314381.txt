#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int t,i;
    cin>>t;
    string str1, str2;
    while(t--)
    {
    	vector<int> arr;
    	cin>>str1>>str2;
    	for(i=0; i<str1.length();i++)
    	{
    		int m = str1[i] - '0';
    		int n = str2[i] - '0';
    		arr.push_back(m^n);
        }
    	for(int i=0; i<str1.length();i++)
    	cout<<arr[i];
    	cout<<"\n";
    }
    return 0;
}
