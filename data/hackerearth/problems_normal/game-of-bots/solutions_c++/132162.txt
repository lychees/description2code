#include<cstdio>
#include<cstring>
using namespace std;
const int MAX=1000001;
char s1[MAX],s2[MAX];
char v1[MAX],v2[MAX];
int p1[MAX],p2[MAX];

int main()
{
	int T;
	scanf(" %d",&T);
	while(T--){
	 scanf(" %s %s",s1,s2);
	 int cnt=0,i;
	 int l1=0,l2=0;
	 for(i=0;s1[i];i++) if((s1[i]=='A') || (s1[i]=='B')) { p1[cnt]=i;v1[cnt++]=s1[i]; }
	 l1=i;
	 v1[cnt]=0;
	 int len1=cnt;
	 cnt=0;
	 for(i=0;s2[i];i++) if((s2[i]=='A') || (s2[i]=='B')) { p2[cnt]=i;v2[cnt++]=s2[i]; }
	 l2=i;
	int len2=cnt;
	v2[cnt]=0;
	if((len1!=len2) || (l1!=l2)){
		printf("No\n");
	} else {
		

		if(strcmp(v1,v2)!=0)
		 printf("No\n");
		else {
			bool bad = false;
			int j;
			i=j=0;
			while(i<len1){
				int x=p1[i];
				int y=p2[i];
				if(s1[x]==s2[y]){ 
					if((s1[x]=='B') && (x > y)) { bad=true; break;}
					if((s1[x]=='A') && (x < y)) { bad=true; break;}
					i++;
				}
				else if(s1[x]=='B' && s2[y]=='A') { bad=true; break;}
				else if(s1[x]=='A' && s2[y]=='B') { bad=true; break;}
			}
			
			

			
			if(bad) printf("No\n");
			else printf("Yes\n");
		}
	}
	
	
	}
	return 0;
}
