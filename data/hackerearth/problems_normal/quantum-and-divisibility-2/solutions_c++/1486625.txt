#include <iostream>
#include <algorithm>
#include <set>
#include <string>
#include <vector>
#include <map>
using namespace std;
#define LL long long
map<int, LL> prime(int num)
{
	map<int, LL>res;
	for (int i = 2; i * i <= num; i++) {
		LL count = 0;
		if (num % i == 0) {
			while (num % i == 0) {
				count++;
				num = num / i;
			}
		}
		res[i] = count;
	}
	if (num > 1) {
		res[num] = 1LL;
	}
	return res;
}

int main()
{
	int t;
	cin >> t;
	while (t--) {
		int a, b, c, d;
		cin >> a >> b >> c >> d;
		map<int, LL> first = prime(a);
		map<int, LL> second = prime(c);
		bool flag = false;
		for (map<int, LL> ::iterator it = second.begin(); it != second.end(); ++it) {
			if (it->second * d > first[it->first] * b) {
				flag = true;
			}
		}
		(flag) ? puts("Not divisible") : puts("Divisible");
	}
	return 0;
}
