#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long int test,n,a[105],i,ans;
    cin>>test;
    while(test--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	
    	}
    	sort(a,a+n);
    	
    	ans=2*(a[n-1]-a[0]);
    	cout<<ans<<endl;
    }
    return 0;
}
