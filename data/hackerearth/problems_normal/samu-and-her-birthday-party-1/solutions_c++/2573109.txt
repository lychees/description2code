#include <bits/stdc++.h>
using namespace std;
int main()
{
    long t;
    scanf("%ld",&t);
    while(t--)
    {
    	int n,k;
    	scanf("%d%d",&n,&k);
    	char dishes[k][n];
    	string str;
    	long i,j,p;
    	for(i=0;i<n;i++)
    	{
    		cin>>str;
    		for(j=0;j<k;j++)
    		{
    			if(str[j]=='1')
    			   dishes[j][i]='1';
    			else
    			   dishes[j][i]='0';
    		}
    	}
    	char uni[n];
    	long long num=pow(2,k);
    	long long count,min_dishes=LONG_LONG_MAX;
    	//cout<<(3&(1<<0))<<" "<<(3&(1<<1))<<"\n";
    	for(i=0;i<num;i++)
    	{
    		count=0;
    		for(p=0;p<n;p++)
    		  uni[p]='0';
    	    for(j=0;j<k;j++)
    	    {
    	    	if((i&(1<<j))>=1)
    	    	{
    	    		for(p=0;p<n;p++)
    	    		{
    	    			if(dishes[j][p]=='1')
    	    			   uni[p]='1';
    	    		}
    	    		count++;
    	    	}
    	    }
    	    //cout<<"i="<<i<<"count="<<count<<"minm="<<min_dishes<<"\n";
    	    long tmp=0;
    	    for(p=0;p<n;p++)
    	      if(uni[p]=='1')
    	        tmp++;
    	    if(tmp==n)
    	      if(count<min_dishes)
    	         min_dishes=count;
    	}
    	printf("%lld\n",min_dishes);
    }
    return 0;
}
