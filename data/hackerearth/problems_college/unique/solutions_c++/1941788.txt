#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    
    int t;
    cin>>t;
    while (t--) {
    	long long int n;
    	cin>>n;
    	vector<long long> v;
    	for (long long i=0;i<n;i++) {
    		long long x;
    		cin>>x;
    		v.push_back(x);
    	}
    	long long cnt=1;
    	sort(v.begin(),v.end());
    	for (long long i=1;i<v.size();i++) {
    		if (v[i]!=v[i-1])cnt++;
    	}
    	cout<<cnt<<"\n";
    }
    return 0;
}
