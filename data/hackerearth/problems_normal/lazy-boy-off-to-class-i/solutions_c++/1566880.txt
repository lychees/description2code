#include<iostream>
#include<algorithm>
#include<cmath>
#include<vector>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		vector<int> arr;
		int total_time = 24*60*60*7;
		int classes=0,n,a,b,cnt=0;
		for(int i=0;i<7;i++)
		{
			scanf("%d",&n); classes+=n;
			for(int j = 0;j<n;j++)
			{
				scanf("%d %d",&a,&b);
				arr.push_back(b-a);
			}
		}
		sort(arr.begin(),arr.end());
		int r_a = ceil(((float)(classes*3)/4.0));
		int i=0,temp=0;
		for(vector<int>::iterator ii = arr.begin();i<r_a;i++,ii++)
		{
			temp+=(*ii);
		}
		printf("%d\n",total_time-temp);
	}
}
