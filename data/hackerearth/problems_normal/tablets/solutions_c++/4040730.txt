#include <bits/stdc++.h>
using namespace std;
#define ll long long int

int main()
{
	ll n,i,j,sum;
	cin >> n;
	ll a[n+1],dp[n+1];
	for(i=0;i<n;i++){
		cin >> a[i];
	}
	dp[0]=1;
	for(i=1;i<n;i++){
		if(a[i]>a[i-1]){
			dp[i]=dp[i-1]+1;
		}
		else{
			dp[i]=1;
		}
	}
	for(i=n-2;i>=0;i--){
		if(a[i]>a[i+1] && dp[i]<=dp[i+1]){
			dp[i]=dp[i+1]+1;
		}
	}
	sum=0;
	for(i=0;i<n;i++){
		sum+=dp[i];
	}
	cout << sum << endl;
    return 0;
}
