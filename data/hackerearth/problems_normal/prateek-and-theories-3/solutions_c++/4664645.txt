#include <iostream>
#include <map>
using namespace std;
typedef long long int ll;
int main()
{
    int t;
    scanf("%d", &t);
    while(t--) {
    	ll n;
    	scanf("%lld", &n);
    	map<ll, ll> m;
    	map<ll, ll> :: iterator it;
    	ll i;
    	ll t1, t2;
    	for(i = 0; i < n; i++) {
    		scanf("%lld %lld", &t1, &t2);
  			m[t1]++;
  			m[t2]--;
    	}
    	ll max = 0;
    	ll val = 0;
    	for(it = m.begin(); it != m.end(); it++) {
    		val = val + it->second;
    		if(val > max) {
    			max = val;
    		}
    	}
    	printf("%lld\n", max);
    }
    return 0;
}
