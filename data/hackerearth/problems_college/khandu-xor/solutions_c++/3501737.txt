#include<bits/stdc++.h>
using namespace std;

int p(string s1)
{
	int i,l;
	l=s1.size();
	for(int i=0;i<l/2;i++){
			if(s1[i]!=s1[l-i-1])
			return 0;
		}
	return 1;
}
int main()
{
	int t;
   cin >> t;
   while(t--){
	string s1,s2,s3,s4;
	cin>> s1; 
	cin >> s2;
	s3= s2 + s1;
	s4= s1 + s2;
	if(p(s4) || p(s3))
	cout<< "YES\n";
	else
	cout<< "NO\n";
	}
 	return 0;
}
