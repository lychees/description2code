#include<iostream>
#include<stdio.h>
#include<string.h>
#define MAX 201
char a[MAX];
using namespace std;
int main()
{
    int t,l,sum;
    scanf("%d",&t);
    while(t--)
    {    
         sum = 0;
         scanf("%s",a);
         l = strlen(a);
         for(int i=0;i<l;i++)
         {
              if(a[i] == '1')
              sum += 1;
              else if(a[i] == '2')
              sum += 2;
              else if(a[i] == '3')
              sum += 3;
              else if(a[i] == '4')
              sum += 4;
              else if(a[i] == '5')
              sum += 5;
              else if(a[i] == '6')
              sum += 6;
              else if(a[i] == '7')
              sum += 7;
              else if(a[i] == '8')
              sum += 8;
              else if(a[i] == '9')
              sum += 9;
              else if(a[i] == '0')
              sum += 0;
         }
         if(sum%9 == 0)
         printf("0 or 9\n");
         else
         printf("%d\n",9- (sum%9));
    }
    
    
    return 0;
}
