#include<bits/stdc++.h>
using namespace std;
int checksmall(char c)
{
    return ('a'<=c && c<='z');
}
char toupper(char c)
{
    return (c-'a'+'A');
}
int main()
{
    ios::sync_with_stdio(0);
    //("input.txt","r",stdin);
    string s;
    getline(cin,s);
    if (checksmall(s[0]))
        s[0]=toupper(s[0]);
    for (int i=1;i<s.size();i++)
    {
        if (s[i-1]==' ' && checksmall(s[i]))
            s[i]=toupper(s[i]);
    }
    cout<<s<<endl;
    return 0;
}
