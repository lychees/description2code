ar = [0 for x in range(row_count)]

while i < row_count:
    row = [int(x) for x in raw_input().split()]
    ar[i] = row
    i += 1

    while True:
        i = 0
        selected_players = []
        if len(players) == 1:
            return players[0]
        for j in range(len(players) / 2):
            if ar[players[i + 1]][players[i]] == 1:
                selected_players.append(players[i + 1])
            else:
                selected_players.append(players[i])
            i += 2
        players = selected_players
