#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
bool prime[1001];
int main()
{
 ll t;
 cin>>t;
 prime[0]=false;
 prime[1]=false;
 for(ll i=2;i<=1000;i++)
 {
 	int f=1;
 	for(ll j=2;j<=sqrt(i);j++)
 	{
 		if(i%j==0)
 		{
 		  f=0;
 		  break;}
 	}
 	if(f)
 	prime[i]=true;
 }
 while(t--)
 {
 	ll n;
 	cin>>n;
 	ll c=0;
 	for(ll i=0;i<n;i++)
 	{
 		ll p;
 		cin>>p;
 		if(prime[p])
 		c++;
 	}
 	cout<<c<<endl;
 }
	return 0;
}
