#include<bits/stdc++.h>

using namespace std;

int main()
{
 int t;
 cin>>t;
 while(t--)
 {
   string s;
   cin>>s;
   int i,k,l=s.length(),vc=0,mf=0,lg=0,cc=0;

   for(i=0;i<l;i++)
    if(s[i]=='a' || s[i]=='e' || s[i]=='i' || s[i]=='o' || s[i]=='u' || s[i]=='y')
     vc++;
    else
     cc++;

   if(vc<cc)
   {
     cout<<"hard"<<endl;
     continue;
   }



   lg=0;
   for(i=0;i<l;i++)
   {
    if(s[i]!='a' && s[i]!='e' && s[i]!='i' && s[i]!='o' && s[i]!='u' && s[i]!='y')
     lg=lg+1;
    else
     lg=0;

     if(lg==3)
      {mf=1; break;}
   }
   // cout<<"mf "<<mf<<endl;
   //cout<<vc<<" vc "<<l/2<<" hf "<<endl;
   if(mf==1)
    cout<<"hard"<<endl;
   else
    cout<<"easy"<<endl;
 }
return 0;
}
