#include <iostream>
#include <iomanip>

using namespace std;

int main(){
	int n;
	double p;
	cin >> n >> p;

	double ans1,ans2,ans3,t;

	p/=100;
	ans1 = 0;
	ans2 = p;
	ans3 = 1-p;

	for(int i = 4; i<=n;i++){
		t = ans1*(1-p) + ans2*p;
		ans1 = ans2;
		ans2 = ans3;
		ans3 = t;
	}

	cout << fixed << setprecision(6) << ans3;

}
