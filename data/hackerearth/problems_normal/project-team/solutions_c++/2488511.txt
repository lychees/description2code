#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
   int tests;
    cin >> tests;
    
    while (tests-- > 0) {
    	int n;
    	cin >> n;
    	
    	int level[n];
    	for (int i = 0; i < n; i++) cin >> level[i];
    	
    	sort(level, level+n);
    	
    	int max = 0, min = 20000;
    	
    	for (int i = 0; i < n/2; i++) {
    		int sum = level[i] + level[n-1-i];
    		if (sum < min) min = sum;
    		if (sum > max) max = sum;
    	}
    	
    	cout << (max - min) << endl;
    }
    return 0;
}
