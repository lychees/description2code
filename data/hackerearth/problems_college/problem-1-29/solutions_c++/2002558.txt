bool flag=0;
bool dfs(vector < vector <int> > v,int n,int x,int y)
{

    if(x==n-1 && y==n-1) flag=1;





        if(y+1<n  && v[x][y+1]==1 )
        {
			dfs(v,n,x,y+1);
        }
        
        if(x+1<n && v[x+1][y]==1)
        {
			dfs(v,n,x+1,y);
        
        }
		
    

return flag;
}




int main()
{

int tt;
cin>>tt;
while(tt--)
{
	flag=0;
    int n;
    cin>>n;
    vector <vector <int> > v(n);

    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            int a;
            cin>>a;
            v[i].push_back(a);

        }

    }

    if(dfs(v,n,0,0))
    {
        cout<<"POSSIBLE"<<endl;
    }
    else cout<<"NOT POSSIBLE"<<endl;


}
return false;
}
