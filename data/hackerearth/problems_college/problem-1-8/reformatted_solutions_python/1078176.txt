class ourclass():
    def icomplete():
        required = raw_input()
        number = int(raw_input())
        dict = []
        for i in range(number):
            dict.append(raw_input())
        ans = []

        for j in range(len(dict)):
            if required == dict[j][:len(required)]:
                ans.append(dict[j])

        if len(ans) > 0:
            print(min(ans))
            return min(ans)
        print(required)
        return required
    icomplete()
