#include <bits/stdc++.h>

using namespace std;
int main()
{
    long long int a,b,d1,d2,ma,mi,carry,c,t;
    cin>>t;
    while (t--)
    {
        c=0;
        carry=0;
        cin>>a>>b;
        ma=max(a,b);
        mi=min(a,b);
        while (ma!=0)
        {
            d1=ma%10;
            d2=mi%10;
            if (d1+d2+carry>9)
            {
                c+=1;
                carry=1;
            }
            else
            {
                carry=0;
            }
            ma/=10;
            mi/=10;
        }
        if (c==0)
            cout<<"No carry operation"<<endl;
        else if (c==1)
            cout<<"1 carry operation"<<endl;
        else
            cout<<c<<" carry operations"<<endl;
    }
    return 0;
}
