#include <iostream>
using namespace std;

int main()
{ 
 int n;
 cin>>n;
 while(n--)
 {
 	string s;
 	cin>>s;
 	int l=s.size();
 	int c=0;
 	int ss=0;
 	bool g,h=false;
 	for(int i=0;;)
 	{
 		while(s[i]!='_'&& i<l)
 		{i++;h=true;}
 		if(i>=l)break;
 		g=false;
 		int k=0;
 		while(s[i]=='_'&& i<l)
 		{i++;g=true;k++;
 		}
 		if(g){c++;ss=ss^k;}
 		if(i>=l)break; 		
 	}
 	if(ss==0||h==0)
 	 cout<<"Bob"<<endl;
 	else
 	 cout<<"Alice"<<endl;
 	
 }
 
 
 return 0;
}
