# include <stdio.h>
 
void swap (char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}
  

void permute(char *a, int i, int n) 
{
   int j; 
   if (i == n)
     printf("%s\n", a);
   else
   {
        for (j = i; j <= n; j++)
       {
          swap((a+i), (a+j));
          permute(a, i+1, n);
          swap((a+i), (a+j)); 
       }
   }
} 
 
int main()
{
   char s[100000];
   int t,L,R;
   scanf("%s",s);
   scanf("%d",&t);
   while (t --) {
   	   scanf("%d %d",&L,&R);
   	   permute(s, L, R);
   	   printf("\n");
   }
   
   return 0;
}
