#include<bits/stdc++.h>
using namespace std;

int main()
{
    int ceo,cto,coo;
    cin>>ceo>>cto>>coo;
    int n;
    cin>>n;
    int a[n];
    int i;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    int total = 3+n;
    int temp[total];
    for(i=0;i<total;i++)
    {
    	temp[i] = 0;
    }
    sort(a,a+n);
    temp[total/2] = ceo;
    if(cto > coo)
    {
    	temp[0] = coo;
    	temp[total-1] = cto;
    }
    else
    {
    	temp[0] = cto;
    	temp[total-1] = coo;
    }
    int j = 0;
    for(i=0;i<total;i++)
	{
		if(temp[i] == 0)
		{
			temp[i] = a[j];
			j++;
		}
	}
	int diff = 0;
	for(i=1;i<total;i++)
	{
		diff +=abs(temp[i]-temp[i-1]);
	}
	cout<<diff<<endl;
}
