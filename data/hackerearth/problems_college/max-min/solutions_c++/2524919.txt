# include<stdio.h>
# include<algorithm>
using namespace std;
int main()
{
	int n,k,m=0;
	scanf("%d%d",&n,&k);
	int diff[100000],a[n],j;
	j=k-1;
	for(int i=0;i<n;i++)
	scanf("%d",&a[i]);
	sort(a,a+n); int p=0;
	while(1)
	{
		diff[m]=a[j]-a[p];  
		m++; j++; p++;
		if(j==n-1) break;
	}
	sort(diff,diff+m);
	if(k==1)
	printf("%d\n",a[0]);
	else
	printf("%d\n",diff[0]);
	return 0;
	
}
