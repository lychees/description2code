#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    while(n--)
    {
    	string s;
    	cin>>s;
    	int odd=0,even=0;
    	int a[26]={0};
    	int len=s.length();
    	for(int i=0;i<len;i++)
    	{
    		a[s[i]-97]++; 
    	}
    	for(int i=0;i<26;i++)
    	{
    	//	cout<<a[i]<<" ";
    		if(a[i]>0)
    		{
    		if(a[i]%2==0)
    		even++;
    		else
    		odd++;
    		}
    	}
    //	cout<<"even "<<even;
    //	cout<<"odd "<<odd<<"\n";
    	if(odd==1)
    	{
    	cout<<"0\n";
    	}
    	else if(odd>1&&even==0)
    	{
    		odd--;
    		cout<<odd<<"\n";
    	}
    	else if(odd==0&&even>0)
    	{
    		cout<<"0\n";
    	}
    	else if(odd>0&&even>0)
    	{
    		odd--;
    		cout<<odd<<"\n";
    	}
    	
    }
    return 0;
}
