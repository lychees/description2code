int main()
{
	int t;

	while(t--)
	{
		int n, len;
		cin >> n;
		fill_n(fact,sizeof(fact)/sizeof(int),0);
		len = 1;
		fact[0] = 1;
		for(int i = 1; i <= n; i++)
		{
			for(int j = 0; j < len; j++)
				fact[j] = fact[j]*i;
			for(int j = 0; j < MX; j++)
			{
				while(fact[j] > 9)
				{
					fact[j+1]+=fact[j]/10;
					fact[j]%=10;
				}
				if (fact[j] > 0)
					len = max(len, j+1);			
			}
		}

		for(int i = len-1; i >= 0; i--)
			printf("%d",fact[i]);
		printf("\n");
	}
	return 0;
}
