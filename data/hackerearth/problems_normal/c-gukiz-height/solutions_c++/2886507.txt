#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
ll l[100000] , pl[100000];
int main()
{
    ll g , n  ;
    cin>>n>>g;
    for(int i=1;i<=n;i++){
        cin>>l[i];
    }
    for(int i=1;i<=n;i++){
        pl[i]=pl[i-1]+l[i];
       // if(pl[i]>=g-(i*(i+1)/2)) {cout<<i<<endl; return 0;}
    }
    ll cycle=pl[n];
    int j=-1;
    bool ok =false;
    while(((j+1)*n*(j+1)*n+1)/2<=1000000000){
        j++;
        for(int i=1;i<=n;i++){
            if(cycle*j+pl[i]>=g-((j*n+i)*(j*n+i+1)/2)) {cout<<j*n+i<<endl; return 0;}
        }
    }
   ll lo=0 , hi=2000000000;
    while(lo<hi){
        ll med=(lo+hi)/2;
        if(g-(med*(med+1))/2 <= ((med-1)/n)*cycle+pl[(med-1)%n+1] ){
            hi=med;
        }
        else lo=med+1;
    }
    cout<<lo<<endl;
    return 0;
}
