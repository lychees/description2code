#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    
    while(t!=0)
    {
    	double a , b , c;
    	cin>>a>>b>>c;
    	
    	double s = (a+b+c)/2;
    	
    	double area = sqrt(s*(s-a)*(s-b)*(s-c));
    	
    	double radius  = (a*b*c)/(4*area);
    	
    	double ans = 3.1415*radius*radius;
    	
    	printf("%.4lf\n",ans);
    	
    	t--;
    }
}
