#include <iostream>
using namespace std;
#include <algorithm>

int main()
{
	int n;
	cin >> n;
	int ar[n];
	for(int i = 0; i < n; i++){
		cin >> ar[i];
	}
	
	sort(ar,ar+n);
	long int sum = 0;
	for(int i = 0; i < n/2; i++){
		sum = sum + 2*ar[n-i-1] -2*ar[i];
	}
	cout << sum << endl;
	
    
    return 0;
}
