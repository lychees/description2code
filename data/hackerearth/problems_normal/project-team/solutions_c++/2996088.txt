#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        int n;
        cin >> n;
        int a[n];
        for(int i=0;i<n;i++)
            cin >> a[i];
        sort(a,a+n);
        int mx = a[0] + a[n-1];
        int mn = a[1] + a[n-2];
        for(int i=1;i<n/2;i++)
        {
            if((a[i] + a[n-1-i]) > mx)
                mx = a[i] + a[n-1-i];
            else if((a[i] + a[n-1-i]) < mn)
                mn = a[i] + a[n-1-i];
        }
        cout << mx - mn << endl;
    }
}
