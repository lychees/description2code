#include<iostream>
#include<stdio.h>

using namespace std;
int main()
{
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		string arr;
		long long int s;
		cin>>arr;
		s = arr.size();
		string s2 = "cde";
		string s7 = "acf";
		string s6 = "b3";
		string s9 = "c6a";
		for(long long int j=0;j<s;j++)
		{
			if(arr[j]=='2')
				cout<<s2;
			else if(arr[j]=='7')
				cout<<s7;
			else if(arr[j]=='6')
				cout<<s6;
			else if(arr[j]=='9')
				cout<<s9;
		}
		cout<<"\n";
	}
	return 0;
}
