#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

bool even(ll N,ll M){
    bool t1,t3;
    if(N<M){
    	return false;
    }
    while(M){
        t1=(N%2==1);
        N=N>>1;
        t3=(M%2==1);
        M=M>>1;
        if(!t1 && t3){//0c1
            return true;
        }
    }
    return false;
}

int main(){
    ll M,N;
    int T;
    cin >> T;
    while(T--){
        cin >> N >> M;
        puts(even(N,M)?"even":"odd");
    }
    return 0;
}
