#include <iostream>
#define N 112345
using namespace std;

int arr[N];
int n;
int dpmax[N][20];
int log_2[N];
void process()
{
    for(int i=0;i<n;++i)
    {
        dpmax[i][0]=i;
    }
    for(int j=1;(1<<j)<=n;++j)
    {
        for(int i=0;i+(1<<j)-1<n;++i)
        {
        	if(arr[dpmax[i][j-1]]>arr[dpmax[i+(1<<(j-1))][j-1]])
        		dpmax[i][j]=dpmax[i][j-1];
        	else dpmax[i][j]=dpmax[i+(1<<(j-1))][j-1];
        }
    }
}
int rmax(int i,int j)
{
	int l=log_2[j-i+1];
    if(arr[dpmax[i][l]]>arr[dpmax[j-(1<<l)+1][l]])
    	return dpmax[i][l];
    else return dpmax[j-(1<<l)+1][l];
}
long long ans=0;
int func(int i,int j)
{
	if(i>j) return -1;
	if(i==j) return i;
	//printf("%d %d\n",i,j);
	int m=rmax(i,j);
	int l=func(i,m-1);
	int r=func(m+1,j);
	//printf("%d %d %d\n",m,l,r);
	if(l!=-1) ans+=arr[m];
	if(r!=-1) ans+=arr[m];
	return m;
}
int main()
{
	log_2[1]=0;
	for(int i=2;i<=100000;++i)
	{
		log_2[i]=log_2[i-1];
		if((i&(i-1))==0) log_2[i]++;
	}
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	for(int i=0;i<n;++i) scanf("%d",&arr[i]);
    //	memset(dp,)
    	process();
    	func(0,n-1);
    	printf("%lld\n",ans);
    	ans=0;
    }
    return 0;
}
