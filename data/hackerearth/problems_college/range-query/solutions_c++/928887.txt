#include <iostream>
using namespace std;

int main()
{
    int a[101][10000],val,x,y,i,count,n,num,posnum[101];
    for(i=0;i<101;i++)
    posnum[i]=-1;
    long q;
    cin>>n;
    for(i=0;i<n;i++){
    cin>>num;
    posnum[num]++;
    a[num][posnum[num]]=i;
    }
    cin>>q;
    while(q-->0)
    {   count=0;
    	cin>>x>>y>>val;
    	i=0;
    	while((a[val][i]<x)&&(i<=posnum[val]))i++;
    	while((a[val][i]<=y)&&(i<=posnum[val]))
    	{
    		count++;
    		i++;
    	}
    		
    	cout<<count<<endl;	
    }
    return 0;
}
