#include <bits/stdc++.h>
using namespace std;
int a[1004],n,sm[100004];
int solve(int s){
	int i=0,j=0,k,ts=0,cnt=0;
	while(i<n && j<n){
		ts+=a[j];
		if(ts==s){
			cnt++;
			i=j+1;
			ts=0;
		}
		else if(ts>s){
			while(ts>s && i<n){
				ts-=a[i];i++;
			}
			if(ts==s){
				cnt++;i=j+1;ts=0;
			}
		}
		j++;
	}
	return cnt;
}
int main()
{
    int i,j,k,s=0,mx=0,c=0;
    scanf("%d",&n);
    for(i=0;i<n;i++){
    	scanf("%d",&a[i]);
    	s+=a[i];
    }
    for(i=1;i<=s;i++){
    	sm[i]=solve(i);
    	//cout<<i<<" "<<sm[i]<<endl;
    	mx=max(mx,sm[i]);
    }
    for(i=1;i<=s;i++)
     if(sm[i]==mx)
     c++;
     cout<<mx<<endl;
    for(i=1;i<=s;i++)
     if(sm[i]==mx)
      printf("%d ",i);
    return 0;
}
