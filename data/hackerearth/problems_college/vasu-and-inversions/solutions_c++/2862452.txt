#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n,k,i;
	int t;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&k);
		int a[n];
		int z=0,o=0;
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			if(a[i]==0)
			z++;
			else
			o++;
		}
		if(z>=k)
		{
			o+=k;
		}
		else
		{
			o+=z;
			int l=k-z;
			if(l%2==1)
			o--;
		}
		printf("%d\n",o);
	}
}
