#include <iostream>
#include<cstring>
using namespace std;

void reverse(char b[],int k)
{
	int i=0,j=k;
	while(i<j)
	{
		char t=b[i];
		b[i]=b[j];
		b[j]=t;
		i++;j--;
	}
}

{

	char b[2*len];
	if(len<=3)
	{
		cout<<a<<endl;
	}
	else
	{
		for(i=j;i>=0;i--)
		{
			b[k++]=a[i];
			count++;

			}

			//cout<<b[k-1];

		for(i=k-1;i>=0;i--)

}
