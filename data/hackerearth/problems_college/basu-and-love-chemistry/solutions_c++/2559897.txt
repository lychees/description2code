#include <bits/stdc++.h>
#define MOD 1000000007

using namespace std;

int main(void) {
    string s;
    cin >> s;

    long long int ans = 0;
    long long int ans2 = 0;

    int n = (int)s.length();
    for (int i = 0; i < n; ++i) {
        ans = (ans * 10) + ((i + 1) * (s[i] - '0'));
        ans += ans2;
        ans2 += (i + 1) * (s[i] - '0');
        ans2 %= MOD;
        ans %= MOD;
    }

    cout << ans << '\n';
}
