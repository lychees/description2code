#include <iostream>
using namespace std;
int maxSubArraySum(int a[], int size)
{
   int max_so_far = 0, max_ending_here = 0;
   int i;
   for(i = 0; i < size; i++)
   {
     max_ending_here = max_ending_here + a[i];
     if(max_ending_here < 0)
        max_ending_here = 0;
     if(max_so_far < max_ending_here)
        max_so_far = max_ending_here;
    }
    return max_so_far;
} 

int main()
{
    
   int z,b;
   scanf("%d",&z);
   while(z-->0){
   scanf("%d",&b); int a[b];for(int r=0;r<b;r++){scanf("%d",&a[r]);}
   int max_sum = maxSubArraySum(a, b);
   printf("%d\n", max_sum);
   getchar();}
   return 0;
}
