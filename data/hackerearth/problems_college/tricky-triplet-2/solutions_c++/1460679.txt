#include <iostream>
using namespace std;

int main()
{
    long int n;
    cin>>n;
    int a[n];
    int s=0;
    for(int i=0;i<n;i++)
    cin>>a[i];
    for(int i=0;i<n;i++)
    {
    	for(int j=i+1;j<n;j++)
    	{
    		for(int k=j+1;k<n;k++)
    		{
    			if((a[k]-a[j])==(a[j]-a[i]))
    			s++;
    		}
    	}
    }
    cout<<s;
    return 0;
}
