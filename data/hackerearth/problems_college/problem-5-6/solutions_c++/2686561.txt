#include <bits/stdc++.h>

using namespace std;



int arr [1000 + 1];
int dp [1000 + 1][1000 + 1];

int n , m , k , d , t , tem1 , tem2 , tem3 , tem4 , sum = 0 , ans = 0, x;
string s , c;

int rec (int index , int len){

    if(index == n + 1 && len == 0) return 0;
    if(index == n + 1) return -1e9;

    int &st = dp[index][len];
    if(st != -1) return st;

    int e = 0;

    e = rec(index + 1 , len);
    int sta  = arr[index] - x;
    int r = sta ;

    int ba2i = len;

    for(int i = 1; i <= len; i++){
        if(ba2i - i * index < 0) break;
        if(i == 1) e = max(e , r + rec(index + 1 , len - i * index) );
        if(i > 1){
            r += sta;
            e = max(e , r + rec(index + 1 , len - i * index) );
        }
    }
    return st = e;
}

int main()
{





    scanf("%d", &t);

    while( t-- ){

        scanf("%d%d", &n , &x);
        for(int i = 0; i <= n; i++)
            for(int j = 0; j <= n; j++)
                dp[i][j] = -1;

        for(int i = 1; i <= n; i++) scanf("%d", &arr[i]);
        printf("%d\n", rec(1 , n) );
    }

    return 0;
}
