#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,x,y;
    	cin>>n>>x>>y;
    	long long int rem=n-x;
    	long long int worse=y+1;
    	if(rem>=worse)
    	{
    		cout<<worse<<endl;
    	}
    	else
    	{
    		cout<<rem<<endl;
    	}
    }
    return 0;
}
