#include <iostream>
using namespace std;

int main()
{
  
    long int t;
    cin>>t;
    long long int a,b,c;
    while(t--){
    			cin>>a>>b>>c;
    			if(a+b>c && a+c>b && b+c>a)
    			{
    				if(a==b && b==c)
    					cout<<"YES. It is a equilateral triangle."<<endl;
	    			else if((a==b ||b==c || a==c))
	    				cout<<"YES. It is a isosceles triangle."<<endl;
	    			else 
	    			cout<<"YES. It is a scalene triangle."<<endl;
    			}
    	
    			else {
    			cout<<"NO"<<endl;
    			}
    }
    return 0;
}
