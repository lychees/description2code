#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long int i,s=0,n,j=0,b[1000000];
    cin>>n;
    for(i=0;i<n;i++)
    cin>>b[i];
    sort(b,b+n);
    for(i=0;i<n;)
    {
    	++j;
    	if(b[i]<=j*j)
         {s+=j*j-b[i];i++;}
    	else
    	s+=j*j;
    }
    cout<<s;
    return 0;
}
