#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define F(i,a,b) for(ll i = (ll)(a); i <= (ll)(b); i++)
#define RF(i,a,b) for(ll i = (ll)(a); i >= (ll)(b); i--)
int main()
{
    #ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
        freopen("output.txt","w",stdout);
    #endif
    ll N;
    ll ans;
    string S;
        cin>>S;
        N = S.length();
        vector<ll> Prefix(N),Suffix(N);
        F(i,0,N-1) Prefix[i] = Suffix[i] = 0LL;
        Prefix[0] = (S.at(0) == '1' ? 1LL : 0LL);
        F(i,1,N-1)
        {
            if(S.at(i) == '1') Prefix[i] = Prefix[i-1] + 1LL;
            else Prefix[i] = Prefix[i-1];
        }
        Suffix[N-1] = (S.at(N-1) == '1' ? 1LL : 0LL);
        RF(i,N-2,0)
        {
            if(S.at(i) == '1') Suffix[i] = Suffix[i+1] + 1LL;
            else Suffix[i] = Suffix[i+1];
        }
        ans = 0LL;
        F(i,1,N-2)
        {
            if(S.at(i) == '0' && i-1>=0 && i+1 <= N-1)
            {
                ans += (ll)(Prefix[i-1]*Suffix[i+1]);
            }
        }
        printf("%lld",ans);
    return 0;
}
