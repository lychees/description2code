#include <bits/stdc++.h>
using namespace std;
vector<int> v;
void pre(){
	int i=1;
	while(i<=(1<<17)){
		int flag=0;
		for(int j=0;j<20;j++){
			if(i&(1<<j) && i&(1<<(j+1)))flag=1;
		}
		if(!flag)v.push_back(i);
		i++;
	}
}
void solve(){
	int n;
	cin>>n;
	cout<<*lower_bound(v.begin(),v.end(),n)<<"\n";
}
int main(){
	pre();
	int t;
	cin>>t;
	while(t--)solve();
}
