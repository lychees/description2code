#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int ar[100005];

long int DP[100005],size;

long int ret( int n){
  if(n >= size ) return 0;
  if(DP[n] != -1) return DP[n];
  if(n == size - 1) {
    DP[n] = ar[n];
    return DP[n];
  }

  DP[n] = max(ret(n+1) , ar[n] + ret(n+2));
  return DP[n];

}







int main()
{
  long int t;
  cin >> t;
  while(t--){
    
    cin >> size;
    
    for(int i = 0; i < size; i++){
      cin >> ar[i];
    }
    for(int i = 0; i < 100005; i++){
      DP[i] = -1;
    }
    long int val = 0;
    
    val = ret(0);
    cout << val << endl;
  }

    
    
    return 0;
}
