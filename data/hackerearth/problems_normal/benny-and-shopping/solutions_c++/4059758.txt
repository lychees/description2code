#include <iostream>
#include <cstring>
using namespace std;

int main()
{
   int n,p,l;
   string s,o="";
   cin>>n;
   while(n--)
   {
   	o = "";
   	cin.ignore();
   	getline(cin,s);
   	l = s.length();
   	p = s.find('$');
   	for(int i=p+1;i<l;++i)
   	{
   		if(((s[i]>='0')&&(s[i]<='9'))||(isspace(s[i])))
   		{
   		if(isspace(s[i]))
   		continue;
   		if((o=="")&&(s[i]=='0'))
   		{
   			continue;
   		}
   		else
         o=o+s[i];
   	
   		
   		}
   		else
   		{
   			break;
   		}
   	}
   	if(o=="")
   	cout<<"$0"<<endl;
   	else
   	cout<<"$"<<o<<endl;
   }
    return 0;
}
