#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
    int t, n;
    cin >> t;

    string s;
    while(t--) {
        cin >> s;
        int xr = 0;
        int counter = 0;
        for (int i = 0; i < s.size(); i++) {
            if (s[i] == '_') counter++;
        }
        if (counter == s.size()) {
            cout << "Bob" << endl;
            } else {
                for (int i = 0; i < s.size();) {
                    if (s[i] == '_') {
                        int counter = 0;
                        while(i < s.size() && s[i] == '_') {
                            counter++;
                            i++;
                        }
                        xr = xr ^ counter;
                    } else {
                        i++;
                    }
                }

                if (xr) cout << "Alice" << endl;
                else cout << "Bob" << endl;
                }
    }
}
