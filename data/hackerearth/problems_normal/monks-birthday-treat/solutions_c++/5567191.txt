 #include <iostream>
 #include <vector>
using namespace std;

vector <int> adj[1000];
bool visited[1000];
int num=0;

void dfs(int s) {
    visited[s] = true;
    for(int i = 0;i < adj[s].size();++i)    {
     if(visited[adj[s][i]] == false)
     {
     	num++;
         dfs(adj[s][i]);
     }
    }
}

void initialize(int n) {
    for(int i = 0;i<=n;++i)
     visited[i] = false;
}

int main() {
    int n,d,x,y,mn=1000,t;
    cin >> n;                      
    cin >> d;                      
    for(int i=0;i<d;++i) 
    {
     cin >> x >> y;
     adj[x].push_back(y);
    }


    for(int i = 1;i <= n;++i) {
    	initialize(n);
     if(visited[i] == false)     
     {
         dfs(i);
         num++;
         if(num<mn)
     		mn=num;
     
     }
     num=0;
    }
    
    cout<<mn<<endl;
    return 0;
}
