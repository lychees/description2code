#include<bits/stdc++.h>
using namespace std;

#include<stdio.h>

 long long C( long long n)
{
   return (n*(n-1)*(n-2))/6;
}
int main(){
	vector<set< long long int> >store(15);
	vector<set< long long int> > exclude(15);
	 long long int n;
	 scanf("%lld",&n);
	
	vector< long long> arr;
	for(int i=0;i<n;i++){
		 long long x;
		 scanf("%lld",&x);
		arr.push_back(x);
	}
	
	for(int i=0;i<n;i++){
		 long long int x=arr[i];
		 int flag2=0;
		 int flag3=0;
		 int flag5=0;
		 int flag7=0;
		while(x>0){
			int rem=x%10;
			if(rem==2)
			{
				store[0].insert(i);
				flag2=1;
				//cout<<"IN TWO";
			}
			else if(rem==3){
				store[1].insert(i);
				flag3=1;
				
			}
			else if(rem==5){
				store[2].insert(i);
				flag5=1;
			}
			else if(rem==7){
				store[3].insert(i);
				flag7=1;
			}
			
			x=x/10;
			if(flag2==1 && flag3==1 && flag5==1 && flag7==1)
			break;
		}
		
		if(flag2 && flag3){
			exclude[0].insert(i);
		}
		if(flag2 && flag5){
			exclude[1].insert(i);
		}
		if(flag2 && flag7){
			exclude[2].insert(i);
		}
		if(flag3 && flag5){
			exclude[3].insert(i);
		}
		if(flag3 && flag7){
			exclude[4].insert(i);
		}
		if(flag5 && flag7){
			exclude[5].insert(i);
		}
		
		if(flag2 && flag3 && flag5){
			store[4].insert(i);
		}
		if(flag2 && flag3 && flag7){
			store[5].insert(i);
		}
		if(flag2 && flag5 && flag7){
			store[6].insert(i);
		}
		if(flag3 && flag5 && flag7){
			store[7].insert(i);
		}
		if(flag2 && flag3 && flag5 && flag7){
			exclude[6].insert(i);
		}
	}
		long long ans=0;
		for(int i=0;i<8;i++){
			//if(store[i].size()>=3)
			ans+=C(store[i].size());
		}
		for(int i=0;i<7;i++){
			//if(exclude[i].size()>=3)
			ans-=C(exclude[i].size());
		}
 		printf("%lld\n",ans);


 



	
	
}
