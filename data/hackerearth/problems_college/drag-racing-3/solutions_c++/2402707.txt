#include <iostream>
#include <cmath>
#include <map>
#include <algorithm>
using namespace std;
int main() {
	int t,i;
	cin>>t;
	int n = pow(2,t) - 1;
	string a[n],z;
	for(i=0;i<2*n;i++)
	{
		cin>>z;
		if(i%2==0)
			a[i/2] = z;
	}
	sort(a,a+n);
	int j = 0;
	map <string,int> b;
	for(i=0;i<n;i++)
	{
		b[a[i]]++;
	}
	string x;
	int max = 0;
	for(auto i:b)
	{
		if(i.second>max)
		{
			max = i.second;
			x = i.first;
		}
	}
	cout<<x;
	return 0;
}
