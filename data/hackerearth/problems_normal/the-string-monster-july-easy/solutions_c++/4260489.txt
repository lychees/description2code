#include<iostream>
#include<algorithm>
#include<string>

using namespace std;

int const N = 41;

int n, test, cnt[N * 5], rem, m, sz[N];

string s[N], l;

bool dfs(int p){
	if(p == n){
		return false;
	}else{
		bool can = true;
		for(int i=0;i<sz[p];++i){
			--cnt[s[p][i]];
			if(cnt[s[p][i]] < 0){
				can = false;
			}
		}
		if(can){
			if(rem == sz[p]){
				return true;
			} 
			rem -= sz[p];
			if(dfs(p+1)){
				return true;
			}
			rem += sz[p];
		}
		for(int i=0;i<sz[p];++i){
			++cnt[s[p][i]];
		}
		return dfs(p+1);
	}
}

int main(){
#ifdef _DEBUG
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
#endif
	cin >> test;
	while(test--){
		cin >> n;
		for(int i=0;i<n;++i){
			cin >> s[i];
			sz[i] = s[i].size();
		}
		cin >> l;
		m = l.size();
		rem = m;
		for(int i=0;i<N*5;++i){
			cnt[i] = 0;
		}
		for(int i=0;i<l.size();++i){
			++cnt[l[i]];
		}
		if(dfs(0)){
			cout << "YES" << endl;
		}else{
			cout << "NO" << endl;
		}
	}





	return 0;
}
