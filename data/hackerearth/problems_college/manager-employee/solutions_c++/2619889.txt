#include <bits/stdc++.h>

using namespace std;

#define ll long long int
#define s(x) scanf("%d", &x)
#define sd(x) scanf("%lf", &x)
#define sl(x) scanf("%lld", &x)
#define pb push_back
#define mp make_pair
#define mod 1000000007

int main()
{
    int t, n, i, j;
    s(t);
    while (t--) {
        s(n);
        int a[n];
        for (i = 0; i < n; i++)
            s(a[i]);
        sort(a, a+n);
        vector <int> v;
        for (i = 0, j = n-1; i < j; i++, j--)
            v.pb(a[i]+a[j]);
        sort(v.begin(), v.end());
        cout << v[v.size()-1]-v[0] << endl;
    }
    return 0;
}
