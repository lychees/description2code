#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		string s[101];
		int i = 0;
		while(i < n)
		{
			cin>>s[i];
			i++;
		}
		int a[30] = {0};
		i = 0;
		while(i < n)
		{
			int len = s[i].length();
			int j = 0;
			int temp[30] = {0};
			while(j < len)
			{
				if(temp[s[i].at(j)-'a'] == 0)
				{
					a[s[i].at(j)-'a']++;
				}
				temp[s[i].at(j)-'a'] = 1;
				j++;
			}
			i++;
		}
		i = 0;
		int max = 0;
		int fmax = a[i];
		while(i < 27)
		{
//			printf("%d ", a[i]);
			if(a[i] > fmax)
			{
				fmax = a[i];
				max = i;
			}
			i++;
		}
		printf("%c\n", max+'a');
	}
	return 0;
}
