#include<bits/stdc++.h>
using namespace std;

const int MOD = 1e9+7;

int main()
{
	long long int vec[1000001];
	vec[0]=0;
	vec[1]=1;
	vec[2]=2;
	for(int i=3;i<=1000000;i++)
	{
		vec[i]=vec[i-1]+(i-1)*vec[i-2];
		vec[i]%=MOD;
	}
	int t;
	cin >> t;
	while(t--)
	{
		int n;
		cin >> n;
		cout << vec[n]%MOD << endl;
	}
	return 0;
}
