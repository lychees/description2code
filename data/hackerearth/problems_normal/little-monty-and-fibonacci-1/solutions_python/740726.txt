def cnt(x, n):
	if x < 3:
		return n
	return n // x

def allCounts(inputNums):
	t = len(inputNums)
	result = [0]*t
	for i in xrange(t):
		res = 0
		n = inputNums[i]
		for x in xrange(1, n+1):
			res += cnt(x, n)
		result[i] = res
	return result

t = input()
inputNums = [0]*t
for i in xrange(t):
	inputNums[i] = input()

result = allCounts(inputNums)
for r in result:
	print r
