#include <iostream>
#include <stdio.h>
using namespace std;

typedef struct node
{
	int data;
	struct node *next;
}node;
node *head;

node* Delete(node *head1, int st, int np) // n is position from beginning
{
  node *temp = head1;
  node *p_temp;
  int offset;
  offset = st % np;
  if(offset == 0)
  {
      while(temp -> next != head1)
      {
          temp = temp -> next;
      }
      (temp -> next) = (temp -> next) -> next;
  }
  else
  {
      for(int i = 0; i<offset; ++i)
    {
        p_temp = temp;
        temp = temp -> next;
    }
  p_temp -> next = temp -> next;
  }
  return temp -> next;
}

int main()
{
    int t, n, k, ans;
    scanf("%d",&t);
    while(t--)
    {
    	head = NULL;
    	scanf("%d %d",&n,&k);
    	node *p;
    	for(int i = 0; i < n; ++i)
    	{
    		node *temp = new node;
    		temp -> data = i+1;
    		if(head == NULL)
    		{
    			head = temp;
    			temp -> next = head;
    		}
    		else
    		{
    			p = head;
    			while(p -> next != head)
    				p = p -> next;
    			temp -> next = p->next;
    			p -> next = temp;
    		}
    	}
    	p = head;
    	while(1)
    	{
            if(head -> next == head)
            {
                ans = head -> data;
                break;
            }
            else
            {
                p = Delete(p,k,n);
                --n;
            }
            head = p;
    	}
    	printf("%d\n",ans);
    }
    return 0;
}
