#include <iostream>

#include<string.h>
int main()
{
    int t,i,j,l,co,k=0,cx;
    char a[1000],b[1000],c[1000];
    scanf("%d",&t);
    while(t--)
    {
    	k=0;
    	co=0;cx=0;
    	a[0]='\0';
    	j=0;
    	scanf("%s",a);
    	l=strlen(a);
    for(i=0;i<l;i++)
    {
    	cx=0;co=0;
    	while(a[i]!='*'&&i<l)
    	{
    		if(a[i]=='x')
    		cx=1;
    		if(a[i]=='o')
    		co=1;
    		i++;
    	}
    	for(j=k;j<i;j++)
    	{
    		if(cx==1&&co==1&&a[j]=='x'||cx==1&&co==0&&a[j]=='x')
    		printf("x");
    		else
    		if(cx==0&&co==1&&a[j]=='o')
    		printf("o");
    	}
    	if(a[i]=='*')
    	printf("*");
    	k=j;
    }
    printf("\n");
    }
}
