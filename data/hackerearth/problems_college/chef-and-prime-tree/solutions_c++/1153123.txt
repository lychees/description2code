#include<iostream>
using namespace std;
int p[5000001];
int calc( int a ,int b, int sum){
	
	 if( a > b)
	 return calc( b , a , sum);
     else
       if( a == b)
         {
         	if( p[a])
         	sum-=a;
         	else
         	sum+=a;
         	return sum;
         }
         else
         {
              if(p[b])
			  sum-=b;
			  else
			  sum+=b;
			  return calc( a , b >> 1 ,sum );      	
         }
 	
}
int main()
{
     int i , j;
	  for(i = 0 ; i <= 5000000; ++i )
	    p[i] = 1;
		for( i = 2; i <= 2240; ++i ){
			  if(p[i])
			  for(j = i*i; j <= 5000000; j=j+i )
			  p[j] = 0;
		}	
		p[0]=0;
		p[1]=0;
	    int t;
		cin >> t ;
		while(t--){
			   int a , b;
			   cin >> a >> b;
			   cout << calc( a , b , 0 ) << endl;	 
		} 
	return 0;
}
