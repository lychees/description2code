//Calculating Inverse Modulus 

// When we do (a^b)%m then we can also do ((a%m)^(b%m))%m...Sometimes in some problems we may have large base which can be treated like this

#include<bits/stdc++.h>
using namespace std;

#define mp make_pair
#define pb push_back

typedef long long ll;

const long long MOD=1e9+7;

long long modex(long long base,long long expo){

     long long ans=1;

     while(expo!=0){    

           if((expo%2)==1){

                ans*=base;
                ans%=MOD;

           }

           base=base*base;
         
           base%=MOD;

           expo/=2;
     }

     return ans;
 
}

void solve(){

    ll n;
 
    cin>>n;

    ll ans=modex(2,n-1);

    ans=modex(ans,MOD-2);

    cout<<ans%MOD<<'\n';
}

int main(){

    int test;

    cin>>test;

    while(test--)
        solve();  
 
    return 0;

}
