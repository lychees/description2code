#include <iostream>
using namespace std;

int main()
{
   int t;cin>>t;
   while(t--){
   	long long int n,i,m,day,sum=0,first=0,last=0;
   	cin>>n;
   	long long int *A;
   	A=new long long int[n];
   	for(i=0;i<n;i++)
   	scanf("%lld",&A[i]);
   	
   	for(i=0;i<n;i++)
   	{if(first==0&&A[i]!=0)first=i+1;break;}
   	for(i=0;i<n;i++)
   	{if(A[i]!=0)last=i+1;}
   	scanf("%lld",&m);
   	if(m==0)day=first;
   	else{
   	for(i=0;i<n;i++)
   	sum+=A[i];
   	m%=sum;
   	if(m==0)day=last;
   	else
   	{for(i=0;i<n;i++)
   		{
   			m-=A[i];
   			if(m<=0)
   			{day=++i;
   			break;}
   		}	
   	  }
   	}
   	cout<<day<<endl;
   }
}
