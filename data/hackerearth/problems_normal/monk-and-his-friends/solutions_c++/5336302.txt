#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;
		long long x;
		cin>>n>>m;
		map<long long,int> M;
		map<long long,int>::iterator it;
		for(int i=0;i<n;i++)
		{
			scanf("%lld",&x);
			M[x]=1;
		}
		for(int i=0;i<m;i++)
		{
			scanf("%lld",&x);
			it=M.find(x);
			if(it==M.end())
			{
				printf("NO\n");
				M[x]=1;
			}
			else
				printf("YES\n");
		}
	}
	return 0;
}
