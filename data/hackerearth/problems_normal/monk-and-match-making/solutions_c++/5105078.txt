#include <algorithm>
#include <iostream>
#include <string>
#include <cstdio>
#include <vector>
using namespace std;

int main()
{
    int n, t, q1, q2, w1, w2, q3=-1, q4=-1, w3=-1, w4=-1, k=0;
    string s1;
    vector<int> v();
    cin >> s1;
    //scanf(" %[^\t\n]s",&s1); 
    cin >> t;
    for(int tt=0; tt<t; tt++)
    {
    	cin >> q1 >> q2 >> w1 >> w2;
    	q1--;
    	q2--;
    	w1--;
    	w2--;
    	if(q3==q1 && q4==q2 && w3==w1 && w4==w2)
    	{
    		if (k==1){cout << "No"<<endl;}
	    	else{cout << "Yes"<<endl;}
    	}
    	else{
    	if((w2-w1)!=(q2-q1)){cout << "No"<<endl;k=1;}
    	else
    	{
    		string s2=s1.substr (q1,q2-q1+1);
    		string s3=s1.substr (w1,w2-w1+1);
    		if (s2.compare(s3) != 0){cout << "No"<<endl;k=1;}
	    	else{cout << "Yes"<<endl;k=0;}
	    	s2.clear();
	    	s3.clear();
    	}}
    	q3=q1;
    	q4=q2;
    	w3=w1;
    	w4=w2;
    }
    return 0;
}
