#include <iostream>
#include<vector>
using namespace std;

int main()
{
	long int arr[26],t,n,j;
	string s,str;
	long int q=0;
	
	cin>>t;
	while(t--){
		for(int i=0;i<26;++i)
		arr[i]=0;
		
		cin>>n>>s;
		for(long int i=0;i<s.length();++i){
			arr[int(s[i])-97]++;
		}
	
	long int maxq=0,index=0,size=0;
	for(long int i=0;i<n;++i){
		cin>>str;
		q=0;
		for(long int j=0;j<str.length();++j){
			q+=arr[int(str[j])-97];
		}
		if(i==0){
			if(q==0){
				size=str.length();
			}
		}
	//	cout<<q<<" ";
		if(q>maxq){
			maxq=q;
			index=i;
			size=str.length();
		}
		if(q==maxq){
			
			if(str.length()<size)
				{
				index=i;
				size=str.length();
				}
				
		}
	
	
	}
	cout<<index+1<<endl;
	
	}
    
    return 0;
}
