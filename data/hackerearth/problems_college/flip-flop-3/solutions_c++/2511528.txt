    #include <iostream>
    #include <algorithm>
    #include <cstring>
    
    
    using namespace std;
    int main()
    {
        int t;
        cin>>t;
        while(t--){
            string str;
            cin>>str;
            int count=0;
            for(int i=0;i<str.length()-1;i++){
                if(str[i]=='X'){
                    if(str[i+1]!='Y'){count++;}
                }else if(str[i]=='Y'){
                    if(str[i+1]!='X'){count++;}
                }
            }
            cout<<count<<"\n";
        }
        return 0;
    }
