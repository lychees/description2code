using namespace std;
const int N = 100005;
int h[N],a[N];
int hc[N];

int main() {
    int n; cin>>n;
    for(int i = 1;i <= n;i++) {
        scanf("%d %d",&h[i],&a[i]);
        hc[h[i]]++;
    }
    for(int i = 1;i <= n;i++) {
        long long ans = 0;
        ans = n - 1 + hc[a[i]];
        cout<<ans<<' '<<(2 * n - 2 - ans)<<endl;
        
    }
    
}
