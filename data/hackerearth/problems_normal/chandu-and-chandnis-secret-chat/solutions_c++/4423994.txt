#include<bits/stdc++.h>
using namespace std;
string str;
bool cmp(int i,int j){
return (str[i]!=str[j])? str[i]>str[j]: i<j;
}
int main()
	{
		int t;
		cin>>t;
		while(t--)
			{
				
				int k;
				cin>>str>>k;
				int n=str.length();
				int in[30001];
				for(int i=0;i<n;++i)in[i]=i;
				sort(in,in+n,cmp);
				k--;
				for(int i=0;i<n;++i,k=in[k])
					cout<<str[k];
				cout<<endl;
			}
		return 0;
	}
