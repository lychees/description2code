#include <iostream>
#include <string>
#include <unordered_set>

using namespace std;

int main()
{
    unordered_set<string> myset;
    string s;
    cin>>s;
    double countways=0;
    //length of substring
    for (int j=1; j<=s.length(); j++)
    {
        //for each j
        for (int i=0; i<s.length()-j+1; i++)
        {
            if (myset.find(s.substr(i, j))==myset.end())
            {
                countways++;
                myset.insert(s.substr(i, j));
            }

        }
        myset.clear();
    }
    cout<<countways<<endl;
    return 0;
}
