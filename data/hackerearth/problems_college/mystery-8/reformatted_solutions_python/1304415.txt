'''
Output is input_string[distance of * from end]
'''
t = input()
while t > 0:
    t -= 1
    string = raw_input()
    i = 0
    starIndex = -1
    for char in string[::-1]:
        if char == '*':
            starIndex = i
        elif char.isalpha():
            i += 1

    for char in string:
        if char.isalpha() or char == '*':
            starIndex -= 1
            if starIndex == -1:
                print char
                break
