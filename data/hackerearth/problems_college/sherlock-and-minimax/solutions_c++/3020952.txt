#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

const int lmt = 105;
int in[lmt];
vector<int> tocheck;

int main(){
    int n,p,q;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
        scanf("%d",&in[i]);
    sort(in,in+n);
    scanf("%d %d",&p,&q);
    tocheck.pb(p);
    tocheck.pb(q);
    for(int i=0;i<n-1;i++){
        int tmp = (in[i]+in[i+1])/2;
        if(tmp>=p && tmp<=q)
            tocheck.pb(tmp);
        tmp++;
        if(tmp>=p && tmp<=q)
            tocheck.pb(tmp);
    }
    int ans = 0 , mx = -1;
    for(int i=0;i<tocheck.size();i++){
        int loc = INT_MAX;
        for(int j=0;j<n;j++)
            loc = min(loc,abs(tocheck[i]-in[j]));
        if(loc > mx){
            mx = loc;
            ans = tocheck[i];
        }
    }
    printf("%d\n",ans);
    return 0;
}
