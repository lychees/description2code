#include <iostream>
using namespace std;

int main()
{
    int t,n,i,c;
    int a[10]={1, 2, 5, 10, 20, 50, 100, 500, 1000};
    cin>>t;
    while(t--){
    	cin>>n;
    	c=0;
    	for(i=8;i>=0;i--){
    		if(n>=a[i]){
    			c+=n/a[i];
    			n=n%a[i];
    		}
    		if(n==0)
    			break;
    	}
    	cout<<c<<endl;
    }
    return 0;
}
