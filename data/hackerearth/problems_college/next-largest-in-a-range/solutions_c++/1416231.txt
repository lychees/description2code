using namespace std;
int read_int()
{
register char c=gc();
while(c<'0'||c>'9')
    c=gc();  //ignore all characters till the first digit
int ret=0;
while(c>='0' && c<='9') //till we get a stream of digits.As soon as we get something else,we stop
{
ret=ret*10+(c-48);//construct the number
c=gc();
}
return ret;
}
int t,i,n,s,m,v,k,l,r,q,x,flag;
int a[100001],b[100001];
int main()
{
scanf("%d%d%d%d",&n,&l,&r,&q);
for(i=0;i<n;i++)
    scanf("%d",&a[i]);

v=0;
for(i=l;i<=r;i++)
    b[v++]=a[i];

sort(b,b+v);
while(q--){
    scanf("%d",&x);
    flag=0;
    for(i=0;i<v;i++){
        if(b[i]>x){
            flag=1;
            break;
            }
    }
    if(flag)
    cout<<b[i]<<endl;
    else
        cout<<"-1"<<endl;

}



	return 0;
}
