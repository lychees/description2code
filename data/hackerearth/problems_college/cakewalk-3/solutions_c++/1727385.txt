#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	string s;
	cin>>s;
	string answer="";
	for(int i=0;i<n-1;i++)
	{
		if(i!=0)
		s=answer;
		answer="";
		int a;
		string p;
		cin>>a>>p;
		for(int j=0;j<a;j++)
		{
			answer=answer+s[j];
		}
		answer=answer+p;
	}
	cout<<answer;
}
