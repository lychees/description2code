#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int n,m;
    	cin>>n>>m;
      	int i;
      	int a[n];
      	int j;
      	j=n-1;
      	for(i=0;i<n;i+=2)
      	{
      		a[i]=j;
      		j--;
      	}
      	j=0;
      	for(i=1;i<n;i+=2)
      	{
      		a[i]=j;
      		j++;
      	}
       	for(i=0;i<n;i++)
      	{
      		if(a[i]==m)
      			break;
      	}    	
      	cout<<i<<endl;
    	
    }
}
