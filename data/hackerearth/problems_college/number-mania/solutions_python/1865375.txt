import math

def findPrimeNumers(noRange):
	sqrt_no = int(math.sqrt(noRange))
	for no in range(2,sqrt_no+1):
		if noRange%no == 0:
			return 0

	return 1

def findFactorial(no):	
	factoril = 1;
	for no in range(1,no+1):
		factoril = factoril*no

	return factoril
