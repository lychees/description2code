#include<iostream>
using namespace std;
int main()
{
    int t,flag;
    string txt,pat;
    cin>>t;
    while(t--)
    {
        flag=1;
        cin>>pat>>txt;
        int arr[26],arr_pat[26];
        for(int i=0;i<26;++i)
        {
            arr[i]=0;
            arr_pat[i]=0;
        }
        for(int i=0;i<pat.length();++i)
        {
             arr[txt[i]-'a']++;
             arr_pat[pat[i]-'a']++;

        }

        for(int i=0;i<26;++i)
        {
            if(arr[i]!=arr_pat[i])
            {
                flag=0;
                break;
            }
        }
        if(flag==1)
            cout<<"YES\n";
        else
        {
            for(int i=pat.length();i<txt.length();i++)
            {

                arr[txt[i-pat.length()]-'a']--;
                arr[txt[i]-'a']++;
                int k=0;
                for(k=0;k<26;++k)
                {
                    if(arr[k]!=arr_pat[k])
                    {
                        flag=0;
                        break;
                    }
                }
                if(k==26)
                {
                    cout<<"YES\n";
                    flag=1;
                    break;
                }

            }
            if(flag==0)
            cout<<"NO\n";

        }


    }
return 0;
}
