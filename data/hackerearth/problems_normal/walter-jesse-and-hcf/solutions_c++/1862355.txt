#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll gcd(ll a,ll b)
{
	if(a==0)return b;
	return gcd(b%a,a);
}
int main()
{
	ll t,x,y,k,tmp,flag,i;
	scanf("%lld",&t);
	while(t--)
	{
		flag=1;
		scanf("%lld%lld%lld",&x,&y,&k);
		tmp=gcd(x,y);
		ll count=0;
		for(i=tmp;i>=1;i--)
		{
			if(x%i==0&&y%i==0)
				count++;
			if(count==k)
			{
				printf("%lld\n",i);
				flag=0;
				break;
			}
		}
		if(flag)
			printf("No crymeth today\n");
	}	
}
