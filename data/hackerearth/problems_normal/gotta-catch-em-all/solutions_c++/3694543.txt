#include <bits/stdc++.h>
using namespace std;

vector<int>pok;

int main()
{
	int k;
	cin >> k;
	for(int i=0; i<k; ++i){
		int a;
		cin >> a;
		pok.push_back(a);
	}
	sort(pok.begin(), pok.end());
	int ans = 0;
	for(int i=k-1; i>=0; --i)
		ans = max(ans, pok[i]+(k-i));
	cout << ans+1 << endl;
    //cout << "Hello World!" << endl;
    return 0;
}
