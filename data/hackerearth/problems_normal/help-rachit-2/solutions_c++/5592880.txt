
#include<cstdio>
#include<cstring>
#include<vector>
#include<cmath>
#define M 10000
using namespace std;
vector<int>vwct;
bool flag[M+11];
double ans[M+11];
void init(){
    memset(flag,true,sizeof(flag));
    memset(ans,0,sizeof(ans));
    vwct.push_back(2);
    for(int i=4;i<=M;i+=2) flag[i]=false;
    for(int i=3;i<=M;i+=2){
        if(flag[i]){
            vwct.push_back(i);
            for(int j=i+i;j<=M;j+=i){
                flag[j]=false;
            }
        }
    }
}
bool prime(int n){
    int q=sqrt(n);
    for(int i=0;i<vwct.size()&&vwct[i]<=q;i++){
        if(n%vwct[i]==0)
            return false;
    }
    return true;
}
void solve(){
    ans[1]=0;
    int step=0,p=0,cnt=1,b=1;
    for(int i=3;i<=M;i+=2){
        step+=2;
        for(int j=1;j<=4;j++){
            b+=step;
            if(prime(b)){
                p++;
            }
        }
        cnt+=4;
        ans[i]=p*1.0/cnt;
    }
}
int main(){
    init();
    int cs;
    solve();
    scanf("%d",&cs);
    while(cs--){
        int n;
        scanf("%d",&n);
        printf("%.6lf\n",ans[n]*100);
    }
    return 0;
}
