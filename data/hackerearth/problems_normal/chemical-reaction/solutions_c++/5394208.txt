ll seg[1500000];
void init(ll nd,ll l,ll r)
{
	seg[nd]=r-l+1;
	if(l==r)
	return;
	ll mid=(l+r)/2;
	init(nd*2,l,mid);
	init(nd*2+1,mid+1,r);
}
void update(ll nd,ll l,ll r,ll ind)
{
	ll mid=(l+r)/2;
	seg[nd]--;
	if(l==r)
	return;
	if(ind<=mid)
	update(nd*2,l,mid,ind);		
	else
	update(nd*2+1,mid+1,r,ind);
}
ll quer(ll nd,ll l,ll r,ll ind)
{
	ll mid=(l+r)/2;
	if(l==r)
	return l;
	if(seg[nd*2]<ind)
	return quer(nd*2+1,mid+1,r,ind-seg[nd*2]);
	else
	return quer(nd*2,l,mid,ind);
}
