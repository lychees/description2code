#include <iostream>

using namespace std;

#define MOD 1000000007

typedef long long int ll;

ll choose(ll N, ll K)
{
	ll ret = 1;
	for (ll i = N - K + 1; i <= N; i++)
	{
		ret = (ret%MOD * i%MOD) % MOD;
	}
	return ret;
}

int main()
{
	int T;
	ll N, K;
	cin >> T;
	while (T--)
	{
		cin >> N >> K;
		cout << choose(K, N) << endl;
	}
	system("pause");
	return 0;
}
