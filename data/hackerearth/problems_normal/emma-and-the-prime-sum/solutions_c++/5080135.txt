#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main()
{
    bitset<1001000> bs;
    bs.set();
    bs[0] = bs[1] = 0;
    for(ll i = 2; i <= 1000000; i++)
    	if(bs[i])
    		for(ll j = i * i; j <= 1000000; j += i)
    			bs[j] = 0;
    
    ll dp[1000010];
    dp[0] = 0;
    for(int i = 1; i <= 1000000; i++)
    	dp[i] = dp[i - 1] + bs[i] * i;
    	
    int t, x, y;
    scanf("%d", &t);
    while(t--)
    {
    	scanf("%d %d", &x, &y);
    	printf("%lld\n", dp[y] - dp[x - 1]);
    }
    return 0;
}
