n = int(raw_input())
while (n != 0):
    n -= 1
    s = raw_input()
    k = s.index("$") + 1
    ans = 0
    while (k < len(s)):
        if (s[k] == " "):
            k += 1
        else:
            if (ord(s[k]) <= 57) and (ord(s[k]) >= 48):
                ans = (10*ans) + int(s[k])
            else:
                k = len(s)
            k += 1
    print("$" + str(ans))
