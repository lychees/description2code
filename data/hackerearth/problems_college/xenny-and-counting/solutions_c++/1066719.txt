#include<bits/stdc++.h>
using namespace std;

int main()
{
    ios::sync_with_stdio(0);
    int t;
    cin>>t;
    while(t--)
    {
        int n,k;
        cin>>n>>k;
        int x=n+k;
        long long ans=1;
        for(int i=1;i<n;i++)
        {
            ans=(ans*(x-i))/i;
        }
        cout<<ans<<endl;
    }
    return 0;
}
