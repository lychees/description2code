#include <bits/stdc++.h>
using namespace std;

string a[1005];
int sz;
map<string,int>st;

int main()
{
	sz=0;
	while(cin>>a[sz])
	{
	    int i=sz;
        for(int j=0;j<a[i].size();j++)
        {
            if(a[i][j]>='a'&&a[i][j]<='z')a[i][j]=a[i][j]-'a'+'A';
        }
		st[a[sz]]++;
		sz++;
	}
	for(int i=0;i<sz;i++)
	{
		int k=st[ a[i] ];
		if(k)
		{
			st[ a[i] ]=0;
			cout<<a[i]<<" "<<k<<endl;
		}
	}
    return 0;
}
