#include <iostream>
using namespace std;

int main()
{
   long int n,i,k,low,high,found,mid,p;
   cin>>n;
   long int a[n];
   for(i=0;i<n;i++)
   {
   	cin>>a[i];
   }
   cin>>k;
   while(k--)
   {
   	cin>>p;
   	low=0,high=n-1,found=0;
   	mid=(low+high)/2;
   	while(low<=high)
   	{
   		if(a[mid]==p)
   		{
   			cout<<mid<<endl;
   			found=1;
   			break;
   		}
   		else if(a[mid]>p)
   		{
   			high=mid-1;	
   		}
   		else
   		{
   			low=mid+1;
   		}
   		mid=(high+low)/2;
   	}
   	if(found==0)
   	{
   		cout<<-1<<endl;
   	}
   }
   
    return 0;
}
