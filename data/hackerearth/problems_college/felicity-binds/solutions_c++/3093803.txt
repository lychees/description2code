#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[1503];
    	int b[1503];
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	for(int j=0;j<n;j++)
    	cin>>b[j];
    	sort(a,a+n);
    	sort(b,b+n);
    	int sum=0;
    	for(int i=0;i<n;i++)
    	{
    		sum+=a[i]*b[i];
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
