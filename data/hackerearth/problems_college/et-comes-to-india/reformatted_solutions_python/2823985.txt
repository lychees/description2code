def sol(n, k):
    L = (2**n) - 1
    if k > L:
        return '-1'
    if k == L/2+1:
        return 'G'
    if k < L/2+1:
        return sol(n-1, k)
    else:
        ans = sol(n, L+1-k)
        if ans == '-1':
            return ans
        if ans == 'G':
            return 'C'
        else:
            return 'G'


for t in range(input()):
    n, k = map(int, raw_input().split())
    print sol(n, k)
