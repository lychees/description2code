#include <iostream>
#include <cstring>
#include<cstdio>
using namespace std;

int main()
{
    char str[100000];
    cin.getline(str,100000);
    int len = strlen(str);
    int n;
    scanf("%d",&n);
    
    for(int i=0;i<len;i=i+n)
    {
    	for(int j =0; (j<n && j+i<len);j++)
    		cout<<str[i+j];
    	cout<<endl;
    }
    
}
