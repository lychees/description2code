#include <iostream>

using namespace std;

int fact(int num) {
	if(num==0)
		return 0;
	if(num==1)
		return 1;
	return num*fact(num-1);
}

int Rsum(int i) {
	if(i/10 == 0)
		return i;
	return Rsum(i/10+i%10);
}

int calc_sum(int num) {
	int f;
	if(num>=6)
		return 9;
	else {
		f = fact(num);
		return Rsum(f);
	}
}

int main() {

	int T, sum;
	cin >> T;
	while(T--) {
		sum = 0;
		int a, b;
		cin >> a >> b;
		if(a<6)
		for(int i=a; i<=(b<6?b:5); i++) {
			sum += calc_sum(i);
		}
		//cout << "Sum1 = " << sum << endl;
		if(b>=6) {
        int val = b-((a>6)?a:6)+1;
        //cout << "Val = " << val << endl;
		sum = sum + val*9;
		//cout << "Sum2 = " << sum << endl;
		}



		//sum = calc_sum(b);
		cout << sum << endl;
	}
}
