#include<bits/stdc++.h>
using namespace std;
int dp[1000001];
int main()
{
	int n,m,q,i,a,b,l,r,mid,ans;
	cin>>n;
	cin>>m;
	for(i=0;i<m;i++)
	{
		cin>>a>>b;
		dp[a]+=1;
		dp[b+1]+=-1;
	}
	for(i=1;i<=n;i++)
	{
		dp[i]=dp[i-1]+dp[i];
			
	}
	sort(dp,dp+n+1);
	cin>>q;
	for(i=0;i<q;i++)
	{
		cin>>a;
		l=1;r=n;
		ans=n+1;
		while(l<=r)
		{
			mid=(l+r)/2;
			if(dp[mid]>=a)
			{
				ans=mid;
				r=mid-1;
			}
			else
			{
				l=mid+1;
			}
		}
		cout<<n-ans+1<<endl;
	}
}
