#include<cstdio>
#include<cstdlib>
#include<iostream>
#define SZ 100000
#define DF 10
#define ci(i) (i-1)
using namespace std;

long long int A[DF][SZ];

long long int GetValueAt(int r,int indx)
{
	return (indx<0)?0:A[r][indx];
}

int main()
{
	int N,Q;
	int B[SZ];
	cin>>N>>Q;
	for(int i=0;i<N;i++)
	{
		cin>>B[i];
	}

	// preprocess array

	for(int i=0;i<10;i++)
	{
		for(int j=0;j<N;j++)
		{
			
				A[i][j] = GetValueAt(i,j-i-1) + B[j];
		}
	}

	/*for(int i=0;i<10;i++)
	{
		for(int j=0;j<N;j++)
		{
			
				cout<<A[i][j]<<" ";
		}
		cout<<"\n";
	}*/
	while(Q--)
	{
		int L,R,K;
        cin>>L>>R>>K;
		
		R=L+((R-L)/K)*K;
		L--;
		R--;
		cout<<(GetValueAt(K-1,R)-GetValueAt(K-1,L-K))<<"\n";
	}
	return 0;
}
