#include <iostream>
#include <string.h>
using namespace std;
#define clear_  memset(arr,0,sizeof(arr)),memset(brr,0,sizeof(brr)),memset(beep,0,sizeof(beep)),memset(intensity,0,sizeof(intensity));
int arr[100002],brr[100002];
int beep[1002],intensity[1002];
int main()
{
	ios_base::sync_with_stdio(false);
	int n,temp;
	clear_;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>temp;

		for(int j=1;j*temp<=100000;j++)
		{
			arr[j*temp]=1;
			brr[j*temp]++;
		}

	}
	for(int i=1;i<=100000;i++)
	{
		arr[i]+=arr[i-1];
		brr[i]=max(brr[i],brr[i-1]);
	}
	int p,q,t;
	cin>>p;
	for(int i=0;i<p;i++)
	{
		cin>>beep[i]>>intensity[i];
	}
	cin>>q;
	for(int i=0;i<q;i++)
	{
		cin>>temp>>t;
		if(arr[t]>beep[temp-1]||brr[t]>intensity[temp-1])
		{
			cout<<"Awake"<<endl;
		}
		else
			cout<<"Asleep"<<endl;
	}
	return 0;
}
