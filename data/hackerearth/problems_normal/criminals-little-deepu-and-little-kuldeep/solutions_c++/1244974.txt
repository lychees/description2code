#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;cin>>n;
    	int arr[n],i,j,k,s=0,brr[n];
    	for(i=0;i<n;i++)
    					cin>>arr[i];
    	sort(arr,arr+n);
    	k=0;
    	for(i=0;i<n;){k=0;
    					j=i;
    					while(arr[j]==arr[i] && j<n){
    											k++;
    											j++;
    					}
    					i=j;
    		brr[s++]=k;			
    	}
    	sort(brr,brr+s);
    	cout<<brr[s-1]<<"\n";
    }
    return 0;
}
