#include <bits/stdc++.h>
using namespace std;
int b[1007][1007];

int main()
{
  int t;
  cin>>t;
  while(t--)
  {
    int n,m;
    cin>>n>>m;
    int a[40][40];
    for (int i = 0; i < n; ++i)
    {
      string b;
      cin>>b;
      for(int j = 0;j< m;j++)
       { if(b[j]=='0')
          a[i][j]=0;
        else
          a[i][j] = 1;
      }
    }
    
    int x =-1,y = -1;
    int p =0,s = 0;
    while(p!=n*m)
    {
     p = 0;
     for (int i = 0; i < n; ++i)
    {
      for(int j = 0;j< m;j++)
        if(a[i][j]==0)
        {
          x = i;
          y = j;
        }
    }
    if((x<0)&&(y<0))
      break;
    for (int i = 0; i <= x; ++i)
    {
      for (int j = 0; j <= y; ++j)
      {
          a[i][j] = a[i][j]^1;
      }
    }
    for (int i = 0; i < n; ++i)
    {
      for (int j = 0; j < m; ++j)
      {
        if(a[i][j]==1)
          p++;
      }
    }
    s = s+1;
  }
  cout<<s<<endl;

  }
    

}
