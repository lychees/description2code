#include <iostream>
using namespace std;

int main()
{
    long long int t,n,id,ballpos,prev;
    char code;
    cin>>t;
    //cout<<t<<endl;
    while(t--)
    {
   		cin>>n>>ballpos;
    	//cout<<"n= "<<n<<" ballpos= "<<ballpos<<" prev= "<<prev<<endl;

   		while(n--)
   		{
    		cin>>code;
    		if(code=='B'){
    			
    			prev=prev^ballpos;
    			ballpos=prev^ballpos;
    			prev=prev^ballpos;
    		}
    		else{
    			cin>>id;
    			//cout<<"PASS TO "<<id<<endl;
    			prev=ballpos;
    			ballpos=id;
    		}
    		//cout<<"ballpos= "<<ballpos<<" prev= "<<prev<<endl;
    	}
    	cout<<"Player "<<ballpos<<endl;
    }
    
    return 0;
}
