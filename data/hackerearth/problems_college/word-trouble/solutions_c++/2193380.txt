#include<cstdio>
#include<cstring>
#include<cmath>
using namespace std;
int main()
{
	int t;
	scanf("%d ",&t);
	while(t--)
	{
		char s1[200],s2[200],ws[10];
		scanf("%s",s1);
		scanf("%s",s2);
		if(t!=0)
			gets(ws);
		int l1=strlen(s1);
		int l2=strlen(s2);
		int cnt=0;
		cnt=abs(l1-l2);
		int ch;
		if(l2<l1)
			ch=l2;
		else
			ch=l1;
		for(int i=0;i<ch;i++)
			if(s1[i]!=s2[i])
				++cnt;
		if(cnt==0)
			printf("Exact\n");
		else
			printf("Non Exact %d\n",cnt);
	}
	return 0;
}
