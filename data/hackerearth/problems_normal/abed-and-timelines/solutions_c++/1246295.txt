#include <iostream>
#include <cmath>

using namespace std;

int main()
{
    int tc;
    cin>>tc;
    int result[tc];
    long long M,N,xi,yi,xf,yf;
    long long price,stepx,stepy;
    for(int k=1;k<=tc;k++)
    {
        cin>>N>>M;
        cin>>xi>>yi;
        cin>>xf>>yf;
        cin>>price;

        stepx=min(abs(xi-xf) , N-abs(xi-xf));
        stepy=min(abs(yi-yf), M-abs(yi-yf));

        result[k]=(stepx+stepy)*price;
    }

    for(int k=1;k<=tc;k++){
        cout<<result[k]<<endl;
        if(result[k]>1000){
            cout<<"Let go of the group."<<endl;
        }
        else{
            cout<<"You saved the group."<<endl;
        }
    }

}
