#include<iostream>
using namespace std;
int main()
{
	int t;
	int n;
	int val;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int a[100001]={0};
		for(int i=0;i<n;i++)
		{
			cin>>val;
			a[val]++;
		}
		for(int i=1;i<100001;i++)
		{
			if(a[i]%2==1)
			{
				cout<<i<<endl;
				break;
			}
		}
	}
	return 0;
}
