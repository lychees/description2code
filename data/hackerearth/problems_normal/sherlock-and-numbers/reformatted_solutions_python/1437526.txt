t = int(raw_input())
for i in range(t):
    (n, k, p) = raw_input().split()
    (n, k, p) = int(n), int(k), int(p)
    desk1 = p
    rem = raw_input().split()
    rem = map(int, rem)

    for num in rem:
        if num <= desk1:
            desk1 += 1
    if desk1 <= n:
        print desk1
    else:
        print -1
