for testcase in xrange(int(raw_input())):
    sorted_k_levels = sorted([int(token)
                             for token in raw_input().strip().split(' ')[1:]])
    lows = sorted_k_levels[:len(sorted_k_levels) / 2]
    reversed_highs = reversed(sorted_k_levels[len(sorted_k_levels) / 2:])
    team_scores = [low + high for low, high in zip(lows, reversed_highs)]
    print max(team_scores) - min(team_scores)
