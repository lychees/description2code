#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;
	
int main()
{
	int t,f;
	int a,b,ans;
	scanf("%d",&t);
	while(t--)
	{
		ans=0;
		scanf("%d %d",&a,&b);
		for(int i=a;i<=b;i++)
		{
			f=1;
			for(int j=2;j*j<=i;j++)
			{
				if(i%j==0)
				{
					f=0;
					break;
				}
			}
			if(f && i!=1)
				ans+=i;
		}
		printf("%d\n",ans);
	}
    return 0;
}
