#include<bits/stdc++.h>
#include<map>
#include<vector>
using namespace std;
typedef long long ll;
struct node
{
	string name;
	ll g;
	ll s;
	ll b;
};
map<string,struct node>mp;
map<string,int>done;
string arr[10001];
ll k=0;
bool comp(string aa,string bb)
{
    if(mp[aa].g>mp[bb].g)
        return true;
    else if(mp[aa].s>mp[bb].s)
        return true;
    else if(mp[aa].b>mp[bb].b)
        return true;
    else if(mp[aa].name>mp[bb].name)
        return true;
        else 
            return false;
}
int main()
{
	ll n;
	string str;
	cin>>n;
	for(ll i=0;i<n;i++)
	{
		for(ll j=0;j<3;j++)
		{
			cin>>str;
			if(done[str]==0)
			{
				done[str]=1;
				arr[k++]=str;
				mp[str].name=str;
			}
			if(j==0)
			mp[str].g++;
			if(j==1)
			mp[str].s++;
			if(j==2)
			mp[str].b++;
		}
	}
//	cout<<k<<endl;
//	sort(arr,arr+k,comp);
	for(ll i=0;i<k;i++)
	{
		for(ll j=i+1;j<k;j++)
		{
			if(mp[arr[i]].g>mp[arr[j]].g)
			{
				swap(arr[i],arr[j]);
				continue;
			}
			if(mp[arr[i]].g==mp[arr[j]].g && mp[arr[i]].s>mp[arr[j]].s)
			{
				swap(arr[i],arr[j]);
				continue;
			}
			if(mp[arr[i]].g==mp[arr[j]].g && mp[arr[i]].s==mp[arr[j]].s && mp[arr[i]].b>mp[arr[j]].b)
			{
				swap(arr[i],arr[j]);
				continue;
			}
			if(mp[arr[i]].g==mp[arr[j]].g && mp[arr[i]].s==mp[arr[j]].s && mp[arr[i]].b==mp[arr[j]].b && mp[arr[i]].name<mp[arr[j]].name)
			{
				swap(arr[i],arr[j]);
				continue;
			}
		}
	}
	for(ll i=k-1;i>=0;i--)
	{
		cout<<mp[arr[i]].name<<" "<<mp[arr[i]].g<<" "<<mp[arr[i]].s<<" "<<mp[arr[i]].b<<endl;
	}
	return 0;
}
