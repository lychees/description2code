#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    bool ans[t];
    for (int i=0;i<t;i++)
    {
    	int n;
    	cin>>n;
    	ans[i]=false;
    	long long a[n];
    	for (int j=0;j<n;j++)
    	{
    		cin>>a[j];
    	}
    	for(int m=0;m<n-1;m++)
    	{
    		for(int k=m+1;k<n;k++)
    		{
    			if(a[m]>a[k])
    				ans[i]=true;
    		}
    	}
    	
    }
    
        for (int i=0;i<t;i++)
    {
    
    if(ans[i]==true)
    {
    	cout<<"NO\n";
    }
    else
    {
    	cout<<"YES\n";
    }
    
    }
    return 0;
}
