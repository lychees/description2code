#include<iostream>
#include<map>
using namespace std;
int main()
{
    string s1;
    getline(cin,s1);
    char s2;
    map<char,int> maps;
    map<char,int>::iterator it;
    int maxi;
    for(int i=0;i<s1.length();i++)
    {
        maps[s1[i]]++;
    }
    maxi=0;
    for (it=maps.begin(); it!=maps.end(); ++it)
    {
        if(maxi<it->second)
        {
            maxi=it->second;
            s2=it->first;
        }
    }
    cout<<s2<<" "<<maxi;
    return 0;
}
