#include<bits/stdc++.h>
#define c cin
#define o cout
using namespace std;

int main(){
	int t;
	int i;
	c>>t;
	map <string, int> m;
	map <string, int>::iterator mit;
	string s;
	for(i=0; i<t; i++){
		c>>s;
		m[s]++;
	}
	for(mit=m.begin(); mit!=m.end(); mit++)
		o<<mit->first<<" "<<mit->second<<"\n";
	return 0;
}
