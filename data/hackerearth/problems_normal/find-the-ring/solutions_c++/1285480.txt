#include<bits/stdc++.h>
using namespace std;

int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		int index,n;
		
		cin>>index>>n;
		
		bool even = false;
		
		if(n%2 == 0) even = true;
		
		if(n==0) cout<<index<<endl;
		else
		{
			if(index == 1 && even == true) 
			{
				cout<<"1"<<endl;
			}
			if(index == 1 && even == false)
			{
				cout<<"0"<<endl;
			}
		
			if(index == 0 && even == true)
			{
				cout<<"0"<<endl;
			}
			if(index == 0 && even == false)
			{
				cout<<"1"<<endl;
			}
		
			if(index == 2 && even == true)
			{
				cout<<"0"<<endl;
			}
			if(index == 2 && even == false)
			{
				cout<<"1"<<endl;
			}
		}
	}
	return 0;
}
