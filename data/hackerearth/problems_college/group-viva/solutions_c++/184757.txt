int gcd(int a,int b)
{
    if(b>a)return gcd(b,a);
    if(b==0)return a;
    return gcd(b,a%b);
}
void solve()
{
    int r,n;
    sd(n);sd(r);
    r = r-1;
    int num = 0;
    for(int i=1;i<=sqrt(r);i++)
    {
        if(r%i==0)num+=2;
    }
    int p = sqrt(r);
    if(r==p*p)num--;
    int den = n;
    num = n - num;
    int g = gcd(num,den);
    num/=g;
    den/=g;
    cout<<num<<"/"<<den<<endl;

}

{
    int t;
    cin>>t;
    while(t--){
   solve();
    }
