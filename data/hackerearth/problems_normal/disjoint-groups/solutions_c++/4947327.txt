#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const ll MOD = 1e9 + 7;

ll binpow(ll a, ll k)
{
	ll res = 1;
	while (k) {
		if (k & 1)
			res = res * a % MOD;
		a = a * a % MOD;
		k >>= 1;
	}
	return res;
}

int main()
{
	ll n;
	cin >> n;

	vector<ll> v(n);
	ll sum = 0;
	for (auto& t : v) {
		cin >> t;
		sum ^= t;
	}

	if (sum) {
		cout << 0;
		return 0;
	}

	cout << binpow(2, n - 1) - 1;
}
