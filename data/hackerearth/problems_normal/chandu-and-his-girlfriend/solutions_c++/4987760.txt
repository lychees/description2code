#include <iostream>
#include <algorithm>
#include <cstdlib>
using namespace std;

int main()
{
	int test;
	cin >> test;
	
	while(test--){
		int elements;
		cin >> elements;
		
		int array[elements];
		
		for(int i = 0; i < elements; i++)
			scanf("%d", &array[i]);
		
		sort(array, array+elements);
		
		for(int i = elements-1; i>=0;i--)
			printf("%d ", array[i]);
		printf("\n");
	}
    
    return 0;
}
