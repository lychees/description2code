int isp[100005];

void sieve()
{
    int i,j;
    isp[0]=isp[1]=1;
    for(i=2;i<=100000;++i)
    {
        if(!isp[i])
        {
            for(j=1;i*j<=100000;++j)        isp[i*j]=i;
        }
    }
}

int cnt[100002];

int main()
{
    int i,n,x,q;
    i(n);
    sieve();
    while(n--)
    {
        i(x);
        while(x>1)
        {
            ++cnt[isp[x]];
            x/=isp[x];
        }
    }
    i(q);
    int mul,ct,ans;
    while(q--)
    {
        i(x);
        ans=INT_MAX;
        while(x>1)
        {
            ct=0;
            mul=isp[x];
            while(mul==isp[x])
            {
                ++ct;
                x/=isp[x];
            }
            ans=min(ans,cnt[mul]/ct);
        }
        o(ans);on;
    }
    return 0;
}
