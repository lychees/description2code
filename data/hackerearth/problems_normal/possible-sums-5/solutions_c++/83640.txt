#include<cstdio>

int poss[110][10010];
int ar[110];

int main(){
	int i,j,n,t;
	for(scanf("%d",&t);t--;){
		scanf("%d",&n);
		for(i=0;i<n;i++)
			scanf("%d",ar+i);
		poss[0][0] = 1;
		for(i=1;i<=n;i++){
			int te = ar[i-1];
			for(j=0;j<=10000;j++){
				poss[i][j] = poss[i-1][j];
				if(j>=te && poss[i-1][j-te])
					poss[i][j] = 1;
			}				
		}
		int count = 0;
		for(j=0;j<=10000;j++){
			if(poss[n][j])
				count++;
		}
		printf("%d\n",count);

	}
	return 0;
}
