#include <iostream>
#include <stdio.h>
#include <vector>
#include <algorithm>

using namespace std;

typedef long long int Lint;

Lint GCD( Lint a, Lint b ) {
	
	if( ( a % b ) == 0 )
		return b;
	else
		return GCD( b, a%b );
}

Lint gcd( vector<Lint>& A, Lint n ) {
	
	Lint i, g;
	g = A[ 0 ];
	
	for( i = 1;i < n;i++ )
		g = GCD( A[ i ], g );
	
	return g;
}

int main()
{
	
	Lint t, n, i, v;
	
	scanf("%lld", &t );
	
	while( t-- ) {
		
		scanf("%lld", &n );
		vector<Lint> A( n );
		
		for( i = 0;i < n;i++ )
			scanf("%lld", &A[ i ] );
		
		v = gcd( A, n );
		
		printf("%lld\n", ( v * n ) );
	}
	
	return 0;

}
