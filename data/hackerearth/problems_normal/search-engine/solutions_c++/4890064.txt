#include <iostream>
#include <bits/stdc++.h>
using namespace std;
struct trie
{
	struct trie *c[26];
	int v;
};
struct trie * getnode()
{
	struct trie *p;
	p=(struct trie *)malloc(sizeof(struct trie));
	int i;
	for(i=0;i<26;i++)
	{
		p->c[i]=NULL;
	}
	p->v=0;
	return p;
}
void insert(struct trie *root,string x,int e)
{
	struct trie *t=root;
	int i,l;
	l=x.size();
	for(i=0;i<l;i++)
	{
		if(t!=NULL&&t->c[x[i]-'a']==NULL)
		{
			t->c[x[i]-'a']=getnode();
		}
		if(t!=NULL)
		{
			t=t->c[x[i]-'a'];
		}
		if(t->v<e)
		{
			t->v=e;
		}
	}
}
int ser(struct trie *root,string x)
{
	int i,l;
	l=x.size();
	struct trie *t=root;
	for(i=0;i<l;i++)
	{
		if(t!=NULL&&t->c[x[i]-'a']==NULL)
		{
			return -1;
		}
		else
		{
			t=t->c[x[i]-'a'];
		}
	}
	return t->v;
}
int main() {
	// your code goes here
	struct trie *root=getnode();
	int n,q,i;
	string s;
	int l;
	cin>>n>>q;
	for(i=1;i<=n;i++)
	{
		cin>>s>>l;
		insert(root,s,l);
	}
	while(q--)
	{
		cin>>s;
		cout<<ser(root,s)<<endl;
	}
	return 0;
}
