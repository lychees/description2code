#include<iostream>
using namespace std;
long long int exp(long long int x,long long int n)
{
	if(n<0)
	return exp(1/x,-n);
	if(n==0)
	return 1;
	if(n==1)
	return x;
	if(n%2==0)
	return exp(x*x,n/2);
	else 
	return (x*exp(x*x,(n-1)/2));
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int a,b,c,d,ans;
		cin>>a>>b>>c;
		d=exp(a,b);
		
		int md=d%c;
		if(md>c/2)
		ans=d+c-md;
		else
		ans=d-md;
		cout<<ans<<"\n";
	}
}
