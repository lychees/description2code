#include <bits/stdc++.h>
using namespace std;
#define MAX 1000001

int lose[2000000];

void queensmove()
{
for(int i=0;i<MAX;i++)
lose[i]=-1;

double fi=(double)((1+sqrt(5))/2);

for(int i=0;i<MAX;i++)
lose[(int)(fi*(double)i)]=(int)((fi*(double)i))+i;
}

int main()
{
    int t;
    cin>>t;
    queensmove();
    while(t--)
    {
    	int n,m;
    	cin>>n>>m;
    	if(lose[n]==m)
    	cout<<"Chandu\n";
    	else
    	cout<<"Chandni\n";
    }
    return 0;
}
