#include <iostream>
using namespace std;

int main() {
    
    int t,n;
    cin>>t;int k=0,rem=0;;
    while(t--)
    {k=0;rem=0;
    	cin>>n;
    	if(n%2 !=0 || n<6 || n==10)
    	cout<<-1<<endl;
    	else
    	  { k=n/8;
    	    rem=n%8;
    	    if(rem==0)
    	    cout<<k<<endl;
    	    else
    	    cout<<k+1<<endl;
    	  	
    	  }
    }
    
	return 0;
}
