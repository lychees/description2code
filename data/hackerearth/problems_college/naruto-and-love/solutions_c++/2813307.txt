#include<iostream>
#include<algorithm>
#include<stdio.h>

using namespace std;

typedef long long ll;
int n, m;
int naruto[150001], food[150001];
bool valid(ll mid)
{
	int i = 0, j = 0;
	while (i < n && j < m)
	{
		if (abs(naruto[i] - food[j]) <= mid)
			i++, j++;
		else
			j++;
	}
	if (i == n)
		return true;
	else
		return false;
}
int main()
{
	int testcases;
	scanf("%d", &testcases);
	while (testcases--)
	{
		scanf("%d%d", &n,&m);
		for (int i = 0; i < n; i++)
			scanf("%lld", &naruto[i]);
		for (int i = 0; i < m; i++)
			scanf("%lld", &food[i]);
		sort(naruto, naruto + n);
		sort(food, food + m);
		ll left = 0, right = 1e12;
		while (left < right)
		{
			ll mid = (right - left) / 2 + left;
			if (valid(mid))
				right = mid;
			else
				left = mid + 1;
		}
		cout << right << endl;
	}
	return 0;
}
