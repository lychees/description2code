typedef long long LL;
using namespace std;
const int MOD = 1e9+7;
const int SIZE = 5*1e3+10;
#define LOCALTEST 0 // change this to 1 to perform localtest on "in.txt"
int n;
string s;
map<PII,LL> z;
LL rec(int i, int j) {
    if (j<i) return 0;
    if (i==j) return 1;
    if (z.count({i,j})) return z[{i,j}];
    return z[{i,j}]=rec(i, j-1) - rec(i+1, j-1) + rec(i+1, j) - rec(i+1, j-1) + rec(i+1, j-1) + (LL)(s[i] == s[j])*(1LL + rec(i+1, j-1));
}
int main(){
    if (LOCALTEST) {
        freopen("in.txt", "r", stdin);
        freopen("out.txt", "w", stdout);
    }
    cin>>n;
    while (n--) {
        cin>>s;
        z.clear();
        printf("%lld\n",rec(0,SZ(s)-1));
    }
    return 0;
}
