#include <iostream>
#include<stdio.h>
using namespace std;

void arrange(int *a,int n,bool &isArranged)
{  	
	int b[n];
	int t=a[0];
	int j=0;
	for(;j<n;j++)
	{
		if(t!=a[j])
		 break;
	}
	if(j!=n)
	{
		isArranged=false;
	
  	for(int i=0;i<n;i++)
  	{  
  		if(i > 0 && i < n-1)
  		{
  			if(a[i-1] ==1 && a[i+1] ==1)
  			{
  				//cout<<"Middle was sitting "<<i<< endl;
  				b[i]=1;
  			}
  			else
  			{
  				b[i]=0;
  			}
  		}
  		else
  		{
  			if(i==0 && a[i+1]==1)
  			{
  				//cout<<"2nd was sitting "<<i<< endl;
  				b[i]=1;
  			}
  			else if(i==n-1 && a[i-1]==1)
  			{
  				//cout<<"2nd Last was sitting "<<i<< endl;
  				b[i]=1;
  			}
  			else
  			b[i]=0;
  		}
  		
  	}
  	for(int i=0;i<n;i++)
  	{
  		a[i]=b[i];
  	}
  }
  else
    isArranged=true;
  	
}

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,m;
    	scanf("%d%d",&n,&m);
    	int a[n];
    	for(int i=0;i<n;i++)
    	scanf("%d",&a[i]);
    	bool isArranged=false;
    	while(m--)
    	{
    		arrange(a,n,isArranged);
    		//cout<<isArranged<<endl;
    		if(isArranged)
    		break;
    	}
    	for(int i=0;i<n;i++)
    	{
    		printf("%d ",a[i]);
    	}
    	cout<<endl;
    }
    return 0;
}
