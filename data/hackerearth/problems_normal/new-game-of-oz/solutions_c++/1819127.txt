#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
#include <unordered_map>

using namespace std;
int main()
{

  int T;
  scanf("%d",&T);
  while(T--)
  {
    int n;
    scanf("%d",&n);
    int a[n];
    unordered_map<int,int>mp;
    for(int i =0;i<n;i++)
    {
       scanf("%d",&a[i]);
       mp[a[i]]= 1;
    }
    sort(a,a+n);
    int count =0;
    for(int i =0;i<n;i++)
    {
     if(mp.find(a[i])!=mp.end())
     {
          count++;
          mp.erase(a[i]);
          int b = a[i];
          b++;
           if(mp.find(b)!=mp.end())
          {
          	  mp.erase(b);
          	

          }
          b= a[i];
          b--;
          if(mp.find(b)!=mp.end())
          {
             mp.erase(b);
             b--;  

          }

     }

    }

   printf("%d\n",count);
  }

return 0;
}
