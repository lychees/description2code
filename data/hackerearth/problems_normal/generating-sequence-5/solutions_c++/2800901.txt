#include<bits/stdc++.h>
using namespace std;
#define lint long long
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        lint g,n;
        cin>>g>>n;
        int j=2;
        for(int i=0;i<n-1;i++)
        {
            cout<<g*j<<" ";
            
        }
        j++;
        cout<<g*j<<endl;
    }

}
