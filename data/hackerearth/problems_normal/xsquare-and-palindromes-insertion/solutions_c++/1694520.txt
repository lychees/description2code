#include<iostream>
#include<string>
#include<string.h>
typedef unsigned long long int ll;

using std::string;
using std::cout;
using std::cin;

int main()
{
   ll occur[26];
   int index;
   string s;
   ll min;
   int T; 
   cin>>T;
   while(T)
   {
      index=-1;
      min=0;
      memset(occur,0,sizeof(occur));
      cin>>s;
      for(ll i=0;i<s.length();i++)
      {
         occur[s[i]-'a']++;
       }
      for(int i=0;i<26;i++)
      {
          if(occur[i]%2==0)
          {
               continue;
           }
           else
           if(index==-1)
           {
                index=i;
            }
            else 
            {
               if(occur[i]>occur[index])
               {
                    min+=1;
                    index=i;
                } 
               else if(occur[i]<=occur[index])
               {
                   min+=1;
                }
              }
          }
          cout<<min<<"\n";
          T--;
       }
}
