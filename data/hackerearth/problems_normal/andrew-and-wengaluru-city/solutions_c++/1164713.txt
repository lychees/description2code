#include <iostream>
#include <vector>
using namespace std;

#define M 1000000007

int trapWaterVolume2D(vector<int> A) {
    int n = A.size();
    vector<int> leftmost(n, 0), rightmost(n, 0);

    int leftMaxSoFar = 0;
    for (int i = 0; i < n; i++){
        leftmost[i] = leftMaxSoFar;
        if (A[i] > leftMaxSoFar) leftMaxSoFar = A[i];
    }

    int rightMaxSoFar = 0;
    for (int i = n - 1; i >= 0; i--){
        rightmost[i] = rightMaxSoFar;
        if (A[i] > rightMaxSoFar) rightMaxSoFar = A[i];
    }

    int vol = 0;
    for (int i = 0; i < n; i++){
        vol = (vol%M + max(0, min(leftmost[i], rightmost[i]) - A[i])%M)%M;
    }
    return vol;
}

int main()
{
	
    int t, n;
    cin >> t;
    while(t--){
    	cin >> n;
    	vector<int> vec(n);
    	for(int i = 0; i < n; i++) cin >> vec[i];
 		cout << trapWaterVolume2D(vec) << endl;	
    }
}
