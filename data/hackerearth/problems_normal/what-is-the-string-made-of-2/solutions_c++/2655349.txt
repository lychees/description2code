#include <iostream>
using namespace std;

int main()
{
   char a[100];
   int b[10]={6,2,5,5,4,5,6,3,7,6},k,sum=0;
   cin>>a;
   int i=0;
   while(a[i]!='\0')
   {
   	 k=a[i]-48;
   	for(int j=0;j<10;j++)
   	{
   		if(k==j)
   		{
   			sum=sum+b[j];
   		}
   	}
   	i++;
   }
   cout<<sum;
    return 0;
}
