#include <iostream>
using namespace std;

int main()
{
    int i,T,N,P,R,U,S,j;
    cin>>T;
    for(i=1;i<=T;i++){
    	cin>>N>>P>>U>>R>>S;
    	for(j=1;j<=N;j++){
    	   if(P+R>U){
    	      if(P-S<0){
    	      	break;
    	      }
    	      else{
    	      	P-=S;
    	      }
    	   }
    	   else{
    	   	P+=R;
    	   }
    	}
    	cout<<"Case #"<<i<<": "<<P<<endl;;
    }
    return 0;
}
