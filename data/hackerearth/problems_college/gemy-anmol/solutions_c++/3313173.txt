#include<stdio.h>
#include<stdlib.h>
int cmpfunc(const void *a, const void *b){
	return (*(int *)a - *(int *)b);
}

int main(){
	long long int t, n, m, i, sum, c; scanf("%lld", &t);
	while(t-- && scanf("%lld%lld", &n, &m)){
		long long int a[n], sum = c = 0;
		for(i = 0; i < n; scanf("%lld", &a[i++]));
		qsort(a, n, sizeof(long long int), cmpfunc);
		for(i = n - 1; i >= 0; i--){
			sum += a[i], c++; 
			if(sum >= m){printf("%lld\n", c); break;}
		}
		if(sum < m)	printf("-1\n");
	}		
	return 0;
}
