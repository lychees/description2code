#include <iostream>

using namespace std;

int main() {
        int T;
        cin >> T;
        while(T--)
        {
                long long int N;
                cin >> N;
                if(N<=3 && N>0) {
                	cout << 1 << endl;
                }
                else if(N%3 == 0) {
                        cout << N/3 << endl;
                } else {
                        cout << N << endl;
                }
        }
}
