#include <bits/stdc++.h>
using namespace std;

struct x {
    long long score, time;
};

struct x a[100005], sums[100005];

bool comp(struct x x1, struct x x2)
{
    if(x1.score == x2.score) {
        return x1.time < x2.time;
    } else return x1.score > x2.score;
}

int main()
{
    int n,q;
    cin>>n>>q;

    for(int i=0;i<n;i++) {
        cin>>a[i].time;
    }

    for(int i=0;i<n;i++) {
        cin>>a[i].score;
    }

    sort(a, a+n, comp);

    sums[0] = a[0];
    for(int i=1;i<n;i++) {
        sums[i].score = sums[i-1].score + a[i].score;
        sums[i].time = sums[i-1].time + a[i].time;
    }

    for(int i=0;i<q;i++) {
        int k;
        cin>>k;
        cout<<sums[k-1].time<<endl;
    }

    return 0;
}
