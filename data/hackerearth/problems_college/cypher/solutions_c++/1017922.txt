#include <iostream>
#include<cstdio>
using namespace std;

int main()
{
    int i,t,n;
    string s;
    string text;
    scanf("%d",&t);
    while(t--)
    {
    	text="";
    	cin>>s;
    	for(i=0;i<s.length();i++)
    	{
    		if((int)(s[i])+i > 122)
    		{
    			text+=(char)((int)(s[i])+i-26);
    		}
    		else
    		{
    			text+=(char)((int)(s[i])+i);
    		}
    	}
    	cout<<text<<endl;
    }
    return 0;
}
