#include <iostream>
using namespace std;
#define ll long long 
ll min(ll a,ll b){ return a<b?a:b;}

int main()
{
	int t;
	cin>>t;
	ll _min,L,B,H;
	while(t--)
	{
		scanf("%lld",&L);
		scanf("%lld",&B);
		scanf("%lld",&H);
		_min=min(L,B);
		_min=min(H,_min);
		if(_min&1)
		{
			printf("%lld",_min/2);
		    printf(".5 \n");
		}
		else
		{
			printf("%lld",_min/2);
		    printf(".0 \n");
		}
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
