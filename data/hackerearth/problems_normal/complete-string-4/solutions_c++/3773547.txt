#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	string str;
    	bool flag =true;
    	cin >> str;
    	vector <bool>v(26,false);
    	for (int  i = 0 ;i <str.length () ;i++)
    		v[str[i]-'a']=true;
    	for(int i = 0;i< 26;i++)
    		if(!v[i])
    			flag = false;
		if(flag)
			cout <<"YES\n";
		else
			cout<<"NO\n";
    }
    return 0;
}
