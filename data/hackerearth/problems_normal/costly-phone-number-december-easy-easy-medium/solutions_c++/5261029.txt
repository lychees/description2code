#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	for(int z=0;z<t;z++)
	{
		int a[10];
		for(int i=0;i<10;i++)
		{
			cin>>a[i];
		}
		int n,bo;
		cin>>n;
		while(1)
		{
		bo=0;	
		for(int i=1;i<10;i++)
		{
			for(int j=i;j<10;j++)
			{
			if(a[i]+a[j]<a[(i+j)%10])
			{
				a[(i+j)%10]=a[i]+a[j];
				bo=1;
			}
			}
			
		}
		if(bo==0)
		break;
		}
		char c[n];
		cin>>c;
		int i=0,sum=0;
		while(c[i]!='\0')
		{
			int p=(int)c[i]-48;
			sum+=a[p];
			i++;
		}
		cout<<sum<<endl;
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
