#include <iostream>
using namespace std;

int binarySearch(long int target, long int arr[], int start,int end)
{
	int mid = (start+end)/2;
	if(arr[mid] >= target && arr[mid-1] < target)
	return mid;
	else if(start <= end)
	{
     if (arr[mid] <= target)
     return binarySearch(target,arr,mid+1,end);
     else if(arr[mid] > target)
     return binarySearch(target,arr,start,mid-1);
	}
}

int main()
{
    long int n,q;
    long int x,sum[100001]={0};
    cin>>n;
    int i = 1;
    while(n--)
    {
    	cin>>x;
    	sum[i] = sum[i-1] + x;
    	i++;
    }
    cin>>q;
    while(q--)
    {
    	cin>>x;
    	if(x > sum[i-1])
    	cout<<"-1"<<endl;
    	else
    	{
    		cout<<binarySearch(x,sum,1,i-1)<<endl;
    	}
    }
    return 0;
}
