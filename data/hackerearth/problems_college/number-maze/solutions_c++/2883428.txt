#include<iostream>
#include<vector>
using namespace std;
int main()
{
    long long int a, b;
    int i;
    vector<long long int> v;
    cin>>a>>b;
    for(i=a+1;i<b;i++)
    {
    	if((i%5!=0 && i%3!=0) || i%15==0 && i!=0)
    		v.push_back(i);
    }

    for(i=0;i<v.size();i++)
    {
        cout<<v[i];
        if(i!=v.size()-1)
            cout<<",";
    }
return 0;
}
