#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int x1,y1,x2,y2;
    	cin>>x1>>y1>>x2>>y2;
    	int x11,y11,x22,y22;
    	cin>>x11>>y11>>x22>>y22;
    	if((x11>x2) || (x22<x1))
    		cout<<"Apart"<<endl;
    	else if((y11>y2) ||(y1>y22))
    		cout<<"Apart"<<endl;
    	else if(x1==x22 && (y1==y22 || y2==y11))
    	{
    		cout<<"Point"<<endl;
    	}
    	else if(x11==x2 && (y1==y22 || y2==y11))
    	{
    		cout<<"Point"<<endl;
    	}
    	else if(x1==x22 && ((y1<=y22 && y1>=y11)||(y2<=y22 && y2>=y11)))
    		cout<<"Line"<<endl;
    	else if(x1==x22 && ((y22<=y2 && y22>=y1)||(y11<=y2 && y11>=y1)))
    		cout<<"Line"<<endl;
    	else if(x2==x11 && ((y11<=y2 && y11>=y1)||(y22<=y2 && y22>=y1)))
    		cout<<"Line"<<endl;
    	else if(x2==x11 && ((y1<=y22 && y1>=y11)||(y2<=y22 && y2>=y11)))
    		cout<<"Line"<<endl;
    	else if(y2==y11 && ((x22>=x1 && x22<=x2)||(x11>=x1 && x11<=x2)))
    		cout<<"Line"<<endl;
    	else if(y2==y11 && ((x2>=x11 && x2<=x22)||(x1>=x11 && x1<=x22)))
    		cout<<"Line"<<endl;
    	else if(y22==y1 && ((x22>=x1 && x22<=x2)||(x2>=x11 && x2<=x22)))
    		cout<<"Line"<<endl;
    	else if(y22==y1 && ((x11>=x1 && x11<=x2)||(x1>=x11 && x1<=x22)))
    		cout<<"Line"<<endl;
    	else
    		cout<<"Rectangle"<<endl;
    }
    return 0;
}
