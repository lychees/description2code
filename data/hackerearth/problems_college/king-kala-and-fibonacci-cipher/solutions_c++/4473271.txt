typedef long long int ll;

bool compare(pair<ll,ll> i, pair<ll,ll> j)
{
    return i.first>j.first || (i.first==j.first && i.second<j.second);
}
int main()

		if ( i <= 1 )
		 next = i;
		else
		{
		 next = first + second;
		 first = second;
		 second = next%26;
		}
		stri[i]=(char)(next%26+97);
		
    	
    }
   
    while(n--)

    	}
    
    	for(i=l;i<=r;i++)
    	{
    		s[stri[i]-97].first++;
    	}
    	
    	sort(s.begin(),s.end(),compare);
