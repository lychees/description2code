#include <iostream>
using namespace std;
#include <string>
#include <bits/stdc++.h>
#define ll long long
#include <cmath>
#include <cstring>
int main()
{
    int test,i,count,c;
    string s3;
    cin>>test;
    assert(0<test<=30);
    while(test--)
    {
    	string s2,s1;
    	cin>>s1>>s2;
    	if(s1.size()==0 && s2.size()==0){cout<<s3<<c<<endl;continue;}
    	count=0;
    	count=(s2.size()-s1.size());
    	count=abs(count);
  //  	cout<<(count)<<endl;
 //   	cout<<s1.size()<<" "<<s2.size()<<endl;
    	if(s2.size()>=s1.size())
    	{
    		for(i=0;i<s1.size();i++)
    			if(s1[i]!=s2[i]){count++;}
    	}
    	else if(s2.size()<s1.size())
    	{
    		for(i=0;i<s2.size();i++)
    			if(s1[i]!=s2[i]){count++;}
    	}
    	if(count==0){cout<<"Exact\n";}
    	else{cout<<"Non Exact "<<count<<endl;s3="Non Exact ";c=count;}
    }
    return 0;
}
