#include<unordered_map>
#include<string>
using namespace std;
int main() {

    cin.tie(0);
    unordered_map<string,int> my;
	register short unsigned int n;
	string name, winner;
	short int point;
	int maxi=0;
	cin>>n;
	while(n--) {
	    cin>>name>>point;
	    my[name]+=point;
	    if(my[name]>maxi) {
	        maxi=my[name];
	        winner=name;
	    }
	}
	cout<<winner<<"\n";

}
