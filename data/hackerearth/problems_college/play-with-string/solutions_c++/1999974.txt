#include <iostream>
using namespace std;
#include<string>
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int i,l;
    	string s,r;
    	cin>>s;
    	l=s.length();
     for(i=0;i<l;i++)
     r[i]=s[l-i-1];
    	for(i=1;s[i];i++)
    	if(abs(s[i]-s[i-1])!=abs(r[i]-r[i-1]))
    	break;
    	if(s[i]=='\0')
    	cout<<"Funny\n";
    	else
    	cout<<"Not Funny\n";
    }
    return 0;
}
