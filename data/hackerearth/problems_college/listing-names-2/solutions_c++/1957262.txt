#include <bits/stdc++.h>
using namespace std;
map<string, int> mp;
vector<string> vs;
int main() {
	// your code goes here
	int t;
	cin>>t;string s;
	
	for(int i=0;i<t;i++)
	{
		cin>>s;mp[s]++;
		if(mp[s]==1)vs.push_back(s);
	}
	sort(vs.begin(),vs.end());
	for(int i=0;i<vs.size();i++)
	{
		cout<<vs[i]<<" "<<mp[vs[i]]<<endl;
	}
	return 0;
}
