#include <iostream>
#include<string>
using namespace std;
int main()
{
    int t;
    string s1,s2;
    cin>>t;
    while(t--)
    {
    
     long long int count=0,i,j,l1,l2;
     long long int ans = 10000000;
     cin>>s1;
     cin>>s2;
     l1=s1.size();
     l2=s2.size();
     
       if(l1<=l2)
	  {
  	        for (i = 0; i<l2; i++)
		   {
		    count = 0;
				for (j = 0; j<l1; j++)
				{
					if (s1[j] != s2[j + i])
						count++;
				}
				ans = min(ans, count);
			}
	    }
     
     else
     {
          for (i = 0; i<l1; i++)
		 {
		  count = 0;
		   for (j = 0; j<l2; j++)
		    {
			 if (s2[j] != s1[j + i])
			   count++;
			 }
			ans = min(ans, count);
		 }
		  
      }
      cout<<ans<<endl;	
      
    }
    return 0;
}
