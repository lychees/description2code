import math


def is_prime(number):

    sq = int(math.sqrt(number))
    for i in range(2, sq):
        if number % i == 0:
            return False

    return True


def frequency(numbers):
    freq = {}

    for number in numbers:
        if number in freq:
            freq[number] += 1
        else:
            freq[number] = 1

    return freq


def main(numbers):

    freq = frequency(numbers)
    max_freq = 0
    freq_num = 0

    for number in numbers:
        if is_prime(number) and number > freq_num:
            max_freq = freq[number]
            freq_num = number

    return freq_num, max_freq


if __name__ == '__main__':

    num = int(raw_input())
    numbers = map(int, raw_input().split())

    freq_num, max_freq = main(numbers)
