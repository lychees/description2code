#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	for(int i=0;i<t;i++){
		int n;
		cin >> n;
		int a[n] ,sum=0;
		for(int j=0;j<n;j++){
			cin >> a[j];
			sum+=a[j];
		}
		int b[sum/2+1];
		
		
		
		int x=10000000,v,half,sum1;
		for(int ans =0;ans<n;ans++){
		sum1 = sum - a[ans];
		for(int k=0;k<sum1/2+1;k++)
			b[k]=0;
		b[0]=1;
		for(int j=0;j<n;j++){
			if(j!=ans){
			for(int k=sum1/2;k>=a[j];k--){
				if(b[k-a[j]])
					b[k]=1;
			}}
		}
		half=sum1/2;
		for (; !b [half]; half--);
		if(sum1-2*half==x && a[v]>a[ans])
			v=ans;
		if(sum1-2*half<x){
			x=sum1-2*half;
			v=ans;
		}
	//	cout<< "a" <<sum1-2*half;
		}
		cout << x << " " << a[v] << "\n";
	}
}
