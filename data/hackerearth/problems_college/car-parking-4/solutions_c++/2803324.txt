#include <cstdio>
#include <iostream>

using namespace std;

char input[101010];

bool isSafe(int i)
{
	if(input[i] != '-') 
		return false;

	if( input[i+1] != '\0' && input[i+1] == 'B'  || input[i+2] != '\0' && input[i+2] == 'B' ) 
		return false;

	if( input[i+1] == 'S' )
		return false;
	if ( i && input[i-1] == 'S')
		return false;

	return true;
}

int Process()
{
	int ans = 0;
	for(int i = 0; input[i] != '\0'; ++i)
	{
		if(isSafe(i))
		++ans;
	}
	return ans;
}

int main()
{
	int T;
	cin >> T;
	while(T--)
	{
		cin >> input;
		printf("%d\n",Process());
	}
	return 0;
}
