#include <iostream>
#include <string.h>
#define ll long long

using namespace std;

ll dp[31][31],N;

ll solve(int d, int i)
{
  if(d==0 && i>=0) return 1;
  else if(dp[d][i]!=-1) return dp[d][i];
  else
  {
    dp[d][i]=0;
    if(d-1>=0)
    dp[d][i] = solve(d-1, i+1);
    if(i-1>=0)
    dp[d][i] += solve(d,i-1);
  }
  return dp[d][i];
}

int main()
{
    int t;
    memset(dp,-1,sizeof(dp));
    cin>>t;
    while(t--)
    {
      cin>>N;
      cout<<solve(N,0)<<endl;
    }
}
