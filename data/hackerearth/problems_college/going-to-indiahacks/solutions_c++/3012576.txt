#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007

long long dp[1010][1010];

int main()
{
    dp[0][0] = 1;
    
    for(int i=1; i<=1005; i++)
    	dp[i][0] = dp[0][i] = 1;
    
    for(int i=1; i<=1005; i++)
    {
    	for(int j=1; j<=1005; j++)
    	{
    		dp[i][j] = dp[i-1][j] + dp[i][j-1] + dp[i-1][j-1];
    		
    		if(dp[i][j]>=MOD)
    			dp[i][j]%= MOD;
    	}
    }
    
    int T;
    scanf("%d", &T);
    
    while(T--)
    {
    	int a, b, c, d;
    	scanf("%d %d %d %d", &a, &b, &c, &d);
    	
    	if(a>c || b>d)
    		printf("0\n");
    	else
    		printf("%lld\n", dp[c-a][d-b]);
    }
    
    return 0;
}
