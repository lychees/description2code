#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
   long long int t,n,m,p,q,r,sum;
    cin >> t;
    while(t--)
     {
        cin >> n >> m;
        p=m-1;
        q=n/m;
        r=n%m;
        sum=q*(p*(p+1)/2)+r*(r+1)/2;
        cout << sum << endl;
        
        
    }
    return 0;
}
