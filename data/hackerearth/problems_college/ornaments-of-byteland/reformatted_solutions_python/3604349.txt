def zerolistmaker(n):
    listofzeros = [0] * n
    return listofzeros


n = input()
l = zerolistmaker(101)
m = n
while n:

    x = raw_input()
    a = set(x)

    for i in a:
        l[ord(str(i))-96] += 1

    n = n-1
g = 0
for k in l:
    if k == m:
        g = g+1

if g > 0:
    print g
else:
    print "0"
