#include <iostream>
#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
int n,k;
string s;
char cub[12][6];
long long dp[12][(1<<10)+10];
int noofbits(int a)
{
    int ct=0;
    while(a>0)
    {
        a=a&(a-1);
        ct++;
    }
    return ct;
}
long long rec(int idx,int bits)
{
    if(idx>n)
    {
        if(noofbits(bits)==k)
        {
            return 1;
        }
        else
            return 0;
    }
    else if(dp[idx][bits]!=-1)
    {
        return dp[idx][bits];
    }
    else
    {
        dp[idx][bits]=rec(idx+1,bits)%mod;
        for(int i=0;i<6;i++)
        {
            for(int j=0;j<k;j++)
            {
                if((bits&(1<<j))==0)
                {
                    if(cub[idx][i]==s[j])
                    {
                        dp[idx][bits]=(dp[idx][bits]%mod+rec(idx+1,bits|(1<<j))%mod)%mod;
                    }
                }
            }
        }
        return dp[idx][bits];
    }
}
int main()
{
    int i,j;
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cin>>n>>k;
    memset(dp,-1,sizeof(dp));
    for(i=0;i<n;i++)
    {
        for(j=0;j<6;j++)
        {
            cin>>cub[i][j];
        }
    }
    cin>>s;
    cout<<rec(0,0);
    return 0;
}
