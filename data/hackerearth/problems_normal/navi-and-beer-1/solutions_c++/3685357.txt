#include <iostream>
#include <math.h>
#include <stdio.h>
using namespace std;

int main()
{
    int t,n,k,i;
    long long m,rest=0,sum=0;
    cin>>t;
    for(i=0;i<t;i++){
    	cin>>n>>k>>m;
    	sum=((n*(n+1))/2);
    	sum=k*sum*sum;
    	rest=sum-m;
    	if(rest>0)
    		cout<<rest<<endl;
    	else
    		 cout<<0<<endl;
    	sum-0;
    	rest=0;
    }
    
    return 0;
}
