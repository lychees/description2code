#include <iostream>
using namespace std;
typedef long long LL;
const LL MAXI=1000000000000001;

int main()
{
	LL N,i,ans,f[100];
	f[0]=1;
	f[1]=2;
	i=1;
	while(f[i]<MAXI){
		i++;
		f[i]=f[i-1]+f[i-2];
	}
	f[i+1]=f[i]+f[i-1];
	int T;
	cin >> T;
	while(T--){
		cin >> N;
		for(i=0;;i++){
			ans=N+i;
			if(f[i]>ans){
				break;
			}
		}
		cout <<ans<<endl;
	}
	return 0;
}
