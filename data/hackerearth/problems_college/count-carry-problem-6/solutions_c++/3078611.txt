#include <iostream>
using namespace std;

int main()
{
    long long i,j,k,l,a,b,m,n,sum,car,T,t;
    cin>>T;
    while(T--){
    	cin>>a>>b;
    	car=0;
    	k=0;
    	while(a!=0 && b!=0){
    		sum=a%10+b%10+car;
    		if(sum>=10){
    			k++;
    			car=sum%10;
    		}
    		else car=0;
    		a/=10;
    		b/=10;
    	}
    	if(b!=0) a=b;
    	while(a!=0){
    		sum=a%10+car;
    		if(sum>=10){
    			k++;
    			car=sum%10;
    		}
    		else car=0;
    		a/=10;
    	}
    	
    	if(k==0) cout<<"No carry operation\n";
    	else if(k==1) cout<<"1 carry operation\n";
    	else{
    		cout<<k<<" carry operations\n";
    	}
    }
    return 0;
}
