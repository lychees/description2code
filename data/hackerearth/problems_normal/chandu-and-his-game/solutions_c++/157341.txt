#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

typedef unsigned long long ull;

int main()
{
    int n, k;
    cin >> n >> k;
    vector<int> x(k), y(k);
    for(int i=0; i<k; ++i) cin >> x[i];
    for(int i=0; i<k; ++i) cin >> y[i];
    
    ull cost = 0;
    for(int i=0; i<k; ++i) {
    	int l = x[i]-1;
    	int r = n-x[i];
    	int t = y[i]-1;
    	int b = n-y[i];
    	
    	cost += min(l,r) + min(t,b);
    }
    
    cout << cost << endl;
}
