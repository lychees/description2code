#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	long long int a, b,s;
    	cin >> a >> b;
    	s = a^b;
    	int count = 0;
    	while(s)
    	{
    		s &= (s-1);
    		count++;
    	}
    	if(count%2==0)
    	  cout << "YES" << endl;
    	else
    	 cout << "NO" << endl;
    }
    return 0;
}
