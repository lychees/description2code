#include <iostream>
using namespace std;

int main()
{
    long long n,i,j,k;
    cin>>n;
    for(i=0;i<n;i++)
    	cout<<"*";
    	cout<<endl;
    for(i=1;i<n-1;i++)
    {
    	for(j=n-1;j>=i;j--)
    	{
    		if(i==j)
    		{
    			cout<<"*";
    		}
    		else
    		{
    			cout<<" ";
    		}
    	}
    	cout<<endl;
    }
    for(i=0;i<n;i++)
    {
    	cout<<"*";
    }
    cout<<endl;
    return 0;
}
