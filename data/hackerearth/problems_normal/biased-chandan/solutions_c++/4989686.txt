#define ll long long

using namespace std;

int main()
{
	stack <ll> S; //STL (Standard Template Library)
	ll N;
	cin>>N;
	
	ll arr[N];
	for(int i=0;i<N;i++)cin>>arr[i];
	
	for(int i=0;i<N;i++)
	{
		if(arr[i] != 0)
		S.push(arr[i]);
		
		else if(arr[i] == 0)
		{
			if(S.empty() == false)
			S.pop();
		}
	}
	
	ll sm = 0;
	while(S.empty() == false)
	{
		sm += S.top();
		S.pop();
	}
	
	cout<<sm<<endl;

   return 0;
}
