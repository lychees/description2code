template<class T>
void in(T &x)
{
    register T c = gc();
    x = 0;
    T neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
 
bool search(int n)
{
	string s="";
	while(n)
	{
		int t=n%10;
		n/=10;
		if(t==0)
			s+='0';
		else
			s+='1'+t-1;
	}
	reverse(s.begin(),s.end());
	for(int i=0;i<s.length()-1;i++)
		if(s[i]=='2' && s[i+1]=='1')
			return true;
	return false;
}

int main()
{
    
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	if(n%21==0 || search(n))

    }
    
    return 0;
}
