#include<iostream>
using namespace std;
int flag;
void func(int a[],int i,int j,int sum)
{
	if(sum==0)
		flag=1;
	if(i>j)
		return ;
	func(a,i+1,j,sum-a[i]);
	func(a,i+1,j,sum);
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		flag=0;
		int n;
		cin>>n;
		int a[n],i,s;
		for(i=0;i<n;i++)
			cin>>a[i];
		cin>>s;
		func(a,0,n-1,s);
		if(flag==1)
			cout<<"YES\n";
		else
			cout<<"NO\n";   
	}
	return 0;
}
