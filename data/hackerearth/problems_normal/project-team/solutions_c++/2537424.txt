#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,max,min;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int array[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>array[i];
    	}
    	
    	int x=0,y=n-1;
    	
    	sort(array,array+n);
    	
    	max=array[x]+array[y];
    	min=array[x]+array[y];
    	
    	while(x<y)
    	{
    	  if((array[x]+array[y])>max)
    	    max=array[x]+array[y];
    	  else if(array[x]+array[y]<min)
    	    min=array[x]+array[y];
    	  x++;
    	  y--;
    	}
    	
    	cout<<max-min<<endl;
    	
    }
    return 0;
}
