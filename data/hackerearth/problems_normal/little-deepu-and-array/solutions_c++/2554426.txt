#include <bits/stdc++.h>
using namespace std;

int main()
{
	int N;
	cin>> N;
	assert(N>= 1 && N <= 100000);
	int i , j;
	int V[N];
	for(i= 0; i < N ; i++)
	cin >> V[i] ;
	int M;
	cin >> M;
	int X[M];
	for(i = 0; i < M ; i++)
	cin >> X[i];
	for(i = 0 ; i < M ; i++)
	for(j = 0 ; j < N ; j++)
	{
		if(V[j] > X[i])
		V[j] = V[j] - 1;
	}
	for(j = 0 ; j < N; j++)
	cout << V[j] << " ";
     
    return 0;
}
