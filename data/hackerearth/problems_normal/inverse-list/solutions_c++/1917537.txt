#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    int n;
    vector <int> a;
    vector <int> b;
    while(t--)
    {
    	int flag=0;
    	cin>>n;
    	a.resize(n+1);
    	b.resize(n+1);
    	for(int i=1;i<=n;i++)
    	{
    		cin>>a[i];
    	}
    	for(int i=1;i<=n;i++)
    	{
    		b[a[i]]=i;
    	}
    	for(int i=1;i<=n;i++)
    	{
    		if(a[i]!=b[i])
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==0)
    	{
    		cout<<"inverse"<<endl;
    	}
    	else
    	{
    		cout<<"not inverse"<<endl;
    	}
    }
}
