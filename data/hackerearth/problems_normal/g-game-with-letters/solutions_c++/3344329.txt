#include<bits/stdc++.h>
using namespace std;
int main()
{
	std::ios::sync_with_stdio(false);
	int t,n,i,j;
	string a,b;
	int game[50];
	cin>>t;
	while(t--)
	{
		cin>>n;
		cin>>a;
		memset(game,0,sizeof(game));
		for(i=0;i<a.length();i++)
		{
			game[a[i]-'a']++;
		}
		int ans[n+1];
		int maxi,maxs,max=INT_MIN;
		for(i=1;i<=n;i++)
		{
			ans[i]=0;
			cin>>b;
			for(j=0;j<b.length();j++)
			{
				ans[i]+=game[b[j]-'a'];
			}
			if(ans[i]>max)
			{
				max=ans[i];
				maxi=i;
				maxs=b.length();
			}
			else if(ans[i]==max)
			{
				if(maxs>b.length())
				{
					maxs=b.length();
					maxi=i;
				}
			}
		}
		cout<<maxi<<endl;
	}
}
