#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,ans1;
	long long int n,ans;
    scanf("%d",&t);
    while(t--)
    {
    scanf("%lld",&n);
    if(n==1)
    cout<<1<<endl;
    else if(n==2)
    cout<<2<<endl;
    else if(n==3)
    cout<<3<<endl;
    else
    {
    long long int p=1;
    ans=n;
    ans1=0;
    while(ans>(2*p))
    {
    	ans=ans-2*p;
    	ans1+=2;
    	p++;
    }
    if(ans!=0)
    {
    if(ans<=p)
    	ans1++;
    else
        ans1+=2;
    }
    cout<<ans1<<endl;
    }
    }
    return 0;
}
