int findMin(int arr[],int i,int n)
{
	int min=INT_MAX; 
	for(;i<=n;i++)	
		if(arr[i]!=0 && arr[i]<min) min=arr[i];
	return min;
}

int findZero(int arr[],int i,int n)
{
	int tot=0;
	for(;i<=n;i++) 	if(arr[i]!=0) tot++;
	return tot;
}
