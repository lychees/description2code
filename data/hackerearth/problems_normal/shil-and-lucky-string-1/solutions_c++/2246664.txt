#include <bits/stdc++.h>
using namespace std;

int n;
char arr[100005];
int ans = 0;


int calc(vector <int> A, vector <int> B) {
	int res = A[25];
	A[0] += A[25];
	A[25] = 0;
	int idx = 1;
	for(int i = 0; i < 25; i++) {
		res += B[i];
		B[25] += B[i];
		B[i] = 0;
		idx = i + 1;
		while(A[i] > 0) {
			while(B[idx] == 0) idx++;
			int x = min(A[i], B[idx]);
			A[i] -= x;
			B[idx] -= x;
		}
	}
	return res;
}

int main() {
	scanf("%d", &n);
	for(int i = 1; i <= n; i++)
		scanf(" %c", &arr[i]);
	vector <int> cnt1(26, 0);
	vector <int> cnt2(26, 0);
	for(int i = 1; i <= n / 2; i++)
		cnt1[arr[i] - 'a']++;
	for(int i = n / 2 + 1; i <= n; i++)
		cnt2[arr[i] - 'a']++;
	
	
	int cnt = 0;
	for(int i = 0; i < 26; i++)
		cnt += abs(cnt1[i] - cnt2[i]);
	
	cnt /= 2;
	ans = cnt;
	
	ans = min(ans, calc(cnt1, cnt2));
	ans = min(ans, calc(cnt2, cnt1));
	
	printf("%d\n", ans);

}
