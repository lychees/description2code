import sys
from collections import defaultdict
from operator import itemgetter
marks = defaultdict(list)
for _ in range(input()):
    student, mark = raw_input().split()
    mark = int(mark)
    if marks.has_key(mark):
        marks[mark].append(student)
    else:
        marks[mark] = []
        marks[mark].append(student)
smark = sorted(marks.iteritems(), key=itemgetter(0), reverse=True)
#print smark
for x in smark:
    lst = sorted(x[1])
    for y in lst:
        print  y, x[0]
