T = input()
while T > 0:
    l = raw_input()
    st = ""
    for char in l:
        if char.isalnum() or char == '*':
            st = st+char
    le = len(st)
    for i in range(0, le/2):
        if (st[i] == '*' or st[le-1-i] == '*'):
            if (st[i] == '*'):
                print st[le-1-i]
            else:
                print st[i]
    T = T-1
