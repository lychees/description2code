const int MAXN = 70;

ll fib[MAXN];

void gen(ll k){
    memset(fib, 0, sizeof(fib));
    fib[1] = 1;
    fib[2] = 1;
    for(ll i =  3;i<MAXN;i+=1)
    {
        for(ll j = i-1;j>i-1-k&&j>0;j--)
        {
            fib[i]+=fib[j];
        }
    }
}



int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    ll n,k;
    cin>>n>>k;
    gen(k);
    long long g = __gcd((1L << n) - fib[n+2], (long long)(1L<< n));
    cout << ((1L << n) - fib[n+2])/g << "/" << (1L<<n)/g<< endl;

}
