#include <iostream>
#include <string>

using namespace std;

int min(int a, int b)
{
	return a<=b?a:b;
}


int matched_characters(string word1, string word2)
{
	int size1 = word1.size();
	int size2 = word2.size();
	
	int min_size = min(size1, size2);
	
	int count = 0;
	for(int i=0;i<min_size;i++)
	{
		if(word1[i] == word2[i])
		{
			count++;
		}
	}
	return count;
}
int main()
{
	int T;
    cin>>T;
    
    while(T--)
    {
    	string A,B;
    	cin>>A;
    	cin>>B;
    	
    	cout<<matched_characters(A,B)<<endl;
    }
    return 0;
}
