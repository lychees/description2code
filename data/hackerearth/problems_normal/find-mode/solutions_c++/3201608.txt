int main()
{
      int t;
      cin>>t;
      while(t--)
      {
                 int n;
                 scanf("%d",&n);
                 int k[n];
                 int mx=0;
                 int mx1=0;
                 vector<int>vc;
                 map<int,int>mp;
              const unsigned int M=1000001;
              bitset<M>str;
                 for(int i=0;i<n;i++)
                 {
                    scanf("%d",&k[i]);
                 // mx1=max(mx1,k[i]);
                     mp[(k[i])]++;
                     mx=max(mx,mp[(k[i])]);
                 }
              // cout<<mx<<endl;

                            if((mp[(k[i])]==mx)&&str[(unsigned int )k[i]]==0)

      }

}
