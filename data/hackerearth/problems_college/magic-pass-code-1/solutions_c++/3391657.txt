#include <bits/stdc++.h>
using namespace std;
typedef long long int L;
typedef unsigned long long int ll;

int getSum(int n){
	int t=0;
	while(n!=0){
		t+=(n%10);
		n/=10;
	}
	return t;
}

int main(){
	int n; int maxSum=0, maxNum=0;
	while(cin>>n){
		int tmp=getSum(n);
		if(tmp>maxSum){
			maxSum=tmp;
			maxNum=n;
		}
	}
	cout<<maxNum<<endl;
}
