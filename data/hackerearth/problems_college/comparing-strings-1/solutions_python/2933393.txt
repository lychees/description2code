t = int(raw_input())
while t>0:
    t-=1
    s1 = raw_input()
    s2 = raw_input()
    len = s1.__len__()
    if len>s2.__len__():
        len = s2.__len__()
        s1,s2 = s2,s1
    flag = 0
    for i in s1:
        if s2.__contains__(i):
            flag = 1
            break
    if flag==1:
        print("YES")
    else:
        print("NO")
