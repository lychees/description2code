#include<iostream>
#include<algorithm>
#include<vector>
#include<string>
#include<cstring>
#include<memory>
#include<iterator>
#include<math.h>
#include<iso646.h>

using namespace std;

int main()
{
    long int t,n,i,j;
    cin>>t;while(t--)
    {
        cin>>n;
        vector<long int> a(n+1);
        for(long int k=1;k<=n;k++)
            cin>>a[k];
        sort(a.begin(),a.end());
        cin>>i>>j;
        cout<<(a[n+1-i]+a[j])<<endl;
    }
return 0;
}
