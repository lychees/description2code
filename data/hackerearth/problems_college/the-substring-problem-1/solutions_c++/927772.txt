#include <iostream>
using namespace std;

int main()
{
    int test;
    cin>>test;
    char str[101];
    int count[26],i,temp,flag[26],max,k;
    int n;
    while(test)
    { 
    	for(i=0;i<26;i++)
    	{
    		count[i]=0;
    	}
    	cin>>n;
    	while(n)
       {
       for(i=0;i<26;i++)
       {
       	flag[i]=0;
       }
       cin>>str;
       
       i=0;
       	while(str[i]!='\0')
       {
       	
       	
       	
       	temp=(int)str[i];
       //	printf("%d \n",temp);
       	if(flag[temp-97]==0)
       	{
       		count[temp-97]++;
       		flag[temp-97]=1;
       	}
       	i++;
       }
       		n--;       	
       }
       		 k=0;
       		max=count[0];
       		for(i=1;i<26;i++)
       		{
       		  if(count[i]>max)
       		  {
       		  	max=count[i];
       		  	k=i;
       		  }
       			}
       	if(max)
       	{
       		k=k+97;
       		printf("%c\n",k);}
       test--;
       }    	
   
   
    	
     
    	
    
    return 0;
}
