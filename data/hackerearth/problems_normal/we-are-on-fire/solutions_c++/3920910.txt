using namespace std;
 
int n,m,total; 
int a[maxN+5][maxN+5];
pair<int,int> parent[maxN+5][maxN+5];
int c[maxN+5][maxN+5];

int ri[]={0,0,1,-1};
int cj[]={1,-1,0,0};

void bfs(int si,int sj)
{
	int k,ui,uj,vi,vj;
	queue<pair<int,int> > q;
	q.push(make_pair(si,sj));
	parent[si][sj]=make_pair(si,sj);
	c[si][sj]=0;
	while(!q.empty())
	{
		ui=(q.front()).first;
		uj=(q.front()).second;
		q.pop();
		c[si][sj]++;
		for(k=0;k<4;k++)
		{
			vi=ui+ri[k];
			vj=uj+cj[k];
			if(vi>=0&&vi<n&&vj>=0&&vj<m&&a[vi][vj]==1&&parent[vi][vj].first==-1)
			{
				parent[vi][vj]=make_pair(si,sj);
				q.push(make_pair(vi,vj));
			}
		}
	}
	total+=c[si][sj];
}

int main()
{
    #ifndef ONLINE_JUDGE
    	//freopen("in.txt","r",stdin);

    #endif
	int i,j,q,x,y,si,sj;
	scanf("%d%d%d",&n,&m,&q);
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
			scanf("%d",&a[i][j]);
	for(i=0;i<=n;i++)
		for(j=0;j<=m;j++)
		{
			parent[i][j]=make_pair(-1,-1);
			c[i][j]=0;
		}
	total=0;
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			if(a[i][j]==1&&parent[i][j].first==-1)
				bfs(i,j);
		}
	}
	while(q--)
	{
		scanf("%d%d",&x,&y);
		si=parent[x-1][y-1].first;
		sj=parent[x-1][y-1].second;
		if(si==-1)
			printf("%d\n",total);
		else
		{
			if(a[si][sj]==1)
			{
				a[si][sj]=0;
				total-=c[si][sj];
			}
			printf("%d\n",total);
		}
	}
    return 0;
}
