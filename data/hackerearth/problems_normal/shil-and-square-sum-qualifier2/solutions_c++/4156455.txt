#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define sci(fd) scanf("%d",&fd)
#define scll(fd) scanf("%lld",&fd)
#define pb push_back
#define mp make_pair
#define MOD 1000000007
#define PI 3.1415926535897932
#define pii pair < int,int > 
#define fi first
#define se second
const ll infi=1000000000000000009;
ll ar[1000009],cumsum[1000009];
int main()
{
    ll i,j,n,k;
    scll(n);
    scll(k);
    for(i=1;i<=n;i++)
    {
        scll(ar[i]);
        cumsum[i]=(cumsum[i-1]+ar[i])%MOD;
    }
    ll ans=0,usity=0;
    for(i=1;i<=k;i++)
    {
        ans=(ans+(((i*i)%MOD)*ar[i])%MOD)%MOD;
        usity=(usity+((2*i-1)*ar[i])%MOD)%MOD;
    }
    printf("%lld ",ans);
    for(i=k+1;i<=n;i++)
    {
        ans=(ans+(((k*k)%MOD)*ar[i])%MOD)%MOD;
        ans=(ans-usity+MOD)%MOD;
        printf("%lld ",ans);
        usity=(usity+((2*k-1)*ar[i])%MOD)%MOD;
        usity=(usity-2*(cumsum[i-1]-cumsum[i-k])+MOD+MOD+MOD)%MOD;
        usity=(usity-ar[i-k]+MOD)%MOD;
    }
    return 0;
}
