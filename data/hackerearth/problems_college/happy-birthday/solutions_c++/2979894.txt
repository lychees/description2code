#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int n;
	cin>>n;
	ll int a[5];
	memset(a,0,sizeof(a));
	ll int i,j,t,ans=0;
	for(i=0;i<n;i++)
	{
		cin>>j;
		a[j]++;
	}
	t=min(a[1],a[3]);
	a[1]-=t;a[3]-=t;
	ans+=t;
	ans+=a[3];a[3]=0;
	t=a[2]/2;ans+=t;a[2]-=2*t;
	t=min(a[2],2*a[1]);ans+=t;a[1]-=t;a[2]-=t;
	ans+=a[1]/4;a[1]=a[1]%4;
	if(a[1]|a[2])ans++;
	ans+=a[4];
	cout<<ans<<endl;
	return 0;
}
