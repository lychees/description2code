#include<bits/stdc++.h>
using namespace std;
int dp[10000001];
int main()
{
	int t,n,i;
	cin>>t;
	dp[6]=1;
	dp[9]=1;
	dp[20]=1;
	for(i=7;i<10000001;i++)
	{
		if(i>6 && dp[i-6]==1)
		dp[i]=1;
		else if(i>9 && dp[i-9]==1)
		dp[i]=1;
		else if(i>20 && dp[i-20]==1)
		dp[i]=1;
		else 
		dp[i]=0;
	}
	while(t--)
	{
		cin>>n;
		if(dp[n]==1)
		cout<<"True\n";
		else
		cout<<"False\n";
		
	}
}
