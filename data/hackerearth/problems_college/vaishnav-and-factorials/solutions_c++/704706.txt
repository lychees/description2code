#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long t,n,i,j,k,c,r;
    long long a[100000];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	memset(a,0,sizeof(a));
    	a[0]=1;
    	k=0;
    	for(i=2;i<=n;i++)
    	{
    		r=0;
    		for(j=0;j<=k;j++)
    		{
    			a[j]=a[j]*i+r;
    			r=a[j]/10;
    			a[j]=a[j]%10;
    		}
    		while(r)
    		{
    			k++;
    			a[k]=r%10;
    			r=r/10;
    		}
    	}
    	c=0;
    	for(i=0;i<=k;i++)
    	{
    		if(a[i]==4||a[i]==7)
    			c++;
    	}
    	cout<<c<<endl;
    }
    return 0;
}
