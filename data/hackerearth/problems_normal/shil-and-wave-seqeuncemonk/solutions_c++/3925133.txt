typedef long long ll;
typedef long double ld;

typedef pair<int, int> pii;
typedef pair<int, pii> piii;
typedef pair<ll, ll> pll;
typedef vector<int> vi;

const int INF = 0x3f3f3f3f, MOD = 1e9+7;
const int N = 1e5+1;

ll BIT[2][N];
ll n, aux;
ll ans;

void add(int bit, ll pos, ll val) {
  for(;pos<N;pos+=pos&-pos) BIT[bit][pos]=(BIT[bit][pos]+val+MOD)%MOD;
}

ll query(int bit, ll pos) {
  ll sum=0;
  for(;pos>0;pos-=pos&-pos) sum=(sum+BIT[bit][pos])%MOD;
  return sum;
}

int main() {
  scanf("%lld", &n);
  for(int i=0; i<n; i++) {
    scanf("%lld", &aux);
    ll q1 = query(0,aux), q2 = query(1,aux);
    ans=(ans+q1+q2)%MOD;
    add(1,1,q1+1);
    add(1,aux,-q1-1);
    add(0,aux+1,q2+1);
  }
  printf("%lld\n", ans);
  return 0;
}
