#include<bits/stdc++.h>
using namespace std;
int n;
int k;
int dp[1000005]={NULL};
int rec(int i,int r);
int main()
{
int test;
scanf("%d",&test);
while(test--)
{
memset(dp,NULL,sizeof(dp));
scanf("%d",&n);
k=rec(1,n);
printf("%d\n",k);
}
return 0;
}
int rec(int i,int r)
{
if(r==0)
    return 0;
if(r<0)
    return 1000000;
if(dp[r]!=NULL)
    return dp[r];
dp[r]=1+min(rec(i,r-i),rec(i+1,r-i-1));
return dp[r];
}
