#include <iostream>
#include<bits/stdc++.h>
using namespace std;
long long DP[270][12][12];
#define MOD 4294967143
int n;
long long solve(int pos,int prev,int last){
	if(pos==0 && prev==last) return 1;
	if(pos==0 || prev<0 || prev>10) return 0;
	if(DP[pos][prev][last]!=-1) return DP[pos][prev][last];
	long long ret=0;
	ret+=solve(pos-1,prev-1,last);
	ret+=solve(pos-1,prev+1,last);
	ret%=MOD;
	DP[pos][prev][last]=ret;
	return ret;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		long long ans=0;
		memset(DP,-1,sizeof(DP));
		for(int i=1 ; i<=10 ; i++){
			if(i-1>=0)
			ans+=solve(n-1,i,i-1);
			if(i+1<=10)
			ans+=solve(n-1,i,i+1);
			ans%=MOD;
		}
		printf("%lld\n",ans);
	}
    return 0;
}
