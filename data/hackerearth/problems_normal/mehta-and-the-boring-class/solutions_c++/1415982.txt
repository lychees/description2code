#include <bits/stdc++.h>
using namespace std;
#define maxn 1000000+10
long long dp[maxn];
bool primes [maxn];
list<int> pn;
list<int>::iterator it;


void precompute() {
	int k=0;
	dp[0] = 0;
	dp[1] = 0;
	
	
	for(int i=2;i<maxn;i++) {
		if(dp[i] > 0) {
				continue;
		}
		for(int j=i;j<maxn;j+=i) {
			int k = j;
			while(k%i == 0) {
				dp[j]++;
				k = k/i;
			}
		}
	}
	
	for(int i=1;i<maxn;i++) {
		dp[i] += dp[i-1];
	}
}
int main()
{
	int t;
	int a, b;
	scanf("%d",&t);
	
	precompute();

	while(t--) {
		scanf("%d %d",&a, &b);
		printf("%lld\n",dp[b]-dp[a-1]);
	}
}
