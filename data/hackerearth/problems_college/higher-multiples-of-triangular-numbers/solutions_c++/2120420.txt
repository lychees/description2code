#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int t,n;
    
    scanf("%lld",&t);
    
    long long int a[1001];
    
    for(int i=1;i<=1000;i++)
    {
    	a[i]=(i*(i+1))/2;
    	//printf("%lld\n",a[i]);
    }
    
    for(;t>0;t--)
    {
    	scanf("%lld",&n);
    	
    	for(int i=1;i++;)
    	{
    		int c=0;
    		//printf("%lld\n",a[i]);
    		for(int j=1;j<=a[i];j++)
    		{
    			//printf("%lld\n",j);
    			if(a[i]%j==0)
    			{
    			//	printf("%lld\n",j);
    				c++;
    				if(c>n)
    				break;
    			}
    		}
    		if(c>n)
    		{
    			printf("%lld\n",a[i]);
    			break;
    		}
    	}
    }
    return 0;
}
