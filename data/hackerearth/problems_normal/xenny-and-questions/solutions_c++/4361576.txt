#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll a[100100];
void merge(ll s, ll e){
	ll mid=(s+e)>>1;
	ll j, i;
	ll l[100100], r[100100];
	for(j=0, i=s; i<=mid; i++, j++)
		l[j]=a[i];
	ll sz1=j;
	for(j=0, i=mid+1; i<=e; i++, j++)
		r[j]=a[i];
	ll sz2=j, k=s;
	i=0,j=0;
	while(i<sz1 && j<sz2){
		if(l[i] < r[j])
			a[k++]=l[i++];
		else
			a[k++]=r[j++];
	}
	while(i<sz1)
		a[k++]=l[i++];
	while(j<sz2)
		a[k++]=r[j++];
}

void mergesort(ll s, ll e){
	if(e == s)return;
	ll mid=(s+e)>>1;
	mergesort(s, mid);
	mergesort(mid+1, e);
	merge(s,e);
}
int main(){
	ios::sync_with_stdio(false);
	ll t;cin>>t;
	while(t--){
		ll n, k;cin>>n>>k;
		
		for(int i=0; i<n; i++)cin>>a[i];
		if(!n){cout<<0<<endl;continue;}
		mergesort(0,n-1);
		ll sum=0;
		for(int i=n-1; i>=max(n-k,0LL); i--)
			sum+=a[i];
		cout<<sum<<endl;
	}
	return 0;
}
