#include <stdio.h>
 
double dp[51][10001]={},answer;
 
int main()
{
    int t,n,m,k,i,j,p;
    scanf("%d",&t);
    while(t--)
    {
    	int count=0;
    	scanf("%d%d%d",&m,&n,&k);
    	
    	for(i=1;i<=m;i++)
    		for(j=1;j<=k;j++)
    		    dp[i][j]=0;
    		    
    	for(i=1;i<=k && i<=n;i++)
    	{
    		dp[1][i]=(double)1/n;
    	}
   		
   		for(i=2;i<=m;i++)
   			for(j=1;j<=k;j++)
   			{
   				for(p=1;p<=n && p<j;p++)
   				{
   					dp[i][j]+=dp[i-1][j-p];
 
   				}
   				dp[i][j]/=n;
   			}
   		
   		answer=dp[m][k];
   		while(answer<1.0 && answer!=0.0)
   		{
   			answer*=10;
   			count++;
   		}
   		printf("%0.3f %d\n",answer,count);
    }
}
