ans=[2]
def chkPrime(n):
    if n==2 or n==3:
        return n
    if n%2==0 or n%3==0:
        return 0
    else:
        i=5
        while i*i<=n:
            if n%i==0 or n%(i+2)==0:
                return 0
            i+=6
    return n
i=3
while i<1050000:
    if str(i)==str(i)[::-1]:
        x=chkPrime(i)
        if x>0:
            ans.append(i)
            i=x
    i+=1

for i in range(input()):
	n=input()
	for j in ans:
		if j>=n:
			print(j)
			break
