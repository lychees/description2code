#include <bits/stdc++.h>
using namespace std;

int main()
{
    int r,k,v,t,g1=0,g2;
    cin>>t;
    while(t--)
    {
    	cin>>r>>k>>v;
    	if((r>=k)&&(r>=v))	//r is greatest
    		{
    			g1=r;
    			if(k>=v)
    				g2=k;	//k is second greatest
    			else
    				g2=v;
    		}
    	else
    		if((k>=r)&&(k>=v))	//k is greatest
    			{
    				g1=k;
    				if(r>=v)
    					g2=r;	//r is second greatest
    				else
    					g2=v;
    			}
    	else
    		if((v>=k)&&(v>=r))	//v is greatest
    			{
    				g1=v;
    				if(r>=k)
    					g2=r;	//r is second greatest
    				else
    					g2=k;
    			}
    	if(g1!=g2)		//g1>g2+1...
    		cout<<2*g2+1<<endl;
    	else
    		cout<<2*g2<<endl;
    }
    return 0;
}
