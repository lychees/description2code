#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
    int t;
    scanf("%d\n",&t);
    while(t--){
    	int num,f=0;long long int a=1,d=2,i;
    	scanf("%d\n",&num);
    	while(1){
    		f=0;
    		for(i=1;i*i<=a;i++){
    			if(a%i==0)
    			{	f++;
    				if((i*i)!=a)
    					f++;
    		}
    	}
    	if(f>num)
    	{
    		printf("%lld\n",a);
    		break;
    	}
    	a=a+d;
    	d++;
    }}
    return 0;
}
