#include <iostream>
using namespace std;

int main()
{
    int tc=0;
    cin>>tc;
    while(tc > 0) {
    	unsigned long int num =0,result=1, turns=2;
    	cin>>num; 
    	while(num > 1) {
    		if(num % 2 != 0) {
    			result += turns;
    		}
    		num /= 2;
    		turns *= 2;
    	}
    	cout<<result<<'\n';
    	tc--;
    }
    return 0;
}
