#include <iostream>
#include <string.h>
#include <stdio.h>
using namespace std;

int get()
{
  char c;
  while(c=getchar(),(c<'0'||c>'9')&&(c!='-'));
  int x=0;
  while(c>='0'&&c<='9')
    {
      x=x*10+c-48;
      c=getchar();
    }
  return x;
}


int main() {
	// your code goes here
	long long int T,N,X,prev,curr,change,i;
	long long int ans;
	const int MAX=100001;
	T=get();
	while(T--){
		ans=0;
		N=get();X=get();
		//N--;
		prev=get();
		for(i=1;i<N;i++){
			curr=get(); 
			if(prev>=curr) {
				//cout<<"prev="<<prev<<"curr="<<curr;
				change=((prev-curr+X)/X); //cout<<"change="<<change;
				curr+=change*X;
				ans+=change;
			}
			prev=curr;
		}
		printf("%lld\n",ans);
	}
	return 0;
}
