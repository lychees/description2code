#include <iostream>
#include <cstdio>
#include<algorithm>
#include<vector>
using namespace std;
bool prime[1000023] = {false};
int factor[1000023] = {0};
int main()
{
	long long res[1000023] = {0};
	res[1] = res[0] = 0;
	for(int i = 2;i * i <= 1000000;++i){
		if(!prime[i]){
			for(int j = 2;j * i <= 1000000;++j){
				prime[j*i] = true;
				if(!factor[j*i])factor[j*i] = i;
			}
		}
	}
	for(int i = 2;i <= 1000000;++i){
		res[i] = res[i - 1] + (prime[i]?factor[i]:0);
	}
	int test,l,r;
    scanf("%d",&test);
    while(test--){
    	scanf("%d%d",&l,&r);
    	long long val = res[r] - res[l - 1];
    	printf("%lld\n",val);
    }
    return 0;
}
