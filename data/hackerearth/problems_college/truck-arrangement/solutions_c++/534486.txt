#include <iostream>
#include <cstdio>
#include <algorithm>
#include <math.h>

using namespace std;

 struct arrnode
{
	int value;
	int index;
};

bool compare(arrnode a,arrnode b)
{
	if(a.value<b.value)
	   return true;
	   
	if(a.value==b.value)
	{
		if(a.index<b.index)
		   return true;
		
		
	}
	
	return false;
	   
}

void check(arrnode arr[],int n)
{
	sort(arr,arr+n,compare);

	int max=0,diff;

	int i;

	for(i=0;i<n;i++)
	{
		diff=abs(i-arr[i].index);
		if(diff>max)
			max=diff;
	}
	printf("%d",max);
}

int main()
{
		
        int test_cases;
        scanf("%d",&test_cases);
        while(test_cases > 0)
        {
                int N,i=0,ans;
                scanf("%d",&N);
                int arr[N];
                while(i<N)
                {
                        scanf("%d",&arr[i]);
                        i = i+1;
                }
               
                arrnode a[N];

                for(i=0;i<N;i++)
                {
                	a[i].value=arr[i];
                	a[i].index=i;
                }
       
                check(a,N);

                printf("\n");

                test_cases--;
        }
        return 0;
}
