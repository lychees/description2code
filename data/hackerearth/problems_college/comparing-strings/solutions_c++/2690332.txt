#include <iostream>
#include<vector>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		string a;
		string b;
		cin>>a;
		cin>>b;
		vector<int> count(256,0);
		int i=0;
		while(i<a.size()){
			count[a[i]]++;
			i++;
		}
		i=0;
		while(i<b.size()){
			if(count[b[i++]]>0){
				cout<<"YES"<<endl;
				break;
			}
		}
		if(i==b.size()){
			cout<<"NO"<<endl;
		}
	}
    return 0;
}
