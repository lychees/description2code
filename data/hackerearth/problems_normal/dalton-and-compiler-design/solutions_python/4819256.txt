t = int(raw_input())
ans = [[] for num in range(t)]

for i in range(t):
    n = int(raw_input())
    ans[i].append(str((n/2)))
    ans[i].append(str((n - n/2)))

for i in range(t):
    print ' '.join(ans[i])
