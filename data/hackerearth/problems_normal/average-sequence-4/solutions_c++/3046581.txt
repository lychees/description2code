#include <iostream>
using namespace std;

int main()
{
    int i,t;
    long long sum = 0;
    long value;
    long long cur = 0;
    cin>>t;
    cin>>sum;
    cout<<sum<<" ";
    for(int i=2;i<=t;i++)
    {
    	cin>>value;
    	cur = (value * i) - sum;
    	cout<<cur<<" ";
    	
        sum+=cur;
    	
    }
    return 0;
}
