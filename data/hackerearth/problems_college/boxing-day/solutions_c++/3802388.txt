#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int n,k;
	cin>>n>>k;
	int arr[n];
	fill(arr,arr+n,1);
	while(k--)
	{
		int x,y;
		cin>>x>>y;
		arr[x-1]=arr[x-1]+arr[y-1];
		arr[y-1]=0;
	}
	
	int cnt=0;
	for(int i=0;i<n;i++)
		if(arr[i]!=0)
			cnt++;
	
	cout<<cnt;
	
}
