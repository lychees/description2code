typedef long long int ll;
typedef pair<int, int > pii;

#define MOD 1000000007
#define pb push_back

void solve()
{
    int n;
    vector<pii> a(100005);
    for(int i=0;i<100005;i++)
    	a[i] = pii(1000000, 1);
    cin>>n;
    a[1].first = 1;
    a[1].second = 1;
    a[2].first = 1;
    a[2].second = 2;
    for(int i = 1;i<=n;i++)
    {
        pii p = a[i];
        int steps = p.first;
        int jump = p.second;
        if(i+jump <= n)
        {
            pii tmp = a[i+jump];
            if(tmp.first > steps+1)
            {
                tmp.first = steps+1;
                tmp.second = jump;
            }
            a[i+jump] = tmp;
        }
        if(i+jump+1 <= n)
        {
            pii tmp = a[i+jump+1];
            if(tmp.first > steps+1)
            {
                tmp.first = steps+1;
                tmp.second = jump+1;
            }
            a[i+jump+1] = tmp;
        }
    }
    cout<<a[n].first<<"\n";
}

int main()
{
    std::ios::sync_with_stdio(false);
    int t;
    cin>>t;
    //t = 1;
    while(t--)
    {
        solve();
    }
    return 0;
}
