using namespace std;
map <string,int> m;
void rec(string s,int a,int h)
{
	if(m.find(s)!=m.end())
	return;
	m[s]=1;
	string p=s;
	for(int i=1;i<p.length();i+=2)
	{
		char c=p[i];
		int j=p[i]-'0';
		j+=a;
		j=j%10;
		p[i]=j+'0';
	}
	rec(p,a,h);
	rotate(s.begin(),s.begin()+s.length()-h,s.end());
	rec(s,a,h);
}
int main()
{
	int t,a,h;
	string s;
	map <string,int>::iterator it;
	scanf("%d",&t);
	while(t--)
	{
		cin>>s;
		scanf("%d %d",&a,&h);
		h=h%s.length();
		rec(s,a,h);
		it=m.begin();

	}
}
