# include<stdio.h>
#include <vector>
using namespace std;
int max(int x, int y)
{ return (y >= x)? y : x; }
 
 
int maxSubArraySum(vector<int> &a)
{
   int max_so_far = a[0], i;
   int curr_max = a[0];
 
   for (i = 1; i < a.size(); i++)
   {
        curr_max = max(a[i], curr_max+a[i]);
        max_so_far = max(max_so_far, curr_max);
   }
   return max_so_far;
}
 
/* Driver program to test maxSubArraySum */
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		vector<int> a(n);
		for(int i = 0; i< n;i++)
			scanf("%d",&a[i]);
		 int max_sum = maxSubArraySum(a);
			printf("%d\n", max_sum);
		}
   return 0;
}
