#include <iostream>
using namespace std;

int main()
{
    long int t;
    typedef long long int ll;
    ll a, b, m, cnt, i, x, y;
    
    cin>>t;
    
    while(t--)
    {
    	cnt=0;
    	cin>>a>>b>>m;
    	x=a/m;
    	y=b/m;
    	cnt=y-x;
    	if(a%m==0)
    		cnt++;
    	cout<<cnt<<endl;
    }
    return 0;
}
