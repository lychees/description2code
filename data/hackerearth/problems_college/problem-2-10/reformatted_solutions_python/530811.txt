l = raw_input()
[a, b] = l.split()
a = int(a)
b = int(b)


def gcd(a, b):
    if b == 0:
        return a
    else:
        while not b == 0:
            c = b
            b = a % b
            a = c
        return a


g = gcd(a, b)
print str(a/g)+' '+str(b/g)
