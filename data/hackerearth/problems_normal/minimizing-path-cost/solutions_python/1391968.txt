INF = 999999999999

def floydWarshall(graph):
	nodes = graph.keys()
	distance = {}
	for n in nodes:
		distance[n] = {}
		for k in nodes:
			distance[n][k] = graph[n][k]
	for k in nodes:
		for i in nodes:
			for j in nodes:
				distance[i][j] = min(distance[i][j], distance[i][k]+distance[k][j])

	return distance
