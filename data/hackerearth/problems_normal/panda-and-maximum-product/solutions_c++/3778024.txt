#include <iostream>
using namespace std;

int main()
{
    long long int p1,p2,pos1=-1e9,pos2=-1e9,neg1=1e9,neg2=1e9,n,t,result=-1e9;
    cin>>n;
    for(p1=0;p1<n;p1++)
    {
    	cin>>t;
    	if(t>=0)
    	{
    		if(t>pos1)
    		{
    			pos2=pos1;
    			pos1=t;
    		}
    		else if(t>pos2)
    			pos2=t;
    	}
    	else
    	{
    		if(t<neg1)
    		{
    			neg2=neg1;
    			neg1=t;
    		}
    		else if(t<neg2)
    			neg2=t;
    	}
    }
    if(pos2==-1e9)
    	result=max(result,pos1);
    else
    	result=max(result,pos1*pos2);
    if(neg1!=1e9)
    {
    	if(neg2!=1e9)
    		result=max(result,neg1*neg2);
    	else
    		result=max(result,neg1);
    }
    cout<<result;
    return 0;
}
