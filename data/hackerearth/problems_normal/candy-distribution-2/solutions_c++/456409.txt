#include <stdio.h>
#include <stdlib.h>
#include <math.h> 
#include <iostream>

using namespace std;


void CandyDistribution2()
{
	int numTests = 0;
	scanf("%d", &numTests);

	int numElem = 0;
	for (int x = 0; x < numTests; x++)
	{
		scanf("%d", &numElem);
		int temp = 0;
		long long minSum =0;

		for (int i = 0; i < numElem; i++)
		{
			scanf("%d",&temp);
			minSum += temp - 1;
		}

		minSum++;

		cout << minSum << endl;
	}



}

int main()
{
    CandyDistribution2();
    return 0;
}
