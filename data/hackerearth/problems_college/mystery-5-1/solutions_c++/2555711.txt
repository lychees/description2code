#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n;
    string str;
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	int b[26]={0};
    	int i,m;
    	n=str.length();
    	for(i=0;i<n;i++)
    	{
    		m=str[i]-'a';
    		++b[m];
    	}
    	cout<<*max_element(b,b+26)<<endl;
    }
    return 0;
}
