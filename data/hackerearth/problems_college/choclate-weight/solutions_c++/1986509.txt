#include <iostream>
using namespace std;

int main() {
	int n;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	int q;
	cin>>q;
	while(q--)
	{
		int x;
		cin>>x;
	int sum=0,flag=0;
	
	for(int i=0;i<n;i++)
	{
		for(int j=i+1;j<n;j++)
		{
			if(x==a[i] ||x==a[i]+a[j])
			flag=1;
		}
	}
	
	if(flag==1)
	cout<<"I Like You"<<endl;
	else
	cout<<"I Hate You"<<endl;
	}
	
	
	return 0;
}
