#include <iostream>
#include <cmath>
#include <algorithm>
#include <cstring>
using namespace std;

int main() {
	// your code goes here
	int t,i,n,c, m;
	cin>>t;
	char a[10005];
	while(t--){
		cin>>a;
		c = m = 0;
		n = strlen(a);
		i = 0;
		int dp[30] = {0};
		while(i < n){
			dp[a[i] - 97]++;
			i++;
		}
		for(i = 0; i < 27; i++)
			m = max(m, dp[i]);
		cout<<m<<endl;
	}
	return 0;
}
