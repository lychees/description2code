#include <iostream>
#include <map>
#include <algorithm>
using namespace std;
multimap<long long int,long long int> cost;
long long int solve(long long int n)
{
	multimap<long long int,long long int>::iterator it=cost.find(n);
	if(it!=cost.end())
	{
		return it->second;
	}
	if(n<=0)
	return 0;
	long long int x=solve(n/2),y=solve(n/3),z=solve(n/4),ret;
	if(x+y+z > n)
	ret=x+y+z;
	else
	ret=n;
	cost.insert(make_pair(n,ret));
	return ret;
}
int main()
{
	long long int n;
	while(cin >> n)
	{
    cout << solve(n) << '\n';
	}
    return 0;
}
