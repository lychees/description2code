noOfTestCases = int(raw_input())
import collections

def customSort(string, orderString):
	ordDic = collections.OrderedDict()
	for c in orderString:
		ordDic[c] = 0
	for c in string:
		ordDic[c] += 1
	output = ""
	for k in ordDic.keys():
		output += ordDic[k] * k
	print output	
			

for caseNo in range(1, noOfTestCases + 1):
	orderString = raw_input()
	string = raw_input()
	customSort(string, orderString)
