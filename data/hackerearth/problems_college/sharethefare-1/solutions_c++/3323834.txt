#include <iostream>
#include <string>

using namespace std;

struct person
{
    string name;
    int amt;
};

struct person p[100];

int n;
void calc(string,int,int);
void print();


int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int q;
        cin>>n;
        cin>>q;
        for(int i=0;i<n;i++)
            {
            	cin>>p[i].name;
            	p[i].amt=0;
            }
        for(int i=0;i<q;i++)
        {
            string payee,others;
            int nop;
            int money;
            cin>>payee;
            cin>>money;
            cin>>nop;
            //float div=(float)money/nop+1;
            //int actual=(int)div;
            int act=money/(nop+1);
            //if(actual!=div)
            int ad=money-(act+(money%(nop+1)));
                calc(payee,ad,0);
            //else
              //  calc(payee,actual,0);
            for(int x=0;x<nop;x++)
            {
                cin>>others;
                calc(others,act,1);
            }
        }
        print();




    }
    return 0;
}

void calc(string pr,int mon,int todo)
{
    for(int i=0;i<n;i++)
    {
        if(pr==p[i].name)
        {
            if(todo==0)
            {
                p[i].amt+=mon;
            }
            else
                p[i].amt-=mon;
        }
    }
}
void print()
{
    for(int i=0;i<n;i++)
    {
        if(p[i].amt==0)
        {
            cout<<p[i].name<<" neither owes nor is owed\n";
        }
        else if(p[i].amt>0)
        {
            cout<<p[i].name<<" is owed "<<p[i].amt<<"\n";
        }
        else
        {
            cout<<p[i].name<<" owes "<<p[i].amt*-1<<"\n";
        }
    }
}
