#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
          int t;
          scanf("%d",&t);
          while(t--)
          {
          char str[1000];
          scanf("%s",str);
          int l,i,sum1=0,sum2=0;
          l=strlen(str);
          for(i=0;i<l/2;i++)
          {
                  sum1=sum1+ (str[i]-'0');
          }
          for(i=l/2;i<l;i++)
          {
                  sum2=sum2+ (str[i]-'0');
          }
          if(sum1==sum2)
                    printf("YES\n");
          else
                    printf("NO\n");
          }
          return 0;
}
