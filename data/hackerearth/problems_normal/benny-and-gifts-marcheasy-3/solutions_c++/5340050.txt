#include <bits/stdc++.h>
using namespace std;

#define rep(i, a, b) for (int i = (a); i < (b); i++)

#define mp make_pair

typedef pair<int, int> ii;

const int MAXN = int(1e5) + 5;

string s;
ii p[MAXN];
int n, sz;

int main() {
//    freopen("input.inp", "r", stdin);

    cin >> s;
    int x(0), y(0);
    p[sz++] = mp(x, y);

    for (int i = 0; i < s.length(); i++) {
        if (s[i] == 'L') y--;
        else if (s[i] == 'R') y++;
        else if (s[i] == 'U') x--;
        else x++;

        p[sz++] = mp(x, y);
    }

    sort(p, p+sz);
    int res(0);
    for (int i = 1; i < sz; i++) {
        if (p[i] == p[i-1]) res++;
    }

    printf("%d", res);
}
