#define MOD 1000000007
using namespace std;

int main()
{
    int t,n,m;
    long long ans;
    cin>>t;
    while(t--)
    {
    long long dp[1001][1001]={0};
    cin>>n>>m; 
    ans=0;
    dp[0][0]=1;
    for(int i=1;i<=n;i++)
    { dp[0][i]=0;
      dp[i][0]=1;
   }
    for(int i=1;i<=n;i++)
    {
    	for(int j=1;j<=n;j++)
    	{
    		if(i<j*m)
    		continue;
    		dp[i][j]=(dp[i-1][j]+dp[i][j-1])%MOD;
    		
    	}
    	
    }
    for(int i=n;i>=0&&i>=(n-i)*m;i--)
    {
    	
        if(i>n-i)
    	 ans=ans+dp[i][n-i];
    	 ans=ans%MOD;
        
    }
    cout<<ans<<"\n"; 
    }
    return 0;
}
