{

	cin>>t;
	while(t--)
	{
    char a[10000],b[10000],c[10000];
    int x,i;

    {
    	if(a[i]==b[i])
    	c[i]='0';
    	else

    }
    for(i=0;i<x;i++)
    cout<<c[i];

    }

}
