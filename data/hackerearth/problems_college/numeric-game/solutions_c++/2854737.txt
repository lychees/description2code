    while(tc--){
    	int a1,b1,a2,b2;
    	cin>>a1>>a2>>b1>>b2;
    	int min_limit[a2 + 1];
    	for(int i = 0;i <= a2;i++)
    		min_limit[i] = MOD;

    	min_limit[1] = 1;
    	min_limit[2] = 2;
    	min_limit[3] = 2;
    	int count = 0;

    	for(int i = 3;i < a2 + 1;i++){
    		for(int j = min_limit[i];j - min_limit[i] < i;j++){
    			if(j <= a2)
    				min_limit[j] = min(min_limit[j],i);
    		}
    	}

    	for(int i = a1;i <= a2;i++){
    		//cout<<min_limit[i]<<endl;
    		for(int j = b1;j <= b2;j++){
    			if(min_limit[i] <= j && min_limit[i] + i - 1 >= j)
    				continue;
    			else
    				count++;
    		}
    	}
    	cout<<"Case #"<<cases<<": "<<count<<endl;
    	cases++;
    }
}
