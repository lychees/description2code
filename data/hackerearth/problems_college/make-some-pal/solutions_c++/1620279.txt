#include <iostream>
#include<string>
using namespace std;
bool p(string str,int a,int b)
{
	int i,j;
	for(i=a,j=b;j>i;i++,j--)
	{
		if(str[i]!=str[j])
		return false;
	}
	return true;
}
int pal(string str)
{	int i=0,j=str.length()-1;
   if(p(str,i,j))
   return str.length();
   else
   {
   i++;
   while(i<j)
   {
   if(p(str,i,j))break;
   i++;
   }
   return i+str.length();
   }
}
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
  	string str;
  	cin>>str;
  	cout<<pal(str)<<endl;
  }
}
