#include<iostream>
#include<string>
using namespace std;

int main(){

int T;
cin>>T;
string S1;
string S2;
int N;
while(T--){
cin>> S1 >> S2 >> N;
int cnt = 0;
if(S1.size() == S2.size()){
   for(int i=0;i< S1.size();i++){
        if(S1[i] != S2[i]){
            cnt++;
        }
   }
   if(cnt == N){
    cout<<"Yes";
   }
   else if(cnt > N)
   {
   	cout<<"No";
   }
   else if(cnt < N){
       if((N-cnt)%2)
          cout<<"No";
       else cout<<"Yes";
   }
   else if(cnt == 0)
   {
       if(N%2)
          cout<<"No";
       else cout<<"Yes";
   }
  
   cout<<endl;

}else{
    cout<<"No"<<endl;
}

}
return 0;
}
