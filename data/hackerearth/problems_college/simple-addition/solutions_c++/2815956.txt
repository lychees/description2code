int gcd(int a, int b){
    return (a == 0) ? b : gcd(b % a, a);
}

int findroot(int a, int b){
    int cnt = 0;
    while (a > 1 || b > 1){
        if (a > b) a -= b;
        else b -= a;
        cnt++;
    }
    return cnt;
}
