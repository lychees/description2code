#include <bits/stdc++.h>

using namespace std;
int arr[50];
int search1(int l,int h,int val)
{
    while(l<h){
        int mid=l+(h-l+1)/2;
        if(arr[mid]<=val)
            l=mid;
        else h=mid-1;
    }
    return l;
}
int main()
{
    int t;
    scanf("%d",&t);
    arr[0]=0;
    arr[1]=4;
    long long int x=4,i=2,sum=0;
    while(sum<=100000000){
        arr[i]=arr[i-1]+x*2;
        x*=2;
        sum=arr[i];
        i++;
    }
    int h=i-1;
    while(t--){
        scanf("%lld",&x);
        x-=4;
        i=search1(0,h,x);
        //cout<<x<<"  "<<arr[i]<<endl;
        x-=arr[i];
        if(x==0){
            printf("1\n");
        }
        else{
            int ans=3+(2*(x-1));
            printf("%d\n",ans);
        }
    }
    return 0;
}
