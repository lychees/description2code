#include <iostream>

using namespace std;


int main()
{
    int t;
    long long int k, a, b, c;
    cin>>t;
    while(t--)
    {
        long long int ans =0ll,ans1 = 0ll;
        cin>>k;
        cin>>a>>b>>c;
        ans = a+b+c;
        ans1= 1ll;
        for(int i=0;i<k;i++)
        {
            ans1 *= ans;
        }
        cout<<ans1<<endl;
    }
    return 0;
}
