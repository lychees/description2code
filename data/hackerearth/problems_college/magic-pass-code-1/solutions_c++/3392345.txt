#include<iostream>
#include<cstdio>
using namespace std;
int main(){
	char arr[100001];
	int max=0;
	int num=0;
	while(scanf("%s",arr)!=EOF){
		int sum=0;
		for(int i=0;arr[i]!='\0';i++){
			sum+=arr[i]-'0';
		}
		if(max<sum){
			max=sum;
			string str(arr);
			num=stoi(str);
		}
	}
	cout<<num<<endl;
}
