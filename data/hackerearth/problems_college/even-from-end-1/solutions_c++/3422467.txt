#include<iostream>
#include<stack>
using namespace std;

int main()
{
	stack<int>A;
	int num, tmp;
	int count=0;
	while(cin>>num){
		if(count++){cout<<endl;count++;}
		for(int a=0;a<num;a++){
			cin>>tmp;
			if(tmp%2==0)A.push(tmp);
		}
		if(A.size()==0){
			cout<<"None to display";
			continue;
		}
		int counter=0;
		while(!A.empty()){
			if(counter++){cout<<" ";counter++;}
			cout<<A.top();
			A.pop();
		}
	}
}
