#include <iostream>
#include <algorithm>
#include <vector>
#include <map>
#include <string>
using namespace std;
 
int main()
{
	map<string, int> nPlayer;
	int nTestCases;
	cin >> nTestCases;
	string fDriver, sDriver;
	for (int i = 0; i < pow(2,nTestCases)-1; i++)
	{
		cin >> fDriver >> sDriver;
		if(nPlayer[fDriver] != -1 )
			nPlayer[fDriver] = 1;
		nPlayer[sDriver] = -1;
	}
	for (map<string, int>::iterator it = nPlayer.begin(); it != nPlayer.end(); it++) {
		if (it->second == 1) {
			cout << it->first;
			break;
		}
 
	}
	return 0;
}
