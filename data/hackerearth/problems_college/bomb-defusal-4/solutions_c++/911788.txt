#include <iostream>
#include <math.h>
#include <cstdlib>
using namespace std;
int main()
{
    int t;
    int x,y,u,v,p,start,finish;
    cin >> t;

    while(t--){
        cin >> p>>x>>y>>u>>v;
        start = u-v;
        finish = x-y;
        if (x == u && y == v){
            cout << "Counter-Terrorists Win !"<<endl;
        }
        else {
        if (start == finish){
            if (p > 1 ){
            cout <<"Counter-Terrorists Win !"<<endl;
            }
            else {
                cout <<"Terrorists Win !"<<endl;
            }
        }
        else {
            if (abs(start)%2 == abs(finish)%2){
                if (p > 2){
                    cout << "Counter-Terrorists Win !"<<endl;
                }
                else {
                    cout <<"Terrorists Win !"<<endl;
                }
            }
            else{
                cout << "Terrorists Win !"<<endl;
            }
        }
    }
    }
    return 0;
}
