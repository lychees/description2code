#include <iostream>
#include<set>
using namespace std;

int main()
{
   int n,m,notest;
   set<long int> setofcandies;
   cin>>notest;
   for(int i=1 ; i <= notest ; i++)
   {
   	 cin >> n >>m;
     setofcandies.clear();
   	  long int nc;
   	  for(int j= 1 ; j <= n ; j++ )
   	  {
   	  	 cin >>nc;
   	  	 setofcandies.insert(nc);
   	  	 
   	  }
   	  
   	  for(int j=1 ; j <= m ; j++ )
   	  {
	   	  	cin >> nc;
	   	  	if(setofcandies.find(nc) != setofcandies.end())
	   	  	{
	   	  		cout<<"YES\n";
	   	  	}
	   	  	else
	   	  	{
	   	  		cout<<"NO\n";
	   	  	}
	   	  	setofcandies.insert(nc);
   	  	
   	  }
   }
    return 0;
}
