    #include<bits/stdc++.h>
    using namespace std;
    int main() {

    	int t,j;
    	scanf("%d",&t);
    	while(t--) {
    		int m,n,res = 0;
    		scanf("%d",&m);
    		scanf("%d",&n);
    		int arr[n];
    		int i;
    		for(i=0;i<n;i++) {
    			scanf("%d",&arr[i]);
    		}
            sort(arr,arr+n);
    		j = n - 1;

    		for(i = 0; i <= j;) {
    			if(arr[j] > m)
    				--j;
    			else if (arr[i] + arr[j] > m) {
    				--j;
    				++res;
    			}
    			else {
    				--j;
    				++i;
    				++res;
    			}
    		}
    		printf("%d\n",res);
    	}


    	return 0;

    }
