#define MAX 1000000
long long int arr[1003];

int fact(int n,int res[],int res_size)
{
	long long int sum=0;
	res_size=multiply(n,res,res_size);
	for(int i=0;i<res_size;i++)
	{
		sum+=res[i];
	}
	arr[n]=sum;
	return res_size;
}
int multiply(int x,int res[],int res_size)
{
	int prod,carry=0;
	for(int i=0;i<res_size;i++)
	{
		prod=res[i]*x+carry;
		res[i]=prod%10;
		carry=prod/10;
	}
	while(carry)
	{
		res[res_size]=carry%10;
		carry=carry/10;
		res_size++;
	}
	return res_size;
}
