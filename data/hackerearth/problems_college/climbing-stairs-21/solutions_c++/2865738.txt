#include<bits/stdc++.h>
using namespace std;
long long sum=0;
long long dp[1001];
long long solve(int n){
	if(n==0){
		return 1;
	}
	if(dp[n]!=-1)
	  return dp[n];
	  long long ans1=0;
	if(n>=2)
	ans1=solve(n-2);
	if(n>=1)
	ans1+=solve(n-1);
	dp[n]=ans1;
	return dp[n];
}
int main(){
  int t;
  scanf("%d",&t);
  for(int i=0;i<=1000;i++)
  dp[i]=-1;
  while(t--){
  	int n;
  	sum=0;
  	scanf("%d",&n);
  	solve(n);
  	printf("%lld\n",dp[n]);
  }
  return 0;	
}
