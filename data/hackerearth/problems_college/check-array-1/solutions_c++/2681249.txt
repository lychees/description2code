#include <bits/stdc++.h>
 
using namespace std;
 
int main()
{
   int t;
   cin>>t;
   while(t--){long long int n;
   cin>>n;
    long long int a[n],c=0,s1=0,s2=0;
    for(int i=0;i<n;i++){
        cin>>a[i];
        c+=a[i];
    }
    for(int i=1;i<n-1;i++)
    {
 
        s1+=a[i-1];
    if(s1==(c-s1-a[i])){
        cout<<"YES"<<endl;s2=1;
        break;
    }
    }
    if(s2==0)
        cout<<"NO"<<endl;
   }
    return 0;
}
