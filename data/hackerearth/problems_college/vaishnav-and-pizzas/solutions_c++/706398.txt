#include <iostream>
#include <stdio.h>
#include <vector>

using namespace std;

typedef long long int Lint;

Lint fi(Lint n) 
{ 
		if( n == 0 || n == 1 )
			return 0;

       Lint result = n, i; 
       for(i=2;i*i <= n;i++) 
       { 
         if (n % i == 0) 
         	result -= result / i; 
         while (n % i == 0) 
         	n /= i; 
       } 
	    if (n > 1) 
	       	result -= result / n; 

       return result; 
} 

int main() {
	
	Lint n = 10001, i, sum, t, v;

	vector<Lint> A( n ), B( n );

	for( i = 0;i < n;i++ )
		A[ i ] = fi( i );

	for( i = 0, sum = 0;i < n;i++ ) {
		sum += A[ i ];
		B[ i ] = sum;
	}

	scanf("%lld", &t );

	while( t-- ) {

		scanf("%lld", &v );
		printf("%lld\n", B[ v ] );	
	}
	return 0;
}
