t = int(raw_input())
for i in range(0, t):
    x, y = map(int, raw_input().split(" "))
    k = 0
    if x > y:
        a = x
        b = y
    else:
        a = y
        b = x
    while a > 1 or b > 1:
        k = k+a/b
        a = a % b
        a, b = b, a
        if a > 1 and b == 1:

            k = k+a-b
            a = 1

    if x == 1:
        k = y-x
    if y == 1:
        k = x-y

    print k
