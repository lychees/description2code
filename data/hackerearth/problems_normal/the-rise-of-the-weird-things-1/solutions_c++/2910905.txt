#include<bits/stdc++.h>
using namespace std;

int main()
{
    int odd[1005], even[1005];
    int n, temp;
    cin>>n;
    int j=0, k=0;
    for(int i=0; i<n; i++)
    {
        scanf("%d",&temp);
        if(temp&1)
            odd[j++]=temp;
        else
            even[k++]=temp;
    }

    sort(&odd[0], &odd[j]);
    sort(&even[0], &even[k]);

    long long int sum=0;
    for(int i=0; i<k; i++)
    {
        printf("%d ",even[i]);
        sum+=even[i];
    }
    cout<<sum<<" ";

    sum=0;
    for(int i=0; i<j; i++)
    {
        printf("%d ",odd[i]);
        sum+=odd[i];
    }
    cout<<sum;

    return 0;
}
