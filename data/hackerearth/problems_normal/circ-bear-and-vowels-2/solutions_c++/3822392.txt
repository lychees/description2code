#include <bits/stdc++.h>
using namespace std;

string temp;
int t;

int main(void)
{
    int l,v,c,cont,mx;
    scanf("%d",&t);
    while(t--)
    {
        cin>>temp;
        l=temp.length();
        v=0;
        c=0;
        cont=0;
        mx=0;
        for(int i=0;i<l;i++)
        {
            if(temp[i]=='a'||temp[i]=='e'||temp[i]=='i'||temp[i]=='o'||temp[i]=='u'||temp[i]=='y')
            {
                v++;
                cont=0;
            }
            else 
            {
                c++;
                cont++;
                if(mx<cont) mx=cont;
            }
        }
        if(c>v || mx>=3) printf("hard\n");
        else printf("easy\n");
    }
    return 0;
}
