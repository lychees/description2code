def outlist(data):

    runMap = {}
    for entry in data:
        cCode, run = int(entry[0]), int(entry[1])
        if (runMap.has_key(cCode) == False):
            runMap[cCode] = [run]
        else:
            runMap[cCode].append(run)

    sortedCCode = sorted(runMap.keys())

    for cCode in sortedCCode:
        runList = reversed(sorted(runMap[cCode]))
