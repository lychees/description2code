#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t,n;
   
   
   
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	int a[2*n];
   	n=2*n;
   	for(int i=0;i<n;i++)
   		cin>>a[i];
   		
   	int sum=0;
   	sort(a,a+n);
   	for(int i=0;i<n;i++)
   	{
   		if(a[i]+a[n-i-1]>sum)
   		sum=a[i]+a[n-i-1];
   	}
   	
   	cout<<sum<<endl;
   }
    return 0;
}
