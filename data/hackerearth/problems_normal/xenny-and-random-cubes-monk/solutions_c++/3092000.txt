#include<bits/stdc++.h>
using namespace std;
#define lx long long
#define mod 1000000007
int n,k;
lx dp[10][1<<10];
char ss[11][7];
string s;
int solve (int pos,int mask);

int main()
{
    ios::sync_with_stdio(0);
    //freopen("input.txt","r",stdin);
    memset(dp,-1,sizeof(dp));
    cin>>n>>k;
    for (int i=0;i<n;i++)
        for (int j=0;j<6;j++)
            cin>>ss[i][j];
    cin>>s;
    lx ans = solve(0,0);
    cout<<ans<<endl;
    return 0;
}
int solve (int pos,int mask)
{
    if(pos == k)
        return 1;
    if(dp[pos][mask] != -1)
        return dp[pos][mask];
    lx ret = 0;
    for (int i=0;i<n;i++)
    {
        if (mask&(1<<i))
            continue;
        int c = 0;
        for (int j=0;j<6;j++)
        {
            if (ss[i][j] == s[pos])
                c++;
        }
        ret += (c*solve(pos+1,mask|(1<<i)))%mod;
        ret %= mod;
    }
    return dp[pos][mask] = ret;
}
