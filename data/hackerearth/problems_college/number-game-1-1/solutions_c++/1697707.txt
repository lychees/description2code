#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   		long long n;
   		cin>>n;
   		long long cu=0;
   		while(n>1)
   		{
   			if(n%3==0)
   			{
   				n/=3;
   				cu++;
   			}
   			else if(n%2==0)
   			{
   				n/=2;
   				cu++;
   			}
   			else
   			{
   				n-=1;
   				cu++;
   			}
   			
   		}
   		cout<<cu<<endl;
   }
    return 0;
}
