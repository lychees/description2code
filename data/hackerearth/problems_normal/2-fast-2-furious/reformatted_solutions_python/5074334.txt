import math
for i in xrange(int(raw_input())):
    try:
        dom = raw_input().split(' ')
    except (EOFError):
        break
    try:
        brian = raw_input().split(' ')
    except (EOFError):
        break
    mdom = 0
    mbrian = 0
    n = len(dom)
    for i in xrange(n-1):
        k = math.fabs(int(dom[i+1])-int(dom[i]))
        if k > mdom:
            mdom = k
        l = math.fabs(int(brian[i+1])-int(brian[i]))
        if l > mbrian:
            mbrian = l
    if mdom > mbrian:
        print "Dom"
        print int(mdom)
    elif mdom < mbrian:
        print "Brian"
        print int(mbrian)
    else:
        print "Tie"
        print int(mdom)
