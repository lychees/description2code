#include <iostream>
#include <cstdio>
#include <map>
using namespace std;
int main()
{
    string s,s1,s2;
    int t,i;
    multimap<string,string> m;
    multimap<string,string> :: iterator it,it1;
    cin >> t;
    while(t--){
        s1 = s2 = "";
        cin >> s;
        i = 0;
        while(s[i] != '-'){
            s1 += s[i];
            i++;
        }
        int j = i+1;
        for(i=j;i<s.size();i++){
            s2 += s[i];
        }
        m.insert(pair<string,string>(s1,"start"));
        m.insert(pair<string,string>(s2,"finish"));
    }
    bool f = true;
    int j = 0;
    string next = "";
    for(it = m.begin(); it != m.end(); it++,j++){
       // cout << (*it).first << " " << (*it).second << endl;
        string curr = (*it).first;
        it1 = it;
        it1++;
        if(it1 != m.end()){
            next = (*it1).first;
        }else{
            next = "";
        }
        if(next == curr){
         //   cout << curr << " " << next << endl;
         //   cout << "hh\n";
            //curr = pre;
            continue;
        }
        string y = (*it).second;
        if(j%2==0 && y == "finished"){
            f = false;
            break;
        }
        if(j%2==1 && y == "start"){
            f = false;
            break;
        }
    }
    if(f){
        cout << "Who needs a moderator?\n";
    }else{

        cout << "Will need a moderator!\n";
    }
    return 0;
}
