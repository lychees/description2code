#include <iostream>
using namespace std;

int main()
{
    int a,b;
    int r1,r2;
    int diff=0;
    scanf("%d %d",&a,&b);
    while(a!=0)
    {
    	r1=a%10;
    	r2=b%10;
    	if(r1!=r2)
    		diff++;
    	a=a/10;
    	b=b/10;
    }
    if(diff<=1)
    	printf("wins");
    else
    	printf("loses");
    return 0;
}
