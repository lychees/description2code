#include<bits/stdc++.h>
using namespace std;
int main()
{
    int te;
    scanf("%d",&te);
    while(te--)
    {
        int n,k,i,j;
        scanf("%d%d",&n,&k);
        char a[n][k+1],b[k][n];
        for(i=0;i<n;i++)
            scanf("%s",a[i]);
        for(i=0;i<n;i++)
            for(j=0;j<k;j++)
                b[j][i]=a[i][j];
        for(i=0;i<k;i++)
        {
            sort(b[i],b[i]+n);
            printf("%c",b[i][(n-1)/2]);
        }
        printf("\n");
    }
    return 0;
}
