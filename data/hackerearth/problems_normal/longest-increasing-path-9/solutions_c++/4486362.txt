#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pii pair<int,int>
#define pll pair<ll,ll>
#define mp make_pair
#define ff first
#define ss second
#define pb push_back
#define pf pop_front
#define IOS ios::sync_with_stdio(false)
int t , n , m , a[105][105] ,dp[105][105];
int dx[] = {-1,0};
int dy[] = {0,-1};
bool check(int x , int y , int ind){
    if(x >= 1 && x <= n && y >= 1 && y <= m ){
        if(a[x][y] < a[x-dx[ind]][y-dy[ind]])return true;
        else return false;
    }
    return false;
}
int main(){

	int i , j , k;
    cin >> t;
    while(t--){
        cin >> n >> m ;
        for(i = 1; i <= n ; i++){
            for(j = 1 ; j <= m ; j++){
                cin >> a[i][j];
            }
        }
        memset(dp,0,sizeof(dp));
        dp[1][1] = 1;
        int ans = 1;
        for(i = 1 ; i <= n ; i++){
            for(j = 1 ; j <= m ; j++){
                if(i==1 && j==1)continue;
                for(k = 0 ; k < 2 ; k++){
                    int oldx = i + dx[k];
                    int oldy = j + dy[k];
                    if(check(oldx,oldy,k) && dp[oldx][oldy]!=0){
                        dp[i][j] = max(dp[i][j],dp[oldx][oldy]+1);
                    }
                }
                ans = max(ans,dp[i][j]);
            }
        }
        cout << ans << endl;
    }
	return 0;
}
