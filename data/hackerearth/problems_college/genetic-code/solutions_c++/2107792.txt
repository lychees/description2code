#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	char prev,arr[101],key;
	int n,count,in=0;
	cin>>arr;
	n=strlen(arr);
	

   	
   	prev=arr[0];
   	count=1;
   	for(int i=1;i<n;i++)
   	{
   		key=arr[i];

   	    if(key==prev)
   	    count++;
   	    
   	    else

		{if(count%2==0)
		in++;
			count=1;
		}
   	    prev=key;

   	}

	 if(count%2==0)
        in++;


	cout<<in;
    return 0;
}
