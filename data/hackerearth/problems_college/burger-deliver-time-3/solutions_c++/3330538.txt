#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    int i,n,a,d;
    scanf("%d",&n);
    vector<pair<int,int> > v(n);
    for(i=0;i<n;i++)
    {
    	scanf("%d%d",&a,&d);
    	v[i].first=a+d;
    	v[i].second=i+1;
    }
    sort(v.begin(),v.end());
    for(i=0;i<n;i++)
    {
    	printf("%d ",v[i].second);
    }
    return 0;
}
