#include <iostream>
#include <map>
#include <queue>
using namespace std;
#define ll long long
int main()
{
    //cout << "Hello World!" << endl;
    ll m , n, sum = 0;
    cin >> m >> n;
    map<ll,ll> mp;
    priority_queue<ll> pq;
    for(ll i = 0 ; i < m ; ++i) {
    	ll x;
    	cin >> x;
    	//a[i] = x;
    	mp[x]++;
    	pq.push(x);
    }
    for(ll i = 0;  i < n ; ++i) {
    	ll x = pq.top();
    	sum += x;
    	pq.pop();
    	pq.push(x - 1);
    }
    cout << sum;
    return 0;
}
