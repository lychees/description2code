#include <iostream>
using namespace std;
bool prime(int x)
{
	for(int i=2;i<=x/2;i++)
	{
		if(x%i==0)return false;;
	}
	return true;
}


int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int c[n],p[n];
    	for(int i=0;i<n;i++)
    	cin>>c[i];
    	for(int i=0;i<n;i++)
    	cin>>p[i];
    	int s1=0,s2=0;
    	for(int i=0;i<n;i++)
    	{
    		bool a=prime(p[i]);
    		bool b=prime(c[i]);
    		if(a && !b && c[i]!=1)
    		{
    			if(c[i]>=p[i])
    			{
    				c[i]-=p[i];
    				p[i]=0;
    			}
    			else
    			{
    				p[i]-=c[i];
    				c[i]=0;
    			}
    		}
    		else if(a)
    		c[i]=0;
    		else if(!b && c[i]!=1)
    		p[i]=0;
    		else
    		{if(c[i]>=p[i])
    			{
    				c[i]-=p[i];
    				p[i]=0;
    			}
    			else
    			{
    				p[i]-=c[i];
    				c[i]=0;
    			}
    			
    		}
    		s1+=c[i];
    		s2+=p[i];
    		//cout<<s1<<" "<<s2<<endl;
    	}
    	if(s1>s2)
    	cout<<"The Composites"<<endl;
    	else if(s1<s2)
    	cout<<"The Primes"<<endl;
    	else
    	cout<<"Tie"<<endl;
    }
    return 0;
}
