#include <iostream>
#include <functional>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T ;
    cin>>T ; 
    while (T--)
    {
    	int N ; 
    	int K ;
    	cin>>N>>K ; 
    	int i ;
    	long long maxm=0 ; 
    	vector<long long >DP(N+1) ;
    	DP[0]=0;
    	for(i=1;i<N+1;i++)
    	{
    		cin>>DP[i] ;
    		
    	}

    	for(i=1 ;i<N+1;i++) 
    	{
    		if(i<K+1)
    		  {
    		  	DP[i]=max(DP[i],DP[i-1]) ;
    		  }
    		
    		 else{  
    		DP[i]=max(DP[i],DP[i-K-1]+DP[i]);
    		if(DP[i]<DP[i-1])
    		   DP[i] =DP[i-1] ;
    		 }
    	   if(DP[i]>maxm) 
    	     maxm = DP[i] ;
    		
    	}
    	
    	cout<<maxm<<"\n"  ;
    }
    	return 0;
}
