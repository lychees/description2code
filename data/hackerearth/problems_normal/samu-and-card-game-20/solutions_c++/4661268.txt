#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	int a,count=0;
	long long int n,m,x,y,posx,posy;
	
	while(t--)
	{
		posx = 1;
		posy = 1;
		count = 0;
		cin>>n>>m>>a;
		while(a--)
		{
			cin>>x>>y;
			while((posx+x<=n)&&(posy+y<=m)&&(posx+x>=1)&&(posy+y>=1))
			{
				posx+=x;
				posy+=y;
				count++;
			}
		}
		cout<<count<<endl;
	}
    return 0;
}
