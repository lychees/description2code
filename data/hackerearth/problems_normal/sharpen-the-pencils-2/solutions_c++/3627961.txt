#include <iostream>
using namespace std;

int main()
{
    int T, i, j;
    cin>>T;
    while (T > 0){
        int N;
        cin>>N;
        if (N == 1) {
            cout<<"1 0"<<endl;
        }
        else if (N == 2) {
            cout<<"1 1"<<endl;
        }
        else{
            int A[N];
            for (int k = 0; k < N; k++){
                cin>>A[k];
            }
            i = 0;
            j = N-1;
            int S = 2*A[N-1];
            int count = N-2;
            while (count > 0){
                if (S - A[i] > 0){
                    S-=A[i];
                    i++;
                    count--;
                }
                else if (S - A[i] < 0){
                    A[i] -= S;
                    j--;
                    count--;
                    S = 2*A[j];
                }
                else{
                    i++;
                    j--;
                    count-=2;
                    S = 2*A[j];
                }
            }
            cout<<i+1<<" "<<N-i-1<<endl;

        }
        T--;

    }
    return 0;
}
