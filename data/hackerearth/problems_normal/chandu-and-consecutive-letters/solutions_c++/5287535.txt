#include <iostream>
#include<cstring>
#define size 1000
using namespace std;

int main()
{
    int t,i;
    char str[size];
    cin>>t;
    while(t--)
    {
     cin>>str;
     for(i=1;i<strlen(str);i++)
     {
      if(str[i-1]!=str[i])
      cout<<str[i-1];
     }
     cout<<str[i-1];
     cout<<"\n";
    }
    return 0;
}
