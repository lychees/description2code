name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
import sys
d = {}
t = int(sys.stdin.readline())
for i in xrange(t):
    name = sys.stdin.readline().strip()
    if name in d:
        d[name] = d[name]+1
    else:
        d[name] = 1

# print d
for name in sorted(d):
    print name, d[name]
