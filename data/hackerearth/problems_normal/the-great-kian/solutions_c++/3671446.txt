#include<iostream>
#include<cmath>
using namespace std;
int main()
{
		int n;
		cin>>n;
		long int * p;
		p=new(nothrow) long int [n];
		for(int i=0;i<n;i++)
		cin>>p[i];
		long int x=0,y=0,z=0;
		for(int i=0;i<n;i+=3)
		x+=p[i];
		for(int i=1;i<n;i+=3)
		y+=p[i];
		for(int i=2;i<n;i+=3)
		z+=p[i];
		cout<<x<<" "<<y<<" "<<z<<"\n";
	return 0;
}
