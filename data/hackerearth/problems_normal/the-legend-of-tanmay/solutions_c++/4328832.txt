#include<bits/stdc++.h>
#define ll long long
#define lg long
#define dd double
#define fl float
#define ch char
#define maxx 1000000007
#define scf(a) scanf("%lld",&a)
#define scdf(a) scanf("%ld",&a)
#define prt(a) printf("%lld",a)
#define prdt(a) printf("%ld",&a)
#define get(a) cin>>a
#define put(a) cout<<a
#define loop(i,a,b) for(i=a;i<=b;i++)
#define rev(i,a,b) for(i=a;i>=b;i--)
#define nxt printf("\n")
#define space printf(" ")
#define mem(s) memset(s,0,sizeof(s))
using namespace std;
int main()
{
    ll i,j,k,l,n,m,t,p,neg,mi,ma;
    get(t);
    while(t--)
    {
        neg=mi=ma=0;
        get(n);
        vector<ll>a;
        loop(i,0,n-1)
        {
            get(j);
            a.push_back(j);
            if(a[i]<0)
                neg++;
        }
        sort(a.begin(),a.end());
        if(a[0]>=0)
        {
                mi=a[0];
                rev(i,n-1,0)
                {
                    if(a[i]!=0)
                    {
                        if(i==n-1)
                            ma=a[i];
                        else
                            ma=ma*a[i];
                    }
                    else
                        break;
                }
        }
        else if(a[n-1]<=0)
        {
            if(neg%2==0)
            {
                mi=ma=a[0];
                loop(i,2,neg-2)
                {
                    mi=mi*a[i];
                }
                loop(i,1,neg-1)
                {
                    ma=ma*a[i];
                }
            }
            else
            {
                mi=1;
                loop(i,0,neg-1)
                mi=mi*a[i];
                ma=a[0];
                loop(i,1,neg-2)
                {
                        ma=ma*a[i];
                }
                if(ma<0&&a[n-1]==0)
                    ma=0;
            }
        }
        else
        {
            if(neg%2==0)
            {
                loop(i,0,n-1)
                {
                    if(a[i]!=0)
                    {
                        if(i==0)
                            ma=a[i];
                        else
                            ma*=a[i];
                    }
                }
                rev(i,n-1,0)
                {
                    if(a[i]<0)
                    {
                        break;
                    }
                }
                mi=ma/a[i];
            }
            else
            {
                loop(i,0,n-1)
                {
                    if(a[i]!=0)
                    {
                        if(i==0)
                            mi=a[i];
                        else
                            mi*=a[i];
                    }
                }
                rev(i,n-1,0)
                {
                    if(a[i]<0)
                        break;
                }
                ma=mi/a[i];
            }
        }
        put(ma);
        space;
        put(mi);
        nxt;
    }
    return 0;
}
