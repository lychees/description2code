#include <iostream>
#include <map>
using namespace std;
 
int main()
{
	map<char,int> a;
	int i,j,t,u,v,n;
	string s;
	char k;
	cin>>t;
	cin.ignore();
	for (i=1;i<=t;i++)
	{

		a[k]=0;

			}
		}

	}

}
