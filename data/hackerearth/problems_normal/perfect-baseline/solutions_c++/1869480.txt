#include <bits/stdc++.h>
#define whas(X) cerr<<#X<<" is "<<X<<endl;
#define vi vector<int>
#define pb(X) push_back(X)
#define si set<int>
#define stlp ::iterator
#define endl '\n'
typedef long long LL;
using namespace std;
char x[21][10000];
int main()
{
   int tans,t,i,n,k,j;
   cin>>t;
   while(t--)
   {
      double places[21][26];
      cin>>n>>k;
      memset(places,0,sizeof(places));
      getchar();
      for(i=0;i<n;i++)
      {
         for(j=0;j<k;j++)
            x[j][i] = getchar(),places[j][x[j][i]-'a']++;
         getchar();
      }
      char ans[21];
      fill(ans,ans+21,'a');
      for(i=0;i<k;i++)
      {
         int mink = -1,mans=0;
         for(j=0;j<26;j++)
         {
            tans = 0;
            int k;
            for(k=0;k<26;k++)
               tans += places[i][k]*(abs(j-k));
            if(tans<mans || mink == -1)
               mans = tans,mink = j;
         }
         ans[i] += mink;
      }
      ans[k]='\0';
      printf("%s\n",ans);
   }
   return 0;
}
