#include<bits/stdc++.h>
using namespace std;
bool isprime(int n){
	if(n<=1)
		return false;
	int val = sqrt(n);
	for(int i=2;i<=val;i++)
		if(n%i==0)
			return false;
	return true;
}
int main()
{
	int t,k;
	string s;
	int i,j,l,high,low;
	cin>>t;
	while(t--)
	{
		high=0;
		low=0;
		cin>>s;
		l=s.length();
		for(k=1;k<l-1;k++)
		{
			
			i=k-1;
			j=k;
			while(i>=0 && s[j]!='\0' && s[i]==s[j])
			{
				if((high-low)<=j-i)
				{
					(high)=j;
					(low)=(i);
				}
				i=i-1;
				j=j+1;
			}
			i=k-1;
			j=k+1;
				while(i>=0 && s[j]!='\0' && s[i]==s[j])
			{
				if((high-low)<=j-i)
				{
					(high)=j;
					(low)=(i);
				}
				i=i-1;
				j=j+1;
			}
		}
	//	cout<<high-low;
			if(isprime(high-low+1))
			{
				cout<<"YES"<<'\n';
			}
			else
			{
				cout<<"NO"<<'\n';
			}
	}
	return 0;
}
