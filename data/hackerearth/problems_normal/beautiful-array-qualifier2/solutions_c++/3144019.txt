#include <bits/stdc++.h>
using namespace std;

const int LIM = 500000; // primes>=4e4

int N,K,M;
bool bad[21][LIM];

int main() {
	scanf("%d%d%d",&N,&K,&M);
	K = min(K,20);
	for (int i=0;i<M;i++) {
		int b;
		scanf("%d",&b);
		assert(b!=1); // do tests not make 1 bad
		if (b<LIM) bad[K][b] = 1;
	}
	bad[0][1] = 1;
	bad[K][0] = 1;
	int on = 0;
	for (int i=0;i<N;i++) {
		bool up = 0;
		while (bad[K][on]) on+=1, up = 1;
		printf("%d ",on);
		if (up) {
			for (int j=1;j<=K;j++) {
				for (int k=0;k<LIM;k+=on) {
					bad[j][k]|=bad[j-1][k/on];
				}
			}
		}
	}
	printf("\n");
}
