n = int(raw_input())
cities = {}
for i in range(n):
    city = raw_input()
    cities[city] = i

# print cities
cost = []
row = []
for i in range(n):
    row = [int(i) for i in raw_input().split()]
    cost.append(row)


vid = 0
tot = 0
p = int(raw_input())
for k in range(p):
    d = raw_input()
    v = cities[d]
    tot += cost[vid][v]
    vid = v

print tot
