#define MOD 1000000009
vector<int> prime;
long long a[555];
void init()
{
	int i,j;
	prime.push_back(2);
	for(i=3;i<500;i++)
	{
		for(j=2;j<i;j++)
		{
			if(i%j==0)
				break;
		}
		if(i==j)
		{
			prime.push_back(i);
		}
	}
}
int cal(int t)
{
	int i,n=prime.size();
	int re=0;
	for(i=0;i<n;i++)
	{
		if(t>=prime[i])
		{
			re++;
		}
	}
	return re;
}
long long mypow(int t)
{
	long long re=1;
	long long x=2;
	while(t)
	{
		if(t&1)
			re*=x;
		x*=x;
		t/=2;
	}
	return re;
}
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int i,n;
	init();
	cin>>n;
	for(i=2;i<=n;i++)
	{
		a[i]=a[i-1]+mypow(cal(i)-1);
	}
	cout<<a[n]<<endl;
//	system("pause");
	return 0;
}
