#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int q;
    cin>>q;
    long long *arr,*sum;
    arr=new long long[n];
    sum=new long long[n];
    cin>>arr[0];
    sum[0]=arr[0];
    for(int i=1;i<n;i++)
    {
    	cin>>arr[i];
    	sum[i]=sum[i-1]+arr[i];	
    }
    for(int i=0;i<q;i++)
    {
    	int g1,g2,p1,p2;
    	scanf("%d %d",&g1,&p1);
    	scanf("%d %d",&g2,&p2);
    	if(p1>g2)
    	{
    		printf("%ld\n",sum[g2]-sum[g1]+arr[g1]+sum[p2]-sum[p1]+arr[p1]);
    	}
    	else if(p1>=g1)
    	{
    		if(g2<p2)
    			printf("%ld\n",sum[p2]-sum[g1]+arr[g1]);
    		else
    			printf("%ld\n",sum[g2]-sum[g1]+arr[g1]);
    	}
    	else if(p2<g1)
    		printf("%ld\n",sum[p2]-sum[p1]+arr[p1]+sum[g2]-sum[g1]+arr[g1]);
    	else if(g2<p2)
    		printf("%ld\n",sum[p2]-sum[p1]+arr[p1]);
    	else
    		printf("%ld\n",sum[g2]-sum[p1]+arr[p1]);
    }
    return 0;
}
