#include<iostream>
#include<math.h>
using namespace std;

int main()
{
	int n,limit;
	cin>>n;
	limit=pow(2,n);
	int matrix[limit+1][limit+1];
	int winner[limit+1];
	for(int i=2;i<=limit;i++)
		{
			for(int j=1;j<i;j++)
				cin>>matrix[i][j];
		}
	for(int i=1;i<=pow(2,n);i++)
		winner[i]=i;
	int k=2;
	int m;
	for(int k=1;k<=n;k++)
	{
		m=1;
		for(int i=1;i<=pow(2,n)/k;i+=2)
		{
			if(matrix[winner[i+1]][winner[i]]==1)
			{
				winner[m]=winner[i+1];
			}
			else
				winner[m]=winner[i];
			m++;
		}	
	}
	cout<<winner[1]<<endl;;
	return 0;
}
