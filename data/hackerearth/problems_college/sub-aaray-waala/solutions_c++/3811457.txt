#include<bits/stdc++.h>
using namespace std;
long long int power(long long int x, long long int y)
{
    long long int temp;
    if( y == 0)
        return 1;
    temp = (power(x, y/2))%1000000007;
    if (y%2 == 0)
        return ((temp%1000000007)*(temp%1000000007))%1000000007;
    else
        return ((((x%1000000007)*(temp%1000000007))%1000000007)*(temp%1000000007))%1000000007;
}
int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,p;
		cin>>n;
		long long int arr[n],ans;
		set<long long int> s;
		for(long int i=0;i<n;i++)
		{
			cin>>arr[i];
			s.insert(arr[i]);
		}
		p=s.size();
		ans=power(n,p);
		ans%=1000000007;
		cout<<ans<<endl;
		
	}
	return 0;
	
}
