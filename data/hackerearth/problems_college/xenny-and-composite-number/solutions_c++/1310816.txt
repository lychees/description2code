#include <iostream>
#include<stdbool.h>
#include<string.h>
bool prime[100001];
using namespace std;
void fun()
{
	memset(prime,true,sizeof(prime));
	

	
	for(int i=2;i<100001;i++)
	{
		if(prime[i])
		{
			int j=2*i;
			while(j<100001)
			{
				prime[j]=false;
				j+=i;
			}
		}
	}
}
int main()
{
    long t,n;
    fun();
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	n++;
    	while(prime[n])
    	  n++;
    	cout<<n<<endl;
    }
    return 0;
}
