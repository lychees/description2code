#include <stdio.h>
#include <string.h>

int main()
{
	char s[51000];
	int l, r, k, d, n, q, x;
	scanf("%d %d %s", &n, &q, s);
	
	for (; q > 0; --q) {
		int m['z'] = {0};
		scanf("%d %d %d", &l, &r, &k);
		x = 0;
		for (--l; l < r; ++l) {
			m[s[l]]++;
			x++;
		}
		if (x < k) {
			printf("Out of range\n");
		} else {
			for (d = 'a'; d <= 'z'; ++d) {
				if ((k -= m[d]) <= 0)
				    break;
			}
			printf("%c\n", d);
		}
	}
	
    return 0;
}
