#include <bits/stdc++.h>

using namespace std;

const int MAXN = 100000 + 2;

int g[100000 + 10];

int main()
{
	//re-submit
    g[1] = 0;
    for(int i = 2; i <= 100000; i++) {
        set<int> a;
        a.insert(0);
        for(int j = 2; j <= (int)(sqrt(i)); j++) {
            if (i % j == 0) {
                a.insert(g[j]);
                a.insert(g[i / j]);
            }
        }
        g[i] = -1;
        int j = -1;
        for(set<int>::iterator it = a.begin(); it != a.end(); it++) {
            j++;
            if (j != *it) {
                g[i] = j;
                break;
            }
        }
        if (g[i] == -1) g[i] = j + 1;
    }
    int x = 0;
    int n;
    cin >> n;
    for(int i = 0; i < n; i++) {
        int j;
        cin >> j;
        x ^= g[j];
    }
    if (x != 0) cout << "ALICE" << endl;
    else cout << "BOB" << endl;
}
