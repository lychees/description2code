#include <bits/stdc++.h>
#include <iostream>
using namespace std;
int dp[1000006];
int print_each_digit(int x)
{
    if(x >= 10)
       print_each_digit(x / 10);

    int digit = x % 10;

    return digit ;
}
int main()
{
	memset(dp,0,sizeof(dp));
	for(int i = 1 ; pow(i,i) <= 1000000 ; i++)
	{
		dp[int(pow(i,i))] =1;
	}
	dp[0] = 0;

	for(int i = 2 ; i <= 1000000; i++)
	{
		vector<int> V;
	int x =i;
		while(x>0)
		{
			int q = x%10;
			x= x/10;
			V.push_back(q);
		}
		
		while(!V.empty())
		{
			int p = pow(V.back(),2) ;
			V.pop_back();
			
			if(dp[ i - p ] == 1)
			{
				dp[i]= 1;
				break;
			}
		}
	}	
	int T;
	cin>>T;
	while(T--)
	{
		int N;
		cin>>N;
		if(dp[N]== 1)
			cout<<"Yes"<<endl;
		else cout<<"No"<<endl;
	}

    return 0;
}
