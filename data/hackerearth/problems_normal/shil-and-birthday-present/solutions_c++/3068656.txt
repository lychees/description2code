#include <set>
#include <stdio.h>
using namespace std;

int main()
{  int n,a[1000001];
   scanf("%d",&n);
   
   int i=0;
   int x;
   int m=0;
   while(i!=1000001)
   {a[i]=0;
   	 i++;
   	
   	}
   i=0;
   while(i!=n)
   {
   	scanf("%d",&x);
   	if(m<x)m=x;
   	a[x]=1;
   	i++;
   	}
   	i=0;
   	long long ans =0;
   	  while(i!=m+1)
   {if(a[i])ans++;
   	 i++;
   	
   	}
   	ans=(ans*(ans-1))/2;
   printf("%ld",ans);
   
    return 0;
}
