#include <bits/stdc++.h>
using namespace std;
int a[26],b[26];
int main()
{
int t,count,flag;
string s,c;
cin>>t;
while(t--)
{
flag=0;
count=0;
cin>>s>>c;
for(int i=0;s[i]!='\0';i++)
a[s[i]-'a']++;

for(int i=0;c[i]!='\0';i++)
b[c[i]-'a']++;
for(int i=0;i<26;i++)
{
if(a[i]>b[i])
{
flag=1;
}
else if(b[i]&&a[i]<b[i])
{
count++;
}

a[i]=0;
b[i]=0;
}
if(flag)
cout<<"-1\n";
else
cout<<count<<"\n";
}
return 0;
}
