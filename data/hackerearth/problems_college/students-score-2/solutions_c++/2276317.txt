#include <iostream>
using namespace std;
int max(int x, int y)
{ return (y > x)? y : x; }
 
int maxSubArraySum(int a[], int size)
{
   int max_so_far = a[0], i;
   int curr_max = a[0];
 
   for (i = 1; i < size; i++)
   {
        curr_max = max(a[i], curr_max+a[i]);
        max_so_far = max(max_so_far, curr_max);
   }
   return max_so_far;
}
int main()
{
	int n;
	scanf("%d",&n);
	int a[100000];
	for(int i =0;i<n;i++)
	scanf("%d",&a[i]);
	printf("%d",maxSubArraySum(a,n));
   // cout << "Hello World!" << endl;
    return 0;
}
