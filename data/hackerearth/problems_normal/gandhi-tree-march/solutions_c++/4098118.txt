#include <bits/stdc++.h>
#define MAX 11111
 
using namespace std;
 
vector<int> X[26];
 
string S;
int N;
int pos;
void compute(int row)
{
	if (row >= N) return;
	if (S[pos] == '.')  
	{ pos++; return; }
	else
	{
		int x = S[pos] - 'a';
		X[x].push_back(row); pos += 2;
		compute(row - 1); compute(row + 1);	
	}
	pos++;
}
 
int main()
{
    int T;
    cin >> T;
    while (T--)
    {
    	for (int i = 0; i < 26; i++) X[i].clear();
    	int id = 0;
    	int C;
    	cin >> C >> S; N = S.size();
    	pos = 0;
    	compute(0);
    	string ans = "";
    	for (int i = 0; i < 26; i++)
    	{
    		for (int j = 0; j < X[i].size(); j++)
    			if (X[i][j] == C) ans += char('a' + i);
    	}
    	if (ans.size() == 0) ans = "Common Gandhijee!";
     	cout << ans << endl;
    }
}
