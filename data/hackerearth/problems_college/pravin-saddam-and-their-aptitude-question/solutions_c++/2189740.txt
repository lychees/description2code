#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,sum,a,b;
	cin>>t;
	while(t--)
	{
		sum=0;
		cin>>a>>b;
		for(int i=a;i<=b;i++)
		{
			if(i==1)
			sum+=1;
			else if(i==2)
			sum+=2;
			else if(i==3)
			sum+=6;
			else if(i==4)
			sum+=4;
		}
		cout<<sum%10<<endl;
		
	}
	
	
	
	return 0;
}
