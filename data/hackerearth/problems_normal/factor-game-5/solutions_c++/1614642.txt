#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

int res=0,n,i,x,xp,d;

int main()
{
    cin>>n;
    while (n--) {
    	cin>>x; d=0; i=2; xp=sqrt(x);
    	while (x>1&&i<=xp) {
    		while (x%i==0) {x/=i;d++;}
    		i++;
    	}
    	if (x>1) d++;
    	res^=d;
    }
    if (res) cout<<"ALICE"; else cout<<"BOB";
}
