#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	map<int,int>maaap;
	map<int ,int >::iterator it;
	for(int i=1;i<=n;++i)
	{
		int temp;
		cin>>temp;
		maaap[temp]++;
	}
	bool flag=false;
	int maxi=INT_MIN;
	for(it=maaap.begin();it!=maaap.end();++it)
	{
		if(it->second==1)
		{
			if(it->first>maxi)
			{
				maxi=it->first;
			}
			flag=true;
		}
	}
	(flag==false)?printf("-1\n"):printf("%d\n",maxi);
	return 0;
}
