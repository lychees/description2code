#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;

int main(){
	int i,sum=0;
	long int n;
	cin>>n;
	int arr[100000];
	for(i=0; i<n; i++){
		cin>>arr[i];
		if((sum + arr[i]) >= 0){
			arr[i] = sum + arr[i];
		}
		else {
			arr[i] = 0;
		}
		sum = arr[i];
	}
	
	sort(arr, arr+n);
	cout<<arr[n-1];
	
}
