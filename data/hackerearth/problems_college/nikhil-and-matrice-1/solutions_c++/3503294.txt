#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
   {
   	int n,s,m;
   	cin>>s;
   	n=s,m=s;
   	int a[n][n],c=1;
    int i,j, k = 0, l = 0;
    while (k < m && l < n)
    {
        /* Print the first row from the remaining rows */
        for (i = l; i < n; ++i)
        {
            //printf("%d ", a[k][i]);
            a[k][i]=c++;
        }
        k++;
 
        /* Print the last column from the remaining columns */
        for (i = k; i < m; ++i)
        {
           // printf("%d ", a[i][n-1]);
           a[i][n-1]=c++;
        }
        n--;
 
        /* Print the last row from the remaining rows */
        if ( k < m)
        {
            for (i = n-1; i >= l; --i)
            {
               // printf("%d ", a[m-1][i]);
               a[m-1][i]=c++;
            }
            m--;
        }
 
        /* Print the first column from the remaining columns */
        if (l < n)
        {
            for (i = m-1; i >= k; --i)
            {
                //printf("%d ", a[i][l]);
                a[i][l]=c++;
            }
            l++;    
        }        
    }

    for(i=0;i<s;i++)
    {
    	for(j=0;j<s;j++)
    	cout<<a[i][j];
    	cout<<"\n";
    }
    cout<<"\n";
	}
}
