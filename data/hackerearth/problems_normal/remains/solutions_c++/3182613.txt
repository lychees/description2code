#include <iostream>
#include <cstdio>

#define ll long long int
using namespace std;



int main()
{
    ll t,x,y,n,temp;
    ll i,count;
    int flag;
    scanf("%lld",&t);
    while(t--){
    	scanf("%lld%lld%lld",&x,&y,&n);
    	count=x+y;
    	flag=0;
    	
    	for(i=3;i<=n;i++){
    		if(x==y){
    			count+= ((n-i+1)/3)*2*x;
    			if((n-i+1)%3 == 2) count+=x;
    			break;
    		}
    		temp=y;
    		y=abs(x-y);
    		x=temp;
    		count+=y;
    	}
    	
    	printf("%lld\n",count);
    }
    return 0;
}
