def factors(n):    
    return len(set(reduce(list.__add__, 
                ([i, n//i] for i in range(1, int(n**0.5) + 1) if n % i == 0))))
    

t=input()
while t:
    t-=1
    x=input()
    if x%2==1 :
        print "0"
    else:
        x=x/2
        print factors(x)
