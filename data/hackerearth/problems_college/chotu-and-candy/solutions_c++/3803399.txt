#include <iostream>
#include <utility>
#include <algorithm>

using namespace std;

pair <int, pair <int, int> > p[100005];

int main()
{
	int n, q, i, j, k, x, s, c;
	cin >> n >> q;
	
	for (i = 0; i < n; ++i) {
		cin >> x >> s >> c;
		p[i] = make_pair(-1 * s, make_pair(c, x));
	}
	
	sort(p, p + n);
	
	for (i = 0; i < q; ++i) {
		cin >> x;
		cout << p[x - 1].second.second << endl;
	}
	
	return 0;
}
