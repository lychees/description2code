#include <iostream>
#include<set>
using namespace std;

int recur(int x,int a[],set<int> &s){
   if(a[x]==-1)
     return 1;
   else if(s.find(a[x])!=s.end())
     return 0;
   else s.insert(a[x]);
   
   recur(a[x],a,s);
}

int main()
{   ios::sync_with_stdio(0);
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;++i)
    		cin>>a[i];
       set<int> s;
       if(recur(0,a,s))
       printf("Win\n");
       else  printf("Lose\n");
      
    }
    return 0;
}
