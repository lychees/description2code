#include <bits/stdc++.h>
#include <iostream>
using namespace std;

int main() {
  int xp, yp, xv, yv;
  scanf("%d%d%d%d", &xp, &yp, &xv, &yv);
  int max = xv > yv ? xv : yv;
  if (xp + yp <= max)
    printf("Polycarp");
  else if (xp <= xv && yp <= yv) {
    printf("Polycarp");
  } else {
    printf("Vasiliy");
  }
  return 0;
}
