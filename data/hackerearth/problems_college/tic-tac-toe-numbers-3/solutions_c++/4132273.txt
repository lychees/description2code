ll fibosum[1000001];
l fibo[1000001];
bool lol[1000001];
void FIBO(ll N)
{
    fibo[0]=0;
    fibo[1]=1;
    lol[0]=lol[1]=1;
    l i=2;
    while(1)
    {
        fibo[i]=fibo[i-1]+fibo[i-2];
        if(fibo[i]>N)
            break;
        lol[fibo[i]]=1;
        i++;
    }
    fibosum[0]=0;
    FORAB(i,1,N)
    {
        if(lol[i])
        {
            fibosum[i]=fibosum[i-1]+i;
        }
        else
            fibosum[i]=fibosum[i-1];
    }
}

bool primecheck(ll N)
{
    ll i;
    for(i=2;i*i<=N;++i)
        if(N%i==0)
            return 0;
    return 1;
}
