import re
     
test_case = raw_input("")

if test_case.isdigit():
	test_case = int(test_case)
	if test_case < 1 or test_case > 10:
		print "Invalid Test"
	else:
		for i in range(0,test_case):
			test_string = raw_input("")
			if len(test_string) > 100 or len(test_string) < 1:
				print 'Invalid Input'
			elif ' ' in test_string:
				print 'Invalid Input'
			elif (re.search(r'[a-zA-Z]', test_string) == None):
				print 'Invalid Input'
			else:
				print min(len(re.findall(r'[a-z]', test_string)), len(re.findall(r'[A-Z]', test_string)))
else:
	print "Invalid Test"
