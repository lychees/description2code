#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,a[100002],inv[100002],i;
    cin>>t;
    while(t--) {
    	memset(inv,0,sizeof(inv));
    	cin>>n;
    	for(i=1;i<=n;++i) 
    		cin>>a[i];
    	for(i=1;i<=n;++i) {
    		inv[a[a[i]]]=a[i];
    	}
    	for(i=1;i<=n;++i) {
    		if(inv[i]!=a[i]) {
    			cout<<"not inverse\n";
    			break;
    		}
    	}
    	
    	//cout<<i<<" ";
    	if(i==n+1)
    		cout<<"inverse\n";
    	//else
    		//cout<<"not inverse\n";
    }
    
    return 0;
}
