#include<iostream>
#include<algorithm>
using namespace std;
 
int main()
{
    int T,n,i,j,count,sum,a[101];
    
    cin >> T;
    while(T--)
    {
    	int b[10001]={0};
        cin >> n;
        count=1;
        sum=0;
        for(i=0 ; i<n ; i++)
        {
            cin >> a[i];
            sum+=a[i];
        }
        b[0]=1;
        for(i=0 ; i<n ; i++)
        {
            for(j=sum ; j>=a[i] ; j--)
            {
                if(b[j-a[i]]==1 && b[j]!=1)
                {
                    count++;
                    b[j]++;
                }
            }   
        }
        cout << count << endl;
    }
    return 0;
}
