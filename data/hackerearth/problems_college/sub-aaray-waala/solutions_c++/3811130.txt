//fast power algorithm:(pow(a,n)%10^9+7)
long long fast_pow(int a,int n)
{
    long long result = 1;
    long long power = n;
    long long value = a;
    while(power>0)
    {
        if(power&1)
		{
			result = result*value;
            result = result%1000000007;
		}
        value = value*value;
        value = value%1000000007;
        power /= 2;
        //power >>= 1;
    }
    return result;
}

int main() {
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while ( t-- ) {
		int n;
		cin>>n;
		vector<int> A(n);
		map<int,int> m;
		int d=0;
		for ( int i=0;i<n;i++ ) {
			cin>>A[i];
			if ( m[A[i]]==0 ) {
				d++;
				m[A[i]]=1;
			}
		}
        ll ans = fast_pow(n,d);
        cout<<ans<<"\n";
	}
	return 0;
}
