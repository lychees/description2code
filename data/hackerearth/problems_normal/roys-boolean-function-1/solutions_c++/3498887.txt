#include<bits/stdc++.h>
using namespace std;
bool prime[1000001];
void seive(){
	memset(prime, true, sizeof(prime));
	prime[0] = prime[1] = false;
	for(int i =2; i*i <= 1000001; i++){
		if(prime[i]){
			for(int j = i*i; j <= 1000001; j += i){
				prime[j] = false;
			}
		}
	}
}
int gcd(int a, int b) {
	if(b == 0)
	return a;
	return gcd(b, a%b); 
}
int main(){
	seive();
	int t, n, c;
	cin >> t;
	while(t--){
	cin >> n;
	c = 0;
	for(int i = 1; i <= n; i++){
		if(gcd(i,n)==1){
			c++;
		}
	}
	//cout << "c= " << c;
	if(prime[c]){
		cout <<"TRUE\n";
	}else cout<<"FALSE\n";
	}
}
