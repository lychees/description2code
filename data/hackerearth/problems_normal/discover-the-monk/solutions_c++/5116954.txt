#include <iostream>
#include <algorithm>
using namespace std;
bool binSearch(int *A,int top,int mid,int bot,int elem) {
	if (top>bot) return false;
	if (A[mid]==elem) return true;
	else if (A[mid]<elem) {
		top=mid+1;
		mid=(top+bot)/2;
		return binSearch(A,top,mid,bot,elem);
	} else {
		bot=mid-1;
		mid=(top+bot)/2;
		return binSearch(A,top,mid,bot,elem);
	}
}
int main()
{
    //cout << "Hello World!" << endl;
    int N,Q,*A,i;
    cin>>N>>Q;
    A=new int[N];
    for(i=0;i<N;i++) cin>>A[i];
    sort(A,A+N);
    while(Q--) {
    	cin>>i;
    	if (binSearch(A,0,(N-1)/2,N-1,i)) cout<<"YES\n";
    	else cout<<"NO\n";
    }
    delete[] A;
    return 0;
}
