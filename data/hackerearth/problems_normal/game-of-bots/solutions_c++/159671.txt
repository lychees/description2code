#include <iostream>
#include<cstring>
using namespace std;

struct X
{
	int idx;
	char c;
};


char c1[1000001],c2[1000001];
X st[1000001];
int main()
{
     
    int t,i,j,l1,l2;
    X a;
    
    cin>>t;
    while(t--)
    {	int flag = 0;
    	cin>>c1>>c2;
    	
    	l1 = strlen(c1);
    	int top = -1;
    	for(i = 0 ; i<l1 ; i++)
    	{
    		if(c1[i] == '#')
    			continue;
    		else
    		{	a.c = c1[i];
    			a.idx = i;
    			st[++top] = a;
    		}
    	}
    	
    	for(i = l1-1; i>=0; i--)
    	{
    		if(c2[i] == '#')
    			continue;
    		else
    			if(c2[i] == 'A' && c2[i] == st[top].c)
    			{
    				if(i>st[top].idx)
    				{
    					flag=1;
    					break;
    				}
    				else
    				{
    					top--;
    				}
    				
    			}
    			else if(c2[i] == 'B' && c2[i] == st[top].c)
    			{
    				if(i < st[top].idx)
    				{
    					flag=1;
    					break;
    				}
    				else
    				{
    					top--;
    				}
    				
    			}
    			else
    			{
    				flag = 1;
    				break;
    			}
    	}	
    	
    	if(flag == 1)
    		cout<<"No\n";
    	else
    		cout<<"Yes\n";
    	
    }
    
    return 0;
}
