def getline():
    return stdin.readline()


def getLineAs(tp):
    return map(tp, getline().split())


def array(n, init=0):
    return [deepcopy(init) for i in range(n)]


def count_if(lst, pred):
    ret = 0
    for i in lst:
        if pred(i):
            ret = ret + 1
    return ret


def toSepStr(lst):
    if len(lst):
        return "".join([str(item) + " " for item in lst])[:-1]
    else:
        return ""
