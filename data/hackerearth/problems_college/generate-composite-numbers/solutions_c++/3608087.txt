#include<bits/stdc++.h>
#define N 10005
int a[N];
using namespace std;
int main()
{
    int i,j;
    a[4]=1;
    a[6]=1;
    a[8] = 1;
    for(i = 2; i<=N; i++)
    {
        if(a[i]==0)
        {
            for(j = 2*i; j<=N; j+=i)
                a[j] = 1;
        }
    }
    int n;
    while(scanf("%d",&n)==1)
    {
        for(i = 1; i<=n; i++)
            if(a[i] == 1)
                printf("%d ",i);
        printf("\n");
    }
    return 0;
}
