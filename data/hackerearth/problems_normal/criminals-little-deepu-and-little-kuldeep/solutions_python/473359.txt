testcase = int(raw_input())
while testcase:
	testcase-=1
	n = int(raw_input())
	box = []
	opbox = [100]
	for i in range(0,n):
		box.append(raw_input())
	box.sort()
	count = 0
	opbox[count] = box[0]
	for i in range(1,n):
		pointer = 0
		for j in range(0,count+1):
			pointer = j
			if box[i] != opbox[j]:
				opbox[j] = box[i]			
				break
			if pointer == count:	
				count+=1
				opbox.append(box[i])
				pointer+=1	
	print count+1
