#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

ll dp[55][50];
void precompute()
{
  for(int i=1;i<=50;i++)
  {
  	for(int j=1;j<=10;j++)
  	{
  		if(i == j)
  		{
  		  dp[i][j] = pow(2,i);
  		  dp[i][j] -= 1;
  		 // cout<<"i : "<<i<<"  j : "<<j<<"   "<<dp[i][j]<<endl;
  	    }
  		else if(j > i)
  		dp[i][j] = pow(2,i);
  		else
  		{
  			int x = j;
  			int k = i-1;
  			while(x > 0)
  			{
  				dp[i][j] += dp[k--][j];
  				x--;
  			}
  		}
  	}
  }
 /* for(int i=1;i<=10;i++)
  {
  	for(int j=1;j<=10;j++)cout<<dp[i][j]<<"   ";cout<<endl;
  }*/
}
int main()
{
	precompute();
	int t;cin>>t;
    while(t--)
    {
    	int m,n;
        scanf("%d %d",&m,&n);
        printf("%lld\n",dp[n][m]);
    }
    return 0;
}
