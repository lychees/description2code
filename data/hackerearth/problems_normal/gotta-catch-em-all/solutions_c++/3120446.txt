#include <iostream>
#include<algorithm>
using namespace std;
int main()
{
    int k;
    cin>>k;
    long days[k];
    for(int i=0;i<k;i++)cin>>days[i];
    sort(days,days+k);
    int j=0;
    long ans=0;
    for(int i=k-1;i>=0;i--)
    {
    	days[i]+=j;
    	j++;
    	ans=max(ans,days[i]);
    }
    cout<<(ans+2)<<endl;
    return 0;
}
