#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
     int t;
    cin>>t;
    for(;t--;)
    {
    	int n,k,i,m=0;
    	cin>>n>>k;
    	int a[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		if(a[i]<=0)
    		m++;
    	}
    	if(m<k)
    	cout<<("YES\n");
    	else
    	cout<<("NO\n");
    }
    return 0;
}
