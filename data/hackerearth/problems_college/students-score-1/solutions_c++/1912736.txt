#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
   int n,m;vector<int> v;vector<int> v2;
   cin>>n;
   for(int i=0;i<n;i++){
   	cin>>m;
   	v.push_back(m);
   }
   int j=0;
   while(j<n-1){
   	int sum=v[j];
   	v2.push_back(sum);
   	for(int i=j+1;i<n;i++){
   		sum+=v[i];
   		v2.push_back(sum);
	   }
   	sort(v2.begin(),v2.end());
   	v[j]=v2[v2.size()-1];
    v2.clear();
	j++;
   }
   sort(v.begin(),v.end());
   cout<<v[v.size()-1];
}
