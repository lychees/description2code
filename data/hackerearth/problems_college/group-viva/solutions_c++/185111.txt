#include<iostream>
#include<algorithm>
#include<cmath>

using namespace std;

int chkp(int a,int x)
{
	int y;
	for(y=2;y<=a;++y)
		{
			if(x%y==0) return 0;
		}
		return 1;

}

int gen(int p[],int b)
{
	int k=0,x;
	for(x=2;x<=b;++x)
	{
		if(chkp(sqrt(x),x)==1) p[k++]=x;
	}
	return k;
}

int gcd ( int a, int b )
{
  int c;
  while ( a != 0 ) {
     c = a; a = b%a;  b = c;
  }
  return b;
}

int main()
{
	int tx,tc,l,n,r,m,x,p[100000],k,g,cn=0,s=0;
	k=gen(p,sqrt(1000000000));
	cin>>tx;
	for(tc=0;tc<tx;++tc)
	{
		cin>>r>>l;
		n=l-1;
		m=sqrt(n);
		cn=0;
		s=1;
		for(x=0;p[x]<=m;++x)
		{
			while(n%p[x]==0) {++cn;n/=p[x];}
			s*=cn+1;
			cn=0;
		}
		if(n>m) s*=2;
		s=r-s;
		g=gcd(s,r);
		cout<<s/g<<"/"<<r/g<<endl;
	}
	//for(x=0;x<k;++x) cout<<p[x]<<" ";
	return 0;
}
