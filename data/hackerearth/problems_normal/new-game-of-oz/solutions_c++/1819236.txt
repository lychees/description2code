#include <iostream>
#include <cstring>
using namespace std;

int main()
{

    int a[1010];
    
    int i,n,t,x,ans;
    
    cin>>t;
    
    while(t--)
    {
    	cin >> n;
    	ans = 0;
    	memset(a,0,sizeof(a));
    	for(i=0;i<n;i++)
    	{
    		cin>>x;
    		a[x] = 1;
    	}
    	
    	for(i=1;i<=1000;i++)
    	{
    		if(a[i])
    		{
    			a[i] = 0;
    			a[i+1] = 0;
    			ans++;
    		}
    	}
    	cout << ans << endl;
    }
    return 0;

}
