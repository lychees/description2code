#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
ll a[200005];
ll b[200005];
int main()
{

    int n,i,j,k;cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>a[i]>>b[i];
    }
      sort(a,a+n);
      sort(b,b+n);
      ll sum=0;
      int x=1;
      for(i=1;i<n;i++)
      {
          sum=((sum%mod+(x%mod*a[i]%mod)%mod))%mod;
          x++;
      }
      x=1;
      for(i=n-2;i>=0;i--)
      {
         // sum=sum-x*a[i];
                   sum=((sum%mod-(x%mod*a[i]%mod)%mod+mod))%mod;

          x++;
      }
      x=1;
      for(i=1;i<n;i++)
      {
          sum=((sum%mod+(x%mod*b[i]%mod)%mod))%mod;
          x++;
      }
      x=1;
      for(i=n-2;i>=0;i--)
      {
         // sum=sum-x*a[i];
                   sum=((sum%mod-(x%mod*b[i]%mod)%mod+mod))%mod;

          x++;
      }
      cout<<sum<<endl;

}
