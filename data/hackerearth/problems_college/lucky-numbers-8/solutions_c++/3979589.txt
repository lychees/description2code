#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	int b[n];
    	int k=n-1;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		b[k--]=a[i];
    	}
    	
    	int max=0;
    	int count=0;
    	for(int i=0;i<n;i++)
    	{
    		if(i==0)
    		{
    			max=b[i];
    			count++;
    		}
    		if(max<b[i])
    		{
    			max=b[i];
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
}
