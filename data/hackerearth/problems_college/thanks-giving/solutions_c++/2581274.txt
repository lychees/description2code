#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	char a[4005];
	cin>>a;
	int l=strlen(a);
	int freq[30]={0};
	char ch;
	for(int i=0;i<l;i++)
	{
		ch=a[i];
		freq[ch-97]++;
	}	
	int maxindex=-1;
	int max=0;
	for(int i=0;i<30;i++)
	{
		if(freq[i]>max)
		{
			maxindex=i;
			max=freq[i];
		}	
	}	
	ch=97+maxindex;

	int flag1=0;
	int flag2=0;
	string b,c;
	b="head rush";
	c="cubing club";
	int l1=b.size();
	for(int i=0;i<l1;i++)
	{
		if(b[i]==ch)
			flag1=1;
	}	
	int l2=c.size();
	for(int i=0;i<l2;i++)
	{
		if(c[i]==ch)
			flag2=1;
	}
	if(flag1==1 && flag2==1)
	{
		cout<<"Thanks both\n";
	}	
	else if(flag1==1)
	{
		cout<<"Thanks head rush\n";
	}	
	else if(flag2==1)
	{
		cout<<"Thanks cubing club\n";
	}
	else
		cout<<"none\n";


}
