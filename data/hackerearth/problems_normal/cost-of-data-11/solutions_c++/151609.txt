#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
#include<utility>
#include<climits>
#include<cstdlib>
#include<algorithm>
#include<map>
#include<string>
using namespace std;
int main()
{
	int n,x=0,i,j;
	cin>>n;
	string a[n];
	for(i=0;i<n;i++)
	cin>>a[i];
	sort(a,a+n);
	x=a[0].size();
	//cout<<x<<endl;
	for(i=1;i<n;i++)
	{
		for(j=0;a[i].size();j++)
		{
			if((a[i-1].size())>=j&&a[i][j]==a[i-1][j])
			continue;
			else
			{
				x+=(a[i].size()-j);
				//cout<<i<<" "<<x+1<<endl;
				break;
			}	
		}
		//cout<<x+1<<endl;
	}
	cout<<x+1<<endl;
	return 0;
}
