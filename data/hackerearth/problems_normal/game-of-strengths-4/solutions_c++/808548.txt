using namespace std;

int main()
{
   int t;
   long long n,a[100000],i,s,j;
   scanf("%d",&t);
   while(t--)
   {
       s=0;
       scanf("%lld",&n);
       for(i=0;i<n;i++)
       {
            scanf("%lld",&a[i]);
       }
       sort(a,a+n);
       i=n-1;
       j=i;
       while(i>=0)
       {
           s=s+(i*(a[i]-a[j-i]))%1000000007;
           i--;
       }
       s=s%1000000007;
      // cout<<a[n-1]*s<<endl;

       printf("%llu\n",a[n-1]*s%1000000007);
   }
    return 0;
}
