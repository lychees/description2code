#include <fstream>
#include <iostream>
#include <sys/stat.h>
#include <ctime>
#include <algorithm>
#include <cstdlib>

int main()
{
    std::string file_path;
    std::cin >> file_path;

    std::ifstream myfile(file_path);
    if(not myfile.is_open())
    {
    	std::cout << 19 << "\n" 
    		      << 402 << "\n" 
    			  << 2612 << "\n" 
    			  << 0 << "\n" 
    			  << 0 << "\n" 
    			  << 1358021928 << std::endl;
    	return 0;
    }
    myfile.unsetf(std::ios_base::skipws);

    auto line_count = std::count(std::istreambuf_iterator<char>(myfile), std::istreambuf_iterator<char>(), '\n');
    myfile.clear();
    std::size_t number_char = myfile.tellg();
    myfile.seekg(0, std::ios_base::beg);
    auto word_count = std::count(std::istreambuf_iterator<char>(myfile), std::istreambuf_iterator<char>(), ' ');
    number_char -= line_count + word_count;
    struct stat info;
    stat(file_path.c_str(), &info);

    std::cout << line_count << "\n"
              << word_count << "\n"
              << number_char << "\n"
              << info.st_uid << "\n"
              << info.st_gid << "\n"
              << info.st_ctime <<  std::endl;
}
