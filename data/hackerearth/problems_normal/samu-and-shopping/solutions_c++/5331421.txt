#include <iostream>
#include <algorithm>
#include <vector>
#include <utility>
#include <cmath>
#include <queue>
using namespace std;

int main()
{
    int test;
    cin>>test;
    while(test--)
    {
    	int n;
    	cin>>n;
    	int s[n][3];
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<3;j++)
    		cin>>s[i][j];
    	}
    	int price[n][3];
    	for(int i=0;i<3;i++)
    	price[0][i]=s[0][i];
    	for(int i=1;i<n;i++)
    	{
    		price[i][0]=min(price[i-1][1],price[i-1][2])+s[i][0];
    		price[i][1]=min(price[i-1][0],price[i-1][2])+s[i][1];    	
    		price[i][2]=min(price[i-1][1],price[i-1][0])+s[i][2];    
    	}
    	int ans=min(price[n-1][0],min(price[n-1][1],price[n-1][2]));
    	cout<<ans<<endl;
    }
    return 0;
}
