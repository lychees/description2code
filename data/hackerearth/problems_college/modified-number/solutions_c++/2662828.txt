#include<iostream>
#include<algorithm>
#include<vector>
#include<memory.h>

using namespace std;

int main()
{
    int p,q;
    cin>>p>>q;

    int arr[]={1 ,9 ,45 ,55 ,99 ,297, 703, 999 ,2223 ,2728 ,4950, 5050, 7272 ,7777 ,9999 ,17344, 22222, 77778, 82656, 95121 ,99999};

int count=0;
    for(int i=0;i<21;i++)
    {
        if(arr[i]>=p && arr[i]<=q)
            {
                cout<<arr[i]<<" ";
                count++;
            }
    }
    if(count==0)
        cout<<"INVALID RANGE";

return 0;
}
