#include<bits/stdc++.h>
using namespace std;

int main(){
	int n;
	long long s,m,p;
	cin>>n>>s>>m>>p;
	long long arr[n],nx=0;
	for(int i=1;i<=n;i++){
		cin>>arr[i];
		nx=max(nx,arr[i]);
	}
	long long dp[2][2510];
	
	for(int j=0;j<=n;j++){
		dp[1][j]=-1000000000000000000;
	}
	long long mxx=-1000000000000000000;
	for(int j=1;j<=n;j++){
		int J = arr[j];
		long long w=0;
		long long a=arr[1];
			if(a>J){
				w = m*(a-J);
			}
			else{
				w = p*(J-a);
			}
		dp[0][j]=-w;
		mxx = max(mxx,dp[0][j]);
	}
	for(int i=2;i<=n;i++){
		long long  a = arr[i],tmp=-1000000000000000000;
		int I;
		if(i%2==0)
			I=1;
		else
			I=0;
		for(int j=1;j<=n;j++){
			int J=arr[j];
			long long w=0;
			if(a>J){
				w = m*(a-J);
			}
			else{
				w = p*(J-a);
			}
			dp[I][j]=dp[1-I][j]+s-w;
			dp[I][j]=max(dp[I][j],mxx-w);
			tmp=max(tmp,dp[I][j]);
		}
		mxx=tmp;
	}
	long long mx=0;
	for(int j=1;j<=n;j++)
			mx=max(mx,dp[1-n%2][j]);
	cout<<mx;
	return 0;
}
