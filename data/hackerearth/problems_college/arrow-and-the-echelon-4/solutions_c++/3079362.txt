#include<iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		long long int n;
		cin >> n;
		if (n % 3 == 0)
		{
			
			for (int i = 0; i < n; i++)
			{
				cout << "5";
			}
			cout << endl;
		}
		else if (n % 3 == 1)
		{
			int n5 , n3 ;
			n3 = 10 ;
			n5 = n - 10 ;
			if(n5 >= 0)
			{
				for (int i = 0; i < n5; i++)
				cout << "5";
				for (int i = 0; i < n3; i++)
					cout << "3";
				cout << endl;
			}
			else if (n % 5 == 0)
			{
				for (int i = 0; i < n; i++)
				{
					cout << "3";
				}
				cout << endl;
			}
			else
				cout << "-1" << endl;
		}
		else
		{
			int n5, n3;
			n5 = n - 5;
			n3 = 5;
			if(n5 >= 0)
			{
				for (int i = 0; i < n5; i++)
				cout << "5";
				for (int i = 0; i < n3; i++)
					cout << "3";
				cout << endl;
			}
			else
			cout << "-1" << endl ;
		}
	}
}
