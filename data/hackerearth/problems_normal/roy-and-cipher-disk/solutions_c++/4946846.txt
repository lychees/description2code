#include <iostream>
using namespace std;

int main()
{
    int count,i,j,k,l,t;
    char a[100];
    cin>>t;
    while(t>0)
    {
    	t--;
    	cin>>a;
    	j=0;
    	for(i=0;a[i]!='\0';i++)
    	{
    		k=a[i]-'a';
    		count=0;
    		l=j;
    		while(1)
    		{
    			if(l>25)
    			  l=0;
    			if(l==k)
    			   break;
    			else
    			{
    				count++;
    				l++;
    			}
    		}
    		j=k;
    		if(count>13)
    		   cout<<count-26<<" ";
    		else
    		   cout<<count<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
