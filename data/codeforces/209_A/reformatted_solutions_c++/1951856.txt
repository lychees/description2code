#include <algorithm>
#include <iostream>
#include <string>
#include <vector>
using namespace std;
int main() {
  vector<int> a(1000000);
  a[0] = 1;
  a[1] = 3;
  a[2] = 6;
  for (int i = 3; i < 1000000; i++)
    a[i] = (a[i - 1] + a[i - 2] + 2) % 1000000007;
  for (int n; cin >> n;)
    cout << a[n - 1] << endl;
  return 0;
}
