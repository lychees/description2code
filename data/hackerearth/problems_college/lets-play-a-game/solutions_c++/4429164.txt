void sieve()
{
    ll i,j;
    dp[0]=0;
    dp[1]=0;
    for(i=2;i<=1000001;i++)
    {
        if(arr[i]==0)
         {
            dp2[i]=1;

        for(j=2*i;j<=1000001;j=j+i)
        {
                arr[j]=1;
                dp2[j]++;
        }

         }
    }
    ll k=0;
    for(i=2;i<=1000001;i++)
    {
        if(arr[i]==0)
        {
         arr[k]=i;
         k++;
        }
    }

}
int fun(int n)
{
    int result=n;
    int i;
    for(i=0;arr[i]<=sqrt(n);i++)
    {
        if(n%arr[i]==0)
        {
            while(n%arr[i]==0)
            {
                n=n/arr[i];
            }
            result=result-result/arr[i];

        }
    }
    if(n>1)
    {
                      result=result-result/n;

    }
    return result;
}
