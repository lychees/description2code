#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    getchar();
    while(t--)
    {
    	string str;
    	getline(cin,str);
    	int len=str.length();
    	int sum=0;
    	for(int i=0;i<len;i++)
    	{
    		sum+=(str[i]);
    	}
    	int f=sum/len;
    	if(f%2==1)
    	{
    		reverse(str.begin(),str.end());
    		cout<<str<<endl;
    	}
    	else
    	{
    		cout<<str<<endl;
    	}
    }
}
