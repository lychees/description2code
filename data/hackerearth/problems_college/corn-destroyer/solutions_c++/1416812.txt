#include <bits/stdc++.h>
using namespace std;
#define For(i,a,b) for(int i=a;i<b;i++)
#define mp make_pair
#define pb push_back
#define mod 1000000007

int n,t,xx,yy,x,y;
double ans,lis[100005],a,inte;

int main( ){
    //freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
    //ios::sync_with_stdio(0);
    cin>>t;
    while(t--)
    {
        cin>>xx>>yy;
        cin>>n;
        For(i,0,n)
        {
            cin>>x>>y;
            if(x==xx)
            {
                if(y>yy) a=acos(-1)/2.0;
                else a=-acos(-1)/2.0;
            }
            else if(y==yy)
            {
                if(x>xx) a=0;
                else a=acos(-1);
            }
            else
            {
                a=atan(1.0*(y-yy)/(x-xx));
                if(a>0&&x<xx) a-=acos(-1);
                if(a<0&&y>yy) a+=acos(-1);
            }
            lis[i]=a;
        }
        sort(lis,lis+n);
        inte=2*acos(-1)+lis[0]-lis[n-1];
        For(i,0,n-1) inte=max(inte,lis[i+1]-lis[i]);
        ans=100.0*(2*acos(-1)-inte)/(2*acos(-1));
        printf("%.6f\n",ans);
    }
    return 0;
}
