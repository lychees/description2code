#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
    long long int a,n,b,cases;
    scanf("%lld",&cases);
    while(cases--) {
    	scanf("%lld %lld %lld",&a,&b,&n);
    	long long int g = __gcd(a,b);
    	g = (a*b)/g;
    	long long int l = 0;
    	long long int r = 1000000000000000000;
    	long long int val;
    	while(l <= r) {
    		long long int m = (l+r)/2;
    		val = (m/a) + m/b - m/g;
    		if(val == n) {
    			long long int ans = max((m/a)*a , (m/b)*b);
    			printf("%lld\n",ans);
    			break;
    		}else if(n > val){
    			l = m+1;
    		}else {
    			r = m-1;
    		}
    	}
    	
    }
    return 0;
}
