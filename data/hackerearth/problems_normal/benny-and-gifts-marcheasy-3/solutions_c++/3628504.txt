#include <bits/stdc++.h>
using namespace std;
#define ll long long 
#define fi first 
#define se second
#define pb push_back  
#define mp make_pair
//ll  a[1000005][2];
pair <ll,ll  >p[100005];
int main()
{
   ll i, j ,k ,x, y, n;   
   string s;
   cin>>s;
   n=s.length();
   p[0]=mp(0,0);
   x=0;
   y=0;
   for(i=0;s[i]!='\0';i++)
   {
     if(s[i]=='L')
      --x;
     else if(s[i]=='R')
      ++x;
     else if(s[i]=='U') 
      ++y;
     else
      --y;
      p[i+1]=mp(x,y);  
   }
   //for(i=0;i<i)
   p[i+1]=mp(1000000000,1000000000);
   
   sort(p,p+s.length()+2);
   ll cnt, hmm=0;
   cnt=0;
   
   //for(i=0;i<=n;i++)
   // cout<<p[i].fi<<" "<<p[i].se<<endl;
   for(i=0;i<=n;i++)
   {
      if(p[i]==p[i+1])
         {
            cnt++;
         }
      else
      {
         hmm+=cnt;
         cnt=0;
      }   
   }
   cout<<hmm;


   return 0;
}
