#include <iostream>
using namespace std;
int pr[1<<20];
int main()
{
    //cout << "Hello World!" << endl;
    int n,k;
    cin>>n>>k;
    for(int i=2;i<=n;i++){
    	if(pr[i]==0){
    		for(int j=i;j<=n;j+=i){
    			pr[j]+=i;
    		}
    	}
    }
    int count=0;
    for(int i=2;i<=n;i++){
    	if(pr[i]==k)
    		count++;
    }
    cout<<count<<endl;
    return 0;
}
