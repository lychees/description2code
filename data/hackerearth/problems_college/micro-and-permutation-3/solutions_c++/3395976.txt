#include <bits/stdc++.h>
using namespace std;

int comp (string X, string Y) {
    string XY = X.append(Y);
    string YX = Y.append(X);
    return XY.compare(YX) > 0 ? 1: 0;
}

int main() {
    int n;
    scanf("%d",&n);
    vector < string > v;
    v.resize(n);
    
    for(int i=0; i<n; i++){
    	cin >> v[i];
    }
    
    sort(v.begin(), v.end(), comp);
    
    for(int i=0; i<n; i++){
    	cout << v[i];
    }
    puts("");
    return 0;
}
