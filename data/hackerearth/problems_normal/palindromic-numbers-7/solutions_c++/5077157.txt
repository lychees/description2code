#include <iostream>
using namespace std;

int ispal(int n)
{
	int a[100],i,j;
	for(i=0;n!=0;i++)
	{
		a[i]=n%10;
		n/=10;
	}
	for(j=0;j<i;j++)
	{
		if(a[j]!=a[i-j-1])
		return 0;
	}
	return 1;
}

int main()
{
    int t,count,i,a,b;
    cin>>t;
    while(t--)
    {
    	count=0;
    	cin>>a>>b;
    	for(i=a;i<=b;i++)
    	if(ispal(i)==1)
    	count++;
    	cout<<count<<endl;
    }
    return 0;
}
