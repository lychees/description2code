#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t,a[3],ans;
	cin >> t;
	assert(t>=1 && t<=100);
	while ( t-- ) {
		ans = 0;
		cin >> a[0] >> a[1] >> a[2];
		if ( a[0] || a[1] || a[2] ) ans = 1;
		for ( int i = 0; i < 3; i++ ) {
			assert(a[i] >= 0 && a[i] <= 1000000);
			for ( int j = i+1; j < 3; j++ ) {
				ans = max(ans, 2*min(a[i],a[j]) + (a[i] != a[j]));
			}
		}
		cout << ans << endl;
	}
	return 0;
}
