
#include<bits/stdc++.h>
using namespace std;
int main(){
	long long  t,r,g,b,f1,f2,f3,t1,t2,t3;
	cin>>t>>r>>g>>b;
	f1=f2=f3=0;
	long long  a[8]={0};
	t1=r;t2=g;t3=b;
	while(t--){
		if(t1==0){
			t1=r;
			f1=(1+f1)%2;
		}
		if(t2==0){
			t2=g;f2=(1+f2)%2;
		}
		if(t3==0){
			t3=b;f3=(1+f3)%2;
		}
		if(f1&&f2&&f3){
		a[6]++;}
		else if(f1&&f2){
		a[3]++;}
		else if(f2&&f3){
		a[4]++;}
		else if(f1&&f3){
		a[5]++;}
		else if(f1){
		a[0]++;}
		else if(f2){
		a[1]++;}
		else if(f3){
		a[2]++;}
		else {
		a[7]++;}
		t1--;t2--;t3--;
	}
	for(int i=0;i<8;i++)cout<<a[i]<<" ";
	cout<<endl;
	return 0;
}
