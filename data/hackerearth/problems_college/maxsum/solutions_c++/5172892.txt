#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		vector<int> v;
		int temp;
		for(int i=0;i<n;i++){
			cin>>temp;
			v.push_back(temp);
		}
		vector<vector<int>>dp;
		dp.resize(n,vector<int>(2,-1));
		int i;
		for(i=0;i<1;i++){
			dp[i][0] = 0;
			dp[i][1] = v[i];
		}
		for(;i<n;i++){
			dp[i][0] = max(dp[i-1][0],dp[i-1][1]);
			dp[i][1] = v[i] + dp[i-1][0];
		}
		cout<<max(dp[n-1][0],dp[n-1][1])<<endl;
	}
	return 0;
}
