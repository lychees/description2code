#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
	int t;
	int n[100],m[100],s[100],x[100];
	
	cin>>t;
	for(int j=0;j<t;j++)
	{
		cin>>n[j]>>m[j]>>s[j];
		cout<<endl;
		x[j]=0;
	}
	for(int i=0;i<t;i++)
	{
		x[i]=m[i]%n[i];
		x[i]=x[i]+s[i]-1;
		if(x[i]<=n[i])
		{cout<<x[i]<<endl;}
		else
		{cout<< x[i]-n[i]<<endl;}
	}
    return 0;
}
