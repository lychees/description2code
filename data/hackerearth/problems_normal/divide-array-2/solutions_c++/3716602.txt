#include <iostream>
using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(0);
	int n;cin>>n;
	int arr[n];
	for(int i=0;i<n;i++)
		cin>>arr[i];
	int q;cin>>q;long long z=1;
	while(q--)
	{
		int a;cin>>a;
		z*=a;
	}
	if(z==0)
	{
		for(int i=0;i<n;i++)
			arr[i]=0;
	}
	else
	{
		for(int i=0;i<n;i++)
		arr[i]/=z;
	}
	for(int i=0;i<n;i++)
		cout<<arr[i]<<" ";
    return 0;
}
