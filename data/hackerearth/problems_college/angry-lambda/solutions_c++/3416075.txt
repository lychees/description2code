#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll t,i,j,k,l,m,c,n;
int main()
{
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;c=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>l;
    		if(l<=0)
    		{
    			c++;
    		}
    	}
    	if(c>=k)
    	{
    		cout<<"NO"<<endl;
    	}
    	else
    		cout<<"YES"<<endl;
    }
}
