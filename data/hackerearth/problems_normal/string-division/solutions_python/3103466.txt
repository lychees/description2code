map = {}
def process(str,count):
	global map
	if count==0:
		return 1
	ln = len(str);
	for i in range(1,ln-count+2):
		if map.get(str[:i],None) == None:
			map[str[:i]] = 1
			if process(str[i:],count-1):
				del map[str[:i]]
				return 1
			del map[str[:i]]
	return 0
		

def solve():
	global map
	map = {}
	str = raw_input()
	if process(str,4):
		return 'YES'
	return 'NO'

def main():
	t=raw_input()
	for i in range(int(t)):
		print solve()
		
main()
