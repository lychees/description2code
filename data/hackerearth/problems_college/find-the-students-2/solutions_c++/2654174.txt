#include<bits/stdc++.h>
using namespace std;

int n,i,c,a[1002];

int search(int x[],int l)
{
    int m=INT_MAX;
    
    for(int j=0;j<l;j++){
        if(x[j]<m && x[j]>0)
        m=x[j];
    }
    
    return m;
}

int main(){
    
    cin>>n;
    
    for(i=0;i<n;i++){
        cin>>a[i];
    }
    sort(a,a+n);
    
    if(a[0]!=0)
    cout<<n<<endl;
    
    else
    cout<<0<<endl;
    
    
    
    while(a[n-1]!=0){
        
        int z=search(a,n);
        int c=0;
        for(i=0;i<n;i++)
        a[i]=a[i]-z;
        
        for(i=0;i<n;i++){
            if(a[i]>0)
            c++;
        }
        if(a[n-1]!=0)
        cout<<c<<endl;
        
    }
    
    
}
