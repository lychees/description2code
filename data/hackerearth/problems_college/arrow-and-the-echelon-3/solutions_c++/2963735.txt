#include<bits/stdc++.h>

using namespace std;


int main()
{
    long long n;
    int t;

    cin>>t;

    while(t--)
    {

        cin>>n;

        int i,j,k,ans;
        int flag=0;
        int five=-1,three=-1;
        int five1=0,three1=0;
        for(i=0;i<=n;i+=3)
        {
            j=n-i;

            if(j%5==0)
            {   flag=1;
                if(i>five1)
                {five1=i;
                 three1=j;
                }
                else if(i==five1&&j>three1)
                {
                    {five1=i;
                 three1=j;
                }
                }

            }
       }

       if(flag)
       {
           for(i=0;i<five1;i++)
            cout<<"5";
           for(i=0;i<three1;i++)
            cout<<"3";
           cout<<endl;
       }

       else
        cout<<"-1\n";

}

}
