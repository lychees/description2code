#include <iostream>
using namespace std;

int main()
{
    unsigned long int n,p=0,q=0,r=0;
    int i;
    cin>>n;
    unsigned long int a[n];
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    	if(i%3==0)
    	p=p+a[i];
    	else if(i%3==1)
   q=q+a[i];
   else if(i%3==2)
   r=r+a[i];
    }
    cout<<p<<" "<<q<<" "<<r;
    return 0;
}
