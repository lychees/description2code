#include <iostream>
using namespace std;
long long arr[1000010][10]={0};
long long mod=1000000007;
int main()
{
	for(int i=1;i<=1000000;i++)
	{
		if(i==1)
		{
			for(int j=0;j<10;j++)
			arr[i][j]=1;
			continue;
		}
		for(int j=0;j<10;j++)
		{
			long long up=0,down=0;
			if(j!=0)
			up=arr[i-1][j-1];
			if(j!=9)
			down=arr[i-1][j+1];
			arr[i][j]=(up+down)%mod;
		}
	}
	int test;
	cin>>test;
	while(test--)
	{
		int n;
		cin>>n;
		long long ans=0;
		if(n==1)
		ans=10;
		else
		for(int i=1;i<10;i++)
		{
			ans=(ans+arr[n][i])%mod;
		}
		cout<<ans<<endl;
	}
    return 0;
}
