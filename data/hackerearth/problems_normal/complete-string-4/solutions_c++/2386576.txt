#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
   		char str[100];
   		int a[150],c=0;
   		for(int j=0;j<150;j++)
   		a[j]=0;
   		cin>>str;
   		for(int i=0;i<strlen(str);i++)
   		{
   			int n=str[i];
   			++a[n];
   			if(a[n]==1)
   			c++;
   		}
   		if(c==26)
   		cout<<"YES\n";
   		else
   		cout<<"NO\n";
   		
    }
    return 0;
}
