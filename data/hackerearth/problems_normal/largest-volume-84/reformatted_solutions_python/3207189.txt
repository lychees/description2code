import math

r, s = map(float, raw_input().split())

max_vol = 0.00

R = r
while R < s+r:
    h = math.sqrt(s**2-(R-r)**2)

    v = (1/3.0)*math.pi*h*(R**2+r**2+R*r)

    if v > max_vol:
        max_vol = v

    R += 0.0001

b = str(max_vol).split('.')
if len(b[1]) < 2:
    while len(b[1]) < 2:
        b[1] += '0'

# print "%s.%s" % (b[0], b[1][:2])

print "%.2f" % max_vol
