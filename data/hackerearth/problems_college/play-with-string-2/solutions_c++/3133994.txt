#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int n;
    cin>>n;
    while(n--)
    {
    	string s,r;
    	cin>>s;
    	int l=s.length(),i,f,g,ch=0,t;
    	for(i=l-1;i>=0;i--)
    	{
    	r[l-1-i]=s[i];
    	}
    	for(i=1;i<l;i++)
    	{
    		t=s[i];
    		g=s[i-1];
    		f=abs(t-g);
    		t=r[i];
    		g=r[i-1];
    		t=abs(t-g);
    		if(t!=f)
    		{
    			ch=1;
    			cout<<"Not Funny"<<endl;
    			break;
    		}
    	}
    	if(ch==0)cout<<"Funny"<<endl;
    }
    return 0;
}
