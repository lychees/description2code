#include <iostream>
#include<vector>
using namespace std;

int main()
{
   long long t,n,i,tmp,j;
   cin>>t;
   while(t--)
   {  int check=1;
   	cin>>n;
   	vector<long long> v(n+1),v1(n+1);
   	for(i=1;i<=n;i++)
   	{   cin>>tmp;
   		v1[i]=tmp;
    //    v[tmp]=i;
   	}
   		for(i=1;i<=n;i++)
   		v[i]=v1[v1[i]];
   		
   		for(i=1;i<=n;i++)
   		if(v[i]!=i) {check=0;break;}
   	
   if(!check) cout<<"not inverse"<<endl;
   else cout<<"inverse"<<endl;
   }
    return 0;
}
