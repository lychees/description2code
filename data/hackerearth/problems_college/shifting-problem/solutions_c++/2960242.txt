#include<bits/stdc++.h>
using namespace std;
 
 main()
{
    int tt;
    cin>>tt;
    long long int a[1001];
    memset(a, 0, sizeof(a));
    a[0] = 1;
    a[1] = 1;
    for(int i=2; i<1001; i++)
        a[i] = a[i-1] + a[i-2];
    while(tt--)
    {
        int n;
        scanf("%d",&n);
        printf("%lld\n",a[n]);
    }
 
}
