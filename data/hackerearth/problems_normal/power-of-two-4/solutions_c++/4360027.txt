#include <bits/stdc++.h>
using namespace std;

bool pow2(long long n)
{
	return (n && !(n & (n - 1)));
}

int main()
{
    ios::sync_with_stdio(false);
    long long total = 0, ans;
    for(int i = 0; i < 32; i++)
    	total += 1 << i;
    int t;
    cin >> t;
    while(t--) {
    	int n;
    	cin >> n;
    	long long ar[n];
    	for(int i = 0; i < n; i++)
    		cin >> ar[i];
    	
    	int flag = 0;	
    	for(int i = 0; i < 32; i++) {
    		ans = total;
    		for(int j = 0; j < n; j++) {
    			if(ar[j] & (1 << i))
    				ans &= ar[j];
    		}
    		if(pow2(ans)) {
    			flag = 1;
    			break;
    		}
    	}
    	
    	if(flag)
    		cout << "YES" << endl;
    	else
    		cout << "NO" << endl;
    }
    return 0;
}
