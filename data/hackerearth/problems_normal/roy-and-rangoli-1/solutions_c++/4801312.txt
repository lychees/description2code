#include<bits/stdc++.h>
#include<stdio.h>
using namespace std;
bool mark[2000005]={false};
long int dp[2000005]={0};
int main()
{
    long int n;
    //scanf("%ld",&n);
    cin>>n;
    long int i,j,k;
    for(i=2;(i*i)<=(2*n);i++){
            if(mark[i]==false){
              for(j=2*i;j<=2*n;j+=i){
                 mark[j]=true;
                }
            }
    }
   // for(k=2;k<=2*n;k++){
     //   if(mark[k]==false){
      //      cout<<k<<endl;
      //  }
    //}
    dp[0]=0;
    dp[1]=0;
    for(i=2;i<=2*n;i++){
        if(mark[i]==false){
            dp[i]=dp[i-1]+1;
        }
        else{
            dp[i]=dp[i-1];
        }
    }
    long int prev=0,curr=n-1;
    long int sum=0;
    for(i=0;i<n;i++){
            if(i==0){
                sum+=dp[curr];
            }
        else{
        sum+=dp[i+curr]-dp[i+prev-1];
        sum=(sum%1000000007);
        }
    }
    cout<<sum<<endl;
}
