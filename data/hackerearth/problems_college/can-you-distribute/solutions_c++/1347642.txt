#include <cstdio>
#include <iostream>
using namespace std;

#define MOD 10000007
#define MAX 100

typedef long long int ll;

ll nCr[MAX][MAX];

void preCompute()
{
	for (int line = 0; line < MAX; line++)
  	{
    	for (int i = 0; i <= line; i++)
    	{
      		if (line == i || i == 0)
           		nCr[line][i] = 1;
      		else 
           		nCr[line][i] = (nCr[line-1][i-1] + nCr[line-1][i])%MOD;
	    }
	}
}


int main(){
	preCompute();
	ll n,b;
	scanf("%lld%lld",&n,&b);
	
	if(n<b) printf("-1\n");
	else
		printf("%lld\n",nCr[n-1][b-1]);
	
	return 0;
}
