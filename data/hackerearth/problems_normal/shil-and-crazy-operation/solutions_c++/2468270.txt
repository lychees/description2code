
#include <bits/stdc++.h>
#define lli long long int
using namespace std;

int a[100005];
int p[100005];
bool f[100005] = {};
vector<int> v[100005];
int ans[100005];

int main()
{
	ios_base::sync_with_stdio(false);
	
	lli n,t;
	cin>>n>>t;
	
	for (int i=1 ; i<=n ; i++) cin>>a[i];
	for (int i=1 ; i<=n ; i++) cin>>p[i];
	
	int vi = 1;
	for (int i=1 ; i<=n ; i++)
	{
		if (f[i] == false)
		{
			int k = i;
			while (f[k]==false) v[vi].push_back(k) , f[k]=true , k=p[k];
			vi++;
		}
	}
	
	for (lli i=1 ; i<vi ; i++)
	{
		lli sz = v[i].size();
		lli s = t%sz;
		for (lli j=0 ; j<sz ; j++)
		{
			ans[ v[i][j] ] = a[ v[i][s] ];
			s = (s+1)%sz;
		}
	}
	
	for (int i=1 ; i<=n ; i++) cout<<ans[i]<<" ";
	
	return 0;
}
