#include <iostream>
using namespace std;

int main()
{
  int n,min,i,t,x;
  string c;
  cin>>t;
  while(t--)
  {
  	int min=100000,a[11]={0};
    cin>>n;
    cin>>c;
    for(i=1;i<=n;i++)
    {
  	x=(c[i-1]-'0');
  	
  	if((a[x]!=0)&&(min>(i-a[x])))
  		min=i-a[x];
  	a[x]=i;
    }
  if(min==100000)
  cout<<"-1"<<"\n";
  else
  cout<<min<<"\n";
    
  }
}
