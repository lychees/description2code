#include<iostream>
#include<cstdio>
using namespace std;
int a[100001],b[100001];
int main()
{
 int t;
 scanf("%d",&t);
 while(t--)
 {
 	int n,c=0,i;
 	cin>>n;
 	for(i=1;i<=n;i++)
 	{
 		cin>>a[i];
 	}
 	for(i=1;i<=n;i++)
 	{
 		b[a[i]]=i;
 	}
 	for(i=1;i<=n;i++)
 	{
 		if(a[i]!=b[i])
 		{
 			c=1;
 			break;
 		}
 	}
 	if(c==1)
 	cout<<"not inverse"<<endl;
 	else
 	cout<<"inverse"<<endl;
 }
}
