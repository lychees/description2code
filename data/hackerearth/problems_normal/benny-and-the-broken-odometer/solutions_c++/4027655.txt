#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */  
    long long int t,n,s,x,k,r,l;
    
    for(scanf("%lld",&t);t--;printf("%lld\n",s))
        {
        for(scanf("%lld",&n),x=n,s=k=0;x;x=x/10,k++)
            {
            l=pow(10,k);
            if(x%10>3)
                s+=l;            
        }
        s=n-s;
        for(;l>1;)
            {
            r=s/l;
            l/=10;
            s-=r*l;
        }
    }
    return 0;
}
