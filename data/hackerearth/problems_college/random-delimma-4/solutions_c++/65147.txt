#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
  int t;
    scanf("%d", &t);
    int k,n;
    double prob = 0;
    while(t--) {
      scanf("%d %d", &k, &n);
      if(k > n) {
        prob = 1.0;
      } else {
        prob = 1.0;
        for(int i = 0; i < k; ++i) {
          prob = prob * (n-i)/n;
        }
        prob = prob * -1;
        prob += 1.0;
      }
      std::cout << std::fixed << std::setprecision(9) << prob << std::endl;
      //printf("%lf\n", prob);
    }
    return 0;
}
