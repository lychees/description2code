test = int(raw_input())
for i in range(0, test):
    st = raw_input()
    i = 0
    count = 0
    while i < len(st):
        # print(st[i],st[i+1:])
        if st[i] in st[(i+1):]:
            count += 1
            break
        else:
            i += 1
            # print(i)
    if count == 0:
        print("No")
    elif count > 0:
        print("Yes")
