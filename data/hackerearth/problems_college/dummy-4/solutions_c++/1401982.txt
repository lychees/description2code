#include <bits/stdc++.h>
using namespace std;

int main()
{
	char s[1001];
    int n;
    long long sum =0, r = 0;
    cin >> n;
    while(n--){
    	cin >> s;
    	for(int i=0; s[i]!='\0'; i++){
    		sum += s[i] - '0';
    	}
    }
    while(sum > 9){
    	r = sum;
    	sum = 0;
    	while(r > 0){
    		sum += r%10;
    		r /= 10;
    	}
    }
    cout << sum << endl;
    return 0;
}
