#include<stdio.h>
#include<string.h>
int test(int count[],int sount[26])
{
	int i;
	for(i=0;i<26;i++)
	{
		if(count[i]!=sount[i])
		return 0;
	}
	return 1;
}
int main()
{
	int i,j,n,t,flag,l1,l2;
	scanf("%d",&t);
	while(t--)
	{
		char x[100006];
		char y[100006];
		int count[26],sount[26];
		for(i=0;i<26;i++)
		{
			count[i]=0;
			sount[i]=0;
		}
		scanf("%s%s",x,y);
		l1=strlen(x);
		l2=strlen(y);
		if(l1<l2)
		printf("NO\n");
		else
		{
		for(i=0;i<l2;i++)
		{
		count[x[i]-'a']++;
		sount[y[i]-'a']++;
		}
		if(test(count,sount))printf("YES\n");
		else
		{
			int flag=0;
			for(i=l2;i<l1;i++)
			{
				count[x[i-l2]-'a']--;
				count[x[i]-'a']++;
				if(test(count,sount))
				{
					flag=1;
					break;
				}
			}
			if(flag==1)
			printf("YES\n");
			else
			printf("NO\n");
		}
		
		}
		
		
	}
	return 0;
}
