#include <iostream>

using namespace std;

int main(int argc, char *argv[]) {
    int ll, rr, max = 0, a, b;
    register int temp;

    cin >> ll >> rr;
    for (a = ll; a < rr; ++a)
        for (b = a + 1; b <= rr; ++b) {
            max = (temp = a ^ b) > max ? temp : max;
        }

    cout << max << endl;
    return 0;
}
