#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<int,int> ii;
typedef vector<int> vi;
typedef vector<ii> vii;

const int  MOD=1000000007;
const int  INF= int(1e9);
const int MAX=1001;

ll c[MAX][MAX];
ll f[MAX];
ll dp[MAX];
inline void add(ll &a, ll b) {
    a += b;
    if (a >= MOD) a -= MOD;
}
inline int mul(ll  a, ll b) {
    return (long long)a * b % MOD;
}
int main()
{
    int n,k;
    cin >> n >> k;
    f[0] = 1;
    for(int i=1;i<=n;i++)
    {
        f[i]= mul(f[i-1],i);
    }
    for(int i=0;i<=n;i++)
    {
        for(int j=0;j<=i;j++)
        {
            if(j == 0 || j == i)
            {
                c[i][j] = 1;
            }
            else
            {
                c[i][j] = (c[i-1][j] + c[i-1][j-1])%MOD;
            }
        }
    }

    dp[0] = 1;
    for(int i=1;i<=n;i++)
    {
        for(int j=0;j<i;j++)
        {
            ll ways = mul(c[i-1][j],mul(f[j+1],dp[i-j-1]));
            add(dp[i],ways);
        }
    }
    ll res= mul(dp[n],c[n][k]);
    cout <<res  << "\n";
    return 0;
}
