for _ in range(int(raw_input())):
    a1, b1, a2, b2, a3, b3 = map(int, raw_input().split())
    slope1 = (b2-b1)/((a2-a1)*1.0) if a2 != a1 else None
    slope2 = (b3-b2)/((a3-a2)*1.0) if a3 != a2 else None
    # print slope1,slope2
    if slope1 == slope2:
        print "YES"
    else:
        print "NO"
