for _ in range(input()):
	num = input()
	oddsum = 1
	evensum = 0
	totsteps = 0
	while num > 1:
		if num % 2 == 0:
			evensum += num
			num /= 2
		else:
			oddsum += num
			num = (3 * num) + 1
		totsteps += 1
	evenrem = evensum % totsteps
	oddrem = oddsum % totsteps
	if oddrem > evenrem:
		print "Odd Rules"
	elif oddrem < evenrem:
		print "Even Rules"
	else:
		print "Tie"
