#include <bits/stdc++.h>
using namespace std;
int dp[1000000];
int fun(int idx,int a[],int n)
{
	if(idx>=n)
		return 0;
	if(dp[idx]!=-1)
		return dp[idx];
	int ans=0;
	ans=max(a[idx]+fun(idx+2,a,n),fun(idx+1,a,n));
	dp[idx]=ans;
	return ans;
}
