#include <iostream>
using namespace std;
#include<bits/stdc++.h>
#define ll long long int
int main()
{ ll n,m;
  int t;
  cin>>t;
  while(t--)
  {
  	ll finc=0ll;
   scanf("%lld%lld",&n,&m);
   if(m==1ll)
   {cout<<n<<endl; continue;}
   while(n!=0ll)
   {
   	finc = finc+n%m;
   	n = n/m;
   	
   }
   cout<<finc<<endl;
  	
  }
    return 0;
}
