#include <bits/stdc++.h>
using namespace std;

bool isPal(string s)
{
	string t=s;
	reverse(t.begin(),t.end());
	if(t==s)
		return true;
	return false;	
}


int main()
{
    int t;
    string a,b;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	int cnt=0;
    	string temp=a;
    	for(int i=0;i<=a.size();i++)
    	{
    		temp.insert(i,b);
    		if(isPal(temp))
    			cnt++;
    		
    		temp=a;
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
