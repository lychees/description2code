    ll read()
    {
        ll i=0;
        char ch=gc;
        while(ch<'0'||ch>'9')
        ch=gc;
        while(ch<='9'&&ch>='0')
        {
            i=(i<<1)+(i<<3)+ch-'0';
            ch=gc;
        }
        return i;
    }
