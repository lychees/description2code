#include <bits/stdc++.h>
#define ll long long
using namespace std;
string mysum(string a,string b)
{
	int l1=a.length();
	int l2=b.length();
	int i,x,j=max(l1,l2);
	if(l1<l2)
	{
		for(i=0;i<(l2-l1);i++)
		a="0"+a;
	}
	else
	{
		for(i=0;i<(l1-l2);i++)
		b="0"+b;
	}
	string c="";
	int carry=0;
	for(i=j-1;i>=0;i--)
	{
		x=((a.at(i)-'0')+(b.at(i)-'0')+carry);
		carry=x/10;
		x=x%10;
		c=((char)('0'+x))+c;
	}
	if(carry!=0)
	c=((char)(carry+'0'))+c;
	return c;
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	//~ string a,b;
	//~ cin>>a>>b;
	//~ cout<<mysum(a,b)<<endl;
	string a[3];
	int i,j,k;
	cin>>a[0]>>a[1]>>a[2];
	int flag=0;
	for(i=0;i<3;i++)
	{
		for(j=0;j<3;j++)
		{
			for(k=0;k<3;k++)
			{
				if(mysum(a[i],a[j])==a[k])
				{
					flag=1;
					break;
				}
			}
			if(flag)
			break;
		}
		if(flag)
		break;
	}
	if(flag)
	{
		cout<<"YES"<<endl;
	}
	else
	{
		cout<<"NO"<<endl;
	}
	return 0;
}
