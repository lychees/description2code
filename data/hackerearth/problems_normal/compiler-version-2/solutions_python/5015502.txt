arr = []
while(True):
	try:
		str1 = raw_input()
		arr.append(str1)
	except: break

for index, each_str in enumerate(arr):
	idx = 0
	new_str = ''
	while(idx < len(each_str)):
		if each_str[idx] == "/" and idx+1 < len(each_str) and each_str[idx+1] == "/":
			new_str += each_str[idx:]
			break
		elif each_str[idx] == "-" and idx+1 < len(each_str) and  each_str[idx+1] == ">":
			new_str += "."
			idx +=  1
		else:
			new_str += each_str[idx]
		idx += 1
