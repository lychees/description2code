#include <iostream>
using namespace std;

int main()
{
int a,b,n,temp,add = 0;
cin>>a>>b>>n;
int old_v = a;
int new_v = b;
if(n == 1){
    cout<<a<<endl;
}else if(n == 2){
    cout<<b<<endl;
}else{
for (int i = 3 ; i <= n ;i++){
    add = old_v + new_v;
    temp = new_v;
    new_v = add;
    old_v = temp;
}
cout<<add<<endl;
}

}
