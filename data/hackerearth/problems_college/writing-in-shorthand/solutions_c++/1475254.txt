#include<bits/stdc++.h>
using namespace std;

map<string , string>mp;

int main() {
    int n,m;
    scanf("%d%d",&n,&m);
    for(int i=0;i<m;i++) {
        string str1,str2;
        cin>>str1>>str2;
        mp[str1]=str2;
    }
    for(int i=0;i<n;i++) {
        string str;
        cin>>str;
        string s=mp[str];
        if(s.size()<str.size()) {
            cout<<s<<" ";
        } else {
            cout<<str<<" ";
        }
    }
    return 0;
}
