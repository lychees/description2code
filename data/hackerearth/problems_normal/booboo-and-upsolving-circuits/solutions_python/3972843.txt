from math import ceil
n,m=map(int,raw_input().split())
lis=map(int,raw_input().split())
l=1
r=10**18
while l<r:
    mid=(l+r)/2
    days=0
    cur=0
    worked=0
    while cur<n and days<m:
        worked=0
        while cur<n and worked<mid:
            worked+=lis[cur]
            cur+=1
        if worked>mid:
            cur-=1
        days+=1
    if cur==n and days<=m:
        r=mid
    else:
        l=mid+1
print r
