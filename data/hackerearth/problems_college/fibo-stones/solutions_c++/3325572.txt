#include <iostream>
using namespace std;
int main()
{
   long  int n,m,fib[1000],sum=0,sum1=0,i,j,first=0,next,second=1,stones[1000],total;
    cin>>n>>m;
    for(i=0;i<m;i++)
    {
    	cin>>stones[i];
    }
   for(i=1;i<=n;i++)
   {
   	if(i<=1)
   	next=i;
   	else
   	{
   		next=first+second;
   		first=second;
   		second=next;
   	}
   	fib[i-1]=next;
   }
  
   
   
    for(i=0;i<n;i++)
    {
    	sum1=sum1+fib[i];
    
    }
    //cout<<sum1<<endl;
    for(i=0;i<m;i++)
    {
    	for(j=0;j<n;j++)
    	{
    		if(stones[i]==fib[j])
    		{
    		//cout<<fib[j]<<endl;
    		sum=sum+fib[j];
    		break;
    	
    		}
    	}
    	
    	
    
    }
    total=sum1-sum;
    if(n==1&&m==1)
    cout<<0;
    cout<<total;
    
    return 0;
}
