#include<iostream>
#include<vector>
using namespace std;
class Solver{
	int n;
	vector<int> val;
	int get_gcd(int a , int b){
		if(a==0 || b==0){
			return a+b;
		}
		if(a<b){
			a = a^b;
			b = a^b;
			a = a^b;
		}
		if(a%b==0){
			return b;
		}
		return get_gcd(b , a%b);
	}
public:
	Solver(){
		cin>>n;
		val = vector<int>(n);
		for(int i=0;i<n;++i){
			cin>>val[i];
		}
	}
	int solve(){
		if(n==1){
			return val[0];
		}
		int gcd = get_gcd(val[0] , val[1]);
		for(int i=2;i<n;++i)if(val[i]!=0){
			gcd = get_gcd(val[i] , gcd);
		}
		
		return gcd;
	}
};
int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	Solver s;
    	cout<<s.solve()<<endl;
    }
    
    return 0;
}
