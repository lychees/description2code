#include <bits/stdc++.h>
 
#define fi first
#define se second
#define pb push_back
#define mp make_pair
#define pii pair<int,int>
#define pll pair<ll,ll>
#define MOD 1000000007
#define ll long long int
#define MAXINT 2000000000
#define SET(x,y) memset(x,y,sizeof(x));
 
using namespace std;

ll DP[1000];
int A[1000];

int main() {
	int n,i,j,c;
	cin>>n;
	for(i=1;i<=n;i++)	cin>>A[i];
	DP[0]=1;
	for(i=1;i<=n;i++) {
		DP[i]=0;
		c=0;
		for(j=i;j>=1;j--) {
			if(A[j])	c++;
			if(c==1)	DP[i]+=DP[j-1];
			if(c>1)		break;
		}
	}
	cout<<DP[n]<<endl;
	return 0;
}
