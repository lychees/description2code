#include <iostream>
using namespace std;

int main()
{
    unsigned long long int i,j,k,t,n,flag=0;
    cin>>t;
    while(t--)
    {	flag = 0;
    	cin>>n;
    	int a[100000];
    	int temp = n;
    	k=0;
    	while(temp)
    	{   
    		a[k] = temp%10;
    		temp = temp/10;
    		k++;
    	}
    	//cout<<k<<endl;
    	for(i=0;i<k;i++)
    	{  // cout<<a[i]<<" ";
    		if(a[i] == 1 && a[i+1] == 2)
    		  flag = 1;
    	}
    	if(n == 21 || n%21 == 0)
    		flag = 1;
    	if(flag)
    		cout<<"The streak is broken!"<<endl;
    	else
    		cout<<"The streak lives still in our heart!"<<endl;
    	
    }
    return 0;
}
