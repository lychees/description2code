#include <iostream>
#include <limits.h>

using namespace std;

long max_price_f(int price[], int n) {
	long total_price[n+1];
	total_price[0] = 0;
	for(int i=1; i<=n; i++) {
		long max_price = INT_MIN;
		for(int j=0; j<i; j++) {
			max_price = max(max_price, price[j] + total_price[i-j-1]);
		}
		total_price[i] = max_price;
	}
	
	return total_price[n];
}

int main()
{
    int t;
    cin>>t;
    int n;
    while(t--) {
    	cin>>n;
    	int price[n];
    	for(int i=0; i<n; i++) {
    		cin>>price[i];
    	}
    	cout<<max_price_f(price, n)<<endl;
    }
    return 0;
}
