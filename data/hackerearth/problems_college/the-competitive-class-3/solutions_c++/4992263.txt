#include <iostream>
#include<algorithm>
using namespace std;
int cmp(int a,int b){
	return a>b;
}
int main()
{
    int i,n,rank[1010],j=1;
    for(int i=0;i<1010;i++)rank[i]=0;
  	scanf("%d",&n);
  	int a[n],dup[n];
  	for(i=0;i<n;i++){
  		scanf("%d",&a[i]);
  		dup[i]=a[i];
  	}
  	sort(a,a+n,cmp);
  	
  	for(i=0;i<n;i++){
  		if(rank[a[i]]==0)
  		rank[a[i]]=(i+1);
  	}
  	printf("\n\n");
  	for(i=0;i<n;i++){
  		printf("%d ",rank[dup[i]]);
  	}
    return 0;
}
