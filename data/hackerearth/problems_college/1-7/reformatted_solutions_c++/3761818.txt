#include <bits/stdc++.h>
using namespace std;

int main() {
  int tc;
  ;
  cin >> tc;
  while (tc--) {
    string s;
    cin >> s;
    char c;
    cin >> c;
    int f[26] = {0};
    int len = s.length();
    for (int i = 0; i < len; i++) {
      f[s[i] - 'a']++;
    }
    cout << f[(int)c - 97] << endl;
  }
  return 0;
}
