#include "bits/stdc++.h"
using namespace std;
typedef long long int ll;
int main()
{
    ll t,n,i,day,x;
    i=1;
  
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	ll a[n],flag=1;
    	if(!n) cout<<"0\n";
    	else{
    	for(i=0;i<n;i++) cin>>a[i];
    	day=0;
    	
    	while(day<n)
    	{sort(a,a+n,greater<ll>());
    		for(i=0;i<=day;i++)
    		{
    		a[i]--;
    		if(a[i]<0)
    		{
    			flag=0;
    			break;
    		}}
    		if(!flag) break;
    		day++;
    	}
    	cout<<day<<endl;}
    }
    return 0;
}
