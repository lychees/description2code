#include <iostream>
#include <string.h>

using namespace std;
#define STUD 1000
#define SUBJ 20
int mrksheet[STUD][SUBJ];
int worst[STUD];

void findworst(int sub, int std)
{
	int temp = -1;
	int j = 0;
	do
	{	if (worst[j] > 0)
		{
			if(temp < 0)
				temp = mrksheet[worst[j]-1][sub-1];
			else
				if(temp > mrksheet[worst[j]-1][sub-1])
					temp = mrksheet[worst[j]-1][sub-1];
		}
		j++;
	}while(j<std);

	j = 0;
	do
	{
		if (worst[j] > 0)
		{
			if(temp != mrksheet[worst[j]-1][sub-1])
				worst[j] = 0;
		}
		j++;
	}while(j<std);

}
int main()
{
	int std, sub;

	cin>>std;
	cin>>sub;
	for(int i = 0;i<std;i++)
		for(int j = 0; j<sub;j++)
			cin>>mrksheet[i][j];

	int tc;
	cin>>tc;
	int qry;
	int subprlst;
	do
	{
		cin>>qry;
		for(int j = 0; j<std; j++)
			worst[j] = j + 1;
		for(int i =0; i<qry ;i++)
		{
			cin>>subprlst;
			findworst(subprlst,std);
		}

		int j = 0;
		do
		{
			if(worst[j] > 0)
			{
				cout<<worst[j]<<endl;
				break;
			}
			j++;
		}while(1);

		tc--;
	}while(tc>0);
	return 0;
}
