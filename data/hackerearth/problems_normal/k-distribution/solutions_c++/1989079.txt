#include <bits/stdc++.h>

using namespace std;

const int MOD=1000000007;
int N, K;
int dp[101][100001];

int main()
{
    scanf("%d%d", &N, &K);
    dp[0][0]=1;
    long long s=0;
    for(int i=1; i<=N; i++)
    {
        int a;
        scanf("%d", &a);
        for(int j=0; j<K; j++)
            dp[i][j]=dp[i-1][j];
        for(int j=a; j<K; j++)
            dp[i][j]=(dp[i][j]+dp[i-1][j-a])%MOD;
        s+=a;
    }
    int one=0, two=0;
    for(int i=0; i<K; i++)
    {
        if(i<K && s-i<K)
            two=(two+dp[N][i])%MOD;
        else
            one=(one+2LL*dp[N][i])%MOD;
    }
    int ans=1;
    for(int i=0; i<N; i++)
        ans=2LL*ans%MOD;
    ans=(ans-(one+two)%MOD+MOD)%MOD;
    printf("%d\n", ans);
    return 0;
}
