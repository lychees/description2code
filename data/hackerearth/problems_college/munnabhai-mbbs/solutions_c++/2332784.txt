#include<bits/stdc++.h>
#define N 1100
#define mp make_pair
using namespace std;
int edges[N];
int arr[N];
vector<int> v[N];
int main(){
	int t;
	cin>>t;
	while(t--){
		for(int i=0; i<N; i++){
			if(v[i].size()) v[i].clear();
			edges[i]=0;	arr[i]=0;
		} 
		int n,m; cin>>n>>m;
		while(m--){
			int x,y; cin>>x>>y;
			v[x].push_back(y);
			edges[y]++;
		}
		queue<int>q;
		for(int i=1; i<=n; i++){
			if(!edges[i]) q.push(i);
		}
		int f=0;
		while(!q.empty()){
			int cur=q.front();
			q.pop();
			f++;
			arr[cur]=1;
			for(int i=0; i<v[cur].size(); i++){
				int node=v[cur][i];
				if(edges[node]>0) edges[node]--;
				if(!edges[node] && !arr[node]){
					q.push(node);
					arr[node]=1;
				}
			}
		}
		cout<<f<<endl;

	}
}
