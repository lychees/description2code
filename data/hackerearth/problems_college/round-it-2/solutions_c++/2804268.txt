#include <bits/stdc++.h>
using namespace std;

int main()
{
	int T,A,B,rem;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%d %d",&A,&B);
		rem=A%B;
		if((float)rem>(float)B/2)
			printf("%d\n",A+(B-rem));
		else if((float)rem<(float)B/2)
			printf("%d\n",A-rem);
		else
			printf("%d\n",A+(B/2));
	}
return 0;
}
