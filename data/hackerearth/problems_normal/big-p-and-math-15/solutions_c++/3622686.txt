#include<iostream>

using namespace std;

int main()
{
	int a,b,temp;
	cin>>a>>b;

	int min_a[7],min_b[7],max_a[7],max_b[7],i=0,j=0;

	int x=10;

	while(a>x/10)
	{
		temp=a%x;
		temp=temp/(x/10);
		if(temp==5)
		{
			max_a[i]=6;
			min_a[i]=5;
		}
		else if(temp==6)
		{
			min_a[i]=5;
			max_a[i]=6;
		}
		else
		{
			min_a[i]=temp;
			max_a[i]=temp;
		}
		x=x*10;
		i++;
	}
	x=10;
	while(b>x/10)
	{
		temp=b%x;
		temp=temp/(x/10);
		if(temp==5)
		{
			max_b[j]=6;
			min_b[j]=5;
		}
		else if(temp==6)
		{
			min_b[j]=5;
			max_b[j]=6;
		}
		else
		{
			min_b[j]=temp;
			max_b[j]=temp;
		}
		x=x*10;
		j++;
	}
	int a_min=0,b_min=0,a_max=0,b_max=0;
	x=1;
	for(int k=0;k<i;k++)
	{
		a_min+=min_a[k]*x;
		x*=10;
	}
	x=1;
	for(int k=0;k<j;k++)
	{
		b_min+=min_b[k]*x;
		x*=10;
	}
	x=1;
	for(int k=0;k<i;k++)
	{
		a_max+=max_a[k]*x;
		x*=10;
	}
	x=1;
	for(int k=0;k<j;k++)
	{
		b_max+=max_b[k]*x;
		x*=10;
	}
	cout<<a_min+b_min<<" "<<a_max+b_max;
}
