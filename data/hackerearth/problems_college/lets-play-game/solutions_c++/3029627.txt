#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main(){

  cin.sync_with_stdio(false);
  int t;
  cin >> t;
  while(t--){

    ll n , k;
    cin >> n >> k;
    ll ans = 0;
    ll ar[n+1];
    for(int i = 0 ; i < n ; ++i ){
      ll x;
      cin >> x;
      ar[i] = k/x;
    }
    for(int i = 0 ; i < n; ++i ){
      ll x;
      cin >> x;
      ans = max( ans , x*ar[i]);
    }
    cout << ans << endl;

  }

}
