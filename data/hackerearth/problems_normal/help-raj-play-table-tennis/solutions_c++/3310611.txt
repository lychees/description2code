#include <bits/stdc++.h>
using namespace std;

#define mod 1000000009

long long int mem[40][40];
long long int a,x2,b,y2;

long long int solve(long long int x, long long int y)
{
	long long int ans;
	long long int k,m;

	if (mem[x][y] != -1) {
	    // cout <<x<<" "<<y<<endl;
		return mem[x][y];
	}

	if (x > x2 || y > y2)
		return 0;

	m = min(x,y);
	if (m >= 11 && abs(x-y) > 2) {
		//cout <<x<<" "<<y<<endl;
		return 0;
	}

	k = max(x,y);

	if (k >= 11 && abs(x-y) > 2) {
        return 0;
	}

	if (k >= 11 && abs(x-y) == 2) {
     //   cout <<"hi "<<x<< " "<<y<<endl;
		if (x == x2 && y == y2) {
			mem[x][y] = 1;
			return mem[x][y];
		}
		mem[x][y] = 0;
		return mem[x][y];
	}

	ans = solve(x+1,y);
	ans  = (ans + solve(x,y+1)) % mod;
    //if (x == 12 && y == 11)
	// cout <<x<<" "<<y<<" "<<ans<<endl;

	mem[x][y] = ans;
	return mem[x][y];
}


int main()
{
	long long int t,x,i,j,cas = 0;
	long long int ans;
	cin >> t;

	while (t--) {
		cas++;
		cout <<"Case "<<cas<<": ";
		ans = 0;

		for (i = 0; i < 35; i++) {
			for (j = 0; j < 35; j++) {
				mem[i][j] = -1;
			}
		}

		cin >> a >> b >> x2 >> y2;

		if (x2 < a || y2 < b) {
			cout <<"-1\n";
			continue;
		}

		x = max(x2,y2);
		if (x < 11) {
			cout <<"-1\n";
			continue;
		}

		x = max(a,b);


		if (x >= 11 && abs(a-b) == 2 && (a != x2 || b != y2)) {
			cout <<"-1\n";
			//cout <<"hello\n";
			continue;
		}

		if (x >= 11 && abs(a-b) > 2) {
            cout <<"-1\n";
            continue;
		}

		x = max(x2, y2);

		if (x >= 11 && abs(x2-y2) != 2) {
			cout <<"-1\n";
			continue;
		}


		ans = solve(a,b);
		ans = (ans + mod) % mod;
		cout <<ans<<endl;

	}
	return 0;
}
