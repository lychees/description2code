#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >>t;
    while(t--)
    {
    	int n;
    	cin >> n;
    	long long int arr[n+1];
    	arr[0]=0;
    	arr[1]=1;
    	arr[2] = 2;
    	for(int i=3;i<n+1;i++)
    		arr[i] = arr[i-1]+arr[i-2];
    	cout << arr[n] <<endl;
    }
    return 0;
}
