#include<bits/stdc++.h>
using namespace std;
#define ff first
#define ss second
#define pb push_back
#define mp make_pair
#define all(x) x.begin(),x.end()
#define sz(x) ((int)x.size())
const int MOD = 1e9+7;
typedef long long ll;
typedef pair<int,int> pii;
const int MAXN = 502;
ll dp[MAXN][MAXN];
int ar[MAXN];
int main() {
	// freopen("TASK.in","r",stdin);	
	// freopen("TASK.out","w",stdout);
	int n;
	cin>>n;
	for(int i=1;i<=n;i++) scanf("%d",&ar[i]);
	for(int i=1;i<=n;i++) {
		dp[i][ar[i]]=1;
		for(int j=i-1;j>=1;j--) {
			if(ar[j]<ar[i]) {
				for(int k=1;k<=100;k++) {
					int g=__gcd(k,ar[i]);
					dp[i][g]+=dp[j][k];
					if(dp[i][g]>=MOD) dp[i][g]-=MOD;
				}
			}
		}
	}
	ll ans=0;
	for(int i=1;i<=n;i++) {
		ans+=dp[i][1];
		if(ans>=MOD) ans-=MOD;
	}
	cout<<ans<<endl;
	return 0;
}
