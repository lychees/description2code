#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		string s;
		cin>>s;
		
		int len=s.length();
		int a[100000]={0};
		int ct=0;
		for(int i=0;i<len;i++)
		{
			if(s.at(i)=='-')
			{
				continue;
			}
			else
			if(s.at(i)=='B')
			{
				a[i]=1;
				if(i-1>=0)
				a[i-1]=1;
				if(i-2>=0)
				a[i-2]=1;
			}
			else
			if(s.at(i)=='S')
			{
				a[i]=1;
				if(i+1<len)
				a[i+1]=1;
				if(i-1>=0)
				a[i-1]=1;
			}
			else
			if(s.at(i)=='D')
			{
				a[i]=1;
			}
		}
		
		for(int i=0;i<len;i++)
		{
			if(a[i]==0)
			{
				ct++;
			}
		}
		
		cout<<ct<<"\n";
	}
}
