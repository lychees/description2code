#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
#define LL long long
int main()
{
	LL A,B,N;
	cin >> A >> B >> N;
	if(A == N)	
		cout << "1\n";
	else	{
		LL start = A, difference = A - B, count = 1;
		while(true) {
			start += difference;
			count++;
			if(start >= N)
			{cout << count << endl; break;}
		}
	}
    
    return 0;
}
