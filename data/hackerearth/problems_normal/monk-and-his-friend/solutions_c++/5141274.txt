#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	long long p,m;
    	cin >> p >> m;
    	long long res= p ^ m;
    	int count=0;
    	while(res)
    	{
    		res = res & res-1 ;
    		count++;
    	}
    	cout << count << endl;
    }
    return 0;
}
