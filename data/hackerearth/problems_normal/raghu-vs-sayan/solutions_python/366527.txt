t=int(input())
while(t):
    A,B,N=map(int, raw_input().split())
    
    # B=int(input())
    # N=int(input())
    Ncals=map(int, raw_input().split())
    Ncals=sorted(Ncals)
    y=0
    z=0
    Ncals.sort()
    for i in Ncals:
        if(i<=A):
            A-=i
            y=y+1
        if(i<=B):
            B-=i
            z=z+1
    if(y>z):
        print("Raghu Won")
    elif(y<z):
        print("Sayan Won")
    else:
        print("Tie")
    
    t=t-1
