#include<cstdio>
#include<iostream>
using namespace std;

int max_sum(int a[], int size)
{
   int max_so_far = a[0], i;
   int curr_max = a[0];

   for (i = 1; i < size; i++)
   {
        curr_max = max(a[i], curr_max+a[i]);
        max_so_far = max(max_so_far, curr_max);
   }
   return max_so_far;
}

int main()
{
    int t;
    scanf(" %d",&t);

    while(t--)
    {
        int res;
        int n;
        scanf(" %d",&n);
        int a[n];
        int i;
        for(i=0;i<n;i++)
        {
            scanf(" %d",&a[i]);
        }
        res = max_sum(a, n);
        printf("%d\n",res);
    }

   return 0;
}
