#include<iostream>
#include<string.h>
#define ll long long int
using namespace std;
int main()
{
	int t, x, y, f;
	ll ans;
	char s[1005];
	char a[10][6]={".,?!1","abc2","def3","ghi4","jkl5","mno6","pqrs7","tuv8", "wxyz9","_0"};
	cin>>t;
	while(t--)
	{
		ans=0;
		scanf("%s",s);
		int j=0;
		for(int i=0; i<strlen(s); i++)
		{
			f=0;
			for(x=0; x<10; x++)
			{
				for(y=0; y<5; y++)
				{
					if(s[i]==a[x][y])
					{f=1; break;}
				}
				if(f==1)
				break;
			}
			if(x!=j)
			{j=x; ans++;}
			ans=ans+y+1;
			//cout<<ans<<endl;
		}
		
	cout<<ans<<endl;
	}
}
