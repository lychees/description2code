#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long double a,b,c,k,r;
    	cin>>a>>b>>c;
    	k=(a+b+c)/2;
    	r=(a*b*c)/(4*sqrt(k*(k-a)*(k-b)*(k-c)));
    	printf("%.4llf\n",3.1415*r*r);
    }
    return 0;
}
