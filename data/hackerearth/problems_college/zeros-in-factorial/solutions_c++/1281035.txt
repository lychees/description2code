#include <iostream>
using namespace std;
long long int val[1000001];

long long int no(long long int num){
	long long res;
	for(res=0;num>0;){
		  num/=5;
		  res+=num;
		}
	return res;
}

void pre(){
	for(long long int i=0;i<=1000000;i++){
		val[i]=no(i);
		if(i!=0)
			val[i]+=val[i-1];
	}	
}

int main()
{
    long long int a,b,c,sum;
    cin>>a;
    pre();
    while(a--){
    	cin>>b>>c;
    	sum=val[c]-val[b-1];
    	cout<<sum<<endl;
    }
    return 0;
}
