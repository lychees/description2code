#include <math.h>
#include <iostream>
using namespace std;

class GET1
{
	public:
		int counter=0;
		void GETONE(int num, int tc)
		{
    		if (num == 1)
    		{
       			cout<<"Case #"<<tc<<":YES\n";
    		}
    		else
    		{
     		  counter++;
       			if (counter >= 20)
            		cout<<"Case #"<<tc<<":NO\n";
       			else
       			{
       				int tempSum=0;
            		while (num > 0)
            		{
                		tempSum += pow((num % 10),2);
                		num = num / 10;
            		}
            		GETONE(tempSum, tc);
       			}
    		}
		}
};

int main()
{
	GET1 get;
	int tc;
    cin>>tc;
    if (tc >= 1 && tc <= 1000)
    {
        for (int i = 0; i < tc; i++)
        {
            get.counter = 0;
            int num;
            cin>>num;
            get.GETONE(num, i+1);
        }
    }
    return 0;
}
