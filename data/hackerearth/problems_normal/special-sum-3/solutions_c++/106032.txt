int gcd(long int max,long int min)
{
  
  
   while(max!=min)
   {
    if(max<min)
    {
     min=min-max;
    }
    else
    max=max-min;
    
   }
   
   if(max==1 && min==1)
   return 1;
   else
   return 0;

}
long int foo(long int n) 
{
        long int ret = 0;
        for (int i = 1; i <= n; i++) 
        {
            if (gcd(n, i) == 1) 
            {
                ret++;
            }
        }
 return ret;
}
    

long int SpecialSum(long int n)
{
        long int ret = 0;

        for (int i = 1; i <= n; i++) 
        {
            if(n % i == 0)
            {
                ret += foo(i);
            }
        }

  return ret;
}
