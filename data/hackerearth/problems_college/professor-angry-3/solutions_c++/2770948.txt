#include <iostream>
using namespace std;

int main()
{
    int a[1000],t,n,k,early=0;
    cin>>t;
    while(t--){
    	cin>>n>>k;
    	for(int i=0;i<n;i++){
    		cin>>a[i];
    	}
    	for(int i=0;i<n;i++){
    		if(a[i]<=0){
    			early++;
    		}
    	}
    	if(early>=k){
    		cout<<"NO"<<endl;
    	}
    	else
    		cout<<"YES"<<endl;
    		early=0;
    }
    return 0;
}
