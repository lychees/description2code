#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define MAX 1000002

ll arr[MAX];
ll dp[MAX];
int main()
{
  int test;cin>>test;
  int t;ll n,k;int i;
  for(t=0;t<test;t++)
  {
    cin>>n>>k;
    for(i=0;i<n;i++)
    {
      cin>>arr[i];
    }
    ll result;
    if (arr[0]<=k)
    dp[0]=arr[0];
    else
    dp[0]=0;
    result=dp[0];
    for(i=1;i<n;i++)
    {
      if (arr[i]<=k)
      {
        dp[i]=dp[i-1]+arr[i];
      }
      else
      dp[i]=0;
      result=max(result,dp[i]);
    }
    cout<<result<<"\n";
  }

return 0;
}
