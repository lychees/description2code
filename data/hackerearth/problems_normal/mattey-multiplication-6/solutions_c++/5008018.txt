#include <bits/stdc++.h>

using namespace std;

vector < int > res;

int main()
{

    long long n , m , t;

    cin >> t;

    while(t--){

        cin >> n >> m;

        int bit = 0;
        res.clear();

        while(m != 0){

            if(m % 2 != 0){         // binary bit = 1

                res.push_back(bit);
            }

            m /= 2;                 // shift right
            bit++;                  
        }

        printf("(%lld<<%d)", n, res[res.size() - 1]);

        for(int i = res.size() - 2; i >= 0; i--){       //traverse in a reverse order

            printf(" + (%lld<<%d)", n, res[i]);
        }

        printf("\n");
    }

    return 0;
}
