/*input
5
1 4 6 2
4 1 4 2
8 2 1 2
2 8 8 2
1 2 1 3
*/

#include <bits/stdc++.h>
using namespace std;
#define M 50007
#define NUM(x) (x - '0')
#define CHAR(x) (x + '0')
char num[M];

int main()
{
	int T, a0, a1, c, n;

	cin >> T;
	while (T--) {	
		cin >> a0 >> a1 >> c >> n;
		sprintf(num, "%d%d", a0, a1);

		for (int i = 2; i < n; i++) {
			num[i] = CHAR((NUM(num[i - 1])*c + NUM(num[i - 2])) % 10);
		}
		num[n] = '\0';

		int mod = 0;
		for (int i = 0; i < n; i++) {
			mod = (mod * 10 + NUM(num[i])) % 41;
		}
		if (mod == 0)
			cout << "YES\n";
		else
			cout << "NO\n";
	}
}
