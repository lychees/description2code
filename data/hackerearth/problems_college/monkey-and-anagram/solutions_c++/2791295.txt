#include<stdio.h>
#include<string.h>
#include<iostream>
#include<math.h>
#include<stdlib.h>
#include<vector>
#include<algorithm>
#include <bits/stdc++.h>
using namespace std;                       


int main() {
	int t,i;
	cin>>t;
	cin.ignore(254,'\n');
	while(t--)
	{
		int ct=0,flag=0;
		string a,b;
		int cta[26]={0},ctb[26]={0};
		cin>>a>>b;
	
		//sort(a,a+strlen(a));
		//sort(b,b+strlen(b));
		for(i=0;i<a.length();i++)
		{
			cta[a[i]-'a']++;
			
		}
		
		for(i=0;i<b.length();i++)
		{
			ctb[b[i]-'a']++;
		}
		
		
		for(i=0;i<26;i++)
		{
			if(cta[i]==ctb[i]) 
			continue;
		    if(cta[i]>ctb[i])
			{
				flag=1;
				cout<<"-1"<<"\n";
				break;
			}
			
			if(ctb[i]>0)
			ct++;
		}
	if(flag!=1)
	{
		
		cout<<ct<<"\n";
	}
	
	
	//cin.ignore(254,'\n');
	}
	
	return 0;
}
