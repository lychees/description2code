#include<stdio.h>
#include<cmath>
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
int main()
{
long long int t=0;
cin>>t;
while(t>0)
{
char arr[100001];
	cin>>arr;
	long long int l_s=0;
	long long int a_h=0;
	for(long long int i=0;i<ceil((strlen(arr))/2);i++)
	{
		l_s+=arr[i]-'0';
		}
	for(long long int i=strlen(arr)-1;i>=strlen(arr)/2;i--)
	{
	a_h+=arr[i]-'0';
		}
	//	cout<<l_s<<" "<<a_h;
	if(l_s==a_h)
	cout<<"YES\n";
	else
	cout<<"NO\n";
	t--;
	}
	return 0;
	}//end fot the amin
