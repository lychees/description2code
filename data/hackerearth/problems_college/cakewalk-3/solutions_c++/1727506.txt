#include <iostream>
using namespace std;

int main()
{
    int n;
    cin >> n;
    string s;
    cin >> s;
    n--;
    string t;
    while (n--) {
    	int m;
    	cin >> m;
    	cin >> t;
    	string tmp = "";
    	for (int i = 0; i < m; i++) {
    		tmp += s[i];;	
    	}
    	t = tmp + t;
    	//cout << t << endl;
    	s = t;
    }
    cout << t << endl;
    return 0;
}
