#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	int i, a[t];
	for(i = 0; i < t; i++)
	{
		cin >> a[i];
	}
	int q;
	cin >> q;
	while(q--)
	{
		int flag = -1;
		long long int target;
		cin >> target;
		long long int sum = 0;
		for(i = 0; i < t; i++)
		{
			sum = sum + a[i];
			if(sum >= target)
			{
				cout << i + 1 << endl;
				flag = 1;
				break;
			}
		}
		if(flag == -1)
		{
			cout << "-1\n";
		}
	}
}
