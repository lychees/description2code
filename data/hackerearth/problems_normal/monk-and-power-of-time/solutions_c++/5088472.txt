#include <iostream>
#include<cstdlib>
using namespace std;

int main()
{
    int x,l,count=0,p,i;
    cin>>x;
    int a[100]={0},b[100]={0};
    for(i=0;i<x;i++)
    {cin>>l;
    a[i]=l;}
    for(i=0;i<x;i++)
    {cin>>l;
    b[i]=l;}
    i=0;
    while(i!=x){
    if(a[i]==b[i]){++count;i++;}
    else{p=a[i];
    	for(int j=i;j<x;j++)
    {if(j==x-1)a[j]=p;
    else
    	a[j]=a[j+1];
    
    	
    }
    i=i;++count;
    }}
    cout<<count;
    return 0;
}
