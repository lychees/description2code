#include<iostream>
using namespace std;
int dp[1002][1002][2];
int gcd(int a, int b)
{
	if (b == 0)
		return a;
	else
		return gcd(b, a%b);
}
void callf()
{
	for (int i = 1; i <= 1000; i++)
	{
		for (int j = 1; j <= 1000; j++)
		{
			for (int k = 0; k < 2; k++)
			{
				if (i == 1 && j == 1)
				{
					dp[i][j][k] = -1;
				}
				else if (i == 1)
				{
					dp[i][j][k] = 0;
				}
				else if (j == 1)
				{
					dp[i][j][k] = 1;
				}
				else if (!k)
				{
					int x = gcd(i, j);
					if (x > 1)
					{
						dp[i][j][k] = dp[i][j - 1][!k] || dp[i][j / x][!k];
					}
					else
					{
						dp[i][j][k] = dp[i][j - 1][!k];
					}
				}
				else if (k)
				{
					int x = gcd(i, j);
					if (x > 1)
					{
						dp[i][j][k] = dp[i - 1][j][!k] && dp[i / x][j][!k];
					}
					else
					{
						dp[i][j][k] = dp[i - 1][j][!k];
					}
				}
			}
		}
	}
}

int main()
{
	int t;
	int a, b;
	cin >> t;
	callf();
	while (t--)
	{
		cin >> a >> b;
		if (dp[a][b][0] == -1)
			cout << "Draw" << endl;
		else if (dp[a][b][0])
			cout << "Arjit" << endl;
		else
			cout << "Chandu Don" << endl;
	}
}
