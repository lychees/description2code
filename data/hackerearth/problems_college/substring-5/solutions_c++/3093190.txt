#include <iostream>
using namespace std;
int main()
{
    string s;
    cin>>s;
    int l=s.length();
    int i,z;
    z=1;
    int total=l;
    for(i=0;i<l;i++)
    {
    	if(s[i]==s[i+1])
    	{
    		total=total+z;
    		z++;
    	}
    	else
    	{
    		z=1;
    	}
    }
    cout<<total<<endl;
    return 0;
}
