#include <iostream>
#define MAX 100000
using namespace std;
int dp[MAX]={0};
int f(int n)
{ dp[0]=1;
 dp[1]=1;
 dp[2]=2;
 dp[3]=3;
 for(int j=3;j<=n;j++)
 dp[j]=dp[j-1]+dp[j-2];
 return dp[n];
}
int g(int n )
{ dp[0]=1;
 dp[1]=1;
 dp[2]=2;
 dp[3]=4;
 for(int j=3;j<=n;j++)
 dp[j]=dp[j-1]+dp[j-2]+dp[j-3];
 return dp[n];
}

int main()
{
   int n;
   cin>>n;
  int j=f(n);
  int k=g(n);
  if(j==0&&k==0)
  {cout<<"ERROR"<<endl;
   return 0;
  }
  
  
  cout<<"Jack-"<<k<<endl;
  cout<<"Jill-"<<j<<endl;
   
    return 0;
}
