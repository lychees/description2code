from collections import Counter


def main():
    for _ in range(input()):
        l = input()
        seq = map(int, raw_input().split())
        c = Counter(seq)
        m = max([c[i] for i in c])
        ans = []
        for i in c:
            if c[i] == m:
                ans.append(i)
        ans.sort()
        ans.reverse()
        st = ""
        for i in ans:
            st += str(i)
            st += " "
        print st


if __name__ == "__main__":
    main()
