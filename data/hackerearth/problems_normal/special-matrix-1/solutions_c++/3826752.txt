#include <iostream>
using namespace std;

int abbs(int a)
{
	if(a<0)
		return -a;
	return a;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<n;j++)
    		{
    			char k;
    			cin>>k;
    			if(k=='*')
    				cout<<(abbs(i-n/2)+abbs(j-n/2))<<endl;
    		}
    	}
    }
    return 0;
}
