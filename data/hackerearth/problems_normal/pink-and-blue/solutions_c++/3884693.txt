#include<bits/stdc++.h>
using namespace std ;
//-------------------------------
typedef long long ll ;
typedef vector<int> vi ;
typedef pair<int,int> ii ;
//-------------------------------
 
#define pb(a) push_back(a)
#define mp(a,b) make_pair(a,b)
#define rep(i,a,b) for(int i=(a) ; i<(b) ; ++i)
#define inf 2000000000
#define endl "\n"
#define max(a,b) ( (a>b) ? (a) : (b)  )
#define min(a,b) ( (a<b) ? (a) : (b)  )
//------------------------------
int ri()
{
	char c = getchar() ;
	while(c<'0' || c>'9') c = getchar() ;
	int ret = 0 ;
	while(c>='0' && c<='9')
	{
		ret = 10*ret+c-48;
		c = getchar();
	}
	return ret;
}
const int maxn = 1e5+5;
int n,m,u,v;
vi gr[maxn];
char color[maxn];
int col[maxn],visit[maxn];
void dfs(int x,int pa)
{
	if(visit[x]) return;
	visit[x]=1;
	rep(i,0,gr[x].size())
	{
		int u=gr[x][i];
		if(u==pa) continue;
		col[u]=(col[x]==0)?1:0;
		dfs(u,x);
	}
}
ll ans=inf;
int main()
{
//	freopen("input.txt","r",stdin);
	n=ri();m=ri();
	rep(i,1,n+1) cin>>color[i];
	rep(i,0,m)
	{
		u=ri();v=ri();
		gr[u].pb(v); gr[v].pb(u);
	}

	memset(col,0,sizeof col);
	memset(visit,0,sizeof visit);
	dfs(1,-1);
	int cnt=0;
	rep(i,1,n+1) if( (color[i]=='B'&&col[i]==1) || (color[i]=='G'&&col[i]==0) ) cnt++;
	ans = min(cnt,ans);
	memset(col,0,sizeof col);
	memset(visit,0,sizeof visit);
	col[1]=1;
	dfs(1,-1);
	cnt=0;
	rep(i,1,n+1) if( (color[i]=='B'&&col[i]==1) || (color[i]=='G'&&col[i]==0) ) cnt++;
	ans = min(cnt,ans);
	cout<<ans;
}
