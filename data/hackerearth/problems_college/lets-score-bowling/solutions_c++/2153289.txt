#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int a[3],i=0;
		int ans=0;
		cin>>a[0]>>a[1]>>a[2];
int	index=-1;
		if(a[0]!=9&&a[1]!=9&&a[2]!=9)
		ans=0;
		else
		{
		for(i=0;i<3;i++)
		{
		if(a[i]==9)
		{
			ans+=10;
			index=i;
			break;
		}	
		else if(a[i]==0)
		ans=ans;
		else if(a[i]>0&&a[i]<7)
		ans++;
		else if(a[i]>6&&a[i]<9)
		ans+=2;
		}
		if(index==1)
		ans-=3;
		if(index==2)
		ans-=6;
		}
		cout<<ans<<"\n";
	}
//    cout << "Hello World!" << endl;
    return 0;
}
