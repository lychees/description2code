def yo(a,size):
     
    max_so_far =a[0]
    curr_max = a[0]
     
    for i in range(1,size):
        curr_max = max(a[i], curr_max + a[i])
        max_so_far = max(max_so_far,curr_max)
         
    return max_so_far
n=int(raw_input())
#s=map(int,raw_input().strip().split(' '))
if n>0:
	s=raw_input()
	s=map(int,s.split())
	print yo(s,n)
else:
