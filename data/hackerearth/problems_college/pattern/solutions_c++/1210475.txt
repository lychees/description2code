#include <iostream>
using namespace std;

int main()
{
   int n,i,j;
 
   cin>>n;
     int d= n+1;
   for(i=1;i<=n;i++){
	   	for(j=1;j<=n;j++){
	   		if(i==1 || i==n ||(i+j)%d==0)
	   			cout<<"*";
	   		else
	   			cout<<" ";
	   	
	   		if((i+j)%d==0 && i!=n){
	   			cout<<"\n";
	   			break;
	   		}
	   			
	   	}
	   
   		
   	
   	
   }
   	
   
}
