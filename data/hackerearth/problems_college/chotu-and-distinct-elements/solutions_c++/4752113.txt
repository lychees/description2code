#include <bits/stdc++.h>
using namespace std;
int a[100005],b[100005];
int main()
{
    int x,t,i,n,l,h,m,flag,sz;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>b[i];
    	}
    	sort(b,b+n);
    	sz=0;
    	for(i=0;i<n;i++)
    	{
 			x=b[i];
    		l=0;
    		h=sz-1;
    		flag=0;
    		while(l<=h)
    		{
    			m=(l+h)/2;
    			if(a[m]==x)
    			{
    				flag=1;
    				break;
    			}
    			else if(a[m]>x)
    			{
    				h=m-1;
    			}
    			else
    			{
    				l=m+1;
    			}
    		}
    		if(!flag)
    		{
    			a[sz]=x;
    			sz++;
    		}
    	}
    	for(i=0,x=sz-1;i<=x;i++,x--)
    	{
    		if(i!=x)
    		{
    			cout<<a[x]<<" "<<a[i]<<" ";
    		}
    		else
    		{
    			cout<<a[x]<<" ";
    		}
    	}
    	cout<<endl;
    }
}
