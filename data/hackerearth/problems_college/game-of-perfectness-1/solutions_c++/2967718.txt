#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    int n;
    cin>>t;
    for(int test=1; test<=t; test++)
    {
        int flag=0;
        float temp,temp1;
        cin>>n;
        int br[n],cr[n];
        float ar[1][n];
        for(int i=0; i<n; i++)
            scanf("%d",&br[i]);
        for(int i=0; i<n; i++)
            scanf("%d",&cr[i]);
        printf("Case #%d: ",test);
        for(int i=0; i<n; i++)
        {
            ar[1][i]=i;
            ar[0][i]=(float)cr[i]/br[i];
        }
        for(int k=1; k<n; k++)
        {
            for(int l=0; l<n-k; l++)
                if(ar[0][l]>=ar[0][l+1])
                {
                    temp=ar[0][l];
                    ar[0][l]=ar[0][l+1];
                    ar[0][l+1]=temp;
                    temp1=ar[1][l];
                    ar[1][l]=ar[1][l+1];
                    ar[1][l+1]=temp1;
                }
        }
        for(int i=n-1;i>=0;i--)
            cout<<ar[1][i]<<" ";
        cout<<endl;
    }

    return 0;
}
