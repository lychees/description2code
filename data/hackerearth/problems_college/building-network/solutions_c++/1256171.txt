#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const int N=1e3+10;
const int inf=1e9+9;

int a[N];

int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int n;
        scanf("%d",&n);
        int minVal=inf;
        for(int i=0;i<n;i++) {
            scanf("%d",a+i);
            minVal=min(minVal,a[i]);
        }
        ll ans=0;
        for(int i=0;i<n;i++) {
            ans+=(a[i]+minVal);
        }
        ans-=2*minVal;
        printf("%lld\012",ans);
    }
    return 0;
}
