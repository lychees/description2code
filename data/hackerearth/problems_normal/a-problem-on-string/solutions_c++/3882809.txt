#include <iostream>
using namespace std;
 
int main()
{
    long long int i,j,k,l,m,n,a[30]={0};
    string s;
    char x,y;
    cin>>n;
    cin>>s;
    for(i=0;i<n;i++)
    {
    	a[s[i]-'a']++;
    }
    cin>>k;
    while(k--)
    {
    	cin>>x>>y;
    	if(x==y)
    	{
    			m=a[x-'a']*(a[y-'a']-1)/2;
    		
    	}
    	else
    	{
    		m=a[x-'a']*(a[y-'a']);
    	}
    	cout<<m<<endl;
    }
    return 0;
}
