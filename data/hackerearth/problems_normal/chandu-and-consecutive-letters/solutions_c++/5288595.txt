#include <iostream>
#include <string>
using namespace std;

int main()
{
    unsigned int count = 0;
    cin >> count;
    
    string str;
    for(unsigned int i =0 ; i< count; i++)
    {
    	cin >>str;
    	for(unsigned int index=0; index< str.size()-1; index++)
    	{
    		if(str[index] == str[index+1])
    		{
    			str.erase(index+1, 1);
    			index--;
    		}
    	}
    	
    	cout << str << '\n';
    }
}
