#include <iostream>
using namespace std;
#include<algorithm>

int main()
{
    //cout << "Hello World!" << endl;
    int t;
   	cin>>t;
    for(;t--;)
    {
    	int b,i,j,sum=0,count=0,n;
    	cin>>n>>b;
    	int a[n];
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	sort(a,a+n);
    	for(i=0;i<n;i++)
    	{
    		sum+=a[i];
    		if(sum<=b)
    		count++;
    		else

    	}
    	cout<<(count)<<endl;
    }
    return 0;
}
