#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
#define MAX 100005
int arr[105];
int mark[10005];
int dp[10005];
int main()
{
		int t;
		cin>>t;
		while(t--)
		{
				memset(mark,0,sizeof(mark));
				memset(dp,-1,sizeof(dp));
				int n,m;
				int i,tmp,j;
				cin>>m>>n;
				for(i=0;i<n;++i)
				{
						cin>>tmp;
						arr[i]=tmp;
						dp[tmp] = 1;
						mark[tmp]++;;
				}
				sort(arr,arr+n);
				dp[0]=1;
				for(i=1;i<=m;++i)
				{
						if(dp[i] == -1)
						{
								if(mark[i] > 0) dp[i]=1;
								else
								{
										dp[i] = 0;
										for(j=0;j<n&&i+arr[j]<=m;++j)
										{
												dp[i+arr[j]]=1;
										}
								}
						}
				}
				if(dp[m] == 1) puts("Little Deepu");
				else puts("Kate");

		}
		return 0;
}
