#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
#define ll long long int
#define MAX 1000006
ll fact[MAX],ifact[MAX];
ll modexp(ll a, ll b)
{
	ll res = 1;
	while(b){
		if(b&1) res=(res*a)%MOD;
		a=(a*a)%MOD;
		b/=2;
	}
	return res;
}
string str;
int fre[26];
int main()
{
    //cout << "Hello World!" << endl;
    int i;
    fact[0]=1;
    for(i=1;i<MAX;++i)
    {
    	fact[i] = (fact[i-1]*i)%MOD;
    	ifact[i] = modexp(fact[i],MOD-2);
    }
    ifact[0] = modexp(fact[0],MOD-2);
    int t;
    cin>>t;
    while(t--)
    {
    	memset(fre,0,sizeof(fre));
    	cin>>str;
    	for(i=0;i<str.size();++i)
    	{
    		fre[str[i]-'a']++;
    	}
    	int con,vow;
    	vow = 0;
    	con =0;
    	vow = fre[0] + fre[4] + fre[8] + fre[14] + fre[20];
    	con = str.size() - vow;
    	//cout<<"con="<<con<<" vow="<<vow<<endl;
    	if(con+1 < vow) puts("-1");
    	else
    	{
    		ll ans,tmp;
    		tmp = fact[con+1]*ifact[con+1-vow];
    		//cout<<"tmp="<<tmp<<endl;
    		tmp%=MOD;
    		tmp = (tmp*fact[con])%MOD;
    		//cout<<"tmp="<<tmp<<endl;
    		for(i=0;i<26;++i)
    		{
    			if(fre[i]!=0) tmp = (tmp*ifact[fre[i]])%MOD;
    		}
    		cout<<tmp<<endl;       
    	}
    	
    	
    }
    
    return 0;
}
