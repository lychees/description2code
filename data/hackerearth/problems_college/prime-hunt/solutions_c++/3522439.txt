#include<bits/stdc++.h>
using namespace std;
bool vis[100000002];
void cal()
{
	memset(vis,false,sizeof(vis));
	for(int i=2;i*i<=100000000;i++)
	{
		if(vis[i]==false)
		{
			for(int j=2;j*i<100000000;j++)
			{
				vis[i*j]=true;
			}
		}
	}
}
int main()
{
	ios_base::sync_with_stdio(false);
	cal();
	vis[1]=vis[0]=true;
	long long t,a,b;
	cin>>t;
	while(t--)
	{
		long long total=0;
		cin>>a>>b;
		for(long i=a;i<b;i++)
		{
			if(!vis[i])
				total++;
		}
		cout<<total<<endl;
	}
	return 0;
}
