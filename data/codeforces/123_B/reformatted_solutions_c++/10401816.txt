#include <bits/stdc++.h>
// #include <bits/stdc++.h>
#include <algorithm>
#include <cmath>
#include <iostream>
using namespace std;

int main() {
  double a, b, x1, y1, x2, y2;
  // cout << (-1)/4 << endl;
  cin >> a >> b >> x1 >> y1 >> x2 >> y2;
  int xx = fabs(floor((x1 + y1) / (2 * a)) - floor((x2 + y2) / (2 * a)));
  int yy = fabs(floor((x1 - y1) / (2 * b)) - floor((x2 - y2) / (2 * b)));
  cout << max(xx, yy) << endl;
  return 0;
}
