#include <iostream>
#include<vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
   	vector< vector<long long int> > memo(1001,vector<long long int>(1001,1));
   	for(int i=0;i<1001;i++)
   	memo[i][1]=i;
   	for(int i=1;i<1001;i++)
   	{
   		for(int j=1;j<i;j++)
   		{
   			memo[i][j]=(memo[i-1][j]+memo[i-1][j-1])%1000000007;
   		}
   	}
    while(t--)
    {
    	int n,p;
    	cin>>n>>p;
    	long long int ans=0;
    	for(int i=0;i<=p;i++)
    	ans=(ans+memo[n][i])%1000000007;
    	cout<<ans<<endl;
    }
}
