#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);
    long long tc,a,b,n;
    cin>>tc;
    while(tc--) {
    	cin>>n>>a>>b;
    	if(a == 0 || b == 0)
    	cout<<"-1\n";
    	else {
    	long long one = (n/a);
    	long long two = (n/b);
    	long int k = __gcd(a,b);
    	long long th = ((a*b)/k);
    	th = (n/th);
    	long long ans = one + two - th;
    	cout<<ans<<endl;
    	}
    }
    return 0;
}
