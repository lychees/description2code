#include <cstdio>
using namespace std;
int gcd(int a, int b) {
	if (a == 0) return b;
	return gcd(b%a, a);
}
const int MOD = 1e9+7;
int main()
{
    int t, a, b, c;
    scanf("%d", &t);
    while (t--) {
    	scanf("%d %d %d", &a, &b, &c);
    	int d = gcd(a, b);
    	d = gcd(d, c);
    	a /= d;
    	b /= d;
    	c /= d;
    	int res = a;
    	res = 1LL*res*b % MOD;
    	res = 1LL*res*c % MOD;
    	printf("%d %d\n", d, res);
    }
    return 0;
}
