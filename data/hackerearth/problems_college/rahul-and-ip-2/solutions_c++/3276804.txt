#include <bits/stdc++.h>

using namespace std;
bool isvalid(string a)
{
	if(a.size() > 1 && a[0] == '0') {
		return false;
	}	
	if(stoi(a) <= 255 && stoi(a) >= 0) {
		return true;
	}
	return false;
} 


vector<string> restoreIpAddresses(string a) {
    vector<string> as;
	if(a.size() > 12 || a.size() < 4) {
		return as;
	}
	string first, second, third, forth, final;
	for(int i = 1; i < 4; i++) {
		first = a.substr(0,i);
		if(!isvalid(first)) {
			continue;
		}
		for(int j = 1; j < 4 && (i+j) < a.size(); j++) {
			second = a.substr(i,j);
			if(!isvalid(second)) {
				continue;
			}
			for(int k = 1; k < 4 && (i+j+k) < a.size(); k++) {
				third = a.substr(i+j,k);
				forth = a.substr(i+j+k);
				if(isvalid(third) && isvalid(forth)) {
					final = first + "." + second + "." + third + "." + forth;
					as.push_back(final);
				}
			}
		}
	}
	return as;	
}

int main()
{
	string x;
	cin >> x;
	vector<string> as = restoreIpAddresses(x);
	cout << as.size() << endl;
	for(int i = 0; i < as.size(); i++) {
		cout << as[i] << endl;
	}
	return 0;
}
