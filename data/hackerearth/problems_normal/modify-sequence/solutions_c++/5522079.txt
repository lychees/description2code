#include <iostream>
using namespace std;

int main()
{ int n,i,f=1;
  cin>>n;
  int* a=new int[n];
  for(i=0;i<n;i++)
   cin>>a[i];
  for(i=0;i<n-1;i++)
  { if(a[i]==a[i+1])
  	{ a[i]=a[i+1]=0; }
  	if(a[i]<a[i+1]) 
  	{ a[i+1]=a[i+1]-a[i];
  	  a[i]=0;	
  	}
  	else if(a[i]>a[i+1]) 
  	{ f=0;
  	  break;	
  	}
  }
  if(f==0 || a[n-1]!=0)
  cout<<"NO";
  else
  cout<<"YES";
  delete[] a;   
  return 0;
}
