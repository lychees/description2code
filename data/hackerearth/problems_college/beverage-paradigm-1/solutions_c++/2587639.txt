#include<bits/stdc++.h>
#define LL long long

using namespace std;
#define loop(i,a,b) for(i=a;i<b;++i)
#define plln(n) printf("%lld\n",n)
#define s3ll(n1,n2,n3) scanf("%lld%lld%lld",&n1,&n2,&n3)
#define s2ll(n1,n2) scanf("%lld%lld",&n1,&n2)
#define sll(n) scanf("%lld",&n)



#define MOD 1000000007

LL cnt[30005]={0};
int main()
{
    LL n,i,j,k,flag=0;
    sll(n);
    LL a[n+5];
    loop(i,0,n) { sll(a[i]); cnt[a[i]]++;}
    sll(k);
    if(cnt[k]) flag=1;
    cnt[0]=1;
    loop(i,0,n)
    {
        for(j=i+1;j<n;j++)
        {
            if(((k-a[i]-a[j])>0)&&(cnt[k-a[i]-a[j]]))
            {
                flag=1;
                break;
            }
        }
        if(flag) break;
    }
    
    if(flag) cout<<"True\n";
    else cout<<"False\n";
    return 0;
}
