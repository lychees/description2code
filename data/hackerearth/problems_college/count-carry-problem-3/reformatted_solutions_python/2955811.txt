for i in range(int(raw_input())):
    a, b = map(str, raw_input().split())

    a = a[::-1]
    b = b[::-1]

    count = 0
    if (len(a) > len(b)):
        count = len(b)
    else:
        count = len(a)

    carry = 0
    for i in range(count):
        t = int(a[i]) + int(b[i])

        if (t > 9):
            carry += 1

    if (carry == 0):
        print("No carry operation")
    elif (carry == 1):
        print("1 carry operation")
    else:
        print("%d carry operations" % carry)
