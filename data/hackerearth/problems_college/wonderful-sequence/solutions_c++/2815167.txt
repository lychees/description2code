#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n;
    	cin >> n;
    	long long a[n];
    	int i;
    	for(i=0;i<n;i++)
    	{
    		cin >> a[i];
    	}
    	int count=0;
    	for(i=0;i<n-1;i++)
    	{
    		if(a[i]>=a[i+1])
    		  count++;
    	}
    	cout << count << endl;
    }
    return 0;
}
