#include <iostream>
using namespace std;

int main()
{
    long long int t,n,m;
    cin>>t;
    while(t!=0){
    long long	int a1[100]={0},i,b1[100]={0},count1=0,count2=0;
    	cin>>n>>m;
    	for(i=0;i<n;i++)
    	cin>>a1[i];
    	while(m!=0){
           count1=0,count2=0;
    		for(i=0;i<n;i++)
    		{ 
    		 if(i==0){if(a1[1]==1)b1[i]=1;else b1[i]=0;}
    		    if(i==n-1){if(a1[n-2]==1)b1[i]=1;else b1[i]=0;}
    		    if(i!=0 && i!=n-1){if(a1[i-1]==1 && a1[i+1]==1)
    		    b1[i]=1;
    		    else
    		    b1[i]=0;}
    		  
    		      
    		    }
    		     
    		  
    		for(i=0;i<n;i++){a1[i]=b1[i]; if(b1[i]==1)count1++;if(b1[i]==0)count2++;}
    		m--;
    			if(count1==n || count2==n)break;

    			
    		}
    		
    		for(i=0;i<n;i++)cout<<a1[i]<<" ";
    		cout<<"\n";
    		t--;
    	}
    	
    
    return 0;
}
