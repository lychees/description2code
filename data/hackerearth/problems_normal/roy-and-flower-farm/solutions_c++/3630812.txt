#include <bits/stdc++.h>
using namespace std;

int a[105][10005];

int main() 
{
	int t;
	cin>>t;
	while(t--)
	{
		int n, p;
		cin>>n>>p;
		int sell[n+1], cost[n+1], i, j;
		for(i=0;i<n;i++)
			cin>>sell[i]>>cost[i];
			
		for(i=0;i<=n;i++)
			a[i][0]=0;
		for(j=0;j<=p;j++)
			a[0][j]=0;
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=p;j++)
			{
				if(cost[i-1]>j)
					a[i][j]=a[i-1][j];
				else
					a[i][j]=max(a[i-1][j], a[i-1][j-cost[i-1]]+sell[i-1]);
			}
		}
		int ans=0, res;
		for(i=0;i<=p;i++)
		{
			if(a[n][i]+(p-i) > ans)
			{
				res=i;
				ans=a[n][i]+(p-i);
			}
		}
		cout<<res<<" "<<ans<<endl;
	}
	return 0;
}
