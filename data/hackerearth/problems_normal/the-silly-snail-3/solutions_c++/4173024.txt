// the silly snail
#include <bits/stdc++.h>
using namespace std;
#define ull unsigned long long
#define lli long long int

const int mod = 1e9 + 7;
vector < vector <int> > graph;

void dfs(int src)
{
	cout << src << ' ';
	for(int i = 0; i < graph[src].size(); i ++)
	dfs(graph[src][i]);
}

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	int t, n, i, x, y, z;
	cin >> t;
	while(t --)
	{
		cin >> n;
		graph.clear();
		graph.resize(100003);
		for(i = 1; i <= n; i ++)
		{
			cin >> x >> y >> z;
			if(y)
			graph[x].push_back(y);
			if(z)
			graph[x].push_back(z);
		}
		dfs(1);
		cout << '\n';
	}
	return 0;
}
