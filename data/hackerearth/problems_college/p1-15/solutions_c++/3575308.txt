#include<bits/stdc++.h>
using namespace std;
int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		string	str1,str2;
		cin>>str1;
		cin>>str2;
		int l1 = str1.length();
		int l2 = str2.length();
		int p,i,flag=0;
		for(i=0;i<l1;i++)
		{
			if(str1[i]==str2[0] && str1.substr(i,l1)==str2.substr(0,l1-i))
			{
				p = i;
				flag=1;
				break;
			}
		}
		if(flag==1)
		cout<<str1.substr(0,p)<<str2<<"\n";
		else
		cout<<str1<<str2<<"\n";
	}
}
