#include <bits/stdc++.h>
#include <iostream>
using namespace std;
#define mod 1000000007
long long  modpow(long long a, long long  b,long long  MOD) {
long long x = 1, y = a;
	while(b > 0) {
		if(b%2 == 1) {
			x=(x*y);
			if(x>MOD) x%=MOD;
		}
		y = (y*y);
		if(y>MOD) y%=MOD;
		b /= 2;
	}
//	cout <<"gg" << x << endl;
	return x;
	
}
	int n ;
    
long long  modInverse(long long  a,long long  m) {
    return modpow(a,m-2,m);
}
long long  func(long long num,long long deno){
	if(deno==0)return 0 ;
	long long   ans = (num%mod);
	ans *= (modInverse(deno,mod));
	ans %= mod; 
	return ans ;
}
int  change(bool flg[]){
	int cout = 0; 
	for(int i=0;i<n;i++){
		if(flg[i])flg[i]=false; 
		else{ flg[i]=true; break;}
	}
	for(int i=0;i<n;i++){
		if(flg[i])cout++;
	}
	if(cout%2 == 0)return cout ;
	else return change(flg);
}
int main()
{
    int t; 
    ios::sync_with_stdio(false) ; 
    cin >> t ;
    while(t--){
    	cin >> n; 
    	long long ar[n+1] ; 
    	bool flag[n+1] ;
    	for(int i=0;i<n;i++)flag[i]=false; 
    	for(int i=0;i<n;i++)cin >> ar[i] ;
    	sort(ar,ar+n) ; 
    	long long ans =0 ;
    	for(int j=0;j<pow(2,n);j++){
    		int cont= change(flag) ; 
    	///	for(int i=0;i<n;i++)cout << flag[i] << " ";cout << endl;
    		if(cont == 0)break; 
    		long long left = 1 ;
    		long long right=  1;
    		cont/=2;
    		queue<long long>as ; 
    		for(int i=0;i<n;i++){
    			if(flag[i]){
    				if(cont > 0){left*=ar[i] ;cont--;}
    				else right*=ar[i] ;
    				if(right >= mod)
    						right %=mod; 
    				as.push(modInverse(left,mod)) ; 
    				left =1 ;
    			}
    		}
    		if(right >= mod)right %=mod; 
    		right *= modInverse(left,mod) ; 
    		right %= mod; 
    		
    		while(!as.empty()){
    			right *= as.front();
    			as.pop();
    			if(right >= mod)right %=mod; 
    		}
    		ans = max(right,ans) ; 
    	}
    	cout<< ans << endl;
    	
    }
    return 0;
}
