#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int setbits(int n){
    int count=0;
    while(n>0){
        count+=(n&1);
        n=n>>1;
    }
return count;
}

int func(int arr[],int n,int k){
    vector<int>v;
    int count=0;
    for(int i=0;i<n;i++){
        int temp=setbits(arr[i]);
        v.push_back(temp);
    }
    sort(v.begin(),v.end());
    for(int i=n-1;i>=(n-k);i--)
        count+=v[i];
    return count;
}

int main()
{
    int t;
    cin>>t;
    while(t>0){
        t--;
        int n,k;
        cin>>n>>k;
        int arr[n];
        for(int i=0;i<n;i++)
            cin>>arr[i];
        cout<<func(arr,n,k)<<'\n'; 
    }
    return 0;
}
