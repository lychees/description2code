#include <bits/stdc++.h>
using namespace std;
#define ll long long

vector<int> g[100005];
int a[100005];
int b[100005];
int c[100005];
 

int main() {
  ios_base::sync_with_stdio(0);
  int n;
  cin>>n;
  for(int i=1; i<=n; i++) {
    int x;
    cin>>x;
    a[i] = x;
    for(int j=2; j*j<=x; j++) {
    	if(x%j==0) {
    		g[j].push_back(i);
    		while(x%j==0) x/=j;
    	}
    }
    if(x>1) g[x].push_back(i);
  }
  int ans = 1;
  memset(c,-1,sizeof(c));
  for(int i=1; i<=n; i++) {
    b[i] = 1;
    int x = a[i];
    for(int j=2; j*j<=x; j++) {
      if(x%j==0) {
        if(c[j]>=0) b[i] = max(b[i],b[g[j][c[j]]]+1);
        c[j]++;
        while(x%j==0) x/=j;
      }
    }
    if(x>1) {
      if(c[x]>=0) b[i] = max(b[i],b[g[x][c[x]]]+1);
      c[x]++;
    }
    ans = max(ans,b[i]);
  }
  cout<<ans<<endl;
        
  return 0;
}
