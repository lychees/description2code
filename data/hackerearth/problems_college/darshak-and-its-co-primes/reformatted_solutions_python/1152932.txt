def gcd(a, b):
    if b == 0:
        return a
    else:
        return gcd(b, a % b)


def main():
    for _ in xrange(input()):
        n, m = map(int, raw_input().split())
        if gcd(n, m) == 1:
            print "Is a Co-Prime"
        else:
            print "Not a Co-Prime"
    return 0


if __name__ == "__main__":
    main()
