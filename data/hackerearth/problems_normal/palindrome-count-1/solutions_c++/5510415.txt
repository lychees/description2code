# include <bits/stdc++.h>
using namespace std;
int dp[50][50];
void sol(string s){
	int n=s.length();
	memset(dp,0,sizeof(dp));
	for(int i=0;i<n;i++){
		dp[i][i]=1;
	}
	for(int i=n-2;i>=0;i--){
		if(s[i]==s[i+1]){
			dp[i][i+1]=1;
		}
	}
	for(int i=n-3;i>=0;i--){
		for(int j=i+2;j<n;j++){
			if(s[i]==s[j] && dp[i+1][j-1]==1){
				dp[i][j]=1;
			}
		}
	}
	int ans=0;
	for(int i=0;i<n;i++){
		for(int j=i;j<n;j++){
			if(dp[i][j]){
				ans++;
			}
		}
	}
	cout<<ans<<endl;
}
int main(){
	string s;
	cin>>s;
	sol(s);
}
