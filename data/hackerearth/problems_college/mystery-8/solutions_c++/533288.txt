#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    int test_cases;
    cin>>test_cases;
    getchar();
    while(test_cases>0)
    {
        string str;
        getline(cin,str);
        for(int i=0,j=str.size()-1;i<j;)
        {
        	while(str[i]!='*'&&(str[i]<'a'||str[i]>'z'))
        		i++;
        	while(str[j]!='*'&&(str[j]<'a'||str[j]>'z'))
        		j--;
        	if(str[i]=='*')
        		{cout<<str[j]<<endl;break;}
        	else if(str[j]=='*')
        		{cout<<str[i]<<endl;break;}
        	i++;
        	j--;
        }
        // your code here
        
        test_cases--;
    }
    return 0;
}
