#include <iostream>
#include<cstdlib>
#include<cstdio>
#include<algorithm>

using namespace std;

typedef struct state
{
	int id;
	int ar;
}state;

void loc(int i,int *s,state *a);

int main()
{
    state a[10+1],b[10+1];
    int j,i,k,p;
    int s[11];

   for(i=1;i<=10;i++)
   {
   scanf("%d",&a[i].ar);
   a[i].id=i;

   b[i]=a[i];

   }


   int min=a[1].ar;
   int maxe =a[1].ar;

   j=1;p=1;
   for(i=2;i<=10;i++)
   {
   	if(a[i].ar<min)
   	{
   		min=a[i].ar;
   		j=i;
   	}
   	if(a[i].ar>maxe)
   	{
   		maxe=a[i].ar;
   		p=i;
   	}
   }

   s[1]=j;

   int pivot, l,u;

   loc(j,s,a);


   for(i=1;i<11;i++)
    if(s[i]==p)
   	   pivot =i;

   l=1;u=10;

   for(i=1;i<10;i++)
   {
   		if(l==pivot || u==pivot)break;
   		if(a[s[l]].ar<a[s[u]].ar)
   		{
   		printf("%d ",a[s[l]].ar);
   		l++;
   		}
   		else
   		{
   			printf("%d ",a[s[u]].ar);
   			u--;
   		}

   	}
   	
   	if(l==pivot)
   	{
   		while(u!=pivot)
   		{printf("%d ",a[s[u]].ar);
   		u--;}
   		
   	}

   	else if(u==pivot)
   	{
   		while(l!=pivot)
   		{printf("%d ",a[s[l]].ar);
   		l++;}
   	}



    return 0;
}

void loc(int i,int * a,state * s)
{
	int j,k,m;



	k=(i+3);
	if(k>10)k=k%10;

	j=i+7;
	if(j>10)j=j%10;

	if(s[k].ar<=s[j].ar)
	{ 	a[2]=k;
		for(m=3;m<=10;m++)
		{
			k=k+3;
			if(k>10)k=k%10;

			a[m]=k;
		}
	}
	else
	{
		a[2]=j;
		for(m=3;m<=10;m++)
		{
			j=j+7;
			if(j>10)j=j%10;

			a[m]=j;
		}
	}




}
