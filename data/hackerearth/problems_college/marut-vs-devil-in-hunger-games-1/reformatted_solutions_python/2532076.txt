import string
import collections

points = raw_input().split()
Q = int(raw_input())

map = {
    k: int(
        points[v]) for k, v in zip(
            string.ascii_lowercase, range(
                0, len(points)))}

mpoints = 0
dpoints = 0
while (Q > 0):
    M = raw_input().strip()
    D = raw_input().strip()
    mcounter = collections.Counter(M)
    dcounter = collections.Counter(D)
    mlist = (mcounter - dcounter).elements()
    for mchar in mlist:
        mpoints += map[mchar]
    dlist = (dcounter - mcounter).elements()
    for dchar in dlist:
        dpoints += map[dchar]
    Q -= 1

if (mpoints == dpoints):
    print "Draw"
elif (mpoints > dpoints):
    print "Marut"
else:
    print "Devil"
