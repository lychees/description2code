#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long int n,i,j,l,x,m=0,c=0;
    cin>>n;
    long long int b[n][26];
    for(i=0;i<n;i++){
    	for(j=0;j<26;j++){
    		b[i][j]=0;
    	}
    }
for(i=0;i<n;i++){
	string a;
	cin>>a;
	l=a.length();
	for(j=0;j<l;j++){
		x=(int)a[j]-97;
		b[i][x]++;
	}
}
for(i=0;i<26;i++){
	c=0;
	for(j=0;j<n;j++){
		if(b[j][i]>0)
		{c++;}
	}
	if(c==n){
		m++;
	}
	
}
cout<<m;

    return 0;
}
