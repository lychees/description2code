#include <iostream>
using namespace std;

int main()
{
    int t,c,a,b,temp,i;
    cin>>t;
    while(t--)
    {   a=2;b=3;
    	cin>>c;
    	if(c==1)
    	cout<<4<<endl;
    	else if(c==2)
    	cout<<9<<endl;
    	else
    	{ 
    		for(i=3;i<=c;++i)
    		{
    			temp=a+b;
    			a=b;
    			b=temp;
    		}
    		cout<<b*b<<endl;
    		
    	}
    }
    return 0;
}
