#include <bits/stdc++.h>
using namespace std;

int F(int a)
{
	if(a < 0)
	a *= -1;
	return a;
}

int calc(int a,int b,int c,int d)
{
	if( (a + b)%2 != (c + d)%2 ) return -1;
	if(F(a-c) == F(b - d)) return 0;
	return 1;
}

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int x1,y1,x2,y2;
		cin >> x1 >> y1 >> x2 >> y2;
		int ans = calc(x1,y1,x2,y2);
		cout << ans << "\n";
	}
	return 0;
}
