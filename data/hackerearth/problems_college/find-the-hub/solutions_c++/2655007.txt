#include <bits/stdc++.h>
using namespace std;


typedef vector<int> vi;

int isPrime(int x)
{
	for(int i=2;i<=x/2;i++)
		if(x%i==0)
			return false;
	return true;
}

int main()
{
    
    int n;
    cin >> n ;
    int count=0;
    int ans=-1;
    
    while(n--)
    {
    	int x;
    	cin >> x;
    
    	if(x<ans)
    		continue;
    	if(isPrime(x))
    	{ 
    		if(x>ans)
    		{
    			count=1;
    			ans=x;
    		}
			
			else
			{
				count++;
			}
    	}
    	
    }
    cout << ans << '\n' << count ;
    return 0;
}
