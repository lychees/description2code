pair<pair<LL,LL>,LL> extended_gcd(LL a, LL b)
{
      pair<pair<LL,LL>,LL> ret;
      if(!b)
      {
            ret.first = mp(1,0);
            ret.second = a;
      }
      else
      {
            pair<pair<LL,LL>,LL> temp = extended_gcd(b,a%b);
            ret.first = mp(temp.first.second, temp.first.first - (a/b)*(temp.first.second));
            ret.second = temp.second;
      }
      return ret;
}

				pair<pair<LL,LL> , LL > abc = extended_gcd(a,p);
				LL ans = (abc.first.first + p)%p;
				ans += a;
