def main(g, n):
    output = []
    for i in xrange(1, n):
        output.append(g*2)
    output.append(g*3)
    print ' '.join(map(str, output))


t = input()
for i in xrange(t):
    g, n = map(long, raw_input().split())
    main(g, n)
