from operator import itemgetter

a1,a2=map(float,raw_input().split())
b1,b2=map(float,raw_input().split())
c1,c2=map(float,raw_input().split())
 
x1=a1+b1-c1
y1=b1+c1-a1
z1=c1+a1-b1

x2=a2+b2-c2
y2=b2+c2-a2
z2=c2+a2-b2

points=[(x1,x2),(y1,y2),(z1,z2)]
points.sort(key=itemgetter(0,1),reverse=False)

print ("%.4f %.4f" %(points[0][0],points[0][1]))
