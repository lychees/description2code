#include<bits/stdc++.h>
using namespace std;
bool f(int i,int j,char a[])
{
	if(a[i]=='1' || a[j]=='1')
		return 0;
	while(i!=j)
	{
		if(i>j)
		{
				i/=2;
				if(a[i]=='1')
					return 0;
		}
		else
		{
				j/=2;
				if(a[j]=='1')
				return 0;
		}
	}
	return 1;
}
int main()
{
		int n,q,i,j;
		char a[100005];
		cin>>n>>q;
		cin>>(a+1);
		while(q--)
		{
			cin>>i>>j;
			if(f(i,j,a))
				puts("Yes");
			else
				puts("No");
		}
		return 0;
}
