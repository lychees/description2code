#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n,r,c,i,s;
    scanf("%d",&n);
    int a[n];
    for(i=0;i<n;i++)
    	scanf("%d",&a[i]);
    scanf("%d",&r);
    std:: sort(a,a+n);
    s=0;
    c=0;
    for(i=n-1;i>=0;i--)
    {
    	s+=a[i];
    	c++;
    	if(s>=r)
    	break;
    }
    printf("%d",c);
    return 0;
}
