#include <iostream>
#include <cmath>
using namespace std;

int a[100001];
int b[100001];

int main()
{
	a[0] = 0;
	b[0] = 0;
	
	for (int i = 1; i < 100001; i++) {
		int j = i;
		int ans = 0;
		while (j > 0) {
			if (j % 2 == 1) {
				ans++;
			}
			j = j / 2;
		}
		if (ans >= b[i - 1]) {
			b[i] = ans;
			a[i] = i;
		} else {
			b[i] = b[i - 1];
			a[i] = a[i - 1];
		}
	}
	a[0] = 1;
    int t;
    cin >> t;
    while (t--) {
    	int n;
    	cin >> n;

		cout << a[n] << endl;
    }
    return 0;
}
