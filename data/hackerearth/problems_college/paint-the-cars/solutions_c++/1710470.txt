#include<cstdio>
#include<iostream>
#include<climits>

int a[20][3];

inline int Read()
{
	register int c=getchar();
	int x=0;
	for(;(c<48 || c>57);c=getchar());
	for(;c>47 && c<58;c=getchar())
		x=(x<<1)+(x<<3)+c-48;
	return x;
}

int solve(int n,int level,int prev,int sum)
{
	if(level==n-1)
	{
		int min=INT_MAX;
		for(int i=0;i<3;i++)
		{
			if(i==prev)
				continue;
			min=std::min(min,a[level][i]);
		}
		return min;
	}
	int Min=INT_MAX;
	for(int i=0;i<3;i++)
	{
		if(i==prev && level!=0)
			continue;
		Min=std::min(Min,solve(n,level+1,i,sum+a[level][i])+a[level][i]);
	}
	return Min;
}

int main()
{
	int t=Read();
	while(t--)
	{
		int n=Read();
		for(int i=0;i<n;i++)
			a[i][0]=Read(),a[i][1]=Read(),a[i][2]=Read();
		if(n==1)
		{
			printf("%d\n",std::min(a[0][0],std::min(a[0][1],a[0][2])));
			continue;
		}
		printf("%d\n",solve(n,0,0,0));
	}
	return 0;
}
