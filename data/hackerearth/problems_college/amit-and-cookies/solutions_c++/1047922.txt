#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int n,k;
    int *arr;
    cin>>n>>k;
    arr=new int[n];
    for(int i=0;i<n;i++)
    	cin>>arr[i];
    sort(&arr[0],&arr[n]);
  
    int temp=0,i;
    for(i=0;temp<=k;i++)
    	temp+=arr[i];
    cout<<i-1<<endl;
    return 0;
}
