
t = int(raw_input())
for i in xrange(t):
    tmp = map(int, raw_input().split())
    n, arr = tmp[0], tmp[1:]
    arr.sort()
    mx = 0
    mi = 9999999999999999
    while (arr):
        t1 = arr.pop(0)
        t2 = arr.pop(-1)
        if t1 + t2 > mx:
            mx = t1 + t2
        if t1 + t2 < mi:
            mi = t1 + t2
    print mx - mi
