#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		vector<int>v;
		vector<int>v1;
		long long int n,x,total=0,cnt=0;
		cin>>n;
		for(int i=0;i<n;i++){
			cin>>x;
			total+=x;
			v.push_back(x);
		}
		
		for(int i=0;i<n;i++){
			cin>>x;
			v1.push_back(x);
		}
		sort(v1.begin(),v1.end());
		for(int i=n-1;i>-1;i--)
		{
			total-=v1[i];
				cnt++;
			if(total<=0)
				break;
		}
		cout<<cnt<<endl;
	}

}
