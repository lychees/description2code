#include <iostream>
using namespace std;

int main()
{
    int n,num,den;
    cin>>n;
    for (int i=0;i<n;i++)
    {
    	int k=0;
    	string st;
    	cin>>st;
    	while(st[k++]!='\0')
    	{
    		;
    	}
    	den=k-1;
    	k=0;
    	if ((st[k]=='w')&&(st[k+1]=='w')&&(st[k+2]=='w')&&(st[k+3]=='.'))
    	{
    		k+=4;
    	}
    	num=den-k;
    	while (st[k+4]!='\0')
    	{
    		if (st[k]=='a'||st[k]=='e'||st[k]=='i'||st[k]=='o'||st[k]=='u')
    		{
    			num--;
    		}
    		k++;
    	}
    	cout<<num<<"/"<<den<<endl;
    }
    return 0;
}
