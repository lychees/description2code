#include<iostream>

int a[1000];

{

    {

        for(j=0;j<20;j++)
            a[j]=0;
        f=i=0;
        k=2;
        while(h>1)
        {
            while(h%k==0&&h>1)
            {
                a[i]++;
                h/=k;
                f=1;
            }

                i++;
                f=0;
            }
            k++;

        }

}
