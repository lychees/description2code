#include <iostream>
using namespace std;

int main()
{
	int n;
    long long int ans=0;
    long long int i,j,k;
     cin>>n;
   for(i=1;i<=n;i++)
   {
   	for(j=1;j<=n;j++)
   	{
   		if(i<j && i!=j)
   		{
   			for(k=1;k<=n;k++)
   			{
   				if(j>k && j!=k && i!=k)
   				ans++;
   			}
   		}
   	}
   }
   cout<<ans;
}
