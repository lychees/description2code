#include <bits/stdc++.h>
#include <stdio.h>    
#include <string.h>
using namespace std;
#define mod 1000000000
#define lim 200003

typedef struct node{
    long long ind;
    long long p;
}nody;
long long diff;

bool comp(nody i,nody j) { return (i.p-(diff-i.ind) < j.p-(diff-j.ind)); }
nody A[lim];

//long long B[lim];

int main(){
    long long n,i,j,k;
    cin>>n;
    for(i=0;i<n;i++){
      cin>>j;
      A[i].ind=i;
      A[i].p=j;
    }
    diff=n-1;

    sort(A,A+n,comp);
    for(i=0;i<n;i++){
      A[i].p=A[i].p+(A[i].ind-i);
    }
    k=1;
    for(i=1;i<n;i++){
      if(A[i].p<A[i-1].p){
        k=0;
        break;
      }
    }
    if(k==0) cout<<":(";
    else for(i=0;i<n;i++) cout<<A[i].p<<" ";
    return 0;
}
