#include <iostream>
using namespace std;

int main()
{
    int size;
    cin>>size;
    int team[size];
    for(int i=0; i<size; i++)
    {
    	cin>>team[i];
    }
    int maxValue=0;
    int maxValueTill=0;
    
    for(int i=0; i<size; i++)
    {
    	maxValue=maxValue + team[i];
    	if(maxValue<0)
    	{
    		maxValue=0;
    	}
    	if(maxValue>maxValueTill)
    	{
    		maxValueTill=maxValue;
    	}
    }
    cout<<maxValueTill;
}
