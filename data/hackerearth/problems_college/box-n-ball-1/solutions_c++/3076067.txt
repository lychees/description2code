#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int t,a,b,c,small;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>c;
    	
    	 if((a<=b) && (a<=c)){
        small = a;
         }else if((b<=c) && (b<=a)){
        small = b;
        }else{
       small = c;
       }
       
    	if(small%2==0)
    	printf("%ld.0\n",small/2);
    	else
    	printf("%ld.5\n",small/2);
    }
    return 0;
}
