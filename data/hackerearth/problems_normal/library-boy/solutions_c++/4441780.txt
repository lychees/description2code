#include <iostream>
#include<cstdio>
using namespace std;
int main()
{
    int n;
    scanf("%d",&n);
    string s;
    int arr[26],count=0,i;
    std::fill(arr,arr+26,0);
    for(i=0;i<n;i++)
    {
        cin>>s;
        int c=(int)s[0];
        arr[c-97]=arr[c-97]+1;
    }
    for(i=0;i<26;i++)
    {
        if(arr[i]%10==0)
            count=count+arr[i]/10;
        else if(arr[i]>0)
            count=count+arr[i]/10+1;
    }
    cout<<count<<"\n";
    return 0;
}
