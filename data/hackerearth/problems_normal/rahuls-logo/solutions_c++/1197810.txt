long long int exp_modulo(long long int a,long long int b,long long int c){
    long long int x=1,y=a; // long long is taken to avoid overflow of intermediate results
    while(b > 0){
        if(b%2 == 1){
            x=(x*y)%c;
        }
        y = (y*y)%c; // squaring the base
        b >>= 1;
    }
    return x%c;
}

void printOdd(ll n)
{
    ll i, j, k=1;
    for (i=n-1; i>=0; i--)
    {
        for (j=0; j<i; j++)
            printf(" ");

        for(j=0; j<k; j++)
        {
            if (j==k-1 && i%2==0)
                printf("/");
            else
                printf("/ ");
        }

//        if (i%2)
//            printf(" ");

        for(j=0; j<k; j++)
        {
            if (j==0 && i%2==0)
                printf("\\");
            else
                printf(" \\");
        }


        if (i%2)
            k+=1;
        printf("\n");
    }



    for (i=0; i<n; i++)
    {
        printf("  ");
        for (j=0; j<i; j++)
            printf(" ");

        for(j=0; j<k; j++)
        {
            if (j==k-1 && i%2==0)
                printf("\\");
            else
                printf("\\ ");
        }

//        if (i%2)
//            printf(" ");

        for(j=0; j<k; j++)
        {
            if (j==0 && i%2==0)
                printf("/");
            else
                printf(" /");
        }


        if (i%2==0)
            k--;
        printf("\n");
    }
}
