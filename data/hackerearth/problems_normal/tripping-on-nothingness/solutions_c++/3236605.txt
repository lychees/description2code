using namespace std;
ll gcd(ll A, ll B)
{
    if(B==0) return A;
    else return gcd(B,A%B);
}
int main()
{
    /*freopen("1in.txt", "r", stdin);
    freopen("1out.txt", "w", stdout);*/
    ll S,Q;
    ll A;
    sll(S);sll(Q);
    ll gcds[Q+10];
    int last = 0;
    fill(gcds,gcds+Q+10,-1);
    while(Q--)
    {
        sll(A);
        ll ans = gcd(max(A,S),min(A,S));
        ll* p = find(gcds,gcds+last,ans);
        if(p == gcds+last)
        {
            gcds[last] = ans;
            last++;
            printf("%lld\n",ans);
        }
        else puts("-1");
    }
    return 0;
}
