n,k = map(int, raw_input().split())
arr = sorted(map(int, raw_input().split()))

lo, hi = 0, 10**7

while(lo < hi):
	mid = lo + (hi-lo)//2
	
	count, pos = 1, arr[0] + mid
	for x in xrange(1,n):
		if arr[x] - pos > mid:
			count += 1
			pos = arr[x] + mid
	
#	print lo, mid, hi, count # debug
		
	if count > k:
		lo = mid + 1
	else:
		hi = mid

print lo
