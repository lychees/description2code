#include <iostream>
#include <set>

using namespace std;
int main()
{
	std::ios::sync_with_stdio(false);
	//http://stackoverflow.com/questions/1042110/using-scanf-in-c-programs-is-faster-than-using-cin
	
	int no_of_features;std::cin>>no_of_features;
	std::set<int> features;
	
	for(int i=0;i<no_of_features;i++)
	{
		int feature;std::cin>>feature;
		features.insert(feature);
	}
	
	int no_of_chicks;
	std::cin>>no_of_chicks;
	int ok_chicks=0;
	
	for(int i1=0;i1<no_of_chicks;i1++)
	{
		int matches=0;
		
		while(1)
		{
			int chick_feature;
			std::cin>>chick_feature;
			bool is_there=features.find(chick_feature)!=features.end();
			if(is_there)
			{
				matches++;
			}
			
			char next=std::cin.peek();
			
			if((next=='\n') || (next==EOF))
			{
				break;
			}
		}
		
		if(matches==no_of_features)
		{
			ok_chicks++;
		}
	}
	
	cout<<ok_chicks;
    
    return 0;
}
