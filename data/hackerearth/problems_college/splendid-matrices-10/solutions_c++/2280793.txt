int f21(int *mat, int i, int j, int size, int count)
{
	*(mat+i*size+j) = count++;
	*(mat+i*size+(j+1)) = count++;
	*(mat+(i+1)*size+j) = count++;
	*(mat+(i+1)*size+(j+1)) = count++;
	return count;	
}
