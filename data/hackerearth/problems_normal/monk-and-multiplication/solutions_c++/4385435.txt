#include <bits/stdc++.h>
#define ll long long 
using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(0);
	
 	int n;
 	cin>>n;
 	int arr[n][4];
 	
 	int a,b,c;
 	cin>>a>>b>>c;
 	arr[0][0]=0,arr[1][0]=0,arr[2][0]=3;
 	arr[2][1]=a,arr[2][2]=b,arr[2][3]=c;
 	
 	/*for(int i=0;i<n;i++)
 		cin>>arr[i];
 	
 	ll int p=1;*/
 	
 	
 	for(int i=3;i<n;i++)
 	{
 		int x;
 		cin>>x;
 		
 		int temp[5];
 		for(int j=1;j<4;j++)
 			temp[j]=arr[i-1][j];
 		temp[4]=x;
 		sort(temp+1,temp+5);
 		
 		arr[i][0]=3;
 		for(int j=4;j>=2;j--)
 			arr[i][j-1]=temp[j];
 			
 	}
 	
 	
 	for(int i=0;i<n;i++)
 	{
 		ll int p=1;
 		if(arr[i][0]<3)
 			cout<<"-1\n";
 		else
 		{
 			p*=arr[i][1];
 			p*=arr[i][2];
 			p*=arr[i][3];
 			cout<<p<<endl;
 		}
 	}
 	
    return 0;
}
