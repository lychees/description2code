#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define lli long long int 
#define s(x) scanf("%lld",&x)
#define p(x) printf("%lld\n",x)
lli aa[1000005],ss[1000005];
int main()
{
    lli n,q,l,r,ans,i;
    s(n);
    for(i=1;i<=n;i++)
    {
    	s(aa[i]);
    }
    ss[1]=aa[1];
    for(i=2;i<=n;i++)
    {
    	ss[i]=ss[i-1]+aa[i];
    }
    s(q);
    while(q--)
    {
    	s(l);
    	s(r);
    	if(l>1)
    	{
    		ans=ss[r]-ss[l-1];
    	}
    	else
    	{
    		ans=ss[r];
    	}
    	p(ans);
    }
    return 0;
}
