#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <iomanip>
#include <iostream>
using namespace std;
int a[5555];
int main() {
  int n, m;
  cin >> n >> m;
  for (int i = 0; i < n; i++)
    cin >> a[i];
  sort(a, a + n);
  int ans = 0;
  for (int i = 0; i < n; i++)
    ans += (a[i] != a[(i + (n / 2)) % n]);
  cout << ans << endl;
  for (int i = 0; i < n; i++)
    cout << a[i] << ' ' << a[(i + (n / 2)) % n] << endl;
  return 0;
}
