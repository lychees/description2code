#include<bits/stdc++.h>
using namespace std;
 

int main()
{
    long long int n,ans,x;
    cin>>n;
   vector <long long int> v;
    for(int i=0; i<n; i++)
    {
    	cin>>x;
    	v.push_back(x);
    }
    sort(v.begin(),v.end());
    
    ans=max((v[0]*v[1]),(v[n-1]*v[n-2]));
    cout<<ans<<endl;
    
    
    return 0;
}
