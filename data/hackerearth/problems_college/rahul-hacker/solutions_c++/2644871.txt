#include <iostream>
using namespace std;

int flip(int n) {
	int a = 0;
	while (n!=0) {
		a = (a*10) + (n%10);
		n /= 10;
	}

	return a;
}

int main()
{
    int t;
    cin >> t;
    while (t--) {
    	int a,b;
    	cin >> a >> b;

    	int n = flip(a) + flip(b);

    	cout << flip(n) << endl;
    }
    return 0;
}
