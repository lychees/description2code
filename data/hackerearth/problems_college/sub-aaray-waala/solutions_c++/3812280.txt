ll power(ll p,ll q)
{
	ll sum=1;
	while(q)
	{
		if(q&1)
		sum=(sum*p)%mod;
		q>>=1;
		p=(p*p)%mod;
	}
	return sum;
}

int main()
{
	int t;

	{
		int n;

		map<ll,ll> mp;
		for(int i=0;i<n;i++)
		{
			ll x;
			cin>>x;
			mp[x]++;
		}
		
		ll sum=0;
		map<ll,ll>::iterator it;
		for(it=mp.begin();it!=mp.end();it++)
		{
			sum++;
		}
		cout<<(power(n,sum)%mod)<<endl;
	}
    
    return 0;
}
