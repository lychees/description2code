#include <bits/stdc++.h>
using namespace std;

typedef vector<int> vi;


int main()
{
    int n;
    cin >> n ;
   	vi a(n+1,0);
   	
   	for(int i=2;i<=n;i++)
   	{
   		if(a[i]==0)
   		for(int j=i;j<=n;j+=i)
   			a[j]++;
   	}
   	
   	unordered_map<int,int> m;
   	for(int i=2;i<=n;i++)
   		m[a[i]]++;
   	int q;
   	cin >> q;
   	while(q--)
   	{
   		int k;
   		cin >> k;
   		cout << m[k] << '\n';
   	}
   	
   	
   	
    return 0;
}
