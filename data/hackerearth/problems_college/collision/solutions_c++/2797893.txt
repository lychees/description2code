#include <iostream>
#include<map>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int arr[n];
    	for(int i=0; i<n; i++)
    	cin>>arr[i];
    	map<int,int> num;
    	int count = 0;
    	for(int i=0; i<n; i++)
    	{
    		if(num[(arr[i]%10)]==0)
    		num[(arr[i]%10)] = 1;
    		else
    		count++;
    	}
    	cout<<count<<"\n";
    }
    return 0;
}
