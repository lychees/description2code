#include <bits/stdc++.h>
#define ll long long
using namespace std;
string s[100],v;
ll l[100],j;
int main()
{
    int t,n,m,i;
    cin>>t;
    while(t--)
	{	
		cin>>n>>m;
		for(i=1;i<=n;i++)
		{
			cin>>s[i];
			l[i]=l[i-1]*2+(s[i].size());
		
		}
		while(m--)
		{
			cin>>j;
			for(i=n;i;i--)
			{
				if(j>=2*l[i-1])
				{
					cout<<s[i][j-2*l[i-1]];
					break;
				}
				else if(j>=l[i-1])
				j=2*l[i-1]-j-1;
			}
		}
		//memset(l,0,90*8);
		cout<<endl;
	}   

    return 0;
}
