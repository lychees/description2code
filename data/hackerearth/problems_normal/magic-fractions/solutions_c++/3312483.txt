#include <iostream>
using namespace std;
#define ll long long

bool prime[505];
void sieve()
{
	prime[1]=false;
	for(int i=2;i<505;i++)
	prime[i]=true;
	for(int i=2;i*i<505;i++)
	{
		if(prime[i])
		{
			for(int j=i+i;j<505;j+=i)
			prime[j]=false;
		}
	}
}

int main()
{
	sieve();

	int dp[505];
	long long  p2[505];
	ll n;
	p2[0]=1;
	for(int i=1;i<100;i++)
	p2[i]=p2[i-1]*2;
	dp[0]=0;
	for(int i=1;i<505;i++)
	{
		if(prime[i]==true)
		dp[i]=dp[i-1]+1;
		else
		dp[i]=dp[i-1];
   
    }
    cin>>n;
    ll sum=0;
    if(n==1|| n==0)
    {
    cout<<0;
    }
    else
    {
    	for(int i=2;i<=n;i++)
    	{
    		sum+=p2[dp[i]-1];
    		
    	}
    	 cout<<sum;
    }
   
    
}
