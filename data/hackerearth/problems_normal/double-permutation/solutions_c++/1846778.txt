typedef long long int uli;
using namespace std;

const int mx=1e5+10;
const int mxt=mx+mx;
const uli mod=1e9+7;
const uli i4=250000002;
int d[mxt];
int ft[mxt];
int cnt[mx];
void upd(int i,int v){
   for(;i<mxt;i+=(i&-i))
      ft[i]+=v;
}
uli qry(int i){
   if(i<=0)return 0;
   int ans=0;
   for(;i>0;i-=(i&-i))
      ans+=ft[i];
   return ans;
}
int main(){
//   freopen("data.in","r",stdin);
   int n;
   scanf("%d",&n);
   for(int i=0;i<n+n;i++)
      scanf("%d",d+i);   
   uli pw=1;
   for(int i=0;i<n;i++) pw=(pw+pw)%mod;
   uli ans=0;
   memset(ft,0,sizeof ft);
   memset(cnt,0,sizeof cnt);
   for(int i=n+n-1;i>=0;i--){
      ans+=2ll*qry(d[i]-1)+qry(d[i]-n-1)+qry(n+d[i]-1)+cnt[d[i]];
      ans%=mod;
      cnt[d[i]]++;
      upd(d[i],1);
   }
   ans=ans*i4%mod;
   ans=ans*pw%mod;
   cout<<ans<<endl;
   return 0;
}
