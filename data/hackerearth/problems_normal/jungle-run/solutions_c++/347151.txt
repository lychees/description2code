#include<iostream>
#include<cstring>
#include<vector>
#include<algorithm>
#include<map>
#include<set>
#include<queue>
#include<stack>
#include<cmath>

using namespace std;

#define all(a) 						a.begin(),a.end()
#define pb							push_back
#define sz(a)						sizeof(a)
#define mp 							make_pair
#define fir							first
#define sec							second
#define ll							long long
#define pii							pair<int,int>
#define sd(z)                       scanf("%d",&z);
#define sc(z)                       scanf("%c",&z);
#define pd(z)                       printf("%d",z);
#define pdn(z)                      printf("%d\n",z);

int dx[] = {1,0,-1,0};
int dy[] = {0,1,0,-1};


int main()
{
    int n,sx,sy,ex,ey;
    sd(n);
    
    char a[n+2][n+2];
    for(int i=0;i<=n+1;i++)
        for(int j=0;j<=n+1;j++)
            a[i][j]='T';
    
    for(int i=1;i<=n;i++)
        for(int j=1;j<=n;j++)
        {
            cin>>a[i][j];
            if(a[i][j]=='S')
            {
                sx=i;
                sy=j;
            }
            else if(a[i][j]=='E')
            {
                ex=i;
                ey=j;
            }
        }
    
    queue<pair<pii, int> > q;
    q.push(mp(mp(sx,sy),0));
    bool vis[n+2][n+2];
    memset(vis, false, sz(vis));
    
    while(!q.empty())
    {
        pair<pii, int> u = q.front();
        q.pop();
        
        if(u.fir.fir==ex&&u.fir.sec==ey)
        {
            pdn(u.sec);
            break;
        }
        
        for(int i=0;i<4;i++)
        {
            if(a[u.fir.fir+dx[i]][u.fir.sec+dy[i]]!='T'&&!vis[u.fir.fir+dx[i]][u.fir.sec+dy[i]])
            {
                q.push(mp(mp(u.fir.fir+dx[i],u.fir.sec+dy[i]),u.sec+1));
                vis[u.fir.fir+dx[i]][u.fir.sec+dy[i]]=true;
            }
        }
    }
            
    
	return 0;
}
