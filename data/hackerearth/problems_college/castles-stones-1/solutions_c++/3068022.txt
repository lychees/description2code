#include<bits/stdc++.h>
using namespace std;
int main()
{
int t,cnt,n,m,i,a[1000],j,temp;
cin>>t;
while(t--)
{
cnt=0;
cin>>m;
cin>>n;
for(i=0;i<n;i++)
cin>>a[i];
for(i=1;i<n;i++)
{
for(j=0;j<n-i;j++)
{
if(a[j]<a[j+1])
{
temp=a[j];
a[j]=a[j+1];
a[j+1]=temp;
}
}
}
for(i=0;i<n;i++)
{
for(j=i+1;j<n;j++)
{
if((a[i]+a[j])<=m && a[i]!=0 && a[j]!=0)
{
//cout<<a[i]<<"and"<<a[j]<<endl;
a[i]=0;
a[j]=0;
cnt++;
break;
}
if(j==n-1 && a[i]<=m && a[i]!=0)
{
//cout<<a[i]<<endl;
cnt++;
a[i]=0;
}
}
if(i==n-1 && a[i]!=0 && a[i]<=m)
{
//cout<<a[i]<<endl;
cnt++; 
}
}
cout<<cnt<<endl;
}
return 0;
}
