import string
t = input()
for i in range(t):
    x = raw_input()
    hist = {}
    for letter in string.lowercase:
        hist[letter]=0
    for letter in x:
        hist[letter]+=1
    freqa = hist['a']
    b=True
    for letter in hist:
        if letter == 'a':continue
        if hist[letter]!=freqa:
            print "No"
            b=False
            break
    if b:print "Yes"
