vector<pii> pr,ls;

bool way1(pii a, pii b) {
	return (a.F<b.F);
}
bool way2(pii a, pii b) {
	if(a.S==b.S) return (a.F<b.F);
	return (a.S>b.S);
}
ll func(ll bal) {
	rep(i,0,pr.size()) {
		if(bal>=pr[i].F) bal+=(pr[i].S-pr[i].F);
		else return false;
	}
	rep(i,0,ls.size()) {
		if(bal>=ls[i].F) bal+=(ls[i].S-ls[i].F);
		else return false;
	}
	return true;
}
ll bsearch(ll l, ll h) {
	ll mid;
	while(h-l>1) {
		mid=l+(h-l)/2;
		if(func(mid)) h=mid;
		else l=mid;
	}
	return h;
}
int main() {
	int n,t,x;
	rdi(n);

		rdi(t); rdi(x);
		if(t<=x) pr.pb(mp(t,x));
		else ls.pb(mp(t,x));
	}
	sort(pr.begin(),pr.end(),way1);
	sort(ls.begin(),ls.end(),way2);
	ll ans=bsearch(0,1000000000000000LL);
	prlong(ans);
}
