#include <iostream>
using namespace std;
int a[100005];
void solve()
{
	
	for(int i=1;i<100002;i++)
	{	int k=0;
		for(int j=0;j<18;j++)
		{
			if( (i&1<<j) && !(i&1<<(j+1)) && (i&1<<(j+2))) { k++; j++;}
				
			}
		//	cout<<i<<" "<<k<<" "<<endl;
			a[i]=k;
		}
	}
	
	
	
	
	

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    solve();
    while(t--)
    {
    	int ans=0,r,k;
    	cin>>r>>k;

    	for(int i=1;i<=r;i++)
    	{
    		if(a[i]>=k)  ans++;
    	}
    	
    	cout<<ans<<endl;
    }
    return 0;
}
