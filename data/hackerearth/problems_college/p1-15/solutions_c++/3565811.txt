#include<bits/stdc++.h>
using namespace std;
typedef pair<int,int> pp;
typedef long long ll;

string a,b;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int i;
		cin>>a>>b;
		int la=a.length(),lb=b.length();
		for(i=min(la,lb);i>0;i--)
		{
			int j=0;
			int s=la-i;
			while(s<la && a[s]==b[j] && j<lb)
			{
				s++; j++;
			}
			if(s==la || j==lb)
				break;
		}
		cout<<a+b.substr(i,lb-i)<<endl;

	}
	return 0;
}
