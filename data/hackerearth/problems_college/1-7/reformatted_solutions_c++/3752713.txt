#include <algorithm>
#include <iostream>
#include <stdio.h>
#include <string.h>

using namespace std;

char s[1005], l;

int main() {
  int t;
  scanf("%d", &t);
  for (int i = 0; i < t; i++) {
    scanf("%s %c", s, &l);
    int c = 0;
    for (int j = 0; s[j] != '\0'; j++) {
      if (s[j] == l)
        c++;
    }
    printf("%d\n", c);
  }
  return 0;
}
