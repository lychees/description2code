#include <stdio.h>
#include<string.h>
int find(int rem, int pover)
{
int i,val=1,res;
for(i=1;i<=pover;i++)
val=val*rem;
res=val%10;
printf("%d\n",res);
}
int chk( int rem, char b[])
{
unsigned long len=strlen(b);
if(len<=1)
return((b[0]-'0')%rem);
int val,val2,num;
switch(rem)
{

case 1 : return 0;
break;
case 2 : val=b[len-1]-'0';
return(val%2);

break;
case 4 : val=b[len-1]-'0';
val2=b[len-2]-'0';
num=val2*10+val;
return(num%4);
break;
}
}

int main()
{
int t,i,x,pover;
char a[10000];
char b[10000];
int len;
int cycle[10]={1,1,4,4,2,1,1,4,4,2};
scanf("%d",&t);
for(i=0;i<t;i++)
{
scanf("%s",a);
len=strlen(a);
scanf("%s",b);
int rem=a[len-1]-'0';
x=cycle[rem];
pover=chk(x,b);
if(pover==0)
pover=x;
find(rem,pover);
}

}
