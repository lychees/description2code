#include <iostream>
#include <algorithm>
#include<cstdio>
int a[100000], b[100001];
 
int lis(int n) {
	b[0] = 0;
	int len = 0;
 
	for (int i = 0; i < n; i++) {
		if (a[i] <= 0) 
		continue;
		int low = 0, up = len, key;
		while (low < up) {
			key = (low + up >> 1) + 1;
			if (b[key] <= a[i])
				low = key;
			else
				up = key - 1;
		}
 
		if (low == len) len++;
		b[low + 1] = a[i];
	}
	return len;
}
 
int main() {
	int t;
	scanf("%d", &t);
	while (t--) {
		int n;
		scanf("%d", &n);
		for (int i = 0; i < n; i++) {
			scanf("%d", &a[i]);
			a[i] -= i;
		}
		printf("%d\n", n - lis(n));
	}
	return 0;
}
