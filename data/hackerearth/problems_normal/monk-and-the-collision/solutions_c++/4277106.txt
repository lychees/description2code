#include<bits/stdc++.h>
using namespace std;

bool hashtable[10] = {false};

int hashfunc(int x)
{
    int val = x%10;
    if(hashtable[val]==false)
    {
        hashtable[val] = true;
        return 0;
    }
    else
        return 1;
}

int main()
{
    int t;
    cin>>t;

    int res[t];
    for(int i=0;i<t;i++)
    {
        int n;
        cin>>n;

        int collisions = 0,x;
        memset(hashtable,false,sizeof(hashtable));
        while(n--)
        {
            cin>>x;
            collisions += hashfunc(x);
        }
        res[i] = collisions;
    }

    for(int i=0;i<t;i++)
        cout<<res[i]<<endl;

    return 0;
}
