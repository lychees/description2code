#include<bits/stdc++.h>
using namespace std;
int mini[15];
bool vis[11];
int main() {
	int t,i,j;
	scanf("%d",&t);
	while(t--) {
		for(i=0;i<=9;i++)
			vis[i]=false,scanf("%d",&mini[i]);
		int rem=10,ind;
		while(rem--) {
			int val=10003;
			for(i=0;i<=9;i++) {
				if(!vis[i]&&mini[i]<val) {
					ind=i;
					val=mini[i];
				}
			}
			for(i=0;i<=9;i++)
				mini[(i+ind)%10]=min(mini[(i+ind)%10],mini[i]+mini[ind]);
			vis[ind]=true;
		}
		int n;
		scanf("%d",&n);
		string str;
		cin>>str;
		long long ans=0;
		for(i=0;i<str.size();i++)
			ans+=mini[str[i]-48];
		printf("%lld\n",ans);
	}
	return 0;
}
