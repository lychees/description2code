#include<bits/stdc++.h>
using namespace std;
struct burger{
int index;
int time1;
int time2;
int time3;
};
bool comp1(struct burger c,struct burger d)
{
    return (c.time3<d.time3);
}
int main()
{
    int i;
    int n;
    struct burger bug1[100000];
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
    cin>>bug1[i].time1>>bug1[i].time2;
    }
    for(i=1;i<=n;i++)
    {
        bug1[i].index=i;
    }
    for(i=1;i<=n;i++)
     {
       bug1[i].time3=(bug1[i].time1+bug1[i].time2);
     }
     sort(bug1,bug1+(n+1),comp1);
       for(i=1;i<=n;i++)
     {
         cout<<bug1[i].index<<" ";
     }
      return 0;
}
