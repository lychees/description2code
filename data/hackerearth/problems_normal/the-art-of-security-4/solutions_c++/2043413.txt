#include<bits/stdc++.h>
using namespace std;
int a[10],n,l,r,ans=0;
void solve(int x)
{
	if(x==n)
	{
		ans++;
		return;
	}
	if(l+a[x]<=r)
	{
		l+=a[x];
		solve(x+1);
		l-=a[x];
	}
	r+=a[x];
	solve(x+1);
	r-=a[x];
}

int main()
{
	int sum=0;
	cin >> n;
	for(int i=0;i<n;i++){cin >> a[i];sum+=a[i];}
	sort(a,a+n);
	while(true)
	{
		l=0,r=0;
		solve(0);
		if(!next_permutation(a,a+n))break;
	}
	cout << ans << " " << sum << endl;
	if(sum>=ans)cout << "Got no way out!\n";
	else cout << "We will win!\n";
	return 0;
}
