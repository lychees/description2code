#include <iostream>
#include<stdio.h>
#include<algorithm>
#include<string>
using namespace std;
string s;
int c[3];
int main()
{
    int t,i;
    long long int ans;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	ans=0;
    	c[0]=c[1]=c[2]=-1;
    	
    	for(i=0;i<s.length();i++)
    	{
    		c[s[i]-'x']=i;
    		if(s[i]=='x')
    		{
    			ans+=(i-min(c[1],c[2]));
    		}
    		else if(s[i]=='y')
    		{
    			ans+=(i-min(c[0],c[2]));
    		}
    		else
    		{
    			ans+=(i-min(c[0],c[1]));
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
