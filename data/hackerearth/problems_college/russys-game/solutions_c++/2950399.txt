#include<bits/stdc++.h>
using namespace std;
main()
{
	long long n,m,i,j,k,d;int flag,t;
	scanf("%lld",&n);
	while(n--)
	{
		flag=0;t=0;
		scanf("%lld",&m);
		long long a[m],s[m];
		for(i=0;i<m;i++)
		scanf("%lld",&a[i]);
		i=0;k=0;
		for(j=a[i];i<m;j++)
		{
			if(j==0)
			break;
			for(d=0;d<k;d++)
			{
				if(j==s[d])
				{
				t=1;
				break;}
			}
			if(t==1)
			break;
			if(j==-1)
			{
				printf("Win\n");
				flag=1;
				break;
			}
			i=j;
			s[k]=j;
			k++;
			j=a[i]-1;
		}
		if(flag==0)
		printf("Lose\n");
	}
}
