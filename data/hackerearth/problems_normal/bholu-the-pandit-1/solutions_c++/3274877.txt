#include <iostream>
using namespace std;

int gcd(int p,int q)
{int t;
	while(q>0)
	{   
		p=p%q;
		t=p;
		p=q;
		q=t;
	}
	return p;
}
int main()
{
	int t;
	cin>>t;
	long long a,b,c;
	while(t--)
	{
		cin>>a>>b>>c;
		long long l=gcd(gcd(a,b),c);
		a/=l;
		b/=l;
		c/=l;
		unsigned long long ans=(((a*b)%1000000007)*c)%1000000007;
		cout<<l<<" "<<ans<<endl;
	}
	return(0);
}
