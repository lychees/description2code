#include <iostream>
#include<string>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t-->0){
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++){
    		string s ;
    		cin>>s;
    		a[i]=s[0];
    	}
    	int flag=0;
    	 for(int i=0;i<n&&flag==0;i++){
            for(int j=i+1;j<n;j++){
                if(a[i]==a[j]){
                    flag++;
                    cout<<"NO"<<endl;
                    break;
                }
            }
        }
        if(flag==0){
            cout<<"YES"<<endl;
        }
        
    }
    return 0;
}
