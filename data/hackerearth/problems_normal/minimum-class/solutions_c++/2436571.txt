{
    long long int a[1000002],i,j;

    for(i=3;i<1000001;i++)

        a[i]=a[i-1]+(2*(i-1));
    }
    long long int n,t;
    cin>>t;
    while(t--)
    {
        cin>>n;
        if(n==1)
            cout<<"1"<<endl;
        else
        {i=1;
        while(1)
        {
           if(n>=a[i]&&n<a[i+1])
            break;

        }
        if(n<a[i]+i)
            cout<<(2*i)-1<<endl;

            cout<<(2*i)<<endl;
        }
    }

}
