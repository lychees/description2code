#include <iostream>
#include <cmath>
#define MAX 100005
typedef long long int ll;
int a[MAX];
int count[MAX];
int in[MAX];
using namespace std;

int main() {
	// your code goes here
int i,j,z,n,q,x;

for(i=4;i<=MAX;i=i+2)
a[i]=2;

for(i=3;i<=sqrt(MAX);i=i+2)
{

while(a[i]>0)
i=i+2;

j=i*(i-2);
z=i<<1;

while((j=j+z)<=MAX)
{
if(a[j]==0)
a[j]=i;
}

}


cin>>n;

for(i=0;i<n;i++)
cin>>in[i];

for(i=0;i<n;i++)
{
	while(a[in[i]]!=0)
	{
		count[a[in[i]]]++;
		in[i]=in[i]/a[in[i]];
	}
	count[in[i]]++;
}

int mini=1e8;
cin>>q;
while(q--)
{
	
    int mini=1e8;
	cin>>x;
	if(a[x]>0)
	{
	while(a[x]>0)
	{
		int k=a[x];
		int z=0;
		while(x%k==0)
		{
			z++;
			x/=k;
		}
		mini=min(mini,count[k]/z);
	}
	
	if(x!=1)
	mini=min(mini,count[x]);
	
	
	}
	else
	mini=min(mini,count[x]);
	
	
	cout<<mini<<endl;
}

	return 0;
}
