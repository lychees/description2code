#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,i,j=0,k=0,flag=0;
    string s;
    cin>>t;
    
    
    if(t>=1&&t<=10)
        {
        while(t--)
        { flag=0;
        j=0;k=0;
            cin>>s;
            if(s.length()<=100&&s.length()>=1)
            {for(i=0;i<s.length();i++)
            {
                if(s[i]>='A'&&s[i]<='Z')
                  {
                     j++;
                     flag=1;
                  }

                if(s[i]>='a'&&s[i]<='z')
                    {k++;
                    flag=1;
                    }
            }
            if(j==0&&k==0)
                cout<<"Invalid Input"<<endl;
            else
                cout<<min(j,k)<<endl;}
            else
            cout<<"Invalid Input"<<endl;



        }
    }
    else
    cout<<"Invalid Test";
}
