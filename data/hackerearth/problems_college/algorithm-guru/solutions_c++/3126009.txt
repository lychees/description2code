#include <iostream>
#include <climits>
using namespace std;

int func(int **dp, int *a, int idx, int x)
{
	if(x < 0)
		return INT_MIN;
	if(x == 0)
		return 0;
	if(idx < 0)
		return INT_MIN;
	if(dp[idx][x] != INT_MIN)
		return dp[idx][x];
	if(x%a[idx] == 0)
	{
		dp[idx][x] = x;
		return x;
	}
	
	int p1 = 0;
	if(a[idx] <= x)
		p1 = a[idx] + func(dp,a,idx,x-a[idx]);
	int p2 = func(dp,a,idx-1,x);
	dp[idx][x] = p1>p2?p1:p2;
	//cout << idx << " " << x << " " << dp[idx][x] << endl;
	return dp[idx][x];
	
}

int main()
{
	int tc;
	cin >> tc;
	while(tc-- > 0)
	{
		int n,x;
		cin >> n >> x;
		int a[n];
		for(int i=0;i<n;i++)
			cin >> a[i];
		
		//int dp[n][x+1];
		int **dp;
		dp = new int *[n];
		for(int i=0;i<n;i++)
			dp[i] = new int[x+1];
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<=x;j++)
				dp[i][j] = INT_MIN;
		}
		int ans = func(dp,a,n-1,x);
		cout << ans << endl;
		
	}
    return 0;
}
