#include <bits/stdc++.h>

using namespace std;

int BIT[100001+7], a[100001+7], n,t;
void update(int x, int val)
{
      for(; x <= n; x += x&-x)
        BIT[x] += val;
}
int query(int x)
{
     int sum = 0;
     for(; x > 0; x -= x&-x)
        sum += BIT[x];
     return sum;
}

int main()
{
     scanf("%d", &n);
     int i;
     for(i = 1; i <= n; i++)
     {
           scanf("%d", &a[i]);
           update(i, a[i]);
     }
     int q;
     cin >> q;
     while(q--)
     {
     	int l,r;
     	cin >> l >> r;
     	cout << query(r) - query(l-1) << endl;
     }
     
     return 0;
}
