for i in range(int(raw_input())):
	line = raw_input()
	if len(line)==1:
		print(line)
		continue
	l = len(line)-1
	a=int(line[l])
	b=int(line[l-2])
	op = line[l-1]
	if op=='+':
			res=(a+b)
	elif op=='-':
			res=(a-b)
	elif op=='*':
			res=a*b
	for i in range(l-3,0,-2):
		b=int(line[i-1])
		op=line[i]
		if op=='+':
			res+=b
		elif op=='-':
			res -=b
		else:
			res*=b
	print(res)
