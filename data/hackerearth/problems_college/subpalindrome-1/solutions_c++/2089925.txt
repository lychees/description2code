#include <bits/stdc++.h>
using namespace std;

int N;
string S;
vector<vector<long long> > best;
 
long long solve(int i, int j) {
    long long &res = best[i][j];
    if (res >= 0) return res;
    res = 0;
    if (i == j) return res = 1;
    if (i+1 == j) return res = 2;
    res = solve(i,j-1) + solve(i+1,j);
    if (S[i] != S[j-1]) res -= solve(i+1,j-1);
    return res;
}
 
int main() {
    int tt; cin >> tt;
    while (tt--) {
        cin >> S;
        best.clear();
        N = S.size();
        best.resize(N + 1, vector<long long>(N+1,-1));
        cout << solve(0, N) - 1 << endl;
    }
}
