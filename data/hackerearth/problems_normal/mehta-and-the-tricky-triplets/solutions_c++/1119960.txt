#include<bits/stdc++.h>
#define si(n)scanf("%d",&n)
#define pi(n)printf("%d\n",n)
#define sll(n)scanf("%lld",&n)
#define pll(n)printf("%lld\n",n)
using namespace std;using ll=long long;
const int N=1e5+50;
string s[N];
int main(){
    ll n,ans=0;sll(n);
    for(int i=0;i<n;i++){
            cin>>s[i];
    }
    const char p[]={'2','3','5','7'};
    for(int i=1;i<(1<<4);i++){
            int x=i,ind=0,sb=0;vector<char>vc;
            while(x){
                int r=x%2;
                if(r){
                    vc.push_back(p[ind]);sb++;
                }
                ind++;
                x>>=1;
            }
            ll cx=0;
	        for(int j=0;j<n;j++){
	            bool ok=true;
	            for(char c:vc){
	                if(s[j].find(c)==string::npos){
	                    ok=false;break;
	                }
	            }
	            if(ok)cx++;
	        }
	        if(sb&1)ans+=(cx*(cx-1)*(cx-2))/6;
	        else ans-=(cx*(cx-1)*(cx-2))/6;
    }
    pll(ans);
    return 0;
}
