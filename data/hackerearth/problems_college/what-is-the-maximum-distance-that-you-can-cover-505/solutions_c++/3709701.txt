#include <iostream>
#include<iomanip>
using namespace std;
int main()
{
    double c[100],s[100],d[100],k,res,fres,m;
    int N,i,j;
    long int mres;
    cin>>N;
    cout<<endl;
    for(i=1;i<=N;i++)
    {
    	cin>>s[i];
    }
    cout<<endl;
    for(i=1;i<=N;i++)
    {
    	cin>>c[i];
    }
    cout<<endl;
    cin>>m;
    for(i=1;i<=N;i++)
    {
    	d[i]=c[i]/s[i];
    }
    for(i=1;i<=(N-1);i++)
    {
    	for(j=i+1;j<=N;j++)
    	{
    		if(d[i]>d[j])
    		{
    			k=d[i];
    			d[i]=d[j];
    			d[j]=k;
    		
    		}
    	}
    }
    res=m/d[1];
    mres=res*1000;
    fres=mres*0.001;
    cout<<fixed<<setprecision(3)<<fres;
    return 0;
}
