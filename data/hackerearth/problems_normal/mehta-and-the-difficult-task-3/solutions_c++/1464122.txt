vector<vector<long long> > one(long sz)
{
 vector<vector<long long> > res;
 res.resize(sz);
 for (int i=0;i<res.size();i++)
  res[i].resize(sz);
 for (int i=0;i<res.size();i++)
  for (int j=0;j<res[i].size();j++)
   res[i][j]=(i==j);
 return res;
}

vector<vector<long long> > mult(vector<vector<long long> > a,
vector<vector<long long> > b)
{
 vector<vector<long long> > res;
 res=one(a.size());
 for (int i=0;i<res.size();i++)
  for (int j=0;j<res.size();j++)
   res[i][j]=0;
 for (int i=0;i<a.size();i++)
  for (int j=0;j<a.size();j++)
   for (int k=0;k<a.size();k++)
    res[i][j]=(res[i][j]+a[i][k]*b[k][j])%bs;
  return res;
}

vector<vector<long long> > pw(vector<vector<long long> > a,
long long b)
{
 if (b==0)
  return one(a.size());
 if (b%2) return mult(a,pw(a,b-1));
 return pw(mult(a,a),b/2);
}
