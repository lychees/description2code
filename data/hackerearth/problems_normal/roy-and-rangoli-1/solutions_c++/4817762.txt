#include <iostream>
using namespace std;
long long int mark[2000005]={0};
long long int dp[2000005]={0};
void sieve(long long int n)
{
	mark[0]=1;
	mark[1]=1;
	for(int i=2;i*i<=n;i++)
	{
		if(mark[i]==0)
		for(int j=2*i;j<=n;j+=i)
		{
			mark[j]=1;
		}
	}
}

int main()
{
	long long int n,sum=0;
	dp[0]=0;
	dp[1]=0;
	cin>>n;
	sieve(2*n);
    for(int k=2;k<=(2*n);k++)
    {
    	if(mark[k]==0)
    	dp[k]=dp[k-1]+1;
    	else
    	dp[k]=dp[k-1];
    }
    sum+=(dp[n-1]);
    sum=sum%1000000007;
    for(int l=1;l<n;l++)
    {
    	sum+=(dp[l+n-1]-dp[l-1]);
    	sum=sum%1000000007;
    }
    cout<<sum<<endl;
}
