#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define gc getchar_unlocked

void sc(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
int main()
{
    int n;
    sc(n);
    int arr[n];
    for(int i = 0;i<n;i++){
    	sc(arr[i]);
    }
    sort(arr,arr+n);
    ll p = 1;
    for(int i = 1;i<n;i++){
    	if(arr[i]==arr[i-1]){
    		continue;
    	}
    	else{
    		p++;
    	}
    }
    p = (p*(p-1))/2;
    printf("%lld\n",p);
    return 0;
}
