#include <stdio.h>
#include <string.h>
#include<vector>
#include<iostream>
using namespace std;
vector<int> prime;
void markMultiples(bool arr[], int a, int n)
{
    int i = 2, num;
    while ( (num = i*a) <= n )
    {
        arr[ num-1 ] = 1;
        ++i;
    }
}
 
void SieveOfEratosthenes(int n)
{
    if (n >= 2)
    {
        bool arr[n];
        memset(arr, 0, sizeof(arr));
        for (int i=1; i<n; ++i)
        {
            if ( arr[i] == 0 )
            {
                prime.push_back(i+1);
                markMultiples(arr, i+1, n);
            }
        }
    }
}
 
int main()
{
	
 	int x;
    SieveOfEratosthenes(8000);
    cin>>x;
    cout<<prime[x-1];
    return 0;
}
