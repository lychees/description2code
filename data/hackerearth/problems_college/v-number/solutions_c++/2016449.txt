#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	int n,i,s1=0,s2=0;
    	string s;
    	cin>>s;
    	n=s.length();
    	for(i=0;i<=(n/2)-1;i++)
    	{
    		s1=s1+s[i];
    	}
    	for(i=n/2;i<n;i++)
    	{
    		s2=s2+s[i];
    	}
    	if(s1==s2)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
   
}
