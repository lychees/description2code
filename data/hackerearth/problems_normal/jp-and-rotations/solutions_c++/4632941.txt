#include <iostream>
using namespace std;

const int maxn = 1e5 + 5;
int st[maxn];
int en[maxn];

int main()
{
	int n, m;
	cin >> n >> m;
	int nn = n;
	while (nn <= 1000) nn += n;
	for (int i = 0; i < n; ++i) cin >> st[i];
	for (int i = 0; i < n; ++i) cin >> en[i];
	
	int sti = 0;
	for (int i = 1; i <= m; ++i) {
		char c; cin >> c;
		int x; cin >> x;
		if (c == 'L') {
			sti = (sti + x) % n;
		}
		else {
			sti = (sti - x + nn) % n;
		}
		if (st[sti] == en[0]) {
			cout << i << endl;
			return 0;	
		}
	}
	cout << -1 << endl;

    return 0;
}
