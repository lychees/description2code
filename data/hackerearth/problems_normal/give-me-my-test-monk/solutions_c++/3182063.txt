#include<bits/stdc++.h>

using namespace std;
#define ll long long int
#define mq 1000000007
#define pb push_back
#define mk make_pair
#define fi first
#define se second
#define pii pair<ll, ll>

ll fact[105];
ll dp[105];			// for counting derangements...

ll invmod(ll a) {

	ll x=1, y=a, b=mq-2;
	while (b>0) {
		if(b%2==1) {
			x=(x*a)%mq;
		}
		a=(a*a)%mq;
		b=b/2;
	}
	
	return x;
}

ll comb(int n, int k) {

	if(k==0 || n-k == 0){
		return 1;
	}

	ll ans=fact[n];
	ans = (ans*invmod(fact[n-k]))%mq;
	ans = (ans*invmod(fact[k]))%mq;
	return ans;

}

int main() {

	fact[0]=1;
	fact[1]=1;
	dp[0]=1;
	dp[1]=0;

	for(int i=2; i<105; i++) {
		fact[i]=(i*fact[i-1])%mq;
		dp[i]=((i-1)*(dp[i-1]+dp[i-2]))%mq;
	}

	int n, l , r;
	cin>>n>>l>>r;

	ll ways=0;
	for (int i=l; i<=r ; i++) {
		ways = (ways + (comb(n,i)*dp[i])%mq)%mq;
	}
	cout<<(ways*invmod(fact[n]))%mq<<endl;
	return 0;
}
