#include<iostream>
using namespace std;
int main()
{
	long long int n,t;
	cin>>t;
	for(int i=0; i<t; i++)
	{
		cin>>n;
		long long int sum=0;
		long long int s=0;
		for(long long int j=1; s<=n; j++)
		{
			
			s=1;
			s=s<<j;
			s+=1;
			long long int c=s;
			while(c<=n)
			{
				sum+=c%1000000007;
				c=c<<1;
			}
		}
		cout<<sum%1000000007<<endl;
	}
	return 0;
}
