#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long int n,i,max=0,max2=0;
    cin>>n;
    long int a[n];
    cin>>a[0];
    for(i=1;i<n;i++){
    	cin>>a[i];
    	if(fabs(a[i]-a[i-1])==1){
    		max2++;
    	//	cout<<max2;
    		if(max<max2){
    			max=max2;
    		}
    	}else{
    		if(max<max2){
    			max=max2;
    		}
    		max2=0;
    	}
    }
    cout<<max+1;
    return 0;
}
