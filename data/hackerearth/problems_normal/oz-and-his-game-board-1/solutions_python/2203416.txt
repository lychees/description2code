tst = int(raw_input())
while tst > 0:
	temp = raw_input()
#	print temp
	t = temp.split(' ')
	N = int(t[0])
	M = int(t[1])
	A = int(t[2])
	B = int(t[3])
	C = int(t[4])
	
	ans = N*A + M*B
	
	other = 0
	if N > M:
		other = C*(M+M) + (N-M)*A
	else:
		other = C*(N+N) + (M-N)*B
		
	if other > ans:
		ans = other
	print ans
	
	tst-=1
