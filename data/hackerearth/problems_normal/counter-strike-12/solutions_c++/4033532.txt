#include<bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int test;
    cin>>test;
    while(test--)
         {
         
         	int n,p,dis;
         	cin>>n>>p>>dis;
         	//cout<<"dis   "<<n<<endl; 
         	//cout<<"n   "<<p<<endl; 
         //	cout<<"p   "<<dis<<endl; 
         	
         	int a[n][2],b[p][2];
         	for(int i=0;i<n;i++)
         	    cin>>a[i][0]>>a[i][1];
         	for(int i=0;i<p;i++)
         	    cin>>b[i][0]>>b[i][1];
         	  // cout<<"dis   "<<endl; 
         	 int count=0;
         	 //cout<<"aa"<<endl;
         	 for(int i=0;i<n;i++)
         	    {
         	    //	cout<<"aa"<<endl;
         	    
         	    	for(int j=0;j<p;j++)
         	    	   {
         	    	   	int d,x,y;
         	    	   	x=abs(a[i][0]-b[j][0]);
         	    	   	y=abs(a[i][1]-b[j][1]);
         	    	   	d=x+y;
         	    	  	//cout<<d<<" "<<"aa"<<endl;
         	    	   	 if(d<=dis)
         	    	   	   {
         	    	   	   	//cout<<"aaaa"<<endl;
         	    	   	   	count++;
         	    	   	   	break;
         	    	   	   }
         	    	   	
         	    	   }
         	    }
         	   // cout<<"count   "<<count<<endl;
         	if(count>=(n/2))  
         	cout<<"YES"<<endl;
         	else
         	cout<<"NO"<<endl;
         }
    return 0;
}
