#include <bits/stdc++.h>
using namespace std;

bool primes[10000001];
long long ans[1000001];

int main()
{
    primes[1]=1;
    for(int i=2; i*i<=1000000; ++i)
     if(primes[i]==0)
     for(int j=i*i; j<=1000000; j+=i)
      primes[j]=1;
      
      
     for(int i=1; i<=1000000; ++i)
     {
      if(i==4)
       ans[i]=(ans[i-1]);
      else
      ans[i]=ans[i-1]+(primes[i]==1);
     }
      
     int t;
     cin>>t;
     
     while(t--)
     {
     	int l,r;
     	scanf("%d%d",&l,&r);
     	printf("%lld\n",ans[r]-ans[l-1]);
     }
    return 0;
}
