#include <bits/stdc++.h>
using namespace std;

int row[]={0,0,+1,-1};
int col[]={+1,-1,0,0};

int vis[25][25];

bool isvalid(int a[25][25], int n, int x, int y)
{
	if(x>=0 && x<n && y>=0 && y<n && a[x][y]==0)
		return true;
	else
		return false;
}

void dfs(int a[25][25], int n, int x, int y, int &ans)
{
	if(vis[x][y])
		return;
	if(x==n-1 && y==n-1)
	{
		ans+=1;
		return;
	}
	
	vis[x][y]=1;
	
	for(int i=0;i<4;i++)
	{
		int xn=x+row[i];
		int yn=y+col[i];
		if(isvalid(a,n,xn,yn)==true)
			dfs(a,n,xn,yn,ans);
	}
	
	vis[x][y]=0;
}

int main() 
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a[25][25], i, j;
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				cin>>a[i][j];
				vis[i][j]=0;
			}
		}
		int ans=0;
		dfs(a, n, 0, 0, ans);
		cout<<ans<<endl;
	}
	return 0;
}
