#include<iostream>
using namespace std;
long long int mod=1000000007,mod1=10000,fib[15005]={0},prefib[15005]={0},mod2=15000;
int main()
{
	fib[1]=0;
	fib[2]=1;
	prefib[1]=0;
	prefib[0]=0;
	for(int i=3;i<=15000;i++)
		fib[i]=(fib[i-1]+fib[i-2])%mod1;
	for(int i=2;i<=15000;i++)
		prefib[i]=(prefib[i-1]+fib[i]);
	int t;
	cin >> t;
	while(t--)
	{
		long long int l,r,a,b,ans=0,temp=0,temp1=0;
		cin >> l >> r;
		if(l%mod2==0)
		a=l/mod2;	
		else
		a=(l/mod2)+1;
		b=r/mod2;
		if(b<a)
		{
			cout << (prefib[r%mod2]-prefib[(l%mod2)-1])%mod << '\n';
		}
		else
		{
			temp=((b-a))%mod;
			temp1=(temp*(prefib[mod2]))%mod;
			temp=0;
			if(l%mod2==0)
			{
			l=mod2;
			temp=(temp+prefib[mod2]-prefib[l-1])%mod;
			}
			else if(l%mod2>0)
			temp=(temp+prefib[mod2]-prefib[(l%mod2)-1])%mod;
			if(r%mod2>0)
			temp=(temp+prefib[r%mod2])%mod;
			ans=(ans+temp)%mod;
			ans=(ans+temp1)%mod;
			cout << ans << '\n';
		}		
	}
	return 0;
}
