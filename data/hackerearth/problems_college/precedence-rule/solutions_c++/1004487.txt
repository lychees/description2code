#include<bits/stdc++.h>
int main() {
	char a[50];
	int p,l;
	long long ans;
	int t,i;
	scanf("%d",&t);
	while(t--) {
	    long long sum=0;
		scanf("%s",a);
		l=strlen(a);
		if(l==1) sum=a[0]-48;
		for(i=l-1;i>=0;i--) {
			if(a[i]>='0'&&a[i]<='9') sum=p=a[i]-48;
			else {
				if(a[i]=='*') {
					sum=(sum*(a[i-1]-48));
					i--;
				}
				else if(a[i]=='+') {
					sum=sum+(a[i-1]-48);
					i--;
				}
				else if(a[i]=='-') {
					sum=sum-(a[i-1]-48);
					i--;
				}
				//printf("%lld\n",sum);
			}
		}
		printf("%lld\n",sum);
	}
	return 0;
}
