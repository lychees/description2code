#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int X[n], Y[n], R[n];
    for(int i=0; i<n; i++){
    	cin>>X[i]>>Y[i]>>R[i];
    }
    int c=0;
    for(int i=1; i<=1000; i++){
    	for(int j=1; j<=1000; j++){
    	int check=0;
    		for(int k=0; k<n; k++){
	    		if(((X[k]-i)*(X[k]-i) + (Y[k]-j)*(Y[k]-j)) <= R[k]*R[k]){
	    			check++;
	    			if(check==2){
	    				break;
	    			}
	    		}
    		}
    		if(check==2){
    			c++;
    		}
    	}
    }
    
    cout<<c<<endl;
    return 0;
}
