#include <iostream>
#include <cstring>
using namespace std;
inline void sort(char *a,int n){
	int k,j;
	for(k=1;k<=n-1;k++){
		for(j=0;j<=n-k-1;j++){
			if(*(a+j)>*(a+j+1)){
				*(a+j)=*(a+j)+*(a+j+1);
				*(a+j+1)=*(a+j)-*(a+j+1);
				*(a+j)=*(a+j)-*(a+j+1);
			}
		}
	}
}
int main(){
    char str[1000];
	cin>>str;
	int len=strlen(str);
	sort(str,len);
	cout<<str<<endl;
    return 0;
}
