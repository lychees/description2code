#include<bits/stdc++.h>
using namespace std;
long long int count=0;
int nFactors (unsigned long long int n)
{
	int i;
	int pro = 1;
	int c=0;
	for(i=2;i<=sqrt(n);i++)
	{
		while(n%i == 0)
		{
			n = n/i;
			c++;
		}
		pro = pro*(c+1);
		c=0;
	}
			if(n>1)
		{
			pro = pro*2;
		}
	return pro;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		unsigned long long int n;
		cin>>n;
		long long int pro = nFactors (n);
		cout<<pro<<"\n";
	}
}
