#include <bits/stdc++.h>
using namespace std;
//palindromic index
int main() {
	int t;
	cin>>t;
	while(t--){
		char ch[1000000],c[1000000];
		int i,j,temp=0,flag=0,k=0,indexi=0,indexj=0;
		cin>>ch;
		int len=strlen(ch);
		for(i=0,j=len-1;i<=len/2;i++,j--){
			if(ch[i]!=ch[j])
			{
				indexi=i;
				indexj=j;
				temp=1;
				break;
			}
		}
		if(temp==0)
		cout<<-1<<endl;
		else{
			//cout<<indexi<<" "<<indexj<<endl;
			for(i=0;i<indexi;i++)
			c[k++]=ch[i];
			for(i=indexi+1;i<len;i++)
			c[k++]=ch[i];
			for(i=0,j=k-1;i<=k/2;i++,j--){
			if(c[i]!=c[j])
			{
				flag=1;
				break;
			}
		}
		if(flag==0)
		cout<<indexi<<endl;
		else
		cout<<indexj<<endl;
		}
	}
	return 0;
}
