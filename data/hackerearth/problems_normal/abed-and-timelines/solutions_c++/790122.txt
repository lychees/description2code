#include <iostream>
using namespace std;

int main()
{
	int t,p;
	long n,m,x1,x2,y1,y2,min,max;
	long long ans,ans1,ans2;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%ld%ld",&n,&m);
		scanf("%ld%ld%ld%ld",&x1,&y1,&x2,&y2);
		scanf("%d",&p);
		if(x1<x2)
		{min=x1;
		max=x2;
		}
		else
		{min=x2;max=x1;}
		ans=max-min;
		ans1=n-max+min;
		if(ans>ans1)
		ans=ans1;
		
		if(y1<y2)
		{min=y1;
		max=y2;
		}
		else
		{min=y2;max=y1;}
		ans1=max-min;
		ans2=m-max+min;
		if(ans1>ans2)
		ans1=ans2;
		ans+=ans1;
		ans=ans*p;
		printf("%lld\n",ans);
		if(ans<=1000)
		{	printf("You saved the group.\n");
		}
		else
		printf("Let go of the group.\n");
	}
	return 0;
}
