VI primes;
int marker[10001] = {0};
void sieve(int limit = 10000)
{
	FOR(i,2,limit+1) marker[i] = 1;
	FOR(i,2,limit+1)
	{
		if(!marker[i]) continue;
		primes.pb(i);
		int j = 2*i;
		while(j<=limit)
		{
			marker[j] = 0;
			j += i;
		}
	}
}

int ncr[5010][5010];

void pre()
{
	ncr[0][0] = ncr[1][1] = ncr[1][0] = 1;
	for(int i = 2; i <= 5000; i++)
	{
		ncr[i][i] = ncr[i][0] = 1;
		for(int j = 1; j < i; j++)
			ncr[i][j] = (ncr[i-1][j] + ncr[i-1][j-1]) % MOD;
	}
}

int pr(int a)
{
	if(a < 2) return 1;
	for(int i = 2;i*i <= a; i++)
	{
		if(a %i == 0) return 0;
	}
	return 1;
}

int main()
{
	sieve();
	pre();

	int a , b;
	cin >> a >> b;
	if(a == 0 and b == 0) cout << 0 << endl;
	else if(a == 0 && !pr(b)) cout << 1 << endl;
	else if(b == 0 && !pr(a)) cout << 1 << endl;
	else
	{
		int ans = 0;
		for(int i = 0; i < sz(primes); i++)
		{
			if(a % primes[i] == 0 && b % primes[i] == 0)
			{
				ans = (ans + ncr[a/primes[i] + b/primes[i]][a/primes[i]]) % MOD;
			}
		}
		cout << ans << endl;
	}
	return 0;
}
