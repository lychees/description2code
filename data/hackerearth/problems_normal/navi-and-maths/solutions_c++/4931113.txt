#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;

ll fermat(ll base,ll exponent)
{
    ll result = 1;
    while (exponent>0){
        if (exponent & 1)
            result=(result*base)%mod;
        exponent = exponent>>1;
        base = (base * base)%mod;
    }
    return result;
}



int main(){
	ll mul,val,m,count,t,n,sum;
	scanf("%lld",&t);
	
	for(ll k=1;k<=t;k++){
		
		m=(-1)*mod;
		scanf("%lld",&n);
		ll a[n+1];
		
		for(ll i=0;i<n;i++) scanf("%lld",&a[i]);
		ll lim=1<<n;
		
		for(ll i=0;i<lim;i++){
			sum=0;  val=1; count=0; 
			for(ll j=0;j<n;j++){
				if(i & (1<<j)){
					count++;
					val=(val*a[j])%mod;
					sum=(sum+a[j])%mod;
				}
			}
			if(count>=2){
			sum=fermat(sum,mod-2);
			val=val*sum;
			val%=mod;
			m=max(m,val);
			}
			//cout<<"val="<<val<<" sum="<<sum<<endl;
		}
		printf("Case #%lld: %lld\n",k,m);
	}
	return 0;
}
