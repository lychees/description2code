L = int(raw_input())
N = int(raw_input())

for i in range(0,N):
	dim = raw_input().split()
	W,H = int(dim[0]), int(dim[1])
	if W == H and W >= L and H >= L:
		print "ACCEPTED"
	elif W > L and H > L:
			print "CROP IT"
	elif W == L and H == L:
		print "ACCEPTED"
	elif  W < L and H < L:
		print "UPLOAD ANOTHER"
	elif W == L and H > L:
		print "CROP IT"
	elif H == L and W > L:
		print "CROP IT"
	else:
		print "UPLOAD ANOTHER"
