#include<bits/stdc++.h>
//zone
using namespace std;
struct ar{
    int x;
    int y;
};
int kuroko(ar a,ar b)
{
    return a.x < b.x  ;
}
int main()
{
    int t;
    cin >> t;
    while(t--){
            int n;
    cin >> n;
    int i,j;
    ar a[n + 1];
    long int dp[n + 1];

    int u,v;
    for(i = 0 ; i < n ; i++)
    {
        cin >> u >> v;
        a[i].x = u;
        a[i].y = v;
    }

    sort(a,a+n,kuroko);
    for(i = 0 ; i < n ; i++)
    {
        dp[i] = a[i].y;
    }
    /*for(i = 0 ; i < n ; i++)
    {
        cout << a[i].x << " " << a[i].y<<endl;
    }*/
    long int maxi = a[0].y;
    for(i = 1 ; i < n ; i++)
    {
        for(j = 0 ; j < i ; j++)
        {
            if(a[i].x > a[j].x && a[i].y > a[j].y )
            {
                dp[i] = max(dp[i], dp[j] + a[i].y);
            }
        }
        maxi = max(dp[i],maxi);
    }
    cout<<maxi<<endl;
    }
    return 0;
}
