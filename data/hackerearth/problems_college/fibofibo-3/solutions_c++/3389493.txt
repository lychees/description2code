#include<iostream>
using namespace std;
int count=0;
int count2=0;
int f(int n)
{
    if(n==1)
    {
        count2++;
        return 1;
    }
    else if(n==0)
    {
        count++;
        return 0;
    }
    else
        return f(n-1)+f(n-2);

}
int main()
{
    int t;
    cin>>t;
    while(t-->0)
    {
        count=0;
        count2=0;
        int n;
        cin>>n;
        f(n);
        cout<<count<<" "<<count2<<endl;
    }
}
