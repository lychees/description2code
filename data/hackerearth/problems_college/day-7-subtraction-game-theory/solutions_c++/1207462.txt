#include <iostream>
#define MOD 1000000007
using namespace std;

int main()
{
	int x=0;
	int y=0;
	int n=0;
    cin>>x>>y;
    long nQuery=0;
    cin>>nQuery;
    long long res[1000002];
    
    int i=1;
    res[0] = 1;
    for(i=1;i<=1000000;i++)
    res[i]=(res[i-1]*x-y+MOD)%MOD;
    
   for(;nQuery>0;nQuery--)
	{
    	scanf("%d",&n);
    	cout<<res[n]<<"\n";
    }
    return 0;
}
