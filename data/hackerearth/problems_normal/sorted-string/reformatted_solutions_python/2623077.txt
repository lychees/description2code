t = input()
for i in range(t):
    d = dict()
    st = raw_input()
    for j in range(1, 27):
        c = chr(96+j)
        co = st.count(c)
        if co > 0:
            if co in d:
                d[co].append(c)
            else:
                d[co] = [c]
    k = sorted(d.keys())
    st = ""
    for key in k:
        val = sorted(d[key])
        for v in val:
            st = st+(v*key)
    print st
