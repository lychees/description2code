min_len = int(raw_input())
test = int(raw_input())
hei = []
wid = []
for i in range(test):
    h, w = map(int, raw_input().split())
    hei.append(h)
    wid.append(w)
for i in range(test):
    if (hei[i] < min_len or wid[i] < min_len):
        print "UPLOAD ANOTHER"
    elif (hei[i] >= min_len and wid[i] >= min_len and hei[i] == wid[i]):
        print "ACCEPTED"
    else:
        print "CROP IT"
