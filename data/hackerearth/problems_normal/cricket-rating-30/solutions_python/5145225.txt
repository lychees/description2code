def max_subarray(A):
    max_ending_here = max_so_far = 0
    for x in A:
        max_ending_here = max(0, max_ending_here + x)
        max_so_far = max(max_so_far, max_ending_here)
    return max_so_far
   
n = int(raw_input())
score = []
if(n!=0):
	score = [int(x) for x in raw_input().split()]
	ans = max_subarray(score)
else:
	ans = 0
