#include <iostream>
using namespace std;
#include <cstring>
int main()
{
	int t,count,i,j;
	char p[100000],q[100000];
   cin>>t;
   while(t--)
   {
   	count=0;
   	cin>>p>>q;
   	for(i=0;i<strlen(p)-1;i++)
   	{
   		for(j=0;j<strlen(q)-1;j++)
   		{
   			if(p[i]==q[j])
   			count++;
   		}
   	}
   	if(count>0)
   	cout<<"YES"<<endl;
   	else
   	cout<<"NO"<<endl;
   }
    return 0;
}
