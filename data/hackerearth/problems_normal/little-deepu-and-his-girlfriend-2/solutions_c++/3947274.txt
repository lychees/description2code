#include <cstdio>
#include <algorithm>
#include <cstring>
#include <cassert>
using namespace std;
#define MAX 110
#define SIZE 10010
int ar[MAX];
bool win[SIZE];
int main(){
	int T, N, M;
	scanf("%d", &T);
	assert(T>0 and T<=1000);
	while(T--){
		scanf("%d %d", &M, &N);
		assert(M>0 and M<=10000);
		assert(N>0 and N<=100);
		for(int i=0;i<N;i++){
			scanf("%d", &ar[i]);
			assert(ar[i]>0 and ar[i]<=M);
		}
		sort(ar, ar+N);
		
		memset(win, false, sizeof(win));
		for(int i=0;i<N;i++)
			win[ar[i]] = true;
	
		for(int i=1;i<=M;i++){
			if(not win[i]){
				for(int j=0;j<N;j++){
					if(i-ar[j]>=0 and not win[i-ar[j]]){
						win[i] = true;
						break;
					}
				}
			}
		}
		if(win[M])
			printf("Little Deepu\n");
		else
			printf("Kate\n");
	}
	return 0;
}
