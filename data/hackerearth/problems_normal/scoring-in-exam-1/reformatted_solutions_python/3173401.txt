def read_ints():
    return map(int, raw_input().split())


def main():
    n, q = read_ints()
    ts = read_ints()
    ss = read_ints()
    pairs = zip(ss, ts)
    pairs.sort(reverse=True)
    ret = [0] * (n + 1)
    for i in xrange(1, n + 1):
        ret[i] = ret[i - 1] + pairs[i - 1][1]

    for i in xrange(q):
        k = read_ints()[0]
        print ret[k]


if __name__ == '__main__':
    main()
