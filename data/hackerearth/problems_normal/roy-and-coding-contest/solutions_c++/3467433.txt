#include <bits/stdc++.h>

using namespace std;

int main()
{

    long long t , n, m , ans = 0, st, con = 0;
    bool found = false;

    cin >> t;

    while(t--){

        cin >> n >> m;
        st = 1, con = 0, ans = 1;
        found = false;

        if(n == 1){
            cout << "0\n";
            continue;
        }

        while(1){

            ans += (st /2);
            con++;

            if(st > m){

                found = true;
                break;
            }
            st *= 2;
            if(ans >= n) break;
        }

        if(!found || ans >= n) cout << con << "\n";
        else cout << con + (n - ans) / m +  (((n - ans) % m) != 0) << "\n";
    }

    return 0;
}
