#include<bits/stdc++.h>
using namespace std;
#define ll long long int 
ll ones(ll n)
{
	ll ans=0;
	while(n>0)
	{
		ans++;
		n=n&(n-1);
	}
	return(ans);
}
bool cmp(pair<ll,pair<ll,ll> > a,pair<ll,pair<ll,ll> > b)
{
	
	if(a.second.first>b.second.first)
	return(false);
	else if(a.second.first<b.second.first)
	return(true);
	else if(a.second.second>b.second.second)
	return(false);
	else
	return true;
}
int main()
{
	
	//cout<<ones(656478043)<<endl;
	ll t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		ll arr[100009];
		//vector<pair<value,pair<numOfOne,i> > >
		vector< pair<ll,pair<ll,ll> > >res;
		for(ll i=0;i<n;i++)
		{
		 cin>>arr[i];
		 res.push_back(make_pair(arr[i],make_pair(ones(arr[i]),i)));
		}
		sort(res.begin(),res.end(),cmp);
		for(ll i=0;i<res.size();i++)
		cout<<res[i].first<<" ";
		cout<<endl;
		
	    
	}
	return(0);
}
