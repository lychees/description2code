#include <iostream>
#include<algorithm>
#include<limits.h>
using namespace std;


int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	int n,k;
    	cin>>n>>k;
    	long long int temp;
    	long long int maximum=INT_MIN;
    	long long int cnt=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>temp;
    		if(temp<=k)
    			cnt=cnt+temp;
    		else 
    		{
    			maximum=max(maximum,cnt);
    			cnt=0;
    		}
    	}
    	maximum=max(maximum,cnt);
    	t--;
    	cout<<maximum<<endl;
    }
}
