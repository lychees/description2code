#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define MAX 100010

char s[MAX];
int a[26];
int n;

int main(){
    scanf("%s", s);
    n = strlen(s);
    for (int i = 0; i < n; i++)
    	a[s[i] - 'a']++;
    n = 0;
    for (int i = 0; i < 26; i++)
    	n += a[i] & 1;
    printf("%s\n", n < 2 ? "YES" : "NO");
    return 0;
}
