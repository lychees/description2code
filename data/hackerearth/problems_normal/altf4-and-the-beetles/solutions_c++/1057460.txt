#include <bits/stdc++.h>

using namespace std;

const int MAXN = 100000 + 2;

int main()
{
    int test;
    cin >> test;
    while (test --) {
        int n;
        long long a, b, c;
        scanf("%d %lld %lld %lld", &n, &a, &b, &c);
        vector<pair<int, int> > t;
        for(int i = 0; i < n; i++) {
            int u, v;
            scanf("%d %d", &u, &v);
            t.push_back(make_pair(u, 1));
            t.push_back(make_pair(v + 1, -1));
        }
        long long res = max(a, c) * n;
        sort(t.begin(), t.end());
        t.push_back(make_pair((int)(1e9) + 10, 0));
        int j = 0, aa = n, bb = 0, cc = 0;
        for(int i = 1; i < t.size(); i++) {
            if (t[i].first > t[i - 1].first) {
                for(int k = j; k < i; k++) {
                    if (t[k].second == 1) aa--, bb++; else bb--, cc++;
                }
                res = max(res, a * aa + b * bb + c * cc);
                j = i;
            }
        }
        printf("%lld\n", res);
    }
}
