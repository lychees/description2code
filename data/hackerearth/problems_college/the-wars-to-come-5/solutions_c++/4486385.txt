#include <iostream>
using namespace std;
#include<vector>
#include<algorithm>
int main()
{
    vector <int> l;
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	int temp;
    	cin>>temp;
    	l.push_back(temp);
    }
    sort(l.begin(),l.end(),greater<int>());
    int tot;
    cin>>tot;
    int sum=0;
    int count=0;
    for(int j=0;(j<n)&&(sum<tot);j++)
    {
    	sum=sum+l[j];
    	count++;
    }
    cout<<count;
}
