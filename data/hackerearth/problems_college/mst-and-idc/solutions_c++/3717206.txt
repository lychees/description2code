#include <bits/stdc++.h>
using namespace std;
int main(void)
{
	ios_base::sync_with_stdio(0);
	int t;
	cin >> t;
	while(t--)
	{
		int x;
		cin >> x;
		vector<int> v;
		int val;
		for(int i=0;i<x;i++)
		{
			cin >> val;
			v.push_back(val);
		}	
		sort(v.begin(), v.end());
		cout << 2 * (v[x-1] - v[0]) << endl;
	}
	return 0;
}
