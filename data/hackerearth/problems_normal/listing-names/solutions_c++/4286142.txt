#include<bits/stdc++.h>
using namespace std;
void sol()
{
	map<string,int> m;
	string s;
	int n,i;
	cin>>n;
	for(i=0;i<n;++i)
	{
		cin>>s;
		m[s]++;
	}
	map<string,int> ::iterator z;
	for(z=m.begin();z!=m.end();++z)
	{
		cout<<(*z).first<<' '<<(*z).second<<endl;
	}
}
int main()
{
	sol();
	return 0;
}
