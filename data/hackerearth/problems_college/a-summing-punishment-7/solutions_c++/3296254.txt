#include <iostream>
using namespace std;
 
int main()
{
int N;
cin>>N;
int a[N];
   for(int i=0;i<N;i++)
   {
	cin>>a[i];
    }
int Q,l,r;
cin>>Q;
for(int i=0;i<Q;i++)
    {
    	int sum=0;
    	cin>>l;
    	cin>>r;
    	for(int j=l-1;j<r;j++)
    	{
    		sum=sum+a[j];
    	}
    	cout<<sum<<endl;
    	
    }
}
