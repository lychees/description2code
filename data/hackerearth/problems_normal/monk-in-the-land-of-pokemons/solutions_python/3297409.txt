T = int(raw_input())
for test in range(T):
	N = int(raw_input())
	array = []
	
	maximum = 0
	for pond in range(N):
		string = raw_input()
		appendage = string.split()
		array.append(appendage)
		food = int(appendage[0])
		pokemon = int(appendage[1])
		if pokemon > maximum:
			maximum = pokemon
		if food > maximum:
			maximum = food
	
	initial_backpack_number = 0
	backpack = {x: 0 for x in range(maximum + 1)}
	
	number_of_ponds = len(array)
	for pond in range(number_of_ponds):
		food = int(array[pond][0])
		pokemon = int(array[pond][1])
		
		if food != pokemon:
			if backpack[pokemon] > 0:
				backpack[pokemon] += -1
			else:
				initial_backpack_number += 1
			backpack[food] += 1
	
	print(initial_backpack_number)
