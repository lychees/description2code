def gcd(a, b):
    if b == 0:
        return a
    else:
        return gcd(b, a % b)


cases = int(raw_input())
for i in xrange(cases):
    a, b = map(int, raw_input().split())
    if a == 1 and b == 1:
        print "Is a Co-Prime"
    elif gcd(a, b) == 1:
        print "Is a Co-Prime"
    else:
        print "Not a Co-Prime"
