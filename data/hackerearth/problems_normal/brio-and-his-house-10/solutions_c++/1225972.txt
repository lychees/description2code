#include <iostream>
#include<algorithm>
#include<iomanip>
#include<cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	double x,y,z;
    	cin>>x>>y>>z;
    
    	double m = (x*y*z)/sqrt((x+y+z)*(x+y-z)*(x+z-y)*(z+y-x));
    	double area = 3.1415*m*m;
    	cout<<setprecision(4)<<fixed<<area<<endl;
    }
}
