#include <iostream>
using namespace std;
#define max 100000
int fun(int a[max],int n){
	int maxsofar=0,maxending=0;
	for(int i=0;i<n;i++){
		maxending=maxending+a[i];
		if(maxending<0){
			maxending=0;
		}
		if(maxending>maxsofar){
			maxsofar=maxending;
		}
	}
	return maxsofar;
}
int main()
{
	int n;
	int a[max];
   cin>>n;
   for(int i=0;i<n;i++){
   	cin>>a[i];
   } 
   cout<<fun(a,n);
}
