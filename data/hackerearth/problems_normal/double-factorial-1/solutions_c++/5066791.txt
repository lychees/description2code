#include <iostream>
#include <vector>
#include <cmath>
using namespace std;
vector<string> fact;
int main()
{
   	long long int p1,p2,t,n,m;
   	bool allZero;
   	fact.resize(5);
   	fact[0]="1";
   	fact[1]="1";
   	fact[2]="2";
   	fact[3]="720";
   	fact[4]="620448401733239439360000";
   	cin>>t;
   	while(t--)
   	{
   		cin>>n>>m;
   		if(m==0)
   		{
   			cout<<0<<endl;
   			continue;
   		}
   		if(n>=5)
   		{
   			cout<<0<<endl;
   			continue;
   		}
   		allZero=true;
   		for(p1=max(0ll,(long long int)(fact[n].size()-m));p1<fact[n].size();p1++)
   			if(fact[n][p1]!='0')
   			{
   				allZero=false;
   				break;
   			}
   		if(allZero)
   		{
   			cout<<0<<endl;
   			continue;
   		}
   		for(p1=max(0ll,(long long int)(fact[n].size()-m));p1<fact[n].size();p1++)
   			if(fact[n][p1]!='0')
   				break;
   		for(;p1<fact[n].size();p1++)
   			cout<<fact[n][p1];
   		cout<<endl;
   	}
    return 0;
}
