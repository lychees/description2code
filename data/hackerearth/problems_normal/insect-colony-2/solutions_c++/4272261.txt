#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    for(int i=1;i<=t;i++)
    {
    	int n;
    	cin>>n;
    	long long a[n],c=0;
    	for(int j=0;j<n;j++)
    	{
    		cin>>a[j];
    		if(a[j]%2!=0)
    		c++;
    	}
    	
    	if(c%2==0)
    	cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
    }
    
    return 0;
}
