#include<iostream>
#include<vector>
using namespace std;
#define maxv 10001
struct spe{
	int value;
	int count;
};
int main()
{
   vector<int> vecp(maxv,1);
   vecp[1]=0;
   for(int i=2;i<maxv;i++){
   	int j=2;
   	while(j*i<maxv){
   		vecp[j*i]=0;
   		j++;
   	}
   }
   int n,k;
   cin>>n>>k;
   vector<vector<int> >vec(n+1,vector<int> (n+1));
   vector<spe> vecs(maxv);
   for(int i=1;i<maxv;i++){ 
   vecs[i].value=vecp[i];
   vecs[i].count=0;
   }
   for(int i=maxv-1;i>=1;i--){
   	if(vecs[i].value==0)
   	vecs[i].count=1000000;
   	else{
   		if(i-k>=1){
   			if(vecs[i-k].value==0){
   			vecs[i-k].count=vecs[i].count+1;
   			vecs[i-k].value=1;
   			}
   		}
   	}
   }
   //for(int i=1;i<50;i++) cout<<i<<" "<<vecs[i].count<<endl;
   int temp,ans=1000,minx,miny;
   for(int i=1;i<=n;i++){
   	for(int j=1;j<=n;j++)
   	cin>>vec[i][j];
   }
   for(int i=2;i<n;i++){
   	for(int j=2;j<n;j++){
   		temp=vecs[ vec[i][j] ].count+vecs[ vec[i-1][j-1] ].count+vecs[ vec[i+1][j-1] ].count+vecs[ vec[i-1][j+1] ].count+vecs[ vec[i+1][j+1] ].count;
   		if(temp<ans)
   		{
   			ans=temp;
   			minx=i;
   			miny=j;
   		}
   	}
   }
   if(ans==1000)
   cout<<"no"<<endl;
   else{
   	cout<<"yes"<<endl;
   	cout<<ans<<endl;
   	cout<<minx<<" "<<miny;
   }
    return 0;
}
