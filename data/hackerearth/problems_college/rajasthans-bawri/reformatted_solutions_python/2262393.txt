test = int(raw_input())
while (test):
    flag = 1
    arr = map(int, raw_input().split(' '))
    if (arr[0] >= arr[2]):
        if (arr[1] >= arr[3]):
            flag = 0
        elif (arr[0] >= arr[3]):
            if (arr[1] >= arr[2]):
                flag = 0
    elif (arr[0] >= arr[3]):
        if (arr[1] >= arr[2]):
            flag = 0
    if (flag):
        print"Not Possible"
    else:
        print"Possible"

    test -= 1
