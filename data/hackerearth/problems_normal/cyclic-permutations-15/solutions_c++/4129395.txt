#include <iostream>
#include <string>
#include<vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a,b;
    	cin>>a>>b;
    	b+=b;
    	string str = a+b;
    	vector<int> z_fn(str.size());
    	int L = a.size();
    	for(int i=1,l=0,r=0;str[i];++i)
    	{
    		int k = 0;
    		if(i<=r)
    		{
    			k = min(r-i+1,z_fn[i-l]);
    		}
    		while(i+k<str.size() && str[k]==str[i+k])++k;
    		z_fn[i] = k;
    		if(i+k-1>r)
    		{
    			l = i,r =i+k-1;
    		}
    	}
    	int ans=0;
    	for(int i=L;i<2*L;++i) ans+=z_fn[i]>=L;
    	cout<<ans<<endl;
    }
    return 0;
}
