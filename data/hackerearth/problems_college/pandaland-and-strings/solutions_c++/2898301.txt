typedef unsigned long long int uint64;

struct node{
	bool done;
	struct node *links[26];
	node(){
		done=false;
		memset(links,false,sizeof(links));
	}
};

bool ins(struct node *root,string s){
	bool ret=false;
	for(int i=0;i<s.length();i++){
		if(root->links[s[i]-'a']==NULL)
		root->links[s[i]-'a']=new node();
		root=root->links[s[i]-'a'];
		ret|=root->done;
	}
	root->done=true;
	return ret;
}

int main(){
	int t,n,i;
	string s;
	cin>>t;
	while(t--){
		struct node *root=new node();
		cin>>n;
		vector<string>v;
		
		bool flag=false;
		for(i=0;i<n;i++){
			cin>>s;
			v.pb(s);
		}
		sort(all(v));
		for(i=0;i<v.size();i++)
		flag|=ins(root,v[i]);
		
		if(flag)
		printf("NO\n");
		else
		printf("YES\n");
	}
	return 0;
}
