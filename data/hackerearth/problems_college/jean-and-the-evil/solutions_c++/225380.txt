int  readint()
{
    int cc = getc(stdin);
    while (cc < '0' || cc > '9')
                cc = getc(stdin);
        int ret = 0;
        while (cc >= '0' && cc <= '9') {
                ret = ret * 10 + cc - '0';
                cc = getc(stdin);
        }
        return ret;
}
inline void writeint (LL n)
{
    LL N = n, rev, count = 0;
    rev = N;
    if (N == 0) { pc('0'); pc('\n'); return ;}
    while ((rev % 10) == 0) { count++; rev /= 10;}
    rev = 0;
    while (N != 0) { rev = (rev<<3) + (rev<<1) + N % 10; N /= 10;}
    while (rev != 0) { pc(rev % 10 + '0'); rev /= 10;}
    while (count--) pc('0');
    pc('\n');
    return ;
}
