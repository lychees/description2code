#include <bits/stdc++.h>
using namespace std;


int main()
{
	int n,x,y=0,ind=0,p=0,x5;
	cin >> x >> n;
	int a[n][2],b[n][2];
	memset(a,0,sizeof(a));
	memset(b,0,sizeof(b));
	
	for(int i=0;i<n;i++){
		cin >> a[i][0] >> a[i][1] >> b[i][0] >> b[i][1];
	}
	if(n==0){
		cout << x << endl;
		return 0;
	}
	
	for(int i=0;i<n;i++){
		if(x>=a[i][0] && x<=b[i][0]){
		y = max(y,(b[i][1]-a[i][1])*(x-a[i][0])/(b[i][0]-a[i][0])+a[i][1]);
		if(y==(b[i][1]-a[i][1])*(x-a[i][0])/(b[i][0]-a[i][0])+a[i][1])
		ind = i;
		}
	}
	if(b[ind][1]>a[ind][1]){
		y=a[ind][1];
		x=a[ind][0];
	}
	else{
		y=b[ind][1];
		x=b[ind][0];
	}
	a[ind][0]=-1;a[ind][1]=-1;b[ind][0]=-1;b[ind][1]=-1;
	if(n==1){
		cout << x << endl;
		return 0;
	}
	
	p=0;
	int u=-1;
	for(int j=0;j<n;j++){
		for(int i=0;i<n;i++){
			if(a[i][0]!=-1 ){
			int l = (b[i][1]-a[i][1])*(x-a[i][0])/(b[i][0]-a[i][0])+a[i][1];
			if(x>=a[i][0] && x<=b[i][0] && l<=y){
		p = max(p,(b[i][1]-a[i][1])*(x-a[i][0])/(b[i][0]-a[i][0])+a[i][1]);
		if(p==(b[i][1]-a[i][1])*(x-a[i][0])/(b[i][0]-a[i][0])+a[i][1])
		ind = i;
			}
			}
	}
	if(u!=ind){
	if(b[ind][1]>a[ind][1] && a[ind][0]!=-1){
		y=a[ind][1];
		x=a[ind][0];
	}
	else if(a[ind][0]!=-1){
		y=b[ind][1];
		x=b[ind][0];
	}
	u=ind;
	}
	a[ind][0]=-1;
	p=0;
	
	}
	
	cout << x << endl;
	
    return 0;
}
