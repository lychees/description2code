               // cout<<"Value of log[i] = " <<tempx<<endl;
                tempx++;
                tempy++;
        //cout<<"Value of log[i] = " <<tempx<<endl;
            }else{
                if(x < tempx){
                    x = tempx;
                }
                if(y < tempy){
                    y = tempy;
                }
                tempx = 0;
                tempy = 0;
            }
        }
        if(x < tempx){
            x = tempx;
        }

            y = tempy;
        }

    }

}
