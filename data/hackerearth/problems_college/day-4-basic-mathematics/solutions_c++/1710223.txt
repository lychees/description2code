#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int T;
    long long int D = 0;
    long double alpha = 0;
    long long result = 0;
    
    cin>>T;
    
    while(T--){
    	cin>>D;
    	alpha = (sqrt(8*D + 1) - 1)/2;
    	result = int(ceil(alpha));
    	if(result % 3 == 0){
    		cout<<"SL"<<endl;
    	} else if(result % 3 == 1){
    		cout<<"LB"<<endl;
    	} else {
    		cout<<"BS"<<endl;
    	}
    }
    return 0;
}
