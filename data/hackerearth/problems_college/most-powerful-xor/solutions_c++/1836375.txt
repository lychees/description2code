#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int t, L, R;
    cin >> t;
    while (t--) {
    	cin >> L >> R;
    	int maxRet = 0;
    	int i, j;
    	for (i = L; i <= R; i++)
    	  for (j = i + 1; j <= R; j++)
    	    maxRet = max(maxRet, i ^ j);
    	
    	cout << maxRet << endl;
    }
    
    return 0;
}
