const int lim=1e5+5;

int main()
{
  ios_base::sync_with_stdio(false);cin.tie(NULL);
  //freopen("input.txt", "r", stdin);
  ll t,i,j,k,n;
  cin>>t;
  while(t--)
  {
  	string s;
  	cin>>n;
  	cin>>s;
  	
	ll ans1=0,ans2=0,o=0,z=0;
  	i=0;
  	while(i<n)
  	{
  	  if(s[i]=='1')
		o++;
	  else if(s[i]=='0')
	    z++;	
	    
		i++;
	}
	
	for(i=0;i<o;i++)
	{
		if(s[i]=='0')
		ans1++;
	}
	
	for(i=0;i<z;i++)
	{
		if(s[i]=='1')
		ans2++;
	}
	
	cout<<min(ans1,ans2)<<endl;
	
  }
  return 0;
}
