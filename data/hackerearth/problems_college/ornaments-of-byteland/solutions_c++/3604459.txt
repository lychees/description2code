#include<iostream>
using namespace std;

int main()
{
   int i,n,al[26]={},j,cnt=0;
   cin>>n;
   string arr[n];
   for(i=0;i<n;i++)
   {	cin>>arr[i];
	for(j=0;j<arr[i].size();j++)
		if(al[arr[i][j]-'a']==i)
			al[arr[i][j]-'a']++;
   }
   for(i=0;i<26;i++)
	if(al[i]==n)
		cnt++;
   cout<<cnt<<endl;
   return 0;
}
