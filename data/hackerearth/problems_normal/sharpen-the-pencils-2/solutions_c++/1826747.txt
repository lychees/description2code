#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
	int n;
	cin>>n;
	int *arr = new int[n];

	for(int i=0; i<n; i++)
	    cin>>arr[i];

	int left = 0;
	int right = n-1;

	int sum_right = arr[n-1];
	int sum_left = arr[0];
	while(left < right) {
	    while(1) {
		if(sum_left <= 2 * sum_right && left + 1 < right) {
		    left++;
		    sum_left += arr[left];
		}
		else {
		    break;
		}
	    }
	    if(left<right) {
		right--;
		sum_right += arr[right];
	    }
//	    cout<<left<<" "<<right<<endl;
	}

	cout<<left+1<<" "<<n-right-1<<endl;
    }
}
