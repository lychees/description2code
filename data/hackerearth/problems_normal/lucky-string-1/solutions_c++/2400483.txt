#include<bits/stdc++.h>
using namespace std;
int main(){
	long long i,t,n,a,b,j,k;
	scanf("%lld",&t);
	long long ar[48];
	ar[1]=2;
	for(i=2;i<46;i++)
		ar[i]=ar[i-1]+(pow(2,i)*i);
	//printf("%lld\n",ar[45]);
	while(t--){
		long long c=1;
		scanf("%lld",&n);
		for(i=1;i<46;i++){
			if(ar[i]>n){
				k=i;
				break;
			}
			if(ar[i]==n){
				c=0;
				printf("Earth\n");
				break;
			}
		}
		if(c==1){
		a = n-ar[k-1];
		if(a%k==0){
			a/=k;
			b=k;
		}
		else{
			long long x=a;
			a = (a/k)+1;
			b = x%k;
		}
		b = k-b;
		b = pow(2,b);
		j = (a-1)/b;
		if(j%2==0)
			printf("Hacker\n");
		else
			printf("Earth\n");
		}
	}
	return 0;
}
