#include <iostream>
using namespace std;

long* pData;
long Size;
long MaxData=0;

void InitData()
{
	cin>>Size;
	pData=new long[Size];
	for(long i=0; i<Size; i++)
	{
		cin>>pData[i];
		if(pData[i]>MaxData)
		{
			MaxData=pData[i];
		}
	}
}

void PrintData()
{
	for(long i=0; i<Size; i++)
	{
		cout<<pData[i]<<" ";
	}
}

void DoWork()
{
	long numDiv;
	cin>>numDiv;
	long div=1;
	bool allZero=false;
	for(long i=0; i<numDiv; i++)
	{
		long data;
		cin>>data;
		div*=data;
		if(div>MaxData)
		{
			allZero=true;
			break;
		}
	}
	
	for(long i=0; i<Size; i++)
	{
		if(!allZero)
		{
			pData[i]/=div;
		}
		else
		{
			pData[i]=0;
		}
	}
	// cout<<div<<endl;
}

int main()
{
    // cout << "Hello World!" << endl;
    InitData();
    // PrintData();
    DoWork();
    PrintData();
    delete[] pData;
    pData=NULL;
    return 0;
}
