#include<bits/stdc++.h>
using namespace std;
main()
{
	int n,m,i,sum;
	scanf("%d",&n);
	while(n--)
	{
		sum=0;
	scanf("%d",&m);
	int a[m],b[m];
	for(i=0;i<m;i++)
	scanf("%d",&a[i]);
	for(i=0;i<m;i++)
	scanf("%d",&b[i]);
	sort(a,a+m);
	sort(b,b+m);
	for(i=0;i<m;i++)
	sum=sum+a[i]*b[i];
	printf("%d\n",sum);
	}
}
