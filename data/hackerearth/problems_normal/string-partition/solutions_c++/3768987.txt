# include<iostream>
# include<string>
# include<cstring>
using namespace std;

const long long MAXN=1e18;
int n,k;
long long z;
long long dp[128][128];
string s;
long long solve(int i, int k)
{
	if(i==n)
	{
		if(k) return -MAXN;
		return 0;
	}
	if(dp[i][k]!=-1) return dp[i][k];
	long long ans=-MAXN,ch=0;
	for(int j=i; j<n; j++)
	{
		ch=ch*10+(s[j]-'0');
		if(ch>z) break;
		ans=max(ans,ch+solve(j+1,k-1));
	}
	return dp[i][k]=ans;
}


int main()
{
    int t;
    cin>>t;
    for(int tc=0; tc<t; tc++)
    {
    	cin>>k>>z;
    	cin>>s;
    	n=s.size();
    	memset(dp,-1,sizeof(dp));
    	long long ans=solve(0,k);
    	if(ans<0) cout<<-1<<endl;
    	else cout<<ans<<endl;
    }
    return 0;
}
