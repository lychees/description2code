#include<bits/stdc++.h>
using namespace std;
long long arr[100005];
int main()
{
    long long n,q,l,r,i,lef,ryt,mid,mini,maxi;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    sort(arr,arr+n);

    cin>>q;
    for(i=0;i<q;i++)
    {
        cin>>l>>r;
        lef=0;
        ryt=n;
        mini=LLONG_MAX;
        maxi=-1;
        while(lef<ryt)
        {
            mid=(lef+ryt)/2;
            if(arr[mid]>=l)
            {
                mini=min(mini,mid);
                ryt=mid;
            }
            else
            {

                lef=mid+1;
            }
        }
        lef=0;
        ryt=n;
        while(lef<ryt)
        {
            mid=(lef+ryt)/2;
            if(arr[mid]<=r)
            {
                lef=mid+1;
                maxi=max(maxi,mid);
            }
            else
            {

                ryt=mid;
            }
        }
       // cout<<maxi<<" ** "<<mini<<endl;
        if(maxi-mini+1>=0)
        cout<<maxi-mini+1<<"\n";
        else cout<<0<<"\n";
    }
    return 0;
}
