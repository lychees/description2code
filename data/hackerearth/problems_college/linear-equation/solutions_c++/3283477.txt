#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	int sum=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		sum+=a[i];
    	}
    	sum=sum/(n-1);
    	for(int i=0;i<n;i++)
    	{
    		cout<<sum-a[i]<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
