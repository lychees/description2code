#include <iostream>
#define ll long long 
using namespace std;

int main()
{
	
    int t,numMisC;
    ll sum=0;
    char c;
    cin>>t;
    while(t--){
    	string word;
    	sum=0;
    	cin>>word;
    	int l = word.length();
    	for(int i=0;i<l;i++)
    	sum+=(word[i]-'a'+1);
    	cin>>numMisC;
    	for(int j=0;j<numMisC;j++){
    		cin>>c;
    		
    		for(int k=0;k<l;k++)
    			if(word[k]==c)
    				sum-=(c-'a'+1);
    	}
    	
    	cout<<sum<<"\n";
    	
    	
    }
    
}
