#include <iostream>
#define ll long long
#define MOD 1000000007
using namespace std;
ll powmod(ll base,ll exponent,ll mod){
	ll ans=1;
	while(exponent>0){
		while(exponent%2==0){
			base*=base;
			base%=MOD;
			exponent/=2;
		}
		exponent--;
		ans*=base;
		ans%=MOD;
	}
	return ans;
}
int main() {
	// your code goes here
	int t;
	ll n,answer;
	cin>>t;
	while(t--){
		cin>>n;
		n--;
		n%=(MOD-1);
		answer=powmod(2,n*(MOD-2),MOD);
		cout<<answer<<endl;
	}
	return 0;
}
