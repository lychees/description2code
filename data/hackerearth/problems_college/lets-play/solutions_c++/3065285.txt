#include<bits/stdc++.h>
using namespace std;
int f1[1000001],f2[10000001],f3[1000001],f4[1000001],k;
void solve(int i){
	if(i<=0) return ;
	int &ref = f1[i];
	int &ref2 = f2[i];
	if(ref!=-1 && ref2!=-1) return ;
	long long int cur = INT_MAX;
	for(int j=i-1;j>=i-k && j>=1;j--){
		solve(j);
	//	if(i==2) cout << f1[i] << f2[i] << endl;
	   if(cur>max(f1[j],f3[i])*max(f2[j],f4[i])){
	   	cur = max(f1[j],f3[i])*max(f2[j],f4[i]);
	   	f1[i] = max(f1[j],f3[i]);
	   	f2[i] = max(f2[j],f4[i]);
	   }
	}
	return ;	
}
int main(){
	int n;
	cin >> n >>k;
	for(int i=1;i<=n;i++) {
		cin >> f3[i] >> f4[i];
	}
	memset(f1,-1,sizeof(f1));
	memset(f2,-1,sizeof(f2));
	f1[0]  = 0;
	f2[0] = 0;
	for(int i=1;i<=k;i++)
	{
		f1[i] = f3[i];
		f2[i] = f4[i];
	}
	
	solve(n+1);
//	for(int i=1;i<=n;i++)
//	cout << f1[i] << endl;
	cout << f1[n+1]*f2[n+1] << endl;
	
	
	return 0;
}
