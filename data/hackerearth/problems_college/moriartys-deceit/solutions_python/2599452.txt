import sys
def gcd(a,b):
	if b==0:
		return a
	else:
		return gcd(b,a%b)
t = int(input())
while(t!=0):
    n,m = map(int,sys.stdin.readline().split())
    a = list(map(int,sys.stdin.readline().split()))
    output = []
    flag = 0
    for i in a:
        if gcd(i,m)==1:
        	flag = 1
        	output.append(i)
    for i in output:
    	print i,
    if flag==1:
    	print ""
    else:
    	print "-1"
    t = t-1
