def next_lucky(digs, ind):
        if ind == len(digs):
                return True
        if digs[ind] < 3:
                for i in xrange(ind, len(digs)):
                        digs[i] = 3
                return False
        if digs[ind] == 4:
                digs[ind] = 5
                for i in xrange(ind+1, len(digs)):
                        digs[i] = 3
                return False           
        if digs[ind] > 5:
                for i in xrange(ind, len(digs)):
                        digs[i] = 3
                return True
        if digs[ind] == 3:
                carry = next_lucky(digs, ind+1)
                if carry:
                        digs[ind] = 5
                return False
        if digs[ind] == 5:
                carry = next_lucky(digs, ind+1)
                if not carry:                  
                        return False
                digs[ind] = 3
                return True
