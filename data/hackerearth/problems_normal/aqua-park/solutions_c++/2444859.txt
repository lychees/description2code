#include <bits/stdc++.h>
using namespace std;
int a[102][2], b[102][2];
int cross(int A[2], int B[2], int C[2] ) 
{
	if(B[1] == INT_MAX) return -1;
    int AB[2];
    int AC[2];
    AB[0] = B[0]-A[0];
    AB[1] = B[1]-A[1];
    AC[0] = C[0]-A[0];
    AC[1] = C[1]-A[1];
    int cross = AB[0] * AC[1] - AB[1] * AC[0];
    return cross;
}
int main()
{
    int n, X, Y = INT_MAX;
    scanf("%d %d",&X, &n);
    for(int i=0;i<n;i++) scanf("%d %d %d %d",&a[i][0], &a[i][1], &b[i][0], &b[i][1]);
    
    while(true) {
    	//cout << X << ' ' << Y << endl;
    	bool f = false;
    	int id = -1;
    	double nxt = -1000000000;
    	for(int i=0;i<n;i++) {
    		if((a[i][0] <= X && X <= b[i][0]) || (a[i][0] >= X && X >= b[i][0])) {
    			int B[2];
				B[0] = X;
				B[1] = Y;
    			int ch;
				if(a[i][0] < b[i][0]) ch = cross(a[i], B, b[i]);
				else ch = cross(b[i], B, a[i]);
				if(ch >= 0) continue;
				double cur = a[i][1] + ((b[i][1]-a[i][1])/(b[i][0]-a[i][0]))*(X-a[i][0]);
				//cout << cur << endl;
				if(cur > nxt) {
					f = true;
					nxt = cur;
					id = i;
				}
    		}
    	}
    	if(!f) break;
    	if(a[id][1] < b[id][1]) {
    		X = a[id][0];
    		Y = a[id][1];
    	}
    	else {
    		X = b[id][0];
    		Y = b[id][1];
    	}
    }
    cout << X << endl;
    return 0;
}
