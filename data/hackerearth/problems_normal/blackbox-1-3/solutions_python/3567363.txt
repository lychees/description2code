def gcd(a, b):
	if(a == 0):	return b;
	return gcd(b % a, a);

def pt(a, b, c):
	if(((a * a) + (b * b) == (c * c)) or ((b * b) + (c * c) == (a * a)) or ((c * c) + (a * a) == (b * b))):	return 1;
	return 0;

t = input();
while(t > 0):
	a, b, c = map(int, raw_input().split());
	if(pt(a, b, c) == 1 and gcd(a, gcd(b, c)) == 1):	print "YES";
	else:	print "NO";
	t -= 1;
