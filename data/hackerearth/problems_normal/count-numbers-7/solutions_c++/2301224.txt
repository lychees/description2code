#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int ans,a,b,x,k,t;
	cin>>k>>t;
	ll int r[k];
	ll int i,j;
	for(i=0;i<k;i++)
	{
		//~ scanf("%lld",r+i);
		cin>>r[i];
	}
	int arr[k],mc;
	while(t--)
	{
		cin>>a>>b;
		ans=0;
		memset(arr,0,sizeof(arr));
		ll int n=(1<<k);
		for(i=1;i<n;i++)
		{
			mc=0;
			x=1;
			for(j=(k-1);j>=0;j--)
			{
				arr[j]=(arr[j]+1)%2;
				if(arr[j])
				{
					break;
				}
			}
			for(j=0;j<k;j++)
			{
				if(arr[j])
				{
					x*=r[j];
					mc++;
				}
			}
			if(mc&1)//odd
			{
				ans+=(b/x - (a-1)/x);
			}
			else
			{
				ans-=(b/x - (a-1)/x);
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
