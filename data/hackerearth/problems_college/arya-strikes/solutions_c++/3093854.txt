#include <cstdio>
#include <string.h>
using namespace std;

bool dp[55][55];
int reja[55][55];

void mdp(int x, int y, int n){
	if(x == y && x + 1 == n){
		dp[n - 1][n - 1] = true;
		printf("POSSIBLE\n");
	}else{
		dp[x][y] = true;
		if(y + 1 < n && !dp[x][y + 1] && reja[x][y + 1] == 1){
			
			mdp(x, y + 1, n);
		}
		if(x + 1 < n && !dp[x + 1][y] && reja[x + 1][y] == 1){
			
			mdp(x + 1, y, n);
		}
	}
}


int main() {
	int t, n, i, j;
	scanf("%d", &t);
	while(t > 0){
		scanf("%d", &n);
		for(i = 0; i < n; i++){
			for(j = 0; j < n; j++){
				scanf("%d", &reja[i][j]);
				dp[i][j] = false;
			}
		}
		mdp(0,0,n);
		if(!dp[n - 1][n - 1]){
			printf("NOT POSSIBLE\n");
		}
		t--;
	}
	
	return 0;
}
