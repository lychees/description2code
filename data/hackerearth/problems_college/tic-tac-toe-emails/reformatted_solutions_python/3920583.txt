import collections
noOfTestCases = int(raw_input())


def uniqueSubjects(subjects):
    subjectMap = collections.OrderedDict()
    for subject in subjects:
        if (subjectMap.has_key(subject) == False):
            subjectMap[subject] = 1
        else:
            subjectMap[subject] += 1

    uniqueSubjects = ""

    for subject in subjectMap.keys():

        uniqueSubjects += subject + " "
    print uniqueSubjects


for x in range(1, noOfTestCases + 1):
    uniqueSubjects(raw_input().split(" "))
