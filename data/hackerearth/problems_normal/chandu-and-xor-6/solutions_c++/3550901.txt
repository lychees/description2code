#include<iostream>
using namespace std;
int matrix[1000][1000];
int rowSum[1000];
int colSum[1000];
int row,col;
int calcMax()
{
	int MaxXOR = 0;
    for(int i=0; i<row; i++) {
        for(int j=0; j<col; j++) {
            int xorValue = rowSum[i]^colSum[j];
            if(xorValue > MaxXOR)
                MaxXOR = xorValue;
        }
    }
    return MaxXOR;
}
int main() {
    
    cin>>row>>col;
    int temp=0;
    for(int i=0; i<row; i++) {
        temp=0;
        for(int j=0; j<col; j++) {
            cin>>matrix[i][j];
            temp += matrix[i][j];
        }
        rowSum[i] = temp;
    }
    for(int j=0;j<col;j++) {
        temp=0;
        for(int i=0; i<row; i++) {
            temp += matrix[i][j];
        }
        colSum[j] = temp;       
    }
    cout<<calcMax();
    return 0;
}
