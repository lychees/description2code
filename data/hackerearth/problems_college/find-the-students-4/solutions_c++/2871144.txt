#include <cstdio>
#include <iostream>
#include <vector>
#include <algorithm>

#define DEBUG if(0)

using namespace std;

int main()
{
    int N,tmp,minE = 1010101,newMin = 10101010;
    cin >> N;
    vector<int> input;

    for(int i = 0; i < N; ++i)
    {
        cin >> tmp;
        input.push_back(tmp);
    }
    sort(input.begin(),input.end());

    DEBUG {
        for(int i = 0; i < input.size(); ++i)
            cout << input[i] << " ";
        cout << endl;
    }

    int ans = input.size();

    cout << ans << endl;
    int i = 0;
    
    while(ans > 0 && i < input.size())
    {
        int minV = input[i];
        int rem = 1;
        while(input[++i] == minV)
            ++rem;
        ans = ans - rem;
        DEBUG cout << minV << " : " << rem << endl;
        if(ans > 0) cout << ans << endl;
    }
    
    return 0;
}
