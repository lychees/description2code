#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int tc;
	cin >> tc;
	map<string,long int> mymap;
	while(tc--)
	{
		string s;
		cin >> s;
		mymap[s]++;
	}
	
	for(map<string,long int>::iterator it=mymap.begin();it!=mymap.end();it++)
		cout << it->first << " " << it->second << endl;
	return 0;
}
