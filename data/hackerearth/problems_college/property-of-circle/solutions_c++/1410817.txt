#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int n,x,y,r1,x1,y1,r2;
    cin>>n;
    while(n-->0)
    {
    	cin>>x;
    	cin>>y;
    	cin>>r1;
    	cin>>x1;
    	cin>>y1;
   		cin>>r2;
   		double d=sqrt((x-x1)*(x-x1)+(y-y1)*(y-y1));
   	if(d<abs(r1-r2))
    {
        if(r1>r2)
        {
            cout<<"C1CC2"<<endl;
        }
        if(r1<r2)
        {
            cout<<"C2CC1"<<endl;
        }
 
    }
    if(d==abs(r1-r2))
    {
        if(r1>r2)
        {
            cout<<"C1CC2"<<endl;
        }
        if(r1<r2)
        {
            cout<<"C2CC1"<<endl;
        }
        if(r1==r2)
            cout<<"C2~C1"<<endl;
    }
    if(d>abs(r1-r2))
        cout<<"NOT SATISFIED"<<endl;
    }
}
