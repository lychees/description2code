#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,i;
    float d;
    scanf("%d",&n);
    float a[n];
    for(i=0;i<n;i++)
    {
    	scanf("%f",&a[i]);
    }
    if(n==3)
    {
    	printf("0\n%f %f %f",a[0],a[1],a[2]);
    }	
    else
    {
    	d=1000000000;
    	sort(a,a+n);
    	for(i=0;i<n-1;i++)
    	 d=min(d,float(a[i+1]-a[i])/2);
    	printf("%f\n%f ",d,a[0]); 
    	for(i=1;i<n-1;i++) 
    	 if(a[i+1]-a[i]>2*d)
    	 {
    	 	printf("%f ",a[i]+d);
    	 }
    	 else
    	 {
    	 	printf("%f ",a[i]+d);
    	    if(-1*a[i++]+a[i]!=2*d&&i<n-1)
    	 	  ;
    	 }
    }
    return 0;
}
