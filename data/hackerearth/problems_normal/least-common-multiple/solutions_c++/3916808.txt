// Miller Rabin Primarility Testing
ll  bigmod(ll a,ll b,ll mod) { //Bigmod
    ll ans=0;
    while(b){
        if (b&1)
            ans=(ans+a)%mod;
        a=(a+a)%mod;
        b>>=1;
    }
    return ans;
}
// Random Number Generator
ll  randomll () {
    ll  ans = 0;
    FOR(i,0,3)
        ans = (ans << 16) ^ rand();
    return ans;
}
// Binary Represt.
ll  binByMod(ll  a, ll  to, ll  mod) {
    ll ans=1;
    while(to){
        if(to&1)
            ans=bigmod(ans,a,mod);
        a=bigmod(a,a,mod);
        to>>=1;
    }
    return ans;
}
// Prime Testing
bool isPrime(ll  n) {
    if (n % 2 == 0 && n != 2||n==1)
        return false;
    FOR(i,0,20) {
        ll  a = (randomll () % (n - 1));
        if (a < 0) a += n - 1;
        a++;
        a = binByMod(a, (n - 1) / 2, n);
        if (a != n - 1 && a != 1)
            return false;
    }
    return true;
}
bool mark[40000005];
int prime[3000005];
int  prime_gen(int n)
{
    for(int i=3;i*i<=n;i+=2)
        if(!mark[i])
            for(int j=i*i,k=i<<1;j<=n;j+=k)
            mark[j]=true;
    int k=0;
    prime[k++]=2;
    for(int i=3;i<=n;i+=2)if(!mark[i])prime[k++]=i;
    return k;
}

{
    ll T,n;

    scanf("%lld",&n);
    ll sq=sqrt(n+1);
    ll pc=prime_gen(sq);

    ll now=n/2+1;
    while(!isPrime(now))now++;


    FOR(i,0,pc-1)
    {
        ll mx=0;
        if(prime[i]<=n)
        {
            ll prr=prime[i];
            while(prr<=n)
            {
                mx=max(mx,prr);
                prr*=prime[i];
            }
          //  pr(mx);
            ll nn=n/mx;

        }
    }

}
