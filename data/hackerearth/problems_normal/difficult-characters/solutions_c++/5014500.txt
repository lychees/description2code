#include<bits/stdc++.h>
using namespace std;
#define N '\n'
#define ll long long
#define ii int
#define S string
#define I cin
#define O cout
#define fr(j) for(i=0;i<j;i++)
#define fe(k) for(l=1;l<=k;l++)
#define ft(test) for(;test;--test)

bool adj[10000];
int level[10000];
ll dist[10000];

int main()
{
ll test,e,i,j,k,x,y,m=0,n=0,q,flag=0,sum,x1,y1,x2,y2,x3,y3,a,b,c,l;
I>>test;
S s;
ft(test)
{
I>>s;
ll arr[26];
vector <pair<ll,char> >v;
memset(arr,0,sizeof(arr));
for(i=0;i<s.size();i++)
arr[int(s[i]-'a')]++;

for(i=25;i>=0;--i)
{
if(arr[i]==0) O<<char(i+'a')<<" ";
else 
{
	v.push_back(make_pair(arr[i],char(i+'a')));
}
}
sort(v.begin(),v.end());
for(i=1;i<=v[v.size()-1].first;i++) 
{

	for(j=v.size()-1;j>=0;--j) if(v[j].first==i) O<<v[j].second<<" ";

}

O<<N;
}
return 0;
}
