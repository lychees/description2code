#include<bits/stdc++.h>
#define ff first
#define ss second
#define mp make_pair
#define pb push_back
using namespace std;
int main()
{
	int n;
	cin>>n;
	vector<pair<int,int> > v;
	for(int i=0;i<n;i++)
	{
		int a,b,c;
		cin>>a>>b>>c;
		v.pb(mp(a,a+c));
	}
	sort(v.begin(),v.end());
	int ans=v[0].ss-v[0].ff+1;
	int end=v[0].ss;
	for(int i=1;i<n;i++)
	{
		   if(end>=v[i].ss)
		   continue;
		   else if (v[i].ff>=end)
		   {
		   	   ans+=v[i].ss-v[i].ff+1;
		   	   if(end==v[i].ff)
		   	   ans--;
		   	   end=v[i].ss;
		   }
		   else
		   {
		   	    ans+=v[i].ss-end;
		   	    end=v[i].ss;
		   }
	}
	cout<<ans<<endl;
	return 0;
}
