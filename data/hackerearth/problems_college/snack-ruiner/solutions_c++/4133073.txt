#include<bits/stdc++.h>
using namespace std;
#define ld long double
#define eps 1e-9
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    int i;
    cin>>n;
    ld d[31],f[31];
    for(i=0;i<n;++i)
    	cin>>d[i];
    for(i=0;i<n;++i)
    	cin>>f[i];
    ld mex[31];
    ld mval = 0.0;
    ld sum = 0;
    int ind;
    for(i=0;i<n;++i)
    {
    	mex[i] = d[i]*f[i];
    	mval=max(mval,mex[i]);
    	if(mval - mex[i] < eps)
    	{
    		mval = mex[i];
    		ind = i; 
    	}
    	sum += mex[i];
    }
    ld ans = mval/sum;
    //cout<<ind<<" "<<ans<<endl;
    printf("%d %.3Lf\n",ind,ans);
    }
    return 0;
}
