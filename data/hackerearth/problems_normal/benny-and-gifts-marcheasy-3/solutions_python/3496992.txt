inp=raw_input()
x,y=0,0
icy={}
icy[(0,0)]=1
count=0
for i in inp:
	if i=='L':
		y=y-1
		if(icy.get((x,y),0)>0):
			count=count+1
		else:
			
			icy[(x,y)]=1


	if i=='R':
		y=y+1
		if(icy.get((x,y),0)>0):
			count=count+1
		else:
			
			icy[(x,y)]=1

	if i=='U':
		x=x-1
		if(icy.get((x,y),0)>0):
			count=count+1
		else:
			
			icy[(x,y)]=1

	if i=='D':
		x=x+1
		if(icy.get((x,y),0)>0):
			count=count+1
		else:
