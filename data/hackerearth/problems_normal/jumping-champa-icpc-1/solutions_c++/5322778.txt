#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,q,i,j;
    scanf("%d",&t);
    for(i=1;i<=t;i++)
    {
    	scanf("%d %d",&n,&q);
    	long int a[n],ans=0;
    	for(j=0;j<n;j++)
    	scanf("%ld",&a[j]);
    	sort(a,a+n);
    	for(j=0;j<n-1;j++)
    		ans=ans+q*(a[j+1]-a[j]);
    	printf("%ld\n",ans);
    }
    return 0;
}
