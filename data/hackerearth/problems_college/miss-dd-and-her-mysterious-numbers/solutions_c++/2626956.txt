#include<iostream>
#include<stdio.h>
int arr[1000001];
using namespace std;
void make(int n)
{
    int i,j;
    for(i=0;i<=n;i++)
    {
        arr[i]=0;
    }
    arr[1]=0;
    for(i=2;i<=n;i++)
    {
        if(arr[i]==0)
        {
            arr[i]++;
            if(i<=n/2)
            for(j=i*2;j<=n;j+=i)
            {
                arr[j]++;
            }
        }
    }
}
int main()
{
    int n,t,k,counter,counter_1,i,j;
    scanf("%d",&n);
    scanf("%d",&t);
    make(n);
    while(t--)
    {
        scanf("%d",&k);
        if(k>79000||k>n)
            printf("%d\n",0);
        else
        {
        counter_1=0;
         for(i=2;i<=n;i++)
         {
             if(arr[i]==k)
             {
                 counter_1++;
             }
         }
         printf("%d\n",counter_1);
        }
    }
}
