#include <iostream>
#include <cstdlib>
#include <cstdio>
#include<stdlib.h>
using namespace std;
int min(int x,int y)
{
	return x<y?x:y;
}
int main()
{
    int t;
    //freopen("input.txt","r",stdin);
    cin>>t;
    while(t--)
    {
    	int N,M,price;
    	cin>>N>>M;
    	int x1,x2,y1,y2;
    	cin>>x1>>y1;
    	cin>>x2>>y2;
    	cin>>price;
    	int amount=0;
    	int x=abs(x1-x2);
    	int y=abs(y1-y2);
    	amount=min(x,(N-x))+min(y,(M-y));
    	amount=amount*price;
    			
    				
    	if(amount<=1000)
    	cout<<amount<<"\n"<<"You saved the group."<<"\n";	
    	else
    	cout<<amount<<"\n"<<"Let go of the group."<<"\n";	
    	
    	
    }
    return 0;
}
