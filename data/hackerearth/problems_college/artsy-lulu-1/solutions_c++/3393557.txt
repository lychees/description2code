#include<iostream>

using namespace std;

int main()
{
	char c;
	int h;
	while(cin >> c >> h)
	{
		char temp = c + (h - 1);
		if(h < 2)
			cout << "INVALID" << endl;
		else if((temp >= 'a' && temp <= 'z') || (temp >= 'A' && temp <= 'Z'))
		{
			for(int i = 0 ; i < h ; i++)
			{
				for(int j = 0 ; j <= i ; j++)
				{
					cout << char(c + i);
					if(j != i)
						cout << " " ;
				}
				cout << endl;
			}
		}
		else
		{
			cout << "INVALID" << endl;
		}
		cout << endl;

	}
	return 0;
}
