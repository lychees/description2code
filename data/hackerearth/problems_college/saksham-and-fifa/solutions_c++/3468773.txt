#include <iostream>
using namespace std;
int a[10000005];
int main()
{
	int i,j,n,k;
	scanf("%d",&n);
	int ans = -1, pos = 0;
	for(i=0;i<n;i++)
	{
		scanf("%d %d",&a[i],&j);
		if(a[i]>ans)
		{
			ans = a[i];
			pos = j;
		}
	}
	if(pos>0)
		printf("Left Corner\n");
	else if(pos<0)
		printf("Right Corner\n");
	else
		printf("Either Corner\n");
	
    return 0;
}
