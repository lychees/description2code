#include <bits/stdc++.h>
using namespace std;
int a[100][4];
int fun(int index,int i,int n)
{
	if(index==n)
	return 0;
	int ans1=0;
	int ans2=0;
	int res;
	if(i==0)
	{
	ans1=a[index][1]+fun(index+1,1,n);
	ans2=a[index][2]+fun(index+1,2,n);
	res=min(ans1,ans2);
	}
	else if(i==1)
	{
	ans1=a[index][0]+fun(index+1,0,n);
	ans2=a[index][2]+fun(index+1,2,n);
	res=min(ans1,ans2);	
	}
	else 
	{
		ans1=a[index][1]+fun(index+1,1,n);
	ans2=a[index][0]+fun(index+1,0,n);
	res=min(ans1,ans2);
	}
	return res;
}
int main()
{
	int t,n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<3;j++)
			{
				scanf("%d",&a[i][j]);
			}
		}
		int res=INT_MAX;
		for(int i=0;i<3;i++)
		{
			int ans=a[0][i]+fun(1,i,n);
			res=min(res,ans);
		}
		cout<<res<<"\n";
	}
	return 0;
}
