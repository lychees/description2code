#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long n;
		scanf("%lld",&n);
		if(n%2!=0)
		printf("0\n");
		else
		{
			int i;long long co=0;
			for(i=1;i<=sqrt(n);i++)
			{
				if(n%i==0&&i*i!=n)
				{
					if(i%2==0)
					co++;
					int x=n/i;
					if(x%2==0)
					co++;
				}
				if(n%i==0&&i*i==n)
				{
					if(i%2==0)
					co++;
				}
			}
			printf("%lld\n",co);
		}
	}
	return 0;
}
