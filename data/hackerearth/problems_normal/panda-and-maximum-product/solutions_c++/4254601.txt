#include <iostream>
using namespace std;

using ll = long long;

int main()
{
	int n;
	cin >> n;
	ll mx0 = -1000000001;
	ll mx1 = -1000000001;
	ll mn0 = 1000000001;
	ll mn1 = 1000000001;
	while (n--) {
		ll x;
		cin >> x;
		if (x > mx0) { mx1 = mx0; mx0 = x; } else if (x > mx1) mx1 = x;
		if (x < mn0) { mn1 = mn0; mn0 = x; } else if (x < mn1) mn1 = x;
	}
    cout << (mx1 * mx0 > mn1 * mn0 ? mx1 * mx0 : mn1 * mn0) << endl;
    return 0;
}
