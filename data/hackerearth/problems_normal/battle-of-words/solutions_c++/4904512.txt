#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;

int main()
{
	int t;
	bool pos,neg;
	unsigned int i;
	string s1,s2;
	int arr[256];
	cin>>t;
	
	getline(cin,s1);
	while(t--)
	{
		pos = false;
		neg = false;
		getline(cin,s1);
		getline(cin,s2);
		memset(arr,0,sizeof(arr));
		
		for(i=0;i<s1.length();i++)
		{	
			if(s1[i]!=' ')
				arr[s1[i]]++;
		}
		
		for(i=0;i<s2.length();i++)
		{
			if(s2[i]!=' ')
				arr[s2[i]]--;
		}	
		
		for(i=0;i<256;i++)
		{
			if(arr[i]>0)
				pos = true;
			else if(arr[i] <0)
				neg = true;
		}
		
		//for(i=97;i<=122;i++)
			//cout<<arr[i]<<" ";
		if(pos&&neg)
			cout<<"You draw some.";
		else if(pos)
			cout<<"You win some.";
		else
			cout<<"You lose some.";
		cout<<endl;
		
	}
    return 0;
}
