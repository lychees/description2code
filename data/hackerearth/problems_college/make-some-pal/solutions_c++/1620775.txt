#include <iostream>
#include <string>


using namespace std;

int main()
{
	int t, a[50], ctr;
	string s;
	cin >> t;
	while(t--)
	{
		ctr = 0;
		char last;
		int l;
		cin >> s;
		l = s.length();
		last = s[l-1];
		for(int i=0; i<l; i++)
		{
			if(s[i] == last)
			{
				a[ctr] = i;
				ctr++; 
			}
		}
		int len = 1;
		for(int j=0; j<ctr; j++)
		{
			int length = (l) - a[j];
			/*if(length % 2 == 0)
				continue;*/
			int t = 1, flag = 0;
			for(int k=a[j]; k<=a[j]+length/2; k++)
			{
				if(s[k] == s[l-t])
				{
					t++;
					continue;
				}
				else
				{
					flag = 1;
					break;
				}
			}
			if(flag == 0)
				{
					len = length;
					break;
				}

		
		}
		cout << l - len + l << endl;
	}
	return 0;
}
