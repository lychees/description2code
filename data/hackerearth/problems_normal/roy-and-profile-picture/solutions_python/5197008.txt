def main():
	l = int(raw_input())
	n = int(raw_input())
	answers = []
	for i in range(n):
		w,h = raw_input().split()
		w,h = int(w), int(h)
		if w < l or h < l:
			answers.append("UPLOAD ANOTHER")
		elif w == h:
			answers.append("ACCEPTED")
		else:
			answers.append("CROP IT")
	for i in range(len(answers)):
		print answers[i]		

if __name__ == "__main__":
	main()
