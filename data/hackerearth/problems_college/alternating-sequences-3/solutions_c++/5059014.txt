#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n];
     int dp[n];
    for(int i=0;i<n;i++){
    	cin>>a[i];
    	dp[i]=1;
    }
    for(int i=1;i<n;i++){
    	for(int j=0;j<i;j++){
    		if(abs(a[j])<abs(a[i])&&a[j]<0&&a[i]>0)
    			dp[i]=max(dp[i],dp[j]+1);
    		if(abs(a[j])<abs(a[i])&&a[j]>0&&a[i]<0)
    			dp[i]=max(dp[i],dp[j]+1);
    		
    	}
    }
    int maxm=0;
    for(int i=0;i<n;i++)
    	maxm = max(dp[i],maxm);
   cout<<maxm<<endl;
    return 0;
}
