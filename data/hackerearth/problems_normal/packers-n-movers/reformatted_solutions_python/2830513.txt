m = int(input())
p = int(input())
a = map(int, raw_input().split())
part = [[-1 for _ in xrange(p+1)]for _ in xrange(m+1)]


def fun(part, mt, pt):
    if pt >= p:
        return 0
    if mt >= m:
        return 2**31
    if part[mt][pt] != -1:
        return part[mt][pt]
    temp, wt = 2**31, 0
    for i in xrange(pt, p):
        wt += a[i]
        temp = min(temp, max(fun(part, mt+1, i+1), wt))
    part[mt][pt] = temp
    return temp


print fun(part, 0, 0)
