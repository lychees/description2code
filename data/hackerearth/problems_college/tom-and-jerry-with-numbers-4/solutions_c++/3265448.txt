#include <iostream>
#include <cstring>
using namespace std;
int m1, m2;
//int *dp;
bool winjerry(int n , int *dp) {
	if (n < m1 )
		return false;
	if(dp[n] != -1)
		return dp[n];
	dp[n] = ((n >= m1 && !winjerry(n-m1, dp)) || (n >= m2 && !winjerry(n-m2, dp)));
	return dp[n];
}
int main()
{
  //  cout << "Hello World!" << endl;
    ios_base::sync_with_stdio(false);
    int t, n;
    cin >> t;
   // cout << t;
    while (t--) {
    	cin >> n >> m1 >> m2;
    	int dp[n+1];
    	memset(dp, -1, sizeof(dp));
    	if(winjerry(n,dp))
    		cout << "1\n";
    	else 
    		cout << "0\n";
    	//delete[] dp;
    }
    return 0;
}
