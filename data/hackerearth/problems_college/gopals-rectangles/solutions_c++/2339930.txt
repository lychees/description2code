#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	unsigned long long a,b;
    	scanf("%llu %llu",&a,&b);
    	unsigned long long s,m1,m2;
    	m1=(unsigned long long)(a*(a+1)/2);
    	m2=(unsigned long long)(b*(b+1)/2);
    	s=(unsigned long long)(m1*m2);
    	printf("%llu\n",(unsigned long long)s);
    }
    return 0;
}
