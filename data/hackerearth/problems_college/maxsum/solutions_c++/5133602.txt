#include<iostream>
using namespace std;
int t,n,ans;
int arr[10001];
int dp[10001];

int max(int a,int b){
	return (a>b)? a : b;
}

int main(){
	cin>>t;
	for(int i=0;i<t;i++){
		cin>>n;
		for(int j=1;j<=n;j++){
			cin>>arr[j];
		}
		
		ans = 0;
		dp[0] = 0;
		dp[1] = arr[1];
		dp[2] = arr[2];
		
		for(int j=3;j<=n;j++){
			dp[j] = arr[j]+max(dp[j-2],dp[j-3]);
		}
		cout<<max(dp[n-1],dp[n])<<endl;
	}
	return 0;
}
