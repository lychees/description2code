#include <iostream>
#define MOD 1000000007
#define ll long long
using namespace std;
ll arr[902][902]={0};
void ncr(){
	ll i,j;
	arr[0][0]=1;
	for(i=1;i<902;i++){
		for(j=0;j<=i;j++){
			if (i==j || j==0)
				arr[i][j]=1;
			else
				arr[i][j]=(arr[i-1][j]+arr[i-1][j-1])%MOD;
		}
	}
}
int main(){
	ncr();
	ll i,j,n,m,ans;
	int t;
	cin>>t;
	while(t--){
		ans=0;
		cin>>n>>m;
		if (m&1){
			printf("0\n");
			continue;
		}
		ll p=18*n,q=6*n;
		for(ll i=0;i<=m/2;i++){
			ll x=(arr[p][i]*arr[q][m/2-i])%MOD;
			ll y=(arr[q][i]*arr[p][m/2-i])%MOD;
			x=(x*y)%MOD;
			ans+=x;
			ans%=MOD;
		}
		cout<<ans<<endl;
	}	
}
