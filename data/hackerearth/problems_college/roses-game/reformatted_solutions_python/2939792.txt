def outCome(values):
    alreadyVistedPlaces = {}
    currentLoc = '0'
    while (alreadyVistedPlaces.has_key(currentLoc)
           == False and currentLoc != '-1'):
        alreadyVistedPlaces[currentLoc] = 1
        currentLoc = values[int(currentLoc)]

    if (currentLoc == '-1'):
        return "Win"
    else:
        return "Lose"
