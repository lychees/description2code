#include <bits/stdc++.h>

using namespace std;

int prime[2222222]={};
vector <int> primes;

int ispal(int n){
    char str[1000];
    int i=0,rem,j;
    while(n){
        rem = n%10;
        str[i++]=(char)(rem + 48);
        n/=10;
    }
    for(j=0;j<=i/2;j++){
        if(str[j]!=str[i-j-1]){
            return 0;
        }
    }
    return 1;
}

void seive(){
    int i,j;
    for(i=2;i<2222222;i++){
        if(!prime[i]){
            primes.push_back(i);
            for(j=2*i;j<2222222;j+=i){
                prime[j]=1;
            }
        }
    }
}

int main(){
    seive();
    int n;
    scanf("%d",&n);
    while(n--){
        int x,i;
        scanf("%d",&x);
        for(i=0;i<primes.size();i++){
            if(primes[i]>=x && ispal(primes[i])){
                printf("%d\n",primes[i]);
                break;
            }
        }
    }
    return 0;
}
