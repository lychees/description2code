#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int a[5];
		for(int i=0;i<3;i++)
		scanf("%lld",&a[i]);
		sort(a,a+3);
		long long int ans=a[0];
		if(ans>=MOD)ans=ans%MOD;
		ans=ans*(a[1]-1);
		if(ans>=MOD)ans=ans%MOD;
		ans=ans*(a[2]-2);
		if(ans>=MOD)ans=ans%MOD;
		printf("%lld\n",ans);
	}
    return 0;
}
