#include <iostream>
#include <list>
using namespace std;
long printLst(list<long> a) {
        list<long>::iterator it;
        long count = 0,num=1;
        for(it=a.begin();it!=a.end();) {
                long num2=num*num;
                if(*it>num2) {
                        count += num2;
                        num++;
                } else {
                        long temp = num2 - *it;
                        count += temp;
                        it++;
                        num++;
                }
        }
        return count;
}

int main() {
        long N;
        list<long> A;

        cin >> N;
        for(long i=0;i<N;i++) {
                long temp;
                cin >> temp;
                A.push_back(temp);
        }
        A.sort();
        cout << printLst(A) << endl;
        return 0;
}
