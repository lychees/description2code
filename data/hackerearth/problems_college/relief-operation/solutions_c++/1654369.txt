#include <bits/stdc++.h>
using namespace std;
#define long long long
int rint() {int ret;scanf("%d", &ret);return ret;}
int rlong() {long ret;scanf("%lld", &ret);return ret;}

int a[2001], b[2001], dp[2001];

int main()
{
#if _DEBUG
    freopen("input.txt", "r", stdin);
#endif

	for (int tt = rint(); tt > 0; tt--)
	{
		int n = rint();
		int m = rint();
		for (int i = 0; i < n; i++)
			a[i] = rint();
		for (int i = 0; i < n; i++)
			b[i] = rint();
		memset(dp, 0, sizeof(dp));
		for (int i = 0; i < n; i++)
			for (int j = m - a[i]; j >= 0; j--)
				dp[j + a[i]] = max(dp[j + a[i]], dp[j] + b[i]);
		printf("%d\n", dp[m]);
	}

    return 0;
}
