#include <iostream>
using namespace std;
int pow(int , int);
int a[1024][1024];
void fun(int n);
int main()
{
	int n;
	int i,j,k;
	int prod = 1;
	cin>>n;
	prod = pow(2,n);
	a[0][0] = 1;
	a[1][0] = 3;
	a[0][1] = 2;
	a[1][1] = 4;
	for(i = 2;i <= n;i++)
	{
		fun(i);
	}

	for(i = 0;i < prod;i++)
	{
		for(j = 0;j < prod;j++)
		{
			cout<<a[i][j]<<" ";
		}
		cout<<endl;
	}
	return 0;

}
void fun(int n)
{
	int x = n-1;
	int resultindex = pow(2,x);
//	cout<<"resultindex = "<<resultindex<<endl;
	int y = pow(2,n);
	y = (y*y)/4;
	for(int i = 0;i < resultindex;i++)
	{
		for(int j = 0;j < resultindex;j++)
		{
	//		cout<<"i = "<<i<<" j = "<<j<<"resultindex ="<<resultindex<<endl;
			a[i][j+resultindex]  = a[i][j] + y;
	//		cout<<"	a[i][j+resultindex] = "<<	a[i][j+resultindex]<<endl;
			a[i+resultindex][j] = a[i][j] + 2*y;
	//		cout<<"a[i+resultindex][j] = "<<a[i+resultindex][j]<<endl;
			a[i+resultindex][j+resultindex] = a[i][j] + 3*y;
	//		cout<<"a[i+resultindex][j+resultindex] = "<<a[i+resultindex][j+resultindex]<<endl;
		}
	}
}
int pow(int n,int pow)
{
	int result = 1;
	for(int i = 0;i < pow;i++)
	{
		result = result*n;
	}
//	cout<<"n = "<<n<<" pow = "<<pow<<" result = "<<result<<endl;
	return result;
}
