#include<bits/stdc++.h>
using namespace std;

int main()
{
    
    long long a[1000];
    a[1]=1;
    a[2]=2;
    for(int i=3;i<1000;i++)a[i] = a[i-1]+a[i-2];
    int t;
    cin>>t;
    while(t--)
    {
        
        int n;
        cin>>n;
        cout<<a[n]<<endl;
    }
    return 0;
}
