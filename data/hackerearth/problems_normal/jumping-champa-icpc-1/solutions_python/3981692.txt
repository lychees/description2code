t = int(raw_input())
#print t
for _ in range(t):
	inP = raw_input()
	inA = raw_input()
	
	params = [int(j) for j in inP.split()]
	
	arr = [int(i) for i in inA.split()]
	
	arr.sort()
	
	res = 0
	for x in range(params[0]-1):
		res += params[1] * abs(arr[x] - arr[x+1])
	print res
