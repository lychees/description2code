#include<bits/stdc++.h>
using namespace std;
bool prime[1000001];

void sieve()
{
	
   memset(prime,true,1000001);
   prime[0]=false;
   prime[1]=false;
   for(long long i=2;i<1000001;i++)
   {
   	   if(prime[i]==true)
   	   {
   	   	   for(long long j=i+i;j<1000001;j=j+i)
   	   	   {
   	   	   	prime[j]=false;
   	   	   	
   	   	   }
   	   	
   	   	
   	   }
   	
   	
   	
   }
	
	
	
}

int main()
{
	int t;
	cin>>t;
	sieve();
	while(t--)
	{

	long long n,a=0,b=100000;
	cin>>n;
	
	for(long long i=2;i<n;i++)
	{
		if(prime[i]==true && prime[n-i]==true)
		{
			if(abs(i-(n-i))<(abs(a-b)) && i!=(n-i) && (i*(n-i))!=n && n*i!=(n-i) && ((n-i)*n)!=i){
			a=i;
			b=n-i;
			}
			
		}
		
		
	}
	if(a==0 && b==100000)
	cout<<"-1"<<" "<<"-1"<<endl;
	else
	cout<<a<<" "<<b<<endl;	
		
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
