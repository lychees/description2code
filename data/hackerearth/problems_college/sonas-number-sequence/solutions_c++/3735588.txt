#include<iostream>
#include<cstdlib>

using namespace std;

int main(){
	int n;
	cin>>n;
	int arra[n],arrb[n];
	for(int i=0;i<n;i++){
		cin>>arrb[i];
	}
	for(int i=0;i<n;i++){
		int sum = 0;
		for(int j=0;j<i;j++){
			sum = sum+arra[j];
		}
		arra[i] = (i+1)*arrb[i]-sum;
	}
	for(int i=0;i<n;i++){
		cout<<arra[i]<<" ";
	}
return 0;
}
