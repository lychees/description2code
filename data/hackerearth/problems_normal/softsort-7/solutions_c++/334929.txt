#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

long long int p[1000001];

void fact()
{
	p[0]=1;
	for(long long int i=1;i<=1000000;i++)
	{
		if(p[i-1]*i>1000000007)
			p[i]=(p[i-1]*i)%1000000007;
		else
			p[i]=p[i-1]*i;
	}
}

int main()
{
    long int t,n;
    long long int k,ans;
    fact();
    scanf("%ld",&t);
    while(t--)
    {
    	scanf("%ld",&n);
    	k = p[n];
    	ans = (3*(k + 1))%1000000007;
   		printf("%lld\n",ans);
    	
    }
    return 0;
}
