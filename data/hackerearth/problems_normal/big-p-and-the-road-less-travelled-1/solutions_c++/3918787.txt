using namespace std;
int n;
vector<int>v[100005];
bool visited[100005]={0};
int dp[100005]={0};
int func(int node)
{
    if(node==n)
        return 1;
     if(visited[node]==true)
        return dp[node];
    visited[node]=true;
    int ans=0;
    for(int i=0;i<v[node].size();i++)
    {
        ans+=func(v[node][i]);
    }
    return dp[node]=ans;
}
int main()
{

        if(a==0 && b==0)
            break;
        v[a].pb(b);
    }
    cout<<func(1)<<endl;


}
