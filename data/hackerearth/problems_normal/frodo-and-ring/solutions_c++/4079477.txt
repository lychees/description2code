#include<iostream>
#include<cstdio>
using namespace std;
inline void scanchar(char *a)
{
    register char c=0;
    while (c<33) 
	c=getchar();
    *a=c;
}
int main()
{
	int t,h,n,m,i,j;
	char use;
	cin>>t;
	for(h=0;h<t;h++)
	{
	  cin>>n>>m;
	  int A[n][m];
	  int dp[n][m];
	  for(i=0;i<n;i++)
	   for(j=0;j<m;j++)
	   {
	   scanchar(&use);
	   if(use=='T')
	   A[i][j]=150;
	   else 
	   A[i][j]=-100;
       }
      A[n-1][m-1]=0;
      if(A[0][0]>=0)
      dp[0][0]=A[0][0];
      else
      {
      cout<<-1<<endl;
      continue;
      }
	  for(i=1;i<n;i++)
	   {
	   if(dp[i-1][0]!=-1)	
	   {
	   dp[i][0]=A[i][0]+dp[i-1][0];
	   if(dp[i][0]<0)
	   dp[i][0]=-1;
       }
	   else
	   dp[i][0]=-1;
	   }
	   for(j=1;j<m;j++)
	   {
	   if(dp[0][j-1]!=-1)	
	   {
	   dp[0][j]=A[0][j]+dp[0][j-1];
	   if(dp[0][j]<0)
	   dp[0][j]=-1;
       }
	   else
	   dp[0][j]=-1;
	   }
	  for(i=1;i<n;i++)
	    for(j=1;j<m;j++)
		{
		  if(dp[i-1][j]!=-1 || dp[i][j-1]!=-1)	
		  {
		  dp[i][j]=max(A[i][j]+dp[i-1][j],A[i][j]+dp[i][j-1]);
		  if(dp[i][j]<0)
		  dp[i][j]=-1;
	      }
		  else
		  dp[i][j]=-1;
		}
	   cout<<dp[n-1][m-1]<<endl;
    }
	return 0;
}
