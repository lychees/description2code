#include <bits/stdc++.h>

using namespace std;

#define s(x) scanf("%d", &x)

int main()
{
    int t, n, l, b, r, ld, rd;
    cin >> t;
    while (t--) {
        cin >> n >> b;
        l = n/b;
        r = l+1;
        ld = n-l*b;
        rd = r*b-n;
        if (rd > ld)
            cout << l*b << endl;
        else
            cout << r*b << endl;
    }
    return 0;
}
