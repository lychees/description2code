#include <iostream>
#include <map>
#include <unordered_map>

using namespace std;

int mod = 1000000007 ;

unordered_map<int , int > fact ;

void compute (int max)
{
	long long int f = 1 ;
	
	for(int i = 1 ; i <= max ; i++)
	{
		f = ((f%mod)  *  (i%mod))%mod ; 
		fact[i] = f ;
	}
	
}

int main()
{
    int n , m ; 
    
    cin >> n >> m ; 
    
    map<int,int> mp ; 
    
    
    int max = -1 ; 
    
    for(int i = 1 ; i <= n ; i++)
    {
    	for(int j = 1 ; j <= m ; j++)
    	{
    		mp[i^j] ++ ;
    		
    		if(mp[i^j] > max)
    			max = mp[i^j] ;
    	}
    }
    
    long long int sum = 1 ;
    
    compute(max) ;
    
    for(map<int,int> :: iterator it = mp.begin() ; it != mp.end() ; it++)
    {
    	
    	
    	sum = ( ( sum % mod ) * (fact[it->second]%mod) )%mod ;
    	
    }
    
    cout << sum ;
    
    return 0;
}
