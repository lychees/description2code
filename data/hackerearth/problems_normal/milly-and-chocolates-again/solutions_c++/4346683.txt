#include <iostream>
using namespace std;

int binary(int a[],int l,int r,int n){
	if(l==r)
		return l;
	else{
		int m=(l+r)>>1;
		if(a[m]>n)
			r=m;
		else
			l=m+1;
		return binary(a,l,r,n);
	}
}

int main()
{
    int x,n,q,index;
    cin>>n;
    int a[n+1];
    a[0]=1;
    for(int i=1;i<=n;i++)
    {
    	cin>>x;
    	a[i]=a[i-1]+x;
    }
    cin>>q;
    while(q--){
    	cin>>index;
    	int i=binary(a,0,n,index);
    	cout<<i<<endl;
    }
    return 0;
}
