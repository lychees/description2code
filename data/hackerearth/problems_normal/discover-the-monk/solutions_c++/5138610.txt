#include <iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main()
{
    int n,q,i,j;
    cin>>n>>q;
    
    int a[n],b[q];
    
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    
    vector<int> v(a,a+n);
    sort (v.begin(),v.end());
    
    for(i=0;i<q;i++)
    {
    	cin>>b[i];
    }
    
    for(i=0;i<q;i++)
    {
    	if(binary_search(v.begin(), v.end(), b[i]))
        	cout << "YES" << endl;
    	else
        	cout << "NO" << endl;
    }
    return 0;
}
