import sys
import os
t = int(sys.stdin.readline().strip('\n'))

while t > 0:

    values = sys.stdin.readline().strip('\n').split()
    G = int(values[0])
    N = int(values[1])
    i = 1
    if N != 1:
        while i < N:
            print(G*2),
            i = i+1
        print(G*3)
    else:
        print(G*2)
    t -= 1
