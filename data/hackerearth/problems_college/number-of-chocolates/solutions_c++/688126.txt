#include <cstdio>
using namespace std;
long long K[1000000];
int main()
{
    int N,Q,G1,G2,P1,P2;
    long long num;
    scanf("%d%d",&N,&Q);
    scanf("%lld",K);
    for(int i=1;i<N;i++){
    	scanf("%lld",&num);
    	K[i] = K[i-1] + num;
    }
    	while(Q--){
    		num = 0;
    		scanf("%d%d",&G1,&P1);
    		scanf("%d%d",&G2,&P2);
    		num = K[G2] - K[G1-1];
    		num += K[P2] - K[P1-1];
    		
    		if(G1<=P1 && G2>=P2){
    		num -= K[P2] - K[P1-1];
    		}
    		else if(P1<=G1 && P2>=G2)
    		num -= K[G2] - K[G1-1];
    		
    		
    		
    		else if(P1<=G2 && G1<=P1 && G2<=P2){
    			num -= K[G2] - K[P1-1];
    		}else if(G1<=P2 && P1<=G1 && P2<=G2)
    		{
    			num -= K[P2] - K[G1-1];
    		}
    		printf("%lld\n",num);
    	}
    return 0;
}
