M = int(raw_input())
P = int(raw_input())
w = map(int, raw_input().strip().split())
w = [0]+w
INF = 1e8
cw = [0]*(P+1)
for i in xrange(1,P+1):
	cw[i] = cw[i-1]+w[i]
dp = [[INF for i in xrange(P+1)] for j in xrange(M+1)]
dp[1] = cw
for i in xrange(2,M+1):
	for j in xrange(1,P+1):
		for k in xrange(i-1,j):
			dp[i][j] = min(dp[i][j],max(cw[j]-cw[k],dp[i-1][k]))
ans = INF
for i in xrange(1,M+1):
	ans = min(ans,dp[i][P])
print ans
