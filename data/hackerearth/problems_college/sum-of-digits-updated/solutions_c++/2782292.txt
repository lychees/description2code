#include <iostream>
using namespace std;

int main()
{
    int t,len,i;
    long long sum=0;
    cin>>t;
    string str;
    while(t>0){
    	cin>>str;
    	len=str.length();
    	sum=0;
    	for(i=0;i<len;i++){
    		sum=sum+(str[i]-'0');
    	}
    	cout<<sum<<endl;
    	t--;
    }
    return 0;
}
