#include <cstdio>

int letterCount[10][26] = {{0}}, letters[10], n, k, taken[10] = {0};

long long perm(int ind)
{
	if (ind == k)
	{
		return 1;
	}
	long long ans = 0;
	for (int i = 0; i < n; i++)
	{
		if (taken[i] == 0 && letterCount[i][letters[ind]] > 0)
		{
			taken[i] = 1;
			ans += letterCount[i][letters[ind]] * perm(ind+1);
			ans %= 1000000007;
			taken[i] = 0;
		}
	}
	return ans;
}

int main()
{
	scanf("%d %d", &n, &k);
	char c, str[100];
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < 6; j++)
		{
			scanf(" %c", &c);
			letterCount[i][c-'a']++;
		}
	}
	scanf("%s", str);
	for (int i = 0; i < k; i++)
	{
		letters[i] = str[i] - 'a';
	}
	printf("%lld\n", perm(0));
}
