#include<bits/stdc++.h>
using namespace std;
long long int solve(long long int n)
{
    long long int ans=0,i;
    for(i=0;i<n;i++)
    ans=ans+(i*(n-i));
    return 2*ans;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long int n;
        cin>>n;
        cout<<solve(n)<<endl;
    }
    return 0;
}
