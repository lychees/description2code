#include <iostream>
#include<stack>
using namespace std;

int main()
{
    // cout << "Hello World!" << endl;
    int t;
    cin >> t;
    while(t--) {
    	int n, k, num;
    	cin >> n >> k;
    	stack<int> friends;
    	for(int i = 0; i < n; i++) {
			cin >> num;
			while(k > 0 && !friends.empty() && (num > friends.top())) {
				friends.pop();
				k--;
			}
			friends.push(num);
    	}
    	while(k--) {
    		friends.pop();
    	}
    	
    	stack<int> temp;
    	while(!friends.empty()) {
    		temp.push(friends.top());
    		friends.pop();
    	}
    	
    	while(!temp.empty()) {
			cout << temp.top() << " ";
			temp.pop();
    	}
    	cout << endl;
    }
    return 0;
}
