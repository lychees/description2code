#include <iostream>
#include <algorithm>
#include <math.h>
using namespace std;
int h[3001][3000]={};
int main()
{
    int n;
   cin>>n;
   
   for(int i=0;i<(pow(2,n)-1);i++)
   {
   		for(int j=0;j<=i;j++)
   			cin>>h[i][j];
   }
   vector <int> p;
   for(int i=1;i<=pow(2,n);i++)
   	p.push_back(i);
   while(p.size()!=1)
   {
   		vector <int> p1;
   		for(int j=1;j<p.size();j+=2)
   			if(h[p[j]-2][p[j-1]-1]==0)
   				p1.push_back(p[j-1]);
   			else
   				p1.push_back(p[j]);
   		p=p1;
   }
   cout<<p[0]<<endl;
    return 0;
}
