#include <iostream>
#include <bits/stdc++.h>
using namespace std;
typedef unsigned long long int thisInt;
typedef unsigned int Tint;
void ScrapVol(thisInt N){
	if(N==1){
		cout << N*N*N << endl;
	}
	else if(N >= 2){
		cout << N*N*N-(N-2)*(N-2)*(N-2) << endl;
	}

}
int main(){
	Tint T;
	cin >> T;
	thisInt N;
	while(T--){
	    cin >> N ;
        ScrapVol(N);
	}


	return 0;
}
