#include<iostream>
using namespace std;
string s;
int i, l, w;

int main()
{
  cin >> s;
  l = 1;
  w = 0;
  for(i = 1; i < s.size(); i++)
  {
    if (s[i] != s[i - 1])
    {
      if(l % 2 == 0)
      {
	w++;
      }
      l = 1;
    }
    else
    {
      l++;
    }
  }
  
  if (l % 2 == 0)
  {
    w++;
  }
  cout << w << endl;
  
return 0;
}
