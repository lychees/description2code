#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int tc;
	cin >> tc;
	while(tc--)
	{
		long int n,m;
		cin >> n >> m;
		long int arr[n],brr[m];
		for(long int i=0;i<n;i++)
			cin >> arr[i];
		for(long int j=0;j<m;j++)
			cin >> brr[j];
		long int arrC=0,brrC=0;
		while(arrC<n && brrC<m)
		{
			if(arr[arrC] > brr[brrC])
			{
				cout << arr[arrC] << " ";
				arrC++;
			}
			else
			{
				cout << brr[brrC] << " ";
				brrC++;
			}
		}
		while(arrC < n)
		{
			cout << arr[arrC] << " ";
			arrC++;
 		}
 		while(brrC < m)
 		{
 			cout << brr[brrC] << " ";
			 brrC++;	
		}
		cout << endl; 
	}
	return 0;
}
