#include<bits/stdc++.h>
using namespace std;
#define max 10000005

int phi[max];
void sieve()
{
	int i,j;
	
	for (i = 1; i < max; ++i)
		phi[i] = i;
		
	for(i=2;i<max;i++)
	{
		if(phi[i] == i)
		{
			phi[i]=(i-1);
			for(j=i*2;j<max;j+=i)
			{
				phi[j]/=i;
				phi[j]*=(i-1);
			}
		}
	}
}

int main()
{
	sieve();
	int t,n;
	cin>>t;
	while(t--)
	{
		scanf("%d", &n);
		printf("%d\n", phi[n]);
	}
}
