#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int main()
{
	ll t,n,a,b,i,j,ans,temp,k;
	scanf("%lld%lld",&k,&t);
	ll arr[k+1];
	for(i=0;i<k;i++)
	{
		scanf("%lld",&arr[i]);
	}
	while(t--)
	{
		scanf("%lld%lld",&a,&b);
		ans=0;
		for(i=1;i<(1<<k);i++)
		{
			ll cnt=0;
			ll ac=(a-1),bc=b;
			for(j=0;j<k;j++)
			{
				if((i>>j)&1)
				{
					cnt++;
					ac=(ac/arr[j]);
					bc=(bc/arr[j]);
				}
			}
			if(cnt%2==1)
			{
				ans=(ans+(bc-ac));
			}
			else
			{
				ans=(ans-(bc-ac));
			}
		}
		printf("%lld\n",ans);
	}
}
