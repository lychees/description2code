#include <iostream>
#include <cstring>
using namespace std;
 
bool isPalin(string str)
{
	int len = str.length();
	len--;
	int start=0;
 
	while(start<=len){
		if(str[start++]!=str[len--])
			return false;
	}
	return true;
}
 
int main() {
	int t;
	string a,b,temp;
	cin>>t;
 
	while(t--){
		cin>>a>>b;
		int len = a.length();
		temp = a;
 
		int count=0;
 
		for(int i=0; i<=len; i++){
			a.insert(i,b);
			if(isPalin(a)) count++;
			a = temp;
		}
 
		cout<<count<<endl;
	}
	return 0;
}
