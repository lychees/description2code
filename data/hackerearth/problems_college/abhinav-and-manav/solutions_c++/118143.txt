#include <bits/stdc++.h>
using namespace std;
#define lli long long int
lli f(lli n)
{
	int c=0;
	while(n!=0)
	{
		c++;
		n/=10;
	}
	return c;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		lli l,r;
		cin>>l>>r;
		int n1,n2;
		n1=f(l);n2=f(r);
		lli ans=0;
		for(int i=1;i<=10;i++)
		{
			if(i<n1 || i>n2) continue;
			lli fath=0,sm=0;
			for(int j=1;j<=i;j++)
				fath=fath*10+9;
			if(l<=(fath/2) && r>=(fath/2))
			{
				if(((fath+1)/2)>=l && ((fath+1)/2)<=r)
				{
					sm=fath*(fath/2)-(fath/2)*(fath/2);
					sm=max(fath,fath*((fath+1)/2)-((fath+1)/2)*((fath+1)/2));
				}
				else sm=fath*(fath/2)-(fath/2)*(fath/2);
			}
			else if(l>(fath/2))
			{
				sm=fath*l-(l*l);
			}
			else if(r<(fath+1)/2)
			{
				sm=fath*r-(r*r);
			}
			ans=max(ans,sm);
		}
		cout<<ans<<"\n";
	}
	return 0;
}
