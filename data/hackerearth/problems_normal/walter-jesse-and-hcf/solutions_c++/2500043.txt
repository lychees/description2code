

    #include <bits/stdc++.h>
    #define lli long long
    using namespace std;
    bool func(lli p1, lli p2)
    {
    return p1 > p2;
    }
    int main()
    {
    int t;
    lli x,y,k;
    cin >> t;
    assert(t>=1&&t<=1000);
    while ( t-- ) {
    cin >> x >> y >> k;
    assert(x>=1&&x<=10000000000LL);
    assert(y>=1&&y<=10000000000LL);
    assert(k>=1&&k<=10000000000LL);
    vector <lli> v;
    lli val = __gcd(x,y);
    lli tt = (lli)sqrt(val);
    for ( lli i = 1; i <= tt; i++ ) {
    if ( val%i == 0 ) {
    v.push_back(i);
    if ( i != val/i ) v.push_back(val/i);
    }
    }
    sort(v.begin(),v.end(),func);
    if ( v.size() < k ) cout << "No crymeth today" << endl;
    else cout << v[k-1] << endl;
    }
    return 0;
    }
