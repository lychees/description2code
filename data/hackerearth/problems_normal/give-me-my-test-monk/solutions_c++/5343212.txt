#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
long long power(long long a,long long b)
{
	if(b==0)
		return 1;
	else if(b%2==0)
	{
		long long s=power(a,b/2)%mod;
		return (s*s)%mod;
	}
	else
		return (a*power(a,b-1))%mod;
}
int  main()
{
	long long f[101];
	long long dearrange[101];
	long long ncr[101][101];
	 f[0]=1;
	for(long long i=1;i<=100;i++)
 	f[i]=((f[i-1]%mod)*(i%mod))%mod;
 	dearrange[0]=1;
 	dearrange[1]=0;
 	for(long long i=2;i<=100;i++)
 	{
 		for(long long j=2;j<=i;j++)
 		{

				long long val=power(f[j],(mod-2))%mod;
 			if(j%2==0)
 				dearrange[i]=(dearrange[i]+val)%mod;
 			else
 				dearrange[i]=(dearrange[i]-val+mod)%mod;
 		}
 		dearrange[i]=(dearrange[i]*f[i])%mod;
 	}
 	ncr[0][0]=1;
 	for(long long i=0;i<=100;i++)
 	{
		for(long long j=0;j<=100;j++)
 		{
 			if(j==0)
 				ncr[i][j]=1;
 			else if(i==0)
 				ncr[i][j]=0;
 			else
 			ncr[i][j]=(ncr[i-1][j]+ncr[i-1][j-1])%mod;
 		}
 	}
 	long long n,l,r;
 	cin>>n>>l>>r;
 	long long ans=0;
 	for(long long i=l;i<=r;i++)
 	{
 		ans=(ans+(((ncr[n][i]%mod)*(dearrange[i]%mod))%mod))%mod;
 	}
 	long long finalans=(ans*(power(f[n],mod-2)%mod))%mod;
 	cout<<finalans<<endl;
}
