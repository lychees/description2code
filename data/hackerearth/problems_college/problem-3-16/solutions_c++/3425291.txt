#include <bits/stdc++.h>
using namespace std;

int main()
{
int t;
cin>>t;
int n,k;
while(t--){
	cin>>n;
	cin>>k;
	int b[11];
	int temp;
	memset(b,0,sizeof(b));
	for(int i = 0; i < n; i++){
		cin>>temp;
		b[temp]++;
	}
	int ans = 0;
	for(int i = 1; i <= k; i++){
		ans = max(ans,b[i]);
	}
	ans = n-ans;
	cout<<ans<<endl;
}
}
