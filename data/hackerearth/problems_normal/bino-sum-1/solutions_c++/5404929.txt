#include <iostream>
using namespace std;
#define MOD 1000000007
long long ar[1006][1006];
inline void pre(){
	ar[0][0] = 1;
	for (int i = 1; i <= 1000; ++i)
	{
		ar[i][0] = 1;
		ar[i][i] = 1;
		for (int j = 1; j<i; ++j)
		{
			ar[i][j] = (ar[i - 1][j] + ar[i - 1][j - 1]) % MOD;
		}
	}
}
int main() {
	ios_base::sync_with_stdio(0);
	pre();
	int t;
	cin >> t;
	while (t--)
	{
		long long n, m, sum = 0;
		cin >> n >> m;
		for (int i = 0; i <= m; ++i)
			sum = (sum + ar[n][i]) % MOD;

		cout << sum << "\n";
	}
	return 0;
}
