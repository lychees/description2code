#include <bits/stdc++.h>
using namespace std;

int a[25],b[25],c[25],p,q,r,n;
bool flag;
void dfs(int i,int p,int q,int r)
{
//	if(flag)
//		return;
    if (p==0 && q==0 && r==0) 
    	flag=true;
    if (i==n+1)
    	return;
    dfs(i+1,p,q,r);
    dfs(i+1,p-a[i],q-b[i],r-c[i]);
}
int main()
{
	int t,temp=1;
	cin>>t;
    while(t--)
    {
        cin>>p>>q>>r;
        cin>>n;
        for (int i=1; i<=n; i++)
            cin>>a[i]>>b[i]>>c[i];
        flag=false;
        dfs(1,p,q,r);
        cout<<"Case #"<<temp++<<": ";
        if (flag)
        	cout<<"yes"<<endl;
        else 
	        cout<<"no"<<endl;
    }
    return 0;
}
