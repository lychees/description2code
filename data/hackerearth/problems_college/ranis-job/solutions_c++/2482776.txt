#include <iostream>
using namespace std;

void quickSort(long long arr[],long long start,long long end){
	if(start<=end){
        long long i=start,j=end;
        long long mid = (i+j)/2;
        long long pivot = arr[mid];

        while(i<=j){
            while(arr[i]>pivot)
                i++;
            while(arr[j]<pivot)
                j--;
            if(i<=j){
                swap(arr[i],arr[j]);
                i++;
                j--;
            }
        }
        quickSort(arr,start,j);
        quickSort(arr,i,end);
	}
}
