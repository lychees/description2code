#include <iostream>
using namespace std;

int main()
{
    long long int n,m,arr[100000]={0},sum=0,a,b,k,avg=0;
    cin>>n>>m;
    for(int i=0;i<m;i++)
    {
    	cin>>a>>b>>k;
    	for(int j=a;j<=b;j++)
    	arr[j-1]+=k;
    	
    	
    }
    for(int i=0;i<n;i++)
    {
    	sum=sum+arr[i];
    }
    
    avg=sum/n;
    cout<<avg<<endl;
    return 0;
}
