#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
    ll t,b,w,x,y,z,amt,a1,a2;
    cin>>t;
    while(t--)
    {
        cin>>b>>w;
        cin>>x>>y>>z;
        amt=0;
        if(z>=x&&z>=y)
            amt=x*b+y*w;
        else
        {
            if(x<y)
            {
                a1=y*w;
                amt+=b*x;
                a2=(x+z)*w;
                if(a1<a2)
                    amt+=a1;
                else
                    amt+=a2;
            }
            else
            {
                amt+=w*y;
                a2=(y+z)*b;
                a1=x*b;
                if(a1<a2)
                    amt+=a1;
                else
                    amt+=a2;
            }
        }
        cout<<amt<<'\n';
    }
    return 0;
}
