int a,b;ll c;
int dp[2][1005];
int solve(int s,int k){
    if(s*b>=c){return k;}
    int &ret=dp[k][s];
    if(ret+1)return ret;
    ret=0;
    for(int i=s*a;i<=s*b;i++){
        if(solve(i,1-k)==k)return ret=k;
        //j=s/i;
        //i=s/j;
    }
    return ret=1-k;
}
