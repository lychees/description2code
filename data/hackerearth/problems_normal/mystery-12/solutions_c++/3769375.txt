#define MOD 1000000007
#define ll long long int
#define MAXINT 1000000000001ll
#define SET(x,y) memset(x,y,sizeof(x));

using namespace std;
/*
7*x = a+y*d;
A+x*D = 0 %MOD
x*D = MOD - A%MOD
*/

ll powers(ll a,ll b,ll m) {
	ll ans=1;
	while(b) {
		if(b&1)
			ans=(ans*a)%m;
		a=(a*a)%m;
		b>>=1;
	}
	return ans;
}

void solve(void) {
	ll a,d,p,ans;
	cin>>a>>d>>p;
	ans=((p - a%p )%p * powers(d%p,p-2,p))%p;
	if(ans == 0) {
		if(a%p ==0 && a!=0)
			cout<<ans<<endl;
		else
			cout<<-1<<endl;
		return;
	}
	cout<<ans<<endl;
}

int main() {
	int i,t;
	scanf("%d",&t);
	for(i=1;i<=t;i++) {
		solve();
	}
	return 0;
}
