#include<iostream>
using namespace std;
int main()
{
   long  int n;
    cin>>n;
    long int j=0;
    while(j<n)
    {
        long int a,b,m;
        cin>>a>>b>>m;
       if(a>b)
       {
           long int temp=a;
           a=b;
           b=temp;
       }
        cout<<(b/m)-(a-1)/m<<endl;
        j++;
    }
}
