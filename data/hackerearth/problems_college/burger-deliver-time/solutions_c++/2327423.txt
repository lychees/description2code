#include <iostream>
#include <algorithm>
using namespace std;

class Order
{
	public:
	int Sum, i;
	
	bool operator <(Order a)
	{
    	return Sum < a.Sum;
	}
};
struct comp
{
	bool operator()(const Order &x,const Order &y){return x.Sum<y.Sum;}
};

int main()
{
	int TC;
	cin>>TC;
	int A[TC],B[TC];
	Order C[TC];
	for(int i=0;i<TC;i++)
	{
		cin>>A[i]>>B[i];
		C[i].i = i+1;
		C[i].Sum = A[i] + B[i];
	}
	
	sort(C,C+TC,comp());
	
	for(int i=0;i<TC;i++)
	{
		cout<<C[i].i<<" ";
	}
	
	return 0;
}
