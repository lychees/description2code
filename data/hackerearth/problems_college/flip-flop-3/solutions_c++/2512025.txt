#include<bits/stdc++.h>// Try try and try again for 100
using namespace std;
int main(){
	long long n;
	cin>>n;
	while(n--){
		string str;
		cin>>str;
		long long count=0;
		for(long long i=0;i<str.length();){
			char ch=str[i];
			i++;
			while(ch==str[i]){
				count++;
				i++;
				if(i==str.length())
					break;
			}
		}
		cout<<count<<endl;
	}
}
