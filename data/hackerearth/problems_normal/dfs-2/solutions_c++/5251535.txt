#include<bits/stdc++.h>
#include <stdio.h>

using namespace std;
int cmp(const void* a,const void *b)
{   return (*(int*)a-*(int*)b );}
int b_search(int key, int*arr, int n,int mode)
{
    int ind=-1, low=0, high=n-1, mid;

    while(low<=high)
    {
        mid = (low + high) >> 1;
        if(arr[mid]==key)
        {
            ind = mid;
            if(mode)
                high = mid-1;
            else
                low = mid+1;
        }
        else if(arr[mid]>key)
            high=mid-1;
        else
            low= mid+1;
    }

    return ind;


}
int main()
{
    int n, q;
    scanf("%i",&n);
    int arr[n], i, l, r, x, y;

    for(i=0;i<n;i++)
        scanf("%i",&arr[i]);

    sort(arr,arr+n);

    scanf("%i",&q);
    while(q--)
    {
        scanf("%i %i",&l,&r);
        if(l<=arr[n-1])
        {
            // Making it within range
            x = b_search(l,arr,n,1);
            while( x == -1 && ++l <= arr[n-1])
                x = b_search(l,arr,n,1);

            // Making it within range   
            if( r > arr[n-1] )
                r = arr[n-1];
            y = b_search(r,arr,n,0);
            while( y == -1 && --r >=l)
                y = b_search(r,arr,n,0);

            printf("%i\n",y-x+1);

        } 
        else
            puts("0");
    }
    return 0;
}
