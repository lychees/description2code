#include<bits/stdc++.h>
#define pii pair<long long,long long>
#define fi first
#define se second
#define mp make_pair
#define vpi vector< pii >
#define pb push_back
#define ll long long int
#define mod 1000000007
#define gcd(a,b) __gcd(a,b)
#define sf(n) scanf("%lld",&(n))
#define pf(n) printf("%lld\n",(n))
#define setbits(n) __builtin_popcount((n))
#define pq priority_queue
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,m;
        cin>>n>>m;
        vector<int>x;
        vector<int>y;
        vector<int>ch;
        int ans=0;
        for(int i=0;i<n;i++)
        {
            int a,b;
            cin>>a>>b;
            x.pb(a);y.pb(b);ch.pb(0);
        }
        for(int i=0;i<m;i++)
        {
            int a,b;
            cin>>a>>b;
            x.pb(a);y.pb(b);ch.pb(1);
        }
        n=x.size();
        int dis[n][n];
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                dis[i][j]=(x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]);
            }
        }
        for(int i=0;i<n;i++)
        {
            if(ch[i]==1)
            {
            for(int j=0;j<n;j++)
            {
                for(int k=0;k<n;k++)
                {
                 dis[j][k]=min(dis[j][k],max(dis[j][i],dis[i][k]));
                }
            }
            }
        }
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(ch[i]==1 && ch[j]==0)
                    ans=max(ans,dis[i][j]);
            }
        }
        cout<<fixed;
        cout<<setprecision(6);
        cout<<sqrt(ans)<<"\n";
    }
}
