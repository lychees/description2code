#include <stdio.h>
#define max 1000007
int a[1000009];
void sieve(void)
{
	int i,j;
	for(i=2;i*i<=max;i++)
	{
		if(a[i]==0)
		{
			for(j=i;j*i<=max;j++)
			{
				a[i*j]=1;
			}
		}
	}
	a[1]=1;
	a[4]=0;
	for (i = 2; i <= max; ++i)
	{
		a[i]+=a[i-1];
	}
}


int main(int argc, char const *argv[])
{
	int i,j,t,l,r,ans,x;
	sieve();
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&l,&r);
		ans=(a[r] - a[l-1]);
		printf("%d\n",ans);
	}
	return 0;
}
