// time-complexity O(MAXN*MAXN) for precalculation, O(1) per query

#include <cstdio>
#include <algorithm>
#include <iostream>
#include <cstdio>

using namespace std;

const int MAXN = 10100;
bool win[MAXN];

void precalc() {
    win[0] = win[1] = false;
    for (int i = 2; i < MAXN; i++) {
        win[i] = false;
        for (int j = 1; j < i; j++) {
            if (!win[j] && !win[i - j]) {
                win[i] = true;
                break;
            }
        }
    }
}

void solve() {
    int a, b;
    cin>>a>>b;
    if (win[a] || win[b]) {
        cout<<"Shekhar"<<"\n";
    }
    else {
        cout<<"Nilesh"<<"\n";
    }
}

int main() {
    precalc();
    int tests;
    cin>>tests;
    while (tests--) {
        solve();
    }
    return 0;
}
