#include <iostream>
using namespace std;
struct A
{
    unsigned long price;
}ok;
int main()
{
    int t;
    cin>>t;
    while(t)
    {
    	int n;
    	cin>>n;
        struct A prices[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>prices[i].price;
    	}
   
      //  cout<<prices[n-1].price<<"\n";
    	for(int i=1;i<n;i++)
    	{
    		
    		int low=0,high=i-1;
    		unsigned long sum=0,maxi=0;
    		while(low<=high)
    		{
    			sum=prices[low].price;
    			sum+=prices[high].price;
    			if(sum>maxi)
    			 maxi=sum;
    			low++;
    			high--;
    		}
    		if(maxi>prices[i].price)
    		   prices[i].price=maxi;
    	}
        cout<<prices[n-1].price<<"\n";
    	
    	
    	t--;
    }
    return 0;
}
