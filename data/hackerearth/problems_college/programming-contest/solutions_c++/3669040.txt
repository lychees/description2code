#include<bits/stdc++.h>
using namespace std;
int main()
{
	string str1,str2,str3;
	cin>>str1;
	cin>>str2;
	cin>>str3;
	string str = str1+str2+str3;
	int l = str1.length();
	int i = l-1;
	int j = l;
	while(i>=0 && j<str2.length()+l)
	{
		if(str[i] == str[j])
		{
			str[j] ='0';
			j++;
			i--;
		}
		else
		{
			break;
		}
	}
	int l1 = str1.length()+ str2.length();
	i= l1-1;
	j = l1;
	while(i>=0 && j<str.length())
	{
		if(str[i] == str[j])
		{
			str[j] ='0';
			j++;
			i--;
		}
		else
		{
			break;
		}
	}
	string s ="";
	for(i=0;i<str.length();i++)
	{
		if(str[i]!='0')
		{
			s = s + str[i];
		}
	}
	cout<<s.length()<<"\n";
}
