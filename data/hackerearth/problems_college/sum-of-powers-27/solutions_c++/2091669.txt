#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
vector<long long>v;
int main()
{
    int t,k,j,x=0,i;
    cin>>t;v.push_back(1);x++;
    for(k=2;k<=1000000;k++)
    {
        if(k*k<=1000000){v.push_back(k*k);x++;}else break;
        for(j=1;;j++){if(v[x-1]*k>1000000){break;}else {v.push_back(v[x-1]*k);x++;}}
    }
    sort(v.begin(),v.end());
    while(t--)
    {
        long long n,y=0,z=0,p=0,ans=0;
        cin>>n;
        for(i=0;i<v.size();i++)
        {
            y=v[i];
            z=n-y;
            if(binary_search(v.begin(),v.end(),z)){ans++;break;}
        }
        if(ans==1){cout<<"Yes\n";}else cout<<"No\n";
    }
    return 0;
}
