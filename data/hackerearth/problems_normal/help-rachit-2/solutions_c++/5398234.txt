#include<cstdio>
#include<cstring>
#include<vector>
#include<cmath>
#define MAXN 10000
using namespace std;
vector<int>v;
bool flag[MAXN+11];
double ans[MAXN+11];
void init(){
    memset(flag,true,sizeof(flag));
    memset(ans,0,sizeof(ans));
    v.push_back(2);
    for(int i=4;i<=MAXN;i+=2) flag[i]=false;
    for(int i=3;i<=MAXN;i+=2){
        if(flag[i]){
            v.push_back(i);
            for(int j=i+i;j<=MAXN;j+=i){
                flag[j]=false;
            }
        }
    }
}
bool isPrime(int n){
    int q=sqrt(n);
    for(int i=0;i<v.size()&&v[i]<=q;i++){
        if(n%v[i]==0)
            return false;
    }
    return true;
}
void solve(){
    ans[1]=0;
    int step=0,p=0,cnt=1,base=1;
    for(int i=3;i<=MAXN;i+=2){
        step+=2;
        for(int j=1;j<=4;j++){
            base+=step;
            if(isPrime(base)){
                p++;
            }
        }
        cnt+=4;
        ans[i]=p*1.0/cnt;
    }
}
int main(){
    init();
    int cs;
    solve();
    scanf("%d",&cs);
    while(cs--){
        int n;
        scanf("%d",&n);
        printf("%.6lf\n",ans[n]*100);
    }
    return 0;
}
