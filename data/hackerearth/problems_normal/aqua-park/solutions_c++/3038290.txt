#include<stdio.h>

typedef long long ll;

double X,Y,x1[101],y1[101],x2[101],y2[101];
ll n;

 int visit[101] = {0};

ll get_slide()
{
    ll res= -1;
    double y_max = -1;
    double y_line;//represents the y coordinate of a point on the line segment for a given X coordinate of that point.

    for(ll i=0;i<n;++i)
    {
        if(visit[i]||x1[i]>X||x2[i]<X)
        {
            continue;
        }
        y_line = y1[i] + (y2[i]-y1[i])*(X-x1[i])/(x2[i]-x1[i]);
        //line formula (y-y1) = m(x-x1) where m = (y2-y1)/(x2-x1)
        if(y_line > Y)
            continue;
        else
        {
            if(y_line > y_max)
            {
                y_max = y_line;
                res = i;
                visit[i] = 1;
            }
        }
    }
    return res;
}

int main()
{
    scanf("%lf%lld",&X,&n);
    Y = (1e+7);
    for(ll i=0;i<n;++i)
    {
        scanf("%lf%lf%lf%lf",&x1[i],&y1[i],&x2[i],&y2[i]);
    }
    for(ll i=0;i<n;++i)
    {
        ll id = get_slide();
        if(id == -1)
            break;
        //printf("%lld\n",id);
        if(y1[id]<y2[id])
            {
                Y = y1[id];
                X = x1[id];
            }
        else
            {
                Y = y2[id];
                X = x2[id];
            }
    }
    printf("%lld",(ll)X);
    return 0;
}
