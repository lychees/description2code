#include <iostream>
#include <map>
#include <set>
using namespace std;

int leastDiv(int a, int ld) {
	for (int j = ld+1; j <= a; j++) 
		if (a%j == 0)
			return j;
}

int main()
{
    int t;
    cin >> t;
    while (t > 0) {
    	t--;
    	
    	int n;
    	cin >> n;
    	
    	if (n == 1) {
    		cout << "FALSE" << endl;
    		continue;
    	}
    	set <int> divs;
    	int res = n;
    	int ld = 1;
    	
    	while (res > 1) {
    		ld = leastDiv(res, ld);
    		divs.insert(ld);
    		while (res%ld == 0) {
    			res /= ld;
    		}
    	}
    	
    	set <int>::iterator sitr;
    	
    	int count = n - 1;
    	for (int i = 2; i < n; i++) {
	    	for (sitr = divs.begin(); sitr != divs.end(); sitr++) {
	    		if(i%(*sitr) == 0) {
	    			count--;
	    			break;
    			}
	    	}
    	}
    		
    		
    	n = count;
    	
    	if (n == 1)
    		cout << "FALSE" << endl;
    	else {	
	    	int j;
	    	bool exit = false;
	    	for (j = 2; j*j <= n && !exit; j++) {
	    		if (n%j == 0) {
	    			cout << "FALSE" << endl;
	    			exit = true;
	    		}
	    	}
	    	
	    	if (!exit)
	    		cout << "TRUE" << endl;
    	}
    }
    return 0;
}
