long long bs(long long x){
	long long lb = 0;
	long long ub = 10000000;
	while(ub-lb>1){
		long long med = (lb+ub)/2;
		long long val = ((1+med)*med)>>1LL;
		if(val < x){
			lb = med;
		}else{
			ub = med;
		}
	}
	return lb;
}

long long get_sum(long long x){
	if(x==0) return 0;
	long long N = bs(x);
	long long ret = 0;
	if(N%2==1){
		ret += ((1+N)*(N/2 + 1))/2;
	}else{
		ret += ((1+N-1)*(N/2))/2;
	}
	x -= ((1+N)*N)/2;
	if(x%2==1){
		ret += N+1;
	}
	return ret;
}

int main(){
	int T;
	cin >> T;
	long long l,r;
	while(T--){
		cin >> l >> r;
		cout << get_sum(r) - get_sum(l-1) << endl;
	}
	return 0;
}
