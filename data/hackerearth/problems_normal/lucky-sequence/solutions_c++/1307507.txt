#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        ll n;
        scanf("%lld",&n);
        ll res=2;
        int l;
        for(int i=1;i<=60;i++) {
            if(res>=n) {
                l=i;
                break;
            } else {
                n-=res;
                res*=2;
            }
        }
        ll t1=(n-1);
        vector<int>v;
        while(t1) {
            v.push_back(t1%2);
            t1>>=1;
        }
        for(int i=v.size();i<l;i++) {
            v.push_back(0);
        }
        for(int i=l-1;i>=0;i--) {
            if(v[i]==1) {
                printf("7");
            } else {
                printf("4");
            }
        }
        puts("");
    }
    return 0;
}
