#include<bits/stdc++.h>
using namespace std;
#define ll long long
 
int main()
{
	int a[1005],b[1005],x[1005],y[1005];
	int n,k;
	cin>>n>>k;
	for(int i=1;i<=n;i++)
	cin>>a[i]>>b[i];
	
	for(int i=1;i<=n&&i<=k;i++)
	{
		x[i]=a[i];
		y[i]=b[i];
	}
	
	for(int i=k+1;i<=n;i++)
	{
		int temp=INT_MAX;
		for(int j=i-1;j>=i-k;j--)
		{
			int p=max(a[i],x[j]);
			int q=max(b[i],y[j]);
			if(temp>=p*q)
			{
				x[i]=p;
				y[i]=q;
				temp=p*q;
			}
		}
	//	cout<<x[i]<<" "<<y[i]<<endl;
	}
	
	int ans=x[n]*y[n];
	for(int i=n-1;i>0&&i>=n-k+1;i--)
	{
		ans=min(ans,x[i]*y[i]);
	}
	cout<<ans;
    return 0;
}
