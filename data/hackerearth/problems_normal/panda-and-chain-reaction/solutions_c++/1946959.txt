    #include <bits/stdc++.h>
    #define m 1000003
    using namespace std;
     
    long long  dp[m] ;
     
    void ini(){
    	dp[0]=1;
    	for(long long  i = 1 ;i<m;i++)
    		dp[i]=(dp[i-1]*i)%m ;
    }
    int main()
    {	int t ; 
    	cin >> t ;ini();
    	while(t--){
    		long long n, x ;
    		cin>>n>>x ;
    		long long ans =1 ;
    		if(n>=m) cout<<"0\n" ;
    		else{
    			 ans = (dp[n]*(x%m))%m ;
    			cout<<ans<<"\n" ;
    		}
    	}
        return 0;
    }
