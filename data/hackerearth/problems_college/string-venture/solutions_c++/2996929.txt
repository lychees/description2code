#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstring>

using namespace std;

int weight(string & str)
{
	int w = 0;
	for (char ch : str) {
		w += ch;
	}
	
	return w/str.size();
}

int isEven(int n)
{
	return ((n & 1) == 0);
}

int main()
{	
	string str;
	
	getline(cin, str);
	
	//printf(" w = %d\n", weight(str));
	
	if (isEven(weight(str))) {
		std::reverse(std::begin(str), std::end(str));
		std::cout << str << std::endl;
	} else {
		int table[256] = {0};
		
		int max_index = 0;
		for (char ch : str) {
			table[ch]++;
			if (table[ch] > table[max_index]) {
				max_index = ch;
			}
		}
		printf("%c\n", (char)max_index);
	}

	return 0;
}
