#include<bits/stdc++.h>
#define ll long long int
using namespace std;
map<ll, vector<ll> > mp;
ll hash_fn(char *s)
{
	ll hash=0,i;	
	for(i=0;s[i];i++)
	{
		hash=hash*26 + s[i]-'a';
	}
	return hash;
}
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("out.txt","w",stdout);
	char str[12],tmp[12];
	ll n,q,l,r,i,j,cnt=0;
	scanf("%lld",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%s",str);
		ll has=hash_fn(str);
		mp[has].push_back(i);
	}
	scanf("%lld",&q);
	while(q--)
	{
		ll m=0;
		scanf("%lld%lld%s",&l,&r,tmp);
		ll val=hash_fn(tmp);
		ll res=upper_bound(mp[val].begin(),mp[val].end(),r)-lower_bound(mp[val].begin(),mp[val].end(),l);
		printf("%lld\n",res);
	}
	//fclose(stdout);
	return 0;
}
