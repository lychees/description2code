#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int t,i,j,k,n,m_coins;
	long int a[]={2,3,5,7};
	cin>>t;
	while(t--)
	{
		cin>>n;
		long int C[n+1];
		C[0]=0;
		for(i=1;i<=n;i++)
		{
			m_coins=INT_MAX;
			for(j=0;j<4;j++)
			{
				if(i>=a[j])
					m_coins=min(m_coins,C[i-a[j]]);
			}
			if(m_coins!=INT_MAX)
				C[i]=m_coins+1;
			else
				C[i]=INT_MAX;
		}
		cout<<C[n]<<"\n";
	}
}
