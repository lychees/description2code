#include <stdio.h>
#include <math.h>
#define pi 3.14159265358979323
#define e  2.71828182845904523536 
int main()
{
float s;
long long int t,n,i,zeros,ans,res;
scanf("%lld",&t);
while(t--)
{
s=0;
scanf("%lld",&n);
if(n==0 || n==1)
ans=1;
else
ans=(long long int)floor(0.5*log10(2*pi) + (n+0.5)*log10(n)-n*log10(e) + log10(e)/(12*n+1)) + 1;
res=5;
zeros=0;
while(n/res>0)
{
	zeros+=n/res;
	res=res*5;
}
printf("%lld\n",ans-zeros);

}
return 0;
}
