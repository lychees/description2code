int vis[100000];
int bfs(vector<int> adj[],int n)
{
    int s,i,l,col;
    queue<int> q;
    q.push(0);
    vis[0]=1;
    while(!q.empty())
    {
        s=q.front();
        q.pop();
        col=vis[s];
        l=adj[s].size();
        for(i=0;i<l;i++)
        {
            if(!vis[adj[s][i]])
            {
                vis[adj[s][i]]=(col==1)?2:1;
                q.push(adj[s][i]);
            }
            else if((col==1 && vis[adj[s][i]]==1) || (col==2 && vis[adj[s][i]]==2) )
            {
                return -1;
            }
        }

    }
    return 1;

}
int main()
{
    int t,n,e,i,j,a,b;
    cin>>t;
    while(t-->0)
    {
        cin>>n>>e;
        vector<int> adj[n+1];
        for(i=0;i<e;i++)
        {
            scanf("%d%d",&a,&b);
            adj[a].push_back(b);
            adj[b].push_back(a);
        }
        memset(vis,0,sizeof(vis));
        int ans=bfs(adj,n);
        if(ans==1)cout<<"BICOLORABLE."<<endl;
        else cout<<"NOT BICOLORABLE."<<endl;
    }


    return 0;
}
