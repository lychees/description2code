#include<iostream>
#include<cstdlib>
using namespace std;
int main()
{

int t;
cin>>t;
int div[1000009]={0};
int ans[1000009];
ans[1]=0;
for(int i=1;i<=1000000;i++)
{
	int div2=2*i;
	while(div2<=1000000)
	{
		div[div2]++;
		div2+=i;
	}
}
for(int i=2;i<=1000000;i++)
{
	ans[i]=ans[i-1]+div[i];
}
while(t--)
{
	int n;
	cin>>n;
	cout<<ans[n]<<endl;
}

return 0;
}
