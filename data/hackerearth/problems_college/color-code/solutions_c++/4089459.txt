#include<bits/stdc++.h>
using namespace std;


void convert(int decimal)
{
	int q;
	int i=1,temp;
	char hex[100];
	
	q= decimal;
	while(q!=0)
	{
		temp= q%16;
		
		if(temp<10)
		temp +=48;
		
		else
		temp+=55;
		
		hex[i++]=temp;
		
		q=q/16;
	}
	if(i==2)
	cout<<"0";
	for(int j=i-1;j>0;j--)
	{
		if(hex[j]=='F')
		hex[j]='Z';
		if(hex[j]=='E')
		hex[j]='Y';
		if(hex[j]=='D')
		hex[j]='X';
		if(hex[j]=='C')
		hex[j]='W';
		if(hex[j]=='B')
      hex[j]='V';
      if(hex[j]=='A')
      hex[j]='U';
      
      cout<<hex[j];
		
		
		
		
		
	}
	
	
	
	
}


int main()
{
	
	int t;
	cin>>t;
	for(int i=1;i<=t;i++)
	{
		int a,b,c;
		cin>>a>>b>>c;
		
		cout<<"Case "<<i<<": ";
		if(a==0)
		cout<<"00";
		else
		convert(a);
		cout<<" ";
		if(b==0)
		cout<<"00";
		else
		convert(b);
		cout<<" ";
		if(c==0)
		cout<<"00";
		else
		convert(c);
		cout<<endl;
		
		
		
		
	}
  
    return 0;
}
