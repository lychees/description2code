#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
    pair<string,int> Ar[105];
    int N;
    cin>>N;
    string s;
    for(int i=0;i<N;i++)
    {
        cin>>s;
        Ar[i].first=s;
        Ar[i].second=i;
    }
    sort(Ar,Ar+N);
    pair<int,string> Br[105];
    int c=0;
    Br[0].first=1;
    Br[0].second=Ar[0].first;
    for(int i=1;i<N;i++)
    {
        if(Ar[i].first==Br[c].second)
        {
            Br[c].first++;
        }
        else
        {
            c++;
            Br[c].first=1;
            Br[c].second=Ar[i].first;
        }
    }
    c++;
    sort(Br,Br+c);
    for(int i=c-1;i>=0;i--)
        cout<<Br[i].second<<" "<<Br[i].first<<"\n";
    return 0;
}
