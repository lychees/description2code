#include <iostream>
#include<stdio.h>
using namespace std;

int n,m,ans;
int max(int a,int b)
{
	if(a>b)
	return a;
	else
	return b;
}
int main()
{
    int t;
    char c;
    scanf("%d",&t);
    while(t--)
    {
    	ans=0;
    	scanf("%d %d",&n,&m);
    	long long int d[n][m];
    	int a[200][200];
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<m;j++)
    		{
			//	cin>>c;
				scanf(" %c",&c);
    			if(c=='T')
    			a[i][j]=150;
    			else if(c=='D')
    			a[i][j]=-100;
    			else 
    			a[i][j]=0;
    			d[i][j]=-1;
    		}
    	}
      	
      	if(a[0][0]>=0)
      	{
      		d[0][0]=a[0][0];
      	
    	for(int i=1;i<n;i++)
    	{
    		if(d[i-1][0]>=0)
    		d[i][0]=a[i][0]+d[i-1][0];
    	}
    	for(int i=1;i<m;i++)
    	{
 
    	if(d[0][i-1]>=0)
    	d[0][i]=a[0][i]+d[0][i-1];
    	}
    	for(int i=1;i<n;i++)
    	{
    		for(int j=1;j<m;j++)
    		{
    			
    			int gg=max(d[i-1][j],d[i][j-1]);
    			if(gg>=0)
    			d[i][j]=a[i][j]+gg;
    		}
    	}
    	ans=d[n-1][m-1];
    	if(ans>=0)
    	printf("%lld\n",ans);
    	else
    	printf("-1\n");
      	}
      	else
      		printf("-1\n");
    }
    return 0;
}
