#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
	string s;
	cin>>s;
	int count=0,count1=0;
	for(int i=0;i<s.length();i++){
		if(s[i]=='0'){
			count++;
		}else{
			count1++;
		}
	}
	if(count==count1){
		cout<<0<<"\n";
	}
	else{
		int m=min(count,count1);
		int x=(s.length()-(2*m));
		cout<<x<<"\n";
	}
	}
    return 0;
}
