#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
   long long n;
   cin>>n;
   vector<long long> a(n);
   vector<long long> b(n);
   long long product=0;
   for(long long i=0;i<n;i++){
   	cin>>a[i];
   }
   for(long long j=0;j<n;j++){
   	cin>>b[j];
   }
   	sort(a.begin(),a.end());
   	sort(b.begin(),b.end());
   for(long long i=0;i<n;i++){
   
   	
   	long long x=a.back();
   	long long y=b.back();
   	if(a.size()!=0){
   	a.pop_back();
   	b.pop_back();}
   	product=product+(x*y);
   }
   
   cout<<product;
    return 0;
}
