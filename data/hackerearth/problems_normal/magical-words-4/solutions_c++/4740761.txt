#include<bits/stdc++.h>
using namespace std;
char str[1001];
int pali[1001][1001];
long long dp()
{
	int n = strlen(str);
	int i,j,k,len;
	long long count = 0;
	for(i=0;i<n;i++)
	{
		pali[i][i] = 1;
		count++;	
	}
	for(len=2;len<=n;len++)
	{
		for(i=0;i<n-len+1;i++)
		{
			j = i + len - 1 ;
			if(str[i]==str[j] && len==2)
			{
				pali[i][j] = 1;
				count+=4;
			}
			else if(str[i]==str[j])
			{
				pali[i][j] = pali[i+1][j-1];
				if(pali[i][j])
				{
					count+=(j-i+1)*(j-i+1);
				}
			} 
		}
	}
	return count;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		memset(pali,0,sizeof(pali));
		cin>>str;
		cout<<dp()<<endl;
	}
}
