#include <bits/stdc++.h>
using namespace std;


int main(){
	map<string,int> mp,mp2;

	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		string a;
		int b;
		cin>>a>>b;
		mp[a]+=b;
		mp2[a] = i;
	}
	string ans = mp.begin()->first;
	int mx = mp.begin()->second;
	for(map<string,int>::iterator it = mp.begin();it!=mp.end();it++){
		int v = it->second;
		string s = it->first;
		if(v>mx){ans = s;mx = v;}
		else if(v==mx){
			if(mp2[s]<mp2[ans]){ans = s;mx = v;}
		}
	}
	cout<<ans<<endl;
}
