#include <bits/stdc++.h>
using namespace std;
#define MAX 1000000
int main()
{
    int n,q;
    scanf("%d%d",&n,&q);
    int aa[MAX]={0};
    int i;
    long int num,a,b,count=0;
    scanf("%ld%ld%ld",&num,&a,&b);
    for(i=a;i<=b;i++)
    {
    	aa[i]=1;
    }
    while(--q)
    {
    	count=0;
    	scanf("%ld%ld%ld",&num,&a,&b);
    	for(i=a;i<=b;i++)
    	{
    		if(aa[i]==1)
    		{
    			count++;
    		}
    	}
    	printf("%ld\n",count);
    	
    }
    
    
}
