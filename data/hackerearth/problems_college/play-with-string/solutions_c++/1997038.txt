#include<cstdio>
#include<cstring>
#include<cmath>
using namespace std;
int main()
{
	int j,T,i,n;
	scanf("%d",&T);
	while(T--)
	{
		char s[10000],r[10000];
		scanf("%s",s);
		n=strlen(s);
		for(j=0;j<n;j++)
		{
			r[j]=s[n-j-1];
		}
		int flag=1;
		for(i=1;i<n;i++)
		{
			if(abs(s[i]-s[i-1])!=abs(r[i]-r[i-1]))
			{	
				flag=0;
				
				break;
			}
		}
		if(flag==1)
			printf("Funny\n");
		else
			printf("Not Funny\n");
	}
	return 0;
}
