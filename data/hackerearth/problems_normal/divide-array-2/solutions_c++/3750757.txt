#include <iostream>
using namespace std;

void divide(int *arr, int size, int final_D)
{
for(int i=0;i<size;i++)
{
arr[i]=arr[i]/final_D;
}
}

int main()
{
    int n;
    cin>>n;
    
    int A[n];
    for(int i=0;i<n;i++)
    {cin>>A[i];}
    
    int q,d;
    cin>>q;
    long long int final_d=1;
    
    while(q--)
    {cin>>d;
       final_d*=d;
       
       
    }
     
    if(final_d!=0)
   { 
     divide(A,n,final_d);
    
     for(int i=0;i<n;i++)
     {cout<<A[i]<<" ";
     }
   }
   else
   {
     for(int i=0;i<n;i++)
     {A[i]=0;
     cout<<A[i]<<" ";
     }
     return 0;
   }
}
