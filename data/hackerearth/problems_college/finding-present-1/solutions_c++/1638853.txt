#include<bits/stdc++.h>

using namespace std;

int main(){
    int m,n,min1;
    cin>>m;int a[m];
    cin>>a[0];
    min1=a[0];
    for(int i=1;i<m;i++){
        cin>>a[i];
        if(min1>a[i])min1=a[i];
    }
    cin>>n;int b[n];
    int min2;
    cin>>b[0];
    min2=b[0];
    for(int i=1;i<n;i++){
        cin>>b[i];
        if(min2>b[i])min2=b[i];
    }
    int num1[101]={0},num2[101]={0};
    //cout<<min1<<" "<<min2<<endl;
    for(int i=0;i<n;i++){
        num2[b[i]-min2]++;
    }
    for(int i=0;i<m;i++){
        num1[a[i]-min1]++;
    }
    for(int i=0;i<101;i++){
        if(num2[i]-num1[i]>0)cout<<i+min2<<" ";
    }
    cout<<endl;
}
