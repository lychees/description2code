{

    int n;
    cin>>n;
    vector<long long int> a(n);
    for(int i=0;i<n;i++)
    	cin>>a[i];
    sort(a.begin(),a.end());
    long long int d[n];
    for(int i=0;i<n;i++)
    	d[i]=1;
    d[0]=1;
    long long int mx=-1;
    for(int i=1;i<n;i++)
    {
    	for(int j=0;j<i;j++)
    	{
    		if(a[i]%a[j]==0)
    		{
    		   if(d[i]<d[j]+1)
    		   	d[i]=d[j]+1;
    		}
      	mx=max(mx,d[i]);
    	}
    
    }
    /*for(int i=0;i<n;i++)

    if(mx==1)
    cout<<"-1";
