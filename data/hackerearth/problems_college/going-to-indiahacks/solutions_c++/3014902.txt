#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;

int main()
{
    cin.sync_with_stdio(0);
    cin.tie(0);
    
   
    long long int a[1005][1005];
    
    for(int i = 0; i <= 1004; i++)
    {
    	a[i][0] = 0;
    	a[0][i] = 0;
    }
    
    a[1][1] = 1;
    
    for(int i = 1; i <= 1004; i++)
    {
    	for(int j = 1; j <= 1004; j++)
    	{
    		if(i == 1 && j == 1)
    			continue;
    		a[i][j] = (a[i-1][j-1] + a[i-1][j] + a[i][j-1]) % MOD;
    		
    	}
    }
    
    int t, m1, m2, m3, m4;
    cin >> t;
    while(t--)
    {
    	cin >> m1 >> m2 >> m3 >> m4;
    	if(m1 > m3 || m2 > m4)
    		cout << "0\n";
    	cout << a[m3-m1 + 1][m4-m2 + 1 ] << '\n';
    }
    return 0;
}
