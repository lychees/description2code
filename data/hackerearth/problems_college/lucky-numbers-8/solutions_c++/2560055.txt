#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int n,cnt=0;
   	scanf("%d",&n);
   	//cin>>n;
   	int arr[n],i,j;
   	for(i =0;i<n;i++)
   		scanf("%d",&arr[i]);
   		//cin>>arr[i];
   	//int flag;
   	for( i =0;i<n;i++)
   		{
   			for(j =i+1;j<n;j++)
   				if(arr[j] >= arr[i])
   					break;
   			if(j==n)
   				cnt++;
   		}
   	cout<<cnt<<endl;
   }
}
