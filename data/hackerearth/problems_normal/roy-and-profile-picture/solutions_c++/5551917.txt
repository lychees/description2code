#include <iostream>
using namespace std;

int main()
{
	int L,N,i=0;
	cin>>L;
	cin>>N;
	int W[N],H[N];
    while(i<N)
    {
    	cin>>W[i]>>H[i];
    	i++;
    }
    i=0;
    while(i<N)
    {
    if(W[i]<L||H[i]<L)
    {
    	cout<<"UPLOAD ANOTHER"<<endl;
    	
    }
    else
    {
    	if(W[i]==H[i])
    	cout<<"ACCEPTED"<<endl;
    	else
    	cout<<"CROP IT"<<endl;
    	
    }
    i++;
	}
    return 0;
}
