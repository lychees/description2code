#include <iostream>
using namespace std;
#include<algorithm>

int main()
{
    int t,n,i;
    long long int raghu,syan,raghuc,syanc,tot;
    cin >> t;
    int dishes[10000];
    while(t--)
    {
    	cin >>raghu >>syan >>n;
    	for(i=0;i<n;i++)
    	{
    		cin >>dishes[i];
    	}
    	sort(dishes,dishes+n);
    	tot=raghuc=syanc=0;
    	for(i=0;i<n;i++)
    	{
    		tot+=dishes[i];
    		if(tot<=raghu)
    			raghuc++;
    	
    		if(tot<=syan)
    			syanc++;
    		else
    			break;
    
    	}
    
    if(raghuc>syanc)
    	cout<< "Raghu Won\n";
    if(syanc>raghuc)
    	cout<< "Sayan Won\n";
    if(syanc==raghuc)
    	cout<< "Tie\n";
    }
    
}
