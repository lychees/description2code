int endDates[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

int countify(int d, int m, int y)
{
	if(y == 0) // Corner case
		return 0;	
	int ans;
	ans = (y/100) * 11; // Before this century, how many good dates are there
	ans += max(0, min(11, ((y%100) - 1) - 2)); // In this century, before this year, how many good dates are there
	if(m > (y%100) - 1 and y%100 > 2) // In this century, in this year, how many good dates are there
		ans++;
	else if(m == (y%100) - 1 and d >= m - 1 and y%100 > 2)
		ans++;
	return ans;
}

int leap(int y)
{
	if(y % 4 == 0)
	{
		if(y % 100 == 0)
		{
			if(y % 400 == 0)
				return 1;
			else
				return 0;
		}
		else
			return 1;
	}
	else
		return 0;
}

int main()
{
	// ios::sync_with_stdio(false);
	int t;
	int d1, m1, y1, d2, m2, y2;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d:%d:%d", &d1, &m1, &y1);
		scanf("%d:%d:%d", &d2, &m2, &y2);
		if(d1 == 1)
		{
			if(m1 == 1)
			{
				y1--;
				m1 = 12;
				d1 = 31;
			}
			else
			{
				m1--;
				if(m1 == 2)
				{
					if(leap(y1))
						d1 = 29;
					else
						d1 = 28;
				}
				else
					d1 = endDates[m1];
			}
		}
		else
			d1--;
		printf("%d\n", countify(d2, m2, y2) - countify(d1, m1, y1));
	}
	return 0;
}
