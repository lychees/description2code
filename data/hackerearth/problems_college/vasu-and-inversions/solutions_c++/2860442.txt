#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int main()
{
	int t;
	cin>>t;
	while(t--) {
		int n,k,i,x;
		scanf("%d%d",&n,&k);
		int c = 0;
		priority_queue<int,vector<int>,greater<int> >q;
		for(i = 0; i < n; i++) {
			scanf("%d",&x);
			q.push(x);
			if(x) {
				c++;
			}
		}
		for(i = 0; i < k; i++) {
			x = q.top();
			x ^= 1;
			if(!x) {
				c--;
			}
			else {
				c++;
			}
			q.pop();
			q.push(x);
		}
		printf("%d\n",c);
	}
	return 0;
}
