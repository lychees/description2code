#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,i,j;
    cin>>n;
    for(j=1;j<=n;j++) cout<<'*';
    cout<<endl;
    for(i=2;i<=n-1;i++)
    {
    	for(j=1;j<=n-i;j++) cout<<" ";
    	cout<<"*"<<endl;
    }
    for(j=1;j<=n;j++) cout<<"*";
    return 0;
}
