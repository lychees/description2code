from fractions import gcd
for t in range(input()):
    a, b, c = map(long, raw_input().split())
    if gcd(a, b) == 1 and gcd(b, c) == 1 and gcd(c, a) == 1:
        a = a*a
        b = b*b
        c = c*c
        if a+b == c or b+c == a or a+c == b:
            print "YES"
        else:
            print "NO"
    else:
        print "NO"
