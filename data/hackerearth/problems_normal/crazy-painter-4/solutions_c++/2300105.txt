#include<iostream>
using namespace std;

main()
{
    long long int N,Pts,rem,q;
    long int T;
    long int C[26];
    int i;
    
    cin>>T;
    
    long long int *Sum;
    Sum = new long long int [T];
    long int K=T;
    while(T-- > 0)
    {
        cin>>N;
        Pts = 12*N;
        
        for(i=0;i<26;i++)
        {
            cin>>C[i];
        }
        rem = Pts%26;
        q = Pts/26;
        
        for(i=0;i<26;i++)
        {
        		Sum[T] += q*C[i] + q*(q-1)/2;
        }
        for(i=0;i<rem;i++)
        {
            Sum[T] += (q + C[i]);
        }
    }
    
    for(i=K-1;i>=0;i--)
    {
        cout<<Sum[i]<<endl;
    }
}
