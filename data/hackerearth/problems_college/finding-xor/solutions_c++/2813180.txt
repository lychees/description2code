#include <iostream>
using namespace std;
 
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
	   string s1,s2;
	   int i=0,l;
	   cin>>s1>>s2;
	   l=s1.size();
	   for(i=0;i<l;i++)
	   {
	   	 if(((s1[i]=='1')&&(s2[i]=='1'))||((s1[i]=='0')&&(s2[i]=='0')))
	   	 cout<<0;
	   	 else
	   	 cout<<1;
       }
       cout<<endl;
	}
    return 0;
}
