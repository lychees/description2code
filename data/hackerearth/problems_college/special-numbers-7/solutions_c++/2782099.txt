#include<bits/stdc++.h>
#define ll long long
using namespace std;
bool isprime(ll n)
{
    for(ll i=2;i<=n/2;i++)
    {
           if(n%i==0)
               return false;
    }
    return true;
}
bool isfact(ll n)
{
    ll fac=0;
    for(ll i=1;i<=n/2;i++)
    {
        if(isprime(i) && n%i==0)
            fac++;
    }
    if(fac>3)
        return true;
    else
        return false;
}
int main()
{
    //cout<<isfact(30);
    //cout<<isfact(10);
    ll t;
    cin>>t;
    ll input[t];
    for(ll i=0;i<t;i++)
      cin>>input[i];
    ll max1=input[0];
    for(ll i=1;i<t;i++)
        if(input[i]>max1)
            max1=input[i];
    vector<ll> v;
   for(ll i=3;v.size()<=max1;i++)
    {
          if(isfact(i))
          {
              v.push_back(i);
          }
    }
     for(ll i=0;i<t;i++)
    {
        cout<<v[input[i]-1]<<endl;
    }
    return 0;
}
