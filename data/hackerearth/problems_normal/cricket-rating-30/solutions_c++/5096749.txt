#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main()
{
    ll n;
    cin >> n;
    ll sum = 0, ans = 0, tem ;
    
    for(ll i = 0; i < n; i++){
    	
    	cin >> tem ;
    	
    	if(sum <= 0){
    		if(tem < 0) continue ;
    		sum = tem ;
    	}
    	else {
    		sum += tem ;
    	}
    	
    	ans = max(ans, sum);
    }
    
    cout << ans << endl ;
    return 0;
}
