#include<bits/stdc++.h> 
using namespace std;
	int a[123456];
	vector<int>v[123456];
	int x[123456];
int main() {

	memset(a,0,sizeof(a));
	a[1] = 1;
	int i,j;
	for(i=2;i<=123455;i++) {
		if(a[i]==0) {
			v[i].push_back(i);
			for(j=i+i;j<=123455;j=j+i) {
				v[j].push_back(i);
				a[j] = 1;	
			}	
		}
	}
	int n;
	scanf("%d",&n);
	
	memset(x,-1,sizeof(x));
	int q[n];
	memset(q,0,sizeof(q));
	int arr[n];
	for (int i=0;i<n;i++) {
		scanf("%d",&arr[i]);
	}
	q[n-1] = 1;
	for (int i=0;i<v[arr[n-1]].size();i++) {
		x[v[arr[n-1]][i]]=n-1;
	}
	
	for (int i=n-2;i>=0;i--) {
		int flag = -2;
		for (int j=0;j<v[arr[i]].size();j++) {
	//		printf("%d ",v[arr[i]][j]);
			if(x[v[arr[i]][j]]!=-1) {
				if(q[i]<1 + q[x[v[arr[i]][j]]]) {
					q[i] = 1 + q[x[v[arr[i]][j]]];
					x[v[arr[i]][j]] = i;
					flag = j;
				}
				else {
					x[v[arr[i]][j]] = i;
				}
			}
			else {
				q[i] = max(q[i],1);
				x[v[arr[i]][j]] = i;
			}
		}
	//	printf("\n");
	}
	int s=0;
	for(i=0;i<n;i++) {
	//	printf("%d ",q[i]);
		s = max(s,q[i]); 
	}
	printf("%d\n",s);
}
