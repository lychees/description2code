int main() {
	
	ios_base::sync_with_stdio(false);

    int n,p; cin >> n >> p;
	dp[0] = 1;
	dp[1] = 0;
	dp[2] = p / 100.0;
	dp[3] = 1 - dp[2];

	printf("%.6lf\n",dp[n]);

}
