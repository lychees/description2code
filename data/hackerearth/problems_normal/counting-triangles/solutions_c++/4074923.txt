#include <bits/stdc++.h>
#define MAXN 100001
using namespace std;

struct t{ long long b[3]; };
t a[MAXN];

bool cmp(t a1, t a2) {
	if(a1.b[0] < a2.b[0]) return true;
	if(a1.b[0] > a2.b[0]) return false;
	if(a1.b[1] < a2.b[1]) return true;
	if(a1.b[1] > a2.b[1]) return false;
	if(a1.b[2] < a1.b[2]) return true;
	return false;
}

int main()
{
    int n, cnt = 0; scanf("%d", &n);
    for(int i = 1; i <= n; i++) {
    	scanf("%lld%lld%lld", &a[i].b[0], &a[i].b[1], &a[i].b[2]);
    	sort(a[i].b, a[i].b+3);
    }
    sort(a, a+n+1, cmp);
    for(int i = 1; i <= n; i++) {
    	if((a[i].b[0] != a[i-1].b[0] || a[i].b[1] != a[i-1].b[1] || a[i].b[2] != a[i-1].b[2]) && (a[i].b[0] != a[i+1].b[0] || a[i].b[1] != a[i+1].b[1] || a[i].b[2] != a[i+1].b[2])) cnt++;
    }
    printf("%d", cnt);
    return 0;
}
