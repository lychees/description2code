#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
#define MAXN 1000001

int a[MAXN], b[MAXN];

int main()
{
    int N,M;
    scanf("%d%d", &N, &M);

    while(M--)
    {
        int L,R;
        scanf("%d%d", &L, &R);
        a[L] += 1;
        a[R+1] -= 1;
    }

    for(int i=2; i<=N; i++)
        a[i] += a[i-1];

    for(int i=1; i<=N; i++)
    {
        int noOfCoins = a[i];
        b[noOfCoins] += 1;
    }

    for(int i=N-1; i>=1; i--)
        b[i]  += b[i+1];

    int Q;
    scanf("%d", &Q);
    while(Q--)
    {
        int X;
        scanf("%d", &X);
        printf("%d\n", b[X]);
    }

    return 0;
}
