from itertools import groupby
def checkIfOK(st):
    groups = []
    for k, g in groupby(st):
        groups.append(list(g))
    for x in groups:
        if len(x) != len(st)/3:
            return False
    return True    

for _ in range(input()):
    inst  = raw_input().strip()
    ins = set(inst)
    lns = len(ins)

    if lns != 3:
        print "Not OK"
        continue
    if checkIfOK(inst):
        print "OK"
    else:
        print "Not OK"
