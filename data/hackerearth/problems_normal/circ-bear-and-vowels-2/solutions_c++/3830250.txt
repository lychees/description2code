#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
	int T;
	cin >> T;
	while(T--){
		string s;
		cin >> s;
		int cons = 0, vow = 0, continuous = 0;
		for(int i = 0 ; i < s.size(); ++i){
			if(s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u' || s[i] == 'y'){
				++vow;
				continuous = 0;
			}
			else{
				++cons;
				++continuous;
				if(continuous == 3)
					break;
			}
		}
		if(continuous == 3 || cons > vow)
			cout << "hard" << endl;
		else
			cout << "easy" << endl;
	}
	return 0;
}
