#include <bits/stdc++.h>
using namespace std;
char a[1000005];
int main()
{
   // cout << "Hello World!" << endl;
   int t;
   scanf("%d",&t);
   while(t--)
   {
   	scanf("%s",a);
   	int i;
   	int co[26];
   	for(i=0;i<26;i++)co[i] = 0;
   	for(i=0;a[i];i++)co[a[i]-'a']++;
   	set<int> s;
   	for(i=0;i<26;i++)s.insert(co[i]);
   	set<int>::iterator I;
   	for(I=s.begin();I!=s.end();I++)
   	{
   		for(i=25;i>=0;i--)
   		if(co[i]==*I)printf("%c ",i+'a');
   	}
   	printf("\n");
   }
    return 0;
}
