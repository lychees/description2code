#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    long long int t,n,i,k,count=0;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	count=0;
    	cin>>n;
        while(n>0)
        {
        	n/=2;
        	count++;
        }
        k=pow(2,count)-1;
    	cout<<k<<endl;
    }
    return 0;
}
