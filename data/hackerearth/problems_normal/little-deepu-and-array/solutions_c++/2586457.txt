#include <iostream>
using namespace std;
int m,n;

void hit(int *a,int x)
{
	for(int i=0;i<n;i++)
	{
		if(a[i]>x)
		{
			a[i]=a[i]-1;
		}
	}
}
int main()
{
	int x;
    cin>>n;
    int q[n];
    for(int i=0;i<n;i++)
    {
    	cin>>q[i];
    }
    cin>>m;
    for(int i=0;i<m;i++)
    {
    	cin>>x;
    	hit(q,x);
    }
    for(int i=0;i<n;i++)
    {
    	cout<<q[i]<<" ";
    }
    return 0;
}
