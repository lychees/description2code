#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
        long double a, b, c, d, D, r, angle, ans = 0.00;;
        cin >> a >> b >> c >> d;

        D = (c-b);
        r = (b-a);
        if(2*r <= D){
            cout << "0.000000" << endl;
            continue;
        }
        angle = acos(D/(2*r));
        ans = angle/acos(-1);
        cout << setprecision(6) << fixed << ans << endl;
    }
    return 0;
}
