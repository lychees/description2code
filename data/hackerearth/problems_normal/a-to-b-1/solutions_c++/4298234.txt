#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
long long int power(long long int a,long long int b)
{
	long long int res=1;
	while(b>0)
	{
		if(b%2==1)
		{
			res=res*a;
			res=res%mod;
			
		}
		a=a*a;
		a=a%mod;
		b=b/2;
	}
	return res;
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int n,count0=0;
    cin>>n;
    int *a=new int[n];
    //int *b=new int[n];
    int i;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    long long int pro=1;
    for(i=0;i<n;i++)
    {
    	if(a[i]==0)
    	{
    		count0++;
    	}
    	else
    	{
    		pro=(pro*a[i])%mod;
    	}
    }
    int q;
    cin>>q;
    while(q--)
    {
    	int o;
    	cin>>o;
    	if(o==0)
    	{
    		int x,y;
    		cin>>x>>y;
    		if(y==a[x-1])
    		{
    			continue;
    		}
    		else if(y==0)
    		{
    			count0++;
    			pro=(pro*power(a[x-1],mod-2))%mod;
    		}
    		else if(a[x-1]==0)
    		{
    			count0--;
    			pro=(pro*y)%mod;
    		}
    		else
    		{
    			pro=(pro*power(a[x-1],mod-2))%mod;
    			pro=(pro*y)%mod;
    		}
    		a[x-1]=y;
    	}
    	else
    	{
    		int x;
    		cin>>x;
    		if(count0==0)
    		{
    			cout<<(pro*power(a[x-1],mod-2))%mod;
    		}
    		else if(count0==1)
    		{
    			if(a[x-1]==0)
    			{
    				cout<<(pro)%mod;
    			}
    			else
    			{
    				cout<<"0";
    			}
    		}
    		else
    		{
    			cout<<"0";
    		}
    		cout<<"\n";
    	}
    }
    return 0;
}
