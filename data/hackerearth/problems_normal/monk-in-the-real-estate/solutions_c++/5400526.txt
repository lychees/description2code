#include<bits/stdc++.h>
using namespace std;
typedef struct node
{
	int data;
	struct node *next;
}node;
node* g[10100];

int main()
{
	int t,i;
	cin>>t;
	while(t--)
	{
		int e;
		cin>>e;
		for(i=1;i<=10000;i++)
		{
			g[i]=NULL;
		}
		int temp_e=e;
		while(e--)
		{
			int a,b;
			cin>>a;
			cin>>b;
			node* temp=(node*)malloc(sizeof(node));
			temp->data=b;
			temp->next=NULL;
			if(g[a]==NULL)
			{
				g[a]=temp;
			}
			else
			{
				node *prev,*current=g[a];
				while(current)
				{
					prev=current;
					current=current->next;
				}
				prev->next=temp;
			}
			temp=(node*)malloc(sizeof(node));
			temp->data=a;
			temp->next=NULL;
			if(g[b]==NULL)
			{
				g[b]=temp;
			}
			else
			{
				node *prev,*current=g[b];
				while(current)
				{
					prev=current;
					current=current->next;
				}
				prev->next=temp;
			}			
		}
		int c=0;
		for(i=1;i<=10000;i++)
		{
			if(g[i]==NULL)
			{
				c++;
			}
		}
		cout<<10000-c<<"\n";
	}
	return 0;
}
