#include<bits/stdc++.h>
using namespace std;
bool contains(string &a,char sd)
{
	for(int i=0;i<a.length();i++)
		{
		   if(a[i]==sd)
			   return 1;
		}
	return 0;
}
int main()
{
	string str;
	cin>>str;
	pair<int,int> count[26];
	for(int i=0;i<26;i++)
	{
        count[i].first=0;
        count[i].second=i;
	}
	for(int i=0;i<str.length();i++)
	{
		count[str[i]-'a'].first--;
	}
    sort(count,count+26);
    char sd=count[0].second + 'a';
    string str1="head rush";
    string str2="cubing club";
    bool flag1=contains(str1,sd);
    bool flag2=contains(str2,sd);
    if(flag1)
    {
        if(flag2)
        {
        	cout<<"Thanks both"<<endl;
        }
        else
        {
        	cout<<"Thanks head rush"<<endl;
        }
    }
    else
    {
    	if(flag2)
    	        {
    	        	cout<<"Thanks cubing club"<<endl;
    	        }
    	        else
    	        {
    	        	cout<<"none"<<endl;
    	        }
    }
	return 0;
}
