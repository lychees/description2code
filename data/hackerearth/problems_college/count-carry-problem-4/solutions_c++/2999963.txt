#include <iostream>
using namespace std;

int main()
{  int t;
  cin>>t;
  while(t--)
   { long long int a,b,sum,c=0,count=0;
   cin>>a>>b;
    while(a!=0||b!=0)
     { sum=(a%10)+(b%10)+c;
       c=sum/10;
       if(c!=0)
         count++;
        a/=10;
        b/=10;
     } 
     if(count==0)
       cout<<"No carry operation\n";
     else  if(count==1)
        cout<<"1 carry operation\n";
      else
         cout<<count<<" carry operations\n";
   }     
   return 0;
}
