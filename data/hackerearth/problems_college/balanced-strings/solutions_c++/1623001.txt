#include <iostream>
#include <cstring>

using namespace std;

int main()
{
    char str[1001];
    int n,i,len,flag;
    int count[26];
    cin>>n;
    while(n--)
    {
    	cin>>str;
    	flag=0;
    	for(i=0;i<26;i++)
    	count[i]=0;
    	len=strlen(str);
    	for(i=0;i<len;i++)
    	{
    		count[str[i]%97]++;
    	}
    	for(i=0;i<25;i++)
    	{
    	if(count[i]!=count[i+1])
         	{
    			flag=1;
    		}
          }
          if(flag==1)
          cout<<"No"<<endl;
          else
          cout<<"Yes"<<endl;
    }
    return 0;
}
