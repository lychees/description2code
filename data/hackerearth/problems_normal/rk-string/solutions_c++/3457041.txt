#include <bits/stdc++.h>

using namespace std;

int main() {
    
    int t;

    cin >> t;

    string str;
    int dp[1000001];
    
    while (t--) {
        int a, b;
        cin >> a >> b;
        cin >> str;
        memset(dp, 0, sizeof dp);
        dp[0] = 1;
        long long ans = 0;
        int sum = 0;
        for (int i = 0; i < str.length(); i++) {
        	if (str[i] == 'R') {
        		sum += 1;
        	}
        	if (sum >= a)
        		ans += dp[sum - a];
        	dp[sum]++;
        }
        cout << ans << " ";
        memset(dp, 0, sizeof dp);
        ans = 0;
       	sum = 0;
       	dp[0] = 1;
        for (int i = 0; i < str.length(); i++) {
        	if (str[i] == 'K') {
        		sum += 1;
        	}
        	if (sum >= b)
        		ans += dp[sum - b];
        	dp[sum]++;
        }
        cout << ans << endl;
    }
    return 0;
}
