//unique strings.cpp
#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	string s,fl;
	getline(cin,fl);
	vector< char > vc;
	string::iterator iter;
	while(t--)
	{
		vector< bool > v(26,false);
		getline(cin,s);
		iter=s.begin();
		for(;iter!=s.end();iter++)
		{
			if(!v[*iter-'a'])
			{
				v[*iter-'a']=true;
				vc.push_back(*iter);
			}
		}
		for(int loop=0;loop<vc.size();loop++)
			cout<<(char)vc[loop];
		printf("\n");
		vc.clear();
	}

	return 0;
}
