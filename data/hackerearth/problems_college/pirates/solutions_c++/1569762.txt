#include <iostream>
#include<cstdlib>
using namespace std;
int main()
{
	int i,n,a[100000];
	scanf("%d",&n);
	for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    int m=1;
    int cur=1;
    for(i=1;i<n;i++){
    	while(i<n && abs(a[i]-a[i-1])==1){
    	   cur++;
    	  // cout<<"i="<<i<<" ";
    	   i++;
    	}  
    	if(cur>m)
    		m=cur; 
    		//cout<<endl<<"cur="<<cur<<"i="<<i<<endl;
         cur=1;
    		
    	  
    }    
    printf("%d\n",m);
       
    return 0;
}
