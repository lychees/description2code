const ll mod = 1000000007;
typedef pair<int,int> pii;
int main(){
    //printf("Hello World..!\n");
    int t;
    cin>>t;
    while(t--){
        int n, x;
        cin>>n;
        int ans=0;
        map<int, int> mp;
        while(n--){
            cin>>x;
            if(mp[x]){
                mp[x]++;
            }
            else{
                mp[x]=1;
            }
            ans = max(mp[x], ans);
        }
        cout<<ans<<"\n";
    }
    return 0;

}
