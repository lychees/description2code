#include <bits/stdc++.h>
using namespace std;

int main(){
	int t; cin>>t;
	while(t--) {
		string x,y;
		cin>>x>>y;
		int cnt=0;
		int maxi=INT_MAX;
		if(x.size()==y.size()){
			for(int i=0;i<x.size();i++){
				if(x[i]!=y[i])
				cnt++;
			}
			maxi=cnt;
		}
		else{
			if(x.size()>y.size()){
				string tmp=x;
				x=y;
				y=tmp;
			}
			for(int i=0;i<=(y.size()-x.size());i++){
			    cnt=0;
				for(int j=0;j<x.size();j++){
					if(x[j]!=y[j+i]) ++cnt;
				}
				if(cnt<maxi)  maxi=cnt;
			}
		}
		cout<<maxi<<endl;
	}
    return 0;
}
