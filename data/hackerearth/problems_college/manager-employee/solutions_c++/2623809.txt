#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    int low=0;
    int high=n-1;
    int min=1000000;
    int max=-100000;
    while(low<high)
    {
    	if((a[high]+a[low])>max)
    	max=a[high]+a[low];
    	if((a[high]+a[low])<min)
    	min=a[high]+a[low];
    	high--;
    	low++;
    }
    cout<<max-min<<endl;
    }
    return 0;
}
