unsigned long long mm[1024][1024];

int numberOfPaths(int m, int n)
{
	for (int i = 0; i < m; i++)
		mm[i][0] = 1;

	for (int j = 0; j < n; j++)
		mm[0][j] = 1;

	for (int i = 1; i < m; i++)
	{
		for (int j = 1; j < n; j++)
			mm[i][j] = (mm[i - 1][j]% 1000000007 + mm[i][j - 1]% 1000000007 + mm[i-1][j-1]% 1000000007)% 1000000007;

	}
	return mm[m - 1][n - 1];
}

int main()
{
	int N, a, b, c, d;
	cin >> N;
	int ans = numberOfPaths(1024, 1024);
	while (N--)
	{
		cin >> a >> b >> c >> d;
		int ans = mm[c-a][d-b];
		cout << ans << endl;
	}
	
	
	return 0;
}
