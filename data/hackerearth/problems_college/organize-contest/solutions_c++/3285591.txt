#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	#define pii pair<int,int>

	scanf("%d", &t);
	map<string,pii >m;
	string str;
	int tmp;

	for(int i = 0; i< t; i++) {
		cin >> str>> tmp;
		if(m.find(str) != m.end()) {
			pii tt = m[str];
			tt.first += tmp;
			tt.second = i;
			m[str] = tt;
		}
		else {
			m[str] = make_pair(tmp , i);
		}
	}

	map<string,pii>:: iterator it = m.begin();
	string astr;
	int msc =-2000;
	int ind;
	while(it != m.end()) {
		string ts = it->first;
		pii tt = it->second;
		if(tt.first > msc || (tt.first == msc && tt.second < ind)) {
			msc = tt.first;
			astr = ts;
			ind = tt.second;
		}
		it++;
	}
	cout << astr << endl;
return 0;
}
