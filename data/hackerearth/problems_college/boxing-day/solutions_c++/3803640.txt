#include <iostream>
#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
    ll n,k,a,b,arr[100001]={0},i;
    cin>>n>>k;
    for(i=1;i<=n;i++)
    	arr[i]++;
    for(i=0;i<k;i++)
    {
    	cin>>a>>b;
    	arr[a]++;
    	arr[b]=0;
    }
    ll count=0;
    for(i=1;i<=n;i++)
    	if(arr[i]!=0){count++;}
    cout<<count<<endl;
    return 0;
}
