#include <bits/stdc++.h>
using namespace std;

int main() {
	string x,y;
	int t,cx1,cx0,cxq,cy1,cy0,cq1,i,count,test;
	cin>>t;
	for(test=1;test<=t;test++){
		cx1=cx0=cxq=cy1=cy0=cq1=count=0;
		cin>>x;
		cin>>y;
		//cout<<x<<" "<<y<<endl;
		//cout<<x.length()<<" "<<y.length()<<endl;
		if(x.length()!=y.length()){
			cout<<"Case "<<test<<": "<<-1<<endl;
		}else{
			for(i=0;i<x.length();i++){
				if(x[i]!=y[i]){
					if(x[i]=='1')
					cx1++;
					else if(x[i]=='?')
					cxq++;
					else
					cx0++;
					if(y[i]=='1')
					cy1++;
					else
					cy0++;
					if(x[i]=='?'&&y[i]=='1')
					cq1++;
				}
			}
			//cout<<cx0<<" "<<cx1<<" "<<cxq<<" "<<cy0<<" "<<cy1<<" "<<cq1<<" "<<count<<endl;
			if(cx1>cy1){
				cout<<"Case "<<test<<": "<<-1<<endl;
			}else{
				cxq-=cq1;
				cy1-=cq1;
				count+=cq1;
				cq1=0;
				//cout<<cx0<<" "<<cx1<<" "<<cxq<<" "<<cy0<<" "<<cy1<<" "<<cq1<<" "<<count<<endl;
				if(cy1>cx1){
					count+=(2*(cy1-cx1));
					//cy1=cx1;
					cq1-=(cy1-cx1);
					cx0-=(cy1-cx1);
					cx1-=(cy1-cx1);
					//cout<<cx0<<" "<<cx1<<" "<<cxq<<" "<<cy0<<" "<<cy1<<" "<<cq1<<" "<<count<<endl;
				}
				count+=cxq;
				cy0-=cxq;
				cxq=0;
				//cout<<cx0<<" "<<cx1<<" "<<cxq<<" "<<cy0<<" "<<cy1<<" "<<cq1<<" "<<count<<endl;
				count+=(cx1);
				//cout<<cx0<<" "<<cx1<<" "<<cxq<<" "<<cy0<<" "<<cy1<<" "<<cq1<<" "<<count<<endl;
				cout<<"Case "<<test<<": "<<count<<endl;
			}
		}
	}
	return 0;
}
