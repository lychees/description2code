#include <iostream>
using namespace std;
#define mod 1000000007

int power(int x, long long int y)
{
    long long int temp;
    if( y == 0)
        return 1;
    temp = power(x, y/2);
    if (y%2 == 0)
        return (temp*temp)%mod;
    else
        return (((x*temp)%mod)*temp)%mod;
}

int main()
{
   // cout << "Hello World!" << endl;
    int t;	cin >> t;
    while(t--) {
    	long long int n;	cin >> n;
    	int ans = power(2,n);
    	
    	cout << ans - 1 << endl;
    }
    return 0;
}
