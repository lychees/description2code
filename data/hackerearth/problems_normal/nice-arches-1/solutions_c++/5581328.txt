#include <iostream>
#include <stack>
using namespace std;

int main()
{
    int m;
    cin>>m;
    string s;
    
    int count=0;
    while(m--)
    {
    	cin>>s;
    	stack<char>word;
    	int len=s.length();
    	for(int i=0;i<len;i++)
    	{
    		if(word.empty())
    		{
    			word.push(s[i]);
    		}
    		else
    		{
    			if(word.top()==s[i])
    			{
    				word.pop();
    			}
    			else
    			word.push(s[i]);
    		}
    	}
    	if(word.empty())
    	count++;
    	//word.clear();
    }
    cout<<count<<endl;
    return 0;
}
