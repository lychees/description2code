VI p;
int sito[100], wynik[100005], tab[100];;
void prepro()
{
	int N = 50;
	FOR(i, 2, N)
	{
		if(!sito[i])
		{
			p.PB(i);
			for(int j = i; j <= N; j += i)
				sito[j] = i;
		}
	}
}

void jebaj()
{
	int n;
	cin >> n;
	REP(i, n)
	{
		int a;
		cin >> a;
		int maska = 0;
		REP(j, 15)
			if(a % p[j] == 0)
				maska += (1 << j);
		tab[i] = maska;
	}
	REP(i, n)
	{
		REP(j, (1 << 15))
		{
			if(!(tab[i] & j))
				wynik[j ^ tab[i]] = max(wynik[j ^ tab[i]], wynik[j] + 1);
		}
	}
	int res = 0;
	REP(j, (1 << 15))
	{
		res = max(res, wynik[j]);
		wynik[j] = 0;
	}
	cout << res << "\n";
}
