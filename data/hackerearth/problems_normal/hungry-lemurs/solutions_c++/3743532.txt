#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   int n,k;
    cin>>n>>k;
    int ans=10000000;
    for(int i=1;i<=100000;i++)
    {
    	for(int j=0;j<=100000;j+=i)
    	{
    		if(j%i==0)
    		{
    			ans=min(ans,abs(n-j)+abs(k-i));
    		}
    	}
    }
    cout<<ans<<endl;
    return 0;
}
