#include <iostream>
using namespace std;
int nums[] = {1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1};
string x[] = {"G", "EG", "F", "EF", "E", "CE", "D", "CD", "C", "AC", "B", "AB", "A"};

int findIndex(char a) {
	for(int i=0;i<13;i+=2) {
		
		if(x[i][0] == a) return i;
	}
	return -1;
}
long toNum(string roman) {
	long res = 0;
	//res = nums[findIndex(roman[0])];
	int len = roman.length();
	int value[len];
	for(int i=0;i<len;i++) {
		value[i] = nums[findIndex(roman[i])];
	}
	for(int i=0;i<len;i++) {
		
		if(i+1 < len && value[i+1] > value[i]) {
			res += value[i+1] - value[i];
			i++;
		}
		else {
			res += value[i];
		}
	}
	return res;
}
 string ToRoman(int number)
    {
        string value = "";
        for (int i = 0; i < 13 && number != 0; i++)
        {
            while (number >= nums[i])
            {
                number -= nums[i];
                value += x[i];
            }
        }
        return value;
    }
int main()
{
    int t;
    cin>>t;
    string s;
    while(t--) {
    	s.clear();
    	cin>>s;
		long num = toNum(s);  
		cout<<ToRoman(num)<<endl;
    }
    return 0;
}
