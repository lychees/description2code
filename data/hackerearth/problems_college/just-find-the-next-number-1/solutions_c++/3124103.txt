#include<string.h>
using namespace std;

int check(int a)
{
    long int i,j,temp=0;
    for(j=a;j<=100000;j++)
    {
        char c[50]="n";
        int i=j;
            while(i>0)
            {
                if(i%2==0)
                {
                    strcat(c,"0");

                }
                else{

                    strcat(c,"1");
                }
                i=i/2;

           }
           for(int k=1;k<strlen(c)-1;k++)
           {
               if(c[k]=='1'&&c[k+1]=='1')
               {
                 goto l;
               }

           }
           return j;
           l:;

    }
}

main()
