#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;
int main()
{

int D, G, M,i,j,temp, sum;
vector<int> group;
cin>>D;

	for( i=0; i<D; i++)
		{
			sum = 0;
			cin>>G>>M;
			for( j=0; j<G; j++)
			{
				cin>>temp;
				group.push_back(temp);
			}
			sort(group.begin(),group.end());
			j = 0;
			while( (sum += group[j++]) <= M);
			cout<<j-1<<"\n";
			group.clear();
		}
return 0;
}
