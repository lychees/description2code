#define MOD 4294967296
#define pb push_back
#define mp make_pair

typedef long long int ll;

using namespace std;

int pos[100001];

int main()
{
    ios::sync_with_stdio(0);
    ll n,x,i,t;
    cin>>t;
    while(t--) {
        cin>>n;
        vector <ll> v;
        v.pb(0);
        int temp = 0;
        memset(pos,0,sizeof(pos));
        for(i=1;i<=n;i++) {
            cin>>x;
            v.pb(x);
            pos[x] = i;
        }
        for(i=1;i<=n;i++) {
            if(pos[v[i]] != v[v[i]]) {
                temp=1;
                break;
            }
        }
        if(temp == 1) {
            cout<<"not inverse\n";
        } else cout<<"inverse\n";
    }
    return 0;
}
