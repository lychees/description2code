import fileinput

if __name__ == "__main__":
    line = fileinput.input().readline().strip()
    X = 0
    Y = 0
    count = 0
    s = set()
    s.add(0)
    for c in line:
        if c == 'L':
            Y = Y-1
        elif c == 'R':
            Y = Y+1
        elif c == 'U':
            X = X-1
        elif c == 'D':
            X = X+1
        value = X*1000000 + Y
        if value in s:
            count = count+1
        s.add(value)
    print count
