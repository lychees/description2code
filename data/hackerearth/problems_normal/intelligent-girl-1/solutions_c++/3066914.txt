# include<bits/stdc++.h>
using namespace std;
int main()
{
    string s;
    cin>>s;

    int j,len=s.length();
    for(int i=0;i<len;i++)
    {
        int c=0;
        for(j=i;j<len;j++)
        {   if(s[j]%2==0)
                c++;
        }
        cout<<" "<<c;
    }
    return 0;
}
