#include <iostream>
using namespace std;

int main()
{
    int  test;
    cin>>test;
    while(test--)
    {
       int size,i;
   	   cin>>size; 
       int a[size],sum=0;
       for(i=0;i<size;i++)
       {
           cin>>a[i];
           sum+=a[i];
       }
       int prev=0;
       for(i=0;i<size;i++)
       {
           sum=sum-a[i];
           if(prev==sum)
           {
              cout<<"YES"<<endl;
              break;
           } 
           prev+=a[i];                    
       }
       if(i==size)
           cout<<"NO"<<endl;          
    }
    return 0;
}
