#include <iostream>
#include <queue>
#include <math.h>
using namespace std;

int main()
{
	queue<int> win;
	int n,temp,winner=0,player1,player2;
	cin>>n;
	int players = (int)pow(2,n);
	int matrixLength = players -1;
	vector<int> strength[matrixLength];
	for(int i=0;i<matrixLength;i++)
	{
		for(int j=0;j<=i;j++)
		{
			cin>>temp;
			strength[i].push_back(temp);
		}
	}
	
	for(int i=1;i<players;i=i+2)
	{
		player1 = i;
		player2 = i+1;
		if(strength[player2-2][player1-1] == 1)
			win.push(player2);
		else
			win.push(player1);
	}

	while(win.size()!=1)
	{
		player1=win.front();
		win.pop();
		player2=win.front();
		win.pop();
		
		if(strength[player2-2][player1-1] == 1)
			win.push(player2);
		else
			win.push(player1);
	}
	winner = win.front();
	win.pop();
	cout<<winner<<endl;
    return 0;
}
