#include<iostream>
#include<stdio.h>
#define DIV 1000000007
 
using namespace std;
 
typedef unsigned long long int Lint;
 
int main() {
	
	Lint t, a, b, c, small1, small2, small3, pro1, pro2, pro3;
 
	scanf("%llu", &t );
 
	while( t-- ) {
	
		scanf("%llu %llu %llu", &a, &b, &c );
 
		if( ( a <= b ) && ( a <= c ) ) {
 
			small1 = a;
 
			if( b <= c ) {
		
				small2 = b;
				small3 = c;			
			}
			else {
		
				small2 = c;
				small3 = b;			
			}
		}
 
		else if( ( b <= a ) && ( b <= c ) ) {
 
			small1 = b;
 
			if( a <= c ) {
		
				small2 = a;
				small3 = c;			
			}
			else {
		
				small2 = c;
				small3 = a;			
			}
		}
 
		else if( ( c <= a ) && ( c <= b ) ) {
 
			small1 = c;
 
			if( a <= b ) {
		
				small2 = a;
				small3 = b;			
			}
			else {
		
				small2 = b;
				small3 = a;			
			}
		}
 
		pro1 = ( ( small1  % DIV ) * ( ( small2 - 1 )  % DIV ) ) % DIV;
		pro2 = ( ( pro1  % DIV ) * ( ( small3 - 2 )  % DIV ) ) % DIV;
		
		printf("%llu\n"  , pro2 );
	}
	
	return 0;
}
