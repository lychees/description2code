#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
        int t;
        scanf("%d",&t); 
        long long int arr[5];
        while(t--)
        {
                scanf("%lld%lld%lld",&arr[0],&arr[1],&arr[2]);
                sort(arr,arr+3);
                long double ans=((long double)(arr[0]))/2;
                printf("%.1llf\n",ans);
        }
        return 0;
}
