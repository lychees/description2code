#include <iostream>
using namespace std;
int roundUp(int n, int b){
        if(n % b >= b/2.0){
            return n/b*b+b;
        }
        return n/b*b;
    }
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,b;
    	cin>>n>>b;
    	cout<<roundUp(n,b)<<endl;
    }
    return 0;
}
