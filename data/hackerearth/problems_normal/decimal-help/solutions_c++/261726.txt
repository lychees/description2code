#include <iostream>
#include<math.h>
using namespace std;

int calc(long int N,long int D,long int R)
{
	int flag=0;
	long int temp;
	temp=0;
	if(N<D)
		N=N%D;
	while(R)
	{
	
		N=N*10;
		while(N<D&&R)
		{
			N=N*10;
			temp=0;
			R--;
			if(R==0)
				flag=1;
		}
		if(flag)
			break;
		temp=N/D;
		N=N%D;
		R--;
	}
	return temp;
}


int main()
{
    int T;
    long int R;
    long int N,D;
    int i;
    cin>>T;
    for(i=0;i<T;i++)
    {
    	cin>>N>>D>>R;
    	cout<<calc(N,D,R)<<"\n";

    }
    return 0;
}
