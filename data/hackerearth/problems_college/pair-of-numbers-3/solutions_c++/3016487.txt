#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int n;
    while(t--)
    {
    	cin>>n;
    	int count=0;
    	for(int i=1;i<=n;i++)
    	{
    		for(int j=1;j<=i;j++)
    		{
    			int mul=i*j;
    			int sum=i+j;
    			//cout<<mul<<" "<<sum<<"\n";
    			if(mul%sum==0&&j<i)
    			 count++;
    		}
    	}
    	
    	cout<<count<<"\n";
    }
    return 0;
}
