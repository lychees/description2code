def mod(x):
	if(x<0):
		return -x
	return x
t=input()
for _ in range(0,t):
	a=raw_input()
	b=a[::-1]
	k=0
	for i in range(1,len(a)):
		if(mod(ord(a[i])-ord(a[i-1]))!=mod(ord(b[i])-ord(b[i-1]))):
			k=1
			print "Not Funny"
			break
	if(k==0):
		print "Funny"
