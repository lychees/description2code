#include<iostream>
#include<cstdio>
using namespace std;
int t,l;
int main(){
	cin>>t;
	while(t--){
		int i;
		cin>>l;
		char ch[1000001];
		scanf("%s",ch);
		char ch1[1000001],ch2[1000001];
		int f=0;
		ch1[0]='Y';
		for(i=1;i<l;i++){
			if(f==0)
			{
				ch1[i]='N';
				f=1;
			}
			else {
				ch1[i]='Y';
				f=0;
			}
		}
		f=0;
		ch2[0]='N';
		for(i=1;i<l;i++){
			if(f==0)
			{
				ch2[i]='Y';
				f=1;
			}
			else {
				ch2[i]='N';
				f=0;
			}
		}
		
		int ans1=0;
		for(i=0;i<l;i++){
			if(ch[i]!=ch1[i]) ans1++;}
		
		int ans2=0;
		for(i=0;i<l;i++){
			if(ch[i]!=ch2[i]) ans2++;}
		
		int ans=0;
		ans=((ans1>ans2)?ans2:ans1);
		cout<<ans<<endl;
		}
		return 0;
	}
