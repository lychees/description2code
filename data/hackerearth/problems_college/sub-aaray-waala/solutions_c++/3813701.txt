#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
long long big_power(long long n,long long p);
int main()
{
   int t;
   std::ios::sync_with_stdio(false);
   cin>>t;
   while(t--)
   {
      long long n;
      cin>>n;
      long long i,j;
      long long arr[n];
      for(i=0;i<n;i++)
        cin>>arr[i];
      long long ans=0;
      map<long,long> m;
      for(i=0;i<n;i++)
      {
          m[arr[i]]=1;
      }
      ans=big_power(n,m.size());
      cout<<ans<<"\n";
   }
   return 0;
}
long long big_power(long long n,long long p)
{
    long long tmp;
    if(p==0)
        return 1;
    tmp=big_power(n,p/2)%mod;
    tmp=(tmp*tmp)%mod;
    if(p%2!=0)
        return (tmp*n)%mod;
    else
        return tmp;
}
