#include <iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;

int main()
{    int n,i,a[1000000],b[1000000],r,l,mid,diff,max;
    scanf("%d",&n);
    for(i=0;i<n;i++){
    scanf("%d",&a[i]);	
    }
    std::sort(a,a+n);
    mid=n/2;
    b[mid]=a[0];
    r=1;
    l=1;
    for(i=1;i<n;i++){
    	if(i%2==0)
    	{b[mid+r]=a[i];r++;}
    	else 
    	{b[mid-l]=a[i];l++;}
    }max=0;
   for(i=1;i<n;i++){
   	diff=b[i]-b[i-1];
   	if(diff<0)diff=diff*(-1);
   	if(max<diff)
   	max=diff;
   }
   diff=b[0]-b[n-1];
   if(diff<0)diff=diff*(-1);
   if(max<diff)max=diff;
   printf("%d",max);
    
    return 0;
}
