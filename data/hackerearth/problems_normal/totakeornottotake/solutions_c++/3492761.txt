#include<bits/stdc++.h>
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
 
long long take(long long a,char b,long long c){
if(b=='N')
        return -a;
else if(b=='+')
    return a+c;
else if(b=='/')
    return a/c;
else if(b=='-')
    return a-c;
else if(b=='*')
    return a*c;
//else
   // assert(false);
}
 
int main()
{
int test,t;
char b;
long long maxi,mini,c;
cin >> test;
while(test--){
    cin >> t;
    vector<long long> ar;
    maxi=mini=1;
    while(t--){
        cin >> b;
        if(b!='N'){
                cin >> c;
        }
        ar.push_back(mini);
        ar.push_back(maxi);
       ar.push_back(take(maxi,b,c));
       ar.push_back(take(mini,b,c));
       sort(ar.begin(),ar.end());
       mini=ar[0];
       maxi=ar.back();
       long long M = 1000 * 1000 * 1000;
		M *= M;
		assert(max(abs(maxi), abs(mini)) <= M);
    }
    cout << maxi << endl;
}
    return 0;
}
