#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		uint32_t n,a[500000];
		double s = 0;
		cin>>n;
		
		for(uint32_t i = 0;i < n;i++)
		{
			cin>>a[i];
		}
		for(uint32_t i = 0;i < n;i++)
		{
			s += a[i];
		}
		cout<<setprecision(8)<<fixed<<(s/n)<<endl;
	}
    return 0;
}
