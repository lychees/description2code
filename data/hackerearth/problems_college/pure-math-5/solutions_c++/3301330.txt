    #include<iostream>
    
    using namespace std;
     
    int main()
    {
        unsigned long int n,m,k,count,i,j,t,s;
        cin>>t;
        while(t--)
        {
            count=0;
            cin>>k;
            for(j=2;j<=k;j++)
            {
                i=1;
                if(((j*j)-1)%i==0&&((i*i)-1)%j==0)
                {
                    count++;
                    n=j;
                    m=((j*j) -1)/i;
                    while(m<=k)
                    {
                        count++;
                        s=m;
                        m=((s*s)-1)/n;
                        n=s;
                    }
                }
            }
            cout<<count<<"\n";
        }
        return 0;
    }
