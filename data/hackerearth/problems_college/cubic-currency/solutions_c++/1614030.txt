#include<bits/stdc++.h>
using namespace std;
long long memo[10000+10][23];
int A[22];
void k()
{
    for(int i=1;i<=22;i++)
        A[i]=i*i*i;
}
long long recurese(int val,int i)
{
     if(i==22)
     {
         if(val==0)
            return 1;
         else
            return 0;
     }
    if(val-A[i]==0)  // including this is a bliss :D
        return 1;
    if(val-A[i]>0) // can be included
     { if(memo[val-A[i]][i]==-1)
        memo[val-A[i]][i]=recurese(val-A[i],i);
        if(memo[val][i+1]==-1)
       memo[val][i+1]= recurese(val,i+1);
       return memo[val-A[i]][i]+memo[val][i+1];
     }
    else
          if(memo[val][i+1]==-1)
       memo[val][i+1]= recurese(val,i+1);
       return memo[val][i+1];
}
int main()
{ //freopen("tomar.txt","w",stdout);
     k();
  int t,n;
  for(int i=0;i<=10000;i++)
  {
      for(int j=0;j<=22;j++)
        memo[i][j]=-1;
  }
  scanf("%d",&t);
  while(t--)
  {
      scanf("%d",&n);
      printf("%lld\n",recurese(n,1));
  }
    return 0;
}
