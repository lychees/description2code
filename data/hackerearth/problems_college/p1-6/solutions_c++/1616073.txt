#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
using namespace std;
long long arr[1000000];
char str[10000001];
int main()
{
    int i,len;
    gets(str);
    if(str[0]=='1')
        arr[0]=1;
        len=strlen(str);
       for(i=1;i<len;i++)
        if(str[i]=='1')
            arr[i]=arr[i-1]+1;
            else
                arr[i]=arr[i-1];
    long long p=0;
    for(i=0;i<len;i++)
        if(str[i]=='0')
            p=p+(arr[i]*(arr[len-1]-arr[i]));

    printf("%lld\n",p);
}
