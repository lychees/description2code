#include <iostream>
#define mod 1000000007
#define ll long long 
using namespace std;
ll modPow(ll a,ll b,ll MOD)
{
ll res=1;
 a%=MOD;
 for(;b;b>>=1)
    {
        if(b&1)res=res*a%MOD;
          a=a*a%MOD;
    }
 return res;
}
ll modInverse(ll a){
    return modPow(a,mod-2,mod);
    }

int main()
{
	long long int t;
	cin>>t;
	while(t--)
	{
		long long int k,m,i,sum=0,x=0,y=0,p=1,q=1,ans=0,left=0;
		cin>>k>>m;
		long long int a[k];
		for(i=0;i<k;i++)
		{
			cin>>a[i];
			sum+=a[i];
		}
	//	cout<<sum<<endl;
		left=m-sum;
	//	cout<<left<<endl;
		if(sum> m)
        cout<<0<<endl;
		else
		{
			x=k+left-1;
			y=k-1;
			for(i=x;i>y;i--)
			{
				p=((p%mod)*(i%mod))%mod;
			}
	//		cout<<p<<endl;
			for(i=1;i<=left;i++)
			{
				q=((q%mod)*(i%mod))%mod;
			}
	//		cout<<q<<endl;
			ans=(p*modInverse(q))%mod;
			cout<<ans<<endl;
		}
	}
    return 0;
}
