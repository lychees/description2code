#include <iostream>
using namespace std;
#define ll long long
#include <algorithm>
#include <vector>

int main()
{	ll n,m,q,i;
	cin>>n>>m;
	vector<ll> x;
	for(i=0;i<2*n;++i)
	{	cin>>q;
		x.push_back(q);
	}
	sort(x.begin(),x.end());
	q=0;
	ll s=0,k=0;
	for(i=0;i<2*n;++i)
	{	s+=x[i];	
		++q;
		if(s>m)
		{	if(q-1>n)
				cout<<n;
			else
				cout<<q-1;
			k=1;
			break;
		}
	}
	if(!k)
	{	if(q>n)
			cout<<n;
		else
			cout<<q;
	}
    return 0;
}
