test_cases = int(raw_input(""))
result = []
for i in range(test_cases):
    item_number = int(raw_input(""))
    items = raw_input("").split(" ")[:item_number]
    item_number = float(item_number)
    items = map(float, items)
    total = sum(items)
    percentage = format(round(total/item_number, 8), '0.8f')
    result.append(percentage)

for each in result:
    print each
