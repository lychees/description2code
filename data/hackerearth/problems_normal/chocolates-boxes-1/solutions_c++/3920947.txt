#include <iostream>
#include<bits/stdc++.h>
#include<math.h>
using namespace std;


 

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
		//cout<<n;
		int map[200];
		memset(map,0,sizeof(map));
		priority_queue<long int>P;
    	int a[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	int k;
    	cin>>k;
    	for(int i=0;i<n;i++)
    	{
		    
    		if(a[i]>k) continue;
			map[a[i]]++;
			if(map[a[i]]==1)
    		P.push(a[i]);
    	}
    	
    	int x=0;
		int arr[100000];
		
		while(!P.empty())
		{
		 arr[x++]=P.top();
		 P.pop();
		}

	
		int ans=0;
		for(int i=0;i<x;i++)
		{
		 for(int y=x-1;y>=i;y--)
		 {
		  if((arr[i]+arr[y])>k) break;
		 // cout<<arr[i]<<" "<<arr[y]<<endl;
		  if((arr[i]+arr[y])==k)
		  {
		    //cout<<"sex"<<endl;
		    if(i!=y)
			{
			
			  ans=ans+(map[arr[i]]*map[arr[y]]);
			}
			else
			{
			  ans=ans+(map[arr[i]]*(map[arr[i]]-1))/2;
			}
		  }
		 }
      }
	cout<<ans<<endl;
		
    
}	
	
}
