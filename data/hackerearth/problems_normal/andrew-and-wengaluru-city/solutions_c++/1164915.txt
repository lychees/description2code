#include <stdio.h>
#include <string.h>
#include<iostream>
using namespace std;
int main(void) {
    int t,i,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        long long ar[n+1],m_a[n+1],m_b[n+1],a=0;
        for(i=0;i<n;i++)
        scanf("%lld",&ar[i]);
        for(i=0;i<n;i++)
        {
            if(i==0)
            m_a[i]=ar[i];
            else
            m_a[i]=max(m_a[i-1],ar[i]);
            //printf("%d ",m_a[i]);
        }
        for(i=n-1;i>=0;i--)
        {
            if(i==n-1)
            m_b[i]=ar[i];
            else
            m_b[i]=max(ar[i],m_b[i+1]);
            //printf("%d ",m_b[i]);
        }
        for(i=0;i<n;i++)
        {
            a=(a+(min(m_a[i],m_b[i])-ar[i]))%1000000007;
        }
        printf("%lld\n",a);
    }

}
