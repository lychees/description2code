def comp(a):
    t = 0
    res = 0
    for i in range(1, len(a)):
        t += i * (a[i] - a[i-1])
        res += t
    return res


X, Y = [], []
for _ in range(int(raw_input())):
    x, y = map(int, raw_input().split())
    X.append(x)
    Y.append(y)
X.sort()
Y.sort()
print(comp(X) + comp(Y)) % 1000000007
