#include <iostream>
#include<algorithm>
#include<stdlib.h>
using namespace std;

int main()
{
	int M, N,t;
	int A[100000];

	cin>>t;
	while(t--)
	{
		long int cost=0;
		cin>>N;
		for(int i=0; i<N; i++)
		{
			cin>>A[i];
		}
		sort(A, A+N);
		int indx = N/2;
		if(N%2)
			M = A[indx];
		else
			M = (A[indx] + A[indx-1])/2;
		
		//computing the cost
		for(int i=0; i<N; i++)
			cost+= abs(A[i]- M);

		cout<<cost<<"\n";
	}
	return 0;
}
