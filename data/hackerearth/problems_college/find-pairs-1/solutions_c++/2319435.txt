#include <bits/stdc++.h>
 
using namespace std;
 
int main()
{
long long int t;
cin>>t;
while(t--){
    long long int n,c=0;
    cin>>n;
    long long int a[n];
    for(int i=0;i<n;i++)
        cin>>a[i];
    for(int i=0;i<n;i++)
    for(int j=i;j<n;j++)
    if(a[i]==a[j])
      c++;
    cout<<c<<endl;
        
}
    
    return 0;
}
