#include<bits/stdc++.h>
using namespace std;
long long  o , e;
void solve(long long n , int step )
{
         if(n&1)
          o+=n;
          else
          e+=n;
     if( n == 1)
     {
       if(e%step < o%step)
       cout <<"Odd Rules"<<endl;
       else
       if(e%step==o%step)
       cout <<"Tie"<<endl;
       else
       cout <<"Even Rules"<<endl;
       return ;
      }
      else

      if(n%2==0)
      solve(n/2,step+1);
      else
      solve(3*n+1,step+1);


}
int main(){


   int t;
   cin >> t;
   while(t--)
   {

         long long n;
         cin >> n;
         o = 0;
         e = 0;
         solve(n , 0 );


   }

}
