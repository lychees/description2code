#include <bits/stdc++.h>
void solution(int n)
{
    int a[100],i,j,w1=0,temp;
    char w[260];
        w1=0;
        if(n==0||n==1)
                std::cout<<"1"<<std::endl;
        else {
            a[0]=1;
        for(i=1;i<n;i++)
                a[i]=0;
        for(j=0;j<n;j++)
        {
            while(a[j]>0)
            {
                temp=a[j]%10;
                for(i=j+1;i<n;i++)
                {
                            a[i]=(i+1)*temp+a[i];
                            temp=a[i]%10;
                        }
                w[w1++]=a[n-1]%10+48;
                for(i=j;i<n;i++)
                            a[i]=a[i]/10;
            }
        }
        unsigned long int ans = 0;
        for(i=w1-1;i>=0;i--)
                ans+=w[i]-'0';
        std::cout<<ans<<std::endl;
    }

}

int main(int argc, char const *argv[])
{
    int t,n;
    std::cin>>t;
    while(t--)
    {
        std::cin>>n;
        solution(n);
    }
    return 0;
}
