#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
#define LL long long int
LL modpow(LL base, LL exp) {
  base %= MOD;
  LL result = 1;
  while (exp > 0) {
    if (exp & 1) result = (result * base) % MOD;
    base = (base * base) % MOD;
    exp >>= 1;
  }
  return result;
}
int main()
{
    LL t,n,ans;
    scanf("%lld",&t);
    while(t--){
    	scanf("%lld",&n);
    	ans=(3*(((modpow(3,n-1)-modpow(2,n)+1)%MOD+MOD)%MOD))%MOD;
    	printf("%lld\n",ans);
    }
    return 0;
}
