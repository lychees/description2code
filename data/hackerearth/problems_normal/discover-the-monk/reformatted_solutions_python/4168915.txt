def bsearch(A, x, l, r):
    while l <= r:
        mid = (l + r) / 2
        if A[mid] == x:
            return mid
        elif A[mid] < x:
            l = mid + 1
        else:
            r = mid - 1
    return -1


n, r = map(int, raw_input().split())
# print n, r
A = sorted(map(int, raw_input().split()))
# print A
for i in xrange(r):
    x = int(raw_input())
    if bsearch(A, x, 0, len(A)-1) != -1:
        print "YES"
    else:
        print "NO"
