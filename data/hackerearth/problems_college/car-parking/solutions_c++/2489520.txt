#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	int len=str.length();
    	vector<int>a;
    	a.resize(len);
    	//memset(a,0,sizeof(a));
       	int count=0;
    	for(int i=0;i<len;i++)
    	{
    		if(str[i]=='B')
    		{
    			a[i]=1;
    			if(i-1>=0)
    			a[i-1]=1;
    			if(i-2>=0)
    			a[i-2]=1;
    		}
    		else if(str[i]=='S')
    		{
    			a[i]=1;
    			if(i-1>=0)
    			a[i-1]=1;
    			if(i<len-1)
    			a[i+1]=1;
    		}
    		else if(str[i]=='D')
    		{
    			a[i]=1;
    		}
    		else
    		{
    			continue;
    		}
    	}
    	for(int i=0;i<len;i++)
    	{
    		if(a[i]==0)
    		{
    			count++;
    		}
    	}
    	
    	cout<<count<<endl;
    	
    }
}
