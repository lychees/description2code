#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long t,n,a,sum;
    scanf("%lld",&t);
    while(t--){
        vector<long long> arr;
        sum=0;
        scanf("%lld",&n);
        for(int i=0;i<n;i++){
            scanf("%lld",&a);
            sum+=a;
        }
        //cout<<sum<<endl;
        for(int i=0;i<n;i++){
            scanf("%lld",&a);
            arr.push_back(a);
        }
        sort(arr.begin(),arr.end());
        reverse(arr.begin(),arr.end());
        long long count=0;
        int flag=0;
        for(int i=0;i<arr.size();i++){
            //cout<<sum<<" "<<i<<endl;
            if(sum<=0){
                printf("%lld\n",count);
                flag=1;
                break;
            }
            else{
                sum=sum-arr[i];
            }
        count++;
        }
        if(flag==0)printf("%lld\n",n);
    }
    return 0;
}
