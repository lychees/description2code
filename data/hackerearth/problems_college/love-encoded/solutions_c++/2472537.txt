#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int test;
	cin>>test;
	string s,t;
	while(test--)
	{
		int n;
		cin>>n;
		cin>>s;
		t=s;
		int i,l=s.length();
		char c;
		for(i=0;i<l;i++)
		{
			c=s.at(i)-'A';
			t.at(i)=('A'+(c-n+26)%26);
		}
		cout<<t<<endl;
	}
	return 0;
}
