#include <bits/stdc++.h>

#define s(x) scanf("%d",&x);
#define sl(x) scanf("%lld",&x);
#define ll long long
#define ull long long
#define pii pair <int, int>
#define vi vector <int>
#define pb push_back
#define mp make_pair

using namespace std;

char q[100];
ll dp[100][5];

void ff(ll n)
{
	stringstream ss;
	ss << n;
	ss >> q;
}
ll fun (ll l, ll flag)
{
	if (l == 0 && flag == 1) return 1;
	if (l == 0) return 0;
	if (dp[l][flag] != -1) {
		return dp[l][flag];
	}
	ll ans = 0;
	for (ll i = 0; i <= 9; i++) {
		if (i == 2 || i == 3 || i == 5 || i == 7) {
			ans += fun (l - 1, 1);
		} else {
			ans += fun (l - 1, flag);
		}
 	}
	dp[l][flag] = ans;
	return ans;
}
int main()
{
	memset (dp, -1, sizeof dp);
	ll t; sl(t);
	while (t--) {
		ll n;
		sl(n);
		ff(n);
		ll l = strlen(q);
		ll ans = 0;
		ll flag = 0;
		for (ll i = 0; i < l; i++) {
			int cur = q[i] - '0';
			
			for (ll j = 0; j < cur; j++) {
				if (j == 2 || j == 3 || j == 5 || j == 7) {
					ans += fun (l - i - 1, 1);	
				} else {
					ans += fun (l - i - 1, flag);
				}
			}
			if (cur == 2 || cur == 3 || cur == 5 || cur == 7) {
				flag = 1;
			}
		}
		printf ("%lld\n", ans + flag);
		
		//cout << ans + flag << endl;
	}
	return 0;
}
