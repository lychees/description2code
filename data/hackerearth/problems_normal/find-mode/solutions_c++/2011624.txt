#include <bits/stdc++.h>
using namespace std;
#define MAX 1000001
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long int n,i,temp,max=0;
    	scanf("%ld",&n);
    	int flag[MAX]={0};
    	priority_queue<long int>mypq;
    	for(i=0;i<n;i++)
    	{
    		scanf("%ld",&temp);
    		flag[temp]++;
    		if(max<flag[temp])
    		{
    			max=flag[temp];
    		}
    	}
    	for(i=0;i<MAX;i++)
    	{
    		if(max==flag[i])
    		{
    			mypq.push(i);
    		}
      	}
       while (!mypq.empty())
  		{
    		cout<< mypq.top()<<" ";
    		 mypq.pop();
 		}
 
      	cout<<endl;
    	
    }
}
