#include <iostream>
#include<cstring>

using namespace std;

int main()
{
    int tno,i;
    char A[100];
    char B[100];
    char C[100];
    
    cin>>tno;
    
    while(tno--)
    {
    	cin>>A;
    	cin>>B;
    	
    	
    	for(i=0;i<strlen(A);i++)
    	{

    		if(A[i]-B[i]==0  )
    			C[i]='0';
    		else
    			C[i]='1';
    	}
    	C[i]='\0';
    		
    	
    	printf("%s\n",C);
    }
    return 0;
}
