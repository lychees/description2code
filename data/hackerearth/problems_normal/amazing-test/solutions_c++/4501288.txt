#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n,x;
    	long int sum=0;
    	cin>>n>>x;
    	int a[n+1];
    	for(int i=1;i<=n;i++)
    	cin>>a[i],sum=sum+a[i];
    	if(sum<=x)
    	cout<<"YES\n";
    	else
    	{
    		int b[2][x+1];
    		for(int i=0;i<=x;i++)
    		b[0][i]=0;
    		
    		for(int i=0;i<2;i++)
    		b[i][0]=0;
    		
    		for(int i=1;i<=n;i++)
    		{
    			for(int j=1;j<=x;j++)
    			{
    				if(j-a[i]>=0)
    				b[i%2][j]=max(b[(i-1)%2][j],b[(i-1)%2][j-a[i]]+a[i]);
    				else
    				b[i%2][j]=b[(i-1)%2][j];
    			}
    		}
    		
    		/*for(int i=1;i<=n;i++)
    		{
    			for(int j=1;j<=x;j++)
    				cout<<b[i%2][j]<<" ";
    				cout<<"\n";
    		}
    		cout<<sum<<" "<<b[n%2][x]<<" "<<x<<"\n";*/
    		if(sum-b[n%2][x]<=x)
    		cout<<"YES\n";
    		else
    		cout<<"NO\n";
    	}
    	
    }
    return 0;
}
