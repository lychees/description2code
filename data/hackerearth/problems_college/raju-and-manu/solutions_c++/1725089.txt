#include<iostream>
#include<bitset>
using namespace std;

      int arr[1000001];
int main()
{
      int mx = 0;
      int tmp = 0;
      for(int i = 1; i <= 100000 ; ++i )
      {
            bitset<32> bs(i);
              int val = bs.count();
              if(val > mx)
                {
                mx = val;
                tmp = i;
              }
              else
                if(val == mx)
                tmp = i;

              arr[i] = tmp;
      }
      int t;
      cin >> t;
      while(t--)
      {
           int n;
           cin >> n;
           cout << arr[n] << endl;
      }
       return 0;
}
