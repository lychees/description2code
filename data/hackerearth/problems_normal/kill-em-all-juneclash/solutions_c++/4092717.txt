typedef long long ll;

const int inf=1e9;
const int MOD=1e9+7;
const int N=1e6+10;
const int ans[9]={0,1,1,2,3,4,7,11,21};

ll power(ll x,ll y){
    ll ans=1;
    while (y){
        if (y&1) ans=ans*x%MOD;
        x=x*x%MOD;
        y>>=1;
    }
    return ans;
}

int solve(int n){
    if (n<=6) return ans[n];
    ll ans=1;
    ans+=power(2,(n-1)/2-2);
    ans+=power(2,n-4);
    if (!(n&1)) ans+=power(2,n/2-3);
    return ans%MOD;
}

    while (test--){
        scanf("%d",&n);
        printf("%d\n",solve(n));
    }
}
