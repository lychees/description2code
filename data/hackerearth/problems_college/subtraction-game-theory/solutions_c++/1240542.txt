#include <iostream>
using namespace std;

int sod(long long int n)
{
		int s=0;
		while(n)
		{
				int d=n%10;
				s+=d;
				n/=10;
		}
		return s;
}
int main()
{
    int n;
    std::cin>>n;
    int a,sum1=0;long long int sum2=0;
    while(n)
    {		
    		cin >> a;
    		sum2+=a;
    		sum1+=sod(a);
    		n--;
    }
    cout<< sum1%9-sod(sum2)%9;
	return 0;
}
