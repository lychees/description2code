def isBotMovedRight(bot, initial_state, final_state):
	return isBotMovedLeft(bot, initial_state[::-1], final_state[::-1])

def isBotMovedLeft(bot, initial_state, final_state):
	assert(len(initial_state) == len(final_state))
	count_bot_initial_state = 0
	count_bot_final_state = 0
	for ichar, fchar in zip(initial_state, final_state):
		if ichar == bot:
			count_bot_initial_state += 1
		if fchar == bot:
			count_bot_final_state += 1
		if count_bot_initial_state < count_bot_final_state:
			return True
	return False

def isStateVaild(initial_state, final_state):
	if len(initial_state) != len(final_state):
		return False
	if initial_state.replace('#','') != final_state.replace('#',''):
		return False
	if isBotMovedLeft('B', initial_state, final_state):
		return False
	if isBotMovedRight('A', initial_state, final_state):
		return False
	return True

no_of_testcases = raw_input()

for _ in range(0, int(no_of_testcases)):
	states = raw_input().split(' ')
	assert(len(states) == 2)
	state_valid = isStateVaild(states[0], states[1]) 
	print "Yes" if state_valid else "No"
