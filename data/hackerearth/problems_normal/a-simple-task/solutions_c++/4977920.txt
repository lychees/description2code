#include<iostream>
#include<algorithm>
#include<bits/stdc++.h>
#include<math.h>
#include<cstdlib>

int GCD(int A,int B);

using namespace std;

int main(void)
{
	int i,j,N,min,gcd;
//	cout<<"Enter the value of N:";
	cin>>N;
	int a,b;
	cin>>a;
	
	gcd=0;
	for(i=1;i<N;i++)
	{
		cin>>b;
		gcd=GCD(gcd,abs(a-b));
		a=b;
	}
	for(i=2;i<=gcd/2;i++)
	{
		if(gcd%i==0)
		  cout<<i<<" ";
	}
	cout<<gcd<<endl;
	
		return 0;
}
int GCD(int A,int B)
{
	if(B==0)
	  return A;
	else
	   return GCD(B,A%B);  
}
