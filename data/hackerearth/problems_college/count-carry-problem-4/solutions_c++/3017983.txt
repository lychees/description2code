#include <stdio.h>  
typedef unsigned long long int ll;  
int main(void) {  
 ll t;  
 ll a,b;  
 ll carry;  
 scanf("%llu",&t);  
 while(t--)  
 {  
  carry=0;  
  scanf("%llu %llu",&a,&b);  
  while(a&&b)  
  {  
   if(((a%10)+(b%10))>=10)  
    carry++;  
   a/=10;  
   b/=10;  
  }  
  if(carry==0)  
   printf("No carry operation\n");  
  else if(carry==1)  
   printf("1 carry operation\n");  
  else  
   printf("%llu carry operations\n",carry);  
 }  
 return 0;  
}
