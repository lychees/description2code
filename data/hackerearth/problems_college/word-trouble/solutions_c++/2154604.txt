#include<stdio.h>
#include<string.h>

int main()
{
  int t;
  scanf("%d",&t);
  while(t--)
  {
  	char a[50],b[50];
  	int p,l,count=0,i;
    scanf("%s%s",a,b);
    p=strlen(a);
    l=strlen(b);
    if(p<=l)
    {
      count=l-p;
    for(i=0;i<p;i++)
      count+=a[i]!=b[i];
    if(count)
      printf("Non Exact %d\n",count);
    else
      printf("Exact\n");
    }
    else
    {
    count=p-l;
    for(i=0;i<l;i++)
      count+=a[i]!=b[i];
    if(count)
      printf("Non Exact %d\n",count);
    else
      printf("Exact\n");
    }
  }
  return 0;
}
