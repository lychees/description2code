#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    for(int i=1;i<=T;i++) {
    	int T;
    	long N;
    	cin>>N>>T;
    	long X=N;
    	for(int j=0;j<T;j++) {
    		if(X%2 ==0) {
    			X-=X/2;
    		} else {
    			X-=(X+1)/2;
    		}
    			X-=(X/4);
    	}
    	cout<<X<<endl;
    }
    return 0;
}
