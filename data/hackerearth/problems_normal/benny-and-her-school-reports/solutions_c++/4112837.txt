#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n,m,x;
    	cin>>n>>m>>x;
    	long long int sum = 0;
    	int iter = n;
    	while(iter--){
    		int v;
    		cin>>v;
    		sum+=v;
    	}
    	
    	
    	long long int ans = (long long int)x * (n+1) - sum;
    	//cout << ans<<" : ";
    	if(ans>m){
    		cout<<"Impossible\n";
    	}
    	else if(ans < 0){
    		cout<<1<<"\n";
    	}
    	else{
    		cout<<ans<<"\n";
    	}
    }
    return 0;
}
