
nums = []


def process():
    global nums
    nums = [0]*(10**6 + 3)
    nums[1] = 2
    nums[2] = 3

    for i in xrange(3, 10**6 + 3):
        nums[i] = (nums[i-1] + nums[i-2]) % 1000000007


def main():
    global nums
    process()
    for _ in xrange(int(raw_input())):
        print(nums[int(raw_input())+2]-3)


main()
