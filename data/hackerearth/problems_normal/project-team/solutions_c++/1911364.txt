#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int t;
	cin>>t;
	int n=0;
	while(t--){
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++)
			cin>>a[i];
		
		sort(a,a+n);
		int i=0,j=n-1;
		
		long max=-1,min=1000000000;
		while(i<j){
			if((a[i]+a[j]) > max) max=a[i]+a[j];
			if((a[i]+a[j]) < min) min=a[i]+a[j];
			
			i++;
			j--;
		}
		cout<<max-min<<endl;	
	}
	return 0;
}
