#include <bits/stdc++.h>
using namespace std;
#define mod 1000000009
long long int po(long long int a,long long int b)
{
	if(b==0)
	{
		return 1;
	}
	long long int temp;
	temp=po(a,b/2);
	if(b%2==0)
	{
		temp=((temp%mod)*(temp%mod))%mod;
		return temp;
	}
	else
	{
		temp=((a)*((((temp%mod)*(temp%mod))%mod)%mod))%mod;
		return temp;
	}
}
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   		long long int n;
   		cin>>n;
   		long long int res=po(26,(n+1)/2);
   		cout<<res%mod<<endl;
   }
}
