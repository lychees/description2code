for t in xrange(input()):
    r, c = map(int, raw_input().split())
    arr = [[0 for i in xrange(c)] for j in xrange(r)]
    s = 0
    for i in xrange(r):
        for j in xrange(c):
            if arr[i][j] == 0:
                s += 1
                arr[i][j] = 1
                if j+2 < c:
                    arr[i][j+2] = 1
                if i+2 < r:
                    arr[i+2][j] = 1
    print s
