#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
#define MAX 100005
#define ll long long int
ll fact[MAX];
ll ifact[MAX];
ll modexp(ll a, ll b){ll res=1;while(b){if(b&1)res=(res*a)%MOD;a=(a*a)%MOD;b/=2;}return res;}
int main()
{
		int i;
		fact[0]=1;
		for(i=1;i<MAX;++i)
				fact[i] = (fact[i-1]*i)%MOD;
		for(i=0;i<MAX;++i)
				ifact[i] = modexp(fact[i],MOD-2);
		int n;
		int m;
		int t;
		int tmp;
		ll sum;
		ll t1,t2;
		cin>>t;
		while(t--)
		{
				cin>>m>>n;
				sum = 0;
				for(int i=0;i<m;++i)
				{
						cin>>tmp;
						sum+=tmp;
				}
				n-=sum;
				if(n < 0)
				{
					cout<<0<<endl;
					continue;
				}
				t1 = (fact[n+m-1]*ifact[m-1])%MOD;
				t1 = (t1*ifact[n])%MOD;
				if(t1<0) t1+=MOD;
				cout<<t1<<endl;

		}
		return 0;
}
