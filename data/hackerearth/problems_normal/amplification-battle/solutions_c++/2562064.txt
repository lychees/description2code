#include <iostream>
#include <algorithm>
using namespace std;
long long int bi(long long int x,long long int y){
	if(x%y==0) return x/y;
	return x/y+1;
}
int main(){
	int t; cin>>t;
	while(t--){
	long long int a,b,c;
	cin>>a>>b>>c;
	int res=0;
	while(c>1){
		if(res == 0){
			c = bi(c,b);
			res = 1;
		}
		else{
			res = 0;
			c = bi(c,a);
		}
	}
	if(res == 1) cout<<"Adam"; else cout<<"Bob"; cout<<endl;
}
	return 0;
}
