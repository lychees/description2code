for i in range(int(raw_input())):
    x, y = map(int, raw_input().split(' '))
    l = list(map(int, raw_input().split(' ')))
    a = {}
    for i in range(x):
        a[l[i]] = 0
    for i in range(x, x+y):
        if l[i] not in a:
            print("NO")
            a[l[i]] = 0
        else:
            print("YES")
