def solve(l,k,m):
	l.sort(key=lambda x:x[0:m])
	print(l[k-1])

	n,k,m=map(int,raw_input().split())

	l=[]

	for i in range(n):
		s=raw_input()
		l.append(s)

	solve(l,k,m)
