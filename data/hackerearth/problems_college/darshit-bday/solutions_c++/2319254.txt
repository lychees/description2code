#include<iostream>
using namespace std;

int main()
{
	long testcases,b,w,x,y,z,res;
	cin>>testcases;
	while(testcases--){
		cin>>b>>w>>x>>y>>z;
	
	
		if(x>y && x>(y+z)){
			res=(y+z)*b+y*w;
		}
		else if(y>x && y>(x+z)){
			res=x*b+(x+z)*w;
		}
		else{
			res=b*x+w*y;
		}
		cout<<res<<endl;
	}
}
