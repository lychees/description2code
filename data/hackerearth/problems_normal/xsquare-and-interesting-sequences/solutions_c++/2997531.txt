#include<bits/stdc++.h>
#define M 1000000009
#define LL long long 
using namespace std;


LL power(LL a,LL b){
	LL res = 1 ;
	while(b){
		if(b%2){
			res = res*a ;
			res %= M ;
		}
		b /= 2 ;
		a = a*a ;
		a %= M ;
	}
	
	return res ;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{	
		long long int n;
		cin>>n;
		long long int ans=((power(26,(n+1)/2))%M);
		cout<<(ans%M)<<endl;
	}
}
