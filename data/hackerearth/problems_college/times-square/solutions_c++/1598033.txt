#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n ,m, i, j, k, l, a, k1, k2;
	scanf("%lld%lld%lld", &n, &m, &a);
	if(n < m) {
		k1 = n;
		k2 = m;
	} else {
		k1 = m;
		k2 = n;
	}
	i = k1/a;
	if(k1%a != 0) 
		i++;
	j = k2/a;
	if(k2%a != 0) 
		j++;
	printf("%lld\n", i*j);		
	return 0;
}
