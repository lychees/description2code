from __future__ import print_function
t = input()
while (t > 0):
    p, k = map(int, raw_input().split())
    k += 1
    d = k
    if (p == 1):
        print(k, end="\n")
    else:
        while (p >= 1):
            print(k, end=" ")
            k += d
            p -= 1
        print(end="\n")

    t -= 1
