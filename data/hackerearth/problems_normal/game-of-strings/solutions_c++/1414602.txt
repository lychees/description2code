#include <iostream>
using namespace std;


string a,b; int la,lb;
int dlcs[5002][5002];
int lcs(int i, int j){
	if(i==la||j==lb){
		return 0;
	}
	if(dlcs[i][j]!=-1)
	return dlcs[i][j];
	if(a[i]==b[j]&&a[i]!='a'&&a[i]!='e'&&a[i]!='i'&&a[i]!='o'&&a[i]!='u'){
		return dlcs[i][j]=1+lcs(i+1,j+1);
		
	}
	else{
		return dlcs[i][j]=max(lcs(i+1,j),lcs(i,j+1));
	}

}
int main()
{
	 cin>>a>>b;
	la=a.length(), lb=b.length();
	for(int i=0;i<=la;i++){
		for(int j=0;j<=lb;j++){
			dlcs[i][j]=-1;
		}
	}
	
	cout<<lcs(0,0);

    return 0;
}
