#include<iostream>
#include<stdio.h>
#include<math.h>
#include<string>
using namespace std;
int main()
{
    string a, b;
    int temp2;
    int la, lb, val_a, val_b;
    cin>>a;
    cin>>b;
    la = a.length();
    lb = b.length();
    int s[lb], ar[la], br[lb];
    for(int i = 0; i<la; ++i)
        ar[i] = int(a[i] - 48);
    for(int i = 0; i<lb; ++i)
        br[i] = int(b[i] - 48);
    for(int i = 0; i<la; ++i)
    {
        for(int p = 0; p<lb; ++p)
            s[p] = 0;
        temp2 = 0;
        val_a = 0;
        val_b = 0;
        ar[i] = (!ar[i]);
        for(int j = 0; j<la; ++j)
            val_a += ar[j] * pow(2,la-j-1);
        val_b = val_a;
        int m = 0;
        while(val_b > 0)
        {
            s[m++] = val_b % 3;
            val_b /= 3;
        }
        for(int j = 0; j<lb; ++j)
        {
            if(br[j] != s[lb-j-1])
                temp2++;
            if(temp2 == 2)
                break;
        }
        if(temp2 < 2)
            break;
        ar[i] = (!ar[i]);
    }
    printf("%d",val_a);
    return 0;
}
