import sys


def read_int(stream):
    return int(stream.readline().rstrip())


def read_array(stream):
    return [int(x) for x in stream.readline().rstrip().split()]


def sell(stock):
    Max_index = stock.index(max(stock))
    Max_price = max(stock)
    profit = Max_index*Max_price-sum(stock[0:Max_index])
    if Max_index != (len(stock)-1):
        stock = stock[(Max_index+1):(len(stock))]
        profit = sell(stock)+profit
    return profit
