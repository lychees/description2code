for x in xrange(input()):
    N, M = map(int, raw_input().split())
    a, b = map(int, raw_input().split())
    c, d = map(int, raw_input().split())
    cst = input()
    ans = (min(abs(a-c), N-abs(a-c))+min(abs(b-d), M-abs(b-d)))*cst
    print ans
    print "You saved the group." if ans < 1000 else "Let go of the group."
