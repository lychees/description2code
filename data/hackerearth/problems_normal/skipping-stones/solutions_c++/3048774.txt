#include <iostream>
#include <vector>
#include <algorithm>
#include <map>
#include <cstdio>
using namespace std;
vector<float> v1,v2;
map<int,float> mp1;
float findprob(int a,float b,float c)
{
//	cout<<a<<"->";
	if(a==c)
		return v1[a];
	int p3;
	float res=-1.0;
	for(p3=a+1;p3<=c&&v2[p3]-v2[a]<=b;p3++)
	{
		if(mp1[p3]==0)
			mp1[p3]=findprob(p3,b,c);
		res=max(res,mp1[p3]);
	}
	return res*v1[a];
}
int main()
{
    int n,p1,p2;
    float result=1,d,l;
    cin>>n>>l>>d;
    v1.resize(n+2);
    v2.resize(n+2);
    v1[0]=1.0;
    v1[n+1]=1.0;
    v2[0]=0;
    v2[n+1]=d;
    for(p1=1;p1<=n;p1++)
    	cin>>v1[p1];
    for(p1=1;p1<=n;p1++)
    	cin>>v2[p1];
    result=findprob(0,l,n+1);	
    //cout<<result;
    if(result<0)
    	cout<<"IMPOSSIBLE\n";
    else
    	printf("%.6f",result);
    return 0;
}
