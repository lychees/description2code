dbg = False
t = int(raw_input())
for _ in xrange(t):
    n, p = map(int, raw_input().split())
    data = [p]
    for _ in xrange(n):
        n = raw_input()
        if n == "B":
            data += [data[-2]]
        else:
            n = int(n.split()[1])
            data += [n]
        if dbg:
            print "player %i" % data[-1]
    print "Player %i" % data[-1]
