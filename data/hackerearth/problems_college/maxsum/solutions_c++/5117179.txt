#include <iostream>
#include <cmath>
using namespace std;

unsigned int a[10000],dp[10000];

unsigned int solve(int n)
{
	if (n < 0) return 0;
	if (dp[n] != 0) return dp[n];
	return dp[n] = max(solve(n - 1),solve(n - 2) + a[n]);
}

int main()
{
    int t,n,i;
    cin >> t;
    while (t--)
    {
    	cin >> n;
    	for (i = 0; i < n; i++)
    	{ 
    		cin >> a[i];
    		dp[i] = 0;
    	}
    	cout << solve(n-1) << "\n";
    }
    return 0;
}
