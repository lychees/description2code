#include<bits/stdc++.h>

using namespace std;
#define ll long long int
#define mq 1000000007
#define pb push_back
#define mk make_pair

int main() {

	int t;
	cin>>t;
	int tin=t;
	while(t--) {
		int m, n;
		cin>>m>>n;

		priority_queue <int> pm[100005];
	
		int x,y,g,h,good=0, great=0;

		for(int i=0; i<m; i++) {
			cin>>x>>y;
			pm[x-1].push(y);
		}

		int k;
		for(int i=0; i<n; i++) {
			cin>>x>>y;

			if(!pm[x-1].empty()) {
				good++;
				k=pm[x-1].top();
				pm[x-1].pop();
				if(k<=y) {
					great++;
			//		if(tin-t>1)
			//		cout<<"k ki value"<<k<<" "<<x-1<<" y ki val "<<y<<endl;
				}
			}
		}

		cin>>g>>h;

		if(h<=great) {
			cout<<"Great"<<h <<" "<<great<<"\n";
		}
		else if(g<=good) {
			cout<<"Good"<<g <<" "<<good<<"\n";
		}
		else {
			cout<<":(\n";
		}

	}
	return 0;
}
