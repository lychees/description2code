#include <bits/stdc++.h>

using namespace std;
vector<string> v[100005];
int main()	
{
	int test;
	cin>>test;
	string str;
	cin.ignore();
	for(int a=0;a<test;a++)
	{
		getline(cin,str);
		for(int i=0;i<100005;i++)
			v[i].clear();
		int c=-1,len=str.size(),oldc=-1,i=0;
		if(str.find(" ")==-1)
		{
			cout<<str<<endl;
			continue;
		}
		for(i=0;;i++)
		{
			c=str.find(" ",c+1);
			v[i].push_back(str.substr(oldc+1,c-oldc));
			if(c==-1)
				break;
			oldc=c;
		}
		string tmp=v[i][0];
		v[i].pop_back();
		v[i].clear();	
		tmp=tmp+" ";
		v[i].push_back(tmp);
		for(int j=i;j>0;j--)
		{
			cout<<v[j][0];
		}
		tmp=v[0][0];
		for(int p=0;p<tmp.size()-1;p++)
			cout<<tmp[p];
		cout<<endl;
	}
}
