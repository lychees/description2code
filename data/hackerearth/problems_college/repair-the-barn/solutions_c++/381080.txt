#include <iostream>
#include<stdlib.h>
using namespace std;
int cmp(const void*a,const void *b){
return (*(int*)a-*(int*)b);
}
int pmc(const void*a,const void*b){
return (*(int*)b-*(int*)a);
}

int main() {
	int m,s,c,p[200],diff[200],i,len;
	cin>>m>>s>>c;
	for (i=0;i<c;i++)
	cin>>p[i];
	qsort(p,c,sizeof(int),cmp);
	for (i=1;i<c;i++)
	diff[i-1]=p[i]-p[i-1];
	qsort(diff,c-1,sizeof(int),pmc);
	if (m>=c)
	cout<<c;
	else{
	len=p[c-1]-p[0]+1;
	for (i=1;i<m;i++)
	len-=(diff[i-1]-1);
	cout<<len;
	}
	return 0;
}
