#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

const int lmt = 100005;
unordered_map<int,int> stor;
int in[lmt];
int ans[lmt];

int main(){
    int t,n;
    scanf("%d",&t);
    while(t--){
        stor.clear();
        scanf("%d",&n);
        for(int i=1;i<=n;i++)
            scanf("%d",&in[i]);
        ans[0]=0;
        for(int i=1;i<=n;i++){
            if(stor.find(in[i])==stor.end())
                ans[i] = ans[i-1] + 1;
            else
                ans[i] = min(ans[i-1]+1,i-stor.find(in[i])->second);
            stor[in[i]]=i;
        }
        int st=0,en=0,len=0;
        for(int i=1;i<=n;i++){
            if(ans[i]>len){
                len = ans[i];
                en = i;
                st = i-len+1;
            }
        }
        printf("%d %d\n",st,en);
    }
    return 0;
}
