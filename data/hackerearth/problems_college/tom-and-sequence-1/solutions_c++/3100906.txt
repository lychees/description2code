#include<bits/stdc++.h>
#define mod (1000000007)
using namespace std;
long long primes[200000];
long long size=0;
void sieve(long long n)
{
    vector<bool> composite(n+100,false);
	for(long long i=2;i<=n;++i)
	{
		if(!composite[i])
		{
            for(long long b=2*i;b<=n;b+=i)
                composite[b]=true;
		}
	}
	for(long long i=2;i<=n;++i)
		if(!composite[i])
		{
			primes[size]=i;
			if(size!=0)
			primes[size]=((primes[size]%mod)*(primes[size-1]%mod))%mod;
			size++;
		}
}
int main()
{
	sieve(2000000);
	int t;
	cin>>t;
	while(t--)
	{
		long long n;
		cin>>n;
		cout<<primes[n-1]<<'\n';
	}
    return 0;
}
