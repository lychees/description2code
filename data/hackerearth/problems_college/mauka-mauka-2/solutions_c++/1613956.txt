#include <bits/stdc++.h>
using namespace std ;
#define MAXN 100001

bool P[MAXN] ;
vector<int> adj[MAXN] ;
int A[MAXN] , DP[MAXN] ,ans , N , B[MAXN];
void sieve(){
    int c = 0 ;
    for(int i=2;i<MAXN;i++){
        if(!P[i]){
            c ++ ;
            int j = i ;
            while(j < MAXN){
                adj[j].push_back(i) ;
                P[j] = 1 ;
                j += i ;
            }
        }
    }
}
int main(){
    sieve() ;
    cin >> N ;
    for(int i=1;i<=N;i++) cin >> A[i] ;
    for(int i=1;i<=N;i++){
        int mx = 0 ;
        for(int j=0;j<adj[A[i]].size();j++){
            mx = max(mx,B[adj[A[i]][j]]) ;
        }
        DP[i] = mx + 1 ;
        ans = max(ans,DP[i]) ;
        for(int j=0;j<adj[A[i]].size();j++){
            B[adj[A[i]][j]] = DP[i] ;
        }
    }
    
    cout << ans << endl ;
    return 0 ;
}
