#include <iostream>
using namespace std;
#include<stdlib.h>
int main()
{
    long int n,k,i,ar[1000001],c=0;
    cin>>n>>k;
    for(i=0;i<n;i++)
    cin>>ar[i];
    i=0;
    while(i<n)
    {
    	if((i+k)>=n)
    	break;
    	if(abs(ar[i]+ar[i+k])%k==0 ||abs(ar[i]-ar[i+k])%k==0 )
    	c++;
    	i++;
    }
    cout<<c;
    return 0;
}
