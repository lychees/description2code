#include <iostream>
using namespace std;

int main() 
{
	int t;
	int x,y,z;
	cin>>t;
	while(t--)
	{
		cin>>x>>y>>z;
		int maxi=0;
		for(int i=0;i<3;i++)
		{
			int xrem=x-i;
			int yrem=y-i;
			int zrem=z-i;
			if(xrem<0 || yrem<0 ||zrem<0)break;
			maxi=max(maxi,i+xrem/3+yrem/3+zrem/3);
		}
		cout<<maxi<<endl;
	}
	// your code goes here
	return 0;
}
