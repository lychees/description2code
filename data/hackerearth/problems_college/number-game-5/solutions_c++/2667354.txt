#include <iostream>
#include <cstdio>
using namespace std;
const int maxn = 1001000;
long long mul[maxn], sum[maxn];

void pre()
{
    for (long long i = 1; i < maxn; i++) {
         mul[i] = i * (i + 1) * (i + 1);
    }
    sum[1] = mul[1];
    for (int i = 2; i < maxn; i++)
        sum[i] += sum[i - 1] + mul[i];
}

int Binceil(int L,int R,int x)
{
    while(L<=R) {
          int mid = (L + R) >> 1;
          if(sum[mid]<=x && sum[mid+1]>x) return mid+1;
          else if(sum[mid]>x) R = mid-1;
          else L = mid + 1;
    }
    return -1;
}

int main() {
    int cases;
    scanf("%d", &cases);
    pre();
    while(cases--) {
          long long n;

          if (n < 4) {

           }
                   if (i == 1) {

}
