    #include<bits/stdc++.h>
    using namespace std;
    int main()
    {
    	int t;    	
    	cin >> t;
    	while(t--)
    	{
    		long int n, s=0;		
    		cin >> n;
    		long int a[n];
    		for(int i = 0; i < n; i++)
    			cin >> a[i];
    		long int temp[n];
    		temp[n - 1] = a[n - 1];
    		for(int i = n - 2; i >= 0; i--)
    		{
    			if(a[i] > temp[i + 1])
    				temp[i] = a[i];
				else
					temp[i] = temp[i + 1];
    		}
    		for(int i = 0; i < n; i++)
    		    s += temp[i] - a[i];
    		cout << s<< endl;
    	}
    }
