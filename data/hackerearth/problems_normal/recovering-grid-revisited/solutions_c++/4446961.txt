#include <bits/stdc++.h>
using namespace std;
long long int modinverse(long long int a){
	int n=1000000005;
	long long int ans=1;
	while(n){
		if(n&1){
			ans=(ans*a)%1000000007;
		}
		n=n>>1;
		a=(a*a)%1000000007;
	}
	return ans;
}
int main()
{	long long int n,m,t,var;
	cin>>t;
	
	while(t--){
		long long int ans=1;
    	cin>>n;
    	cin>>m;
    	if(n==1){
    		cout<<0<<endl;
    		continue;
    	}
    	var=n*n-1;
    	//cout<<var;
    	for(int i=m+var-1;i>var-1;i--){
    		ans=(ans*i)%1000000007;
    	}
    	long long int prod=1;
    	for(int i=1;i<=m;i++){
    		prod=(prod*i)%1000000007;
    	}
    	
    	ans=(ans*modinverse(prod))%1000000007;
    	
    	cout<<ans<<endl;
	}
    return 0;
}
