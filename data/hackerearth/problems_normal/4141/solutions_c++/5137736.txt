#include <iostream>
using namespace std;
typedef long int ll;
int main()
{
	ll t;
	scanf("%lld",&t);
	while(t--)
	{
		ll c,n,x,a[50002],i;
		scanf("%lld%lld%lld%lld",&a[0],&a[1],&c,&n);
		if(n==1)
		{
			if(a[0]==0)
			printf("YES\n");
			else
			printf("NO\n");
			
		}
		else
		{
			x=(a[0]*10+a[1])%41;
			for(i=2;i<n;i++)
			{
				a[i]=(a[i-1]*c+a[i-2])%10;
				x=x*10+a[i];
				x=x%41;
				
			}
			//cout<<a[0]<<a[1]<<a[2];
			if(x==0)
			printf("YES\n");
			else
			printf("NO\n");
		}
	}
    return 0;
}
