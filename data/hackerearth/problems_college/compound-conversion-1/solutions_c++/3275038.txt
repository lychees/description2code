#include <iostream>
using namespace std;

int main()
{
    int t,k;
    cin>>t;
    k=1;
    while(t--)
    {
    	int q=0,o=0,z=0,oo=0;
    	string s1,s2;
    	cin>>s1>>s2;
    	for(int i=0;i<s1.length();i++)
    	{
    		if(s1[i]=='?'){
    			
    		 q++;
    		 if(s2[i]=='1')oo++;
    		}
    		else if(s1[i]!=s2[i])
    		{
    			if(s1[i]=='1') o++;
    			else z++;
    		}
    		
    	}
    	if(o<=z)
    	{
    		cout<<"Case "<<k<<": "<<(z+q)<<endl; 
    	}
    	else {
    		if(oo+z>=o)cout<<"Case "<<k<<": "<<(q+o)<<endl;
    		else cout<<"Case "<<k<<": "<<"-1"<<endl;
    	}
    	k++;
    }
    return 0;
}
