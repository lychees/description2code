const int N = 1e5 + 5;

char s[N];
int main() {
  int n, m, k, l, r, c, q;
  scanf("%d%d%d%s", &n, &m, &k, s + 1);
  m += k;
  while (m--) {
    scanf("%d%d%d%d", &q, &l, &r, &c);
    if (q == 1) {
      memset(s + l, c + '0', r - l + 1);
    } else {
      puts(memcmp(s + l, s + l + c, r - l - c + 1) ? "NO" : "YES");
    }
  }
  return 0;
}
