#include <bits/stdc++.h>

using namespace std;

void constructDP(int dp[])
{
	int i = 1;
	dp[0] = 0;
	dp[1] = 1;
	while(i <= 50000)
	{
		dp[2*i] = 2*dp[i] - 1;
		dp[(2*i)+1] = 2*dp[i] + 1;
		i++;
	}
}

int main()
{
	int dp[200010]={0};
	constructDP(dp);
	int t = 0;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		printf("%d\n", dp[n]);
	}
	return 0;
}
