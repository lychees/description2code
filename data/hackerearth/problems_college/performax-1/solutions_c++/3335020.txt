#include <iostream>
#include<vector>
using namespace std;


int main()
{
    int n,d,ans=0,last=0; cin >> n;cin >> d;
    if(n<=1){
    	cout << 0;return 0;
    }
    vector<int> v(n); cin >> v[0];last=v[0];
    for(int i=1;i<n;i++){
    	cin >> v[i];
    	if(v[i]<=last){
    		ans+= ((last-v[i])/d)+1;
    		last=v[i]+(d*(((last-v[i])/d)+1));
    	}
    	else
    		last=v[i];
    }
    cout << ans;
    return 0;
}
