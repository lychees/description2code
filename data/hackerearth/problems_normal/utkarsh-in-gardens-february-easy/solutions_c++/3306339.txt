#include <bits/stdc++.h>

using namespace std;

bitset < 2000 > g[2000];
bitset < 2000 > temp;
int a[2030][2030];

int main () {
	int n;
	scanf("%d",&n);
	for (int i=0; i<n; i++) {
		for (int j=0; j<n; j++) {
			cin >> a[i][j];
			if (a[i][j])
				g[i][j] = 1;
		}
	}
	long long ans = 0;

	for (int i=0; i<n; i++) {
		for (int j=i+1; j<n; j++) {
			temp =  (g[i] & g[j]);
			int x = temp.count();
			ans = ans + x * (x - 1) / 2;
		}
	}
	cout << ans / 2 << endl;
	return 0;
}
