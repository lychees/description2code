#include <iostream>
#include<algorithm>
using namespace std;

int A[26];
int main(){
    string s,s2,s3;
    cin >> s;
    for(int i=0;i<s.size();i++){
    	A[s[i]-'a']++;
    }
    
    int cnt=0;
    for(int i=0;i<26;i++){
   		if(A[i]%2){
   			s3+=i+'a';
   			cnt++;
   			if(cnt>1){
   				cout << -1 << endl;
   				return 0;
   			}
   		}
   		while(A[i]){
   			if(A[i]==1){
   				break;
   			}
   			s2+=i+'a';
   			A[i]-=2;
   		}
    }
    s=s2;
    reverse(s2.begin(),s2.end());
    s=s+s3+s2;
    
    cout << s << endl;
    return 0;
}
