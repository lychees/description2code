#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        vector<int> v(n+1);
        vector<int> r;
        vector<int> l;
        for(int i=1;i<=n;i++)
            cin>>v[i];
        int i=1;
        int level=0;
        while(i<=n)
        {
            int j=pow(2,level);
            int k=i;
            int left=0,right=0;
            int f=1;
            while(k<=n&&k<(i+j))
            {
                if(v[k]&&f==1){
                    right=v[k];
                    f=2;
                }
                else if(v[k]&&f==2)
                {
                	if(left==0)
                    left=right;
                    right=v[k];
                }
                    k++;
            }
           // cout<<level<<" "<<left<<" : "<<right<<endl; 
            if(left)
                l.push_back(left);
            if(right)
                r.push_back(right);
            level++;
            i*=2;
        }
        for(int j=0;j<r.size();j++)
            cout<<r[j]<<endl;
        for(int j=0;j<l.size();j++)
            cout<<l[j]<<endl;
        cout<<endl;
    }
    return 0;
}
