#include<bits/stdc++.h>
#define  den 1000001
typedef long long int lu;
using namespace std;
int main()
{
	lu i,j,x,t,n;
	vector<int> ar;
	vector<lu>br;
	ar.assign(1000001,0);
	ar[0]=ar[1]=1;
	for(i=2;i<den;i++)
	{
		if(ar[i]==0)
		{
			x=3;
			for(j=i*2;j<den;)
			{
				ar[j]=1;
			j=i*x;
			x++;}
		}
	}
	for(i=0;i<den;i++)
	{
		if(ar[i]==0)
		br.push_back(i);
	}
	
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		if(ar[n]==0)
		{
			printf("2^0\n");
			continue;
		}
		i=0;
		vector<lu> cr;
	cr.assign(br.size(),0);
		while(n>1)
		{
			while(n%br[i]==0&&n!=0)
			{
				cr[i]++;
				n=n/br[i];
			}
			i++;
		}
		printf("2^%lld",cr[0]);
		for(j=1;j<i;j++)
		{
			if(cr[j]!=0)
			{
				printf("*%lld^%lld",br[j],cr[j]);
			}
			
		}
		cout<<endl;
	}

	return 0;
}
