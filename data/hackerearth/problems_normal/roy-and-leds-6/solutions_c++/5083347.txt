#include <iostream>
using namespace std;

int main()
{
    int N,r,g,b,i;
    int *R,*G,*B;
    int ir=0,ig=0,ib=0;
    int fr=0,fg=0,fb=0;
    int light[8] = {0};
    
    cin>>N;
    
   	R = (int *)malloc(sizeof(int)*N);
   	G = (int *)malloc(sizeof(int)*N);
   	B = (int *)malloc(sizeof(int)*N);
   	
   	cin>>r>>g>>b;
   	
   	for(i=0;i<N;i++){
   		R[i] = fr;
   		ir++;
   		
   		G[i] = fg;
   		ig++;
   		
   		B[i] = fb;
   		ib++;
   		
   		if(ir==r){
   			ir %= r;
   			if(fr){
   				fr = 0;
   			}else{
   				fr = 1;
   			}
   		}
   		
   		if(ig==g){
   			ig %= g;
   			if(fg){
   				fg = 0;
   			}else{
   				fg = 1;
   			}
   		}
   		
   		if(ib==b){
   			ib %= b;
   			if(fb){
   				fb = 0;
   			}else{
   				fb = 1;
   			}
   		}
   	}
    
    for(i=0;i<N;i++){
    	if(R[i]==1 && G[i]==0 && B[i]==0){
    		light[0]++;
    	}else if(R[i]==0 && G[i]==1 && B[i]==0){
    		light[1]++;
    	}else if(R[i]==0 && G[i]==0 && B[i]==1){
    		light[2]++;
    	}else if(R[i]==1 && G[i]==1 && B[i]==0){
    		light[3]++;
    	}else if(R[i]==0 && G[i]==1 && B[i]==1){
    		light[4]++;
    	}else if(R[i]==1 && G[i]==0 && B[i]==1){
    		light[5]++;
    	}else if(R[i]==1 && G[i]==1 && B[i]==1){
    		light[6]++;
    	}else{
    		light[7]++;
    	}
    }
    
    for(i=0;i<8;i++){
    	cout<<light[i]<<" ";
    }
    
    return 0;
}
