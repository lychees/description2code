#include<iostream>
using namespace std ;
#include <bits/stdc++.h>

int main()
{
	int t ; 
	cin >> t ; 
	while ( t--)
	{
	
	long long int x1 ,  y1  ,  x2 , y2 ; 
	cin >> x1>>y1 >>x2>>y2 ;
	long long int r ; 
	cin >> r ; 
	long long int d1  = abs(x1*( y2-y1) - y1*(x2-x1) ); 
	long long int d2 = sqrt( (y2-y1)*( y2-y1) + ( x2-x1)*( x2-x1)) ; 
	long double d = d1/d2 ; 
	if ( d < r)
	{
		cout << "REPLANNING" <<endl; 
	}
	else if ( d == r )
	{
		cout << "JUST MISSED" <<endl; 
	}
	else
	{
		cout << "SAFE" <<endl; 
	}
	
   }
	  return 0 ;
	
}
