int task(string s, char from, char to) {
    for(int i = 0; s[i]; i++) {
        if(s[i] == from) {
            s[i] = to;
        }
    }

    istringstream iss(s);
    int a;
    iss >> a;
    return a;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);

    string s;
    cin >> s;
    int mxa = task(s, '5', '6');
    int mna = task(s, '6', '5');
    cin >> s;
    int mxb = task(s, '5', '6');
    int mnb = task(s, '6', '5');

    cout << mna + mnb << ' ' << mxa + mxb << endl;
    End;
}
