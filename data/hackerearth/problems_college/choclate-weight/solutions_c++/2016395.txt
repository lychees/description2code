#include <bits/stdc++.h>
#include<iostream>
using namespace std;
 int main()
{
    int n,a[200000],i,tc;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    cin>>tc;
    while(tc--)
    {	int f,x,j;
    	cin>>x;
    	f=0;
    	for(i=0;i<n;i++)
    	{
    	if(a[i]==x)
    	{
    		f=1; 
    		break;
    	}
    	else
    	{
    	for(j=i+1;j<n;j++)
    	{
    		if(x==a[i]+a[j])
    		{
    			f=1; 
    			break;
    		}
    	}
    	}
    	if(f==1)
    	break;
    	}
    	if(f==0)
    	cout<<"I Hate You"<<endl;
    	else
    	cout<<"I Like You"<<endl;
    }
}
