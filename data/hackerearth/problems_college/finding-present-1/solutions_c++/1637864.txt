#include <iostream>
using namespace std;

int main()
{  int x[10000]={0},y[10000]={0};
    int p;
    cin>>p;
    int a[p];
    for(int  i=0;i<p;i++)
    {
    cin>>a[i];
    x[a[i]]++;
    }
    int q;
    cin>>q;
    int b[q];
    for(int i=0;i<q;i++)
    {
    cin>>b[i];
    y[b[i]]++;
    }
    for(int i=0;i<10000;i++)
    if(y[i]>x[i]||y[i]<x[i])
    cout<<i<<" ";
    
    
    return 0;
}
