#include<bits/stdc++.h>
using namespace std;
char a[10005][25];
#define abso(a,b) (a>b?(a-b):(b-a))
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,k,s;
		scanf("%d %d",&n,&k);
		int i,j;
		char ans[25];
		for(i=0;i<n;i++)
		{
			scanf("%s",a[i]);
		}
		for(i=0;i<k;i++)
		{
			int fr[26]={0};
			for(j=0;j<26;j++)
			{
				for(s=0;s<n;s++)
				{
					int d=a[s][i]-97;
					fr[j]+=abso(j,d);
				}
			}
			int min=1e9;int ind;
			for(j=0;j<26;j++)
			{
				//printf("%d ",fr[j]);
				if(fr[j]<min)
				{
					min=fr[j];
					ind=j;
				}
			}
			ans[i]=ind+97;
			
		}
		ans[i]='\0';
		printf("%s\n",ans);
	}
	return 0;
}
