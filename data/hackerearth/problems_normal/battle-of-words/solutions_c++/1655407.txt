#include <bits/stdc++.h>
using namespace std;

int main()
{
    int a; cin >> a;    	string aa; getline(cin, aa); 

    for (int g=0; g<a; g++)
    {
    	string b,c; getline(cin,b);getline(cin, c); 
    	map <char, int> first, second; 
    	int alicescore=0, bobscore=0; 
    	for (int g=0; g<b.length(); g++){if (b[g]!=' '){first[b[g]]++; alicescore++;} }
    	for (int g=0; g<c.length(); g++){if (c[g]!=' ') {second[c[g]]++; bobscore++;} }
    	for (map <char, int> :: iterator x=first.begin(); x!=first.end(); x++)
    	{
    		char l=x->first;
    		int z=min((int)first[l], (int)second[l]);
    		alicescore-=z; 
    		bobscore-=z; 
    	}
    	if (alicescore==0 && bobscore==0) cout << "You draw some." << '\n'; 
    	else if (alicescore>0 && bobscore>0) cout << "You draw some." << '\n'; 
    	else if (alicescore==0) cout << "You lose some." << '\n'; 
    	else cout << "You win some." << '\n';
    }
    return 0;
}
