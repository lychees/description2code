ll gcd ( ll a, ll b ) {
    a = ABS ( a ); b = ABS ( b );
    while ( b ) { a = a % b; swap ( a, b ); } return a;
}

ll ext_gcd ( ll A, ll B, ll *X, ll *Y ){
    ll x2, y2, x1, y1, x, y, r2, r1, q, r;
    x2 = 1; y2 = 0;
    x1 = 0; y1 = 1;
    for (r2 = A, r1 = B; r1 != 0; r2 = r1, r1 = r, x2 = x1, y2 = y1, x1 = x, y1 = y ) {
        q = r2 / r1;
        r = r2 % r1;
        x = x2 - (q * x1);
        y = y2 - (q * y1);
    }
    *X = x2; *Y = y2;
    return r2;
}

inline ll modInv ( ll a, ll m ) {
    ll x, y;
    ext_gcd( a, m, &x, &y );
    if ( x < 0 ) x += m; //modInv is never negative
    return x;
}

inline ll power ( ll a, ll p ) {
    ll res = 1, x = a;
    while ( p ) {
        if ( p & 1 ) res = ( res * x );
        x = ( x * x ); p >>= 1;
    }
    return res;
}

inline ll bigmod ( ll a, ll p, ll m ) {
    ll res = 1 % m, x = a % m;
    while ( p ) {
        if ( p & 1 ) res = ( res * x ) % m;
        x = ( x * x ) % m; p >>= 1;
    }
    return res;
}

int mod = 1000000000 + 7;
ll fact[2000000], inv[2000000];

void precal() {
    fact[0] = 1;
    rep(i,1,1000000) {
        fact[i] = fact[i-1] * i;
        fact[i] %= mod;
    }

    rep(i,0,1000000) {
        inv[i] = modInv( fact[i], mod );
    }
}

ll combo ( int n, int k ) {
    if ( n == k ) return 1;
    if ( n == 0 ) return 0;
    if ( k == 0 ) return 1;

    int x = n;
    int y = k;

    ll res = ( inv[x-y] * inv[y] ) % mod; ///Calculate nck(x,y)
    res *= fact[x];
    res %= mod;
    return res;
}
