#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10;

map <string, int> names;

int main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; i++) {
		string str;
		cin >> str;
		names[str]++;
	}
	for (auto it = names.begin(); it != names.end(); ++it) {
		cout << it->first << " " << it->second << '\n';
	}
	return 0;
}
