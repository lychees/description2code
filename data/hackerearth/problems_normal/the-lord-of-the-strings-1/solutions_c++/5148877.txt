#include <iostream>
#include <string>

using namespace std;

int main()
{
    int testcase;
    cin >> testcase;
    while(testcase--)
    {
        int rs[30] = {0};
        string s;
        cin >> s;
        int len = s.length();
        for (int i = 0; i < len; i++)
        {
            if (rs[s[i] - 'a'] != 1) cout << s[i];
            rs[s[i] - 'a'] = 1;
        }
        cout << endl;
    }

    return 0;
}
