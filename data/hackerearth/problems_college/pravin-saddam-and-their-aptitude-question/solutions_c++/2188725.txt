#include <iostream>
using namespace std;
unsigned long long a[19];
	long long fact(int n){
	if(n==0||n==1)
	return 1;
	
	return n*fact(n-1);
	}
void solve(){
	int i=1;
	a[0]=1;
	a[1]=1;
	for(int i=2;i<19;i++){
		a[i]=fact(i);
	}
}
int main()
{
	solve();
int t;
cin>>t;
int x,b;
while(t--){
	cin>>x>>b;
	unsigned long long sum=0;
	for(int i=x;i<=b;i++){
		sum+=a[i];
	}
	cout<<sum%10<<"\n";
}
return 0;
}
