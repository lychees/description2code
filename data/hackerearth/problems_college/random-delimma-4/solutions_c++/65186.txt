int main()
{
    //ios::sync_with_stdio(false);
    int t,i,j,n,k;
    double ans,num;
    cin>>t;
    for(i=0;i<t;++i)
    {
        cin>>k>>n;
        ans=1.0;
        num=(double)n;
        if(k>n)
        {
            cout<<"1.000000000\n";
            continue;
        }
        for(j=0;j<k;++j)
        {
            ans*=(num-j)/num;
        }
        ans=1.0-ans;
        printf("%0.9lf\n",ans);
    }
    return 0;
}
