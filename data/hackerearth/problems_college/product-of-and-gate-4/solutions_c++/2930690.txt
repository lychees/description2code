#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	int te;
	cin>>te;
	while(te--)
	{
		long long int a,i,b;
		cin>>a>>b;
		long long int ans=b;
		for(i=a;i<b;i++)
		{
			ans=(ans&i);
			//cout<<ans<<endl;
		}
		cout<<ans<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
