#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;

long long maxim(long long* ans, long long* ar, long long a, long long n)
{
	long long yo = 0;
	for(long long i=1;i<=sqrt(a);i++)
	{
		if(a%i==0)
		{
			if(yo<ans[i])
				yo = ans[i];
			if(yo<ans[a/i])
				yo = ans[a/i];
		}
	}
	return yo;
}

int main()
{
    long long n;
    cin>>n;
    long long* ar = new long long[n+1];
    long long* ans = new long long[n+1];
    for(long long i=1;i<=n;i++)
    	cin>>ar[i];
    ans[1] = ar[1];
    cout<<ans[1]<<" ";
    for(long long i=2;i<=n;i++)
    {
    	ans[i] = ar[i] + maxim(ans,ar,i,n);
    	cout<<ans[i]<<" ";
    }
}
