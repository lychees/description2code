from sys import stdin


def sumOfDigits(number):
    sum = 0
    while number > 0:
        rem = number % 10
        number = int(number / 10)
        sum = sum + rem
    return sum


inputCount = int(stdin.readline())
i = 0
sumOfNumbers = 0
sumOfSums = 0
while i < inputCount:
    i += 1
    input = int(stdin.readline())
    sumOfNumbers = sumOfNumbers + input
    sumOfSums = (sumOfSums % 9 + input % 9) % 9

print(sumOfSums % 9 - sumOfDigits(sumOfNumbers) % 9)
