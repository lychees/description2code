#include<bits/stdc++.h>
using namespace std;

long long reverse(long long num) {
	long long rev = 0 ;

	while(num) {
		rev = 10 * rev + num % 10;
		num /= 10;
	}
	return rev;
}

int main() {
	int t;
	cin >> t;

	while(t--) {
		long long n ,revn ;
		cin >> n;
		revn = reverse(n);

		if(reverse(pow(revn ,2)) == n*n && n != revn){
			printf("%lld\n",revn );
		}else{
			printf("No\n");
		}
	}
	return 0;
}
