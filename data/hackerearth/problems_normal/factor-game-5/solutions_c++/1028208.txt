template<class T> inline T tmax(T a, T b) {return (a > b) ? a : b;}

const int maxn = 100001;
int n, a[maxn], fact[maxn], cnt[maxn];
bool e[maxn];

int main(int argc, char *argv[]) {
    ios_base::sync_with_stdio(false);
    int val;
    
    mem(e);
    mem(cnt);
    cin >> n;
    repu(i, 2, maxn) {
        if (!e[i]) {
            fact[i] = i;
            for (int j = i * 2; j < maxn; j += i) {
                e[j] = true;
                fact[j] = i;
            }
        }
    }
    repu(i, 2, maxn) {
        cnt[i] = 1 + cnt[i / fact[i]];
    }
    int ans = 0;
    repu(i, 0, n) {
        cin >> val;
        ans ^= cnt[val];
    }
    if (ans != 0) printf("ALICE\n");
    else printf("BOB\n");
    return 0;
}
