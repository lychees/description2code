#include<bits/stdc++.h>
using namespace std;
int main ()
{
int T,num,i,k,row,col,mid,swap1,swap2;
char ch;
cin>>T;
while(T--)
{
	cin>>num;
	for(i=1;i<=num*num;++i)
	{
	cin>>ch;
	if(ch=='*')
	k=i;
	}
	if(k%num==0)
	row=k/num;
	else
	row=k/num+1;
	if(k%num==0)
	col=num;
	else
	col=k%num;
	mid=(num+1)/2;
	swap1=abs(mid-row);
	swap2=abs(mid-col);
	cout<<swap1+swap2<<endl;
}
return 0;
}
