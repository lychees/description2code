#include "bits/stdc++.h"
using namespace std;
int prime(int n)
{
	int f=0;
	if(n==1)
	return 1;
	if(n==2)
	return 0;
	if(n%2==0)
	return 1;
	for(int i=3;i<=sqrt(n)&&f==0;i+=2)
	{
		if(n%i==0)
		f=1;
	}
	return f;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char s[1005];
    	cin>>s;
    	int l=strlen(s),ans=-1;
    	int p,ansa=-1;
    	for(int i=0;i<l;i++)
    	{
    		for(int j=i,k=i;j>=0&&k<l;j--,k++)
    		{
    			if(s[j]==s[k])
    			{
    				p=(k-j+1);
    				//cout<<p<<endl;
    			}
    			else
    			break;
    		}
    		if(p>ansa)
    		ansa=p;
    		
    	}
    	for(int i=0;i<l-1;i++)
    	{
    		for(int j=i,k=i+1;j>=0&&k<l;j--,k++)
    		{
    			if(s[j]==s[k])
    			{
    				p=(k-j+1);
    			}
    			else
    			break;
    		}
    		if(p>ansa)
    		ansa=p;
    		
    	}
    	int q=prime(ansa);
    	if(q==0)
    	{
    		ans=ansa;
    	}
    	//cout<<ans<<endl;
    	if(ans>1)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
    return 0;
}
