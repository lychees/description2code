#include <iostream>
using namespace std;

int arr[101][101];
long long DP[101][101];
int main()
{
    int T;
    cin>>T;
    long long maxPath=1;
    while(T--)
    {
    	int N,M;
    	cin>>N>>M;
    	for(int i=1; i<=N; i++)
    	{
    		for(int j=1; j<=M; j++)
    		{
    			cin>>arr[i][j];
    			DP[i][j]=0;
    		}
    	}
		
		DP[1][1]=1;
		maxPath=1;
		for(int i=1; i<=N; i++)
		{
			for(int j=1; j<=M; j++)
			{
				
				if(!(i==1 && j==1))
				{
					//cout<<arr[i][j]<<"\n";
					if(DP[i-1][j] >= DP[i][j-1] && arr[i][j] > arr[i-1][j] && DP[i-1][j])
					{
						
						DP[i][j] = 1 + DP[i-1][j];
					}
					else if(arr[i][j] > arr[i][j-1] && DP[i][j-1])
					{
						DP[i][j] = 1 + DP[i][j-1];
					}
				}
				if(DP[i][j] > maxPath)
					maxPath=DP[i][j];
				//cout<<"("<<i<<","<<j<<")="<<DP[i][j]<<"\n";
			}
		}
		cout<<maxPath<<"\n";
    }
    return 0;
}
