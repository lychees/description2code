#include <iostream>
#define mod 1000000007
using namespace std;

unsigned long long int fact[1000001] = {0};

void calcFact()
{
	fact[0] =1;
	fact[1] =1;
	unsigned long long int i;
	for(i=1;i<=1000000;i++)
	{
		fact[i] = (i*fact[i-1])%mod;
	}
}

int main()
{
	int t;
	unsigned long long int n,ans;
	cin>>t;
	calcFact();
	while(t--)
	{
		cin>>n;
		ans = (3*fact[n])%mod;
		ans = (ans+3)%mod;
		cout<<ans<<endl;
	}
    return 0;
}
