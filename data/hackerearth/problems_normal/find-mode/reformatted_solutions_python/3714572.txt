while _:

    input()
    nums = map(int, raw_input().split())
    di = {}
    for i in nums:
        di[i] = di.get(i, 0) + 1
    k = di.keys()
    v = di.values()
    m = max(v)
    s = v.index(m)
    li = []
    while True:
        li.append(k[s])
        try:
            s = v.index(m, s+1)
        except BaseException:
            break
    for i in sorted(li, reverse=True):
        print i,
