#include <iostream>
#include <vector>
using namespace std;
#define max 4e+16

int main()
{
	unsigned long long int first,second,total,sum,i,n,t;
    first = 1;
    second = 2;
    vector <long long int> h;
    h.push_back(2);
    total=2;
    while(sum<=max)
    {
    	sum=first+second;
    	if((sum%2)==0)
    		h.push_back(sum);
    	first=second;
    	second=sum;
    }
   // for(i=0;i<5;i++)
    // cout<<h[i]<<endl;
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	i=0;
   	total=0;
   	while(h[i]<=n)
   	{
   		total+=h[i];
   		i++;
   	}	
    cout<<total<<endl;
   }
    return 0;
}
