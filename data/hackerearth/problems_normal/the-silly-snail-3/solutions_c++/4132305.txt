#include <bits/stdc++.h>
using namespace std;

void preOrder(vector< pair< int,int > > &tree, int root ) {
	if(root == 0)
		return;
	if(tree[root].first == 0  && tree[root].second == 0) {
		cout << root << " ";
		return;
	}
	
	cout << root << " ";
	preOrder(tree, tree[root].first);
	preOrder(tree, tree[root].second);
	
}

int main()
{
	int t = 1; 
	cin >> t;
	while(t--) {
		int n; cin >> n;
		vector<pair<int,int> > tree(100001, {0,0});
		int x, y, z;
		for(int i = 0; i < n; i++) {
			cin >> x >> y >> z;
			tree[x] = {y,z};
		}
		preOrder(tree, 1);
	
		cout << endl;
	}
    return 0;
}
