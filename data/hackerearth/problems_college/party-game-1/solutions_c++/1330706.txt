#include <iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int a[10005];
int main()
{
    long long int t,x,y,n,p,q,i;
    cin>>t;
    while(t--)
    {
    	cin>>x>>y>>n;
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	sort(a,a+n);
    	
    	p=0;
    	q=0;
    	
    	for(i=0;i<n;i++)
    	{
//    		p+=(x/a[i]);
//    		x=x-(x/a[i]);
//    		
//    		q+=(y/a[i]);
//    		y=y-(y/a[i]);
			
			if(x>=a[i])
			{
				x-=a[i];
				p++;
			}
			if(y>=a[i])
			{
				y-=a[i];
				q++;
			}

    	}
    	if(p>q)
    	{
    		cout<<"Chef Won\n";
    	}
    	else if(p<q)
    	{
    		cout<<"Jhool Won\n";
    	}
    	else
    	{
    		cout<<"Tie\n";	
    	}
    }
    return 0;
}
