#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {   ll e=0,o=0,i,a;
        ll n; cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>a;
            if(a%2==0)
                e++;
            else
                o++;
        }
        cout<<(e*o)<<endl;
    }
    return 0;
}
