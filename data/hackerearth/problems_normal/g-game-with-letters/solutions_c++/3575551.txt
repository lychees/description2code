#include <bits/stdc++.h>
using namespace std;
int l[39];
void solve(){
    memset(l,0,sizeof(l));
    int n;
    string s;
    cin>>n;
    cin>>s;
    for(int i=0;i<s.size();i++){
        l[s[i]-'a']++;
    }
    int best = 0, sz  = -1 , id = -1;
    for(int i=0;i<n;i++){
        string f;
        cin>>f;
        int curr = 0;
        for(int j=0;j<f.size();j++){
            curr += l[f[j]-'a'];
        }
        if(curr > best){
            best = curr;
            sz = f.size();
            id = i;
        }
        else if(curr == best && sz > f.size()){
            sz = f.size();
            id = i;
        }
    }
    cout<<id+1<<endl;
}
int main(){
    int t;
    cin>>t;
    while(t--)solve();
}
