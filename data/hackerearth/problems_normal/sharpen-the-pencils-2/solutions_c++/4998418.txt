#include <iostream>
#include <cmath>
#define u unsigned

using namespace std;

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
	u int t,n,i,x,y;
	u long a[100000],ca,cb;
    cin >> t;
    while (t--)
    {
    	cin >> n;
    	for (i = 0; i < n; i++)
    	 cin >> a[i];
    	ca = cb = x = 0;
    	y = n - 1;
    	while (abs(x-y) > 1)
    	{
    		cb += a[y] * 2;
    		while (ca < cb && x < y) ca += a[x++];
    		if (abs(x-y) < 1) break;
    		y--;
    	}
    	cout << x << " " << n - y << "\n";
    }
    return 0;
}
