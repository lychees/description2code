#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    int T;
    cin>>T;
    while(T--)
    {
    	int n;
    	cin>>n;
    	map<int,int> m1,m2;
    	for(int i=1;i<=n;i++)
    	{
    		int e;
    		cin>>e;
    		m1.insert(make_pair(e,i));
    	}
    	
    	for(auto i : m1)
    	{
    		m2.insert(make_pair(i.second,i.first));
    		
    	}
    	/*
    	cout<<"Map 1 contains: \n";
    	for(auto i : m1)
    	cout<<i.first<<" "<<i.second<<"\n";
    	
    	cout<<"Map 2 contains: \n";
    	for(auto i : m2)
    	cout<<i.first<<" "<<i.second<<"\n";
    	*/
    	
    	if(m1==m2)
    	cout<<"inverse\n";
    	else
    	cout<<"not inverse\n";
    }
    return 0;
}
