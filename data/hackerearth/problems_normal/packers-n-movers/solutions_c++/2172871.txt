#include <bits/stdc++.h>
using namespace std;

int arr[251];
int dps[20][251];

int main()
{
	int i,j,m,n,k;
	cin>>m>>n;
	for(i=0;i<n;i++)
	{
		cin>>arr[i];
	}
	
	int sum[251]={0};
	
	for(i=1;i<=n;i++)
	dps[1][i]=sum[i]=sum[i-1]+arr[i-1];
	
	for(i=2;i<=m;i++)
	{
		for(j=1;j<=n;j++)
		{
			dps[i][j]=100000000;
			for(k=1;k<j;k++)
			{
				dps[i][j]=min(dps[i][j],max(dps[i-1][k],sum[j]-sum[k]));
			}
		}
	}
	cout<<dps[m][n];
	
	return 0;
}
