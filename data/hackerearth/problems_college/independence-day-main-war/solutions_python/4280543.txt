def getPrecise(x):
    return sum(x)/len(x)
    
for _ in xrange(input()):
    x=input()
    xcr,ycr = [],[]
    for _ in xrange(x):
        x,y = map(float,raw_input().split())
        xcr.append(x)
        ycr.append(y)
    print "("+str(("%.2f"%getPrecise(xcr)))+" , "+str(("%.2f"%getPrecise(ycr)))+")"
