using namespace std;

int main()
{
    int n,m,q,i;
    long long int mul = 1 ;

    	scanf("%d",&a[i]);
    	
    }

    int post[n];
    pre[0] = 1;

    	pre[i] = ((a[i-1])*(pre[i-1]))%m;

    }

    {
    	post[i] = ((a[i+1])*(post[i+1]))%m;
    	//cout<<post[i]<<endl;
    }
    
    
    
    while(q--)
    {
    	long long int l,ans;
