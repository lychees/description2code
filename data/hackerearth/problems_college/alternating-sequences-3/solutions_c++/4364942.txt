#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
 ll n;
 cin>>n;
 ll a[n];
 ll dp[n];


for(ll i=0;i<n;i++){
cin>>a[i];
dp[i]=1;
}

for(ll i=0;i<n;i++)
{   ll ma=0;
    for(ll j=0;j<i;j++)
    {
        if((abs(a[i])>abs(a[j]))&&(a[i]*a[j]<0)&&(dp[j]>ma))
            ma=dp[j];
        dp[i]=ma+1;
    }
}
 ll ma=dp[0];
for(ll i=0;i<n;i++)
    if(dp[i]>ma)
        ma=dp[i];
    cout<<ma<<endl;

}
