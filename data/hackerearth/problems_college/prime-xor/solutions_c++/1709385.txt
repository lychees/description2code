#include<bits/stdc++.h>
using namespace std;
long int  p[10000001];
void prime()
{
p[0]=1; p[1]=1; /// 1 depects no a prime no,.....
long int i,j;
for(i=2;i<10000001;i++)
{
if(p[i]==0)
{
for(j=2*i;j<10000001;j+=i) p[j]=1;
}
}
}
long int compute(long int b)
{
long int i ,sum=0;
for(i=3;i<=b;i++) if(p[i]==0) sum++;
return sum;
}
using namespace std;
int main()
{
prime();
int t; cin>>t; while(t--)
{
long int a,b,k=0;
cin>>a>>b;
if(a!=1&&a!=2)
cout<<"0\n";
else
{
k  = compute(b);
cout<<k<<"\n";
}
}
return 0;
}
