#include <bits/stdc++.h>
using namespace std;

int main()
{
    int q, o, c, top = -1;
    int a[100000];
    scanf("%d",&q);
    while(q--){
    	scanf("%d",&o);
    	if(o==1){
    		if(top >= 0){
    			printf("%d\n",a[top--]);
    		}
    		else
    		printf("No Food\n");
    	}
    	else{
    		scanf("%d",&c);
    		a[++top] = c;
    	}
    }
    return 0;
}
