#include <iostream>
using namespace std;

int main()
{
	long long int N, T, K, arr[100000] =  {0}, count, i, flag, add;
	
	cin >> T;
	while(T--) {
		count = 0;
		cin >> N >> K;
		i = 0;
		while(i < N) {
			cin >> arr[i];
			i++;
		}
		while(true) {
			flag = 0;
			for(i = 1; i < N; i++) {
				
				if(arr[i] <= arr[i-1]) {
					add = (arr[i-1] - arr[i]) / K + 1;
					arr[i] += K * add;
					count+= add;
					flag = 1;
				}
			}
			if(flag == 0)
				break;
		}
		cout << count << endl;
		
	}
    return 0;
}
