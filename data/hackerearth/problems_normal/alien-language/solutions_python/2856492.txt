test=int(raw_input())
while(test):
	test-=1
	s=raw_input()
	r=raw_input()
	lens=len(s)
	lenr=len(r)
	flag=0
	ls=[0 for _ in range(26)]
	lr=[0 for _ in range(26)]
	for i in xrange(lens):
		ls[ord(s[i])-97]+=1
	for i in xrange(lenr):
		lr[ord(r[i])-97]+=1
	for i in range(26):
		if(ls[i]!=0 and lr[i]!=0):
			print "YES"
			flag=1
			break
	if(flag==0):
		print "NO"
