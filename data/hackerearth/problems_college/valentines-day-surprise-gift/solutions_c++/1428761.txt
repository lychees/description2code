#include<bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair

typedef long long LL;

LL R[20];
LL Q=1000000007;

long long binpow(long long val, long long deg, long long mod) {
        if (!deg) return 1 % mod;
        if (deg & 1) return binpow(val, deg - 1, mod) * val % mod;
        long long res = binpow(val ,deg >> 1, mod);
        return (res*res) % mod;
}


LL f(LL n, LL t){
    LL te = n/t;
    if (n%t != 0){
        te++;
    }
    return te;
}

LL gcd(LL a, LL b){
    if (b == 0){
        return a;
    }
    return gcd(b, a%b);
}

LL val(LL n)
{
    if (n <= 10){
        return R[n];
    }
    LL ans = 1;
    ans *= binpow(2, f(n, 4), Q);
    ans %= Q;
    ans *= val(f(n, 5));
    ans %= Q;
    ans *= val( f(n, 10) ) ;
    ans %= Q;
    return ans;
}

int main(){
    R[0] = R[1] = R[2] = 1;
    R[3] = 3;
    R[4] = 3;
    R[5] = 5*3;
    R[6] = 5*3;
    R[7] = 3*5*7;
    R[8] = 3*5*7;
    R[9] = 3*5*7;
    R[10] = 3*5*7;
    //cout<<val(15)<<"\n";
    LL test;
    cin>>test;
    while (test--){
        LL n;
        cin>>n;
        LL r;
        r = val(n);
        //cout<<r<<"\n";
        LL tt = -1;
        for (LL i=1;i<=5;i++){
            for (LL j=1;j<=5;j++){
                if (gcd(i, j)==1){
                    LL temp = binpow(i*r, j*r, Q);
                    tt = max(tt, temp);
                }
            }
        }
        cout<<tt<<"\n";
    }
    return 0;
}
