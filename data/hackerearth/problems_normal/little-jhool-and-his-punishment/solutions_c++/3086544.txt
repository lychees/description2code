#include <iostream>
using namespace std;
#define min(x,y) (x>y) ? (x):(y)
#define min(x,y) (x<y) ? (x):(y)
int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
    	int n,b,g;
    	cin>>n>>b>>g;
    	int mn = min(b,g);
    	int mx = max(b,g);
    	if(mx>(mn+1))
    		cout<<"Little Jhool wins!\n";
    	else
    		cout<<"The teacher wins!\n";
    }
    return 0;
}
