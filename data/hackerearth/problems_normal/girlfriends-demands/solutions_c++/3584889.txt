#include<iostream>
using namespace std;
#include<string>
#include<algorithm>
int main()
{
	string s;
	int n;
	cin>>s>>n;
	int len=s.size();
	unsigned long long int a[n],b[n];
	for(int i=0;i<n;i++)
	   cin>>a[i]>>b[i];
	for(int i=0;i<n;i++)
	{
		unsigned long long int x,y;
		x = (a[i]-1)%len;
		y = (b[i]-1)%len;
		if(s[x]==s[y])
			cout<<"Yes"<<endl;
		 else
		     cout<<"No"<<endl;
    }
}
