def sd(n):
    r = 0
    while (n > 0):
        r += n % 10
        n /= 10
    return r


def d(n):
    if n <= 10:
        return 0.25
    elif n >= 11 and n <= 20:
        return 0.3
    else:
        return 0.2


n = input()
p = map(int, raw_input().split())
l, b = map(int, raw_input().split())
a = l*b
ans = 0
for item in p:
    ans += int(a*(1-d(item))*item*sd(item))
print ans
