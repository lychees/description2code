m, n = map(int, raw_input().split(" "))

m -= n


def binomial(n, k):
    if (n < k):
        return 0
    if (k == 0 or k == n):
        return 1
    if (k > n-k):
        k = n-k
    t = 1
    for i in range(1, k+1):
        t *= n-k+i
        t /= i
    return t


if (m < 0):
    print -1
else:
    print binomial(m+n-1, n-1) % 10000007
