#include<bits/stdc++.h>
#define ll long long int
int main()
{
	int t,n,a0,a1,c,rem,y,anow;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d%d%d",&a0,&a1,&c,&n);
		if(n==1)
		rem=a0%41;
		else
		{
		
		rem=(a0*10+a1)%41,anow,y=n-2;
		while(y--)
		{
			anow=(a1*c+a0)%10;
			a0=a1;
			a1=anow;
			rem=(rem*10+anow)%41;
		}
		}
		if(rem==0)
		{
			printf("YES\n");
		}
		else
		{
			printf("NO\n");
		}
	}
	return 0;
}
