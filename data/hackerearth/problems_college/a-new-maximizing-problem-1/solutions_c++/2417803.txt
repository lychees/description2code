#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
#define MOD 1000000007
int main()
{
	int t,n,i,j;
	long long int arr[100100],res,sum;
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		sum=0;
		for(i=0;i<n;i++){
			scanf("%lld",arr+i);
		}
		sort(arr,arr+n);
		j=1;
		res=0;
		for(i=0;i<n;i++){
			if(j<n)
			{
				res=(res+(MOD-arr[i]))%MOD;
				j+=2;
			}
			else
			{
				if(arr[i]>0)
					res=(res+arr[i])%MOD;
				else
					res=(res+MOD+arr[i])%MOD;
			}
			if(arr[i]>0)
				sum=(sum+arr[i])%MOD;
			else
				sum=(sum+(MOD-arr[i]))%MOD;
		}
		res=(res*sum)%MOD;
		printf("%lld\n",res);
	}
    return 0;
}
