#include<bits/stdc++.h>
using namespace std;
const int N=123456;
int dp[N][4];
int a[N][4];
map<string,char> m;

int main() {
   int t;
   cin>>t;
   while(t--){
   int n;
   scanf("%d",&n);
   for(int i=1;i<=n;i++)
   {
    for(int j=1;j<=3;j++)
      scanf("%d",&a[i][j]);
   }
   for(int i=1;i<=3;i++)
    dp[1][i]=a[1][i];
   for(int i=2;i<=n;i++)
   {
    for(int j=1;j<=3;j++)
    {
      if(j==1)
      {
        dp[i][j]=a[i][j]+min(dp[i-1][2],dp[i-1][3]);
      }
      else if(j==2)
      {
        dp[i][j]=a[i][j]+min(dp[i-1][1],dp[i-1][3]);
      }
      else if(j==3)
      {
        dp[i][j]=a[i][j]+min(dp[i-1][2],dp[i-1][1]);
      }
    }
   }
   int m=1e9;
   for(int i=1;i<=3;i++)
    m=min(dp[n][i],m);

  printf("%d\n",m);
}
  
}
