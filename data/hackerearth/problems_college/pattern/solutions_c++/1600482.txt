#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,flag;
cin>>n;
for(int i=0;i<n;i++)
{

    for(int j=0;j<n;j++)
    {
        if(i==0||i==(n-1))
        cout<<"*";
        else
        {
            flag=n-i-1;
            if(j==flag){
            cout<<"*"<<endl;
            break;
            }
            else
                cout<<" ";
        }
    }
    if(i==0)
        cout<<endl;
}
return 0;
}
