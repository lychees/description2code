#include <iostream>
using namespace std;
#include<string>
#include<algorithm>
#define ll long long
int main()
{
    ll i,j,t,len;
    string s,s2;
    vector<string> svec;
    cin>>s;
    len=s.size();
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>s2;
    	svec.push_back(s2);
    }
    sort(svec.begin(),svec.end());
    for(i=0;i<svec.size();i++)
    {
    	ll count=0;
    	for(j=0;j<len;j++)
    	{
    		if(svec[i][j]==s[j]){count++;}
    	}
    	if(count==len){cout<<svec[i];break;}
    }
    if(i==svec.size()){cout<<s;}
    return 0;
}
