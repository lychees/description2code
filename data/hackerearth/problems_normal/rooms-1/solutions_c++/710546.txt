#include <bits/stdc++.h>
#define vi vector<int>
using namespace std;

int main()
{
	int  t;
	cin>>t;
	while(t--)
	{
		int n,j;
		cin>>n;
		vi arr,dep;
		for(int i=0;i<n;++i){
			cin>>j;
			arr.push_back(j);
		}
		for(int i=0;i<n;++i){
			cin>>j;
			dep.push_back(arr[i]+j);
		}
		sort(arr.begin(),arr.end());
		sort(dep.begin(),dep.end());
		int i=0,ans=0,occ=0;
		j=0;
		while(i<n)
		{
			while(dep[j]<=arr[i])
				--occ,++j;
			occ++;
			if(ans<occ)
				ans=occ;
			++i;
		}
		cout<<ans<<endl;
	}
}
