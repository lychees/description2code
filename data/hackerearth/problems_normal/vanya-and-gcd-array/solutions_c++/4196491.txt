#include <iostream>
#define MOD 1000000007
using namespace std;
int gcd ( int a, int b )
{
  int c;
  while ( a != 0 ) {
     c = a; a = b%a;  b = c;
  }
  return b;
}
int a[501];
int N;
long long dp[501][101];
long long rec(int n ,int g){
	if(n>=N-1)
	return 0;
	if(dp[n][g]!=-1)
	return dp[n][g];
	long long ans =0;
 for(int i = n+1;i<N;i++){
 	if(a[i]>a[n]){
 		int val = gcd(g,a[i]);
 		if(val ==1){
 			ans +=( 1+rec(i,val))%MOD;
 			ans%=MOD;
 		}
 		else
 	{	ans+=rec(i,val);
 		ans%=MOD;
 	}
 	}
 }

	return dp[n][g]=ans%MOD;
	
}


int main()
{
   
   cin>>N;
   for(int i=0;i<N;i++)
   cin>>a[i];
   for(int i=0;i<N;i++)
     for(int j=0;j<=100;j++)
     dp[i][j]=-1;
   long long ans =0;
   for(int i=0;i<N;i++)
   {ans+=rec(i,a[i]);
   if(a[i]==1)
   ans+=1;
   ans%=MOD;
   	
   }
   cout<<ans<<endl;
}
