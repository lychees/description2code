#include <bits/stdc++.h>
using namespace std;
string s;
    
    set<long long> ss;
int main()
{
    
    cin>>s;
    
    for(int i=0;i<s.size();i++)
    {
    	long long st=0;
    	for(int j=i;j<s.size();j++)
    	{
    		st=st*173+s[j];
    		ss.insert(st);
    	}
    }
    cout<<ss.size()<<endl;
    return 0;
}
