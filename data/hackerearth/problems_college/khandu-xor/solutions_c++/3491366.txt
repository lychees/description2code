#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s1,s2,s3;
    int t,l,flag,i;
    cin>>t;
    while(t--)
    {
    	cin>>s1>>s2;
    	s3=s1+s2;
    	flag=0;
    	l=s3.size();
    	for(i=0;i<l/2;i++)
    	{
    		if(s3[i]!=s3[l-1-i])
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==1)
    	{
    		s3=s2+s1;
    		flag=0;
    		l=s3.size();
    		for(i=0;i<l/2;i++)
    		{
    			if(s3[i]!=s3[l-1-i])
    			{
    				flag=1;
    				break;
    			}
    		}    		
    	}
    	if(!flag)
    	{
    		cout<<"YES"<<endl;
    	}
    	else
    	{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
