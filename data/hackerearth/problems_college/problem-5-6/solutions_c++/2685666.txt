using namespace std;
int arr[1009];
int dp[1009];
int n;
int recur(int s)
{
    if(s < 0)return -mod;
    if(s==0)return 0;
    if(dp[s]!=-1)return dp[s];
    int ret=0;
    for(int i=1;i<=n;i++)
    {
        ret = max(ret , recur(s-i) + arr[i]);
    }
    dp[s]=ret;
    return ret;
}
int main()
{
    int t,i,j,m,a,b,x;
    cin>>t;
    while(t--)
    {
        memset(dp,-1,sizeof(dp));
        cin>>n>>x;
        for(i=1;i<=n;i++){
            cin>>arr[i];
            arr[i]-=x;
        }
        int ans =recur(n);
        cout<<ans<<endl;
    }




    return 0;
}
