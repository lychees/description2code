#include <iostream>
using namespace std;

int main()
{
    int t;
    long long int m,n,k,r;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	r=0;
    	if(m==1||n<m)
    	{
    		cout<<n<<"\n";
    		continue;
    	}
    	while(n>0)
    	{
    		r=r+n%m;
    		n=n/m;
    	}
    	cout<<r<<"\n";
    }
    return 0;
}
