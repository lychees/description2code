#include<bits/stdc++.h>
using namespace std;
long long num=1e9+7;
long long pwr(long long a,long long b)
{
	if(b==0)return 1;
	long long temp=pwr(a,b/2);
	temp=(temp*temp)%num;
	if(b%2)temp=(temp*a)%num;
	return temp;
}
main()
{
	int tc;
	scanf("%d",&tc);
	while(tc--)
	{
		long long n;
		scanf("%lld",&n);
		long long ans=(pwr(3,n)-((((3*((pwr(2,n)-2+num)%num))%num)+3)%num)+num)%num;
		printf("%lld\n",ans);
	}
}
