#include <iostream>
#include <vector>
using namespace std;

vector<int> e[10005];
int q[10005];
int ans[10005];

int k;

int solve(int v, int parent) {
	q[v] = 1;
	ans[v] = 0;
	for(int i = 0; i < e[v].size(); i++) {
		int v0 = e[v][i];
		if(v0 == parent) continue;
		solve(v0, v);
		q[v] += q[v0];
		ans[v] += ans[v0];
	}
	if(q[v] >= k) {
		q[v] = 0;
		ans[v]++;
	}
}

int main()
{
    int n;
    cin >> k >> n;
    for(int i = 2; i <= n; i++) {
    	int x;
    	cin >> x;
    	e[x].push_back(i);
    	e[i].push_back(x);
    }
    solve(1, 0);
    cout << ans[1] << endl;
    return 0;
}
