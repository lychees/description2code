#include<iostream>
using namespace std;
long int reverse(long int x)
{
	long int c=0;
	while(x!=0)
		{
			c=c*10+(x%10);
			x=x/10;
		}
		return c;
}
int main()
{
	long int N,a,b,s;
	int i;
	cin>>N;
	for(i=0;i<N;i++)
	{
		cin>>a>>b;
		a=reverse(a);
		b=reverse(b);
		s=a+b;
		cout<<reverse(s)<<endl;
	}
	return 0;
}
