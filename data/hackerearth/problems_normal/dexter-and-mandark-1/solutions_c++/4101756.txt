#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long i,j,m,n;
        scanf("%lld%lld",&n,&m);
        long long dp[2][n+1];
        for(i=0;i<=n;i++)
        {
            dp[0][i]=0;
            dp[1][i]=0;
        }
        dp[0][0]=1;
        for(i=1;i<=n;i++)
        {
            for(j=0;j<i;j++)
            {
                long long prev_dscore,prev_mscore;
                if(dp[0][j]==0)
                {
                    dp[1][j]=0;
                    continue;
                }
                prev_dscore=j;
                prev_mscore=(i-1-j);
                if(prev_dscore>=m*(prev_mscore+1))
                {
                    dp[1][j]=(dp[1][j]+dp[0][j])%mod;
                }
                if((prev_dscore+1)>=m*prev_mscore)
                {
                    dp[1][j+1]=(dp[1][j+1]+dp[0][j])%mod;
                }
            }
            for(j=0;j<=n;j++)
            {
                dp[0][j]=dp[1][j];
                dp[1][j]=0;
            }
        }
        long long ans=0;
        for(i=n/2+1;i<=n;i++)
        {
            ans=(ans+dp[0][i])%mod;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
