#include <iostream>
#include<string>
using namespace std;

int main()
{int t;
    cin>>t;
    while(t--)
    {string s;
    	cin>>s;
    	string tmp="";
    	int c=0,ans=0;
    	for(int i=0;i<s.length();i++)
    	{
    		if(s[i]!='*')
    			tmp+=s[i];
    		else
    		{
    			for(int j=0;j<tmp.length();j++)
    		{
    			if(tmp[j]=='O')
    				c++;
    			else
    			{
    				c=0;
    				break;
    			}
    		}ans+=c;c=0;tmp="";
    		}
    	}
    	for(int j=0;j<tmp.length();j++)
    		{
    			if(tmp[j]=='O')
    				c++;
    			else
    			{
    				c=0;
    				break;
    			}
    		}ans+=c;c=0;tmp="";
    	cout<<ans<<endl;
    }return 0;
}
