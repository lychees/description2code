#include <iostream>

using namespace std;

int main()
{
	int n=0,i,j,a[1000];
	unsigned long int f[1000][1000];
	cin>>n;
	for(i=0;i<n;i++)
	cin>>a[i];
	for(i=0;i<n;i++)
	{
		f[i][0]=0;
		f[i][1]=1;
		for(j=2;j<a[i];j++)
			f[i][j]=f[i][j-1]+f[i][j-2];
	}
	for(i=0;i<n;i++)
	{
		cout<<"\n";
		for(j=a[i]-1;j>=0;j--)
		cout<<f[i][j]<<" ";
	}
	}
