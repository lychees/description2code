#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
   ll i,t,n;
   vector<ll> v(107);
   v[1]=1;
   v[2]=2;
   for(i=3;i<=106;i++)
   {
   	v[i]=v[i-1]+v[i-2];
   }
   scanf("%lld",&t);
   while(t--)
   {
   	scanf("%lld",&n);
   	printf("%lld\n",v[n]);
   }
    return 0;
}
