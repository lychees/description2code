#include <iostream>
#include <math.h>
using namespace std;

int main()
{
  int T,N,M,X[2000],Y[2000],D,count;
  cin>> T;
 
  for(int i=0; i<T;i++){
  	cin>>N>>M>>D;
  	count=0;
  	for(int k=0;k<(N+M);k++){
  		cin>>X[k]>>Y[k] ;
  	}
  	for(int l=N;l<(N+M);l++){
  		for(int j=0;j<N;j++){
  			if ((abs(X[l]-X[j])+abs(Y[l]-Y[j]))<=D){
  				count++;
  				break;
  			}
  		}
  	}
  	if (count>M/2){
  		cout<<"YES"<<endl;
  	}
  	else{
  		cout<<"NO"<<endl;
  	}
  }
}
