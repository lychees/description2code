#include <iostream>
using namespace std;
string s[105];
int k;
bool bs(int i,int j)
{
	if(i==k-1 && j==k-1) return true;
	if(s[i][j]=='1')
	{
	 if(i==k-1) return bs(i,j+1);
	else if(j==k-1) return bs(i+1,j);
	else return (bs(i+1,j) || bs(i,j+1));
	}
	else return false;
}
int main()
{
    int t;
    scanf("%d",&t);
    int i;
    while(t--)
    {
    	int n;
    	scanf("%d %d",&n,&k);
    	
    	for(i=0;i<n;i++) cin>>s[i];
    	if(s[k-1][k-1]=='0') printf("%d\n",-1);
    	else
    	{	
    		 
    	
    	if(bs(0,0)) {k--;k*=2;printf("%d\n",k);}
    	else printf("%d\n",-1);
    	}
    	
    }
    return 0;
}
