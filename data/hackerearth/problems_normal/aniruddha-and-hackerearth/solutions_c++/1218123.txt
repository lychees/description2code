#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define MOD 1000000007

ll gcd(ll a,ll b)
{
	return (b==0)?a:gcd(b,a%b);
}

ll DP[1000010];

void compute()
{
	DP[1] = 2;
	DP[2] = 3;
	for(int i=3;i<=1000000;++i)
	{
		DP[i] = DP[i-1] + DP[i-2];
		DP[i] %= MOD;
	}
	for(int i=2;i<=1000000;++i)
	{
		DP[i] += DP[i-1];
		DP[i] %= MOD;
	}
}

int main()
{
	int t,n;
	compute();
	cin >> t;
	while(t--)
	{
		cin >> n;
		cout << DP[n] << endl;
	}
	return 0;
}
