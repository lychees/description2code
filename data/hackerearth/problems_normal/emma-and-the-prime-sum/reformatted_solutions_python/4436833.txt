def mysieve(n):
    for i in xrange(2, n):
        if i * i <= n and not prime[i]:
            for j in xrange(i * i, n, i):
                prime[j] = 1


def comp():
    for i in xrange(2, n):
        if not prime[i]:
            pre[i] = pre[i-1] + i
        else:
            pre[i] = pre[i - 1]
