#include<iostream>
#include<cstdio>
#include<vector>
#include<cmath>
using namespace std;
vector<int> v(1000000);
vector<int> res(78498);
int prime(long long n)
{
    long long i;
    if(n==0)
        return 0;
    if(n==2)
        return 1;
    for(i=2;i<=sqrt(n);i++)
    {
        if(n%i==0)
            return 0;
    }
    return 1;
}
int main()
{
    int i,j,index=0,t,m,n,flag;
    long long sum;
    for(i=2;i<=1000000;i++)
    {
        if(v[i]==0)
        {
            res[index]=i;
            index++;
            for(j=2*i;j<=1000000;j=j+i)
            {
                v[j]=1;
            }
        }
    }
    scanf("%d",&t);
    while(t--)
    {
        sum=0;
        scanf("%d%d",&m,&n);
        for(i=0;i<78498;i++)
        {
            if(res[i]>n)
                break;
            else
            {
                if((res[i]>=m)&&(res[i]<=n))
                    sum+=res[i];
            }
        }
        flag=prime(sum);
        if(flag)
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
