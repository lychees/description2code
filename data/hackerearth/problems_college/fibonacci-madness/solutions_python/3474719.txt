#print 'Hello World!'
t=input()
while t>0:
	t-=1
	n=input()
	if n==1:
		print"0"
		continue
	elif n==2:
		print"1 0"
		continue
	l=[0]*n
	l[0]=0
	l[1]=1
	for i in range(2,n):
		l[i]=l[i-1]+l[i-2]
	l=l[::-1]
	print " ".join(str(i) for i in l)
