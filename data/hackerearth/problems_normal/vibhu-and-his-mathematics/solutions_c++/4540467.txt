#include <iostream>
#include <cmath>
using namespace std;
long long int A[1000001];
long long int buy(int n)
{
    if (A[n] != -1)
    	return A[n];
    A[n] = (buy(n-1) + (n-1)*buy(n-2)%1000000007)%1000000007;
    return A[n];
}

int main() {
    int T;
    cin >> T;
    A[0] = 0;
    A[1] = 1;
    A[2] = 2;
    for (int j=3;j<=1000000;++j)
    	A[j]=-1;
    for (int i=0;i<T;++i)
    {
        int N;
        cin >> N;
        cout << buy(N) << endl;
    }
    return 0;
}
