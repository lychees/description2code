#include <iostream>
using namespace std;
typedef long long int ll;
int main()
{
    ll t;
    cin >> t;
    while(t--) {
    	ll n, m, x, d;
    	cin >> n >> m >> x;
    	ll s = x * (n + 1);
    	while(n--) {
    		cin >> d;
    		s-=d;
    	}
    	if(s > m) {
    		cout << "Impossible\n";
    	}
    	else if(s <= 1) {
    		cout << "1\n";
    	}
    	else {
    		cout << s << endl;
    	}
    }
    return 0;
}
