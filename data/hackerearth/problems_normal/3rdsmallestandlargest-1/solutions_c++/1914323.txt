#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string s;
    while(t--){
    	cin>>s;
    	if(s.length()<3){
    		cout<<"Not possible!"<<endl;
    		continue;
    	}
    	sort(s.begin(),s.end());
    	next_permutation(s.begin(),s.end());
    	next_permutation(s.begin(),s.end());
    	cout<<s<<" ";
    	
    	sort(s.begin(),s.end());
		reverse(s.begin(),s.end());
    	prev_permutation(s.begin(),s.end());
    	prev_permutation(s.begin(),s.end());
    	cout<<s<<endl;
    	
    }
    return 0;
}
