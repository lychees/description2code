#include <iostream>
using namespace std;

int main()
{
//    cout << "Hello World!" << endl;
    int T,A,B;
    cin>> T;
    long sum=0;
    	int rsum[7]={0,1,2,6,6,3,9};
    for ( int i=0;i<T;i++){
    	
    	cin>> A>>B;
    	
    	sum=0;
    
    	for (int j=A;j<=B;j++){
    		
    	    if( j>=1 and j <=6 ){
    	    	sum+=rsum[j];
    	    }else if( j==0){
    	    	sum+=0;
    	    }else{
    	    	
    	    	sum+=9*(B-j+1);
    	    	break;
    	    }
    		
    	}
    	cout<<sum<<endl;
    	
    }
    return 0;
}
