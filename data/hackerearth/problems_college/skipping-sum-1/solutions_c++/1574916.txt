#include <bits/stdc++.h>
using namespace std;
long long arr[57][100005];

int main()
{
    long long brr[100001]={0},n,q,i,x,y,K,z,j,temp,count,india,sum;
   
    cin>>n>>q;
    for(i=1;i<=n;i++)
    {
    	cin>>brr[i];
    }
    
    
    count=1;
    for(K=1;K<=10;K++)
    {
    	for(i=1;i<=K;i++)
    	{
    		temp=0;
    		for(j=i;j<=n;j=j+K)
    		{
    			arr[count][j]=temp+brr[j];
    			temp=arr[count][j];
    		}
    		count++;
    	}
    }
    
    while(q--)
    {sum=0;
    	cin>>x>>y>>z;
    	
    	if(x%z==0)
    		india=(z*(z-1))/2,india=india+z;
    	else
    		india=(z*(z-1))/2,india=india+x%z;
    	while(arr[india][y]==0)y--;
    	sum=arr[india][y]-arr[india][x]+brr[x];
    	cout<<sum<<endl;
    }
    return 0;
}
