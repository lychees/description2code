#include <bits/stdc++.h>
using namespace std;
long long a[1000001];

long long solve(long long n)
{
	long long r;
	while(n>0)
	{
		r=n%10;
		if(r==7)
			return 1;
		n=n/10;
	}
	return 0;
}

void preprocessor()
{
	long long i,flag,j;
	memset(a,0,sizeof(a));
	for(i=2;i<1000001;i++)
	{
		if(a[i]==0);
		{
			for(j=2;i*j<1000001;j++)
			{
				a[i*j]=1;
			}
		}
	}
	for(i=2;i<1000001;i++)
	{
		if(a[i]==1)
		{
			flag=solve(i);
			if(flag==1)
			{
				a[i]=2;
			}
		}
	}
}

int main()
{
    long long t,l,r,i,ans,k;
    preprocessor();
    /*for(i=0;i<100;i++)
    {
    	cout<<i<<" "<<a[i]<<endl;
    }*/
    cin>>t;
    while(t--)
    {
    	cin>>l>>r;
    	ans=0;
    	for(i=l;i<=r;i++)
    	{
    		if(a[i]==2)
    			ans++;
    	}
    	if(ans==0)
    		ans=-1;
    	cout<<ans<<endl;
    }
    return 0;
}
