int main()
{

    ll n , m  , k , d , t , tem1 , tem2 , tem3 , tem4 , y = 1, sum = 0 , ans = 0;

    cin >>   t             ;

    while(t--){

        cin >> n;

        ll low = 1, high = 1e9, mid;
        while(low < high){

            mid = low + (high - low) / 2;
            sum = mid * (mid - 1) + mid;
            if(sum == n){high = mid; break;}
            else if (sum > n) high = mid;
            else low = mid + 1;
        }
        if(high * (high - 1) + high > n ) high--;
        tem1 = n - high * (high - 1) - high;
        int num = (high - 1) * 2 + 1;
        if(tem1 == 0) cout << num << "\n";
        else {

            ll ex = 0;
            if(tem1 % high != 0) ex = 1;
            cout << num + tem1 / high + ex << "\n";
        }
    }


    return 0;
}
