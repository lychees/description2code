#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
    int t,n,i;
    long long x,y,cap1,cap2,count1,count2,arr[100000];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld%lld%d",&x,&y,&n);
        i=0;
        cap1=0;cap2=0;count1=count2=0;
        while(i<n)
        {
            scanf("%lld",&arr[i]);
            i++;
        }
        sort(arr,arr+n);
        i=0;
        while(cap1<x&&i<n)
        {
            count1++;
            cap1=cap1+arr[i];
            i++;

        }
        if(cap1>x)
            count1--;
        i=0;
        while(cap2<y&&i<n)
        {
            count2++;
            cap2=arr[i]+cap2;
            i++;
        }
        if(cap2>y)
            count2--;
        if(count1>count2)
            printf("Chef Won\n");
        else if(count2>count1)
            printf("Jhool Won\n");
        else
            printf("Tie\n");

    }



    return 0;
}
