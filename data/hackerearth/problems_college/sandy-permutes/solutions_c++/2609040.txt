#include <bits/stdc++.h>

using namespace std;

int main()
{
     int n,k,i,j;
    cin>>n>>k;
    long long int a[n];
    for(i=0;i<n;i++)
        cin>>a[i];
   if(k>=n)
   k=n-1;
    for(i=0;i<k;i++){
        int temp=0,m=i;
        for(j=i;j<n;j++)
        {
            if(a[j]>temp){
                temp=a[j];m=j;}
               
               }
       
        
      if (m==i && k<n)
        k++;
          
          a[m]=a[i];
          a[i]=temp;
       
        
    }
    for(j=0;j<n;j++)
        cout<<a[j]<<" ";

    return 0;
}
