#include <iostream>
#include <limits>
#include <algorithm>

bool canReachFromAtoB(int p, int x, int y, int u, int v)
{
	int time = std::numeric_limits<int>::max();
	
	if (u - x == 0 && v - y == 0)
	{
		time = 0;
	}
	else if (std::abs(u - x) == std::abs(v - y))
	{
		time = 1;
	}
	else if (((u - y) & 1) == ((v - x) & 1))
	{
		time = 2;
	}
	
	return time < p;
}

int main()
{
	int t, p, x, y, u, v;
	std::cin >> t;
	
	while (t--)
	{
		std::cin >> p >> x >> y >> u >> v;
		std::cout << (canReachFromAtoB(p, x, y, u, v) ? "Counter-Terrorists Win !" : "Terrorists Win !") << std::endl; 
	}
	
	return 0;
}
