int main()
{
    int t;
    scanf("%d", &t);
    deque<ll> d;
    while(t--)
    {
        ll n;
        cin >> n;
        if(n < 1){
        	printf("0\n");
        	continue;
        }
        d.clear();
        for(ll i = 1; i <= n; ++i) d.push_back(i);
        while(!d.empty()){
            ll cur = d.front();
            d.pop_front();
            if (!d.empty()) d.pop_front();
            else {
                printf("%d\n", cur);
                break;
            }
            d.push_back(cur);
        }
    }
    return 0;
}
