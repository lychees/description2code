#include<iostream>
#include<cstring>
using namespace std;
int IsCharDuplication(string s)
{
    int Count = 0;
 for (int i = 0; i < s.length(); i++)
    for (int j = i+1; j < s.length(); j++)
      if (s[i] == s[j])
      {
        //return true;
        Count++;
        int length = (s.length()-Count);
        return length;
      }
      return s.length();
}

int main()
{
    string A;
    cin>>A;

}
