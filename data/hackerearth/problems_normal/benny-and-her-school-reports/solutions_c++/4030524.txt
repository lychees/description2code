#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;
#define ll long long
int main()
{
ll i,t,j,k,n,m,x,a;
ll r;
cin>>t;
for(i=0;i<t;i++)
{
	cin>>n>>m>>x;
	ll s=0;
	for(j=0;j<n;j++)
	{
		cin>>a;
		s+=a;
	}
	ll q=(n+1)*x;
	if(q<=s)
	cout<<1<<"\n";
	else
	{
		if(q-s>m)
		cout<<"Impossible\n";
		else
		cout<<q-s<<"\n";
	}
}
	return 0;
	
}
