#include <bits/stdc++.h>
using namespace std;
int bit[100005][26];int n;
void update(int x,int val,int pos)
{
    for(;x<=n;x+=x&(-x))
        bit[x][pos]+=val;
}
int query(int x,int pos)
{
    int sum=0;
    for(;x>0;x-=x&-x)
        sum+=bit[x][pos];
    return sum;
}
int main()
{
    int q;
    cin>>n>>q;
    string s;
    cin>>s;
    int i;
    for(i=0;i<s.length();i++)
    {
        update(i+1,1,s[i]-'a');
    }
    while(q--)
    {
        int t1;
        cin>>t1;
        if(t1==1)
        {
            int pos;char ch;
            cin>>pos>>ch;
            update(pos,1,ch-'a');
            update(pos,-1,s[pos-1]-'a');
            s[pos-1]=ch;
        }
        else if(t1==2)
        {
            int l,r;
            cin>>l>>r;int od=0;
            for(i=0;i<26;i++)
            {
                int ni=query(r,i)-query(l-1,i);
                if(ni%2==1)
                    od++;
            }
            if(od<=1)
                cout<<"yes\n";
            else
                cout<<"no\n";
        }

    }


}
