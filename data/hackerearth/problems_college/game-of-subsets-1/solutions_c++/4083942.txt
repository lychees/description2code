#include<bits/stdc++.h>
//zone
using namespace std;
long long powmod(int p)     
{
    long long x,ans;
    ans=1; x=3;
    while(p>0)
    {
        if (p%2==1)
            ans=(ans*x)%1000000007;
        x=(x*x)%1000000007;
        p/=2;
    }
    return ans%1000000007;
}

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        int n;
        cin >> n;
        long long ans=powmod(n);
        cout<<ans<<endl;
    }
    return 0;
}
