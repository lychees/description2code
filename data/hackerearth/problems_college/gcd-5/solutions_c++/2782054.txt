#include <iostream>
using namespace std;

int main()
{long long int t,n,i,a[100000],flag;
    cin>>t;
    while(t--)
    {flag=0;
    	cin>>n;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	for(i=0;i<n-1;i++)
    	{
    		if(a[i]>a[i+1])
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==0)
    	cout<<"YES\n";
    	else
    	
    	cout<<"NO\n";
    }
    return 0;
}
