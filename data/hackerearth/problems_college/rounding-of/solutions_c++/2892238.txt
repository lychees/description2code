#include <bits/stdc++.h>
using namespace std;
int main() {
	int tc,n,b;
	cin>>tc;
	while(tc--) {
		cin>>n>>b;
		int a = n/b;
		int e = a+1;
		
		int c = a*b,f = e*b;
		
		int x1 = abs(c-n);
		int x2 = abs(f-n);
		//cout<<x1<<" "<<x2<<"\n";
		if(x1 < x2)
		cout<<a*b<<endl;
		else
		cout<<e*b<<endl;
	}
	return 0;
}
