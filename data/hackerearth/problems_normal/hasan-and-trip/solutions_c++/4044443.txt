#include <iostream>
#include <vector>
#include <utility>
#include <cmath>
using namespace std;
vector<pair<pair<double,double>,double> > city;
vector<double> happy;
double solve(int a)
{
	//cout<<a<<endl;
	if(a==0)
		return city[0].second;
	if(happy[a]!=-1e9)
		return happy[a];
	double res=-1000000000.000000;
	double x=city[a].first.first,y=city[a].first.second,xx,yy,dist=0.0000000,unhappy;
	int p1;
	for(p1=a-1;p1>=0;p1--)
	{
		xx=city[p1].first.first;
		yy=city[p1].first.second;
		dist=(x-xx)*(x-xx)+(y-yy)*(y-yy);
		unhappy=sqrt(dist);
		res=max(res,solve(p1)+city[a].second-unhappy);
	}
	return happy[a]=res;
}
int main()
{
    int p1,p2,n;
    cin>>n;
    city.resize(n);
    happy.resize(n,-1e9);
    for(p1=0;p1<n;p1++)
    	cin>>city[p1].first.first>>city[p1].first.second>>city[p1].second;
    double result=0.0000000;
    result=solve(n-1);
    printf("%.6lf",result);
    return 0;
}
