#include <iostream>
#include <cmath>
using namespace std;
float check_the_point(int x1,int y1,int x2,int y2,int x3,int y3)
{
    float M;
    M=abs((float)(x1*(y2-y3)+x2*(y3-y1)+x3*(y1-y2))/2);
    return M;
}

int main()
{
    int t,a,b,c,d,e,f,g,h,i;
    cin>>t;
    int oup[t];
    float r1,r2,r3,r4;
    for(i=0;i<t;++i)
    {
        cin>>a>>b>>c>>d>>e>>f>>g>>h;
        r1=check_the_point(a,b,c,d,e,f);
        r2=check_the_point(a,b,g,h,e,f);
        r3=check_the_point(c,d,e,f,g,h);
        r4=check_the_point(a,b,c,d,g,h);

        if(r1==(r2+r3+r4))
            oup[i]=1;
        else oup[i]=0;
    }
   for(i=0;i<t;++i)
    {
        if(oup[i]==1)
        cout<<"INSIDE\n";
        else
        cout<<"OUTSIDE\n";
    }
    return 0;
}
