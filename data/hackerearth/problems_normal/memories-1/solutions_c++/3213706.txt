#include <bits/stdc++.h>
#include <unistd.h>
using namespace std;
#define ll long long
#define ull unsigned long long
#define inf std::numeric_limits<ll>::max();
#define MOD 1000000007

int pow_2[]={1,2,4,8,16,32,64,128,256,512,1024,2048,4096,8192,16384,32768,65536};

int main(){
	ios::sync_with_stdio(0);
	int t;
	cin>>t;
	for(int z=0;z<t;z++){
		int n,p;
		cin>>n>>p;
		vector< pair<int,int> > non_pair;
		for(int i=0;i<p;i++){
			int a,b;
			cin>>a>>b;
			non_pair.push_back(make_pair(a,b));
		}
		int ans=0;
		for(int i=1;i<pow_2[n];i++){
			ans+=1;
			for(int j=0;j<p;j++){
				if((i>>(non_pair[j].first-1))&1 && (i>>(non_pair[j].second-1))&1){
					ans-=1;
					break;
				}
			}
		}
		cout<<ans<<'\n';
	}
	return 0;
}
