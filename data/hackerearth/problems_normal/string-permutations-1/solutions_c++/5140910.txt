#include <bits/stdc++.h>
using namespace std;

int main()
{
	string str;
	map<string,int>m;
	map<string,int>::iterator ii;
    int n; cin>>n; while(n--)
    {
    	cin>>str;
    	m[str]++;
    }
    int max=0,ans=0;
    for(ii=m.begin();ii!=m.end();ii++)
    {
    	if(ii->second>max) max=ii->second;
    }
    for(ii=m.begin();ii!=m.end();ii++)
    {
    	if(ii->second==max) ans+=max;
    }
    cout<<ans;
    return 0;
}
