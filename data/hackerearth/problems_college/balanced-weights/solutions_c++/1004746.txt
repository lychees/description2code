#include <iostream>
using namespace std;
#define DAP long long int
#define scan(v) scanf("%lld",&v)

DAP gcd(DAP d,DAP b)
{
	DAP r;
	while(d>0)
	{
		r=b%d;
		b=d;
		d=r;
	}
	return b;
}
int main()
{
	DAP t,n,i,v,ans;
	cin>>t;
	while (t--)
	{
		//DAP 
		scan(n);
		for (i=0; i<n;i++)
		{
			if (i==0)
			{
				scan(v);
				ans=v;
			}
			else
			{
				scan(v);
				ans = gcd(ans,v);
			}
		}
		ans = ans*n;
		printf("%lld\n", ans);
			
	}
    
    return 0;
}
