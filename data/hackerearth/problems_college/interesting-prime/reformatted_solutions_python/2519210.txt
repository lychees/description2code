import itertools


def primes(n):
    primfac = []
    d = 2
    while d*d <= n:
        while (n % d) == 0:
            primfac.append(d)
            n //= d
        d += 1
    if n > 1:
        primfac.append(n)
    return primfac


for _ in range(input()):
    stuff = primes(input())
    st = set()
    for L in range(0, len(stuff)+1):
        for subset in itertools.combinations(stuff, L):
            st.add(subset)

    print len(st) - 1
