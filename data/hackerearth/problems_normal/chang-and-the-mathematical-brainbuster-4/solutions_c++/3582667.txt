#include<bits/stdc++.h>
#define ll long long

using namespace std;

int arr[10005];

int main()
{
    int n;
    cin>>n;
   
    for(int i=0;i<n;i++)
        cin>>arr[i];
    sort(arr,arr+n);
    int ans=0;
    for(int i=n-1;i>=0;i--)
    {
    	for(int j=0;j<n-i;j++)
    	{
    		if(3*arr[j]>=arr[j+i])
    		{
    			cout<<n-(i+1);
    			return 0;
    		}
    	}
    }
    cout<<ans;
    
}
