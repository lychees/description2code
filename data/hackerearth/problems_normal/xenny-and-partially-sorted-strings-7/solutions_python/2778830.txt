#!/usr/bin/env python

def main():
    test_cases =  int(raw_input())
    for x in range(0, test_cases):
        N,K,M = map(int,raw_input().split(' '))
        strings = []
        for i in range(0,N): strings.append(raw_input())
        print sorted(strings, key=lambda x: x[0:M])[K-1]



if __name__ == '__main__':
    main()
