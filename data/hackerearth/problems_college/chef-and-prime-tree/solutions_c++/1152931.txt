#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
inline bool isprime(int n){
 if(n==1)
    return false;
 for(int i=2;i<=sqrt(n);i++){
    if(n%i==0)
     return false;
 }
 return true;
}
int main(){
int t;
int x,y;
scanf("%d",&t);
while(t--){
  scanf("%d %d",&x,&y);
  ll ans = 0;
  while(x!=y){
        if(x>y){
          if(isprime(x))
            ans -= x;
        else
            ans += x;
          x = x/2;
        }
        if(x<y){
        if(isprime(y))
            ans -= y;
        else
            ans += y;
          y = y/2;
        }
  }
  if(x==y){
    if(isprime(x))
        ans -=x ;
    else
        ans +=x;
  }
  printf("%lld\n",ans);
}
return 0;
}
