// A O(Logn) complexity program to count set bits in all numbers from 1 to n
#include <stdio.h>
#include<bits/stdc++.h>
#define ll long long int
using namespace std;
ll combi(ll n,ll k)
{
    ll ans=1;
    k=k>n-k?n-k:k;
    ll j=1;
    for(;j<=k;j++,n--)
    {
        if(n%j==0)
        {
            ans*=n/j;
        }else
        if(ans%j==0)
        {
            ans=ans/j*n;
        }else
        {
            ans=(ans*n)/j;
        }
    }
    return ans;
}
