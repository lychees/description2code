#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;
long long int dp[1005][1005] = {0};
int main()
{
    int tc,n,m;
    cin >> tc;
 
    while(tc--){
    cin >> n >> m;
 
    for (int i = 0;i <= 1001 ;i++)
        for (int j = 0;j <= 1001;j++)
            dp[i][j] = 0;
 
    for (int i = 0;i <= n;i++)
        dp[i][0] = 1;
    for (int i = 0;i <= n;i++)
        dp[0][i] = 0;
 
    dp[0][0] = 0;
  /*  for ( int i = 0; i<=n ; i++)
    {
    	for ( int j = 0; j<= n ; j++)
    	{
    		cout << dp[i][j] << " " ; 
		}
		cout << endl; 
	}
	cout << endl; */
 
    for (int i = 1;i <= n;i++)
    {
        for (int j = 1;j <= n;j++)
        {
            if (i < j * m)
                continue;
           // else
                dp[i][j] = (dp[i-1][j] + dp[i][j-1]) % mod;
        }
    }
    /*cout << endl; 
    for ( int i = 0; i<=n ; i++)
    {
    	for ( int j = 0; j<= n ; j++)
    	{
    		cout << dp[i][j] << " " ; 
		}
		cout << endl; 
	}
	cout << endl;*/
 
    long long int ans = 0;
 
    for (int i = 0;i <= n /*&& n-i >= i*m*/;i++)
    {
        if (n-i > i)
            ans += dp[n-i][i];
           ans %= mod;
    }
 
    cout << ans << endl;
    }
 
    return 0;
}
