#include <bits/stdc++.h>
using namespace std;

#define ll long long int

string fun(string curr)
{
	int i;
	int len=curr.size();
	string res="";
	for(i=0;i<len;i++)
	  {
	  	if(i%2==0)
	  	res.push_back(curr[i]);
	  }
	for(i=len-1;i>=0;i--)
	  {
	  	if(i%2)
	  	 res.push_back(curr[i]);
	  }
	  
	  return res;
	
}

int main()
{
	string tmp;
	ll k,i,p;
	cin>>k;
	string str,curr;
	map<int,string> mp;
	cin>>str;
	curr=str;
	int flag=0;
	for(i=1;i<=k;i++)
	{
		 tmp=fun(curr);
		if(tmp.compare(str) == 0)
		{
			flag=1;
			p=i;
			break;
		}
		mp[i]=tmp;
		
		curr=tmp;
		
	}
	
	if(flag)
	{
	//	k%=p;
	cout<<mp[k%p];
	}
	
	else
	{
		cout<<tmp;
	}
	
    
    return 0;
}
