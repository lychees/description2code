#include <bits/stdc++.h>


using namespace std;
const long long N=251 , md = 4294967143;
long long n,m,a[N][N],ans,b[N];
int main()
{
  //  freopen("in","r",stdin);
    cin>>n>>m;
    for (int i=1;i<=m;i++)
        cin>>b[i];
        sort(b+1,b+m+1);
    a[0][1]=1;
    for (int i=1;i<=n;i++)
        for (int j=1;j<=m;j++)
            if (i-b[j]>=0)
            {
                for (int k=1;k<=j;k++)
                    a[i][j]+=a[i-b[j]][k];
            }
    for (int i=1;i<=m;i++)
        ans+=a[n][i];
    cout<<ans;
    return 0;
}
