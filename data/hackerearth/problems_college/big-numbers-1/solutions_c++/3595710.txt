#include <bits/stdc++.h>

using namespace std;

const int MD = 1e9 + 7;

int n, m, k;
int dp[51][2001][100];

int countNos(int curr, int sq, int divi) {
	int& res = dp[curr][sq][divi];
	if(res != -1) {
		return res;
	}
	if(curr == 0) {
		if(sq == k && divi == 0) return 1;
		return 0;
	}
	res = 0;
	for(int i=0; i<10; i++) if(sq + i*i <= k) {
		res = (res + countNos(curr-1, sq+i*i, (divi * 10 + i) % m)) % MD;
	}
	return res;
}

int main(int argc, char const *argv[]){

	memset(dp, -1, sizeof(dp));
	scanf("%d%d%d", &n, &m, &k);
	int ans = 0;
	for(int i=1; i<10; i++) if(i*i <= k) {
		ans = (ans + countNos(n-1, i*i, i%m)) % MD;
	}
	printf("%d\n", ans);
	return 0;
}
