#include <bits/stdc++.h>
using namespace std;

int main()
{long long int t,b,w,x,y,z,d,e,f,ans;
    cin>>t;
    while(t--)
    {
    	cin>>b>>w;
    	cin>>x>>y>>z;
    	d=b*x+w*y;
    	e=b*(y+z)+w*y;
    	f=b*x+w*(x+z);
    if(d<e&&d<f)
    ans=d;
    else if(e<d&&e<f)
    ans=e;
    else ans=f;
    cout<<ans<<endl;
    }
    return 0;
}
