N = int(raw_input())
L = sorted(map(int, raw_input().split()))
Uniq = []
flag = False
result = -1
for x in range(1, N+1):
    if not L[N-x] in Uniq:
        Uniq.append(L[N-x])
        if L[N-x] != L[N-x-1]:
            result = L[N-x]
            flag = True
    if flag:
        break
print result
