#include<bits/stdc++.h>
using namespace std;
void prime();
long long p[1000001],ans[1000001];
int main()
{
	long long int t,x,i;
	cin>>t;
	for(i=0;i<=1000000;i++)
	{
		p[i]=1;
	}
	prime();
	while(t--)
	{
		cin>>x;
		
		cout<<ans[x]<<endl;
	}
}
void prime()
{
   	long long int i,j,count;
	p[0]=0;
	p[1]=0;
	ans[0]=0;
	ans[1]=0;
	for(i=2;i<=1000000;i++)
	{
		count=1;
		if(p[i]==1)
		{
			
			for(j=2*i;j<=1000000;j+=i)
			{
				if(p[j]==1)
				{
					p[j]=0;
					count++;
				}
				p[j]=0;
			}
			ans[i]=count;
			
		}
	}
}
