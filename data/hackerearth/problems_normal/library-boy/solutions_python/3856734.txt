n=int(raw_input())
a=[]
for x in range (0,n):
    a.append(raw_input())
b=sorted(a)
temp=b[0][0]
count=0
flag=1
for x in range (0,n):
    if b[x][0]==temp:
        count=count+1
    if count==10:
        flag=flag+1
        count=0
    if b[x][0]!=temp:
        flag=flag+1
        temp=b[x][0]
        count=0
