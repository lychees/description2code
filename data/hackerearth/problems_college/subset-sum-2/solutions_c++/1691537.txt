#include <iostream>
using namespace std;

int main()
{
	long long mod = 1000000007; 
	int T=0;
    cin>>T;
    while(T--)
    {
    	int n;
    	cin>>n;
    	long long k=1;
    	int i;
    	for(i=0;i<n-1;++i)
    	{
    		k=(k*2)%mod;
    	}
    	long long sum = (n*(n+1))/2;
    	sum = (k*sum)%mod;
    	cout<<sum<<endl;
    	
    }
    return 0;
}
