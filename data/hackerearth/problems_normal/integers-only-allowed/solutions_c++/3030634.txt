#include<stdio.h>

typedef long long ll;

ll gcd(ll a, ll b)
{
    while(b)
    {
        ll temp = b;
        b = a%b;
        a = temp;
    }
    return a;
}

ll mod(ll a)
{
    return (a>=0?a:(-a));
}

int main()
{
   // freopen("C:\\Users\\Nilanshu Rajmane\\Downloads\\ip.txt","r",stdin);
//freopen("C:\\Users\\Nilanshu Rajmane\\Downloads\\output.txt","w",stdout);
    ll n,x,y,prev_x,prev_y,first_x,first_y;
    scanf("%lld",&n);
    ll A=0,border_pts=n,inner_pts;
    n--;
    scanf("%lld%lld",&x,&y);
    first_x = x;
    first_y = y;
    while(n--)
    {
        prev_x = x;
        prev_y = y;
        scanf("%lld%lld",&x,&y);
        A += prev_x*y - x*prev_y;

        border_pts += gcd(mod(x-prev_x),mod(y-prev_y)) - 1;
    }
    border_pts += gcd(mod(x-first_x),mod(y-first_y))-1;
    A += x*first_y - first_x*y;

    A = mod(A/2);

    inner_pts = A - (border_pts/2) + 1;
    printf("%lld",inner_pts);
    return 0;
}
