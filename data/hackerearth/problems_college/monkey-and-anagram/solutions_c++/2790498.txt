#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int h1[26],h2[26],i,count=0,flag=0;
        char j;
        memset(h1,0,sizeof(h1));
        memset(h2,0,sizeof(h2));
        string s,t;
        cin>>s;
        cin>>t;
        for(i=0; i<s.length(); i++)
        {
            j=s[i];
            h1[j-97]++;

        }
        for(i=0; i<t.length(); i++)
        {
            j=t[i];
            h2[j-97]++;
        }
        for(i=0; i<26; i++)
        {
            if(h2[i]>h1[i])
            {
                count++;
            }
            if(h2[i]-h1[i]<0)
            {
                flag=1;
            }

        }
        if(flag==1)
            cout<<"-1"<<endl;
        else
            cout<<count<<endl;
    }
    return 0;

}
