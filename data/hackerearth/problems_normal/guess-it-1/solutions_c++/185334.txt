#include <iostream>
using namespace std;

int main()
{
    int T=0;
    cin>>T;
    while(T--) {
    	string s;
    	cin>>s;
    	int sum=0;
    	for(int i=0;i<s.length();i++) {
    		if(s[i]!='_')
    		    sum += s[i]-'0';    		
    	}
    	if(sum%9==0)
    	    cout<<"0 or 9"<<endl;
    	else
    	    cout<<9-(sum%9)<<endl;
    }
    return 0;
}
