#include <iostream>
#include <string>
using namespace std;

int main()
{
    string str;
    char c;
    getline(cin,str);
    
    int freq[500],sum=0,k,max=0;
    for(int i=0;i<500;i++)
    	freq[i] = 0;
    
    string::size_type n = str.size();
    
    for(int i=0; i<n;i++)
    {
    	c = str[i];
    	k = c - 'a';
    	freq[k]++;
    	sum = sum + k;
    }
    
    sum = sum/n;
    
    if(sum % 2 == 0)
    {
    	for(int i=0;i<500;i++)
    	{
    		if(freq[i] > max)
    		{
    			max = freq[i];
    			c = 'a' + i;
    		}
    	}
    	cout << c;
    }
    else
    {
    	for(int i=n-1;i>=0;i--)
    	cout << str[i];
    }
    
    return 0;
}
