#include<iostream>
#include<cmath>
#include<cstdio>
using namespace std;

typedef long long int lli;
typedef long int li;

// input macros
#define inp_int(t) scanf("%d",&t)

#define copy(a,b) strcpy(b,a)
// for loops
#define REP(i,j,n) for(int i=j;i<n;i++)
#define rep(i,n) REP(i,0,n)

int main()
{
    int t; inp_int(t);
    while(t--)
    {
        int xx; inp_int(xx);
    int arr[700];
    for(int i=0;i<700;i++)
        arr[i]=0;
    arr[0]=1;
    int c=699;
    for(int i=2;i<=xx;i++)
    {
        for(int j=0;j<=c;j++)
            arr[j]*=i;
        for(int j=0;j<=c;j++)
        {
            if(arr[j]>9)
            {
                arr[j+1]+=(arr[j]/10);
                arr[j]=arr[j]%10;
            }
        }
    }
    int sum=0;
    while(arr[c]==0)
        c--;
    for(int j=c;j>=0;j--)
        if(arr[j] == 4 || arr[j] == 7) sum++;
        cout<<sum<<endl;
    }
}
