N,Q=map(int,raw_input().split())
Monsters=raw_input()
def escape(P,M):
    if Monsters[M-1]=="1" or Monsters[P-1]=="1":
        print "No"
    elif P==M:
        print "Yes"
    else:
        if P>M:
            P=int(P/2)
        else:
            M=int(M/2)
        escape(P,M)
        
for i in range(Q):
    P,M=map(int,raw_input().split())
    escape(P,M)
