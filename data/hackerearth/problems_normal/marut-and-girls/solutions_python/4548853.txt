M=input()
rq=[]
rq = [int(n) for n in raw_input().split()]
N=input()
prop=0
for i in range(N):
    pq=[0]*10001
    count=0
    for n in raw_input().split():
        pq[int(n)]=1
    for j in range(M):
        if pq[rq[j]]!=1:
            break
        else:
            count+=1
    if count==M:
        prop+=1
print prop
