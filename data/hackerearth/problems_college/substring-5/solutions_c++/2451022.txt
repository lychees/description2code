#include<bits/stdc++.h>
using namespace std;
int main()
{
    string str;
    cin>>str;
    int len=str.length();
    int j=1;
    long long int sum=0;
    for(int i=1;i<len;i++)
    {
    	if(str[i-1]==str[i])
    	{
    		j++;
    	}
    	else
    	{
    		sum+=((j*(j+1))/2);
    		j=1;
    	}
    	
    }
    sum+=((j*(j+1))/2);
    cout<<sum<<endl;
}
