#include <iostream>
#include <string>
#include <math.h>
#include <algorithm>
using namespace std;

int countones(string arr){
	int count;
	int i;
	count = 0;
	for(i=0; i<arr.size();i++){
		if(arr.at(i) == '1' ){
			count++;
		}
	}
	//cout << count << endl;
	return count;
}

int main()
{
    int t;
    cin >> t;

    while(t--) {
    	long int n;
    	string arr;
    	cin >> arr;
    	cin >> n;
    	n = n * (n + 1)/2;
    	int a = countones(arr);

    	int siz = arr.size();

    	if(n % 2 == 0) {
    		cout << a << endl;
    	} else {
    		cout << abs(siz - a) << endl;
    	}

    }
    return 0;
}
