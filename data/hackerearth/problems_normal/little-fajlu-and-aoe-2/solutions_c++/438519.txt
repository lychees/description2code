#include <iostream>

using namespace std;

int main()
{
	int i,j,k; 	//number of warriors
	int t,n,s,p,c,power,f,g;
	cin>>t;
	while(t)
	{
		cin>>f>>g>>s>>p>>c>>power;
		int max=0, tempMax=0;

		for (i = 0; 100*i <=f ; i++)
		{
			//cout<<i<<f<<endl;
			for (j = 0; 100*i+125*j <=f && 50*j<= g;j++)
			{
				k = min((f-100*i-125*j)/50,(g-50*j)/100);
				//cout<<k<<endl;
				tempMax = s*i+p*j+c*k;
				//cout<<tempMax<<endl;
				if(tempMax>max)
					max = tempMax;
			}
		}
		if(max>power)
		{
			cout<<max-power<<endl;
		}
		else
		{
			cout<<"-1"<<endl;
		}
		t--;
	}
	return 0;
}
