#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,i,j,k;
	cin >> n;
	for(i=1;i<=n;i++)
	{
		if(i == 1 || i == n)
		{
		for(k=1;k<=n;k++)
		cout << "*";
		cout  << endl;
     	}
     	else
     	{
     		for(j=1;j<=n;j++)
     		{
     			if(j == n+1-i)
     			{
     			cout << "*";
     			break;
     		}
     			else 
     			cout << " ";
     		}
     		cout << endl;
     	}
	}
	return 0;
}
