from collections import Counter
import operator

val = raw_input()
counter = Counter(val)
# print counter
sort = sorted(counter.items(), key=operator.itemgetter(1), reverse=True)
# print sort
score = sort[0][1]
point = ord(sort[0][0])
points = []
for ele in sort:
    if ele[1] == score:
        if point > ord(ele[0]):
            point = ord(ele[0])
    else:
        break


print chr(point), score
