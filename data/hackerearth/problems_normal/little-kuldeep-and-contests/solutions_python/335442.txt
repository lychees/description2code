n = int(input())
start=[]
end=[]
for i in xrange(n):
	a = raw_input()
	a = a.split('-')
	s = a[0].split(':')
	st = int(s[0])*100+int(s[1])
	e = a[1].split(':')
	en = int(e[0])*100+int(e[1])
	start = start + [st]
	end = end + [en]

ans=0
for i in xrange(n):
	j=i+1
	while(j<n):
		if (not (start[i]>=end[j] or start[j]>=end[i]) ):
			ans=1
			i=n
			j=n
		j = j+1

if(ans):
	print "Will need a moderator!"
else:
	print "Who needs a moderator?"
