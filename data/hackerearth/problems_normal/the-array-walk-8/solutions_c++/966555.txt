#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define l long long int
long long int fff(l a,l b,l &p,l &q )
{
	if(a == 0)
	{
		p = 0;
		q = 1;
		return b;
	}
	l m,n;
	l t = fff(b%a,a,m,n);
	p = n-(b/a)*m;
	q = m;
	return t;
}
bool function1(l a,l b, l c,l &i,l &j,l &g)
{
	g = fff(abs(a),abs(b),i,j);
	if(c%g != 0)
	{
		return false;
	}
	i = i*c/g;
	j = j*c/g;
	if(a < 0) i = i*-1;
	if(b < 0) j = j*-1;
	return true;
}

int main()
{
	int test;
	cin >> test;
	while(test--)
	{
		l a,b,c,d,e,f,g,v;
		cin >> a >> b >> c;
		bool p = function1(b,0-c,a,e,f,g);
		if(p == false)
		{
			cout<<"-1\n";
			continue;
		}
		l tt = b/g,ss = c/g;
	    v = max(abs(e)/ss,abs(f)/tt)+1;
		e = e + v*ss;
		f = f + v*tt;
		l tt1 = e/ss;
		l tt2 = f/tt;
		l mn = min(tt1,tt2);
		e = e - mn*ss;
		f = f-mn*tt;
		cout<<e+f<<endl;
	}
    return 0;
}
