#include<bits/stdc++.h>
using namespace std;

int main()
{
   vector < pair < int, int > > j;
   int n, x, y, d;
   pair <int,int> bar;
   cin >> n;
   for ( int i = 0; i < n; i++ ) {
      cin >> x >> y >> d;
      bar = make_pair ( x, x+d );
      j.push_back( bar );
   }
   sort ( j.begin(), j.end() );
   int num = 0;
   for ( int i = 0; i < n; )
   {
      int k = i+1;    
      int max1 = j[i].second;
      while ( k < n && j[k].first <= max1 )
      {
         if ( max1 < j[k].second )
            max1 = j[k].second;
         k++;
      }
      num += (max1 - j[i].first + 1);
      i = k;
   }
   cout << num << endl;
   return 0;
}
