template<typename T>
T getNum() {
   T res=0;
   char c;
   while(1)
   {
      c=getchar_unlocked();
      if(c==' ' || c=='\n') continue;
      else break;
   }
   bool negatif;
   if (c=='-') {
       negatif = true;
       res = 0;
   }
   else {
       res=c-'0';
       negatif = false;
   }
   while(1)
   {
      c=getchar_unlocked();
      if('0'<=c && c<='9') res=10*res + c-'0';
      else break;
   }
   if (negatif) res*=-1;
   return res;
}

int main() {
	char s[10005];
	int prefsum[10005];
	prefsum[0] = 0;
	int n,k,A,B;
	int tests;
	scanf("%d",&tests);
	while (tests--) {
		scanf("%d %d %d %d",&n,&k,&A,&B);
		getchar();
		gets(s);
		set<char> spesial;
		REP(i,k) {
			char c;
			scanf("%c",&c);
			spesial.insert(c);
		}
		
		REP(i,n) {
			prefsum[i+1] = prefsum[i];
			if (spesial.count(s[i])) prefsum[i+1]++;
		}

		int ans = 0;
		FOR(i,1,n) {
			int kiri = 0; //indeks terkecil sehingga prefsum[i] >= A
			int lo = i,hi = n;
			while (lo<=hi) {
				int mid = (lo+hi)>>1;
				if (prefsum[mid]-prefsum[i-1] >= A) {
					hi = mid-1;
					kiri = mid;
				}
				else lo = mid+1;
			}
			
			int kanan = 0; //indeks terbesar sehingga prefsum[i] <= B
			lo = i; hi = n;
			while (lo<=hi) {
				int mid = (lo+hi)>>1;
				if (prefsum[mid]-prefsum[i-1] <= B) {
					lo = mid+1;
					kanan = mid;
				}
				else hi = mid-1;
			}
			
			if (kiri and kanan) ans += (kanan-kiri+1);
		}
		printf("%d\n",ans);
	}
	return 0;
}
