#include <iostream>
#include <algorithm>

using namespace std;

int noc(int n){
	if(n<=0)
		return -1;
	else{
		if(n==6 || n==8)
			return 1;
		int a=noc(n-6);
		int b=noc(n-8);
		if(a==-1 && b==-1)
			return -1;
		else if(a==-1)
			return 1+b;
		else if(b==-1)
			return 1+a;
		else
			return 1+min(a,b);
	}
}

int main()
{
   	int t,n;
   	cin>>t;
   	for(int it=1;it<=t;it++){
   		cin>>n;
   		cout<<noc(n)<<endl;
   	}
   	return 0;
}
