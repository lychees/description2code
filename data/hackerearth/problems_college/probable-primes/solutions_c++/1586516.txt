VI primes;
int arr[2000001] = {0};
void sieve(int limit)
{
      FOR(i,2,limit+1) arr[i] = 1;
      int i = 2;
      while(i<=limit)
      {
            primes.pb(i);
            int j = 2*i;
            while(j<=limit)
            {
                  arr[j] = 0;
                  j+=i;
            }
            i++;
            while(i<=limit && !arr[i])
                  i++;
      }
}

int main()
{
      //inp_s;
      sieve(1100000);
      FOR(i,1,1100000) arr[i] += arr[i-1];
      int t;
      cin >> t;
      while(t--)
      {
            int l,r;
            cin >> l >> r;

            float nums = r-l+1;
            float ctr = arr[r] - arr[l-1];
            float ans = ctr / nums;
            
         //   assert(ans >=0 && ans <= 1);
            printf("%.6f\n",ans);
      }
      return 0;
}
