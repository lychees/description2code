#include <bits/stdc++.h>
#define ll long long int
#define s(a) scanf("%lld",&a)
#define sd(a) scanf("%lf",&a)
#define f first
#define sc second
#define pb push_back
#define mp make_pair
#define inf 10e16

using namespace std;

ll dp[100001][11][11];
ll n,maxx;
ll mod=1000000007;

ll recur(ll ind,ll lst,ll mx)
{
    if(ind<0 || mx>maxx) return 0;
    if(ind==0) {
        if(mx==maxx) {return 1;}
        else return 0;
    }
    if(dp[ind][lst][mx]>=0) {
        return dp[ind][lst][mx];
    }
    ll w=0;
    ll i;
    if(lst==0) {
        for(i=1;i<=9;i++) {
            w=(w+recur(ind-i,i,mx))%mod;
        }
    }
    else {
        ll k=min(9LL,lst+maxx);
        for(i=lst;i<=k;i++) {
            w=(w+recur(ind-i,i,max(mx,i-lst)))%mod;
        }
    }
    dp[ind][lst][mx]=w;
    return w;
}

int main()
{
    ll t,i,j,k,l,w,ww,x,y,z;
    s(t);
    for(ll tt=1;tt<=t;tt++) {
        for(i=0;i<=100000;i++) {
            for(j=0;j<=10;j++) {
                for(k=0;k<=10;k++) {
                    dp[i][j][k]=-1;
                }
            }
        }
        s(n);s(maxx);
        cout<<"Case #"<<tt<<": "<<recur(n,0,0)<<endl;
    }
    return 0;
}
