#include <bits/stdc++.h>
using namespace std;

int dp[1010][1010][2];

int gcd(int x,int y)
{
    if (x > y) {
        int t = x;
        x = y;
        y = t;
    }

    if (y % x == 0)
        return x;
    else
        return (gcd(y%x, x));
}

int main()
{
    int a,b,t,i,j,x;

    dp[1][1][0] = 0;
    dp[1][1][1] = 0;

    for (i = 2; i <= 1000; i++) {
        dp[1][i][0] = 2;
        dp[1][i][1] = 2;
    }

    for (i = 2; i <= 1000; i++) {
        for (j = 1; j <= 1000; j++) {
            if (j == 1) {
                dp[i][j][0] = 1;
                dp[i][j][1] = 1;
                continue;
            }
            x = gcd(i,j);
            if (x == 1) {
                dp[i][j][0] = dp[i][j-1][1];
                dp[i][j][1] = dp[i-1][j][0];
            } else {
                dp[i][j][0] = min(dp[i][j-1][1], dp[i][j/x][1]);
                dp[i][j][1] = max(dp[i-1][j][0], dp[i/x][j][0]);
            }
        }
    }

    cin >> t;
    while (t--) {
        cin >> a >> b;
        if (dp[a][b][0] == 0) {
            cout <<"Draw\n";
        } else {
            if (dp[a][b][0] == 1) {
                cout <<"Arjit\n";
            } else {
                cout <<"Chandu Don\n";
            }
        }
    }
    return 0;
}
