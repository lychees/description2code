#include <bits/stdc++.h>
using namespace std;
vector<int>v;
int main()
{
	int t;
	cin>>t;
	int n;
	while(t--){
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++){
			cin>>a[i];
		}
		sort(a,a+n);
		int i=0,j=n-1,z;
		for(;i<=j;i++,j--){
			z=a[i]+a[j];
			v.push_back(z);
			
		}
		sort(v.begin(),v.end());
		
		cout<<abs(v[0]-v[v.size()-1])<<"\n";
		v.clear();
	}
    return 0;
}
