#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
	ll t;
	cin>>t;
	while(t--){
		ll c,r;
		ll k=0;
		ll k1=-1;
		ll sum=0;
		ll sum1=0;
		cin>>r>>c;
		for(ll i=0;i<r/2;i++){
			k+=2;
			if(k<=c){
				sum+=k;
			}
			else{
				sum+=c;
			}
		}
		ll limit=0;
		if(r%2==0){
			limit=r/2;
		}
		else{
			limit=r/2+1;
		}
		for(ll i=0;i<limit;i++){
			k1+=2;
			if(k1<=c){
				sum1+=k1;
			}
			else{
				sum1+=c;
			}
		}
		cout<<sum<<" "<<sum1<<endl;
	}
}
