#include<bits/stdc++.h>
#define MAX   53
#define FOR(i,a,b) for (int i=(a),_b=(b);i<=_b;i=i+1)
#define REP(i,n) for (int i=0,_n=(n);i<_n;i=i+1)
using namespace std;
const char win[]="Stark";
const char lose[]="Lannister";
int sg[MAX][MAX][MAX];
int getValue(const set<int> &s) {
    int res=0;
    while (s.find(res)!=s.end()) res++;
    return (res);
}
void precount(void) {
    REP(i,MAX) REP(j,MAX) REP(k,MAX) {
        if (i==0 && j==0 && k==0) {
            sg[i][j][k]=0;
            continue;
        }
        set<int> s;
        FOR(l,1,i) s.insert(sg[i-l][j][k]);
        FOR(l,1,j) s.insert(sg[i][j-l][k]);
        FOR(l,1,k) s.insert(sg[i][j][k-l]);
        sg[i][j][k]=getValue(s);
    }
}
void process(void) {
    int n;
    scanf("%d",&n);
    int res=0;
    REP(zz,n) {
        int x,y,z;
        scanf("%d%d%d",&x,&y,&z);
        res^=sg[x][y][z];
    }
    if (res>0) printf("%s\n",win); else printf("%s\n",lose);
}
int main(void) {
    precount();
    int t;
    scanf("%d",&t);
    REP(zz,t) process();
    return 0;
}
