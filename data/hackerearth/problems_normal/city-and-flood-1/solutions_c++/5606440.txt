using namespace std;
int root(int i,int *parent)
{
	while(parent[i]!=i)
	{
		i=parent[i];
	}
	return i;
}
void join(int i,int j,int *parent,int *size)
{
	int root1=root(i,parent);
	int root2=root(j,parent);
	if(size[root1]<=size[root2])
	{
		parent[root1]=root2;
		size[root2]+=size[root1];
	}
	else
	{
		parent[root2]=root1;
		size[root1]+=size[root2];
	}
}
int main()
{
	ios::sync_with_stdio(false);
	int n,k;
	cin>>n>>k;
	int parent[n+1],size[n+1];
	for(int i=1;i<=n;i++)
	{
		parent[i]=i;
		size[i]=1;
	}
	int i,j;
	while(k--)
	{
		cin>>i>>j;
		join(i,j,parent,size);
	}
	int count=0;
	bool isRoot[n+1];
	for(i=1;i<=n;i++)
	{
		isRoot[i]=false;
	}
	for(i=1;i<=n;i++)
	{
		isRoot[root(i,parent)]=true;
	}
	for(i=1;i<=n;i++)
	{
		if(isRoot[i])
			count++;
	}
	cout<<count<<br;
	return 0;
}
