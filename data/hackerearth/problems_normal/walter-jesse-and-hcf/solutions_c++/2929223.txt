#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int x,y,k,d;
    cin>>d;
    
    for(int i=0;i<d;i++){
    	
    	cin>>x>>y>>k;
    	long long int gcd=__gcd(x,y);
    	vector<long long int>v;
    	
    	for(int j=1;j*j<=gcd;j++){
    		
    		if(gcd%j==0){
    			v.push_back(j);
    			if(j!=gcd/j){
    				v.push_back(gcd/j);
    			}
    		}
    	}
    	
    sort(v.begin() , v.end());
    if (k > v.size()) cout<<"No crymeth today\n";
       else cout<<v[v.size() -k]<<"\n";
    	
    
    }
    
    return 0;
}
