#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;
long long mark[1000009];
vector< long long >p;

long long t,i,j,k,l,m,n,q,ar[1000000],br[1000000],ans,u;


  void sieve(long long n)
{
    long long i,j;
    mark[0]=mark[1]=1;
    for(i=4;i<=n;i=i+2)
    mark[i]=1;

    for(i=3;i<=n;i++)
    {
        if(!mark[i])
        for(j=2*i;j<=n;j=j+i)
        {

            mark[j]=1;
        }
}

for(i=0;i<n;i++)
{
    if(mark[i]==0)
    p.push_back(i);
}
}









int main()
{


sieve(100000);

scanf("%lld",&t);
u=p.size();
while(t--)
{
    scanf("%lld %lld %lld %lld",&n,&q,&m,&k);
    for(i=0;i<u;i++)
    {

        while(n%p[i]==0&&n>0)
        {
            ar[p[i]]++;
            n=n/p[i];

    }
    if(n==1)
    break;
}
for(i=0;i<=100000;i++)
{
ar[i]=ar[i]*q;
}

   for(i=0;i<p.size();i++)
    {
        while(m%p[i]==0)
        {
            br[p[i]]++;
            m=m/p[i];
    }
    if(m==1)
    break;
}
for(i=0;i<=100000;i++)
{
br[i]=br[i]*k;
}

ans=0;
   for(i=0;i<=100000;i++)
   {
       ar[i]=ar[i]-br[i];
       if(ar[i]<0)
       ans=1;

    }
    if(m!=1)
    if(n%m!=0)
    ans=1;

    if(ans==0)
    printf("Divisible\n");
    else
    printf("Not divisible\n");

    for(i=0;i<=100000;i++)
    {
        ar[i]=0;br[i]=0;//

    }



    }








    return 0;
}
