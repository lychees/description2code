#include<bits/stdc++.h>
using namespace std;
long long arr[100005];
long long answ[100005];
int n;
int main()
{
	scanf("%d",&n);
	int i,j;
	for(i=1;i<=n;i++)
	scanf("%lld",&arr[i]);
	for(i=1;i<=n;i++)
	answ[i]=0;
	answ[1]=arr[1];
	for(i=2;i<=n;i++)
	{
		int x=i;
		for(j=1;j<=sqrt(x);j++)
		{
			if(i%j==0)
			{
				int p,s;
				p=j;
				s=i/j;
				if(p!=s)
				{
					if(p!=x)
					answ[i]=max(answ[i],answ[p]+arr[i]);
					if(s!=x)
					answ[i]=max(answ[i],answ[s]+arr[i]);
				}
				else if(p==s)
				{
					if(p!=x)
					answ[i]=max(answ[i],answ[p]+arr[i]);
				}
			}
		}
	}
	for(i=1;i<=n;i++)
	printf("%lld ",answ[i]);
	printf("\n");
	return 0;
}
