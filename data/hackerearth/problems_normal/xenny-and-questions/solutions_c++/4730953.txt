#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	int n,tim;
    	cin>>n>>tim;
    	vector<long long int> vec;
    	vec.resize(n);
    	for(int i=0;i<n;i++)
    		scanf("%lld",&vec[i]);
    	sort(vec.begin(),vec.end());
    	int cnt=0;
    	long long int sum=0;
    	for(int i=n-1;i>=0;i--)
    	{
    		if(tim==cnt)
    			break;
    			
    		sum+=vec[i];
    		cnt++;
    		
    	}
    	cout<<sum<<endl;
    	t--;
    }
}
