#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
char ss[11][7];
string s;
int n,k;
long long dp[10][1<<10];
int solve(int pos,int mask)
{
	if(pos==k)
	return 1;
	if(dp[pos][mask]!=-1)
	return dp[pos][mask];
	long long ret = 0;
	int i,j;
	for(i=0;i<n;i++)
	{
		if(mask&(1<<i))
		continue;
		int c=0;
		for(j=0;j<6;j++)
		{
			if(ss[i][j]==s[pos])
			c++;
		}
		ret+=(c*solve(pos+1,mask|(1<<i)))%mod;
		ret%=mod;
		
	}
	return dp[pos][mask]=ret;
}
int main()
{
	memset(dp,-1,sizeof(dp));
	int i,j;
	cin >> n >>k;
	for(i=0;i<n;i++)
	{
		for(j=0;j<6;j++)
		{
			cin >> ss[i][j];
		}
	}
	cin >> s;
	long long ans=solve(0,0);
	cout<< ans<< endl;
	return 0;
}
