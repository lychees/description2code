#include <bits/stdc++.h>
#define s(T) scanf("%d",&T)
#define us(T) scanf("%u",&T)
#define uls(T) scanf("%lu",&T)
#define SIZE 65535
#define mx 105
#define mod 1000000009
using namespace std;
typedef long long ll;

int main(){
   char s[mx][mx];
   int i,j,k,l,freq[27],T,n;
   bool a[27];
   scanf("%d",&T);
   while(T--){
    for(i=0; i<27; ++i)
        freq[i]=0;
    scanf("%d",&n);
    for(i=0; i<n; ++i)
        scanf("%s",s[i]);
    for(i=0; i<n; ++i){
        for(j=0; j<27; ++j)
            a[j]=false;
        l=strlen(s[i]);
        for(j=0; j<l; ++j){
            k=s[i][j]-'a';
            if(!a[k]){
                a[k]=true;
                ++freq[k];
            }
        }
    }
    k=0;
    for(i=0; i<27; ++i){
        //cout<<a[i];
        if(freq[i]>k){
            k=freq[i];
        }
    }
    //cout<<endl;
    for(i=0; i<27; ++i)
    if(freq[i]==k){
        j=i;
        break;
    }
    printf("%c\n",'a'+j);
   }
return 0;
}
