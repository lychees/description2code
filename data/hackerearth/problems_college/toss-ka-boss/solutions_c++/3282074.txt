#include <bits/stdc++.h>
using namespace std;

char a[1000005];


int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		scanf("%s",a);
		int l = strlen(a);
		int i=0;
		long tot=0;
		long cur=0;
		while(i<l){
			if(a[i]=='H'){
				cur++;
			}else{
				tot+=(cur*(cur+1))/2;
				cur=0;
			}
			i++;
		}
		tot+=(cur*(cur+1))/2;
		printf("%ld\n",tot);
		
	}
}
