#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    ll n,x[10005],y[10005],f[10004];
    double dp[10005];
    cin>>n;
    for(int i=0;i<n;i++)
    cin>>x[i]>>y[i]>>f[i];
    dp[0]=f[0];
    //cout<<dp[0]<<" ";
    for(int i=1;i<n;i++){
    	dp[i]=-1000000000007;
    	for(int j=0;j<i;j++){
    		dp[i]=max(dp[i],dp[j]-(sqrt((x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]))));
    	}
    	dp[i]=dp[i]+f[i];
    }
    cout<<fixed<<setprecision(6)<<dp[n-1]<<endl;
    return 0;
}
