#include <iostream>
#include <string.h>
using namespace std;

int main()
{
    int t;
    string s,temp="";
    cin>>t;
    while(t--){
    	cin>>s;
    	for(int j = 65; j<=90;j++){
    		for(int i=0;i<s.size();i++){
    			if(s[i]==j){
    				temp+=s[i];	
    			}
    		}
    	}
    	cout<<temp<<endl;
    	temp="";
    }
    return 0;
}
