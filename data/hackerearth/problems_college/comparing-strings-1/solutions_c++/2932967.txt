#include<stdio.h>
#include<iostream>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<stdlib.h>
#include<vector>
#include<map>
#include<set>

using namespace std;

main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		char a[100000],b[100000];
		cin>>a>>b;
		
		int l1=strlen(a);
		int l2=strlen(b);
		
		int flag=0;
		for(int i=0;i<l1;i++)
		{
			for(int j=0;j<l2;j++)
			{
				if(a[i]==b[j])
				{
					flag=1;
					break;
				}
			}
		}
		if(flag==1)
		cout<<"YES\n";
		else
		cout<<"NO\n";
	}
}
