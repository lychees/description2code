#include <bits/stdc++.h>

using namespace std;

int primec(long x){
	int j,y;
	y=sqrt(x);
	for(j=3;j<=y;j=j+2){
		if(x%j==0){
			return 0;
		}
	}
	return 1;
}

{
    int tc;
	scanf("%d", &tc);
	long int m[tc],n[tc],j,sum,i;
	unsigned int prime[1000000]={0};
	for(i=3;i<1001;i++){
		if(prime[i]==0){
			for(j=i;j*i<1000001;j++){
				prime[i*j]=1;
			}
		}
	}
	for(i=0;i<tc;i++){
		scanf("%ld %ld", &m[i], &n[i]);
	}
	for(i=0;i<tc;i++){
		if(m[i]%2==0){
			j=m[i]+1;
		}
		else{
			j=m[i];
		}
		if(m[i]<=2){
            sum=2;
            j=3;
