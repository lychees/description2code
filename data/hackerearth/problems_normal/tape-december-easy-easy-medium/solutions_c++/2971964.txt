#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1e5+3;
int ar[MAXN];
int previ[MAXN];
int summ[MAXN];
int main() {
    int i,n,k,d;
    scanf("%d%d%d",&n,&k,&d);
    int last=-1;
    bool noAns=false;
	int sum=0;
	ar[0]=0;
    for(i=1;i<=k-1;i++) {
        scanf("%d",&ar[i]);
        if(ar[i]!=0)
            last=i;
        previ[i]=last;
    	sum+=ar[i];
	}
    for(;i<=n;i++) {
        scanf("%d",&ar[i]);
		sum+=ar[i]-ar[i-k];
		summ[i]=sum;
        if(sum==0)
			noAns=true;
		if(ar[i]!=0)
            last=i;
        previ[i]=last;
    }
	sum=0;
    int ans=0;
    if(!noAns) {
	    for(i=k;i<=n;i++) {
	        sum=summ[i];
	        if(sum<d) {
	            ans++;
				int val=previ[i]+k-1;
				if(val>n)
					break;
				else
					val=previ[val];
				i=val+k-1;
	        }
   	 	}
    }
    if(noAns)
        printf("-1\n");
    else
        printf("%d\n",ans);
    return 0;
}
