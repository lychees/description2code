#include <iostream>
#include<bits/stdc++.h>
using namespace std;
long long a[100][100];
long long int c(long long int n,long long int m)
{  
	if(n==m || m==0)
	return 1;
	return a[n][m]=c(n-1,m-1)+c(n-1,m);
}
	
long long int fact(long long int n)
{ 
	int i;
	long long int ans=1;
	for(i=1;i<=n;i++)
	ans*=i;
	return ans;
}
	
int main()
{
   long long int t,n,m,i,j;
    cin>>t;
    while(t--)
    {
    	cin>>i>>j;
    	m=fact(i+j)*c(7,i)*c(5,j);
    	cout<<m<<"\n";
    }
    return 0;
}
