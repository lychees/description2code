#include <iostream>
using namespace std;

int main()
{
    int t;
    long long int n1,n2,n3,n,pro;
    cin>>t;
    while(t--)
    {
    	cin>>n1>>n2>>n3;
    	n=n1<n2?(n1<n3?n1:n3):(n2<n3?n2:n3);
    	pro=1;
    	if(n==n1)
    	{
    		if(n2<=n3)
    		{
    			pro=(n1*(n2-1))%1000000007;
    			pro=(pro*(n3-2))%1000000007;
    			cout<<pro%1000000007<<endl;
    		}
    		else
    		{
    				pro=(n1*(n3-1))%1000000007;
    				pro=(pro*(n2-2))%1000000007;
    				cout<<pro%1000000007<<endl;
    		}
    	}
    	else if(n==n2)
    	{
    		if(n1<=n3)
    		{
    			pro=(n2*(n1-1))%1000000007;
    			pro=(pro*(n3-2))%1000000007;
    			cout<<pro%1000000007<<endl;
    		}
    		else
    		{
    				pro=(n2*(n3-1))%1000000007;
    				pro=(pro*(n1-2))%1000000007;;
    				cout<<pro%1000000007<<endl;
    		}
    	}
    else	if(n==n3)
    	{
    		if(n1<=n2)
    		{
    			pro=(n3*(n1-1))%1000000007;
    			pro=(pro*(n2-2))%1000000007;
    			cout<<pro%1000000007<<endl;
    		}
    		else
    		{
    				pro=(n3*(n2-1))%1000000007;
    				pro=(pro*(n1-2))%1000000007;
    				cout<<pro%1000000007<<endl;
    		}
    	}
    }
    return 0;
}
