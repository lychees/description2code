#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main()
{
    char  st[1000] ;
    
    cin.getline(st, 1000);
    
    ll a[256]  = {0} ;
    
    ll n = strlen(st) ;
    
    for(ll i = 0; i < n; i++){
    	a[st[i]] += 1;
    }
    
    // for(ll i = 0; i < 256; i++){
    // 	cout << a[i] << " " ;
    // }
    
    ll max = a[0]; 
    ll index = 0;
    for(ll i = 0; i < 256; i++){
    	if(a[i] > max){
    		max = a[i] ;
    		index = i ;
    	}
    }
    
    printf("%c %d",'\0'+ index, max);
    return 0;
}
