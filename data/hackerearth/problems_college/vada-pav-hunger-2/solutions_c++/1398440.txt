int bfs(int src)
{
    queue<int> q;
    q.push(src);
    int ans=0;
    vis[src]=true;
    while(!q.empty())
    {
        int node= q.front(); q.pop();
        for(int i=0;i<adj[node].size();i++)
        {
            int V= adj[node][i];
            if(vis[V]==false)
            {
                vis[V]=true;
                q.push(V);
                level[V]= level[node]+1;
                ans= max(ans,level[V]);
            }
        }
    }
    return ans;
}
