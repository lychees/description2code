#include<bits/stdc++.h>
#define ll long long int
#define inp(x) scanf("%lld",&x)
using namespace std;
static ll a[1000009];
int main(){
	ll t,n,x,m,l;
	inp(t);
	while(t--){
		m = -1;l = -1;
		inp(n);
		while(n--){
			inp(x);
			++a[x];
			if(m<a[x]){
				m = a[x];
			}
			if(l<x){
				l=x;
			}
		}
		for(ll i = l;i>=1;--i){
			if(a[i]==m){
				printf("%lld ",i);
			}
		}printf("\n");
		memset(a,0,sizeof(a));
	}
	return 0;
}
