#include <iostream>
using namespace std;
#include <iomanip>

int main()
{
    int n;
    double p;
    cin>>n>>p;
    p /= 100;
    double dp[n+1];
    dp[0] = 0;
    dp[1] = 0;
    dp[2] = p;
    dp[3] = 1-p;
    for(int i=4;i<=n;i++)
    {
        dp[i] = p*dp[i-2] + (1-p)*dp[i-3];
    }
    cout<<fixed<<setprecision(6)<<dp[n]<<endl;
}
