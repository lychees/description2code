#include <iostream>
using namespace std;

int main()
{
    int test,n,count1,count2,i;
    string s;
    cin >> test;
    while(test--) {
    	cin >> n;
    	cin >> s;
    	count1=0;
    	count2=0;
    	for(i=0;i<2*n;i=i+2) {
    		if(s[i]=='B')
    			count1++;
    		else if(s[i]=='G')
    			count2++;
    	}
    	//cout << count1 << " " << count2 << "\n";
    	if(count1<count2)
    		cout << count1 << "\n";
    	else
    		cout << count2 << "\n";
    }
    return 0;
}
