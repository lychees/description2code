#include <iostream>

using namespace std;

int main(void)
{
	int n,flag;
	cin>>n;
	
	int s[n];
	for(int i=0; i<n; i++)
		s[i] = 0;
	
	for(int i=0; i<n; i++)
		cin>>s[i];
	for(int j=0; j<n; j++)
	{
		flag=0;
	for(int i=0; i<n && !flag; i++)
	{
		if(i==j)
			continue;
		else
		{
			if(s[j]%s[i]==0)
				flag=1;
		}
	}
	if(!flag)
		cout<<s[j]<<" ";
	}
	return 0;
}
