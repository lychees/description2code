#include<iostream>
using namespace std;
int main()
{
	long long int t;
	cin>>t;
	while(t>0)
	{
		long long int n;
		cin>>n;
		long long int a[n];
		long long int i;
		for(i=0;i<n;i++)
			cin>>a[i];
		long long int count=0,count1=0;
		for(i=0;i<n;i++)
		{
			if(a[i]%2==0)
				count++;
			else
				count1++;
		}
		cout<<count*count1<<endl;
		t--;
	}
}
