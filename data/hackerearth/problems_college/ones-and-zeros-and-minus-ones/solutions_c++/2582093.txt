#define ll long long
#define ld double
#define sf(x) sort(x.begin(),x.end(),func);
#define s(x) sort(x.begin(),x.end())
#define pb push_back                                                                                                                                                                                                                                                                                            
#define set(v, val) memset(v, val, sizeof(v));
#define rs(v) { s(v) ; r(v) ; }
#define all(v) v.begin(),v.end()
#define r(v) { reverse(all(v)) ; }
#define mod 1000000007

using namespace std;

int main() 
{    
    ios_base::sync_with_stdio(0);
    ll t;
    cin>>t;
    while(t--)
    {
        ll n;
        cin>>n;
        ll a[n];
        for(ll i=0;i<n;i++)
        {
            cin>>a[i];
        }
        ll c=0;
        ll f=1;
        vector < ll > v;
        for(ll i=0;i<n;i++)
        {
            if(a[i]==1)
            {
                c++;
            }
            if(a[i]==-1)
            {
                f=0;
                v.pb(c);
                c=0;
            }
        }
        v.pb(c);
        if(f)
        {
            cout<<"0/1\n";
            continue;
        }
        ll ans=0;
        for(ll i=0;i<v.size()-1;i++)
        {
            //cout<<v[i]<<"\n";
            ans = max(ans,v[i]+v[i+1]);
        }
        cout<<ans<<"/1\n";
    }
    return 0;
}
