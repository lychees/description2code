#include <bits/stdc++.h>

using namespace std;

int main()
{
	int test,m,n;
	cin >> test;
	int dp[10001];
	while (test--) {
		cin >> m >> n;
		int a[n];
		for (int i = 0;i <= m; i++) {
			dp[i] = 0;
		}
		for (int i = 0; i < n; i++) {
			cin >> a[i];
		}

		for (int i = 1; i <= m; i++) {
			for (int j = 0; j < n; j++) {
				if (i - a[j] >= 0 && dp[i-a[j]] == 0) {
					dp[i] = 1;
					break;
				}
			}
		}

		if (dp[m]) {
			cout << "Little Deepu\n";
		} else {
			cout << "Kate\n";
		}
	}

	return 0;
}
