int dp[100000][3];
int price[100000][3];
int n;
int f(int x,int y)
{
	if(x==n-1)
	return price[n-1][y];
	if(dp[x][y]!=-1)
	return dp[x][y];
	else
	{
		if(y==0)
		{
			dp[x+1][1]=f(x+1,1);
			dp[x+1][2]=f(x+1,2);
			return dp[x][y]=price[x][y]+min(dp[x+1][1],dp[x+1][2]);
		}
		if(y==1)
		{
			dp[x+1][0]=f(x+1,0);
			dp[x+1][2]=f(x+1,2);
			return dp[x][y]=price[x][y]+min(dp[x+1][0],dp[x+1][2]);
		}
		if(y==2)
		{
			dp[x+1][1]=f(x+1,1);
			dp[x+1][0]=f(x+1,0);
			return dp[x][y]=price[x][y]+min(dp[x+1][1],dp[x+1][0]);
		}
	}
	return 0;
}

int main()
{
	std::ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
	n=0;
	for(int i=0;i<100000;i++)
	{
		for(int j=0;j<3;j++)
		price[i][j]=0;
	}
	for(int i=0;i<100000;i++)
	{
		for(int j=0;j<3;j++)
		dp[i][j]=-1;
	}
	cin>>n;	
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<3;j++)
		cin>>price[i][j];
	}
	cout<<min(min(f(0,0),f(0,1)),f(0,2))<<endl;
	}
	return 0;
}
