#include <iostream>

using namespace std;

int main(int argc, char *argv[])
{
        int T, N, tmp, A[200];
        cin >> T;
        while(T--){
                cin >> N;
                for(int i = 0; i < N; i++){
                        cin >> A[i];
                }
                bool found;
                for(int bit = 0; bit < 32; bit++){
                        found = false;
                        int subset_and = 0xFFFFFFFF;
                        for(int i = 0; i < N; i++){
                                if(A[i] & (0x01 << bit)){
                                        subset_and &= A[i];
                                }
                        }
                        if((subset_and & (subset_and - 1)) == 0){
                                found = true;
                                break;
                        }
                }
                found ? cout << "YES\n" :  cout << "NO\n";
        }
        return 0;
}
