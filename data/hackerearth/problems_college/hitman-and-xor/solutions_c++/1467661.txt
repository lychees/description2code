#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	int arr[n];
    	long long even=0,odd=0;
    	for(int i=0;i<n;i++)
    	{
    		scanf("%d",&arr[i]);
    		if(arr[i]%2==0)
    		even++;
    		else
    		odd++;
    	}
    	printf("%lld\n",odd*even);
    }
    return 0;
}
