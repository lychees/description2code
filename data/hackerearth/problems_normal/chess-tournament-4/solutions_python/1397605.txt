def winner(p, q, c):
	if p>q:
		p,q = q,p
	if c[q-1][p] == 0:
		return p
	else:
		return q
		
n = input()
l = 2**n
c = []
for ____ in xrange(l-1):
	c.append(list(map(int, raw_input().split())))

L = [i for i in xrange(l)]
L1 = []

while(len(L) != 1):
	for k in xrange(l/2):
		p = 2*k
		q = 2*k+1
		L1.append(winner(L[p],L[q], c))
	L = L1
	L1 = []
	l = l/2
	
print L[0]+1
