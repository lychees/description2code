#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    unsigned int T;
    cin>>T;
    while(T--)
    { unsigned int n;
    cin>>n;
    unsigned int a[n];
    for(int i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    unsigned int temp=0;
    for(int i=0;i<n-1;i++)
    {if(a[i]==a[i+1])
    {
    	temp++;
    }
    }
    cout<<temp<<endl;
    
}
}
