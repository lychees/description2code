#include <iostream>
#include<vector>
using namespace std;

int main()
{
   string s;
   long int n;
   cin>>n;
   cin>>s;
   long int q,i;
   cin>>q;
   long int a[26] = {0};
   for(i=0;i<n;i++){
   	a[(int)(s[i]-97)]++;
   }
   while(q>0){
   	char x,y;
   	cin>>x>>y;
   	if(x==y){
   		cout<<((a[(int)(x-97)]-1) * a[(int)(y-97)])/2<<endl;
   	} else {
	   	cout<<a[(int)(x-97)] * a[(int)(y-97)]<<endl;
   	}
   	q--;
   }
    return 0;
}
