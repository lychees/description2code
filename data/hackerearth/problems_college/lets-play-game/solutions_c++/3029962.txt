#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,n;
	long long int k; 
	cin>>t;
	while(t--) {
		cin>>n>>k;
		long long int a[n],b[n],c[n],max=0;
		for(int i=0;i<n;i++) {cin>>a[i]; c[i]=k/a[i];}
		for(int i=0;i<n;i++) {cin>>b[i]; if(max<b[i]*c[i]) max=b[i]*c[i];}
		cout<<max<<endl;
	}
    return 0;
}
