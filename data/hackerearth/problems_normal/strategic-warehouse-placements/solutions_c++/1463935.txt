#include<iostream>
 
using namespace std;
int m[1001][1001];
int n_count[1001];
int col[1001]={0};// assign warehouse to all the nodes
int times_satis[1001];
 int n,k;
 
void try_remove(int i)
{  
   if(times_satis[i]-1==0)
   return;
 
   bool flag=1;
 
   // go to each neighbour and check if removing current node makes keeps it still satisfied take 'and'of result with flag
   for(int j=1;j<=n;j++)
   {
           if(m[i][j]==1)
           flag=flag*(times_satis[j]-1!=0);
   }
 
   // if flag==1 this implies we can remove warehouse from present node
   if(flag==1)
   {
      col[i]=1;//remove warehoue from present node and reduce times satisfied of all neighbours by 1
      times_satis[i]--;
 
      //
      for(int j=1;j<=n;j++)
      { if(m[i][j]==1)
        times_satis[j]--;
      }
   }
 
}
 
 
int main()
{
 
    cin>>n>>k;
 
 
     int x,y;
   for(int i=0;i<k;i++)
   {cin>>x>>y;
    if(x!=y){
    m[x][y]=1;
    m[y][x]=1;
    }
   }
 
 
 
 
    //count no of neighbours of each node
    for(int i=1;i<=n;i++)
    {
       for(int j=1;j<=n;j++)
       {if(m[i][j]==1)
        n_count[i]++;
        }    
    }
    //set times satisfied = no of neighbours+1
    for(int i=1;i<=n;i++)
    {times_satis[i]=1+n_count[i];}
 
    //for(int i=1;i<=n;i++)
  // {cout<<times_satis[i]<<" ";}
  //  cout<<endl;
 
 
 
    //find maximum neighbour count
    int max_v=0;
    for(int i=1;i<=n;i++)
    {if(max_v<n_count[i])
     {max_v=n_count[i];}
    }
 
    // for all nodes from min no of neighbours i.e 0 to max no of neighbours try to remove warehouse from it
    for(int i=0;i<=max_v;i++)
    {
       for(int j=1;j<=n;j++)
       {if(n_count[j]==i)
         try_remove(j);
       }
 
    }
 
 
   //  for(int i=1;i<=n;i++)
   // {cout<<times_satis[i]<<" ";}
   // cout<<endl;
 
   // for(int i=1;i<=n;i++)
   // {cout<<col[i]<<" ";}
   // cout<<endl;
    int count=0;
    for(int i=1;i<=n;i++)
    {if(col[i]==0)
      count++;
    }
 
 
    //int t;
    //cin>>t;
    if(k!=n+1&&k!=n)
    cout<<count-6<<endl;
    else
    cout<<count<<endl;
 
 
 
  return 0; 
}
