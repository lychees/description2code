#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    string s;
    map<string,int> m; 
    while(t--)
    {
        
        cin>>s;
        m[s]++;
    }
    map<string,int>::iterator it;
    for(it = m.begin();it != m.end();it++)
    {
        cout<<it->first<<" "<<it->second<<endl;
    }
}
