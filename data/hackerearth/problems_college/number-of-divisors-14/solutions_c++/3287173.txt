#include <bits/stdc++.h>

using namespace std;

int isP[1000005];
vector <int> p;

void sieve()
{
    int i, j;
    for (i = 2; i <= 1000000; ++i) {
        if (!isP[i]) {
            p.push_back(i);
            for (j = 2; i*j <= 1000000; ++j)
                isP[i*j] = 1;
        }
    }
}

int main()
{
    sieve();
    int t, i, cnt;
    long long int n, ans;
    cin >> t;
    while (t--) {
        cin >> n;
        ans = 1;
        for (i = 0; i < p.size(); ++i) {
            cnt = 1;
            while (n % p[i] == 0) {
                ++cnt;
                n /= p[i];
            }
            ans *= cnt;
        }
        assert(n == 1);
        cout << ans << endl;
    }
    return 0;
}
