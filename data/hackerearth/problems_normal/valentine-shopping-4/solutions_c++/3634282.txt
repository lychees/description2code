#include <iostream>
#include<cmath>
using namespace std;

int main()
{
	int t,l;
	cin>>t;
	for(l=0;l<t;l++)
	{
		int n,m,i,j,x,index=0;
		cin>>n>>m;
		for(i=0;i<n;i++)
		{
			double min=100;
			for(j=0;j<m;j++)
			{
                x=100;
				double a,b,c;
				double d,d1,d2;
				cin>>a>>b>>c;
				double a1=a/100;
				double a2=b/100;
				double a3=c/100;
			//	cout<<a1<<" "<<a2<<" "<<a3<<"\n";
				 d=x-(x*(a1));
				 d1=d-(d*(a2));
				 d2=d1-(d1*(a3));
			//	cout<<d<<" "<<d1<<" "<<d2<<"\n";
				if(d2<min)
				{ 
		     	min=d2;
				index=j;	
				}        
			}
			cout<<index+1<<" ";
			
		}
		cout<<"\n";
	}
	
   // cout << "Hello World!" << endl;
    return 0;
    
}
