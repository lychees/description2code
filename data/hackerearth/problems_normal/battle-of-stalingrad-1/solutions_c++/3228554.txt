#include<bits/stdc++.h>
using namespace std;
long  double cal(int a,int b,int c,int d,int e,int f)
{
      long  double  result=(  ((a*(d-f))+(c*(f-b))+(e*(b-d)))/2.0  );
        if(result<0)
        {
              result*=-1;
        }
        return result;
}
bool check(int a,int b,int c,int d,int e,int f,int g,int h)
{
         long double sum=cal(a, b,c,d,g,h);
         long double sum1=cal(a, b,g,h, e, f);
         long double sum3=cal(c, d,e,f,g, h);
         long double sum4=cal(a, b,c, d, e, f);
    //   cout<<sum4<<" "<<sum3<<" "<<sum1<<" "<<sum<<endl;
         if(sum4==(sum+sum1+sum3))
         {
              return true;
         }
         else
         {
              return false;
         }

}

int main()
{
         int t;
         cin>>t;
         while(t--)
         {
                 int a,b,c,d,e,f,g,h;
                 cin>>a>>b>>c>>d>>e>>f>>g>>h;
                 if(check(a,b,c,d,e,f,g,h)==true)
                 {

}
