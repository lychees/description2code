int INF = INT_MAX / 4;

int main() {
	int T; cin >> T;
	while (T--) {
		int n; cin >> n;
		vector<ll> a(n), b(n);
		for (int i = 0; i < n; i++)
			cin >> a[i];
		for (int i = 0; i < n; i++)
			cin >> b[i];
		vector<ll_ll> p(n);
		for (int i = 0; i < n; i++)
			p[i] = ll_ll(a[i], b[i]);
		sort(p.begin(), p.end());
		for (int i = 0; i < n; i++) {
			a[i] = p[i].first;
			b[i] = p[i].second;
		}
		vector<ll> x(n);
		ll sum = 0;
		for (int i = 0; i < n; i++) {
			x[i] = a[i] - sum;
			sum += b[i];
		}
		ll x0 = x[n / 2], ans = 0;
		for (int i = 0; i < n; i++)
			ans += abs(x[i] - x0);
		cout << ans << endl;
	}
}
