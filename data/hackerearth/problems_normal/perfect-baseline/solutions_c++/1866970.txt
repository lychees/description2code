#define MOD 1000000007
#define INF 1000000000000000000
#define PI acos(-1)

using namespace std;

long long GCD (long long a,long long b) {
	if (b == 0) return a;
	return(a % b == 0 ? b : GCD(b,a % b));
}

long long POW (long long base,long long exp) {
	long long val;
	val = 1;
	while (exp > 0) {
		if (exp % 2 == 1) {
			val = (val * base) % MOD;
		}
		base = (base * base) % MOD;
		exp = exp / 2;
	}
	return val;
}

vector <string> v;
string s,ans;

int main()
{
	int t;
	scanf("%d",&t);
	while (t --) {
		int n,k,i,j,mn,cost;
		char ch,cur;
		scanf("%d%d",&n,&k);
		v.clear();
		for (i = 0; i < n; i++) {
			cin >> s;
			v.push_back(s);
		}
		ans.clear();
		for (i = 0; i < k; i++) {
			mn = INT_MAX;
			cur = '#';
			for (ch = 'a'; ch <= 'z'; ch++) {
				cost = 0;
				for (j = 0; j < n; j++) {
					cost = cost + abs(v[j][i] - ch);
				}
				if (cost < mn) {
					mn = cost;
					cur = ch;
				}
			}
			ans.push_back(cur);
		}
		cout << ans << '\n';
	}
	return 0;
}
