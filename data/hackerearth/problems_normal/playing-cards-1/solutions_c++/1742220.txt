#include<bits/stdc++.h>
using namespace std;

typedef long long LL;

LL dp[1001][1001] = {0};

void precomp()
{
      dp[0][0] = dp[1][0] = dp[1][1] = 1;
      for(int i=2;i<=1000;i++)
      {
            dp[i][i] = dp[i][0]=  1;
            for(int j=1;j<i;j++) dp[i][j] = (dp[i-1][j] + dp[i-1][j-1]) % 1000000007;
      }
}

int main()
{
      precomp();
      int t;
      cin >> t;
      while(t--)
      {
            int n,m;
            cin >> n >> m;
            if(m%2) cout << 0 << endl;
            else
            {
                  LL ans = 0;
                  LL mod = 1000000007;
                  int q = m/2;
                  for(int redFace = 0;redFace <= q;redFace += 1)
                  {
                        int redRank = q - redFace;
                        int blackRank = redFace;
                        int blackFace = redRank;

                        LL temp = (((dp[6*n][redRank] * dp[6*n][blackRank]) % mod) * ((dp[18*n][blackFace] * dp[18*n][redFace]) % mod)) % mod;
                        ans = (ans + temp) % mod;
                  }
                  //for(int i = 1;i<=m;i++)
                        //ans = (ans * i) % (1000000007);
                  cout << ans << endl;
            }
      }
      return 0;
}
