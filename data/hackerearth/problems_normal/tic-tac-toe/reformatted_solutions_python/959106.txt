def function(n):
    if (n % 2 == 0):
        res = n-1
        s1 = res/2
        s2 = res-s1
        total = (n/2)*(s1**2) + (n/2)*(s2**2)
        return total
    else:
        res = n-1
        s1 = res/2
        total = (n)*(s1**2)
        return total


test = input()
for i in range(0, test):
    number = input()
    mod = 10**9+7
    k = number-1
    maxofn = k*(k+1)*(2*k+1)/6
    minofn = function(number)
    print minofn % mod, maxofn % mod
