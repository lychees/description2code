#include <iostream>
using namespace std;

int main()
{
    int t,n,f[100001],a[100001],max1,max2;
    int i,c;
    long int sum;
    cin>>t;
    while(t--)
    {
        cin>>n;
        sum=0;
        for(i=0;i<n;i++)
        {
            cin>>f[i];
            cin>>a[i];
            sum=sum+a[i];
            f[i]=a[i]+f[i];
        }
        max1=f[0];
        c=0;
        for(i=0;i<n;i++)
        {
            if(max1<f[i])
            {
                max1=f[i];
                c=i;
            }
        }

        if(c==0)
        max2=f[1];
        else
        max2=f[0];

        for(i=0;i<n;i++)
        {
            if(max2<f[i] && i!=c)
            {
                max2=f[i];
            }
        }
        cout<<"\n"<<(max1+max2)-sum;
    }
    return 0;
}
