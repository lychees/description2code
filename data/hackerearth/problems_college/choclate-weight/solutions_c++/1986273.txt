#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long t, x, flag, i, j;
    vector<long long> v;
    scanf("%lld",&t);
    while(t--){
        scanf("%lld",&x);
        v.push_back(x);
    }
    sort(v.begin(),v.end());
    scanf("%lld",&t);
    while(t--){
        scanf("%lld",&x);
        if(binary_search(v.begin(),v.end(),x)){
            printf("I Like You\n");
            continue;
        }
        flag=0;
        i=0;j=v.size()-1;
        while(i<j){
            if(v[i]+v[j]<x)
                i++;
            else if(v[i]+v[j]>x)
                j--;
            else{
                flag=1;
                break;
            }

        }
        if(flag==1)
           printf("I Like You\n");
        else{
           printf("I Hate You\n");
           //cout<<"s";
        }
    }
    return 0;
}
