#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define F(i,a,b) for(int i = (int)(a); i <= (int)(b); i++)
#define RF(i,a,b) for(int i = (int)(a); i >= (int)(b); i--)

int main()
{
    #ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
        freopen("output.txt","w",stdout);
    #endif
    int T,L,R,ans;
    scanf("%d",&T);
    while(T--)
    {
    	ans = INT_MIN;
    	scanf("%d%d",&L,&R);
    	F(i,L,R)
    	{
    		F(j,i,R)
    		{
    			ans = max(ans,i^j);
    		}
    	}
    	printf("%d\n",ans);
    }
    return 0;
}
