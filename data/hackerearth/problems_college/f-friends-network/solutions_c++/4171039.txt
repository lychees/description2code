#include<bits/stdc++.h>
using namespace std;

int n;
int a[1000][1000];
int v[1000];
//int mark[1000];
int dfs(int x,int p)
{
    v[x] = 1;

    for(int i=0;i<n;i++){
        if(a[x][i]==1 && i!=p && v[i]==0 && dfs(i,x))
            return 1;
        else if(a[x][i]==1 && i!=p && v[i]==1)
            return 1;
    }

    v[x] = 2;
    return 0;
}

int main()
{
    int t;
    cin>>t;

    while(t--){
        cin>>n;
        int f = 0;

        for(int i=0;i<=n+10;i++){
            v[i] = 0;
            //mark[i] = 0;
        }

        for(int i=0;i<n;i++)
            for(int j=0;j<n;j++)
                cin>>a[i][j];

        for(int i=0;i<n;i++){
            if(v[i]==0 && dfs(i,-1)){
                f = 1;
                break;
            }
        }

        if(f==1)
            cout<<"Aye"<<endl;
        else
            cout<<"Nae"<<endl;
    }

    return 0;
}
