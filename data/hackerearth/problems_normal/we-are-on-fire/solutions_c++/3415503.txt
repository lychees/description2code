#include <bits/stdc++.h>
using namespace std;
int a[1005][1005];
bool vis[1005][1005];
queue <pair<int,int> > q;
pair<int,int> p;
int ans,n,m;
int dx[4] = {1,-1,0,0};
int dy[4] = {0,0,1,-1};
void bfs(int i,int j){
	vis[i][j] = true;
	q.push({i,j});
	while(!q.empty()){
		p = q.front();
		q.pop();
		int i = p.first;
		int j = p.second;
		ans--;
		for(int k = 0;k < 4;k++){
			if(a[i+dx[k]][j + dy[k]]&& (i+ dx[k] > 0) && (i+ dx[k] <= n) && (j+dy[k] > 0) && (j +dy[k] <= m) && !vis[i+dx[k]][j+dy[k]]){
				q.push({i+dx[k],j+dy[k]});
				vis[i+dx[k]][j+dy[k]] = true;
		}
	}
	}
	printf("%d\n",ans);
}
int main()
{
    int q,x,y;
    scanf("%d %d %d",&n,&m,&q);
    for(int i = 1;i<= n;i++){
    	for(int j = 1;j<=m;j++){
    		scanf("%d",&x);
    		a[i][j] = x;
    		if(a[i][j])
    			ans++;
    	}
    }

    for(int i = 0; i < q;i++){
    	scanf("%d %d",&x,&y);
    	if(a[x][y] && !vis[x][y]){
    	//	cout << " hhh" << endl;
    		bfs(x,y);
    	}
    	else
    		printf("%d\n",ans);
    }
    return 0;
}
