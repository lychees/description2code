#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
LL n,i;
double p1,p2,ans,p,dp[1000008];
double solve(LL n)
{
	if(n <= 3)
	{
		if(n == 2)
		return p1;
		else if(n == 3)
		return p2;
		else 
		return 0;
	}
	else
	{
	   return (p1*solve(n-2) + p2*solve(n-3) );
	}
}
int main()
{
	cin>>n>>p;
	p1 = p /100;
	p2 = 1 - p1;
//	cout<<p1 <<" "<<p2<<endl;
//	ans  = solve(n);
    dp[0] = 1;
    dp[1] = 0;
    dp[2] = p1;
    dp[3] = p2;
    for(i = 4;i <= n;i++)
    dp[i] = p1*dp[i-2] + p2*dp[i-3];
	printf("%.6f\n",dp[n]);
}
