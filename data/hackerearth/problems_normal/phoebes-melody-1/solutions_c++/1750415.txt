#include <iostream>
using namespace std;

int main()
{
    int stime=0,N,k,T;
    scanf("%d",&T);
    for(int i=0;i<T;i++)
    {
    	scanf("%d",&N);
    	scanf("%d",&k);
    	int *v= new int[N];
    	for(int i=0;i<N;i++)
    	scanf("%d",&v[i]);
    	for(int j=0;j<N;j++)
    	{
    		for(int l=k;l<N;l++)
            { 
    		    if((j+l<N) && v[j]>v[j+l])
    		   { stime=stime+abs(j-(j+l));
    		    break;
    		   }
    		    else if((j-l>=0 ) && v[j]>v[j-l])
                { stime=stime+abs(j-(j-l));
                 break;
                }
            }
    	}
    	
         printf("%d\n",stime);
    	stime=0;
    }
    return 0;
}
