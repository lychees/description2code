#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	vector<int> a;
    	for(int i=0;i<n;i++){
    		int x;
    		cin>>x;
    		a.push_back(x);
    	}
    	sort(a.begin(),a.end());
    	int total=0;
    	for(int i=0;i<n;i++){
    		total++;
    		if(i!=n-1 && a[i]+1==a[i+1])i++;
    	}
    	cout<<total<<endl;
    }
    return 0;
}
