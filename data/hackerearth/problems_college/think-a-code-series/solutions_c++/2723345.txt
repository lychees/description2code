#include <cstdio>
typedef long long ll;
using namespace std;

int main()
{
    ll t, l, p, c;
    scanf("%lld",&t);
    for (ll i=1;i<=t;i++)
    {
    	scanf("%lld %lld %lld", &l, &p, &c);
    	ll m=l;
    	ll count=0;
    	while (m<p)
    	{
    		m*=c;
    		count++;
    	}
    	ll ans=0;
        while((1<<ans)<count)
        {
            ++ans;
        }
    	printf("Case #%lld: %lld\n", i, ans);
    }
    return 0;
}
