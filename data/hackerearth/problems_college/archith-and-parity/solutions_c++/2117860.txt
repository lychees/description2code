#include <iostream>
using namespace std;

int main()
{
    int t,n,count;
    cin>>t;
    while(t--){
    	cin>>n;
        count=0;
    //	cin>>n;
    	while(n){
    		if(n&1)
    		count++;
    		n=n>>1;
    	}
    	if(count%2==0){
    		cout<<"even"<<"\n";
    	}else {
    		cout<<"odd"<<"\n";
    	}
    }
    return 0;
}
