using namespace std;
typedef unsigned long long int ull;
typedef long long int ll;
typedef vector<vector<ll> > mat;
#define N 1000005

int dp[1002][1002];

int main(){
	int t;
	cin >> t;
	while(t--){
		int n,w;
		vector<int> Am(1002),Vi(1002);
		cin >> n >> w;
		forp(i,1,n){
			dp[i][0] = 0;
			dp[0][i] = 0;
		}
		forp(i,1,n) {
			cin >> Am[i];
		}
	    forp(i,1,n){
	    	cin >> Vi[i];
	    }
		forp(i,1,n){
			forp(j,1,w){
			   dp[i][j] = dp[i-1][j];
			   if(j >= Am[i])
			   		dp[i][j] = max(dp[i][j],dp[i-1][j-Am[i]]+Vi[i]);
			}
		}
		cout << dp[n][w] << "\n";
	}
 return 0;
}
