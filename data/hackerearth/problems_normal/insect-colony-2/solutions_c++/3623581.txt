#include <bits/stdc++.h>
typedef long long int ll;
using namespace std;
 
void fun()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int i;
    	ll a[n];
    	ll x=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		x=x^a[i];
		}
		if(x%2==0)
		{
			cout<<"Yes\n";
		}
		else
		cout<<"No\n";
		
		
	}
}
int main() {
	
	fun();
	return 0;
}
