#include <bits/stdc++.h>
using namespace std ;
int main()
{
	long long int test,temp,n,l ;
	cin >> test ;
	std::vector<long long int> v;
	while(test--)
	{
		cin >> n ;
		v.clear() ;
		for(long long int i=0;i<n;i++)
		{
			cin >> temp ;
			v.push_back(temp) ;
		}
		cin >> l ;
		long long int count1=1 ;
		long long int a[n] ;
		memset(a,0,sizeof(a)) ;
		a[0] = v[0] ;
		for(long long int i=0;i<n;i++)
		{
			if(v[i]<a[0])
			{
				a[0] = v[i] ;
			}
			else if(v[i]>a[count1-1])
			{
				if(count1<l)
				{
					a[count1] = v[i] ;
					count1++ ;
				}
			}
			else
			{
				auto it = lower_bound(a,a+count1,v[i]) -a;
				a[it] = v[i] ;
			}
		}
		if(count1>=l)
		{
			cout << a[count1-1] << "\n" ;
		}
		else
		{
			cout << -1 << "\n" ;
		}
	}
	return 0 ;
}
