#include <iostream>

using namespace std;

int main()
{
    int t,i,j,n,maxi,c;
    cin>>t;
    for(i=0;i<t;i++)
    {
        maxi=0;
        c=0;
        cin>>n;
        int arr[n];
        for(j=0;j<n;j++)
        {
            cin>>arr[j];
            if(arr[j]>maxi)
            {
                maxi=arr[j];
            }
        }
        int arr2[maxi+1];
        for(j=0;j<=maxi;j++)
        {
            arr2[j]=0;
        }
        for(j=0;j<n;j++)
        {
            arr2[arr[j]]++;
        }
        for(j=0;j<=maxi;j++)
        {
            if(arr2[j]>0)
            {
                c++;
            }
        }
        cout<<c<<"\n";
    }
    return 0;
}
