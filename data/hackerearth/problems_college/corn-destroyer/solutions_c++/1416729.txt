int INF = INT_MAX / 2;

int main() {
	int t; cin >> t;
	while (t--) {
		int x0, y0;
		cin >> x0 >> y0;
		int N; cin >> N;
		vector<double> a(N);
		for (int i = 0; i < N; i++) {
			int x, y;
			scanf("%d%d", &x, &y);
			a[i] = atan2((double)y - y0, x - x0);
		}
		if (N == 1) {
			printf("%.6f\n", 0);
			continue;
		}
		sort(a.begin(), a.end());
		double maxi = 0;
		if (N >= 2) maxi = a[0] - a[N - 1] + 2*M_PI;
		for (int i = 0; i + 1 < N; i++)
			maxi = max(maxi, a[i + 1] - a[i]);
		printf("%.6f\n", 100 - maxi / (2*M_PI) * 100);
	}
}
