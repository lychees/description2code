#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int num;
    	cin>>num;
    	long long int ocount=0, ecount=0;
    	while(num--)
    	{
    		long long int l;
    		cin>>l;
    		if (l&1)
    		ocount++;
    		else
    		ecount++;
    	}
    	cout<<ocount*ecount<<endl;
    }
    return 0;
}
