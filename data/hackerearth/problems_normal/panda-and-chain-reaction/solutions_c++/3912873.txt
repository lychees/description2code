#include <iostream>
#include <bits/stdc++.h>
#define ll long long int 
#define mod 1000003
#define val 1000000000000000000
using namespace std;
vector<ll> f;

int main()
{	
	ll temp;
	f.push_back(1);
	for(ll i=1;i<=mod;i++){
		temp = ((f[i-1]%mod)*(i%mod))%mod;
    		f.push_back(temp);
	}
	
    int t;
    scanf("%d",&t);
    while(t--){
    	ll n,x;
    	scanf("%ld%ld",&n,&x);
    	x = x%mod;
    	if(n>mod)
    		n = mod;
    	ll result = (x*f[n])%mod;
    	printf("%ld\n",result);
    }
    return 0;
}
