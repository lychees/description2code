#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int t;
    cin>>t;
    while(t--)
    {
    	long long int n;
    	cin>>n;
    	long long int x=0;
    	while(n>0)
    	{
    		n=(long long int)(n-(long long int)(pow(2,((long long int)(log2(n))))));
    		x++;
    	}
    	cout<<x<<endl;
    }

}
