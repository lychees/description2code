#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n, q, i, x, l, r;
    cin>>n>>q;
    int a[n];
    for(i=0;i<n;i++)
    cin>>a[i];
    while(q--)
    {
        int pizza[5]={16,9,26,26,1};
    	int s=0, k;
    	cin>>x>>l>>r;
    	if(x==1)
    	{
    	    for(i=l-1;i<r;i++)
            s+=a[i];
            cout<<s<<"/"<<(r-l+1)<<endl;
    	}
    	else if(x==0)
    	{
    		for(i=l-1,k=0;i<r;i++,k++)
    		{
    			a[i]=a[i]+pizza[k];
    			if(k==4)
                {
                    k=-1;
                }
    		}
    	}
    }
return 0;
}
