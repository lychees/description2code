#include<bits/stdc++.h>
using namespace std;
#define ll long long

int main()
{
	ll n,k;
	cin>>n>>k;
	map<int,int> m;
	bool flag=false;
	for(int i = 0;i<n;++i)
	{
		ll x;
		cin>>x;
		if(m.find(k-x)!=m.end())
		{
			cout<<"YES"<<endl;
			flag=true;
			break;
		}
		else
		m[x] = 1;
	}
	if(flag==false)
	cout<<"NO"<<endl;
	return 0;
}
