#include <iostream>
using namespace std;

int main()
{
    long long t,a,b,res;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	res=a;
    	for (long long i = a+1; i <=b; ++i)
    	{
    		/* code */
    		res=res&i;
    		if(res==0)
    		break;
    	}
    	cout<<res<<endl;
    }
    return 0;
}
