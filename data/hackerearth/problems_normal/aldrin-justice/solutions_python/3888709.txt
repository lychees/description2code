for _ in range(0, t):
    bt1, bt2, mt1, mt2 = map(int, raw_input().split())
    if (bt2 < mt1 and bt1 < mt1 and bt2 < mt2 and bt1 < mt2) or (mt1 < bt1 and mt2 < bt1 and mt1 < bt2 and mt2 < bt2):
        print "Line"
    else:
        if bt2 == mt1 or bt2 == mt2 or mt1 == mt2 or bt1 == bt2:
            print "Point"
        else:
            print "Nothing"
