#include <iostream>
using namespace std;
 
int main()
{
    int n,k,i,j,m,flag;
    cin>>n>>k;
    int a[n];
    int b[k];
    for(i=0;i<n;i++)
    	cin>>a[i];
    for(i=0;i<k;i++)
    	cin>>b[i];
    for(i=0;i<k;i++)
    {	
    	flag=0;
    	for(j=0;j<n;j++)
    	{
    		for(m=1;m<=360;m++)
    		{
    			if((m*a[j])%360==b[i])
    			{
    				flag=1;
    				break;
    			}
    		}
    		if(flag)
    			break;
    	}
    	if(flag)
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    
    return 0;
}
