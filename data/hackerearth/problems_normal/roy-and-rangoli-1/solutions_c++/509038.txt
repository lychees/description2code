#include <iostream>
#include<vector>
using namespace std;
bool arr[2001000];
vector<int> primes;
void sieve()
{
	long long int i,j;
	for(i=0;i<=2000000;i++)
	{
		arr[i]=true;
	}
	arr[0]=arr[1]=false;
	for(i=2;i<=2000000;i++)
	{
		if(arr[i])
		{
			primes.push_back(i);
			for(j=i*i;j<=2000000;j+=i)
			{
				arr[j]=false;
			}
		}
	}
}
int main()
{
    sieve();
    int n,i,res=0;
    scanf("%d",&n);
    for(i=0;i<primes.size()&&primes[i]<=2*(n-1);i++)
    {
    	if(primes[i]<n)
    	{
    		res=(res+primes[i]+1)%1000000007;
    	}
    	else
    	{
    		res=(res+(n-(primes[i]-n+1)))%1000000007;
    	}
    }
    printf("%d\n",res);
    return 0;
}
