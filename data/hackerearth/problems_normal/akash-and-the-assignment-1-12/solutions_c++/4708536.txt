#include <bits/stdc++.h>
#include <string>

using namespace std;

int arr[50005][26];
int main()
{
	int num,q;
	cin>>num>>q;
	string str;
	cin>>str;
	
	int x;
	for(int i=0;i<str.size();i++)
	{
		x=str[i]-'a';
		for(int j=0;j<26;j++)
		{
			if(j==x)
				arr[i+1][j]=arr[i][j]+1;
			else
				arr[i+1][j]=arr[i][j];
		}
	}
	
	int left,right,rank,k;
	for(int a=0;a<q;a++)
	{
		k=0;
		cin>>left>>right>>rank;
		int i;
		for(i=0;i<26;i++)
		{
			k+=arr[right][i]-arr[left-1][i];
			
			if(k>=rank)
				break;
		}
		char c='a'+i;
		
		if(i==26)
			cout<<"Out of range"<<endl;
		else
			cout<<c<<endl;
	}
	

}
