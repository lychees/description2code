#include<iostream>
#include<stdlib.h>
using namespace std;
int main(){
	long long int t;
	cin>>t;
	while(t--){
		long long int a,b,i,j;
		cin>>a>>b;
		long long int max=0;
		for(i=a;i<=b;i++){
			for(j=i+1;j<=b;j++){
				if(max<(i^j)){
					max=i^j;
				}
			}
		}
		cout<<max<<"\n";
	}
	return 0;
}
