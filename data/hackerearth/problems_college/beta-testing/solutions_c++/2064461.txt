#include<bits/stdc++.h>
using namespace std;
int main(){
	int N;
	string str;
	cin>>N;
	map<string,int>m;
	for(int i=0;i<N;i++){
		cin>>str;
		m[str]++;
	}
	map<string,int>::reverse_iterator it=m.rbegin();
	for(it;it!=m.rend();++it)
		cout<<it->first<<" "<<it->second<<endl;
	return 0;
}
