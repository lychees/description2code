#include<bits/stdc++.h>
#define mod 1048576
using namespace std;
typedef long long ll;

string s[1048576];

int main()
{
	
int n,k,q,l,atr;
cin>>n>>k>>l>>q;
string str;
for(int i=0;i<n;i++)
{
	cin>>str;
	int id=0;
	for(int j=0;j<k;j++)
	{
	 cin>>atr;
	 id=(id*(l+1))+atr;
	}
	s[id%mod]=str;
}
	for(int i=0;i<q;i++)
	{
		int id=0;
	  for(int j=0;j<k;j++)
	  {
	  	cin>>atr;
	  	id=(id*(l+1))+atr;
	  }
	  
	  if(s[id%mod].length()!=0)
	   cout<<s[id%mod]<<endl;
	   else
	   cout<<"You cant fool me :P\n";
	}
	system("pause");
return 0;	
}
