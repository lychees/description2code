#include <iostream>
using namespace std;

int main()
{
    int a,b=0,n,i,c=2;
    cin>>n;
    if(n%2==0){
    	b=1;
    	c=3;
    }
    for(;b<c;b++,n++){
    for(i=1;i<=n;i++)
    {
    	for(a=i;a<n;a++)
    		cout<<" ";
    	for(a=1;a<i;a+=2)
    		cout<<"/ ";
    	if(i%2==1)
    		cout<<"/\\";
    	for(a=1;a<i;a+=2)
    		cout<<" \\";
    	cout<<endl;
    }
    for(i=1;i<=n;i++)
    {
    	cout<<"  ";
    	for(a=1;a<i;a++)
    		cout<<" ";
    	for(a=n;a>i;a-=2)
    		cout<<"\\ ";
    	if((i%2)!=(b%2))
    		cout<<"\\/";
    	for(a=n;a>i;a-=2)
    		cout<<" /";
    	cout<<endl;
    }
    }
    return 0;
}
