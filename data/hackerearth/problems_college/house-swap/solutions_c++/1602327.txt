/* CodeMasters */

#include<bits/stdc++.h>

using namespace std;

#define sci(x) scanf("%d",&x)
#define scl(x) scanf("%lld",&x)
#define pri(x) printf("%d",x)
#define pll(x) printf("%lld",x)
#define nl printf("\n")
#define MOD 1000000007

typedef long long ll;

int main()
{
        int n;
        sci(n);
        int a[n],i,counter=0,mm[n],k=0;
        for(i=0;i<n;++i)
        {
                sci(a[i]);
                if(a[i]==i)
                {
                        counter++;
                }
                else
                {
                        mm[k++]=i;
                }
        }
        int ans=counter;
        if(counter==n)
        {
                pri(n-2); nl;
                return 0;
        }
        int flag=0;
        for(i=0;i<k;++i)
        {
//              pri(a[mm[i]]);
//              nl;
//              pri(a[a[mm[i]]]); nl;
                if(mm[i] == a[a[mm[i]]])
                {
                        flag=1;
                        break;
                }
        }
        if(flag)
        pri(ans+2);
        else
        pri(ans+1);
        nl;
        return 0;
}
