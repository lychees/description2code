#include <bits/stdc++.h>
using namespace std;

int main() 
{
    int T; 
    cin >> T;
    for(int i=1;i<=T;i++)
    {
        string s;
        cin >> s;
        string t = "", p = "";
        for(int i=0;i<s.size();i++)
        {
            p = t;
            t += s[i];
            p = s[i] + p;
            if(t < p) swap(t, p);
        }

        cout << t << endl;
    }
}
