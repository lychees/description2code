#include<bits/stdc++.h>
using namespace std;

int findSET(int x)
{

	int count = 0;
	while(x)
	{

		if(x%2==1)count++;
		x/=2;
	}
	return count;
}
int main()
{
	ios_base::sync_with_stdio(false);
    cin.tie(NULL);
   int t;
	cin>>t;
	while(t--)
	{
        int n, k;
		cin>>n>>k;
		int *arr = new int[n];
		//int *count = new int[n];
		int x;
		for(int i=0; i<n; i++){cin>>x;arr[i]=findSET(x);}
		sort(arr, arr+n);int sum = 0;
		for(int i=n-1; i>=n-k; i--)
			sum+=arr[i];
		cout<<sum<<endl;
			delete arr;
		
    }
return 0;
}
