#define sz(x) (int)x.size()

int w[50],h[50];

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i;
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            scanf("%d",&w[i]);
        }
        for(i=0;i<n;i++)
        {
            scanf("%d",&h[i]);
        }
        sort(w+1,w+n);
        sort(h,h+n,greater<int>());
        int flag = 1;
        for(i=1;i<n;i++)
        {
            if(h[i]*w[i] >= h[0]*w[0])
            {
                flag = 0;
                break;
            }
        }
        if(flag)
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
