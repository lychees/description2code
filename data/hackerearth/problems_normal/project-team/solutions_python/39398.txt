import sys

tests = int(raw_input(''))

for test in xrange(tests):
    inp = map(int, raw_input('').strip().split())
    noOfStudents = inp[0]
    knowledgeLevels = inp[1:]
    knowledgeLevels.sort()

    maxKnow = knowledgeLevels[0] + knowledgeLevels[-1]
    minKnow = knowledgeLevels[0] + knowledgeLevels[-1]

    for i in xrange(noOfStudents/2):
        s = knowledgeLevels[i] + knowledgeLevels[noOfStudents - i - 1]
        if s > maxKnow:
            maxKnow = s
        if s < minKnow:
            minKnow = s

    print maxKnow - minKnow
sys.exit(0)
