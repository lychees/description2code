#include <iostream>
#include <vector> 

using namespace std;

int main()
{
	int n; int max_e = 0, max_s = 0;
	cin >> n;
	vector<int> v(n);
	for (int i = 0; i < n; i++)
	{
		cin >> v[i];
		max_e += v[i];
		if (max_e < 0)
			max_e = 0;
		if (max_s < max_e)
		{
			max_s = max_e;
		}
	}
	cout << max_s << "\n";
	return 0;
}
