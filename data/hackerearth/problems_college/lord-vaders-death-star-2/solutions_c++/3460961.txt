#include <iostream>
using namespace std;
#include<map>
map<long long,int > ma;
int main()
{
	int n; cin>>n;
	int ans=-1;
	for(int i=0;i<n;i++)
	{
		long long x; cin>>x;
		ma[x]+=1;
	}
	map<long long,int>::iterator it;
	for(it=ma.begin();it!=ma.end();it++)
	{
		if(it->second==1)
		{
			ans=it->first;
		}
	}
	if(ans!=-1) cout<<ans; 
	else
	cout<<-1<<endl;
    return 0;
}
