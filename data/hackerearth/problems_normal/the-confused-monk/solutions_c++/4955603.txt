#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;
long long mod=1000000007;

void modExp(long long gcd,long long f){
long long ans=1;
while(gcd!=0){
    if(gcd%2==1){
        ans= (ans*f)% mod;
    }
    f = (f*f)%mod;
    gcd = gcd/2;
}
cout<<ans<<endl;
}
