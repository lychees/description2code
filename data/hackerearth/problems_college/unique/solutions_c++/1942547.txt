#include<stdio.h>
#include<stdlib.h>
#include<queue>
using namespace std;
typedef struct tree
{
int data;
struct tree* left;
struct tree* right;
}tree;
void insert(tree* &root,int data)
{
if(root==NULL)
{
root=new tree;
root->data=data;
root->left=NULL;
root->right=NULL;
}
if(root->data==data)
return ;
if(root->data<data)
insert(root->right,data);
if(root->data>data)
insert(root->left,data);
}
void print(tree* root)
{
queue<tree*> q;
q.push(root);
int c=0;
tree* s;
s=new tree;
while(!q.empty())
{
s=q.front();
q.pop();
c++;
if(s->left!=NULL)
q.push(s->left);
if(s->right!=NULL)
q.push(s->right);
}
printf("%d\n",c);
}
int main()
{
int t;
scanf("%d",&t);
while(t>0)
{
tree* root;
root=new tree;
root=NULL;
long long int n,a[1000000],i;
scanf("%lld",&n);
for(i=0;i<n;i++)
{
scanf("%lld",&a[i]);
insert(root,a[i]);
}
print(root);
t--;
}
return 0;
}
