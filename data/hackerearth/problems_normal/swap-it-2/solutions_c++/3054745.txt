#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t, n, k, temp;
	vector<int> arr;
	cin >> t;
	while(t--) {
		cin >> n >> k;
//		memset(pos, 0, 1000005 * sizeof(int));
		arr.clear(); // sorted.clear();

		for(int i = 0; i < n; i++) {
			cin >> temp;
			arr.push_back(temp);
//			sorted.push_back(temp);
//			sort(sorted.begin(), sorted.end());
		}
		
		for(int i = 0; i < n && k > 0; i++) {
			int min = 1000005;
			int pos = -1;

			for(int j = i; j < i + k + 1 && j < n; j++) {				
				if(arr[j] < min) {
					min = arr[j];
					pos = j;
				}
			}
			for(int j = pos; j > i; j--) {
				int temp = arr[j];
				arr[j] = arr[j - 1];
				arr[j - 1] = temp;
				k--;
			}
		}
			
		for(int i = 0; i < n; i++)
			cout << arr[i] << " ";
		cout << endl;
	}
return 0;
}
