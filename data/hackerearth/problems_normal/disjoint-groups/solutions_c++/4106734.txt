#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define pb push_back
#define mod 1000000007
ll pow_mod(ll a, ll b)
{
    ll r = 1;
	while (b > 0)
	{
		if(b % 2 == 1)
			r = (r * a) % mod;
		b = b/2;
		a = (a*a) % mod;
	}
	return r;
}
int main()
{
	ll n,x;
	scanf("%lld",&n);
	ll cur=0,i;
	for(i=0;i<n;i++)
	{
		scanf("%lld",&x);
		cur^=x;
	}
	if(cur==0)
	{
		ll ans=pow_mod(2,n-1);
		ans-=1;
		if(ans<0)
			ans+=mod;
		printf("%lld\n",ans);	
	}
	else
		printf("0\n");
	return 0;
}
