#include <iostream>
using namespace std;

int main()
{
	int i,j,n,temp;
   long long int a;
    cin>>n>>a;
    long long int k[100000];
   
    for( i=0;i<n;i++)
    {
    	cin>>k[i];
    }
    for(i=0;i<n;i++)
    {
    	for (j=0;j<n;j++)
    	{
    		if(k[i]<k[j])
    		{
    			temp=k[i];
    			k[i]=k[j];
    			k[j]=temp;
    		}
    		
    	}
    }
    i=0;
   while(a>k[i])
   {
   	a=a-k[i];
   	i++;
   }
   cout<<i;
    return 0;
}
