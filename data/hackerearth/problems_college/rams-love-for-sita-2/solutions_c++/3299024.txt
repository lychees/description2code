#include<iostream>
using namespace std;
int main()
{
int n,c,m,u,r,x,t,s;
cin>>x;
while(x--)
{
cin>>n>>c>>m;
  r=n/c;
  u=r;
  if(r<m)
      cout<<u<<endl;
  else
      {
      while(r>=m)
          {
          s=r/m;
          t=r%m;
          r=s+t;
          u+=s;
          }
      cout<<u<<endl;
      }
  }
  return 0;
  }
