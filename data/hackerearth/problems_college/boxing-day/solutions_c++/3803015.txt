#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
int i,j,k,n,cou[100001],x,y;
cin>>n;
for(i=1;i<=n;i++)
{
	cou[i]=1;
}
cin>>k;
for(i=0;i<k;i++)
{
	cin>>x>>y;
	cou[x]+=cou[y];
	cou[y]=0;
}
int s=0;
for(i=1;i<=n;i++)
{
	if(cou[i]!=0)
	{
		s+=1;
	}
}
cout<<s;
	return 0;
}
