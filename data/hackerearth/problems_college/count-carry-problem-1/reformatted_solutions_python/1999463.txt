def carry(x, y):
    carry = 0
    while x and y != 0:
        a1 = x % 10
        a2 = y % 10
        if a1 + a2 >= 10:
            carry = carry + 1
        x = int(x/10)
        y = int(y/10)
    if carry == 0:
        print "No carry operation"
    elif carry == 1:
        print "1 carry operation"
    else:
        print "%s carry operations" % carry


inp = int(raw_input())
for i in range(inp):
    arr = map(int, raw_input().split())
    carry(arr[0], arr[1])
