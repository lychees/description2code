T = int(raw_input().strip())
for x in range(0, T):
    a = raw_input().strip()
    flag = 0
    for x in a:
        if int(x) == 0 or int(x) == 1 or int(x) == 8:
            flag = 0
        else:
            flag = 1
            break
    if flag == 1:
        print "NO"
    else:
        if a == a[::-1]:
            print "YES"
        else:
            print "NO"
