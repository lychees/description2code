#include <iostream>
using namespace std;

typedef long long ll;
ll MOD = 1000000007;

int pow_mod(ll x, ll n, int m) {
	ll res = 1;
	for (; n > 0; n >>= 1) {
		if (n & 1) res = (res * x) % m;
		x = (x * x) % m;
	}
	return res;
}

int main() {
	int t; cin >> t;
	while (t--) {
		ll n; cin >> n;
		cout << (pow_mod(2, n, MOD) + MOD - 1) % MOD << endl;
	}
}
