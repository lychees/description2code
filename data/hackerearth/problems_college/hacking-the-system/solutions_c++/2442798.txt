#include <iostream>
#include <cstring>
using namespace std;

int main()
{
   char str1[9],str2[9];
   cin>>str1>>str2;
   int n=strlen(str1);
   int flag=0;
   for(int i=0;i<n;i++)
   	if(str1[i]!=str2[i])
   	flag++;
   	
   	if(flag<=1)
   	cout<<"wins\n";
   	else
   	cout<<"loses\n";
    return 0;
}
