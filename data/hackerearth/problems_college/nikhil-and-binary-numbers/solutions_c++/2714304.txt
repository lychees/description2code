#include<iostream>
#include<stdlib.h>
using namespace std;
int main(){
	long long int t;
	cin>>t;
	while(t--){
		long long int n,c=0,c1=0,x;
		cin>>n;
		for(int i=0;i<n;i++){
			cin>>x;
			if(x==1){
				c++;
			}
			else{
				c1++;
			}
			
		}
		if(c%2 == 0){
			cout<<c1<<"\n";
		}
		else{
			cout<<c<<"\n";
		}
	}
	
}
