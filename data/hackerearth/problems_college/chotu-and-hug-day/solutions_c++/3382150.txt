#include <iostream>
#include <stdlib.h>
#include <cmath>

using namespace std;

int comp(const void *a, const void *b)
{
	return (*((int *)a) - *((int *)b));
}

int main()
{
    int t, n, h, start, mid, end, min, max, i, ans;
    cin>>t;
    
    for ( ; t; t--) {
    	cin>>n>>h;
    	int hei[n];
    	for (i = 0; i < n; i++) {
    		cin>>hei[i];
    	}
    	
    	qsort(hei, n, sizeof(int), comp);
    	
    	start = 0; end = n - 1; ans = 999999;
    	while (start <= end) {
    		mid = start + (end - start) / 2;
    		if (hei[mid] == h) {
    			ans = 0;
    			break;
    		} else if (hei[mid] < h) {
    			if ((h - hei[mid]) < ans) {
    				ans = h - hei[mid];
    			}
    			start = mid + 1;
    		} else if (hei[mid] > h) {
    			if ((hei[mid] - h) < ans) {
    				ans = hei[mid] - h;
    			}
    			end = mid - 1;
    		}
    	}
    	cout<<ans<<endl;
    }
    
    return 0;
}
