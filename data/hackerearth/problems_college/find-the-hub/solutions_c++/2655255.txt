#include<bits/stdc++.h>
using namespace std;
int gcd(int a,int b) {
    if(b==0)
        return a;
    printf("%d %d\n",a,b);
    return gcd(b,a%b);
}
int main(void) {
    int n,a[100005],i;
    scanf("%d",&n);
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    sort(a,a+n);
    int f=0;
    int j,save;
    for(i=n-1;i>=0;i--) {
        save=a[i];
        f=0;
        for(j=2;j*j<=save;j++) {
            if(save%j==0) {
                f=1;
                break;
            }
        }
        if(f==0)
            break;
    }
    //printf("%d\n",i);
    int ss=i;
    save=a[i];
    i-=1;
    int cnt=1;
    while(i>=0) {
        if(a[i]==save)
            cnt++;
        else
            break;
        i--;
    }
    printf("%d\n%d\n",save,cnt);
    return 0;
}
