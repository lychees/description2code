#include <iostream>
#include <cstdio>

using namespace std;
char s1[35];
int ans = 0;

struct node
{
    node *ch[26];
};


node *getnode()
{
    node *p = new node;
    ans++;
    for(int i = 0;i < 26;++i)
        p->ch[i] = NULL;
    return p;
}


void insert(node *root, string s)
{
    node *p = root;
    int idx;
    for(int i = 0;i < s.length();++i)
    {
        idx = s[i] - 'a';
        if(p->ch[idx] == NULL)
        {
            p->ch[idx] = getnode();
        }
        p = p->ch[idx];
    }
}

int main()
{
    int n;
    string s;
    node *root = getnode();
    scanf("%d", &n);
    while(n--)
    {
        scanf("%s", s1);
        s = s1;
        insert(root, s);
    }
    printf("%d\n", ans);
    return 0;
}
