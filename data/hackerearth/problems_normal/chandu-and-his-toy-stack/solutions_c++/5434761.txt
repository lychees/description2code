#include <bits/stdc++.h>
using namespace std;
int a[100000];
int b[100000];
int main() {
	int t;
	scanf("%d",&t);
	while(t--){
		int n,x,y;
		scanf("%d%d%d",&n,&x,&y);
		for(int i=0;i<n;i++){
			scanf("%d%d",&a[i],&b[i]);
		}
		sort(a,a+n);
		sort(b,b+n);
		long long int ans=0;
		for(int i=0;i<n;i++){
			if(a[i]==b[i]){
				continue;
			}
			else if(a[i]>b[i]){
				ans+=(a[i]-b[i])*y;
			}
			else{
				ans+=(b[i]-a[i])*x;
			}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
