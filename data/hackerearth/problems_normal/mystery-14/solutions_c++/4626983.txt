#include <iostream>
using namespace std;

int main()
{
    int t;
	cin >> t;
	while (t--)
	{
		int n;
		cin >> n;
		int a[100000];
		int x;
		for (x = 0; x < n; x++)
		{
			cin >> a[x];
		}
		int i, sum=0;
		for (i = 0; i < n; i++)
		{
			if (a[i] % 2 != 0)
			{
				sum += a[i];
		    }
		}
		cout << sum << endl;
	}
    return 0;
}
