#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
    {
    	int n;
    	cin>>n;
    	int num=0;
    	while(n!=0)
    	 {
    	 	num+=n%10;
    	 	n/=10;
    	 }
    	 int cnt=0;
    for(int i=1;i<=num;i++)
     {
     	if(num%i==0)
     	 cnt++;
     }
     if(cnt==2)
     cout<<"YES"<<endl;
     else
     cout<<"NO"<<endl;
    }
    return 0;
}
