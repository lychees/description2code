#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str;
		cin>>str;
		int i;
		int alp[26]={'0'};
		alp[0]=0;
		int l=str.size();
		//cout<<l<<endl;
		if(l>=10)
		cout<<"YES"<<endl;
		else
		{
			int count=0;
			for(i=0;i<l;i++)
			{
			 if(alp[(int)str[i]-97]==0)
			 {
			 count++;
			 alp[(int)str[i]-97]++;
			 }
			}
	//cout<<count;
			if(count>=4)
			cout<<"YES"<<endl;
			else
			{
				if(count==1)
				cout<<"NO"<<endl;
				else if(count==2 && l>=7)
				cout<<"YES"<<endl;
			    else	if(count==3 && l>=5)
				cout<<"YES"<<endl;
				else
				cout<<"NO"<<endl;
			}
			
				
			}
		
	}
   // cout << "Hello World!" << endl;
    return 0;
}
