#include <bits/stdc++.h>

using namespace std;

int main()
{
    
    long long n, sum = 0, tem1;
    
    cin >> n;
    
    for(int i = 1; i <= n; i++){
        
        cin >> tem1;
        
        tem1 *= i;
        tem1 -= sum;
        sum += tem1;
        
        cout << tem1 << " ";
    }
    
    
    
    return 0;
}
