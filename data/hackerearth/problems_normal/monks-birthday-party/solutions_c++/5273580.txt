#include <iostream>
#include<set>
#include<string>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    set<string> setstr;
    int tc;
    int nostr;
    cin >>tc;
    string str;
    for(int it =1 ; it <= tc ; it++)
    {
    	cin >> nostr;
    	setstr.clear();
    	for(int i = 1 ; i <= nostr ; i++)
    	{
    		
    		
	    	cin>>str;
	    	if(setstr.find(str) == setstr.end())
	    	{
	    		setstr.insert(str);
	    	}
    	}
    	
    	for(auto s : setstr)
	    {
	    	cout<<s<<"\n";
	    }
    }
    
    
    return 0;
}
