#include <iostream>
using namespace std;

int main()
{
	int n;
	long long int a,b,cnt,x,y,c;
	cin>>n;
	while(n--) {
		cin>>a>>b;
		c=0;
		cnt=0;
		while(a>0 && b>0) {
			x = a%10;
			y = b%10;
			if(x+y+c>=10)
				cnt++;
			c = (x+y+c)/10;
			a/=10;
			b/=10;
		}
		if(cnt==0) 
			cout<<"No carry operation"<<endl;
		else if(cnt==1)
			cout<<"1 carry operation"<<endl;
		else 
			cout<<cnt<<" carry operations"<<endl;
	}
    return 0;
}
