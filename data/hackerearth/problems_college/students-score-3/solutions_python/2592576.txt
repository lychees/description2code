if __name__=='__main__':
    n = int(raw_input())
    arr = list(map(int,raw_input().split()))
    runs = 0
    score = 0
    if len(arr)==1 and arr[0]==0:
            print(0)
    else:
        for i in range(0,n):
            runs += arr[i]
            if runs < 0:
                runs = 0
            if runs > score:
                score = runs
        print score
