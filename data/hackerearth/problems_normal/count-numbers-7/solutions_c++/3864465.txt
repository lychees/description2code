#define rep(i,a,n) for(int i=a;i<n;i++)
#define srt(x) sort(x.begin(),x.end())
#define st(x) memset(x,0,sizeof x)
#define rl(x) scanf("%lld",&x)
#define ri(x) scanf("%d",&x)
#define gc getchar_unlocked
#define pi2 pair<pii,int>
#define pii pair<int,int>
#define MOD 1000000007
#define LL long long
#define MAX 100000001
#define ss second
#define INF 1<<25
#define ff first
int main()
{
	
	int k,t;
	ri(k),ri(t);
	LL prime[t];
	for(int i=0;i<k;i++)rl(prime[i]);
	std::vector< std::vector<LL> > v(12);
	for(int i=0;i<(1<<k);i++)
	{
		int c=0;
		LL res=1;
		for(int j=0;j<k;j++)
		{
			if(!(i&(1<<j)))
			{
				c++;
				res*=prime[j];
			}
		}
		v[c].push_back(res);
	}
	while(t--)
	{
		LL a,b,ans=0;
		cin>>a>>b;
		for(int i=0;i<k;i++)
		{
			ans+=(b/prime[i] - (a-1)/prime[i]);
		}
		for(int i=2;i<=k;i++)
		{
			for(int j=0;j<v[i].size();j++)
			{
				if(i&1)
				ans+=(b/v[i][j] - (a-1)/v[i][j]);
				else
				ans-=(b/v[i][j] - (a-1)/v[i][j]);	
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
