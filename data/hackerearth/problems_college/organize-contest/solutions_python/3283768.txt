d,dd = {},{}
a,b = [],[]
z = input()
for i in range(z):
	st = raw_input().split()
	nm,n = st[0], int(st[1])
	a.append(nm)
	b.append(n)
	try:
		d[nm]+=n
	except:
		d[nm]=n

mx = max(d.values())

for i in range(z):
	try:
		dd[a[i]]+=b[i]
	except:
		dd[a[i]]=b[i]
	if dd[a[i]]==mx:
		print a[i]
		break
