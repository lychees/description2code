#include <iostream>
using namespace std;

int main()
{
    long int N,d,b,cd=0,cb=0,id,ib,i,md=0,mb=0;
    cin>>N;
    cin>>id;
    for(i=1;i<N;i++)
    {
    	cin>>d;
    	cd=d-id;
    	id=d;
    	if(cd<0)
    		cd*=-1;
    	if(cd>md)
    		md=cd;
    }
    cin>>ib;
    for(i=1;i<N;i++)
    {
    	cin>>b;
    	cb=b-ib;
    	ib=b;
    	if(cb<0)
    		cb*=-1;
    	if(cb>mb)
    		mb=cb;
    }
    if(md>mb)
    	cout<<"Dom\n"<<md;
    else if(mb>md)
    	cout<<"Brian\n"<<mb;
    else
    	cout<<"Tie\n"<<md;
    return 0;
}
