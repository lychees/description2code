//Simple Probability

//If p is the probabilty that it rains in 24 hours 

//Then probabilty that it rains in 10 hours p*(10/24) 

//Probabilty that it rains in 2 hours of day p*(2/24) 

#include<bits/stdc++.h>
using namespace std;

int main(){

    int t;

    scanf("%d",&t);

    while(t--){

        double p;

        int time;
 
        scanf("%lf %d",&p,&time);

        double norain=pow(1-p,time/1440.0);

        double rain=1-norain;

        printf("%0.4lf\n",rain);                      


    }

    return 0;

}
