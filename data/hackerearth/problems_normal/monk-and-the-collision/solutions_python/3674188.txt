t=input()
assert(t>0 and t<11)
flag=[]
for i in range(0,t):
    hash1=0
    hash2=[]
    temp=0
    n=input()
    assert(n>0 and n<101)
    x=map(int,raw_input().split())
    for j in x:
        hash1=j%10
        hash2.append(hash1)
    for k in range(10):
        test=hash2.count(k)
        if test>0:
            temp=temp+test-1
    flag.append(temp)
    del hash2[:]
for o in range(0,t):
    print flag[o]
