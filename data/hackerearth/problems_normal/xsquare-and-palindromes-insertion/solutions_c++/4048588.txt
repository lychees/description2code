#include <iostream>
using namespace std;
#include<cstring>
#include<string.h>
int main()
{
	int t,i,j,k,l,n;
	cin>>t;
	char s[10000];
	int a[26];
	while(t--){
		cin>>s;
		l=strlen(s);
		memset(a,0,sizeof(int)*26);
		for(i=0;i<l;i++){
			a[s[i]-'a']++;
		}
		j=0;
		k=0;
		for(i=0;i<26;i++){
			if(a[i]==1){
				j++;
			}
			else if(a[i]>1 && a[i]%2==1){
				k++;
			}
		}
		if(j==1){
			cout<<k<<endl;
		}
		else if(k!=0){
			cout<<k+j-1<<endl;
		}
		else{
			cout<<0<<endl;
		}
	}
    return 0;
}
