#include <bits/stdc++.h>
using namespace std;

long gcd(long a, long b){
	if(b==0)
		return a;
	else
		return gcd(b,a%b);
}
long combination(long n, int r){
	long result=1;
	for(int i=0;i<r;i++)
		result=result*(n-i)/(i+1);
	return result;
}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	long n,k;
    	cin>>n>>k;
    	long kCount=0;
    	for(long i=0;i<n;i++){
    		long num;
    		cin>>num;
    		if(num==k)
    			kCount++;
    	}
    	long nr,dr;
    	if(kCount>=3)
    		nr = combination(kCount,3);
    	else
    		{cout<<"0/1\n";continue;}
    	dr = combination(n,3);
    	long f = gcd(nr,dr);
    	cout<<nr/f<<"/"<<dr/f<<"\n";
    }
    return 0;
}
