#include <iostream>
#include<bits/stdc++.h>
using namespace std;
long long pwr[10000];
long long sz;
void init()
{
    long long i=5;
    sz=0;
    while(i<=100000001000)
    {
        pwr[sz]=i;
        sz++;
        i=i*5;
    }
}
int main()
{
    int te;
    long long n;
    init();
    scanf("%d",&te);
    long long ans,i;
    while(te--)
    {
        ans=0;
        scanf("%lld",&n);
        for(i=0;i<sz;i++)
        {
            if(n<pwr[i])
                break;
            ans=ans+(n/pwr[i]);
        }
        printf("%lld\n",ans);
    }
    return 0;
}
