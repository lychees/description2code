n = int(raw_input())
z=[]
v=[]
k = []
b =(map(eval,raw_input().split()))
for i in range(len(b)):
    if(b[i]%2==0):
        z.append(b[i])
    else:
        v.append(b[i])
z.sort()
v.sort()
z.append(sum(z))
v.append(sum(v))
for j in range(len(z)):
    k.append(z[j])
for h in range(len(v)):
    k.append(v[h])
print ' '.join(map(str,k))
