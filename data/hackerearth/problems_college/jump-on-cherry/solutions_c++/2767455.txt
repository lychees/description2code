#include <iostream>
using namespace std;

int gcd(int a, int b)
{
	if (b==0)
	return 1;
	if (a%b==0)
	return b;
	else if (b%a==0)
	return a;
	else 
	{
	if (a>b)
	{
		gcd(b,a%b);
	}
	else gcd(a,b%a);
	}
}
int main()
{
    int t;
    cin>>t;
    while(t)
    {
    	int a,b;
    	cin>>a>>b;
    	int c;
    	c = (a+b)*a/(gcd(a+b,a));
    	c = c/a;
    	cout<<c<<endl;
    	t--;
    }
    return 0;
}
