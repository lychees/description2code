#include <iostream>
#include<string.h>
#include<algorithm>
using namespace std;
char array_palindrome[10001];
int main() {
	// your code goes here
	
	int n,length_of_array,cost;
	
	cin>>n;
 while(n--)
 {
	cost=0;
	cin>>array_palindrome;
	length_of_array=strlen(array_palindrome)-1;
	int i=0;
	while(i<length_of_array)
	{
	  if(array_palindrome[i]!=array_palindrome[length_of_array])
	  {
	   cost+=abs(int(array_palindrome[i]-array_palindrome[length_of_array]));
	  }
	  i++;
	  length_of_array--;
	}
	cout<<cost<<endl;
 }
	return 0;
}
