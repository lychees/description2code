#include<iostream>
#include<vector>

using namespace std;
int main()
{
	int n,a[5000];
	vector<int>b;
	long long int sum = 0;
	cin>>n;
	for(int i = 0; i < n; i++)
		cin>>a[i];
	b.push_back(a[0]);
	for(int i = 1; i < n; i++)
	{
		if(a[i] == 0)
		{
			b.pop_back();
		}
		else
			b.push_back(a[i]);
	}
	for(int i = 0; i < b.size(); i++)
		sum += b[i];
	cout<<sum;
	return 0;
}
