unsigned long long fibonacci(unsigned long long n)
{
    int i;
    unsigned long long f1 = 0;
    unsigned long long f2 = 1;
    unsigned long long fi;

    if(n == 0)
        return 0;
    if(n == 1)
        return 1;

    for(i = 2 ; i <= n ; i++ )
    {
        fi = f1 + f2;
        f1 = f2;
        f2 = fi;
    }
    return fi;
}
using namespace std;

int main(){
   int cases;
   scanf("%d",&cases);
    while(cases--)
    {
        int n;
        scanf("%d",&n);
        unsigned long long res = fibonacci(n+2);
cout<<res*res<<endl;    }
    return 0;
}
