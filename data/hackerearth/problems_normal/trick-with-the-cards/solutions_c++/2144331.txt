#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    long long int t,n;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld",&n);
    	if(n%3==0)
    		printf("%lld\n",n/3);
    	else if(n>3)
	   		printf("%lld\n",n);
    	else if(n>0)
    		printf("1\n");
    	else
    		printf("0\n");
    }
    return 0;
}
