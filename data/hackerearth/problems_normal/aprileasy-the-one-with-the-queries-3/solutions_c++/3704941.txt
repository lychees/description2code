#include<bits/stdc++.h>
using namespace std;
 
const int MAX =100005;
int n[MAX][26];
string str1,str2;
int main()	{
cin>>str1;
int x= str1.length();
for(int i=0;i<26;i++)	{
n[x][i]=x;

}
for(int i=x-1;i>=0;i--)	{
for(int j=0;j<26;j++)	{

n[i][j]=n[i+1][j];
//cout<<n[i][j];
}
n[i][str1[i]-'a']=i;
//cout<<"reached2";

}
int q;
cin>>q;
while(q-->0)	{
cin>>str2;
int pos=0;
bool flag=true;
int y=str2.length();
for(int i=0;i<y&&flag;i++)	{
//cout<<n[pos][str2[i]-'a'];
if(n[pos][str2[i]-'a']>=x){
flag=false;
//cout<<"ok";
}
else	{
pos=n[pos][str2[i]-'a']+1;
//cout<<"oka";
}
}
if(flag)
cout<<"Yes\n";
else
cout<<"No\n";
}
return 0;
}
