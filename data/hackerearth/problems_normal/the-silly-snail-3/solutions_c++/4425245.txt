#include <stdio.h>
#include<math.h>
#include<vector>
#include<iostream>
using namespace std;
long long int result=0;
int vis[100005]={0};
vector <int >v[100005];
void dfs(int a)
{
    vis[a]=1;
    if(a!=0)
    printf("%d ",a);
    for(int i=0;i<v[a].size();i++)
    {
        if(vis[v[a][i]]!=1)
        {
            dfs(a[v][i]);
        }
    }
}
int main()
{
    int t ;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        for(int i=0;i<=100005;i++)
        {
            vis[i]=0;
            v[i].clear();
        }
       for(int i=1;i<=n;i++)
       {
          int a,b,c;
          cin>>a>>b>>c;
          v[a].push_back(b);
          v[a].push_back(c);
       }
       dfs(1);cout<<endl;
    }
    return 0;
}
