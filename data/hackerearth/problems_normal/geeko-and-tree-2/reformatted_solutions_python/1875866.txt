def greek():
    i = 0
    input = int(raw_input())
    while i < input:
        val, power = raw_input().split()
        val = int(val)
        power = int(power)+1
        power = val**power
        i += 1
        power = power-1
        value = str(power/(val-1))
        value = list(value)
        sum = 0
        for l in value:
            sum = sum+int(l)
        print sum


if __name__ == "__main__":
    greek()
