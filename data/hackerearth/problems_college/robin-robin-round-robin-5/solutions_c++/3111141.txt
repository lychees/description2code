#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,q,c;
    scanf("%d%d",&n,&c);
    pair<long long int,int>p[100003];
    
    int i,dist[n+1];
    for(i=1;i<=n;i++)
    {
    	scanf("%d",&dist[i]);
    }
    scanf("%d",&q);
    long long int ans[n+1];
    ans[1]=0;
    for(i=1;i<n;i++)
    {
    	ans[i+1]=ans[i]+min(dist[i],c);
    }
    for(i=1;i<=n;i++)
    {
    	p[i].first=ceil(((double)dist[i])/c);
    	p[i].second=i;
    }
    sort(p+1,p+n+1);
    long long int ans2[100003];
    for(i=1;i<=n;i++)
    {
    //	cout<<p[i].first<<" "<<p[i].second<<endl;
    	ans2[p[i].second]=i;
    }
    while(q--)
    {
    	int m,d;
    	scanf("%d%d",&m,&d);
    	if(m==1)
    	{
    		printf("%lld\n",ans[d]);
    	}
    	else
        {
        	printf("%lld\n",ans2[d]);
        }
    }
    return 0;
}
