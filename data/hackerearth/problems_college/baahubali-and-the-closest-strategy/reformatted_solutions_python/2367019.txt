n, k = (int(i) for i in raw_input().split())
l = sorted([int(i) for i in raw_input().split()])
for kkk in range(0, k):
    q = input()
    i = 0
    j = n-1
    ans = 99999999
    soln = 0
    while i < j:
        s = l[i]+l[j]
        if s > q:
            j -= 1
        elif s < q:
            i += 1
        else:
            soln = q
            break
        if ans > abs(q-s):
            ans = abs(q-s)
            soln = s
    print soln
