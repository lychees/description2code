from math import sqrt


def isPrime(n):
    if n == 2 or n == 3:
        return True
    if n % 2 == 0 or n < 2:
        return False
    for i in range(3, int(n**0.5)+1, 2):   # only odd numbers
        if n % i == 0:
            return False

    return True


t = int(raw_input())
while (t > 0):
    s = 0
    l, m = map(int, raw_input().split())
    for i in range(l, m+1):
        if (isPrime(i)):
            s += i
    print(s)
    t -= 1
