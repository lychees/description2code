#include <bits/stdc++.h>
using namespace std;
long long int p[10000001],mod=1000000009,ans,p2[10000001],p3[10000001];
int main()
{
    long long int T=0,i,N;
	p[0]=1;
	for(i=1;i<=10000000;i++)
	p[i]=(26*p[i-1])%mod;
	p2[0]=1;
	for(i=1;i<=10000000;i++)
	p2[i]=(p2[i-1]*p[10000000])%mod;
	p3[0]=1;
	for(i=1;i<=10000000;i++)
	p3[i]=(p3[i-1]*p2[10000000])%mod;

    scanf("%lld",&T);
    while(T--)
    {
    	ans=1;
    	scanf("%lld",&N);
    	if(N%2==1)
    	N=N/2+1;
    	else
    	N=N/2;
    	long long int x=N/10000000;
    	long long int z=x/10000000;
    	long long int k=x%10000000;
    	long long int y=N%10000000;
    	ans=(ans*p3[z])%mod;
    	ans=(ans*p2[k])%mod;
    	ans=(ans*p[y])%mod;
    	printf("%lld\n",ans);
    }
    return 0;
}
