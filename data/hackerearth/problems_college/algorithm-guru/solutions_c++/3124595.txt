#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

const int max_x= 3010;
const int max_n= 3010;

ll arr[max_n], xarr[max_x];

int main()
{
	ll t, n, m, i, j, x;
	
	scanf("%lld", &t);
	
	while(t--)
	{
		scanf("%lld %lld", &n, &x);
		
		for(i=0;i<n;++i)
		{
			scanf("%lld", &arr[i]);
		}
		
		memset(xarr, 0, sizeof(xarr));
		
		xarr[0]= 1;
		
		for(i=0;i<n;++i)
		{
			for(j=0;j<=max_x-arr[i]-1;++j)
			{
				if(xarr[j])
				{
					//cout<<j+arr[i]<<"-------------"<<endl;
					xarr[j+arr[i]]= 1;
				}
			}
		}
		
		for(i=x;i>=0;--i)
		{
			if(xarr[i])
			break;
		}
		
		printf("%lld\n", i);
	}
	
	return 0;
}
