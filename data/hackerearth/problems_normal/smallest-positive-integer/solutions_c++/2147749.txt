#include<iostream>
#include<cstring>
using namespace std;

int main() {
	int count[10];
	long mod = 1000000007;
	long long i,j,n,temp;
	cin>>n;
	if (n == 0){
		cout<<"10";
		}else if ( n<10){
			cout<<n;
		} else {
			for(i=0;i<mod;i++)
			{
				memset(count,0,sizeof(count));
				temp = i*mod + n;
				for(j = 9; j>1 ; j--){
					while(temp%j == 0){
						temp/=j;
						count[j]++;
					}
				}
				if(temp == 1)
					break;
			}
			
			for(i=2;i<10;i++){
				for(j=1;j<=count[i];j++)
					cout<<i;
			}
			cout<<endl;
		}
		return 0;
	}
