t = input()
for _ in range(t):
    s = raw_input()
    s = list(s)
    q = 1
    l = 0
    for i in range(len(s)-1, 0, -1):
        if s[i] > s[i-1]:
            k = i-1
            q = 0
            break
    if (q == 1):
        print "no answer"
        continue

    for i in range(len(s)-1, k, -1):
        if s[i] > s[k]:
            l = i
            break
    s[k], s[l] = s[l], s[k]
    s[k+1:len(s)] = reversed(s[k+1:len(s)])
    print ''.join(s)
