#include<iostream>
using namespace std;
int FindMaxSum(long int arr[], int n)
{
  long int incl = arr[0];
  long int excl = 0;
  long int excl_new;
  int i;

  for (i = 1; i < n; i++)
  {
     excl_new = (incl > excl)? incl: excl;
     incl = excl + arr[i];
     excl = excl_new;
  }
   return ((incl > excl)? incl : excl);
}

int main()
{
    int t;
    cin>>t;
    long int a[10000],n;
    while(t--)
    {
        cin>>n;
        for(int i=0;i<n;i++)
            cin>>a[i];
        n=FindMaxSum(a,n);
        cout<<n<<endl;
    }
}
