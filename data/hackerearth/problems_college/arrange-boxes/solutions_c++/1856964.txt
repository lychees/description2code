#define REP(i,n)	for(int i=0;i<n;i++)
#define RE(i,j,n)	for(int i=j;i<n;i++)
using namespace std;
typedef long long LL;
typedef long L;
int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	int n;
	while(t--)
	{
		cin>>n;
		int a[n];
		int b[n];
		REP(i,n)	cin>>a[i];
		REP(i,n)	cin>>b[i];
		LL ans=0;
		REP(i,n)
		{
			int val=b[i];
			int pos=-1;
			if(a[i]==b[i])	continue;
			for(int j=i+1;j<n;j++)
			{
				if(a[j]==val)
					pos=j;
			}
			ans+=pos-i;
			int temp=a[i];
			a[i]=b[i];
			for(int j=i+1;j<=pos;j++)
			{
				swap(temp,a[j]);
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
