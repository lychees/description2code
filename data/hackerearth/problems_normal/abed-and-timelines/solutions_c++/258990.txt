#include <iostream>
#include <stdlib.h>
#include <string.h>
using namespace std;
typedef struct tim{
    int val;
    int id;
    int bit;
}tim;
#define gc getchar_unlocked
inline long long rd(){
    long long n = 0;
    int flag = 1;
    char c;
    while(1)
    {
        c = gc();
        if(c >= '0' && c <= '9')
        {  n = (n<<3) + (n<<1) + c - '0'; flag = 0;  }
        else if(flag != 1)
            break;
    }
    
    return  n;
}
int compare(const void *a, const void *b){
    
    tim *p = (tim *)a, *q = (tim *)b;
    return p->val-q->val;
    
}
long long mini(long long x,long long y){
    return x<y?x:y;
}
long long abso(long long x){
   return  x<0?(-x):x;
}
int main()
{
    long long n,m,t,x1,y1,x2,y2,price;
    t = rd();
    long long mi;
    while(t--){
        mi = 9999999999999;
        long long r,c,p,q;
        n = rd(); m =rd();
        x1 = rd(); y1 = rd();
        x2 = rd(); y2 = rd();
        price = rd();
        r = abso(x1-x2);
        c = abso(y1-y2);
        p = n-r;
        q = m-c;
        mi = mini(mi, r+c);
        mi = mini(mi, p+c);
        mi  = mini(mi, q+r);
        mi  = mini(mi,p+ q);
        printf("%lld\n",price*mi);
        if(price*mi <= 1000)
            printf("You saved the group.\n");
        else
            printf("Let go of the group.\n");
    }
    
    return 0;
}
