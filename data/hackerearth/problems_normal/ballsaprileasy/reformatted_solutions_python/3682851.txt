from math import *

n = input()
a = sorted(map(int, raw_input().split()))

sq = int(sqrt(a[0]))
if (sq*sq != a[0]):
    sq += 1

s = (sq*(sq+1)*(2*sq+1))/6

sq += 1

i = 1
while (i < n):
    while (sq*sq < a[i]):
        s += (sq*sq)
        sq += 1
    s += (sq*sq)
    sq += 1
    i += 1

print(s-sum(a))
