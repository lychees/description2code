#include <iostream>
#include <algorithm>

using namespace std;

long long gcd(long long a, long long b) {
    while (a && b) {
        if (a > b) a %= b;
        else b %= a;
    }
    return a + b;
}

long long a,b;
long long ans;
long long lcm[42];
int len[42];

int main() {
    lcm[1] = 1;
    for (int i = 2; i < 42; i++) {
        lcm[i] = lcm[i-1] / gcd(lcm[i-1], i) * i;
    }

    len[1] = len[2] = 1;
    for (int i = 3; i < 42; i++) {
        int j;
        for (j = 2; j < i; j++)
            if (i % j != 0) break;
        len[i] = len[j] + 1;
    }

    cin >> a >> b;

    while (a < 42 && a <= b) {
        ans += len[a];
        a++;
    }

    if (a > b) {
        cout << ans << '\n';
        return 0;
    }

    long long num = 0;
    for (int i = 41; i > 1; i--) {
        long long cur = b / lcm[i-1] - (a-1) / lcm[i-1];
        ans += (len[i] + 1) * (cur - num);
        num = cur;
    }

    cout << ans << '\n';
}
