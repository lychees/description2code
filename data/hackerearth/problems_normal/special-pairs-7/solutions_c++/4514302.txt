#include<bits/stdc++.h>
using namespace std;

int dp[23][(1<<22)+2];
int a[1000007];

long long func(int n)
{
for(int mask=0;mask<(1<<21);mask++)
{
for(int j=0;j<21;j++)
if(mask&(1<<j))
dp[j+1][mask]=dp[j][mask]+dp[j][mask^(1<<j)];
else
dp[j+1][mask]=dp[j][mask];
}
long long ans=0;
for(int i=1;i<=n;i++)
ans+=dp[21][((1<<21)-1)^a[i]];
for(int i=0;i<(1<<21);i++)
dp[0][i]=0;
return ans;
}

int main()
{
ios_base::sync_with_stdio(false);
//freopen("in.txt","r",stdin);
int t;
cin>>t;
while(t--)
{
int n;
cin>>n;
for(int i=1;i<=n;i++)
{
cin>>a[i];
dp[0][a[i]]++;
//cout<<a[i]<<endl;
}
cout<<func(n)<<endl;
}
return 0;
}
