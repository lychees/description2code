#include<iostream>
#include<algorithm>

using namespace std;

int no_of_sides(long long int n, int p)
{
	if(p<=0)
		return n;
	else if(p>0)
	{
		if(n<=3)
			return 3;
		else if(n%2==0)	 
			return no_of_sides((n/2)+1, p-1);
		else
			return no_of_sides((n/2)+2, p-1);
	}
}

int main()
{
	int tc, p;
	long long int n;
	cin >> tc;
	while(tc--)
	{
		cin >> n;
		cin >> p;
		cout<<no_of_sides(n,p)<<endl;
	}

}
