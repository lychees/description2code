#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;

int n;
//char A[101][202];
vector<char> A[101];

int main() {
	scanf("%d", &n);
	n++;
	for (int i = 0; i < n; i++) {
		int j = 0, t = n - i - 1;
		while (j < t) {
		    j++;
		}
		A[i].push_back('/');
		t = t % 2;
		for (j++; j < n; j++) {
			if (j % 2 == t) {
				A[i].push_back('/');
			}
			else {
				A[i].push_back(' ');
			}
		}
		if (i % 2) {
			A[i].push_back(' ');
			j++;
		}
		A[i].push_back('\\');
		t = j % 2;
		for (j++; j <= n + i; j++)
			if (j % 2 == t) {
		        A[i].push_back('\\');
			}
			else {
				A[i].push_back(' ');
			}
	}
	n--;
	for (int i = 0; i < n; i++) {
	    for (int j = n - i - 1; j > 0; j--)
			printf(" ");
		for (int j = 0; j < A[i].size(); j++)
    		printf("%c", A[i][j]);
    	printf("\n");
	}
	int t = n / 2 * 2;
	if (n % 2 == 0)
		t--;
	for (int i = n - 1, k = 0; i >= 0; i--, k++) {
		for (int j = n - t + k; j >= 0; j--)
			printf(" ");
		for (int j = A[i].size() - 1; j >= 0; j--)
			printf("%c", A[i][j]);
		printf("\n");
	}
	
	n++;
	t = n / 2 * 2;
	if (n % 2 == 0)
		t--;
	for (int i = 0; i < n; i++) {
	    for (int j = n - i - 1; j > 0; j--)
			printf(" ");
		for (int j = 0; j < A[i].size(); j++)
    		printf("%c", A[i][j]);
    	printf("\n");
	}
	for (int i = n - 1, k = 0; i >= 0; i--, k++) {
		for (int j = n - t + k; j >= 0; j--)
			printf(" ");
		for (int j = A[i].size() - 1; j >= 0; j--)
			printf("%c", A[i][j]);
		printf("\n");
	}
	
	return 0;
}
