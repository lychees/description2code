#include <iostream>
using namespace std;
int main()
{
	int a; cin >> a;
	for (int g=0;g<a; g++)
	{
		int tot=0; 
		string b;cin>>b;int cur=0; int s=0; 
		for (int g=0;g<b.length(); g++)
		{
			if (b[g]=='|') 
			{tot^=cur; cur=0;s++;}
			else cur++; 
		}tot^=cur;
		if (!s)
		{
			cout << "Bob" << '\n'; continue; 
		}
		if (tot==0)
		{
			cout << "Bob" << '\n'; 
		}
		else cout << "Alice" << '\n';
	}
    return 0;
}
