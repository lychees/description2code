#include <bits/stdc++.h>
using namespace std;
double distance(long int x,long int y,long int a,long int b)
{
    return sqrt((a-x)*(a-x)+(b-y)*(b-y));
}
int main()
{
    long long int n;
    cin>>n;
    vector <long double> x(n),dp(n),y(n),f(n);
    for(int i=0;i<n;i++)
    {
        cin>>x[i]>>y[i]>>f[i];
    }
    dp[0]=f[0];
    for(long long  int i=1;i<n;i++)
    {
        dp[i]=-1ll<<60;
        for(long int j=0;j<i;j++)
        {
            dp[i]=max(dp[i],dp[j]-distance(x[j],y[j],x[i],y[i])+f[i]);
        }
    }
    cout<<fixed<<setprecision(6)<<dp[n-1];
    return 0;
}
