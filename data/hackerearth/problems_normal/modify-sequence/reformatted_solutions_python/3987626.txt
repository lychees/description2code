'''
nInt = int(raw_input())
seq = [int(word) for word in raw_input().split()]

for i in range(nInt-1):
    if seq[i] > seq[i+1]:
        print "NO"
        break
    else:
        seq[i+1] -= seq[i]
else:
    if seq[nInt-1] > 0:
        print "NO"
    else:
        print "YES"
