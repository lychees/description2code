#include <iostream>
#include <cmath>
using namespace std;

int power(int n)
{
        int sum=1;
        for(int i=1;i<=n;i++){
            sum = sum*10;
        }

        return sum;
}

int total_digits(int n)
{
    int count=0;
    while(n){
        n = n/10;
        count++;
    }
    return count;
}

int KP(int n)                                  //n = 123
{
    int upper,lower,temp;
    int digits = total_digits(n);           //digits = 3; upper = 1; lower=23;

    if(digits%2!=0){
        temp = power((digits/2)+1);
        //cout<<temp;
        upper = n / temp;
        lower =  n % temp;
    }
    else{
        temp = power(digits/2);
        upper = n / temp;
        lower = n % temp;
    }

    return upper+lower;
}


int main()
{
    int low,up;
    cin>>low>>up;
    int count=0;

    for(int i=low; i<=up;i++){
        if(KP(i*i) == i){
                count++;
                cout<<i<<" ";
        }
    }

    if(count==0)    cout<<"INVALID RANGE"<<endl;
    return 0;
}
