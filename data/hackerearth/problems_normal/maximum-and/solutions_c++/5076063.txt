#include <iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    long long int a,i,j,max,p,b,k;
    for(i=0;i<t;i++)
    {
    	max=0;
    	cin>>a;
    	cin>>b;
    	
    	for(k=b;k>=b-10&&k>=a;k--)
    	{
    		for(p=k-1;p>=b-10&&p>=a;p--)
    		{
    		j=k&p;
    		//cout<<k<<"&"<<p<<"  "<<j<<endl;
    		if (j>max)
    		max=j;
    		}
    		
    	}
    	cout<<max<<endl;
    	
    }
    return 0;
}
