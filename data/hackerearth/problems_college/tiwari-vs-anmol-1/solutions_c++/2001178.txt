#include <bits/stdc++.h>
#define max1 1000001
#define m 1000000007
using namespace std;
int isprime[max1]={0};
void seive(){

for(int i=2;i<sqrt(max1);i++){
	if(isprime[i]==0){
	    
	  for(int j=i*i;j<max1;j+=i){
			if(isprime[j]==0){
				isprime[j]=i;
			}
		}
	}
}
}
int main()
{
	seive();
	
	isprime[0]=isprime[0];
	for(int i=1;i<max1;i++){
		isprime[i]=(isprime[i-1]%m+isprime[i]%m)%m;
	}
    int t,y;
    cin>>t;
    int l,r;
    while(t--){
    	cin>>l>>r;
    y=isprime[r]-isprime[l-1];
    cout<<y<<"\n";
    }
    return 0;
}
