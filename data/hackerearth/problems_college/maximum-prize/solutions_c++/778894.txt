#include<iostream>
#include<algorithm>

using namespace std;

int main() {
	long long int t, n, k, a[100001], i, sum;
	cin>>t;
	while(t--) {
		sum=0;
		cin>>n>>k;
		for(i=0; i<n; i++) {
			cin>>a[i];
		}
		sort(a, a+n);
		if(k>=n){
			for(i=0; i<n; i++) {
			sum+=a[i];
		}
		}
		else{
		for(i=n-1; i>n-1-k; i--) {
			sum+=a[i];
		}
		}
		cout<<sum<<endl;
	}
	return 0;
}
