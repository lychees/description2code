#include <iostream>
using namespace std;

int main()
{
    int a,b,c,n=0,test;
    cin>>a;
    while(a--){
    	n=0;
    	scanf("%d",&b);
    	for(int i=1;i<=b;i++){
    		for(int j=i+1;j<=b;j++){
    			c=i^j;
    			
    			if(c<=b)
    			n++;
    		}
    	}
    	cout<<n<<endl;
    }
    return 0;
}
