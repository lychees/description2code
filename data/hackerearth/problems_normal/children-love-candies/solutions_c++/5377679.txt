#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	unsigned long long int n,q;
    	cin>>n>>q;
    	for(int i=1;i<=q;i++)
    	{
    		if(n%2==0)
    		n=n/2;
    		else
    		n=n-(n+1)/2;
    		if(n>4)
    		n=n-n/4;
    	}
    	cout<<n<<endl;
    }
    return 0;
}
