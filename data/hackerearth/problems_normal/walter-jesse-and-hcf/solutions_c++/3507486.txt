#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
const int N = 512345;
LL A, B, K;
void solve() {
    cin >> A >> B >> K;
    assert(A>=1 && A<=10000000000);
    assert(B>=1 && B<=10000000000);
    assert(K>=1 && K<=10000000000);
    LL C = __gcd(A , B);
    vector<LL> res;
    for (LL i = 1 ; i * i <= C ; ++ i) {
        if (C % i == 0) {
            res.push_back(i);
            if (i * i != C)
                res.push_back(C / i);
        }
    }
    sort(res.begin() , res.end());
    if (K > res.size()) cout<<"No crymeth today\n";
    else cout<<res[res.size() - K]<<"\n";
}
int main() {
    int t; cin>>t; 
    while (t--) solve();//
}
