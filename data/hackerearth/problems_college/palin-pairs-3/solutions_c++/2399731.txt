#include<iostream>
#include<map>
#include<string>
#include<algorithm>
#include<vector>

using namespace std;

int main()
{
	int n;
	cin>>n;
	string s;
	long long int out = 0;
	map<string, int> mymap;

	for(int i=0;i<n;i++)
	{
		cin>>s;
		string cp = s;
		reverse(s.begin(), s.end());
		//cout<<"Reversed: "<<s;
		out=out + mymap[s];
		mymap[cp]++;
	}
	

	cout<<out;
	return 0;
}
