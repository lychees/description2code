t=int(raw_input())
for w in range(t):
    s=raw_input()
    b=len(s)
    if b % 2 != 0:
        s1=s[b/2+1::]
        s2=s[:b/2]
    else:
        s1=s[b/2::]
        s2=s[:b/2]
    if s1[::-1] == s2:
        print 'Palindrome'
    else:
        values = dict()
        for index, letter in enumerate(string.ascii_lowercase):
               values[letter] = index + 1
        mul=1
        for i in s:
            mul=mul*values[i]
        print mul
