T = int(raw_input())
s = set(['0', '1', '8'])
for i in range(0, T):
    M = (raw_input())
    N = set(str(M))

    if len(N) <= 3:
        if N.issubset(s) and M == M[::-1]:
            print "YES"
        else:
            print "NO"

    else:
        print "NO"
