#include<bits/stdc++.h>
using namespace std;
int dig(int n)
{
	  int ct = 0;
	  while(n)
	  {
	  ct++;
	  n=n/10;
	  }
	   return ct;
}
int rev(int n)
{
    int s=0;
    while(n)
    {
        s=s*10+n%10;
        n=n/10;
    }
     return s;
}
int main()
{
      int t;
      cin >> t;
      while(t--)
      {


            int n;
            cin >> n;
              int val = rev(n*n);
  if(ceil(sqrt(val))==floor(sqrt(val)) && sqrt(val)!= n && dig(val)==dig(n*n) && n == rev(sqrt(val)))
                cout << sqrt(val) << endl;
              else
                cout <<"No"<<endl;
      }

}
