#include <iostream>
#include<stdio.h>
using namespace std; 
#define ll long long
ll cross_product(ll x1, ll y1,ll x2,ll y2,ll cx,ll cy){ 
return (x2 - x1) * (cy - y1) - (y2 - y1) * (cx - x1);
}
bool inside(ll x1, ll y1,ll x2,ll y2,ll cx,ll cy)  {
return (min(x1,x2)<=cx && cx<=max(x1,x2) && 
min(y1,y2)<=cy && cy<=max(y1,y2));
}

bool intersect (ll x1, ll y1,ll x2,ll y2,ll cx,ll cy,ll dx,ll dy) 
{
    ll abc,abd,cda,cdb;
abc=cross_product(x1, y1, x2, y2, cx, cy);
abd=cross_product(x1, y1, x2, y2, dx, dy);
cda=cross_product(cx, cy, dx, dy, x1, y1);
cdb=cross_product(cx, cy, dx, dy, x2, y2);

if (((abc>0 && abd<0)||(abc<0 && abd>0)) &&
((cda>0 && cdb<0)||(cda<0 && cdb>0)))
return true;

if (abc == 0 && inside(x1, y1, x2, y2, cx, cy))return true;
if (abd == 0 && inside(x1, y1, x2, y2, dx, dy))return true;
if (cda == 0 && inside(cx, cy, dx, dy, x1, y1))return true;
if (cdb == 0 && inside(cx, cy, dx, dy, x2, y2))return true;

return false;
}



int main()
{
int t;
ll x, y, x1, y1, x2, y2;
cin >> t;
while(t--){
cin >> x1 >> y1 >> x2 >> y2 >> x >> y;
if(intersect(0, 0, x, y, x1, y1, x2, y2))
         cout <<"NO\n";
         else
            cout<<"YES\n";
}
return 0;
}
