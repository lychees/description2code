#include <iostream>
#include <string>
using namespace std;

int main()
{
    int n;
    cin>>n;
    string a[n];
    
    for(int i=0;i<n;i++){
    	cin>>a[i];	
    }
    for(int i=0;i<n;i++){
    	string word=a[i];
    	for(int j=word.size()-1;j>=0;j--){
    		cout<<word[j];
    	}
    	cout<<endl;
    }
    
    return 0;
}
