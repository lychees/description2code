#include<bits/stdc++.h>
using namespace std;
map<int, int> m;
int main()
{
    int n, x;
    int ans = -1;
    cin >> n;


    while(n--) {
        cin >> x;

        m[x]++;

    }



    for(map<int, int>::iterator it = m.begin(); it != m.end(); it++) {
        if(it->second == 1) {
            ans = max(ans, it->first);
        }
    }

    if(ans == -1)
        cout << -1 <<endl;
    else
        cout << ans << endl;
}
