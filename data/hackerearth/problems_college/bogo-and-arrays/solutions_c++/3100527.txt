int A[1<<21], closest[1<<21];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	
        int ans=0;
        int n,k;
        scanf("%d %d",&n,&k);
        for(int i=1;i<=n;i++)
            {
                scanf("%d",&A[i]);
            }
            closest[n+1]=-1;
        for(int i=n;i>=1;i--)
        {
            if(A[i]>k)
            {
           //     printf("%d ",A[i]);
                closest[i]=i;
            }
            else
            {
                closest[i]=closest[i+1];
            }
        }
        for(int i=1;i<=n;i++)
        {
            if(closest[i]==-1)
                continue;
            ans+=(n-closest[i]+1);
        }
        printf("%d\n",ans);

    }

 return 0;
}
