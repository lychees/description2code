#include <stdio.h>
using namespace std;

int main()
{
    	long t,i,b,a[1000001],x,y,sum1,sum;
	scanf("%ld%ld",&t,&b);
	
		for(i=1;i<=t;i++)
		scanf("%ld",&a[i]);
		while(b--)
		{
			sum=0;sum1=0;
		scanf("%ld%ld",&x,&y);	
		for(i=x; i<=y;i++)
		{sum+=a[i];
		sum1-=a[i];}
		printf("%ld %ld\n",sum,sum1+(2*a[y]));
		}
	return 0;
}
