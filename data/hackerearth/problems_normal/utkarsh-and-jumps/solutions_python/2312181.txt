s = raw_input()
s = s.split(' ')
n = int(s[0])
p = float(s[1])

steps = []
steps.append(0)
steps.append(0)

p2=float(p/100)
p3=float(1-p2)

#print p2, p3

steps.append(p2)
steps.append(p3)
i=4
while i<=n:
    steps.append((steps[i-2]*p2) + (steps[i-3]*p3))
    i+=1
print "%0.6f"%steps[n]
