#include <iostream>
#include <algorithm>
#include <map>
using namespace std;
map<int, int> posi;
int main()
{
    int N, K; cin >> N >>K;
    int arr[100010];
    for (int i = 0; i < N; i++) {
    	cin >> arr[i];
    	posi[arr[i]] = i;
    }
    for (int i = N, j = 0; i >= 1 && K > 0; i--, j++) {
    	if (posi[i] != j) {
    		int a = arr[j];
    		int b = posi[i];
    		swap(arr[j], arr[b]);
    		posi[i] = j;
    		posi [a] = b;
    		K--;
    	}
    }
    cout << arr[0];
    for (int i = 1; i < N; i++) cout << " " << arr[i]; cout <<"\n";
}
