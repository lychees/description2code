#define Max(a,b) ((a)>(b)?(a):(b)) 
#define Min(a,b) ((a)<(b)?(a):(b)) 
 
typedef long long ut; 
 
int ms1(int w, int h) { 
    return (w * h + 1) / 2; 
    } 
   
  int test(int w, int h) { 
    
    return ms1(w/2, h/2) + 
      ms1((w+1)/2, (h)/2) + 
      ms1((w+1)/2, (h+1)/2) + 
      ms1((w+0)/2, (h+1)/2); 
    } 
  
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int w,h;
        cin>>w>>h;
        ut ans = test(w,h);
        cout<<ans<<endl;
    }
    return 0;
}
