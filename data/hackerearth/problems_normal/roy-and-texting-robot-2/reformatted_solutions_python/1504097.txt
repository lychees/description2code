#!/usr/bin/python

T = int(raw_input())
S = None
PAD = {
    1: '.,?!1',
    2: 'abc2',
    3: 'def3',
    4: 'ghi4',
    5: 'jkl5',
    6: 'mno6',
    7: 'pqrs7',
    8: 'tuv8',
    9: 'wxyz9',
    0: '_0'
}

# current and last positoin of hand
HAND = 1
LAST = 0
COUNT = 0
result = []

# returns the number of press needed


def get_index(l):
    global HAND
    return PAD[HAND].find(l)

# increment total count


def inc_count():
    global COUNT
    COUNT += 1

# change the position of hand


def inc_hand():
    global HAND
    global LAST

    LAST += 1

    if HAND == 9:
        HAND = 0
    elif HAND == 0:
        HAND = 1
    else:
        HAND += 1

# checks if the letter is @ current hand position


def check_substr(l):
    global COUNT
    index = get_index(l)
    if index >= 0:
        return index + 1
    else:
        inc_hand()
        return check_substr(l)

# returns the presses needed for string


def get_time(S):
    global COUNT
    global LAST
    global HAND
    COUNT = 0
    LAST = 0
    HAND = 1
    for L in S:
        COUNT += check_substr(L)
        if LAST > 0:
            COUNT += 1
            LAST = 0
    return COUNT


while (T > 0):
    S = raw_input()
    time = get_time(S)
    result.append(time)
    T -= 1

for res in result:
    print res
