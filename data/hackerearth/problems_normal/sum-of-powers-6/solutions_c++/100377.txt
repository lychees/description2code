#include <stdio.h>
#include <math.h>
#include <map>
using namespace std;
int main()
{
	map <int,int> hm;
	int T,x,i,j;
	hm[1]=1;
	for(i=1;i<=1000;i++)
	{
		hm[pow(i,2)]=1;
	}
	j=0;
	for(i=1;i<=100;i++)
	{
		hm[pow(i,3)]=1;
	}
	j=0;
	for(i=1;i<=15;i++)
	{
		hm[pow(i,5)]=1;
	}
	j=0;
	for(i=1;i<=7;i++)
	{
		hm[pow(i,7)]=1;
	}
	j=0;
	for(i=1;i<=3;i++)
	{
		hm[pow(i,11)]=1;
	}
	hm[pow(2,13)]=1;
	hm[pow(2,17)]=1;
	hm[pow(2,19)]=1;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%d",&x);
		map<int,int>:: iterator it;
		bool flg=false;
		for(it=hm.begin();it!=hm.end();it++)
		{
			int val=x-it->first;
			if(val>0)
			{
				if(hm.find(val)!=hm.end())
				{
					printf("Yes\n");
					flg=1;
					break;
				}
			}
		}
		if(!flg)
			printf("No\n");
	}
}
