#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long n,m,a,b,c,temp;
    	cin>>n>>m>>a>>b>>c;
    	if(m>n)
    	temp=2*n*c+(m-n)*b;
    	else
    	temp=2*m*c+(n-m)*a;
    	if(m*b+n*a > temp)
    	cout<<m*b+n*a<<endl;
    	else
    	cout<<temp<<endl;
    }
    return 0;
}
