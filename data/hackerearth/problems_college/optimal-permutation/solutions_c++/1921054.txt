#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
    long long int t;
    cin>>t;
    while(t--)
    {
        long long int n,i,a[200005],x=0,y=0,ans=0,z=0;
        cin>>n;
        x=n;
        y=n/2;
        a[0]=y;
        for(i=1;i<n;i++)
        {
            if(i%2!=0)
            {
                a[i]=x;
                x--;
            }
            if(i%2==0)
            {
                y--;
                if(y==0)y=(n/2)+1;
                a[i]=y;

            }
            ans+=(abs(a[i]-a[i-1]));
        }
        cout<<ans<<endl;
    }
    return 0;
}
