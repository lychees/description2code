#include<bits/stdc++.h>
using namespace std ;
int n , a[102] , dp[10002] ;
int solve ( int m ) {
    int i ;
    for ( int it = 1 ; it <= m ; it++ ) {
        if ( dp[it] != 0 ) {
            for ( i = 0 ; i < n ; i++ ) {
                if ( it - a[i] >= 0 ) {
                    dp[it] = (!dp[it-a[i]]) ;
                    if ( dp[it] == 0 )
                        break ;
                }
            }
        }
    }
}
int main () {
    int t ;
    scanf ( "%d" , &t ) ;
    while ( t-- ) {
        int m ;
        scanf ( "%d %d" , &m , &n ) ;
        dp[0] = 1 ;
        for ( int i = 1 ; i <= m ; i++ ) {
            dp[i] = -1 ;
        }
        for ( int i = 0 ; i < n ; i++ ) {
            scanf ( "%d" , &a[i] ) ;
            dp[a[i]] = 0 ;
        }
        if ( dp[m] == 0 ) {
            cout << "Little Deepu\n" ;
            continue ;
        }
        solve(m) ;
        if ( dp[m] == 0 )
            cout << "Little Deepu\n" ;
        else
            cout << "Kate\n" ;
    }
    return 0 ;
}
