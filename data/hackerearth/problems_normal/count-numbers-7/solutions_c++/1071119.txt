#include<iostream>
#include<vector>
using namespace std;
long int gcd( long int a, long int b)
{
	 if( b == 0)
	 return a;
	 return gcd(b,a%b);
}
int main()
{
	vector< pair<long int,long int> > pp;
	  long int k,t,i,s,e,c,val,gc;
	  cin>>k>>t;
   long int a[k+1];
      for(i=0;i<k;i++)
      cin>>a[i];

    for (s = 1; s < (1 << k); ++s)   
    {
    	 c=0;
    	 val=1;
    	 gc=0;
        for (e = 0; e < k; ++e)      
        {
        	
            if (s & (1 << e))                  
             {
                c++;
                gc=gcd( gc , a[e]);
                val=val*a[e];
            
            }
        }
  
        if( c == 1)
           gc=1;
           pp.push_back( make_pair((val/gc),c) );      
    }
       while(t--)
       {
       	  long int ans=0;
       	   cin>>s>>e;
       	   for( i = 0; i < pp.size(); i++)
       	   {
   	       	   if(pp[i].second & 1)
   	       	   ans=ans+(e/pp[i].first-(s-1)/pp[i].first);
   	       	   else
   	       	   ans=ans-(e/pp[i].first-(s-1)/pp[i].first);
   	       }
   	       cout<<ans<<endl;
       }
    return 0;
}
