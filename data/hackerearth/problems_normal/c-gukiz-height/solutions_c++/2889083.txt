#include<iostream>
#include<cmath>
#include<algorithm>

using namespace std;

long long N, H, T, d, res = (1LL<<60);
long long S[100001];

int main(){
	cin >> N >> H;
	for(int i = 0; i < N; i++){
		cin >> S[i];
		if( i ) S[i] += S[i-1];
	}
	T = S[N-1];
	for(int i = 0; i < N; i++){
		d = i+1;
		if( H - (d * (d+1) / 2 ) <= S[i] ){
			res = min( res , d );
			break;
		}
		long double a = -N*N;
		long double b = -(2.0*T + 2.0*d*N + N);
		long double c = 2.0*(H-S[i])-d*d-d;
		//a /= (N*N); b /= (N*N); c /= (N*N); // normalize
		long double x = ( -b - sqrt( b*b - 4.0*a*c ) )/(2.0*a);
		res = min( res , d + N * (long long)ceil(x) );
	}
	cout << res << endl;
}
