#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	//
		string s;
	    int len;
	    int maxi;	
	    int temp_back;
	    int temp_for;
	    int ans;
	//
	
	
    for(int i=0;i<t;i++)
    {
    	//string s;
    	//cout<<"a"<<endl;
	    cin>>s;
	    len = s.length();
	    maxi = 1;
	    for(int j=1;j<len-1;j++)
	    {
	    	ans = 1;
	    	if(s[j-1]==s[j+1])
	    	{
	    		//cout<<"b"<<endl;
	    		ans=3;
	    		temp_back = j-2;
	    		temp_for = j+2;
	    		while(temp_back>=0 && temp_for<len)
	    		{
	    			//cout<<"c"<<endl;
	    			if(s[temp_back]==s[temp_for])
	    			{
	    				ans = ans + 2;
	    				temp_back--;
	    				temp_for++;
	    			}
	    			else
	    			{
	    				break;
	    			}
	    		}
	    	}
	    	maxi = (ans>maxi?ans:maxi);
	    }
	    cout<<maxi<<endl;
    }
    return 0;
}
