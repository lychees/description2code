#include <iostream>
using namespace std;
int main() {
	int t,n,c,ar[200][200],ans;
	cin>>t;
	while(t--) {
		cin>>n>>c;
		for(int i=1;i<=n;i++) {
			for(int j=1;j<=n;j++) {
				cin>>ar[i][j];
			}
		}
		
		for(int i=0;i<=n;i++) {
			ar[0][i]=0;
			ar[i][0]=0;
		}
		for(int i=1;i<=n;i++) {
			for(int j=1;j<=n;j++) {
				ar[i][j]=max(ar[i-1][j]+ar[i][j],ar[i][j-1]+ar[i][j]);
			}
		}
		ans=ar[n][n]-c;
		if(ans>=0)
		cout<<ans<<endl;
		else cout<<-1<<endl;
	}
	return 0;
}
