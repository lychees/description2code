string s,re;
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int T;
	int i,n;
	int t;
	cin>>T;
	while(T--)
	{
		re="OK";
		cin>>s;
		n=s.size();
		if(n%3!=0)
		{
			re="Not OK";
		}
		else
		{
			t=n/3;
			for(i=1;i<t;i++)
				if(s[i]!=s[0])
					re="Not OK";
			for(i=1+t;i<t*2;i++)
				if(s[i]!=s[t])
					re="Not OK";
			for(i=1+t*2;i<n;i++)
				if(s[i]!=s[t*2])
					re="Not OK";
			if(s[0]==s[t]||s[0]==s[t+t]||s[t]==s[t+t])
				re="Not OK";
		}
		cout<<re<<endl;
	}
//	system("pause");
	return 0;
}
