#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T,i,j,k,sqt;
    string S;
    char c;
    vector<char> v;
    cin>>T;
    while(T--)
    {
        cin>>S;
        int flag=0;
        int l=S.length();
        i=0;
        while(i<l)
        {
            j=S[i];
            flag=0;
            sqt=floor(sqrt(j));
            for(k=2;k<=sqt;k++)
            {
                if(j%k==0)
                {
                    flag=1;
                    break;
                }
            }
            if(flag==1)
            {
                c=j;
                v.push_back(c);
            }
            i++;
        }
        if(v.size()!=0)
        {
            for(i=0;i<v.size();i++)
                cout<<v.at(i);
        }
        else
        {
            cout<<"NULL";
        }

        cout<<"\n";
        v.clear();
    }

}
