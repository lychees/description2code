#include <iostream>
#include <algorithm>
using namespace std;
typedef struct newcitizens{
	int u;int v;
}nu;
typedef struct oldcitizens{
	int w;int x;
}old;
bool comp(nu x,nu y)
{
	return ((x.u)<(y.u));
}
int main()
{
	int test;
	cin>>test;
	while(test--)
	{
		int m,n;
		int g,h;
		int hctr=0;
		int gctr=0;
		cin>>m;
		cin>>n;
		nu arr[m];
		old ar[n];
		for(int i=0;i<m;i++)
		{cin>>arr[i].u;cin>>arr[i].v;}
		for(int i=0;i<n;i++)
		{cin>>ar[i].w;cin>>ar[i].x;}
		cin>>g;cin>>h;
		sort(arr,arr+m,comp);
		int j=0;
		for(int i=0;i<n && j<m;i++)
		{
			if(arr[j].u==ar[i].w)
			{
				if(arr[j].v==ar[i].x)
				hctr++;
			j++;
			gctr++;
			}
		}
		if(hctr==h)
		{
		cout<<"Great\n";continue;}
		if(gctr==g)
		cout<<"Good\n";
		else
		cout<<":(\n";
		
	}
	return 0;
}
