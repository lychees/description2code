#include<iostream>
#include<cmath>
#include<iomanip>
using namespace std;
int main()
{
    long long t, l, b,h;
    scanf("%lld", &t);
    long long ans;
    while(t--)
    {
        scanf("%lld %lld %lld", &l, &h, &b);
        ans = l<b ? (l<h ? l : h) : (b<h ? b : h);
        if(ans&1)
        printf("%lld.5 \n", ans>>1);
        else
        printf("%lld.0 \n", ans>>1);
    }
    return 0;
}
