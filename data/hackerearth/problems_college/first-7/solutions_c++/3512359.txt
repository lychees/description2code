typedef long long ll;
typedef pair<int,int> ii;

const int MAX = 100005;
const int MOD = 1000000000+7;
const int INF = 1000000000;

int main() {
    int t;
    cin >> t;
    while (t--) {
        ll n;
        scanf("%lld", &n);
        ll answer = (n*(n+1)*(2*n+1))/3 + (n*(n-1)*(2*n-1))/3;
        answer -= n*(n+1)/2;
        answer -= n*(n-1)/2;
        answer += n*n*(n-1);
        answer += n*n*(n-1);
        cout.setf(ios::fixed);
        cout << setprecision(6);
        cout << double(answer) / (n*n) << endl;
    }
}
