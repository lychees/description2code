#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1e2+5;

bool vis[MAXN];
vector<int> idx,val;
int ar[MAXN];
vector<int> adj[MAXN];
char s[MAXN];

void DFS(int s) {
	vis[s]=true;
	idx.push_back(s);
	val.push_back(ar[s]);
	for(int i=0;i<(int)adj[s].size();i++) if(!vis[adj[s][i]]) DFS(adj[s][i]);
}
int main() {
	int n;
	cin>>n;
	for(int i=1;i<=n;i++) scanf("%d",&ar[i]);
	for(int i=1;i<=n;i++) {
		scanf("%s",s+1);
		for(int j=1;j<=n;j++)
			if(s[j]=='Y') adj[i].push_back(j),adj[j].push_back(i);
	}
	for(int i=1;i<=n;i++) {
		if(!vis[i]) {
			DFS(i);
			sort(idx.begin(),idx.end());
			sort(val.begin(),val.end());
			for(int j=0;j<(int)idx.size();j++) ar[idx[j]]=val[j];
			idx.clear();
			val.clear();
		}
	}
	for(int i=1;i<=n;i++) printf("%d ",ar[i]);
	printf("\n");
	return 0;
}
