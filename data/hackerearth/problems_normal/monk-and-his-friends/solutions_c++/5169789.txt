#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--){
    	int n,m;
    	cin>>n>>m;
    	set<long long int>clas;
    	long long int x;
    	for(int i=0;i<n;i++){
    		cin>>x;
    		clas.insert(x);
    	}
    	long long int arr[m];
    	for(int i=0;i<m;i++)
    		cin>>arr[i];
    	for(int i=0;i<m;i++){
    		
    		if(clas.find(arr[i])!=clas.end())
    			cout<<"YES"<<endl;
    		else
    			cout<<"NO"<<endl;
    		clas.insert(arr[i]);
    	}
    }
    return 0;
}
