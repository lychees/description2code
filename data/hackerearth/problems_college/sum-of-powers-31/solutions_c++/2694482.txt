using namespace std;
bool perfect[1000007];
const int MAX = 1000000;
int main()
{
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);
	for (int i = 2; i <= MAX; i++)
	{
		for (long long j = 1LL*i*i; j <= MAX; j *= i)
		{
			perfect[j] = true;
		}
	}
	set<int>go;
	go.insert(1);
	for (int i = 1; i <= MAX; i++)
	{
		if (perfect[i])
		{
			go.insert(i);
		}
	}
	int t;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++)
	{
		int x;
		scanf("%d", &x);
		bool ok = false;
		for (auto cur : go)
		{
			if (go.count(x - cur))
			{
				ok = true;
				break;
			}
		}
		if (ok)
		{
			printf("Yes\n");
		}
		else
		{
			printf("No\n");
		}
	}
}
