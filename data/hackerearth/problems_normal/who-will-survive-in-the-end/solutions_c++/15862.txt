#include <iostream>
using namespace std;
int main()
{
        int tst;
        cin>>tst;
        while(tst--) {
                int value,siz=0,get=1;
                //cout<<"asdfa\n";
                cin>>value;
                int val=value;
                while(val>1) {
                  //    cout<<"asdfa\n";
                        val/=2;
                        siz++;
                }
                //cout<<siz<<"\t"<<val<<endl;
                while(siz>0) {
                        get*=2;
                        siz--;
                }
                //cout<<siz<<"\t"<<value<<"\t"<<get<<endl;
                int diff=value-get;
                cout<<1+diff*2<<endl;
        }
        return 0;
}
