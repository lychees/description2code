#include<bits/stdc++.h>
using namespace std;
struct node
{
    int x;
    int y;
};
node ar[1000100];
bool operator <(node a,node b)
{
    if (a.x==b.x) return a.y>b.y;
    else return a.x<b.x;
}

int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        scanf("%d%d",&ar[i].x,&ar[i].y);
    }
    sort(ar,ar+n);
    for(int i=0;i<n;i++)
    {
        printf("%d %d\n",ar[i].x,ar[i].y);
    }

}
