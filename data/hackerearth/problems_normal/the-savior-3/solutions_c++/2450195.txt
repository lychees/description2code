#include <iostream>
using namespace std;

int main()
{	int sum=0;
  int t=0,n=0;
  cin>>t;
  for(int i=1;i<=t;i++)
  {
  	cin>>n;
  	int*a=new int[n];
  	for(int j=0;j<n;j++)
  	cin>>a[j];
  	for(int j=0;j<n-1;j++)
  	{
  		for(int k=j+1;k<n;k++)
  		{
  			if((a[j]+a[k])%2==0 && a[j]!=a[k])
  			sum=sum+1;
  		}
  		
  	}
  	cout<<sum<<"\n";
  	sum=0;
  	
  	
  }
    return 0;
}
