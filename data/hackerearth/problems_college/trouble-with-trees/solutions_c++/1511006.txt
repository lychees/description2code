#include<iostream>
using namespace std;
long catalan[1001];
int MOD=1000000007;

void cat(){

    for (int i=0;i<=1000;i++){
        if (i==0 || i==1){
            catalan[i]=1;
        }
        else{
            long sum =0;long left, right;
            for (int k=1;k<=i;k++){
                left = catalan[k-1] % MOD;
                right= catalan[i-k] % MOD;
                sum =(sum+ (left * right)%MOD)%MOD;
            }
            catalan[i]=sum;
        }
    }

}
int main()
{
    int t;
    cin>>t;
    cat();
    while(t--)
    {
              long int n;
              cin>>n;
              cout<<catalan[n]<<"\n";
    }
    
              return 0;
}
