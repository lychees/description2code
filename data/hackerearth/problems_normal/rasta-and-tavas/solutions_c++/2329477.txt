#include <iostream>
#include <stdio.h>
using namespace std;
bool solve(long long int n,long long int k){
	long long int p=0,i;
	if(n%2 == 0){
		p += 2;
		while(n%2 == 0)
			n = n/2;
	}
	for(i=3; i*i<=n; i+=2){
		if(n%i == 0){
			p += i;
			while(n%i == 0)
				n = n/i;
		}
	}
	if(n>2)
		p += n;
	//printf("p %lld\n",p);
	if(p == k)
		return true;
	else 
		return false;
}
int main()
{
    long long int n,k,i,count=0;
    scanf("%lld%lld",&n,&k);
    for(i=n; i>=1; i--){
    	if(solve(i,k))
    		count++;
    }
    printf("%lld",count);
    return 0;
}
