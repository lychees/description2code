const int MAXN = 2e3 + 3;
const long long INFLL = 100000000000000000LL;

int pr[MAXN], ta[MAXN];
vector <pair <int, int> > lst;
long long DP[MAXN];

int main() {
	int t; cin >> t;
	while(t--) {
		int n, k; cin >> n >> k;
		for(int i = 0; i < n; i++) {
			cin >> pr[i];
		}
		for(int i = 0; i < n; i++) {
			cin >> ta[i];
		}
		lst.clear();
		for(int i = 0; i < n; i++) {
			lst.pb(mp(pr[i], ta[i]));
		}
		sort(lst.begin(), lst.end());
		for(int i = 0; i < n; i++) {
			long long&res = DP[i];
			res = INFLL;
			int total_tasks = 0;
			for(int j = i; j >= 0; j--) {
				total_tasks += lst[j].second;
				long long now_price = (total_tasks + k) * (long long) lst[i].first + (j > 0 ? DP[j - 1] : 0);
				res = min(res, now_price);
			}
		}
		cout << DP[n - 1] << '\n';
	}
	return 0;
}
