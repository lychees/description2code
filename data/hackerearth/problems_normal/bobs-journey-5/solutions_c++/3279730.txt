#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	map<char,int> map;
    	for(int i=0;i<n;i++){
    		string s;
    		cin>>s;
    		char k=s.at(0);
    		//cout<<k<<endl;
    		map[k] +=1;
    	}
    	int flag=0;
    	for(auto i=map.begin();i!=map.end();i++){
    		if(i->second>1){
    			flag=1;
    			cout<<"NO"<<endl;
    			break;
    		}
    	}
    	if(flag==0) cout<<"YES"<<endl; 
    }
    return 0;
}
