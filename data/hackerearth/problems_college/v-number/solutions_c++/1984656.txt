#include <iostream>
using namespace std;

int main()
{
    int t; cin >> t;
    while(t--)
    {
    	string s; cin >> s;
    	int i=0,l=s.length(),sum1=0,sum2=0;
    	while(i<l)
    	{
    		if(i<l/2)
    			sum1 += int(s[i]-'0');
    		else
    			sum2 += int(s[i]-'0');
    		i++;
    	}
    	cout << (sum1==sum2?"YES":"NO") << endl;
    }
    return 0;
}
