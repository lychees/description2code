#include <iostream>
using namespace std;

int main() {
	int n,k,arr[1000005];
	cin>>n>>k;
	if(k>n) {
		cout<<0<<endl;
		return 0;
	}
	for(int i=0;i<n;i++)
		cin>>arr[i];
	int ans=0;
	for(int i=0;i<n-k;i++) {
		if((arr[i]+arr[i+k])%k==0 || (arr[i]-arr[i+k])%k==0 ) {
			ans++;
		}
	}
	cout<<ans<<endl;
	// your code goes here
	return 0;
}
