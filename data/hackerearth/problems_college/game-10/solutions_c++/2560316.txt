#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long n,m,k,diff;
    	cin>>n>>m;
    	k=n/2;
    	if(m>=k)
    	 {
    		   diff=m-k;
    		   if(n%2==0)
    		   cout<<n-2-2*diff<<endl;
    		   else
    		   cout<<n-1-2*diff<<endl;
    	 }
    	 else
    	 {
    	 	   k--;
    	 	   diff=k-m;
    		   if(n%2==0)
    		   cout<<n-1-2*diff<<endl;
    		   else
    		   cout<<n-2-2*diff<<endl;
    	 }
    	
    }
    return 0;
}
