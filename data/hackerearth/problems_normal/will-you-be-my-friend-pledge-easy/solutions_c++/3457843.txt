#include <iostream>
using namespace std;

int f(int p, int q)
{
  if(p==1 || q==1)
	return 0;
  
  else if(p>q)
   f(p-q,q);
  
  else if(q>p)
   f(p,q-p);
  
  else if(p==q)
  return 1;
  
  else
  return 0;
}

int main()
{
    int a,n,c=0;
    
    cin>>a>>n;
    int x[n];
    for(int i=0;i<n;i++)
    {
    	cin>>x[i];
        if(f(a,x[i]))
        c++;
    }
    cout<<c;
    
    
    
    return 0;
}
