#include <iostream>

using namespace std;

int main()
{
    int n;

    cin>>n;

    int v[n];

    for(int i=0;i<n;i++)
    {
        cin>>v[i];
    }

    int s=0,s1=0;

    for(int i=0;i<n;i++)
    {
        s+=v[i];

        if(s<0)
        {
            s=0;
        }

        if(s1<s)
        {
            s1=s;
        }
    }

    cout<<s1;

    return 0;
}
