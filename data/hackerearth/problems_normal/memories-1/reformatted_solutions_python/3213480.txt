import sys
import operator
import copy


def makeImage(i, N):
    arr = []

    element = 1
    while (i):
        if i % 2 == 1:
            arr.append(element)
        i /= 2
        element += 1
    return arr


T = input()
for _ in xrange(T):
    N, P = map(int, raw_input().strip().split(' '))

    pairs = []

    for _ in xrange(P):
        A, B = map(int, raw_input().strip().split(' '))
        pairs.append((A, B))

    Range = 2**N
    count = 0
    for i in range(1, Range):
        image = makeImage(i, N)

        correctImage = True

        for p in pairs:
            if p[0] in image and p[1] in image:
                correctImage = False
                break
        if correctImage:
            count += 1
    print count
