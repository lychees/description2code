//vector<int> g[2*100000+5];int visited[2*100000+5];
lld ans[1000];
int main()
{
	for(int a=1;a<=100;++a)
	{
		for(int b=1;b<=100;++b)
		{
			for(int c=1;c<=100;++c)
			{
				if(a==b or b==c or a==c)continue;
				int p=__gcd(a,b);
				p=__gcd(p,c);
				ans[p]++;
			}
		}
	}
	int t;
	cin>>t;
	lld x,y;
	while(t--)
	{
		scanf("%lld%lld",&x,&y);
		lld te=0;
		for(int i=1;i<=100;++i)
		{
			if(abs(y-x)%i==0)te+=ans[i];
		}
		printf("%0.6f\n",te/(100.0*99.0*98.0));
	}
}
