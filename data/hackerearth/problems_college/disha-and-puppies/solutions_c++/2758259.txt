void sb(int sum)
{
	for (int i = 0; i <= n; i++)
        dp[0][i]=true;

	for (int i = 1; i <= sum; i++)
        dp[i][0]=false;

	for (int i = 1; i <= sum; i++){
        for (int j = 1; j <= n; j++){
            dp[i][j] = dp[i][j-1];
            if (i >= a[j-1])
                dp[i][j] = dp[i][j] || dp[i - a[j-1]][j-1];
        }
	}
}
