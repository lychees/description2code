#include<stdio.h>
#include<string.h>
#include<ctype.h>
int main(){
	int i, s = 0, inl[26] = {0}, inu[26] = {0}, l, m, n, p;char a[101]; scanf("%[^\n]s", a); l = strlen(a);
	for(i = 0; i < l; s += a[i++]); s = s / l;
	if(s % 2 == 0)	for(i = l - 1; i >= 0; printf("%c", a[i--]));
		else{
			for(i = 0; i < l; i++)	isupper(a[i]) ? (inu[a[i] - 65]++) : inl[a[i] - 97]++;
			m = inl[0], l = 0; for(i = 0; i < 26; i++)	if(inl[i] > m)	l = i, m = inl[i];
			p = inu[0], n = 0; for(i = 0; i < 26; i++)	if(inu[i] > p)	n = i, p = inu[i];
			(m > p) ? printf("%c", l + 97) : printf("%c", n + 65);
		}		
	return 0;
}
