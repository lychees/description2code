#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		double d[n];
		double f[n];
		int i,j;
		for(i=0;i<n;i++)
		cin>>d[i];
		for(i=0;i<n;i++)
		cin>>f[i];
		double e[n];
		double sum=0.0;
		for(i=0;i<n;i++)
		{
			e[i]=d[i]*f[i];
			sum+=e[i];
		}
		for(i=0;i<n;i++)
		e[i]=e[i]/sum;
		int ind;
		double maxi=-1;
		for(i=0;i<n;i++)
		{
			if(e[i]>maxi)
			{
				maxi=e[i];
				ind=i;
			}
		}
		printf("%d %.3lf\n",ind,maxi);
	}
}
