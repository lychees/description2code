#include <bits/stdc++.h>
using namespace std;
#define ll long long
const ll mod=1e9+7;
ll fact[105],ifact[105],n,l,r,ans[105],C[105][105],derange[105];
ll inverse(ll a,ll pow)
{
    ll ret=1;
    while(pow)
    {
        if(pow&1)
        {
            ret*=a;
            ret%=mod;
        }
        a*=a;
        a%=mod;
        pow>>=1;
    }
    return ret;
}
int main()
{
    fact[0]=1;
    for(ll i=1;i<105;i++)
        fact[i]=(i*fact[i-1])%mod;
    ifact[104]=inverse(fact[104],mod-2);
    for(ll i=103;i>=0;i--)
        ifact[i]=((i+1)*ifact[i+1])%mod;
    derange[0] = 1;
    derange[1] = 0;
    for(ll i=2;i<=104;i++)
    {
        derange[i] = ((i-1)*(derange[i-1]+derange[i-2]))%mod;
    }
    for(ll i=0;i<=104;i++)
    {
        for(ll j=0;j<=i;j++)
        {
            if(i == j || j == 0)
                C[i][j] = 1;
            else
                C[i][j] = (C[i-1][j] + C[i-1][j-1])%mod;
        }
    }
    cin>>n>>l>>r;
    ans[0]=1;
    ans[1]=0;
    ll mul=-1;
    ll ret=0,q=0;
    for(ll i=0;i<=n;i++)
    {
        ans[i]=(derange[i]*C[n][i])%mod;
        if(i>=l && i<=r)
            ret=(ret+(ans[i]))%mod;
        q=(q+ans[i])%mod;
        //cout<<ret<<" "<<q<<endl;
    }
    ret=(ret*inverse(q,mod-2))%mod;
    cout<<ret<<endl;
    return 0;
}
