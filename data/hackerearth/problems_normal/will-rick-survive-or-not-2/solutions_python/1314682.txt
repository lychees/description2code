for T in range(input()):
    A = [0] * 50001
    n, a = input(), map(int, raw_input().split())
    for i in a: A[i] += 1
    c = x = i = 0
    while i < n:
        while A[x] == 0: x += 1
        A[x] -= 1
        if c >= x: break
        c += 1
        if i % 6 == 5: c += 1
        i += 1
    if i == n:
        print "Rick now go and save Carl and Judas"
    else:
        print "Goodbye Rick\n", i
