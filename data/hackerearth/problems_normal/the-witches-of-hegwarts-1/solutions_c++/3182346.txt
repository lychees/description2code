using namespace std;

int solve(int n)
{
        map <int,int> mp;
        queue <int> Q;
        Q.push(n);
        int x,r;
        mp[n] = 0;
        while(!Q.empty() && mp.find(1) == mp.end())
        {
                x = Q.front();
                r = mp[x];
                Q.pop();
                if(x % 2 == 0 && mp.find(x >> 1) == mp.end())
                {
                        Q.push(x >> 1);
                        mp[x >> 1] = r + 1;
                }
                if(x % 3 == 0 && mp.find(x / 3) == mp.end())
                {
                        Q.push(x / 3);
                        mp[x / 3] = r + 1;
                }
                if(mp.find(x - 1) == mp.end())
                {
                        Q.push(x - 1);
                        mp[x - 1] = r + 1;
                }
        }
        return mp[1];
}

int main()
{
        int t,n;
        cin >> t;
        while(t--)
        {
                cin >> n;
                cout << solve(n) << endl;
        }
        return 0;
}
