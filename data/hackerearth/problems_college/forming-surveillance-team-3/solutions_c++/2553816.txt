#include<bits/stdc++.h>
using namespace std;
typedef map<int,int> maps;
int main()
{
maps map1;
maps::iterator t;
int T,M,x;
cin>>T;
while(T)
{
    map1.clear();
    cin>>M;
    while(M)
    {
    cin>>x;
    t=map1.find(x);
    if(t==map1.end())
        map1[x]=1;

    M--;
    }
    cout<<map1.size()<<endl;


T--;
}
return 0;
}
