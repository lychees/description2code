#include<iostream>
using namespace std;
#include<bits/stdc++.h>

// n==eggs //k==floors

int eggFloor[501][501]={{0}};

void eggDrop(int n,int k)
 {
 	int i,j,res;
 	for(i=1;i<=n;i++)
 		{eggFloor[i][1]=1;
 		 eggFloor[i][0]=0;	
 		}
 	for(i=1;i<=k;i++)
 	  eggFloor[1][i]=i;
 	
 	for(i=2;i<=n;i++)
 	  {
 	  	for(j=2;j<=k;j++)
 	  	   {
 	  	   	eggFloor[i][j]=INT_MAX;
 	  	   	for(int x=1;x<=j;x++)
 	  	   	   {res=1+max(eggFloor[i-1][x-1],eggFloor[i][j-x]);
 	  	   	   if(res<eggFloor[i][j])
 	  	   	     eggFloor[i][j]=res;
 	  	   	   }
 	  	   }
 	  }
 	
 	
 	
 }
int main()
 {
 	eggDrop(500,500);
 	int t;
 	cin>>t;
 	while(t--)
 	 {int n,i,j,copy,k;
 	 	cin>>n>>k;
 	 	cout<<eggFloor[k][n]<<endl;
 	 }
 }
