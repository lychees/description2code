#include<iostream>
#include<string>
using namespace std;

int main()
{
    unsigned long int no_of_test_case,weight,lenght_of_input;
    int no_of_skipping_letters,temp,flag;
    string input;
    cin>>no_of_test_case;
    while(no_of_test_case--)
    {
        flag=0;
        weight=0;
        cin>>input;
        cin>>no_of_skipping_letters;
        char arr[no_of_skipping_letters];
        for(int i=0;i<no_of_skipping_letters;i++)
        {
            cin>>arr[i];
        }
        lenght_of_input=input.length();
        for(unsigned long int i=0;i<lenght_of_input;i++)
        {
            for(int j=0;j<no_of_skipping_letters;j++)
            {
                if(input[i]==arr[j])
                {
                    flag=1;
                    break;
                }
            }
            if(flag==1)
            {
                flag=0;
            }
            else
            {
                temp=input[i]-96;
                weight=weight+temp;
            }
        }
       cout<<weight<<"\n";
    }
    return 0;
}
