#include <iostream>
#define ll long long
using namespace std;

int main()
{
    ll n = 0;
    ll ctr  = 0, tc = 0;
    cin >> n;
    while (n--){
    	cin >> tc;
    	while (tc != 1){
    		if (tc % 3 == 0){
    			tc = tc / 3;
    			ctr++;
    		}
    		else if (tc % 2 == 0){
    			tc = tc / 2;
    			ctr++;
    		}
    		else{
    			tc = tc - 1;
    			ctr++;
    		}
    		
    	}
    	cout << ctr << endl;
    	ctr = 0;
    }
    return 0;
}
