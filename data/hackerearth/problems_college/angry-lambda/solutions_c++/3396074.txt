#include<bits/stdc++.h>
using namespace std;

long long int gcd(long long int a,long long int b)
{
	if(a==0)
	return b;
	else 
	return gcd(b%a,a);	
}

long long int max(long long int a,long long int b)
{
	return a>b?a:b;
}

int main()
{
	int t;
	cin>>t;
	
	for(;t>0;t--)
	{
		int n,k;
		cin>>n>>k;
		
		int c=0;
		for(int i=0;i<n;i++)
		{
			int x;
			cin>>x;
			if(x<=0)
			c++;
		}
		
		if(c<k)
		cout<<"YES\n";
		else
		cout<<"NO\n";
	}
	return 0;
}
