#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;


int ctr=0,i,a[26];
int main() {
   
    string s;
    cin>>s;
     
    int flag = 1;
    for(i=0;i<s.length();i++)
        {
        a[s[i]-'a']++;
        }
    for(i=0;i<26;i++)
        {
        if(a[i]%2)
            ctr++;
        
        if(ctr>1)
            {
            flag=0;
            break;
            }
    }
    if(flag==0)
        cout<<"NO";
    else
        cout<<"YES";
    return 0;
}
