#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
#define MAX 1000006
int lp[MAX];
int sig[MAX];
void sieve()
{
	int i,j;
	for(i=2;i<MAX;++i)
	{
		if(lp[i] == 0)
		{
			for(j=i;j<MAX;j+=i)
				lp[j] = i;
		}
	}
	sig[1] = 1;
	for(i=2;i<MAX;++i)
	{
		int cnt = 2;
		int d = lp[i];
		int tmp = i/d;
		while(tmp%d == 0)
		{
			tmp/=d;
			cnt++;
		}
		sig[i] = sig[tmp] * cnt;
	}
	//for(i=1;i<=10;++i)
	//	cout<<sig[i]<<" ";
	//cout<<endl;
}
ll arr[MAX];
int main()
{
	int t;
	int i;
	sieve();
	arr[1] = 0;
	arr[2] = 1;
	for(i=3;i<MAX;++i)
	{
		arr[i] = (arr[i-1]+sig[i]-1)%MOD;
	}
	ios_base::sync_with_stdio(false);
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		
		cout<<arr[n]<<endl;
		
	}
    return 0;
}
