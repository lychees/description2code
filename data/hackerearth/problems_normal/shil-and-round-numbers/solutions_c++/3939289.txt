#include<bits/stdc++.h>
using namespace std;
#define MAX 200010
#define ll long long
ll bit[MAX];
int n;
void update(int ind,int val){
	while(ind<=n){
		bit[ind]+=val;
		ind+=(ind)&(-ind);
	}
}

int query(int r){
	int sum=0;
	int i=r;
	while(i>0){
		sum+=bit[i];
		i-=i&(-i);
	}
	return sum;
}

bool round(string val){
	if(val[0]=='-')return false;
	if(val[0]==val[val.size()-1]){
		return true;
	}
	else return false;
}

int main(){
	string arr[MAX];
	int q;
	cin>>n>>q;
	for(int i=1;i<=n;i++){
		cin>>arr[i];
		if(round(arr[i])){
			update(i,1);
		}
	}
	while(q--){
		int t;
		cin>>t;
		if(t==1){
			int l,r;
			cin>>l>>r;
			cout<<query(r)-query(l-1)<<endl;
		}
		else{
			int i;
			string k;
			cin>>i;
			cin>>k;
			if(round(arr[i])){
				update(i,-1);
			}
			arr[i]=k;
			if(round(k)){
				update(i,1);
			}
		}

	}
	return 0;
}
