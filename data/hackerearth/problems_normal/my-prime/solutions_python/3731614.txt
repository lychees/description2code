def get_prime(length_of_set, num_set):
    """Return all primes in set"""
    num_set = num_set.split(' ')
    num_set = map(int, num_set)
    prime_list = [] #list to store all prime numbers
    for num in num_set:
        for all_num in num_set:
            if num % all_num == 0 and num != all_num:
                prime_list.append(num)
                break
    prime_list = [str(num) for num in num_set if num not in prime_list]
    return prime_list
