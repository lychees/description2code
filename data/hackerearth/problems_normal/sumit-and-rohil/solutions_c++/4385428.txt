#include<iostream>
#include<string>
#include<vector>
#include<map>

using namespace std;

bool IsAnagram(string s1, string s2)
{
	int len1 = 0;
	int len2 = 0;
	map<char, int> str1Map;
	map<char, int> str2Map;



	for (auto ch : s1)
	{
		str1Map[ch] += 1;
		len1++;
	}
	for (auto ch : s2)
	{
		str2Map[ch] += 1;
		len2++;
	}

	if ( (len1 != len2) || (s1[len1-1] != s2[len2-1]) )
		return false;

	if ((str1Map.size() == str2Map.size()) &&  equal(str1Map.begin(), str1Map.end(), str2Map.begin()))
		return true;
	
	return false;

}

int main()
{
	int n;
	string input;

	cin >> n;
	vector<string> group;
	bool alreadyAdded = false;
	for (int i = 0; i < n; i++)
	{
		cin >> input;

		alreadyAdded = false;
		for (auto item : group)
		{
			if ( (item[0] == input[0]) && IsAnagram(item, input) )
			{
				alreadyAdded = true;
				break;
			}
		}

		if (alreadyAdded == false)
			group.push_back(input);
	}
	
	
	cout << group.size();
	return 0;
}
