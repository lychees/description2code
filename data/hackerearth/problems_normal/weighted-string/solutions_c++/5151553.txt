#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main(){
    int tc;cin>>tc;
    while(tc--){
        int k;cin>>k;
        string s;cin>>s;
        vector<int>pref(s.size());
        pref[0]=s[0]-'a'+1;
        for(int i=1;i<s.size();++i){
            pref[i]+=pref[i-1];
            pref[i]+=s[i]-'a'+1;
        }
        ll ans=0;
        for(int i=0;i<s.size();++i){
            for(int j=i;j<s.size();++j){
                if(i-1>=0 && pref[j]-pref[i-1]==k)ans++;
                else if(i-1<0 && pref[j]==k)ans++;
                else if(i-1>=0 && pref[j]-pref[i-1]>k)break;
                else if(i-1<0 && pref[j]>k)break;
            }
        }
        cout<<ans<<"\n";
    }
    return 0;
}
