#include <iostream>
#include <cstring>

using namespace std;

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
    int t,l,j,c,i,ch[25];
    char p[25],s[100000];
    cin >> t;
    while (t--)
    {
    	for (i = 0; i < 26; i++)
    	  ch[i] = 0;
    	cin >> p;
    	cin >> s;
    	l = strlen(s);
    	for (i = 0; i < l; i++) ch[s[i] - 97]++;
    	for (i = 0; i < 26; i++)
    	{
    		c = ch[p[i] - 97];
    	 	for (j = 0; j < c; j++)
    	 	 cout << p[i];
    	}
    	cout << "\n";
    }
    return 0;
}
