#include<stdio.h>
#include <string.h>

int main(){
	int n, i, j, flaga = 0, flage = 0, flagi = 0, flago = 0, flagu = 0;
	scanf("%d", &n);
	char a[n];
	scanf("%s", a);
	
	
	for(i = 0; i < strlen(a); i++){
		if(a[i] == 'a')	flaga = 1;
			else if(a[i] == 'e')	flage = 1;
				else if(a[i] == 'i')	flagi = 1;	
					else if(a[i] == 'o')	flago = 1;
						else if(a[i] == 'u')	flagu = 1;
		
		if(flaga == 1 && flage == 1 && flagi == 1 && flago == 1 && flagu == 1){
			printf("YES\n");
			return 0;
		}	
	}
	printf("NO\n");
	
	return 0;
}
