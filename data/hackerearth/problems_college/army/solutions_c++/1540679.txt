#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
bool present(ll p,ll a[], ll m)
{
	while(p)
	{
		ll c=p%10;
		for(ll i=0;i<m;i++)
		{
			if(a[i]==c)
			return true;
		}
		p=p/10;
	}
	return false;
}
int main()
{
 ll n;
 cin>>n;
 ll q;
 cin>>q;
 while(q--)
 {
 	ll m;
 	cin>>m;
 	ll a[m];
 	for(ll i=0;i<m;i++)
 	cin>>a[i];
 	ll count=0;
 	for(ll i=1;i<=n;i++)
 	{
 		ll p=i; 
 		if(present(p,a, m))
 		count++;
 	}
 	cout<<count<<endl;
 	

 }
 return 0;
}
