#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long a, b, n;
    cin>>a>>b>>n;
    long long counter = 0, days = 0;
    while(1){
        days = days + a;// - b;
        counter++;
        if(days>=n)
            break;
        days = days - b;
    }
    cout<<counter<<endl;
}
