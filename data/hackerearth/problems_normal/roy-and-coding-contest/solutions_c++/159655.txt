#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        double n,m;
        scanf("%lf%lf",&n,&m);
        if(n == 1)
        {
            printf("0\n");
            continue;
        }
        int ans = 0;
        double a1 = log2(m);
        int a11 = ceil(a1);
        a11++;
        int  temp = pow(2,a11-1);
        if(temp >= n)
        {
            ans = ceil(log2(n)) + 1;
        }
        else
        {
            ans += (a11);
            n -= temp;
            //cout<<n<<endl;
            //cout<<ans<<endl;
            ans += ceil(n/m*1.0);
        }
       printf("%d\n",ans);
    }
    return 0;
}
