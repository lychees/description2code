#include <iostream>
#include <cstring>
#include <cstdio>
#include <iomanip>
using namespace std;

int main()
{
	
	int n,i;
	scanf("%d",&n);
	double arr[n],brr[n];
	
    for(i=0;i<n;i++)
    	cin>>arr[i];
    
    for(i=0;i<n;i++)
       cin>>brr[i];
       
       double man;
       cin>>man;
       
       double result=0;
       
       for(i=0;i<n;i++)
       {
       double	yax=man/brr[i];
       	      yax=yax*arr[i];
       	
       	if(yax>result)
       		result=yax;
       	
       }
       
       long long zz=result*1000;
       
       result=zz*1.0/1000;
       
       cout<<std::fixed<<std::setprecision(3)<<result<<endl;
       
    return 0;
}
