#include<bits/stdc++.h>
using namespace std;
int v[1005];
int A[1005][1005];
int n,m;
void dfs(int in,int p)
{
    v[in]=1;
    for(int i=1;i<=n;i++)
    {
        if(A[in][i]==-1) continue;
        if(v[i]==0 && A[in][i]<p)
            dfs(i,p);
    }
}
int gcd(int a,int b)
{
    if(b==0) return a;
    return gcd(b,a%b);
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int x,y,pp,z;
        memset(A,-1,sizeof(A));
        cin>>n>>m;
        for(int ii=0;ii<m;ii++)
        {
            scanf("%d%d%d",&x,&y,&pp);
            A[x][y]=A[y][x]=pp;
        }
        scanf("%d",&z);
        int ans=0;
        for(int ii=0;ii<z;ii++)
        {
            scanf("%d%d",&x,&y);
            memset(v,0,sizeof(v));
            dfs(x,A[x][y]);
            if(v[y]==0) ans++;
        }
        int gc=gcd(z,ans);
        cout<<(ans/gc)<<"/"<<(z/gc)<<"\n";
    }
    return 0;
}
