#include<iostream>
using namespace std;
 
int main(void){
	long long t;
	cin>>t;
	while(t--){
		long long a, b, count = 0;
		cin>>a>>b;
		while(a!=0 && b!=0){
			int sum = (a%10)+(b%10);
			if(sum>9)
				count++;
			a /= 10;
			b /= 10;
		}
		if(count == 0){
			cout<<"\nNo carry operation";
		}
		else if(count == 1){
			cout<<"\n1 carry operation";
		}
		else{
			cout<<endl<<count<<" carry operations";
		}
	}
}
