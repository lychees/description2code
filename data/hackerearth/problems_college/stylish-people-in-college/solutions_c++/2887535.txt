#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll a[100005],sum[100005];
int main()
{
   // cout << "Hello World!" << endl;
   int t;
   scanf("%d",&t);
   int c;
   for(c=1;c<=t;c++)
   {
   	int n;
   	scanf("%d",&n);
   	int i;
   	for(i=1;i<=n;i++)
   	scanf("%lld",&a[i]);
   	sum[1] = a[1];
   	for(i=2;i<=n;i++)
   	sum[i] = max(sum[i-1],a[i]+sum[i-2]);
   	printf("Case %d: %lld\n",c,sum[n]);
   }
    return 0;
}
