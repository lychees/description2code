#include <iostream>
using namespace std;
int binarySearch(int *a,int n,int k)
 {int lp=0,up=n-1,mp;
  while(lp<=up)
   {mp=(lp+up)/2;
    if(a[mp]==k)
        return mp;
    else if(a[mp]<k)
        lp=mp+1;
    else
        up=mp-1;
   }
  return -1;
 }
main()
 {int i,k,m,n,*a;
  cin>>n;
  a=new int[n];
  for(i=0;i<n;i++)
   {cin>>a[i];
   }
  cin>>m;
  for(i=0;i<m;i++)
   {cin>>k;
    cout<<"\n"<<binarySearch(a,n,k);
   }
  delete a;
 }
