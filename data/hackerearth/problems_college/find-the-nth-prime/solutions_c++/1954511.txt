#include <iostream>
#include<cstdio>
#include<cstring>

using namespace std;
int prime[1005];
bool status[10000];
void gen()
{
	memset(prime,0,sizeof prime);
	memset(status,0,sizeof status);
	for(int i=3;i<=100;i+=2)
	{
		if(status[i]==0)
		{
			for(int j=i*i;j<=10000;j+=i+i)
			status[j]=1;
		}
	}
	prime[0]=2;
	int k=1;
	for(int i=3;i<=10000;i+=2)
	if(!status[i])
	prime[k++]=i;
}
int main()
{
    gen();
    int n;
    scanf("%d",&n);
    cout<<prime[n-1];
    return 0;
}
