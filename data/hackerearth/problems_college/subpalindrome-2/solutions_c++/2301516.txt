#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll arr[100][100];
ll subcount(string s,int i,int j){
	if(i>j){
		return 0;
	}
	if(i==j)
		return 1;
	if(arr[i][j]!=-1)
		return arr[i][j];
	ll ans;
	ans=subcount(s,i,j-1)+subcount(s,i+1,j)-subcount(s,i+1,j-1);
	if(s[i]==s[j]){
		ans+=1+subcount(s,i+1,j-1);
	}
	arr[i][j]=ans;
	return arr[i][j];
}
int main(){
	int t;
	cin>>t;
	while(t--){
		memset(arr,-1,sizeof(arr));
		string str;
		cin>>str;
		cout<<subcount(str,0,str.length()-1)<<endl;
	}
}
