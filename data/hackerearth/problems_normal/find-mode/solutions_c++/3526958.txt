#include <iostream>
#include<string>
#include<stdio.h>
#include<vector>
#include<algorithm>

using namespace std;
//int c[26]; int odd;
int a[1000000];
vector<long long int>v,w;
vector<long long int>::iterator it;
long long int n;
void initialise()
{
    for (long long int i=0;i<1000000;i++)
        a[i]=0;
}
long long int max_freq()
{	
    long long int i=0; long long int temp=0, index=0;
   // cout <<"Printing freq array";
    //for(i=0;i<n;i++)
    //cout <<a[i]<<'\n';
    for(i=0;i<1000000;i++)
    {
        if(a[i]>temp)
            {
                temp=a[i];
                index=i;
            }
    }
    return temp;
}

int main()
{
int t;
long long int b,cnt;
cin >>t;
while(t--)
{	w.clear();
	v.clear();
    initialise();
    cin >>n;
    cnt=n;
    while(cnt--)
    {
        cin >>b;
        v.push_back(b);
        a[b]++; //a[b] denotes how many times b has occured
    }
    b=max_freq();
  // cout <<"Max freq is "<<b<<'\n';
    for(long long int y=0;y<n;y++)
    {	//cout <<"element being processed is "<<v[y]<<" and the freq is "<<a[v[y]]<<'\n';
    	if(a[v[y]]==b) 
    	{
    		w.push_back(v[y]); 
    	//	cout <<"Element being pushed is "<<v[y]<<'\n';
    		
    	}
    }
    sort(w.begin(),w.end());
   // it=unique(w.begin(),w.end());
    for(long long int z=w.size()-1;z>=0;z--)
    {
    //	cout <<w[z]<<' ';
    	if(z>=1)
    	{
    		if(w[z]!=w[z-1])
    		cout <<w[z]<<' ';
    	}
    	else cout <<w[z]<<' ';
    }
    cout <<'\n';
    //cout <<b<<'\n';
    //for(long long b=0;b<n;b++)
      //  cout <<a[b]<<'\n';

}
    return 0;
}
