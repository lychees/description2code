#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	string str;
    	cin >> str;
    	int l = str.size();
    	int h = l / 2;
    	int sum1 = 0, sum2 = 0;
    	for(int i = 0; i < h; i++) sum1 += (str[i] - '0');
    	for(int i = h; i < l; i++) sum2 += (str[i] - '0');
    	if(sum1 == sum2) {
    		puts("YES");
    	}
    	else {
    		puts("NO");
    	}
    }
    return 0;
}
