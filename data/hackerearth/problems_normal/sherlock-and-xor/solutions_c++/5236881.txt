#include <iostream>
#include <stdio.h>

using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    for(int tt=1;tt<=t;++tt)
    {
        int n;
        long long odd=0,even=0;
        scanf("%d",&n);
        while(n--)
        {
            int a;
            scanf("%d",&a);
            if(a%2==0)even++;
            else odd++;
        }
        long long k=odd*even;
        printf("%lli\n",k);
    }
}
