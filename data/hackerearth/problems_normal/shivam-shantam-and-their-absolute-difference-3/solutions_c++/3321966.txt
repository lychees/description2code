#include <bits/stdc++.h>
using namespace std;

int mod = (1e9)+7 ;
int main()
{
    int n ,p=0;cin>>n;int base = n*2000 ;int result = 0 ;
    int maxwidth = n*2000*2 ;
    vector<int> arr(2*n);
    vector<int> sum(maxwidth,0);sum[n*2000]=1 ;
    vector<int> tsum(maxwidth,0);
    for(;p<n;p++)cin>>arr[p];
    for(;p<2*n;p++){cin>>arr[p];arr[p]=-arr[p];}
    for(int i=0;i<2*n;i++){
    	for(int k=0;k<maxwidth;k++){
    		if(sum[k]>0){
    			tsum[k+arr[i]] = (tsum[k+arr[i]] + sum[k])%mod;
    		}
    	}
    	for(int j=0;j<maxwidth;j++)if(tsum[j]>0){sum[j] = (sum[j] +tsum[j])%mod;tsum[j]=0;}
    }
    int q ;cin>>q;
    for(int i=base-q;i<=base+q;i++)result = (result + sum[i])%mod ;
    cout<<result<<endl;
    return 0;
}
