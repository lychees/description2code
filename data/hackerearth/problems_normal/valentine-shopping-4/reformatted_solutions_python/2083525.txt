for _ in xrange(int(raw_input())):
    N, M = map(int, raw_input().strip().split())
    ans = []
    for n in xrange(N):
        best = (0, 2)
        for m in xrange(M):
            temp = reduce(lambda x, y: x * (1-(float(y)/100)),
                          [1]+raw_input().strip().split())
            if best[1] > temp or (best[1] == temp and best[0] > m+1):
                best = m, temp
        ans.append(str(best[0]+1))
    print " ".join(ans)
