#include <iostream>
#include <stdio.h>
#include <stdbool.h>
using namespace std;
#define lli long long int
#define till(n) for(int i = 0 ; i <= n ; i++)
#define foran(a,n) for(int i = a ; i <= n ; i++)

int main() {

  lli n, max = 0;
  scanf("%lld",&n);

  lli a[n+5];
  pair<int,int> maxf[n+5], maxb[n+5];
  bool hash[100000] = {0};
  for (int i = 0; i < n ; i++) {
      scanf("%lld",&a[i]);
  }

  maxf[1] = make_pair(a[0], 1) ;
  foran(2, n){
    if(maxf[i-1].first < a[i-1])
    {
        maxf[i] = make_pair(a[i-1], i);
    }
    else
        maxf[i] = maxf[i-1];
  }

  maxb[1] = make_pair(a[n-1], n);
  foran(2, n){
      if(maxb[i-1].first < a[n-i])
          maxb[i] = make_pair(a[n-i], n-i+1);
      else
          maxb[i] = maxb[i-1];
  }

  foran(1, n){

      int a = maxf[i].second ;
      int b = maxb[i].second ;
      hash[a] = 1;
      hash[b] = 1;
  }

  till(n){
      if(hash[i])
        printf("%d ",i);
  }

  printf("\n");
  return 0;
}
