#include<iostream>
using namespace std;
#define MOD 1000000007
#define ll long long int
ll gcd(ll a, ll b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);	
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	ll a, b, c, g, r;
    	cin>>a>>b>>c;
    	g=gcd(gcd(a,b),c);
    	a/=g;
    	b/=g;
    	c/=g;
    	r=(((a*b)%MOD)*c)%MOD;
    	cout<<g<<" "<<r<<endl;
    }
return 0;
}
