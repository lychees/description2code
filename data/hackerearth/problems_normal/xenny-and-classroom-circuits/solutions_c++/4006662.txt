#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t, n;
    int c1,c2;
    char s[2000005];
    cin>>t;
    while(t--){
    	cin>>n;
    	for(int i=0;i<2*n;i++){
    		cin>>s[i];
    	}
    	c1=c2=0;
    	for(int i=0;i<2*n;i++){
    		if(i%2==0 and s[i]=='B'){
    			c1++;
    		}
    		else if(i%2 !=0 and s[i]=='B'){
    			c2++;
    		}
    	}
    	cout<<(c1>c2?c2:c1)<<endl;
    	
    }
    return 0;
}
