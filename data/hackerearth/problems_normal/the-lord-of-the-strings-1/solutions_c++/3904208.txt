#include <iostream>

using namespace std;

int main()
{
	int t;
	string testcase[100];
	string input,output;
	int alpha[26]={0};
	
	cin >> t;
	cin.clear(); cin.ignore(10,'\n');
	
	for(int i=0;i<t+1;++i){
		cin >> testcase[i];
	}
	
	
	
	
	for(int j=0;j<t+1;++j){
		
		for(int k=0;k<26;++k){ alpha[k]=0;}
		
		input = testcase[j];
		output="";
		for(int i=0;i<input.length();++i){
			if(alpha[ input[i] -97 ] == 0){
				output+=input[i];
				alpha[ input[i] -97 ] = 1;
			}
		}
		
		cout << output << endl;
	}
    return 0;
}
