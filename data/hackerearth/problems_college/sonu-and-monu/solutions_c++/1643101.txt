#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

ll gcd(ll a, ll b)
{
while (b^=a^=b^=a%=b);
return a;
}

int main()
{
ios_base::sync_with_stdio(false);
ll t;
cin >> t;
while (t--)
{
ll n, p; 
cin >> n >> p;
string s;
cin >> s;
vector<ll> ones; 
for (int i = 0; i < n; i++)
if (s[i] == '1') 
ones.push_back(i);
ll l = ones.size();
if (l)
{
ll ans = l;
ll choices = (n * (n-1)) + n;
for (int i = 0; i < l-1; i++)
{
ll cnt = l-i;
for (int j = i+1; j < l; j++)
{
if (ones[j] - ones[i] <= p)
ans += 2;
else 
break;
}
}
ll g = gcd(ans, choices);
cout << ans/g << "/" << choices/g << "\n";
}
else
cout << "0/1" << "\n";
} 
return 0;
}
