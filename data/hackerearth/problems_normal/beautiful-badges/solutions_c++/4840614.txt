#include <bits/stdc++.h>
using namespace std;

int main()
{
	vector<int>blue,yellow;
    int n,b=0,y=0,ans=0,j=0;
    cin>>n;
    int color,value;
    for(int i=0;i<n;i++)
    {
    	cin>>color>>value;
    	if(color==1) blue.push_back(value);
    	else  yellow.push_back(value);
    }
    b = blue.size();
    y = yellow.size();

    //cout<<"blue "<<b<<" yellow "<<y<<"\n";
	sort(blue.begin() , blue.end());
	sort(yellow.begin() , yellow.end());

    for(int i=0;i<y;i++)
    {

    	while(j<b && blue[j]*sqrt(2)<yellow[i]) j++;

    	if(j<b && blue[j]<=yellow[i]*sqrt(2))
        {
            ans++;
            j++;
        }
    }
    cout<<ans;

    return 0;
}
