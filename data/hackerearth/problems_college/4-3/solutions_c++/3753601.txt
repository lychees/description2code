#include<stdio.h>
#include<math.h>
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,i;
		scanf("%d",&n);
		long long j=1,sum=0;
		int a[n];
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		for(i=0;i<n-1;i++)
		{
			j=(j*2)%1000000007;
		}
		for(i=0;i<n;i++)
		{
			sum=(sum+(a[i]*j))%1000000007;
		}
		printf("%lld\n",sum);
	}
	return 0;
}
