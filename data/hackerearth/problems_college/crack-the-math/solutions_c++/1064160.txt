#include <iostream>
#include <string.h>
#include <cstdio>
using namespace std;

int main()
{
   unsigned long long T;
   cin>>T;
   while(T--)
   {
   	char str1[100],str2[100];
   scanf("%s%s",str1,str2);
   	int i=0;
   	char str3[30];
   	while(str1[i] && str2[i])
   	{
   	   int d=str1[i]-str2[i];
   	   if(d==0)
   	   {
   	   	str3[i]='0';
   	   }
   	   else
   	   {
   	   	str3[i]='1';
   	   }
   	   i++;
   	}
   	str3[i]='\0';
   	cout<<str3<<endl;

   }
   return 0;
}
