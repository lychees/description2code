#include <iostream>
#include <map>
using namespace std;

int main() {
	int n;
	cin >> n;
	int lines = (1 << n) - 1;
	map<string, bool> match;
	string a, b;
	for (int i = 0; i < lines; i++) {
		cin >> a >> b;
		map<string, bool>::const_iterator search = match.find(a);
		if (!(search != match.end())) {
			match[a] = true;
		}
		match[b] = false;
	}
	string winner;
	map<string, bool>::const_iterator it;
	for (it = match.begin(); it != match.end(); ++it) {
		if (it->second) {
			winner = it->first;
			break;
		}
	}
	cout << winner;
	return 0;
}
