#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define PII pair<ll, ll>
#define f first
#define s second
#define F(i,a,b) for(ll i = (ll)(a); i <= (ll)(b); i++)
#define RF(i,a,b) for(ll i = (ll)(a); i >= (ll)(b); i--)
#define inf LLONG_MAX
#define mod 1000000007
#define MAXN 100005
#define pb(x) push_back(x)

ll t, n;
ll arr[35][26];
ll fin[26];
string st, req;

bool isPossible(ll N, ll rem[])
{
	bool flag=1, ans;
	F(i,0,25)
		if(rem[i]!=0)
			flag=0;
	if(flag)
		return true;
	if(N==0 && !flag)
		return false;
	bool f=1;
	F(i,0,25)
	{
		if(arr[N][i]>rem[i])
			f=0;
	}
	if(!f)
	{
		//cout<<N<<endl;
		ans = isPossible(N-1, rem);
		return ans;
	}
	ll tmp[26];
	F(i,0,25)
		tmp[i]=rem[i]-arr[N][i];
	ans = ans | isPossible(N-1, tmp);
	ans = ans | isPossible(N-1, rem);
	return ans;
}

int main() 
{
	ios_base::sync_with_stdio(false);cin.tie(0);
	cin>>t;
	
	while(t--)
	{
		cin>>n;
		memset(arr,0,sizeof(arr));
		memset(fin,0,sizeof(fin));
		F(i,1,n)
		{
			cin>>st;
			ll len=st.length();
			F(j,0,len-1)
				arr[i][st[j]-'a']++;
		}
		cin>>req;
		ll len=req.length();
		F(i,0,len-1)
			fin[req[i]-'a']++;	
		if(isPossible(n, fin))
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
	return 0;
}
