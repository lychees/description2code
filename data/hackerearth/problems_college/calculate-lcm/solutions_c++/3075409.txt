#include<iostream>
#include<math.h>
using namespace std;
long long mod=pow(2,30);
long long gcd(long long a, long long b)
{
    if(b==0)return a;
    else return gcd(b,a%b);
}
int main()
{
    int t,d;
    long long hcf,i,s,s2,j,a,b,ans;
    cin>>t;
    while(t--)
    {
        ans=0;
        cin>>a>>b;
        for(i=1;i<=a;i++)
        {
            for(j=1;j<=b;j++)
            {
                d=0;
                hcf=gcd(i,j);
                for(s=2;;s++)
                {
                    s2=s*s;
                    if(hcf/s2)
                    {
                        if(hcf%s2==0)
                        {
                            d=1;
                            break;
                        }
                    }
                    else
                        break;
                }
                if(d==1)
                    continue;
              ans=(ans+(((i/hcf)*j)%mod))%mod;
            }
        }
        cout<<ans<<endl;
    }
}
