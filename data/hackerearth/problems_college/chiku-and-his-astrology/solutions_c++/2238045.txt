#include <iostream>
#include <cmath>
using namespace std;

char str[200002];
int main()
{
    int t;
    cin>>t;
    while(t--){
    	cin>>str;
    	long i=0,one=0,zero=0;
    	while(str[i]!='\0'){
    		if(str[i]=='1')
    			one++;
    		else
    			zero++;
    		i++;
    	}
    	cout<<abs(one-zero)<<endl;
    }
    return 0;
}
