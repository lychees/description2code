#include<stdio.h>
#include<iostream>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<stdlib.h>
#include<vector>
#include<map>
#include<set>

using namespace std;

main()
{
	long int n,a[100000],i,ct=1;
	cin>>n;
	
	for(i=0;i<n;i++)
	cin>>a[i];
	
	sort(a,a+n);
	long int range=a[0]+4;
	for(i=0;i<n;i++)
	{
		if(a[i]>range)
		{
			range=a[i]+4;
			ct++;
		}
	}
	cout<<ct;
}
