#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    long long int i,n;
    scanf("%lld",&n);
    long long arr[n],sum=0;
    for(i=0;i<n;i++){
        scanf("%lld",&arr[i]);
    }
    sort(arr,arr+n);
    for(i=0;i<n;i=i+2){
        sum += arr[i]*arr[i+1];
    }
    printf("%lld\n",sum);


}
return 0;
}
