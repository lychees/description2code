#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		char str[100001];
		scanf("%s", str);
		int cnt=0;
		int n=strlen(str);
		for(int i=0; i<n; i++)
		{
			if(str[i]!='-')
				continue;
			if(i<n-1 && (str[i+1]=='B' || str[i+1]=='S'))
				continue;
			if(i<n-2 && str[i+2]=='B')
				continue;
			if(i>0 && str[i-1]=='S')
				continue;
			cnt++;
		}
		printf("%d\n", cnt);
	}
	return 0;
}
