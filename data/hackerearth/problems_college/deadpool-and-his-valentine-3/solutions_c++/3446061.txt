#include <iostream>
using namespace std;

int main()
{
    long long int k,m,i,p,x;

    cin>>k>>m;
    bool flag,a[105]={false};
    while(k--)
    {
    	cin>>x;
    	for(i=0;i<x;i++)
    	{
    		cin>>p;
    		a[p]=true;
    	}
    }
    flag=0;
    for(i=1;i<=m;i++)
    {
    	if(!a[i])
    	{
    		flag=1;
    	}
    }
    if(flag)
    {
    	printf("NO\n");
    }
    else
    {
    	printf("YES\n");
    }
    return 0;
}
