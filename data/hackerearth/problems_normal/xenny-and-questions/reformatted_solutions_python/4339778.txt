    x = map(int, raw_input().split())
    n = x[0]
    k = x[1]
    arr = []
    sum1 = 0
    for j in range(n):
        arr.append(int(raw_input()))
    if k >= n:
        print sum(arr)
        continue
    if k == 0:
        print 0
        continue

    arr = sorted(arr)
    for j in range(k):
        sum1 += arr[n-j-1]
    print sum1
