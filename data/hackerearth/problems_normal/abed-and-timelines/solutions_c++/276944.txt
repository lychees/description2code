#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc!=0)
    {
    	int n,m ;
    	cin>>n>>m;
    	int x1,y1,x2,y2;
    	cin>>x1>>y1>>x2>>y2;
    	x1++;y1++;x2++;y2++; 
    	int p;
    	cin>>p;
    	
    	int moves=0;
    	
    	if(abs(x1-x2)>abs(n-abs(x1-x2)))  moves+=abs(n-abs(x1-x2));
    	else moves+=abs(x1-x2);
    	
    	if(abs(y1-y2)>abs(m-abs(y1-y2)))  moves+=abs(m-abs(y1-y2));
    	else moves+=abs(y1-y2);
    	
    	cout<<p*moves<<endl;
    	
    	if(p*moves<=1000) {cout<<"You saved the group."<<endl;}
    	else {cout<<"Let go of the group."<<endl;}  
    	
    	tc--;
    }
    return 0;
}
