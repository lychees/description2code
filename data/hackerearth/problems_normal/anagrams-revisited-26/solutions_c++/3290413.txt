#include <iostream>
#include <algorithm>
#include <string>
#include <vector>
using namespace std;
vector<string> V;
int main(){
        int N,ct = 1,maxx =0;
        cin>>N;
        V.resize(N);
        for(int i = 0;i<N;i++){
            cin>>V[i];
            sort(V[i].begin(),V[i].end());
        }

        sort(V.begin(),V.end());

        for(int i = 1;i<V.size();i++){
            if(V[i-1]==V[i])
                ct++;
            else
            {
                maxx = max(maxx,ct);
                ct=1;
            }

        }
       maxx = max(maxx,ct);
       cout<<maxx;
       return 0;
}
