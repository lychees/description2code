#include <iostream>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	int ct=0;
    	string s1,s2;
    	cin>>s1>>s2;
    	int l1=s1.length();
    	int l2=s2.length();
    	if(l1<l2)
    	{
    		for(int i=0;i<l1;i++)
    			if(s1[i]==s2[i])
    				ct++;
    	}
    	else
    	{
    		for(int i=0;i<l2;i++)
    			if(s1[i]==s2[i])
    				ct++;
    	}
    	cout<<ct<<endl;
    }
    return 0;
}
