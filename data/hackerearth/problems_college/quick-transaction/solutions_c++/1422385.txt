#include <iostream>
using namespace std;

int main()
{
    int t,count=0;
    long long int x;
    cin>>t;
    while(t--)
    {
    	count=0;
    	cin>>x;
    	while(x>=1000)
    	{
    		count+=1;
    		x=x-1000;
    	}
    	while(x>=500)
    	{
    		count+=1;
    		x=x-500;
    	}
    	while(x>=100)
    	{
    		count+=1;
    		x=x-100;
    	}
    	while(x>=50)
    	{
    		count+=1;
    		x=x-50;
    	}
    	while(x>=20)
    	{
    		count+=1;
    		x=x-20;
    	}
    	while(x>=10)
    	{
    		count+=1;
    		x=x-10;
    	}
    	while(x>=5)
    	{
    		count+=1;
    		x=x-5;
    	}
    	while(x>=2)
    	{
    		count+=1;
    		x=x-2;
    	}
    	while(x>=1)
    	{
    		count+=1;
    		x=x-1;
    	}
    	cout<<count<<endl;
    }
    
    
    return 0;
}
