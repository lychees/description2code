#include<bits/stdc++.h>
using namespace std;

int mark[1000002],indegree[1000002];
std::vector<int>arr[1000002];

struct compare
{
    bool operator()(const int l,const int r)
    {
        return l>r;
    }
};

std::priority_queue<int,vector<int> ,compare>pq;

int n,k,x,temp,y,i,j;

int main()
{
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&k);
        scanf("%d",&y);
        mark[y]=1;
        for(j=1;j<k;j++)
        {
            scanf("%d",&x);
            arr[y].push_back(x);
            indegree[x]++;
            y=x;
            mark[x]=1;
        }
    }

    for(i=0;i<=1000000;i++)
    {
        if(mark[i] && indegree[i]==0)
            pq.push(i);
    }

    while(!pq.empty())
    {
        temp=pq.top();
        pq.pop();
        printf("%d ",temp);
        for(i=0;i<arr[temp].size();i++)
        {
            if(--indegree[arr[temp][i]]==0)
                pq.push(arr[temp][i]);
        }
    }

return 0;
}
