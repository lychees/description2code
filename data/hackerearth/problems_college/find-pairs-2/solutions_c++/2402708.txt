#include <iostream>
using namespace std;
#include<map>
int main()
{
	int t; cin>>t;
   	while(t--)
	{
		long long n; cin>>n;
		long long a[n];
		long long ans=0;
		for(int i=0;i<n;i++) cin>>a[i];
		for(int i=0;i<n;i++)
		{
			for(int j=i+1;j<n;j++)
			if(a[i]==a[j]) ans++;
		}
		cout<<ans+n<<endl;
	}
	return 0;
}
