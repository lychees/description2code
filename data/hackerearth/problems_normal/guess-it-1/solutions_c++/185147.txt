#include<iostream>
#include<algorithm>
#include<cmath>

using namespace std;

int main()
{
	int tx,tc,x,sum,t;
	cin>>tx;
	string s;
	for(tc=0;tc<tx;++tc)
	{
		cin>>s;
		sum=0;
		for(x=0;x<s.length();++x)
		{
			if(s[x]!='_')
			{
				sum+=s[x]-'0';
			}
			else t=x;
		}
		sum%=9;
		sum=9-sum;
		if(sum==9) cout<<"0 or 9\n";
		else cout<<sum<<endl;
	}
	return 0;
}
