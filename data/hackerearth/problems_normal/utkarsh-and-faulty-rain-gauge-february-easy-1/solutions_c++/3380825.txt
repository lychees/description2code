#include <iostream>
#include<iomanip>
using namespace std;

double func(double n, double r)
{
	return (3*n*(n+1)) / (2*r*(2*n+1));
}




int main()
{
	int t;
	double res = 1.0, n, r;
	cin>>t;
	while(t--)
	{
		cin>>n>>r;
		res *= func(n, r);
	}
	cout<<setprecision(4)<<fixed<<res;
    
    return 0;
}
