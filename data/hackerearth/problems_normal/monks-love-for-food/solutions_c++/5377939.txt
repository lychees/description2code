#include <bits/stdc++.h>

using namespace std;

int Q ;
int main()
{
    long long t, temp;
    stack <long long> S;
    cin >> Q;
    for(int i = 0; i < Q; i++)
    {
        cin >> t;
        if(t == 1)
        {
            if(S.empty())
                cout << "No Food" << endl;
            else
            {
                cout << S.top() << endl;
                S.pop();
            }
        }
        else if(t == 2)
        {
            cin >> temp;
            S.push(temp);
        }
    }
}
