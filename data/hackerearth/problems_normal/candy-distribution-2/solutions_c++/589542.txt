#include <iostream>
#include<cstdio>
using namespace std;
#define ll long long int
int main()
{
    int t;
    ll n;
    scanf("%d",&t);
    while(t--){
    	ll sum=0;
    	scanf("%llu",&n);
    	for(ll i=0;i<n;i++){
    		ll num;
    		scanf("%llu",&num);
    		sum=sum+num;
    	}
    	printf("%llu\n",sum-(n-1));
    }
    return 0;
}
