#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
        long long num,div=0,i=1;
        cin>>num;
        
        if(num%2==0)
        {
            long long x=int(sqrt(num));
            while(i<=x)
            {	
            	if(num%i==0)
            	{
	                if(i%2==0)
	                {
	                    div++;
	                }
	                
	                long long int  p=num/i;
	                
	                if(p%2==0&&i*i!=num)
	                {
	                    div++;
	                }
            	}
            	i++;
            }
        }
        
        cout<<div<<endl;
    }
}
