#include<bits/stdc++.h>
typedef long long ll;
using namespace std;
int const Size = 1005;
int row[Size], col[Size], rc[Size][Size];
int n, m;
char arr[Size][Size];
int main(){
	int n,m;
	cin >> n >> m;
	int ans = 0;
	for(int i=1; i<=n; i++){
		cin >> arr[i] + 1;
		for(int j=1; j<=m; j++){
			if(arr[i][j] == 'B'){
				row[i]++;
				col[j]++;
				rc[i][j]++;
			}
		}
	}
	for(int i=1; i<=n; i++){
		for(int j=1; j<=m; j++)
			ans = (ans+(rc[i][j]+row[i]+col[j])%2);
	}
	cout << ans;
	return 0;
}
