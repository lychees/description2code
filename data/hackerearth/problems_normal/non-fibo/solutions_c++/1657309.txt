#include<bits/stdc++.h>
using namespace std;
long long fib[100];
int main()
{
	long long t, i;
	fib[1]=1;
	fib[2]=2;
	for(i=3; i<100; i++)
		fib[i]=fib[i-1]+fib[i-2];
	cin>>t;
	while(t--)
	{
		long long ans;
		long long n, sum=0, off;	
		cin>>n;
		for(i=2; i<=74; i++)
		{
			off=fib[i]-fib[i-1]-1;
			if(n-off > 0)
				n-=off;
			else
				break;
		}
		cout<<(fib[i-1]+n)<<endl;
	}
	return 0;
}
