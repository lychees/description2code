#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int n,x,i;
    cin>>n>>x;
    int s=0,c=0;
    long long int a[n];
    for(i=0;i<n;i++)
    {
        cin>>a[i];
    }
    for(i=0;i<n;i++)
    {
        if(a[i]>x)
            s++;
        else
            c++;

        if(s>1)
           break;
    }
    cout<<c;
   return 0;
}
