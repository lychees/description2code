#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    long int t;
    scanf("%ld",&t);
    long int n,k;
    while(t--)
    {
    	scanf("%ld %ld",&n,&k);
    	
    	if((n-1)%(k+1))
    	{
    		printf("Rani\n");
    	}
    	else
    	{
    		printf("Nandu\n");
    	}
    	
    }
    return 0;
}
