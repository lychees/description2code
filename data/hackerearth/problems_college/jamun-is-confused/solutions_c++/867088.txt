#include <iostream>
using namespace std;

long long gcd(long long a,long long b){
	if(b==0) return a;
	if(a<b)	return gcd(b,a);
	while(b){
		long long t=b;
		b=a%b;
		a=t;
	}
	return a;
}

int main()
{
    int t,n;
    long long a[1001];
    long long g;
    cin>>t;
    while(t--){
    	cin>>n;
    	for(int i=0;i<n;i++){
    		cin>>a[i];
    		if(i==0) g=a[i];
    		else g=gcd(g,a[i]);
    	}
    	cout<<g<<endl;
    }
    return 0;
}
