#include <iostream>
#include <string>
using namespace std;
int main(){
	ios::sync_with_stdio(false);
	int T = 0;
	cin >> T;
	cin.ignore();
	
	while(T--)
	{
	 string a;
	 string b;
	 a = b = "";
	 
	 getline(cin,a);
	 getline(cin,b);
	 
	 int len = min(a.length(), b.length());
	 int count = 0;
	 
	 for(int i = 0; i < len; i++)
	  if(a[i] == b[i])
	   count++;
	 
	 cout << count << endl;
	}
	
	return 0;
}
