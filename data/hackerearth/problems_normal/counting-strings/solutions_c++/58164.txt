#include<string>
#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
	long long t, i, j, l;
	string m;
	cin>>t;
	while(t--){
		long a[1000010];
		cin>>m;
		long long ans = 0;
		l = m.size();
		a[l] = l;
  		for(i=l-1;i>=0;--i){
  				if(m[i]=='a'||m[i]=='z')
  					a[i] = i;
  				else
  					a[i] = a[i+1];
  				ans = ans + l - a[i];
  		}
  		printf("%ld\n",ans);
	}
  return 0;
}
