#include<bits/stdc++.h>
using namespace std;
int cnt[56]={0};
char s1[104];
char s2[104];
int main()
{
  int t;
  cin>>t;
  int temp=t;
  while(t--)
  {
    scanf("%s",s1);
    scanf("%s",s2);
    //printf("%s %s",s1,s2);
    int len=strlen(s1);
    for(int i=1;i<=5;++i)cnt[i]=0;
    for(int i=0;i<len;++i)
    {
      if(s1[i]=='1' && s2[i]=='0')cnt[1]++;
      if(s1[i]=='0' && s2[i]=='1')cnt[2]++;
      if(s1[i]=='?' && s2[i]=='0' )cnt[3]++;
      if(s1[i]=='?' && s2[i]=='1')cnt[4]++;
    }
    int ans=0;
    if(cnt[1]>cnt[2] && cnt[1]-cnt[2]>cnt[4])ans=-1;
    else if(cnt[1]>cnt[2] && cnt[1]-cnt[2]<=cnt[4] )ans+=cnt[2]+2*(cnt[1]-cnt[2])+cnt[4]-(cnt[1]-cnt[2])+cnt[3];
    else if(cnt[2]>=cnt[1])ans+=cnt[2]+cnt[3]+cnt[4];
  printf("Case %d: %d\n",temp-t,ans);
  }

  return 0;
}
