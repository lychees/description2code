# parallel sort

import operator

N       = int(raw_input())

My_list = []

for i in range(N):
    name,marks  = [x for x in raw_input().split()]
    marks       =  int(marks)
    My_list.append((name,marks))

My_list.sort(key=operator.itemgetter(0))
My_list.sort(key=operator.itemgetter(1),reverse=True)

for tupli in My_list:
    print tupli[0],tupli[1]
