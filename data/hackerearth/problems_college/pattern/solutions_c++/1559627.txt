#include <iostream>
using namespace std;
     
int main(){
   
	int n;
	int x;
    
	cin>>n;
	
	for(int i=0;i<n;i++){
	    cout <<"*";
}
	cout<<endl;

	for(int i=0;i<n-2;i++){
    
		for(x=0;x<n-i-2;x++)
    	cout<<" ";
    
	cout <<"*"<< endl;
    }
    
	while(n--){
    	cout<<"*";
 }
 
 return 0;
}
