#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */  
    int t;
    cin>>t;
    while(t--)
        {
        string a,b;
        cin>>a>>b;
        int i,flag=0;
       int len=a.size();
        int len1=b.size();
        for(i=0;i<len &&i<len1;i++)
            {
            if(a[i]==b[i])
                {
                flag=1;
                break;
            }
            else
                continue;
        }
        if(flag==1)
            {
            cout<<"YES"<<endl;
        }
        else
            cout<<"NO"<<endl;
            
    }
    return 0;
}
