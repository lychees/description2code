#include <iostream>
using namespace std;
typedef long long ll;
int check(ll n){
	ll count=0;
	while(n){
		n&=(n-1);
		count++;
		
	}
	if(count%2==0)
	   return 0;
	else
	   return 1;
}	   
int main()
{
    ll t,n;
    scanf("%lld",&t);
    while(t--){
    	scanf("%lld",&n);
    	if(check(n))
    	  printf("odd\n");
    	else
    	  printf("even\n");
    }	  
    return 0;
}
