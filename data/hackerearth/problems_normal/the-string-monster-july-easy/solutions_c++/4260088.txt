//its not how good you are...its how good you want to be
#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define ll      int
#define inf 1000000000000
#define mod 1000000007
#define pb push_back
#define mp make_pair
#define all(v) v.begin(),v.end()
#define S second
#define F first
#define boost1 ios::sync_with_stdio(false);
#define boost2 cin.tie(0);
#define mem(a,val) memset(a,val,sizeof a)
#define ld  long double
#define endl "\n"

string a[50],b[50],s[50],t;
ll cnt[50],req[50];
map<vector<int>,ll> m;
vector<ll>v1,v2;
int main()
{
	boost1;boost2;
	ll i,j,n,te,mid,k,l,size1,size2,mask,f,f1;
	cin>>te;
	while(te--)
	{
		mem(req,0);
		cin>>n;
		for(i=0;i<n;i++)
		cin>>s[i];
		cin>>t;
		for(i=0;i<t.length();i++)
		req[t[i]-'a']++;
		if(n%2==0)
		mid=n/2;
		else
		mid=n/2+1;
		for(i=0;i<mid;i++)
		a[i]=s[i];
		for(i=mid;i<n;i++)
		b[i-mid]=s[i];
		size1=mid;
		size2=n-mid;
		f1=0;
		for(mask=0;mask<(1<<size1);mask++)
		{
			mem(cnt,0);
			for(i=0;i<size1;i++)
			{
				if(mask&(1<<i))
				{
					for(j=0;j<a[i].length();j++)
					cnt[a[i][j]-'a']++;
				}
			}
			v1.clear();
			for(i=0;i<26;i++)
			v1.pb(cnt[i]);
			m[v1]=1;
		}
		for(mask=0;mask<(1<<size2);mask++)
		{
			mem(cnt,0);
			for(i=0;i<size2;i++)
			{
				if(mask&(1<<i))
				{
					for(j=0;j<b[i].length();j++)
					cnt[b[i][j]-'a']++;
				}
			}
			v2.clear();
			f=0;
			for(i=0;i<26;i++)
			{
				if(cnt[i]>req[i])
				{
					f=1;
					break;
				}
				v2.pb(req[i]-cnt[i]);
			}
			if(f)
			continue;
			if(m[v2])
			{
				f1=1;
				cout<<"YES"<<endl;
				break;
			}
		}
		if(!f1)
		cout<<"NO"<<endl;
	}
	return 0;
}
