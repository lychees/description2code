#define MAX 10010

int main() {	
	char s[MAX];
	int test;

	cin >> test;

	while(test--) {
		int a,b, ans = 0;
		bool val = 1;

		scanf("%s",s);
		cin >> a >> b;
		
		int i = 0, j = strlen(s)-1;

		while(i < j) {
			if(s[i] == '/' && s[j] == '/') {
				ans += 2 * min(a,b);
			}
			else if(s[i] == '/') {
				if(s[j] == 'a')
					ans += a;
				else
					ans += b;
			}
			else if(s[j] == '/') {
				if(s[i] == 'a')
					ans += a;
				else
					ans += b;
			}
			else if(s[i] != s[j]){
				val = 0;
				break;
			}
			i++; j--;
		}


		if(val)
			cout << ans << endl;
		else 
			cout << -1 << endl;
	}

	return 0;
}
