#include<iostream>
#include<cmath>

unsigned long long p(unsigned long long x)
{
	if(!x)
	return 0;
	unsigned long long z=0,n=1;
	while(x/pow(5,n))
	{
		z+=x/pow(5,n);
		n++;
	}
	return z;
}

int main()
{
	int t;
	std::cin>>t;
	while(t--)
	{
		unsigned long long n;
		std::cin>>n;
		std::cout<<p(n)<<"\n";
	}
	return 0;
}
