{
   // cout << "Hello World!" << endl;
    int t;

    {
    	string ch,ch2;
    	cin>>ch;
    	ch2=ch;
    	reverse(ch2.begin(),ch2.end());
    	long long int i,j;
    	int flag=0;
    	for(i=1;i<ch.length();i++)
    	{
    		if(abs(ch[i-1]-ch[i])!=abs(ch2[i-1]-ch2[i]))
    		flag=1;
    	}

    	else cout<<"Funny"<<endl;
    }
