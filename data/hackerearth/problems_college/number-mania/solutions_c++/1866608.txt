int mul(int x,int res[],int res_size){
    int carry=0;
    for(int i=0;i<res_size;i++){
        int pro=res[i]*x+carry;
        res[i]=pro%10;
        carry=pro/10;

    }
    while(carry>0){
        res[res_size]=carry%10;
        carry=carry/10;
        res_size++;
    }
    return res_size;
}
