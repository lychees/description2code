#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	string s;
    	cin>>s;
    	int counter=0,ans1=0,ans2=0;
    	for(int i=0;i<n;i++)
    	{
    		if(s[i]=='0')
    			counter++;
    	}
    	for(int i=0;i<counter;i++)
    	{
    		if(s[i]=='1')
    			ans1++;
    	}
    	for(int i=0;i<n-counter;i++)
    	{
    		if(s[i]=='0')
    			ans2++;
    	}
    	ans1=min(ans1,ans2);
    	cout<<ans1<<endl;
    }
    return 0;
}
