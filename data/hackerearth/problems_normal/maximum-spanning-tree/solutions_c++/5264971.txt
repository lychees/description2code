#include<bits/stdc++.h>
using namespace std;
long long t,n,k,a,b,c,arr[5010],mst=0;
priority_queue<pair<int,pair<int,int> > > q;
void initialise()
{
    for(int i=1;i<=n;i++)
    {
        arr[i]=i;
    }
}
int root(int x)
{
    while(x!=arr[x])
    {
        arr[x]=arr[arr[x]];
        x=arr[x];
    }
    return x;
}
void unn(int a, int b)
{
    arr[root(a)]=root(b);
}
void kruskal()
{
    int w,a,b;
    while(!q.empty())
    {
        w=q.top().first;
        a=q.top().second.first;
        b=q.top().second.second;
        q.pop();
        if(root(a)!=root(b))
        {
            unn(a,b);
            mst+=w;
        }
    }
}
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(NULL);
	cin>>t;
	while(t--)
	{
	    cin>>n>>k;
	    initialise();
	    for(int i=0;i<k;i++)
	    {
	        cin>>a>>b>>c;
	        q.push(make_pair(c,make_pair(a,b)));
	    }
	    mst=0;
	    kruskal();
    	cout<<mst<<endl;
	}
    return 0;
}
