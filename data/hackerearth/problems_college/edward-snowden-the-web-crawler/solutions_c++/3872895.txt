#include <iostream>
using namespace std;

int main()
{
	string websitename;
	cin >> websitename;
    string currentstring;
    while (true)
    {
    	cin >> currentstring;
    	if(currentstring.find("exit")!=string::npos)
    		return 0;
    	if(currentstring.find(websitename)!=string::npos && currentstring.find("href")!=string::npos)
		{
    		int start = currentstring.find(websitename);
    		int end = currentstring.find(">");
    		cout << currentstring.substr(start, end - start - 1) << "\n";
    	}
    	currentstring="";
    }
    return 0;
}
