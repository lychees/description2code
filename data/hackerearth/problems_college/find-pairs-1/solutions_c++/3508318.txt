#include<bits/stdc++.h>
using namespace std;
int ar[1000];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        for(int i=0;i<n;i++)
        {
            cin>>ar[i];
        }
        int ans=0;
        for(int i=0;i<n;i++)
        {
            for(int j=i;j<n;j++)
                if(ar[i]==ar[j])
                ans++;

        }
        cout<<ans<<endl;
    }
    return 0;
}
