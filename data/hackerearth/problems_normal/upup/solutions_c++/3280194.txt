#include <bits/stdc++.h>
using namespace std;

int main()
{
    string str;
    getline(cin,str);
    bool isStart=true;
    for(int i=0;i<str.size();i++) {
    	if(str[i]==' ')
    		isStart=true;
    	else if(str[i]>='a' && str[i]<='z' && isStart) {
    		str[i]=str[i]-32;
    		isStart=false;
    	}
    	else
    		isStart=false;
    }
    cout<<str;
    return 0;
}
