#include <iostream>
using namespace std;

int main()
{
    int T;
    long N;
    cin>>T;
    while(T--)
    {
    	cin>>N;
    	long ans;
    	cin>>ans;
    	ans=ans%5;
    	for(long i=1;i<N;++i)
    	{
    		long temp;
    		cin>>temp;
    		temp=temp%5;
    		ans=ans^temp;
    	}
    	if(ans==0)
    	cout<<"Andrew\n";
    	else
    	cout<<"Aniruddha\n";
    }
    return 0;
}
