#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int a,b,d;
		cin>>a>>b>>d;
		long long int p = pow(a,b);
		long long int c = p/d;
		 a = c*d;
		b = (c+1)*d;
		long long int x = min(b-p,p-a);
		if(c*d == p)
		{
			cout<<p<<"\n";
		}
		else if(x == b-p)
		{
			cout<<b<<"\n";
		}
		else if(x == p-a)
		{
			cout<<a<<"\n";
		}

	}
}
