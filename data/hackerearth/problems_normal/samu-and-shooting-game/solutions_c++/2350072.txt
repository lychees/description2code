const int mod = 1000000007;
double dp[1001][1001];
size_t W,X,Y;
double P1,P2;
double bt(size_t N,size_t cur){
	if(N == 0){
		if(cur >= W)
			return 1.;
		return 0.;
	}
	double &ref = dp[N][min(cur,W)];
	if(ref >= 0)
		return ref;
	ref = 0.;
	ref = max(ref,P1 * bt(N-1,cur + X) + (1. - P1) * bt(N-1,cur));
	ref = max(ref,P2 * bt(N-1,cur + Y) + (1. - P2) * bt(N-1,cur));
	return ref;
}
int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(nullptr);

	size_t T,N;
	cin>>T;
	while(T--){
		cin>>X>>Y>>N>>W>>P1>>P2;
		P1/=100.;
		P2/=100.;
		memset(dp,-1,sizeof dp);
		cout<<fixed<<setprecision(6)<<bt(N,0)*100.<<endl;
	}
 	return 0;
}
