#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main()
{
	vector<ll> A;
    int n;cin>>n;
    //ll A[n];
    ll x;
    for(int i=0;i<n;i++){
    	cin>>x;
    	A.push_back(x);
    }
    	
    sort(A.begin(),A.end());
   
    ll temp1,temp2;
    temp1=A[n-1]*A[n-2];
    temp2=A[0]*A[1];
    
    if(temp1>temp2)
    	cout<<temp1;
    else
    	cout<<temp2;
    	
    return 0;
}
