#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
#define MAX 210
int TC, n, s;
int A[MAX];
 
bool solve(int x, int sum) {
	if (sum == 0) {
		return false;
	}
	if (!(sum & (sum - 1))) {
		return true;
	}
	if (x >= n) {
		return false;
	}
	return (solve(x + 1, sum & A[x]) || ((sum != (sum & A[x])) && solve(x + 1, sum)));
}
 
int main()
{
	scanf("%d", &TC);
	while (TC--) {
		scanf("%d", &n);
		for (int i = 0; i < n; i++) {
			scanf("%d", &A[i]);
		}
		if (solve(0, -1)) {
			printf("YES\n");
		}
		else {
			printf("NO\n");
		}
	}
    return 0;
}
