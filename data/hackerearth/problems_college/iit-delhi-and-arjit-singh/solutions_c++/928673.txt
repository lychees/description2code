int main()
{
	vector< vector<int> > p(1000001);
	for (int i = 2; i <= 1000000; i++)
		if (p[i].empty())
			for (int j = i; j <= 1000000; j += i)
				p[j].push_back(i);
	int t; cin >> t;
	while (t--) {
		int N, M; cin >> N >> M;
		ll ans = 0;
		for (int i = 1; i * M <= N; i++) {
			int n = p[i].size();
			for (int S = 0; S < (1 << n); S++) {
				int x = 1;
				for (int j = 0; j < n; j++)
					if (S & (1 << j)) x *= -p[i][j];
				ans += i / x;
			}
		}
		cout << ans << endl; 
	}
}
