#include<iostream>
#include<vector>
#include<limits.h>
using namespace std;
int fun(vector<vector<int> >& game,vector<int>& query){

	int ans=game[0].size()-1;
	int size=game.size();
	
  
    

	for(int i=0;i<size;i++){
		while(query[i]<game[i][ans])
		ans--;
	}
	return ans;
}
int main(){
	int n,m,q;
	cin>>n>>m>>q;
	vector<vector<int> > game;
	game.resize(n);
	int l=0;
	while(l<n){
		int k=0;
		while(k<m){
			int temp;
				scanf("%d",&temp);
			game[l].push_back(temp);
			k++;
		}
		l++;
	}
	while(q--){
		vector<int> query;
		int k=0;
		query.clear();
		while(k<n){
			int temp;
			scanf("%d",&temp);
			query.push_back(temp);

			k++;
		}
		printf("%d\n",fun(game,query)+1);
	}

}
