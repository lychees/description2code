#include <bits/stdc++.h>

using namespace std;

long int dp[100005] = {0};

int main()
{
    int n;
    cin >> n;
    int a[n];
    for (int i = 0;i < n;i++)
        cin >> a[i];
    dp[0] = 1;
    for (int i = 0;i < n;i++)
    {
        for (int j = 100005;j >= 0;j--)
        {
            if (dp[j])
                dp[j+a[i]] = 1;
        }
    }
    
    int q;
    cin >> q;
    
    while(q--)
    {
        long int num;
        cin >> num;
        
        if (dp[num] == 1)
            cout << "Yes" << endl;
        else 
            cout << "No" << endl;
    }
    
    return 0;
}
