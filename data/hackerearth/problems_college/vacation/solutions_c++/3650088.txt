#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false); 
	cin.tie(NULL);
    long long int t;
    scanf("%lld",&t);
    while(t--)
    {
    	long long int n;
    	scanf("%lld",&n);
    	if(n==1)
    	printf("Bob\n");
    	else if(n>=2)
    	printf("Alice\n");
    }
    return 0;
}
