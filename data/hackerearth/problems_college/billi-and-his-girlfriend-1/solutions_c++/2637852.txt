#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
typedef unsigned long long int ull;
int main()
{
	ull t,a,b,k,sum;
	cin>>t;
	while(t--)
	{
		cin>>a>>b>>k;
		ull r=1;
		sum=0;
		k=pow(10,k);
		while(b>0)
		{
			if(b%2==1)
			{
				r=(r*a)%k;
			}
			a=(a*a)%k;
			b/=2;
		}
		while(r>0)
		{
			sum+=r%10;
			r/=10;
		}
		cout<<sum<<endl;
	}
	return 0;
}
