#include <iostream>
#include <cstdio>
#include <math.h>
using namespace std;

int main()
{
	
	int t;
	long long int n,i,sum,count;
	long double root;
	scanf("%d",&t);
	while(t--)
	{
		sum=1;
		count=0;
		scanf("%lld",&n);
		if(n==1)
		printf("0\n");
		else
		{
			n=n-1;
			n*=8;
			n=n+1;
			root=sqrt(n);
			root=root-1;
			root=root/2;
		sum=ceil(root);		
		printf("%lld\n",sum);
		}
	}
	return 0;
}
