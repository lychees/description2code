#include<bits/stdc++.h>
#include<cstring>
using namespace std;
char matr[123][123];
char s[209];
int main()
{
	int tc;
	char temp;
	scanf("%d",&tc);
	scanf("%c",&temp);
	while(tc--)
	{
		memset(matr ,0,sizeof matr);
		scanf("%[^\n]s",s);
		scanf("%c",&temp);
		//printf("%s\n",s);
		int len=strlen(s);
		int n=sqrt(len);
		int k=0;
		for(int j=0;j<n;++j)
		{
			for(int i=0;i<n;++i)
			{
				matr[i][j]=s[k];
				++k;
			}
		}
		for(int i=0;i<n;++i)
			for(int j=0;j<n;++j)
				printf("%c",matr[i][j]);
		printf("\n");

	}
	return 0;
}
