#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin>>n;
	vector<int> p(n);
	for(int i = 0; i < n; ++i)
		cin>>p[i];
	sort(p.begin(), p.end());
	int q;
	cin>>q;
	while(q--)
	{
		int a, b;
		cin>>a>>b;
		cout<<upper_bound(p.begin(), p.end(), b) - lower_bound(p.begin(), p.end(), a)<<endl;
	}
    return 0;
}
