#include <iostream>
#include <stdlib.h>
#include <sstream>
#include <utility> //pair
#include <queue>
#include <map>

int main()
{
    std::string line;
    std::getline(std::cin, line);
    int num_lines = atoi(line.c_str());

    std::priority_queue< std::pair<long, long> > q;
    std::map<long, long> new_z_map;
    long id, old_z, P, L, C, S;
    long z_increment;
    long new_z;
    for (int i = 0; i < num_lines; i++)
    {
        std::string line;
        std::getline(std::cin, line);
        std::stringstream line_stream(line);
        line_stream >> id >> old_z >> P >> L >> C >> S;
        new_z = P*50 + L*5 + C*10 + S*20;
        z_increment = new_z - old_z;
        q.push(std::make_pair(z_increment, id));
        new_z_map.insert(std::make_pair(id, new_z));

    }

    std::pair<long, long> record;
    for (int i = 0; i < 5; i++)
    {
        record = q.top();
        q.pop();
        std::cout << record.second << " " << new_z_map.at(record.second) << std::endl;
    }

    return 0;
}
