#include<iostream>
using namespace std;
int main () {
    long long i , t , n;
    scanf ( "%lld" , &t );
    while ( t-- ) {
        scanf ( "%lld" , &n );
        long long cnt=0;
        long long i=0;
        while(n>>i)
        {
          if((n>>i)&1)
                cnt++;
            i++;
        }
        printf ( "%lld\n" , cnt );
    }
    return 0;
}
