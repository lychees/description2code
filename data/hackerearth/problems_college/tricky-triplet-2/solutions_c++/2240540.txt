#include<iostream>
using namespace std;
int main(){
    long long n;
    cin>>n;
    long long a[n];
    for(long long i=0;i<n;i++){

        cin>>a[i];
    }
    long long count=0;
    for(long long i=0;i<n;i++){
        for(long long j=i+1;j<n-1;j++){
            for(long long k=j+1;k<n;k++){
                if(a[k]-a[j]==a[j]-a[i]){
                   count++;
                   }
            }
        }
    }
cout<<count;
}
