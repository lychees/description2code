#include <iostream>
using namespace std;
long long int sum_of_divisors(long long int n)
{
	long long int sum = 0;
	for(long long int i =1;i*i<=n;i++ )
	{
		if(n%i == 0 )
		{
			if(i%2!=0)
			 sum += i;
			 if(i*i!=n && (n/i)%2!=0)
			 {
			 	sum += n/i;
			 }
		}
	}
	return sum;
}
int main()
{
    int t;
    long long int n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	long long int sum =0;
    	sum = sum_of_divisors(n);
    	if(n%2==0)
    	 cout<<24*sum<<endl;
    	 else
    	 cout<<8*sum<<endl;
    }
    return 0;
}
