#include<bits/stdc++.h>
using namespace std;
struct price{
	int s;
	int p;
	int f;
};
price arr[100001];
int memo[100001][3];
int dp(int n,int o)
{
	if(n==0)
		return 0;
	if(memo[n][o])
		return memo[n][o];
	if(o==0)
	{
		return memo[n][o] = min(arr[n-1].p + dp(n-1,1),arr[n-1].f + dp(n-1,2));
	}
	else if(o==1)
	{
		return memo[n][o] = min(arr[n-1].s + dp(n-1,0),arr[n-1].f + dp(n-1,2));
	}
	else if(o==2)
	{
		return memo[n][o] = min(arr[n-1].s + dp(n-1,0),arr[n-1].p + dp(n-1,1));
	}
	else 
	{
		return memo[n][o] = min(min(arr[n-1].s + dp(n-1,0),arr[n-1].p + dp(n-1,1)),arr[n-1].f + dp(n-1,2));
	}
}
int main()
{
	int i,t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		memset(memo,0,sizeof(memo));
		for(i=0;i<n;i++)
		{
			cin>>arr[i].s>>arr[i].p>>arr[i].f;
		}
		cout<<dp(n,-1)<<endl;
	}
}
