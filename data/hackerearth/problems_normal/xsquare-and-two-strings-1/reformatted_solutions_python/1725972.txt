import sys

for __ in range(input()):
    sa = raw_input()
    sb = raw_input()
    mark = [0]*150
    for i in sa:
        mark[ord(i)] += 1
    pr = False
    for i in sb:
        if mark[ord(i)] > 0:
            print "Yes"
            pr = True
            break
    if not pr:
        print "No"
