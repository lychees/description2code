#include <iostream>
#include <math.h>
using namespace std;

int binary_decimal(int n) /* Function to convert binary to decimal.*/

{
    int decimal=0, i=0, rem;
    while (n!=0)
    {
        rem = n%10;
        n/=10;
        decimal += rem*pow(2,i);
        ++i;
    }
    return decimal;
}

int main()
{
   // cout << "Hello World!" << endl;
   int t,A,B,xor1,xor2;
   cin>>t;
   while(t--)
   {
   	cin>>A;
   	cin>>B;
   	A=binary_decimal(A);
   	B=binary_decimal(B);
   	
   	//cout<<A<<"  "<<B;
   	
   	xor1=A^B;
   	xor2=B^A;
   	
   //	cout<<xor1<<"  "<<xor2;
   	
   	if(A>=B)
   	cout<<"The Hound"<<endl;
   	else
   	cout<<"The Mountain"<<endl;
   	
   }
    return 0;
}
