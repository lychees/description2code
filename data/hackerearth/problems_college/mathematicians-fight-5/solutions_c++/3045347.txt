#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int myfind(ll int x)
{
	if(x==1)
	{
		return x;
	}
	else
	{
		if(x&1)
		return 2*myfind(x>>1)+1;
		else
		return 2*myfind(x>>1)-1;
	}
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int n;
	cin>>n;
	ll int i,x;
	for(i=0;i<n;i++)
	{
		cin>>x;
		cout<<myfind(x)<<endl;
	}
	return 0;
}
