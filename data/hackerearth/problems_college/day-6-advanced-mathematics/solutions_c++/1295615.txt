#include<bits/stdc++.h>
using namespace std;

int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		string str;
		cin>>str;
		bool flag = false;
		for(int i=0;i<str.length()-1;i++)
		{
			if(str[i] == str[i+1])
			{
				flag = true;
				break;
			}
		}
		if(flag)
		{
			cout<<"SLAP"<<endl;
		}
		else
		{
			cout<<"KISS"<<endl;
		}
	}
	return 0;
}
