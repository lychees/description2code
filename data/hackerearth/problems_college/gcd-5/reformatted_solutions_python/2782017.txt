t = input()
while t > 0:
    t -= 1
    n = input()
    a = map(int, raw_input().split())
    if n == 1:
        print "YES"
        continue
    s = True
    for i in range(1, n):
        if a[i] < a[i-1]:
            s = False
            break
    if s:
        print "YES"
    else:
        print "NO"
