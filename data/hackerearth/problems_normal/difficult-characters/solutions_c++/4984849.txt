#include <iostream>
#include <algorithm>
using namespace std;

struct dd{
	int pos;
	int count;
};

bool compare(dd a,dd b){
	
	if(a.count < b.count || (a.count == b.count && a.pos > b.pos)){
		return 1;
	}
	return 0;
}

int main()
{
	
	string str;
	int t;
	cin>>t;
	while(t>0){
	
    	cin>>str;
    	dd node[26];
    	
    	for(int i=0;i<26;i++){
    		node[i].pos=i;
    		node[i].count=0;
    		}
    	int len=str.length();
    	
    	for(int i=0;i<len;i++){
    		node[str[i]-'a'].count+=1;
    		}
    	
    	sort(node,node+26,compare);
    	
    	for(int i=0;i<26;i++){
    		cout<<char(node[i].pos + 'a')<<" ";
    		}
    		cout<<endl;
    	t--;
    	
	}
    return 0;
}
