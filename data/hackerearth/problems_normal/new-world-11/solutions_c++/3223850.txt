typedef long long ll;
const ll MOD = 1e9+7;
const int INF = 0x7f7f7f7f;
const double eps = 1e-6;

const int maxn = 100010;
int arr[maxn];
int n,k;

bool ispossible(int x) {
	int num = 0;
	int p = 0;
	for (int i = 1; i < n; ) {
		int xx = arr[i] - arr[p];
		if(xx <= x) ++ i;
		else {
			if(abs(p - i) == 1) return false;
			num ++;
			p = i - 1;
		}
	}
	
	return num <= k - 1;
}

int main() {
	ios_base::sync_with_stdio(false);
	
	int t; cin >> t;
	while (t --) {
		cin >> n >> k;
		for (int i = 0; i < n; i ++) cin >> arr[i];

		int lo = 0, hi = arr[n - 1] - arr[0];
		while (lo <= hi) {
			int mid = (lo + hi) / 2;
			if(ispossible(mid)) hi = mid - 1;
			else lo = mid + 1;
		}

		cout << lo << "\n";
	}
	return 0;
}
