#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    char time[6];
    cin>>time;
    int hour= time[1]-'0'+10*(time[0]-'0');
    int minute= time[4]-'0'+10*(time[3]-'0');
    double houra=0.5 * (hour*60 + minute);
    double minutea=minute*6;
    if (houra == 12) houra = 0;
	if (minutea == 60) minutea = 0;
    //cout<<houra<<" "<<minutea<<endl;
    double diff=(houra-minutea);
    if((houra-minutea)<0)
    diff=minutea-houra;
    //cout<<houra<<" "<<minutea<<endl;
    //cout<<diff<<endl;
    cout << std::fixed;
	if(diff<180)
	cout<<std::setprecision(6)<<diff<<endl;
	else cout<<std::setprecision(6)<<360-diff<<endl;    
    return 0;
}
