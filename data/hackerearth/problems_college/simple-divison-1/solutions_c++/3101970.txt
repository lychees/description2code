#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int a,b,c;
    	cin>>a>>b>>c;
    	cout<<a/b;
    	int rem=a%b;
    	if(c!=0)
    	cout<<'.';
    	for(int i=1;i<=c;i++)
    	{
    		if(rem==0)
    		cout<<0;
    		else
    		{
    			rem*=10;
    			while((rem<b)&&(i<c))
    			{
    				rem*=10;
    				cout<<0;
    				i++;
    			}
    			cout<<rem/b;
    			rem%=b;
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
