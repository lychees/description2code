#include <iostream>
#include <cmath>
using namespace std;


int test;

long long d;
int i;
long long coaching=0;
long long diff=0;
int main()
{
	cin>>test;
	cin>>d;
	long long *arr = new long long[d]{0};
	i=0;
	while(test--){
		cin>>arr[i];
	
		if(i==0){
			i++;
			continue;
		}
		diff = arr[i] - arr[i-1];
		
		if(diff<0){
			long long hours = ceil( abs(diff) / d);
			++hours;
			arr[i] +=  d * (hours);
			coaching+= hours;
		}
		//	cout<<arr[i]<<"=i"<<i<<endl;
		i++;
	}
	cout<<coaching<<endl;
   
   return 0;
}
