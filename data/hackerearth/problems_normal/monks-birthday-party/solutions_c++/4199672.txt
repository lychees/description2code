#include<iostream>
#include<string>
#include<set>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        string s[n];
        set <string> a;
        for(int i=0;i<n;i++)
        {
            cin>>s[i];
        }
        for(int i=0;i<n;i++)
        {
            if(!a.count(s[i]))
                a.insert(s[i]);
        }
        set <string>::iterator it;
        for(it=a.begin();it!=a.end();it++)
            cout<<*it<<endl;




    }
    return 0;

}
