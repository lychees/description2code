#include <iostream>
using namespace std;
#define ll long long

int main()
{
    ll n,sum=0;
    cin>>n;
    ll A[n], B[n];
    cin>>B[0];
    A[0] = B[0];
    cout<<A[0];
    sum=A[0];
    for( auto i=1; i<n;i++) {
    	cin>>B[i];
    	A[i]=B[i]*(i+1)-sum;
    	sum+=A[i];
    	cout<<" "<<A[i];
    }
    
    return 0;
}
