#include<bits/stdc++.h>

using namespace std;

int Dom[1000001];
int Brian[1000001];

int main(){

	int n;
	cin>>n;
	
	for(int i=0;i<n;i++)
	   cin>>Dom[i];
	
	for(int i=0;i<n;i++)
	   cin>>Brian[i];
	   
	int maxd=0,maxb=0;   
	
	for(int i=0;i<n-1;i++){
	   int dif=abs(Dom[i+1]-Dom[i]);
	   if(dif>maxd)
	      maxd=dif;
	}      

	for(int i=0;i<n-1;i++){
	   int dif1=abs(Brian[i+1]-Brian[i]);
	   if(dif1>maxb)
	      maxb=dif1;
	}
	
	int ans;
	
	if(maxd>maxb)
	  cout<<"Dom"<<endl<<maxd;
	else
	  cout<<"Brian"<<endl<<maxb;	

return 0;
}
