#include <iostream>
using namespace std;

int main()
{
   int t,a0,a1,a2,c,n,mo;
   cin>>t;
   while(t--)
   {
   	mo = 0;
   	scanf("%d%d%d%d",&a0,&a1,&c,&n);
   	if(n==1){
   		if(!a0) printf("YES\n");
   		else printf("NO\n");
   		continue;
   	}
   	mo = a0*10+a1;
   	mo %= 41;n-=2;
   	while(n>0)
   	{
   		
   		n--;
   		a2 = (a1*c+a0)%10;
   		mo*=10;mo+=a2;
   		mo%=41;
   		a0=a1;
   		//cout<<"ee";
   		a1=a2;
   	}
   	
   	
   		if(mo) printf("NO\n");
   		else printf("YES\n");
   	
   }
}
