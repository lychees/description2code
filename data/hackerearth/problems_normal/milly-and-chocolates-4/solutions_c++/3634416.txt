#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   	int t,n,i,a;
   	double s=0;
   	cin>>t;
   	while(t--)
   	{
   		cin>>n;
   		s=0;
   		for(i=0;i<n;i++)
   		{
   			cin>>a;
   			s+=a;
   		}
   		printf("%.8f\n",(s/(double)n));
   	}
    return 0;
}
