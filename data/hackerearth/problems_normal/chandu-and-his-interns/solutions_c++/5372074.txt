#include <iostream>
#include<cmath>
using namespace std;

bool divisor(int X){
	int count = 2, Y;
	for(int i = 2 ; i <= sqrt(X) ; i++ ){
		if( ( X % i ) == 0 ){
			count++;
			Y = ( X / i );
			if( Y != i )
				count++;
		}
		if( count >= 4 ){
			return true;
		}
	}
	return false;
}
int main()
{
    int N, X;
    cin>>N;
    while(N--){
    	cin>>X;
    	if( divisor(X) ){
    		cout<<"YES"<<endl;
    	}
    	else{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
