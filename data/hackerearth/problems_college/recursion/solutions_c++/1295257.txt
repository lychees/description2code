#include <iostream>
#include<math.h>
#include<vector>
#include<algorithm>
using namespace std;
bool cmp(long long int a,long long int b){
return((a<b)?true:false);	
	
}
void swap(int* a,int* b){
int t=*a;
*a=*b;
*b=t;
	
}
long long int print(int* arr,int n){
	long long int num=0;
	int g=n;
	for(int i=0;i<=n;i++){
		
		num=num+(arr[i]*pow(10,g));
		g--;
	}
	return num;
	
}
void per(int* arr,int s,int e,vector<long long int>& v){
if(s==e){
	
	long long int get=print(arr,e);
	
	v.push_back(get);
}	

else{
	
	for(int j=s;j<=e;j++){
		
		swap(&arr[s],&arr[j]);
		per(arr,s+1,e,v);
		swap(&arr[s],&arr[j]);
		
	}
}
 
	
}
void printPer(int n){
	
 int arr[n];
 
 for(int i=0;i<n;i++){
 	
 	arr[i]=i+1;
 }
 
 vector<long long int> v;
 per(arr,0,n-1,v);
 
 sort(v.begin(),v.end(),cmp);
 
 vector<long long int>::iterator it=v.begin();
 for(it;it!=v.end();it++){
 	
 	cout<<*it<<" ";
 }
 
	
}


int main()
{
	
	int tc,test,ip;
	cin>>test;
	while(test--){
	cin>>ip;
	printPer(ip);
    cout<<endl;
	}
    return 0;
}
