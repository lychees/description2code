t = input()
while t > 0:
    inp = raw_input().split()
    a = int(inp[0])
    b = int(inp[1])
    equ = ""
    for i in range(54, -1, -1):
        multi = 1 << i
        if multi <= b:
            equ += "(%d<<%d)" % (a, i)
            b -= multi
    equ = equ.replace(")(", ") + (")
    print equ
    t-=1
