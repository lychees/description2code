#include<iostream>
#include<stdlib.h>
int main()
{
	int *p;
	p=(int *)malloc(100000000*sizeof(int));
	int N,M,T,i,j;
	std::cin>>T;
	while(T--)
	{
		std::cin>>N;
		std::cin>>M;
		for(i=0,j=N-1;i<N,j>0;i+=2,j-=1)
		{
			p[i]=j;
		}
		for(i=1,j=0;i<N;i+=2,j++)
			p[i]=j;
		for(i=0;i<N;i++)
			if(p[i]==M)
				break;
		std::cout<<i<<std::endl;
	}
	return 0;
}
