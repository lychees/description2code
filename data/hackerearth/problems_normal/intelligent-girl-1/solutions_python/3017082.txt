S = str(input())

mapping = [0] * len(S)

for idx, char in enumerate(S):
	if not int(char)&1:
		mapping[idx] = 1
		
for i in range(len(S)-1, 0, -1):
	mapping[i-1] += mapping[i]
	
print ' '.join(map(lambda x: str(x), mapping))
