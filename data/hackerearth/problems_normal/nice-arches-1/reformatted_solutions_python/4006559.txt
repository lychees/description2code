count = 0
for i in range(int(raw_input())):
    word = raw_input()
    l = []
    for char in word:
        if (len(l) > 0 and l[-1] == char):
            l.pop()
        else:
            l.append(char)
    if len(l) == 0:
        count = count + 1
print(count)
