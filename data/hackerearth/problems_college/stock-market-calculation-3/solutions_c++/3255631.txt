#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long int n;
		cin>>n;
		long int a[n];
		
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		long int p=n-1;
		long int q=p-1;
	
		long int mxp=0;
		while(1)
		{
			if(a[p]>a[q])
			{
				
				long int pr;
				pr=a[p]-a[q];
				mxp=mxp+pr;
				q--;
			}
			else
			{
				p=q;
				q--;
			}
			
			if(q<0 || p<0)
			{
				break;
			}
		}
		
		cout<<mxp<<"\n";
		
	}
}
