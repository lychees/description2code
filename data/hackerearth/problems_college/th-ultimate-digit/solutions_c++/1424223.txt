#include <iostream>

int powr(int a,long long b,int mod)
{
	if(b==0)
	return 1;
	else if(b%2==0)	
	{
		int x= powr(a,b/2,mod);
		return (x*x)%mod;
	}
	else
	{
		return ((a*(powr(a,b-1,mod)))%mod);
	}
}
