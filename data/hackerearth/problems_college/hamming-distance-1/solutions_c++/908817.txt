#include <iostream>
using namespace std;

int main()
{
    int t,x,y;
    cin>>t;
    while(t--)
    {
    	
    	cin>>x>>y;
    	if(x>y)
    	{
    		int count=0;
    		while(x>0)
    		{
    			if((x&1)!=(y&1))
    			count++;
    			x>>=1;
    			y>>=1;
    			
    		}
    		cout<<count<<endl;
    		
    	}
    	else
    	{
    		
    		int count=0;
    		while(y>0)
    		{
    			if((x&1)!=(y&1))
    			count++;
    			x>>=1;
    			y>>=1;
    		}
    		cout<<count<<endl;
    	}
    }
    return 0;
}
