#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long int d;
    cin>>d;
    for(int t=1;t<=d;t++){
        int g,m,cnt,sum;
    	cnt=sum=0;
    	cin>>g>>m;
    	long int a[g];
    	for(int i=0;i<g;i++)
    		cin>>a[i];
    	sort(a,a+g);
    	for(int i=0;i<g;i++){
    		sum+=a[i];
    		if(sum>m)
    			break;
    		cnt++;
    	}
    	cout<<cnt<<"\n";

    }
    return 0;
}
