#include <iostream>
#include <bits/stdc++.h>
#define ll long long 

using namespace std;

int main()
{
    ll t;
    cin >> t;
    while(t--)
    {
    	ll a, b, x, y, m = 0;
    	cin >> a >> b;
    	for(ll i = b; i >a; i--)
    	{
    		x = i;
    		y = i - 1;
    		m = max(m, (x & y));
    		if(m == x || m == y)
    		break;
    		
    	}
    	cout << m << endl;
    }
    return 0;
}
