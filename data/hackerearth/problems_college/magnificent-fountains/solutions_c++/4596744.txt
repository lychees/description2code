#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
int prime[10001];
ll pwr(int a,int b){
	if(b==0) return 1;
	ll x = pwr(a,b>>1);
	x = (x*x)%mod;
	if(b&1) x = (a*x)%mod;
return x;
}
void pre(){
	for(int i=2;i<=10000;i++) prime[i] = 1;
	for(int i=2;i<=100;i++){
		if(prime[i]){
			for(int j=i*i;j<10001;j+=i){
				prime[j]=0;
			}
		}
	}
}
int main()
{
    pre();
    int t;
    scanf("%d",&t);
    while(t--){
    	int n;scanf("%d",&n);
    	int arr[n];
    	for(int i=0;i<n;i++){
    		scanf("%d",&arr[i]);
    	}
    	ll ans = 1;
    	for(int i=2;i<=10000;i++){
    		if(prime[i]){
    			int mx = 0;
    			for(int j=0;j<n;j++){
    				int cnt = 0;
    				int x = arr[j];
    				while(x%i==0){
    					x = x/i;
    					cnt++;
    				}
    				mx = max(cnt,mx);
    			}
    			ans = (ans*pwr(i,mx))%mod;
    		}
    	}
    	printf("%lld\n",ans);
    }
    return 0;
}
