#include <iostream>
using namespace std;

typedef long long int Lint;

int main()
{
	ios_base::sync_with_stdio( false );
	Lint t, n, i, j, flag, v1, v2;
	cin >> t;
	
	while( t-- ) {
		string s;
		cin >> s;
		n = s.length();
		flag = 1;
		
		for( i = 1, j = n - 2 ;i < n;i++,j-- ) {
			v1 = s[ i ] - s[ i - 1 ];
			v2 = s[ j ] - s[ j + 1 ];
			if( v1 < 0 )
				v1 *= -1;
			if( v2 < 0 )
				v2 *= -1;
			if( v1 != v2 ) {
				flag = 0;
				break;
			}	
		}
		
		if( flag )
			cout << "Funny\n";
		else
			cout << "Not Funny\n";
	}
	
	return 0;
}
