#include <iostream>
#include<cmath>
using namespace std;

long long mul_inv(long long a, long long b)
{
	long long int b0 = b, t, q;
	long long x0 = 0, x1 = 1;
	if (b == 1) return 1;
	while (a > 1) {
		q = a / b;
		t = b, b = a % b, a = t;
		t = x0, x0 = x1 - q * x0, x1 = t;
	}
	if (x1 < 0) x1 += b0;
	return x1;
}

long long pow1(long long n)
{

	if(n==0) return 1;
	long long x= pow1(n/2);
	if(n%2==0)
	return x*x %1000000007;
	else
	return (x*x%1000000007)*2%1000000007;
 
}

int main()
{
	long long n,t;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	n=(n%1000000006);
    	cout<<mul_inv(pow1(n-1),1000000007)<<endl;
    }
    return 0;
}
