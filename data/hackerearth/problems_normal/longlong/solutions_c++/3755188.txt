#include <bits/stdc++.h>
using namespace std;

#define pb push_back
#define pf push_front
#define ff first
#define ss second
#define rz resize

typedef long long ll;
typedef pair<int,int> pii;

string s;
vector<ll> pw,H;
const int ML=1000100;

unordered_set<ll> S;
bool pr(int len){

  S.clear();
  ll X;
  for(int i = len;i<=s.length();i++){
    X = (H[i]-H[i-len])*pw[ML-i+len-1];
    if(S.find(X)!=S.end())
      return true;
    S.insert(X);
  }
  return false;
}

int main(){
  ios::sync_with_stdio(0),cin.tie(0);

  cin>>s;

  int i;

  pw.rz(ML+1);
  pw[0]=1;
  for(i=1;i<=ML;i++)
    pw[i]=(pw[i-1]*173);

  H.rz(ML+1);
  H[0]=0;
  for(i=1;i<=s.length();i++){
    H[i]=(H[i-1]+s[i-1]*pw[i]);
  }

  int lo=0,hi=s.length(),m;

  while(lo<hi){
    m=(lo+hi+1)/2;
    if(pr(m))
      lo=m;
    else
      hi=m-1;
  }

  cout<<lo<<endl;

  return 0;
};
