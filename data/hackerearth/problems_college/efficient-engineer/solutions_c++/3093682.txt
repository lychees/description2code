#include<bits/stdc++.h>
using namespace std;
long long ar[100009];
int main()
{
	long long n,k,d,i,s=0,gs=1000000000000,l=-1,last=-1;
	scanf("%lld%lld%lld",&n,&k,&d);
	for(i=0;i<n;i++)
	{
		scanf("%lld",&ar[i]);
	}
	for(i=0;i<k-1;i++)
	{
		s+=ar[i];
		if(ar[i]!=0)
		last=i;
	}
	int count=0;
	for(i=k-1;i<n;i++)
	{
		s+=ar[i];
		if(s<gs)
		gs=s;
		if(ar[i]!=0)
		last=i;
		if(s<d)
		{
			if(l==-1)
			{
				count++;
				l=last+k-1;
			}
			else if(i-k+1>l)
			{
				count++;
				l=last+k-1;
			}
		}
		s-=ar[i-k+1];
	}
	if(gs>=d)
	printf("0\n");
	else if(gs!=0)
	printf("%d\n",count);
	else
	printf("-1\n");
}
