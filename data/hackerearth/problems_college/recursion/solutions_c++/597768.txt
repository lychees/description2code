/*
Map
Maps are associative containers that store elements formed by a combination of a key value and a mapped value, following a specific order.

In a map, the key values are generally used to sort and uniquely identify the elements, while the mapped values store the content associated to this key
*/
#include<iostream>
#include<stdlib.h>
#include<map>
#include<stdio.h>
using namespace std;
void permutation(int data[],int k,int n,map<int,int> &index)
{
    if(k!=n)
    {
        for(int i=k; i<n; i++)
        {
            int temp=data[k];
            data[k]=data[i];
            data[i]=temp;
            permutation(data,k+1,n,index);
            temp=data[k];//back track
            data[k]=data[i];
            data[i]=temp;
        }
    }
    else if(k==n)
    {
        int num=0;
        for(int i=0; i<n; i++)
            num=num*10+data[i];
        index[num]=1;
    }
}
int main()
{
    int test;
    scanf("%d",&test);
    int A[test];
    for(int t=0; t<test; t++)
    {
        map<int,int> index;
        int n;
        scanf("%d",&n);
        int data[n];
        for(int i=0; i<n; i++)
            data[i]=i+1;
        permutation(data,0,n,index);
        for( map<int,int>::iterator it=index.begin(); it!=index.end(); it++ )
            cout<<it->first<<" ";
        printf("\n");
    }
}
