def gcd(a, b):
    if a%b == 0:
        return b
    else:
        return gcd(b, a%b)

def lcm(a, b):
    return (a*b)/gcd(a,b)

T = int(raw_input())

while T>0:
    n = int(raw_input())
    arr = map(int, raw_input().split())
    ans = arr[0]
    for i in range(1,n):
        ans = lcm(ans,arr[i])
    ans = ans%1000000007
    print ans
