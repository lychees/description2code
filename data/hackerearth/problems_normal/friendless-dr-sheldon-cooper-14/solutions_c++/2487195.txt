#include <iostream>
#include <queue>
using namespace std;
int map[1000][1000];
int visit[1000];
int main()
{
    int t;
    cin>>t;
    while(t--){
    	int nodes,cabDrivers;
    	cin>>nodes;
    	cin>>cabDrivers;
    	int from,to;
    	for(int i=0;i<nodes;i++){
    		visit[i]=0;
    		for(int j=0;j<nodes;j++){
    			map[i][j]=0;
    		}
    	}
    	for(int i=0;i<cabDrivers;i++){
    		cin>>from>>to;
    		map[from-1][to-1]=1;
    	}
    	
    	cout<<nodes-1<<endl;
    }
    return 0;
}
