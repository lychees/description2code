template <class T>T Max2(T a,T b){return a<b?b:a;}
template <class T>T Min2(T a,T b){return a<b?a:b;}
template <class T>T Max3(T a,T b,T c){return Max2(Max2(a,b),c);}
template <class T>T Min3(T a,T b,T c){return Min2(Min2(a,b),c);}
template <class T>T Max4(T a,T b,T c,T d){return Max2(Max2(a,b),Max2(c,d));}
template <class T>T Min4(T a,T b,T c,T d){return Min2(Min2(a,b),Max2(c,d));}

using namespace std;
 
int min(string S, string T)
{
  int count = 0;
  stack<int> s1, s0, Is1, Is0;
 
  int size = S.size();
 
  for(int i=0; i<size; i++)
  {
    if(S[i]=='0' && T[i]=='1')
		s0.push(i);
    else if(S[i]=='1' && T[i]=='0')
		s1.push(i);
    else if(S[i]=='?' && T[i]=='1')
		Is0.push(i);
 
    else if(S[i]=='?' && T[i]=='0')
		Is1.push(i);
  }
 
  while(!s1.empty())
  {
    if(s0.empty() && Is0.empty())
		return -1;
    if(!s0.empty())
	{
      s0.pop();
      count++;
    }
	else
	{
      Is0.pop();
      count+=2;
    }
    s1.pop();
  }
  return  count + s0.size() + Is0.size() + Is1.size();
}
 
int main()
{
  int N;
  cin >> N;
  for(int k=1; k<=N; k++)
  {
    string S,T;
    cin >> S >> T;
 
    printf("Case %d: %d\n",k,min(S,T));
  }
 
  return 0;
}
