for i in xrange(t):
    flag = 0
    s = raw_input()
    b = ""
    for i in xrange(len(s)):
        if (s[i] == 'a' or s[i] == 'e' or s[i] == 'i' or s[i] == 'o' or s[i] ==
                'u' or s[i] == 'A' or s[i] == 'E' or s[i] == 'I' or s[i] == 'O' or s[i] == 'U'):
            b += s[i]
            flag = 1
    if flag == 1:
        print b
