#include <iostream>
#include <string>
using namespace std;

int main()
{

    cin>>l;
    string st[l];
    int r=0;
    for(i=0;i<l;i++)
    	cin>>st[i];
    int flag=0;
    for(i=0;i<l;i++){
    	flag=0;
    	for(j=0;j<k;j++){
    		if(st[i]!=s[j])
    			flag++;
    	}
    	if(flag==k){
    		if(r!=0)
    			cout<<".";

    		st[i][0]=toupper(st[i][0]);
    		cout<<st[i][0];
    	}
    	
    }
    
    
    return 0;
}
