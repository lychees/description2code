#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007

long long int power(long long int n, long long int k)
{
  long long int ans, powi;
  if(k==0)
    return 1;
  if(n==0)
    return 0;
  if(n==1)
    return 1;
  ans=n%MOD;
  powi=1;
  while(powi*2<=k)
  {
    powi=powi*2;
    ans=(ans*ans);
    ans=ans%MOD;
  }

  if(k-powi!=0)
  ans=ans*power(n, k-powi);
  ans=ans%MOD;

  return ans;
}

int main() {
  int tt;
  cin>>tt;
  while(tt--)
    {
    int n;
    cin>>n;
    long long ans=0,temp;
    for(int i=0;i<n;i++)
      {
      cin>>temp;
      ans|=temp;
      }
    ans*=power(2,n-1);
    ans%=MOD;
    cout<<ans<<"\n";
   }
    return 0;
}
