#include<bits/stdc++.h>

#define mod 1000000007
#define i_long long long
i_long gcd(i_long a,i_long b);
int main()
{
int t,n;
scanf("%d",&t);
//cin>>t;
while(t--)
{
scanf("%d",&n);
//cin>>n;
i_long arr[n];

for(int i=0;i<n;i++)
{
	scanf("%lld",&arr[i]);
//cin>>arr[i];
}
i_long ans=arr[0];
for(int i=1;i<n;i++)
{
	ans=gcd(ans,arr[i]);
	}

}

}
i_long gcd(i_long a,i_long b)
{
	i_long t;
	while(b!=0)
	{
		t=b;
		b=a%b;
		a=t;
		}
		return a;
	}
