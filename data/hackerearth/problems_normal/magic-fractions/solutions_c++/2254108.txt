#include<bits/stdc++.h>
using namespace std;

#define MAX 502
int arr[MAX];
int siz;
vector<int>primes;

void PREPROCESS()
{
	int i , j;
	arr[0] = 1 , arr[1] = 1;

	for(i = 2 ; i < MAX ; i++)
	{
		if(arr[i] == 0)
		{
			primes.push_back(i);

			for(j = i ; j*i < MAX ; j++)
			{
				arr[i*j] = 1;
			}
		}
	}
	siz = primes.size();
}

int getans(int num)
{
    int no_fact = 0;

    for(int i = 0 ; i < siz && primes[i] <= num ; i++)
    {
        if(num / primes[i] >= 1)
            no_fact++;
    }
        return no_fact;

}

int main()
{
	PREPROCESS();

	int n;
	cin>>n;

	long long ans = 0;

	for(int i = 1 ; i <= n ; i++)
    {
        ans += (long long)pow(2,getans(i)-1);
    }

    cout<<ans<<endl;
}
