#include<bits/stdc++.h>
#define  den 1000001
typedef long long int lu;
using namespace std;
int main()
{
	lu i,j,x,ans,t,z,y,yans;
	vector<int>ar(1000001);
	lu br[den]={0};
	ar[0]=ar[1]=1;
	for(i=2;i<=den-1;i++)
	{
		if(ar[i]==0)
		{
			j=i*2;
			x=3;
			for(j=i*2;j<den;)
			{
				ar[j]=1;
			j=i*x;
			x++;}
		}
	}
	br[0]=br[1]=0;
	ans=0;
	yans=0;
	for(i=2;i<den;i++)
	{
			if(ar[i]==0)
			ans++;
			if(ar[ans]==0)
			{
				yans++;
				br[i]=yans;
			}
			else br[i]=yans;
		
	}
	scanf("%lld",&t);
	while(t--)
	{
		ans=0;
		scanf("%lld%lld",&y,&z);
		ans=br[z]-br[y-1];
		printf("%lld\n",ans);
		
	}
	return 0;
}
