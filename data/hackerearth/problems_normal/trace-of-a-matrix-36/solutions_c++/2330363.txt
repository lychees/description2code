#include <iostream>
using namespace std;

int main()
{
    long long int sum1=0,sum2=0,r,c;
    int t;
    
    cin>>t;
    while(t--)
    {
    	cin>>r>>c;
    	sum1=min(r,c)*(2+(min(r,c)-1)*(r+1))/2;
    	sum2=min(r,c)*(2+(min(r,c)-1)*(c+1))/2;
    	cout<<sum1+sum2<<endl;
    }
    
    return 0;
}
