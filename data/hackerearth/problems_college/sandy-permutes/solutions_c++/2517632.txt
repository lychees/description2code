#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
unsigned long long int n,k,i,j,index,max,temp;
cin>>n>>k;
unsigned long long int arr[n];
for(i=0;i<n;i++)
cin>>arr[i];
for(i=0;i<k&&i<n;i++)
{
    index=i;
    max=arr[i];
    for(j=i+1;j<n;j++)
    {
        if(arr[j]>max){
            index=j;
            max=arr[j];
        }
    }
    if(index==i&&n>100)
        k++;
        temp=arr[index];
        arr[index]=arr[i];
        arr[i]=temp;
}
for(i=0;i<n;i++)
    cout<<arr[i]<<" ";
    return 0;
}
