#include<iostream>

using namespace std;
int main()
{
	int q,t,c,i = 0;
	int a[100000];
	cin>>q;
	a[0] = 0;
	while(q--)
	{
		cin>>t;
		if(t == 1)
		{
			if(a[i] == 0)
				cout<<"No Food\n";
			else{
				cout<<a[i]<<"\n";
				i--;
			}
		}
		if(t == 2)
		{
			cin>>c;
			i++;
			a[i] = c;
		}			
	}
	
	return 0;
}
