#include <iostream>
#include<queue>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n,m,adj[301][301],i,j,x,y,col[301],flag=0;
    	cin>>n>>m;
    	for(i=0;i<=300;i++)
    	{col[i]=-1;
    	for(j=0;j<=300;j++)
    	adj[i][j]=0;
    	}
    	for(i=0;i<m;i++)
    	{
    		cin>>x>>y;
    		adj[x][y]=1;
    		adj[y][x]=1;
    	}
    	queue<int > q;
    	q.push(0);
    	col[0]=0;
    	while(!q.empty())
    	{
    		int v=q.front();
    		q.pop();
    		for(i=0;i<n;i++)
    		if(adj[v][i]==1)
    		{
    			if(col[i]==-1)
    			{col[i]=col[v]==1?0:1;
    			 q.push(i);
    			}
    			else
    			if(col[i]==col[v])
    			{
    				flag=1;
    				break;
    			}
    		}
    		
    	}
    	if(flag==1)
    	cout<<"NOT BICOLORABLE.\n";
    	else
    	cout<<"BICOLORABLE.\n";
    }
    return 0;
}
