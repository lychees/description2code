#include"bits/stdc++.h"
using namespace std;
long int func1(long int *arr,long int n,long int value)
{
	long int ans=0;
	long int left=0,right=n+1;
	while(left<=right)
	{
		long int mid=(left+right)/2;
		if(arr[mid]<value)
		{
			ans=mid;
			left=mid+1;
		}
		else if(arr[mid] >= value)
		{
			right=mid-1;
		}
	}
	return ans;
}
long int func2(long int *arr,long int n,long int value)
{
	long int ans=n+1;
	long int left=0,right=n+1;
	while(left<=right)
	{
		long int mid=(left+right)/2;
		if(arr[mid]>value)
		{
			ans=mid;
			right=mid-1;
		}
		else if(arr[mid]<=value)
		{
			left=mid+1;
		}
	}
	return ans;
}
int main()
{
	long int n;
	cin>>n;
	long int arr[n+2];
	for(long int i=0;i<n;i++)
		cin>>arr[i];
	arr[n]=INT_MAX;
	arr[n+1]=INT_MIN;
	sort(arr,arr+n+2);
	
	long int q;
	cin>>q;
	while(q--)
	{
		long int l,r;
		cin>>l>>r;
		
		long int left=func1(arr,n,l);
		long int right=func2(arr,n,r);
		
		cout<<right-left-1<<endl;
	}
	return 0;
}
