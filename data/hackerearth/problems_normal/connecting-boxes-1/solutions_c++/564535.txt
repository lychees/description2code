#include<bits/stdc++.h>
using namespace std;


int main()
{
    int i,j,k,l,m,n,t,fl,fl1;
    
    scanf("%d%d",&n,&t);
    
    
    int a[n+2];
    for(i=0;i<n+1;i++)
        a[i]=0;

    while(t--)
    {   scanf("%d%d",&l,&m);
       // cin>>l>>m;
        if(l==m || a[l]!=0 || a[m]!=0 )
        {cout<<"NO"<<endl;
        continue;
        }
        
        if(l>m)
        {
            fl1=l;
            l=m;
            m=fl1;
        }
        
        
        
        if(m-l==1)
      {cout<<"YES"<<endl;
       a[l]=1 ;
       a[m]=-1;
        continue;
        }
        
        fl=0;
        for(i=l;i<m;i++)
        {
            if(fl==0 && a[i]==-1)
            {   fl=1;
                //printf("NO\n");
                break;
            }
            if(a[i]==1)
                fl++;
            if(a[i]==-1)
                fl--;


        }
        if(fl==0 && i==m)
         {
            cout<<"YES"<<endl;
       a[l]=1 ;
       a[m]=-1;
         }
         else
            cout<<"NO"<<endl;
    }







    return 0;
}
