#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n];
    int i,j;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    sort(a,a+n);
    int in = 0;
    int count = 0;
    int temp[n];
    temp[0]=1;
    for(i=1;i<n;i++)
    {
    	temp[i]=0;
    }
    for(i=0;i<n-1;)
    {
      		in = a[i];
	    	for(j=i+1;j<n;j++)
	    	{
	    		if(a[j]<=in+4)
	    		{
	    			temp[j]=-1;
	    			i++;
	    		}
	    		else
	    		{
	    			temp[j]=1;
	    			i++;
	    			break;
	    		}
	    	}
    }
    if(temp[n-1]==0)
    {
    	temp[n-1]==1;
    }
    for(i=0;i<n;i++)
    {
    	if(temp[i]==1)
    		count++;
    }
    cout<<count<<endl;
   
}
