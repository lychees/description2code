#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t, h[100], m[100];
    cin>>t;
    
    for(int i=0; i<t; i++)
    {
    	cin>>h[i]>>m[i];
    }
    
    for(int i=0; i<t; i++)
    {
    	int x, y, diff;
    	
    	x=m[i]*6;
    	//cout<<x;
    	y=h[i]*5*6 + (0.5*m[i]);
    	//cout<<y;
    	diff=abs(y-x);
    	
    	cout<<min(diff, (360-diff))<<'\n';
    }
    return 0;
}
