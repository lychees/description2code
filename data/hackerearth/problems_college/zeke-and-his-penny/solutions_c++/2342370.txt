#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define MOD 1000000007

int main()
{
	int T;
	
	cin>>T;
	
	while(T--)
	{
		ll A[5];
		
		for(int i=1;i<=3;i++)
		{
			cin>>A[i];
		}
		
		sort(A+1,A+4);
		
		if((A[3]-2) <=0 || (A[2]-1)<=0 || A[1]<=0)
		cout<<"0"<<endl;
		else
		{
			ll ans=A[1];
			
			if(ans>=MOD)
			ans%=MOD;
			
			ans*=(A[2]-1);
			
			if(ans>=MOD)
			ans%=MOD;
			
			ans*=(A[3]-2);
			
			if(ans>=MOD)
			ans%=MOD;
			
			cout<<ans<<endl;
		}
		
	}
    return 0;
}
