#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    long n,t,i;
    cin>>t;
    while(t--)
    {
        cin>>n;
        long a[n];
        for(i=0;i<n;i++)
            cin>>a[i];
        sort(a,a+n);
        long j=1,c=0;
        i=0;
        while(i<n && j<n)
        {
            if(a[i]==a[j])
            {
                c++;
                j++;
            }
            else
            {

                j++;
                i=j-1;
            }
        }
        cout<<c<<"\n";
    }
    return 0;
}
