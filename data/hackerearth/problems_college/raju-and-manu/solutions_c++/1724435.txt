#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
     {
     	int n;
     	cin>>n;
     	int temp=n;
     	int count=0,c=0,ans=0;
     	while(temp>0)
     	 {
     	 	if(temp%2!=0)
     	 	 c++;
     	 	temp=temp/2;
     	 	count++;
     	 }
     	 if(((n)&(n+1))==0)
          {	
           cout<<n<<"\n";
     	   continue;
           }
         else
         {
         	
         for(int j=count-1;j>=0;j--)
     	 {int temp=0;
     	 for(int i=count-1;i>=0;i--)
     	  {
     	  	if(i!=j)
     	  	temp+=pow(2,i);
     	  }
     	  if(temp<=n)
     	    ans=max(ans,temp);
         }
     	  cout<<ans<<"\n";
         }
     }
}
