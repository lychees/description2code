#include<iostream>
#include<stdio.h>
#include<limits.h>
#include<algorithm>
using namespace std;
long long int binary_search(long long int a[3000],long long int l,long long int h,long long int s)
{
	long long int m=(l+h)/2;
	if(l==h)
	return l;
	if(a[m]>s&&a[m-1]<=s)
	return m-1;
	else if(a[m]<=s&&a[m+1]>s)
	return m;
	else if(a[m]>s)
	return binary_search(a,l,m-1,s);
	else
	return binary_search(a,m+1,h,s);
}
int main()
{
	long long int a[3000],i,n,k,val,ans,j,search;
	scanf("%lld %lld",&n,&k);
	for(i=0;i<n;i++)
	scanf("%lld",&a[i]);
	sort(a,a+n);
	a[n]=INT_MAX;
	ans=0;
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			search=k/(a[i]*a[j]);
			//printf("%lld ",search);
			if(search<a[0])
			continue;
			else if(search>=a[n-1])
			ans=ans+n-2;
			else
			{
				val=binary_search(a,0,n-1,search);
				if(val>=j)
				ans=ans+val-1;
				else if(val>=i)
				ans=ans+val;
				else
				ans=ans+val+1;
			}
		}
	}
	printf("%lld",ans/3);
	return 0;
}
