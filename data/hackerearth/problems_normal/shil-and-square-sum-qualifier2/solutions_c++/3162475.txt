typedef long long ll;
const ll MOD = 1e9+7;
const int INF = 0x7f7f7f7f;
const double eps = 1e-6;

const int maxn = 1000010;
ll arr[maxn];

int main() {
	ll n,k; scanf("%lld%lld",&n,&k);
	for (int i = 1; i <= n; i ++) scanf("%lld",arr + i);

	ll x,y,xx;
	x = y = xx = 0;

	for (ll i = 1; i <= k; i ++) {
		x = (x + ((i * i) % MOD) * arr[i] + MOD) % MOD;
		y = (y + (i + i - 1) * arr[i] + MOD) % MOD;
		xx = (xx + arr[i] + MOD) % MOD;
	}

	printf("%lld ",x);

	for (ll i = k + 1; i <= n; i ++) {
		x -= y;
		x = (x + MOD) % MOD;
		x = (x + (((k * k) % MOD) * arr[i]) % MOD + MOD) % MOD;

		xx -= arr[i - k];
		
		y -= xx * 2 + arr[i - k];
		y = (y + ((k + k - 1) * arr[i]) % MOD + MOD) % MOD;

		xx += arr[i];

		x = (x + MOD) % MOD;
		y = (y + MOD) % MOD;
		xx = (xx + MOD) % MOD;

		printf("%lld ",x);
	}

	printf("\n");
	return 0;
}
