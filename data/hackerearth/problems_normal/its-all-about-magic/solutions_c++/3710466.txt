#include <iostream>
using namespace std;


int main()
{
	int noofcases=0;
	cin>>noofcases;
	for(int i=0;i<noofcases;i++)
	{
		int M=0;
		long long int countofzero=0;
		long long int countofone=1;
		long long int temp = 0;
		cin >>M;
		
		for(int j=1;j<=M;j++)
		{
			if(j==1)
			{
				countofzero=1;
				countofone=0;
			}else if(j==2)
			{
				countofzero=1;
				countofone=1;
			}else
			{
				temp = countofzero;
				countofzero = countofzero + countofone;
				countofone = temp;
			}

		}
		

		cout << countofone <<" "<< countofzero << endl;
	}
    
    return 0;
}
