#include <iostream>
using namespace std;

int main()
{
    int n;
    int x;
    int i = 2;
    cin >> n;
    cin >> x;
    cout << x;
    long int sum = x;
    n -= 1;
    while(n--){
    	cin >> x;
    	int y;
    	y = (x * i) - sum;
    	cout << " " << y;
    	sum += y;
    	i++;
    }
    return 0;
}
