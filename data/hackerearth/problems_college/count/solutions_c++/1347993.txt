#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while (t--)
    {
    	int r, c;
    	cin >> r >> c;
    	
    	int even, odd;
    	if (r <= c)
    	{
    		if (r % 2 == 0)
			{
				even = (r * (2 + r))/4;
				odd = (r * r)/4;
			}
			else
			{
				even = (r * r)/4;
				odd = ((r+1) * (1+r))/4;
			}
    	}
    	else
    	{
    		int a;
    		a = r - c;
    		if (c%2==0)
    		{
    			even = (c * (2+c))/4;    		
    			odd = (c * c)/4;    			
    			if (a % 2 == 0)
    			{
    				even += a/2 * c;
    				odd += a/2 * c;
    			}
    			else
    			{
    				even += a/2 * c;
    				odd += (a+1)/2 * c;
    			}
    		}
    		else
    		{
    			even = (c * c)/4;
    			odd = ((c+1) * (1+c))/4;
    			if (a%2==0)
    			{
    				even += a/2 * c;
    				odd += a/2 * c;
    			}
    			else
    			{
    				even += (a+1)/2 * c;
    				odd += a/2 * c;
    			}
    		}
    		
    	}
    	cout << even << " " << odd << "\n";
    }
    
    return 0;
}
