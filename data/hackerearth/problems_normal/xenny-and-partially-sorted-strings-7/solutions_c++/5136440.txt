#include <iostream>
using namespace std;
int main()
{
int t,n,m,k;
string temp;
cin>>t;
string arr[1000];
string copy[1000];
while(t>0)
{
cin>>n>>k>>m;
for(int i=0;i<n;i++)
{
cin>>arr[i];
}
for(int i=0;i<n;i++)
{
copy[i]=arr[i].substr(0,m);
}
for(int i=0;i<n-1;i++)
{
for(int j=i+1;j<n;j++)
{
if(copy[i]>copy[j])
{
temp=copy[i];
copy[i]=copy[j];
copy[j]=temp;
temp="";
temp=arr[i];
arr[i]=arr[j];
arr[j]=temp;
}
}
}
cout<<arr[k-1]<<endl;
t--;
}
return 0;
}
