#include<iostream>
using namespace std;
int main()
{
    int x;
    cin>>x;
    while(x)
    {
        int d;
        cin>>d;
        char str[50];
        cin>>str;
        for(int i=0; str[i]!=0;i++)
        {
            str[i]=str[i]-d;
            if(str[i]<65)
            {
                str[i]=str[i]+26;
            }
        }
        cout<<str<<endl;x--;
    }
}
