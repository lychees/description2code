#include<iostream>
#include<cstdio>
#include <cmath>
#define gc getchar_unlocked
using namespace std;
void inp(long long &x)
{
    register long long c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
int main(){
	long long t,b,e,i,l;
	ios_base::sync_with_stdio(false);
	inp(t);
	while(t--){
		inp(b);
		inp(e);
		l = b % 10;
		i = e % 4;
		if(i == 0)
			i = 4;
		printf("%d\n",(int)(pow(l, i)) % 10);
	}
}
