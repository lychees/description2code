#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
	int i,j,k,t,wl,ol=0,f=0;
	char o[100][31]={' '};
	string w[100];
	cin>>t;
	for(i=0;i<=t;i++)
	{
		cin>>w[i];
	}
	for(i=0;i<=t;i++)
	{	
		wl=w[i].length();
		o[i][0]=w[i][0];ol=1;
		for(j=0;j<wl;j++)
		{	f=0;
			for(k=0;k<ol;k++)
				if(w[i][j]==o[i][k])
				{
					f++;
				}
				
			if(f==0)
			{		o[i][ol]=w[i][j];
					ol++;
			}
		}	
		cout<<o[i]<<"\n";
		
	}
    return 0;
}
