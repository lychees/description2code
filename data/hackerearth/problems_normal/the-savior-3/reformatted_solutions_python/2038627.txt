t = input()
for _ in xrange(t):
    n = input()
    num = [int(i) for i in raw_input().split()]
    ev = 0
    od = 0
    mx = (10**5)+1
    hs = [0]*(mx)
    for i in num:
        if (i % 2 == 0):
            ev += 1
        else:
            od += 1
        hs[i] += 1
    now = 0
    num = list(set(num))
    for i in num:
        if (i % 2 == 0):
            now += hs[i]*(ev-hs[i])
        else:
            now += hs[i]*(od-hs[i])
    print now >> 1
