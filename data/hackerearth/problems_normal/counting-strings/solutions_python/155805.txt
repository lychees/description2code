for t in xrange(int(raw_input())):
	m = raw_input()
	pi,i = -1,0
	l = len(m)
	ret = 0
	while i<l:
		if m[i]=='a' or m[i]=='z':
			ret += (l - i) * (i-pi)
			pi = i
		i+=1
	print ret
