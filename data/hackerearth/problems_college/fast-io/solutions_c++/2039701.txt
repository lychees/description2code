#include<iostream>
using namespace std;
void fastscan(int &x)
{
bool neg=false;
register int c;
x =0;
c=getchar();
if(c=='-')
{
neg = true;
c=getchar();
}
for(;(c>47 && c<58);c=getchar())
x = (x<<1) + (x<<3) +c -48;
if(neg)
x *=-1;
}
int main(){
	ios_base::sync_with_stdio(false);
	 int n,c=0,num;
	fastscan(n);
	while(n--){
		fastscan(num);
		if(num<300000000)
		c++;
		}
		cout<<c<<endl;

}
