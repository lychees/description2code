#include <stdio.h>
 #include<iostream>
 using namespace std;
int josephus(int n, int k)
{
  if (n == 1)
    return 1;
  else
    
    return (josephus(n - 1, k) + k-1) % n + 1;
}
 
// Driver Program to test above function
int main()
{
  int t,n;
  cin>>t>>n;
  int s=josephus(t,n);
  cout<<s<<"\n";
  return 0;
}
