#include<bits/stdc++.h>
using namespace std;

int main() {
  int tt;
  scanf("%d",&tt);
  while(tt--) {
    int n,d;
    scanf("%d%d",&n,&d);
    for(int i=0;i<=d;i+=2) {
      if(i&&(i%6==0)) {
        n+=2;
      }
      if(i<d&&n==0) {
        puts("No");
        goto me;
      }
      n-=1;
    }
    puts("Yes");
    me:;
  }
  return 0;
}
