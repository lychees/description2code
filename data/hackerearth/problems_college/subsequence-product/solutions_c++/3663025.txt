#include <iostream>
using namespace std;

int main()
{
    long long int t,i,j,h,n,prod;
    cin>>t;
    while(t--)
    {
    	prod=1;
    	cin>>n>>h;
    	long int a[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		prod=(prod%h)*(a[i]);
    		
    	}
    	prod=prod%h;
    	if(prod==0)
    	cout<<"YES";
    	else
    	cout<<"NO";
    	cout<<endl;
    }
    return 0;
}
