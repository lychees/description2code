#include<iostream>
#include<algorithm>
using namespace std;
typedef long long ll;
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		ll n,flag=0;
		cin>>n;
		ll ar[n],sum[n];
		for(int i=0;i<n;i++)
		{
			cin>>ar[i];
			if(i!=0)
			{
				if(ar[i]%ar[i-1]==0)
					ar[i-1]=(ar[i]/ar[i-1])-1;
				else ar[i-1]=ar[i]/ar[i-1];
			}
		}
		sort(ar,ar+(n-1));
		for(int i=0;i<n-1;i++)
			if(ar[i]<i+1)
			{
				cout<<"NO\n";
				flag=1;
				break;
			}
		if(!flag)
		 cout<<"YES\n";
	}
}
