#include <iostream>
using namespace std;

int main()
{
    int t,l;
    int n,pcount,ncount;
    cin>>t;
    while(t--)
    {
    	pcount=ncount=0;
    	cin>>l;
    	while(l--)
    	{
    		cin>>n;
    		if(n==-1)
    		ncount++;
    		else
    		pcount++;
    		
    	}
    	if(ncount >= pcount)
    	{
    		cout<<ncount<<"\n";
    	}
    	else
    	{
    		cout<<ncount+1<<"\n";
    	}
    }
    return 0;
}
