#include<bits/stdc++.h>

using namespace std;

int main()
{
	int t,n,i,j,k;
	cin >> t;
	while(t--)
	{
		cin >> n;
		i = 1;
		j = i*(i+1)/2;
		while(j<=n)
		{
			i++;
			j = i*(i+1)/2;
		}
		i--;
		j = i*(i+1)/2;
		if(j > n)
		{
			cout << i +1<<endl;
		}
		else if(j == n || (j-1) != n)
		{
			cout << i << endl;
		}
		else
		{
			cout << i - 1<<endl;
		}
		
	}
	return 0;
}
