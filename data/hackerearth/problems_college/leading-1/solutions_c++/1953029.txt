#include<iostream>
#include<stdlib.h>
#include<math.h>
using namespace std;
int fact(int n){
	if(n==0 || n==1){
		return 1;
	}
	else{
		return(n*fact(n-1));
	}
}
int main(){
	long long int t;
	cin>>t;
	while(t--){
		

	long long int n0,n1,total;
	cin>>n0>>n1;
	total=fact(n0+n1-1)/(fact(n0)*fact(n1-1));
	cout<<total<<"\n";
	

	
}
	return 0;
}
