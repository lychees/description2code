
#include<iostream>
using namespace std;
int main()
{ long long int n,k,a[10000],i,j,s;
 cin>>n;
 cin>>k;
 for(i=0;i<n;i++)
 cin>>a[i];
 for(i=0;i<n;i++)
{ int temp=0;
 for(j=i+1;j<n;j++)
 { if(a[i]<=a[j])
    { temp=a[i];
     a[i]=a[j];
     a[j]=temp;
     } 
  }    
}  
 s=0;
if(n==k)
{    for(i=0;i<n;i++)
      s+=a[i];  
}
 else
{ s=0;
    int x=n/k;
 if(n%k==0) x=x-1; 
 for(j=1;j<=x;j++)
      { for(i=k*(j-1);i<k*j;i++)
          s+=j*a[i];
        
     }
 
 for(i=x*k;i<n;i++)
     s+=j*a[i];
     
 }     
    
    
  cout<<s;  
   
   return 0;
}
