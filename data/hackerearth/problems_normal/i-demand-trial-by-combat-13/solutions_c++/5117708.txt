#include <iostream>
using namespace std;

int main()
{
    int tc,size,times,i,z;
    cin >> tc;
    while(tc--)
    {
        cin >> size >> times;
        bool arr[size];
        bool tmp[size];
        for(i=0 ; i<size ; i++)
        {
            cin >>  z;
            if(z == 1)
                arr[i] = true;
            else
                arr[i] = false;
        }
        bool flag = true;
        while(times--)
        {
            tmp[0] = arr[1];
            tmp[size-1] = arr[size-2];
            for(i = 1 ; i<size-1 ;i++)
            {
                if(arr[i-1] == 1 && arr[i+1] == 1)
                {
                    tmp[i] = true;
                }
                else
                {
                    tmp[i] = false;
                }
            }
            flag = true;
            for(i=0;i<size;i++)
            {
                if(tmp[i] != arr[i])
                {
                    arr[i] = tmp[i];
                    flag = false;
                }
            }
            if(flag)
                break;
        }
        for(i=0;i<size; i++)
        {
            cout << arr[i] << " ";
        }
        cout << endl;
    }
    return 0;
}
