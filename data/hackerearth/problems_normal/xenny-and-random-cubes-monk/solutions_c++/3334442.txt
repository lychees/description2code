#include <bits/stdc++.h>
using namespace std;
#define ll long long
const ll MOD = 1e9+7;
int arr[11][30];
string str;
int n,k;
ll find(int si, int mask) {
	if(si == k) return 1;
	ll ans = 0;
	for(int i = 0; i <= n; i++) 
		if(!(mask & (1<<i)) && arr[i][str[si]-'a']) 
			ans = (ans + (arr[i][str[si]-'a'] * find(si+1 , mask|(1<<i)))%MOD)%MOD;
	return ans;
}
int main()
{
	memset(arr, 0 , sizeof(arr));
	scanf("%d%d", &n,&k);
	char c;
	for(int i = 0; i < n; i++) {
		for(int j = 0; j < 6; j++) {
			cin >> c;
			arr[i][((int)c-'a')]++;
		}
	}
	cin >> str;
	printf("%lld\n", find(0,0));
return 0;
}
