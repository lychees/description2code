#include<bits/stdc++.h>
using namespace std;
bool cube(int n)
{
	int r = round(cbrt(n));
	return n==r*r*r;
}
int temp[10000000]={0};
int main()
{
	int i;
	for(i=1;i<=100;i++)
	{
		temp[i*i*i]=1;
	}
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int a[n];
		int j,k;
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		int d=0;
		int c=0;
		for(i=0;i<n;i++)
		{
			for(j=i+1;j<n;j++)
			{
				for(k=j+1;k<n;k++)
				{
					d++;
					if(temp[a[i]+a[j]+a[k]])
		            {
		            	c++;
	             	}
				}
			}
		}
		double p = (double)c/d;
	printf("%.9lf\n",p);
	}
}
