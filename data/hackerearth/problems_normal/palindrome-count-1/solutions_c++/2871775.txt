#include <iostream>
using namespace std;

bool isPalindrome(string s) {
	for (int i = 0; i < s.size() / 2; i++) {
		if (s[i] != s[s.size() - i - 1]) return false;
	}
	return true;
}

int numberofPalindrome(string s) {
	int start = 0, end = 0;
	int n = s.size(), res = 0;
	for (int i = 0; i < n; i++) {
		for (int j = i; j < n; j++) {
			string curr = s.substr(i, j - i + 1);
			if (isPalindrome(curr)) {
				res++;
			}
		}
	}
	return res;
}

int main()
{
	string s;
	cin >> s;
	
    cout << numberofPalindrome(s) << endl;
    return 0;
}
