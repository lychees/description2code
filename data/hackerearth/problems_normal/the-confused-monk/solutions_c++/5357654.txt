#include <cstdio>
using namespace std;

long long gcd(long long a, long long b)
{
	return b ? gcd(b, a % b) : a;
}

long long pPow(long long x, long long n, long long m)
{
	long long P;
	while(!(n & 1))
	{
		x = (x * x) % m;
		n >>= 1;
	}
	P = x;
	n >>= 1;
	while(n)
	{
		x = (x * x) % m;
		if(n & 1)
			P = (P * x) % m;
		n >>= 1;
	}
	return P;
}


void patMain()
{
	const long long quotient = 1000000007;
	long long N, f, g;

	scanf("%lld", &N);
	scanf("%lld", &f);
	g = f;
	for(int i = 1; i < N; i++)
	{
		long long x;
		scanf("%lld", &x);
		g = gcd(g, x) % quotient;
		f = (f * x) % quotient;
	}

	printf("%lld\n", pPow(f, g, quotient));
}

int main()
{
	patMain();
	return 0;
}
