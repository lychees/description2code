#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	long long n;
    	cin>>n;
    	long long a[n+1],i,j;
    	for(i=1;i<=n;i++)
    	cin>>a[i];
    	long long cont=0;
    	if(a[n]==0)
    	{
    		cout<<n*(n-1)/2<<" "<<"0"<<endl;
    	}
    	else{
    	for(i=1;i<=n;i++)
    	{
    		j=i;
    		while(a[j]!=0)
    		{
    			cont++;
    			j=a[j];
    		}
    	}
    	long long x=(n*(n-1))/2-cont;
    	cout<<cont<<" "<<x<<endl;
    	}
    }
    return 0;
}
