#include <iostream>
#include <math.h>
using namespace std;

int isPrime(long long int n){
    int i, limit;
	if(n==1 || n==0)
	return 0;
	if(n>1000000000)
	limit=70800;
	else limit=sqrt(n);
	for(i=2;i<=limit;i++){
		if(n%i==0)
			return 0;
	}
	return 1;
}
int main()
{
    long long int t, n;
    cin>>t;
    while(t--){
    	cin>>n;
    	if(isPrime(n))
    		cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
    return 0;
}
