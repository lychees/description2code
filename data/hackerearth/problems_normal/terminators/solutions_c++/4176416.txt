#include <bits/stdc++.h>
using namespace std;
typedef int ll;

ll find_lcs(ll m,ll n,vector<ll> &v1,vector<ll> &v2,vector<vector<ll> > &lcs)
{
	
	for(ll i=1;i<=m+1;i++)
	{
		for(ll j=1;j<=n+1;j++)
		{
			if(i==1 ||j==1)
				lcs[i][j]=0;
			else if(v1[i-1]==v2[j-1])
				lcs[i][j]=lcs[i-1][j-1]+1;
			else
				lcs[i][j]=max(lcs[i-1][j],lcs[i][j-1]);
		
		}
	}
	return lcs[m+1][n+1];
}
int main()
{
	int test;
	cin>>test;
	while(test--)
	{
		int term;
		cin>>term;
		vector<ll> temp1(term+1,0);
		vector<ll> temp2(term+1,0);
		vector<vector<ll> > lcs(term+2,vector<ll>(term+2,-1));
		for(ll i=1;i<=term;i++)
		{
			ll n1,n2;
			cin>>n1>>n2;
			temp1[n1]=i;
			temp2[n2]=i;
		}
		ll len=find_lcs(term,term,temp1,temp2,lcs);
		cout<<(len)<<endl;
		
	}
	return 0;
}
