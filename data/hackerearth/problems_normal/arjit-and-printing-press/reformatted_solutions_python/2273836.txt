t = input()
for _ in range(t):
    a = raw_input()
    b = raw_input()
    b = ''.join(sorted(b))
    a = list(a)
    b = list(b)
    j = 0
    for i in range(len(a)):
        if a[i] > b[j]:
            a[i] = b[j]
            j += 1
            i = 0
        if j >= len(b):
            break
    print "".join(a)
