#include <bits/stdc++.h>
using namespace std;
main()
{
    int a; scanf("%d", &a);
    for (int g=0;g<a; g++)
    {
        int b; scanf("%d", &b); int gcd=0; 
        for (int y=0;y<b; y++){int c;scanf("%d", &c);
        if (y==0)gcd=c;
        else gcd=__gcd(gcd, c);
        }
        printf("%lld\n", 1LL*b*gcd); 
        
    }
    return 0; 
}
