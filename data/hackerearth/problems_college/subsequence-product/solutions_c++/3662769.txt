#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	int a[10001],i,n,k,h;
	while(t--)
	{
		scanf("%d %d",&n,&h);
		for(i=0;i<n;i++)
		scanf("%d",&a[i]);
		for(i=0;i<n;i++)
		{
			k=__gcd(h,a[i]);
			h = h/k;
		}
		if(h==1)
		printf("YES\n");
		else
		printf("NO\n");
	}
	return 0;
}
