#include<cstdio>
#include<queue>
using namespace std;
int main()
{
int c,p,n;

scanf("%d%d%d",&c,&p,&n);
vector<pair<int,int> >temper;
priority_queue<pair<int,int> > q;
int arr[c];
int studs[c];
int monkiq[p];
for(int i=0;i<n;i++)
{
scanf("%d",&arr[i]);
q.push(make_pair(arr[i]*-1,i));
studs[i]=1;
}
for(int i=n;i<c;i++)
{
arr[i]=0;
studs[i]=0;
q.push(make_pair(arr[i],i));
}


for(int i=0;i<p;i++)
{
temper.clear();
scanf("%d",&monkiq[i]);
int y=(q.top().second);
int z=q.top().first;
int min=y;
temper.push_back(make_pair(y,z));
q.pop();

while(q.top().first==z)
{
int temp=q.top().second;
if(temp<min)
{

min=temp;

}
temper.push_back(make_pair(temp,z));
q.pop();

}


studs[min]=studs[min]+1;
int x=(arr[min]+monkiq[i])*(studs[min]);

arr[min]=monkiq[i];


q.push(make_pair(x*-1,min));
printf("%d ",min+1);
while(!temper.empty())
{

if(temper.back().first==min)
temper.pop_back();
else
{q.push(make_pair(temper.back().second,temper.back().first));
temper.pop_back();
}
}


}










return 0;


}
