#include <iostream>
#include <cstdio>
#define mod 1000000009

using namespace std;

long long int fib(long long int n) {
	long long int i, j, k, h, t;
	i = h = 1;
	j = k = 0;
	while (n > 0) {
		if (n%2 == 1) {
			t = (j*h) % mod;
			j = (i*h + j*k + t) % mod;
			i = (i*k + t) % mod;
		}
		t = (h*h) % mod;
		h = (2*k*h + t) % mod;
		k = (k*k + t) % mod;
		n = n/2;
	}
	return j;
}

int main()
{
   long long int t, n, i, j;
   scanf("%lld", &t);

   while(t--) {
        scanf("%lld", &n);
        i = fib(n) * fib(n + 1);
        i = i % mod;
        printf("%lld\n", i);
   }
}
