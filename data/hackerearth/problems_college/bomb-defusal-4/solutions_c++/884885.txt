typedef long long LL;

#define MOD 1000000007
#define MAX 1000001
#define sd(x) scanf("%d", &x)
#define terr "Terrorists Win !"
#define ct "Counter-Terrorists Win !"

using namespace std;



int main() {
	int t;
	
	cin >> t;
	
	while (t--) {
		int p, x, y, u, v;
		scanf("%d%d%d%d%d", &p, &x, &y, &u, &v);
		int a=abs(x-u);
		int b=abs(y-v);
		if ((a+b)%2==1) {
			puts(terr);
		} else {
			if (p>2) {
				puts(ct);
			} else if (p==2 && a==b) puts(ct);
			else if (p==1 && a==0 && b==0) puts(ct);
			else puts(terr);
		}
	}
	
	return 0;
}
