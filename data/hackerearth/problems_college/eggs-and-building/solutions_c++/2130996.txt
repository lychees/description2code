using namespace std;
int dp[1005][15];
int go(int n,int k)
{
    if(n==1 || n==0) return n;
    if(k==1) return n;
    if(dp[n][k]!=-1) return dp[n][k];
    int ans=INT_MAX;
    for(int i=1;i<=n;i++)
    {
        int tt=max(go(i-1,k-1),go(n-i,k));
        ans=min(ans,tt);
    }
    return dp[n][k]=ans+1;
}
int main()
{
    int t; si(t);
    SET(dp,-1);
    while(t--)
    {
        int n,k; si(n); si(k);
        printf("%d\n",go(n,k));
        
    }
	return 0;
}
