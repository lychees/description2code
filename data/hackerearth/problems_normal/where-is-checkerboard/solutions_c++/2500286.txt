#include <bits/stdc++.h>
using namespace std;

int main()
{
	//long long num=(1LL << 32);
	//cout<<num<<"-"<<(1LL<<32)<<"=";
	//cout<< (num &=~(1ll<<32) ) <<endl;
    for( int n, m; cin>>n>>m;) {
    	vector<string>tab(n);
    	for( int i=0; i < n; i++)
    		cin>>tab[i];
    	long long fRow=0, sRow=0, hash;
    	for( int i=0; i < 32; i++){
    		if( i%2==0 ){
    			fRow |= (1ll<<i);
    		} else {
    			sRow |= (1ll<<i);
    		}
    	}
    	//cout<<fRow<<" , "<<sRow<<endl;
    	
    	vector< vector<long long> >calc(n, vector<long long>() );
    	for( int j, i=0; i<n; i++){
    		hash = 0;
    		for( j=0; j < 32 && j<m; j++){
    			hash<<=1;
    			if( tab[i][j]=='#' )
    				hash |=1;
    		}
    		calc[i].push_back(hash);
    		for( ; j < m; j++){
    			hash &=~(1LL<<31);
    			hash<<=1;
    			if( tab[i][j]=='#' )
    				hash |=1;
    			calc[i].push_back(hash);
    		}
    	}
    	
    	int res = 0;
    	for( int i=0; i < n; i++){
    		for( int k, j=0; j < calc[i].size(); j++){
    			if( calc[i][j]==fRow ) {
    				for( k=0; k < 32 && i+k+1<n; k+=2){
    					if( calc[i+k][j]!=fRow || calc[i+k+1][j]!=sRow )
    						break;
    				}
    				if( k==32 ){
    					//cout<<"find in:"<<i<<", "<<j<<endl;
    					//cout<<calc[i][j]<<", "<<fRow<<endl;
    					//cout<<calc[i+1][j]<<", "<<sRow<<endl;
    					res++;
    				}
    			}
    		}
    	}
    	cout<<res<<endl;
    }
    return 0;
}
