#include<bits/stdc++.h>
using namespace std;
#define s 100001
#define ll long long int
#define p 1000000007
vector<int> adj[s];
bool visit[s];
void input();
ll dfs(int);
int main()
{
	int t;
	cin>>t;
	while(t--)input();
	return 0;
}

void input()
{
	int n,k;
	cin>>n>>k;
	for(int i=0;i<=n;i++)
	{
		adj[i].clear();
		visit[i]=false;
	}
	int x,y;
	while(k--)
	{
		cin>>x>>y;
		adj[x].push_back(y);
		adj[y].push_back(x);
	}
	ll ans=1;
	for(int i=1;i<=n;i++)
	{
		if(visit[i]==false)
		{
			ll t=dfs(i);
			if(t>=2){
			t=t*(t-1)/2;
			ans=(ans*t)%p;
			}
		}
	}
	cout<<ans<<endl;
}
ll dfs(int u)
{
	ll t=1;
	visit[u]=true;
	for(int i=0;i<adj[u].size();i++)
	{
		int v=adj[u][i];
		if(visit[v]==false)t+=dfs(v);
	}
	return t;
}
