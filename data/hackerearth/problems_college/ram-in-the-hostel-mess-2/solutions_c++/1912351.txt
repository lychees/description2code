#include<bits/stdc++.h>
using namespace std;
int n;
int arr[1000001][2];
int findpos(int i,int f)
{
	if(arr[i][f] != -1)
		return arr[i][f];
	if(i==n+1)
		return 1;
	if(f==1)
		return arr[i][f] = findpos(i+1,0);
	else
	{
		return arr[i][f] = findpos(i+1,1) +	findpos(i+1,0);
	}
}

int main()
{
	int t;
	scanf("%d",&t);
	
	while(t--)
	{
		scanf("%d",&n);
		memset(arr,-1,sizeof arr);
		long long int ans = (long long int)findpos(1,0);
		ans = ans * ans;
		printf("%lld\n",ans);
	}
}
