bool check_divi(int temp, int k, int m, vector<int>& A)
{
	if(temp%m !=0)
	{
		return false;
	}
	for(int i =k; i<A.size(); i++)
	{
		if(temp % A[i] == 0 )
		{
			return false;
		}

	}
  return true;
}

		int k = B[i];
		int m = A[k-1];
         
		if(i == 0)
		{
			result.push_back(m);

			temp = result[i-1] + (m -(result[i-1] % m));

			while(true)

				if(check_divi(temp, k,m, A))

					result.push_back(temp);
					
					break;
				}
				else
				{
					temp += m;
				}
			}
		}
	}
	
	return result[n-1];
}
