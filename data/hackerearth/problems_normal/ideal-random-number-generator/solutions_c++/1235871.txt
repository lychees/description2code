#include <iostream>
#include <algorithm>
using namespace std;

int gcd(int a,int b)
{
	while(a!=b)
	{
		if(a>b)
		{
			a=a-b;
		}
		else
		{
			b=b-a;
		}
	}
	return a;
}

int main() 
{
	int a=0,b=0,c=0,m1=0,m2=0,p=0,q=0;
	cin>>a>>b>>c;
	
	if(a<=b)
	{
		m1=a;
		m2=b;
	}
	else
	{
		m1=b;
		m2=a;
	}
	
	if(m1+m2<=c)
	{
		cout<<"1/1"<<endl;
	}
	else if(c<=m1 && c<=m2)
	{
		p=c*c;
		q=2*m1*m2;
		int g=gcd(p,q);
		p=p/g;
		q=q/g;
		cout<<p<<"/"<<q<<endl;
	}
	else if(c>=m1 && c<=m2)
	{
		p=(2*c-m1)*m1;
		q=2*m1*m2;
		int g=gcd(p,q);
		p=p/g;
		q=q/g;
		cout<<p<<"/"<<q<<endl;
	}
	else 
	{
		p=2*(c-m2)*m2+(m1+m2-c)*(c-m1+m2);
		q=2*m1*m2;
		int g=gcd(p,q);
		p=p/g;
		q=q/g;
		cout<<p<<"/"<<q<<endl;
	}
	
	return 0;
}
