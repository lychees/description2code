#include <bits/stdc++.h>
using namespace std;
 #define maxi 100000000
int solve(int x)
{
	int cont=0;
	while(x!=0)
	{
		if(x%10==4 || x%10==7)
		cont++;
		
		x/=10;
	}
	return cont;
}
 
int main()
{
    int a,l;
    cin >> a >> l;
    
   int flag=0;
    
    for(int i=a+1;i<maxi;i++)
    {	
    	int j;
    	for(j=0;j<l;j++)
    	{
    		if(solve(a+j)!=solve(i+j))
    		{
    			break;
    		}
    	}
    	if(j==l)
    	{
    		flag=1;
    		cout << i;
    		break;
    	}
    }
    if(!flag)
    {
    	cout << -1;
    }
}
