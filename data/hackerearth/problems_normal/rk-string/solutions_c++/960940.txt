#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

ll arr[1000005],brr[1000005],sumr,sumk,ansr,ansk,m,n,t;
string str;
int main()
{
    cin>>t;
    while(t--)
    {
    	cin>>m>>n;
    	cin>>str;
    	memset(arr,0,sizeof(arr));
    	memset(brr,0,sizeof(brr));
    	arr[0]=brr[0]=1;
    	ansr=ansk=sumr=sumk=0;
    	for(int i=0;str[i];i++)
    	{
    		if(str[i]=='K')
    			sumk++;
    		else
    			sumr++;
    		if(sumr>=m)
    			ansr+=arr[sumr-m];
    		if(sumk>=n)
    			ansk+=brr[sumk-n];
    		arr[sumr]++;
    		brr[sumk]++;
    	}
    	cout<<ansr<<" "<<ansk<<endl;
    }
    return 0;
}
