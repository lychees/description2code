#include<iostream>
using namespace std;
long long int gcd(long long int a,long long int b)
{
    if(!b) return a;
    else
        return gcd(b,a%b);
}
main()
{
    long long int t,n,a,b,ans;
    cin>>t;
    while(t--)
    {
        cin>>n;
        if(n==1)
        {
            cin>>n;
            ans=n;
        }
        else
        {
            cin>>a>>b;
            ans=gcd(a,b);
            for(long long int i=2;i<n;i++)
            {
                cin>>a;
                ans=gcd(ans,a);
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
