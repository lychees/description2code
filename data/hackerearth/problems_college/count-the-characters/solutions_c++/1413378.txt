#include<iostream>
using namespace std;
int main()
{
int a[27]={0};
int real[27]={0};
unsigned int n,n1;
char b[101];
int ctr=0;
cin>>n;
n1=n;
    while(n1--)
    {
      cin>>b;
      for(unsigned int i=0;b[i]!='\0';i++)
           {
           real[b[i]-96]++;
           a[b[i]-96]++;
           }
           for(unsigned int i=1;i<27;i++)
           {
               if(a[i]>0)
               real[i]-=a[i]-1;
                a[i]=0;

           }
    }
    for(unsigned int i =1;i<27;i++)
    {
        if(real[i]>=n)
            ctr++;

    }
    cout<<ctr<<endl;
return 0;
}
