#include <iostream>
#define ll long long
using namespace std;

void swap(string& s,int n){
	ll k=n*(n+1)/2;
	

	if(k%2!=0){
		int i=0;
		while(i<s.length()){
		if(s[i]=='1')
		s[i]='0';
		else if(s[i]=='0')
		s[i]='1';
	
		i++;
		}
	}

}

int countSetBit(string s){
	int count=0;
	for(int i=0;i<s.length();i++){
		if(s[i]=='1')
		count++;
	}
	return count;
}

int main()
{
    int test,n;
    string s;
    cin>>test;
    while(test--){
    	cin>>s>>n;
  
    	swap(s,n);
    	

    	cout<<countSetBit(s)<<endl;
    }
    return 0;
}
