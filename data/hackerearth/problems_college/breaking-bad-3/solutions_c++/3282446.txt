#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,i;
        cin>>n;
        long arr[n];
        long long prod = 0,ans = 0;
        for(i=0;i<n;i++)
        {
            cin>>arr[i];
        }
        sort(arr,arr+n);
        reverse(arr,arr+n);
        i = 0;
        while(i<n)
        {
            prod = arr[i]*arr[i+1];
            ans+=prod;
            i = i+2;
        }
        cout<<ans<<endl;
    }
    return 0;
}
