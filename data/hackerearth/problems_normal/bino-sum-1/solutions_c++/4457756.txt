#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long
ll fact[1002];
ll power(ll a,ll b)
{
	ll res=1;
	while(b>0)
	{
		if(b&1)
		res=(res*a)%mod;
		a=(a*a)%mod;
		b/=2;
	}
	return res;
}
int main()
{
    int t,n,p,i;
    fact[0]=fact[1]=1;
    for(i=2;i<=1000;i++)
    {
    	fact[i]=(fact[i-1]*i)%mod;
    }
    scanf("%d",&t);
    while(t--)
    {
    	ll ans=0;
    	scanf("%d%d",&n,&p);
    	if(n==p)
    	ans=power(2,n);
    	else
    	{
    		for(i=0;i<=p;i++)
    		{
    			ll temp=(fact[i]*fact[n-i])%mod;
    			ans=(ans+(fact[n]*power(temp,mod-2))%mod)%mod;
    		}
    	}
    	printf("%lld\n",ans);
    }
    return 0;
}
