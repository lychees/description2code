#include <cstdio>
#include <iostream>
#include <algorithm>
#include <vector>
#include <set>
#include <queue>
#include <map>
#include <cstring>
#include <string>
#include <cmath>
using namespace std;

int a[1005][1005], bi[1005][1005], bj[1005][1005];

int main()
{
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < n; ++i)
	for (int j = 0; j < m; ++j)
		scanf("%d", &a[i][j]);
	for (int i = 0; i < n; ++i) 
	for (int j = 0; j < m; ++j) {
		bi[i][j] = 1;
		if (i && a[i - 1][j] == a[i][j])
			bi[i][j] = bi[i - 1][j] + 1;
	}
	int ans = 0;
	for (int i = 0; i < n; ++i)
	for (int j = m-1; j >= 0; --j) {
		bj[i][j] = 1;
		if (j < m-1 && a[i][j+1] == a[i][j])
			bj[i][j] = bj[i][j+1] + 1;
		ans = max(ans, bi[i][j] + bj[i][j] - 1);
	}
	cout << ans << endl;

	return 0;
}
