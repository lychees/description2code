#include <bits/stdc++.h>

#define REP(i, a) for (int i = 0; i < (int) (a); i ++)
#define REPP(i, a, b) for (int i = (int) (a); i <= (int) (b); i ++)
#define MST(a, b) memset(a, b, sizeof(a))

using namespace std;

map<int,int> cnt;

int main(){
    int n,x;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&x);
        cnt[x]++;
    }
    int ans = -1;
    for(auto it : cnt){
        if(it.second == 1)
            ans = max(ans,it.first);
    }
    printf("%d\n",ans);
    return 0;
}
