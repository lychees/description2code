#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,t,n,elem,j,maxi,k;
    vector<int> v(30000001);
    scanf("%d",&n);
    while(n)
    {
    	maxi=0;
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&(elem));
    		v[elem]++;
    		if(elem>maxi)
    			maxi=elem;
    	}
    	for(i=0;i<=maxi;i++)
    	{
    		j=v[i];
    		for(k=1;k<=j;k++)
    			printf("%d ",i);
    	}
    	printf("\n");
    	for(i=0;i<=maxi;i++)
    		v[i]=0;
    	scanf("%d",&n);
    }
    return 0;
}
