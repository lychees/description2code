#include <iostream>
using namespace std;
#define ll long long
ll modpow(ll a,ll b,ll c)
{
	ll t=1;
	while(b>0)
	{
		if(b&1)
		t=((t%c)*(a%c))%c;
		a = ((a%c)*(a%c))%c;
		b>>=1;
	}
	return t;
}
int main()
{
	int T;
	scanf("%d",&T);
	while(T--)
	{
	string str; ll mod,ans=0;
	cin>>str>>mod;
	int n  = str.length();
	for(int i=0;i<str.length();i++)
	{
		ans+=(str[n-i-1]-'0')*modpow(10,i,mod)%mod;
	}
	printf("%lld\n",ans%mod);
	}
    return 0;
}
