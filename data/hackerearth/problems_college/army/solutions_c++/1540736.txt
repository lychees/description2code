#include <bits/stdc++.h>
#define LL long long int
#define mp make_pair
#define pb push_back
#define ff first
#define ss second
using namespace std;
int ans[1<<10];
void calc(int n){
	int temp,t1;
	for(int i=1;i<=n;i++){
		for(int j=1;j<((1<<10));j++){
			temp=i;
			while(temp){
				t1 = temp%10;
				if(j&(1<<t1)){ans[j]++;break;}
				temp/=10;
			}
		}
	}
	return ;
}
int main(){
	ios_base::sync_with_stdio(false); cin.tie(NULL);
	int i,j,n,q,m,val,z,x;
	cin >> n;
	calc(n);
	cin >> q;
	while(q--){
		cin >> m;
		val =0;
		while(m--){
			cin >> x;
			val |=(1<<x);
		}
		cout << ans[val] << endl;
	}
	return 0;
}
