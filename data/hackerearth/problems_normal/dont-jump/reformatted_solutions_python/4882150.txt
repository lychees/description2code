import sys
while True:
    line = sys.stdin.readline()
    if not line:
        break
    marble_num = int(line.strip())
    marble_stack = []
    cont_stack = []
    for i in range(1, marble_num):
        marble_stack.append(i)

    line = sys.stdin.readline()
    split = line.strip().split(" ")
    if split != [""]:
        patt = [int(x) for x in split]
    else:
        patt = []
    string = []
    pssb = True
    for i in patt:
        while True:
            if not cont_stack:
                string.append("C")
                if marble_stack:
                    cont_stack.append(marble_stack.pop(0))
                else:
                    pssb = False
                    break
            elif cont_stack[-1] < i:
                string.append("C")
                if marble_stack:
                    cont_stack.append(marble_stack.pop(0))
                else:
                    pssb = False
                    break
            elif cont_stack[-1] > i:
                string.append("K")
                cont_stack.pop()
            else:
                string.append("K")
                cont_stack.pop()
                break
        if not pssb:
            break
    if pssb:
        print "".join(string)
    else:
        print -1
