#include <bits/stdc++.h>
 
using namespace std;

# define MOD 1000000007

template<class T> inline void read(T& x) {
    char ch;
    while(!isdigit(ch = getchar_unlocked()));
    x = ch-'0';
    while( isdigit(ch = getchar_unlocked())) x *= 10, x += ch-'0';
}

int main(){
	long n;
	read <long> (n);
	std::vector<long> x,y,r;
	while(n--){
		long a,b,c;
		read <long> (a);
		read <long> (b);
		read <long> (c);
		x.push_back(a);
		y.push_back(b);
		r.push_back(c);
	}
	long check;
	long answer = 0;
	for(long i = 1 ; i <= 1000 ; i++){
		for(long j = 1 ; j <= 1000 ; j++){
			check = 0;
			for(long k = 0 ; k < x.size() ; k++){
				if((i-x[k])*(i-x[k])+(j-y[k])*(j-y[k])-r[k]*r[k] <= 0){
					check++;
				}
			}
			if(check > 1){
				answer++;
			}
		}
	}
	cout << answer << "\n";
	return 0;
}
