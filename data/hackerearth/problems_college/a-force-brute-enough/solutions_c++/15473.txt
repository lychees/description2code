#include<iostream>
#include<algorithm>
#include<map>
#include<vector>
#include<cstdio>
#include<ctype.h>
#define si(n) scanf("%d",&n)
#define FOR(i,n) for(int i =0 ; i < (int)n ; i++)
#define FORN(i,x,n) for(int i = (int)x ; i < (int)n ; i++)

using namespace std;
vector<int> seq;
vector<int> a;

int n;
int k = 10;

void db(int t,int p){
	if(t > n){
		if(n%p==0){
			FORN(j,1,p+1){
				seq.push_back(a[j]);
			}
		}
	}
	else{
		a[t]= a[t-p];
		db(t+1,p);
		FORN(j,a[t-p]+1,k){
			a[t]=j;
			db(t+1,t);
		}
	}
}

void de_bruijn(int k,int n){
	db(1,1);
}

int main(){
	while(1){
		scanf("%d\n",&n);
		if(n==0) break;
		seq.clear();
		a.clear();
		for(int i =0 ; i < 10*n; i++) a.push_back(0);
		de_bruijn(10,n);
		FOR(i,seq.size()){
			cout << seq[i];
		}
		FOR(i,n-1)
			cout << seq[i];
		cout << endl;
	}
}
