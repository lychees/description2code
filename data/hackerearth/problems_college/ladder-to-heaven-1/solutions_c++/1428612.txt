#include <iostream>
using namespace std;

int main()
{
    long long int t,f,s,temp,sum,n;
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	
    	f=1;
    	s=2;
    	
    	if(n<2)
    	{
    		printf("0\n");
    		continue;
    	}
    	
    	sum=2;
    	while(s<n)
    	{
    		temp = s;
    		s = f+s;
    		if(s>n)
    		{
    			break;
    		}
    		f= temp;
    		if(s%2==0)
    		{
    			//printf("%lld\n",s);
    			sum = sum+s;
    		}
    	}
    	
    	printf("%lld\n",sum);
    }
    
    return 0;
}
