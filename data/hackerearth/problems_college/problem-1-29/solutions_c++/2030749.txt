#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include<string.h>
using namespace std;
int mazeResolve(int arr[100][100],int n,int x,int y)
{
	if(x==n-1 && y==n-1)
		return 1;
	if(arr[x][y]==0)
		return 0;
	else
		return mazeResolve(arr,n,x+1,y)||mazeResolve(arr,n,x,y+1);
}
int solveMaze(int arr[100][100],int n)
{
	if(mazeResolve(arr,n,0,0))
		cout<<"POSSIBLE"<<endl;
	else
		cout<<"NOT POSSIBLE"<<endl;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,arr[100][100];
		cin>>n;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
				cin>>arr[i][j];
		}
		solveMaze(arr,n);
			
	}
}
