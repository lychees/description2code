#include<bits/stdc++.h>
#define ll long long
#define f(a,b,c) for(ll int i=a;i<b;i+=c)
#define mod 1000000007

using namespace std;

ll powmod(ll a,ll b) {
	ll x = 1, y = a;
	while(b) {
		if(b & 1) {
			x = x * y % mod;
		}
		y = y * y % mod;
		b >>= 1;
	}
	return x;
}

int main() {
	cin.tie(0);
	ios_base::sync_with_stdio(false);
	ll int T;
	cin >> T;
	while(T--) {
		ll int N;
		cin >> N;
		ll int a[N];
		for(ll int i=0;i<N;i++)
			cin >> a[i];
		ll p = a[0];
		for(ll int i=1;i<N;i++) {
			p=p|a[i];
		}
		ll int ans = (((ll int)powmod(2,N-1)%mod)*(p%mod))%mod;
		cout << ans << endl;
	}
	return 0;
}
