import math

n = int(raw_input())

for i in range(n):
    z, m, n = map(int, raw_input().split())
    value = int(math.sqrt(z))
    total = 0
    for j in range(1, value+1):
        if z % j == 0:

            value2 = z/j

            if value2 <= n:
                total += 1
            if value2 <= m and j != value2:
                total += 1
    print total
