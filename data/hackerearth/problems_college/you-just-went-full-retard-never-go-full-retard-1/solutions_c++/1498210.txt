#include <iostream>
#include <cmath>
using namespace std;


void isPrime(long long n){
		if(n<2){
			cout<<"NO"<<endl;
			return ;
		}
		else if(n%2==0){
			if(n!=2){
				cout<<"NO"<<endl;
				return ;
			}
			cout<<"YES"<<endl;
			return ;
		}
		else{
			
			long long limit = sqrt(n);
			for(long i=3;i<limit;i+=2){
				if(n%i==0){						
					cout<<"NO"<<endl;
					return ;
				}
			}
			cout<<"YES"<<endl;
		}
		
	
	
}
int main()
{
	int t;
	cin>>t;
	while(t--){
		long long n;
		cin>>n;
		isPrime(n);
	}
	return 0;
}
