#include<iostream>
#include<stdlib.h>
#include<string.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	char s[100000];
	int hash[26]={0};
	cin>>s;
	for(int i=0;s[i]!='\0';i++)
	{
		int x=s[i]-'a';
		//cout<<x<<" ";
		hash[x]++;
	}
	 
	int min=hash[1];
	if(hash[20]<min)
		min=hash[20];
	if(hash[17]<min)
		min=hash[17];
	if(hash[24]<min)
		 min=hash[24];
	
	 cout<<min<<endl;
	 }
return 0;	
}
