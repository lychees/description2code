class Solution(object):
    def solve(self, cipher):       
        tasks = cipher
        tasks.sort(key=lambda t: t[0])
        overshot = -1
        timer = 0
        for task in tasks:
            timer += task[1]
            overshot = max(overshot, timer - task[0])
        return overshot
