#include <iostream>
using namespace std;
int a[2000005];
int main()
{
   int T, N,i,k;
   long long s;
   cin>>T;
   while(T--)
   {
   		cin>>N;
   		s=0;
   		for(i=0;i<2000005;i++)
   		{
   			a[i]=0;
   		}
   		while(N--)
   		{
   			cin>>k;
   			a[k+1000000]++;
   		}
   		for(i=0;i<2000005;i++)
   		{
   			k=a[i];
   			s+=k*(k+1)/2;
   		}
   		cout<<s<<endl;
   }
}
