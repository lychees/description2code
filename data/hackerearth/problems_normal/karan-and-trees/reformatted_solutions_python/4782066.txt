
test = int(raw_input())
for i in range(test):
    a, b = [int(x) for x in raw_input().split()]
    p1 = a
    p2 = b
    while p1 != p2:
        if p1 > p2:
            if p1 % 2 == 0:
                p1 /= 2
            else:
                p1 = (p1 - 1) / 2
        elif p2 > p1:
            if p2 % 2 == 0:
                p2 /= 2
            else:
                p2 = (p2 - 1) / 2
    print p1
