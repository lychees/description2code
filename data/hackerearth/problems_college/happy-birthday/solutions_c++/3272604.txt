#include<iostream>
using namespace std;
main()
{
	int n,i,a[1000];
	int s=0,r=0,k=0,m=0;
	cin>>n;
	for(i=0;i<n;i++)
	cin>>a[i];
	for(i=0;i<n;i++)
	{
		if(a[i]==4)
		s++;
		else if(a[i]==2)
		{
			r++;
		}
		else if(a[i]==3)
		k++;
		else if(a[i]==1)
		m++;
		
	}
	if(r%2==0)
	{
	s=s+(r/2);
	}
	else if(r%2!=0)
	{
		s=s+(r+1)/2;	
	}
	if(k==m&&k!=0)
	{
		s=s+k;
	}
	else
	{
		s=s+k;
		m=k-m;
		if(m>0||k==0)
		{
		if(r%2!=0)
		{
			m=m-2;	
		}
		if(m<0)
		m=-m;
		s=s+(m/4);
	}
}
	
	cout<<s;
	return 0;
}
