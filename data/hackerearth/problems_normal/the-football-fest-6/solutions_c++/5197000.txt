#include <iostream>
using namespace std;
#include <iostream>
#include<stack>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long n,id;
		cin>>n>>id;
		stack<long long> s;
		s.push(id);
		while(n--)
		{
			char ch;
			long long pid;
			cin>>ch;
			if(ch=='P')
			{	
				cin>>pid;
				s.push(pid);
			}
			else
			{
				long long x=s.top();
				s.pop();
				long long y=s.top();
				s.push(x);
				s.push(y);
			}
		}
		cout<<"Player"<<" "<< s.top()<<endl;
	}
return 0;
}
