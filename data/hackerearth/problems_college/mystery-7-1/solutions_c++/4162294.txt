#include <iostream>
#include <string>
#include <math.h>
using namespace std;

int main()
{
    int t,a,i,l;
    string s;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	l=s.length();
    	for(i=0;i<l;i++)
    	{
    		a=s[i]-97;
   // 		cout<<endl<<"a:"<<a<<":";
    		if(a<18)	cout<<(a/3)+2;		
    		else if(a==18)	cout<<"7";
    		else if((a>18)&&(a<22))	cout<<"8";
    		else if((a>=22)&&(a<=25))	cout<<"9";
    	}
    	cout<<endl;
    }
    return 0;
}
