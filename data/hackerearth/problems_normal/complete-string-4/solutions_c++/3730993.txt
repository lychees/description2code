#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a;
    	cin>>a;
    	bool present[26]={false};
    	for(int i=0;i<(a.length());i++)
    		present[((int)a.at(i))-97]=true;
    	bool fail=false;
    	for(int i=0;i<26;i++)
    	{
    		if(!present[i])
    		{
    			fail=true;
    			break;
    		}
    	}
    	if(fail)
    		cout<<"NO"<<endl;
    	else
    		cout<<"YES"<<endl;
    }
    return 0;
}
