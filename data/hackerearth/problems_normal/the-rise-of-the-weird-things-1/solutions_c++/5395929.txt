#include<bits/stdc++.h>
using namespace std;
int main(){
	int n;
	scanf("%d",&n);
	int zs=0,vs=0;
	vector<int>z,v;
	vector<int>::iterator i;
	for(int i=0;i<n;i++){
		int tmp;
		scanf("%d",&tmp);
		if(tmp%2==0){
			zs+=tmp;
			z.push_back(tmp);
		}
		else{
			vs+=tmp;
			v.push_back(tmp);
		}
	}
	sort(z.begin(),z.end());
	sort(v.begin(),v.end());
	for(i=z.begin();i!=z.end();i++)cout<<*i<<" ";
	printf("%d ",zs);
	for(i=v.begin();i!=v.end();i++)cout<<*i<<" ";
	printf("%d",vs);
	return 0;
}
