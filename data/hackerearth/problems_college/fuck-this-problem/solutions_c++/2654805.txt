#include <iostream>
using namespace std;
#include<stdio.h>

int main()
{
    char s[10000];
    int n=0,l;
    scanf(" %[^\n]s",s);
    while(s[n]!='\0')
    {
    	n++;
    }
    cin>>l;
    int j=0;
    for(int i=0;i<(n/l)+1;i++)
    {
    	while(s[j]!='\0')
    	{
    		cout<<s[j];
    		if((j%l)-(l-1)==0)
    			cout<<endl;
    		j++;
    	}
    }
    return 0;
}
