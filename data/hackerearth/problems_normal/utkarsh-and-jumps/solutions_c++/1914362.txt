#include <iostream>
#include <iomanip>
using namespace std;

#define fsize 1000000

double dp[fsize];

double fun(double n,double p)
{
	p=p/100;
	
	dp[0]=0;
	dp[1]=0;
	dp[2]=p;
	
	dp[3]=1-p;

	for(int i=4;i<=n;i++)
	{
		dp[i]=p*dp[i-2]+(1-p)*dp[i-3];
	}

    return dp[int(n)];
}


int main()
{
    double n;
    double p;
    cin>>n;
    cin>>p;
	std::cout.precision(6);
    cout<<fixed<<fun(n,p)<<endl;
    return 0;
}
