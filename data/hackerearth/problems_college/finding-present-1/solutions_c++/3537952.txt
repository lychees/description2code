#include <iostream>
using namespace std;

int main()
{
    long int b,p,q,a[10001]={0};
    cin>>p;
    while(p--)
    {
    	cin>>b;
        a[b]++;
    	
    }
    cin>>q;
    while(q--)
    {
    	cin>>b;
    	a[b]--;
    }
    for(int i=1;i<=10000;i++)
    {if(a[i]!=0)
    cout<<i<<" ";}
    return 0;
}
