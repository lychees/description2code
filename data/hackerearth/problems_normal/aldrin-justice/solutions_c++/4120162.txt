#include <iostream>
using namespace std;

int main()
{
    int t,bt1,bt2,m1,m2,x,y;
    cin>>t;
    while(t--)
    {
    	cin>>x>>y;
    	if(x<y)
    	{
    		bt1=x;
    		bt2=y;
    	}
    	else
    	{
    		bt1=y;
    		bt2=x;
    	}
    	cin>>x>>y;
    	if(x<y)
    	{
    		m1=x;
    		m2=y;
    	}
    	else
    	{
    		m1=y;
    		m2=x;
    	}
    	if((bt2==m1) || (bt1==bt2 && m1<=bt1 && m2>=bt2) || (m1==m2 && bt1<=m1 && bt2>=m1))
    	{
    		cout<<"Point"<<endl;
    	}
    	else if(bt2<m1 || m2<bt1)
    	{
    		cout<<"Line"<<endl;
    	}
    	else
    	{
    		cout<<"Nothing"<<endl;
    	}
    	
    }
    return 0;
}
