#include<stdio.h>
#include<iostream>
#include<string>

using namespace std;

int palindrome(string S,int R,int L)
{
    int pos=1;
    
    while(R<=L)
    {
        if(S[R]!=S[L])
        {
            pos=-1;
            break;
        }
        R+=1;L-=1;
    }
    
    
    return pos;
}


int main()
{
    int C,i,j,Len,pos,v;
    scanf("%d",&C);
    
    while(C--)
    {
        string S;
        cin >> S;
        
        Len = S.size();
        
        i = 0 ;
		j = Len-1;
        
        while(i<=j)
        {
            pos = -1;
            
            if(S[i]!=S[j])
            {
               
                v = palindrome(S,i+1,j);
                
                if(v!=-1)
                    pos=i;
                else
                    pos=j;
               
                break;
            }
            
            i = i + 1;
            j = j - 1;
            
        }
        printf("%d\n",pos);
    }
    return 0;
}
