#include <bits/stdc++.h>
using namespace std;
int main()
{
    int j,count=0;
    cin>>j;
    int a[j];
    for(int i=0;i<j;i++)
    {
        cin>>a[i];
    }
    sort(a,a+j);
    for(int i=0;i<(j-1);i++)
    {
        if(a[i] == a[i+1])
            count++;
    }
    cout<<count<<endl;
}
