#include <bits/stdc++.h>
using namespace std;
#define lli long long int
#define md 1000000007
map<lli,int> m;
map<lli,int> :: iterator it;
lli A[205],ans;
int n,fl;
int X[100005];
void func()
{
	for(int i=2;i<100005;i++)
	{
		for(int j=0;j<n;j++)
		{
			while(A[j]%i==0)
				A[j]/=i,X[i]++;
		}
	}
	for(int i=0;i<n;i++)
		if(A[i]>1)
			m[A[i]]++;
}
int main() {
	scanf("%d",&n);
	fl=0;
	ans=1;
	for(int i=0;i<n;i++) scanf("%lld",&A[i]);
	func();
	
	int fl=0;
	for(int i=3;i<100005;i++)
	{
		if((i-1)%4==0) ans=(ans*((X[i]+1))%md)%md;
		if((i-3)%4==0 && X[i]&1)
		{
			fl=1;
			break;
		}
	}
	if(fl==1)
	{
		printf("0\n");
		return 0;
	}
	for(it=m.begin();it!=m.end();it++)
	{
		if(((it->first)-1)%4==0) ans=(ans*((it->second)+1))%md;
		if(((it->first)-3)%4==0 && (it->second)&1)
		{
			fl=1;
			break;
		}
	}
	if(fl==1) printf("0\n");
	else printf("%d\n",(4*ans)%md);
	return 0;
}
