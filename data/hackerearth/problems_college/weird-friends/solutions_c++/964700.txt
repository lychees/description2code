#include <iostream>
using namespace std;

int main()
{
    long long int i,l,x,y,d=0;
    cin>>l>>x>>y;
    if(y>=1&&x>y&&l>=x&&l<=1000000000)
    {
    for(i=0;d<=l;i++)
    { 
    	d+=x;
    	if(d==l||d>l) 
    	{
    		i++;
    		break;
    	}	
    	else d-=y;
    }
    cout<<i;
    }
    return 0;
}
