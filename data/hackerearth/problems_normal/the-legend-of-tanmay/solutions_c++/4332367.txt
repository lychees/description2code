#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int main()

{

    int x;
    cin >> x;
    while(x--)
    {

        int y;
        cin >> y;
        vector <int> a,b;
        int c=0;
        while(y--)
        {

            int q;
            cin >> q;
            if(q==0)
            {

                c++;
            }
            if(q>0)
            {

                a.push_back(q);
            }
            if(q<0)
            {

                b.push_back(q);
            }

        }



        sort(a.begin(),a.end());
        sort(b.begin(),b.end());
        long long t=1,m=1;
        if(b.size()==0&&a.size()==0)
        {
                t=0;
                m=0;
        }
        else
        {

            if(b.size()==0)
            {
                if(c!=0)
                {

                    m=0;

                }
                else
                {
                    m=a[0];

                }
                for(int d=0;d<a.size();d++)
                {

                    t*=a[d];
                }
            }
            else if(a.size()==0)
            {
                if(b.size()==1)
                {

                    m=b[0];
                    if(c!=0)
                    {

                        t=0;

                    }
                    else
                    {

                        t=b[0];
                    }
                }
                else
                {

                    if(b.size()%2==0)
                    {
                        for(int d=0;d<b.size();d++)
                        {

                            t*=b[d];
                        }
                        for(int d=0;d<b.size()-1;d++)
                        {
                            m*=b[d];
                        }

                    }
                    else
                    {
                        for(int d=0;d<b.size()-1;d++)
                        {

                            t*=b[d];
                        }
                        for(int d=0;d<b.size();d++)
                        {

                            m*=b[d];
                        }
                    }
                }

            }
            else
            {
                if(b.size()==1)
                {

                    m=b[0];


                }
                else
                {

                    if(b.size()%2==0)
                    {

                        for(int d=0;d<b.size();d++)
                        {

                            t*=b[d];

                        }
                        for(int d=0;d<b.size()-1;d++)
                        {

                            m*=b[d];
                        }

                    }
                    else
                    {
                        for(int d=0;d<b.size()-1;d++)
                        {

                            t*=b[d];
                        }
                        for(int d=0;d<b.size();d++)
                        {

                            m*=b[d];
                        }


                    }
             
                }
                       for(int d=0;d<a.size();d++)
                    {

                        t*=a[d];
                        m*=a[d];
                    }


            }
        }
        cout << t << " " << m << endl;
    }
}
