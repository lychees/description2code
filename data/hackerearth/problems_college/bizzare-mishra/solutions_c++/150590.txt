#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
main()
{
  int t,n,c,i,f,d;    
   char s[10001],b[10001],x[10001];   
   scanf("%d",&t);
   while(t--)   
    {
     c=0;d=0;           
     scanf("%d",&n);            
      scanf("%s",s);
      b[0]='N'; 
      x[0]='Y';  
      if(b[0]!=s[0])
       c++;
       if(x[0]!=s[0])
        d++;        
      for(i=1;i<n;i++)          
       {
         if(b[i-1]=='N')
          {                        
          b[i]='Y';
          if(b[i]!=s[i])
           c++;                  
          }
          else
          {
           b[i]='N';
           if(b[i]!=s[i])
            c++;
          }        
          
          if(x[i-1]=='N')
          {                        
          x[i]='Y';
          if(x[i]!=s[i])
           d++;                  
          }
          else
          {
           x[i]='N';
           if(x[i]!=s[i])
            d++;
          }    
                       
       }   
     printf("%d\n",min(c,d));        
    }  
 return 0;     
}
