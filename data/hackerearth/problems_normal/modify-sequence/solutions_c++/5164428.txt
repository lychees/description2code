#include <iostream>
using namespace std;

int main()
{
	long int n,a[10001],i;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		
	}
	for(i=0;i<n-1;i++)
	{
		a[i+1]=a[i+1]-a[i];
		a[i]=0;
		if(a[i+1]<0)
		{	i++;
			break;
		}	
	}
	i--;
	if(n!=1)
	{
	if(a[i]==0&&a[i+1]==0)
	cout<<"YES";
	else 
	cout<<"NO";
	}
	else if(n==1&&a[0]==0)
	cout<<"YES";
	else if(n==1&&a[0]!=0)
	cout<<"NO";
	
    return 0;
}
