#include <iostream>
using namespace std;

int main()
{
    int N,k,A[1000000],B[1000000]={},ans,max;
    cin>>N>>k;
    for(int i=0;i<N;i++)
    {
    	cin>>A[i];
    	B[A[i]]++;
    }
    for(int i=0;i<N;i++)
    {
    	if(i==0)
    	{ 
          max=B[i];
    	  ans=i;
    	}  
    	if(B[i]>max)
    	{  
    	  max=B[i];
    	  ans=i;
    	}  
    }
    cout<<ans;
	return 0;
}
