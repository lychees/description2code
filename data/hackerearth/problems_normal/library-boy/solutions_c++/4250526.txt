//COLOR CODE
#include <bits/stdc++.h>
using namespace std;
int main()
{  int num,i,count=0,arr[26]={0};
   cin>>num;
   string input;
   for(i=0;i<num;i++)
   {
      cin>>input;
      if(input.length()>2 &&input.length()<=20)
         arr[input[0]-97]++;
   }
   for(i=0;i<26;i++)
     if(arr[i]!=0 && arr[i]%10!=0)
         count+=arr[i]/10 +1;
    else if(arr[i]!=0 && arr[i]%10==0)
        count+=arr[i]/10;
   cout<<count<<endl;
   return 0;   
}
