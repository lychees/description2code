#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#include<algorithm>

int main()
{
 string s;
 getline(cin,s);
 if (s[0]>='a' && s[0]<='z')
 s[0]=s[0]-32;
 for(int i=0;i<s.length();i++)
 {
 	if(s[i]==' ' )
 	{
 	if(s[i+1]>='a' && s[i+1]<='z')
 	{
 	s[i+1]=s[i+1]-32;
 	}
 	}
 }
 cout<<s;
 return 0;
}
