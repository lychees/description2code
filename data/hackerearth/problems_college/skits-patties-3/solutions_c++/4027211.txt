#include<iostream>
using namespace std;
int arr[400];
int main()
{
    arr[0]=1;
 
    for(int i=0;i<=360;i++)
        if(arr[i]==1)
            arr[i+6]=1,arr[i+9]=1,arr[i+20]=1;
    int task;
    cin>>task;
    while(task--)
    {
        int p;
        cin>>p;
        if(arr[p%360])
        cout<<"True\n";
        else
            cout<<"False\n";
    }
}
