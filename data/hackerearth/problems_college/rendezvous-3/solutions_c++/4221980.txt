#include<bits/stdc++.h>
using namespace std;
struct mo{
	int a;
	int b;
	int c;
}edge[100001];

struct mo1{
	int to;
	int coast;
	int last;
};

vector<pair<int,int> > adj[100001];
class cmp1{
public:
	bool operator () (mo1 a,mo1 b){
		return (a.last>b.last);
	}
};
mo1 make(int to,int coast,int last){
	mo1 temp;
	temp.to = to;
	temp.coast = coast;
	temp.last = last;
	return temp;
}
int main(){
	int n,m;
	scanf("%d %d",&n,&m);
	for(int i=0;i<m;i++){
		int a,b,c;
		scanf("%d %d %d",&a,&b,&c);
		adj[a].push_back(make_pair(b,c));
		adj[b].push_back(make_pair(a,c));
	}
	long long ans = 0;
	priority_queue<mo1,vector<mo1> ,cmp1 > q;
	q.push(make(1,0,0));
	bool v[100001];
	memset(v,false,sizeof(v));
	while(!q.empty()){
		mo1 temp = q.top();
		q.pop();
		if(v[temp.to])
		   continue;
		//printf("%lld\n",temp.last);
		v[temp.to] = true;
		ans += temp.last;
		for(int i=0;i<adj[temp.to].size();i++){
			if(v[adj[temp.to][i].first]==false){
				q.push(make(adj[temp.to][i].first,temp.coast+adj[temp.to][i].second,adj[temp.to][i].second));
			}
		}
	}
	printf("%lld\n",ans);
	return 0;
}
