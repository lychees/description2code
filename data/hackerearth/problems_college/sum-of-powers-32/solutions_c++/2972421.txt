set<int> S;
void init()
{
    S.insert(1);
    for(int i=2;i<=100000;i++)
    {
        ll x = i;
        ll f = i;
        x = x*f;
        while(x <= 1000000000)
        {
            S.insert(x);
            x = x*f;
        }
    }
   // cout << S.size();
}
int main()
{
    init();
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int x;
        scanf("%d",&x);
        int flag = 0;
        for(set<int>::iterator it = S.begin(); it!=S.end();it++)
        {
            if(*it >= x)
                break;
            int y = *it;
            if(S.find(x - y) != S.end())
            {
                //cout << x - y << " " << y << endl;
                flag = 1;
                break;
            }
        }
        if(flag)
            printf("Yes\n");
        else
            printf("No\n");
    }
    
    
    
return 0;
}
