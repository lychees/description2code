n=0
m=0
for _ in range(input()):
    x,y=map(int,raw_input().split())
    z=map(int,raw_input().split())
    n=(z[0]+z[2])*y
    m=(z[1]+z[2])*x
    if n<z[1]*y:
        print ((x+y)*z[0])+(y*z[2])
    elif m<z[0]*x:
        print ((x+y)*z[1])+(x*z[2])
    else:
        print x*z[0]+y*z[1]
