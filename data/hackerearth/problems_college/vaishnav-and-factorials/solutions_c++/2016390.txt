#include<iostream>
using namespace std;
int main()
{
    int t,count;
    int a[100][2000];
    int n,i,j,temp,m[100],x,o;
    cin>>t;
    for(o=0;o<t;o++)
    {
       cin>>n;
       a[o][0]=1;
       m[o]=1;
 
       temp = 0;
       for(i=1;i<=n;i++)
       {
            for(j=0;j<m[o];j++)
            {
               x = a[o][j]*i+temp;
               a[o][j]=x%10;
               temp = x/10;
            }
             while(temp>0)
             {
               a[o][m[o]]=temp%10;
               temp = temp/10;
               m[o]++;
             }
      }
    }
    for(o=0;o<t;o++)
    {
    	count=0;
        for(i=m[o]-1;i>=0;i--)
        {
          	if(a[o][i]==4||a[o][i]==7)
        		count++;
        }
        cout<<count<<endl;
    }
    return 0;
}
