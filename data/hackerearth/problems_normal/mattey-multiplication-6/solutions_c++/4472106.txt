#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	
	while (t--) {
		long long n, m;
		cin >> n >> m;
		
		bool fl = false;
		for (int i = 63; i >= 0; i--) {
			if (m & (1LL << i)) {
				if (fl) cout << " + ";
				cout << "(" << n << "<<" << i << ")";
				fl = true;
			}
		}
		cout << endl;
	}
}
