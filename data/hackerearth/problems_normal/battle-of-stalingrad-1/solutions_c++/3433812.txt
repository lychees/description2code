#include <iostream>
using namespace std;
int hold(int r);
int area(int x1, int y1, int x2, int y2, int x3, int y3){
	long long A;
	A = (((x1*y2)-(x2*y1))+((x2*y3)-(x3*y2))+((x3*y1)-(x1*y3)));
	if(A > 0){
	return A;
	}else{
		return(-1)*A;
	}
}
int main(){
	long long t, x , y, x1, y1, x2, y2, x3, y3, triA, A1, A2, A3, sum;
	cin >> t;
	cin.ignore();
	for(int index = 0; index < t; index++){
		cin >> x1 >> y1 >> x2 >> y2 >> x3 >> y3 >> x >> y;
		triA = area(x1, y1, x2, y2, x3, y3);
		A1 = area(x ,y, x1, y1, x2, y2);
		A2 = area(x, y, x2, y2, x3, y3);
		A3 = area(x, y, x1, y1, x3, y3);
		sum = A1 + A2 + A3;
		if(triA == sum){
			cout << "INSIDE" << endl;
		}else{
			cout << "OUTSIDE" << endl;
		}
	}
}
