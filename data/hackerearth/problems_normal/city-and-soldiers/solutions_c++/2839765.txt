#define MAX 100005//maximum value of n goes here!!

int n;
ll parent[MAX], rnk[MAX], cnt[MAX], curMin, curMax;

void make_set(int n)
{
    for(int i = 0; i < n; i++)
    {
        rnk[i] = 1;
        parent[i] = i;
    }
}
int find(int i)
{
    if(parent[i] != i)
        parent[i] = find(parent[i]);
    return parent[i];
}
void Union(int x, int y)
{
    int xroot = find(x);
    int yroot = find(y);
    if(xroot == yroot)
        return;
    parent[xroot] = yroot;
}
