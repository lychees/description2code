#include<iostream>
#include<math.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long num;
		long diff=0,diff1=0;
		cin>>num;
		long long rootnum=sqrt(num);
		if(rootnum*rootnum==num)
		cout<<"YES"<<endl;
		else if(rootnum*rootnum<num)
		{
		diff=num-rootnum*rootnum;
		diff1=(rootnum+1)*(rootnum+1)-num;
		if(diff<diff1)
		cout<<"-"<<diff<<endl;
		else
		cout<<"+"<<diff1<<endl;
	   }
	}
	return 0;
}
