#include <iostream>
#include<cstring>
#include<cstdlib>
using namespace std;
int a[1033][1033];
int cur=1;
void fill(int lr,int lc, int rr, int rc){
	//cout<<lr<<" "<<lc<<" "<<rr<<" "<<rc<<endl;
	if(lr>rr || lc>rc)
		return;
	if(lr==rr && lc==rc){
		if(a[lr][lc]==-1){
			a[lr][lc] = cur;
			cur++;
			return;
		}
		else
			return;
	}
	//if(lr<=rr/2 && lc<=rc/2)
	fill(lr,lc,(lr+rr)/2,(lc+rc)/2);
//	if(lr<=rr/2 && (lc/2+1)<=rc)
	fill(lr,(lc+rc)/2+1,(lr+rr)/2,rc);
//	if((lr/2+1)<=rr && lc<=rc/2)
	fill((lr+rr)/2+1,lc,rr,(lc+rc)/2);
//	if((lr/2+1)<=rr && (lc/2+1)<=rc)
	fill((lr+rr)/2+1,(lc+rc)/2+1,rr,rc);
}
int main()
{
    int n;
    cin>>n;
    for(int i=1;i<=1<<n;i++)
    	for(int j=1;j<=1<<n;j++)
    		a[i][j]=-1;
    fill(1,1,1<<n,1<<n);
    //cout<<"asdasd";
    for(int i=1;i<=1<<n;i++){
    	for(int j=1;j<=1<<n;j++){
    		cout<<a[i][j]<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
