#include <bits/stdc++.h>

using namespace std;
int main()
{
	int t,i,j,n,flag;
	cin >> t;
	int a[100001];
	for(i=2;i<=100000;i++)
	{
		if(a[i]==-1)
			continue;
		a[i]=0;
		for(j=2;j<=(100000/i);j++)
		{
			a[j*i]= -1;
		}
	}
	while(t--)
	{
		cin >> n;
		flag=0;
		for(i=2;i<n;i++)
		{
			if(a[i]!=-1 && a[n-i]!=-1){
				cout << "Deepa" << endl;
				flag=1;
				break;
			}
		}
		if(flag!=1)
			cout << "Arjit" << endl;
	}
	return 0;
}
