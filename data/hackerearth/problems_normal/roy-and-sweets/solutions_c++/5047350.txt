#include <bits/stdc++.h>
using namespace std;

int main(){
	int n;
	cin>>n;
	vector< pair<int, int> >sweets;
	for(int i=0; i<n; i++){
		int s, q;
		cin>>s>>q;
		sweets.push_back(make_pair(s, q));
	}
	sort(sweets.begin(), sweets.end());
	int hash[1000005];
	memset(hash, 0, sizeof(hash));
	int maxi=0;
	for(int i=n-1; i>=0; i--){
		maxi = max(maxi, sweets[i].second);
		hash[sweets[i].first] = maxi;
	}
	for(int i=1000000; i>=0; i--){
		if(hash[i]==0 && i<1000000) hash[i]=hash[i+1];
	}

	int g;
	cin>>g;
	int members[g];
	long long r=0;
	for(int i=0; i<g; i++){
		cin>>members[i];
		r=r+(long long)(100*(hash[members[i]]/members[i]));
	}
	cout<<r<<endl;
}
