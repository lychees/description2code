#include<bits/stdc++.h>
using namespace std;
int c1[26],c2[26];
int main()
{
    int T,m,n,l,i;
    string str1,str2;
    cin>>T;
    getchar();
    while(T--)
    {
        getline(cin,str1);
        getline(cin,str2);
        i=0;
        while(i<str1.length())
        {
            if(str1[i]!=' ')
                c1[str1[i]-97]++;
            i++;
        }
        i=0;
        while(i<str2.length())
        {
            if(str2[i]!=' ')
                c2[str2[i]-97]++;
            i++;
        }
        for(i=0;i<26;i++)
        {
            if(c1[i]>c2[i])
            {
                c1[i]=c1[i]-c2[i];
                c2[i]=0;
            }
            else if(c1[i]<c2[i])
            {
                c2[i]=c2[i]-c1[i];
                c1[i]=0;
            }
            else
                c1[i]=c2[i]=0;
        }
        int sum1=0,sum2=0;
        for(i=0;i<26;i++)
        {
            sum1+=c1[i];
            sum2+=c2[i];
        }
        if(sum2==0 && sum1!=0)
            cout<<"You win some.\n";
        else if(sum1==0 && sum2!=0)
            cout<<"You lose some.\n";
        else
            cout<<"You draw some.\n";

        for(i=0;i<26;i++)
            c1[i]=c2[i]=0;


    }
}
