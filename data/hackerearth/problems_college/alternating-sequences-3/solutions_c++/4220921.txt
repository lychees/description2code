ll gcd(ll a,ll b)
{
    if(b==0)
      return a;
    return gcd(b,a%b);
}
 
ll power(ll a,ll n)
{
    if(n==0)
      return 1;
    ll x=power(a,n/2);
    x=(x*x)%mod;
    if(n%2!=0)
    {
      x=x*a;
      x=x%mod;
    }
    return x;
}

ll dp[52][52][52][4];
int n;
string s;
ll rec(int i,int a,int b,int flag)
{
    ll ans=0;
    if(i>=n)
    {
        if(a==0 && b==0 && flag==3)
            return 1;
        return 0;
    }
    if(dp[i][a][b][flag]!=-1)
        return dp[i][a][b][flag];
    if(flag==0)
    {
        while(i<n && s[i]!='a')
            i++;
        if(i<n)
        {
            ans+=rec(i+1,a+1,b,flag);
            ans+=rec(i+1,a,b,flag);
            ans+=rec(i+1,a+1,b,flag+1);
        }
    }
    else if(flag==1)
    {
        while(i<n && s[i]!='b')
            i++;
        if(i<n)
        {
            ans+=rec(i+1,a,b+1,flag);
            ans+=rec(i+1,a,b,flag);
            ans+=rec(i+1,a,b+1,flag+1);
        }
    }
    else if(flag==2)
    {
        if(a==0)
            ans+=rec(i,a,b,flag+1);
        else 
        {
            while(i<n && s[i]!='c')
                i++;
            if(i<n)
            {
                ans+=rec(i+1,a-1,b,flag);
                ans+=rec(i+1,a,b,flag);
            }
        }
    }
    else
    {
        if(b==0)
            ans+=1+rec(i,0,0,0);
        else 
        {
            while(i<n && s[i]!='d')
                i++;
            if(i<n)
            {
                ans+=rec(i+1,a,b-1,flag);
                ans+=rec(i+1,a,b,flag);
            }
        }
    }
    ans=ans%mod;
    //cout<<i<<" "<<flag<<" ans= "<<ans<<" "<<s[i]<<endl;
    dp[i][a][b][flag]=ans;
    return ans;
}

    int dp[n];
    memset(dp,-1,sizeof(dp));
    dp[0]=1;
    for(i=1;i<n;i++)
    {
        k=0;
        for(j=0;j<i;j++)
        {
            if(abs(v[i])>abs(v[j]))
            {
                if(v[j]>0 && v[i]<0)
                    k=max(k,dp[j]);
                else if(v[j]<0 && v[i]>0)
                    k=max(k,dp[j]);
            }
        }
        dp[i]=k+1;
        //cout<<i<<" "<<dp[i]<<endl;
    }
    k=0;
    for(i=0;i<n;i++)
        k=max(k,dp[i]);
