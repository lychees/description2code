
tc = int(raw_input())
while tc > 0:
    enter = raw_input()
    i = 0
    output=[]
    while i <= len(enter)-1:
        if i == len(enter) - 1:
            # print "last"
            output.append(enter[i])
            break
        if enter[i] == enter[i+1]:
            # print enter[i],
            # print i
            i += 1
            continue
        else:
            output.append(enter[i])
        i += 1
    print "".join(map(str, output))
    tc -= 1
