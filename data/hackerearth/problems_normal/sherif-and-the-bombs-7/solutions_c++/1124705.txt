#include <cstdio>
#include <algorithm>
using namespace std;

int main() {
    int n, x[12], y[12], r[12];
    scanf("%d", &n);
    for (int i = 0; i < n; i++)
        scanf("%d %d %d", &x[i], &y[i], &r[i]);
    int ans = 0;
    for (int i = 1; i <= 1000; i++) {
        for (int j = 1; j <= 1000; j++) {
            int cnt = 0;
            for (int k = 0; k < n; k++) {
                int dx = i - x[k], dy = j - y[k];
                if (dx*dx + dy*dy <= r[k]*r[k])
                    cnt++;
            }
            if (cnt >= 2)
                ans++;
        }
    }
    printf("%d\n", ans);
}
