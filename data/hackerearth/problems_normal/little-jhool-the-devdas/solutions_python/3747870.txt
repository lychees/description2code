l = raw_input()
even = []
odd  = []
l = [int(i) for i in l.split()]

for e in l:
	if int(e)%2==0:
		even.append(int(e))
	else:
		odd.append(int(e))

if len(odd)>0 and len(odd)%2 ==0:
	odd.remove(min(odd))

ans = (sum(even)+sum(odd))

if ans%2 == 0:
	print ":("
else:
	print ans
