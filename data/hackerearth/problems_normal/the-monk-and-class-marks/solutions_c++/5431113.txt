#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,i,m;string str;
	cin>>n;
	vector< pair<int,string> > v;
	for(i=0;i<n;i++)
	{
		cin>>str>>m;
		v.push_back(make_pair(-m,str));
	}
	sort(v.begin(),v.end());
	for(i=0;i<n;i++)
	{
		cout<<v[i].second<<" "<<-1*v[i].first<<endl;
	}
	return 0;
}
