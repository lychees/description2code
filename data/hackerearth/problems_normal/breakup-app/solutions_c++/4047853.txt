#include<bits/stdc++.h>
using namespace std;
int a[32];

int main()
{
    memset(a,0,sizeof(a));
    int n;
    bool flag;
    scanf("%d",&n);
    while(n--)
    {
        flag=false;
        string s;
        s.clear();
        int x;
        scanf("%d",&x);
        getline(cin,s);
        string v;
        v.clear();
        int i=0;
        if(s[0]=='G')
            flag=true;
        while(i<s.length())
        {
            if(isdigit(s[i])==true)
            {
                while(isdigit(s[i])==true&&i<s.length())
                {
                    v=v+s[i];
                    i++;
                    //cout<<"v="<<v<<"\n";
                }
                int d;
                if(v.length()==1)
                    d=(v[0]-48);
                else
                    d=(v[0]-48)*10+(v[1]-48);
                if(flag==true)
                    a[d]=a[d]+2;
                else
                    a[d]++;
                v.clear();
            }
            i++;
        }
        //for(int j=0;j<32;j++)
            //printf("%d ",a[j]);
       // printf("\n");
    }
        int m,index;m=-1;
        bool e=true;
        for(int j=0;j<32;j++)
        {
            if(a[j]>m)
            {
                m=a[j];
                index=j;
                e=true;
            }
            else if(a[j]==m)
            {
                e=false;
            }
        }
        if(index==19||index==20)
        {
            if(e==true)
                printf("Date\n");
            else
                printf("No Date\n");
        }
        else
            printf("No Date\n");
    return 0;
}
