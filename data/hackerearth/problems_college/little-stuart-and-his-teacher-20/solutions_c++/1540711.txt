inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}
int phi(int n){
    int result = n; 
    for(int i=2;i*i<=n;i++) { 
        if(n%i==0)result-=result/i; 
        while(n%i==0)n/=i; 
    } 
    if(n>1)result-=result/n; 
    return result; 
}
