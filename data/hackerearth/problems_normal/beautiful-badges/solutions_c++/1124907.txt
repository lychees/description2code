#include <bits/stdc++.h>

using namespace std;

int s[1010][3];

int main() {
	int n;
	scanf("%d", &n);
	for(int i = 0; i<n; i++) {
		int c, siz;
		scanf("%d %d", &c, &siz);
		s[siz][c]++;
	}
	int r = 0;
	for(int i = 1000; i>0; i--) {
		for(int j = 1000; j>0; j--) {
			if(2*i*i > j*j && 2*j*j > i*i) {
				int x = min(s[i][1], s[j][2]);
				r += x;
				s[i][1] -= x;
				s[j][2] -= x;
			}
		}
	}
	printf("%d", r);
	return 0;
	
}
