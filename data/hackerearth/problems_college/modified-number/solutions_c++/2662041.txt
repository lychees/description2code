#include <iostream>
using namespace std;

bool kaprekar(long int n)
    {
   long int sq=n*n;
    string str=to_string(sq);
    if(sq<=9)
        str = "0"+str; 
    int l=str.size();
    string left= str.substr(0,l/2);
    string right=str.substr(l/2,l-1);
   // cout<<left<<" "<<right<<endl;
    int lt=atoi(left.c_str());
    int r=atoi(right.c_str());
    if(lt+r==n)return true;
    else return false;
    
}

int main() {
    int p,q,count=0;
   cin>>p>>q;
     for(long int i=p;i<=q;i++)
       {
        if(kaprekar(i)){cout<<i<<" ";count++;}
  
    }
    kaprekar(9);
    if(count==0)cout<<"INVALID RANGE";
    return 0;
}
