#include <bits/stdc++.h>
using namespace std;

int main()
{
    int a[100005],i,j,k,n,x,q;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
    	if(i==0)
    	{
    		scanf("%d",&a[i]);
    	}
    	else
    	{
    		scanf("%d",&a[i]);
    		a[i]+=a[i-1];
    	}
    }
    scanf("%d",&q);
    while(q--)
    {
    	scanf("%d",&x);
    	int ans=lower_bound(a,a+n,x)-a;
    	ans+=1;
    	printf("%d\n",ans);
    }
    return 0;
}
