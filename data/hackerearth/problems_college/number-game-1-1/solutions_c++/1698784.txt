#include <iostream>
using namespace std;

int main()
{
   int n;
   cin>>n;
   while(n--)
   {
   	int k;
   	cin>>k;
   		int count=0;
   	
   	while(k!=1)
   	{
   	
   		if(k%3==0)
   		{
   			k=k/3;
   			count++;
   		}
   		else if(k%2==0)
   		{
   		k=k/2;	
   		
   		count++;
   		}
   		else 
   		{
   			k--;
   			count++;
   		}
   		
   	}
   	cout<<count<<"\n";
   	
   }
    return 0;
}
