#include<iostream>
#include<math.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int a,b,c;
		cin>>a>>b>>c;
		int u=pow(a,b);
		if(b<0)
		{
			if((c-u)<=u)
			{
				cout<<c<<endl;
			}
			else
			{
				cout<<"0\n";
			}
		}
		else
		{
		
		int p=0,j,q;
		for(int i=0;p==0;i++)
		{
			j=c*i;
			if(j>u)
			{
				q=i;
				p++;
			}
		}
		if((j-u)<(u-(c*(q-1))))
		{
			cout<<c*q<<endl;
		}
		else
		{
			cout<<c*(q-1)<<endl;
		}
	    }
	
	}
	return 0;
}
