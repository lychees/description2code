#include <iostream>
using namespace std;

int main()
{
    int T, N, parity;
    cin>>T;
    while(T--)
    {
    	cin>>N;
    	parity=0;
    	while(N!=0)
    	{
    		if((N & 1)==1)
    		{
    			parity++;
    		}
    		N=N>>1;
    	}
    	if(parity%2==0)
    	{
    		cout<<"even"<<endl;
    	}
    	else
    	{
    		cout<<"odd"<<endl;
    	}
    }
    return 0;
}
