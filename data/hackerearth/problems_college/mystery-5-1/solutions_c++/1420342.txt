#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
int main()
{
    int t,i,k,a[26],len;
    string s;
    cin>>t;
    while(t--)
    {
    	for(i=0;i<26;i++)a[i]=0;
        cin>>s;
        len =s.length();
        for(i=0;i<len;i++)
        {
            k=s[i];
            k=k%97;
            a[k]++;
        }
        sort(a,a+26);
        cout<<a[25]<<"\n";
    }
    return(0);
}
