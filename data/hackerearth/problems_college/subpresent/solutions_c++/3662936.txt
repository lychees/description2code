#include<bits/stdc++.h>
using namespace std;
int main()
{
    string b,s;
    int t,k,c=0;
    cin>>b>>k>>t;
    int bl = b.size();
    while(t--) {
        cin>>s;
        int sl = s.size();
        int bp =0,sp =0;
        while(bp!=bl && sp !=sl) {
            if(b[bp] == s[sp]) {
                bp++;
                sp++;}
            else
                bp++;
        }
        if(sp == sl)
                c++;

    }
    if(c>=k) {
        cout<<c<<" "<<"LUCKY";
    }
    else
        cout<<c<<" "<<"UNLUCKY";
    return 0;

}
