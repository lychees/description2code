#include <iostream>
#include <string>
#include <vector>

using namespace std;

int FindMinMoves(vector<string> B) 
{
  	int R = B.size(), C = B[0].size();
  	bool on[R][C];

  	for (int r=0; r<R; r++) 
  	{
  		for (int c=0; c<C; c++)
  		{
  			on[r][c] = (B[r][c]=='1');
  		}
  	}
  	int res = 0;

  	for (int r=R-1; r>=0; r--)
  	{
  	 	for (int c=C-1; c>=0; c--)
  	 	{ 
  	 		if (!on[r][c]) 
  	 		{
    			for (int x=0; x<=r; x++) for (int y=0; y<=c; y++) on[x][y] = !on[x][y];
    			res++;
 			}
 		}
 	}
  	return res;
}

int main()
{
	int T;
	cin>>T;

	for(;T--;)
	{
		int N,M;
		cin>>N>>M;

		vector<string> S;
		string s; 
		for (int i = 0; i < N; ++i)
		{
			cin>>s;
			S.push_back(s);
		}
		cout<<FindMinMoves(S)<<"\n";
	}
	return 0;
}
