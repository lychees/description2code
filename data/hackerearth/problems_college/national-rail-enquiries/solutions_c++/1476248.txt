#include <iostream>
#include <algorithm>
using namespace std;
long long int gcd(long int a,long int b) {
	return (b == 0 ? a : gcd(b,a%b));
}
long long int fact(long int n) {
	if (n == 1 || n == 0)
		return 1;
	else
		return n*fact(n-1);
}
int main()
{
    int t;
    scanf("%d",&t);
    int ctr = 1;
    while (t--) {
    	long long int n;
    	scanf("%lld",&n);
    	if (n <= 110) {
    	long long int ans = fact(n-1)%110;
    	//long long int ans = 1;
    	//for (long int i = 2; i < n; i++) {
    	//	ans = ans*i;
    	//}
    	//ans = ans%110;
    	ans = ans%110;
    	if  (ans > 0 &&  gcd(ans,n) == 1)
    		printf("Case %d:Yes\n",ctr);
    	else
    		printf("Case %d:No\n",ctr);
    	}
    	else {
    		printf("Case %d:No\n",ctr);
    	}
    	ctr++;
    }
    return 0;
}
