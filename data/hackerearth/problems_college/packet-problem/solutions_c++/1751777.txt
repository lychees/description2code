#include <bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,m;
	cin>>t;
	long long int a[100][2];
	a[1][0] = 0;
	a[1][1] = 1;
	a[2][0] = 1;
	a[2][1] = 1;
	for(int i = 3; i <= 90; i++) {
		a[i][0] = a[i-1][0]+a[i-2][0];
		a[i][1] = a[i-1][1]+a[i-2][1];
	}
	while(t--) {
		cin>>m;
		
		cout<<a[m][0]<<" "<<a[m][1]<<endl;
	}
    return 0;
}
