#include<bits/stdc++.h>

using namespace std;

long long a[10],pwr[10];

long solve(long n)
{
    if(n<3)
        return 0;
    if(n<10)
        return 1;
    else
    {
        int k=log10(n);

        long tmp=n/pwr[k];

    if(tmp<3)
            return tmp*a[k]+solve((n%pwr[k]));
        else if (tmp==3)
        return (tmp)*a[k]+n%pwr[k];
        else
            return (tmp-1)*a[k] + pwr[k] + solve(n%pwr[k]);

    }


}

int main()
{

    a[1]=1;
    pwr[1]=10;
    long x=1;
    for(int i=2;i<10;i++)
    {
        pwr[i]=pwr[i-1]*10;
        x*=10;
        a[i]=a[i-1]*9+x;

    }
    int t;
    long long n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        cout<<n-solve(n)<<"\n";
    }
}
