int readint()
{
    int t=0;
    char c,ch;
    ch=getchar();
    if (ch=='-') c=getchar();
    else c=ch;
    while(c<'0' || c>'9')
        c=getchar();
    while(c>='0' && c<='9')
    {
        t=(t<<3)+(t<<1)+c-'0';
        c=getchar();
    }
    if (ch=='-' ) return -t;
    else return t;
}
ll powerFast(ll b,ll p)
{
   long long int x=1,r;
   x=1;
   while(p>0)
   {
       r=p%2;
       p=p/2;
       if (r==1) x=(x*b)%mod;
       b=(b*b)%mod;
   }
    return x;
}
ll invmod(int a)
{
    return powerFast(a,mod-2);
}
ll fact(int n)
{
    ll ans=1,i;
    fori(1,n)
    {
        ans=(ans*i)%mod;
    }
    return ans;
}
ll ifact(int n)
{
    ll prev=1,i;//ifact[i-1];
    fori(1,n)
    {
        prev=prev*invmod(i)%mod;
    }
    return prev;
}
