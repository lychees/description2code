int main()
{
    int t;
    cin >> t;
    while(t--){
        int n,m;
        cin >> n >> m;
        vector< vector< vector<PII> > > adj;
        adj.resize(n+1);
        for(int i=0; i<=n; i++)
            adj[i].resize(m+1);
        vector<string> str(n);
        vector<vector<bool> > vis(n+1, vector<bool>(m+1, false));
        for(int i=0; i<n; i++)
            cin >> str[i];
        for(int i=0; i<=n; i++)
            for(int j=0; j<=m; j++){
                if(i>=1 and j>=1 and j<m and str[i-1][j-1]!=str[i-1][j])
                    adj[i][j].pb({i-1, j});
                if(i>=1 and i<n and j<m and str[i-1][j]!=str[i][j])
                    adj[i][j].pb({i, j+1});
                if(j>=1 and  j<m and i<n and str[i][j-1]!=str[i][j])
                    adj[i][j].pb({i+1, j});
                if(i>=1 and j>=1 and i<n and str[i-1][j-1]!=str[i][j-1])
                    adj[i][j].pb({i, j-1});
            }
        for(int i=0; i<m; i++){
            adj[0][i].pb({0, i+1});
            adj[0][i+1].pb({0, i});
            adj[n][i].pb({n, i+1});
            adj[n][i+1].pb({n, i});
        }
        for(int i=0; i<n; i++){
            adj[i][0].pb({i+1, 0});
            adj[i+1][0].pb({i, 0});
            adj[i][m].pb({i+1, m});
            adj[i+1][m].pb({i, m});
        }
        stack<PII> mys;
        mys.push({0,0});
        while(not mys.empty()){
            PII p=mys.top();
            mys.pop();
            for(auto &i: adj[p.F][p.S])
                if(not vis[i.F][i.S])
                    vis[i.F][i.S]=true, mys.push(i);
        }
        string ans="YES";
        for(int i=0; i<=n; i++)
            for(int j=0; j<=m; j++)
                if(not adj[i][j].empty() and vis[i][j]==false)
                    ans="NO";
        cout << ans << endl;
    }
    return 0;
}
