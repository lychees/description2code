#include<iostream>
#define N 100002
using namespace std;

int hash_[N];
int arr[N];

int root(int a)
{
    while(a!=arr[a])
    {
        a=arr[a];
    }
    return a;
}

void union_(int a,int b)
{
    int root_a=root(a);
    int root_b=root(b);
    if(root_a == root_b)
    return;
    arr[root_a]=arr[root_b];
}

int main()
{
    ios::sync_with_stdio(false);
    int n,q,i,j,x,y,z;
    cin>>n>>q;
    for(i=1;i<=n;i++)
    {arr[i]=i;  hash_[i]=i;}
    while(q--)
    {
        cin>>x;
        if(x==1)
        {
            cin>>y>>z;
            union_(y,z);
        }
        else if(x==2)
        {
            cin>>y;
            hash_[root(y)]=y;
        }
        else
        {
            cin>>y;
            cout<<hash_[root(y)]<<endl;
        }
    }
    return 0;
}
