#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d\n",&t);
    while(t--){
    	string s,p;
    
    	cin>>s>>p;
    	int j=0;
    	for(int i=0;s[i];i++){
    		if(s[i]==p[j]){
    			j++;
    		}
    	}
    	if(j >= p.length()) printf("YES\n");
    	else printf("NO\n");
    }
    return 0;
}
