#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int px,py,bx,by;
    while(t--)
    {
    	cin>>px>>py>>bx>>by;
    	if((px >= bx && py >=by) || (py >= bx && px >= by) )
    	cout<<"Possible"<<endl;
    	else
    	cout<<"Not Possible"<<endl;
    	
    }
    return 0;
}
