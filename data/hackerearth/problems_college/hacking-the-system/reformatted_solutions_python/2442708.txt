a = int(raw_input())
b = int(raw_input())

# a,b = raw_input().split()
a = list(str(a))
b = list(str(b))

a = list(map(int, a))
b = list(map(int, b))

if len(a) != len(b):
    print('loses')
else:
    count = 0
    for i in range(len(a)):
        if int(a[i]) != int(b[i]):
            count += 1
            if count == 2:
                break

    if count <= 1:
        print('wins')
    else:
        print('loses')
