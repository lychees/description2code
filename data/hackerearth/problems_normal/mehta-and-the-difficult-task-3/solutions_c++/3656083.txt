#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define inf 1000000000000
#define mod 1000000007
#define pb push_back
#define mp make_pair
#define all(v) v.begin(),v.end()
#define S second
#define F first
#define boost ios::sync_with_stdio(false);

ll id[16][16];

void mul(ll a[16][16],ll b[16][16])
{
	ll c[16][16];
	for(int i=0;i<16;i++)
	{
	    for(int j=0;j<16;j++)
	    {
        	c[i][j]=0;
        	for(int k=0;k<16;k++)
        	{
	            c[i][j]+=(a[i][k]*b[k][j])%mod;
	            c[i][j]%=mod;
        	}
    	}
	}
	for(int i=0;i<16;i++)
	{
    	for(int j=0;j<16;j++)
        a[i][j]=c[i][j];
	}
}
	

int main()
{
	ll t,i,j,n,a,b,c,d,mask,fin_mask,tr[16][16],p,ans;
	cin>>t;
	while(t--)
	{
		cin>>n>>a>>b>>c>>d;
		memset(tr,0,sizeof tr);
		memset(id,0,sizeof id);
		ans=0;
		for(mask=0;mask<(1<<4);mask++)
		{
			tr[mask][mask]=6;
			tr[mask][mask^(1<<0)]=1;
			tr[mask][mask^(1<<1)]=1;
			tr[mask][mask^(1<<2)]=1;
			tr[mask][mask^(1<<3)]=1;
		}
		for(i=0;i<16;i++)
		id[i][i]=1;
		p=n;
		while(p>0)
    	{
	        if(p&1) 
	        mul(id,tr);
	        mul(tr,tr);
	        p/=2;
    	}
    	fin_mask=0;
        if (d)fin_mask|=1;
        if (c)fin_mask|=2;
        if (b)fin_mask|=4;
        if (a)fin_mask|=8;
        ans+=id[fin_mask][0];
      	ans%=mod;
        cout<<ans<<endl;
        
	}
	return 0;
}
