#include <iostream>
#define MAX 10000

using namespace std;

int N;
int A[MAX];

int max_1(int x, int y)
{
    if(x>y) return x;
    else return y;
}

int Algo()
{
    int incl = A[0];
    int excl = 0;
    int tmp;
    for(int i=1; i<N; i++)
    {
        tmp = incl;
        incl = max_1(incl, excl + A[i]);
        excl = tmp;
    }

   if(incl >= excl)
     return incl;
     else
    return excl;
}

int main()
{
    int T;
    cin >> T;

    while(T--)
    {
        cin >> N;
        for(int i=0; i<N; i++)
            cin >> A[i];

        cout << Algo() << endl;
    }

    return 0;
}
