#include <iostream>
#include <algorithm>

using namespace std;

long long abso(long long a)
{
	if (a < 0) return -a;
	else return a;
}

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
    long long a[100000],tmp[100000],ans;
    int n,i,k;
    cin >> n;
    for (i = 0; i < n; i++)
      cin >> a[i];
    sort(a,a + n);
    int mid = (n + 1) / 2;
    int l = 0;
    int r = n - 1;
    k = 0;
    while (mid--)
    {
    	if (l == r)
    	{
    		tmp[k++] = a[l++];
    		break;
    	}
    	tmp[k++] = a[l++];
    	tmp[k++] = a[r--];
    }
    ans = abso(tmp[n-1] - tmp[0]);
    for (i = 1; i < n; i++)
      ans += abso(tmp[i-1] - tmp[i]);
    cout << ans;
    return 0;
}
