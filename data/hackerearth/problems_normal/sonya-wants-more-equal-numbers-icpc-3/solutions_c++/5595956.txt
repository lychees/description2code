#include<iostream>
#include<algorithm>
using namespace std;
int a[105],c[105];

int n,x,dp[105][100005],sx[105][100005];

int main()
{

cin>>n;

   for(int i=0;i<n;i++)
   {
     cin>>a[i];
     c[i]=a[i];
     if(i>0)c[i]+=c[i-1];
   }

   for(int i=0;i<n;i++)
    for(int j=0;j<=1e5;j++)
    sx[i][j]=-1;

   for(int i=0;i<n;i++)
    {for(int j=i;j<n;j++)
       {
          int sum=c[j];
          if(i>0)sum-=c[i-1];
          sx[i][sum]=j;
       }
    }

    for(int i=n-1;i>=0;i--)
    {
        for(int j=0;j<=1e5;++j)
        {
            int sc=sx[i][j];
            if(sc==-1)dp[i][j]=dp[i+1][j];
            else dp[i][j]=1+dp[sc+1][j];
        }
    }
    int mx=0;
    for(int i=0;i<=1e5;i++)
        if(dp[0][i]>mx)mx=dp[0][i];
    cout<<mx<<endl;

    for(int i=0;i<=1e5;i++)
        if(dp[0][i]==mx)cout<<i<<" ";

return 0;
}
