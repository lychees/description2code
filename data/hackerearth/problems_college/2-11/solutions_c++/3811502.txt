#include <iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;
string s1,s2;
int main()
{
	int n,a[27][27],i,j,k;
	cin>>n;
	for(i=1;i<=26;i++)
	for(j=1;j<=26;j++)
	scanf("%d ",&a[i][j]);
	for(k=1;k<=26;k++)
	for(i=1;i<=26;i++)
	for(j=1;j<=26;j++)
	if(a[i][j]>a[i][k]+a[k][j])
	a[i][j]=a[i][k]+a[k][j];
	for(i=0;i<n;i++)
	{
		cin>>s1;
		cin>>s2;
		long int c=0;
		for(j=0;s1[j]!='\0';j++)
		if(s1[j]!=s2[j])
		c+=a[(int)(s2[j]-'a')+1][(int)(s1[j]-'a')+1];
		printf("%ld\n",c);
	}
	return 0;
}
