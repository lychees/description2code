#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int x,y;
		cin>>x>>y;
		if(x==y)
		{
			cout<<-1<<endl;
			continue;
		}
		ll int i,mc=0,z;
		z=x-y;
		for(i=1;(i*i)<=z;i++)
		{
			if((z%i)==0)
			{
				if((x%i)==y)
				mc++;
			}
			if((z%i==0) && (i!=z/i) && (z%(z/i)==0))
			{
				if((x%(z/i))==y)
				mc++;
			}
		}
		cout<<mc<<endl;
	}
	return 0;
}
