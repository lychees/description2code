#include<iostream>
#include<cstdio>
#include<algorithm>
#include<string>
#include<vector>
using namespace std;
int abs(int a,int b)
{
	return max(a-b,b-a);
}
int main()
{
	int m,n,t,p;
	int x1,y1,x2,y2;
	cin>>t;
	while(t--)
	{
		cin>>m>>n;
		cin>>x1>>y1>>x2>>y2>>p;
		int ymin=min(abs(y1-y2),min(abs(y1+n-y2),abs(y2+n-y1)));
		int xmin=min(abs(x1-x2),min(abs(x1+m-x2),abs(x2+m-x1)));
		int ans=(xmin+ymin)*p;
		cout<<ans<<"\n";
		if(ans<=1000)
		cout<<"You saved the group.\n";
		else
		cout<<"Let go of the group."<<"\n";
	}
}
