#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
typedef long long ll;
bool isPrime[1000005];
bool visited[1000005];
int arra[1000005];
int arrb[1000005];
int dp[1000005];
void calc()
{
    int i,j;
    for(i=0;i<1000005;i++)
    {
        arra[i]=0;
        arrb[i]=i;
    }
    for(i=2;i<1000005;i++)
    {
        if(arra[i]==0)
        {
            arra[i]=1;
            arrb[i]=i-1;
            for(j=2*i;j<1000005;j=j+i)
            {
                arra[j]=arra[j]+1;
                arrb[j]=arrb[j]-(arrb[j]/i);
            }
        }
    }
}
int main() {
    calc();
    dp[10]=1;
    for(int n=11;n<1000005;n++) {
        int val1=n-arra[n];
        int val2=n-arrb[n];
        if(val1<10&&val2<10) dp[n]=1;
        if(val1>=10&&dp[val1]==1) dp[n]=0;
        else if(val1>=10&&dp[val2]==1) dp[n]=0;
        else dp[n]=1;
    }
    int t;
    cin>>t;
    while(t--) {
        int n;
        cin>>n;
        //cout<<arra[n]<<" "<<arrb[n]<<" ";
        if(dp[n]==1) cout<<"Bob"<<endl;
        else cout<<"Alice"<<endl;
    }
    return 0;
}
