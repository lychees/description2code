#include <iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main()
{
	int tc;
	cin>>tc;
	vector<int>b;
	vector<int>g;
	while(tc--)
	{
		int n;
		cin>>n;
		g.clear();
	b.clear();
	int x;
	for(int i=0;i<n;i++)
	{
		cin>>x;
		b.push_back(x);
	}
	
	for(int i=0;i<n;i++)
	{
		cin>>x;
		g.push_back(x);
	}
	
	sort(b.rbegin(),b.rend());
	sort(g.begin(),g.end());
		
	
	int count=0;
	for(int i=0;i<n;i++)
	{
		if(!(g[i]%b[i])||!(b[i]%g[i]))
		{
			count++;
		}
	}
	cout<<count<<endl;
    //cout << "Hello World!" << endl;
   } return 0;
}
