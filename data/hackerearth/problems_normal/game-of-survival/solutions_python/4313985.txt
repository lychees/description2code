for i in range(int(raw_input())):
	m = int(raw_input())
	s = []
	st = []
	for i in xrange(m):
		s.append(int(raw_input()))
	for i in xrange(m):
		st.append(int(raw_input()))
	s=sorted(s)
	st=sorted(st)
	g = 0
	for i in range(m):
		if s[i] <= st[i]:
			g = 1
			break
	if g == 1:
		print "NO"
	else:
		print "YES"
