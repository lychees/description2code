#include <iostream>
using namespace std;

int main()
{
    int i,n,x,c=0,d=0;
    cin>>n;
    int a[n];
    cin>>x;
    for(i=0;i<n;i++)
       cin>>a[i];
     for(i=0;i<n;i++)
     {
     	if(a[i]<=x)
     	  c++;
     	  else if(a[i]>x)
     	  {
     	  	d++;
     	  	if(d==1)
     	  	  continue;
     	  	  else if(d>1)
     	  	  break;
     	  }
     }
     cout<<c<<endl;
    return 0;
}
