#include <iostream>
#include <vector>
#include <cstdio>
#include <cmath>
#include <algorithm>

using namespace std;

int main()
{
	int n, p;
	cin>>n>>p;
	double two = p/(double)100;
	double three = 1-p/(double)100;
	vector < double > prob(n+5);
	prob[0]=prob[1]=0;
	prob[2]=two;
	prob[3]=three;
	for(int i=4;i<=n;i++) {
		prob[i]=prob[i-2]*two+prob[i-3]*three;
	}
	printf("%.6lf\n",prob[n]);
    return 0;
}
