#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

bool a[4000000] = {0};

int main()
{
	int k;
	for (int i = 2; i < sqrt(4000000); i++) {
		k = 2;
		if (a[i] == 0) {
			while (i * k < 4000000) {
				a[i*k] = 1;
				k++;
			}
		}
	}
	
	int n;
	unsigned long long int sum = 0;
    
    scanf("%d", &n);
    
    for (int i = 2; i < n; i++) {
    	if (a[i] == 0) {
    		sum += i;
    	}
    }
    
    printf("%lld", sum);
    
    return 0;
}
