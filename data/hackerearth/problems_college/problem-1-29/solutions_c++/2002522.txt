#include <iostream>
#include <string.h>
#define gc getchar_unlocked
#define pc putchar_unlocked
using namespace std;
 int a[50][50];
 
inline void println(int a)
{ 
 int i=0;
char S[20];
if(a==0)
    pc('0');
else{
while(a>0)
{
    S[i++]=a%10+'0';
a=a/10;
}
--i;
while(i>=0)
pc(S[i--]);}
pc('\n');
}
 
 
inline int readInt()
{int n=0;char c=gc();int f=1;
if(c==EOF)
    return -9999;
while((c<'0'||c>'9')&&c!='-')
c=gc();
if(c=='-')
    f=-1;
while(c<='9'&&c>='0'){
n=n*10+c-'0';
c=gc();}
 
return (f*n);
}
int checkpos(int i,int j,int n){
	if(i==(n-1)&&j==(n-1))
		return 1;
	else if(i==n||j==n)
		return 0;
	else{
		if(a[i+1][j]==1&&a[i][j+1]==1)
		return(checkpos(i+1,j,n)+checkpos(i,j+1,n));
		else if(a[i+1][j]==1)
		return(checkpos(i+1,j,n));
		else if(a[i][j+1]==1)
		return(checkpos(i,j+1,n));
		else
		return 0;
	}
	
}
int main(){
	int t,n,i,j;
	t=readInt();
	while(t--){
		n=readInt();
		for(i=0;i<n;i++){
			for(j=0;j<n;j++)
				a[i][j]=readInt();
		}
		if(checkpos(0,0,n)>=1)
			printf("POSSIBLE\n");
		else
			printf("NOT POSSIBLE\n");
	}
}
