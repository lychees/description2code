#include <iostream>
using namespace std;

int n, a, res, num;

int main() {
    ios_base::sync_with_stdio(0);
    cin >> n;
    while (n--) {
        res = 0;
        cin >> a;
        for (int i = 1; i <= a; i++) {
            int tmp_res = 0, cr = i;
            while (cr > 0) {
                tmp_res += cr % 2;
                cr /= 2;
            }
            if (tmp_res >= res) {
                num = i;
                res = tmp_res;

            }
        }
        cout << num << endl;
    }
}
