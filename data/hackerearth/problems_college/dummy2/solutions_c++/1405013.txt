#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;

    do{
            string str,ans,temp;
        t--;
        cin>>str;
        for(int i=0;str[i]!='\0';i++)
        {
            if(isdigit(str[i]))
            {
                temp=ans;
                for(int k=0;k<(str[i]-'0')-1;k++)
                {
                    ans+=temp;
                }
            }
            else
            {
                ans+=str[i];
            }
        }
        cout<<ans<<endl;;
    }while(t>0);

}
