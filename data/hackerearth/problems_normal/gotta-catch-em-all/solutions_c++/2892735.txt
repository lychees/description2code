#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   long long int k,i,j;
   cin>>k;
   long long int sum=0,x;
   vector<long long int> a;
   for(i=0;i<k;i++)
   {
   	cin>>x;
   	a.push_back(x);
   }
   sort(a.begin(),a.end());
   reverse(a.begin(),a.end());
   sum=0;
   for(i=0;i<k;i++)
   {
   	if(a[i]+i+1>sum)
   	sum=a[i]+i+1;
   }
   cout<<sum+1<<endl;
    return 0;
}
