#include <iostream>
#include <set>
#include <algorithm>
using namespace std;

int main()
{
	int t,n,m;
	
    cin>>t;
    while(t--)
    {
    	set<string> s;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		string name;
    		cin>>name;
    		s.insert(name);
    	}
    	set <string>::iterator it;
		for(it=s.begin();it!=s.end();it++)
		{
			cout<< *it<<"\n";
		}
    }
    return 0;
}
