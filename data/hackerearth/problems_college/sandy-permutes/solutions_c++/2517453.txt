#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    int n,k;
    cin >> n >> k;
    vector<int> a(n);
    for(int i=0;i<n;i++)
        cin >> a[i];
    
    for(int i=0;i<n;i++){
        int pos = max_element(a.begin()+i,a.end())-a.begin();
        
        if(k && pos!=i){
            k--;
            swap(a[i],a[pos]);
        }
    }
    for(auto i:a)
        cout << i << " ";
    
    return 0;
}
