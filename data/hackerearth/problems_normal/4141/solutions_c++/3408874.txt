
#include<bits/stdc++.h>

using namespace std;

#define si(x) scanf("%d",&x)
#define pri(x) printf("%d",x)
#define sll(x) scanf("%lld",&x)
#define pll(x) printf("%lld",x)
#define sstr(s) scanf("%s",s)
#define pstr(s) printf("%s",s)
#define nl printf("\n")
#define ll long long int
#define pb push_back
#define mp make_pair

int main()
{
    int t;
    si(t);
    while(t--) {
	int a0,a1,c,n;
	si(a0); si(a1); si(c); si(n);
	if(n==1) {
		if(a0)
	    pstr("NO\n");
	    else 
	    pstr("YES\n");
	    continue;
	}
	if(n==2) {
	    if((a0*10+a1)%41==0) {
		pstr("YES\n");
	    }
	    else {
		pstr("NO\n");
	    }
	    continue;
	}
	else {
	    int r = (a0*10+a1)%41, temp;
	    for(int i=2;i<n;++i) {
		temp = (a1*c+a0)%10;
		r = r*10+temp;
		r = r%41;
		a0=a1;
		a1=temp;
	    }
	    if(r%41==0) {
		pstr("YES\n");
	    }
	    else {
		pstr("NO\n");
	    }
	}
    }
    return 0;
}
