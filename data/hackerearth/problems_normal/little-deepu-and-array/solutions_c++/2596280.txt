#include <bits/stdc++.h>

#define ll long long int

using namespace std;

int main(int argc,char **argv)
{
	int t;
	scanf("%d",&t);
	vector<ll> v;
	for(int i=0;i<t;++i)
	{
		ll e;
		scanf("%lld",&e);
		v.push_back(e);
	}
	int m;
	scanf("%d",&m);
	while(m--)
	{
		ll x;
		scanf("%lld",&x);
		for(int i=0;i<v.size();++i)
		{
			if (v[i] > x)
			{
				v[i] -= 1;
			}
		}
    }
    for(int i=0;i<v.size();++i)
    {
    	printf("%d ",v[i]);
    }
    return 0;
}
