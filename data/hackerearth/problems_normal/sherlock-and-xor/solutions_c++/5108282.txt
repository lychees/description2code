#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    int n;
    int even=0,odd=0;
    cin>>t;
    for(int i=0;i<t;i++)
       {
          cin>>n;
          int arr[n];
          for(int j=0;j<n;j++)
             cin>>arr[j];
          sort(arr,arr+n);
           even=0;
           odd=0;
          for(int k=0;k<n;k++)
            {
               if(arr[k]%2==0)
                 even++;
               else
                  odd++;
            	 
           
            }
          unsigned int temp=even*odd;  
          cout<<temp<<endl;  
            
           // cout<<count<<endl;
       }
    return 0;
}
