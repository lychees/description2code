int mult(int x,int y)
{
  ll ans,x1=(ll)x,y1=(ll)y;
  ans=(x1*y1)%mi;
  return (int)ans;
}
int gcd(int a,int b) { return b==0 ? a : gcd (b,a%b);}
int lcm(int a,int b) { return a*(b/gcd(a,b));}

ll pow1(ll a,ll b)
{
  ll ans=1;
  while(b>0)
    {
      if(b&1) ans=(ans*a)%mi;	
      a=(a*a)%mi; b>>=1;
    }	
  return ans;
}

}

vector<vi> v;
int c[100005],e[100005],dp[100005],dpinv[100005];
void dfs(int u,int par)
{
  if(v[u].size()==1 && u!=1)
    {
      //trace1(u);
      if(e[u]==c[u])
	dp[u]=0,dpinv[u]=1;
      else
	dp[u]=1,dpinv[u]=0;
      return;
    }
  int sum1=0,sum2=0;
  for(vi::iterator itr=v[u].begin();itr!=v[u].end();itr++)
    {
      int v1=*itr;
      if(v1!=par)
	{
	  dfs(v1,u);
	  sum1+=dp[v1];
	  sum2+=dpinv[v1];
	}
    }
  if(c[u]==e[u])
    {
      dp[u]=min(sum1,sum2+2);
      dpinv[u]=min(sum1+1,sum2+1);
    }
  else
    {
      dp[u]=min(sum1+1,sum2+1);
      dpinv[u]=min(sum1+2,sum2);
    }
}

    si(c[i]);
  rep(i,1,n+1)
    si(e[i]);
  dfs(1,-1);

}
