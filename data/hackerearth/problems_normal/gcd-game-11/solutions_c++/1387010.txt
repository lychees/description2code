#include <iostream>
#include <algorithm>
#define WIN 1
#define LOSE 0
#define DRAW 2
using namespace std;
int dp[1005][1005][2];
int fun(int a,int b,bool flag)
{
	int x=0,g,ans;
	g=__gcd(a,b);
	ans=LOSE;
	if(a==1 && b==1)
		return DRAW;	
	if(dp[a][b][flag]!=-1)
		return dp[a][b][flag];
	if(flag==0)
	{
		if(a==1)
			return 0;
		if(b-1>0)
			x=fun(a,b-1,1);
		if(x==LOSE)
			ans=WIN;
		else if(x==DRAW)
			ans=DRAW;
		if(ans!=WIN && g>1)
		{
			x=fun(a,b/g,1);
			if(x==LOSE)
				ans=WIN;
			else if(x==DRAW)
				ans=DRAW;
		}
	}
	else if(flag==1)
	{
		if(b==1)
			return 0;
		if(a-1>0)
			x=fun(a-1,b,0);
		if(x==LOSE)
			ans=WIN;
		else if(x==DRAW)
			ans=DRAW;
		if(ans!=WIN && g>1)
		{
			x=fun(a/g,b,0);
			if(x==LOSE)
				ans=WIN;
			else if(x==DRAW)
				ans=DRAW;
		}
	}
	return dp[a][b][flag]=ans;
}
int main()
{
	int t,a,b,x;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d%d",&a,&b);
    	for(int i=0;i<=a;i++)
    		for(int j=0;j<=b;j++)
    			dp[i][j][0]=dp[i][j][1]=-1;
    	x=fun(a,b,0);
    	if(x==1)
    		puts("Arjit");
    	else if(x==0)
    		puts("Chandu Don");
    	else
    		puts("Draw");
	}
	return 0;
}
