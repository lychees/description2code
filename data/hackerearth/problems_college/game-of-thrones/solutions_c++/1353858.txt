#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
	long long T,N,c,i;
	cin>>T;
	while(T--)
	{
		c=0;
		cin>>N;
		long long A[N];
		for(i=0;i<N;i++)
			cin>>A[i];
		sort(A,A+N);
		for(i=0;i<N-1;i++)
		{
			if(A[i]==A[i+1])
			{
				c++;				
			}
		}
		cout<<c<<"\n";	
	}
	return(0);
}
