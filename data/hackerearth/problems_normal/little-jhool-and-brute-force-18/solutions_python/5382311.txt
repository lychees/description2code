width = 100
book = {}
for a in xrange(width - 1):
    for b in xrange(a + 1, width):
        num = a**3 + b**3
        if num not in book:
            book[num] = 0
        book[num] += 1
jst_nums = [k for k, v in book.iteritems() if v > 1]
jst_nums.sort()
for _ in xrange(input()):
    n = input()
    result = -1
    for x in jst_nums:
        if x <= n:
            result = x
    print result
