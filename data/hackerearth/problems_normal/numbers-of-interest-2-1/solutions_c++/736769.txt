#include<bits/stdc++.h>
using namespace std;

vector<long long> f(1000005);
vector<long long> g(1000005);
vector<vector<int> > factors(1000005);
int  a[1000001]={0};
int prime[1000005];
int p=1;
void generate(int y)
    {int i,j;
     prime[0]=2;
	 for(i=3;i*i<=y;i+=2)
        {if(a[i])
          continue;
         prime[p++]=i;
		for(j=i*i;j<=y;j+=2*i)
           {a[j]=1;
            factors[j].push_back(i);
           }
        }
        for(i;i<=y;i+=2)
          if(!a[i])
            prime[p++]=i;
     //   for(i=0;i<p;i++)
             ;//cout<<prime[i]<<" ";
    }
    
int main()
 {generate(1000000);
 	
 	int t,i,j,n;
 	
 	for(i=2;i<=1000000;i++)
 	  {
 	  	if(i%2==0)
 	  	  f[i]=2;
 	  	else
 	  	  {
 	  	   if(factors[i].empty())
 	  	  	  f[i]=i;
 	  	  	else
 	  	  	  f[i]=factors[i][0];
 	  	  }
 	  	f[i]=f[i]+f[i-1];
 	  }
 	for(i=2;i<=1000000;i++)
 	  {//cout<<"factors for "<<i<<endl;
 	  	int copy=i,copy2=i;
 	  	long long ans=1;
 	  	long long ptr=2,temp=1;
 	  	while(copy%2==0)
 	  	  {
 	  	   copy/=2;
 	  	   copy2/=2;
 	  	   temp+=ptr;
 	  	   ptr*=2;
 	  	  }
 	  	ans*=temp; 
 	    for(j=0;j<factors[copy].size();j++)
 	      {
 	      	int fact=factors[copy][j];
 	      //	cout<<fact<<" ";
 	      	temp=1;
 	      	ptr=fact;
 	      	while(copy2%fact==0)
 	  	  		{
 	  	   			copy2/=fact;
 	  	   			temp+=ptr;
 	  	  			 ptr*=fact;
 	  	  	    }
 	  			ans=(temp*ans); 
 	      	
 	      }
 	      if(copy2!=1)
 	        {
 	        	ans*=(1+copy2);
 	        }
 	  	
 	  	g[i]=(ans+g[i-1])-i;
 	  	//cout<<endl;
 	  }
 	cin>>t;
 	while(t--)
 	 {
 	 
 		cin>>n;
 		//cout<<f[n]<<" "<<g[n]<<" ";
 		cout<<(f[n]+g[n])%n<<endl;
    }
 }
