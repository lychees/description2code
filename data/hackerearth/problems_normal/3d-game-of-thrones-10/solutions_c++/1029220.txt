#include <cstdio>
#include <cmath>
#include <cstring>

int main(){
	int t, n, x, y, z, res;

	scanf("%d", &t);
	while (t--){
		scanf("%d", &n);
		res = 0;
		scanf("%d%d%d", &x, &y, &z);
		res = (x ^ y ^ z);
		for (int i = 1; i < n; i++){
			scanf("%d%d%d", &x, &y, &z);
			res ^= (x ^ y ^ z);
		}
	
		printf("%s\n", res ? "Stark" : "Lannister");
	}
	
	return 0;
}
