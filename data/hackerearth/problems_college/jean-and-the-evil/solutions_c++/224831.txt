#include<cstdio>
#include<algorithm>
#include<climits>
using namespace std;
int maxSubArraySum(int a[], int size)
{
   int max_so_far = 0, max_ending_here = 0;
   int i;
   for(i = 0; i < size; i++)
   {
	 max_ending_here = max_ending_here + a[i];
	 if(max_ending_here < 0)
		max_ending_here = 0;
	 if(max_so_far < max_ending_here)
		max_so_far = max_ending_here;
	}
	return max_so_far;
} 
int s,flag,i,a[1000005],t,n;
int main()
{
	scanf("%d",&t);
	while(t--)
	{
		flag=0;
		s=INT_MAX;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",a+i);
			s=max(a[i],s);
			if(a[i]>=0)
				flag=1;
		}
		if(!flag)
			printf("%d\n",s);
		else			
			printf("%d\n",maxSubArraySum(a,n));
	}
	return 0;
}
