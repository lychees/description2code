#include <stdio.h>
#include<vector>
#include <math.h>

using namespace std;

int remove_zeros(int number) {
    while(number%10==0) number/=10;
    return number;
}

int getreverse(int number) {
    vector<int> numbers;
    while(number) {
        numbers.push_back(number%10);
        number/=10;
    }
    
    int power = pow (10,numbers.size()-1);
    //for(int i=1;i<numbers.size();++i) power *= 10;
    
    int reverse = 0;
    for(int i=0;i<numbers.size();++i) {
        reverse += numbers[i]*power;
        power /= 10;
    }
    return reverse;
}

int main() {
    int num_cases;
    scanf(" %d", &num_cases);
    for(int i=0;i<num_cases;++i) {
        int num1, num2;
        scanf("%d %d", &num1, &num2);
        //printf("%d-%d\n", getreverse(num1), getreverse(num2));
        num1 = remove_zeros(getreverse(num1));
        num2 = remove_zeros(getreverse(num2));
        int sum = remove_zeros(num1+num2);
        printf("%d\n", getreverse(sum));
    }
}
