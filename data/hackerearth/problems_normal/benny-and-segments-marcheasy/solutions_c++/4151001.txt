#include<iostream>
#include<algorithm>
#include <cmath>
using namespace std;
typedef long long int ll;
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
        ll a,b,temp=0;;
        cin>>a>>b;
        pair<ll,ll> p[2001];
        for(int i=0;i<a;i++)
        {
            ll n,m;
            cin>>n>>m;
            p[i].first=n;
            p[i].second=m;
        }
        sort(p,p+a);
        for(ll j=0;j<a;j++)
        {
            ll max=p[j].first+b;
            ll right=p[j].second;
            for(ll i=0;i<a;i++)
            {
                if(p[i].first<=right&&p[i].second<=max&&p[i].first>p[j].first)
                {
                   if(p[i].second>right)
                   {
                       right=p[i].second;
                   }
                }
                
            }
            if(right==max)
            {
                temp=1;
            }
            
        }
        if(temp==1)
        {
            cout<<"Yes"<<endl;
        }
        else
        {
            cout<<"No"<<endl;
        }
    }
}
