#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int m,n;
    	cin>>m>>n;
    	int a[m],i,k,count,b[m],l=0,val=0;
    	for(i=0;i<m;i++)
    	cin>>a[i];
    	for(i=0;i<m;i++)
    	{
    		k=a[i];
    		count=0;
    		while(k>0)
    		{
    			count+=k&1;
    			k=k>>1;
    		}
    		b[l]=count;
    		l++;
    	}
    	sort(b,b+m);
    	for(i=m-1;n!=0;i--)
    	{
    		val+=b[i];
    		n--;
    	}
    	cout<<val<<endl;
    }
    return 0;
}
