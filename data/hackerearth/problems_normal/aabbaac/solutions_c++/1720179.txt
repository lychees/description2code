vector<ll> v;
char s[52][104],ans[1003],len[52];
ll temp;

int main(){
    int t,i,j,n,m,ind;
    //freopen("input.txt","r",stdin);
	scd(t);
	while(t--){
		scd(n);scd(m);
		rep(i,0,n-1){
			scs(s[i]);
			len[i]=strlen(s[i]);	
		}
		v.PB(len[0]);
		rep(i,1,n-1)
			v.PB((2*v[i-1])+len[i]);
		//parr(v,n);
		rep(i,0,m-1){
			scl(temp);
			temp+=1;
			ind=(LB(v.B,v.E,temp)-v.B);
			while(1){
				if(temp>(v[ind]-len[ind])){
					temp=temp-v[ind]+len[ind]-1;
					ans[i]=s[ind][temp];
					break;
				}
				else if( temp<=((v[ind]-len[ind])/2) ){
					ind--;
				}
				else{
					temp=v[ind]-len[ind]-temp+1;
					ind--;
				}
			}
		}
		v.clear();
		ans[m]='\0';
		printf("%s\n",ans);
	}
	return 0;
}
