#include<iostream>
using namespace std;
int main(){
	
	int i;
	cin>>i;
	while(i--){
		int a,b;
		int unitplace=0;
		cin>>a>>b;
		for(int j=a;j<=b;j++){
			if(j<=4 && j!=3){
				unitplace +=(j%5);
			}
			else if(j==3){
				unitplace +=6;
				
			}
			unitplace %=10;
		}
		cout<<unitplace<<endl;
	}
	return 0;
}
