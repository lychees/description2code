template <typename T, typename U> 
inline void chkmax(T& x, U y) {
	if (x < y) x = y;
}

template <typename T, typename U>
inline void chkmin(T& x, U y) {
	if (y < x) x = y;
}

const int MAXN = 5555;
int n, K;
int a[MAXN];

int main() {

	scanf("%d%d", &n, &K);
	for (int i = 0; i < n; i++) scanf("%d", a + i);
	sort(a, a + n);
	int q;
	scanf("%d", &q);
	while (q--) {
		int x;
		scanf("%d", &x);
		int ans = K + 1;
		for (int i = 0; i < n; i++) {
			if (a[i] && x % a[i] == 0) {
				chkmin(ans, x / a[i]);
			}
			for (int j = 1; j <= K; j++) {
				int t = x - j * a[i];
				if (t < 0) break;
				for (int k = 1; k <= K - j; k++) {
					if (t % k == 0 && binary_search(a, a + n, t / k)) {
						chkmin(ans, j + k);
						break;
					}
				}
			}
		}
		if (ans == K + 1) puts(":(");
		else printf("%d\n", ans);
	}

	return 0;
}
