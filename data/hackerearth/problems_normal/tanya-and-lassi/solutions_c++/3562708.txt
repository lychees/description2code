#include <iostream>
#include <algorithm>

using namespace std;

long long int dp[10005], prices[10005];

int main(){
	int t;
	cin >> t;
	while( t-- ) {
		long int litre;
		cin >> litre;
		
		for(long int i=0; i<litre; i++)
			cin >> prices[i];

		for(long int j = 1; j < litre+1; j++) {
			dp[j] = 0;
			long long int profit = 0;
			
			for(long int k = 1; k <= j; k++) {
				profit = max(profit, dp[j-k] + prices[k-1]);
			}
			
			dp[j] = profit;
		}
		
		cout << dp[litre] << endl;
	}
	return 0;
}
