int main()
{
	int n,x,s;
	cin >> n >> x;
	int a[n];
	vector < int > ans;
	for(int i=0;i<n;++i)cin >> a[i];
	for(int i=1 ; i < (1<<n) ; ++i)
    {
    	vector <int> t;
    	s = 0;
		for(int j=0;j<n;++j)
		{
			if(i & (1<<j))
			{
				s += a[j];
				t.pb(j+1);
			}
		}
		if(s == x)
		{
			/*for(int i=0;i<t.size();++i)

    		cout << endl;

			if(ans.empty())ans=t;

			else if((t.size()) == ans.size())
			{
				bool f =false; 
				for(int i=0;i<(int)ans.size();++i)
				{
					if(t[i] < ans[i])
					{
						f = true;
						break;
					}
				}
				if(f)
				ans=t;
			}
		}
    }
    if(ans.empty())
    {
    	cout << "-1";
    	return 0;
    }
    
    for(int i=0;i<ans.size();++i)
    {
    	cout << ans[i] << " ";
    }

}
