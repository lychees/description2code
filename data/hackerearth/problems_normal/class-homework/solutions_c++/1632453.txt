#include <bits/stdc++.h>
using namespace std;

#define inf 1000000000000LL

int len;
char S[25];

long long func(int x, int y, int z)
{
	if((S[0]=='0' && x>1) || (S[x]=='0' && y-x>1) || (S[y]=='0' && z-y>1) || (S[z]=='0' && len-z>1))
		return -1;
	
	long long a = 0, b = 0, c = 0, d = 0;
	
	for(int i=0; i<x; i++)
	{
		a = a*10 + (S[i]-'0');
		
		if(a>inf)
			return -1;
	}
	
	for(int i=x; i<y; i++)
	{
		b = b*10 + (S[i]-'0');
		
		if(b>inf)
			return -1;
	}
	
	for(int i=y; i<z; i++)
	{
		c = c*10 + (S[i]-'0');
		
		if(c>inf)
			return -1;
	}
	
	for(int i=z; i<len; i++)
	{
		d = d*10 + (S[i]-'0');
		
		if(d>inf)
			return -1;
	}
	
	return a+b+c+d;
}

int main()
{
    int N;
    scanf("%d", &N);
    
    while(N--)
    {
    	scanf(" %s", S);
    	
    	len = strlen(S);
    	
    	long long ans = -1;
    	
    	for(int i=1; i<len; i++)
    	{
    		for(int j=i+1; j<len; j++)
    		{
    			for(int k=j+1; k<len; k++)
    				ans = max(ans, func(i, j, k));
    		}
    	}
    	
    	if(ans!=-1)
	    	printf("%lld\n", ans);
	    else
	    	printf("unlucky\n");
    }
    
    return 0;
}
