#include <iostream>
#include<vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	int cnt1=0;
    	int cnt2=0;
    	vector<float > vec;
    	int n;
    	cin>>n;
    	vec.resize(n);
    	
    	for(int i=0;i<n;i++)
    	{
    		cin>>vec[i];
    	}
    	
    	int i=0;
    	int j=n-1;
    	while(i<=j)
    	{
    		if(i==j)
    		{
    			i++;
    			j--;
    			cnt1++;
    		}
    		else if(vec[i]==vec[j]*2)
    		{
    			i++;
    			j--;
    			cnt1++;
    			cnt2++;
    		}
    		else if(vec[i]<vec[j]*2)
    		{
    			vec[j]=vec[j]-vec[i]/2;
    			i++;
    			cnt1++;
    			if(i==j)
    			{	cnt2++;break;}	
    			
    		}
    		else if(vec[i]>vec[j]*2)
    		{
    			vec[i]=vec[i]-2*vec[j];
    			j--;
    			cnt2++;
    			if(i==j)
    			{
    				cnt1++;break;
    			}
    			
    		}
    	}
    	cout<<cnt1<<" "<<cnt2;
    	cout<<endl;
    	t--;
    }
}
