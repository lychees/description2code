#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    char ans[1005],in[1005];
    gets(in);
    while(t--)
    {
    	gets(in);
    	long long int  l,inc=0,out;
    	l=strlen(in);
    	//cout<<in<<endl;
    	for(int i=0;i<l;i++)
    	{
    		if(in[i]=='$')
    		{
    			i++;
    			while(((in[i]<='9' && in[i]>='0') || in[i]==' ') && i<l)
    			{
    				if(in[i]<='9' && in[i]>='0')
    				{
    					ans[inc]=in[i];
    					//cout<<ans[inc]<<"    "<<in[i]<<endl;
    					inc++;
    					if(inc==1 && ans[inc-1]=='0') inc--;
    					
    				}
    				i++;
    			}
    			ans[inc]='\0';
    		}
    	}
    	if(strlen(ans)==0)
    	{
    		ans[0]='0';
    		ans[1]='\0';
    	}
    	cout<<'$'<<ans<<endl;
    }
    return 0;
}
