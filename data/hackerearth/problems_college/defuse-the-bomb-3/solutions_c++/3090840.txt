#include<bits/stdc++.h>
using namespace std;
int convert(string str)
{
	int i;
	int len = str.length(),sum=0;
	for(i=0;i<len;i++)
	{
		sum = sum*10 + (str[i]-'0');
	}
	return sum;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str;
		cin>>str;
		long long int n;
		cin>>n;
		int p = n,c=0;
		while(p!=0)
		{
			c++;
			p = p/10;
		}
		for(int i=0;i<=str.length()-c;i++)
		{
			long long int num = convert(str.substr(i,c));
			cout<<num*n<<"\n";
		}
	}
}
