inline ll scan(){
    char c = getchar_unlocked();
    register ll x = 0;
    bool neg=false;
    if(c=='-'){
        neg=true;
    }
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    if(neg==false) return x;
    else if(neg==true) return -1*x;
}

ll beg[112345],fin[112345];

int main(){
	ll t=scan();
	while(t--){
		register  ll i,j,k=1,p,q,x,y,ans=1,n;
		n=scan();
		for(i=1;i<=n;i++){
			beg[i]=scan();fin[i]=scan();fin[i]--;
		}
		sort(beg+1,beg+1+n);
		sort(fin+1,fin+1+n);
		i=2;j=1;
		while(i<=n && j<=n){
			if(beg[i]<=fin[j]){
				k++;i++;
				ans=max(ans,k);
			}else{
				j++;k--;
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
