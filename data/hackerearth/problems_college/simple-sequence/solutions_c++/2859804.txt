#include <stdio.h>
#include<iostream>
using namespace std;
int DP[100000]={0},LISLen=0;
void fillDP(int i);
int main()
{
    int t,n,i,myint;
		LISLen=0;
		cin>>n;
    	for( i=0;i<n;i++)
    	{
			scanf("%d",&myint);
			fillDP(myint-i);
    	}
    	cout<<n-LISLen<<endl;
    }
 
 
void fillDP(int i)
{
 
	if(i>0)
	{
		int left=0,right=LISLen-1,y=-1;
		while(left<=right)
		{
			int mid=(left+right)>>1;
			if(DP[mid]<=i)
				left=mid+1;
			else {
				y=mid;
				right=mid-1;
			}
		}
		DP[(y >= 0)?(y):(LISLen++)] =i;
	}
}
