t=input()
for _ in range(0,t):
	n,k=map(int,raw_input().split())
	a=map(int,raw_input().split())
	a.sort()
	j=0
	for i in range(n):
		if(a[i]>0):
			break
		j+=1
	if(j<k):
		print "YES"
	else:
		print "NO"
