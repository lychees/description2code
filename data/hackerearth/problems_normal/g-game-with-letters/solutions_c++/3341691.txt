#include <bits/stdc++.h>
using namespace std;

int main () {
	int T;
	scanf("%d", &T);
	for (int t = 0; t < T; t++) {
		char s[100005];
		int c[26], n, m = -1, k = 0, l = 100;
		memset(c, 0, sizeof(c));
		scanf("%d %s", &n, s);
		for (int i = 0; s[i]; i++) {
			c[s[i]-'a']++;
		}
		for (int i = 0; i < n; i++) {
			char r[15];
			int t = 0, y = 0;
			scanf("%s", r);
			for (int j = 0; r[j]; j++) {
				t += c[r[j]-'a'];
				y++;
			}
			if (t > m || (t == m && y < l)) {
				m = t;
				k = i;
				l = y;
			}
		}
		printf("%d\n", k+1);
	}
	return 0;
}
