#include <iostream>
using namespace std;
 
typedef long long ll;
 
int main()
{
	ll X;
    int T;
    cin>>T;
    while(T--)
    {
    cin>>X;
    	ll ans=0;
    	for(ll d=1;d*d<=X;++d)
    	{
    		if(X%d==0)
    		{
    			if(d%4!=0)
    			 ans+=d;
    			 if(d*d!=X)
    			  if((X/d)%4!=0)
    			   ans+=X/d;
    		}
    	}
    	ans=8*ans;
    	cout<<ans<<endl;
    }
    return 0;
}
