#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    string s;
    cin>>s;
    for(int i=1;i<s.size();)
      if(s[i]!=s[i-1])
       {
       	s.erase(s.begin()+i);
       	s.erase(s.begin()+i-1);
       //	cout<<"i"<<i<<endl;
       if(i!=1)
       	--i;
       }
      else ++i;
     printf("%d\n",s.size());
    }
    return 0;
}
