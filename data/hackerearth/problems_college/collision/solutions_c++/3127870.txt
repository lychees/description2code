#include<bits/stdc++.h>
using namespace std;
main()
{
	int t,j;
	cin>>t;
	for(j=0;j<t;j++)
	{
		int N,i,s=0;
		int hash[100]={0};
		int hash1[100]={0};
		int a[101];
		cin>>N;
		for(i=0;i<N;i++)
		{
		cin>>a[i];
		hash[a[i]]=a[i]%10;
		}
		for(i=0;i<N;i++)
		{
			if(hash1[hash[a[i]]]==0)
			{
			hash1[hash[a[i]]]=1;
			s++;
			}
		}
		s=N-s;
	cout<<s<<endl;
	
	}

return 0;
}
