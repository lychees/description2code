import sys


def is_prime(number):
    for i in range(2, number-1):
        if (number == 2):
            return True
        if (number % i == 0):
            return False
    return True


def primes(stop_position):
    next_num = 2
    position = 1
    while stop_position >= position:
        if is_prime(next_num):
            yield (position, next_num)
            position += 1
        next_num = next_num + 1


series = [(a, b) for a, b in primes(int(raw_input()))]

series = series.pop()
