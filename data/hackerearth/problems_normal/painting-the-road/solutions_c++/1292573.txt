#include<bits/stdc++.h>
using namespace std;

struct node {
    int l,r,c;
};

const int inf=0x3f3f3f3f;
const int N=1010;

int dp[N];
node a[N];

bool comp(node a,node b) {
    return a.l<b.l;
}
int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int L,n;
        scanf("%d%d",&L,&n);
        for(int i=0;i<n;i++) {
            scanf("%d%d%d",&a[i].l,&a[i].r,&a[i].c);
        }
        sort(a,a+n,comp);
        for(int i=0;i<=L;i++) {
            dp[i]=inf;
        }
        dp[0]=0;
        for(int i=0;i<n;i++) {
            for(int j=a[i].l;j<=a[i].r;j++) {
                dp[a[i].r]=min(dp[a[i].r],dp[j]+a[i].c);
            }
        }
        if(dp[L]==inf) {
            dp[L]=-1;
        }
        printf("%d\012",dp[L]);
    }
    return 0;
}
