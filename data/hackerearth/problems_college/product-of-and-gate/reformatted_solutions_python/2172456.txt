def rangeBitwiseAnd(m, n):
    k = 0
    while True:
        if n > m:
            k += 1
        else:
            return m << k
        m = m >> 1
        n = n >> 1


tc = int(raw_input())
while tc > 0:
    tc = tc - 1
    a, b = map(int, raw_input().split())
    ans = rangeBitwiseAnd(a, b)
    print ans
