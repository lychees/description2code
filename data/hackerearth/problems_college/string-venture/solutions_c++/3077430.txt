#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	string master;
    string s;
    float w=0;
    int cnt=0;
    while(cin>>s){
    	master+=s;
    	master+=" ";
    	for(int i=0;i<s.length();i++){
    		w+=int(s[i]);
    	}
    	cnt++;
    }
    master.pop_back();
    w+=(cnt-1)*(int(' '));
    //cout<<master.length()<<endl;
    w=w/master.length();
    //cout<<w<<endl;
    int v = floor(w);
    //cout<<v<<endl;
    if(v%2!=0){
    	int a[26]={0};
    	char ch;
    	for(int i=0;i<master.length();i++){
    		if(master[i]!=' '){
	    		ch=master[i];
	    		if(isupper(master[i])) ch=tolower(s[i]);
	    		a[ch-'a']+=1;
    		}
    	}
    	int m=0,ans=0;
    	for(int i=0;i<26;i++){
    		if(a[i]>m){
    			m=a[i];
    			ans=i;
    		}
    	}
    	cout<<char(ans+'a')<<endl;
    }else{
    	reverse(master.begin(),master.end());
    	cout<<master;
    }
    return 0;
}
