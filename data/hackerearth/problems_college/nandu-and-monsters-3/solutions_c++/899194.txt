#include <iostream>
#include <vector>
#include <string>
using namespace std;

int main() {
	long long int n,q;
	cin >> n >> q;
	vector<long long int> all;
	string s;
	cin >> s;
	for(long long int i=0;i<n;++i){
		all.push_back(s[i]-48);
	}
	while(q--){
		long long int i,e;
		cin >> i >> e;
		bool found = false;
		while(i >= 1 && e >= 1){
			if(i > e && all[i-1] == 0 && all[e-1] == 0){
				i /= 2;
			}
			else if(i > e && (all[i-1] == 1 || all[e-1] == 1)){
				found = false;
				break;
			}
			else if(e > i && all[i-1] == 0 && all[e-1] == 0){
				e /= 2;
			}
			else if(e > i && (all[i-1] == 1 || all[e-1] == 1)){
				found = false;
				break;
			}
			else if(i == e && all[i-1] == 0 && all[e-1] == 0){
				found = true;
				break;
			}
			else if(e == i && (all[i-1] == 1 || all[e-1] == 1)){
				found = false;
				break;
			}
		}
		if(found == true)
			cout << "Yes" << endl;
		else
			cout << "No" << endl;
	}
	return 0;
}
