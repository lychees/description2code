#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int n,a,max1=-1;
    long long int min1=10000000000,ans=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
    	scanf("%d",&a);
      if(a>max1)
      max1=a;
      if(a<min1)
      min1=a;
     // printf("max= %d min= %d\n",max1,min1);
    }
      ans=max1*min1;
      printf("%lld\n",ans);


    return 0;
}
