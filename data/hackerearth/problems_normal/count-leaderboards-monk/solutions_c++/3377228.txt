#include <iostream>
#include<string.h>
#include <math.h>

using namespace std;

#define ll long long int
#define MOD 1000000007

ll ncr[1003][1003];
int n,k;
ll dp[1002];
ll fact[1002];

void f(int n){
	fact[0]=1;
	fact[1]=1;
	for(int i=2;i<=n;i++)
		fact[i]=(i%MOD*fact[i-1]%MOD)%MOD;
}

void _init(int n){
	memset(ncr,0,sizeof(ncr));
	for(int i=0;i<1001;i++){
		ncr[i][0]=1;
	}
	ncr[1][1]=1;
	for(int i=2;i<n+1;i++){
		for(int j=1;j<=i;j++){
			ncr[i][j]=(ncr[i-1][j]%MOD+ncr[i-1][j-1]%MOD)%MOD;
		}
	}
}

ll solve(int n){
	if(n==1||n==0)
		return 1;
	if(dp[n]!=-1)
		return dp[n];
	ll ans=0;
	for(int i=0;i<n;i++){
		ans+=(((ncr[n-1][i]%MOD*fact[i+1]%MOD)%MOD)*solve(n-i-1)%MOD)%MOD;
		ans%=MOD;
		//cout << ans <<endl;
	}
	dp[n]=ans;
	return dp[n];
}

int main()
{
    //cout << "Hello World!" << endl;
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    
    
    memset(dp,-1,sizeof(dp));
    cin >> n >> k;
    _init(n);
    f(n);
	ll ans=(solve(n)%MOD*ncr[n][k]%MOD)%MOD;
	cout << ans << endl;
    return 0;
}
