from collections import Counter
for _ in range(int(raw_input())):
    raw_input()
    a = Counter(map(int, raw_input().split())).items()
    b = sorted(a, key=lambda x: (-x[1], -x[0]))
    p = b[0][1]
    for k, v in b:
        if v == p:
            print k,
        else:
            break
