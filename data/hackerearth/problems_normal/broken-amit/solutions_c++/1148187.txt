#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;

#define max 20
pair<int,int> v[max];

bool cmp( const pair<int,int> &a,const pair<int,int> &b)
    { return   (a.second>b.second);
    }
     
  int main()
  {
  	int n,i;
  	cin>>n;
  	for(i=0;i<n;i++)
  	{
  		cin>>v[i].first;
  	}
  		for(i=0;i<n;i++)
  	{
  		cin>>v[i].second;
  		v[i].second=i*v[i].second;
  	}
     	long  sum=0;
     	sort(v,v+n,cmp);
     
     
     	/*		for(int i=0;i<n;i++)

     	cout<<endl;
     	for(int i=0;i<n;i++)
     	cout<<v[i].second<<" ";
     		cout<<sum<<endl;*/
     	
     	for(i=0;i<n;i++)
     	{
     		for(int j=0;j<n;j++)
     		{
     			sum+=v[j].first;
     		}
     			v[i].second=0;
     		for(int k=i;k<n;k++)
     		{
     			v[k].first-=v[k].second;
     		}
     	
     	}
     	
     	cout<<sum<<endl;


return 0;
}
