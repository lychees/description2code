#include <bits/stdc++.h>
using namespace std;

int main()
{
  //  cout << "Hello World!" << endl;
  int t;
  scanf("%d",&t);
  while(t--)
  {
  	int n,i,x,ans=0;
  	map<int,int> count;
  	scanf("%d",&n);
  	for(i=0;i<n;i++)
  	{
  		scanf("%d",&x);
  		count[x]++;
  		ans  = max(ans,count[x]);
  	}
  	printf("%d\n",ans);
  }
    return 0;
}
