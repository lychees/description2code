#include <iostream>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--){
		string s;
		int n;
		bool o=true;
		cin>>s;
		scanf("%d",&n);
		char arr[n];
		for(int i=0;i<n;i++) cin>>arr[i];
		int ans=0;
		for(int i=0;i<s.size();i++){
			for(int j=0;j<n;j++){
				if(s[i]==arr[j]){ o=false;break;}
			}
			if(o) ans+=s[i]-96;
			o=true;
		}
		printf("%d\n",ans);
	}
    return 0;
}
