#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
	int n;
	cin >> n;
	vector<pair<int,int> > p(n+1);
	int a,b;
	for(int i=0;i<n;i++){
		scanf("%d%d",&a,&b);
		p[i] = make_pair(a,b);
	}
	ll area = 0;
	p[n] = make_pair(p[0].first,p[0].second);
	for(int i=0;i<n;i++){
		area += (ll) p[i].first * p[i+1].second -(ll) p[i+1].first * p[i].second;
	}
	area /= 2;
	
	b=0;
	for(int i=0;i<n;i++){
		b+= __gcd(abs(p[i].first - p[i+1].first), abs(p[i].second - p[i+1].second));
	}
	//cout << "b " << b << endl;
	printf("%lld",abs(area) - b/2 + 1);
    //cout << "Hello World!" << endl;
    return 0;
}
