#define N 40005
#define DBG(x) cerr<<(#x)<<"="<<x<<endl;

template <class T> inline void Max(T &a,T b){if(a<b)a=b;}
template <class T> inline void Min(T &a,T b){if(a>b)a=b;}

double c[N][61];
int main(){
    int i,j,k,m,T;
    m=60;
    c[0][0]=1;
    vector<ll>v[61];
    for(i=1;i<N;i++){
        c[i][0]=1;
        int r=min(i,m);
        for(j=1;j<=r;j++){
            c[i][j]=c[i-1][j-1]+c[i-1][j];
            if(c[i][j]>1e15)break;
            if(j>=3&&j<=i/2){
                if(j+2==i||j+1==i)continue;
                v[j].pb((ll)c[i][j]);
                if(j*2<i)v[j].pb((ll)c[i][j]);
            }
        }
    }
    for(j=3;j<=m;j++)sort(all(v[j]));
    scanf("%d",&T);
    while(T--){
        ll n;cin>>n;
        if(n==1){
            puts("-1");continue;
        }
        int ans=1;
        if(n>2)ans++;
        ll x=sqrt(n*2.);
        while(x*x-x<n*2)x++;
        if(x*x-x==n*2){
            ans++;
            if(x>4)ans++;
        }
        //DBG(ans)
        if(n==3)ans--;
        for(i=3;i<=m;i++){
            j=lower_bound(all(v[i]),n)-v[i].begin();
            k=upper_bound(all(v[i]),n)-v[i].begin();
            if(k>j){ans+=k-j;}
        }
        printf("%lld\n",ans);
        /*
        ans=0;
        c[0][0]=0;
        for(i=1;i<=100;i++){
            c[i][0]=1;
            for(j=1;j<=i;j++){
                c[i][j]=c[i-1][j-1]+c[i-1][j];
                if(c[i][j]==n){DBG(i)DBG(j)}
            }

        }
        */
    }
    return 0;
}
