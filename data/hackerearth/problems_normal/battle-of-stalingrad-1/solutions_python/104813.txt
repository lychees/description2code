#!/usr/bin/python
import string
def area(a,b,c,d,e,f):
	res=((a*(d-f))+(c*(f-b))+(e*(b-d)))
	if(res<0):
		res*=-1
	return res
	
def find(val):
	a=area(val[0],val[1],val[2],val[3],val[4],val[5])
	b=area(val[2],val[3],val[4],val[5],val[6],val[7])
	c=area(val[0],val[1],val[4],val[5],val[6],val[7])
	d=area(val[0],val[1],val[2],val[3],val[6],val[7])
	
	if(a==b+c+d):
		return 1
	return 0
	
if __name__=="__main__":
	cases=input()
	
	for i in range(cases):
		val=map(int,raw_input().split())
		if(find(val)):
			print "INSIDE"
		else:
			print "OUTSIDE"
