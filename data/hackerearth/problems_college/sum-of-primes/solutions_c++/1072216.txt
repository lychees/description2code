#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

long long primes[1000001];
long long sums[1000001];

bool prime(long long a) {
	long long sq = sqrt(a);
	if(a%2 == 0) return false;
	for(long long int i = 3; i < sq; i += 2)
		if(a%i == 0) return false;
	return true;
}

int main(void) {
	long long lim = 1000;
	for(long long i = 2; i <= lim; i++) {
		if(primes[i] == 0) {
			long long j = i*i;
			while(j <= 1000000) {
				primes[j] = 1;
				j += i;
			}
		}
	}
	
	primes[0] = primes[1] = 1;
	
	for(long long i = 0; i <= 1000000; i++) {
		sums[i] = sums[i-1]+i*(!primes[i]);
	}
	
	long long t;
	scanf("%lld", &t);
	for(long long i = 0; i < t; i++) {
		long long a, b;
		scanf("%lld%lld", &a, &b);
		long long SUM = sums[b] - sums[a-1];
		if(SUM <= 1000000 && !primes[SUM]) {
			printf("YES\n");
		}
		else if(SUM > 1000000 && prime(SUM)) {
			printf("YES\n");
		}
		else
			printf("NO\n");
	}
	return 0;
}
