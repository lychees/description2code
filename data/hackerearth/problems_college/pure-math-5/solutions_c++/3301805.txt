#include <iostream>
#include<stdio.h>

using namespace std;
 
int main() {
    long long int n, m, k, count, i, j, t, size;
    scanf("%lld", &t);
    while(t--) {
        count = 0;
        scanf("%lld", &size);
        for(j = 2; j <= size; j++) {
            i = 1;
            if(((j * j) - 1) % i == 0 && ((i * i) - 1) % j == 0) {
                count++;
                n = j;
                m = ((j * j) - 1) / i;
                while(m <= size) {
                    count++;
                    k = m;
                    m = ((k * k) - 1) / n;
                    n = k;
                }
            }
        }
        
        printf("%lld\n", count);
    }
    
    return 0;
}
