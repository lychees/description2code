#include <iostream>
#include <map>
using namespace std;
int main ()
{
 map<string,int>arr;
 int i,j,n;
  int a=0,p;
  long long int sum;
 long int val[1001][1001];
 string str;
 cin>>n;
 for(i=0;i<n;i++)
 {
 		cin>>str;
 		arr[str]=i;
 } 
 for(i=0;i<n;i++)
 {
 	for(j=0;j<n;j++)
 	{
 	 cin>>val[i][j];	
	}
 }
 cin>>p;
 for(j=0;j<p;j++)
 {
 	cin>>str;
 	sum+=val[a][arr[str]];
 	a=arr[str];
 }
 cout<<sum;
  return 0;
}
