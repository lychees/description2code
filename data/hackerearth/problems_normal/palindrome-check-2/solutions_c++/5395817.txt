#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int l,i=0;
    char str[100];
    cin>>str;
    l=strlen(str);
    while(i<l/2)
    {
        if((str[i])==str[l-i-1])
            i++;
        else
            break;
    }
    if(i==l/2)
        cout<<"YES";
    else
        cout<<"NO";
    return 0;
}
