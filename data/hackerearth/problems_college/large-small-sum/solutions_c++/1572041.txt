#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,i,j;
    cin >> t;
    while(t--)
    {
    	cin >> n;
    	long long int a[n];
    	for(int i=0;i<n;i++)
    	   cin >> a[i];
    	sort(a,a+n);
    	
    	cin >> i >> j;
    	cout << a[n-i]+a[j-1]<< "\n";
    }
    return 0;
}
