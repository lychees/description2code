#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    while(cin >> n){
    	int x[n];
    	for(int i=0; i<n; i++){
    		cin >> x[i];
    	}
    	vector<int> t;
    	for(int i=n-1; i>=0; i--){
    		if(x[i] % 2 ==0)
    			t.push_back(x[i]);
    	}
    	if(t.size() > 0){
    		for(int j=0; j<t.size(); j++){
    			cout << t[j] << " ";
    		}
    		cout << endl;
    	}
    	else
    		cout << "None to display\n";
    	
    }
    return 0;
}
