#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int a[10004];
int main()
{
    int t,i,r,s,n;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d %d %d",&r,&s,&n);
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&a[i]);
    	}
    	sort(a,a+n);
    	for(i=0;i<n;i++)
    	{
    		r-=a[i];
    		s-=a[i];
    		if(r<0 && s<0)
    		{
    			printf("Tie\n");
    			break;
    		}
    		if(r<0)
    		{
    			printf("Sayan Won\n");
    			break;
    		}
    		if(s<0)
    		{
    			printf("Raghu Won\n");
    			break;
    		}
    	}
    	if(i==n)
    	printf("Tie\n");
    }
    return 0;
}
