#include <iostream>
#include <algorithm>

using namespace std;

typedef long long llong;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--) {
    	int l,i,j;
    	scanf("%d",&l);
    	llong a[l+1],dp[l+1];
    	for(i=1;i<=l;i++)
    		scanf("%lld",&a[i]);
    	for(i=1;i<=l;i++) {
    		dp[i]=a[i];
    		for(j=1;j<i;j++) {
    			dp[i]=max(dp[i],a[j]+dp[i-j]);
    		}
    	}
    	printf("%lld\n",dp[l]);
    }
    
    return 0;
}
