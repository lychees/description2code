'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
looper = (2**input())-1
dictaphone = {}
while looper > 0:
    detail = raw_input()
    # detail = input("")
    detail = detail.split()
    if detail[0] not in dictaphone:
        dictaphone[detail[0]] = 1
    else:
        dictaphone[detail[0]] += 1
    looper -= 1
maxx = max(dictaphone.values())
for k, value in dictaphone.items():
    if value == maxx:
        print k
