#include <iostream>
#include <stdio.h>
#include <vector>
#include <stdlib.h>
#include <math.h>

using namespace std;

int main()
{
	
	int n, i, ct, maxVal, v;
	
	scanf("%d", &n );
	
	vector<int> A( n );
	
	for( i = 0;i < n;i++ )
		scanf("%d", &A[ i ] );
	
	ct = 1;
	maxVal = 1;
	
	for( i = 1;i < n;i++ ) {
		
		v = (int)abs( A[ i ] - A[ i - 1 ] );
		
		if( v == 1 )
			ct++;
		else {
			maxVal = max( maxVal, ct );
			ct = 1;
		}
	}
	
	maxVal = max( maxVal, ct );
	
	printf("%d\n", maxVal );
	
	return 0;
}
