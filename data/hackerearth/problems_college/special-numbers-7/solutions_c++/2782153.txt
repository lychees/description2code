#include <cstdio>
#include<set>
using namespace std;
int minPrime[100001];
int lucky[1001];
int factorize(int n) {
    set<int> res;
    while (n != 1) {
        res.insert(minPrime[n]);
        if(res.size()==3){
        	break;
        }
        n /= minPrime[n];
    }
    return res.size();
}
int main()
{
for (int i = 2; i * i <= 5000; ++i) {
    if (minPrime[i] == 0) { //if i is prime
        for (int j = i * i; j <= 5000; j += i) {
            minPrime[j] = i;
        }
    }
}
for (int i = 2; i <= 5000; ++i) {
    if (minPrime[i] == 0) {
        minPrime[i] = i;
    }
}
int factCount=1;
for(int i=30;i<=3000;i++){
	if(factorize(i)>=3){
		lucky[factCount++]=i;
	}
}
//printf("%d\n",factCount);
    int t,n;
    scanf("%d",&t);
    while(t>0){
    	scanf("%d",&n);
    	printf("%d\n",lucky[n]);
    	t--;
    }
    return 0;
}
