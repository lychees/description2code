#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    for(int k=0;k<n;k++)
    {
        int m,p,a[10005],c=0,sum=0;
        cin>>m>>p;
        for(int i=0;i<m;i++){cin>>a[i];if(a[i]==p)c++;}
        if(c==0)cout<<"NO"<<" "<<0<<endl;
        else
        {
           for(int i=0;i<m;i++)
           {
               if(a[i]>p)sum++;
           }
           cout<<"YES"<<" "<<sum<<endl;
        }
    }
    return 0;
}
