from heapq import heapify, heappush, heappop

rows, fans = [int(i) for i in raw_input().split()]
rowSeats = [-int(i) for i in raw_input().split()]
maxRevenue = 0

heapify(rowSeats)
while (fans):
    maxSeats = heappop(rowSeats)
    maxRevenue += -maxSeats
    heappush(rowSeats, maxSeats + 1)
    fans -= 1

print(maxRevenue)
