TC = int(raw_input())

while TC >= 1:
    insects = map(int, raw_input().split())
    N = insects.pop(0)
    #print 'N is: %d' % N
    total_odd = 0
    #print 'insects array is: {0}'.format(insects)

    for index, value in enumerate(insects):
        #print 'i is: %d' % value
        if value % 2 == 0:
            continue
        else:
            total_odd += 1

    if total_odd % 2 == 0:
        print 'Yes'
    else:
        print 'No'

    TC = TC -1
