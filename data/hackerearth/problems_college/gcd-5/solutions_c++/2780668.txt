#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n;
    	cin>>n;
    	long long int arr[n];
    	long long int i;
    	int flag;
    	for(i=0;i<n;i++)
    	{
    		cin>>arr[i];
    	}
    	for(i=0;i<n-1;i++)
    	{
    		if(arr[i]<arr[i+1])
    		{
    			flag=0;
    		}
    		else
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==0)
    	{
    		cout<<"YES"<<endl;
    	}
    	else
    	{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
