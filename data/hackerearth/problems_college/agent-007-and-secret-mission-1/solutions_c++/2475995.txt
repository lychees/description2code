#include <iostream>
#include<cstring>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t,i,cs,cu,d;
   string s;
   cin>>t;
   while(t--){
   	cin>>s;
   	cin>>d;
   	int n=s.size();
   	cs=0;cu=0;
   	for(i=0;i<n;i++){
   		if((char)s[i]=='1')cs++;
   		
   	}
   	
   	cu=n-cs;
   //cout<<cs<<" "<<cu<<endl;
   	if(d%4==0||d%4==3){
   		cout<<cs<<endl;
   	}
   	else cout<<cu<<endl;
   }
    return 0;
}
