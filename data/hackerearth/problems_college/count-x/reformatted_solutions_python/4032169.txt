t = input()
o = []
for i in range(t):
    l = raw_input()
    num = ''
    j = 0
    while j < len(l):
        if l[j] == '-':
            j += 1
            while j < len(l) and l[j] == '-':
                num += '0'
                j += 1
        else:
            c = 0
            while j < len(l) and l[j] == 'X':
                c += 1
                j += 1
            num += str(c)

    o.append(int(num))

for e in o:
    print e
