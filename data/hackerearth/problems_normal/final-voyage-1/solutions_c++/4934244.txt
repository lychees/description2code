using namespace std;
int dp[1005][1005];
int w1,n,d[1005],v[1005];
int compute(int in ,int w)
{
    if(in>n)
        return 0;
    if(dp[in][w]!=-1)
        return dp[in][w];
    if(w>=d[in])
    dp[in][w] = max(v[in]+compute(in+1,w-d[in]),compute(in+1,w));
    else
        dp[in][w] = compute(in+1,w);
    return dp[in][w];
}
int main()
{
    int i,t;
    ios::sync_with_stdio(false);
    cin >> t;

    while(t--)
    {

        memset(dp,-1,sizeof(dp));
        cin >> n >> w1;
        for(i=1;i<=n;i++)
            cin >> d[i];
        for(i=1;i<=n;i++)
            cin >> v[i];
        cout << compute(1,w1) << endl;
    }
}
