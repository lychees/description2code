#include<bits/stdc++.h>
#define gc getchar
#define m 1000000007
using namespace std;
typedef long long ll;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
int main()
{
    ll n,i,j,x,c;
    string s;
    cin>>n;
    while(n--)
    {
        cin>>s;
        c=0;
        x=s.length();
        for(i=0;i<x;)
        {
            if(s[i]=='X')
            {
                for(j=i+1;j<x;j++)
                {
                    if(s[j]=='X')
                        c++;
                    else
                        break;
                }
                i=j;
            }
            else
            {
                for(j=i+1;j<x;j++)
                {
                    if(s[j]=='Y')
                        c++;
                    else
                        break;
                }
                i=j;
            }
        }
        cout<<c<<'\n';
    }
    return 0;
}
