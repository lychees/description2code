#include<bits/stdc++.h>
using namespace std;
#define ll long long

ll gcd(ll a,ll b)
{
	if(b == 0){
		return a;
	}
	gcd(b,a%b);
}

int main() {
	ll n,m;
	ll test;
	cin>>test;
	while(test--) {
	cin>>n>>m;
	if(gcd(n,m) == 1) {
		cout<<"Is a Co-Prime\n";
	} else {
		cout<<"Not a Co-Prime\n";
	}
	}
	return 0;
}
