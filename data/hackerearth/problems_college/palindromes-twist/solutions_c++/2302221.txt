#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		string str;
		cin>>str;
		int ac,bc;
		cin>>ac>>bc;
		int min_cost=min(ac,bc);
		int r=str.length()-1;
		long long cost=0;
		if(((r+1)%2)!=0){
			if(str[r/2]=='/'){
				cost+=min_cost;
			}
		}
		int l=0;
		bool done=false;
		while(l<r){
			if((str[l]!=str[r])&&(!((str[l]=='/')||str[r]=='/'))){
				cout<<"-1"<<endl;
				done=true;
				break;
			}
			else if((str[l]!=str[r])||(str[l]=='/'&&str[r]=='/')){
				if(str[l]=='a'||str[r]=='a'){
					cost+=ac;
				}
				else if(str[l]=='b'||str[r]=='b'){
					cost+=bc;
				}
				else{
					cost+=2*min_cost;
				}
			}
			l++;r--;
		}
		if(!done)
		cout<<cost<<endl;
	}
}
