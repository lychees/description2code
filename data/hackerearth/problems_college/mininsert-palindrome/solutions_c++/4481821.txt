#include <bits/stdc++.h>
using namespace std;
int main() {
	int t,n;
	cin >> t;
	while(t--) {
		cin >> n;
	string str;
	cin >> str;
	//int n = str.size();
	int dp[n][n];
	for(int i = 0; i < n; i++) {
		for(int j = 0; j < n; j++) {
			dp[i][j] = 0;
		}
	} 
	for(int i = 0; i < n; i++) dp[i][i] = 0;
	for(int i = 2; i <= n; i++) {
		for(int j = 0; j < n-i+1; j++){
		int k = j+i-1;
		if(i == 2){ 
		   dp[j][k] = (str[j] == str[k])?0:1;}
		else {
			if(str[j] == str[k]) dp[j][k] = dp[j+1][k-1];
			else dp[j][k] = min(dp[j+1][k],dp[j][k-1])+1;
		}
	    }
	}
	cout << dp[0][n-1] << endl;
   }
	return 0;
}
