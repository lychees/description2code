#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
int main() {
	ll t,n,A[100005],k;
	cin>>t;
	while(t--)
	{
		ll count=0;
		cin>>n>>k;
		for(ll i=1;i<=n;i++)
		{ cin>>A[i];}
		ll pos=0;
		for(ll i=1;i<=n;i++)
		{
			if(A[i] > k)
			pos=i;
			else 
			{
				count += i-pos;
			}
		}
		cout<<(n*(n+1))/2-count<<endl;
		
	}
	
	return 0;
}
