#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
 bool pr(int i)
{
if (i<2)
{
return false;
}
else if (i%2==0&&i!=2)
{
return false;
}
else {
for (int j =3;j<=sqrt(i);j=j+2)
{
if (i%j==0)
{
return false;
}
}
return true;
}
}
int main()
{
    long long int t,p,q,k,sum;
    cin>>t;
    while(t--)
    {
        cin>>p>>q;
        sum=0;
         while(p!=q)
         { if(p>q)
             {
                 if(pr(p)==true)
                   sum=sum-p;
                 else
                    sum=sum+p;
                 p=p/2;

             }
             else
             {
                 if(pr(q)==true)
                    sum=sum-q;
                 else
                    sum=sum+q;
                 q=q/2;
             }
         }
         if(pr(p)==true)
            sum=sum-p;
         else
            sum=sum+p;
         cout<<sum<<endl;
    }
    return 0;
}
