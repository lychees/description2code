#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

ll dp[1000000]={0},fdp[1000000]={0};
	
void divisors ()
{       // calculate number of divisors

    for(ll i = 2; i <= 1000000; i++)
    {
        for(ll j = i + i; j <= 1000000; j += i)
        {
            dp[j]++;
        }
    }
}	

int main()
{
	ll n,t,i,j;
	cin>>t;
	
	divisors();
	
	fdp[1]=0;
	
	for(i=2;i<=1000000;i++)
	{
		fdp[i]=fdp[i-1]+dp[i]+1;
	}
	
	while(t--)
	{
		cin>>n;
		
		cout<<fdp[n]<<endl;
	}
	
	return  0;
}
