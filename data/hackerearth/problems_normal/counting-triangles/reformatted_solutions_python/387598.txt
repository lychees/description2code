'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
n = int(raw_input())
dic = {}
ans = 0
while n:
    n -= 1
    li = raw_input().split()
    for i in range(0, 3):
        li[i] = int(li[i])
    li.sort()
    s = ""
    for i in li:
        s += str(i) + ","
    if dic.has_key(s):
        dic[s] = False
    else:
        dic[s] = True
for elem in dic.values():
    if elem:
        ans += 1
print ans
