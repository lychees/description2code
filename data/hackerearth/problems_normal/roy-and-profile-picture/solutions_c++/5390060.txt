#include <iostream>
using namespace std;

int main()
{
    int n,l,a,b;
    
    cin>>l>>n;
    
    while(n--)
    {
    	cin>>a>>b;
    	
    	if(a>=l && b>=l)
    	{
    		if(a==b)
    			cout<<"ACCEPTED"<<endl;
    		else
    			cout<<"CROP IT"<<endl;	
    	}
    	else
    	{
    		cout<<"UPLOAD ANOTHER"<<endl;
    	}
    	
    }
    return 0;
}
