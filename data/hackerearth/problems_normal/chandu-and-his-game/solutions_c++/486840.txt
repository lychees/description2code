#include <iostream>
#include<algorithm>

using namespace std;

long findmin(long a[])
{
 int i;
 long min=a[0];
 for(i=1;i<4;i++)
  {
   if(a[i]<min)
     min=a[i];

  }
  return min;
}


int main()
{
    long int n ,k,x[100000],y[100000],i,moves,corner[4];
    cin>>n>>k;
    for(i=0;i<k;i++)
      cin>>x[i];
      moves=0;
    for(i=0;i<k;i++)
    {
       cin>>y[i];
       corner[0]=abs(x[i]-1)+abs(y[i]-1);
       corner[1]=abs(x[i]-1)+abs(n-y[i]);
       corner[2]=abs(n-x[i])+abs(y[i]-1);
       corner[3]=abs(n-x[i]) + abs(n-y[i]);
       moves+=findmin(corner);
    }
    cout<<moves<<endl;
    return 0;
}
