def truncate(f, n):
    s = '%.12f' % f
    i, p, d = s.partition('.')
    return '.'.join([i, (d+'0'*n)[:n]])
t = int(raw_input())
speed = map(float , raw_input().split())
fuel = map(float , raw_input().split())
avail = float(raw_input())
for i in range(t):
	speed[i] = (speed[i])* (avail/fuel[i])
print truncate(max(speed) , 3)
