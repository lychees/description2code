#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll fib[31];
ll a[1000001];
int main()
{
    a[1000001]={0};
    fib[0]=0;fib[1]=1;
    for(ll i=2;i<31;i++)
    {
        fib[i]=fib[i-1]+fib[i-2];
        if(fib[i]<=1000000)
            a[fib[i]]=1;
    }
    a[2]=0;a[3]=0;
    for(ll i=4;i<1000001;i++)
    {
        if(a[i]==0)
            a[i]=a[i-1]+i;
        else
            a[i]=a[i-1];
    }
    ll t;
    scanf("%lld",&t);
    while(t--)
    {
        ll n;
        scanf("%lld",&n);
        ll sum=a[n-1];
        bool flag=true;
        ll k;k=sqrt(sum);
        for(ll i=2;i<=k;i++)
        {
            if(sum%i==0)
            {
                flag=false;
                break;
            }
        }
        if(flag==true)
            printf("Tic\n");
        else
        {
            if(sum%2==0)
                printf("Tac\n");
            else
                printf("Toe\n");
        }
    }
    return 0;
}
