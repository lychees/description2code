#include <iostream>
#include <vector>

using namespace std;

#define maxN 10001

vector <bool> sieve;

void ESieve()
{
	sieve.resize(maxN, true);
	sieve[0] = sieve[1] = false;

	for (int i = 2; i < maxN; i++)
	{
		for (int j = i*i; j < maxN; j += i)
		{
			if (sieve[i])
				sieve[j] = false;
		}
	}
}

int main()
{
	int N;

	ESieve();

	cin >> N;

	for (int i = 2; i <= N; i++)
		if (!sieve[i])
			cout << i << " ";

	return 0;
}
