#!/usr/bin/env python
__author__ = "rramchandani"

import sys
import math


def solve(n, x, arr):
    if n != len(arr):
        return -1
    count = 0
    for index in range(0, len(arr)-1):
        if arr[index] >= arr[index+1]:
            diff = arr[index] - arr[index+1]
            val = int(math.ceil(float(diff + 1) / x))
            arr[index+1] += val*x
            count += val
    return count


def main():
    testcases = int(sys.stdin.readline().replace("\n", ""))
    for testcase in range(0, testcases):
        n, x = [int(i)
                for i in sys.stdin.readline().replace("\n", "").split(" ")]
        arr = [int(i)
               for i in sys.stdin.readline().replace("\n", "").split(" ")]
        print(solve(n, x, arr))


if __name__ == "__main__":
    main()
