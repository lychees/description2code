#include <iostream>
#include <bits/stdc++.h>
using namespace std;
long long int n, i, a, b, k, x, y, c, aa[100000];
int main()
{
    cin>>n;
    for(i=0;i<n;i++)
    {
    	aa[i]=1;
    }
    cin>>k;
    for(i=0;i<k;i++)
    {
    	cin>>x>>y;
    	aa[y-1]=0;
    }
    c=0;
    for(i=0;i<n;i++)
    {
    	if(aa[i]==1)
    	{
    		c++;
    	}
    }
    cout<<c<<endl;
    return 0;
}
