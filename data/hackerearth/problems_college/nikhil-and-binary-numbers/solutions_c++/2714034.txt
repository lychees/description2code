#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t; cin >> t;
    while(t--)
    {
    	int n; cin >> n;
    	int a[n];
    	for(int i=0;i<n;i++) cin >> a[i];
    	int xor_prefix[n],xor_suffix[n];
    	xor_prefix[0] = a[0]; xor_suffix[n-1]=a[n-1];
    	for(int i=1;i<n;i++)
    		xor_prefix[i] = a[i] xor xor_prefix[i-1];
    	for(int i=n-2;i>=0;i--)
    		xor_suffix[i] = a[i] xor xor_suffix[i+1];
    	
    	int ans=0;
    	if(xor_suffix[1]==0)
    		ans++;
    	if(xor_prefix[n-2]==0)
    		ans++;
    	for(int i=1;i<n-1;i++){
    		int x = xor_prefix[i-1] xor xor_suffix[i+1];
    		if(x==0)
    			ans++;
    	}
    	cout << ans << endl;
    	
    }
    return 0;
}
