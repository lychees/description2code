#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
    ll t,n,m,a,b,c,a1,a2,a3,ans;
    cin>>t;
    while(t--)
    {
      cin>>n>>m>>a>>b>>c;
      a2=a3=LLONG_MIN;
      
          a1=n*a+m*b;
          if(n<=m)
            a2=(m-n)*b+2*n*c;
          if(m<=n)
            a3=(n-m)*a+2*m*c;
          ans=max(max(a2,a3),a1);
      cout<<ans<<endl;    
    }

}
