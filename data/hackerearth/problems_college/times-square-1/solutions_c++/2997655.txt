#include<iostream>
using namespace std;
int main()
{
    int n,m,a,c,d,cc,dd;
    cin>>n>>m>>a;
    if(m>=n)
        {
            d=m;
            dd=n;
        }
    else
        {
            d=n;
            dd=m;
        }
        if(a<=d)
        {
            if(d%a==0)
                c=d/a;
            else
                c=d/a+1;
        }
        else
        {
            c=1;
        }
        if(a<=dd)
        {
            if(dd%a==0)
                cc=dd/a;
            else
                cc=dd/a+1;
        }
        else
            cc=1;
        cout<<c*cc<<endl;
    return 0;
}
