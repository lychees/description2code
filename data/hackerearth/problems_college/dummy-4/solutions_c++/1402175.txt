#include<iostream>
#include<string.h>
#include<cstdio>
int re(int);
int main()
{
    char str[1001];
    int a,b,k=0;
    char c;
    scanf("%d",&a);
    getchar();
    while(a--)
    {
        gets(str);
        b=0;
        int len=strlen(str);
        for(int i=0;i<len;i++)
            b+=str[i];
        b-=48*len;
k+=b;
    }
    printf("%d\n",re(k));
}
int re(int a)
{
    if(a<10)
        return a;
    int p=0;
    while(a)
    {
        p+=(a%10);
        a/=10;
    }
    re(p);
}
