#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
  int ax;
  cin>>ax;
  vector<long> my;
  for(int i=0;i<ax;i++)
  {
  	long bx =0;
  	cin>>bx;
  	my.push_back(bx);
  }
  sort(my.begin(),my.end());
  long dx = -1,cu=0;
  
  for(int i=0;i<ax;i++)
  {
  	if(my[i]<=dx)continue;
  	else 
  	{
  		cu++;
  		dx = my[i]+4;
  	}
  }
  
  cout<<cu<<endl;
    return 0;
}
