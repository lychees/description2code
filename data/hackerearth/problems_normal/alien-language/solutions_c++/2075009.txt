int lps[30];

void _lps(string pattern){
	int n=pattern.length();
	int i=1;
	int len=0;
	lps[0]=0;
	while(i<n){
		if(pattern[i]==pattern[len]){
			len++;
			lps[i]=len;
			i++;
		}
		else if(len>0){
			len=lps[len-1];
		}
		else if(len==0){
			lps[i]=0;
			i++;
		}
	}
}
