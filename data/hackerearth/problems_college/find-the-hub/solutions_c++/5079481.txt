#include<iostream>
using namespace std;
int main(){
	int N;
	cin>>N;
	int arr[N],hash[1000000]={0};
	for(int i=0;i<N;++i){
		cin>>arr[i];
	}
	int i;
	for(int j=0;j<N;++j){
		for( i=2;i<=(arr[j]/2);i++){
			if(arr[j]%i==0)
			break;
			
		}
		if(i==(arr[j]/2)+1)
		hash[arr[j]]++;
	}
	int t=0;
	for(int j = 0;j < N;++j){
	
		if(hash[arr[j]]!=0){
			if(t<arr[j]){
				t = arr[j];
			}
		}
		
	}
	cout<<t<<endl;
	cout<<hash[t]<<endl;
	return 0;
}
