#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define mp make_pair
int main()
{
	int t;
	vector<pair<int,string> >v;
	cin>>t;
	while(t--)
	{
		v.clear();
		int n,k;
		cin>>n;
		while(n--)
		{
		
			string str;
			cin>>k>>str;
			
			v.pb(mp(k,str));
			sort(v.begin(),v.end());
		}
		cin>>k;
		cout<<v[k-1].second<<endl;
	}
	return 0;
}
