#include <iostream>
#include <iomanip>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,i,ind;
        cin>>n;
        double d[n],f[n],max=-2,tot=0,cur;
        for(i=0;i<n;i++)
        {
            cin>>d[i];
        }
        for(i=0;i<n;i++)
        {
            cin>>f[i];
        }

        for(i=0;i<n;i++)
        {
            tot += (d[i]*f[i]);
            cur = d[i] * f[i];
            if(cur>max)
            {
                ind = i;
                max = cur;
            }
        }
        cout<<ind<<" "<<fixed<<setprecision(3)<<(max/tot)<<endl;

    }
    return 0;
}
