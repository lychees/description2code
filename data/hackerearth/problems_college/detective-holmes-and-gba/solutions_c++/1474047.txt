#include <cstdio>
#include <iostream>
#include <algorithm>

using namespace std;

#define s(a) scanf("%d",&a)
#define p(a) printf("%d",a)
#define sp   printf(" ")
#define nl   printf("\n")

typedef long long ll;

int main()
{
	int T;
	s(T);
	while(T--) {
		int n,m,s;
		s(n);s(m);s(s);
		int temp=m%n;
		temp+=s-1;
		temp%=n;
		if(!temp)temp=n;
		p(temp);nl;
	}
	return 0;
}
