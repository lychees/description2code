#include<bits/stdc++.h>
#define ll long long int
#define MAX 10000009
#define MOD 1000000007
using namespace std;


ll phi_for_N(ll N)
{
    ll i,ans;
    ans=N;
    for(i=2; i*i<=N; i++)
    {
        if(N%i==0)
        {
            while(N%i==0)
            {
                N=N/i;
            }
            ans=ans-ans/i;///phi(N)=N(1- 1/p) where p is prime
        }
    }
    if(N>1)
        ans=ans-ans/N;
    return ans;
}

int main()
{
    ll t,T,n,b;
    scanf("%lld",&T);
    for(t=1; t<=T; t++)
    {
        scanf("%lld",&n);
        printf("%lld\n",phi_for_N(n));
    }
    return 0;
}
