#include<bits/stdc++.h>
using namespace std;
int N;
int a[10];
int recur(int p,int l,int r){
    if(l>r) return 0;
    if(p==N) return 1;
    int res=0;
    for(int i=p;i<N;i++){
        swap(a[i],a[p]);
        res+=recur(p+1,l+a[p],r)+recur(p+1,l,r+a[p]);
        swap(a[i],a[p]);
    }
    return res;
}
int main()
{
    scanf("%d",&N);
    int sum=0;
    for(int i=0;i<N;i++){
        scanf("%d",&a[i]);
        sum+=a[i];
    }
    int res=recur(0,0,0);
    printf("%d %d\n",res,sum);
    if(sum>=res){
        printf("Got no way out!\n");
    }
    else printf("We will win!\n");
}
