#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    long long x[n],y[n],f[n];
    for(int i=0;i<n;++i)
    	cin>>x[i]>>y[i]>>f[i];
    long double dp[n];
    dp[0]=(long double)f[0];
    for(int i=1;i<n;++i){
    	dp[i]=(long double)(-1ll<<60);
    	for(int j=0;j<i;++j){
    		dp[i]=(long double)max(dp[i],dp[j]-sqrt((long double) (((y[i]-y[j])*(y[i]-y[j]))+((x[i]-x[j])*(x[i]-x[j]))))+f[i]);
    	}
    }
    cout<<fixed;
	cout.precision(6);
    cout<<dp[n-1]<<endl;
    return 0;
}
