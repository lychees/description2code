#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	unsigned long long int a, b, steps=0;
    	scanf("%lld %lld", &a, &b);
    	
    	if(a==1 && b==1) {
    		cout<<steps<<endl;
    		continue;
    	}
    	while(a!=1 && b!=1)
    	{
    		if(a<b)
    			swap(a,b);	
    		steps += a/b;
    		a %= b;	
    	}
    	if(a==1 && b!=1)
    		steps += b-1;
    	if(b==1 && a!=1)
    		steps += a-1;
    	
    	cout<<steps<<endl;
    }
    
    return 0;
}
