#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007

int main()
{
    int t;
    long long int n;
    long long int ans, temp, frac, cnt, a, b;
    cin>>t;
    while(t--)
    {
    	scanf("%lld",&n);
    	ans=1;
    	while(n>1)
    	{
    		cnt=1;
    		a=2;
    		while(1){
    			temp=(a*a)%MOD;
    			ans=((ans)*(temp))%MOD;
    			cnt*=2;
    			a=(a*a)%MOD;
    			n-=cnt;
    			if(n<2*cnt) break;
    		}
    	}
    	if(n==1) ans=(ans*2)%MOD;
    	ans=(ans-1)%MOD;
    	if(ans<0) ans=ans+MOD;
    	printf("%lld\n",ans);
    }
    return 0;
}
