alls = []
while True:
    failed = False
    try:
        s = raw_input()
    except:
        failed = True
        break
    if len(s.rstrip().lstrip())==0:
        failed = True
        break   
    alls.append(s) 
    
if len(alls)> 0:
    winner = ''
    maxs   = 0
    for y in alls:
        csum = sum(int(x) for x in y)
        if csum > maxs:
            maxs = csum
            winner= y
    print winner
    exit()
exit()
