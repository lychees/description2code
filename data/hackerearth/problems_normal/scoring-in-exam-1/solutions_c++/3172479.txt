#include<bits/stdc++.h>
using namespace std;
# define ll long long
pair<ll,ll>  ar[111111];
ll pre[111111];
int main()
{
    int n,q,i;
    ll x;
    cin>>n>>q;
    for(i=0;i<n;i++)
    {
        cin>>x;
        ar[i].second=x;
    }
    for(i=0;i<n;i++)
    {
        cin>>x;
        ar[i].first=x;
    }
    sort(ar,ar+n);
    pre[0]=0;
    for(i=0;i<n;i++)
        pre[i+1]=pre[i]+ar[i].second;
    while(q--)
    {
        cin>>x;
        x=n-x+1;
        cout<<pre[n]-pre[x-1]<<'\n';
    }
}
