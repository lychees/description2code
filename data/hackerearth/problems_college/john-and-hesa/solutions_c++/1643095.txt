#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    unsigned long int t,x,p,count;
    cin >> t;
    while(t--)
        {
         count = 0;
           cin >> x >> p;
        
        while(x%p == 0)
            {
              count ++;
              x = x / p;
            }
        if(x == 1)
            cout << count << " " << 0 << endl;
        else
            cout << count << " " << x << endl; 
        
    }
    return 0;
}
