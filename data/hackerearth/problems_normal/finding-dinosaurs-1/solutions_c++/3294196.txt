#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n, k, l, q;
	string s;
	
	scanf("%d %d %d %d", &n, &k, &l, &q);
	

	map<vector<int>, string> m;
	for (int i=0; i<n; i++) {
		cin >> s;
		vector<int> v(k);
		for (int j=0; j<k; j++) {
			scanf("%d", &v[j]);
		}
		m[v] = s;
	}

	for (int i=0; i<q; i++) {
		vector<int> v(k);
		for (int j=0; j<k; j++) {
			scanf("%d", &v[j]);
		}
		if (m.count(v)==0) {
			printf("You cant fool me :P\n");
		} else {
			cout << m[v] << endl;
		}
	}

	return 0;
}
