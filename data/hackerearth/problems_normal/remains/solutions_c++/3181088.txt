#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long int t,i,c,s,x,y,a,b,temp,n;
    cin>>t;
    while(t--)
    {
    	s=0;
    	cin>>x>>y>>n;
    	s=x+y;
    	a=x;
    	b=y;
    	for(i=2;i<n;i++)
    	{
    		temp=abs(a-b);
    		if(temp==0)
    		{
    			i++;
    			break;
    		}
    		s=s+temp;
    		a=b;
    		b=temp;
    	}
 
    	cout<<s+((n-i)/3)*max(a,b)*2+((n-i)%3)*max(a,b)<<endl;
    }
    return 0;
}
