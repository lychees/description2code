void process(){
	cin >> n >> k;
	FOR(i, 1, n) FOR(j, 1, k) cin >> a[i][j];

	int x, y;
	FOR(i, 1, k) a[0][i] = 100;

		memset(dd, false, sizeof(dd));
		cin >> x;
		int res = -1, cnt = n;
		while (x--){
			cin >> y;
			if (cnt < 2) continue;
			int m = 0;

			FOR(i, 1, n){
				if (!dd[i] && a[i][y] < a[m][y]) m = i;
			}
			cnt = 0;
			FORB(i, n, 1){
				if (!dd[i] && a[i][y] == a[m][y]){
					cnt++;
					res = i;
				} else dd[i] = true;
			}
		}
		cout << res << endl;
	}
}
