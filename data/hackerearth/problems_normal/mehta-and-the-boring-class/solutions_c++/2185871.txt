#include <bits/stdc++.h>

using namespace std;

int a[1000001];

int prime_factor(int n)
{
	int i,j;
	int a[100000];
	int k = 0;
	int m = n;
	while (n % 2 == 0) {
		a[k++] = 2;
		n = n/2;
	}
	for (i = 3; i <= sqrt(n); i = i + 2) {
		while (n % i == 0) {
			a[k++] = i;
			n = n/i;
		}
	}
	if (n > 2) {
		a[k++] = n;
	}
	/*for (i = 0; i < k; i++) {
		printf("%d\n",a[i]);
	}*/
	
	return k;
	
}

int main()
{
	int t,i,j;
	scanf("%d",&t);
	a[1] = 0, a[2] = 1, a[3] = 2;
	for (i = 4; i <= 1000000; i++) {
		a[i] = a[i-1] + prime_factor(i);
	}
	int A, B;

		scanf("%d %d",&A,&B);

}
