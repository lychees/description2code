#include <iostream>
#include <bitset>
using namespace std;

int main()
{
    int m, n;
    cin >> m >> n;
    bitset<1000> data[m];
    int bo;
    for (int i = 0; i < m; i++) {
    	for (int j = n-1; j >= 0; j--) {
    		cin >> bo;
    		if (bo)
    			data[i].set(j, true);
    		else
    			data[i].set(j, false);
    	}
    }
    int max = 0;
    int min = 2000;
    for (int i = 0; i < m; i++) {
    	bitset<1000> temp = ~data[i];
    	int satis = 0;
    	int flip = temp.count() - 1000 + n;
    	for (int j = 0; j < m; j++) {
    		bitset<1000> temp1 = temp^data[j];
    		if (temp1.count() == 1000)
    			satis++;
    	}
    	if (satis > max) {
    		max = satis;
    		min = flip;
    	} else if (satis == max) {
    		if (flip < min) {
    			min = flip;
    		}
    	}
    }
    cout << min << " " << max << endl;
    return 0;
}
