noOfTestCases = int(raw_input())


def isPossible(string):
    countOfU = 0
    countOfS = 0

    for s in string:
        if (s == 'S'):
            countOfS += 1
        else:
            countOfU += 1

    if (countOfS > countOfU + 1 or countOfU > countOfS + 1):
        return -1

    firstStringToCompare = ""
    secondStringToCompare = ""

    for i in range(1, len(string)+1):
        if (i % 2 != 0):
            firstStringToCompare += "S"
            secondStringToCompare += "U"
        else:
            firstStringToCompare += "U"
            secondStringToCompare += "S"

    StoUDiff1 = 0
    UtoSDiff1 = 0

    StoUDiff2 = 0
    UtoSDiff2 = 0

    start = 0

    while (start < len(string)):
        if (string[start] == 'S' and firstStringToCompare[start] == 'U'):
            StoUDiff1 += 1
        elif (string[start] == 'U' and firstStringToCompare[start] == 'S'):
            UtoSDiff1 += 1

        if (string[start] == 'S' and secondStringToCompare[start] == 'U'):
            StoUDiff2 += 1
        elif (string[start] == 'U' and secondStringToCompare[start] == 'S'):
            UtoSDiff2 += 1
        start += 1

    if (StoUDiff1 != UtoSDiff1 and StoUDiff2 != UtoSDiff2):
        return -1

    if (StoUDiff1 == UtoSDiff1 and StoUDiff2 != UtoSDiff2):
        return StoUDiff1

    if (StoUDiff2 == UtoSDiff2 and StoUDiff1 != UtoSDiff1):
        return StoUDiff2

    if (StoUDiff2 == UtoSDiff2 and StoUDiff1 == UtoSDiff1):
        return min(StoUDiff2, StoUDiff1)


for caseNo in range(1, noOfTestCases + 1):
    print isPossible(raw_input())
