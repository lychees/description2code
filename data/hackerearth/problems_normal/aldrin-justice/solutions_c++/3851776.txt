#include <iostream>
using namespace std;

int main()
{
    int tc;
    int bt1,bt2,mt1,mt2;
    int left,right;
    cin>>tc;
    while(tc--)
    {
    	cin>>bt1>>bt2>>mt1>>mt2;
    	left=min(max(bt1,bt2),max(mt1,mt2));
    	right=max(min(bt1,bt2),min(mt1,mt2));
    	if(left>right)
    		cout<<"Nothing\n";
    	else if(left==right)
    		cout<<"Point\n";
    	else
    		cout<<"Line\n";
    	
    }
    
    return 0;
}
