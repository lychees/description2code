#include <bits/stdc++.h>
using namespace std;
#define INF 999999999999LL
string s,s1,s2;
long long n;
void floydWarshell (long long graph[][105])
{
    long long i, j, k;

    for (k = 0; k < n; k++)
    {
        for (i = 0; i < n; i++)
        {
            for (j = 0; j < n; j++)
            {
                if (graph[i][k] + graph[k][j] < graph[i][j])
                    graph[i][j] = graph[i][k] + graph[k][j];
            }
        }
    }
}
int main()
{
    long long i,j,k,m;
    cin >> n >> m;
    map<string,long long> mp;
    for(i=0;i<n;i++)
    {
    	cin >> s;
    	mp[s]=i;
    }
    long long graph[105][105];
    for(i=0;i<n;i++)
	for(j=0;j<n;j++)
		graph[i][j]=INF;
    for(i=0;i<n;i++)
    graph[i][i]=0;
    for(i=0;i<m;i++)
    {
	cin >> s1 >> s2 >> j;
	graph[mp[s1]][mp[s2]]=j;
	graph[mp[s2]][mp[s1]]=j;
    }  
    floydWarshell(graph);
    cin >> m;
    while(m--)
    {
    	cin >> s1 >> s2;
    	i=mp[s1];
    	j=mp[s2];
    	cout << graph[i][j] << "\n";
    }
    return 0;
}
