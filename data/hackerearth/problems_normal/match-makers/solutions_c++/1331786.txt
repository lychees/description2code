#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
#include<string>
#include<cstring>
using namespace std;


#define REP(i,n) for(int i=0, i##_len=(n); i<i##_len; ++i)

int T, N, A[10111], B[10111];
int main() {
    scanf("%d", &T);
    REP ($, T) {
	scanf("%d", &N);
	REP (i, N) scanf("%d", A+i);
	REP (i, N) scanf("%d", B+i);
	sort(A, A+N);
	sort(B, B+N);
	reverse(B, B+N);
	int cnt = 0;
	REP (i, N) if (A[i] % B[i] == 0 || B[i] % A[i] == 0) cnt++;
	printf("%d\n", cnt);
    }
    
    return 0;
}
