#include <iostream>
#include<bits/stdc++.h>
using namespace std;
vector<pair<int,int> >a;
    bool cmp(const pair<int,int>&f,const pair<int,int>&g )
    {
    	return f.second<g.second;
    	
    	
    	
    }

int main()
{
    long long int i,j,k,m,n,sum,x,y,t;
    
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	cin>>m;
    	sum=m;
    	bool flag=1;
    	a.clear();
    	for(i=0;i<n;i++)
    	{
    		cin>>x;
    		cin>>y;
    		a.push_back(make_pair(x,y));
    	}
    	sort(a.begin(),a.end(),&cmp);
    for(i=0;i<n;i++)
    {
    		if(sum>=a[i].second)
    		{
    			sum=sum+a[i].first;
    			continue;
    		}
    		else
    		{
    			flag=0;
    			break;
    		}
    		
    }
    	if(flag)
    	{
    		cout<<"YES"<<endl;
    	}
    	else
    	{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
