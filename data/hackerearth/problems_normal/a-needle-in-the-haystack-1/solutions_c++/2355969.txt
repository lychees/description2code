#include <iostream>
#include<string.h>
#include<stdio.h>
using namespace std;

int main()
{
	int t,l,i,k,j,flag=1;
	char s[10000],r[100000],c[10000];
	cin>>t;
	while(t>0)
	{
		t--;
		l=0;
		scanf("%s",s);
		scanf("%s",r);
		for(i=strlen(s)-1;i>=0;i--)
		{
			c[l++]=s[i];
		}	
		
		c[l]='\0';
		flag=1;
		for(i=0;i<strlen(r);i++)
		{
			
			for(k=0,j=i;k<strlen(s);j++,k++)
			{
				
				if(s[k]!=r[j])
				{
					flag=0;
					break;
				}
				else
				flag=1;
			}
			if(flag==1)
			{
				cout<<"YES\n";
				break;
			}
			else
			{
				for(k=0,j=i;k<strlen(s);j++,k++)
				{
				
					if(c[k]!=r[j])
					{
						flag=0;
						break;
					}
					else
					flag=1;
				}
				if(flag==1)
				{
					cout<<"YES\n";
					break;
				}
			}
		}
		if(i==strlen(r))
		cout<<"NO"<<endl;
	}
	return 0;
}
