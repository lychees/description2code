#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
	long int n,price;
	cin>>n;
	vector<long> v;
	for(int i=0;i<n;i++)
	{
		cin>>price;
		v.push_back(price);
	}
	sort(v.begin(),v.end());
	long int x=v[0];
	long int y=v[0]+4,ans=1;
	for(int i=0;i<v.size();i++)
	{
		if(v[i]>=x && v[i]<=y)
		continue;
		else
		{
			ans++;
			x=v[i];
			y=v[i]+4;
		}
	}
	cout<<ans;
}
