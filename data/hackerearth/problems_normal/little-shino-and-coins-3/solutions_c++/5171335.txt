#include<iostream>
#include<vector>
using namespace std;

int main()
{
    int n,k,count,b=0;
    cin>>n;
    string s;
    cin>>s;
    k = s.length();
   	vector<int> vec(26);
    for(int i=0;i<k;i++){
        for(int l=0;l<26;l++) vec[l]=0;
    	count=0;
      	for(int j=i;j<k;j++){
    		if(vec[int(s[j])-97]==0){
    		count++;
    		vec[int(s[j])-97]=1;
    		}
    		if(count==n)
    		  b++;
    		}
    	}
    cout<<b<<endl;
    return 0;
}
