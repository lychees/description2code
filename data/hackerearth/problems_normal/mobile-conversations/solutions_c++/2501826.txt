#include<bits/stdc++.h>
#define ll long long

using namespace std;
#define pll pair<ll ,ll>


int main()
{
    vector<pll> v1, v2;


    ll t ,dur,recharge;
    cin>>t;
    while(t--)
    {
        cin>>dur>>recharge;
        if(recharge>dur)
        v1.push_back(make_pair(dur,recharge));
        else
         v2.push_back(make_pair(recharge,dur));

    }
    sort(v1.begin(),v1.end());
    sort(v2.begin(),v2.end(),greater<pll>() );
    vector<pll>::iterator it;
    ll bal=0;
    ll min_rec=0;
    ll duration,rech;
    for(it=v1.begin();it!=v1.end();it++)
    {

        pll p=*it;
        duration=p.first;
        rech=p.second;


        if(bal==0)
        {
            min_rec+=duration;
            bal=rech;


        }
        else
        {
            if(bal>=duration)
            {
                bal=bal-duration+rech;
            }
            else
            {
                min_rec +=(duration-bal);
                bal=rech;
            }

        }

    }
     for(it=v2.begin();it!=v2.end();it++)
    {
        pll p=*it;
        duration=p.second;
        rech=p.first;


        if(bal==0)
        {
            min_rec+=duration;
            bal=rech;


        }
        else
        {
            if(bal>=duration)
            {
                bal=bal-duration+rech;
            }
            else
            {
                min_rec +=(duration-bal);
                bal=rech;
            }

        }

    }
    cout<<min_rec;
}
