#include <bits/stdc++.h>
using namespace std;
int pal(string a)
{
	int i=0,j=a.length()-1,f=0;
	while(i<j)
	{
		if(a[i]!=a[j])
		{
			f=1;break;
		}
		else
		{
			i++;j--;
		}
	}
if(f==0){return 0;}
else{return 1;}
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a,b;
    	cin>>a;
    	if(pal(a)==0)
    	{
    		cout<<"-1"<<endl;
    	}
    	else
    	{
    	for(int i=0;i<a.length();i++)
    	{
    		b=a;
    		b.erase(i,1);
    		if(pal(b)==0)
    		{
    			cout<<i<<endl;
    			continue;
    		}
    	}
    	}
    }
}
