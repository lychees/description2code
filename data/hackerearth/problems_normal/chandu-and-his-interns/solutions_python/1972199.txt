def factors(n):
	a = []
	if n > 4 and n % 2 == 0:
		return False
	for i in range(1, int(n**0.5) + 1):
		if n % i == 0:
			a.append(i)
			a.append(n//i)
			b = set(a)
			if len(b) >= 4:
				return False
	return b
