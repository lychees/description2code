#include <bits/stdc++.h>
using namespace std;
#define ll long long int 
ll al[50];
ll gcd(ll a,ll b)
{
	ll t;
	while(a%b!=0)
	{
		t=b;
		b=a%b;
		a=t;
	}
	return b;
}
	ll find(ll a,ll b,ll c)
	{
		int val=gcd(a,b);
		//cout<<val<<endl;
		ll i,j=0;
		for(i=1;i*i<val;i++)
		{
			if(val%i==0)
			{al[j++]=i;
			al[j++]=val/i;
			}
		}
		if(i*i==val)
		al[j++]=i;
		//cout<<j<<endl;
		sort(al,al+j);
		if(c>j)
		return 0;
		else
		return al[j-c];
		
	}
int main()
{
   ll t,a,b,c;
   scanf("%lld",&t);
   while(t--)
   {
   	scanf("%lld%lld%lld",&a,&b,&c);
   	a=find(a,b,c);
   	if(a==0)
   	printf("No crymeth today\n");
   	else
   	printf("%lld\n",a);
   }
    return 0;
}
