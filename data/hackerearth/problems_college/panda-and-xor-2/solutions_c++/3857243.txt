#include <iostream>
using namespace std;

int main()
{
   long int a,b,c,n,i;
 //  scanf("%ld",&n);
 cin>>n;
   while(n)
   {
   	i=0;
   	cin>>a>>b;
   	 //scanf("%ld%ld",&a,&b);
   	 c=a^b;
   	 while(c)
   	 {
   	 	c=c&(c-1);
   	 	i++;
   	 }
   	 if(i%2)
   	 cout<<"NO"<<endl;
   	// printf("NO\n");
   	 else
   	 cout<<"YES"<<endl;
   //	printf("YES\n");
   	 n--;
   }
    return 0;
}
