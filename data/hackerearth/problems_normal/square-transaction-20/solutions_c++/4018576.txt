#include <iostream>
#include <vector>
#include <algorithm>
#include <string.h>

#define ll long long
#define iloop(i,j,n) for(ll i=j;i<n;i++)
#define dloop(i,j,n) for(ll i=j;i>=n;i--)

using namespace std;

int main()
{
	ll n;
	scanf("%lld",&n);
	ll a[n];
	iloop(i,0,n)
	{
		scanf("%lld",&a[i]);
	}
	ll sum[n];
	memset(sum,0,sizeof(n));
	sum[0]=a[0];
	iloop(i,1,n)
	{
		sum[i] = sum[i-1] + a[i];
	}
	ll q;
	scanf("%lld",&q);
	while(q--)
	{
		ll num;
		scanf("%lld",&num);
		if(sum[n-1] < num)
		{
			cout << "-1\n";
			continue;
		}
		ll start=0,final=n-1,flag=0,mid,mid1;
		while(start < final-1)
		{
			// cout << start << ' ' << final << endl;
			mid=(start + final)/2;
			mid1=mid-1;
			if(sum[mid] >= num && sum[mid1] < num)
			{
				flag=1;
				cout << mid+1 << endl;
				break;
			}
			else if(sum[mid] < num)
			{
				start = mid;
			}
			else
			{
				final = mid;
			}
		}
		if(flag==1)
			continue;
		if(sum[start] >= num)
		{
			cout << start+1 << endl;	
		}
		else
		{
			cout << final+1 << endl;
		}
	}
    return 0;
}
