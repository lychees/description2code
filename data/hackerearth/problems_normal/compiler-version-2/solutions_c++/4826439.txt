#include <iostream>
#include <cstdio>
#include <cstdlib>

using namespace std;

char code[70];
int main()
{
    while(fgets(code, 65, stdin)!=NULL){
    	int i=0;
    	bool inComment= false;
    	while(code[i]!='\0')
    	{
   			if(code[i]=='/' && code[i+1]=='/')
			{
    			inComment  = true;					
			}
    		if(!inComment && code[i]=='-' && code[i+1]=='>'){
    			printf(".");
    			i+=2;
    		}
    		else{
    			printf("%c", code[i]);
    			i++;
    		}
    	}
    	//printf("\n");
    }
    return 0;
}
