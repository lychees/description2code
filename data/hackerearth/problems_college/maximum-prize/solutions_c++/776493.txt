#include<bits/stdc++.h>
using namespace std;
#define S(n) scanf("%d",&n)

int main()
{
    int t;
    S(t);

    while(t--)
    {
        int n,k;
        S(n);
        S(k);
        vector<int> v(n);

        for(int i = 0 ; i<n ; i++ )
        {
            S(v[i]);
        }

        sort(v.rbegin(),v.rend());
        long long sum = 0;
        for(int i = 0 ; i<k && i<n ; i++)
        {
            sum+=v[i];
        }
        cout<<sum<<endl;

    }




    return 0;
}
