import sys


def solve():
    K = int(sys.stdin.readline())
    S = sys.stdin.readline().strip()

    count = 0
    for start_ind in range(len(S)):
        d = {}
        for char_ind in range(start_ind, len(S)):
            char = S[char_ind]
            if char in d:
                d[char] += 1
            else:
                d[char] = 1
            if len(d) == K:
                count += 1
            elif len(d) > K:
                break
    print count


solve()
