#include <bits/stdc++.h>
using namespace std;
int main() {
	int T;
	cin >> T;
	
	for(int i=0; i<T; i++){
		int L, P, C;
		cin >> L >> P >> C;
		
		int result = 0;
		int temp = L;
		
		while(temp*C < P) {
			result++;
			temp *=C;
		}
		
		result++;
		int count=0;
		while((1<<count) < result)
			count++;
			
			cout<< "Case #" << i+1 << ": " << count << endl;
	}
	return 0;
}
