    #include<bits/stdc++.h>
    using namespace std;
    string n,x;
    int c,i;
     
     
    int main()
    {
        cin>>n;
        cin>>x;
        c=1;
        i=0;
        while(n[i]!=NULL)
        {
            if(n[i]!=x[i])
            c++;
            
            i++;
        }
        
        if(c==1 || c==2)
        cout<<"wins"<<endl;
        else
        cout<<"loses"<<endl;
            
        return 0;
    }
