#include<bits/stdc++.h>
using namespace std;
main()
{
    int T,N,M,t,n;
    long long a;
    cin>>T;
    while(T--)
    {
        cin>>N>>M;
        if(N==1)
            cout<<0<<endl;
        else if(N>M)
        {
            t=ceil( (log10(M)/log10(2))+1.0 );
            a=1<<(t-1);
            n=ceil( ((N-a)/(1.0*M))+1.0 );
            cout<<n+t-1<<endl;
        }
        else
        {
            t=ceil( (log10(N)/log10(2))+1.0 );
            cout<<t<<endl;
        }
    }
}
