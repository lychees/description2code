#include <iostream>
using namespace std;

long long int reverse(long long int x)
{
	long long int ret=0;
	while(x>0)
	{
		ret*=10;
		ret+=(x%10);
		x/=10;
	}
	
	return ret;
}

int main()
{
    long long int t, *a, *b, *sum, *rsum, *ra, *rb, i, j;
    cin>>t;
    a=(long long int*)malloc(t*sizeof(long long int));
    b=(long long int*)malloc(t*sizeof(long long int));
    ra=(long long int*)malloc(t*sizeof(long long int));
    rb=(long long int*)malloc(t*sizeof(long long int));
    sum=(long long int*)malloc(t*sizeof(long long int));
    rsum=(long long int*)malloc(t*sizeof(long long int));
    for(i=0;i<t;i++)
    {
    	cin>>a[i]>>b[i];
    	sum[i]=reverse(a[i])+reverse(b[i]);
    }
    for(i=0;i<t;i++)
    {
    	cout<<reverse(sum[i])<<endl;
    }
    return 0;
}
