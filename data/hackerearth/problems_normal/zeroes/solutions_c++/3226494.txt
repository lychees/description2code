#include<bits/stdc++.h>
using namespace std;
long long int a[100005],primecount[100005],mark[100005],f[100005];
vector<pair<int,int> >factors[100005];
int main()
{
    long long int n,q,x,i,j,b,k;
    scanf("%lld",&n);
    memset(mark,0,sizeof(mark));
    memset(primecount,0,sizeof(primecount));
    for(i=0;i<n;i++)
    {
        scanf("%lld",&a[i]);
        mark[a[i]]++;
    }
    for(i=2;i<=100000;i++)
    {
        if(f[i]!=0)
            continue;
        else
        {
            for(j=i;j<=100000;j+=i)
            {
                b=j;k=0;
                while(b%i==0)
                {
                	f[j] = 1;
                    k++;
                    b=b/i;
                }
                primecount[i]+= k*mark[j];
                factors[j].push_back(make_pair(i,k));
            }
        }
    }
    scanf("%lld",&q);long long int ans,d;
    while(q--)
    {
        ans=100000;
        scanf("%lld",&x);
        for(__typeof((factors[x]).begin()) i = (factors[x]).begin(); i != (factors[x]).end(); i++) {
			d = i->first;
			ans = min(ans, primecount[d] / i->second);
		}
        printf("%lld\n",ans);
    }
    return 0;
}
