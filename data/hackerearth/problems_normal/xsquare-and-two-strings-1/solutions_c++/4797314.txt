#include <stdio.h>
#include <iostream>
#include <algorithm>
#include <string.h>
using namespace std;
int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
    	int flag  = 0,i;
    	char ch1[100001],ch2[100001];
    	scanf("%s%s", ch1,ch2);
    	int a = strlen(ch1);
    	int b = strlen(ch2);
    	
    	sort(ch1, ch1+a);
    	sort(ch2, ch2+b);
    //	printf("%s\n%s\n", ch1,ch2);
  		if(a<=b)
  		{
  			for(i = 0; i<b;i++)
  			{
  				if(ch1[0] == ch2[i])
  				{
  					flag = 1;
  					break;
  				}
  			}
  			if(flag == 0)
  			{
  				printf("No\n");
  			}
  			else
  			{
  				printf("Yes\n");
  			}
  			
  			
  		}
  		else
  		{
  			for(i = 0; i<a;i++)
  			{
  				if(ch1[i] == ch2[0])
  				{
  					flag = 1;
  					break;
  				}
  			}
  			if(flag == 0)
  			{
  				printf("No\n");
  			}
  			else
  			{
  				printf("Yes\n");
  			}
  		}
    }
    return 0;
}
