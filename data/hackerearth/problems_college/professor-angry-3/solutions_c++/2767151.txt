    #include <iostream>
    using namespace std;
     
    int main()
    {
        int t;
        cin >>t;
        int a[t];
        char f[t];
        int n,k,x;
        for(int var=0;var<t;var++)
        {
        	a[var]=0;
        	cin>>n>>k;
        	for(int v=0;v<n;v++)
        	{
        		cin>>x;
        		if(x<=0)
        		{
        			a[var]++;
        		}
        	}
        	if(a[var]>=k)
        	{
        		f[var]='n';
        	}
        	else
        	{
        		f[var]='y';
        	}
        }
        for(int var=0;var<t;var++){
        	if(f[var]=='y')
        	cout<<"YES\n";
        	else 
        	cout<<"NO\n";
        }
        return 0;
    }
