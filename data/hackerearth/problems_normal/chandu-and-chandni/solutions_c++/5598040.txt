#include<bits/stdc++.h>
using namespace std;

int v[2002002],t;
double d=((double)((1+sqrt(5))/2.0));

int main(){
	for(int i=1;i<=1001001;i++)
		v[i]=-1;
	
	for(int i=1;i<=1001001;i++)
		v[(int)(d*(double)i)] = (int)(d*d*i);
	
	cin >> t;
	while(t--){
		int a,b;
		cin >> a >> b;
		if(v[a]==b)
			cout << "Chandu" << endl;
		else
			cout << "Chandni" << endl;
	}
	return 0;
}
