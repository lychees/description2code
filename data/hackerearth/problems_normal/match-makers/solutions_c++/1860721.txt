#include<stdio.h>
int main()
{
	int t,n,a[10000],b[10000],i,j,temp,count;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			//girls..a..ascending
		}
		for(i=0;i<n;i++)
		{
			scanf("%d",&b[i]);
			//boys..b.descending
		}
		for(i=0;i<n;i++)
		{
			for(j=i+1;j<n;j++)
			{
				if(a[j]<a[i])
				{
					temp=a[i];
					a[i]=a[j];
					a[j]=temp;
				}
			}
		}
		//for(i=0;i<n;i++)
			//printf("%d\t",a[i]);
		for(i=0;i<n;i++)
		{
			for(j=i+1;j<n;j++)
			{
				if(b[j]>b[i])
				{
					temp=b[i];
					b[i]=b[j];
					b[j]=temp;
				}
			}
		}
		//printf("\n");
				//for(i=0;i<n;i++)
			//printf("%d\t",b[i]);
	//	printf("\n");
		count=0;
		for(i=0;i<n;i++)
		{	
			if(a[i]%b[i]==0||b[i]%a[i]==0)
				count++;
		}
		printf("%d\n",count);			
	}
	return 0;
}
