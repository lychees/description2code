#include<bits/stdc++.h>
using namespace std;

typedef long long int LL;

int low[1000001] = {0};
int mark[1000001] = {0};
LL F[1000001] = {0};

void sieve()
{
	for(int i=2;i<=1000000;i++) mark[i] = 1,low[i] = i;
	for(int i=2;i<=1000000;i++)
	{
		if(!mark[i]) continue;
		int j = 2*i;
		low[i] = 0;
		while(j<=1000000)
		{
			if(low[j] == j)
			low[j] = i;
			mark[j] = 0; j += i;
		}
	}
}

int main()
{
	sieve();
	for(int i = 1;i<=1000000;i++) F[i] = (F[i-1] + low[i])%1000000007;
	int t; cin>>t;
	while(t--)
	{
		int a,b;
		cin>>a>>b;
		LL ans = F[b] - F[a-1];
		ans = (ans%1000000007 + 1000000007)%1000000007;
		cout<<ans<<endl;
	}
}
