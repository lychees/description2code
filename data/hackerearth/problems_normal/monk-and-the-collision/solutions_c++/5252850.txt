#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--){
	long long temp=0;
	int n;
	cin>>n;
	long long a[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		temp=(temp>a[i]%10)?temp:a[i]%10;
	}
	int b[temp+1];
	for(int i=0;i<temp+1;i++)
	b[i]=0;
	int count=0;
	for(int i=0;i<n;i++)
	{
		b[a[i]%10]++;
		if(b[a[i]%10]>1)
		count++;
	}
	cout<<count<<endl;}
}
