#include <iostream>
using namespace std;

int main()
{
   long int t,n;
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	int count=0;
   	if(n==3)
   	{
   		cout<<3<<endl;
   	}
   	if(n!=3)
   	{
   	for(int i=1;i<=n;i++)
   	{
   		if(n%i==0)
   		{
   			count++;
   		}
   	}
   	if(count==2)
   	{
   		cout<<2<<endl;
   	}
   	else
   	{
   		cout<<1<<endl;
   	}
   	}
   	}
   	
  
   	
   
}
