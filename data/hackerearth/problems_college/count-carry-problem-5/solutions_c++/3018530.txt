#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
    int N;
    cin >> N;
    while (N--) {
    	string a, b;
    	cin >> a >> b;
    	reverse(a.begin(), a.end());
    	reverse(b.begin(), b.end());
    	if (a.length() < b.length()) {
    		swap(a,b);
    	}
    	int count = 0;
    	int carry = 0;
    	for (int i = 0; i < b.length(); i++) {
    		carry = carry + (a[i] - '0') + (b[i] - '0');
    		carry = carry/10;
    		if (carry > 0) {
    			count++;
    		}
    	}
    	
    	for (int i = b.length(); i < a.length(); i++) {
    		carry = carry + (a[i] - '0');
    		carry /= 10;
    		if (carry > 0) count++;
    	}
    	if (count) {
    		if (count == 1) cout << count << " carry operation\n";
    		else cout << count << " carry operations\n";
    	}
    	else
    	cout << "No carry operation\n";
    }
    return 0;
}
