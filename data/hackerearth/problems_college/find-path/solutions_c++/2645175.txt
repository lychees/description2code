#include <iostream>
#include<malloc.h>
using namespace std;

int main()
{
	int t,n, *arr,*diff;
	int i,k;
	int cnt;
    cin>>t;
    while(t>0)
    {
    	cin>>n;
    	cnt=0;
    	arr=new int[n];
    	diff=new int[n];
    	for(int j=0;j<n;j++)
    	{
    		cin>>arr[j];
    	}
    	i=0;k=0;
    	while(i<n)
    	{
    		diff[k]=arr[i]-arr[i+1];
    		i++;
    		k++;
    	}
    	k--;

    	for(i=0;i<k;i++)
    	{
    		if(diff[i]>=0)
    			cnt++;
    	}
    	cout<<cnt<<endl;
    	t--;
    }

    return 0;
}
