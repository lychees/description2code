#include <bits/stdc++.h>
using namespace std;
long long int MOD = 1000000007;
long long int pow_mod(long long int p,long long int n)
{
    long long int x=1;
    while(n)
    {
        if(n&1)
            x=(x*p)%MOD;
        p=(p*p)%MOD;
        n>>=1;
    }
    return x;
}
long long int modInverse(long long int a,long long int m) {
    return pow_mod(a,m-2);
}
int main() 
{
    long long int t,a,b,n,x,y;
    cin>>t;
    a = 2;
    b = 7;
    while(t--) {
        cin>>n;
        if(n <= 3) {
            n = n==1?a:n==2?b:b+2*a;
            cout<<n<<endl;
            continue;
        }
        n-=2;
        x = pow_mod(2,n+1)-2*pow(-1,n)+MOD;
        x %= MOD;
        n++;
        y = (pow_mod(2,n)-pow(-1,n)+MOD);
        y %= MOD;
        x = (x*a+y*b)%MOD;
        y = modInverse(3,MOD);
        x = (x*y)%MOD;
        cout<<x<<endl;
    }
	return 0;
}
