#include<bits/stdc++.h>
using namespace std;
int x[]={-1,0,1,-1,1,-1,0,1};
int y[]={1,1,1,0,0,-1,-1,-1};
int dp[1005][1005];
int ans=0;
int t,m,n;
int check(int xx,int yy)
{
    int l,r;
    ++ans;
    dp[xx][yy]=2;
    for(int i=0;i<8;i++)
    {
        l=xx+x[i];
        r=yy+y[i];

        if(l>=0 && l<n && r>=0 && r<m && dp[l][r]==1)
        {

            check(l,r);
        }
    }

}
int main()
{

    cin>>t;
    while(t--)
    {


    int maxi=0,cnt=0;
    cin>>n>>m;
    for(int i=0;i<n;i++) for(int j=0;j<m;j++) cin>>dp[i][j];
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {


            if(dp[i][j]==1)
            {

            ans=0;
            check(i,j);
            maxi=max(maxi,ans);
            cnt++;
            }
        }
    }
    cout<<cnt<<" "<<maxi<<"\n";
    }
}
