n = long(raw_input())
numstr = raw_input().split(' ')
nums = map(long, numstr)
notzero = 0
for i in range(0, len(nums)-1):
    if (nums[i] > nums[i+1]):
        notzero = 1
    else:
        nums[i+1] = nums[i+1] - nums[i]
if (notzero == 0 and nums[len(nums)-1] == 0):
    print "YES"
else:
    print "NO"
