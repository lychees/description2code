#include <iostream>
#include <memory.h>
using namespace std;
#define N 1048580
typedef long int li;
li A[N], mask, dp[N][21] , freq[N];
int main()
{
    int t ,n ,i ,j;
    cin >> t;
    while(t--) {
    	memset(A , 0 , sizeof(A));
    	memset(dp , 0 , sizeof(dp));
    	memset(freq , 0 , sizeof(freq));
    	cin >> n;
    	for(i = 1 ; i <= n ; i++){
    		cin >> A[i];
    		freq[A[i]]++;
    	}
    	for(mask = 0 ; mask < 1<<20 ; mask++){
    	/*	dp[mask][0]=freq[mask];
			if(mask&1)
				dp[mask][0]+=freq[mask^1];*/
				dp[mask][-1] = freq[mask];
    		for(i = 0 ; i < 20 ; i++){
    			if(mask & (1<<i))
    				dp[mask][i] = dp[mask][i-1] + dp[mask^(1<<i)][i-1]; 
    			else
    				dp[mask][i] = dp[mask][i-1];
    		}
    	}
    	li sol = 0;
    	for(i = 1 ; i <= n ; i++){
    	    //cout << dp[A[i]][19] << " ";
    		sol += dp[(1ll<<20) - 1 - A[i]][19];
    	}
    	cout << sol << "\n";
    }
    return 0;
}
