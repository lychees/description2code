#include <iostream>
using namespace std;

#define ll long long
const ll mod = 1e9+7;

ll pm(ll a, ll b) {
	//cout << a << " " << b << endl;
	ll ans = 1;
	while(b) {
		if(b&1) ans = (ans * a)%mod;
		b >>= 1;
		a = (a*a)%mod;
	}

	return ans;
}
int main()
{
    int t;
    ll tmp;
    scanf("%d", &t);
    while(t--) {
    	scanf("%lld", &tmp);
    	printf("%lld\n", (pm(pm(2,tmp), mod-2)*2)%mod);
    }
    return 0;
}
