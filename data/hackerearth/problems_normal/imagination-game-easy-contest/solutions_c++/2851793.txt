#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t>0)
	{
		t--;
		int n;
		cin>>n;
		vector<int> v(n+1);

		for(int i=1;i<=n;i++)
		{
			cin>>v[i];    		
		}

		int cnt=0, tmp;
		int bb = 0;
		for(int i=1;i<=n;i++)
		{
			cin>>tmp;

			bb ^= (tmp-1);
			bb ^= (v[i]-tmp);
		}

		if(bb==0)
		{
			cout<<"Second\n";
		}
		else
			cout<<"First\n";    	

	}
	return 0;
}
