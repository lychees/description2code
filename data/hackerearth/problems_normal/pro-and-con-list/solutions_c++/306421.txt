using namespace std;

#define ll long long
#define vt vector

typedef struct girl
{
	ll f,a;
}girl;

int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		ll n;
		cin>>n;
		vector<girl> values(n);
		for(ll i=0;i<n;i++)
			scanf("%lld %lld",&values[i].f,&values[i].a);
		vector<ll> ans(n);
		ll tota = 0;
		for(ll i=0;i<n;i++)
		{
			tota += values[i].a;
			ans[i] = (values[i].f+values[i].a);
		}
		sort(ans.begin(),ans.end(),greater<ll>());
		cout << ans[0]+ans[1]-tota << endl;
	}
	return 0;
}
