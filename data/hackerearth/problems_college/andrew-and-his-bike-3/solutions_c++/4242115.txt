#include <bits/stdc++.h>
using namespace std;
bool visited[100001];
int arr[100001];

vector<int> v[100001];
int solve(int i);
int main()
{
   int n;
   cin>>n;
   fill(arr,arr+n+1,1);
   fill(visited,visited+n+1,0);
   arr[1]--;
   int u,x;
   for(int i=0;i<n-1;i++){
    cin>>u>>x;
    v[u].push_back(x);
    v[x].push_back(u);
   }
   solve(1);
   for(int i=1;i<=n;i++)
      cout<<arr[i]<<" ";
    return 0;
}

int solve(int i){
    visited[i]=1;
    for(int j=0;j<v[i].size();j++){

        if(!visited[v[i][j]]){
           //  cout<<i<<" "<<v[i][j]<<endl;
            arr[i]+=solve(v[i][j]);
        }
    }
    return arr[i];
}
