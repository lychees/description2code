#include <iostream>

using namespace std;

int main()
{
    int a[50][50],b[50][50];
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<n;j++)
    		{
    			cin>>a[i][j];
    			b[i][j]=0;
    		}
    	}
    	b[0][0]=1;
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<n;j++)
    		{
    			if((i==0 && j==0 ))
    				continue;
    			else if(i==0)
    			{
    				if(b[i][j-1]==1 && a[i][j-1]==1)
    					b[i][j]=1;
    			}
    			else if(j==0)
    			{
    				if(b[i-1][j]==1 && a[i-1][j]==1)
    					b[i][j]=1;
    			}
    			else
    			{
    				if((b[i-1][j]==1 && a[i-1][j]==1 ) || (b[i][j-1]==1 && a[i][j-1]==1))
    					b[i][j]=1;
    			}
    		}
    	}
    	if(b[n-1][n-1]==1)
    		cout<<"POSSIBLE"<<endl;
    	else
    		cout<<"NOT POSSIBLE"<<endl;
    	
    	
    }
    return 0;
}
