T = int(raw_input())
while T > 0:
    index = 0
    word = list(raw_input())
    reserve = raw_input()

    reserve = sorted(reserve)

    for i in range(len(word)):
        if index > (len(reserve)-1):
            break
        elif word[i] > reserve[index]:
            word[i] = reserve[index]
            index = index + 1

    print ''.join(word)

    T = T - 1
