#include<bits/stdc++.h>
using namespace std;
#define MAX 100005
#define Int long long
Int A[105];
Int dp[MAX]={0};
int main()
{
    Int N,maxm=0,i,j;
    cin>>N;
    for (i=1;i<=N;++i)
        cin>>A[i];
    for (i=1;i<MAX;++i)
    {
        Int cnt=0;
        Int sum=0;
        queue<Int>q;
        for (j=1;j<=N;++j)
        {
            q.push(A[j]);
            sum+=A[j];
            while (sum>i)
            {
                sum-=q.front();
                q.pop();
            }
            if (sum==i)
            {
                while (!q.empty())q.pop();
                ++cnt;
                sum=0;
            }
        }
        dp[i]=cnt;
        maxm=max(maxm,dp[i]);
    }
    cout<<maxm<<"\n";
    for (i=1;i<MAX;++i)
    {
        if (dp[i]==maxm)
            cout<<i<<" ";
    }
    return 0;
}
