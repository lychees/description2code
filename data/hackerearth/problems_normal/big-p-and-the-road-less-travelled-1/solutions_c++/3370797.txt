#include<bits/stdc++.h>
using namespace std;
#define ll long long
vector<int> v[100005];
int n;
ll dp[100005];
int  vis[100005];
ll dfs(int u)
{
	if(u==n)return  1;
	if(vis[u])return dp[u];
	vis[u]=1;
	ll ans=0;
	for(int i=0;i<(int)v[u].size();++i)
	{
		ans+=dfs(v[u][i]);
	}
	dp[u]=ans;
	return ans;
}
int main()
{
	int x,y;
	cin>>n;
	memset(vis,0,sizeof(vis));
	memset(dp,0,sizeof(dp));//asdfasdf
	for(;;)
	{
		cin>>x>>y;
		if(x==0 && y==0)break;
		v[x].push_back(y);
	}
	ll ans=dfs(1);
	cout<<ans<<endl;
	return 0;
}
