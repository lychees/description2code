#include<iostream>
using namespace std;
void swapw(long long *a, long long *b)
{
    long long temp;
   temp=*a;
   *a=*b;
   *b=temp;
}
void quick_sort(long long arr[], int l, int r, long long temp[])
{
    if(l<r)
{
    int k=(l+r)/2;
    int i,j;
    swapw(&arr[l], &arr[k]);
    swapw(&temp[l], &temp[k]);
    long long p=arr[l];
    for(i=l+1, j=l+1;j<=r;j++)
    {
        if(arr[j]<p)
        {
            swapw(&(arr[j]), &(arr[i]));
            swapw(&temp[j], &temp[i]);
            i++;
        }
    }
    swapw(&arr[l], &arr[i-1]);
    swapw(&temp[l], &temp[i-1]);
    k=i-1;
    quick_sort(arr, l, k-1, temp);
    quick_sort(arr, k+1, r, temp);
}
}
main()
{
	int t,n,i;

    //int arr[]={3,8,2,5};

    quick_sort(P,0,n-1,G);

    		}
    }

}
}
