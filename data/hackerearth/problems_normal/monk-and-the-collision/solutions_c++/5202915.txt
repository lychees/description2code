#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
    	int n;
    	cin >> n;
    	unordered_map<int,bool> M;
    	long long ans=0;
    	for(int i=0; i<n; i++){
    		int x;
    		cin >> x;
    		x=x%10;
    		if(M.find(x)!=M.end()) ans++;
    		M[x]=true;
    	};
    	cout << ans << endl;
    	
    };
    return 0;
}
