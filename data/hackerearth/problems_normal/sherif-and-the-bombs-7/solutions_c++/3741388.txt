#include <iostream>
using namespace std;

int plot[1001][1001] = {0};

int main()
{
    int N, x, y, r, curX, curY;
    cin>>N;
    for(int i=0; i<N; i++) {
    	cin>>x>>y>>r;
    	for(curX = max(1, x-r); curX<=min(x+r, 1000); curX++) {
    		for(curY=max(1, y-r); curY<=min(y+r, 1000); curY++) {
    			if((((curX - x) * (curX - x)) + ((curY - y) * (curY - y))) <= r * r) {
    				plot[curX][curY]++;
    			}
    		}
    	}
    }
    int count = 0;
    for(curX = 1; curX<=1000; curX++) {
    	for(curY=1; curY<=1000; curY++) {
    		if(plot[curX][curY] > 1) {
    			count++;	
    		}
    	}
    }
    cout<<count;
    return 0;
}
