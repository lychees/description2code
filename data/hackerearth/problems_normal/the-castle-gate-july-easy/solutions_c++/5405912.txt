#include <iostream>
using namespace std;
int main()
{
    int t,count,i,j,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	count=0;
    	for(i=1;i<=n;i++)
    	{
    		for(j=1;j<=n;j++)
    		{
    			if(i==j)continue;
    			else if((i^j)<=n)count++;
    		}
    	}
    	cout<<count/2<<"\n";
    }
    return 0;
}
