#include <iostream>
#include <set>
#include <vector>
using namespace std;

int main(){
    int n;
    cin >> n;
    int sum = 0;
    for (int i = 0; i < n; i++){
        int tmp;
        cin >> tmp;
        sum += tmp;
    }
    if (n*2-2 == sum) cout << "Yes" << endl;
    else cout << "No" << endl;
}
