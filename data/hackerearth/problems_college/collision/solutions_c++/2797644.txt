#include<iostream>
#include<algorithm>

int hash(int num){
	return num%10;
}
int main(){
	int t,cnt,n,index;
	int arr[10]={-1};
	long long x;
	scanf("%d",&t);
	while(t--){
		std::fill_n(arr, 10, -1);
		cnt=0;
		scanf("%d",&n);
		while(n--){
			scanf("%lli",&x);
			index=hash(x);
			if(arr[index]==-1)
				arr[index]=x;
			else
				cnt++;

		}
		printf("%d\n",cnt);
	}
	system("pause");
}
