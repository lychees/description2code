#include<bits/stdc++.h>
using namespace std;


string s3[6]={"username", "pwd", "profile", "role", "key", "ZZZ"};
int main(){
	int k=0;
	string s;
	cin >> s;
	s+="&ZZZ";
	vector<string> s2(5);
	for(int i=0;i<s.size();i++){
		if(s[i]=='='){
			for(int j=i+1;j<s.size();j++){
				if(s.substr(j,s3[k+1].size())==s3[k+1]){
					// cout << s.substr(i+1,j-i-2) << endl;
					s2[k]=s.substr(i+1,j-i-2);
					k++;
					i=j-1;
					break;
				}
			}
		}
	}
	
	for(int i=0;i<5;i++){
		cout << s3[i] << ": " << s2[i] << endl;
	}
	
	return 0;
}
