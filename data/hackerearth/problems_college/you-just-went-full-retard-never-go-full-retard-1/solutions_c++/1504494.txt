#include<bits/stdc++.h>

using namespace std;


int main()
{
	long long int n,t,flag,i;

	scanf("%lld",&t);
	while(t--)
	{
		flag=0;
		scanf("%lld",&n);
		if(n<1)
		printf("NO\n");
		else
		{
		for(i=2;i<=sqrt(n);i++)
		{
				if(n%i==0)
				{
				printf("NO\n");	
				flag=1;
				break;
				}
		}
		if(flag==0)
		printf("YES\n");
		}
	}
	return 0;
}
