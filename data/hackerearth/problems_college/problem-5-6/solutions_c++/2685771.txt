#include <bits/stdc++.h>

#define mp make_pair
#define pb push_back
#define ll long long

using namespace std;

const int nm = 1010;

int n, x, a[nm];
int f[nm];

int main() {
#ifdef LOCAL
	freopen("input.txt", "r", stdin);
#endif
	int t;
	scanf("%d", &t);
	while (t--) {
		scanf("%d%d", &n, &x);
		for (int i = 1; i <= n; ++i)
			scanf("%d", &a[i]);
		f[0] = 0;
		for (int i = 1; i <= n; ++i) {
			f[i] = 0;
			for (int j = 1; j <= i; ++j) {
				f[i] = max(f[i], f[i - j] + a[j] - x);
			}
		}
		printf("%d\n", f[n]);
	}
#ifdef LOCAL
	printf("\n%.3lf s\n", (double) clock() / CLOCKS_PER_SEC);
#endif
}
