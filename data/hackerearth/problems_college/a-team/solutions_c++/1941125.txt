#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long i,j,k,T;
    double x,y,x1,y1,x2,y2,xc,yc,ans,r,xm,ym;
    cin>>T;
    while(T--){
    	cin>>x1>>y1>>x2>>y2>>xc>>yc;
    	j=k=0;
    	if(yc*(x2-x1)!=xc*(y2-y1)){
	    	x=(y2*(x2-x1)-x2*(y2-y1))/(yc*x2/xc-yc*x1/xc-y2+y1);
	    //	cout<<x<<" "<<y<<endl;
	    	y=yc/xc*x;
	    	r=min(x1,x2);
	    	x2=max(x1,x2);
	    	x1=r;
	    	r=min(y1,y2);
	    	y2=max(y1,y2);
	    	y1=r;
	    	if(x1<=x&&x<=x2) j=1;
	    	if(y1<=y&&y<=y2) k=1;
	    	x1=x;
	    	y1=y;
	    	if(j==1 && k==1){
	    		j=k=0;
	    		r=0;
	    		x=min(r,xc);
	    		y=min(r,yc);
	    		xm=max(r,xc);
	    		ym=max(r,yc);
		    	if(x<=x1&&x1<=xm) j=1;
		    	if(y<=y1&&y1<=ym) k=1;
	    	}
    	}
    	else if(yc*x2==xc*y2){
    		r=0;
    		x=min(r,xc);
    		y=min(r,yc);
    		xm=max(r,xc);
    		ym=max(r,yc);
	    	if(x<=x1&&x1<=xm) j=1;
	    	if(y<=y1&&y1<=ym) k=1;
	    	if(j==1&&k==1) r=0;
	    	else j=k=0;
	    	if(x<=x2&&y2<=xm) j=1;
	    	if(y<=y2&&y2<=ym) k=1;
    	}
    	if(j==1&&k==1) cout<<"NO\n";
    	else cout<<"YES\n";
    	
    }
    return 0;
}
