#include<cstdio>
#include<cstdlib>
#include<cmath>
using namespace std;
int ar[1000000];
int f[1000000];
int g[1000000];
int div1[1000000];
#define MAX 1000000
void smallest()
{  int i,j,x,y,count,prod,y1;
ar[1]=0;
	for(i=2;i<=MAX;i++)
	{
		if(i%2==0)
		ar[i]=2;
		else
		ar[i]=i;
	}
	int check=sqrt(MAX);
	for(i=3;i<=check;i+=2)
	{
		if(ar[i]==i)
		{
			for(j=i;j<=MAX;j=j+i)
			{
				if(ar[j]==j)
				ar[j]=i;
			}
		}
	}
	f[1]=g[1]=0;
	for(i=2;i<=MAX;i++)
	{  x=i;
			 y=ar[i];
			 count=0;
		prod=1;
			while(y!=0)
			{
				count++;
				x=x/y;
				
				y1=ar[x];
				if(y1!=y)
				{
					prod=prod*((pow(y,count+1)-1)/(y-1));
					count=0;
				}
				y=y1;
			}
			prod=prod-i;
			div1[i]=prod;
	}

}
int main()
{
	int t;
	smallest();
	scanf("%d",&t);
	while(t--)
	{
		int n,x,y,y1,count,prod,i;
		scanf("%d",&n);
		if(n==1)
		printf("0\n");
		else
		{
		
		for(i=2;i<=n;i++)
		{
			f[i]=(f[i-1]%n+ar[i]%n)%n;
			
			 
		
		
			g[i]=(g[i-1]%n+div1[i]%n)%n;
		
		}
	
		printf("%d\n",(f[n]%n+g[n]%n)%n);
	}
	}
}
