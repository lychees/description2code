#include<bits/stdc++.h>
using namespace std;
int main()
{
int t,n,i,j,edg,*a,flag;
cin>>t;
while(t--)
{
cin>>n>>edg;
a=new int [n];
for(i=0;i<n;i++)
a[i]=0;
flag=0;
while(edg--)
{
cin>>i>>j;
a[i]++;
a[j]++;
}
for(i=0;i<n;i++)
{
if(a[i]%2!=0)
{
flag=1;
break;
}

}
if(flag)
    cout<<"BICOLORABLE."<<endl;  
else
 cout<<"NOT BICOLORABLE."<<endl;
 
}


return 0;
}
