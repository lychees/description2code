#include<iostream>
#include<cstdio>
#define gc getchar_unlocked
using namespace std;
int readInt()
{
	char c;
	int res=0;
	do{
		c=gc();
	}while(c<'0'||c>'9');
	do{
		res=res*10+c-'0';
		c=gc();
	}while(c>='0'&&c<='9');
	return res;
}
int main()
{
	ios::sync_with_stdio(false);
	int n=readInt();
	int a[n],i,cnt=0,t=0;
	for(int i=0;i<n;i++)
	a[i]=readInt();
	for(i=0;i<n;i++)
	{
		if(a[i]-a[i+1]==1||a[i]-a[i+1]==-1)
		cnt++;
		else
		{  if(cnt!=0)
		   cnt+=1; 
		   if(cnt>t)
			t=cnt;
		    cnt=0;
	}
   }
   cout<<t<<endl;
	return 0;
}
