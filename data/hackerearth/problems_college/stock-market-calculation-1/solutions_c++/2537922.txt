typedef long long ll;

#define INF 1000000000000000
#define MOD 1000000007

int arr[100010];
int main(){

	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;

	while(t--){

		int n;
		cin>>n;

		for(int i=0; i<n; i++){
			cin>>arr[i];
		}

		int dp[n];
		dp[n-1] = 0;
		int max_ele = arr[n-1];

		for(int i=n-2; i>=0; i--){
			if(arr[i] > max_ele)
				max_ele = arr[i];
			dp[i] = max_ele - arr[i];
		}
		int ans = 0;
		for(int i=0; i<n; i++){
			ans += dp[i];
		}
		cout<<ans<<"\n";

	}
	return 0;
}
