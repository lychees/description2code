#include<bits/stdc++.h>

using namespace std;

long long A[100002], B[100002];

int monkiness (int i, int n){
	int l, m, r, monk;
	l = i;
	r = n-1;
	while(r >= l){
		m = (l+r)/2;
		if(B[m] >= A[i]){
			monk = m-i;
			l = m+1;
		}
		else r = m-1;
	}
	return monk;
}
main(){
	int t, n, i, ans;
	cin>>t;
	while(t--){
		ans=0;
		cin>>n;
		for(i=0;i<n;i++) cin>>A[i];
		for(i=0;i<n;i++) cin>>B[i];
		for(i=0;i<n;i++)
			if(ans<monkiness(i,n)) ans = monkiness(i,n);
		cout<<ans<<endl;
	}
}
