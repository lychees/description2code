    int n,j,i;

    for( i=0;i<n;i++)
    {
    	cin>>j;
    	v.push_back(j);
    }
    sort(v.begin(),v.end());
    int ans=0;
    for(i=0;i<n;i++)
    {
    	if(i+2>=n)
    	break;
    	ans=max(ans,v[i+2]-v[i]);
    }
    cout<<ans;
}
