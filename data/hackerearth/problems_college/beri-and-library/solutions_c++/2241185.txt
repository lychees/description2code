#include<iostream>
#include<cstdio>

using namespace std;

int main()
{
    int tc;
    unsigned long long int n;
    scanf("%d",&tc);
    while(tc--){
        long int ans = 0;
        cin>>n;
        while(n!=0){
            if(n%2==1){
                ans++;
            }
            n = n/2;
        }
        printf("%ld\n",ans);
    }
    return 0;
}
