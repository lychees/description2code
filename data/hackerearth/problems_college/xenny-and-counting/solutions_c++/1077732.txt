#include <bits/stdc++.h>
using namespace std;
#define int long long
int dp[20][20]; 
main()
{
    dp[0][0]=1;
    for (int y=1;y<=19; y++)
    	{
    for (int g=0;g<=19; g++)
    {
    		for (int z=0;z<=g; z++)
    		{
    			dp[y][g]+=dp[y-1][g-z]; 
    		}
    	}
    }
    int b;cin>>b;
    for (int y=0;y<b; y++)
    {
    	int c,d;cin >> c >> d;
    	cout <<dp[c][d] << '\n'; 
    }
    return 0;
}
