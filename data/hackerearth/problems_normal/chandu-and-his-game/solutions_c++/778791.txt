#include <bits/stdc++.h>
using namespace std;
#define ll long long

int dist(int x1, int y1, int x2, int y2)
{
    return abs(x1 - x2) + abs(y1 - y2);
}

int main()
{
    ll n, k, a, b;;
    cin >> n >> k;

    ll ans = 0;

    for(int i = 0; i < k; i++) {
        cin >> a >> b;
        int p[4];
        p[0] = dist(1, 1, a, b);
        p[1] = dist(1, n, a, b);
        p[2] = dist(n, 1, a, b);
        p[3] = dist(n, n, a, b);
        sort(p, p + 4);
        ans += p[0];
    }
    cout << ans << endl;
    return 0;
}
