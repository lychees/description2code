#include<bits/stdc++.h>
using namespace std;
#define ll  long long
#define mod ((ll)1e9+9)
ll mpow(ll a, ll b)
{
	ll ret=1;
	//if(b>mod)b%=mod;
	while(b>0)
	{
		if((b&1)==1)ret=ret*a;
		if(ret>mod)ret%=mod;
		a=a*a;
		if(a>mod)a%=mod;
		b=b>>1;
	}
	return ret;
}
int main()
{	
	ll  tc;
	cin>>tc;
	while(tc--)
	{
		ll n;
		cin>>n;
		ll ans=0;
		if((n%2)==0)
		{
	//		puts("evevn");
			ans=mpow(26,n/2);
		}
		else 
		{
			ans=mpow(26,n/2)*26;
		}
		if(ans>mod)ans%=mod;
		cout<<ans<<endl;
	}
	return 0;
}
