#include <iostream>
#include<algorithm>
#include<functional>
using namespace std;

int main()
{
	int t,nr,ng,mywealth,i;
	cin >>t;
	for(int test=0;test<t;test++)
	{
		cin >>nr>>ng;
		int arr[nr];
		for( i=0;i<nr;i++)
		{
			cin >> arr[i];
		}
		mywealth=arr[0];
		sort(arr,arr+nr,greater<int>());
		bool flag=false;
		for( i=0;i<nr && !flag;i++)
		{
			if(arr[i] == mywealth)
			flag=true;
		}
		int strpos =i%ng-1;
		int count =0;
		for(i=strpos;i<nr;i=i+ng)
		{
			if(arr[i] > mywealth)
			count++;
		}
		cout << count << endl;
	}

}
