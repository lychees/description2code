#include <iostream>
using namespace std;

int main()
{
   int T,rem,flag,count;
   long long N,y,z;
   cin>>T;
   count=0;
   while(T--)
   {
   	flag=0;
   	cin>>N;
   	z=N;
   	if(N==1)
   	flag=25;
   	while(N/10!=0)
   	{	z=N;
   	y=0;
   	while(z!=0)
   	{
   	rem=z%10;
    z=z/10;
   	y=y+(rem*rem);
   	}
   	if(y==1)
   	{
   		flag=1;
   		break;
   }
   else
   {
   	flag=0;
   }
   N=y;
   }
   count++;
   if(flag==1)
   cout<<"Case #"<<count<<":YES"<<"\n";
   else if(flag==25)
   cout<<"Case #"<<count<<":YES"<<"\n";
   else
   cout<<"Case #"<<count<<":NO"<<"\n";
   }
    return 0;
}
