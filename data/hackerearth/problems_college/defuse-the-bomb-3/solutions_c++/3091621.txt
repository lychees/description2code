#include <iostream>
using namespace std;
int main() {
  unsigned long long cases,test,multip,mdiv=1,nodd,nod=0,mnod,divi=1,safe;
	cin>>cases;
	while(cases)
	{
		mdiv=1,nodd,nod=0,mnod=0,divi=1;
	    cin>>test;
	    cin>>multip;
	    safe=test;
	    while(safe)
	    {
	     safe=safe/10;
	     nod++;	
	    }
	   // cout<<nod;
	   nodd=nod;
	    nod=nod-2;
	    while(nod)
	    {
	        divi=divi*10;
	        nod--;
	    }
	    safe=multip;
	    
	    while(safe)
	    {
	        safe=safe/10;
	        mnod++;
	    }
	    mnod=mnod-2;
	    while(mnod)
	    {
	        divi=divi/10;
	        mnod--;
	    }
	    
	    while(nodd)
	    {
	        mdiv=mdiv*10;
	        nodd--;
	    }
	    mdiv=mdiv/10;
	   // cout<<mdiv<<endl;
	  while(divi)
	    {
	        cout<<(test/divi)*multip<<endl;
	        
	        test=test-((test/(mdiv))*mdiv);
	        mdiv=mdiv/10;
	       
	        divi=divi/10;
	    }
	    
	    cases--;
	}
	
	return 0;
}
