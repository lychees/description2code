#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t = 1;
    string s[500];
    char temp;
    cin >> t;
    if(t >= 1 && t <= 500){
    	for(int i = 0; i < t; i++){
    		cin >> s[i];
    		if(s[i].length() <= 1000)
    			continue;
    		else
    			break;
    	}
    }
    for(int i = 0; i < t; i++){
    	for(int j = 0; j < (s[i].length() - 1); j++){
    		for(int k = (j + 1); k < s[i].length(); k++){
    			if((int)s[i].at(j) >= (int)s[i].at(k)){
    				temp = s[i].at(j);
    				s[i].at(j) = s[i].at(k);
    				s[i].at(k) = temp;
    			}
    		}
    	}
    	cout << s[i] << endl;
    }
    return 0;
}
