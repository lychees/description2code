#include<iostream>
#define max 2147483646
using namespace std;
int isOdd(int x)
{
	return (x%2)?1:0;
}

int main()
{
	int n,sum=0,odd_val_min=max;
	cin>>n;
	while(n--)
	{
		int i;
		cin>>i;
		sum+=i;
		if(isOdd(i)&&(odd_val_min>i))
		odd_val_min=i;
	}
	if(!isOdd(sum)&&(odd_val_min))
	sum-=odd_val_min;
	if(isOdd(sum))
	cout<<sum<<"\n";
	else
	cout<<":(";
	return 0;
}
