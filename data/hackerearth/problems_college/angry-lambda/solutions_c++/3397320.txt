#include <bits/stdc++.h>
using namespace std;
#define max 1000000007
int main()
{
	ios_base::sync_with_stdio(false);
	int t,k,n,temp;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		int count=0;
		count=0;
		for(int i=0;i<n;i++)
		{
			cin>>temp;
			if(temp<=0)
				count++;
		}
		if(count<k)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
	return 0;
}
