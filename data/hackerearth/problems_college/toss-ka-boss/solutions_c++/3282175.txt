#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main(){

   int t;
   cin >> t;
   while( t-- ){

      string x;
      cin >> x;

      ll ans = 0;
      int ct = 0;
      for(int i = 0 ; i < x.size() ; ++i ){
        if(x[i]=='H'){
        ct++;
        ans = ans + ct;
        }
        else ct = 0;

       }

       cout << ans << endl;


   }

}
