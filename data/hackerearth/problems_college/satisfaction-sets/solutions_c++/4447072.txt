#include <bits/stdc++.h>
using namespace std;

#define LL long long int

int main()
{
    LL n, ans = 0, m, num, i;
    cin >> n;
    vector < LL > M(501, 0);
    for(i=0; i<n; i++){
        cin >> m;
        vector < LL > N(501, 0);
        while(m--){
            cin >> num;
            if(!N[num]){
                M[num]++;
                N[num] = 1;
            }
        }
    }
    for(i=0; i<=500; i++)
        if(M[i] == n)
            ans += i;
    cout << -1*ans << endl;
    return 0;
}
