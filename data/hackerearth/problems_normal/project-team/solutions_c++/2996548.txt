#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        int a1[n],i;
        for(i=0;i<n;i++)
            cin>>a1[i];
        sort(a1,a1+n);
        int a2[n/2];
        for(i=0;i<n/2;i++)
            a2[i]=a1[i]+a1[n-1-i];
        int min=20000,max=0;;
        for(i=0;i<n/2;i++){
            if(max<a2[i])
                max=a2[i];
            if(min > a2[i])
                min=a2[i];
        }
        cout<<max-min<<endl;
    }
    return 0;
}
