const int N=1e6+10;
int cases,x;
bool vld[N];
set<int>s;
set<int>::iterator it,it2;



int main()
{
	std::ios::sync_with_stdio(0);
	cin.tie(0);
	//s.insert(0);
	s.insert(1);
	for(int i=2;i<=1000;i++)
	{
		for(int j=i*i;j<=1000000;j*=i)s.insert(j);
	}
	for(int i=0;i<=1000000;i++)vld[i]=0;
	for(it=s.begin();it!=s.end();it++)
	{
		for(it2=s.begin();it2!=s.end();it2++)
		{
			if((*it)+(*it2)<=1000000)vld[(*it)+(*it2)]=1;
		}
	}
	cin>>cases;
	while(cases--)
	{
		cin>>x;
		if(vld[x])cout<<"Yes\n";
		else cout<<"No\n";
	}
	return 0;
}
