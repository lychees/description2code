#include <iostream>
using namespace std;  
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        int a,b,f=0;
        cin>>a>>b; 
        while(1)
        {
            if(a%2==0&&b%2==0)
            {
                a=a/2;
                b=a/2;f++ ; 
                break;
            }
            if(a%2==0&&b%2!=0)
            {
                a=a/2;
                b=a/2;f++; 
                break;
            }
            if(a%2!=0&&b%2==0)
            {
                a=b/2;
                b=b/2;f++; 
                break;
            }
            if(a%2!=0&&b%2!=0) break;  
        }
        if(f%2==0)  cout<<"Nilesh"<<endl; 
        else cout <<"Shekhar"<<endl; 
    }return 0;
}
