#include <bits/stdc++.h>

using namespace std;

long long ans [100 + 1];

int main()
{

    long long n , mod = 1e9 + 7;
    string s;

    ans[1] = 1, ans[2] = 1, ans[3] = 2;

    for(long long i = 4; i <= 100; i++){

        ans[i] = ( (i - 1) % mod * ((ans[i - 1] % mod + ans[i - 2] % mod) % mod)) % mod;
    }

    cin >> n >> s;

    cout << ans[n];


    return 0;
}
