#include<bits/stdc++.h>
using namespace std;
int main()
{
	 int n,k=0,x[1000010];
	cin>>n;

	bool vis[1000010];

	for( int i=1;i<=n;i++)
	{
		vis[i]=true;
	}

	for( int i=0;i<n;i++)
	{
		 int a;
		cin>>a;

		if(vis[a]==true)
		{
			x[k++]=a;
		}
		vis[a]=false;
	}

	sort(x,x+k);
	
	long long int count=0;

	for( int j=0;j<k;j++)
	{
		count+=j;
	}

	cout<<count<<endl;

	return 0;
}
