int main()
{	
	long long unsigned t;
	int sher=1;
	scanf("%llu", &t);
	while(t--)
	{	
		long long unsigned n, cnt=0;
		scanf("%llu", &n);
		while(n)
		{
			if(n&1)
				cnt++;
			n>>=1;
		}	
		if((cnt&1) == sher)
		{
			printf("Sherlock\n");
			sher=1;
		}
		else
		{
			printf("Watson\n");
			sher=0;
		}
	}
	return 0;
}
