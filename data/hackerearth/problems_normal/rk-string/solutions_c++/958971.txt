int m,n;

int a[1000009];
int b[1000009];
int nn;

ill f(int s[], int x, int req)
{
    int sum = 0;
    if (x != 0) sum=s[x-1];

    int ii = lower_bound (s+x, s+nn, req+sum)-s;
    if (ii == nn) return 0;
    int jj = upper_bound (s+x, s+nn, req+sum)-s;
    jj--;
    return (ill)(jj-ii+1);
}

int main()
{
   // freopen ("input.txt", "r", stdin);

    int t;
    S (t);
    while (t--) {
        S (m);
        S (n);
        cin >> s;
        nn = s.size();
        int i;
        F (i, 0, s.size()) {
            a[i] = b[i] = 0;
            if (s[i] == 'R') a[i] = 1;
            else b[i] = 1;
            if (i == 0) continue;
            a[i]+=a[i-1];
            b[i]+=b[i-1];
        }

        ill aaa,bbb;
        aaa = 0;
        bbb = 0;

        F (i, 0, s.size()) {
            aaa+= f(a, i, m);
            bbb+= f(b, i, n);
        }

        cout << aaa << " " << bbb << endl;

    }

    return 0;
}
