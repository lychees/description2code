#include <bits/stdc++.h>
using namespace std;
 
int main()
{
	int lemurs, bananas;
	cin >> bananas >> lemurs;
	int minutes = INT_MAX;
	for (int i = 1; i <= 100000; i++) {
		for (int j = 0; j <= 100000; j += i)
		    // suppose after certain operations, the final state has i bananas and j lemurs.
		    // no of operations it will take to reach this operation is abs(bananas - i) + abs(lemurs - j)
		    // take minimum of all such possibilities.
		    
		    // outer loop runs 10^6 times, inner loop runs 10^6 / i times;
		    // which is O(N*log(N)) in average case. Acceptable.
		    if (j % i == 0) {
				minutes = min(minutes, abs(lemurs - i) + abs(bananas - j));
		    }
	}
	cout << minutes << endl;
    return 0;
}
