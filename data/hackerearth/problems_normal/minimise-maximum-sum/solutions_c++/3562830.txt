#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t,n;
   scanf("%d",&t);
   while(t--)
   {
   	scanf("%d",&n);
   	vector<int> v;
   	int i,x;
   	n=2*n;
   	for(i=0;i<n;i++)
   	{
   		scanf("%d",&x);
   		v.push_back(x);
   	}
   	sort(v.begin(),v.end());
   	int ans = v[0] + v[n-1];
   	for(i=0;i<n/2;i++)
   	ans = max(ans,v[i]+v[n-1-i]);
   	printf("%d\n",ans);
   }
    return 0;
}
