#include<bits/stdc++.h>
using namespace std;
int ans=0;
vector<int> adj[100001];
int dfs(int cur,int pr,int dis){
	ans=max(ans,dis);
	int ma=0,ma1=0;
	for(int i=0;i<adj[cur].size();i++){
		if(adj[cur][i]!=pr){
			int x;
		  x=dfs(adj[cur][i],cur,dis+1);
		  if(x>ma){
		  	ma1=ma;ma=x;
		  } else if(ma1<x){
		  	ma1=x;
		  }
		}
	}
	ans=max(ans,max(ma+ma1,ma+1));
	return ma+1;
}
int main(){
	int n;
	scanf("%d",&n);
	for(int i=0;i<n-1;i++){
		int a,b;
		scanf("%d %d",&a,&b);
		adj[a].push_back(b);
		adj[b].push_back(a);
	}
	dfs(1,0,0);
	printf("%d\n",ans);
	return 0;
}
