N, M = [int(i) for i in raw_input().split()]

days = [int(i) for i in raw_input().split()]
ans = None
ghost = {}
for i in range(len(days)):
    try:
        ghost[days[i]] += 1
    except KeyError:
        ghost[days[i]] = 1

    if ans is None:
        ans = (days[i], 1)
    else:
        if ans[1] < ghost[days[i]] or (
                ans[1] == ghost[days[i]] and ans[0] < days[i]):
            ans = (days[i], ghost[days[i]])

    print ans[0], ans[1]
