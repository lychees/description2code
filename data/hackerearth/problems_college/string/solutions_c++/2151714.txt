#include <bits/stdc++.h>
using namespace std;

char A[200];

pair<int, int> cnt[30];
int del_me[30];

int main(){
	int n, k, pos = 26;
	cin>>A;
	n = strlen(A);
	cin>>k;

	for(int i = 0; i < 26; i++)	cnt[i].second = i, cnt[i].first = 0;

	for(int i = 0; i < n; i++)	cnt[A[i] - 'a'].first++;

	sort(cnt, cnt + 26);
	for(int i = 0; i < 26; i++){
		if(k <= 0)	break;
		if(k >= cnt[i].first)	del_me[cnt[i].second] = cnt[i].first, k -= cnt[i].first, pos = 25 - i;
		else k = 0;
	}

	cout<<pos<<endl;

	for(int i = 0; i < n; i++){
		if(del_me[A[i] - 'a'] == 0)	cout<<A[i];
		else	del_me[A[i] - 'a']--;
	}
	cout<<endl;
}
