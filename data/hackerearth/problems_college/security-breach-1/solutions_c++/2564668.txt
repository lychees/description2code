#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str;
		cin>>str;
		int i,len = str.length();
		for(i=0;i<len;i++)
		{
			if(str[i]=='2')
			{
				cout<<"cde";
			}
			if(str[i]=='7')
			{
				cout<<"acf";
			}
			if(str[i]=='6')
			{
				cout<<"b3";
			}
			if(str[i]=='9')
			{
				cout<<"c6a";
			}
		}
		cout<<"\n";
	}
    return 0;
}
