#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	int n;
	while(t--){
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++){
			cin>>a[i];
		}
		int b[n];
		b[0]=a[0];
		for(int i=1;i<n;i++){
			b[i]=a[i]+b[i-1];
		}
		int x,y,flag=0;
		if(b[n-1]-b[1]==0){
			cout<<"YES"<<"\n";
		}else{
		for(int i=1;i<n;i++){
			x=b[i]-a[i];
			y=b[n-1]-b[i];
			if(x==y){
				flag=1;
				break;
			}else{
				flag=-1;
			}
		}
	if(flag==1){
		cout<<"YES"<<"\n";
	}else{
		cout<<"NO"<<"\n";
	}
	}
	}
    return 0;
}
