#include <iostream>
#define get getchar_unlocked()
using namespace std;

inline int scan(){
	long long n	=0 , ch = get;
	while(ch<48 || ch>57)
		ch = get;
	while(ch>=48 && ch<=57){
		n = (n<<3) + (n<<1) + ch -48;
		ch = get;
	}
		
	return n;
}


int main()
{
    string s;
    long long n;
    long long len,i,k;
    n = scan();
    while(n--){
    	k=0;
    	cin>>s;
    	len = s.length();
    	for(i=1;i<len;i++){
    		if(s[i]==s[i-1])
    			k++;
    	}
    	printf("%lld\n",k);
    }
    
    return 0;
}
