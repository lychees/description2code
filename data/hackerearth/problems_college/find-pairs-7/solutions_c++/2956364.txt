#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int a[n];
		int i,j;
		long long co=0;
		for(i=0;i<n;i++)
		scanf("%d",&a[i]);
		for(i=0;i<n;i++)
		{
			for(j=i;j<n;j++)
			{
				if(a[i]==a[j])
				co++;
			}
		}
		printf("%lld\n",co);
	}
	return 0;
}
