#include <iostream>
#include<math.h>
typedef long long lld;
using namespace std;

lld gcd(lld a, lld b)
{
  
    if (a == b)
        return a;
 
   
    if (a > b)
        return gcd(a-b, b);
    return gcd(a, b-a);
}
lld kgcd(lld g,lld k)
{
	lld l=1,i;
	for (i = g/2; i>=1; i--)
	{ 
		if (g % i == 0)
		{
			l++;
			if(l==k)
				return i;
		}
   	 		
	}
	return -1;
}

int main()
{
    lld t;
    lld a,b,k,g;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>k;
    	g=gcd(a,b);
    	if(k==1)
    	{
    		cout<<g<<endl;
    		continue;
    	}
    	g=kgcd(g,k);
    	if(g==-1)
    		cout<<"No crymeth today\n";
    	else
    		cout<<g<<endl;
    }
    return 0;
}
