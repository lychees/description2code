#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tc, lens, sr, i, j, k;
    string s;
    
    cin >> tc;
    tc++;
    
    while (tc--) {
    	getline(cin, s);
    	lens = s.length();
    	
    	sr = sqrt(lens);
    	
    	char grid[sr][sr];
    	
    	k = 0;
    	for (i = 0; i < sr; i++) {
    		for (j = 0; j < sr; j++) {
    			grid[i][j] = s[k++];
    		}
    	}
    
    	for (i = 0; i < sr; i++) {
    		for (j = i + 1; j < sr; j++) {
    			swap(grid[i][j], grid[j][i]);
    		}
    	}
    	
    	k = 0;
    	for (i = 0; i < sr; i++) {
    		for (j = 0; j < sr; j++) {
    			s[k++] = grid[i][j];
    		}
    	}
    
    	cout << s << endl;
    }
    
    return 0;
}
