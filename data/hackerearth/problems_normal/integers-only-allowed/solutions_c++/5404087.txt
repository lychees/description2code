lng A=0, B=0, lastX, lastY, X, Y;

lng gcd(lng a,lng b)
{
	return ((b == 0) ? a : gcd(b,a%b));
}
lng cnt()
{
	if(lastX == X) return abs(Y - lastY)-1;
	if(lastY == Y) return abs(X - lastX)-1;
	return gcd(abs(Y-lastY),abs(X-lastX))-1;
}
int main()
{
	int i,j,k,n;
	cin >> n;
	cin >> lastX >> lastY;
	lng xx = lastX, yy = lastY;
	fr(i,1,n)
	{
		cin >> X >> Y;
		A += (X + lastX)*(lastY - Y);
		B += cnt();
		lastX = X;
		lastY = Y;
	}
	X = xx, Y = yy;
	A += (X + lastX)*(lastY - Y);
	B += cnt();
	B += n;
	cout << (A - B + 2)/2;
	return 0;
}
