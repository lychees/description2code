#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	char str[105];
	int i,l;
	cin>>str;
	l=strlen(str);
	int a,b,t;
	cin>>t;
	for(int j=1;j<=t;j++)
	{
	int count=0;
	cin>>a>>b;
	for(i=a;i>=a&&i<b;i++)
		{
		if(str[i-1]==str[i])
			{
				count++;
			}
		}
	cout<<count<<endl;
	}
	return(0);
}
