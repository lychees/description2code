#include <iostream>
#include <map>
#include <string>
using namespace std;

int main()
{
    int T;
    string input;
    map<string,int> names;
    cin>>T;
    while(T--)
    {
    	cin >> input;
    	names[input]++;
    }
    map<string,int>::iterator it = names.begin();
    for(; it != names.end(); ++it)
    cout<< (*it).first << " " << (*it).second << endl;
    return 0;
}
