#include<iostream>
#include<algorithm>
#include<vector>
#include<string.h>
using namespace std;
int main()
{
  int t;
  cin>>t;
  while(t--) {
    char s[105][105];
    int i,j,n,m,f=0,ones=0;
    cin>>n>>m;
    for(i=0;i<=n+1;i++) {
     for(j=0;j<=m+1;j++) s[i][j]='0';
    }
    for(i=1;i<=n;i++) {
      for(j=1;j<=m;j++) cin>>s[i][j];
    }
    for(i=1;i<=n && f==0;i++) {
      for(j=1;j<=m && f==0;j++) {
        if(s[i][j]=='1') {
          ones++;
          if(s[i-1][j]=='1' || s[i+1][j]=='1') f=1;
          else if(s[i][j-1]=='1' || s[i][j+1]=='1') f=1;
        }
      }
    }
    if(ones<1) cout<<"no 2\n";
    else if(f==1) cout<<"yes 0\n";
    else cout<<"no 1\n";
  }
//  system("pause");
  return 0;
}
