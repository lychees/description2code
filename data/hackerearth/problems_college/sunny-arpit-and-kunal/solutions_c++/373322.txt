#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,a[3];
    scanf("%d",&t);
    while(t--){
    	
    	scanf("%d %d %d",&a[0],&a[1],&a[2]);
    	sort(a,a+3);
    	
    	if(a[1]-a[0] >= a[2]-a[1]) printf("%d \n",a[1]-a[0]-1);
    	else printf("%d \n",a[2]-a[1]-1);    	
    	
    }
    return 0;
}
