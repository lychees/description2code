#include <iostream>
#include <cmath>
using namespace std;

long long fun(long long a,long long b)
{
	long long res=1;
	while(b)
	{
		if(b&1)res=res*a;
		a=a*a;
		b>>=1;
	}
	return res;
}


int main()
{
	long long t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		
		if(n==1)
		{
			cout<<0<<" "<<1<<endl;
			continue;
		}
		
		long long bits = (log2(n-1))+1;
		
		long long res=fun(2,bits-1);
		
		long long res1= 2*(n-res);
		
		cout<<(2*res-1)<<" "<<res1<<endl;
		
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
