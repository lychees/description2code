#include <iostream>
#include <string>
using namespace std;
int main()
{
    int t,n,i,j,number[100],temp,max=0,min=0;
    cin >> t;
    while(t--)
    {
        cin >> n;
        temp = n;
        while(temp--)
        {
            cin >> number[temp];
        }
        for (i = 0; i < n-1; i++)
        {
            for (j = i + 1; j < n; j++)
            {
                if (number[i] > number[j])
                {
                    temp =  number[i];
                    number[i] = number[j];
                    number[j] = temp;
                }
            }
        }
        i=(n/2)-1;
    	max=number[i]+number[n-1-i];
    	min=number[i]+number[n-1-i];
    	while(i--){
    		temp=number[i]+number[n-1-i];
    		if(temp>max){
    			max=temp;
    		}
    		if(temp<min){
    			min=temp;
    		}
    	}
    	cout << (max-min) << "\n";
    }
    return 0;
}
