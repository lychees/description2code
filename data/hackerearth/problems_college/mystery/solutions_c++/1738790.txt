#include<iostream>

using namespace std;

long long  sum=0;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long a,b,c;
        cin>>a>>b>>c;
        if(c==0)
            cout<<a<<"\n";
        else if(c==1)
            cout<<b<<"\n";
        else
        {
            for(int i=1;i<c;i++)
            {
                sum=a+b;
              //  cout<<"sum="<<sum<<"\n";
                a=b;
                b=sum;
            }
        cout<<sum<<"\n";
        }
    }
return 0;
}
