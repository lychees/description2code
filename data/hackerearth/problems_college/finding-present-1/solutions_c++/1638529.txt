#include <iostream>
using namespace std;

int main()
{
    int n, m, a[10001] = {0}, b[10001] = {0}, x;
    
    cin >> n;
    for (int i = 0; i < n; ++i)
    {
    	cin >> x;
    	a[x]++;
    }
    
    cin >> m;
    for (int i = 0; i < m; ++i)
    {
    	cin >> x;
    	b[x]++;
    }
    
    for (int i = 0; i < 10001; ++i)
    	if (a[i] != b[i])
    		cout << i << " ";
    return 0;
}
