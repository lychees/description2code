#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int a[110],dp[100010][110];
vector<int> ans;

int main()
{
	ios_base::sync_with_stdio(0);
	int n,maxi,temp,index,len;
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>a[i];
	maxi=0;
	for(int i=1;i<=100000;i++)
	{
		temp=0;
		index=1;
		for(int j=1;j<=n;j++)
		{
			temp+=a[j];
			if(temp==i)
			{
				dp[i][j]=1+dp[i][index-1];
				temp-=a[index];
				index++;
				maxi=max(maxi,dp[i][j]);
			}
			else if(temp>i)
			{
				while(temp>i)
				{
					temp-=a[index];
					dp[i][index]=max(dp[i][index],dp[i][index-1]);
					index++;
				}
				if(temp==i)
				{
					dp[i][j]=1+dp[i][index-1];
					temp-=a[index];
					index++;
					maxi=max(maxi,dp[i][j]);
				}
			}
		}
	}
	cout<<maxi<<"\n";
	len=0;
	for(int i=1;i<=100000;i++)
		for(int j=1;j<=n;j++)
			if(dp[i][j]==maxi)
			{
				ans.push_back(i);
				len++;
				break;
			}
	sort(ans.begin(),ans.end());
	for(int i=0;i<len;i++)
		cout<<ans[i]<<" ";
	cout<<"\n";
	return 0;
}
