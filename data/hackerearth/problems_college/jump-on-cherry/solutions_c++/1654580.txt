#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
  int tt;
  scanf("%d",&tt);
  while(tt--) {
    ll x,y;
    set<ll>s;
    scanf("%lld%lld",&x,&y);
    ll temp=y;
    ll dist=0;
    ll t=0;
    while(true) {
      if(s.find(dist)!=s.end()) {
        printf("%lld\012",t);
        break;
      }
      dist+=x;
      s.insert(dist+temp);
      temp+=y;
      t++;
    }
  }
  return 0;
}
