k = input()
s = raw_input()

l = len(s)
result = 0
for i in range(l):
	t = set(s[i:i+k])
	for j in range(i + k - 1, l):
		t.add(s[j])
		# print t
		if len(t) == k: result += 1
		elif len(t) > k: break
print result
