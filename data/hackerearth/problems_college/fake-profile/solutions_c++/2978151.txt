#include <iostream>
#include <vector>

using namespace std;

int main()
{
	ios_base::sync_with_stdio( false );
	string s;
	int n, i, ct = 0;
	
	cin >> s;
	n = s.length();
	
	vector<bool> A( 26, true );
	
	for( i = 0;i < n;i++ ) {
		if( A[ s[ i ] - 97 ] ) {
			ct++;
			A[ s[ i ] - 97 ] = false;
		}
	}
	
	if( ct & 0x1 ) {
		cout << "IGNORE HIM!\n";
	} else {
		cout << "CHAT WITH HER!\n";
	}
	return 0;
}
