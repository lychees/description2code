#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    int n;
   cin>>n;
   long int a[n],i,j,count=0;
   for(i=0;i<n;i++)
   cin>>a[i];
   for(i=0;i<n;i++)
   {
   	for(j=i+1;j<n;j++)
   	{
   		if(a[i]==a[j])
   		{
   		count++;
   		break;
   		}
   	}
   }
  
   cout<<count;
}
