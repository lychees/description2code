#include <stdio.h>

int main()
{
	long long int testCases;
	scanf("%lld", &testCases);
	long long int n;
	for(int c=0;c<testCases;c++)
	{	
		long long int s = 0;
		scanf("%lld", &n);
		long long int term;
		long long int total = 0;
		for(int i=0;i<n;i++)
		{
			scanf("%lld", &term);
			s += term;
			total += term;
			if(s >= n)
				s %= n;
		}
		if(s == 0)
			printf("YES\n%lld 0\n", (total / n));
		else
			printf("NO\n%lld %lld\n", (total / n), s);
	}
	return 0;
}
