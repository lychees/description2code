#define LL long long
#define MOD 1000000007
#define MAXN 2000007

LL fct[MAXN],ifct[MAXN];

LL fastpow(LL a,LL b)
{
    LL ans=1;
    while(b)
    {
        if(b&1)
            ans=(ans*a)%MOD;
        a=(a*a)%MOD;
        b>>=1;
    }
    return (LL)ans;
}

void fact()
{
    fct[0]=1;
    ifct[0]=1;
    fct[1]=1;
    ifct[1]=1;
    LL i=2;
    while(i<2000005)
    {
        fct[i]=(fct[i-1]*i)%MOD;
        ifct[i]=fastpow(fct[i],MOD-2);
        i++;
    }
}
int main()
{
    int T;
    fact();
    si(T);
    while(T--)
    {
        int N;
        si(N);
        long long ans = ((fct[2*N]*ifct[N]%MOD)*ifct[N+1]%MOD)%MOD;
        pl(ans);pn;
    }
    return 0;
}
