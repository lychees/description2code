#include<bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	vector<int> v;
	for(int i=0;i<n;i++){
		int a;
		cin>>a;
		v.push_back(a);
	}
	sort(v.begin(),v.end());
	vector<int> v1,v2;
	v1.push_back(v[0]);
	v2.push_back(v[0]);
	for(int i=1;i<n-1;i++){
		if(i%2){
			v1.push_back(v[i]);
		}else{
			v2.push_back(v[i]);
		}
	}
	
	v1.push_back(v[n-1]);
	v2.push_back(v[n-1]);
//	for(int i=0;i<v1.size();i++)

	
//	for(int i=0;i<v1.size();i++)cout<<v1[i]<<" ";
//	for(int i=0;i<v2.size();i++)cout<<v2[i]<<" ";
	
	int i=1;
	int j=1;
	int diff=INT_MIN;
	while((i<v1.size()-1) && (j<v2.size()-1)){	
		int t1=abs(v1[i]-v1[i-1]);
		int t2=abs(v2[j]-v2[j-1]);
		diff=max(diff,max(t1,t2));
		i++;
		j++;
	}
	if(i<v1.size()-1){
		int t1=abs(v1[i]-v1[i-1]);
		diff=max(diff,t1);
	}
	if(j<v2.size()-1){
		int t1=abs(v2[i]-v2[i-1]);
		diff=max(diff,t1);
	}
	cout<<diff<<endl;
	return 0;
}
