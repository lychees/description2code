long long solve(long long x,long long y)
{
    long long c=0;
    while(x>1 || y>1)
    {
        if(x<y)
        {
            long long k=y/x;
            y=y-k*x;
            c+=k;
        }
        else
        {
            long long k=x/y;
            x=x-k*y;
            c+=k;
        }
    }
    if(c>0) c--;
    return c;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long a,b;
        scanf("%lld%lld",&a,&b);
        long long ans=solve(a,b);
        printf("%lld\n",ans);
    }
   return 0;
}
