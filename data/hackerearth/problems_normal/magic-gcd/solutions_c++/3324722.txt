#include<iostream>
using namespace std;
int gcd(int a,int b)
{   if(b==0)
        return a;
    return gcd(b,a%b);
}
int maxim(int a, int b)
{   if(a>=b)
        return a;
    else
        return b;
}
int main()
{   int t;
    cin>>t;
    for(;t>0;--t)
    {   int n1,n2,k,ans=0;
        cin>>n1>>n2>>k;
        for(int i=0;i<=k;i++)
	    { 
	    	int imin=maxim(n1-i,1);
	        int jmin=maxim(n2-k+i,1);
	        int imax=n1+i;
	        int jmax=n2+k-i;
	
	        for(int l=imin;l<=imax;l++)
	        {   for(int j=jmin;j<=jmax;j++)
	            {   ans=maxim(ans,gcd(l,j));
	            }
	        }
	    }
        cout<<ans<<endl;
    }
}
