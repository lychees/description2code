#include <iostream>
#include <algorithm>
#include <math.h>
#include <stdlib.h>

using namespace std;

bool double_is_int(double num) {
   int i = num;
   return i == num;
}

bool check(double P,double x,double y,double u,double v){
		int time;
		double x1 = (u+x-y+v)/2;
		double y1 = (u-x+y+v)/2;
		double x2 = (u+x+y-v)/2;
		double y2 = (x-u+y+v)/2;
		if(x==u and y==v)
			time = 0;
		else if((y2-y1)/(x2-x1) == 1 or (y2-y1)/(x2-x1) == -1)
			time = 1;
		else if((double_is_int(x1) and double_is_int(y1)) or (double_is_int(x2) and double_is_int(y2)))
			time = 2;
		else
			time = P+1;
		if(time<P)
			return true;
		return false;
}

int main(){
	double T,P,x,y,u,v;
	cin>>T;
	while(T--){
		cin>>P>>x>>y>>u>>v;
		if(check(P,x,y,u,v))
			cout<<"Counter-Terrorists Win !"<<endl;
		else
			cout<<"Terrorists Win !"<<endl;
	}
	return 0;
}
