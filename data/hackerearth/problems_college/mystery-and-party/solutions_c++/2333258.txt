#include <iostream>
#include<algorithm>
#include<cstdio>
using namespace std;

int main()
{
	int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,k,i;

    	long long int a[100000];
    	for(i=0;i<n;i++)

    	printf("%lld\n",a[k-1]);
    }
   	
    return 0;
}
