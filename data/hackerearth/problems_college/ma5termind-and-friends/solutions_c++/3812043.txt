#include<bits/stdc++.h>
#define ll long long
#define mod 1e9+7
int visit[1000005];
ll ans[1000005];
using namespace std;
struct node
{
    int l;
	int r;
	int cost;
};
typedef struct node node;
node arr[1000005];
int main()
{
    //freopen("in.txt","r",stdin);
    int t;
    cin>>t;
    while(t--)
    {
    	int n,m;
        cin>>n>>m;
        for(int i=0;i<100004;i++)
         visit[i]=0;
        for(int i=0;i<100004;i++)
		 ans[i]=0;
 
        for(int i=1;i<=m;i++)
        {
            cin>>arr[i].l;
			cin>>arr[i].r;
			cin>>arr[i].cost;
	    }
		int q,l,r;
		cin>>q;
        while(q--)
        {
            cin>>l>>r;
            visit[l]+=1;
            visit[r+1]-=1;
        }
        for(int i=1;i<=m;i++)
        {
        	 visit[i]+=visit[i-1];
         }
         ll res=0;
        for(int i=1;i<=m;i++)
        {
            res=arr[i].cost*1LL*visit[i];
            ans[arr[i].l]+=res;
            ans[arr[i].r+1]-=res;
        }
        for(int i=1;i<=n;i++)
        {
            ans[i]+=ans[i-1];
         }
	    for(int i=1;i<=n;i++)
        {
             cout<<ans[i]<<" ";
        }
	   cout<<endl;  
    }
    return 0;
}
