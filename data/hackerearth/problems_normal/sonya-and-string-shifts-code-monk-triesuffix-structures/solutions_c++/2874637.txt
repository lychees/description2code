#include<bits/stdc++.h>
using namespace std;
int main(){

//Helpers for input and output

    int n,k,q;

    string s;
    cin>>n;
    cin>>s;
    cin>>q;
    int *arr=new int[n];
    int i,j,l,fe=0;
    
    arr[0]=0;
    for(i=1;i<n;i++){
        if(s[i]==s[fe]){
        	k=i+1;l=fe+1;
        	for(j=1;j<n;j++){
        		if(k==n)
        			k=0;
        		if(l==n)
        			l=0;
        		if(s[k]<s[l]){
        			arr[i]=i;
        			fe=i;
        			break;
        		}
        		else if(s[l]<s[k]){
        			arr[i]=fe;
        			break;
        		}
        		k++;l++;
        	
        	}
        }
        else if(s[i]<s[fe]){
        	arr[i]=i;
        	fe=i;
        }
        else{
        	arr[i]=fe;
        }
    }
    
    while(q--){
        cin>>k;
        if(arr[k]==0)
            cout<<0;
        else cout<<arr[k];
        cout<<endl;
    }
   return 0;
}
