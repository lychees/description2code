void si(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

int sieve[1000006];
int main(){
	int lim=1000006;
	for(int i=2;i<lim;i++){
		if(sieve[i]==0)
		for(int j=i;j<lim;j+=i){
			if(sieve[j]==0) sieve[j]=i;
		}
	}
	int n;
	si(n);
	while(n--){
		int r;
		si(r);
		printf("%d\n",r-sieve[r]);
	}
}
