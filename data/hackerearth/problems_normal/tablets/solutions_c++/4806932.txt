#include <iostream>
using namespace std;
int main()
{
    long long int n,sum=0,i;
    cin>>n;      
    long long int a[n];
    for(i=0;i<n;i++)
    {
        cin>>a[i];       
    }
    long long int t[n];   
    t[0]=1;   
    for(i=1;i<n;i++)      
    {
    if(a[i]>a[i-1])    
    {
        t[i]=t[i-1]+1;
    }
    else
    t[i]=1;     
    }
  for(i=n-2;i>=0;i--)
  {
    if(a[i]>a[i+1]&&t[i]<=t[i+1])
    {
    t[i]=t[i+1]+1;
    }

  }
  for(i=0;i<n;i++)
  sum+=t[i];    
  cout<<sum;   
}
