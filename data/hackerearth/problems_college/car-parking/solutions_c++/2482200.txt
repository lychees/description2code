#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int l=s.length();
    	for(int i=0;s[i]!='\0';i++)
    	{
    		if(s[i]=='S')
    		{
    			if(i-1>=0 && s[i-1]=='-')
    				s[i-1]='X';
    			if(i+1<l && s[i+1]=='-')
    				s[i+1]='X';
    		}
    		else if(s[i]=='B')
    		{
    			if(i-1>=0 && s[i-1]=='-')
    				s[i-1]='X';
    			if(i-2>=0 && s[i-2]=='-')
    				s[i-2]='X';
    		}
    	}
    	int ans=0;
    	for(int i=0;s[i]!='\0';i++)
    	{
    		if(s[i]=='-')
    			ans++;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
