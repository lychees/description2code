#include <iostream>
#include<climits>
#include<vector>
using namespace std;

int main()
{
   	long long N,M;
   	vector<long long> Ar,Br;
   	cin>>N;
   	long long mina=INT_MAX,minb=INT_MAX,d;
   	for(int i=0;i<N;i++)
   	{
   		cin>>d;
   		mina=min(mina,d);
   		Ar.push_back(d);
   	}
   	cin>>M;
   	for(int i=0;i<M;i++)
   	{
   		cin>>d;
   		minb=min(minb,d);
   		Br.push_back(d);
   	}
   	mina=min(mina,minb);
   	minb=mina;
   	vector<long long> Mm(105),Nn(105);
   	for(int i=0;i<N;i++)
   	{
   		if(Mm[Ar[i]-mina]==0)Mm[Ar[i]-mina]=1;
   		else Mm[Ar[i]-mina]++;
   	}
   	for(int i=0;i<M;i++)
   	{
   		if(Nn[Br[i]-minb]==0)Nn[Br[i]-minb]=1;
   		else Nn[Br[i]-minb]++;
   	}
   	for(int i=0;i<=103;i++)
   	{
   		if(Mm[i]<Nn[i])
   			cout<<mina+i<<" ";
   	}
    return 0;
}
