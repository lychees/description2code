import heapq

t = int(raw_input())

for i in range(t):
	n = int(raw_input())
	start = map(int, raw_input().strip().split(' '))
	duration = map(int, raw_input().strip().split(' '))

	sd = zip(start,duration)
	sd.sort(key=lambda t: t[0])
	rooms=[]
	heapq.heappush(rooms, sd[0][0]+sd[0][1])

	for i in range(1, n):
		finish = sd[i][0] + sd[i][1]
		if sd[i][0] < rooms[0]:
			heapq.heappush(rooms, finish)
		elif sd[i][0] >= rooms[0]:
			heapq.heapreplace(rooms, finish)

	print len(rooms)
