#include <iostream>
#include <stdlib.h>
using namespace std;
int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}
int main() {
	int t;
	long int n,count,temp;
	int ar[100000];
	cin>>t;
	while(t--)
	{
		count=1;
		temp=1;
		cin>>n;
		for(long int i=0;i<n;i++)
		cin>>ar[i];
		qsort (ar, n, sizeof(int), compare);
		for(long int i=0;i<n-1;i++)
		{
			
			if(ar[i]==ar[i+1])
			temp++;
			if(count<temp)
				count=temp;
			if(ar[i]!=ar[i+1])
				temp=1;
		}
		cout<<count<<endl;
	}
	return 0;
}
