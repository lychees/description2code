#include<bits/stdc++.h>
using namespace std;

int main() {

    int t;
    cin>>t;
    int cost,n,k;
    int arr[101];

    while(t--) {
        cin>>n>>k;
        for(int i=0;i<n;i++)
            cin>>arr[i];
        int ans=INT_MAX;
        for(int i=0;i<n;i++) {
            vector<int> v;
            for(int j=0;j<n;j++) {
                if(arr[i]>=arr[j]) {
                    cost=(arr[i]-arr[j])*3;
                }
                else {
                    cost=(arr[j]-arr[i])*5;
                }
                v.push_back(cost);
            }
            sort(v.begin(),v.end());
            cost=0;
            for(int j=0;j<k;j++)
                cost+=v[j];
            ans=min(ans,cost);
        }
        cout<<ans<<endl;
    }

    return 0;
}
