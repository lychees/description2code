#include <iostream>
using namespace std;

void merge(int low,int mid,int high,long *a)
{
	int i,j,k;
	long c[100000];
	i=low;
	j=mid+1;
	k=low;
	while(i<=mid && j<=high)
	{
		if(a[i]<a[j])
		{
			c[k]=a[j];
			k++;
			j++;
		}
		else
		{
			c[k]=a[i];
			k++;
			i++;
		}
	}
	while(i<=mid)
	{
		c[k]=a[i];
		k++;
		i++;
	}
	while(j<=high)
	{
		c[k]=a[j];
		k++;
		j++;
		
	}
	for(i=low;i<k;i++)a[i]=c[i];
	
}
void mergeSort(int low,int high,long *A)
{
	int mid;

	if(low<high)
	{	
	mid=(low+high)/2;
	mergeSort(low,mid,A);
	mergeSort(mid+1,high,A);
	merge(low,mid,high,A);
	}
}
int main()
{
    int i,N,T;
    long *A;
    cin>>T;
    while(T)
    {
    	cin>>N;
    	A=new long[N];
    	for(i=0;i<N;i++)cin>>A[i];
    	
    	mergeSort(0,N-1,A);
    	
    	
    	for(i=0;i<N;i++)cout<<A[i]<<" ";
    	cout<<endl;
    	
    	T--;
    	delete []A;
    }
}
