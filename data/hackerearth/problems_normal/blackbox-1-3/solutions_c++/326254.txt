#include <bits/stdc++.h>
using namespace std;

long long gcd(long long a, long long b)
{
	
	if ( a==0 ) return b;
	return gcd(b%a,a); 

}

int main()
{
    long long t,a,b,c,g1,g2,g3;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>c;
    	
    	g1 = gcd(a,b);
    	g2 = gcd(b,c);
    	g3 = gcd(a,c);
    	long long a1 = 0;
    	if(g1==1 && g2==1 && g3==1)
    	{
    		a1 = 1;
    	}
    	
    	if((a*a) + (b*b) == (c*c)  &&  a1 ==1)
    	cout<<"YES"<<endl;
    	
    	else if((b*b) + (c*c) == (a*a)  &&  a1 ==1)
    	cout<<"YES"<<endl;
    	
    	else if((a*a) + (c*c) == (b*b)  &&  a1 ==1)
    	cout<<"YES"<<endl;
    	
    	else
    	cout<<"NO"<<endl;
    	
    }
    return 0;
}
