#include<bits/stdc++.h>
using namespace std;
const int N = 1000005;
int start[N],en[N];
typedef long long ll;
ll arr[N];
int main()
{
	int n,m,l,r;
	scanf("%d%d",&n,&m);
	while(m--)
	{
		scanf("%d%d",&l,&r);
		start[l]++;
		en[r]++;
	}
	arr[1] = start[1];
	for(int i=2;i<=n;i++)
	{
		arr[i] = arr[i-1] + (start[i]-en[i-1]);
	}
	sort(arr+1,arr+n+1);
	int q;
	scanf("%d",&q);
	while(q--)
	{
		scanf("%d",&l);
		int pos = lower_bound(arr+1,arr+n+1,l)-arr;
		printf("%d\n",n-pos+1);
	}
	return 0;
}
