n=int(raw_input())
ppairs=[]
npairs=[]
for i in xrange(n):
	ti,xi=map(int,raw_input().split())
	if (xi-ti)>=0:
		ppairs.append([ti,xi])
	else:
		npairs.append([ti,xi])
 
ppairs.sort(reverse=True,key=lambda x: (x[1]+0.0)/x[0])
#print ppairs
invest=0
profit=0
for i in ppairs:
	if profit<i[0]:
		invest+=i[0]-profit
		profit=i[1]
	else:
		profit=profit-i[0]+i[1]
#print profit ,invest
npairs.sort(reverse=True,key=lambda x: ((x[1]),(-x[0])))
for i in npairs:
	if profit<i[0]:
		invest+=i[0]-profit
		profit=i[1]
	else:
		profit=profit-i[0]+i[1]
 
print invest
