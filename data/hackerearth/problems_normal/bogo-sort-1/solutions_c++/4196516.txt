#include <iostream>
#include<algorithm>
using namespace std;
typedef long long ll;

int sor(const void *a,const void *b){
	return *(int *)a - *(int *)b;
}

int main()
{
    ll n;
    cin>>n;
    ll A[n];
    for(int i = 0 ; i < n ; i++){
    	cin>>A[i];
    }
    qsort(A,n,sizeof(ll),sor);
    
    for(int i = 0 ; i < n ; i++){
    	cout<<A[i]<<" "<<endl;
    }
    return 0;
}
