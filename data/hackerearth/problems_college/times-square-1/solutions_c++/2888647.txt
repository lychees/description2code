#include <iostream>
using namespace std;

int main()
{
	int n,m,a;
	cin>>n>>m>>a;
	if(m>n)
	{
		int temp = m;
		m = n;
		n = temp;
	}
	int rows = 1,cols = 1;
	if(n>a)
	{
		rows = n/a;
		if(n%a != 0)rows++;
	}
	
	if(m>a)
	{
		cols = m/a;
		if(m%a != 0)cols++;
	}
    cout << (rows*cols) << endl;
    return 0;
}
