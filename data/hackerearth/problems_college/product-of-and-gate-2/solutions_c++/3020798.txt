#include <bits/stdc++.h>
using namespace std;

long long int a, b, diff;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	diff = b-a;
    	int nos = 0, zeros = 0;
    	while(b!=0)
    	{
    		b=b/2;
    		nos++;
    	}
    	
    	long long int i=1;
    	while(i<=diff)
    	{
    		i*=2;
    		zeros++;
    	}
    	zeros--;
    	
    	//cout<<nos<<" "<<zeros<<endl;
    	
    	long long int tpow=1, ans=0;
    	for(; a!=0; a/=2, tpow*=2)
    	{
    		if(zeros>=0)
    		{
    			zeros--;
    			continue;
    		}
    		//cout<<tpow<<" ";
    		if(((a%2)&1) != 0)
    			ans += tpow;
    	}
    	
    	cout<<ans<<endl;
    	
    }
    return 0;
}
