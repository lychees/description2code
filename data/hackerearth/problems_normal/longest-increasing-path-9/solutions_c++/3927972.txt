#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,m,i,j,s;
	cin >> t;
	while(t--){
		cin >> n >> m;
		int a,b;
		int v[n+1][m+1];
		for(i=0;i<=n;i++){
			for(j=0;j<=m;j++){
				v[i][j]=0;
			}
		}
		for(i=0;i<n;i++){
			for(j=0;j<m;j++){
				cin >> s;
				v[i][j]=s;
			}
		}
		int dp[n+1][m+1];
		for(i=0;i<n;i++){
			dp[i][m]=0;
		}
		for(i=0;i<m;i++){
			dp[n][i]=0;
		}
		dp[n-1][m-1]=1;
		for(i=n-1;i>=0;i--){
			for(j=m-1;j>=0;j--){
				if(v[i][j]< v[i][j+1]) a=dp[i][j+1] +1;
				else a=1;
				if(v[i][j] < v[i+1][j]) b= dp[i+1][j] +1;
				else b=1;
				dp[i][j]= max(a,b);
			}
		}
		cout << dp[0][0] << endl;
	}
    return 0;
}
