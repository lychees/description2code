first_input = map(int, raw_input().split())
N, Q = first_input[0], first_input[1]
given_array = map(int, raw_input().split())


def update_array(input_array, given_array):
    if input_array[1] < len(given_array):
        given_array[input_array[1] - 1] = input_array[2]


def find_grater_elem(operation_input, given_array):
    for i, item in enumerate(given_array):
        ind = None
        if item >= operation_input[1]:
            ind = i + 1
            break
    if ind:
        print ind
    else:
        print -1


for _ in xrange(Q):
    operation_input = map(int, raw_input().split())
    if len(operation_input) > 2:
        update_array(operation_input, given_array)
    else:
        find_grater_elem(operation_input, given_array)
