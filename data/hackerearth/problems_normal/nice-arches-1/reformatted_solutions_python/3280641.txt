n = int(raw_input())
b = 0
i = 0
while i < n:
    s = raw_input()
    i += 1
    l = len(s)
    j = 0
    bu = []
    while j < l:
        if bu:
            if bu[-1] == s[j]:
                bu.pop()
            else:
                bu.append(s[j])
        else:
            bu.append(s[j])
        j += 1

    if not bu:
        b += 1
print(b)
