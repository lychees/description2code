#include <iostream>
#include <set>

// https://www.hackerearth.com/problem/algorithm/tripping-on-nothingness/description/

using namespace std;

int s;
set<int> ans;

int gcd(int a, int b) {
	if (b == 0) 
		return a;
	else 
		return gcd(b, a % b);
}

void run_test() {
	int v; 
    cin >> v;
    int g = gcd(s, v);
    if (ans.find(g) == ans.end()) 
		ans.insert(g);	
    else
    	g = -1;
    	
	cout << g << '\n';
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    int T;
    cin >> s >> T;
    for (int i = 0; i < T; ++i)
        run_test();
    return 0;
}
