#include <iostream>
#include<stdio.h>
using namespace std;

void count(long int r,long int c,long int &even,long int &odd)
{
	odd=0;
	even=0;
	long int k1,k2,k;
	if(r<=c)
	{
		k1=(r+1)/2;
		k2=r/2;
		odd=k1*k1;
		even=k2*(k2+1);
	}
	else
	{
		k1=(c+1)/2;
		k2=c/2;
		k=(r-c);
		odd=k1*k1;
		even=k2*(k2+1);
		if(k%2==0)
		{
			odd+=(k/2)*c;
			even+=(k/2)*c;
		}
		else
		{
			if(c%2==0)
			{
				odd+=(((k/2)+1)*c);
				even+=(k/2)*c;
			}
			else
			{
				odd+=(k/2)*c;
				even+=(((k/2))+1)*c;
			}
		}
	}
}

int main()
{
    long int t,r,c,even,odd;
    scanf("%ld",&t);
    while(t--)
    {
    	scanf("%ld",&r);
    	scanf("%ld",&c);
    	count(r,c,even,odd);
    	printf("%ld %ld\n",even,odd);
    }
    return 0;
}
