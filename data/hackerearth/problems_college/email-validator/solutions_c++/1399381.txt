#include <bits/stdc++.h>
#define Max(a, b) (a > b) ? a : b
#define Min(a, b) (a < b) ? a : b
#define gcd __gcd
#define FOR(i, a, n) for (i = a; i < n; ++i)
#define FORe(i, a, n) for (i = a; i <= n; ++i)
#define FORi(i, a, n) for (int i = a; i < n; ++i)
#define FORie(i, a, n) for (int i = a; i <= n; ++i)		
#define FastIO ios_base::sync_with_stdio(false); cin.tie(0);
#define endl '\n'
using namespace std;
typedef long long ll;
typedef unsigned long long ull;

bool isAlphaNumeral(char A) {
	return (A >= 'a' && A <= 'z') || (A >= 'A' && A <= 'Z') || (A >= '0' && A <= '9');
}

int main() {
	FastIO
	ll N;
	cin>>N;
	while (N--) {
		char A[100];
		cin>>A;
		int n = strlen(A), i = 0;
		bool flag = true;
		while (i < min(16, n) && flag) {
			if (!isAlphaNumeral(A[i]) && !(A[i] == '_')) {
				if (A[i] == '@' && i != 0) break;
				flag = false;
			} else ++i;
		}
		if (!flag || A[i] != '@' || (n - i > 28)) {
			cout<<"NO\n";
			continue;
		} else ++i;
		int start = i, l = min(i + 16, n);
		bool alpha = false;
		while (i < l && flag) {
			if ((A[i] >= 'a' && A[i] <= 'z') || (A[i] >= 'A' && A[i] <= 'Z')) alpha = true;
			if (!isAlphaNumeral(A[i]) && !(A[i] == '-')) {
				if (A[i] == '.' && (i != start) && alpha) break;
				flag = false;
			} else ++i;
		}
		if (!flag || A[i] != '.' || (n - i > 11) || (i == n - 1)) {
			cout<<"NO\n";
			continue;
		} else ++i;
		while (i < n && flag) {
			if (!(A[i] >= 'a' && A[i] <= 'z')) flag = false;
			++i;
		}
		if (!flag) cout<<"NO\n";
		else cout<<"YES\n";
	}
	return 0;
}
