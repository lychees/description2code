int main()
{
	long int N ,i ,j,*count,max;
	long long int *NO;
	int T, l;
	cin>>T;
	for(l=0; l<T; l++)
	{
		cin>>N;
	
		NO = new long long int[N];
		count = new long int[N];
	
		for(i=0;i<N;i++)
			cin >> NO[i];
	
		for(i=0;i<N;i++)
			count[i] = countbit( NO[i] );
		
		max = Max(count,N);
	
		for(i=0; i<=max; i++)
			for(j=0; j<N; j++)
				if(i == count[j])
					cout << NO[j] << " ";
		cout << "\n";
				
		delete []NO;
		delete []count;
	}
	return 0;
}

long int countbit(long long int n)
{
	long int count=0;
    while(n)
    {
        n = n & (n-1);
        count++;
    }
    return count;
}

long int Max(long int a[],long int s)
{
	long int max = 0;
	for(long int i=0; i<s ;i++)
		if(max < a[i])
			max=a[i];
	return max;
}
