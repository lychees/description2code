#include <iostream>
#include <stdio.h>
using namespace std;

#define MOD 1000000007

int main()
{
    //cout << "Hello World!" << endl;
    long long int n;
	scanf("%lld", &n);
	
	if(n == 0)
		printf("10\n");
	else if( n  == 1)
    	printf("0\n");
    else{
	    int j = 0;
	    int count[10];
	    
	    for(int i = 0 ; i < 10 ; i++ )
	    	count[i] = 0;
	    	
	 	while(true)
	 	{
	    	long long int temp = MOD * j + n;
	    	for(int i = 9 ; i >= 2 ; i--)
	    	{
	    		while(temp % i == 0)
	    		{
	    			temp = temp / i;
	    			count[i]++;
	    		}
	    	}
	    	
	    	if(temp == 1)
	    	{
	    		break;
	    	}else
	    	{
	    		for(int i = 0 ; i < 10 ; i++ )
	    			count[i] = 0;
	    		j++;
	    	}
	 	}
	 	
	 	for(int i = 2 ; i < 10 ; i++ )
	 	{
	 		while(count[i] != 0)
	 		{
	 			printf("%d", i);
	 			count[i]--;
	 		}
	 	}
    }
    return 0;
}
