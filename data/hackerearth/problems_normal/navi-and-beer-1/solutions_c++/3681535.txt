#include <iostream>
using namespace std;

int main()
{
    
    long int N,K,i;
    long int T, M, sum, Ans;
    
    //taking input for number of test cases, T.
    cin>>T;
    while(T)
    {
    //For every test case, taking input for N , K and M.
      sum=0;
      cin>>N>>K>>M;
      for(i=1;i<=N;i++)
      {
      	sum = sum + (K*i*i*i);
      }
      if ((sum - M) > 0)
      {
      	Ans=sum-M;
      }
      else
      {
      	Ans=0;
      }
      //let the answer be in variable **Ans**
      cout << Ans << endl; //print your answer for every test case.
      T--;
    }

    
    return 0;
}
