t = int(raw_input())
for _ in range(t):
	n = map(str,raw_input().split(" "))
	le = len(n)
	k={}
	ma=0
	le-=1
	while le >=0 :
		k[ma]=n[le]
		ma+=1
		le-=1
	st=""
	for i in range(ma):
		st += k[i]
		st+=' '
		ma-=1
	print st
