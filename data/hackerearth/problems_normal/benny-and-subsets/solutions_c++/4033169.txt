#include <bits/stdc++.h>
#define MAX (1 << 22)
using namespace std;
const int mod=1e7+7;
typedef long long ll;
ll big[MAX];
vector<int> a,b;
int dp[1002][1 << 12];
int main() {
	int n,x,i,j,num;
	cin >>n >>x;
	
	for(i=0;i<n;i++){
	    cin >>num;
	    if(num>(1<<10))
	        b.push_back(num);
	    else
	        a.push_back(num);
	}
	int size_a=(int)a.size();
	int size_b=(int)b.size();
	int val;
	for(i=0;i<(1<<size_b);i++){
	    val=0;
	    for(j=0;j<size_b;j++){
	        if(i & (1<<j))
	            val^=b[j];
	    }
	    big[val]++;
	}
	
	dp[0][0]=1;
	for(i=1;i<=size_a;i++){
	    for(j=0;j<(1<<12);j++){
	        dp[i][j]=((dp[i-1][j])%mod+(dp[i-1][j^a[i-1]])%mod)%mod;
	    }
	}
	int sum=0;
	for(i=0;i<(1<<12);i++){
	    sum=(sum+(dp[size_a][i]*big[i^x])%mod)%mod;
	}
	cout <<sum;
	return 0;
}
