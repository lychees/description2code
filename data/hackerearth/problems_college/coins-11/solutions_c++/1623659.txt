#include <bits/stdc++.h>
using namespace std;

bool arr[100001];
void sieve()
{
	int i,j;
	arr[1]=1;
	for(i=2;i<100001;i++)
	{
		if(arr[i]==0)
		{
			for(j=i+i;j<=100001;j=j+i)
			arr[j]=1;
		}
	}
}

int main()
{
	sieve();
	
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	if(n==3)
    	cout<<"3"<<endl;
    	else if(arr[n])
    	cout<<"1"<<endl;
    	else
    	cout<<"2"<<endl;
    }
    
    
}
