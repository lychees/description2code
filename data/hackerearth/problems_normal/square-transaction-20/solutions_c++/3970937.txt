#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	int a[t];
	int b[t];
	int sum=0;
	int i;
	for(i=0;i<t;i++)
	{
		cin>>a[i];
		sum+=a[i];
		b[i]=sum;
		//cout<<b[i]<<" ";
	}
	//cout<<endl;
	int Q;
	cin>>Q;
	int q;
	while(Q--)
	{
		cin>>q;
		if(q>sum)
		{
			cout<<"-1"<<endl;
			continue;
		}
		for(int i=0;i<t;i++)
		{
			if(b[i]>=q)
			{
				//cout<<b[i]<<" "<<q<<endl;
				cout<<i+1<<endl;
				break;
			}
		}
	}
    //cout<<count;
    return 0;
}
