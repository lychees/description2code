#include<stdio.h>
#include<iostream>
#include<map>

using namespace std;

int main()
{
	int test;
	scanf("%d",&test);
	
	while(test>0)
	{
		int i;
		char str[35]; 
		scanf("%s",str);
		map<char,int> m; 
		m.clear();
		for(i=0;str[i];i++)
		{
			if(!m[str[i]])
				m[str[i]]=1;
		}
		
		for(i=0;str[i];i++)
		{
			if(m[str[i]])
			{
				printf("%c",str[i]);
				m[str[i]]=0;
			}
		}
		printf("\n");
		test--;
	}
}
