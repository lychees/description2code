T = input()

for t in xrange(T):
    a, b = map(int, raw_input().split())
    if a < b:
        print a
        # break
    else:
        V = []
        # C = a
        while a >= b:
            V.append(a % b)
            a = a/b
        V.append(a)
        V.reverse()
        S = map(str, V)
        print "".join(S)
