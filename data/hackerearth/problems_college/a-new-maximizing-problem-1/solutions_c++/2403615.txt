#include <iostream>
#include <algorithm>
#define d 1000000007;
using namespace std;
void swap(double* a,double* b)
{
	double temp = *a;
	*a = *b;
	*b = temp;
	
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		double x[n];
		for(int i=0;i<n;i++) //take the input
		{
			cin>>x[i];
		}
		sort(x,x+n); //sort the array the sort every even index in the resultant array is smallest possible
		double a[n];
		int k=0;
		/*converting the given array to array so that even idices have smaller values of array and odd indices
		have larger values to compute maximum A */
		if(n%2==0)
		{
			for(int i=0;i<n/2;i++)
			{
				a[k++]=x[i];
				a[k++]=x[i+n/2];
			}
			
		}
		else
		{
			for(int i=0;i<n/2;i++)
			{
				a[k++]=x[i];
				a[k++]=x[i+(n/2)+1];
			}
			a[k]=x[(n/2)];
		}
    	long long asum=0;
    	long long bsum=0;
    	for(int i=0;i<n;i++)
    	{
    	    asum+= a[i]*((i%2==0)?-1:1); //if odd -1 even 1
    		bsum+= a[i]*((a[i]>0)?1:-1); //if -negative multiply with -1 else 1
    		asum=asum%d;
    		bsum=bsum%d;
    	}
    	//cout<<asum<<" "<<bsum<<endl; //to check individual sums
    	long long csum = (asum*bsum)%d; // result
    	cout<<csum<<endl;
    	
    }
    return 0;
}
