#include<bits/stdc++.h>
using namespace std;
int main()
{
vector<int> v;
int tat;
int tot;
cin>>tat;
int n,d;
while(tat--)
{tot=0;
cin>>n;
int t=n;
while(n--)
{
scanf("%d",&d);
v.push_back(d);
}
sort(v.begin(),v.end());
for(int i=0;i<t;)
{int f=0;
while(v[i]==v[i+1]){
f=1;
i++;
}//if(f)tot++;
if(!f){tot++;i++;}
}
cout<<tot<<endl;
v.resize(0);
}
return 0;
}
