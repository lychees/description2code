#include<bits/stdc++.h>
using namespace std;
int main()
{
	int test,hex[2] = {0},q,j,i,y,t,u,m;
	int a[3] = {0};
	cin >> test;
	for(u = 0;u < test;u++)
	{
		for(i = 0;i < 3;i++)
		{
			cin >> a[i];
		}
		cout << "Case "<< u+1 << ":" << " ";
		for(j = 0;j < 3;j++)
		{
			y = a[j];
			t = 1;
			for(m = 0;m < 2;m++)
			{
				hex[m] = 0;
			}
			while(y != 0)
			{
				
				hex[t] = y % 16;
				t--;
				y = y / 16;
			}
			for(q = 0;q < 2;q++)
			{
				if(hex[q] == 10)
				{
					cout << "U";
				}
				else if(hex[q] == 11)
				{
					cout << "V";
				}
				else if(hex[q] == 12)
				{
					cout << "W";
				}
				else if(hex[q] == 13)
				{
					cout << "X";
				}
				else if(hex[q] == 14)
				{
					cout << "Y";
				}
				else if(hex[q] == 15)
				{
					cout << "Z";
				}
				else
				{
					cout << hex[q];
				}
			}
			cout << " ";
		}
		cout << endl;
	}
	return 0;
}
