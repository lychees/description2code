#include<bits/stdc++.h>
using namespace std;
struct stu{
int a;
int b;
};
bool compare(struct stu p,struct stu q)
{
if(p.a<q.a)
{
return 1;
}
else
if(p.a==q.a && p.b>q.b)
return 1;
else
return 0;
}
int main()
{
int n,i;

{

}
