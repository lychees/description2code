#include<bits/stdc++.h>
#define s(a) scanf("%d",&a)
#define sll(a) scanf("%lld",&a)
#define ss(a) scanf("%s",a)
using namespace std;
typedef long long int ll;
ll n,ans1[20],ans2[20];
int prime[4]={2,3,5,7};
inline bool contains(ll num,int dig)
{
    while(num)
    {
        if(num%10 == dig)return true;
        num = num/10;
    }
    return false;
}
void precomp()
{
      int i,j;
      ll res1,res2;
      ans1[0] = 1;
      ans2[0] = 0;
      for(i=1;i<20;i++)
      {
            res1 = res2 = 1;
            for(j=0;j<i;j++)
                  res1 = res1*10, res2 = res2*6;
            ans1[i] = res1;
            ans2[i] = res1-res2;
      }
}
inline ll solve(ll num)
{
      bool primeOccured = false;
      int len=0,i,j;
      int dig[20];
      ll temp = num;
      ll res = 0 ;
      while(temp)
      {
            dig[len]=temp%10;
            len++;
            temp = temp/10;
      }
      for(i=len-1;i>=0;i--)
      {
            for(j=0;j<dig[i];j++)
            {
                  if(primeOccured || (j==2 || j==3 || j==5 || j==7 ))
                        res += ans1[i];
                  else res += ans2[i];
            }
            if(dig[i] == 2 || dig[i] == 3 || dig[i] == 5 || dig[i] == 7)
                  primeOccured = true;
      }
      for(i=0;i<4;i++)
            if(contains(num,prime[i]))res++, i=4;
      return res;
}
int main()
{
      int t;
      precomp();
      s(t);
      while(t--)
      {
            sll(n);
            printf("%lld\n",solve(n));
      }
      return 0;
}
