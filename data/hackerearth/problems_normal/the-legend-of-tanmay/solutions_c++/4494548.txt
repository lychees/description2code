#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i;
		cin>>n;
		int arr[20];
		for(int i=0;i<n;i++)
		{
			cin>>arr[i];
		}
		sort(arr,arr+n);
		int neg=0,pos=0,zr=0;
		for(i=0;i<n;i++)
		{
			if(arr[i]<0)
				neg++;
			else if(arr[i]>0)
				pos++;
			else
				zr++;
		}
		long long maxAns=1,minAns=1;
		if(neg%2)
		{
			if(neg==1 && pos==0)
			{
				if(zr>0)
					maxAns=0;
				else
					maxAns=arr[0];
			}
			else
			{
				for(i=0;i<neg-1;i++)
				{
					maxAns=arr[i]*maxAns;
				}
				for(;i<n;i++)
				{
					if(arr[i]>0)
					{
						maxAns=arr[i]*maxAns;
					}
				}	
			}
			
			for(i=0;i<n;i++)
			{
				if(arr[i]!=0)
				{
					minAns=arr[i]*minAns;
				}
			}
		}
		else
		{
			if(neg==0 && zr==0)
			{
				minAns=arr[0];
				for(i=0;i<n;i++)
				{
					maxAns=arr[i]*maxAns;
				}
			}
			else if(neg==0 && zr>0)
			{
				minAns=0;
				if(pos==0)
					maxAns=0;
				else
				{
					for(i=0;i<n;i++)
					{
						if(arr[i]!=0)
						maxAns=arr[i]*maxAns;
					}	
				}
			}
			else
			{
				if(pos==0)
				{
					for(i=0;i<neg-1;i++)
					{
						minAns=arr[i]*minAns;
					}
					for(i=0;i<neg;i++)
					{
						maxAns=arr[i]*maxAns;
					}
				}
				else
				{
					for(i=0;i<n;i++)
					{
						if(arr[i]!=0)
						maxAns=arr[i]*maxAns;
					}
					for(i=0;i<neg-1;i++)
					{
						minAns=arr[i]*minAns;
					}
					for(;i<n;i++)
					{
						if(arr[i]>0)
							minAns=arr[i]*minAns;
					}
				}
			}
			
		}
		cout<<maxAns<<" "<<minAns<<endl;
	}
	
	return(0);
}
