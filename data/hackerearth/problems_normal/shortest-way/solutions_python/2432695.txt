t=int(raw_input())
for _ in xrange(t):
    a,b=map(int,raw_input().split())
    ans=0
    while True:
        if a==1 and b==1:
            break
        if a>b:
            ans+=int((a-1)/b)
            a=a-(int((a-1)/b)*b)
        else:
            ans+=int((b-1)/a)
            b=b-(int((b-1)/a)*a)
    print(ans)
