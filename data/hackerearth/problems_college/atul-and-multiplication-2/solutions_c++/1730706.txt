#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
int main()
{
    long int n,i;
    cin>>n;
    long long int p,q,t;
    for(i=0;i<n;i++)
    {
        cin>>p>>q>>t;
        if((p*q)==t)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    return 0;
}
