#include<iostream>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        int a[n],ans=0;
        for(int i=0;i<n;i++){
            cin>>a[i];
        }
        for(int i=0;i<n;i++){
                int count=0;
            for(int j=i+1;j<n;j++){
                if(a[i]>a[j])
                    count++;
                else
                	break;
                
            }
            if(count==n-i-1)
                ans++;
        }
        cout<<ans<<endl;
    }
return 0;
}
