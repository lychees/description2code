#include <bits/stdc++.h>
using namespace std;

int main(){
	int n;
	cin>>n;
	int a[n];
	map<int, vector< pair<int, int> > > sums;

	for (int i = 0; i < n; ++i){
		cin>>a[i];
	}

	for(int i=0; i<n; i++){
		for(int j=i; j<n; j++){
			int sum=0;
			for(int k=i; k<=j; k++){
				sum+=a[k];
			}
			sums[sum].push_back(make_pair(i, j));
		}
	}
	vector<int>ans;
	int maxi=-1;
	map<int, vector< pair<int, int> > >::iterator it;
	for (it = sums.begin(); it != sums.end(); it++){
		vector< pair<int, int> > sub_array = it->second;
		int count=0;
		int prev = -1;
		for(int i=0; i<sub_array.size(); i++){
			if(sub_array[i].first>prev){
				count++;
				prev = sub_array[i].second;
			}
		}
		if(count>maxi){
			ans.clear();
			maxi=count;
		}
		if(count==maxi){
			ans.push_back(it->first);
		}
	}
	cout<<maxi<<endl;
	for(int i=0; i<ans.size(); i++){
		cout<<ans[i]<<" ";
	}
}
