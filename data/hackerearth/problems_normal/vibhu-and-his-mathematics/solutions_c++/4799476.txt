#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;
long long t,n,dp[1000010];
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	dp[0]=0;
	dp[1]=1;
	dp[2]=2;
	dp[3]=4;
	for(int i=4;i<=1000000;i++)
	{
		dp[i]=(dp[i-1]%MOD+((i-1)*dp[i-2])%MOD)%MOD;
	}
	cin>>t;
	for(int i=0;i<t;i++)
	{
		cin>>n;
		cout<<dp[n]<<endl;
	}
	return 0;
}
