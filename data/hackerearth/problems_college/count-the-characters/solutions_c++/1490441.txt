#include <iostream>
using namespace std;

int main()
{
	char str[100];
	int n,i,count=0,c,tmp;
	int b[26]={0};
	
    scanf("%d",&n);
    tmp=n;
    while(n--)
    {
    	int a[26]={0};
    	scanf("%s",str);
    	for(i=0 ; str[i]!='\0' ; i++)
    	{
    		c=str[i]-'a';
    		a[c]++;
    	}
    	for(i=0 ; i<26 ; i++)
    	{
    		if(a[i]>0)
    		b[i]++;
    	}
    }
    for(i=0 ; i<26 ; i++)
    {
    	if(b[i]==tmp)
    	count++;
    }
    printf("%d",count);
    return 0;
}
