#include <bits/stdc++.h>
using namespace std;

int sol(int start,string s1,string s2)
{
        if(start==s1.length()-1)
        {
                return abs(s1[start]-s2[start]);
        }
        else
        {
            int mi=abs(s1[start]-s2[start])+sol(start+1,s1,s2);
            
            for(int i=start+1;i<s1.length();i++)
            {
                string temp1=s1;
                string temp2=s2;

                swap(temp1[start],temp1[i]);
                //cout<<<<endl;
               
                mi=min(mi,1+abs(temp1[start]-s2[start])+sol(start+1,temp1,s2));
            }
            return mi;
        }


}

{

        string s1,s2;
        cin>>s1>>s2;
        cout<<sol(0,s1,s2)<<endl;

    }
}
