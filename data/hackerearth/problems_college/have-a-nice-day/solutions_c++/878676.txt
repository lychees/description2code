#include <iostream>
#include<stdio.h>
#include <algorithm>
using namespace std;


long int function1(unsigned int N)
{
   long int sum=0;unsigned int i,j;
   static int a[1000],b[1000];
   for(i=0;i<N;i++)
   {
       cin>>a[i];
   }
   for(j=0;j<N;j++)
   {
       cin>>b[j];
   }
   sort(a,a+N);
   sort(b,b+N);
   for(i=0,j=0;i<N && j<N;i++)
   {
       sum+=a[i]*b[j];
       j++;
   }
   return(sum);
}
int main()
{
    unsigned int t,N;
    long int count1;
    cin>>t;
    if(t<1 || t>1500)
    {
        exit(0);
    }
    else {
            while(t--)
            {
                cin>>N;
                count1=function1(N);
                cout<<count1<<endl;
            }
}
return(0);
}
