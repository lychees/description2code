#include <iostream>
#include<stdlib.h>

using namespace std;

int main()
{
    int m,i,j;
    cin>>m;
    int mat[m][m],sum1=0,sum2=0;
    for( i=0;i<m;i++)
    	for( j=0;j<m;j++)
		 cin>>mat[i][j];
    for(i=0;i<m;++i) sum1+=mat[i][i];
    for(i=m-1,j=0;i>=0;i--,j++)
    {
    	if(j==m)break;
    	sum2+=mat[i][j];
    }
    cout<<endl<<abs(sum1-sum2);
    
    return 0;
}
