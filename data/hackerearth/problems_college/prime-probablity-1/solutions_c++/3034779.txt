#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
	long long notc,i,j,k,temp,l,r,nop;
	int arrsum[1000001];
	bool arr[1000001];
	for(i=0;i<1000001;i++)
	    arr[i]=true;
	arr[1]=false;
	for(i=2;i<1000001;i++)
	{
		if(arr[i])
		{
			for(j=i;i*j<1000001;j++)
			    arr[i*j]=false;
		}
	}
	arrsum[0]=arrsum[1]=0;
	for(i=2;i<1000001;i++)
	    if(arr[i])
	        arrsum[i]=arrsum[i-1]+1;
	    else
	        arrsum[i]=arrsum[i-1];
	cin>>notc;
	while(notc--)
	{
		cin>>l>>r;
		cout<<fixed<<setprecision(6)<<((float)(arrsum[r]-arrsum[l-1])/(r-l+1))<<"\n";
	}
    return 0;
}
