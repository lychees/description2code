#include <bits/stdc++.h>

using namespace std;
int a[10005];
int dp[10005];

int rec(int i, int n)
{
	if (i >= n) {
		return 0;
	}
	
	if (dp[i] > -1) {
		return dp[i];
	}
	
	dp[i] = max(a[i] + rec(i+2, n), rec(i+1,n));
	
	return dp[i];
}

int main()
{
	int t, n;
	cin >> t;
	while (t--) {
		memset(dp, -1, sizeof(dp));
		cin >> n;
		for (int i = 0; i < n; i++) {
			cin >> a[i];
		}
		cout << rec(0, n) << endl;
	}
	
	return 0;
}
