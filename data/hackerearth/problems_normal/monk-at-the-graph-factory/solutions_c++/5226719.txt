#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n],sum=0;
    
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    	sum+=a[i];
    }
    
    if(sum%2==1)
    {
    	cout<<"No"<<endl;
    	return 0;
    }
    sum/=2;
    int v=sum+1;
    if(v==n) cout<<"Yes"<<endl;
    else cout<<"No"<<endl;
    return 0;
}
