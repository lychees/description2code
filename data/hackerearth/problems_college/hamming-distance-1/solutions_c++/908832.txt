string tostring(int n)
{
    string s;
    while (n) {
        s.push_back(n % 10 + '0');
        n /= 10;
    }
    reverse(s.begin(), s.end());
    return s;
}

        int cnt = 0;
        for (int i = 0; i <= 30; i++) {
