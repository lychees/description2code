#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int n,m,i,j,x,y,q,p;
   cin>>n>>m;
   int arr[n][m];
   for(i=0;i<n;i++)
   	for(j=0;j<m;j++)
   		cin>>arr[i][j];
   cin>>q;
   while(q--){
   	cin>>p;
   	x=y=-1;
   	int flag=0;
   	for(i=0;i<n;i++){
   		if(arr[i][0]<=p&&arr[i][m-1]>=p);
   		else
   			continue;
   		for(j=0;j<m;j++){
   			if(arr[i][j]==p){
   				x=i;
   				y=j;
   				flag=1;
   				break;
   			}
   		}
   		if(flag)	
   			break;
   	}
   	cout<<x<<" "<<y<<endl;
   }
    return 0;
}
