#include<stdio.h>
int main()
{
    int t,arr[110],c,l,i;
    scanf("%d\n",&t);
    while(t--)
    {
        l=0;
        
        while((c=getchar())!='\n'&&c!=EOF)
        {
            arr[l++]=c;
        }
        for(i=0;i<l/2;i++)
        {
            if(arr[i]!=arr[l-1-i])
            break;
        }
        if(i>=l/2)
        printf("YES\n");
        else
        printf("NO\n");
    }
    return 0;
}
