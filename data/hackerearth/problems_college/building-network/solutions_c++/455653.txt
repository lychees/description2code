#include<stdio.h>
#include<limits.h>
long long int min=INT_MAX;
main()
{
	long long int i,j,t,n,a[1005],rem,sum=0;
	
	scanf("%lld",&t);
	while(t--)
	{
		sum=0;
		scanf("%lld",&n);
		for(i=0;i<n;i++)
		{
		scanf("%lld",&a[i]);
		if(a[i]<min)
		{
		min=a[i];
		rem=i;
		}
		}
		
		for(i=0;i<n;i++)
		{
			if(rem==i)
			continue;
			else
			sum+=a[i]+a[rem];
		}
		
		
		printf("%lld\n",sum);
		
	}
	return 0;
}
