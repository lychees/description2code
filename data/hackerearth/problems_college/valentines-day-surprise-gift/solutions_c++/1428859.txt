#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
long long p(long long n,long long m)
{
    if(m==0) return 1%MOD;
    long long x=p(n,m/2);
    if(m%2==0)
        return (x*x)%MOD;
    else
        return (((x*x)%MOD)*n)%MOD;
}
long long mulmod(long long a,long long b,long long c)
{    
	long long x = 0,y=a%c;    
	while(b > 0)
	{
	    if(b%2 == 1)
		{
			x = (x+y)%c;
		}        
		y = (y*2)%c;
		b /= 2;
	}
	return x%c;
}
long long R(long long n)
{
    if(n==0 || n==1 || n==2)
        return 1;
    if(n<=10){
        if(n<=4)	return 3;
        else if(n<=6)	return 15;
        else	return 105;
    }
    return mulmod(mulmod(p(2,(n+3)/4),R((n+4)/5),MOD),R((n+9)/10),MOD);
}
int main(){
    int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        long long s=R(n);
        long long ans=0;
        for(int i=1;i<=5;i++){
         	for(int j=1;j<=5;j++){
         	  	if(__gcd(i,j)!=1)continue;
         	  	ans=max(p(s*i,s*j),ans);
         	}
        }
        for(int i=1;i<=5;i++){
         	for(int j=1;j<=5;j++){
         	  	if(__gcd(j,i)!=1)continue;
         	  	ans=max(p(s*j,s*i),ans);
         	  }
         }  
        cout<<ans<<"\n";
    }
}
