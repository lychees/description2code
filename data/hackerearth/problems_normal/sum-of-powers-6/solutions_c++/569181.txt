#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
int t,x,i,j,f,l,r;
vector<int> v;

for(i=2;i<=1010;i++)
{
j=i*i;
while(j<=1000010)
{
v.push_back(j);
j*=i;
}
}

//v.push_back(0);
v.push_back(1);

sort(v.begin(),v.end());

scanf("%d",&t);

while(t--)
{
scanf("%d",&x);

l=0;
r=v.size()-1;
f=0;

while(l<=r)
{
if(v[l]+v[r]>x)
r--;
else
if(v[l]+v[r]<x)
l++;
else
{
f=1;
break;
}
}

if(f==1)
printf("Yes\n");
else
printf("No\n");
}

return 0;
}
