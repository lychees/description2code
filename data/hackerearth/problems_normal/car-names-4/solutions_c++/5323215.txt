#include<iostream>
using namespace std;
bool a( string s, int l )
{
    for(int i=1; i<l; i++)
    {
        if(s[i]==s[i-1]) continue;
        else return false;
    }return true;
}
int main()
{
    int n;
    cin>>n;
    for(int j=0; j<n; j++)
    {
        string s; cin>>s;
    int l=s.length(); if(l==3) {cout<<"OK"<<endl; continue;}
    else if(l%3==0)
      { string s1=s.substr(0,l/3); 
        string s2=s.substr(l/3, l/3);
        string s3=s.substr(2*l/3, l/3);
        if((a(s1,l/3 ) && a(s2, l/3)) && a(s3,l/3))
        {if(((s1[0]!=s2[0]) && (s2[0]!=s3[0])) && (s3[0]!=s1[0]))
        {cout<<"OK"<<endl;}
          else cout<<"Not OK"<<endl;  
        }
        else cout<<"Not OK"<<endl;
      }
      else cout<<"Not OK"<<endl;
      
    }
}
