ll n, x, y;


ll fnc(int id, int x, int y)
{
    if(id == n + 1)
        return 0;

    ll ret =  fnc(id + 1, y, abs(x - y)) + x;

    //ret = min(ret, fnc(id + 1, y, x + y) + x);

   // cout << id <<" "<<x <<" "<<y <<" "<<ret << endl;

    return ret;
}

int main()
{
    int t;
    cin >> t;
    while(t--) {
        cin >> x >> y >> n;

        ll a = x;
        ll b = y;
        ll ch = __gcd(x, y);
        ll pr = x + y;
        n -= 2;

        while(a != ch || b != ch) {
            if(!n)
                break;
        	ll tmp = abs(b - a);
        	a = b;
        	b = tmp;
        	n--;
            pr += tmp;
        }
        //cout << pr << endl;
        if(n) {
            n += 2;
            pr-= (2 * ch);
           
            ll d1, d2;
            d1 = n / 3;
            d2 = n % 3;
           //  cout << pr <<" "<<n <<" "<<d1 <<" "<<d2<< endl;

            pr += (d1 * 2* ch);
            if(d2 == 1)
                pr += ch;;
            if(d2 == 2)
                pr += 2*ch;
        }
        cout << pr << endl;
        continue;

        cout << a <<" "<<b<<endl;
        if(n > 1000) {
        	cout << "0 "<<endl;
        	continue;
        }

       // memset(dp, -1, sizeof(dp));
        cout << fnc(1, x, y) << endl;
    }
    return 0;
}
