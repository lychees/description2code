#include <iostream>
using namespace std;
int max(int a,int b)
{
	return a>b?a:b;
}
int main()
{
  int t;
  cin>>t;
  int n,i,j;
  while(t--)
  {
  	  cin>>n;
  	  int a[n];
  	  long dp[n];
  	  for(i=0;i<n;i++)
  	    cin>>a[i];
  	  
  	  dp[0]=a[0];
  	  dp[1]=max(a[0],a[1]);
  	  for(i=2;i<n;i++)
  	      dp[i]=max(dp[i-1],dp[i-2]+a[i]);
  cout<<dp[n-1]<<endl;	  
  }
    return 0;
}
