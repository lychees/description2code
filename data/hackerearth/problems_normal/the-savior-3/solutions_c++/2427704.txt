using namespace std;

int main()
{
	int n,t;
	scanf("%d",&t);
	while(t--){
		int a[100005]={0};
		scanf("%d",&n);
		int o=0,e=0,ans=0;
		for(int i=0 ; i<n ; i++){
			int x; scanf("%d",&x);
			if(x%2) ans+=o-a[x], o++, a[x]++;
			else ans+=e-a[x], e++, a[x]++;
		}
		printf("%d\n",ans);
	}
}
