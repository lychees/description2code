a = input()
lis = []
for i in range(a):
    x = input()
    lis.append(x)


def fun(n):
    matches = 0
    while n > 1:
        if n % 2 == 1:
            x = n/2
            n = (n+1)/2
            matches = matches+x
        else:
            x = n/2
            n = n/2
            matches = matches+x
    print matches


for i in lis:
    fun(i)
