#include <iostream>
using namespace std;

int main()
{
	int in = 0;
	cin >> in;
	for(int i =0; i < in; i++)
	{
		int k,n;
		cin >> k >> n;
		n+=1;
		int pascal[n][n];
		//generate
		for(int a=0; a<n; a++)
			for(int b = 0; b<n ; b++)
				pascal[a][b]  = 0 ;
		for(int a=0; a<n; a++)
			pascal[a][a]  = 1 ;
		for(int a=0; a<n; a++)
			pascal[a][0]  = 1;
			
		//display

		for(int a=2; a<n; a++)
		{
			for(int b = 1; b<n ; b++)
			{
				if(pascal[a][b] == 1) break;
				pascal[a][b] = pascal[a-1][b]+pascal[a-1][b-1];
			}
		}
		// for(int a=0; a<n; a++)
		// {
		// 	for(int b = 0; b<n ; b++)
		// 	{
		// 		cout << pascal[a][b] << " ";
		// 	}
		// 	cout << endl;
		//}
		if(pascal[n-1][k-1] != 1)
			cout << pascal[n-1][k-1] << " ";
		if(n-k-1 == 1)
			cout <<"x";
		else if(n-k > 0)
			cout << "x^" << n-k << " ";
		if(k-1 == 1)
			cout <<"y";
		else if(k-1 > 0)
			cout << "y^" << k-1;
		cout <<endl;
		

	}
    return 0;
}
