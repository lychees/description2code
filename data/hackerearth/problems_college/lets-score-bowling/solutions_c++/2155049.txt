#include <iostream>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int a,b,c,score=0;
    	scanf("%d%d%d",&a,&b,&c);
    	
    	//a
    	
    	if(a==0)
    	{
    		;
    	}
    	else if(a>0 && a<7)
    	{
    		score+=1;
    	}
    	else if(a>6 && a<9)
    	{
    		score+=2;
    	}
    	else if(a==9)
    	{
    		score+=10;
    		printf("%d\n",score);
    		continue;
    	}
    	
    	//b
    	
    	if(b==0)
    	{
    		;
    	}
    	else if(b>0 && b<7)
    	{
    		score+=1;
    	}
    	else if(b>6 && b<9)
    	{
    		score+=2;
    	}
    	else if(b==9)
    	{
    		score+=7;
    		printf("%d\n",score);
    		continue;
    	}
    	
    	//c
    	
    	if(c==0)
    	{
    		;
    	}
    	else if(c>0 && c<7)
    	{
    		score+=1;
    	}
    	else if(c>6 && c<9)
    	{
    		score+=2;
    	}
    	else if(c==9)
    	{
    		score+=4;
    		printf("%d\n",score);
    	}
    	if(a!=9 && b!=9 &&c!=9)
    	{
    		printf("0\n");
    	}
    }
    return 0;
}
