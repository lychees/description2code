#include <stdio.h>
#include<iostream>
#include <vector>

using namespace std;

vector<int> dp[10005];
long long len[10005];

void add(int x) {
	int c = 0, d;
	
	//printf("len= %d %d\n", len[x - 2], len[x - 1]);
	len[x] = (len[x - 1] > len[x - 2] ? len[x - 1] : len[x - 2]);
	//printf("%d %d\n", x, len[x]);
	
	for(int i = 0;i < len[x];++i) {
		d = c;
		if(i < len[x - 1]) d += dp[x - 1][i];
		if(i < len[x - 2]) d += dp[x - 2][i];
		dp[x].push_back(d % 10);
		c = d / 10;
	}
	if(c != 0) dp[x].push_back(c), len[x]++;
}

int main()
{
    int T, n;
    
    dp[1].push_back(2); len[1] = 1;
    dp[2].push_back(3); len[2] = 1;
    for(int i = 3;i < 10001;++i) add(i);
    
    
    scanf("%d", &T);
    while(T--) {
    	scanf("%d", &n);
    	
    	for(int i = len[n] - 1;i >= 0;--i) printf("%d", dp[n][i]);
    	printf("\n");
    }
    
    return 0;
}
