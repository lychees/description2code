#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,i,left,right,dist;
	vector <pair< long long int,long long int> >vect;

	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>left>>right>>dist;
		//vect.push_back(make_pair(left,right),dist);
		vect.push_back(make_pair(left,left+dist));
	}

	sort(vect.begin(),vect.end());

	//  for(i=0;i<n;i++)
	// {
	//       cout<<vect[i].first.first<<endl;
	// }

	right=-1000000004;
	long long int count=0;
	for(i=0;i<n;i++)
	{
		if(vect[i].first > right)
		{
			count+=( vect[i].second - vect[i].first) + 1;
			right=vect[i].second;
		}


		else if(vect[i].first <= right && vect[i].second > right)
		{
			count+=vect[i].second - right;
			right=vect[i].second;
		}
		//else if(vect[i].first<=right && vect[i].second <=right)
		//{
//	 cout<<"the right is "<<count<<endl;
	}	
	cout<<count<<endl;

	}
