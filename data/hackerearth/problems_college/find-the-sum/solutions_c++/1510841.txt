#include<iostream>

using namespace std;

int main()
{
    int t;
    long long n1,n2,l,r,ans;

    cin>>t;

    while(t--)
    {
        cin>>l>>r;

        if(l>r)
        {
            long long temp=l;
            l=r;
            r=temp;
        }
        n1=ans=0;
        while(((n1*(n1+1))>>1) < l)
        n1++;

        n2=n1;
        while(((n2*(n2+1))>>1) < r)
        n2++;

        if(n1%2==0)
        {
            if((((n1*(n1+1))>>1) - l)%2 == 0)
            ans -= n1;

            n1--;

        }else
        {
            if((((n1*(n1+1))>>1) - l)%2 == 0)
                n1 -= 2;

        }

        if(n2%2==0)
        {
            if((r - ((n2*(n2+1))>>1) + n2)%2 == 1)
            ans += n2;

            n2--;

        }else
        {
           if((r - ((n2*(n2+1))>>1) + n2)%2 == 0)
            n2 -=2;
        }

        n2 = (n2>>1) +1;
        n1 = (n1>>1) +1;
        ans += n2*n2 - n1*n1;

        cout<<ans<<'\n';


    }
    return 0;
}
