#include <iostream>
using namespace std;

int main()
{
    string s;
    int i;
    getline(cin,s);
    if(s[0]>=97 && s[0]<=122)
    	s[0]=s[0]-32;
    for(i=1;i<s.size()-1;i++)
    {
    	if(s[i]==' ')
    	{
    		if(s[i+1]>=97 && s[i+1]<=122)
    			s[i+1]=s[i+1]-32;
    	}
    }
    cout<<s;
    return 0;
}
