#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,k,m;
    	cin>>n>>k>>m;
    	
    	long long int sum=k*n*n*(n+1)*(n+1);
    	sum=sum/4;
    	
    	if(sum<=m)
    	{
    		cout<<"0"<<endl;
    		
    	}
    	else
    	{
    		cout<<sum-m<<endl;
    	}
    	
    	
    }
    return 0;
}
