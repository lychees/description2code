#include "bits/stdc++.h"
using namespace std;
bool isPal(string s)
{
	for(long int i=0,j=s.length()-1;i<j;i++,j--)
	{
		if(s[i]!=s[j])
			return 0;
	}
	return 1;
}
int main()
{
	long int tc;
	cin>>tc;
	while(tc--)
	{
		string s;
		cin>>s;
		bool wrong=0;
		for(long int i=0;i<s.length()&&wrong==0;i++)
			if(s[i]!='1' && s[i]!='0' && s[i]!='8')
				wrong=1;
		if(wrong == 0 && isPal(s)==true)
			cout<<"YES\n";
		else
			cout<<"NO\n";
	}
	return 0;
}
