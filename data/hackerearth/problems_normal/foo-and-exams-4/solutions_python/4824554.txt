import sys
import math

def binary():
	sys.stdin.readline()
	line = sys.stdin.readline()
	while line:
		a, b, c, d, k = [int(x) for x in line.strip().split(" ")]
		
		func = lambda x: a * math.pow(x, 3) + b * math.pow(x, 2) + c * x + d
		upper = k
		lower = 0

		while True:
		    i = (upper + lower) / 2
		    if func(i) == k or upper - lower <= 1:
		        break
		    elif func(i) > k:
		        upper = i
		    elif func(i) < k:
		        lower = i

		print lower
		line = sys.stdin.readline()
	

binary()
