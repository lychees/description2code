#include <iostream>
using namespace std;
 
int main()
{
    //cout << "Hello World!" << endl;
    int tc;
    cin>>tc;
    while(tc--)
    {
    	long long int n,m;
    
    	cin>>n>>m;long long int ans=n;
    	for(long long int i=n+1;i<=m;i++)
    	{
    		ans=ans&i;
    		
    	}
    //	ans=4&4;
    	cout<<ans<<endl;
    }
    return 0;
}
