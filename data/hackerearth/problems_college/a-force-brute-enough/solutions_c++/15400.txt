#include<iostream>
#include<stdio.h>

using namespace std;

int n,k=10;

int a[100] = {0};

int temp[10],curr;

void db(int t, int p)
{
	int j;
        if(t > n)
        {
            if(n%p == 0)
        	{
                for(j=1;j<=p;j++)
                {
                	if(curr<n-1)
                	{
                		temp[curr++] = a[j];
                	}
                    printf("%d",a[j]);
                }
            }
        }
        else
        {
            a[t] = a[t - p];
            db(t + 1, p);
            for(j=a[t-p]+1;j<k;j++)
            {
                a[t] = j;
                db(t + 1, t);
            }
        }
}

int main()
{
	scanf("%d",&n);
	while(n!=0)
	{
		db(1,1);
		curr = 0;
		for(int i=0;i<n-1;i++)
		{
			printf("%d",temp[i]);
		}
		printf("\n");
		scanf("%d",&n);
	}
	return 0;
}
