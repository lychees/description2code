#include <iostream>
#include<math.h>
using namespace std;
 
 
int main(){
   int a;
   cin>>a;
   while(a--)
   {
   	unsigned long long int b,tc;
   	cin >> b;
   	unsigned long long ans=log2(b);
   	unsigned long long ans2=pow(2,ans+1);
   	tc = b - (ans2-b);
   	if(tc==0)
   	{ tc = b;
   	 cout << tc - 1 << " " << tc <<endl;
   	}
   	else
   	{
   		cout<<ans2-1<<" "<<tc<<endl;
   	}
   	
   	
   }
   
    return 0;
}
