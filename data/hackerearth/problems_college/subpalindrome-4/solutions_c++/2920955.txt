#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char s[51];
		cin>>s;
		int dp[51][51], l=strlen(s);
		for(int i=0;i<l;i++)
		for(int j=0;i<j;i++)
		dp[i][j]=0;
		for(int i=l-1;i>=0;i--)
		{
			int c=1;
			dp[i][i]=1;
			if(i+1!=l)
			{
				if(s[i]==s[i+1])
				{
					c++;
					dp[i][i+1]=3;
				}
				else
					dp[i][i+1]=2;
			}
			for(int j=i+2;j<l;j++)
			{
				if(s[i]==s[j])
				{
					c++;
					c+=dp[i+1][j-1];
				}
				dp[i][j]=dp[i+1][j]+c;
			}
		}
		cout << dp[0][l-1] << endl;
	}
return 0;
}
