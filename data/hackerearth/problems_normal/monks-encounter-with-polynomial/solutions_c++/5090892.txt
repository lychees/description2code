#include <bits/stdc++.h>
using namespace std;
#define lln long long int

int main()
{
    lln t;
    cin >> t;
    for(lln i=0;i<t;i++){
    	lln a,b,c,k,z;
    	cin >> a >> b >> c >> k;
    	for(lln j=0;;j++){
    		z = a*j*j + b*j + c;
    		if(z>=k){
    			cout << j << endl;
    			break;
    		}
    	}
    	
    }
    return 0;
}
