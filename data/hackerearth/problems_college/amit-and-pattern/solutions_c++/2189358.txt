#include<iostream>
#define gc getchar_unlocked
using namespace std;
inline int getn(){
	int n=0, c=gc();
	while(c < '0' || c > '9') c = gc();
	while(c >= '0' && c <= '9')
		n = (n<<3) + (n<<1) + c - '0', c = gc();
	return n;
}
int main()
{
    int r=getn(),c=getn(),arr[r][c],i,j;
    for(i=0;i<r;i++)
    {
        for(j=0;j<c;j++)
        {
            arr[i][j]=getn();
        }
    }

    //Printing Upper diagonal elements
    for(int startPoint=0;startPoint<c;startPoint++)
    {
        i=0;
        j=startPoint;
        while(i<r && j>=0)
        {
            cout<<arr[i][j]<<" ";
            i=i+1;
            j=j-1;
        }
        cout<<"\n";
    }

    //Printing the remaining elements
    for(int startPoint=1;startPoint<r;startPoint++)
    {
        i=startPoint;
        j=c-1;
        while(i<r && j>=0)
        {
            cout<<arr[i][j]<<" ";
            i=i+1;
            j=j-1;
        }
        cout<<"\n";
    }
}
