#include<iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t>0)
	{
		
	t--;
	int n;
	cin>>n;
	int a[n+1];
	
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	int m=0;
	int li[n+1];
for(int i=1;i<=n;i++)
{
	li[i]=1;
	
	for(int j=1;j<i;j++)
	{
		if(a[i]>a[j])
		li[i]=li[j]+1;
	}
	m=max(m,li[i]);

}

cout<<n-m<<endl;

}
}
