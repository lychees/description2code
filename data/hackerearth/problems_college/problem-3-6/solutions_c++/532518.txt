#include <iostream>
using namespace std;
int main()
{
   int colh[100],colt[100],h[100],t[100],i,j,n;
   cin>>n;
   for (i=0;i<n;i++){
   	cin>>colh[i]>>colt[i];
   	h[i]=0;
   	t[i]=0;
   }
   
   for (i=0;i<n-1;i++){
   	for (j=i+1;j<n;j++){
   		if (colh[i]==colt[j]){
   			h[i]+=1;
   			h[j]+=1;
   		}else{
			h[i]+=1; t[j]+=1;	
   		}
   		if (colh[j]==colt[i]){
   			h[i]+=1;
   			h[j]+=1;
   		}
   		else{
   			h[j]+=1; t[i]+=1;
   		}
   	}
   }
   
   for (i=0;i<n;i++){
   	cout<<h[i]<<" "<<t[i]<<"\n";
   	
   }
}
