#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;scanf("%d",&t);
	while(t--){
		string str;cin>>str;
		if(str.size()<3){
			printf("Not possible!\n");
			continue;
		}
		sort(str.begin(),str.end());
		next_permutation(str.begin(),str.end());
		next_permutation(str.begin(),str.end());
		cout<<str<<" ";
		sort(str.begin(),str.end());
		reverse(str.begin(),str.end());
		prev_permutation(str.begin(),str.end());
		prev_permutation(str.begin(),str.end());
		cout<<str<<"\n";
	}
	return 0;
}
