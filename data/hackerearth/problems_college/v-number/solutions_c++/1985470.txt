#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--)
   {
   	string s;
   	cin>>s;
   	int len=s.length();
   	int l_sum=0,h_sum=0;
   	for(int i=0;i<len/2;i++)
   	l_sum+=s[i]-48;
   	if(len%2!=0)
   	for(int i=len/2+1;i<len;i++)
   	h_sum+=s[i]-48;
   	else
   	for(int i=len/2;i<len;i++)
   	h_sum+=s[i]-48;
   	if(l_sum==h_sum)
   	cout<<"YES"<<endl;
   	else
   	cout<<"NO"<<endl;
   	
   	
   }
    return 0;
}
