#include<bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    ll t,n,p,u,r,s,i,cas;
    scanf("%lld",&t);
    for(cas=1;cas<=t;cas++)
    {
        scanf("%lld%lld%lld%lld%lld",&n,&p,&u,&r,&s);
        for(i=0;i<n;i++)
        {
            if((p+r)<=u)
            p=p+r;
            else if((p-s)>=0)
            p=p-s;
            else
            break;
        }
        printf("Case #%lld: %lld\n",cas,p);
    }




    return 0;
}
