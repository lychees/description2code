import math
t = int(raw_input())
while t > 0:
    a, b = [int(x) for x in raw_input().split()]
    if a == b - 1:
        print a & b
    else:
        y = int(math.log(b, 2))
        if b == int(math.pow(2, y)):
            print(b-1) & (b-2)
        else:
            if b % 2 == 0:
                print(b-1) & (b-2)
            else:
                print b & (b-1)

    t -= 1
