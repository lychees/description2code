#include <iostream>
#include <string>
using namespace std;

int main()
{
	unsigned long long int count,bits[100000],temp,key,i,tests,n,k,prod;
	cin >> tests;
	while(tests--)
	{
		cin >> n >> k;
		key = 1;
		count= 0;
		
		if(k == 0)
		{

		    cout << "NO" << endl;
		    continue;
		}
		
		temp = k;
		
		for(i = 0 ; i < 100000 ; i++)
			  bits[i] = 0 ;
			  
			  
		for(i = 0 ; temp > 0 ; i++)
		{
			bits[i] = temp % 2;
			temp /= 2;
			count++;
		}
		

		prod = 1;
				
		for(i = 0 ; prod <= n && key == 1;i++,prod++)
		{
		

			if(bits[i] == 0)
					   key = 0;

		}
		

		if(key == 1)
			   cout << "YES"  << endl;
			   
		else
			cout << "NO" << endl;


	}
	return 0;
}
