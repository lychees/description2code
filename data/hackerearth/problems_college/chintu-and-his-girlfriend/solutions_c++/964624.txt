#include<bits/stdc++.h>
using namespace std;
int main(){
  int n,k,i,j,count=0;
  cin>>n>>k;
  int a[26];
  memset(a,0,sizeof(a));
  j=n;
  while(j--){
    char s[100];
    cin>>s;
    int x=strlen(s);
    sort(s,s+x);
    for(i=0;i<x;i++){
      if (i>=1)
	if (s[i]==s[i-1]) continue;
      a[s[i]-'a']++;}
  }
  for(i=0;i<26;i++){
    // cout<<a[i]<<" "<<endl;
    if (a[i]==n) count++;}
  if (count>=k) cout<<":-)"<<endl;
  else cout<<":-("<<endl;
  return 0;

}
