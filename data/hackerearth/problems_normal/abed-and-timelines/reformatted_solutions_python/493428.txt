tc = int(raw_input())
while (tc > 0):
    tc = tc - 1
    n, m = map(int, raw_input().split())
    r1, c1 = map(int, raw_input().split())
    r2, c2 = map(int, raw_input().split())
    price = int(raw_input())
    x = abs(r1-r2)
    y = abs(c1-c2)
    ans = min(x, n-x) + min(y, m-y)
    finalans = ans * price
    if (finalans > 1000):
        print finalans
        print "Let go of the group."
    else:
        print finalans
        print "You saved the group."
