#include <iostream>
using namespace std;

int dp[110][110];
int arr[101][101];
int main()
{
    int n;
    cin>>n;
    for(int i=1;i<=n;++i)
    {
    	for(int j=1;j<=n;++j)
    		cin>>arr[i][j];
    }
    
    for(int i=n;i>=1;--i)
    {
    	for(int j=1;j<=n;++j)
    	{
    		dp[i][j]=dp[i+1][j];
    		int num=2*j;
    		while(num<=n)
    		{
    			dp[i][j]=max(dp[i][j],dp[i+1][num]);
    			num*=2;
    		}
    		num=j/2;
    		while(num>=1){
    			dp[i][j]=max(dp[i][j],dp[i+1][num]);
    			if(num*2!=j)
    			{
    				int c=num*2;
    				while(c<=n)
    				{
    					dp[i][j]=max(dp[i][j],dp[i+1][c]);
    					c*=2;
    				}
    			}
    			num/=2;
    		}
    		dp[i][j]+=arr[i][j];
    	}
    }
    int ans=dp[1][1];
    for(int i=2;i<=n;++i)
    {
    	ans=max(ans,dp[1][i]);
    }
    cout<<ans;
    return 0;
}
