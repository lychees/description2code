#include <bits/stdc++.h>
using namespace std;

int main()
{
   set<int> s;
   set<int>::iterator it;
   int i,cal,val=2,t,n;
   s.insert(1);
   cal=1;
   for(i=1;i<=29;i++)
   {
   		cal=cal*val;
   		s.insert(cal);
   }
   scanf("%d",&t);
   while(t--)
   {
   		scanf("%d",&n);
   		if((n&(n-1)))
   		{
   			it=s.lower_bound(n);
   			it--;
   			val=n-(*it);
   			val=(2*val)+1;
   			
   		}
   		else
   			val=1;
   		printf("%d\n",val);
   }
    return 0;
}
