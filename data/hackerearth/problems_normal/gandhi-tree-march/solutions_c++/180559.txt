void calc(int x, int y, int c)
{
    //cout << x << " " << y << " " << c - 20000 << endl;
    if(x > y)
        return;
    if(x == y)
    {
        if(s[x] != '.')
            V[c].pb(s[x]);
        return;
    }
    V[c].pb(s[x]);
    if(s[x+2] == '.')
    {
        calc(x+3,y-1,c+1);
    }
    else
    {
        int cur = 0;
        int tmp = -1;
        for(int i=x+3;i<=y;i++)
        {
            if(s[i] == ')')
                cur--;
            else if(s[i] == '(')
                cur++;
            if(cur == 0)
            {
                tmp = i;
                break;
            }
        }
        calc(x+2,tmp,c-1);
        calc(tmp+1,y-1,c+1);
    }
}
