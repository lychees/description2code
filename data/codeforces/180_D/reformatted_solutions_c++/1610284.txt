#include <algorithm>
#include <climits>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <queue>
#include <set>
#include <string>
#include <vector>
using namespace std;
string S, T;
int main() {
  cin >> T >> S;
  int N = S.length(), M = T.length();
  S += ('a' - 1);
  for (int i = 0; i < M; ++i) {
    // put the earliest letter that works
    for (int j = i; j < M; ++j)
      if (T[j] == S[i]) {
        swap(T[i], T[j]);
        break;
      }
    if (T[i] == S[i]) {
      // check rest
      sort(T.begin() + i + 1, T.end(), greater<char>());
      if (T > S)
        continue;
    }
    int pick = -1;
    for (int j = i; j < M; ++j)
      if (T[j] > S[i] && (pick == -1 || T[j] < T[pick]))
        pick = j;
    if (pick == -1) {
      printf("-1\n");
      return 0;
    }
    swap(T[i], T[pick]);
    sort(T.begin() + i + 1, T.end());
    break;
  }
  cout << T << endl;
}
