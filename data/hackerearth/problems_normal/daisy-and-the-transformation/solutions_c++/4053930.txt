#include <bits/stdc++.h>
using namespace std;

int main()
{
    // cout << "Hello World!" << endl;
    int t; 
    cin>>t;
    while(t--) {
    	int n, i, m, k;
    	string s1, s2;
    	cin>>s1;
    	cin>>s2;
    	cin>>k;
    	n = s1.size(), m = s2.size();
    	if(m != n) {
    		cout<<"No\n";
    	}
    	else {
	    	int c = 0;
	    	for(i = 0; i < n; i++) {
	    		if(s1[i] != s2[i]) c++;
	    	}
	    	if(c == k) cout<<"Yes\n";
	    	else if(c < k) {
	    		int s = k-c;
	    		if(s%2 == 0) cout<<"Yes\n";
	    		else cout<<"No\n";
	    	} else cout<<"No\n";
	    }
    }
    return 0;
}
