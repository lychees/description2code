#include<bits/stdc++.h>
using namespace std;

int a[5555];
long long dp[5555][5555];
long long s[5555];

int main() {
  int n,m,k;
  cin>>n>>m>>k;
  for(int i=1;i<=n;i++) {
    cin>>a[i];
    s[i]=s[i-1]+a[i];
  }
  long long ans=0;
  for(int i=1;i<=n;i++) {
    for(int j=0;j<=k;j++) {
      dp[i][j]=max(dp[i-1][j],dp[i][j]);
      if(i>=m && j>0) {
        dp[i][j]=max(dp[i][j],dp[i-m][j-1]+s[i]-s[i-m]);
      }
      ans=max(ans,dp[i][j]);
    }
  }
  cout<<ans<<endl;
}
