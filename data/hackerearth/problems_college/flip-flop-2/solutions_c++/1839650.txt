#include <iostream>
#include <string>
using namespace std;

int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		int deletions = 0;
		string inp;
		cin>>inp;
		for(int i = 0; i < inp.size() - 1; i++)
		{
			char cur = inp[i];
			char next = inp[i+1];
			if(cur == next)
			{
				deletions++;
			}
		}
		cout<<deletions<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
