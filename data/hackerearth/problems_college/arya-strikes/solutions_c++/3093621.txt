#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;
int maze[55][55];
bool check(int x,int y,int n){
	if(x<n and x>=0 and y>=0 and y<n and maze[x][y]==1)return true;
	return false;
}
bool backtracking(int x,int y,int n){
	if(x==n-1 and y==n-1)return true;
	if(check(x,y,n)){
		if(backtracking(x+1,y,n))//go down
			return true;
		if(backtracking(x,y+1,n))//go right
			return true;
		return false;//if from (x,y), destination can't be reached
	}
	return false;//otherwise if (x,y) is not safe
}
int main(void){
	int tests;
	scanf("%d",&tests);
	while(tests--){
		memset(maze,0,sizeof(maze));
		int n;
		scanf("%d",&n);
		for(int i=0;i<n;i++){
			for(int j=0;j<n;j++){
				scanf("%d",&maze[i][j]);
			}
		}
		if(backtracking(0,0,n))
			printf("POSSIBLE\n");
		else printf("NOT POSSIBLE\n");
	}
	return 0;
}
