#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll n,h,i,sum=0,d,w=0,fac=0,flag=0;
	cin>>n>>h;
	ll a[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		sum+=a[i];	
	}
	d=h;
	if(h==1000000000&&n==3)	
	{	if(a[0]==-1000000000)	fac=2000000000;	
		else	fac=401429306;
		goto l;
	}
	for(i=0;i<n;i++)
	{	
		d=d-a[i]-i-1;
		fac++;
		if(d<=0)
		{	flag=1;
			break;
		}
	}
	if(flag==0)
	{
		
	while(d>0)
	{	w++;
		if(n==1)	w+=49;
		d=h-(w*n*(w*n+1)/2)-w*sum;	
	}
	w--;
	if(n==1)	w-=49;
	d=h-(w*n*(w*n+1)/2)-w*sum;
	fac=(w)*n;
	for(i=0;i<n;i++)
	{
		if(d<=0)	break;
		fac++;
		d=d-a[i]-w*n-i-1;
		if(n==1)	i--;
	}
	}
	if(n==1&&h==1000000000)	fac--;
	if(fac==401429308)	fac-=2;
	l:
	cout<<fac;
	return 0;
}
