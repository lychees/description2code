#include<bits/stdc++.h>
#define ll long long int
using namespace std;
bool prime[1000006];
ll cnt[1000006];
int main()
{
	ll t,l,r,i,j,temp=0;
	prime[0]=prime[1]=1;
	//cnt[0]=cnt[1]=0;
	for(i=2;i*i<=1000006;i++)
	{
		if(!prime[i])
		{
			j=2;
			while(j*i<=1000006)
			{
				prime[j*i]=1;
				j++;
			}
			//temp=temp+i;
		}
		//cnt[i]=temp;
	}
	cnt[0]=0;
	temp=0;
	cnt[1]=cnt[2]=cnt[3]=cnt[4]=cnt[5]=1;
	temp=1;
	for(i=6;i<1000006;i++)
	{
		if(prime[i])
		{
			temp++;
		}
		cnt[i]=temp;
	}
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&l,&r);
		printf("%lld\n",cnt[r]-cnt[l-1]);
	}
}
