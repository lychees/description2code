#include <iostream>
#include <bits/stdc++.h>
#define MAXN 100000
using namespace std;

//resubmitting
int pr[MAXN+1];
int cnt[MAXN+1];
void sieve()
{
	int limit=sqrt(MAXN);
	for(int i=2;i<=limit;++i)
	{
		if(pr[i]==0)
		{
			for(int j=i*i;j<=MAXN;j+=i)
			{
				pr[j]=i;
			}
		}
	}
	for(int i=2;i<=MAXN;++i)
	{
		if(pr[i]==0) cnt[i]=1;
		else cnt[i]=1+cnt[i/pr[i]];
	}
}
int main()
{
    sieve();
    //for(int i=2;i<=10;++i) cout<<cnt[i]<<endl;
    int n,a,ans=0;
    cin>>n;
    for(int i=0;i<n;++i)
    {
    	cin>>a;
    	ans=ans^cnt[a];
    }
    if(ans!=0) cout<<"ALICE"<<endl;
    else cout<<"BOB"<<endl;
    return 0;
}
