#include <iostream>
#include<queue>
#define ll long long 
#define endl "\n"
using namespace std;

int main()
{
    ll m,n,t;
    cin >> m >>n;
    priority_queue<int> p;
    while(m--){
    	cin >> t;
    	p.push(t);
    }
    ll p1,sum=0;
    while(n--){
    	p1=p.top();
    	p.pop();
    	sum+=p1;
    	p.push(p1-1);
    }
    cout << sum <<endl;
    return 0;
}
