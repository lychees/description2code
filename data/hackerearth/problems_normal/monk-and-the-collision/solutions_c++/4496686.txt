#include <iostream>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
        int n,a[10]={0},x,sum=0;
        cin>>n;
        while(n--)
        {
            cin>>x;
            a[x%10]++;
        }
        for(int i=0;i<10;i++)
            if(a[i]>1)
                sum += (a[i]-1);
        cout<<sum<<endl;
    }
}
