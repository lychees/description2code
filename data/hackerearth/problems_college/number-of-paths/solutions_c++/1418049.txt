#include<bits/stdc++.h>
using namespace std;

const int N=1010;
const int md=1e9+7;

vector<pair<int,int> >v;
int dp[N][N];

void preprocess() {
    for(int i=0;i<N;i++) {
        dp[i][0]=dp[0][i]=1;
    }
    for(int i=1;i<N;i++) {
        for(int j=1;j<N;j++) {
            dp[i][j]=dp[i-1][j]+dp[i][j-1];
            if(dp[i][j]>=md) {
                dp[i][j]-=md;
            }
        }
    }
    return;
}
int main() {
    preprocess();
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        v.clear();
        int n,q;
        scanf("%d%d",&n,&q);
        v.push_back(make_pair(0,0));
        v.push_back(make_pair(n-1,n-1));
        for(int i=0;i<q;i++) {
            int a,b;
            scanf("%d%d",&a,&b);
            v.push_back(make_pair(a,b));
        }
        sort(v.begin(),v.end());
        int ans=1;
        for(int i=1;i<v.size();i++) {
            int x=v[i].first-v[i-1].first;
            int y=v[i].second-v[i-1].second;
            ans=1ll*ans*dp[x][y]%md;
        }
        printf("%lld\012",ans);
    }
    return 0;
}
