#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int f=0,n,i,j,k,l;
   char a[100][15],b;
   scanf("%d",&n);
   for(i=0;i<n;i++)
   scanf("%s",&a[i]);
   for(i=0;i<n;i++)
   {
   	l=strlen(a[i]);
   	for(j=i+1;j<n;j++)
   	{
   		if(strlen(a[i])==strlen(a[j]))
   		{
   		
   			for(k=0;k<l;k++)
   			{
   				if(a[i][k]==a[j][l-k-1])
   				continue;
   				else
   				break;
   			}
   			if(k==l)
   			{
   				f=1;
   				break;
   			}
   		}
   	}
   	if(f==1){
   	printf("%d %c\n",l,a[i][l/2]);
   	break;
   	}
   }
    return 0;
}
