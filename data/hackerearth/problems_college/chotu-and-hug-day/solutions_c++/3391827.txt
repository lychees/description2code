#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,h,i,l,temp=11111111;
		scanf("%d %d",&n,&h);
		int a[n];
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		for(i=0;i<n;i++)
		{
			l = abs(a[i] - h);
			if(l<temp)
			temp = l;
		}
		printf("%d\n",temp);
		
		
	}
	
}
