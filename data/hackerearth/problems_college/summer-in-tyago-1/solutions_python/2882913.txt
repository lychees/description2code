n = input()
arr = map(int,raw_input().split())
f=0
for i in range(n-1):
	if arr[i]==arr[i+1]+1:
		print ":("
		f=1
		break 
if f==0:
    for i in range(n):
        for j in range(1,n):
            if arr[j]<arr[j-1]:
                arr[j]^=arr[j-1]
                arr[j-1]^=arr[j]
                arr[j]^=arr[j-1]
                arr[j-1]+=1
                arr[j]-=1
if f==0:
	arr = map(str,arr)
	print ' '.join(arr)
