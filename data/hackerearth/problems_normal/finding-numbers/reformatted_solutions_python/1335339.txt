for T in range(input()):
    Z, M, N = map(int, raw_input().split())
    l, ans = [], 0
    for i in range(1, int(Z ** .5) + 1):
        if Z % i == 0:
            l.append(i)
            if i != Z / i:
                l.append(Z / i)
    for i in l:
        if i <= min(M, N) and Z / i <= max(M, N):
            ans += 1
    print ans
