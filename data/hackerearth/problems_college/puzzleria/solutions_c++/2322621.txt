#include<cstdio>
#include<iostream>
using namespace std;
long long fact(int n)
{
	long long fact=1,i;
	for(i=1;i<=n;i++)
	{
	fact= fact*i;	
	}
	return fact;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	int a,b;
	cin>>a>>b;
	int n = fact(a+b);
	long long m = fact(7)/(fact(7-a)*fact(a));
	
	long long o = fact(5)/(fact(5-b)*fact(b));
	cout<<m*o*n<<"\n";	
	}
}
