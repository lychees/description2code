int main()
{

    long long i,j,k,m,n,x,y,t;
    cin>>t;
    for(j=1;j<=t;j++)
    {
    	cin>>n;long long a[n+1];long long b[n+1];
    	for(i=0;i<n;i++) scanf("%lld",&a[i]);
    	x=0;y=0;
    	b[0]=a[0];
    	if(n>=2)
    	b[1]=max(a[0],a[1]);
    	for(i=2;i<n;i++)
    	{
    		b[i]=max(b[i-1],b[i-2]+a[i]);
    	}
    	cout<<"Case "<<j<<": "<<b[n-1]<<endl;
    }
    return 0;
}
