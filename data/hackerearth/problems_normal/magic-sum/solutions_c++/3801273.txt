ll power(ll a,ll b)
{
	if(!b)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2);
	temp=(temp*temp);
	if(b&1)
		temp=(temp*a);
	return temp;
}
ll solve( int i, int j, vi& v)
{
	ll ans = 0;
	while( i ^ j)
	{
		if( i > j)
		{
			ans += v[i];
			i >>= 1;
		}
		else
		{
			ans += v[j];
			j >>= 1;
		}
	}
	ans += v[i];
	return ans;
}
int main()
    {
		int i, t, n, val, j;
		
		sc("%d",&t);
		while( t-- )
		{
			sc("%d",&n);
			vi v( n+1 );
			for( i=1;i<=n;i++ )
			{
				sc("%d",&(v[i]));
			}
			int leaves = floor ((n+1) / 2);
			vi check( leaves);
			val = n;
			for( i=0;i<leaves;i++ )
			{
				check[i] = val;
				val--;
			}
			ll maxi = -1e18;
			for( i=0;i<leaves;i++ )
			{
				for( j=i;j<leaves;j++ )
				{
					maxi = max( maxi, solve( check[i], check[j], v) );
				}
			}
			cout << maxi <<"\n";
		}
		
    return 0;
}
