def is_palindrome(s):
    if s == s[::-1]:
        return True
    return False


def count(a, b):
    cnt = 0
    for i in range(len(a)+1):
        if is_palindrome(a[:i]+b+a[i:]):
            cnt += 1
    return cnt


t = int(raw_input())
while t > 0:
    a = raw_input()
    b = raw_input()
    print count(a, b)
    t -= 1
