#include<iostream>
#define ll long long
using namespace std;


int main()
	{
	                    
	 int t;
	 ll n,b;
	 cin>>t;
	 while(t--){
	 	cin>>n>>b;
	 	ll temp = n%b;

			if(temp >= (float)b/2)
			{
				n = n + (b - temp);
            	}

			else
			{
				n = n - temp;
			}

		         
	 	cout << n << endl; 
	 	
	 }
	      

	return 0;
	}
