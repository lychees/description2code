ll dp[100][100];
ll func(int fullpieces,int halfpieces)
{
    if(dp[fullpieces][halfpieces]!=-1)
        return dp[fullpieces][halfpieces];
    if(fullpieces==0)
        return dp[fullpieces][halfpieces] =1;
    else if(halfpieces==0)
        return dp[fullpieces][halfpieces] = func(fullpieces-1,halfpieces+1);
    else 
        return dp[fullpieces][halfpieces] =func(fullpieces-1,halfpieces+1)+func(fullpieces,halfpieces-1);
    
}
