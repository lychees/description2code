#include <bits/stdc++.h>
using namespace std;

#define MAX 1000

long long int answer[MAX];

long long int getGcd(long long int a, long long int b) {
    if (a%b == 0) return b;
    else return getGcd(b, a%b);
}

long long int getAns(long long int a) {
    if (a <= 2) return 0;
    long long int i, gcd, ans = 0, lcm = 1, left = a-2, newLeft;
    for (i=2; left > 0; i++) {
        gcd = getGcd(lcm, i);
        lcm = (lcm/gcd) * i;
        newLeft = a/lcm;
        if (i == 2) newLeft--;
        ans += ((left-newLeft)*answer[i]);
        left = newLeft;
        // cout<<"ans = "<<ans<<", left = "<<left<<", lcm = "<<lcm<<endl;
    }
    return ans;
}

int main() {
	long long int i, j, a, b, ans;
	answer[2] = 1, answer[3] = 2, answer[4] = 3;
	for (i=5; i<MAX; i++) {
	    for (j=2; j<i; j++) {
	        if (i%j != 0) {
	            answer[i] = answer[j]+1;
	            break;
	        }
	    }
	}
	cin>>a>>b;
	ans = getAns(b) - getAns(a-1);
	ans += (b-a+1);
	cout<<ans;
	return 0;
}
