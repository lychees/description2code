char maxChar(string str)
{
    int arr[256];
    memset(arr,0,sizeof(arr));
    int n= str.length();
    for(int i=0;i<n;i++) {
    	arr[str[i]]++;
    }
    int max = INT_MIN,c=0;
    for(int i=0;i<256;i++) {
    	if(arr[i]>max) {
    		max = arr[i];
    		c=i;
    	}
    	
    }
    return (char)c;
}
