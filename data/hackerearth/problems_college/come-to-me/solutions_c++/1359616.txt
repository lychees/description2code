template<class T>
inline void chkmax(T& x, T y) {
	if (x < y) x = y;
}

template<class T>
inline void chkmin(T& x, T y) {
	if (y < x) x = y;
}

typedef long long LL;
typedef pair<int, int> PII;
typedef vector<int> VI;
int n;
char s[10005];
int chk[30];

int main() {
	int tn, ti, st, en, c, rlt;
	
	scanf("%d", &tn);
	for (ti = 1; ti <= tn; ti++) {
		scanf("%s", s);
		n = strlen(s);
		memset(chk, 0, sizeof chk);
		st = 0;
		rlt = 0;
		for (en = 0; en < n; en++) {
			c = s[en] - 'A';
			chk[c]++;
			for (; chk[c] > 1; st++) chk[s[st] - 'A']--;
			chkmax(rlt, en - st + 1);
		}
		printf("%d\n", rlt);
	}
	return 0;
}
