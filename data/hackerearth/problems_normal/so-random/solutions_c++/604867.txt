#include<bits/stdc++.h>
using namespace std;
int main()
{
	//ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
	    int time;
	    double p,rootp;
	    cin>>p>>time;
	    rootp=pow(1-p,(double)1/1440);
	  //  cout<<"rootp is "<<rootp;
	    double a=1;
	    int i;
	    for(i=0;i<time;i++)
	        a*=(rootp);
	    printf("%.4lf\n",1-a);
	}
	return 0;
}
