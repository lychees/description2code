#include<stdio.h>
int main()
{
	int t,n,i,a[100001],b[100001],flag;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=1;i<=n;i++)
			scanf("%d",&a[i]);
		for(i=1;i<=n;i++)
		{
			b[a[i]]=i;
		}	
		flag=1;
		for(i=0;i<n;i++)
		{
			if(a[i]!=b[i])
			{
				printf("not inverse\n");
				flag=0;
				break;
			}
		}
		if(flag==1)
			printf("inverse\n");
	}
	return 0;
}
