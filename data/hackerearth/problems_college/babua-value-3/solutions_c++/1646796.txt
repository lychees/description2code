#include <iostream>
#include <algorithm>
using namespace std;

long long gcd(long long a, long long b){
	if(b==0) return a;
	else return gcd(b,a%b);
}

int main()
{
    long long n;
    cin>>n;
    long long a[n];
    long long cum[n];
    for(int i = 0; i<n; i++) cin>>a[i];
    sort(a,a+n);
    cum[0] = a[0];
    for(int i = 1; i<n; i++) cum[i] = cum[i-1] + a[i];
    long long ans = 0;
    long long tmp;
    for(int i = 0; i<n; i++){
    	if(i==0){
    		tmp = cum[n-1]-cum[i] - (n-1)*a[i];
    		ans += tmp;
    	}else if(i==n-1){
    		tmp = (n-1)*a[i] - cum[i-1];
    		ans += tmp;
    	}else{
    		tmp = (i)*a[i] - cum[i-1];
    		tmp += (cum[n-1] - cum[i]) - (n-1-i)*a[i];
    		ans += tmp;
    	}
    	//cout<<i<<" "<<tmp<<endl;
    }
    ans += cum[n-1];
    long long g = gcd(ans,n);
    cout<<ans/g<<"/"<<n/g<<endl;
    return 0;
}
