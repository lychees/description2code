#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    long long t,n,*p,b,i,j,*s,c,temp,*d,l;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	p = new long long[n];
    	for(i=0;i<n;i++)
    	 cin>>p[i];
    	b = pow(2,n);
    	s = new long long[b];
    
    	for(c= 0; c < b; c++)
    	{
    		s[c]=0;
      		for(j = 0; j < n; j++)
       		{
          /* Check if jth bit in the counter is set
             If set then pront jth element from set */
          		if(c & (1<<j))
            	{
            	
            		s[c]+=p[j];
            		
            	}	
            }
       
        }
        l = s[b-1]+1;
        d = new long long[l];
        for(i=0;i<b;i++)
          	d[s[i]]++;
        c=0;
        for(i=0;i<l;i++)
        {
        	if(d[i]!=0)
        	 c++;
        }
        
        cout<<c<<endl;
       
    }
    return 0;
}
