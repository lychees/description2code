#include <iostream>
#include <algorithm>
using namespace std;

int cnt[26];
int solve()
{
	string s;
	cin>>s;
	int len=s.length();
	int l=0,r=0,ans=1;
	cnt[s[0]-'A']++;
	for(int i=1;i<len;++i)
	{
		if(cnt[s[i]-'A']==0)
		{
			cnt[s[i]-'A']=1;
			r=i;
			ans=max(ans,r-l+1);
		}
		else
		{
			for(;l<i;++l)
			{
				if(s[l]==s[i]){
					l++;
					break;
				}
				cnt[s[l]-'A']--;
			}
			cnt[s[i]-'A']=1;
			r=i;
			ans=max(ans,r-l+1);
		}
	}
	printf("%d\n",ans);
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	for(int i=0;i<26;++i) cnt[i]=0;
    	solve();
    }
    return 0;
}
