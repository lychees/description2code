#include <iostream>
#include <set>
#include <algorithm>
using namespace std;

int main()
{
    
    int n;
    cin>>n;
    set<string> s;
    while(n--) {
    	string str;
    	cin>>str;
    	if(str.length()>3) {
    		sort(str.begin()+1,str.end()-1);
    	}
    	s.insert(str);
    }
    cout<<s.size()<<endl;
    return 0;
}
