def f():
    return map(int, raw_input().split())


n, k = f()
a = sorted([(v, i) for i, v in enumerate(f())])
r, v = 0, []
for i in range(n):
    if k - a[i][0] >= 0:
        v.append(a[i][1] + 1)
        r += 1
        k -= a[i][0]
print r
if r > 0:
    print ' '.join(map(str, v))
