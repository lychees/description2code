#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
int f[1000001];
long long ff[1000001];
int main()
{
    memset(f,0,sizeof(f));
    for(int i=2;i<=1000000;i++){
    	for(int j=2*i;j<=1000000;j+=i)
    	if (!f[j]) f[j] = i;
    }
    ff[0] = 0;
    ff[1] = 0;
    for(int i=2;i<=1000000;i++) ff[i] = ff[i-1]+f[i];
    int t;
    scanf("%d",&t);
    int L,R;
    while(t--){
    	scanf("%d %d",&L,&R);
    	printf("%lld\n",ff[R]-ff[L-1]);
    }
    return 0;
}
