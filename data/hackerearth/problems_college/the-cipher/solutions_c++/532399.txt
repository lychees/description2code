#include <bits/stdc++.h>
using namespace std;
#define ll long long


int main() {
	ll T;
	cin>>T;
	while(T--) {
		ll N;
		cin>>N;
		ll array[N];
		int i;
		for(i=0;i<N;i++) {
			cin>>array[i];
		}
		for(i=0;i<N;i++) {
			int val = ~array[i];
			cout<<(char)val;
		}
		cout<<endl;
	}	
	return 0;
}
