def divisors(n):
	from math import sqrt
	s = int(sqrt(n + 0.0))
	for i in xrange(1, s + 1):
		if n % i == 0:
			yield i
			yield (n / i)
 
def winning(p, q, s):
	if p % 2 == 0:
		return False
	
	for d in divisors(q):
		if d >= 2 and q / d >= s and not winning(d, q / d, s):
			return True
			
	return False
 
def main():
	T = int(raw_input().strip())
	for t in xrange(T):
		p, q, s = map(int, raw_input().split())
		print "Bernard" if winning(p, q, s) else "Havanese"
 
main()
