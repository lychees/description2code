using namespace std;

string a[10];
bool used[10]={false};
int b[10];
int mn=1000000000;


void solve(int n,int i){
	if(i==n){
		
		string ans="";
		for(int j=0;j<n;j++){
//			cout<<a[b[j]]<<" ";
			if(j==0){
				ans+=a[b[j]];
			}else{
				int k,p,q;
				for(k=0;k<ans.size();k++){
					
					for(p=k,q=0;p<ans.size() && q<a[b[j]].size();p++,q++)
						if(ans[p]!=a[b[j]][q])
							break;
					
					if(p==ans.size()){
						for(p=k,q=0;p<ans.size();)
							p++,q++;
						for(;q<a[b[j]].size();q++)
							ans+=a[b[j]][q];
						break;
					}
				}
				
				if(k==ans.size()){
					ans+=a[b[j]];
				}
			}
		}
//		cout<<endl;
//		cout<<ans<<endl;
		int sz =  ans.size();
		mn = min(mn,sz);
		return;
	}
		
	for(int j=0;j<n;j++){
		if(used[j]==false){
			b[i]=j;
			used[j]=true;
			solve(n,i+1);
			used[j]=false;
		}
	}		
	
}

int main(){
	int n,i,j,x,y;
	cin>>n;
	
	for(i=0;i<n;i++){
		cin>>a[i];
	}
	
	solve(n,0);
	
	cout<<mn;
	return 0;
}
