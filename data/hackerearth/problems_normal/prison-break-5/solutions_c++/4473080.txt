#include <bits/stdc++.h>
using namespace std;

int a[25][25];
int ans,t,n;
int visited[25][25];

int solve(int x,int y){
	if(x==n-1 && y==n-1){
		ans++;
		return 1;
	}
	visited[x][y]=1;
	if(x+1<n && a[x+1][y]==0 && visited[x+1][y]==0) 
		solve(x+1,y);
	if(y+1<n && a[x][y+1]==0 && visited[x][y+1]==0) 
		solve(x,y+1);
	if(x-1>=0 && a[x-1][y]==0 && visited[x-1][y]==0) 
		solve(x-1,y);
	if(y-1>=0 && a[x][y-1]==0 && visited[x][y-1]==0) 
		solve(x,y-1);
	visited[x][y]=0;
	return 1;
}

int main(){
	cin>>t;
	while(t--){
		cin>>n;
		ans=0;
		for(int i=0;i<n;i++){
			for(int j=0;j<n;j++){
				cin>>a[i][j];
				visited[i][j]=0;
			}
		}
		solve(0,0);
		cout<<ans<<endl;
	}
	return 0;
}
