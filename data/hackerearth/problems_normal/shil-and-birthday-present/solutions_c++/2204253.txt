#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int i,j,k,n;
	cin>>n;
	long long int a[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	sort(a,a+n);
	long long int c=1;
	for(i=0;i<n-1;i++)
	{
		if(a[i]!=a[i+1])
		c++;
	}
	long long int ans;
	ans=c*(c-1)/2;
	cout<<ans<<endl;
	return 0;
}
