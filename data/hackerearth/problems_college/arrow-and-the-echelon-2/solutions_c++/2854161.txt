#include <iostream>
using namespace std;
 
int main()
{
 
    int t;
    cin>>t;
    while(t--)
    {
    	long long int i,n;
    	cin>>n;
 
    	if(n%5==0 && n%3==0)
    	{
            for(i=0;i<n;i++)
                cout<<"5";
    	}
    	else if(n%3==0)
    	{
            for(i=0;i<n;i++)
                cout<<"5";
            cout<<endl;
    	}
    	else if(n%5==0)
    	{
            for(i=0;i<n;i++)
                cout<<"3";
            cout<<endl;
    	}
    	else{
 
            int finalx,finaly;
            int x=0,y=0;
            bool flag=true;
            for(;x*5+y*3<=n;x++)
            {
                for(;;y++)
                {
                    if(x*5+y*3 == n)
                    {
                        finalx=x;
                        finaly=y;
                        flag=false;
                        break;
                    }
                    else if(x*5+y*3 > n)
                    {
                        y=0;
                        break;
                    }
 
                }
            }
 
            if(5*x+ 3*y>n &&flag)
            {
                cout<<"-1"<<endl;
            }
            else{
                
                for(i=0;i<3*finaly;i++)cout<<"5";
                for(i=0;i<5*finalx;i++)cout<<"3";
                cout<<endl;
            }
 
    	}
 
 
    }
    return 0;
}
