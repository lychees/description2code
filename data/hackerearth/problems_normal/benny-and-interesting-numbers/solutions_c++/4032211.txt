#include<bits/stdc++.h>
using namespace std;
//zone
int check(long long int x)
{
    for(long long int i = 2; i * i <= x; i++)
    {
        if(x % i == 0)
        {
            int cnt = 0;
            while(x % i == 0)
            {
                x = x / i;
                cnt++;
            }
        if(cnt > 1)
        {
            return 0;
        }
    }
}
return 1;
}
int main()
{

    int t;
    cin >> t;
    while(t--)
    {
        long long int n;
        cin >> n;
        if(check(n))
        {
            cout<<"Yes\n";
        }
        else
            cout<<"No\n";
    }
    return 0;
}
