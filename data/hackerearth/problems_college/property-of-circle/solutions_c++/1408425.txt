#include<bits/stdc++.h>
using namespace std;
int main()
{
	long double h1,k1,h2,k2,r1,r2,dis,difr,t;
	cin>>t;
	while(t--)
	{
		cin>>h1>>k1>>r1>>h2>>k2>>r2;
		dis=pow((h2-h1)*(h2-h1)+(k2-k1)*(k2-k1),0.5);
		difr=abs(r1-r2);
		if(difr>=dis)
		{
			if(r1>r2)
			cout<<"C1CC2"<<endl;
			else if(r2>r1)
			cout<<"C2CC1"<<endl;
			else
			cout<<"C2~C1"<<endl;
		}
		else
		cout<<"NOT SATISFIED"<<endl;
	}
	return 0;
}
