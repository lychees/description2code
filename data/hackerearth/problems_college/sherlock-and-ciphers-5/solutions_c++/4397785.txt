#include <iostream>
#include<math.h>
#include<string.h>
using namespace std;
 
int main()
{
	int n;
	cin>>n;
	char s[2000];
	gets(s);
	while(n--)
	{
			
		//cout<<n;
	
		
	gets(s);
		
		
	
		int l=strlen(s);
		int m=sqrt(l);
		
		for(int i=0;i<m;i++)
		{
			for(int j=i;j<l;j+=m)
			{
		cout<<s[j];
			}
		}
	
		cout<<"\n";
	
			//getchar();
	}
 //   cout << "Hello World!" << endl;
    return 0;
}
