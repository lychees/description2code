
#include <bits/stdc++.h>
using namespace std;
long long z,m,n,i,j;
vector <long long int>a,b;
int main() {
	// your code goes here
	int t,count,k;
	cin>>t;
	while(t--)
	{
		count=0;
		k=0;
		cin>>z>>m>>n;
		for(i=1;i*i<=z;i++)
		{
			if(z%i==0)
			{
				if(i<=m)
				a.push_back(i);
				if(i!=z/i && z/i<=m)
				a.push_back(z/i);
				if(i<=n)
				b.push_back(i);
				if(i!=z/i && z/i<=n)
				b.push_back(z/i);
			}
		}
			sort(a.begin(),a.end());
			sort(b.begin(),b.end());
			j=b.size()-1;
			for(int i=0;i<a.size();i++)
			{
				if(a[i]*b[j]==z)
				count++,j--;
			}
			cout<<count<<endl;
			a.clear();
			b.clear();
		}
	return 0;
}
