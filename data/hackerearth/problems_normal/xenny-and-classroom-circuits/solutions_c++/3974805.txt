#include <iostream>
using namespace std;

int main()
{
    int t; cin>>t; 
    while(t--){
    	int N; cin>>N; N*=2;
    	string A; cin>>A;
    	int cnt = 0; 
    	for(int i =0; i<N; i+=2){if(A[i]=='G')cnt++; if(A[i+1]=='B')cnt++;}
    	if(N-cnt < cnt) cnt = N - cnt; 
    	cout<<cnt/2<<endl;
    }
    return 0;
}
