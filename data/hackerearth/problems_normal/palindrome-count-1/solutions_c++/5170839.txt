int countPalindrome(string str)
{
    int l = len(str);
    int i,j,k,count=0;
    for(i=0;str[i];i++)
    {
        count++;
        k=i-1;j=i+1;
        while(k>=0 && j<l && str[k]==str[j])
        {
            ++count;
            k--;j++;
        }
        k=i;j=i+1;
        while(k>=0 && j < l && str[k]==str[j])
        {
            ++count;
            k--;j++;
        }
    }
    return count;
}
