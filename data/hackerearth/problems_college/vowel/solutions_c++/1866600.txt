#include<bits/stdc++.h>

using namespace std;

int main()
{
	int T;
	cin>>T;
	string s;
	int flag;
	while(T--){
		cin>>s;
		flag=0;
		for(int i=0;i<s.length();i++){
			if(s[i]=='A'||s[i]=='E'||s[i]=='I'||s[i]=='O'||s[i]=='U'||s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u'){
				cout<<s[i];
				flag=1;
			}
		}
		if(flag==0){
			cout<<"No";
		}
		cout<<endl;
	}
	return 0;
}
