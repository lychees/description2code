#include <iostream>
using namespace std;
long int ncr(int n,int r){
	long result=1;
	for(int i = 1 ; i <= r; i++){
		result = result*(n - r +i)/i;
	}
	return result;
}
int main()
{
    int t;
    cin >>t;
    for(int i=0;i<t;i++)
    {	int total =0;
    	int n;
    	cin >> n;
    	for(int p = 0;p <= int((n+1)/2);p++){
    		total = total + ncr(n-p+1,p);
    	}
    	long result = total*total;
   		cout << result <<endl;
   	}
}
