#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ll int t,n,k;
    cin>>t;
    while(t--)
    {
        cin>>n;
        pair<pair<ll,ll >,double> apa[n];
        ll int ar[n],br[n],cr[n];
        for(int i=0; i<n; i++)
        {
            scanf("%lld%lld%lld",&ar[i],&br[i],&cr[i]);
            k=__gcd(ar[i],br[i]);
            if(k==0)
                k=1719;
            apa[i].first.first=ar[i]/k;
            apa[i].first.second=br[i]/k;
            apa[i].second=(double)cr[i]/k;
        }
        ll int ma=0,coun=1;
        sort(apa,apa+n);
        for(int i=0; i<n-1; i++)
        {
            if((apa[i].first.first==apa[i+1].first.first)&&(apa[i].first.second==apa[i+1].first.second))
            {
                if((apa[i].second!=apa[i+1].second))
                {
                    coun++;
                }
            }
            else coun=1;
            if(coun>ma)
                ma=coun;

        }
        if(n==1)
            cout<<"1\n";
        else
            printf("%lld\n",ma);

    }
}
