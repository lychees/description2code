#include <iostream>
using namespace std;
int t, gp, gv, gc, n, p[20], v[20], c[20];
bool checker(int gp, int gv, int gc, int i){
	if(!gp && !gv && !gc) return true;
	if(gp<0 || gv<0 || gc<0) return false;
	if(i>=n) return false;
	return (checker(gp, gv, gc, i+1) || checker(gp-p[i], gv-v[i], gc-c[i], i+1));
}
int main()
{
	cin>>t;
	for(int q=1; q<=t; q++){
		cin>>gp>>gv>>gc>>n;
		cout<<"Case #"<<q<<": ";
		//cout<<" gp= "<<gp<<" gv= "<<gv<<" gc= "<<gc<<" n= "<<n<<endl;
		for(int j=0; j<n; j++){
			cin>>p[j]>>v[j]>>c[j];
		}
		if (checker(gp, gv, gc, 0)) cout<<"yes"<<endl;
		else cout<<"no"<<endl;
	}
	return 0;
}
