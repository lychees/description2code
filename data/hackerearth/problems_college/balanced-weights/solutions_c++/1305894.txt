#include <iostream>
using namespace std;
long gcd(long a,long b)
{
	if(b==0)
	return a;
	else if(b>a)
	return gcd(b,a);
	else
	return gcd(a%b,b);
	
}
int main()
{
	long n,s,g;
	long a;
scanf("%ld",&n);
while(n--)
{
	scanf("%ld",&s);
	scanf("%ld",&g);
	
	for(long i=0;i<s-1;i++)
	{
		scanf("%ld",&a);
		g=gcd(g,a);
		
		
	}
	printf("%ld\n",g*s);
}
    return 0;

}
