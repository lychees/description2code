#include <stdio.h>
int factorial(int n)
{
	if(n==0||n==1)
	return 1;
	else
	return n*factorial(n-1);
}
int main()
{
    int t,n,m,A[10000],i,j,p,q;
    scanf("%d",&t);
    for(i=1;i<=t;i++)
    {
    	scanf("%d%d",&n,&m);
    	p=m-1;
    	q=n+p;
    	A[i]=factorial(q)/(factorial(p)*factorial(n));
    	
    	
    
    }
    for(i=1;i<=t;i++)
    {
    	printf("%d\n",A[i]%1000000007);
    }
}
