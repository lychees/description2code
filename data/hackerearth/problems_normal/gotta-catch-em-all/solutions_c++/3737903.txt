ll power(ll a,ll b)
{
	if(!b)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2);
	temp=(temp*temp);
	if(b&1)
		temp=(temp*a);
	return temp;
}
int main()
    {

		cin>>n;
		vi v(n);
		for(i=0;i<n;i++)

			cin>>v[i];
		}
		sort(v.begin(),v.end());
		for(i=n-1;i>=0;i--)
		{
			ans=max(ans,v[i]+n-i);
		}
		ans++;
		cout<<ans;
		return 0;
    }
