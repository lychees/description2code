#include <iostream>
#include <vector>
#include <stdio.h>

using namespace std;

int main()
{
	
	int t, n, i, v, count;
	
	scanf("%d", &t );
	
	while( t-- ) {
		
		scanf("%d", &n );
		vector<bool> B( 10001, true );
		count = 0;
		
		for( i = 0;i < n;i++ ) {
			scanf("%d", &v );
			if( B[ v ] ) {
				count++;
				B[ v ] = false;
			}	
		}
		
		printf("%d\n", count );
	}
	
	return 0;
}
