n = int(raw_input())

calling = raw_input().split(" ")
ideal = raw_input().split(" ")
count = 0
x = 0
while n > 0:
	count += 1
	if calling[x] == ideal[x]:
		calling.pop(0)
		ideal.pop(0)
		n -= 1
	else:
		t = calling.pop(0)
		calling.append(t)

print count
