#include <iostream>
#include <sstream>
#include <iomanip>
using namespace std;

std::string int_to_hex(int x){
	stringstream stream;
	stream<<setfill('0')<<setw(2)
		<<hex<<x;
	string str2=stream.str();
	
return str2;	
}


void mars_lang(string &str){
	
	if(str.at(0)>90){
		str.at(0)-=12;
	}
	if(str.at(1)>90){
		str.at(1)-=12;
	}
}



int main()
{
    int cases;
    cin>>cases;
    int n=1;
    while(cases--){
    	int r,g,b;
    	cin>>r>>g>>b;
    	std::string str;
    
    	cout<<"Case "<<n++<<": ";
    	str=int_to_hex(r);
     	mars_lang(str);
    	cout<<str<<" ";
    	
    	str=int_to_hex(g);
    	mars_lang(str);
    	cout<<str<<" ";
    	
    	str=int_to_hex(b);
    	mars_lang(str);
    	cout<<str<<endl;
   
    }
    return 0;
}
