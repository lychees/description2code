#include <iostream>
#include <vector>
#include <algorithm>

long long linear_search(std::vector<long long> a, long long X) {
  for (const auto c : a) {
    if (c>X) return c;
  }
  return -1;
}


int main() {
  
  long long n, l, r, Q;
  std::cin >> n >> l >> r >> Q;

  std::vector<long long> a;
  for (int i = 0; i<n; i++) {
    long long aux;
    std::cin >> aux;
    if (i<l) continue;
    if (i>r) continue;
    a.push_back(aux);
  }

  std::sort(std::begin(a), std::end(a));

  while (Q--) {
    long long X;
    std::cin >> X;
    std::cout << linear_search(a, X) << '\n';
  }
  
}
