from math import factorial
t = int(raw_input())
dic = {}
for i in range(t):
    input = int(raw_input())
    if input in dic.keys():
        print dic[input]
        continue
    fact = factorial(input)
    num = fact % 100000
    dic[input] = '{0:0{width}}'.format(num, width=5)
    print dic[input]
