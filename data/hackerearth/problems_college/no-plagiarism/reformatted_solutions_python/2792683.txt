a = raw_input()
b = raw_input()
len_a = len(a)
len_b = len(b)
if a == b:
    c = 0
else:
    d = [[0 for i in xrange(len_b+1)] for j in xrange(len_a+1)]
    for i in xrange(len_a+1):
        d[i][0] = i
    for i in xrange(len_b+1):
        d[0][i] = i
    for j in xrange(1, len_b+1):
        for i in xrange(1, len_a+1):
            if a[i-1] == b[j-1]:
                d[i][j] = d[i-1][j-1]
            else:
                d[i][j] = min(d[i-1][j], d[i][j-1])+1
    c = d[len_a][len_b]
print c
