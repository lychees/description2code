#include <iostream>
typedef long long int ll;
typedef long int lg;
typedef unsigned long long int ull;
using namespace std;

int main()
{
    ll arr[20],n,i;
    cin>>arr[1]>>arr[2]>>n;
    if(n<3)
    {
    	cout<<arr[n]<<endl;
    }
    else
    {
    	for(i=3;i<=n;i++)
    	{
    		arr[i]=arr[i-1]+arr[i-2];
    	}
    	cout<<arr[n]<<endl;
    }
    return 0;
}
