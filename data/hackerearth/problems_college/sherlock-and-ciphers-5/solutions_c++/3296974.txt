#include <bits/stdc++.h>
using namespace std;
int main() {
    int n;
    //cin >> n;
    char str[200];
    gets(str);
    n = atoi(str);
    while(n--) {
        gets(str);
        int l = strlen(str);
        int s = sqrt(l);
        int i,j,k;
        char a[s+1][s+1];
        k = 0;
        for(i = 0; i < s; i++) {
            for(j = 0; j < s; j++) {
                a[i][j] = str[k];
                k++;
            }
        }
        for(i = 0; i < s; i++) {
            for(j = 0; j < s; j++) {
                cout<<a[j][i];
            }
        }
        for(i = 0; i < s; i++) {
            for(j = 0; j < s; j++) {
                a[i][j] = '\0';
            }
        }
        cout<<endl;
    }
    return 0;
}
