#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
int t;
cin>>t;
while(t--){
int n,m;
cin>>n>>m;
vector<int>v1(n),v2(n);
for(int i=0;i<n;i++){cin>>v1[i];}
while(m--){
if(v1[1]==0){v2[0]=0;}
else{v2[0]=1;}
if(v1[n-2]==0){v2[n-1]=0;}
else{v2[n-1]=1;}
for(int i=1;i<=n-2;i++){
if(v1[i-1]==0||v1[i+1]==0){v2[i]=0;}
else{v2[i]=1;}
}
v1=v2;
if(std::find(v1.begin(),v1.end(),0)==v1.end()){break;}
if(find(v1.begin(),v1.end(),1)==v1.end()){break;}
}
for(int i=0;i<n;i++){cout<<v1[i]<<" ";}
cout<<endl;
}
return 0;
}
