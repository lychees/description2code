/************************************************************************************/

int main()
{
   ios::sync_with_stdio(false);
   int n;
   cin >> n;
   map<long long, int> cnt;
   for (int i = 0; i < n; i++) {
   	    int x;
   	    cin >> x;
   	    cnt[x]++;
   }
   long long ans = -1;
   for (map<long long,int>::iterator it = cnt.begin(); it != cnt.end(); it++) {
   	    if (it->second == 1) {
   	    	ans = max(ans, it->first);
   	    }
   }
   cout << ans << endl;
   return 0;
}
