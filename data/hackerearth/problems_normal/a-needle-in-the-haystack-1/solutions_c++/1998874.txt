#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string p,txt;
        cin>>p>>txt;

        int lp = p.length();
        int lh = txt.length();

        int p_alpha[26] = {0}; 
        int flag = 0;
        for(int i=0;i<lp;i++)
            p_alpha[p[i]-'a']++;
            
        int t_alpha[26] = {0};

        for(int i=0;i<lp;i++)
        {
            t_alpha[txt[i]-'a']++;
        }

        for(int i=lp;i<lh;i++)
        {
            int found = 1;
            for(int k=0;k<26;k++)
            {
                if(p_alpha[k] != t_alpha[k])
                {
                    found = 0;
                    break;
                }
            }
            if(found)
            {
                flag=1;
                break;
            }
            
            t_alpha[txt[i-lp]-'a']--;
            t_alpha[txt[i]-'a']++;
        }
        int found = 1;
        for(int k=0;k<26 && !flag;k++)
            {
                if(p_alpha[k] != t_alpha[k])
                {
                    found = 0;break;
                }
            }
            if(found)
             flag=1;
        if(!flag)
        cout<<"NO"<<endl;
        else
        cout<<"YES"<<endl;
    }
    return 0;
}
