const int N = 100100;

int cnt[N][26];

int Get(int x, int l, int r) {
	if (l == 0)
	    return cnt[r][x];
	return cnt[r][x] - cnt[l - 1][x];
}

int main()
{
   ios::sync_with_stdio(false);
   int n;
   cin >> n;
   string s;
   cin >> s;
   for (int i = 0; i < s.length(); i++) {
   	    cnt[i][ s[i] - 'a' ] = 1;
   }
   for (int i = 1; i < s.length(); i++) {
   	    for (int j = 0; j < 26; j++) {
   	    	 cnt[i][j] += cnt[i - 1][j];
   	    }
   }
   
   int m;
   cin >> m;
   while (m--) {
   	      int l, r;
   	      cin >> l >> r;
   	      int odd = 0, even = 0, len = r - l + 1;
   	      for (int i = 0; i < 26; i++) {
   	      	   int C = Get(i, l - 1, r - 1);
   	      	   if (C % 2) odd++;
   	      	   else if (C > 0) even++;
   	      }
   	      //cout << odd << " " << even << endl;
   	      if (len % 2) {
   	      	  if (odd == 1 && len > 1)
   	      	      cout << "YES" << endl;
   	      	  else
   	      	      cout << "NO" << endl;
   	      } else {
   	      	  if (odd <= 2 && len > 0)
   	      	      cout << "YES" << endl;
   	      	  else
   	      	      cout << "NO" << endl;
   	      }
   }
   return 0;
}
