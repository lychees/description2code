#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
main()
{
int tc,i=1;
int count[2]={0};
char str[1000]={""};
gets(str);
cin>>tc;
while(i<=tc+1)
{
    char test[1000]={""};
    gets(test);
    if(strlen(test)!=strlen(str))
    {i++;continue;}
    else if(test[0]==str[0])
    {
        count[0]++;
        i++;
        continue;
    }
    else if(test[0]==char(str[0]+3))
    {
        count[1]++;
        i++;
        continue;
    }
}
cout<<count[0]<<" "<<count[1];
}
