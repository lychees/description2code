LIM = 1000000
isPrime = [True for i in xrange(LIM + 1)]
hasMinP = [0 for i in xrange(LIM + 1)]


def sieve():
    for p in xrange(2, LIM + 1):
        if not isPrime[p]:
            continue
        else:
            hasMinP[p] = 1
            for j in xrange(p + p, LIM + 1, p):
                if isPrime[j]:
                    hasMinP[p] += 1
                isPrime[j] = False


sieve()
tcase = int(raw_input())
for i in xrange(tcase):
    n = int(raw_input())
    print hasMinP[n]
