from sys import stdin
t = int(stdin.readline())
for _ in xrange(t):
    a, b = map(int, stdin.readline().split())
    ans = 0
    if a <= 7 and b >= 7:
        ans += 8
    if a <= 13 and b >= 13:
        ans += 144
    print ans
