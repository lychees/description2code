
#include <cstdio>
#include <string>
#include <iostream>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;
#define ll long long

ll t, n, m;
vector<ll> v;

int main(){
    //freopen("test.txt","r",stdin);
    for(cin >> t; t--; ){
        cin >> n >> m;
        if( m == 1 ){
            cout << n << endl;
            continue;
        }
        v.clear();
        ll ans = 0;
        ll q = 1;
        while(true){
            v.push_back(q);
            if( 1e18/m <= q ) break;
            q *= m;
        }
        for(int i = v.size() - 1; i >= 0; i--){
            ans += n / v[i];
            n %= v[i];
        }
        cout << ans << endl;
    }

	return 0;
}
