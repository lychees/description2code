struct matc{
	int st,en,p;
};
matc a[55];
bool cmp(matc x,matc y){
	return x.en<y.en;
}
int main(){
	int t,i,n,j;
	cin>>t;
	while(t--){
		cin>>n;
		for(i=1;i<=n;i++)
		cin>>a[i].st>>a[i].en>>a[i].p;
		sort(a+1,a+n+1,cmp);
		int ans[n+1];
		int tmp=0;
		for(i=1;i<=n;i++){
			ans[i]=a[i].p;
			for(j=i-1;j>0;j--){
				if(a[i].st>=a[j].en)
				ans[i]=max(ans[i],ans[j]+a[i].p);
			}
			tmp=max(tmp,ans[i]);
		}
		double res=tmp/100.0;
		printf("%.2lf\n",res);
	}
	return 0;
}
