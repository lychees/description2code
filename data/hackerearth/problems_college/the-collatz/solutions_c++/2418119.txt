#include <iostream>
using namespace std;

int main()
{
    int t;cin>>t;
    while(t--)
       {
       	int a;//cout<<"a";
       	long long int steps=0,even=0,odd=0;
       	cin>>a;
       	while(a!=1)
        {	if(a%2==0) 
       	{
       		even+=a;//cout<<"even"<<even<<endl;
       	  a=a/2;steps++;//cout<<"s"<<steps<<endl;
       	 
       	}
       	else
       	{
       	 odd=odd+a;//cout<<"odd"<<odd<<endl;
       	 a=(3*a)+1;
       	 steps++;//cout<<"s"<<steps<<endl;
       	}
        }
        odd+=1;
        if((odd%steps)>(even%steps))cout<<"Odd Rules\n";
        else if((odd%steps)==(even%steps))cout<<"Tie\n";
        else cout<<"Even Rules\n";
       }
    return 0;
}
