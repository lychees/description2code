#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<int,int> ii;
typedef vector<int> vi;
typedef vector<ii> vii;

const int  MOD=1000000007;
const int  INF= int(1e9);
const int  MAX=50;
int n,k,res;
string s;
int cnt[MAX][MAX];
int main()
{
    ios_base::sync_with_stdio(false);
    cin>>n>>k;
    for(int i=0;i<n;i++) {
        char dice;
        for(int j=0;j<6;j++) {
            cin>>dice;
            cnt[i][dice-'a']++;
        }
    }
    cin>>s;
    for(int mask=0;mask<(1<<n);mask++) {
        if(__builtin_popcount(mask) == k) {
            vi cube;
            for(int i=0;i<n;i++) {
                if( mask & (1<<i)) {
                    cube.push_back(i);
                }
            }
            sort(cube.begin(),cube.end());
            do {
                ll ways=1;
                for(int i=0;i<k;i++) {
                    int req=s[i]-'a';
                    int cubeNo=cube[i];
                    if(cnt[cubeNo][req]>0) {
                        ways*=cnt[cubeNo][req];
                        ways%=MOD;
                    } else {
                        ways=0;
                    }
                }
                res+=ways;
                res%=MOD;
            }while(next_permutation(cube.begin(),cube.end()));


        }

    }

    cout<<res<<"\n";
    return 0;

}
