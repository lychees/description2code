#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int tc;
	cin>>tc;
	if(tc>10 || tc<1)	
		cout<<"Invalid Test\n";
	else
	{
		for(long int i1=0;i1<tc;i1++)
		{
			string p;
			cin>>p;
			
			long int up=0,lo=0;
			for(long int i=0;i<p.length();i++)
				if(p[i]>=97 && p[i]<=122)
					lo++;
				else if(p[i]>=65 && p[i]<=90)
					up++;
			
			if((lo==0 && up==0) || p.length()>100 || p.length()<1)
				cout<<"Invalid Input\n";
			else
			{
				cout<<min(lo,up)<<endl;
			}
		}
	}
	return 0;
}
