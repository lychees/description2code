#include <iostream>
#include <vector>
using namespace std;

void get_input(vector<long> & v, int n){
	for(int i=0; i<n; i++){
    	long k;
    	cin >> k;
    	v.push_back(k);
    }
}

long get(long a[], int i){
	if(i<0){
		return 0;
	}else{
		return a[i];
	}
}

int main()
{
    int n,q;
    vector<long> a,b;
    cin >> n;
    cin >> q;
    get_input(a,n);
    get_input(b,n);
    long sum1[n],sum2[n];
    sum1[0] = a[0];
    sum2[0] = b[0];
    for(int i=1; i<n; i++){
    	if(i%2 == 0){
    		sum1[i] = sum1[i-1] + a[i];
    		sum2[i] = sum2[i-1] + b[i];
    	}else{
    		sum1[i] = sum1[i-1] + b[i];
    		sum2[i] = sum2[i-1] + a[i];
    	}
    }
    for(int i=0; i<q; i++){
    	int k,l,r;
    	cin >> k;
    	cin >> l;
    	cin >> r;
    	if((k+l) % 2 == 0){
    		cout << get(sum1, r-1) - get(sum1,l-2) << endl;
    	}else{
    		cout << get(sum2,r-1) - get(sum2,l-2) << endl;
    	}
    }
    return 0;
}
