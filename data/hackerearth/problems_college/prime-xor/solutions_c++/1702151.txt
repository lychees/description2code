#include <iostream>
using namespace std;

int dp[1000005] = {0};

int main()
{
	dp[0] = dp[1] = 1;
	for (int i = 2; i < 1000004; i++) {
		for (int j = 2 * i; j < 1000004; j += i) {
			dp[j] = 1;
		}
	}
    int t;
    cin >> t;
    while (t--) {
    	int a, b;
    	cin >> a >> b;	    	
    		int c = 0;
    		if (2 >= a && 2 <= b) {
    			for (int i = a; i <= b; i++) {
    				if (dp[i] == 0) {
    					c++;
    				}
    			}
    			c--;
    			cout << c << endl;
    		} else {
    			cout << 0 << endl;
    			
    		}
    	
    }
    return 0;
}
