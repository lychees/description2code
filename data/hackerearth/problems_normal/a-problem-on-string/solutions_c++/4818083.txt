#include<bits/stdc++.h>
#define MAX 10000000
#define mpr(x,y) make_pair(x,y)
#define pb(x) push_back(x)
#define MOD 1000000007ll
#define INF 9223372036854775807
#define S(x) scanf("%lld",&x)
#define Ss(x) scanf("%s",x)
#define Ps(x) printf("%s\n",x)
#define P(x) printf("%lld\n",x)
#define P_(x) printf("%lld ",x)
#define Ps_(x) printf("%s ",x)
#define nL printf("\n")
using namespace std;
typedef long long ll;
ll rem[1000000][26];
ll ans[26][26];
ll n,q;
string s;
void pre(){
	for(ll i=0; i<s.length(); i++){
		if(i>0){
			for(ll j=0; j<26; j++){
				rem[i][j] = rem[i-1][j];
			}
		}
		rem[i][s[i]-'a']++;
	}
}
void populateAns(){
	for(ll i=1; i<n; i++){
		for(ll j=0; j<26; j++){
			ans[j][s[i]-'a']+=rem[i-1][j];
		}
	}
}
int main(){
	char x,y;
	S(n);
	cin>>s;
	pre();
	
	populateAns();
	
	S(q);
	while(q--){
		cin>>x>>y;
		if(x!=y)
		P(ans[x-'a'][y-'a']+ans[y-'a'][x-'a']);
		else
		P(ans[x-'a'][y-'a']);
	}
	
	return 0;
}
