#define maxn 100005
int arr[105][maxn];
int sum[105];
int mat[105];
int main()
{
	int n,i,j;
	si(n);
	int v,mm=0;
	sum[0]=0;
	for(i=1;i<=n;i++){
		si(mat[i]);
		sum[i]=sum[i-1]+mat[i];
	}
	for(i=1;i<=n;i++){
		for(j=0;j<maxn;j++) arr[i][j]=arr[i-1][j];
		for(j=1;j<=i;j++){
			int x=sum[i]-sum[j-1];
			arr[i][x]=max(arr[i][x],1+arr[j-1][x]);
			mm=max(arr[i][x],mm);
		}
	}
	vector<int> vv;
	for(i=0;i<maxn;i++) if(arr[n][i]==mm) vv.pb(i);
	printf("%d\n",mm);
	for(i=0;i<vv.size();i++) printf("%d ",vv[i]); printf("\n");
	return 0;	
}
