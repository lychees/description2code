#include <iostream>
using namespace std;

int main()
{
    int t,k,n,i;
    long long m,xr;
    
    cin>>t;
    
    for(k=0;k<t;k++)
    {
    	cin>>n;
    	xr=0;
    
    	for(i=0;i<n;i++)
    	{
    		cin>>m;
    		
    		m=m%5;
    		xr=xr^m;
    	}
    	
    	if(xr)
    	cout<<"Aniruddha\n";
    	else
    	cout<<"Andrew\n";
    	
    }
    return 0;
}
