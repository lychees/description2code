#include <iostream>
#include <cstring>
using namespace std;
int dashes[10]={6,2,5,5,4,5,6,3,7,6};
int toInt(char c)
{
	return (c-'0');
}
int main()
{
    char str[100];
    
    cin>>str;
    int count=0;
    for(int i=0;i<strlen(str);i++)
    {
    	count+=dashes[toInt(str[i])];
    }
    cout<<count;
    return 0;
}
