#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
        while(t--)
	{
	    double x1,x2,y1,y2,r;
	    cin>>x1>>y1>>x2>>y2;
	    cin>>r;
        double d = sqrt((x2-x1)*(x2-x1) + (y2-y1)*(y2-y1) );
        double D = x1*y2-x2*y1;
        double delta = r*r*d*d-D*D;
        if(delta < 0)
            cout<<"SAFE\n";
        else
            if(delta == 0)
                cout<<"JUST MISSED\n";
            else
                cout<<"REPLANNING\n";
	}
    return 0;
}
