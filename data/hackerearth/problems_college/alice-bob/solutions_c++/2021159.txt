#include<cstdio>
#include<algorithm>
using namespace std;

int num[105];

int gcd(int a, int b)
{
	return b ? gcd(b, a % b) : a;
}

int main(void)
{
	int n, temp = 0;
	scanf("%d", &n);
	for (int i = 0; i < n; ++i)
	{
		scanf("%d", &num[i]);
		temp = max(temp, num[i]);
	}
	int maxn = temp;
	for (int i = 0; i < n - 1; ++i)
		for (int j = i + 1; j < n; ++j)
		{
			int k = gcd(num[i], num[j]);
			temp = gcd(temp, k); 
		}
	int movetimes = maxn / temp - n; 
	puts(movetimes & 1 ? "Alice" : "Bob");
	return 0;
}
