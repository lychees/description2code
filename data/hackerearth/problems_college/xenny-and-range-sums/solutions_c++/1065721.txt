#include <bits/stdc++.h>
using namespace std;
#define max 1000003
long long int a[max],b[max];
int main() 
{
	long long int n,q,l,r,i;
	cin>>n>>q;
	//long long int a[n],b[n+2];
	b[0] = 0;
	for(i = 1; i <= n; i++) {
		cin>>a[i];
		b[i]=a[i]+b[i-1];
	}
	while(q--) {
		cin>>l>>r;
		cout<<b[r]-b[l-1]<<" ";
		r--;
		cout<<a[r+1]-(b[r]-b[l-1])<<endl;
	}
	return 0;
}
