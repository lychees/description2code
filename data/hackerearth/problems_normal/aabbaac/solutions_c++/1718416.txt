#include <iostream>
#include <cstdio>
#include <string>

#define MAXN 10000

using namespace std;

string S[MAXN];
int N, M;

void read()
{
	cin >> N >> M;
	for(int i = 0; i < N; i++)
		cin >> S[i];
}

long long sz[MAXN];
char rec(int n, long long k)
{
	/*
	for(int i = 0; i < n; i++) printf(" ");
	printf("%d %d\n", n, k);
	*/

	if(n == 0)
		return S[0][k];

	long long currSize = sz[n] - (long long)S[n].size();
	
	//cout << n << " " << k << endl;

	if(currSize <= k)
		return S[n][k - currSize];

	if(k >= currSize / 2)
		return rec(n - 1, currSize / 2 - (k - currSize / 2) - 1);
	else
		return rec(n - 1, k);
}

void solve()
{
	sz[0] = S[0].size();
	for(int i = 1; i < N; i++)
		sz[i] = (long long)sz[i - 1] * 2 + (long long)S[i].size();

	/*for(int i = 0; i < N; i++)
		printf("%d, ", sz[i]);
	printf("\n");
	*/

	for(int i = 0; i < M; i++)
	{
		long long x;
		cin >> x;

		cout << rec(N - 1, x);
	}

	cout << endl;
}

int main()
{
	int T;
	cin >> T;
	for(int i = 0; i < T; i++)
	{
		read();
		solve();
	}
	return 0;
}
