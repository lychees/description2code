import sys

t = input()

while t > 0:
    k = []
    s = sys.stdin.readline().strip().split("_")
    if s[0] != "":
        k.extend(list(s[0]))
    if s[1] != "":
        k.extend(list(s[1]))
    k = map(int, k)
    s = sum(k)
    s = s % 9
    if s == 0:
        print "0 or 9"
    else:
        print 9-s
    t = t-1
