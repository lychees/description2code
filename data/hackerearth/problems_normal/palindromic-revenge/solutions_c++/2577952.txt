#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main() {
	int t;
	cin >> t;
	while(t--) {
		vector <int> v;
		string s;
		int count = 0;
		cin >> s;
		for(int i=0; s[i]!='\0'; i++) {
			v.push_back(s[i]);
		}
		int len = v.size();
		for(int i=0,j=len-1; i<j; i++,j--) {
			if(v[i] != v[j]) {
				count += abs(v[i] - v[j]);
			}
		}
		cout << count << endl;
	}
	return 0;
}
