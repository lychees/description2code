#include<bits/stdc++.h>
#include<algorithm>
using namespace std;
int main()
{
          long long int n,i,j;
          cin>>n;
          char str[n][100];
          char temp[100];
          for(i=0;i<n;i++)
                    cin>>str[i];
          for(i=0;i<n;i++)
           {
           for(j=i+1;j<n;j++)
           {
            if(strcmp(str[i],str[j])>0)
            {
            strcpy(temp,str[i]);
            strcpy(str[i],str[j]);
            strcpy(str[j],temp);
            }
           }
           }
          long long int c=1;
           for(i=1;i<n;i++)
           {
                     if(strcmp(str[i],str[i-1])==0)
                     {
                               c++;
                     }
                     else
                     {
                               cout<<str[i-1]<<" "<<c<<endl;
                               c=1;
                     }
           }
           cout<<str[n-1]<<" "<<c<<endl;
           
          return 0;
}
