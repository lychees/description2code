#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
int a[1000006];
int b[1000006];
void sieve(){
	a[0]=0;
	a[1]=0;
for (int i = 2; i < 100001; i++) {
		if (a[i] == 1) { 
			for (int j = 1; i*j <= 100000; j++) {
				if (j > 1&&a[i*j]==1)
				a[i*j] = i;
			}
		}	
	}
}
int main()
{
	for (int i = 0; i <= 100000; i++) 
		a[i] = 1;
	sieve();
    int t;

    	if(a[i]==1)
    	b[i]=(b[i-1])%mod;
    	else
    	b[i]=(b[i-1]+a[i])%mod;
    }
    while(t--){
    	int l,r,sum=0,j=2;

    //	for(int i=2;i<=8;i++)
