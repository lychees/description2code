#include <iostream>
#include<cstring>
using namespace std;
int pal(string s){
	long long int l=s.size();
	long long int i;
	for(i=0;i<l/2;i++){
		if(s[i]!=s[l-1-i])
		return 0;
	}
	return 1;
}
int main()
{
    int t;
    cin >> t;
    while(t--){
    	string s1,s2,s3,s4;
    	cin >>s1;
    	cin >>s2;
    	s3=s1+s2;
    	s4=s2+s1;
    	if((pal(s3))||(pal(s4)))
    	cout << "YES\n";
    	else
    	cout << "NO\n";
    }
}
