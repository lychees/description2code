freq=[0]*26
let=raw_input()
maxi=0
for j in let:
	i=ord(j)
	freq[i-97]+=1
	if freq[i-97]>freq[maxi]:
		maxi=i-97
mr=chr(maxi+97)
if mr in 'head rush':
	if mr in 'cubing club':
		print 'Thanks both'
	else :
		print 'Thanks head rush'
elif mr in 'cubing club':
	print 'Thanks cubing club'
else :
	print 'none'
