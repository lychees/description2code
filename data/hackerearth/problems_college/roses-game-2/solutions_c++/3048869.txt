#include<iostream>
using namespace std;
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        int N;
        cin>>N;
        int a[N],v[N];
        int i;
        for(i=0;i<N;i++)
        {
            cin>>a[i];
            v[i]=0;
        }
        i=0;
        int flag=0;
        if(a[0]==-1)
            flag=1;
        while(a[i]!=-1)
        {
 
            if(v[i]==0)
            {
                v[i]=1;
                i = a[i];
            }
            if(v[i]==1)
            {
                flag=0;
                break;
            }
            if(a[i]==-1){
                flag=1;
                break;
            }
        }
        if(flag==0)
            cout<<"Lose"<<endl;
        else
            cout<<"Win"<<endl;
    }
}
