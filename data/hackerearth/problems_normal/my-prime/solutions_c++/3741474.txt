#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int n, flag;
	vector<int> arr;
	cin >> n;
	for (int temp; cin>>temp;)
		arr.push_back(temp);
	for (int x : arr) {
		for (int y : arr) {
			flag = 0;
			if (x%y==0 && x!=y)
				break;
			else
				flag++;
		}
		if (flag)
			cout << x << " ";
	}
	
	
    return 0;
}
