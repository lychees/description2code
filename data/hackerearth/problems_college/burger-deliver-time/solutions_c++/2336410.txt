#include<iostream>
using namespace std;
int main()
{
	int n,i,j,tmp;
	scanf("%d",&n);
	int t[n],d[n],s[n],c[n];
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&t[i],&d[i]);
		s[i]=t[i]+d[i];
		c[i]=s[i];
	}
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(s[j]<s[i])
			{
				tmp=s[j];
				s[j]=s[i];
				s[i]=tmp;
			}
		}
	}
	printf("\n");
	for(i=0;i<n;i++)
	{
		//printf("%d ",c[i]);
		for(j=0;j<n;j++)
		{
			if(c[j]==s[i])
			{
				printf("%d ",j+1);
				c[j]=-1;
			}
			
		}
	}
return 0;
}
