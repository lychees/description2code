#include <iostream>
#include <algorithm>

using namespace std;

bool cmp(int x, int y)
{
	return (x > y);
}

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	
	int T, N, K, A[100001];
	
	cin >> T;
	
	while (T--)
	{
		cin >> N >> K;
		
		for (int i = 0; i < N; i++)
			cin >> A[i];
			
		sort(A, A+N, cmp);
		
		long long int ans = 0;
		
		for (int i = 0; i < N && K; i++, K--)
		{
			ans += A[i];
		}
		
		cout << ans << endl;
	}
    return 0;
}
