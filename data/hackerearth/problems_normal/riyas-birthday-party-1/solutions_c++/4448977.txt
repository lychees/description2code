#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define lli long long int
long int m=1e9+7;
lli mexp(lli a,lli b)
{
	if(b==0)
	{
		return 1;
	}
	lli t=mexp(a,b/2);
	if(b%2==0)
	{
		return ((t%m)*(t%m))%m;
	}
	else
	{
		return ((t%m*t%m)%m*(a%m))%m;
	}
}
int main()
{
	ios_base::sync_with_stdio(false);
    lli t,n,ans;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	ans=mexp(n,2);
    	ans=2*ans;
    	ans%=m;
    	ans=(ans%m-n%m+m)%m;
    	cout<<ans<<endl;
    }
    return 0;
}
