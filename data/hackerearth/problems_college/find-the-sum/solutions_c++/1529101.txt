long long int che(long long int k)
{
	long long int t;
	long double r;
	long long int f,ct=-1;
	do
   {
   	k--;
   	r=(((sqrt(8*k+1))-1)/2);
   	f=(floor(r));
    ct++;
   }while((r-f)!=0);
   if(f%2==0)
   {
   	t=(r/2)*(r/2);
   	if(ct%2!=0)
   	t=t+(r+1);	
    }
   else
   {
    t=((r+1)/2)*((r+1)/2);
    if(ct%2!=0)
    t=t+(r+1);
   }
   return t;
}
