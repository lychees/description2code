#include<bits/stdc++.h>
using namespace std;

char s[10005];
int main() {
	long int i,n,j,t,u,v;
	cin >> t;
	while(t--) {
		cin >> s;
		u=v=0;
        for(i=0;i<strlen(s);i++) {
        	if(s[i]=='U')
        		u++;
        	else
        		v++;
        }
        if(!(abs(u-v)==1 || v==u )) { 
        	cout << "-1" << endl;
        	continue;
        }

        
        if(u==v) {
        	u=v=0;
        	for(i=0;i<strlen(s);i=i+2) {
        		if(s[i]!='U')
        			u++;
        		else
        			v++;
        	}
        	cout << min(u,v) << endl;

        }
        else if(u>v) {
        	u=0;
            for(i=0;i<strlen(s);i=i+2) {
        		if(s[i]!='U')
        			u++;
        	}	
        	cout << u << endl;
        }
        else {
        	u=0;
        	for(i=0;i<strlen(s);i=i+2) {
        		if(s[i]!='S')
        			u++;
        	}	
        	cout << u << endl;
        }
        
	}
	return 0;
}
