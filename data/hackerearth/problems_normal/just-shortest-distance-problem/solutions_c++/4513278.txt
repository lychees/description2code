const int MAXM = 1e5+2; 
int x;
vector<int> v[MAXM];
int h[MAXM];
queue<int> q;
bool visit[MAXM];

void update(int x)
{
    for(int i=0;i<v[x].size();i++)
    {
        int t=v[x][i];
        if(h[t]>h[x]+1){
            h[t]=h[x]+1;
            update(t);
        }
    }
}

int main()
{
    int n,m,i,j,k,x,y,opt;
    cin>>n>>m;
    for(i=2;i<=n;i++)
        h[i]=1e9;
    while(m--)
    {
        cin>>opt;
        if(opt==2){
            cin>>x>>y;
            v[x].pb(y);
            if(h[y]>h[x]+1){
                h[y]=h[x]+1;
                update(y);
            }
        }else{
            cin>>x;
            if(h[x]>1e8){
                cout<<-1<<endl;
            }else{
                cout<<h[x]<<endl;
            }
        }

    }
    return 0;
}
