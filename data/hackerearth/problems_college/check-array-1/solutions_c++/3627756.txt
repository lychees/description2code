#include <iostream>

using namespace std;

int main()
{
  long long int i,n,t,s,a[100003],b[100003];
  cin>>t;
  while(t--)
  {
      cin>>n;
      s=0;
      for(i=0;i<n;i++)
      {cin>>a[i];
      b[i]=s;
      s=s+a[i];}
      s=0;
      for(i=n-1;i>=0;i--)
      {if(s==b[i])
      {
          cout<<"YES\n";
          break;
      }
      else
        s=s+a[i];

      }
      if(i==-1)
        cout<<"NO\n";

  }

    return 0;
}
