#include<bits/stdc++.h>
using namespace std;
#define cn(n) scanf("%lf",&n)
int main()
{
	int t;
	cin>>t;
	double con[t],sp[t],ttl,max[t];
	for(int i=0;i<t;i++)
        cn(sp[i]);
    for(int i=0;i<t;i++) 
        cn(con[i]);
    cn(ttl);
    for(int i=0;i<t;i++) 
    {
       max[i]=(sp[i]/con[i])*ttl;
       //printf("%lf",sp[i]);
    }
    for(int i=1;i<t;i++)
      if(max[0]<max[i])
           max[0]=max[i];
    max[0]=(int)(max[0]*pow(10,3));
    max[0]/=pow(10,3);
    printf("%0.3lf\n",max[0]);
}
