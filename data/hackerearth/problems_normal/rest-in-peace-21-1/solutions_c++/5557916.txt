#include <iostream>
using namespace std;

int main()
{
    int t,a,x,flag;
    cin>>t;
    while(t--)
    {
    	flag=1;
    	cin>>x;
    	if(x%21==0)
    	flag=0;
    	while(x>0)
    	{
    		a=x%100;
    		if(a==21)
    		{
    			flag=0;
    			break;
    		}
    		x=x/10;
    	}
    	if(flag==1) cout<<"The streak lives still in our heart!"<<endl;
    	else cout<<"The streak is broken!"<<endl;
    }
    return 0;
}
