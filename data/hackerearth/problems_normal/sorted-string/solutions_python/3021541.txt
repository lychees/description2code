from collections import Counter

 
for _ in range(input()):
	s = Counter(raw_input().strip())
	l = [(x,s[x]) for x in s]
	l = sorted(l,key = lambda x:(x[1],x[0]))
	ans = ""
	for x in l:
			ans = ans + x[0]*x[1]
		
	print ans
