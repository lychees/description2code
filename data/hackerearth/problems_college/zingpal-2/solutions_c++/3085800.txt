#include <bits/stdc++.h>
#define maxN 1000
int parent[maxN];
void init(){
	for(int i=0;i<maxN;i++){
		parent[i]=i;
	}
}
int root(int x){
	if(x!=parent[x]){
		parent[x]=root(parent[x]);
	}
	return parent[x];
}
void unions(int x,int y){
	x=root(x);
	y=root(y);
	if(x>y)parent[y]=parent[x];
	else parent[x]=parent[y];
}
using namespace std;

int main()
{
    int n,e;
    cin>>n>>e;
    init();
    int x,y;
    for(int i=0;i<e;i++){
    	cin>>x>>y;
    	if(root(x)==root(y)){
    		cout<<"FHTAGN!\n";
    		return 0;
    	}else{
    		unions(x,y);
    	}
    }
    cout<<"NO\n";
    return 0;
}
