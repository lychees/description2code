#include <iostream>

using namespace std;

struct Rec
{
	int x1,y1,x2,y2;
};

int main()
{
    int T;
    scanf("%d",&T);
    Rec F,S;
    while(T--)
    {
    	scanf("%d%d%d%d",&F.x1,&F.y1,&F.x2,&F.y2);
    	scanf("%d%d%d%d",&S.x1,&S.y1,&S.x2,&S.y2);
    	if(S.x1<F.x2&&S.y1<F.y2&&F.x1<S.x2&&F.y1<S.y2)
    	 cout<<"Rectangle"<<endl;
    	else if((S.x1<F.x2&&F.x1<S.x2&&(S.y1==F.y2||F.y1==S.y2))||(S.y1<F.y2&&F.y1<S.y2&&(F.x1==S.x2||F.x2==S.x1)))
    	 cout<<"Line"<<endl;
    	else if((F.x1==S.x2||F.x2==S.x1)&&(F.y1==S.y2||F.y2==S.y1))
    	 cout<<"Point"<<endl;
    	else 
    	 cout<<"Apart"<<endl;
    } 
    return 0;
}
