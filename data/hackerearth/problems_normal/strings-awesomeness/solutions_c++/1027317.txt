#include<bits/stdc++.h>
#include<bits/stdtr1c++.h>
using namespace std;
using namespace tr1;
#define long int64_t
#define ii pair<int,int>
#define eps numeric_limits<double>::epsilon()

char A[500010];
int S[500010], Require[26];
int n;

void init ()
{
    if (n < 11) return;

    char t[] = "hackerearth";
    int C[26] = {};
    int i;

    for (i = 0; t[i]; ++i) ++Require[t[i] - 'a'];

    for (i = 1; i < 11; ++i)
    {
        if (isupper(A[i])) A[i] += 'a' - 'A';
        ++C[A[i] - 'a'];
    }

    for (i = 11; i <= n; ++i)
    {
        if (isupper(A[i])) A[i] += 'a' - 'A';
        ++C[A[i] - 'a'];
        if (i > 11) --C[A[i - 11] - 'a'];

        S[i] = S[i - 1];
        if (mismatch(C, C + 26, Require).first - C == 26) ++S[i];
    }
}

void solve ()
{
    int q, x, y;
    scanf("%d", &q);

    while (q--)
    {
        scanf("%d%d", &x, &y);
        if (y - x >= 10) printf("%d\n", S[y] - S[x + 9]);
        else
            puts("0");
    }
}

int main ()
{
	#ifdef isMyPC
		freopen("test.inp","r",stdin);
	#endif

    gets(A + 1); n = strlen(A + 1);
	init ();
	solve ();

	return 0;
}
