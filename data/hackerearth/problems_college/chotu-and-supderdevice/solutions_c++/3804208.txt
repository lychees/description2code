#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
long long int add(long long int a,long long int b)
{
	if(a==b)
		return 0;
	if(a>=b/2)
	{
		if(a==b/2)
		{
			if(b%2==0)
			return 1;
			else
			return a*a*a-b; 
		}
		else
		{
			return min(a-b/2+1,2*a-b);
		}
	}
	else
	return(a,b/2);
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int m,p,ad;
		scanf("%lld %lld",&m,&p);
		if(p>m)
		ad=add(m,p);
		else
		ad=m-p;
		printf("%lld\n",ad);
	}
	return 0;
}
