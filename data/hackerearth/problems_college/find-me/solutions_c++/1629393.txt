#include <iostream>
#include <algorithm>
using namespace std;

bool cmp(int i,int j)
{ 
	return (i<j); 
}

int main()
{
    long int t,n,p;
    cin >> t;
    while(t--)
    {
    	cin >> n >> p;
    	long long int arr[n];
    	int ans=0;
    	for(int i=0;i<n;i++)
    	{
    		cin >> arr[i];
    	}
    	sort(arr,arr+n,cmp);
    	
    	for(int i=0;i<n;i++)
    	{
    		if(arr[i]==p)
    		{
    			ans = 1;
    			cout << "YES " << n-i-1 << endl;
    			break;
    		}
    	}
    	
    	if(ans==0)
    	{
    		cout << "NO 0" << endl;
    	}
    }
    return 0;
}
