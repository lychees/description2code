#include <iostream>
using namespace std;

int main()
{
    int n,a=1,b=1,c;
    cin>>n;
    
    if(n==1)
    	cout<<a<<"\n";
    else if(n==2)	
    	cout<<a<<"\n"<<b<<"\n";
    else
    {
    	cout<<a<<"\n"<<b<<"\n";
    	while(n-2>0)
    	{
	    	
    		c = a+b;
    		cout<<c<<"\n";
    		a=b;
    		b=c;
    		n--;
    	}
    }
    return 0;
}
