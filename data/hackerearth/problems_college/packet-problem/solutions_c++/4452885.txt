#include <iostream>
using namespace std;
long long a[91],b[91];
int main()
{
	int i,t,n;
	a[1] = 0;
	b[1] = 1;
	a[2] = 1;
	b[2] = 1;
	for(i = 3;i <= 90;i++){
		a[i] = a[i-1] + a[i-2];
		b[i] = b[i-1] + b[i-2];
	}
	cin >> t;
	while(t--){
		cin >> n;
		cout << a[n] << " " << b[n] << endl;
	}
    return 0;
}
