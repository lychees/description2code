#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define F(i,a,b) for(int i = (int)(a); i <= (int)(b); i++)
#define RF(i,a,b) for(int i = (int)(a); i >= (int)(b); i--)
int convert(vector<int> A)
{
    int factor = 1,ans = 0;
    F(i,0,9)
    {
        if(A[i] == 1) ans += factor;
        factor <<= 1;
    }
    return ans;
}
int main()
{
    #ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
        freopen("output.txt","w",stdout);
    #endif
    int N,Q,M,temp,Count,K;
    int arr[1000005];
    F(i,1,1000000)
    {
        temp = i;
        vector<int> A(10);
        while(temp > 0)
        {
            A[temp%10] = 1;
            temp /= 10;
        }
        arr[i] = convert(A);
    }
    scanf("%d%d",&N,&Q);
    while(Q--)
    {
        scanf("%d",&M);
        K = 0;
        F(i,0,M-1)
        {
            scanf("%d",&temp);
            K += (1<<temp);
        }
        Count = 0;
        F(i,1,N)
        {
            if(arr[i] & K) Count++;
        }
        printf("%d\n",Count);
    }
    return 0;
}
