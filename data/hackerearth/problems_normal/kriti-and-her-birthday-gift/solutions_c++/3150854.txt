#include <bits/stdc++.h>
using namespace std;
typedef unsigned long long ull;
const ull base  = 173;
const int maxn = 1e5+10;
pair<ull,int> a[maxn];
int n,q,l,r,p1,p2;
char s[20];
ull z;
ull geth(char z[],int len){
    ull ans=0;
    for (int i=0; i<len; i++) ans=ans*base+z[i]-'a';
    return ans;
}
int main(){
  
    scanf("%d\n",&n);
    for (int i=1; i<=n; i++){
        scanf("%s\n",s);
        a[i]=make_pair(geth(s,strlen(s)),i);
    }
    sort(a+1,a+n+1);
    scanf("%d\n",&q);
    while (q--){
        scanf("%d %d %s\n",&l,&r,s);
        z=geth(s,strlen(s));
        printf("%d\n",upper_bound(a+1,a+n+1,make_pair(z,r))-lower_bound(a+1,a+n+1,make_pair(z,l)));
    }
 
}
