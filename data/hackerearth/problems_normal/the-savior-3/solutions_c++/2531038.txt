#include <iostream>
using namespace std;
#include<cassert>
int main()
{
   int T;
   cin>>T;
   while(T)
   {
   	int N;
   	cin>>N;
   	
   	int array[N];
   	for(int i=0;i<N;i++)
   	{
   		cin>>array[i];
   	}
   	int count;
   	count=0;
   	for(int a=0;a<N-1;a++)
   	{
   		for(int a1=a+1;a1<N;a1++)
   		{
   			if((array[a]+array[a1])%2==0&&array[a]!=array[a1])
   			{
   				count++;
   			}
   		}
   	}
   	cout<<count<<endl;
   	T--;
   }
    return 0;
}
