#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
const int mx=1e3+10;
double f[mx][mx];

int main(){
   int t,n,x,y,w;
   double p1,p2;
   scanf("%d",&t);
   while(t--){
      scanf("%d %d %d %d %lf %lf",&x,&y,&n,&w,&p1,&p2);
      p1/=100.0;
      p2/=100.0;
      f[0][0]=1.0;
      for(int i=1;i<=n;i++){
         f[i][0]=1.0;
         for(int s=1;s<=w;s++){
            f[i][s]=0.0;
            double e1=1.0;
            if(s-x>=0)e1=f[i-1][s-x];
            f[i][s]=max(f[i][s],e1*p1+f[i-1][s]*(1.0-p1));
            double e2=1.0;
            if(s-y>=0)e2=f[i-1][s-y];
            f[i][s]=max(f[i][s],e2*p2+f[i-1][s]*(1.0-p2));
         }
      }
      double ans=f[n][w]*100.0;
      printf("%lf\n",ans);
   }

   return 0;
}
