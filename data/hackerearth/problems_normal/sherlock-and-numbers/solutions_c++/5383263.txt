#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long int n,k,p;
		scanf("%ld%ld%ld",&n,&k,&p);
		long int a[100100],i;
		for(i=1;i<=k;i++)
		{
			scanf("%ld",&a[i]);
		}
		sort(a+1,a+k+1);
		//int count=0,pos;
		int final_pos=p;
		for(i=1;i<=k;i++)
		{
			if(a[i]<=final_pos)
			{
				final_pos++;			
			}			
		}		
		if(final_pos>n)
		{
			cout<<"-1\n";
		}
		else
		{
			cout<<final_pos<<"\n";
		}
	}
	return 0;
}
