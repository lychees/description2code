#include <cstdio>
#include <iostream>
#include <algorithm>

using namespace std;

char s[10009];
int n;

char a[] = {'Y', 'N'};

int f(int i, int l) {
	if(i == n) return 0;
	return (s[i] != a[l]) + f(i + 1, !l);
}

int main() {
	int t;
	cin >> t;
	while(t--){
		cin >> n >> s;
		cout << min(f(0, 0), f(0, 1)) << endl;
	}
	return 0;
}
