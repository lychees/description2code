#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <iterator>
#include <vector>
#include <cstring>
#include <map>

#define gc getchar_unlocked
static  long int get_i()
{
    char c=gc();
    while(c<'0'||c>'9')
        c=gc();
    long int ret = 0;
    while(c>='0'&&c<='9')
    {
       ret = ret * 10 + c - '0';
       c=gc();
    }
   
    return ret;
}
int main(int argc, char const *argv[])
{
    int t = get_i();
    while(t--)
    {
        std::string s;
        std::cin>>s;
        bool check = true;
        int count=0;
        std::map<char, bool> map;
        if(s.length()<26)
        {
            printf("NO\n");
            continue;
        }
        else
        {
            for(int i=0;i<s.length();i++)
                map.insert(std::pair<char,bool>(s[i],true));

            for (std::map<char,bool>::iterator it = map.begin(); it != map.end(); ++it)
                count++;
            
            (count==26)?printf("YES\n"):printf("NO\n");
        }
    }
    return 0;
}
