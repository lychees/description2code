#include <iostream>
using namespace std;

int main()
{
    int p[1001];
	int n=1000;   
    for(int i=0;i<=1000;i++){
    	p[i]=1;
    }
    
    for(int i=2;i*i<=n;i++){
    	if(p[i]==1)
    		for(int j=i*2;j<=n;j=j+i)
    			p[j]=0;
    }
    
    p[0]=p[1]=0;
    
    //for(int i=1;i<=n;i++)
   // 	p[i]=p[i-1]+p[i];
    
   int t;
   cin>>t;
   while(t--){
   	int alen,temp;
   	cin>>alen;
   	int count=0;
   	while(alen--){
   		cin>>temp;
   	
   		count=count+(p[temp]);
   	}
   	
   	cout<<count<<endl;
   	
   }
    
    return 0;
}
