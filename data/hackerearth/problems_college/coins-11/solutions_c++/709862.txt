# include <iostream>
# include <vector>
using namespace std;

int iscomposite(int n)
{
	int i;
	for (i=2;i<n;i++)
		if (n%i == 0)
			return 1;
	if (n == 3)
		return 3;
	if (n == 1)
		return 1;
	else
		return 2;
}

int main()
{
	int i,n,a;
	cin >> n;
	vector <int> v;
	for (i=1;i<=n;i++)
	{	
		cin >> a;
		v.push_back(a);
	}
	for (i=0;i<n;i++)
	{	
		v[i] = iscomposite(v[i]);
		cout << v[i] << endl;
	}
	return 0;
}
