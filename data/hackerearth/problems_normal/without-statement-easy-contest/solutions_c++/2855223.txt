#include <bits/stdc++.h>
using namespace std;

int pre[1001][10001],p[10];

inline int precompute(){
    for (register int i=1;i<=729;i++)
        pre[i][0]=i;

    for (register int i=0;i<10;i++)
        p[i]=i*i;

    for (register int i=1;i<=729;i++){
        for (register int j=1;j<=10000;j++){
            int x=pre[i][j-1];
            int sum=0;
            while (x){
                int d=x%10;
                sum+=p[d];
                x/=10;
            }
            pre[i][j]=sum;
        }
    }
    return 0;
}

int main(){
    int t,n,m;
    precompute();
    cin >> t;
    while (t--){
        scanf("%d%d",&n,&m);
        int temp=n;
        n=0;
        while (temp){
            int d=temp%10;
            n+=p[d];
            temp/=10;
        }
        m--;
        while (m>10000){
            n=pre[n][10000];
            m-=10000;
        }
        cout << pre[n][m] << endl;
    }
    return 0;
}
