#include <bits/stdc++.h>

using namespace std;

int main()
{	
	ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int T;
    long long A,B,C,K;
    double d,x;
    cin>>T;
    while(T--) {
    	cin>>A>>B>>C>>K;
 		d = sqrtl(B*B - 4*A*(C-K));
    	x = (-B+d)/(2*A);
    	if(x < 0 || x == 0)
    		cout<<0;
    	else {
    		if((C-K) >= 0)
    			cout<<0;
    		else {
    			if(x > (int)x)
    				x++;
    			cout<<(int)x;
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
