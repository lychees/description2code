k = int(raw_input())
str = raw_input()
cnt = 0
for i in range(0, len(str)):
    j = i
    window = set()
    while (len(window) <= k and j < len(str)):
        window.add(str[j])
        if (len(window) == k):
            # print i ,'-' , j
            cnt = cnt + 1
        j = j+1
print cnt
