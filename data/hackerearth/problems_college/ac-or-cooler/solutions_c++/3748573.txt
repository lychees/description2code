#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int j,n,m,i,a[100002],c[100002];
    cin >> n >> m;
    for(i=0;i<n;i++)cin >> a[i];
    for(i=0;i<n;i++)cin >> c[i];
    
    sort(a,a+n);
    sort(c,c+n);
    int ans = 0;int cost = 0;
    i = j = 0;
    while(ans < n && cost < m && i < n && j < n){
    	//cout << "xxx\n";
    	if(a[i] < c[j] && ( cost + a[i] ) <= m){
    		cost += a[i];
    		i++;
    	}else if( c[j] <= a[i] && ( cost + c[j] ) <= m ){
    		cost += c[j];
    		j++;
    	}else{
    		break;
    	}
    	ans++;
    }
    //cout << ans << endl;
    if(i >= n && cost <= m){
    	while(ans < n && j < n && ( cost + c[j] ) <= m){
    		cost += c[j];
    		ans++;
    		j++;
    	}
    }
    if(j >= n && cost <= m){
    	while(ans < n && i < n && ( cost + a[i] ) <= m){
    		cost += a[i];
    		ans++;
    		i++;
    	}
    }
    cout << ans << endl;
    return 0;
}
