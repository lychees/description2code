#include <bits/stdc++.h>
#define ll long long
using namespace std;
bool h[100000][26];
int flag=0;
int visited[100000];
string s;
int n;
void ispossible(int st,int en)
{
	if(st>en)
	{
		flag=1;
		return;
	}
	int i;
	for(i=0;i<n;i++)
	{
		if(visited[i])
		continue;
		else
		{
			if(h[i][s.at(st)-'a']==1)
			{
				visited[i]=1;
				ispossible(st+1,en);
				if(flag)
				return;
				visited[i]=0;
			}
		}
	}
		
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	memset(h,0,sizeof(h));
	cin>>s>>n;
	int i,j;
	for(i=0;i<n;i++)
	{
		for(j=0;j<26;j++)
		{
			h[i][j]=0; //Not present
		}
		visited[i]=0;
	}
	string t;
	for(i=0;i<n;i++)
	{
		for(j=0;j<6;j++)
		{
			cin>>t;
			if(t.at(0)>='a' && t.at(0)<='z')
			h[i][t.at(0)-'a']=1;//set
		}
	}
	int l=s.length();
	ispossible(0,l-1);
	if(flag)
	cout<<"YES"<<endl;
	else
	cout<<"NO"<<endl;
	return 0;
}
