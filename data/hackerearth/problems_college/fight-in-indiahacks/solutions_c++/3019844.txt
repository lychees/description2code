#include<bits/stdc++.h>
using namespace std;

long long int dp[1000005][2];

long long int calWin(long long int n,long long int a,long long int b,long long int chance)
 {

      if(n<a && n<b)
        return (!chance);
      if(n==a || n==b)
      {
          dp[n][chance]=chance;
          return chance;
      }
      if(dp[n][chance]!=-1)
             return dp[n][chance];
      if(n-a>=0)
            if(chance==calWin(n-a,a,b,!chance))
                {
                    dp[n][chance]=chance;
                    return chance;
                }
      if(n-b>=0)
            if(chance==calWin(n-b,a,b,!chance))
                 {
                     dp[n][chance]=chance;
                     return chance;
                 }

           dp[n][chance]=(!chance);
            return (!chance);
 }

void findWinner()
  {
      long long int test,n,a,b;
      scanf("%lld",&test);
      while(test--)
      {
          memset(dp,-1,sizeof(dp));
          scanf("%lld%lld%lld",&n,&a,&b);
          if(a==0 || b==0)
             cout<<"1"<<endl;
          else
           cout<<calWin(n,a,b,1)<<endl;
      }

  }

int main()
{

   findWinner();
    return 0;
}
