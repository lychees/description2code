def isprime(n):
    n = abs(int(n))
    if n < 2:
        return False
    if n == 2: 
        return True   
    if not n & 1: 
        return False
    for x in range(3, int(n**0.5) + 1, 2):
        if n % x == 0:
            return False

    return True
n=input()
for i in range(n):
	s=raw_input().split(' ')
	a=int(s[0])
	b=int(s[1])
	sum=0
	for j in range(a,b+1):
		if j==1:
			continue
		if(isprime(j)):
			sum+=j
	print sum
