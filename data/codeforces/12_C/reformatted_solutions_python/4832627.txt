from collections import defaultdict as d


def r():
    return map(int, raw_input().split())


n, m = r()
c = sorted(r())
D = d(int)
for i in m*[0]:
    D[raw_input()] += 1
v = sorted(D.values())

sm = 0
sM = 0
for i in range(len(v)):

    sm += v[i]*c[len(v)-i-1]
    sM += v[i]*c[-(len(v)-i)]
print sm, sM
