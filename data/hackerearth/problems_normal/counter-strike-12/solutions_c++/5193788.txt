#include <iostream>
#include<cmath>
using namespace std;
int a[10000],b[10000],x[10000],y[10000],n,m,d,t,k;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {cin>>n>>m>>d;
    for(int i=0;i<n;i++)
    cin>>a[i]>>b[i];
    for(int i=0;i<m;i++)
    cin>>x[i]>>y[i];
    k=0;
    for(int i=0;i<m;i++)
      {for(int j=0;j<n;j++)
         {if((abs(b[j]-y[i])+abs(a[j]-x[i]))<=d)
          { k++;
           break;
          }
         }
       
      }
         if(k>m/2 )
         cout<<"YES"<<endl;
         else
         cout<<"NO"<<endl;
         }
    return 0;
}
