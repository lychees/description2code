#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t;
    long long int n,s,temp,i;
    
    scanf("%d",&t);
    while(t--){
    	scanf("%lld",&n);
    	s=0;
    	if(n==2){
    		printf("%d\n",1);
    	}else if(n%2==1){
    		printf("%d\n",0);
    	}else{
    		temp=n/2;
    		s=2;
    		
    		for(i=2;i<=sqrt(temp);i++){
    			if(temp%i==0){
    				if(i == temp/i){
    					s+=1;
    				}else{
    					s+=2;
    				}
    			}
    		}
    		printf("%d\n",s);    		
    	}
    }
    
    return 0;
}
