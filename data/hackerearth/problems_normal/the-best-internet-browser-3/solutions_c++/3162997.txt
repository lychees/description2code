#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;string s;int num1,num2,i;char c;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	num1=(s.size());
    	num2=0;
    	for(i=0;i<s.size();i++)
    	{
    		c=s[i];
    		if((c=='a')||(c=='e')||(c=='i')||(c=='o')||(c=='u'))
    		{
    			continue;
    			
    		}
    		num2++;
    		
    	}
    	num2++;
    	cout<<(num2-4)<<"/"<<num1<<endl;
    }
    
    return 0;
}
