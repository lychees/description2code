int find()
{
	int p = 0 , i = 0;
	while (i<sl)
	{
		if (b[p][s[i]-'a'] == 0) return -1;
		p = b[p][s[i]-'a'];
		i++;
	}
	return p;
} 

int calc(int l , int r)
{
	int id = find();
	if (id==-1) return 0;
	
	set<int>::iterator L = m[id].lower_bound(l);
	set<int>::iterator R = m[id].upper_bound(r);
	int ans = distance(L , R);
	
	return ans;
}

int Random(int min,int max)
{
	static bool inited = false;
	if (!inited)
	{
		unsigned seed = time(NULL);     // uses the system clock
		srand(seed);                    // set system clock to seed of random
		inited = true;
	}
	int range = max - min + 1;      // calculate range of random numbers
	int r = rand()%range + min;     // create randomly number in range
	return r;
}
