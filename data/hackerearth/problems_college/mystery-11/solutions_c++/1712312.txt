#include <iostream>
#include<map>
using namespace std;
char arr[1001];
int main()
{
	ios::sync_with_stdio(false);
	map<char, int>mymap = { { '0', 6 }, { '1', 2 }, { '2', 5 }, { '3', 5 }, { '4', 4 }, { '5', 5 }, { '6', 6 }, { '7', 3 }, { '8', 7 }, { '9', 6 } };
	int t;
	cin >> t;
	while (t--)
	{
		int sum = 0;
		cin >> arr;
		for (int i = 0; arr[i] != '\0'; i++)
		{
			sum += mymap[arr[i]];
		}
		cout << sum << endl;
	}
	return 0;
}
