#include <iostream>
#include<map>
using namespace std;
bool answer(string s){
	int count=0;
	int n=s.size();
	map<int,char> m;
	m[0]='l';
	m[1]='o';
	m[2]='v';
	m[3]='e';
	int j=0;
	for(int i=0;i<n;i++){
		if(s[i]==m[j]){
			j++;
		}
		if(j>3){
			break;
		}
	}
	if(j>3){
		return true;
	}
	return false;
}
int main()
{
    // cout << "Hello World!" << endl;
    string s;
    cin>>s;
    if(answer(s)){
    	cout<<"I love you, too!\n";
    }else{
    	cout<<"Let us breakup!\n";
    }
    return 0;
}
