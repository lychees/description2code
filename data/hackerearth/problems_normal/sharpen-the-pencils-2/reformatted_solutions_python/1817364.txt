for _ in range(int(raw_input())):
    noOfPencils = int(raw_input())
    lengthOfPencils = raw_input().split()
    lengthOfPencils = [int(i) for i in lengthOfPencils]
    summationOfLengths = [lengthOfPencils[0]]
    for i in range(1, noOfPencils):
        summationOfLengths.append(
            summationOfLengths[i - 1] + lengthOfPencils[i])
    totalSum = sum(lengthOfPencils)
    toCover = totalSum / 3 * 2
    for i in range(noOfPencils):
        if summationOfLengths[i] == toCover:
            print i + 1, noOfPencils - i - 1
            break
        elif summationOfLengths[i] > toCover:
            if i == 0:
                print i + 1, noOfPencils - i - 1
            elif i == noOfPencils - 1:
                print i, noOfPencils - i
            else:
                a = lengthOfPencils[:i]
                b = lengthOfPencils[i + 1:]
                if sum(a) > sum(b) * 2:
                    print i, noOfPencils - i
                else:
                    print i + 1, noOfPencils - i - 1
            break
