#include<bits/stdc++.h>
using namespace std;
int main()
{
    set<long int> S;
    set<long int>::iterator it1,it2;
    long int i,n,j,k,diff;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>k;
        S.insert(k);
    }
    while(1)
    {
       int index=0;
        for(it1=S.begin();it1!=S.end();)
        {
           int flag=0;
            for(it2=it1;it2!=S.end();it2++)
            {
                if(it2!=it1)
                {
                    diff=abs(*it2-*it1);
                    if(S.find(diff)==S.end())
                    {
                        flag=1;
                        break;
                    }
                }
            }
            if(flag==1)
            {
                S.insert(diff);
                index++;
            }
            else
                it1++;
        }
        if(it1==S.end())
        {
            if(index%2==1)
                cout<<"Alice\n";
            else
                cout<<"Bob\n";
            break;
        }
    }

}
