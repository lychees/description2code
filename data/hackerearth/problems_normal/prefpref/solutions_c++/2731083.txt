int   z[1000005];

void Z_Func(int t){
    int l = -1, r = -1;
    z[0] = t;

    for(int i = 1; i < t; i++){
        z[i] = 0;
        if(i < r){ z[i] = min(z[i-l], t-i); }
        while(i + z[i] < t && en[z[i]] == en[ z[i]+i ] ) z[i]++;
        if(i+z[i] > r){ r = i+z[i]; l = i; }
    }
}
