#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,i,sum;
	string str;
	cin>>t;
	while(t--)
	{
		cin>>str;
		sum=0;
		int len=str.length();
		for(i=0;i<len;i++)
		{
			if(str[i]=='-' && ((i+1<len && str[i+1]!='B') || i==len-1) && ((i+2<len && str[i+2]!='B')|| i+2>=len) && ((i+1<len && str[i+1]!='S')||i==len-1) && str[i-1]!='S'  )
			sum++;
			
		}
		cout<<sum<<"\n";
	}
    return 0;
}
