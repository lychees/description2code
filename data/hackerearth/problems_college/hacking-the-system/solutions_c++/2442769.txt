#include <iostream>
#include<cmath>
using namespace std;

int main()
{
	int A,B,i=1,diff=0,j=0;
    cin>>A;
    cin>>B;
int lenA=log10(A)+1;
    int lenB=log10(B)+1;
    if(lenA==lenB)
    {    
        while(j<lenA)
      {
            if(((A/i)%10)!=((B/i)%10))
                {
                    diff++;
                    if(diff>1) {cout<<"loses";return 0;}
                } 
                i=i*10;
                j++;
       }
     
    }
    
      cout<<"wins";
    return 0;
}
