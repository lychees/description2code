#include <stdio.h>
using namespace std;

int main() {
  int t, r, l, rem, c;
  scanf("%d", &t);
  for (int i = 0; i < t; i++) {
    c = 0;
    scanf("%d%d", &r, &l);
    rem = 0;
    if ((l * 2) <= r && l != 0)
      c = 10;
    else {
      if ((l * 2) % r != 0)
        rem = 1;
      c = (((l * 2) / r) + rem) * 5;
    }
    printf("%d \n", c);
  }
  return 0;
}
