#include <iostream>
using namespace std;

int main() {
	long int t,n,m,i,k,f;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		long int a[n];
		k=n-1,f=0;
		for(i=0;i<n;i++)
		{
			if(i%2==0)
			a[i]=k--;
			else
			a[i]=f++;
		}
		for(i=0;i<n;i++)
		{//cout<<a[i]<<" ";
		if(a[i]==m)
		{cout<<i<<"\n";
		break;}}
	}
}
