#include <iostream>
#define ll long long
using namespace std;

int main()
{
	ll t, a, b, d1, d2;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	ll carry=0, c=0;
    	while(a!=0||b!=0)
    	{
    		d1=a%10;d2=b%10;
    		if((carry+d1+d2)>=10)
    		{
    			carry=(carry+d1+d2)/10;
    			if(carry==1)
    				c++;
    		}
    		a/=10;b/=10;
    	}
    	if(c==0)
    		cout<<"No carry operation"<<endl;
    	else if(c==1)
    		cout<<c<<" carry operation"<<endl;
    	else
    		cout<<c<<" carry operations"<<endl;
    }
    return 0;
}
