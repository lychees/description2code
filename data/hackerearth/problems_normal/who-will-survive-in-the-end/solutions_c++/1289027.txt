#include <iostream>
#include <math.h>
#define ll long long int
using namespace std;

ll lastPosition(ll n){

	ll m,l,res;
	for(m=1;m<=n/2;m++){
		if(pow(2,m)>n)
			break;
	}
	
	m=m-1;
//	cout<<m<<endl;
	l= n-pow(2,m);
	res = 2*l+1;
	return res;
}

int main()
{
	ll n;
    int t;
    cin>>t;
    while(t--){
    	cin>>n;
  //  	lastPosition(n);
    	cout<<lastPosition(n)<<endl;
    }
}
