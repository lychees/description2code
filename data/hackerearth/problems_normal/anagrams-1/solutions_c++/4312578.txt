#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
#define MAX 100005
string str1,str2;
string sub(string nstr1, string nstr2)
{
	int fre1[26], fre2[26];
	memset(fre1,0,sizeof(fre1));
	memset(fre2,0,sizeof(fre2));
	int i;
	if(nstr2.size() > nstr1.size()) swap(nstr1,nstr2);
	for(i=0;i<nstr1.size();++i)
	{
		fre1[nstr1[i]-'a']++;
	}
	for(i=0;i<nstr2.size();++i)
	{
		fre2[nstr2[i]-'a']++;
	}
	for(i=0;i<26;++i)
	{
		if(fre1[i] >= fre2[i]) fre1[i]-=fre2[i];
		else return "NO LUCK";
	}
	int cnt;
	string ans;
	int flag = 0;
	char z;
	char g;
	for(i=0;i<26;++i)
	{
		if(fre1[i] % 2 == 0)
		{
			cnt = fre1[i]/2;
			g = 'a'+i;
			while(cnt--)
			{
				ans+=g;
			}
			
		}
		else
		{
			flag++;
			z = 'a'+i;
			cnt = fre1[i]/2;
			while(cnt--)
			{
				ans+=z;	
			}
		}
	}
	string rev;
	rev = ans;
	reverse(rev.begin(),rev.end());
	if(flag == 0)
	{
		ans+=rev;
		return ans;
	}
	else if(flag == 1)
	{
		ans+=z;
		ans+=rev;
		return ans;
	}
	else return "NO LUCK";
}
int main()
{
	int t;
	scanf("%d\n",&t);
	while(t--)
	{
		getline(cin,str1);
		getline(cin,str2);
		string line1,line2;
		char z;
		int i;
		for(i=0;i<str1.size();++i)
		{
			z = str1[i];
			if((z >= 65 && z <= 90) || (z >= 97 && z <= 122))
			{
				z = tolower(z);
				line1+=z;
			}
		}
		for(i=0;i<str2.size();++i)
		{
			z = str2[i];
			if((z >= 65 && z <= 90) || (z >= 97 && z <= 122))
			{
				z = tolower(z);
				line2+=z;
			}
		}
		sort(line1.begin(),line1.end());
		sort(line2.begin(),line2.end());
		if(line1 == line2) puts("YES");
		else
		{
			cout<<sub(line1,line2)<<endl;
			
		}

	}
    return 0;
}
