#include <iostream>
#include <string>
using namespace std;

int main()
{
    int inputs;
    cin>>inputs;
    while(inputs--)
    {
    	string s;
    	cin>>s;
    	int len=s.length();
    	int a[26]={0};
    	int i=0;
    	if(len<26)
    	  cout<<"NO"<<endl;
    	else
    	{
    		for(int j=0;j<len;j++)
    			a[s[j]-'a']+=1;
    	
    	
    	for(i=0;i<26;i++)
    	{
    		if(a[i]<1)
    		{
    			cout<<"NO"<<endl;
    			break;
    		}
    	}
    	}
    	if(i==26)
    	cout<<"YES"<<endl;
    	
    }
    return 0;
}
