n,m=(int(e) for e in raw_input().split())
def bsearch(arr,n):
	h=len(arr)-1
	l=0
	while l<h:
		m=(l+h)/2
		if arr[m][0] >=n:
			h=m
		else:
			l=m+1
	return l
