#include<iostream>
#include<math.h>
using namespace std;
int main()
{
	long int t;
	long long int total;
	long long int l, r, sq1, sq2;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		cin>>l>>r;
		sq1=sqrt(l);
		sq2=sqrt(r);
		if(sq1*sq1>l)
			sq1--;
		if(sq2*sq2>r)
			sq2--;
		total=sq2-sq1+((sq1*sq1)==l);
		cout<<total<<endl;
	}
	
	return 0;
}
