#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++)
		   cin>>a[i];
		int count[n];
		int visit[n];
		memset(visit,0,sizeof(visit));
		memset(count,0,sizeof(count));
		for(int i=0;i<n;i++)
		{ 
		  if(!visit[i])
		  {
		   // visit[i]=1;
		    int temp=0;
		    int j=i;
		    vector<int> x;
		    while(!visit[j])
		    {
		    	x.push_back(j);
		    	visit[j]=1;
		    	j=a[j];
		    	temp++;
			}
			visit[j]++;
			int k=count[j];
		    count[i]=k+temp;
			bool flag=false;
			int p1=0;
			for(vector<int>::iterator p=x.begin();p!=x.end();p++)
			{
				if(!flag)
				{
					if(visit[*p]>=2)
					{
					    flag=true;
					    count[*p]=temp-p1+k;  
				    }
				    else
				    {
				    	count[*p]=temp-p1 +k;
				    	p1++;
					}
				}
				else
				{
					count[*p]=temp-p1+k;
				}
			}
		  }
		}
		int max1=0,ans=0;
		for(int i=0;i<n;i++)
		   if(max1<count[i])
		   {
			 ans=i;
			 max1=count[i];
		   }
		cout<<ans<<endl;
	}
	return 0;
}
