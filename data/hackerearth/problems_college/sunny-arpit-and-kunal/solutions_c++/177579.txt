#include<iostream>
using namespace std;
int abs(int n)
{
    if(n<0)
    return -1*n;
    else
    return n;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int a,b,c;
        int g,s,x,y;
        int flag1=0,flag2=0,flag3=0;
        cin>>a>>b>>c;
        if(a>b&&a>c)
        {
            g=a;
            flag1=1;
        }
        else if(b>a&&b>c)
        {
            g=b;
            flag2=1;
        }
        else
        {
            g=c;
            flag3=1;
        }
        if(a<b&&a<c)
        {
            s=a;
            flag1=1;
        }
        else if(b<a&&b<c)
        {
            s=b;
            flag2=1;
        }
        else
        {
            s=c;
            flag3=1;
        }
        if(flag1==0)
        {
            x=b-a;
            x=abs(x);
            y=c-a;
            y=abs(y);
        }
        else if(flag2==0)
        {
            x=a-b;
            x=abs(x);
            y=c-b;
            y=abs(y);
        }
        else
        {
            x=a-c;
            x=abs(x);
            y=b-c;
            y=abs(y);
        }
        if(x>y)
        cout<<x-1<<endl;
        else
        cout<<y-1<<endl;
    }
    return 0;
}
