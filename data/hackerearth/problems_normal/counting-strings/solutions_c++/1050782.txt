#include <iostream>
#include <string>

using namespace std;

int main()
{
	int testCount = 0;
	string s = "";
	long index = 0;
	long index1 = 0;
	long index2 = 0;
	long result = 0;
	long len = 0;
	long i = 0;
	
	cin >> testCount;
	cin.clear();
	cin.ignore();
	
	while (testCount > 0)
	{
		cin >> s;
		
		i = 0;
		result = 0;
		len = s.length();
		
		while (i < len)
		{
			if (s[i] == 'a' || s[i] == 'z')
			{
				result += (len - i);
				i++;
			}
			else
			{
				index1 = s.find('a', i);
				index2 = s.find('z', i);
				
				if ((index1 != string::npos) || (index2 != string::npos))
				{
					if (index1 == string::npos)
					{
						index = index2;
					}
					else if (index2 == string::npos)
					{
						index = index1;
					}
					else
					{
						if (index1 < index2)
						{
							index = index1;
						}
						else
						{
							index = index2;
						}
					}
				}
				else
				{
					break;
				}
				
		    	result += ((index - i) * (len - index));
			    i = index;
			}
		}
		
		cout << result << endl;
		testCount--;
	}
	
    return 0;
}
