#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	int L,R,n,i;
	map<int, long long> mp;
	scanf("%d",&t);
	while (t --) {
		scanf("%d",&n);
		int a[n];
		for (i = 0; i < n; i++) {
			scanf("%d",&a[i]);
		}
		scanf("%d %d",&L,&R);
		for (i = 0; i < n; i++) {
			for (map<int,long long> :: iterator it = mp.begin(); it != mp.end(); it++) {
				mp[__gcd(a[i],it->first)] += (long long)it->second;
			}
			mp[a[i]]++;
		}
		long long ans = 0;
		for (i = L; i <= R; i++) {
			ans += mp[i];
		}
		printf("%lld\n",ans);
		mp.clear();
	}
	
	return 0;
}
