#include <iostream>
using namespace std;

int main()
{
	int t ;
	cin >> t;
	
	long long mo = 1000000007;

	long long dp[1002][1002];
	long long top[1002];
	
	for(int i=0 ; i < 1002 ; i++)
		for(int j=0 ; j < 1002 ; j++)
			dp[i][j]=0;

		dp[1][1]=1 ;
		top[1]=1;
	
	
		for(int i=2 ; i < 1002 ; i++){
			long long temp = 0 ;
			for(int j=1; j<=i; j++){
			dp[i][j]=j*dp[i-1][j]%mo + dp[i-1][j-1]%mo;
			temp=(temp+dp[i][j])%mo;
			}
			top[i]=temp;
		}
			
	
	
	while(t--){
		
		int val  ;
		cin >> val ;
		cout << top[val] << endl ;
		
	}
	

    return 0;
}
