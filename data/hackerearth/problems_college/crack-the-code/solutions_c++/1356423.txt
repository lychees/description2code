#include<bits/stdc++.h>
typedef long long int ll;
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
        ll x,y,a=0,b=1,cnt=0;
        cin>>x>>y;
        while(a<=y)
        {
            if(a>=x)
                cnt++;
            ll c=a+b;
            a=b;
            b=c;
        }
        
        if(cnt%2==0)
            cout<<"VALID"<<endl;
        else
            cout<<"INVALID"<<endl;
    }
    return 0;
}
