#include<bits/stdc++.h>

typedef long long ll;
using namespace std;

int main()
{
  //freopen("input.txt","r",stdin);
    int t;
    cin>>t;
    while(t--)
    {
        ll n,i,mi=0,ma=0,num=0;
        cin>>n;
        cin>>num;
        ma=num;
        mi=num;
        for(i=1;i<n;i++)
        {
            cin>>num;
            if(num<mi)
                mi=num;
            if(num>ma)
                ma=num;
        }
        //cout<<ma<<" "<<mi<<endl;
        if(abs(ma-mi)%2==1)
            cout<<"Yes"<<endl;
        else
           cout<<"No"<<endl;
    }
    return 0;
}
