long long sqRoot(long long num)
{
	long long h=1e9,l=0,m,ans;
	while(l<=h){
		m=(h+l)/2;
		if(m*m<=num){
			l=m+1;
			ans=m;
		}
		else
		h=m-1;
	}

	return ans;
}
