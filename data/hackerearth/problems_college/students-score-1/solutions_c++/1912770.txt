#include<bits/stdc++.h>
using namespace std;
int max(vector<int> a)
{
   int m1 = 0, m2 = 0;
   int i;
   for(i = 0; i <a.size(); i++)
   {
     m2 = m2 + a[i];
     if(m2 < 0)
        m2 = 0;
     if(m1 <m2)
        m1 =m2;
    }
    return m1;
} 
 
int main()
{
	int n;
	cin>>n;
	
	vector<int> v;
	
	for(int i=0;i<n;i++)
	{
		int x;
		cin>>x;
		v.push_back(x);
	}
	
	cout<<max(v)<<endl;
	
	
	
		
}
