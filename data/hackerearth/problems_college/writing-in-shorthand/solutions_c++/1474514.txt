#include <iostream>
#include<algorithm>
#include<cstdio>
#include<string>
using namespace std;

int main()
{
    pair <string,string> p[3005];
    int n,m,i,j;
    cin>>n>>m;
    for(i=0;i<m;i++)
    {
    	cin>>p[i].first>>p[i].second;
    }
    string str;
    for(i=0;i<n;i++)
    {
    	cin>>str;
    	for(j=0;j<m;j++)
    	{
    		if(str==p[j].first)
    		{
    			if(p[j].first.length()<=p[j].second.length())
    			cout<<p[j].first<<" ";
    			else
    			cout<<p[j].second<<" ";
    			break;
    		}
    	
    	}
    }
    return 0;
}
