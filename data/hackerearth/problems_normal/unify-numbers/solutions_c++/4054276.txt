#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int i,n,tcase,median;
	long long sum=0;
	int arr[100000];
    cin >> tcase;
    while(tcase--){
    	sum=0;
    	cin >> n;
    	for(i=0 ; i<n ; i++){
    		cin >> arr[i];
    	}
    	sort(arr,arr + n);
    	if(n&1){
    		//odd
    		median = (arr[n/2]);
    	} else {
    		median = (arr[n/2] + arr[n/2 - 1])/2;
    	}
    	for(i=0 ; i<n ; i++){
    		sum+=abs(arr[i]-median);
    	}
    	cout << sum << endl;
    }
    return 0;
}
