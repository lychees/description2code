template <typename T>
T modpow(T base, T exp) {
  /// base %= modulus;
  T result = 1;
  while (exp > 0) {
    if (exp & 1) result = (result * base); ///  % modulus;
    base = (base * base); ///  % modulus;
    exp >>= 1;
  }
  return result;
}


ll hcf (ll s,ll a)
{
    ll m=max(s,a);
    ll n=min(s,a);
    if (m%n==0)
        return n;
    else
        return hcf(n,m%n);
}
