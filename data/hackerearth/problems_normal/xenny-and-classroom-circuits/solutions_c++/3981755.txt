#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int main()
{
	int t=0;
	 cin>>t;
	 while(t--)
	 {
		int n=0,count1=0,count2=0;
		cin>>n;
		string s,s2;
		cin>>s;
		s2=s;
		int i=0,j=0;
		while(i<s.size())
		{
			if(i%2==0 && s[i]!='B')
			{
				s[i]='B';
				count1++;
			}
			if(i%2!=0 && s[i]!='G')
			{
				s[i]='G';
				count1++;
			}
			if(j%2==0 && s2[j]!='G')
			{
					s2[i]='G';
					count2++;
				}
				if(j%2!=0 && s2[j]!='B')
				{
					count2++;
				}
				i++;j++;
		}
			count1/=2;
			count2/=2;
			int y=count1<count2?count1:count2;
			cout<<y<<endl;
	 }
	 return 0;
}
