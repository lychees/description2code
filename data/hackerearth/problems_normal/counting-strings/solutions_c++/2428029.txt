#include <iostream>
#define ll long long int
#include<algorithm>
using namespace std;

int main()
{
    ll t,i,j,n;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	ll count=0,sum=0;
    	for(i=0;i<s.size();i++)
    	{
    		if(s[i]=='a'||s[i]=='z')
    		{
    			sum+=count*(count+1)/2;
    			count=0;
    		}
    		else
    		count++;
    	}
    	sum+=count*(count+1)/2;
    	ll k=s.size();
    	cout<<(k*(k+1))/2-sum<<endl;
    	
    }
    return 0;
}
