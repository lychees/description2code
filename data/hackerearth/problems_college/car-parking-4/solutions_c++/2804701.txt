#include<bits/stdc++.h>

using namespace std;

const int N=1234567;

char s[N];

int main() {
    int test;
    scanf("%d", &test);
    while(test--) {
        scanf("%s", s);
        int len=strlen(s);
        int cnt=0;
        for(int i=0;i<len;i++) {
            bool ok=true;
            if(s[i]=='D' || s[i]=='B' || s[i]=='S') ok=false;
            if(i+1<len && s[i+1]=='B') ok=false;
            if(i+2<len && s[i+2]=='B') ok=false;
            if(i+1<len && s[i+1]=='S') ok=false;
            if(i-1>=0 && s[i-1]=='S') ok=false;
            if(ok) cnt++;
        }
        printf("%d\n", cnt);
    }
}
