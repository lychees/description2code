#include<iostream>

using namespace std;
double rootone=-1.0,roottwo=-1.0,rootthree=-1.0;
void cuberoot(double a,double b, double c, double d){
double f,h,g,r,s,t,u;
f = ((3.0*(c/a)) - ((b*b)/(a*a)))/3.0;
g = ((2.0*((b*b*b)/(a*a*a))) - (9.0*((b*c)/(a*a))) + (27.0*(d/a)))/27.0;
h = ((g*g)/4.0) + ((f*f*f)/27.0);
if(h>0){
    r = -(g/2.0) + sqrt( h);
    s = cbrt(r);
    t = -(g/2.0) - sqrt( h);
    u = cbrt(t);
    rootone = (s + u) - (b/(3.0*a));
    roottwo = -1.0;
    rootthree= -1.0;
//  cout<<r<<" "<<s<<" "<<t<<" "<<u<<endl;
}
else if(h==0&&g==0&&f==0){
rootone = roottwo = rootthree = cbrt(d/a) * -1.0;
}

else {
    double i,j,k,l,m,n,p;
    i = sqrt(((g*g)/4.0) - h);
    j = cbrt(i) ;
    k = acos(- (g / (2.0*i)));
    l = j * -1.0 ;
    m = cos(k/3.0);
    n= sqrt(3.0) * sin(k/3.0);
    p= (b/(3.0*a)) * -1.0;
    rootone = 2.0*j * cos(k/3.0)   -(b/(3.0*a));
    roottwo = l * (m + n) + p;
    rootthree =  l * (m - n) + p ;
   // cout<<i<<" "<<j<<" "<<k<<" "<<l<<" "<<m<<" "<<n<<" "<<p<<endl;
}

}
int main(void){
   double r , l,height ,volumeone=-1.0,volumetwo,tempvol;
   cin>>r;
   cin>>l;
   double a,b,c ,d;
   a=3.0;
   b=-3.0*r;
   c= -2.0*l*l;
   d=-r*l*l;
 /*  a=2.0;

   d = 24.0;*/

  // cout<<"value of a,b,c and d "<<endl;
  // cout<<a<<" "<<b<<" "<<c<<" "<<d<<endl;
cuberoot(a,b,c,d);
 //  cout<<"root is "<<rootone<<" "<<roottwo<<" "<<rootthree<<endl;
if(rootone>=0){
    height = sqrt((l*l)-((rootone-r)*(rootone-r)));
    tempvol = (3.141592653589*height*(rootone*rootone+r*r+rootone*r))/(3.0);
    volumeone = tempvol>volumeone?tempvol:volumeone;
  //  cout<<"value of volume one "<<tempvol<<endl;
}

if(roottwo>=0){
    height = sqrt((l*l)-((roottwo-r)*(roottwo-r)));
    tempvol = (3.141592653589*height*(roottwo*roottwo+r*r+roottwo*r))/(3.0);
    volumeone = tempvol>volumeone?tempvol:volumeone;
  //  cout<<"value of volume two "<<tempvol<<endl;
}

if(rootthree>=0){
    height = sqrt((l*l)-((rootthree-r)*(rootthree-r)));
    tempvol = (3.141592653589*height*(rootthree*rootthree+r*r+rootthree*r))/(3.0);
    volumeone = tempvol>volumeone?tempvol:volumeone;
  //  cout<<"value of volume three "<<tempvol<<endl;
}
volumetwo = (3.141592653589*r*r*l)/(1.0);
volumeone = volumeone>volumetwo?volumeone:volumetwo;
cout<<fixed<<setprecision(2)<<volumeone<<endl;
}
