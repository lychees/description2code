#include <bits/stdc++.h>

using namespace std;

#define maxN 1000
#define maxVal (int)1000000000

int a[maxN+5][maxN+5];
int n,m,x,y;
set<pair<int,pair<int,int> > > s;
set<pair<int,pair<int,int> > >::iterator it;
int d[maxN+5][maxN+5];

int ri[]={0,0,1,-1};
int cj[]={1,-1,0,0};

int compute()
{
	int k,ui,uj,vi,vj;
	d[0][0]=a[0][0];
	s.insert(make_pair(d[0][0],make_pair(0,0)));
	while(!s.empty())
	{
		it=s.begin();
		ui=(*it).second.first;
		uj=(*it).second.second;
		if(ui==x&&uj==y)
			break;
		s.erase(it);
		for(k=0;k<4;k++)
		{
			vi=ui+ri[k];
			vj=uj+cj[k];
			if(vi>=0&&vi<n&&vj>=0&&vj<m)
			{
				if(d[vi][vj]>(d[ui][uj]+a[vi][vj]))
				{
					if(d[vi][vj]!=maxVal)
						s.erase(s.find(make_pair(d[vi][vj],make_pair(vi,vj))));
					d[vi][vj]=d[ui][uj]+a[vi][vj];
					s.insert(make_pair(d[vi][vj],make_pair(vi,vj)));
				}
			}
		}
	}
	return d[x][y];
}

int main()
{
	int t,i,j,p,ans;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&m);
		for(i=0;i<n;i++)
			for(j=0;j<m;j++)
			{
				scanf("%d",&a[i][j]);
				d[i][j]=maxVal;
			}
		s.clear();
		scanf("%d%d%d",&x,&y,&p);
		x--;
		y--;
		ans=compute();
		if(ans<p)
			printf("YES\n%d\n",(p-ans));
		else
			printf("NO\n");
	}
    return 0;
}
