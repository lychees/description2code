#include<iostream>
using namespace std;
int main(){
	int k,w,i,j,count,c;
	cin>>k;
	string s[k];
	for(i=0;i<k;i++)
	cin>>s[i];
	cin>>w;
	string s1[w];
	for(i=0;i<w;i++)
	cin>>s1[i];
	for(i=0,count=0;i<w;i++){
		for(j=0,c=0;j<k;j++){
			if(!s1[i].compare(s[j])){
				c=1;
				break;
			}
		}
		if(c!=1){
			if(count>0)
				cout<<".";
				++count;
			cout<<(char)(s1[i][0]-32);
		}
	}
	return 0;
}
