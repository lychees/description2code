#include <iostream>
using namespace std ; 
int main()
{
	string s ; 
	int add = 0; 
	string p ; 
	while ( cin >> s )
	{
		int i ; 
		int sum = 0; 
		for ( i = 0 ; i< s.length() ; i++)
		{
			sum = sum + (s[i]- '0' ) ;
		}
		if ( add <= sum )
		{
		  p = s ; 
		  add = sum ;
		}
	
	}
	cout << p <<endl; 
	return 0 ;
}
