#include <bits/stdc++.h>
using namespace std;
//priority_queue<int, vector<int> ,greater<int> >pq;
//priority_queue<int>pq2;
int main()
{
    int n;
    cin>>n;
    int f[n],m[n];
    for(int i=0;i<n;i++){
    	cin>>f[i];
  //  	pq2.push(f[i]);
    }
    for(int i=0;i<n;i++){
    	cin>>m[i];
    //	pq.push(m[i]);
    }
   float count=0;
    sort(f,f+n);
    for(int i=0;i<n;i++){
    	for(int j=0;j<n;j++){
    		if(f[i]>m[j] && m[j]!=-1){
    			m[j]=-1;
    			count++;
    			break;
    		}else if(f[i]==m[j]){
    			count+=(0.5);
    		}
    	}
    }
    count*=100;
    cout<<count;
    return 0;
}
