#include <iostream>
using namespace std;

int main()
{
    long long int avg=0,n,t,m,x,a[100000];
    long long int i,j;
    cin>>t;
    
    while(t>0)
    {
    	avg=0;
    	t--;
    	cin>>n>>m>>x;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		avg+=a[i];
    	}
    	//cout<<avg<<" ";
    	long int y;
    	y=(n+1)*x-avg;
    	if(y>m)
    	cout<<"Impossible\n";
    	else if(y>0)
    	cout<<y<<"\n";
    	else
    	cout<<1<<"\n";
    }
    return 0;
}
