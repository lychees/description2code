#include <bits/stdc++.h>
#define ultrax_007 ios_base::sync_with_stdio(false);cin.tie(NULL)
#define ll long long
#define pb push_back
#define pr pair<ll int,ll int>
#define mp make_pair
using namespace std;
ll int gcd(ll int a ,ll int b)
{
   if(b==0) return a;
   a%=b;
   return gcd(b,a);
}
int main(){
   int n;
   cin>>n;
   vector<pair<ll int,ll int> > v;

   for(int i=0;i<n;i++){
    ll int x,y;
    cin>>x>>y;
    v.pb(mp(x,y));
   }
   ll int area=0;
   ll int bound=0;
   for(int i=0;i<n;i++){
        int j=i+1;
        if(j==n){j=0;}
    area+=(v[i].first*v[j].second)-(v[j].first*v[i].second);
    int p=abs(v[j].first-v[i].first);
    int q=abs(v[j].second-v[i].second);
    if(p==0){
        bound+=q;
    }else if(q==0){
        bound+=p;
    }else{
        bound+=gcd(p,q);
    }
    bound--;

   }
   area=abs(area/2);
   bound+=n;
   ll int ans=area-(bound/2)+1;
   cout<<ans;


}
