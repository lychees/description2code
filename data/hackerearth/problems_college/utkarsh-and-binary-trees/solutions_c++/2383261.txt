#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
#define mod 1000000007
LL DP[501][501];
int N,H,n,h;
int main()
{
    ios::sync_with_stdio(false);
    cin>>N>>H;
    for(int h = 0; h<H; h++)
    {
        DP[0][h] = 1;
        DP[1][h] = 1;
    }
    DP[1][0] = 0;
    for(int n = 2; n<N; n++)
    {
        for(int h = 2; h<H; h++)
        {   DP[n][h]=0;
            for(int l = 0; l < n; l++)
            {
                int r = n-(1+l);
                DP[n][h] = (DP[n][h] + DP[l][min(l,h-1)]*DP[r][h-1]) % mod;
            }
        }
    }
    LL ans =0;
    for(int root = 1; root<=N; root++)
    {
        ans = (ans + ((root*DP[root-1][H-1])%mod)*DP[N-root][H-1])% mod;
    }
    cout<<ans<<endl;
    return 0;
}
