#include <iostream>
#include <stdio.h>
#include <map>
using namespace std;

int D[200];
int mark[200];

int sum_dig(long long int i){
	int s = 0;
	while(i){
		s = s + i%10;
		i = i/10;
	}
	return s;
}

int fun(long long int v){
	
	if(v==1 || v==4)
	{
		D[v] = 1;
		mark[v] = 1;
		return D[v];
	}

	int z = sum_dig(v*v);

	if(D[z]==1 || D[z]==0){
		mark[z] = 1;
		return D[z];
	}

	if(mark[z]==1 && D[z]==-1)
	{
		D[z] = 0;
		return D[z];
	}

	if(z==1 || z==4){
		mark[z] = 1;
		D[z] = 1;
		return D[z];
	}


	if(D[z]==-1){
		mark[z] = 1;
		D[z] = fun(sum_dig(v*v));
		return D[z];
	}
	return 0;
}
int main(){
	D[1] = 1;
	D[4] = 1;
		
	for(int i=1; i<200; i++)
		D[i] = -1;

	for(int i=1; i<200; i++)
		D[i] = fun(i);

	int T;
	cin>>T;
	while(T){
		long long int n;
		scanf("%lld",&n);
		int z = sum_dig(n*n);
		if(n==1 || n==4 || D[z]==1)
			printf("YES\n");
		else
			printf("NO\n");
		
		T--;
	}
	
	return 0;
}
