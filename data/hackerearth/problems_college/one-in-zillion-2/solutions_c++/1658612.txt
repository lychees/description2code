#include<stdio.h>
#define gc() getchar()
//#define gc() getchar_unlocked()
inline int fastread()
{
int i=0;
char c=gc();
while(c<48) c=gc();
while(c>47)
{
i=i*10+c-48;
c=gc();
}
return i;
}
int main()
{
    int t,n,a[10100],k,l,i,j;
    unsigned long long x;
    t=fastread();
    while(t--)
    {
        n=fastread();
        n/=30;
        n+=1;
        a[0]=2;
        l=0;
        k=0;
        for(i=1;i<n;i++)
        {
            for(j=0;j<=l;j++)
            {
                k=a[j]*2+k;
                a[j]=k%10;
                k=k/10;
            }
            while(k!=0)
            {
                a[++l]=k%10;
                k/=10;
            }
        }
        for(i=l;i>=0;--i) printf("%d",a[i]);
        printf("\n");
    }
    return 0;
}
