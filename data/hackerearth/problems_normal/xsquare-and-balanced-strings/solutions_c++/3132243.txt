#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000000007
char arr[100005];
int a[27];
int main()
{
    int t,i,n;
    sd(t);
    while(t--)
    {
        ss(arr);
        n=strlen(arr);
        for(i=0;i<=26;i++)
            a[i]=0;
        for(i=0;i<n;i++)
            a[(int)(arr[i]-'a')]++;
        int flag=0;
        for(i=0;i<26;i++)
        {
            if(a[i]%2==1)
            {
                flag=1;
                break;
            }
        }
        if(flag==0)
            printf("1\n");
        else
            printf("-1\n");
    }
	return 0;
}
