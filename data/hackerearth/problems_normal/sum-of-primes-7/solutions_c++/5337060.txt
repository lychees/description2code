#include <iostream>
#include <cstdio>
using namespace std;
#define max 1000006
#define ll long long 
ll a[max];
bool p[max];

void sieve()
{
	int i,j;
	p[0]=p[1]=false;
	for(i=2;i<max;i++)	p[i]=true;

	for(i=2;i*i<=max;i++)
	{
		if(p[i]==true)
		{
			for(j=i*i;j<max;j+=i)
				p[j]=false;
		}
	}
	a[0]=a[1]=0;
	for(i=2;i<max;i++)
	{
		if(p[i]==true)
			a[i]=a[i-1]+i;
		else
			a[i]=a[i-1];
	}
}

int main()
{
	int n,l,r;
	long long sum;
	scanf("%d",&n);
	sieve();
	while(n--)
	{
		scanf("%d %d",&l,&r);
		sum=a[r]-a[l-1];
		printf("%lld\n",sum);
	}
	return 0;
}
