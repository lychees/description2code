inline void scan_f(int *a)
{
    char c = 0;
    while(c<33)
        //c = fgetc_unlocked(stdin);
        c = getc(stdin);
    *a = 0;
    while(c>33)
        {
            *a = (*a)*10 + c - '0';
    //c = fgetc_unlocked(stdin);
    c = getc(stdin);
    }
}

int position[105];

int main(){
    int n,m,a[105][105];
    int row=0;
    scan_f(&n);
    scan_f(&m);
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            scan_f(&a[i][j]);
        }
    }
    while(position[n-1]<m){
        for(int i=0;i<=row;i++){
            if(position[i]<m)
            printf("%d ",a[i][position[i]]);
            position[i]++;
        }
        printf("\n");
        if(row<n-1)
            row++;
    }
    return 0;
}
