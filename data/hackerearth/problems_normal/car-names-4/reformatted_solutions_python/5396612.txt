for _ in xrange(input()):
    name = raw_input()
    if len(name) % 3 != 0:
        print 'Not OK'
    else:
        r = len(name) / 3
        a, b, c = set(name[:r]), set(name[r:2*r]), set(name[2*r:])
        if len(a) == 1 and len(b) == 1 and len(
                c) == 1 and not a == b and not a == c and not b == c:
            print 'OK'
        else:
            print 'Not OK'
