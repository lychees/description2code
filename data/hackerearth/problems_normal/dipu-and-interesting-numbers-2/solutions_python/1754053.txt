import math
def perfectSqrCount(start,end):
	start_sqrt=long(math.sqrt(start))
	end_sqrt=long(math.sqrt(end))
	if start_sqrt * start_sqrt!= start:
		start_sqrt+=1
	if end_sqrt * end_sqrt> end:
		end_sqrt-=1	
	return (end_sqrt-start_sqrt+1)
