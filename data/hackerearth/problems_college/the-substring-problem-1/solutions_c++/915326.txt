#include<iostream>
#include<algorithm>
#include<math.h>
#include<string.h>
#include<stack>
#include<queue>
#include<set>
#include<stdio.h>
using namespace std;
int main()
{
    int t,n,i,j,k,maxvl,maxchar,maxch;
    cin>>t;
    while(t--)
    {
	    char str[105];
	    int *h=new int[26];
	    set<char> s;
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>str;
    		int len=strlen(str);
    		for(j=0;j<len;j++)
    		   s.insert(str[j]);
    		while(!s.empty()) 
			{
				h[*s.begin() - 97]++;
				s.erase(s.begin());
			}  
    	}
    	maxvl=h[0];
    	maxchar=0;
    	for(k=1;k<26;k++)
    	{
    		if(h[k]>maxvl)
    		{
    			maxvl=h[k];
    			maxchar=k;
    		}
    	}
    	maxch=maxchar+97;
    	printf("%c\n",maxch);
    }
	return 0;
}
