#include <iostream>
using namespace std;

long long bits(int n)
{
	long long c=0,rest=0;
	long long ans=0;
	while(n>0)
	{
		if(n&1)
		{
			if(c==0) ans++;
			else{
				ans+=(1<<(c-1))*c+rest+1;
			}
			rest+=(1<<c);
		}
		n>>=1;
		c++;
	}
	return ans;
}
int main()
{
    int t,a,b;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	long long x=bits(a-1),y=bits(b);
    	//cout<<x<<" "<<y<<endl;
    	cout<<(y-x)<<endl;
    }
    return 0;
}
