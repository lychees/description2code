#include <stdio.h>
 
unsigned long long  abs(unsigned long long a,unsigned long long b)
{
	if(a>b)
	return a-b;
	else
	return b-a;
}
int main()
{
	int t,p;
	scanf("%d",&t);
	unsigned long long x,y,x1,y1,x2,y2,r,c,a,l;
	while(t--){
	scanf("%llu %llu %llu %llu %llu %llu",&x,&y,&x1,&y1,&x2,&y2);
 
	scanf("%d",&p);
	l=abs(x1,x2);
	if(l>abs(x,l))
	r=abs(x,l);
	else
	r=l;
	l=abs(y1,y2);
	if(l>abs(y,l))
	c=abs(y,l);
	else
	c=l;
	a=(r+c)*p;
	printf("%llu\n",a);
	if(a<=1000)
	printf("You saved the group.\n");
	else
	printf("Let go of the group.\n");
	}
    return 0;
}
