#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n, p, current, previous;
    	cin>>n>>p;
    	char c;
    	current = p;
    	while(n--)
    	{
    		cin>>c;
    		if(c == 'P')
    		{
    			cin>>p;
    			previous = current;
    			current = p;
    		}
    		else
    		{
    			long long int tmp;
    			tmp = previous;
    			previous = current;
    			current = tmp;
    		}
    	}
    	cout<<"Player "<<current<<endl;
    }
    return 0;
}
