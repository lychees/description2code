#include<cstdio>
#include<cstring>
using namespace std;
char n[10000000],m[10000000];
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		
		scanf("%s %s",&n,&m);
		//printf("%s %s\n",n,m);
		int l1=strlen(n);
		int l2=strlen(m);
		int i=l1-1,j=l2-1,cnt=0,c=0;
		for(;i>=0 && j>=0;i--,j--)
		{
			c=((n[i]-48)+(m[j]-48)+c)/10;
			if(c>=1)
				++cnt;
		}
		if(i==-1)
		{
			while(c!=0 && j>=0)
			{
				c=(m[j]-48+c)/10;
				++cnt;
				j--;
			}
		}
		else
		{
			while(c!=0 && i>=0)
			{
				c=(n[i]-48+c)/10;
				++cnt;
				i--;
			}
		}
		if(cnt==0)
			printf("No carry operation\n");
		else if(cnt==1)
			printf("1 carry operation\n");
		else
			printf("%d carry operations\n",cnt);
	}
	return 0;
}
