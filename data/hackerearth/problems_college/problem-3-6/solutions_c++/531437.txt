int main()
{
  
  long int n;cin>>n;
  vector<long int> home(100005,0); //no. of teams with home jersey no. i
  vector<long int> away(100005,0); //no. of teams wid away jersey no. j
  

 vector< pair<long int,long int> > info(n);
 for (int i=0;i<n;i++){
  cin>>info[i].first>>info[i].second;
  ++home[info[i].first];++away[info[i].second];
 }

/*   long int sumh=0,suma=0;

 vector< pair<long int,long int> > ans(n);
 for (int i=1;i<=n;i++){

  ans[i-1].first= n-1+home[info[i-1].second];
  ans[i-1].second= 2*(n-1)-ans[i-1].first ;

 }

 for (int i=0;i<n;i++)
  cout<<ans[i].first<<" "<<ans[i].second<<"\n";


   return 0;
}
