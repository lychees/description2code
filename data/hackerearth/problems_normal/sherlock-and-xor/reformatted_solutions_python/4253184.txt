t = int(raw_input())


def count(lst):
    even = odd = 0
    for i in lst:
        if i % 2 == 0:
            even += 1
        else:
            odd += 1
    return odd*even


for i in range(t):
    n = int(raw_input())
    s = raw_input()
    b = s.split(' ')
    lst = []
    for j in range(n):
        lst.append(int(b[j]))
    n1 = count(lst)
    print n1
