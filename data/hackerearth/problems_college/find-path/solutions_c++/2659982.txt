#include <iostream>
using namespace std;

int main()
{
    int tc,n,i;
    cin>>tc;
    while(tc--)
    {
    cin>>n;
    int arr[n];
    for(i=0;i<n;i++)
    cin>>arr[i];
    int c=0;
   
    for(i=1;i<n;i++)
    {
    	if(arr[i]<0||arr[i-1]>=arr[i])
    	c++;
    }
    cout<<c<<endl;
    }
    return 0;
}
