   "Hard Work Beats Talent
   When Talent Doesn't Work Hard"
*/
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,flag,i,j,temp,k;
    int arr[100];
    scanf("%d",&n);
    for(i=0;i<n;i++)
        scanf("%d",&arr[i]);
    for(j=2;j<=400000000;j++)
    {
        flag=1;
        temp=arr[0]%j;
        for(k=1;k<n;k++)
        {
            if(arr[k]%j==temp)
                continue;
            else
            {
                flag=0;
                break;
            }
        }
        if(flag==1)
            printf("%d ",j);
    }
    printf("\n");
    return 0;
}
