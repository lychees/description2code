#include <iostream>
using namespace std;
#include<string>
#define ll long long
#include<bits/stdc++.h>
int main()
{
    ll test,i,j;
    string s;
    cin>>test;
    while(test--)
    {
    	cin>>s;
    //	cout<<s<<endl;
    	ll count=0;
    	for(i=0;i<s.size();i++)
    	{
    		if(s[i]=='B' || s[i]=='S' || s[i]=='D'){continue;}
    		else if(s[i+1]=='B' || s[i+2]=='B' || s[i+1]=='S' || s[i-1]=='S'){continue;}
    		else{
    			//cout<<i<<" ";
    			count++;}
    	}
    	//cout<<endl;
    	cout<<count<<endl;;
    }
    return 0;
}
