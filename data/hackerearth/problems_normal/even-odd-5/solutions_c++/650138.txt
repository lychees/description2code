#include <bits/stdc++.h>
#define ll long long
using namespace std;

void solve()
{
    ll r,c;
    ll even = 0;
    ll odd = 0;
    scanf("%lld%lld",&r,&c);
    if(c>=r)
    {
        if(r&1)
        {
            odd = (r+1)>>1;
            odd = odd*odd;
            even = (r-1)>>1;
            even = even*(even+1);
        }
        else
        {
            odd = r>>1;
            odd = odd*odd;
            even = r>>1;
            even = even*(even+1);
        }
    }
    else
    {
        ll a = r;
        ll b = c;
        r = c;
        if(r&1)
        {
            odd = (r+1)>>1;
            odd = odd*odd;
            even = (r-1)>>1;
            even = even*(even+1);
        }
        else
        {
            odd = r>>1;
            odd = odd*odd;
            even = r>>1;
            even = even*(even+1);
        }
        for(int i=b+1;i<=a;i++)
        {
            if(i&1)odd+=c;
            else even+=c;
        }
    }
    printf("%lld %lld\n",even,odd);
}

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        solve();
    }
}
