#include <bits/stdc++.h>
#define ll long long
using namespace std;
int myfind(int x)
{
	int i,ans=0;
	for(i=2;(i*i)<=x;i++)
	{
		if((x%i)==0)
		{
			ans++;
			while((x%i)==0)
			{
				x/=i;
			}
		}
	}
	if(x!=1)
	ans++;
	return ans;
}
bool isprime(int x)
{
	if(x==1)
	return 0;
	int i;
	bool flag=1;
	for(i=2;(i*i)<=x;i++)
	{
		if((x%i)==0)
		{
			flag=0;
			break;
		}
	}
	return flag;
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int test;
	cin>>test;
	while(test--)
	{
		int a,b;
		cin>>a>>b;
		int lcm = (a*b)/__gcd(a,b);
		int f = myfind(lcm);
		if(isprime(f))
		{
			cout<<"Yes"<<endl;
		}
		else
		cout<<"No"<<endl;
	}
	return 0;
}
