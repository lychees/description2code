#include<bits/stdc++.h>
using namespace std;
bool compare(const pair<pair<int,int>,int> &x,const pair<pair<int,int>,int> &y){
    if(x.first.first!=y.first.first)
        return x.first.first<y.first.first;
    else
        return x.second < y.second;        
}
int main()
{
    int t;
    cin>>t;
    while(t--){
        int n,m,i,s,e,c,j;
        int hash[105];
        memset(hash,0,sizeof(hash));
        cin>>n>>m;
        pair<pair<int,int>,int> p[n];
        for(i=0;i<n;i++){
            cin>>s>>e;
            p[i].first.first = s;
            p[i].first.second = e;            
            p[i].second = i;
        }
        sort(p,p+n,compare);        
        for(i=0;i<n;i++){
            s = p[i].first.first;
            e = p[i].first.second;
            c = 0;
            for(j=0;j<=s;j++){
                c += hash[j]; 
            }
            if(c<m){
                hash[s]++;
                hash[e]--;
            }
        }
        c = 0;
        int f =0;
        double ans=0,x;
        for(i=0;i<=100;i++){            
            c += hash[i];
            if(c>=m){
                f=1;
            }
            x = c;
            if(c<2){
                ans += (x*10.0);
            }
            else if(c==2){
                ans += (x*9.5);
            }
            else{
                ans += (x*9.3);
            }
        }               
        ans = round(ans);
        c = ans;
        cout<<c;
        if(f)
            cout<<" Cab was full";
        cout<<endl;
    }
    return 0;
}
