#include<iostream>
#include<cstdio>
#include<cmath>

using namespace std;

int main()
{
    int Test;
    scanf("%d",&Test);
    while(Test--)
    {
        int N,M,K=5;
        scanf("%d%d",&M,&N);
        if(N==0)
        {
            printf("0\n");
            continue;
         }
        if(N<=M&&N>=1)
            printf("%d\n",(2*K));
        else
        {
            if(N%M==0)
            {
                N=2*(N/M);
                printf("%d\n",(N*K));
            }
            else
            {
               N=ceil(2*((float)N/(float)M));
               printf("%d\n",(N*K));
            }
        }
    }
    return 0;
}
