//#include "stdafx.h"
#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
const int N = 1000001;
long long int a[N];
int main(int argc, char** argv) {
	//freopen("in.txt", "r", stdin);
	//freopen("out.txt", "w", stdout);
	long long int n, q;
	scanf(" %lli %lli", &n, &q);
	for (int i = 1; i <= n; i++) {
		scanf(" %lli", a + i);
	}
	for (long long int it = 0; it < q; it++) {
		int op;
		scanf(" %d", &op);
		if (op == 0) {
			long long int t1, t2;
			scanf(" %d %d", &t1, &t2);
			a[t1] = t2;
		}
		else {
			long long int ele;
			scanf(" %d", &ele);
			long long int res = -1;
			for (long long int i = 1; i <= n; i++) {
				if (a[i] >= ele) {
					res = i; 
					break;
				}
			}
			printf("%d\n", res);
		}
	}
	return 0;
}
