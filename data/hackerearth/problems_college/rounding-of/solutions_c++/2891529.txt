#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
        int n,b,m,q;
    	cin>>n>>b;
    	m=n%b;
    	q=n/b;
    	if(b%2==0){
    		if(m>=(b/2))
    		  m=b;
    		 else
    		  m=0;
    	}
    	else{
    		if(m>(b/2))
    		  m=b;
    		 else
    		  m=0;
    	}
    	cout<<q*b+m<<endl;
    }
    return 0;
}
