#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		string w,r;
		int l1,l2;
		cin>>w>>r;
		l1=w.size();
		l2=r.size();
		sort(r.begin(),r.begin()+l2);
		for(int k=0;k<l1;k++)
		{
		for(int j=0;r[j]!='\0';j++)
			{
				if(r[j]<w[k])
					{
						w[k]=r[j];
						r.erase(r.begin()+j);
						break;
					}
			}
		}
	cout<<w;
	cout<<"\n";
	}
	return 0;
}
