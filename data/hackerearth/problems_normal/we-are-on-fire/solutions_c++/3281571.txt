#include <bits/stdc++.h>
using namespace std;
int a[1005][1005];
int visit[1005][1005];
int xmove[4]={1,0,-1,0};
int ymove[4]={0,1,0,-1};
int cnt;
int n,m;
void dfs(int x,int y)
{
	visit[x][y]=1;
	cnt--;
	for(int i=0;i<4;i++)
	{
		int xx=x+xmove[i];
		int yy=y+ymove[i];
		if(xx>=1&&xx<=n&&yy>=1&&yy<=m&&visit[xx][yy]==0&&a[xx][yy]==1)
		{
			dfs(xx,yy);
		}
	}
	
}
int main()
{
	int q,x,y;
	scanf("%d%d%d",&n,&m,&q);
	cnt=0;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			scanf("%d",&a[i][j]);
			if(a[i][j]==1)
			cnt++;
		}
	}
	while(q--)
	{
		scanf("%d%d",&x,&y);
		if(a[x][y]==0||visit[x][y]==1)
		{
			cout<<cnt<<"\n";
			continue;
		}
		
		dfs(x,y);
		cout<<cnt<<"\n";
		
	}
	return 0;
}
