#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

#define ll long long int

int main()
{
    ll t,n;
    cin>>t;
    
    int now,prev,i;
    bool flag;
    
    while(t--)
    {
    	flag = true;
    	now = prev = 0;
    	cin>>n;
    	
    	if(n%21==0)
    	{
    		cout<<"The streak is broken!"<<endl;
    		continue;
    	}
    	
    	//cout<<"Size"<<log10(n)+1<<endl;
    	
    	for(i=0;n!=0;i++)
    	{
   
    		now=n%10;
    		//cout<<"Now :"<<now<<endl;
    		//cout<<"Prev :"<<prev<<endl;
    		if(now==2 && prev ==1)
    		{
    			cout<<"The streak is broken!"<<endl;
    			flag=false;
    			break;
    		}
    		
    		prev=now;
    		n=n/10;
    	
    	}
    	
    	if(flag)
    		cout<<"The streak lives still in our heart!"<<endl;
    	
    }
    
    return 0;
}
