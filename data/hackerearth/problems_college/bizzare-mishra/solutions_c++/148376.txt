#include <cstdio>
#include <iostream>
#include <cstring>

#define MAX 10000

int main ()
{
	int t,l,i,n1,n2;
	char str[MAX + 2], cpstr[MAX + 2],c;
	
	scanf ("%d", &t);
	
	while (t--)
	{
		n1 = n2 = 0;
		
		scanf ("%d", &l);
		scanf ("\n%s", str);
		
		c = str[0];
		
		strcpy (cpstr, str);
		
		if (c == 'Y')
		{
			//make like YNYN
			for (i = 0;i < l - 1;i++)
			{
				if (str[i] == str[i + 1])
				{
					if (i % 2)
						str[i + 1] = 'Y';
					else
						str[i + 1] = 'N';
					n1++;
				}
			}
			
			//make like NYNY
			n2 = 1;
			cpstr[0] = 'N';
			for (i = 0;i < l - 1;i++)
			{
				if (cpstr[i] == cpstr[i + 1])
				{
					if (i % 2)
						cpstr[i + 1] = 'N';
					else
						cpstr[i + 1] = 'Y';
					n2++;
				}
			}
		}
		else
		{
			//make like YNYN
			n1 = 1;
			str[0] = 'Y';
			for (i = 0;i < l - 1;i++)
			{
				if (str[i] == str[i + 1])
				{
					if (i % 2)
						str[i + 1] = 'Y';
					else
						str[i + 1] = 'N';
					n1++;
				}
			}
			
			//make like NYNY
			for (i = 0;i < l - 1;i++)
			{
				if (cpstr[i] == cpstr[i + 1])
				{
					if (i % 2)
						cpstr[i + 1] = 'N';
					else
						cpstr[i + 1] = 'Y';
					n2++;
				}
			}
		}
		printf ("%d\n", (n1 < n2) ? n1 : n2);
	}
	return 0;
}
