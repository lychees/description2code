#include <bits/stdc++.h>
using namespace std;


int main(){
    int T;
    long long a,b,c;
    cin>>T;
    while(T--){
        cin>>a>>b>>c;
        if( ((a*a+b*b==c*c) || (a*a+c*c==b*b) || (c*c+b*b==a*a) ) && (__gcd(a,__gcd(b,c))==1) ) puts("YES");
        else    puts("NO") ;
    }
}
