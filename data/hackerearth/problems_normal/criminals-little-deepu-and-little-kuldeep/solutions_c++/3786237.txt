#include <bits/stdc++.h>
#define ll long long int
#define gt(i) scanf("%lld", &i)

using namespace std;

int main()
{
	ll test;
	gt(test);
	while(test--) {
		ll i, j, k, n, m;
		map <ll, ll> mp;
		map <ll, ll> :: iterator pos;
		gt(n);
		for(i = 0; i < n; i++) {
			gt(j);
			mp[j]++;
		}
		ll maxv = 0;
		for(pos = mp.begin(); pos != mp.end(); pos++) {
			if(maxv < pos->second)
				maxv = pos->second;
		}
		cout << maxv << endl;
	}
	return 0;
}
