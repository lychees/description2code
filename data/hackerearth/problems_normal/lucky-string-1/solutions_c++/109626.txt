#include <iostream>
#include <cmath>

using namespace std;

int findN(long long int n )
{
    int i=1;
    while(n > 0)
    {
            n -= i*pow(2.0,i);
            i++;
    }
    return i-1;
}


int main()
{
    int b;
    long long int  T, blocknum, n, S;
    cin>>T;
    while(T--){
    //cin>>n;
    scanf("%lld",&n);
    b=findN(n);
    S = 2 + pow(2.0,b)*(b-2);
    blocknum = ceil((1.0*(n-S))/b);
    long long int offset = n - S - b*(blocknum - 1);
    
    if((blocknum-1) & ((long)1 << (b-offset)))
            printf("Earth\n");
    else
            printf("Hacker\n");
    }
    return 0;
}
