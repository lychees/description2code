#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll size = 1e6 + 1;
ll arr[1000001];
ll mod = 1e9+7;
int main()
{
	arr[0] = 1;
	arr[1] = 1;
	for(ll i=2;i<size;i++)
	{
		arr[i] = (arr[i-1]%mod + (i-1)*arr[i-2]%mod)%mod;
	}
	ll t;
	cin>>t;
	while(t--)
	{
		ll n;
		cin>>n;
		cout<<arr[n]<<endl;
	}
}
