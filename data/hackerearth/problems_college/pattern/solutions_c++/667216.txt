#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int k=n-2;
    for(int i=1;i<=n;i++)
    {cout<<"*";}
    cout<<endl;
    for(int j=1;j<=n-2;j++)
    {
		for(int p=1;p<=k;p++)
		{
			cout<<" ";
		}k--;
		cout<<"*";
		cout<<endl;
    	
    }
    for(int m=1;m<=n;m++)
    {cout<<"*";}
    return 0;
}
