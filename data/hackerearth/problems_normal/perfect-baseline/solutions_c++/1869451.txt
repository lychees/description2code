#include<iostream>
#include<stdlib.h>

using namespace std;
int main(){

	int n,k,test;
	cin>>test;
	while(test--){
	cin>>n>>k;
	
	string s[100003];
	
	int i,j,m,cost=0;
	char ans,c;
	
	string res="";
	
	for(i=0;i<n;i++){
		cin>>s[i];
	}
	
	for(i=0;i<k;i++){
		
		int maxcost = 1000000000;

		for( c = 'a';c<='z'; c++){
			cost=0;
			
			for(m=0;m<n;m++){
				cost+= abs(c-s[m][i]);
			}
			
			if(cost<maxcost){
				maxcost = cost;
				ans = c;
			}
		}
	//	cout<<ans<<endl;
		res+=(char)ans;
	}
	
	cout<<res<<endl;
	}
	return 0;
}
