#include <bits/stdc++.h>
using namespace std;

int T, N, m1, m2, L, S, dp[1000005];

int main () {
	scanf("%d", &T);
	while (T--) {
		scanf("%d%d%d", &N, &m1, &m2);
		for (int i = 1; i <= N; i++) {
			dp[i] = 0;
			if (i >= m1 && dp[i-m1] == 0) dp[i] = 1;
			if (i >= m2 && dp[i-m2] == 0) dp[i] = 1;
		}
		printf("%d\n", dp[N]);
	}
	return 0;
}
