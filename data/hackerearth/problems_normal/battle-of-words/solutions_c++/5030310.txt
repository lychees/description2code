#include <iostream>
#include <string.h>
using namespace std;

int main()
{
    int t;

    cin>>t;

    cin.ignore();

    while(t--)
    {
        int ar[124];
        bool pos=false,neg=false;
        string s1,s2;

        getline(cin,s1);
        getline(cin,s2);

        memset(ar,0,sizeof(ar));

        for(int i=0;s1[i]!=NULL;i++)
        {
            if(s1[i]!=' ')
            {
                ar[s1[i]]++;
            }
        }

        for(int i=0;s2[i]!=NULL;i++)
        {
            if(s2[i]!=' ')
            {
                ar[s2[i]]--;
            }
        }

        for(int i=97;i<=122;i++)
        {
            if(ar[i]>0)
            {
                pos=true;
            }else if(ar[i]<0)
            {
                neg=true;
            }
        }

        if(pos && neg)
        {
            cout<<"You draw some.\n";
        }else if(pos)
        {
            cout<<"You win some.\n";
        }else
        {
            cout<<"You lose some.\n";
        }
    }


    return 0;
}
