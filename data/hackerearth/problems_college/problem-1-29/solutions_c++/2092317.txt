#include <iostream>
using namespace std;
bool safe(int maze[51][51],int i,int j,int n)
{
if(i>=0 && i<n && j>=0 && j<n &&maze[i][j]==1)
return true;
return false;
	
	
}
bool solveMazeUtil(int maze[51][51],int x,int y,int n)
{
	if(x==n-1 && y==n-1)
	return true;
	if(safe(maze,x,y,n))
	{
		if(solveMazeUtil(maze,x+1,y,n))
		return true;
		if(solveMazeUtil(maze,x,y+1,n))
		return true;
		return false;
	}
	return false;
	
}
bool solveMaze(int maze[51][51],int n)
{
if(!solveMazeUtil(maze,0,0,n))
return false;
return true;

}
	

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    int n;
    int i,j;
    int maze[51][51];
    cin>>t;
    bool solution;
    while(t--)
    {
    	solution=false;
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<n;j++)
    		{
    			cin>>maze[i][j];
    		}
    	}
    	solution=solveMaze(maze,n);
    	if(solution)
    	cout<<"POSSIBLE"<<endl;
    	else
    	cout<<"NOT POSSIBLE"<<endl;
    }
    return 0;
}
