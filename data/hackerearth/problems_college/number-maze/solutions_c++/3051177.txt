#include<iostream>
using namespace std;
int main()
{
    long int a,b,flag=0;
    cin>>a>>b;
    for(int i=a+1;i<b;i++)
    {
        if((i%3!=0 && i%5!=0) || (i%3==0 && i%5==0))
        {
            cout<<i;

        }
        if(i%3==0 && i%5==0)
            flag=1;
            else if(i%5==0 || i%3==0)
                flag=0;
                else
                    flag=1;

        if(i<b-1 && flag==1)
            cout<<',';


    }

}
