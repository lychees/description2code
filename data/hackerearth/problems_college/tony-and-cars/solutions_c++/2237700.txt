using namespace std;

int main(int argc, char const *argv[])
{
	long long t;
	cin>>t;
	while(t--){
	long long n;
	cin>>n;
	long long j=n,dig=0,sum=0;
	while(j>0){

		dig++;
		j=j/10;
	}
	long long gret=9,sm=1;
	for(int i=1;i<dig;i++){
		sum+=(gret-sm+1)*i;
		gret=(gret*10)+9;
		sm*=10;
	}
	sum+=(n-sm+1)*dig;
	cout<<sum<<endl;
}
	return 0;
}
