#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t!=0)
    {
    	string a,b;
    	cin>>a>>b;
    	
    	vector<int> arra(26,0);
    	vector<int> arrb(26,0);
    	
    	for(int i=0;i<a.size();i++)
    	arra[a[i]-97]++;
    	
    	for(int i=0;i<b.size();i++)
    	arrb[b[i]-97]++;
    	
    	int flag=0;
    	
    	for(int i=0;i<26;i++)
    	{
    		if(arra[i]!=arrb[i])
    		{
    			flag=1;
    			break;
    		}
    	}
    	
    	if(flag)
    	cout << "No\n";
    	else
    	cout << "Yes\n";
    	
    	t--;
    }
}
