#include<bits/stdc++.h>
using namespace std;
long int count( long int S[], long int m, long int n )
{
	long int i;
    long int table[n+1];
    memset(table, 0, sizeof(table));
    table[0] = 1;
    for( i=0; i<m; i++)
        for(int j=S[i]; j<=n; j++)
            table[j] += table[j-S[i]];

    return table[n];
}
int main()
{
long int t,n;
scanf("%ld",&t);
long int arr[]={1,8,27,64,125,216,343,512,729,1000,1331,1728,2197,2744,3375,4096,4913,5832,6859,8000,9261};
long int m = sizeof(arr)/sizeof(arr[0]);
while(t--)
{
scanf("%ld",&n);
cout<<count(arr,m,n)<<endl;
}
return 0;
}
