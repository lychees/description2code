for _ in range(input()):
    arr = []
    for i in range(input()):
        arr.append(raw_input())
    hor = 0
    ver = 0
    for item in arr:
        if item != item[::-1]:
            break
    else:
        ver = 1
    if arr == arr[::-1]:
        hor = 1
    if hor and ver:
        print "BOTH"
    elif hor:
        print "HORIZONTAL"
    elif ver:
        print "VERTICAL"
    else:
        print "NO"
