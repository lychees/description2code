#include <iostream>
#include <algorithm>
#include <limits.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i = 0 ; i < n ; i++)
    		cin>>a[i];
    	sort(a,a+n);
    	int max = INT_MIN;
    	int min = INT_MAX;
    	for(int i = 0 ; i < n/2 ; i++)
    	{
    		if(a[i] + a[n-i-1] > max)
    			max = a[i] + a[n-i-1];
    		if(a[i] + a[n-i-1] < min)
    			min = a[i] + a[n-i-1]; 
    	}
    	cout<<max-min<<endl;
    }
}
