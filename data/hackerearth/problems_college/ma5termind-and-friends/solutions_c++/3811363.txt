#include<bits/stdc++.h>
#define ll long long
#define pb push_back
#define f first
#define s second
#define mk make_pair
#define ms(a,v) memset(a,v,sizeof(a))
using namespace std;

const int lim=1e5+5;
const int M=1e9+7;

typedef struct f
{
	ll l,r,c;
}f;

int main()
{
  ios_base::sync_with_stdio(false);cin.tie(NULL);
 // freopen("input.txt", "r", stdin);
  ll t,i,j,k;
  cin>>t;
  while(t--)
  {
  	ll n,m;
  	cin>>n>>m;
  	
	  f fn[lim];
  	
  	for(i=1;i<=m;i++)
  	cin>>fn[i].l>>fn[i].r>>fn[i].c;
  	
  	ll q;
  	cin>>q;
  	
  	ll freq[lim]={0};
  	while(q--)
  	{
  		ll x,y;
  		cin>>x>>y;
  		freq[x]++;
		freq[y+1]--;	
	}
	
	for(i=1;i<=m;i++)
	freq[i]+=freq[i-1];
	
	//for(i=1;i<=m;i++)
	//cout<<freq[i]<<"  ";
	ll sum[lim]={0};
    for(i=1;i<=m;i++)
    {
    	ll l=fn[i].l,r=fn[i].r,c=fn[i].c;
    	c*=(freq[i]);
    	
    	sum[l]+=c;
    	sum[r+1]-=c;
	}
	
	for(i=1;i<=n;i++)
	{
		sum[i]+=sum[i-1];
	}
	
	for(i=1;i<=n;i++)
	cout<<sum[i]<<" ";
	cout<<endl;
  }
  return 0;
}
