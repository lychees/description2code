#include <iostream>
#include <bits/stdc++.h>
#define ll long long
using namespace std;

bool dp[100002][102];
int main()
{
    int tc,n,a[100],i,j,q;
    cin>>n;
    for(i=1;i<=n;i++)
    cin>>a[i];
    for(i=0;i<n;i++)
    dp[0][i]=true;
    for(i=1;i<100002;i++)
    dp[i][0]=false;
    for(i=1;i<100002;i++)
    {
    	for(j=1;j<=n;j++)
    	{
    		//if(i<a[j])
    		dp[i][j]=dp[i][j-1];
    		if(i>=a[j])
    		dp[i][j]=dp[i][j]||dp[i-a[j]][j-1];
    	}
    }
    cin>>tc;
    while(tc--)
    {
    	cin>>q;
    	if(dp[q][n])
    	cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
    }
    
    
    
    return 0;
}
