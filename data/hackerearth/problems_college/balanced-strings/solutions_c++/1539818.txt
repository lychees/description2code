#include <bits/stdc++.h>

using namespace std;
int has[100];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        memset(has,0,sizeof(has));
        string s;
        cin>>s;
        for(int i = 0; i < s.size(); i++)
        {
            has[s[i] - 97]++;
        }
        int flag = 0;
        for(int i = 0; i < 25; i++)
        {
            if( has[i] != has[i+1])
            {
                flag = 1;
            }
        }
        if( flag)
        {
            cout<<"No"<<endl;
        }
        else {
            cout<<"Yes"<<endl;
        }
    }


}
