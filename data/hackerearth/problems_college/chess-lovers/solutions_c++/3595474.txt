#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,x1,y1,x2,y2;
	cin>>t;
	while(t--)
	{
		cin>>x1>>y1>>x2>>y2;
		int diff=abs(x1-x2)+abs(y1-y2);
		if(diff&1)
		cout<<-1<<'\n';
		else
		{
			if((x1+y1==x2+y2)||(x1-x2==y1-y2))
			cout<<0<<'\n';
			else
			cout<<1<<'\n';
		}
	}
    return 0;
}
