#include <iomanip>
#include <math.h>
#include <cstdlib>
#include <string>
#include <iostream>
#include <map>

using namespace std;

int main() {
    int l, r, max = 0;
    cin >> l;
    cin >> r;
    for (int i = l; i <= r; i++) {
        for (int j = i; j <= r; j++) {
            if ( (i ^ j) > max ) {
                max = i ^ j;
            }
        }
    }
    cout << max << endl;
    return 0;
}
