#include <iostream>
using namespace std;
long long arr[100002];
long long dpInc[100002];
long long dpDec[100002];




int main()
{
	long n,m,l,r;
	cin>>n>>m;
	for(long i=0;i<n;i++)
		cin>>arr[i];
	dpInc[0]=dpDec[0]=0;
	
	for(long i=1;i<n;i++){
		if(arr[i-1]<arr[i]){
			dpInc[i]=dpInc[i-1];
			dpDec[i]=i;
		}
		else if(arr[i-1]>arr[i]){
			dpDec[i]=dpDec[i-1];
			dpInc[i]=i;
		}
		else{
			dpInc[i]=dpInc[i-1];
			dpDec[i]=dpDec[i-1];
		}
	}
	for(long i=0;i<m;i++){
		cin>>l>>r;
		l--;
		r--;
		if(dpInc[dpDec[r]]>l)
			cout<<"No";
		else
			cout<<"Yes";
		cout<<"\n";
	}
    return 0;
}
