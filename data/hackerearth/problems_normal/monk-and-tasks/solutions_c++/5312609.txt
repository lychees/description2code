#include <iostream>
#include <map>
using namespace std;
int count_one(int n)
{
	int temp = 0; 
	while (n)
	{
		n = n & (n-1);
		temp++;
	}
	return temp;
}

int main()
{
	multimap<int,int> m;
    int t;
    cin >> t;
    
    while(t--)
    {
    	int n;
    	cin >> n;
    	int arr;
    	for( int i = 0 ; i < n ; i++ )
    	{
    		cin >> arr;
    		m.insert(pair<int,int>(count_one(arr),arr));
    	}
    	
    	multimap<int,int>::iterator it = m.begin();
    	for( it ; it != m.end(); it++)
			cout<<it->second<<" ";
		
		m.clear();
		cout<<endl;
    }
    return 0;
}
