#include <bits/stdc++.h>

using namespace std;
map <string, int> m;
int main()
{
	int t;
	cin >> t;
	string s;
	while (t--) {
		cin >> s;
		m[s] = m[s] + 1;
	}
	
	map <string, int> :: iterator i;
	
	for (i = m.begin(); i != m.end(); ++i) {
		cout << i->first << " " << i->second << endl;
	}
	
	return 0;
}
