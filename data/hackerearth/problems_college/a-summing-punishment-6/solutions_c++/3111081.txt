#include <cstdlib>
#include <iostream>

using namespace std;
int a[1000001],n,m,x,i,j,l,r;
int main(int argc, char *argv[])
{   cin>>n;
    cin>>a[1];
    for(i=2;i<=n;i++)
    {   scanf("%d",&x);
        a[i]=a[i-1]+x;
    }

    cin>>m;
    for(j=1;j<=m;j++)
    {   scanf("%d%d",&l,&r);
        printf("%d\n",a[r]-a[l-1]);
                         
    }
    
    
    
    return EXIT_SUCCESS;
}
