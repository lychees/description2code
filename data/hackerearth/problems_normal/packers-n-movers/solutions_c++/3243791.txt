#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
vector<int> v1;
vector<vector<int> > v2;
int n,m;
int getanswer(int a,int b)
{
	int p1,res_final,res_current,rem;
	if(v2[a][b]==-1)
	{
		if(b==n)
		{
			res_final=0;
			for(p1=a;p1<m;p1++)
				res_final+=v1[p1];
		}
		else
		{
			res_final=1e9;
			res_current=0;
			rem=n-b;
			for(p1=a;p1<m-rem;p1++)
			{
				res_current+=v1[p1];
				res_final=min(res_final,max(res_current,getanswer(p1+1,b+1)));
			}
		}
		v2[a][b]=res_final;
	}
	return v2[a][b];
}
int main()
{
    int p1,p2,res,result=0,sum=0;
    cin>>n>>m;
    v1.resize(m);
    v2.resize(m);
    for(p1=0;p1<m;p1++)
    {
    	cin>>v1[p1];
    	sum+=v1[p1];
    	v2[p1].resize(n+1,-1);
    }
    result=getanswer(0,1);
    cout<<result;
    return 0;
    //
}
