#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   scanf("%d ",&t);
   while(t--){
   		string s,t;
   		getline(cin,s);
   		getline(cin,t);
   		sort(t.begin(),t.end());
   		int i=0,j=0,n=s.length(),m=t.length();
   		while(i<n&&j<m){
   			if(s[i]>t[j]){
   				s[i]=t[j];
   				i++;
   				j++;
   			}
   			else if(s[i]<=t[j])
   				i++;
   		}
   		cout<<s<<endl;
   	
   	
   	
   }
   
    return 0;
}
