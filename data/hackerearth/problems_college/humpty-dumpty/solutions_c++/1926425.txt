#include <bits/stdc++.h>

using namespace std;

int a[1001][1001];

int main()
{
    int t, n, i;
    double x, y, sum;
    cin >> t;
    while (t--) {
        cin >> n;
        double a[n+1];
        cin >> x >> y;
        a[0] = sqrt(x*x+y*y);
        sum = a[0];
        for (i = 1; i <= n; i++) {
            cin >> a[i];
            sum += a[i];
        }
        for (i = 0; i <= n; i++) {
            if ((sum-a[i]) < a[i]) {
                printf("Not able\n");
                goto l1;
            }
        }
        printf("Able\n");
        l1: ;
    }
    return 0;
}
