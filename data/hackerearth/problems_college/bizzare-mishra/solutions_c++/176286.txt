#include <iostream>
using namespace std;

int main()
{
    int t;
    char arr[10000];
    int one[10000],zero[10000],in[10000],len,i,j,k;
    cin>>t;
    
    while(t--)
    {
    	cin>>len;
    	cin>>arr;
    	
    	j=0,k=0;
    	for(i=0;i<len;i++)
    	{
    		if(arr[i]=='Y')
    		  in[i]=1;
    		else
    		  in[i]=0;
    	}
    	one[0]=1;
    	zero[0]=0;
    	for(i=1;i<len;i++)
    	{
    		one[i] = one[i-1]^1;
    		zero[i] = zero[i-1]^1;
    	}
    	
    	for(i=0;i<len;i++)
    	{
    		if(one[i] ^ in[i])
    		  j++;
    		if(zero[i] ^ in[i])
    		  k++;
    	}
    	if(j<k)
    	  cout<<j<<endl;
    	else
    	  cout<<k<<endl;
    }
    
    return 0;
}
