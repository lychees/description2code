#include <bits/stdc++.h>
using namespace std;
bitset<2015>arr[2015];
int main()
{
    int m;
    cin>>m;
    long long ans=0;
    for(int i=0;i<m;i++)
    {
    	for(int j=0;j<m;j++)
    	{
    		int x;
    		cin>>x;
    		arr[i][j]=x;
    	}
    }
    for(int i=0;i<m;i++)
    {
    	for(int j=i+1;j<m;j++)
    	{
    		long long a=(arr[i]&arr[j]).count();
    		ans+=(a*(a-1))/2;
    	}
    }
    cout<<ans/2<<endl;
    return 0;
}
