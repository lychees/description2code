#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(0);
    long long int t,n,orl,len,i,j;
    cin>>t;
    string s;
    while(t--){
        int a[26];
        orl=0;
        cin>>s;
        for(i=0;i<s.size();i++){
            memset(a,0,sizeof (a));
            len=0;
            for(j=i;j<s.size();j++){
                if(a[s[j]-'A']==1)
                    break;
               a[s[j]-'A']=1;
                len++;
            }
            if(orl<len)
              orl=len;
        }
        cout<<orl<<'\n';
    }
    return 0;
}
