# include<iostream>
using namespace std;
int a[505];
int occ[502][2002];
int cube[18]={1, 8, 27, 64, 125, 216, 343, 512, 729, 1000, 1331, 1728, 2197, 2744, 3375, 4096, 4913, 5832};
int main()
{
	int t,i,j,n,k,x;
	double c,den,ans;
	cin>>t;
	while(t--)
	{
		c=0;
		cin>>n;
		for(i=1;i<=n;i++)
		{
			cin>>a[i];
		}
		for(i=n;i>=1;i--)
		{
			for(j=1;j<=2000;j++)
			{
				if(i==n)
					occ[i][j]=0;
				else
					occ[i][j]=occ[i+1][j];
			}
			occ[i][a[i]]++;
		}
		for(i=1;i<=n-2;i++)
		{
			for(j=i+1;j<=n-1;j++)
			{
				for(k=0;k<18;k++)
				{
					x=cube[k]-a[i]-a[j];
					if(x>0 && x<=2000)
					{
						c+=occ[j+1][x];
					}
				}
			}
		}
		den=(n*(n-1)*(n-2))/6.0;
		ans=c/(den*1.0);
		printf("%.9f\n",ans);
	}
	return 0;
}
