int main()
{
      int t;
      scanf("%d\n",&t);
      while(t--)
      {
            char s1[1000010],s2[1000010];
            scanf("%s %s",s1,s2);
            int flag = 0;
            
            int j=0;
            int l1 = strlen(s1);
            int l2 = strlen(s2);
            for(int i=0;i<(l1);i++)
            {
                  while(i<(l1) && s1[i]!='A')
                  i++;
                  while(j<(l2) && s2[j]!='A')
                  j++;
                  if(i-j<0)
                  {
                        flag=1;
                        break;
                  }
                  j++;
            }
            if(flag)
            {
                  cout<<"No"<<endl;
                  continue;
            }
            else
            {
                  j=0;
                  for(int i=0;i<(l1);i++)
                  {
                  while(i<(l1) && s1[i]!='B')
                        i++;
                  while(j<(l2) && s2[j]!='B')
                        j++;
                  if(i-j>0)
                  {
                        flag=1;
                        break;
                  }
                  j++;
                  }
                  if(flag)
                  cout<<"No"<<endl;
                  else
                  cout<<"Yes"<<endl;
            }

      }
      return 0;
}
