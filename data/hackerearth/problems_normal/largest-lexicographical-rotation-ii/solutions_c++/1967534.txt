#include<bits/stdc++.h>
using namespace std;
    int min(string s)
    {
    	s=s+s;
    	int len=s.length(),i=0,j=1,k=0;
    	while(i+k<len && j+k <len)
    	{
    		if(s[i+k]==s[j+k])
    		k++;
    		else
    		if(s[i+k]<s[j+k])
    		{
    			i=i+k+1;
    			if(i<=j)
    			i=j+1;
    			k=0;
    		}
    		else
    		if(s[j+k]<s[i+k])
    		{
    			j=j+k+1;
    			if(j<=i)
    			j=i+1;
    			k=0;
    		}
    	}
    	return min(i,j);
    }
    int main()
    {
    	string s;
    	cin>>s;
    	int len=s.size();
    	int res =min(s);
    	for(int i=res;i<len;i++)
    	printf("%c",s[i]);
    	for(int i=0;i<res;i++)
    	printf("%c",s[i]);
        return 0;
    }
