#include <bits/stdc++.h>
using namespace std; 
int main()
 {
   int n,q;
   cin>>n>>q;
   int a[n];
   for(int i=0;i<n;i++)
    {
    	if(i==0){cin>>a[i];}
    	else{cin>>a[i];a[i]+=a[i-1];}
    }
   // for(int i=0;i<n;i++) cout<<a[i]<<" ";
    int k,k1;
    while(q--)
     {
     	cin>>k>>k1;
     	if(k==1)
     	 {
     	 	cout<<a[k1-1]<<" ";
     	 }
     	else
     	 {
     	 	cout<<a[k1-1]-a[k-2]<<" ";
     	 }
     	if(k==1&&k1==1){cout<<a[0]<<endl;} 
     	else if(k==1)
     	 {
     	 	int z1=(a[k1-2]);
     	 	int z=(a[k1-1]-z1)-z1;
     	 	//cout<<z<<" "<<z1<<endl;
     	 	cout<<z<<endl;
     	 }
     	else
     	 {
     	 	int z1=(a[k1-1]-a[k1-2]);
     	 	int z=(a[k1-1]-a[k-2])-z1;
     	 	//cout<<z<<" "<<z1<<endl;
     	 	cout<<z1-z<<endl;
     	 }
     }
 	
 }
