#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	int len=str.size();
    	int cur=0;
    	for(int i=0;i<len;i++)
    	{
    		int posc=(26+str[i]-97-cur)%26;
    		int posa=26-posc;
    		if(posc<=posa)
    		cout<<posc<<" ";
    		else
    		cout<<-posa<<" ";
    		cur=str[i]-97;
    	}
    	cout<<"\n";
    }
}
