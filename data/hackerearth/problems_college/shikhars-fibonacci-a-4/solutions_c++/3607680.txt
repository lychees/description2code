#include <bits/stdc++.h>
using namespace std;

#define MAXX 100000
#define mod 1000000007
long long int d[MAXX],a,b;

long solve(int n){
	if(d[n]!=-1)return d[n];
	if(n==0)return 0;
	if(n==1)return a;
	if(n==2)return b;
	long T;
	T=solve(n-1)%mod+solve(n-2)%mod;
	d[n]=T%mod;
	return d[n];
}

int main()
{
	int n;cin>>a>>b>>n;
	for(int i=0;i<=n;i++)d[i]=-1;
	long ans=solve(n)%mod;
	cout<<ans<<endl;
    return 0;
}
