#include <bits/stdc++.h>
using namespace std;

int main()
{
   long double tc,l,b,h, small ;
   cin>>tc ;
   while(tc--)
   {
   	cin>>l>>b>>h ;
   	if(l<b)
   	   if(l<h)
   	     small=l;
   	     else
   	     small=h;
   	else
   	   if(b<h)
   	     small=b;
   	     else
   	     small=h ;
   	    
   	    
   	printf("%.1llf\n",small/2) ;
   	  
   }
    return 0;
}
