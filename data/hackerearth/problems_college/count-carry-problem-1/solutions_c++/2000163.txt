#include<bits/stdc++.h>
using namespace std;

int len(int num)
{
    int s=0;
    while(num)
    {
        num/=10;
        s++;
    }
    return s;
}

int main()
{
    long long int t,num1,num2;
   long long int carry,sum,cn;
    int sa , sb,diff;

        sa = len(num1);
        sb = len(num2);

      //  diff = abs(sa-sb);

        carry=0;
        cn=0;
        while(num1)
        {
           sum = num1%10+num2%10+carry;

            {
                sum=sum%10;
                carry=1;

            num1/=10;
            num2/=10;

        if(sa>sb && carry>0)
        {
            while(num1 && carry!=0)
            {
                sum=num1%10+carry;
                if(sum>=10)
                {
                 sum=sum%10;
                 carry=1;
