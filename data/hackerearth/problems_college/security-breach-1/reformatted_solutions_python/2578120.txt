# nput from stdin and provide input before running code

for __ in range(input()):
    ss = raw_input()
    s = ""
    for I in ss:
        if I == '2' or I == '6' or I == '7' or I == '9':
            s += I
    s = s.replace('2', "cde")
    s = s.replace('6', "b3")
    s = s.replace('9', "c6a")
    s = s.replace('7', "acf")
    print s
