#include <iostream>
#include<string>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s,s2;
    	s2="";
    	cin>>s;
    	int n=(int)s.length();
    	while(n!=0)
    	{
    		if(n&1)
    		{
    			s2+=s[(n)/2];
    			s.erase((n)/2,1);
    		}
    		else
    		{
    			if(s[(n)/2]<=s[(n-1)/2])
    			{
    				s2+=s[(n)/2];
    				s.erase((n)/2,1);
    			}
    			else
    			{
    				s2+=s[(n-1)/2];
    				s.erase((n-1)/2,1);
    			}
    		}
    		n=(int)s.length();
    	}
    	cout<<s2<<"\n";
    }
    return 0;
}
