#include<bits/stdc++.h>
using namespace std;
vector<int>adj[100009];
int mark[1000009],value[1000009],zero=0,one=0,h[1000008];
void dfs(int s,int c)
{
    //cout<<s<<endl;
	if(c==1)
	one++;
	else
	zero++;
	mark[s]=1;
	int new1=0;
	if(c==0)
		new1=1;
	for(int i=0;i<adj[s].size();i++)
	{
		if(mark[adj[s][i]]==0)
		dfs(adj[s][i],new1);
	}
}
int main()
{
	ios::sync_with_stdio(false);
	int i,n,t,a,b,v,ans,j=0;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=1;i<=100008;i++)
		adj[i].clear(),h[i]=0;
		for(i=1;i<=n;i++)
		{
			cin>>a>>b;
			adj[a].push_back(b);
				adj[b].push_back(a);
				h[a]=1;
				h[b]=1;
		}
		ans=0;
		for(i=1;i<=100009;i++)
		mark[i]=0;
		for(i=1;i<=100008;i++)
		{
			if(mark[i]==0&&h[i]==1)
			{
				zero=0;
				one=0;
				dfs(i,0);
				ans+= max(zero,one);
			}
		}
		j++;
		printf("Case %d: %d\n",j,ans);
	}
	return 0;
}
