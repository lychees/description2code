#include <bits/stdc++.h>
using namespace std;
double pro[1003][1003],res[1003][1003],ans;
int main()
{
//	std::ios::sync_with_stdio(false);
//	cin.tie(0);
    int t,n,m,i,j,k;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	for(i=1;i<=n;i++)
    		for(j=1;j<=m;j++)
    			res[i][j]=0,scanf("%lf",&pro[i][j]);
    	//	cout<<pro[1][1];
    	res[0][0]=1;
    	for(i=0,j=1;j<=m;j++)
    		res[i][j]=res[i][j-1]*pro[i+1][j];
    	ans=res[0][m]*m;
    	res[0][m]=0;
    	for(i=1;i<n;i++)
    	{
    		res[i][0]=res[i-1][0]*(1-pro[i][1]);
    		for(j=1;j<m;j++)
    			res[i][j]=res[i-1][j]*(1-pro[i][j+1])+res[i][j-1]*pro[i+1][j];
    		res[i][m]=res[i][m-1]*pro[i+1][m];
    		ans+=res[i][m]*(m+i);
    		res[i][m]=0;
    	}
    		res[i][0]=res[i-1][0]*(1-pro[i][1]);
    		res[i][0]=0;
    		for(j=0;j<m;j++)
    		{
    			res[i][j]=res[i-1][j]*(1-pro[i][j+1]);
    			ans+=res[i][j]*(i+j);
    		}
   
    	printf("%.6f\n",ans);
    }
    return 0;
}
