#include<iostream>
#include<assert.h>
using namespace std;

#define MAX 1000007
int ph[MAX];
int main()
{
    int t,n,i,j;
    scanf("%d",&t);
    for(i=0;i<=1000000;i++)
    {
      ph[i]=i;
    }
    for(i=2;i<=1000000;i++)
    {
        if(ph[i]==i)
        {
                     for(j=i;j<=1000000;j+=i)
                     {
                         ph[j]=(ph[j]/ i) * (i-1);
                     }
        }
    }
    
    while(t--)
    {
              scanf("%d",&n); 
              assert(n<1000000);  
              printf("%d\n",ph[n]);      
    }
    return 0;
}
