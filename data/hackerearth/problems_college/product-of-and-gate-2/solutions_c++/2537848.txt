#include <iostream>
using namespace std;
long long andx(long long a, long long b) {
    long long x = a^b;
    long long s = x>>1;

    while (s) {
        x = x|s;
        s >>= 1;
    }
    return a&b&~x;
}
int main()
{
   int t;
   cin >>t;
   while(t--)
   {
   	long long a,b;
   	cin >>a>>b;
   	cout << andx(a,b) <<endl;
   }
    return 0;
}
