#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k;
    	cin>>n>>k;
    	vector<int> a(n),sum(n);
    	
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	
    	sort(a.begin(),a.end());
    	
    	
    	
    	sum[0]=a[0];
    	for(int i=1;i<n;i++)
    	sum[i]=sum[i-1]+a[i];
    	
    	/*for(int i=0;i<n;i++)
    	cout<<sum[i]<<"\n";*/
    	
    	int ans=1000000;
    	for(int i=0;i<=n-k;i++)
    	{
    		int temp=5*(sum[i+k-1]-sum[i]-a[i]*(k-1));
    	
    		for(int j=i+1;j<i+k;j++)
    		{
    			int x=3*(a[j]*(j-i)-(sum[j-1]-sum[i]+a[i]));
    			int y=5*(sum[i+k-1]-sum[j]-(i+k-j-1)*a[j]);
    		//	cout<<temp<<" "<<x<<" "<<y<<"\n";
    			temp=min(temp,x+y);
    		}
    		ans=min(ans,temp);
    	}
    	cout<<ans<<"\n";
    }
    return 0;
}
