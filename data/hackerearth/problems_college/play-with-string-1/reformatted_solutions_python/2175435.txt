import math
t = int(raw_input())
while t:
    t = t-1
    s = raw_input()
    l = len(s)
    c = 0
    if l == 1:
        print("Funny")
        continue
    for i in range(0, l/2):
        if abs(ord(s[i])-ord(s[i+1])) != abs(ord(s[l-i-1])-ord(s[l-i-2])):
            c = 1
            print("Not Funny")
            break
    if c == 0:
        print("Funny")
