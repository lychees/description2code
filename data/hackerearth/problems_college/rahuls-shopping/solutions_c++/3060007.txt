#include<iostream>
#include<vector>
using namespace std;

int calcPrice(int a,int ar){
        int sod = 0,n=a;
        while(a){
                sod+=a%10;
                a/=10;
        }
        if((n-1)/10 == 0) return 0.75*ar*sod*n;
        else if ((n-1)/10 == 1) return 0.7*ar*sod*n;
        else return 0.8*ar*sod*n;
}

int main(){
        int n;
        cin>>n;
        vector<int> a(n);
        for(int i=0;i<n;i++){
                cin>>a[i];
        }
        int l,b;
        cin>>l>>b;
        int ar = l*b;
        int total = 0;
        for(int i=0;i<n;i++){
                total+=calcPrice(a[i],ar);
        }
        cout<<total;
        return 0;
}
