#include <iostream>
using namespace std;

int main()
{
   long n,m,left,right,i=1;
   cin>>n>>m;
   start:
   if(m%2==0)
   {
   left=(m/2);
   right=left+1;
   cout<<left<<endl;
   n--;
   if(n>0){
   cout<<right<<endl;
   n--;
   }
   }
   else
   {
   	left=right=(m+1)/2;
   	cout<<left<<endl;
   	n--;
   }
    if(right==m)
    if(n>0)
    goto start;
   while(n>0)
   {
   	left--;
   	right++;
   	cout<<left<<endl;
   	n--;
   	if(n>0)
   	{
   		cout<<right<<endl;
   		n--;
   	}
    if(right==m)
    if(n>0)
    goto start;
   }
    return 0;
}
