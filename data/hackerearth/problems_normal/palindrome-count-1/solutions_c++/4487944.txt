#include <iostream> 
using namespace std ; 
#include <bits/stdc++.h>
int palindrome ( string s  )
{
	string p ; 
	p = s ;
	reverse( p.begin() , p.end()) ; 
	if ( p ==s)
	{
		return 1 ; 
	}
	else
	{
		return 0 ; 
	}
}
int main()
{
	string s ; 
	cin >> s  ; 
	int i ; 
	int n = s.length() ; 
	int ans = 0 ; 
	for ( int i = 0 ; i< n ; i++)
	{
		for ( int j = i+1 ; j <=n ; j++)
		{
			if ( palindrome(s.substr(i , j-i)) == 1)
			{
				ans++ ; 
			}
		}
	}
	cout << ans <<endl; 
	return 0 ; 
}
