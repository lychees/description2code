#include <iostream>
#include<limits.h>
using namespace std;

int main()
{

    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n],min=INT_MAX,pos=0;
    	for(int i=0;i<n;i++)
    	{cin>>a[i];
    	if(min>a[i])
    	{min=a[i];
    	pos=i;}

    	}
    	long long sum=0;
    	for(int i=0;i<n;i++)
    	{
    		if(pos!=i)
    		sum+=min+a[i];
    	}
    	cout<<sum<<"\n";
    }
    return 0;
}
