#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str,str1;
		cin>>str>>str1;
		string temp1="",temp2="";
		int a_c=0,b_c=0,a_f_c=0,b_f_c=0;
		vector<int> posia,posib;
		vector<int> posfa,posfb;
		for(int i=0;i<str.length();++i)
		{
			if(str[i]!='#')
				temp1+=str[i];
			if(str1[i]!='#')
				temp2+=str1[i];
			if(str[i]=='A')
			{
				a_c++;
				posia.push_back(i);
			}
			else if(str[i]=='B')
			{
				b_c++;
				posib.push_back(i);
			}
			if(str1[i]=='A')
			{
				a_f_c++;
				posfa.push_back(i);
			}
			else if(str1[i]=='B')
			{
				b_f_c++;
				posfb.push_back(i);
			}
		}
		if(a_c!=a_f_c || b_c!=b_f_c || (temp1.compare(temp2)!=0))
			cout<<"No\n";
		else
		{
			bool flag=true;
			for(int i=0;i<posia.size();++i)
			{
				//cout<<posia[i]<<" "<<posfa[i]<<endl;
				if(posia[i]<posfa[i])
				{
					flag=false;
					break;
				}
			}
			for(int i=0;i<posib.size();++i)
			{
				//cout<<posib[i]<<" "<<posfb[i]<<endl;
				if(posib[i]>posfb[i])
				{
					flag=false;
					break;
				}
			}
			if(flag)
				cout<<"Yes\n";
			else
				cout<<"No\n";
		}
	}
	return 0;
}
