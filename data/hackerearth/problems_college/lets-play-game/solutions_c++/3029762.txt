#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[100010],p[100010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n,k;
		cin>>n>>k; //type of stones & total time
		ll int i;
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0;i<n;i++)
		cin>>p[i];
		ll int ans=0;
		for(i=0;i<n;i++)
		{
			ans=max(k/a[i]*p[i],ans);
		}
		cout<<ans<<endl;
	}
	return 0;
}
