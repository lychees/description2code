using namespace std;
 
typedef unsigned long long int ull;
typedef long long int ll;

int pr[1005]={0};
void seive()
{
	pr[0]=1;pr[1]=1;
	for(int i=2;i*i<=1003;i++)
	{
		if(pr[i]==1)
		continue;
		for(int j=i*2;j<=1003;j+=i)
		pr[j]=1;
	}
}

int main()
{ios_base::sync_with_stdio(false);
int t;
gl(t);seive();
while(t--)
{
	int n,c[105],p[105],p1=0,p2=0;
	cin>>n;
	rep(i,0,n-1)gl(c[i]);
	rep(i,0,n-1)gl(p[i]);
	rep(i,0,n-1)
	{
		if(pr[c[i]]==1&&pr[p[i]]==0)
		{
			if(c[i]>p[i])
			p1+=c[i]-p[i];
			else if(p[i]>c[i])
			p2+=(p[i]-c[i]);
		}
		else
		{
			if(pr[c[i]]==1)
			p1+=c[i];
			else if(pr[p[i]]==0)
			p2+=p[i];
			else
			{
				if(c[i]>p[i])
				p1+=c[i]-p[i];
				else if(p[i]>c[i])
				p2+=p[i]-c[i];
			}
		}
	}
	if(p1>p2)
	ol("The Composites\n");
	else if(p2>p1)
	ol("The Primes\n");
	else if(p1==p2)
	ol("Tie\n");
}
return 0;
}
