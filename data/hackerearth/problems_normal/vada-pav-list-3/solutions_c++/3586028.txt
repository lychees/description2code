#include <iostream>
#include <set>
using namespace std;

int main()
{
    long long int n;
    scanf("%d", &n);
    set<string>names;
    string name;
    while(n--)
    {
    	cin >> name;
    	names.insert(name);
    }
    printf("%d\n", names.size());
    for(auto it = names.begin(); it != names.end(); it++)
    	printf("%s\n",(*it).c_str());
}
