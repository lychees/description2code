#define ff first

#define pb push_back
#define mp make_pair
#define all(x) x.begin(),x.end()
#define sz(x) ((int)x.size())

const int MOD = 1e9+7;

typedef long long ll;
typedef pair<int,int> pii;
const int MAXN = 1e5+5;
ll dp[MAXN];
vector<int> adj[MAXN];
void DFS(int s, int par=-1) {
	ll sum=1;
	for(auto it:adj[s]) {
		if(it==par) continue;
		DFS(it,s);
		sum+=dp[it];
	}
	dp[s]=sum;
}
int main() {
	// freopen("TASK.in","r",stdin);freopen("TASK.out","w",stdout);
	int n;
	cin>>n;
	for(int i=1;i<n;i++) {
		int x,y;
		cin>>x>>y;
		adj[x].pb(y);
		adj[y].pb(x);
	}
	DFS(1);
	dp[1]--;
	for(int i=1;i<=n;i++) cout<<dp[i]<<" ";
	cout<<endl;
	return 0;
}
