#include <bits/stdc++.h>
using namespace std;
int main()
{
    long long n,m,q;
    std::ios::sync_with_stdio(false);
    cin>>n>>m>>q;
    vector<vector<long long> > mat(n);
    long long i,j;
    for(i=0;i<n;i++)
    {
    	long long tmp;
    	for(j=0;j<m;j++)
        {
            cin>>tmp;
            mat[i].push_back(tmp);
        }
    }
    while(q--)
    {
    	long long arr[n];
    	long long i,j;
    	for(i=0;i<n;i++)
    	    cin>>arr[i];
        long long level=LONG_MAX;
    	for(i=0;i<n;i++)
    	{
    		long long tmp;
    		vector<long long>::iterator it=upper_bound(mat[i].begin(),mat[i].end(),arr[i]);
    		if(it==mat[i].end())
            {
                tmp=m;
                level=min(level,tmp);
            }
            else
            {
                if((*it)>arr[i])
                {
                    if(it==mat[i].begin())
                        tmp=0;
                    else
                    {
                        it--;
                        tmp=it-mat[i].begin()+1;
                    }
                }
                else
                {
                     tmp=it-mat[i].begin()+1;
                }
                level=min(level,tmp);
            }
    	}
    	cout<<level<<"\n";
    }
    return 0;
}
