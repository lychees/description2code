#include <iostream>
#include <string>
using namespace std;

int main()
{
   int t;
  scanf("%d",&t);
  while(t--)
  {
  	string s;
  	cin>>s;
  	int len = s.length();
  	int sc = 0;
  	int uc=0;
  	for(int i=0;i<len;++i)
  	{
  		if(s[i]=='U')++uc;
  	}
  	sc=len-uc;
  		int cnt = 0;
  			int cnt1 = 0;
  	bool bflag = true;
  	if((sc<uc-1 || uc<sc-1)&&uc!=sc)bflag = false;
  	if(bflag)
  	{
	  	if(sc>uc)
	  	{
	  		cnt=0;
	  		for(int i=1;i<len;i=i+2)
	  		{
	  			if(s[i]=='S')++cnt;
	  		}
	  		cout<<cnt<<endl;
	  	}
	  	else if(uc>sc)
	  	{
	  		cnt = 0;
	  		for(int i=1;i<len;i=i+2)
	  		{
	  			if(s[i]=='U')++cnt;
	  		}
	  		cout<<cnt<<endl;
	  	}
	  	else
	  	{
	  		 cnt;
	  		 cnt1=0;
	  		for(int i=1;i<len;i=i+2)
	  		{
	  			if(s[i]=='S')++cnt;
	  		}
	  		for(int i=1;i<len;i=i+2)
	  		{
	  			if(s[i]=='U')++cnt1;
	  		}
	  		if(cnt1<cnt)cnt=cnt1;
	  		cout<<cnt<<endl;
	  	}
  	}
  	else
  		cout<<-1<<endl;
  }
    return 0;
}
