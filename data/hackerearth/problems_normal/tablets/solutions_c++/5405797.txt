#include <iostream>
using namespace std;

int main()
{
    long long n,i;
    cin>>n;
    long long a[100001];
    for(i=0;i<n;i++)
        cin>>a[i];
    long long dp[100001];
    for(i=0;i<n;i++)
        dp[i]=1;
    for(i=1;i<n;i++)
        if(a[i]>a[i-1])
            dp[i]=dp[i-1]+1;
    for(i=n-2;i>=0;i--)
        if(a[i]>a[i+1]&&dp[i]<=dp[i+1])
            dp[i]=dp[i+1]+1;
    long long sum=0;
    for(i=0;i<n;i++)
        sum+=dp[i];
    cout<<sum<<endl;
}
