#include <bits/stdc++.h>
using namespace std;

const long long maxn = 1000;

long long N,M,dp[maxn];

int main(){
	
	scanf("%lld%lld",&N,&M);
	
	dp[0] = 1;
	for( long long i=1,val ; i <= M ; i++ ){
		scanf("%lld",&val);
		for( long long x=0 ; x+val <= N ; x++ )
			if( dp[x] )
				dp[x+val] += dp[x];
	}
	
	printf("%lld\n",dp[N]);
	
	return 0;
}
