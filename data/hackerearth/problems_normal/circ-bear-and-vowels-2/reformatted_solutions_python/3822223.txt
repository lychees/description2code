
n = int(raw_input())


def hard_easy(input_string):
    vowels = ['a', 'e', 'i', 'o', 'u', 'y']
    total_count = len(input_string)
    vowel_count = 0
    continous_consonant = 0
    second_condition = False
    for elem in input_string:
        if elem in vowels:
            vowel_count += 1
            continous_consonant = 0
        else:
            continous_consonant += 1
        if continous_consonant == 3:
            second_condition = True
    if vowel_count < (total_count - vowel_count) or second_condition:
        return "hard"
    else:
        return "easy"


for i in xrange(0, n):
    input_string = raw_input().strip()
    print hard_easy(input_string)
