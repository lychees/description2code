#include <bits/stdc++.h>
using namespace std;

map<string,map<string,int> > mp;
#define ll long long 
ll f(string str)
{
	ll n = str.length(),c=0;
	string pre,s;
	pre="";
	ll j=1;
	for(ll i=1;i<n;)
	{
		while(i<n && str[i]!='/') i++;
		s = str.substr(j,i-j);
		if(!mp[s][pre]) c++;
		mp[s][pre] = 1;
		pre=s;
		j=i+1;
		i++;
	}
	//c++;
	return c;
}
int main()
{
   int t,tnum=1,n,m,c;
   string str;
   cin>>t;
   while(tnum<=t)
   {
   	mp.clear();
   	c=0;
   	scanf("%d%d",&n,&m);
   	while(n--)
   	{
   		cin>>str;
   		f(str);
   	}
   	while(m--)
   	{
   		cin>>str;
   		c += f(str);
   	}
   	printf("Case #%d: %d\n",tnum,c);
   	tnum++;
   }
}
