#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long brr[1000000];
    int n;
    long long d;
    scanf("%d %lld",&n,&d);
    for(int i=1;i<=1000000;i++)
    brr[i]=d*i;
    long long arr[n];
    for(int i=0;i<n;i++)
    	scanf("%lld",&arr[i]);
    long long count=0;
    for(int i=1;i<n;i++)
    {
    	if(arr[i-1]>arr[i])
    	{
    		for(int j=1;j<=1000000;j++)
    		{
    			if(arr[i]+brr[j]>arr[i-1])
    			{
    				count+=j;
    				arr[i]+=brr[j];
    				//printf("%lld ",arr[i]);
    				break;
    			}
    		}
    	}
    //	printf("%lld ",arr[i]);
    }
    printf("%lld\n",count);
    return 0;
}
