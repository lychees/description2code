def fact(x):
    if (x <= 1):
        return 1
    s = 1
    while (1):
        s *= x
        x -= 1
        if (x == 1):
            break
    return s


t = input()
for _ in range(0, t):
    a = map(int, raw_input().split())
    y = 0
    for i in range(a[0], a[1]+1):
        y += fact(i)
    print y % 10
