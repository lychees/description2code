#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--)  {
    	int a[10];
    	memset(a,0,sizeof(a));
    	long long n;
    	cin >> n;
    	int c=0;
    	while(n) {
    		a[n%10]++;
    		n=n/10;
    		c++;
    	}
    	n = a[4]+a[7];
    	memset(a,0,sizeof(a));
    	c=0;
    	while(n) {
    		a[n%10]++;
    		n=n/10;
    		c++;
    	}
    	if(a[4]+a[7] == c && c != 0)
    	  cout << "YES" << endl;
    	else
    	  cout << "NO" << endl;
    }
    return 0;
}
