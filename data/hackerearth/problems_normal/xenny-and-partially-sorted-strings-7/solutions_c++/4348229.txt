#include <iostream>
#include <string>
#include <stdlib.h>

#include <algorithm>
using namespace std;
int main()
{

    int x;
    cin >> x;
    while(x--)
    {

        int q,w,e;
        cin >> q >> w >> e;
        vector <pair<string,string>> s1;
        while(q--)
        {
                string a;
                cin >> a;
                string b= a.substr(0,e);
                s1.push_back(make_pair(b,a));


        }
        sort(s1.begin(),s1.end());
        cout << s1[w-1].second<<endl;

    }
}
