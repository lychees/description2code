#include<bits/stdc++.h>
using namespace std;
vector<long long int>arr;
int main()
	{
		int N,Q,lindex,rindex;
		long long int x,l,r;
		cin>>N;
		for(int i=0;i<N;i++)
			{
				cin>>x;
				arr.push_back(x);
			}
		cin>>Q;
		sort(arr.begin(),arr.end());
		for(int i=0;i<Q;i++)
			{
				cin>>l>>r;
				if(l>arr[N-1])
					cout<<"0"<<endl;
				else
				{
				int low=0;
				int high=arr.size()-1;
				lindex=0;
				rindex=0;
				while(1)
					{//cout<<"bye"<<endl;
						int mid=(low+high)/2;
						if(arr[mid]==l)
							{
								while(arr[mid]==l && low<=high)
									{
										mid--;
									}
								
								lindex=mid+1;
								break;
							}
						else if(arr[mid]<l)
							low=mid+1;
						else
							high=mid-1;
						if(low>high){
							//cout<<"hi"<<endl;
							l=arr[low];
							low=0;
							high=N-1;
						}
						
					}
				low=0;
				high=arr.size()-1;
				if(lindex==N-1)
					cout<<"1"<<endl;
				else if(r>arr[N-1] && lindex==0)
					cout<<N<<endl;
				else if(r>arr[N-1] && lindex!=1)
					cout<<N-lindex<<endl;
				else{
				while(1)
					{
						int mid=(low+high)/2;
						if(arr[mid]==r)
							{
								while(arr[mid]==r && low<=high)
									{
										mid++;
									}
							
								rindex=mid-1;
								break;
							}
						else if(arr[mid]<r)
							low=mid+1;
						else
							high=mid-1;
						if(low>high){
							//cout<<"hi"<<endl;
							r=arr[high];
							low=0;
							high=N-1;
						}
						
					}
				int cnt=0;
				for(int j=lindex;j<=rindex;j++)
					cnt++;
				cout<<cnt<<endl;
				}}
			}
		return 0;
	}
