#include<bits/stdc++.h>
using namespace std;
int main()
{
unsigned long long int test;
cin>>test;
while(test--)
{
	char s[1009];
	char t[1009];
	cin>>s>>t;
	int sa[26]={0};
	int ta[26]={0};
	for(int i=0;i<strlen(s);i++)
	{
		int x=s[i];
		sa[x-97]++;
	}
	for(int i=0;i<strlen(t);i++)
	{
		int x=t[i];
		ta[x-97]++;
	}
	int count=0;
	int ans=0;
	for(int i=0;i<26;i++)
	{
		if(sa[i]>ta[i])
		{
			
			count=-1;
		}
		else if(sa[i]<ta[i])
		{
			ans=ans+1;
		}
	}
//	cout<<count<<endl;
	if(count==-1)
	{
		cout<<"-1"<<endl;
	}
	else
	{
		cout<<ans<<endl;
	}
	
	
}
return 0;	
}
