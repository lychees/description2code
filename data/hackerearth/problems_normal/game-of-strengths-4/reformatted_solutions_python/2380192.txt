m = (10**9)+7
for i in range(int(raw_input())):
    n = int(raw_input())
    num = sorted(map(int, raw_input().split()))
    maxnum = num[n-1]
    productconstant = n-1
    abssum = 0
    for j in range(n-1, -1, -1):
        abssum += (num[j]*productconstant)
        productconstant -= 2
    print(abssum*maxnum) % m
