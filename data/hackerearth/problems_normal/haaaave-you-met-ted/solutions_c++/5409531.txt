#include <bits/stdc++.h>
using namespace std;
#define int long long
long long f(int n)
{
  long long count = 0;
  while(n)
  {
    count += n & 1LL;
    n >>= 1LL;
  }
  return count;
}

main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	int aa=INT_MAX;
    	for(int i=0;i<n;i++)
    	{
    		int no = a[i];
    		aa= min(aa,f(no));
    	}
    	cout<<aa<<endl;
    }
    return 0;
}
