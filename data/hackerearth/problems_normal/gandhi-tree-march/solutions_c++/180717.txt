vector<int> Data;
vector<char> Label;
string str;
int len;
int i;
void setValues(int col){
		if(str[i]>='a' && str[i]<='z'){
			Data.pb(col);
			Label.pb(str[i]);
			i+=2;
			if(str[i]>='a' && str[i]<='z'){
				setValues(col-1);
			}
			i++;
			if(str[i]>='a' && str[i]<='z'){
				setValues(col+1);
			}
			i++;
			if(str[i]==')'){
				return;
			}
		}
}
