#include <iostream>
using namespace std;
int main()
{
    int t,a,b;
    int p[5];
    cin>>t;
    while(t--)
    {
        int i=0;
        cin>>a>>b;
        if(b>a)
            cout<<a;
        else
        {
            while(a>=b)
            {
                p[i++]=a%b;
                a=a/b;
            }
            p[i]=a;
            for(int j=i;j>=0;j--)
                cout<<p[j];
        }
        cout<<endl;
    }
    return 0;
}
