#include<stdio.h>
#include<string.h>
#include<iostream>
#include<math.h>
#include<stdlib.h>
using namespace std; 

int main() {
	long long int t,n;
	int k;
	cin>>t;
	while(t--)
	{
		long long int flag=0,i,sum=0;;
		cin>>n;
		while(n>0)
		{
			k=n%10;
			n=n/10;
			sum=sum+k;
		}
		if(sum==1)
		{
			cout<<"NO"<<"\n";
			continue;
		}
		for(i=2;i<=sqrt(sum);i++)
		{
			if(sum%i==0&&sum!=1)
			{
				flag=1;
				break;
			}
		}
		
		
		if(flag==1)
		{
			cout<<"NO"<<"\n";
		}
		else
		{
			cout<<"YES"<<"\n";
		}
	}
	return 0;
}
