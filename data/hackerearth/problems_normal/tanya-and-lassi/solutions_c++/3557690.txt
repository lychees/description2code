#include <bits/stdc++.h>
using namespace std;
#define ll long long

ll int maxi(ll int a, ll int b) {
	if(a>b) return a; 
	return b;
}


int main()
{
    ll int t,l;
    cin>>t;
    while(t--) {
    	cin>>l;
    	ll int a[l];
    	ll int dp[l];
    	memset(dp,0,sizeof(dp));
    	for(int i = 0;i<l;i++) {
    		cin>>a[i];
			if(i == 0) {
				dp[i] = a[i];
			}else {
				for(int j = 0;j<= (i-1)/2;j++){
					dp[i] = maxi((dp[j] + dp[i-1 - j]) , dp[i] );
				}
				dp[i] = maxi(a[i] , dp[i]);
			}
			
    	}
    	cout<<dp[l-1]<<endl;
    }
    return 0;
}
