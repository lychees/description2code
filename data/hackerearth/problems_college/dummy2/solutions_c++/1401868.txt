#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
typedef long long ll;
#define sz 1000004

char a[sz],b[sz],c[sz];

int main(){
	int t,i,p,j;
	cin>>t;
	while(t--){
		cin>>a;
		for(i=0;a[i];){
			if(a[i]>='1' && a[i]<='9'){
				p=a[i]-'0';
				for(j=0;j<i;j++)
					b[j]=a[j];
				b[j]='\0';
				for(j=i+1;a[j];j++)
					c[j-i-1]=a[j];
				c[j-i-1]='\0';
				strcpy(a,"");
				for(j=0;j<p;j++)
					strcat(a,b);
				strcat(a,c);
				i=p*i;
			}
			else
				i++;
		}
		cout<<a<<endl;
	}
	return 0;
}
