#include<iostream>
#define MAX 100000

using namespace std;

int main()
{
  int i,j,a[MAX],sum;
  a[0]=0;a[1]=1;
  // cout<<a[0]<<" "<<a[1];
  i=2;
  sum=1;
  while(sum<=100000)
    {
      sum=a[i-2]+a[i-1];
      a[i]=sum;
      //    cout<<" "<<a[i];
      i++;
    }
  //  cout<<i<<endl;
  int t,x,y,k;
  cin>>t;
  
  for(j=0;j<t;j++)
    {
      k=0;
      int count=0;
      cin>>x>>y;
      if(x==1)
	{
	  k=2;
	  count=1;
	}
      while(a[k]<x)
	k++;
      while(x<=y)
	{
	  if(a[k]==x)
	    {
	      count++;
	      k++;
	    }
	  x++;
	}
      if(count%2==0)
	cout<<"VALID"<<endl;
      else
	cout<<"INVALID"<<endl;
    }
  
  return 0;
}
