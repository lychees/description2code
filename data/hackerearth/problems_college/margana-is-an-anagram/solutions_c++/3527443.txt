#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	int a1[27]={0},b1[27]={0};
    	string a,b;
    	cin>>a>>b;
    	
    	if(a.size()!=b.size()){
    		cout<<"No"<<endl;
     	}
     	else{
     		int f=0;
     		for(int i=0;i<a.size();i++){
     			a1[a[i]-96]++;
     			b1[b[i]-96]++;
     		}
     		
     	
     		for(int i=0;i<26;i++){
     			if(a1[i]!=b1[i]){
     				f=1;break;
     			}
     		}
     		
     		if(f==0) cout<<"Yes"<<endl;
     		else cout<<"No"<<endl;
     	}
     	
    }
    return 0;
}
