#include <iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;

int main()
{
    int t,n;
    long long int a,b;
    scanf("%d",&t);
    while(t--){
    	scanf("%lld %lld %d",&a,&b,&n);
    	long int*arr=new long int[n];
    	for(int i=0;i<n;i++){
    		scanf("%ld",&arr[i]);
    	}
    	sort(&arr[0],&arr[n]);
    	long int i=0;
    	while((a>=0)&&(b>=0)&&(i<n)){
    		a-=arr[i];
    		b-=arr[i];
    		i++;
    	}
    	if((a<0)&&(b<0)){
    		printf("Tie\n");
    	}
    	else if(a<0){
    		printf("Sayan Won\n");
    	}
    	else if(b<0){
    		printf("Raghu Won\n");
    	}
    	else{
    		printf("Tie\n");
    	}
    }
    return 0;
}
