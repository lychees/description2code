#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,l;
    long int sum;
    string s;
    cin>>t;
    cin.ignore();
    while(t--)
    {
    	sum=0;
    	getline(cin,s);
    	for(int i=0;s[i]!='\0';i++)
    	sum+=s[i];
    	l=s.length();
    	if((sum/l)%2==1)
    	{
    		for(int i=l-1;i>=0;i--)
    		cout<<s[i];
    		cout<<"\n";
    	}
    	else
    	cout<<s<<"\n";
    	
    }
    return 0;
}
