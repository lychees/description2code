#include <stdio.h>
#include <math.h>
#include <algorithm>
#include <queue>
using namespace std;



int main(int argc, char const *argv[])
{
    priority_queue<pair<long long,pair<long long,long long> > > pq;
    long long n,id,old,p,l,c,s;
    scanf("%lld",&n);
    for(int i=0;i<n;i++){
    	scanf("%lld %lld %lld %lld %lld %lld",&id,&old,&p,&l,&c,&s);
    	long long curr=50*p+5*l+10*c+20*s-old;
    	pq.push(make_pair(curr,make_pair(id,old)));
    }
    for(int i=0;i<5;i++){
    	printf("%lld %lld\n", (pq.top().second).first,(pq.top().second).second+pq.top().first);
    	pq.pop();
    }
}
