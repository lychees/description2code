#include <bits/stdc++.h>
using namespace std;
int palindrome(string s)
{
	int ans = 0, i;
	int lenggth = s.size();
	int temp1[51] = {}, temp2[51] = {};
	ans += s.size();
	for(int leng = 2; leng <= lenggth; leng++)
	{
		if((leng&1)==0)
		{
			for(i = 0; i <= s.size() - leng; i++)
				{
					if(s[i] == s[i+leng-1] && temp2[i+1] == 0)
						ans++, temp2[i] = 0;
					else	
						temp2[i] = 1;
				}
		}
		else
		{
			for(i = 0; i <= s.size()-leng; i++)
				{
					if(s[i] == s[i+leng-1] && temp1[i+1] == 0)
						ans++, temp1[i] = 0;
					else
						temp1[i] = 1;
				}
		}
	}
	return ans;
}
int main()
{
    string s;
    cin >> s;
    cout << palindrome(s) << endl;
    return 0;
}
