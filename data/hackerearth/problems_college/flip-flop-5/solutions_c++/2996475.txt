#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
	int t;
	scanf("%d ",&t);
	while(t--)
	{
		char a[100001];
		scanf("%s",a);
		int set,cnt=0;
		if(a[0]=='X')
			set=1;
		else
			set=0;
		int n=strlen(a);
		for(int i=1;i<n;)
		{
			if(set==0)
			{
				while(i<n && a[i]=='Y')
				{
					cnt++;
					i++;
				}
				if(i<n && a[i]=='X')
				{
					set=1;
					i++;
				}
			}
			else
			{
				while(i<n && a[i]=='X')
				{
					cnt++;
					i++;
				}
				if(i<n && a[i]=='Y')
				{
					set=0;
					i++;
				}
			}
		}
		printf("%d\n",cnt);
	}
	return 0;
}
