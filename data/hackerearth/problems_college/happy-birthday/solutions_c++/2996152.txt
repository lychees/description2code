#include<iostream>
using namespace std;
int main()
{

        if(t==1)

            b++;
        else if(t==3)
            c++;

    }
    m=d;
    if(c>=a)
    {
        m=m+a;
        c=c-a;
        if(c!=0)
            m=m+c;
        if(b%2==0)
            m=m+b/2;
        else
            m=m+b/2+1;
    }
    else
    {
        m=m+c;
        a=a-c;
        if(b%2==0)
            {
            	m=m+b/2;
            	if(a%4==0)
            	m=m+a/4;
            	else
            	m=m+a/4+1;
            }
        else
        {
            m=m+b/2;
            a=a+2;
            if(a%4==0)
            m=m+a/4;
            else
                m=m+a/4+1;
        }
    }
    cout<<m<<endl;

}
