#include <iostream>
using namespace std;

int main()
{
    long int t,x,p;
    cin >> t;
    while(t--)
    {
    	cin >> x >> p;
    	int ctr=0;
    	while(x%p==0)
    	{
    		x /= p;
    		ctr++;
    		if(x==1)
    		{
    			x = 0;
    			break;
    		}
    	}
    	cout << ctr << " " << x << endl;
    }
    return 0;
}
