#include<iostream>
using namespace std;
int main()
{int t,n,temp;
cin>>t;
long res[t],sum=0;
for(int z1=0;z1<t;z1++)
{  cin>>n;
int m[n],w[n];
sum=0;
for(int j=0;j<n;j++)
cin>>m[j];
for(int j=0;j<n;j++)
cin>>w[j];
//sort start
for(int i=0;i<n;i++)
{
for(int j=0;j<n-1;j++)
{
if(m[j]>m[j+1])
{
temp=m[j];
m[j]=m[j+1];
m[j+1]=temp;
}
if(w[j]>w[j+1])
{
temp=w[j];
w[j]=w[j+1];
w[j+1]=temp;
}
}
}//sort end
for(int i=0;i<n;i++)
sum=sum+(m[i]*w[i]);

res[z1]=sum;
}
for(int k=0;k<t;k++)
cout<<res[k]<<"\n";
	
}
