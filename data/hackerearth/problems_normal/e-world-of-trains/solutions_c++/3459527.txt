const int maxn = 3343;
int n, l, t, k;
int f[maxn][maxn];
int g[maxn << 2];

void solve() {
	scanf("%d%d%d%d", &n, &l, &t, &k);
	if (l == 1) {
		if (t != n) {
			puts("0");
		}
		else {
			printf("%d\n", fpow(k, n));
		}
		return;
	}
	f[0][0] = g[maxn] = 1;
	FOR(j, 0, t + 1) {
		int tot = f[0][j], cur = 0;
		FOR(i, 1, n + 1) {
			while (cur < i - l + 1) {
				submod(tot, f[cur][j]);
				cur++;
			}
			addmod(f[i][j], (ll) (k - 1) * tot % MOD);
			addmod(f[i][j], (ll) (k - 1) * g[i - j - l + 1 + 2 * maxn] % MOD);
			if (i - l + 1 <= 0) {
				addmod(f[i][j], f[0][j]);
			}
			if (i >= l && i - j - l + 1 <= 0) {
				addmod(f[i][j], f[0][j - i + l - 1]);
			}
			addmod(tot, f[i][j]);
		}
		FOR(i, 0, n + 1) {
			addmod(g[i - j + 2 * maxn], f[i][j]);
		}
	}
	printf("%d\n", f[n][t]);
}

int main() {
//	ios_base::sync_with_stdio(0); cin.tie(0);
#ifdef _LOCAL_

#endif
	solve();
#ifdef _LOCAL_

	return 0;
}
