#include <bits/stdc++.h>

#define fi first
#define se second
#define pb push_back
#define mp make_pair
#define pii pair<int,int>
#define pll pair<ll,ll>
#define MOD 274877906944ll
#define ll long long int
#define MAXINT 1000000001ll
#define SET(x,y) memset(x,y,sizeof(x));

using namespace std;

typedef struct node {
    int val,x,y,z,t;
    bool operator < (const node &b) const{
        return val<=b.val;
    }
}node;

pii A[100005];

void solve(void) {
	int n,l,i,j,x,y,lim,tar,f=0;
	scanf("%d%d",&n,&l);
	for(i=0;i<n;i++) {
		scanf("%d%d",&x,&y);
		A[i]=mp(x,y);
	}
	sort(A,A+n);
	for(i=0;i<n;i++) {
		lim=A[i].se;
		tar=A[i].fi+l;
		for(j=i+1;j<n;j++) {
			if(A[j].fi>lim)	break;
			if(A[j].se<=tar)	lim=max(lim,A[j].se);
		}
		if(lim==tar)	f=1;
	}
	if(f)	printf("Yes\n");
	else	printf("No\n");
}

int main() {
	int t;
	scanf("%d",&t);
	while(t--)
		solve();
}
