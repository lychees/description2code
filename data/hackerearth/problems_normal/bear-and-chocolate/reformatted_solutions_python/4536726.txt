for _ in xrange(input()):
    size = input()
    hor, vert = [0]*size, [0]*size
    choco = 0
    for j in xrange(size):
        val = raw_input()
        for chk in xrange(len(val)):
            if val[chk] == '#':
                hor[chk] += 1
                vert[j] += 1
                choco += 1
    flag = False
    present_sum = 0
    for a in xrange(size):

        present_sum += hor[a]
        if present_sum == choco - present_sum:
            flag = True
            break
    if not flag:

        present_sum = 0
        for a in xrange(size):
            present_sum += vert[a]
            if present_sum == choco - present_sum:
                flag = True
                break
    print "YES" if flag else "NO"
