#include <iostream>
using namespace std;

int main()
{
    int t=0,i=0,j=0,n=0,p=0,q=0,sum=0,temp=0;
    cin>>t;
    while(t>0)
    {
    	cin>>n;
    	
    	int a[n];
    	for(p=0;p<n;p++)
    	    cin>>a[p];
    	 
    	 for(p=0;p<n;p++)
{
  for(q=0;q<n;q++)
  {
    if(a[p]<a[q])
    {
     temp = a[p];
     a[p] = a[q];
     a[q] = temp;
    }
  }
}
    
       	   
       	   cin>>i>>j;
       	   
       	   sum=a[j-1]+a[n-i];
       	   
       	 
       	   cout<<sum<<endl;
    	t--;
    }
    
    
    return 0;
}
