typedef long long int ll;
const int mx = 110;
using namespace std;
int t, n;
map <int, int> m;
vector <int> v;
int main(){
//    ios_base::sync_with_stdio(0);
//    cin.tie(0);
    //freopen("data.in","r",stdin);
    scanf("%d", &t);
    while(t--){
        scanf("%d", &n);
        for(int i = 0; i < n; i++){
            int a ;
            scanf("%d", &a);
            if(m[a] == 0)v.push_back(a);
            ++m[a];
        }
        ll ans = 0;
        for(int i = 0; i <(int)v.size(); i++){
            ans += (1LL * m[v[i]] * (1LL * m[v[i]] - 1) / 2);
            ans += (1LL * m[v[i]] );
        }
        printf("%lld\n", ans);
        m.clear();
        v.clear();
    }
    return 0;
}
