#include<bits/stdc++.h>
using namespace std;
void fn(string x)
{
	for(int i = 0 ; i < x.size()-1 ; ++i )
	 {
	 	cout << x[i]<<"-";
	 }
	 cout<<x[x.size()-1] << endl;
}
string rev(string x)
{
	string y;
	for(int i = x.size()-1 ; i >= 0 ; --i )
	  y+=x[i];
	  
	  return y;
	
}
int main(){
	
	 string x;
	 cin >> x;
	    fn(x);
	    fn(rev(x));
}
