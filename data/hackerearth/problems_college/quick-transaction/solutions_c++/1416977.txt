#include<iostream>
using namespace  std;
#define ll long long int
int main()
{
      int t;
	  std::ios::sync_with_stdio(false);
	  cin>>t;
	  while(t--)
	  {
	     ll x,count=0,m;
		 cin>>x;
		 ll arr[9]={1000,500,100,50,20,10,5,2,1};
		 ll sum=x;
		 for(ll i=0;i<9;i++)
		 {
		     if((sum/arr[i]>0)&&(sum>0))
			 {
			     m=(sum/arr[i]);
				 sum=sum-(m*arr[i]);
				 count=count+m;
	
				 
			 }
			 else if(sum==0)
			 break;
		 }
		 cout<<count<<endl;
	  }
	  return 0;
}
