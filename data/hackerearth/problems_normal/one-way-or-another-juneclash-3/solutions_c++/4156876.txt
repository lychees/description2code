#include <iostream>

using namespace std;

int main()
{
    int n,t,count=0;
    long long int r,c,x,j,sum;
    char par,p;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        cin>>n;
        for(j=0;j<n;j++)
        {
            cin>>r;
            cin>>c;
            cin>>x;
            sum=r+c;
            if(j==0)
            {
            if((sum%2!=0&&x%2!=0)||(sum%2==0&&x%2==0))
            {
                par='o';
            }

            else if((sum%2==0&&x%2!=0)||(sum%2!=0&&x%2==0))
            {
                par='e';
            }
           }
           if(j!=0)
           {
               if((sum%2!=0&&x%2!=0)||(sum%2==0&&x%2==0))
                {
                p='o';
                }

                else if((sum%2==0&&x%2!=0)||(sum%2!=0&&x%2==0))
                {
                p='e';
                }
                if(p!=par)
                {
                    count=1;
                }
            }
         }
         if(count==0)
            cout<<"Yes\n";
            else
                cout<<"No\n";
            count=0;
    }
}
