#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
using namespace std;
int cmpfunc (const void *a, const void *b){
 return (*(char *)a >*(char *)b);
}
int main(){
    int T,len,i,pos,minimum,j;
    char s[102];
    scanf("%d",&T);
    while(T--){
        scanf("%s",s);
        len=strlen(s);
        for(i=len-2;i>=0;i--){
            if(s[i]<s[i+1])
                break;
        }
        if(i!=-1){
            minimum='z'+1;
        for(j=i+1;j<=len-1;j++){
            if(s[j]<minimum && s[j]>s[i]){
                minimum=s[j];
                pos=j;
            }
        }
        swap(s[pos],s[i]);
        qsort(s+i+1,len-i-1,sizeof(char),cmpfunc);
        printf("%s\n",s);
        }
        else{
        printf("no answer\n");
        }
    }
        return 0;
}
