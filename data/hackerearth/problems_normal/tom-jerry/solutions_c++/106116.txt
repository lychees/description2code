#include<iostream>
#include<cstdio>
using namespace std;
int res = 0;
int matrix[101][101];
int vis[101][101];
void dfs(int i, int j, int n)
{
    if(i>n || j>n || i<=0 || j<=0 || matrix[i][j] == 1 || vis[i][j] == 1)
    {
        return;
    }
    else
    {
        vis[i][j] = 1;
        if(i == n && j == n)
        {
            res++;
            vis[i][j] = 0;
            return;
        }
        dfs(i-1,j,n);
        dfs(i+1,j,n);
        dfs(i,j-1,n);
        dfs(i,j+1,n);
        vis[i][j] = 0;
        return;
    }
}
int main()
{
    int n,i,j;
    scanf(" %d",&n);

    for(i=0; i<=n; i++)
    {
        matrix[i][0] = 1;
        vis[i][0] = 0;
    }
    for(i=0; i<=n; i++)
    {
        matrix[0][i] = 1;
        vis[0][i] = 0;
    }
    for(i=1; i<=n; i++)
    {
        for(j=1; j<=n; j++)
        {
            vis[i][j] = 0;
            scanf(" %d",&matrix[i][j]);
        }
    }
    dfs(1,1,n);
    printf("%d\n",res);
    return 0;
}
