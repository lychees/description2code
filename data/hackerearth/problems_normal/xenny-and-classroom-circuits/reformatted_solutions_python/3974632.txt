for _ in xrange(int(raw_input())):
    n = input()
    q = raw_input()
    ct, ct_2 = 0, 0
    for i in xrange(2*n):
        if i % 2 and q[i] == 'G':
            ct += 1
        elif i % 2 == 0 and q[i] == 'B':
            ct += 1
    for i in xrange(2*n):
        if i % 2 and q[i] == 'B':
            ct_2 += 1
        elif i % 2 == 0 and q[i] == 'G':
            ct_2 += 1
    print min(ct, ct_2)/2
