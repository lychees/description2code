def min_balanced_strings(st):
	dicti = {}
	ret = 1
	for i in range(0,len(st)):
		if st[i] in dicti:
			dicti[st[i]] = dicti[st[i]]+1
		else:
			dicti[st[i]] =1
		
	for v in dicti.values():
		if v%2 != 0:
			return -1
		
	return 1

loops = raw_input()

	print(min_balanced_strings(j))
