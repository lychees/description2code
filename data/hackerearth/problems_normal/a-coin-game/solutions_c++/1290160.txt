#define fori(i, a, b) for(int (i) = (a); (i) < (b); (i)++)
#define ford(i, a, b) for(int (i) = (a)-1; (i) >= (b); --(i))
#define test(t) while((t)--)
#define MP make_pair
#define mod 1000000007

int main() {
    int t, diff = 1;
    int dp[1000001] = {0};
    fori(i, 1, 1000001) {
        if (!dp[i]) {
            dp[i] = diff;
            if (i+diff <= 1000000)
                dp[i+diff] = diff;
            diff++;
        }
    }
    cin >> t;
    int a, b;
    test(t) {
        cin >> a >> b;
        if (!(a == 0 && b == 0)) {
            if (a == b || a == 0 || b == 0) {
                cout << "Play" << endl;
                continue;
            }
        }
        if (dp[a] == dp[b])
            cout << "Don't Play" << endl;
        else
            cout << "Play" << endl;
    }

    return 0;
}
