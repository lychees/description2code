	for(i=0;i<n;i++)

	sort(a,a+n);
	if(n<3)
	{
		if(n == 1)
			cout << "-1"<<endl;
		else
		{
			int d = a[1] - a[0];

			{
				cout << "2"<<endl;
				cout << a[0]-d<<" "<<a[1]+d<<endl;
			}
			else
			{
				cout<<"3"<<endl;
				cout<< a[0]-d<<" "<<(a[0]+a[1])/2<<" "<<a[1]+d<<endl;
			}
		}
	}

		int y,d = 99999999, x = 0;
		for(i=1;i<n;i++)
			d = min(d,a[i]-a[i-1]);
//			cout << d<<endl;
		for(i=1;i<n;i++)
		{
			if(a[i]-a[i-1] == 2*d)
			{

				y = a[i] - d;
			}
			else if(a[i]-a[i-1] != d)
			{
				cout <<"0"<<endl;
				return 0;
			}
		}
		if(x == 1)
		{
			cout <<"1"<<endl<<y<<endl;
		}

		if(x==0)
			cout << "2"<<endl<<a[0]-d<<" "<<a[n-1]+d<<endl;

			cout<<"1"<<endl<<a[0]<<endl;
		else

		}
	}
}
