#include<stdio.h>
long long int sum(long long int num)
	{
		long long int sum=0;
		while(num)
			{
				sum+=num%10;
				num/=10;
			}
		//printf(" %lld",sum);
		return sum;
	}
	
int main()
	{
		int t,i;
		long long int n;
		int flag;
		bool a[200];
		scanf("%d",&t);
		while(t-->0)
			{
				for(i=0;i<200;++i)
					a[i]=0;
				scanf("%lld",&n);
				
				flag=0;
				if(n==1||n==4)
					{
						printf("YES\n");
					}
				else
				{
				while(1)
					{
						n=sum(n*n);
						if(n==1||n==4)
							{
								flag=1;
								break;
							}
						if(a[n]==1)
							{
								break;
							}
						else
							a[n]=1;
					}
				if(flag==1)
					printf("YES\n");
				else
					printf("NO\n");
				}
			}
	}
