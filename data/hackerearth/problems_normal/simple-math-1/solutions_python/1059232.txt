def choose(n, k):
    if 0 <= k <= n:
        ntok = 1
        ktok = 1
        for t in xrange(1, min(k, n - k) + 1):
            ntok *= n
            ktok *= t
            n -= 1
        return ntok // ktok
    else:
        return -1



a,b = map(int,raw_input().split())
answer = choose(a-1, b-1)
if answer == -1:
	print -1
else:
	print answer % 10000007
