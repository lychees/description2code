#include <iostream>
using namespace std;

int main()
{
    long long int t, r, c, i, j, k, n, m, l;
    scanf("%lld", &t);
    while (t--) {
    	scanf("%lld%lld", &r, &c);
    	m = min(r, c);
    	if (m % 2 == 0) {
    		n = m / 2;
    		i = (n * (2 + m)) / 2;
    		j = (n * m) / 2;
    	} else {
    		n = m / 2;
    		i = (n * (1 + m)) / 2;
    		j = ((n + 1) * (1 + m)) / 2;
    	}
    	if ((r - m) > 0) {
    		n = r - m;
    	if (n % 2 == 0) {
    		i = i + (m * (n / 2));
    		j = j + (m * (n / 2));
    	} else {
    		if (m % 2 == 0) {
    			j = j + (m * (n / 2 + 1));
    			i = i + (m * (n / 2));
    		} else {
    			i = i + (m * (n / 2 + 1));
    			j = j + (m * (n / 2));
    		}
    	}
    	}
    	printf("%lld %lld\n", i, j);
    }
    return 0;
}
