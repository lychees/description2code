#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
LL n,h,mod=1000000007;
#define MAX(a,b) (a>b?a:b)
#define MIN(a,b) (a>b?b:a)
LL dp[600][600];
LL func(LL num,LL height){
	if(height<0){return 0;}
	if(num<0){return 0;}
	if(dp[num][height]!=-1){return dp[num][height];}
	if(num==0){dp[num][height]=1;return 1;}
	if(height==0){
		dp[num][height]=0;
		return 0;
	}
	LL ret=0;
	for(LL i=1;i<=num;i++){
		LL val1=func(i-1,height-1);val1%=mod;
		LL val2=func(num-(i+1)+1,height-1);val2%=mod;
		ret+=(val1*val2)%mod;
		ret%=mod;
	}
	dp[num][height]=ret;
	return ret;
}
int main(){
	cin>>n>>h;
	for(LL i=0;i<=510;i++)for(LL j=0;j<=510;j++)dp[i][j]=-1;
	LL rett=0;
	if(h!=0){
		for(LL ret=1;ret<=n;ret++){
			LL val1=func(ret-1,h-1);val1%=mod;
			LL val2=func(n-(ret+1)+1,h-1);val2%=mod;
			LL val=(val1*val2)%mod;
			rett+=((val*ret)%mod);
			rett%=mod;
		}
	}else{
		rett=0;
	}
	cout<<rett<<"\n";
	return 0;
}
