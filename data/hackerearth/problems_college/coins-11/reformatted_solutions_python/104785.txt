A = []


def fun():
    for i in range(0, 100005):
        A.append(0)
    A[0] = 1
    A[1] = 1
    for i in range(2, 1000):
        if (A[i] == 0):
            j = i*i
            while (j <= 100002):
                A[j] = 1
                j = j+i


fun()
t = int(raw_input())
while (t != 0):
    n = int(raw_input())
    if (n <= 3):
        print n
    else:
        if (A[n] == 1):
            print "1"
        else:
            print "2"
    t = t-1
