#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,a[100001],n,valid=0;
	scanf("%d",&t);
	while(t--)
	{
		valid=0;
		scanf("%d",&n);
		if(n<3)
		{
			for(int i=0;i<n;i++)
			scanf("%d",&a[i]);
			printf("-1\n");
		}
		else
		{
		for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
		sort(a,a+n);
		for(int i=n-1;i>=0;i--)
		{
			for(int j=i-1;j>=0;j--)
			{
				for(int k=j-1;k>=0;k--)
				{
					if(a[k]>a[i]-a[j])
					{
					  printf("%d %d %d \n",a[k],a[j],a[i]);
					  valid=1;
					  break;
				   }
			  }
			  if(valid==1)
			  break;
					
			}
			if(valid==1)
			break;
		}
		if(valid==0)
		printf("-1\n");
	}
}
return 0;
}
