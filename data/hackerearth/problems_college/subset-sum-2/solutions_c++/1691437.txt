#include <iostream>
using namespace std;
#define MOD 1000000007
long long po[10004];

int main()
{
    po[0] = 1;
    for(int i = 1; i <= 2000; i++) {
    	po[i] = po[i-1] * 2;
    	po[i] %= MOD;
    }
    
    
    int t;
    cin >> t;
    
    while(t--) {
    	int n;
    	
    	cin >> n;
    	long long ans = 0;
    	for(int i = 1; i <= n; i++) {
    		ans = (ans + i*po[n-1])%MOD;
    	}
    	cout << ans << endl;
    }
    return 0;
}
