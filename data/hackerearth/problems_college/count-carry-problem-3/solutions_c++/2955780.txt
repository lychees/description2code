#include <iostream>
using namespace std;

int main()
{
    int t,temp1,temp2;
    cin>>t;
    long int a[2],i;
    int ans=0;
    for(i=0;i<t;i++)
    {
    	ans=0;
    	cin>>a[0];
    	cin>>a[1];
    	do
    	{
    		temp1=a[0]%10;
    		temp2=a[1]%10;
    		if(temp1+temp2>=10)
    		ans+=1;
    		a[0]=a[0]/10;
    		a[1]=a[1]/10;
    	}while(a[0]>0 &&a[1]>0);
    	if(ans==0)
    	cout<<"No carry operation\n";
    	else if(ans==1)
    	cout<<ans<<" carry operation\n";
    	else
    	cout<<ans<<" carry operations\n";
    }
    return 0;
}
