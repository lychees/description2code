#include <iostream>
using namespace std;

int main()
{
	int n,m,i;
    cin>>n>>m;
    long int a[m];
    for(i=0;i<m;i++)
    {
    	cin>>a[i];
    }
    long int b[n],j,sum=0;
    b[0]=b[1]=1;
    for(i=2;i<n;i++)
    {
    	b[i]=b[i-1]+b[i-2];
    }
    for(i=0;i<m;i++)
    {
    	for(j=0;j<n;j++)
    	{
    		if(a[i]==b[j])
    		{
    			a[i]=-1;
    			b[j]=0;
    			break;
    		}
    	}
    }
    for(i=0;i<n;i++)
    {
    	sum+=b[i];
    }
    cout<<sum;
    return 0;
}
