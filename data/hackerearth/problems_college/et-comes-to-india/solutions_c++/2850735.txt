#include<cstdio>
#define ll long long
//NOW will start
using namespace std;
char ch[4];
long long len[55];
int inv;
#define gc getchar_unlocked
inline void scanint(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
    
}
int get(int N,long long K)
{
	//printf("searching %d %lld\n",N,K);
	if(N==1)
		{
			
			if(K==1) return 'G'; else return -1; 
		}
		if(N==2)
		{
			if(K<3) return 'G'; if(K==3) return 'C';
			return -1;
		}
		
		long long t=len[N-1];  
		//USEd FOR
		if(t>=K) return get(N-1,K);
		if(t+1 == K) return 'G';
		inv++;
		K=K-t-1;
		//RETURNING
		return get(N-1,t-K+1); 		
		
}
int main()
{

	ch[0]='G'; ch[1]='G'; ch[2]='C';
	
	len[1]=1; len[2]=3;
	for(int i=3;i<54;i++) len[i]=2*len[i-1]+1; 

	int ntc; scanint(ntc);
	while(ntc--)

		int N; long long K; scanint(N); scanf("%lld",&K);
		//TAKING INPUT

			if(K==1) printf("G\n"); else printf("-1\n"); continue;

			if(K<3) printf("G\n"); if(K==3) printf("C\n");
			if(K>3) printf("-1\n"); continue;
		}
		
		 inv=0;
		
		if(len[N]<K) 
		{
			printf("-1\n"); continue;
		}
		
		int aa=get(N,K);

	}
