using namespace std;
typedef long long ll;
//#define DEBUG
#define mod 1000000007
#define pb push_back

int arr[102], N;

int lis(){
    int L[N];
    for(int i=0;i<N;i++)
        L[i] = 1;
    for(int i=1;i<N;i++){
        for(int j=0;j<i;j++){
            if(arr[i] > arr[j] && L[i] < L[j]+1)
                L[i] = L[j] + 1;
        }
    }
    int maxi=-1;
    for(int i=0;i<N;i++)
        maxi = max(maxi, L[i]);
    return maxi;
}

int main(){
    //ios::sync_with_stdio(false);
    #ifdef DEBUG
    freopen("input.txt", "r", stdin);
    #endif // DEBUG

    scanf("%d", &N);
    for(int i=0;i<N;i++)
        scanf("%d", &arr[i]);
    int ans = lis();
    //cout<<ans<<endl;

    vector<int> v;
    while(ans>0){
        v.pb(ans%2);
        ans/=2;
    }
    for(int i=v.size()-1;i>=0;i--)
        printf("%d", v[i]);
    printf("\n");

    return 0;
}
