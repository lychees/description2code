#include <bits/stdc++.h>
using namespace std;
//...................max sum...................................
int max_sum(int arr[], int n) {
   int max[100000], i = 0;
   max[0] = 0;
   max[1] = arr[0];  
   for(i = 1 ; i < n ; i++) {
      int val = max[i-1] + arr[i];
      if(val > max[i]) {
         max[i+1] = val;
      }
      else
         max[i+1] = max[i];
   }
   return max[i];
}
//...............main function................................
int main() {
   int arr[100000], n, i, t, res;
   scanf("%d", &t);
   while(t--) {
      scanf("%d", &n); 
      for(i = 0 ; i < n ; i++) 
         scanf("%d", &arr[i]);  
      res = max_sum(arr, n);    
      printf("%d\n", res);
   }
   return 0;
}
