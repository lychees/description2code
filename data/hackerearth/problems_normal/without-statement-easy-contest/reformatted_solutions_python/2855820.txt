t = int(raw_input())
while t > 0:
    n, m = map(int, raw_input().split())
    s = str(n)
    visited = []
    while m > 0:
        if s in visited:
            j = visited.index(s)
            s = visited[j+((m) % (len(visited)-j))]
            break
        sm = 0
        visited.append(s)
        for i in s:
            sm += int(i)**2
        s = str(sm)
        m -= 1
    print s
    t -= 1
