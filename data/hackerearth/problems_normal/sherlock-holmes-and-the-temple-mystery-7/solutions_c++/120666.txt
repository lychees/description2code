#include <iostream>
#include <map>
#include <vector>
#include <algorithm>

using namespace std;

bool isSubsetSum(int set[], int n, int sum)
{
    
    bool subset[sum+1][n+1];

   
    for (int i = 0; i <= n; i++)
      subset[0][i] = true;

  
    for (int i = 1; i <= sum; i++)
      subset[i][0] = false;

   
     for (int i = 1; i <= sum; i++)
     {
       for (int j = 1; j <= n; j++)
       {
         subset[i][j] = subset[i][j-1];
         if (i >= set[j-1])
           subset[i][j] = subset[i][j] || subset[i - set[j-1]][j-1];
       }
     }

     return subset[sum][n];
}

int main()
{
    int n, t, sum;

    cin >> t;

    while (t--) {
        cin >> n;

        int a[n];
        sum = 0;

        for (int i = 0; i < n; i++) {
            cin >> a[i];
            sum += a[i];
        }

        if (sum % 2 == 0 && isSubsetSum(a, n, sum / 2)) {
            cout << "YES\n";
        } else {
            cout << "NO\n";
        }
    }
}
