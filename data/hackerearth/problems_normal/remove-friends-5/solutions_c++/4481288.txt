using namespace std;

vector<int> ans;

int main()
{
    int t,n,k,x,p,i;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d%d",&n,&k);
    	ans.clear();
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&p);
    		if(!i)
    			ans.push_back(p);
    		else
    		{
    			while(!ans.empty()&&k>0)
    			{
    				if(ans[(int)ans.size()-1]<p)
    				{
    					ans.pop_back();
    					k--;
    				}
    				else
    					break;
    			}
    			ans.push_back(p);
    		}
    	}
    	while((k--)>0)
    		ans.pop_back();
    	for(i=0;i<(int)ans.size();i++)
    		printf("%d ",ans[i]);
    	printf("\n");
    }
    return 0;
}
