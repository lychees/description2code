#include <iostream>
using namespace std;
int rec(string s)
{
	int n = s.size();
	int l =0;
	int h = n-1;
	 while (h > l)
    {
        if (s[l++] != s[h--])
        {
            
            return 0;
        }
    }
    return 1;
}
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	string r, s;
    	cin >> r >> s;
    	string a;
    	int i,n = r.size();
    	int c=0;
    	for(i=0;i<=n;i++)
    	{
    		a = r;
    		a.insert(i,s);
    		//cout << a << endl;
    		if(rec(a)==1)
    		c++;
    	}
    	cout << c << endl;
    }
    return 0;
}
