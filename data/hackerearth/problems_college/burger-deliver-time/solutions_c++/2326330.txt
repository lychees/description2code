#include <bits/stdc++.h>
#define ll long long
using namespace std;
typedef struct node
{
	int i,t; //index,time & processing time
}node;
bool my(node x,node y)
{
	if(x.t<y.t)
	return true;
	if(x.t==y.t)
	{
		if(x.i<y.i)
		return true;
	}
	return false;
}
node a[100010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int n;
	cin>>n;
	int i,x,y;
	for(i=0;i<n;i++)
	{
		cin>>x>>y;
		x+=y;
		a[i].i=i+1;
		a[i].t=x;
	}
	sort(a,a+n,my);
	for(i=0;i<n;i++)
	{
		cout<<a[i].i<<" ";
	}
	cout<<endl;
	return 0;
}
