#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pii pair<int,int>
#define pll pair<ll,ll>
#define mp make_pair
#define ff first
#define ss second
#define pb push_back
#define pf pop_front
#define IOS ios::sync_with_stdio(false)
ll n , a[100005] , dp1[100005] ,dp2[100005];
int main(){

    //freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
	ll i , j , k;
    cin >> n;
    for(i = 1 ; i <= n ; i++)cin >> a[i];
    dp1[1] = (a[1] > a[2])?2:1;
    for(i = 2 ; i <= n ; i++){
        if(a[i] > a[i-1])
            dp1[i] = dp1[i-1] + 1;
        else dp1[i] = 1;
    }
    for(i = n-1 ; i >= 1 ; i--){
        if(a[i] > a[i+1] && dp1[i] <= dp1[i+1])
            dp1[i] = dp1[i+1] + 1;
    }
    dp2[n] = (a[n] > a[n-1])?2:1;
    for(i = n-1 ; i >= 1 ; i--){
        if(a[i] > a[i+1])
            dp2[i] = dp2[i+1]+1;
        else dp2[i] = 1;
    }
    ll sum = 0;
    for( i = 1 ; i <= n ;i++){
        sum+= max(dp2[i],dp1[i]) ;
    }
    cout << sum << endl;
	return 0;
}
