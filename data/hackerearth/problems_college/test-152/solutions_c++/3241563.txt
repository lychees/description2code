#include<cstdio>
#include<cstdlib>
#include<iostream>
using namespace std;
int main()
{	int t;
	cin>>t;
	while(t--)
	{	int n;
		cin>>n;
		char  a[n];
		cin>>a;
		int hash[27];
		int max=0;
		int i;
		for(i=0;i<27;i++)hash[i]=0;
		for(i=0;a[i]!='\0';i++)
		{	hash[(int)a[i]-97]+=1;
			if(hash[(int)a[i]-97]>max)max=hash[(int)a[i]-97];
		}
		if(max%2==0)cout<<"Yes\n";
		else cout<<"No\n";
		//cout<<max;
	}
	return 0;
}
