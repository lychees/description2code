#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mset(m,v) memset(m,v,sizeof(m))
#define pb push_back
#define mp make_pair
#define fi first
#define se second
#define pi 3.1415926535897932384
#define mod 786433
 
const int MAXM = 3e2+5;
map<ll,ll>m,mm;

vector<ll> v,v1;

int main()
{
    int i,j,n,q;
    ll k,x;
    scanf("%d%d",&n,&q);
    for(i=1;i<=n;i++){
        scanf("%lld",&k);
        mm[i]=k;
        v.pb(k);
    }

    while(v.size()!=1){
        for(i=1;i<v.size();i+=2){
            m[v[i-1]]++;
            m[v[i]]++;
            ll p=max(v[i],v[i-1]);
            v1.pb(p);
        }
        if(v.size()&1)  v1.pb(v[v.size()-1]);
        v=v1;
        v1.clear();
    }

    while(q--){
        scanf("%lld",&x);
        printf("%lld\n",m[mm[x]]);
    }

    return 0;
}
