#include <iostream>
using namespace std;
int T, N, C, M;
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cin >> T;
    while (T--) {
        cin >> N >> C >> M;
        int tot = 0;
        int wrappers = 0;
        while (N >= C) {
            N -= C;
            ++tot;
            ++wrappers;
            if (wrappers == M) {
                wrappers = 1;
                ++tot;
            }
        }
        cout << tot << "\n";
    }
    return 0;
}
