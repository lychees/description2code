#include<iostream>
#include<vector>
#include<cstdio>
using namespace std;
long long int  numDistinct(string S, string T)
{

        vector<long long int > f(T.size()+1);

        //set the last size to 1.
        f[T.size()]=1;

        for(int i=S.size()-1; i>=0; --i){
            for(int j=0; j<T.size(); ++j){
                f[j]+=(S[i]==T[j])*f[j+1];
              //  printf("%d\t", f[j] );
            }
          //  cout<<"\n";
        }
        return f[0];
    }
int main()
{
    string str;
    string arr="101";
    cin>>str;
    long long int ans=numDistinct(str,arr);
    cout<<ans;
    return 0;
}
