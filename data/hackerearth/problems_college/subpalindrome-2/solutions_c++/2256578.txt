#include<bits/stdc++.h>
using namespace std;
char s[1000];
int n;
long long int a[1002][1002];
void find()
{
	int i,j,l;
	for(i=0;i<n;++i)
	a[i][i]=1;
	for( l=2;l<=n;++l)
	{
		for(i=0,j=i+l-1;j<n;++i,++j)
		{
			if(s[i]!=s[j])
			a[i][j]=a[i+1][j]-a[i+1][j-1]+a[i][j-1];
			else
			a[i][j]=a[i+1][j]-a[i+1][j-1]+a[i][j-1]+1+a[i+1][j-1];
		}
	}
	printf("%lld\n",a[0][n-1]);
}
int main()
{
	int i,t;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",&s);
		n=strlen(s);
		find();
		memset(a,0,n*n*sizeof(int));
	}
	return 0;
}
