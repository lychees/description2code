#include <iostream>
#include <vector>
#include <math.h>
using namespace std;
int l,b;

int cost(int p){
	int digit_sum=0;
	if(p<10)
		digit_sum=p;
	else{
		digit_sum=(p/10)+(p%10);
	}
	float discount;
	if(p<=10)
		discount=0.75;
	else if(p<=20)
		discount=0.7;
	else
		discount=0.8;
		
	return floor(discount*l*b*p*digit_sum);
}

int main()
{
    //cout << "Hello World!" << endl;
    int N;
    cin>>N;
    vector<int> plank;
    for(int i=0;i<N;i++){
    	int p;
    	cin>>p;
    	plank.push_back(p);
    }
    //cout<<"plank size is"<<plank.size()<<endl;
    cin>>l>>b;
    int cost_p=0;
    for(int i=0;i<plank.size();i++){
    	//cout<<"cost of plank is"<<cost(plank[i])<<endl;
    	cost_p+=cost(plank[i]);
    }
    cout<<cost_p;
    return 0;
}
