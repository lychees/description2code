#include<bits/stdc++.h>
#define ll long long int

using namespace std;

ll fib[1000005];

ll fibo(int n)
{
    if(n==1)
        return 1;
    if(n==2)
        return 2;
    if(fib[n]==0)
    {
        fib[n]=fibo(n-1)+fibo(n-2);
    }
    return fib[n];
}

int main()
{
    fib[1]=1; fib[2]=2;
    ll x=fibo(100);

    int t;
    ll cnt;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld",&x);
        cnt=0;
        for(int i=1;i<=100;i++)
        {
            if(fib[i]<=x+cnt)
                cnt++;
            else
                break;
        }
        printf("%lld\n",x+cnt);
    }

    return 0;
}
