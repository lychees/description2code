#include <iostream>
using namespace std;
int arr[32];
int DtoBin(int x)
{
	int r=0;
	
	while(x)
	{
	  arr[r++]=x%2;
	  x/=2;
	}
	return r;
}
	
int fun(int n)
{
	int x=n;
	for(int i=n;i<100000;i++)
	{
		int r=DtoBin(i);
		int f=0;
		for(int j=0;j<r-1;j++)
		{
			if(arr[j] == 1 && arr[j]==arr[j+1])
			{f=1;
				break;
			}
		}
	if(f==0)	
	return i;
		
	}
	return x;
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	if(n & (n-1) == 0)
    	cout<<n<<endl;
    	else
         cout<<fun(n)<<endl;
    }
    return 0;
}
