#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin >> t;
    while(t--) {
    	int n, x, odd = 0;
    	cin >> n;
    	for(int i = 0 ; i < n ; ++i) {
    		cin >> x;
    		if(x % 2 == 1)
    			odd++;
    	}
    	if(odd % 2 == 0)
    		cout << "Yes";
    	else
    		cout << "No";
    	cout << endl;
    }
    return 0;
}
