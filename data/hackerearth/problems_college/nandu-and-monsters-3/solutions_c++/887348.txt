#include <bits/stdc++.h>
using namespace std;
int monster[100000];
int main()
{
   long N,Q,k,i,j,t;
    char c;
    scanf("%ld %ld",&N,&Q);
    for(i=0;i<N;++i)
        {
            scanf("%1d",&monster[i]);
        }
    for(k=0;k<Q;++k)
    {
        scanf("%ld %ld",&i,&j);
        if(i>j)
        {t=i;i=j;j=t;}
        while(1)
        {
            if(i==j)
            {   if(monster[i-1]==0)
                {printf("Yes\n");break;}
                else
                {printf("No\n");break;}
            }    
            else if(i>j)
            {   if(monster[j-1]==1)
                {printf("No\n");break;}
                else
                
                {t=i;i=j;j=t;}
            }
            else
            {
                if(monster[j-1]==1)
                {printf("No\n");break;}
                else
                {
                    if(j%2==0)
                    j=j/2;
                    else
                    j=(j-1)/2;
                }
            }
        }
    }
    return 0;
}
