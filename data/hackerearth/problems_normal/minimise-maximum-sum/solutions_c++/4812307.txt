#include <bits/stdc++.h>
using namespace std;
void minSum(long *a,long n)
{
	sort(a,a+(2*n));
	long i=0,maxSum=INT_MIN,j=(2*n)-1;
	while(i<=j)
	{
		if(a[i]+a[j] > maxSum)
		{
			maxSum=(a[i]+a[j]);
		}
		i++;j--;
	}
	cout<<maxSum<<"\n";
}
int main()
{
	long t;
	cin>>t;
	while(t--)
	{
		long n,i=0;
		cin>>n;
		long a[2*n];
		while(i<(2*n))
		{
			cin>>a[i];
			i++;
		}
		minSum(a,n);
	}
}
