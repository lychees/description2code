
N = int(raw_input())
l = [(0, 0, 0)] * N

for i in range(N):
    l[i] = map(int, raw_input().split(' '))

l = sorted(l, key=lambda a: a[0])

s = 0
end = None
for i in range(N):
    (x, _, d) = l[i]
    if end is None:
        start = x
    else:
        start = max(x, end + 1)
    end = max(x + d, end)
    s += max(0, end - start + 1)
print s
