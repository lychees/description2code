#include<bits/stdc++.h>
using namespace std;
struct bomb {
    int x,y,r;
    bomb(int a,int b,int c) {
        x=a;y=b;r=c;
    }
};
vector<bomb> v;
bool check(int x1,int y1) {
    int cnt=0;
    for(int i=0;i<v.size();i++) {
        int a=x1-v[i].x; a*=a;
        int b=y1-v[i].y; b*=b;
        int r=v[i].r; r*=r;
        a=a+b-r;
        if(a<=0) cnt++;
    }
    return cnt>=2?true:false;
}
int main() {
    int n,x,y,r;
    scanf("%d",&n);
    for(int i=0;i<n;i++) {
        scanf("%d%d%d",&x,&y,&r);
        v.push_back(bomb(x,y,r));
    }
    int cnt=0;
    for(int i=1;i<=1000;i++) {
        for(int j=1;j<=1000;j++) {
            if(check(i,j)) cnt++;
        }
    }
    printf("%d\n",cnt);
}
