#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
void inc(int index,int s,vector<int > &d){
d[index]=0;
if(index!=(d.size()-1)){
    d[index+1] +=s;
}
}
void fun(vector<int > &d,int index,int &diff){
int a = -1*d[index];
int ex=0;
    diff += a/5;
    ex = a%5;
    diff += ex/2;
    ex = ex%2;
    diff+=ex;
    inc(index,-1*a,d);
}
void calcu(vector<int > &d,vector<int > &arr){
for(int i=0;i<arr.size()-1;i++){
            d[i]=arr[i]-arr[i+1];
        }
}
int main(){
    int t,n,temp,count=0;
    cin>>t;
    while(t--){
        cin>>n;
        count=0;
        vector<int > arr(n);
        for(int i=0;i<n;i++){
            cin>>temp;
            arr[i]=temp;
        }
        sort(arr.begin(),arr.end());
        vector<int > d(n-1);
        calcu(d,arr);
        for(int i=0;i<n-1;i++){
            if(d[i]!=0){
                fun(d,i,count);
            }
        }
        cout<<count<<endl;
        arr.clear();
        d.clear();
    }
return 0;
}
