#include <iostream>
#define ll long long
using namespace std;

int main()
{
    ll t,pos,times;
    cin>>t;
   
    while(t--)
    {
    	 cin>>pos>>times;
    	if(times==0)
    	cout<<pos<<"\n";
    	else
    	{
    	if(pos==0 || pos ==2)
    	{
    		if(times%2==0)
    		cout<<"0\n";
    		else
    		cout<<"1\n";
    		
    		
    	}
    	else
    	{
    		if(times%2==0)
    		cout<<"1\n";
    		else
    		cout<<"0\n";
    		
    	}
    	}
    }
    return 0;
}
