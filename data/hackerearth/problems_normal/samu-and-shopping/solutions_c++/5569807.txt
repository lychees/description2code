#include <bits/stdc++.h>

using namespace std;


int T, N;
long long Store[100005][3];
int Shops[100005][3];

void Input()
{
    cin >> N;
    for(int i = 0; i < N; i++)
    {
        cin >> Shops[i][0] >> Shops[i][1] >> Shops[i][2];
        Store[i][0] = Store[i][1] = Store[i][2] = -1;
    }
}

int Expendit(int shops, int last)
{
    if(shops == 0)
    {
        if(last == 0)
            return min(Shops[0][1], Shops[0][2]);
        else if(last == 1)
            return min(Shops[0][0], Shops[0][2]);
        else
            return min(Shops[0][0], Shops[0][1]);
    }

    if(Store[shops][last] != -1)
        return Store[shops][last];
    else
    {
        long long x;
        if(last == 0)
            x = min(Shops[shops][1]+Expendit(shops-1, 1), Shops[shops][2]+Expendit(shops-1, 2));
        else if(last == 1)
            x = min(Shops[shops][0]+Expendit(shops-1, 0), Shops[shops][2]+Expendit(shops-1, 2));
        else
            x = min(Shops[shops][0]+Expendit(shops-1, 0), Shops[shops][1]+Expendit(shops-1, 1));
        Store[shops][last] = x;
        return x;
    }
}

int main()
{
    ios::sync_with_stdio(false);
    cin >> T;
    for(int i = 0; i < T; i++)
    {
        Input();
        cout << min( min(Expendit(N-1, 0), Expendit (N-1,1)), Expendit(N-1, 2)) << '\n';
    }
}
