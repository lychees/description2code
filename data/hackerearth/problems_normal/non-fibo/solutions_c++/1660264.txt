#include<bits/stdc++.h>
using namespace std;
#define ll long long

int main()
{
    ll t;
    scanf("%lld",&t);
    while(t--)
    {
        ll n,a,b,s,pos,cnt,ans;
        scanf("%lld",&n);
        a=1,b=2,s=3;
        cnt=0;
        while(cnt<n)
        {
            cnt=cnt+(s-a-1);
            a=b;
            b=s;
            s=a+b;

        }
        pos=cnt-n+1;
        ans=s-pos;
        printf("%lld\n",ans);
    }
    return 0;
}
