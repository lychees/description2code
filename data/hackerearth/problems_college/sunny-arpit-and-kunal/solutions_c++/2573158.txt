#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long arr[3];
    	long mid,low,high;
    	cin>>arr[0]>>arr[1]>>arr[2];
    	sort(arr,arr+3);
    	low=arr[0];
    	mid=arr[1];
    	high=arr[2];
    	//cout<<low<<mid<<high<<"\n";
    	long diff;
    	diff=high-mid-1;
    	diff=max(diff,mid-low-1);
    	if(diff<1)
    	 diff=0;
        cout<<diff<<endl;

    }
    return 0;
}
