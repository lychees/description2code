#include <iostream>
#include <map>
using namespace std;
typedef long long int ll;
ll calc(ll a, ll b, ll mod) {
	ll ans = 1;
	while(b > 0) {
		if(b % 2 == 1) {
			ans = (ans * a) % mod;
		}
		b/=2;
		a = (a * a) % mod;
	}
	return ans;
}
int main()
{
    ll n;
    cin >> n;
    ll a[n + 1];
    ll b[n + 1];
    ll i;
    ll p = 1;
    ll mod = 1000000007;
    map<ll, ll> m;
    map<ll, ll> :: iterator it;
    for(i = 1; i <= n; i++) {
    	cin >> a[i];
    	if(a[i] != 0) {
    		p = (p * a[i])%mod;
    	}
    	else {
    		m[i]++;
    	}
    }
    ll q;
    ll id, v;
    cin >> q;
    while(q--) {
    	int ch;
    	cin >> ch;
    	if(ch == 0) {
    		cin >> id >> v;
    		if(a[id] == 0) {
    			if(v != 0) {
    				m.erase(id);
    				a[id] = v;
    				p = (p * v)%mod;
    			}
    		}
    		else {
    			p = p * calc(a[id], mod - 2, mod);
    			p = p%mod;
    			if(v == 0) {
    				m[id]++;
    				a[id] = v;
    			}
    			else {
    				p = p * v;
    				p = p%mod;
    				a[id] = v;
    			}
    		}
    	}
    	else {
    		ll ans, k;
    		cin >> id;
    		k = m.size();
    		if(k >= 2) {
    			ans = 0;
    		}
    		else if(k == 1) {
    			it = m.begin();
    			if(it->first == id) {
    				ans = p;
    			}
    			else {
    				ans = 0;
    			}
    		}
    		else {
    			ans = p * calc(a[id], mod - 2, mod);
    			ans = ans%mod;
    		}
    		cout << ans << endl;
    	}
    }
    return 0;
}
