#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

#define pii pair<int,int>
#define pll pair<ll,ll>
#define PB push_back
#define MP make_pair

ll gcd(ll a, ll b){
	if(a<b)
		swap(a,b);

	if(b == 0)
		return a;

	ll g = a%b;

	while(g != 0){
		a = b;
		b = g;
		g = a%b;
	}

	return b;
}

int c[50];

int main(){
	ll a,b,i,j,k,l,x,y,ans=0,g;

	c[2] = c[1] = 2;

	cin >> a >> b;

	if(b>2)
		if(a<3){
			ans = 2 - a + 1;
			a = 3;
		}
		else;
	else{
		ans = b - a + 1;
		a = b + 1;
	}

	x = b - a + 1;

	for(i=2,l=1;x>0;i++){
		if(c[i]==0){
			for(j=1;i%j==0;j++);

			c[i] = 1 + c[j];
		}

		g = gcd(i,l);

		if(i == g)
			continue;

		l = (i/g)*l;

		y = b/l - (a-1)/l;

		ans += (c[i]*(x-y));

		// cout << i << ' ' << l << ' ' << c[i] << ' ' << x << ' ' << y << ' ' << ans << '\n';

		x = y;
	}

	cout << ans << '\n';

	return 0;
}
