#include<bits/stdc++.h>
#define ll long long int
#define s(x) scanf("%lld",&x)
#define p(x) printf("%lld\n",x)
using namespace std;
struct node
{
	ll x;
	ll y;
	ll d;
};
int main()
{	
	ll t;
	s(t);
	while(t--)
	{
		ll x,i,n,k,j,z=-1;
		s(n);
		s(k);
		char a[105][105];
		for(i=0;i<n;i++)
		scanf("%s",a[i]);
		node temp,temp1,temp2;
		temp.x=0;
		temp.y=0;
		temp.d=0;
		queue<node> q;
		if(a[0][0]=='1')
		q.push(temp);
		while(!q.empty())
		{
			temp1=q.front();
			q.pop();
			if(temp1.x==k-1 && temp1.y==k-1)
			{
				z=temp1.d;
				break;
			}
			if(temp1.x>=n || temp1.y>=n)
			continue;
			temp2.x=temp1.x+1;
			temp2.y=temp1.y;
			temp2.d=temp1.d+1;
			if(temp2.x<n && temp2.y<n && a[temp2.x][temp2.y]=='1')
			q.push(temp2);
			temp2.x=temp1.x;
			temp2.y=temp1.y+1;
			temp2.d=temp1.d+1;
			if(temp2.x<n && temp2.y<n && a[temp2.x][temp2.y]=='1')
			q.push(temp2);
		}
		p(z);
	}
	return 0;
}
