#include <bits/stdc++.h>
using namespace std;
int a[10000005];
int main()
{
   int  i, j, m, t, mm = 0;
   long long int k, n;
  scanf("%lld %lld", &n, &k);
 
   for(i = 0 ; i < n; i++)
   cin>>a[i];
   if(k >= n) {
   	sort(a, a+n);
   	for(i = n; i >= 1 ; i--)
   	cout<<i<<" ";
   	cout<<"\n";
   }
   else {
   for(i = 0 ; i < n-1; i++) {
   	m = i;
   	for(j = i+1 ; j < n; j++)
   		if(a[j] >= a[m])
   			m = j;
   			if( m != i) {
   			t = a[i];
   			a[i] = a[m];
   			a[m] = t;
   			mm++;
   			if(mm == k)
   			break;
   		}
   }
   for(i = 0 ; i < n; i++)
  	printf("%lld ", a[i]);
   	printf("\n");
   }
    return 0;
}
