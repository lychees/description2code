#include <iostream>
using namespace std;

int main()
{
    
    int t,z,n,s;
    cin>>t;
    for(z=0;z<t;z++)
    {
       	cin>>n;
       	int b[n],c[n],w,i;
       	char a[n][10];
       	for(int j=0;j<n;j++)
       	{
       		cin>>b[j]>>a[j];
       		c[j]=b[j];
       	}
       	int tem;
       	char te[10];
       	for(int j=0;j<n;j++)
       	{
       	for(int k=j+1;k<n;k++)
       	{
       		if(b[j]>b[k])
       		{
       			
       			tem=b[j];
       			b[j]=b[k];
       			b[k]=tem;
       		}
       	}
       	}
       	cin>>s;
       	w=b[s-1];
       	for(i=0;i<n;i++)
        {
            if(c[i]==w)
            {
                cout<<a[i]<<endl;
                break;
            }
        }
       	
    }
    return 0;
}
