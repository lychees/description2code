#include <bits/stdc++.h>
using namespace std;

long long dp[66][33],dp1[100];
long long nCr(int n,int r){
	if(n==r) return dp[n][r]=1;
	if(r==0) return dp[n][r]=1;
	if(r==1) return dp[n][r]=(long long)n;
	if(dp[n][r]) return dp[n][r];
	return dp[n][r]=nCr(n-1,r)+nCr(n-1,r-1);
}
long long fact(int n){
	if(n==0) return dp1[n]=1;
	if(n==1) return dp1[n]=1;
	else return dp1[n]=n*fact(n-1);
}

int main()
{
	int t; cin>>t;
	while(t--){
		int c,v; cin>>c>>v;
		long long ways=nCr(7,c)*nCr(5,v)*fact(c+v);
		//if()
		cout<<ways<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
