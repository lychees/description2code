#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	int i,n;
	char str[10010];
	scanf("%d%s",&n,str);
	int h[26];
	memset(h,0,sizeof(h));
	for(i=0;str[i]!='\0';i++)
	{
		h[str[i]-'a']=1;
	}
	int flag=1;
	if(!(h['a'-'a']==1 && h['e'-'a']==1 && h['i'-'a']==1 && h['o'-'a']==1 && h['u'-'a']==1))
	flag=0;
	if(flag)
	{
		cout<<"YES\n";
	}
	else
	{
		cout<<"NO\n";
	}
	return 0;
}
