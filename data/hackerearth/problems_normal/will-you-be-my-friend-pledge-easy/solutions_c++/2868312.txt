#include <iostream>
using namespace std;
int ma(int q,int w)
{
    return q>w?q:w;
}
int mi(int q,int w)
{
    return q<w?q:w;
}
int gcd(int q,int w)
{
    if(w==0)
        return q;
    else
        return gcd(w,q%w);
}

int main()
{
   int a;
 cin>>a;
 int n;
 cin>>n;
 int arr[n];
 for(int i=0;i<n;i++)
    cin>>arr[i];
    int x,y;
    int c=0;
    int g;
 for(int i=0;i<n;i++)
 {
     x=ma(a,arr[i]);
     y=mi(a,arr[i]);
     g=gcd(x,y);
     if(g!=1)
        c++;
 }
 cout<<c<<endl;
}
