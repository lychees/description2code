#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
 //   cout << "Hello World!" << endl;
 int n,a[1000],b[1000],i,j=0,p=0,s=0,q=0;
 cin>>n;
 while(n--)
    {  
    	cin>>i;
    	if(i%2==0)
       {
    	a[j]=i;
    	j++;
         }
        else
        {
    	b[p]=i;
    	p++;
         }
  
    }
    sort(a,a+j);
    sort(b,b+p);
    for(i=0;i<j;i++)
    {
    	cout<<a[i]<<" ";
    	s=s+a[i];
    }
    cout<<s<<" ";
    for(i=0;i<p;i++)
    {
    	cout<<b[i]<<" ";
    	q=q+b[i];
    }
    cout<<q<" ";
    
    return 0;
}
