using namespace std;

typedef long long ll;
typedef pair<int,int> ii;
typedef vector<int> vi;

typedef long double ld;
const int MOD = 1e9 + 7;
#define fi first
#define se second
#define pb push_back
#define mp make_pair

ll a[200001];

ll power(ll a, ll b)
{
	ll r = 1;
	while(b)
	{
		if(b%2) r = (r *a)%MOD;
		a = (a*a)%MOD;
		b /= 2;
	}
	return r;
}

int main()
{
	ios_base::sync_with_stdio(false); cin.tie(0);
	int n; cin >> n;
	ll ans = 0;
	if(n == 1)
	{
		int x;
		cin >> x;
		cout << x;
		return 0;
	}
	ll forever = power(2, n - 2);
	for(int i = 0; i < n; i++) 
	{
		cin >> a[i];
		ans = (ans + ((((i+2)*forever)%MOD)*a[i])%MOD)%MOD;
	}
	cout << ans;
	return 0;
}
