#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int a[1000005],rmx[1000005];
int main()
{
    int i,n,s,q,fm,m;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>s>>q;
    	a[s]=q;
    }
    int mx=0;
    for(i=1000000;i>0;i--)
    {
    	if(mx<a[i])
    	{
    		mx=a[i];
    	}
    	rmx[i]=mx;
    }
    cin>>fm;
    long long int ans=0;
    while(fm--)
    {
    	cin>>m;
    	ans+=100*(rmx[m]/m);
    }
    cout<<ans;
    return 0;
}
