    #include<bits/stdc++.h>
    using namespace std;
    int main(){
    	int m;
    	cin >> m;
    	int z = 0;
    	while(m--){
    		z++;
    		string s1,s2;
    		cin >>s1 >>s2;
    		int c = 0;
    		int c1 = 0,c2 = 0,s=0;
    		for(int i = 0 ; i < s2.length(); i++){
    			if(s1[i] == '?'){
    				c++;
    				if(s2[i] == '1')
    				s++;
    			}else if(s1[i] != s2[i]){
    				if(s1[i] != '1')
    					c2++;
    				else
    					c1++;
    			}
    		}
    		if(c1 - c2 <= 0)
    			cout <<"Case " << z << ": " <<  c+c2 << endl;
    		else{
    			if(s>=c1-c2)
    			cout <<"Case " << z << ": " <<  c+c1 << endl;
    			else
    			cout <<"Case " << z << ": " <<  -1 << endl;
    		}
    	}
    	return 0;
    }
