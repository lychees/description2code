#include <iostream>
using namespace std;
#include<cstdio>
int count( int S[], int m, int n )	{
    int i, j, x, y;

    int table[n+1][m];

    for (i=0; i<m; i++)
        table[0][i] = 1;
 
    for (i = 1; i < n+1; i++)
    {
        for (j = 0; j < m; j++)
        {
            x = (i-S[j] >= 0)? table[i - S[j]][j]: 0;
            y = (j >= 1)? table[i][j-1]: 0;
            table[i][j] = x + y;
        }
    }
    return table[n][m-1];
}
int main()
{
	int coin,a[3];
    scanf("%d",&coin);
    scanf("%d%d%d",&a[0],&a[1],&a[2]);
    printf("%d\n", count(a, 3, coin));
    return 0;
}
