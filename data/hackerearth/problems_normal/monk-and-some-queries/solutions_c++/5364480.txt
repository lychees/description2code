#include<bits/stdc++.h>
using namespace std;
int n,a,x,cnt[100005];
priority_queue<int>q;
priority_queue<int>lq;
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a;
		if(a==1)
		{
			cin>>x;
			cnt[x]++;
			q.push(x);
			lq.push(-x);
		}
		else
		if(a==2)
		{
			cin>>x;
			if(cnt[x]>0)cnt[x]--;
			else cout<<"-1"<<endl;
		}
		else
		if(a==4)
		{
			if(cnt[-lq.top()]!=0)
				cout<<-lq.top()<<endl;
			else
			{
				while(1!=0)
				{
					lq.pop();
					if(cnt[-lq.top()]!=0)
					{
						cout<<-lq.top()<<endl;
						break;
					}
					if(lq.size()==0)
					{
						cout<<-1<<endl;
						break;
					}
				}
			}}
		else
		if(a==3)
		{
			if(cnt[q.top()]!=0)
				cout<<q.top()<<endl;
			else
			{
				while(1!=0)
				{
					q.pop();
					if(cnt[q.top()]!=0)
					{
						cout<<q.top()<<endl;
						break;
					}
					if(q.size()==0)
					{
						cout<<-1<<endl;
						break;
					}
				}
			}
		}
	}
	return 0;
}
