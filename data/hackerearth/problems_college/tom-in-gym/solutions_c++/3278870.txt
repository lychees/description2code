#include<iostream>
using namespace std ;
int main()
{
int t,n,i,j,k,fo,temp;
int f[100000],m;
cin>>t;
for(i=0;i<t;i++)
{
	//cout<<endl;
cin>>n;
	f[0]=1;
	m=1;
        for(j=1,fo=1;fo<=n;j+=2,fo++)
	{	temp=0;
		for(k=0;k<m;k++)
		{
			temp+=f[k]*j;
			f[k]=temp%10;
			temp=temp/10;
		}
		while(temp>0)
		{
			f[k]=temp%10;
			temp=temp/10;
			k++;m++;
		}		
	}
	for(j=m-1;j>=0;j--)
	cout<<f[j];
	cout<<endl;
         
}
return 0;
}
