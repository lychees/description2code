#include <iostream>
#include <math.h>
using namespace std;
long int mirror(long int n,int e)
{
	long int a=0,b;
	for(int i=e-1;i>=0;i--)
	{
		b=n/pow(10,i);
		a=(a*10)+9-(b%10);
	}
	return a;
}
int main()
{
    int t,e1,e2;
    cin>>t;
    long int l,r,x,a,b,y;
    while(t--)
    {
    	cin>>l>>r;
    	x=r;y=l;e1=0;e2=0;
    	while(x>0){x=x/10;e2++;	}
    	while(y>0){y=y/10;e1++;	}
    	b=pow(10,e2)/2;
    	a=b-1;
    	if((a>=l&&a<=r)||(b>=l&&b<=r))
    	cout<<a*b<<endl;
    	else if(e1==e2)
    	{
    		if(r<a)
    		{
    			cout<<r*mirror(r,e2)<<endl;
    		}
    		else if(l>b)
    		{
    			cout<<l*mirror(l,e1)<<endl;
    		}
    	}
    	else
    	{
    		cout<<r*mirror(r,e2)<<endl;
    	}
    }
    return 0;
}
