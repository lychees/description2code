#include <iostream>
#include<algorithm>
#include<cmath>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n,m;long long int x=0;
    	cin>>n;
    	 int a[n];
    	 for(int i=0;i<n;i++)
    	 {cin>>a[i];}
    	sort(a,a+n);
    	
    	if(n%2==0)
    	{
    		m=(a[(n/2)-1]+a[n/2])/2;
    	}
    	else
    		m=a[(n-1)/2];
    		//cout<<m<<endl;
    	for(int i=0;i<n;i++)
    		{
    		    x=x+abs(m-a[i]);	
    		}
    	cout<<x<<endl;
    }
    return 0;
}
