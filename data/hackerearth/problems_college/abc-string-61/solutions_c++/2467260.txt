#include<bits/stdc++.h>
using namespace std;
long long func(long long a,long long b)
{
	long long res=1;
	a%=1000000007;
	while(b>0)
	{
		if(b&1) res=(res*a)%1000000007;
		a=(a*a)%1000000007;
		b>>=1;
	}
	return res;
	}

int main()
{
 int test,n;
 scanf("%d",&test);
 while(test--)
 {
 	scanf("%d",&n);
 	long long a1=func(3,n-1);
 	long long a2=func(2,n);
 	a1=a1-a2+1;
 	if(a1<0)
 	a1+=1000000007;
 	a1=a1*3;
 	a1=a1%1000000007;
 	printf("%lld\n",a1);
 }
}
