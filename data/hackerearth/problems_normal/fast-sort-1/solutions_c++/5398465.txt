#include<bits/stdc++.h>
using namespace std;
int main ()
{
	int T,a[30],i;
	string str1,str2;
	cin>>T;
	while(T--)
	{
	str1.erase();
	str2.erase();
	memset(a,0,sizeof(a));
	cin>>str1>>str2;
	for(i=0;i<str2.length();++i)
	{
		a[str2[i]-97]++;
	}
	for(i=0;i<str1.length();++i)
	{
		while(a[str1[i]-97]>0)
			{
			cout<<str1[i];	
			a[str1[i]-97]--;
			}	
	}
	for(i=0;i<str2.length();++i)
	{
		while(a[str2[i]-97]>0)
			{
			cout<<str2[i];	
			a[str2[i]-97]--;
			}	
	}
	
	cout<<endl;
	}
	return 0;
}
