import operator

a = 'head rush'
b = 'cubing club'
inp = raw_input()
count = {}

for char in inp:
		if char in count:
			continue
		else:
			count[char] = inp.count(char)

max_count = max(count.iteritems(), key = operator.itemgetter(1))[1]

max_items = []

for key in count:
	if count[key] == max_count:
		max_items.append(key)
		
max_items.sort()
max_char = max_items[0]

if a.find(max_char) != -1:
	if b.find(max_char) != -1:
		print 'Thanks both'
	else:
		print 'Thanks head rush'
elif b.find(max_char) != -1:
	print 'Thanks cubing club'
else:
	print 'none'
