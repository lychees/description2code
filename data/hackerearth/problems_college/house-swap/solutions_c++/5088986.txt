#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
int a[1000000];
int main()
{
	int  n, i , j, k=0,t=0;
	scanf("%d", &n);
	for(i=0; i<n; i++)
	scanf("%d", &a[i]);
	
	for(j=1; j<n; j++)
	{
		if(a[j-1] == j-1)
		{
			k++;
			continue;
		}
		if(a[j] == j)
		{
			k++;
			continue;
		}
		if(a[j+1] == j+1)
		{
			k++;
			continue;
		}
		if(a[j-1] == j && a[j] == j-1)
		{
			swap(a[j-1],a[j]);
			t++;
			k = k+2;
			j++;
			continue;
		}
		if(a[j] == j+1 && a[j+1] == j)
		{
			swap(a[j],a[j+1]);
			t++;
			k = k+2;
			j++;
			continue;
		}
		else
		t++;
	}
	if(t ==0)
	printf("%d\n", k-1);
	else
	printf("%d\n", k);
}
