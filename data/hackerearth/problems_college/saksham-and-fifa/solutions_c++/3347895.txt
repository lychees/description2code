#include<stdio.h>
#include<algorithm>
#include<limits.h>
using namespace std;
int main(void)
{
	int n,t=INT_MIN,i=0,max=INT_MIN;
	long long p=INT_MIN,f;
	scanf("%d",&n);
	while(n--)
	{
		scanf("%d %lld",&t,&p);
		if(t>max)
		{	max=t;
			f=p;
		}
		//i++;
	}
	//f=p[index];
	if(f<0)
		printf("Right Corner\n");
	else if(f>0)
		printf("Left Corner\n");
	else if(f==0)
		printf("Either Corner\n");
	return 0;
}
