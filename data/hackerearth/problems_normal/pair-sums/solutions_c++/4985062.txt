#include <iostream>
#include <algorithm>
using namespace std;

bool binary_search(int *arr,int beg,int end,int key){
	if(beg<=end){
		int mid=beg+(end-beg)/2;
		if(arr[mid]==key){return 1;}
		if(arr[mid]<key){
			return binary_search(arr,mid+1,end,key);
			
		}
		else return binary_search(arr,beg,mid-1,key);
	
	}
	return 0;
}
int main()
{
    int n,sum;
    cin>>n;
    cin>>sum;
    int *arr=new int[n];
    for(int i=0;i<n;i++){
    	cin>>arr[i];
    }
    sort(arr,arr+n);
    
    
    int i;
    for(i=0;i<n;i++){
    	int find=sum-arr[i];
    	bool f=binary_search(arr,i+1,n,find);
    	if(f==1){
    		cout<<"YES"<<endl;
    		break;
    	}
    }
    if(i==n){cout<<"NO"<<endl;}
    
    return 0;
}
