#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
 long long  int d,g,m;
   cin>>d;
   int count;
   long long int served;
   long long int G[3000];
   for(long long int ii=0;ii<d;ii++){
   		served=0; count=0;
   		cin>>g>>m;
   		for(long long int kk=0;kk<g;kk++ ) cin>>G[kk];
   		sort(G,G+g);
   		for(int jj=0;jj<g;jj++){
   			served+=G[jj];
   			if(served>m) break;
   			count++;
   		}
   		cout<<count<<endl;
   	
   }
}
