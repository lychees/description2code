#include<iostream>
using namespace std;

int reverse(int n){
    int rev = 0;

    while(n>0){
        rev = rev*10 + n%10;
        n = n/10;
    }
    return rev;
}

int main(){
    int t;
    int n,r;
    cin>>t;
    while(t--){
        cin>>n;
        r = reverse(n);
        if(n!=r && (n*n==reverse(r*r)))cout<<r;
        else cout<<"No";
        cout<<endl;
    }
    return 0;
}
