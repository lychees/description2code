int a[105];
int dp[10005];
vector<int>inc;
void go(int val){
	for(int i=0;i<inc.size();i++)
	dp[inc[i]+val]=1;
	inc.clear();
}
int main(){
	int t,n,i,j;
	cin>>t;
	while(t--){
		int su=0;
		cin>>n;
		for(i=1;i<=n;i++){
			cin>>a[i];
			su+=a[i];
		}
		if(su%2!=0){
			cout<<"NO"<<endl;
			continue;
		}
		else
		{
			for(i=1;i<=su/2;i++)
			dp[i]=0;
			dp[0]=1;
			for(i=1;i<=n;i++){
				for(j=0;j<=su/2;j++){
					if(dp[j])
					inc.pb(j);
				}
				go(a[i]);
			}
			if(dp[su/2])
			cout<<"YES"<<endl;
			else
			cout<<"NO"<<endl;
		}
	}

}
