#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n,count=0;
    cin>>n;
    unsigned long a[n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    
    sort(a,a+n);
    
    for(int i=0;i<(n-1);i++)
    {
    	while(a[i]<a[n-1])
    	{
   	 		a[i]*=2;
    	}
    }
    
    for(int i=0;i<(n-1);i++)
    {
    	if(a[i]!=a[i+1])
    	{
    		count=1;
    		break;
    	}
    }
    
    if(count==0)
    cout<<"YES";
    else
    cout<<"NO";
    
    return 0;
}
