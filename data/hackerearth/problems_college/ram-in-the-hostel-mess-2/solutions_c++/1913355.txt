#include <bits/stdc++.h>
using namespace std;
# define ll long long
ll counts(ll n)
{
    ll a[n], b[n];
    a[0] = b[0] = 1;
    for (int i = 1; i < n; i++)
    {
        a[i] = a[i-1] + b[i-1];
        b[i] = a[i-1];
    }
    return a[n-1] + b[n-1];
}
int main()
{
    ll t,n,ans;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	ans=counts(n);
    	ans=pow(ans,2);
    	cout<<ans<<endl;
    }
    return 0;
}
