#include <iostream>
using namespace std;

int main()
{
	int N;
	cin >> N;
	int numr = N;
	bool* p = new bool[N];
	for(int i = 0; i < N; ++i)
	{
		p[i] = true;
	}
	string str;
	cin >> str;
	int size = str.size();
	int k = 0;
	int cindex = 0;
	while(numr > 1)
	{
		int keffect = k%size;
		if (str[keffect] == 'b')
		{
			p[cindex] = false;
			--numr;
		}
		cindex = (cindex + 1)%N;
		while(p[cindex] == false)
		{
			cindex = (cindex + 1)%N;
		}
		++k;
	}
	cout << (cindex + 1)<< endl;
	
    //cout << "Hello World!" << endl;
    return 0;
}
