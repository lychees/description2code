#include <bits/stdc++.h>

using namespace std;

int boy[10001];
int girl[10001];

bool large(const int lhs,const int rhs)
{
	return lhs>rhs;
}
int main()
{
	int t,n,i,count;

	scanf("%d",&t);

	while(t--){
		scanf("%d",&n);
		for(i=0;i<n;i++)
			scanf("%d",&girl[i]);
		for(i=0;i<n;i++)
			scanf("%d",&boy[i]);
		count = 0;
		sort(girl,girl+n);
		sort(boy,boy+n,large);
		
		for(i=0;i<n;i++){
			if((boy[i]%girl[i]==0)||(girl[i]%boy[i]==0))
				count++;
		}
		
		cout<<count<<endl;
	}
	return 0;
}
