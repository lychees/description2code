#include <bits/stdc++.h>
using namespace std;
long long sum[1005];
long long a[1000000];
void preprocessor()
{
	long long i,j,k,l,x,y,r;
	a[0]=1;
	sum[0]=1;
	l=1;
	for(i=1;i<1005;i++)
	{
		r=0;
		for(j=0;j<l;j++)
		{
			x=a[j]*i+r;
			a[j]=x%10;
			r=x/10;
		}
		while(r)
		{
			a[l++]=r%10;
			r=r/10;
		}
		k=0;
		for(y=0;y<l;y++)
		{	
			k=k+a[y];
		}
		sum[i]=k;
		//cout<<sum[i]<<endl;
	}
}

int main()
{
	long long t,n;
	preprocessor();
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<sum[n]<<endl;
	}
    return 0;
}
