map < int, vector < pi > > m;
int ISP(int i){
	vector< pi > v(m[i]);
	sort(all(v));
	int n = v.size(),cnt=1,cur = v[0].y;
	for(int i=1;i<n;i++) {
			if(v[i].x > cur){
				cur = v[i].y;
				cnt++;
			}
	}
	return cnt;
}
