#include <iostream>
#include <stdio.h>
#include <iomanip> 
using namespace std;

int main()
{
	long n,i;
	long speed[10000],cons[10000],requi;
	long ans = -1, temp;
	cin>>n;
	for(i=0;i<n;i++){
		cin>>speed[i];
	}
	for(i=0;i<n;i++){
		cin>>cons[i];
	}
	cin>>requi;
	requi = requi;
	for(i=0;i<n;i++){
		temp = (requi*speed[i]*1000)/(cons[i]);
		//cout<<temp<<"\t";
		if(temp>ans){
			ans = temp;
		}
	}
	printf("%.3f\n",(float)ans/1000);
    return 0;
}
