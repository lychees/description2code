#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t; while(t--)
    {
       map<string,int>m;
       map<string,int>::iterator it;
       string s;
       int r,c,no=0;
       cin>>r>>c;
       for(int i=0;i<r;i++)
       {
           cin>>s;
           m[s]++;
       }
       for(it=m.begin();it!=m.end();it++)
       {
           if((it->second)>1)
           {
               no=1; break;
           }
       }
       if(no) cout<<"No\n";
       else cout<<"Yes\n";
    }

    return 0;
}
