#include <iostream>
#include <cstdio>

using namespace std;


void FindSymmetricCut(int n, int p) {
	if (3 == n) {
		printf("%d\n",3);
		return;
	}
	
	else {
		while (p != 0) {
			
			if (n == 3) {
				break; 
			}
			
			else {
				
				if (n % 2 == 0) 
					n = (n/2) + 1;
			
				else {
					int t;
					//Solve the AP: a+(t-1)d = n. Find t. Use t to solve the symmetric cut AP. 
					// For 3, Cut = 3 ; For 5, Cut = 4 ; For 6, Cut = 5 ......... So a = 3; d = 2;
					t = ((n - 3) / 2) + 1;
					
					// Now the cut vertex is: a+ (t-1)*d where d = 1
					n = 3 + (t - 1);
				}
			
			}
			--p;
		}
			
			
		printf("%d\n",n);
	}
}




int main()
{
    int t;
    scanf("%d", &t);
    for (int i = 0; i < t; i++) {
    	int n, p;
    	scanf("%d %d", &n, &p);
		FindSymmetricCut(n, p);
    }
    return 0;
}
