#include<iostream>
#include<algorithm>
using namespace std;
int fn(int a,int b)
{
	return a>b?b:a;
}
int main()
{
	int n,a[10001],ans=999999,i,j;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	sort(a,a+n);
	  for(i=0;i<n-1;i++)
	  {
  		for(j=n-1;j>i;j--)
  		{
		     if(a[j]-3*a[i]<=0)
			 break; 	  
		  }
		  if(j>i)
		  {
  			ans=fn(ans,n-(j-i+1));
  		}
  	}
  	cout<<ans<<endl;
	return 0;
}
