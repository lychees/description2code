#include <stdio.h>
#include <iostream>
using namespace std;
const int MAX = 100000+7;
long long a[MAX];
long long count[MAX];
long long hit[20000];


long long find_less(long long num,long long hit[],int M){
	long long count = 0;
	for(int i=0;i<M;i++)
	{
		if (num > hit[i]){
			count++;
			num--;
		}
	}
	return count;

}

int main()
{
	long long N;
	cin >> N;
	for(int i=0;i<N;i++){
		cin >> a[i];
	}
	long long M;
	cin >> M;
	for(int i=0;i<M;i++)
		cin >> hit[i];
	for(int i=0;i<N;i++)
		count[i] = find_less(a[i],hit,M);

	for(int i=0;i<N;i++)
		cout<<a[i]-count[i]<<" ";
}
