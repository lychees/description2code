using namespace std;
int a[1000001];
int sum[1000002];
int main()
{
    int t;
    cin>>t;
    for(int i=2;i<1000001;i++)
    {
        if(a[i]==0)
        {
            for(int j=2;i*j<1000001;j++)
            {
                if(a[i*j]==0) a[i*j]=i;
            }
        }
    }
    for(int i=2;i<1000002;i++) sum[i]=sum[i-1]+a[i];
  //  for(int i=2;i<10;i++) cout<<a[i]<<" ";
    while(t--)
    {
        int l,r;
        cin>>l>>r;
        cout<<sum[r]-sum[l-1]<<endl;
    }
}
