#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int t,n,k,i,mn,mx;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	mn=min(k,n-k);
    	mx=max(k,n-k);
		for(i=1;i<=mn-1;i++)
		{
			cout<<"9";
		}
		cout<<"8";
		for(i=mn+1;i<=mx;i++)
		{
			cout<<"9";
		}
		for(i=mx+1;i<=n-1;i++)
		{
			cout<<"0";
		}
    	cout<<"1"<<endl;
    }
    return 0;
}
