#include <iostream>
using namespace std;


int a[4] ;
int s[4] ;

void change(){
	
  int t1 = min( s[2] , s[3] ) + a[1] ;
  int t2 = min( s[1] , s[3] ) + a[2] ;
  int t3 = min( s[2] , s[1] ) + a[3] ;
   
   s[1] = t1 ;
   s[2] = t2 ;
   s[3] = t3 ;
	
}

int min(){
	int temp = 999999;
	for( int k = 1  ; k <= 3 ; k ++ ){
	
	    if( s[k] < temp  )
	    
	        temp = s[k];
	
	}
	return temp ;
}

int main()
{
	int T ;
	cin>>T ;
	while( T-- ){
	
		s[1] = 0 ; s[2] = 0 ; s[3] = 0 ;
		int N ;
		
		cin>>N;
		for( int i = 0 ; i < N ; i ++ ){
			cin>>a[1]>>a[2]>>a[3] ;
			change() ;
			
		}
		cout<<min()<<endl ;
	}
   
    return 0;
}
