ll gc(ll a,ll b)
{
    if(b==0)
       return a;
    else
       return gc(b,a%b);
}

int isp[N+2];

void seive()
{
    int i,j;
    isp[0]=isp[1]=1;
    for(i=2;i<=N;++i)
    {
        if(!isp[i])
        {
            for(j=2;i*j<=N;++j)        isp[i*j]=i;
        }
    }
}

int main()
{
    ll t,n,i,j;
    i(t);
    seive();
    while(t--)
    {
        int idx=0;
        i(n);
        if(!isp[n])
        {
            idx=n-1;
        }
        else
        {
            f(i,0,n-1)
            {
                if(gc(i,n)==1)  idx+=1;
            }
        }
        if(!isp[idx])
            o("Correct\n");
        else
            o("Incorrect\n");
    }
    return 0;
}
