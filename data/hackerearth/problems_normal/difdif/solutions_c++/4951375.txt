#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int lcp(string a,string b){
	int i=0;
	while(a[i]==b[i]){
		i++;
	}
	return i;
}
int main()
{
    string str;
    cin>>str;
    int len=str.length();
    string str2[len];
    for(int i=0;i<len;i++){
    	str2[i]=str.substr(i);
    	
    }
    sort(str2,str2+len);
    int ans=str2[0].length();
    for(int i=1;i<len;i++){
    	ans+=str2[i].length()-lcp(str2[i],str2[i-1]);
    }
    cout<<ans;
    
    return 0;
}
