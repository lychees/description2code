int main(void)
{
       int t,n,i,j;
       long dp[201][201];
       for(i=0;i<201;i++)
        {
            dp[i][0]=1;
            dp[i][1]=0;
            dp[1][i]=0;
        }

        for(i=2;i<201;i++)
        {
            for(j=2;j<201;j++)
            {
                if(j<i)
                    dp[i][j]=dp[i-1][j];
                else
                    dp[i][j]=dp[i-1][j]+dp[i][j-i];
            }
        }


    cin >> t;

    while(t--)
    {
        scan(n);
        cout << dp[n][n] << endl;
    }

    return 0;
}
