#include <bits/stdc++.h>
#define MAX 1000006
using namespace std;
double prob,nprob,cache[MAX];
int main(int argc, char const *argv[]) {
    int n,p,i;
    scanf("%d%d",&n,&p);
    for(i=0;i<n;i++)
        cache[i]=0;
    prob=(double)p/100;
    nprob=1-prob;
    cache[0]=1;
    cache[1]=0;
    cache[2]=prob;
    cache[3]=nprob;
    for(i=4;i<=n;i++)
    {
        cache[i]=prob*cache[i-2]+nprob*cache[i-3];
    }
    printf("%0.6f\n",cache[n]);
    return 0;
}
