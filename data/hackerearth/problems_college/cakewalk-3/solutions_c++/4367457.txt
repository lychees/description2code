#include <iostream>
#include <string>

using namespace std;

int main()
{
    int n, letras;
    string palavra, sufix;
    ios_base::sync_with_stdio(false);
    cin>>n>>palavra;

    while(--n){
        cin>>letras>>sufix;
        palavra.erase(letras, -1);
        palavra.append(sufix);
    }
    cout<<palavra<<'\n';
}
