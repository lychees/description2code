#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10;

set < pair <int, int> > s;

int main() {
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		int x, y, d;
		scanf("%d %d %d", &x, &y, &d);
		auto line = make_pair(x, x + d);
		if (s.empty()) {
			s.insert(line);
		} else {
			auto it = upper_bound(s.begin(), s.end(), line);
			auto cur = it, t = it;
			--t;
			if (it == s.begin() || line.first > t->second) {
				t = s.insert(line).first;
				cur = t;
				++cur;
			} else {
				auto newline = make_pair(t->first, max(t->second, line.second));
				s.erase(t);
				t = s.insert(newline).first;
				cur = t;
				++cur;
			}
			while (cur != s.end()) {
				auto line = *cur;
				if (line.first > t->second) {
					break;
				}
				s.erase(cur);
				auto newline = make_pair(t->first, max(t->second, line.second));
				s.erase(t);
				t = s.insert(newline).first;
				cur = t;
				++cur;
			}
		}
	}
	int res = 0;
	for (auto it = s.begin(); it != s.end(); ++it) {
		res += it->second - it->first + 1;
	}
	printf("%d\n", res);
	return 0;
}
