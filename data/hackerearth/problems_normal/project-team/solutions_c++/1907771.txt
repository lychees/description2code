#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,a[10001];
	cin>>t;
	while(t--)
	{
	   cin>>n;
	   int b[100001];
	   for(int i=0;i<n;i++)
	   cin>>a[i];
	   sort(a,a+n);
	   int j=n-1;
	   for(int i=0;i<n/2;i++)
	   {
	     b[i]=a[i]+a[j];
	     j--;
	   }
	   int ma=0;
	   int mi=1000001;
	   for(int i=0;i<n/2;i++)
	   {
	     if(b[i]>ma)
	     ma=b[i];
	     if(b[i]<mi)
	     mi=b[i];
	   }
	   cout<<ma-mi<<endl;
	}
	return 0;
}
