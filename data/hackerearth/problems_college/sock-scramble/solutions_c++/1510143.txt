#include <iostream>
using namespace std;

    int findpair(int A[], int n) {
     int x=A[0];

     for(int i=1;i<n;i++)
        x=x^A[i];

    return x;
    }



int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int A[n];
    	for(int i=0;i<n;i++)
    		cin>>A[i];


    int p = findpair(A,n);
    cout<<p<<endl;
    }
    return 0;
}
