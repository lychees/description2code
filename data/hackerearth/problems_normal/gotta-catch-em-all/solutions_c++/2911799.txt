#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<string>
#include<vector>
#include<list>
#include<utility>
#include<set>
#include<map>
#include<stack>
#include<queue>
using namespace std;
int main(){
    int k,i,max=0;
    scanf("%d",&k);
    int arr[k];
    for(i=0;i<k;i++)
        scanf("%d",&arr[i]);
    i=sizeof(arr)/sizeof(arr[i]);
    sort(arr,arr+i);
    for(i=k-1;i>=0;i--){
        arr[i]+=(k-i);
        if(arr[i]>max)
            max=arr[i];
    }
    max++;
    printf("%d\n",max);
    return 0;
}
