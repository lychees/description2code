#include<iostream>
using namespace std;
int main()
{
    int tt;
    cin>>tt;
    cin.ignore();
    while(tt--)
    {
        string str;
        cin>>str;
        int l=str.size();
        int flag=1;
        string ans,temp,test;
        int f,s;
        for(int i=0;i<l;i++)
        {
            if(str[i]=='o')
            {
                int k1=i;
                temp=test;
                while(k1<l)
                {


                    if(str[k1]=='*'){
                        temp+='*';
                        ans+=temp;
                        temp=test;
                        flag=1;
                       // cout<<"ans=br==>"<<ans<<endl;
                        break;
                    }
                    else if(str[k1]=='x')
                    {
                        flag=0;
                        temp='x';
                        ans+=temp;
                        temp=test;

                        //cout<<"ans===>"<<ans<<endl;


                    }
                    else{
                            if(flag==1)
                            temp+=str[k1];
                        //cout<<"temp===>>>"<<temp<<endl;
                    }
                    k1++;

                }
                i=k1;
                ans+=temp;

            }
            else{
                    if(str[i]=='x')flag=0;
                    else if(str[i]=='*')flag=1;
                ans+=str[i];

                        //cout<<"ans===>"<<ans<<endl;
            }
        }
        cout<<ans<<endl;

    }

}
