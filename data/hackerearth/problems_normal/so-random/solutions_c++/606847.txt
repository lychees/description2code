#include <iostream>
#include <stdio.h>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin >> t;
    
    while(t--) {
    	int time;
    	double p;
    	
    	cin >> p >> time;
    	int c = 1440/time;
    	double neg = 1-p;
    	
    	double ans = pow(neg, 1.0/c);
    	printf("%.4f\n",1-ans);
    	//cout << 1-ans << "\n";
    }
    return 0;
}
