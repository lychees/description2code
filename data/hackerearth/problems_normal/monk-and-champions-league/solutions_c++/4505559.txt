#include<bits/stdc++.h>
using namespace std;
priority_queue<int> pq;
int main()
{
    int n,m,x;
    scanf("%d %d",&n,&m);
    for(int i=0;i<n;i++)
    {
        cin>>x;
        pq.push(x);
    }
    long long sum1=0;
    while(m--)
    {
        sum1+=pq.top();
        int y= pq.top();
        y=y-1;
        pq.pop();
        pq.push(y);
    }
    printf("%lld\n",sum1);
    return 0;
}
