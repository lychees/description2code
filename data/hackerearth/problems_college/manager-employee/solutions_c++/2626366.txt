#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long t;
    cin >> t;
    while(t--)
    {
    	long long n;
    	cin >> n;
    	long long a[n],maxa=0,mina=1000000,temp;
    	for(long long j=0 ; j<n ; j++)
    		cin >> a[j];
    	sort(a,a+n);
    	for(long long i=0 ; i<n/2 ; i++)
    	{
    		temp = a[n-i-1] + a[i];
    		mina = min(mina,temp);
    		maxa = max(maxa,temp);
    	}
    	cout << maxa-mina << endl;    	
    }
}
