#include <iostream>
#include<string.h>
using namespace std;
int T,i,j,S;
char A[10][50];

int main()
{
    do
    {
    	cin>>T;	
    }
    while(T<=0 || T>10);
    for(i=0;i<T;i++)
    {
 
    	cin>>S;
    	cin>>A[i];
    	int l=strlen(A[i]);
    	for(j=0;j<l;j++)
    	{
    		if(A[i][j]>=65 && A[i][j]<=90)
    		{
    			if((A[i][j]-S)>=65)
    			{
    				A[i][j]=A[i][j]-S;	
    			}
    			else
    			{
    				A[i][j]=A[i][j]-S+26;
    			}
    		}
    		else
    		{
    			cout<<"not in range";
    		}	
    	}
    	cout<<A[i]<<endl;
    	
    }
  
    return 0;
}
