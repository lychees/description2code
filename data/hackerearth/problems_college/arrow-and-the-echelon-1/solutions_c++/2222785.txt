#include<iostream>
using namespace std;
int main()
{
    int tt;
    cin>>tt;
    while(tt--)
    {
        int n;
        cin>>n;
        string ans;
        if(n%3==0 || n%5==0 )
        {
            if(n%3==0)
            {

                while(n--)
                ans+='5';

            }else if(n%5==0)
            {
                while(n--)
                {
                    ans+='3';
                }

            }
        }else{

                int x=0,y=0;
                for(int i=0;i<=n;i++)
                {
                    for(int j=0;j<=n;j++)
                    {
                        if(i*3+j*5==n)
                        {
                            x=i;y=j;
                            break;
                        }
                        if(i*3+j*5>n)
                            break;

                    }


                }
                //cout<<x<<y<<endl;
                if(x && y){
                while(x--)
                    ans+="555";
                while(y--)
                {
                    ans+="33333";
                }
                }else
                            ans+="-1";
        }

        cout<<ans<<endl;
    }

}
