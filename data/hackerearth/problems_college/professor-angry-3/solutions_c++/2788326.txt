#include<iostream>
int t,cs=0,tme[1000];
int main(){
	std::cin>>t;
	while(cs<t){
		int n,k,count=0;
		std::cin>>n>>k;
		for(int i=0;i<n;i++)
			std::cin>>tme[i];
		for(int j=0;j<n;j++){
			if(tme[j]<=0)
				count++;
		}
		if(count<k)
			std::cout<<"YES\n";
		else
			std::cout<<"NO\n";
			cs++;
	}
	return 0;
}
