#include<bits/stdc++.h>
using namespace std;

vector<int> comp;

void gen()
{
    int i, j;
    for(i=2; i<=10000/2; i++){
        for(j=2; i*j<=10000; j++){
            comp.push_back(i*j);
        }
    }

    sort(comp.begin(), comp.end());
}

int main()
{
    gen();

    int i, n;
    vector<int> ans;

    cin >> n;
    for(i=0; i<comp.size(); i++){
        if(comp[i]>n) break;
        if(i==0) ans.push_back(comp[i]);
        if(i>0 and comp[i]!=comp[i-1]) ans.push_back(comp[i]);
    }

    cout << ans[0];
    for(i=1; i<ans.size(); i++)
        cout << " " << ans[i];

    cout << '\n';

    return 0;
}
