#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--) {
        string s;
        int h[26] = {0};
        cin >> s;
        for(int i = 0; i < s.length(); i++) {
            if(h[s[i]-'a'] == 0)
                h[s[i]-'a']++;
            else {
                h[s[i]-'a']++;
                s[i] = 'X';
            }
        }
        for(int i = 0; i < s.length(); i++)
            if(s[i] != 'X')
                cout << s[i];
        cout << endl;
    }
    return 0;
}
