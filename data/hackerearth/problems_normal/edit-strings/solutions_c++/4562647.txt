#include <iostream>
#include<string>
using namespace std;
string s;int n;
int func(int x)
{
	int count=0;
	while(x>1)
	{
		
		count+=x/2;
		x-=x/2;
	}
	return count ;
}
int main()
{
	int q;
	cin >> n>>q>>s ;
	int i=0;
	int count=0;
	while(i<n)
	{
		if(s[i]=='.')
		{int temp=0;
			while(s[i]=='.')
		{
			i++;
			temp++;
		}
		int x=temp;
		
		count+=func(temp);
		}
		else
		if(s[i]!='.')
		i++;
		
	}
	
	
	for(i=1;i<=q;i++)
	{
		int k;
		cin >> k;
		char a;
		cin >> a;
		
		int left=0,right=0;
		int j=k-1;
		while(s[j-1]=='.'&&j>=1)
		{left++;j--;}
		j=k+1;
		while(s[j-1]=='.'&&(j-1)<n)
		{
			right++;j++;
		}
		
		int l=func(left);
		int r=func(right);
		 if(a=='.' && s[k-1]!='.' )
		{int ne=func(left+right+1);
		count=count-l-r+ne;
		
		}
		else if(a!='.'&&s[k-1]=='.')
		{
			int ne=(func(left+right+1));
			count=count-ne+l+r;
			
		}
		cout << count << endl;
		s[k-1]=a;
	}
    
    return 0;
}
