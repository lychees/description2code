#include<iostream>
#include<string.h>
using namespace std;
main()
{
	int n;
	cin>>n;
	int num=1,i;
	string a[n];
	for(i=0;i<n;i++)
	cin>>a[i];
	int flag[26]={0};
	int vis[n],k;
	
	for(i=0;i<n;i++)
	vis[i]=0;
	for(i=0;i<n;i++)
	{
		if(vis[i]==0)
		{
			vis[i]=num;
			string s=a[i];
			for(k=0;k<26;k++)
			flag[k]=0;
			for(int j=0;j<s.length();j++)
			flag[s[j]-'a']++;
			for(int j=i+1;j<n;j++)
			{
				string s1=a[j];
				for(k=0;k<s1.length();k++)
				flag[s1[k]-'a']--;
				for(k=0;k<26;k++)
				{
					if(flag[k]!=0)
					break;
				}
				if(k==26 && s[0]==s1[0] && s[s.length()-1]==s1[s1.length()-1])
				vis[j]=num;
				for(k=0;k<26;k++)
				flag[k]=0;
				for(k=0;k<s.length();k++)
				flag[s[k]-'a']++;
			}
			num++;
		}
		
	}
	int m=vis[0];
	for(i=1;i<n;i++)
	if(vis[i]>m)
	m=vis[i];
	cout<<m<<endl;
	
}
