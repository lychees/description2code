#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n;
    long a,b;
    cin>>t;
    while(t)
    {
    	cin>>a>>b>>n;
    	long cal[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>cal[i];
    		
    	}
       sort(cal,cal+n);
    	
        int num_raghu=0,num_sayan=0;
        //cout<<a<<" "<<b<<"\n";
   
        for(int i=0;i<n;i++)
        {
        	if(cal[i]<=a)
        	{
        		
        	
        		a-=cal[i];
        	
        		num_raghu++;
        	}
        	else
        	 break;
        }
          for(int i=0;i<n;i++)
        {
        	if(cal[i]<=b)
        	{
        		b-=cal[i];
        		num_sayan++;
        	}
        	else
        	break;
        }
      //  cout<<num_raghu<<" "<<num_sayan<<"\n";
        if(num_raghu==num_sayan)
        cout<<"Tie\n";
        else if(num_raghu>num_sayan)
        cout<<"Raghu Won\n";
        else cout<<"Sayan Won\n";
    	t--;
    }
    return 0;
}
