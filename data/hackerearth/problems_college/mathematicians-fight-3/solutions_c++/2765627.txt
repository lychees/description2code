#include <iostream>
using namespace std;
long long find(long long x){
	if(x==1)
	return 1;
	if(x%2==0){
		return find(x/2)*2-1;
	}
	else{
		return find(x/2)*2+1;	
	}
	
}
int main()
{
   int t;
   long long x,p;
   cin>>t;
   while(t--){
   	cin>>x;
   	p=find(x);
   	cout<<p<<endl;
   }
    return 0;
}
