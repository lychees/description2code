#include <iostream>
#define ll long long

using namespace std;

int main()
{
    string s = "";
    int letters[26] = {0};
    
    while(cin >> s){
    	for (ll i = 0; i < s.length(); i++)
    		letters[(int)s[i] - 'a']++;
    		
    	for (int i = 0; i < 26; i++){
    		for (int j = 0; j < letters[i]; j++)
    			cout << (char)(i+97);
    	}
    	cout << endl;
    	
    	for (int i = 0; i < 26; i++)
    		letters[i] = 0;
    	
    }
    
    return 0;
}
