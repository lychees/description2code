vector<int>v;
int main()
{
    int t,n,i,j,k,tmp_count,count;
    sieve();
    int num = 30;
    count = 0;
    while(count != 1000){
        tmp_count = 0;
        for (i = 1; i*i <= num; i++){
            if(num%i == 0){
                if(p[i] == 0 || p[num/i] == 0){
                    tmp_count++;
                }
            }
        }
        if(tmp_count >= 3){
            v.push_back(num);
            count++;
        }
        num++;
    }
    cin >> t;
    while(t--){
        cin >> n;
        P(v[n-1]);
    }
}
void sieve()
{
    int i,j;
    p[0] = 1;
    p[1] = 1;

    for (i = 2; i*i <= 1004; i++){
        if(p[i] == 0){
            j = 2;
            while(i*j <= 100004){
                p[i*j] = 1;
                j++;
            }
        }
    }
}
