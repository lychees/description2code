#include <bits/stdc++.h>
using namespace std;

#define MAXN 100000

vector<int> edge[MAXN];
bool visited[MAXN];
int ans[MAXN];

int search (int i) {
	
	ans[i] = 1;
	
	visited[i] = true;
	
	for (int j=0; j<edge[i].size(); j++) {
		if (!visited[edge[i][j]]) {
			ans[i] += search(edge[i][j]);
		}
	}
	
	visited[i] = false;
	
	return ans[i];
}

int main()
{
	int N;
	scanf("%d", &N);
	
	int u, v;
	for (int i=0; i<N-1; i++) {
		scanf("%d %d", &u, &v);
		edge[u-1].push_back(v-1);
		edge[v-1].push_back(u-1);
	}
	
	memset(visited, false, sizeof(visited));
	
	search(0);
	
	ans[0]--;
	
	for (int i=0; i<N; i++) {
		printf("%d ", ans[i]);
	}
	
    return 0;
}
