ll MOD = (ll)1e9 + 7;

ll modexpo(ll a,ll b)
{
	ll res = 1;
	while(b > 0)
	{
		if(b & 1)
		{
			res = (res*a)%MOD;
		}

		a = (a*a)%MOD;
		b >>= 1 ;
	}

	return res%MOD;
}

ll ncr(ll n,ll r = 6)
{
	if(r>n)
		return 0;

	if(r >= n-r)
		r = n - r;

	ll i;
	ll fact = 1;
	for(i = 1; i<=r; i++)

		fact *= (n-i+1);
		fact %= MOD;
		fact *= (modexpo(i,MOD-2));

}
