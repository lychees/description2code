t = int(raw_input())
for i in range(t):
    s = raw_input()
    res = 4
    ot = s[4:-4]
    ot = ot.replace(
        "a",
        "").replace(
        "o",
        "").replace(
            "i",
            "").replace(
                "e",
                "").replace(
                    "u",
        "")
    res += len(ot)
    print str(res)+"/"+str(len(s))
