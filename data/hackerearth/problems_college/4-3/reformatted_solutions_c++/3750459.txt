#include <iostream>
#include <stdio.h>
using namespace std;
int main() {
  int t;
  cin >> t;
  while (t > 0) {
    long int n, c, s = 0, sum = 0, h = 1;
    cin >> n;
    for (int i = 0; i < n; i++) {
      cin >> c;
      s += c;
    }
    for (int i = 1; i < n; i++)
      h = (h * 2) % 1000000007;
    sum = (s * h) % 1000000007;

    cout << sum << endl;
    --t;
  }
  return 0;
}
