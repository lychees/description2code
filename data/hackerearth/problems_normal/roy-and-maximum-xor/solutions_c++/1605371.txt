int main()
{
	int t;
	si(t);
	while(t--)
	{
		ll n;
		sl(n);
		if(n==1)
		{
			printf("0 1\n");
			continue;
		}
		n--;
		ll mx;
		int bit = 63;
		while(!((n>>bit)%2)) 
		{
			bit--;
		}	
		bit++;

		ll xr = mx^n;
		ll answer = n-xr+1;

	}
