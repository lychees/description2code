#include <iostream>
using namespace std;
void check(long long int n )
{
		if(n % 4 == 0 || n % 4 == 1)
		{
		     
		     
		     cout<< (n/4) * (n/4) << endl;
		}
		else
		{
			cout<<(n/4) * ((n/4)+1)<<endl;
		}
}
int main()
{
	long long int n ;
	int t;
	cin >> t;
	while(t--)
	{
		cin >> n;
		check(n);
	}
}
