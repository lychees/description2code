#include<bits/stdc++.h>
using namespace std;
void sol()
{
    int i,n,j,x;
    long long int c=0;
    cin>>n;
    int a[n+1];
    for(i=0;i<n;++i) cin>>a[i];
    sort(a,a+n);
    for(i=0;i<n;)
    {
       ++c;
       j=i+1;
       while(j<n&&a[j]<=a[i]+4) ++j;
       i=j;
    }
    cout<<c<<endl;
}
int main()
{
    sol();
    return 0;
}
