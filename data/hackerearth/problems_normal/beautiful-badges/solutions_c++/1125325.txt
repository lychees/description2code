using namespace std;

int N, color, d;

vector<int> b;
vector<int> y;

int main() {
    scanf("%d", &N);
    for(int i = 0; i < N; i++) {
        scanf("%d %d", &color, &d);
        if(color == 1) b.PB(d);
        else y.PB(d);
    }

    double raiz2 = sqrt(2);
    sort(b.begin(), b.end());
    sort(y.begin(), y.end());

    int i = 0, j = 0, ans = 0;
    while(i < b.size() && j < y.size()) {
        double calc1 = b[i] * 1.0 * raiz2;
        double calc2 = y[j] * 1.0 * raiz2;
        if(calc1 > y[j] && calc2 > b[i]) i++, j++, ans++;
        else if(calc1 > y[j]) j++;
        else i++;
    }

    printf("%d\n", ans);

}
