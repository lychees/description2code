#include <iostream>
#include<bits/stdc++.h>
#include<vector>
#include<map>
#include<algorithm>
#include<stdio.h>
#define all(c) (c).begin(),(c).end()
#define r_all(c) (c).rbegin(),(c).rend()
#define tr(c,i) for(decltype((c).begin()) i =(c).begin();i!=(c).end();i++)
using namespace std;
map<string,pair<int,int> > mp;
int main()
{
    int n,m,j,i;
    string str;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>str;
        mp[str].first++;
        mp[str].second=n-i;
    }
    vector<pair<pair<int,int>,string> > vec;
    tr(mp,it)
    {
        vec.push_back(make_pair(it->second,it->first));

    }
    sort(r_all(vec));
    tr(vec,it)
    {
        cout<<it->second<<" "<<it->first.first<<endl;
    }
    return 0;
}
