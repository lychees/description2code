#include <bits/stdc++.h>
using namespace std;
int n,m;
int mat[1002][1002];
int row[1002];
int col[1002];

int main()
{
    cin>>n>>m;
    for(int i=0;i<n;i++)
     {
         for(int j=0;j<m;j++)
           cin>>mat[i][j];
     }
    
    for(int i=0;i<n;i++)
     {  row[i]=0;
         for(int j=0;j<m;j++)
          row[i]+=mat[i][j];
     }
     
     for(int i=0;i<m;i++)
     {  col[i]=0;
         for(int j=0;j<n;j++)
          col[i]+=mat[j][i];
     }
      int maxm= INT_MIN;
      for(int i=0;i<n;i++)
     {  
         for(int j=0;j<m;j++)
          maxm=max(maxm,row[i]^col[j]);
     }
    
   cout<<maxm<<endl;
   return 0;    
}
