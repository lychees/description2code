#include <iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
int n;
long long s[1000001],c[1000001];



int main()
{
 	long long sum=0;
	cin>>n;
	int i;
	for(i=0;i<n;i++){
		cin>>s[i];
		
	}
	for(i=0;i<n;i++){
		cin>>c[i];
	}
	sort(c,c+n);
	sort(s,s+n);
	for(i=0;i<n;i++){
		sum+=c[i]*s[i];
	}
	cout<<sum;
	//we have to find out maximum strength
    return 0;
}
