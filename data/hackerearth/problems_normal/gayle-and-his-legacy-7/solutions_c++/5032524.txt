#include<iostream>
#include<map>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		long k;
		cin>>n>>k;
		long add[n],p[n];
		for(int i=0;i<n;i++)
		{
			cin>>add[i]>>p[i];
		}
		multimap<long,long> m;
		for(int i=0;i<n;i++)
		{
			 m.insert(pair<long , long>(p[i],add[i]));
		}
		int f=1;
		 for (multimap< long,  long>::iterator it = m.begin();
       it != m.end();
       ++it)
       {
       	if((*it).first<=k)
       	{
       		k=k+(*it).second;
       	}
       	else 
       	{
       		f=0;
       		break;
       	}
       }
       if(f)
       {
       	cout<<"YES"<<endl;
       }
       else cout<<"NO"<<endl;
	}
}
