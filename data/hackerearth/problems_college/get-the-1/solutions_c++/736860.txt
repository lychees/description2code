#include <iostream>
#include <cmath>
using namespace std;

int main() {
	int a , n ;
	cin>>n;
	for(int i = 0 ; i < n ; i++)
	{
	cin>>a;	
	while(a%10 != a)
	{
		if((a%1000 == a)&&(a%100 != a))
		{
			a = pow((a/100),2) + pow(((a%100)/10),2) + pow((a%10),2);
		}
		else if(a%100 == a)
		{
			a = pow((a/10),2) + pow((a%10),2);
		}
	}
	
	if(a == 1)
	cout<<"Case"<<" "<<"#"<<i+1<<":"<<"YES"<<"\n";
	
	else
	cout<<"Case"<<" "<<"#"<<i+1<<":"<<"NO"<<"\n";
	}
	return 0;
}
