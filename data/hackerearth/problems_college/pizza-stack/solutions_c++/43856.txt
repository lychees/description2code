#include <stdio.h>
#include <string.h>
#define  MOD 1000000007

long long dp[1010][1010] , tot[1010] ;
int t , n ;

int main()
{   
	memset(dp,0,sizeof(dp)) ; dp[1][1] = 1 ; tot[1] = 1 ;
    for( int i = 2 ; i <= 1000 ; i++ )
    { 
      long long tmp = 0 ;	
      for( int j = 1 ; j <= i ; j++ )  	
      { 	
      	dp[i][j] = ( ((long long)j*dp[i-1][j])%MOD+dp[i-1][j-1])%MOD ;
        tmp = ( tmp+dp[i][j] )%MOD ; 
      }
      tot[i] = tmp ;
    }
    
    scanf("%d",&t) ;
    while( t-- )
    {
        scanf("%d",&n) ;
        printf("%lld\n",tot[n]) ;
    }
    
    return 0;
}
