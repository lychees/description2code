#include <iostream>
using namespace std;

int main()
{
    int i,count=0,j,k;
    int n;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
          cin>>a[i];
    for(i=0;i<n;i++)
    {
    	for(j=0;j<n&&j!=i;j++)
    	{
    		for(k=0;k<n&&k!=i&&k!=j;k++)
    		{
    			if((a[k]-a[j])==(a[j]-a[i]))
    			       count++;
    		}
    	}
    }
    cout<<count;
    return 0;
}
