#include <cstdio>
#include <algorithm>
using namespace std;
#define N 1024
bool dp[N];
int in[N];
int main() {
	int t;
	scanf("%d",&t);
	while(t--) {
		int m,n,ans=0;
		scanf("%d%d",&n,&m);
		for(int i=0;i<N;i++) dp[i]=0;
		dp[0]=1;
		while(m--) {
			int a; scanf("%d",&a);
			for(int i=N-1; i-a>=0; i--) if(dp[i-a]) dp[i]=1;
		}
		for(int i=0;i<n;i++) scanf("%d",in+i);
		sort(in,in+n);
		int cnt=0;
		for(int i=0; i<=n; i++) {
			if(i>0 && i<n  && in[i-1]==in[i]) { cnt++; }
			else {
				if(cnt>0 && dp[cnt]) ans++;
				cnt=1;
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
