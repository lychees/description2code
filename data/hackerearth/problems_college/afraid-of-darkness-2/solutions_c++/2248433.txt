#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;
int candles[25005];

int main()
{
	ios_base::sync_with_stdio(false);
	bool fg=0;
	int t,N,i,days,j,k;
	cin>>t;
	while(t--)
	{
		cin>>N;
		for(i=0;i<N;i++)
			cin>>candles[i];

		days=0;
		if(N==0)
		{
			cout<<0<<endl;
			continue;
		}

		for(k=0;k<N;k++)
		{
			sort(candles,candles+N);
			i=N-1;
			fg=0;
			for(j=0;j<=days;j++)
			{
				candles[i]--;
				if(candles[i]<0)
				{
					fg=1;
					break;
				}
				i--;
			}
			if(fg)
				break;
			days++;
		}

		cout<<days<<endl;
	}
	return 0;
}
