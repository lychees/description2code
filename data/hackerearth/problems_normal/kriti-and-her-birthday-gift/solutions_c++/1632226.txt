char ss[13];
map < ll , vi > M;
ll eval(char *s)
{
    ll ret = 0;
    int i;
    for(i=0;s[i];i++)
    {
        assert('a' <= s[i] && s[i] <= 'z');
        ret = 26*ret + s[i]-'a';
    }
    return ret;
}

int main()
{
    int n,q,i;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        scanf("%s",ss);
        M[eval(ss)].pb(i);
    }
    scanf("%d",&q);
    for(i=0;i<q;i++)
    {
    	int l,r;
        scanf("%d%d%s",&l,&r,ss);
        ll val = eval(ss);
        int ans = upper_bound(all(M[val]),r) - lower_bound(all(M[val]),l);
        printf("%d\n",ans);
    }
    return 0;
}
