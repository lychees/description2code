#include <iostream>
const int len = 20000;
 
int main()
{
    int arr[20000] = {0};
 
    for (int i = 2; i < len; i++)
    {
        for (int j = i * i; j < len; j+=i)
        {
            arr[j - 1] = 1;
        }
    }
    int n;
    scanf("%d",&n);
    for (int i = 1; i <= n; i++)
    {
        if (arr[i - 1] == 1)
            std::cout << i << "\n";
    }
}
