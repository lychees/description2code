def isprime(n):
    n = abs(int(n))
    if n < 2:
        return False
    if n == 2:
        return True
    if not n & 1:
        return False
    for x in range(3, int(n**0.5)+1, 2):
        if n % x == 0:
            return False
    return True


def inv():
    count = input()
    lista = map(int, raw_input().split())
    listb = map(int, raw_input().split())
    incr = 0
    for index in range(count):
        if lista[index] > listb[index]:
            incr += 1
    if isprime(incr):
        print "MAGIC INVERSION"
    else:
        print "SIMPLE INVERSION"


inv()
