#include<bits/stdc++.h>

#define s(n)                        scanf("%d",&n)
#define p(n)                        printf("%d\n",n)
#define forall(i,a,b)               for(int i=a;i<b;i++)
#define print_v(x)                  for(int i=0;i<x.size();i++) cout << x[i].first << " "
#define ll long long int
int main()
{

     forall(k,0,t)
     {

     forall(i,0,n)

          s(p);
          child.push_back(p);

     sort(tree.begin(),tree.end());

     forall(i,0,n)

          }

     p(tot);
