#include<iostream>
using namespace std;
typedef long long ll;
ll ar[1000005];
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		ll value,N,m1,m2;
		cin>>N>>m1>>m2;
		if(N==0 || m1==0)
		{
			cout<<"1\n";
			continue;
		}
		if(N<m1)
			cout<<"0\n";
		else if(m1<=N && N<m2)
		{
			if(m1!=0)
			{
				value=N/m1;
				if(value%2==0)
					cout<<"0\n";
				else cout<<"1\n";
			}
			else cout<<"1\n";
		}
		else
		{
			ar[0]=0;
			for(ll i=0;i<m1;i++)
				ar[i]=0;
			if(m1!=0)
			{
				for(ll i=m1;i<m2;i++)
				{
					value=i/m1;
					if(value%2==1)
						ar[i]=1;
					else ar[i]=0;
				}	
				for(ll i=m2;i<=N;i++)
				{
					if(ar[i-m1]==0 || ar[i-m2]==0)
						ar[i]=1;
					else ar[i]=0;
				}
				cout<<ar[N]<<endl;
			}
			else if(m2!=0)
			{
				value=N/m2;
				if(value%2==0)
					cout<<"0\n";
				else cout<<"1\n";
			}
			else cout<<"1\n";
		}
	}
	return 0;
}
