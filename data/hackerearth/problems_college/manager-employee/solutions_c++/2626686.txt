#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n, i, j, t;
    cin >> t;
    while(t--) {
    	cin >> n;
    	vector<int> a;
    	for(i = 0; i < n; i++) {
    		cin >> j;
    		a.push_back(j);
    	}
    	sort(a.begin(), a.end());
    	vector<int> v;
    	for (i = 0, j = n-1; i < j; i++, j--)
            v.push_back(a[i]+a[j]);
        sort(v.begin(), v.end());
        cout << v[v.size()-1]-v[0] << endl;
    }
    
    return 0;
}
