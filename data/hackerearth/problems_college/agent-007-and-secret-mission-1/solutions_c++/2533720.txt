#include <iostream>
#include<string.h>
#include<stdio.h>
using namespace std;
int main()
{
    long t,i,n,cnt1,cnt0;
    char a[1000000];
    cin>>t;
    while(t!=0)
    {
     cnt1=0,cnt0=0;
     cin>>a;
     cin>>n;
     for(i=0;i<strlen(a);i++)
     {
     	if(a[i]=='1')
     	   cnt1++;
     	else
     	   cnt0++;
     }
     int k=0;
     for(i=1;i<=n;i+=2)
         k++;
     if(k%2==0)
        cout<<cnt1<<"\n";
     else
        cout<<cnt0<<"\n";
     t--;
    }
    return 0;
}
