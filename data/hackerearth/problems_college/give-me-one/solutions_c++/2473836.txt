#include<bits/stdc++.h>
using namespace std;
int minf(int a,int b){
	return (a<b)?a:b;
}
int a[20000001];
int main()
{
    int t;
    a[0]=0;a[1]=0;a[2]=1;a[3]=1;
    for(int i=4;i<=20000001;i++){
        	int two=pow(10,7),three=pow(10,7);
        	if(i%2==0)
        		two=a[i/2];
        	if(i%3==0)three=a[i/3];
        	a[i]=1+minf(minf(two,three),a[i-1]);
        }
    scanf("%d",&t);
    while(t--)
    {
        long long  n,c;
        c=0;
        scanf("%lld",&n);
        printf("%lld\n",a[n]);
    }
    return 0;
}
