#include<bits/stdc++.h>

using namespace std;
vector<int>adj[10050];
int ans=0;
int dfs(int v,int p,int k)
{
    int s=1;

    for(int i=0;i<adj[v].size();i++)
    if(adj[v][i]!=p){
        s += dfs(adj[v][i],v,k);
    }

    if(s>=k)
    {
        ans++;
        return 0;
    }else
    return s;
}
int main()
{
    int k,n,i,x;

    cin>>k>>n;

    for(i=2;i<=n;i++)
    {
        cin>>x;

        adj[i].push_back(x);
        adj[x].push_back(i);
    }

    dfs(1,-1,k);

    cout<<ans;

    return 0;
}
