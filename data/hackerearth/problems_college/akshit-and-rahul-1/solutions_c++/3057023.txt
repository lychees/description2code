#include<iostream>
using namespace std;
int main(){
	int t;
	cin>>t;
	for(int u=1;u<=t;u++){
		long long a1,a2,b1,b2;long h;
		cin>>a1>>a2>>b1>>b2;long long int s=0;
		if(a1==b1==b2==a2)s=0;
		else{
		
		for(long i=a1;i<=a2;i++){
			if(i%2==0)h=i/2;
			else h=(i+1)/2;
			if(b1<=h){
				if(b2<=h){
					s=s+b2-b1+1;
				}
				else{if(b2<=h+i){s=s+h-b1+1;
				}
				else{s=s-b1+1+b2-i;
				}
				}
			}
			else{if(b1<=h+i){
				if(b2>h+i){s=s+b2-h-i;
				}
			}
			else{s=s+b2-b1+1;
			}
			}}}
			cout<<"Case #"<<u<<": "<<s<<endl;
		}
	}
