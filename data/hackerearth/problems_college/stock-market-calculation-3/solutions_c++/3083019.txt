const int N = 1e5 + 23;
int a[N];

#undef int
int main()
{
#define int long long
	int t; scanf("%lld", &t);
	while (t--)
	{
		int n; scanf("%lld", &n);
		for (int i = 0; i < n; i++)
			scanf("%lld", a + i);
		int mn = 0, ret = 0;
		for (int i = n - 1; i >= 0; i--)
		{
			if (a[i] < mn) ret -= a[i] - mn;
			mn = max(mn, a[i]);
		}
		printf("%lld\n", ret);
	}
	return 0;
}
