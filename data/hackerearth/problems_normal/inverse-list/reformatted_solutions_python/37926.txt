import sys
stdin = sys.stdin


def check_inverse(n, numbers):

    a = {}
    j = 1
    new_numbers = []
    for i in numbers:
        a[i] = j
        j += 1
    for i in a:
        new_numbers.append(a[i])

    if new_numbers == numbers:
        print "inverse"
    else:
        print "not inverse"
