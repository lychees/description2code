#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
	long int T,N,j;
	scanf("%ld",&T);
	for(;T--;)
	{
		scanf("%ld",&N);
		long int t1=1,t2=2;
		if(N==1)
			printf("%ld\n",t1);
		else if(N==2)
			printf("%ld\n",t2);
		else
		{
			long int key;
			for(j=2;j<N;j++)
			{
				key=t1;
				t1=t2;
				t2=key+t2;
			}
			printf("%ld\n",t2);
		}
	}
}
