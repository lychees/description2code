#include <iostream>
using namespace std;

int main()
{
    int t;cin>>t;while(t--)
    {
    	int n;
    	cin>>n;
    	int arr[n],i,flag=0;
    	int c=n;
    	for(i=0;i<n;i++)
    	cin>>arr[i];
    	for(i=0;i<n-1;i++)
    	{ flag=0;
    		for(int j=i+1;j<n;j++)
    		{
    			if(arr[j]>=arr[i])
    			{flag=1;break;}
    		}
    		if(flag==1)c--;
    	}
    	cout<<c<<"\n";
    }
    return 0;
}
