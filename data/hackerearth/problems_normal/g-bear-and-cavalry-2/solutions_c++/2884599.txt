#define MAXN 100
#define xx first 
#define yy second

#define MOD 1000000007
#define INF 1000000000000000000LL
using namespace std;

int n;
int W[MAXN] , H[MAXN];
bool used[MAXN] = {0};
string calc () {
    memset (used, false, sizeof (used));
    int idx = 0;
    for (int i = 0; i < n; i ++ )
        if (H[idx] < H[i]) idx = i;
    used[idx] = 1;
    int brave =  W[0] * H[idx];
    for (int i = 1; i < n; i ++ ) {
        int best = -1;
        
        for ( int j = 0; j < n; j ++ ) {
            if ( used [ j ] == 0 && H[j] * W[i] < brave ) {
                if ( best == -1 || H[j] * W[i] > W[i] * H[best] ) best = j;
            }    
        }    
        if (best == - 1) return "NO";
        used[best] = 1;
        
    }             
    return "YES";
}
int main(){
    int T; cin >> T;
    while (T--) {
    cin >> n;
    for ( int i = 0; i < n; i ++ )
        cin >> W[i];    
    
    for (int j = 0; j < n; j ++ ) 
        cin >> H[j];
    cout << calc () << endl; 
    }
//system("pause");    
return 0;
}
