#include <bits/stdc++.h>
using namespace std;
vector <int> mx;
int md=1e9+7;
void init()
{
	mx.clear();
}
void solve()
{
	long long ans=1;
	int n;
	init();
	scanf("%d",&n);
	int t;
	for(int i=0;i<n;i++)
	{
		scanf("%d",&t);
		mx.push_back(t);
		
	}
	sort(mx.begin(),mx.end());
	for(int i=0;i<mx.size();i++)
	{
		ans=(ans*((mx[i]-i)*1ll))%md;
		if(mx[i]-i<=0)
			ans=0;
	}
	printf("%lld\n",ans);
}
int main()
{
   int t;
   scanf("%d",&t);
   for(int i=0;i<t;i++)
   	solve();
   return 0;
}
