#include<bits/stdc++.h>
using namespace std;
 



int main()
{

	
      ios_base::sync_with_stdio(false);
		cin.tie(NULL);
		int t;
	cin>>t;
	while(t--)
	{

		int n, m;
		cin>>n>>m;
		int *arr=new int[n];
		int i=n-1;
		for(int k=0; k<n; k+=2)
		{

			arr[i]=k;i--;
			
		}
		i=0;
		for(int k=1; k<n; k+=2)
		{

			arr[i]=k;i++;
		}
		cout<<arr[m]<<endl;
		delete arr;
		      
	}
	return 0;
}
