ll int power(ll int a,ll int b)
{
    ll int ans = 1;
    while(b)
    {
        if(b&1)ans*=a;
        a=a*a;
        b/=2;
    }
    return ans;
}
int main()
{
    fast;
    int t;
    cin>>t;
    while(t--)
    {
        ll int a,b,c;
        cin>>a>>b>>c;
        if(b<0){
        	if(a==1)cout<<"1\n";
        	else cout<<"0\n";
        }
        else {
            ll int temp = power(a,b);
            ll int temp2 = temp/c;
            ll int temp3=temp2+1;
            if(abs(temp - c*temp2) <= abs(temp - c*temp3))
                cout<<c*temp2*1LL<<endl;
            else cout<<1LL*c*temp3<<endl;
        }
    }

    return 0;
}
