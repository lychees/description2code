#include <cstdio>
#include <iostream>
#include <cstring>

using namespace std;

int main()
{
	int t,c1,r1,c2,r2,ans;
	scanf("%d",&t);
	while (t --) {
		scanf("%d %d %d %d",&r1,&c1,&r2,&c2);
		if (c1 == r2 && c2 == r1) {
			ans = 2;
		} else if ((c1 == r2 && c2 != r1) || (c2 == r1 && c1 != r1)) {
			ans = 1;
		} else {
			ans = 0;
		}
		printf("%d\n",ans);
	}
	
	return 0;
}
