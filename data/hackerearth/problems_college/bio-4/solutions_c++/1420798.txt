#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    int n,i,j,l1,l2,c,ans;
    char a[1003],b[1003];
    cin>>t;
    while(t--)
    {
    	ans=l2;
        scanf(" %s %s",a,b);
        l1=strlen(a);
        l2=strlen(b);
        for(i=0;i<=l1-l2;i++)
        {
        	c=0;
        	for(j=0;j<l2;j++)
        		if(b[j]!=a[i+j])
        			c++;
        	if(c<ans)
        		ans=c;
        }
        printf("%d\n",ans);
    }
    return 0;
}
