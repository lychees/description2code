A = int(raw_input())
N = int(raw_input())
X = [int(x) for x in raw_input().split()]


def gcd(a, b):
    if a > b:
        a, b = (b, a)
    while a != 0:
        b %= a
        a, b = (b, a)
    return b


count = 0
for x in X:
    if gcd(A, x) != 1:
        count += 1

print count
