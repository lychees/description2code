#include <bits/stdc++.h>
using namespace std;

int main() {
  int tc;
  cin >> tc;
  while (tc--) {
    int n, k;
    cin >> n >> k;
    int freq[n + 1];
    for (int i = 0; i <= n; i++) {
      freq[i] = 0;
    }
    int roll;
    for (int i = 0; i < k; i++) {
      cin >> roll;
      freq[roll]++;
    }
    for (int i = 1; i <= n; i++) {
      if (freq[i] % 2 != 0) {
        cout << i << endl;
        break;
      }
    }
  }
  return 0;
}
