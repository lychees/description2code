#include <bits/stdc++.h>
//SIEVE
//KISE RYOTA
//EASY
using namespace std;
int isprime[1000001];
int ans[1000001];
void pre()
{
 
int i=4,j;	
	while(i<=1000000)
	{
		isprime[i]=1;
		i=i+2;
	}
	
	for(i=3;i<=sqrt(1000000);i++)
	{
		if(isprime[i]==0)
		{
			for(j=3*i;j<=1000000;j=j+2*i)
			{
				isprime[j]=1;	
			}
		}
	}
 
}
 
void calc()
{
	int i;
	for(i=2;i<=1000000;i++)
	{
		if(isprime[i]==1)
		ans[i]=ans[i-1];
		else
		ans[i]=ans[i-1]+1;
	}
}
 
 
int main()
{
	int t,a,b;
	float answer;
    pre();
    calc();
   scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&a);
    	scanf("%d",&b);
    answer=(float)(ans[b]-ans[a-1])/(b-a+1);
		printf("%.6f\n",answer);
    
    }
    return 0;
}
