#include <bits/stdc++.h>

using namespace std;

int main() {
	int t; scanf("%d", &t);
	while(t--) {
		long long int n; scanf("%lld", &n);
		int sq = sqrt(n), count = 2 * sq - 1, diff = n - sq * sq;
		if(diff > 0) {
			if(diff >= sq)
				count += diff / sq;
			if(diff % sq) count++;
		}
		printf("%d\n", count);
	}
	return 0;	
}
