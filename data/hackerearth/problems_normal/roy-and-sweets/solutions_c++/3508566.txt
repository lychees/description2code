#include <iostream>
using namespace std;

long long dp[1000001];

int main()
{
	long long n, s, q, g, m;
	long long i, maxx = 0, r = 0;
	
	cin>>n;
	
	for(i = 1; i <= n; i++)
	{
		cin>>s>>q;
		dp[s] = q;
	}
	
	for(i = 1000000; i >= 1; i--)
	{
		dp[i] = max(dp[i], maxx);
		if(maxx < dp[i])
			maxx = dp[i];
	}
	
	cin>>g;
	
	for(i = 1; i <= g; i++)
	{
		cin>>m;
		r += (dp[m]/m) * 100;
	}
	cout<<r;
	
    return 0;
}
