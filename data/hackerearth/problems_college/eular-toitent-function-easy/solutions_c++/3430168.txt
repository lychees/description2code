#include<bits/stdc++.h>
using namespace std;
int gcd(int a,int b)
{
	if(a==0)
	return b;
	if(b==0)
	return a;
	else
	return gcd(b%a,a);
}
int main()
{
	int t;
	cin >> t;
	int n,count;
	while(t--){
		count=0;
		cin >> n;
		for(int i=1;i<n;i++){
		
			if(gcd(i,n)==1)
			count++;
		}
		cout << count <<endl;
	}
	return 0;
}
