#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;

int main()
{ int tc;
  scanf("%d",&tc);
  while(tc--)
  { int n;
    cin>>n;
    n=n*2;
    vector <int> a(n);
    int i,j;
    for(i=0;i<n;i++)
      cin>>a[i];
    sort(a.begin(),a.end());
    int max=-1;
    for(i=0,j=n-1;i<=j;i++,j--)
    { 
      int sum=a[i]+a[j];
      if(sum>max)
         max=sum;
    }
    cout<<max<<"\n";
  }
 return 0;
}
