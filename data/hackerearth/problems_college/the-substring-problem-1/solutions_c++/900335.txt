#include<iostream>
        
        using namespace std;
        int main()
        {
          int n, temp, max;
          char maxch, curr;  
         string a;   
         char arr[26];
         bool flag[26];
         for(int k=0; k<26; k++)
         {
         flag[k]=false;
         arr[k]=0;
         }
            int len = a.length();
            int test;
            
            cin>>test;
            for(int i = 0; i<test; i++)
            {
                 for(int k=0; k<26; k++)
                 arr[k]=0;
                    
            cin>>n;
            while(n--)
            {
                  for(int k=0; k<26; k++)
                  flag[k]=false;        
             cin>>a;
             for(int j = 0; j<a.length(); j++)
             {
             curr = a[j] - 'a';
             if(!flag[curr])
             {
             arr[curr] += 1;  
             flag[curr] = true;
             }       
             }
             }      
             max=0;
         for(int k=0; k<26; k++)
         {
                 temp = arr[k];
                 if(temp>max)
                 {
                 max=temp;
                 maxch = k;
                 }    
         }      
         cout<<(char)(maxch+'a')<<endl;     
            }
            return 0;
        }
