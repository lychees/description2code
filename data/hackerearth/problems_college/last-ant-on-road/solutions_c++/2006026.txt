#include<bits/stdc++.h>
using namespace std;
typedef struct node{
float val;
int dir;
}node;
int main()
{
 int t;
 cin>>t;
 while(t--)
 {
        int n,x;
        cin>>n;
        node rod[n+1];
        for(int i = 1 ; i <=n;i++)
        {
        cin>>x;
        rod[i].dir = x;
        rod[i].val = i;
        }
        vector<int> ll;
        while(1)
        {
            for(int i = n;i>=1;i--)
            {
                if(rod[i].val < 0.0 || rod[i].val > (n+1)*1.0)
                continue;

                if(rod[i].dir == -1)
                rod[i].val -= 0.5;
                else
                rod[i].val += 0.5;
                if(rod[i].val < 0.0 || rod[i].val > (n+1)*1.0)
                ll.push_back(i);

            }
            int flag = 0;
            for(int i = 1;i<=n;i++)
            {
                if(rod[i].val < 0.0 || rod[i].val > (n+1)*1.0)
                continue;
                flag++;
                if(rod[i].val == rod[i+1].val)
                {
                rod[i].dir = -1;
                rod[i+1].dir = 1;
                i++;
                }
            }
        //    for(int i = 1;i<=n;i++)
         //   {
         //   cout<<rod[i].val<<"\t";
          //  }
          //  cout<<endl;
          //  for(int i = 1;i<=n;i++)
          ///  {
          //  cout<<rod[i].dir<<"\t";
          //  }
          //  cout<<endl;
          //  getchar();
            if(flag==0)
            break;


        }
        cout<<ll[n-1]<<endl;

 }
}
