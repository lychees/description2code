#include <bits/stdc++.h>

using namespace std;
int a[100][100];
int b[100][100];

int dp(int n,int m,int x,int y)
{
    if(x<n-1 && y<m-1){
        if(b[x][y]>-1)
            return b[x][y];
        if(a[x+1][y]>a[x][y] && a[x][y+1]>a[x][y])
            return b[x][y]=max(1+dp(n,m,x+1,y),1+dp(n,m,x,y+1));
        if(a[x+1][y]>a[x][y])
            return b[x][y]=1+dp(n,m,x+1,y);
        if(a[x][y+1]>a[x][y])
            return b[x][y]=1+dp(n,m,x,y+1);
    }
    else if(x==n-1 && y<m-1){
        if(b[x][y]>-1)
            return b[x][y];
        if(a[x][y+1]>a[x][y])
            return b[x][y]=1+dp(n,m,x,y+1);
    }
    else if(x<n-1 && y==m-1){
        if(b[x][y]>-1)
            return b[x][y];
        if(a[x+1][y]>a[x][y])
            return b[x][y]=1+dp(n,m,x+1,y);
    }
    return 0;
}

int main()
{
    int t,n,m,i,j;
    cin>>t;
    while(t--){
        cin>>n>>m;
        for(i=0;i<n;i++){
            for(j=0;j<m;j++){
                cin>>a[i][j];
                b[i][j]=-1;
            }
        }
        cout<<1+dp(n,m,0,0)<<endl;
    }
    return 0;
}
