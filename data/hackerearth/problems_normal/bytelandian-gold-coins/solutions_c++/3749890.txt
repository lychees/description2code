#include <bits/stdc++.h>
using namespace std;
map<long long int,long long int>arr;
long long int DP(long long int n)
{
	if(n==0)
	return 0;
	if(arr.count(n)==0)
	{
		return arr[n]=max(n,DP(n/2)+DP(n/3)+DP(n/4));
	}
	return arr[n];
}
int main()
{
    long long int n,i;
    ios_base::sync_with_stdio(0);
    while(cin>>n)
   	{
    	arr[0]=0;
    	cout<<DP(n)<<"\n";
    	arr.clear();
    }
    return 0;
}
