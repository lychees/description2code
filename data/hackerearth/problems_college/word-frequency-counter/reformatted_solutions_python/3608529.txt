arr = raw_input().upper().split()
d = []
count = [0 for i in xrange(len(arr))]
j = 0
for i in arr:
    flag = 0
    for k in xrange(j):
        if i == d[k]:
            flag = 1
            break
    if flag:
        count[k] += 1
    else:
        d.append(i)
        count[j] = 1
        j += 1
for i in xrange(j):
    print d[i], count[i]
