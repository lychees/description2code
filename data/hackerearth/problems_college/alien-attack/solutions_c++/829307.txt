#include <bits/stdc++.h>
#define lld long long int
#define MOD 1000000007
using namespace std;

lld pow_lld(lld b) {
	lld ans = 1;
	while(b--) {
		ans = (ans*2) % MOD;
	}
	return ans;

}

lld power(lld y)
{
    lld temp,x = 2;
    if( y == 0)
        return 1;
    temp = power(y/2);
    if (y%2 == 0)
        return (temp*temp)%MOD;
    else
        return (x*temp*temp)%MOD;
}

int main(int argc, char const *argv[]) {
	ios_base::sync_with_stdio(false);
	int t;
	scanf("%d", &t); // t = 1;
	while(t--) {
		lld n,ans;
		scanf("%lld", &n);
		n--;
		if(n%2 == 0)
			ans = (3*power(n))%MOD - 1;
		else
			ans = (3*power(n))%MOD + 1;
		printf("%lld\n", ans % MOD);
	}
}
