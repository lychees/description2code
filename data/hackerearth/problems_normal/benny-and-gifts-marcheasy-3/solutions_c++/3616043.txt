#include<bits/stdc++.h>
using namespace std;

set<pair<long int,long int> > st;
set<pair<long int,long int> >::iterator it;
char str[100010];
int main()
{
	long int i,j,x,y,count;
	x=0;y=0;
	count=0;
	st.insert(make_pair(x,y));
	scanf("%s",str);
	for(i=0;str[i]!='\0';i++)
	{
		switch(str[i])
		{
			case 'L':
				y--;
				break;
			case 'R':
				y++;
				break;
			case 'U':
				x--;
				break;
			case 'D':
				x++;
				break;
		}
		it = st.find(make_pair(x,y));
		if(it==st.end())
		st.insert(make_pair(x,y));
		else
		{
			count++;
			//printf("%d %d ",it->first,it->second);	
		}
	}
	printf("%ld\n",count);
	return 0;
}
