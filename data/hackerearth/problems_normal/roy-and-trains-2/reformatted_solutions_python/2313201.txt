import math
for x in range(input()):
    t0, t1, t2, v1, v2, d = map(float, raw_input().split())
    l1 = int(math.ceil((d/v1)*60)+t1)
    l2 = int(math.ceil((d/v2)*60)+t2)
    if t1 < t0 and t2 < t0:
        print "-1"
    elif (t1 >= t0 and t2 < t0):
        print l1
    elif (t1 < t0 and t2 >= t0):
        print l2
    else:
        print min(l1, l2)
