#include <iostream>
using namespace std;

int main()
{
    long long int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++)
    	 cin>>a[i];
        int count=0;
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<n;j++)
    		{
    			if(j>=i &&(a[i]==a[j]))
    			{
    			 count++;
    			}
    		}
    	}
    	cout<<count<<"\n";
    }
    return 0;
}
