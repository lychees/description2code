#include <bits/stdc++.h>
using namespace std;
#define ll long long

int max(int a,int b)
{
	if(a>b)
		return a;
	return b;
}

int main()
{
    int n;
    cin >> n;
    
    ll *A=new ll[n+1];
    ll *sum=new ll[n+1];
    for(int i=1;i<=n;i++)
    {
    	cin >> A[i];
    	sum[i]=0;
    }
    
    sum[1]=A[1];
    for(int i=2;i<=n;i++)
    {
    	int m=A[1];
    	for(int j=2;j<=sqrt(i);j++)
    	{
    		if(i%j==0)
    		{
    			m = max(m,sum[j]);
    			m = max(m,sum[i/j]);
    		}
    	}
    	sum[i]=A[i]+m;
    }
    for(int i=1;i<=n;i++)
    {
    	cout << sum[i] << " ";
    }
}
