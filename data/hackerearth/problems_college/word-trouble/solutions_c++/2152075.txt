#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t,i,x;char a[20],b[20];
    x=0;
    cin>>t;
    while(t--)
    {cin>>a;x=0;
    cin>>b;
    for(i=0;i<strlen(a)&&i<strlen(b);i++)if(a[i]!=b[i])x++;
    if(strlen(a)>strlen(b))x+=strlen(a)-strlen(b);else x+=strlen(b)-strlen(a);
    if(x==0)cout<<"Exact\n";
    else cout<<"Non Exact "<<x<<"\n";
    }
    return 0;
}
