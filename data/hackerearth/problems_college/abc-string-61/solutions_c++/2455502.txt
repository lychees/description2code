#include <iostream>
#define MOD 1000000007

using namespace std;

long long int solve(int N);
long long int exp(long long int a, long long int n);

int main()
{
	int T, N;
	cin>>T;
	while(T--)
	{
		cin>>N;
		cout<<solve(N)<<endl;
	}
	
    return 0;
}

long long int solve(int N)
{
	if(N<3)
		return 0;
		
	long long int ans = exp(3, N-1);
	ans = ans - exp(2, N);
	if(ans<0)
		ans = ans+MOD;
	ans = (ans+1)%MOD;
	
	return (3*ans)%MOD;
	
}

long long int exp(long long int a, long long int n)
{
	if(n==0)
		return 1;
		
	long long int ans = exp(a, n/2);
	
	if(n%2==0)
	{
		ans = (ans*ans)%MOD;
	}
	else
	{
		ans = (((ans*ans)%MOD)*a)%MOD;
	}
	
	return ans;
}
