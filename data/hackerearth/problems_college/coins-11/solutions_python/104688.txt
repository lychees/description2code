import math
def Sieve(A,limit):
	A = [True] * (limit+1)
	A[0] = False
	A[1] = False
	for i in xrange(2,int(math.sqrt(limit+1))):
		if(A[i]):
			for j in xrange(i*i,limit+1,i):
				A[j] = False
	return A
