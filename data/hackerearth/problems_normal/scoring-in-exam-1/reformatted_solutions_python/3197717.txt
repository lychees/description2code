

str = raw_input()
N, Q = map(lambda x: int(x), str.split())
time = raw_input()
TList = [int(t) for t in time.split()[:N+1]]
score = raw_input()
SList = [int(s) for s in score.split()[:N+1]]
sortedTuple = sorted(zip(TList, SList), key=lambda tup: tup[1], reverse=True)

sumList = []
sumList.append(sortedTuple[0][0])
for i in xrange(1, len(sortedTuple)):
    sumList.append(sumList[i-1]+sortedTuple[i][0])

for x in xrange(0, Q):
    query = int(raw_input())
    print sumList[query-1]
