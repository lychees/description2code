#include <bits/stdc++.h>
using namespace std;

void func(int n){
	if(n>3){
	bool prime[n+1];
	memset(prime,true,sizeof(prime));
	for(int p=2;p*p<=n;p++){
		if(prime[p]==true){
			for(int i=p*2;i<=n;i+=p){ 
				prime[i]=false;
				// cout<<i<<endl;
			}
		}
	}
	for(int p=2;p<=n;p++){
		if(!prime[p]){
			cout<<p<<" ";
		}
	}
	}
	else{
		cout<<"0";
	}
}
int main()
{
   int t;
   cin>>t;
   func(t);
    return 0;
}
