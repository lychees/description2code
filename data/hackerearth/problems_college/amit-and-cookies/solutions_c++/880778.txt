#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long int n,k;
    cin >> n>> k;
    long long int arr[n],sum=0,cnt=0;
    for(int i=0;i<n;i++)
      cin>> arr[i];
    sort(arr,arr+n);
    for(int i=0;i<n;i++)
    {
    	sum+=arr[i];
    	
    	if(sum>k)
    	  break;
    	 else 
    	cnt++;  
    }
    cout << cnt;
    return 0;
}
