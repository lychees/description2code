#include <iostream>
#include <string>

using namespace std;

int main()
{
	string s; cin >> s;
	int n; cin >> n;
	int sy = 0, an = 0;
	
	while (n--) {
		string s1; cin >> s1;
		int len = s.length(), len1 = s1.length();
		
		if (len == len1) {
			if (s[0] == s1[0]) {
				sy++;
			}
			else if (s[0] == (s1[0]-3)) {
				an++;
			}			
		}	
	}
	
	cout << sy << " " << an;
}
