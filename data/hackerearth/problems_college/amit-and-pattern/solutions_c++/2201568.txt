#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	int ar[n+10][m+10];
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		scanf("%d",&ar[i][j]);
	}
	int i = 1, j = 1;
	while(j<=m)
	{
		for(int p=i,q=j; p<=n && q>=1 ; p++,q--)
		printf("%d ",ar[p][q]);
		printf("\n");
		++j;
	}
	++i;
	j=m;
	while(i<=n)
	{
		for(int p=i,q=j; p<=n && q>=1; p++,q--)
		printf("%d ",ar[p][q]);
		printf("\n");
		++i;
	}
 	return 0;
}
