def summ(f):
    a = 0
    for j in range(1, f+1):
        a = a+j
    return a


def setbits(s):
    nn = []
    z = 0
    o = 0
    for i in xrange(len(s)):
        if s[i] == '0':
            z += 1
        else:
            o += 1
    nn.append(z)
    nn.append(o)
    return nn


N = int(raw_input())

    s = raw_input()
    n1 = setbits(s)
