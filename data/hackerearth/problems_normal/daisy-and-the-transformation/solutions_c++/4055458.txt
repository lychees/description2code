#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t >0)
    {
    	string s1, s2;
    	int n;
    	cin >> s1 >> s2 >> n;
    	int length = s1.size(), count = 0;
    	for (int i = 0; i < length; i++)
    	{
    		if (s1[i] != s2[i])
    			count++;
    		if (count > n)
    			break;
    	}
    	
    	if (count > n)
    		cout << "No" << endl;
    	else
    	{
    			n = n-count;
    			if (n%2 == 0)
    				cout << "Yes" << endl;
    			else
    				cout << "No" << endl;
    	}
    	t--;
    }
    return 0;
}
