#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define inf 1000000000000
#define mod 1000000007
#define pb push_back
#define mp make_pair
#define all(v) v.begin(),v.end()
#define S second
#define F first
#define boost1 ios::sync_with_stdio(false);
#define boost2 cin.tie(0);
#define mem(a,val) memset(a,val,sizeof a)

ll par[1000001];
vector<pair<ll,pair<ll,ll> > >edge;
ll find(ll x)
{
	if(par[x]==x)
	return x;
	return par[x]=find(par[x]);
}	
int main()
{
	ll i,j,n,m,x,xp,yp,y,z,ans=0;
	cin>>n>>m;
	for(i=1;i<=m;i++)
	{
		cin>>x>>y>>z;
		edge.pb(mp(z,mp(x,y)));
	}
	sort(all(edge));
	for(i=1;i<=n;i++)
	par[i]=i;
	for(i=0;i<edge.size();i++)
	{
		x=edge[i].S.F;
		y=edge[i].S.S;
		xp=find(x);
		yp=find(y);
		if(xp==yp)
		continue;
		par[xp]=yp;
		ans+=edge[i].F;
	}
	cout<<ans;
	return 0;
}
