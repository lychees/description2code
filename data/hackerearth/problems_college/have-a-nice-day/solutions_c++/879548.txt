#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
 int t,n,i;
 
 cin>>t;
 while(t>0){
 	cin>>n;
  vector<int> arr1(n);
 vector<int> arr2(n);
   for(i=0;i<n;i++)
   cin>>arr1[i];
   cout<<endl;
   for(i=0;i<n;i++)
   cin>>arr2[i];
   std::sort(arr1.begin(),arr1.begin()+n);
   std::sort(arr2.begin(),arr2.begin()+n);
   int sum=0;
   for(i=0;i<n;i++){
   	sum=sum+arr1[i]*arr2[i];
   }
   cout<<sum;
   t--;
 }
   
   return 0;
}
