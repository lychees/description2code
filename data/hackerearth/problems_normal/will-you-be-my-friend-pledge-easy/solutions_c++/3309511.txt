#include<iostream>

using namespace std;

int main()
{
    int num, n;

    cin>>num;
    cin>>n;

    int numberwithchild[n];

    for (int i = 0; i<n; i++)
        cin>>numberwithchild[i];

    int arr[num/2];
    for(int i=2;i<num/2;i++)
    {
        if(num%i == 0)
            arr[i] = 1;
        else
            arr[i] = 0;
    }

    int count = 0;

    for(int i = 0; i<n; i++)
    {
        for(int j = 2;j<num/2;j++)
        {
            if(arr[j] == 1)
            {
                if(numberwithchild[i]%j==0)
                {
                    count++;
                    break;
                }
            }
        }
    }

    cout<<count;

    return 0;
}
