#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
	    int c1, c0, n, i;
	
	    string s;
	
	    cin >> s;
		n = s.size();
	    c1 = 0;
	    c0 = 0;
	    for (i = 0; i < n; i++) {
	        if (s[i] == '1')
	            c1++;
	
	        else
	            c0++;
	    }
	
	    cout << max(c1, c0) - min(c1, c0) << endl;
	}
    return 0;
}
