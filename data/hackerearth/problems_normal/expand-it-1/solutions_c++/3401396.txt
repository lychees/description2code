

#include <vector>
#include <cctype>
#include <iostream>
using namespace std;

int main() {
    vector<long long int> vec(26, 0);
    string str;
    long long int counter = 0, Q, sum = 0, temp;
    cin >> str >> Q;
    int pos = str[0] - 'a';
    for(int i = 1; i < str.size(); i++) {
        if(isalpha(str[i]) == 0) {
            counter = counter*10 + (str[i] -'0');
        }
        else {
            vec[pos] += counter;
            pos = str[i] - 'a';
            counter = 0;
        }
    }
    vec[pos] += counter;
    sum += vec[0];
    for(int i = 1; i < 26; i++) {
        sum += vec[i];
        vec[i] += vec[i - 1];
    }
    for(int i = 0; i < Q; i++) {
        cin >> temp;
        if(temp > sum)
            cout << "-1" << endl;
        else {
            for(int i = 0; i < 26; i++) {
                if(temp <= vec[i]) {
                    cout << char(i + 'a') << endl;
                    break;
                }
            }
        }
    }
    return 0;
}
