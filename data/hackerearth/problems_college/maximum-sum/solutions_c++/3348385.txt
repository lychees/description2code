/*Ack - http://codeforces.com/blog/entry/18051 */
#include<bits/stdc++.h>
using namespace std;
#define RI(x) scanf("%d",&(x))
const int N = 1000006;
int n, cc, ins, val;
char qq;
int t[2*N];

void build(){
    for(int i=n-1; i; i--){
        t[i]=max(t[i<<1],t[i<<1|1]);
    }
    return;
}

int query(int l, int r){/* [l,r) */
    int res=0;
    for(l+=n, r+=n; l<r; l>>=1,r>>=1){
        if(l&1){
            res=max(res,t[l++]);
        }
        if(r&1){
            res=max(res,t[--r]);
        }
    }
    return res;
}

void modify(int location, int value){
    for(t[location+=n]=value; location>1; location>>=1){
        t[location>>1]=max(t[location],t[location^1]);
    }
}

int main(){
    RI(n);
    for(int i=0; i<n; i++){
        scanf("%d",&t[n+i]);
    }
    build();
    RI(cc);
    while(cc--){
        cin >> qq;
        RI(ins);
        RI(val);
        if(qq == 'Q'){
            cout << query(ins-1,val) << endl;
        }else{
            modify(ins-1, val);
        }
    }
    return 0;
}
