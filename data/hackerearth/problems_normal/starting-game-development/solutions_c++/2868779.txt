#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main()
{
	
    int  n , m , q ; 
    scanf("%d%d%d",&n,&m,&q) ; 
    
    vector<vector<int>>array ; 
    
    for(int i = 0 ; i < n ; i ++)
    {
    	vector<int> temp ; 
    	array.push_back(temp) ; 
    	for(int j = 0 ; j < m ; j ++)
    	{
    		int temp ; 
    		scanf("%d",&temp) ;
    		array[i].push_back(temp) ;
    		//printf("%d " , array[i][j])
    		
    	}
    }
    
    while(q--)
    {
    	
    	int ans = 1000000 ;  
    	for(int i = 0 ; i < n ; i++)
    	{
    		int temp ;
    		scanf("%d",&temp) ; 
    		vector<int>:: iterator it = upper_bound(array[i].begin(),array[i].end(),temp) ; 
    		int level = it-array[i].begin() ;
    		ans = min(ans , level) ; 
    	}
    	
    	printf("%d\n",ans) ;
    }
    
    
    return 0;
}
