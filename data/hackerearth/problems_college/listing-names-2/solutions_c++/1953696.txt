#include <bits/stdc++.h>
#define ll long long int
#define pb push_back
#define mp make_pair
#define s(a) scanf("%lld",&a)
#define f first
#define sc second
#define sd(a) scanf("%lf",&a)

using namespace std;

vector<string>v;
map<string,ll> mpp;
string s;

int main()
{
    ll t,i,j,k;
    s(t);
    for(i=0;i<t;i++) {
        s.clear();
        cin>>s;
        if(mpp[s]>0) {
            mpp[s]++;
        }
        else {
            mpp[s]=1;
            v.pb(s);
        }
    }
    sort(v.begin(),v.end());
    for(i=0;i<v.size();i++) {
        cout<<v[i]<<" "<<mpp[v[i]]<<endl;
    }
    return 0;
}
