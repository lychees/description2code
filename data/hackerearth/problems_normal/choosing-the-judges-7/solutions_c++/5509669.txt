#include<bits/stdc++.h>
#define Int long long
using namespace std;

Int dp[100005];
int cnt=1;

int main(){
	
	int w;
	cin>>w;
	while(w--){
		
		memset(dp,0,sizeof dp);
		int n,i;
		cin>>n;
		Int A[n];
		for(i=0;i<n;i++)
			cin>>A[i];
		dp[0]=A[0];
		dp[1]=max(A[0],A[1]);
		for(i=2;i<n;i++)
			dp[i]=max(dp[i-1],dp[i-2]+A[i]);
		cout<<"Case "<<cnt<<": "<<dp[n-1]<<endl;
		cnt++;
		
	}
	
}
