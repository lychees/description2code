#include<bits/stdc++.h>
using namespace std;
long long gcd(long long x2,long long x3)
{
    if(x3==0)
    {
        return x2;
    }
    return gcd(x3,x2%x3);
}
int main()
{
    int t;
    scanf("%d",&t);
    //cin>>t;
    long long x1,y1,y,maximum,minimum,n,x,n1;
    while(t--)
    {
        maximum=0,minimum=0,x=1;
        scanf("%lld",&n);
       // cin>>n;
        int a[10];
        int b[10];
        int d[10];
        int c=0,flag=0,r,i,j=0,k,l=0;
        n1=n;
        while(n1 > 0)
        {
            r=n1%10;
            if(!(r==0 || r==2 || r==4 || r==6 || r==8))
            {
                flag=1;
                //break;
            }
            a[c]=r;
            n1=n1/10;
            c++;
        }

        for(i=0; i<c; i++)
        {
            b[j]=a[c-i-1];
            d[j]=a[c-i-1];
            j++;
        }
        if(flag==1)
        {
              for(i=0; i<j; i++)
                {
                    if(b[i] % 2 != 0)
                    {
                        break;
                    }
                }
                if(b[i]==9)
                {
                    if(i == 0)
                    {
                        b[i]=20;
                    }
                    else
                    {
                        b[i]=0;
                        int m=(i-1);
                        while(b[m] == 8 && m > 0)
                        {
                            b[m]=0;
                            m--;
                        }
                        if(b[m] == 8)
                        {
                            b[m]=20;
                        }
                        else
                        {
                            b[m]=b[m]+2;
                        }
                    }
                }
                else
                {
                    b[i]=b[i]+1;
                }
                for(k=i+1; k<j; k++)
                {
                    b[k]=0;
                }


                for(i=0; i<j; i++)
                {
                    if(d[i] % 2 !=0)
                    {
                        break;
                    }
                }
                //cout<<i<<endl;

                d[i]=d[i]-1;
                for(k=i+1; k<j; k++)
                {
                    d[k]=8;
                }
                minimum=d[j-1];
                maximum=b[j-1];
                for(i=j-2; i>=0; i--)
                {
                    x=x*10;
                    minimum=minimum+(d[i]*x);
                    maximum=maximum+(b[i]*x);
                }
                //  cout<<maximum<<endl;
                x1=(n-minimum);
                y1=(maximum-n);
                y=gcd(x1,y1);
                printf("%lld/%lld\n",(y1/y),(x1/y));

               // cout<<(y1/y)<<"/"<<(x1/y)<<endl;

            }
        else
        {
            printf("Unlimited Power\n");
            //cout<<"Unlimited Power"<<endl;
        }
    }
    return 0;
}
