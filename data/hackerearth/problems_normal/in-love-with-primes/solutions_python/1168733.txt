def is_prime(num):
	for i in range(2, num/2 +1):
		if num % i == 0:
			return False
	return True


def is_sum_primes(num):
    for i in range(2, num/2+1):
        if is_prime(i):
            if i == num-i:
                return True
            if is_prime(num-i):
                return True
    return False
