#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    vector<int>v;
    int t,a,b,c,m,mini;
    cin>>t;
    while(t--)
    {
              cin>>a>>b>>c;
              v.push_back(a);
              v.push_back(b);
              v.push_back(c);
              sort(v.begin(),v.end());
              if(v[2]-v[1]>=v[1]-v[0])
              cout<<v[2]-v[1]-1<<endl;
              else
              cout<<v[1]-v[0]-1<<endl;
              v.clear();
    }
   //  system("pause");
    return 0;
}
