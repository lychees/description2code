long long p[52];

int main(){
	long long two=2,i,j,x,y,q,n,m,t,l;	
	p[0]=1;
	for(i=1;i<=50;i++){
		p[i]=p[i-1]*two;	
	}
	
	cin>>t;
	
	while(t--){
		cin>>n;
		
		for(l=1;;l++){
			if( (n- p[l]*l)<=0 )	break;
			n-=(p[l]*l);
		}		
		y= n/l;
		if(n%l)	y++;
		vector<int> v;
		
		y--;
		while(y){
			v.push_back(y%2);
			y/=2;
		}
		
		while(1){
			x=v.size();
			if(x==l)	break;
			v.push_back(0);
		}
		
		reverse(v.begin(),v.end());
		
		if( n%l == 0){
			if(v[l-1]==0)	cout<<"Hacker";
			else			cout<<"Earth";
		}else{
			if(v[n%l - 1]==0)	cout<<"Hacker";
			else			cout<<"Earth";
		}
		
		cout<<endl;
	}
	
	return 0;
}
