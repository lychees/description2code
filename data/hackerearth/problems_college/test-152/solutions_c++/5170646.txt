#include <iostream>
#define ll long long int
#include <string>
#include <algorithm>
using namespace std;
int main()
{
    string s;
    ll len,i,t,a[160];
    cin>>t;
    while(t--)
    {
    	for(i=0; i<=150; i++)
    	{
    		a[i]=0;
    	}
    	cin>>len;
    	cin>>s;
    	for(i=0; i<s.length(); i++)
    	{
    		int z;
    		z = int(s[i]);
    		a[z]++;
    	}
    	sort(a,a+150,greater<ll>());
    	if(a[0] % 2 == 0)
    	{
    		cout<<"Yes\n";
    	}
    	else
    	{
    		cout<<"No\n";
    	}
    }
    return 0;
}
