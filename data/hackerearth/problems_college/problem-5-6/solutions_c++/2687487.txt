#include <bits/stdc++.h>
#define ll long long
#define ull unsigned long long
#define MOD 1000000007
#define INF 999999999
using namespace std;
int answer = 0;
int a[1002] = {};
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	int tc;
	cin >> tc;
	while(tc--)
	{
		int ans = 0;
		int n,x;
		cin >> n >> x;
		for(int i=1;i<=n;i++)
		{
			cin >> a[i];
			a[i] -= x;
		}
		int dp[1001] = {};
		for(int i=1;i<=n;i++)
		{
			dp[i] = a[i];
			for(int j=1;j<i;j++)
			{
				dp[i] = max(dp[i], dp[j]+dp[i-j]);
			}
		}
		cout << dp[n] << "\n";
	}
}
