#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,k,sum,i;
    	cin>>n>>k;
    	int flag = 0;
    	sum = 0;
    	for(i = n;i>=0;i--)
    	{
    		sum = sum + i;
    		if(sum >= k)
    		{
    			flag = 1;
    			break;
    		}
    	}
    	if(flag == 1)
    	{
    		cout<<"YES"<<endl;
    	}
    	else
    	{
    		cout<<"NO"<<endl;
    	}
    	flag = 0;
    	
    }
    return 0;
}
