#include <iostream>
#include <cstdlib>
#include <string>
#include <vector>
#include <algorithm>
#include <sstream>
#include <map>
#include <cctype>
#include <cmath>
#include <utility>
using namespace std;

typedef pair<int,int> ii;
typedef long long ll;
typedef vector<int> vi;
typedef vector<ii> vii;
typedef vector<ll> vl;


int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	
	int t;
	cin>>t;
	
	while(t--){
		vi duration;
		for(int i=0; i<7; ++i){
			int n;
			cin>>n;
			while(n--){
				int s,e;
				cin>>s>>e;
				duration.push_back(abs(s-e));
			}
		}
		sort(duration.begin(), duration.end());
		
		int classes = ceil((double)duration.size()*0.75);
		
		ll ans = 0;
		for(int i=0; i<classes; ++i){
			ans+=duration[i];
		}
		cout<<(604800-ans)<<"\n";
		
	}
	
	return 0;
}
