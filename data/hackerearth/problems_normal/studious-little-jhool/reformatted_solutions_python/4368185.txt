#!/usr/bin/env python
# -*- coding: utf-8 -*-

a = [1000] * 150
a[0] = 0
a[10] = 1
a[12] = 1
for i in xrange(13, len(a)):
    a[i] = min(a[i - 10] + 1, a[i - 12] + 1)

for i in xrange(0, len(a)):
    if a[i] >= 1000:
        a[i] = -1

t = int(raw_input())
for _ in xrange(0, t):
    n = int(raw_input())
    print a[n]
