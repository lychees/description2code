#include<stdio.h>
#include<iostream>
#include<string.h>
#include<cmath>
using namespace std;
int dp[100000]={0};
int ans(long int n,long int i)
{
    if(n==0)
    return 0;
    else if(n<0)
    return 100000;
    else if(dp[n]!=NULL)
    return dp[n];
    else
    dp[n]=1+min(ans(n-i,i),ans(n-i-1,i+1));
    return dp[n];
    }
int main()
{
int t;
scanf("%d",&t);
while(t--)
{
long int n;
scanf("%ld",&n);
int k=ans(n,1);
printf("%d\n",k);
memset(dp,0,sizeof(dp));
}
return 0;
}
