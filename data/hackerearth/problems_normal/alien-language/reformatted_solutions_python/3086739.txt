t = input()
while (t):
    t -= 1
    a = [0]*26
    b = [0]*26
    ori = raw_input()
    sub = raw_input()
    for i in ori:
        a[ord(i)-ord('a')] += 1
    for i in sub:
        b[ord(i)-ord('a')] += 1
    flag = False
    for i in range(26):
        if (a[i] > 0 and b[i] > 0):
            flag = True
    if (flag):
        print "YES"
    else:
        print "NO"
