#include<iostream>
#include<bits/stdc++.h>

using namespace std;



bool mycomparison( pair<int, int> p1, pair<int,int> p2){
	
	if(p1.second==p2.second){
		return p1.first > p2.first  ;
	}
else{
return p1.second<p2.second;

}
};



int main(){
	
 int t   ;
 cin >> t ;
int arr[26] = {0};
 string str ; 	

 vector< pair<char,int> > vec  ;

 vec.reserve(26) ; 

 char c = 'a';

  for(int i =0 ; i<26 ; i++){
 
  vec.push_back(make_pair(c+i, 0));


}

while( t-- ){
	
	cin >> str ; 
	
int arr[26] = {0};
	
for(int j = 0 ; j< str.length();j++){
	
arr[str[j]-'a']++;
	
}



vec.clear();


  for(int i =0 ; i<26 ; i++){
 
  vec.push_back(make_pair(c+i, arr[i]));


}
sort(vec.begin(),vec.end(),mycomparison);
for(int i =0 ; i<26 ; i++){
 
 cout << vec[i].first<<" ";


}

cout << endl ;


}	
	
	
	




}
