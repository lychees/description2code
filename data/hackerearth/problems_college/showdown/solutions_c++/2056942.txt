#include <bits/stdc++.h>
using namespace std;

bool comp(int a, int b) {
	return a > b;
}

int main()
{
	int n; cin >> n;
	int a[n], b[n];
	for (int i = 0; i < n; i++) {
		cin >> a[i];
	}
	for (int i = 0; i < n; i++) cin >> b[i];
	
	sort(a, a+n, comp); 
	
	sort(b, b+n, comp);
	int sum = 0, s2 = 0;
	
	int i = 0, j = 0;
	int ans = 0;
	while (i < n && j < n) {
		if (a[i] > b[j]) {
			ans+=100;
			i++; j++;
		} else {
			j++;
		}
	}
	sum = ans; ans = 0;
	i = 0;j = 0;
	while (i < n && j < n) {
		if (a[i] == b[j]) {
			i++; j++;
			ans += 50;
		} else if (a[i] > b[j]) {
			i++; j++;
			ans += 100;
		} else j++;
	}
	cout << max(sum,ans) << endl;
	return 0;
}
