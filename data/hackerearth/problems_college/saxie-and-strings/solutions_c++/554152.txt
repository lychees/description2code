#include<stdio.h>
#include<string.h>
#include<iostream>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        char s[105];
        scanf("%s",s);
        int l=strlen(s);
        int flag=0,i,j;
        for(i=0,j=l-1;i<j;i++,j--)
        {
            if(s[i]!=s[j])
            {
                flag=1;
                break;
            }
        }
        if(flag==0)
        printf("YES\n");
        else
        printf("NO\n");
    }
    return 0;
}
