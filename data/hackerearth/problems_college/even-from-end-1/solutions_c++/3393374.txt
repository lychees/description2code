#include <iostream>
#include <vector>

using namespace std;

int main(){
	int N;
	
	while(cin>>N){
		vector<int> even;
		int x;
		while(N--){
			cin>>x;
			if(x%2==0)
				even.push_back(x);
		}
		for(int i=even.size()-1;i>=0;i--){
			cout<<even[i]<<" ";
		}
		if(even.size()>=1)
			cout<<endl;
		else
			cout<<"None to display"<<endl;
	}
	
	return 0;
}
