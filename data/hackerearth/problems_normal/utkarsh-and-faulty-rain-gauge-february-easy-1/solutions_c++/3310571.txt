#include <iostream>
typedef long long int ll;
#define MAX 1005
using namespace std;
double a[MAX];

int main() {
	// your code goes here
	ll n,i;
	double x,y;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>x>>y;
		a[i]=(3*x*(x+1))/(2*y*(2*x+1));
	}
	double ans=a[0];
	
	for(i=1;i<n;i++)
	ans*=a[i];
	
	
	printf("%0.4lf",ans);
	return 0;
}
