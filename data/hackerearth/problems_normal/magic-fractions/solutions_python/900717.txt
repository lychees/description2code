def checkprime(a):
	for i in range(2,a):
		if(a%i==0):
			return False
	return True

res=1
if(numb<=2):
	res=numb-1
else:
	for j in range(3,numb+1):
		if(checkprime(j)==True):
			adder=adder*2
		res=res+adder
