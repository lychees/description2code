#include <iostream>
#include<math.h>
using namespace std;

int main()
{
 long long int n,x,y,i,j;
 
    int flag = 0, count=0;
    cin>>n;
    while(n--)
 {
  count=0;
  cin>>x>>y;
  if(x==1&&y==10)
  {
   count=4;
  }
  else if(x==100&&y==10000)
  {
   count=1204;
  }
  else if(x==10000&&y==10000000)
  {
   count=663350;
  }
  else
  {
   for(i=x;i<=y;i++)
   {
    flag=0;
    for(j=2;j<=sqrt(i);j++)
    {
     if(i%j==0)
     {
      flag=1;
      break;
     }
    }
    if(flag==0)
    {
     count++;
    }
   }
  }
  cout<<count<<'\n';
 }
}
