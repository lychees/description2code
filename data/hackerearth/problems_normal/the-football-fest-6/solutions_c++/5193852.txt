#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
	ll t;
	cin >> t;
	while(t--)
	{
    ll n, id;
    stack <ll> s;
    char c;
    cin >> n >> id;
    for(ll i = 0; i < n; i++)
    {
    	cin >> c;
    	if(c == 'P')
		{
			s.push(id);
			cin >> id;
		}
		else
		{
			ll p = id;
			id = s.top();
			s.pop();
			s.push(p);
		}
    }
    cout << "Player " << id << endl;
    }
    return 0;
}
