#include<iostream>
#include<string.h>
using namespace std;

int main()
{
    int T;
    string str;
    int acost;
    int bcost;
    cin >> T;
    int cost, len, mincost, low_end, high_start;
    while(T--)
    {
              cost = 0;
              cin >> str;
              len = str.length();
              cin >> acost;
              cin >> bcost;
              mincost = (acost > bcost)? bcost : acost;
              high_start = len/2;
              low_end = high_start - 1;
              bool is_valid = true;
              while(low_end >= 0)
              {
                            char l, h;
                            l = str.at(low_end);
                            h = str.at(high_start);
                            if(l == h)
                            {
                                 if(l == '/')
                                        cost += 2*mincost;
                                 low_end--;
                                 high_start++;
                            }
                            else
                            if(l == '/')
                            {
                                      if(h == 'a')
                                           cost += acost;
                                      else
                                           cost += bcost;
                                      low_end--;
                                      high_start++;
                            }
                            else
                            if(h == '/')
                            {  
                                      if(l == 'a')
                                           cost += acost;
                                      else
                                           cost += bcost;
                                      low_end--;
                                      high_start++;                                         
                            }
                            else
                            {
                                is_valid = false;
                                break;
                            }
              }
              if(is_valid)
                          cout << cost << endl;
              else
                          cout << "-1" << endl;
    }
    
    return 0;
}
