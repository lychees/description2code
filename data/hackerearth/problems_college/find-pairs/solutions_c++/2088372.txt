#include <iostream>
#include<map>
using namespace std;

int main()
{
    int n,tno;
    int i,x;
    int count;
    map <int,int> hash;
    
    cin>>tno;
    
    while(tno--)
    {
    	cin>>n;
    	
    	hash.clear();
    	
    	
    	
    	count=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>x;
    		if(hash[x]!=0)
    			count=count+hash[x];
    			
    		hash[x]++;;
    	}
    	
    	
    	
    	cout<<count+n<<endl;
    	
    	
    }
    return 0;
}
