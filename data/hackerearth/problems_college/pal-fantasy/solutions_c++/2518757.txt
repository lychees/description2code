    #include <iostream>
    #include<string>
    using namespace std;
     
    int main()
    {
    	int t;
    	cin>>t;
    	string s;
    	while(t--){
    		cin>>s;
    		int n=s.length();
    		for(int i=0;i<n;i++){
    			int start=i,end=n-1;
    			int flag=1;
    			
    			while(start<=end){
    				if(s[start++]!=s[end--]){flag=0;  break;}
    			}
    			if(flag==1){
    				{cout<<n+i<<endl;break;}
    			}
    		}
    	}
    	return 0;
    }
