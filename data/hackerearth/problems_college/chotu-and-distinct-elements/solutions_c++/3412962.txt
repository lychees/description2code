#include <iostream>
using namespace std;
#include <algorithm>
#include <stdio.h>

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	int a[n],i,j;
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&a[i]);
    	}
    	std::sort(a,a+n);
    /*	for(i=0;i<n;i++)
    	{
    		printf("%d ",a[i]);
    	}
    	printf("\n");
    */
    	int f=0;
    	i=0,j=n-1;
    	while(i<=j)
    	{
    		if(f==0){
    			printf("%d ",a[j]);
    			while(a[j-1]==a[j])
    				j--;
    			j--;
    			f=1;
    			continue;
    		}
    		else if(f==1){
    			printf("%d ",a[i]);
    			while(a[i]==a[i+1])
    				i++;
    			i++;
    			f=0;
    		}
    	}
    	printf("\n");
    }
    return 0;
}
