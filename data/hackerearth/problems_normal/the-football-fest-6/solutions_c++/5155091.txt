#include <iostream>
using namespace std;

int main()
{
	int t,x,i,j,top;
	char c;
	cin>>t;
	int s[100000];
	while(t!=0)
	{
		top=0;
		cin>>i;
		cin>>x;
		
		s[top++]=x;
		while(i!=0)
		{
			cin>>c;
			switch(c)
			{
				case'P':cin>>j;
						s[top++]=j;
						break;
				case 'B':j=s[top-2];
						s[top++]=j;
						break;
			}
			i--;
		}
		
    cout << "Player "<<s[top-1]<< endl;
    t--;
	}
	//for(i=0;i<top;i++) cout<<s[i]<<" ";
}
