#include <iostream>
#include <limits.h>
using namespace std;

int main()
{
    int n;
    cin >> n;
    int Jack[n],Jill[n];
    Jack[0]=1;
    Jack[1]=2;
    Jack[2]=4;
    for(int i=3;i<=n-1;i++)
    {
    	Jack[i]=Jack[i-1]+Jack[i-2]+Jack[i-3];
    }
    Jill[0]=1;
    Jill[1]=2;
    for(int i=2;i<=n-1;i++)
    {
    	Jill[i]=Jill[i-1]+Jill[i-2];
    }
    if(n>0 )
    {
    	cout << "Jack-" << Jack[n-1] << endl;
    	cout << "Jill-" << Jill[n-1] << endl;
    }
    else cout << "ERROR" << endl;
    return 0;
}
