#include <iostream>
using namespace std;

int main()
{
    int T,l=0,i,j;
    char s[105];
    cin>>T;
    cin.get();
    while(T--)
    {
    	l=0;
    	cin.getline(s,105);
    	for(i=0;s[i]!='\0';i++)
    	{
    		l+=s[i];
    	}
    	l/=i;
    	if(l%2==0)
    	{
    		cout<<s<<endl;
    	}
    	else
    	{
    		for(j=i-1;j>=0;j--)
    		{
    			cout<<s[j];
    		}
    		cout<<endl;
    	}
    }
}
