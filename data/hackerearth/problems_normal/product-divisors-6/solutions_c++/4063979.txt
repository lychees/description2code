#include<bits/stdc++.h>
//konoha's yellow flash
using namespace std;
map<int,int>m;
vector<int> v;
void solve(int num)
{
    int i = 2;
   while(num > 1)
   {
       while(num % i == 0)
       {
           m[i]++;
           num=num/i;
       }
       i++;

   }
}
int main()
{
    int t;
    cin >> t;
    int a[t + 1];
    int i;
    for(i = 0; i < t; i++)
    {
        cin >> a[i];
        solve(a[i]);
    }
    long long int ans = 1;
    map<int,int>::iterator it;
    for(it = m.begin(); it != m.end() ; it++)
    {
        ans = (ans * (it->second + 1))% 1000000007;
    }
    cout<<ans<<endl;
    return 0;
}
