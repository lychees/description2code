#include<iostream>
#include<cstdio>
#define gc getchar_unlocked
using namespace std;
int readInt()
{
	char c;
	int res=0;
	do{
		c=gc();
	}while(c<'0'||c>'9');
	do{
		res=res*10+c-'0';
		c=gc();
	}while(c>='0'&&c<='9');
	return res;
}
