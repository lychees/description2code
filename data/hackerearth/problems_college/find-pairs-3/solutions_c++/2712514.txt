#include <bits/stdc++.h>
using namespace std;
#define mx 2000000
int main()
{
///////////////	cout<<"hello";
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
    int t;
	cin>>t;
	while(t--)
	{
      int hash[mx]={0};
		int n;
		cin>>n;
		int arr[n];
		for(int i=0; i<n; i++)
		{

			cin>>arr[i];
			arr[i]+=(1000000);
			hash[arr[i]]++;
		}
		
		int count = n;
		for(int i=0; i<n; i++)
		{

			if(hash[arr[i]]!=0){int a = hash[arr[i]];
				count+=(a*(a-1)/2);hash[arr[i]]=0;}
		}
		cout<<count<<endl;
	}
	return 0;
}
