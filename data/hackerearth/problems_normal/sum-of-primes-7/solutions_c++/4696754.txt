#include<bits/stdc++.h>
#define ll long long int
using namespace std;
bool prime[1000006];
ll sum[1000006];
void seive()
{
	ll i,j;
	prime[0]=prime[1]=1;
	for(i=2;i*i<=1000006;i++)
	{
		if(!prime[i])
		{
			j=2;
			while(j*i<=1000006)
			{
				prime[j*i]=1;
				j++;
			}
		}
	}
	sum[0]=0;
	for(i=1;i<=1000006;i++)
	{
		if(!prime[i])
		{
			sum[i]=sum[i-1]+i;
		}
		else
		{
			sum[i]=sum[i-1];
		}
	}
}

int main()
{
	seive();
	ll t,l,r,i,j;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&l,&r);
		printf("%lld\n",sum[r]-sum[l-1]);
	}
	return 0; 
}
