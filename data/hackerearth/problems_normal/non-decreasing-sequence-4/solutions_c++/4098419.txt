#include<bits/stdc++.h>
#define ll long int
using namespace std;
int main()
{
	ll t,n,len,a[123456],i,temp;
	for(cin>>t;t;t--)
	{
		cin>>n;
		memset(a,0,sizeof(a));
		ll top=0;
		a[0]=-1;
		for(i=0;i<n;i++)
		{
			cin>>temp;
			if(temp>a[top])
			{
				a[++top]=temp;
			}
			else
			{
				ll l=1,h=top,mid;
				while(l<=h)
				{
					mid=(l+h)/2;
					if(temp>a[mid])
					{
						l=mid+1;
					}
					else
					{
						h=mid-1;
					}
				}
				a[l]=temp;
			}
		}
		cin>>len;
		if(len<=top)
		{
			cout<<a[len]<<endl;
		}
		else
		{
			cout<<"-1"<<endl;
		}
	}
	return 0;
}
