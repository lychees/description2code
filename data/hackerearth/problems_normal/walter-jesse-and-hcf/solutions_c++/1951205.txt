#include <iostream>
#include<vector>
#include<algorithm>


using namespace std;


long long int gcd(long long int a,long long int b)
{
    long long int r;
    while(b!=0)
    {
        r = a%b;
        a = b;
        b = r;
    }
    return a;
}

int sto[200005];

int main()
{
    long long int D,X,Y,K;
    cin>>D;
    while(D--)
    {
        long long int k=0,i;
        cin>>X>>Y>>K;
        long long int num = gcd(X,Y);
        for(i=1;i*i<num;i++)
        {
            if(num%i==0)
            {
                sto[k] = i;
                k++;
                sto[k] = num/i;
                k++;
            }
        }
        if(i*i == num)
         {
            sto[k] = i;
            k++;
         }
        sort(sto,sto+k);
        if(K<=k)
        cout<<sto[k-K];
        else
        cout<<"No crymeth today";
        cout<<endl;
    }
    return 0;
}
