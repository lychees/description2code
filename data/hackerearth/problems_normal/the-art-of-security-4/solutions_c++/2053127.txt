#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;
long long int sum1, sum2;
long long int ans = 0;
int n;
long long int a[10];
void solve(int k) {
    if(k == n) {
        ans++;
        return;
    }
    
    if(sum1+a[k] <= sum2) {
        sum1 += a[k];
        solve(k+1);
        sum1 -= a[k];
    }
    sum2 += a[k];
    solve(k+1);
    sum2 -= a[k];
}

int main()
{
   ans = 0;
   long long int sum = 0;
   scanf("%d", &n);

   for(int i = 0; i < n; i++) {
       scanf("%lld", &a[i]);
       sum += a[i];
   }
   sort(a, a+n);
   sum1 = sum2 = 0;
   do {
       sum1 = sum2 = 0;
       solve(0);
   }while(next_permutation(a, a+n));
   printf("%lld %lld\n", ans, sum);
   if(ans > sum) printf("We will win!\n");
   else printf("Got no way out!\n");
   
   return 0;
}
