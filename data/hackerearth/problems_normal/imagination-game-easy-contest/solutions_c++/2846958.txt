#define SIZE 20

int d[SIZE];
int pos[SIZE];

int main(){


    int kase;
    scanf("%d",&kase);

    while ( kase-- ){

        int n;
        scanf("%d",&n);

        for (int i=0;i<n;++i){
            scanf("%d",&d[i]);
        }
        for (int i=0;i<n;++i){
            scanf("%d",&pos[i]);
        }

        int ans = 0;
        for (int i=0;i<n;++i){
            if ( d[i]-pos[i] >= 1 ){
                ans ^= d[i]-pos[i];
            }
            if ( pos[i] >= 2 ){
                ans ^= pos[i]-1;
            }
        }

        if ( ans >0 ){
            printf("First\n");
        }else{
            printf("Second\n");
        }

    }

    return 0;
}
