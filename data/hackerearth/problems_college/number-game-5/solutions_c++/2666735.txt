ll get(ll n)
{
	return (n*n)*(n+1)*(n+1)/4 + n*(n+1)*(2*n+1)/3 + n*(n+1)/2;
}
int main() 
{
	ios_base::sync_with_stdio(0);
	ll t;
	cin>>t;
	while(t--)
	{
		ll n;
		cin>>n;
		if(n<4)
		{
			cout<<"0\n";
			cout<<"Draw\n";
			continue;
		}
		ll l=0,r=1000000000;
		while(l<=r)
		{
			ll mid = (l+r)/2;
			if(get(mid)>n)
			{
				r = mid-1;
			}
			else
			{
				l = mid+1;
			}
		}
		cout<<l-1<<"\n";
		if((l-1)%2)
		{
			cout<<"Dhruv\n";
		}
		else
		{
			cout<<"Ketan\n";
		}
	}
	return 0;
}
