#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long sum,i,j,test,num;
    vector<long long int> arr(1000001);
    scanf("%lld",&test);
    while(test--)
    {
        sum=0;
        scanf("%lld",&num);
        for(i=0;i<num;i++)
            scanf("%lld",&arr[i]);
        for(i=0;i<num;i++)
        {
            for(j=i;j<num;j++)
            {
                if(arr[i]==arr[j])
                    sum++;
            }
        }
        cout<<sum<<endl;
    }
}
