#include <iostream>
using namespace std;
 
int main()
{
    long long int i,j,test,ra=0,size,changes=0;
    cin>>test;
    for(i=0;i<test;++i)
    {
    	changes=0;
    	cin>>size;
    	long long int a[size];
    	for(j=0;j<size;++j)
    		cin>>a[j];
    	for(j=0;j<size-1;++j)
    	{
    		if(a[j]<0)
    		{
    			a[j]=ra;
    			changes++;
    		}
    		if(a[j]>=a[j+1])
    		{
    			a[j]=ra;
    			changes++;
    		}
    		ra++;
       	}
    	cout<<changes<<endl;
    }
    return 0;
}
