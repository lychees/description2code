#include <bits/stdc++.h>
using namespace std;
const int N = 1e3 + 7;
string s[N];
int ans[N][N];

int main()
{
	ios::sync_with_stdio(false);

	int n, m;
	cin >> n >> m;
	for (int i=0; i<n; i++)
		cin >> s[i];
	for (int i=0; i<n; i++)
	{
		int cnt = 0;
		for (int j=0; j<m; j++)
			if (s[i][j] == 'B')
				cnt++;
		for (int j=0; j<m; j++)
			ans[i][j] += cnt;
	}
	for (int i=0; i<m; i++)
	{
		int cnt = 0;
		for (int j=0; j<n; j++)
			if (s[j][i] == 'B')
				cnt++;
		for (int j=0; j<n; j++)
			ans[j][i] += cnt;
	}
	for (int i=0; i<n; i++)
		for (int j=0; j<m; j++)
			if (s[i][j] == 'B')
				ans[i][j]++;
	int out = 0;
	for (int i=0; i<n; i++)
		for (int j=0; j<m; j++)
			out += ans[i][j] % 2;
	cout << out << "\n";
	return 0;
}
