#include <iostream>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    long long int tc;
    cin>>tc;
    while(tc--)
    {
        long long int n,b,coun=0,ans1=0;
        cin>>n>>b;
        long long int a[n];
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
        }
        for(int i=0;i<n;i++)
        {
            if(a[i] == b)
                coun++;
        }
        if(coun>0)
        {
            for(int i=0;i<n;i++)
            {
                if(a[i] > b)
                    ans1++;
            }
            cout<<"YES"<<" "<<ans1<<endl;
        }
        else
        {
            cout<<"NO"<<" "<<0<<endl;
        }
    }
}
