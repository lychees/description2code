s=str(raw_input());
q=int(raw_input());
while(q!=0):
    buff=str(raw_input());
    [l1,r1,l2,r2]=buff.split();
    if(s[int(l1)-1:int(r1)]==s[int(l2)-1:int(r2)]):
        print("Yes");
    else:
        print("No");
    q-=1;
