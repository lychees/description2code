#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
typedef long long lli;
int main()
{
    int n,a,i,j;
    cin>>n;
    lli max_size=1;
    vector<lli> num;
    vector<lli> sub_size(n);
    i=0;
    while(i<n)
    {
        cin>>a;
        num.push_back(a);
        i++;
    }
    sort(num.begin(),num.end());
    sub_size[n-1]=1;
    i=n-2;
    while(i>=0)
    {
        sub_size[i]=1;
        j=i+1;
        while(j<n){
            if(num[j]%num[i]==0)
                sub_size[i]=max(sub_size[i],sub_size[j]+1);
            j++;
        }
        max_size=max(max_size,sub_size[i]);
        i--;
    }
    if(max_size==1)
        max_size=-1;
    cout<<max_size<<endl;
    return  0;
}
