from sys import stdin
def get(n):
	ans = []
	while True:
		ans.append(n)
		n/=2
		if n==0:
			break
	ans.reverse()
	return ans
t = int(stdin.readline())
for _ in xrange(t):
	n,m = map(int,stdin.readline().split())
	a = get(n)
	b = get(m)
	l = len(a)
	if len(b) <l:
		l = len(b)
	prev = -1
	for i in xrange(l):
		if a[i]!=b[i]:
			break
		prev = a[i]
