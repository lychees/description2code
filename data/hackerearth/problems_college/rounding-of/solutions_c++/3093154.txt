#include<iostream>
using namespace std;
int main()
{
	int t,x;
	long n,b,res;
	cin>>t;
	while(t--)
	{
		cin>>n>>b;
		x=n%b;
		if(b%2==1)
		{
			if(x==b/2+1)
				res=n+b-x;
			else if(x<b/2+1)
				res=n-x;
			else
				res=n+b-x;
		}
		else
		{
			if(x<b/2)
				res=n-x;
			else
				res=n+b-x;
		}
		cout<<res<<endl;
	}
	return 0;
}
