#include<bits/stdc++.h>
using namespace std;

// recursive approch

int coin ( int a[] , int m , int n)
{
    int i , j;
    if (  n ==0)
        return 1;
    if ( n < 0)
        return 0;
    if (m<=0 && n>=1)
        return 0;
    else
        return coin( a, m-1, n ) + coin( a, m, n-a[m-1] );
}
int main()
{

    int m , n,t ;
    cin >> t;
    while(t--){
        cin >> n;
        int a[]= {1,3,5};
        cout << coin ( a,3,n) << endl;
    }
    return 0;

}
