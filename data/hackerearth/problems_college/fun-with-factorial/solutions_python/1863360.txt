def factorial(n):
    if n==0:
     return 1
    else:
     return n*factorial(n-1)
t=input()
while t>0:
    n=input()
    x=str(factorial(n))
    if len(x)==4:
        print "0"+x
    elif len(x)==3:
        print "00"+x
    elif len(x)==2:
        print "000"+x
    elif len(x)==1:
        print "0000"+x
    else:
     print x[len(x)-5:len(x)] 
    t=t-1
