#include<bits/stdc++.h>
using namespace std;

long long int mod=1000000007;
int main() {
    long long int i,j,k,t,a[3],ans;
    cin >> t;
    while(t--) {
    	cin >> a[0] >> a[1] >> a[2];
    
    	for(i=0;i<2;i++)
    		for(j=0;j<2;j++)
    			if(a[j] > a[j+1])
    				swap(a[j],a[j+1]);
        a[0]=a[0]%mod;
    	a[1]=(a[1]-1)%mod;
    	a[2]=(a[2]-2)%mod;
        ans=(a[0]*a[1])%mod;
        ans=(ans*a[2])%mod;
        cout << ans << endl;
    	
    }

	return 0;
}
