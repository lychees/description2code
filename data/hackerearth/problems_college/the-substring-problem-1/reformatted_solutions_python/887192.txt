import string
tc = int(raw_input())

for t in range(tc):
    n = int(raw_input())
    words = []
    for i in range(n):
        words.append(raw_input())

    letters = string.ascii_lowercase

    m = -1
    maxl = ""
    current = 0
    l = ""

    for letter in letters:
        current = 0
        for word in words:
            if letter in word:
                current += 1
        if (current > m):
            m = current
            l = letter
            maxl = letter
        elif (current == m and letter < maxl):
            l = letter

    print l
