const int MAXN = 1050;
int status[(MAXN/32)+10];

bool check(int n, int pos) { return (bool)(n & (1<<pos)); }
int SET(int n, int pos){ return n=n|(1<<pos);}

void sieve()
{
    int sqrtN=int (sqrt(MAXN));
    status[1>>5]=SET(status[1>>5],1&31);
    for(int j=4; j<=MAXN; j=j+2)
        status[j>>5]=SET(status[j>>5],j&31);
    for(int i=3; i<=sqrtN; i=i+2)
    {
        if(check(status[i>>5],i&31)==0)
        {
            for(int j=i*i; j<=MAXN; j= j + (i<<1))
            {
                status[j>>5]=SET(status[j>>5],j&31);
            }
        }
    }
}

LL arr[100004];

int main()
{
    int test;
    scanf("%d", &test);
    while(test--)
    {
        int n;
        LL k;
        scanf("%d %lld", &n, &k);
        string str = "";
        for(int i=0; i<n; i++)
        {
            scanf("%lld", &arr[i]);
            if(arr[i]<=k) str = str + 'T';
            else str = str + 'F';
        }

        LL maxx = 0;
        int sz = str.length();
        for(int i=0; i<sz; i++)
        {
            LL cnt = 0;
            if(str[i]=='T'){
                while(str[i]=='T'){
                    cnt = cnt + arr[i];
                    i++;
                }
                maxx = max(maxx, cnt);
            }
        }
        printf("%lld\n", maxx);
    }
}
