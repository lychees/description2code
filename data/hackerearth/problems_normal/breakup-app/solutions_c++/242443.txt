int search(string s, int len)
{ int l = len;
  int value =0;
  for(int i = 0; i < l-1; i++)
  {  if(isdigit(s[i]) && isdigit(s[i+1]))
      { value = (s[i]-'0')*10 + (s[i+1]-'0');
        i++;
        if(s[0] == 'G')
        {  arr[value]++;
           arr[value]++;
        }
        else if (s[0] == 'M')
        { arr[value]++;
        }
    //    cout<<value<<endl;
      }
  }
}
int main()
{ int max_arr = 0, max_arr_2 = 0;
  char s[1005];

    // getline(cin,s);
     lnth = strlen(s)-1;
   //  cout<<s<<" "<<lnth<<endl;
     search(s,lnth);
   }

   for(int i = 1 ; i < 30; i++)
   { if(arr[i] > arr[max_arr])
      { max_arr = i;
      }
   }
   for(int i = 1 ; i < 30; i++)
   { if(arr[i] == arr[max_arr] && max_arr != i)
      { max_arr_2 = i;
      }
