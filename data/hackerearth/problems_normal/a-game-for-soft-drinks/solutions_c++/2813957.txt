#include<bits/stdc++.h>
using namespace std;
int dp[105][105][105][2];
int mem[105][105][105][2];
int recurseit(int n1,int n2,int n3,int ch)
{
    if(n1==0&&n2==0&&n3==0)
    {
        return 1-ch;
    }
    else if(n1<0||n2<0||n3<0)
        return -1;
    if(mem[n1][n2][n3][ch])
        return dp[n1][n2][n3][ch];
    else
    {
        mem[n1][n2][n3][ch]=1;
        int val1,val2,val3,val4,val5,val6,val7;
        val1=recurseit(n1-1,n2,n3,1-ch);
        val2=recurseit(n1,n2-1,n3,1-ch);
        val3=recurseit(n1,n2,n3-1,1-ch);
        val4=recurseit(n1-1,n2-1,n3,1-ch);
        val5=recurseit(n1-1,n2,n3-1,1-ch);
        val6=recurseit(n1,n2-1,n3-1,1-ch);
        val7=recurseit(n1-1,n2-1,n3-1,1-ch);
        if(val1==ch || val2==ch || val3==ch || val4==ch || val5==ch || val6==ch || val7==ch)
            return dp[n1][n2][n3][ch]=ch;
        else
            return dp[n1][n2][n3][ch]=1-ch;
    }
}
int main() {
    int q;
    cin>>q;
    while(q--)
    {
        int n1,n2,n3;
        cin>>n1>>n2>>n3;
        if(!recurseit(n1,n2,n3,0))
        cout<<"Ashima"<<endl;
        else
        cout<<"Aishwarya"<<endl;
    }
    return 0;
}
