#include <iostream>
using namespace std;

int main()
{
    int l;
    long long n,m,sum;
    cin>>l;
    for(int k1=0;k1<l;k1++){
    	cin>>n>>m;
    		long long k=n/m;
    		sum = ((m*(m-1))/2ll)*k;
    		k = n%m;
    		sum+= ((k*(k+1))/2ll);
    		cout<<sum<<endl;
    }
    return 0;
}
