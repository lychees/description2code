#include <bits/stdc++.h>
using namespace std;
int a[110][110];
int visit[110][110];
long long int dp[110][110];
int xmove[4]={1,0,-1,0};
int ymove[4]={0,1,0,-1};
long long int ans;
void fun(int x,int y,int n)
{
	if(x==n-1&&y==n-1)
	ans++;
	
	

	for(int i=0;i<4;i++)
	{
		int xx=x+xmove[i];
		int yy=y+ymove[i];
		if(xx>=0&&xx<n&&yy>=0&&yy<n&&visit[xx][yy]==0&&a[xx][yy]==0)
		{
			visit[xx][yy]=1;
			fun(xx,yy,n);
			visit[xx][yy]=0;
			
		}
	}
return;
}
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			scanf("%d",&a[i][j]);
		}
	}
	ans=0;
	visit[0][0]=1;
	fun(0,0,n);

	cout<<ans;
	return 0;
}
