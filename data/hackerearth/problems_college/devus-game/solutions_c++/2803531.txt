#include <bits/stdc++.h>
#define ll long long int
#define size 100005
using namespace std;

int main() 
{
	ios::sync_with_stdio(false);
	ll t,n,a[size];
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		ll flag[size];
		memset(flag,0,sizeof(flag));
		ll tmp=a[0],f;
		flag[0]=1;
		while(1)
		{
			if(tmp==-1)
			{
				f=0;
				break;
			}
			else if(flag[tmp]==0)
			{
				flag[tmp]=1;
				tmp=a[tmp];
			}
			else
			{
				f=1;
				break;
			}
		}
		if(f==1)
		{
			cout<<"Lose"<<endl;
		}
		else
		{
			cout<<"Win"<<endl;
		}
	}
	return 0;
}
