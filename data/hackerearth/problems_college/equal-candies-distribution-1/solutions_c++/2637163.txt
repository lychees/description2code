#include<iostream>
#include<algorithm>
using namespace std;

int operations(int);
int main()
{
    ios::sync_with_stdio(false);
    int t;
    cin>>t;

    while(t--)
    {
        int n,count=0;
        cin>>n;

        int arr[n];
        for(int i=0;i<n;i++)
            cin>>arr[i];

        sort(arr,arr+n);
        for(int i=0;i<n-1;i++)
        {
            count+=operations(arr[i+1]-arr[i]);
            arr[i+1]=arr[i];
        }

        cout<<count<<"\n";
    }
}

int operations(int a)
{
    int count=0;

    count+=a/5;
    a=a-(a/5)*5;
    count+=a/2;
    a=a-(a/2)*2;
    count+=a;

    return count;
}
