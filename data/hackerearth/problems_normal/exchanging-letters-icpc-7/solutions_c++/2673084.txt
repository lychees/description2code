#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define all(a) a.begin(), a.end()
#define mod 1000000007
#define endl "\n"
#define si(a) scanf("%d",&a)
#define slld(a) scanf("%lld",&a)
#define slf(a) scanf("%lf",&a)

using namespace std;
typedef long long LL;
typedef vector<int> VI;
typedef vector< vector<int> > VVI;
typedef pair< LL, LL > PII;

const int N = 2010;
int mark[N];

map< pair<int,LL>, LL> M;
inline LL f(int idx, LL sum)
{
	if(sum < 0) return 0LL;
	if(idx == 0)
	{
		if(sum == 0) return 1LL;
		return 0LL;
	}
	pair<int,LL> state = mp(idx,sum);
	if(M.count(state)) return M[state];
	if(!mark[idx]) return f(idx-1,sum);
	LL ret = f(idx-1,sum-(idx+1));
	ret = ret + f(idx-1,sum-(idx-1));
	if(ret >= mod) ret -= mod;
	return M[state] = ret;
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int n,k,t,idx; cin>>n>>k>>t;
	for(int i=0; i<k; i++)
	{
		cin>>idx;
		mark[idx] = 1;
	}
	cout<<f(n,t)<<endl;
	return 0;
}
