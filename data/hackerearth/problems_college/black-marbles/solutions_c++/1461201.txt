#include <bits/stdc++.h>

using namespace std;
vector<string>v;
int dx[]={-2,-2,2,2};
int dy[]={-2,2,-2,2};
int m,n;
bool check(int x,int y){
	if(x>=0 && x<m&& y>=0 && y<n) return true;
	return false;
}
bool proc() {
	int cnt = 0;
	cin>>m>>n;
	if(m==0) return false ;
	string s;
	for(int i=0;i<m;i++) {
		cin>>s;
		v.push_back(s);
	}
	for(int i=0;i<m;i++) {
		for(int j=0;j<v[i].length();j++){
			if(v[i][j]=='F'){
				bool ok = true;
				for(int k=0;k<4;k++) {
					int ddx = dx[k]+i;
					int ddy = dy[k]+j;
					
					if(check(ddx,ddy)){
						if(v[ddx][ddy]=='P'){
						}
						else if(v[ddx][ddy]=='G'){
							if(k==0){
								if(v[i-1][j-1]=='P'){
								}else ok = false;
							}
							else if(k==1) {
								if(v[i-1][j+1]=='P'){
								}else ok = false;
							}
							else if(k==2) {
								if(v[i+1][j-1]=='P'){
								}else ok = false;
							}
							else {
								if(v[i+1][j+1]=='P'){
								}else ok = false;
							}
						}
						else if(v[ddx][ddy]=='F'){
						}
						else ok = false;
					}
					
					
				}
				if(ok){
						v[i][j] = 'G';
						cnt++;
				}
				else v[i][j] = 'P';
			}
		}
	}
	//for(int i=0;i<m;i++) cout<<v[i]<<endl;
	cout<<cnt<<endl;
	v.clear();
	return true;
}

int main()
{
	int t;
	while(proc());
    return 0;
}
