#include<iostream>
#include<set>
#include<vector>
#include<algorithm>
using namespace std;

vector <int> vn;
int main()
{
	long T, n, k, x;
	long long a;
	cin>>T;
	while(T--)
	{
	 	cin>>n>>k;
		for(int i = n; i; i--)
		{
			cin>>x; vn.push_back(x);
		}
		sort(vn.rbegin(), vn.rend());
		x = min(n,k);
		a = 0;
        for(int i = 0;i < x; i++) 
            a += vn[i];
        
		cout<<a<<endl;
		vn.clear();
	}
	return 0;
}
