#include<bits/stdc++.h>
using namespace std;




pair<int,int> a[101];



int cal(int n)
{
    int c=1;
    int E=a[1].first;
    int i=1;
    for (int j=2;j<=n;j++)
    {
      if (a[j].second>=a[i].first)
      {
          //printf ("%d ", j);
          c++;
          i = j;
      }
    }
    return c;
}

int _cal(int n)
{
    int c=1;
    int E=a[1].first;
    int i=2;
    
    while(i<=n)
    {
        while(a[i].second<E)
      	{
            i++;
            if(i>n)
            break;
        }
        
        if(i>n)
        break;
        
        c++;
       	E=a[i].first;
        
    }
    return c;
}
int main()
{
    int n,S,E;
    cin>>n;
    int A,B,C,D;
    for(int i=1;i<=n;i++)
    {
        scanf("%d:%d-%d:%d",&A,&B,&C,&D);
        //cout<<A<<" "<<B<<" "<<C<<" "<<D<<endl;
        S=(A*60)+B;
        E=(C*60)+D;
        a[i].second=S; // start
        a[i].first=E;  // end
    }

    sort(a+1,a+n+1);

    int ans=_cal(n);

    //cout<<ans<<endl;

    if(ans==n)
        cout<<"Who needs a moderator?"<<endl;
    else
        cout<<"Will need a moderator!"<<endl;
    return 0;
}
