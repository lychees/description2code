#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	int t,aCost,bCost,minCost,out,i;
	string inp;
	
	cin>>t;
	while(t--){
		out=0;
		cin>>inp;
		cin>>aCost;
		cin>>bCost;
		minCost=aCost<bCost?aCost:bCost;
		
		for(i=0;i<inp.length()/2;i++){
			if(inp[i]=='/' && inp[inp.length()-1-i]=='/'){
				out+=2*minCost;
			}
			else if(inp[i]=='/' || inp[inp.length()-1-i]=='/'){
				if(inp[i]=='a' || inp[inp.length()-1-i]=='a'){
					out+=aCost;
				}
				else if(inp[i]=='b' || inp[inp.length()-1-i]=='b'){
					out+=bCost;
				}
			}
			else if(inp[i]!=inp[inp.length()-1-i]){
				cout<<"-1\n";
				break;
                                cout<<i<<" ";
			}
		}
		
                if(inp[i]!='/' && inp[inp.length()-1-i]!='/'){
                    if(inp[i]!=inp[inp.length()-1-i]){
                            continue;
                    }
		}
                
		cout<<out<<endl;
		
	}
	
    return 0;
}
