void shift(char * st,int i,int j)
{
char a=st[j];
while(i<j)
{


st[j]=st[j-1];
j--;
}
st[i]=a;
}
void combination(char *c,int i,int n)
{
if(i==n)
cout<<c<<" ";
char st[10];
for(int j=i;j<=n;j++)
{
strcpy(st,c);
shift(st,i,j);
combination(st,i+1,n);
}
}
int main()
{
char st[10]={'1','2','3','4','5','6','7','8','9','\0'};
int t;
cin>>t;
int n;
char c[10];
while(t--)
{
cin>>n;
strcpy(c,st);
c[n]='\0';
combination(c,0,n-1);
cout<<"\n";
}
}
