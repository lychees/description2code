#include <iostream>
using namespace std;

long long prob(long long x, long long y)
{
	if(y == 0)
		return x;
	else 
		return prob(y, x%y);
}

int main()
{
    int t;
    scanf("%d", &t);
    long long n, k, i, var;
    long long var1, var2, ans;
    while(t--) {
    	scanf("%lld", &n);
    	scanf("%lld", &k);
    	long long arr[n];
    	var = 0;
    	for(i = 0; i < n; i++) {
    		scanf("%lld", &arr[i]);
    	}
    	
    	for(i = 0; i < n; i++) {
    		if(arr[i] == k) {
    			var++;
    		}
    	}
    	
    	var1 = var*(var-1)*(var-2);
    	var2 = n*(n-1)*(n-2);
    	ans = prob(var1, var2);
    	
    var1 = var1/ans;
    var2 = var2/ans;
    printf("%lld/%lld\n",var1,var2);
    }
    
    return 0;
}
