#include <bits/stdc++.h>
using namespace std;

int main()
{
	long n,m;
	long a;
	long long profit = 0;
    priority_queue<long> q;
    cin>>n>>m;
    for(int i=0;i<n;i++){
    	cin>>a;
    	q.push(a);
    }
    for(int i=0;i<m;i++){
    	profit += q.top();
    	a=q.top()-1;
    	q.pop();
    	q.push(a);
    }
    cout<<endl<<profit;
    return 0;
}
