#include <iostream>
using namespace std;

int main()
{
    int T;
    
    cin >> T;
    
    for (int i = 0; i < T; i++) {
    	long N;
    	
    	cin >> N;
    	
    	long temp = N - 1;
    	
    	long highestOne = 0;
    	
    	while (temp > 0) {
    		temp >>= 1;
    		highestOne++;
    	}
    	
    	long max = (1L << highestOne) - 1;
    	
    	long count = max - 2 * (max - N + 1) + 1;
    	
    	cout << max << " " << count << endl;
    }
}
