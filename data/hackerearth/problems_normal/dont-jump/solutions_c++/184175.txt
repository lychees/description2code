#include <iostream>
#include <stdio.h>
#include <string.h>

using namespace std;

char str[1000000];

int main()
{
    
         int n;
         char str[1000];
         
    while(scanf("%d",&n)!=EOF)
      {  
         int A[100000];
         int B[100001]={0};                     
         if(getchar()==EOF)
            break;                        
         int j=0;
         gets(str); 
         //puts(str);
         int len=strlen(str);
         int m=0;
         for(int i=0;i<len;i++)
           {
              if(str[i]==' ')
                {
                   if(m!=0)
                     A[j++]=m;
                     
                   m=0;
                   while(str[i]==' '&&i<len) 
                     i++;
                   i--;           
                }
              else
                {
                   m=m*10+(str[i]-48);
                }                             
           }
           
          if(m!=0)
            A[j++]=m; 
         
         int t=0;
         int no=0;
         int flag=0;
         int flag1=0;
         for(int k=0;k<j;k++)
           {
               flag1=1;  
              // printf("%d\n",A[k]);  
               int temp=no;    
               if(B[A[k]]==1||A[k]>n)
                 {
                    flag=1;
                    break;
                 }      
                      
               while(no<A[k])
                 {
                    str[t++]='C';
                    no++;
                    if(no==A[k])
                     {
                       str[t++]='K';
                       B[no]=1;
                     }  
                 }
               
               
               while(temp>A[k])
                 {
                    if(B[temp]!=1)
                      {
                         str[t++]='K';
                         B[temp]=1;
                      }
                    temp--;  
                     if(temp==A[k])
                      {
                         str[t++]='K';
                         B[temp]=1;           
                      }    
                 }  
                       
                 
                
             }
             
            str[t]='\0';    
            
            if(flag==0&&flag1==1)
              printf("%s\n",str);
            else if(flag1==1)
              printf("-1\n");
              
            else
             printf("\n");  
              
      }
              
//system("pause");
return 0;              
}
