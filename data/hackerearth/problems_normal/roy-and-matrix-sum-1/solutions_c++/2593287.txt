#include <iostream>
#include<math.h>

using namespace std;

int main()
{
	int tc;
   long size;
   long sum=0;
   cin>>tc;
   while(tc--)
   {
      cin>>size;
      sum=0;
      for(long i=1;i<=size-1;i++)
      {
      		sum+=(i)*(i+1)/2;
      }
      cout<<sum*2<<endl;
   }
	//getch();
   return 0;
}
