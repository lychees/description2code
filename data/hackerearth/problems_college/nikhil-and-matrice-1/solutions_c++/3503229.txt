#include <bits/stdc++.h>

using namespace std;

#define lb      lower_bound
#define ub      upper_bound
#define pb      push_back
#define pf      push_front
#define mp      make_pair
#define gcd     __gcd
#define all(x)  x.begin(), x.end()
#define S(x)    scanf("%d", &x)
#define Sl(x)   scanf("%lld", &x)
#define P(x)    printf("%d", x)
#define Pl(x)   printf("%lld", x)
#define PS(x)   printf("%s", x)
#define MOD     1000000007LL

typedef long long       ll;
typedef vector<int>     vi;
typedef vector<vi>      vvi;
typedef pair<int, int>  ii;
typedef vector<ii>      vii;

int main()
{
	int t, n;
	S(t);
	while(t--) {
		int a[20][20];
		memset(a, 0, sizeof(a));
		S(n);
		int n1, i = 0, j = 0, m;
		m = n;
		n1 = n;
		m -= 1, n -= 1;
		int u = 1;
		while(i <= m && j <= n) {
			for(int k = j; k <= n; k++) {
				a[i][k] = u;
				u++;
			}
			i++;
			if(i > m) break;
			for(int k = i; k <= m; k++) {
				a[k][n] = u;
				u++;
			}	
			n = n-1;
			if(j > n) break;
			for(int k = n; k >= j; k--) {
				a[m][k] = u;
				u++;
			}
			m = m-1;
			if(i > m) break;
			for(int k = m; k >= i; k--) {
				a[k][j] = u;
				u++;
			}
			j++;
			if(j > n) break;
		}
		for(int i = 0; i < n1; i++) {
			for(int j = 0; j < n1; j++) {
				cout << a[i][j];
			}
			cout << endl;
		}
		if(t > 0) {
			cout << endl;
		}
	}
	return 0;
}
