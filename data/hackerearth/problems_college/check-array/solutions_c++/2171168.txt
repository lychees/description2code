#include<stdio.h>
int main()
{
    long long t,n,sum,sum1,c,a[100000],i,count,count1;
scanf("%lld",&t);
while(t--)
{
    scanf("%lld",&n);
    sum1=0;
    sum=0;
    count=0;
    count1=0;
   for(i=0;i<n;i++)
   {
        scanf("%lld",&a[i]);
        sum1=sum1+a[i];
    }
      sum1=sum1-a[0];
      c=0;
        while(c<n)
          {
            if(sum1==sum)
             {
                count++;
                break;
             }
            else if(sum1<sum)
            {
                count1++;
                break;
            }
            else
            {
                sum=sum+a[c];
                 c++;
                sum1=sum1-a[c];
            }
          }
if(count!=0)
printf("YES\n");
if((count==0) || (count1!=0))
printf("NO\n");
}
return 0;
}
