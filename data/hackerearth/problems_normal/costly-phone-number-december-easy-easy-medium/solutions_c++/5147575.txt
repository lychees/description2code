#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   		int c[10];int x;
   		for(int i=0;i<10;++i)
   			cin>>c[i];
   		string s;
   		cin>>x;
   		cin>>s;
   		for(int k=0;k<10;++k)
   		{
   			for(int i=0;i<10;++i)
   			{
   				for(int j=0;j<10;++j)
   				{
   					c[(i+j)%10] = min(c[(i+j)%10],c[i]+c[j]);
   				}
   			}
   		
   			
   		}
   		int ans=0;
   		for(int i=0;i<s.length();++i)
			ans+=c[s[i]-'0'];
		cout<<ans<<endl;
   }
    return 0;
}
