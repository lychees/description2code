#include<bits/stdc++.h>
#define pii pair<long long,long long>
#define fi first
#define se second
#define mp make_pair
#define vpi vector< pii >
#define pb push_back
#define ll long long int
#define mod 1000000007
#define gcd(a,b) __gcd(a,b)
#define sf(n) scanf("%lld",&(n))
#define pf(n) printf("%lld\n",(n))
#define setbits(n) __builtin_popcount((n))
#define pq priority_queue
using namespace std;
string arr[105];
bool visited[26];
int oudeg[26];
int arr2[26][26];
int indeg[26];
vector<int>v[26];
map<int,string>m;
int co=0;
int dep[26];
void dfs(int node,int depth)
{
    for(int i=0;i<v[node].size();i++)
    {
        dfs(v[node][i],depth+1);
    }
    dep[node]=max(dep[node],depth);
}
int main()
{
    int n;
    cin>>n;
    memset(oudeg,0,sizeof(oudeg));
    memset(indeg,0,sizeof(indeg));
    memset(visited,0,sizeof(visited));
    memset(arr2,0,sizeof(arr2));
   for(int i=0;i<n;i++)
    {
        cin>>arr[i];
        for(int j=0;j<arr[i].length();j++)
        {
            visited[arr[i][j]-'a']=true;
        }
        if(i==0)
            continue;
        int l=min(arr[i].length(),arr[i-1].length());
        int j;
        int flag=0;
        for(j=0;j<l;j++)
        {
            if(arr[i-1][j]!=arr[i][j])
            {
                flag=1;
                break;
            }
        }
        if(flag==1 && arr2[arr[i-1][j]-'a'][arr[i][j]-'a']==0)
        {
            v[arr[i-1][j]-'a'].pb(arr[i][j]-'a');
            indeg[arr[i][j]-'a']++;
            oudeg[arr[i-1][j]-'a']++;
            arr2[arr[i-1][j]-'a'][arr[i][j]-'a']=1;
        }


    }
    fill(dep,dep+26,1);
    for(int i=0;i<26;i++)
    {
        if(indeg[i]==0 && oudeg[i]!=0)
            dfs(i,1);
    }
    for(int i=0;i<26;i++)
    {
        if(visited[i]==true)
        {
            m[dep[i]]+=(i+'a');
        }
    }
    for(map<int,string>::iterator it=m.begin();it!=m.end();it++)
    {
        sort((it->second).begin(),(it->second).end());
        cout<<(it->second)<<"\n";
    }

}
