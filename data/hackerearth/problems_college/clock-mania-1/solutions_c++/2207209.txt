#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    int h,m;
    while(t--){
    	cin>>h>>m;
    	double H=h*30+m/2;
    	double M=m*6;
    	double a=abs(H-M);
    	cout<<min(a,360-a)<<endl;
    }
    return 0;
}
