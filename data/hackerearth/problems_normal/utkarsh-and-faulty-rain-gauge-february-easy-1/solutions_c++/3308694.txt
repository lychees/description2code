#include <iostream>
using namespace std;

int main()
{
	double avgt=1;
	long long t,r,n;
	cin>>t;
	while(t--)
	{
		cin>>n>>r;
		double avg=0.0;
		long long x = (n*(n+1))/2;
		x = x*x; x=0;
		for(long long i=0;i<=n;i++)
		{	avg += ((1.0*i)/r)*(i*i); x += i*i;}
		avg = avg/x;
		// cout<<avg<<endl;
		// avg = avg/(n+1);
		avgt = avgt*avg;
	}
	printf("%.4lf",avgt);
    return 0;
}
