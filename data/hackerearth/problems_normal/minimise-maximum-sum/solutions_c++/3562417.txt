#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,k,n,T,ans;
    int arr[200005];
    scanf("%d",&T);
    while(T--){
    	scanf("%d",&n);
    	for(i=0;i<2*n;i++)
    		scanf("%d",&arr[i]);
    	sort(arr,arr+2*n);
    	ans=0;
    	for(i=0;i<n;i++){
    		k=arr[i]+arr[2*n-i-1];
    		if(k>ans)
    			ans=k;
    	}
    	printf("%d\n",ans);
    }
    return 0;
}
