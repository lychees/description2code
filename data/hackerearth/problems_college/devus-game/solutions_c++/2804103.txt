#include<bits/stdc++.h>
using namespace std;
int main()
{
    //freopen("input.txt","r",stdin);
    int t,n,i,j,flag=0,c=0;
    scanf("%d",&t);
    while(t--)
    {   flag=0;
        scanf("%d",&n);
         int A[n];
    	for(i=0;i<n;i++)
    	scanf("%d",&A[i]);
    	i=0;
    	c=0;
    	int cpos=0;
    	while(c<=n){
                if(A[cpos]==-1)
                {
                    flag=1;break;
                }
                cpos=A[cpos];
                c++;

    	}

    (flag==1)?puts("Win"):puts("Lose");
}
return 0;
}
