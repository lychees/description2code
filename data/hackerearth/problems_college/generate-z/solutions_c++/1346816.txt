#include <cstdio>

int main(){
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
		printf("*");
	printf("\n");
	for (int i = n - 1; i > 1; i--){
		for (int j = 1; j < i; j++)
			printf(" ");
		printf("*\n");
	}
	for (int i = 0; i < n; i++)
		printf("*");
	printf("\n");
	return 0;
}
