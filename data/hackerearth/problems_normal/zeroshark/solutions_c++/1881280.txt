#include<cstdio>
#include<iostream>
#include<algorithm>
#include<vector>
#include<map>
#include<set>
using namespace std;
#define FR(i,x,n) for(int i=x;i<=n;i++)
#define FRR(i,n,x) for(int i=n;i>=x;i--)
#define s(x) scanf("%d",&x)

#define ll long long

int mod=1000000007;
long long dp[100004][2][2];

void pre()
{
	dp[1][0][0]=1; dp[1][0][1]=1;
	
	for(int i=2;i<100003;i++)
	{
		dp[i][1][0] = dp[i-1][1][1] + dp[i-1][0][0] ; dp[i][1][0]%=mod;
		dp[i][1][1] = dp[i-1][1][0] + dp[i-1][1][1]; dp[i][1][1]%=mod;
		dp[i][0][0] = dp[i-1][0][1];										dp[i][0][0]%=mod;
		dp[i][0][1] = dp[i-1][0][0] + dp[i-1][0][1];			dp[i][0][1]%=mod;
	}
}
int main()
{
	int ntc; s(ntc);
	pre();
	while(ntc--)
	{
		int n; s(n);
		int ans=dp[n][1][0]+dp[n][1][1]; if(ans>=mod) ans-=mod;
		printf("%d\n",ans);
	}
}
