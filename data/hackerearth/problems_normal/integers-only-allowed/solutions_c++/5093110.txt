#include <stdio.h>

using namespace std;
long long int p[1000002][2];
long long int gcd(long long int a,long long int b)
{
    if(a<0)
        a*=-1;
    if(b<0)
        b*=-1;
    if(b==0)
        return a;
    if(a==0)
        return b;
    else
        return gcd(b,a%b);
}
int main()
{
    int n,i;
    long long int ans,area=0ll,b=0ll;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%lld %lld",&p[i][0],&p[i][1]);
    }
    p[n][0]=p[0][0];
    p[n][1]=p[0][1];

    for(i=0;i<n;i++)
    {
//printf("ix:%lld iy:%lld i+1x:%lld i+1y:%lld\n",p[i][0],p[i][1],p[i+1][0],p[i+1][1]);
        area+=p[i+1][0]*p[i][1] - p[i][0]*p[i+1][1];
        b+=gcd(p[i+1][0]-p[i][0],p[i+1][1]-p[i][1]);
  //       printf("i:%d area: %lld  b:%lld\n",i,area,b);
    }
  //  printf("area: %lld b:%lld\n",area/2,b);
    if(area<0)
        area*=-1;
    ans= area +2- b;
    printf("%lld\n",ans/2);
    return 0;
}
