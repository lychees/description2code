#include <iostream>
#define MAX 100000
using namespace std;

int a[MAX];

int compareTo (const void* a, const void* b)
{
   return -(*(int*)a - *(int*)b);
}

int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin >> t;
	while(t--){
		int len;
		cin >> len;
		
		for(int i=0; i<len; i++)
			cin >> a[i];
			
		qsort(a, len, sizeof(int), compareTo);
		
		for(int i=0; i<len; i++)
			cout << a[i] << " ";
		cout << endl;
	}
	return 0;
}
