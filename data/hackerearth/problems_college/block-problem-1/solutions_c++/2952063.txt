#include <iostream>
#include <algorithm>
#include <cstring>
#include <vector>
#include<stdio.h>
#define MAX 1010
using namespace std;




void mod(long long &x){
    x += 1000000007;
    x = x % 1000000007;
}


long long A[MAX], B[MAX][MAX], C[MAX][MAX];

long long solving(int x, int y){
    if(C[x][y]) return C[x][y];
    C[x][1] = 1;
    for(int i = 2; i <= y; i++){
        C[x][i] = B[x][i];
        long long temp = 0, temp2 ;
        for(int j = 1; j < i; j++){
            temp2 = B[x][j] * C[x][i-j];
            mod(temp2);
            temp += temp2;
            mod(temp);
        }
        C[x][i] -= temp;
        mod(C[x][i]);
    }
    return C[x][y];
}

int main(){
    A[1] = 1;
    A[0] = 1;
    for(int i = 2; i < MAX; i++){
        A[i] = A[i-1];
        if(i - 2 >= 0) A[i] += A[i-2];
        mod(A[i]);
        if(i - 3 >= 0) A[i] += A[i-3];
        mod(A[i]);
        if(i - 4 >= 0) A[i] += A[i-4];
        mod(A[i]);
    }
    for(int i = 1; i < MAX; i++){
        for(int j = 1; j < MAX; j++){
            C[i][j] = 0;
            if(i == 1) B[i][j] = A[j];
            else {
                B[i][j] = B[i-1][j] * A[j];
                mod(B[i][j]);
            }
        }
    }
    int T, x, y;
    scanf("%d", &T);
    for(int i =0 ; i<T;i++)
    {//while(T--){
        scanf("%d%d", &x, &y);
        cout << solving(x, y) << endl;
    }
    return 0;
}
