#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int jack[3]={1,2,3};
int dpk[100003];
int dpl[100003];
int jill[2]={1,2};
int target;
int fun(int sum)
{
    if(sum==target)
    return 1;
    if(sum > target)
    return 0;
    if(dpk[sum]!=-1)
    return dpk[sum];
    int ans=0;
    for(int i=0;i<3;i++)
    {
        ans+=fun(sum+jack[i]);
       
    }
     
    return dpk[sum]=ans;
}

int funl(int sum)
{
    if(sum==target)
    {
      
    return 1;
    }
    if(sum > target)
    return 0;
    if(dpl[sum]!=-1)
    return dpl[sum];
    int ans=0;
    for(int i=0;i<2;i++)
    {
        ans+=funl(sum+jill[i]);
    }
    return dpl[sum]=ans;
}
int main()
{

    cin >> target;
    if(target < 0)
    {
    cout << "ERROR" << endl;
    
    	return 0;
    }
    memset(dpk,-1,sizeof(dpk));
    memset(dpl,-1,sizeof(dpl));
    cout << "Jack-"<<fun(0) << endl;
    cout << "Jill-"<<funl(0) << endl;
    
    
    return 0;
}
