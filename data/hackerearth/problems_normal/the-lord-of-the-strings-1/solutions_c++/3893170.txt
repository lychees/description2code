#include <iostream>
using namespace std;
#include<string.h>

int main()
{
    int t,i,j,k=0,len,count;
    
    cin>>t;
    while(t--)
    {
    	char a[32],b[32];
    	k=0;
    	cin>>a;
    	len=strlen(a);
    	count=len;
    	for(i=0;i<len;i++)
    	{
    		for(j=i+1;j<len;j++)
    		{
    			if(a[i]=='!')
    			continue;
    		   if(a[i]==a[j])
    		   {
    		   a[j]='!';
    		   count--;
    		   }
    		}
    	}
    	for(i=0;i<len;i++)
    	{
    		if(a[i]=='!')
    		continue;
    		else
    		{
    			b[k]=a[i];
    			k++;
    		}
    	}
    	for(k=0;k<count;k++)
    	{
    	cout<<b[k];
    	}
    	cout<<"\n";
    }
    return 0;
}
