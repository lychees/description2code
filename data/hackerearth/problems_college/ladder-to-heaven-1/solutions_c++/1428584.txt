#include <bits/stdc++.h>
#define ll long long int
#define s(a) scanf("%lld",&a)

using namespace std;

ll fibo[901]={0};
ll sum[901]={0};

int main()
{
    ll i,j,l,r,mid,n,t;
    fibo[1]=1;
    fibo[2]=2;
    for(i=3;i<=90;i++) {
        fibo[i]=fibo[i-1]+fibo[i-2];
        //cout<<fibo[i]<<endl;
    }
    sum[1]=0;
    sum[2]=2;
    for(i=3;i<=90;i++) {
        if(fibo[i]%2==0) sum[i]=sum[i-1]+fibo[i];
        else sum[i]=sum[i-1];
    }
    s(t);
    while(t--) {
        s(n);
        for(i=1;i<=90;i++) {
            if(fibo[i]>n) break;
        }
        cout<<sum[i-1]<<endl;
    }
    return 0;
}
