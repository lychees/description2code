#include <iostream>
#include <cstring>
using namespace std;

int main()
{
   char str[100];
   cin>>str;
   int len = strlen(str);
   int count =0,x=1;
   for(int i=1;i<len;i++)
   	{
   		if(str[i]==str[i-1])x++;
   		else
   		{
   		count=count+(x*(x+1)/2);
   		x=1;
   		}
   	}
   	count=count+(x*(x+1)/2);
   	cout<<count;
    return 0;
}
