#include <iostream>
#include <vector>

using namespace std;

int main()
{
	ios_base::sync_with_stdio( false );
	int t, n1, i, n2, ct;
	
	cin >> t;
	
	while( t-- ) {
		
		string s1, s2;
		cin >> s1;
		cin >> s2;
		n1 = s1.length();
		n2 = s2.length();
		ct = 0;
		
		for( i = 0;i < min( n1, n2 );i++ )
			if( s1[ i ] == s2[ i ] )
				ct++;
		
		cout << ct << endl;
	}
	
	return 0;
}
