#include<bits/stdc++.h>
#define bs 1000000007
const int N = 1200050;
using namespace std;
int n, k;
long long ar[N];
long long pref[N];
long long s;
long long deriv;
int main(){
	ios_base::sync_with_stdio(0);
	//cin.tie(0);
	cin >> n >> k;
	for (int i = 1; i <= n; i++)
	{
		cin >> ar[i];
	}
	for (int i = 1; i <= n; i++)
	{
		pref[i] = pref[i - 1] + ar[i];
		pref[i] %= bs;
	}
	for (int i = 1; i <= k; i++)
	{
		s += 1ll * i*i%bs*ar[i] % bs;
	}
	s %= bs;
	cout << s;
	for (int i = 2; i <= k; i++)
	{
		deriv += (2 * i - 1)*ar[i] % bs;
	}
	for (int i = 2; i <= n - k + 1; i++)
	{
		s = s - ar[i - 1] + bs;
		s %= bs;
		s -= deriv;
		while (s < 0)
			s += bs;
		
		deriv = deriv - 3 * ar[i] + bs;
		deriv %= bs;
		deriv -= 2 * (pref[i + k - 1] - pref[i]) % bs;
		while (deriv < 0)
			deriv += bs;
		s += ar[i + k - 1] * k%bs*k%bs;
		s %= bs;
		deriv += (2 * k + 1) * 1ll * ar[i + k - 1] % bs;
		deriv %= bs;
		cout << " " << s;
	}
	cout << endl;
	return 0;
}
