 #include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,k,m;
		cin>>n>>k>>m;
		set<string> s;
		set<string>::iterator it;
		char v[1001][1000];

		for(int i=0;i<n;i++)
		{
			cin>>v[i];
			string b="";
			for(int j=0;j<m;j++)
				b+=v[i][j];
			//cout<<b<<endl;
			s.insert(b);
		}

		int count=1;
		string ans;
		for (it=s.begin(); it!=s.end(); ++it)
		{
			//cout<<*it<<endl;
			if(count==k)
			{
				ans += *it;
				break;
			}
			count++;
		}

		//cout<<ans<<" "<<count<<endl;

		for(int i=0;i<n;i++)
		{
			int flag=0;
			for(int j=0;j<m;j++)
			{
				if(ans[j]!=v[i][j])
				{
					flag=1;
					break;
				}
			}
			if(flag==0)
			{
				cout<<v[i]<<endl;
				break;
			}
		}

		// int index = stoi (str_dec,&s[m]);
		// cout<<v[index]<<endl;

	}
}
