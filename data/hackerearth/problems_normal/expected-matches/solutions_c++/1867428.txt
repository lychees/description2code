#include <stdio.h>

int n, m;

double dp[1005][1005];
int used[1005][1005];
double p[1005][1005];

double solve(int a, int b) {
    if (a == n || b == m) {
        return 0;
    }
    if (used[a][b] != -1) {
        return dp[a][b];
    }

    double aux = 1.0 + p[a][b] * solve(a, b + 1) + (1.0 - p[a][b]) * solve(a + 1, b);
    dp[a][b] = aux;
    used[a][b] = 1;
    return aux;
}

int main(void) {
    int t;
    double r;

    scanf(" %d", &t);

    while(t--) {
        scanf(" %d %d", &n, &m);
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                scanf(" %lf", &p[i][j]);
                used[i][j] = -1;
            }
        }

        r = solve(0, 0);
        printf("%.6lf\n", r);
    }
    return 0;
}
