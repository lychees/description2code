typedef long long int ll;
typedef pair<int, int > pii;

#define MOD 1000000007
#define pb push_back

vector<int> primes(1000005, 1);
vector<int> sums(1000005, 0);
vector<int> finals;
void sieve()
{
	finals.reserve(100000);
	primes[0] = 0, primes[1] = 0;
	primes[2] = 1;
	for(int i=2;i<1000005;i++)
	{
		if(primes[i] == 1)
		{
			for(int j=2*i;j<1000005;j+=i)
			{
				sums[j] += i;
				primes[j] = 0;
			}
			sums[i] = i;
			finals.push_back(i);
		}
	}
	//cout<<finals.size()<<"\n";
}

void solve()
{
	ll n, k;
	cin>>n>>k;
	int cnt = 0;
	for(int i = 2;i<=n;i++)
	{
		if(sums[i] == k)
			cnt++;
	}
	cout<<cnt<<"\n";
}

int main()
{
	std::ios::sync_with_stdio(false);
	//freopen ("C:\\Users\\Ritesh\\Desktop\\Coding\\Codeforces\\input.txt","r",stdin);
	int t;
	//cin>>t;
	sieve();
	t = 1;
	while(t--)
	{
		solve();
	}
	return 0;
}
