#include <iostream>
using namespace std;
int main()
{

	int t;
	cin >> t;
	for (int i= 1;i<=t;i++)
	{
		int n, m, k;
		cin >> n >> m >> k;
		int ans = (n - 1) * (m / k);
		while (m-k >=k)
		{
			m -= k;
			ans++;
		}
		if (m == k)
			ans=ans+ k;
		else
			ans=ans+ k + 1;
		cout <<"Case #"<<i<< ": "<<ans<<endl;
	}
	return 0;
}
