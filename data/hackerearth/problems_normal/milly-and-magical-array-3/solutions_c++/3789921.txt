#include <iostream>
using namespace std;
#define ll long long
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	ll N,X;
    	cin>>N>>X;
    	ll arr[N];
    	for(int i=0;i<N;i++)
    	cin>>arr[i];
    	
    	ll operation = 0;
    	ll deff = 0;
    	ll times = 0;
    	for(int i=1;i<N;i++)
    	{
    		if(arr[i-1]>=arr[i])
    		{
    			deff = arr[i-1]-arr[i];
    			if(deff == 0)
    			times = 1;
    			else
    			times = deff/X+1;
    			
    			arr[i] = arr[i]+times*X;
    			operation = operation+times;
    		}
    	}
    	cout<<operation<<endl;
    }
    return 0;
}
