t = int(raw_input())
while (t > 0):
    a = raw_input().split(' ')
    n = int(a[0])
    m = int(a[1])
    x = raw_input().split(' ')
    x = map(long, x)
    l1 = x[0:n]
    l1 = set(l1)
    l2 = x[n:n+m]
    # print l1
    for i in l2:
        if i not in l1:
            print "NO"
            l1.add(i)
        else:
            print "YES"
    t = t-1
