#include <set>
#include <cstdio>
#include <vector>
#include <string.h>
#include <algorithm>
#define max 1000002
using namespace std;


int main() 
{
    set<int> q;
    set<int>::iterator it;
    vector<vector<int> >v(1000002);;
    int *vis, *de;
    vis = (int*)malloc( (sizeof(int))*max);
    de = (int*)malloc( (sizeof(int))*max);
    memset(vis , 0 , sizeof(vis));
    memset(de , 0 , sizeof(de));
    int i , j , k ,  x , n ,nn, y;
    scanf("%d" , &n);
    nn = n;
    //puts("tte");
    while(n--)
    {
        scanf("%d %d" , &k , &x);
        vis[x] =1 ;
        for( j = 1 ; j < k ; j++)
        {
            scanf("%d" , &y);
            vis[y] = 1;
            de[y]++;
            v[x].push_back(y);
            x = y;
        }
    }
    for( i = 1 ; i < max ; i++)
        if( vis[i] && de[i] == 0)
    {
        q.insert(i);   
    }
    while( !q.empty())
    {
        it = q.begin();
        printf("%d " , *it);
        x = *it;
        q.erase(it);
        for( i = 0 ; i < v[x].size() ; i++)
        {
            de[ v[x][i]]--;
            if( de[ v[x][i] ] == 0)
                q.insert( v[x][i]);
        }
    }
    free(vis);
    free(de);
      
    return 0;
}
