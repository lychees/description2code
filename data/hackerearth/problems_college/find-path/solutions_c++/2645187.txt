#include <iostream>
using namespace std;

int main()
{
    int t1, t2, n[20], i, c;
    cin>>t1;
    for(; t1>0; t1--)
    {
    	c=0;
    	cin>>t2;
    	{
    		for(i=0; i<t2; i++)
    		{
    			cin>>n[i];
    		}
    		for(i=0; i<t2-1; i++)
    		{
    			if(n[i]-n[i+1]>=0)
    				c++;
    		}
    		
    	}
    	cout<<c<<"\n";
    }
    
}
