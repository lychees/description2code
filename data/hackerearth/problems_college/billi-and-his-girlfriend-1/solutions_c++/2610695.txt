#include<bits/stdc++.h>
#define ll long long

ll mod;



using namespace std;

int power(ll x, unsigned ll y)
{
    if( y == 0)
        return 1;
    else if (y%2 == 0)
    {
        ll k;
        k=power(x, y/2) %mod;


        return (k*k)%mod;
    }
    else
        return (x*power(x, y-1))%mod;

}

int sum(ll num)
{
    int s=0;
    while(num!=0)
    {
        s+=(num%10);
        num/=10;
    }
    return s;
}

int main()
{

    ll t,a,b,c;
    cin>>t;
    while(t--)
    {
        cin>>a>>b>>c;
        mod=pow(10,c);

        cout<<sum(power(a,b))<<"\n";


    }
}
