typedef long long ll;
typedef unsigned long long ull;
typedef pair<int,char> pii;
bool comp(pii lhs, pii rhs) {
    if (lhs.first < rhs.first) return true;
    else if (lhs.first == rhs.first) return lhs.second > rhs.second;
    else return false;
    }
int main() {
//    read("in.txt");
//rite("out.txt");
    int tc;
    int i;
    scanf("%d",&tc);
    ll ans,temp;
    vector<pii > line;
    int n,p,q,a,b,c;
    while(tc--) {
        scanf("%d %d %d %d",&n,&a,&b,&c);
        line.clear();
        for(i = 0 ; i < n ; ++i) {
            scanf("%d %d",&p,&q);
            line.push_back(make_pair(p-1,'-'));
            line.push_back(make_pair(q,'+'));
            }
        sort(line.begin(),line.end(),comp);
        ans = n;
        ans *= c;
        temp = ans;
        i = line.size()-1;
        while (i >= 0) {

            while (i >= 0 ) {
                if (line[i].first < 0) break;
                if (line[i].second == '+') temp = temp - c + b;
                else temp = temp - b + a;
//                cout << line[i].first << " " << temp << endl;
                i -= 1;
                if (line[i].first != line[i+1].first) break;
                }
//            cout << " " << temp << endl;
            ans = max(temp,ans);
            if (line[i].first < 0) break;
            }
        cout << ans << endl;
        }
    return 0;
    }
