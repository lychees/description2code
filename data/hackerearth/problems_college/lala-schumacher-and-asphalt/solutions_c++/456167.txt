#include<stdio.h>
#include<iostream>
#include<cmath>
using namespace std;
int main()
{
long long t,n,i,j,k,min,a,b,t1,t2,temp;;
scanf("%lld",&t);
while(t--)
{
scanf("%lld",&n);
long long x[n],y[n];
for(i=0;i<n;i++){scanf("%lld",&x[i]);}
for(j=0;j<n;j++) {scanf("%lld",&y[j]);}
scanf("%lld%lld",&a,&b);
scanf("%lld%lld",&t1,&t2);

min=abs(a)*t1+abs(b)*t1;
for(i=0;i<n;i++)
{
temp=(abs(x[i])+abs(y[i]))*t1+(abs(a-x[i])+abs(b-y[i]))*t2;
if(min>temp){min=temp;}
}

cout<<min<<"\n";
}

return 0;
}
