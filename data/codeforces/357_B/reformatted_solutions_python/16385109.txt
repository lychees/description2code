def R():
    return map(int, raw_input().split())


n, m = R()
COLOR = [None for i in xrange(n+1)]
for i in xrange(m):
    inp = R()
    colors = list(set([1, 2, 3]) - set([COLOR[dancer] for dancer in inp]))
    for dancer in inp:
        if COLOR[dancer] is None:
            COLOR[dancer] = colors.pop(0)
print " ".join(map(str, COLOR[1:]))
