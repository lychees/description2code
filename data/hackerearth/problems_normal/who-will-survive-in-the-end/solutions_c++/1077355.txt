#include <stdio.h>
#include <math.h>

long long int josephus(long long int );

int main()
{
    int N,i;
    long long int res;
    long long int no_players[1000];
    scanf("%d",&N);
    for(i=0;i<N;i++)
    {
        scanf("%lld",&no_players[i]);
    }
    for(i=0;i<N;i++)
    {
        res=josephus(no_players[i]);
        printf("%lld\n",res);
    }
    return 0;
}

long long int josephus(long long int n)
{
    long long int m;
    long long int l;
    long long int res;
    for(m=1;m<=n/2;m++)
    {
        if(pow(2,m)>n)
            break;
    }
    m=m-1;
    l=n-pow(2,m);
    res=2*l+1;
    return res;
}
