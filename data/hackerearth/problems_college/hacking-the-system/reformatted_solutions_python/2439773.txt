import sys
comp = raw_input()
stu = raw_input()
comp = [int(i) for i in comp]
stu = [int(i) for i in stu]
non_match = 0
iter = 0
while iter < len(comp):
    if comp[iter] != stu[iter]:
        non_match += 1
        if non_match > 1:
            print "loses"
            sys.exit(0)
    iter += 1
print "wins"
