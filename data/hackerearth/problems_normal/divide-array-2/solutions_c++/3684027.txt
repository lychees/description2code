#include <bits/stdc++.h>
using namespace std;
int main()
{
	long int x,p,m=1,q,n;
	
	ios_base::sync_with_stdio(false);
	
	cin>>n;
	
	long int arr[n];
	
	for(x=0;x<n;x++)
	
	{
		cin>>p;
		arr[x]=p;
	}
	cin>>q;
	while(q--)
	
	{
		cin>>p;
		m=m*p;
	}
	
	for(x=0;x<n;x++)
	
	{
		if(m!=0)
		cout<<arr[x]/m<<" ";
		else
		cout<<"0"<<" ";
	}
	
	return 0;
}
