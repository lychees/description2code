#include<bits/stdc++.h>
#define gc getchar
#define pc putchar
#define pii pair<int,int>
#define MOD 1000000007
using namespace std;
typedef long long int LL;
template <class T>
inline void rd(T &x)
{
	x=0;
	bool check=false;
	register int c;
	do{
		c=gc();
		if(c=='-')
		check=true;
	}while(c<48 || c>57);
	for(;c<48 || c>57;c=gc());
	for(;c>47 && c<58;c=gc()){
		x=(x<<1)+(x<<3)+c-48;
	}
	if(check)
	x=-x;
}
template <class T>
inline void pd(T x)
{
	char c[21];
	bool check=false;
	int k=-1;
	if(x<0)
	check=true,x=-x;
	do
	{
		c[++k]=x%10+48;
		x/=10;
	}while(x);
	if(check)
	c[++k]='-';
	while(k>=0)
	pc(c[k--]);
	pc('\n');
}

bool cmp(int  a,int b)
{
	return a>b;
}
int cnt[24*60*60];
int gcd(int a,int b)
{
	if(b==0)
	return a;
	return gcd(b,a%b);
}
int main()
{
	
	int i,j,k,l,m,n,t,cntt=0,total,h,s;
	for(i=0;i<24;i++)
	{
		for(j=0;j<60;j++)
		{
			m=gcd(i,j);
			for(k=0;k<60;k++)
			{
				n=gcd(m,k);
				
				if(n==1)
				{
					cntt++;
				}
				cnt[(i*60*60)+(j*60)+k]=cntt;
			}
		}
	}
	
	rd(t);
	while(t--)
	{
		rd(h);
		rd(m);
		rd(s);
		total=h*60*60;
		total+=m*60;
		total+=s;
		total=24*60*60-total;
		i=cntt-cnt[(h*60*60)+(m*60)+(s-1)];
		total-=i;
		
		n=1;
		if(total!=0 && i!=0)
		n=gcd(total,i);
		
		cout<<total/n<<":"<<i/n<<endl;
	}
	
}
