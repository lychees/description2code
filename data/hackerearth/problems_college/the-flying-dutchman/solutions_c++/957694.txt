#include<stdio.h>
int main()
{
    long long int ntc,i,n,a1,a2,b1,b2;
    scanf("%lld",&ntc);


    while(ntc--)
    {
    long long int arr[100009];
    long long int A[100009]={};
    long long int B[100009]={};
    scanf("%lld",&n);

    for(i=n-1;i>=0;i--)
        scanf("%lld",&arr[i]);

    A[0]=arr[0];
    B[0]=0;
    A[1]=arr[0]+arr[1];
    B[1]=0;
    for(i=2;i<n;i++)
    {
        a1=arr[i]+B[i-1];
        b1=A[i-1];
        a2=arr[i]+arr[i-1]+B[i-2];
        b2=A[i-2];

        if(a1>b1&&a2>b2)
        {
            if(a1-b1>a2-b2)
                A[i]=a1,B[i]=b1;
            else
                A[i]=a2,B[i]=b2;
        }
        else if(a1>b1)
            A[i]=a1,B[i]=b1;
        else if(a2>b2)
            A[i]=a2,B[i]=b2;
        else
        {
            if(b1-a1>b2-a2)
                A[i]=a2,B[i]=b2;
            else
                A[i]=a1,B[i]=b1;
        }
    }
    if(A[n-1]>B[n-1])
        printf("LOSE\n");
    else if(A[n-1]<B[n-1])
        printf("WIN\n");
        else
        printf("DRAW\n");
    }
}
