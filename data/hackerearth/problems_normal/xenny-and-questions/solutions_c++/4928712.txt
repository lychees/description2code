#include<iostream>
#include <bits/stdc++.h>
#include<algorithm>
using namespace std;
int main()
{
	int t,n,m,i;
	long long int sum;
	ios_base::sync_with_stdio(false);
    cin.tie(NULL);
	cin>>t;
	while(t--)
	{
		
		sum=0;
		cin>>n>>m;
		int a[n];
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a,a+n);
		for(i=n-1;i>=0;i--)
		{
			if(m==0)
			{
				break;
			}
			else
			{
				sum = sum+a[i];
				m--;
			}
		}
		cout<<sum<<endl;
	}
}
