#include<bits/stdc++.h>
using namespace std;
#define ll long long
vector<ll> dp(20000002);
int main()
{
   std::ios_base::sync_with_stdio(false);
    ll t,n;
    cin>>t;
   dp[0]=0;
   dp[1]=0;
        dp[2]=1;
        for(ll i=3;i<=20000001;i++)
        {
            if(i%2==0&&i%3!=0)
                dp[i]=1+min(dp[i/2],dp[i-1]);
            else if(i%3==0&&i%2!=0)
                dp[i]=1+min(dp[i/3],dp[i-1]);
            else if(i%2==0&&i%3==0)
                dp[i]=1+min(dp[i/2],min(dp[i/3],dp[i-1]));
            else
                dp[i]=1+dp[i-1];
        }
  




while(t--)
    {
    	int n;
        cin>>n;
        cout<<dp[n]<<"\n";
    }

   return 0;
}
