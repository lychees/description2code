#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {

        int n;
        cin>>n;
        long int a[n+1];
        int i,ind=0;
        cin>>a[0];
        long int min=a[0];

        for(i=1;i<n;i++)
           {
               cin>>a[i];
               if(min>a[i])
                {
                min=a[i];
                ind=i;
                }
           }
           long int sum=0;
           for(i=0;i<n;i++)
           {
               if(i!=ind)
                sum+=a[i]+min;
           }
           cout<<sum<<endl;

    }
    return 0;

}
