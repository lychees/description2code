#include <iostream>
using namespace std;
 
int search(int x,int a[],int beg,int end){
	if(beg<=end){
		int mid=((beg+end)/2);
		if (x==a[mid]) return mid;
		if (x>a[mid])  return search(x,a,mid+1,end);
		return search(x,a,beg,mid-1);
	
	}	
	else
			return -1;
	
}
 
int main()
{
   int q,x,mat[1000][1000],n,m;
   cin>>n>>m;
   for(int i=0;i<n;i++){
    	for(int j=0;j<m;j++){
    		cin>>(mat[i][j]);
    	}
    }
   
    cin>>(q);
    while(q--){
    	cin>>(x);
    	for(int i=0;i<n;i++){
    		int c=search(x,mat[i],0,m-1);
    		if(c!=-1){
    			cout<<i<<" "<<c<<endl;
    			break;
    		}
    		else if(i==n-1)
    			cout<<"-1 -1\n";
    			
    	}
    	
    	
    }
        
    return 0;
}
