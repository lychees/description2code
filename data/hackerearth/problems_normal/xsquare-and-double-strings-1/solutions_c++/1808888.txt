#include <iostream>
using namespace std;

int main()
{
    int t;cin>>t;
    while(t--) {
    		getchar();
    		string s;
    		cin>>s;
    		int f[256]={0};
    		int n=s.size();
    		for(int i=0;i<n;i++) {
    				f[s[i]]++;
    		}
    		bool flag=false;
    		for(int i=0;i<256;i++) {
    			if(f[i]>1) {
    				flag=true;
    					break;
    			}
    		}
    		if(!flag)
    			cout<<"No"<<endl;
    		else
    			cout<<"Yes"<<endl;
    }
    return 0;
}
