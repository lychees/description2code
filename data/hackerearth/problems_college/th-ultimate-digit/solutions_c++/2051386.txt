#include<math.h>
#include <iostream>
#include<string.h>
using namespace std;
int main() 
{int n,x,len,len1,y,z,b;
char arr[1000],array[10000];
cin>>n;
while(n--)
{		cin>>arr;
		cin>>array;
		len=strlen(arr);
		len1=strlen(array);
		x=arr[len-1]-48;
		y=array[len1-1]-48;
		z=array[len1-2]-48;
		if(len1==1&&y==0)
		cout<<'1'<<endl;
		else
		{
			if(x==0||x==1||x==5||x==6)
				{
					switch(x)
					{
						case 0:cout<<'0'<<endl;break;
						case 1:cout<<'1'<<endl;break;
						case 5:cout<<'5'<<endl;break;
						case 6:cout<<'6'<<endl;break;
					}
				}
					else
					{
						if(x==9||x==4)
						{
							if(y%2==0)
							{
								if(x==4)
								cout<<'6'<<endl;
								else
								cout<<'1'<<endl;
							}
							else
							{
								cout<<x<<endl;
							}
						}
						else
						{
							b=(z*10+y)%4;
							if(b!=0)
							{
								cout<<((int)(pow(x,b)))%10<<endl;
							}
							else
							cout<<((int)(pow(x,4)))%10<<endl;
						}		
			    }	
			
		}
		
	}	
	return 0;
}
