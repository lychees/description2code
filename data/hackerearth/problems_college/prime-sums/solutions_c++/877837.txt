#include <iostream>
using namespace std;
int main()
{
    int i,j,n;
    long long sum = 0;
    cin>>n;
    bool a[4000000];
    for(i=2;i<n;i++)
    	a[i]=1;
    for(i=2; i*i <n;i++)
    {
 	for(j=i; i*j< n; j++)
    	a[i*j]=0;
    }
 for(int i=2; i<n;i++)
	{
		if(a[i])
		sum+=i;
    }
    cout<<sum;
    return 0;
}
