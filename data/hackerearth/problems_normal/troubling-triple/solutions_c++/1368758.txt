#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

#define MAX 2005

int main()
{
    int n;
    long long k;
    int a[MAX];
    scanf("%d%lld", &n, &k);
    for (int i = 0; i < n; i++) {
        scanf("%lld", a + i);
    }
    sort(a, a+n);
    long long count = 0;
    for (int i = 0; i < n; i++) {
        for (int j = i+1; j < n; j++) {
            long long largest = k/(a[i]*a[j]);
            count += upper_bound(a+j+1, a+n, largest) - (a + j + 1);
        }
    }
    printf("%lld\n", count);
}
