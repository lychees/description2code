#include <iostream>
using namespace std;
int main() 
{	int t;
	cin>>t;
	while(t--)
	{	int n, minimum =0;
		string s;
		cin>>n>>s;
		for(int i =0;i<n;++i)
		{	if(s[i] == '0' || s[i] == '2' || s[i] == '3' || s[i] == '5' || s[i] == '7')
				continue;
			else
			{	if(s[i] == '1'||s[i] == '4'||s[i] == '6' || s[i] == '8')
					minimum +=1;
				else if(s[i] == '9')
					minimum +=2;
			}
		}
		// cout<<s<<" "<<minimum<<endl;
		cout<<minimum<<endl;
	}
return 0;
}
