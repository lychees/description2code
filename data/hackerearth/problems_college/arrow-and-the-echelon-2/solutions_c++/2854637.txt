#include <cstdio>
#include <iostream>

using namespace std;

void Print(char a,int n)
{
    for(int i = 0; i < n; ++i)
        cout << a;
}

int main()
{
    int T,n;
    cin >> T;
    while(T--)
    {
        cin >> n;
        int R3 = n % 3;
        int R5 = 0;

        if( R3 == 0 )
            R3 = n;
        else if( R3 == 1)
        {
            R5 = 10;
            R3 = n - R5;
        }
        else if(R3 == 2)
        {
            R5 = 5;
            R3 = n - R5;
        }

        R3 = n - R5;
        if(R3 < 0)
        {
            cout << "-1";
        }
        else
        {
            Print('5',R3); Print('3',R5);
        }
        cout << endl;
    }
    return 0;
}
