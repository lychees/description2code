#include <iostream>
using namespace std;

int main()
{
	int t,i,j;
	char a[100000];
	cin>>t;
	while(t--){
		int oc=0,c=0;
		cin>>a;
		i=0;
	
		while(a[i]){
			if(a[i]=='O')
				oc++;
				i++;
			}
			i=0;
		while(a[i]){
		
				j=i+1;
			if(a[i]=='X'){
				j=i-1;
				c=0;
				while(j>=0 && a[j]!='*'){
					if(a[j]=='O')
						c++;
						
					j--;
					
				}
				j=i+1;
					
				while(a[j] && a[j]!='*'){
				
					if(a[j]=='O')
						c++;
						j++;
				}
			oc=oc-c;
			}
			i=j;
		}
		
	  cout << oc << endl;	
	}
  
    return 0;
}
