#include<bits/stdc++.h>
#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long dp[100001];
    long long int i,j,k,l;
     i=l=1;k=2;
    while(i<100001)
    {
    	for(j=1;j<=k&&i<100001;j++)
    	dp[i++]=l;
    	l++;
    	k++;
    }
    int tc;
    cin>>tc;
    while(tc--)
    {
    	int n;
    	cin>>n;
    	cout<<dp[n]<<endl;
    }
    return 0;
}
