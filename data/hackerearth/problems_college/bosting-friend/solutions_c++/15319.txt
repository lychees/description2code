#include <iostream>
#include<stdio.h>
using namespace std;
#define FOR(i,a,b) for (int i = (a); i < (b); i++)
long long gcd(long long a, long long b){
if(b==0)return a;
return gcd(b,a%b);
}
int main(){
int tc;
scanf("%d",&tc);
FOR(tcc,1,tc+1){
long long N,p1,p2;
cin>>N>>p1>>p2;
printf("#%d:",tcc);
long long q1 =100,q2=100;
if(p2==0&&p1!=0){
printf("Broken\n");
} else if(p2==100&&p1!=100){
printf("Broken\n");
} else {
long long gc = gcd(p1,q1);
q1/=gc;
if(q1<=N)printf("Possible\n");
else printf("Broken\n");
}
}
return 0;
}
