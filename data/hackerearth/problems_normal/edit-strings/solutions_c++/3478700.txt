#include <iostream>

using namespace std;

int main()
{
   int n,m,p,coun=0;
   char c;
   cin>>n>>m;
   char a[n];
   for(int i=0;i<n;i++)
    cin>>a[i];
    while(m--)
        {cin>>p;
        cin>>c;
        a[p-1]=c;
for(int j=0;j<n;j++)
{
    if(a[j]=='.' && a[j+1]=='.')
        coun++;

}

cout<<coun<<endl;
coun=0;
        }



    return 0;
}
