#include <iostream>
#include<cmath>
#include <iomanip>
using namespace std;

int main()
{

  int N, i, j , k;
  double Max[3000], val, newVal, Y[3000], F[3000], X[3000];
  cin>>N;

  for(i=0; i<N; i++){
    cin>>X[i];
    cin>>Y[i];
    cin>>F[i];
  }
  Max[0] = F[0];
  for (j = 1; j < N; j++) {
      val = F[0] + F[j] - sqrt((X[j]-X[0])*(X[j]-X[0]) + (Y[j] - Y[0])*(Y[j] - Y[0]));
      for (k=1; k < j; k++) {
        if (val < ( newVal = (Max[k] + F[j] - sqrt((X[j]-X[k])*(X[j]-X[k]) + (Y[j] - Y[k])*(Y[j] - Y[k]))))) {
          val = newVal;
        }
      }
      Max[j] = val;
      //cout<<Max[j]<<" ";

  }
    cout<< setprecision(6) << fixed <<Max[N-1];
	return 0;
}
