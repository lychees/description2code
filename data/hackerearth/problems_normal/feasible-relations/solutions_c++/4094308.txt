#include<bits/stdc++.h>
using namespace std;
#define ull long long int
ull arr[1000005];
ull sizee[1000005];
ull root(ull i)
{
    while(i!=arr[i])
    {
        arr[i]=arr[arr[i]];
        i=arr[i];
    }
    return i;
}
void union_nodes(ull a, ull b)
{
    ull ra=root(a);
    ull rb=root(b);
    if(sizee[ra]>sizee[rb]){
        arr[rb]=ra;
        sizee[ra]+=sizee[rb];
    }
    else{
        arr[ra]=rb;
        sizee[rb]+=sizee[ra];
    }
    return;
}
bool connected(ull a,ull b)
{
    if(root(a)==root(b))
        return true;
    return false;
}
int main()
{
    ios::sync_with_stdio(false);
    ull t,n,k,x,y,i,flag;
    string str;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        for(i=0;i<=n;i++)
        {
            arr[i]=i;
            sizee[i]=1;
        }
        flag=0;
        while(k--)
        {
            cin>>x>>str>>y;
            if(str=="=" && !flag)
                union_nodes(x,y);
            else if(!flag){
               if(connected(x,y))
                    flag=1;
            }
        }
        if(flag)
            cout<<"NO\n";
        else
            cout<<"YES\n";
    }
    return 0;
}
