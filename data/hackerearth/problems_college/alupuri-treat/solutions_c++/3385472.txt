#include <iostream>
using namespace std;

int main()
{
	
	long long int mat[200][200],m, i,j, sum=0;
	cin >> m;
	for(i=0; i<m; i++)
	{
		for(j=0; j<m; j++)
		{
			cin >> mat[i][j];
			if(i==j)
				sum += mat[i][j];
			if(i==m-1-j)
				sum -= mat[i][j];
		}
	}
	sum = abs(sum);
	cout << sum << endl;
    return 0;
}
