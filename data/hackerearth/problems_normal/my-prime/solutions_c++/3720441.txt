#include <iostream>
using namespace std;

int main()
{
    int n,i=0,j,cnt;
    scanf("%d",&n);
    int a[n];
    while(i<n)
    {
    	cin>>a[i++];
    }
    for(i=0;i<n;i++)
    {
    	cnt=0;
    	for (j=0;j<n;j++)
    	{
    	if ((a[i]%a[j])==0)
    		cnt++;
    	}
    	if (cnt<=1)
    		cout<<a[i]<<' ';
    }
    
    return 0;
}
