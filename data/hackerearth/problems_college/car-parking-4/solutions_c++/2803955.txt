#include <bits/stdc++.h>
#define si(x) scanf("%d", &x)
#define slli(x) scanf("%lld", &x)
#define sf(x) scanf("%f", &x)
#define slf(x) scanf("%lf", &x)
#define sc(x) scanf("%c", &x)
#define pb push_back
#define mp make_pair

using namespace std;

typedef pair <int, int> pi;
typedef vector <int> vi;
typedef vector <pi> vpi;

bool valid(int i, string s) {
	int len = s.length();
	if( s[i] != '-' || 
		(i + 1 < len && 
			(s[i + 1] == 'S' ||
			s[i + 1] == 'B')) || 
		(i - 1 >= 0 && s[i - 1] == 'S') || 
		(i + 2 < len && s[i + 2] == 'B'))
		return false;
	return true;
}

int main () {
	int t; si(t);
	while(t--) {
		string s;
		cin>>s;
		int len = s.length(), cnt = 0;
		for(int i = 0; i < len; i++) {
			if(valid(i, s))
				cnt++;
		}
		printf("%d\n", cnt);
	}
	return 0;
}
