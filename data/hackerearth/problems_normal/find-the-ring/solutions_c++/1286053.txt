#include <iostream>
using namespace std;

int main()
{
    int t,a,b;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	if(b==0)
    	cout<<a<<"\n";
    	else if(a==0&&b%2==0)
    	cout<<"0\n";
    	else if(a==0&&b%2!=0)
    	cout<<"1\n";
    	else if(a==1&&b%2==0)
    	cout<<"1\n";
    	else if(a==1&&b%2!=0)
    	cout<<"0\n";
    	else if(a==2&&b%2==0)
    	cout<<"0\n";
    	else
    	cout<<"1\n";
    }
    return 0;
}
