a = [0]*1000001
p = [1]*1000001
for i in range(2, 1000001):
    if p[i]:
        for j in range(i, 1000001, i):
            p[j] = 0
            x = j
            while x % i == 0:
                a[j] += 1
                x /= i
    a[i] += a[i-1]
for t in range(input()):
    x, y = map(int, raw_input().split())
    print a[y]-a[x-1]
