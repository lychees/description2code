#include <bits/stdc++.h>
#include <bits/stdc++.h>
#define si(n) scanf("%d",&n)
#define sl(n) scanf("%lld",&n)
#define ss(n) scanf("%s",n)
#define fr(i, n) for(int i = 0; i < n; i++)
#define REP(i, a, b) for (int i = int(a); i <= int(b); i++)
#define REN(i, a, b) for (int i = int(a); i >= int(b); i--)
#define ms(i, n) memset(i, n, sizeof(i))
#define INF 1002000000
#define MOD 1000000007
#define IND 1000000000000
typedef long long LL;
using namespace std;
int main ()
{
	//freopen ("input.txt","r",stdin);
	//freopen ("output.txt","w",stdout);
	LL dp[23][23], ans;
    int n;si(n);
    char s[100];
    while(n--) {
        ss(s);
        int sz = strlen(s);
        fr(i, 22)
            fr(j, 22)
                dp[i][j] = -1;
        fr(i, sz) {
            dp[i][i] = s[i] - '0';
            if(s[i] != '0') {
            REP(j, i + 1, sz - 1) {
                dp[i][j] = dp[i][j - 1] * 10 + s[j] - '0';
            }
            }
        }
        ans = -2;
        REP(j, 1, sz - 3) {
            REP(k, j + 1, sz - 2) {
                REP(l, k + 1, sz - 1) {
                    if(dp[0][j - 1] <= IND && dp[j][k - 1] <= IND&& dp[k][l - 1] <= IND && dp[l][sz - 1] <= IND ) {
                    if(dp[0][j - 1] != -1 && dp[j][k - 1] != -1 && dp[k][l - 1] != -1 && dp[l][sz - 1] != -1 ) {
                        ans = max(ans, dp[0][j - 1] + dp[j][k - 1] + dp[k][l - 1] + dp[l][sz - 1]);
                    }
                    }
                }
            }
        }
        if(ans == -2)
            printf("unlucky\n");
        else printf("%lld\n", ans);
    }
	return 0;
}
