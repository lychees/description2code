#include <bits/stdc++.h>
using namespace std;
#define ll long long 
#define fi first 
#define se second
#define pb push_back  
#define mp make_pair
int main()
{
  
    int t;
    ll n, cnt, temp ;
    cin>>t;
    while( t-- )
    {
    	cnt=0;
    	cin>>n;                      // When N is 1 we simply know that
    	if(n==1)					 //  the answer is 0	
    		cout<<0;
    	else
    	{
    	  n++;
    	  temp=n;
    	  // counting the number of bits in the
    	  //binary representation of the number.
    	  while(temp)				
    	  {
    	  	temp>>=1;    // right-shifting the binary value of N, which is same as 
    	  	cnt++;		 // dividing by 2.	
    	  }
    	 cout<<cnt-2;    // We print the count-2 for the reason given in editorial.
    	}
    	cout<<endl;
    }
	return 0;
}
