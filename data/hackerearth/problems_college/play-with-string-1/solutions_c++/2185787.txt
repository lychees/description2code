#include <iostream>
#include <string>
#include <math.h>
using namespace std;

int calcString(string s){
	int i,k=0;
	char *r = new char[s.length()];
	
	
	for(i=s.length()-1; i>=0 ; --i)
		r[k++]=s[i];
		
	r[k]='\0';
	
		
	for(i=1;i<s.length();i++){
	   if( abs(s[i]-s[i-1]) != abs(r[i]-r[i-1]))
	    		return 0;				
	}
	

	
return 1;	
}

int main()
{
	int i=0,t;
	string s;

	cin>>t;
    while(i<t){
    	cin>>s;
    	if(calcString(s))
    		cout<<"Funny\n";
    	else
    		cout<<"Not Funny\n";
    	
    	++i;    	
    }
    
    return 0;
}
