#include <bits/stdc++.h>
using namespace std;
#define PI 3.14159265359
//(log(2?n)/2+n(logn?loge))/log10]+1,
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,i,digit,z=0;
    	cin>>n;
    	for(i=5;n/i>=1;i=i*5)
    	{
	z=z+n/i;                  	
	}

	digit=(floor((log(2*PI*n)/2 + n*(log(n) - 1) + 1/(12*n))/log(10)) + 1);//no of didgits in n fact
	cout<<digit-z<<"\n";
    }
    return 0;
}
