t = int(raw_input())

for _ in range(t):
	
	d = {}
	alpha = raw_input()
	word = raw_input()
	
	for a in alpha:
		d[a] = 0
		
	for b in word:
		d[b] += 1
		
	res = ''
	for c in alpha:
		if d[c] > 0:
			while d[c] > 0:
				res += c
				d[c] -= 1
		
	print res
