#include <iostream>
using namespace std;
#define LL long long int 
int N = 0 , X , T , A[1111] ;
LL DP[11111];
int main()
{
   // cout << "Hello World!" << endl;
	int i = 1 ;
	while(i*i*i <= 9261){
		A[++N] = i*i*i ;
		i ++ ;
	}
	cin >> T ;
	while(T --){
		cin >> X ;
		DP[0] = 1 ; 
		for(int i=1;i<=X;i++) DP[i] = 0 ;
		for(i=1;i<=N;i++){
			for(int j=1;j<=X;j++){
				if(j-A[i] >= 0)
					DP[j] += DP[j-A[i]] ;
			}
		}
		cout << DP[X] << endl ;
	}
    return 0;
}
