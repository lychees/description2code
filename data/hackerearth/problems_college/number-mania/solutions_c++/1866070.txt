#include <iostream>
#include <algorithm>
#include <vector>
#include <cmath>
using namespace std;
int multiply(int x,int res[],int res_size)
{
	int carry=0;
	for(int i=0;i<res_size;i++)
	{
		int prod=res[i]*x+carry;
		res[i]=prod%10;
		carry=prod/10;
	}
	while(carry)
	{
		res[res_size]=carry%10;
		carry=carry/10;
		res_size++;
	}
	return res_size;
}
void factorial(int n)
{
	int res[1000];
	res[0]=1;
	int res_size=1;
	for(int x=2;x<=n;x++)
	{
		res_size=multiply(x,res,res_size);
	}
	for(int i=res_size-1;i>=0;i--)
	{
		cout<<res[i];
	}
	cout<<""<<endl;
}
bool isPrime(int n)
{
	if(n==1)
	return false;
	for(int i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
		return false;
	}
	return true;
}
int main()
{
	int t,p=2;
	cin>>t;
	vector<int> A;
	while(A.size()<=100)
	{
		if(isPrime(p))
		{
			A.push_back(p);
		}
		p++;
	}
	int i=0;
	while(A[i]<=t)
	{
		factorial(A[i]);
		i++;
	}
}
