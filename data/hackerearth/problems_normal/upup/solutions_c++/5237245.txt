#include <iostream>
#include<vector>
#include<string>
using namespace std;

int main()
{
    string s,temp;
    getline(cin,s);
    vector<string> v;
    for(int i=0;i<s.size();i++)
    {
    	if(s[i]==' ')
    	{
    		v.push_back(temp);
    		temp.clear();
    		
    	}
    	else
    	temp=temp+s[i];
    }
    v.push_back(temp);
    for(int i=0;i<v.size();i++)
    {
    	temp=v[i];
    	if(temp[0]>='a' && temp[0]<='z')
    	{
    	temp[0]=temp[0]-32;
    	v[i]=temp;
    	}
    }
    for(int i=0;i<v.size();i++)
    cout<<v[i]<<" ";
    cout<<endl;
}
