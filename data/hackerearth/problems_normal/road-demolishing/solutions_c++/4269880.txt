#include<iostream>
using namespace std;
typedef long long ll;
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
        ll n,q,ans=0;
        cin>>n>>q;
        q--;
        ans=n*n-(n%q)*((n+q-1)/q)*((n+q-1)/q);
        ans=ans-(q-n%q)*(n/q)*(n/q);
        ans/=2;
        ans=(n*(n-1))/2-ans;
        cout<<ans<<endl;
    }
}
