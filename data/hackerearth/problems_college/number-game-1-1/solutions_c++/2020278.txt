#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	long int N,j,cnt;
	cin>>t;
	while(t--)
	{
		scanf("%ld",&N);
		j=N;
		cnt=0;
		while(j!=1)
		{
			if(j%3==0)
			{
				j=j/3;
			}
			else if(j%2==0)
			{
				j=j/2;
			}
			else
			{
				j-=1;
			}
			cnt++;
		}
		cout<<cnt<<"\n";
	}
}
