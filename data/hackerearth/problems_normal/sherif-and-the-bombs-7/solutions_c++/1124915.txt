#include <stdio.h>
#include <math.h>

typedef struct {
	int x, y, r;
} circulo;

int n, res;
circulo mapa[11];

double calculateDistance(double x1, double y1, double x2 , double y2) {
	double distance;
	double distance_x = x1 - x2;
	double distance_y = y1 - y2;
	distance = sqrt( (distance_x * distance_x) + (distance_y * distance_y));
	return distance;
}


int verifica(int x, int y, circulo z) {
	if(calculateDistance((double)x, (double)y, (double)z.x, (double)z.y) <= z.r)
		return 1;
	return 0;
}

int main() {
	scanf("%d", & n);
	for(int i = 0; i < n; i++) {
		scanf("%d %d %d", & mapa[i].x, & mapa[i].y, & mapa[i].r);
	}
	res = 0;
	for(int i = 1; i <= 1000; i++) {
		for(int j = 1; j <= 1000; j++) {
			int q = 0;
			for(int k = 0; k < n; k++) {
				q += verifica(i, j, mapa[k]);
				if(q >= 2) {
					res++;
					break;
				}
			}
		}
	}
	printf("%d\n", res);
}
