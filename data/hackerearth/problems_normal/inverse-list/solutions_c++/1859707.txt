#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n+1];
    	for(int i=1;i<=n;i++)
    	cin>>a[i];
    	int f=1;
    	for(int j=1;j<=n;j++)
    	{
    		if(a[a[j]]!=j)
    		{
    			f=0;
    			break;
    		}
    	}
    	if(f==1)
    	cout<<"inverse"<<endl;
    	else
    	cout<<"not inverse"<<endl;
    }
    return 0;
}
