#include <iostream>
using namespace std;
int main()
{
	int t ;
	int n,k ;
	cin >> t ;
	int i,j ;
	while(t--)
	{
		int max = 0 ;
		cin >> n >> k ;
		int A[n];
		int B[n];
		for(i=0;i<n;i++)
		{
			cin >> A[i];
		}
		for(i=0;i<n;i++)
		{
			cin >> B[i];
		}
		for(i=0;i<n;i++)
		{
			if(B[i]*(k/A[i]) > max)
			max = B[i]*(k/A[i]) ;
		}
		cout << max << endl;
	}
}
