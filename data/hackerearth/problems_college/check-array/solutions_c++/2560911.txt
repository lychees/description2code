#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int T,i,j,k,Arr[1000001],sum,N;
	cin>>T;
	while(T--)
	{
		cin>>N;
		sum=0;
		for(i=0;i<N;i++)
		{
			cin>>Arr[i];
			sum+=Arr[i];
		}
		int flag=0;
		long int left=0,right;
		for(i=0;i<N;i++)
		{
			right=sum-left-Arr[i];
			if(left==right)
			{
				flag=1;
				break;
			}
			left=left+Arr[i];
		}
		if(flag==1)
		cout<<"YES\n";
		else
		cout<<"NO\n";
		
	}
}
