#include <iostream>
#include<vector>
#include<algorithm>
#include<string>
#include<sstream>

using namespace std;

int main()
{
    string s;
    cin>>s;
    vector<long long int> arr(26,0);
    int i=0,index;
    while(i<s.size())
      {
      	index = i;
      	if(s[i]>='a' && s[i]<='z')
      	{
      	
      		int j=i+1;
      		while(j<s.size())
      		 {
      		 	if(s[j]>='a' && s[j]<='z'){
      		 	 i = j;
      		 	break;
      		 	}
      		 	else
      		 	 j++;
      		 }
      		string t = s.substr(index+1,j-index-1);
      		//cout<<s[index]<<" "<<t<<endl;
      		stringstream ss(t);
      		long long int len;
      		ss >> len;
      		arr[s[index]-'a'] += len;
      		if(j==s.size())break;
      		
      	}
       
      }
      for(int i=1;i<26;i++){
      	//cout<<arr[i]<<"  ";
      	arr[i]+= arr[i-1];
      }
      long long int test,tmp;
      cin>> test;
      while(test--){
      	scanf("%lld", &tmp);
      	int flag = 0;
      	for(int i=0;i<26;i++){
      		if(tmp<=arr[i]){
      			//cout<<i<<endl;
      			printf("%c\n", i + 'a');
      			flag = 1;
      			break;
      		}
      	}
      	if(flag==0){
      		printf("-1\n");
      	}
      }
}
