#include <bits/stdc++.h>
using namespace std;
#define MAX 1005
int main()
{
    int N,M;
    int a[MAX][MAX];
    cin >> N >> M;
    for(int i = 0; i < N;i++)
    	for(int j = 0;j < M;j++)
    		cin >> a[i][j];
    		
    long long row[MAX],col[MAX];
    memset(row,0,sizeof(row));
    memset(col,0,sizeof(col));
    
    // summing value-------
    long long sum  = 0;
    for(int i = 0; i < N;i++) {
    	sum = 0;
    	for(int j = 0;j < M;j++) {
    		sum += a[i][j];
    	}
    	row[i] = sum;
   	}
   	
   	for(int j = 0; j < M;j++) {
    	sum = 0;
    	for(int i = 0; i < N;i++) {
    		sum += a[i][j];
    	}
    	col[j] = sum;
   	}
   	
   	int xo = 0,max_xor = 0;
   	for(int i = 0; i< N;i++) {
   		for(int j = 0;j < M;j++) {
   			xo = row[i]^col[j];		
   			if(xo > max_xor)
   			max_xor = xo;
   		}
   	}
   		cout << max_xor << endl;
    return 0;
}
