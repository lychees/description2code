#include<iostream>
using namespace std;
typedef long long int lli;
lli N;
bool flag[10];
lli a[13];
lli get(int i)
{
    lli ans=0;
    lli j=0;
    while(i>0)
    {
        if((i&1)==1)
        {
            ans++;
            flag[j]=true;
        }
        j++;
        i=i>>1;
    }
    return ans;
}
int main()
{
    int k,tc;
    cin>>k>>tc;
    for(int i=0;i<k;++i)
        cin>>a[i];
    while(tc--)
    {
        lli A,B;
        cin>>A>>B;
        lli ans=0;
        lli D=(1<<k)-1;
        for(int i=1;i<=D;++i)
        {
            for(int j=0;j<k;++j)
                flag[j]=false;
            lli  setbits=get(i);
            lli num=1;
            for(int j=0;j<k;++j)
            {
                if(flag[j]==true)
                    num*=a[j];
            }
            if(setbits%2==0)ans-=((B/num)-((A-1)/num));
            else ans+=((B/num)-((A-1)/num));;
        }
        cout<<ans<<endl;
    }
    return 0;
}
