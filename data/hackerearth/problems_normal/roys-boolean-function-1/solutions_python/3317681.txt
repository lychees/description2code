def prime(n):
	c=0
	i=1
	while i<=n:
		if n%i==0:
			c+=1
		i+=1
	if c==2:
		print "TRUE"
	else:
		print "FALSE"
def gcd(i,j):
	if j==0:
		return i
	else:
		return gcd(j,i%j)
def go(n):
	i=1
	c=0
	while i<n:
		if gcd(i,n)==1:
			c+=1
		i+=1
	prime(c)
