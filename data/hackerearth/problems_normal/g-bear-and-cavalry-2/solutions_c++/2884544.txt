#include <iostream>
#include <algorithm>
#define endl '\n'
using namespace std;

const int maxN = 45;

int t, n, w[maxN], h[maxN], first, currMax;

bool cmp(int x, int y)
{
    return x > y;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    cin>>t;
    for(int cs = 0; cs < t; cs++)
    {
        cin>>n;
        for(int i = 0; i < n; i++)
        {
            cin>>w[i];
        }
        for(int i = 0; i < n; i++)
        {
            cin>>h[i];
        }


        sort(w + 1, w + n);
        sort(h, h + n, cmp);

        bool p = true;

        first = w[0] * h[0];
        for(int i = 1; i < n; i++)
        {
            if(w[i] * h[i] >= first)
            {
                p = false;
                break;
            }
        }

        if(p) cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
    }
}
