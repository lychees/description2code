#include<bits/stdc++.h>
#define inf 2000000
using namespace std;

int f(int a, int b){
    int s=0;
    while(a!=1&&b!=1&&a>0&&b>0){
        if(a>b){
            a=a-b;
        }
        else{
            b=b-a;
        }
        s++;
    }
    if(a==1)return b-1+s;
    if(b==1)return a-1+s;
    return inf;
}

int main(){
    int l,aux,res,cases,i;
    cin>>cases;
    while(cases--){
        cin>>l;
        res=-1;
        for(i=1;i<=l/2;i++){
            aux = f(i, l-i)+1;
            if(res==-1||res>aux)res=aux;
        }
        if(res==-1)cout<<0<<endl;
        else cout<<res<<endl;
    }
}
