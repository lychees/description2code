#include <iostream>
using namespace std;

int main()
{
	int t,n,ans,temp,p;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	ans=0;
    	if(n==0||n==1)
    	cout<<ans<<endl;
    	else
    	{
    		temp=-1;
    		p=1;
    		while(temp<0)
    		{
    			p=3*p;
    			temp=p-n;
    			ans++;
    		}
    		cout<<ans<<endl;
    	}
    }
    return 0;
}
