#include<bits/stdc++.h>
using namespace std;
#define LL long long 
LL digsum(LL n)
{
	LL i;
	while (n >= 10)
	{
		for (i = 0; n; n /= 10)
	    i = i + n % 10;
		n = i;
	}
	return n;
}

int main()
{
	LL n,t;
	scanf("%lld",&t);
	while(t--){
	scanf("%lld",&n);
	printf("%lld\n",digsum(n));
	}
	return 0;
}
