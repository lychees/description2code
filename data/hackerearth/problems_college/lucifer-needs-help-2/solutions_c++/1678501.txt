#include<bits/stdc++.h>
using namespace std;
long long a[1000006];
long long sum[1000006];

int main() {
	long long n,i,a1,b,c;
	scanf("%lld",&n);
	for(i=0;i<n;i++)
	scanf("%lld",&a[i]);
	long long q;
	scanf("%lld",&q);
	for(i=0;i<q;i++) {
		scanf("%lld%lld%lld",&a1,&b,&c);
		a1--;
		b--;
		sum[a1] += c;
//		printf("sum[%lld]=%lld\n",a1,sum[a1]);
		if(b+1 < n) {
			sum[b+1] -= c;
//			printf("sum[%lld]=%lld\n",b+1,sum[b]);
		}
	}
//	for(i=0;i<n;i++) 
//	printf("sum[%lld] =%lld\n",i,sum[i]);	

	for(i=1;i<n;i++) {
		
		sum[i] += sum[i-1];
	}
	for(i=0;i<n;i++) {
		printf("%lld\n",(sum[i] + a[i]));	
	}
	return 0;
}
