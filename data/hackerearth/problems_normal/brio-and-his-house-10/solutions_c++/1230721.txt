#include <iostream>
#include<cmath>
#include<iomanip>
using namespace std;

int main()
{
    int t;
    double a,b,c, r;
    cin>>t;
    while(t--)
    {
     
      cin>>a>>b>>c;
      r=(a*b*c)/(sqrt((a+b+c)*(b+c-a)*(a+c-b)*(a+b-c)));
      cout<<fixed<<setprecision(4)<<3.1415*r*r<<endl;
    }
    
    return 0;
}
