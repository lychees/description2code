#include <iostream>
#include <math.h>
using namespace std;

int gcd(int n,int i)
{
	int j;
	for(j = 2; j<=i;j++)
	{
		if(n%j == 0 && i%j == 0)
		{
			return 0;
		}
	}
	return 1;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,i;
    	cin>>n;
    	int count = 1;
    	for(i = 2; i<n;i++)
    	{
    		if((n%i != 0) && (gcd(n,i) == 1))
    		{
    			count++;
    		}
    	}
    	int flag = 0;
    //	cout<<" n = "<<n<<" count "<<count<<endl;
    	for(i = 2; i<=sqrt(count);i++)
    	{
    		if(count%i == 0)
    		{
    			flag = 1;
    			break;
    		}
    	}
    	if(flag == 1 || count == 1)
    	{
    		cout<<"Incorrect\n";
    	}
    	else
    	{
    		cout<<"Correct\n";
    	}
    }
    return 0;
}
