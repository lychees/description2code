#include <iostream>
#include <queue>
#include <algorithm>
using namespace std;

int main() {
	int T, N;
	cin >> T;
	while (T > 0) {
		cin >> N;
		queue<int> q;
		int *b = new int[N];
		for (int i = 0; i < N; i++) {
			cin >> b[i];
		}
		sort(b, b+N);
		for (int i = 0; i < N; i++)
			q.push(b[i]);
		int res = 0, i = 0, prev = 0;
		while (!q.empty()) {
			if (q.front() == q.back()) {
				res += q.size();
				break;
			}
			int x = q.front();
			q.pop();
			if (x > q.front()) {
				res++;
			} else if (x == q.front()) {
				while (x == q.front()) {
					q.push(x); q.pop();
				}
				if (x > q.front())
					res++;
			}
		}
		cout << res << endl;
		delete[] b;
		T--;
	}
	return 0;
}
