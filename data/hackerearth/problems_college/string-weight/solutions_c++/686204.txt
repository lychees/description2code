#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
       char str[10001];
       cin>>str;
       int n,i,s[26]={0};
       cin>>n;
       char ch;
       for(i=0;i<n;i++)
       {
            cin>>ch;
            s[ch-'a']=1;
       }
       int score=0;
       i=0;
       while(str[i]!='\0')
       {
        if(s[str[i]-'a']==0)
        score+=(str[i]-'a'+1);
        i++;
       }
       cout<<score<<endl;
    }
    return 0;
}
