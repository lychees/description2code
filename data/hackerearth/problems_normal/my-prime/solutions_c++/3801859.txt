#include <iostream>
#include <cstdlib>

using namespace std;
#define MAX 100
#define loop(i,n) for(int i=0; i<n; ++i)

int isFactor(int a, int b){
	if(b>0){
		if(a%b == 0) return 1;
	}
	return 0;
}

int checkFactorTable(int a[], int n, int num){
	loop(i,n){
		if(a[i]==0)
			break;
		if(num%a[i]==0)
			return 1;
	}
	return 0;
}

int main(){
    int n,s[MAX],t[MAX],tSize=0,i;
    string line;
	cin >> n;
	loop(i,n){
		cin >> s[i];
	}
	for(int i=0;i<n;i++){
		int fnd=0;
		for(int j=i+1;j<n;j++){
			if(s[i]==0) continue;
			if(s[i]<s[j]) {
				if(isFactor(s[j],s[i])){
					//check for factors in orginal array
					s[j]=0;
					fnd=1;
				}
			}else{
				if(isFactor(s[i],s[j])){
					s[i]=0;
					continue;
				}
			}
		}
		//search in factor array
		if(!fnd && checkFactorTable(t,tSize,s[i])){
			s[i]=0;
		}
	}
	loop(i,n){
		if(s[i]!=0)
			cout << s[i] << " ";
	}
	cout << "\n";
    return 0;
}
