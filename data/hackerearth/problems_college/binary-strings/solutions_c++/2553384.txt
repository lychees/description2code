#include <iostream>
#include<math.h>
using namespace std;

int main()
{
	long long int t,n,i,fib[1000006];
    cin>>t;
    
    fib[0]=1;
    fib[1]=1;
    for(i=2;i<=1000000;i++)
     fib[i] = (fib[i-1] + fib[i-2])%1000000007;
     
    while(t--)
    {
      cin>>n;
     
      cout<<fib[n]<<endl;
    }
    return 0;
}
