#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pii pair<int,int>
#define pll pair<ll,ll>
#define mp make_pair
#define ff first
#define ss second
#define pb push_back
#define pf pop_front
#define IOS ios::sync_with_stdio(false)
int t , n , L ;
pii p[100005];
int main(){

	int i , j , k ;
    cin >> t;
    while(t--){
        cin >> n >> L;
        for(i = 0 ; i < n ; i++){
            cin >> p[i].ff >> p[i].ss;
        }
        sort(p,p+n);
        int ans = 0  , l = 0 , r = 0 , lm = 0 , rm = 0;
        int flag = 0;
        for(i = 0 ; i < n ; i++){
            l = p[i].ff;
            r = p[i].ss;
            rm = p[i].ff + L ;
            if(r == rm){
                ans = 1;
                break;
            }
            for(j = i + 1 ; j < n ; j++){
                if(p[j].ff > r)continue;
                if((p[j].ss > rm))continue;
                r = max(r,p[j].ss);
                if(r==rm){ans = 1;
                    break;
                }
            }
        }
        if(ans)cout << "Yes\n";
        else cout << "No\n";

    }
	return 0;
}
