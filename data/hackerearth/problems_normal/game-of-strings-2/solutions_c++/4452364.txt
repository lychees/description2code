#include <bits/stdc++.h>
using namespace std;
char str[1000003],pat[1000003];
int lps[1000003],n,l;
bool kmp(char pat[],char str[],int k)
{
	int i=0,j=0;
	for(i=1;i<k;)
	{
		if(j==l)
		return 1;
		if(str[i]==pat[j])
		{
			i++;
			j++;
			
		}
		else
		{
			if(j>0)
			j=lps[j-1];
			else i++;
		}
	}
	if(j==l)
	return 1;
	return 0;
}

{
    int t,i,k,j;

    	n=strlen(str);
    	lps[0]=0;
    	i=1;
    	j=0;

    	{
    		if(str[i]==str[j])
    		{
    			lps[i]=j+1;
    			i++;
    			j++;
    			
    		}
    		else
    		{
    		if(j>0)
    		{
    			j=lps[j-1];
    		}
    		else
    		{
    			lps[i]=0;
    			i++;
    		}
    		}
    	}
    	j=lps[n-1];
    	while(j>k)
    	{
    		j=lps[j-1];
    	}
    	l=j;
    	int f=1;
    	while(l>0)
    	{
    	for(i=0;i<l;i++)
    	pat[i]=str[i];
    	pat[i]='\0';
    	if(kmp(pat,str,k))

    		f=0;
    	printf("%s\n",pat);

}
