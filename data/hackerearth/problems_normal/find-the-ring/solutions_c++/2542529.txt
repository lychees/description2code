#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int t,index,n;
    t=0;
    scanf("%d",&t);
    while(t--){
    	index=0; n=0;
    	scanf("%d%d",&index,&n);
    while(n!=0){
    		while(index!=0 && n!=0)
    		 {
    		   index--;
    		   n--;
    		 }
    		  if(index==0 && n!=0)
    		   {
    		   index++;
    		   n--;
    		   }
    		   
    	
    	}
    	printf("%d\n",index);
    }    
}
