#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int n, m, i, j, a[100005], c[100005], cost=0, count=0;

    cin >> n >> m;
    for(i=0; i<n; i++)
        cin >> a[i];
    for(i=0; i<n; i++)
        cin >> c[i];

    sort(a, a+n);
    sort(c, c+n);

    i = 0; j = 0;
    while(count<n && cost<m && i<n && j<n){
        if(a[i]<c[j] && (cost+a[i])<=m ){
            cost=cost+a[i];
            i++;
        }
        else if(c[j]<=a[i] && (cost+c[i])<=m){
                cost=cost+c[j];
                j++;
            }
            else{
                break;
            }
        count++;
    }

    if(i>=n && cost<=m){
    	while(count<n && j<n && (cost+c[j])<=m){
    		cost=cost+c[j];
    		count++;
    		j++;
    	}
    }
    if(j>=n && cost<=m){
    	while(count<n && i<n && (cost+a[i])<=m){
    		cost=cost+a[i];
    		count++;
    		i++;
    	}
    }
    cout << count <<endl;

    return 0;
}
