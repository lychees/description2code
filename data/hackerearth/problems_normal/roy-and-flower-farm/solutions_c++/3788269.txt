#include<bits/stdc++.h>
#define lp(i,n) for(int i=0;i<n;i++)
#define FIO ios::sync_with_stdio(0)
using namespace std;
typedef pair<int,int> ii;
typedef pair<double,double> dd;
typedef long long ll;
typedef pair<ll,ll> pll;
typedef vector<vector<int> > vvi;
typedef vector<int> vi;
typedef unsigned long long ull;
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--){
    int n,p;
    cin>>n>>p;
    ii a[n+1];
    for(int i=1;i<=n;i++)
        cin>>a[i].first>>a[i].second;
    ll dp[n+1][p+1];
    memset(dp,0,sizeof(dp));
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=p;j++)
        {
            if(j>=a[i].second)
                dp[i][j]=max(dp[i-1][j-a[i].second]+a[i].first-a[i].second,dp[i-1][j]);
            else
                dp[i][j]=dp[i-1][j];
        }
    }
    ll Max=-1e9-9,Min=1e9+9;
    for(int i=1;i<=n;i++)
    {
        for(int j=0;j<=p;j++)
        {
            if((dp[i][j]>Max)||(dp[i][j]==Max && j<Min))
                Min=j , Max=dp[i][j];
        }
    }
    cout<<Min<<" "<<p+Max<<endl;
    }
    return 0;
}
