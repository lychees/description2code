#include <bits/stdc++.h>
using namespace std;

int main()
{
    string str;
    cin >> str;
    int x = 0, y = 0;
    set<pair<int, int> >s;
    s.insert(make_pair(0, 0));
    int l = str.size();
    int slip = 0;
    for(int i = 0; i < l; i++) {
    	if(str[i] == 'R') {
    		x = x;
    		y = y + 1;
    		if(s.find(make_pair(x, y)) != s.end()) slip++;
    		s.insert(make_pair(x, y));
    	}
    	else if(str[i] == 'L') {
    		x = x;
    		y = y - 1;
    		if(s.find(make_pair(x, y)) != s.end()) slip++;
    		s.insert(make_pair(x, y));
    	}
    	else if(str[i] == 'U') {
    		x = x - 1;
    		y = y;
    		if(s.find(make_pair(x, y)) != s.end()) slip++;
    		s.insert(make_pair(x, y));
    	}
    	else if(str[i] == 'D') {
    		x = x + 1;
    		y = y;
    		if(s.find(make_pair(x, y)) != s.end()) slip++;
    		s.insert(make_pair(x, y));
    	}
    }
    cout << slip << endl;
    return 0;
}
