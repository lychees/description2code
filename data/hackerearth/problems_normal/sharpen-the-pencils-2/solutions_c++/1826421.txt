#include "iostream"

using namespace std;

void countSticks(int arr[] , int size) {
	int st = 0 , end = size - 1;
	int count = 1;
	int credit = arr[end]*2 - arr[st];
	//cout<<"credit : "<< credit<<"\nans : "<<count<<endl;
	while(st < (end-1)) {
		while(credit >= 0) {
			
			if (st < (end-1))
			{
				credit -= arr[++st];
				count++;
				//cout<<"credit : "<< credit<<"\nans : "<<count<<endl;
			}
			else 
				break;
		}

		if (st < (end-1))
		{
			credit += arr[--end]*2;
			//cout<<"credit : "<< credit<<"\nans : "<<count<<endl;
		}
		
	}
	cout<<count<<" "<<(size-count)<<endl;
}


int main() {
	int test = 0;
	cin>>test;
	while ( test-- ) {
		int size = 0;
		cin>>size;
		int arr[size];
		for (int i = 0; i < size; ++i)
		{
			cin>>arr[i];
		}
		if(size == 1) {
			cout<<"1 0"<<endl;
		}
		else
			countSticks(arr , size);
	}
}
