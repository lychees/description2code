#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    for(int j=n;j>0;j--)
    {
    	if(j==n || j==1)
    	 {
    	 	for(int k=0;k<n;k++)
    	 	cout<<"*";
    	 	cout<<"\n";
    	 }
    	 else
    	 {
    	 	for(int m=1;m<j;m++)
    	 	cout<<" ";
    	 	cout<<"*\n";
    	 }
    }
}
