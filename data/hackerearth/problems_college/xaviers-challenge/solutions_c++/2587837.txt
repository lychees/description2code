#include <iostream>
#include <cmath>
#include <algorithm> 
using namespace std;

int main()
{
   	int t , sum  , len ;
   	double temp ;
   	string str;
   	cin >> t ;
   	cin.ignore();
   	while(t--)
   	{
   		sum = 0;
   		getline(cin , str);
   		len = str.size();
   		for(int i = 0 ; i < len ; ++i)  sum = sum + str[i];
   		temp = (double) sum / len;
   		sum = floor(temp);
   		if(sum % 2 != 0) 
   		{
   			reverse(str.begin() , str.end());
   			cout << str <<endl;
   		}
   		else cout << str << endl;
   	}
    return 0;
}
