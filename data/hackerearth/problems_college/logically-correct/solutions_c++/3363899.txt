#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int f,s,n;
    int i;
    cin>>f>>s>>n;
    long long int array[n];
    array[1]=f;
    array[2]=s;
    for(i=1;i<=(n-2);i++)
  {  
    array[i+2]=array[i+1]*array[i+1]+array[i];
  }
    cout<<array[n];
    return 0;
}
