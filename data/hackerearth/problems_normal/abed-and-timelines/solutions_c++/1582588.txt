#include <stdio.h>
#include<bits/stdc++.h>
int main()
{
    int n,m,i,j,k,l,a1,b2,b1,a2,d1,d2,t,p;
    long long int ans;
    scanf("%d",&t);
    for(i=0;i<t;i++)
    {
      scanf("%d%d",&m,&n);
   	scanf("%d%d",&a1,&b1);
    	scanf("%d%d",&a2,&b2);
    	scanf("%d",&p);
    	d1=abs(a2-a1);
    	if(d1>m-d1)
    	d1=m-d1;
    	d2=abs(b2-b1);
    	if(d2>n-d2)
    	d2=n-d2;
    	
    	ans=(d1+d2)*p;
    	printf("%lld",ans);
    	printf("\n");
    	if(ans>1000)
        printf("Let go of the group.");
        else
        printf("You saved the group.");
        printf("\n");
    }
    return 0;
}
