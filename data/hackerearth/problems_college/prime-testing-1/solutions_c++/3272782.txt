#include <iostream>
#include<math.h>
#include<vector>
using namespace std;
bool is_prime(int n){
	if(n==0){return false;}
	if (n==1){return false;}
	else if(n<4){return true;}
	else if(n%2==0){return false;}
	else if(n<9){return true;}
	else if(n%3==0){return false;}
	else{
		int r = floor(sqrt(n));
		int f=5;
		while(f<=r){
			if(n%f==0){return false;}
			else if(n%(f+2)==0){return false;}
			f+=6;
		}
	}
	return true;
}

int main()
{
   int t;cin>>t;
   while(t--){
        long long n;cin>>n;long long count =0;
        for(long long i=0;i<n;i++){
            int x;cin>>x;if(is_prime(x)) count++;
        }
        cout<<count<<endl;
   }

    return 0;
}
