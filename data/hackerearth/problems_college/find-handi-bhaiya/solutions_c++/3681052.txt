#include <iostream>
using namespace std;

int main()
{
    long int t,a,b,c,i;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>a>>b>>c;
    	if((a==b)&&(b==c))
    		cout<<"YES. It is a equilateral triangle."<<endl;
    		
    	else if((a+b>c)&&(b+c>a)&&(c+a>b))
    	{
    		if((a==b)||(b==c)||(c==a))
    			cout<<"YES. It is a isosceles triangle."<<endl;
    	
    		else
    			cout<<"YES. It is a scalene triangle."<<endl;
    	}
    	
    	else
    		cout<<"NO"<<endl;
    }
}
