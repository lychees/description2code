#include<iostream>
#include<cstdio>
#include<string>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        double h,d,v;
        cin>>h>>d>>v;
        double ans=v*sqrt((d*d)-(h*h))/h;
        printf("%.9lf\n",ans);
    }
    return 0;
}
