#include <iostream>
using namespace std;

int main()
{
  string s;
  cin >> s;
  for(int i = 0 ; i < s.length() ; i++ )
  	{
  		cout << s[i];
  		if ( i != s.length() - 1 )
  		cout<<"-";
  	}
  	cout << endl;
  for(int i = s.length()-1; i >= 0 ; i-- )
  	{
  		cout << s[i];
  		if ( i != 0 )
  		cout<<"-";
  	}
}
