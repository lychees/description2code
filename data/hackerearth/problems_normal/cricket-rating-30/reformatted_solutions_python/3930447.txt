def max_subarray(A):
    max_ending_here = max_so_far = 0
    for x in A:
        max_ending_here = max(0, max_ending_here + x)
        max_so_far = max(max_so_far, max_ending_here)
    return max_so_far


n = int(raw_input())
if n > 0:
    a = [int(x) for x in raw_input().split()]
    print(max_subarray(a))
else:
    print(0)
