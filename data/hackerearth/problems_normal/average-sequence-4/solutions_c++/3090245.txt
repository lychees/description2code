#include <iostream>
#include <math.h>
#include <vector>
using namespace std;
int main()
{

    int x;
    cin >> x;
    vector <long long> a;
    int c=0;
    int m=2;
    long long t=0;
    while(x--)
    {

        long long q;
        cin >> q;
        if(c==0)
        {
            a.push_back(q);
            t+=q;
            c++;
        }
        else
        {

            a.push_back(q*m-t);
            t+=a[c];
            c++;
            m++;
        }

    }
    int d;
    for(d=0;d<a.size();d++)
    {

        cout << a[d]<<" " ;
    }
    cout <<endl;

}
