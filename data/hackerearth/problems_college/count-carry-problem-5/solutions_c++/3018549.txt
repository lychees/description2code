#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    string a,b;
    int test;
    cin>>test;
    while(test--)
    {
    cin>>a>>b;
    auto i=a.end()-1,j=b.end()-1;
    
    int ct=0;
    for(;i>=a.begin()&&j>=b.begin();i--,j--)
    {
    	if(*i-'0'+*j-'0' > 9)
    	ct++;
    }
    if(ct==0)
    	cout<<"No carry operation"<<endl;
    else if(ct==1) 
    	cout<<ct<<" carry operation"<<endl;
	
	else 
    	cout<<ct<<" carry operations"<<endl;

}

return 0;
}
