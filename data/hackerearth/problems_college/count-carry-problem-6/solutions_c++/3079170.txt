#include<iostream>
using namespace std;
long int dig(long int t)
{
    long int a;
    while(t!=0)
    {
        a=t%10;
        t=t/10;
    }
    return a;
}
int main()
{
    long int t;
    cin>>t;
    while(t--)
    {
        long int a,b,c=0,q=0;
        cin>>a>>b;
        while(a!=0 && b!=0)
        {
            long int z=(a%10)+(b%10)+q;
            if(z>=10)
                {
                    c++;
                    q=dig(z);
                }
            a/=10;
            b/=10;
        }
        if(c==0)
            cout<<"No carry operation"<<endl;
        else if(c==1)
            cout<<c<<" carry operation"<<endl;
           else
           cout<<c<<" carry operations"<<endl;
    }
    return 0;
}
