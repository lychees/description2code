#include <iostream>
#include <string>
#include <cstring>
#include <math.h>
using namespace std;
int main()
{
  long long int t,n,k,p,r,f1,f2,f3,i,j,m,te;
  cin>>te;
  for(m=0;m<te;m++)
  {
      cin>>n;k=0;
      if(n%2==1)
        p=n/2+1;
      else
        p=n/2;
      for(i=1;i<=p;i++)
      {
          t=n-i+1;
          r=n-2*i+1;
          f1=1;f2=1;f3=1;
          for(j=1;j<=t;j++)
            f1*=j;
          for(j=1;j<=r;j++)
            f2*=j;
          for(j=1;j<=i;j++)
            f3*=j;
            k+=f1/(f2*f3);
      }
      k+=1;
      cout<<(k*k)<<"\n";
  }
return 0;
}
