#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
typedef int w;
typedef long long l;
struct hi{
    string fin;
    w important;
    string lastname;
};
vector<hi> v;
w n;
string s;
char tc(w x){
    return x+'a';
}

bool cmp(hi a,hi b){

    if(a.lastname!=b.lastname) return a.lastname<b.lastname;
    else{
        return a.important>b.important;
    }
}

void solve(){
    v.clear();
    cin>>n;
    getline(cin,s);
    for(int i=1;i<=n;++i){
        getline(cin,s);
        hi tmp;
        string tm="";
        for(int j=0;j<s.size();++j) if(s[j]==' ') tm=""; else tm=tm+tc(tolower(s[j]));
        tmp.lastname=tm;
        tmp.important=i;
        tmp.fin=s;
        v.push_back(tmp);
    }
    sort(v.begin(),v.end(),cmp);
    for(int i=0;i<v.size();++i){
        cout<<v[i].fin<<endl;
    }


}

int main(){
    ios_base::sync_with_stdio(false);
    //freopen("in","r",stdin);
    w T;
    cin>>T;
    getline(cin,s);
    while(T--) solve();
}
