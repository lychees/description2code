#include <iostream>
#define LL long long
using namespace std;
int main(){
	int tst;
	cin>>tst;
	while(tst--){
		string str;
		cin>>str;
		int size=str.size(),i=0,pre=0;
		LL sum=0;
		bool flag=false;
		for(int i=0;i<size;i++){
			if(str[i]=='a'||str[i]=='z'){
				sum=sum+i-0+1;
				pre=i;
				flag=true;
			}else{
				if(flag){
					sum=sum+pre+1;
				}
			}
		}
		cout<<sum<<endl;
	}
	return 0;
}
