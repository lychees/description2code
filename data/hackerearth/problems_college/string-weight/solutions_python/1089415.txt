t=int(raw_input())
while t>0:
    t-=1
    s=raw_input()
    n=int(raw_input())
    forgot_list = map(str, raw_input().split())
    n=len(s)
    ans=[]
    i=0
    while i<n:
        if s[i] not in forgot_list:
            ans.append(s[i])
        i+=1
    weight=0
    n=len(ans)
    i=0
    while i<n:
        weight+=ord(ans[i])-96
        i+=1
    print weight
