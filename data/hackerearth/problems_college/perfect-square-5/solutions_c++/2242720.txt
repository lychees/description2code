#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    long long a,t,m,s,ss,n,b;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	m=sqrt(n);
    	s=m*m;
    	b=s;
    	ss=(m+1)*(m+1);
    
    	
    	if(n==b)
    	cout<<"YES\n";
    	else if(abs(s-n)<abs(ss-n))
    	   cout<<s-n<<"\n";
    	else
    	   cout<<"+"<<ss-n<<"\n";
    	
    }
    return 0;
}
