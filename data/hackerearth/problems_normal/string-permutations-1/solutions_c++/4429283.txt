#include <iostream>
#include<unordered_map>
using namespace std;

int main()
{
    int t;
   // cin>>t;
   // while(t>0)
    {
    	int n;
    	cin>>n;
    	string temp;
    	unordered_map<string ,int> mymap;
    	for(int i=0;i<n;i++)
    	{
    		cin>>temp;
    		if(mymap.find(temp)==mymap.end())
    		{
    			mymap[temp]=1;
    		}
    		else
    		{
    			mymap[temp]+=1;
    		}
    	}
    	unordered_map<string ,int>::iterator it;
    	it=mymap.begin();
    	int maxcount=0;
    	int maxvalue=0;
    	for(it=mymap.begin();it!=mymap.end();it++)
    	{
    		if(it->second > maxvalue)
    		{
    			maxvalue=maxcount=it->second;
    		}
    		else if(it->second==maxvalue)
    		{
    			maxcount=maxcount+it->second;
    		}
    	}
    	cout<<maxcount<<endl;
    	t--;
    }
}
