#include<iostream>
#include<bits/stdc++.h>
using namespace std;
long int root(long int a[],long int x)
{
	while(x!=a[x])
	{x=a[x];
	}
	return x;
}
void uni(long int a[],long int x,long int y)
{
long int rx=root(a,x),ry=root(a,y);
if(x!=y)
{a[rx]=ry;
	
	}	
	
}
int main()
{
	long int n,k,i,j,count;
	cin>>n;
	cin>>k;
	long int arr[n+1];
	for(i=1;i<=n;i++)
	{arr[i]=i;
	}
	while(k--)
	{cin>>i;
	cin>>j;
	uni(arr,i,j);
		
	}
	
	for(i=1;i<=n;i++)
	{if(arr[i]==i)
	count++;
	}
	cout<<count;
}
