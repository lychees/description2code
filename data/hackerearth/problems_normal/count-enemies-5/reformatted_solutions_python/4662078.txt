# print 'Hello World!'
no = int(raw_input())
for i in range(no):
    inputStr = raw_input()
    inputStrArr = list(inputStr)
    stack = []
    for j in inputStrArr:
        # print j
        if len(stack) == 0:
            stack.append(j)
        elif stack[-1] is 'O' and j is 'X':
            stack.pop()
            while len(stack) != 0 and stack[-1] is 'O':
                stack.pop()
            stack.append(j)
        elif stack[-1] is 'O' and j is 'O':
            # print 'hello'
            stack.append(j)
        elif stack[-1] is '*' or j is '*':
            stack.append(j)
        elif stack[-1] is 'X' and j is not 'O':
            stack.append(j)
        # print stack
    count = 0
    for k in stack:
        if k is 'O':
            count = count + 1

    print count
