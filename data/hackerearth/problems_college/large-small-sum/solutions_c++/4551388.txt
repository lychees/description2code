#include <bits/stdc++.h>
using namespace std;

typedef long long int ll;

int main()
{
    ll t,n,i,a[100001],x,y;
    
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	
    	for(i=1;i<=n;i++)
    	cin>>a[i];
    	
    	sort(a+1,a+n+1);
    	
    	cin>>x>>y;
    	cout<<a[n+1-x]+a[y]<<endl;
    }
    
    return 0;
}
