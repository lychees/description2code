#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n;
    	cin >> n;
    	vector<int> A(n);
    	for(int i=0;i<n;++i)
    		cin >> A[i];
    	int sum;
    	cin >> sum;
    	bool flag = false;
    	for(int i=0;i<(1<<n) and !flag;++i)
    	{
    		int total = 0;
    		for(int j=0;j<n;++j)
    			if(i&(1<<j))
    				total += A[j];
    		if(total == sum)
    			flag = true;
    		// cout << i << " : " << total << endl;
    	}
    	if(flag)
    		cout << "YES" << endl;
    	else	
    		cout << "NO" << endl;
    }
    return 0;
}
