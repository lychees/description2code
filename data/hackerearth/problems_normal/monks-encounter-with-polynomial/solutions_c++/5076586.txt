#include<bits/stdc++.h>
#include<algorithm>
using namespace std;

long long int binarySearch(long long int A,long long int B,long long int C,long long int K)
{
	long long int low = 0;
	long long int high = 100000;
	long long int mid,temp;
	while(low<=high)
	{
		mid = low + (high-low)/2;
		temp = A*mid*mid + B*mid + C;
		if(temp == K)
		{
			return mid;
		}
		else if(temp < K)
		{
			low = mid + 1;
		}
		else
		{
			high = mid - 1;
		}
	}
	return low;
}

int main()
{
   long long int t;
   cin >> t;
   while(t--)
   {
   	long long int A,B,C,K;
   	cin >> A >> B >> C >> K;
   	long long int X = binarySearch(A,B,C,K);
   	cout << X << endl;
   }
   return 0;
}
