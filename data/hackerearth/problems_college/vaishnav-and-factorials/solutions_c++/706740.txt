#include <iostream>
using namespace std;

int main()
{
    int t, a[500]={0,};
    cin>>t;
    while(t--)
    {
        a[0]=1;
        int m=1;
        int temp=0;
        int n;
        cin>>n;
        int i,j;
        for(i=1;i<=n;i++)
        {
            for(j=0;j<m;j++)
            {
                int x=a[j]*i+temp;
                a[j]=x%10;
                temp=x/10;
            }

            while(temp>0)
            {
                a[m]=temp%10;
                temp=temp/10;
                m++;
            }
        }
        
        int cnt=0;
        for(i=m-1;i>=0;i--)
        {
            if(a[i]==4||a[i]==7)
            cnt++;
        }
        cout<<cnt<<endl;
    }
    return 0;
}
