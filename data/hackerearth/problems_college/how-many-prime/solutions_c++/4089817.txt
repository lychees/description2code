using namespace std;

int powr (int a, int b)
{
    if (b == 0)
        return 1;
    int x = powr(a, b/2);
    if (b % 2 == 0)
        return (x*x);
    else
        return (((x*x))*a);
}

map <int,int> v;
int mark[1100000];

void pre() {
    int i,j,cnt = 0;
    for(i = 2; i <= 1000000; i++) {
        if(mark[i] == 0) {
            v[i] = 1;
            cnt++;
            for(j = i+i; j <= 1000000; j+=i) {
                mark[j] = 1;
            }
        }
    }
}

bool check(int a, int b) {
    if(v[a] == 1 && v[b] == 1) return true;
    return false;
}

int main()
{
    ios::sync_with_stdio(0);
    pre();
    int t,i,x;
    cin>>t;
    while(t--) {
        cin>>x;
        for(i = x/2-(x%2 == 0); i >= 2; i--) {
            if(check(i,x-i)) {
                cout<<i<<" "<<x-i<<endl;
                goto l1;
            }
        }
        cout<<"-1 -1\n";
        l1: ;
    }
    return 0;
}
