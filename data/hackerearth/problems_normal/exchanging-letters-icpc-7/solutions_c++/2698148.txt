int f[2005],inv[2005];
int pow(int a,int b){
    int ans=1;
    while(b){
        if(b&1)ans=1LL*ans*a%mod;
        b>>=1;a=1LL*a*a%mod;
    }
    return ans;
}
int C(int a,int b){
    return 1LL*f[a]*inv[a-b]%mod*inv[b]%mod;
}
int main(){
    int T,i,j,k,n,m,K;
    f[0]=1;
    scanf("%d%d%d",&n,&K,&m);
    for(i=1;i<=K;i++)f[i]=f[i-1]*(ll)i%mod;
    inv[K]=pow(f[K],mod-2);
    for(i=K-1;i>=0;i--)inv[i]=inv[i+1]*(ll)(i+1)%mod;
    for(i=0,k=0;i<K;i++){
        scanf("%d",&j);
        m-=j;
    }
    k=K;k-=m;
    if(k<0||(k&1)||(k/2>K))puts("0");
    else{   
        k/=2;
        printf("%d\n",C(K,k));
    }
    return 0;
}
