#include<iostream>
using namespace std;
#include<cmath>

#define mx 1000000000000000000
#define ll long long
int main(){
    ll t,n;
    cin>>t;
    while(t--){
        cin>>n;
        ll x,y;
        x=sqrt(n);
        y=n-x*x;
        cout<<2*x-1+y/x+((y%x)?1:0)<<"\n";
    }
    return 0;
}
