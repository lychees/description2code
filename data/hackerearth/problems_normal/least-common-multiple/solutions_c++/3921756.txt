#include <bits/stdc++.h>
using namespace std;
#define ll long long int
bool magicFun(ll a, ll n)
{
	ll power = a;
	while(a < 10000000000000000 && power <= n){
		if(power == n)
			return true;
		power = power*a;
	}
	return false;
}
