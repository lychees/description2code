
def is_palin(s, a, b):
    while a < b:
        if s[a] != s[b]:
            return False
        a += 1
        b -= 1
    return True


def palin_index(s):
    i, j = 0, len(s)-1
    while i < j:
        if s[i] != s[j]:
            if is_palin(s, i+1, j):
                return i
            else:
                return j
        i += 1
        j -= 1
    return -1
