#include <iostream>
using namespace std;

int main()
{
    int q,t;
    long long int n;
    cin>>q;
    while(q--)
    {
    	cin>>n;
    	cin>>t;
    	while(t--)
    	{
    		if(n%2==0)
    			n=n/2;
    			else
    				n=n-(n+1)/2;
    			n=n-n/4;
    	}
    	cout<<n<<endl;
    }
    return 0;
}
