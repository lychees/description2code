#include<bits/stdc++.h>
using namespace std;
#define Max 100005
#define ll long long
int ara[Max];
int Sum[Max];
int main()
{
    ll n,m,q;
    while(cin>>n>>m)
    {
        ll A[n+5],B[n+5];
        for(int i=1;i<=n;i++){
            cin>>A[i];
        }
        for(int i=1;i<=n;i++){
            cin>>B[i];
        }
        for(int i=2;i<=n;i+=2){
            swap(A[i],B[i]);
        }
        A[0]=0,B[0]=0;
        for(int i=1;i<=n;i++){
            A[i]+=A[i-1];
            B[i]+=B[i-1];
        }
        ll l,r;
        for(int i=0;i<m;i++){
            cin>>q>>l>>r;
            if(q==1){
                if(l%2==0) cout<<B[r] - B[l-1]<<endl;
                else cout<<A[r] - A[l-1]<<endl;
            }
            else{
                if(l%2==0) cout<<A[r] - A[l-1]<<endl;
                else cout<<B[r] - B[l-1]<<endl;
            }
        }
    }
    return 0;
}
