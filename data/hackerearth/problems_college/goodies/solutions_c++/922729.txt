#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	
	int ntc,n,i,j,a;
	scanf("%d",&ntc);
	while(ntc--)
	{
		
		scanf("%d",&n);
		int arr[10001]={};
		for(i=0;i<n;i++)
		{
		scanf("%d",&a);
		arr[a]++;
		}
		
		for(i=1;;i++)
		{
			
			for(j=1;j<=10000;j++)
			{
				
				if(arr[j]>0)
				{
					arr[j]--;n--;
				}
			}
			if(n==0)
			break;
		}
		
		printf("%d\n",i);
		
		
	}
    return 0;
}
