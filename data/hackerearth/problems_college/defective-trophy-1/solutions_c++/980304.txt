#include <iostream>
#include <cmath>
 
using namespace std;
 
int divide (int n) {
    if (n == 0 || n == 1) return 0;
    return 1 + divide ((int) ceil (n / 3.0));
}
 
int main()
{
   int t,sol,n;
   scanf ("%d", &t);
   while (t--) {
       scanf ("%d", &n);
       sol = divide(n);
       printf ("%d\n", sol);
   }
}
