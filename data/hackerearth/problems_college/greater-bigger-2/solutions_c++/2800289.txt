char getNext(vector<char> &cols,char ch)
{
	int i,n=cols.size();
	int diff=11111111;
	for(i=0;i<n;i++)
	{
		if(cols[i]>ch)
		{
			if(diff>(cols[i]-ch))
				diff=cols[i]-ch;
		}
	}
	if(diff==11111111)
		return -1;
	else
		return ch+diff;
}
