#include <iostream>
#include <vector>
#define endl '\n'

using namespace std;

const long MAX = 1030;

int arr[MAX][MAX];

int winner(int a, int b) {
	// cout << "Winner of (" << a << ", " << b << ") is " << (arr[a][b] ? a : b) << endl;
	return arr[a][b] ? a : b;
}

int main() {
	int n, val;
	vector<int> players;

	cin >> n;

	players.push_back(0); // 1st player

	for (int i = 1; i < (1 << n); ++i)
	{
		players.push_back(i); // Push other players
		for(int j = 0; j < i; j++) {
			cin >> val;
			arr[i][j] = val;
			arr[j][i] = !val;
		}
	}

	while(players.size() != 1) {
		vector<int> next_players;
		for (int i = 0; i < players.size() - 1; i += 2)
		{
			next_players.push_back(winner(players[i], players[i + 1]));
		}
		players = next_players;
	}

	cout << players[0] + 1;
}
