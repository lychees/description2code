#include <iostream>
#include <string.h>
#include <string>
#include <stdint.h>
using namespace std;

int main()
{
   int64_t t;
   string a,b;
   int64_t len;
    cin>>t;
    while(t--){
    	cin>>a;
    	cin>>b;
    	len=a.length();
    	for(int i=0;i<len;i++){
    		if((a[i]=='0' && b[i]=='0') || (a[i]=='1' && b[i]=='1'))
    			cout<<"0";
    		else
    			cout<<"1";
    	}
    	cout<<"\n";
    }
    return 0;
}
