#include <iostream>
#include <cstdio>
#include <climits>
using namespace std;
#define ll unsigned long long
#define l long
ll sum[100005];
int n, m;

int check(ll a) {
	int i, c = 0; 
	ll sum1 = 0;

	for ( i = 1 ; i <= n ; i++ ) {
		sum1 += (sum[i] - sum[i-1]);

		if ( sum1 > a) {
			c++;
			sum1 = sum[i] - sum[i-1];
		}
		if ( sum[i] - sum[i-1] > a )
			return 1000000;
		if ( i == n )
			c++;
	}
	//cout << a << " \n";
	
	return c ;
}

int main() {

	int i;
    ll a, beg, end, mid, ans;
  
    cin >> n >> m;
    sum[0] = 0;
    for ( i = 1 ; i <= n ; i++) {
    	cin >> a;
    	sum[i] = a + sum[i-1];
    }

    beg = 0;
    end = ULONG_MAX;
    while ( beg <= end ) {
    	mid = (beg + end) / 2;

    	if ( check(mid) <= m) {
    		end = mid - 1;
   			ans = mid;
    	}
    	else
    		beg = mid + 1;

    }
    cout << ans << endl;
	return 0;

}
