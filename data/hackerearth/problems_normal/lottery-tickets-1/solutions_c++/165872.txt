#include <iostream>
#include <iomanip> 
using namespace std;

void swap(long long int *x,long long int *y )
{
	long long int temp;
	temp=*x;
	*x=*y;
	*y=temp;
}

int main()
{
    int t,n;
    long long int st[50],et[50];
    long long int prob[50];
    int lis[50],max;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		cin>>st[i];
    		cin>>et[i];
    		cin>>prob[i];
    	}
    	for(int i=0;i<n-1;i++)
    	{
    		for(int j=i+1;j<n;j++)
    		{
    			if(st[i]>st[j])
    			{
    				swap(&st[i],&st[j]);
    				swap(&et[i],&et[j]);
    				swap(&prob[i],&prob[j]);
    			}
    		}
    	}
    	
    	for(int i=0;i<n;i++)
    	{
    		lis[i]=prob[i];
    		for(int j=0;j<i;j++)
    		{
    			if( (et[i]>et[j]) && (et[j]<=st[i]) )
    			{
    				max=lis[j]+prob[i];
    			
    			    if(max>lis[i])
    			   {
    				 lis[i]=max;
    			   }
    			}   
    		}
    	}
    	
    	max=lis[0];
    	for(int i=1;i<n;i++)
    	{
    		if(lis[i]>max)
    		  max=lis[i];
    	}
    	std::cout << std::fixed;
        std::cout << std::setprecision(2);
        std::cout << (double)max/100<<endl;
    	
    }
    return 0;
}
