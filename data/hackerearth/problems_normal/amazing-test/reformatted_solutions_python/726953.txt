import sys

t = input()

while t > 0:
    s = sys.stdin.readline().strip().split()
    s = map(int, s)
    n, x = s
    s = sys.stdin.readline().strip().split()
    s = map(int, s)
    d = {0: 1}
    for i in s:
        dn = {}
        for j in d:
            dn[j+i] = 1
        d.update(dn)
    sm = sum(s)
    fl = 0
    for i in d:
        if i <= x and (sm-i) <= x:
            fl = 1
            break
    if fl == 0:
        print "NO"
    else:
        print "YES"
    # print d
    t = t-1
