#include <bits/stdc++.h>
using namespace std;

class todo
{
	long long dolo;
	public:
	todo(long long d=2): dolo(d){}
	friend bool operator <(const todo& a, const todo& b)
	{
		if(a.dolo<b.dolo)
			return true;
		return false;
	}
	friend long long operator !(const todo& a)
    {
        long long t=a.dolo;
            return t;
    }
	friend ostream& operator <<(ostream& os, const todo& a)
	{
		os<<a.dolo<<" ";
	}
};

int main()
{
	priority_queue<todo> todol;
	long long n;
	cin>>n;
	for(long long i=0;i<n;i++)
	{
		long long t;
		cin>>t;
		todol.push(todo(t));
		if(todol.size()<3)
		{
			cout<<"-1"<<endl;
			continue;
		}
		long long z[3];
		for(long long j=0;j<3;j++)
		{

			z[j]=!todol.top();
			//cout<<z[j];
			todol.pop();
		}
		long long ans=z[0]*z[1]*z[2];
		for(long long j=0;j<3;j++)
		{
			todol.push(z[j]);
		}
		cout<<ans<<endl;
	}

    //cout << "Hello World!" << endl;
    return 0;
}
