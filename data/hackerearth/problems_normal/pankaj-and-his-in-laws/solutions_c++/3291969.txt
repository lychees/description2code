using namespace std;

void in(int &x){
    register int c=gc();
    x=0;
    for(;(c<48||c>57);c=gc());
    for(;c>47&&c<58;c=gc()){x=(x<<1)+(x<<3)+c-48;}
}

int n,a[105],lis[105],b[105];

int solve(){
	int i,j,ans=0;
	for(i=0;i<n;i++) lis[i]=1;
	for(i=1;i<n;i++)
		for(j=0;j<i;j++)
			if(a[i]>a[j]&&lis[i]<lis[j]+1)
				lis[i]=lis[j]+1;
	for(i=0;i<n;i++) ans=max(ans,lis[i]);
	return ans;
}

int main()
{
	in(n);int i,shit;
	for(i=0;i<n;i++) ii(a[i]);
	shit=solve();i=0;
	while(shit>0){
		b[i++]=shit%2;
		shit>>=1;
	}i--;
	while(i>=0){
		printf("%d", b[i]);
		i--;
	}
    
    return 0;
}
