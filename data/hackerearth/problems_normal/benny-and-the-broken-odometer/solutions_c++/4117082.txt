using namespace std;

typedef pair<int,int> p;
typedef pair<int,p> pii;

long long int count1[15];
long long dp[15];

void init()
{
	count1[1] = 1;
	int temp = 10;
	for(int i = 2 ; i < 11 ; i++)
	{
		count1[i] = 9*count1[i-1] + temp;
		temp *= 10;
	}
	dp[1] = 1;
	for(int i = 2 ; i < 11 ; i++)
	{
		dp[i] = count1[i];
	}
}

void solve()
{
	vector<int> v;
	int N;
	int tN;
	scanf("%d",&N);
	tN = N;
	long long temp = 1;
	while(N > 0)
	{
		v.pb(N%10);
		N /= 10;
		temp *= 10;
	}
	temp /= 10;
//	temp /= 10; 
	long long sub = 0;
	for(int i = v.size()-1 ; i >= 0 ; i--)
	{
		int n = v[i];
		for(int j = 0 ; j < n ; j++)
		{
			if(j==3)
				sub += temp;
			else
				sub += dp[i];
		}
		temp /= 10;
	}
	printf("%lld\n",tN-sub);
}

int main()
{
	int T;
	init();
	scanf("%d",&T);
//	T = 1;
	while(T--)
	{
		solve();
	}
	return 0;
}
