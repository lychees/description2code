#include <iostream>
using namespace std;
#include<vector>
int main()
{
   int n;
   cin>>n;
   for(int i=1;i<=n;i++){
   	vector<int> v;
   	
   	int k;
   	cin>>k;
   	int n1=k;
   	for(int g=0;g<k;g++){
   	int s;
   	cin>>s;
   	v.push_back(s);
   }
   	for(int g=0;g<k;g++){
   			for(int d=g+1;d<k;d++)
   			{
   	if(v[d]==v[g])
   	n1++;
   			}
   
   }
   cout<<n1<<"\n";
}
return 0;}
