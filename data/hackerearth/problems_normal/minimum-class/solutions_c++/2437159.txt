#include <iostream>
#include<cmath>

using namespace std;

bool sq(int n) {
    if (n < 0)
        return false;
    int root(round(sqrt(n)));
    return n == root * root;
}

int  main(){
	int t;
	cin>>t;
	for(int i=0;i<t;i++){
		int n;
		cin>>n;
		if(sq(n)){
			cout<<2*sqrt(n) - 1<<'\n';
		}
		else{
			int p = sqrt(n);
			int q = (p+1)*(p+1);
			if( q - p >n){
				cout<<2*(p+1) - 2<<'\n';
			}
			else{
				cout<<2*(p+1) - 1<<'\n';
			}
		}
	}
}
