import math
tc= int(raw_input())
while(tc):
	x1,y1,x2,y2 = map(int,raw_input().split())
	r = int(raw_input())
	d1 = x2-x1
	d2 = y2-y1
	D = x1*y2-x2*y1
	res1 = math.sqrt(d1*d1 + d2*d2)
	res2 = r*r*res1*res1-D*D
	if res2 < 0:
		print "SAFE"
	elif res2 == 0:
		print "JUST MISSED"
	else:
		print "REPLANNING"
	  
	tc = tc - 1
