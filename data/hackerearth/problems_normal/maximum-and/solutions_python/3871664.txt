
tc = int(raw_input())

while(tc > 0):
    s = raw_input()
    s = s.split()
    l = [int(c) for c in s]
    num = l[1]
    ans = 0
    d = l[1] - l[0]
    if d > 10:
        for i in range(0,10):
            tmp = (num-i) & (num-i-1)
            if tmp > ans:
                ans = tmp
    else:
        for i in range(0,d):
            tmp = (num-i) & (num-i-1)
            if tmp > ans:
                ans = tmp

    print (ans)
    tc-=1
