#include <cstdio>
#include <iostream>
#include <string>
using namespace std;

int previ,curr,times;

void get_num(char x){
	int num = (int)	 x; // ascii conversion
	previ = curr;
	if(num == 95){
		curr = 0;	times = 1;
	}
	else if(num> 118){ // wxyz
		curr = 9;	times = num - 118;
	}
	else if(num> 115){ // tuv
		curr = 8;	times = num - 115;
	}
	else if(num> 111){ // pqrs
		curr = 7;	times = num - 111;
	}
	else if(num> 108){ // mno
		curr = 6;	times = num - 108;
	}
	else if(num> 105){ // jkl
		curr = 5;	times = num - 105;
	}
	else if(num> 102){ // ghi
		curr = 4;	times = num - 102;
	}
	else if(num> 99){ // def
		curr = 3;	times = num - 99;
	}
	else if(num> 96){ // abc
		curr = 2;	times = num - 96;
	}
	else if(num<= 57 && num>= 48){ // 0-9
		curr = num - 48;
		times = 4;
		if(curr == 0)	times = 2;
		else if(curr == 1 || curr == 7 || curr == 9){
			times = 5;
		}
	}
	else{
		curr = 1;
		switch (x){
			case '.' :
				times = 1;
				break;
			case ',' :
				times = 2;
				break;
			case '?' :
				times = 3;
				break;
			case '!' :
				times = 4;
				break;				
		}
	}
}

int main()
{
    int t,ans,len;
    scanf("%d",&t);
    for(int x=0;	x<t;	x++){
    	string line;
    	cin >> line;
    	ans = 0;
    	len = line.length();
    	curr = 1;
    	for(int i=0;	i<len;	i++){
    		get_num(line[i]);
    		if(curr != previ)	ans++;
    		ans += times;
    	}
    	printf("%d\n",ans);
    }
    return 0;
}
