#include <bits/stdc++.h>

using namespace std;

#define ll long long int
#define md 1000000007

ll power(ll a, ll b) {
	if (b == 0) return 1;
	if (b == 1) return a % md;
	
	ll temp = power(a, b/2);
	temp = temp % md;
	ll temp1 = (temp * temp) % md;
	
	if (b&1) return ( temp1 * (a%md) ) % md;
	else return temp1;
}

int main()
{
	ll t, n;
	cin >> t;
	
	while(t--) {
		cin >> n;
		cout << power(2, n) - 1 << endl;
	}
	return 0;
}
