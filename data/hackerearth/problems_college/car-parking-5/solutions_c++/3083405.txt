#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    int t,l,c;
    char s[100];
    cin>>t;
    while(t--)
    {
    	c=0;
    	cin>>s;
    	l=strlen(s);
    	int a[l];
    	for(int i=0;i<l;i++)
    	a[i]=0;
    	for(int i=0;i<l;i++)
    	{
    		if(s[i]=='B')
    		{
    			a[i]=1;
    			if((i-2)>=0)
    			{
    				a[i-1]=1;
    				a[i-2]=1;
    			}
    			else if((i-1)>=0)
    			{
    				a[i-1]=1;
    			}
    		}
    		else if(s[i]=='S')
    		{
    			a[i]=1;
    			if((i-1)>=0)
    			{
    				a[i-1]=1;
    			}
    			if((i+1)<l)
    			a[i+1]=1;
    		}
    		else if(s[i]=='D')
    		a[i]=1;
    		
    	
    	}
    	for(int i=0;i<l;i++)
    		{
    			if(!a[i])
    			c++;
    		}
    	cout<<c<<endl;
    }
    return 0;
}
