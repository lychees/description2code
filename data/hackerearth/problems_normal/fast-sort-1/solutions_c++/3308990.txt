#include<iostream>
#include<algorithm>
using namespace std;
char a[100005];int b[100004];int c[100005];
int main(){
	int t;
	cin>>t;string s1,s2;
	for(long i=0;i<t;i++){
		cin>>s1>>s2;
		for(long j=0;j<s1.size();j++){
			int d=s1[j]-'a';
			c[d]=j;
			a[j]=s1[j];
		}
		for(long k=0;k<s2.size();k++){
			int g=s2[k]-'a';
			b[k]=c[g];
		}
		sort(b,b+s2.size());
		for(long k=0;k<s2.size();k++){
			cout<<a[b[k]];
		}
		cout<<endl;
	}
}
