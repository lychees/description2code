#include <iostream>
using namespace std;
#include<string>
int main()
{
	long long int i,l,cnt,ans=0;
    string s="";
    cin>>s;
    char st;
    l=s.length();
    i=1;
    cnt=1;
    st=s[0];
    while(s[i]!='\0') {
    	if(s[i]==st)
    		cnt++;
    	else {
    		st=s[i];
    		ans+=(cnt*(cnt+1)/2);
    		cnt=1;
    	}
    	i++;
    }
    ans+=(cnt*(cnt+1)/2);
    cout<<ans<<"\n";
    return 0;
}
