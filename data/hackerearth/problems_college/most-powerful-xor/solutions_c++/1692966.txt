#include <iostream>
using namespace std;
int maximumXOR(int l, int r) {
    int q = l ^ r, a = 1;
    while(q){
        q /= 2;
        a <<= 1;
    }
    return --a;
}
int main()
{
	int t; cin >> t;
	while(t--)
	{

     	cout << maximumXOR(l,r) << endl;
	}
    return 0;
}
