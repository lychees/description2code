count = 0
def bubbly2(s):
	if len(s) %2 != 0:
		return False
	l = []
	for ch in s:
		if not l:
			l.append(ch)
		else:
			if l[-1] == ch:
				del l[-1]
			else:
				l.append(ch)
	return l == []
		

for t in xrange(int(raw_input())):
	res = bubbly2(raw_input())
	#print res
	if res:
		count += 1
