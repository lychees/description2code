#include<iostream>

using namespace std;

int main()

{
    int n,q,x,y,i,j,a[1001][3]={0};
    cin>>n>>q;
    for(i=0;i<q;i++)
    {
        cin>>x>>y;
        
        a[i][0]=x;
        a[i][1]=y;
        if(x>y)
        {
        	int temp;
        	temp=a[i][0];
        	a[i][0]=a[i][1];
        	a[i][1]=temp;
        	
        }
        int flag=1;
        if(x==y)
        {
            flag=0;
        }
        for(j=0;j<i&&flag==1;j++)
        {
            if(a[j][2]==1){
            if(a[i][1]==a[j][0]||a[i][0]==a[j][0]||a[i][0]==a[j][1]||a[i][1]==a[j][1])
            {
                flag=0;
                break;
            }
            if(a[i][0]<a[j][0]&&a[i][1]>a[j][0]&&a[i][1]<a[j][1]||a[i][0]>a[j][0]&&a[i][0]<a[j][1]&&a[i][1]>a[j][1])
            {
                flag=0;
                break;
            }
            }
        }
        if(flag==1)
        {
            a[i][2]=1;
            cout<<"YES"<<endl;
        }
        else
        {
            a[i][2]=0;
            cout<<"NO"<<endl;
        }
    }
    return 0;
}
