#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000000007
#define bitcount __builtin_popcountll
int b[15][27];
int selno[15];
int w[10005];
int po[15];
int x;
int n,k;
char c,a[15];
int ways()
{
    int pos=bitcount(x);
    //printf("%d\n",pos);
    int i;
    if(pos==k){
        w[x]=1;
        //pos--;
        return 1;
    }
    /*if(w[x]!=0){
        //pos--;
        return w[x];
    }*/
    int ans=0;
    for(i=0;i<n;i++)
    {
        if(selno[i]==0&&b[i][(int)(a[pos]-'a')]>0)
        {
            //printf("%d %d\n",(a[pos]-'a'),b[i][(int)(a[pos]-'a')]);
            selno[i]=1;
            x+=po[i];
            //printf("%d\n",i);
            //pos++;
            ans=(ans+(b[i][(int)(a[pos]-'a')]*ways())%mod)%mod;
           // printf("%d %d %d %d %d\n",pos,b[i][(int)(a[pos-1]-'a')],(int)(a[pos-1]-'a'),i,ways());
            //pos--;
            x-=po[i];
            selno[i]=0;
            w[x]=(w[x]+ans)%mod;
            //printf("%d %d\n",i,ans);
        }
    }

    return ans;
}
int main()
{
    int i,j;
    sd(n);
    sd(k);
    po[0]=1;
    for(i=1;i<=10;i++)
        po[i]=po[i-1]*2;
    for(i=0;i<n;i++)
    {
        for(j=0;j<6;j++)
        {
            c=getchar();
            c=getchar();
            //printf("%c\n",c);
            b[i][(int)(c-'a')]++;
        }
    }
    ss(a);
    int ans=ways();
    printf("%d\n",ans%mod);
}
