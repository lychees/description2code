#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int k,a,b,c;
    	cin>>k;
    	cin>>a>>b>>c;
    	a=a+b+c;
    	long long int sum=1,cnt=0;
    	while(cnt!=k)
    	{
    		sum=sum*a;
    		cnt++;
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
