
#include <iostream>
using namespace std ;
int main()
{

    int t ;
    cin >> t ;
    while ( t--)
    {
        int n ;
        cin >>  n ;
        int m ;
        m = n ;
        int i ;
        while (n %3  != 0)
        {
            n = n -5 ;
        }
        if ( n < 0)
        {
            cout << "-1" << endl;

        }
        else{
            for ( i = 0; i< n ; i++)
            {
                cout << "5" ;
            }
            for ( i = 0; i<m-n ; i++)
            {
                cout << "3" ;
            }
            cout << endl;
        }


    }
    return 0 ;
}
