#include<bits/stdc++.h>
using namespace std;
#define X first
#define Y second
#define PB push_back
#define MP make_pair
#define MAX 100005
#define scanl(n) scanf("%lld",&n)
typedef long long int ll;
int main()
{
	ll test,n,arr[MAX],i,flag;
	cin>>test;
	while(test--){
		flag=0;
		cin>>n;
		for(i=0;i<n;i++)
			cin>>arr[i];
		i=0;
		while(n--){
		//	cout<<i<<" "<<arr[i]<<" ";
			
			i=arr[i];
			
			if(i==-1){
				flag++;
				break;
			}
			
		}
		if(flag==0)
			cout<<"Lose\n";
		else
			cout<<"Win\n";
	}
	return 0;
}
