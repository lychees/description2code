#include <bits/stdc++.h>
using namespace std;
int dp[1000009];
int arr[] = {2,3,5,7};
int recur(int sum) {
	if(sum == 0) {
		return 0;
	}
	if(sum < 0) {
		return 1e9;
	}
	if(dp[sum]!= -1) {
		return dp[sum];
	}
	int ans= 1e9;;
	for(int i= 0; i < 4; i++) {
		ans = min(ans, 1+recur(sum - arr[i]));
	}
	return dp[sum] = ans;
}
int main()
{
    int t;
    cin >> t;
    while(t--) {
    	memset(dp,-1,sizeof dp);
    	int n;
    	cin >> n;
    	int ans= recur(n);
    	if(ans == 1e9) {
    		ans = -1;
    	}
    	cout << ans << endl;
    }
    return 0;
}
