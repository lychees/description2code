t=input()
while t:
	t-=1
	n=input()
	l=[]
	l=map(int, raw_input().split())
	l.sort()
	c=0
	while n>0:
		a=(l[0]+1) in l
		b=(l[0]-1) in l
		
		if a==b and b==False:
			c=c+1
			n=n-1
			l.remove(l[0])
		elif a==False and b==True:
			c=c+1
			n=n-2
			l.remove(l[0]-1)
			l.remove(l[0])

		elif a==True and b==False:
			c=c+1
			n=n-2
			l.remove(l[0]+1)
			l.remove(l[0])
	
	print c
