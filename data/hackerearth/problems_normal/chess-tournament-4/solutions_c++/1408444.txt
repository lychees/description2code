#include<iostream>
#include<math.h>
using namespace std;

int main()
{
	int n,range;
	cin>>n;
	range=pow(2,n);
	int matrix[range+1][range+1];
	int win[range+1];
	for(int i=2;i<=range;i++)
		{
			for(int j=1;j<i;j++)
				cin>>matrix[i][j];
		}
	for(int i=1;i<=pow(2,n);i++)
		win[i]=i;
	int k=2;
	int m;
	for(int k=1;k<=n;k++)
	{
		m=1;
		for(int i=1;i<=pow(2,n)/k;i+=2)
		{
			if(matrix[win[i+1]][win[i]]==1)
			{
				win[m]=win[i+1];
			}
			else
				win[m]=win[i];
			m++;
			/*
			cout<<"Win matrix"<<endl;
			for(int j=1;j<=range;j++)
				cout<<win[j]<<endl;
			*/
		}	
	}
	cout<<win[1]<<endl;;
	return 0;
}
