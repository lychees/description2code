typedef long long ll;
typedef pair<int,int> pii;

const int N = 510;
const int M = 10;
const ll MOD = 1e9 + 7;

int n, m, k;
ll dp[N][N*5];
ll ways[M];

int main(){
	sd3(n,m,k);
	
	for(int i = 0; i < (1<<m); i++){
		int sz = __builtin_popcount(i);
		if(sz < 2 or sz > (m+1)/2) continue;
		bool flag = true;
		for(int j = 0; j < m; j++){
			if((i&(1<<j)) > 0 and (i&(1<<(j-1))) > 0){
				flag = false;
				break;
			}
		}
		if(flag){
			ways[sz]++;
		}
	}

	dp[0][0] = 1;
	for(int i = 1; i <= n; i++){
		for(int j = 2*i; j <= 5*n; j++){
			for(int l = 2; l <= m and l <= j; l++){
				dp[i][j] += ways[l] * dp[i-1][j-l] % MOD;
				if(dp[i][j] >= MOD) dp[i][j] -= MOD;
			}
		}
	}
	
	ll ans = 0;
	for(int i = k; i <= 5*n; i++){
		ans = (ans + dp[n][i]) % MOD;
	}
	cout << ans << endl;
	
	
	
	return 0;
}
