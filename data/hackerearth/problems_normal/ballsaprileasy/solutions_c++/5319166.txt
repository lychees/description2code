#include <bits/stdc++.h>
using namespace std;

int main()
{
   int n;
   cin>>n;
   long long int a[n];
   for(int i=0;i<n;i++)
   {
   		cin>>a[i];
   }
   sort(a,a+n);
   long long int ans=0;
   long long int i=0,layer=1;
   while(i < n)
   {
   	    if(layer*layer<a[i])
   	    {
   	    	ans+=(layer*layer);
   	    }
   	    else
   	    {
   	    	ans+=((layer*layer)-a[i]);
   	    	i++;
   	    }
   	    layer++;
   }
   cout<<ans<<endl;
}
