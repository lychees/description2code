int f(string a, string b, string c){
	string s = a + b + c;
	int n = s.length();
	int i = a.length() - 1, j = 0;
	int ln = 0;
	while (i >= 0 && j < b.length() && a[i] == b[j]){
		++ln; --i; ++j;
	}
	n -= ln;
	ln = 0;
	i = b.length() - 1; j = 0;
	while (i >= 0 && j < c.length() && b[i] == c[j]){
		++ln; --i; ++j;
	}
	n -= ln;
	return n;
}

int main(){
	string s1, s2, s3;
	cin >> s1 >> s2 >> s3;
	int ans = f(s1, s2, s3);
	ans = min(ans, f(s1, s3, s2));
	ans = min(ans, f(s2, s1, s3));
	ans = min(ans, f(s2, s3, s1));
	ans = min(ans, f(s3, s1, s2));
	ans = min(ans, f(s3, s2, s1));
	cout << ans; el;
	return 0;
}
