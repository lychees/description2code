#include <bits/stdc++.h>
using namespace std;

int main()
{
	
	int t,n,i,j,score,flag;
	
    cin>>t;
    while(t--){
    	
    	vector<pair<int,int> >v;
		vector<pair<int,int> >v1;
		int c1[30]={0};
		string s,a[100002];
    	cin>>n>>s;
    	for(i=0;i<n;i++)
    		cin>>a[i];
    		
    	for(i=0;i<s.size();i++){
    		
    		c1[s[i]-'a']++;
    	}
    	for(i=0;i<n;i++){
			score=0;
    		for(j=0;j<a[i].size();j++){
    			score+=c1[a[i][j]-'a'];
    		
    		}
    		v.push_back(make_pair(score,i));
    		score=0;
    	}
    	
    	sort(v.begin(),v.end());
    	flag=0;
    	for(i=v.size()-1;i>=0;i--){
    		
    		if(v[i].first==v[i-1].first){
    			v1.push_back(make_pair(a[v[i].second].size(),v[i].second));
    		}
    		else{
    			v1.push_back(make_pair(a[v[i].second].size(),v[i].second));
    			flag=1;
    			break;
    		}
    	}
    	if(flag==0){
    		if(v[v.size()-2].first==v[v.size()-1].first){
    			v1.push_back(make_pair(a[v[v.size()-1].second].size(),v[v.size()-1].second));
    		}
    	}
    	if(v1.empty()){
    		cout<<v[v.size()-1].second+1<<endl;
    	}
    	else{
    		sort(v1.begin(),v1.end());
    		int ans=v1[0].second+1;
    		cout<<ans<<endl;
    	}
    }
    return 0;
}
