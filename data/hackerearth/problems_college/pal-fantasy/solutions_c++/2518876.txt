#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t,i,j,k,l;
    scanf ("%d",&t);
    char str[100];
    getchar();
    while (t--)
    {
    	scanf ("%s",str);
    	l=strlen(str);
    	int fmax=0,flag;
    	for (i=0;i<l;++i)
    	{
    		flag=0;
    		for (j=l-1;j>=i;--j)
    		{
    			if (str[j]!=str[l-1-j+i])
    			{
    				flag=1;
    				break;
    			}
    		}
    		if (flag==0)
    		{
    			if ((l-i)>fmax)
    				fmax=(l-i);
    		}
    	}
    	l+=(l-fmax);
    	printf ("%d\n",l);
    }
    return 0;
}
