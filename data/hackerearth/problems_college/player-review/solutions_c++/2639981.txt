#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,i,j,k,a[100001],me=0,ms=0,c=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		if(a[i]<0)
			c++;
	}
	if(c==n)
		cout<<"0\n";
	else
	{
		for(i=0;i<n;i++)
		{
			me=me+a[i];
			if(me<0)
				me=0;
			if(ms<me)
				ms=me;
		}
		cout<<ms<<"\n";
	}
}
