#include<iostream>
#include<cstdio>
#define ll long long
#include<cstring>
#include<algorithm>
#include<cmath>
#define mod 10000007
ll arr[101][101];
void solve()
{
	ll i,j;
	for(i=0;i<101;i++)
	{
		arr[i][0]=1;
		arr[i][i]=1;
	}
	for(i=1;i<101;i++)
	{
		for(j=1;j<i;j++)
			arr[i][j]=(arr[i-1][j-1]+arr[i-1][j])%mod;
	}
}
using namespace std;
int main()
{
	ll n,r,i,j;
	solve();
	scanf("%lld%lld",&n,&r);
	if(n<r)
		printf("-1");
	else
	{
		n-=r;
		n=n+r-1;
		r--;
		printf("%lld",arr[n][r]);
	}
	return 0;
}
