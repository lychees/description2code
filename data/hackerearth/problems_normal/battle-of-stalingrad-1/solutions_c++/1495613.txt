#include <iostream>
using namespace std;
long double area(int x1,int y1,int x2,int y2,int x3,int y3)
{
	long double res =  ( ( x1*(y2-y3) + x2*(y3-y1) + x3*(y1-y2) )/2.0 );
	if(res<0)
	res=res*(-1);
	return res;
	
}
bool issafe(int x1,int y1,int x2,int y2,int x3,int y3,int x4,int y4)
{
	long double large_area=area(x1,y1,x2,y2,x3,y3);
	long double area1=area(x1,y1,x2,y2,x4,y4);
	long double area2=area(x3,y3,x2,y2,x4,y4);
	long double area3=area(x1,y1,x3,y3,x4,y4);
	return(large_area==area1+area2+area3);

}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		 int x1,y1,x2,y2,x3,y3,x4,y4;
		cin>>x1>>y1>>x2>>y2>>x3>>y3>>x4>>y4;
		if(issafe(x1,y1,x2,y2,x3,y3,x4,y4))
		{
			cout<<"INSIDE"<<endl;
		}
	else
		{
			cout<<"OUTSIDE"<<endl;
		}
	}
    //cout << "Hello World!" << endl;
    return 0;
}
