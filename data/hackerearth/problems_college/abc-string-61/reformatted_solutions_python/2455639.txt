def apowerb(a, b):
    b = int(bin(b)[2:])
    s = 1
    p = a
    while (b != 0):
        if (b % 10 == 1):
            s = s*p
        p = p*p
        p = getmod(p)
        s = getmod(s)
        b = b/10
    return s
