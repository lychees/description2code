#include <iostream>
using namespace std;
long has[1000015]={0};
int main()
{
   long n;
   cin>>n;
   //long a[n];
   long a;
   
   for(int i =0 ;i<n;i++)
   {
   	cin>>a;
   	has[a]++;
   }
   int n1;
   cin>>n1;
   //cin>>n;
   for(int i=0;i<n1;i++)
   {
   	cin>>a;
   	has[a]--;
   }
   
   for(int i=0;i<1000015;i++)
   {
   		if(has[i]<0)cout<<i<<" ";
   		
   }
   cout<<endl;
    return 0;
}
