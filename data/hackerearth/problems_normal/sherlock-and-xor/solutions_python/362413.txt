_FAC_TABLE = [1, 1]
def f(n):
    if n < len(_FAC_TABLE):
        return _FAC_TABLE[n]

    last = len(_FAC_TABLE) - 1
    total = _FAC_TABLE[last]
    for i in range(last + 1, n + 1):
        total *= i
        _FAC_TABLE.append(total)

    return total
    
def ncr(n,r):
	p = n-r
	y = f(n)/f(r)/f(p)
	return y
