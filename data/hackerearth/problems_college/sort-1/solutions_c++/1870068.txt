#include <iostream>
#include <vector>

using namespace std;

#define max 100000

int T;
int in[max];


void quicksort(int x[max],int first,int last){
    int pivot,j,temp,i;
    if(first<last){
        pivot = first;
        i=first;
        j=last;
        while(i<j){
            while(x[i]<=x[pivot] && i<last)
                i++;
            while(x[j]>x[pivot])
                j--;
            if(i<j){
                temp=x[i];
                x[i]=x[j];
                x[j]=temp;
            }
        }
        temp=x[pivot];
        x[pivot]=x[j];
        x[j]=temp;
        quicksort(x,first,j-1);
        quicksort(x,j+1,last);
    }
}
int main()
{
	
	//int swap;
	//memset(in, 0, sizeof(int));
	cin>>T;
	int N[T];
	while(T>0)
	{
		cin>>N[T];
		for(int i=0;i<N[T];i++)
			cin>>in[i];
		
		quicksort(in, 0, N[T]-1);
        
    	for(int k = 0; k<N[T]; k++)
    		cout<< in[k] <<" ";
    	
    	cout << endl;
    	T--;
	}		
    return 0;
}
