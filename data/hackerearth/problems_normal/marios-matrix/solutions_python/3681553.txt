import itertools
t = int(raw_input())
for i in range(t):
    c = 0
    a = list()
    n = list(map(int, raw_input().split()))
    for j in range(n[0]):
        b = list(map(int, raw_input().strip()))
        a.append(b)
    a.sort()
    b = list(a for a,_ in itertools.groupby(a))
    if len(a) == len(b):
        print("Yes")
    else:
        print("No")
