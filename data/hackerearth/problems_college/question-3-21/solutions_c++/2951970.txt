#include<math.h>
int prime(int x)
{
	int count=0,flag;
	for(int i=2;i<=x;i++)
	{
		flag=1;
		for(int j=2;j<=sqrt(i);j++)
		{
			if(i%j==0)
			{	
				flag=0;
				break;
			}	
		}
		if(flag)
			count++;
	}
	return count;
}	
int main()
{
    int n,t,x,k;
    cin>>n>>t;
    while(t--)
    {
    	cin>>x;
    	k=prime(x);
    	cout<<k<<endl;
    }
    return 0;
}
