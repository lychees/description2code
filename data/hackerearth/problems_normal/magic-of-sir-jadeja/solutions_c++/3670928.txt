#include <bits/stdc++.h>
#define inf 1000000000
using namespace std;

int abs(int n) {
    if (n < 0) return -n;
    return n;
}

int main()
{
    int t, k;
    string s[7] = {"Rohit", "Dhawan", "Kohli", "Yuvraj", "Raina" , "Dhoni", "Sir Jadeja"};

    cin >> t;
    while (t--) {
        cin >> k;

        if (k <= 7) {
            cout << s[k - 1] << endl;
            continue;
        }
        int d = 0;
        int tmp = 7;
        while (k >= tmp) {
            k -= tmp;
            tmp *= 2;
            d++;

        }
        k--;
        cout << s[k / (1<<d)] << endl;
    }

    return 0;
}
