#include<bits/stdc++.h>
using namespace std;

int main(){
 int t;
scanf("%d",&t);
while(t--){
long long int b,w,x,y,z;
long long int cost = 0;

scanf("%lld %lld %lld %lld %lld",&b,&w,&x,&y,&z);
if(x<y && y>(z+x))
cost = b*x + w*x + w*z;
else if(x>y && x>z+y)
cost = b*y + w*y + b*z;
else
cost = b*x + w*y;

printf("%lld\n",cost);

}
return 0;
}
