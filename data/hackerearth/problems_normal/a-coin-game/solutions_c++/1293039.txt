int main() {
	set<i_i> m;
	vector<bool> a(1000001);
	int j = 0;
	for (int i = 0; i + j <= 1000000; i++)
		if (!a[i] && !a[i + j]) {
			a[i] = a[i + j] = true;
			m.insert(i_i(i, i + j));
			j++;
		}
	int T; cin >> T;
	while (T--) {
		int N1, N2; scanf("%d%d", &N1, &N2);
		if (N1 > N2) swap(N1, N2);
		cout << (m.count(i_i(N1, N2)) ? "Don't Play" : "Play") << endl;
	}
}
