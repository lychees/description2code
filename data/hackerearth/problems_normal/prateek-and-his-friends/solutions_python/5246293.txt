t = int(raw_input())
while t > 0 :
    t-=1
    n,x = map(int , raw_input().split())
    bad = 1
    l = [0] * n
    for i in range(n):
        l[i] = int(raw_input())
    s = 0
    b = 0
    for i in range(n):
        s+= l[i]
        while s > x :
            s-=l[b]
            b+=1
        if s == x :
            bad = 0
            break

    if bad:
        print 'NO'
    else:
        print 'YES'
