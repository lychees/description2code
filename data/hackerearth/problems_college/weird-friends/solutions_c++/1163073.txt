//Dance Floor
#include <iostream>
using namespace std;


int main(){
	int JackX;
	int JackY;
	int JackCoordinate=0;
	int danceFloor;
	int music=0;
	cin>>danceFloor;
	cin>>JackX;
	cin>>JackY;
	while(JackCoordinate<=danceFloor){
		JackCoordinate+=JackX;
    music++;
    if(JackCoordinate >= danceFloor)
      break;
    JackCoordinate-=JackY;
	}
	cout<<music;
}
