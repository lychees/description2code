#include<bits/stdc++.h>
using namespace std;
vector<int>ar[100000];
int visited[100000];
void dfs(int u)
{
	visited[u]=1;
	for(int i =0;i<ar[u].size();i++)
	{
		if(visited[ar[u][i]]==0)
		{
			dfs(ar[u][i]);
		}
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i;
		cin>>n;
		int m;
		cin>>m;
		for(i=0;i<=n;i++)
		{
			ar[i].clear();
		}
		int b,c;
		for(i=0;i<m;i++)
		{
			cin>>b>>c;
		    ar[b].push_back(c);
		    ar[c].push_back(b);
		}
		memset(visited,0,sizeof(visited));
		int ans=0;
		for(i=0;i<n;i++)
		{
			if(visited[i]==0)
			{
				ans++;
				dfs(i);
			}
		}
		cout<<ans<<"\n";
	}
}
