#include<iostream>
#include <iomanip>
using namespace std;

int main()	{
	char time[5];
	int hour,min;
	cin >> time;
	hour = (time[0]-'0')*10 + (time[1]-'0');
	min = (time[3]-'0')*10 + (time[4]-'0');
	if ( (hour <= 12 && hour > 0) && (min <= 60 && min >= 0) )	{
		float mindeg = min*6.0;
		float hourdeg = (hour*60 + min)/2.0;
		if (mindeg > hourdeg)	{
			if(mindeg-hourdeg > 180)	{
				mindeg=360-mindeg;
				cout << setiosflags(ios::fixed) << setiosflags(ios::showpoint) << setprecision(6) << (mindeg + hourdeg);
			}
			else
				cout << setiosflags(ios::fixed) << setiosflags(ios::showpoint) << setprecision(6) << (mindeg - hourdeg);
		}
		else	{
			if(hourdeg-mindeg > 180)	{
				hourdeg=360-hourdeg;
				cout << setiosflags(ios::fixed) << setiosflags(ios::showpoint) << setprecision(6) << (mindeg + hourdeg);
			}
			else
				cout << setiosflags(ios::fixed) << setiosflags(ios::showpoint) << setprecision(6) << (hourdeg - mindeg);
		}
	}

	return 0;
}
