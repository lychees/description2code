#include <bits/stdc++.h>
using namespace std ;

#define sc(x) scanf("%d",&x)
#define sc2(x,y) scanf("%d%d",&x,&y)
#define sc3(x,y,z) scanf("%d%d%d",&x,&y,&z)


long long int mod = 1000000007;
long long int pow2(long long int a , long long int b){
	long long int res=1;
	while(b){
		if(b&1) res=(res*a)%mod;
		a= (a*a)%mod;
		b = b>>1;
	}
	return res;
}

int main(){
	int t;
	sc(t);
	while(t--){
		vector<vector<long long int> > adj(10001);	
		long long int max1=INT_MIN, len1=INT_MAX, min1=INT_MAX, len2=INT_MIN, ans=1;
		long long int k1;
		int n,k;
		int i,j;
		cin >> n >> k;
		for(i=0;i<=k-1;i++){
		for(j=0;j<=n-1;j++) {
			long long int l;
			cin >> l;
			adj[i].push_back(l);
			}
		}
	
	
	for(i=1;i<1<<n;i++){
		long long int len=0, sum=1;
		for(j=0;j<n;j++){
			long long int sum1=0;
			if(i&(1<<j)){
				for(k1=0;k1<=k-1;k1++)
					{	
						sum1+=(adj[k1][j]);
						sum1%=mod;
						
					}
				len++;
				sum*=sum1;
				if(sum>=mod)
				sum%=mod;
			}
			
		}
			ans = sum*pow2(len,mod-2)%mod;
			if(ans>=mod)
				ans%=mod;
			if(min1>ans||(min1==ans && len>len2)){ 
				min1 = ans;len2=len;
			}
			if(max1<ans||(max1==ans && len<len1)){ 
				max1 = ans;len1=len;
			}
		
		}
		printf("%lld %lld \n",(max1^min1),(len1^len2));
	}
}
