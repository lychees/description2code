#include <bits/stdc++.h>
using namespace std;
 
int main() {
	// your code goes here
	int t,g;
	long long m;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%lld",&g,&m);
		int a[g];
		for(int i=0;i<g;i++)
		scanf("%d",&a[i]);
		sort(a,a+g);
		int cnt=0;
		unsigned long long sum=0;
		for(int i=0;i<g;i++)
		{
			sum=sum+a[i];
			if(sum<=m)
			cnt++;
			else
			break;
		}
		printf("%d\n",cnt);
	}
	return 0;
}
