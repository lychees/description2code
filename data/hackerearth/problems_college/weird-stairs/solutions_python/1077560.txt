
def solve():	
	[W,N]=[int(x) for x in raw_input().split()]
	ans = W*N
	#print ans
	a=[int(x) for x in raw_input().split()]
	
	i=N-1
	while i>=0:
		if a[i] == 1:
			break
		i -= 1
	
	cur = 2*(i+1)
	i += 1
	
	
	while i<N:
		cur = cur + 1
		i += 1
		
	print ans + cur




def main():
	for _ in xrange(int(raw_input())):
		solve()
		
main()
