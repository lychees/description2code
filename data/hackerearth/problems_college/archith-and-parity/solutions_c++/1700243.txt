#include <iostream>
#include <bitset>

int main(){
	ios::sync_with_stdio(false);
	int T=0;
	
	cin >> T;
	
	while(T--)
	{
	 int in=0;
	 cin >> in;
	 bitset<30> bin(in);
	 int count=0;
	 
	 for(int i=0;i<30;i++)

	}

}
