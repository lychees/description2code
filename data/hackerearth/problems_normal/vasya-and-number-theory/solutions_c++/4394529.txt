#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define d1(a)cout<<#a<<": "<<a<<"\n";
#define d2(a,b)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<"\n";
#define d3(a,b,c)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<" , "<<#c<<": "<<c<<"\n";
#define d4(a,b,c,d)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<" , "<<#c<<": "<<c<<" , "<<#d<<": "<<d<<"\n";

int dp[1005],a[1005];

int main() {
	int n;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	sort(a,a+n);
	for(int i=0;i<n;i++){
		dp[i] = 1;
	}
	for(int i=0;i<n;i++){
		for(int j=i+1;j<n;j++){
			if(a[j]%a[i] == 0 && dp[j] < dp[i]+1){
				dp[j] = dp[i] + 1;
			}
		}
	}
	int ans = -1;
	for(int i=0;i<n;i++){
		ans = max(ans,dp[i]);
	}
	if(ans>1){
		cout<<ans;
	}
	else{
		cout<<"-1";
	}
	return 0;
}
