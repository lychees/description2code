T = input()
for _ in xrange(T):
    N = raw_input()
    pencils = map(int, raw_input().strip().split())
    mPCount = 0
    rPCount = 0
    mPencil = 0
    rPencil = 0
    while len(pencils) > 0:
        if mPencil == 0:
            mPencil = pencils.pop(0)
            mPCount += 1
        if rPencil == 0 and len(pencils) > 0:
            rPencil = pencils.pop() * 2
            rPCount += 1

        if mPencil > rPencil:
            mPencil -= rPencil
            rPencil = 0
        else:
            rPencil -= mPencil
            mPencil = 0

    print "%s %s" % (mPCount, rPCount)
