#include <iostream>
#include <string>
#include <vector>
#include <cmath>

using namespace std;

bool check_prime(long long int n){
	if(n==1)	return false;
	if(n==2)	return true;
	if(n%2==0)	return false;
	long long int x=sqrt(n);
	for(long long int i=3;i<=x;i+=2){
		if(n%i==0)	return false;
	}
	return true;
}

int main(){
	long long int T;
	long long int m,n,sum;
	cin >> T;
	while(T--){
		cin >> m >> n;
		sum=0;
		for(long long int i=m;i<=n;i++){
			if(check_prime(i))	sum++;
		}

		cout<<sum<<endl;
	}
	return 0;
}
