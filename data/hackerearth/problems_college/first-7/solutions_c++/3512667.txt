# include <bits/stdc++.h>

using namespace std;

typedef long long ll;

const int max_n = 1e5 + 10;

double dp[max_n], arr[max_n];

void precompute()
{
	ll i, j, n;

	dp[0] = arr[0] = 0;
	dp[1] = arr[1] = 1;

	for(n=2;n<max_n;++n)
	{
		arr[n] = ( (arr[n-2] + 8*(n-2)*(n-2)) + ( 1.0 * (3*n - 2) * (n*n - (n-2)*(n-2)) ) );
		dp[n] = arr[n]/(n*n) ;
	}
}

int main()
{
	ll t, i, j, n;

	precompute();

	scanf("%lld", &t);

	while(t--)
	{
		scanf("%lld", &n);

		printf("%.6lf\n", dp[n]);
	}

	return 0;
}
