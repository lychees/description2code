#include <iostream>
#include <algorithm>
#define ll long long
using namespace std;
ll int a[10000001];
ll int b[10000001];
ll int comparator(ll int a, ll int b)
{
	return a>b;
}
int main()
{
   ll int n;
   cin>>n;
   for(ll int i=0; i<n; i++)
   {
   		cin>>a[i];
   }
   for(ll int i=0; i<n; i++)
   {
   		cin>>b[i];
   }
   sort(a,a+n,comparator);
   sort(b,b+n,comparator);
   ll int sum=0;
   for(ll int i=0; i<n; i++)
   {
   		sum=sum+a[i]*b[i];
   }
   cout<<sum<<endl;
    return 0;
}
