#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	long int a[n],max;
    	long long int profit=0;
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	
    	max=a[n-1];
    	for(int i=n-1;i>0;i--)
    	{
    		if(a[i-1] > max)
    		max=a[i-1];
    		else
    		profit=profit +max - a[i-1];
    	}
    	cout<<profit<<endl;
    }
    return 0;
}
