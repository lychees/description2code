using namespace std;
ll arr[1001]={0};
ll stairs(ll a)
{
	if(a<=2)
	return a;
	else if(arr[a]!=0)
	return arr[a];
	else
	{
		arr[a]=stairs(a-1)+stairs(a-2);
		return arr[a];
	}
}
int main()
{  
	boost;
	arr[1]=1;
	arr[2]=2;
	ll t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<stairs(n)<<endl;
	}
	return 0;
}
