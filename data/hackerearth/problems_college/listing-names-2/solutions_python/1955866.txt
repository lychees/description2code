mydict = {}

T = int(raw_input())
while T>0:
	val = raw_input()
	if mydict.has_key(val):
		mydict[val] = mydict[val] + 1
	else:
		mydict[val] = 1
	T = T -1
	
m_list = mydict.keys()
m_list.sort()
for val in m_list:
	print "%s %d" %(val , mydict[val])
