#include<iostream>
#include<cstdio>
using namespace std;
int prime[100001];
void seive()
{

prime[0]=-1;
prime[1]=-1;
prime[2]=0;
for(int i=2;i<100;i++)
{
  if(prime[i]==0)
  {
     for(int j=i*i;j<=1001;j+=i)
     {
        prime[j]=-1;
     }
  }
}
}

string pre(string s) {
  int n = s.length();
  if (n == 0) return "^$";
  string ret = "^";
  for (int i = 0; i < n; i++)
    ret += "#" + s.substr(i, 1);
 
  ret += "#$";
  return ret;
}
 
int longpal(string s) {
  string T = pre(s);
  int n = T.length();
  int *P = new int[n];
  int C = 0, R = 0;
  for (int i = 1; i < n-1; i++) {
    int i_mirror = 2*C-i; 
     
    P[i] = (R > i) ? min(R-i, P[i_mirror]) : 0;
     
    while (T[i + 1 + P[i]] == T[i - 1 - P[i]])
      P[i]++;
 
      if (i + P[i] > R) {
      C = i;
      R = i + P[i];
    }
  }
 
  int maxLen = 0;
  int centerIndex = 0;
  for (int i = 1; i < n-1; i++) {
    if (P[i] > maxLen) {
      maxLen = P[i];
      centerIndex = i;
    }
  }
  delete[] P;
return maxLen;
}

int main()
{
 seive();
 int t;

 string s;
 scanf("%d",&t);
 while(t--)
 {
   cin>>s;
   if(prime[longpal(s)]==0)
   printf("YES\n");
   else printf("NO\n");
 }


}
