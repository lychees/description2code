#include <iostream>
using namespace std;

typedef long long ll;

int main(int argc, char const *argv[])
{
    ios_base::sync_with_stdio(false);
    ll n,m,q;
    cin>>n>>m>>q;
    ll arr[n+1];
    arr[0]=1;
    for(int i=1;i<=n;i++)
    {
        cin>>arr[i];
    }

    ll first[n+2], last[n+2];
    first[0]=1;
    last[n+1]=1;

    for(int i=1;i<=n;i++)
    {
        first[i]=first[i-1]*arr[i];
        first[i]%=m;
    }

    for(int i=n;i>=1;i--)
    {
        last[i]=last[i+1]*arr[i];
        last[i]%=m;
    }

    while(q--)
    {
        int in;
        cin>>in;
        ll ans = first[in-1]*last[in+1];
        ans%=m;
        cout<<ans<<"\n";
    }
    return 0;
}
