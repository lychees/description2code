item = raw_input()
itemLen = len(item)
numSet = int(raw_input())
for setCtr in range(numSet):
	intsStr = raw_input()
	ints = intsStr.split(" ")
	a = int(ints[0])
	b = int(ints[1])
	a %= itemLen
	b %= itemLen
	if item[a - 1] == item[b - 1]:
		print "Yes"
	else:
		print "No"
