#include<bits/stdc++.h>
using namespace std;
#define MAX 100005
#define MOD 1000000007
#define ll long long int
ll x[MAX];
ll x2[MAX];
ll q[MAX];
ll frnds[MAX];
multiset<pair<ll,int> > mt;
int main()
{
    int c;
    int p;
    int n;
    int i;
    
    pair<ll,int> tmp;
    cin>>c>>p>>n;
    for(i=0;i<=n;++i)
    	x[i]=x2[i]=frnds[i]=q[i]=0;
    for(i=1;i<=n;++i)
    {
    	cin>>x[i];
    	q[i]++;
    }
    for(i=1;i<=p;++i)
    	cin>>frnds[i];
    for(i=1;i<=c;++i)
    {
    	mt.insert(make_pair((x[i]*q[i])%MOD,i));
    }
    for(i=1;i<=p;++i)
    {
    	tmp = *mt.begin();
    	int pos = tmp.second;
		x2[pos] = x[pos];
    	x[pos] = frnds[i];
    	q[pos]++;
    	cout<<pos<<" ";
    	ll pr = ((x[pos]  + x2[pos])*q[pos]) % MOD;
    	mt.erase(mt.begin());
    	mt.insert(make_pair(pr,pos));
    	
    	
    }
    cout<<endl;
    return 0;
}
