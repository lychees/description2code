#include<bits/stdc++.h>
using namespace std;
typedef unsigned long long ui64;
int gcd(int a,int b)
{
int t;
while(a)
{
t = a;
a = b%a;
b = t;
}
return b;
}
int power(ui64 x, ui64 y, ui64 p)
{
int res = 1;
x = x % p;
while (y > 0)
{
if (y & 1)
res = (res*x) % p;
y = y>>1;
x = (x*x) % p;
}
return res;
}
