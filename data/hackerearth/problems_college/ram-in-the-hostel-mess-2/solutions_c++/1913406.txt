#include <iostream>
#include <cmath>
using namespace std;

typedef long long ll;

ll n;

ll calculate(ll size,ll lastIndex){
	if(size==1)
		return (n-lastIndex);
	
	ll sum=0;
	while(lastIndex<=n){
		
		sum+=calculate(size-1,lastIndex);
		lastIndex++;
	}
	return sum;
}

int main()
{
    ll t,i,total;
    cin>>t;
    
	while(t--){
		cin>>n;
		if(n==0){
			cout<<0<<endl;
			continue;
		}
		total=0;
		ll upto=ceil(n/2.0);
		for(i=1;i<=upto;i++)
			total+=calculate(i,2*i-2);
		
		total++;
		cout<<total*total<<endl;
	}
    return 0;
}
