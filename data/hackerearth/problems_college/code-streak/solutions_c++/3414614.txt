#include <iostream>
#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll a[1000008]={0};
int main()
{
   
  ll t,n,t1,t2;
  cin>>t;
  while(t--){
      cin>>n;
      fill(a,a+1000003,0);
      ll max=-1,c=0;
      while(n--){
          cin>>t1>>t2;
          if(t2==1){
              if(a[t1]==0){
                  c++;a[t1]=1;
              }
          }
         else {
            
             c=0;
         } 
          if(c>max)max=c;
      }
      cout<<max<<endl;
  }
    return 0;
}
