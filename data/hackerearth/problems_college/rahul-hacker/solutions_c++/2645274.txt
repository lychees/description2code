#include <iostream>
using namespace std;
long int rev(long int n)
{
	long int rem;
	long int r=0;
	while(n>0)
	{
		rem = n%10;
		r=rem+r*10;
		n=n/10;
	}
	return r;
}
	
int main()
{
    int t;
    long int x,y;
    cin>>t;
    while(t--)
    {
    	cin>>x>>y;
    	cout<<rev(rev(x)+rev(y))<<endl;
    }
    
    return 0;
}
