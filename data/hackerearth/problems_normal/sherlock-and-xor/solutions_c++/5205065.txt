#include <iostream>

using namespace std;

int main()
{
	int cases;
	cin >> cases;
	
	int oldDays = 0;
	bool* tasks = nullptr;
	while (cases > 0)
	{
		int days;
		cin >> days;
		if (days > oldDays)
		{
			if (tasks != nullptr)
				free(tasks);
			tasks = (bool*)malloc(days * sizeof(bool));
		}
		
		long odd = 0, even = 0;
		for (int i = 0; i < days; i++)
		{
			int t;
			cin >> t;
			if (t & 1)
				odd++;
			else
				even++;
		}
		
		cout << odd*even << endl;
		
		cases--;
	}
    return 0;
}
