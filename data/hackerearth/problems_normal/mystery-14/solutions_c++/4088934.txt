#include <stdio.h>
#include<iostream>
#include<cstring>
#define gc getchar_unlocked
using namespace std;
typedef long long ll;
void scanint(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
void scanlong(long long &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;

    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
int main(void) {
    int t;
    scanint(t);
    while(t--)
    {
    	int n;
    	long long sum=0;
    	scanint(n);
    	while(n--){
    		int temp;
    		scanint(temp);
    		if(temp&1)
    		sum+=temp;
    	}
    	cout<<sum<<endl;
    }
	return 0;
}
