#include <iostream>
#include <vector>
#include <algorithm>
typedef long l;
using namespace std;

int main()
{
    int N, num;
    l inp;
    cin>>N;
    num=N;
    vector<l> arr;
    while(num--)
    {
    	cin>>inp;
    	arr.push_back(inp);
    }
    sort(arr.begin(), arr.end());
    for(int i=0; i<N; i++)
        cout<<arr[i]<<endl;
    return 0;
}
