#include<iostream>
using namespace std;
int main()
{
	long long int m,n,a;
	cin>>n>>m>>a;
	long long int i=1;
	long long int j=1;
	long long int c=0,d=0;
	while(1)
	{
		if(c==0)
		{
			if(a*i>n || a*i==n)
			{
				c=i;	
			}
			i++;
		}
		if(d==0)
		{
			if(a*j>m || a*j==m)
			{
				d=j;
			}
			j++;
		}
		if(c!=0 && d!=0)
		{
			break;
		}
	}
	cout<<c*d<<"\n";
}
