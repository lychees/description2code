#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int a;
    	string s;
    	cin>>a>>s;
    	int l = s.size();
    	int ans=0;
    	for(int i=0;i<l;i++)
    	{
    		if(s[i]=='1'||s[i]=='4'||s[i]=='6'||s[i]=='8')ans++;
    		if(s[i]=='9')ans+=2;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
