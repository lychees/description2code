#include<bits/stdc++.h>
using namespace std;
map<char,int> mp;
int main()
{
	mp['a']=mp['e']=mp['i']=mp['o']=mp['u']=mp['y']=1;
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		int ct=0,v=0,f=0,i,l=s.size();
		for(i=0;i<l;i++)
		if(mp[s[i]])
		v++;
		int x=l-v;
		if(x>v)
		{
		
		cout<<"hard"<<endl;
		continue;
		}
		
		for(i=0;i<l;i++)
		{
			if(!mp[s[i]])
			ct++;
			else
			ct=0;
			if(ct>=3)
			{
				f=1;
				break;
			}
		}
		if(f )
		cout<<"hard"<<endl;
		else
		cout<<"easy"<<endl;
	}
	return 0;
}
