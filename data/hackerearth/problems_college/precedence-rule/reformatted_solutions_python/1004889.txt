T = int(raw_input())
for t in range(T):
    exp = raw_input()
    exp += "+"
    val = 0
    for i in range(1, len(exp)):
        if (exp[-i] == "+"):
            val += int(exp[-i-1])
        elif (exp[-i] == "-"):
            val -= int(exp[-i-1])
        elif (exp[-i] == "*"):
            val *= int(exp[-i-1])
    print val
