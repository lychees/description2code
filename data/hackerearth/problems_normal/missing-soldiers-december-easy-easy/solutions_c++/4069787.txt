#include<bits/stdc++.h>
using namespace std;
const int maxv=1e5;
pair<int,int>v[maxv];
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		int x,y,d;
		cin>>x>>y>>d;
		v[i]=make_pair(x,x+d);
	}
	sort(v,v+n);
	int r=0;
	int c=1;
	for(int i=0;i<n;i++)
	{
		c=max(c,v[i].first);
		if(v[i].second>=c)
		{
			r+=(v[i].second-c+1);
			c=v[i].second+1;
		}
	}
	cout<<r;
	return 0;
}
