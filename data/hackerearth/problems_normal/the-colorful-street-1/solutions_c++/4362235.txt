#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	int arr[3][n];
    	for(int i=0;i<n;++i){
    		cin>>arr[0][i]>>arr[1][i]>>arr[2][i];
    	}
    	int mini;
    	for(int i=1;i<n;++i){
    		for(int j=0;j<3;++j){
    			mini=INT_MAX;
    			for(int k=0;k<3;++k){
    				if(k==j)
    					continue;
    				mini=min(mini,arr[k][i-1]);
    			}
    			arr[j][i]+=mini;
    		}
    	}
    	cout<<min(arr[0][n-1],min(arr[1][n-1],arr[2][n-1]))<<endl;
    }
    return 0;
}
