#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
   
    	string s[t];
    	for(int i=0;i<t;i++)
    	cin>>s[i];
    	sort(s,s+t);
    	int j=1,i;
    	for(i=1;i<t;i++)
           {//cout <<s[i-1]<<" i"<<j<<endl;j=1;
           	if(s[i].compare(s[i-1])==0)
             j++;
            else {cout <<s[i-1]<<" "<<j<<endl;j=1;}
           }
           cout <<s[i-1]<<" "<<j<<endl;
  
    return 0;
}
