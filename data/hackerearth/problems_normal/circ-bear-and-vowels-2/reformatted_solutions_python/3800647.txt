vow = "aeiouy"
for _ in range(input()):
    ins = raw_input().strip()
    vowels = sum([1 for x in ins if x in vow])
    if vowels < (len(ins) - vowels):
        print "hard"
        continue
    else:
        concnt = 0
        isHard = False
        for x in ins:
            if x not in vow:
                concnt += 1
                if concnt == 3:
                    isHard = True
                    concnt = 0
                    break
            else:
                concnt = 0
    print "hard" if isHard else "easy"
