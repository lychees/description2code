#include <bits/stdc++.h>
using namespace std;
#define For(i,a,b) for(int i=a;i<b;i++)
#define mp make_pair
#define pb push_back
#define mod 1000000007

int n,q,x,lis[1111],l,r,ans;

int main( ){
    //freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
    //ios::sync_with_stdio(0);
    cin>>n>>l>>r>>q;
    For(i,0,n) cin>>lis[i];
    while(q--)
    {
        cin>>x;
        ans=mod;
        For(i,l,r+1) if(lis[i]>x) ans=min(ans,lis[i]);
        if(ans==mod) cout<<-1;
        else cout<<ans;
        cout<<endl;
    }
    return 0;
}
