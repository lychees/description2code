#include <iostream>
using namespace std;
long long int n, q, i, j, d;
long long int a[1000000];
int main()
{
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    cin>>q;
    j=1;
    for(i=0;i<q;i++)
    {
    	cin>>d;
    	j=j*d;
    }
    for(i=0;i<n;i++)
    {
    	if(j==0)
    	{
    		a[i]=0;
    	}
    	else
    	{
    		a[i]=a[i]/j;
    	}
    	cout<<a[i]<<" ";
    }
    cout<<endl;
    return 0;
}
