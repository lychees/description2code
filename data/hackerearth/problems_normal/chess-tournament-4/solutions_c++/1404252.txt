#include <iostream>
#include<queue>
using namespace std;
int main()
{
    queue<int>winners;
    int n, i, j, p1, p2, players, matches[1025][1025];
    cin>>n;
    players = 1<<n;
    for( i=2; i<=players; i++ ) {
    	for( j=1; j<i; j++ ) {
    		cin>>matches[i][j];
    	}
    }
    for( i=1; i<=players; i++)
    	winners.push(i);
    while(1) {
    	p1 = winners.front();
    	winners.pop();
    	if(winners.empty()) {
    		break;
    	}
    	p2 = winners.front();
    	winners.pop();
    	winners.push( matches[p2][p1] ? p2 : p1 );
    }
    cout<<p1<<endl;
    return 0;
}
