#include <iostream>
#include<bits/stdc++.h>
typedef long long int ll;
using namespace std;

int main()
{
  ll test;
  scanf ("%lld", &test);
  while (test--){
  	ll n, mod;
  	scanf ("%lld %lld ", &n, &mod);
  	ll i;
  	ll ans = 1;
  	ll num;
  	for (i =0 ; i < n; i++) {
  		scanf ("%lld", &num);
  		ans = ((ans % mod) * (num % mod) ) % mod;
  	}
  	if (ans == 0 ){
  		printf ("YES\n");
  	}else{
  		printf("NO\n");
  	}
  }
    return 0;
}
