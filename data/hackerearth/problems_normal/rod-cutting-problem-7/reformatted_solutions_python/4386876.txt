def Two(x):
    if (bin(x).count('1')) == 1:
        return 0
    else:
        return 1


for x in range(input()):
    rod = input()
    rods = [rod]
    while rod != 1:
        if rod % 2:
            rods.append(rod/2)
        else:
            rods.append(rod/2+1)
        rod /= 2
    ans = 0
    for currentRod in rods:
        ans += Two(currentRod)
    print ans
