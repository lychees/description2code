#include<bits/stdc++.h>
using namespace std;
int main ()
{	int T,n,m,d,i,j,count;
	int x1[1007],x2[10007],y1[1006],y2[1004];
	cin>>T;	
	while(T--)
	{count=0;	
		cin>>n>>m>>d;
		for(i=0;i<n;i++)
		cin>>x1[i]>>y1[i];
		for(i=0;i<m;i++)
		cin>>x2[i]>>y2[i];
		for(i=0;i<m;i++)
			{	
				for(j=0;j<n;j++)
				{
				if(abs(x1[j]-x2[i])+abs(y1[j]-y2[i])<=d)
						{count++;break;}
				}
			}
		if(m&1==1)
		{
		if(count>=m/2)
		cout<<"YES\n";	
		else
		cout<<"NO\n";
		}
		else
		{
		if(count>m/2)
		cout<<"YES\n";	
		else
		cout<<"NO\n";
		}
	}
return 0;
}
