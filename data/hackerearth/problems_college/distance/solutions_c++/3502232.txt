#include <iostream>
#include<cstdio>
#include<cmath>
#include<algorithm>
#include<string.h>
using namespace std;
int main()
{
   double gx,gy,sx,sy,ans,a,b;
   cin>>gx>>gy;
   cin>>sx>>sy;
   a=gx-sx;
   a=a*a;
   b=gy-sy;
   b=b*b;
   a=a+b;
   ans=sqrt(a);
   printf("%.5lf\n",ans);
   return 0;
}
