#include <iostream>
#include<functional>
#include<algorithm>
#include<math.h>
using namespace std;
int sum[100000], a[100000];
int TarnadoSum(int j, int ML, int M){
  int i = j-1;
  while(i>=0){
   if(a[j]-a[i]>ML)
    break;
    i--;
   }
  if(i<0)
   return M;
  else
    return sum[i]+M;
}

int main() {
  int i, S, M, ML, n;
  cin>>S>>M>>ML>>n;
  for(i=0; i<n; i++)
    cin>>a[i];
  sort(a,a+n);
  sum[0] = min(S,M);
  
  for(i= 1; i<n; i++)
      if(a[i] == a[i-1])
      sum[i] = min(sum[i-1],TarnadoSum(i,ML,M));
      else
      sum[i] = min(sum[i-1]+S,TarnadoSum(i,ML,M));
      
  cout<<sum[n-1];
 return 0;
}
