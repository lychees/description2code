#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    int m;
    long long x;
    cin >> m;
    if(m > 0) {
    	vector <long long> v;
    	for(int i = 0; i < m; i++) {
    		cin >> x;
    		v.push_back(x);
    	}
    	sort(v.begin(), v.end());
    	long long diff = 0;
    	for(int i = 0; i < m - 1; i++) {
    			diff = __gcd(diff, v[i + 1] - v[i]);
    	}
    	set <long long> s;
    	for(long long i = 1; i * i <= diff; i++) {
    		if(!(diff % i)) {
    			if(i != 1)
    				s.insert(i);
    			s.insert(diff / i);
    		}
    	}
    	set <long long> :: iterator it;
    	for(it = s.begin(); it != s.end(); it++)
    		cout << *it << " ";
    	cout << endl;
    }
    return 0;
}
