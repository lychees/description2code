#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
long int check(long int *elem,long int *mask,long int n){
    long int cnt=0;
for(int i=0;i<n;i++){
    if(mask[i])
        cnt+=elem[i];
}
return cnt;


}




int nexmask( long int *mask,long int n){
int i;
for(i=0;i<n&&mask[i];i++)
    mask[i]=0;
if(i<n){
    mask[i]=1;
    return 1;
}
return 0;

}

int main(){

int t;
cin>>t;
while(t--){
   long int n;
    long int s;
    int flag=0;
    cin>>n;
    long int elem[n],mask[n];
    for(int i=0;i<n;i++){
        mask[i]=0;
        cin>>elem[i];
       }
    cin>>s;
    if(s==0){
    cout<<"YES"<<endl;
    continue;
    }
    else{

        while(nexmask(mask,n)){
        if(s==check(elem,mask,n))
            {
                flag=1;
                break;
                }


        }
        if(flag==0)
            cout<<"NO"<<endl;
        else
            cout<<"YES"<<endl;



    }




    }

return 0;
}
