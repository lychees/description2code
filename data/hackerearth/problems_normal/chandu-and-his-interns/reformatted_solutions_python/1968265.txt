import math


def divisors_new(num):
    limit = num
    num_divisors = 0

    for i in xrange(1, int(math.sqrt(limit)+1)):
        if num_divisors > 3:
            break
        if num % i == 0:
            limit = num / i
            if limit != i:
                num_divisors += 1
            num_divisors += 1

    return num_divisors


N = int(raw_input())
for i in xrange(0, N):
    X = int(raw_input())
    count = divisors_new(X)
    if count < 4:
        print 'NO'
    else:
        print 'YES'
