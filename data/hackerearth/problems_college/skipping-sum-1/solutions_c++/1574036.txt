using namespace std;

#define ll long long
#define vt vector

ll dp[100001][11];


int main()
{
	int n, q;
	cin >> n >> q;
	for(int i = 1; i <= n; i++)
		scanf("%lld", &dp[i][0]);
	for(int k = 1; k < 11; k++)
		for(int i = 1; i <= n; i++)
			if (i-k >= 0)
				dp[i][k] = dp[i-k][k] + dp[i][0];
			else
				dp[i][k] = dp[i][0];
	while(q--)
	{
		int l, r, k;
		scanf("%d %d %d", &l, &r, &k);
		printf("%lld\n", dp[l+k*((r-l)/k)][k] - dp[l-k>=1?l-k:0][k]);
	}
	return 0;
}
