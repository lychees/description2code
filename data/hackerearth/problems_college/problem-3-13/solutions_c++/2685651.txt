#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int main()
{
	int n,m,i,j;
	cin>>n>>m;
	int a[n][m];
	int dp[n][m];
	for(i = 0; i < n; i++) {
		for(j = 0; j < m; j++) {
			cin>>a[i][j];
			dp[i][j] = 1;
		}
	}
	for(i = 0; i < n; i++) {
		for(j = m-2; j >= 0; j--) {
			if(a[i][j] == a[i][j+1]) {
				dp[i][j] = dp[i][j+1]+1;
			}
		}
	}
	for(j = 0; j < m; j++) {
		for(i = n-2; i >= 0; i--) {
			if(a[i][j] == a[i+1][j]) {
				dp[i][j] = dp[i+1][j]+1;
			}
		}
	}
	int maxi = 0;
	for(i = 0; i < n; i++) {
		for(j = 0; j < m; j++) {
			maxi = max(dp[i][j],maxi);
		}
	}
	cout<<maxi<<endl;
	return 0; 
}
