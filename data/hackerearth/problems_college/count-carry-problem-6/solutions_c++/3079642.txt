//#include "stdafx.h"
#include<string>
#include<iostream>
#include<cstdio>
using namespace std;
inline int add(string a, string b)
{
	int count = 0;
	for (int i = 0; i < a.length() / 2; i++)
	{
		char t = a[i];
		a[i] = a[a.length() - i - 1];
		a[a.length() - i - 1] = t;
	}
	for (int i = 0; i < b.length() / 2; i++)
	{
		char t = b[i];
		b[i] = b[b.length() - i - 1];
		b[b.length() - i - 1] = t;
	}
	string c = a + '0';
	int i;
	for (i = 0; i < b.length(); i++)
		c[i] = b[i];
	for (; i < c.length(); i++)
		c[i] = '0';
	int carry = 0;
	for (i = 0; i < a.length(); i++)
	{
		int x = a[i] - '0' + c[i] - '0' + carry;
		c[i] = x % 10 + '0';
		carry = x / 10;
		if (carry > 0)
		{
			count++;
		}
	}
	c[i] = carry + '0';
	for (int i = 0; i < c.length() / 2; i++)
	{
		char t = c[i];
		c[i] = c[c.length() - i - 1];
		c[c.length() - i - 1] = t;
	}
	return count;
}

int main()
{
	int n;
	scanf("%d", &n);
	while (n--)
	{
		string a,b;
		cin >> a >> b;
		int c=add(a, b);
		if (c == 1)
			printf("%d carry operation\n",c);
		else if (c == 0)
			printf("No carry operation\n");
		else
			printf("%d carry operations\n", c);
	}
    return 0;
}
