using namespace std;

map < char , int > mp;
map < char , int > :: iterator it;

int main()
{
    //freopen("in.txt","r",stdin);
    //freopen("out.txt","w",stdout);
    char str;
    while(scanf("%c",&str)==1){
        if(str=='\n')break;
        mp[str]++;
    }
    int count=0;
    for(it=mp.begin();it!=mp.end();it++){
        if((it->second)&1)count++;
    }
    if(!count||count==1)printf("YES\n");
    else printf("NO\n");
    return 0;
}
