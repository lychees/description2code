#include <iostream>
#include <cstring>
using namespace std;
int C[26];
int main()
{
    int n;
    string S;
    bool P[26];
    cin>>n;
    for(int i=0;i<n;++i)
    {
    	cin>>S;
    	for(int j=0;j<26;++j)
    	P[j]=false;
    	for(int j=0;j<S.length();++j)
    	{
    		if(P[S[j]-'a']==false)
    		{
    			++C[(S[j]-'a')];
    			P[S[j]-'a']=true;
    		}
    	}
    }
    int cnt=0;
    for(int i=0;i<26;++i)
    if(C[i]==n)
    ++cnt;
    cout<<cnt;
    return 0;
}
