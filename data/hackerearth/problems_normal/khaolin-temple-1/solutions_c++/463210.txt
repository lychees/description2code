typedef pair<int, int> pii;
 
int compare(pii a,pii b){return(a.second<b.second);}
 
void mergesort(vector< pii > &arr,int x,int y){
    int i,j,mid,val,k;
    if(x>=y)
	return;
	
    mid = (x+y+1)/2;
    mergesort(arr,x,mid-1);
    mergesort(arr,mid,y);
	
    val=0;
    vector< pair<int,int> > brr(y-x+1);
    i=x;j=mid;k=0;
    while(i<mid && j<=y){
        if(arr[i].first+val>=arr[j].first){
            brr[k++]=arr[j++];
            val++;
        }
        else{
            brr[k++]=pii(arr[i].first+val,arr[i].second);
            i++;
        }
    }
    if(i<mid){
        while(i!=mid){
            brr[k++]=pii(arr[i].first+val,arr[i].second);
            i++;
        }
    }
    if(j<=y){
        while(j<=y)
		brr[k++]=arr[j++];
    }
	
    for(i=0;i<(y-x+1);i++)
	arr[x+i]=brr[i];
	
    return;
}
