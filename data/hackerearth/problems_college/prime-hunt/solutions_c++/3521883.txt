#include <bits/stdc++.h>
using namespace std;
int main() {
   int t, j;
   long l, u, i, c;
   scanf("%d", &t);
   while(t--) {
      scanf("%ld %ld", &l, &u);
      c = 0;
      for(i = l ; i <= u ; i++) {
         if((i== 2 || i % 2) && i != 1) {
            bool flag = true;
            for(j = 3 ; j <= sqrt(i) ; j += 2)
               if(i % j == 0) {
                  flag = false;
                  break;
               }
            if(flag)
               c++;
         }    
      } 
      printf("%ld\n", c);
   }
   return 0;
   }
