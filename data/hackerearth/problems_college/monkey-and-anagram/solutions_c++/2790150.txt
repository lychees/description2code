#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	string a,b;
    	cin>>a>>b;
    	if(a.length()>b.length()) cout<<-1<<endl;
    	else
    	{
    		int c[26]={0};
    		for(int i=0;i<b.length();i++)
    				c[b[i]-'a']++;
    				
    			for(int i=0;i<a.length();i++)
    			     c[a[i]-'a']--;
    		int ans=0;
    			for(int i=0;i<26;i++)
    			{
    				if(c[i]) ans++;
    				if(c[i]<0) {ans=-1;break;}
    			}
    			cout<<ans<<endl;
    		
    	}
    }
    return 0;
}
