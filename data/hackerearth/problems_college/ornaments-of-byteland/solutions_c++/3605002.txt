#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n=0;
	int a[26]={0};
	int f=0;
	scanf("%d",&n);
	while(n--){
		char s1[111];
		scanf("%s",s1);
		if(f==0){
		for(int i=0;i<strlen(s1);i++){
			a[s1[i]-'a']++;   
		}
		f=1;
		//cout<<"First\n";
	}
	else{
	int	b[26] = {0};
		for(int i=0;i<strlen(s1);i++)
		b[s1[i]-'a']++;
		
		for(int k=0;k<26;k++){
			if(a[k]>0 && b[k]>0);
			else
			a[k] = 0;
			
		///	cout<<k<<"  a[k] "<<a[k]<<" b[k]  "<<b[k]<<endl;
		}// for ends
		//cout<<"Second\n";
	}// else ends
	
	} // while ends
	int ctr=0;
	for(int j=0;j<26;j++)
	{
		if(a[j]>0)
		ctr++;
	}
	printf("%d\n",ctr);
    //cout << "Hello World!" << endl;
    return 0;
}
