using namespace std;

	bool KMP(char * text,char * pat)
	{	
		int i,j,x=0,y,n,m;	
		
		n=strlen(text);
		m=strlen(pat);
		
		int lps[m],len=0;
		
		lps[0]=0;
		i=1;
		
		while(i<m)
		{
			if(pat[i]==pat[len]){
				len++;
				lps[i]=len;
				i++;
			}
			else{
				if(len)
					len=lps[len-1];
				else{
					lps[i]=len;
					i++;
				}
			}
		}
		
		for(i=j=0;i<n;)
		{
			if(text[i]==pat[j]){			
				i++,j++;
			
				if(j==m){
					return true;
				}
			}
			else{
				if(j==0)	i++;
				else		j=lps[j-1];
			}						
		}
		
		return false;
	}

int main()
