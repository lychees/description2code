#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
int main()
{
        int t;
        cin>>t;
        while(t--)
        {
	    char s[10010];
	    int aCost,bCost;
	    cin>>s;
	    cin>>aCost;
	    cin>>bCost;
	    int l=strlen(s)-1;
	    int i,j;
	    int flag=0;
	    long cost=0;
	    for(i=0,j=l;i<=j;i++,j--)
	    {
		if(s[i]!='/' && s[j]!='/' && s[i]!=s[j])
		{
		        flag=1;
		        break;
		}
		else if(s[i]=='/' && s[j]=='/')
		{
		        cost=cost+2*min(aCost,bCost);
		}
		else if(s[i]=='/' && s[j]!='/')
		{
		        if(s[j]=='a')
			    cost+=aCost;
		        else
			    cost+=bCost;
		}
		else if(s[i]!='/' && s[j]=='/')
		{
		        if(s[i]=='a')
			    cost+=aCost;
		        else
			    cost+=bCost;
		}
		
	    }
	    if(flag==1)
		cout<<"-1\n";
	    else
	    {
		cout<<cost<<"\n";
	    }
        }
        return 0;
}
