#include<bits/stdc++.h>
using namespace std;
int main() {
	int t;
	scanf("%d",&t);
	while(t--) {
		int n,d;
		scanf("%d %d",&n,&d);
		while(d>0 && n>0) {
			if(n>=3) {
				n-=1;
				d-=6;
			}
			else {
				n-=1;
				d-=2;
			}
		}
		if(d>0) {
			printf("No\n");
		}
		else {
			printf("Yes\n");
		}
	}
	return 0;
}
