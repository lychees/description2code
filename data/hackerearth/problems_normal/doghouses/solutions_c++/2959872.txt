#define mt 1010
#define mod

int ys[mt],yl[mt],yr[mt];

int main ()
{
//    fop;
    cffi;
    int i,j,n,y;
    cin>>n;
    rep(i,mt)
    {
        yl[i]=n;
        yr[i]=1;
    }

    rep1(i,n)
    {
        cin>>y;
        ys[y]++;
        yl[y]=min(yl[y],i);
        yr[y]=max(yr[y],i);
    }
    int ans=0,t1,t2;
    int lx,ly;
    for(i=0; i<=1000; i++)
    {
        t1=ys[i];
        ans=max(ans,t1);
        lx=n,ly=1;
        for(j=i+1; j<=1000; j++)
        {
            if(!ys[j])
                continue;
            t2=ys[j];
            if(lx<yl[i] && lx<yl[j])
                t2++;
            if(ly>yr[i] && ly>yr[j])
                t2++;
            ans=max(ans,t1+t2);
            lx=min(lx,yl[j]);
            ly=max(ly,yr[j]);
        }
        if(lx<yl[i])
            t1++;
        if(ly>yr[i])
            t1++;
        ans=max(ans,t1);
    }

    deb(ans);

    return 0;
}
