#include <iostream>
using namespace std;

int db(int n){
	
	int r=0,p=0,a=n;
	
	while(a){
		r=a%2;
		if(r&p==1){
			return 0;
		}
		p=r;
		a=a/2;
	}
	return 1;
}

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	int n;
    	cin>>n;
    	
    	while(1){
    		if(db(n)){
    			cout<<n<<endl;
    			break;
    		}
    		else{
    			n=n+1;
    		}
    	}
    }
    return 0;
}
