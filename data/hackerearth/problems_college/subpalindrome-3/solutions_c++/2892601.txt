#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;

long long dp[1001][1001];

long long count_palin(string &s, int i, int j)
{
    if (i > j)
        return 1;
    long long ans = 1;
    for (int m = i; m < j; m++) {
        if (s[j] == s[m]) {
            ans += count_palin(s, m+1, j-1);
        }
    }
    ans += count_palin(s, i, j-1);
    //cout << i << " " << j << " " << ans << endl;
    return ans;
}

 int main()
{
    ios::sync_with_stdio(false);
    int t;
    cin >> t;
    while (t--) {
        string s;
        cin >> s;
        if (s.size() == 1)
        	cout << "1" << endl;
        else
        	cout << count_palin(s, 0, s.size()-1)-1 << endl;
    }

    return 0;
}
