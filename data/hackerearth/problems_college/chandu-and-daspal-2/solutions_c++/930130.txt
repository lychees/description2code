#include <iostream>
using namespace std;
int gcd(int a,int b)
{
	if(b==0)
	return a;
	
	return gcd(b,a%b);
}
int main()
{	
	int T,N,x,y,z,flag;
	string S;
	cin>>T;
	while(T--)
	{	flag=0;
		cin>>x>>y>>S;
		if(S[0]=='C')
		flag=1;
		while(1)
		{
		if(flag==1)
		{
			z=gcd(x,y);
			x=x-z;
			y=y-z;
			
			if(x==0 || y==0)
			{
				cout<<"Chandu"<<endl;
				break;
			}
			flag=0;
		}
		else
		{
			z=gcd(x,y);
			x=x-z;
			y=y-z;
			if(x==0 || y==0)
			{
				cout<<"Daspal"<<endl;
				break;
			}
			flag=1;
		}
		}
	}
   
    return 0;
}
