    #include <iostream>
    using namespace std;
    #include <vector>
    #include <algorithm>
     
    int main()
    {
        vector<int> a;
        long long int s, i;
        cin>>s;
        i=s;
        while(i--)
        {
        	int d;
        	cin>>d;
        	a.push_back(d);
        }
        sort(a.begin(),a.end());
        long long int p=0;
        for(i=0; i<s; ++i)
        {
        	cout<<a[i]-p<<" ";
        	p++;
        }
        return 0;
    }
