#include <iostream>
using namespace std;

int main()
{
    string n;
    cin>>n;
    int i,sum;
    sum = 0;
    for(i=0;i<n.length();i++){
    	if(n[i] == '0'){
    		sum += 6;
    	}else if(n[i] == '1'){
    		sum += 2;
    	}else if(n[i] == '2'){
    		sum += 5;
    	}else if(n[i] == '3'){
    		sum += 5;
    	}else if(n[i] == '4'){
    		sum += 4;
    	}else if(n[i] == '5'){
    		sum += 5;
    	}else if(n[i] == '6'){
    		sum += 6;
    	}else if(n[i] == '7'){
    		sum += 3;
    	}else if(n[i] == '8'){
    		sum += 7;
    	}else if(n[i] == '9'){
    		sum += 6;
    	}
    }
    cout<<sum;
    return 0;
}
