import sys

t=int(input())

while t>0:
  s=sys.stdin.readline().strip().split()
  i,f=s
  le=len(i)
  bfi=i.find("B")
  bff=f.find("B")
  ali=i[::-1].find("A")
  alf=f[::-1].find("A")
  ali=le-ali-1
  alf=le-alf-1
  #al=len(f)-al-1
  #print bfi,bff,ali,alf
  if bfi<=bff and ali>=alf and i.count("A")==f.count("A") and i.count("B")==f.count("B"):
     print "Yes"
  else:
     print "No"
  t=t-1
