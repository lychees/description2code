#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <string>
#include <vector>

using namespace std;

typedef long long LL;

LL n;

int main() {
  cin >> n;
  if (n % 3 != 0) {
    printf("0\n");
    return 0;
  }
  LL ans = 0;
  for (LL x = 2; x * x * x <= n; x++) {
    if (n % (x * 3) != 0)
      continue;
    LL newN = n / (x * 3);
    for (LL y = (long long)sqrt(newN); y >= x; y--) {
      if (newN % y != 0)
        continue;
      LL z = newN / y;
      if ((x + y + z) & 1)
        continue;
      if (x + y <= z)
        break;
      if (x == y && y == z)
        ans++;
      else if ((y == z) || (x == y))
        ans += 3;
      else
        ans += 6;
    }
  }
  cout << ans << endl;
}
