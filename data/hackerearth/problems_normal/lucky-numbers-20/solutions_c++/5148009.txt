#include <iostream>
#define mod 1000000007
#define lli long long int
using namespace std;

int main()
{
	lli tc,N,count,tmp,ans;
	bool flag;
	cin >> tc;
	while(tc--)
	{
		cin >> N;
		ans = 0;
		for(lli i = 2;i<N ;i*=2)
		{
			ans= (ans + (i%mod) + 1)%mod;
			for(lli j = i*2 ; j+i <=N ;j*=2)
			{
				ans=(ans + i%mod +j%mod)%mod;
			}
		}
		cout << ans << endl;
	
	}
    return 0;
}
