#include <iostream>
#include <cstdint>
#include <cstdio>
#define ull int64_t 

using namespace std;

int main()
{
	ull t;
	cin>>t;
	while(t--)
	{
		ull n,m,x,te,s=0;
		cin>>n>>m>>x;
		for(ull i=0;i<n;i++){
			cin>>te;
			s+=te;
		}
		ull g;
		g=((n+1)*x)-s;
		if(g>m)
			cout<<"Impossible"<<endl;
		else if(g<0)
			cout<<"1"<<endl;
		else 
			cout<<g<<endl;
	}	

}
