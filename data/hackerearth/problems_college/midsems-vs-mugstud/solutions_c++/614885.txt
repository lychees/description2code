#include<iostream>
using namespace std;
int main()
{
int p,q,r;
int tests;
int i,j,k,l;
int b[101][101][101]={0};
cin>>tests;
for(l=0;l<tests;l++)
{
cin>>p;
for(i=1;i<=p;i++)
for(j=1;j<=p;j++)
for(k=1;k<=p;k++)
cin>>b[i][j][k];
for(i=p;i>0;i--)
for(j=p;j>0;j--)
for(k=p;k>0;k--)
b[i][j][k]=b[i][j][k]-b[i-1][j-1][k-1]+b[i-1][j-1][k]
+b[i-1][j][k-1]+b[i][j-1][k-1]-b[i-1][j][k]
-b[i][j-1][k]-b[i][j][k-1];
 
for(i=1;i<=p;i++)
for(j=1;j<=p;j++)
{
cout<<"\n";
for(k=1;k<=p;k++)
cout<<b[i][j][k]<<" ";
}
}
}
