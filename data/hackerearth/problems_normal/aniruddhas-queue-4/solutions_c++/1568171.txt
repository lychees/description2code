#include <iostream>
using namespace std;

int main()
{
    int t;
    
    cin >> t;
    
    while(t--) {
    	long long int n,m;
    	long long int s=0,i;
    	
    	cin >> n;
    	
    	long long int a[n];
    	
    	for(int i=0;i<n;i++) {
    		cin >> a[i];
    		s+=a[i];
    	}
    	cin >> m;
    	
    	m = m%s;
    	
    	if(m==0) {
    		for(i=n-1;i>=0;i--) if(a[i]!=0) break;
    	} else {
    	
    	s=0;
    	
    	for(i=0;i<n;i++) {
    		s+=a[i];
    		if(s>=m) break;
    	}
    }
   
    	cout<<i+1<<endl;
    }
}
