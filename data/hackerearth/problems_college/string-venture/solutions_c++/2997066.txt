#include<bits/stdc++.h>
using namespace std;
int main()
{
    char s[102],sa;
    int b,c,i,j,l,q;
    cin.getline(s,102);
    int a[26]={0};
    c=0;
    for(i=0;s[i]!='\0';i++)
        {
            c=c+s[i];
            a[s[i]-97]++;
        }
        j=strlen(s);
        c=c/j;
        q=0;
        if(c%2!=0)
        {
            for(i=0;i<26;i++)
            {
                    if(a[i]>q)
                    {
                        q=a[i];
                        sa=i+97;
                    }

            }
            cout<<sa<<endl;
        }
        else
        {
            for(i=j-1;i>=0;i--)
                cout<<s[i];
            cout<<endl;
        }
        return 0;
}
