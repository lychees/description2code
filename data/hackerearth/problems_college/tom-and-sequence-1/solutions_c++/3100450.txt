#include <bits/stdc++.h>
using namespace std;
#define m 1000000007
#define MAX 1300000
int a[MAX];
long long int b[100001];
#define gc() getchar_unlocked()
inline long long int read_int()
{
    char c = gc();
    while(c<'0' || c>'9')
    c = gc();
    long long int ret = 0;
    while(c>='0' && c<='9')
    {
    	ret = (ret<<3) + (ret<<1) + c - 48;
    	c = gc();
    }
    return ret;
}
void sieve()
{
	long long int i,j,val,ctr,k;
	a[0] = a[1] = false;
	for(i=2;i<MAX;i++)
        	a[i]=true;
    	for(i=4;i<MAX;i+=2)
        	a[i]=false;
    	for(i=3;i*i<MAX;i+=2)
    	{
        	if(a[i])
        	{
        		for(j=i*i;j<MAX;j+=i)
                		a[j]=false;
        	}
    	}
    	k = 0;
    	for(i=1;i<=MAX;i++)
    	{
    		if(a[i]==1)
    		{
    			b[k] = i;
    			k++;
    		}
    	}
}
int main()
{
	sieve();
	long long int i,val,q,n;
	long long int res[1000001];
	res[0] = 2;
	for(i=1;i<=100000;i++)
	{
		res[i] = ((res[i-1]%m)*((b[i])%m))%m;
	}
	q = read_int();
	while(q--)
	{
		n = read_int();
		cout << res[n-1]%m << endl;
	}
	return 0;
}
