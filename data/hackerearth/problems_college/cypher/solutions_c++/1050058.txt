#include <iostream>
#include <string>

using namespace std;

int main()
{
	int testCount = 0;
	string s;
	int len = 0;
	
	cin >> testCount;
	
	while (testCount > 0)
	{
		cin >> s;
		len = s.length();
		
		for (int i = 0; i < len; i++)
		{
			s[i] = 'a' + ((s[i] - 'a') + i) % 26;
		}
		
		cout << s << endl;
		testCount--;
	}
    return 0;
}
