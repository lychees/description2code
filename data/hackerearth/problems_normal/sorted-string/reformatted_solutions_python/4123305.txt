s = "abcdefghijklmnopqrstuvwxyz"
for i in range(input()):
    a = raw_input()
    l = []
    s1 = ""

    for j in range(26):
        l.append(a.count(s[j]))
    for k in range(1, max(l)+1):
        for m in range(26):
            if (k == l[m]):
                for n in range(k):
                    s1 += s[m]
    print(s1)
