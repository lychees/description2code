#include <bits/stdc++.h>
# define mod 1000000007
using namespace std;

int main()
{
    long long int t,a,b,c,ans;
    vector<long long int>v;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>c;
    	v.push_back(a);
    	v.push_back(b);
    	v.push_back(c);
    	sort(v.begin(),v.end());
    	ans=1;
    	ans=(ans*v[0])%mod;
    	ans=ans*(v[1]-1)%mod;
    	ans=ans*(v[2]-2)%mod;
    	cout<<ans<<endl;
    	v.clear();
    }
    
    return 0;
}
