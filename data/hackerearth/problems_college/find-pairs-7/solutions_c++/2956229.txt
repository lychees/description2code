#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int arr[10000007];
int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--){
   		int n;
   		cin>>n;
   		int b;
   		for(int i=0;i<n;i++)
   		{
   			scanf("%d",&b);
   			arr[b+1000000]++;
   		}
   		ll sm=0;
   		for(int i=0;i<10000007;i++){
   			sm+=1LL*arr[i]*(arr[i]+1)/2;
   			arr[i]=0;
   		}
   		printf("%lld\n",sm);
   }
    return 0;
}
