typedef long double ld;
typedef long long ll;
typedef unsigned long long ull;

int main(int /* argc */, char** /* argv */)
{
    ios_base::sync_with_stdio(false);
    #ifdef LOCAL
    assert(freopen("inp", "r", stdin));
    #endif
    int t;
    cin >> t;
    while (t--) {
        ll n, k;
        cin >> n >> k;
        ll f = 1;
        ll d = 1;
        ll e = 0;
        ll ans;
        do {
            ll t = (n - f) / d + 1;
            if (t == 1) {
                ans = f;
                break;
            }
            ll l = f + (t - 1) * d;
            ll s = f + d * e;
            d *= 2;
            ll r = (n - s) / d + 1;
            if (r >= k) {
                ans = s + d * (k - 1);
                break;
            }
            k -= r;
            if (e == 0) {
                f += d / 2;
            }
            if (s + d * (r - 1) == l) {
                e = 1;
            } else {
                e = 0;
            }
        } while (true);
        cout << ans << endl;
    }
    return 0;
}
