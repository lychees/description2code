#include<bits/stdc++.h>
using namespace std;
#define mod (int)(1e9+7)
#define int long long
long long inv_mod(long long a,long long b)
{
    if(b==0) return 1ll;
    if(b==1) return a%mod;
    if(b&1)
    {
     long long temp=inv_mod(a,b/2)%mod;
     temp=(temp*temp)%mod;

     return (a*temp)%mod;
    }
    else
    {
        long long temp=inv_mod(a,b/2)%mod;
        return (temp*temp)%mod;
    }
}
int sol()
{
    int i,n,j;
    cin>>n;
    int a[n+1];
    for(i=0;i<n;++i) cin>>a[i];
    int32_t x=(int32_t)pow(2,n+1)-1;
    int s=0,p=0,ans=0;
    for(i=0;i<x;++i)
    {
        s=0;p=1;
        for(j=0;j<n;++j)
        {
            if(i&(1<<j))
            {
                s+=a[j];
                p*=a[j];
                p%=mod;
            }
        }
       s=inv_mod(s,mod-2);
       p=(p*s)%mod;
       ans=max(p,ans);
    }
    return ans;
}
int32_t main()
{
    int t,i=1;
    cin>>t;
    while(i<=t)
    {
        int s=sol();
        cout<<"Case #"<<i<<':'<<' '<<s<<endl;
        ++i;
    }
    return 0;
}
