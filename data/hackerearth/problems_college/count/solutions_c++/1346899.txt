#include <bits/stdc++.h>
using namespace std;
#define mod 10000007

int main()
{
    long long int t,i,c,r,anso,anse;
    cin >> t;
    while (t--) {
        cin >> r >> c;
        anse = 0;
        anso = 0;
        for (i = 1; i <= r; i++) {
            if (i % 2 != 0) {
                anso = anso + min(c,i);
            } else {
                anse += min(c,i);
            }
        }
        cout << anse << " " << anso << endl;
    }

    return 0;
}
