#include <iostream>
#define m 1000000007
#define ll long long int 
using namespace std;
int gcd(int a, int b)
{
	if(b==0) return a;
	return gcd(b,a%b);
}
ll powM(ll base, ll ex) {
  base %= m;
  ll result = 1;
  while (ex > 0) {
    if (ex & 1) result = (result * base) % m;
    base = (base * base) % m;
    ex >>= 1;
  }
  return result;
}
int main()
{
    ll n,a,g=0;
    cin>>n;
    ll ans=1;
    for(int i=0;i<n;i++)
    {
    	cin>>a;
    	ans*=a;
    	ans%=m;
    	g=gcd(a,g);
    }
    //cout<<ans<<" "<<g<<endl;
    ans = powM(ans,g);
    cout<<ans<<endl;
    return 0;
}
