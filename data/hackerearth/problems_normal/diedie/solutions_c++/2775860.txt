#include <iostream>
using namespace std;

#define mod 1000000007

long long pw(long long x, long long y)
{
	if(y==0)
		return 1;
	long long u = pw(x,y/2);
	u = u*u % mod;
	if(y&1)
		u = u*x % mod;
	return u;
}

int main()
{
	int t;
	long long n,a;
	
	cin>>t;
	while(t--)
	{
		cin>>n;
		a=pw(2,n-1);
		cout<<pw(a,mod-2)<<endl;
	}
    return 0;
}
