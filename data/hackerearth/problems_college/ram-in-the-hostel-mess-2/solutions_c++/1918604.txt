#include <iostream>
#include <cmath>
#include <algorithm>
#include <cstdio>
#include <functional>

using namespace std;
int nlist[1000000];
int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n;
		cin >> n; 
	if(n==1)
		{ cout << 4 << endl;}
	else
		{	
			nlist[0] = 0;
			nlist[1] = 2;
			nlist[2] = 3;

			for(int i =3;i<=n;i++)
			{
				nlist[i] = nlist[i-1] + nlist[i-2];
			}

				cout << nlist[n] * nlist[n] <<endl;
		}


	}

	return 0;
}
