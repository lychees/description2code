#include <iostream>
#include <math.h>
using namespace std;

unsigned nChoosek( unsigned n, unsigned k )
{
    if (k > n) return 0;
    if (k * 2 > n) k = n-k;
    if (k == 0) return 1;

    int result = n;
    for(int i = 2; i <= k; ++i)
    {
        result *= (n-i+1);
        result /= i;
    }
    return result;
}

long long calc_odd(int num)
{
	int i;
	long long t = 0;
	for(i=0; i<=num; i++)
	{
		t += nChoosek(num,i);
		num--;
	}
	return t;
}
int main()
{
	int t, num;
	long long total, total_odd;
	cin>>t;
	while(t>0)
	{
		total_odd = total = 0;
		cin>>num;
		total_odd = calc_odd(num);
		cout<<total_odd<<endl;
		t--;
	}

    return 0;
}
