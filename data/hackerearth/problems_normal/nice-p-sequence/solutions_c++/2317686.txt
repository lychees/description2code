ll canhai(ll n)
{
    ll fi,la,mid;
    fi=0;
    la=n+1;
    while (fi+1<la)
    {
        mid=fi/2+la/2;
        if (fi%2 && la%2) mid++;
        if (mid>n/mid) la=mid;
        else fi=mid;
    }
    return fi;
}

ll power(ll a, ll n, ll m)
{
    if (n==0) return 1;
    ll tmp,c;
    tmp=c=a;
    n--;
    while (n)
    {
        if (n%2) tmp=tmp*c%m;
        c=c*c%m;
        n/=2;
    }
    return tmp;
}

int test;
ll a1,p,a2,n;

II inv(ll a, ll b)
{
    if (a==0)return ( II(0,1) );
    II pa=inv(b%a,a);
    II pa2;
    pa2.st=pa.nd-(b/a)*pa.st;
    pa2.nd=pa.st;
    return pa2;
}
