import math
n = int(raw_input())
a = sorted(map(int, raw_input().split()))
maxav = 0.0
lsum = 0
tsum = sum(a)
minav = float('inf')

for i in xrange(n-1):
    lsum += a[i]
    maxav = max(maxav, (tsum-lsum+0.0)/(n-i-1))
    minav = min(minav, (lsum+0.0)/(i+1))
if n == 1:
    print 0
else:
    print int(math.floor(maxav-minav))
