import math


def nCr(n, r):
    return math.factorial(n)/math.factorial(r)/math.factorial(n-r)


t = input()
for _ in range(t):
    n = int(raw_input())
    # print math.ceil(n/2)
    # print math.floor(n/2)
    s = 0
    for i in range(0, n/2+n % 2+1):
        k = (n-i+1)
        s += nCr(k, i)
    print s*s
