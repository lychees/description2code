#include <iostream>
using namespace std;

int main()
{
    int t;
    string a,b;
    cin>>t;
    while(t--){
    	cin>>a>>b;
    	if(b.length()>a.length()){
    		string c=a;
    		a=b;
    		b=c;
    	}
    	int ans=0;
		for(int i=0;i<a.length();i++){
			int y=a[i]-'0';
			if(i<b.length()){
				int r=b[i]-'0';
				if(y+r>9){
					ans++;
				}
			}
		}    
		if(ans>0){
			if(ans==1){
				cout<<"1 carry operation"<<endl;	
			}else{
				cout<<ans<<" carry operations"<<endl;	
			}
			
		}else{
			cout<<"No carry operation"<<endl;
		}
    }
    return 0;
}
