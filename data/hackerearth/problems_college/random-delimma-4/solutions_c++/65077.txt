#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
    int t, n, k;
    
    cin >> t;
    
    while (t--) {
    	cin >> k >> n;
    	double d = 1.0;
    	if (k < n) {
    		for (int i = 1; i < k; i++) {
    			d *= (double)(n - i) / (double)(n);
    		}
    		d = 1.0 - d;
    	}
    	
    	printf("%.9lf\n", d);
    }
    return 0;
}
