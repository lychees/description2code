typedef long long ll;
typedef long long LL;
typedef double D;

const int inft = 1000000009;
const int mod = 1000000007;
const int MAXN = 1000006;

int t[MAXN];
ll pref[MAXN],sum;
int n,h;
ll get(int i){
	D a=1.*n*n;
	D b=1.*n*(i+1)+1.*n*i+2.*sum;
	D c=2.*pref[i]+1.*(i+1)*i-2.*h;
	if(2LL*pref[i]+1LL*(i+1)*i-2*h>=0)return i;
	b/=a;c/=a;a/=a;
	D del=b*b-4.*c;
	ll k=ceil((-b+sqrt(del))/2.);
	return 1LL*n*k+i;
}
void solve() {
	scanf("%d%d",&n,&h);
	fru(i,n)scanf("%d",&t[i]);
	pref[0]=0;
	fru(i,n)pref[i+1]=pref[i]+t[i];
	sum=pref[n];
	ll ans=1LL<<60;
	fru(i,n){
		ans=min(ans,get(i+1));
	}
	printf("%lld\n",ans);
}

	fru(ti,te) solve();
