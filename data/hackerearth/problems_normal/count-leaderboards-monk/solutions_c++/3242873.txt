#include <iostream>
#include <cstdio>
#include <cstring>
#define MOD 1000000007
#define ll long long int
int N, K;
using namespace std;
ll C[1005][1005];
ll fact[1005];
int main()
{
    scanf("%d %d", &N, &K);
    for(int i = 0;i<=N;i++)
	{
		C[i][0] = C[i][i] = 1;
		int k = i>>1;
		for(int j = 1;j<k+1;j++)
		{
			C[i][j] = C[i][i-j] = (C[i-1][j-1] + C[i-1][j])%MOD;
		}
	}
	fact[0] = 1;
	fact[1] = 1;
	for(int i = 2;i<=N;i++)
	{
		fact[i] = (i*fact[i-1])%MOD;
	}
	
	ll L[N+5];
	memset(L,0,sizeof(L));
	L[1] = 1;L[0] = 1;
	for(int i = 1;i<N;i++)
	{
	   for(int j = 0;j<=i;j++)
	   {
	   	  L[i+1] = (L[i+1] + (((C[i][j]*fact[j+1])%MOD)*L[i-j])%MOD)%MOD;
	   }
	}
	ll ans = (L[N]*C[N][K])%MOD;
	printf("%lld", ans);
    return 0;
}
