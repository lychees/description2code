#include <iostream>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    long int n;
    while(t--)
    {
    	cin>>n;
    	cout<<(n/2)+1<<" ";
    	
    	if(n%2==0)
    	{
    		cout<<(n/2)-1<<"\n";
    	}
    	else
    	{
    		cout<<(n/2)<<"\n";
    	}
    	
    }
    return 0;
}
