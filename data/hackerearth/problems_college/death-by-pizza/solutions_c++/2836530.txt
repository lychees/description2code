#include<iostream>

using namespace std;

int main(){
	int N;
	int Q;
	int data[20000];
	int layer[20000][3];
	int pizza[5]={16,9,26,26,1};
	int sum=0;
	int k=0;
	cin>>N>>Q;
	for(int i=0;i<N;i++){
		cin>>data[i];
	}
	for(int i=0;i<Q;i++){
		sum=0;
		cin>>layer[i][0]>>layer[i][1]>>layer[i][2];
		if(layer[i][0]==1){ //firewall
			for(int j=layer[i][1];j<=layer[i][2];j++){
				sum += data[j-1];
			}
			cout<<sum<<"/"<<(layer[i][2]-layer[i][1]+1)<<"\n";
		}
		else{//filter
			k=0;
			for(int j=layer[i][1];j<=layer[i][2];j++){
				data[j-1]=data[j-1]+pizza[k%5];
				k++;
			}
		}
	}
	return 0;
}
