	string correct, modified;
	rep(t, T){
		cin>>correct;	//def

		int clen = correct.length();
		int mlen = modified.length();
		int maxsize = 0;
		for(i=0;i<=clen-mlen;i++){

					size++;
				}

				maxsize = size;
			}
		}
