#include<iostream>
#include<algorithm>
#include<set>
using namespace std;

int main()
{
    int n,q ,x;
    set<int> used;
    cin >> n >> q;
    while (q--)
    {
        int p;
        cin >> p;
        x = __gcd(n,p);

        if(used.find(x) == used.end())
        {
            cout << x;
            used.insert(x);
        }
        else {
            cout << -1;
        }

        cout << endl;
    }

    return 0;

}
