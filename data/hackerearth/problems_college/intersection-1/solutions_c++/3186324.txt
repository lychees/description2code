ll power(ll a,ll b)
{
      if(!b) return 1;
      ll x = power(a,b/2);
      x = (x*x);
      if(b&1) return (x*a);
      else return x;
}

int search(ll a , int p)
{
	ll low = 1,high = pow(a , (1.0)/p) + 100;
	while(low < high) {
		ll mid = (1 + low + high) / 2;
		if(power(mid,p) > a) {
			high = mid - 1;
		}
		else {
			low = mid;
		}
	}
	return low;
}

int main()
{
	int t,tc=1;
	cin>>t;
	while(t--) {
		int p,q,i;
		ll k,yo;
		cin>>p>>q>>k;
		int y = search(k,q);
		int ans = 0;
		for(i = 0; i <= 10000000; i++) {
			yo = power(i,p) + i;
			if(yo > k) break;
			while(power(y,q) > k-yo) y--;
			if(power(y,q) == k-yo) ans++;
		}
		printf("Case #%d: %Ld\n",tc++,ans);
	}
	return 0;
}
