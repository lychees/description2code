#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
int main()
{
    int t;
    cin>>t;
    long long int v[3];
    while(t--)
    { 
    	cin>>v[0]>>v[1]>>v[2];
    	//cout<<v[0]<<v[1]<<v[2];
    	sort(v,v+3);
    	//cout<<v[0]<<v[1]<<v[2];
    	for(int i=0;i<3;i++)
    	{
    		v[i]=v[i]-i;
    	}
    	long long int res=((((v[0]%mod)*(v[1]%mod))%mod)*(v[2]%mod))%mod;
    	cout<<res<<endl;
    }
    return 0;
}
