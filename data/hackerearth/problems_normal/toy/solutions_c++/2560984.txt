#include <bits/stdc++.h>
using namespace std;

int main() {
	long long int t,n,i,j,count=0,count1=0,a[100002];
	cin>>n;
	for(i=0;i<n;i++)
	cin>>a[i];
	sort(a,a+n);j=0;count=0;count1=0;
	while(1)
	{
		for(i=j;i<n;i++)
		{
			if(a[i]-a[j]<=4)
			count++;
			else
			break;
		}
		count1++;j=count;
		if(count==n)
		break;
	}
	cout<<count1<<endl;
	return 0;
}
