#include<iostream>

using namespace std;

int main()
{
        int t;
        cin >> t;
        while(t--)
        {
                long long int n,w,a,count=0,preva=0;
                long long int ans=0;
                cin >> w;
                cin >> n;
                for(int i=1;i<=n;i++)
                {
                        cin >> a;
                        if(a==1 && preva < i)
                        {
                                preva=i;
                                count=0;
                        }
                        else
                                count++;
                }
                ans = ans + n*w + count + 2*preva;
                cout << ans << endl;
        }
        return 0;
}
