#include <iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main() {
	// your code goes here
	int t,i;
	scanf("%d",&t);
	while(t--)
	{
		getchar();
		char in[1000005],final[1000005],inAA[1000005],inBB[1000005];
		scanf("%s %s",in,final);
		int inlen=strlen(in),inA=0,inB=0,finalA=0,finalB=0;
		for(i=0;i<inlen;i++)
		{
			if(in[i]=='A') inA++;
			else if(in[i]=='B') inB++;
			if(final[i]=='A') finalA++;
			else if(final[i]=='B') finalB++;
		}
		if(finalA==inA && finalB==inB)
		{
			//BOT A
			bool flag=0;
			strcpy (inAA,in);
			for(i=inlen-1;i>=0;i--)
			{
				if(flag && in[i]=='#') inAA[i]='A';
				if(in[i]=='A') flag=1;
				if(in[i]=='B') flag=0;
			}
			//BOT B
			strcpy (inBB,in);
			flag=0;
			for(i=0;i<inlen;i++)
			{
				if(flag && in[i]=='#') inBB[i]='B';
				if(in[i]=='B') flag=1;
				if(in[i]=='A') flag=0;
			}
			//printf("%s\n",in);
			flag=1;
			for(i=0;i<inlen;i++)
			{ 
				if(final[i]=='A')
				{
					if(inAA[i]!='A'){
						flag=0;
						break;
					}
				}
				if(final[i]=='B')
				{
					if(inBB[i]!='B')
					{
						flag=0;
						break;
					}
				}
			}
			if(flag)
				printf("Yes\n");
			else
				printf("No\n");
			
		}
		else
			printf("No\n");
		
	}
	return 0;
}
