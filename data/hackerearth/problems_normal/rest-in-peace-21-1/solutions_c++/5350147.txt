#include<bits/stdc++.h>
#include<string>
#include <stdlib.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    bool flag;
    while(n--)
    {   flag=true;
        char k[555];
        cin>>k;
        int l=strlen(k);
        for(int i=0;i<l-1;i++)
        {   //cout<<i<<endl;
            if(k[i]=='2' && k[i+1]=='1')
            {   flag=false;
                break;
            }
        }
        int z;
        z=atoi(k);  ///////////for the conversion
        if(z%21==0)
            flag=false;

        if(flag)
        cout<<"The streak lives still in our heart!"<<endl;
        else
        cout<<"The streak is broken!"<<endl;


    }

}
