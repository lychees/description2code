#include <cstdio>
#include <vector>
#include <map>
#include <string>
#include <algorithm>
#include <iostream>

#define DEBUG if(0)

using namespace std;

bool Process(string &a,string &b)
{
    vector<int> fa,fb;
    fa.resize(26);
    fb.resize(26);
    
    for(int i = 0; i < a.size(); ++i)
      ++fa[ a[i] - 'a' ];
    
    // init
    for(int i = 0; i < a.size(); ++i)
      ++fb[ b[i] - 'a' ];
    
    for(int i = 0; i <= b.size() - a.size(); ++i)
    {
	bool nFound = true;
	if(i != 0)
	{
	    //slide
	    --fb[ b[i-1] - 'a' ]; // remove entry
	    ++fb[ b[i - 1 + a.size()] - 'a' ]; // Add entry
	}
	
	for(int i = 0; i < 26; ++i)
	{
	    if( fa[i] > fb[i] )
	    {
	         nFound = false;
		 break;
	    }
	}
	
	if(nFound)
	  return true;
    }
    
    DEBUG {
      
      for(int i = 0; i < 26; ++i)
	  if(fa[i] || fb[i])
	  cout <<(char)('a' + i) << " " << fa[i] << " " << fb[i] << endl;
     
    }
    
    for(int i = 0; i < 26; ++i)
    {
	if( fa[i] > fb[i] )
	  return false;
    }
    
    return true;
}

int main()
{
    int T; string a,b;
    cin >> T;
    while(T--)
    {
	cin >> a;
	cin >> b;
	
	Process(a,b) ? printf("YES\n") : printf("NO\n");
    }
   
    return 0;
}
