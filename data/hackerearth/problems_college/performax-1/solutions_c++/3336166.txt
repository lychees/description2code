#include<bits/stdc++.h>
using namespace std;


int main()
{
	int n,z;
	double prev,ans,cur,d,x=0;
	
	scanf("%d",&n);
	scanf("%lf",&d);
	prev=0; ans=0;
	while(n--){
		cin>>cur;
	///	cout<<cur<<"-";
		if(cur<prev){
			z=((prev-cur)/d);
			z++;
			cur=cur+(z*d);
			ans+=z;			
		}
		prev=cur;
	//	cout<<endl;
		
	}
	z=ans;
	cout<<z;
	return 0;
}
