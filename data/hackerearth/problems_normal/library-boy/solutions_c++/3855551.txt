int main()
{
	int freq[26] = {0};
	int ans = 0;
	int n;
	cin >> n;
	while(n--)
	{
		string s;
		cin >> s;
		freq[s[0] - 'a'] += 1;
	}
	for(int i = 0; i < 26; i++)
		ans += (freq[i] + 9) / 10;
	cout << ans << endl;
	return 0;
}
