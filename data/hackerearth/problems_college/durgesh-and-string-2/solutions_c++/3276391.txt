#include <bits/stdc++.h>
 
#define fi first
#define se second
#define pb push_back
#define mp make_pair
#define pii pair<int,int>
#define pll pair<ll,ll>
#define MOD 1000000009
#define ll long long int
#define MAXINT 100000000
#define SET(x,y) memset(x,y,sizeof(x));
 
using namespace std;

char s[1000001];
pii A[1000];
int DP[20][502][503];

int main() {
	int n,x,y,i,j,k,l;
	cin>>n;
	for(i=1;i<=n;i++) {
		cin>>s;
		l=strlen(s);
		x=y=0;
		for(j=0;j<l;j++) {
			if(s[j]=='1')	x++;
			else			y++;
		}
		A[i]=mp(x,y);
	}
	cin>>x>>y;
	for(i=1;i<=n;i++) {
		for(j=0;j<=x;j++) {
			for(k=0;k<=y;k++) {
				DP[i][j][k]=DP[i-1][j][k];
				if(A[i].fi<=j && A[i].se<=k)	DP[i][j][k]=max(DP[i][j][k],DP[i-1][j-A[i].fi][k-A[i].se]+1);
			}
		}
	}
	cout<<DP[n][x][y]<<endl;
	return 0;
}
