long long get(int ps,int span)
{
	long long res=s[ps+span]-s[ps];
	res*=pw[N-ps];
	return res;
}

int lcp(int a,int b)
{
	/*
	cout<<a<<"%"<<b<<endl;
	for (int i=a;i<st.size();i++)
		cout<<st[i];
	cout<<endl;
	for (int i=b;i<st.size();i++)
		cout<<st[i];
	cout<<endl;
	*/
	int l,r;
	l=0;
	r=n;
	while (l<r)
	{
		int mid=l+r+1;
		mid/=2;
		if (get(a,mid)==get(b,mid))
			l=mid;
		else
			r=mid-1;
	}
//	cout<<l<<endl;
	return l;
}

bool cmp(int l,int r)
{
	int C=lcp(l,r);
	if (C==n)
		return (l<r);
	
	return st[l+C]<st[r+C];
}
