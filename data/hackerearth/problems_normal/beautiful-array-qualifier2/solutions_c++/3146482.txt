// LUCIFER <3 SLS <3

#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define s(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)
#define sd(x) scanf("%lf", &x)
#define mod 1000000007
#define get getchar//_unlocked

int mrk[1000005];

int main()
{
    int n, k, m, i, j, tmp, cnt, x;
    s(n);
    s(k);
    s(m);
    set <int> bad;
    while (m--) {
        s(x);
        bad.insert(x);
    }

    for (i = 1; i <= 1000000; ++i)
        mrk[i] = 100000000;

    cnt = 0;
    i = 1;

    while (cnt < n) {
        if (bad.find(i) == bad.end() && mrk[i] > k) {
            printf("%d ", i);
            ++cnt;
            mrk[i] = 1;
            for (j = 1; (tmp = i*j) <= 1000000; ++j)
                mrk[tmp] = min(mrk[tmp], mrk[j]+1);
        }
        ++i;
    }

    printf("\n");
    return 0;
}
