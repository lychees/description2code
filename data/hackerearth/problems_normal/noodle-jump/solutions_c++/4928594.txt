#include<bits/stdc++.h>
using namespace std;
#define ull long long int
int main()
{
    ios::sync_with_stdio(false);
    ull n,k,i;
    cin>>n>>k;
    ull arr[n+1];
    for(i=0;i<n;i++)
        cin>>arr[i];
    sort(arr,arr+n);
    ull ans=arr[0];
    for(i=1;i<n;i++)
    {
        if(arr[i]-arr[i-1]>k)
            break;
        ans=arr[i];
    }
    cout<<ans<<"\n";
    return 0;
}
