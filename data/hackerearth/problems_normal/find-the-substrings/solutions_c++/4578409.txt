#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    string s;
    while(t--)
    {
    	cin>>s;
    	long long l = s.length();
    	int a=0;
    	int b=0;
    	int c=0;
    	long long ans = l*(l+1) / 2;
    	for(long long i=0;i<l;i++)
    	{
    		if(s[i]=='a')
    		{
    			a=i+1;
    			ans-=min(b,c);
    		}
    		else if(s[i]=='b')
    		{
    			b=i+1;
    			ans-=min(a,c);
    		}
    		else if(s[i]=='c')
    		{
    			c=i+1;
    			ans-=min(a,b);
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
