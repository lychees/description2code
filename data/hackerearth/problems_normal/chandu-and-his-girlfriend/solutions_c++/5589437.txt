
#include <iostream>
#include <string>
#include <algorithm>

		
int main()
{
	using namespace std;
	int t;
	cin >> t ;
	for(int z=0;z<t;z++)
	{
		int n;
		cin >> n;
		int* a = new int[n];
		int k;
		for (k = 0; k < n; k++)
		{
			cin >> a[k];
		}

		sort(a,a+n);
		for (int j = n-1; j >= 0; j--)
		{
			cout << a[j] << " ";
		}
		cout << endl;
	}
}
