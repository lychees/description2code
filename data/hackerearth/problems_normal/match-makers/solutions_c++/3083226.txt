#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        int N,i;
        cin>>N;
        int A[N],B[N];
        for(i=0;i<N;i++)
            cin>>A[i];
        sort(A,A+N,greater<int>());
        for(i=0;i<N;i++)
            cin>>B[i];
        sort(B,B+N);
        int count=0;
        for(i=0;i<N;i++)
        {
            if(A[i]%B[i]==0 || B[i]%A[i]==0)
                count++;
        }
        cout<<count<<endl;
    }
}
