#include <bits/stdc++.h>
using namespace std;

int main(void)
{
	int T,x,f,t,d,i;
	char dec[100009];
	scanf("%d",&T);
	while(T--)
	{ scanf("%d",&x);
	if(x%3==0) {f=x;
	t=0;}
	else if(x%5==0) {f=0;
	t=x;}
	else if(x<8) {f=-1;
	t=-1;}
	else
	{ d=x%10;
	if(d==8||d==3) {f=3;
	t=x-3;}
	else if(d==1||d==6){f=6;
	t=x-6;}
	else if(d==4||d==9){f=9;
	t=x-9;}
	else {f=12;
	t=x-12;}}
	if(f!=-1&&t!=-1) { for(i=0;i<f;i++) dec[i]='5';
	for(i=f;i<x;i++) dec[i]='3';
	dec[i]='\0';
	printf("%s\n",dec);}
	else printf("-1\n");}
    return 0;
}
