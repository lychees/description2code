#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;

int main() 
{
    
    int t;
    int bt1, bt2, mt1, mt2;
    int t1, t2;
    int arr[4];
    
    scanf("%d", &t);
    for(int i =0; i < t; i++)
    {
        
        scanf("%d%d%d%d", &bt1, &bt2, &mt1, &mt2);
        
        t1 = (bt1 > bt2)? bt1:bt2;
        t2 = (bt1 < bt2)? bt1:bt2;
        
        bt1 = t2;
        bt2 = t1;
        arr[0] = t2;
        arr[1] = t1;
        
        t1 = (mt1 > mt2)? mt1:mt2;
        t2 = (mt1 < mt2)? mt1:mt2;
        
        mt1 = t2;
        mt2 = t1;
        arr[2] = mt1;
        arr[3] = mt2;
        if( bt1 == mt2 || bt2 == mt1 )
        {
            printf("Point\n");
            continue;
        }
        
        sort(arr, arr+4);
        
        
        if((arr[3]== mt2 && arr[2] == bt2) ||(arr[3] == bt2 && arr[2] == mt2))
        {
            if(mt2 == mt1 || bt2 == bt1)
                printf("Point\n");
            else
                printf("Nothing\n");
        }
        else
            printf("Line\n");
    }
	// your code goes here
	return 0;
}
