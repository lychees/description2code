void pre() {
    p[1]=1;
    for (int i=2 ; i*i <= 1e6 ; i++) 
        if (!p[i])
            for (int j=i*i ; j <= 1e6 ; j+=i)
                p[j]=1;
}

ll phi(ll n, ll left) {
    ll idx = n;
    ll ret = n;
    for (ll i=0 ; i < g[idx-left].size(); i++) {
        ll f = g[idx-left][i];
        while (n%f==0)
            n/=f;
        ret = (ret/f)*(f-1);
    }
    if(n > 1) {
    	ret = (ret/n)*(n-1);
    }
    return ret; 
}
