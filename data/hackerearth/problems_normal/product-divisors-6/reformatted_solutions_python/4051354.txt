import sys

MOD = 1000000007

n = int(sys.stdin.readline())

ar = map(int, sys.stdin.readline().split(' '))

factors = dict()
factors[2] = 0
factors[3] = 0
largest_prime_till_now = 3


def factorize(n, factors, largest_prime_till_now):
    for factor in factors.iterkeys():
        while n % factor == 0 and n != 1:
            factors[factor] += 1
            factors[factor] %= MOD
            n /= factor

    while n != 1:
        largest_prime_till_now += 2
        while n % largest_prime_till_now == 0 and n != 1:
            if factors.has_key(largest_prime_till_now):
                factors[largest_prime_till_now] += 1
            else:
                factors[largest_prime_till_now] = 1
            factors[largest_prime_till_now] %= MOD
            n /= largest_prime_till_now


result = 1
found_zero = False
for i in ar:
    if i == 0:
        found_zero = True

if not found_zero:
    for i in ar:
        factorize(i, factors, largest_prime_till_now)
    if factors[2] == 0:
        factors.pop(2)
    if factors[3] == 0:
        factors.pop(3)
    for i in factors.keys():
        result = result*(factors[i]+1)
        result %= MOD

print result
