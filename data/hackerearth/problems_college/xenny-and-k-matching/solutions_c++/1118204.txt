#include<bits/stdc++.h>
#define limit 1000000

using namespace std;

int main()
{
	long long N,k,a[limit],i=0;
	long count=0;

	cin>>N>>k;
	assert(N>=2 && N<=limit);

	for(long j=0;j<N;j++)
		cin>>a[j];

	for(long rp=k;rp<N;rp++)
	{
		if(abs(a[i]-a[rp])%k == 0 || abs(a[i]+a[rp])%k==0)
			count++;
		i++;
	}

	cout<<count++;

	return 0;
}
