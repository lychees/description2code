#include <iostream>
using namespace std;

int main()
{
    int t,a[1000],c;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>a[i];
    }
    for(int i=0;i<t;i++)
    {   c=0;
    	for(int j=0;j<t;j++)
    	{
    		if(a[i]<a[j])
    		{
    			c++;
    		}
    	}
    	c=1+c;
    	cout<<c<<" ";
    }
    return 0;
}
