#include <bits/stdc++.h>
#define mod 1000000007
#define MOD 1000000007
using namespace std;

typedef long long int ll;

ll powe(ll a, ll b)
{
 ll x=1,y=a; 
 while(b > 0)
 {
 		if(b%2 == 1)
 		{
 			x=(x*y);
 			
 			if(x>MOD) 
 			x%=MOD;
 		}
 	y = (y*y);
 	
 	if(y>MOD) 
 	y%=MOD; 
 	
 	b /= 2;
 }
 return x;
}
