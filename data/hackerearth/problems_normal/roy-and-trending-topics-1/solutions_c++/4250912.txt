#include <iostream>
#include <queue>
using namespace std;


class node{
    public:
     long long int id;
    long long int del;
       long long int new_score;
    };
    
/**Overloading operator '<' **/
bool operator<(const node& a,const node& b){
    if(a.del==b.del)
        return a.id<b.id;
    return a.del<b.del ;
}

int main()
{
    
    int N;
    cin>>N;
    
    
    priority_queue<node> p;
    
    for(int i=0;i<N;i++){
        
        node key;
        long long int post,like,comment,share,score;
        
        cin>>key.id>>score>>post>>like>>comment>>share;
        
        key.new_score = post*50 + like*5 + comment*10 + share*20;
        
        key.del =  key.new_score -score;
        
        p.push(key);
        
    }
  
   for(int i=0;i<5;i++){
       
       cout<<p.top().id<<" "<<p.top().new_score<<"\n";
       p.pop();
   }
  
}
