#include "bits/stdc++.h"
using namespace std;
const int N = 1e5 + 5;
int n;
int arr[N];
int pre[N];
int suf[N];
int main(){
	cin >> n;
	for(int i = 1 ; i <= n ; ++i){
		cin >> arr[i];
	}
	pre[0] = 0;
	for(int i = 1 ; i <= n ; ++i){
		pre[i] = max(pre[i - 1] , arr[i]);
	}
	suf[n + 1] = 0;
	for(int i = n ; i >= 1 ; --i){
		suf[i] = max(suf[i + 1] , arr[i]);
	}
	for(int i = 1 ; i <= n ; ++i){
		if(pre[i - 1] < arr[i] || suf[i + 1] < arr[i]){
			cout << i << " ";
		}
	}
}
