#include <bits/stdc++.h>
using namespace std;

int dp[1000005];

struct node
{
	int x;
	int y;
	int val;
	node(int a,int b,int c)
	{
		x=a;
		y=b;
		val=c;
	}
}temp(0,0,0);

std::queue<node>q;

void func()
{
	q.push(node(2,1,1));
	while(!q.empty())
	{
		temp=q.front();
		q.pop();
		
		if(temp.x >=1000001 && temp.y>=1000001)
			continue;
			
		if(temp.x<1000001 && temp.y<1000001)	
		{
		if(dp[temp.x]<temp.val && dp[temp.y]<temp.val)
		continue;
		}
		
			if(temp.x<1000001)
			{
			dp[temp.x]=min(dp[temp.x],temp.val);
			q.push(node(temp.x+temp.y,temp.y,temp.val+1));
			}
			if(temp.y<1000001)
			{
			dp[temp.y]=min(dp[temp.y],temp.val);
			q.push(node(temp.x,temp.x+temp.y,temp.val+1));
			}
	}
}

int t,n,cnt;
int main()
{
	memset(dp,1000005,sizeof(dp));
	dp[1]=0;
   func();
   scanf("%d",&t);
   while(t--)
   {
   	scanf("%d",&n);
   	if(dp[n]>1000000)
   	dp[n]=n-1;
   	printf("%d\n",dp[n]);
   }
   
    return 0;
}
