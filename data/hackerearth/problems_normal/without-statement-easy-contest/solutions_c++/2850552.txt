#include <bits/stdc++.h>
#define ll long long
using namespace std;
bool visited[900];
char str[30];
int myget(char *str)
{
	int i,j;
	int ans=0;
	for(i=0;str[i]!='\0';i++)
	{
		j=(str[i]-'0');
		ans+=(j*j);
	}
	return ans;
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int test;
	cin>>test;
	while(test--)
	{
		int m;
		cin>>str>>m;
		int i,t1,t2,t3,mc,ans;
		memset(visited,0,sizeof(visited));
		t1=myget(str);
		mc=0;
		while(!visited[t1])
		{
			visited[t1]=1;
			t2=0;
			while(t1!=0)
			{
				t3=(t1%10);
				t2+=(t3*t3);
				t1/=10;
			}
			t1=t2;
			mc++;
		}
		m--;
		if(m<mc)
		{
			t1=myget(str);
			for(i=0;i<m;i++)
			{
				t2=0;
				while(t1!=0)
				{
					t3=(t1%10);
					t2+=(t3*t3);
					t1/=10;
				}
				t1=t2;
			}
			ans=t1;
		}
		else
		{
			t1=myget(str);
			for(i=0;i<(mc-1);i++)
			{
				t2=0;
				while(t1!=0)
				{
					t3=(t1%10);
					t2+=(t3*t3);
					t1/=10;
				}
				t1=t2;
			}
			ans=t1;
			m-=(mc-1);
			memset(visited,0,sizeof(visited));
			mc=0;
			while(!visited[t1])
			{
				visited[t1]=1;
				t2=0;
				while(t1!=0)
				{
					t3=(t1%10);
					t2+=(t3*t3);
					t1/=10;
				}
				t1=t2;
				mc++;
			}
			mc=m%mc;
			t1=ans;
			for(i=0;i<mc;i++)
			{
				t2=0;
				while(t1!=0)
				{
					t3=(t1%10);
					t2+=(t3*t3);
					t1/=10;
				}
				t1=t2;
			}
			ans=t1;
		}
		
		cout<<t1<<endl;
	}
	return 0;
}
