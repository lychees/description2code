#include<iostream>
using namespace std;
main()
{
	int T,digit,sumtemp,sum2;
	long int value;
	cin>>T;
	while(T--)
	{
		cin>>value;
		sum2=0;
		while(value)
		{
		digit=value%10;
		sum2=sum2+digit;
		sumtemp=sum2;
		if(sumtemp>9)
		{
			sum2=0;
			while(sumtemp)
			{
			digit=sumtemp%10;
			sum2=sum2+digit;
			sumtemp=sumtemp/10;	
			}
		}
		value=value/10;
		}
		cout<<sum2<<endl;
	}
}
