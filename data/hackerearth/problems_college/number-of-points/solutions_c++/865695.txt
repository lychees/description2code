#include <iostream>
#include <cmath>
using namespace std;

int gcd(int a, int b) {
	if(b==0) return a;
	return gcd(b,a%b);
}

void irred(int x, int y, int &m1, int &m2) {
	x=abs(x);
	y=abs(y);
	int g=gcd(x,y);
	m1=x/g;
	m2=y/g;
}

int main()
{
    int T,x1,y1,x2,y2;
    cin>>T;
    while(T--) {
    	cin>>x1>>y1>>x2>>y2;
		cout<<max(0,gcd(abs(x1-x2),abs(y2-y1))-1)<<endl;
    }
    return 0;
}
