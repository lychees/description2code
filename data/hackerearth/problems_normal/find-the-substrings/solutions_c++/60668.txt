#define S(n)					scanf("%d",&n)

#define Sf(n) 					scanf("%lf",&n)
#define Ss(n) 					scanf("%s",n)
using namespace std;
int main()
{
	int t,i,j,k,n,m;
	#ifndef ONLINE_JUDGE
	//freopen("example.txt","r",stdin);
	#endif
	S(t);
	while(t--)
	{
		string s;
		cin>>s;
		int ca=0,cb=0,cc=0;
		int l,r;
		l=0,r=1;
		if(s[0]=='a')
			ca++;
		else if(s[0]=='b')
			cb++;
		else if(s[0]=='c')
			cc++;
		long long int ans=1;
		for(;r<s.size();r++)
		{
			if(s[r]=='a')
				ca++;
			else if(s[r]=='b')
				cb++;
			else if(s[r]=='c')
				cc++;
			if(ca==0||cb==0||cc==0)
			{
				ans+=r-l+1;
			}
			else
			{
				while(1)
				{
					if(s[l]=='a')
						ca--;
					else if(s[l]=='b')
						cb--;
					else if(s[l]=='c')
						cc--;
					
					l++;
					if(ca==0||cb==0||cc==0)
					{
						ans+=r-l+1;
						break;
					}	
				}		
			}
			//cout<<l<<" "<<r<<" "<<ans<<" "<<ca<<" "<<cb<<" "<<cc<<endl;
		}
		cout<<ans<<endl;
	}
	return 0;
}
