#include <bits/stdc++.h>
using namespace std;
int arr[21][21],vis[21][21];
int n;
int temp=0;
/*
 * count all possible path from start to destination
 *
 * start is (0,0) and destination is (n-1,n-1)
 * solve using dfs
 */
void DFS(int x,int y)
{
//	cout<<"JJJ"<<endl;
	if(x==n-1&&y==n-1)
	{
		temp++;	 // increament the count of possible path
		return ; // we reached to destination
	}
	vis[x][y]=1;  // mark the current as visited and go for all its adjacent which are not visited
	if(x+1<n&&!vis[x+1][y]&&!arr[x+1][y])
	{
		DFS(x+1,y); // call 1st adjacent if it is nt visited and arr[x+1][y]==0 if arr[x+1][y]==1 dnt go bcz it is blocked (geven is problem definition)
	}
	if(x>0&&!vis[x-1][y]&&!arr[x-1][y])
	{
		DFS(x-1,y);
	}
	if(y+1<n&&!vis[x][y+1]&&!arr[x][y+1])
	{
		DFS(x,y+1);
	}
	if(y>0&&!vis[x][y-1]&&!arr[x][y-1])
	{
		DFS(x,y-1);
	}
	vis[x][y]=0;
	return ;
}
int main() {
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		memset(arr,0,sizeof(arr));memset(vis,0,sizeof(vis));
		temp=0;
		cin>>n;
		for(int i=0;i<n;i++)
			for(int j=0;j<n;j++)
				cin>>arr[i][j];
		DFS(0,0);
		cout<<temp<<endl;
	}
	return 0;
}
