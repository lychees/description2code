#include <bits/stdc++.h>

#define mod 1000000007

long long fast_pow(long long int n) {
    long long result = 1;
    long long power = n;
    long long value = 2;
    while(power > 0) {
        if(power & 1) {
            result = result * value;
            result = result % mod;
        }
        value = value * value;
        value = value % mod;
        power /= 2;
    }
    return result;
}

}
