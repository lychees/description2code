def issubstr(substr, mystr, start_index=0):
    for letter in substr:
        start_index = mystr.find(letter, start_index) + 1
        if start_index == 0:
            return False
    return True


findIn = raw_input()
for x in xrange(int(raw_input())):
    findStr = raw_input()
    if issubstr(findStr, findIn):
        print "Yes"
    else:
        print "No"
