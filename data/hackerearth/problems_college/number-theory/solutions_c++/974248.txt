#include<iostream>
#include<cmath>
#include<cstring>
#include<string>
#include<algorithm>
using namespace std;

int x;
bool arr[10];


bool test(int i)
{
  while(i)
  {
    if(arr[i%10]) 
         return 1;
    i/=10;
  }
  return 0;
}

int main()
{
  int t;

  scanf("%d",&t);
  while(t--)
  {
     scanf("%d",&x);
     memset(arr,0,sizeof(arr));
     int m=x;
     while(m)
     {
       arr[m%10]=1;
       m/=10;
     }

     int res=0;
     for(int i=1;i*i<=x; i++)
     {
        if(x%i==0)
        {
          res+=test(i);
          if(i*i!=x)
	          res+=test((int)(x/i));    
        }
     }
     printf("%d\n", res);
   }
   return 0;
}
