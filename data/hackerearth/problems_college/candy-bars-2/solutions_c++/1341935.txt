#include<bits/stdc++.h>
#include<math.h>
using namespace std;
typedef long long int ll;
#define s(x) scanf("%lld",&x)
#define s1(x) scanf("%s",&x)
#define p(x) printf("%lld\n",x)
#define f1(x,r) for(x=0;x<r;x++)
#define f2(x,r) for(x=1;x<=r;x++)
#define f3(x) for(x=0;;x++)
#define f4(x,r) for(x=r;x>0;x--)
int main()
{
    ll i,j,k,l,n,max,max1,flag;
    s(n);
    ll a[n],ls[n];
    f1(i,n)
    {
        s(a[i]);
    }
    max1=-1;
    f1(i,n)
    {
        
        ls[0]=1;
        flag=0;
        max=-1;
        f1(j,i)
        {
            if(a[i]>a[j])
            {
                flag=1;
                if(ls[j]>max)
                    max=ls[j];
            }
        }
            if(flag==1)
                ls[i]=1+max;
            if(flag==0)
                ls[i]=1;
        if(ls[i]>max1)
            max1=ls[i];
    }
    k=n-max1;
    p(k);
}
