#include <iostream>
#include <vector>
using namespace std;

int dp[100010];
vector<int> adj[100010];

void dfs(int s, int p) {
	if(adj[s].size()==1 && s!=1) {
		dp[s]=0;
		return;
	}
	for(int i=0; i<adj[s].size(); ++i) {
		if(adj[s][i]!=p) {
			dfs(adj[s][i], s);
			dp[s]+=dp[adj[s][i]]+1;
		}
	}
	return;
}

int main()
{
	int n, i, u, v, j;
	cin>>n;
	for(i=2; i<=n; ++i) {
		cin>>u>>v;
		adj[u].push_back(v);
		adj[v].push_back(u);
	}
	dfs(1, 0);
	cout<<dp[1]<<" ";
	for(i=2; i<=n; ++i)
		cout<<dp[i]+1<<" ";
    return 0;
}
