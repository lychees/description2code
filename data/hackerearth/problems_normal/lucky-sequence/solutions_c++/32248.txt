#define S(n)	    scanf("%d",&n)
#define P(n)	    printf("%d\n",n)
#define Sl(n)	    scanf("%lld",&n)
#define Pl(n)	    printf("%lld\n",n)

#define Ss(n)       scanf("%s",n)
#define LL long long
#define ULL unsigned long long

using namespace std;

char ans[10000];

int main()
{
    LL t,n,m,i,j,k,l;
    Sl(t);
    while(t--)
    {
        Sl(n);
        l=0;

        while(n>0)
        {
            m=n;
            k=2;

            while(k<m)
            {
                m=m-k;
                k=k*2;
            }
            k=k/2;

            if(m<=k)
            {
                ans[l]='4';
                l++;
                n=n-k;
            }
            else
            {
                ans[l]='7';
                l++;
                n=n-k-k;
                m=m-k;
            }

            break;
        }

        while(n>0)
        {
            k=k/2;

            if(m<=k)
            {
                ans[l]='4';
                l++;
                n=n-k;
            }
            else
            {
                ans[l]='7';
                l++;
                n=n-k-k;
                m=m-k;
            }
        }

        for(i=0;i<l;i++)
            printf("%c",ans[i]);

        printf("\n");
    }
    return 0;
}
