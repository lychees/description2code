#include<stdio.h>
#define mod 1000000007
#include <bits/stdc++.h>
#include<iostream>
using namespace std;
int main()
{
	long int t,sum,num1,num2,num3;
	int n;
	char ch;
	scanf("%d",&t);
	while(t--)
	{
		num2=num3=1;
		scanf("%d",&n);
	    while(n--)
		{
			scanf(" %c",&ch);
		 
		  if(ch=='N')
		  {
		  	 num2=max(num2,-num3);
		     num3=min(num3,-num2);	//printf("%d\n",num2);
		  }
		  else if(ch=='+')
		  {
		  	scanf("%d",&num1);
		  	num2=max(num2,num2+num1);
		  	num3=min(num3,num3+num1);
		  }
		  else if(ch=='-')
		  {
		  	scanf("%d",&num1);
		  	num2=max(num2,num2-num1);
		  	num3=min(num3,num3-num1);
		  }
		  else if(ch=='*')
		  {
		  	scanf("%d",&num1);
		  	 num2= max(max(num2, num2 * num1), max(num3, num3 * num1));
            num3= min(min(num2, num2 * num1), min(num3, num3 * num1));
		  	
		  }
		  else if(ch=='/')
		  {
		  	scanf("%d",&num1);
		  	 num2= max(max(num2, num2 / num1), max(num3, num3 / num1));
            num3= min(min(num2, num2 /num1), min(num3, num3 /num1));
		  	
		  }
		}
		printf("%ld\n",num2);
	}
	return 0;
}
