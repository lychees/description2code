#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	ios::sync_with_stdio(false);
    int T,N,K,P,i,temp,*removeList,ans;
    long long ct,prevCtr;
    cin>>T;
    while(T--) {
    	cin>>N>>K>>P;
    	removeList=new int[K];
    	for(i=0;i<K;i++) {
    		cin>>removeList[i];
    	}
    	
    	//if (removeList[0]<P) {
    	for(i=1,prevCtr=ct=removeList[0]-1;i<K/* && removeList[0]<P*/;i++) {
    		ct+=removeList[i]-removeList[i-1]-1;
    		if (ct==P) {
    			ans=removeList[i]-1;
    			break; 
    		} else if (ct>P) {
    			ans=removeList[i-1]+P-prevCtr;
    			break;
    		}
    		prevCtr=ct;
    	} 
    	//if (N<K+P) ans=-1;
    	if (removeList[0]>P) ans=P;
    	else if (K==1) ans=++P;
    	else if (ct<P) {
    		i--;
    		if (ct==0) ct=K;
    		ans=removeList[i]+P-ct;
    	}
    	if (ans>N) ans=-1;
    	cout<<ans<<'\n';
    	delete[] removeList;
    }
    return 0;
}
