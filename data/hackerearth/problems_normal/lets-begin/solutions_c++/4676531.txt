int dp[1000006];
int min(int a,int b,int c,int d)
{
	int m=10000;
	if(a!=0 && a<m)
	m=a;
	if(b!=0 && b<m)
	m=b;
	if(c!=0 && c<m)
	m=c;
	if(d!=0 && d<m)
	m=d;
	

	
return m;	
}
long int calldp(int i,int level)
{
	if(i<0)
	return -1;
	if(dp[i]!=-2)
	return dp[i];
	if(i==0)
	return level-1;
	int a=1+calldp(i-2,level+1);
	int b=1+calldp(i-3,level+1);
	int c=1+calldp(i-5,level+1);
	int d=1+calldp(i-7,level+1);
	//printf("%d %d %d %d\n",a,b,c,d);
	if(a==0 && b==0 && c==0 && d==0)
	dp[i]=-1;
	else{
		dp[i]=min(a,b,c,d);
	}
	return dp[i];
}
