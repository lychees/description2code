using namespace std;
#include<iostream>
#include<string.h>
int main()
{
	int t,i,k,max=0;
	long n;
	cin>>t;
	for(int j=0;j<t;j++)
	{
		cin>>n;
		int x[n],ans=0,var=-1;
		for(i=0;i<n;i++)
			cin>>x[i];
		for(k=0;k<n;k++)
		{
			max=0;
			for(i=0;i<n;i++)
			{
				if(max<x[i])
				{
					max=x[i];
					var=i;
				}
			}
			x[var]=-1;
			for(i=0;i<n;i++)
			{
				if(x[i]<max && x[i]>=0)
				{
						x[i]=-1;
				}
			}
			if(max>0)
				ans++;
		}
		cout<<ans<<endl;
	}
}
