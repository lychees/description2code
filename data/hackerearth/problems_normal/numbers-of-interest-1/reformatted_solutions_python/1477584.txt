tc = int(raw_input())
while tc > 0:
    tc -= 1
    x, y, n = map(int, raw_input().split())
    ans = [y]*x
    if n <= x:
        print y
    else:
        last = x*y
        ans.append(last)
        ind = 0
        current = x*y
        while len(ans) <= n:
            tmp = last+current-ans[ind]
            ans.append(tmp)
            last = current = tmp
            ind += 1
        print ans[n-1]
