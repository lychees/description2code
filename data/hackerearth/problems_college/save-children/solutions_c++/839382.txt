#include<bits/stdc++.h>
using namespace std;

int main()
{   int t;
    cin>>t;
    while(t--)
    {
        int n;
        scanf("%d",&n);
        vector<int> v(n);
        vector<int> t(n);

        for(int i = 0 ; i<n ; i++)
            scanf("%d",&v[i]);

        for(int i = 0 ; i<n ; i++)
            scanf("%d",&t[i]);

        sort(v.begin(),v.end());
        sort(t.begin(),t.end());

        int diff = 0;

        for(int i = 0 ; i<n ; i++)
        {
            diff = max(diff,abs(v[i]-t[i]));
        }
        cout<<diff<<endl;
        int xx = 0;
        int ans = INT_MAX;
        int idx = -101;
        for(int i = -100 ; i<=100; i++)
        {
            xx = 0;
            for(int j = 0; j<n ; j++)
            {
                xx+=abs(i-v[j]);
            }

            if(ans > xx)
            {   ans = xx;
                idx = i;
            }
        }

        cout<<idx<<endl;
    }

    return 0;
}
