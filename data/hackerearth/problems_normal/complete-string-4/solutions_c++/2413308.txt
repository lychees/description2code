#include<iostream>
#include<cstring>
#include<cassert>
#include<cmath>
#include<algorithm>

using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    while(n--){
    	string str;
    	cin>>str;
    	int len=str.length(),flag=0;
    	int a[26]={0};
    	for(int i=0;i<len;i++){
    		a[str[i]-97]++;
    	}
    	for(int i=0;i<26;i++){
    		if(a[i]==0){
    			flag=1;
    			break;
    		}
    	}
    	if(flag==1)cout<<"NO"<<endl;
    	else
    	cout<<"YES"<<endl;
    }
    return 0;
}
