#include <iostream>
using namespace std;
int a[1000][1000]={0};
void put(int i,int j)
{
	a[i][j]=1;
	if((i+2)<1000)
	a[i+2][j]=-1;
	if((j+2)<1000)
	a[i][j+2]=-1;
}
int main()
{
	for(int i=0;i<1000;i++)
	{
		for(int j=0;j<1000;j++)
		{
			if(a[i][j]==0)
			put(i,j);
		}
	}
	int t;
	cin>>t;
	while(t>0)
	{
		int count=0;
		int m,n;
		cin>>m>>n;
		for(int i=0;i<m;i++)
		{
			for(int j=0;j<n;j++)
			{
				if(a[i][j]==1)
				count++;
			}
		}
		cout<<count<<endl;
		t--;
	}
    return 0;
}
