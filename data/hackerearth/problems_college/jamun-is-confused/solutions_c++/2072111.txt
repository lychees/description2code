#include <bits/stdc++.h>
using namespace std;

int gcd(int a,int b){
if(b==0){
	return a;
}
return gcd(b,a%b);

}

int main()
{
	
	int t,n;
	cin>>t;
	while(t--){
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++){
			cin>>a[i];
		}
		int x=a[0];
		for(int i=1;i<n;i++){
			x=gcd(x,a[i]);
		}
		cout<<x<<"\n";
	}
    return 0;
}
