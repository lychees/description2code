#include <iostream>
using namespace std;
const long long r=10000000; 
int main()
{	int i,j,k,x[r],y[r],n[r],m;
	long long t,seq[50]={0};
    cin>>t;
    for(i=0;i<t;++i)
    {
    	cin>>x[i]>>y[i]>>n[i];
	}
	for(i=0;i<t;++i)
	{	for(m=0;m<50;++m)
    	{
		seq[m]=0;
		
		}	
    	for(j=0;j<x[i];++j)
    	{
    			seq[j]=y[i];
    	}
    		for(j=x[i];j<=n[i];++j)
    	{
    			for(k=1;k<=x[i];++k)
    			{
    				seq[j]=seq[j]+seq[j-k];
    			}
    	}
    
    	cout<<"\n"<<seq[n[i]-1]<<" ";
	}
    return 0;
}
