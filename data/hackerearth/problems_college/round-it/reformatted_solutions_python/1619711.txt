from math import floor, ceil
t = int(raw_input())
while t:
    t -= 1
    n, b = map(int, raw_input().split())
    if n <= b:
        print b
    else:
        a = floor(float(n)/float(b))
        c = ceil(float(n)/float(b))
        if abs(n - a*b) == abs(n - c*b):
            print int(c*b)
        elif abs(n - a*b) < abs(n - c*b):
            print int(a*b)
        else:
            print int(c*b)
