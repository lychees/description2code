#include<bits/stdc++.h>
#define sz(x) (int)(x.size())
 
using namespace std;
typedef long long lld;
 
int main()
{
	
	string s; cin >> s;
	int flag = 0;
	for(int i = 0; i < sz(s); i++)
	{
		if(s[i] >= 'a' && s[i] <= 'z')
		{
		   if(flag == 0)
		   {
		   	 s[i] = ( (s[i]-'a') + 'A');
		   	 flag = 1;
		   }
		}
		else flag = 0;
	}
	cout << s << endl;
	return 0;
}
