t=input()
while t>0:
	t-=1
	k=input()
	s=raw_input()
	r=""
	for i in range(len(s)):
		c=ord(s[i])-k
		if c>=65:
			r+=chr(c)
		else:
			r+=chr(c+26)
		
	print r
