#include <iostream>
using namespace std;
#include<cstring>
#include<set>
#define ul unsigned long
#define ll long long
int f(ul mid, ll m, ul *a,ll n){
	ll i=0,j,k;
	while(m){
		ul c=mid;
		while(1 && i<n){
			if(c>=a[i])
				c-=a[i++];
			else{
				break;
			}
		}
		m--;
	}
	if(i==n){
		return 1;
	}
	return 0;
}
int main()
{
	ll t,i,j,k,n,m;
	ul a[1000006];
	cin>>n>>m;
	ul mi=0;
	for(i=0;i<n;i++){
		cin>>a[i];
		if(mi<a[i]){
			mi=a[i];
		}
	}
	ul l=0,h=m*mi*mi*m*100;
	ul mid;
	while(l<=h){
		mid =(l+h)/2;
		int v=f(mid,m,a,n);
		if(v==1 && f(mid-1,m,a,n)!=1){
			 break;
 
		}
		else if(v==1){
			h=mid-1;
		}
		else
			l=mid+1;
	}
	cout<<mid<<endl;
	return 0;
}
