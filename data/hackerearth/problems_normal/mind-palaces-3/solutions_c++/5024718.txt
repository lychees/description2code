#include<iostream>
using namespace std;
int main()
{
    int arr[1000][1000],i,j,k;
    int m,n;
    int N,c;
    cin>>m;
    cin>>n;
    for(i=0;i<m;i++)
        for(j=0;j<n;j++)
            cin>>arr[i][j];
    cin>>N;
    for(c=0;c<N;c++)
    {
        i=0;
        j=n-1;
        cin>>k;
     while ( i < m && j >= 0 )
     {
      if ( arr[i][j] == k )
      {
         cout<<i<<" "<<j<<endl;
         break;
      }
      if ( arr[i][j] > k )
        j--;
      else
        i++;
    }
    if(i==m || j==-1)
    {
    	cout<<"-1 -1"<<endl;
    }
    }
    return 0;
}
