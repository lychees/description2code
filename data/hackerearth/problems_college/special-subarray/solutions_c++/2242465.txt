#include <stdio.h>
using namespace std;
 
const int N = 100000 +10;
 
bool ans;
 
int n;
int tdnum;
int mlo;
int mhi;
int a[N];
int down_p[N];
int up_p[N];
 
void ri()
{
	int i;
	
	scanf("%d%d", &n, &tdnum);
	for (i = 1; i <= n; i++) scanf("%d", &a[i]);
	up_p[n] = n;
	for (i = n - 1; i >= 1; i--) if (a[i] <= a[i + 1]) up_p[i] = up_p[i + 1]; else up_p[i] = i;
	down_p[1] = 1;
	for (i = 2; i <= n; i++) if (a[i - 1] >= a[i]) down_p[i] = down_p[i - 1]; else down_p[i] = i;
}
 
void print() {if (ans) printf("Yes\n"); else printf("No\n");}
 
int main()
{
	ri();
	while (tdnum--)
	{
		scanf("%d%d", &mlo, &mhi);
		ans = up_p[mlo] >= down_p[mhi];
		print();
	}
	return 0;
}
