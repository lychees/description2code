
import sys
import itertools

for _ in range(int(sys.stdin.readline())):
	n = int(sys.stdin.readline())
	arr = map(int, sys.stdin.readline().strip().split())
	s = int(sys.stdin.readline())
	if s == 0 or s in arr:
		sys.stdout.write('%s\n' % 'YES')
	else:
		y = 0
		for i in range(1, n+1):
			for b in itertools.combinations(arr, i):
				if sum(b) == s:
					sys.stdout.write('%s\n' % 'YES')
					y = -1
					break
			if y == -1:
				break
		else:
			sys.stdout.write('%s\n' % 'NO')
