#include <iostream>
using namespace std;
#define rem 1000000007
int main()
{
   int t;
   long long n;
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	if(n<4) cout<<0<<endl;
   	else if(n==4) cout<<1<<endl;
   	else if(n==5) cout<<5<<endl;
   	else
   	{
   		cout<<(n*(n-3))/2 + n*(n-5)<<endl; 
   	}
   }
}
