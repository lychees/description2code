#include <bits/stdc++.h>
#define ll long long
#define pb push_back
#define mk make_pair
#define pi pair<int, int>
#define fi first
#define se second
using namespace std;

int main() {
	int t;
	cin >> t;
	while(t--) {
		int n;
		cin >> n;
		vector < string > v;
		for(int i = 0; i < n; i++) {
			string su;
			cin >> su;
			v.pb(su);
		}
		string s;
		cin >> s;
		int ans[26];
		memset(ans, 0, sizeof ans);
		for(int i = 0; i < s.length(); i++) {
			ans[s[i] - 'a']++;
		}
		vector < string > v1(n / 2), v2(n - n / 2);
		for(int i = 0; i < n / 2; i++) {
			v1[i] = v[i];
		}
		
		for(int i = n / 2, j = 0; i < n; i++, j++) {
			v2[j] = v[i];
		}
		map < vector < int >, int> m;
		for(int i = 0; i < (1 << (n - n / 2)); i++) {
			vector < int >  ct(26, 0);
		//	memset(ct, 0, sizeof ct);
			for(int j = 0; j < n - n / 2; j++) {
				if(i & (1 << j)) {
					for(int k = 0; k < v2[j].length(); k++) {
						ct[v2[j][k] - 'a']++;
					}
				}
				//return 0;
			}
			m[ct] = 1;
			
		}
		int ch = 0;
		for(int i = 0; i < (1 << (n / 2)); i++) {
			vector < int > ct(26, 0), ct1(26, 0);
			for(int j = 0; j < n / 2; j++) {
				if(i & (1 << j)) {
					for(int k = 0; k < v1[j].length(); k++) {
						ct[v1[j][k] - 'a']++;
					}
				}
			}
			int tot = 0;
			for(int j = 0; j < 26; j++) {
				ct1[j] = ans[j] - ct[j];
				if(ct1[j] == 0) tot++;
			}
			if(tot == 26 || m[ct1] == 1) {
				ch = 1;
				break;
			}
		}
		if(ch) {
			cout << "YES\n";
		} else {
			cout << "NO\n";
		}
	}
	return 0;
}
