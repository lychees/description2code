t = input()
fact = []
modulo = 10**6 + 3

fact.append(1)
i = 1
while i < (10**6+4):
    fact.append((fact[i-1] * i) % modulo)
    i = i+1

for i in range(t):
    (n, x) = map(int, raw_input().split(" "))
    x = x % modulo

    if n >= modulo:
        print 0
    else:
        print(x*fact[n]) % modulo
