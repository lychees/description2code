#include <iostream>
#include<string.h>
using namespace std;
int values(char a)
{
	if(a=='0'||a=='9'||a=='6') return 6;
	else
	if(a=='1') return 2;
	else
	if(a=='2'||a=='3'||a=='5') return 5;
	else
	if(a=='4') return 4;
	else
	if(a=='7') return 3;
	else
	if(a=='8') return 7;
	else
	return 0;
}
int main()
{
	int sum;
	char A[50];
	cin>>A;
	int n=strlen(A);
	for(int i=0;i<n;i++)
    	sum=sum+values(A[i]);
    cout<<sum;
   // cout<<A[0];
    return 0;
}
