using namespace std;
#define size 200
int a[size];
int p[102];
void pre()
{
 for(int i=2;i<=102;i++)
 {
  if(!p[i])
  {
  p[i]=1;
   for(int j=i+i;j<=102;j+=i)
    if(!p[j])
     p[j]=-1;
  // res[i]=res[i-1]+i;
  }
  //else
  // res[i]=res[i-1]+p[i];
 }
}
int main()
{
	pre();
 int n,index,temp,i,num;
 a[0]=1;

 scanf("%d",&num);
 for(int j=1;j<=num;j++)
 { 
 if(p[j]==1)
 {
 n=j;
  index=0;
  for(int k=0;k<size;k++)
  a[k]=1;
 for(;n>=2;n--)
 {
 temp=0;
 for(i=0;i<=index;i++)
 {
 temp=(a[i]*n)+temp;
 a[i]=temp%10;
 temp=temp/10;
 }
 while(temp>0)
 {
 a[++index]=temp%10;
 temp=temp/10;
 }
 }
 
 for(i=index;i>=0;i--)
 printf("%d",a[i]);
 printf("\n");
}
}
 return 0;
}
