#include <bits/stdc++.h>
using namespace std;


int BIT[1000005];

int sum_bit(int size, int idx)
{
	int sum = 0;
	while (idx != 0) {
		sum += BIT[idx];
		idx = idx - (idx & -idx);
	}
	return sum;
}

void update_bit(int size, int idx, int val)
{
	while (idx <= size) {
		BIT[idx] += val;
		idx = idx + (idx & -idx);
	}

}
