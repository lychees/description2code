score=[[0,-1,1],[1,0,-1],[-1,1,0]]
def rps(x):
	if x=='R':return 0
	if x=='P':return 1
	if x=='S':return 2
for _ in xrange(input()):
	n=input()
	maxi=0
	winner=0
	a=[]
	for i in xrange(n):
		a.append(raw_input())
	for i in xrange(n):
		tmp=0
		for j in xrange(n):
			if i!=j:
				c=0
				for k in xrange(5):
					c+=score[rps(a[i][k])][rps(a[j][k])]
				if c>0:
					tmp+=1
		if tmp>maxi:
			maxi=tmp
			winner=i
	print winner
