#include <iostream>
using namespace std;

int main()
{
    int c[20],w[20],n,k,angle,temp;
    scanf("%d %d",&n,&k);
    int arr[366]= {0};
    for(int i=0; i<n; i++)
    {
        scanf("%d",&c[i]);
    }

    for(int i=0; i<n; i++)
    {
        angle=c[i];
        arr[angle]=1;
        for(int j=0; j<360; j++)
        {
            angle+=c[i];
            temp=angle%360;
            arr[temp]=1;
        }
    }
    for(int i=0; i<k; i++)
    {
        scanf("%d",&w[i]);
        if(arr[w[i]]==1) printf("YES\n");
        else printf("NO\n");
    }
    return 0;
}
