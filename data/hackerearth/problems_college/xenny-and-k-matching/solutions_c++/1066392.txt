// A O(Logn) complexity program to count set bits in all numbers from 1 to n
#include <stdio.h>
#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int main(){
ll n,k,i,j,count=0;
cin>>n>>k;
ll a[n];
for(i=0;i<n;i++)
    cin>>a[i];
for(i=0;i<n-k;i++){
    if (abs(a[i]-a[i+k])%k==0) count++;
   else if (abs(a[i]+a[i+k])%k==0) count++;
   else continue;
}
cout<<count<<endl;

return 0;
}
