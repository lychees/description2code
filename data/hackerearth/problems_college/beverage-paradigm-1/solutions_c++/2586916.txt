#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;
int chose(int ar[],int n,int k)
{
	int i;
	sort(ar,ar+n);
	for(i=0;i<n-2;i++)
	{
		int l=i+1;
		int r = n-1;
		while(l<r)
		{
		if(ar[i]+ar[l]+ar[r]==k)
		{
			return 1;
			}
			else
			if(ar[i]+ar[l]+ar[r]>k)
			{
				r--;
				}
				else
				{
					l++;
					}	
		}
	}
	return 0;
}
int main()
{
	int n,i;
	cin>>n;
	int a[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	int k ;
	cin>>k;
	int p = chose(a,n,k);
	if(p==1)
	{
		cout<<"True"<<"\n";
	}
	else
	{
		cout<<"False"<<"\n";
	}
  return 0;
}
