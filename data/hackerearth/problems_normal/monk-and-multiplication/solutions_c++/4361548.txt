#include <bits/stdc++.h>
using namespace std;
#define dbg(x) { cout<<#x<<" : "<<x<<endl; }

long long int pq[3];

long long int calc_sum()
{
	long long int sum=pq[0]*pq[1]*pq[2];
	return sum;
}

void pq_insert(int num)
{
	if(num < pq[0])
		return;
	if( num <= pq[1] ) {
		pq[0] = num;
		return;
	}
	if( num<= pq[2] ) {
		pq[0] = pq[1];
		pq[1] = num;
		return;
	}
	if( num > pq[2]) {
		pq[0]=pq[1];
		pq[1]=pq[2];
		pq[2]=num;
		return;
	}
}

int main()
{
	 int n;
	cin>>n;
	int a[n];
	////sum = 0;
	memset(pq,0,sizeof(pq));
	for( int i=0; i<n; i++)
		cin>>a[i];
	for(int i=0; i<n; i++) {
		if ( i<= 1)
			cout<<"-1"<<endl;
		else if( i ==2 ) {
			pq[0]=a[0];
			pq[1]=a[1];
			pq[2]=a[2];
			sort(pq,pq+3);
			//dbg(pq[0]);
			//dbg(pq[1]);
			//dbg(pq[2]);
			cout<<calc_sum()<<endl;
		}
		else {
			pq_insert(a[i]);
			cout<<calc_sum()<<endl;
			
		}
	}
    return 0;
}
