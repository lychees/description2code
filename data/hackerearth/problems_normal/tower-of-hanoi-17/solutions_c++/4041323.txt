#include <bits/stdc++.h>
#define ll long long int
using namespace std;

struct disc {
	int r,h;
} a[201];

bool comp(const disc &a, const disc &b) {
	if (a.r < b.r) {
		return 1;
	}
	if (a.r == b.r && a.h < b.h) {
		return 1;
	}
	return 0;
}

ll dp[201];

int main()
{
	int test;
	cin >> test;

	while (test--) {
		int n;
		cin >> n;
		for (int i = 0; i < 201;i++) {
			dp[i] = -1;
		}
		for (int i = 0; i < n; i++) {
			cin >> a[i].r >> a[i].h ;
		}
		sort(a,a+n,comp);
		dp[0] = a[0].h;
		ll mx = LLONG_MIN;

		for (int i = 1; i < n; i++) {
			dp[i] = a[i].h;
			for (int j = 0; j < i; j++) {
				if (a[i].r > a[j].r && a[i].h > a[j].h) {
					dp[i] = max(dp[i],dp[j] + a[i].h);
				}
			}
			mx = max(mx,dp[i]);
		}
		cout << mx << endl;
	}

	return 0;
}
