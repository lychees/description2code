#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
long long int A[100005];
long long int B[100005];
long long int C[100005];
int main()
{
	int te;
	scanf("%d",&te);
	while(te--)
	{
		long long a,b,c;
		scanf("%lld %lld %lld",&a,&b,&c);
		
		int k=0;
		int i=0;
		int j;
		for (j = 1; j <= sqrt(a); j++) {
			if (j * j == a) {
				A[k++]= j;
			} else if (a % j == 0) {
				A[k++]=j;
				A[k++]=a/j;
			}
		}
		for (j = 1; j <= sqrt(b); j++) {
			if (j * j == b) {
				B[i++]=j;
			} else if (b % j == 0) {
				B[i++]=j;
				B[i++]=b/j;
			}
		}
		sort(A,A+k);
		sort(B,B+i);
		long long low=0;
		long long high=i-1;
		int s=0;
		long long mid;
		for(j=0;j<k;j++) 
		{
			int co=0;
			while(low<=high)
			{
				mid=(low+high)/2;
				if(B[mid]==A[j])
				{C[s++]=A[j];co=1;break;}
				else if(B[mid]<A[j])
				low=mid+1;
				else
				high=mid-1;
			}
			if(co==1)
			low=mid+1;
			else
			low=0;
			high=i-1;
		}
		if(s<c)
		printf("No crymeth today\n");
		else
		{
			printf("%lld\n",C[s-c]);
		}
	}
	return 0;
}
