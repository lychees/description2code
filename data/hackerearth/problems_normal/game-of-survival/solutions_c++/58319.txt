#include<stdio.h>

#include<stdlib.h>

int compare(const void *a, const void *b)
{
    return (*(int *)a-*(int *)b );
}
int main()
{
    int test;
    scanf("%d",&test);
    
    while(test>0)
    {
                 int n; 
                 scanf("%d",&n);
                 int sh[100000],st[100000];
                 for(int i=0;i<n;i++)
                 {
                         scanf("%d",&sh[i]);
                 }
                 for(int i=0;i<n;i++)
                 {
                         scanf("%d",&st[i]);
                 }
                 qsort(sh,n,sizeof(int),compare);
                 qsort(st,n,sizeof(int),compare);
                 
                 int i;
                 for(i=0;i<n;i++)
                 {
                         if(sh[i]<st[i])
                                        break;
                 }
                 if(i==n)
                         printf("YES\n");
                 else
                         printf("NO\n");
                 test--;
    }
    
}
