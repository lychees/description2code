
#include<bits/stdc++.h>

using namespace std;
int main(){
  
  int t;
  string temp;
  vector<string> str;
  cin>>t;
  
  for(int i=0;i<t;i++)
  {
  	cin>>temp;
  	sort(temp.begin(),temp.end());
  	str.push_back(temp);
  }
  
  sort(str.begin(),str.end());
  
  int current=1,ans=1;
  
  for(int i=1;i<t;i++)
  {
  	if(str[i]==str[i-1])
  	   current++;
  	else
  	   current=1;
  	
  	if(current>ans)
  	   ans=current;
  	else
  	   ans=ans;
  	
  }
  
  cout<<ans;
  
}
