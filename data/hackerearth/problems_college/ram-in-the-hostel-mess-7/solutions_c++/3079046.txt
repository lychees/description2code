inline lld func(lld n)
{
    lld a[n], b[n];
    a[0] = b[0] = 1;
    for (int i = 1; i < n; i++)
    {
        a[i] = a[i-1] + b[i-1];
        b[i] = a[i-1];
    }
    return a[n-1] + b[n-1];
}

{
	int i,j,t,N;
    lld ans,res;
    inp1(t);
    while(t--)
    {
        inp1(N);
        ans=func(N);
        res=ans*ans;
        cout<<res<<endl;
    }
    return 0;
