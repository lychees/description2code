    #include <cstdio>
    #include <algorithm>
    using namespace std;
    #define MAX 100010
    int left[MAX];
    long long ar[MAX];
    int main(){
    int T, N;
    scanf("%d", &T);
    while(T--){
    scanf("%d", &N);
    for(int i=1;i<=N;i++)
    scanf("%lld", &ar[i]);
    for(int i=0;i<=N;i++)
    left[i] = 0;
    int l, r;
    int ans = MAX, tmp;
    long long sum = 0;
    for(int i=1;i<=N;i++){
    sum = sum + ar[i];
    sum = sum%N;
    if(left[sum]==0){
    if(sum==0)
    tmp = i;
    else
    tmp = MAX;
    }else{
    tmp = i - left[sum];
    }
    if(tmp<ans){
    ans = tmp;
    l = left[sum] + 1;
    r = i;
    }
    left[sum] = i;
    }
    printf("%d %d\n", l, r);
    }
    return 0;
    }
