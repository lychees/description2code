#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while (t--) {
        int n, m;
        cin >> n >> m;
        vector<string> grid(n);
        for (int i = 0; i < n; i++) {
            cin >> grid[i];
        }
        int cnt = 0;
        bool flag = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == '1') {
                    cnt++;
                    if (i > 0 && grid[i-1][j] == '1') {
                        flag = 1;
                    } if (j > 0 && grid[i][j-1] == '1') {
                        flag = 1;
                    } if (j < m-1 && grid[i][j+1] =='1') {
                        flag = 1;
                    } if (i < n-1 && grid[i+1][j] == '1') {
                        flag = 1;
                    }
                }
            }
        }
        if (n == 1 && m == 1) {
            if (cnt == 1) {
                    cout << "yes 0" << endl;
            } else {
                cout << "no 1" << endl;
            }
            continue;
        }
        if (flag) {
            cout << "yes 0" << endl;
        } else {
            if (cnt > 1) {
                cout << "no 1" << endl;
            } else {
                cout << "no 2" << endl;
            }
        }
    }
}
