def Gcd(a, b):
    if b % a == 0:
        return a
    else:
        return Gcd(b % a, a)


C = 1000000007
N = int(raw_input())
A = [int(x) for x in raw_input().split()]

if N == 1:
    gx = A[0]
else:
    temp = A[0]
    for i in range(N):
        temp = Gcd(temp, A[i])
    gx = temp

X = 1
for i in range(N):
    X *= A[i]
    X %= C

ans = 1
a = X
while (gx != 0):
    if gx % 2 == 1:
        ans *= a
        ans %= C
    a *= a
    a %= C
    gx = gx/2
print ans
