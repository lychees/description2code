#include <iostream>
using namespace std;

long long int mod = 1000000007;

long long int power(long long int x, long long int y)
{
    long long int temp;
    if( y == 0)
        return 1;
    temp = power(x, y/2);
    if (y%2 == 0)
        return (temp*temp) % mod;
    else
        return ((((x*temp)% mod)*temp) % mod);
}

int main()
{
    int t, n, i, k;
    
    cin>>t;
    
    while(t--) {
    	cin>>n;
    	
    	k = power(3, n);
    	
    	cout<<k%mod<<endl;
    }
    return 0;
}
