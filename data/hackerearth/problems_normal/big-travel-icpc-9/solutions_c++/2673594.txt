#include<bits/stdc++.h>
using namespace std;
struct node
{
	long long x,y;
};
bool cmp(node a,node b)
{
	return a.x<b.x;
}
bool cmp2(node a,node b)
{
	return a.y<b.y;
}
int main() 
{
	int n;
	cin>>n;
	node a[200009]={0};
	for(int i=0;i<n;i++)cin>>a[i].x>>a[i].y;
	sort(a,a+n,cmp);//x
	long long ans=0,sum=0;
	#define mod 1000000007
	for(int i=1;i<n;i++)
	{
		sum=a[i].x-a[i-1].x;
		sum=(((sum*(n-i))%mod)*i)%mod;
		ans=(ans+sum)%mod;
	}
	sort(a,a+n,cmp2);//y
	sum=0;
	for(int i=1;i<n;i++)
	{
		sum=a[i].y-a[i-1].y;
		sum=(((sum*(n-i))%mod)*i)%mod;
		ans=(ans+sum)%mod;
	}
	cout<<ans%mod<<endl;
	return 0;
}
