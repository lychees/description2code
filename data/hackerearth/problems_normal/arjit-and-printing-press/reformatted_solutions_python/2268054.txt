t = int(raw_input())
while t != 0:
    o = list(raw_input())
    d = raw_input()
    d = sorted(d)
    i = 0
    j = 0
    while i < len(o):
        if j == len(d):
            break
        else:
            if o[i] > d[j]:
                o[i] = d[j]
                j += 1
        i += 1
    print "".join(o)
    t -= 1
