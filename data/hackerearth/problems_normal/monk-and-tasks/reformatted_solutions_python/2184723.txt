def mySort(t, N):
    ele = [[] for i in range(61)]
    i = 0
    while (i < N):
        ele[bin(t[i]).count('1')].append(t[i])
        i = i + 1
    j = 1
    while (j < 60):
        if (len(ele[j]) > 0):
            print ' '.join(map(str, ele[j])),
        j = j+1
    print ' '.join(map(str, ele[60]))


T = int(raw_input())
