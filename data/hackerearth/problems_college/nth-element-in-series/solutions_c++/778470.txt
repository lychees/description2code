#define MOD 1000000007

int n;
int x[5];

inline void mat_mult(int a[5][5], int b[5][5], int c[5][5])
{
	int i, j, k;
	int temp[5][5];

	FOR(i, 0, 4) {
		FOR(j, 0, 4) {
			temp[i][j] = 0;
			FOR(k, 0, 4) {
				temp[i][j] = (temp[i][j] + (a[i][k] *1LL* b[k][j]) % MOD) % MOD;
			}
		}
	}

	FOR(i, 0, 4) {
		FOR(j, 0, 4) {
			c[i][j] = temp[i][j];
		}
	}
}

inline int solve()
{
	int res[5][5], base[5][5];
	int i, j;

	FOR(i, 0, 4) {
		FOR(j, 0, 4) {
			res[i][j] = base[i][j] = 0;
		}
		res[i][i] = 1;
	}

	base[0][0] = base[0][1] = base[0][2] = base[0][4] = 1;
	base[1][0] = base[2][1] = base[3][2] = base[4][3] = 1;
	base[0][3] = 2;

	int p = n-4;

	while (p) {
		if (p & 1) {
			mat_mult(res, base, res);
		}
		mat_mult(base, base, base);
		p >>= 1;
	}

	int ans = 0;

	FOR(i, 0, 4) {
		ans = (ans + (res[0][i] * 1LL * x[4-i]) % MOD )% MOD;
	}

	return ans;
}
