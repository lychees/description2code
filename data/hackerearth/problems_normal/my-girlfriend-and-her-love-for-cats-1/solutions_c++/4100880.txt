#include <bits/stdc++.h>

using namespace std;

int main(){
	int n;
	cin >> n;
	int A[n],B[n];
	for(int i=0;i<n;i++){
		cin >> A[i];
	}
	for(int i=0;i<n;i++){
		cin >> B[i];
	}

	sort(A,A+n);
	sort(B,B+n);
	long long int ans,crt;
	ans = 0;
	for(int i=0;i<n;i++){
		crt = A[i];
		crt*=B[i];
		ans+=crt;
	}
	cout << ans;

}
