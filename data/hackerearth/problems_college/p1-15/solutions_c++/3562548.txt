#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t,n1,n2,i,j,flag,k,p;
    char a[1000010],b[1000010],c[2000010];
    scanf("%d",&t);
    while(t--)
    {
    	cin>>a>>b;
    	c[0]=NULL;
    	n1=strlen(a);
    	n2=strlen(b);
    	for(i=(min(n2,n1)-1);i>=0;i--)
    	{	flag=0;
    		for(j=n1-1,p=i;j>=0 && p>=0;j--,p--)
    		{
    			if(b[p]!=a[j])
    			{	//cout<<j<<" "<<p<<endl;
    				flag=1;
    				break;
    			}
    		}
    		if(flag!=1)
    		{
    			for(k=0,p=0;k<=j && p<=j;k++,p++)
    			c[k]=a[p];
    			for(j=0;j<n2;j++,k++)
    			c[k]=b[j];
    			c[k]=NULL;
    			break;
    		}
    	}
    	if(c[0]==NULL)
    	{
    		for(i=0;i<n1;i++)
    		c[i]=a[i];
    		for(j=0;j<n2;j++,i++)
    		c[i]=b[j];
    		c[i]=NULL;
    	}
    	cout<<c<<endl;
    	//cout<<a<<b<<endl;
    }
    return 0;
}
