#include <bits/stdc++.h>
using namespace std;
#define max 100000
#define mod 1000000007

long long int a[10000001] = {0};
long long int p[10000001] = {0};
//vector<int> v;
void low_sieve()
{
	long long int i,j,k=0;
	a[1] = 1;
	
	for(i=2; i<= 10000001; i++)
	{
		if(a[i]==0) 
		{   p[k++] = i;
			
			for(j=i*2; j< 10000001; j = j + i)
			{   if(a[j]==0)
			       a[j] = i; 
			}
		}
	}
}
int main()
{
	low_sieve();
	long long int t;
	scanf("%lld",&t);
	while(t--)
	{
		int n,m;
		scanf("%d %d",&n,&m);
		long long int ans=1;
		//cout<<p[n-1]<<" "<<p[p[n-1]-1]<<" "<<p[m-1]<<" "<<p[p[m-1]-1]<<endl;
		ans = (p[p[n-1]-1]%mod*p[p[m-1]-1]%mod)%mod;
		printf("%lld\n",ans);
		
	}
    return 0;
}
