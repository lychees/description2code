#include <iostream>
using namespace std;

int main()
{
    bool prime[100010]={0};
    prime[0]=1,prime[1]=1;
    for(int i=2;i<100010;i++){
    	if(prime[i]==0){
	    	for(int j=2;i*j<100010;j++){
	    		prime[i*j]=1;	
	    	}
    	}
    }
    int t; cin>>t;
    bool flag=0;
    while(t--){
    	flag=0;
    	int n; cin>>n;
    	for(int i=2;i<n;i++){
    		if(prime[i]==0&&prime[n-i]==0){
    			flag=1;
    			break;
    		}
    	}
    	if(flag) cout<<"Deepa"<<endl;
    	else cout<<"Arjit"<<endl;
    }
    return 0;
}
