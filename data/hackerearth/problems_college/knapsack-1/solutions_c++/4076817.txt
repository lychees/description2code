//#include<bits/stdc++.h>
#include<cstdio>
#include<iostream>
#include<memory.h>
using namespace std;
 
const int N=2010;
 
int dp[N];
int a[N];
 
int main() {
  int tt;
  scanf("%d",&tt);
  while(tt--) {
  	memset(dp,0,sizeof(dp));
    int n,k;
    scanf("%d%d",&n,&k);
    for(int i=0;i<n;i++) {
      scanf("%d",a+i);
    }
    dp[0]=1;
    for(int i=0;i<=k;i++) {
      for(int j=0;j<n;j++) {
        if(dp[i]==1) {
          if(i+a[j]<=k)
          dp[i+a[j]]=1;
        }
      }
    }
    for(int i=k;i>=0;i--) {
      if(dp[i]==1) {
        printf("%d\012",i);
        break;
      }
    }
  }
  return 0;
}
