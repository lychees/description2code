/*
 * SEIZE THE DAY
 */
 
 #include<bits/stdc++.h>
 using namespace std;
 int main()
 {
 	int t;
 	cin>>t;
 	while(t--)
 	{
 		int n;
 		cin>>n;
 		int ar[n],br[n];
 		for(int i=0;i<n;i++)
 		cin>>ar[i];
 		map<int,int> hash;
 		int swap=0;
 		for(int i=0;i<n;i++)
 		{
 			cin>>br[i];
 			hash[br[i]]=i;
		 }
		 
		 for(int i=0;i<n;i++)
		 {
		 	ar[i]=hash[ar[i]]+1;
		 	br[i]=hash[br[i]]+1;
		 }
		 
		 for(int i=0;i<n;i++)
		 {
		 	int x=0;
		 	for(int j=0;j<n-i-1;j++)
		 	{
		 		if(ar[j]>ar[j+1])
		 		{
		 			x++;
		 			int tmp=ar[j];
		 			ar[j]=ar[j+1];
		 			ar[j+1]=tmp;
				 }
			 }
			 swap+=x;
			 if(x==0)
			 break;
		 }
		 
		 cout<<swap<<endl;
	 }
	 return 0;
 }
