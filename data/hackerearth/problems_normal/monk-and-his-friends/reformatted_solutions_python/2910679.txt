t = int(raw_input())
while t > 0:
    instd = set()
    outstd = []
    n, m = map(int, raw_input().split())
    array = map(int, raw_input().split())
    for i in xrange(n):
        instd.add(array[i])
    i += 1
    for j in xrange(m):
        outstd.append(array[i+j])
    for i in xrange(m):
        if outstd[i] in instd:
            print 'YES'
        else:
            print 'NO'
            instd.add(outstd[i])
    t -= 1
