#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main()
{
    string s;
    vector<string> v;
    vector<string>:: iterator it;
    int n;
    cin >> n;
    for ( int i = 0; i < n; i++) {
        cin >> s;
        v.push_back(s);
    }
    sort(v.begin(),v.end());
    v.erase(unique(v.begin(),v.end()),v.end());
    //cout << "HELLLLLLLLLOOOOOOO\n";
    for ( it = v.begin(); it != v.end(); it++)
    {
        cout << *it << endl;
    }
    return 0;
}
