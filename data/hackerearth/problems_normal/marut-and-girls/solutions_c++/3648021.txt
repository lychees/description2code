#include <iostream>
#include <string.h>
#include <stdlib.h>
using namespace std;

int main()
{
    int M, N, flag = 0, sumM=0, sumN=0, ans=0, tempi;
    int arrQ[10001] = {0};
    char S[1001], prevS[1001];
    scanf("%d", &M);
    for(int i=0;i<M;i++) {
    	scanf("%d", &tempi);
    	arrQ[tempi] = 1;
    	sumM += tempi;
    }

    cin >> N;

    for(int i=0;i<N;i++) {
    	sumN=0;
    	scanf("%s", S);
		strcpy(prevS,S);

    	if(arrQ[atoi(S)]==1){
    		sumN += atoi(S);
    	}
    		while(std::cin.peek() != '\n') {
    			scanf("%s", S);
    			if(!strcmp(prevS,S)) {
    				break;
    			} else {
    				if((sumN!=sumM) && arrQ[atoi(S)]==1){
    					sumN += atoi(S);
    				}
    				strcpy(prevS,S);
    			}
    		}
    	
    	//cout << "sumN = " << sumN << endl;
    	if(sumM == sumN) {
    		ans++;	
    	}
    }
   	printf("%d\n", ans);
    return 0;
}
