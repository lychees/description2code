import math
import sys


def gcd(a, b):
    if (b == 0):
        return a
    else:
        return int(gcd(b, a-b*math.floor(a/b)))


for _ in xrange(int(input())):
    p, q = map(int, sys.stdin.readline().split())
    if gcd(p, q) == 1:
        print("Is a Co-Prime")
    else:
        print("Not a Co-Prime")
