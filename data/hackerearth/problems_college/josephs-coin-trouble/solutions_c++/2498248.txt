#include<bits/stdc++.h>
using namespace std;
int coins[10000];
int findWays(long long int n,int m)
{
	//cout<<n<<" "<<m<<endl;
	long long int mat[n+1][m];
	for(int i=0;i<m;i++)
		mat[0][i]=1;
	for(int i=1;i<=n;i++)
	{
		for(int j=0;j<m;j++)
		{
			int x=(i>=coins[j])?mat[i-coins[j]][j]:0;
			int y=(j>=1)?mat[i][j-1]:0;
			mat[i][j]=x+y;
		}
	}
	return mat[n][m-1];
}
int main()
{
	char x[10];
	long long int amount;
	char str[100000];
	int ci=0;
	cin>>amount;
	cin.getline(x,10);
	cin.getline(str,sizeof(str));
	for(int i=0;str[i]!='\0';i++)
	{
		if(str[i]!=' ')
			coins[ci++]=str[i]-'0';
	}
	cout<<findWays(amount,ci)<<endl;
}
