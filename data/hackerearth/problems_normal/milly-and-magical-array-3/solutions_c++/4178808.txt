#include <bits/stdc++.h>
#define lli long long int
#define M 1000000007

using namespace std;

void swap(int *a, int *b) {
    int t = *a;
    *a = *b;
    *b = t;
}

int f[1000005];

int main()
{
    lli n, t, t1, t2, x, y, z;
    lli A[100005];

    cin >> t;
    while (t--) {
        cin >> n >> x;
        lli cnt = 0;

        for (int i = 0; i < n; i++) {
            cin >> A[i];
        }
        for (int i = 1; i < n; i++) {
            if (A[i] <= A[i - 1]) {
                int tmp = ((A[i - 1] - A[i]) / x) + 1;
                cnt += tmp;
                A[i] += tmp * x;
            }
        }
        cout << cnt << endl;
    }

    return 0;
}
