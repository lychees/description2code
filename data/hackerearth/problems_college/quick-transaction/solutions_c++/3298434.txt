#include<stdio.h>
int main(){
	long long int t, n, c, r; scanf("%lld", &t);
	while(t-- && scanf("%lld", &n)){
		c = 0;
		while(n){
			if(n >= 1000)	c += (n / 1000), r = n % 1000;
			else if(n >= 500)	c += (n / 500), r = n % 500;
			else if(n >= 100)	c += (n / 100), r = n % 100;
			else if(n >= 50)	c += (n / 50), r = n % 50;
			else if(n >= 20)	c += (n / 20), r = n % 20;
			else if(n >= 10)	c += (n / 10), r = n % 10;
			else if(n >= 5)	c += (n / 5), r = n % 5;
			else if(n >= 2)	c += (n / 2), r = n % 2;
			else if(n >= 1)	c += (n / 1), r = n % 1;
			n = r;
		}
		printf("%lld\n", c);
	}		
	return 0;
}
