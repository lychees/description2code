#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

#define MAX 31
#define MOD 1000000007

ll R[MAX];

ll power(ll a, ll b){
	if(!b) return 1;
	if(b == 1) return a;
	ll temp = power(a,b/2);
	if(b&1)
		return temp*temp%MOD*a%MOD;
	return temp*temp%MOD;
}

void init(){

	R[0] = R[1] = R[2] = 1;
	R[3] = R[4] = 3;
	R[5] = R[6] = 15;
	R[7] = R[8] = R[9] = R[10] = 105;

	ll i,b,ans1,ans2,ans3;

	for(i=11;i<=31;++i){

		b = i/4;
		if((i%4)!= 0) b++;
		ans1 = power(2,b);

		b = i/5;
		if((i%5) != 0) b++;
		ans2 = R[b];

		b = i/10;
		if((i%10) !=0) b++;
		ans3 = R[b];

		R[i] = (((ans1*ans2)%MOD)*ans3)%MOD;
	}
}

ll gcd(ll a, ll b){
	if(!b) return a;
	return gcd(b,a%b);
}

ll max(ll a, ll b){
	return a>b?a:b;
}

int main(){
	init();

	ll t,n,_max,i,j,temp;
	scanf("%lld",&t);

	while(t--){
		scanf("%lld",&n);

        if(!n) printf("0\n");
        else{
            temp = R[n];
            _max = 0;

            for(i=1;i<=5;++i){
                for(j=1;j<=5;++j){
                    if(gcd(i,j) == 1)
                        _max = max(_max,power(i*temp,j*temp));
                }
            }
            printf("%lld\n",_max);
        }
	}

	return 0;
}
