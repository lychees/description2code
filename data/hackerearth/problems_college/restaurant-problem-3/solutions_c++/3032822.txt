#include<iostream>
using namespace std;
#include<string.h>
int main()
{
long len,i,dig1,dig2,dig3;
long long count=0;
char str[1000005];
 
cin>>str;
 
len=strlen(str);
int k=0;
 
for(i=0;i<len;i++)
{
    dig1= str[i]-'0';
    if(dig1%8==0)
    count++;
 
    if(i<len-1)
    {
    dig2=dig1*10+(str[i+1]-'0');
    if(dig2%8==0 && dig2>10)
    count++;
    }
 
    if(i<len-2)
    {
    dig3=dig2*10+(str[i+2]-'0');
    if(dig1>0)
    k++;
    if(dig3%8==0)
    count+=k;
    }
}
 
cout<<count;
//Shalvin
//Shalvin
return 0;
}
