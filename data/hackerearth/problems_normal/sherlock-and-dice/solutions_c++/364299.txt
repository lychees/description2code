using namespace std;
typedef long long LL;

LL Q=1000000007;
int k;
int n, m;
double dp[60][100002];
bool ff[60][100002];

double f(int i, int s)
{
    if (s > 10000){
        return 0;
    }
    if (i == m){
        if (s == k){
            return 1;
        }
        return 0;
    }
    if (ff[i][s] == 1){
        return dp[i][s];
    }
    double ans=0;
    for (int j=1;j<=n;j++){
        ans += ((1.0)/n)*f(i+1, s+j);
    }
    ff[i][s]=1;
    dp[i][s]=ans;
    return ans;
}
double eps = 1e-9;

int main()
{
    int test;
    cin>>test;
    while (test--){
        memset(dp, 0, sizeof(dp));
        memset(ff, 0, sizeof(ff));
        cin>>m>>n>>k;
        double ans=f(0, 0);
        int y=0;
        int cnt=0;
        int fff=0;
        while (1){
            if (cnt > 100){
                fff=1;
                break;
            }
            cnt++;
            int t=(int)ans;
            if ( t > 0){
                break;
            }
            y++;
            ans *= 10;
        }
        if (fff == 0){
            printf("%.3f ", ans);
            cout<<y<<"\n";
        } else {
            cout<<"0.000 0\n";
        }
    }
    return 0;
}
