#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    string s;
    int i,j;
    cin>>s;
    for(i=0;i<s.length();i++)
    {
    	for(j=0;j<s.length()-i-1;j++)
    	{
    		if(s[j]>=s[j+1])
    		{
    			char c=s[j];
    			s[j]=s[j+1];
    			s[j+1]=c;
    		}
    	}
    }
    cout<<(s);
    return 0;
}
