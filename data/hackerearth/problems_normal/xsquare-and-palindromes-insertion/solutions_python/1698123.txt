def main(string):
    ch_map = {}
    strlen = 0
    for ch in string:
        try:
            ch_map[ch] += 1
        except:
            ch_map[ch] = 1
        strlen += 1
    odd_count = 0
    for key, value in ch_map.items():
        if value % 2 == 0:
            pass
        else:
            odd_count += 1
    print 0 if odd_count == 0 else odd_count-1
