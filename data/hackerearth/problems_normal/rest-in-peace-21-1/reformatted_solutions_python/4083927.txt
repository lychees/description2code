tc = int(raw_input())
while (tc > 0):
    tc = tc - 1
    n = int(raw_input())
    m = str(n)
    if (m.find('21') >= 0) or (n % 21 == 0):
        print "The streak is broken!"
    else:
        print "The streak lives still in our heart!"
