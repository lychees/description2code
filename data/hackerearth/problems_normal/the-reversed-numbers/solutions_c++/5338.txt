#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int flip(int n)
{
    int sum=0,temp;
    while(n>0)
    {
        temp=n%10;
        sum=sum*10 + temp;
        n=n/10;
    }
    return sum;
}

int main()
{
    int test;
    scanf("%d",&test);
    while(test--)
    {
        int a,b;
        scanf("%d %d",&a,&b);
        int a_temp=flip(a);
        int b_temp=flip(b);
        int temp=flip(a_temp+b_temp);
        printf("%d\n",temp);
    }
    return 0;
}
