#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	int n,p;
    	cin>>n>>p;
    	int dp[p+1],tmp,temp,i,j,sum=0;
    	memset(dp,0,sizeof(dp));
    	i=1,dp[0]=1;
    	sum=1;
    	for(i=2;i<=n;i++) {
    			temp=dp[0];
    			dp[0]=sum;
    		for(j=1;j<p;j++) {
    			tmp=temp;
    			temp=dp[j];
    			dp[j]=tmp;
    			sum+=dp[j];
    			sum=sum%mod;
    		}
    	}
    	sum=sum%mod;
    	cout<<sum<<endl;
    }
    return 0;
}
