#include <iostream>
#include<math.h>
#include<stdlib.h>

using namespace std;

int main()
{
    int n,x,l,m,result;
	cin>>n;
    
    for(int i=0;i<n;i++)
    {
    	cin>>x;
        m=log((double)x)/log(double(2));
		//cout<<"\nm is :"<<m;
        l=x-pow(2.0,m);
		//cout<<"\nl is :"<<l;
        result=2*l+1;
        cout<<result<<"\n";
    	
    }
	system("pause");
    return 0;
}
