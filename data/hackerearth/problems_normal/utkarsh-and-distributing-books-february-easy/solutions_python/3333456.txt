import sys

T=int(input())

for t in xrange(T):
    n=int(input())
    
    min=sys.maxint
    sum=0
    arr=map(int, raw_input().split())
        
    for x in arr:
        if(x < min):
            min = x
        
        sum += x    
                
    sys.stdout.write(str(min-1) + " " + str(sum-n) + "\n")
