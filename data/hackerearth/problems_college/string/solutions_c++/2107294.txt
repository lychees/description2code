using namespace std;

#ifdef WIN32
	#define I64 "%I64d"
#else
	#define I64 "%lld"
#endif

typedef long long ll;

#define f first
#define s second
#define mp make_pair
#define pb push_back
#define all(s) s.begin(), s.end()
#define sz(s) (int(s.size()))
#define fname "a"
#define ms(a,x) memset(a, x, sizeof(a))
#define forit(it,s) for(__typeof(s.begin()) it = s.begin(); it != s.end(); ++it)
#define MAXN 2002
#define MAXC 26

int n, k;
char s[MAXN];
int a[MAXN];
pair <int, int> b[MAXN];
int ok[MAXN];

int main()
{
	#ifdef LOCAL
	freopen(fname".in", "r", stdin);
	freopen(fname".out", "w", stdout);
	#endif

	scanf("%s", s);
	n = strlen(s);
	scanf("%d", &k);
	for (int i = 0; i < n; ++i)
		++a[s[i] - 'a'];
	for (int i = 0; i < MAXC; ++i)
		b[i] = mp(a[i], i);
	sort(b, b + MAXC);
	int sum = 0, z = MAXC;
	for (int i = 0; i < MAXC; ++i)
	{
		if (sum + b[i].f > k)
		{
			z = i;
			break;
		}
		sum += b[i].f;
	}
	for (int i = z; i < MAXC; ++i)
		ok[b[i].s] = 1;
	printf("%d\n", MAXC - z);
	string ans = "";
	for (int i = 0; i < n; ++i)
		if (ok[s[i] - 'a'])
			ans += s[i];
	printf("%s\n", ans.c_str());
	return 0;
}
