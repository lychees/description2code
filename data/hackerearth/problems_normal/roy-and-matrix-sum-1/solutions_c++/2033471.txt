#include<bits/stdc++.h>
 
#define lli long long
#define mp make_pair
#define pi pair<int,int>
#define pli pair<lli,lli>
#define pb push_back
 
 
using namespace std;
inline lli Pr(lli num)
{
	return (num*(num+1));
}
 
int main()
{
lli t,n,sum=0;
cin>>t;
while(t--)
{
	sum=0;
	cin>>n;
	for(lli i=0;i<n;i++)
	{
		sum+=Pr(i-0);
	}
	cout<<sum<<endl;
}
return 0;
}
