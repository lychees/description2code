using namespace std;
#define MAX 1001
int t;
int n;
int dp[MAX];
vector<pair<int, int> > v;
int main(){
	scanf("%d", &t);
	while (t--){
		v.clear();
		scanf("%d", &n);
		memset(dp, 0, sizeof(dp));
		for (int i = 0; i < n; i++){
			int kari, kari1;
			scanf("%d%d", &kari, &kari1);
			v.push_back(make_pair(kari, kari1));
		}
		sort(v.begin(), v.end());
		for (int i = 0; i < v.size(); i++){
			dp[i] = 1;
			for (int j = 0; j < i; j++){
				if (v[i].second > v[j].second){
					dp[i] = max(dp[i], dp[j] + 1);
				}
			}
		}
		int maxt = 0;
		for (int i = 0; i < v.size(); i++){
			maxt = max(maxt, dp[i]);
		}
		printf("%d\n", maxt);
	}
	return 0;
}
