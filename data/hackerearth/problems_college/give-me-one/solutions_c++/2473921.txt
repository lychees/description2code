#include <stdio.h>
#define ll long long int
static long long int ar[20000002];

ll min(ll a, ll b)
{
	if(a<b)
	return a;
	else 
	return b;
}

ll f(ll n)
{
	if(ar[n]!=-1)
	return ar[n];
	else
	{
		ll temp;
	temp=min(1+f(n-1),(1+(n%2)+f(n/2)));
	temp=min(temp,(1+(n%3)+f(n/3)));
	return (ar[n]=temp);
	}
	
}

int main()
{
    int t;
    long long int i;
    for(i=0;i<=20000001;i++)
   	ar[i]=-1;
   	ar[0]=0;
    ar[1]=0;
    ar[2]=1;
    ar[3]=1;
      for(i=4;i<=20000001;i++)
     ar[i]=f(i);
    scanf("%d",&t);
    while(t--)
    {
    	long long int n;
    	scanf("%lld",&n);
    //	long long int ans=f(n);
    	printf("%lld\n",ar[n]);
    }
    return 0;
}
