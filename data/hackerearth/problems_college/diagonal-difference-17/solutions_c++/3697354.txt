#include <iostream>
#include <cmath>

using namespace std;

int main()
{
    int N, A[5001][5001];
    
    cin >> N;
    
    for (int i = 0; i < N; i++)
    {
    	for (int j = 0; j < N; j++)
    		cin >> A[i][j];
    }
    
    int d1 = 0, d2 = 0;
    
    for (int i = 0; i < N; i++)
    {
    	for (int j = 0; j < N; j++)
		{
			if (i == j)
				d1 += A[i][j];
			
			if (i + j == N-1)
				d2 += A[i][j];
		}
    }
    
    int ans = abs(d1-d2);
    
    cout << ans << endl;
    
    return 0;
}
