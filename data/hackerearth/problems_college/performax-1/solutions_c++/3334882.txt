#include <iostream>
#include <stdio.h>
using namespace std;
int main(){
	int n,d;
	cin>>n>>d;
	int a,temp1,temp2,temp3;
	cin>>a;
	int result=0;
	for (int i = 1; i < n; i += 1)
	{
		cin>>temp1;
		if(temp1<=a){
			if((a-temp1)%d==0)
				temp3 = 1;
			else
				temp3 = 0;
			temp2 = (a-temp1+d-1)/d+temp3;
			result = result + temp2;
			temp1 = temp1 + d * temp2;
		}			
		a = temp1;
	}
	printf("%d",result);	
	return 0;
}
