#include "bits/stdc++.h"
using namespace std;
int main()
{
	long long dp[1000001];
	dp[0]=0;
	for(long long i=1;i<=1000000;i++)
		dp[i] = dp[i-1] + (i*(i-1))/2 ;
	
	long int tc;
	cin >> tc;
	while(tc--)
	{
		long long n;
		cin >> n;
		cout << 2*dp[n] << endl;
	}
	return 0;
}
