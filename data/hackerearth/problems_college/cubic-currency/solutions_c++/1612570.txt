#include <iostream>
#include<cmath>
using namespace std;
typedef long long ll;
int main()
{ 
	ll t,i,j,cnt,n,a[22],dp[10001]={0};
	for(i=1;i<=21;i++){
		a[i]=i*i*i;
	}
	dp[0]=1;
	for(i=1;i<=21;i++){
		for(j=a[i];j<10001;j++){
		    dp[j]+=dp[j-a[i]];
		}
	}
	scanf("%lld",&t);
	while(t--){
		
		scanf("%lld",&n);		
		printf("%lld\n",dp[n]);
	}
    
    return 0;
}
