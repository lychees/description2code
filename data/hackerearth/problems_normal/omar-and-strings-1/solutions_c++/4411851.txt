#include<bits/stdc++.h>
using namespace std;

const int maxn = 1e5+5;
char s[2*maxn],t[2*maxn];
int mp[2*maxn],cnt[2*maxn];

void process(char *s, int *mp) {
    mp[0] = -1;
    int n = strlen(s);
    for(int i = 0, j = -1; i < n;) {
        while(j > -1 && s[i] != s[j]) j = mp[j];
        mp[++i] = ++j;
    }
}
int main() {
  //  freopen("input.txt","r",stdin);
    scanf("%s",s);
    int n = strlen(s);
    for(int i = n; i < 2*n; ++i) s[i] = s[2*n-i-1];
    s[2*n] = 0;
    process(s,mp);
    int x = min(n,mp[2*n]);
    for(int i = 0; i < n; ++i) swap(s[i],s[n+i]);
    process(s,mp);
    int y = min(n,mp[2*n]);
    for(int i = 0; i < y; ++i) t[i] = s[i];
    for(int i = 0; i < x; ++i) t[i+y] = s[n-1-i];
    t[x+y] = 0;
    puts(t);
    process(t,mp);
    n = strlen(t);
    for(int i = n; i >= 1; --i) {
        ++cnt[i];
        cnt[mp[i]]+= cnt[i];
    }
    for(int i = 1; i <= n; ++i) printf("%d ",cnt[i]);
}
