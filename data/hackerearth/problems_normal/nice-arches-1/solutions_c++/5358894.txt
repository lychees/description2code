#include <iostream>
#include<cstring>
using namespace std;
const int limit = 100001;
class String{
	private :
	char str[limit];
	public :
	void getstr();
	char charAt(int i);
	int tot_len();
};
void String::getstr(){
	cin >> str;
}
char String::charAt(int i){
	return str[i];
}
int String::tot_len(){
	return strlen(str);
}

int main()
{
    int m;
    cin >> m;
    String arr[m];
    char stk[limit];
    int top,count;
    for(int i=0;i<m;i++)
       arr[i].getstr();
    count = 0;
    for(int i=0;i<m;i++){
    	top = -1;
    	stk[++top] = arr[i].charAt(0);
    	 for(int j=1;j<arr[i].tot_len();j++){
    	 	if(arr[i].charAt(j) == stk[top])
    	 	     --top;
    	 	 else
    	 	   stk[++top] = arr[i].charAt(j);
    	 }
    	 if(top == -1)
    	  count++;
    }
    cout << count << endl;
    return 0;
}
