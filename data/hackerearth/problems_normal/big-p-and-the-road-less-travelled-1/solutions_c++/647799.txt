#include <iostream>
using namespace std;

int a[100005]={0},b[100005]={0},flag[100005],v,n;
int countp(int s)
{
	if(s==n)
	return 1;
	else 
	{
		int c=0;
		for(int i=0;i<v;i++)
		{
		if(a[i]==s)
		{
			if(flag[b[i]]!=-1)
			c=c+flag[b[i]];
			
			else
			{
			flag[b[i]]=countp(b[i]);
			c=c+flag[b[i]];
				
			}
		}
		}
		return c;
	}
	
}
int main()
{
   cin>>n;
   int x,y;
   cin>>x;
   cin>>y;
   v=0;
   for(int i=1;i<=n;i++)
   flag[i]=-1;
   while(!(x==0&&y==0))
   {
   	a[v]=x;
   	b[v]=y;
   	v++;
   	cin>>x>>y;
   }
int c=countp(1);
   cout<<c<<endl;
    return 0;
}
