T = int(raw_input())
while T > 0:
    new_l = []
    num = int(raw_input())
    l = list(str(num))
    for x in l:
        if x == '2':
            new_l.append('cde')
        elif x == '7':
            new_l.append('acf')
        elif x == '6':
            new_l.append('b3')
        elif x == '9':
            new_l.append('c6a')
        else:
            new_l.append("")
    print "".join(new_l)
    T -= 1
