#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll n,k;
ll dp[203][203];
ll f(ll n, ll m)
{
	if(n==0) return 1;
	if(m==0) return 0;
	if((n<k) || (n<0) || (m<k)) return 0;
	else if(dp[n][m]!=-1) return dp[n][m];
	else 
	return dp[n][m]=f(n-m,m)+f(n,m-1);
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	memset(dp,-1,sizeof(dp));
    	ll ans=f(n,n);
    	cout<<ans<<endl;
    }
    return 0;
}
