#include <iostream>
using namespace std;

const int N = 1010;
char s[N];
int a[N], b[N];
int c[N][N];

int main()
{
	int n, m;
	scanf("%d%d", &n, &m);
	for (int i = 0; i < n; i++)
	{
		scanf(" %s ", s);
		for (int j = 0; j < m; j++)
		{
			if (s[j] == 'B')
			{
				a[i] ^= 1;
				b[j] ^= 1;
				c[i][j] = 1;
			}
		}
	}
	int ans = 0;
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
			ans += a[i] ^ b[j] ^ c[i][j];
	}
	printf("%d\n", ans);
	
    return 0;
}
