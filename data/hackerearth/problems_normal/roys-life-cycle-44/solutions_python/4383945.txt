def find_longest_streak(s,slen):
	longest_streak = 0
	streak = 0
	for i in xrange(slen):
		if s[i] == 'C':
			streak += 1
		else:
			streak = 0
		longest_streak = max(streak,longest_streak)
	return longest_streak
