#include <iostream>
using namespace std;

int main()
{
    int t;
    int r,c,w;
    cin>>t;
    int a[t];
    for(int i=0;i<t;i++)
    {
    	cin>>r>>c>>w;
    	int ans=c/w;
    	ans=ans*r;
    	ans=ans+w-1;
    	a[i]=ans;
    }
    for(int i=0;i<t;i++){
    	cout<<"Case #"<<i+1<<": "<<a[i]<<"\n";
    }
    return 0;
}
