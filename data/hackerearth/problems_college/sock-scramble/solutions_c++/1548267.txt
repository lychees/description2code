#include <iostream>
#include<bits/stdc++.h>
#include<algorithm>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		long long int visited[100005];
		long long int arr[100005];
		//memset(visited,0,sizeof visited);
		for(long long int i=0;i<=100000;i++)
		{
			visited[i]=0;
		}
		for(int i=1;i<=n;i++)
		{
		   
            cin>>arr[i];
			visited[arr[i]]++;
		}
		for( int i=1;i<=n;i++)
		{
			if(visited[arr[i]]==1)
			{
			      cout<<arr[i]<<endl;
			}
		}
	}
    //cout << "Hello World!" << endl;
    return 0;
}
