#include <bits/stdc++.h>
using namespace std;
#define MP(x,y) make_pair(x,y)
int main()
{
    string s;
    cin>>s;
    map<pair<int,int>,int > mp;
    int n = s.size();
    int x=0, y=0;
    mp[make_pair(x,y)] = 1;
    int ans = 0;
    mp[MP(x,y)]=1;
    for(int i=0; i<n; i++){
    	if(s[i]=='U') {
    		if(mp[MP(x-1,y)]==1){
    			ans++;
    		}
    		else
    		 mp[MP(x-1,y)]=1;
    		 
    		 x--;
    	}
    	else if(s[i]=='D'){
    		if(mp[MP(x+1,y)]==1)ans++;
    		else mp[MP(x+1,y)]=1;
    		
    		x++;
    	} 
    	else if(s[i]=='R'){
    		if(mp[MP(x,y+1)]==1)ans++;
    		else mp[MP(x,y+1)]=1;
    		
    		y++;
    	} 
    	else if(s[i]=='L'){
    		if(mp[MP(x,y-1)]==1)ans++;
    		else mp[MP(x,y-1)]=1;
    		
    		y--;
    	} 
    }
    cout<<ans;
    return 0;
}
