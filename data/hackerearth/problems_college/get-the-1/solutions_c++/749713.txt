#include <iostream>
using namespace std;
int main() {
	long long int t;
	cin>>t;
	for(int i=1;i<=t;++i)
	{
		long long int n,sum;
		cin>>n;
		while(n>9)
		{
			sum=0;
			int temp;
			while(n!=0)
			{
				temp=n%10;
				sum+=temp*temp;
				n/=10;
			}
			n=sum;
		}
		if(n==1) cout<<"Case #"<<i<<":YES"<<endl;
		else cout<<"Case #"<<i<<":NO"<<endl;
	}
	return 0;
}
