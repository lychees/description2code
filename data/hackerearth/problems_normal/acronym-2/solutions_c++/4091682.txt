#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,i;
    cin>>t;
    string s;
    vector<string> v1,v2;
    vector<string>::iterator it1,it2;
    while(t--)
    {
    	cin>>s;
    	v1.push_back(s);
    }
    cin>>n;
    while(n--)
    {
    	cin>>s;
    	v2.push_back(s);
    }
    for(it1=v1.begin();it1!=v1.end();it1++)
    {
    	for(it2=v2.begin();it2!=v2.end();)
    	{
    		if(*it1==*it2)
    		{
    			v2.erase(it2);
    		}
    		else
    		{
    			it2++;
    		}
    	}
    }
    for(i=0;i<v2.size()-1;i++)
    {
    	s=v2.at(i);
    	putchar(toupper(s[0]));
    	putchar('.');
    }
    if(v2.size()!=0)
    {
    	s=v2.at(i);
    	putchar(toupper(s[0]));
    }
    
    return 0;
}
