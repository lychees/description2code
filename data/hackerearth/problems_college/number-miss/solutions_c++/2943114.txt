#include <iostream>
using namespace std;

int main()
{
	int i,x;
    int hash[10001];
    for(i=0;i<10001;i++)
    hash[i]=0;
    int n,m;
    scanf("%d",&n);
    
    for(i=0;i<n;i++)
    {
        scanf("%d",&x);
    	hash[x]--;
    }
    scanf("%d",&m);
    for(i=0;i<m;i++)
    {
    	scanf("%d",&x);
    	hash[x]++;
    }
    for(i=0;i<10001;i++)
    {
    	if(hash[i]!=0)
    	printf("%d ",i);
    }
    printf("\n");
    return 0;
}
