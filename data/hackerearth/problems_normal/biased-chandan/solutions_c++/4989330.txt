#include <bits/stdc++.h>
#include <sstream>
#include <set>
#include <map>
#include <vector>
#include <functional>
#define ll long long
#define MOD 1000000007
#define PI acos(-1)
#define mx 100005

using namespace std;

int main()
{
 	ll N,x;
 	cin>>N;
 	vector<ll> v;
 	for(int i=0; i<N; i++)cin>>x,v.push_back(x);
 	
	ll sm  = 0;
	
	bool er[N];
	for(int i=0;i<N;i++)er[i] = false;
	
	for(int i=0;i<N;i++)
	{
		if(v[i] == 0)
		{
			er[i] = true;
			int j = i-1;
			while(j >= 0 and er[j] == true)j--;
			
			if(j >= 0)
			er[j] = true;
		}
	}
	 
	for(int i=0;i<N;i++)
	if(er[i] == false)
	sm += v[i];
	cout<<sm<<endl;
	
	return 0;
}
