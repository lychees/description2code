#include<bits/stdc++.h>
using namespace std;
#define S(x) scanf("%lld", &x)
#define ll long long

int main()
{
	ll t;
	S(t);
	while(t--) {
		
		ll n,k;
		S(n);
		S(k);
		ll i,j;
		ll a[n + 5];
		ll p = 0;
		for(i = 0;i < n;i++) {
			S(a[i]);
		}
		for(i = 0;i < n;i++) {
			ll mi = a[i],pos = i;
			for(j = i;j <= i + k&& j < n;j++) {
				if(mi > a[j]) {
					mi = a[j];
					pos = j;
				}
 			}
 			for(j = pos;j > i;j--) {
 				a[j] = a[j - 1];
 				k--;
 			}
 			a[i] = mi;
		}
		for(i = 0;i  < n;i++) {
			cout << a[i] << " ";
		}
		cout << "\n";
	}
	
}
