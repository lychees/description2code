#include <iostream>
using namespace std;

int main()
{
    int a[30];
    string s;
    int t;
    for(cin >> t;t--;)
    {
    	cin >> s;
    	for(int i = 0;i < 30;++i)
    		a[i] = 0;
    	for(int i = 0;i < s.length();++i)
    		a[s[i] - 'a']++;
    	bool c = true;
    	int m = a[0];
    	for(int i = 1;i < 26;++i)
    		if(a[i] != m)
    			c = false;
    	if(c)
    		cout << "Yes" << endl;
    	else
    	
    		cout << "No" << endl;
    } 
    return 0;
}
