#include <iostream>
using namespace std;


int main()
{
	long long n;
	cin >> n;
    long long result = 0;

	for (int i=1; 3*i < n; i++) {
//		cout << i << " ";
		result += 3*i;
	}
//	cout << endl;

	for (int i=1; 5*i < n;i++) {
//		cout << i << " ";
		result += 5*i;
	}
	cout << endl;

	for (int i=1; 15*i < n;i++) {
//		cout << i << " ";
		result -= 15*i;
	}
//	cout << endl;
	cout  <<  result;
    return 0;
}
