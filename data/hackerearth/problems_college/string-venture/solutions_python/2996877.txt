from collections import defaultdict
s = raw_input()
slist = list(s)
alist = [ord(c) for c in slist]
tot = sum(alist)/len(alist)
if tot%2 == 0:
    clist = slist[::-1]
    print(("".join(str(x) for x in clist)))
else:
    
    L = slist[:]
    d = defaultdict(int)
    for i in L:
       d[i] += 1
    result = max(d.iteritems(), key=lambda x: x[1])
    mlist = list(result)
    print mlist[0]
