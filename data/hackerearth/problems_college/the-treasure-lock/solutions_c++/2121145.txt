{
	long long int t,n,i;

	while(t--)
	{
		cin>>n;
		if(n%3==0&&n%5==0)
		{
			for(i=1;i<=n;i++)
			cout<<"5";
		}
		else
		if(n%3==0)
		{
			for(i=1;i<=n;i++)
			cout<<"5";
		}
		else
		if(n%5==0)
		{
			for(i=1;i<=n;i++)
			cout<<"3";
		}
		else
		if((n-5)%3==0)
		{
			for(i=1;i<=n-5;i++)
			cout<<"5";
			cout<<"33333";
		}
			else

			for(i=1;i<=n-5;i++)

	}

}
