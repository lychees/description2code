#include <iostream>
using namespace std;

int main()
{
    int t;
    long long int i,n;
    long long int *a;
    
    cin>>t;
    
    while(t--){
    	cin>>n;
    	a = (long long int *)malloc(sizeof(long long int)*n);
    	int neg[1000001] = {0};
    	int pos[1000001] = {0};
    	
    	for(i=0;i<n;i++){
    		cin>>a[i];
    	}
    	
    	for(i=0;i<n;i++){
    		if(a[i]>0){
    			pos[a[i]]++;
    		}else{
    			neg[-a[i]]++;
    		}
    	}
    	
    	long long int ans = n;
    	for(i=0;i<1000001;i++){
    		if(pos[i]>1){
    			ans+=pos[i]*(pos[i]-1)/2 ;
    		}
    		
    		if(neg[i]>1){
    			ans+=neg[i]*(neg[i]-1)/2 ;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
