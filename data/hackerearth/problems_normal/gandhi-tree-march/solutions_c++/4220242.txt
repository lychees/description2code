#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T,c;
    char s[10009];
    cin>>T;
    while(T--)
    {
    	cin>>c>>s;
    	int arr[10009];
    	char lr[10009];
    	for(int i=0;s[i]!='\0';i++)
    	{
    		arr[i] = 1000000009;
    		lr[i] = ' ';
    	}
    	
    	int n = 0;
    	arr[0] = 0;
    	for(int i=1;s[i]!='\0';i++)
    	{
    		if(s[i]=='.' || (s[i]>=97 && s[i]<=122))
    		{
    			if(s[i-1]=='(')
    				lr[i]='l';
    			else if(s[i-1]==')' || (s[i-1]>=97 && s[i-1]<=122) || s[i-1]=='.')
    				lr[i] = 'r';
    		}
    	}
    	int top = -1;
    	int sta[10009];
    	char f[10009];
    	
    	for(int i=2;s[i]!='\0';i++)
    	{
    		if(lr[i]=='l')
    		{
    			sta[++top] = --n;
    			if(s[i]!='.')
    				arr[i] = n;
    		}
    		if(lr[i]=='r')
    		{
    			n = sta[top--];
    			if(s[i]!='.')
    				arr[i] = n+2;
    			n = n+2;
    		}
    	}
    	top = 0;

    	for(int i=0;s[i]!='\0';i++)
    	{
    		if(arr[i]==c)
    			f[top++] = s[i];
    		
    	}
    	if(top==0)
    		cout<<"Common Gandhijee!";
    	else
    	{
    		sort(f,f+top);
    		for(int i=0;i<top;i++)
    			cout<<f[i];
    	}
    	cout<<endl;
    }
}
