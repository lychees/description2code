#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int main()
{
	int n,x,a[1010],b[1010];
	ll ans=0;
	cin>>n;
	cin>>a[0];
	b[0]=a[0];
	for(int i=1;i<n;i++)
	{
		cin>>a[i];
		b[i]=b[i-1]+a[i];
	}
	for(int i=1;i<=n;i++)
	{
		for(int j=0;j<a[i-1];j++)
		{
			ans+=i;
		}
		ans-=(i-1);
	}
	cout<<ans<<endl;
	return 0;
}
