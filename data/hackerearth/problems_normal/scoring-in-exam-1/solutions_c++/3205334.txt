#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,q;
    cin>>n>>q;
    vector<pair<int,int> > t(n,make_pair(0,0));
    for(int i=0;i<n;i++)
    {
    	cin>>t[i].second;
    }
    for(int i=0;i<n;i++)
    {
    	cin>>t[i].first;
    }
    sort(t.begin(),t.end(),[](pair<int,int> a,pair<int,int> b)
    {
    	return a.first>b.first;
    });
    int sum=0;
    while(q--)
    {
    	cin>>n;
    	sum=0;
    	for(int i=0;i<n;i++)
    	{
    		sum+=t[i].second;
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
