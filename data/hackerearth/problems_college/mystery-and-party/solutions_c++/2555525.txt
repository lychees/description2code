
#include<bits/stdc++.h>
using namespace std;

int main()
{
    long int t,n,k,i,l;
    
    vector<long int> v;
    scanf("%ld",&t);
    while(t--)
    {
    	scanf("%ld %ld",&n,&k);
    	for(i=0;i<n;i++){
    	scanf("%ld",&l);
    	v.push_back(l);
    	}
    
    sort(v.begin(),v.end());
    cout<<v[k-1]<<endl;
    v.clear();
    }
    
   
}
