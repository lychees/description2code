#include <bits/stdc++.h>
using namespace std;

int main()
{
    string str,sub;
    cin>>str;
    int c=0,count=0,total,i,j,k,z,p=2;
    for(i=0;i<str.length();i++)
    {
    	for(j=i+1;j<str.length();j++)
    	{
    	    sub=str.substr(i,p);
    	  
    	    
    	    for(k=0;k<sub.length()-1;k++)
    	    {
    	    	if(sub[k]==sub[k+1])
    	    	{
    	    		c++;
    	    	}
    	    	if(c==sub.length()-1)
    	    	{
    	    		count++;
    	    	}
    	    	//p++;
    	    	
    	    }
    	    p++;
    	    c=0;
    	    if(p+i>str.length()+1)
    	           break;
    	}
    	p=2;
    }
    total=count+str.length();
    cout<<total<<endl;
    return 0;
}
