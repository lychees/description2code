#include <iostream>
using namespace std;

int main()
{
    int tc;
    int qqq=1;
    cin>>tc;
    while(tc--){
    	int r,g,b;
    	int r1=0,g1=0,b1=0;
    	cin>>r>>g>>b;
    	string ans1="",ans2="",ans3="";
    	string fans1="",fans2="",fans3="";
    	char t1,t2,t3;
    	for(int i=0;i<2;i++){
    		r1=r%16;
    		b1=b%16;
    		g1=g%16;
    		if(r!=0){
    			if(r1>9){
    				r1=r1+75;
    			}else{
    				r1=r1+48;
    			}
    			t1=r1;
    			ans1=ans1+t1;
    			r=r/16;
    		}else{
    			ans1=ans1+'0';
    		}
    		if(g!=0){
    			if(g1>9){
    				g1=g1+75;
    			}else{
    				g1=g1+48;
    			}
    			t2=g1;
    			ans2=ans2+t2;
    			g=g/16;
    		}else{
    			ans2=ans2+'0';
    		}
    		if(b!=0){
    			if(b1>9){
    				b1=b1+75;
    			}else{
    				b1=b1+48;
    			}
    			t3=b1;
    			ans3=ans3+t3;
    			b=b/16;
    		}else{
    			ans3=ans3+'0';
    		}
    	}
    	for(int i=1;i>=0;i--){
    		fans1=fans1+ans1[i];
    		fans2=fans2+ans2[i];
    		fans3=fans3+ans3[i];
    	}
    	
    	cout<<"Case "<<qqq<<": "<<fans1<<" "<<fans2<<" "<<fans3<<endl;
    	qqq++;
    }
}
