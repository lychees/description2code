import re

pattern = []
text = []

T = int(raw_input( ))

for i in range(T):
	pattern.append(raw_input( ).lower())
	text.append(raw_input( ).lower())
	
for ptrn,txt in zip(pattern,text):
	p = re.compile(ptrn, re.IGNORECASE)
	yn = p.search(txt)
	if yn:
		print 'YES'
	else:
		yn = p.search(txt[::-1])
		if yn:
			print 'YES'
		else:
			print 'NO'
