#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[1000010];

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		vector<ll int> v;
		vector<ll int> r;
		ll int n,l;
		cin>>n;
		ll int i,j,ans,e=0;
		for(i=0;i<n;i++)
		cin>>a[i];
		cin>>l;
		v.push_back(a[0]);e++;
		if(l==e)
		r.push_back(a[0]);
		for(i=1;i<n;i++)
		{
			if(a[i]>v[e-1])
			{
				v.push_back(a[i]);
				e++;
				if(e==l)
				r.push_back(a[i]);
				//~ cout<<e<<" "<<l<<endl;
			}
			else
			{
				j=(ll int)(lower_bound(v.begin(),v.end(),a[i])-v.begin());
				v[j]=a[i];
				if((j+1)==l)
				r.push_back(a[i]);
				//~ cout<<(j+1)<<" "<<l<<endl;
			}
			//~ for(j=0;j<e;j++)
			//~ cout<<v[j]<<" ";
			//~ cout<<endl;
		}
		j=r.size();
		if(j==0)
		ans=-1;
		else
		{
			ans=LONG_MAX;
			for(i=0;i<j;i++)
			{
				ans=min(ans,r[i]);
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
