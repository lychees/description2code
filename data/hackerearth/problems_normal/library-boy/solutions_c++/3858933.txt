#include <iostream>
#include <algorithm>

using namespace std;

#define MAXN 201

int arr[MAXN];

int main(){
	int n;
	string a;
	cin >> n;
	for(int k=0;k<n;k++){
		cin >> a;
		char i= a[0];
		int j= i;
		arr[j]++;
	}
	//97-122
	int t=0;
	for(int k=96;k<124;k++){
		if(arr[k]>0){
			t=t+(((arr[k]-1)/10)+1);
		}
	}
	cout << t;
	return 0;
}
