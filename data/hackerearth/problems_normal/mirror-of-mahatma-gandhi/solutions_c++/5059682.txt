#include<iostream>
#include<string.h>
using namespace std;
bool ispalin(string s)
{
	int n=s.length();
	int l=0,r=n-1;
	while(l<r && s[l]==s[r])
	{
		l++;
		r--;
	}
	if(l>=r)
	return true;
	else
	return false;
}
main()
{
	int t;
	cin>>t;
	while(t--)
	{
			string s;
	cin>>s;
	int n=s.length();
	int i;
	for(i=0;i<n;i++)
	{
		if(s[i]=='2' || s[i]=='3' || s[i]=='4' || s[i]=='5' || s[i]=='6' || s[i]=='7' || s[i]=='9')
		break;
	}
	if(i==n && ispalin(s))
	cout<<"YES"<<endl;
	else
	cout<<"NO"<<endl;
	}
}
