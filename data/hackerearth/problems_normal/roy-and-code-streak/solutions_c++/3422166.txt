#include <iostream>
#include <set>
using namespace std;

int main()
{
    int R,T;
    set <long long> subset;
    long long max,len,N,S;
    cin>>T;
    for(int i=0;i<T;i++){
    	cin>>N;  len=0; max=0;
    	for(long long j=0;j<N;j++){
    		cin>>S>>R;
    		if(R==1){
    		if(!subset.count(S)){
    		  subset.insert(S); len++;
    		    if(len>max)
    			    max=len;
    		}
    		else
    		  { }
    	   }
    	   else 
    	     len=0;
    	  }
    	  cout<<max<<endl;
    	subset.clear(); }
    return 0;
}
