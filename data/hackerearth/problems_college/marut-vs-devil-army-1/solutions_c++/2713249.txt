#include<cstdio>
#include<algorithm>
using namespace std;
int main(){
	int tests;
	scanf("%d",&tests);
	while(tests--){
		int n,e,v;
		scanf("%d%d%d",&n,&e,&v);
		long long int fill=0;
		for(int i=0;i<n;++i){
			int enemy;
			scanf("%d",&enemy);
			int cnt=max(enemy-(e-1),0)/v;
			if(max(enemy-(e-1),0)%v!=0)cnt++;
			fill+=cnt;
			e=e+cnt*v-enemy;
		}
		printf("%lld\n",fill);
	}
	return 0;
}
