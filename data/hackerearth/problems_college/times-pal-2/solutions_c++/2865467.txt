#include <iostream>
using namespace std;
bool check(string s)
{
	int j =s.length()-1,i=0;
	while(i<j)
	{
		if(s[i]!=s[j])
		return false;
		i++; j--;
	}
	return true;
}
int main()
{
   int t; cin >> t;
   string a,b,s;
   while(t--)
   {
   	cin >> a >> b;
   	int count =0;
   	for(int i=0;i<=a.length();i++)
   	{
   		s = a;
   	 s = s.insert(i,b);
   	 //cout<<s<<endl;
   	 if(check(s))
   	 count++;
   	}
   	cout << count <<endl;
   }
    return 0;
}
