#include<bits/stdc++.h>
using namespace std;
int co=0;
void fun(int n,int p,int a[],int b[],int cu,set<int> se)
{
	if(cu==n+1)
	{
		int br=0;
		set<int>::iterator it1,it2;
		for(int i=0;i<p&&se.size()!=0;i++)
		{
			it1=se.find(a[i]);
			it2=se.find(b[i]);
			if(it1!=se.end()&&it2!=se.end())
			{
				br=1;
				break;
			}
		}
		if(br==0&&se.size()!=0)
		{
			co++;
		}
		return ;
	}
	fun(n,p,a,b,cu+1,se);
	se.insert(cu);
	fun(n,p,a,b,cu+1,se);
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		co=0;
		int n,p;
		scanf("%d%d",&n,&p);
		int a[p],b[p];
		for(int i=0;i<p;i++)
		{
			scanf("%d%d",&a[i],&b[i]);
		}
		set<int> se;
		fun(n,p,a,b,1,se);
		//cout<<kl<<"\n";
		printf("%d\n",co);
	}
}
