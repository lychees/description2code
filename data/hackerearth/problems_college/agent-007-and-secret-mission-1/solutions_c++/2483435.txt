#include<iostream>
#include<string>
using namespace std;
int main()
{ int T;
  string S;
  int N;
  int res,s,ns;
  cin>>T;
  while(T--)
  {  cin>>S;
     cin>>N;
     res = (N*(N+1))/2;
     int len = S.length();
     s = 0 ,ns =0;
     for(int i = 0 ; i < len; i++)
     {   if (S[i] == '1')
            { s++;
            }
         else
            { ns++;
            }
     }
     if(res%2 == 0)
     {  cout<<s<<endl;
     }
     else
     {  cout<<ns<<endl;
     }
  }
  return 0;
}
