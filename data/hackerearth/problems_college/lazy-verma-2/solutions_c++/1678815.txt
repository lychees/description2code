#include<iostream>
#include<cstdio>
#define ll long long
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
	ll n,m,i,sum,t,x;
	t=sum=0;
	scanf("%lld%lld",&n,&m);
	for(i=0;i<n;i++)
	{
		scanf("%lld",&x);
		t+=x;
	}
	for(i=0;i<m;i++)
	{
		scanf("%lld",&x);
		sum+=(x*t);
	}
	cout<<sum;
	return 0;
}
