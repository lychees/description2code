#include <bits/stdc++.h>
using namespace std;
#define maxn 100005
int a[105];
bool dp[maxn];
void subset(int sum,int n){

	if(dp[sum])
		return;
	if(sum<0)
		return;
	if(sum>=0 && n==0){
		dp[sum] = true;
		return;	
	}
	subset(sum-a[n-1],n-1);
	subset(sum,n-1);
	return;
}
int main(){

	int t,n,x,ans,total_sum,flag;
	cin >> t;
	while(t--){
		total_sum=0;
		flag=0;
		cin >> n >> x;
		memset(dp,false,sizeof(dp));
		for(int i=0;i<n;i++){
			cin >> a[i];
			total_sum += a[i];
		}
		subset(x,n);
		for(int i=0;i<=100000;i++){
			if(dp[i] && total_sum - (x-i) <= x){
                                cout << "YES" << endl;
                                flag=1;
                                break;
                        }
		}
		if(flag==0)
			cout << "NO" << endl;	
	}

	return 0;
}
