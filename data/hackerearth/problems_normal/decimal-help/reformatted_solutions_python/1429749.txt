def nth_digit(a, b, c):
    r = a
    digit = 0
    for _ in xrange(c):
        if r == 0:
            return 0
        digit = (r*10)/b
        r = (r*10) % b
    return digit


for _ in xrange(input()):
    temp = map(int, raw_input().split())
    a, b, c = temp[0], temp[1], temp[2]
    print nth_digit(a, b, c)
