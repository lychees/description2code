import sys
import heapq

ff = sys.stdin
N = int(ff.readline())
number_list = [int(x) for x in ff.readline().strip().split()]
max_3 = None
for i in xrange(N):
    if i < 2:
        print(-1)
    else:
        if i == 2:
            max_3 = number_list[0:3]
            print(max_3[0] * max_3[1] * max_3[2])
        else:
            max_3.append(number_list[i])
            x, y, z = heapq.nlargest(3, max_3)
            print(x * y * z)
            max_3 = [x, y, z]
