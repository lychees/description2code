#include <iostream>
using namespace std;
int main() {
	int t;
	cin>>t;
	while(t--)
	{
		unsigned long long n,i,a,k,ans;
		cin>>n;
		i = 2;
		a = 1;
		if(n==0)
			cout<<"0\n";
		else
		{
			while(1)
			{
				a+=i;
				i++;
				k = 1;
				ans = 0;
				while(k<=a)
				{
					if(a%k==0)
						ans++;
					k++;
				}
				if(ans>n)
				{
					cout<<a<<endl;
					break;
				}
			}
		}
	}
	return 0;
}
