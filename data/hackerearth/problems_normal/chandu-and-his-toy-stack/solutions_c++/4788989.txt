#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   	int t;
   	scanf("%d",&t);
   	while(t--){
   		int n,x,y,a[1000000],b[1000000],sum=0,i;
   		scanf("%d%d%d",&n,&x,&y);
   		for(i=0;i<n;i++)
   		{
   			scanf("%d",&a[i]);
   			scanf("%d",&b[i]);
   		}
   		sort(a,a+n);
   		sort(b,b+n);
   		for(i=0;i<n;i++){
   			if(a[i] < b[i])
				sum=sum + (b[i]-a[i])*x;
			else if(b[i] < a[i])
				sum+=(a[i] - b[i])*y;
			else
				continue;
   		}
   		printf("%d\n",sum);
   		
   	}
   
    return 0;
}
