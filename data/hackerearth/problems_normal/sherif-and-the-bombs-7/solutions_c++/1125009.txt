#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cmath>
using namespace std;
int p[1001][1001]={0};

int addpt(int x,int y)
{
	p[x][y]++;
	if(p[x][y]==2)
		return 1;
	return 0;
}

int main()
{
    int n,x,y,r,ans=0,t;
    scanf("%d",&n);
    while(n--)
    {
    	scanf("%d%d%d",&x,&y,&r);
    	for(int i=(x-r);i<(x+r+1);i++)
    	{
    		if(i>0 && i<=1000)
    		{
    			t = (int)(sqrt( (r*r) - ((i-x)*(i-x)) ));
    			for(int j=(y-t);j<(y+t+1);j++)
    			{
    				if(j>0 && j<=1000)
    				{
    					ans = ans + addpt(i,j);
    				}
    		
    			}
    		}
    	}
    }
    printf("%d\n",ans);
    return 0;
}
