#include <iostream>
using namespace std;

void min_max(int *a, int *b) {
	int temp = *a;
	if(*a>*b) {
		*a = *b;
		*b = temp;
	}
}

int main()
{
    int T;
    cin >> T;
    while(T--) {
    	int b1, b2, m1, m2;
    	cin >> b1 >> b2 >> m1 >> m2;
    	min_max(&b1, &b2);
    	min_max(&m1, &m2);

    	if(b2 < m1 || m2 < b1) {
    		/* no touch */
    		cout << "Line" << endl;
    	} else if (m1>=b1 && m1<b2 || b1>=m1 && b1<m2) {
    		/* full overlap */
    		if(m1==m2 || b1==b2)
                cout << "Point" << endl;
            else
                cout << "Nothing" << endl;
    	} else if (b2==m1 || m2==b1){
    		/* one point */
    		cout << "Point" << endl;
    	}
    }
    return 0;
}
