#include <iostream>
#include <vector>
using namespace std;
long dp[10001];
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    int test;
    cin >> test;
    while(test--){
    	int l;
    	cin >> l;
    	vector<int> price;
    	for(int i = 0;i < l;++i){
    		int lassi;
    		cin >> lassi;
    		price.push_back(lassi);
    		dp[i] = 0;
    	}
    	dp[l] = 0;
    	for(int i = 1;i <= l;++i){
    		for (int j = 1;j <= i;++j){
    			dp[i] = max(dp[i], dp[i - j] + price[j - 1]);
    		}
    	}
    	cout << dp[l] << "\n";
    }
}
