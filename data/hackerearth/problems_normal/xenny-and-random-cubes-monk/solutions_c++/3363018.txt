#include <iostream>
#include <string.h>
#include <math.h>
#define ll long long int

#define  MOD 1000000007

using namespace std;

ll dp[12][1<<11];
char cub[12][6],words[12];
int n,k;

ll solve(int pos,int used){
	if(pos==k)
		return 1;
	if(dp[pos][used]!=-1)
		return dp[pos][used];
	int ans=0;
	for(int i=0;i<n;i++){
		if(used&(1<<i))
			continue;
		int count=0;
		for(int j=0;j<6;j++){
			if(cub[i][j]==words[pos])
				count++;
		}
		ans+=(count*solve(pos+1,used|(1<<i)))%MOD;
	}
	dp[pos][used]=ans;
	return ans;
}

int main()
{
//    cout << "Hello World!a" << endl;
	
	ll ans;
	cin >> n >> k;
	char a;
	for(int i=0;i<n;i++){
		for(int j=0;j<6;j++){
			cin >> a;
			cub[i][j] = a;
		}
	}
	for(int i=0;i<k;i++)
		cin >> words[i];
	memset(dp,-1,sizeof(dp));
	ans=solve(0,0);
	cout << ans << endl;
    return 0;
}
