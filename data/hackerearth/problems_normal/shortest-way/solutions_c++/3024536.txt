#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    long long int t,a,b,result,temp,ma,mi;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	mi=min(a,b);
    	ma=max(a,b);
    	result=0;
    	while(mi!=0)
    	{
    		result+=ma/mi;
    		ma=ma%mi;
    		if(ma<mi)
    		{
    			temp=mi;
    			mi=ma;
    			ma=temp;
    		}
    		
    	}
    	cout<<result-1<<"\n";
    	
    	
    }
    return 0;
}
