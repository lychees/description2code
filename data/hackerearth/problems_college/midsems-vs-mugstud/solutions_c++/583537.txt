#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		ll n,i,j,k;
		cin>>n;
		vector< vector< vector<ll> > > v;
		v.resize(n);
		for(i=0;i<n;i++)
		{
			v[i].resize(n);
			for(j=0;j<n;j++)
			{
				v[i][j].resize(n);
				for(k=0;k<n;k++)
				{
					cin>>v[i][j][k];
				}
			}
		}
		for(i=n-1;i>0;i--)
		{
			for(j=n-1;j>=0;j--)
			{
				for(k=n-1;k>=0;k--)
				{
					v[i][j][k]=v[i][j][k]-v[i-1][j][k];
				}
			}
		}
		for(i=n-1;i>=0;i--)
		{
			for(j=n-1;j>0;j--)
			{
				for(k=n-1;k>=0;k--)
				{
					v[i][j][k]=v[i][j][k]-v[i][j-1][k];
				}
			}
		}
		for(i=n-1;i>=0;i--)
		{
			for(j=n-1;j>=0;j--)
			{
				for(k=n-1;k>0;k--)
				{
					v[i][j][k]=v[i][j][k]-v[i][j][k-1];
				}
			}
		}
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				for(k=0;k<n;k++)
				{
					cout<<v[i][j][k]<<" ";
				}
				cout<<endl;
			}
		}
	}
	return 0;
}
