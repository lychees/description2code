#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int t,n,b,c,d,coun;
    cin>>t;
    while(t--)
    {
        coun=0;
        cin>>n;
        if(n==1)
            coun=1;
        while(n!=1)
        {


        coun++;
        if(n%3==0)
            n/=3;
            else if(n%2==0)
                n/=2;
                else
                    n--;
        }
        cout<<coun<<endl;
    }



    return 0;
}
