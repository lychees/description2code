#include <iostream>
using namespace std;
void fact (int n){
	int a[250]={0};
	if(n==0 || n==1)
		cout<<"1"<<endl;
	else{
		int cnt =1 ;
		int mul,cry;
		a[0]=1;
		
		for(int i=2;i<=n;i++){
			cry=0;
			for(int j=0;j<cnt;j++){
				mul = a[j]*i+cry;
				cry=mul/10;
				a[j]= mul%10;
				if(cry>0 && (cnt-1<=j))
				{
					cnt++;
				}
			}
			
		}
		for(int k=cnt-1;k>=0;k--)
		{
			cout<<a[k];
		}
		cout<<endl;
	}
}
int main()
{
	int T;
	long long n;
	cin>>T;
	for(int i=0;i<T;i++){
		cin>>n;
		fact(n);
	}
    return 0;
}
