#include<bits/stdc++.h>
using namespace std;
#define lli long long int
int main()
{
    string str;
    lli T,counter;
    cin>>T;
    while(T--)
    {
        counter=0;
        cin>>str;
        lli n = str.length();
        int A[n];
        memset(A,0,sizeof(A));
        for(int i=0;i<n;i++)
        {
            if(str[i]=='B')
            {
                A[i]=1;
                A[i-1]=1;
                A[i-2]=1;
            }
            else if(str[i]=='D')
                A[i]=1;
            else if(str[i]=='S')
            {
                A[i]=1;
                A[i-1]=1;
                A[i+1]=1;
            }

        }
        for(lli i=0;i<n;i++)
        {
           // cout<<A[i]<<" ";
            if(A[i]==0)
                counter++;
        }
        cout<<counter<<endl;

    }
    return 0;
}
