#include <bits/stdc++.h>
using namespace std;

int main()
{
	long t, a, b, c, l, r, mid, ans, val;
	long k;
	cin>>t;
	while(t--){
		cin>>a>>b>>c>>k;
		l = 0;
		r = 100000;
		ans=0;
		while(l<=r){
			mid = (l + r) / 2;
			val = (a*mid*mid) + (b*mid) + c;
			if(val>=k){
				ans = mid;
				r = mid-1;
			}
			else l = mid+1;
		}
		cout<<ans<<endl;
	}

    return 0;
}
