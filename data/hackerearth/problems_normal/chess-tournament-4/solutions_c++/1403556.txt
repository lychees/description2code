#include <cstdio>

int A[1025], wins[1025][1025];

int main() {
	int N, i, j, score, candidates;
	
	scanf("%d", &N);
	candidates = 1 << N;
	for (i = 0; i < candidates; ++i)
		A[i] = i+1;
	for (i = 2; i <= candidates; ++i)
		for (j = 1; j < i; ++j) {
			scanf("%d", &score);
			wins[i][j] = score; wins[j][i] = score ? 0 : 1;
		}
	for (j = candidates; j > 1; j >>= 1)
		for (i = 0; i < j; i += 2)
			A[i>>1] = wins[A[i]][A[i+1]] ? A[i] : A[i+1];
	printf("%d\n", A[0]);		
	
	return 0;
}
