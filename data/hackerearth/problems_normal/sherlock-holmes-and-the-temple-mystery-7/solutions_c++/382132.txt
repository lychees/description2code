#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

int isSubsetSum(int arr[],int n,int sum)
{
	int dp[sum+1][n+1];
	
	for(int i=0;i<=n;i++)
		dp[0][i]=1;
	for(int i=1;i<=sum;i++)
		dp[i][0]=0;
	for(int i=1;i<=sum;i++)
	{
		for(int j=1;j<=n;j++)
		{
			dp[i][j]=dp[i][j-1];
			if(i >= arr[j-1])
				dp[i][j] = dp[i][j] || dp[i-dp[j-1][j-1]];
		}
	}
	return dp[sum][n];
}

int main()
{
    int n,t,cost;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int arr[n];
    	cost=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>arr[i];
    		cost+=arr[i];
    	}
    	if(cost&1)
    		cout<<"NO\n";
    	else
    	{
    		if(isSubsetSum(arr,n,cost/2))
    			cout<<"YES\n";
    		else
    			cout<<"NO\n";
    	}
    }
    return 0;
}
