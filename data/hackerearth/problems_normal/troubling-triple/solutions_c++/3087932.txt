#include <bits/stdc++.h>
using namespace std;

int main()
{
    for( int n, k; cin>>n>>k; ) {
    	vector<int>nums(n);
    	for(int i=0; i <n ; i++)
    		cin>>nums[i];
    	sort( nums.begin(), nums.end() );
    	long long val, res = 0;
    	for( int i=0 ; i < n ; i++) {
    		for( int j=i+1; j < n ; j++) {
    			val = nums[i]*nums[j];
    			if( val>k )
    				continue;
    			//cout<<(upper_bound( nums.begin(), nums.end(), k/val)-nums.begin()-j-1)<<"\n";
    			int aux = (upper_bound( nums.begin(), nums.end(), k/val)-nums.begin()-j-1 );
    			res += max(0, aux);
    		}
    	}
    	cout<<res<<"\n";
    }
    return 0;
}
