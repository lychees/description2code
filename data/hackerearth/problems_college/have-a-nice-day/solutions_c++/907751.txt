#include <iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;

int main()
{
	int t,a[10000],b[10000],i,sum,n;
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	sum=0;
   	for(i=0;i<n;i++)
   	{
   		cin>>a[i];
   	}
  for(i=0;i<n;i++)
  cin>>b[i];
   	sort(a,a+n);
   	sort(b,b+n);
   	for(i=0;i<n;i++)
   	sum=sum+a[i]*b[i];
   	cout<<sum<<endl;
   }
    return 0;
}
