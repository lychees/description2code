#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
     	cin>>str;
    	int n=str.length();
    	int j=1,count=0;
    	
    	for(int i=0;i<n;i++)
    	{
    		if(str[i]=='A' || str[i]=='a' || str[i]=='b' || str[i]=='B' || str[i]=='C' || str[i]=='c' )
    		{
    			count+=j;
    			j++;
    		}else
    		{
    			j=1;
    		}
    		
    	}
    	cout<<count<<endl;
    }
    return 0;
}
