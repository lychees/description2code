#include <iostream>
#define mod 1000000007
#define ll long long
using namespace std;
ll modPow(ll a,ll b,ll MOD)
{
	ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
ll modInverse(ll a){
    return modPow(a,mod-2,mod);
    }

int main()
{
	long long int t;
	cin>>t;
	while(t--)
	{
		long long int k,m,i,sum=0,x=0,y=0,p=1,q=1,ans=0,left=0;
		cin>>k>>m;
		long long int a[k];
		for(i=0;i<k;i++)
		{
			cin>>a[i];
			sum+=a[i];
		}

		left=m-sum;

		if(sum> m)
        cout<<0<<endl;
		else
		{
			x=k+left-1;
			y=k-1;
			for(i=x;i>y;i--)
			{
				p=((p%mod)*(i%mod))%mod;
			}

			for(i=1;i<=left;i++)
			{
				q=((q%mod)*(i%mod))%mod;
			}

			ans=(p*modInverse(q))%mod;
			cout<<ans<<endl;
		}
	}
    return 0;
}
