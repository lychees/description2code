#include <bits/stdc++.h>
using namespace std;

int main()
{
	int i,j,k,t,mini,sum[100],n;
	int arr[100];
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		for(i=0;i<n;i++)
		{
			cin>>arr[i];
		}
		sort(arr,arr+n);
		mini=1000000000;
		for(i=1;i<=100;i++)
		{
			
			for(j=0;j<n;j++)
			{
				if(arr[j]>i)
				{
					sum[j]=(arr[j]-i)*5;
				}
				else
				{
					sum[j]=(i-arr[j])*3;
				}
			}
			for(j=1;j<n;j++)
			{
				sum[j]+=sum[j-1];
			}
			mini=min(mini,sum[k-1]);
			for(j=0;j<n-k;j++)
			{
				mini=min(sum[j+k]-sum[j],mini);
			}
		}
		cout<<mini<<"\n";
	}
	return 0;
}
