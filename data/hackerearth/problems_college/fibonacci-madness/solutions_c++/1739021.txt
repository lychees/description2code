#include <iostream>

using namespace std;

long long int fib[100];

void pfib(int n)
{
	int i;
	
	fib[0]=0;
	
	fib[1]=1;
	
	for(i=2;i<n;i++) { fib[i]=fib[i-1]+fib[i-2]; }
	
	for(i=n-1;i>=0;i--) { cout<<fib[i]<<" "; }
	
	cout<<endl;
}	

int main()
{
    int t,n;
    
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	
    	pfib(n);
    }
    
    return 0;
}
