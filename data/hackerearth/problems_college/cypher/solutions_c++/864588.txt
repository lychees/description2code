#include <iostream>
using namespace std;

int main()
{
	int n;
	cin >>n;
	char l[26] = {'a','b','c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','v','w','x','y','z'};
	while (n > 0){
		string s;
		cin >> s;
		string t2 = "";
		int tt = 1;
		for (int i = 1; i < s.length(); i++){
			int t1 = s[i]-97;
			t1 = t1 + tt;
			t1 = t1 % 26;
			t2 = t2 + l[t1];
			tt++;
		}
		t2 = s[0] + t2;
		cout << t2 << endl;
		n -= 1;
	}
    return 0;
}
