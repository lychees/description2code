long long multiply(long long x,long long a[],long long size)
{
	long long carry=0;
	for (int i=0; i<size; i++)
    {
        int prod = a[i] * x + carry;
        a[i] = prod % 10;
        carry  = prod/10;
    }

    while (carry)
    {
        a[size] = carry%10;
        carry = carry/10;
        size++;
    }
    return size;
}

long long fact(long long n,long long a[])
{
	long long x;
	long long size=1;
	a[0]=1;
	for(x=2;x<=n;x++)
	size=multiply(x,a,size);
	return size;
}
