using namespace std;
long long dp[5007][5007];
int main()
{
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);
	dp[0][0] = 1;
	string a, b;
	cin >> a >> b;
	int n = a.length();
	int m = b.length();
	for (int i = 0; i <= n; i++)
	{
		for (int j = 0; j <= m; j++)
		{
			if (j > 0)
			{
				dp[i][j] += dp[i][j - 1];
			}			
			if (i>0 && j>0 && a[i - 1] == b[j - 1])
			{
				dp[i][j] += dp[i - 1][j - 1];
			}
		}
	}
	cout << dp[n][m] << endl;
}
