#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long t,n,m,i,j;
	long long e,o;
	cin>>t;
	while(t--)
	{
		scanf("%lld%lld",&n,&m);
		if(n<=m)
		{
				o=(n+1)/2;
				o*=o;
				e=(n/2);
				e*=(e+1);
		}
		else
		{
			o=(m+1)/2;
			o*=o;
			e=(m/2);
			e*=(e+1);
			if(m%2==0)
			{
				o=o+m*((n-m+1)/2);
				e=e+m*((n-m)/2);
			}
			else
			{
				e=e+m*((n-m+1)/2);
				o=o+m*((n-m)/2);
			}
		}
		printf("%lld %lld\n",e,o);
	}
    return 0;
}
