#include <iostream>
using namespace std;

int main()
{
    int i,x[30],n;
    cin>>n;
    x[0]=1; x[1]=1;
   
    for (i=2;i<80;i++)
    {
        x[i]=x[i-1]+x[i-2];
        if (x[i]>n)
           break;
    }
    for (i=0;i<=80;i++)
    {
        if(x[i]<=n)
            cout<<x[i]<<endl;
        else 
            break;
    }
    return 0;
}
