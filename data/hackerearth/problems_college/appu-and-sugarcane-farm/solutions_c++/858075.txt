#include <iostream>
#include<math.h>
using namespace std;
long long int poer(int i)
{
	long long int d=1;
	while(i--)
	{
		d=d*2;
	}
	return d;
}
int check(int g)
{
		int i=0;
		while(g>poer(i)&&g%poer(i)==0)
		{
			i++;
		}
		if(g==poer(i))
		return 0;
		else return 1;
		
}
int main()
{
    int n;
    scanf("%d",&n);
    long long int a[n];
    int i;
    long long int min=1000000000;
    for(i=0;i<n;i++)
    {
    	scanf("%lld",&a[i]);
    	if(a[i]<min)
    	min=a[i];
    }
    for(i=0;i<n;i++)
    if(a[i]%min==0)
    a[i]=a[i]/min;
    else
    break;
    if(i!=n)
    printf("NO\n");
    else
    {
    	for(i=0;i<n;i++)
    	if(check(a[i]))
    	break;
    	if(i!=n)
    	printf("NO");
    	else
    	printf("YES");
    }
    return 0;
}
