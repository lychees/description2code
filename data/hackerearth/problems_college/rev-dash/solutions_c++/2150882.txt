#include<bits/stdc++.h>
using namespace std;
int main()
{
    char s[10000];
    cin>>s;
    int x=strlen(s);
    int i;
    for(i=0;i<x;i++)
    {
        cout<<s[i];

        if(i!=x-1)
            cout<<"-";
    }
    cout<<endl;
    for(i=x-1;i>=0;i--)
    {
        cout<<s[i];
        if(i!=0)
            cout<<"-";
    }
    return 0;

}
