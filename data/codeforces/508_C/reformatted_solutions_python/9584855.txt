m, t, r = map(int, raw_input().split())
d = sorted(map(int, raw_input().split()))
c = []
rt = 0
if t < r:
    rt = -1
    d = []
for i in d:
    c1 = []
    for j in c:
        if i-j < t:
            c1 += [j]
    c = c1
    if r > len(c):
        rc = r-len(c)
    else:
        rc = 0
    rt += rc
    for j in xrange(rc):
        c += [i-(rc-1-j)]
print rt
