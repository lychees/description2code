#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n ,res = 0, sum = 0, a;

    cin >> n;

    for(int i = 0; i < n; i++){

        cin >> a;

        if(sum <= 0){               

            if(a < 0) 
            {
            	continue;     
            }
            sum = a;                

        } 
        else 
        {
        sum += a;   
        }

        res = max(res , sum);          
    }

    cout << res << "\n";

    return 0;
}
