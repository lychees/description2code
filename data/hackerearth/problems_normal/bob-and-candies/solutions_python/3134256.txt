def select(a,start,end,x):
	if start>end:
		return end+1,[]
	y=x-a[start-1]
	if y==0:
		return 1,[start]
	if y<0:
		return select(a,start+1,end,x)
	counts,ss=select(a,start+1,end,y)
	countns,ns=select(a,start+1,end,x)
	if counts+1<=countns:
		return counts+1,[start]+ss
	else:
		return countns,ns

n,x=raw_input().split()
n,x=int(n),int(x)
ax=raw_input().split()
a=[int(i) for i in ax]
count,s=select(a,1,n,x)
if count>n:
	print -1
else:
	ss=[str(i) for i in s]
	print ' '.join(ss)
