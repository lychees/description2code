#include <iostream>
using namespace std;
typedef long long int lli;
class Cards{
	lli number_of_cards;
	lli min_taps;
	public:
		void setNumberOfCards(lli n){
			number_of_cards = n;
		}
		lli getMinTaps(){
			return min_taps;
		}
		void  calculateTaps();
};

void Cards::calculateTaps(){
	min_taps = number_of_cards /(3ll);
	if(number_of_cards == 0)
		min_taps = 0;
	else if(number_of_cards <= 3)
		min_taps = 1;
	else if(number_of_cards%3ll != 0ll)
		min_taps = number_of_cards;
	
}
int main()
{
    lli t,n;
    cin>>t;
    Cards card;
    while(t--){
    	cin>>n;
    	card.setNumberOfCards(n);
    	card.calculateTaps();
    	cout<<card.getMinTaps()<<endl;
    	 	
    }
    return 0;
}
