#include<iostream>
 
using namespace std;
 
 
long count[101] = {0};
 
int main(){
    int t;
    cin >> t;
    while(t--){
        int n;
        long long c = 0;
        cin >> n;
        for(int i=0;i<n;i++){
            int a;
            cin >> a;
            count[a]++;
        }
        int k;
        cin >> k;
        for(int i=0;i<=100;i++){
            if(k-i <= 100 && k-i >= 0){
                if(count[i]){
                    if(i != k-i)
                        c += count[i]*count[k-i];
                    else
                        c += count[i]*(count[i]-1)/2;
                    count[i] = 0;
                    count[k-i] = 0;
                }
            }
            count[i] = 0;
        }
        cout << c << endl;
    }
    return 0;
}
