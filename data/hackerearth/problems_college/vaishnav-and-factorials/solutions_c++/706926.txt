int factorial(int n,int len) 
{ 
int i,j; 
for(i=2;i<=n;i++){ 
j=multiply(i,j); 
} 
return j;
} 

int multiply(int n,int len) 
{ 
long i,r=0; 
int arr[MAX]; 
for(i=0;i<=len;i++) 
{ 
arr[i]=fact[i]; 
} 

for(i=0;i<=len;i++) 
{ 
fact[i] = (arr[i]*n + r)%10; 
r = (arr[i]*n + r)/10; 
} 
if(r!=0) 
while(r!=0) 
{ 
fact[i]=r%10; 
r= r/10;

} 
len = i-1;

}
