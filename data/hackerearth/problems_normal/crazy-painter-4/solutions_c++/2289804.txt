#include<bits/stdc++.h>
using namespace std;


int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long n;
        int c[30];
        int i,j;
        cin>>n;
        for(i=0;i<26;i++)
            cin>>c[i];
        long long sum=0;
        for(i=0;i<26;i++)
            sum+=c[i];
        long long x = n*12;
        //cout<<"x = "<<x<<endl;
        long long y = x/26;
        //cout<<"y = "<<y<<endl;
        long long cost=0;
        cost = (y*sum) + 13*(y-1)*y;
        int p = x%26;
        //cout<<"p = "<<p<<endl;
        for(i=0;i<p;i++)
            cost+=c[i];
        cost += (y*p);
        cout<<cost<<endl;
    }
    return 0;
}
