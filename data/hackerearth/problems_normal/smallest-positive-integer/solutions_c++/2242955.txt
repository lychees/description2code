#include <iostream>
#include <vector>
#include <string>
#include <stdlib.h>
#include <algorithm>
#include <math.h>
#define mod 1000000007
using namespace std;


int main()
{
    unsigned long long int n ;
    cin >> n ;
    if( n==1 )
    	cout << 1 ;
    else if( n == 0 )
    	cout << 10 ;
    else
    {
    	for(unsigned long long int i = 0 ; i< mod ;++i )
    	{
    		unsigned long long int temp = mod*i+n ;
    		vector<int> v(10,0);
    		for(int j=9;j>1;--j)
    		{
    			while(temp%j == 0)
    			{
    				temp = temp/j ;
    				v[j] +=1 ;
    			}
    		}
    		if(temp==1)
    		{
    			for(int k=2;k<=9;++k)
    			{
    				for(int l=1;l<=v[k];++l)
    					cout << k ;
    			}
    			break ;
    		}
    	}
    	
    }
    return 0;
}
