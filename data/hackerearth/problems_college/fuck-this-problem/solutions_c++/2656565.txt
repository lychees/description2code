#include <cstdlib>
#include <iostream>
#include <stdio.h>
#include <string.h>

using namespace std;
char s[10001];
int N,n,i,k;
int main(int argc, char *argv[])
{   gets(s);
    scanf("%d",&N);
    n=strlen(s);
    k=0;
    for(i=0;i<n;i++)
    {  printf("%c",s[i]);
       k++;
       if(k==N)
       {   k=0;
           printf("\n");
       }
       
    }  
    if(n%N!=0)
       printf("\n");
    
    return EXIT_SUCCESS;
}
