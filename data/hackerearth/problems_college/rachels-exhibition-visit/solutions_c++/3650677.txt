#include <iostream>
#include <map>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k,l,r;
    	string s,sp;
    	map <char,int> msp;
    	cin>>n>>k>>l>>r;
    	cin>>s>>sp;
    	for(int i=0;i<sp.size();++i)
    	{
    		msp[sp[i]]=1;
    	}
    	int count=0,cnt;
    	for(int i=0;i<s.size();++i)
    	{
   		for(int j=i;j<s.size();++j)
    	{
    		cnt=0;
    		for(int m=i;m<=j;++m)
    		{
    			if(msp[s[m]]==1){++cnt;}
    		}
    		if(cnt>=l&&cnt<=r){++count;}
    	}
    	
    	}
    	cout<<count<<endl;
    }
}
