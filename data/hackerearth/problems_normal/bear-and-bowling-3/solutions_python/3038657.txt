mod=1000000007
ans=0
num=[]
n=int(raw_input())
if (n==1):
    num=int(raw_input())
    print num%mod
else:
    s=pow(2,n-1,mod)
    k=pow(2,n-2,mod)
    num=map(int,raw_input().split())
    for i in (num):
        x=(i*s)%mod
        ans+=x
        s+=k
        ans%=mod
    print ans
