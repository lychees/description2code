#include <bits/stdc++.h>
using namespace std;
long long int a[1000009];
#define LL long long
int main()
{
   int t,n,i,j;
   LL m,sum=0;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	sort(a,a+n);
    	sum=0;
    	for(i=n-1;i>=0;i--)
    	{
    		sum+=a[i];
    		if(sum>=m)
    		break;
    	}
    	if(i==-1)
    	cout<<i<<endl;
    	else
    	cout<<n-i<<endl;
    }
    return 0;
}
