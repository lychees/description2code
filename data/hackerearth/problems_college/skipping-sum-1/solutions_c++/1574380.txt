#include <bits/stdc++.h>
using namespace std;

int n , q , l , r , k , a[100001] ;
long long sum[100001][11] ;

int main()  {
  
  scanf("%d%d",&n,&q) ;
  for(int i=0 ; i<n ; i++) scanf("%d",&a[i]) ; 
  
  for(int i=n-1 ; i>=0 ; i--) for(int k=1 ; k<=10 ; k++) {
  	 long long tmp=0 ;
  	 if(i+k<n) tmp=sum[i+k][k] ; 
  	 sum[i][k] = tmp+a[i] ;  
  }
  
  while(q--) {	 
  	 scanf("%d%d%d",&l,&r,&k) ;
  	 l-- ; 
  	 while(r<n && (r-l)%k) r++ ; 
  	 
  	 long long ans=sum[l][k] ;
  	 if(r<n) ans-=sum[r][k] ;
  	 
  	 printf("%lld\n",ans) ;
  }
 	
  return 0;
}
