#include <bits/stdc++.h>
using namespace std;
 
char str[1001],cj[1001];
int M[1001][1001]={0};
 
int main(){
	scanf("%s",cj);scanf("%s",str);
    int t,i,j,k,l,lcj=strlen(cj);
    for (j=1;j<=1000;j++)
        M[0][j]=1;
    for (j=1;j<=lcj;j++)
        M[j][0]=0;   
    l=strlen(str);
    for (i=1;i<=lcj;i++){
        M[i][0]=0;
        for (j=1;j<=l;j++){
            if (cj[i-1]==str[j-1])
            M[i][j]=(M[i-1][j]+M[i][j-1])%10000;
            else M[i][j]= M[i][j-1];
        }        
    }
    printf("%d",M[lcj][l]);
    return 0;
}
