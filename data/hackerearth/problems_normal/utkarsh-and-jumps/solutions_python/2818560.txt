n,p = [int(i) for i in raw_input().split()]
lis = [1,0,p/100.0,1-p/100.0]
for i in range(4,n+1):
	val = lis[i-2]*p/100.0 + lis[i-3]*(1-p/100.0)
	lis.append(val)

print "%.6f" %lis[-1]
