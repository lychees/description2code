t = int(input())
while t > 0:
    m = int(input())
    res = 0
    while m > 0:
        l = raw_input()
        l, d = l.split()
        l = int(l)
        d = int(d)
        res = res+l*d
        m = m-1
    while res >= 10:
        num = res
        res = 0
        while num > 0:
            res = res+num % 10
            num = num/10
    print(str(res))
    t = t-1
