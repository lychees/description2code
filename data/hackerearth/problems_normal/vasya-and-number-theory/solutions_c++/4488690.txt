ll dp[MAX], a[MAX];
int  n;
ll solve(int pos)
{
    if(pos==n)
        return 1;
    if (dp[pos] != -1)
        return dp[pos];
    int res  = 1;
    int tmp = 0;
    for (int x = pos + 1; x <= n; x++)
    {
        if (a[x] % a[pos] == 0|| a[pos]==0)
            tmp = solve(x) + 1;
        if (tmp > res)
            res = tmp;
    }
    dp[pos] = res;
    return dp[pos];
}
