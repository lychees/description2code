#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

vector <int> p,n;
int main()
{
   	long long int pp, pn,maximum,minimum,zcount,x;
   	int t,num,e; 
   	cin>>t;
   	while(t--)
   	{
   		p.clear(); n.clear();
   		pp=1; pn=1; zcount=0;
   		
   		cin>>e;
   		int ce=e;
   		while(e--)
   		{
   			cin>>num;
   			if(num<0)
   			{n.push_back(num);  pn*=num;}
   			else if(num>0)
   			{p.push_back(num); pp*=num;}
   			else 
   			zcount++;
   		}
   		
   		sort(p.begin(),p.end()); sort(n.begin(),n.end());
   		if(zcount==ce)
   		{ minimum=0; maximum=0;}
   		
   		else if(n.size()==0)
   		{
   			if(zcount>0)
   			{
   				minimum=0;
   			}
   			else
   			minimum=p[0];
   			maximum=pp;
   		}
   		
   		else if(p.size()==0)
   		{
   			if(pn<0)
   			{
   				minimum=pn;
   				if(n.size()==1 && zcount>0)
   				maximum=0;
   				else if(n.size()==1)
   				maximum=n[0];
   				else
   				maximum=pn/(n[n.size()-1]);
   			}
   		
   			else
   			{
   				maximum=pn;
   				minimum=pn/(n[n.size()-1]);
   			}
   		}
   		
   		else {
   			x=pp*pn;
   			if(x<0)
   			{
   				minimum=x;
   				maximum=x/n[n.size()-1];
   			}
   			else
   			{maximum=x;
   			minimum=x/n[n.size()-1];}
   			}
   		
   		cout<<maximum<<" "<<minimum<<"\n";
   		
   	}
    return 0;
}
