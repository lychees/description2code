#include <iostream>
using namespace std;

#include<string.h>

int main()
{
    int t,i,j,len,v1,v2,ans;
    char s[100];
    cin>>t;
    gets(s);
    for(i=0;i<t;i++)
    {
    	gets(s);
    //	puts(s);
    	len=strlen(s);
    	for(j=0;j<len/2;j++)
    	{
    		ans=0;
    		v1=s[j+1]-s[j];
    		v2=s[len-1-j]-s[len-2-j];
    	//	cout<<v1<<endl<<v2<<endl;
    		if(v1<0)
    		v1=-1*v1;
    		if(v2<0)
    		v2=-1*v2;
    		//cout<<v1<<endl<<v2;
    		if(v1!=v2)
    		{
    			cout<<"Not Funny\n";
    			ans=10;
    			break;
    		}
    		
    	}
    	if(ans!=10)
    	cout<<"Funny\n";
    }
    return 0;
}
