#define IOS 		ios_base::sync_with_stdio(false); cin.tie(NULL)
#define IN          int t; scanf("%d",&t); while(t--)

int main()
{
    
    IN
    {
    	string s;
    	unordered_map <int,int> m;
    	m['r'] = m['u'] = m['b'] = m['y'] = 0;
    	cin >> s;
    	int l = s.length();
    	for(int i=0;i<l;++i)
    	m[s[i]]++;
    	int ans = min(m['y'],min(m['b'],min(m['r'],m['u'])));
    	cout << ans << '\n';
    }
    
    return 0;    
}
