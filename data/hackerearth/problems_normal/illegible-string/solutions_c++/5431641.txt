#include <iostream>
using namespace std;
char s[1000002];

int main()
{
int n;
cin>>n;
cin>>s;
int flag=0;
int minl=0,maxl=0;
for(int i=0;i<n;i++)
{
	if(s[i]=='v'){
		if(flag==1){
			minl++;
			flag=0;
		}
		else{
			flag=1;
		}
	}
	else if(s[i]=='w'){
		maxl++;
		if(flag==1){
			if(s[i+1]=='v'){
				minl++;
				i++;
				flag=0;
			}
		}
	}
	else{
		flag=0;
	}
}
minl=n-minl;
maxl+=n;
cout<<minl<<" "<<maxl;
}
