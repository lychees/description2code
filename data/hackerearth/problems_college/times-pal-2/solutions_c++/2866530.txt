#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
char a[10000],b[10000];
int check(char *p)
{
    char q[10000];
    strcpy(q,p);
    for(int i=0,j=strlen(p)-1;i<strlen(p);i++,j--)
        q[i]=p[j];
    q[strlen(p)]='\0';
    //cout<<p<<endl<<q<<endl;
    if(!strcmp(p,q))
        return 1;
    else
        return 0;
}
void add(int j,char *p)
{
    int i,k,l=strlen(b);
    for(i=j,k=0;k<strlen(p);k++,i++)
    {
        int q=strlen(b);
        while(q>=i)
        {
            b[q]=b[q-1];
            q--;
        }
        b[i]=p[k];
    }
    b[l+strlen(p)]='\0';
    //cout<<b<<endl;
}
int main()
{
    int n;
    cin>>n;
    while(n--)
    {
        char c[10000];
        int f=0;
        cin>>a>>c;
        for(int i=0;i<=strlen(a);i++)
        {
            strcpy(b,a);
            add(i,c);
            if(check(b))
                f++;
        }
        cout<<f<<endl;
    }
}
