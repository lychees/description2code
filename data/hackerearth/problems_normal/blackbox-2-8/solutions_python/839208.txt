n = int(raw_input())
def factorial(n):
	if n == 0 or n == 1:
		return 1
	else:
		return n*factorial(n-1)

for _ in range(n):
	m = int(raw_input())
	l = factorial(m)
	k = str(l)
	s = 0
	for j in range(len(k)):
		s += int(k[j])
	print s
