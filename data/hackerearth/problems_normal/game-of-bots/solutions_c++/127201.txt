char s1[1000005],s2[1000005];
int a1[1000005],a2[1000005];
int p1[1000005];
int p2[1000005];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s %s",s1,s2);
        int l = strlen(s1);
        int k1=0,k2=0;
        for(int i=0;i<l;i++)
        {
            if(s1[i] == 'A')
            {
                p1[k1] = i;
                a1[k1++] = 1;
            }
            else if(s1[i] == 'B')
            {
                p1[k1] = i;
                a1[k1++] = 2;
            }
            if(s2[i] == 'A')
            {
                p2[k2] = i;
                a2[k2++] = 1;
            }
            else if(s2[i] == 'B')
            {
                p2[k2] = i;
                a2[k2++] = 2;
            }
        }
        if(k1 != k2)
        {
            printf("No\n");
            continue;
        }
        int flag = 1;
        FOR(i,k1)
        {
            if(a1[i] != a2[i])
            {
                flag = 0;break;
            }
        }
        if(!flag)
        {
            printf("No\n");
            continue;
        }
        FOR(i,k1)
        {
            if(a1[i] == 1)
            {
                if(p2[i] > p1[i])
                {
                    flag = 0;
                    break;
                }
            }
            else
            {
                if(p2[i] < p1[i])
                {
                    flag = 0;
                    break;
                }
            }
        }
        if(!flag)
            printf("No\n");
        else
            printf("Yes\n");
    }
return 0;
}
