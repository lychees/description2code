#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <math.h>
#include<stdio.h>
#define  ll long long int
#define mod 1000000009
using namespace std;



void Multiply(ll F[2][2] , ll M[2][2]){
 ll x =  (F[0][0]*M[0][0] + F[0][1]*M[1][0])%mod;
 ll y =  (F[0][0]*M[0][1] + F[0][1]*M[1][1])%mod;
 ll z =  (F[1][0]*M[0][0] + F[1][1]*M[1][0])%mod;
 ll w =  (F[1][0]*M[0][1] + F[1][1]*M[1][1])%mod;

  F[0][0] = x%mod;
  F[0][1] = y%mod;
  F[1][0] = z%mod;
  F[1][1] = w%mod;


}

void power( ll F[2][2] ,ll n){

  if( n == 0 || n == 1)
      return;
  ll M[2][2] = {{1,1},{1,0}};

  power(F, n/2);
  Multiply(F, F);

  if (n%2 != 0)
     Multiply(F, M);
}
ll fib(ll n){

 ll F[2][2] = {{1,1},{1,0}};
if(n==0)
    return 0;
power(F,n-1);
return F[0][0];

}
int main()
{
    int test;
    ll n;
    scanf("%d",&test);
    while(test--){
    scanf("%lld",&n);
printf("%lld\n",(fib(n)*fib(n+1))%mod);

    }
	return 0;
}
