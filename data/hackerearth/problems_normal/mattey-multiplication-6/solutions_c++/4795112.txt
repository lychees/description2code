#include <iostream>
#include<stack>
using namespace std;

int main()
{
    int test;
    stack<int> hash;
    cin>>test;
    long long m,n;
    while(test--)
    {
    //	for(int i=0;i<64;i++)
    //	hash[i]=0;
    	cin>>n>>m;
    	int k=0;
    	while(m)
    	{
    		if(m&1)
    		{
    			hash.push(k);
    		}
    		k++;
    		m>>=1;
    	}
    	while(hash.empty()==false)
    	{
    		cout<<"("<<n<<"<<"<<hash.top()<<")";
    		hash.pop();
    		if(hash.empty()==false)
    		cout<<" + ";
    	}
    	
    	cout<<"\n";
    	
    
    }
    return 0;
}
