#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int inp[1030][1030];
int res[12][1030];

int main()
{
	
    int N;
    cin>>N;
    int len = (int) pow(2, N);
    for(int i = 2;i<=len;i++){
        for(int j = 1;j<i; j++){
            int temp;
            cin>>temp;
            //cout<<temp;
            inp[i][j] = temp;
        }
    }
    for(int i = 1;i<=len;i++){
        res[i][0] = i;
    }
    int winner;
    for(int i = 1; i<=N; i++){
        int temp = (int) pow(2,i-1);
        for(int j = 1;j<len;){
            int first = res[j+temp][i-1];
            int second = res[j][i-1];
            //cout<<"first: "<<first<<"second"<<second;
            if(inp[first][second] == 1){
                res[j][i] = first;
                winner = first;
            }
            else{
                res[j][i] = second;
                winner = second;
            }            
            //cout<<"winner: "<<res[j][i]<<" "<<"level"<<i<<" "<<j<<"\n";
            j = j + temp + temp;
        }
    }
    for(int i = 1;i<=len;i++){
    	
    }
    cout<<winner;
    return 0;
}
