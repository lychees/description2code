#include <stdio.h>
int R,C;
int a[20][20];
void spiralPrint(int m, int n)
{
    int i, k = 0, l = 0;
    int cn = 1;
    while (k < m && l < n)
    {
        /* Print the first row from the remaining rows */
        for (i = l; i < n; ++i)
        {
            a[k][i] = cn++;
        }
        k++;

        /* Print the last column from the remaining columns */
        for (i = k; i < m; ++i)
        {
            a[i][n-1] = cn++;
        }
        n--;

        /* Print the last row from the remaining rows */
        if ( k < m)
        {
            for (i = n-1; i >= l; --i)
            {
                a[m-1][i] = cn++;
            }
            m--;
        }

        /* Print the first column from the remaining columns */
        if (l < n)
        {
            for (i = m-1; i >= k; --i)
            {
                a[i][l] = cn++;
            }
            l++;
        }
    }
}

/* Driver program to test above functions */
int main()
{
    int t;
    scanf("%d",&t);

    while(t--) {
    	int n;
    	scanf("%d",&n);

    	int
    	R = n;
    	C = n;

    	int i,j;
    	int cn = 1;

	    spiralPrint(R, C);
        for(i = 0; i < n; ++i) {
            for(j = 0; j <n ; ++j)
                printf("%d", a[i][j]);
            printf("\n");
        }
        printf("\n");
    }

    return 0;
}
