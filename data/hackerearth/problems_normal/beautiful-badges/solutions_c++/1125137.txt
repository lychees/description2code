#include <cstdio>
#include <algorithm>
using namespace std;

#define MAX 100005

int n;
int qb, b[MAX];
int qy, y[MAX];

int main() {
    scanf("%d", &n);
    qb = qy = 0;
    for (int i = 0; i < n; i++) {
        int c, d;
        scanf("%d %d", &c, &d);
        if (c == 1)
            b[qb++] = d;
        else
            y[qy++] = d;
    }
    sort(b, b+qb);
    sort(y, y+qy);
    int ans = 0;
    for (int i = 0, j = 0; i < qb && j < qy; ) {
        if (b[i]*b[i] * 2 <= y[j]*y[j]) {
            i++;
            continue;
        }
        if (y[j]*y[j] * 2 <= b[i]*b[i]) {
            j++;
            continue;
        }
        ans++;
        i++;
        j++;
    }
    printf("%d\n", ans);
}
