#include <iostream>
using namespace std;
long long int loser[1000006];
void precompute()
{
 long long int d=0;
 for(long long int i=0;i<1000006;++i)
 {
   if(loser[i])continue;
    loser[i]=i+d;
    if(i+d<1000006)loser[i+d]=i;
    d++;
 }
}
int main() {
long long int temp,t,n1,n2;
cin>>t;
precompute();
while(t--)
{
 cin>>n1>>n2;
 if(n1<n2)
 {
  temp=n1;
  n1=n2;
  n2=temp;
  }
 if(n1==loser[n2])
  cout<<"Don't Play\n";
 else
  cout<<"Play\n";
}
	return 0;
}
