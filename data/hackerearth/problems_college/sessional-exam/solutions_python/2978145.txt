def main():
	a=map(int,raw_input().split(" "))
	n,m=a[0],a[1]
	b=[]
	for i in range(n):
		b.append(raw_input())
	s=[]
	for j in range(m):
		c=[]
		for i in range(n):
			c.append(int(b[i][j]))
		#print len(c)
		m=max(c)
		for k in range(n):
			if c[k]==m :
				if k not in s:
					s.append(k)
	print len(s)
if __name__=='__main__':
	main()
