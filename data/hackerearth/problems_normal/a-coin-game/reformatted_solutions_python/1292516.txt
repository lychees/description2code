import math
t = input()
phi = 1.61803398874989484
for i in range(0, t):
    x = raw_input()
    x = x.split()
    a = int(x[0])
    b = int(x[1])
    minimum = min(a, b)
    maximum = max(a, b)
    r = maximum
    r = r-minimum
    p = phi*r
    x = int(math.floor(p))
    if x == minimum:
        print("Don't Play")
    else:
        print("Play")
