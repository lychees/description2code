#include <bits/stdc++.h>
using namespace std;
int a[100005],b[100005];
int main()
{	
	int t,n,i;
	cin >> t;
	while(t--)
	{
		cin>>n;
		for( i=1;i<=n;i++)
		{
			cin>>a[i];
			b[i]=a[i];
		}
		sort(a+1,a+n+1);
		int flag=1;
		for( i=1;i<=n;i++)
		{
			if(a[i]!=b[i])
			{
				flag=0;
				break;
			}
		}
		if(flag)
		{
			cout<<-1<<" "<<-1<<endl;
			continue;
		}
		for(i=1;i<n;i++)
		{
			if(b[i]>b[i+1])
			{
				break;
			}
		}
		int x=i,y=i+1;
		int p=x-1,q=y+1;
		while(p>=1 &&b[p]>b[x] )
		{
		p--;
		x--;	
		}
		
			while(q<=n &&b[q]<b[y] )
		{
			q++;
			y++;
		}
		
		for(int i=x,j=y;i<j;i++,j--)
		{
			int temp=b[i];
			b[i]=b[j];
			b[j]=temp;
		}
		
		 flag=1;
		for( i=1;i<=n;i++)
		{
			if(a[i]!=b[i])
			{
				flag=0;
				break;
			}
		}
		if(flag==0)
		{
			cout<<-1<<" "<<-1<<endl;
			continue;
		}
		cout<<x<<" "<<y<<endl;
		
	}
}
