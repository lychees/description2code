#include <bits/stdc++.h>
using namespace std;

bool isPalindrome(string s)
{
	int i=0,j=s.size()-1;
	while(i<j)
	{
		if(s[i++]!=s[j--])
			return false;
	}
	return true;
}

int main()
{
    int t;
    string s;
    cin >> t;
    while(t--)
    {
    	cin >> s;
    	for(int i=0;i<s.size();i++)
    	{
    		if(isPalindrome(s.substr(i,s.size()-i)))
    		{
    			cout << s.size()+i << endl;
    			break;
    		}
    	}
    }
    return 0;
}
