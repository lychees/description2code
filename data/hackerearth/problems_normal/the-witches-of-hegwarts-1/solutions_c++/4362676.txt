#include <iostream>
using namespace std;
#include<bits/stdc++.h>
typedef long long int ll;



ll bfs(ll n)
{
	queue<ll> q;
	map<ll,ll> mp;
	q.push(n);
	
	while(!q.empty())
	{
		
		ll s = q.size();
		
		while (s--)
		{
		
		ll x  = q.front();
	
	//	cout << x <<"  " <<  mp[x] <<endl;
		q.pop();
		
		if ((x%2==0) and (mp[x/2]==0))
			{
			q.push(x/2);
			mp[x/2] = mp[x]+1;
			}
			
		if ((x%3 == 0) and (mp[x/3]==0))
		{
			q.push(x/3);
			mp[x/3]=mp[x]+1;
		}
		
		if (mp[x-1]==0)
		{q.push(x-1);
		mp[x-1]= mp[x]+1;
		}
		
		}
		
		if (mp[1] > 0)
		{
			return mp[1];
		}
		
	}
	
	
}



int main()
{
	ll T,n;
	cin >> T;
	while (T--)
	{
		cin >> n;
		if (n==1)
			cout << 0;
		else
		cout << bfs(n);
		cout << endl;
		
		
	}
    
}
