//HackerEarth--Angles
#include<bits/stdc++.h>
#define MAX 15

using namespace std;

int reqdAngle,angles[MAX],givenAngles;

int dp[365][365];
int solve(int idx,int currentAngle){
// cout<<"\nAt "<<idx<<" we have "<<currentAngle;
   if(currentAngle==reqdAngle)
	   return 1;
 
   if(idx==360) 
   
   return 0;
     
  if(dp[idx][currentAngle]!=-1) return dp[idx][currentAngle];   
   
   int ans=0,i;
   
 // cout<<"\nAt "<<idx<<" we have "<<currentAngle;
  //system("pause");
     
   for(i=0;i<givenAngles;i++){
	
	  ans=ans|solve(idx+1,(currentAngle+angles[i])%360);
	  ans|=solve(idx+1,(currentAngle-angles[i]+360)%360);   
   
   }


   return dp[idx][currentAngle]=ans;
   
}

int main(void){
  
  int testCases,i;
     
  cin>>givenAngles>>testCases;
  
  for(i=0;i<givenAngles;i++)
    cin>>angles[i];
	
  while(testCases--){
  	
  	cin>>reqdAngle;
  	memset(dp,-1,sizeof dp);
  	
  	int ans=solve(0,0);
  	if(ans) cout<<"YES\n";
  	else cout<<"NO\n";
  	
  	
  	
  }	  

   
  
  return 0;
   
}
