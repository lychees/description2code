#include <iostream>
using namespace std;

int main()
{
    int t;cin>>t;
    while(t--)
    {
    	unsigned long long a,sum=0;
    	int no;
    	cin>>no;
    	for(int i=0;i<no;i++)
    	{
    		cin>>a;
    		sum+=a;
    	}
    	if(sum%no==0)
    	{
    		cout<<"YES"<<endl;
    	} else {
    		cout<<"NO"<<endl;
    	}
    	cout<<(sum/no)<<" "<<(sum%no)<<endl;
    }
}
