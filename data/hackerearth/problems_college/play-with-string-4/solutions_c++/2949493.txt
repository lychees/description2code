#include<iostream>
#include<stdio.h>
#include<cstring>
#include<cmath>
using namespace std;
char a[1000001],b[1000001],c[1000001];
int main()
{
	long long tc;
	scanf("%lld",&tc);
	while(tc--)
	{
	int n,i,j,f=0;
scanf("%s",a);
n=strlen(a);
for(i=n-1,j=0;i>=0;i--,j++)
{
	b[j]=a[i];
}b[j]='\0';
for(i=0;i<n-1;i++)
{
	if(abs((int)a[i+1]-(int)a[i])==abs((int)b[i+1]-(int)b[i]))
	f=1;
	else
	{
		f=0;
		printf("Not Funny\n");
		break;
	}
}
if(f==1)
printf("Funny\n");
}
return 0;
}
