#include <iostream>
using namespace std;

int main()
{
	static bool a[4000001]={0};
	long i=0;
	int j=0;
	a[0]=a[1]=1;
	for(i=4;i<4000001;i+=2)
	a[i]=1;
	for(i=3;i<=2000;i+=2)
	{
		if(a[i]==1)
		continue;
		for(j=i;j*i<4000001;j++)
		a[i*j]=1;
	}
	long long ans=0;
	long p;
	scanf("%ld",&p);
	if(p>2)
	ans=2;
	for(i=3;i<p;i+=2)
	if(!a[i])
	ans+=i;
	printf("%lld",ans);
	return 0;
}
