#include <iostream>
#include <cstring>
using namespace std;

int main() {
	// your code goes here
	int t;
	scanf("%d",&t);
	while(t>0){
	string s;
	cin>>s;
	long long int n=s.size();
	if(n%2==0){
	long long int j,c=0;
	for(long long int i=0;i<=n/2;i++){
	if(s[i]!=s[n-1-i]){ c++;
	j=i;
	break;}
	}
	if(c==0) printf("-1\n");
	else printf("%lld\n",n-j-1);
	
	}
	else{
	long long int k,d=0;
	for(long long int i=0;i<n/2;i++){
	if(s[i]!=s[n-1-i]){ d++; 
	k=i;
	break;}
	}
	if(d==0) printf("-1\n");
	else printf("%lld\n",k);
	
	}
	t--;
	}
	return 0;
}
