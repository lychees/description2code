#include <bits/stdc++.h>
using namespace std;

const int maxn = 1e5+123;

int n,ar[maxn];
bool used[maxn];

int main(){
	
	int t;
	
	scanf("%d",&t);
	
	while( t-- ){
		scanf("%d",&n);
		for( int i=0 ; i < n ; i++ )
			scanf("%d",&ar[i]);
		
		int cur = 0;
		for( ; cur != -1 ; cur = ar[cur] ){
			if( used[cur] )
				break;
			used[cur] = 1;
		}
		if( cur == -1 )
			puts("Win");
		else
			puts("Lose");
		
		cur = 0;
		for( ; cur != -1 ; cur = ar[cur] ){
			if( !used[cur] )
				break;
			used[cur] = 0;
		}
	}
	
	return 0;
}
