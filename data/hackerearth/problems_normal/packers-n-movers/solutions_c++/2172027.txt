#include <iostream>
using namespace std;

int a[101];
int dp[20][305];
int solve(int m,int p)
{
	if(dp[m][p]!=-1)
		return dp[m][p];
	if(m==1)
	{
		int sum=0;
		for(int i=p-1;i>=0;i--)
			sum+=a[i];
		dp[m][p]=sum;
		return sum;
	}
	if(p==0)
	{
		dp[m][p]=0;
		return 0;
	}
	int ans=1000000000;
	int sum=0;
	for(int i=p-1;i>=0;i--)
	{
		sum+=a[i];
		ans=min(ans,max(sum,solve(m-1,i)));
	}
	dp[m][p]=ans;
	return ans;
	
}
int main()
{
    int n,m;
    cin>>n>>m;
    
    for(int i=0;i<m;i++)
    	cin>>a[i];
    for(int i=0;i<=n;i++)
    	for(int j=0;j<=m;j++)
    		dp[i][j]=-1;
    cout<<solve(n,m);
    return 0;
}
