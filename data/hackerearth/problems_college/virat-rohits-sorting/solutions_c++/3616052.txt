#include <iostream>
#include <utility>
#include <algorithm>
#define lli long long int 
using namespace std;
pair <lli,lli>a[1000006];
int main()
{
    lli n;
    cin>>n;
    
    for(lli i=0;i<n;i++)
		{
			scanf("%lld%lld",&a[i].first,&a[i].second);
		}
		sort(a,a+n, []( const std::pair<int, int> &p1, const std::pair<int, int> &p2 )
               {
               		
                   return p1.first<p2.first || (p1.first==p2.first && p1.second>p2.second) ;
                   
               });
	for(lli i=0;i<n;i++)
		{
			printf("%lld %lld\n",a[i].first,a[i].second);
		}
		
}
