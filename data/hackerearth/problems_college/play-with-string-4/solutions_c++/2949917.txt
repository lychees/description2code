#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	string s,s1;
   	cin>>s;
   	int n=s.length();
   	s1=s;
   	int j=0,i;
   	reverse(s1.begin(),s1.end());
   	for(i=1;i<n;i++)
   	{
   		if(abs(s[i]-s[i-1])!=abs(s1[i]-s1[i-1]))
   		j=1;
   	}
   	if(j)
   	cout<<"Not Funny"<<endl;
   	else
   	cout<<"Funny"<<endl;
   	
   }
}
