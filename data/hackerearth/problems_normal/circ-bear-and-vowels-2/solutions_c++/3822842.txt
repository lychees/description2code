#include <iostream>
#include <cstring>
#include <cmath>
using namespace std;

int main()
{
    int t;
	cin >> t;
	while (t--){
		string s;
		cin >> s;
		int vowels = 0;
		int size = s.size();
		int run = 0;
		bool pass = true;
		for (int i = 0; i < size; i++){
			if (s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u' || s[i] == 'y'){
				vowels++;
				run = 0;
			}
			else{
				run++;
			}
			if (run > 2){
				pass = false;
				break;
			}
		}
		if (pass && vowels >= ceil((float)size/2)){
			cout << "easy\n";
		}
		else{
			cout << "hard\n";
		}
	}
    return 0;
}
