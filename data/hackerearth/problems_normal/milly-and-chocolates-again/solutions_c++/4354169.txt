#include <iostream>
#include <unordered_map>
using namespace std;

int main()
{
    int N,Q;
    unordered_map<int,int> maps;
    int cnt=1,num;
    cin>>N;
    for(int i=0;i<N;i++)
    {
    	cin>>num;
    	while(num!=0)
    	{
    		maps.insert({cnt,i+1});
    		num--;
    		cnt++;
    	}
    }
    
    cin>>Q;
    while(Q>0)
    {
    	Q--;
    	cin>>num;
    	auto search = maps.find(num);
    	cout<<search->second<<"\n";
    }
    
    return 0;
}
