#include <iostream>
using namespace std;

long long int ncr(int n,int r)
{
     if(n==r) return 1;
     if(r==1) return n;
     return ncr(n-1,r) + ncr(n-1,r-1);
}
int main()
{
    int powx,powy;
    int t,k,n;
    cin>>t;
    while(t--){
    	cin>>k>>n;
    	powx = n-k+1;
    	powy = k-1;
    	if(powx==0){
    		cout<<"y^"<<powy<<endl;
    	}else if(powy==0){
    		cout<<"x^"<<powx<<endl;
    	}else if(powy==1){
    		cout<<ncr(n,k-1)<<" x^"<<powx<<" "<<"y "<<endl;
    	}else if(ncr(n,k-1)==1){
    		cout<<"x^"<<powx<<" "<<"y^"<<powy<<endl;
    	}
    	else{
    		cout<<ncr(n,k-1)<<" x^"<<powx<<" "<<"y^"<<powy<<endl;
    	}
    }
    return 0;
}
