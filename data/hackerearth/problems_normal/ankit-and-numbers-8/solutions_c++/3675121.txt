#include <iostream>
#include <cassert>
typedef long long int lli;
using namespace std;
void sumP(lli n){
    lli p2=1;
    for(lli i=0;i<n;i++){
        p2=p2*2;
    }
    lli p3=p2/2;
    lli sum=0;
    for(lli i=1;i<=n;i++){
        sum=sum+i*p3;
    }

    cout << sum << endl;
}
int main(){
    lli t;
    cin >> t;
    assert(t>=1 and t<=42);
    lli n;
    for(lli i=0;i<t;i++){
        cin >> n;
        assert(n>=1 and n<=42);
        sumP(n);
    }
    return 0;
}
