#include <iostream>

using namespace std;

int main()
{
    int a[4][2]={
    {0,1},
    {2,1},
    {0,3},
    {2,1}
    };
    long int t,r,p[100001],s,k,c;
    for(long int i=1;i<100001;i++)
    {
        s=0;
        c=0;
        r=i;
        while(r!=0)
        {
            k=r%2;
            s=a[s][k];
            r=r/2;
            if(s==3)
                c++;
        }
        p[i]=c;
    }
    cin>>t;
    while(t--)
    {
        cin>>r>>k;
        c=0;
        for(long int i=1;i<=r;i++)
        {
            if(p[i]>=k)
                c++;
        }
        cout<<c<<endl;
    }
    return 0;
}
