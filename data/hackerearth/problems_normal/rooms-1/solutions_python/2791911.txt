T = input()

while(T>0):
	n = input()
	A = map(int,raw_input().split())
	D = map(int,raw_input().split())
	E = []
	l = len(A)
	for i in range(l):
		E.append(A[i]+D[i])
	A.sort()
	E.sort()
	i = 0
	j = 0
	m = 0
	g = 0
	while((i<n) and (j<n)):
		if(A[i] < E[j]):
			g = g+1
			if(g>m):
				m = g
			i = i+1
		else:
			g = g - 1
			j = j +1
	print(m)
	T = T-1
