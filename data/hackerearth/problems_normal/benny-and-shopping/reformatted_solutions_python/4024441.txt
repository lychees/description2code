n = int(raw_input())
for k in range(n):
    p = 0
    temp = raw_input()
    for i in range(len(temp)):
        if temp[i] == '$':
            i += 1
            break
    for j in range(i, len(temp)):
        try:
            r = int(temp[j])
            p = p*10+r
        except BaseException:
            if temp[j] != ' ':
                break
    print '$'+str(p)
