#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int te;
    cin>>te;
	
   long long product;
    for(int ei=0;ei<te;++ei){
    	product=1;
    	string str,rev;
    	cin>>str;
    	rev=str;
    	reverse(rev.begin(),rev.end());
    	if(str==rev){
    		cout<<"Palindrome"<<endl;
    	}else{
            
    		for(int ei=0;ei<str.length();++ei){
    			product=product*(str[ei]-96);

    		}
    		cout<<product<<endl;
    	}

    }
    return 0;
}
