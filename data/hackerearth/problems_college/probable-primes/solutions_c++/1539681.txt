#include <iostream>
#include <vector>
#include <string.h>
#include <stdio.h>
#define max 1e+6
using namespace std;
bool h[1000005];
int main()
{
	memset(h,false,sizeof(h));
	long int i,j;
	for(i=2;i<=max;i++)
	{
		for(j=2*i;j<=max;j+=i)
		 h[j]=true;
	}
	vector<long int> a;
	a.push_back(0);
	a.push_back(0);
	double count=0;
	for(i=2;i<=max;i++)
	{
		if(h[i]==false)
		 count++;
		a.push_back(count);
	}
	int t;
	cin>>t;
	while(t--)
	{
		long int l,r;
		cin>>l>>r;
		printf("%.6f\n",(float)(a[r]-a[l-1])/(r-l+1));
		//printf("%.6f\n",a[1000000]);
	}
    return 0;
}
