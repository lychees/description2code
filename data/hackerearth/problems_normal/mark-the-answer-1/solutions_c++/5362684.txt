#include <stdio.h>
 
inline int fastread(int& x)
{
	register char c=getchar_unlocked();
	x=0;
	for(;c>47 && c<58;c = getchar_unlocked()) {x = (x<<1) + (x<<3) + c - 48;}
}



int main()
{
    int n,x,temp,ans=0,flag1=0;
    fastread(n);
    fastread(x);
 	for(;ans<n;ans++)
 	{
 		fastread(temp);
 		if(temp>x && flag1==1) 
 		break;
 		else
 		if(temp>x)
 		flag1=1;
 		
 	}
    
    printf("%d",ans-flag1);
    return 0;
}
