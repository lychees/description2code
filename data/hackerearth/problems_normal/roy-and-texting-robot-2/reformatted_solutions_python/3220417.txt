button_groups = (
    '.,?!1',
    'abc2',
    'def3',
    'ghi4',
    'jkl5',
    'mno6',
    'pqrs7',
    'tuv8',
    'wxyz9',
    '_0')
key_weights = {}
key_groups = {}
for button_group in button_groups:
    key_weights.update({char: button_group.index(
        char) + 1 for char in button_group})

for index, button_group in enumerate(button_groups):
    key_groups.update({char: index for char in button_group})


def get_key_press_count(word):
    key_press = []
    previous_group = key_groups['1']
    current_group = key_groups['1']
    for char in word:
        current_group = key_groups[char]
        if previous_group != current_group:
            key_press.append(1)
        key_press.append(key_weights[char])
        previous_group = current_group
    return sum(key_press)


def app():
    num_test_cases = int(raw_input())
    if num_test_cases > 50:
        return
    test_cases = []
    for _ in range(num_test_cases):
        case = raw_input()
        if len(case) > 1000:
            continue
        test_cases.append(case)

    for word in test_cases:
        print get_key_press_count(word)


app()
