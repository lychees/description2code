#include <iostream>
#include<stdlib.h>
#include<stdio.h>
using namespace std;
float div(int x1,int y1,int x2,int y2,int x3,int y3)
{
    float area = ((x1*(y2-y3) + x2*(y3-y1)+ x3*(y1-y2))/2.0);
    if(area<0)
    {
        return (area*(-1));
    }
    return area;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	float a,b,c,d,e,f;
    	float x,y;

    	cin>>a>>b>>c;
    	cin>>d>>e>>f;
    	cin>>x>>y;
    	 float are;
    	 are = div(a,b,c,d,e,f);
    	 float a1,a2,a3;
    	 a1 = div(x,y,c,d,e,f);
    	 a2 = div(a,b,x,y,e,f);
    	 a3 = div(a,b,c,d,x,y);

    	 if(a1+a2+a3 == are)
    	 {
    	 	cout<<"INSIDE\n";
    	 }
    	 else
    	 {
    	 	cout<<"OUTSIDE\n";
    	 }
    }
    return 0;
}
