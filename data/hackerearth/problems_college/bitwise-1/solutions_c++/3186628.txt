#include<iostream>
using namespace std;
int main()
{
	long long int x,y;
	long int k,count=0;
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		cin>>x>>k;
		count=0;
		for(y=1;y<90000000000000;y++)
		{
			if(x+y==(x^y))
			count++;
			if(count==k)
			{
				cout<<"Case #"<<i+1<<": "<<y<<"\n";
				break;
			}
		}
		//cout<<"hello";
	}
	
	return 0;
}
