#include<bits/stdc++.h>

#define gcd(a,b) __gcd(a,b)
#define sf(n) scanf("%d",&(n))
#define pf(n) printf("%d\n",(n))
#define setbits(n) __builtin_popcount((n))
#define pq priority_queue
using namespace std;
int n;
int arr[15];
int sum=0;
int dp[1500][1500][15];
int func(int gr1,int gr2,int i)
{
    if(i==n)
    {
        int gr3=sum-(gr1+gr2);
        if(gr3>=gr1 && gr3>=gr2)
            return gr3;
        else
            return INT_MAX;
    }
    if(dp[gr1][gr2][i]!=-1)
        return dp[gr1][gr2][i];
    else
    return dp[gr1][gr2][i]=min(func(gr1,gr2,i+1),min(func(gr1+arr[i],gr2,i+1),func(gr1,gr2+arr[i],i+1)));
}
int main()
{

    sf(n);
    memset(dp,-1,sizeof(dp));
    for(int i=0;i<n;i++)
        {
            sf(arr[i]);
            sum+=arr[i];
        }
    pf((func(0,0,0)));
}
