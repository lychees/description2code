def kadness(arr, n):
    maxSoFar = 0
    maxEndingHere = 0
    for i in range(n):
        maxEndingHere += arr[i]
        if maxEndingHere < 0:
            maxEndingHere = 0
        maxSoFar = max(maxEndingHere, maxSoFar)
    return maxSoFar

noe = input()
if noe == 0:
    print 0
    exit()
ele = map(int, raw_input().split())
if noe == 1:
    print ele[0]
    exit()

print kadness(ele, noe)
