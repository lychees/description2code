#include <bits/stdc++.h>
using namespace std;
int dp[1000005];
template<typename T>
inline void fi(T *a)
{
 register char c=0;
 while (c<33) c=getchar();//_unlocked();
 *a=0;
 int tmp = 0;
 while (c>33)
 {
     if ( c == 45 ) tmp = 1;
     else *a=*a*10+c-'0';
     c=getchar();//_unlocked();
 }
 if ( tmp == 1 ) *a = 0-(*a);
}
void pre()
{
    dp[0] = dp[1] = 0;
    for ( int i = 2; i <= 1000000; i++ ) {
        if ( dp[i] > 0 ) continue;
        for ( int j = i; j <= 1000000; j += i ) {
            int k = j;
            while ( k%i == 0 ) dp[j]++, k /= i;
        }
    }
    for ( int i = 2; i <= 1000000; i++ ) dp[i] += dp[i-1];
    return;
}
int main()
{
    pre();
    int t,a,b;
    fi(&t);
    assert(t <= 1000000);
    while ( t-- ) {
        fi(&a), fi(&b);
        assert(a <= b && a >= 1 && b <= 1000000);
        printf("%d\n", dp[b]-dp[a-1]);
    }
    return 0;
}
