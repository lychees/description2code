#include <bits/stdc++.h>

using namespace std;


int coins(int arr[], int m, int n) //m is no of elements
{
if(m==0 || n<0) return 0;
	if(n==0) return 1;
	
	return coins(arr, m-1, n) + coins(arr, m, n-arr[m-1]);
	
}

int main()
{

	int n;
	cin>>n;
	int arr[3];
	int i=0 ,x;
	while(cin>>x){arr[i++]=x;}
	
	  cout<<coins(arr, i, n)<<endl;
	
}
