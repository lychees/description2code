#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		
	
	  int m,n,p;
	  int x1,y1,x2,y2;
	  cin>>m>>n;
	  cin>>x1>>y1;
	  cin>>x2>>y2;
	  cin>>p;
	  int a=min(abs(x1-x2),min(abs(m+x1-x2),abs(m+x2-x1)));
	  int b=min(abs(y1-y2),min(abs(n+y1-y2),abs(n+y2-y1)));
	 // cout<<a+b<<endl;
	  int temp=(a+b)*p;
	  cout<<temp<<endl;
	  if(temp<=1000)
	  cout<<"You saved the group."<<endl;
	  else
	  cout<<"Let go of the group."<<endl;
	}  
    return 0;
}
