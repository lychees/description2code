#include<iostream>
using namespace std;
long a[100005];int v[100005];
int main(){
	long t;
	cin>>t;long n;
	for(long i=0;i<t;i++){
		cin>>n;
		for(long j=0;j<n;j++){
		cin>>a[j];v[j]=0;}
		int l=0;int o=0;
		long j=0;if(a[j]==-1)l=1;
		while(l==0&&o==0){
			v[j]=1;
			j=a[j];
			if(v[j]==1)o=1;
			if(a[j]==-1)l=1;
		}
		if(l==1)cout<<"Win"<<endl;
		if(o==1)cout<<"Lose"<<endl;
		
	}
}
