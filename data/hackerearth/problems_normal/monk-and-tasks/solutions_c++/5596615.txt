#include <iostream>
using namespace std;
string output[61];
int main() {
    int t, n;
    long int number;
    scanf("%d", &t);
    while (t--){
        scanf("%d", &n);
        int j = 61;
        while (j--){
            output[j]="";
        }
        while (n--) {
            scanf("%ld", &number);
            int no_ones = __builtin_popcount (number);
            if (no_ones == 1){
                printf("%ld ", number);
            } else{
                output[no_ones] = output[no_ones].append(to_string(number));
                output[no_ones] = output[no_ones].append(" ");
            }

        }
        for (int i = 2; i < 61; ++i) {
            printf("%s", output[i].c_str());
        }
        printf("\n");
    }
    return 0;
}
