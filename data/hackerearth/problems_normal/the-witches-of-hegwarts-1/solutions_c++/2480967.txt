#include<bits/stdc++.h>
using namespace std;
#define mp make_pair
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		unordered_map<int , int> check;
		queue<pair<int , int> > q;
		q.push(mp(n , 0));
		check.insert(mp(n , 1));
		while(!q.empty()){
			pair<int , int > x = q.front();
			q.pop();
			int w = x.second;
			int num = x.first;
			if(num == 1){
				cout<<w<<endl;
				break;
			}
			if(num > 0 && num%2==0){
				int temp = num/2;
				if(check.find(temp) == check.end()){
					q.push(mp(temp , w+1));
					check.insert(mp(temp , 1));
				}
			}
			if(num > 0 && num%3==0){
				int temp = num/3;
				if(check.find(temp) == check.end()){
					q.push(mp(temp , w+1));
					check.insert(mp(temp , 1));
				}
			}
			if(num-1>0){
				int temp = num-1;
				if(check.find(temp) == check.end()){
					q.push(mp(temp , w+1));
					check.insert(mp(temp , 1));
				}
			}
		}
	}
	return 0;
}
