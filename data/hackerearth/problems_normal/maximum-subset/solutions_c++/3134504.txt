#include<bits/stdc++.h>
using namespace std;

long long mod = 1000000007;
map<long long,long long> mod_inv;

long long val(vector<long long> &v)
{
  long long ans = 1;
  sort(v.begin(),v.end());
  for(int i=0;i<v.size()/2;i++)
  {
    ans = (ans * mod_inv[v[i]] )%mod;
  }

  for(int i=v.size()/2; i<v.size(); i++)
    ans = (ans * v[i]%mod)%mod;

  return ans;
}

int main()
{
  long long t,n;
  cin>>t;
  mod_inv[1] = 1;
  for(int i = 2; i <= 100000; i++)
        mod_inv[i] = (-(mod/i) * mod_inv[mod % i]) % mod + mod;

  while(t--)
  {
    cin>>n;
    long long a[n];
    for(int i=0;i<n;i++)
      cin>>a[i];

    long long size = pow(2,n) , ans = INT_MIN;
    for(long long i = 1; i<size; i++)
    {
      long long temp = i , len = 0;
      while(temp)
      {
        len++;
        temp &= temp-1;
      }
      if(len%2)
        continue;
      //cout<<i<<"\n";
      vector<long long> v;
      for(int j=0;j<n;j++)
        if(i & 1<<j)
          v.push_back(a[j]);
      ans = max(ans,val(v));
    }

    cout<<ans<<"\n";

  }
}
