#include<iostream>
#include<algorithm>

using namespace std;

int main()
{
    int n;
    cin>>n;
    string s1,temp;

    while(n>0)
    {
        temp="";
        cin>>s1;
        char s[s1.length()];


        for(int i=0;i<s1.length();i++)
        {s[i]=s1[i];}
    int elements = sizeof(s) / sizeof(s[0]);
    sort(s, s + elements);


        for(int i=1;i<s1.length();i++)
        {
            if(s[i-1]!=s[i])
            {temp=temp+s[i];}
        }
        temp=s[0]+temp;


        if(temp.length()<26)
        {cout<<"NO\n";
        }
        else
        {cout<<"YES\n";}
       

    n--;}


return 0;}
