n = int(raw_input())
l = raw_input()

q = [x for x in range(1, n+1)]

o, p = 0, 0
while len(q) > 1:
    if l[p] == 'a':
        o += 1
        if o >= len(q):
            o = 0
    else:
        q.remove(q[o])
        if o >= len(q):
            o = 0
    p += 1
    if p >= len(l):
        p = 0
print q[0]
