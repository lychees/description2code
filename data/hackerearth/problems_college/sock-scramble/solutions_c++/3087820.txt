#include <iostream>

using namespace std;

int main()
{
    int t,n,count=0;
    
    cin>>t;
    while(t--)
    {
    	count=0;
    	cin>>n;
    	long int a[n];
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<n;j++)
    		{
    			
    				if(a[i]==a[j])
    				{
    				count++;
    				}
    			
    		}
    		if(count!=2)
    		{
    		cout<<a[i]<<'\n';
    		goto theme;
    		}
    		count=0;
    	
    	}
    	theme:;
    }
    return 0;
}
