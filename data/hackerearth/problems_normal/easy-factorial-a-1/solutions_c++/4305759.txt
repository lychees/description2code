#include <bits/stdc++.h>
using namespace std;
long long solve(long long t,long long x)
{
  long long ans = 1;
  for(long long i = 1;i<= t; ++i)
    ans = (ans * (i  % x) % x);
  return ans;
}
int main(int argc, char const *argv[])
{
  std::ios::sync_with_stdio(false);
  long long t,x;
  cin >> t >> x;
  cout << solve(t,x) << endl;
  return 0;
}
