#include <iostream>
using namespace std;

int main()
{
    string s;

    if(s[0]!='_')
    {
    
    s[0] = s[0]-('a'-'A');
    int i =1;
    
    while(i<s.size())
    {
    	if(s[i]=='_')
    	{
    	while(i<s.size() && s[i]=='_')
    	{
    		i++;
    	}
    	
    	s[i]=s[i]-('a'-'A');
    	
    	}
    	
    	i++;
    }
    
    }
    
    else
    {
    	
    	int i =0;
    
    while(i<s.size())
    {
    	if(s[i]=='_')
    	{
    	while(i<s.size() && s[i]=='_')
    	{
    		i++;
    	}
    	
    	s[i]=s[i]-('a'-'A');
    	
    	}
    	
    	i++;
    }
    
    	
    }

}
