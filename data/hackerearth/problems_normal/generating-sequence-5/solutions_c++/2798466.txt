#include <iostream>
#include<vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    vector<long int> *v = new vector<long int>[t];
    int no = 0;
    while(no < t){
    	long int g;
    	int n;
    	cin>>g;
    	cin>>n;
    	
    	int cnt = 2;
    	
    	long int x;
    	x = g*cnt;
    	
    	if(n<=2){
    		v[no].push_back(g*2);
    		v[no].push_back(g*3);		
    	}
    	else{
    		for(int i=0; i<n-1; i++){  
				v[no].push_back(x);
    		}
    		v[no].push_back(g*3);
    	}
    	no++;
    }
    
    for(int i=0; i<no; i++){
    	for(int j=0; j<v[i].size();j++)
    		cout<<v[i][j]<<" ";
    	cout<<"\n";
    }
    
    return 0;
}
