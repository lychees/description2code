from collections import defaultdict

test_cases = input()

for _ in xrange(test_cases):
    count_dict = defaultdict(int)
    _2 = input()
    l = map(int, raw_input().split())
    for el in l:
        count_dict[el] += 1

    highest_val = max(count_dict.values())
    # print highest_val
    result = sorted([_ for _ in count_dict if count_dict[_] == highest_val])
    result.reverse()
    # print result
    print ' '.join([str(_) for _ in result])
