t = int(raw_input())
while t > 0:
    t -= 1
    n = int(raw_input())
    l = []
    m = []
    for i in range(n):
        a, b = raw_input().split(" ")
        a, b = int(a), int(b)
        l.append(a+b)
        m.append(b)
    l.sort()
    l = l[::-1]
    s = sum(m)

    print l[0]+l[1]-s
