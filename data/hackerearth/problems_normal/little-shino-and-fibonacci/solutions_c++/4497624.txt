#include<bits/stdc++.h>
using namespace std;
#define fore(i,a,b) for(i=a;i<=b;i++)
#define forl(i,a,b) for(i=a;i<b;i++)
#define si(d) scanf("%d",&d)
//#define ss(s) scanf("%s",s)
#define sll(d) scanf("%lld",&d)
#define pin(d) printf("%d\n",d)
#define pln(d) printf("%lld\n",d)
#define ll long long
#define ull unsigned long long
ll a[15005]={0};
ll ans[15005];
//ans[0]=0;
ll mod=1e9+7;
ll mode=10000;
ll fib(ll n)
{
	a[0]=0;
	a[1]=1;
	if(n==0)
	return 0;
	if(n==1)
	return 1;
			if(a[n]!=0)
			return a[n]%mode;
			a[n]=fib(n-1)+fib(n-2);
			a[n]%=mode;
			return a[n];
		return (fib(n-1)+fib(n-2))%mode;
}
int main()
{
	
		int t;
		si(t);
		a[15005]=fib(15005);
		ans[0]=0;
		for (int i=1;i<=15005;i++ )
		{
			ans[i]=ans[i-1]+a[i];
			ans[i]%=mod;
		}
		while(t--)
		{
			ll l,r;
			sll(l),sll(r);
			--l,--r;
			ll sum=0,sumr=0,suml=0;
			ll fr=r/15000,fl=l/15000;
			r%=15000;
			l%=15000;
			fr%=mod;
			fl%=mod;
			sumr=(fr*ans[14999])%mod;
			sumr+=(ans[r]-ans[0]);
			sumr%=mod;
			suml=(fl*ans[14999])%mod;
			suml+=(ans[l-1]-ans[0]);
			suml%=mod;
		//	cout << sumr <<" " << suml << endl;
	     sum=(sumr-suml+mod)%mod;
	     pln(sum);
		}
	return 0;
}
