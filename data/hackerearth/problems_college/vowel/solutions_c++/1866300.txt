#include <bits/stdc++.h>
using namespace std;
int i,t,ctr;
int main()
{
    cin>>t;
    string s;
    while(t--)
    {
cin>>s;
ctr=0;
for(i=0;i<s.length();i++)
{
    if(s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u'||s[i]=='A'||s[i]=='E'||s[i]=='I'||s[i]=='O'||s[i]=='U'){
        cout<<s[i];
        ctr++;
        }

}

if(ctr==0)
    cout<<"No"<<endl;
    else
    cout<<endl;


}
	return 0;
}
