#include <bits/stdc++.h>
using namespace std;
 
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,k;
		cin>>n>>k;
		long long int dp[n],arr[n];
		memset(dp,0,sizeof(dp));
			long long int ans=-1000000009;
		for(int i=0;i<n;i++) 
		{
			cin>>arr[i];
			if(arr[i]<0) arr[i]=0;
			if(i<k+1) dp[i]=arr[i];
			ans=max(ans,dp[i]);
		}
		for(int i=k+1;i<n;i++)
		{
			dp[i]=arr[i];
			for(int j=i-k-1;j>=0 && j>=i-k-k-1;j--)
			{
				dp[i]=max(dp[i],dp[j]+arr[i]);
			}
			ans=max(dp[i],ans);
		}
//		if(ans<0) ans=0;
//		for(int i=0;i<n;i++) cout<<dp[i]<<" ";
		cout<<ans<<"\n";
	}
	return 0;
}
