using namespace std;


int arr[MAXN + 1], a;
LL inv[MAXN + 1];

LL power(LL m)
{
    if(m == 0)
        return 1;
    LL temp = power(m/2);
    temp = temp * temp;
    temp %= M;
    if(m % 2)
        temp *= a;
    temp %= M;
    return temp;
}

void pre()
{
    inv[1] = 1;
    for(int i=2;i<=MAXN;i++)
    {
        a = i;
        inv[i] = inv[i - 1] * power(M - 2);
        inv[i] %= M;
    }

}

int main()
{
    int t, i, j, n, sum, m;
    LL ans;
    sc(t);
    pre();
    while(t--)
    {
        sc(n);sc(m);
        sum = 0;
        for(i=1;i<=n;i++)
        {
             sc(arr[i]);
             sum += arr[i];
        }

        n = n - 1;
        m = m - sum;
        if(m + n < n)
        {
            printf("0\n");
        }
        else
        {
            m = m + n;
            ans = m;
            ans *= inv[n];
            ans %= M;
            for(i=2;i<=n;i++)
            {
                ans *= (m - i + 1);
                if(ans >= M)
                ans %= M;
            }
            if(ans < 0)
                ans += M;
            printf("%lld\n",ans);
        }
    }
}
