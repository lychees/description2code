#include <iostream>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
    int t,t1;

    t1 = t;
    long long n,x,n1;
    while(t--)
    {
    	cin>>n;
    	while(n>=10)
    	{
    		x = 0;
    		n1 = n;
    		
    		while(n1)
    		{
    			x+=(n1%10)*(n1%10);
    			n1/=10;
    		}

    	}

    	}
    	else

    	}
    	
    	
    }
    
    return 0;
}
