#include<bits/stdc++.h>
using namespace std;
const int N = 100005;
char arr[N];
bool ispalin(char arr[],int l,int r)
{
	while(l<=r)
	{
		if(arr[l]!=arr[r])
			return false;
		l++,r--;
	}
	return true;
}
int main()
{
	int n,q,p,l,r,x;
	scanf("%d%d",&n,&q);
	scanf("%s",arr);
	char ch;
	while(q--)
	{
		scanf("%d",&x);
		if(x==0)
		{
			scanf("%d %c",&p,&ch);
			arr[p-1] = ch;
		}else{
			scanf("%d%d",&l,&r);
			bool val = ispalin(arr,l-1,r-1);
			if(val) printf("Yes\n");
			else printf("No\n");
		}
	}
	return 0;
}
