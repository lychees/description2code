#include<bits/stdc++.h>
using namespace std;
int main(){
	int T;
	cin>>T;
	while(T--){
		long long int a,b,c;
		cin>>a>>b>>c;
		long long int ll,lu,wl,wu;
		ll=c;
		bool wcurr=false,lcurr=false;
		lcurr=true;
		while(true){
			if(wcurr){
				ll=ceil((float)wl/a);
				lu=wl-1;
				wcurr=false;
				lcurr=true;
				//cout<<"losing "<<ll<<" "<<lu<<endl;
				if(ll<=1)break;
			}
			else{
				wl=ceil((float)ll/b);
				wu=ll-1;
				lcurr=false;
				wcurr=true;
				//cout<<"winning "<<wl<<" "<<wu<<endl;
				if(wl<=1)break;
			}

		}
		if(wcurr){
		     cout<<"Adam\n";
		}
		else{
		    cout<<"Bob\n";
		}
	}
}
