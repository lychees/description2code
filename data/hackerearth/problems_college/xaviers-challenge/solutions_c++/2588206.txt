#include <iostream>
using namespace std;

int main()
{
    int tt;
    cin>>tt;
    cin.ignore();
    while(tt--){
    	string str;
    	getline(cin,str);
    	string rev;
    	int l=str.size();
    	long int sum=0;
    	for(int i=0;i<l;i++){
    		sum+=(int)(str[i]);
    		rev+=str[l-i-1];
    	}
    	long int avg=sum/l;
    	//cout<<sum<<"  "<<avg<<"  "<<l<<endl;
    	if(avg%2==0)
    		cout<<str<<endl;
        else
            cout<<rev<<endl;


    }
    return 0;
}
