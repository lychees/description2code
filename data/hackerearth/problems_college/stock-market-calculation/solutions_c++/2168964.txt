#include<bits/stdc++.h>
int main(){
	long long int t,i,days,arr[100001],result,max;
	scanf("%lld",&t);
	while(t--){
		scanf("%lld",&days);
		for(i=0;i<days;i++){
			scanf("%lld",&arr[i]);	
		max=arr[i];
		}
		result=0;
		for(i=days-2;i>=0;i--){
			if(arr[i]>=max)
				max=arr[i];
			else if(arr[i]<max)
				result = result + max - arr[i];
		}
		printf("%lld\n",result);
	}
}
