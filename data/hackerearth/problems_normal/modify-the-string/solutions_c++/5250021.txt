#include <iostream>
using namespace std;

int main()
{
    char name[100];
    
    int i = 0;
    gets(name);
    while (name[i] != '\0')
     {
         if (name[i]>=65 && name[i]<=90)        // if uppercase
            name[i] += 32;             // change to lowercase
         else
         if (name[i] >=97 && name[i] <= 122)// if lowercase
             name[i] -=32;             // change to lowercase
             ++i;
     }
     puts(name);    
    return 0;
}
