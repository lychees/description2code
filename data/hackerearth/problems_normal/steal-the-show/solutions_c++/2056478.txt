#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long a,b,c,d,i,j,ar[10000];
    scanf("%lld %lld %lld",&a,&b,&c);
    scanf("%lld",&d);
    for(i=0;i<d;i++){
    	scanf("%lld",&ar[i]);
    }
    sort(ar,ar+d);
    if(b>c){
    	j=c;c=b;b=j;
    }
    j=0;
    if(d==0){
    	j=abs(c-a)+abs(a-b);
    	cout<<j<<endl;
    	return 0;
    }
    if(d==1){
    	j=abs(ar[0]-b)+abs(a-ar[0])+abs(c-a);
    	if(j>(abs(a-b)+abs(ar[0]-a)+abs(c-ar[0]))){
    		j=(abs(a-b)+abs(ar[0]-a)+abs(c-ar[0]));
    	}
    	cout<<j<<endl;
    	return 0;
    }
    j=ar[d-1]-ar[0];
    j+=abs(c-ar[d-1]);
    j+=abs(ar[0]-b);
    
    if(d%2==0){
    	i=d/2;
    	j+=abs(ar[i]-a)+abs(a-ar[i-1])-(ar[i]-ar[i-1]);
    }else{
    	i=d/2;
    	b=abs(ar[i]-a)+abs(a-ar[i-1])-(ar[i]-ar[i-1]);
    	if(b>(abs(ar[i+1]-a)+abs(a-ar[i])-(ar[i+1]-ar[i]))){
    		b=abs(ar[i+1]-a)+abs(a-ar[i])-(ar[i+1]-ar[i]);
    	}
    	j+=b;
    }
    cout<<j<<endl;
    return 0;
}
