#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int i;
    	vector<int>v;
    	int num;
    	for(i=1;i<=n;i++)
    	{
    	 cin>>num;
    	 v.push_back(num);
    	}
        sort(v.begin(),v.end());
        vector<int>a;
        for(i=0;i<n;i++)
        {
        	a.push_back(v[i]+v[n-i-1]);
        }
        sort(a.begin(),a.end());
        cout<<a[n-1]-a[0]<<"\n";
    }
    //cout << "Hello World!" << endl;
    return 0;
}
