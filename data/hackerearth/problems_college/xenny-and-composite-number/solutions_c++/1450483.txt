#include <iostream>
#include<math.h>
using namespace std;
int prime(int n)
{
	int sqr=(int)sqrt(n);
	int count=0;
	for(int i=2;i<=sqr;i++)
	{
		if(n%i==0)
		count=1;
	}
	if(count==1)
	return 0;
	else
	return 1;

}
int main()
{
	int T;
	cin>>T;
	int n;

	for(int i=0;i<T;i++)
	{
		cin>>n;
		if(n==0||n==1)
		cout<<"4"<<"\n";
		else
		{

				if(n%2==0)
				{
				if(prime(n+1))
				{
					cout<<n+2<<"\n";;
				}else
				cout<<n+1<<"\n";;
			
				}else
				{
            		if(prime(n+1))
					{
						cout<<n+2<<"\n";;
					}
					else
					cout<<n+1<<"\n";
				}
		}
	}
    return 0;
}
