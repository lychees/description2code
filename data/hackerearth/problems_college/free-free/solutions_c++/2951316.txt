#include <iostream>
#include<stdio.h>
#include<ctype.h>
using namespace std;

int main()
{
    int t,i,sum;
    char s[200];
    cin>>t;
    while(t--)
    {
    	sum=0;
    	int count[26]={0},countcaps[26]={0};
    	cin>>s;
    	for(i=0;s[i];i++)
    	{
    		if(islower(s[i]))
    		count[s[i]-'a']++;
    		else
    		countcaps[s[i]-'A']++;
    	}	
    	for(i=0;i<26 ;i++)    
    	{
    		if(count[i]%2==0 &&count[i])
    		  sum+=(count[i]/2);
    		else if(count[i])
    		 sum+=((count[i]/2)+1);
    		if(countcaps[i]%2==0 &&countcaps[i])
    		  sum+=(countcaps[i]/2);
    		else if(countcaps[i])
    		 sum+=((countcaps[i]/2)+1); 
    	}
    	
    	
    	cout<<sum<<endl;
    }
    return 0;
}
