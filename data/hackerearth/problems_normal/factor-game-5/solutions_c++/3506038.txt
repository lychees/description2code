#include <iostream>
#include <cstring>
#include <cmath>
using namespace std;

int primeFactors(int n)
{
    int cnt=0;
    while (n%2 == 0)
    {
        cnt++;
        n = n/2;
    }
 
    for (int i = 3; i <= sqrt(n); i = i+2)
    {
        while (n%i == 0)
        {
            cnt++;
            n = n/i;
        }
    }
 
    if (n > 2)
        cnt++;
    return cnt;
}

int main()
{
    int n;
    cin>>n;
    int xr=0;
    for(int i=0;i<n;i++)
    {
    	int x;
    	cin>>x;
    	xr^=primeFactors(x);
    }
if(xr==0) cout<<"BOB\n";
else cout<<"ALICE\n";
return 0;
}
