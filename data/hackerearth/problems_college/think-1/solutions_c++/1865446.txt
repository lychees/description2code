#include<bits/stdc++.h>
using namespace std;
int main()
{ int t,x,d,s,c,v;
  float y;
  cin>>t;
  while(t--)
  { s=0;
    cin>>x;
    c=x;
    x=x*x;
    while(x>0)
    {
      d=x%10;
      s=s*10+d;
      x=x/10;
    }
    y=sqrt(s);
    //cout<<y<<"\n";
    v=0;
    int u=c;
    while(c>0)
    {
      d=c%10;
      v=v*10+d;
      c=c/10;
    }
    if(v==y&&v!=u&&ceil(y)==floor(y)&&u%10!=0)
    {
        cout<<y<<"\n";
    }
    else
    {
        cout<<"No\n";
    }
  }

return 0;
}
