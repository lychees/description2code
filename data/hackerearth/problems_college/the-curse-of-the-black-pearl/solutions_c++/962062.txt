#include <iostream>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
    string s,s1;
    long l,tim,c,i,x;
    int t = 882;
    
    while(t--)
    {
    	tim = 0;
    	cin>>s;
    	l = s.length();
    	s1 = s;
    	for(i=l-1;i>0;i--)
    	{
    		if(s[i]=='0')
    		{
    			tim++;
    		}
    		else
    		{
    			while(s[i]=='1' && i>=0)
    			{
    				s[i] = '0';
    				i--;
    				tim++;
    			}
    			if(i>=0)s[i]='1';
    			tim++;
    			i++;
    		}
    	}

    	cout<<tim<<"\n";
    }
    return 0;
}
