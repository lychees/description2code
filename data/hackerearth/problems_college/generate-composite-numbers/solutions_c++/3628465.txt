#include <iostream>
using namespace std;

int main()
{
    int n;
    int a[10000]={0};
    cin>>n;
    for(int i=2;i<=100;i++){
    	int j=i+i;
    	while(j<=n){
    		a[j]=1;
    		j+=i;
    	}
    }
    //cin>>n;
    for(int k=1;k<=n;k++){
    	if(a[k]==1)
    	cout<<k<<" ";
    }
    return 0;
}
