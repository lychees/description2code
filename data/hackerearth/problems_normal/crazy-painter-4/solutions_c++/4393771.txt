#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long int n,freq,j;
        cin>>n;
        n=n*12;
        freq=n/26;
        j=n%26;
        vector <long long int> a(27),f(27);
        for(int i=1;i<=26;i++)
        {
            cin>>a[i];
        }
        for(int i=1;i<=26;i++)
        {
            f[i]=freq;
        }
        for(int i=1;i<=j;i++)
        {
            f[i]++;
        }
        long long int sum=0;
        for(int i=1;i<=26;i++)
        {
            sum+=(a[i]*f[i])+(((f[i])*(f[i]-1))/2);
        }
        cout<<sum<<endl;
    }
}
