#include <bits/stdc++.h>
#define M 1000000007
using namespace std;

int dp[10005][15];

int main()
{
	int t, n, p;
	cin >> t;
	while (t--) {
		cin >> n >> p;
		memset(dp, 0,sizeof(dp));
		for (int i = 1; i <= n; i++) {
			dp[i][0] = 1;
		}
		
		for (int i = 2; i <= n; i++) {
			int w = 0;
			for (int j = 0; j < p; j++) {    //insert y
				w += dp[i - 1][j];
				w %= M;
			}
			dp[i][0] = w;
			for (int j = 1; j < p; j++) {  //insert x
				dp[i][j] = dp[i-1][j-1];
			}
		}
		
		int w = 0;
		for (int i = 0; i < p; i++) {
			w += dp[n][i];
			w %= M;
		}
		
		cout << w << endl;
	}
	return 0;
}
