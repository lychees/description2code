for t in range(0, int(raw_input())):
	even = 0
	odd = 1
	step = 0
	n = int(raw_input())
	while n != 1:
		if n%2 == 0:
			even += n
			n = n/2
		else:
			odd += n
			n = 3*n + 1
		step += 1

	even, odd = even%step, odd%step
	
	if odd > even:
		print "Odd Rules"
	elif even > odd:
		print "Even Rules"
	else:
		print "Tie"
