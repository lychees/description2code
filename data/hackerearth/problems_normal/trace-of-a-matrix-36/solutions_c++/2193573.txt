#include <iostream>

using namespace std;

typedef unsigned long long int Lint;

int main()
{
	ios_base::sync_with_stdio( false );
	Lint t, r, c, v1, v2, mini, v3;
	
	cin >> t;
	
	while( t-- ) {
		
		cin >> r >> c;
		mini = min( r, c );
		
		v1 = ( ( mini ) * ( mini - 1 ) ) / 2;
		v2 = v1 * ( r + c + 2 );
		v3 = v2 + ( mini * 2 );
		cout << v3 << endl;
	}
	return 0;
}
