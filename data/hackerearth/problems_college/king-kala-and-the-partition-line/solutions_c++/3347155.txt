#include<bits/stdc++.h>
using namespace std;
#define lld long long int
lld MAX(lld a,lld b)
{
    if(a>b)return a;
    return b;
}
int main()
{
    
    int T;
    cin>>T;
    while(T--)
    {
        int N;
        cin>>N;
        int A,B,C;
        cin>>A>>B>>C;
        int count1=0,count2=0;
        for(int i=0;i<N;i++)
        {
            int X,Y;
            cin>>X>>Y;
            int T=A*X+B*Y+C;
            if(T>0)count1++;
            else if(T<0)count2++;
        }
        cout<<MAX(count2,count1)<<endl;
    }
    
    return 0;
}
