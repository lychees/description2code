#include<cstdio>
#include<vector>
#include<cmath>
int z, i, j, n, q, pom;
bool w[1000001], w2[1000001];
std::vector<int>v;

int main()
{
  scanf("%d", &z);
  
  v.push_back(1);
  
  for(q = 2; q <= 30; q++)
  {
    for(i = 2; pow(i, q) <= 1000000; i++)
    {
      pom = pow(i, q);
      if (w[pom] == 0)
      {
	w[pom] = 1;
	v.push_back(pom);
      }
    }
  }
  
  for(i = 0; i < v.size(); i++)
  {
    for(j = 0; j < v.size(); j++)
    {
      if (v[i] + v[j] <= 1000000)
      {
	w2[v[i] + v[j]] = 1;
      }
    }
  }
  
  while(z--)
  {
    scanf("%d", &n);
    if (w2[n] == 0){printf("No\n");}else{printf("Yes\n");}    
  }  
  
return 0;
}
