A = ord("a")
for _ in range(int(raw_input())):
    p = 26 * [0]
    t = raw_input().strip()
    lp = len(t)
    for i in t:
        p[ord(i)-A] += 1
    c = 26 * [0]
    t = map(ord, raw_input().strip())
    lt = len(t)
    t += [A]
    for i in t[:lp]:
        c[i-A] += 1
    ans = "NO"  
    for i in range(lt-lp+1):
        if c == p:
            ans = "YES"
            break
        c[t[i]-A] -= 1
        c[t[i+lp]-A] += 1
    print ans
