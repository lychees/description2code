#include <bits/stdc++.h>
using namespace std;
 
typedef long long LL;
 
#define PII       pair<int,int>
#define all(c)    c.begin(),c.end()
#define sz(c)     (int)c.size()
#define clr(c)    c.clear()
#define pb        push_back
#define mp        make_pair
#define cin(x)    scanf("%d",&x)
#define MOD       1000000000
#define EPS       1E-10



int main()
{
	int t;
	cin >> t;
	for(int ii = 1; ii <= t; ii++)
	{
		LL x , k;
		cin >> x >> k;
		int arr[100] = {0};
		int i = 0;
		while(x)
		{
			arr[i] = x%2;
			x /= 2;
			i += 1;
		}
		unsigned long long ans = 0;
		i = 0;
		unsigned long long f = 1;
		while(i < 100 && k)
		{
			if(arr[i])
			{
				f *= 2;
				i += 1;
				continue;
			}
			if(k%2)
			{
				ans += f;
				k /= 2;
				i += 1;
			}
			else
			{
				k /= 2;
				i += 1;
			}
			f *= 2;
			assert(i <= 65);
		}
		cout << "Case #" << ii << ": " << ans << "\n";
	}
	return 0;
}
