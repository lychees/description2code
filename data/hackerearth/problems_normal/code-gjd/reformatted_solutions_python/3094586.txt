for _ in range(int(raw_input())):
    n = int(raw_input())
    a = map(int, raw_input().split())
    a = sorted(a, reverse=True)
    if n == 1:
        print 0
        continue
    if n == 2:
        print a[0]+a[1]
        continue
    s = (a[0]+a[1])*(n-1)
    for i in range(2, n):
        s += a[i]*(n-i)
    print s
