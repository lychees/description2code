#include <iostream>
using namespace std;

int main() {
	int a[1000001]={0};
	for(int i=2;i<=1000;i++)
	{
		if(a[i]==0)
		{
			for(int j=i*i;j<=1000000;j+=i)
			a[j]=1;
		}
	}
long	int b[1000001]={0};
	b[2]=2;
	for(int i=3;i<=1000000;i++)
	{
		if(a[i]==0)
		{
			b[i]=b[i-1]+i;
		}
		else
		b[i]=b[i-1];
	}
	
	int t;
	cin>>t;
	while(t--)
	{
		int x,y;
		cin>>x>>y;
		cout<<b[y]-b[x-1]<<endl;
		
	}
	return 0;
}
