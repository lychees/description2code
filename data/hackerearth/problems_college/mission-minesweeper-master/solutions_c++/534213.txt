#include<bits/stdc++.h>
using namespace std;
int vis[100][100],arr[100][100],n,m,mat1[100][100],mat[100][100];
int ch(int i,int j)
{
    if(i>=0&&i<n&&j>=0&&j<m)
        return 1;
    else
        return 0;
}
void rec(int i,int j)
{
    if(vis[i][j]==1)
        return;
    vis[i][j]=1;
    mat1[i][j]=mat[i][j];
    if(ch(i-1,j))
    mat1[i-1][j]=mat[i-1][j];
    if(ch(i-1,j+1))
    mat1[i-1][j+1]=mat[i-1][j+1];
    if(ch(i-1,j-1))
    mat1[i-1][j-1]=mat[i-1][j-1];
    if(ch(i,j+1))
    mat1[i][j+1]=mat[i][j+1];
    if(ch(i,j-1))
    mat1[i][j-1]=mat[i][j-1];
    if(ch(i+1,j+1))
    mat1[i+1][j+1]=mat[i+1][j+1];
    if(ch(i+1,j))
    mat1[i+1][j]=mat[i+1][j];
    if(ch(i+1,j-1))
    mat1[i+1][j-1]=mat[i+1][j-1];
    if(ch(i-1,j)&&mat[i-1][j]==0)
        rec(i-1,j);
    if(ch(i-1,j+1)&&mat[i-1][j+1]==0)
        rec(i-1,j+1);
    if(ch(i-1,j-1)&&mat[i-1][j-1]==0)
        rec(i-1,j-1);
    if(ch(i,j-1)&&mat[i][j-1]==0)
        rec(i,j-1);
    if(ch(i,j+1)&&mat[i][j+1]==0)
        rec(i,j+1);
    if(ch(i+1,j)&&mat[i+1][j]==0)
        rec(i+1,j);
    if(ch(i+1,j-1)&&mat[i+1][j-1]==0)
        rec(i+1,j-1);
    if(ch(i+1,j+1)&&mat[i+1][j+1]==0)
        rec(i+1,j+1);

}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    int r;
    cin>>n>>m;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            mat1[i][j]=-2;
            vis[i][j]=0;
        }
    }
    int indx,indy;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            char temp;
            cin>>temp;
            if(temp=='*')
            {
                arr[i][j]=1;
            }
            else if(temp=='c')
            {

                indx=i;
                indy=j;
                arr[i][j]=0;
            }
            else
                arr[i][j]=0;
        }
    }

    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            if(arr[i][j]==0)
            {
                int cnt=0;
                if(ch(i-1,j)&&arr[i-1][j]==1)
                    cnt++;
                if(ch(i-1,j-1)&&arr[i-1][j-1]==1)
                    cnt++;
                if(ch(i-1,j+1)&&arr[i-1][j+1]==1)
                    cnt++;
                if(ch(i+1,j)&&arr[i+1][j]==1)
                    cnt++;
                if(ch(i+1,j-1)&&arr[i+1][j-1]==1)
                    cnt++;
                if(ch(i+1,j+1)&&arr[i+1][j+1]==1)
                    cnt++;
                if(ch(i,j-1)&&arr[i][j-1]==1)
                    cnt++;
                if(ch(i,j+1)&&arr[i][j+1]==1)
                    cnt++;
                mat[i][j]=cnt;
            }
            else
                mat[i][j]=-1;
        }
    }
    if(mat[indx][indy]==0)
    rec(indx,indy);
    else
	mat1[indx][indy]=mat[indx][indy];
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            if(mat1[i][j]!=-2)
            {
                if(mat1[i][j]==-1)
                    cout<<"*";
                else
                    cout<<mat1[i][j];
            }
            else
            {
                if(arr[i][j]==1)
                    cout<<"*";
                else
                    cout<<".";
            }

        }
        cout<<endl;
    }
    }
}
