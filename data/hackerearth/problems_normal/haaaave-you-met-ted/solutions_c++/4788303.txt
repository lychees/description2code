#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long n;
		cin>>n;
		int a,min=100,c=0;;
		for(int i=0;i<n;i++)
		 {
		 	cin>>a;
		 	c=0;
		 	while(a)
		 	{
		 		a=a&(a-1);
		 		c++;
		 		
		 	}
		 	if(c<min)
		 		min=c;
		 }
		
		 cout<<min<<"\n";
		
	}
	return 0;
}
