#include <iostream>
using namespace std;

int main()
{
	int test;
    cin>>test;
    while(test>0)
    {
    	int a,n,arr[100][10001],counting=0,i,j;
    	for(i=0;i<100;i++)
    	{
    		for(j=0;j<10001;j++)
    		{
    			arr[i][j]=0;
    		}
    	}
    	cin>>n;
    	cin>>a;
    	
    	arr[0][a]++;
    	arr[0][0]++;
    	
			for(i=1;i<n;i++)
			{
				cin>>a;
				for(j=0;j<10001;j++)
				{	
					if(arr[i-1][j]>0)
					{
					
						arr[i][j]++;
						arr[i][j+a]++;

					}
				}
			}
			for(i=0;i<10001;i++)
			{
				if(arr[n-1][i] > 0)
				{
				
					counting++;
				}
			}
		
			
    	
    	cout<<counting<<endl;
    	test--;
    }
    return 0;
}
