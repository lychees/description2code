#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,l,r,q,t,nn;
    cin>>n>>l>>r>>q;nn=r-l+1;
    vector<int>v(n),vv(nn);
    vector<int>::iterator it;
    for(int i=0;i<n;i++)cin>>v[i];
    for(int i=0;i<nn;i++)vv[i]=v[i+l];
    sort(vv.begin(),vv.end());
    while(q--){
    	cin>>t;
    	it=upper_bound(vv.begin(),vv.end(),t);
    	if(it==vv.end())cout<<-1<<endl;
    	else cout<<*it<<endl;
    }
    return 0;
}
