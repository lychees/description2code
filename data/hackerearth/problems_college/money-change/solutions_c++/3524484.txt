#include <bits/stdc++.h>
using namespace std ; 
#define ll long long int 
int main()
{
	 ll m  ; 
	 cin >> m ; 
	 ll n ; 
	 cin >>  n ; 
	 ll i ;
	 ll a[100000];  
	 for ( i = 0; i< n ; i++)
	 {
	 	 cin >> a[i] ; 
	 }
	 sort ( a , a+n) ; 
	 ll dp[n+1][m+1] ; 
	 for ( i = 0; i<= m ; i++)
	 {
	 	dp[0][i] = 1 ; 
	 }
	 for ( i = 0 ; i< n ; i++)
	 {
	 	dp[i][0] = 1 ; 
	 }
	 ll j ;
	
	  
	 for ( i = 1 ; i< n ; i++)
	 {
	 	  for ( j = 1 ; j<= m ; j++)
	 	  {
	 	  	    if ( j < a[i])
	 	  	    {
	 	  	        dp[i][j] = dp[i-1][j] ; 	
				}
				else
				{
					dp[i][j] = dp[i-1][j] + dp[i][j-a[i]] ; 
 				}
		   }
	 	
	 }
	cout << dp[n-1][m] <<endl; 
	
	return 0 ; 
}
