#include<bits/stdc++.h>
using namespace std;
#define FOR(i,a,b) for(i= a ; i < b ; ++i)
#define rep(i,n) FOR(i,0,n)
#define pb push_back
#define mp make_pair
#define pln(n) printf("%lld\n",n)
#define sl(n) scanf("%lld",&n)
#define F first
#define S second
#define mod (ll)(1e9 + 7)
#define ll long long int
#define MAX 1000006
ll modpow(ll a,ll n,ll temp){ll res=1,y=a;while(n>0){if(n&1)res=(res*y)%temp;y=(y*y)%temp;n/=2;}return res%temp;} 
vector<pair<ll,ll> > arr;
int main()
{
	ll d,n,c,v1,v2,w,wit,cit,i,j,k,calc,ans,cnt;
	sl(d);
	rep(k,d)
	{
		arr.clear();
		sl(n);
		rep(i,n)
		{
			sl(v1); sl(v2);
			arr.pb(mp(v1,v2));
		}
		sl(w); sl(c);
		calc=1<<n;
		ans=-1;
		FOR(i,1,calc)
		{
			cnt=0;
			wit=0; cit=0;
			rep(j,n)
			{
				if(i&(1<<j))
				{
					cnt++;
					cit+=arr[j].F;
					wit+=arr[j].S;
				}
			}
			if(cnt<=c && wit<=w)
				ans=max(ans, cit);
		}
		printf("For Day #%lld:\n",k+1);
		pln(ans);
	}
	return 0;
}
