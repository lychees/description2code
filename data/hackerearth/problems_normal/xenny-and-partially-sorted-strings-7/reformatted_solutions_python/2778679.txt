test = input()
for j in range(test):
    a = {}
    n, k, m = map(int, raw_input().split())
    for i in range(n):
        inp = raw_input()
        a[inp] = inp

    for i in a:
        a[i] = a[i][0:m]

    print sorted(a, key=a.get)[k-1]
