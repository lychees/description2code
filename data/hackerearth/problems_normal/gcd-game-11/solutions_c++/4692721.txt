#include <bits/stdc++.h>
using namespace std;
int dp[1008][1008][2];

int rec(int i,int j,int turn){
	int &res = dp[i][j][turn-1];
	if(res != -1)
		return res;

	if(i == 1 && j == 1)
		return 0;
	if(i == 1)
		return 2;
	if(j == 1)
		return 1;

	int gcd =  __gcd(i,j);
	if(turn == 1){
		int r1 = rec(i,j-1,2);
		int r2 = (gcd == 1)?-1:rec(i,j/gcd,2);
		if(r1 == 1 || r2 == 1)
			res = 1;
		else res = 2;
	}
	else{
		int r1 = rec(i-1,j,1);
		int r2 = (gcd == 1)?-1:rec(i/gcd,j,1);
		if(r1 == 2 || r2 == 2)
			res = 2;
		else res = 1;
	}
	return res;
}
int main(int argc, char const *argv[])
{
	int T;
	memset(dp,-1,sizeof dp);

	scanf("%d",&T);
	while(T--){
		int a,b;
		scanf("%d%d",&a,&b);
		int ans = rec(a,b,1);
		//cout<<ans<<endl;
		switch(ans){
		case 1:
			printf("Arjit\n");
			break;
		case 2:
			printf("Chandu Don\n");
			break;
		case 0:
			printf("Draw\n");
		}
	}
	return 0;
}
