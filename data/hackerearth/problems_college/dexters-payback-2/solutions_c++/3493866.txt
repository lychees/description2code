#include <bits/stdc++.h>
#define ll long long int

using namespace std;

bool check(int i, int j) {
    if ((i % 4 == 0 && j % 4 == 0) || (i % 4 == 0 && j % 4 == 1) || (i % 4 == 1 && j % 4 == 0) || (i % 4 == 1 && j % 4 == 1)
        || (i % 4 == 2 && j % 4 == 2) || (i % 4 == 2 && j % 4 == 3) || (i % 4 == 3 && j % 4 == 2) || (i % 4 == 3 && j % 4 == 3)) {
            return true;
        }
        return false;
}
int main()
{
    int t;
    int w, h;

    cin >> t;

    while (t--) {
        cin >> w >> h;
        int cnt = 0;
        for (int i = 0; i < w; i++) {
            for (int j = 0; j < h; j++) {
                if (check(i, j)) {
                    cnt++;
                }
            }
        }
        cout << cnt << endl;
    }
    return 0;
}
