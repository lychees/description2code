#include <iostream>
#include <math.h>
using namespace std;

int solve (int s) {
	if (s == 1) {
		return 1;
	}
	int temp = sqrt(s);
	int sum = 0;
	while (temp > 0) {
		sum = sum + (s / temp) - (temp - 1);
		temp = temp - 1;
	}
	return sum;
}

int main()
{
	int t;
    cin >> t;
    for (int i = 0; i < t; i++) {
    	int s;
    	cin >> s;
    	cout << solve(s) << endl;
    }
    return 0;
}
