#include <iostream>
#include<string.h>
#include<bits/stdc++.h>

using namespace std;
typedef long long ll; 
ll ma = 1000000007;

char s[100005],ta[100005];
ll a[27][100005], b[27][100005];
ll p[100005];
int xx[26],yy[26];
int main()
{
   int t;
   int i,j,k;
   scanf("%d",&t);
   p[0] = 1;
   for(i=1;i<=100001;i++)
  {
  	p[i] = p[i-1]*2;
  	if(p[i] > ma)p[i]-=ma;
  }
   while(t--)
   {
   	int n;
   	scanf("%s",&s);
   	int q;
   	n = strlen(s);
   	scanf("%s",&ta);
   //	printf("gave %s of len %d\n",s,n);
   	scanf("%d",&q);
   	for(i=0;i<26;i++)
   	for(j=0;j<=n+1;j++)
   	a[i][j] = b[i][j] = 0;
   	for(i=0;i<n;i++)
   	{
   		int ch1,ch2;
   		ch1 = s[i] - 'a';
   		ch2 = ta[i] - 'a';
   		a[ch1][i+1] = 1;
   		b[ch2][i+1] = 1;
   		
   	}
   	for(i=0;i<26;i++)
   	{
   		for(j=1;j<=n;j++)
   		{
   			a[i][j] = a[i][j-1]*2 + a[i][j] ;
   				b[i][j] = b[i][j-1]*2 + b[i][j];
   				if(a[i][j] > ma)a[i][j]-=ma;
   				if(b[i][j] > ma)b[i][j]-=ma;
   		}
   	}
   	while(q--)
   	{
   		scanf("%d%d%d",&i,&j,&k);
   	//	printf("for query %d %d %d\n",i,j,k);
   		int found ;
   		int x,y,z;
   		long po = p[k];
   	//	printf("shift of %lld\n",po);
   		ll fir,sec;
   		for(x=0;x<26;x++)
   		{
   		//	break;
   			fir = a[x][i-1];
   			sec = a[x][i+k-1];
   			ll u,v;
   			u = fir;
   			v = sec;
   			fir = (fir * po)%ma;
   			
   			sec = (sec + ma - fir)%ma;
   			
   		xx[x] = sec;
   		//	printf("in first %c has val %lld  %lld  %lld\n",x+'a', sec,u,v);
   				fir = b[x][j-1];
   			sec = b[x][j+k-1];
   			u = fir;
   			v  = sec;
   			
   			fir = (fir * po)%ma;
   			sec = (sec + ma - fir)%ma;
   		yy[x] = sec;
   	   		//	printf("in second %c has val %lld  %lld  %lld\n",x+'a', sec,u,v);
   		}
   		sort(xx,xx+26);
   		sort(yy,yy+26);
   		
   		int valid = 1;
   		for(i=0;i<26;i++)
   		if(xx[i]!=yy[i]){
   			valid = 0;
   			break;
   		}
   		if(valid)printf("yes\n");
   		else printf("no\n");
   		
   	}
   }
    return 0;
}
