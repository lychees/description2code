#include <bits/stdc++.h>
using namespace std;
int arr[110];
//int m,n;
//long long int fa;
int main()
{
	int m, n;
	long long int sum = 0;
	scanf("%d %d", &m,&n);
	for(int i = 0; i < m; i++) {
		cin >> arr[i];
		sum+=arr[i];
	}
	
	
	sort(arr,arr+m);
	reverse(arr,arr+m);
	if(n >= m) {
		printf("%lld\n", sum);
	}
	else {
		sum = 0;
		int k = 1;
		for(int i = 0; i < m; i++) {
			sum += k*arr[i];
			if((i+1)%n == 0 && i!= 0) {
				k++;
			}
		}
		printf("%lld\n", sum);
	}
    return 0;
}
