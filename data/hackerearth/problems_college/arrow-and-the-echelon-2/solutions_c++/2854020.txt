#include <iostream>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,i,j,x,y;
    	cin>>n;
    	x=n;
    	if(x%5==0) {x=n;y=0;}
    	else 
    
    	{
	    	x=n-1;
	    	for(;x>=0;x--)
	    	{
	    		if((n-x)%3==0 && x%5==0) {y=n-x;break;}
	    	}
    	}
    	if(x<0) printf("%d",-1);
    	else
    	{
    		for(i=0;i<y;i++) printf("%d",5);
    		for(i=0;i<x;i++) printf("%d",3);
    	}
    
    	cout<<endl;
    }
    return 0;
}
