#include <iostream>
#include<cmath>
typedef unsigned long long int ll;
using namespace std;
bool prime[4000001];
void init()
{
	for(ll i=0;i<4000001;i++)
		prime[i]=true;
	for(ll i=2;i<=sqrt(4000001);i++)
		for(ll j=i;i*j<4000001;j++)
			if(prime[i*j]==true)
				prime[i*j]=false;
}
int main()
{
    ll n=0;
    ll sum = 0;
    init();
    cin>>n;
    for(ll i=2;i<n;i++)
    {
    	if(prime[i])
    		sum+=i;	
    }
    cout<<sum<<endl;
    return 0;
}
