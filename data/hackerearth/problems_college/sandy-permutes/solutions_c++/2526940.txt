typedef long long ll;

void next_perm(vector<int> &v, int n, int k, int ind){
	int i=0;
	for(i=ind; i<n; i++){
		//cout<<v[i]<<endl;
		if(v[i] == k){
			
			break;
			}
			
		}
	
	
	int temp = v[ind];
	v[ind] = v[i];
	v[i] = temp;

}

int main(){

	ios_base::sync_with_stdio(false);

	int n;
	cin>>n;

	ll k;
	cin>>k;

	vector<int> v;
	int x;

	for(int i=0; i<n; i++){
		cin>>x;
		
		v.push_back(x);
	}

	
	int temp = n;
	if(k >= n){
		for(int i=n; i>0; i--){
			cout<<i<<" ";
		}
		return 0;
	}
	int j = 0;
	temp = n;
	for(int i=0; i<k; i++){
		while(j<n && v[j] == temp){
			j++;
			temp--;
		}
		//cout<<j;
		next_perm(v, n, temp, j);
	}
	for(int i=0; i<n; i++){
			cout<<v[i]<<" ";
	}
	return 0;
}
