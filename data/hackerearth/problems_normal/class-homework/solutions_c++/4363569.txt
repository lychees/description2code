#include<bits/stdc++.h>
using namespace std;
#define ll long long int
char str[1000100];
ll dp[6][30];
ll compute(ll numbersleft,ll endingpoint)
{
	if(numbersleft==0 && endingpoint==0)
		return 0;
	if(numbersleft<=0 || endingpoint<=0)
	{
		return -100000000000000;
	}
	if(dp[numbersleft][endingpoint]!=-100000000000000)
	{
		return dp[numbersleft][endingpoint];
	}
	ll i,num=0,base=1,cnt=0;
	for(i=endingpoint; i>=1; i--)
	{
		cnt++;
		num+=base*(str[i]-'0');
		//printf("%lld is num\n",num);
		if(str[i]=='0' && cnt>1)
		{
			;//do nothing
		}
		else if(num<=1000000000000)
		{
			dp[numbersleft][endingpoint]=max(dp[numbersleft][endingpoint],num+compute(numbersleft-1,i-1));
		}
		else
		{
			break;
		}
		if(i==numbersleft)
			break;
		base*=10;
	}
	return dp[numbersleft][endingpoint];
}
int main()
{
	ll n;
	scanf("%lld",&n);
	ll i;
	for(i=1; i<=n; i++)
	{
		ll j,k;
		for(j=0; j<=5; j++)
		{
			for(k=0; k<30; k++)
				dp[j][k]=-100000000000000;
		}
		scanf(" %s",str+1);
		ll len=strlen(str+1);
		ll ans=compute(4,len);
		if(ans<0)
		{
			printf("unlucky\n");
		}
		else
		{
			printf("%lld\n",ans);
		}
	}
	return 0;
}
