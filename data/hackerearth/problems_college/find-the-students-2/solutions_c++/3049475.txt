#include <bits/stdc++.h>
using namespace std;


int main()
{
    int n, k, i, t;
    cin>>n;
    int a[1010];
    int b[1010];
    for(i = 0  ; i < 1010; i++)
    	b[i] = 0;
    for(i = 0 ; i < n; i++) {
    	cin>>a[i];
    	b[a[i]]++;
    }
    sort(a, a+n);
    k = 0;
    for(i = 0  ;  n > 0;) {
   		cout<<n<<"\n";
   		n = n - b[a[i]];
   		i += b[a[i]];
    }
    return 0;
}
