void markMultiples(int arr[],int a)
{
	int i=2,num,n;
	n=100000;
	while((num=i*a)<=n)
	{
		arr[num-1]=1;
		++i;
	}
}
void SieveOfEratosthenes(int arr[])
{
	int n=100000;
    if (n >= 2)
    {
        arr[n]=0;
 
        for (int i=1; i<n; ++i)
        {
            if ( arr[i] == 0 )
            {
                markMultiples(arr, i+1);
            }
        }
    }
}
