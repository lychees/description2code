double area(double val,double a,double b) {
	return   a *( (b - val)*(b - val)) / (2*b);
} 
int main(){
	double a, b ,p;
	cin >> a >> b >> p;
	double ar = p*(a*b)/200;
	double ans = b-.01;
	double l = 0.0 , h = b;
	while(l<=h){
		double m = (l+h)/2.0;
		if(area(m,a,b)<=ar){
			h = m - 0.000001;
			ans = min(m,ans);
		}
		else {
			l = m + 0.000001;
		}		
	}
printf("%.2f",ans);
return 0;
}
