#include <iostream>
#include<math.h>
#include<iomanip>
using namespace std;

int main()
{
     int t,i,j;
    cin>>t;
    for(j=1;j<=t;j++)
    {
    	int n;
    	cin>>n;
    	double x,y,z,vx,vy,vz;
    	double sx=0,sy=0,sz=0,svx=0,svy=0,svz=0,t1,d;
    	for(i=0;i<n;i++)
    	{
    		cin>>x>>y>>z>>vx>>vy>>vz;
    		sx+=x; sy+=y;sz+=z;svx+=vx;svy+=vy;svz+=vz;
    	}
    	sx=sx/n; sy=sy/n; sz=sz/n; svx=svx/n; svy/=n;svz/=n;
    	if(svx==0&&svy==0&&svz==0)
         	t1=0;
        else         
        	t1=-(sx*svx+sy*svy+sz*svz)/(svx*svx+svy*svy+svz*svz);
    	d=sqrt(pow(sx+svx*t1,2)+pow(sy+svy*t1,2)+pow(sz+svz*t1,2));
    	cout<<fixed;
    	cout<<"Case #"<<j<<": "<<setprecision(8)<<d<<" "<<setprecision(8)<<t1<<"\n";
    
    }
    return 0;
}
