ll modpow(ll a,ll n,ll temp){ll res=1,y=a;while(n>0){if(n&1)res=(res*y)%temp;y=(y*y)%temp;n/=2;}return res%temp;} 
ll arr[1000006];
int main()
{
	ll MAX,i,t,a,n,calc;
	MAX=1000004;
	arr[0]=1;
	FOR(i,1,MAX)
	{
		arr[i]=i*arr[i-1];
		arr[i]%=mod;
	}
	sl(t);
	while(t--)
	{
		sl(n);
		sl(a);
		if(n>=MAX)
			pin(0);
		else
		{
			a%=MAX;
			calc=arr[n]*a;
			calc%=mod;
			pln(calc);
		}
	}
	return 0;
}
