#include <bits/stdc++.h>
using namespace std;

int main(){
    int t; cin >> t; char s[300];
    while(t--){
        int dig=0;
        cin >> s;
        for(int i=0; s[i]!='\0'; ++i){
            if(48<=s[i] && s[i]<=57)
                dig+=s[i]-48;
        }

        dig%=9; dig=9-dig;
        if(dig==0 || dig==9)puts("0 or 9");
        else printf("%d\n", dig);

    }
    return 0;
}
