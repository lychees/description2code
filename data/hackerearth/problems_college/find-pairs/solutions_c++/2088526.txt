#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i;
		cin>>n;
		int sum = n;
		int a[n];
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		int count =1;
		sort(a,a+n);
		for(i=0;i<n;i++)
		{
			if(a[i]==a[i+1])
			{
			count++;	
			}
			else
			{
				sum = sum + (count*(count-1))/2;
				count =1;
			}
		}
		sum = sum + (count*(count-1))/2;
		count =1;
		cout<<sum<<"\n";
	}
}
