t=int(raw_input())
while t:
    abn=map(int,raw_input().split())
    a=abn[0]
    b=abn[1]
    n=abn[2]
    disheskcal=map(int,raw_input().split())
    disheskcal.sort()
    tmp=0
    for i in xrange(0,n):
	tmp+=disheskcal[i]
	if tmp<=a and tmp>b:
	    print "Raghu Won"
	    tmp=-1
	    break
	if tmp<=b and tmp>a:
	    print "Sayan Won"
	    tmp=-1
	    break
    if tmp!=-1:
	print "Tie"
    t-=1
