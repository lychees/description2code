#include <iostream>
#include <set>
using namespace std;
int main()
{
    int t,n;
    std::cin>>t;
    while(t--)
    {
      std::cin>>n;
      int a[n];
      for(int i =0;i<n;i++)
      {
          std::cin>>a[i];
      }
       const size_t len = sizeof(a) / sizeof(a[0]);
      std::set<int> sa(a, a + len);
      std::cout << sa.size() << std::endl;
    }
}
