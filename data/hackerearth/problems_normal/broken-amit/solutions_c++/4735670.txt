#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll l[25],d[25];
ll dp[(1<<21)+5];
int n;
ll ma;
ll solve(int pos, int mask)
{
//	printf("solve %d %d\n",pos,mask);
	if(pos==n)return 0;
//	printf("max is %lld %lld\n",ma,dp[mask]);
	if(dp[mask]!=ma)return dp[mask];
	int i;
	ll cur = ma;
//	printf("mask is %d\n",mask);
	for(i=0;i<n;i++)
	{
		if(!(mask&(1<<i))){
		ll tmp = n*l[i] + (i*d[i]*(pos)*(pos+1))/2 - n*i*d[i]*pos;
	//	printf("%d at step %d will give %lld\n",i,pos,tmp);
		cur = max(cur, tmp + solve(pos+1, mask|(1<<i)));
		}
	}
	return dp[mask] = cur;
}
int main()
{
    //cout << "Hello World!" << endl;
    scanf("%d",&n);
    int i;
    for(i=0;i<n;i++)
    scanf("%lld",&l[i]);
    for(i=0;i<n;i++)
    scanf("%lld",&d[i]);
    ma = 9999999;
    ma = ma*ma;
    ma = -ma;
  //  printf("max is %lld\n",ma);
  //  memset(dp,ma,sizeof(dp));
  for(i=0;i<=(1<<21);i++)dp[i] = ma;
    //printf("%lld %lld\n",dp[0],dp[1]);
    printf("%lld\n",solve(0,0));
    return 0;
}
