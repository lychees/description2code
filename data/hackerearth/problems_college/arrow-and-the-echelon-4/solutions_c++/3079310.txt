    cin >> t;
    while(t--){
        cin >> x;
        bool b = false;
        int r = 0, l = 0;
        for(int i = 0; i * 3 <= x; i++){
            if((x - i * 3) % 5 == 0){
                l = i * 3;
                r = x - l;
                b = true;
            }
        }
        if(b){
            while(l--)cout << "5";
            while(r--)cout << "3";
        }
        else cout << -1;
        cout << endl;
    }
    return 0;
