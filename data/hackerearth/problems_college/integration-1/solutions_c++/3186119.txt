#include <iostream>

using namespace std;
int main(void)
{

    int T,i,j;
    cin>>T;
    for(i=1;i<=T;i++)
    {
        long long int n,p,u,r,s,max_degree=0;
        cin>>n>>p>>u>>r>>s;
        int temp=p;
        for(j=1;j<=n;j++)
        {
            if(temp+r<=u)
            {
                temp+=r;
            }
            else
                if(temp-s>=0)
                temp-=s;
            else
                break;
        }
        cout<<"Case #"<<i<<": "<<temp<<endl;

    }

}
