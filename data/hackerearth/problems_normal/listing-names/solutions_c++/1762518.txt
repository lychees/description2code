#include <bits/stdc++.h>
using namespace std;

string idx[100009];

map< string, int> mp;
int main()
{
	ios_base::sync_with_stdio(0); cin.tie(0);
	int t, i = 0;
	string s;
	cin >> t;
	while(t--){
		cin >> s;
		if(mp[s]){
			mp[s]++;
		}else{
			idx[i++] = s;
			mp[s]++;
		}
	}
	for(int j = 0; j < i; ++j){
		cout << idx[j] << " " << mp[idx[j]] << "\n";
	}
	
}
