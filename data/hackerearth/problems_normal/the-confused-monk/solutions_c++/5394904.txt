#include<bits/stdc++.h>
using namespace std;

#define mod 1000000007
#define W while
#define LL long long int

LL gcds(LL a,LL b)
{
	if(b==0)
		return a;
	else
		return gcds(b,a%b);
}
LL power(LL a,LL b)
{
	if(b==0)
		return 1;
	if(b==1)
        return a;
	LL temp = power(a,b/2);
	if(b%2)	
		return (((a%mod)*(temp%mod))%mod*(temp%mod))%mod;
	else 
		return ((temp%mod)*(temp%mod))%mod;
}


int main()
{
	LL n,ans,i,temp;
	cin>>n;
	LL a[n];
	cin>>a[0];
	temp=a[0];
	for(i=1;i<n;i++)
	{
		cin>>a[i];
		temp=gcds(temp,a[i]);
	}
	
	ans=1;
	for(i=0;i<n;i++)
	{
		ans=((ans%mod)*(power(a[i],temp)%mod))%mod;
	}
	cout<<ans<<endl;
	return 0;
}
