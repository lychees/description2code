#include <bits/stdc++.h>
using namespace std;

long long int *l;

int main()
{
	int t;
	cin >> t;
	while (t > 0){
		int n;
		long long int d;
		cin >> n >> d;
		l=new long long int[n];
		sort(l,l+n);
		int total = 0;
		for (int i = 0; i < n-1; i++){
			if (l[i+1] - l[i] <= d){
				total++;
				i++;
			}
		}
		cout << total << endl;
		t--;
	}
    return 0;
}
