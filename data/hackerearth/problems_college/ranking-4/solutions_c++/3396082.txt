#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		vector < pair < int , int > > a(n);
		int ranks[n];
		for(int i=0;i<n;i++){
			cin>>a[i].first;
			a[i].second=i;
		}
		sort(a.begin(),a.end());
		reverse(a.begin(),a.end());
		int cur = 1;
		ranks[a[0].second]=cur;
		for(int i=1;i<n;i++){
			if(a[i].first==a[i-1].first){
				ranks[a[i].second]=cur;
			}
			else{
				cur=i+1;
				ranks[a[i].second]=cur;
			}
		}
		for(int i=0;i<n;i++){
			cout<<ranks[i]<<" ";
		}
		cout<<endl;
	}
    return 0;
}
