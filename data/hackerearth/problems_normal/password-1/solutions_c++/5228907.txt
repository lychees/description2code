#include<iostream>
#include<map>
using namespace std;
string rev(string s)
{
	int i=0;
	int j=s.length()-1;
	for(;i<j;i++,j--)
	swap(s[i],s[j]);
	
	return s;
}
int main()
{
	int n;
	cin>>n;
	map<string,int> mp;
	while(n--)
	{
		string s;
		cin>>s;
		if(mp[s]==1)
		{
			int len=s.length();
			cout<<len<<" "<<s[len/2]<<" ";
		}
		s=rev(s);
		mp[s]=1;
	}
}
