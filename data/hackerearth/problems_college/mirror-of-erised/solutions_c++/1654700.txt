typedef long long ll;
#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <vector>
using namespace std;
ll solve( char *X, char *Y, ll n)
{
   ll L[n+1][n+1];
   ll i, j;
  
  
   for (i=0; i<=n; i++)
   {
     for (j=0; j<=n; j++)
     {
       if (i == 0 || j == 0)
         L[i][j] = 0;
  
       else if (X[i-1] == Y[j-1])
         L[i][j] = L[i-1][j-1] + 1;
  
       else
         L[i][j] = max(L[i-1][j], L[i][j-1]);
     }
   }
    
 
   return L[n][n];
}
int main()
{
	ll t,i,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		char *a=new char[n];
		char *b=new char[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0;i<n;i++)
		cin>>b[i];
		cout<<solve(a,b,n)<<endl;
	}
return 0;
}
