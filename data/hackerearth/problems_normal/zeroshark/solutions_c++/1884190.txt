#include <bits/stdc++.h>
#define ll long long int
#define s(a) scanf("%lld",&a)
#define pb push_back
#define mp make_pair
#define f first
#define sc second
#define inf 10e16

using namespace std;

ll dp[100001][2];
ll mod=1000000007;

int main()
{
    ll t,n,i,j,k,ans;
    dp[1][0]=1;
    dp[1][1]=1;
    dp[0][0]=0;
    dp[0][1]=0;
    for(i=2;i<=100000;i++) {
        dp[i][0]=dp[i-1][1];
        dp[i][1]=(dp[i-1][1]+dp[i-1][0])%mod;
    }
    s(t);
    while(t--) {
        ans=0;
        s(n);
        if(n==1 || n==2) {cout<<n-1<<endl;continue;}
        if(n==3) ans=1;
        else ans=(dp[n-3][0]+dp[n-3][1])%mod;
        ans=(ans+dp[n-2][1])%mod;
        for(i=1;i<n-2;i++) {
            ll w;
            if(i==n-3) w=dp[i][1];
            else w=dp[i][1]*(dp[n-3-i][0]+dp[n-3-i][1]);
            w%=mod;
            ans=(ans+w)%mod;
        }
        cout<<ans<<endl;
    }
    return 0;
}
