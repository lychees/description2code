using namespace std;

typedef long long LL;

string s;
LL dp[100][10][10];
LL Q = 1e9 + 7;

LL go(LL i, LL carry, LL valid)
{

    if (i == -1){
        if (carry == 0 && valid != 0){
            return 1;
        }
        return 0;
    }
    if (dp[i][carry][valid] != -1){
        return  dp[i][carry][valid];
    }
    LL ans = 0;
    for (LL j=0;j<=1;j++){
        for (LL k=0;k<=1;k++){
            for (LL l=0;l<=1;l++){
                LL t = (j&k&l) > 0 ? 1 : 0;
                LL sum = j + k + l;
                sum += carry;
                if (sum%2 == s[i] - '0'){
                    ans += go(i-1, sum/2, valid | t);
                    ans %= Q;
                }
            }
        }
    }
    return dp[i][carry][valid] = ans;
}

int main()
{
    LL test;
    cin>>test;
    while (test--){
        memset (dp, -1, sizeof(dp));
        LL n;
        s = "";
        cin>>n;
        while (n){
            s += (char)(n%2 + '0');
            n /= 2;
        }
        reverse(s.begin(), s.end());
      //  cout<<s<<"\n";
        LL len = s.size();
        cout<<go(len - 1, 0, 0)<<"\n";
    }
    return 0;
}
