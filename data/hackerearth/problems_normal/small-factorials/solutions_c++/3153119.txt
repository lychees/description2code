#include <iostream>
using namespace std;
int main()
{
int t;
cin>>t;
while(t--)
{
	int temp=0,x=0,index;
    int i,n, a[200],m=1;
	a[0]=1;
	cin>>n;
	for(i=1;i<=n;i++)
	{ 
		for(index=0;index<m;index++)
		{
			x=a[index]*i+temp;
			a[index]=x%10;
			temp=x/10; 
		} 
		
		while(temp>0)
		{
			a[index]=temp%10;
            temp=temp/10; 
            index++;
            m++;
        }
    }
    for(i=m-1;i>=0;i--)
    {
        cout<<a[i];
    }
    cout<<endl;
}
return 0;
}
