#include <iostream>
#include<string.h>
int a[1025][1025];
int b[1025];
using namespace std;

int main()
{
    int n,i,j,k,l;
    cin>>n;
   // memset(a,0,sizeof(a));
    l=1;
    for(i=1;i<=n;++i)
    l=l*2;
    for(i=2;i<=l;++i)
    {
    	for(j=1;j<i;++j)
    	cin>>a[i][j];
    }
    for(i=1;i<=l;++i)
    b[i]=i;
    while(l!=1)
    {
    	j=1;
    	for(i=1;i<=l;i+=2)
    	{
    		if(a[b[i+1]][b[i]]==1)
    		b[j]=b[i+1];
    		else
    		b[j]=b[i];
    		j++;
    	}
    	l=l/2;
    }
    cout<<b[1]<<endl;
    return 0;
}
