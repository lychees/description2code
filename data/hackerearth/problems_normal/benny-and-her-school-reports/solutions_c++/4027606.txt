#include<iostream>
#include<vector>
using namespace std;

int main()
{
    long long int numT;
    cin>>numT;
    while(numT--)
    {
        long long int N,M,X,sum=0,diff;
        vector<long long int> marks;
        cin>>N>>M>>X;
        long long int i=0;
        while(i<N)
        {
            cin>>diff;
            marks.push_back(diff);
            sum=sum+marks[i];
            i++;
        }
        long long int avg=(N+1)*X;
        diff=avg-sum;
        if(diff<0)
        {
            cout<<1<<endl;
        }
        else if(diff>M)
        {
            cout<<"Impossible"<<endl;
        }
        else
            cout<<diff<<endl;
    }
}
