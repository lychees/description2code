#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n, k, cnt=0;
    	cin>>n>>k;
    	int a[n];
    	for(int i=0; i<n; i++){
    		cin>>a[i];
    	}
    	
    	for(int i=0; i<n; i++){
    		if(a[i]<=0)
    			cnt++;
    	}
    	if(cnt>=k)
    		cout<<"NO\n";
    	else
    		cout<<"YES\n";
    }
    return 0;
}
