#!/usr/bin/python
def minMax(a, b):
   if(a-b>0):
      max=a
      min=b
   else:
      max=b
      min=a
   return max,min

T=int(input())
for t in range(0,T):
   a,b=raw_input().split()
   a=int(a)
   b=int(b)
   count=0
   while (True):
      max,min=minMax(a,b)
      if(max==1 and min==1):
         count=count+0
         break
      if(max==1):
         count=count+min-1
         break
      if(min==1):
         count=count+max-1
         break
      count=count + max/min
      r=max%min
      if (r==1):
         count=count+min-1
         break
      else:
          a=r
          b=min
   print count
