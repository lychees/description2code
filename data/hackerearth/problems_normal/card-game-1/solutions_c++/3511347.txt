#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,k,n,t;
    cin>>t;
    int a[100005],b[100005];
    string s;
    while(t--)
    {
    	cin>>n;
    	for(i=1;i<=n;i++)
    	cin>>a[i];
    	for(i=1;i<=n;i++)
    	cin>>b[i];
    	int ans=0;
    	for(i=1;i<=n;i++)
    	if(a[i]!=b[i])
    	ans++;
    	cin>>s;
    	
    	if(s=="Devu"&&(ans==0||ans==2) || n==2)
    	cout<<"Devu\n";
    	else
    	cout<<"Churu\n";
    }
    return 0;
}
