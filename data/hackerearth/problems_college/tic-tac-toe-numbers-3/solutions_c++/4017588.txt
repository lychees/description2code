#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	    long long int n;
	    cin>>n;
	    long long int a=1,b=1,c=0,sum=0;
	    while(c<=n)
        {
            c=a+b;
           
            if(c>n)break;
            sum=sum+c;
            a=b;
            b=c;

        }
        
        long long int s,l;
        s=(n*(n+1))/2;
        l=s-sum-1-n;
        

        if(l%2==0)cout<<"Tac"<<endl;
        else
        {int c=0;
            for(int i=2;i<=sqrt(l);i++)
            {
                if(l%i==0)
                {
                    c++;
                    break;
                }
            }
            if(c==0)cout<<"Tic"<<endl;
            else cout<<"Toe"<<endl;
        }

	}
	return 0;
}
