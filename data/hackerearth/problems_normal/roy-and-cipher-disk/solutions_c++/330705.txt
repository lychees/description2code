#include <iostream>
#include <string>
using namespace std;

int main()
{
    int n;
    cin >> n;
    string input[n];
    int i=0;

    while(n > 0)
    {
    	cin >> input[i];
    	i++;
    	n--;
    }

    n = i;
    int firstTime;
    i = 0;
    while(n > 0)
    {
    	int firstTime = 1;
    	int length = input[i].length();
    	int k  = 0;
    	int finalValue;
    	while(k < length)
    	{
    		if(firstTime)
    		{
    			firstTime = 0;
    			finalValue = input[i].at(k) - 97;
    		}
    		else
    		{
    			finalValue = input[i].at(k) - input[i].at(k-1);
    		}

    		if(finalValue > 13)
    			cout << finalValue - 26 << " ";
    		else if(finalValue < -13)
    			cout << finalValue + 26 << " ";
    		else if(finalValue == 13 || finalValue == -13)
    			cout << "13 ";
    		else
    			cout << finalValue << " ";

    		k++;
    	}
    	cout << endl;
    	i++;
        n--;
    }

    return 0;
}
