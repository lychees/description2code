using namespace std;


int main()
{
    long long int t;
    cin>>t;
    while (t--) {
          long long int n;
            cin>>n;
            int ct1 = 0;
            int st = 1;
            while (n%2 == 0)
            {
                ct1++;
                n = n/2;
            }
     
            for (int i = 3; i <= sqrt(n); i = i+2)
            {
        
                int ct2 = 0;
                while (n%i == 0)
                {
                    ct2++;
                    n = n/i;
                }
                st = st * (ct2 + 1);
            }
            if (n > 2) {
                    st = st *2;
            }
    
            cout<<ct1 * st << endl;
              
          
    }

   // getch();

    return 0;
}
