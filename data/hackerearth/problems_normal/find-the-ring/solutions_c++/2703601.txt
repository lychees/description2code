#include <iostream>
using namespace std;

int solution_1(int n) {
	if(n&0x1) return 0;
	else return 1;
}

int main()
{
    int T;
    cin >> T;
    while(T--) {
    	int i,n;
    	cin >> i >> n;
    	if(i==1) cout << solution_1(n) << endl;
    	else if(i==0 || i==2) {
    		if(n==0) cout << i << endl;
    		else cout << solution_1(n-1) << endl;
    	}	
    }
}
