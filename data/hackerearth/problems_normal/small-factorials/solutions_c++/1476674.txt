#include<stdio.h>
 
int main()
{
    int a[200],temp,n,x,i,index,t;
    scanf("%d",&t);
    while(t--)
    {
        a[0]=1;
    for(i=1;i<200;i++)
    {
        a[i]=0;
    }
    scanf("%d",&n);
    int r=1;
    for(i=1;i<=n;i++)
    {
        temp=0;
        for(index=0;index<r;index++)
        {
            x=i*a[index]+temp;
            a[index]=x%10;
            temp=x/10;
        }
        while(temp>0)
        {
            a[index++]=temp%10;
            temp/=10;
            r++;
        }
 
    }
    for(i=r-1;i>=0;i--)
        printf("%d",a[i]);
     printf("\n");
    }
    return 0;
 
}
