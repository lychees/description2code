#include <stdio.h>

#define ll unsigned long long
 
ll gcd(ll num1, ll num2){
    if(num2== 0){
        return num1;
    }
 return gcd(num2, num1%num2);
}
 
int main(){
 
int t,p,l,i;
ll c,sq,g,on;
scanf("%d",&t);
 
while(t--)
{
	scanf("%d%d",&l,&p);
	char s[2*l];
	scanf("%s",s);
	c=on=0;
	
	p=p+1;
	for(i=0;i<p;i++)
	{
		if(s[i]=='1')
			on++;
	}
	
	c=on*on;
	
	for(i=p;i<l;i++)
	{
		if(s[i-p]=='1')
			on-=1;
		if(s[i]=='1')
		{
			on+=1;
			c+=(2*on)-1;
		}
	}
	
	if(c>0)
	{
	sq=((ll)l)*((ll)l);
	g=gcd(c,sq);
	c=c/g;
	sq=sq/g;
	}
	else
	sq=1;
	printf("%lld/%lld\n",c,sq);
}
return 0;
}
