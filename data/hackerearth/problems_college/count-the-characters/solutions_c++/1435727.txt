#include <iostream>
#include<cstring>
using namespace std;
int a[1000];

int main()
{
    int n,i;
    cin>>n;
    for(i=1;i<=n;i++)
    {
    	string s;
    	cin>>s;
    	for(int j=0;j<s.length();j++)
    	if(a[s[j]]==i-1)
    	a[s[j]]=i;
    }
    int c=0;
    for(i=0;i<1000;i++)
    if(a[i]==n)
    c++;
    cout<<c<<endl;
    return 0;
}
