#include <bits/stdc++.h>
using namespace std;
int A[251];
long int DP[251][251];
long int R(int C,int I) {
	if(C == 0)
		return  1;
	if(I < 0 or C < 0)
		return 0;
	if(DP[C][I] != -1)
		return DP[C][I];
	return DP[C][I] = R(C-A[I],I) + R(C,I-1);
}
int main()
{
    int M,n;
    cin>>M>>n;
    for(int i=0;i<n;i++) {
    	cin>>A[i];
    }
    memset(DP,-1,sizeof(DP));
    cout<<R(M,n-1)<<endl;
    
    return 0;
}
