'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
t = int(raw_input())
dic = {}
while (t > 0):
    x = raw_input()
    if x in dic:
        dic[x] += 1
    else:
        dic[x] = 1
    t -= 1

c = 0
pre = 0
for w in sorted(dic, key=dic.get, reverse=True):
    if dic[w] >= pre:
        c += dic[w]
        pre = dic[w]
    else:
        break
print c
