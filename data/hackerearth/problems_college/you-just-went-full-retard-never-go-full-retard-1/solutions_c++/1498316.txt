#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

bool isPrime(ll n) {
  if(n==1||n==0) {
    return false;
  }
  for(int i=2;1ll*i*i<=n;i++) {
    if(n%i==0) {
      return false;
    }
  }
  return true;
}
int main() {
  int tt;
  scanf("%d",&tt);
  while(tt--) {
    ll num;
    scanf("%lld",&num);
    if(isPrime(num)==false) {
      puts("NO");
    } else {
      puts("YES");
    }
  }
  return 0;
}
