#include<bits/stdc++.h>
using namespace std;
vector<int>v[100005];
int vis[100005];
int size[100005];
void dfs(int ind)
{
	vis[ind]=1;
	size[ind]=1;
	for(int i=0;i<v[ind].size();i++)
	{
		if(!vis[v[ind][i]])
		{
			dfs(v[ind][i]);
			size[ind]+=size[v[ind][i]];
		}
	}
}
int main()
{
	int n;
	scanf("%d",&n);
	int i,j;
	int a,b;
	for(i=1;i<=n-1;i++)
	{
		scanf("%d %d",&a,&b);
		v[a].push_back(b);
		v[b].push_back(a);
	}
	dfs(1);
	for(i=1;i<=n;i++)
	{
		if(i==1)
		printf("%d ",size[i]-1);
		else
		printf("%d ",size[i]);
	}
}
