def gcd(a,b):
    while b != 0:
        t = b
        b = a % b
        a = t
    return a;

n = int(raw_input())
while n > 0:
	l,b,h = raw_input().split()
	tt = gcd(int(l),int(b))
	gcf = gcd(tt,int(h))
	
	
	a = int(l) / gcf
	b = int(b) / gcf
	c = int(h) / gcf
	
	ans = (((a*b) % 1000000007)*c)%1000000007
	
	print gcf, ans
	n-=1
