#include<bits/stdc++.h>
using namespace std;
long long arr[100005];
int n;
int main()
{
	scanf("%d",&n);
	int i,j;
	for(i=1;i<=n;i++)
	scanf("%lld",&arr[i]);
	long long sub[n+2];
	for(i=1;i<=n;i++)
	sub[i]=1;
	for(i=2;i<=n;i++)
	{
		for(j=i-1;j>=1;j--)
		{
			if(arr[j]>0&&arr[i]<0&&(abs(arr[j])<abs(arr[i])))
			{
				sub[i]=max(sub[i],sub[j]+1);
			}
			else if(arr[j]<0&&arr[i]>0&&(abs(arr[j])<abs(arr[i])))
			{
				sub[i]=max(sub[i],sub[j]+1);
			}
		}
	}
	long long ans=-1;
	for(i=1;i<=n;i++)
	{
		ans=max(ans,sub[i]);
	}
	printf("%lld\n",ans);
	return 0;
}
