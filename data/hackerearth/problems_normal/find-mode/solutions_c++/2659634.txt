#include <bits/stdc++.h>
using namespace std;
long int arr[1000001];
int main()
{
    int tc;
    long int N,c,max;
    cin>>tc;
    while(tc--)
    {
    	for(long int i=1;i<1000001;i++)
    	{
    		arr[i]=0;
    	}
    	
    	cin>>N;
    	for(long int i=0;i<N;i++)
    	{
    		scanf("%ld",&c);
    		arr[c]++;
    	}
    	max=arr[1];
    	for(long int i=1;i<1000001;i++)
    	{
    		if(arr[i]>max)
    		{
    		max=arr[i];
    		}
    		
    	}
    	for(long int i=1000000;i>0;i--)
    	{
    		if(arr[i]==max)
    		{
    			printf("%ld ",i);
    		}
    	}
    	cout<<endl;
    }
}
