#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long

int main()
{
    ll t,n;
    vector<ll>dp(100001);
    dp[1]=1;
    dp[2]=2;
    dp[3]=4;

    for(int i=4; i<=100000; i++)
    {
        dp[i]=(dp[i-1]+dp[i-2])%mod;
        dp[i]=(dp[i]+dp[i-3])%mod;
    }

    cin>>t;
    while(t--)
    {
        cin>>n;
        cout<<dp[n]<<endl;
    }
    return 0;
}
