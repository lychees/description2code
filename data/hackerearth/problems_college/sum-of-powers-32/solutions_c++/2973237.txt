#include <iostream>
#include <iostream>
#include <cmath>
using namespace std;
 
int main()
{   int c,t,n,j,pos,a[1000000]={0},i;
    a[1]=1;
    for(i=2;i<=1000;i++){
        for(j=2;pow(i,j)<=1000000;j++)
        {   pos=pow(i,j);
            a[pos]=1;
        }
    }
    cin>>t;
    while(t--)
    {   cin>>n;
        c=0;
        for(i=1;i<=n/2;i++)
        {   if(a[i] && a[n-i]) 
            {   c=1;
                break;
            }
        }
        if(c) 
            cout<<"Yes\n";
        else 
            cout<<"No\n";
    }
    return 0;       
}
