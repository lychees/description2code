inline int f(ll x)
{
	int res = 0;
	while(x)
	{
		if (x % 10 == 4 || x % 10 == 7) ++res;
		x /= 10;
	}
	return res;
}
ll n, l;

int main()
{
	#ifdef LOCAL
	freopen(fname".in", "r", stdin);
	freopen(fname".out", "w", stdout);
	#endif

	cin >> n >> l;
	for (ll b = n + 1;; ++b)
	{
		bool ok = 1;
		for (int i = 0; i < l && ok; ++i)
			if (f(n + i) != f(b + i))
				ok = 0;
		if (ok)
		{
			cout << b << endl;
			return 0;
		}
	}

	return 0;
}
