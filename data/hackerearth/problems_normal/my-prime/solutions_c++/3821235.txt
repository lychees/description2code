#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int n;
	cin >> n;
	int a[n];
	int i=0;
	vector<int> nums;
	while(i<n)
	{
		cin >> a[i];
		i++;
	}
	for(int i=0; i<n; i++)
	{
		int flag = 1;
		for(int j=0; j<n && flag!=0; j++)	
		{
			if(a[i] != a[j] && a[i]%a[j] == 0)
				flag = 0;
		}	
		if(flag == 1)
			nums.push_back(a[i]);
	}
	for(int i=0; i<nums.size(); i++)
		cout << nums[i] << " ";
	return 0;
}
