#include <bits/stdc++.h>
using namespace std;
long long s[1000009],dp[1000009];
int main()
{
	long long n,a,b,maxa = -1;
	cin >> n;
	memset(dp,0,sizeof dp);
	memset(s,0,sizeof s);
	for(int i =0 ; i < n; i++) {
		cin >> a >> b;
		s[a] = b;
		maxa = max(a,maxa);
	}
    for(int i = maxa; i > 0; i--) {
    	dp[i] = max(s[i],dp[i+1]);
    }
    int q;
    cin >> q;
    long long int ans = 0;
    for(int i =0; i < q; i++) {
    	cin >> a;
    //	cout << "dp[" << a << "] = " << dp[a] << endl;
    	ans+=(100*(dp[a]/a));
    }
    cout << ans << endl;
    return 0;
}
