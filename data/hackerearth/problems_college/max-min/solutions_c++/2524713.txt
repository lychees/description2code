void solve()
{
	int n, k;
	cin>>n>>k;
	vector<int> a(n);
	for(int i=0;i<n;i++)
		cin>>a[i];
	sort(a.begin(), a.end());
	ll min_diff = 99999999999999;
	for(int i=0;i<=n-k;i++) {
		if(a[k-1+i] - a[i] < min_diff)
			min_diff = a[k-1+i] - a[i];
	}
	cout<<min_diff<"\n";
}
