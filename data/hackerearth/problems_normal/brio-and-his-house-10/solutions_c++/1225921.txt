#include<stdio.h>
#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int t=0;
	cin>>t;
	while(t--)
	{
		double a=0,b=0,c=0,r=0;
		cin>>a>>b>>c;
		r=(a*b*c)/sqrt(((a+b+c)*(-a+b+c)*(a+b-c)*(a-b+c)));
		
		printf("%.4lf\n",(3.1415)*r*r);
		
		
		
		}//end of the while loop
	
	return 0;
	
	}
