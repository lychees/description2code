using namespace std;
long a[10100]={0};
void compo()
{
    long n,k,i;
           n=10050;


	for(i=3;i<=sqrt(n);i=i+2)
	{
		if(a[i]!=1)
		{
			for(k=i*i;k<=n;k=k+2*i)
            a[k]=1;
		}
	}


}
int main()
{
    compo();
    long p;
    while(cin>>p)
    {
         if(4<=p)
            cout<<"4";
        for(int i=5;i<=p;i++)
        {
            //cout<<a[i]<<" ";
            if(a[i]==0&&i%2!=0)
            {
                continue;
            }
            else
                cout<<" "<<i;
        }
        cout<<"\n";
    }
    return 0;
}
