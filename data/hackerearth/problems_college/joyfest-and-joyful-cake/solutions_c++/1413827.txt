#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    long long n;
    while(t--){
    	scanf("%lld",&n);
    	if (n==1){
    		cout << "0\n";
    		continue;
    	}
    	if (n==2){
    		cout << "1\n";
    		continue;
    	}
    	double y = (sqrt(1.0+8.0*(n-1))-1.0)/2.0;
    	long long x = y-10;
    	if (x<0) x = 2LL;
    	while (x*x+x<2*(n-1)) x++;
    	printf("%lld\n",x);
    }
    return 0;
}
