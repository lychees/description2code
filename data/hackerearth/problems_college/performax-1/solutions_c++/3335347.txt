#include <iostream>
using namespace std;

int main()
{
   long long int n,d,t,s,p;
    long long int l;
    cin>>n>>d>>l;
    s=0;
   for(int i=1;i<n;i++)
    {
    	cin>>t;
    	p=l-t+1;
    	if(p>0){
    		if(p%d==0)
    		{
    			s=s+p/d;
    			//l=t;
    			l=l+1;
    		}
    		else
    		{
    			s=s+p/d+1;
    			l=d*(p/d+1)+t;
    		}
    	}
    	else{
    		l=t;
    	}
    }
    cout<<s<<endl;
    return 0;
}
