#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
	int testcases,n,i,aux;
	long long rta;
	vector<int> hombres, mujeres;
	scanf("%d", &testcases);
	while(testcases > 0){
		scanf("%d", &n);
		hombres.clear();
		mujeres.clear();
		for(i = 0; i < n; i++){
			scanf("%d", &aux);
			hombres.push_back(aux);
		}
		sort(hombres.begin(), hombres.end());
		for(i = 0; i < n; i++){
			scanf("%d", &aux);
			mujeres.push_back(aux);
		}
		sort(mujeres.begin(), mujeres.end());
		rta = 0;
		for(i = 0; i < n; i++){
			rta += (hombres[i] * mujeres[i]);
		}
		printf("%lld\n", rta);
		testcases--;
	}
	return 0;
}
