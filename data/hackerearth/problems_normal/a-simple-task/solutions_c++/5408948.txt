#include<bits/stdc++.h>
using namespace std;
int check(int a[],int test,int n)
{
	int i;
	int flag=a[0]%test;
	for(i=1;i<n;i++)
	{
		if(a[i]%test!=flag)
			return 0;
	}
	return 1;
}
int main()
{
    int n;
    cin>>n;
    int a[n],i,temp[100005];
    for(i=0;i<n;i++)
    	cin>>a[i];
    sort(a,a+n); 
    int diff=a[1]-a[0];
    int j=0;
    for(i=2;i<n;i++)
    	diff=min(diff,a[i]-a[i-1]);
    if(check(a,diff,n))
			temp[j++]=diff;
    for(i=2;i*i<=diff;i++)
    {
      if(diff%i==0)
      {
        if(check(a,i,n))
			temp[j++]=i;
		if((diff/i)!=i&&check(a,diff/i,n))
			temp[j++]=diff/i;
        }
    }
    sort(temp,temp+j);
    for(i=0;i<j;i++)
       cout<<temp[i]<<" ";
    return 0;
}
