#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll ans[100005];
ll temp[100005][3];
ll val[100005];
int main()
{
	ll i,j,k,n,m,T,Q,x,y;
	scanf("%lld",&T);
	while(T--){
		scanf("%lld%lld",&n,&m);
		for(i=1;i<=m;i++){
			scanf("%lld%lld%lld",&temp[i][0],&temp[i][1],&temp[i][2]);
		}
		for(i=0;i<=n+1;i++){
			ans[i]=0;
		}
		for(i=0;i<=m+1;i++)
		{
			
			val[i]=0;
		}
		scanf("%lld",&Q);
		while(Q--){
			scanf("%lld%lld",&x,&y);
			val[x]+=1;
			val[y+1]-=1;
		}
		for(i=2;i<=m;i++)
			val[i]+=val[i-1];
		for(i=1;i<=m;i++){
			ans[temp[i][0]]+=temp[i][2]*val[i];
			ans[temp[i][1]+1]-=temp[i][2]*val[i];
		}
		for(i=2;i<=n;i++)
			ans[i]+=ans[i-1];
		for(i=1;i<=n;i++){
			printf("%lld ",ans[i]);
		}
		printf("\n");
	}
    return 0;
}
