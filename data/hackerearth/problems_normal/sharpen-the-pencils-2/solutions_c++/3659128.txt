#include<iostream>
using namespace std;

int
main ()
{
  int tc = 0;
  cin >> tc;
  while (tc--)
    {
      long long n = 0;
      cin >> n;

      long long *arr = new long long[n];
      for (long long i = 0; i < n; i++)
	cin >> arr[i];

      long long i = 0;
      long long j = n - 1;
      long long w = arr[0], worka = 0, workb = 0;

      while (j > i)
	{
	  workb = arr[j];
	  worka = 2 * workb;
	  while (w < worka)
	    {
	      if (i + 1 < j)
		i++;
	      w = w + arr[i];
	    }
	  w = w - worka;

	  if (w == 0 && i + 1 < j)
	    {
	      i++;
	      w = w + arr[i];
	    }

	  if (j - 1 > i)
	    j--;
	  else
	    break;
	}

      cout << i + 1 << " " << n - (i + 1) << endl;

    }
}
