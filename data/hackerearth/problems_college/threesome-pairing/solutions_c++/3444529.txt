#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t, i, j, n, ans;
    cin >> t;
    while (t--) {
        ans = 0;
        cin >> n;
        int v[n+1];
        for (i = 0; i < n; i++) {
            cin >> v[i];
        }
        for(i = 0; i < n; i++) {
            for(j = i+1; j < n; j++) {
                if(i == j) {
                    continue;
                }
                if ((v[i] + v[j]) % 3 == 0) {
                    ans++;
                }
            }
        }
        cout << ans << endl;
    }

    return 0;
}
