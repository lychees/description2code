#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
     cin>>t;
     while(t--)
     {
     	string s;
     	int dump,dope[26]={0};
     	cin>>s;
     	for(int i=0;i<s.length();i++)
     	{
     		dump=s[i];
     		dump=dump%97;
     		dope[dump]++;
     	}
     	sort(dope,dope+26);
     	cout<<dope[25]<<"\n";
     }
    return 0;
}
