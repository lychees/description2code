#include<bits/stdc++.h>
using namespace std;
long long b[10000000];
int main()
{
	long long n,m,i,j,k;
	scanf("%lld",&n);
	long long a[n+1];
	for(i=1;i<=n;i++)
	{
		scanf("%lld",&a[i]);
	}
	scanf("%lld",&m);
	while(m--)
	{
		scanf("%lld%lld%lld",&i,&j,&k);
		b[i]+=k;
		b[j+1]-=k;
	}
	for(i=2;i<=n;i++)
	{
		b[i]+=b[i-1];
	}
	for(i=1;i<=n;i++)
	{
		a[i]+=b[i];
		cout<<a[i]<<'\n';
	}
	return 0;
}
