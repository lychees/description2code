#include<bits/stdc++.h>
using namespace std;
long int dp[1000005];
int main()
{
    int t;
    long int n;
    cin>>t;
    int a[]={2,3,5,7};
    for(int i=0;i<=1000000;i++){
        dp[i]=1000003;
    }
    dp[0]=0;
    dp[1]=-1;
    dp[2]=1;
    dp[3]=1;
    for(long int i=4;i<=1000000;i++){
        for(int j=0;j<=3;j++){
            if(a[j]<=i && (i-a[j])!=1){
                dp[i]=min(dp[i],(1+(dp[i-a[j]])));
            }
        }
    }
    while(t--)
    {
        cin>>n;
        cout<<dp[n]<<endl;
    }
}
