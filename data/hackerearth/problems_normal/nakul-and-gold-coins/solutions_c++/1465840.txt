#include<bits/stdc++.h>
using namespace std;
int sum[1000010];
int mark[1000010];
int primes[1000010],sz=0;
void compute()
{

    for(int i =2 ; i <=1000;i++)
    {
        if(mark[i]==0){
        for(int j = i*i;j<=1000000;j = j+i)
        {
            mark[j]=1;
        }
        }
    }
for(int i = 2;i<=1000000;i++)
{
    if(mark[i] == 0)
        {sum[i] = 1;
        primes[sz] = i;
        sz++;
        }
    sum[i]=sum[i]  + sum[i-1];
}

}
int main()
{
int t;
compute();
//return 1;
scanf("%d",&t);
while(t--)
{
long long l,r;
scanf("%lld %lld",&l,&r);
long long ans= 0;
for(int i = 0; i <sz && primes[i] <r;i++)
{
    long long a = r/primes[i];
    long long b = max((long long)primes[i],(l-1)/primes[i]);
    //cout<<a<<" "<<b<<endl;
    //cout<<sum[a]<<" "<<sum[b]<<endl;
    if(b>a)
    continue;

    long long x,y;
    if(a>1000000)
        x = sum[1000000];
    else
        x = sum[a];
    if(b> 1000000)
        y = sum[1000000];
    else
        y = sum[b];
    ans = ans + (x - y);
}
cout<<ans<<endl;
}
}
