from sys import stdin
# to get line from stdin


def getLine():
    return stdin.readline().strip()


def findStreak():
    n = getLine()
    # minutes=1444
    n = int(n)
    x = -1
    y = -1
    tempX = 0
    tempY = 0
    for i in range(0, n):
        line = getLine()
        tempX = 0
        for j in range(0, len(line)):
            if (line[j] == 'C'):
                tempX = tempX+1
                tempY = tempY+1
            else:
                if (tempX > x):
                    x = tempX
                if (tempY > y):
                    y = tempY
                tempX = 0
                tempY = 0
        if (tempX > x):
            x = tempX
    if (tempY > y):
        y = tempY
    return x, y


x, y = findStreak()
print(str(x)+" "+str(y))
