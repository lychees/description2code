#include <vector>
#include <cstdlib>
#include <iostream>
#include <iomanip>
#include <string>
using namespace std;
typedef long long LL;

int main()
{
  int n,d,mi,ma;
  LL ans;
  scanf("%d",&n);
  scanf("%d",&mi);
  ma = mi;
  for(int i=1;i<n;i++)
  {
      scanf("%d",&d);
      if(ma<d)ma=d;
      if(mi>d)mi=d;
  }
  ans = 1;
  ans = ans*ma;
  ans*=mi;
  printf("%lld\n",ans);
  return 0;
}
