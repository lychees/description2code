x, n = (int(i) for i in raw_input().split())
l = []
for kkk in range(0, n):
    x1, y1, x2, y2 = (int(i) for i in raw_input().split())
    if y1 <= y2:
        l.append([y1, y2, x1, x2])
    else:
        l.append([y2, y1, x2, x1])
l.sort(reverse=True)
index = 0
while index < n:
    if min(l[index][2], l[index][3]) <= x <= max(l[index][2], l[index][3]):
        x = l[index][2]
    index += 1
print x
