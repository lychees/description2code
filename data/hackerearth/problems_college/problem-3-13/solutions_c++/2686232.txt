#include<bits/stdc++.h>
using namespace std;

int x[1000][1000],ss[1000][1000];
int main()
{
    int r,c;
    cin>>r>>c;
    
   
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            cin>>x[i][j];
        }
    }
    int max=0;
    for(int i=0;i<r-1;i++)
    {
        for(int j=0;j<c;j++)
        {
            if(x[i+1][j]==x[i][j])
            {
            ss[i+1][j]=ss[i][j]+1;
            }
            if(ss[i+1][j]>max)
                max=ss[i+1][j];
        }
        
    }
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c-1;j++)
        {
            if(x[i][j+1]==x[i][j])
            {
            ss[i][j+1]=ss[i][j]+1;
            }
            if(ss[i][j+1]>max)
                max=ss[i][j+1];
        }
        
    }
    /*for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            cout<<ss[i][j]<<" ";
        }
        cout<<endl;
    }*/
    cout<<max+1;
    return 0;
}
