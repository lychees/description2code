#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
#define ll long long int 
int min(int a,int b)
    {
    return (a<b)?a:b;
}
ll dp[10005],arr[10005];
int main() {
    
    ll s[3],i,j,n,maxi,mini;
    s[0]=1;
    s[1]=2;
    s[3]=5;
    dp[0]=0;
    for(i=1;i<=10004;i++)
        {
        maxi=10007;
        
            {
                    if(i>=1&&maxi>dp[i-1])
                        maxi=dp[i-1];
                if(i>=2&&maxi>dp[i-2])
                        maxi=dp[i-2];
                if(i>=5&&maxi>dp[i-5])
                        maxi=dp[i-5];
                        
             }
        dp[i]=maxi+1;
        //if(i<=15)
          //  cout<< dp[i]<<"  "<<maxi<<endl;
    }
    
    ll t;
    cin>>t;
    ll count,count1;
    while(t--)
        {
        count=0;
        cin>>n;
        mini=10009;
        for(i=0;i<n;i++)
        {
            cin>>arr[i];
         if(arr[i]<mini)
             mini=arr[i];
        }
       // cout<<mini<<endl;
        count=0;
        count1=0;
        for(i=0;i<n;i++)
        {count+=dp[arr[i]-mini];
         count1+=dp[arr[i]];
        }
        count=min(count,count1);
        cout<<count<<endl;
        
    }
    return 0;
}
