int main()
{
	int t,x,i,j,n,a[100000];
	sc(t);
	while(t--)
	{
		sc(n);
		for(i=0;i<n;i++)
		{
			sc(a[i]);
		}
		for(i=n-2;i>=0;i--)
		{
			if(a[i]>a[i+1])
				break;
		}
		if(i==-1)
		{
			for(i=0;i<n;i++)
				printf("%d ",a[i]);
			printf("\n");
		}
		else
		{
			x=i+1;
			for(j=i+2;j<n;j++)
				if(a[j]<a[i])
					x=j;
			swap(a[i],a[x]);
			reverse(a+i+1,a+n);
			for(i=0;i<n;i++)
				printf("%d ",a[i]);
			printf("\n");
		}
	}
	return 0;
}
