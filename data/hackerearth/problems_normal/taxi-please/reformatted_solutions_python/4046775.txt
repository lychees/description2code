N, M = map(int, raw_input().split())
taxi = [0 for i in xrange(N)]
ans = [-1 for i in xrange(N)]
request = []
for n in xrange(N):
    S, J = map(int, raw_input().split())
    request.append((S, J, n))
request = sorted(request, key=lambda x: x[0])
for S, J, idx in request:
    found = False
    for i in xrange(M):
        if taxi[i] <= S:
            taxi[i] = S + J
            ans[idx] = i+1
            break
print ' '.join(map(str, ans))
