#include <iostream>
using namespace std;
int t[100003]={0}; //time
int c_b[100003]= {0} ;// cumalative beeps
int m_i[100003]= {0} ;// max intensity
int p_f[100003]= {0} ;//person frequency
int p_i[100003]= {0} ;//person intensity
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>t[i];
	
	for(int i=1;i<=n;i++)
	{
		for(int j=t[i] ; j<=100000 ; j += t[i] )
		{
			c_b[j]++;
			m_i[j]++;
		}
	}

	for(int i=1;i<=100000;i++)
		c_b[i] = !!c_b[i];

	for(int i=2;i<=100000;i++)
	{
		m_i[i]=max( m_i[i] , m_i[i-1] );
		c_b[i]=c_b[i]+c_b[i-1];
	}
	// cout<<"\nCummalative frequency\n";
	// for(int i=1;i<=n;i++)
	// {
	// 	cout<<c_b[i]<<" ";
	// }
	// cout<<"\nIntensity\n";
	// for(int i=1;i<=n;i++)
	// 	cout<<m_i[i]<<" ";
	// cout<<"\n";
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>p_f[i]>>p_i[i];
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		int x,y;
		cin>>x>>y;
		if( c_b[y] > p_f[x] || m_i[y] > p_i[x] )
			cout<<"Awake\n";
		else
			cout<<"Asleep\n";
	}
    return 0;
}
