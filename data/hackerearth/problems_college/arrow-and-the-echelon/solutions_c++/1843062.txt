#include"bits/stdc++.h"
using namespace std;
void print(int _5,int _3){
	for(int i=0;i<_5;++i)putchar('5');
	for(int i=0;i<_3;++i)putchar('3');
	puts("");
}
int main(){
	std::ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		bool flag=false;
		for(int i=0;i<=n;i+=5){
			if((n-i)%3==0){
				print(n-i,i);
				flag=true;
				break;
			}
		}
		if(!flag)puts("-1");
	}
	return 0;
}
