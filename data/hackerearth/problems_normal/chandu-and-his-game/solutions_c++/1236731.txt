#include<iostream>
using namespace std;
int min(int a,int b,int c,int d){
	int x,y;
	x=(a>b)?b:a;
	y=(c>d)?d:c;
	return (x>y)?y:x;
}


int main(){
	int n,k;
	cin>>n>>k;
	int x[k+1],y[k+1];
	int i;
	for(i=1;i<=k;i++)
		cin>>x[i];
	for(i=1;i<=k;i++)
		cin>>y[i];
	long long int s=0;
	for(i=1;i<=k;i++){
		s+=min(x[i]+y[i]-2,(x[i]-1)+n-y[i],n-x[i]+(y[i]-1),(n-x[i])+(n-y[i]));
		}
	cout<<s<<endl;
return 0;
}
