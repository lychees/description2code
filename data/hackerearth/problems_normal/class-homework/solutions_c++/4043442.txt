#include <iostream>
#include <bits/stdc++.h>
#define ll long long

using namespace std;

int main()
{
    int tc,i,j,k,n,flag,x;char s[22];
    
    ll maxi=(ll)1e12;
    scanf("%d",&tc);
    while(tc--)
    {
    	scanf("%s",s);
    	n=strlen(s);
    	if(n<4)
    	{
    	printf("unlucky\n");
    	continue;
    	}
    	ll a,b,c,d,sum;
    	sum=LLONG_MIN;
    	for(i=0;i<=n-4;i++)
    		for(j=i+1;j<=n-3;j++)
    			for(k=j+1;k<=n-2;k++)
    			{
    				a=0;b=0;c=0;d=0;flag=0;
    				for(x=0;x<=i;x++)a=a*10+(s[x]-'0');
    				for(x=i+1;x<=j;x++)b=b*10+(s[x]-'0');
    				for(x=j+1;x<=k;x++)c=c*10+(s[x]-'0');
    				for(x=k+1;x<n;x++)d=d*10+(s[x]-'0');
    				if(i>=1&&s[0]=='0')flag=1;
    				if(j-i>1&&s[i+1]=='0')flag=1;
    				if(k-j>1&&s[j+1]=='0')flag=1;
    				if(n-1-k>1&&s[k+1]=='0')flag=1;
    				if(a>maxi||b>maxi||c>maxi||d>maxi)flag=1;
    				if(flag==1)
    				continue;
    				//cout<<a<<" "<<b<<" "<<c<<" "<<d;
    				sum=max(sum,a+b+c+d);
    			}
    	if(sum==LLONG_MIN)
    	printf("unlucky\n");
    	else
    	printf("%lld\n",sum);
    }
	return 0;
}
