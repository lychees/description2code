#include <iostream>
using namespace std;

int main()
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int mat[n][3];
		int dp[n][3];
		for(int i=0; i<n; i++)
		{
			for(int j=0; j<3; j++)
			{
				cin>>mat[i][j];
			}
		}

		for(int i=0;i<3;i++)
			dp[0][i] = mat[0][i];

		for(int i=1;i<n;i++)
		{
			dp[i][0] = min(dp[i-1][1],dp[i-1][2]) + mat[i][0]; 
			dp[i][1] = min(dp[i-1][0],dp[i-1][2]) + mat[i][1]; 
			dp[i][2] = min(dp[i-1][0],dp[i-1][1]) + mat[i][2]; 
		}

		cout<<min(min(dp[n-1][0],dp[n-1][1]),dp[n-1][2])<<endl;
	}
	return 0;
}
