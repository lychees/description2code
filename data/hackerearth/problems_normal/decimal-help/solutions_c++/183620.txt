#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int findRemainder(int numerator, int denominator, int position)
{
     int remainder = numerator;
    int digit = 0;
    int i = 0;
    for( i = 0; i < position; i++)
    {
        if(remainder == 0)
            return 0;
        digit = (10*remainder) / denominator;
        remainder = (10*remainder) % denominator;
    }
    return digit;
}

int main()
{
    int cases;
    scanf("%d",&cases);
    while(cases--)
    {
        int  numerator, denominator, position;
    scanf("%d %d %d", &numerator, &denominator, &position);
    printf("%d\n",findRemainder(numerator, denominator, position));


    }


    return 0;
}
