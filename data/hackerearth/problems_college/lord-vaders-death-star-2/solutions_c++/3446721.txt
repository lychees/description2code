 void INSERTION_SORT(long long int a[],long long int n)
    {
        long long int key,i,j;
        for(j=2;j<=n;j++)
        {
            key=a[j];
            i=j-1;
            while((i>0)&&(a[i]>key))
            {
                a[i+1]=a[i];
                i=i-1;
            }
            a[i+1]=key;
        }
    }
