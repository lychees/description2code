#include <bits/stdc++.h>
using namespace std;
#define max 100005
int tim[max], prior[max];
priority_queue<pair<int, int>> q;
int main()
{
	int t;
	cin>>t;
	int n;
	for(int te=1;te<=t;te++) {
		cin>>n;
		for(int i=0;i<n;i++) cin>>tim[i];
		for(int i=0;i<n;i++) cin>>prior[i];
		for(int i=0;i<n;i++) {
			q.push(make_pair(prior[i]/tim[i], -i));
		}
		cout<<"Case #"<<te<<": ";
		pair<int, int> node;
		for(int i=0;i<n;i++) {
			node = q.top();
			q.pop();
			cout<<-node.second<<" ";
		}
		cout<<endl;		
	}
    return 0;
}
