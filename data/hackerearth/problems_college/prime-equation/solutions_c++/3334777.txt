#include<bits/stdc++.h>
#define ll long long 
#define ull unsigned long long 
#define ldo long double
#define db double
#define pb push_back
#define vi vector<int>
#define vll vector<long long>
#define  boost ios_base::sync_with_stdio(0)
#define MOD 1000000007
#define SIZE 10000001
using namespace std;
bool arr[SIZE]={0};
int main()
{  
	boost;
	arr[0]=1;
	arr[1]=1;
	ll i,p,x,t,n;
	ll value[1000];
	set<ll> myset;
	set<ll>::iterator it;
	myset.clear();
	for(i=3;i<SIZE;i=i+2)
	{
		if(arr[i]==0)
		{
			p=3;
			while(p*i<SIZE)
			{
				arr[p*i]=1;
				p=p+2;
			}
		}
	}
	x=3;
	for(i=5;i<SIZE;i=i+2)
	{
		if(arr[i]==0)
		{
			myset.insert(i-x);
			x=i;
		}
	}
	it=myset.begin();
	ll temp=0;
	while(it!=myset.end())
	{
		//cout<<temp<<" "<<*it<<endl;
		value[temp]=*it;
		temp++;
		it++;
	}
	//cout<<endl<<temp<<endl;
	myset.clear();
	for(i=0;i<temp;i++)
	{
		for(ll j=0;j<temp;j++)
		{
			myset.insert(value[i]+value[j]);
		}
	}
	//printf("Hello\n");
	//cout<<myset.size()<<endl;
	cin>>t;
	while(t--)
	{
		cin>>n;
		if(myset.find(n)!=myset.end())
		cout<<"YES\n";
		else
		cout<<"NO\n";
	}
	myset.clear();
	return 0;
}
