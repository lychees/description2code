import itertools

t = int(raw_input())
for i in range(1, t+1):
    n = int(raw_input())
    _str = raw_input()
    if _str[-1] == " ":
        # Unecessary line to get trid of some shitty inputs with spaces at the
        # end
        _str = _str[:-1]
    _nos = _str.split(" ")
    _values = map(lambda v: int(v), _nos)
    _sum = []
    for x in range(0, n+1):
        _combinations = list(itertools.combinations(_values, x))
        for c in _combinations:
            _sum.append(sum(c))
    print len(set(_sum))
