 #include <bits/stdc++.h>

using namespace std;

int DP[1002][1002][2];
bool visited[1002][1002][2];
int GCD[1002][1002];

int gcd(int a, int b) {
	if(b == 0) {
		return a;
	}
	return gcd(b, a % b);
}

void solve() {
	for(int i = 1; i <= 1000; i ++) {
		for(int j = 1; j <= i; j ++) {
			GCD[i][j] = GCD[j][i] = gcd(i, j);
		}
	}
}

int evaluate(int a, int b, int turn) {
	if((turn == 0 && a == 1) || (turn == 1 && b == 1)) {
		return 0;
	}
	if(visited[a][b][turn]) {
		return DP[a][b][turn];
	}
	visited[a][b][turn] = true;
	int result = 0;
	if(turn == 0) {
		result |= (!evaluate(a, b - 1, turn ^ 1));
		if(GCD[a][b] != 1) {
			result |= (!evaluate(a, b / GCD[a][b], turn ^ 1));				
		}
	} else {
		result |= (!evaluate(a - 1, b, turn ^ 1));
		if(GCD[a][b] != 1) {
			result |= (!evaluate(a / GCD[a][b], b, turn ^ 1));				
		}
	}
	DP[a][b][turn] = result;
	return result;
}

int main() {
	int t, a, b;
	solve();
	cin >> t;
	while(t --) {
		cin >> a >> b;
		if(a == 1 && b == 1) {
			cout << "Draw" << endl;
		} else if(a == 1) {
			cout << "Chandu Don" << endl;
		} else if(b == 1) {
			cout << "Arjit" << endl;
		}else if(evaluate(a, b, 0)) {
			cout << "Arjit" << endl;
		} else {
			cout << "Chandu Don" << endl;
		}
	}
	return 0;
}
