#include <iostream>
#include <map>
using namespace std;

int main(){
	map<string,int> mp;
	
	int t;
	string s;
	
	cin >> t;
	while(t--){
		cin >> s;
		if(mp.count(s) == 0){
			mp[s] = 1;
		} else {
			mp[s] += 1;
		}
	}
	
	map<string, int>::iterator it;
	int max = 0, count = 0;
	for(it = mp.begin(); it != mp.end(); it++){
		if(it->second > max){
			max = it->second;
			count = max;
		} else if(it->second == max){
			count += max;
		}
	}
	
	cout << count << endl;
    return 0;
}
