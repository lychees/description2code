#include <iostream>
using namespace std;

int main()
{
   long t;
   cin>>t;
   while(t--){
   	long long n,p,a,q=0;
   	cin>>n;//cout<<n<<endl;
   	p=n;
   	while(n!=1){
   		p=n;
   		long sum=0;
   		while(p!=0){
   			a=p%10;
   			sum+=a;
   			p/=10;
   		}
   	//	cout<<sum<<endl;
   		if(sum%3==0){n/=3;q++;}
   	else if(n%2==0){n/=2;q++;}
   		else {n--;q++;}
   	}
   	cout<<q<<endl;
   }
}
