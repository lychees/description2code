#include <iostream>
using namespace std;

int main()
{
    int A, B, N;
    cin>>A>>B>>N;
	int sum = A+B;
    if(N == 1)
    	sum = A;
    else if(N == 2)
    	sum = B;
    else{
	    for(int i = 3; i < N; i++){
	    	A = B;
	    	B = sum;
	    	sum = A + B;
	    }
    }
	cout<<sum;
    return 0;
}
