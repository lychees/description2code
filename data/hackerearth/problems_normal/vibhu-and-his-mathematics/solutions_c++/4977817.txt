#include <iostream>
typedef unsigned long long ull;
using namespace std;

const long int MOD = 1000000007;

ull dp[1000001],t,n;

ull solve(ull n)
{
	if (dp[n] != 0) return dp[n];
	return dp[n] = (solve(n-1) + (n-1)*solve(n-2)) % MOD;
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    dp[1] = 1;
    dp[2] = 2;
    cin >> t;
    while (t--)
    {
    	cin >> n;
    	cout << solve(n) << "\n";
    }
    return 0;
}
