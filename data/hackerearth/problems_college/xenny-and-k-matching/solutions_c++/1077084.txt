#include <bits/stdc++.h>
#define MAX 1000001
#define LL long long
using namespace std;
int main()
{
    LL n=0;
    LL k=0,i=0,a[MAX]={0},j=0;
    #ifndef ONLINE_JUDGE
    freopen("input.txt","r",stdin);
    #endif // ONLINE_JUDGE
    cin>>n>>k;
    int counter=0;
    for(i=0;i<n;i++)
        cin>>a[i];
    for(i=0,j=k;j<n;j++,i++)
        {
            if((abs(a[i]-a[j])%k==0 )|| (abs(a[i]+a[j])%k==0))
            counter++;
        }
    cout<<counter<<endl;
    return 0;
}
