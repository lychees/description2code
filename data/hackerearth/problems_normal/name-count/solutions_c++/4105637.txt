#include<stdio.h>
#define M 1000000007
long long fact[10000000];
long long int modpow(long long int base,long long int exp)
{
    long long int val=1;
    while(exp!=0)
    {
        if(exp%2==1)
            val=(val*base)%M;
        exp=exp/2;
        base=(base*base)%M;
    }
    return val;
}
int main()
{
    long long t,i;
    
	fact[0]=1,fact[1]=1;
    for(i=2;i<=2000000;i++)
        fact[i]=(fact[i-1]*i)%M;
	
	scanf("%lld",&t);
    
        
    long long n,nCr,r;
    while(t--)
    {
       
	   
	    scanf("%lld %lld",&r,&n);
      //  r=n;
       // n=2*n-1;
        nCr=fact[n]*(modpow((fact[n-r]*fact[r])%M,M-2)%M)%M;
        
        printf("%lld\n",(nCr*fact[r])%M);
    }
    return 0;
}
