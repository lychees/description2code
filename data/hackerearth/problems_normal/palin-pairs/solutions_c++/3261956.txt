#include <bits/stdc++.h>
using namespace std;
#define ff first
#define ss second
int main()
{
	map<string,int> m;
	int n;
	cin>>n;
	for(int i=1;i<=n;i++) {
		string str;
		cin>>str;
		m[str]++;
	}
	long long ans=0,ans2=0;
	for(map<string,int>::iterator mit=m.begin();mit!=m.end();mit++) {
		string sss=mit->ff;
		string s2=mit->ff;
		reverse(sss.begin(),sss.end());
		if(s2==sss) {
			int val=mit->ss;
			ans+=1LL*val*(val-1)/2;
		}
		else if(m.find(sss)!=m.end()) {
			int val1=mit->ss,val2=m[sss];
			ans2+=1LL*val1*val2;
		}
	}
	cout<<ans2/2+ans<<endl;
    return 0;
}
