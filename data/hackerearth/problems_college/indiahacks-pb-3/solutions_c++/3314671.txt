#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,k;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		int a[n],i;
		for( i=0;i<n;i++)
		{
			cin>>a[i];
		}
		
		int dpi[n],dpe[n];
		dpi[0]=a[0];
		dpe[0]=0;
		for(i=1;i<=k;i++)
		{
			dpi[i]=a[i];
			dpe[i]=max(dpi[i-1],dpe[i-1]);
		}
		for(;i<n;i++)
		{
			dpi[i]=a[i]+max(dpi[i-k-1],dpe[i-k-1]);
			dpe[i]=max(dpi[i-1],dpe[i-1]);
		}
		cout<<max(dpi[n-1],dpe[n-1])<<'\n';
	}
	return 0;
}
