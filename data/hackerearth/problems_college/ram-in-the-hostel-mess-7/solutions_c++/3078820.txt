#include <bits/stdc++.h>
using namespace std;
 
int main()
{   
    long long int t,i,j,k,l,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	long long int arr[n+1];
		
    	
		if(n==1)
		{
			cout<<"4\n";
		}
		
		else{
		arr[0]=0;
		arr[1]=2;
		arr[2]=3;
		for(int i=3;i<=n;i++)
		arr[i]=arr[i-1]+arr[i-2];
		j=arr[n]*arr[n];
		cout<<j<<"\n";
		}
    }
    return 0;
}
