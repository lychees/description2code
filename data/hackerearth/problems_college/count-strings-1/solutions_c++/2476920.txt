lli *fact;
lli *infact;
lli exp(lli a, lli p )
{
lli result = 1;
if(p==0)
    return 1;
if(p==1)
    return a;
while(p)
{
    if(p&1)
        result *= a;
    result%=mod;
    p >>=1;
    a*=a;
    a%=mod;
}
return result;
}
void init()
{
    fact = new lli[1000007];
    infact = new lli[1000007];
    lli i,j,k,x,y,z;
    fact[0]=1;
    infact[0]=1;
    fl(i,1,1000007)
    {
        fact[i]=i*fact[i-1];
        fact[i]%=mod;
        infact[i]=exp(fact[i],mod-2);
        infact[i]%=mod;
    }
}
