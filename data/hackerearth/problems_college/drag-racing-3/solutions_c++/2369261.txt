#include<bits/stdc++.h>
using namespace std;
int main()
{
  int n;
   cin>>n;
   int k=pow(2,n);
   std::map<string,bool> m;
   string win;
   for(int i=0;i<k-1;i++)
   {
   	string s1,s2;
   	cin>>s1>>s2;
   	if(m.find(s1)!=m.end())
   	{
   		if(m[s1]==false)
   		{
   		 m[s2]=false;
   		 continue;	
		}
		
	}
	else
	{
	m[s1]=true;
    }
	m[s2]=false;
   }
   std::map<string,bool>::iterator it;
   for(it=m.begin();it!=m.end();++it)
   {
   	if((*it).second==true)
   	{
   	win=(*it).first;
	   break;	
	}
   }

  cout<<(*it).first<<endl;
    return 0;
}
