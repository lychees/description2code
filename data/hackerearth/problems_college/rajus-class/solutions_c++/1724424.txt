#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	ll n;
    	cin>>n;
    	vector<pair< int,string> >mp;
    	while(n--)
    	{
    		ll n;
    		string str;
    		cin>>n>>str;
    		mp.push_back(make_pair(n,str));
    		
    	}
    	sort(mp.begin(),mp.end());
    	ll r;
    	cin>>r;
    	cout<<mp[r-1].second<<endl;
    	
    }
	return 0;
}
