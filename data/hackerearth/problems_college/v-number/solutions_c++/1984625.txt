#include<iostream>
#include<string>
using namespace std;
int main()
{
	int t;
	cin >> t;
	for (int tt = 1; tt <= t; tt++)
	{
		string s;
		cin >> s;
		int a = 0, b = 0;
		for (int i = 0; i < s.length() / 2; i++)
		{
			a += s[i] - '0';
		}
		for (int i = s.length()-s.length() / 2; i < s.length(); i++)
		{
			b += s[i] - '0';
		}
		if (a == b)
		{
			cout << "YES" << endl;
		}
		else
		{
			cout << "NO" << endl;
		}
	}
}
