#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

bool flag[150];

int main()
{
    int t;
    cin>>t;
    while(t--){
        char s[1000000];
        scanf("%s", s);
        //cout<<"accepted "<<s<<endl;
        int n = strlen(s);
        int k = 0;
        for(int i=0; i<150; i++)
            flag[i] = false;
            
        for(int i=0; i<n; i++){
            if(!flag[s[i]]){
                k++;
                flag[s[i]] = true;
            }
        }
        //cout<<"Found "<<k<<" different chars"<<endl;
        
        int ans = k*(k+1)/2;
        cout<<ans<<endl;
    }
    return 0;
}
