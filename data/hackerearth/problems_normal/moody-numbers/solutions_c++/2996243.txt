#include<bits/stdc++.h>
using namespace std;

int func(long long int n)
{
	int sum=0;
	n=n*n;
	while(n)
	{
		sum+=n%10;
		n/=10;
	}
	return sum;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n;
		scanf("%lld",&n);
		int flag=0;
		while(1)
		{
			if(n==1||n==4)
			{
				flag=1;
				break;
			}
			if(n==9||n==16)
				break;
			n=func(n);
		}
		if(flag)
			printf("YES");
		else
			printf("NO");
		printf("\n");
	}
}
