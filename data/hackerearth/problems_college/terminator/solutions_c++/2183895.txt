#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int j=0,i;
    char str[101],a[101];
    cin>>str;
    for(i=0;str[i]!='\0';i++)
    {
    	if(str[i]=='o')
    	{
    	   a[j]=str[i];
    	   j++;    	   
    	   }
    else if(str[i]=='x')
    {
        while(j>0 && a[j-1]=='o')
        j--;
        while(str[i+1]=='o' && str[i]!='\0')
        i++;
        a[j]='x';
        j++;
    }
      else
      {
        a[j]='*';
        j++;
       }
   }
      a[j]='\0';
     cout<<a<<endl; 
 }
    
    return 0;
}
