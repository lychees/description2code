#include <iostream>
#include<cstring>
using namespace std;

int main()
{
	int n,k;
	string s;
	cin>>n>>k;
	int count1=0;
	int end[26]={0};
	int r=n;
	while(n--)
	{
		int ha[26]={0};
		cin>>s;
		for(int i=0;i<s.length();i++)
			ha[s[i]-'a']=1;
		for(int i=0;i<26;i++) end[i]=end[i]+ha[i];
		//for(int i=0;i<26;i++) cout<<end[i]<<" ";
		//cout<<endl;
	}
	for(int i=0;i<26;i++)
		if(end[i]==r) count1++;
	//cout<<count1;
	if(count1==k) cout<<":-)";
	else cout<<":-(";
    return 0;
}
