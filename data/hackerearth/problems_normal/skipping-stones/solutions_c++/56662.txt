#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<vector>

using namespace std;

int main()
{
	int n;
	double l,d;
	scanf("%d%lf%lf",&n,&l,&d);
	double arr[n+2],dist[n+2],dp[n+2];
	int i;//ini=1;
	arr[0]=1.0;
	dist[0]=0;
	dp[0]=1.0;
	for(i=0;i<n;i++)
	{
		scanf("%lf",&arr[i+1]);
	}
	for(i=0;i<n;i++)
	{
		scanf("%lf",&dist[i+1]);
	}
	arr[n+1]=1.0;
	dist[n+1]=d;
	
	int kk=1,can=-7;
	double mymax=-8.0;
	int p;
	for(i=1;i<n+2;i++)
	{
			p=1;
			double max=-9.0;
			while(float(dist[i]-dist[i-p]) <= float(l))
			{
				if(float(arr[i]*dp[i-p]) > float(max))
				{
					max=arr[i]*dp[i-p];
				}
				
				p++;
				if(p>i)
					break;
			}
			if(max==-9.0)
			{
				printf("IMPOSSIBLE\n");
				return 0;
			}
			else
			{
				dp[i]=max;
			}
	}
	printf("%.6lf\n",dp[n+1]);
	return 0;
}
