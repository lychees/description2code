#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    char code[10],copy[10],b[500000][10];
    int n=0,g,s,i,j,k;
    cin>>code>>n;
    for(i=0;i<n;i++)
    cin>>b[i];
    for(i=0;i<10;i++)
    copy[i]=code[i];
    
    for(i=0;i<n;i++)
    {
    	g=s=0;
    	for(j=0;j<10;j++)
    	{
    		if(copy[j]==b[i][j])
    		{
    			b[i][j]='.';
    			copy[j]='.';
    			g++;
    			
    		}
    	}
    	for(j=0;j<10;j++)
    		for(k=0;k<10;k++)
    			if(b[i][k]==copy[j]&&copy[j]!='.')
    			{
    				s++;
    				b[i][k]='.';
    				copy[j]='.';
    			
    			}
    	cout<<"\n"<<g<<" "<<s;
    	 for(k=0;k<10;k++)
    	copy[k]=code[k];
    }
    			
}
