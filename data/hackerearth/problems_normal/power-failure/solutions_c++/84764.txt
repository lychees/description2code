#include<vector>
#define max 100001
#define mod 1000000007
using namespace std;
int a[10000];
vector<int>dp(max);
vector<bool>ar(max,true);
void sieve()
{ 
int count;
count=1;
	long i,j,k;
	for(i=2;i*i<=max;++i)
	{
		if(ar[i])
		{
			for(j=i;i*j<=max;j++)
			ar[i*j]=false;
		}
	}
	dp[2]=1;
	for(i=3;i<=max-1;++i)
	{
	if(ar[i])
	{
		count++;
		dp[i]=count;
	}
	else
	dp[i]=dp[i-1];
    } 
	
}


int main()
{
	sieve();
	long prod;
int m,n,a[100000],i,b[100000],count,t,j;
	scanf("%d",&t);
	for(j=1;j<=t;++j)
	{
		count=0;
		prod=1;
		memset(b,0,sizeof(b));
         scanf("%d%d",&n,&m);
         for(i=0;i<n;++i)
         scanf("%d",&a[i]);
         sort(a,a+n);
         for(i=n-1;i>=0;--i)
         {
         	b[i]=(m-a[i]+1)-(dp[m]-dp[a[i]-1]);
         	b[i]=b[i]-count;
         	prod=((prod%mod)*(b[i]%mod))%mod;
         	count++;
         }
         printf("%ld\n",prod%mod);
	}
	return 0;
}
