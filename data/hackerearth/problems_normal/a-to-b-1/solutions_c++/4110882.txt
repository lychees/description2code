#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MAX 100005
#define MOD 1000000007
ll arr[MAX];
ll modexp(ll a, ll b)
{
		ll res = 1;
		while(b)
		{
				if(b&1) res = (res*a)%MOD;
				a=(a*a)%MOD;
				b/=2;
		}
		return res;
}
int main()
{
		int n;
		int q;
		int i;
		cin>>n;
		ll pro = 1;
		int zc = 0;
		for(i=1;i<=n;++i)
		{
				cin>>arr[i];
				if(arr[i] != 0) pro = (pro * arr[i])%MOD;
				else zc++;
		}
		cin>>q;
		while(q--)
		{
				int type,id,v;
				cin>>type>>id;
				if(type)
				{
						if(zc == 0)
						{
								ll tmp = modexp(arr[id],MOD-2);
								ll ans = (pro*tmp)%MOD;
								//cout<<"pro="<<pro<<endl;
								cout<<ans<<endl;
						}
						else if(zc == 1)
						{
								if(arr[id] == 0) cout<<pro<<endl;
								else cout<<0<<endl;
						}
						else
						{
								cout<<0<<endl;
						}

				}
				else
				{
						cin>>v;
						ll tmp = modexp(arr[id],MOD-2);
						if(v == 0)
						{
								if(arr[id] == 0) zc=zc;
								else 
								{
										zc++;
										pro = (pro*tmp)%MOD;
								}
						}
						else
						{
								if(arr[id] == 0) zc--;
								else pro = (pro*tmp)%MOD;
								pro = (pro * v)%MOD;	
						}
						arr[id] = v;
				}
		}
		return 0;
}
