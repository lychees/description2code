#include <iostream>
using namespace std;
#include <cstring>
int main()
{

    int t;
    cin>>t;
    char a[500002];
    char b[500002];
    char r[500002];
    int la,lb;
    int i;
    int c;
    char *p;
    while(t--)
    {
        cin>>a;
        cin>>b;
        la=strlen(a);
        lb=strlen(b);
        i=0;c=0;

        {
            r[i]=(a[i]+b[i]+c-96)%10+48;
            c=(a[i]+b[i]+c-96)/10;
            i++;
        }
        while(i<la)
        {
           r[i]=(a[i]+c-48)%10+48;
            c=(a[i]+c-48)/10;
            i++;
        }
         while(i<lb)
        {
           r[i]=(b[i]+c-48)%10+48;
            c=(b[i]+c-48)/10;
            i++;
        }
        if(c==1)
            r[i++]=49;
        r[i]='\0';
        p=r;

        while(p[0]=='0')
            p++;
        cout<<p<<"\n";
        }





    }
