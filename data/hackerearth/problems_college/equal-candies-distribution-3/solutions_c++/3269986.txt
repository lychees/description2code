#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
int memo[1005];
int f(int n)
{
	if(memo[n]!=0)
		return memo[n];
	if(n==0)
		return 0;
	int a, b, c;
	a=b=c=10000;
	if(n>=5)
		a=1+f(n-5);
	if(n>=2)
		b=1+f(n-2);
	if(n>=1)
		c=1+f(n-1);
	return memo[n]=min(min(a, b), c);
}
int main()
{
	int t, n, i, A[10005], m, x;
	cin>>t;
	while(t--)
	{
		m=10000;
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>A[i];
			m=min(m, A[i]);
		}
		x=0;
		for(i=0;i<n;i++)
		{
			A[i]-=m;
			x+=f(A[i]);
		}
		cout<<x<<endl;
	}
    return 0;
}
