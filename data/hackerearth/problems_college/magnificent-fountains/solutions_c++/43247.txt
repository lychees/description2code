#include <iostream>
#include <memory.h>
using namespace std;

const int MAXN = 10100;
const long long MOD = 1000000007;
int r[MAXN];

int main() {
    int T;
    cin >> T;
    for (int t = 0; t < T; t++) {
    	//memset(r, 0, sizeof r);
    	int n;
    	cin >> n;
    	for (int i = 0; i < n; i++) {
    		int a;
    		cin >> a;
    		int b = a;
    		for (int x = 2; x <= b; x++) {
    			int c;
    			for (c = 0; a % x == 0; a /= x, c++);
    			r[x] = max(r[x], c);
    		}
    	}
    	long long ans = 1;
    	for (long long x = 2; x < MAXN; x++) {
    		for (int i = 0; i < r[x]; i++) {
    			ans = (ans * x) % MOD;
    		}
    	}
    	cout << ans << endl;
    }
    return 0;
}
