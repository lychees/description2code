#include<bits/stdc++.h>
using namespace std;

int main(){
     int dp[101][101],m,a[101][101],n,t;
    cin>>t;
    while(t--){
        cin>>n>>m;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++)
                cin>>a[i][j];
        }
        
        dp[n-1][m-1]=1;
        for(int i=n-2;i>=0;i--)
          {  if(a[i][m-1]<a[i+1][m-1]) dp[i][m-1]=dp[i+1][m-1]+1;
             else                      dp[i][m-1]=1;
          }
        for(int j=m-2;j>=0;j--)
          {  if(a[n-1][j]<a[n-1][j+1]) dp[n-1][j]=dp[n-1][j+1]+1;
             else                      dp[n-1][j]=1;
          }
          
        for(int i=n-2;i>=0;i--){
            for(int j=m-2;j>=0;j--){
            	dp[i][j]=1;
               if(a[i+1][j]>a[i][j] && a[i][j+1]<=a[i][j])
                 dp[i][j]=1+dp[i+1][j];
               if(a[i][j+1]>a[i][j] && a[i+1][j]<=a[i][j])
                 dp[i][j]=1+dp[i][j+1];
                if(a[i+1][j]>a[i][j] && a[i][j+1]>a[i][j])
                 dp[i][j]=1+max(dp[i+1][j],dp[i][j+1]);
            }
        }
        cout<<dp[0][0]<<endl;
    }
    return 0;
}
