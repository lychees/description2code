#include <bits/stdc++.h>
using namespace std;
#define ll long long
vector<int> v(1000101);
int main()
{
    map<int,int> m;
    int i,j,t,n,num;
    ll ans;
    for(i=2;i<=1000100;i++)
    {
    	if(v[i]==0)
    	{
    		m[i-1]=i;
    		for(j=2*i;j<=1000100;j=j+i)
    			v[j]=1;
    	}
    }
    m[1]=1;
    map<int,int>::iterator it;
    scanf("%d",&t);
    while(t--)
    {
    	ans=0;
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    	{	scanf("%d",&num);
    	it=m.lower_bound(num);
    	ans=ans+(it->second);
    	}
    	printf("%lld\n",ans);
    }
    return 0;
}
