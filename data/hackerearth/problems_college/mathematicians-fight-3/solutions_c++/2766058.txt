#include<bits/stdc++.h>

using namespace std;


int check(int m)
    {
    
        if(m==1||m==2)
            return 1;
    if(m%2==0)
        return 2*check(m/2)-1;
    else
        return 2*check(m/2)+1;
    }

int main()
    {
        int t;
        cin>>t;
        while(t--)
            {
                int n;
                cin>>n;
                cout<<check(n)<<endl;
            }
    
    
    
    
    }
