#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int t,n,i,j,k,s,f,a[100000];
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld",&n);
    	s=0;f=0;k=0;
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&a[i]);
    		//if(i>0)
    		s+=a[i];
    	}
    	for(i=0;i<n-1;i++)
    	{
    		if(i>0)
    		k+=a[i-1];
    		s-=a[i];
    		if(k==s)
    		{
    			f=1;
    			break;
    		}
    	}
    	if(f==1)
    	printf("YES\n");
    	else
    	printf("NO\n");
    }
    return 0;
}
