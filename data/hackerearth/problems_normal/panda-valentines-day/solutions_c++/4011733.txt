#include<bits/stdc++.h>
using namespace std;
#define MAX 1000006
int sieve[MAX];
int ans[MAX];
void siever(void)
{
	int i,j;
	int tmp;
	for(i=2;i<MAX;++i)
	{
		if(sieve[i] == 0)
		{
			for(j=i;j<MAX;j+=i)
			{
				
				tmp = j;
				while(tmp%i==0)
				{
					sieve[j]++;
					tmp/=i;
				}
			}
		}
	}
}

int main()
{
    int i;
    int n;
    siever();
    int t;
    ans[2]=1;
    ans[3]=2;
    for(i=4;i<MAX;++i)
    	ans[i] = ans[i-1] + sieve[i];
    cin>>t;
    while(t--)
    {
    	
    	cin>>n;
    	cout<<ans[n]<<endl;
    }
    return 0;
}
