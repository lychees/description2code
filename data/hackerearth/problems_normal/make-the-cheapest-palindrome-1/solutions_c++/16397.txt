#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int T,aCost,bCost, lengthOfString,result;
	bool flag;
	cin>>T;
	while(T--)
	{
		flag = 1;
		result = 0;
		string givenString;
		cin>>givenString;
		cin>>aCost;
		cin>>bCost;
		lengthOfString = givenString.length();
		for(int i=0;i<lengthOfString/2;i++)
		{
			if(((givenString[i] == 'a')&&(givenString[lengthOfString-1-i] == '/'))||((givenString[i] == '/')&&(givenString[lengthOfString-1-i] == 'a')))
			{
				//cout<<"Step 1"<<endl;	
				result += aCost;	
			}
			else if(((givenString[i] == 'b')&&(givenString[lengthOfString-1-i] == '/'))||((givenString[i] == '/')&&(givenString[lengthOfString-1-i] == 'b')))
			{
				//cout<<"Step 2"<<endl;
				result += bCost;
			}				
			else if((givenString[i] == '/')&&(givenString[lengthOfString-1-i] == '/'))
			{
				if(aCost <bCost)
				{
					result += 2*aCost;
				}
				else
				{
					result += 2*bCost;
				}
			//	cout<<"Step 3"<<endl;
			}
			else if(((givenString[i] == 'a')&&(givenString[lengthOfString-1-i] == 'b'))||((givenString[i] == 'b')&&(givenString[lengthOfString-1-i] == 'a')))
			{
				cout<<-1<<endl;
				flag = 0;
				break;
			}
		}
		if(flag == 1)
		cout<<result<<endl;
	}
	return 0;
}
