#include<bits/stdc++.h>
#define ll long long
#define mp make_pair
#define pb push_back
using namespace std;
bool v[100003];
vector <int> g[100003];
ll mini=10000000;
ll w[100003];
void dfs(int s)
{
	if(v[s])
		return;
	v[s]=1;
	if(w[s]<mini)
		mini=w[s];
	for(int i=0;i<g[s].size();i++)
		if(!v[g[s][i]])
			dfs(g[s][i]);
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;
		cin>>n>>m;
		memset(v,0,n+2);
		memset(w,0,n+2);
		for(int i=0;i<n;i++)
			cin>>w[i],g[i].clear();
		for(int i=0;i<m;i++)
		{
			int u,v;
			cin>>u>>v;
			u--;
			v--;
			g[u].pb(v);
			g[v].pb(u);
		}
		vector <ll> rec;
		for(int i=0;i<n;i++)
		{
			if(!v[i])
			{
				dfs(i);
				rec.pb(mini);
			}
			mini=10000000;
		}
		ll ans=0;
		if(rec.size()>1)
		{
			sort(rec.begin(),rec.end());
			for(int i=1;i<rec.size();i++)
				ans+=rec[i];
			cout<<ans*rec[0]<<endl;
		}
		else
			cout<<0<<endl;
	}
	return 0;
}
