#include <stdio.h>
#include <cstring>
#include <iostream>
using namespace std;


int main()
{
    int t,n,k,p,temp;
    scanf("%d",&t);
    while(t--)
	{
		scanf("%d%d%d",&n,&k,&p);
        while(k--)
		{
			scanf("%d",&temp);
			if(temp<=p)
				p++;
		}
		if(p<=n)
			printf("%d\n",p);
		else
			printf("-1\n");
	}
	return 0;
}
