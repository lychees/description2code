#include <iostream>
using namespace std;

int t;
int A,B;

int en[6] = {1,1,2,6,4,0};

int main()
{
    scanf("%d\n",&t);
    while(t--){
    	scanf("%d%d",&A,&B);
    	if(A >= 5){ printf("0\n");}
    	else{
    		int sum = 0;
    		while(A<= B && A<= 5){
    			sum+= en[A];
    			A++;}
    		sum =sum%10;
    		printf("%d\n",sum);
    		}
    }
    return 0;
}
