#include <iostream>
using namespace std;

int main()
{
               int n,b,x,y,ans;
               int t;
               cin>>t;
               while(t--)
               {
               	 cin>>n>>b;
               	 x=n;
               	 y=n;
               	 if(x%b==0)
               	  cout<<x<<endl;
               	  else
               	  {
               	     while(1)
               	    {
               	          x=x-1;
               	 	      y=y+1;
               	        if(x%b==0 && y%b==0)
               	       {
               	   	       ans=x>y?x:y;
               	   	      break;
               	       }
               	         if(x%b==0)
               	       {
               	   	     ans=x;
               	   	     break;
               	       }
               	       if(y%b==0)
               	      {
               	   	     ans=y;
               	   	    break;
               	      }
               	    }
               	   cout<<ans<<endl;
               	  }
               }
    return 0;
}
