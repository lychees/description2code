#include<iostream>
#include<math.h>
#include<algorithm>
#define mod 1000000007

using namespace std;

int pmax(int numb)
{
	int i, max=0, max_i=0, count;
	for(i=2; i<=sqrt(numb) && numb>1; i++)
	{
		count = 0;
		while(numb%i == 0 && numb>0)
		{
			count++;
			numb /= i;
		}
		if(count>max)
		{
			max = count;
			max_i = i;
		}
	}
	return max_i;
}

int main()
{
	int n, l, i, pos=0;
	scanf("%d %d", &n, &l);
	int arr[n];
	for(i=0; i<n; i++)
	{
		scanf("%d", &arr[i]);
	}
	
	sort(&arr[0], &arr[n]);
	
	arr[n-1] = (arr[n-1] * 2) % mod;
	
	for(i=2; i<=l && pos<n; i++)
	{
		if( (pmax(i)%i) % 2 == 0)
		{
			arr[pos] /= 2;
			if(arr[pos]==0)
			{
				pos++;
			}
		}
		else
		{
			arr[n-1] = (arr[n-1] * 2) % mod;
		}
	}
	
	printf("%d\n", n-pos);
	
	for(i=0; i<n; i++)
	{
		if(arr[i])
		{
			printf("%d ", arr[i]);
		}
	}
	return 0;
}
