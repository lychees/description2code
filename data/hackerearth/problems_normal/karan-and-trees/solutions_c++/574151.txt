#include <iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long a,b;
        cin>>a>>b;
        long long i,j;
        i=a;
        j=b;

        while(1)
        {
            if(i>j)
           {
               if(i%2==0)
                i=i/2;
            else
                i=(i-1)/2;
           }
            else if(i<j)
             {if(j%2==0)
                j=j/2;
            else
                j=(j-1)/2;
             }
             else
                break;



        }
        cout<<i<<"\n";

    }
    return 0;
}
