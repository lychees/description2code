#include <iostream>
using namespace std;
const int MAX=10000;
int main()
{
    int T,N,A[MAX],U[MAX],i,j,ans;
    long long int sum;
    cin>>T;
    while(T--)
    {
    	sum=ans=0;
    	cin>>N;
    	for(i=0;i<N;i++)
    	{
    		cin>>A[i];
    		sum+=A[i];
    	}
    	
    	for(i=0;i<=MAX;i++) U[i]=0;
    	U[0]=1;
    	
    	for(i=0;i<N;i++)
    	{
    		for(j=sum;j>=A[i];j--)
    		{
    			if(U[j-A[i]]==1 && U[j]==0)
    			{
    				U[j]=1;
    				ans++;
    			}
    		}
    	}
    	cout<<ans+1<<endl;
    }
    return 0;
}
