    using namespace std;
    const int maxn=1E6+100;
    int A[maxn],N,MOD;
    long long cnt[maxn];
    void solve()
    {
    cin>>A[0];
    //cout<<" ";
    cin>>A[1];
    // cout<<" ";
    cin>>N;
    // cout<<" ";
    cin>>MOD;
    
    for(int j=0;j<MOD;j++)
    {
    cnt[j]=0;
    }
    for(int j=2;j<N;j++)
    {
    A[j]=A[j-1]+A[j-2];
    A[j]%=MOD;
    }
    for(int j=0;j<N;j++)
    {
    cnt[A[j]]++;
    }
    long long ans=0;
    for(int j=0;j<MOD;j++)
    {
    ans+=cnt[j]*cnt[j];
    }
    cout<<ans<<endl;
     
    }	
     
    int main()
    {
    int T;
     
    cin>>T;
    for(int i=0;i<T;i++)
    {
    solve();
    }
   
    return 0;
    }
