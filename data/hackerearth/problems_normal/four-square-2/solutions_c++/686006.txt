#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int t;
    long long int N;
    cin>>t;
    while(t--){
    	cin>>N;
    	long long int sum = 0;
    	for(long int i=1;i<=(long long int)sqrt(N);i++){
    		if(N%i==0){
    			if( i%4!=0)
    				sum+=i;
    			if(i*i!=N){
					if((N/i)%4!=0)
						sum+=(N/i);
				}
    		}
    	}
    	
    	cout<<(8*sum)<<endl;
    }
    return 0;
}
