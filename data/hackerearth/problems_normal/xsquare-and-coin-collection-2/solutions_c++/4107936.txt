#include <bits/stdc++.h>
using namespace std;
#define lli long long int
int main()
{
   vector<lli> v;
   lli t;
   lli n,k,best_upto=0,curr_best=0;
   cin>>t;
   while(t--)
   {
	   	cin>>n>>k;
	   	for(lli i=0;i<n;i++)
	   	{
	   		lli temp;
	   		cin>>temp;
	   		v.push_back(temp);
	   	}
	   	for(lli i=0;i<n;i++)
	   	{
	   		if(v[i]>k)
	   		{
	   			if(curr_best>best_upto)
	   			{
	   				best_upto=curr_best;
	   			}
	   			curr_best=0;
	   		}
	   		else
	   		curr_best=curr_best+v[i];
	   	}
	   	if(curr_best>best_upto)
	   			{
	   				best_upto=curr_best;
	   			}
	   	cout<<best_upto<<endl;
	   	v.clear();
	   	curr_best=0;
	   	best_upto=0;
   }
   
}
