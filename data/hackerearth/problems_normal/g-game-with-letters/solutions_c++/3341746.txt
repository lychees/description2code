#include <iostream>
using namespace std;
#include<string.h>
int main()
{
	int t,n,size,k,i,a[26],max,sum,m,index,pos;
	char s[1000002],b[50];
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d%s",&n,s);
    	max=-1;size=n+1;
    	for(i=0;i<26;i++)
    	a[i]=0;
    	m=strlen(s);
    	for(i=0;i<m;i++)
    	{
    		a[(int)(s[i]-97)]++;
    	}
    	index=0;pos=0;
    	while(n--)
    	{
    		scanf("%s",b);
    		sum=0;
    		k=strlen(b);
    		for(i=0;i<k;i++)
    		{
    			sum=sum+a[(int)(b[i]-97)];
    		}
    		if(max<=sum)
    		{
    			if(max==sum)
    			{
    				if(k<size)
    				{
    					size=k;
    					pos=index;
    					max=sum;
    				}
    			}
    			else {
    				max=sum;
    				size=k;
    				pos=index;
    			}
    		}
    		index++;
    	}
    	printf("%d\n",pos+1);
    }
    return 0;
}
