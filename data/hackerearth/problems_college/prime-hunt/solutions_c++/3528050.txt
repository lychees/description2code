#include <iostream>
using namespace std;
bool isprime(long int n)
{
	for (long int i=3;i*i<=n;i+=2)
	{
		if (n%i==0)
		return 0;
	}
	return 1;
}
int main()
{
	int t;
	cin>>t;
	long int j,ctr,a,b;
	for (int i=0;i<t;i++)
	{
		cin>>a;
		cin>>b;
		ctr=0;
		if (a<=2)
		{
			ctr++;
			for (j=3;j<=b;j+=2)
			{
				if (isprime(j)==1)
				ctr++;
			}
		}
		else
		{
			if (a%2==0)
			a+=1;
			for (j=a;j<=b;j+=2)
			{
				if (isprime(j)==1)
				ctr++;
			}
		}
		cout<<ctr<<endl;
	}
	
}
