#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll solve(ll sum, vector<int> v, ll x, ll req){
	ll ret=0;
	ll arr[2][200]={0}, prev=1, cur=0, pmax=0;
	for(int j=0; j<=9; j++)
		arr[prev][j+sum]=1;
	pmax=9;
	for(int i=v.size()-2; i>x; i--){
		for(int j=0; j<=9; j++){
			for(int k=j; k<=j+pmax; k++){
				arr[cur][k+sum]+=arr[prev][k-j+sum];
			}		
		}
		pmax+=9;
		for(int j=0; j<=162; j++)
			arr[prev][j]=0;
		cur=!cur;prev=!prev;
	}
	for(int j=0; j<v[x]; j++){
		for(int k=j; k<=j+pmax; k++){
			arr[cur][k+sum]+=arr[prev][k-j+sum];
		
		}
	}
	return arr[cur][req];
}

ll find(ll x, ll k){
	if(!x)return 0;
	vector<int> v;
	while(x){
		v.push_back(x%10);
		x/=10;	
	}	
	reverse(v.begin(), v.end());	
	ll sum=0;
	ll ret=0;
	for(int i=0; i<v.size()-1; i++){
		ret+=solve(sum,v,i,k);	
		sum+=v[i];
	}
	for(int i=0; i<=v.back(); i++)
		if(sum+i == k)ret++;
	return ret;
}
int main(){
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	ll l, r, k;cin>>l>>r>>k;	
	if(k > 162)cout<<"0"<<endl;
	else{
		cout<<(find(r, k)-find(l-1, k))<<endl;		
	}	
	return 0;
}
