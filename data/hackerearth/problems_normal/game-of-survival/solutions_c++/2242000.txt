#include<bits/stdc++.h>
using namespace std;

int main()
{

    int t;
    cin>>t;

    while(t--)
    {

            int n;
            scanf("%d",&n);

            long long int s[n],c[n];

            for(int i=0;i<n;i++)
                scanf("%lld",&s[i]);
            for(int i=0;i<n;i++)
                scanf("%lld",&c[i]);

            bool flag=true;

            sort(s,s+n);
            sort(c,c+n);

            for(int i=0;i<n;i++)
                {
                    if(s[i] <= c[i])
                    {
                        flag=false;
                        break;
                    }
                    
                }

            if(flag)
                cout<<"YES"<<endl;
            else
                cout<<"NO"<<endl;


    }


}
