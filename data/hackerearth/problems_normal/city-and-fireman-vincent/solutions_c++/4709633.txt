#include<bits/stdc++.h>
using namespace std;
int n,m,a[1004],x,y,l,cnt,ans=1;
bool visited[1005];
vector<int>adj[1004];
void dfs(int i)
{
    visited[i]=1;
    if(a[i]==l)cnt++;
    if(a[i]<l)
    {
        l=a[i];
        cnt=1;
    }
    for(int j=0;j<adj[i].size();j++)
    {
        if(visited[adj[i][j]]==0)
            dfs(adj[i][j]);
    }
}
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++)
        cin>>a[i];
    cin>>m;
    while(m--)
    {
        cin>>x>>y;
        adj[x].push_back(y);
        adj[y].push_back(x);
    }
    for(int i=1;i<=n;i++)
    {
        if(!visited[i])
        {
            cnt=1;
            l=100000;
            dfs(i);
            ans*=cnt;
            ans%=1000000007;
        }
    }
    cout<<ans<<endl;
    return 0;
}
