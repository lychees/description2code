#include <bits/stdc++.h>
using namespace std;

void reverse(char s[])
{
	int start=0;
	char temp;
	int end=strlen(s)-1;
	while(start<end)
	{
		temp=s[start];
		s[start]=s[end];
		s[end]=temp;
		start++;
		end--;
	}
}
int main() {
long long int n;
char s[100005],r[100005];
cin>>n;
while(n--)
{
cin>>s;
strcpy(r,s);
reverse(r);
int i=1,j=1;
int a,b,flag=0;
//cout<<strlen(s)-1;
while(i<= (strlen(s)-1) )
{int p,q;
	a=s[i];
	b=s[i-1];
	a=abs(a-b);
	//	cout<<a<<" "<<b<<endl;
	p=r[j];
	q=r[j-1];
	b=abs(p-q);
	//cout<<a<<" "<<b<<endl;
//	cout<<i<<endl;
	if(a!=b)
	{
		flag=1;
		break;
	}
	i++;
	j++;
}
if(flag==1)
cout<<"Not Funny"<<endl;
else

}
	return 0;
}
