#include<iostream>
#include<cmath>
using namespace std;
typedef long long LL_t;
#define abmody(a, b, y) ((a * b) % y)

int find_digit(LL_t x, LL_t y, LL_t pos)
{
  	LL_t m = pos - 1;
  	LL_t A = 1, B = 10;
  	while (m > 0) {
		if (m & 1)
			A = abmody(A,B,y);
		B = abmody(B,B,y);
		m >>= 1;
	}

	x %= y;
	A = abmody(A, x, y);

	int ans = floor(10 * A / y);
	ans %= 10;
	
 	return ans;
}

int main()
{
	long T;
	cin>>T;
	
	LL_t N, D, R;
	while(T--) {
	 	cin >> N >>D >> R;
	 	cout << find_digit(N, D, R) << endl;
	}
	return 0;
}
