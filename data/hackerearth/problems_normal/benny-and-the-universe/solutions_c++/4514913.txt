#include<bits/stdc++.h>
using namespace std;
#define inf 1000000007
#define ll long long int
int mp[10004];
int arr[10004];
int main()
{
		int n;
		int i;
		int q;
		scanf("%d %d",&n,&q);
		for(i=0;i<n;++i)
		{
				scanf("%d",arr+i);
		}
		sort(arr,arr+n);
		int base=arr[0];
		fill(mp , mp+10004 , inf);
		int mv;
		//mp[0]=0;
		priority_queue<pair<ll,int> > pq;
		pq.push(make_pair(0,0));
		while(!pq.empty())
		{
			int eqc;
			ll num;
			num=-pq.top().first;
			eqc=pq.top().second;
			pq.pop();
			if(mp[eqc]<num) continue;
			for(i=0;i<n;++i)
			{
				ll new_num,neqc;
				new_num = num+arr[i];
				neqc = new_num%base;
				if(mp[neqc] > new_num && new_num < inf)
				{
					mp[neqc] = new_num;
					pq.push(make_pair(-new_num,neqc));
				}
			}
			
		}
		while(q--)
		{
				int x;
				scanf("%d",&x);
				base = x%arr[0];
				if(base == 0) puts("YES");
				else if(mp[base] != inf && x >= mp[base]) puts("YES");
				else puts("NO");
		}

}
