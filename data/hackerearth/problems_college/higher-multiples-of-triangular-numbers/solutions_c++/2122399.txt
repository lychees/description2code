#include<bits/stdc++.h>
using namespace std;
int main(void)
{
    int t,n;
    long long int s,x,i,a;
    cin>>t;
    while(t--)
    {
        cin>>n;
        x=3;
        a=3;
        s=0;
        while(s<=n)
        {   s=2;
            for(i=2;i*i<=x;i++)
            {
                if(x%i==0)
                {
                    if(i*i==x)
                        s++;
                    else s+=2;
                }
            }
            if(s>n)
            {
                cout<<x<<endl;
                break;
            }
            else x+=a;
            a++;
            }
    }

}
