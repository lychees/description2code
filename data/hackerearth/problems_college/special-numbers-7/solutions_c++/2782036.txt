#include<bits/stdc++.h>

#define FOR(i,a,b) for(int i=a;i<b;i++)
#define rep(i,n) for(int i=0;i<n;i++)
#define pint(n) printf("%d\n",n)
#define plong(n) printf("%lld\n",n)
#define sint(n) scanf("%d",&n)
#define slong(n) scanf("%lld",&n)
#define MAX 10000
typedef long long int ll;
int prime[MAX]={0};
void sieve()
{
    for(int i=0;i<MAX;i++)
        prime[i]=1;
    for(int i=2;i<MAX;i+=2)
        prime[i]=0;
    for(int i=3;i*i<MAX;i+=2)
    {
        if(prime[i])
        {
            for(int j=i*i;j<MAX;j+=i)
                prime[j]=0;
        }
    }
     prime[2]=1;
     prime[1]=0;
}
int no_of_div(int x)
{
    int count=0;
    for(int i=1;i<=sqrt(x);i++)
    {
        if(x%i==0)
        {
            if(prime[i])
                count++;
            if(i!=(x/i) && prime[x/i])
                count++;
            if(count>=3)
                return count;
        }
    }
    return count;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        int cnt=0;
        cin>>n;
        sieve();
        for(int i=0;i<100000;i++)
        {
            if(no_of_div(i)>=3)
                cnt++;
            if(cnt==n)
            {
                printf("%d\n",i);
                break;
            }
        }

    }
    return 0;
}
