#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,q,i,j;
	cin>>n>>m;
	vector<long long>v[n];
	long long p;
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
		cin>>p;
		v[i].push_back(p);
		}
	}	
	vector<long long>::iterator it;
	cin>>q;
	while(q--)
	{
	int ch=0;
	cin>>p;
	for(i=0;i<n;i++)
	{
		it=lower_bound(v[i].begin(),v[i].end(),p);
		if((*it)==p)
		{
		ch=1;
		cout<<i<<" "<<(it-v[i].begin())<<endl;
		break;
		}
	}
	if(ch==0)
	cout<<-1<<" "<<-1<<endl;
	
}
return 0;
}
