#include <bits/stdc++.h>
#include <stdio.h>
using namespace std;
int main(int argc, char const *argv[])
{
	int i,x,y;
	cin>>i;
	while(i--)
	{
		scanf("%d%d",&x,&y);
		if(x == 0 && y == 0)
			cout<<1;

		else if(x == 0)
			cout<<2;

		else if(x < 0)
			cout<<0;
		else 
			cout<<4;

		cout<<endl;
	}
	return 0;
}
