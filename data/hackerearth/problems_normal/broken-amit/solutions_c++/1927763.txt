#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
struct node{
ll lv,dv;
int ind;

}a[1000];

bool co(struct node a,struct node b)
{

   return ( a.dv*a.ind)>(b.dv*b.ind);
}
int main()
{
    ll sum=0;ll dec=0;
    int t;
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {cin>>a[i].lv;a[i].ind=i;
    sum+=a[i].lv;}
    for(int i=0;i<n;i++)
    {cin>>a[i].dv;dec+=(a[i].dv*i);}
    sort(a,a+n,co);
   ll  tot=sum;
   for(int i=1;i<n;i++)
   {
            dec-=(a[i-1].dv*a[i-1].ind);
       sum-=dec;
       tot+=sum;

   }
   cout<<tot<<endl;
    return 0;
}
