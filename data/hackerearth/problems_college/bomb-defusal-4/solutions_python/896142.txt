t = input()
while t:
	P,x,y,u,v = map(int,raw_input().split())
	if abs(x-u) == abs(y-v):
		if P>1 or abs(x-u)==0: print "Counter-Terrorists Win !"
		else: print "Terrorists Win !"
	elif (x%2 == y%2 and u%2 == v%2) or (x%2 != y%2 and u%2 != v%2):
		if P>2: print "Counter-Terrorists Win !"
		else: print "Terrorists Win !"
	else: print "Terrorists Win !"
	t -= 1
