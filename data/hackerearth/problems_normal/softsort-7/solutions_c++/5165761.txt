#include <iostream>
using namespace std;
long long int fact[1000001];
void preProcess()
{
	fact[0]=1;
	for(int i=1; i<1000001; i++)
	{
		fact[i]=(fact[i-1]*i)%1000000007;
	}
}
int main()
{
	int t;
	cin>>t;
	preProcess();
	while(t--)
	{
	   long long int num;
	   long long int lines=0;
	    cin>>num;
	  
	    lines=((3*fact[num])%1000000007+3)%1000000007;
	    cout<<lines<<endl;
	    
	    
	    
	}
	return 0;
}
