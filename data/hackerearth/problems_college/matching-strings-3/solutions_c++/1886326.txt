#include <iostream>
#include<math.h>
#define ll long long
using namespace std;
ll min(ll a,ll b)
{
    if(a<b)
        return a;
    return b;

}

int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
        ll count=0;
        string a,b;
        cin>>a;
        cin>>b;
        for(ll i=0;i<min(a.length(),b.length());i++)
        {
            if(a[i]==b[i])
                count++;

        }
        cout<<count<<endl;

    }

    return 0;
}
