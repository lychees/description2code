#include <iostream>
#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int k;
    	cin>>k;
    	
    	long long int output[100001];
    	output[0] = 1;
    	output[1] = 2;
    	output[2] = 4;
    	
    	for(int i = 3; i < k; i++){
    		output[i] = output[i-1] + output[i-2] + output[i-3];
    		output[i] = output[i]%1000000007;
    	}
    	
    	cout<<output[k-1]<<"\n";
    }
    return 0;
}
