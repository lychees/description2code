#include <iostream>
using namespace std;
# define ll long long
# define mod 1000000007
ll bp(ll base,ll exp)
{
	ll ret=1;
	while(exp)
	{
		if(exp&1)
		ret=(ret*base)%mod;
		base=(base*base)%mod;
		exp>>=1;
	}
	return ret;
}
int main()
{
	int t;
	ll n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<bp(3,n)<<'\n';
	}
    return 0;
}
