#include<cstdio>
#include<cmath>
#define phi 1.61803398874989484 
int main(){
  	int t,a,b;
	scanf("%d",&t);
	while(t--){
	scanf("%d%d",&a,&b);
	int max;
	int min;
	if(a>b){
	max=a;
	min=b;
	}
	else {
	max=b;
	min=a;
	}
	int r;
	r=max;
	r=r-min;
	int p;
	p=phi*r;
	int x;
	x=int(floor(p));
	if(x==min){
	printf("Don't Play\n");
	}
	else{
	printf("Play\n");
	}

}}
