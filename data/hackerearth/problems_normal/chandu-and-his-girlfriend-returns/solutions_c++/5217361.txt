#include <iostream>
using namespace std;

int main()
{
    int t,n,m;
    cin >> t;
    while(t--){
        cin >> n >> m;
        int a[n],b[m],c[m+n];
        for(int p=0; p<n; p++)
            cin >> a[p];
        for(int i=0; i<m; i++)
            cin >> b[i];
        int i=0,j=0,k=0;
        while(i<n && j<m){
        if(a[i]>=b[j])
            c[k++] = a[i++];
        else
            c[k++] = b[j++];
		}
		while (i<n) c[k++] = a[i++];
		while (j<m) c[k++] = b[j++];
		for(int p=0; p<m+n; p++)
            cout << c[p] << " ";
        cout << endl;
    }

    return 0;
}
