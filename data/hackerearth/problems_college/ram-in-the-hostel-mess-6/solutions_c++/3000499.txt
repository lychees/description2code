int64 dp[100005][2];

int main(){
	dp[1][0]=1;
	dp[1][1]=1;
	for(int i=2;i<=100000;i++){
		dp[i][0]=dp[i-1][1];
		dp[i][1]=dp[i-1][0]+dp[i-1][1];
	}
	int t,n;
	cin>>t;
	while(t--){
		cin>>n;
		int64 ans=dp[n][0]+dp[n][1];
		ans=ans*ans;
		printf("%lld\n",ans);
	}
