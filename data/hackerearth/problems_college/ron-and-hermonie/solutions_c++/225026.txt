#include<cmath>
#include<cstdio>
unsigned long long modexpo(long long a,long long b,long long n)
{
    long long d=1;
    while(b)
    {
        if(b%2)
            d=(d*a)%n;
        b>>=1;
        a=(a*a)%n;
    }
    return d;
}
int main()
{
    unsigned long long int a , b , n, i , k;
    scanf("%llu", &n);
    for (i=0;i<n;i++)
    {
        scanf("%llu %llu", &a , &b);
                
        if(b==0&&a==0)
            k=1;
        else if (b==0 )
            k=1;
        else if (a==0)
            k=0;
        else
        {                    
            if (b%4==0)
                b = 4;
            else
                b=b%4;
            k = modexpo(a, b,10);
        }    
            //k = k%10;    
            printf("%llu\n", k);        
    }
    return 0;
}
