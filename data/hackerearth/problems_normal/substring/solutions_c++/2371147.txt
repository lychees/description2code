#include <bits/stdc++.h>
using namespace std;

#define F(i, n) for(int i=0; i<n; i++)
#define ll long long
int main()
{
    int t;
   	scanf("%d\n", &t);
   	string s;
   	while(t--)
   	{
   		getline(cin, s);
   		ll _count=0;
   		ll ans=0;
   		for(int i=0; i<s.size(); i++)
   		{
   			if(s[i]=='A'||s[i]=='a'||s[i]=='B'||s[i]=='b'||s[i]=='C'||s[i]=='c')
   			_count++;
   			else
   			{
   				ans+=(_count*(_count+1))/2;
   				_count=0;
   			}
   		}
   		ans+=(_count*(_count+1))/2;
   		printf("%lld\n", ans);
   	}
    return 0;
}
