#include <iostream>
#include <cmath>
#include <algorithm>
#include <cstdio>
#include <functional>
 
using namespace std;
int l[1000000];
int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n;
		cin >> n; 
	if(n==1)
		{ cout << 4 << endl;}
	else
		{	
			l[0] = 0;
			l[1] = 2;
			l[2] = 3;
 
			for(int i =3;i<=n;i++)
			{
				l[i] = l[i-1] + l[i-2];
			}
 
				cout << l[n] * l[n] <<endl;
		}
 
 
	}
 
	return 0;
}
