def gcd(x, y):
    if y == 0:
        return x
    else:
        return gcd(y, x % y)


t = int(raw_input())
while t > 0:
    t = t-1
    l, b, h = (int(i) for i in raw_input().split())
# print l, b, h
    c = gcd(l, b)
    c = gcd(c, h)
    l = l/c
    b = b/c
    h = h/c

    ans = (l*b*h) % 1000000007
    print c, ans
