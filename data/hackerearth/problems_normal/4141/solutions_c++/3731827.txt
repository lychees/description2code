#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main(){
	ll t,c,n,a[50005],i;
	ll ans;
	cin >> t;
	while(t--){
		cin >> a[0] >> a[1] >> c >> n;
	     if(n==1)
	      ans = a[0];
	     else 
		  ans = a[0]*10+a[1];
		
		for(i=2; i<n; i++){
			a[i] = a[i-1]*c+a[i-2];
			a[i] = a[i]%10;
			ans = ans*10+a[i];
			ans = ans%41;
		}
		if(ans%41==0)
		 printf("YES\n");
		else
		 printf("NO\n");
	}
	return 0;
}
