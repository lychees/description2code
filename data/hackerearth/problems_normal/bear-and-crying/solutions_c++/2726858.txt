#include <iostream>

#define mod 1000000007
using namespace std;

int main()
{
  	string s;
  	cin>>s;
  	int n = s.length();
  	int curr_sum = 0, colons = 0,ans = 0;
  	for(int i=0;i<n;i++)
  	{
  		if(s[i] == ';')
  		{
  			++colons;
  			ans = (curr_sum+ans)%mod;
  		}
  		else
  		{
  			curr_sum = (colons+curr_sum*2)%mod;
  		}
  	}
  	
  	cout<<ans;
  	return 0;
}
