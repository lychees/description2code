#include <bits/stdc++.h>

using namespace std;

long long int a[10000007];

int main()
{
    long long int t,n,i;

    a[1] = 2;
    a[2] = 3;

    for(i = 3; i < 10000007; ++i){
        a[i] = a[i-1] + a[i-2];
    }

    cin >> t;

    while(t--){
        cin >> n;
        cout << a[n]*a[n] << endl;
    }
    return 0;
}
