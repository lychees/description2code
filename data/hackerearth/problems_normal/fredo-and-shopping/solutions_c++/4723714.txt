#include <bits/stdc++.h>
using namespace std;
int bit[1000005],a[1000005];
int n;
void update(int pos)
{
	while(pos<=n){
		bit[pos]++;
		pos += pos&(-pos);
	}
}
int query(int pos)
{
	//printf("query %d\n",pos);
	int res = 0;
	while(pos > 0)
	{
		res += bit[pos];
		pos -= pos&(-pos);
	}
	return res;
}
int main()
{
   // cout << "Hello World!" << endl;
   int i,nn;
   set<int> s;
   int in = 0;
   scanf("%d",&nn);
   for(i=0;i<nn;i++){scanf("%d",&a[i]);bit[i] = a[i]; }
   sort(bit,bit+nn);
   int pr = 0;
   unordered_map<int,int> ran;
   for(i=0;i<nn;i++)
   {
   	if(bit[i]==pr);
   	else {
   		in++;
   		ran[bit[i]] = in;
   	//	printf("rank %d %d\n",bit[i],in);
   		pr = bit[i];
   	}
   	bit[i] = 0;
   }
   for(i=0;i<nn;i++){
   a[i] = ran[a[i]];
   //printf("%d ",a[i]);
   }
   n = nn;
   for(i=0;i<nn;i++)
   {
   	printf("%d ",query(a[i]-1));
   	update(a[i]);
   }
   
    return 0;
}
