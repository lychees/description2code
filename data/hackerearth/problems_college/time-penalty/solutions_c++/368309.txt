#include <iostream>
using namespace std;

int main()
{
   int date,hour,min;
   unsigned long long attempts, penaltytime;
   
   cin >> date >> hour >> min;
   cin >> attempts;
   
   if (date>5)
   cout << "Invalid Date";
   else if (hour > 23)
   cout << "Invalid hour";
   else if (min > 59)
   cout << "Invalid minute";
   else
   {
   	penaltytime = (attempts*20)+((date-1)*24*60)+(hour*60)+min;
   	cout << penaltytime;
   }
   
}
