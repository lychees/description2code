Len = int(raw_input(''))
Array = (raw_input('')).split()
OperationCount = int(raw_input(''))
DevDigit = 1
for _ in range(OperationCount):
    DevDigit *= int(raw_input(''))

for ArrayIndex in range(len(Array)):
    Array[ArrayIndex] = str(int(int(Array[ArrayIndex]) / DevDigit))

print ' '.join(Array)
