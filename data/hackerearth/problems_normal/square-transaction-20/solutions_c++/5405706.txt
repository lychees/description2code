#include <iostream>
using namespace std;

int main()
{
    long int t,i;
    cin>>t;
    long int a[t];
    for(i=0;i<t;i++)
    {
    	cin>>a[i];
    }
    for(i=0;i+1<t;i++)
    {
    	a[i+1]=a[i+1]+a[i];
    }
    //for(i=0;i<t;i++)
    //cout<<a[i];
    long int q;
    cin>>q;
    while(q--)
    {
    	long int k;
    	cin>>k;
    	if(k>a[t-1])
    	{
    		cout<<"-1"<<endl;
    	}
    	else
    	{
    	    long int low=0,high=t-1,mid,result=-1;
    	    while(low<=high)
    	    {
    		    mid=low+(high-low)/2;
    		    if(a[mid]==k)
    		    {
    			    result=mid;
    			    break;
    		    }
    		    else if(a[mid]>k)
    		    {
    			    result=mid;
    			    high=mid-1;
    		    }
    		    else
    		    {
    			    low=mid+1;
    		    }
    	    }
    	    cout<<result+1<<endl;
    	}
    }
    return 0;
}
