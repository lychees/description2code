#include<math.h>
#include <iostream>
using namespace std;
#define get getchar_unlocked
inline long long int scan()
{
  long long int n=0,s=1;
    char p=get();
    if(p=='-')  s=-1;
    while((p<'0'||p>'9')&&p!=EOF&&p!='-') p=get();
    if(p=='-') s=-1,p=get();
    while(p>='0'&&p<='9') { n = (n<< 3) + (n<< 1) + (p - '0'); p=get(); }
    return n*s;
}

int main() {
	
	long long int t;
t=scan();//	cin>>t;
	while(t--)
	{
long long int n;
n=scan();//	cin>>n;
	
	long long int arr[20];
	long long int x=n;
	long long int len=0;
	long long int i=0;
	
	while(x)
	{
		arr[i]=x%10;
		x=x/10;
		
		len++;
	i++;
	}
	i=0;
	long long int j=len-1;
	while(i<j)
	{
		long long int temp=arr[i];
		arr[i]=arr[j];
		arr[j]=temp;
		i++;
		j--;
	}long long int flag=0;
	long long int sum=0;
	for(long long int i=0;i<len;i++){
                   for(long long int j=0;j<arr[i];j++){
                    if(j==2||j==3||j==5||j==7)
                        continue;
                    sum = sum + pow(6,len-i-1);
                }
                if(arr[i]==2||arr[i]==3||arr[i]==5||arr[i]==7){
                    flag = 1;
                    break;
                }
            }
            sum = n-sum;
            
            sum = sum+flag;	
		
		cout<<sum<<"\n";
		
		
	}
	// your code goes here
	return 0;
}
