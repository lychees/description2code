#include <iostream>
#include<cstring>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
  {
  	char str[32];
  	int arr[256]={0};
  	cin>>str;
  	for(int i=0;i<strlen(str);i++)
  	{
  	  if(arr[str[i]-'a']==0)
  	  {
  	  	arr[str[i]-'a']=1;
  	  	cout<<str[i];
  	  }
  		
  	}
  	cout<<endl;
  	
  }
}
