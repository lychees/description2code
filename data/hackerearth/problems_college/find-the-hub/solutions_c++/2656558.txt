#include <bits/stdc++.h>
#define li long long int
using namespace std;
int prime(li num){
	li p = 1;
	if(num==0 || num==1)
       return 0;
	for(li i = 2; i <= sqrt(num); i++){
		if((num%i) == 0){
			p = 0;
			break;}}
	return p;}
int main()
{
   int n;
   cin>>n;
   li a[n],t=n,k,i=0,max=0,c=0,flag=0;
   while(i<n){
      cin>>k;
     if(max==k)
        ++c;
      if(prime(k)==1){
        if(max<k)
        {
          max=k;
          c=1;
          t=k;}
        flag=1;}
        i++;}
    i=0;
     cout<<t<<endl<<c<<endl;
return 0;
}
