#include <bits/stdc++.h>
#define si(x) scanf("%d", &x)
#define slli(x) scanf("%lld", &x)
#define sf(x) scanf("%f", &x)
#define slf(x) scanf("%lf", &x)
#define sc(x) scanf("%c", &x)

#define mp make_pair

using namespace std;

typedef pair <int, int> pi;
typedef vector <int> vi;
typedef vector <pi> vpi;

int main () {
	int t; si(t);
	while(t--) {
		int n; si(n);
		int arr[n], ex;
		for(int i = 0; i < n; i++)
			si(arr[i]);
		bool visited[n];
		memset(visited, 0, n);
		int source = 0;
		while(true) {
			source = arr[source];
			if(arr[source] == -1) {
				printf("Win\n");
				break;
			}
			else if(visited[source]) {
				printf("Lose\n");
				break;
			}
			visited[source] = true;
		}
	}
	return 0;
}
