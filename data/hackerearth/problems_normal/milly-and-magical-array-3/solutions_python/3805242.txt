tc=int(raw_input())
while tc:
    tc=tc-1
    N,X=map(int,raw_input().split())
    n=map(int,raw_input().split())
    magic=0
    for i in xrange(1,N):
        #print "i,n[i]", i,n[i]
        if n[i]<=n[i-1]:
            add=(n[i-1] - n[i])/X
            add+=1
            magic +=add
            n[i]=add*X + n[i]
            #print "add ",add
            #print "n,n[i]",n,n[i]
    print magic
