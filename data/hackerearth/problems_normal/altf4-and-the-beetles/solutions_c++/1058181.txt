#define pii pair<int,int>
using namespace std;
int main()
{
	// freopen("input.txt","r",stdin);
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		long long int a,b,c;
		scanf("%d %lld %lld %lld",&n,&a,&b,&c);
		pii t[2*n];
		for (int i = 0; i < n; ++i)
		{
			scanf("%d %d",&t[2*i].first,&t[2*i + 1].first);
			t[2*i].second=-1;
			t[2*i + 1].second=1;
		}
		sort(t,t+2*n);
		long long int ans=((long long int)n)*a;
		long long int curr=ans;
		for (int i = 0; i < 2*n; ++i)
		{
			if(t[i].second==-1)
				curr=curr+b-a;
			else
				curr=curr+c-b;
			if(curr>ans)
				ans=curr;
		}
		printf("%lld\n", ans);
	}
	return 0;
}
