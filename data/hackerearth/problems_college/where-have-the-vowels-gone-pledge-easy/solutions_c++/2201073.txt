#include <iostream>
#include <map>

using namespace std;

int main(){
	
	int N;
	while(cin >> N){
		map<char,int> garbage; char t;
		for(int i = 0; i < N; i++){
			t = getchar();
			garbage[t]++;
		}
		if(garbage['a'] >= 1 && garbage['e'] >= 1 && garbage['i'] >= 1 && garbage['o'] >= 1 && garbage['u'] >= 1)
			cout << "YES\n";
		else
			cout << "NO\n";
	}

	return 0;
}
