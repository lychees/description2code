#include<iostream>
using namespace std;

int main(){
    long long int t, r, c, R, C, even, odd;
    cin>>t;
    
    while(t--){
        cin>>r>>c;
        even = odd =0;
        
        if(c>=r){
            //cout<<"r : "<<r<<endl;
            
            R=r;
            
            if(R%2)
                R--;
            
            //cout<<"r for even : "<<R<<endl;
            even = (R*R+2*R)/4;
            
            R = r;
            if(R%2==0)
                R--;
            //cout<<"r for odd : "<<R<<endl;
            odd = (R*R+2*R+1)/4;
            cout<<even<<" "<<odd<<endl;
        }else{
            
            R=c-1;
            
            if(R%2)
                R--;
            
            even = (R*R+2*R)/4;
            
            R = c-1;
            if(R%2==0)
                R--;
            odd = (R*R+2*R+1)/4;
            
            //cout<<"first odd : "<<odd<<endl;
            
            R=r; C=c;
            
            if(R%2)
                R--;
            if(C%2)
                C++;
            
            if(C<=R){
                int rows = (R-C)/2+1;
                even = even + rows*c;
            }
        
            R=r; C=c;
            if(R%2==0)
                R--;
            if(C%2==0)
                C++;
        
            if(C<=R){
                int rows = (R-C)/2+1;
                odd = odd + rows*c;
            }
            cout<<even<<" "<<odd<<endl;
        }
    }
    
    return 0;
}
