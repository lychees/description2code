#include <bits/stdc++.h>
using namespace std;
stack <string> v;

int main()
{
	int t,l,i;
	cin >> t;
	getchar();
	while(t--){
		string s,s1;
		getline(cin,s);
		s=s+' ';
		s1="";
		l=s.size();
		for(i=0;i<l;i++){
			if(s[i]!=' ')
			{
				s1+=s[i];	
			} 
			else
			{
				v.push(s1);
				s1="";
			}
		}
		while(!v.empty()){
			cout << v.top() << " ";
			v.pop();
		}
		cout << endl;
	}
    return 0;
}
