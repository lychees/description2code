#include <iostream>
#include <string>
#include <cstring>
using namespace std;

int main()
{
     int N,cnt=0,t=0;
    string num;
    cin>>N;
    while(N--)
    {
    	cnt=0;t=0;
    	cin>>num;
    	for(int i=0;i<num.size();i++)
    	{
    		t=num[i]-'0';
    	
    		if(t==0||t==6||t==9)
    		cnt+=6;
    		else if(t==1)
    		cnt+=2;
    		else if(t==2||t==3||t==5)
    		cnt+=5;
    		else if(t==4)
    		cnt+=4;
    		else if(t==7)
    		cnt+=3;
    		else if(t==8)
    		cnt+=7;
  		    	
    	}
   	cout<<cnt<<endl;
    }
}
