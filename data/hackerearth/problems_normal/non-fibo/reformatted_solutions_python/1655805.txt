num = [(4, 4), (6, 7)]
non = [1, 2]
tot = [1, 3]
while (tot[-1] <= (10**15)):
    num += [(num[-1][0]+num[-2][0]-1, num[-1][1]+num[-2][1]+1)]
    non += [non[-1]+non[-2]+1]
    tot += [tot[-1]+non[-1]]
l = len(tot)
t = input()
for i in xrange(t):
    n = input()
    for j in range(0, l):
        if (tot[j] >= n):
            diff = tot[j]-n
            print num[j][1]-diff
            break
