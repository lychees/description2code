#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    int t,p;
    ll n, sum,i;
    double percentage;
    
    scanf("%d", &t);
    while(t--) {
    	sum =0;
    	scanf("%lld", &n);
    	
    	for (i=0;i<n;i++){
    	  scanf("%d", &p);
    	  sum += p;
    	}
    	
    	percentage = ((double) sum)/((double) (n));
    	
    	printf("%0.8f\n", percentage);
    }
    return 0;
}
