vector<string> split(const string &s, char c) {
	vector<string> v;
	stringstream ss(s);
	string x;
	while (getline(ss, x, c)) v.push_back(x);
	return v;
}

typedef long long ll;
const int MOD = 1000000007;

template<class T, class U> inline T tmin(T a, U b) {return (a < b) ? a : b;}

struct UF {
    vector<int> par, rk, sub;
    int sz;
    
    UF() {}
    UF(int _n) {
        sz = _n;
        par.resize(sz + 1), rk.assign(sz + 1, 0);
        sub.assign(sz + 1, 1);
        for (int i = 0; i <= sz; ++i) {
            par[i] = i;
        }
    }
    
    int find(int x) {
        if (x != par[x]) par[x] = find(par[x]);
        return par[x];
    }
    
    int size(int x) {
    	return sub[x];
    }
    
    bool same(int x, int y) {
        return find(x) == find(y);
    }
    
    void unite(int x, int y) {
        x = find(x), y = find(y);
        if (x == y) return;
        if (rk[x] < rk[y]) {
        	par[x] = y; sub[y] += sub[x];
        }
        else {
            par[y] = x; sub[x] += sub[y];
            if (rk[x] == rk[y]) rk[x]++;
        }
    }
};

const int N = 100005;
ll dp[N];

int main(int argc, char *argv[]) {
    ios_base::sync_with_stdio(false);
    int n, k, u, v;
    
    dp[0] = 1;
    repu(i, 1, N) dp[i] = (i * dp[i - 1]) % MOD;
    
    cin >> n >> k;
    UF uf(n);
    repu(i, 0, k) {
    	cin >> u >> v;
    	uf.unite(u, v);
    }
    ll ans = 1;
    repu(i, 0, n) {
    	if (uf.find(i) == i) ans = (ans * dp[uf.size(i)]) % MOD;
    }
    printf("%lld\n", ans);
    return 0;
}
