#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n,a[100000],i,q,lb,ub,ans;
    long long int l,r;
    
    cin>>n;
    
    for(i=0;i<n;i++)
    	cin>>a[i];
    	
    sort(a,a+n);
    	
    cin>>q;
    
    for(i=0;i<q;i++)
    {
    	cin>>l>>r;
    	
    	lb=lower_bound(a,a+n,l)-a;
    	ub=upper_bound(a,a+n,r)-a;
    	ans=ub-lb;
    	
    	cout<<ans<<endl;
	}
	
	return 0;
}
