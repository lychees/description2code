#include <bits/stdc++.h>
#define ll long long int
#define s(a) scanf("%lld",&a)

using namespace std;

ll a[1001][1001];
vector<ll> org,vir;

int main()
{
    ll n,k,i,j,q,x,m,w;
    s(n);s(k);
    for(i=1;i<=n;i++) {
        for(j=1;j<=k;j++) {
            s(a[i][j]);
        }
    }
    s(q);
    while(q--) {
        s(w);
        org.clear();
        vir.clear();
        for(i=1;i<=n;i++) {
            org.push_back(i);
        }
        for(i=0;i<w;i++) {
            s(x);
            m=INT_MAX;
            for(j=0;j<org.size();j++) {
                //cout<<a[org[j]][x]<<endl;
                if(a[org[j]][x]<m) {m=a[org[j]][x];}
            }
            //cout<<m<<endl;
            for(j=0;j<org.size();j++) {
                if(a[org[j]][x]==m) vir.push_back(org[j]);
            }
            org=vir;
            vir.clear();
        }
        cout<<org[0]<<endl;
    }
    return 0;
}
