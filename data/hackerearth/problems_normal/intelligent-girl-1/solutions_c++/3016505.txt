#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    string s;
    cin>>s;
    int i;
    int n=s.length();
    int a[n];
    memset(a,0,sizeof(a));
    for(i=0;i<n;i++)
    {
        if(int(s[i])%2==0)
        a[i]=1;
        else
        a[i]=0;
    }
    for(i=n-2;i>=0;i--)
    a[i]=a[i]+a[i+1];
    for(i=0;i<n;i++)
    cout<<a[i]<<" ";
    return 0;
}
