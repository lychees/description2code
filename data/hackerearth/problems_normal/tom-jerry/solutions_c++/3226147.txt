#include <bits/stdc++.h>
#define UP 0
#define DOWN 2
#define LEFT 3
#define RIGHT 1
using namespace std;
long dp[101][101][4];
int a[101][101];
bool visit[101][101];
long count_(int x, int y);
int n;
bool valid(int sx, int sy);
int inc[4][2]={{1, 0}, {0, 1}, {-1, 0}, {0, -1}};
int main(){
cin>>n;
memset(dp, -1, sizeof(dp));
memset(visit, false, sizeof(visit));

for(int i=0; i<n; i++){
	for(int j=0; j<n; j++){
		cin>>a[i][j];
	}
}
visit[0][0]=true;
cout<<count_(0, 0);
}

long count_(int x, int y){
if(x==n-1 && y==n-1){
	return 1;
}
long c=0;
int sx, sy;
for(int i=0; i<4; i++){
	sx=x+inc[i][0]; sy=y+inc[i][1];
	if(valid(sx, sy)){
		visit[sx][sy]=true;
			c+=count_(sx, sy);
		visit[sx][sy]=false;
	}
}
return c;
}

bool valid(int sx, int sy){
	if(sx<0 || sx>n-1 || sy<0 || sy>n-1 || a[sx][sy] == 1 || visit[sx][sy]==true){
		return false;
	}
	return true;
}
