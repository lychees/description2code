#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
bool com(int a,int b)
{
    if(a>b)
    {
        return true;
    }
    return false;
}
int main()
{
    vector<int>vec;
    vector<int>::iterator it;
    int n,i,k,mvalue,res;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>k;
        vec.push_back(k);
    }
    make_heap(vec.begin(),vec.end(),com);
    while(!vec.empty())
    {
    mvalue=vec.front();
    it=vec.begin();
    while(it!=vec.end())
    {
        *it=*it-mvalue;
        it++;
    }
    while(!vec.empty()&&vec.front()<=0)
    {
        pop_heap(vec.begin(),vec.end(),com);
        vec.pop_back();
    }
    cout<<n<<endl;
    n=vec.size();
    }
    return 0;
}
