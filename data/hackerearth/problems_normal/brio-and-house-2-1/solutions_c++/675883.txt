inline LD mod(LD x){
    if(x > 0) return x;
    return -x;
}

inline void Solve(){
    LD x1, y1, x2, y2, a, b, c, m, r, d;
    cin>>x1>>y1>>x2>>y2>>r;
    if(x1 == x2){
        a = 1;
        b = 0;
        c = x1;
    }
    else{
        a = (y2 - y1 + 0.0) / (x2 - x1 + 0.0);
        b = -1.0;
        c = y2 - a * x2;
    }

    d = mod(c) / (sqrt(a * a + b * b));
    if(d < r){
        cout<<"REPLANNING"<<endl;
    }
    else if(d == r){
        cout<<"JUST MISSED"<<endl;
    }
    else{
        cout<<"SAFE"<<endl;
    }

    return;
}

int main(){
    int t = 1;
    cin>>t;
    while(t--){
        Solve();
    }
    return 0;
}
