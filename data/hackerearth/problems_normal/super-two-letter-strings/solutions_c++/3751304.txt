#define MOD 1000000007
using namespace std ;
ll dp[11][10010];
void initialise(){
	ll i,j,k,temp;
	for(i=1;i<10002;i++){
		dp[0][i]=1;
	}
	for(i=1;i<=10;i++){
		dp[i][1]=1;
		//dp[i][0]=0;
		temp = 1;
		for(j=2;j<=i+1;j++){
			dp[i][j]=2*dp[i][j-1];
			temp += dp[i][j];
		}
		k = 1;
		for(j=i+2;j<=10001;j++){
			dp[i][j]=temp%MOD;
			temp=(temp%MOD+dp[i][j])%MOD;
			temp=(temp-dp[i][k]%MOD+MOD)%MOD;
			k++;
		}
		
	}
}
int main(){
	int t,p,n,k,i,j;
	si(t);
	initialise();
	while(t--){
		si(n);si(p);
		printf("%lld\n",dp[p-1][n]);
	}
}
