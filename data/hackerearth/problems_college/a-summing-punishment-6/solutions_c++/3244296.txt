#include <iostream>
using namespace std;

int main()
{
    long int n,a[100005],i,q;
    
    cin>>n;
    for(i=0;i<n;i++){
    	cin>>a[i];
    }
    
    cin>>q;
    while(q--){
    	long int l,r,j;
    	cin>>l>>r;
    	long long int sum=0;
    	
    	for(j=l-1;j<r;j++){
    		sum+= a[j];
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
