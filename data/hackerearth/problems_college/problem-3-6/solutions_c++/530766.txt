#include <iostream>
using namespace std;

int main()
{
    int teamnum;
    cin>>teamnum;
    int matches=(teamnum-1);
    int x[teamnum],y[teamnum],home[teamnum],away[teamnum];
    for (int i=0;i<teamnum;i++)
    {
    	cin>>x[i]>>y[i];
    	home[i]=matches;
    	away[i]=0;
    }
    for (int i=0;i<teamnum;i++)
    {
    	for (int j=i+1;j<teamnum;j++)
    	{
    		if(x[i]==y[j])
    	    	home[j]++;
            else
                away[j]++;
            if(y[i]==x[j])
                home[i]++;
            else
                away[i]++;
    	}
    }
    for (int i=0;i<teamnum;i++)
    {
    	cout<<home[i]<<" "<<away[i]<<endl;
    }
    return 0;
}
