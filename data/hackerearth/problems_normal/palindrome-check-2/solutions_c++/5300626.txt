#include<iostream>
#include<string.h>
using namespace std;

int main()
{
	string s;
	cin>>s;
	if(s.length()%2 == 0)
	{
		int a=0,b=s.length()-1,flag=0;
		while(a<b)
		{
			if(s.at(a) != s.at(b))
			{
				flag = 1;
				break;
			}
			a += 1;
			b -= 1;
		}
		if(flag == 0)
		{
			cout<<"YES";
		}
		else
		{
			cout<<"NO";
		}
	}
	else
	{
		int a=0,b=s.length()-1,flag=0;
		while(a!=b)
		{
			if(s.at(a) != s.at(b))
			{
				flag = 1;
				break;
			}
			a += 1;
			b -= 1;
		}
		if(flag == 0)
		{
			cout<<"YES";
		}
		else
		{
			cout<<"NO";
		}
	}
	return 0;
}
