#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
int main ()
{
   int x, y, z, max, stalls, cows, used[200], gap[199], unused;

   cin >> max >> stalls >> cows;

   for (x = 0;x < 199;x++)
       gap[x] = 1;

   for (x = 0;x < cows;x++)
       cin >> used[x];

   sort(used, used + cows);

   for (x = 0;x < cows - 1;x++)
       gap[x] = used[x + 1] - used[x];

   for (x = 0;x < cows;x++)
   {
       for (y = 0;y < cows - 1;y++)
       {
           if (gap[y] < gap[y + 1])
           {
               z = gap[y];
               gap[y] = gap[y + 1];
               gap[y + 1] = z;
           }
       }
   }

   unused = used[0] + stalls - used[cows - 1] - 1;

   for (x = 0;x < max - 1;x++)
       unused = unused + gap[x] - 1;

   cout << stalls - unused << "\n";

   return 0;
}
