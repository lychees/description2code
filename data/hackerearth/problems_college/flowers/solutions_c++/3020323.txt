#include <iostream>
# include<algorithm>
using namespace std;
# include<set>
int main() {
	// your code goes here
	int n,k,ct=0,i,j,l=0;
	cin>>n>>k;
	int a[n];
	for(i=0;i<n;i++)cin>>a[i];
	sort(a,a+n);
	for(i=n-1;i>=0;){
		j=k;
		while(j--){
			ct+=((l+1)*a[i--]);
			if(i<0)break;
		}
		l++;
	}
	cout<<ct<<endl;
	return 0;
}
