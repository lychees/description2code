#include <bits/stdc++.h>
using namespace std;
#define s second
#define f first
#define ll long long int

int main()
{
	ll t,n,i,j,k,a,b;
	cin >> t;
	while(t--){
		cin >> n;
		pair <ll,ll> p[n];
		ll dp[n];
		for(i=0;i<n;i++){
			cin >> a >> b;
			p[i]={a,b};
		}
		sort(p,p+n);
		for(i=0;i<n;i++){
			dp[i]=p[i].s;
		}
		for(i=1;i<n;i++){
			for(j=i-1;j>=0;j--){
				if(p[i].s>p[j].s && p[i].f!=p[j].f){
					dp[i]=max(dp[i],dp[j]+p[i].s);
				}
			}
		}
		cout << *max_element(dp,dp+n) << endl;
	}
    return 0;
}
