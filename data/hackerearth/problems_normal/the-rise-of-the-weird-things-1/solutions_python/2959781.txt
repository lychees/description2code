#!/usr/bin/python

def isZombie(num):
    return num & 1 and True or False

Zombies=[]
Vampires=[]

N = int(raw_input())
if 1 <= N <= 1001:
    t = map(int, raw_input().split())
    for i in range(N):
        Vampires.append(t[i]) if isZombie(t[i]) else Zombies.append(t[i])

Zombies.append(sum(Zombies))
Vampires.append(sum(Vampires))
Zombies.sort()
Vampires.sort()
Weird = Zombies + Vampires
W =map(str, Weird)
print ' '.join(W)
