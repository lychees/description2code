#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   for(int i=1;i<=t;i++)
   {
   	string str1,str2;
   	cin>>str1>>str2;
   	int j=0;
   	int count=0;
   	while(j<str1.length() && j<str2.length())
   	{
   		if(str1[j]==str2[j])
   			count++;
   			j++;
   	}
   	cout<<count<<endl;
   }
    return 0;
}
