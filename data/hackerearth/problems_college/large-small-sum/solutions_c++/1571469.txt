#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	long long int i, arr[1005], n, k, j, sum, t;
	scanf("%lld", &t);
	while(t--)
	{
		scanf("%lld", &n);
		for(k=0; k<n; k++)
		{
			scanf("%lld", &arr[k]);
		}
		sort(arr, arr+n);
		scanf("%lld %lld", &i, &j);
		sum=arr[j-1] + arr[n-i];
		printf("%lld\n", sum);
	}
	return 0;
}
