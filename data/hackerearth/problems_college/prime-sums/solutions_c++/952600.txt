#include<iostream>
#include<cmath>
using namespace std;
int A[4000001];
int main(){
long int i,n,l,k,j;
long long int s=0;
cin>>n;
if(n>2){
for(i=0;i<n;i++)
    A[i]=1;
    k=sqrt(n);
for(i=2;i<=k;i++){
    if(A[i]){
    j=(i*i);
    for(l=j;l<n;l=l+i){
        A[l]=0;
    }
    }
}
for(i=2;i<n;i++){
    if(A[i])
        s=s+i;
}
}
cout<<s<<endl;
return 0;
}
