#include <bits/stdc++.h>
using namespace std;
int main()
{
	int tot=60*60*24;
    int a[60*60*24]={0};
    for(int i=0;i<24;i++)
    {
    	for(int j=0;j<60;j++)
    	{
    		for(int k=0;k<60;k++)
    		{
    			 if(__gcd(i,__gcd(j,k))!=1)
    				a[i*3600+j*60+k]=1;
    		}
    	}
    }
    for(int i=tot-2;i>=0;i--)
    	a[i]+=a[i+1];
    int t;
    cin>>t;
    while(t--)
    {
    	int h,m,s;
    	cin>>h>>m>>s;
    	int idx=h*3600+m*60+s;
    	int g1;
    	if(a[idx]!=0 && tot-a[idx]-idx!=0)
    		g1=__gcd(a[idx],tot-a[idx]-idx);
    	else
    		g1=1;
    	cout<<a[idx]/g1<<":"<<(tot-a[idx]-idx)/g1<<endl;
    }
    return 0;
}
