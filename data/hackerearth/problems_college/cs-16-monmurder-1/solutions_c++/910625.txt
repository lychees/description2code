#include <iostream>
#include <math.h>

using namespace std;

void primeFactors(int p[], int n)
{
    int idx=0;
    int max = sqrt(n);
    //int flag = 0;
    // Print the number of 2s that divide n
    while (n%2 == 0)
    {
        //cout<<"Divisible by 2"<<endl;
        p[idx]++;
        n = n/2;
        //flag = 1;
    }

 
    
    // n must be odd at this point.  So we can skip one element (Note i = i +2)
    for (int i = 3; i <= max; i = i+2)
    {
        idx++;
        //cout<<"idx : "<<idx<<endl;
        //flag = 0;
        // While i divides n, print i and divide n
        while (n%i == 0)
        {
            //cout<<"Divisible by "<<i<<endl;
            p[idx]++;
            n = n/i;
            //flag = 1;
        }
        if(n==1)
            break;
    }
    idx++;
 
    // This condition is to handle the case whien n is a prime number
    // greater than 2
    if (n > 2){
        //cout<<"number is prime and idx="<<idx<<endl;
        p[idx]++;
    }
}

int main(){
    int t;
    cin>>t;
    int p[1000];
    int H;
    while(t--){
        cin>>H;
        int max = sqrt(H);
        for(int i=0; i<= max+1; i++)
            p[i] = 0;

        primeFactors(p, H);
        int nimsum = 0;
        for(int i=0; i<=max+1; i++){
            //cout<<p[i]<<", ";
            nimsum = p[i]^nimsum;
        }
        //cout<<"\n";
        if(nimsum == 0)
            cout<<"Pappu Bhujia"<<endl;
        else
            cout<<"Printf{}"<<endl;
    }
    return 0;
}
