#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
int main()
{
    ios::sync_with_stdio(false);
    
    int t;
    cin>>t;
    assert(t>=1 && t<=10);
    while(t--)
    {
        long long dp[1001][1001]={0};
        int N,M,i,j;
        cin>>N>>M;
        assert(N>=1 && N<=1000 && M>=1 && M<=N);
        dp[0][0]=1;
        for(i=1;i<=N;i++)
        {
            dp[i][0]=1;
            dp[0][i]=0;
        }
        for(i=1;i<=N;i++)
        {
            // for(j=1;i>=j*M;j++)
            for(j=1;j<=N;j++)
            {
                if(i<j*M)
                {
                    continue;
                }
                dp[i][j]=(dp[i-1][j]+dp[i][j-1])%MOD;
            }
        }
     /*   for(i=0;i<=N;i++)
        {
            for(j=0;j<=N;j++)
            {
                cout<<dp[i][j]<<" ";
            }
            cout<<endl;
        }*/
        long long ans=0;
        for(i=0;i<=N && N-i>=i*M;i++)
        {
            if(N-i>i)
                ans+=(dp[N-i][i]);
            ans%=MOD;
        }
        cout<<ans<<endl;
    }
    return 0;
}
