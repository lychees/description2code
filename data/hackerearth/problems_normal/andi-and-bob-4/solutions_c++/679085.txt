#include <cstdio>

typedef long long int ll;
int a[(int)5e6+5];
ll  oa[(int)5e6+5];
ll ea[(int)5e6+5]; //no of even subarrays starting anywhere but ending at index i

int main()
{
	int t,n;
	int i,j;
	scanf("%d",&t);
	ll o,e;
	for(i=0;i<t;i++)
	{
		scanf("%d",&n);
		for(j=0;j<n;j++){
			scanf("%d",&a[j]);
			a[j] = a[j]%2;
		}
		oa[0]=ea[0]=0;
		if(a[0]) //odd
			oa[0]++;
		else
			ea[0]++;
		for(j=1;j<n;j++)
		{
			if(a[j])
			{
				ea[j] = oa[j-1];
				oa[j] = ea[j-1]+1;
			}
			else
			{
				ea[j] = ea[j-1]+1;
				oa[j] = oa[j-1];
			}
		}
		
		e = o = 0;
		for(j=0;j<n;j++)
		{
			e+= ea[j];
			o+=oa[j];
		}
		
		if(e>o)
			printf("ANDI\n");
		else
			printf("BOB\n");
	}
	return 0;
}
