#include <bits/stdc++.h>
using namespace std;
int main(){
	int t,n,i,j,k;
	unsigned long long length,prev;
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		char a[n+1];
		scanf("%s",a);
		length=n;
		prev=1;
		for(i=1;i<n;i++){
			if(a[i]>=a[i-1]){
				length+=prev;
				prev++;
			}else prev=1;	
		}
		cout<<length<<endl;
	}
	return 0;
}
