t = int(raw_input())
answer = []
while t > 0:
    nkp = map(int, raw_input().split(' '))
    kvals = map(int, raw_input().split(' '))

    p = nkp[2]

    if nkp[0] - nkp[1] < nkp[2]:
        p = -1
    else:
        for k in kvals:
            if k <= p:
                p += 1

    answer.append(p)
    t -= 1

for a in answer:
    print a
