#include <bits/stdc++.h>
#define MAX 1002
using namespace std;
int A[MAX];
int main()
{
	int n;
	cin >> n;
	
	assert(n >= 1 && n <= 1000);
	
	for ( int i = 0; i < n; i++ ) {
		cin >> A[i];
		assert(A[i] >= 1 && A[i] <= 100000);
	}
	
	sort(A, A + n);
	int fs = 0, sc = n - 1;
	
	vector <int> v1, v2;
	
	while ( fs <= sc ) {
		v1.push_back(A[fs]);
		v2.push_back(A[sc]);
		if ( fs != sc ) v1.push_back(A[sc]), v2.push_back(A[fs]);
		fs++, sc--;
	}
	
	int ans1 = 0, ans2 = 0;
	for ( int i = 0; i < n - 1; i++ ) ans1 += abs(v1[i] - v1[i + 1]), ans2 += abs(v2[i] - v2[i + 1]);
	
	
	cout << max(ans1,ans2) << endl;
	
	return 0;
}
