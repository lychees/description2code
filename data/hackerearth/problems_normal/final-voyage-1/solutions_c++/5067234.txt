#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

ll n, w, wei[10010], val[10010], dp[10010][10010] ;

ll rec(ll index, ll tot){
	if(tot < 0)
	return -1e9 ;
	
	if(index == n)
	return 0;
	
	ll &st = dp[index][tot] ;
	
	if(st != -1)
	return st ;
	
	ll maxi = rec(index+1, tot) ;
	
	maxi = max(maxi, (val[index] + rec(index+1, tot-wei[index])));
	
	return st = maxi ;
}

int main(){
	ll t;
	cin >> t;
	while(t--){
		
		cin >> n >> w;
		
		for(ll i = 0; i < n; i++){
			cin >> wei[i];
		}
		
		for(ll i = 0; i < n; i++){
			cin >> val[i];
		}
		
		for(ll i = 0; i <= n; i++){
			for(ll j = 0; j <= w; j++){
				dp[i][j] = -1 ;
			}
		}
		
		ll ans = rec(0,w) ;
		cout << ans << endl ;
		
	}
	return 0;
}
