def main():
    s=raw_input();
    l=len(s);
    t= input();
    for i in range (t):
        x,y=(raw_input().split());
        x=int(x)%l-1;
        y=int(y)%l-1;
        if(s[x]==s[y]):
            print "Yes"
        else:
            print "No"
main();
