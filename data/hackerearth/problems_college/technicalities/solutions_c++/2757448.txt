#include <iostream>
#include <string.h>
#define mod 1000000007
using namespace std;

int main()
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		if(n < 18)
		{
			long long int num;
			cin>>num;
			cout<<num%mod<<endl;
		}
		else
		{
			string s;
			int a[n];
			cin>>s;
			for(int i = 0; i < n; i++)
			{
				a[i] = s[i]-'0';
			}
			long long int num = a[0];
			int i = 1;
			while(i < n)
			{
				num = num*10 + a[i];
				if(num > mod)
					num = num%mod;
				i++;
			}
			cout<<num<<endl;
		}
		
	}
    
}
