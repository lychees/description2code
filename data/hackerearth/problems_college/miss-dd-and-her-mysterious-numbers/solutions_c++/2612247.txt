#include <iostream>
#include <cmath>
#include <stdio.h>
#define N 10000000
using namespace std;

int num_div[N];
int cnt[N];

int findnum(int n){
    for (int i=4;i<=n;i++){
        if (num_div[i]!=1)
        for (int j=2;j<=i;j++){
            if (i%j==0){
                if ((i/j)%j==0) num_div[i]=num_div[i/j];
                else num_div[i]=1+num_div[i/j];
                break;
            }
        }
    }
    return 0;
}

int sieve(int n){
    for( int i = 2; i <= n; i++ )
    {
        if( num_div[i] == 0 )
        {
            num_div[i] = 1;
            for(int j = i+i; j <= n; j+=i )
            {
                num_div[j] = -1;
            }
        }
    }
    return 0;
}

int main(){
    int n;
    scanf("%d",&n);
    sieve(n);
    findnum(n);

    for (int i=0;i<=n;i++){
        cnt[num_div[i]]++;
    }
    int q,query;
    scanf("%d",&q);
    while (q--){
        scanf("%d",&query);
        if (query>=1000000) cout << "0" << endl;
        else{
            cout << cnt[query] << endl;
        }
    }
    return 0;
}
