#include<iostream>
#include<algorithm>

using namespace std;

int a[105];

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n, k;
		cin >> n >> k;
		for(int i = 0; i < n; i++)
			cin >> a[i];
		int ans = 1e9;
		sort(a, a + n);
		for(int i = 1; i <= a[n - 1]; i++)
		{
			int l, r = 0;
			while(a[r] < i)
				r++;
			l = r - 1;
			int sum = 0;
			for(int j = 0; j < k; j++)
				if(l >= 0 && (r == n || (i - a[l]) * 3 < (a[r] - i) * 5))
					sum += (i - a[l--]) * 3;
				else
					sum += (a[r++] - i) * 5;
			ans = min(ans, sum);
		}
		cout << ans << endl;
	}
    return 0;
}
