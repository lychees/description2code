#include<bits/stdc++.h>
using namespace std;
#define max 100005
int main()
{
    int cnt,temp,i,j,k,q,n,a[max],primes[max],temp1,temp2,zeroes,freq[max];
    //construction of multidimensional vector for queries
    vector<vector<int> > c(max);
    memset(primes,0,sizeof(primes));
    memset(freq,0,sizeof(freq));
    // sieve //
    bool sieve[max];
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
        freq[a[i]]++;
    }

    for(i=2;i<max;i++)
    {
        if(sieve[i]==false)
        {
            c[i].push_back(i);
            primes[i]+=freq[i];
            for(j=2;i*j<max;j++)
            {
                sieve[i*j]=true;
                c[i*j].push_back(i);
                temp=i*j;
                cnt=0;
                while(temp%i==0)
                {
                    ++cnt;
                    temp/=i;
                }
                primes[i]+=cnt*freq[i*j];
            }
        }

    }
    //query input
    scanf("%d",&q);
    while(q--)
    {
        zeroes=10000000;
        scanf("%d",&k);
        for(i=0;i<c[k].size();i++)
        {
            temp2=0;
            temp=k;
            while(temp%c[k][i]==0)
            {
                temp=temp/c[k][i];
                temp2++;
            }
            zeroes=min(zeroes,(int)(primes[c[k][i]]/temp2));
        }
        printf("%d\n",zeroes);
    }


   return 0;
}
