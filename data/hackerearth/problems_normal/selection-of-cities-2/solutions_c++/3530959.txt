#include<bits/stdc++.h>
using namespace std ;
#define mod 1000000007
typedef long long lint;

lint power(lint a, lint n)  //a^n
{
    lint ans=1;
    while(n)
    {
        if(n & 1)
            ans=(ans*a)%mod;
        a=(a*a)%mod;
        n>>=1;
    }
    return ans;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		lint n;
		cin>>n;
		cout<<power(2,n)-1<<endl;
		
	}
}
