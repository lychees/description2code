#include <iostream>
using namespace std;
long long a[1000010]={0};
long long b[1000010]={0};
int main()
{
    long long n;
   	long long max=0,min=1000012,i,num,n1;
   	cin>>n;
   	for(i=0;i<n;i++)
   	{
   		cin>>num;
   		a[num]++;
   		if(num>max)
   			max=num;
   		if(num<min)
   			min=num;
   	}
   	cin>>n1;
   	for(i=0;i<n1;i++)
   	{
   		cin>>num;
   		b[num]++;
   	}
   	for(i=min;i<=max;i++)
   	{
   		if(a[i])
   		{
   			if(b[i]==0)
   				cout<<i<<" ";
   			else if(b[i]!=a[i])
   				cout<<i<<" ";
   		}
   	}
   
    return 0;
}
