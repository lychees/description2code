#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	ll t,n,i,j,ans,a,b;
	cin >> t;
	while(t--){
		vector<pair<ll,ll> > v;
		cin >> n;
		for(i=0;i<n;i++){
			cin >> a >>b;
			v.push_back(make_pair(a,b));
		}
		sort(v.begin(),v.end());
		ll dp[n];
		dp[0]=v[0].second;
		ans=dp[0];
		for(i=1;i<n;i++){
			dp[i]=v[i].second;
			for(j=0;j<i;j++){
				if( v[i].first>v[j].first && v[i].second>v[j].second )
				{
					dp[i]=max( dp[i],dp[j]+v[i].second );	
				}
			}
			ans = max(ans,dp[i]);
		}
		cout << ans << endl;
		
	}
    return 0;
}
