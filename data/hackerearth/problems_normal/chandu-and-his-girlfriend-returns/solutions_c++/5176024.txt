#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main(){
	vector <int> vec,vec2;
	int t=0;
	
	cin >> t;
	
	for(int x=0; x<t;x++){
		int n1=0, n2=0;
		cin >> n1 >> n2;
		for(int y = 0; y < n1; y++){
			int temp=0;
			cin >> temp;
			vec.push_back(temp);
		}
		for(int y = 0; y < n2; y++){
			int temp=0;
			cin >> temp;
			vec2.push_back(temp);
		}
		vec.insert(vec.end(), vec2.begin(), vec2.end());
		sort(vec.begin(), vec.end());
		
		for(int y = vec.size()-1; y >= 0; y--)
			cout<< vec.at(y)<< " ";
		cout<<endl;
		vec.clear();
		vec2.clear();
	}
	
	return 0;
}
