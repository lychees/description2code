#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int m,n,i;
	scanf("%d%d",&m,&n);
	int a[m],b[n];
	for(i=0;i<m;i++)
	scanf("%d",&a[i]);
	for(i=0;i<n;i++)
	scanf("%d",&b[i]);
	sort(a,a+m);
	sort(b,b+n);
	int count;
	count=a[m-1];
	while(1)
	{
		if(2*a[0]<=count&&count<b[0])
		break;
		else
		count++;
	}
	printf("%d\n",count);
}
