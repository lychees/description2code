#include<bits/stdc++.h>
using namespace std;
#define MAX 1002
int seen[104][104];
char table[100][100];
int tb_count[104][104];
int tb[104][104];
char _itoc(int i)
{
	if(i==0) return '0';
	else if(i==1) return '1';else if(i==2) return '2';else if(i==3) return '3';
	else if(i==8) return '8';else if(i==7) return '7';else if(i==6) return '6';
	else if(i==4) return '4';else if(i==5) return '5';else if(i==9) return '9';
}
	
void solve(int i,int j,int N,int M)
{
	if(seen[i][j]==1 || i<0 || j<0 || j>M-1 || i>N-1)
	{
		return ;
	}
	if(tb_count[i][j]!=0)
	{
		table[i][j]=_itoc(tb_count[i][j]);
		seen[i][j]=1;
		return ;
	}
	else
	{
		table[i][j]=_itoc(tb_count[i][j]);
		seen[i][j]=1;
		solve(i-1,j-1,N,M);solve(i,j-1,N,M);solve(i+1,j-1,N,M);
		solve(i-1,j,N,M);solve(i+1,j,N,M);
		solve(i-1,j+1,N,M);solve(i,j+1,N,M);solve(i+1,j+1,N,M);
		return ;
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--){
	int N,M;
	cin>>N>>M;
	for(int i=0;i<104;i++)
	{
		for(int j=0;j<104;j++)
		{
			tb[i][j]=0;
			tb_count[i][j]=0;
			seen[i][j]=0;
		}
	}
    table[100][101]={'\0'};
    int I,J;
	char x='\0';
	for(int i=0;i<N;i++)
	{
		for(int j=0;j<M;j++)
		{
			cin>>x;
			table[i][j]=x;
		}
	}
	for(int i=0;i<=N+1;i++)
	{
		for(int j=0;j<=M+1;j++)
		{
		    if(i==0 || j==0 ||i==N+1 || j==M+1)
			tb[i][j]=0;
			else
			{
				if(table[i-1][j-1]=='*')
				tb[i][j]=-1;
				else if(table[i-1][j-1]=='c')
				{
					tb[i][j]=0;
					I=i-1;
					J=j-1;
				}
				else
				tb[i][j]=0;
			}
		}
	}
	for(int i=0;i<N;i++)
	{
		for(int j=0;j<M;j++)
		{
			if(tb[i+1][j+1]==-1)
			tb_count[i][j]=-1;
			else
			{
				tb_count[i][j]=-(tb[i+2][j+2]+tb[i+1][j+2]+tb[i][j+2]+tb[i+2][j+1]+tb[i][j+1]+tb[i+2][j]+tb[i+1][j]+tb[i][j]);
			}
		}
	}
	solve(I,J,N,M);
	for(int i=0;i<N;i++)
	{
		for(int j=0;j<M;j++)
		{
			cout<<table[i][j];
		}
		cout<<endl;
	}
    }
}
