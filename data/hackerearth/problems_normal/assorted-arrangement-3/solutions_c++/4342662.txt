#include <iostream>
#include <algorithm>
#define MAXM 55

using namespace std;

int n,m;
int c[MAXM];
int ans;

int main() {
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        cin >> c[i];
    }

    while (n--) {
        int k;
        cin >> k;
        ans = (ans / c[k] + 1) * c[k];
        while (1) {
            bool ok = 1;
            for (int i = k + 1; i <= m && ok; i++) {
                if (ans % c[i] == 0) ok = 0;
            }
            if (ok) break;
            ans += c[k];
        }
    }

    cout << ans << '\n';
}
