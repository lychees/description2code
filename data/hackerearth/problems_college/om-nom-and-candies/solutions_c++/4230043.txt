#include<bits/stdc++.h>
using namespace std;

int main(){
	int n;
	cin>>n;
	int a[n+1];
	for(int i=1;i<=n;i++)
	 cin>>a[i];
	int sum[n+1];
	for(int i=1;i<=n;i++){
		sum[i]=a[i];
	}
	for(int i=1;i<=n;i++){
		for(int j=1;j<=sqrt(i);j++){
			if(i%j==0 && j*j!=i && i!=j && j!=1){
				int x=i/j;
				
			 sum[i]=max(sum[i],a[i]+sum[j]);
			 sum[i]=max(sum[i],a[i]+sum[x]);
				}
			else{
				if(i%j==0 && i!=j){
					 sum[i]=max(sum[i],a[i]+sum[j]);
				}
			}	
		}
	}
	for(int i=1;i<=n;i++)
	 cout<<sum[i]<<" "; 
}
