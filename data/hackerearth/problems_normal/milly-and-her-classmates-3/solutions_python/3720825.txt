def is_sorted(array, n):
    for index in xrange(0, n-1):
        if array[index] > array[index+1]:
            return False
    return True


def decresing_range(array, n):
    start, end = None, None
    for index in xrange(0, n-1):
        if start is None and array[index] > array[index+1]:
            start = index
        elif start is not None and end is None and array[index] < array[index+1]:
            end = index
            break
    if start is not None and end is None:
        end = n - 1
    return start, end

if __name__ == "__main__":
    t = input()
    for _ in xrange(t):
        n = input()
        array = map(int, raw_input().strip().split())
        start, end = decresing_range(array, n)
        if start is None:
            print -1, -1
        else:
            sorted_list = array[:start] + list(reversed(array[start:end+1])) + array[end+1:]
            if is_sorted(sorted_list, n):
                print start + 1, end + 1
            else:
                print -1, -1
