#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--) {
		int x,y,i;
		cin>>x>>y;
		int z = x-y;
		if(z < 0) {
			puts("0");
			continue;
		}
		if(x == y) {
			puts("-1");
			continue;
		}
		int c = 0;
		for(i = 1; i <= sqrt(z); i++) {
			if(z%i == 0) {
				if(x%i == y) c++;
				int l = z/i;
				//cout<<i<<" "<<l<<endl;
				if(l != i && x%l == y) c++;
			}
		}
		cout<<c<<endl;
	}
    return 0;
}
