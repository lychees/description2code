#include<bits/stdc++.h>
using namespace std;
#define ll long long int 
#define mod 1000000007
ll po(ll a,ll b)
{
	if(b==0)
		return 1;
	ll ans=po(a,b/2);
	ans=ans*ans;
	if(ans>=mod)
		ans%=mod;
	if(b&1){
		ans*=a;
		if(ans>=mod)
			ans%=mod;
	}
	return ans;
}
int main()
{
 	int t;
 	scanf("%d",&t);
 	while(t--)
 	{
		ll n,i,x;
		set<ll> se;
		cin>>n;
		for(i=0;i<n;i++)
		{
			scanf("%lld",&x);
			se.insert(x);
		}
		ll ans=0;
		ans=po(n,se.size());
		printf("%lld\n",ans);
 	}
	return 0;
}
