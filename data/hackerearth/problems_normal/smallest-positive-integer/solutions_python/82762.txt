import sys

def get_min_length_single_digit_factors(x):
	original_x = x
	factor_counts = [0]*10
	for i in xrange(8):
		if x < 2:
			break
		factor = 9-i
		while x % factor == 0:
			x = x/factor
			factor_counts[factor] += 1
	if not x==1:
		return get_min_length_single_digit_factors(original_x+1000000007)

	return factor_counts

def get_smallest_positive_int(x):
	if x==0:
		return str(10);
	factor_counts = get_min_length_single_digit_factors(x)
	ret =""
	for digit,count in enumerate(factor_counts):
		if count > 0:
			for j in xrange(count):
				ret += str(digit)
	return ret

def main():
	X = input()
	sys.stdout.write(get_smallest_positive_int(X))
	
if __name__=="__main__":
	main()
