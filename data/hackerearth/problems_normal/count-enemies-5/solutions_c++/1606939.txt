

    #include<iostream>
    using namespace std;
    int main(){
     
    long int t,j,c,d,sum;
    cin>>t;
    char s[100001];
    while(t--)
    {
    c=0;
    d=0;
    sum=0;
    cin>>s;
    for(j=0;s[j];j++)
    {
    if(s[j]=='O')
    c++;
    if(s[j]=='X')
    d++;
    if(s[j]=='*')
    {
    if(d==0)
    sum+=c;
    d=0;
    c=0;
    }
    }
    if(d==0)
    sum+=c;
    cout<<sum<<endl;
    }
    return 0;
    }
