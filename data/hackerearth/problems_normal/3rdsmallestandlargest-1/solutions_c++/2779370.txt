#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t,i;
    cin>>t;
    while(t--)
    {
        string s;
        cin>>s;
        if(s.length()<3)
        {
            cout<<"Not possible!"<<endl;
            continue;
        }
        sort(s.begin(),s.end());
        for(i=0;i<2;i++)
        {
            next_permutation(s.begin(),s.end());
        }
        cout<<s<<" ";

        sort(s.begin(),s.end());
        reverse(s.begin(),s.end());
        for(i=0;i<2;i++)
        {
            prev_permutation(s.begin(),s.end());
        }
        cout<<s<<endl;


    }
    return 0;
}
