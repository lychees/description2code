#include <cstdio>

int main()
{
    int t;
    long long int n, i, sum ;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%lld",&n);
    	sum=0;
    	for(i=1;i*i<=n;i++)
    	{
    		if(n%i==0)
    		{
    			if(i%4!=0)
    			sum+=i;
    			if((n/i)%4!=0&&(n/i)!=i)
    			sum+=n/i;
    		}
    	}
    	
    	printf("%lld\n", 8*sum);
   
    }
}
