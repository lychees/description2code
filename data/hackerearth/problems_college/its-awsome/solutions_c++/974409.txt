#include <iostream>

#define LLU long long unsigned int

using namespace std;

int main()
{
	int T,d,count;
	cin>>T;
	while(T--)
	{
		count=0;
	
	LLU N;
	cin>>N;
	while(N)
	{
		d=N%10;
		if(d==4 || d==7)
		{
			count++;
		}
		N=N/10;
	}
	if(count==4 || count==7)
	{
		cout<<"YES\n";
	}
	else
	{
		cout<<"NO\n";
	}
	}
    //cout << "Hello World!" << endl;
    return 0;
}
