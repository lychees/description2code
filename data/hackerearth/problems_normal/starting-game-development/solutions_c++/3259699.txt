#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n,m,q,i,j,x;
    vector<int> v[501];
    vector <int> a;
    cin>>n>>m>>q;
    for(i=0;i<n;i++)
    {
    	for(j=0;j<m;j++)
    	{
    		cin>>x;v[i].push_back(x);
    	}
    }
    while(q--)
    {
    	a.clear();
    for(i=0;i<n;i++)
    {
    cin>>x;a.push_back(x);	
    }
    int mini=INT_MAX;
    for(i=0;i<n;i++)
    {
    	x=upper_bound(v[i].begin(),v[i].end(),a[i])-v[i].begin();
    	mini=min(mini,x);
    }
    cout<<mini<<endl;
    }
    
}
