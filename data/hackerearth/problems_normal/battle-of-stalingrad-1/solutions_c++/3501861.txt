#include<bits/stdc++.h>
using namespace std;



int main()
{
	int t;
	cin>>t;
	while(t--){
		int a[3];
		int b[3];
		int c[3];
		int point[3];
		cin>>a[1]>>a[2]>>b[1]>>b[2]>>c[1]>>c[2]>>point[1]>>point[2];
		
		
		pair<int,int>AB;
		pair<int,int>BC;
		pair<int,int>CA;
		AB.first = b[1] - a[1];
		AB.second = b[2] - a[2];
		BC.first = c[1] - b[1];
		BC.second = c[2] - b[2];
		CA.first = a[1] - c[1];
		CA.second = a[2] - c[2];
		
		int f1,f2,f3;
		
		f1 = AB.first * ( point[2] - a[2]   ) - AB.second*( point[1] - a[1]  );
		
		f2 = BC.first * ( point[2] - b[2]   ) - BC.second*( point[1] - b[1]  );
		
		f3 = CA.first * ( point[2] - c[2]   ) - CA.second*( point[1] - c[1]  );
		///cout<<f1<<" "<<f2<<" "<<f3<<endl;
		if(( f1<=0 && f2<=0 && f3<=0  )  || ( f1>=0 && f2>=0 && f3>=0  ))
			cout<<"INSIDE"<<endl;
		else
			cout<<"OUTSIDE"<<endl;
	}
	return 0;
}
