#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,r;
    cin>>n>>r;
    vector<int> a;
    for(int i=1;i<=n;i++)
    	a.push_back(i);
    int prev=0;
    while(a.size()>1)
    {
    	int temp=((r%a.size())+prev)%a.size();
    	if(temp==0)
    	{
    		temp=a.size();
    		prev=0;
    	}
    	prev=temp-1;
    	a.erase(a.begin()+temp-1);
    }
    cout<<a[0];
    return 0;
}
