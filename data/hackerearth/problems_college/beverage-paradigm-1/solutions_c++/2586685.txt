#include <iostream>
using namespace std;

int main()
{
	int n,a[10000],x,c=0,i,j,k;
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	cin>>x;
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			for(k=j+1;k<n;k++)
			{
				if(a[i]+a[j]+a[k]==x)	c=1;
			}
		}
	}
	if(c==1)	cout<<"True";
	else	cout<<"False";
    return 0;
}
