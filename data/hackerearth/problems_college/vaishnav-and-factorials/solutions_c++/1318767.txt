#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t!=0)
	 {
	 	t--;
	 	int n;
	    cin>>n;
	    int r=0;
	    int a[80000];
	    //int p=100;
	    a[79999]=1;
	    int k=n;
	    int temp=0;
	    int count=1;
	    for(int i=0;i<79999;i++)
	    a[i]=0;
	    while(k>1)
	    {
	        int i;
	        for(i=79999;i>=80000-count;i--)
	        {
	            a[i]=a[i]*k;
	            a[i]=a[i]+temp;
	            temp=0;
	            if(a[i]>=10)
	            {
	                temp=a[i]/10;
	                a[i]=a[i]%10;
	
	            }
	        }
	        while(temp!=0)
	        {
	            a[i]=temp%10;
	            temp=temp/10;
	            count++;
	            i--;
	        }
	        k--;
	    }
	    for(int i=80000-count;i<80000;i++)
	    {
	    	if(a[i]==4 || a[i]==7)
	    	r++;
	    }
	    cout<<r<<endl;
	 }
    return 0;
}
