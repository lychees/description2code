#include <iostream>
#include <map>
#include <stdio.h>
#include <string.h>
using namespace std;

char s[1000005];
int g[1000005],v[1000005];
int main()
{
    int T,n,m,i,j;
    scanf("%d",&T);
    while(T--)
    {
    	scanf("%d%d%s",&n,&m,s);
    	int a=0,b=0,len=strlen(s);long long ans=0,res=0;
    	for(i=0;i<=len;i++)g[i]=v[i]=0;
    	g[0]=v[0]=1;
    	for(i=0;s[i];i++)
    	{
    		j=s[i]=='R';
    		a+=j;b+=!j;
    		if(a>=n)ans+=g[a-n];
    		if(b>=m)res+=v[b-m];
    		g[a]++,v[b]++;
    	}
    	cout<<ans<<" "<<res<<"\n";
    }
    return 0;
}
