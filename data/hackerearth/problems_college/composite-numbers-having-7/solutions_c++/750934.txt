//Composite with number 7
#include<iostream>
#include<math.h>
#include<stdio.h>
#include<stdlib.h>
using namespace std;

int main()
{
	int test, start, end, flag, flag_1, cnt, i, j, index, temp;
	int *arr;
	arr = new int[1000005];
	for(i=0;i < 1000005;i++)
		arr[i] = 0;
	for(i=2;i <= 1000;i++)
	{
		for(j=2, flag=2;j <= sqrt(i);j++)
		{
			if(i % j == 0)
			{
				flag = 0;
				break;
			}
		}
		if(flag != 0)
		{
			index = 2;
			while(index * i <= 1000000)
			{
				flag_1 = 0;
				temp = index * i;
				while(temp)
				{
					if(temp % 10 == 7)
					{
						flag_1 = 1;
						break;
					}
					else
						temp /= 10;
				}
				if(flag_1 == 1)
					arr[(index*i)] = 1;
				index++;
			}
		}
	}
	scanf("%d", &test);
	while(test--)
	{
		scanf("%d %d", &start, &end);
		for(i=start, cnt=0;i <= end;i++)
			if(arr[i] == 1)
				cnt++;
		if(cnt != 0)
			printf("%d\n", cnt);
		else
			printf("-1\n");
	}
	return 0;
}
