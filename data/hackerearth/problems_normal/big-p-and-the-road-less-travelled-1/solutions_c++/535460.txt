#include <bits/stdc++.h>
#define ll long long
using namespace std;

vector<int>g[10010];
ll dp[10010];
bool vis[10010];

int n;
ll dfs(int u){
    if(u==n){
        return 1;
    }
    if(vis[u])
    	return dp[u];
   vis[u] = true;
   ll num = 0;
   for(int i =0;i<g[u].size();i++){
       num+=dfs(g[u][i]);
   }
   	dp[u]= num;
     return num;
    
}

int main()
{
    cin >> n;
    memset(dp,0,10010);
    memset(vis,false,10010);
    int a=-1,b=-1;
    while ( a !=0 && b!=0){
        cin >> a >> b;
        if (a > 0 && b > 0)
            g[a].push_back(b);
    }
   dfs(1);
   cout << dp[1];
      
}
