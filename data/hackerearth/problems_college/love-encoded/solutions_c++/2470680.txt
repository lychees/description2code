#include<bits/stdc++.h>
#define ll long long
#define pb push_back
#define mp make_pair
using namespace std;
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		int n;
		cin>>n;
		string s;
		cin>>s;
		char a[(int)s.length()];
		int i;
		for(i=0;i<s.length();++i)
		{
			if(s[i]-'A'<n)
			{
				a[i]=s[i]-n+26;
			}
			else a[i]=s[i]-n;
		}
		for(i=0;i<s.length();++i)
		cout<<a[i];
		cout<<"\n";
	}
	return 0;
}
