#include <iostream>
#include <vector>
using namespace std;
int q=1;
bool visited[1001];
vector <int> a[1001];
void init()
{
	for(int i =0;i<1001;i++)
	{
		visited[i] = false;
		
	}
}
void dfs(int s)
{
	visited[s] = true;
	for(int i =0;i<a[s].size();i++)
	{
		if(visited[a[s][i]]==false)
		{
			dfs(a[s][i]);
			q++;
			
		}
	}
}

int main()
{
    int n,e;
    init();
    cin>>n>>e;
    while(e--)
    {
    	int x,y;
    	cin>>x>>y;
    	a[x].push_back(y);
    	
    }
    
    int min=1001,temp;
    for(int i=1;i<=n;i++)
    {
    	dfs(i);
    	init();
    	temp = q;
    	q=1;
    	if(min>temp)
    	min =temp;
    }
    cout<<min<<endl;
    
    return 0;
}
