#include <bits/stdc++.h>
using namespace std;
long long arr[10];
int main() 
{
    long long n,S=0,ans=0,i,x,y,k,j;
    cin>>n;
    for(i=0;i<n;i++)
        {
        	scanf("%d",arr+i);
        	S+=arr[i];
        }
    sort(arr,arr+n);
    do
    {
    for(i=0;i<(1<<n);i++)
    {
    	x=y=0;
    	k=i;
    	for(j=1;j<=n&&x<=y;j++)
    	{
    		if(k%2==1)
    		    x+=arr[j-1];
    		else
    		    y+=arr[j-1];
    		k/=2;
    	}
    	if(x<=y)
    	ans=ans+1;
    }
    }while(next_permutation(arr,arr+n));
    cout<<ans<<" "<<S<<"\n";
    if(S>=ans)
        cout<<"Got no way out!\n";
    else 
        cout<<"We will win!\n";
}
