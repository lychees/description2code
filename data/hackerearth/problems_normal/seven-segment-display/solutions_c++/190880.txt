#include <stdio.h>
#include <cstring>
#include <algorithm>
using namespace std;

int t ;
char a[100001] ;
int b[]={6,2,5,5,4,5,6,3,7,6} ;

int main() 
{
  scanf("%d",&t) ;
  while(t--)
  { 
     scanf("%s",&a) ; int ans=0 ; 
     for(int i=0 ; a[i] ; i++) ans+=b[a[i]-'0'] ;
     printf("%d\n",ans) ;   
  }

 return 0;
}
