#include<bits/stdc++.h>
using namespace std;
#define MAX 10000006
#define MOD 1000000007
int primes[MAX];
long long int sum[MAX];
int lp[MAX];
int pr;
void siever(void)
{
		int i,j;
		pr = 0;
		for(i=2;i<MAX;++i)
		{
				if(lp[i] == 0)
				{
						primes[pr++] = i;
						for(j=i;j<MAX;j+=i)
						{
								lp[j]=1;
						}
				}
		}
}
inline int give(int a, int b)
{
		int ia,ib;
		ia = lower_bound(primes,primes+pr,a) - primes;
		ib = upper_bound(primes,primes+pr,b) - primes;
		return ib-ia;
}
int main()
{
		siever();
		int t;
		int n;
		int i;
		//scanf("%d",&t);
		t=1;
		while(t--)
		{
				scanf("%d",&n);
				long long int sum = 0;
				int tmp;
				for(i=0;i<n;++i)
				{
						tmp = give(i,i+n-1);
						//cout<<"tmp="<<tmp<<endl;
						sum = (sum+tmp)%MOD;
				}
				if(n > 1){
						printf("%lld\n",sum);
				}
				else puts("0");

		}
		return 0;
}
