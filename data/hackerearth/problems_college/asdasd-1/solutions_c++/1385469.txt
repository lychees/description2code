#include<iostream>
#include<cmath>
using namespace std;
long gcd(long a,long b);
long gcd(long a,long b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long a,b;
		cin>>a>>b;
		a=(1<<a)-1;
		b=(1<<b)-1;
		a=gcd(a,b);
		b=(int)sqrt(a);
		int count=0;
		for(int i=2;i<=b;i++)
		{
			if(a%i==0)
			count++;
			while(a%i==0)
			a/=i;
			if(a==1)
			break;
		}
		if(a>1)
		count++;
		cout<<count<<endl;
	}
	return 0;
}
