import sys
flag = True


class Node:
    def __init__(self, c):
        self.c = c
        self.next = {}
        self.cnt = 0
        self.ends = 0


def insert(node, word):
    for w in word:
        if w not in node.next:
            node.next[w] = Node(w)
        node = node.next[w]
        node.cnt += 1
    node.ends += 1


def traverse(node, path):
    global flag
    if len(path) > 0:
        if len(node.next) > 0 and node.ends != 0:
            flag = False
    # if len(path)>0:
    #     print path, len(node.next), node.cnt, node.ends
    for w in node.next:
        traverse(node.next[w], path+w)


tc = int(raw_input())
while tc > 0:
    root = Node(u"$")
    n = input()
    tc = tc - 1
    flag = True
    while n > 0:
        n = n - 1
        s = raw_input()
        insert(root, s)
    traverse(root, "")
    if flag:
        print "YES"
    else:
        print "NO"
