#include <iostream>
using namespace std;

int main()
{
    long T,i,B,W,x,y,z,a,b;
    long res;
    cin>>T;
    i=0;
    while(i++<T){
    	cin>>B;
    	cin>>W;
    	cin>>x;
    	cin>>y;
    	cin>>z;
    	a=x<(y+z)?x:(y+z);
    	b=y<(x+z)?y:(x+z);
    	res=B*a+W*b;
    	cout<<res<<endl;
    }
    return 0;
}
