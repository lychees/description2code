#include<iostream>
using namespace std;
main()
{
    int i,j,n;
    cin>>n;
    int A[n],B[n],b=0;
    //get input from user
    for(i=0;i<n;i++)
    {
        cin>>A[i];
    }
    for(i=0;i<n;i++)
    {
        int prime=0;
        for(j=0;j<n;j++)
        {
            if(A[i]%A[j]==0)
                prime++;
        }
        if(prime==1)
        {
            B[b]=A[i];
            b++;
        }
    }
    for(i=0;i<b;i++)
    {
        cout<<B[i]<<" ";
    }
}
