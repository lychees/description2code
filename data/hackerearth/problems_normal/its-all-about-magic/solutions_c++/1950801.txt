unsigned long long fib(int n)
{
	if(n==1)
	return 0;
	else if(n==2)
	return 1;
	else{
	unsigned long long f =0, s=1,nT;  
	for (int i= 3; i<=n;i++){
		nT = f+s;
		f=s;
		s=nT;
	}
	return nT;
	}
}
