#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t>0){
    	int n;
    	cin >> n;
    	string s[n];
    	for(int i=0;i<n;i++){
    		cin >> s[i];
    	}
    	int hash[26]={0};
    	for(int i=0;i<n;i++){
    		int len = s[i].length();
    		int presenthash[26]={0};
    		for(int j=0;j<len;j++){
    			if(presenthash[s[i][j]-'a']==0){
    			hash[s[i][j]-'a']++;
    			presenthash[s[i][j]-'a']=1;
    			}
    		}
    	}
    	int index=0;
    	int maxvalue=hash[0];
    	for(int i=1;i<26;i++){
    		//cout << hash[i] << endl;
    		if(hash[i]>maxvalue){
    			index = i;
    			maxvalue = hash[i];
    		}
    	}
    	cout << char(index+'a') << endl;
    	t--;
    }
}
