#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int n,l,r,i,q,x,k=0;
    cin>>n>>l>>r>>q;
    int arr[n],j,brr[n];
    for(i=0;i<n;i++)
    {
    	cin>>arr[i];
    }
    for(i=l;i<=r;i++)
    {
    	brr[k++]=arr[i];
    }
    sort(brr,brr+k);
    //for(i=0;i<k;i++) cout<<brr[i]<<" ";
    for(i=0;i<q;i++)
    {
    	cin>>x;
    	int ans=-1;
    	for(j=0;j<k;j++)
    	{
    		if(brr[j]>x)
    		{
    			ans=brr[j];
    			break;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
