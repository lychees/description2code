'''input
RRULDL
'''

s = raw_input().rstrip()
start = "0 0"
data = set()
data.add(start)
for i in s:
    start = start.split()
    if i == 'L':
        start[1] = str(int(start[1]) - 1)
    elif i == 'R':
        start[1] = str(int(start[1]) + 1)
    elif i == 'U':
        start[0] = str(int(start[0]) - 1)
    else:
        start[0] = str(int(start[0]) + 1)
    start = " ".join(start)
    data.add(start)
print len(s) - len(data) + 1
