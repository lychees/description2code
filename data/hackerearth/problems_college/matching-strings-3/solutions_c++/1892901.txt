#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		
		string a,b;
		cin>>a;
		cin>>b;
		long long int count=0;
		
		int size=max(a.length(),b.length());
		int exit=min(a.length(),b.length());
		for(int i=0;i<size;i++){
			if(i>=exit){break;}
			if(a[i]==b[i]){
				count++;
				}
				
		}
		
		cout<<count<<endl;
	}
	}
