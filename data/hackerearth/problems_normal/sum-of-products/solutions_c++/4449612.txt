#include <iostream>
#include <math.h>
using namespace std;
typedef long long int ll;
ll sum(ll n, ll mod) {
	ll ans = n * (n + 1);
	ans = ans>>1;
	return ans%mod;
}
int main()
{
    ll t;
    scanf("%lld", &t);
    ll mod = 1000000007;
    while(t--) {
    	ll n;
    	scanf("%lld", &n);
    	ll p = (ll)(sqrt(n));
    	ll i;
    	ll ans = 0;
    	for(i = 1; i <= p; i++) {
    		ans = ans + (n/i)*i;
    		ans%=mod;
    	}
    	for(i = n/p; i > 1; i--) {
    		ll x = n/i;
    		ll y = n/(i - 1);
    		ll val = (sum(y, mod) - sum(x, mod) + mod)%mod;
    		val = val * (i - 1);
    		val%=mod;
    		ans = (ans + val)%mod;
    	}
    	printf("%lld\n", ans);
    }
    return 0;
}
