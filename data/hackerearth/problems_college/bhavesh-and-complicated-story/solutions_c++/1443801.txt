#include <bits/stdc++.h>
using namespace std;
#define MAX 1000005
#define ll long long
int bittree[MAX];

#define gc getchar

ll read(int idx)
{
    ll sum = 0;
    while(idx > 0) {
        sum += (ll)bittree[idx];
        idx -= (idx & -idx);
    }
    return sum;
}

void update(int idx, int val)
{
    while(idx <= MAX - 2) {
        bittree[idx] += val;
        idx += (idx & -idx);
    }
}

void solve()
{
    int n;
    scanf("%d", &n);
    for(int i = 0; i < n; i++) {
        int temp;
        scanf("%d", &temp);
        temp++;
        update(temp, 1);
    }
    int q;
    scanf("%d", &q);
    while(q--) {
    	int type;
    	scanf("%d", &type);
    	if(type == 0) {
    		int l, r;
    		scanf("%d%d", &l, &r);
    		l++; r++;
    		printf("%lld\n", read(r) - read(l - 1)); 
    	}
    	else if(type == 1){
    		int num;
    		scanf("%d", &num);
    		num++;
    		update(num, -1);
    	}
    	else if(type == 2) {
    		int num;
    		scanf("%d", &num);
    		num++;
    		update(num, 1);
    	}
    }
}

int main()
{
	solve();
    return 0;
}
