#include<bits/stdc++.h>
using namespace std;
pair<int,int> ans[51];
bool v[51];
pair<int,int> fibo(int n){
	if(v[n])
	   return ans[n];
	v[n]=true;
	if(n==0){
		return make_pair(1,0);
	}
	if(n==1){
		return make_pair(0,1);
	}
	pair<int,int> temp,temp1;
	temp=fibo(n-1);
	temp1=fibo(n-2);
	ans[n].first=temp.first+temp1.first;
	ans[n].second=temp.second+temp1.second;
	return ans[n];
}
int main(){
	int t;
	ans[1]=make_pair(0,1);
	ans[0]=make_pair(1,0);
	memset(v,false,sizeof(v));
	fibo(50);
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		printf("%d %d\n",ans[n].first,ans[n].second);
	}
	return 0;
}
