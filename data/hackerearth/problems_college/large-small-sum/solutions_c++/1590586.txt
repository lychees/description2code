#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t,i,l,r,n;
    int arr[1001];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=1;i<=n;i++)
    		cin>>arr[i];
    	sort(arr+1,arr+n+1);
    	cin>>l>>r;
    	cout<<arr[n-l+1]+arr[r]<<endl;
    }
    return 0;
}
