#include<bits/stdc++.h>
#define si(x) scanf("%d",&x)
#define sii(x,y) scanf("%d%d",&x,&y)
#define sl(x) scanf("%ld",&x)
#define sll(x,y) scanf("%ld%ld",&x,&y)
using namespace std;
typedef long long ll;
int main(){
     string str;
	 cin>>str;
	 string h=str.substr(0,2);
	 string m=str.substr(3);
	 int hr=stoi(h);
	 int min=stoi(m);
	 float hr_ang=hr*30+(min*(1/2.0));
	 float min_ang=min*6.0;
	 float res=abs(hr_ang-min_ang);
	 if(res>180){
	 	res=360-res;
	 }
	 printf("%.6f",res);
}
