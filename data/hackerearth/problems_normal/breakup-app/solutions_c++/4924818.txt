#include <iostream>
#include <cstring>
#include <vector>
#include <utility>
using namespace std;

int main()
{
    int n;
    cin>>n;
   	vector<pair<int,int>>a(30);;
	for(int i=0;i<30;i++)
	{
		a[i].first = i+1;
		a[i].second = 0;
	}
    for(int i=0;i<n;i++)
    {
    	string s;
    	getline(cin,s);
    	for(int i =0;i<30;i++)
    	{
    		if(s[0]=='G')
    		{
    			if(i<10 && s.find(to_string(a[i].first))!=std::string::npos)
    			{
    				int k = s.find(to_string(a[i].first));
    				if(s[k-1]<'0' && s[k+1]<'0')
    					a[i].second+=2;
    			}
    			else if(s.find(to_string(a[i].first))!=std::string::npos)
    			{
    				a[i].second +=2;
    			}
    		}
    		else
    		{
    			if(i<10 && s.find(to_string(a[i].first))!=std::string::npos)
    			{
    				int k = s.find(to_string(a[i].first));
    				if(s[k-1]<'0' && s[k+1]<'0')
    					a[i].second+=1;
    			}
    			else if(s.find(to_string(a[i].first))!=std::string::npos)
    			{
    				a[i].second +=1;
    			}	
    		}
    	}
    }
    int max=0,pos, flag=0;
	for(int i=0;i<30;i++)
		if(a[i].second>max)
		{
			max=a[i].second;
			pos =i;
		}
	for(int i=0;i<30;i++)
		if(a[i].second==max && i!=pos)
		{
			flag=1;
		}
	if(flag==1)
		cout<<"No Date";
	else if(pos==18 || pos==19)
		cout<<"Date";
	else
		cout<<"No Date";
    return 0;
}
