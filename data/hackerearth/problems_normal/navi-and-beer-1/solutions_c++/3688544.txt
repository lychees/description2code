#include <iostream>
using namespace std;

int main()
{
    int test;
    cin>>test;
    while(test--){
       long long  int n,k,m,i;
        cin>>n>>k>>m;
       long long  int sum=0;
        for(i=1;i<=n;i++){
            sum+=k*(i*i*i);
            
        }
        if(sum>=m)
        cout<<sum-m<<endl;
        else
        cout<<"0"<<endl;
    }
    return 0;
}
