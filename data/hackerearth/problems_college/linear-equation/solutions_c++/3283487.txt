#include <iostream>
#include <vector>

using namespace std;

int main(){

	int T;
	int N;
	long long int sum;
	cin >> T;

	while(T--){
		cin >> N;

		sum=0;
		vector<long long int> a(N);
		for(int i=0;i<N;i++){
			cin >> a[i];
			sum+=a[i];
	
		}
		sum/=(N-1);
		for(int i=0;i<N;i++){
			cout << (sum-a[i])<< " ";
		}
		cout << endl;

	}

	return 0;
}
