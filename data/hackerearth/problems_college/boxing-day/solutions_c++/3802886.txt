#include<bits/stdc++.h>
using namespace std;
int parent[100005];
int find_parent( int x )
{
	if( parent[x] == x )
		return parent[x];
	return ( parent[x] = find_parent(parent[x]));
}
int main()
{
	int n;
	cin>>n;
	int k;
	cin>>k;
	for( int i = 1;i <=n ;i++ ){
		parent[i] = i;
	}
	int c = 0;
	while( k-- )
	{
		int x,y;
		cin>>x>>y;
		int px = find_parent( x );
		int py = find_parent( y );
		if( px != py )
		{
			parent[py] = px;	
		}
	}
	for( int i = 1 ; i <= n ; i++ )
	{
		if( find_parent(i) == i)
		{
			c++;
		}	
	}
	cout<<c<<"\n";
	
	
}
