using namespace std;

string func(string s){
	string tmp = "";

	auto is_number = [&](int pos){
		if(s[pos] - '0' < 0 || s[pos] - '0' > 9) return false;
		return true;
	};

	for(int i=0; i<s.size(); i++){
		if(is_number(i)){
			string new_s = "";
			int times = s[i] - '0';
			while(times--){
				new_s += tmp;
			}
			if(i+1<s.size()) new_s += s.substr(i+1, string::npos);

			s = new_s;
			i = -1;
			tmp = "";
		}else{
			tmp += s[i];
		}
	}

	return s;
}

int main(){
	int t;
	cin >> t;
	while(t--){
		string s;
		cin >> s;
		string ret = func(s);
		cout << ret << endl;
	}
	return 0;
}
