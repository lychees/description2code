#include <iostream>
using namespace std;
#include<cmath>

int gcd(int n1, int n2)
{
    if (n2!=0)
       return gcd(n2, n1%n2);
    else 
       return n1;
}

int main()
{
    //printf("Hello World!\n");
    int t;
    cin>>t;
    for(;t--;)
    {
    	int a,b,x,y,m;
    	cin>>a>>b>>x>>y;
    	m=gcd(abs(a-x),abs(b-y));
    	if(m==0)
    	cout<<("0\n");
    	else
    	cout<<(m-1)<<endl;
    }
    return 0;
}
