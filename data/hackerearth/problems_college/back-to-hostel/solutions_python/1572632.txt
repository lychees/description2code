def isprime(num):
	if num < 2: return False
	if num == 2: return True
	for i in range(2,int(num**0.5)+1):
		if num%i == 0:
			return False
	return True

def ispal(s):
	for i in range(0,len(s)/2):
		if s[i] != s[len(s)-i-1]:
			return False
	return True

n = input()
count = 0
while n:
	s = raw_input()
	if isprime(len(s)):
		if ispal(s):
			count += 1
	n -= 1
print count
