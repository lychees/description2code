#include <iostream>
using namespace std;
int visited[1001][1001];
void activate(int * count, int x,int y,int r)
{
	
	
		for(int j = x - r;j <= x+r;j++)
		{
			for(int k = y-r;k<=y+r ;k++)
			{
				if(j>=1 && j <= 1000 && k >=1 && k <=1000  )
				{
			
				if( ((x-j)*(x-j) + (y-k)*(y-k)) <= (r * r))
			     { 
			     	if(visited[j][k]==0)visited[j][k]=1;
			      	else if(visited[j][k]==1)
			   	   {
			   		(*count)++;	
				    visited[j][k]++;
			     
		          }
		         
		         
				}
			
			}
		}
		}
}
int main()
{
    int n;
    cin>>n;
    int x,y,r;
    int count=0;
    while(n--)
    {
    	cin>>x>>y>>r;
    	activate(&count,x,y,r);
    }
    cout<<count;
    
    return 0;
}
