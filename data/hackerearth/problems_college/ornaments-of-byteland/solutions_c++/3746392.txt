#include <bits/stdc++.h>
using namespace std;
int cnt[10000][30];
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		string s;
		cin>>s;
		int x = s.size();
		for(int j=0;j<x;j++)
		{
			cnt[i][s[j]-'a']++;
		}
	}
	int ans = 0;
	for(int j=0;j<26;j++)
	{
		bool f = true;
		for(int i=1;i<=n;i++)
		{
			if(cnt[i][j]==0)
			{
				f= false;
				break;
			}
		}
		if(f)ans++;
	}
	cout<<ans;
    //cout << "Hello World!" << endl;
    return 0;
}
