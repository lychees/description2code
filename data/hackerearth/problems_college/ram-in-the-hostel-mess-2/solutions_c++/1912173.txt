#include <bits/stdc++.h>
using namespace std;
 
typedef long long ll;

ll countStrings(ll n)
{
    ll a[n], b[n];
    a[0] = b[0] = 1;
    for (ll i = 1; i < n; i++)
    {
        a[i] = a[i-1] + b[i-1];
        b[i] = a[i-1];
    }
    return a[n-1] + b[n-1];
}
 
int main()
{
	int t;
	scanf("%d", &t);
	
	while (t--) {
		ll n;
		scanf("%lld", &n);
		ll x = countStrings(n);
		printf("%lld\n", x*x);
	}
	
    return 0;
}
