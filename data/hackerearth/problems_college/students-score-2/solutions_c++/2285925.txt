#include <iostream>
using namespace std;

int main()
{
    long n,max=0,sum=0,i;
    cin>>n;
    long a[n];
    for(i=0;i<n;i++)
    	cin>>a[i];
    for(i=0;i<n;i++)
    {
    	if(sum==0 && a[i]>=0)
    	{
    		sum=sum+a[i];
    		if(sum>max)
    			max=sum;
    	}
    	else if(sum > 0)
    	{
    		sum=sum+a[i];
    		if(sum>max)
    			max=sum;
    	}
    	else if(sum<0)
    		sum=0;
    }
    cout<<max;
}
