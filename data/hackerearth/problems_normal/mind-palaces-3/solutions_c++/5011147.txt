#include<stdio.h>
using namespace std;
#include<map>

int main() {
	int n, m, i, j, q;
	long val;
	std::map<long, std::pair<int, int> > myMap;
	
	scanf("%d%d", &n, &m);
	for(i=0;i<n;i++)
		for(j=0;j<m;j++) {
			scanf("%ld", &val);
			myMap[val] = make_pair(i, j);
		}
	
	scanf("%d", &q);
	for(i=0;i<q;i++) {
		scanf("%ld", &val);
		if(myMap.find(val) == myMap.end())
			printf("-1 -1");
		else
			printf("%d %d", myMap[val].first, myMap[val].second);
		printf("\n");
	}
	
	return 0;
}
