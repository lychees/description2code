#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin >> t;
	while ( t-- ) {
		int n,m1;
		cin >> n >> m1;
		vector <string> v;
		map <string,int> m;
		map <string,int> :: iterator it;
		for ( int i = 0; i < n; i++ ) {
			string s;
			cin >> s;
			m[s]++;
			v.push_back(s);
		}
		for ( it = m.begin(); it != m.end(); it++ ) {
			if ( (*it).second > 1 ) {
				cout << "No" << endl;
				goto p1;
			}
		}
		cout << "Yes" << endl;
		p1: { }
		
	}
    return 0;
}
