#include<bits/stdc++.h>
#define ll long long int
using namespace std;


int main() {
    ll t,n,tim,a[100005],b[100005];
  cin >> t;
    while(t--)
    {
        cin >> n >> tim;
        for(int i = 0;i < n;i++)
            cin >> a[i];
        for(int i = 0;i < n;i++)
            cin >> b[i];
        ll ans = 0;
        for(int i = 0;i < n;i++)
        {
            ans = max(ans,( (tim/a[i]) * (b[i])));  
        }
        cout << ans << endl;
    }
}
