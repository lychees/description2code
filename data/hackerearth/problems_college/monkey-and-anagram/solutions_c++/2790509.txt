#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--)
	{
		string s,t;
		int l1,l2,flag=0,i,j,c=0,ans=0,k=0,flag1=0;
		cin>>s>>t;
		l1=s.length();
		l2=t.length();
		int visited[26];
		if(l1>l2)
		flag=1;
		else
		{
			for(i=0;i<26;i++)
			{
			//	counted[i]=0;
				visited[i]=0;
			}
			
			for(i=0;i<l1;i++)
			{
				for(j=0;j<l2;j++)
				{
					if(s[i]==t[j])
					{
			//			cout<<j<<" ";
						flag1=1;
						t[j]='X';
						break;
					}
				}
				if(flag1!=1)
				{
					flag=1;
					break;
				}
				flag1=0;
			}
			
		//	cout<<t;
		//	cout<<flag<<endl;
		if(flag!=1)
		{
			for(i=0;i<l2;i++)
			{
				if(t[i]!='X')
				{
					int x=(int)t[i]-97;
					if(visited[x]==0)
					{
						c++;
						visited[x]=1;
					}
				
				}
			}
			ans=c;
		}
		}
		if(flag==1)
		cout<<"-1"<<endl;
		else
		cout<<ans<<endl;
	}
	return 0;
}
