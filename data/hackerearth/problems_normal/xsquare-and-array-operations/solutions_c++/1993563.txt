#include <bits/stdc++.h>
using namespace std;
long int maximum(long int a,long int b)
{
	return((a>b)?a:b);
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long int n,i,temp,op1,op2,op3,sum=0;
    	queue<long int>myqueue;
    	scanf("%ld",&n);
    	for(i=0;i<n;i++)
    	{
    		scanf("%ld",&temp);
    		myqueue.push(temp);
    	}
    	while(myqueue.size()!=1)
    	{
    		op1=myqueue.front();
    		myqueue.pop();
    		op2=myqueue.front();
    		op3=maximum(op1,op2);
    		sum+=op3;
    	}
    	cout<<sum<<endl;
    //	myqueue.clear();
    }
}
