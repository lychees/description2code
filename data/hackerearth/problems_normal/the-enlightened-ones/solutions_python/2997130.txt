def check(t, nm, d):
	m = t[0] + d
	nm -= 1
	for i in xrange(1, n):
		if((m + d) >= t[i]):
			continue
		else :
			if nm <= 0:
				return 0
			m = t[i] + d
			nm -= 1
	return 1
	
n, k = map(int, raw_input().split())
t = map(int, raw_input().split())
t.sort()
low, high, pos = 0, 10000000, -1
while low <= high :
	mid = (low + high) / 2
	if check(t, k, mid) :
		pos = mid
		high = mid-1
	else :
		low = mid + 1
print pos
