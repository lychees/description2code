#include <bits/stdc++.h>
using namespace std;
int p1[1100];
int p2[1100];
int lcs[1100][1100];
int main()
{
    ios_base::sync_with_stdio(0);
    int t,n,pos;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(int i=1;i<=n;i++)
    	{
    		cin>>pos;
    		p1[pos]=i;
    		cin>>pos;
    		p2[pos]=i;
    	}
    	
    	for(int i=1;i<=n;i++)
    	{
    		for(int j=1;j<=n;j++)
    		{
    			if(p1[i]==p2[j])
    			lcs[i][j]=lcs[i-1][j-1]+1;
    			else
    			lcs[i][j]=max(lcs[i-1][j],lcs[i][j-1]);
    		}
    	}
    	cout<<lcs[n][n]<<"\n";
    }
    return 0;
}
