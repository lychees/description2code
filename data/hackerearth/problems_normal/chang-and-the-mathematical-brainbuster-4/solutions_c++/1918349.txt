#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int n;
    cin >> n;
    
    int a[n+1];
    
    for (int i = 0; i < n; i++) scanf("%d", &a[i]);
    
    sort(a, a+n);
    
    int ans = 100000;
    
    for (int i = 0; i < n; i++) {
    	for (int j = i+1; j < n; j++) {
    		if (a[j] - 3*a[i] <= 0) {
    			ans = min (ans, n-(j-i+1));
    		}
    	}
    }
    
    cout << ans << endl;
    return 0;
}
