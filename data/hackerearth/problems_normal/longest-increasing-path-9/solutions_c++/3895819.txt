#include <iostream>
#include <iomanip>
#include <cassert>
#include <vector>
#include <string.h>

using namespace std;

typedef long long int int64;
int main()
{
	int64 i,j,k,n,m,t,a[101][101],ans[101][101],ga;
	cin>>t;
	while(t--)
	{
		scanf("%lld %lld",&n,&m);
		memset(ans,0,sizeof(ans[0][0])*101*101);ans[0][0]=1;ga=-1;
		for(i=0;i<n;i++)
			for(j=0;j<m;j++)
				scanf("%lld",&a[i][j]);
		for(i=0,j=1;j<m;j++)
			if(a[i][j-1]<a[i][j]&&ans[i][j-1]==1)
				ans[i][j]=1;
		for(i=1,j=0;i<n;i++)
			if(a[i-1][j]<a[i][j]&&ans[i-1][j]==1)
				ans[i][j]=1;
		for(i=1;i<n;i++)for(j=1;j<m;j++)
		{
			if(a[i][j-1]<a[i][j]&&ans[i][j-1]==1)
				ans[i][j]=1;
			if(a[i-1][j]<a[i][j]&&ans[i-1][j]==1)
				ans[i][j]=1;
		}
		for(i=0;i<n;i++)
			for(j=0;j<m;j++)
				if(ans[i][j]==1&&i+j+1>ga)
					ga=i+j+1;
					
		cout<<ga<<endl;
	}
	return 0;
}
