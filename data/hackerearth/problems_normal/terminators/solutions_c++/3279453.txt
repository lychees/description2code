#include <bits/stdc++.h>
#define UP 1
#define LEFT 2
#define DIAG 3
using namespace std;
int a[1005], b[1005];
int dp[1005][1005];
int main(){
int n, test_case, x, y;
scanf("%d ", &test_case);
while(test_case--){
	scanf("%d", &n);
	for(int i=1; i<=n; i++){
		scanf("%d %d", &x, &y);
		a[x]=i; b[y]=i;
	}
	for(int i=0; i<=n; i++){
		for(int j=0; j<=n; j++){
			dp[i][j]=0;
		}
	}
	for(int i=1; i<=n; i++){
		for(int j=1; j<=n; j++){
			if(a[i]==b[j]){
				dp[i][j] = dp[i-1][j-1]+1;
			}else{
				dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
			}
		}
	}
	cout<<dp[n][n]<<endl;
}
}
