#include <iostream>
#include <string.h>
using namespace std;


int calcIndex(string s){
	int i,j;
	
	for(i=0, j=s.size()-1; i<j ; ){
		if(s[i]==s[j]) { ++i; --j;}
		else{
			if(j-i>1){ 
				if(s[i]==s[j-1]) return j;  
				else return i;
				}
			else 
			    return i;
		}
	}
	return -1;
}

int main()
{
    int i=0,t;
    string s;
    
    cin>>t;
    while(i<t){
    	cin>>s;
    	cout<<calcIndex(s)<<"\n";    
    	++i;
    }
    
   
    return 0;
}
