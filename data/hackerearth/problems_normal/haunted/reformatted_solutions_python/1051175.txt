N, M = map(int, raw_input().split(' '))
awardsMap = {}
maxAge = maxAwards = 0
ages = map(int, raw_input().split(' '))
ans = ''
for i in range(N):
    if ages[i] in awardsMap:
        awards = awardsMap[ages[i]]+1
    else:
        awards = 1
    awardsMap[ages[i]] = awards
    if awards > maxAwards or (awards == maxAwards and ages[i] > maxAge):
        maxAge = ages[i]
        maxAwards = awards
    ans += "%d %d\n" % (maxAge, maxAwards)
print ans
