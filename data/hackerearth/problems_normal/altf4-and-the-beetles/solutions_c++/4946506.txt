#include <bits/stdc++.h>
using namespace std;

#define IO ios_base::sync_with_stdio(false); cin.tie(0);
#define endl "\n"
#define DBG 666
#define mp make_pair
#define revsort(x,y) sort((x),(x+y));reverse((x),(x+y));

typedef long long ll;

int main(){

	IO;
	int t;
	cin >> t;

	while (t--){

		ll n, a, b, c;
		vector< pair<ll, ll > > v;
		cin >> n >> a >> b >> c;
		for (int i = 0; i < n; i++){
			ll l, r;
			cin >> l >> r;
			v.push_back(mp(l, 1));
			v.push_back(mp(r, 2));
		}

		sort(v.begin(), v.end());

		ll res = n*a, curr = n*a;
		for (int i = 0; i < v.size(); i++){

			if (v[i].second == 1){
				curr += (b - a);
			}
			if (v[i].second == 2){
				curr += (c - b);
			}
			res = max(res, curr);
		}
		cout << res << endl;
	}

	return 0;
}
