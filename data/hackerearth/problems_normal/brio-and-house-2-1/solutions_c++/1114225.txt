#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	int t;
	cin >> t;
	
	while(t--) {
		double x1,y1,x2,y2,r;
		cin >> x1 >> y1 >> x2 >> y2;
		cin >> r;
		double an;
		
		double b = x2-x1;
		double a = y2-y1;			
		double c = a*x1 - b*y1;
		c = abs(c);
		double d = sqrt(a*a + b*b);
		if(d != 0) an = c / d;
		
		if(an == r) {
			cout << "JUST MISSED" << endl;
		} else if(an < r) {
			 cout << "REPLANNING" << endl;
		} else {
			cout << "SAFE" << endl;
		}
	}
    return 0;
}
