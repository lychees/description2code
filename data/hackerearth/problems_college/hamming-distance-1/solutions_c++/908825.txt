#include<stdio.h>
void work(int a,int b)
{
    int x,y,i,c=0;
    for(i=0;i<32;i++)
    {
        x=a%2;
        y=b%2;
        if(x!=y)
            c++;
        a=a/2;
        b=b/2;
    }
    printf("%d\n",c);
}




int main()
{
    int ntc,a,b;
    scanf("%d",&ntc);
    while(ntc--)
    {
        scanf("%d%d",&a,&b);
        work(a,b);
    }
}
