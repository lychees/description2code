#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        if(n==0 || n==1)
            cout<<4<<endl;
        else if(n&1)
            cout<<n+1<<endl;
        else
        {
            int flag=1;
            for(int i=2;(i*i)<=(n+1);i++)
            {
                if(((n+1)%i)==0)
                {
                    flag=0;
                    break;
                }
            }
            if (flag==0)
                cout<<n+1<<endl;
            else
                cout<<n+2<<endl;
        }
    }
    return 0;
}
