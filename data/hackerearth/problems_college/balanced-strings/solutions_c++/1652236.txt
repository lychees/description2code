#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {int flag=1;
   	string str;
   	cin>>str;
   	int a[26]={0};
   	for(int i=0;i<str.length();i++)
   	a[str[i]-97]++;
   	for(int i=0;i<25;i++)
   	if(!((a[i]==a[i+1])&&a[i]&&a[i+1]))flag=0;
   	if(flag)cout<<"Yes"<<endl;else
   	cout<<"No"<<endl;
   }
    return 0;
}
