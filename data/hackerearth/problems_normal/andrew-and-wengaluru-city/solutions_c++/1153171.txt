#include <iostream>
using namespace std;

int main() {
	int tow[100001];
	int t,n,i;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(int a=0;a<n;a++)
		cin>>tow[a];
	
int hl[100001] = {0}, hr[100001] = {0}; // highest-left and highest-right
for (int i = 0; i < n; i++) hl[i] = max(tow[i], (i!=0)?hl[i-1]:0);
for (int i = n-1; i >= 0; i--) hr[i] = max(tow[i],i<(n-1) ? hr[i+1]:0);
long long ans = 0;
for (int i = 0; i < n; i++) {ans += min(hl[i], hr[i]) - tow[i];ans%=1000000007;}
cout<<ans<<endl;
	}
	
	return 0;
}
