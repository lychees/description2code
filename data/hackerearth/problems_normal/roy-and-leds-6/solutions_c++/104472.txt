#include <iostream>
using namespace std;

int main()
{
	int t,r,g,b,rflag=0,gflag=0,bflag=0,yflag=0,cflag=0,mflag=0,wflag=0,blkflag=0;
	int rcnt = 0,gcnt = 0,bcnt = 0,ycnt = 0,ccnt = 0,mcnt = 0,wcnt = 0,blkcnt = 0;
	int a[8] = {0,0,0,0,0,0,0,0};
    cin>>t>>r>>g>>b;
    for(int i=0; i<t; i++)
    {
    	if(rcnt == r){
    		if(rflag==0){
    			rflag=1;
    		}
    		else{
    			rflag=0;
    		}
    		rcnt=0;
    	}
    	
    	if(gcnt == g){
    		if(gflag==0){
    			gflag=1;
    		}
    		else{
    			gflag=0;
    		}
    		gcnt=0;
    	}
    	
    	if(bcnt == b){
    		if(bflag==0){
    			bflag=1;
    		}
    		else{
    			bflag=0;
    		}
    		bcnt=0;
    	}
    	
    	rcnt++;
    	gcnt++;
    	bcnt++;
    	
    	if(rflag && !gflag && !bflag){
    		++a[0];
    	}
    	else if(!rflag && gflag && !bflag){
    		++a[1];
    	}
    	else if(!rflag && !gflag && bflag){
    		++a[2];
    	}
    	else if(rflag && gflag && !bflag)
    	{
    		yflag = 1;
    		++a[3];
    	}
    	else if(gflag && bflag && !rflag)
    	{
    		cflag = 1;
    		++a[4];
    	}
    	else if(rflag && bflag && !gflag)
    	{
    		mflag = 1;
    		++a[5];
    	}
    	else if(rflag && gflag && bflag)
    	{
    		wflag = 1;
    		++a[6];
    	}
    	else if(!rflag && !gflag && !bflag)
    	{
    		blkflag = 1;
    		++a[7];
    	}
    	else{
    		cout<<rflag<<gflag<<bflag;
    	}
    	
    }
    
    for(int i=0;i<8;i++)
    	{
    		cout<<a[i]<<" ";
    	}
    
    return 0;
}
