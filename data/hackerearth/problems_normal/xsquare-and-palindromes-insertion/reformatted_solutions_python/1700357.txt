for t in range(int(raw_input())):
    string = raw_input()
    num = [0 for i in range(26)]
    for s in string:
        num[ord(s)-97] += 1
    total = 0
    odd = False
    for i in num:
        if i % 2 == 1:
            if not odd:
                odd = True
            else:
                total += 1
    print total
