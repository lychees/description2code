#define pi pair<long long,long long>

#define mod 1000000007

using namespace std;
int prime[6000];
vector<int> p;

void sieve(int n)
{
    for(int i=2;i<=n;++i)prime[i]=1;
    for(int i=2;i<=sqrt(n);++i)
    {
        if(prime[i])
        for(int j=i*i;j<=n;j+=i)
        {
            prime[j]=0;
        }
    }
    for(int i=2;i<=n;++i)
    {
        if(prime[i])p.push_back(i);
    }
}


int n,a[6000],csum[6000],dp[6000];


main()
{
    sieve(6000);
    ios_base::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    int t;
    //cin>>t;
    t=1;

    for(int tx=1;tx<=t;++tx)
    {
        cin>>n;
        for(int i=0;i<=n-1;++i)
        {
            cin>>a[i];
            csum[i]=a[i];
            if(i>0)csum[i]+=csum[i-1];
        }
           //csum[n]=csum[n-1];

        for(int i=n-1;i>=0;i--)
        {
            for(int j=0;j<p.size();++j)
            {
                int sum=0,siz=p[j];
                if(i+siz>n)continue;

                sum=csum[i+siz-1];
                if(i-1>=0)sum-=csum[i-1];

                dp[i]=max(dp[i],sum+dp[i+siz+1]);
           //  if(i==0)cout<<p[j]<<" "<<dp[i]<<"\n";
            }
            dp[i]=max(dp[i],dp[i+1]);
            //cout<<dp[i]<<" ";
        }
        cout<<dp[0]<<endl;


         for(int i=0;i<=n;++i)
         {
             dp[i]=0;a[i]=0;csum[i]=0;
         }
    }

    return 0;
}
