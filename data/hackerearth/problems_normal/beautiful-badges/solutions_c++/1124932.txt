#include <bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
#define REP(i,n) for(int i=0; i<n; i++)
#define R(x) scanf("%d",&x)
#define F first
#define S second
typedef long long int64;
typedef pair<int,int> pii;

int main() {
	int n,a,b;

	while (R(n)==1) {
		vector<int> v[2];
		v[0].clear();
		v[1].clear();
		
		REP(i,n) {
			R(a),R(b);
			v[a-1].push_back(b);
		}

		sort(v[0].begin(),v[0].end());
		sort(v[1].begin(),v[1].end());

		int64 res=0, j=0, k=0;

		REP(i,v[k].size()) {
			int tmn=ceil(v[k][i]*sqrt(2)/2), tmx=floor(v[k][i]*sqrt(2));

			while (j<v[1].size() && v[1][j]<tmn) j++;

			if (j<v[1].size() && v[1][j]<=tmx) {
				res++;
				j++;
			}
		}

		printf("%lld\n",res);
	}
	return 0;
}
