#include <iostream>
#define ll long long

using namespace std;

int main()
{
    ll n = 0;
    ll a[26] = {0};
    string s = "";
    int flag = 0;
    
    cin >> n;
    while (n--){
    	
    	cin >> s;
    	for (ll i = 0; i < s.length(); i++){
    		a[int(s[i])-97]++;
    	}
    	// check
    	/*
    	for (ll i = 1; i < 26; i++){
    		cout << a[i] << " ";
    	}
		cout << endl;
		*/
		
    	for (ll i = 1; i < 26; i++){
    		if (a[0] != a[i]){
    			flag = 1;
    			break;
    		}
    	}
    	//cout << endl;
    	// output
    	if (flag)
    		cout << "No" << endl;
    	else
    		cout << "Yes" << endl;
    		
    	for (ll i = 0; i < 26; i++){
    		a[i] = 0;
    	}
    	flag = 0;
    }
    return 0;
}
