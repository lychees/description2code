#include <iostream>
#include <algorithm>
using namespace std;
typedef long long int ll;
int main()
{
    int t;
    cin >> t;
    while(t--) {
    	ll n;
    	cin >> n;
    	ll a[n];
    	ll i;
    	for(i = 0; i < n; i++) {
    		cin >> a[i];
    	}
    	sort(a, a + n);
    	ll p, q, r;
    	for(i = n - 3; i >= 0; i--) {
    		p = a[i];
    		q = a[i + 1];
    		r = a[i + 2];
    		if(p + q > r) {
    			break;
    		}
    	}
    	if(i != -1) {
    		cout << p << " " << q  << " " << r << endl;
    	}
    	else {
    		cout << i << endl;
    	}
    }
    return 0;
}
