#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t=0;
    scanf("%d",&t);
    while(t--)
    {
    	int n=0,k=0,i=0,j=0,max1=0,ans=0;
    	scanf("%d%d",&n,&k);
    	int a[n],b[n];
    	bitset<16> s;
    	for(i=0;i<n;i++)
    		scanf("%d",&a[i]);
    	for(i=0;i<n;i++)
    	{
    		bitset<16> s;
    		s=a[i];
    		b[i]=s.count();
    	}
    	sort(b,b+n);
    	j=n-1;
    	while(k--)
    		ans+=b[j--];
    	
    	cout<<ans<<endl;
    }
    return 0;
}
