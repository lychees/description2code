n = input()
hill = []
for i in range(n):
    hill.append(map(int, raw_input().split()))
seen = set()

for i in range(n):
    maxSlope = -10**50
    for j in range(i+1, n):
        x1, y1 = hill[i][0], hill[i][1]
        x2, y2 = hill[j][0], hill[j][1]
        newSlope = float(y2-y1)/float(x2-x1)
        if newSlope >= maxSlope:
            seen.add((x1, x2))
            maxSlope = newSlope
print len(seen)
