#include <iostream>
#include<stack>
using namespace std;

int main()
{
    int n,x;
    int ch;
    cin>>n;
    stack <int> s;
    while(n--)
    {
    	cin>>ch;
    
    	switch(ch)
    	{
    		case 1: if(s.empty())
    		          cout<<"No Food";
    		       else{
    		       cout<<s.top();
    		        s.pop();
    		       }
    		        cout<<endl;
    		        break;
    		       
    		case 2: cin>>x;
    		         s.push(x);
    	}
    
    }
    return 0;
}
