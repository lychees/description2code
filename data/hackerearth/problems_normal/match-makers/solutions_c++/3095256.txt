#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    long long int n,a[10000],b[10000],i,j,k,c=0;
    cin>>t;
    for(i=1;i<=t;i++)
    {c=0;
    cin>>n;
    for(j=0;j<n;j++)
    cin>>a[j];
    for(k=0;k<n;k++)
    cin>>b[k];
    sort(a,a+n);
    sort(b,b+n);
    for(j=0;j<n;j++)
    {
    	if(a[j]%b[n-1-j]==0||b[n-1-j]%a[j]==0)
    	c++;
    }
    cout<<c<<"\n";
    	
    }
    
    return 0;
}
