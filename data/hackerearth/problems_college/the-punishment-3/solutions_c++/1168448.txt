#include <iostream>
using namespace std;

int main()
{
    int t;cin>>t;
    while(t--){
    	int n,k,i;
    	cin>>n>>k;
    	
    	int x=min(k,n-k);
    	int y=abs(2*k-n);
    	
    	for(i=1;i<x;i++)
    	cout<<"9";
    	cout<<"8";
    	for(i=0;i<y;i++)
    	cout<<"9";
    	for(i=1;i<x;i++)
    	cout<<"0";
    	cout<<"1";
    	cout<<endl;
    }
}
