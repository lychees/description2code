#define MOD 1000000007
#define N 1000004
ill fact[N], ifact[N];

ill pow(ill a, ill b, ill mod) {
    ill x = 1, y = a;
    while ( b > 0 ) {
        if ( b%2 == 1 ) {
            x = x*y;
            if ( x >= mod )
                x %= mod;
        }
        y = y*y;
        if ( y >= mod )
            y%=mod;
        b /= 2;
    } return x;
}

void pre() {
    fact[0] = ifact[0] = 1;
    F(i,1,N) {
        fact[i] = i*fact[i-1];
        fact[i] %= MOD;
        ifact[i] = pow(fact[i],MOD-2,MOD);
    }
}

char str1[N], str2[N];

int main() {
    // input();
    pre();
    int t; S(t);
    while ( t-- ) {
        scanf("%s", str1); scanf("%s", str2);
        // cout << str1 << " " << str2 << endl;
        int cnt1[26], cnt2[26];
        M(cnt1,0); M(cnt2,0);
        for ( int i = 0; str1[i] != '\0'; ++i )
            cnt1[str1[i]-'a']++;
        for ( int i = 0; str2[i] != '\0'; ++i )
            cnt2[str2[i]-'a']++;
        ill ans = 1;
        F(i,0,26) {
            // cout << cnt1[i] << " " << cnt2[i] << endl;
            if ( cnt1[i] < cnt2[i] ) {
                ans = 0;
                break;
            }
            ill temp = (fact[cnt1[i]] * ifact[cnt1[i]-cnt2[i]]) % MOD;
            ans = (ans * temp) % MOD;
        }
        printf("%lld\n", ans);
    }
    return 0;
}
