#include<bits/stdc++.h>
using namespace std;
int n,k,t;
int a;
#define mod 1000000007
int ans=0;
long long f[2009]={0},inv[2009]={0};
long long power(long long a,long long n)
{
	if(n==1)return a;
	if(n==0)return 1;
	long long t=power(a,n/2);
	t=(t*t)%mod;
	if(n&1)
		t=(t*a)%mod;
	return t;
}
void init()
{
	f[0]=f[1]=1;
	for(int i=2;i<=2006;i++)
		f[i]=(f[i-1]*i)%mod;
	inv[2006]=power(f[2006],mod-2);
	for(int i=2005;i>=0;i--)
	{
		inv[i]=(inv[i+1]*(i+1))%mod;
	}
	
//	for(int i=0;i<=10;i++)cout<<inv[i]<<" ";cout<<endl;
//	cout<<(f[2000]*inv[2000])%mod<<endl;
}
long long ncr(int k,int y)
{
	//cout<<"done"<<endl;
	if(y==0||y==k)return 1;
	//cout<<"k="<<k<<" y="<<y<<endl;
	return (((f[k]*inv[y])%mod)*inv[k-y])%mod;
}
int main()
{
	init();
	cin>>n>>k>>t;
	for(int i=1;i<=k;i++)
	{
		cin>>a;
		ans+=a-1;
	}
	int y=t-ans;
//	cout<<"y="<<y<<endl;
	if(y&1 || y<0)cout<<0<<endl;
	else
	{
		y/=2;
		if(k<y)cout<<0<<endl;
		else
		{
		//	cout<<"done"<<endl;
			cout<<ncr(k,y)<<endl;
		//	cout<<"done"<<endl;
		}
	}
	return 0;
}
