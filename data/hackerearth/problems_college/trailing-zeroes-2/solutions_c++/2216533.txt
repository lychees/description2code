#include<bits/stdc++.h>
int a[10000000];
using namespace std;
int findTrailingZeros(int  n)
{
    int count = 0;
    for (int i=5; n/i>=1; i *= 5)
          count += n/i;

    return count;
}
int main()
{
	int t,i;
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
		scanf("%d",&a[i]);
        cout << findTrailingZeros(a[i])<<endl;
	}
    return 0;
}
