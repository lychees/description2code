#include <iostream>
#include<cstdio>
using namespace std;

void swap (char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}
void permute(char *a, int i, int n)
{
   int j;
   if (i == n)
     printf("%s\n", a);
   else
   {
        for (j = i; j <= n; j++)
       {
          swap((a+i), (a+j));
          permute(a, i+1, n);
          swap((a+i), (a+j)); //backtrack
       }
   }
}
int main()
{     char str[10000];
      cin>>str;
      int t;
     cin>>t;
     while(t--)
     { int x,y;
   	  cin>>x>>y;
   	permute(str, x, y);cout<<endl;
     }

   return 0;
}
