#include <iostream>
using namespace std;

int main()
{
    string s,t;
    cin>>s;
    cin>>t;
    int i=0,j=0,l,l1,count=0;
    l=s.length();
    l1=t.length();
    for(j=0;j<l1;j++)
    {
        if(i>l)
        break;
    	if(s[i]==t[j])
    	{
    		count++;
    		i++;
    	}
    }
    cout<<count<<endl;
    return 0;
}
