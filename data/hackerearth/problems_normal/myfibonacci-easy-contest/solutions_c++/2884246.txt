#include <iostream>
#include<cmath>
#include<limits.h>
using namespace std;

#define MAX 100
int fab(int A,int B,int N);
int main()
{
	int A,B,N;
	cin>>A>>B>>N;
	cout<<fab(A,B,N);
    return 0;
}
int fab(int A,int B,int N)
{
	int FAB[MAX];
	for(int i=0;i<MAX;i++)
		FAB[i] = 0;
	FAB[1]=A;
	FAB[2]=B;
	if(N == 1)
		return FAB[1];
	else if(N == 2)
		return FAB[2];
	else
	{
		for(int i=3;i<=N;i++)
		FAB[i]=FAB[i-1] + FAB[i-2];
	}
	return FAB[N];
}
