#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	double x1,y1,x2,y2,r,a,b,c,d,e;
    	cin>>x1>>y1>>x2>>y2;
    	cin>>r;
    	a=y2-y1;
    	b=x2-x1;
    	c=sqrt((a*a+b*b)*1.0);
    	d=abs(y2*x1-x2*y1);
    	e=(d*1.0)/c;
    	if(e>r)
    	cout<<"SAFE"<<endl;
    	if(e<r)
    	{
    	cout<<"REPLANNING"<<endl;
    	}
    	if(e==r)
    	{
    	cout<<"JUST MISSED"<<endl;
    	}
    }
    return 0;
}
