#include <iostream>
using namespace std;

int main()
{
	int cases;
	cin >> cases;
	bool arr[1000001] = {0};
	int prime[1000000];
	int in = 0;
	for(int i = 2;i <= 1000000;i++) {
		if(arr[i] == 0) {
			for(int j = 2 * i;j <= 1000000;j += i) {
				arr[j] = 1;
			}
			prime[in] = i;
			in++;
		}
		
	}
	while(cases--) {
		int n;
		scanf("%d", &n);
		if(n == 1 || n == 2) {
			cout << "-1" << " " << -1 << endl;
		}else {
			int low = 0;
			int high = in - 1;
			int flag = 0;
			int mxl = 0;
			int mxh = 0;
			while(low < high) {
				if(prime[low] + prime[high] == n) {
					if(prime[low] > mxl) {
						mxl = prime[low];
						mxh = prime[high];
					}
					flag = 1;
					low++;
					high--;
				}else if(prime[low] + prime[high] < n) {
					low++;
				}else {
					high--;
				}
			}
			if(flag == 1) {
				cout << mxl << " " << mxh << endl;
			}else {
				cout << -1 << " " << -1 << endl;
			}
		}
	}
   
    return 0;
}
