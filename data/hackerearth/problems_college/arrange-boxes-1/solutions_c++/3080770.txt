const int mx = 1e5 + 10;
using namespace std;

int a[mx];
int b[mx];
int c[mx];

long long int cnt;
void merge(int l,int m,int h)
{
     int i=l,j=m+1,k=l,ii;
     while(i<=m&&j<=h)
     {
                      if(c[a[i]]<=c[a[j]])
                      {
                                    b[k]=a[i];
                                    i++;
                      }
                      else
                      {
                          b[k]=a[j];
                          cnt+=(m-i+1);
                          j++;
                      }
                      k++;
     }
     if(i>m)
     {
            for(ii=j;ii<=h;ii++)
            {
                                b[k]=a[ii];
                                k++;
            }
     }
     else
     {
         for(ii=i;ii<=m;ii++)
         {
                             b[k]=a[ii];
                             k++;
         }
     }
     for(k=l;k<=h;k++)a[k]=b[k];
}
void merge_sort(int l,int h)
{
     int m;
     if(l<h)
     {
            m=(l+h)/2;
            merge_sort(l,m);
            merge_sort(m+1,h);
            merge(l,m,h);
     }
}
int main(){
    int t,n,i,num;
    cin>>t;
    while(t--) {
              scanf("%d",&n);
              for(i=0;i<n;i++)
              {
                              scanf("%d",&a[i]);
              }
              for(i=0;i<n;i++)
              {
                       scanf("%d",&num);
                       c[num]=i;
              }
                 cnt=0;
                 merge_sort(0,n-1);
                 printf("%lld\n",cnt);
    }
    return 0;
}
