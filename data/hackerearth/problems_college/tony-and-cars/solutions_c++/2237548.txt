ll n, ans;

void solve(ll n){
	ans = 0;
	ll m = n, k = 0, o = 0;

	while(m){
		o = o*10 + 1;
		k++;
		m = m/10;
	}

	ll d[10];
	mem(d, 0);

	ll t=ceil(pow(10,k-1));
	int i;
	while(t){
		int dig = (n/t)%10;
		int rest = n%t;
		k--;
		for(i = 0; i < 10; i++){
			d[i]=d[i]+((dig*(t/10)*k));
		}
		for(i = 0; i < dig; i++){
			d[i]=d[i]+t;
		}
		d[dig]+=(rest+1);
		t/=10;
	}
	
	d[0]=d[0]-o;
	
	for(int i = 0; i < 10; ++i){
		ans += d[i];
	}
	cout<<ans<<endl;
}
