#include <iostream>
using namespace std;

int main()
{
    int t, n;
    cin>>t;
    int a[4]={7, 5, 3, 2};
    while(t--){
    	cin>>n;
    	int c[n+1];
    	fill(c, c+n+1, 0);
    	for(int i=0; i<4; i++){
    		for(int j=a[i]; j<=n; j++){
    			if(j-a[i]==0 || c[j-a[i]] > 0){
    				if(c[j]==0){
    					c[j]=c[j-a[i]]+1;
    				}else{
    					c[j]=min(c[j], c[j-a[i]]+1);
    				}
    			}
    		}
    	}
    	if(c[n]){
    		cout<<c[n]<<endl;
    	}else{
    		cout<<"-1\n";
    	}
    }
    return 0;
}
