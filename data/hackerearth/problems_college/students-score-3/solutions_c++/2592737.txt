#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
    int n;
    scanf("%d", &n);
    int max=0, tmax=0;
    while(n--){
    	int temp;
    	scanf("%d", &temp);
    	tmax+=temp;
    	if(tmax<0){
    		tmax=0;
    	}
    	if(tmax>max){
    		max=tmax;
    	}
    }
    printf("%d", max);
}
