#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int N = 500 + 100;
const int K = 10 + 100;

string q[N];

int main()
{
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	int t;
	cin >> t;
	while (t--) {
		int n, k;
		cin >> n >> k;
		for (int i = 0; i < n; ++i) cin >> q[i];
		
		int ans = k;
		for (int i = 1; i < (1<<k); ++i) {
			bool ok = true;
			for (int j = 0; j < n; ++j) {
				bool now = false;
				for (int p = 0; p < k; ++p) {
					if ( (i & (1<<p)) ) {
						now |= q[j][p] == '1';
					}
				}
				ok &= now;
			}
			if (ok) {
				int cnt = 0;
				for (int j = 0; j < k; ++j) {
					if (i & (1 << j)) ++cnt;
				}
				ans = min(ans, cnt);
			}
		}
		cout << ans << '\n';
	}
    return 0;
}
