#include <iostream>
using namespace std;
typedef long long int ll;
ll ans(ll a,ll b,ll c,ll k){
ll low=0,high=100000,mid;
while(low<high){
mid = (low+high)/2;
if((a*mid)*mid + b*mid + c >= k){
high=mid;
}else{
low = mid+1;
}
}
return high;
}


int main()
{
    ll A,B,C,K;
    int t;
    cin>>t;
    while(t--){
    cin>>A>>B>>C>>K;
    cout<<ans(A,B,C,K)<<"\n";
    }
    return 0;
}
