#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
	int t;
	scanf("%d",&t);
	while(t--) {
		string s;
		cin >> s;
		int x,y,z,j=0; x=y=z=0;
		ll ans = 0;
		for(int i=0;i<s.length();i++) {
			if(s[i]=='x') x++;
			if(s[i]=='y') y++;
			if(s[i]=='z') z++;
			while(j<=i && x && y && z) {
				if(s[j]=='x') x--;
				if(s[j]=='y') y--;
				if(s[j]=='z') z--;
				j++;
			}
			ans += (i-j+1);
		}
		printf("%lld\n",ans);
	}
	return 0;
}
