def main():
    a = int(input())
    b = sorted([int(k) for k in raw_input().split(' ')])
    history = -1
    mem = [False] * 1010
    m = 0
    res = len(set(b)) - 1
    for i in range(a):
        if b[i] == history:
            m += 1
            if mem[m]:
                res += 1
            mem[m] = True
        else:
            m = 0
        history = b[i]
    print res


main()
