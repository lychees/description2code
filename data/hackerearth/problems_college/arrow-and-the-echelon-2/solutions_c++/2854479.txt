#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		if(n<3)
		{
			cout<<"-1"<<"\n";
		}
		else if(n%3==0)
		{
			for(int i=0;i<n;i++)
			{
				cout<<"5";
			}
			cout<<"\n";
		}
		
		else if(n%5==0)
		{
			for(int j=0;j<n;j++)
			{
				cout<<"3";
			}
		    cout<<"\n";
		}
		else 
		{
			int a=1;
			int c=0;
			while(1)
			{
				
				int s1=n-a;
				int s2=a;
				if(s1==0)
				{
					c=1;
					break;
				}
				if(s1%3==0 && s2%5==0)
				{
					for(int i=0;i<s1;i++)
					{
						cout<<"5";
					}
					for(int j=0;j<s2;j++)
					{
						cout<<"3";
					}
					cout<<"\n";
					break;
				}
				else if(s1%5==0 && s2%3==0)
				{
					for(int i=0;i<s1;i++)
					{
						cout<<"3";
					}
					for(int j=0;j<s2;j++)
					{
						cout<<"5";
					}
					cout<<"\n";
				}
				a++;
			}
			if(c==1)
			{
				cout<<"-1"<<"\n";
				break;
			}
		}
	}
}
