#include<bits/stdc++.h>
using namespace std;

string str;

int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        cin>>str;
        int idx1=-1;
        if(str.find("@")!=string::npos) {
            idx1=str.find("@");
        }
        int idx2=-1;
        if(str.find(".")!=string::npos) {
            idx2=str.find(".");
        }
        if(idx1==-1||idx2==-1) {
            puts("NO");
            continue;
        }
        int cnt=0;
        string str1=str.substr(0,idx1);
        string str2=str.substr(idx1+1,idx2-idx1-1);
        string str3=str.substr(idx2+1,str.size()-idx2);
        if(str1.size()<1||str1.size()>16) {
            puts("NO");
            goto hell;
        }
        if(str2.size()<1||str2.size()>16) {
            puts("NO");
            goto hell;
        }
        if(str3.size()<1||str3.size()>10) {
            puts("NO");
            goto hell;
        }
        for(int i=0;i<str1.size();i++) {
            if(str1[i]>='a'&&str1[i]<='z') {
            } else if(str1[i]>='A'&&str1[i]<='Z') {
            } else if(str1[i]>='0'&&str1[i]<='9') {
            } else if(str1[i]=='_') {
            } else {
                puts("NO");
                goto hell;
            }
        }
        for(int i=0;i<str2.size();i++) {
            if(str2[i]>='a'&&str2[i]<='z') {
                cnt++;
            } else if(str2[i]>='A'&&str2[i]<='Z') {
                cnt++;
            } else if(str2[i]>='0'&&str2[i]<='9') {
            } else if(str2[i]=='-') {
            } else {
                puts("NO");
                goto hell;
            }
        }
        if(cnt==0) {
            puts("NO");
            goto hell;
        }
        for(int i=0;i<str3.size();i++) {
            if(str3[i]>='a'&&str3[i]<='z') {
            } else {
                puts("NO");
                goto hell;
            }
        }
        puts("YES");
        hell:;
    }
    return 0;
}
