n = int(raw_input())

for i in xrange(n):
	n1, n2 = raw_input().split()
	l = min(len(n1), len(n2))
	
	carries = 0
	
	for j in xrange(1, l+1):
		if int(n1[-j]) + int(n2[-j]) > 9:
			carries += 1
	if carries == 1:
		print str(carries) + ' ' + 'carry operation'
	elif carries > 1:
		print str(carries) + ' ' + 'carry operations'
	else:
		print 'No carry operation'
