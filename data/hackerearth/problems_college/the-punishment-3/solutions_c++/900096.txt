#include <iostream>
#include <cstring>
using namespace std;


int main()
{
    int t;
    cin >> t;
    while(t>0){
    	int n,k;
        cin >> n >> k;
        int l = min(k,n-k);
        int h = max(k,n-k);
        int a[l];
        for(int i=0;i<l-1;i++){
        	a[i]=9;
        }
        a[l-1]=8;
        int b[h];
        for(int j=0;j<h;j++){
        	b[j]=9;
        }
        for(int i=h-1,j=l-1;j>=0;i--,j--){
        	b[i]=b[i]-a[j];
        }
        for(int i=0;i<l;i++){
        	cout << a[i] ;
        }
        for(int i=0;i<h;i++){
        	cout << b[i] ;
        }
        cout << endl;
    	t--;
    }
}
