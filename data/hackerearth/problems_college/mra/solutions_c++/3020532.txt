#include<bits/stdc++.h>
using namespace std;
int col[505][505],row[505][505];
int n,m;
char str[504][504];
int main()
{
    memset(col,0,sizeof(col));
    memset(row,0,sizeof(row));
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;i++)
	scanf("%s",str[i]);


    for(int i=0;i<m;i++)
    {
	for(int j=0;j<n;j++)
	{
	    if(j!=0)
	    col[i][j]=col[i][j-1];
	    if(str[j][i]=='x')
		col[i][j]++;
	}

    }

    for(int i=0;i<n;i++)
	for(int j=0;j<m;j++)
	{
	    if(j!=0)
		row[i][j]=row[i][j-1];
		if(str[i][j]=='x')
		    row[i][j]++;
	}

    int ans=0;
    for(int i=0;i<n;i++)
	for(int j=i+1;j<n;j++)
	{
	    int f=-1,s=-1;
	    for(int k=0;k<m;k++)
	    {
		if(str[i][k]=='x' || str[j][k]=='x')
		{
		    f=s=-1;
		    continue;

		}
		if(i==0)
		{
		    if(col[k][j]==0)
		    {
			if(f==-1)
			f=k;
			else
			{
			    s=k;
			    ans = max(ans,2*(j-i+s-f));
			}

		    }
		}
		else
		if(col[k][j]-col[k][i-1] == 0)
		{
			if(f==-1)
			f=k;
			else
			{
			    s=k;
			    ans = max(ans,2*(j-i+s-f));
			}
		}
	    }

	}
    if(ans==0)
	printf("impossible\n");
    else
	printf("%d\n",ans);



}
