vector<vector<ll> > arr(201,vector<ll>(201));
int main()
{
    ios_base::sync_with_stdio(false);
   ll test,i,j,a,b,maxx,minn,num,m;
    cin>>test;
    while(test--)
    {
        cin>>num>>m;
        char ch,ini;
        for(ll i=0;i<num;i++)
        {
            for(ll j=0;j<m;j++)
            {
                cin>>ch;
                if(i==0&&j==0)
                	ini=ch;
                if(ch=='T')
                    arr[i][j]=150;
                else if(ch=='D')
                    arr[i][j]=-100;
                else
                    arr[i][j]=0;
            }
        }
        
        for(ll i=1;i<num;i++)
        {
            arr[i][0]+=arr[i-1][0];
            if(arr[i][0]<0)
                arr[i][0]=-2000000000;
        }
        for(ll i=1;i<m;i++)
        {
            arr[0][i]+=arr[0][i-1];
            if(arr[0][i]<0)
                arr[0][i]=-2000000000;
        }
        for(ll i=1;i<num;i++)
        {
            for(ll j=1;j<m;j++)
            {
                arr[i][j]+=max(arr[i][j-1],arr[i-1][j]);
                if(arr[i][j]<0)
                    arr[i][j]=-2000000000;
            }
        }
       ll ans;
       if(arr[num-1][m-1]<0||ini=='D')
        ans=-1;
       else
        ans=arr[num-1][m-1];
        printf("%lld\n",ans);
    }

}
