#include <bits/stdc++.h>
using namespace std;
string x="aaceehhkrrt";

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	
	string s;
    cin >> s;
    int n=s.size();
    int q,query[555555];
    int le=x.size();
    
    for(int i=0;i<n;i++)
    	s[i]=tolower(s[i]);
    
    memset(query,0,sizeof(query));
    
    for(int i=0;i+x.size()-1<n;i++)
    {
    	int c1=0,c2=0,c3=0,c4=0,c5=0,c6=0,c7=0,fl=0;
    	
    	for(int j=i;j<=i+le-1;++j)
    	{
    		if(s[j] == 'a')	c1++;
    		if(s[j] == 'c')	c2++;
    		if(s[j] == 'e')	c3++;
			if(s[j] == 'h')	c4++;
			if(s[j] == 'k')	c5++;
			if(s[j] == 'r')	c6++;
			if(s[j] == 't')	c7++;
    	}
		if(c1 == 2 and c2 == 1 and c3 == 2 and c4 == 2 and c5 == 1 and c6 == 2 and c7 == 1)	fl=1;
    	if(fl)
    		query[i+le-1] = query[i+le-2]+1;
    	else	query[i+le-1] = query[i+le-2];
    }
    
    cin >> q;
    while(q--)
    {
    	
    	int l,r;
    	cin >> l >> r;
    	if(r<l+le-2)
    		cout << 0 << "\n";
    	else	cout << query[r-1] - query[l+le-3] << "\n";
    }
    return 0;
}
