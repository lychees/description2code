
#include <iostream>


using namespace std;


int main(){
    int t,i,n;
    cin >> t;
    while(t--)
        {
        cin>>n;
        int div=n%3;
        if(div==0)
            {
            for(i=0;i<n;i++)
            cout<<"5";
            cout<<endl;
        }
        else if(div==1&&n>=10)
            {
            for(i=0;i<n-10;i++)
            cout<<"5";
            for(i=0;i<10;i++)
                cout<<"3";
                cout<<endl;
        }
        
        else if(div==2&&n>=5)
            {
            for(i=0;i<n-5;i++)
                cout<<"5";
            for(i=0;i<5;i++)
                cout<<"3";
            cout<<endl;
        }
        
        else
            cout<<"-1"<<endl;
    }
    return 0;
}
