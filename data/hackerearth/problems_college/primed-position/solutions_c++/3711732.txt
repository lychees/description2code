#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
#define sz 1500010
using namespace std;
bool p[sz];
ll int a[sz],b[sz];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	memset(p,0,sizeof(p));
	p[0]=1;p[1]=1;
	ll int i,j,k;
	for(i=2;i<sz;i++)
	{
		if(p[i])
		continue;
		for(j=2;j*i<sz;j++)
		{
			p[j*i]=1;
		}
	}
	j=0;
	for(i=0;i<sz;i++)
	{
		if(!p[i])
		{
			a[j++]=i;
		}
	}
	//~ cout<<j<<endl;
	k=0;
	for(i=0;i<j;i++)
	{
		if(!p[(i+1)])//prime
		{
			b[k++]=a[i];
		}
	}
	//~ cout<<k<<endl;
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n,m;
		cin>>n>>m;
		n--;m--;
		ll int ans=(b[n]*b[m])%mod;
		cout<<ans<<'\n';
	}
	return 0;
}
