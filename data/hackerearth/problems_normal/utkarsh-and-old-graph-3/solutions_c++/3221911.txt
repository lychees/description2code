#include <bits/stdc++.h>
using namespace std;
int parent[100005] ;
long long sz[100005] ;
bool vis[100005] ;
int find(int x){
    if(x == parent[x])return x ;
    return parent[x] = find(parent[x]) ;
}
void un(int x,int y){
    x = find(x) ;
    y = find(y) ;
    if(x == y)return ;
    parent[x] = y ;
    sz[y] += sz[x] ;
    sz[x] = 0 ;
}
int main()
{
    int n , m ;scanf("%d%d",&n,&m) ;
    for(int i = 1 ; i <= n ; i++){
        sz[i] = 1 ;
        parent[i] = i ;
    }
    for(int i = 0 ; i < m ; i++){
        int x , y ;
        scanf("%d%d",&x,&y) ;
        un(x,y) ;
    }
    long long ans = 0 , sum = 0 ;
    int c = 0 ;
    for(int i = 1 ; i <= n ; i++){
        int x = find(i) ;
        if(sz[x] == 0 || vis[x])continue ;
        vis[x] = true ;
        ans += sz[x]*(sz[x]-1)/2 - sz[x]+1 ;
        ans += (sz[x])*sum ;
        sum += sz[x] ;
        c ++ ;
    }
    ans -= c-1 ;
    cout << ans ;
    return 0;
}
