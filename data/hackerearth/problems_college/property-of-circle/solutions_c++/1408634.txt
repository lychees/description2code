int main()
{
ios_base::sync_with_stdio(false);
cin.tie(0);
cout.tie(0);
int t;
cin>>t;
while(t--)
{
	double x1,y1,r1,x2,y2,r2;
	cin>>x1>>y1>>r1;
	cin>>x2>>y2>>r2;
		if(x1==x2&&y1==y2)
	{
		if(r1<r2)
		cout<<"C2CC1\n";
		else
		if(r1>r2)
		cout<<"C1CC2\n";
		else
		cout<<"C2~C1\n";
		continue;
	}
	double d=(x2-x1)*(x2-x1)+(y2-y1)*(y2-y1);
	d=sqrt(d);
	if(r1>r2)
	{
		if(d+r2<=r1)
		cout<<"C1CC2\n";
		else
		cout<<"NOT SATISFIED\n";
		continue;
	}
	if(r2>r1)
	{
		if(d+r1<=r2)
		cout<<"C2CC1\n";
		else
		cout<<"NOT SATISFIED\n";
		continue;
	}
cout<<"NOT SATISFIED\n";
}
return 0;
}
