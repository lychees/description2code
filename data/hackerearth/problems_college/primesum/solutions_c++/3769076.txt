bool prime[N];

ll ans[N];

void seive() {
 prime[0] = prime[1] = true;
 for (int i = 4; i <= N; i += 2) {
 	prime[i] = true;
 }
 for (int i = 3; i * i <= N; i += 2) {
 	if (!prime[i]) {
 		for (int j = i*i; j <= N; j+= i) {
 			prime[j] = true;
		 }
	 }
 }
 for (int i = 2; i <= N ;i++) {
 	if (!prime[i]) {
 		pr.pb(i);
	 }
 }
}
