#include <bits/stdc++.h>
using namespace std;
int p[100005];
int fset(int x){
	if(p[x]==x) return x;
	return p[x]=fset(p[x]);
}
int main(){
	int n,q;
	cin>>n>>q;
	for(int i=1;i<=100000;++i)
		p[i]=i;
	while(q--){
		int t;
		cin>>t;
		if(t==1){
			int a,b;
			cin>>a>>b;
			p[fset(a)]=fset(b);
		}
		else if(t==2){
			int tt;
			cin>>tt;
			p[fset(tt)]=tt;
			p[tt]=tt;
		}
		else {
			int tt;
			cin>>tt;
			cout<<fset(tt)<<endl;
		}
	}
}
