#include <bits/stdc++.h>
#define md 1000000007
#define ll long long
using namespace std;

ll mpow(ll a, ll b)
{
    ll ans = 1;
    while(b)
    {
      if (b % 2 == 1) ans=(ans*a)%md;
      b=b>>1;
      a=(a*a)%md;
    }
    return ans;
}

ll func(ll n) 
{
  ll p,q;
  if (n == 0) return 0;
  if (n == 1) return 2;
  if (n == 2) return 7;
  p=mpow(2,n-1);
  q=mpow((-1),n-1);
  ll b = ((3%md*p%md)%md-q%md)%md;
  return b;
}

int main()
{
  ll t,n,ans;
  cin>>t;
  while(t--)
  {
      cin>>n;
      ans=func(n);
      ans%=md ;
      cout<<ans<<endl;
  }
  return 0;
}
