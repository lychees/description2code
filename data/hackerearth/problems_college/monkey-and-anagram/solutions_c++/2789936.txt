#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	string r, s;
    	cin >> r;
    	cin >> s;
    	int n = r.size();
    	int m = s.size();
    	int a[26];
    	memset(a, 0, sizeof a);
    	int b[26];
    	memset(b, 0, sizeof b);
    	int i, j;
    	for(i=0;i<n;i++)
    	{
    		a[r[i]-97]+=1;
    	}
    	for(i=0;i<m;i++)
    	{
    		b[s[i]-97]+=1;
    	}
    	int flag = 0;
    	for(i=0;i<26;i++)
    	{
    		//if((a[i]>0 && b[i]==0))
    		if(a[i]>b[i])
    		{
    			flag =1;
    			break;
    		}
    	}
    	if(!flag)
    	{
    		int count=0;
    		for(i=0;i<26;i++)
    		{
    			if(b[i]!=a[i])
    			{
    			count++;
    			}
    		}
    		cout << count << endl;
    	}
    	else
    	{
    		cout << -1 << endl;
    	}
    }
    return 0;
}
