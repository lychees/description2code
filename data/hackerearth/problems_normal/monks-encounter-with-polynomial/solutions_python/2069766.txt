t= int(raw_input())
while t>0:
    a, b, c, k = map(int, raw_input().split())
    lower, upper = 0, 100000
    ans = -1
    while lower<=upper:
        mid = (upper+lower)/2
        val = a*(mid*mid) + b*mid + c
        if val>=k:
            ans = mid
            upper = mid - 1
        else:
            lower = mid + 1
    print ans
    t-=1
