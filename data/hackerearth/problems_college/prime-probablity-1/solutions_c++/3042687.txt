#include <bits/stdc++.h>
using namespace std;
#define maxi 1000000
int main()
{
    int t;cin>>t;
    int arr[1000001]={0},count=0;
    arr[0]=0;arr[1]=0;
   	for(int i=2;i<=maxi;i++){
    	if(arr[i]==0) count++;
    	arr[i]=count;
    	if(!arr[i]) continue;
    	int j=i;
    	while(j<=maxi) {
    		arr[j]=1;
    		j+=i;
    	}
    	arr[i]=count;
    }
    while(t--){
    	int l,r;cin>>l>>r;
    	float prime;
    	float ans;
    	if(l==r){
    		if(arr[l]!=arr[l-1]) prime = arr[r]-arr[l]+1;
    		else prime = arr[r]-arr[l];
    	}
    	else{
    		if(arr[l]!=arr[l-1]) prime = arr[r]-arr[l]+1;
    		else prime = arr[r]-arr[l];
    	}
    	//cout<<prime<<endl;
        ans=prime/(r-l+1);
    	printf("%.6f\n",ans);
    }
    return 0;
}
