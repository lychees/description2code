#include<bits/stdc++.h>
using namespace std;
#define M 1000000007
long long int mypow(int a,int b)
{
	if(b==0)
	return 1;
	long long int temp=mypow(a,b/2);
	if(b%2==0)
	return (temp*temp)%M;
	else
	return ((a*temp)%M*temp)%M;
}
int main()
{
	ios::sync_with_stdio(false);
	set<long long int> s;
	long long int t,n,temp;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(int i=0;i<n;i++){
		cin>>temp;
		s.insert(temp);
		}
		cout<<mypow(n,s.size())<<endl;
		s.clear();
	}
	return 0;
}
