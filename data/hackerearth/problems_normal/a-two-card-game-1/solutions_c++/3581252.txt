
#include <iostream>
#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int main(int argc, char *argv[])
{
    long long N, num_combs=0, num_win_and_lost;
    cin >> N;
    num_combs = N * (N-1);
    num_win_and_lost = num_combs;
    vector <long long> cards(N, 0);
    vector <long long> repeats( pow ( 10. , 6) + 1 , 0);
    for ( long long i = 0 ; i < N; i++ )
    {
        cin >> cards[i];
        repeats[cards[i]]++;
    }
    for ( long long i = 1 ; i <= N; i++)
    {
        num_win_and_lost -= repeats[i] * (repeats[i] - 1);
    }
   
  
    printf("%.06f %.06f",  double(num_win_and_lost/2)/(num_combs),double(num_win_and_lost/2)/(num_combs));
    return 0;
}
