#include <iostream>
#include<bits/stdc++.h>
#define ll long long
using namespace std;



int main()
{
	
   ll t,n,k,p;
  
   cin>>t;
   while(t--)
   {
   
   	cin>>n>>k;
   	ll a[n],v[n];
   	for(ll i=0;i<n;i++)
   	{
   		 cin>>a[i];
   	}
    v[0]=max((ll)0,a[0]);
    
    for(ll i=1;i<=k;i++)
    {
    	 v[i]=max(v[i-1],a[i]);
    }
       for(ll i=k+1;i<n;i++)
       {
       	v[i]=max(v[i-k-1]+a[i],v[i-1]);
       }
       cout<<v[n-1]<<endl;
   }
  
    return 0;
}
