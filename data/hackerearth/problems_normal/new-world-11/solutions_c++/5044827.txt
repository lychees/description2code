    #include<bits/stdc++.h>
    using namespace std;
     
    #define forn(i, n) for(i = 0; i < n; i ++)
    #define for1(i, n) for(int i = 1;i <= n;i ++)
    #define FOR(i, a, b) for(int i = a; i<=b;i++)
    const int N = 1e5 + 10;
    int n, k;
    int local[N];
     
    bool check(int dis)
    {
        int pos = 1;
        int jump = 0;
        for(int i = 2;i <=n;i ++)
        {
            if(local[i] - local[pos] <= dis)
            {
                if(i == n)
                {
                    jump ++;
                    break;
                }
                else if(local[i + 1] - local[pos] > dis)
                {
                    pos = i;
                    jump ++;
                }
            }
            else return false;
        }
        return jump <= k;
    }
     
    int main()
    {
       // freopen("new world.inp","r",stdin);
        ios_base:: sync_with_stdio(false);
        int test;
        cin >> test;
        while(test --)
        {
            cin >> n >> k;
            for1(i, n) cin >> local[i];
            sort(local + 1, local + n + 1);
            int l = 0, r = local[n] - local[1] + 1;///x : ans , val < x : false, val > x : true
            while(r - l > 1)
            {
                int mid = (l + r)/2;
                if(check(mid)) r = mid;
                else l = mid;
            }
            cout<< r  << endl;
        }
    }
