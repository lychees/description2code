#define ll long long
ll power(ll base ,ll exp,ll mod)
{
	ll res=1;
	while(exp>0)
	{
		if(exp&1)
		{
			res=(res*base)%mod;
		}
		exp=exp/2;
		base=(base*base)%mod;
	}
	return res;
}

ll scan()
{
    ll n = 0, ch = getchar_unlocked();
    while (ch < '0' || ch > '9')
        ch = getchar_unlocked();
    while (ch >= '0' && ch <= '9') {
        n = (n << 3) + (n << 1) + ch - '0';
        ch = getchar_unlocked();
    }
    return n;
}
