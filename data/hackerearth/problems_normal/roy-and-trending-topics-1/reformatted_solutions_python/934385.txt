n = long(raw_input())
final = []
ids = []
idChangeMap = {}
idNewMap = {}
for i in range(n):
    ns = map(long, raw_input().split(' '))
    ids.append(ns[0])
    idNewMap[ids[i]] = ns[2]*50+ns[3]*5+ns[4]*10+ns[5]*20
    idChangeMap[ids[i]] = idNewMap[ids[i]] - ns[1]
result = sorted(idChangeMap, reverse=True)
result = sorted(result, key=idChangeMap.get, reverse=True)
s = 0
for i in result:
    if (s < 5):
        print str(i)+" "+str(idNewMap[i])
    s += 1
