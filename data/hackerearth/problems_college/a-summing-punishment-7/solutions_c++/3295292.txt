#include<iostream>
using namespace std;
int main()
{
   long int n;
    cin>>n;
   long int a[n];
    for(long int i=1;i<=n;i++)
    {
        cin>>a[i];
    }
    long int q,l,r,temp;
    cin>>q;
    while(q--)
    {   temp=0;
        cin>>l>>r;
        for(long int i=l;i<=r;i++)
        {
            temp=temp+a[i];
        }
        cout<<temp<<endl;
    }
}
