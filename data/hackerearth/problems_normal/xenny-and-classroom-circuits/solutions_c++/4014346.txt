#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        string str;
        cin>>str;
        int sumpair1=0,sumpair2=0;
        for(int i=0;i<2*n;i++)
        {
            if(i&1)
            {
                if(str[i]=='G')
                    sumpair1++;
                else
                    sumpair2++;

            }
            else
            {
                if(str[i]=='B')
                sumpair1++;
               else
                sumpair2++;

            }
        }
        int ans=min(sumpair1,sumpair2);
        ans/=2;
        cout<<ans<<endl;;


    }






    return 0;
}
