const int N = 2600 + 7;
const int M = 1e9;
const int inf = 1e9 + 7;

const double ep = 1e-9;


int C[N][N];

void init(){
    rep(i, 0, N){
        C[i][0] = C[i][i] = 1;
        rep(j, 1, i)
            C[i][j] = (C[i-1][j-1] + C[i-1][j]) % inf;
    }
}

int main(){
    //freopen("in.txt", "r", stdin);
    int test, n, m;
    init();
    cin >> test;
    while (test-- > 0){
        cin >> n >> m;
        if (m % 2 == 1){
            cout<<0<<endl;
            continue;
        }
        ll res = 0;
        int t = m / 2;
        rep(x, 0, t + 1){
            if (x <= n * 6 && t - x <= n * 6){
                ll tmp = ((ll)C[n*6][x] * C[n*6][t-x]) % inf;
                ll tmp1 = ((ll)C[n*18][t-x] * C[n*18][x]) % inf;
                res = (res + tmp1 * tmp) % inf;
            }
        }
        cout<<res<<endl;
    }
}
