#include <iostream>
#include <new>
using namespace std;

int main()
{
	int n,k,i,j;
	cin >> n;
	int *a= new int[n]; 
	for(i=0;i<n;i++)
	cin >> a[i];
	for(i=0;i<n;i++)
	{
	    k=0;
	    for(j=0;j<n;j++)
	    {
	        if((a[i]%a[j]!=0)&&(i!=j))
	        {
		        k++;
	        }
	    }
	    if(k==n-1)
	    cout << a[i] << " ";
	}
    return 0;
}
