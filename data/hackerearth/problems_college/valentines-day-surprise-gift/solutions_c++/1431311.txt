#include <bits/stdc++.h>
#define ll long long
#define ld long double
using namespace std;
ll gcd(ll a,ll b){ return (b==0)?a:gcd(b,a%b); }
ll modpow(ll a, ll n, ll mod){ ll res=1; while(n){ if(n&1)res=(res*a)%mod; a=(a*a)%mod; n>>=1; } return res; }
ll lmax(ll a,ll b){ return (a>b)?a:b; }
ll mod=1e9 + 7;
map<ll,ll> m1,m2;
ll func1(ll n){
    if(n<=2) return 1;
    if(n<=4) return 3;
    if(n<=6) return 15;
    if(n<=10) return 105;
    if(m1[n]!=0) return m1[n];
    
    ll ret = modpow(2,ceil((ld)n/4),mod);
    ret = (ret * func1(ceil((ld)n/5)))%mod;
    ret = (ret * func1(ceil((ld)n/10)))%mod;
    
    return ret;
}

ll func2(ll n){
    if(n<=2) return 1;
    if(n<=4) return 3;
    if(n<=6) return 15;
    if(n<=10) return 105;
    if(m2[n]!=0) return m2[n];
    
    ll ret = modpow(2,ceil((ld)n/4),mod-1);
    ret = (ret * func1(ceil((ld)n/5)))%(mod-1);
    ret = (ret * func1(ceil((ld)n/10)))%(mod-1);
    
    return ret;
}

int main() {
    ll t,n,x,y,ans,x1,y1;
    scanf("%Ld",&t);
    while(t--){
        scanf("%Ld",&n);
        x=func1(n);
        y=func2(n);
        ans=0;
        for(int i=1;i<=5;i++){
            for(int j=1;j<=5;j++){
                if(gcd(i,j)==1){
                    x1=(x*i)%mod;
                    y1=(y*j)%(mod-1);
                    ans=lmax(ans,modpow(x1,y1,mod));
                }
            }
        }
        printf("%Ld\n",ans);
    }
    return 0;
}
