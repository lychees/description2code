#include<bits/stdc++.h>
using namespace std;
int a[1000000] ={0};

int main()
{
	int test=0;
	scanf("%d" ,&test);
	while(test--)
	{
		int n=0, max=0, maxs=0;
		scanf("%d" , &n);
		
		for(int i=0;i<n;i++)
		{
			scanf("%d", &a[i] );
			maxs+=a[i];
			if(maxs<0 )
			{
				maxs =0;
			}
			if(max<maxs )
			{
				max=maxs;
			}
			
		}
		printf("%d \n",max );
	}
}
