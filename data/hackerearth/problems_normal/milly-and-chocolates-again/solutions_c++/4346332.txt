#include <iostream>
using namespace std;

int main()
{
   int N,Q,i,pre=1,next,j,q,arrayA[1000001]={0};
   cin>>N;
   for(i=0;i<N;i++)
   	{
   		cin>>next;
   		for(j=1;j<=next;j++)
   			{
   				arrayA[pre]=i+1;
   				pre++;
   			}	
   	}	
   cin>>Q;
   while(Q--)
   	{
   		cin>>q;
   		cout<<arrayA[q]<<"\n";
   		
   	}
    return 0;
}
