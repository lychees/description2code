#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int cnt=0;
    
    int pp=sqrt(n);
    
    if(pp*pp==n)
    {
    cnt=1;
    cout<<cnt<<endl;
    return 0;
   }
   int dp[100003]={0};
   int f=0;

   for(int j=1;j<=n;j++)
   {dp[j]=INT_MAX;
    for(int i=1;i*i<=j;i++)
    {
            int k=j-i*i;
            int p=(int)sqrt(k);
          if(k>=0)
            dp[j]=min(dp[j],1+dp[k]);
    }
 
   }
   
    cout<<dp[n]<<endl;
    return 0;
}
