#include <iostream>
using namespace std;
long long a[1000000];
void swap(int x, int y)
{
	long long temp;
	temp = a[x];
	a[x] = a[y];
	a[y] = temp;
}

int main()
{
    long long n,x,y,z;
    cin >> n;
    for(int i=0;i<n;i++){
    	cin >> a[i];
    }
    for(int i=0;i<n;i++){
    	if(i<2)
    		cout << "-1" << endl;
    	else if(i==2){
    		if(a[0] < a[1])
    			swap(0,1);
    		if(a[0] < a[2])
    			swap(0,2);
    		if(a[1] < a[2])
    			swap(1,2);
    		x = a[0];y=a[1];z=a[2];
    		cout << x*y*z << endl;
    		//cout << a[0] << " " << a[1] << " " << a[2];
    	}else{
    		if( a[i]>x && a[i]>y && a[i]>z ){
    			z = y;
    			y = x;
    			x = a[i];
    		}else if( a[i]>y && a[i]>z ){
    			z = y;
    			y = a[i];
    		}else if( a[i]>z ){
    			z = a[i];
    		}
    		cout << x*y*z << endl;
    	}
    }
    return 0;
}
