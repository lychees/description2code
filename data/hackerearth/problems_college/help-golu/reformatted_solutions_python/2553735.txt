def isprime(x):

    if (x == 2):
        return 1
    for i in range(2, x):
        if (x % i == 0):
            return 0
    return 1


t = input()
for i in range(t):
    n = input()
    if (isprime(n)):
        print 1
        continue
    if (n % 2 == 0 and n > 2):
        print 2
    else:
        if (isprime(n-2)):
            print 2
        else:
            print 3
