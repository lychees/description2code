#include <iostream>
#include <map>
using namespace std;

int main()
{
	int i;
    string s;
    cin>>s;
    map<char,int> m;
    for(i=0;i<s.size();i++){
    	m[s[i]]++;
    }
    
    map<char,int> :: iterator it;
    for(it=m.begin();it!=m.end();it++){
    	for(i=1;i<=it->second;i++){
    		cout<<it->first;
    	}
    }
    return 0;
}
