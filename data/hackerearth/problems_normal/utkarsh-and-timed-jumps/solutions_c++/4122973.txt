#include<iostream>
#define ull unsigned long long int

inline ull FindNum(ull Current_Num)
{
	return 	(Current_Num+(Current_Num*Current_Num))/2;
}

void Binary_Search(ull lo,ull hi,ull key)
{
	ull mid=(lo+hi)/2;
	if ((FindNum(mid)>=key) & (FindNum(mid-1)<key))
		std::cout<<mid<<'\n';
	else if(FindNum(mid)>key)
		Binary_Search(lo,mid,key);
	else if(FindNum(mid)<key)
		Binary_Search(mid,hi,key);
}
int main()
{
	int t;
	std::cin>>t;
	while(t--)
	{
		ull lower_Bound=1, Higher_Bound=4294967296, x;
		std::cin>>x;
		Binary_Search(lower_Bound,Higher_Bound,x);
	}
}
