#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    map<int,int> dp;
    dp[1]=0;
    dp[2]=1;
    dp[3]=1;
    dp[4]=2;
    dp[5]=1;
    dp[6]=2;
    dp[7]=1;
    dp[8]=2;
    while(t--)
    {
    	int n;
    	cin >> n;
    	if(dp[n])
    	{
    		cout << dp[n] << endl;
    		continue;
    	}
    	int temp = n/7;
    	int rem = n%7;
    	while(rem==1)
    	{
    		temp--;
    		rem+=7;
    	}
    	dp[n]=temp+dp[rem];
    	cout << dp[n] << endl;
    }
    return 0;
}
