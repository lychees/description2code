ll l_then(int a[], int l, int h, int k)
{
    int m;
    while((h-l) > 1) {
        m = (l+h)/2;

        if(a[m] <= k)
            l = m;
        else
            h = m-1;
    }
    if(a[h] <= k)
        return h;
    else if(a[l] <= k)
        return l;
    return 0;
}

ll g_then(int a[], int l, int h, int k, int z)
{
    int m;
    while((h-l) > 1) {
        m = (l+h)/2;

        if(a[m] >= k)
            h = m;
        else
            l = m+1;
    }
    if(a[l] >= k)
        return l;
    if(a[h] >= k)
        return h;
    return z;
}
