#include <iostream>
using namespace std;

long long int compute(long long int n) 
{
	if(n==0) 
	return 0;
	long long int k=(n+1)/2;
	return k*k+compute(n/2);
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n, s;
    	cin>>n;
    	s=compute(n);
       	cout<<s<<endl;
    }
    return 0;
}
