long int mod(long int ct)
{
	if(ct<0)
	return -ct;
	else return ct;
}

main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		char a[100000];
		cin>>a;
		
		long int l=strlen(a);
		long int i=0,ct=0;
		
		//if(l%2!=0)
		//{
		//cout<<l;
			for(i=0;i<(l/2);i++)
			{
				if((a[i]-a[l-i-1])!=0)
				{
					if(a[i]>a[l-i-1])
					ct=ct+a[i]-a[l-i-1];
					else
					ct=ct+a[l-i-1]-a[i];
				}
			}
		//}
		
		cout<<mod(ct)<<endl;
	}
}
