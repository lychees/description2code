#include <bits/stdc++.h>
using namespace std;
#define MX 10000

bool visit[MX];

int main() {
    int T, N, M, a, b;
    cin >> T;
    while (T--) {
        vector<pair<int, int> > edge;
        cin >> N >> M;
        for (int i = 0; i < M; i++) {
            cin >> a >> b;
            edge.push_back(make_pair(a, b));
        }

        memset(visit, false, sizeof(visit));
        for (int i = 0; i < N - 1; i++) {
            cin >> a;
            pair<int, int> p = edge[a - 1];
            visit[p.first] = visit[p.second] = true;
        }

        bool f = true;
        for (int i = 1; i <= N; i++) {
            if (!visit[i]) {
                f = false;
                break;
            }
        }
        if (!f)
            cout << "NO" << endl;
        else
            cout << "YES" << endl;
    }
}
