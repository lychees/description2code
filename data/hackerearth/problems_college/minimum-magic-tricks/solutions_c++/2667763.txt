#include <iostream>
#include <algorithm>
using namespace std;
const int maxn = 1010;
const int mod = 1000000007;
long long dp[maxn];

void makeDPtable()
{
    dp[1] = 1;
    for (int i = 2; i < maxn; i++) {
         int le =  i - 2;
         dp[i] = 1;
         while (le >= 1) {
                dp[i] += (2 * dp[le]);
                if (dp[i] >= mod)
                    dp[i] %= mod;
                le--;
         }
         //cout << i << " " << dp[i] << endl;
    }
}

int main() {
	makeDPtable();
	int cases;
	cin >> cases;
	while (cases--) {
	       int H;
	       cin >> H;
	       cout << dp[H] % mod << endl;
	}
	return 0;
}
