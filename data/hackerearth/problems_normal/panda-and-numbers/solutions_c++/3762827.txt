#include <bits/stdc++.h>
using namespace std;
int memo[1000100];
set<int>nums;
int DP(int n)
{
	if(n<=0)
	return 0;
	if(memo[n]!=-1)
	return memo[n];
	if(nums.find(n)!=nums.end())
	return 1;
	int dig,temp,n1;
	temp=0;
	n1=n;
	while(n)
	{
		dig=n%10;
		if(dig)
		{
			temp^=DP(n1-(dig*dig));
			if(temp)
			break;
		}
		n/=10;
	}
	return memo[n]=temp;
}
int main()
{
    ios_base::sync_with_stdio(0);
   	int t,n,i;
    for(i=0;i<=1000000;i++)
      memo[i]=-1;
    for(i=1;i<=8;i++)
    nums.insert(pow(i,i));
    cin>>t;
    while(t--)
    {
      cin>>n;
      if(DP(n))
      cout<<"Yes\n";
      else
      cout<<"No\n";
    }
    return 0;
}
