using namespace std;

typedef long long int llt;

vector< vector<int> > graph;
int n;
bool flag;
vector<bool> visited,cur;

bool traverse(int i)
{
    if(!visited[i])
    {
        visited[i]=true;
        cur[i]=true;
        for(int j = 0 ; j < n ; j++)
        {
            if(j==i)
                continue;
            if(graph[i][j]==1 && !visited[j] && traverse(j))
                return true;
            else if(graph[i][j]==1&& cur[j])
                return true;
        }
    }
    cur[i]=false;
    return false;
}

int main()
{
    ios::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
        cin>>n;
        visited.assign(n,false);
        cur.assign(n,false);
        flag = false;
        graph.assign(n,vector<int>(n));
        for(int i =0 ; i < n ; i++)
        {
            for(int j = 0 ; j < n ; j++)
            {
                cin>>graph[i][j];
            }
        }

        for(int i = 0 ; i < n && !flag ; i++)
        {
            flag = traverse(i);
          //  cout<<"i : "<<flag<<"\n";
        }
        if(flag)
            cout<<"Aye\n";
        else
            cout<<"Nae\n";
        visited.clear();
        cur.clear();
        graph.clear();
    }

    return 0;
}
