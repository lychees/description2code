template <class T1, class T2, class Pred = std::less<T2> >
struct sort_pair_second {
    bool operator()(const std::pair<T1,T2>&left, const std::pair<T1,T2>&right) {
        Pred p;
        return p(left.second, right.second);
    }
};
using namespace std;

int divisors(int n)
{
	int ans=1,i,k;
	for (i = 2;i<n/i ; i++)
		if(n%i==0)
		{
			if(i%2==0)
				ans++;
			k=n/i;
			if(k%2==0)
				ans++;
			//cout<<i<<" "<<k<<" "<<ans<<endl;
		}
  	if (i*i==n and i%2==0)
  		ans++;
  	return ans;
}

int main() 
{
	std::ios_base::sync_with_stdio(false);cin.tie(0);cout.tie(0);
	int t,n;
	ll ans;
	cin>>t;
	while(t--)
	{
		cin>>n;
		if(n&1)
		{
			cout<<"0"<<endl;
			continue;
		}
		ans=divisors(n);
  		cout<<ans<<endl;
	}
	return 0;
}
