#include <cstdlib>
#include <iostream>
#include <cmath>
#include <algorithm>

using namespace std;
long i,n,j,k,s;
int main(int argc, char *argv[])
{   cin>>n;
    int a[n];
    for(i=0;i<n;i++)
       cin>>a[i];
    sort(a,a+n);
        
    i=0;
   
  do
  { 
    while(a[i]==0)
       i++;
   
    if(i<n)
       cout<<n-i<<endl;
    k=a[i];
    
    for(j=i;j<n;j++)
    {   a[j]=abs(a[j]-k);
        
    }
    
  }while(i<n);
    
   
    return EXIT_SUCCESS;
}
