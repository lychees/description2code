#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int ar[10];
        for(int i=0;i<10;i++)
            ar[i]=0;
        int result=0;
        int n;
        cin>>n;
        for(int i=0;i<n;i++)
        {
            int a;
            cin>>a;
            a=a%10;
            if(ar[a]!=0)
                result++;
            else
            ar[a]=1;
        }
        cout<<result<<endl;

    }
    return 0;
}
