#include<bits/stdc++.h>
#define MAX 2000002
using namespace std;

int a[MAX]= {0};
int main()
{
    int n,k,x;
    cin>>n>>k;

    for(int i=0; i<n; i++)
    {
        cin>>x;
        if(k-x>=0)
            if(a[k-x])
            {
                printf("YES\n");
                return 0;
            }
        a[x]++;
    }

    printf("NO\n");
}
