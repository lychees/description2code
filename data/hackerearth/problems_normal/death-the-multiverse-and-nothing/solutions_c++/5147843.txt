#include <iostream>
using namespace std;

int main()
{
	int a[10] = {0};
	int count = 0;
	int temp = 0;
	while (count<=10) {
		cin >> temp;
		if (!cin) {
			break;
		}
		if (count==0) {
			count++;
			continue;
		}
		a[count-1] = temp;
		count++;
	}
	for (int i=0; i<count-1; i++) {
		cout << a[i] - 1 << endl;
	}
    return 0;
}
