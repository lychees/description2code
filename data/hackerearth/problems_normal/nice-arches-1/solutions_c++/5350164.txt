#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;

    int count=0;
    while(n--)
    {
      string a;
      cin>>a;
      int len=a.length();
      for(int i=0;i<len;i++)
      {
      	if(a[i]!='0')
      	{
     	  char t = a[i];
     	  //cout<<a[i]<<endl;
    	  for(int j=i+1;j<len ;j++)
    	  {
    		//cout<<a[i]<<' '<<a[j]<<endl;
    		if(t==a[j] && (j-i)%2)
    		{
    			a[i]='0';
    			a[j]='0';
    			break;
    		}
    	  }
      	}
       }
    int i=0;
    //cout<<a<<endl;
    for(;i<len;i++)
    {
    	if(a[i]!='0')
    	break;
    }
    if(i==len)
    count++;
    }
    cout<<count;
    return 0;
}
