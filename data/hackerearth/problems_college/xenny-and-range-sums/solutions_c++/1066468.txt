#include <iostream>
using namespace std;
long int a[1000001];
long int dp[1000001];
long int ndp[1000001];
int main()
{
  	int n,q;
  	cin>>n>>q;
  	dp[0]=0;
  	for(int i=1;i<=n;i++)
  	{
  		cin>>a[i];
  		dp[i]=dp[i-1]+a[i];
  		//cout<<dp[i]<<" ";
  	}
  	//cout<<"\n";
  	ndp[n+1]=0;
  	for(int i=n;i>=1;i--)
  	{
  		ndp[i]=ndp[i+1]+a[i];
  		//cout<<"n"<<ndp[i]<<" ";
  	}
  		//cout<<"\n";
  	while(q--)
  	{
  		int l,p;
  		cin>>l>>p;
  		cout<<(dp[p]-dp[l-1])<<" "<<a[p]-(ndp[l]-ndp[p])<<"\n";
  	}
}
