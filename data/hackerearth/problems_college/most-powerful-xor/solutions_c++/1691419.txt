#include <iostream>
using namespace std;

int main()
{
   	int t;
   	
   	cin >> t;
   	
   	while(t--) {
   		int l, r;
   		int ans = 0;
   		cin >> l >> r;
   		for(int i = l; i <= r; i++) {
   			for(int j = i+1; j <= r; j++) {
   				ans = max(ans, i^j);
   			}
   		}
   		cout << ans << endl;
   	}
    return 0;
}
