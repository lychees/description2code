#include<bits/stdc++.h>
using namespace std;

#define si(x) scanf("%d",&x)
typedef long long LL;

const int mod = 1000000007;
inline int mul(int x, int y){ return ((LL)x * y)%mod;}

int F[100001];
bool isn_p[1300000];
int main()
{
    F[0] = 1;
    int n = 0;
    for(int i = 2; n<100000; i++)
    {
        if(isn_p[i])continue;
        n++;
        F[n] = mul(F[n-1],i);
        for(LL j = (LL)i*i; j<1300000; j+=i)isn_p[j] = true;
    }
    int t; si(t); while(t--)
    {
        int n; si(n); cout<<F[n]<<endl;
    }
    return 0;	
}
