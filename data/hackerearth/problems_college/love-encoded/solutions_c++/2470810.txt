#include<stdio.h>
#include<cstring>
char arr[210],tar[210];
int main(){
    int t,i,j,l;
    scanf("%d",&t);
    while(t--){
        scanf("%d",&l);
        scanf("%s",arr);
        for(i=0;i<strlen(arr);i++){
            if(arr[i]-l >= 65)
                tar[i] = (char)arr[i]-l;
            else
            tar[i] = (char)(arr[i]-l+26);
        }
        tar[i]='\0';
        printf("%s\n",tar);
    }
}
