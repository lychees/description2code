#include <iostream>
#include <vector>
using namespace std;

unsigned long zeroCount(unsigned long);
int main()
{
	int t;
	long long numt;
	vector<long long> num;
    
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>numt;
    	
    	num.push_back(numt);
    }
    for(int i=0;i<t;i++)
    {
    	cout<<zeroCount(num.at(i))<<endl;
    }
    return 0;
}

unsigned long zeroCount(unsigned long n)
{
	
	
	unsigned long count=0;
	if(n<=4)
	{
		return 0;
	}
	if(n==5)
	{
		return 1;
	}
	
	for(unsigned long j=5;n/j>=1;j*=5)
	{
		count+=n/j;
	}
	
	return count;
}
