for _ in range(input()):
    cost = map(int, raw_input().split())
    for x in range(10):
        for y in range(1, 10):
            for z in range(y, 10):                
                cost[(y + z) % 10] = min(cost[(y + z) % 10], cost[y]+cost[z])
               
    input()
    ans = 0
    nums = raw_input()
    for  x in nums:
        ans += cost[int(x)]
    print ans
