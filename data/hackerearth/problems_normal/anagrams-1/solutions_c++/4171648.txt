#include<bits/stdc++.h>
using namespace std;
char extra;
int odd;

void input(char* S,int* H,bool time)
{
      gets(S);
      for(int i=0; S[i]!=NULL; i++)
       {
         if(S[i]>=65 && S[i]<=90)
           S[i]=S[i]+32;
         if(S[i]>=97 && S[i]<=122)
            {
             if(time==0)
              H[S[i]-'a']++;
             else
              H[S[i]-'a']--;     
            }   
       }
       
}

bool palintrue(int* H, int slen)
{ 
  odd=0;
  for(int i=0; i<=25; i++)
  {
    if(H[i]%2==1)
     {
       odd++;
       extra=(char)(i+'a');
       H[i]--;            
     }
  }
  
  if(slen%2==0 && odd!=0)
   {
     return false;
   }
  if(slen%2!=0 && odd!=1)
   {
     return false;
   }
  return true;  
}

void makepalin(int* H,char* S, int slen)
{
  int stindex_from_mid,midindexb,midindexa,stindex,endindex;
  bool Z[100];
  for(int i=0; i<=99; i++)
   {
    Z[i]=false;
   }
   
  if(slen%2==1)
   slen--;
  midindexb=slen/2-1;
  midindexa=slen/2;
  for(int i=25; i>=0; i--)
  {
    if(H[i]!=0)
    {
      stindex_from_mid=H[i]/2-1;
      stindex=midindexb-stindex_from_mid;
      endindex=midindexa+stindex_from_mid;
      for(int j=stindex; j<=endindex; j++)
        { if(Z[j]==false)
           {
            S[j]=(char)(i+'a');
            Z[j]=true;
           } 
        }
        midindexb=stindex-1;
        midindexa=endindex+1;
    }
  }
  
}

void display(char* S, int slen)
{
   if(slen%2==1)
    slen--;
   if(slen==0)
    cout<<extra;
   else  
   for(int i=0; i<=slen-1; i++)
    {
      if(i==slen/2 && odd==1)
       cout<<extra;
      cout<<S[i]; 
    }
   cout<<"\n"; 
}

int main()
{
   short t;
   char S[101],w;
   int H[26];
   scanf("%d%c",&t,&w);
   while(t--)
   {  
      for(int i=0; i<=25; i++)
       H[i]=0;
      
      input(S,H,0);
      input(S,H,1);
      
      int slen=0; bool neg=0,pos=0;
      for(int i=0; i<=25; i++)
       {
        slen=slen+H[i];
        if(H[i]>0)
         pos=1;
        if(H[i]<0)
         neg=1; 
       }
      if(pos==1 && neg==1)
       cout<<"NO LUCK\n";
      else if(slen==0)
       cout<<"YES\n";
      else
      {  
         slen<0?slen=(-1)*slen:1;           //look
         for(int i=0; i<=25; i++)
          {
		H[i]<0?H[i]=(-1)*H[i]:1;
          }
          
         if(palintrue(H,slen))
          {
            makepalin(H,S,slen);
		display(S,slen);
          }
	   else
          { 
		cout<<"NO LUCK\n";
          } 
      } 
   }
}
