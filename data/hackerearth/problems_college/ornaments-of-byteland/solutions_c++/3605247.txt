#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;

    map<char,int> m;
    map<char,int>::iterator it;

    for(int i=0;i<n;i++)
    {
        string s;
        cin>>s;

        map<char,bool> visited;

        for(int j=0;j<s.size();j++)
        {
            if(visited[s[j]] == false)
            {
                m[s[j]]++;
                visited[s[j]]=true;
            }
        }
    }

    int counter =0;

    for(it=m.begin();it!=m.end();it++)
        if((*it).second == n)
            counter++;

    cout<<counter;


}
