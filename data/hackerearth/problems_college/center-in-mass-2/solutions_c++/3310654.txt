lld modpow(lld a,lld n,lld temp){lld res=1;while(n>0){if(n&1)res=(res*a)%temp;a=(a*a)%temp;n>>=1;}return res%temp;} 
 
int main()
{
  int t;
  cin>>t;
  int T=t;
  while (t--) {
    long double x,y,z,vx,vy,vz, X=0,XX=0,Y=0,YY=0,Z=0,ZZ=0;
    int n;
    cin>>n;
    int i;
rep (i, n) {
      cin>>x>>y>>z>>vx>>vy>>vz;
      X+=x;
      Y+=y;
      Z+=z;
      XX+=vx;
      YY+=vy;
      ZZ+=vz;
    }
    X/=n;
    XX/=n;
    Y/=n;
    YY/=n;
    Z/=n;
    ZZ/=n;
    long double tr = -(X*XX+Y*YY+Z*ZZ)/(XX*XX+YY*YY+ZZ*ZZ);
    if (tr < 0) tr = 0;
    long double d = sqrt((X+XX*tr)*(X+XX*tr) + (Y+YY*tr)*(Y+YY*tr) + (Z+ZZ*tr)*(Z+ZZ*tr));
    printf("Case #%d: %.8Lf %.8Lf\n", T-t, d, tr);
  }
	return 0;
}
