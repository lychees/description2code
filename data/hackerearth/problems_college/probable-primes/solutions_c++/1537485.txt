#include <iostream>
#include<stdio.h>
#include<math.h>
using namespace std;

long maxx=1000001;
int primes[1000001];
int main()
{
 int t;
 for(long i=2;i<maxx;i++)
 	primes[i]=1;

 primes[0]=primes[1]=0;
 
 int count[maxx];
 
 for(long i=2;i*i<maxx;i++)
 	{
 		if(primes[i]==1)
		 	for(long j=i;  i*j<maxx ;j++)
		 	{
		 		primes[i*j]=0;
		 	}
 	}
 	 
 	int c=0;
 for(long i=0;i<maxx;i++)
 	{
 		if(primes[i]==1)
 			c++;
 		count[i]=c;
 	}
// for(int i=0;i<10;i++)
 //		cout<<i<<" "<<count[i]<<" "<<primes[i]<<"\n";
 cin>>t;
	while(t--) {
		int l,r;
		cin>>l>>r;
		float ans=(float)(count[r]-count[l-1])/(r-l+1);
		printf("%1.6f\n",ans);
		
	}
    return 0;
}
