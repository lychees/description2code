#include <iostream>
using namespace std;

int main()
{
	int k,m;
	cin>>k>>m;
	int arr_b[m],count=0,i;
	for(i=0;i<m;i++)
	{
		arr_b[i]=i+1;
		
	}
	while(k--)
	{
		int n,i;
		cin>>n;
		int arr[n];
		for(i=0;i<n;i++)
		{
			cin>>arr[i];
			
		}
		for(i=0;i<m;i++)
		{
			for(int j=0;j<n;j++)
			{
				if(arr_b[i]==arr[j])
				{
					arr_b[i]=-1;break;
					
				}
				
			}
			
		}
		
		
		
	}
	for(i=0;i<m;i++)
	{
	
	   if(arr_b[i]==-1)
	   {
	   	 
	   	 count++;
	   	
	   }
		
	}
	
	if(count==m)
	{
		cout<<"YES"<<"\n";
	}else{
		cout<<"NO"<<"\n";
	}
	
	
    return 0;
}
