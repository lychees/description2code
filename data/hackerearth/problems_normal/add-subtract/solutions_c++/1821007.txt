int aa[1005],d[105][105];
int add_sub(int k,int n,int val)
{
	int& ans = d[n][k];
	
	if( n < 0 && k != 0) 
	{
		return INF;
	}
	
	if( k == 0)
	{
		return 0;
	}
	
	
	if(ans != -1) 
	{
		return ans;
	}
	
	ans = INF;
	ans = add_sub(k,n-1,val);
	if( val < aa[n])
	{
		ans = min(ans,add_sub(k-1,n-1,val)+5*(aa[n]-val));
	}
	else 
	{
		ans = min(ans,add_sub(k-1,n-1,val)+3*(val-aa[n]));
	}
	return ans;
}

			res = min(res,add_sub(k,n-1,i));

	}
