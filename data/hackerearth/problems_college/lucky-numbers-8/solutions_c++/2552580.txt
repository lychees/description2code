#include<iostream>
using namespace std;
main()
{
    int arr[100000],t,n,c;
    cin>>t;
    while(t--)
    {
        c=1;
        cin>>n;
        for(int i=0;i<n;i++)
            cin>>arr[i];
        int flag;
        for(int j=0;j<n;j++)
        {
            flag=0;
            if(j==n-1)
                goto en;
            for(int k=j+1;k<n;k++)
            {
                if(arr[j]>arr[k])
                    flag=1;
                else
                {
                    flag=0;
                    goto en;
                }
            }
            en:;
            if(flag==1)
                c++;
        }
        cout<<c<<endl;
    }
}
