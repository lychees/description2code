#define fr(i, a, b) for(i=a; i<=b; i++)
#define fb(i, a, b) for(i=a; i>=b; i--)

#include <ext/pb_ds/assoc_container.hpp>
#include <ext/pb_ds/tree_policy.hpp>

const int maxn = 1e5;
int n, k, ans;
vector<int> adj[ maxn ];
int sz[ maxn ];
void dfs( int u, int p = 0) {
    int i;
    sz[u] = 1;
    fr(i, 0, (int)adj[u].size()-1) {
        if( adj[u][i] != p ) {
            dfs(adj[u][i], u);
            sz[u] += sz[adj[u][i]];
        }
    }
    if(sz[u] >= k) {
        ans ++;
        sz[u] = 0;
    }
}
int main() {
    sc2(k, n);
    int i; fr(i, 2, n) {
        int x; sc1( x );
        adj[x].pb( i );
        adj[i].pb( x );
    }
    dfs(1);
    cout << ans << "\n";
    return 0;
}
