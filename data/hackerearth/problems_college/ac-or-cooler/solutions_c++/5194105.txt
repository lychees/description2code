#include <cstdio>
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int main()
{
    int n,m,i,c=0,cost=0;
    scanf("%d%d", &n,&m);
    int *a=new int[2*n];
    for(i=0; i<2*n; i++)
    {
    	scanf("%d", (a+i));
    }
    std::sort(a,a+2*n);
    for(i=0; i<2*n; i++)
    {
    	if(cost+a[i]<=m && c<n)
    	{
    		c++;
    		cost+=a[i];
    		//printf("%d ",cost);
    	}
    	else
    		break;
    }
    printf("%d\n",c);
    return 0;
}
