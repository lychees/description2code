#include<iostream>
using namespace std;
int main()
{
    int Q;
    cin>>Q;
    while(Q--){
        unsigned long long int N;
        int T;
        cin>>N>>T;
        while(T--)
        {
            if(N%2==0)
                N=N-N/2;
            else
                N=N/2;
            N= N - N/4;
        }
        cout<<N<<endl;
    }
    return 0;
}
