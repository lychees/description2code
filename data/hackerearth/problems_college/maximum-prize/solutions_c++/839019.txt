#include <cstdio>
#include <algorithm>

using namespace std;

int main()
{
    int t,i,n,k;
    scanf ("%d",&t);
    while (t--)
    {
        scanf ("%d %d",&n,&k);
        int p[n];
        long long int max=0;
        for (i=0; i<n; ++i)
            scanf ("%d",&p[i]);
        sort(p,p+n);
        if (k!=0)
        {
            for (i=(n-1); i>=0; --i)
            {
                max+=p[i];
                --k;
                if (k==0)
                    break;
            }
        }
        printf ("%lld\n",max);
    }
    return 0;
}
