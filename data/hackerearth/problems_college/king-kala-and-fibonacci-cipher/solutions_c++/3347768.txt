int main()
{
    int t;
    cin>>t;
    int s[100002];
    s[0]=0;
    s[1]=1;
    s[2]=1;
    s[3]=2;
    for(int i=4;i<100002;i++)
    {
    	s[i]=(s[i-1]+s[i-2])%26;
    }
    while(t--)
    {
    	int l,r;
    	cin>>l>>r;
    	char str[1000];
    	cin>>str;
    	int ll=strlen(str);
    	//int jj=0;
    	int b[30];
    	for(int i=0;i<26;i++)
    	b[i]=0;
    	for(int i=l;i<=r;i++)

    		b[s[i]]+=1;
    	}

    	for(int i=0;i<26;i++)

    		int max=b[0],pos=0;
    		for(int j=1;j<26;j++)
    		{
    			if(b[j]>max)
    			{
    				max=b[j];
    				pos=j;
    			}
    		}
