#include<bits/stdc++.h>
#define ll long long int
#define mod 1000000007
ll powerfunc(ll a,ll b)
{
	if(b==0)
	return 1;
	ll p=powerfunc(a,b/2)%mod;
	p=(p*p)%mod;
	if(b&1)
	p=(p*a)%mod;
	return p;
}
int main()
{
	ll t,n;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		ll p=powerfunc(2,n-1);
		p=powerfunc(p,mod-2);
		printf("%lld\n",p);
	}
}
