#include<bits/stdc++.h>
using namespace std;
long long gcd(long long a,long long b){
     if(b==0)
     return a;
     else
     gcd(b,a%b);
}
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		long long n;
		int a,b;
		scanf("%lld %d %d",&n,&a,&b);
		if(a==0||b==0){
			printf("-1\n");
			continue;
		}
		else{
	    long long x;
	      x=(a*b)/gcd(a,b);
	    long long y=((n/a)+(n/b)-n/(x));
		printf("%lld\n",y);
		}
	}
	return 0;
}
