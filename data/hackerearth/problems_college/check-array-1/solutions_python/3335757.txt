t = int(raw_input())
for _ in range(t):
	n = int(raw_input())
	a = [int(i) for i in raw_input().split()]
	sumleft = [None] * n
	sumright = [None] * n
	sumleft[0] = 0
	sumright[n - 1] = 0 
	for i in range (1, n):
		sumleft[i] = sumleft[i - 1] + a[i - 1]
		sumright[n - i  -1] = sumright[n - i] + a[n - i]
	flag = 0
	for i in range(n):
		if(sumleft[i] == sumright[i]):
			print 'YES'
			flag = 1
			break
	if flag == 0:
		print 'NO'
