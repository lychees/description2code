#include <bits/stdc++.h>
#define ll long long int
using namespace std;

ll max(ll a, ll b)
{
	return ((a>b)?a:b);
}

int main()
{
	ll t,i,j,n,k;
	cin >> t;
	while(t--) {
		cin >> n >> k;
		ll a[n];
		ll b[n];
		for(i=0;i<n;i++) {
			cin >> b[i];
		}
		for(i=0;i<n;i++) {
			cin >> a[i];
		}
		ll knap[n+1][k+1];
		for(i=0;i<=n;i++) {
			for(j=0;j<=k;j++) {
				if(i==0||j==0) {
					knap[i][j]=0;
				}
				else if(b[i-1]<=j) {
					knap[i][j]=max(knap[i-1][j],knap[i-1][j-b[i-1]]+a[i-1]);
				}
				else knap[i][j]=knap[i-1][j];
		//		cout << knap[i][j] << " ";
			}
		//	cout << endl;
		}
		
		cout << knap[n][k] << endl;
	}
    return 0;
}
