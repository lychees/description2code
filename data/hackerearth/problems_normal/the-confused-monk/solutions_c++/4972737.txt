#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
int power(long long p,int s)
{
	long long ans=1;
	while(s!=0){
		if(s%2==1)
		{
			ans=((ans%MOD)*(p%MOD))%MOD;
			//cout<<ans<<endl;
		}
		p=((p%MOD)*(p%MOD))%MOD;
		s=s/2;



	}
	return ans;
}
int gcd(int u,int v)
{

   while(v!=0)
    {
        int temp=u;
        u=v;
        v=temp%v;
    }

    return u;
}
