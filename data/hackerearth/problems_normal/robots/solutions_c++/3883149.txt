#include <iostream>
#include <stdio.h>
#include <algorithm>
#define MOD 1000000007
using namespace std;

int main()
{
    int T;
    scanf("%d", &T);
    while(T--)
    {
    	int N;
    	scanf("%d", &N);
    	int a[N];
    	for(int i=0;i<N;++i)
    	{
    		scanf("%d", &a[i]);
    	}
    	sort(a, a + N);
    	//for(int i =0;i<N;++i)
    	//	printf("%d ", a[i]);
    	long long int ans = a[0];
    	for(int i=1;i<N;++i)
    	{
    		if(a[i] > i)
    			ans = (ans * (a[i] - i)) % MOD;
    		else
    		{
    			ans = 0;
    			break;
    		}
    	}
    	printf("%d\n", ans);
    }
    return 0;
}
