T = int(raw_input())
A = [-1 for i in range(1000001)]
A[0] = 1
A[1] = 1
maxn = 1
for i in range(T):
    n = int(raw_input())
    if n > maxn:
        for j in range(maxn+1, n+1):
            A[j] = (A[j-1]+(j-1)*A[j-2]) % 1000000007
        maxn = n
    print A[n]
