#include <bits/stdc++.h>
using namespace std;

long gangster_power[10001];

int bin_search (long a[], int n, long search);

int main()
{
    int no_of_gangsters,TC,sam_power,i,k;
    cin>>TC;
    
    while(TC--)
    {
    	cin>>no_of_gangsters>>sam_power;
    	
    	for(i=0; i<no_of_gangsters; i++)
    	cin>>gangster_power[i];
    	
    	sort(gangster_power, gangster_power + no_of_gangsters);
    	
    	k=bin_search(gangster_power,no_of_gangsters,sam_power);
    	
    	if(k==-1)
    	cout<<"NO 0"<<endl;
    	else
    	cout<<"YES "<<no_of_gangsters-k-1<<endl;
    }
    
    return 0;
}

int bin_search (long a[], int n, long search)
{
	int first,last,middle;
	first=0;
	last=n-1;
	middle = (first+last)/2;
	
	while(first<=last)
	{
		if(a[middle]>search)
		last=middle-1;
		
		else if(a[middle]==search)
		return middle;
		
		else
		first=middle+1;
		
		middle=(first+last)/2;
	}
	return -1;
}
