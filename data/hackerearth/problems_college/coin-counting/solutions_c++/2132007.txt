int calc(int prev,int val)
{
	if(val>no)
	{
		return 0;
	}
	if(val==no)
		return 1;
	int sm=0;
	for(int i=prev;i<3;i++)
	{
		sm+=calc(i,val+arr[i]);
	}
	return sm;
}
