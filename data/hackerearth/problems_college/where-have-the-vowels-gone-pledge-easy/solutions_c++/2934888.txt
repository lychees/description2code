#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    long n,i,ctr1=0,ctr2=0,ctr3=0,ctr4=0,ctr5=0;
    char a[10005];
    cin>>n;
    cin>>a;
    for(i=0;i<strlen(a);i++)
    {
    	if(a[i]=='a')
    		ctr1++;
    	if(a[i]=='e')
    		ctr2++;
    	if(a[i]=='i')
    		ctr3++;
    	if(a[i]=='o')
    		ctr4++;
 		if(a[i]=='u')
    		ctr5++;
 		if(ctr1 && ctr2 && ctr3 && ctr4 && ctr5)
 			break;
 	}
 	if(ctr1 && ctr2 && ctr3 && ctr4 && ctr5)
 		cout<<"YES";
 	else
 		cout<<"NO";
    return 0;
}
