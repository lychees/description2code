#include<iostream>
#include<string>
using namespace std;

int main()
{
    int t;
    long m,n,k;
    cin>>t;
    while(t--)
    {   k=0;
        cin>>n>>m;
        string *s=new string[n];
        for(long i=0;i<n;i++)
        {
            cin>>s[i];
            for(long j=0;j<i;j++)
            {
                if(s[j]==s[i])
                {
                    k=1;
                    break;
                }
            }

        }
        if(k==1)
            cout<<"No\n";
        else cout<<"Yes\n";
    }




}
