#include <iostream>
#include <string>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
    int left = 0;
    int right = 0;
    int al = 0;
    int ar = 0;
    string s;
    cin>>s;
    int i=0;
    while(s[i]!='\0')
    {
    	if(s[i]=='(')
    	{
    		++left;
    	}
    	else if(s[i]==')')
    	{
    		if(left>0)--left;
    		else
    			++right;
    	}
    	++i;
    }
    while(right>0)
    {
    	cout<<"(";
    	--right;
    }
    cout<<s;
    while(left)
    {
    	cout<<")";
    	--left;
    }
    cout<<endl;
}
    return 0;
}
