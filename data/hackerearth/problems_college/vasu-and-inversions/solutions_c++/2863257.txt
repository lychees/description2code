#include<bits/stdc++.h>
using namespace std;


int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
   int t;
	cin>>t;
	while(t--)
	{

		int n, k, zero=0, one=0 ;
		cin>>n>>k;
		int x;
		for(int i=1; i<=n; i++)
		{

			cin>>x;
			if(x==0)zero++;
			else one++;
		}
		int total = 0;
		if(zero>=k)
		{

			total = one + k;
		}
		else
		{

			
			total = zero + one ;
			if((zero%2==0 && k%2==1) || (zero%2==1 && k%2==0))total-=1;
		}
		cout<<total<<endl;
	}
    return 0;
}
