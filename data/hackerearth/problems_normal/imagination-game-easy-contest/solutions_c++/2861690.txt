#include <stdio.h>
using namespace std;

int main()
{
    int t, n, dim[10], mark[10];
    scanf("%d", &t);
    while(t--)
    {
    	scanf("%d", &n);
    	int i=0;
    	while(i<n)
    	{
    		scanf("%d", &dim[i]);
    		i++;
    	}
    	i=0;
    	scanf("%d", &mark[i]);
		int _xor = mark[i]-1;
		_xor ^= dim[i] - mark[i];
		i++;
    	while(i<n)
    	{
    		scanf("%d", &mark[i]);
    		_xor ^= mark[i]-1;
    		_xor ^= dim[i] - mark[i];
    		i++;
    	}
    	
    	if(_xor)
    		printf("First\n");
    	else
    		printf("Second\n");
    }

    return 0;
}
