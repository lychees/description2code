#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    long long n,a,b,i;
    cin>>n;
    vector<pair<long long, long long> >v;
  //  long long a[n],b[n];
    for(i=0;i<n;i++)
    {
    	cin>>a>>b;
    	v.push_back(make_pair(a+b,i));
    }
    sort(v.begin(),v.end());
    for(i=0;i<n;i++)
    cout<<v[i].second+1<<" ";
}
