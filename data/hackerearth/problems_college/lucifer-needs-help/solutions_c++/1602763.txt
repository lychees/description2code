#include <iostream>
#include <algorithm>
using namespace std;
bool aa (int i,int j) { return (i>j); }
int main()
{
    int arr[1000],n,m,flag=0;
    cin>>n>>m;
    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    sort(arr,arr+n);
    for(int i=0;i<n;i++)
    {

        if(arr[i]<=m)
        {
            for(int j=0;j<i;j++)
            {

                if(arr[i]+arr[j]<=m)
                {
                    for(int k=0;k<j;k++)
                    {
                       // cout<<i<<' '<<j<<' '<<k;
                        if(arr[i]+arr[j]+arr[k]==m)
                        {
                            flag=1;
                            cout<<"YES";
                            break;
                        }
                    }
                }
                else
                    break;
                if(flag==1)
                    break;
            }
        }
        else break;
        if(flag==1)
            break;
    }
    if(flag==0)
        cout<<"NO";
    return 0;
}
