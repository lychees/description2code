#include <iostream>
using namespace std;

void solve(){
	string s;
	cin>>s;
	long long m;
	cin>>m;
	long long ans = 0;
	long long p10 = 1;
	int len = s.size();
	for(int i = len-1; i>=0; i--){
		ans += (((long long)(s[i]-'0'))*(p10))%m;
		p10 *= 10;
		p10 %= m;
		ans %= m;
	}
	cout<<ans<<endl;
}

int main()
{
    int t;
    cin>>t;
    while(t--) solve();
    return 0;
}
