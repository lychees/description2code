def factorial(input):
	fact =1
	for val in range(1,input+1):
		fact = fact*val
	return fact

count = int(raw_input())
for val in range(0,count):
	input = int(raw_input())
	fact = str(factorial(input))
	if len(fact) <=5:
		print ("0"*(5-len(fact)))+fact
