#include<bits/stdc++.h>
using namespace std;
#define ll long long

ll p[100003];

ll root(ll x)
{
    while(p[x]!=x)
    {
        p[x]=p[p[x]];
        x=p[x];
    }
    return x;
}

void union1(ll a,ll b)
{
    ll root_a=root(a);
    ll root_b=root(b);
    if(root_a!=root_b)
    p[root_a]=p[root_b];
}

bool find1(ll a,ll b)
{
    ll root_a=root(a);
    ll root_b=root(b);
    if(p[root_a]==p[root_b])
        return true;
    else
        return false;
}

int main()
{
    ll n,m,i,j,k,l,x,y,z,q;
    cin>>n>>q;
    for(i=1;i<=n;i++)
    {
        p[i]=i;
    }
    while(q--)
    {
        cin>>k>>x;
        if(k==1)
        {
            cin>>y;
            if(find1(x,y)==false)
            {
                union1(x,y);
            }
            //union1(x,y);
        }
        else if(k==2)
        {
            m=root(x);
            //cout<<m<<" ";
            p[m]=x;
            p[x]=x;
        }
        else
        {
            cout<<root(x)<<"\n";
        }
    }
}
