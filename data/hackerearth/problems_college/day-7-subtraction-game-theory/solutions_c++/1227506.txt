#include <iostream>
#include<stdio.h>
#define mod 1000000007
using namespace std;

long long int a[1000005];

void f(long long int x,long long int y)
{
    long long int i;
    a[0]=1;
    for(i=1;i<=1000002;i++)
    {
        a[i]=( (a[i-1]*x)-(y))%mod;
    }
}

int main()
{
    long long int t,i,n,q,x,y,c;

        scanf("%lld%lld",&x,&y);
        //precompute
        f(x,y);

        scanf("%lld",&q);
        for(c=1;c<=q;c++)
        {
            scanf("%lld",&n);
            printf("%lld\n",a[n]);
        }

    return 0;
}
