#include <stdio.h>
#include<algorithm>
using namespace std;
int main() 
{
	int test;
	scanf("%d",&test);
	while(test--)
	{
		long long int N,ans;
		scanf("%lld",&N);
		long long int count=1,power2=2,prev_sum=0,sum=2,diff;
		while(sum<N)
		{
			prev_sum=sum;
			power2=power2*2;
			count++;
			sum+=power2*count;
		}
		if(sum==N)
			printf("%lld\n",2*count-1);
		else 
		{
			long long int POS=prev_sum+((2*count)-1);
			if(N>=POS)
				printf("%lld\n",2*count-1);
			else{
				diff=POS-N;
				ans=(max(2*(count-1)-1,(2*(count)-1)-diff));
				printf("%lld\n",ans);
			}
		}
	}
	return 0;
}
