#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    //cout << "Hello World!" << endl;
    ll n,t,i,k,a[200005];
    cin>>t;
    while(t--){
    	cin>>n>>k;
    	for(i=1;i<=n;i++) cin>>a[i];
    	ll mi=0,ans=0;
    	for(i=1;i<=n;i++){
    		if(a[i]>k){
    			ans+=i;
    			mi=i;
    		}
    		else{
    			ans+=mi;
    		}
    	}
    	
    	
    	cout<<ans<<endl;
    }
    
    
    return 0;
}
