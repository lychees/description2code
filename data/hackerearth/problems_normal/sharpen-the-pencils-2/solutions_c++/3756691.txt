typedef long long LL;
#define MOD 1000000007
LL mpow(LL a, LL n) 
{LL ret=1;LL b=a;while(n) {if(n&1) 
    ret=(ret*b)%MOD;b=(b*b)%MOD;n>>=1;}
return (LL)ret;}
#define MAXN 100000
#define MAXV 10000000
int ar[MAXN+10];
LL pre[MAXN+10];
int main()
{
    int t;
    sd(t);
    assn(t,1,1000);
    while(t--){
        int n;
        sd(n);
        assn(n,1,MAXN);
        for(int i=1; i<=n; i++){
            sd(ar[i]);
            assn(ar[i],1,MAXV);
            pre[i]=pre[i-1]+(LL)ar[i];
        }
        int i,ans;
        for(i=1; i<=n; i++){
            LL y=2*(pre[n]-pre[i]);
            LL x=pre[i-1];
            LL sol=y+2*ar[i]-x;
            if(sol==0){ans=i-1;break;}
            if(sol==3*ar[i]){ans=i;break;}
            else if(sol>0 and sol<3*ar[i]){
                if(x<=y)ans=i;
                else ans=i-1;
                break;
            }
        }
        cout << ans << " " << n-ans << endl;
    }
    return 0;
}
