#include <bits/stdc++.h>
using namespace std;
#define INF 1<<24
#define inp(n) scanf("%lld",&n)
long long N,M;
long long d[100001];
bool vis[100001];
struct node{
	long long dest,weight;
	struct node* next;
};
struct node * g[100001];
 
 
struct node * new_node(long long d,long long w){
	struct node * n= new node;
	n->dest=d;
	n->weight=w;
	n->next=NULL;
	return n;
}
 
void addedge(long long a,long long b,long long c){
	struct  node * t=new_node(b,c);
	t->next=g[a];
	g[a]=t;
	t=new_node(a,c);
	t->next=g[b];
	g[b]=t;
 
 
}
long long min_w(){
	long long m=INT_MAX,idx=-1;
	for(long long i=1;i<=N;i++){
		if(d[i]<m && !vis[i])
			m=d[i],idx=i;
	}
	return idx;
}
void dijkstra(int u){
	d[u]=0;
	for(int i=1;i<=N-1;i++){
		int p=min_w();
		//cout<<p<<endl;
		vis[p]=1;
		struct  node *t=g[p];
		while(t){
			if( d[p]!=INF && !vis[t->dest] && d[p]+t->weight<d[t->dest])
				d[t->dest]=d[p]+t->weight;
				t=t->next;
		}
	}
}

    long long a,b,c;
 
   for(long long i=0;i<M;i++) {
   	inp(a);inp(b);inp(c);
