typedef long long ll;

#define foreach(it, v) for(__typeof((v).begin()) it = (v).begin(); it != (v).end(); ++it)
#define sz(v) int((v).size())

const ll MODULO = 1000000000 + 7;

int main()
{
    ios_base::sync_with_stdio(0);
    string s;
    cin >> s;
    ll count_before = 0, result_before = 0, result = 0;
    for(size_t i = 0; i < s.size(); i++)
    {
        ll digit = (s[i] - '0');
        ++count_before;
        result_before = (10 * result_before + count_before * digit) % MODULO;
        result = (result + result_before) % MODULO;
    }
    cout << result << endl;
    return 0;
}
