#include<bits/stdc++.h>
#define ll long long
bool arr[10000000];
using namespace std;
int main()
{
	ll i,c=0;
	ll t;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>s;
		c=0;
		ll len=s.length();
		for(i=0;i<len;i++)
		arr[i]=1;
		for(i=0;i<len;i++)
		{
			if(s[i]=='D')
			arr[i]=0;
			else if(s[i]=='B')
			{
				arr[i]=0;
				if(i>0)
				arr[i-1]=0;
				if(i>1)
				arr[i-2]=0;
			}
			else if(s[i]=='S')
			{
				arr[i]=0;
				if(i>0)
				arr[i-1]=0;
				if(i<len-1)
				arr[i+1]=0;
			}
		}
		for(i=0;i<len;i++)
		{
			if(arr[i]==1)
			c++;
		}
		cout<<c<<"\n";
	}
return 0;
}
