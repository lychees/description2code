 #include <bits/stdc++.h>
    using namespace std;
 
    int main()
    {
    char ch[60];
    while(gets(ch))
    {
    int flag=0;
    int cnt=0;
    int l=strlen(ch);
    for(int i=0;i<l;i++){
    if(ch[i]=='/' && ch[i+1]=='/')
    {
    flag=1;
    }
    else
    if(flag==0 && ch[i]=='-' && ch[i+1]=='>')
    {
    ch[i]='.';
    cnt++;
    for(int j=i+1;j<l;j++)
    {
    ch[j]=ch[j+1];
    }
    }
    }
    for(int i=0;i<l-cnt;i++)
    cout << ch[i];
    cout << "\n";
    }
    return 0;
    }
