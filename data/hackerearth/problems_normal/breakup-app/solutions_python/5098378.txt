c=0
d=0
e=0
for i in range(input()):
	a=raw_input()
	l=[int(s) for s in a.split() if s.isdigit()]
	if(len(l)>0):
		if(a[0]=="G"):
			for j in range(0,len(l)):
				if(l[j]==19):
					c+=2
				elif(l[j]==20):
					e+=2
				else:
					d+=2
		elif(a[0]=="M"):
			for k in range(0,len(l)):
				if(l[k]==19):
					c+=1
				elif(l[k]==20):
					e+=1
				else:
					d+=1
if(c>d or e>d):
	print("Date")
else:
	print("No Date")
