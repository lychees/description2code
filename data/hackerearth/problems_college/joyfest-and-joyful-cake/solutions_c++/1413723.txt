#include <bits/stdc++.h>
#define lli long long

using namespace std;

int main()
{
	int t;
	scanf("%d", &t);
	while ( t-- ) {
		lli n;
		scanf("%lld", &n);
		lli l,r,m,ans;
		l = 0, r = 1000000000;
		while ( l <= r ) {
			m = (l+r)>>1;
			lli val = m*(m+1);
			val += 2;
			val /= 2;
			if ( val >= n ) {
				ans = m;
				r = m-1;
			}
			else l = m+1;
		}
		printf("%lld\n", ans);
	}
	return 0;
}
