i = int(raw_input())
while i > 0 :
    n,m = raw_input().split()
    j = len(n)-1
    f = 0
    c = 0
    while j >= 0 :
        k = int(n[j])+int(m[j]) + c
        if k >= 10 :
            f += 1
            c = k//10
        else :
            c = 0
        j -= 1 
    i-=1
    if f == 0 :
        print 'No carry operation'
    elif f == 1 :
        print '1 carry operation'
    else :
        print f,'carry operations'
