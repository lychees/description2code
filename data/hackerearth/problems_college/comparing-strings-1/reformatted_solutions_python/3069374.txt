# print 'Hello World!'
t = input()
while t > 0:
    t -= 1
    s1 = raw_input()
    s2 = raw_input()
    c = 0
    for i in s1:
        if i in s2:
            c += 1
    if c == 0:
        print "NO"
    else:
        print "YES"
