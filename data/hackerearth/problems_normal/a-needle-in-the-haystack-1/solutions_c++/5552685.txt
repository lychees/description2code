#include<iostream>
#include<string>
#include<vector>
using namespace std;

int main()
{
    int t,flag;
    string sp,st;
    cin>>t;
    while(t--){
    	flag=0;
    	cin>>sp;
    	cin>>st;
    	vector<int> vecp(26,0);
    	vector<int> vect(26,0);
    	int sizep=sp.size();
    	int sizet=st.size();
    	if(sizet>=sizep){
    	for(int i=0;i<sizep;i++) vecp[sp[i]-97]++;
    	for(int i=0;i<sizep;i++) vect[st[i]-97]++;
    	for(int i=0;i<26;i++){
    		if(vecp[i]!=vect[i]){
    			flag=1;
    			break;
    		}
    	}
    	if(flag==0)
    	cout<<"YES"<<endl;
    	else{
    		for(int i=sizep;i<sizet;i++){
    			vect[st[i]-97]++;
    			vect[st[i-sizep]-97]--;
    			flag=0;
    			for(int i=0;i<26;i++){
    		    if(vecp[i]!=vect[i]){
    			flag=1;
    			break;
    	    	}
    	      }
    	      if(flag==0){
    	      	cout<<"YES"<<endl;
    	      	break;
    	      }
    		}
    		if(flag==1)
    		cout<<"NO"<<endl;
    	 }
    	}
    	else
    	cout<<"NO"<<endl;
    }
    return 0;
}
