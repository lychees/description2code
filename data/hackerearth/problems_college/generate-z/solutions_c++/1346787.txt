LL BigMod(LL a,LL b,LL mod)
{
      if(!b) return 1;
      LL x = BigMod(a,b/2,mod);
      x = (x*x)%mod;
      if(b%2)
            return (x*a)%mod;
      else
            return x;
}

LL modinv(LL a,LL mod)
{
      return BigMod(a,mod-2,mod);
}
VLL fact(1000);

LL choose(int n,int r)
{
      if(n < r) return -1;
      LL ans = fact[n];
      ans = (ans * modinv(fact[r],MOD))%MOD;
      ans = (ans * modinv(fact[n-r],MOD))%MOD;
      return ans;
}
