#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,i,j;
        cin>>n;
        vector<long long > a;
        long long s=0;
        long long p,q;
        for(i=0;i<n;i++)
        {
            cin>>p>>q;
            a.push_back(p+q);
            s+=q;
        }
        sort(a.rbegin(),a.rend());


        cout<<a[0]+a[1]-s<<endl;
        a.clear();
    }
    return 0;
}
