#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    long n,m;
    cin>>n>>m;
    int arr[n],brr[n];
    int i1=0,i2=0;
    for(int i=0;i<n;i++) cin>>arr[i];
    for(int i=0;i<n;i++) cin>>brr[i];
    sort(arr,arr+n);
    sort(brr,brr+n);
    int s = 0;
    while(s<m)
    {
    	if(arr[i1]<=brr[i2]) s+=arr[i1++];
    	else s+=brr[i2++];
    }
    cout<<i2+i1-1<<endl;
}
