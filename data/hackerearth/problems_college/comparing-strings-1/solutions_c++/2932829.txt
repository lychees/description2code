#include <iostream>
using namespace std;

int main()
{
	int t;
    cin>>t;
    while(t--)
    {
    	string s1, s2;
    	cin>>s1>>s2;
		bool flag=false;
    	int a[26]={0}, b[26]={0}, i;
    	for(i=0; i<s1.length(); i++)
    	a[s1[i]-'a']++;
    	for(i=0; i<s2.length(); i++)
    	b[s2[i]-'a']++;
    	
    	for(i=0;i<26;i++)
    	if(a[i]!=0&&b[i]!=0)
    	{
    		flag=true;
    		break;
    	}
    	if(flag)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
return 0;
}
