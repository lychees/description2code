
t=input()
while t>0:
	t-=1
	n,k=map(int, raw_input().split(" "))
	while True:
		prizes=map(int, raw_input().strip().split(" "))
		if len(prizes)==n:
			break
	if n!=k:
		prizes.sort()
		prizes.reverse()
	if k==0:
		print "0"
	else:
		max_prize=0
		for i in range(min(k,n)):
			max_prize+=prizes[i]
		print max_prize
