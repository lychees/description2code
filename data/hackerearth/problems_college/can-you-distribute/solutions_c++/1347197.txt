#include<bits/stdc++.h>
using namespace std;
#define mod 10000007
void EE(long long int a, long long int b, long long int& x, long long int& y)
{
    if(a%b == 0)
    {
        x=0;
        y=1;
        return;
    }
    EE(b,a%b,x,y);
    long long int temp = x;
    x = y;
    y = temp - y*(a/b);
}
long long int inverse(long long int a, long long int m)
{
    long long int x,y;
    EE(a,m,x,y);
    if(x<0) x += m;
    return x;
}
int main()
{
	long long int n,r,s,p,q,x=1,i;
	cin>>n>>r;
	if(n<r) {
		printf("-1\n");
		
	}
	else {
		p=n-1;
		q=r-1;
		s=1;
		for(i=1;i<=p;i++) {
			s=(s*i)%mod;
		}
		for(i=1;i<=q;i++) {
			s=(s*(inverse(i,mod)))%mod;
		}
		for(i=1;i<=p-q;i++) {
			s=(s*(inverse(i,mod)))%mod;
		}
		cout<<s<<endl;
	}
	return 0;
}
