#include <iostream>
using namespace std;

int main()
{
	int t; cin>>t;
	while(t--)
	{
		bool a[256]={0};
		bool res=false;
		string s,b; cin>>s>>b;
		for(int i=0;i<s.length();i++)
		a[s[i]]=true;
	for(int i=0;i<b.length();i++)
		{
			if(a[b[i]]) {res=true; break;}
		}
		if(res) cout<<"YES\n"; else cout<<"NO\n";
	}
    
    
    return 0;
}
