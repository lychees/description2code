#include<bits/stdc++.h>
using namespace std;
long long int n,k,kc,c[1000],sum=0;
int main() {
	cin>>n>>k;
	for (int i=0;i<n;i++) cin>>c[i];
	if (k>=n) {
		for (int i=0;i<n;i++) sum+=c[i];
		cout<<sum;
		return 0;
	}
	sort(c,c+n);
	int j=n-1,cnt=0,mult=1;kc=k;
	while (j>=0) {
		sum+=(mult*c[j]);
		cnt++;j--;
		if (cnt%k==0) mult++;
	}
	cout<<sum;
}
