#include <iostream>
#include<cstdio>
using namespace std;
#define MOD 1000000007
int a[1000010],f[1000010],g[1000010];
int main()
{
    int t,i,n;
    f[0]=1;
    g[0]=1;
    a[0]=2;
    for(i=1;i<1000005;i++)
    {
    	g[i]=(f[i-1]+g[i-1])%MOD;
    	f[i]=(g[i-1])%MOD;
    	a[i]=((f[i]+g[i])%MOD+a[i-1])%MOD;
    }
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	printf("%d\n",a[n-1]);
    }
    return 0;
}
