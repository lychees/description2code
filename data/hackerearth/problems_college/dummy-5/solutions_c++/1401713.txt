#include <bits/stdc++.h>
using namespace std;

int t,n,gp,gc,gf;
vector<int> p,c,f;

bool fff(int i,int pp,int cc,int ff){
    if(pp==gp&&cc==gc&&ff==gf)return 1;
    if(i==n)return 0;
    bool flag=0;
    flag|=fff(i+1,pp,cc,ff);
    pp+=p[i];cc+=c[i];ff+=f[i];
    flag|=fff(i+1,pp,cc,ff);
    return flag;
}

int main()
{
    cin>>t;
    for(int z=1;z<=t;z++){
        cin>>gp>>gc>>gf;
        cin>>n;
        p.resize(n);c.resize(n);f.resize(n);
        for(int i=0;i<n;i++)cin>>p[i]>>c[i]>>f[i];
        cout<<"Case #"<<z<<": ";
        if(fff(0,0,0,0))cout<<"yes\n";
        else cout<<"no\n";
    }
    return 0;
}
