#include <iostream>
#include <bits/stdc++.h>
using namespace std;
map <int,int> m;
int main()
{
    int n;
    int i,x;
    int ans=0,f=0;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>x;
    	m[i]=x;
    }
    for(i=0;i<n;i++)
    {
    	if(m[i]!=i)
    	{
    		if(m[m[i]]==i)
    		{
    			m[m[i]]=m[i];
    			m[i]=i;
    			f=1;
    		}
    	}
    	if(f)
    	{
    		break;
    	}
    }
    for(i=0;i<n;i++)
    {
    	if(m[i]==i)
    	{
    		ans++;
    	}
    }
    if(!f)
    {
    	ans-=2;
    }
    cout<<ans<<endl;
    return 0;
}
