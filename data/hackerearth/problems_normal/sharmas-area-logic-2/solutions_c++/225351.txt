#include <iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int i,x[n],y[n];
    	for(i=0;i<n;i++)
    		cin>>x[i]>>y[i];
    	sort(x,x+n);
    	sort(y,y+n);
    	if(x[n-1]-x[0]==0)
    		cout<<y[n-1]-y[0]<<endl;
    	else if(y[n-1]-y[0]==0)
    		cout<<x[n-1]-x[0]<<endl;
    	else
    		cout<<(x[n-1]-x[0])*(y[n-1]-y[0])<<endl;
    }
    return 0;
}
