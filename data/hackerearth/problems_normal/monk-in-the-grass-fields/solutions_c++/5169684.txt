#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
//    cout << "Hello World!" << endl;
    int T,K,N,i,j,k;
    cin >> T;
    while(T--)
    {
    	cin >> N >> K;
    	int arr[N][N];
    	for(i=0; i  <N;i++)
    	{
    		for(j=0;j<N;j++)
    		{
    			cin >> arr[i][j];
    		}
    	}
    	priority_queue<int, vector<int>, greater<int> > q1,q2;
    	for(i=0; i  <N;i++)
    	{
    		k = 0;
    		for(j=0;j<N;j++)
    		{
    			k += arr[i][j];
    		}
    		q1.push(k);
    	}
    	for(j=0; j  <N;j++)
    	{
    		k = 0;
    		for(i=0;i<N;i++)
    		{
    			k += arr[i][j];
    		}
    		q2.push(k);
    	}
    	int r[K+1],c[K+1];
    	r[0] =0; 
    	c[0] = 0;
    	long long int ans = 1e18,row=0,col=0;
    	for(i=1;i<=K;i++)
    	{
    		j = q1.top();
    		k = q2.top();
    		r[i] = r[i-1] + j;
    		c[i] = c[i-1] + k;
    		q1.pop();
    		q1.push(j+N);
    		q2.pop();
    		q2.push(k+N);
 //   		cout << "......................." << j << " " << k << endl; 
/*    		if((j+col) < (k+row))
    		{
    			cout << "......................." << j+col << " " << k+col << endl;
    			ans += (j+col);
    			row++;
    			q1.pop();
    			q1.push(j+N);
    		}
    		else
    		{
    			cout << "......................." << j+col << " " << k+col << endl;
    			ans += (k+row);
    			col++;
    			q2.pop();
    			q2.push(k+N);
    		}*/
    	}
    	for(i=1;i<=K;i++)
    	{
    		ans = min((r[i] + c[K-i] + (long long int) i*(K-i)),ans);
    		ans = min((r[K-i] + c[i] + (long long int) i*(K-i)),ans);
    	}
    	cout << ans << endl;
    	
    }
    
    return 0;
}
