#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<math.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    for(int i=0;i<t;i++)
    {    
         int a,b,c,n;
         scanf("%d%d%d%d",&n,&a,&b,&c);
         long long int co=0;
         int v=n;
         for(int i=0;i<=a;i++)
         {
                 
                 for(int j=0;j<=b;j++)
                 {
                         v=n-i-j;
                         if(v<0)
                         {    j=b+1;
                         }
                         else co+=1+min(c,v);
                 }
         }
         cout<<co<<"\n";
    }
    return 0;
}
