#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;


char str[11];


int find(char a,int l,int h)
{
    int x=l;
    for(int i=l+1;i<=h;i++)
    {
        if(str[i]>a&&str[i]<str[x])
            x=i;
    }
    return x;
}

void solution(int l)
{
    sort(str,str+l);
    int i;

    while(1)
    {
        printf("%s\n",str);
        for(i=l-2;i>=0;--i)
        {
            if(str[i]<str[i+1])
                break;
        }

    if(i==-1)
        break;
    else
    {
        int firstChar=find(str[i],i+1,l-1);
        swap(str[firstChar],str[i]);

        sort(str+i+1,str+l);
    }
    }

}
int main()
{
    scanf("%s",str);
    int l=strlen(str);
    solution(l);
    return 0;
}
