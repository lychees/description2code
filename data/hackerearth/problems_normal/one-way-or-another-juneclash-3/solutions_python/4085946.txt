from sys import stdin
t = int(stdin.readline())
for _ in xrange(t):
	n = int(stdin.readline())
	di = {}
	ans = 'Yes'
	for __ in xrange(n):
		x,y,z = map(int,stdin.readline().split())
		x = (x+y)%2
		z = z%2
		if x in di:
			if di[x] !=z:
				ans = 'No'
		else:
			di[x] = z
		if len(di)==2:
			if di[0]==di[1]:
				ans = 'No'
	print ans
