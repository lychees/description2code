#include <iostream>
using namespace std;
int t,i,j,k,l;
long int n;
int main()
{
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>n;
    	long int a[n],co=0;
    	for(j=0;j<n;j++)
    	{
    		cin>>a[j];
    	}
    	for(k=0;k<n;k++)
    	{
    		for(l=k;l<n;l++)
    		{
    			if(a[k]==a[l])
    			{
    				co++;
    			}
    		}
    	}
    	cout<<co<<endl;	
    }
    
    return 0;
}
