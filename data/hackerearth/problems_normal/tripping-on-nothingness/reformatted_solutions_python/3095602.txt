s, q = raw_input().split(" ")
s = int(s)
q = int(q)

saved_ans = []


def gcd(a, b):
    while a != 0:
        c = a
        a = b % a
        b = c
    return b


for i in xrange(q):
    a = raw_input()
    a = int(a)
    ans = gcd(s, a)

    if ans in saved_ans:
        print "-1"
    else:
        print ans

    saved_ans.append(ans)
