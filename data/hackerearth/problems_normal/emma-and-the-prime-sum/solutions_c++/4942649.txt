#include<bits/stdc++.h>
#define  den 1000001
typedef long long int lu;
using namespace std;
int main()
{
	lu i,j,x,ans,t,z,y;
	vector<int>ar(1000001);
	lu br[den]={0};
	for(i=2;i<=den-1;i++)
	{
		if(ar[i]==0)
		{
			j=i*2;
			x=3;
			for(j=i*2;j<den;)
			{
				ar[j]=1;
			j=i*x;
			x++;}
		}
	}
	
	ans=0;
	for(i=2;i<den;i++)
	{
		if(ar[i]==0)
		{
			br[i]=ans+i;
			ans=br[i];
		}
		else
		br[i]=ans;
	}
	
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&y,&z);
		ans=br[z]-br[y-1];
		cout<<ans<<endl;
		
	}
	return 0;
}
