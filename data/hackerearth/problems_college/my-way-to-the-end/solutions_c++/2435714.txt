#include <bits/stdc++.h>
using namespace std;
int gr[105][105];
int fl[105][105];
int di[105][105];
int main()
{
   // cout << "Hello World!" << endl;
   int t;
   scanf("%d",&t);
   while(t--)
   {
   	int n;
   	scanf("%d",&n);
   	int m;
   	scanf("%d",&m);
   	char a[105];
   	int i,j;
   //	printf("test\n");
   	for(i=1;i<=n;i++)
   	{
   		scanf("%s",&a);
   		for(j=1;j<=n;j++)
   		gr[i][j] = a[j-1]-'0';
   		for(j=1;j<=n;j++)
   	{
   		fl[i][j] = 0;
   		di[i][j] = -1;
   	}
   	}
   	di[1][1] = 0;
   	fl[1][1] = 1;
   	queue< pair<int,int> > q;
   	if(gr[1][1])
   	q.push(make_pair(1,1));
   	while(!q.empty())
   	{
   		pair<int,int> cur = q.front();
   		q.pop();
   		int x,y;
   		x = cur.first;
   		y = cur.second;
   		int u,v;
   		u = x; 
   		v = y+1;
   		if(v<=n && gr[u][v] && fl[u][v]==0)
   		{
   			q.push(make_pair(u,v));
   			fl[u][v] = 1;
   			di[u][v] = di[x][y] + 1;
   		}
   		u = x+1; 
   		v = y;
   		if(u<=n && gr[u][v] && fl[u][v]==0)
   		{
   			q.push(make_pair(u,v));
   			fl[u][v] = 1;
   			di[u][v] = di[x][y] + 1;
   		}
   	}
   	printf("%d\n",di[m][m]);
   }
    return 0;
}
