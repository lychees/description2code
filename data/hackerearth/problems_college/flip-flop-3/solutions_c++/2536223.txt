#include <iostream>
using namespace std;

int main()
{
    int a;
    cin>>a;
    string str;
    int count=0;
    while(a--)
    {
    	count=0;
    	cin>>str;
    	for(int i=0;i<str.length();i++)
    	{
    		if(str[i]==str[i+1])
    		{
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
    
    return 0;
}
