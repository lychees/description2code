#include <stdio.h>
#define mod 1000000007
   unsigned long  int  result[1000000]={0};
int main()
{
	
    unsigned long  int test,N,n,i,a,b,t;
    scanf("%lu",&test);
   
    
  result[1]=2;
  result[2]=5;
 while(test--)
  {
    scanf("%lu",&N);
     b=3;
    a=2;
   
  i=3;
   while(i<=N&&result[N]==0)
   {  t=a+b;
      a=b;
     t=t%mod;
     b=t;
   
     result[i]=result[i-1]+t;
 
    i++;
  }
   printf("%lu\n",result[N]%mod);
  
  }
return 0;
}
