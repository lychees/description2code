#define ll int
using namespace std;
int mat[102][102];
int coinm[102][102];
int n;
ll dfs(ll x,ll y)
{
       if(x>=n||y>=n)
		return 0;
		if(coinm[x][y]!=-1)
		return coinm[x][y];
	 return coinm[x][y]=mat[x][y]+max(dfs(x+1,y),dfs(x,y+1));
}
int main()
{
	int  t;
scanf("%d",&t);
while(t--)
{
	int c;
	cin>>n>>c;
	ll i,j;
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
			cin>>mat[i][j],coinm[i][j]=-1;
	}
    ll ans=dfs(0,0)-c;
   /* cout<<"dp is "<<endl;
    for(i=0;i<n;i++)
    {
    	for(j=0;j<n;j++)
    		cout<<coinm[i][j]<<" ";
    	cout<<endl;
    }*/
    if(ans>=0)
    	printf("%lld\n",ans);
    else
    	printf("-1\n");
}
return 0;
}
