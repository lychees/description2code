#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	int a,b;
    	bool flag = true;
    	int i = 0;
    	cin>>a>>b;
    	while(b>0 && flag) {
    		if(a == 0)
    			flag = false;
    			i += 2;
    			if(i%6 == 0){
    				a += 2;
    				i = 0;
    			}
    			a -= 1;
    			b -= 2;
    	}
    	if(flag)
    		cout<<"Yes"<<endl;
    	else
    		cout<<"No"<<endl;
    }
    return 0;
}
