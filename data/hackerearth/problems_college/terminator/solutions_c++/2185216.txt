#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	for(int i=0;i<s.length();i++)
    	{
    		if(s[i]=='x' and s[i-1]=='o')
    		{
    			s.erase(s.begin()+i-1);
    			i-=2;
    		}
    	}
    	for(int j=0;j<s.length();j++)
    	{
    		if(s[j]=='x' and s[j+1]=='o')
    		{
    			s.erase(s.begin()+j+1);
    			j--;
    		}
    	}
    	cout<<s<<endl;
    }
    return 0;
}
