

    #include <bits/stdc++.h>
    using namespace std;
    #define ll long long int
    int main()
    {
    	int test;
    	cin>>test;
    	while(test--)
    	{
    		ll n,power=1;
    		cin>>n;
    		if(n==1)
    			cout<<0<<" "<<1<<endl;
    		else{
    			//Row and Column starts from 0-index
    			n--;
    			for(int i=1;;i++)
    			{
    				if(power*2>n)
    					break;
    				power*=2;
    			}
    			ll max_xor=power^(power-1);
    			if(power==n)
    				cout<<max_xor<<" "<<2<<endl;
    			else{
    					ll count= 2*(n-power+1);
    					cout<<max_xor<<" "<<count<<endl;
    			}
    		}
    	}
    	return 0;
    }
