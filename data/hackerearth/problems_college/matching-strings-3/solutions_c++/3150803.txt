#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t , la ,lb;
    cin >> t ;
    while(t--){
    	string a , b ;
    	int count = 0 ;
    	cin >> a >> b ;
    	la = a.size();
    	lb = b.size();
    	for(int i = 0 ; i < la && i < lb ; i++){
    		if(a[i] == b[i]){
    			count++ ;
    		}
    	}	cout << count << endl ;
    }
    return 0;
}
