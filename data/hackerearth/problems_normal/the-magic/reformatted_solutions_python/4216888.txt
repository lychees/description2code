T = int(raw_input())
for t in range(T):
    N = int(raw_input())
    collect = 0
    sumi = 1
    count = 0
    num = N

    while (collect < N):

        target = (num//2)*2
        if (target == 0):
            count += 1
            break

        while sumi <= target:

            sumi = sumi << 1
        sumi = sumi >> 1

        if (collect < N):
            collect += sumi
            num = N-collect
            sumi = 1
            count += 1
    print count
