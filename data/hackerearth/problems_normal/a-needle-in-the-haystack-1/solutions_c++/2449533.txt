#include <iostream>
#include<string>
#include<cstdio>
#define ll long long int
using namespace std;


bool ispatterni(string a,string b)
{
	ll i,lena,lenb,j,k;
	lena=a.length();
	lenb=b.length();
	bool flag;
	if(lena>lenb)
	return false;
	
    ll hasha[26],hashb[26];
   	
    for(i=0;i<26;i++)
    {
    	hasha[i]=0;
    	hashb[i]=0;
    }
    
    for(i=0;i<lena;i++)
    hasha[a[i]-'a']++;
    
    for(i=0;i<lena;i++)
    hashb[b[i]-'a']++;
    
    
    for(i=0;i<lenb-lena+1;i++)
    {
    	j=i+lena;
    	flag=true;
    	for(k=0;k<26;k++)
    	{
    		if(hasha[k]!=hashb[k])
    		{
    			flag=false;
    		}
    	}
    	
    	if(flag)
    	return true;
    	
    	hashb[b[i]-'a']--;
    	if(j<lenb)
    	hashb[b[j]-'a']++;
    	
    }
    
    
    return false;
    
    
}

int main()
{
   ll t,i,j;
   scanf("%lld",&t);
   while(t--)
   {
   	string a,b;
   	cin>>a>>b;
   	
    if(ispatterni(a,b))
    {
    	cout<<"YES\n";
    }
    else
    {
    	cout<<"NO\n";
    }
   	
   	
   }
    return 0;
}
