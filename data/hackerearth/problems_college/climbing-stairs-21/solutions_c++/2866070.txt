#include <cstdlib>
#include <iostream>

using namespace std;
long long I,T,n,i;
int main(int argc, char *argv[])
{   cin>>T;
    for(I=1;I<=T;I++)
    {   cin>>n;
        long long  a[n];
        a[1]=1;
        a[2]=2;
        for(i=3;i<=n;i++)
           a[i]=a[i-1]+a[i-2];
        cout<<a[n]<<endl;
                        
    }
    
    return EXIT_SUCCESS;
}
