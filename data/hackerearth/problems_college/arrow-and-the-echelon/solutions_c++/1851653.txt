#include <iostream>
using namespace std;

int main()
{
    int t;
    long x[20];
    cin>>t;
    long int i,j,k,l;
    
    for(i=0;i<t;i++)
    cin>>x[i];
    
    int check=0;
    
    for(k=0;k<t;k++)
    {
      for(i=0;i<=x[k]/5;i++)
    	{
    		check=0;
    		
    		for(j=0;j<=x[k]/3;j++)
    		{
    			if(x[k]==(3*j+5*i))
    			{
    				check=1;
    				break;
    			}
    			
    		}
    	    if(check==1)
    			break;
    	}
    if(check==1)
    {
    	for(l=0;l<3*j;l++)
    	cout<<5;
    	for(l=0;l<5*i;l++)
    	cout<<3;
    	cout<<endl;
    }
    
    else 
    cout<<-1<<endl;
    
    }
    
    return 0;
}
