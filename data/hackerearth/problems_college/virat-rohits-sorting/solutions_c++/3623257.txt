#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int main()
{
    int n,i;
    ll c,r;
    map<ll, vector<ll> >a;
    map<ll, vector<ll> >::iterator it;
    vector<ll>v;
    vector<ll>::iterator ft;
    pair<ll, vector<ll> >mp;
    scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%lld%lld",&c,&r);
    a[c].push_back(r);
  }
  for(it=a.begin();it!=a.end();it++)
  {
      mp=*it;
      c=mp.first;
      v=mp.second;
      r=v.size();
      sort(v.rbegin(),v.rend());
      for(ft=v.begin();ft!=v.end();ft++)
         cout<<c<<' '<<*ft<<'\n';
  }
  return 0;
}
