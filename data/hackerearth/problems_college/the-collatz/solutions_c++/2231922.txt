#include <bits/stdc++.h>
using namespace std;
typedef unsigned long long int ll;
int main()
{
	ll T;
	cin>>T;
	while(T--)
	{
		ll N;
		cin>>N;
		ll sumEven = 0, sumOdd = 1, steps = 0;
		while(N != 1)
		{
			if(N%2 == 0)
			{
				sumEven += N;
				N = N / 2;
				steps++;
			}
			else
			{
				sumOdd += N;
				N = (3 * N)  + 1;
				steps++;
			}
		}
		if((sumEven%steps) > (sumOdd%steps))
			cout<<"Even Rules"<<endl;
		else if((sumEven%steps) < (sumOdd%steps))
			cout<<"Odd Rules"<<endl;
		else
			cout<<"Tie"<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
