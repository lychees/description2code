bool isprime(long long int x)
    {
    int i=0;
    for(i=2;i<=sqrt(x);i++)
        {
        if(x%i==0)
            return 0;
    }
    return 1;
}
int main() {
    int t,n;
    int i;
    
        cin>>n;
        switch(n)
            {
            case 1:cout<<2<<endl;break;
            case 2:cout<<3<<endl;break;
            case 3:cout<<5<<endl;break;
            case 4:cout<<7<<endl;break;
            default:{
                   
                n=n-4;
                  for(i=2;;i++)
                      {
                      if(isprime(6*i-1))
                          {
                          n--;
                          if(n==0)
                              {
                              cout<<6*i-1<<endl;
                              break;
                          }
                      }
                      if(isprime(6*i+1))
                          {
                          n--;
                          if(n==0)
                              {
                              cout<<6*i+1<<endl;
                              break;
                          }
                      }
                      
                  }
                
            }
        }
        
   
    return 0;
}
