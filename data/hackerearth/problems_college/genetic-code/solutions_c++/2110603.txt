#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
	long long int i,count=0,k=0;
	string a;
	cin>>a;
	for(i=0;i<a.length()-1;i++)
	{
		if(a[i]==a[i+1])
		count++;
		else
		{
			if(count%2==1)
			k++;
			count=0;
		}
	}
	if(count%2==1)
	k++;
	printf("%lld",k);
}
