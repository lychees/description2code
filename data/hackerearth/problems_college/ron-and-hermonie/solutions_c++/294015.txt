#include<iostream>
#include<cstdio>
#include<math.h>
using namespace std;


int Last_Digit(long long int a,long long int b)
{
	int m[6][4] = { { 6, 2, 4, 8 }, { 1, 3, 9, 7 }, { 6, 4 }, { 1, 7, 9, 3 }, { 6, 8, 4, 2 }, { 1, 9 }};

	switch (a % 10){
	case 0:return 0;
	case 1:	return 1;
	case 5:return 5;
	case 6:return 6;
	case 2:return m[0][b % 4];
	case 3:return m[1][b % 4];
	case 4:return m[2][b % 2];
	case 7:return m[3][b % 4];
	case 8: return m[4][b % 4];
	case 9:return m[5][b % 2];
	};

	if (b == 0)return 1;
	
	return -1;
}


int main()
{
	long long int T = 0, B = 0, E = 0;
	scanf("%ld",&T);
	while (T--)
	{
		scanf("%ld %ld",&B,&E);
		printf("%ld",Last_Digit(B,E));
		printf("\n");}
	return 0;
}
