#include <iostream>

using namespace std;
 
int main()
{
    int t,n,m;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	int arr[n*m][3];
    	for(int i=0;i<n*m;i++)
    		cin>>arr[i][0]>>arr[i][1]>>arr[i][2];
    	
    	for(int i=0;i<n;i++)
    	{
            double priceArr[m];
    		int k=0;
    		for(int j=i*m;j<(i+1)*m;j++)
    		{
    			if(arr[j][0]==100||arr[j][1]==100||arr[j][2]==100)
    			     priceArr[k]=0;
    			else
    			{
    				double price=100000;
    				price = price - price*((double)arr[j][0]/(double)100);
    				if(price!=0)
    				{
    					price = price - price*((double)arr[j][1]/(double)100);
    					if(price!=0)
    					{
    						price = price - price*((double)arr[j][2]/(double)100);
    						priceArr[k]=price;
    					}
    					else
    						priceArr[k]=0;
    				}
    				else
    					priceArr[k]=0;
    			}
    			k++;
    		}
    		int minIndex = 0;
    		double minValue = priceArr[0];
    		for(int j=1;j<m;j++)
    		{
                if(priceArr[j]<minValue)
                {
    				minValue=priceArr[j];
    				minIndex=j;
    			}
    		}
    		cout<<minIndex+1<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
