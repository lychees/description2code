#include<bits/stdc++.h>
using namespace std;

int ans[10000001];
void sieve()
{   int d;
    for (int i=2; i<=10000005; i++)
    {
        if(ans[i]==0){

            for(int j=i; j<=10000005; j+=i){
                ans[j]++;
                d = j/i;
                while(d%i==0){
                    ans[j]++;
                    d = d/i;
                }
            }
        }
    }
    for(int i=1; i<=10000000; i++){
        ans[i] = ans[i] + ans[i-1];
    }
}


int main()
{
    sieve();
    int t,n;
    scanf("%d", &t);
    while(t--){
        scanf("%d", &n);
        cout<<ans[n]<<"\n";
    }
    return 0;
}
