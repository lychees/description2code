#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const ll N=1e7+6;

ll f(ll n) {
  return (n*(n+1))/2;
}
void solve(ll l,ll r) {
  ll lo=1,hi=N;
  ll lb;
  while(lo<=hi) {
    ll mi=(lo+hi)/2;
    if(f(mi)>=l) {
      lb=mi;
      hi=mi-1;
    } else {
      lo=mi+1;
    }
  }
  ll ub;
  lo=1,hi=N;
  while(lo<=hi) {
    ll mi=(lo+hi)/2;
    if(f(mi)>=r) {
      ub=mi;
      hi=mi-1;
    } else {
      lo=mi+1;
    }
  }
  ll ans=0;
  if(lb==ub) {
    ll lbb=lb-1;
    ll lbc=((lbb)*(lbb+1))/2;
    ll rem=(l-lbc)%2;
    bool flag=false;
    if(rem==0) {
      flag=true;
    }
    ll diff=(r-l+1);
    if(diff%2==1) {
      if(flag==true) {
        ans=-lb;
      } else {
        ans=lb;
      }
    }
  } else {
    ll lbb=lb-1;
    ll lbc=((lbb)*(lbb+1))/2;
    ll lbcc=((lb)*(lb+1))/2;
    ll rem1=(l-1-lbc)%2;
    ll rem2=(lbcc-lbc)%2;
    if(rem1==1) {
      rem1=lb;
    }
    if(rem2==1) {
      rem2=lb;
    }
    ans+=(rem2-rem1);
    ll rbb=ub-1;
    ll rbc=((rbb)*(rbb+1))/2;
    ll rem=(r-rbc)%2;
    if(rem==1) {
      ans+=ub;
    }
    for(int i=lb+1;i<=ub-1;i++) {
      if(i%2==1) {
        ans+=i;
      }
    }
  }
  printf("%lld\012",ans);
  return;
}
int main() {
  ll tt;
  scanf("%lld",&tt);
  while(tt--) {
    ll l,r;
    scanf("%lld%lld",&l,&r);
    if(l>r) {
    	swap(l,r);
    }
    solve(l,r);
  }
  return 0;
}
