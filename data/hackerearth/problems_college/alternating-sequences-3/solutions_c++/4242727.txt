#include<bits/stdc++.h>
#define ll long long
#define maxe 2147483647
#define INF 1e16
#define si(x) scanf("%d",&x)
#define sc(x) scanf("%c",&x)
#define sl(x) scanf("%ld",&x)
#define sll(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define pi(x) printf("%d\n",x)
#define pl(x) printf("%ld\n",x)
#define pll(x) printf("%lld\n",x)
#define ps(x) printf("%s\n",x)
#define iter(i,a,b) for(int i=a; i<b; i++)
#define ited(i,a,b) for(int i=a;i>=b;i--)
#define mp make_pair 
#define pb push_back 
#define max(a,b) a>b?a:b
#define min(a,b) a<b?a:b
//#define pop pop_back 
#define mod 1000000007
using namespace std;
int main()
{
	int n;
	si(n);
	int a[n+1],sgn[n+1];
	iter(i,1,n+1)
	{
		si(a[i]);
		if(a[i]<0)
		{
			a[i]=-a[i];
			sgn[i]=1;
		}
		else
			sgn[i]=0;
	}
	int dp[5001][2]={0};
	iter(i,1,n+1)
	{
		dp[i][0]=1-sgn[i];
		dp[i][1]=sgn[i];
		iter(j,1,i)
		{
			if(abs(sgn[i]-sgn[j])==1 && a[i]>a[j])
			{
				dp[i][sgn[i]]=max(dp[j][1-sgn[i]]+1,dp[i][sgn[i]]);
			}
		}
	}
	int an=0;
	iter(i,1,n+1)
		iter(j,0,2)
			an=max(an,dp[i][j]);
	pi(an);
	return 0;
}
