typedef unsigned long long int uint64;

int64 conv(int64 a,char x,int64 b){
	if(x=='-')
	return a-b;
	if(x=='+')
	return a+b;
	
	return a*b;
}
int main(){
	int t;
	cin>>t;
	string s;
	while(t--){
		cin>>s;
		reverse(all(s));
		int64 ans=int(s[0])-48;
		for(int i=1;i<s.length();i+=2){
			int val=int(s[i+1])-48;
			ans=conv(ans,s[i],val);
		}
		cout<<ans<<endl;
	}
	return 0;
}
