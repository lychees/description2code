#include <bits/stdc++.h>

using namespace std;

int main()
{
	long long a[15];
	int i;
	int c,v;
	a[0] = 1;
	a[1] = 1;
	for (i = 2; i <= 13; i++) {
		a[i] = i * a[i-1];
	}
	int t;
	scanf("%d",&t);
	while (t --) {
		scanf("%d %d",&c,&v);
		long long ans = a[c+v];
		long long p = a[7]/(a[c] * a[7-c]);
		ans *= p;
		p = a[5]/(a[v] * a[5-v]);
		//p = p / a[5-v];
		ans *= p;
		printf("%lld\n",ans);
	}
	
	return 0;
}
