#include <bits/stdc++.h>
using namespace std;

int main()
{
    long k,i,tmp;
    cin>>k;
    priority_queue<long> poke;
    
    for(i=0;i<k;++i)
    {
    	cin>>tmp;
    	poke.push(tmp);
    }
    long max=0,days=poke.top();
    
    for(i=0;i<k;++i)
    {
    	if(max < poke.top())
    	{
    		max = poke.top();
    		days++;
    	}
    	max--;
    	poke.pop();
    }
    cout<<days+1<<endl;
    return 0;
}
