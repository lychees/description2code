first = 0
second = 0
third = 0
n = input()
list = [j for j in raw_input().split()]
for i in range(0, n):
    curr = list[i]
    if i == 0:
        first = curr
    else:
        if int(curr) >= int(first):
            third = second
            second = first
            first = curr
        else:
            if int(curr) < int(first) and int(curr) >= int(second):
                third = second
                second = curr
            else:
                if int(curr) < int(second) and int(curr) >= int(third):
                    third = curr
    if i < 2:
        print(-1)
    else:
        ans = int(first)*int(second)*int(third)
        print(ans)
        ans = 0
