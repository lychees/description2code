#include<bits/stdc++.h>
#define ll long long
#define maxn 100010
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		ll n, m, l, r, c, i;
		ll l1[maxn], r1[maxn], c1[maxn];
		scanf("%lld%lld",&n,&m);
		for( i = 1 ; i <= m ; i++ )
		{
			scanf("%lld%lld%lld",&l1[i],&r1[i],&c1[i]);
		}
		ll q, times_q[maxn], x, y;
		memset(times_q, 0, sizeof times_q);
		scanf("%lld", &q);
		for( i = 1 ; i <= q ; i++ )
		{
			scanf("%lld%lld",&x, &y);
			times_q[x]++;
			times_q[y + 1]--;
		}
		for( i = 1 ; i <= m + 1 ; i++ )
		{
			times_q[i] += times_q[i - 1];
		}
		ll ans[maxn];
		memset(ans, 0, sizeof ans);
		for( i = 1 ; i <= m ; i++ )
		{
			ans[l1[i]] += c1[i]*times_q[i];
			ans[r1[i] + 1] -= c1[i]*times_q[i];
		}
		for( i = 1 ; i <= n ; i++ )
		{
			ans[i] += ans[i - 1];
		}
		for( i = 1 ; i <= n ; i++ )
		{
			printf("%lld ",ans[i]);
		}
		printf("\n");
	}
	return 0;
}
