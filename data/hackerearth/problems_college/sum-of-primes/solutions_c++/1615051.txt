#include <bits/stdc++.h>
using namespace std ;
#define MAXN 1000001

bool P[MAXN] ;
long long arr[MAXN];
void sieve(){
   // int c = 0 ;
  arr[0]=arr[1]=0;
    for(int i=2;i<MAXN;i++){
        if(!P[i]){
        	arr[i]=arr[i-1]+i;
        	//cout<<arr[i]<<endl;
            //c ++ ;
            int j = i ;
            while(j < MAXN){
            //    adj[j].push_back(i) ;
                P[j] = 1 ;
                j += i ;
            }
        }
        else
        arr[i]=arr[i-1];
     //   cout<<arr[i]<<endl;
    }
}
int main()
{
	sieve();
  int t;
  cin>>t;
  while(t--)
  {
  	long long int n,m;
  	cin>>n>>m;
  	long long int p=arr[m]-arr[n-1];
  	if(p==0)
  	cout<<"NO"<<endl;
  	else
  	{
  long long int s=sqrt(p);
  	int flag=0;
  	for(int i=2;i<=s;i++)
  	{
  		if(p%i==0)
  		{
  			flag=1;
  			break;
  		}
  	}
  	if(flag)
  	cout<<"NO"<<endl;
  	else
  	cout<<"YES"<<endl;
  }
  }
    return 0;
}
