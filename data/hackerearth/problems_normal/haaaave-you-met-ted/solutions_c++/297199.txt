#include <iostream>
using namespace std;

unsigned int countbits(unsigned int y)
{
	int count=0;
	
	for(;y;y = y>>1)
	{
		if(y & 1)
		{
			count++;
		}
	}
	
	return count;
		
}

int main() 
{
	// your code goes here
	int t,n,i;
	unsigned int ui[100001],x;
	unsigned int min = 0;
	cin >> t;
	while(t--)
	{
		cin >> n;
		for(  i = 0;i < n; i++ )
			cin >> ui[i];
		if(n == 1)
		{
			cout << "1" << '\n';
		}
		else
		{
			min = ~0;
			for(i = 0 ;i<n;i++)
			{	
			//	cout << "UI:" << ui[i] << '\n';  
				x = countbits(ui[i]);
				//cout << "X:" << x << '\n';
				if( x == 1)
				{
					cout << x << '\n';
					break;
				}
				else if( x < min )
					min = x;
			}
			if(x == 1)
				continue;
			else
				cout << min << '\n';
		}
	}
	return 0;
}
