
min_fact_cnt = [0]*1000001
primes = [True]*1000001


def sieve():
    for i in range(2, 1000001):
        if primes[i]:
            j = 2*i
            min_fact_cnt[i] = min_fact_cnt[i]+1
            while j < 1000001:
                if primes[j]:
                    min_fact_cnt[i] = min_fact_cnt[i]+1
                primes[j] = False
                j = j+i


sieve()
t = int(raw_input())
for i in range(t):
    p = int(raw_input())
    # print p
    print min_fact_cnt[p]
