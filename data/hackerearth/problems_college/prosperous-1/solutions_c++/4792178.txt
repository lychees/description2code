#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;
int main() {
	int T;
	scanf("%d",&T);
	long A[100009][10][2];
	for(int i = 1; i <= T; ++i){
		long S,D;
		long long sol = 0;
		scanf("%ld %ld",&S,&D);
		for(long j = 0; j <= S; ++j){
			for(int k = 0; k < 10; ++k){
				A[j][k][1] = 0;
				A[j][k][0] = 0;
			}
		}
		for(int j = 1; j <= 9; ++j)	A[j][j][0] = 1;

		for(long j = 1; j <= S; ++j){
			for(int k = 0; k <= D; ++k){
				for(int l = 1; l+k < 10; ++l){
					A[j+k+l][l+k][1] = (A[j+k+l][l+k][1] + A[j][l][1])%1000000007;
					if(k == D){
						A[j+k+l][l+k][1] = (A[j+k+l][l+k][1] + A[j][l][0])%1000000007;
					}
					else{
						A[j+k+l][l+k][0] = (A[j+k+l][l+k][0] + A[j][l][0])%1000000007;
					}
				}
			}
		}

		for(int l = 0; l < 10; ++l){
			sol = (sol + A[S][l][1])%1000000007;
			if(D == 0) sol = (sol + A[S][l][0])%1000000007;
		}
		printf("Case #%d: %lld\n",i,sol);
	}
	return 0;
}
