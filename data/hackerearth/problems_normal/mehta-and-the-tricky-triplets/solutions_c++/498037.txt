int data[100009];
ll cat[20];
int main()
{
	int n;
	s(n);
	ll x;
	fill(cat,0);
	for(int i=0;i<n;i++)
	{
		sl(x);
		data[i] = 0;
		while(x > 0)
		{
			if(x%10 == 2)
			{
				data[i] = setbit(data[i],0);
			}
			else if(x%10 == 3)
			{
				data[i] = setbit(data[i],1);
			}
			else if(x%10 == 5)
			{
				data[i] = setbit(data[i],2);
			}
			else if(x%10 == 7)
			{
				data[i] = setbit(data[i],3);
			}
			x = x/10;
		}

		for(int j=1;j < (16);j++)
		{
			if((j&data[i]) == j)
			{
				cat[j]++;
			}
		}
	}

	ll ans = 0;

	for(int j=1;j < (16);j++)
	{
		if(bitcount(j)%2 == 1)
			ans += (cat[j]*(cat[j]-1)*(cat[j]-2))/6;
		else
			ans -= (cat[j]*(cat[j]-1)*(cat[j]-2))/6;
	}
	cout<<ans<<endl;
}
