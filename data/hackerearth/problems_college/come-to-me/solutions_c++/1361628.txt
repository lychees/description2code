#include<bits/stdc++.h>
#define ll long long
#define ld double
#define f first
#define s second
#define pb push_back
#define M 1000000007
#define MX 1000003
#define INF 10000000000
#define PI 3.141593
#define mmx(a,b) a>b?a:b
#define mmn(a,b) a<b?a:b
using namespace std;
typedef pair<ll,ll> ii;
typedef vector<ll> vv;
ll arr[10004][30];
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
        string s;
        ll ans,n;
        cin>>s;
        n=s.size();
        memset(arr,0,sizeof(arr));
        arr[0][s[0]-'A']=1;
        for(ll i=1;i<n;i++)
        {
            for(ll j=0;j<26;j++)
                arr[i][j]=arr[i-1][j];
            arr[i][s[i]-'A']++;
        }
        for(ll sz=min(26LL,n);sz;sz--)
        {
            ll f=0;
            for(ll i=0;i<=n-sz;i++)
            {
                ll last=i+sz-1,fl=0;
                for(ll k=0;k<26;k++)
                {
                    ll x=arr[last][k];
                    if(i)
                        x-=arr[i-1][k];
                    if(x>1)
                    {
                        fl=1;
                        break;
                    }
                }
                if(!fl)
                {
                    f=1;
                    break;
                }
            }
            if(f)
            {
                ans=sz;
                break;
            }
        }
        cout<<ans<<"\n";
    }
}
