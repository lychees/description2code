const int  maxn  = 5*100004;

const int INF  = 1000000008;

ll arr[maxn],Next[maxn],Prev[maxn];

inline void solve(void){
    int T;
    cin >> T;
    while(T--){
    	int N;
    	cin >> N;
	for(int i=1;i<=N;i++)
		cin >> arr[i];
	vector<pair<ll,ll> > A(N);
	for(int i=1;i<=N;i++){
		A[i-1].first = arr[i];
		A[i-1].second = i;
	}
	sort(A.begin(),A.end());
	for(int i=1;i<=N;i++){
		Next[i] = i+1;
		Prev[i] = i-1;
	}	
	long long int ans = 0;
	for(int i=0;i<N-1;i++){

		int idx = A[i].second;

		int p1 = Prev[idx];
		int p2 = Next[idx]; 
		ll m = INF;
		if(p1 > 0){
			m = min(arr[p1] , m);
		}
		if(p2<=N) m= min(arr[p2] , m);

		ans = ans + m;

		Next[Prev[idx]] = Next[idx];
		Prev[Next[idx]] = Prev[idx];
	}
	cout << ans << endl;
}
	return ;
	}   
int main(int argc, const char * argv[]){
    ios_base::sync_with_stdio(false);
    solve();
    return 0;
}
