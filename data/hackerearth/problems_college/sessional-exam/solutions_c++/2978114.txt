#include <bits/stdc++.h>
using namespace std;
int main() {
	int n,m;
	cin>>n>>m;
	int subjects[n][m],c=0,d=0;
	int max[m];
	for(int i=0;i<m;i++){
		max[i]=0;
	}
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			cin>>subjects[i][j];
			if(max[j]<subjects[i][j]){
				max[j]=subjects[i][j];
				d=1;
			}
		}
		if(d==1){
			c++;
			d=0;
		}
	}
	cout<<c;
	return 0;
}
