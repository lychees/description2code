#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,j,i,l;
    char s[1201];
    scanf("%d\n",&t);
    while(t--){
    	scanf("%d\n",&n);
    	char c[26];
    	int a[26]={0};
    	for(i=0;i<n;i++)
    		scanf("%c %d\n",&c[i],&a[i]);    
    	for(i=0;i<n;i++){
    		for(j=i+1;j<n;j++){
    			if(a[j]<a[i] || ((a[i]==a[j])&&c[i]>c[j])){
    				int tc = a[i];
    				a[i] = a[j];
    				a[j]= tc;
    				char ch = c[i];
    				c[i] = c[j];
    				c[j] = ch;
    			}
    		}
    	}
    	scanf("%s\n",&s);
    	l = strlen(s);
    	int k=1;
    	j=0;
    	for(i=0;i<l;i++){
    		k++;
			if(s[i]=='0'){
				printf("%c",c[i-j]);
				j = i+1;
				k=1;
			}
			if(k==n){
				printf("%c",c[n-1]);    		
				j=i+1;
				k=1;
			}
    	}
    	printf("\n");
    }
    return 0;
}
