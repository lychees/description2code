#include <iostream>
using namespace std;

int main()
{
    int t,i,j,a[27],x,temp;
    char temp2;
    cin>>t;
    string s;
    while(t)
    {
    	cin>>s;
    	char p[] = {'a','b','c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','v','w','x','y','z'};
    	for(i=0;i<26;i++)
    	{
    		a[i] = 0;
    	}
    	a[27]=-1;
    	for(i=0;i<s.size();i++)
    	{
    		a[s[i]-'a']++;
    	}
    	
    	for(i=0;i<26;i++)
    	{
    		for(j=0;j<26-i-1;j++)
    		{
    			if(a[j]>a[j+1])
    			{
    				temp = a[j];
    				temp2 = p[j];
    				a[j] = a[j+1];
    				p[j] = p[j+1];
    				a[j+1] = temp;
    				p[j+1] = temp2;
				}
			}
    	}
    
    	
    	for(i=0;i<27;i=j)
    	{
    		for(j=i;j<27;j++)
    		{
    			if(a[i]!=a[j])
    			{
    				for(x=j-1;x>=i;x--)
					{
						cout<<p[x]<<" ";
					}
					//i=j;
					break;	
				}
			}
		}
 
    	cout<<"\n";
    	
    	t--;
    }
    
    return 0;
}
