#include <cstdio>
#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a[n],i;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		int count=1;
		int val = a[n-1];
		for(i=n-2;i>=0;i--)
		{
			if(a[i] > val)
			{
				count++;
				val = a[i];
			}
		}
		cout<<count<<"\n";
	}
    //printf("Hello World!\n");
    return 0;
}
