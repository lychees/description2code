#include <bits/stdc++.h>
using namespace std;

int main(int agrc, char** argv) {
	
	int N = 0;
	cin >> N;
	map<string, vector<int> > m;
	for(int i = 0; i < N; ++i) {
		string s;
		cin >> s;
		for(int j = 0; j < s.size(); ++j) {
			string temp = "";
			for(int k = j; k < s.size(); ++k) {
				temp.push_back(s[k]);
				m[temp].push_back(i);
			}
		}
	}
	
	for(auto it = m.begin(); it != m.end(); ++it) {
		auto sz = unique((it->second).begin(), (it->second).end());
		(it->second).resize(distance((it->second).begin(), sz));
	}
	
	int Q = 0;
	cin >> Q;
	while(Q--) {
		int l = 0, r = 0;
		string str;
		cin >> l >> r >> str;
		--l;
		--r;
		int ans = 0, ans1 = 0, ans2 = 0;
		if(str.empty())
			ans = r - l + 1;
		else {
			ans1 = upper_bound(m[str].begin(), m[str].end(), r) - m[str].begin();
			ans2 = lower_bound(m[str].begin(), m[str].end(), l) - m[str].begin();
			ans = abs(ans2 - ans1);
		}
		cout << ans << endl;
	}
	
	return 0;
}
