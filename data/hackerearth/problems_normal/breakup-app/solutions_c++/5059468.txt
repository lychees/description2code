#include<bits/stdc++.h>
using namespace std;

int main(){
    int n;
    cin>>n;
    cin.ignore();

    string s;
    int hsh[31]={0};

    for(int j=0;j<n;j++){
        s.clear();
        getline(cin,s);

        int i=0;

        if(s[0]=='G'){
            while(i<s.size()){
                if(i+1<s.size())
                    if((s[i]>'0' && s[i]<'3') && (s[i+1]>='0' && s[i+1]<='9'))
                        hsh[(s[i]-'0')*10 + (s[i+1]-'0')]+=2;

                else if(s[i]>'0' && s[i]<='9')
                    hsh[s[i]-'0']+=2;

                i++;
            }
        }

        else if(s[0]=='M'){
            while(i<s.size()){
                if(i+1<s.size())
                    if((s[i]>'0' && s[i]<'3') && (s[i+1]>='0' && s[i+1]<='9'))
                        hsh[(s[i]-'0')*10 + (s[i+1]-'0')]++;

                    else if(s[i]>'0' && s[i]<='9')
                        hsh[s[i]-'0']++;

                    i++;
            }
        }
    }

    int maxx=INT_MIN, max_p=0;

    for(int i=1;i<31;i++){
        if(hsh[i]>maxx){
            maxx=hsh[i];
            max_p=i;
        }
    }

    if(max_p==19 || max_p==20)
        cout<<"Date"<<endl;
    else
        cout<<"No Date"<<endl;

    return 0;
}
