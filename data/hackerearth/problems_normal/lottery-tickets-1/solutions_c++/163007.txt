#include<bits/stdc++.h>
#define F first
#define S second
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
	int n,i,j,val[109],ans=0;
	vector < pair < int , pair < int, int >  > > arr;
	cin >> n;
	arr.resize(n);
	for(i=0; i<n; i++)
	    cin >> arr[i].S.F >> arr[i].F >> arr[i].S.S ;
	sort(arr.begin(),arr.end());
	for(i=0; i<n; i++)
	{
	    val[i]=arr[i].S.S;
	    for(j=0; j<i; j++)
		if(arr[i].S.S+val[j]>val[i])
		{
		    if(arr[j].F<=arr[i].S.F)
			val[i]=val[j]+arr[i].S.S;
		}
	    if(val[i]>ans)ans=val[i];
	}
	printf("%.2lf\n",(double)ans/100);
    }
    return 0;
}
