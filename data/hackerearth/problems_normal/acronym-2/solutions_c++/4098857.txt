#include<iostream>
#include<string>
#include<cstring>
#include<cctype>
using namespace std;
int main()
{int k,n,count=0,len=0;
 string a[1000],b[1000];
 char word[100];
 cin>>k;
 for(int i=0;i<k;i++)
  cin>>a[i];
 cin>>n;
 for(int i=0;i<n;i++)
  cin>>b[i];
 for(int i=0;i<n;i++)
  {count=0;
   for(int j=0;j<k;j++)
     {if(b[i]==a[j])
       {count=1;
        break;
	   }
	 }
	if(count==0)
	 {if(b[i].at(0)>=97 && b[i].at(0)<=122)
	    {word[len]=b[i].at(0)-32;
	     len++;
		}
	   else
	    {word[len]=b[i].at(0);
	     len++;
		}	
	 } 
  }
  for(int i=0;i<len;i++)
   {if(i==len-1)
     cout<<word[i];
    else
	 cout<<word[i]<<"."; 
   } 
  return 0;  
}
