#include<bits/stdc++.h>
#define MOD 1000000007
using namespace std;
typedef long long int ll;
ll arange[101],c[101][101];

void calculate()
{
    ll i,j;
    arange[0]=1;arange[1]=0;
    for(i=2;i<=101;i++)
        arange[i]=((i-1)*(arange[i-1]+arange[i-2]))%MOD;

    for(i=0;i<101;i++)
    {
        for(j=0;j<=i;j++)
        {
            if(j==0||i==j)
                c[i][j]=1;
            else c[i][j]=(c[i-1][j]+c[i-1][j-1])%MOD;
        }
    }
}

ll cal_pow(ll a,ll b)
{
    ll ret=1;
    while(b)
    {
        if(b&1)ret=(ret*a)%MOD;
        b/=2;
        a=(a*a)%MOD;
    }

    return ret;
}



int main()
{
    ll num,i,l,r,p,q,ans,ways;

    calculate();

        cin>>num>>l>>r;
        p=0;q=0;
        for(i=0;i<=num;i++)
        {
            ways=(arange[i]*c[num][i])%MOD;
            if(i>=l&&i<=r)
            {
                p+=ways;
                p%=MOD;
            }
            q+=ways;
            q%=MOD;
        }

        ll m=cal_pow(q,MOD-2);
        ans=(p*m)%MOD;
        if(ans<0)
            ans+=MOD;

        ans%=MOD;
        cout<<ans<<endl;

    return 0;
}
