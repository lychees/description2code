for t in range(int(raw_input())):
    n = int(raw_input())
    e = 0
    o = 1
    c = 0
    while (n > 1):
        c += 1
        if (n % 2 == 0):
            e += n
            n = n/2
        else:
            o += n
            n = 3*n+1
    e = e % c
    o = o % c
    if (e > o):
        print "Even Rules"
    elif (e < o):
        print "Odd Rules"
    else:
        print "Tie"
