#include <iostream>
using namespace std;
int dp[1000005];
int sa[1000005];
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<1000005;i++){dp[i]=0;sa[i]=0;}
	for(int i=0;i<n;i++)
	{
		int s,q;
		cin>>s>>q;
		dp[s]=q;
	}
	sa[1000004]=dp[1000004];
	for(int i=1000003;i>=0;i--)
		sa[i]=max(sa[i+1],dp[i]);
	int g;
	cin>>g;
	long long int r=0;
	while(g>0)
	{
		g--;
		int m;
		cin>>m;
		r+=(long long int)((sa[m]/m)*100);
	}
	cout<<r<<endl;
    return 0;
}
