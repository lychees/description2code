t=int(raw_input())
for i in range(t+1):
	try:
		
		s=raw_input()
		j=len(s)
		count=0
		for k in range(j/2):
			if s[k]==s[j-1-k]:
				count+=1
			else:
				if s[k]==s[j-2-k]:
					c=j-1-k
				else:
					c=k
				break
		if count==j/2:
			print(-1);
		else:
			print(c);
	except (EOFError):
		break
