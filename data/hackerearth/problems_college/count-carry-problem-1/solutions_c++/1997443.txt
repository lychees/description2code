#include <iostream>
#include<cstring>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   scanf("%d", &t);
   while(t--){
   	char a[3], b[3];
   	scanf("%s %s", &a, &b);
   	int n=strlen(a);
   	int m=strlen(b);
   	int sum, carry=0, count=0;
   	m--;
   	n--;
  
   	while(m>=0 &&n>=0){
   		int sum=(a[n]-'0')+(b[m]-'0')+carry;
   	   
   		if(sum/10>0){
   			carry=sum/10;
   			count++;
   			m--;
   			n--;
   		}
   		else{
   		   m--;
   		   n--;
   		}
   	}
   	if(count>1){
   		printf("%d carry operations\n", count);
   	}
   	else if(count==1){
   		printf("%d carry operation\n", count);
   	}
   	else
   	   printf("No carry operation\n");
   }
    return 0;
}
