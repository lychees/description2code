#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<string>
#include<vector>
#include<map>
#include<stack>
#include<queue>
#include<algorithm>

using namespace std;

int main()
{
	int n;

	cin>>n;

	while(n--)
	{
		string a;
		int m[1000000] = {0};
		
		cin>>a;

		int last = a.length();

		for(int i=a.length()-1; i>=0; i--)
		{
			if(a[i] == 'a' || a[i] == 'z')
				last = i;

			m[i] = last;
		}

		int z = a.length();
		long long int ans = 0;

		for(int i=0; i<z; i++)
		{
			ans += z-m[i];
		}
		
		cout<<ans<<endl;
	}

	return 0;
}
