#include <iostream>
using namespace std;

int main()
{
    int x,y,cas,i,j,count;
    char s[1000][1000];
    cin >> cas;
    while(cas--)
    {
    	cin >> x >> y;
    	for(i=0;i<x;i++)
    	{
    		for(j=0;j<y;j++)
    		cin>>s[i][j];
    	}
    	 
    	count=0;
    	for(i=0;i<x-1;i++)
    	{
    		for(j=0;j<y-1;j++)
    		{
    			if(s[i][j]=='/'&&s[i][j+1]=='\\')
    			{
    				if(s[i+1][j]=='\\'&&s[i+1][j+1]=='/')
    				count++;
    			}
    		}
    	}
    	 
        cout << count << endl;
    }
    return 0;
}
