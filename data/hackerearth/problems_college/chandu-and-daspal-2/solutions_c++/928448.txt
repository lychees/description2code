#include<iostream>
#include<string.h>
using namespace std;
int gcdxy(int a,int b)
{
	int c;
  while ( a != 0 ) {
     c = a; a = b%a;  b = c;
  }
  return b;
}
int main()
{
	int gcd,t,x,y,turn;
	char str[10];
	cin>>t;
	while(t--)
	{
		cin>>x>>y>>str;
		if(!strcmp(str,"Chandu"))
		{
			turn=1;
		}
		else
		{
			turn=0;
		}
		
		while(x!=0&&y!=0)
		{
			gcd=gcdxy(x,y);
			x-=gcd;
			y-=gcd;
			turn=(turn+1)%2;
		}	
		if(turn == 1)
			cout<<"Daspal\n";
		else
			cout<<"Chandu\n";	
	}
	return 0;
}
