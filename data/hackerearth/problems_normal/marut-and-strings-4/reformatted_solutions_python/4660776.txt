# print 'Hello World!'

no = raw_input()
if no is not None and no.isdigit():
    no = int(no)
    if no <= 10 and no >= 1:
        for i in range(no):
            lowerCase = 0
            upperCase = 0
            count = 0
            inputStr = raw_input()
            inputStrArr = list(inputStr)
            if len(inputStrArr) >= 0 and len(inputStrArr) <= 100:
                for j in inputStrArr:
                    if j.isalpha():
                        count = count + 1
                        if j.islower():
                            lowerCase = lowerCase + 1
                        elif j.isupper():
                            upperCase = upperCase + 1
                if count == 0:
                    print "Invalid Input"
                else:
                    print min(lowerCase, upperCase)
            else:
                print "Invalid Input"
    else:
        print "Invalid Test"
else:
    print "Invalid Test"
