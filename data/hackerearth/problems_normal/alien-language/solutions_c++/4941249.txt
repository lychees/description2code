#include <iostream>
#include<cstring>
#include<map>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	string s1;
    	string s2;
    	cin>>s1;
    	cin>>s2;
    	map<char,int> mymap;
    	bool flag=false;
    	for(int i=0;i<s1.length();i++)
    	{
    		mymap[s1[i]-'a']++;
    	}
    	for(int i=0;i<s2.length();i++)
    	{
    		if(mymap.find(s2[i]-'a')!=mymap.end())
    		{
    			flag=true;
    			break;
    		}
    	}
    	if(flag)
    		cout<<"YES"<<endl;
    	else
    		cout<<"NO"<<endl;
    }
    return 0;
}
