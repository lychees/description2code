#include <iostream>
#include <map>
#include <vector>
#define pb push_back

using namespace std;

map <int, vector <int> > m;
vector <int>::iterator low, high;

int main()
{
    int n, q, i, x, y, val;

    cin >> n;

    for (i = 0; i < n; i++) {
        cin >> x;
        m[x].pb(i);
    }

    cin >> q;

    while (q--) {
        cin >> x >> y >> val;

        low = lower_bound(m[val].begin(), m[val].end(), x);
        high = lower_bound(m[val].begin(), m[val].end(), y + 1);

        cout << (high - low) << endl;
    }

    return 0;
}
