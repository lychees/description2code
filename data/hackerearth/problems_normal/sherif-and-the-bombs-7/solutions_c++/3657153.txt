int findpoint(int x, int y, int r)
{
	int bombpoints=0,i,j;
	double area=0;
	for (i=x-r;i<=x+r;i++)
		{for (j=y-r;j<=y+r;j++)
			{
			if(i>=1 && i<=1000 && j>=1 && j<=1000)
			{
				area=(i-x)*(i-x)+(j-y)*(j-y);
				if((r*r-area)>=0)
					{bombpoints++;
					bombSpots[i][j][1]=bombSpots[i][j][1]+1;
				//	cout<<"Bomb found";
					}
				//cout<<"Point\n"<<i<<"\t"<<j<<"\n";			
			}			
			}
		}	
			return bombpoints;
	
}
