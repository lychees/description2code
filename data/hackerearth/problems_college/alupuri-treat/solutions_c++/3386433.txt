#include<bits/stdc++.h>
using namespace std;
int main()
{
		int n; cin>>n; int arr[n][n]; int sum1=0,sum2=0,i,j;
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				cin>>arr[i][j];
			}
		}
		for(i=0;i<n;i++)
		{
			sum1+=arr[i][i];
			sum2+=arr[i][n-i-1];
		}
		cout<<abs(sum1-sum2);
}
