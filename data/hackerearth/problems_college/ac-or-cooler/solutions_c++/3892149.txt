#include <bits/stdc++.h>
using namespace std;

int main(){
    long n,m;
    cin>>n>>m;
    
    long a[2*n];
    for(long i=0; i<2*n; i++){
        cin>>a[i];
    }
    
    sort(a, a+2*n);
    
    long sum = 0;
    long count = 0;
    
    while(count!=n && sum+a[count]<=m){
        sum += a[count];
        count++;
    }
    
    cout<<count<<endl;
    return 0;
    
}
