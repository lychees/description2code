#include<bits/stdc++.h>
using namespace std;
int a[2010][2010]={0};
void fill(int n,int v,int i,int j)
{
	if(n==1)
	{
	 a[i][j]=v;
	 return;
	}
	if(!a[i][j])
	fill(n/2,v,i,j);
	if(!a[i][j+n/2])
	fill(n/2,v+(n*n)/4,i,j+n/2);
	if(!a[i+n/2][j])
	fill(n/2,v+(n*n)/2,i+n/2,j);
	if(!a[i+n/2][j+n/2])
	fill(n/2,v+(3*n*n)/4,i+n/2,j+n/2);
}
int main()
{
	int t,n,b[100],i,j;
	b[0]=1;
	b[1]=2;
	for(i=2;i<=10;i++)
	b[i]=2*b[i-1];
	cin>>n;
	fill(b[n],1,1,1);
	for(i=1;i<=b[n];i++,cout<<endl)
	for(j=1;j<=b[n];j++)
	cout<<a[i][j]<<" ";
	return(0);
}
