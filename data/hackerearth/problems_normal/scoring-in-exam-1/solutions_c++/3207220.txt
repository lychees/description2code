#include <iostream>
#include<algorithm>
#include<vector>
#include<utility>
using namespace std;

int main()
{
	int n,x,q;
	cin>>n>>q;
	vector<int>a;
	vector<int>b;
	for(int i=0;i<n;i++)
	{
		cin>>x;
		a.push_back(x);
	}
	for(int i=0;i<n;i++)
	{
		cin>>x;
		b.push_back(x);
	}
	vector<pair<int,int>>tim;
	
	for(int i=0;i<n;i++)
	{
		tim.push_back(make_pair(b[i],a[i]));
	}
	
	sort(tim.rbegin(),tim.rend());
	int t,total=0,i=0;
	while(q--)
	{
		cin>>t;
		while(t--)
		{
		total+=tim[i].second;
		i++;
		}
		cout<<total<<endl;
		i=0;
		total=0;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
