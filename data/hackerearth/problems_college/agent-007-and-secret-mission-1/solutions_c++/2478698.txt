#include <iostream>
#include<string>
#include<vector>
#include<algorithm>
#include<cstdio>

using namespace std;



int main()
{
    //cout << "Hello World!" << endl;
    long long int t,i,n,set;
    string str;
    cin>>t;
    while(t--)
    {
    	set=0;
    	cin>>str;
    	cin>>n;
    	for(i=0;i<str.length();++i)
    		if(str[i]=='1')
    			++set;
    	if((n%2==1 && ((n+1)/2)%2==1) || (n%2==0 && (n/2)%2==1))
    		set=str.length()-set;
    	cout<<set<<"\n";
    }
    return 0;
}
