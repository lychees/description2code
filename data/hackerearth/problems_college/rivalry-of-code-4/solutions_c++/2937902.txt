#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int tc;
    cin>>tc;
    while(tc--)
    {
    	string a,b,s;
    	cin>>a>>b;
    	if(a.length()>b.length())
    	{
    	   s=a;
    	   a=b;
    	   b=s;
    	}
    	int min=a.length();
    	int cnt;
    	for(int i=1;i<=b.length()-a.length()+1;i++)
    	{
    		cnt=0;
    		for(int j=0;j<a.length();j++)
    		{
    			if(a[j]!=b[j+i-1])
    			cnt++;
    		}
    		if(cnt<min)
    		min=cnt;
    	}
    	cout<<min<<endl;
    }
    return 0;
}
