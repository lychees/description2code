#include <bits/stdc++.h>
#include <assert.h>
using namespace std;
typedef long long ll;
#define MOD 1000000007LL
#define PB push_back
#define INF 999999999999999LL
ll a[10];
ll x[10];
int main() {
	ios_base::sync_with_stdio(false);
	ll T; cin >> T;
	for (ll ii = 0; ii < T; ii++) {
		ll n;
		cin >> n;
		for (ll i = 0; i < n; i++) {
			cin >> a[i];
		}
		for (ll i = 0; i < n; i++) {
			cin >> x[i];
		}
		ll nim = 0;
		for (ll i = 0; i < n; i++) {
			nim = nim ^ (x[i]-1);
			nim = nim ^ (a[i]-x[i]);
		}
		if (nim == 0) {
			cout << "Second" << endl;
		}
		else {
			cout << "First" << endl;
		}
	}
}
