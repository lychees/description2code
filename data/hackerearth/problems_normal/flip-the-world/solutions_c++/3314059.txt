#include<iostream>
using namespace std;
long a[100][100];string s[104];
void flip(long x,long y){
	for(long i=0;i<=x;i++){
		for(long j=0;j<=y;j++){
			if(s[i][j]=='0')s[i][j]='1';
			else s[i][j]='0';
		}
	}
}
int main(){
	int t;
	cin>>t;
	for(int i=0;i<t;i++){
		int n,m;
		cin>>n>>m;long l=0;
		for(long j=0;j<n;j++){
			cin>>s[j];
		}char c=s[n-1][m-1];if(c=='0'){c='1';l=1;flip(n-1,m-1);
		}
		for(long k=m-1;k>=0;k--){
			for(long j=n-1;j>=0;j--){
				if(s[j][k]!=c){
					l=l+1;flip(j,k);
				}
			}
		}
		cout<<l<<endl;
	}
}
