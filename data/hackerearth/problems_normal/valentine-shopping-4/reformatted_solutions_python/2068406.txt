from sys import stdin
testCases = int(stdin.readline())
for i in range(testCases):
    n, m = map(int, stdin.readline().split())
    selectedStores = []
    for j in range(n):
        thresholdPrice = 10000000
        selectedStore = 1
        for k in range(m):
            discount1, discount2, discount3 = map(
                int, stdin.readline().split())
            discountPrice = (100 - discount1) * \
                (100 - discount2) * (100 - discount3)
            if discountPrice < thresholdPrice:
                thresholdPrice = discountPrice
                selectedStore = k + 1
        selectedStores.append(selectedStore)
    print ' '.join(map(str, selectedStores))
