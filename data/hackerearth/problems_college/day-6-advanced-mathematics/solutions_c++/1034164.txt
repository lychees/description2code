#include <iostream>
using namespace std;

int main()
{
    int n = 0, flag = 0;
    string s = "";
    
    cin >> n;
    while (n--){
    	cin >> s;
    	for (int i = 1; i < s.length(); i++){
    		if (s[i] == s[i-1]){
    			flag = 1;
    			break;
    		}
    	}
    	if (flag)
    		cout << "SLAP" << endl;
    	else
    		cout << "KISS" << endl;
    	flag = 0;
    }
    
    return 0;
}
