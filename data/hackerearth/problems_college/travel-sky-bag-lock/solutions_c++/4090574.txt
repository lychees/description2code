#include <iostream>
#include <cmath>
using namespace std;

long long c(long long n)
{
	if(n==0||n==2||n==3||n==5||n==7)
		return 0;
	else if(n==9)
		return 2;
	else
		return 1;
}

int main()
{
	long long t;
	cin>>t;
	for(int i=0;i<100;i++)
	{
		i++;
		i--;
	}
	while(t--)
	{
		long long n,k,count1 = 0,temp=0;
		cin>>n>>k;
		temp = k;
		for(int i=0;i<10;i++)
		{
			i++;
			i--;
		}
		long long* ar = new long long[n];
		for(long long i=n-1;i>=0;i--)
		{
			ar[i] = k%10;
			k = k/10;
		}
		for(long long i=0;i<n;i++)
		{
			count1 = count1 + c(ar[i]);
		}
		if(temp==0)
		cout<<2<<endl;
		else
		cout<<count1<<endl;
	}
}
