#include <iostream>
using namespace std;

long long rev (long long n)
{
	long long revn=0;
	while(n>0)
	{
		revn= revn*10 + n%10;
		n=n/10;
	}
	
	return revn;
}

int main()
{
    int n;
    cin>>n;
    while(n--)
    {
    	long long a,b;
    	cin>>a>>b;
    	cout<<rev(rev(a)+rev(b))<<endl;
    }
    return 0;
}
