#include <iostream>
#include <cstdlib>
#include <string>
#include <vector>
#include <algorithm>
#include <sstream>
#include <map>
#include <cctype>
#include <cmath>
#include <utility>
using namespace std;

typedef pair<int,int> ii;
typedef long long ll;
typedef vector<int> vi;
typedef vector<ii> vii;
typedef vector<ll> vl;


int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	
	int ceo, cto, coo;
	cin>>ceo>>coo>>cto;
	
	int n;
	cin>>n;
	
	vi ht(n);
	
	for(auto &v: ht){
		cin>>v;
	}
	
	sort(ht.begin(), ht.end());
	
	int minval = min(coo, cto);
	int maxval = max(coo, cto);
	
	ll ans = abs(minval - ht[0]) + abs(maxval - ht[n-1]);
	
	if((ht.size() % 2) == 0){
		ht.insert(ht.begin() + ht.size()/2 , ceo);
		for(int i=0; i<n; ++i){
			ans += abs(ht[i] - ht[i+1]);
		}
		cout<<ans;
		return 0;
	}
	else{
		vi temp = ht;
		ht.insert(ht.begin() + ht.size()/2 , ceo);
		temp.insert(temp.begin() + temp.size()/2 + 1, ceo);
		ll ans1 = ans;
		for(int i=0; i<n; ++i){
			ans += abs(ht[i] - ht[i+1]);
		}
		for(int i=0; i<n; ++i){
			ans1 += abs(temp[i] - temp[i+1]);
		}
		//for(auto &v : ht) cout<<v<<" "; cout<<"\n";
		//for(auto &v : temp)  cout<<v<<" "; cout<<"\n";
		cout<<min(ans, ans1);
		return 0;
	}
	
	return 0;
}
