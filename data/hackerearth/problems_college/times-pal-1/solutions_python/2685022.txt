def is_palidrome(input_string):
    if len(input_string) < 2:
        return True

    if input_string[0] == input_string[len(input_string) - 1]:
        return is_palidrome(input_string[1:len(input_string) - 1])
    else:
        return False


def check_variations(A, B):
    counts = 0
    for i in xrange(len(A)):
        if is_palidrome(A[:i] + B + A[i:]):
            counts += 1

    if is_palidrome(A+B):
        counts += 1

    return counts
