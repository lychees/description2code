#include <iostream>
using namespace std;
#define num 1000000
int main()
{
    int t,a[num],n,d=0,i;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<=n;i++)
    	{
    		if(i==1 || i==0)
    		{
    			d=0;
    			a[i]=0;
    		}
    		else
    		{
    		d=d+1;
    		a[i]=a[i-1]+d;
    		}
    	}
    	cout<<a[n]<<endl;
    }
    return 0;
}
