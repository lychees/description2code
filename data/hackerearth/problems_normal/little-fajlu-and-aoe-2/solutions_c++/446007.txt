#define pn printf("\n")
int main()
{
    int T;
    si(T);
    while(T--)
    {
        int F,G,S,P,C,TP1,TP2=0;
        scanf("%d%d%d%d%d%d",&F,&G,&S,&P,&C,&TP1);
        for(int i=0;i*100<=F;i++)
        {
            for(int j=0;i*100+j*125<=F && j*50<=G;j++)
            {
                int f_l = F-i*100-j*125;
                int g_l = G-j*50;
                int k = min(f_l/50,g_l/100);
                TP2 = max(TP2,i*S + j*P + k*C);
            }

        }
        if(TP1 > TP2)
            pi(-1);
        else
            pi(TP2-TP1);
        pn;
    }
	return 0;
}
