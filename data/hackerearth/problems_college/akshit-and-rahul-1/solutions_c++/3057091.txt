int dp[maxn][maxn];

int solve(int a, int b)
{
	if(dp[a][b]!=-1)
		return dp[a][b];
	int i=1, ret=0, x;
	while(i*b < a){
		x = solve(max(a-(i*b), b), min(a-(i*b), b));
		if(x==0){
			ret = 1;
			break;
		}
		i++;
	}
	return dp[a][b] = ret;
}
