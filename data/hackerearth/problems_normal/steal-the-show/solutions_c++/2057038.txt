#include <iostream>
#include <math.h>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{

    int n, a, ceo, coo, cto, len, sum = 0;
    vector<int> v;
    vector<int>::iterator it;

    cin >> ceo >> coo >> cto;
    cin >> n;

    for(int i=0; i<n; i++){
    	cin >> a;
    	v.push_back(a);
    }

    sort(v.begin(), v.end());

    v.insert(v.begin(), min(coo, cto));
    v.push_back(max(coo, cto));

    it = v.begin();
    len = v.size();

    if(len % 2 == 0)
        v.insert(v.begin() + (len/2), ceo);
    else{
        it += len/2;
        if(abs(*(it - 1) - ceo) < abs(*(it + 1) - ceo))
            v.insert(v.begin() + (len/2), ceo);
        else
            v.insert(v.begin() + (len/2) + 1, ceo);
    }

    len = v.size();

    for(int i=0; i<len-1; i++)
        sum += abs(v[i+1] - v[i]);

    cout << sum;

    return 0;
}
