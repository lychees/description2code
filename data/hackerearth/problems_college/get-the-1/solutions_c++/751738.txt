#include <iostream>
#include <set>

bool happyNum(int n)
{
	std::set<int> cycle;
	cycle.insert(n);
	
	while (n != 1)
	{
		int res = 0;
		
		while (n != 0)
		{
			res = res + (n % 10) * (n % 10);
			n /= 10;
		}
		
		if (cycle.find(res) != cycle.end())
		{
			return false;
		}
		
		n = res;
		cycle.insert(n);
	}
	
	return true;
}

int main()
{
	int t, n;
	std::cin >> t;
	
	for (int i = 1; i <= t; ++i)
	{
		std::cin >> n;
		std::cout << "Case #" << i << ":" << (happyNum(n) ? "YES\n" : "NO\n");
	}

	return 0;
}
