ll getSum(ll v)
{
	ll t, s;
	if (v % 2 == 0)
	{
		t = v / 2;
		s = t * (v + 1);
	}
	else
	{
		t = (v + 1) / 2;
		s = v * t;
	}
	return s;
}

	ll ans = getSum(v1) * 3 + getSum(v2) * 5 - getSum(v3) * 15;
