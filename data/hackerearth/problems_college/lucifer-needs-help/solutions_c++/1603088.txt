#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
long long n,m,i,j,k;
int flag =0;
cin>>n>>m;
long long a[n];
for(i=0;i<n;i++)
{
cin>>a[i];
}
long long sum;
sort(a,a+n);
for(i=0;i<n;i++)
{
for(j=i+1;j<n;j++)
{
for(k=j+1;j<n;j++)
{
sum = a[i]+a[j]+a[k];
if(sum == m)
{
flag=1;
break;
}
}
if(flag==1)
{
break;
}
}
if(flag==1)
{
break;
}
}
if(flag==1)
{
cout<<"YES"<<"\n";
}
else
cout<<"NO"<<"\n";
return 0;
}
