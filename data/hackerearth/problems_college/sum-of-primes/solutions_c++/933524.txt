#include<bits/stdc++.h>
using namespace std;
#define ll	long long int
#define MOD	1000000007
vector<ll>primes;
bool flag[1000005];
ll ans[1000005];
void sieve()
{
	memset(flag,true,sizeof(flag));
	flag[1]=false;
	for(int i=2;i<=1000000;++i)
	{
		if(flag[i])
		{
			primes.push_back(i);
			for(int j=2*i;j<=1000000;j+=i)
				flag[j]=false;
		}
	}
	ans[0]=0;
	ans[1]=0;
	for(int i=2;i<=1000000;++i)
	{
		if(flag[i])
			ans[i]=ans[i-1]+i;
		else
			ans[i]=ans[i-1];
	}
}
bool check_prime(ll n)
{
	if(n==1 || n==0)
		return false;
	for(int i=2;i<=(ll)(sqrt(n));++i)
		if(n%i==0)
			return false;
	return true;
}
int main()
{
	sieve();
	int t;
	cin>>t;
	while(t--)
	{
		int m,n;
		cin>>m>>n;
		ll sum=ans[n]-ans[m-1];
		if(check_prime(sum))
				cout<<"YES\n";
		else
				cout<<"NO\n";
	}
	return 0;
}
