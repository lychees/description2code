#include<bits/stdc++.h>
using namespace std;
main()
{
	ios_base::sync_with_stdio(0);cin.tie(0);
	long long int t,x,i,k,l,m,z,c;
	cin>>t;
	while(t--)
	{
		k=0;l=0;
		cin>>x;
		if(x%2==0)
		{
			x++;
		}
		while(k!=1)
		{
			for(i=3;i<=sqrt(x);i=i+2)
			{
				if(x%i==0)
				{
					goto ll;
				}
			}
			c=0;
			m=x;
			while(m>0)
				{
					z=m%10;
					c=(c*10)+z;
					m/=10;
				}
				if(c==x)
				{
				k=1;
				cout<<x<<endl;
			    }
				else
				ll: x=x+2;
		}
	}
}
