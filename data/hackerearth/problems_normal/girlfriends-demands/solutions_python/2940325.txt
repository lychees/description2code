str_ = raw_input()
l = len(str_)

q = raw_input()
input_lst = []


for a in range(int(q)):
    input_ = raw_input().split()
    input_lst.append(input_)

for b in range(int(q)):

    a1 = input_lst[b][0]
    a2 = input_lst[b][1]

    m_a1 = int(a1) % l
    m_a2 = int(a2) % l

    if(str_[m_a1 - 1] == str_[m_a2 - 1]):
        print 'Yes'

    else :
        print 'No'
