#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int arr[1000];
		int n,h,x;
		cin>>n>>h;
		for (int i = 0; i < n; ++i)
		{
			cin>>x;
			if(x >= h)
				x = x - h;
			else
				x = h - x;
			arr[i] = x;
		}
		sort(arr,arr+n);
		
		cout<<arr[0]<<endl;
	}
	return 0;
}
