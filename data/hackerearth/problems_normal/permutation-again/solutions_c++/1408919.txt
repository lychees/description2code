#include <iostream>
using namespace std;

int main()
{
	long int t,n,i,j,result;
	cin >> t;
	for(i = 0; i < t; i++)
	{
		cin >> n;
		if(n == 1)
			result = 1;
		else
		{
			result = n-1;
			result = result + n-2;
		}
		for(j = 2; j <= (n-j); j++)
		{
			result = result + (n-j-j);
			result = result + (n+2-j) - j;
		}
		cout << result <<endl;
	}
    return 0;
}
