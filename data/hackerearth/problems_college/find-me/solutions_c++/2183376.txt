#include<bits/stdc++.h>
using namespace std;
int main(){
    int T,N,P,i,temp;
    cin>>T;
    while(T--){
        cin>>N>>P;
        vector<int>v;
        for(i=0;i<N;i++){
            cin>>temp;
            v.push_back(temp);
        }
        sort(v.begin(),v.end());
        int counter=0;
        vector<int>::iterator it=v.begin();
        it=find(v.begin(),v.end(),P);
        if(it!=v.end()){
            temp=*it;
            for(it;it!=v.end();++it){
                if((*it)>temp)
                    counter+=1;
            }
            cout<<"YES "<<counter<<endl;
        }
        else{
            cout<<"NO "<<counter<<endl;
        }
    }
    return 0;
}
