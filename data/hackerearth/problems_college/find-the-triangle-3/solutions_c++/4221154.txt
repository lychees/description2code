lli modpow(lli base,lli exp,lli mod)
{   if(mod==1) return 0;
    lli res=1;
    base%=mod;
    while(exp>0)
    {   if(exp%2) res=(res*base)%mod;
        base=(base*base)%mod;
        exp/=2;
    }
    return res;
}
lli power(lli base,int exp)
{   lli res=1;
    while(exp>0)
    {   if(exp%2) res*=base;
        base*=base;
        exp/=2;
    }
    return res;
}
bool chk(lli x,lli y,lli z)
{   return (x+y>z)&&(y+z>x)&&(x+z>y);
}
