#include <bits/stdc++.h>
using namespace std;
#define ll long long int
#define pb push_back
bool cmp( pair < int,string> v, pair < int,string> u)
{
    if(v.first<u.first)
        return false;
    else if(v.first>u.first)
        return true;
    else if(v.first==u.first&&v.second<u.second)
        return true;
    else if(v.first==u.first&&v.second>u.second)
        return false;
    else return false;
}
int main()
{
    int n;
    cin>>n;
    vector < pair < int,string> > v;
    int t;string s;
    for(int i=1;i<=n;i++)
    {
        cin>>s>>t;
        v.pb(make_pair(t,s));
    }
    sort(v.begin(),v.end() ,cmp);
    std::vector < pair < int,string> >::iterator it;
    for(it=v.begin();it!=v.end();it++)
        cout<<it->second<<" "<<it->first<<endl;
}
