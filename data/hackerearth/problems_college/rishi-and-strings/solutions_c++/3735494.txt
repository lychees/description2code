#include<bits/stdc++.h>
using namespace std;
typedef long long int lli;
#define ff first
#define ss second
#define mp make_pair
#define pb push_back
string s1,s2;
int dp[150][150];
int main()
{
	 cin>>s1>>s2;
	 int i,j,k,l1=s1.size(),l2=s2.size();
	 for(i=0;i<=l1;i++)
	 dp[0][i]=i;
	 
	 for(i=0;i<=l2;i++)
	 dp[i][0]=i;
	 
	 for(i=1;i<=l2;i++)
	 {
	 	for(j=1;j<=l1;j++)
	 	{
	 		if(s1[j-1]==s2[i-1])
	 		{
	 			dp[i][j]=dp[i-1][j-1];
			 }
			 else
			 {
			 	dp[i][j]=min(dp[i-1][j-1],min(dp[i][j-1],dp[i-1][j]))+1;
			 }
		 }
	 }
	 cout<<dp[l2][l1]<<endl;
	 
	
}
