T = int(raw_input())

for _ in range(T):
	S1 = str(raw_input())
	S2 = str(raw_input())
	num = [0]*26
	for c in S1:
		if c != " ":
			num[ord(c)-97]+=1
	for c in S2:
		if c != " ":
			num[ord(c)-97]-=1
	win = "none"
	for i in range(26):
		if num[i] > 0:
			if win == "lose":
				win = "draw"
				break
			else: win = "win"
		elif num[i] < 0:
			if win == "win":
				win = "draw"
				break
			else: win = "lose"
	print "You " + win + " some."
