using namespace std;
#define pii pair<long long,long long>
#define F(i,a,b) for(ll i = (ll)(a); i <= (ll)(b); i++)
#define RF(i,a,b) for(ll i = (ll)(a); i >= (ll)(b); i--)
#define PI 3.14159265
#define ll long long
#define ff first

#define INF 1000000009

vector <ll> v;
ll arr[1005];
ll m,p;
ll ans1[1005];
ll dp[1265][120];
ll solve(ll i,ll packer)
{
	if(i>p-1)
		return dp[i][packer] = INT_MAX;
	if(dp[i][packer]!=-1)
		return dp[i][packer];
	if(packer==m)
	{
		ll sum = 0;
		F(j,i,p-1)
			sum += arr[j];
		return dp[i][m] = sum;
	}
	//cout<<i<<" "<<packer<<endl;
	ll sum = 0;
	ll ans = INT_MAX;
	F(j,i,p-m+packer)
	{
		sum = sum + arr[j];
		ll temp = solve(j+1,packer+1);
		ans = min(ans,max(sum,temp));
		//cout<<ans<<endl;
	}
	return dp[i][packer] = ans;
}
int main()

	ll ans = solve(0,1);
	cout<<ans<<endl;

}
