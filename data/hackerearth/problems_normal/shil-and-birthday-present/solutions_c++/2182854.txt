#include<iostream>
#include<cstdio>
#include<algorithm>
void scanint (int &);
#define gc getchar_unlocked
using namespace std;
int main()
{
	int n;
	long long int count=1;
	  scanint(n);
	 int a[n];
	 for(int i=0;i<n;i++)
	    scanint(a[i]);
	    sort(a,a+n);
	    for(int i=0;i<n-1;i++)
	    {
	    	if(a[i]!=a[i+1])
	    	   count++;
	    }
	  count=(count*(count-1))/2;
	    printf("%lld \n",count);
	
	
}

 	 register int c =gc();

 	{

	 }
}
