#include <iostream>
using namespace std;

int main()
{
    int i,N,M,A[200000];
    cin>>N>>M;
    for(i=0;i<N;i++)
    cin>>A[i];
    while(M--)
    {
              char ch;
              cin>>ch;
              if(ch=='U')
              {
                         int X,Y;
                         cin>>X>>Y;
                         A[X-1]=Y;
              }
              else
              {
                  int X,C,len,max,min;
                  cin>>X>>C;
                  i=X-1;
                  min=max=A[i];
                  while(i<N && A[i]<=A[X-1]+C && A[i]>=A[X-1]-C)
                  {
                            if(A[i]>max)
                            max=A[i];
                            if(A[i]<min)
                            min=A[i];
                            i++;
                  }
                  len=i-(X-1);
                  if(len==0)
                  cout<<"-1 -1"<<endl;
                  else
                  {
                      int diff1,diff2;
                      diff1=A[X-1]-min;
                      diff2=max-A[X-1];
                      if(diff1>=diff2)
                      cout<<len<<" "<<diff1<<endl; 
                      else
                      cout<<len<<" "<<diff2<<endl;
                  }
                      
              }
    }
    return 0;
}
