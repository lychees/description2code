#include<cstdio>
#include<iostream>
#include<stack>
#include<queue>
#include<math.h>
using namespace std;
int main()
{
int t,x1,y1,r1,x2,y2,r2;
cin>>t;
while(t--)
{
  cin>>x1>>y1>>r1;
  cin>>x2>>y2>>r2;  
  
  double d= sqrt( (x2-x1)*(x2-x1)+(y2-y1)*(y2-y1) ); 
  if(r2>r1)
  {
     if(d<=r2 && d+r1<=r2)
     {
        cout<<"C2CC1"<<endl;
     }
     else
     {
        cout<<"NOT SATISFIED"<<endl;
         }
          
   }
else if(r1>r2)
{if(d<=r1 && d+r2<=r1)
     {
        cout<<"C1CC2"<<endl;
     }
     else
     {
        cout<<"NOT SATISFIED"<<endl;
         }
    
    }
    else if(r1==r2 && x1==x2 && y1==y2)
    { cout<<"C2~C1"<<endl;
        }
        
        else
        {cout<<"NOT SATISFIED"<<endl;
        }
}
}
