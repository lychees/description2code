#include <iostream>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
    	string a,b;
    	cin>>a;
    	cin>>b;
    	int k=a.length() + b.length();
    	n=a.length();
    	int ans=0;
    	for(int i=0;i<n+1;i++)
    	{
    		string s,p;
    		p=b;
    		if(!i) s=b+a;
    		else if(i==n) s=a+b;
    		else
            {
            	p+=a.substr(i,n-1);
            	s=a.substr(0,i);
            	s+=p;
            }
            int flag=0;
            for(int j=0;j<(k+1)/2;j++)
            {
            	if(s[j]!=s[k-1-j])
            	{
            		flag=1;
            		break;
            	}
            }
            //cout<<s<<endl;
            if(!flag) ans++;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
