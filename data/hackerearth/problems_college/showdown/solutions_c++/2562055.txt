#include <bits/stdc++.h>
using namespace std;
vector<int>f,m;
int n;
bool fun1(int i,int j)
{
	return (i>j);
}
int fun(int i, int j)
{
	int sum;
	if(i==n||j==n)
	{
		sum = 0;
	}
	else if(f[i]>m[j])
	{
		sum = 100+fun(i+1,j+1);
	}
	else if(f[i]==m[j])
	{
		sum = max(50+fun(i+1,j+1),fun(i,j+1));
	}
	else
	{
		sum = 0+fun(i,j+1);
	}
	return sum;
}
int main()
{
    cin>>n;
    int temp;
    for(int i=0;i<n;i++)
    {
    	cin>>temp;
    	f.push_back(temp);
    }
    for(int i=0;i<n;i++)
    {
    	cin>>temp;
    	m.push_back(temp);
    }
    sort(f.begin(),f.end(),fun1);
    sort(m.begin(),m.end(),fun1);
    cout<<fun(0,0)<<endl;
   
}
