void swap(long *x,long *y){
	long t = *x;
	*x = *y;
	*y = t;
}
void revert(long a[],long n,long start, long end){
	if(start == -1 && end == -1)return;
	
	for(long i=start; i <= (start+end)/2; i++){
		swap(&a[start], &a[end]);
		start++;
		end--;
	}
}
