#include <string>
#include <iostream>
#include <map>

using namespace std;

int main() {
    map<string, int> m;
    string input;
    cin >> input;
    int x = 0, y = 0;
    int iced = 0;
    
    m[ to_string(x) + ' ' + to_string(y) ] = 1;
    
    for ( int i = 0; i < input.length(); i++ ) {
        if ( input[i] == 'L' ) {
            y--;
        }
        else if ( input[i] == 'R' ) {
            y++;
        }
        else if ( input[i] == 'U' ) {
            x--;
        }
        else {
            x++;
        }
        
        string key = to_string(x) + ' ' + to_string(y);
        if ( m.find(key) != m.end() ) {
            iced++;
        }
        else {
            m[ to_string(x) + ' ' + to_string(y) ] = 1;
        }
    }
    cout << iced << endl;
    
    return 0;
}
