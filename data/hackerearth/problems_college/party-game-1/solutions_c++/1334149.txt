#include<stdio.h>
#include<algorithm>
#include<string>
#include<queue>
#include<string.h>
#include<map>
#include<vector>
#include<math.h>
#include<iostream>
#define MAX 100
#define M 1000000007
#define PB push_back
#define scll(t) scanf("%lld",&t)
#define scl(t) scanf("%ld",&t)
#define max(a,b) a>b?a:b
#define min(a,b) a<b?a:b
#define ll long long
#define ll long long
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long ar[10000];
        long n,i;
        long x,y;
        scl(x);
        scl(y);
        scanf("%ld",&n);
        for(i=0;i<n;i++)
            scl(ar[i]);
        sort(ar,ar+n);
        ll su1=0;
        ll su2=0;
        i=0;
        while(i<n)
        {
            if(x-ar[i]>=0)
            {
                su1++;
                x-=ar[i];
            }
            i++;
        }
        i=0;
        while(i<n)
        {
            if(y-ar[i]>=0)
                {
                    su2++;
                    y-=ar[i];
                }
            i++;
        }
        if(su1==su2)
            printf("Tie\n");
        else if(su1>su2)
            printf("Chef Won\n");
        else
            printf("Jhool Won\n");
    }
    return 0;
}
