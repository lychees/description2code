#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		sort(a,a+n);
		int k=0,c=1;
		for(int i=1;i<n;i++)
		   if(a[k]!=a[i])
		   {
		   	k++;
		   	c++;
		   }
		   else k++;
		   cout<<c<<endl;
	}
	
	return 0;	   
	}
