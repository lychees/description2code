#include<iostream>
#include<utility>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        int n;
        cin>>n;
        int count=0;
        vector<pair<int,int> > pos;
        for(int i=0;i<n;i++)
        {
            int f,s;
            cin>>f>>s;
            pos.push_back(make_pair(f,s));
        }
        
        sort(pos.begin(),pos.end());

        int LIS[n];
        
        LIS[0]=1;

        for(int i=1;i<n;i++)
        {
            int maximum=0;
            for(int j=0;j<i;j++)
            {
                if(pos[i].second > pos[j].second)
                {
                    if(LIS[j] > maximum)
                    {
                        maximum=LIS[j];
                    }
                }
            }
            LIS[i]=1 + maximum;
        }

        int _max=0;
        for(int i=0;i<n;i++)
            if(LIS[i]>_max)
                _max=LIS[i];

        cout<<_max<<endl;

    }
}
