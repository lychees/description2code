#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,a,b;
	scanf("%d", &t);
	while ( t-- ) {
		scanf("%d%d", &a, &b);
		a = 1<<a, b = 1<<b;
		a--, b--;
		a = __gcd(a,b);
			int ans = 0;
			int tt = (int)sqrt(a);
			for ( int i = 2; i <= tt; i++ ) {
				if ( a%i == 0 ) {
					ans++;
					while ( a%i == 0 ) a /= i;
				}
				if ( a == 1 ) break;
			}
			if ( a != 1 ) ans++;
			printf("%d\n", ans);
		
	}
	return 0;
}
