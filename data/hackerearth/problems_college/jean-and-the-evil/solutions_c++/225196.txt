#include <iostream>
#include<algorithm>
#include<cstring>
using namespace std;
long long b[1000003]={0};
int main()
{
    //cout << "Hello World!" << endl;
    int t,n,i;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	int a[n];
    	memset(b,0,sizeof(b));
    	for(i=0;i<n;i++)
    	scanf("%d",&a[i]);
    	for(i=0;i<n;i++)
    	{
    		b[i]=b[i-1]+a[i];
    		if(b[i]<0)
    		b[i]=0;
    	}	
    	long long maxi=-1;
    	for(i=0;i<n;i++)
    	maxi=max(maxi,b[i]);
    	if(maxi<0)
    	maxi=0;
    	printf("%lld\n",maxi);
    }
    return 0;
}
