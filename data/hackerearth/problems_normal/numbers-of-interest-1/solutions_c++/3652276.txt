#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t){
    	t--;
    	int x,y,n;
    	cin>>x>>y>>n;
    	if (n<=x)
    		cout<<y<<endl;
    	else{
    		long a[50];
    		for (int i=1;i<=x;i++)
    			a[i]=y;
    		int top = x+1;
    		while(n>=top){
    			a[top]=0;
    			for(int i=1;i<=x;i++)
    				a[top]+=a[top-i];
    			top++;
    		}
    		cout<<a[n]<<endl;
    	}
    }
    return 0;
}
