#include <iostream>
#include<stdio.h>
using namespace std;
int gcd(int i,int j){
	int temp;
	while(j!=0){
		temp = j;
		j = i%j;
		i = temp;
	}
	return i;
}
int main()
{
	int n,pair=0;
	scanf("%d",&n);
	int *a;
	a = new int[n];
	for(int i=0;i<n;++i){
		scanf("%d",&a[i]);
	}
	for(int i=0;i<n-1;++i){
		for(int j=i+1;j<n;++j){
			if(a[i] > a[j] && gcd(a[i],a[j])==1)
			++pair;
		}
	}
	printf("%d",pair);
    return 0;
}
