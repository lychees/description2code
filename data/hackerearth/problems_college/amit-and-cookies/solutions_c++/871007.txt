#include<iostream>
#include<stdlib.h>
using namespace std;
int main()
{
	int n,k,i,z,sum=0;
	cin>>n>>k;
	int a[n];
	for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
	for(int j=0;j<n;j++)
	{
	  for(i=0;i<n-1;i++)
	  {
		if(a[i]>a[i+1])
		{
			z=a[i];
			a[i]=a[i+1];
			a[i+1]=z;
		}
	  }  
    }
	for(i=0;i<n;i++)
	{
		sum+=a[i];
		if(sum>=k)
		{
			cout<<i;
			exit(0);
		}
	}
	return 0;
}
