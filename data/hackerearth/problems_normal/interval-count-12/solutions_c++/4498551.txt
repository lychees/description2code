#include<bits/stdc++.h>


using namespace std;

long long int n,l,r;

long long int primek[11];
vector<long long int> v;

long long int ebbol (long long int z) {
    long long int ls=l+(z-(l % z));
    if (l%z==0) ls-=z;
    long long int rs=r-(r%z);
    if (rs>=ls) return ((rs-ls)/z+1);
    else return 0;
}
long long int maxi=1000000000000000000;
long long int szamok (int k2) {
    long long int k=k2;
    long long int valasz=0;
    long long int akt=1;
    int tart=1;
    int ossz=0;

    while (k!=0 && akt<=maxi) {
        if (k % 2 == 1) {
            akt*=primek[tart];
            k-=1;
            ossz+=1;
        }
        k=k/2;
        tart++;
        if ((k % 2 ==1) && akt>(maxi/primek[tart])) akt=maxi+1;
    }
    if (akt<=maxi) {

    valasz=ebbol(akt);
    if (ossz % 2 ==0) valasz=0-valasz;
    }
    return valasz;

}

void egesz () {
    long long int vegso=0;
    cin>>n>>l>>r;
    for (int i=1; i<=n; i++) {
        cin>>primek[i];
    }
    for (int i=1; i<(1<<n); i++) {
        vegso+=szamok(i);
    }
    v.push_back(vegso);
}
int t;
int main () {
    cin>>t;
    for (int i=1; i<=t; i++) {
        egesz();
    }
    for (int i=0; i<t; i++) {
        cout<<v[i]<<endl;
    }
}
