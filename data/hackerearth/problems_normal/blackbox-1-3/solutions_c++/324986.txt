#include <iostream>
#include<cstdio>
using namespace std;
int gcd ( int a, int b )
{
  if ( a==0 ) return b;
  return gcd ( b%a, a );
}


int main() 
{
	int a,b,c,t,flag1,flag2;
	scanf("%d",&t);
	while(t--)
	{
		flag1=flag2=0;
		scanf("%d%d%d",&a,&b,&c);
		if(a*a+b*b==c*c||c*c+b*b==a*a||a*a+c*c==b*b)
		flag1=1;
		if(gcd(a,b)==1&&gcd(b,c)==1)
		flag2=1;
		
		if(flag1==1&&flag2==1)
		printf("YES\n");
		else 
		printf("NO\n");
		
		
		
		
		
		
		
		
		
		
	}
	return 0;
}
