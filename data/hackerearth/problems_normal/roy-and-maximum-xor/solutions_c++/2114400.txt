#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int myfind(ll int n)
{
	ll int j=0;
	while(n!=0)
	{
		n>>=1;
		j++;
	}
	return j;
}
int main()
{
	//~ ll int z=1;
	//~ printf("%lld\n",(ll int)(z<<35));
	//~ cout<<LONG_LONG_MAX<<endl;
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int n;
		scanf("%lld",&n);
		ll int v,x,ans;
		x=myfind(n);
		//~ cout<<"X is "<<x<<endl;
		v=1;
		v=v<<x;
		//~ cout<<v<<endl;
		if(v==n*2)
		{
			v/=2;
		}
		v=v-1;
		ans=v-(n-1);
		ans=n-ans;
		printf("%lld %lld\n",v,ans);
	}
	return 0;
}
