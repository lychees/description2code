#include<iostream>
#include<algorithm>
#include<vector>
#define ll long long int
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll n,k,p;
		cin>>n>>k>>p;
		ll diff=2*p;
		vector<ll> input(n);
		for(int i=0;i<n;i++)
		{
			cin>>input[i];
		}
		sort(input.begin(),input.end());
		bool flag=true;
		for(int i=0;i<n;i++)
		{
			ll check=input[i]+diff;
			int upb=upper_bound(input.begin(),input.end(),check)-input.begin();
			if(upb-i>=k){
				flag=false;
			}
		}
		if(flag)cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
	return 0;
}
