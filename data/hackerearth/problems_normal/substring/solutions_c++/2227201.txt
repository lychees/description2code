#include <iostream>
#include <string>
using namespace std;
unsigned long long subs[100001];
int main()
{
	subs[0] = 0;
	for(long i=1; i<100001; i++) {
		subs[i] = i + subs[i-1];
	}
	int t, count, total, length, i, k, l;
	string s;
	cin>>t;
	while(t--) {
		count = 0;
		total = 0;
		cin>>s;
		length = s.length();
		for(i=0; i<length; i++) {
			k = s[i] - 'A';
			l = s[i] - 'a';
			if((l>=0 && l<=2) || (k>=0 && k<=2)) {
				count++;
			} else {
				total += subs[count];
				count = 0;
			}
		}
		total += subs[count];
		cout<<total<<endl;
	}
    return 0;
}
