#include <bits/stdc++.h>

using namespace std;
 
#define MAX 1005
#define INF 2ll * (1e18)
#define MOD 1000000007
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define ll long long

int dp[MAX][MAX];

int main(){
	ios_base::sync_with_stdio(false);
	int t;	
	cin>>t;

	while(t--){
		int n, m;
		cin>>n>>m;

		memset(dp, 0, sizeof(dp));
	
		dp[1][1] = 1;
		
		for(int i=2 ; i<=n ; i++){
			for(int j=1 ; j<=i ; j++){
				int dex, mad;
				dex = j, mad = i - j;
				if(dex < mad * m)
					continue;
				dp[i][j] = (dp[i-1][j-1] + dp[i-1][j]) % MOD;		
			}

		}

		int ans = 0;

		for(int i=1 ; i<=n ; i++){
			int dex = i, mad = n - i;
			if(dex < mad * m || dex == mad)
				continue;
			ans = (ans + dp[n][i]) % MOD;
		}

		cout<<ans<<"\n";
	}

	return 0;
}
