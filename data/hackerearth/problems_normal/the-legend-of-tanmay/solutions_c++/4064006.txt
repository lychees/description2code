#include<bits/stdc++.h>
#include<stdio.h>
#include<stdlib.h>
using namespace std;
#define ll long long int
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(false);
    ll t,n,i,j,c0,cp,cn,k,l;
    cin>>t;
    while(t--)
    {
        c0=0;
        cp=0;
        cn=0;
        cin>>n;
        ll a[n],pos[n],neg[n];
        for(i=0;i<n;i++)
        {
            cin>>a[i];
            if(a[i]==0)
            c0++;
            else if(a[i]>0)
            {pos[cp++]=a[i];}
            else if(a[i]<0)
            {neg[cn++]=abs(a[i]);}
        }
        j=1;
        l=1;
        sort(pos,pos+cp);
        sort(neg,neg+cn);
        for(i=0;i<cp;i++)
        {
            j=j*pos[i];
        }
        for(i=0;i<cn;i++)
        {
            l=l*neg[i];
        }
        if(cp==0)
        {
            if(c0==0)
            {
                if(cn%2==0)
                cout<<l<<" ";
                else if(cn==1)
                cout<<-1*neg[0]<<" ";
                else
                cout<<l/neg[0]<<" ";
            }
            else
            {
                cout<<"0 ";
            }
        }
        else
        {
            if(cn==0)
            cout<<j<<" ";
            else
            {
                if(cn%2==0)
                cout<<j*l<<" ";
                else
                {
                    cout<<(j*l)/neg[0]<<" ";
                }
            }
        }
        if(cp==0)
        {
            if(cn==0)
            {
                cout<<"0\n";
            }
            else
            {
                if(cn%2==1)
                cout<<-1*l<<endl;
                else
                cout<<-1*l/neg[0]<<endl;
            }
        }
        else
        {
            if(cn==0)
            {
                if(c0>0)
                cout<<0<<endl;
                else
                cout<<pos[0]<<endl;
            }
            else
            {
                if(cn%2==0)
                {
                    cout<<-1*j*l/neg[0]<<endl;
                }
                else
                cout<<-1*j*l<<endl;
            }
        }
    }
    return 0;
}
