#include <stdio.h>
#include <algorithm>

using namespace std;

int s[100005],a[100005];

int main(int argc, char **argv)
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		int i,n;
		scanf("%d", &n);
		for(i=0;i<n;i++)
			scanf("%d", &s[i]);
		for(i=0;i<n;i++)
			scanf("%d", &a[i]);
		
		sort(s, s+n);
		sort(a, a+n);
		
		for(i=0;i<n;i++)
			if(s[i]<=a[i])
			{
				printf("NO\n");
				break;
			}
			
		if(i>=n)
			printf("YES\n");
	}
	return 0;
}
