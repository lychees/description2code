t=int(raw_input());
x=[];
for i in range(0,t):
	s="";
	a=raw_input();
	b=raw_input();
	l=len(a);
	w=1;
	while(w<=l):
		dig1=a[w-1];
		dig2=b[w-1];
		if((dig1=='0' and dig2=='0') or (dig1=='1' and dig2=='1')):
			s+="0";
			w=w+1;
		else:
			s+="1";
			w=w+1;

	x.insert(i,s);
	
for i in range(0,t):
	print(x[i]);
