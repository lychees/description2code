#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
using namespace std;

int n;
int ans = 6;
string s[1010];

int main() {
  scanf("%d", &n);
  for (int i = 0; i < n; i++)
    cin >> s[i];
  for (int i = 0; i < n; i++)
    for (int j = i + 1; j < n; j++) {
      int tmp = 0;
      for (int k = 0; k < 6; k++)
        if (s[i][k] != s[j][k])
          tmp++;
      ans = min(ans, (tmp - 1) / 2);
    }
  printf("%d\n", ans);
  return 0;
}
