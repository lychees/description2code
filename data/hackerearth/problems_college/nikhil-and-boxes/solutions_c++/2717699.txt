#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    int ts;
    cin>>ts;
    while(ts--){
    long	double min=0;
    unsigned	long l,b,h;
    	cin>>l>>b>>h;
    	if (l<=b && l<=h)
    	{
     	min = l;
		}
    	else if (b<=l && b<=h)
    	{
    	min = b;
    	}
		else if (h<=l && h<=b)
    	{
    	min = h;
     	}
    	
    	cout<<setprecision(1)<<fixed<<min/2<<endl;
    }
    return 0;
}
