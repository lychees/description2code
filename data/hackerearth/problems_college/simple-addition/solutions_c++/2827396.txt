	#include<stdio.h>
int min,cur;
main()
{
int t,l;
scanf("%d",&t);
while(t--)
{
int i,t1,a,b,q;
scanf("%d",&l);
min=l;
if(l==1) min=0;
//process(...);
t1=l/2;
for(i=1;i<=t1;++i)
{
cur=1;
a=i;
b=l-i;
while(1)
{
q=(b-1)/a;
if(q==0)break;
cur+=q;
b=b-q*a;
a^=b^=a^=b;
if(cur>min) break;
}
if(cur<min && a==1 && b==1) min=cur;
}
printf("%d\n",min);
}
return 0;
}
