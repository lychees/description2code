#include <iostream>
using namespace std;

int main()
{
   long long int n,flag,t;
   cin>>t;
   while(t)
   {
   	cin>>n;
   	 flag=0;
   	while(n>0)
   	{
   		int l=n%10;
   		if(l==4||l==7)
   		flag++;
   		n=n/10;
   		
   	}
   	if(flag==4||flag==7)
   	cout<<"YES"<<endl;
   	else 
   	cout<<"NO"<<endl;
   	t--;
   	
   }
    return 0;
}
