#include <bits/stdc++.h>
using namespace std;

long long dp[66][66][3];
long long f(int n, int k) {
	memset(dp, 0, sizeof dp);
	
	for (int c = 0; c <= n; c++)
		dp[n][c][1] = 1;
	
	for (int i = n - 1; i >= 0; i--) {
		for (int c = 0; c <= i; c++) {
			for (int s = 0; s < 2; s++) {
				dp[i][c][s] = dp[i + 1][c + 1][s || (c + 1) >= k] + dp[i + 1][0][s];
			}
		}
	}
	
	return dp[0][0][0];
}

int main() {
    int n, k;
    cin >> n >> k;
    long long num, den;
   	num = f(n, k);
   	den = (1LL << n);
   	
   	long long g = __gcd(num, den);
   	num /= g;
   	den /= g;
   	
   	cout << num << "/" << den << endl;
   	
    return 0;
}
