#include <iostream>
using namespace std;
int root(int arr[],int i)
{
	while(i!=arr[i])
	{
		i=arr[i];
		
	}
	return i;
}
void union1(int arr[],int size[],int i,int j)
{
	int x=root(arr,i);
	int y=root(arr,j);
	if(size[x]<size[y])
	{
		arr[x]=arr[y];
		size[y]+=size[x];
	}
	else
	{
		arr[y]=arr[x];
		size[x]+=size[y];
	}
}
void init(int arr[],int size[],int n)
{
	for(int i=1;i<=n;i++)
	{
		arr[i]=i;
		size[i]=1;
	}
}
int main()
{
   // cout << "Hello World!" << endl;
    int n,k;
    cin>>n >>k;
    int arr[n],size[n];
    init(arr,size,n);
    int x,y;
    for(int i=0;i<k;i++)
    {
    	cin>>x >>y;
    	union1(arr,size,x,y);
    }
    bool b[n];
    for(int i=1;i<=n;i++)
    {
    	b[i]=false;
    }
    int count=0;
    for(int i=1;i<=n;i++)
    {
    	int r=root(arr,i);
    	if(b[r]==false)
    	{
    		count++;
    		b[r]=true;
    	}
    		
    }
    cout<<count<<endl;
    return 0;
}
