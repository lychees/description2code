#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    for(;t>0;t--)
    {
          int flag=0,i=0;
          char s[10000],r[10000];
          scanf("%s", s);
          
          int l=strlen(s);

         for(i=0;i<l;i++)
         {
             r[i]=s[l-1-i];
         }
 
         for(i=1;i<l;i++)
         {
              if(abs((int) s[i] - (int) s[i-1])!= abs((int) r[i] - (int) r[i-1]))
              {
                      flag=1;
                      break;
               }
         }

         if(flag==1)
         cout << "Not Funny" << endl;

         else
         cout << "Funny" << endl;
   }
return 0;
}
