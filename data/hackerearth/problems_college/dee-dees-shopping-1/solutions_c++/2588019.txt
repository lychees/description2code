
#include<bits/stdc++.h>

using namespace std;

int fuc(int n, int cn1, int cn2){

if(n == 0){

return 0;
}
else if( n < 6) {

return 100;
}

else {
int x =    min(   ( fuc(n-cn1,cn1,cn2 ) + 1) , ( fuc( n - cn2 , cn1 , cn2 ) + 1) )   ;

x = min ( x , ( fuc ( n - cn1 - cn2 , cn1, cn2) + 2));
return x;
}
}

int main(){

int t,n;
cin>>t;
while(t--){

cin>>n;

int cn1 = 8, cn2 = 6;


if( n % 2 != 0)
cout<<"-1"<<endl;
else {
int res = min(  min(   ( fuc(n-cn1,cn1,cn2 ) + 1) , ( fuc( n - cn2 , cn1 , cn2 ) + 1) ) ,  ( fuc ( n - cn1 - cn2 , cn1, cn2) + 2) );

if(res >= 100)
cout<<"-1"<<endl;
else
cout<<res<<endl;
}
}
return 0;
}
