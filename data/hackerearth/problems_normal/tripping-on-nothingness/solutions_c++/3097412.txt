#include <iostream>
#include <algorithm>
#define M 100010

using namespace std;


bool Set[M];

int main()
{
        int S, Q;
        cin>>S>>Q;

        for(int i=0; i<=S; i++)
            Set[i]=false;

        int n;
        while(Q--)
        {
            cin>>n;
            n=__gcd(S, n);
            if(Set[n])
                cout<<"-1\n";
            else
            {
                cout<<n<<endl;
                Set[n]=true;
            }
        }


}
