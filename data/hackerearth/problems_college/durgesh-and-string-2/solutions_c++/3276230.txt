#include<bits/stdc++.h>
#define sz(x) (int)(x.size())
 
using namespace std;
typedef long long lld;
 
 
int nO, nZ;
vector<string> vec;
int n; 
 
int possible(int bitmask)
{
	int one = nO;
	int zero = nZ;
	for(int i = 0; i < 32; i++)
	{
		if(bitmask &(1<<i))
		{
			for(int j = 0; j < sz(vec[i]); j++)
			{
				if(vec[i][j] == '0') zero--;
				else one--;
			}
		}
	}
	if(one < 0 || zero < 0) return 0;
	return 1;
		
}
 
int main()
{
	//freopen("Input8.txt", "r", stdin);
	//freopen("Output8.txt", "w", stdout);
	//cout << "kya be" << endl;
	scanf("%d\n", &n);
	
	//cout << "FF " << endl;
	for(int i = 0; i < n; i++)
	{
	   string s;
	   cin >> s;
	   vec.push_back(s);	
	}
	cin >> nO >> nZ;
	
	//cout << "Input" << endl;
	int ans = 0;
	for(int i = 0; i < (1<<n); i++)
	{
	    if(possible(i))
	    {
			int bui = __builtin_popcount(i);
			ans = max(ans, bui);
		}
	}
	cout << ans << endl;
	// csdkflfjslfjs
	return 0;
}
