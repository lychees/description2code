#include <iostream>
#include<string.h>
#include<stdlib.h>
using namespace std;
int vis[10]={0};
char a[10][20];
int minlen=200;
int n;
void fill(int in,char b[200],int len)
{
//	cout<<"\nreceived"<<in<<"\t";
//	for(int i=0;i<len;i++)
//	   cout<<b[i];
//	cout<<"\n";
	vis[in]=1;
	int ff=0;
	int akas=strlen(a[in]);
//	cout<<" in akas "<<in<<" "<<akas<<"\n";
	int abk=len;
	for(int l=max(0,len-akas);l<len;l++)
	{
		if(b[l]==a[in][0])
		{
			int fff=0;
			for(int pp=l+1;pp<len;pp++)
			{
				if(b[pp]!=a[in][pp-l])
				{
					pp=len;
					fff=1;
				}
			}
//			cout<<l<<" "<<fff;
			if(fff==0)
			{
				for(int kk=len;kk<akas+l;kk++)
				{
					b[kk]=a[in][kk-l];
					abk++;	
				}
				ff=1;
				l=len;
//				cout<<" ab abk is "<<abk<<"\n";
			}
//			cout<<"\n";
		}
	}
	if(ff==0)
	{
		for(int kk=len;kk<akas+len;kk++)
		{
			b[kk]=a[in][kk-len];
			abk++;
		}
	}
//	cout<<" ab len is "<<abk<<"\t"<<ff<<"\n";
	int c=0;
	for(int i=0;i<n;i++)
	{
		if(vis[i]==1)
		{
			c++;		
		}
		else
		{
			vis[i]=1;
			fill(i,b,abk);
			vis[i]=0;
		}
	}
	if(c==n)
	{
	    minlen=min(minlen,abk);
//	   cout<<"min"<<minlen<<"\n";
	   
	}
	vis[in]=0;
	
}
int main()
{
	scanf("%d",&n);
	char b[200];
	for(int k=0;k<200;k++)
	      b[k]=' ';
	for(int j=0;j<n;j++)
	{
		scanf("%s",a[j]);
	}
	
	for(int j=0;j<n;j++)
	{
//		cout<<"j "<<j<<"\n";
		fill(j,b,0);
	}
	cout<<minlen<<"\n";
    return 0;
}
