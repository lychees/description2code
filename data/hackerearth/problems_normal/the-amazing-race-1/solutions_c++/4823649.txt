#include <iostream>
#include <stack>

using namespace std;

int main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		int n;
		cin>>n;
		
		long long int arr[n], nl, nr, s, max, max_i, left[n], right[n];
		
		stack<int> l, r;
		
		for(int i = 0; i < n; ++i)
			cin>>arr[i];
			
		for(int i = 0; i < n; ++i)
		{
			
			if(l.empty())
			{
				l.push(i);
				nl = 0;
			}
			else
			{
				while(!l.empty() && arr[l.top()] < arr[i])
				{
					l.pop();
				}
				
				if(l.empty())
				{
					nl = i;		
				}
				else
				{
					nl = i - l.top();	
				}
				l.push(i);
			}
			
			if(r.empty())
			{
				r.push(n - i - 1);
				nr = 0;
			}
			else
			{
				while(!r.empty() && arr[r.top()] < arr[n - i - 1])
				{
					r.pop();
				}
				
				if(r.empty())
				{
					nr = i;		
				}
				else
				{
					nr = r.top() - n + i + 1;	
				}
				r.push(n - i - 1);
			}
			
			left[i] = nl;
			right[n - i - 1] = nr;
		}
		
		
		for(int i = 0; i < n; ++i)
		{
			s = ((i + 1) % 1000000007 * (left[i] + right[i]) % 1000000007) ;
			
			
			//cout<<"i : "<<i<<" left : "<<left[i]<<" right : "<<right[i]<<" s : "<<s<<endl;
			
			if(i == 0)
			{
				max = s;
				max_i = i;
			}
			else
			{
				if(max < s)
				{
					max = s;
					max_i = i;
				}
			}
		}
		cout<<max_i + 1<<endl;
	}
    return 0;
}
