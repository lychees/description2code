def main():
	t = int(raw_input().strip())
	
	while t!= 0:
		stk = []
		S = raw_input()
		stk.append(S[0])
		for i in xrange(1,len(S)):
			temp = stk.pop()
			if temp == S[i]:
				stk.append(temp)
			else:
				stk.append(temp)
				stk.append(S[i])
		print ''.join(stk)
		t -= 1

if __name__ == '__main__':
	main()
