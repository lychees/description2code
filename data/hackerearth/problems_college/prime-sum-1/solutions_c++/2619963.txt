#include <bits/stdc++.h>

using namespace std;

#define ll long long int
#define s(x) scanf("%d", &x)
#define sd(x) scanf("%lf", &x)
#define sl(x) scanf("%lld", &x)
#define pb push_back
#define mp make_pair
#define INF 100000000

int dp[1000005];

int main()
{
    dp[0] = dp[1] = -1;
    dp[2] = dp[3] = dp[5] = dp[7] = 1;
    dp[4] = dp[6] = dp[8] = dp[9] = 2;
    for (int i = 10; i <= 1000000; i++)
        dp[i] = min(min(dp[i-2],dp[i-3]),min(dp[i-5],dp[i-7]))+1;
    int t, n;
    s(t);
    while (t--) {
        s(n);
        printf("%d\n", dp[n]);
    }
    return 0;
}
