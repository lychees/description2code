#include <bits/stdc++.h>
using namespace std;
#define PII pair <long double,int>
vector <PII> edges; int area[11]; map <int,int> hit; long double maxi=0; 
vector <int> z; vector <int> daans; //map <int,int> missile; 
void dfs (long double strength, long double ans)
{
    
    if (z.size()==9)
    {
    	//for (int g=0;g<z.size(); g++) cout << z[g] << ' '; cout << '\n' << ans <<'\n'; 
        if (ans>maxi)
        {
            maxi=ans; daans=z; 
        }
        return; 
    }
    for (int g=0; g<edges.size(); g++)
    {
        if (hit[edges[g].second]) continue;
        if (hit[edges[g].first])continue;
        hit[edges[g].second]=1; 
        z.push_back(edges[g].second); 
        dfs(strength*pow(11,-2),ans+(strength/double(area[edges[g].second]))); 
        z.erase(z.end()-1); 
        hit[edges[g].second]=0;
    }
}
int main()
{   
    for (int g=1; g<=10; g++)
    {
        int z=(g+3)%10; 
        int zz=(g+7)%10; if (z==0) z=10; if (zz==0) zz=10;
        edges.push_back(PII(g,zz)); edges.push_back(PII(g,z)); 
    }
    for (int g=1; g<=10; g++) cin >> area[g]; 
    dfs(100000, 0); 
    for (int g=0; g<daans.size(); g++) cout << area[daans[g]] << ' '; 
    return 0; 
 
}
