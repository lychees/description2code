#include <iostream>
using namespace std;
#define ll long long 
ll a[1002];
ll fun(ll n, ll m)
{
	ll i,j,k;
	ll sum=0;
	for(i=0;i<n-2;i++)
	{
		for(j=i+1;j<n-1;j++)
		{
			for(k=j+1;k<n;k++)
			{
				sum = a[i]+a[j]+a[k];
				if(sum==m)
				{
					printf("YES");
					return 1;
				}
			}
		}
	}
	return 0;
}
int main()
{
	ll n, m,i, flag=0;
	scanf("%lld%lld", &n, &m);
	ll  sum;
	for(i=0;i<n;i++)
		scanf("%lld", &a[i]);
	flag=fun(n,m);
	if(flag==0)
		printf("NO");
	return 0;
}
