#include<iostream>
#include<cstdio>
#include<math.h>

using namespace std;
int arr[100001];
void sieve()
{
           int i,j,count=0;
           for(i=1;i<=100001;i++)
           {
                      arr[i]=1;
           }
           arr[1]=0;

           for(i=2; i<=sqrt(100001); i++)
           {
                      for(j=2; i*j<=100001; j++)
                      {
                                     arr[i*j] = 0;
                      }
           }
}
int main()
{
           int t;
           cin>>t;
           sieve();
           while(t--)
           {
                      int a;
                      cin>>a;
                      if(a == 3) cout<<3<<endl;
                      else if(arr[a] == 1)
                      cout<<"2"<<endl;
                      else cout<<1<<endl;
           }
           return 0;
}
