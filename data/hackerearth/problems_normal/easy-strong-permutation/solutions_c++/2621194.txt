#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int test;
    vector<long long int> vect;
    long long int y;
    cin>>test;
    for(int i=0;i<test;i++)
    {
    	cin>>y;
    	vect.push_back(y);
    }
    sort(vect.begin(),vect.end());
    long long int z=0;
    for(int i=0;i<test/2;i++)
    {
    	z=z+vect[test-i-1]-vect[i];
    }
    if(test%2!=0)
    z=z+vect[test/2+1];
    cout<<z*2<<endl;
    return 0;
}
