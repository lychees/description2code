#include<iostream>
#include<string>
#include<algorithm>
#include<math.h>


using namespace std;

int main()
{
      long long t,q,count,z,m,n,rem,min,temp,i;
      cin>>t;
      for(q=0; q<t; q++)
      {
          count=0;
          cin>>z>>m>>n;
          if(m<n)
          {
              min=m;
          }
          else
          {
              min=n;
          }
          temp=sqrt(z);
          for(i=1; i<=min && i<=temp; i++)
          {
              if(z%i==0)
              {
                  rem=z/i;
                  if(i==rem)
                  {
                      count++;
                  }
                  else
                  {
                  if(i<=m && rem<=n)
                  {
                      count++;
                  }
                  if(i<=n && rem<=m)
                  {
                      count++;
                  }
                  }


              }
          }
          cout<<count<<endl;
      }





    return 0;

}
