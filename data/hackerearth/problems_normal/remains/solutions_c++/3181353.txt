#include<iostream>
#include<cmath>
using namespace std;
int main(){
    long long int t,x,y,n,i,sum,temp;
    cin>>t;
    while(t--){
        sum=0;
        cin>>x>>y>>n;
        if(x==0){
           if(n%3!=0)
           sum = (n/3)*2*y+(n%3-1)*y;
           else
                sum = (n/3)*2*y;
           cout<<sum<<endl;
           continue;
        }
        sum = x+y;
        for(i=3;i<=n;i++){
            if(y==0){
                sum+=(((n-i+1)/3)*2+(n-i+1)%3)*x;
                break;
            }
            temp = abs(x-y);
            sum += temp;
            x=y;
            y = temp;

        }
        cout<<sum<<endl;
    }
    return 0;
}
