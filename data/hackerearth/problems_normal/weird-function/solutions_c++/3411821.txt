#include <iostream>
using namespace std;
#define mo 1000000007
long long int ans[10001]={0};
long long powe(int a, int b, int MOD)
{
    long long x=1,y=a; 
    while(b > 0)
    {
        if(b%2 == 1)
        {
            x=(x*y);
            if(x>MOD) x%=MOD;
        }
        y = (y*y);
        if(y>MOD) y%=MOD; 
        b /= 2;
    }
    return x;
}
 
int main()
{
	int i,j,a,b;
	//for(i=1;i<=10000;i++)
	//ans[i]=(i*(i+1))/2;
	for(i=1;i<=10000;i++)
		for(j=i;j<=10000;j+=i)
			ans[j]=(ans[j]+powe(i,j,mo))%mo;
	int t;
	cin >> t;
		for(i=2;i<=10000;i++)
		ans[i]=(ans[i]+ans[i-1])%mo;
	while(t--)
	{
		cin >> a >> b;
		cout << (ans[b]-ans[a-1]+mo)%mo << endl;
	}
	
	
    return 0;
}
