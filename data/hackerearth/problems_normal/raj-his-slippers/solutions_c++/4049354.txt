#include<bits/stdc++.h>
using namespace std;
#define in(n) scanf("%d",&n)
int main()
{
    int t,n,m,h,in,up,q,op,r,s,i,a,mx,ans;
    in(t);
    q=t;
    string str;
    while(t--)
    {
        in(n);
        vector<int> v(5);
        cin >> str;
        v[0]=str[0]-48,v[1]=str[1]-48,v[3]=str[3]-48,v[4]=str[4]-48;
        m=v[4]+10*v[3],h=v[1]+10*v[0];
        in=m+60*h;
        cin >> str;
        v[0]=str[0]-48,v[1]=str[1]-48,v[3]=str[3]-48,v[4]=str[4]-48;
        m=v[4]+10*v[3],h=v[1]+10*v[0];
        up=m+60*h;
        cin >> str;
        v[0]=str[0]-48,v[1]=str[1]-48,v[3]=str[3]-48,v[4]=str[4]-48;
        m=v[4]+10*v[3],h=v[1]+10*v[0];
        op=m+60*h;
        in(r);
        in(s);
        vector<int> st(n),en(n);
        for(i=0;i<n;i++)
        {
            cin >> str;
            v[0]=str[0]-48,v[1]=str[1]-48,v[3]=str[3]-48,v[4]=str[4]-48;
            m=v[4]+10*v[3],h=v[1]+10*v[0];
            st[i]=m+60*h;
            cin >> str;
            v[0]=str[0]-48,v[1]=str[1]-48,v[3]=str[3]-48,v[4]=str[4]-48;
            m=v[4]+10*v[3],h=v[1]+10*v[0];
            en[i]=m+60*h;
        }
        if(up+2*r+s>=in || op+r+s>=in)
            ans=-1;
        else
        {
            a=max(up,op-r),mx=1440,ans=-1;
            for(i=0;i<n;i++)
            {
                if(a+2*r+s<st[i])
                    m=a+2*r+s;
                else if(a>=en[i])
                    m=a+2*r+s;
                else
                    m=en[i]+2*r+s;
                if(m<mx && m<=in)
                    ans=i+1,mx=m;
            }
        }
        cout << "Case " << q-t << ": " << ans << endl;
    }
    return 0;
}
