int dp[1001];
int price[1001];
int n,x;

int go(int idx){
	if(idx==0)
	return 0;
	if(dp[idx]!=-1)
	return dp[idx];
	int ret=0;
	for(int i=1;i<=idx;i++){
		ret=max(ret,price[i]-x+go(idx-i));
	}
	dp[idx]=ret;
	return ret;
}

	cin>>price[i];

	}

}
