#include <bits/stdc++.h>
using namespace std;

int main() {
  long int u, t, i, j, k, l, n, o, p, f;
  int a[] = {2, 3, 5, 7, 11, 13, 17, 19};
  cin >> t;
  while (t--) {
    f = 0;
    cin >> n;
    for (i = 0; i < 8; i++) {
      j = a[i];
      k = n;
      l = j;
      o = 0;
      while (k != 0) {
        p = k / l;
        k = k / l;
        o += p;
        l *= j;
      }

      if (o >= f) {
        u = a[i];
        f = o;
      }
    }

    cout << u << "\n";
  }
}
