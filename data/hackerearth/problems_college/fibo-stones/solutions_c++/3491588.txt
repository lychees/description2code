#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    ll f[85];
    f[1] = f[2] = 1;

    for (int i = 3; i < 85; i++) {
        f[i]  = f[i - 1] + f[i - 2];
    }

    ll n, m, x, cnt = 0;

    cin >> n >> m;
    for (ll i = 0; i < m; i++) {
        cin >> x;
        cnt += x;
        //f[x] = 0;
        //cout << x << " " << f[x] << endl;
    }
    ll sum = 0;
    for (ll i = 1; i <= n; i++) {
        sum += f[i];
        //cout << f[i] << " ";
    }
    cout << sum - cnt << endl;

    return 0;
}
