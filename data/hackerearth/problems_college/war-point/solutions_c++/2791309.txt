#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin >> t;
	while ( t -- )
	{
		long long int n , sum=0;
		scanf("%lld",&n);
		vector<long long int> v1(n) , dp(n);
		for( int i=0 ; i < n ; i++) {
			scanf("%lld",&v1[i]);
			sum+=v1[i];
		}
		dp[0] = v1[0];
		for( int i = 1 ; i < n ; i++ ) {
			dp[i] = dp[i-1] + v1[i] ;
		}
		long long int max1 = INT_MAX;
		for( int i = 0 ; i < n ; i++ ) {
			max1 = min(abs(dp[i]-(sum-dp[i])), max1);
		}
		
		 printf("%lld\n",max1);
	}
	return 0;
}
