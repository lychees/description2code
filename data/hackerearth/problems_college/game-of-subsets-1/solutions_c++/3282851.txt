#include <iostream>
#include<cmath>
using namespace std;
#define mod 1000000007
long long pow_mod(long long b,long long e)
{
	long long ans=1;
	b%=mod;
	while(e>0)
	{
		if(e&1)
			ans=(ans*b)%mod;
			b=(b*b)%mod;
			e>>=1;
	}
	return ans%mod;
}
