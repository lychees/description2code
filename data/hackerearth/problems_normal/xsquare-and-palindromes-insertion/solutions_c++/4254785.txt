#include<stdio.h>
#include<string.h>
int main()
{
	int tcases,i;
	char s[100000];
	
	scanf("%d",&tcases);
	while(tcases--)
	{
		int c[100000]={0};
		scanf("%s",s);
		int len=strlen(s);
		for(i=0;i<len;i++)
		{
			c[s[i]-'a']++;
		}
		int count=0;
		for(i=0;i<len;i++)
		{
			if(c[i]%2!=0)
			{
				count++;
			}
		}
		if(count==0)
		{
			printf("0\n");
		}
		else
		{
			printf("%d\n",count-1);
		}
	}
}
