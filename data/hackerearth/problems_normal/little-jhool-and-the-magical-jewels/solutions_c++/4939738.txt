#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    
    int t;
    cin>>t;
    while(t--)
    {
      string str;
      cin>>str;
      int n=str.length();
      int count[26];
      memset(count,0,sizeof(count));
      for(int i=0;i<n;i++)
      {
      	count[str[i]-'a']++;
      }
      int val=min(count['r'-'a'],min(min(count['u'-'a'],count['b'-'a']),count['y'-'a']));
      cout<<val<<endl;
    }
    return 0;
}
