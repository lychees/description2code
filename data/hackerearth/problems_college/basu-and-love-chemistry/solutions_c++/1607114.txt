#include <bits/stdc++.h>
using namespace std;
#define ll long long int
#define mod 1000000007
int main() {
    char arr[200007];
	int i,j,len;
	ll ctr=0, seq=0;
	cin>>arr;	
	len = strlen(arr);
	for(i=0;i<len;i++) {
		j = i+1;
		ctr = ctr*10;		
		seq += j*(arr[i]-'0');	
		ctr += seq;
		ctr = ctr%mod;
	}
	cout<<ctr<<endl;
}
