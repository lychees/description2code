#include<bits/stdc++.h>
using namespace std;
  
  
int main()
{
    long long N,l,r,sum;
    cin>>N;
    long long a[N];
      
    for(int i=0;i<N;i++)
        cin>>a[i];
    cin>>sum;
    sort(a,a+N);
      
    bool flag=false;
      
    for (int i = 0; i < N - 2; i++)
    {
   
        l = i + 1; 
        r = N-1;
        while (l < r)
        {
            if( a[i] + a[l] + a[r] == sum)
            {
                flag=true;
                break;
            }
            else if (a[i] + a[l] + a[r] < sum)
                l++;
            else
                r--;
        }
        if(flag)
            break;
    }
      
    if(flag)
    cout<<"True\n";
    else
    cout<<"False\n";
   
  
      
    return 0;
}
