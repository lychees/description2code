#include <iostream>
using namespace std;
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int ar[21][3];
   	int dp[21][3];
   	int no;
   	cin>>no;
   	for(int i=0;i<no;i++)
   	{
   		for(int j=0;j<3;j++)
   		    cin>>ar[i][j];
   	}
   
   	dp[0][0]=ar[0][0];
   	dp[0][1]=ar[0][1];
   	dp[0][2]=ar[0][2];
   	for(int i=1;i<no;i++)
   	{
   		dp[i][0]=min(dp[i-1][1],dp[i-1][2])+ar[i][0];
   		dp[i][1]=min(dp[i-1][0],dp[i-1][2])+ar[i][1];
   		dp[i][2]=min(dp[i-1][0],dp[i-1][1])+ar[i][2];
   	}
   	cout<<min(min(dp[no-1][0],dp[no-1][1]),dp[no-1][2])<<endl;
   	
   }
    return 0;
}
