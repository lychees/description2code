#include <bits/stdc++.h>
#define MAX 100005
using namespace std;
int arr[MAX];
vector <pair<int,int> > adj[MAX];
int intt(int nodes){
	for (int i=0;i<=nodes;i++)
		arr[i]=i;
}

int root(int x){
	while(x!=arr[x]){
		//arr[x]=arr[arr[x]];
		x=arr[x];
	}
	return x;
}
int union1(int a,int b){
	arr[root(a)]=arr[root(b)];
}

int main()
{	
    int nodes,edges,x,y,q;
    scanf("%d %d",&nodes,&edges);
    intt(nodes);
    for(int i=0;i<edges;i++){
    	scanf("%d %d",&x,&y);
    	union1(x,y);
    }
    scanf("%d",&q);
    while(q--){
    	scanf("%d %d",&x,&y);
    	if(arr[x]==y || arr[y]==x)
    		printf("Direct Connection\n");
    	else if(root(x)==root(y))
    		printf("Indirect Connection\n");
    	else
    		printf("No Connection\n");
    }
    
    return 0;
}
