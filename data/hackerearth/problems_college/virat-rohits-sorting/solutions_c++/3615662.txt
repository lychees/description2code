#include<bits/stdc++.h>
using namespace std;
pair<int,int> ar[1000100];
int br[1000100];
int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        scanf("%d%d",&ar[i].first,&ar[i].second);
    }
    sort(ar,ar+n);
    int top=0;
    for(int i=0;i<n;i++)
    {
        while(i<n-1&&ar[i].first==ar[i+1].first)
        {
            br[top++]=ar[i].second;
            i++;
        }
        br[top++]=ar[i].second;
        for(int j=top-1;j>=0;j--)
        {
            printf("%d %d\n",ar[i].first,br[j]);
        }
        top=0;
    }

}
