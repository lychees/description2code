#include <iostream>
using namespace std;

int main(){
	string s;
	while(getline(cin,s)){
		int flag=0;
		int len = s.length();
		//cout<<len<<endl;
		for(int i=0;i<len-1;++i){
			if(s.at(i)=='/' && s.at(i+1)=='/'){
				break;
			}
			else{
				if(s.at(i)=='-' && s.at(i+1)=='>'){
					++flag;
					s.at(i)='.';
					for(int j=i+1;j<len-1;++j){
						s.at(j)=s.at(j+1);
					}
				}
			}
		}
		for(int k=1;k<=flag;++k){
			s.erase(s.begin()+len-k);
		}
		cout<<s<<endl;
	}
	return 0;
}
