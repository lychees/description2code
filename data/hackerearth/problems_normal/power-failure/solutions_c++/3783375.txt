#include <stdio.h>
#include <vector>
#include <iostream>
#include <algorithm>
#define M 100005
#define mod 1000000007
using namespace std;
 
int pre[M],b[M];
 
int main()
{
	long int i,j;
	int t,n,m,arr[M];
	for(i=0;i<M;i++)
		b[i]=i;
	b[0]=b[1]=pre[0]=0;
	for(i=2;i<M;i++)
		if(b[i])
			for(j=i*i;j<M;j+=i)
				b[j]=0;
	
	for(i=1;i<M;i++)
	{
		pre[i]=pre[i-1];
		if(!b[i])	pre[i]++;
	}
 
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&m);
		for(i=1;i<=n;i++)
			scanf("%d",&arr[i]);
		sort(arr+1,arr+n+1);
		long long int ans=1;    
		for(i=1;i<=n;i++)
		{
			j=pre[m]-pre[arr[i]-1];
			j-=(n-i);                    
			if(j<=0)
			{
				ans=0;
				break;
			}
			ans=(ans*j)%mod;
		}
		printf("%lld\n",ans);
	}
	return 0;
}
