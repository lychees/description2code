#include <iostream>
using namespace std;

int main()
{
    string a,b;
    int i=0,c=0;
    cin>>a>>b;
    
    for(i=0;i<a.size();i++){
    	if(a[i]!=b[i]){
    		c++;
    	}
    }
    
    if(c==1 || c==0){
    	cout<<"wins"<<endl;
    }else{
    	cout<<"loses"<<endl;
    }
    
    
    return 0;
}
