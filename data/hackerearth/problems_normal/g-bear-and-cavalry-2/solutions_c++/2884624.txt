void solve(void) {
  int n;
  scanf("%d", &n);
  vector<int> h(n), w(n);
  for (int i = 0; i < n; ++i)
    scanf("%d", &w[i]);
  for (int i = 0; i < n; ++i)
    scanf("%d", &h[i]);
  int id = 0;
  for (int i = 1; i < n; ++i)
    if (h[i] > h[id]) id = i;
  int mm = w[0]*h[id];
  w.erase(w.begin());
  h.erase(h.begin()+id);
  sort(w.begin(), w.end());
  sort(h.begin(), h.end(), greater<int>());
  for (int i = 0; i < n-1; ++i)
    if (mm <= w[i]*h[i]) {
      printf("NO\n");
      return;
    }
  printf("YES\n");
}

int main(void) {

  int t;
  scanf("%d", &t);
  for (; t; --t) solve();

  return 0;
}
