#include <bits/stdc++.h>
using namespace std;
void swap(char *a,char *b)
{
	char t;
	t=*a;
	*a=*b;
	*b=t;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	int n=str.length();
    	if(str.find_first_not_of(str[0])==string::npos)
    		cout<<"no answer\n";
    	else
    	{
    	for(int i=n-1;i>0;i--)
    	{
    		if(i>1)
    		{
    			if(str[i]>str[i-1])
    			{
    				
    			swap(&str[i],&str[i-1]);
    			cout<<str<<endl;
    			break;
    			}
    		}
    		else 
    		{
    			char temp=str[0],t;
    			string str1=str;
    			char str2[n];
    			sort(str1.begin(),str1.end());
    			for(int i=0;i<n;i++)
    			{
    				if(str1[i]==temp)
    				{
    					 t=str1[i+1];
    					 break;
    				}
    			}
    			for(int i=0;i<n;i++)
    			{
    				if(str[i]==t)
    				{
    					 swap(&str[0],&str[i]);
    					 break;
    				}
    			}
    			for(int i=1;i<n;i++)
    				str2[i-1]=str[i];
    			/*for(int i=0;i<n-1;i++)
    				cout<<str2[i];*/
    			sort(str2,str2+(n-1));
    			cout<<str[0];
    			for(int i=0;i<n-1;i++)
    				cout<<str2[i];
    			cout<<endl;
    		}}
    		
    	}
    	
    }
    return 0;
}
