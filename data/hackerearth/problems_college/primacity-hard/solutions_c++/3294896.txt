int pc[10000002]={0};
vector<int> v[10];

void sieve() {
    for(int i=2;i<=10000000;++i) {
        if(!pc[i]) {
            for(int j=1;i*j<=10000000;++j) {
                pc[i*j]++;
            }
        }
    }
}

void func() {
    for(int i=1;i<=8;++i) {
        v[i].clear();
    }
    for(int i=1;i<=10000000;++i) {
        if(pc[i]) {
            v[pc[i]].pb(i);
        }
    }
}

int main()
{
    sieve();
    func();
    int mx = 0;
  //  for(int i=0;i<=10000000;++i) {

//    cout<<mx;

    int t;
    si(t);
    while(t--) {
        int a,b,k;
        si(a); si(b); si(k);
        if(k>8) {
            pri(0); nl;
            continue;
        }
        int ans = (upper_bound(v[k].begin(),v[k].end(),b)-v[k].begin()) - (lower_bound(v[k].begin(),v[k].end(),a)-v[k].begin());
        pri(ans); nl;
    }
    return 0;
}
