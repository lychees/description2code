#include <iostream>
#include <math.h>
#define ll long long
ll p[1000001]={0};
using namespace std;

int main()
{
	ll t, num, a, b, n, m, f;
    cin>>t;p[1]=1;
    for(a=1;a<1000;a++)
    {
    	for(n=2;n<=20;n++)
    	{
    		if(pow(a,n)<=1000000)
    			p[(ll)pow(a,n)]=1;
    		else
    			break;
    	}
    }
    while(t--)
    {
    	f=0;
    	cin>>num;
    	for(b=1;b<1000;b++)
    	{
    		for(m=2;m<=20;m++)
    		{
    			if(pow(b,m)<=num)
    			{
    				if(p[abs(num-(ll)pow(b,m))]==1)
    				{
    					f=1;break;
    				}
    			}
    			else
    				break;
    		}
    		if(f==1)
    			break;
    	}
    	if(f==1)
    		cout<<"Yes"<<endl;
    	else
    		cout<<"No"<<endl;
    }
    return 0;
}
