#include <bits/stdc++.h>
using namespace std;
char maps(int n)
{
	if(n==10)
	return 'U';
	if(n==11)
	return 'V';
	if(n==12)
	return 'W';
	if(n==13)
	return 'X';
	if(n==14)
	return 'Y';
	if(n==15)
	return 'Z';
}

int main()
{
    int t,r,g,b,c;
    string res="";
    cin>>t;
    int i=1;
    while(t--)
    {
    	cin>>r>>g>>b;
    	c=0;
    	res="";
    	cout<<"Case "<<i<<": ";
    	while(r)
    	{
    		c=r%16;
    		if(c>9)
    		res+=maps(c);
    		else
    		res+=(c+48);
    		r=r/16;
    	}
    	if(res.length()!=2)
        {
            while(res.length()!=2)
                res+='0';
        }
    	reverse(res.begin(),res.end());
    	cout<<res<<" ";
    	res="";
    	while(g)
    	{
    		c=g%16;
    		if(c>9)
    		res+=maps(c);
    		else
    		res+=(c+48);g=g/16;
    	}
    	if(res.length()!=2)
        {
            while(res.length()!=2)
                res+='0';
        }
    	reverse(res.begin(),res.end());
    	cout<<res<<" ";
    	res="";
    	while(b)
    	{
    		c=b%16;
    		if(c>9)
    		res+=maps(c);
    		else
    		res+=(c+48);b=b/16;
    	}
    	if(res.length()!=2)
        {
            while(res.length()!=2)
                res+='0';
        }
    	reverse(res.begin(),res.end());
    	cout<<res<<endl;

i++;
    }
    return 0;
}
