void solve()
{
	int n;
	cin>>n;
	int a[1000001], b[1000001];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		b[i] = 0;
	}
	int max_end_here = 0, max_so_far = 0;
	for(int i=0;i<n;i++)
	{
		if(max_end_here + a[i] > 0)
			max_end_here += a[i];
		else
			max_end_here = 0;
		if(max_so_far < max_end_here)
			max_so_far = max_end_here;
	}
	cout<<max_so_far<<endl;
	
}
