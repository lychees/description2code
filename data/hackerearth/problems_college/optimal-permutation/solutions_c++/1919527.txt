#include<bits/stdc++.h>
using namespace std;
#define ll long long 
int main()
{
	ios :: sync_with_stdio(0);
	ll t;
	scanf("%lld",&t);
	while(t--)
	{
		ll n;
		scanf("%lld",&n);
		ll count=0;
		if(n%2==0 )
		{
			count+=n-1;
			ll ti=(n-2)/2;
			ll sum=ti*(4+(ti-1)*2);
			count+=sum;
			printf("%lld\n",count);
		}
		else
		{
			if(n==1)
			printf("%lld\n",count);
			else
			{
			count+=n;
			ll ti=(n-3)/2;
			ll sum=ti*(6+(ti-1)*2);
			count+=sum;
			printf("%lld\n",count);
		}
		}
		
		
	}

	return 0;
}
