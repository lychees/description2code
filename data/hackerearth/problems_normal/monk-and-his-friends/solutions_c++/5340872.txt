#include <iostream>
#include<unordered_set>
using namespace std;
int main(){
  int T;
  cin>>T;
   while(T--){
    long long N,M;
    cin>>N>>M;
    unordered_set<long long> s;
    unordered_set<long long>::iterator it;
    while(N--){
      long long x;
      cin>>x;
      s.insert(x);
    }
    while(M--){
     long long x;
     cin>>x;
     it=s.find(x);
     if(it!=s.end()){
       cout<<"YES\n";
     }
     else{
       cout<<"NO\n";
     }
     s.insert(x);
    }
}
return 0;
}
