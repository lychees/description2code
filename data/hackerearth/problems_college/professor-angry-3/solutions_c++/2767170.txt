#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	int n, k;
    	cin>>n>>k;
    	
    	while(n--){
    		int a;
    		cin>>a;
    		
    		if(a<=0){
    			//cout<<a<<" "<<k<<endl;
    			k--;
    		}
    	}
    	
    	//cout<<k<<endl;
    	
    	if(k>0){
    		cout<<"YES"<<endl;
    	}else{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
