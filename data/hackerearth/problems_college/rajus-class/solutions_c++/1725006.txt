#include<iostream>
#include<cmath>
#include<map>
#include<cstdio>
#define gc getchar
using namespace std;
typedef long long int ll;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
int main()
{
    ll t,n,r,i,a;
    string nm;
    cin>>t;
    while(t--)
    {
        n=read_int();
        multimap<ll,string> m;
        multimap<ll,string>::iterator it;
        for(i=0;i<n;i++)
        {
            cin>>a>>nm;
            m.insert(make_pair(a,nm));
        }
        r=read_int();
        it=m.begin();
        while(r!=0)
        {
            it++;
            r--;
        }
        it--;
        //for(it=m.begin();it!=m.end();++it)
            cout<<(*it).second<<'\n';
    }
    return 0;
}
