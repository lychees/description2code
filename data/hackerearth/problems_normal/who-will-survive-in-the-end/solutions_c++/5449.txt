#include <iostream>
#include<math.h>
using namespace std;


int check(long a)
{
    int i=1,j=0;
    while(i<a)
    {
        i=i<<1;
        j++;
    }
    j--;
    i=i>>1;
    //cout<<i<<endl;
    a=a-i;
    a=2*a+1;
    return a;


}

int main()
{
