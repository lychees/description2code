#include<iostream>
using namespace std;

int main()
{
    int t,m,n,u[10000],v[10000],w[10000],x[10000],g,gr,i,j,good,great;
    cin>>t;
    while(t--)
    {
        good=great=0;
        cin>>m>>n;
        for(i=0;i<m;i++)
            cin>>u[i]>>v[i];
        for(i=0;i<m;i++)
            cin>>w[i]>>x[i];
        cin>>g>>gr;
        for(i=0;i<n;i++)
        {
            for(j=0;j<m;j++)
            {
                if(u[i]==w[j])
                {
                    good++;
                    if(v[i]==x[j])
                        great++;
                }
            }
        }
        if(g==good && gr==great)
            cout<<"Great"<<endl;
        else if(g==good)
            cout<<"Good"<<endl;
        else
            cout<<":("<<endl;
    }
}
