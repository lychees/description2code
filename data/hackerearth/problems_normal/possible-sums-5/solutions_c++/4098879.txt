#include <bits/stdc++.h>

using namespace std;

int main()
{
    int tc;
    cin >> tc;

    while(tc--){
    int n;
    cin >> n;

    int a[n];

    for (int i = 0;i < n;i++)
        cin >> a[i];

    int dp[10005] = {0};
    dp[0] = 1;

    for (int i = 0;i < n;i++)
    {
        for (int j = 10000;j >= 0;j--)
        {
            if (dp[j])
                dp[j+a[i]] = 1;
        }
    }

    int count = 0;
    for (int i = 0;i <= 10005;i++)
    {
        if (dp[i] == 1)
            count++;
    }

    cout << count << endl;
    }

    return 0;
}
