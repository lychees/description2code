
#include <bits/stdc++.h>
//PERFECT COPY :D
using namespace std;
int leftdp[200002];
int rightdp[200002];
long int a[200002];
int main()
{
        int t;
        cin >> t;
        while(t--)
        {
            int n;
            cin >> n;
            int i;
            for(i = 1 ;i <=n; i++)
            {
                cin >> a[i];
            }
            for(i = 1 ; i <= n ; i++)
            {
                if(a[i] >= 0)
                {
                    leftdp[i] = leftdp[i - 1] + 1;
                }
                else
                    leftdp[i] = leftdp[i - 1];
            }
            for(i = n ; i >= 1 ; i--)
            {
                if(a[i] <= 0)
                {
                    rightdp[i] = rightdp[i + 1] + 1;
                }
                else
                    rightdp[i] = rightdp[i + 1];
            }
            int ans = INT_MAX;
            for(i = 1 ; i < n ;i++)
            {
              ans = min(ans,leftdp[i] + rightdp[i + 1]);
            }
            cout<<ans<<endl;
        }
        return 0;
}
