int dp[2][800001];
int n,k;
int a[201],b[201];
int solve(int idx,int diff){
    if(diff>400000||diff<-400000)return 0;
    if(idx==n){
        if(abs(diff)<=k)return 1;
        return 0;
    }
    int &ret = dp[idx&1][diff+400000];
    if(ret!=-1)
        return ret;
    ret = 0;
    ret += solve(idx+1,diff);
    ret += solve(idx+1,diff+a[idx]);
    ret%=MOD;
    ret += solve(idx+1,diff-b[idx]);
    ret%=MOD;
    ret += solve(idx+1,diff+a[idx]-b[idx]);
    ret %=MOD;
    return ret;
}
 
int main()
{
    scanf("%d",&n);
    fore(i,n)scanf("%d",&a[i]);
    fore(i,n)scanf("%d",&b[i]);
    scanf("%d",&k);
    for(int i=n;i>=0;i--){
        memset(dp[i&1],-1,sizeof dp[i&1]);
        for(int j=-400000;j<=400000;j++){
            solve(i,j);
        }
    }
    printf("%d",solve(0,0));
 	return 0;
}
