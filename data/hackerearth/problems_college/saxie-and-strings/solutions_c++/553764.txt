#include <iostream>
#include<string.h>
using namespace std;
int fun(){
	char s[105];
	cin>>s;
	int l=strlen(s);
	int a=0,b=l-1;
	while(a<b)
		if(s[a++]!=s[b--])
			return 1;
	return 0;	
}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	if(!fun())
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    return 0;
}
