
#include<iostream>
#include<algorithm>
using namespace std;


int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		long long int arr[3] , count = 0 ;
		cin >> arr[0] >> arr[1] >> arr[2];
		sort(arr, arr + 3);
		long long int q = arr[1] - arr[0] - 1 ;
		long long int w = arr[2] - arr[1] - 1 ;
		if(q >= w)
		count = q ;
		else
		count = w ;
		cout << count << endl;
	}
    return 0;
}
