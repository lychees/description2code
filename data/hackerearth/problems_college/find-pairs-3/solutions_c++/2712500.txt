#include<bits/stdc++.h>
using namespace std;
int main(){
	int t,n,arr[1000000];
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		for(int i=0;i<n;i++){
			scanf("%d",&arr[i]);
		}
		long long count=0;
		for(int i=0;i<n;i++){
			for(int j=i;j<n;j++){
				if(arr[j]==arr[i]){
					count++;
				}
			}
		}
		printf("%lld\n",count);
	}
	return 0;
}
