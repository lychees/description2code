#include <iostream>
using namespace std;

int main()
{
    int T;
    cin >> T;
    for(int i=0;i<T;i++){
    	int N,done=0;
    	int64_t even;
    	cin >> N;
    	for(int j=0;j<N;j++){
    		int64_t r,c,x,k;
    		cin >> r >> c >> x;
    		if(done==1)continue;
    		k = c-r;
    		if(j==0){
    			if(x%2==0){
    				even = k;
	    		}else{
	    			even = k+1;
	    		}
    		}else{
    			if(x%2==0){
    				if((k-even)%2==0){//even gap between same parities
    					
    				}else{
    					done=1;
    				}
	    		}else{
	    			k++;
	    			if((k-even)%2==0){//even gap between same parities
    					
    				}else{
    					done=1;
    				}
	    		}
    		}
    	}
    	if(done==0)cout << "Yes" << endl;
    	else cout << "No" << endl;
    }
    return 0;
}
