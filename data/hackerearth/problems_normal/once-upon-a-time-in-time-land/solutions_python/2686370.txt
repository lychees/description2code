test_cases = input()
for i in range(test_cases):
	n,k = map(int,raw_input().split())
	input1 = map(int,raw_input().split())
	a = [0 for i in range(n)]
	for i in range(n-1,-1,-1):
		try:
			g = a[i+1]
		except:
			g = 0
		try:
			m = a[i+k+1]
		except:
			m = 0
		a[i] = max(g,m+input1[i])
		
	print a[0]
