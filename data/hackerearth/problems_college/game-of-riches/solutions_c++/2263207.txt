#include <iostream>
#include<cmath>
using namespace std;
void arrang(int c[],int a)
{
	int i,j;
	for(i=0;i<a-1;i++)
	for(j=i+1;j<a;j++)
	{
		if(c[j]>c[i])
		{
			int d = c[j];
			c[j]= c[i];
			c[i] = d;
		}
	}
	return;
}
int main()
{
   int i,j,k,t,a,b,l,n;
   cin>>t;
   for(i=0;i<t;i++)
   {
   	 cin>>a>>b;
   	 int c[a];
   	 for(j=0;j<a;j++)
   	 cin>>c[j];
   	 int p = c[0];
   	 arrang(c,a);
   	 for(k=0;k<a;k++)
   	 {
   	 	//cout<<c[k]<<endl;
   	 	if(p==c[k])
   	 
   	 	break;
   	 }
   	 k++;
   	 //cout<<k<<endl;
   	 int x = k%b;
   	 int y=0;
   	 for(n=1;n<=a;n++)
   	 {
   	 	if(n%b==x && c[n-1]>p)
   	 	y++;
   	 }
   	 cout<<y<<endl;
   }
    return 0;
}
