#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long int t,n,A[100000],i;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>A[i];
    	}
    	sort(A,A+n);
    	for(i=0;i<n;i++)
    	{
    		cout<<A[i]<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
