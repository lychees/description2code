#include <bits/stdc++.h>

using namespace std;

int a[1000001];

int main()
{
    int t;
    cin >> t;

    while(t--) {
        int n;
        cin >> n;
        int cnt =0 ;
        for(int i = 0 ;i < n; i++) {
            cin >> a[i];
            if(a[i] == 1) cnt++;
        }
        if(cnt == n) cout << "0/1" << endl;
        else {
            int ans =0,last = 0;
            int cnt =0 ;
            for(int i =0 ; i <= n ;i++) {
                if(i == n || a[i] == -1) {
                    ans = max(ans,cnt+last);
                    last = cnt;
                    //cout << last << " " << cnt << endl;
                    cnt = 0;
                }
                else {
                    cnt += a[i];
                }
            }
            cout << ans << "/1" << endl;
        }
    }
    return 0;
}
