#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t, count;
    long long n;
    scanf("%d", &t);
    while (t--) {
        scanf("%lld", &n);
        long long x = 1;
        count = 0;
        while (n) {
                x = 1;
                while (x <= n)
                    x *= 2;
                x /= 2;
                n -= x;
                count++;
        }
        printf("%d\n", count);
    }
    return 0;
}
