#include <iostream>
#include <bits/stdc++.h>
#define MAX 10001
using namespace std;

typedef struct Power{
	long long int G, P;
}Pwr;

bool acompare(Pwr lhs, Pwr rhs){return lhs.P < rhs.P;}

int main()
{
    int T,N,flag,i,cnt;
    long long int M;
    Pwr power[MAX];
    cin>>T;
    while(T--){
    	cin>>N>>M;
    	for(i=0;i<N;i++){
    		cin>>power[i].G>>power[i].P;
    	}
    	sort(power,power+N, acompare);
    	flag = 0;
    	cnt = 0;
    	for(i=0;i<N;i++){
    		if(power[i].P < M){
    			M = M + power[i].G;
    		}
    		else{
    			flag = 1;
    			break;
    		}
    	}
    	cout<<((!flag)?"YES":"NO")<<endl;
    	
    }
    return 0;
}
