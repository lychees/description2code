#include <bits/stdc++.h>

using namespace std;

int main()
{
    int tc;
    cin >> tc;
    char a[10000];
    while(tc--)
    {
        cin >> a;
        char ans[strlen(a)];

        for (int i = 0;i < strlen(a);i++)
            ans[i] = 'a';

        for (int i = 0;i < strlen(a);i++)
        {
            if (a[i] == 'B')
            {
                ans[i-1] = 'B';
                ans[i-2] = 'B';
                ans[i] = 'B';
            } else if (a[i] == 'D')
            {
                ans[i] = 'D';
            } else if (a[i] == 'S')
            {
                ans[i-1] = 'S';
                ans[i+1] = 'S';
                ans[i] = 'S';
            }
        }
        int count = 0;
        for (int i = 0;i < strlen(a);i++)
        {
            //cout << ans[i];
            if (ans[i] == 'a')
                count++;
        }

        //cout << endl;
        cout << count << endl;
    }

    return 0;
}
