#include <iostream>
using namespace std;
int rev(int a)
{
	int temp=0;
	while(a>0)
	{
		int aq=a%10;
		temp=temp*10+aq;
		a/=10;
	}
	return temp;
	}

int main()
{
    int t,x;
    cin>>t;
    while(t--)
    {
    	cin>>x;
    	int sq=x*x;
    	int z=rev(x);
    	int sqz=z*z;
    	int revz=rev(sqz);
    	if(z!=x && sq==revz)
    	cout<<z<<endl;
    	else 
    	cout<<"No"<<endl;
    	
    }
}
