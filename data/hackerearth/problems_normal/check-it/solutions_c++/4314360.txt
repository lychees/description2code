#include<bits/stdc++.h>
#define ll long long
#define lg long
#define dd double
#define maxx 1000000007
#define scf(a) scanf("%lld",&a)
#define scdf(a) scanf("%d",&a)
#define prt(a) printf("%lld",a)
#define prdt(a) printf("%d",a);
#define get(a) cin>>a
#define put(a) cout<<a
#define loop(i,a,b) for(i=a;i<=b;i++)
#define nxt printf("\n")
#define space printf(" ")
ll find_root(long a[],lg n)
{
    while(1)
    {
      if(a[n]==n)
            return n;
      else
        n=a[n];
    }
}
using namespace std;
int main()
{
    ll i,k,j,l,m,n,t;
    get(t);
    while(t--)
    {
        l=0;
        get(n);
        get(m);
        vector<pair<ll,ll>>v;
        vector<ll>s[n+1];
        lg a[n+1],b[n+1],c[n+1];
        loop(i,1,n)
        {
            b[i]=i;
            c[i]=1;
        }
        memset(a,0,sizeof(a));
        v.push_back(make_pair(0,0));
        loop(i,1,m)
        {
            get(j);
            get(k);
            v.push_back(make_pair(j,k));
            s[j].push_back(k);
            s[k].push_back(j);
        }
        loop(i,1,n-1)
        {
            get(m);
            if(l==0)
            {
                a[v[m].first]=1;
                a[v[m].second]=1;
                j=v[m].first;
                k=v[m].second;
                j=find_root(b,j);
                k=find_root(b,k);
                if(j==k)
                {
                    l=1;
                }
                else
                {
                    if(c[j]>c[k])
                    {
                        c[j]+=c[k];
                        b[k]=j;
                    }
                    else
                    {
                        c[k]+=c[j];
                        b[j]=k;
                    }
                }
            }
        }
        if(l==1)
            cout<<"NO\n";
        else
        {
            loop(i,1,n)
            {
                if(a[i]==0)
                    break;
            }
            if(i>n)
                cout<<"YES\n";
            else
                cout<<"NO\n";
        }
    }
    return 0;
}
