#include <bits/stdc++.h>
#define c 1000000007
#define ll long long

using namespace std;

int main () {
    ll t, p, e;
    cin >> t;
    while(t--) {
        cin >> p >> e;
        double np = 0, ep = p + (e * p / 100.00);
        ll n = ep;
        while(n != 0) {
            ll r = n % 10;
            np = np * 10 + r;
            n /= 10;
        }
        double pr = (1.0*p - 1.0*np);
        double pc = double(pr) / double(p);
        double pl = (pc * ep);
        cout << (ll)(pl + ep) << endl;
    }
    return 0;
}
