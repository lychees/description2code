#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   for(int uu=1;uu<=t;uu++)
   {
   	  int n,p,u,s,r,i,j,k=0;
   	  cin>>n>>p>>u>>r>>s;
   	  for(i=0;i<n;i++)
   	  {
   	      if(p+r<u)	
   	      {
   	      	p+=r;
   	      }
   	      else
   	      {
   	      	if(p-s<0)
   	      	break;
   	      	else
   	      	p-=s;
   	      }
   	      
   	  }
   	  	cout<<"Case #"<<uu<<": "<<p<<endl;
   }
    return 0;
}
