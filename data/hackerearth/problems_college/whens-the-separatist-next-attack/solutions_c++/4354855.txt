#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T;
    scanf("%d", &T);
    
    while(T--)
    {
    	string S;
    	cin >> S;
    	
    	map<char, int> X;
    	
    	for(int i=0; i<S.size(); i++)
    		X[S[i]] = -1;
    	
    	long long basex = X.size();
    	long long ans = 0, ptr = -1;
    	
    	if(basex==1)
    		basex = 2;
    	
    	for(int i=0; i<S.size(); i++)
    	{
    		if(X[S[i]]==-1)
    		{
    			if(ptr==-1)
    			{
    				X[S[i]] = 1;
    				ptr = 0;
    			}
    			else if(ptr==0)
    			{
    				X[S[i]] = 0;
    				ptr = 2;
    			}
    			else
    			{
    				X[S[i]] = ptr;
    				ptr++;
    			}
    		}
    		
    		ans = ans*basex + X[S[i]];
    	}
    	
    	cout << ans << endl;
    }
    
    return 0;
}
