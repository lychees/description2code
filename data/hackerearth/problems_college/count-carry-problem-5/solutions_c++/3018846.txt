#include <iostream>
#include <stdio.h>
#include <string>
using namespace std;

int main()
{
    int n;
    unsigned long n1, n2;
    int carry = 0;
    int count = 0;
    scanf("%d\n",&n);
    for(int i = 0; i < n; i++){
    	scanf("%lu %lu\n",&n1,&n2);
    	carry = 0; count = 0;
    	while(n1 > 0 && n2 > 0){
    		if(n1%10 + n2%10 + carry >= 10){
    			carry = 1;
    			count++;
    		}
    		n1 = n1/10;
    		n2 = n2/10;
    	}
    	if(count == 0) printf("No carry operation\n");
    	else if(count == 1) printf("1 carry operation\n");
    	else printf("%d carry operations\n",count);
    }
    return 0;
}
