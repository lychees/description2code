#include <iostream>
#define M 1000000007
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long l,b,d,ans;
    	cin>>l>>b>>d;
    	if(d>l||d>b)
    		cout<<"0"<<endl;
    	else
    	{
    		ans=((l/d)%M)*((b/d)%M);
    		//cout<<(l%d)%M<<" "<<(b%d)%;
    		cout<<ans%M<<endl;
    	}
    }
    return 0;
}
