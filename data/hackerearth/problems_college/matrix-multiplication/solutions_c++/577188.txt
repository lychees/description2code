#include <iostream>
using namespace std;

int main()
{
        int test_cases;
        cin>>test_cases;
        while(test_cases )
        {
                int X1,X2,Y1,Y2;
                cin>>X1>>Y1>>X2>>Y2;
                if(Y1==X2 && Y2==X1)
                {
                	cout<<"2\n";
                }
                else if(Y1==X2|| Y2==X1)
                {
                	cout<<"1\n";
                }
                else
                cout<<"0\n";
                test_cases --;
        }
        return 0;
}
