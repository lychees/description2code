#include<stdio.h>
#include<algorithm>
#include<vector>
using namespace std;
struct node
{
	long long int sum;
	int i;
};
bool comp(const node &a, const node &b)
{
	if(a.sum==b.sum)
		return a.i>b.i;
	return a.sum>b.sum;
}
int main()
{	
	int t;
	scanf("%d", &t);
	vector<node> arr;
	for(int i=0;i<t;i++)
	{
		long long int t,d;
		scanf("%lld %lld", &t, &d);
		node n;
		n.sum=t+d;
		n.i=i;
		arr.push_back(n);
	}
	sort(arr.begin(), arr.end(), comp);
	for(int i=arr.size()-1;i>=0;i--)
		printf("%d ",arr[i].i+1);
	return 0;
}
