#include <iostream>
#include<string.h>
using namespace std;

int main()
{
 	char a[103];
 	char freq[300];
 	int i,j,T;
 	int total_odd;
 	cin>>T;
 	while(T--){
 		cin>>a;
 		total_odd=0;
 		for(i=0;i<300;i++)
 			freq[i]=0;
 		int len=strlen(a);
 		for(i=0;i<len;i++){
 			freq[a[i]]++;
 		}
 		for(i=0;i<300;i++){
 			if(freq[i]%2!=0)
 				total_odd++;
 		}
 		if(total_odd<1||(total_odd==1&&(len%2!=0)))
 			cout<<"Yes"<<endl;
 		else
 			cout<<"No"<<endl;
 	}
}
