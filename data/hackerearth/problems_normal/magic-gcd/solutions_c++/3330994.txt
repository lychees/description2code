#include <iostream>
using namespace std;
int gcd(int j,int b)
{
 if(b == 0) {
            return j;
    }
    else {
        return gcd(b, j % b);
    }
}
int main()
{int max=0;
    int n,p,q,k;
    cin>>n;
    for(int i=n;i>=1;i--)
    {cin>>p;
    cin>>q;
    cin>>k;
    max=0;
    	for(int j=p-k;j<=p+k;j++)
    	{
    		for(int b=q-k;b<=q+k;b++)
    		{int num1,num2;
    				if(j<p)
    				{num1=p-j;
    					
    				}
    				else{
    					num1=j-p;
    				}
    				if(b<q)
    				{num2=q-b;
    					
    				}
    				else{
    					num2=b-q;
    				}
    				
    				if( (num1+num2)<=k)
    				{
    				int gcd1=gcd(j,b);
    					if(gcd1>max)
    						{
    							max=gcd1;
    							}
    				}
    			
    			
    		}
    	}
    	
    	cout<<max<<endl;
    	
    	
    	
    }
    
    return 0;
}
