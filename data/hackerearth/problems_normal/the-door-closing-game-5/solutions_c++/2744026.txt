#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
   int T;
   cin>>T;
   while(T--){
   	string s;
   	cin>>s;
   	vector<int> arr;
   	int cnt=0;
   	for(int i=0;i<s.length();i++){
   		if(s[i]=='|'){
   			if(cnt){
   				arr.push_back(cnt);
   				cnt=0;
   			}
   		}
   		else{
   			cnt++;
   		}
   	}
   	if(cnt)
   	   arr.push_back(cnt);
   	int n=arr.size();

   	int ans=n>0?arr[0]:0;
   	for(int i=1;i<n;i++){
   		ans^=arr[i];
   	}
    if(n==1&&arr[0]==s.length()){
    	cout<<"Bob"<<endl;
    }
   	else if(!ans){
   		cout<<"Bob\n";
   	}
   	else
   	   cout<<"Alice\n";
   }
}
