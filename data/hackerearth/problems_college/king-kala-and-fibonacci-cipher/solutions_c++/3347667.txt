#include <iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;

pair<int,int> p[26];
int f[100005][30];
char s[105];

int main()
{
	int t,l,r,i,j;
	int f1,f2,f3;
	f1=0;
	f2=1;
	
	for(i=0;i<26;i++)
	{
		f[0][i]=0;
		f[1][i]=0;
		f[2][i]=0;
	}

	f[1][0]=1;
	f[2][0]=1;
	f[2][1]=1;

	for(i=3;i<=100001;i++)
	{
		for(j=0;j<26;j++)
			f[i][j]=f[i-1][j];
		
		f3=(f1+f2)%26;
		f[i][f3]++;
		
		f1=f2;
		f2=f3;
	}
	
	// for(i=0;i<10;i++)
	// {
	// 	for(j=0;j<26;j++)
	// 		printf("%d ",f[i][j]);
	// 	printf("\n");
	// }
	
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d\n",&l,&r);
		l++;
		r++;
		for(i=0;i<26;i++)
			p[i]=make_pair(-f[r][i]+f[l-1][i],i);
		
		// for(i=0;i<26;i++)
		// 	printf("%d ",p[i].first);
		// printf("\n");
		
		sort(p,p+26);
		
		gets(s);
		for(i=0;s[i]!='\0';i++)
		{
			s[i]=(p[s[i]-'a'].second)+'a';
		}
		printf("%s\n",s);
	}
    return 0;
}
