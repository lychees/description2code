#include<bits/stdc++.h>
using namespace std;
int main()
{
    int i,j,n0,n1,nq,res,Nq;
    string a,b;
    int t;
    cin>>t;
    for(j=1;j<=t;j++)
    {
        n0=n1=nq=Nq=0;
        cin>>a>>b;
         cout<<"Case "<<j<<": ";
        for(i=0;a[i]!='\0';i++)
        {
            if(a[i]=='1'&&b[i]=='0')
            n1++;
            else if(a[i]=='0'&&b[i]=='1')
                n0++;
            else if(a[i]=='?'&&b[i]=='1')
                nq++;
                else if(a[i]=='?')
                Nq++;
        }
       // cout<<endl<<n0<<" "<<n1<<" "<<nq<<endl;
            if(n1>n0)
            {
            	res=n0;
                n1=n1-n0;
                n0=0;
            }
            else
            {
            	res=n1;
                n0=n0-n1;
                n1=0;
            }
            //cout<<" after first pass "<<res<<endl;
            if(n1>0)
            {
                if(nq>=n1)
                {
                	res=res+n1;
                    n1=0;
                }
            }
         //   cout<<" final p0 p1 and p? "<<n0<<" "<<n1<<" "<<nq<<endl;
            if(n1>0)
                cout<<"-1\n";
            else
                cout<<n0+nq+res+Nq<<endl;
    }
   return 0;
}
