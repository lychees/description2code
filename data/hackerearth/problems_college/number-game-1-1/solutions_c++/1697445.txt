#include <iostream>
using namespace std;
int calculate(int a){
		int res = 0;
		while(a>1){
	  if(a%3==0){
	  	a = a/3;
	  	res++;
	  }	else
	   if(a%2==0){
	  	a = a/2;
	  	res++;
	  }	
	  else {
	  	a=a-1;
	  	res++;
	  }
		}
	  return res;
	  
	}
int main()

{
	
   int n,a;
    cin>>n;
    for(int i=0;i<n;i++){
    	cin>>a;
    int result =	calculate(a);
    	cout<<result<<"\n";
    }
    return 0;
}
