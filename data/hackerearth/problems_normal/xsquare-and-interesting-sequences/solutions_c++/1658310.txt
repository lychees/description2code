#include <bits/stdc++.h>
using namespace std ;
#define LL long long 
LL M = 1000000009LL ;
LL power(LL a,LL b){
	LL resp = 1 ;
	while(b){
		if(b%2){
			resp = resp*a ;
			resp %= M ;
		}
		
		
		b /= 2 ;
		a = a*a ;
		a %= M ;
	}
	
	return resp ;
}
int main(){
	int T ;
	LL N ;
	cin >> T ;
	while(T--){
		cin >> N ;
		cout << power(26,(N+1)/2) << endl ;
	}
	return 0;
}
