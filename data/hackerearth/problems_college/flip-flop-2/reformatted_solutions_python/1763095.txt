import sys
ip = []
for line in sys.stdin:
    ip.append(line)
for i in range(0, len(ip)):
    ip[i] = ip[i].rstrip('\n')
for i in ip[1:len(ip)]:
    deletion = 0
    if i[0] == 'X':
        word = 'Y'
    else:
        word = 'X'
    for j in range(1, len(i)):
        if i[j] != word:
            deletion = deletion+1
        elif word == 'X':
            word = 'Y'
        elif word == 'Y':
            word = 'X'
    print deletion
