#include<iostream>
#include<set>
#include<vector>
#include<algorithm>
#define ll long long
using namespace std;
ll h[55],w[55];
bool vis[55];
void doit()
{
	int n;
	cin>>n;
	for(int i=0;i<n;++i)cin>>w[i];
	for(int i=0;i<n;++i)cin>>h[i];
	for(int i=0;i<n;++i)vis[i]=0;
	sort(h,h+n);
	ll ans=h[n-1]*w[0];
	vis[n-1]=1;
	bool poss=0;
	for(int i=1;i<n;++i)
	{
		int last=-1;
		for(int j=0;j<n;++j)
			if(!vis[j])
			{
				ll now=w[i]*h[j];
				if(now<ans)last=j;
			}
		if(last==-1){cout<<"NO\n";return;}
		vis[last]=1;
	}
	cout<<"YES\n";
	return;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		doit();
	}
	return 0;
}
