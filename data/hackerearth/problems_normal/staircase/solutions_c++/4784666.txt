#include <iostream>
using namespace std;

int main()
{
    int n, i;
    cin>>n;
    int a[n], b[n];
    if(n<0)
    cout<<"ERROR"<<endl;
    else
    {
    	a[0]=b[0]=a[1]=b[1]=1;
	    a[2]=b[2]=2;
	    a[3]=4, b[3]=3;

	    for(i=4;i<=n;i++)
	    {
	    	a[i]=a[i-3]+a[i-2]+a[i-1];
	    	b[i]=b[i-2]+b[i-1];
		}
		cout<<"Jack-"<<a[n]<<endl;
		cout<<"Jill-"<<b[n]<<endl;
    }
    return 0;
}
