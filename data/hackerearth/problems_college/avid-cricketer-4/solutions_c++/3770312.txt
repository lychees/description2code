#include <iostream>
#include<bits/stdc++.h>
using namespace std;

double dp[1002][1002];


double findprob(int balls,int target) 
{
	
	if(target <= 0) 
		return 1;
	if(balls == 0) 
		return 0;

	if(dp[balls][target]!=-1) 
		return dp[balls][target];

	dp[balls][target] = (findprob(balls-1,target) + findprob(balls-1,target)  + findprob(balls-1,target-1) + findprob(balls-1,target-2) + findprob(balls-1,target-3) + findprob(balls-1,target-4) + findprob(balls-1,target-6));
	dp[balls][target]/=7;
	return (dp[balls][target]);
}

int main() {
	int t;
	scanf("%d",&t);
	
	while(t--) {
		for(int i=1;i<1002;i++)
		for(int j=1;j<1002;j++)
		dp[i][j]=-1;

		int r,b;
		scanf("%d%d",&r,&b);
		double res = 0.0;
		res = findprob(b,r);

		printf("%.6lf\n",res);

	
	}
	return 0;
}
