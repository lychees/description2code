#include <iostream>
#include <math.h>
using namespace std;

bool isPrime(long n)
{
	long i;
	
	for(i=2; i<=sqrt(n); i++)
	{
		if(n%i==0)
			return false;
	}
	return true;
}

int main()
{
    int t;
    long int n, i, j;
    bool flag;
    
    cin>>t;
    
    while(t--)
    {
    	flag=1;
    	cin>>n;
    	
    	for(i=2; i<=n/2; i++)
    	{
    		if(isPrime(i))
    		{
    			j=n-i;
    			if(isPrime(j))
    			{
    				cout<<"Deepa\n";
    				flag=0;
    				break;
    			}
    		}
    	}
    	
    	if(flag)
    	{
    		cout<<"Arjit\n";
    	}
    }
    return 0;
}
