#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t=0;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int flag=0;
    	int l=s.size();
    	for(int i=0;i<l/2;i++)
    	{
    		if(!(s[i] == s[l-1-i]))
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag ==0)
    	{
    		cout<<"Palindrome"<<endl;
    	}
    	else
    	{
    		long long int res=1;
    		for(int j=0;j<l;j++)
    		{
    			res=res*((int)s[j]-96);
    		}
    		cout<<res<<endl;
    	}
    }
    return 0;
}
