#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
bool comp(const pair<long long,long long> &a,const pair<long long,long long> &b)
{
	if(a.first==b.first)
	return a.second<b.second;
	else
	return a.first>b.first;

}
int main()
{
	long int n,k,i;
	cin>>n>>k;
	long long int arr[n+1];
	for(i=1;i<=n;i++)
	cin>>arr[i];
	vector<pair<long long,long long> > v;
	for(i=0;i<n;i++)
	v.push_back(make_pair(arr[i+1]%k,arr[i+1]));
	sort(v.begin(),v.end(),comp);
	for(i=0;i<v.size();i++)
	cout<<v[i].second<<" ";
}
