#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long n=0, k=0, m=0;
    	cin>>n>>k>>m;
    	long long sum = (k*n*n*(n+1)*(n+1)/4);
    	//cout<<sum;
    	if(sum>m)
    		cout<<sum-m;
        else
        	cout<<"0";
    	cout<<endl;
    }
    return 0;
}
