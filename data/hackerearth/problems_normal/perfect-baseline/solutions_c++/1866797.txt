#include <iostream>
#include <stdio.h>
#include <vector>
#include <string>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <queue>
#include <deque>
#include <string.h>
#include <iomanip>
#include <stack>

using namespace std;

#define mx 10000

string S;
int vis[200][200];
char ans [200];

int main() {
    cin.sync_with_stdio(false);
    
   
    
    int t;
    cin >> t;
    
    int n, k;
    int i, j, i1;
    while (t--) {
        cin >> n >> k;
        
        memset(vis, 0, sizeof(vis));
        
        for (i = 0; i < n; i++) {
            cin >> S;
            for (j = 0; j < k; j++) {
                for (i1 = 'a'; i1 <= 'z'; i1++)
                    vis[j][i1] += abs(S[j]-i1);
            }
        }
        
        for (i = 0; i < k; i++) {
            int mini = 1000000;
            for (j = 'a'; j <= 'z'; j++) {
                if (vis[i][j] < mini) {
                    mini = vis[i][j];
                    ans[i] = j;
                }
            }
        }
        
        for (i = 0; i < k; i++)
            cout << ans[i];
        cout << endl;
    }
    
    return 0;
}
