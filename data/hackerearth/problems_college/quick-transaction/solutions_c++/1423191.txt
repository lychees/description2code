#include <iostream>
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<math.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	 int arr[9]={1,2,5,10,20,50,100,500,1000};
	 int x,sum=0;
	 cin>>x;
	 for(int i=8;i>=0&&x!=0;i--)
	 {
	 	sum+=x/arr[i];
	 //	cout<<x/arr[i]<<" ";
	 	x%=arr[i];
	 }
	 cout<<sum<<endl;
	}
}
