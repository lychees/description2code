#include<bits/stdc++.h>
using namespace std;


double dp[55][2501];
double solve(int m,int n,int sum)
{
    double p = 1.0/double(m) ;
    if(m*n<sum)
        return 0;
    else
    {

        for(int i=1;i<=m;i++)
        {


                 dp[1][i]=p;
        }
        for(int i=m+1;i<=sum;i++)
            dp[1][i]=0;

        for(int i=2;i<=n;i++)
        {
            for(int j=0;j<=sum;j++)
            {
                dp[i][j]=0;
                for(int k=1;k<=m;k++)
                {
                    if(j-k>=0)
                        dp[i][j]=dp[i][j]+(p*dp[i-1][j-k]);
                }

            }

        }
    }

    return dp[n][sum];

}

int main()
{
   int t,m,n,k;
    cin>>t;
    while(t--)
    {
        int cnt=0;
        cin>>n>>m>>k;
        double x = solve(m,n,k);
        if(x==0)
        {
           printf("%.3f %d\n",x,cnt);
            continue;
        }

while(x<1.0)
{
    x*=10;
    cnt++;

}
printf("%.3f %d\n",x,cnt);


    }

}
