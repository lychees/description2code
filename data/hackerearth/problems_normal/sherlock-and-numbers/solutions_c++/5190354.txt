
#include <iostream>
#include <string>
#include <string.h>

#include <algorithm>

using namespace std;

#define MAX 100020

int arr[MAX];
bool res[MAX];



int main(){
	std::ios_base::sync_with_stdio(false);
	//freopen("input.txt", "rt", stdin);
	int T;
	cin >> T;
	for (int t = 0; t < T; ++t){
		long long n, p, k;
		cin >> n >> k >> p;

		for (int i = 1; i <= k; ++i){
			cin >> arr[i];
		}

		sort(arr, arr + k + 1);
		bool kt = false;
		//1 - > arr[k-1] remove k number

		if (p >= arr[k]){
			cout << p + k -1 << endl;
			continue;
		}
		//p<arr[k-1]
		else{
			//arrr[i] - i - 1  1->arr[i]
				for (int i = 1; i <= k; ++i){
					if (arr[i] - i >= p){
						cout << p + i -1 << endl;
						kt = true;
						break;
					}
				}
		}
		if (!kt)
		cout << "-1" << endl;
	}
	return 0;
}
