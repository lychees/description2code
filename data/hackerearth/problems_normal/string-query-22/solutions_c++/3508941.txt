#include <iostream>
#define ll long long
#define maxx 400002
using namespace std;
ll bit[30][400008],remov[400005]={0};
void upd(ll ch,ll pos,ll val)
{
    for(ll i=pos;i<maxx;i+=(i&-i))
     bit[ch][i]+=val;
}
ll qry(ll ch,ll pos)
{
    ll sum=0;
    for(ll i=pos;i>=1;i-=(i&-i))
    sum+=bit[ch][i];
    return sum;
}
ll bin(ll ch,ll rank)
{
    ll mid,l=1,h=maxx,res;
    while(l<=h){
        mid=(l+h)/2;
        if(qry(ch,mid)>=rank){
            res=mid;
            h=mid-1;
        }
        else
         l=mid+1;
    }
    upd(ch,res,-1);
    return res;
}
int main()
{
    string s,s1;
    cin>>s;
    ll m,temp,count,len=s.length();char A;
    for(ll i=0;i<len;i++)
    {
        upd(s[i]-'a',i+1,1);
    }
    cin>>m;
    while(m--){
        cin>>temp>>A;

        //cout<< remov[pos-1]<<endl;

    }

    cout<<s[i];
    }

}
