#include <bits/stdc++.h>
using namespace std;
int main()
{
   int n;
   	scanf("%d",&n);
	assert(n >= 1 && n <= 1000);
	int A[n],v[1100];
	memset(v,0,sizeof(v));
	
	for(int i=0;i<n;i++){
	scanf("%d",&A[i]);
	assert(A[i] >= 1 && A[i] <= 1000);
	v[A[i]]++;
	}	
	
	for(int i=1;i<=1000;i++)
	v[i]+=v[i-1];
	
	for(int i=0;i<n;i++)
	printf("%d ",(v[1000]-v[A[i]])+1);
	return 0;
}
