#define ll long long
#define MOD 1000000007
using namespace std;
ll g(ll a,ll b){
if(b%a==0)return a;
return g(b%a,a);}

ll pow(ll x,ll n,ll M){


    if(n==0)
        return 1;
    else if(n%2 == 0)        // n is even
        return pow((x*x)%M,n/2,M);
    else                             // n is odd
        return (x*pow((x*x)%M,(n-1)/2,M))%M;

}
