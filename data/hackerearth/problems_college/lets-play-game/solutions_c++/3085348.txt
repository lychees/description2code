#include<iostream>
#include<math.h>
using namespace std;
int main()
{
	int T;
	long int N,K,i,s;
	long int *A,*B,*profit;
	cin>>T;
	while(T>0 && T<11)
	{
		cin>>N;
		cin>>K;
		if(N>0 && N<=pow(10,5) && K>0 && K<=pow(10,9))
		{
			A=new long int[N];
			B=new long int[N];
			profit=new long int[N];
			for(i=0;i<N;i++)
				profit[i]={0};
			for(i=0;i<N;i++)
				cin>>A[i];
			for(i=0;i<N;i++)
				cin>>B[i];
			for(i=0;i<N;i++)
			{
				long int time=0;
				long int count=0;
				s=A[i];
				while(time<=K)
				{
					time=s+time;
					count++;
				}
				count--;
				profit[i]=count*B[i];
			}
			long int max=profit[0];	
			for(i=1;i<N;i++)
			{
				if(max<profit[i])
					max=profit[i];
			}
			cout<<max<<endl;
		}
		T--;
	}
	return 0;
}
