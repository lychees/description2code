#include <iostream>
#include <cstdio>
#include <cstdlib>

#include <algorithm>
using namespace std;


int main()
{
    long t,i,j;
    cin >> t;
    while(t--)
    {
    	long n;
    	cin>> n;
    	long a[n];
    	for(long i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	sort(a,a+n);
        scanf("%ld%ld",&i,&j);
        cout<<a[j-1]+a[n-i]<<endl;
    	
    }
    return 0;
}
