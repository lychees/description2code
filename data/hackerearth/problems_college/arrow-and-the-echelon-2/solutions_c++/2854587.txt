#include <iostream>
using namespace std;

int main()
{
                int t;
                long int x,i,j;
                long int x5,x3;
                bool f;
                cin>>t;
                while(t--)
                {
                	cin>>x;
                	x5=x;
                	x3=0;
                	
                	f=0;
                	while(x3<=x && f==0 && x5>=0)
                	{
                		if(x3%5==0 && x5%3==0)
                		 f=1;
                		else
                		{
                			x5-=5;
                			x3+=5;
                		}
                	}
                	if(f==1)
                	{
                		for(i=1;i<=x5;i++)
                		 cout<<5;
                		for(i=1;i<=x3;i++)
                		 cout<<3;
                		cout<<endl; 
                	}
                	else
                	 cout<<-1<<endl;
                }
                
    return 0;
}
