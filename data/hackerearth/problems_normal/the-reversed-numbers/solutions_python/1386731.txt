def flip(n):
	n=n[::-1]
	k=1
	m=[]
	for i in range(0,len(n)):
		if n[i]=='0' and k==1:
			continue
		else:
			m.append(n[i])
			k=0
	#print int(''.join(m))
	return int(''.join(m))


t=input()
for i in range(0,t):
	p=raw_input()
	a,b=[j for j in p.split()]
	a=flip(a)
	b=flip(b)
	print flip(str(a+b))
