#include <iostream>
using namespace std;
 unsigned long long gcd ( unsigned long long a, unsigned long long b ) { unsigned long long c; while ( a != 0 ) { c = a; a = b%a; b = c; } return b; } 
int main()
{
    unsigned long long  temp,t,p,l,i,ans,pr;
    int a[100005];
    char str[100005];
    scanf("%llu",&t);
    while(t--)
    {
    	a[0]=0;
    	scanf("%llu%llu",&l,&p);
    	scanf("%s",str);
    	
    	for(i=0;str[i]!='\0';i++)
    	{
    		if(str[i]=='1')
    		a[i+1]=a[i]+1;
    		else
    		a[i+1]=a[i];
    	}
    	ans=0;
    	a[0]=0;
    	if(a[l]==0)
    	{
    		printf("0/1\n");
    		continue;
    	}
    	for(i=1;i<=l;i++)
    	{
    		if(str[i-1]=='0')
    		continue;
    	
    		if(i+p<=l)
    		ans+=a[i+p]-a[i-1];
    		else
    		ans+=a[l]-a[i-1];
    		//cout<<ans<<" "<<i<<" "<<p<<endl;
    		temp=0;
    		if(i<p)
    		{
    		//	cout<<"i less p ";
    			temp=a[i]-a[0];
    		}
    		else
    		{
    		//	cout<<"i great p ";
    			temp=a[i]-a[i-p-1];
    		}
    	
    		
    		ans+=temp-1;
    		//cout<<temp<<" "<<ans<<endl;
    	}
    	pr=l*l;
    	temp=gcd(pr,ans);
    	printf("%llu/%llu\n",ans/temp,pr/temp);
    	
    }
    
    return 0;
}
