#include<iostream>
#include<queue>
using namespace std;
int main()
{
	int n;
	cin>>n;
	priority_queue<long long> pq;
	long long x,a,b,c;
	for(int i=1;i<=n;i++)
	{
		cin>>x;
		pq.push(x);
		if(i<=2)
		cout<<"-1"<<endl;
		else if(i>2)
		{
			a=pq.top();
			pq.pop();
			b=pq.top();
			pq.pop();
			c=pq.top();
			pq.push(a);
			pq.push(b);
			cout<<a*b*c<<endl;
		}
	}
}
