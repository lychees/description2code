#include<bits/stdc++.h>
using namespace std;

int main()
{
	int a[1003],b[1003],c,k=0,i,j,m;
	for(i=2;i<10000;i++)
	{
		c=0;
		for(j=2;j<i;j++)
		{
			if(i%j==0)
			{
				c++;
				break;
			}
		}
		if(c==0)
		{
			a[k]=i;
		//	cout<<a[k]<<" ";
			k++;
		}
	}
	k=0;
	for(i=30;i<6005;i++)
	{
		c=0;
		m=0;
		for(j=0;j<1000;j++)
		{
			if(i%a[j]==0)
			{
				c++;
				if(c==3)
				{
					m=1;
					break;
				}
			}
		}
		if(m==1)
		{
			b[k]=i;
			k++;
			if(k==1001)
			break;
		}	
	}
	int n,a1;
	cin>>n;
	while(n--)
	{
		cin>>a1;
		cout<<b[a1-1]<<"\n";
	}
	return 0;
}
