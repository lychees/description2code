#include <bits/stdc++.h>
using namespace std;
int min(int x, int y){
	return (x<y)?x:y;
}
int min(int x, int y, int z){
	return (min(x,y)<z) ? min(x,y):z;
}
int main()
{
    int t,n;
    int arr[20][3];
    cin >> t;
    while( t-- ){
	cin >> n;
        for(int i=0; i<n; i++){
	  cin >> arr[i][0] >> arr[i][1] >> arr[i][2];
	}
        for(int i=1; i<n; i++){
	  arr[i][0]+= min(arr[i-1][1], arr[i-1][2]);
	  arr[i][1]+= min(arr[i-1][0], arr[i-1][2]);
	  arr[i][2]+= min(arr[i-1][0], arr[i-1][1]);
	}
	cout<<min(arr[n-1][0], arr[n-1][1], arr[n-1][2])<<endl;
    }
    return 0;
}
