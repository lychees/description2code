
MOD = 1000000007
trib = []
def calc_trib():
    trib.append(0)
    trib.append(1)
    trib.append(2)
    trib.append(4)
    for i in xrange(4,100010):
        temp = (trib[i-1] + trib[i-2] + trib[i-3])%MOD
        trib.append(temp)
def roy_and_mario():
    calc_trib() #we figured that it is nothing but a tribonacci series, so we precalculate
    for t in xrange(input()):
        n = input()
        print trib[n]
roy_and_mario()
