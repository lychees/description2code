#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll power(ll base,int exp,ll n){
	ll res=1;
	while(exp){
		if(exp&1){
			res=(res*base)%n;
		}
		base=(base*base)%n;
		exp>>=1;
	}
	return res;
}

int main(){
	int t;
	cin>>t;
	while(t--){
	    ll a,b;
		int n;
		cin>>a>>b>>n;
		a=a%n;
		b=b%n;
		if(a!=0&&b==0)
			cout<<"-1\n";
		else{
			ll res=power(b,n-2,n);
			res=(res*(n-a))%n;
			cout<<res<<endl;
		}
	}
}
