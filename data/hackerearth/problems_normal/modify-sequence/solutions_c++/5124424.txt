#include <iostream>
using namespace std;

int main()
{
	long size,i;
	cin >> size;
	long arr[size];
	for(i=0;i<size;i++)
		cin >> arr[i];
	for(i=0;i<size-1;i++)
	{
		if(arr[i] < 0)
		{
			cout << "NO";
			return 0;
		}
		arr[i+1] -= arr[i];
		arr[i] = 0;
	}
	if(arr[size-1] == 0)
		cout << "YES";
	else
		cout << "NO";
    return 0;
}
