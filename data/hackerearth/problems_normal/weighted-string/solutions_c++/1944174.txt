#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    int k;
    string s;
    while(t--)
    {
    	cin>>k;
    	cin>>s;
    	int len=s.length();
    	//v.resize(len);
    	int count=0;
    	int sum;
    	int j;
    	for(int i=0;i<len;i++)
    	{
    		sum=0;
    		j=0;
    		while(sum<k && (i+j)<len)
    		{
    			sum+=(int)(s[i+j]-96);
    			j++;
    		}
    		if(sum==k)
    		{
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
    
}
