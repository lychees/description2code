#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,c;
    scanf("%d",&t);
    while(t--)
    {
       int  n,i,j;
       c=0;
       long long s;
       s=0;
       scanf("%d",&n);
       int a[10001],b[10001];
       for(i=0;i<n;i++)
       {
        scanf("%d",&a[i]);
        s=s+a[i];
       }
       for(j=0;j<n;j++)
       {
        scanf("%d",&b[j]);

       }
       i=n-1;
       sort(b,b+n);
       while(1)
       {
         s=s-b[i];
         c++;
         i--;
         if(s<=0)
            break;
        }
       printf("%d\n",c);
    }
    return 0;
}
