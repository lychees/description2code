#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s,r,m;
    	cin>>s>>r;
    	sort(r.begin(),r.end());
    	int j=0;
    	for(int i=0;i<s.size() && j<r.size();i++)
    	{
    		if(s[i]>r[j])
    		{
    			s[i]=r[j];
    			j++;
    		}
    	}
    	cout<<s<<"\n";
    }
    return 0;
}
