#include<iostream>
using namespace std;
int isPrime(int n)
{
    int ctr=0;
    for(int i=2;i<n;i++)
    {
        if(n%i==0)
            ctr++;
    }
    if(ctr==0)
        return 1;
    else
        return 0;
}
int main(void)
{
    int cases,n,i,ctr,input;

    cin>>cases;

    while(cases--)
    {
        ctr=0;
        cin>>n;

        for(i=0;i<n;i++)
        {
            cin>>input;
            if(input!=0&&input!=1)
            {
                if(isPrime(input))
                    ctr++;
            }
        }
        cout<<ctr<<endl;
    }

    return 0;
}
