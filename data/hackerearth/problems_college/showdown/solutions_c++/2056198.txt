#include <bits/stdc++.h>
using namespace std;

int main() {
	
	int t;
	cin>>t;
	vector<int> f,p,f1,p1;
	int i;
	for(i=0;i<t;i++){
		int x;
		cin>>x;
		f.push_back(x);
	}
	for(i=0;i<t;i++){
		int x;
		cin>>x;
		p.push_back(x);
	}
	
	sort(f.begin(),f.end());
	sort(p.begin(),p.end());
	f1=f;
	p1=p;
	long long ans=0,ans1=0;
	for(i=p.size()-1;i>=0;i--){
		vector<int>::iterator it = upper_bound(f.begin(),f.end(),p[i]);
		if(it==f.end()){
			it = lower_bound(f.begin(),f.end(),p[i]);
			if(it!=f.end()){
				ans+=50;
				f.erase(it);
			}
		}
		else{
			ans+=100;
			f.erase(it);
		}
	}
	for(i=0;i<=p1.size()-1;i++){
		vector<int>::iterator it = upper_bound(f1.begin(),f1.end(),p1[i]);
		if(it==f1.end()){
			it = lower_bound(f1.begin(),f1.end(),p1[i]);
			if(it!=f1.end()){
				ans1+=50;
				f1.erase(it);
			}
		}
		else{
			ans1+=100;
			f1.erase(it);
		}
	}
	cout<<max(ans,ans1)<<endl;
	return 0;
}
