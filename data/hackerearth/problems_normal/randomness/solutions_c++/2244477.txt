#include <iostream>
using namespace std;

int hcf(int n1, int n2)
{
    if (n2!=0)
       return hcf(n2, n1%n2);
    else 
       return n1;
}
int main()
{
	int t;
	cin >> t;
	while(t--){
   	int x,y,c;
   	cin >>x>> y >> c;
   	int num,den;
   	if(c >= (x+y)){
   		cout << "1/1" << endl;
   	}
   	else{
   		if(c>=x && c>=y)
   		{
   			num = 2*x*y - (x+y-c)*(x+y-c);
   			den = 2*x*y;
   		}
   		else if(c <= x && c<=y){
   			num = c*c;
   			den = 2*x*y;
   		}
   		else if(c<=x && c>y){
   			num = c*c - (c-y)*(c-y);
   			den = 2*x*y;
   		}
   		else{
   			num = c*c - (c-x)*(c-x);
   			den = 2*x*y;
   		}
   		int h = hcf(den,num);
   		num = num/h;
   		den = den/h;
   		cout << num << "/" << den <<endl;
   	}
	}
    return 0;
}
