def binary_search(query, choc_cumulative):
	min = 0
	max = choc_cumulative.__len__() - 1
	result = -1		#starting form -1 contains the result just greater than query
	while min<=max:
		mid = (min+max)/2
		if choc_cumulative[mid]>query:
			result = mid
			max = mid-1
		elif choc_cumulative[mid]==query:
			result = mid
			return result+1
		else:
			min = mid+1
	return result+1



N = input()
chocs = map(int, raw_input().split())
cumulative_chocs = []
sum = 0

for c in chocs:
	sum+=c
	cumulative_chocs.append(sum)

Q = input()

for i in range(Q):
	query = input()
	index = binary_search(query, cumulative_chocs)
