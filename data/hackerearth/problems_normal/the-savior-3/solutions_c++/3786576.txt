#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int n,t;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int arr[n];
		
		for(int i=0; i<n; i++)
		{
			cin>>arr[i];
		}
	
		int cnt = 0;
		
		for(int j=0;j<n;j++)
		{
			for(int k=j+1;k<n;k++)
			{
				if((arr[j]+arr[k])%2 == 0 && arr[j]!=arr[k])
					cnt++;	
			}
		}
		
		cout<<cnt<<endl;
	}
	
	return 0;
}
