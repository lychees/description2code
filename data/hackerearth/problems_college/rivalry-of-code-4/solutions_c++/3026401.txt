#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	char a[5005],b[5005];
    	scanf("%s",a);
    	scanf("%s",b);
    	
    	int n=strlen(a);
    	int m=strlen(b);
    	char t[5005];
    	int c;
    	if(n>m){
    		strcpy(t,a);
    		strcpy(a,b);
    		strcpy(b,t);
    		c=n;n=m;m=c;
    	}
    	
    	int i,j,min=n;
    	
    	for(i=0;i<m-n+1;i++){
    		c=0;
    		for(j=0;j<n;j++){
    			if(b[j+i]!=a[j]){
    				c++;
    			}
    		}
    		if(c<min) min=c;
    	}
    	
    	cout<<min<<endl;
    }
    return 0;
}
