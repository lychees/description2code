ll pow_mod(ll a, ll b) {
	ll res = 1;
	while(b) {
		if(b & 1)
			res = (res * a) % MOD;
		a = (a * a) % MOD;
		b >>= 1;
	}
	return res;
}
int main() {
	int t;
	si(t);
	while(t--) {
		int n;
		si(n);
		int res = INT_MAX;
		for(int i = 1; i < n; ++i) {
			int a = i, b = n - i;
			int tmp_res = 0;
			while(!(a == 1 && b == 1)) {
				if(a == b) {
					tmp_res = INT_MAX;
					break;
				}
				if(a < b) {
					b -= a;
				} else {
					a -= b;
				}
				tmp_res++;
			}
			res = min(res, tmp_res);
		}
		if(n == 1) {
			res = -1;
		}
		pi(res + 1);
		pn;
	}
	return 0;
}
