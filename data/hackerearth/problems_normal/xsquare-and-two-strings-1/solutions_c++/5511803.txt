#include<iostream>
#include<stdlib.h>
#include<string.h>
using namespace std;
int cmp(const void*a, const void*b)
{
	return *(int*)a-*(int*)b;
}

int main()
{
	int t;
	char s1[100005],s2[100005];
	cin >> t;
	for(int i=0; i<t; i++)
	{
		cin >> s1 >> s2;
		long long int n1=strlen(s1);
		long long int n2=strlen(s2);
		qsort(s2,n2,sizeof(s2[0]),cmp);
		bool flag = false;
		for(long long int j=0; j<n1; j++)
		{
			int key=s1[j];
			int l=0,r=n2-1;
			while(l<=r)
			{
				int m=(l+r)/2;
				if(key==s2[m])
				{
					flag=true;
					break;
				}
				else if(key<s2[m])
					r=m-1;
				else l=m+1;
			}
		}

		if(flag)
			cout<<"Yes"<<endl;
		else cout<<"No"<<endl;
	}
	return 0;
}
