#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int x, y;
    
    cin >> x >> y;
    
    if (x == y)
    	cout << 1;
    else
   	{
   		if (x > y)
   		{
   			while (y > 0)
   			{
   				x = x/2 + 1;
   				y--;
   			}
   			
   			cout << x;
   		}
   		else
   		{
   			while (x > 0)
   			{
   				y = y/2 + 1;
   				x--;
   			}
   			
   			cout << y;
   		}
   	}
    
    return 0;
}
