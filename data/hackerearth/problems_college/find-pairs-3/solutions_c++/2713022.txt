typedef pair<int,int> PI;
typedef vector<int> VI;
typedef long long LL;
LL pos[1000001],neg[1000001];
int main()
{
    int test,i,n,num;
    LL ans=0;
    cin>>test;
    while(test--)
    {
       ans=0;
       cin>>n;
       for(i=0;i<=1000000;i++){pos[i]=0;neg[i]=0;}
       for(i=0;i<n;i++)
       {
	  cin>>num;
	  if(num>=0)pos[num]++;
	  else neg[abs(num)]++;
       }
       for(i=0;i<=1000000;i++)if(pos[i])ans+=pos[i]+(pos[i]*(pos[i]-1))/2;
       for(i=0;i<=1000000;i++)if(neg[i])ans+=neg[i]+(neg[i]*(neg[i]-1))/2;
       cout<<ans<<endl;
       
    }
    return 0;
}
