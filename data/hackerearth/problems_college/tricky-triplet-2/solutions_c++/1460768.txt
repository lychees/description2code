#include<bits/stdc++.h>
using namespace std;
int a[100000];
int main()
{
	int i,j,n,k,ans,t=1;
	//scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		for(i=0;i<n;i++){
			scanf("%d",&a[i]);
		}
		ans=0;
		for(i=0;i<n-2;i++){
			for(j=i+1;j<n-1;j++){
				for(k=j+1;k<n;k++){
					if((2*a[j])==(a[i]+a[k])){
						ans++;
					}
				}
			}
		}
		printf("%d\n",ans);
	
	}
	return 0;
}
