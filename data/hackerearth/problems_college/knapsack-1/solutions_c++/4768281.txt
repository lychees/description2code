#include <bits/stdc++.h>

using namespace std;

int main()
{
    
   	int t,n,k,K;
   	cin>>t;
   	while(t--)
   	{
   		map<int,bool> mp;
   		cin>>n>>k;
   		int a[n];
   		K=k;
   		for(int i = 0 ;i<n;i++)
   		{
   			cin>>a[i];
   			mp[a[i]] = true;
   		}
   		
   		int min = INT_MAX;
   		for(int j = 0;j<n;j++)
   		{
   			for(int i = a[j];i>0;i--)
   			{
   				//cout<<"i----->   "<<i<<endl;
   				if(mp[i] == true)
   				{
   					
   					k = k%i;
   					i = k;
   					//cout<<K<<"firstk===------------>    "<<k<<endl;
   					
   				}
   			}
   			if(k<min)
   				min = k;
   		}
   		//cout<<"min---->       "<<min<<endl;
   		k = min;
   		if(k==0)
   		{
   			cout<<K<<endl;
   		}
   		else
   		{
   			//cout<<K<<"k===------------>    "<<k<<endl;
   			int i = 1;
   			while(i<k)
   			{
   				if(mp[k-i]==true)
   				{
   					cout<<K-i<<endl;
   					break;
   				}
   				i++;
   			}
   			//cout<<K<<"ik===------------>    "<<k<<endl;
   			
   			if(k == i)
   				cout<<K-k<<endl;
   		}
   	}
    return 0;
}
