#include <iostream>
using namespace std;
int a[100000];
void fibo(int n)
{
	a[0] = 0;
	a[1] = 1;
	for(int i = 2;i <=n;i++)
	{
		a[i] = a[i-1] + a[i-2];
	}
}
int main()
{
    int t,n;
    cin >> t;
    while(t--)
    {
    	cin >> n;
    	fibo(n);
    	if(n == 0)
    	{
    		cout << "1 0"<< endl;
    	}
    	else
    	{
    		cout << a[n-1] << " " << a[n] << endl;
    	}
    }
    return 0;
}
