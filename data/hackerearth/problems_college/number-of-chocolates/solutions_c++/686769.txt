    #include <iostream>
    #include <vector>
    #include <algorithm>
    #include <fstream>
    #include <stdio.h>
    using namespace std;

#define sld(x) scanf("%lld", &x)
    long long  a[2000000];
    #define PB push_back
    #define MP make_pair
    long long  gcd(long long  x, long long  y){
        if(x < y) return gcd(y, x);
        if(y == 0) return x;
        return gcd(y, x % y);
    }
    vector<pair<long long , long long > > v;

    int main()
    {
        long long  n, q, ans, i, x, x1, x2, x3, x4,x5,x6,x7,x8;
        //freopen("input.txt", "r", stdin);
        cin>>n>>q;
        a[0] = 0;
        for(i = 1; i <= n; i++){
            sld(x);
            a[i] = a[i - 1] + x;
        }
        for(i = 1; i <= q; i++){
            cin>>x1>>x3;
            cin>>x2>>x4;
            x1++;
            x2++;
            x3++;
            x4++;
            x5 = min(x1, x2);
            x6 = max(x1, x2);
            x7 = min(x3, x4);
            x8 = max(x3, x4);

            v.clear();
            v.PB(MP(x5, x6));
            v.PB(MP(x7, x8));
            sort(v.begin(), v.end());
            x1 = v[0].first;
            x2 = v[0].second;
            x3 = v[1].first;
            x4 = v[1].second;

            //cout<<x1<<" "<<x2<<" "<<x3<<" "<<x4<<endl;
            if(x3 > x2){
                //cout<<"Yes";
                ans = a[x2] - a[x1 - 1] + a[x4] - a[x3 - 1];
            }
            else{
                ans = a[ max(x4, x2) ] - a[x1 - 1];
            }
            cout<<ans<<endl;
        }
        return 0;
    }
