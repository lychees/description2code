#include <iostream>
#include<cstdio>
#include<cstdlib>
#include<vector>
#include<algorithm>
#include<map>
#include<list>
using namespace std;

#define MAX 2000
int main() 
{
	long long n,k,i,j,prod,ans=0;
	cin>>n>>k;
	vector<long long> values;
	
	// sort and find product of a pair and then perform a Binary Search	
	
	for(i=0;i<n;i++)
	{
		cin>>j;
		values.push_back(j);
	}
	
	sort(values.begin(),values.end());
	
	for(i=0;i<values.size();i++)
	{
	//	cout<<values[i]<<" ";
	}
	
	for(i=0;i<n-2;i++)
	{
		for(j=i+1;j<n-1;j++)
		{
			//cout<<values[i]*values[j]<<" ";
			prod=values[i]*values[j];
			long long key = (long long) k/prod;
			if(key>=1)
			{
				// perform binary Search on key
				vector<long long>::iterator up;
				up=upper_bound(values.begin(),values.end(),key);
				
				if(up-values.begin()>j)
               ans = ans + (up - (values.begin()+j) - 1);
				
			}
		}
	//	cout<<endl;
	}
	
	cout<<ans<<endl;
//	cout<<endl;
	
	
	return 0;
}
