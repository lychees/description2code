#include <bits/stdc++.h>
using namespace std;
long long int a[1000000];
int main()
{
    long long int i,j,t,n,s,z,o,ans;
    cin>>t;
    while(t--){
    	cin>>n;
    	j=0;
    	z=0;
    	o=0;
    	for(i=0;i<n;i++){
    		scanf("%lld",&a[i]);
    		if(a[i]&(1<<0)){
    			o++;
    		}
    		else{
    			z++;
    		}
    	}
    	ans=o*z;
    	cout<<ans<<endl;
    	
    }
    return 0;
}
