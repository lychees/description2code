#include <iostream>
using namespace std;

int main()
{
    string str;
    cin>>str;
    int n = str.length();
    int count = n;
    int m = 1;
    for(int i=0; i<n; i++)
    {
        if(str[i]==str[i+1]){
        count = count+m;
        m++;
        }
        else
        	m = 1;
    }
    cout<<count<<"\n";
    return 0;
}
