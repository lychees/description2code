using namespace std;
typedef long long ll;

int main()
{
    ios_base::sync_with_stdio(false);
    ///freopen("input.txt", "r", stdin);
    int t;
    cin>>t;
    while(t--)
    {
        ll n,k,nz=0,temp;
        cin>>n>>k;
        for(int i=0;i<n;i++)
        {
            cin>>temp;
            if(temp==0)
                nz++;
        }
        if(nz>k)
            cout<<n-nz+k<<endl;
        else
            cout<<n-((k-nz)%2)<<endl;
    }
    return 0;

}
