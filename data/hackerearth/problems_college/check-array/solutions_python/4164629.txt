for _ in range(input()):
	n=input()
	arr=map(int,raw_input().split())
	total=0
	for item in arr:
		total+=item
		
	prevsum=0
	current_item=0
	flag=1
	for key in arr:
		current_item=key
		x=(total - current_item)-prevsum
		if x==prevsum:
			print "YES"
			flag=5
			break
		prevsum=prevsum+current_item
	if flag==1:
		print "NO"
