#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,t,i;
	scanf("%d %d",&n,&t);
	int a[n],b[n];
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	b[0] = a[0];
	for(i=1;i<n;i++)
	{
		b[i] = a[i] + b[i-1];
		//printf("%d ",b[i]);
	}
	//printf("\n");
	for(i=0;i<t;i++)
	{
		int c,d;
		scanf("%d %d",&c,&d);
		printf("%d\n",b[d-1]-b[c-1]+a[c-1]);
	}
	
}
