def chandu(text):
    ans = ""
    ans += text[0]
    for i in range(1, len(text)):
        if text[i] != text[i-1]:
            ans += text[i]
    return ans


T = int(raw_input())

for i in range(T):
    text = raw_input()
    print chandu(text)
