#include <bits/stdc++.h>
using namespace std;
int main()
{
    int tc;
    cin>>tc;
    map<string,int> m;
    while(tc--)
    {
        string s;
        cin>>s;
        m[s]++;
    }
    map<string,int>::iterator it;
    int n = m.size();
    string a[n];
    int b[n];
    int i=0;
    for(it = m.begin();it != m.end();++it)
    {
        a[i] = it->first;
        b[i] = it->second;
        i++;
    }
    for(int i=n-1;i>=0;i--)
    {
        cout<<a[i]<<" "<<b[i]<<endl;
    }
}
