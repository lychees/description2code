#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
   int tests,n,i;
   double sumx,sumy,x,y;
   sumx=sumy=0;
   cin>>tests;
   while(tests>0)
   {
   	sumx=sumy=0;
   	cin>>n;
   	i=n;
   	while(i>0)
   	{
   		cin>>x>>y;
   		sumx=sumx+x;
   		sumy=sumy+y;
   		i--;
   	}
   	sumx=sumx/double(n);
   	sumy=sumy/double(n);
   	cout<<fixed<<setprecision(2);
   	cout<<"("<<sumx<<" ";
   	cout<<fixed<<setprecision(2);
   	cout<<", "<<sumy<<")\n";
   	tests--;
   }
}
