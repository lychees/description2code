#include <bits/stdc++.h>
#define _ ios_base::sync_with_stdio(false);cin.tie(0);
using namespace std;
#define pb push_back
#define pob pop_back
#define pf push_front
#define pof pop_front
#define mp make_pair
#define all(a) a.begin(),a.end()
#define bitcnt(x) __builtin_popcountll(x) 
#define MOD 1000000007
#define total 500005
#define MAX 2000002
#define CMAX 26
 
typedef unsigned long long int uint64;
typedef long long int int64;
 
vector<int>v;

bool chck(int val,int n,int k){
	int count=1;
	int prev=0;
	for(int i=1;i<n;i++){
		if(v[i]-v[prev]>=val){
			count++;
			prev=i;
		}
	}
	if(count>=k)
	return true;
	return false;
}

int main(){
	int t,n,k,i,val;
	cin>>t;
	while(t--){
	cin>>n>>k;
	v.clear();
	for(i=0;i<n;i++){
		scanf("%d",&val);
		v.pb(val);
	}
	sort(all(v));
	int lo=0,hgh=1e7,ans=-1;
	
	while(lo<=hgh){
		int mid=(lo+hgh);
		mid/=2;
		if(chck(mid,n,k)){
			ans=mid;
			lo=mid+1;
		}
		else{
			hgh=mid-1;
		}
	}
	cout<<ans<<endl;
	}
	return 0;
}
