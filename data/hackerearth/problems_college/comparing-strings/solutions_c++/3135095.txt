#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char s1[100000],s2[100000];
		scanf("%s",s1);
		scanf("%s",s2);
		int i,j;
		int l1=strlen(s1);
		int l2=strlen(s2);
		int d=0;
		for(i=0;i<l1;i++)
		{
			int c=0;
			for(j=0;j<l2;j++)
			{
				if(s1[i]==s2[j])
				{
					cout<<"YES"<<"\n";
					c++;
					d++;
					break;
				}
			}
			if(c!=0)
			{
				break;
			}
		}
		if(d==0)
		{
			cout<<"NO"<<"\n";
		}
		
	}
}
