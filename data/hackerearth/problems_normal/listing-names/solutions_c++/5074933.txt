 #include<bits/stdc++.h>
using namespace std;
#define ll long long
#define li long int
#define w while

int main()
{
	ll n,i;
	cin>>n;
	map<string,ll> m;
	vector<string> v;
	string s;
	for(i=0;i<n;i++)
	{
		cin>>s;
		m[s]++;
		if(m[s]==1)
		v.push_back(s);
	}
	sort(v.begin(),v.end());
	vector<string>:: iterator it;
	it=v.begin();
	while(it!=v.end())
	{
		cout<<*it<<" "<<m[(*it)]<<endl;
		it++;
	}
	return 0;
}
