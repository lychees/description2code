#include<iostream>
#include <iomanip>
#include <cstdio>
using namespace std;

int main()
{
    int t,i,j;
    double a,p=1,s[2005],n1,r;
    
    cin >> t;
    
    for(i=0;i<t;i++)
    {
                    cin >> n1; 
                    cin >> r;                
                    a = r*r*6/(n1*(n1+1)*(2*n1+1));
                    s[i] = a * (n1*n1*(n1+1)*(n1+1))/(4*r*r*r);
    }
    for(i = 0;i<t;i++)
    {
          p = p *s[i];          
    }
    
   // cout <<setprecision(4) << p << endl;
   printf("%.4lf\n", p);
    
    return 0;
}
