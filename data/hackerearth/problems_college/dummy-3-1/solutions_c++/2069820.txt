#include<iostream>
using namespace std;

int main()

{
    
    int t;
    cin>>t;


    while(t--)
    {

        int n;
        cin>>n;

        int a[n+1],temp[n+1],k;

        bool res=true;

        for(int i=1;i<=n;i++)
            cin>>a[i];

        for(int i=1;i<=n;i++)
            temp[a[i]]=i;

        for(int i=1;i<=n;i++)
            if(a[i]!=temp[i])
                {res=false;}

        res?(cout<<"inverse"<<endl):(cout<<"not inverse"<<endl);

    }

}
