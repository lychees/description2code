#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long i,n,sum=0,j,k,num;
	cin >> n;
	std::vector<long long> vec;
	for(i=0;i<n;i++)
	{
		cin >> num;
		vec.push_back(num);
	}
	sort(vec.begin(),vec.end());
	if(n==1)
		cout << vec[0] << endl;
	else if(n==2)
		cout << abs(vec[0]-vec[1]);
	else
	{
		for(j=0,k=vec.size();j<=vec.size()/2;j++,k--)
		{   if(j!=vec.size()/2)
			{
			sum+=vec[k-1]-vec[j];
			sum+=vec[k-1]-vec[j+1];
			}
			else
			sum+=vec[j]-vec[0];

		}
		cout<<sum;
	}
	return 0;
}
