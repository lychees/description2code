#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
using namespace std;
int hashTable[26];
int table[26];
int dp[500005];
int main() {
    string str;
    str="hackerearth";
    int len=str.size();
    for(int i=0;i<len;i++) hashTable[str[i]-'a']++;
    cin>>str;
    for(int i=0;i<len-2;i++) dp[i]=0;
    for(int i=len-1;i<str.size();i++) {
        dp[i]=dp[i-1];
        bool flag=false;
        memset(table,0,sizeof(table));
        for(int j=i;j>i-len;j--) {
            if(str[j]>='A'&&str[j]<='Z') table[str[j]-'A']++;
            else if(str[j]>='a'&&str[j]<='z') table[str[j]-'a']++;
        }
        for(int j=0;j<26;j++) if(table[j]!=hashTable[j]) {
            flag=true;
            break;
        }
        if(!flag) dp[i]+=1;
    }
    int q;
    scanf("%d",&q);
    while(q--) {
        int low,high;
        scanf("%d%d",&low,&high);
        low--,high--;
        int lowVal=min(high,low+len-2);
        int ans=dp[high]-dp[lowVal];
        printf("%d\012",max(ans,0));
    }
    return 0;
}
