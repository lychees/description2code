w = raw_input()
leastLen = 10000000
n = input()
temp = w
while n:
	n -= 1
	a = raw_input()
	if len(a) <= leastLen and a[:len(w)] == w:
		if len(a) == len(w):
			temp = a
			break
			
		temp = a
		leastLen = len(a)
	
print temp
