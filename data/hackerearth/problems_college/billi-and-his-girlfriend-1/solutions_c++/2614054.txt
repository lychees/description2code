#include<iostream>
#include<math.h>
using namespace std;

int calc(long long int n,long long int b,long long int mod,long long int prev)
{
	long long int i,m=n;
	for(i=1;i<=b/2;i=i*2)
		{
			n=(n*n)%mod;
		}
		n=(prev*n)%mod;
	if(i==b)
	{
	return n;
	}
	else if(i==b-1)
	{
	return (m*n)%mod;
	}
	else
	calc(m,b-i,mod,n);
}

int main()
{
	long long int a,b,n;
	int t,k,mod,j,r,p,s,i;
	cin>>t;
	while(t--)
	{
		cin>>a>>b>>k;
		mod=pow(10,k);
		n=a%mod;
		n=calc(n,b,mod,1);
		s=0;
		while(k--)
		{
			r=n%10;
			n=n/10;
			s=s+r;
		}
		cout<<s<<endl;
	}
	
}
