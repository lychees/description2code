#include <iostream>
#include<math.h>
using namespace std;
int isnotfab(long num)
{
	int a=0;
	int b=1;
	int c=a+b;
	while(c<num)
	{
		a=b;
		b=c;
		c=a+b;
	}
	if(c==num)
	return 0;
	else
	return 1;
}
int isprime(long num)
{
	int ctr=0;
	int i;
	for(i=2;i<=sqrt(num);i++)
	{
		if(num%i==0)
		ctr++;
	}
	if(ctr==0)
	return 1;
	return 0;
}
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	long n,flag=0,i,sum=0;
    	cin>>n;
    	if(n<=3)
    	flag=2;
    	for(i=4;i<n;i++)
    	{
    		if(isnotfab(i))
    		{
    			sum+=i;
    		}
    	}
    	if(sum%2!=0)
    	{
    		if(isprime(sum))
    		flag=1;
    		else
    		flag=0;
    	}
    	else
    	flag=2;
    	if(flag==0)
    	cout<<"Toe"<<endl;
    	else if(flag==1)
    	cout<<"Tic"<<endl;
    	else
    	cout<<"Tac"<<endl;
    }
    return 0;
}
