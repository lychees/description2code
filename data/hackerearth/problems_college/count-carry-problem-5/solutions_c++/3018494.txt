#include <bits/stdc++.h>
#define ll long long 
using namespace std;

int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	ll x,y,k=0,k1=0,i;
    	cin>>x>>y;
    	
    	ll a[1000];
    	ll b[1000];
    	
    	memset(a,0,sizeof(a));
    	memset(b,0,sizeof(b));
    	
    	while(x !=0)
    	{
    		a[k++]=x%10;
    		x/=10;
    	}
    
    	while(y !=0)
    	{
    		
    		b[k1++]=y%10;
    		y/=10;
    	}
    	ll c=0,cn=0;
    	i=0;
    	while(i<k || i<k1)
    	{
    		ll g=c+a[i]+b[i];
    		if(g>9)
    		{
    		c=1;
    		cn++;
    		
    		}
    		else
    		c=0;
    		i++;
    	}
    	if(cn==0)
    	{
    		cout<<"No carry operation"<<endl;
    	}
    	
    	else if(cn==1)
    	{
    		cout<<cn<<" carry operation"<<endl;
    	}
    	else
    	{
    		
    		cout<<cn<<" carry operations"<<endl;
    	}
    	
    	
    	
    }
    return 0;
}
