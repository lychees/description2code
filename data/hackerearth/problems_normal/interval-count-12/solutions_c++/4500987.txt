#include <bits/stdc++.h>
#define ll long long
#define pb push_back
#define mk make_pair
#define pi pair<int, int>
#define fi first
#define se second
using namespace std;

int main() {
	int t;
	cin >> t;
	while(t--) {
		int n;
		ll l, r;
		scanf("%d%lld%lld", &n, &l, &r);
		int a[n];
		for(int i = 0; i < n; i++) {
			cin >> a[i];
		}
		ll ans = 0;
		for(int i = 1; i < (1 << n); i++) {
			ll p = 1;
			int ch = 0, ct = 0;
			for(int j = 0; j < n; j++) {
				if(i & (1 << j)) {
					if(p <= r / a[j]) {
						p *= a[j];
					} else {
						ch = 1;
					}
					ct++;
				}
			}
			if(ch == 0) {
				ll val = r / p;
				val -= l / p;
				if(l % p == 0) {
					val++;
				}
				if(ct % 2) {
					ans += val;
				} else {
					ans -= val;
				}
			}
		}
		cout << ans << endl;
	}
	return 0;
}
