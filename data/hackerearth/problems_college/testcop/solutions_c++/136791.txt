#include<stdio.h>

int gcd(int a,int b)
{
       if(b==0)
           return a;
       else
           return gcd(b,a%b);


}


int main()
{
  int t,n;
  scanf("%d",&t);
  while (t>0)
  {
    scanf("%d",&n);
    int a[n];
    int i=0;
    for(i=0;i<n;i++)
         scanf("%d",&a[i]);
    int x=a[0];
    i=0;
    for(i=1;i<n;i++)
        x=gcd(x,a[i]);
    if (x==1)
       printf("YES\n");
    else
       printf("NO\n");
    t=t-1;
   }

  return 0;
}
