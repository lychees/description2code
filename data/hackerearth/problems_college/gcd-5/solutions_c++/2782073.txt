#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		scanf("%d",&n);
		long long a,b,f=0;
		cin>>a;
		for(int i=1;i<n;i++)
		{
			cin>>b;
			if(a<=b && f==0)
			a=b;
			else
			f=1;
		}
		if(f==0)
		cout<<"YES\n";
		else
		cout<<"NO\n";
	}
	return 0;
}
