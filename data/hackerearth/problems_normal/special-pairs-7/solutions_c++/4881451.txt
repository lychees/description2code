#include<bits/stdc++.h>
using namespace std;
#define ll long long

int t,n,a[100005],freq[1500005],p,dp[1500005][21],ans;

int main()
{
	cin>>t;
	while(t--)
	{
		ans=0;
		cin>>n;
		memset(freq,0,sizeof(freq));
		memset(dp,0,sizeof(dp));

		for (int i = 0; i < n; ++i)
		{
			cin>>a[i];
			freq[a[i]]++;
		}

		for (int i = 0; i < (1<<20) ; ++i)
		{
			// if(freq[i])
				// cout<<i<<" ";
			dp[i][0]=freq[i];

			if(i&1)
				dp[i][0]+=dp[i^1][0];

			for (int j = 1; j <= 20 ; ++j)
			{
				dp[i][j]=dp[i][j-1];

				if(i&(1<<j))
					dp[i][j]+=dp[i^(1<<j)][j-1];
			}
		}
		// cout<<endl;

		for (int i = 0; i < n; ++i)
		{
			// cout<<dp[((1<<20)-1)^a[i]][20]<<" ";
			ans+=dp[((1<<20)-1)^a[i]][20];
		}

		cout<<ans<<endl;
	}
	return 0;
}
