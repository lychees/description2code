def gcd(a, b):
    if b == 0:
        return a
    else:
        return gcd(b, a % b)


def main():
    test_cases = int(raw_input())
    for _ in xrange(test_cases):
        L, B, H = raw_input().split()
        l, b, h = int(L), int(B), int(H)
        Volume = l*b*h
        possible_dim = gcd(l, gcd(b, h))
        number = Volume/(possible_dim**3)

        print int(possible_dim % 1000000007), int(number % 1000000007)

    return 0


if __name__ == "__main__":
    main()
