#include <iostream>
#include <string>
#include <cctype>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int date[31];
	 for(int i=1;i<31;i++){
    	date[i]=0;
    }
	string str;
	getline(cin,str);
    for(int i=0;i<n;i++){
    	getline(cin,str);
    	int len = str.length();
    	int currDate =-1;
    	for(int j=0;j<len;j++){
    		if(isdigit(str[j])){
    			currDate =-1;
    			currDate = (str[j]-'0');
    			if(isdigit(str[j+1]))
    				currDate = currDate*10 +(str[j+1]-'0');
    			j++;
    			if(str[0]=='G' && currDate>0)
    				date[currDate]+=2;
    			else
    				date[currDate]+=1;
    		}	
    	}
    }
    	
    int max = 0;
    int finalDate =-1;
    for(int i=1;i<31;i++){
    	if(date[i]>max){
    		max=date[i];
    		finalDate = i;
    	}
    }
   if(finalDate ==19 ||finalDate==20)
   	cout<<"Date";
   else
   	cout<<"No Date";
    	
    return 0;
}
