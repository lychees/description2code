#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,q,qr,i,j,a[1000][1000];
    cin>>n>>q;
    for(i=0;i<n;i++)
    for(j=0;j<n;j++)
    cin>>a[i][j];
    
    while(q--)
    {
    	cin>>qr;
    	qr=(qr%360);
    	if(qr==0||qr==360)
    	{
    		for(i=0;i<n;i++)
    		{
    		for(j=0;j<n;j++)
    		cout<<a[i][j]<<" ";
    		cout<<endl;
    		}
    		cout<<endl;
    	}
    	if(qr==90)
    	{
    		for(i=0;i<n;i++)
    		{
    			for(j=n-1;j>=0;j--)
    			cout<<a[j][i]<<" ";
    			cout<<endl;
    		}cout<<endl;
    	}
    	if(qr==180)
    	{
    		for(i=n-1;i>=0;i--)
    		{
    			for(j=n-1;j>=0;j--)
    			cout<<a[i][j]<<" ";
    			cout<<endl;
    		}cout<<endl;
    	}
    	if(qr==270)
    	{
    		for(i=n-1;i>=0;i--)
    		{
    			for(j=0;j<n;j++)
    			cout<<a[j][i]<<" ";
    			cout<<endl;
    		}cout<<endl;
    	}
    }
    
    return 0;
}
