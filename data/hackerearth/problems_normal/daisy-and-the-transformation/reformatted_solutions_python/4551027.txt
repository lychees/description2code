for i in xrange(input()):
    orig, repl, step = raw_input().split()
    size, step, count = len(orig), int(step), 0
    for j in xrange(size):
        if orig[j] != repl[j]:
            count += 1

    if count > step:
        print "No"
    elif count == step or (step - count) % 2 == 0:
        print "Yes"
    else:
        print "No"
