#include<bits/stdc++.h>
using namespace std;
#define max 1000000
int prime[max];
vector <int>v;
void seive(){
	for(int i=2;i*i<max;i++){
		if(prime[i]==0 ){
			
			for(int j=i*2;j<max;j+=i){
				prime[j]=1;
				
			}
		}
	}
}
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
	seive();
	for(int i=2;i<max;i++){
		if(!prime[i])
			v.push_back(i);
	}
	int n;
	scanf("%d",&n);
	int elt=n;
	int i=0;
	while(n>1){
		if(n%v[i]==0){
				elt*=(v[i]-1);
				elt/=v[i];
			
		}
		while(n%v[i]==0){
			n/=v[i];
		}
	
		
	//	printf("v[i]=%d   n=%d elt=%d\n",v[i],n,elt);
		i++;
	}
	printf("%d\n",elt);
	}
}
