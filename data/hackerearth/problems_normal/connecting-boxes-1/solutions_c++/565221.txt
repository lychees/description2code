#include<stdio.h>
#include<string.h>
int main(){
    int N,Q,i,j,l,m;
    scanf("%d%d",&N,&Q);
    int hash[N+1][N+1],flag[N+1];
    memset(hash, 0, sizeof(hash[0][0]) * (N+1) * (N+1));
    memset(flag, 0, sizeof(flag));
    while(Q--){
        scanf("%d%d",&i,&j);
        if(i==j || flag[i] || flag[j] || hash[i][j] || hash[j][i])printf("NO\n");
        else{
            if(i>j){
                int temp=i;
                i=j;
                j=temp;
            }
            printf("YES\n");
            flag[i]=flag[j]=1;
            l=(i%N)+1;
            while(l!=j){
                m=(j%N)+1;
                while(m!=i){
                    hash[l][m]=hash[m][l]=1;
                    m=(m%N)+1;
                }
                l=(l%N)+1;
                
            }
        }
           }
}
