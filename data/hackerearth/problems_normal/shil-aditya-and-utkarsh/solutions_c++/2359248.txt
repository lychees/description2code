#include<bits/stdc++.h>
using namespace std;
long long int foo(int n,int r)
{
	 if(n==r) return 1;
     if(r==1) return n;
     return foo(n-1,r) + foo(n-1,r-1); 
}
int main()
{
    int n;
    cin>>n;
    vector<long long int>v(1001);
    v[0]=0;
    v[1]=0;
    v[2]=0;
    v[3]=2;
    //cout<<foo(4,2)<<endl;
    for(int i=4;i<=1000;i++)
    {
    	v[i]=foo(i-1,2)*2+v[i-1];
    }
    cout<<v[n]<<endl;
}
