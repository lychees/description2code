#include <iostream>
#include<string.h>
#include <algorithm>
using namespace std;

int main()
{
    int n;
    cin>>n;
    string s[n];
    for(int i=0;i<n;i++){
    	cin>>s[i];
    }
    int count;
    for(int i=0;i<n;i++){
    	reverse(s[i].begin(),
                s[i].end());
    	for(int j=i+1;j<n;j++){
    		if(s[j]==s[i])
    		{count=i;break;}
    	}
    }
    int len=s[count].length();
     string str = s[count];
    int a  = (len/2)  ;
    //cout<<s[count];
    char b = str[a];
    cout<<len<<" "<<b;
    return 0;
}
