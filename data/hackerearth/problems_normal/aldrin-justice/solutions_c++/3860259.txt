#include <iostream>
using namespace std;

int main()
{
	int p,q,r,s,t,bt1,bt2,mt1,mt2;
	cin>>t;
	while(t--){
	cin>>bt1>>bt2>>mt1>>mt2;
	if(bt1>bt2){p=bt1;q=bt2;}
	else {p=bt2;q=bt1;}
	if(mt1>mt2){r=mt1;s=mt2;}
	else {r=mt2;s=mt1;}
	if(q>r || s>p)cout<<"Line\n";
	else if(((p==s) && p!=q) || (r==q && r!=s) || (p<=r && p>=s && p==q) || (r<=p && r>=q && r==s)){cout<<"Point\n";}
	else {cout<<"Nothing\n";}
	
	}
    return 0;
}
