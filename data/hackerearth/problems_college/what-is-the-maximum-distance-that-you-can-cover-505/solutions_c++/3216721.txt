#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    map<int, int> vmap;
    double num_speed[n], num_consum[n], fuel;
    for(int i=0; i<n; i++)
       cin>>num_speed[i];
    for(int i=0; i<n; i++)
       cin>>num_consum[i]; 
    cin>>fuel;
    for(int i=0; i<n; i++)
       vmap[num_speed[i]]=num_consum[i];
       
    double max_capacity=0.00;
    map<int, int>::iterator it;
    for(it=vmap.begin(); it != vmap.end(); ++it)
        if((fuel/it->second)*it->first > max_capacity)
            max_capacity=(fuel/it->second)*it->first;
   // max_capacity = max_capacity*1000;
    //printf("%.3f \n",(max_capacity)/1000);  
    printf("%.3f", ((signed long)(max_capacity * 1000) * 0.001f));
  //  std::cout << std::fixed <<max_capacity*10.0/10.0<<endl;
    return 0;
}
