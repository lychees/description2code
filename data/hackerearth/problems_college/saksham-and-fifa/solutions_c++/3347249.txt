#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int n,t,p,t1,i,p1;
	cin>>n;
	cin>>t>>p;
	for(i=1;i<n;i++)
	{
		cin>>t1>>p1;
		if(t1>t)
		{
			t=t1;
			p=p1;
		}

		
	}
	//cout<<p<<" ";
	if(p==0)
		cout<<"Either Corner";
	else if(p<0)
		cout<<"Right Corner";
	else
		cout<<"Left Corner";
	return 0;
}
