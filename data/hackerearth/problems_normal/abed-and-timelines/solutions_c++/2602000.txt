int T, N, M, P;

int main()
{
    scanf("%d", &T);
    while (T--)
    {
       scanf("%d %d", &N, &M);
       int x1, y1, x2, y2;
       scanf("%d %d %d %d", &x1, &y1, &x2, &y2);
       scanf("%d", &P);
       int x = MIN(ABS(x2 - x1), N - ABS(x2 - x1));
       int y = MIN(ABS(y2 - y1), M - ABS(y2 - y1));
       
       int ans = (x + y)*P;
       if (ans <= 1000)
       {
           printf("%d\n", ans);
           printf("You saved the group.\n");
       }
       else
       {
           printf("%d\n", ans); 
           printf("Let go of the group.\n"); 
       }     
    }
    return 0;    
}
