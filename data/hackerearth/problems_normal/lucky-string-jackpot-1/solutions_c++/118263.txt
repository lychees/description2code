i64 calc(i64 n) {
	i64 ans = 0;
	i64 temp = 2;
	
	if(n <= 2) {
		return 1;
	}

	while(1) {
		if(n <= temp*(ans+1)) {
			i64 t = 2*(ans+1)-1;

			if(n >= t)
				return t;
			else 
				return max(n,2*ans-1);
		}
		else {
			n -= temp*(ans+1);
			temp *= 2;
			ans++;
		}
	}
	
}

int main() {
	int t;
	i64 n;

	cin >> t;
	
	while(t--) {
		cin >> n;
		cout << calc(n) << endl;
	}

	return 0;
}
