#include<iostream>
#include<algorithm>
typedef long long ll;
const ll p=1e9+7;
const int maxi=1e5+5;
using namespace std;

ll arr[maxi],n,sum;

int main()
{
 cin>>n;
 
 for(int i=0;i<n;i++)
  cin>>arr[i];
 
 sort(arr,arr+n);
 for(int i=0;i<n/2;i++)
 {
 	 sum=sum-2*arr[i];
 	 sum=sum+2*arr[n-i-1];
 }
  cout<<sum; 
}
