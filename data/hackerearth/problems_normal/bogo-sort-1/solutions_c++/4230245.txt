#include <iostream>
using namespace std;

#define MAX 10000

int N, A[MAX];

int main()
{
	cin >> N;
	for(int i = 0; i < N; i++)
		cin >> A[i];
	for(int i = 0; i < N; i++)
		for(int j = i + 1; j < N; j++)
			if(A[i] > A[j])
			{
				int temp = A[i];
				A[i] = A[j];
				A[j] = temp;
			}
	for(int i = 0; i < N; i++)
		cout << A[i] << endl;
	return 0;
}
