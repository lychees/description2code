#include<iostream>
using namespace std;
long int a[1000000],l,u,i,j;
void quick(long int *,long int,long int);
int main()
{
	int t;
	long int N;
	cin>>t;
while(t--)
{
	cin>>N;
	for(i=0;i<N;i++)
		cin >> a[i];
	l=0;
	u=N-1;
	quick(a,l,u);
	for(i=0;i<N;i++)
		cout<<a[i]<<" ";
	cout<<endl;
}
return 0;
}
 
void quick(long int a[],long int l,long int u)
{
   long int p,temp;
   if(l<u)
   {
   p=a[l];
   i=l;
   j=u;
    while(i<j)
   {
      while(a[i] <= p && i<j )
	 i++;
      while(a[j]>p && i<=j )
	   j--;
      if(i<=j)
      {
      temp=a[i];
      a[i]=a[j];
      a[j]=temp;}
  }
  temp=a[j];
  a[j]=a[l];
  a[l]=temp;
  quick(a,l,j-1);
  quick(a,j+1,u); 
 }
}
