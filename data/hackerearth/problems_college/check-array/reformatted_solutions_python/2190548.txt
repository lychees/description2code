for t in range(input()):
    n = int(raw_input())
    lst = map(int, raw_input().split())
    l = len(lst)
    summ1 = lst[0]
    summ2 = 0
    flag = 0
    for i in range(2, l, 1):
        summ2 += lst[i]
    for i in range(1, l-1, 1):
        summ1 = summ1+lst[i]
        summ2 = summ2-lst[i+1]
        if summ1 == summ2:
            flag = 1
            break
    if flag == 1:
        print "YES"
    else:
        print "NO"
