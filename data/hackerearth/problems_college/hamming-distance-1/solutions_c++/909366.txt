#include<stdio.h>
#include<stdlib.h>
long long int  hammingdistance(long long int x, long long int  y)
{
  long long int dist1 = 0, val = x ^ y;

  while(val)
  {
    ++dist1;
    val &= val - 1;
  }

  return dist1;
}
int main()
{
	long long int dec1, dec2,x,t;
	scanf("%lld",&t);
	while(t--){
	scanf("%lld",&dec1);
    scanf("%lld",&dec2);
	x = hammingdistance(dec1,dec2);
	printf("%lld\n", x);
}
return 0;
}
