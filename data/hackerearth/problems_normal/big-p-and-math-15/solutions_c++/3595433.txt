#include<iostream>
#include<string>
using namespace std;
int main()
{
	long A;
	long B;
	cin >> A;
	cin >> B;
	long t=1;
	long A1=0,B1=0;
	long A2=0,B2=0;
	while(A!=0 || B!=0)
	{
		int rA = A%10;
		int rB = B%10;
		if(rA==5)
			rA++;
		if(rB==5)
			rB++;
		A1 = A1 + rA*t;
		B1 = B1 + rB*t;

		rA = A%10;
		rB = B%10;
		if(rA==6)
			rA--;
		if(rB==6)
			rB--;
		A2 = A2 + rA*t;
		B2 = B2 + rB*t;

		A=A/10;
		B=B/10;
		t=t*10;
	}
	cout << A2+B2 <<" "<<A1+B1<<"\n";
	return 0;
}
