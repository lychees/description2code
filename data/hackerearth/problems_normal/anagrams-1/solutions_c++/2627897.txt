 #include <stdio.h>
 #include<string.h>
    char s[200],s1[200],ans[200];        
    int abs(int a) 
    {
            return a>0?a:0-a;
    }    
    int main()
    {        
        int t,i,j;
        scanf("%d",&t);
        getchar();
        while(t--)
        {
            int f[30]= {0},f1[30]= {0};
            gets(s);
            gets(s1);
            int l=strlen(s);
            for(i=0; i<l; i++)
            {
                if(s[i]<'a')
                    s[i]+='a'-'A';
                if(s[i]>='a'&&s[i]<='z')
                    f[s[i]-'a']++;
            }
            l=strlen(s1);
            for(i=0; i<l; i++)
            {
                if(s1[i]<'a')
                    s1[i]+='a'-'A';
                if(s1[i]>='a'&&s1[i]<='z')
                    f1[s1[i]-'a']++;
            }

            i=0;
            while(i<26&&f[i]==f1[i])
                i++;
            if(i==26)
                printf("YES\n");
            else
            {
                int j=i;
                while(i<26&&f[i]>=f1[i])
                    i++;
               if(i==j)
                while(i<26&&f[i]<=f1[i])
                    i++;
                if(i==26)
                {
                    int c=0,flag=0;
                    for(i=j; i<26; i++)
                    {
                        f[i]=abs(f1[i]-f[i]);
                        if(f[i]%2!=0)
                        {
                            if(flag)
                            {
                                printf("NO LUCK\n");
                                break;
                            }
                            flag=i+'a';
                        }
                        for(j=0; j<f[i]/2; j++)
                            ans[c++]=i+'a';
                    }
                    if(i==26)
                    {
                        for(i=0; i<c; i++)
                            printf("%c",ans[i]);
                        if(flag)
                            printf("%c",flag);
                        for(i=c-1; i>=0; i--)
                            printf("%c",ans[i]);
                        printf("\n");
                    }
                }
                else
                    printf("NO LUCK\n");
            }           
        }
        return 0;
    }
