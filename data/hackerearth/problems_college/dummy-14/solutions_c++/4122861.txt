#include<iostream>
#include<map>
#include<algorithm>
using namespace std;
long long int v[100006];long long int a[1000004];
map<long long int ,int >m;
void lo(){long long k=0;
	for(long i=1;i<=1003;i++){
		for(long j=i+1;j<=1003;j++){
			long long int s=i*i*i+j*j*j;
			m[s]=m[s]+1;
			if(m[s]>1){
				v[k]=s;k=k+1;
			}
		}
	}
	sort(v,v+k);long u=1;
	a[u]=v[0];u=u+1;
	
	for(long j=1;j<k;j++){
		if(v[j]!=v[j-1]){a[u]=v[j];u=u+1;
		}
	}
}
int main(){
	long t;
	cin>>t;
	lo();long long x;
	for(long i=0;i<t;i++){
		cin>>x;cout<<a[x]<<endl;
	}
}
