#include <bits/stdc++.h>
using namespace std;

#define MAXN 704977

int main()
{
	map<int, int> mp;
	for(int i = 1; i < 100; ++i) {
		for(int j = i; j*j*j + i*i*i <= MAXN; ++j) {
			mp[i*i*i + j*j*j]++;
		}
	}
	
	
	int t;
	cin >> t;
	while(t--) {
		
		int n;
		cin >> n;
		
		int ans = -1;
		for(map<int, int>::iterator it = mp.begin(); it != mp.end() && it->first < n; ++it) {
			if(it->second > 1) ans = it->first;
		}
		
		cout << ans << endl;
	}
	
}
