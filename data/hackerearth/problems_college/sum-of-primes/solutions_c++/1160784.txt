#include <iostream>
using namespace std;
#define N 1000001
int check_prime(unsigned long long int a){
	unsigned long long int c;
	for ( c = 2 ; c <= a/c ; c++ ){
		if ( a%c == 0 )
		return 0;
	}
	return 1;
}
int main()
{
	bool prime[N];
	for(int i=2;i<N;i++){
		prime[i]=true;
	}
	prime[1]=false;
	for(int i=2;i<N/i;i++){
		if(prime[i])
		for(int j=i;j*i<N;j++){
			prime[j*i]=false;
		}
	}
	int t;
	cin>>t;
	while(t--){
		long int s,e;
		cin>>s>>e;
		unsigned long long int sum = 0;
		for(long int i=s;i<=e;i++){
			if(prime[i]){
				sum = sum + i;
			}
		}
		if(sum>N-1){
			if(check_prime(sum))
		    cout<<"YES"<<endl;
		    else
		    cout<<"NO"<<endl;
		}
		else{
			if(prime[sum])
		    cout<<"YES"<<endl;
		    else
		    cout<<"NO"<<endl;
		}
	}
	return 0;
}
