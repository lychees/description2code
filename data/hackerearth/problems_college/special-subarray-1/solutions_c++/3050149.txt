#include <stdio.h>
int a[1000006],b[1000006]={0};
int main()
{
    int i,j,k,n,m,l,r;
    scanf("%d %d",&n,&m);
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    for(i=1;i<n-1;i++) {
    	if(a[i-1]>a[i]&&a[i+1]>a[i]) {
    		b[i]=1;
    	}
    }
    for(i=1;i<n;i++) {
    	b[i]+=b[i-1];
    }
    for(i=0;i<m;i++) {
    	scanf("%d %d",&l,&r);
    	if(r-l+1<=2)
    	printf("Yes\n");
    	else {
    		r=r-2;
    		if(b[r]-b[l-1]>0)
    		printf("No\n");
    		else
    		printf("Yes\n");
    	}
    }
    return 0;
}
