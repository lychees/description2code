def srt(l):
    l1 = []
    l2 = []
    e = 0
    o = 0
    s = ""
    for i in l:
        h = int(i)
        if (h % 2 == 0):
            l1.append(h)
            e += h
        else:
            l2.append(h)
            o += h
    l1.append(e)
    l2.append(o)
    list.sort(l1)
    list.sort(l2)
    l3 = l1+l2
    for i in l3:
        s += str(i)+" "
    return (s)
