#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    long long int n,ans;
    double ans1;
    int t;
    short a;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	ans1=(-1+sqrt(1+8*n))/2;
    	if(fmod(ans1,1)==0)
    	ans=ans1;
    	else
    	ans=(int)ans1+1;
    	a=ans%3;
    	switch(a)
    	{
    		case 0:cout<<"SL"<<endl;
					break;
    		case 1:cout<<"LB"<<endl;
					break;
    		case 2:cout<<"BS"<<endl;
					break;
    	}
    	
    }
    return 0;
}
