def main():
    x = raw_input()
    c = 1
    ans = 0
    p_c = x[0]
    for i in range(1, len(x)):
        if x[i] == p_c:
            c += 1
        else:
            ans += (c*(c+1))/2
            c = 1
        p_c = x[i]
    ans += (c*(c+1))/2
    print ans


if __name__ == '__main__':
    main()
