#include<bits/stdc++.h>
#define ll long long
#define gc getchar_unlocked
using namespace std;
void scan(ll &x)
{
    register char c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
 
int main()
{
	ll t;
	scan(t);
	while(t--)
	{
		ll a[100], i, j, k, n, cnt=0, size=0;
		map<ll, ll> m, prev; 
		vector<ll>v;
		scan(n);
		for(i=0; i<n; i++)
			scan(a[i]);
		for(i=0; i<n; i++)
		{
			size=v.size();
			for(j=0; j<size; j++)
			{
				if(!m[a[i]+v[j]])
					v.push_back(a[i]+v[j]);
				m[a[i]+v[j]]+=prev[v[j]];
			}
			if(!m[a[i]])
				v.push_back(a[i]);	
			m[a[i]]++;
			prev=m;
		}	
		printf("%lld\n", m[0]);
	}
	return 0;
}
