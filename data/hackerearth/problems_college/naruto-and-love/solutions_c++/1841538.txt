#include<bits/stdc++.h>
#include<vector>
#include<cstdio>
#include<algorithm>
#define FOR(i,n) for(int i=0;i<n;i++)
#define si(x) scanf("%d",&x)
#define INF 2000000000
using namespace std;
int mouse[150069],holes[150069],m,h;
bool test(int x)
{
    int i=0,j=0;
    while(i<m&&j<h)
    {
        if(abs(mouse[i]-holes[j])<=x)
            i++,j++;
        else
            j++;
    }
    if(i==m)
        return true;
    else
        return false;
}

int main()
{
    int t;
    si(t);
    while(t--)
    {
        si(m),si(h);
        FOR(i,m)
            si(mouse[i]);
        FOR(i,h)
            si(holes[i]);
        sort(mouse,mouse+m);
        sort(holes,holes+h);
        int high=INF,low=0,mid=0;
        while(low<high)
        {
            mid=(high+low)/2;
            if(test(mid))
                high=mid;
            else
                low=mid+1;
        }
        printf("%d\n",high);
    }
    return 0;
}
