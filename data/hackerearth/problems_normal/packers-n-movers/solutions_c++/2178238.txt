#include <bits/stdc++.h>

using namespace std;

#define ll long long 

int main()
{
	ll n,k;
	cin>>k;
	cin>>n;
    ll arr[n];
    ll cum[n+1];
    cum[0]=0;
    
    for(int i=0;i<n;i++)
    {
    	cin>>arr[i];
    	cum[i+1] = cum[i] + arr[i];
    }


    ll dp[n+1][k+1];

    for(ll i=0;i<=n;i++)
    	dp[i][0]=0;
    for(ll i=0;i<=k;i++)
    dp[0][i]=0;

    for(ll i=1;i<=n;i++)
    dp[i][1] = cum[i];
    
    for(ll i=1;i<=k;i++)
    dp[1][i] = arr[0];

    for(ll i=2;i<=k;i++)
    {
    	for(ll j=2;j<=n;j++)
    	{
    		int ans = INT_MAX;
    		for(ll p =1;p<=j;p++)
    		{
    		    //cout<<ans<<endl;
    		    ll temp =max(dp[p][i-1],cum[j]-cum[p]);
    			if(temp < ans)
    			ans = temp;
    		}
    		dp[j][i]=ans;
    	}
    }
    
    cout<<dp[n][k]<<endl;
    
    return 0;
}
