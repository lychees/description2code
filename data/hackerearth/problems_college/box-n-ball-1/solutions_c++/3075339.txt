#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long int l,b,h,min,ans;
    long int t,i;
    scanf("%ld",&t);
    while(t--)
    {
    	scanf("%lld%lld%lld",&l,&b,&h);
    	min=l;
    	if(b<min)
    	min=b;
    	if(h<min)
    	min=h;
    	ans=min/2;
    	printf("%lld",ans);
    	if(min%2==0)
    	printf(".0\n");
    	else
    	printf(".5\n");
    }
    return 0;
}
