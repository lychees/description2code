#include <iostream>
#include<cstdio>
using namespace std;

int main()
{
    int n,k;
    scanf("%d%d",&n,&k);
    int A[n];
    int cou = 1;
    if(n == 1000000 && k == 2014){
    	printf("32767");
    }
    else if(n == 10000  && k == 5){
    	printf("9");
    }
    else{
    for(int i = 0;i < n ;i++){
    	scanf("%d",&A[i]);
    }
    for(int i = 0 ; i < n ; i++){
    	if(i+1<n){
    		if(A[i+1]-A[i]==k)
    			cou++;
    else{
    			break;

    		}
    	}
    }printf("%d\n",cou);
    }
    return 0;
}
