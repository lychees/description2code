#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    scanf("%d",&n);
    vector<int>A(n);
    int m;
    //cin>>m;
    scanf("%d",&m);
    for(int i =0;i<m;i++)
    {
        int l,r;
         scanf("%d",&l); scanf("%d",&r);
      //  cin>>l>>r;
        l--;
        r--;
        A[l]+=1;
        if(r+1<n)
            A[r+1]-=1;
    }
    partial_sum(A.begin(),A.end(),A.begin());
    sort(A.begin(),A.end());
    int q;
   // cin>>q;
    scanf("%d",&q);
    while(q--)
    {
        int idx;
        //cin>>idx;
         scanf("%d",&idx);
        printf("%d\n",A.size()-(lower_bound(A.begin(),A.end(),idx)-A.begin()));//<<endl;

    }


    return 0;
}
