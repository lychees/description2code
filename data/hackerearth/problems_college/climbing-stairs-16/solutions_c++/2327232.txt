#include<iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    long long int fib[101];
    fib[1]=1;
    fib[2]=2;
    for(int i=3;i<=100;i++)
    {
    fib[i]=fib[i-1]+fib[i-2];
    }
    
    while(t--)
    {int n;
    cin>>n;
    cout<<fib[n]<<endl;
}
return 0;


}
