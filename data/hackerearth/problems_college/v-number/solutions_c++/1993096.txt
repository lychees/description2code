#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	int n=str.size();
    	if(n%2!=0)
    	{
    		cout<<"NO"<<endl;
    	}
    	else
    	{
    		int mid=n/2,i,sum1=0,sum2=0;
    		for(i=0;i<mid;i++)
    		{
    			sum1+=str[i]-48;
    		}
    		for(i=mid;i<n;i++)
    		{
    			sum2+=str[i]-48;
    		}
    		if(sum1==sum2)
    		cout<<"YES"<<endl;
    		else
    		cout<<"NO"<<endl;
    	}
    }
}
