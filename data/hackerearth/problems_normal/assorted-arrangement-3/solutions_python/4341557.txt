def find_num(j,d) :
    val = d/c[j]
    arr = c[j+1::]
    while 1 :
        val += 1
        count = 0
        z = val*c[j]
        for _ in xrange(len(arr)) :
            if z % arr[_] == 0 :
                count = 1
                break
        if count == 0 :
            return z

n,m = map(int,raw_input().split())
c = map(int,raw_input().split())
a = map(int,raw_input().split())
pre = 0
min_v = c[0]

for i in a :
    min_v = min(c[i-1],min_v)
    pre = find_num(i-1,pre)
print pre
