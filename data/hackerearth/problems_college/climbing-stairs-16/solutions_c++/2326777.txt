#include <iostream>
using namespace std;
 
int main()
{
    int t,i,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(n<3)
    	cout<<n<<endl;
    	else
    	{
    	long long int a=1,b=2,c;
    	for(i=3;i<=n;i++)
    	{
    		c=a+b;
    		a=b;
    		b=c;
    	}
    	cout<<c<<endl;
    	}
    }
    return 0;
}
