unsigned long long mulmod(unsigned long long a,unsigned long long b,unsigned long long c)
{
unsigned long long sum = 0,y = a % c;
while( b )
{
if(b & 1)
{
sum += y;
if( sum >= c)
sum -= c;
}
y+=y;
if(y>=c) y-=c;
b >>= 1;
}
return sum;
}
