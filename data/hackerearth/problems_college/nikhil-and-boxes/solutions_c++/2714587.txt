#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
     	int n;
		long double l,b,h,r;
    	cin>>n;
		while(n!=0)
		{
			n--;
			cin>>l;
			cin>>b;
			cin>>h;
			if((l<=b)&&(l<=h))
			 r=l/2;
			else if((b<=l)&&(b<=h))
			 r=b/2;
			else
			 r=h/2;
			std::cout << std::fixed << std::setprecision(1) << r<<endl;
		}
    
    return 0;
}
