using namespace std;
int h1[26];
int h2[26];
int main()
{
    int t,i;
    string s1,s2;

    cin >> t;

    while(t--){
        mem(h1,0);
        mem(h2,0);
        cin >> s1;
        cin >> s2;

        for (i = 0; i < s1.size(); i++){
            h1[s1[i]-97]++;
        }
        for (i = 0; i < s2.size(); i++){
            h2[s2[i]-97]++;
        }
        int flag = 0;
        for (i = 0; i < 26; i++){
            if(h1[i] > h2[i]){
                flag = 1;
                break;
            }
        }
        if(flag){
            cout << "-1"<<endl;
        }
        else {
            int count = 0;
            for (i = 0; i < 26; i++){
                if(h1[i] < h2[i]){
                    count++;
                }
            }
            cout << count << endl;
        }
    }
}
