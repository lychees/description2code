#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int n,i,count;
    cin >> n;
    char a[100000];
    while(n--)
    {
    	cin >> a;
    	count = 0;
    	for(i=0;i<(strlen(a)-1);i++)
    	{
    		if(a[i] == 'X' && a[i+1] == 'X')
    			count++;
    		else if(a[i] == 'Y' && a[i+1] == 'Y')
    			count++;
    	}
    	cout << count << endl;
    	count = 0;
    }
    return 0;
}
