import sys


def max_conversion(num, arr):
    if num in arr:
        return arr[num]
    else:
        arr[num] = max(
            num,
            max_conversion(
                num /
                2,
                arr) +
            max_conversion(
                num /
                3,
                arr) +
            max_conversion(
                num /
                4,
                arr))
    return arr[num]


for line in sys.stdin:
    inp = line
    if inp:
        num = int(inp)
        arr = {0: 0, 1: 1}
        print max_conversion(num, arr)
    else:
        break
