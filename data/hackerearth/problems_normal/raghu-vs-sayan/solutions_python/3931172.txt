for _ in xrange(int(raw_input())):
	r,s,n=map(int,raw_input().split());	A=map(int,raw_input().split());	A.sort();
	i,j,val_r,val_s=0,0,0,0;
	while i<n and s>=val_s+A[i]:	val_s+=A[i];	i+=1;
	while j<n and r>=val_r+A[j]:	val_r+=A[j];	j+=1;
#	print i,j
	if i>j:	print "Sayan Won"
	elif i<j:	print "Raghu Won"
	else:	print "Tie"
