#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000000007
int main()
{
    long long int T,a,b;
    double c;
    slld(T);
    while(T--)
    {
        slld(a);
        slld(b);
        scanf("%lf",&c);
        int steps=0;
        while(c>b)
        {
            c/=b;
            steps++;
            if(c<=b)
                break;
            c/=a;
            steps++;
        }
        if(c>a)
        {
            printf("Adam\n");
        }
        else
        {
            if(steps%2==0)
                printf("Adam\n");
            else
                printf("Bob\n");
        }
    }
	return 0;
}
