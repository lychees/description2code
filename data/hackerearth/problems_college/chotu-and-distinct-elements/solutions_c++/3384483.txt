#include <iostream>
using namespace std;

int comp(const void *a, const void *b)
{
	return (*((int *)b) - *((int *)a));
}

int main()
{
    int t, n, i, j;
    cin>>t;
    for ( ; t; t--) {
    	cin>>n;
    	int rose[n];
    	for (i = 0; i < n; i++) {
    		cin>>rose[i];
    	}
    	qsort(rose, n, sizeof(int), comp);
    	i = 0; j = n-1;
    	while (i <= j) {
    		cout<<rose[i]<<" ";
    		while ((i < j) && (rose[i] == rose[i+1])) {
    			i++;
    		}
    		i++;
    		if (i < j) {
    			cout<<rose[j]<<" ";
    		}
    		while ((i < j) && (rose[j] == rose[j-1])) {
    			j--;
    		}
    		j--;
    	}
    	cout<<endl;
    }
    
    return 0;
}
