#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int n, itr;
	vector<int> buckets;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		int temp;
		cin>>temp;
		for(int j=0;j<temp;j++)
		buckets.push_back(i+1);
	}
	cin>>itr;
	for(int i=0;i<itr;i++)
	{
		int caseval;
		cin>>caseval;
		cout<<buckets[caseval-1]<<"\n";
	}
	return 0;
}
