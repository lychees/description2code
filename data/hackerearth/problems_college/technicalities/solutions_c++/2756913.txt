#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define mod 1000000007



int main()
{
	
	ll t;
	cin>>t;
	while(t--)
	{
		ll n;
		cin>>n;
		string s;
		cin>>s;
		ll sz = s.size();
		ll ans=0;
		for(int i=0;i<sz;i++)
		{
			ans = (ans*10+s[i]-'0')%mod;
		}
		cout<<ans<<"\n";
	}
	return 0;
}
