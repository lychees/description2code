#include <bits/stdc++.h>
#define f(i,a,b) for(int i=a;i<b;++i)
using namespace std;

typedef long long int ll;

ll a[1001];
ll b[1001];
ll ans;
int n;


void copy(){
	f(i,0,n){
		b[i]=a[i];
	}

}


bool pos(ll c){
	copy();
	b[0]=a[0]-c;
	if(b[0] < 1)
		b[0]=1;
	f(i,1,n){
		if(b[i] <= b[i-1]){
			if(b[i]+c <= b[i-1])
				return false;
			else 
				b[i]=b[i-1]+1;
		}
		else{
			if(b[i]-c <= b[i-1]){
				if(c)
					b[i]=b[i-1]+1;
			}
			else{
				b[i]-=c;
				if(b[i] < 1)
					b[i] = 1;
			}
		}
	}
	return true;
}

void solve(){
	ans = 0;
	ll left = 0;
	ll right = (ll)1e10;
	while(left < right){
		ll mid = (left+right)/2;
		if(pos(mid)){
			ans = mid;
			right = mid-1;
		}
		else{
			left = mid+1;
		}
	}
	if(left < ans)
		if (pos(left))
				ans = left;
	if(right < ans)
		if (pos(right))
			ans = right;
}


int main(){
	int t;
	cin>>t;
	while(t--){
		cin>>n;
		for(int i=0;i<n;++i)
			cin>>a[i];
		solve();
		cout<<ans<<endl;
	}
	return 0;
}
