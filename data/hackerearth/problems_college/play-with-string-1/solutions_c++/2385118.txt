#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
string s1,s2;
cin>>t;
while(t--)
{int len=0;
cin>>s1;
len=s1.size();

s2=s1;
reverse(s2.begin(),s2.end());
bool flag=1;
for(int i=1;i<len-1;i++)
{
	if(abs(s1[i]-s1[i-1])!=abs(s2[i]-s2[i-1]))
	{flag=0;
	break;
	}
}
if(flag)
cout<<"Funny"<<endl;
else
cout<<"Not Funny"<<endl;
}
return 0;
}
