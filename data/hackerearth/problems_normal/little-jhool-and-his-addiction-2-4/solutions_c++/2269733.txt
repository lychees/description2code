
#include<bits/stdc++.h>

using namespace std;

#define si(x) scanf("%d",&x)
#define pri(x) printf("%d",x)
#define sll(x) scanf("%lld",&x)
#define pll(x) printf("%lld",x)
#define sstr(s) scanf("%s",s)
#define pstr(s) printf("%s",s)
#define nl printf("\n")
#define ll long long int

int main()
{
    int t;
    si(t);
    while(t--)
    {
	int n,k;
	si(n); si(k);
	int a[n];
	for(int i=0;i<n;++i)
	    si(a[i]);
	sort(a,a+n);
	if(n==2)
	{
	    pri(0); nl;
	    if(k>0)
		pstr("Chick magnet Jhool!");
	    else
		pstr("Lucky chap!");
	    nl;
	    continue;
	}
	int sum[n/2],K=0;
	int lim=n/2;
	for(int i=0,j=n-1;i<=lim;++i,--j)
	{
	    sum[K++]=a[j]+a[i];
	}
	sort(sum,sum+K);
	if(sum[K-1]-sum[0]>k)
	{
	    pri(sum[K-1]-sum[0]); nl;
	    pstr("No more girlfriends!");
	}
	else if(sum[K-1]-sum[0]<k)
	{
	    pri(sum[K-1]-sum[0]); nl;
	    pstr("Chick magnet Jhool!");
	}
	else
	{
	    pri(sum[K-1]-sum[0]); nl;
	    pstr("Lucky chap!");
	}
	nl;
    }
    return 0;
}
