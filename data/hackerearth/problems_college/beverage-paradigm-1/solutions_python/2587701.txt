# print 'Hello World!'
from itertools import permutations
from itertools import combinations

n = int(raw_input())
a = []
while len(a) != n:
	a += map(int,raw_input().split())
b = int(raw_input())
f = 0
for i in range(0,len(a)+1):
	for subset in combinations(a,i):
		if len(subset) == 3:
			# print subset
			if sum(subset) == b:
				print "True"
				f = 1
				break
			else:
				continue
if f == 0:
	print "False"
