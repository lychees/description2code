#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int a,b;
    	cin>>a>>b;
    	int count=0;
    	for(int i=a;i<=b;i++)
    	{
    		int k=i;
    		int len=0;
    		while(k>0)
    		{
    			k/=10;
    			len++;
    		}
    		k=i;
    		int check=1;
    		int pow1=pow(10,len-1);
    		int pow2=10;
    		for(int j=0;j<len/2;j++)
    		{
    	//		cout<<k<<endl;
    			int a=k/pow1;
    			int b=k%pow2;
    			if(a!=b)
    			{
    				check=0;
    				break;
    			}
    			else{
    				k%=pow1;
    				k/=pow2;
    				pow1/=100;
    				
    				
    			}
    			
    		}
    	//	cout<<i<<" :"<<check<<endl;
    		if(check)count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
