
 #include <bits/stdc++.h>
using namespace std;
int main() {
	int n, q, x, y, val, hash[102][10005] = {0};
	cin >> n;
	for (int i = 0; i < n; ++i) {
		cin >> x;
		hash[x][i] = 1;
	}
	for (int i = 0; i < 102; ++i)
		for (int j = 1; j < 10005; ++j)
			hash[i][j] += hash[i][j - 1];
	cin >> q;
	while (q--) {
		cin >> x >> y >> val;
		int ans = hash[val][y];
		if (x) ans -= hash[val][x - 1];
		cout << ans << endl;
	}
	return 0;
}
