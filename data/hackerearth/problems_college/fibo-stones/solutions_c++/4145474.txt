// Example program
#include <iostream>
#include <string>
using namespace std;

int main()
{
    unsigned long int n,m,a,temp=0,res=0,i=0,j=1,x;
    cin>>n>>m;
    for(x=0;x<n;x++)
    {
        res=res+j;
        temp=j;
        j=i+j;
        i=temp;
    }
    for(x=0;x<m;x++)
    {
        cin>>a;
        res=res-a;
    }
    cout<<res;
    return 0;
}
