#include<iostream>
#define ll long long
#define M 1000000007
using namespace std;

ll int modular_expo(ll int base,ll int exp,ll int MOD)
{
    ll res=1;
    while(exp>0) {
       if(exp%2==1) res=(res*base)%MOD;
       base=(base*base)%MOD;
       exp>>=1;
    }
    return res%MOD;
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
        int t;
        cin>>t;

        while(t--)
        {
            ll int v,n;
            cin>>n;
            n--;
            n%=M-1;
            v=n*(M-2);
            v%=M-1;

            cout<<modular_expo(2,v,M)<<"\n";
        }

}
