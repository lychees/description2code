#include<iostream>
#include<map>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a,b;
		map<int,int> mymap;
		for(int i=0;i<n;i++)
		{
			cin>>a>>b;
			mymap[a]++;
			mymap[b]--;
		}
		map<int,int> :: iterator it=mymap.begin();
		int ans=0;
		int mx=0;
		for(;it!=mymap.end();it++)
		{
			ans+=it->second;
			mx=max(mx,ans);
		}
		cout<<mx<<endl;
	}
	
	return 0;
}
