#include <stdio.h>
using namespace std;

int arr[100005];

bool check(int n, int k, int max){
	int i=0, j=1;
	//printf ("mid=%d\n", max);
	//printf ("k=%d\n", k);
	//printf ("i=%d\nj=%d\n", i, j);
	while(j<n){
		if (k==0)
			return false;
		if(arr[j]-arr[i]<=max){
			j++;
		}
		else{
			k--;
			i=j-1;
			//j=i+1;
		}
	}
	//printf ("true\n\n");
	//printf ("k=%d\n", k);
	//printf ("i=%d\nj=%d\n\n", i, j);
	
	return true;
}


int bs(int n, int k){
	int left=1, right=1000000000, mid, ans;
	while(left<=right){
		mid=(left+right)/2;
		if (check(n, k, mid)){
			ans=mid;
			right=mid-1;
		}
		else
			left=mid+1;
	}
	return ans;
}

int main(){
	int t;
	scanf ("%d", &t);
	while (t--){
		int n, k;
		scanf ("%d %d", &n, &k);
		for (int i=0; i<n; i++)
			scanf ("%d", &arr[i]);
		printf ("%d\n", bs(n, k));
		
		
	}
	
	
	
	return 0;
}
