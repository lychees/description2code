#include <iostream>
#include <string>
using namespace std;
int main()
{
	int t;
	cin>>t;
	
	while( t-- )
	{
		string st1, st2, st3;;
		cin>>st1>>st2;
		for( int i = 0; i < st1.size(); i++ )
		{
			if( st1[i] != st2[i] )
				st3 += "1";
			else
				st3 += "0";
		}
		cout<<st3<<endl;
	}
}
