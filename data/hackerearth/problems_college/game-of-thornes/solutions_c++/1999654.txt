#include <iostream>
using namespace std;

int main()
{
    int mp[26],i,len;
    string str;
    cin>>str;
    for(i=0;i<26;i++)
    	mp[i]=0;
    len=str.length();
    for(i=0;i<len;i++){
    	mp[str[i]-'a']++;
    }
    
    if(len%2==0){
    	for(i=0;i<26;i++)
    		if(mp[i]%2!=0){
    			cout<<"NO"<<endl;
    			return 0;
    		}
    		cout<<"YES"<<endl;	
    }
    else{
    	int count=2;
    	for(i=0;i<26;i++){
    		if(mp[i]%2!=0)
    			count--;
    		if(count==0){
    			cout<<"NO"<<endl;
    			return 0;
    		}
    	}
    	cout<<"YES"<<endl;
    		
    }
    
    return 0;
}
