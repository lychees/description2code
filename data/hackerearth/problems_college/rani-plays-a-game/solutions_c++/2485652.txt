#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    long long n,k;
    scanf("%d",&t);
    while(t--){
        scanf("%lld %lld",&n,&k);
        int ans[n+5];
        ans[1]=0;
        int flag;
        for(int i=2;i<=n;i++){
            flag=0;
            for(int j=1;j<=k;j++){
                if(ans[i-j]==0){
                    flag=1;
                }
            }
            if(flag==1)ans[i]=1;
            else ans[i]=0;
            //cout<<i<<" "<<ans[i]<<endl;
        }
        if(ans[n]==1)printf("Rani\n");
        else printf("Nandu\n");
    }
    return 0;
}
