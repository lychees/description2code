#include<bits/stdc++.h>
using namespace std;
int main()
{
    long int n,x;
    cin>>n;
    long a[n],i;
    vector<long int>vec;
    for(i=0;i<n;i++)
    {
       cin>>x;
       vec.push_back(x);
    }
    sort(vec.begin(),vec.end());
   // for(i=0;i<n;i++)
     //   cout<<vec[i]<< " ";
    long layer=1,buy=0,pos=0;
    while(pos<n)
    {
        long z=layer*layer;
         if(z<vec[pos])
            buy+=z;
         else
         {
             buy+=(z-vec[pos]);
             pos++;
         }
         layer++;
    }
    cout<<buy<<endl;

}
