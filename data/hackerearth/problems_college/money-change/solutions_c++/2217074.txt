#include<bits/stdc++.h>
using namespace std;
long long int count( long long int S[],long long  int m,long long  int n )
{

       long long  int table[10000];
    int i;

        memset(table, 0, sizeof(table));
        table[0] = 1;
        for(int i=0; i<m; i++)
                for(int j=S[i]; j<=n; j++)
                        table[j] += table[j-S[i]];
        return table[n];
}
int main()
{
        long long int i;

        }
