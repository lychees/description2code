
#include<string>
#include<algorithm>
#include<iostream>
#include<cstdio>
#include<stack>
using namespace std;

int main()
	{int t,T;
	 
	 cin >> T;
	 getchar();
	 for(t=0;t<T;t++){
		string s;
		stack<string> S;
		getline(cin,s);
		string delimiter = " ";
		size_t pos = 0;
		string token;
		while ((pos = s.find(delimiter)) != string::npos) {
    			token = s.substr(0, pos);
    		//	cout << token << endl;
			S.push(token);
    			s.erase(0, pos + delimiter.length());
		}
		//cout << s << endl;
		S.push(s);
		while(S.size()!=0)
			{
				string ele=S.top();
				cout << ele <<" ";
				S.pop();
			}
		printf("\n");
	}
		return 0;
	}
