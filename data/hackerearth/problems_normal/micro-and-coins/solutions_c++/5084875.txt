#define ll long long int
#define maxN 10
#define maxVal 100000000
#define minVal -100000000
#define mod 1000000007LL

#define gcd(a,b) __gcd(a,b)

using namespace std;

int n, m;
int g[maxN + 5][maxN + 5];
int dp[(1 << maxN) + 5][maxN + 5];

int main()
{
	int t, i, u, v, j, k, z;
	scanf("%d", &t);
	while (t--)
	{
		scanf("%d%d", &n, &m);
		memset(g, 0, sizeof(g));
		memset(dp, 0, sizeof(dp));
		while (m--)
		{
			scanf("%d%d", &u, &v);
			g[u - 1][v - 1] = g[v - 1][u - 1] = 1;
		}
		for (i = 0; i<n; i++)
			dp[(1 << i)][i] = 1;
		for (i = 0; i<(1 << n); i++)
			for (j = 0; j<n; j++)
			{
				if ((i&(1 << j)))
					for (k = 0; k<n; k++)
						if (j != k&&g[k][j] && dp[(i ^ (1 << j))][k])
						{
							dp[i][j] = 1;
							break;
						}
			}
		z = 0;
		for (i = 0; i<n; i++)
			if (dp[(1 << n) - 1][i])
			{
				z = 1;
				break;
			}
		if (z)
			printf("Yes\n");
		else
			printf("No\n");
	}
	return 0;
}
