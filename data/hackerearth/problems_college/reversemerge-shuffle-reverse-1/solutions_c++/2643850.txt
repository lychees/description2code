#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
#include<string>
#include<cassert>
using namespace std;
int i,n,j,k,ind,mn;
string res;
char str[1000001];
int S[257],C[257],M[257][5000];
int main()
{
  scanf("%s",str);
  n=strlen(str);
  for(i=0;i<n;i++)
  {
    C[str[i]]++;
    if(i)
      for(j='a';j<='z';j++)
        M[j][i]=M[j][i-1];
    M[str[i]][i]++;
  }
  for(i='a';i<='z';i++)
    C[i]/=2;
  res="";
  ind=n;
  for(i=0;i<n/2;i++)
  {
    mn=-1;
    for(j=ind-1;j>=0;j--)
    {
      for(k='a';k<='z' && M[k][j]+S[k]>=C[k];k++);
      if(k<='z')
        break;
      if(S[str[j]]<C[str[j]])
      {
        if(mn==-1)
          mn=j;
        if(str[j]<str[mn])
          mn=j;
      }
    }
    assert(mn!=-1);
    ind=mn;
    res+=str[ind];
    S[str[ind]]++;
  }
  for(i='a';i<='z';i++)
    assert(C[i]==S[i]);
  for(i=0;i<res.size();i++)
    C[res[i]]--;
  for(i='a';i<='z';i++)
    assert(!C[i]);
  printf("%s\n",res.c_str());
  return 0;
}
