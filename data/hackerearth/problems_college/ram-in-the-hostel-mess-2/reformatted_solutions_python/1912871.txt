from operator import mul
from fractions import Fraction


def nCk(n, k):
    return int(reduce(mul, (Fraction(n-i, i+1) for i in range(k)), 1))


for i in xrange(input()):
    n = int(raw_input())
    sum = 0
    if n % 2 == 0:
        k = n/2
        for j in xrange(k+1):
            sum += nCk(n-j+1, j)
        print sum**2
    else:
        k = (n+1)/2
        for j in xrange(k+1):
            sum += nCk(n-j+1, j)
        print sum**2
