#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--){
    string s;
    cin >> s;
    stack<char> q;
    q.push(s[0]);
    for(int i = 1; i < s.size(); i++)
    {
        if(s[i] != q.top()) q.push(s[i]);
    }
    vector<char> v;
    while( !q.empty())
    {
        v.push_back(q.top());
        q.pop();
    }
    for(int i = v.size() - 1 ; i >= 0; i--)
    {
        cout << v[i];
    }
    cout << endl;
    }
    return 0;
}
