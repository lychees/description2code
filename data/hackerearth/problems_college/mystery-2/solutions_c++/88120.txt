#include <iostream>
using namespace std;

int main()
{
    int t,arr[50],i,j,a,b;
    cin>>t;
    while(t--)
    {
    	cin>>a;
    	cin>>b;
    	i=-1;
    	while(a)
    	{
    		i++;
    		arr[i]=a%b;
    		a=a/b;
    		
    	}
    	for(j=i;j>=0;j--)
    	cout<<arr[j];
    	cout<<endl;
    }
    return 0;
}
