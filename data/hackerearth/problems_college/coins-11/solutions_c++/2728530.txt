using namespace std;
inline ll getn(){
	ll n=0, c=getchar();
	while(c < '0' || c > '9')
		c = getchar();
	while(c >= '0' && c <= '9')
		n = (n<<3) + (n<<1) + c - '0', c = getchar();
	return n;
}

bool prime(ll n)
{
	if(n==1)
	return 0;
	ll i;
	for( i=2;i*i<=n;i++)
	{
		if(n%i==0)
		return 0;
		
	}
	return 1;
}

int main()
{
		std::ios_base::sync_with_stdio(0);
	ll t,n;
	t=getn();
while(t--)
{
	n=getn();
	if(n<=3)
	{cout<<n<<endl;
	continue;
	}
	if(prime(n))
	cout<<"2"<<endl;
	else
	cout<<"1"<<endl;
	
}
	
	return 0;
}
