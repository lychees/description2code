#include<bits/stdc++.h>
using namespace std;
struct node
{
	int w;
	node *a[26];
};
node *head;
void add(string s,int w)
{
	node *temp=head;
	for(int i=0;i<s.length();i++)
	{
		if(temp->a[s[i]-'a']==NULL)
		{
			node *t=new node;
			t->w=w;
			for(int j=0;j<26;j++) t->a[j]=NULL;
			temp->a[s[i]-'a']=t;
			temp=temp->a[s[i]-'a'];
		}
		else {temp=temp->a[s[i]-'a'];temp->w=max(temp->w,w);}
	}
}
int query(string s)
{
	node *temp=head;
	for(int i=0;i<s.length();i++)
	{
		if(temp->a[s[i]-'a']==NULL) return -1;
		temp=temp->a[s[i]-'a'];
	}
	return temp->w;
}
int main()
{
    head=new node;
    head->w=0;
    for(int i=0;i<26;i++) head->a[i]=NULL;
    int n,w,q;
    cin>>n>>q;
    string s;
    while(n--)
    {
    	cin>>s>>w;
    	add(s,w);
    }
    while(q--)
    {
    	cin>>s;
    	cout<<query(s)<<endl;
    }
}
