#include <iostream>
#include <string>
#include <set>
using namespace std;

int main() {
	int len, i;
	cin >> len;
	string str;
	cin >> str;
	
	int max=0, j;
	string newStr;
	set<char> characters;
	set<char>::iterator it;
	
	for (i=0; i<len; i++) {
		if (characters.find(str[i]) == characters.end()) {
			j=0;
			while (j<newStr.length() && newStr[j] > str[i]) {
				j++;
			}
			string temp(newStr.begin(), newStr.begin()+j);
			string temp2(1, str[i]);
			string temp3(newStr.begin()+j, newStr.end());
			temp.append(temp2);
			newStr = temp;
			string::iterator sit;
			
			for (sit=temp3.begin(); sit!=temp3.end(); sit++) {
				characters.erase(*sit);
			}
			characters.insert(str[i]);
		}
		if (str[i] > max) {
			string newChar(1, str[i]);
			max = (int)str[i];
			newStr = newChar;	
		}
	}
	
    cout << newStr << endl;
    return 0;
}
