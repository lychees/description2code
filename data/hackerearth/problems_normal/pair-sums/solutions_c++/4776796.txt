#include<iostream>
#include<unordered_map>
using namespace std;
main()
{
    int num,sum,i;
    int *arr;
    cin>>num>>sum;
    arr=new int[num];
    for(i=0;i<num;i++)
        cin>>arr[i];
    unordered_map<int,int> mymap;
    for(i=0;i<num;i++)
    {
        if(mymap.find(arr[i])!=mymap.end())
        {
            if(arr[i]*2==sum)
            {
                cout<<"YES";
                mymap.clear();
                exit(0);
            }
        }
        else
        {
            mymap[arr[i]]=sum-arr[i];

        }
    }
    for(i=0;i<mymap.size();i++)
    {
        if(mymap.find(sum-arr[i])!=mymap.end())
            {
                cout<<"YES";
                mymap.clear();
                exit(0);
            }
    }

    cout<<"NO";
    delete(arr);
    mymap.clear();
}
