#include <iostream>
using namespace std;

int main() {
  int t;
  cin >> t;
  while (t--) {
    int m, n;
    cin >> m >> n;
    if (n == 0)
      cout << "0" << endl;
    else if (m >= n)
      cout << "10" << endl;
    else {
      n = n * 2;
      if (n % m == 0)
        cout << (n / m) * 5 << endl;
      else {
        while (n % m != 0)
          n++;
        cout << (n / m) * 5 << endl;
      }
    }
  }
  return 0;
}
