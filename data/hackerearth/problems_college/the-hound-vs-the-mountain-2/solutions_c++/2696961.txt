#include <iostream>
using namespace std;
int power(int a,int b)
{
	if(b==0)
		return 1;
	if(b==1)
		return a;
	int temp=power(a,b/2);
	temp*=temp;
	if(b&1)
		temp*=a;
	return temp;
}
int main()
{
    int t,a,b,bina,binb,i,val,r,temp;
    scanf("%d",&t);
    while(t--)
    {
    	bina=binb=0;
    	scanf("%d%d",&a,&b);
    	temp=a;
    	for(i=0;i<8;i++)
    	{
    		r=temp%10;
    		val=power(2,i);
    		bina+=(r*val);
    		temp/=10;
    	}
    	temp=b;
    	for(i=0;i<8;i++)
    	{
    		r=temp%10;
    		val=power(2,i);
    		binb+=(r*val);
    		temp/=10;
    	}
    	if(bina>=binb)
    	{
    		printf("The Hound\n");
    	}
    	else
    		printf("The Mountain\n");
    }
    return 0;
}
