#include <iostream>
#include<math.h>
void fun(long);
using namespace std;
long s,r;
int i=1;
int main()
{
    int t;
    long n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(n==1)
    	 cout<<"Case #"<<i++<<":"<<"YES"<<endl;
    	else
    	fun(n);
    }
    	return 0;
}
void fun(long n)
{
	    s=0;
    	while(n>0)
    	 {
    	 	r=n%10;
    	 	if(r!=0)
    	 	s+=pow(r,2);
    	 	n=n/10;
    	 }
    	
    	 	if(s==1)    	 
    	       cout<<"Case #"<<i++<<":"<<"YES"<<endl;
    	    else if(s>1 && s<=9)
    	       cout<<"Case #"<<i++<<":"<<"NO"<<endl;
    	       else fun(s);   
    return ;
}
