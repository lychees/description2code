#include <bits/stdc++.h>
using namespace std;

typedef long long LL;

const int N = 1e5 + 5;

int A[N];
LL C[N];
vector <int> factors[N];

void generate_factors(int n)
{
    for(int i=1;i<N;++i)
    {
        for(int j=i+i;j<N;j+=i)
        {
            factors[j].push_back(i);
        }
    }
}

void solve(int n)
{
    C[1] = A[1];
    for(int i=2;i<=n;++i)
    {
        LL x = -1;
        for(int j: factors[i])
        {
            x = max(x, C[j]);
        }
        C[i] = A[i] + x;
    }
}

int main()
{
    ios::sync_with_stdio(0); cin.tie(0);
    generate_factors(N);
    int n;
    cin>>n;
    for(int i=1;i<=n;++i) cin>>A[i];
    solve(n);
    for(int i=1;i<=n;++i) cout<<C[i]<<" ";
    cout<<"\n";
    return 0;
}
