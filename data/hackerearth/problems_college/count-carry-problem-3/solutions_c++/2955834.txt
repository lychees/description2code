#include <iostream>
using namespace std;
typedef long long int ll;
int main()
{
	ll rec,n1,n2,d1,d2;
	cin >>rec;
	while(rec--)
	{
		ll ctr=0;
		cin >> n1;
		cin >> n2;
		while(n1!=0&&n2!=0)
		{
			d1=n1%10;
			d2=n2%10;
			if(d1+d2>=10)
			ctr++;
			n1/=10;
			n2/=10;
		}
		if(ctr==0)
		{
			cout << "No carry operation"<<"\n";
			continue;
		}
		if(ctr==1)
		{
			cout << "1 carry operation"<<"\n";
			continue;
		}
		else
		{
			cout << ctr << " carry operations"<<"\n";
		}
	}
    return 0;
}
