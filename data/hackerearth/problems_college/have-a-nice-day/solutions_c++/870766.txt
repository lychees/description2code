#include <iostream>
#include<algorithm>
#define ll long long
using namespace std;

int main()
{
    ll t,n,i,j,sum;
    cin>>t;
    while(t--){
    	sum=0;
    	cin>>n;
    	ll m[1005]={0},w[1005]={0};
    	for(i=0;i<n;i++)
    		cin>>m[i];
    	for(j=0;j<n;j++)
    		cin>>w[j];
    	sort(m,m+n);
    	sort(w,w+n);
    	for(i=0;i<n;i++)
    		sum+=(m[i]*w[i]);
    	cout<<sum<<"\n";
    	
    }
    return 0;
}
