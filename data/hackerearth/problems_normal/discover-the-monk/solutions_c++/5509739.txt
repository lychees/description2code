#include <iostream>
#include<algorithm>
using namespace std;
long long a[1000000],n,q,b;
int bs(int n,int k)
{ int l=0;
  int h=n;
  int m;
 while(l<h)
 {  m=(l+h)/2;
 	if(a[m]<k)
      l=m+1;
      else if(a[m]>k)
      h=m-1;
      else 
      return m;
 }
 return l;
}
 
int main()
{
    cin>>n>>q;
    for(int i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    while(q--)
    {cin>>b;
     if(b>a[n-1])
      cout<<"NO"<<endl;
      else
      {
     if(a[bs(n-1,b)]==b)
      cout<<"YES"<<endl;
      else
      cout<<"NO"<<endl;
      }
    }
    
    return 0;
}
