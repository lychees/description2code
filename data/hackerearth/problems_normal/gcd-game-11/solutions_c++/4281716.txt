typedef long long ll;
ll MOD=1000000007;
ll pw(ll a,ll b){
  ll r=1;
  while(b>0){
     if(b&1){
       r=(r*a)%MOD;
        }
     b=b>>1;
     a=(a*a)%MOD;
  }
return r;
}
int rv[1001][1001][2];
int vis[1001][1001][2]={{}};
int fun(int a,int b,int player){
     if((player==0&&a==1)||(player==1&&b==1))return 0;
     if(vis[a][b][player])return rv[a][b][player];
     vis[a][b][player]=1;
    int g=__gcd(a,b);
    int ans=0;
      if(player==0){
        ans|=(!fun(a,b-1,!player));
        if(g!=1)ans|=(!fun(a,b/g,!player));
      }
      else{
           ans|=(!fun(a-1,b,!player));
           if(g!=1)ans|=(!fun(a/g,b,!player));
      }

     return rv[a][b][player]=ans;
}
