#include <bits/stdc++.h>
#define MOD (1000000007LL)
#define pb push_back
#define ff first
#define ss second
#define LEFT(n) (2*n)
#define RIGHT(n) (2*n+1)
using namespace std;

typedef long long ll;
typedef pair<int, int> ii;


int a, h;
set<string> done;

string add(string str, int a){
    int n = str.length();
    for(int i=1;i<n;i+=2)
        str[i] = '0' + ((str[i]-'0')+a)%10;
    return str;
}

string rotate(string str, int h){
    int n = str.length();
    string temp = str + str;
    return temp.substr(n-h, n);
}

void dfs(string str){
    if(done.find(str) != done.end())    return;
    done.insert(str);
    dfs(rotate(str, h));
    dfs(add(str, a));
}


int main()
{
    ios_base::sync_with_stdio(0);
    // freopen("input.txt", "r", stdin);

    int t;
    cin>>t;
// t=1;
    while(t--){

        string str;
        cin>>str;
        cin>>a>>h;

        int n = str.length();
        h %= n;
        a %= 10;

        done.clear();
        dfs(str);
        string ans = str;
        for(set<string>::iterator it=done.begin();it!=done.end();it++)
            ans = min(ans, (*it));
        cout<<ans<<endl;
    }

    return 0;
}
