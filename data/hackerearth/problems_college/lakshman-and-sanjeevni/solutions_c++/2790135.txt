#define REP(i,a,b) for(int i=a;i<=b;i++)
#define N 10000005

bool t[10000005];
int dp[10000005],minp[10000005];

int main(){
	int n;
	ll ans=0,k;
	int temp;
	scni(n);

	for(int i=2;i*i <= N;i++){
		if(minp[i] == 0){
			for(int j = i*i;j<= N ;j+=i)
				if(minp[j] == 0) minp[j] = i;
		}
	}

	for(int i=2;i<= N;i++) if(minp[i] == 0)minp[i] = i; 

	rep(i,n){
		scnl(k);
		temp = 0;
		int prev = 1;
		while(k!=1){
			if(minp[k] != prev) temp++;
			prev = minp[k];
			k /= minp[k];

		}

		ans += (ll)pow(2,temp);
	}

	printf("%lld\n",ans );


	return 0;
}
