#include <stdio.h>
long long a[100005];
int main()
{
    long n,q,i,l,r,x;
    scanf("%ld %ld",&n,&q);
    a[0]=0;
    for(i=1;i<=n;i++)
    {
    	scanf("%ld",&x);
    	a[i]=a[i-1]+x;
    }
    for(i=0;i<q;i++)
    {
    	scanf("%ld %ld",&l,&r);
    	printf("%lld\n",a[r]-a[l-1]);
    }
    return 0;
}
