#include <iostream>
using namespace std ; 
#include <math.h>
int main()
{
	
	long long int n ; 
	cin >> n ; 
	long long int i ; 
	long long int j ; 
	long long int a[100][100] ; 
	for ( i = 0; i< n ; i++)
	{
		for ( j = 0 ; j <n ; j++)
		{
		 cin >> a[i][j] ; 
		}
	}
	long long int sum1= 0 ; 
	for ( i = 0; i< n ; i++)
	{
		for ( j = 0; j< n ; j++)
		{
			if ( i == j )
			{
				sum1= sum1 + a[i][j] ; 
			}
		}
	}
	long long int sum2 = 0 ;
	for ( i = 0 ; i< n ; i++)
	{
		for ( j = 0; j < n ; j++)
		{
			if ( i +j  == n-1 )
			{  
			 //  cout << i <<  " " << j <<endl; 
			//	cout << "sanjeev " <<endl; 
				sum2 = sum2 + a[i][j] ; 
			}
		}
	}
	//cout << sum1 <<  " " << cout << sum2 <<endl; 
	long long int r = abs( sum1- sum2) ;
	cout << r <<endl; 
	return 0 ; 
}
