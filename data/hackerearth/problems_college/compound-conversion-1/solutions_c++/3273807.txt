int main()
{
	int ts;
	si(ts);
	FOR(t,0,ts)
    {
        string a,b;
        cin>>a>>b;
        
        int ans=0;
        
        if(a.length()!=b.length())
        {
            printf("-1\n");
            continue;
        }
        
        FOR(i,0,a.length())
        {
            if(a[i]=='1'&&b[i]=='0')
            {
                int f=0;
                FOR(j,i+1,a.length())
                {
                    if(a[j]=='0'&&b[j]=='1')
                    {
                        char c;
                        c = a[j];
                        a[j] = a[i];
                        a[i] = c;
                        f=1;
                        ans++;
                        break;
                    }
                }
                if(f==0)
                {
                    FOR(j,0,a.length())
                    {
                        if(a[j]=='?'&&b[j]=='1')
                        {
                            char c;
                            c = a[j];
                            a[j] = a[i];
                            a[i] = c;
                            f=1;
                            ans++;
                            break;
                        }
                    }
                }
            }
            else if(a[i]=='0'&&b[i]=='1')
            {
                int f=0;
                FOR(j,i+1,a.length())
                {
                    if(a[j]=='1'&&b[j]=='0')
                    {
                        char c;
                        c = a[j];
                        a[j] = a[i];
                        a[i] = c;
                        f=1;
                        ans++;
                        break;
                    }
                }
                if(f==0)
                {
                    FOR(j,0,a.length())
                    {
                        if(a[j]=='?'&&b[j]=='0')
                        {
                            char c;
                            c = a[j];
                            a[j] = a[i];
                            a[i] = c;
                            f=1;
                            ans++;
                            break;
                        }
                    }
                }
                if(f==0)
                {
                    a[i] = '1';
                    ans++;
                }
            }
        }
        // cout<<a<<endl<<b<<endl;
        FOR(i,0,a.length())
            if(a[i]=='?'){
                a[i]=b[i];
                ans++;
            }
            
       // cout<<a<<endl<<b<<endl;

    }

}
