#include <iostream>
#include<math.h>
using namespace std;
 
int array[1000001];
 
int main()
 
{
 
int t,n,f;
 
array[1]=1;
 
for(long long int i=2;i<1000001;i++)
 
 {
    if(array[i]!=0)
        continue;
    for(long long int j=i*i;j<1000001;j*=i)
        array[j]=1;
 }
scanf("%d",&t);
 while(t--)
   {
    scanf("%d",&n);
    f=1;
    for(int i=1;i<=n/2;i++)
     {
        if(array[n-i] && array[i])
         { 
            f=0;
         printf("Yes\n");
         break;
         }
     }
        if(f) 
        printf("No\n"); 
   }
}
