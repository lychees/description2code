int LuckyString(long int num){
    long int sum=0,step=0,prevSum=0,actual=0,stringNum=0,charNum=0,quotient=0,remainder=0;
    while(num >= sum){
              if(num==sum){
                           return 1;
                           }
              prevSum = sum;
              sum = sum + (pow((long double)2.0,(long double)step)*step);
              step++;
              }
    step--;
    //cout<<"prevSum = "<<prevSum<<endl;
    //cout<<"Sum = "<<sum<<endl;
    //cout<<"Step = "<<step<<endl;
              
    actual = num-prevSum;
    
    //cout<<"Actual = "<<actual<<endl;
    
    if(actual%step==0){
         stringNum = actual/step;
                       }
    else{
         stringNum = (actual/step)+1;
         }
    
    //cout<<"stringNum = "<<stringNum<<endl;

    if(actual%step==0){
         charNum = 0;
                       }
    else{
         charNum = actual%step;
         charNum = step-charNum;
         }    
    
    //cout<<"charNum = "<<charNum<<endl;
    
    quotient = stringNum/pow((long double)2.0,(long double)charNum);
    remainder = stringNum%(long int)(pow((long double)2.0,(long double)charNum));
    
    //cout<<"quotient = "<<quotient<<endl;
    //cout<<"remainder = "<<remainder<<endl;
    
    if(quotient%2==0 && remainder==0){
                     return 1;
                     }
    else if(quotient%2!=0 && remainder>0){
                     return 1;
                     }
    else if(quotient%2==0 && remainder>0){
                     return 0;
                     }
    else if(quotient%2!=0 && remainder==0){
                     return 0;
                     }
    }
