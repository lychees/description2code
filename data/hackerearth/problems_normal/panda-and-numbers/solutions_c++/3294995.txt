#include <bits/stdc++.h>
#define mp make_pair
#define pb push_back
#define ll long long
#define debug(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define fr(i,beg,end) for(i=beg;i<end;i++)
#define itfr(it,stl) for(it=stl.begin();it!=stl.end();it++)
#define PII pair<int,int>
#define init(x,val) memset(x,val,sizeof(x))
#define fst first
#define snd second
using namespace std;
bool dp[1000001];
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    dp[1]=true;
    dp[4]=true;
    dp[27]=true;
    dp[256]=true;
    dp[3125]=true;
    dp[46656]=true;
    dp[823543]=true;
    int i,sq[10];
    sq[1]=1;
    sq[2]=4;
    sq[3]=9;
    sq[4]=16;
    sq[5]=25;
    sq[6]=36;
    sq[7]=49;
    sq[8]=64;
    sq[9]=81;
    fr(i,11,1000001)
    {
        if(dp[i])
            continue;
        int num=i;
        while(num)
        {
            if((num%10)&&i>=sq[num%10]&&dp[i-sq[num%10]])
            {
                dp[i]=true;
                break;
            }
            num/=10;
        }
    }
    int t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        if(dp[n])
            cout<<"Yes"<<endl;
        else
            cout<<"No"<<endl;
    }
    return 0;
}
