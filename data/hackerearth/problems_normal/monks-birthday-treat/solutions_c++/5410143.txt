    #include <bits/stdc++.h>
    using namespace std;
    #define ll long long
    #define PB push_back
    #define si(n) scanf("%d",&n)
    #define sl(n) scanf("%lld",&n)
    #define pi(n) printf("%d",n)
    #define pl(n) printf("%lld",n)
    vector<vector<ll> >adj;
    int n;
    int visi[1001];
    int dfs(int node)
    {
    	int i;
    	for(i=1;i<=n;i++)
    	visi[i]=0;
    	int count=0;
    	visi[node]=1;
    	stack<int>s;
    	s.push(node);
    	count++;
    	while(s.size()!=0)
    	{
    		int to=s.top();
    		s.pop();
    		for(i=0;i<adj[to].size();i++)
    		{
    			int k=adj[to][i];
    			if(visi[k]==0)
    			{
    				count++;
    				s.push(k);
    				visi[k]=1;
				}
			}
		}
		return count;
		
	}
    int main()
    {
    	
    	int d,x,y;
    	cin>>n>>d;
    	adj.resize(n+2);
    	while(d--)
    	{
    		cin>>x>>y;
    		adj[x].PB(y);
    		
		}
		int ans=dfs(1);
    	for(int j=2;j<=n;j++)
    	{
    		ans=min(dfs(j),ans);
		}
		cout<<ans<<endl;
    }
