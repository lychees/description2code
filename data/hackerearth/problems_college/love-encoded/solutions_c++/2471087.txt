#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    ll t,k,i,d,c,x;
    string s;
    cin>>t;
    while(t--)
    {
        cin>>k;
        cin>>s;
        k=k%26;
        for(i=0;i<s.length();i++)
        {
            int x=s[i];
            if((x-k)>=65)
                cout<<(char)(x-k);
            else
            {
                d=x-65;
                c=k-d;
                cout<<(char)(91-c);
            }
        }
        cout<<'\n';
    }
    return 0;
}
