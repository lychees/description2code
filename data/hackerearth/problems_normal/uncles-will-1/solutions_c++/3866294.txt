#include<iostream>
#include<stdio.h>
#include<vector>
#include<string.h>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    for(int ii=0;ii<t;ii++)
    {
            int m,s,c;
            scanf("%d%d%d",&m,&s,&c);
            bool h[210]={0};
            for(int i=0;i<c;i++)
            {
                    int y;
                    cin>>y;
                    h[y]=1;
            }
            int po=-1;
            int x=0;
            for(int i=0;i<=s;i++)
            {
                    if(h[i]==1)
                    {
                          x++;
                          while(h[1+i]==1)
                            i++;     
                    }
            }
         //   cout<<" x is "<<x<<"\n";
            if(m>=x)
               cout<<c<<"\n";
            else
            {
                
                 for(int k=0;k<x-m;k++)
                 {
                    vector<int>v;
                    int pp=-1;
                    for(int i=0;i<=s;i++)
                    {
                        if(h[i]==1)
                        {
                            if(pp==-1)
                               pp=i;
                            else 
                            {
                                 v.push_back(i-pp-1);
                            }
                            while((h[i]==1)&&(i<210))
                               i++;
                            pp=i-1;
                        }   
                         
                    }
                    sort(v.begin(),v.end());
                    int d=v[0];
                    pp=-1;
                    for(int i=0;i<=s;i++)
                    {
                          if(h[i]==1)
                        {
                            if(pp==-1)
                               pp=i;
                            else 
                            {
                                 if((i-pp-1)==d)
                                 {
                                                 for(int uu=pp;uu<i;uu++)
                                                   h[uu]=1;
                                                  i=210; 
                                 }
                            }
                            while((h[i]==1)&&(i<210))
                               i++;
                            pp=i-1;
                        }   
                    }
                 }
                 int co=0;
                 for(int i=0;i<=s;i++)
                      if(h[i]==1)
                         co++;
                 cout<<co<<"\n";
                 
            }
    
    }
    return 0;
    
}
