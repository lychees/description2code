t = int(raw_input().strip())

while t:
    t = t-1
    n = int(raw_input().strip())
    a = [0]*(1000001)
    count = 0
    temp = 0
    for i in range(n):
        s, c = map(int, raw_input().strip().split())
        if c is 1 and a[s] is 0:
            # print s , c
            temp += 1
            a[s] = 1
        elif c is 0:
            count = max(count, temp)
            temp = 0
    print max(count, temp)
