#include <iostream>
#include <cstring>

using namespace std;

void show (char str[], int i) {
	while (i >= 0) {
		cout << str[i] ;
		i--;
	}
	cout << " ";
}

void rev_sen (char sen[]) {
	int j, k;
	char res[100] = "";

	k = strlen(sen);
	j = 0;           
	
	while (--k >= 0) {
		if (sen[k] == ' ' ) {              
			show (res, j - 1);
			j = 0;
		} else if ( k == 0) {
			res [j] = sen[k];
			show (res, j);
		} else {
			res[j] = sen[k];
			j++;
		}   
	}
}

int main() {
	
	char sen[100] = "";
	int t = 0;

	cin >> t;
	cin.ignore();
	
	while (t--) { 
		cin.getline (sen, 100);
		rev_sen (sen);				
		cout << endl; 	
	}     
	return 0;
}
