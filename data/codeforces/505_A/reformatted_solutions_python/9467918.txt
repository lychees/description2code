import string


def R():
    return map(float, raw_input().split())


def Rs():
    return raw_input().split()


s = raw_input()


def ver(s, ind):
    for i in string.lowercase:
        c = s[:ind]+i+s[ind:]
        if c == c[::-1]:
            return c
    return ''


ss = ''
for i in range(len(s)+1):
    ss = ver(s, i)
    if ss != '':
        break
if ss != '':
    print ss
else:
    print 'NA'
