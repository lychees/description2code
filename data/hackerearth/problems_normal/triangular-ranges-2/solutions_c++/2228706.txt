#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<algorithm>
long long int a[2000005];
using namespace std;
long long int size=0;
long long int ub(long long int ff,long long int val,int flag){
	long long int start=ff;
	long long int end=size;
	long long int m;
	while(start<end){
		m=(start+end)/2;
		if(a[m]>val){
			end=m;
		}else{
			start=m+1;
		}
	}
 
return end;
}
long long int lb(long long int ff,long long int val,int flag){
	long long int start=ff;
	long long int end=size;
	long long int m;
	while(start<end){
		m=(start+end)/2;
		if(a[m]>=val){
			end=m;
		}else{
			start=m+1;
		}
	}
	return start;
}
int main()
{
	a[0]=0;
	a[1]=1;
	for(long long int i=2;i<2000005;i++){
		a[i]=a[i-1]+i;;
		if(a[i]>=1000000000000LL){
			size=i;
			break;
		}
	}
	int t;
	cin>>t;
	while(t--){
	long long int L,R,count=0;
	cin>>L>>R;
	for(long long int i=1;i<size;i++){
		if(a[i]>R){
			break;
		}
		count=count+ub(i,R-a[i],0)-lb(i,L-a[i],1);
	
	
	}
	cout<<count<<endl;	
	
	}
	return 0;
}
