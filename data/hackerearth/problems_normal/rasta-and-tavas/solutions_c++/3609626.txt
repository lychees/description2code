#include <iostream>
#include <cmath>

using namespace std;

int sumOfPrimeFactors(int n) {

	int sum = 0;
	if(n%2 == 0) {
		sum += 2;
		while(n%2 == 0) n = n/2;
	}

	for(int i = 3; i <= sqrt(n); i += 2) {
		if(n%i == 0) {
			sum += i;
			while(n%i == 0) n = n/i;
		}
	}

	if(n > 2) {
		sum += n;
	}
	return sum;
}

int main() {

	int a, k;
	cin >> a >> k;
	int count = 0;
	for(int i = k; i <= a; ++i) {
		if(sumOfPrimeFactors(i) == k) count++;
	}
	cout << count;
}
