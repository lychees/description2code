#include<bits/stdc++.h>
using namespace std;

int main()
{
	string s;
	cin>>s;
	int t,i,l=s.size();
	cin>>t;
	set<string> st;
	while(t--)
	  {string r;
	  cin>>r;
	  if(r==s)
	    {
	     cout<<r;
	     return 0;
	    }
	    int l1=r.size();
	   if(l<l1)
	   for(i=0;i<l;i++)
	    {if(s[i]!=r[i])
	       break;
	    }
	    if(i==l)
	      st.insert(r);
	  }
	if(st.size())
	  cout<<*st.begin();
	else
	 cout<<s<<endl;
}
