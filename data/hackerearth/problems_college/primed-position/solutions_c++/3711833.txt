#include<bits/stdc++.h>
  
  using namespace std;
  long int n=10000000;
  bool prime[10000005];
  long long int result[10005]={0};
  #define ll long long
  ll mod=1000000007;
  void sieve( )
  {
  	
memset(prime,true,sizeof(prime));
prime[0]=false;
prime[1]=false;
//result[0]=0;
//result[1]=0;
long int i,j;
for(i=2;i*i<=10000000;i++)
{
	if(prime[i]==true)
	{
		for(j=i*i;j<=n;j+=i)
		prime[j]=false;
	}
}
int cnt=0,k=1;
for(i=1;i<=n;i++)
{
    //result[i]=result[i-1];
	if(prime[i]==true)
	{
		prime[i]=i;
		cnt++;
		if(prime[cnt]==true)
		{
			result[k]=i;
			k++;
		}
	}
}
//return 0;
  }
  int main()
  {
  	int t;
  	sieve();
  	scanf("%d",&t);
  	while(t--)
  	{
  		int n,m;
  		scanf("%d%d",&n,&m);
  		ll a=result[m]%mod;
  		ll b=result[n]%mod;
  		ll ans=a*b;
  		ans%=mod;
  		printf("%lld\n",ans);
	  }
	  return 0;
  }
