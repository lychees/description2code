T = input()
for i in xrange(T):
    count = 0
    a, b = map(long, raw_input().split())
    x = str(a)
    y = str(b)
    for j in xrange(len(y)):
        c = 0
        c = long(x[j])+long(y[j])
        if c >= 10:
            count += 1
            long(x[i])+1

    if count == 0:
        print "No carry operation"
    elif count == 1:
        print "1 carry operation"
    else:
        print count, "carry operations"
