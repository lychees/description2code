#include <iostream>
using namespace std;
#include <set>
int main()
{
	int t,j,i,n;
	set<string> s;
	set<string>::iterator it;
	cin >> t;
	for(j=0;j<t;j++){
		cin >> n;
		string a[n];
		for(i=0;i<n;i++){
			cin >> a[i];
			s.insert(a[i]);
		}
		
		for(it=s.begin();it!=s.end();it++){
			cout << *it << endl;
		}
		s.clear();
		
	}
    return 0;
}
