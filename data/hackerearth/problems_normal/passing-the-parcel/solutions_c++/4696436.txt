#include <iostream>
#include <stdlib.h>
using namespace std;

struct node{
	int data;
	struct node* next;
};

typedef struct node * Node;

Node newNode(int data)
{
	Node temp;
	temp = (struct node*)malloc(sizeof(struct node));
	if(temp == NULL)
	{
		cout<<"Mem alloc fail"<<endl;
		exit(0);
	}
	temp->data = data;
	temp->next = NULL;
	return temp;
}

void display(Node head)
{
	Node cur =head;
	do
	{
		cout<<(cur->data)<<" "<<endl;
		cur = cur->next;
	}while(cur!=head);
	cout<<endl;
}

int main()
{
	int i,n,part,len;
	string song;
	cin>>n;
	cin>>song;
	len = song.length();
	Node cur,prev,new_node,temp,head = newNode(1);
	head->next = head;
	
	for(i=n;i>=2;i--)
	{
		new_node= newNode(i);
		temp = head->next;
		if(temp == head)
			prev = new_node;
		head->next = new_node;
		new_node->next = temp;
	}
	cur = head;
//	display(head);
	
	// Lets begin game
	cur = head;
	part = -1;
	while( ! (cur->next == cur ))
	{
		part = (part+1)%len;
		if(song[part] == 'a')
		{
			prev =cur;
			cur = cur->next;
		}
		else
		{
			temp = cur;
			prev->next = cur->next;
			cur = prev->next;
			free(temp);
		}
		
	//	display(cur);
	}
	
	cout<<(cur->data)<<endl;
    return 0;
}
