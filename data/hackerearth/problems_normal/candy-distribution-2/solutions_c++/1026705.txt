#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	long int k,n;long long int a,sum=0;
    	cin>>n;
    	k=n;
    	if(n==1)
    	{
    		cin>>a;
    		cout<<a<<endl;
    	}
    	else
    	{
    	while(n--)
    	{
    		cin>>a;
    		sum=sum+a;
    	}
    	cout<<sum-(k-1)<<endl;
    	}
    }
    return 0;
}
