#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int x=0,y=0,count=0;
    map< pair<int, int>, int > m;
    m[make_pair(0,0)]=1;
    for(int i=0;i<s.size();i++)
    {
    	char ch=s[i];
    	if(s[i]=='L')
    	y=y-1;
    	else if(s[i]=='R')
    	y=y+1;
    	else if(s[i]=='U')
    	x=x-1;
    	else
    	x=x+1;
    	
    	if(m[make_pair(x,y)]==1)
    	count++;
    	m[make_pair(x,y)]=1;
    	
    }
    cout<<count;
    return 0;
}
