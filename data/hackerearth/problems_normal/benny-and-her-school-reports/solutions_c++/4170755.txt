void solve() {
	int n, m;
	scanf("%d%d", &n, &m);
	int xx;
	scanf("%d", &xx);
	long long x = 1LL * (n + 1) * xx;
	for (int i = 1; i <= n; i++) {
		int val;
		scanf("%d", &val);
		x -= (1LL * val);
	}
	x = max(x, 1LL);
	if (x > 1LL * m) {
		puts("Impossible");
	} else {
		printf("%d\n", (int)(x));
	}
}

	scanf("%d", &tc);
	while (tc--) {
		solve();
