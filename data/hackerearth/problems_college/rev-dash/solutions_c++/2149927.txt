#include <iostream>
using namespace std;

int main()
{
    string a;
    cin>>a;
    int len;
    len=a.length();
  	for(int i=0;i<len;i++)
  	{
  		if(i==len-1)
  			cout<<a.at(i);
  		else 
  			cout<<a.at(i)<<"-";
  	}
  	cout<<endl;
  	for(int i=len-1;i>=0;i--)
  	{
  		if(i==0)
  			cout<<a.at(i);
  		else 
  			cout<<a.at(i)<<"-";
  	}
    return 0;
}
