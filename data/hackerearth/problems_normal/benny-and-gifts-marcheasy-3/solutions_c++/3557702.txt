#include<bits/stdc++.h>
#define ll long long 
using namespace std;
typedef pair<int , int > p;
int main()
{
	char s[100008];
	gets(s + 1);
    int n = (int)strlen(s + 1) + 1;
	int i;
	ll x=0;
	ll y=0;
	ll ans=0;
	set<p> lg;

	for(i=1;i<=n;i++)
	{
		if(lg.find(make_pair(x, y))!=lg.end())
		{
			ans++;
		}
		lg.insert(make_pair(x, y));
		if(s[i]=='U')
		{
			++x;
		}
		else if(s[i]=='D')
		{
			--x;
		}
		else if(s[i]=='R')
		{
			++y;
		}
		else if(s[i]=='L')
		{
			--y;
		}
		//printf("%lld\n", ans);
	
	}
	printf("%lld\n", ans);
	return 0;

}
