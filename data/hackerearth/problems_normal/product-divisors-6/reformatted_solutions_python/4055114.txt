import math


def factorize(arr, prime_factors):
    i = 2
    num = arr
    while num > 1 and i <= int(math.sqrt(num)):
        if num % i == 0:
            if i in prime_factors:
                prime_factors[i] = prime_factors[i] + 1
            else:
                prime_factors[i] = 1
            num = num/i
        else:
            i = i+1

    if num > 1:
        if num in prime_factors:
            prime_factors[num] = prime_factors[num] + 1
        else:
            prime_factors[num] = 1

    return prime_factors


n_str = raw_input()
n = int(n_str)
arr = []
res = dict()
final_res = 1
temp_str = raw_input()
for i in range(n):
    arr.append(int(temp_str.split()[i]))

for val in arr:
    res = factorize(val, res)

for key, val in res.items():
    final_res = (final_res * (val+1)) % 1000000007

print final_res
