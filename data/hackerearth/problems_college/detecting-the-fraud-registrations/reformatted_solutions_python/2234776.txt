names = []
count = 0
false = []
n = input()
for i in range(n):
    name = raw_input()
    names.append(name)
nameSet = list(set(names))
for i in range(n-1):
    if names[i] in names[i+1:]:
        if names[i] in nameSet:
            nameSet.remove(names[i])
            false.append(names[i])
            count += 1
print count
for i in sorted(false):
    print i
