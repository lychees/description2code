#include <bits/stdc++.h>
using namespace std;

string s;
int k;
map<char, int> m;
bool used[ 26 ];

bool comp(pair<char, int> p1, pair<char, int> p2)
{
	return p1.second < p2.second;
}

int main()
{	
	cin >> s >> k;
	memset(used, false, sizeof(used));
	for(int i = 0; i < s.size(); i++) m[s[i]]++;
	
	vector< pair <char, int> > v(m.begin(), m.end());
	
	sort(v.begin(), v.end(), comp);
	
	int count = 0;
	
	for(int i = 0; i < v.size(); i++)
	{
		if(k >= v[i].second)
		{
			used[v[i].first - 'a'] = true;
			k -= v[i].second;
			count++;
		}
		else break;
	}
	
	
	
	cout << v.size() - count << endl;
	
	int i = 0;
	
	while(i < s.size())
	{
		if(!used[s[i] - 'a'])
			cout << s[i];
		i++;
	}

	return 0;
}
