#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int w;
    	int n;
    	cin>>w>>n;
    	int a[n+1],y[n+1];
    	for(int i=1;i<=n;i++)
    	{
    		cin>>a[i];
    	}
    	long long int sum=0;
    	if(a[1]==1)
    	{
    		y[1]=2;
    	}
    	else
    	{
    		y[1]=1;
    	}
    	sum+=w+y[1];
    	for(int i=2;i<=n;i++)
    	{
    		if(a[i]==1)
    		{
    			y[i]=2*i;
    			sum+=(w+y[i]-y[i-1]);
    		}
    		else
    		{
    			y[i]=y[i-1]+1;
    			sum+=(w+y[i]-y[i-1]);
    		}
    	}
    	cout<<sum<<endl;
    }
}
