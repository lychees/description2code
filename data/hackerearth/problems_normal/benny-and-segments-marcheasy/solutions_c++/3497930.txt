#include <bits/stdc++.h>
#define FOR(x,N) for(int x = 0; x < N; x++)
#define ll long long
#define ALL(a) (a).begin(), (a).end()
using namespace std;

int main(){
    ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);
    int T; cin >> T;
    while(T--){
        int N, L; cin >> N >> L;
        vector< pair<int,int> > A;
        FOR(x,N) {
            int a, b;
            cin >> a >> b;
            if(b-a <= L)A.push_back({a,b}); 
        }
        sort(ALL(A));

        int f = 0, idStart = -1, i = 0;
        while(i < A.size()){
            if(A[i].first > f){
                f = A[i].second, idStart = i;
            }else if(A[i].second > A[idStart].first + L){
                while(idStart < i && A[i].second - A[idStart].first > L) idStart++; 
            }
            f = max(A[i].second,f);            
            if(f == A[idStart].first + L) break;
            i++;
        }

        cout << (idStart >= 0 && f == A[idStart].first + L ? "Yes\n" : "No\n");
    }
}
