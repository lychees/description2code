#xor repeats pattern after every 4th iteration
def xornum(n):
    if n%4 == 0:
        return n
    elif n%4 == 1:
        return 1
    elif n%4 == 2:
        return n+1
    elif n%4 == 3:
        return 0
    
def calculate(a,b):
    xorA = xornum(a-1)
    xorB = xornum(b)
    val =xorA ^ xorB
    print val
    
queries = int(raw_input())
counter=0
while counter < queries:
    LL=[int(x) for x in raw_input().split(' ')]
    L=LL[0]
    R=LL[1]
    calculate(L,R)
    counter += 1
