#include<cstdio>
#include<algorithm>
#include<cstring>

using namespace std;

long long t,i,n,ar[1000009];

int main()
{

scanf("%lld",&t);
ar[1]=1;
ar[2]=2;
for(i=3;i<1000000;i++)
{
ar[i]=(ar[i-1]+ar[i-2]+1)%1000000007;
}

while(t--)
{

scanf("%lld",&n);
printf("%lld\n",ar[n]);


}






return 0;
}
