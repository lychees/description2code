#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    long int t,i,count;
    cin>>t;
    while(t--)
    {
    	char a[100000];
    	count=0;
    	cin>>a;
    	for(i=1;a[i]!=NULL;i++)
    	{
    		if(a[i-1]==a[i])
    		count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
