#include <iostream>
using namespace std;

int main()
{
	int t,A,B,C,k,p;
	cin>>t;
	while(t--)
	{
		cin>>A>>B>>C;
		if(A==B&&B==C)
		{
			cout<<(A+A)<<endl;
		}
		else
		{
		if(A>B)
		{
			if(A>C)
			{
				k=A;
				p=1;
			}
			else
			{
				k=C;
				p=3;
			}
			
		}
		else
		{
			if(B>C)
			{
				k=B;
				p=2;
			}
			else
			{
				k=C;
				p=3;
			}
		}
		switch(p)
		{
			case 1:
			if(B>C)
			p=B;
			else
			p=C;
			break;
			case 2:
			if(A>C)
			p=A;
			else
			p=C;
			break;
			case 3:
			if(A>B)
			p=A;
			else
			p=B;
			break;
			default:
			break;
		}
		if(k==p)
		cout<<(k+k)<<endl;
		else if(p==0)
		cout<<1<<endl;
		else
		cout<<(p+p+1)<<endl;
		}
	}
    return 0;
}
