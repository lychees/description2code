#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    
    int t;
    cin >> t;
    
    for (int i=0; i<t; ++i) {
    	int n,p;
    	cin >> n >> p;
    	int count = 0;
    	int array[n];
    	
    	for (int j=0; j<n; ++j) {
    		cin >> array[j];
    	}
    	
    	bool check = false;
    	for (int j=0; j<n; ++j) {
    		if (array[j] == p) check =true;
    		
    		if (array[j] > p) ++count;
    	}
    	if (check) cout << "YES " << count <<"\n";
    	else cout << "NO " << 0 << '\n';
    }
    return 0;
}
