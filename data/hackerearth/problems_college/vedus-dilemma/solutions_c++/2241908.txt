#include<bits/stdc++.h>

using namespace std;

int gcd(int a,int b){
	if(b==0)
	return a;
	return gcd(b,a%b);
}
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		vector<int> A(n);
		int i;
		
		for(i=0;i<n;i++)
		cin>>A[i];
		
		int p,q;
		cin>>p>>q;
		sort(A.begin(),A.end());
		vector <int> :: iterator r = A.begin();
	
		r= lower_bound(A.begin(),A.end(),p);
		if(*r!=p)
		r = r - 1;
		
		vector <int> :: iterator s = A.begin();
		s = upper_bound(A.begin(),A.end(),p);
		
		
		int c=0;
		vector <int> res;
		while(s!=A.end() && r!=A.begin()-1){
			if(abs(*s - p)< abs(*r - p)){
				if(gcd(*s,p)!=1)
				{
					res.push_back(*s);
					c++;
				}
				s++;
			}
			else{
				if(gcd(*r,p)!=1)
				{
					res.push_back(*r);
					c++;
				}
			
				r = r - 1;
			}
			if(c==q)
			break;
		}
		

		if(c!=q){
			if(s==A.end()){
				
				while(r!=A.begin() - 1 && c!=q){
				//	cout<<"hi"<<endl;
					if(gcd(*r,p)!=1){
						res.push_back(*r);
						c++;
						
					}
					r = r- 1;
				}
			}
			
			else if(r==A.begin() - 1){
				while(s!=A.end() && c!=q){
					if(gcd(*s,p)!=1){
						res.push_back(*s);
						c++;
						//r = r- 1;
					}
					s++;
				}
			}
		}
		if(c!=q){
			cout<<"-1"<<endl;
		}
		else{
			sort(res.begin(),res.end());
		vector <int> :: iterator z = res.begin();
		while(z!=res.end()){
			cout<<*z<<" ";
			z++;
		}
		cout<<endl;
	}
		}
		
	return 0;
}
