#include <iostream>

int main()
{
    int num_cases;
    std::cin >> num_cases;
    for (int c = 0; c < num_cases; c++)
    {   
        int num_entries;
        std::cin >> num_entries;

        int entry;
        for (int i = 0; i < num_entries; i++) std::cin >> entry;

        if (num_entries == 1)
            std::cout << entry << std::endl;
        else
            std::cout << 0 << std::endl;

    }

}
