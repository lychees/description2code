#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,a[100001],i,j,k,max;
    scanf("%d",&t);
    while(t--){
    	int h[11]={0};
    	scanf("%d %d",&n,&k);
    	for(i=0;i<n;i++){scanf("%d",&a[i]);h[a[i]]++;}
    	max=h[1];
    	for(i=2;i<=k;i++)if(max<h[i])max=h[i];
    	printf("%d\n",n-max);
    }
    return 0;
}
