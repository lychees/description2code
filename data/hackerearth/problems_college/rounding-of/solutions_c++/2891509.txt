#include <iostream>
using namespace std;

int main()
{
    long long notc;
    cin>>notc;
    while(notc--)
    {
    	long long n,b,mullow,mulhigh,mid;
    	cin>>n>>b;
    	mullow=b*(n/b);
    	mulhigh=mullow+b;
    	mid=mullow+(b+1)/2;
    	if(n<mid)
    	    cout<<mullow<<"\n";
    	else
    	    cout<<mulhigh<<"\n";
    }
    return 0;
}
