#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
int poww(int a, int b){
    LL ret = 1;
    while(b--){
        ret = ret*a;
        assert(ret <= 1e9);
    }
    return ret;
}

int main(){
    int T,a,b,x,y,g,ans,dist;
    cin >> T;
    while(T--){
        scanf("%d %d %d", &a, &b, &x);
        assert(1 <= x && x <= 1e9);
        if(b < 0) y = (a == 1);
        else y = poww(a, b);
        g = y%x;
        ans = y - g, dist = g;
        g = x - g;
        if(g < dist) ans = y + g;
        printf("%d\n", ans);
    }
}
