#include<bits/stdc++.h>
using namespace std;
int mark[100000];

int main() {
int n;
cin>>n;
for(int i = 1; i <= n; i++) {
string s;
cin>>s;
for(int j = 0 ; j < s.length(); j++) {
if(mark[s[j]] == i - 1) {
mark[s[j]] = i;
}
}
}
int cnt = 0;
for(int i = 0; i < 10000; i++) {
if(mark[i] == n) {
cnt++;
}
}
cout<<cnt<<"\n";
return 0;
}
