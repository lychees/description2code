using namespace std;
ll n,k,arr[MAX];
ll inf=1e14;
ll dp[MAX];
ll solve(ll idx)
{
    if(idx>=n)
        return 0;
    if(dp[idx]!=-inf)
        return dp[idx];
    ll ans=max(solve(idx+1),solve(idx+k+1)+arr[idx]);
    return dp[idx]=ans;
}
int main()
{
    int t;
    si(t);
    while(t--)
    {
        slli(n);
        slli(k);
        ll ans=0;
        ll i;
        for(i=0;i<MAX;i++)
            dp[i]=-1e14;
        for(i=0;i<n;i++)
            slli(arr[i]);
        cout<<solve(0)<<endl;
    }
    return 0;
}
