#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int r,k;
    	cin>>r>>k;
    	
    	int a[r+4];
    	
    	
    	for(int i=0; i<5; i++){
    		a[i] = 0;
    	}
    	
    	a[5] = 1;
    	
    	//for each element check precvious values and add
    	
    	for(int i=6; i<=r; i++){
    		//check remainder of i
    		int j =i;
    		a[i] = 0;
				while(j>4){    		   
    		   		a[i] += a[j%8];
    		   		j >>= 1;
				}
    	}
    	
    	int count = 0;
    	for(int i=0; i<=r; i++){
    		if(a[i]>=k){
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
