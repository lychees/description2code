#include <iostream>
#include <cmath>
#include <cstdio>
using namespace std;
int n;
int res;
int main() {
   cin >> n;
   int x;
   for(int i = 1; i <= n; ++i) {
      scanf("%d", &x);
      int cnt = 0;
      while (x % 2 == 0) {
         ++cnt;
         x /= 2;
      }
      for(int j = 3; j <= sqrt(x); j += 2)
         while (x % j == 0) {
            ++cnt;
            x /= j;
         }
      //cout << cnt << endl;
      if (x > 1) ++cnt;
      //cout << cnt << endl;
      res^= cnt;
   }
   //cout << res << endl;
   if (res) puts("ALICE");
   else puts("BOB");
}
