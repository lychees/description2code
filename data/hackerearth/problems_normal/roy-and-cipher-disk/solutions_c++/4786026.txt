#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t, c=0, cw, acw;
	string s;
	char p;
	map<char, int> m;
	for(char ch = 'a'; ch <= 'z'; ch++) {
		m[ch] = c++;
	}
	cin >> t;
	while(t--) {
		p = 'a';
		cin >> s;
		for(int i = 0; s[i]; i++) {
			if(s[i] == p) cout << "0 ";
			else if(s[i] > p) {
				cw = m[s[i]] - m[p];
				acw = cw - 26;
				if(abs(cw) <= abs(acw)) 
					cout << cw << " ";
				else cout << acw << " ";
			}
			else {
				acw =  m[s[i]] - m[p];
				cw = 26 + acw;
				if(abs(cw) <= abs(acw)) 
					cout << cw << " ";
				else cout << acw << " ";
			}
			p = s[i];
		}
		cout << "\n";
	}
}
