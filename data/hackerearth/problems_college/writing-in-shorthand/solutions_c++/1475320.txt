#include <iostream>
#include <map>
#include <string>
using namespace std;

int main() {
	int n, m;
	cin >> n >> m;
	string first, second;
	map<string, string> mymap;
	while(m--) {
		cin >> first >> second;
		if (first.length() <= second.length())
			mymap[first] = first;
		else
			mymap[first] = second;
	}
	while(n--) {
		cin >> first;
		cout << mymap[first] << " ";
	}
	return 0;
}
