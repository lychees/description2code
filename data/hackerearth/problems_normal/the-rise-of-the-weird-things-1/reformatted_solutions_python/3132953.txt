n = int(raw_input())
a = raw_input().split()
le = []
lo = []
se = 0
so = 0
for i in a:
    i = int(i)
    if (i % 2 == 0):
        se = se+i
        le.append(i)
    else:
        so = so+i
        lo.append(i)
lo.sort()
le.sort()
print ' '.join(str(x) for x in le), se, ' '.join(str(x) for x in lo), so
