#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007 
long long f[2009]={0};
void init()
{
	f[0]=f[1]=1;
	for(int i=2;i<=2005;i++)
		f[i]=(f[i-1]*i)%mod;
}
int main()
{
	init();
    int n,m;
    cin>>n>>m;
    long long a=1;
    int c[2009]={0};
   // for(int i=1;i<=n;i++)
   // for(int j=1;j<=m;j++)cout<<(i^j)<<" ";cout<<endl;
    for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
    	c[i^j]++;
   // for(int i=0;i<5;i++)cout<<c[i]<<" ";cout<<endl;
    for(int i=0;i<=2000;i++)
    	if(c[i])
    		a=(a*f[c[i]])%mod;
    cout<<a<<endl;
    return 0;
}
