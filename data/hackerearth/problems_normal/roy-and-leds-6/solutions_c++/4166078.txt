#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

{
    //cout << "Hello World!" << endl;
    int t,r,g,b;
    
    cin>>t>>r>>g>>b;
    int r1,g1,b1;
    r1=r*-1;g1=-1*g;b1=-1*b;
    vector<int> a(8,0);
    while(t--)
    {
    	if(r1>=0 && g1<0 && b1<0 )
    	++a[0];
    	if(r1<0 && g1>=0 && b1<0)
    	++a[1];
    	if(r1<0 && g1<0 && b1>=0)
    	++a[2];
    	if(r1>=0 && g1>=0 && b1<0)
    	++a[3];
    	if(r1<0 && g1>=0 && b1>=0)
    	++a[4];
    	if(r1>=0 && g1<0 && b1>=0)
    	++a[5];
    	if(r1>=0 && g1>=0 && b1>=0)
    	++a[6];
    	if(r1<0 && g1<0 && b1<0)
    	++a[7];
    	
    	if(r1==r-1)
    	r1=-1*r-1;
    	if(g1==g-1)
    	g1=-1*g-1;
    	if(b1==b-1)
    	b1=-1*b-1;
     	++r1;++g1;++b1;
    	
    	
    }
    for(int i=0;i<8;i++)
    cout<<a[i]<<" ";
    
    return 0;
}
