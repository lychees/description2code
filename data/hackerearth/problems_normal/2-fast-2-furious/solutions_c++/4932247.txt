#include <bits/stdc++.h>
using namespace std;
//7:15
int main()
{
	int n;
	cin>>n;
	int prev;
	cin>>prev;
	int maxdif1=INT_MIN;
	for(int i=1;i<n;i++)
	{
		int x;cin>>x;
		maxdif1=max(maxdif1,abs(prev-x));
		prev=x;
	}
	
	cin>>prev;
	int maxdif2=INT_MIN;
	for(int i=1;i<n;i++)
	{
		int x;cin>>x;
		maxdif2=max(maxdif2,abs(prev-x));
		prev=x;
	}
	
	if(maxdif1>maxdif2)
		cout<<"Dom"<<endl<<maxdif1;
	else if(maxdif1<maxdif2)
		cout<<"Brian"<<endl<<maxdif2;
	else
		cout<<"Tie";
	
}
