class FastScanner:
    def __init__(self):
        import sys
        self.inp = sys.stdin.read().split("\n")
        self.readAt = -1

    def nextInt(self):
        self.readAt += 1
        return int(self.inp[self.readAt])

    def nextString(self):
        self.readAt += 1
        return self.inp[self.readAt]

    def nextRange(self, n):
        self.readAt += n
        return self.inp[self.readAt-n+1:self.readAt+1]

    def skip(self, n):
        self.readAt += n


class PrintWriter:
    def __init__(self):
        self.pw = []

    def println(self, n):
        self.pw.append(n)

    def flush(self):
        print "\n".join(map(str, self.pw))
        self.pw = []


def main():
    fs = FastScanner()
    pw = PrintWriter()
    n = fs.nextInt()
    from string import ascii_lowercase as al, ascii_uppercase as au
    mapping = {i: i for i in al}
    revmap = {i: i for i in al}
    for i in xrange(n):
        a, b = fs.nextString().split()
        character1 = a.lower()
        character2 = b.lower()
        c = revmap[character1]
        d = revmap[character2]
        mapping[c] = character2
        mapping[d] = character1
        revmap[character2] = c
        revmap[character1] = d
    s = fs.nextString()
    ans = ""
    for i in s:
        p = "print"
        if ord(i) > 96:
            ans += mapping[i]
            p = mapping[i]
        elif ord(i) > 63:
            ans += mapping[i.lower()].upper()
            p = mapping[i.lower()].upper()
        else:
            ans += i
            p = i
    pw.println(ans)
    pw.flush()


if __name__ == "__main__":
    main()
