#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll check(ll n ,ll x)
{
	ll p=n/x;
	ll ans =x*(p*(p+1))/2;
	return ans;
}
int main()
{
    //cout << "Hello World!" << endl;
    ll n;
    cin>>n;
    if(n%3==0 || n%5==0 || n%15==0)
    cout<<(check(n,3)+check(n,5)-check(n,15)-n)<<endl;
    else
    cout<<(check(n,3)+check(n,5)-check(n,15))<<endl;
    return 0;
}
