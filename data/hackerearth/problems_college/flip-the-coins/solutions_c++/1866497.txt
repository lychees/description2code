#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    int q;
    int i;
    int l;
    int r;
    int c;
    int a[10000];
    int count;

    scanf("%d",&n);
    scanf("%d",&q);
    memset(a,0,sizeof(a));

    while(q--) {
        scanf("%d%d%d",&c,&l,&r);
        count = 0;
        if(c == 0) {
            for(i = l; i <= r; i++) {
                if(a[i] == 0) {
                    a[i] = 1;
                }
                else {
                    a[i] = 0;
                }
            }
        }
        if(c == 1) {
            for(i = l; i <= r; i++) {
                if(a[i] == 1) {
                    count++;
                }
            }
            printf("%d\n",count);
        }
    }
}
