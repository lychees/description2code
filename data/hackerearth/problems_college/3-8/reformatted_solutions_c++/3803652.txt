#include <cstdio>
#include <vector>
using namespace std;
vector<int> push_f(vector<int> v1, int i) {
  vector<int> v;
  v.push_back(i);
  vector<int>::iterator it;
  for (it = v1.begin(); it != v1.end(); ++it) {
    v.push_back(*it);
  }
  return v;
}
int main() {
  int i, j, t;
  scanf("%d", &t);
  char St[1001];
  for (; t; --t) {
    scanf("%s", St);
    vector<int> v1;
    vector<int>::iterator it;
    v1.push_back(St[0]);
    for (i = 1; St[i] != '\0'; ++i) {
      if (St[i] >= v1[0]) {
        v1 = push_f(v1, St[i]);
      } else {
        v1.push_back(St[i]);
      }
    }
    for (it = v1.begin(); it != v1.end(); ++it) {
      printf("%c", (*it));
    }
    printf("\n");
  }
}
