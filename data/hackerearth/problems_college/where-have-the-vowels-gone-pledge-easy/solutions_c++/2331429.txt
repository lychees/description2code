#include <iostream>
using namespace std;

int main()
{
    int N;
    int a=0, e=0, i=0, o=0, u=0;
    cin>>N;
    char C[N];
    cin>>C;
    
    for(int x=0;x<N;x++)
    {
    	
    	if(C[x]=='a')	a=1;
    	if(C[x]=='e')	e=1;
    	if(C[x]=='i')	i=1;
    	if(C[x]=='o')	o=1;
    	if(C[x]=='u')	u=1;
    }
    if(a==1 && e==1 && i==1 && o==1 && u==1)	cout<<"YES";
    else	cout<<"NO";
    return 0;
}
