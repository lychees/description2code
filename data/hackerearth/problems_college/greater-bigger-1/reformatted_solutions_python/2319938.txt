from itertools import permutations
num = int(raw_input())
for i in range(num):
    flag = "False"
    word = raw_input()
    perms = sorted([''.join(p) for p in permutations(word)])
    for wrd in perms:
        if wrd > word:
            print wrd
            flag = "True"
            break

    if flag == "False":
        print "no answer"
