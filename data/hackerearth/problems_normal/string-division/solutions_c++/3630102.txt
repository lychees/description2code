#include <iostream>
#include <sstream>
using namespace std;

int main()
{
	int N; cin >> N;
	while (N--) {
    	string s; cin >> s;
    	int n = s.size();
    	bool poss = false;
    	for (int a = 1; a + 3 <= n; ++a) {
    		string sa = s.substr(0, a);
    		for (int b = 1; a + b + 2 <= n; ++b) {
    			string sb = s.substr(a, b);
    			if (sb == sa) continue;
    			for (int c = 1; a + b + c + 1 <= n; ++c) {
    				string sc = s.substr(a + b, c);
    				if (sc == sa || sc == sb) continue;
    				string sd = s.substr(a + b + c);
    				if (sd == sa || sd == sb || sd == sc) continue;
    				poss = true;
    				goto hell;
    			}
    		}
    	}
    	hell:;
    	cout << (poss ? "YES" : "NO") << endl;
	}
    return 0;
}
