t = input()
while t>0:
	t-=1
	a = raw_input()
	d = [0]*26
	dc = [0]*26
	j = 26
	for i in range(26):
		dc[i] = j
		j-=1
#	print dc
	for i in range(len(a)):
		d[ord(a[i]) - ord('A')] += 1
#	print d
	flg = True
	for i in range(26):
		if d[i] > 0:
			if d[i] != dc[i]:
				flg = False
				break
	if flg:
		print "Yes"
	else:
		print "No"
