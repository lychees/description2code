ll t , n , l , a[1000005];

inline ll right(ll node){return 2*node+1;}

ll bit[1000005];
ll query(ll ind){
    ll ans = 0;
    while(ind > 0){
        ans = max(ans,bit[ind]);
        ind-=(ind&-ind);
    }   
    return ans;
}
void update(ll ind , ll val){
    while(ind < 1000002LL){
        bit[ind]=max(val,bit[ind]);
        ind+=(ind&-ind);
    }
}
 main(){

    //freopen("input.txt", "r", stdin);

    ll i , j , k ;

        cin >> n;

        for(i = 0 ; i < n ; i++){
            cin >> a[i];
            a[i]++;
        }
        cin >> k;
        ll res = -1;
        for(i = 0 ; i < n ; i++){
            ll temp = query(a[i]-1);
            //cout << temp << " " << a[i] - 1 << endl;
            if(temp+1>=k)
            {
                if(res==-1)
                    res=a[i];
                else
                    res=min(res,a[i]);
            }
            update(a[i],temp+1);
        }
        if(res!=-1)
        cout << res-1 << endl;
        else cout << -1 << endl;
