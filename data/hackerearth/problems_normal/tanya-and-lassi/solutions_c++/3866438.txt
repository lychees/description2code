#include <iostream>
using namespace std;

int main()
{
    
    int test;
    cin>>test;
    while(test--){
    	int l;
    	cin>>l;
    	long long price[l+1];
    	long long dp[l+1];
    	dp[0] = 0;
     	for(int i=1;i<=l;i++){
    		cin>>price[i];
    		dp[i] = 0;
    	}
    	for(int i=1;i<=l;i++){
    		for(int j=1;j<=l;j++){
    			if(j<=i){
    				long long p = dp[i-j] + price[j];
    				if(p>dp[i])
    					dp[i] = p;
    				
    			}
    			
    		}
    	}
    	cout<<dp[l]<<endl;
    }
    return 0;
}
