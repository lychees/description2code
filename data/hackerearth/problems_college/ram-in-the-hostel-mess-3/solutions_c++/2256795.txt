#include <iostream>
using namespace std;
int T, N;
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cin >> T;
    while (T--) {
        cin >> N;
        long long dp[1000005] = {1, 2, 3}; // 0, 1, 2
        for (int a = 3; a <= N; ++a)
            dp[a] = dp[a - 1] + dp[a - 2];
        cout << dp[N] * dp[N] << "\n";
    }
    return 0;
}
