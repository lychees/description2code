#include<iostream>
using namespace std;
long long dp[82],arr[81];
int main(){
long long n,m;
long long sum=0,dpsum=2;
cin>>n>>m;
for(int i=0;i<m;i++){
cin>>arr[i];
sum+=arr[i];}
dp[1]=1;
dp[2]=1;
for(int i=3;i<=n;i++){
	dp[i]=dp[i-1]+dp[i-2];
	dpsum+=dp[i];
}
printf("%lld",dpsum-sum);
}
