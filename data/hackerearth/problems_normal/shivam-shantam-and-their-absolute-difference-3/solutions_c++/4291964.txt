#include <bits/stdc++.h>
#define ll long long
#define last 400000
#define MOD 1000000007
using namespace std;

ll int dpa[2][400010],dpb[2][400010];
ll int a[210],b[210];
int main()
{
    ll int n,i,j,q,prev,now,ans=0LL,minn,maxx,tot;

    //freopen("input.txt","r",stdin);
    cin>>n;

    for(i=1;i<=n;i++)
        cin>>a[i];

    for(i=1;i<=n;i++)
        cin>>b[i];

    cin>>q;

    dpa[0][0]=1;
    dpb[0][0]=1;
    for(i=1;i<=n;i++)
    {
        prev=(i-1)%2;
        now=i%2;
        for(j=0;j<=last;j++)
        {
            dpa[now][j]=dpa[prev][j];
            dpb[now][j]=dpb[prev][j];
        }
        for(j=0;j<=last;j++)
        {

            //cout<<"got"<<dpa[now][j]<<"from previous for j"<<j<<endl;
            if( j+a[i]<=last)
            {
                dpa[now][j+a[i]]=(dpa[prev][j]+dpa[now][j+a[i]])%MOD;
            }
            if( j+b[i]<=last)
            {

                dpb[now][j+b[i]]=(dpb[prev][j]+dpb[now][j+b[i]])%MOD;
            }

            //cout<<"for j"<<j<<" "<<dpa[prev][j]<<"dpa"<<dpa[now][j+a[i]]<<" "<<dpb[now][j+b[i]]<<endl;

        }
    }
    now=n%2;

    for(j=1;j<=last;j++)
    {
        //dpa[now][j]=(dpa[now][j-1]+dpa[now][j])%MOD;
        //cout<<"number of ways coming to j  "<< j<< "is"<<dpb[now][j]<<endl;
        dpb[now][j]=(dpb[now][j-1]+dpb[now][j])%MOD;

    }

    for(j=0;j<=last;j++)
    {
        minn=j-q;
        maxx=min(j+q,400000LL);
        tot=dpb[now][maxx];
        if(minn>0)
            tot=(tot-dpb[now][minn-1]+MOD)%MOD;
        ans=(ans+(dpa[now][j]*tot)%MOD)%MOD;
    }

    cout<<ans<<endl;
}
