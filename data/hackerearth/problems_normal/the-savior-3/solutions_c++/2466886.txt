#include <iostream>
#include<vector>
#include<set>
using namespace std;
int has[101001];
int main()
{
    long long int t;
    cin>>t;
    while(t--)
    { 
    long long int n,i,tmp;
    set<long long int> z,v;
    for(i=0;i<101001;i++) has[i]=0;
    cin>>n;
    long long int sumv=0,sumz=0;
    for(i=0;i<n;i++)
    {
    	cin>>tmp;has[tmp]++;
    	if(tmp&1) {z.insert(tmp);sumz+=tmp;}
    	else       {v.insert(tmp);sumv+=tmp;}
    }
    long long int ans1=0,ans2=0;
    vector<long long int>::iterator it,it1;
    vector<long long int>v1(z.begin(),z.end());
     for(it=v1.begin();it!=v1.end();++it)
        for(it1=it+1;it1!=v1.end();++it1)
          ans1+=has[*it]*has[*it1];
    vector<long long int>v2(v.begin(),v.end());
    
    for(it=v2.begin();it!=v2.end();++it)
        for(it1=it+1;it1!=v2.end();++it1)
          ans2+=has[*it]*has[*it1];
     
    
     cout<<ans1+ans2<<endl;
    }
    return 0;
}
