#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[100010];
int main()
{
	ll int n;
	scanf("%lld",&n);
	ll int i;
	for(i=0;i<n;i++)
	scanf("%lld",(a+i));
	ll int ans=0,s=0,t=0;
	for(i=0;i<n;i++)
	{
		t+=a[i];
		if(t<0)
		t=0;
		if(t>s)
		s=t;
	}
	if(s>0)
	ans=s;
	else
	ans=0;
	printf("%lld\n",ans);
	return 0;
}
