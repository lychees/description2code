#include <bits/stdc++.h>
using namespace std;

int binarysearch(long long int arr[], int u, int val) {
	int l = 0, mid;
	while (l <= u) {
		mid = (l+u)/2;
		if (arr[mid] == val) {
			return mid;
		} else if (arr[mid] > val) {
			u = mid - 1;
		} else {
			l = mid + 1;
		}
	}
	return -1;
}

int main() {
	int n, i, j;
	scanf("%d", &n);
	long long int arr[n];
	long long int k, s, x;
	for (i = 0; i < n; i++) {
		scanf("%lld", &arr[i]);
	}
	sort(arr,arr+ n);
	scanf("%lld", &k);
	for (i = n-1; i > 0; i--) {
		s = k - arr[i];
		for (j = i-1; j > 0; j--) {
			x = s - arr[j];
			if (x == 0) 
			{
				continue;
			}
			if (binarysearch(arr, j-1, x) != -1)
			{
				printf("YES\n");
				return 0;
			}
		}
	}
	printf("NO\n");
	return 0;
}
