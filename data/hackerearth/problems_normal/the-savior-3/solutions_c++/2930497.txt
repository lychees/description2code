#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int t;
	cin>>t;
	for(int k=0;k<t;k++){
		int n;
		cin>>n;long a[n];long e[n];long p[n];int en=0,on=0;
		for(int j=0;j<n;j++){
			cin>>a[j];
			if(a[j]%2==0){e[en]=a[j];en=en+1;
			}
			else{p[on]=a[j];on=on+1;
			}
		}
		sort(e,e+en);
		sort(p,p+on);int l=1;long long int u=0;int r=1;
		while(r<en){if(e[r]==e[r-1])l=l+1;
		if(e[r]!=e[r-1]){u=u+l*(l-1)/2;l=1;
		}
		r=r+1;
		}if(e[en-1]==e[en-2])u=u+l*(l-1)/2;
		r=1;l=1;
		while(r<on){if(p[r]==p[r-1])l=l+1;if(p[r]!=p[r-1]){u=u+l*(l-1)/2;l=1;
		}r=r+1;
		}
		if(p[on-1]==p[on-2])u=u+l*(l-1)/2;
		long long int f=en*(en-1)/2;
		f=f+on*(on-1)/2;
		f=f-u;
		cout<<f<<endl;
	}
}
