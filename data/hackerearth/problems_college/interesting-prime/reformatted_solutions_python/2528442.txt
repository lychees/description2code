import math
test_cases = input()
factors = set()
for j in range(1, test_cases+1):
    n = input()
    root_n = int(math.sqrt(n))
    for i in range(1, root_n+1):
        a = n/i
        b = n % i
        if b == 0:
            factors.add(a)
            factors.add(i)
    if n == 1:
        print 1
    else:
        print len(factors)-1
    factors.clear()
