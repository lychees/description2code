#include <iostream>
#include <vector>
#include <utility>
#include <algorithm>
#include <map>
using namespace std;
vector<pair<int,int> > v1;
map<int,int> mp1;
int main()
{
    int n,q,p1,p2,p3,l,r;
    cin>>n>>q;
    while(q--)
    {
    	cin>>p1>>p2;
    	if(p1==p2)
    	{
    		cout<<"NO\n";
    		continue;
    	}
    	if(mp1[p1]>0)
    	{
    		cout<<"NO\n";
    		continue;
    	}
    	if(mp1[p2]>0)
    	{
    		cout<<"NO\n";
    		continue;
    	}
    	l=min(p1,p2);
    	r=max(p1,p2);
    	for(p3=0;p3<v1.size();p3++)
    	{
    		if(l<v1[p3].first)
    		{
    			if(r>v1[p3].first)
    			{
    				if(r<v1[p3].second)
    				{
    					cout<<"NO\n";
    					break;
    				}
    			}
    		}
    		else
    		{
    			if(l<v1[p3].second)
    			{
    				if(r>v1[p3].second)
    				{
    					cout<<"NO\n";
    					break;
    				}
    			}
    		}
    	}
    	if(p3==v1.size())
    	{
    	v1.push_back(make_pair(l,r));
    	mp1[p1]++;
    	mp1[p2]++;
    	cout<<"YES\n";
    	}
    }
    
    return 0;
}
