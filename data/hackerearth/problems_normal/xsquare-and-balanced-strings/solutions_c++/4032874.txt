#include <iostream>
#include <string>


#define MAX_N 10000

using namespace std;

//char inputString[MAX_N];
string inputString;
long letters[26];
//string letters;



int main() {
    long t;
    cin>>t;
    cin.get();
    while (t) {
        //fgets(inputString, sizeof(inputString), stdin);
        cin>>inputString;
        for (int i = 0; i < 26; ++i) {
            letters[i] = 0;
        }

        for (int i = 0; inputString[i] != '\0'; ++i) {
            letters[inputString[i] - 'a']++;
        }

        bool isBalanced = true;
        for (int i = 0; i < 26; ++i) {
            if (letters[i] & 1) {
                cout<<"-1\n";
                isBalanced = false;
                break;
            }
        }
        if (isBalanced) {
            cout<<"1\n";
        }
        t--;
    }
}
