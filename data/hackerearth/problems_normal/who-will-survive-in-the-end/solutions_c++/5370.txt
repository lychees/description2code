#include<iostream>
#include<algorithm>
using namespace std;
main()
{   int t,n,i,index;
    cin>>t;
    while(t--)
    {
         cin>>n;
         for(i=0;i<31;i++) if(n &(1<<i)) index=i;
         i=index;
         if((1<<i)==n) cout<<1<<endl;
         else
         {   n=2*(n-(1<<i))+1;
             cout<<n<<endl;
         }  
    } 
    return 0; 
   
}
