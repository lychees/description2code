#include <bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cin>>n;
    int a[n+1];
    int maxi=0;
    for(int i=0; i<n; i++) cin>>a[i], maxi=max(maxi,a[i]);
    int pos=0;
    for(int i=0; i<n; i++){
    	if(a[i]==maxi) pos=i;
    }
    vector<int> v;
    v.push_back(pos+1);
    int lol=0;
    for(int i=0; i<pos; i++){
    	if(a[i]>lol){
    		v.push_back(i+1);
    		lol=a[i];
    	}
    }
    lol=0;
    for(int i=n-1; i>pos; i--){
    	if(a[i]>lol){
    		v.push_back(i+1);
    		lol=a[i];
    	}
    }
    sort(v.begin(),v.end());
    for(int i=0; i<v.size(); i++) cout<<v[i]<<" ";
}
