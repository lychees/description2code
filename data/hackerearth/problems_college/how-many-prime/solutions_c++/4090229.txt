#include<bits/stdc++.h>
using namespace std;

void sieve(int N) {
    bool isPrime[N+1];
    for(int i = 0; i <= N;++i) {
        isPrime[i] = true;
    }
    isPrime[0] = false;
    isPrime[1] = false;
    for(int i = 2; i * i <= N; ++i) {
         if(isPrime[i] == true) {
             // Mark all the multiples of i as composite numbers
             for(int j = i * i; j <= N ;j += i)
                 isPrime[j] = false;
        }
    }
    int i;
    for( i=N/2;i>0;i--)
    {
        if(isPrime[i]==true && isPrime[N-i]==true && (i!=(N-i)))
        {
            cout<<i<<" "<<N-i<<endl;
            break;
        }
    }
    if(i==0)
    cout<<"-1 -1"<<endl;

}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        sieve(n);
    }
    return 0;
}
