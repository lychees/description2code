#include <iostream>
#include <algorithm>
#define m 1000000007
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int i;
    	long long int n[3];
    	for(i=0;i<3;i++)
    	{
    		scanf("%lld",&n[i]);
    	}
    	sort(n,n+3);
    	long long int a;
    	a = n[0];
    	a = (((a*(n[1]-1))%m)*(n[2]-2))%m;
    	printf("%lld\n",a);
    }
    return 0;
}
