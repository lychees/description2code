from math import sqrt
t = input()
nums = [int(x) for x in raw_input().split()]
d = {}
M = 10**9 + 7
for i in nums:
    s = int(sqrt(i))+2
    temp = i
    for j in range(2, s):
        while temp % j == 0:
            if j in d:
                d[j] += 1
            else:
                d[j] = 1
            temp /= j
    if temp > 1:
        if temp in d:
            d[temp] += 1
        else:
            d[temp] = 1

ans = 1
for i in d:
    ans *= (d[i]+1) % M
print ans % M
