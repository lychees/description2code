#include <iostream>
using namespace std;

int main()
{
    long long int tc, i, j, k, lens;
    string s;
    
    cin >> tc;
    
    while (tc--) {
    	cin >> s;
    	lens = s.length();
    	
    	long long int a[lens], sum;
    	
    	a[0] = (s[0] == 'H') ? 1 : 0;
    	
    	sum = a[0];
    	
    	for (i = 1; i < lens; i++) {
    		if (s[i] == 'H') {
    			a[i] = a[i - 1] + 1;
    		} else {
    			a[i] = 0;
    		}
    		sum += a[i];
    	}
    	
    	cout << sum << endl;
    	
    }
    
    return 0;
}
