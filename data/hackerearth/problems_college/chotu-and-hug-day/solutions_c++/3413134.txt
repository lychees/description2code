#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,h;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d %d",&n,&h);
    	int a[n],i;
    	for(i=0;i<n;i++)
    	scanf("%d",&a[i]);
    	int min,min1,c;
    	min=abs(h-a[0]);
    	for(i=1;i<n;i++)
    	{
    		min1=abs(h-a[i]);
    		if(min1<min)
    		min=min1;
    	}
    	printf("%d\n",min);
    }
    return 0;
}
