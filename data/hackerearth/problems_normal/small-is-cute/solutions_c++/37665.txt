bool fminstr(char * s1, char * s2) {
	
	for (int i = 0; s1[i] != '\0' && s2[i] != '\0';  ++i)
	{
		if(s1[i] > s2[i]) return true;
		if(s1[i] < s2[i]) return false;	
	}

	char t1[34],t2[34];

	strcpy(t1,s1);
	strcpy(t2,s2);
	strcat(t1,s2);
	strcat(t2,s1);

	for (int i = 0; i < strlen(t1); ++i)
	{
		if( t1[i] > t2[i] ) return true;
		if( t1[i] < t2[i] ) return false;
	}

	return false;
}


int main() {
	
	int N;
	scanf("%d", &N);

	for(int i = 0; i < N; i++) {
		scanf("%s",str[i]);
	}

	//sort 
	char temp[17];
	for (int i = 0; i < N; ++i)
	{
		int min = i;
		for (int j = i+ 1; j < N; j++) {
			if( fminstr(str[min],str[j]) ) // is first is min string..
				min = j;
		}
		strcpy(temp, str[i]);
		strcpy(str[i], str[min]);
		strcpy(str[min], temp);
	}


	for(int i = 0; i < N ; i++) {
		printf("%s",str[i]);
	}
	printf("\n");
	return 0;
}
