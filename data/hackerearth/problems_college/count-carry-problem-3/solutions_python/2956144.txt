for _ in range(int(raw_input())):
	a, b = map(int, raw_input().split())
	ans = 0
	carry = 0
	while 1:
		c, d = a % 10, b % 10
		if c + d + carry > 9:
			ans += 1
			carry += 1
		else:
			carry = 0
		a /= 10
		b /= 10
		if (a, b) == (0, 0): break
			
	print  ans if ans else "No",
	print "carry operation" if ans < 2 else "carry operations"
