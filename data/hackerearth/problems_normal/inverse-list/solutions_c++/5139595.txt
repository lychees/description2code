#include <iostream>
using namespace std;

int main()
{
    int t,flag=1;
    cin>>t;
    while(t--)
    	{
    		int n;
    		cin>>n;
    		int arr[n+1],b[n+1];
    		for(int i=1;i<=n;i++)
    			cin>>arr[i];
    		for(int j=1;j<=n;j++)
    			{
    				b[arr[j]]=j;
    			}
    		for(int i=1;i<=n;i++)
    			{
    				if(arr[i]!=b[i])
    					{
    						flag=0;
    						break;
    					}
    			}
    		if(flag==0)
    			cout<<"not inverse\n";
    		else
    			cout<<"inverse\n";
    		flag=1;
    	}
    return 0;
}
