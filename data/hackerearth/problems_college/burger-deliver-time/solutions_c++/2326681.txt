#include<bits/stdc++.h>
using namespace std;
#define lli long long int
struct burger
{
    lli a,b;
    lli total_time;
    lli pos;
};
struct myfunc
{
    bool operator()(struct burger m,struct burger n)
    {
        return (m.total_time<n.total_time);
    }
};
struct burger arr1[1000001];
lli num;
int main()
{
    lli a,b,total;
    scanf("%lld",&num);
    //vector<lli> arr1(num),arr2(num);
    for(int i=0;i<num;i++)
    {
        scanf("%lld%lld",&a,&b);
        arr1[i].a=a,arr1[i].b=b,arr1[i].total_time=a+b,arr1[i].pos=i+1;
    }
    sort(arr1,arr1+num,myfunc());
    for(int i=0;i<num;i++)
    {
        printf("%lld ",arr1[i].pos);
    }
}
