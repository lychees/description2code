#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {int key;
    char ch[100000];
    cin>>key>>ch;
    int len=strlen(ch);
    for(int i=0;i<len;i++)
    {
    if(ch[i]>=65&&ch[i]<=90)
    {
      int res=ch[i]+key;
      if(res>90)
      {
      res=res-91;
      res=res%26;
      char p=65+res;
      cout<<p;
      }
      else
      cout<<(char)res;
    }
    else if(ch[i]>=97&&ch[i]<=122)
    {
     int res=ch[i]+key;
      if(res>122)
      {
      res=res-123;
      res=res%26;
      char p=97+res;
      cout<<p;
      }
      else
      cout<<(char)res;
    }
    
    }
    cout<<endl;
    }
    return 0;
}
