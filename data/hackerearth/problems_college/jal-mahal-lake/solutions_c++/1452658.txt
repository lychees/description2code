#include<bits/stdc++.h>
using namespace std;
struct node
{
	int x;
	int y;
	int z;
};
bool cmp(struct node a, struct node b)
{
	return a.x < b.x;
}
int main()
{
    long long int t,i,n,a,b,ans;
	cin>>t;
	struct node A[100];
	while(t--)
	{
		ans=0;
		cin>>n;
		for(i=1;i<=n;i++)
		{
			cin>>a;
			A[i].x=a;
		}
		for(i=1;i<=n;i++)
		{
			cin>>b;
			A[i].y=b;
		}
		sort(A,A+(n+1),cmp);
		for(i=1;i<=n;i++)
		{
			A[i].z=(A[i].x+A[i].y);
//			cout<<A[i].z;
		}
		for(i=1;i<=n-1;i++)
		{
			ans=ans+min(i*abs(A[i].z-A[i+1].x),(n-i)*abs(A[i].z-A[i+1].x));
		}
		/*for(i=1;i<=n;i++)
		{
		cout<<A[i].x<<'\n';
		}*/
		cout<<ans<<'\n';

	}
	return 0;
}
