#include <iostream>
using namespace std;
int main()
    {
    long long int t,m,p,z,n,i,j,k,x,y,small,a[100000];
    cin>>t;
    while(t--)
        {
        cin>>n>>k;
       if(k>=n-k)
       {
       	x=k;
       	y=n-k;
       }
       else{
       	x=n-k;
       	y=k;
       }
       //inside=x-3;
       small=y-1;
       j=0;
       for(i=0;i<small;i++)
       {//cout<<"enterted small loop"<<endl;
       	a[j++]=9;
       }
       a[j++]=8;
       for(m=0;m<x-y;m++)
       {
       	a[j++]=9;
       }
       for(p=0;p<small;p++)
       {
       	a[j++]=0;
       }
       a[j++]=1;
       for(z=0;z<n;z++)
       {
       	cout<<a[z];
       }
       cout<<endl;
    }
    return 0;
}
