#include <iostream>
#include <set>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	string str;
    	//vector<string>v;
    	set<string> s;
    	set<string>::iterator it;
    	
    	for(int i=0;i<n;i++)
    	{
    		cin>>str;
    		s.insert(str);
    	}
    	for( it=s.begin();it!=s.end();it++)
    	{
    		cout<<*it<<endl;
    	}
    	
    	
    }
    return 0;
}
