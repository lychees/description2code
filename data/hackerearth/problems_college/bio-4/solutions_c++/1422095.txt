#include <iostream>
#include<string>
typedef long long LL;
using namespace std;

int main()
{
	LL t;
	cin>>t;
	while(t--)
	{
		string a,b;
		cin>>a;
		cin>>b;
		LL la=a.length();
		LL lb=b.length();
		LL ans=2000,temp;
		for(LL i=0;i<la-lb+1;i++)
		{
			temp=0;
			LL k=i;
			for(LL j=0;j<lb;j++)
			{
				if(a[k]!=b[j])
					temp++;
					k++;
			}
			if(temp<ans)
				ans=temp;
		}	
		cout<<ans<<endl;
		
	}
    
    return 0;
}
