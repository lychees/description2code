li sgn(double x)
{
	if(x < 0) return -1;
	if(x == 0) return 0;
	if(x > 0) return 1;
}

li powr(li i, li j)
{
	if(j == 0) return 1;
	li temp = powr(i, j / 2);
	if(j % 2 == 0) return temp * temp;
	return temp * temp * i;
}

li rec(li n, li x)
{
	li t1 = (n / ((2 * n) - (2 * x) + 1));
	if(t1 == 0) t1 = 0;
	else t1 = (log2(t1));
	li temp = powr(2, t1 + 1) - sgn((n / 2) / x);
	//cout << temp << " " << t1 << "\n";
	li ans = (2 * n) + 1 - (((2 * n) - (2 * x) + 1) * temp);
	return ans;
}
