using namespace std;
ll arr[17]={0};
int main()
{
	ll t,i,j,k,n,m,a2,a3,a5,a7,x,y;
	sl(n);
	for(i=0;i<n;i++)
	{
		a2=a3=a5=a7=0;
		m=0;
		sl(j);
		while(j!=0)
		{
			k=j%10;
			if(k==2)
				a2=1;
			else if(k == 3)
				a3=1;
			else if(k==5)
				a5=1;
			else if(k==7)
				a7=1;

			j/=10;
		}
		if(a2==1)
			m+=1;
		if(a3==1)
			m+=2;
		if(a5==1)
			m+=4;
		if(a7==1)
			m+=8;
		for(x=1;x<=15;x++)
		{
			if( (m&x) == x)
				arr[x]++;
		}
	}
	ll ans=0;
	for(i=1;i<=15;i++)
	{
//		if(arr[i]>0)
//			printf("Type %lld : %lld\n",i,arr[i]);
		k=i;
		m=0;
		while(k!=0)
		{
			if((k&1)==1)
				m++;
			k>>=1;
		}
		if(arr[i]<3)
			continue;
		k=arr[i];
		if(m%2==1)
		{
			ans+= (k*(k-1)*(k-2))/6;
		//	printf("After %lld : %lld\n",i,ans);
		}
		else
			ans-= (k*(k-1)*(k-2))/6;
	}
	pln(ans);
}
