#include<stdlib.h>
#include<iostream>
#include<vector>


using namespace std;

int visited[10009],t,n,f,s,count,j,i,a,b,k;
  
  vector<int> friends[10009];

void dfs(int f)
{
  if(visited[f]==0)
  {
    visited[f]=1;
   for(int k=0;k<friends[f].size();k++)
      dfs(friends[f][k]);
  }
    else
      return;
  
}

int main()
{
  
  cin>>t;
  
  while(t--)
  {
	cin>>n>>f>>s;

	for(i=0;i<=10009;i++)
	{
	  visited[i]=0;
	  friends[i].clear();
	}
	
	for(i=0;i<f;i++)
	{
	    cin>>a>>b;
	    friends[a].push_back(b);
	}
	
	count=0;
	
	for(i=0;i<s;i++)
	{
	    cin>>f;
	    dfs(f);
	}
	
	for(i=1;i<=n;i++)
	{
		if(visited[i]==1)
		count++;
	}
	
	cout<<count<<endl;
	
    
  }
  
  return 0;
}
