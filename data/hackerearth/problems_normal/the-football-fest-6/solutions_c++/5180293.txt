#include <iostream>
using namespace std;
int top=-1;
int stack[1000005];
void push(int stack[] , int x){
     top++;
     stack[top]=x;
}

void pop(int stack[])
{
    if(top > 0)
    {
        int temp=0;
        temp = stack[top];
        stack[top] = stack[top-1];
        stack[top - 1] = temp;
 
    }
}


int main()
{
    int n,t_case,p,to_player;
    char choice;
   
    cin>>t_case;
    while(t_case--){
    	top=-1;
    	cin>>n>>p;
        push(stack,p);
    	while(n--){
            cin>>choice;
            if(choice=='P'){
                cin>>to_player;
                push(stack,to_player);
            }
        else {
               pop(stack);
             }
        }
        cout<<"Player "<<stack[top]<<endl;
     
    }
    return 0;
}
