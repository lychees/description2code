#include<bits/stdc++.h>
using namespace std;
int fun(pair<long int ,int>a,pair<long int,int>b)
{
	return (a.first==b.first?a.second<b.second:a.first<b.first);
}
int main()
{
    int n;
    cin>>n;
    vector<pair<long int,int>>v;
    vector<pair<long int,int>>::iterator it;
    int inp1,inp2;
    long int inp3;
    int i=1;
    while(n--)
    {
    	cin>>inp1>>inp2;
    	inp3=inp1+inp2;
    	v.push_back(make_pair(inp3,i));
    	i++;
    }
    sort(v.begin(),v.end(),fun);
    for(it=v.begin();it!=v.end();++it)
    {
    	cout<<it->second<<" ";
    }
    
}
