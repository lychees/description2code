#include <iostream>
using namespace std;
int days(int x)
{
	int z = 1;
	int count = 0;
	while (count<x)
	{
		count+=z;
		z++;
	}
	return z-1;
}

int main()
{
	int date,t,n;
	cin >> t;
	while (t)
	{
		t--;
		cin>>n;
		date = days(n);
		if (date%3 == 0)
			cout<< "SL" << endl;
		else if (date%3 == 1)
			cout<< "LB" << endl;
		else
			cout<< "BS" << endl;
	}
}
