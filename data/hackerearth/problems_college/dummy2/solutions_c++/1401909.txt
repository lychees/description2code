#include <iostream>
#include <string>
using namespace std;
int main()
{
	int t, len, count;
	string x;
	string decompressed, temp;
	cin>>t;
	while(t--){
		cin>>x;
		len = x.length();
		for(int i=0; i<len; i++){
			if(x[i]>='0' && x[i]<='9'){
				count = x[i]-'1';
				temp = decompressed;
				while(count--){
					decompressed.append(temp);
				}
			}
			else
				decompressed +=x[i];
		}
		cout<<decompressed<<endl;
		decompressed.clear();
	}
	return 0;
}
