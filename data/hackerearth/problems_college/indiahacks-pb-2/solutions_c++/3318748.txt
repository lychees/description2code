#include<bits/stdc++.h>
using namespace std;

int palindrom(string s1, string s2)
{
	int a,b,i,k=0,flag=0;
	a = s1.length();
	b = s2.length();
	if(a!=b)
	{
		return 0;
	}
	else
	{
		for(i=a-1;i>=0;i--)
		{
			if(s1.at(i)==s2.at(k))
			{
				k++;
			}
			else
			{
				flag=1;
				break;
			}
		}
		if(flag==1)
		{
			return 0;	
		}
		else
		{
			return 1;
		}
	}
}


int main()
{
	int n;
	cin>>n;
	string str[n];
	int i,j,k;
	for(i=0;i<n;i++)
	{
		cin>>str[i];
		
	}
	for(i=0;i<n-1;i++)
	{
		for(j=i+1;j<n;j++)
		{
			k=palindrom(str[i],str[j]);
			if(k==1)
			{
				break;
			}		
		}
		if(k)
		{
			cout<<str[i].length()<<" "<<str[i].at(str[i].length()/2)<<endl;
		}
	}
	
}
