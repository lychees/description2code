ll cal(ll x){
    if(x==0)return 0;
    int a[20],m=0,i,j;
    while(x)b[m++]=x%10,x/=10;
    ll ans=0;
    if(m>1)ans=9;

    if(m==1){
        ans+=b[0];return ans;
    }
   // DBG(ans)
    int flag=0,c=(m-1)%2;
    for(i=m-1;i>=0;i--){
        for(j=i==m-1?1:0;j<b[i];j++){
            if(i==m-1)ans+=dp[i+1][j][1]+dp[i+1][j][0];
            else if(i==m-2){
                if(j<b[i+1])ans+=dp[i+1][j][1];
                else if(j>b[i+1])ans+=dp[i+1][j][0];
            }
            else if(flag==1){
                if(i%2==c&&j>b[i+1])ans+=dp[i+1][j][0];
                if(i%2==1-c&&j<b[i+1])ans+=dp[i+1][j][1];
            }
            else{
                if(i%2==1-c&&j>b[i+1])ans+=dp[i+1][j][0];
                if(i%2==c&&j<b[i+1])ans+=dp[i+1][j][1];
            }
        } 
        if(i==m-1)continue;
        if(i==m-2){
            if(b[i]==b[i+1])return ans;
            if(b[i]<b[i+1])flag=1;
            else flag=2;
        }
        else{
           // DBG(flag)
            if(flag==1){
                if(i%2==c&&b[i]<=b[i+1])return ans;
                if(i%2==1-c&&b[i]>=b[i+1])return ans;
            }
            else{
                if(i%2==1-c&&b[i]<=b[i+1])return ans;
                if(i%2==c&&b[i]>=b[i+1])return ans;
            }
        }
    }
   // DBG(i)
    return ans+1;
}
