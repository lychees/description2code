#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s;
        cin>>s;
        int flag=0;
        long long int i,product=1;
        for(i=0;i<s.size();i++)
        {
            if((s.at(i))!=(s.at(s.size()-1-i)))
            {
                flag=1;
                break;
        }
                }
                if(flag==0)
                {
                    cout<<"Palindrome"<<endl;
                }

            else

            {
            for(i=0;i<s.size();i++)
            {
                product*=((long long int)(s.at(i))-96);
            }
            cout<<product<<endl;
        }

    }
    return 0;
}
