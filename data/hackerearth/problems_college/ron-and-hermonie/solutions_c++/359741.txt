#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>

using namespace std;

#define gc getchar

long long int read() {
 long long int ret=0;
 char c=gc();
 while(c<'0'||c>'9') c=gc();
 while(c>='0'&&c<='9') {
   ret=10*ret+c-48;
   c=gc();
 }
 return ret;
}

void solve(int n,long long int e)
{
	
	int ans,m;
	if(n==0||n==1||n==5||n==6)
		ans=n;
	else if(n==2)
	{
		m=e&3;
		if(m==0)
			ans=6;
		else if(m==1)
			ans=2;
		else if(m==2)
			ans=4;
		else
			ans=8;
	}
	else if(n==3)
	{
		m=e&3;
		if(m==0)
			ans=1;
		else if(m==1)
			ans=3;
		else if(m==2)
			ans=9;
		else
			ans=7;
	}
	else if(n==4)
	{
		m=e&1;
		if(m==0)
			ans=6;
		else
			ans=4;
	}
	else if(n==7)
	{
		m=e&3;
		if(m==0)
			ans=1;
		else if(m==1)
			ans=7;
		else if(m==2)
			ans=9;
		else
			ans=3;
	}
	else if(n==8)
	{
		m=e&3;
		if(m==0)
			ans=6;
		else if(m==1)
			ans=8;
		else if(m==2)
			ans=4;
		else

	}
	else if(n==9)
	{
		m=e&1;
		if(m==0)
			ans=1;
		else
			ans=9;
	}
	printf("%d\n",ans);
}

int main()
{
    int l1,m;

    t=read();
    while(t--)
    {
    	b=read();
