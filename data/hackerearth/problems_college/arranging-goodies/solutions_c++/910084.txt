struct compare
{
    int M;
    compare(int M = 0) : M(M) {}
    bool operator()(int a, int b) const{
        return abs(M + 1 - 2 * a) < abs(M + 1 - 2 * b);
    }
};


int main(){

		int N, M;
		cin >> N >> M;
		vector<int> order(M);
		for (int i = 0; i < M; i++)
		order[i] = i + 1;
		stable_sort(order.begin(), order.end(), compare(M));
		for (int i = 0; i < N; i++)cout << order[i % M] << '\n';
        return 0;
}
