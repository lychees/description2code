#include<bits/stdc++.h>

using namespace std;

int a[111];

int main() {
    int t,i,j,n;
    cin>>t;
    while(t--) {
        cin>>n;
        for(i = 0; i < n ; i++) {
            cin>>a[i];
        }
        int ans = 0;
        for(i = 0; i < n ; i++) {
            for(j = i+1; j < n ; j++) {
                if((a[i]+a[j])%3 == 0) {
                    ans++;
                }
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
