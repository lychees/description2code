#include<bits/stdc++.h>
using namespace std;
int mat[251][251],n,m;
int last,x;
int binary_search(int s,int e,int o1,int o2,int l){
	if(s>e)
	   return last;
	int mi=(s+e)/2;
	int y=x-mat[o1][mi]+mat[o2][mi];
	//printf("binarySearch=%d\n",y);
	if(y>=l)
	  last=mi;
	if(y>=l)
	  binary_search(s,mi-1,o1,o2,l);
	else
	  binary_search(mi+1,e,o1,o2,l);
}
int binary_search1(int s,int e,int o1,int o2,int l){
	if(s>e)
	   return last;
	int mi=(s+e)/2;
	int y=x-mat[o1][mi]+mat[o2][mi];
	//printf("binarySearch1=%d\n",y);
	if(y<=l)
	  last=mi;
	if(y>l)
	  binary_search1(s,mi-1,o1,o2,l);
	else
	  binary_search1(mi+1,e,o1,o2,l);
}
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		scanf("%d %d",&n,&m);
		int l1,r1;
		scanf("%d %d",&l1,&r1);
		for(int i=0;i<=m;i++)
		  mat[0][i]=0;
		for(int i=0;i<=n;i++)
		  mat[i][0]=0;
		for(int i=1;i<=n;i++){
		   for(int j=1;j<=m;j++){
		   	scanf("%d",&mat[i][j]);
		   	  mat[i][j]+=mat[i-1][j];
		   	  mat[i][j]+=mat[i][j-1];
		   	  mat[i][j]-=mat[i-1][j-1];
		   }
		   }
		   int ans=0;
		   for(int i=1;i<=n;i++){
		   	for(int j=1;j<=m;j++){
		   		for(int k=i;k<=n;k++){
		   			    x=0,last=-1;
		   		//	    x-=(mat[i-1][l]);
		   			    x+=(mat[i-1][j-1]);
		   			    x-=(mat[k][j-1]);
		   		//	    x+=mat[k][l];
		   		        int s=binary_search(j,m,i-1,k,l1);
		   		        if(s==-1)
		   		           s=m+1;
		   		        last=-1;
		   		        int e=binary_search1(j,m,i-1,k,r1);
		   		        if(e!=-1&&s!=m+1)
		   			    ans+=e-s+1;
		   			 //    printf("i=%d j=%d k=%d s=%d e=%d %d\n",i,j,k,s,e,ans);
		   		}
		   	}
		   }
		   printf("%d\n",ans);
	}
	return 0;
}
