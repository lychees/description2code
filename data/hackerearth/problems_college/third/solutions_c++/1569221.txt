#include <bits/stdc++.h>
#define si(n) scanf("%d",&n);
#define pi(n) printf("%d\n",n);
#define pl(n) printf("%lld\n",n);
#define sl(n) scanf("%lld",&n);
#define pd(n) printf("%lf\n",n);
#define ss(s) scanf("%s",s);
#define ps(s) printf("%s\n",s);
#define ll long long
#define mod 1000000007
#define pb push_back
#define maxn 10005
int arr[maxn];
int dp[maxn];
using namespace std;
int main()
{
    int t;
    si(t);
    while(t--)
    {
        int n;
        si(n);
        for(int i=0;i<n;i++)
            si(arr[i]);
            dp[0]=arr[0];
            dp[1]=arr[1];
            for(int i=2;i<n;i++)
            {
                dp[i]=max(dp[i-1],dp[i-2]+arr[i]);
            }
            pi(dp[n-1]);
    }

}
