#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
typedef long long ll;
int ways=100*99*98;
int a[105];
int gcd(int a,int b) {
    return b==0?a:gcd(b,a%b);
}
void preprocess() {
    for(int i=1;i<=100;i++) for(int j=1;j<=100;j++) for(int k=1;k<=100;k++) {
        if(i!=j&&j!=k&&i!=k) a[gcd(gcd(i,j),k)]++;
    }
    return;
}
int main() {
    preprocess();
    int t;
    scanf("%d",&t);
    while(t--) {
        ll ans=0;
        ll x,y;
        scanf("%lld%lld",&x,&y);
        for(int i=1;i<=100;i++) {
            if((x-y)%i==0) ans+=a[i];
        }
        printf("%.6lf\012",double(ans)/ways);
    }
    return 0;
}
