#include<bits/stdc++.h>
#define ll long long
#define pb push_back
#define mp make_pair
#define NL printf("\n")
#define mem(a,b) memset(a, b, sizeof(a))
#define ft first
#define sd second
using namespace std;
const ll mod = 1000000007;
typedef pair<int,int> pii;
int main(){
    //printf("Hello World..!\n");
    int t;
    cin>>t;
    while(t--){
        int n;
        ll k;
        cin>>n>>k;
        int a[n];
        for(int i=0;i<n;i++)
            cin>>a[i];
        int ans[n];
        ll x=n;
        for(int i=0;i<n;i++){
            ans[i]=x;
            x--;
        }
        int f[n];
        mem(f, 0);
        x=0;
        int start=0;
        for(int i=0;i<n;i++){
            x+=a[i];
            if(x>k){
                //f[i]=-1;
                while(x>k && start<=i){
                    f[i-start]-=1;
                    f[n-start]+=1;
                    x-=a[start];
                    start++;
                }
            }
        }
        for(int i=1;i<n;i++){
            f[i]=f[i]+f[i-1];
        }
        for(int i=0;i<n;i++){
            printf("%d ",max(0,ans[i]+f[i]));
        }
        printf("\n");
    }
    return 0;
}
