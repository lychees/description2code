#include<bits/stdc++.h>
using namespace std;

int compare(const void *a, const void *b)
{
	return(*(int *)a - *(int *)b);
}

int main()
{
    int n,i,j,k;
    cin>>n;
    int arr[n+1];
    for(i=0;i<n;i++)
    	cin>>arr[i];
    	
    if(n==1)
    {
    	cout<<arr[0]<<endl;
    	return 0;
    }
    	
    qsort(arr,n,sizeof(arr[0]),compare);
    int len1,len2;
    len1 = ceil(n/2);
    len2 = floor(n/2);
    int A[len1+1], B[len2+1];
    
    i = k = 0;
    j = n-1;
    while(i<j)
    {
    	A[k] = arr[i];
    	i++;
    	B[k] = arr[j];
    	j--;
    	k++;
    }
    if(i==j)
    	A[k] = arr[i];
    	
    long long diff;
    
    i = j = k = 0;
    while(i<floor(n/2))
    {
    	arr[k++] = A[i];
    	arr[k++] = B[i];
    	i++;
    }
    
    if(n%2 == 1)
    	arr[k] = A[i];
    	
    diff = 0;
    for(i=0;i<n-1;i++)
    {
    	diff = diff + abs(arr[i]-arr[i+1]);
    }
    
    diff = diff + abs(arr[0]-arr[n-1]);
    
    cout<<diff;
    
    return 0;
}
