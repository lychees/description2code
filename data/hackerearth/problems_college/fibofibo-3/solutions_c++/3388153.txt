#include <iostream>
using namespace std;
int fib(int n,int &c0,int &c1)
{
	if(n==0)
	{
		c0++;
		return 0;
	}
	if(n==1)
	{
		c1++;
        return 1;
	}
	return fib(n-1,c0,c1)+fib(n-2,c0,c1);
}

int main()
{
    int t;
    int p1,p2;
    p1=0;
    p2=0;
    cin>>t;
    int a[t];
    for(int i=0;i<t;i++)
    cin>>a[i];
    for(int i=0;i<t;i++)
    {   p1=0;
        p2=0;
    	fib(a[i],p1,p2);
    	cout<<p1<<" "<<p2<<endl;
    }
    return 0;
}
