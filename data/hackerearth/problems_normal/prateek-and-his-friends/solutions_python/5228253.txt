t = int(raw_input())
while t>0 :
	t-=1
	flag = 0	
	n, x = map(int, raw_input().split())
	value=[]
	for i in range(n):
		a= int(raw_input())
		value.append(a)
	csum = value[0]
	start = 0 
	for i in range(1,n+1):
		while csum>x and start<i-1 :
			csum -=value[start]
			start+=1

		if csum == x :
			print "YES"
			flag = 1
			break

		if i<n :
			csum += value[i]
	
	if flag==0 :
		print "NO"
