#include<cstdio>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	int hc[100001]={0},ac[100001]={0},hct[t],act[t];
	for(int i=0;i<t;i++)
	{
		scanf("%d %d",&hct[i],&act[i]);
		hc[hct[i]]++;
		ac[act[i]]++;
	}
	for(int i=0;i<t;i++)
	{
		int x=hc[act[i]];
		printf("%d %d\n",t-1+x,t-1-x);
	}
	return 0;
}
