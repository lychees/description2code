#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t,i,j,k;
    scanf ("%d",&t);
    getchar();
    while (t--)
    {
    	string a,b,s;
    	cin>>a;
    	cin>>b;
    	int l,flag,p=0;
    	l=a.size();
    	for (i=0;i<=l;++i)
    	{
    		s=a;
    		s.insert(i,b);
    		//cout<<s;
    		flag=0;
    		for (j=0;j<(s.size()/2);++j)
    			if (s[j]!=s[s.size()-1-j])
    			{
    				flag=1;
    				break;
    			}
    		if (flag==0)
    			p++;
    	}
    	printf ("%d\n",p);
    }
    return 0;
}
