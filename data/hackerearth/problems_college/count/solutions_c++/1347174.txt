#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	int r,c;
    	cin >> r >> c;
    	long long ev,od;
    	ev = od = 0;
    	if(r <= c) {
    		ev = ((r/2)*(r/2 + 1));
    		od = (r*(r+1))/2 - ev;
    	} else {
    		ev = ((c/2)*(c/2+1));
    		od = (c*(c+1))/2 - ev;
    		if(r%2 == 0) {
    			if(c%2 == 0) {
    				ev += c*((r-c)/2);
    				od += c*((r-c)/2);
    			} else {
    				od += c*((r-c)/2);
    				ev += c*(r-c-((r-c)/2));
    			}
    		} else {
    			ev += c*((r-c)/2);
    			od += c*(r-c-((r-c)/2));
    		}
    	}
    	cout << ev << " " << od << endl;
    }
    return 0;
}
