#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int t;
    
    long l, b, h;
    
    cin>>t;
    
    while(t--)
    {
    	cin>>l>>b>>h;
    	
    	long mini = min(l, min(b, h));
    	
    	if(mini % 2 == 0)
    	{
    		mini /= 2;
    		cout<<mini<<".0"<<endl;
    	}
    	else
    	{
    		mini /= 2;
    		cout<<mini<<".5"<<endl;
    	}
    	
    	
    }
    
    return 0;
}
