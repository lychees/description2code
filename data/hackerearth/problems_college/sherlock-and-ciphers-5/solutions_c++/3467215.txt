#include <iostream>
#include<stdio.h>
#include<string.h>
#include<math.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    n++;
    while(n--)
    {
    	char ch[200000];
   		gets(ch);
    	int len = strlen(ch);
    	int i;
    	i = sqrt(len);
    	int t = i,j = 0,k = 0;
    	char b[t][t];
    	for(i = 0;i<t;i++)
    	{
    		for(j = 0; j<t;j++)
    		{
    			b[j][i] = ch[k];
    			k++;
    		}
    	}
    	for(i = 0;i<t;i++)
    	{
    		for(j = 0; j<t;j++)
    		{
    			cout<<b[i][j];
    		}
    	}
    	cout<<endl;

    }
    return 0;
}
