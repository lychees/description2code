#include <cstdio>
int sv[1000005]={0};
int pr[1000005];
int main()
{
  	int t,l,r,k,n,d;
  	long long int i,j;
  	k=0;
  	pr[0]=0;
  	pr[1]=0;
  	sv[0]=1;
  	sv[1]=1;
  	for(i=2;i<=1000000;i++)
  	{
  	
  		if(sv[i]==1)
  		{
  			pr[i]=k;
  			continue;
  		}
  		k++;
		pr[i]=k;
  		for(j=i*i;j<=1000000;j+=i)
  			sv[j]=1;
  	}
  	scanf("%d",&t);
  	while(t--)
  	{
  		scanf("%d %d",&l,&r);
  		n=pr[r]-pr[l-1];
  		d=r-l+1;
//  		printf("%d %d\n",n,d);
  		printf("%.6f\n",(float)n/(float)d);
  	}
    return 0;
}
