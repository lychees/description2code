from collections import defaultdict
t = int(raw_input())
while t:
	t -= 1
	n = int(raw_input())
	ans = 0
	s = defaultdict(int)
	while n:
		n -= 1
		i,p = map(int, raw_input().split())
		s[i] += 1
		if s[p]:
			s[p] -= 1
		else:	
			ans += 1
	print ans
