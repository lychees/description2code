int main()
{
	int tstcse=getint();
	long long a=0,b=0;
	long long tmp=0;
	long long rmndr=0;

	while(tstcse>0){
		a=getint();
		b=getint();
		tmp=0;
		while(a!=1&&b!=1)
		if(a>b){
			rmndr=(a%b);
		tmp=tmp+(a-rmndr)/b;
		a=rmndr;
		}
		else{
			rmndr=(b%a);
		tmp=tmp+(b-rmndr)/a;
		b=rmndr;
		}
		
		if(a>b)
		tmp=tmp+a-b;
		else
		tmp=tmp+b-a;
		printf("%lld\n",tmp);
		
		--tstcse;
	}
    return 0;
}


long long getint(){
	char input=getchar_unlocked();
	long long value=0;
	
	while(input<'0'||input>'9'){
		input=getchar_unlocked();
