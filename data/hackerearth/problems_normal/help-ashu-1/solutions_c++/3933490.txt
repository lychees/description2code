ll arr[100000],segtree[300000];/// 2 * (closest bigger power of 2) - 1

ll maketree(ll l,ll h,ll pos)
{
    if(l==h)
    {
        segtree[pos] =arr[l]%2;    ///counting odd
        return segtree[pos];
    }
    ll mid = (l+h)/2;
    segtree[pos] = maketree(mid+1,h,2*pos +2)+ maketree(l,mid,2*pos +1) ; ///R+l=J
    return segtree[pos];

}

ll querytree(ll ql, ll qh,ll l,ll h,ll pos)
{
    if(ql<=l && qh>=h)
    {
        return segtree[pos];
    }
    else if((ql>l && ql<=h) || (qh>=l && qh<h))     ///((ql>=l && ql<=h) || (qh>=l && qh<=h))
    {
        return querytree(ql,qh,l,(l+h)/2,2*pos+1) + querytree(ql,qh,(l+h)/2 +1,h,2*pos+2);
    }
    else
    {
        return 0;
    }
}

ll updatetree(ll indx, ll n,ll val,ll l,ll h,ll pos)
{
    ll ql=indx , qh=indx;
    if(ql<=l && qh>=h)      ///l==h && l==indx for this case
    {
        segtree[pos]=val%2;
        return segtree[pos];
    }
    else if(ql>h || qh<l)
    {
        return segtree[pos];
    }
    else
    {
        segtree[pos]= updatetree(ql,n,val,l,(l+h)/2,2*pos+1) + updatetree(ql,n,val,(l+h)/2 +1,h,2*pos+2);
        return segtree[pos];
    }
}
