   #include<bits/stdc++.h>
    using namespace std;
    int main()
    {
    int t,n,i;
    cin>>t;
    unsigned long long int s=0,m,k;
    while(t--)
    
    {
    cin>>n;
    int a[n];
    for(i=0,s=0;i<n;i++)
    {
    cin>>a[i];
    s=s+a[i];
    }
    cin>>m;
    if(m==0)
    cout<<1;
    else
    {
    k=m%s;
    if(k==0)
    m=s;
    else
    m=k;
    for(i=0,s=0;i<n;i++)
    {
    s=s+a[i];
    if(s>=m)
    break;
    }
    cout<<(i+1);
    }
    cout<<endl;
    }
    return 0;
    }
