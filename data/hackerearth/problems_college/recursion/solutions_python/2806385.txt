import itertools
def fun(k):
	l=[]
	for i in range(1,k+1):
		l.append(i)
	a=[]
	b=''
	for i in itertools.permutations(l,k):
		for j in i:
			b+=str(j)
		a.append(b)
		b=''
		
	a.sort(key=lambda x:int(x))
	print(' '.join(a))
	#print(' '.join(a))
	
n=int(raw_input())
for i in range(n):
	k=int(raw_input())
	fun(k)
