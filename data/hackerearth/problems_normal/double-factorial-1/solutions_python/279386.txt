def fact(n):
	if n<=1:
		return 1
	else:
		return n*fact(n-1)



T = int(raw_input())





if T > 0:
	while(T > 0):
		s = raw_input().split(' ')
		n,m = [int(x) for x in s]
		if n>=5:
			print("0")
		else:
			f = fact(fact(n))
			print(f%(10**m))
		T -=1
