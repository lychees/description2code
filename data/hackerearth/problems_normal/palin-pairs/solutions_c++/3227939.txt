#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef vector<ll> vi;

ll hsh(string s){
  ll n = s.length();
  ll hsh = 0;
  for(ll i = 0; i < n; i++){
    hsh = hsh * 29 + (s[i] - 'a');
  }
  return hsh;
}
int main(int argc, char const *argv[]) {
  ll n;
  cin>>n;
  vi a;
  vi b;
  for(ll i = 0; i < n; i++){
    string s; cin>>s;
    a.push_back(hsh(s));
    string t = s;
    reverse(s.begin(), s.end());
    if(t != s)
    b.push_back(hsh(s));
  }
  sort(a.begin(), a.end());
  sort(b.begin(), b.end());
  long long  ans = 0;
  for(ll i = 0; i < a.size(); i++){
    ll x = upper_bound(b.begin(), b.end(), a[i]) - lower_bound(b.begin(), b.end(), a[i]);
    ans += x;
  }
  cout<<ans/2ll<<endl;
  return 0;
}
