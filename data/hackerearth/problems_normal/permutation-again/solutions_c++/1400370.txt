#include <iostream>
#include <cmath>
using namespace std;


int main()
{
   int t;
   cin >> t;
   while(t>0)
   {
   	unsigned long long int n,s;
   	
        cin >>n;
        if(n==1)
        {
        	cout << 1<<"\n";
        }
        else
        {
        	s = floor(((n*n)/2)-1);
   			cout << s<<"\n";	
        }
   	
   	t--;
   }
}
