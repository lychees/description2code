#include <bits/stdc++.h>
using namespace std;
#define ll long long int

int main()
{
	ll t,i,n,maxl,minl,a,b;
	cin >> t;
	while(t--){
		cin >> n;
		ll arr[n];
		for(i=0;i<n;i++){
			cin >> arr[i];
		}
		minl=maxl=arr[0];
		for(i=1;i<n;i++){
			a=max(max(maxl,arr[i]),max(maxl*arr[i],minl*arr[i]));
			b=min(min(minl,arr[i]),min(maxl*arr[i],minl*arr[i]));
			maxl=a;
			minl=b;
		}
		cout << maxl << " " << minl << endl;
	}
    return 0;
}
