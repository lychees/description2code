#include<iostream>
using namespace std;

#define MAX 100000
int stack[MAX];
long long int top=0;

int main(int argc,char** argv)
{
 int testcases=0;
 cin>>testcases;

 int ch=0,bubbly=0;
 string str;

 while(testcases--)
 {
    //clear the stack
    top=0;

    cin>>str;

    for(int i=0;i<str.length();i++)
    {
        //cout<<"ch= "<<ch<<endl;
        if( !(top==0) && (stack[top-1] == str[i]))
            {
                //cout<<"pop()"<<endl;
                stack[--top];
            }
        else
            {
                //cout<<"push()"<<endl;
                stack[top++]=str[i];
            }
    }

    if(top==0)
        bubbly++;

 }//end while

 cout<<bubbly<<endl;

 return 0;
}
