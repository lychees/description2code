T=int(raw_input())
for a in range(T):
    b=raw_input()
    X,P=b.split()
    X=int(X)
    P=int(P)
    ans=0
    if X%P!=0:
        print '0',X
    else:
        while X%P==0:
            r=X/P
            ans+=1
            X=r
        if X==1:
            X=0
        print ans,X
