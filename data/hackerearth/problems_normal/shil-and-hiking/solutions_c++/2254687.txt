#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	std::ios::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);
	cin>>n;
	vector<pair<long long ,long long> >myvect;
	long long ans=0;
	for(int i=0;i<n;i++) 
	{
		long long t1,t2;
		cin>>t1>>t2;
		myvect.push_back(make_pair(t1,t2));
	}
	for(int i=1;i<n;i++)
	{
    	double slope=INT_MAX;
    	for(int j=i-1;j>=0;j--)
    	{
        	double t=(double)(myvect[j].second-myvect[i].second)/(myvect[j].first-myvect[i].first);
        	if(t<=slope)
        	{
            	slope=t;
            	ans++;
        	}
    	}
	}
	cout<<ans<<endl;
}
