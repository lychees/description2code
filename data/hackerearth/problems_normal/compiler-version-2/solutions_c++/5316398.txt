#include <bits/stdc++.h>

using namespace std;

int main()
{
	string line;
	while(getline(cin,line))
	{
		
		string s;
		int found1=line.find("->");
		if(found1!=-1)
		{
			int found2=line.find("//");
			if(found2!=-1)
			{
				for(int i=0;i<found2;i++)
				{
					if(line[i]=='/'&&line[i+1]=='/')
						break;
					if(line[i]=='-'&&line[i+1]=='>')
					{
						line[i]='.';
						line.erase(i+1,1);
					}
				}
			}
			else
			{
				for(int i=0;i<line.length();i++)
				{
					if(line[i]=='/'&&line[i+1]=='/')
						break;
					
					if(line[i]=='-'&&line[i+1]=='>')
					{
						line[i]='.';
						line.erase(i+1,1);
					}
				}
			}
		}
		cout<<line<<endl;

	}

}
