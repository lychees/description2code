#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
	int T, X;
	int simple[9] = {-1, -1, 1, 1, 2, 1, 2, 1, 2};
    scanf("%d", &T);
    while(T--)
    {
    	scanf("%d", &X);
    	int ans = -1;
    	if (X > 9)
    	{
	    	int rem = X%7;
	    	if (rem == 1) {
	    		int div = (X-7)/7;
	    		rem = X - (div*7);
	    		ans = simple[rem] + div;
	    	} else {
	    		int div = X/7;
	    		ans = div;
	    		if (rem > 0)
	    			ans += simple[rem];
	    	}
    	} else {
    		ans = simple[X];
    	}
    	printf("%d\n", ans);
    }
    return 0;
}
