#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;
int compare(string s,string k)
{
	string sk=s+k;
	string ks=k+s;
	if(sk.compare(ks)>0)
	{return 0;}
	else 
	{return 1;}
}
void calculateMin(vector<string> v,long int t)
{
	long long int i=0,j=0;
	sort(v.begin(),v.end(),compare);
	i=0;
	while(i<t)
	{
		cout<<v[i];
		i++;
	}
	cout<<endl;
}
int main()
{
	long int t,i=0;
	cin>>t;
	vector<string> v;
	string s;
	
//	long long int a[t];
	while(i<t)
	{
		cin>>s;
		v.push_back(s);
		i++;
	}

}
