#include <iostream>
using namespace std;
int main()
{
	int c[361],n,k,x,sum;
	cin>>n;
	cin>>k;
	for(int i=0;i<=360;i++)
	c[i]=0;
	for(int i=0;i<n;i++)
	{
		cin>>x;
		sum=x;
        c[sum]=1;
		while(sum!=0)
		{
			sum=(sum+x)%360;
			c[sum]=1;
           // cout<<sum<<endl;
		}
	}
    for(int i=0;i<k;i++)
   {
       cin>>x;
       int flag=0;
       for(int j=0;j<=360;j++)
       {
            if( (x-j)>=0 && (x-j)<=360 && c[j] && c[x-j])
              flag=1; 
            if( (x+j)>=0 && (x+j)<=360 && c[j] && c[x+j])
             flag=1;
           if(c[x])
            flag=1; 
       }
       if(flag)
        cout<<"YES"<<endl;
      else
		cout<<"NO"<<endl;
   }

    return 0;
}
