#include <iostream>

using namespace std;

int main()
{
	int n,a[10001],i,j,k,x;
	string flag;
	cin>>n;
	for (i=0;i<n;i++) cin>>a[i];
	cin>>x;
	flag="False";
	for (i=0;i<n;i++)
	{
		for (j=i+1;j<n;j++)
		{
			for (k=i+1;k<n;k++)
			{
				if (a[i]+a[j]+a[k]==x)
				{
					flag="True";
				}
			}
		}
	}
	cout<<flag<<endl;
	return 0;
}
