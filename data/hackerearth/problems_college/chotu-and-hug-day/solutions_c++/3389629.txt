#include <iostream>
using namespace std;

int main()
{
    int t,n,h,a[1000],i,min;
    cin>>t;
    while(t--)
    {	cin>>n>>h;
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	min=abs(a[0]-h);
    	for(i=1;i<n;i++)
    	{	if(min>abs(a[i]-h))
    			min=abs(a[i]-h);
    	}
    	cout<<min<<"\n";
    }
    return 0;
}
