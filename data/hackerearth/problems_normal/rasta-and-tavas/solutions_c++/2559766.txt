#include <iostream>
#include <string.h>
#include <map>
using namespace std;
 

int main()
{
  std::ios_base::sync_with_stdio(false);
  int n,k,ans=0;
  cin >> n >> k;
  int sum[1000001]={0};
  for(int i=2;i<=n;i++){
    if(sum[i] == 0){
      for(int j=i;j<=n;j+=i){
	sum[j] += i;
      }
    }
    if(sum[i] == k)
      ans ++;
  }

  cout << ans << endl;
  
  return 0;
}
