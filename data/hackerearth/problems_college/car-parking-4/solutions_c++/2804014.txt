#include <iostream>
#include<string.h>
using namespace std;

int main()
{
                int t,i,count;
                string str;
                cin>>t;
                while(t--)
                {
                	 cin>>str;
                	int l=str.length();
                	 count=0;
                	for(int i=0;i<l;i++)
                	{
                		if(str[i]=='B')
                		{
                			if(i>=1 && str[i-1]=='-')
                		     str[i-1]='*';
                		    if(i>=2 && str[i-2]=='-') 
                		      str[i-2]='*';
                		    str[i]='*';
                		}
                	    if(str[i]=='D')
                		 str[i]='*';
                	   if(str[i]=='S')
                		 {
                		 	str[i]='*';
                		 	if(i>=1 && str[i-1]=='-')
                		 	str[i-1]='*';
                		 	if(i<l-1 && str[i+1]=='-')
                		 	str[i+1]='*';
                		 }
                	}
                	// cout<<str<<endl;
                	 for(i=0;i<l;i++)
                	 {
                	 	if(str[i]!='*')
                	 	 count++;
                	 }
                	  cout<<count<<endl;
                }
    return 0;
}
