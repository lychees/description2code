#include <iostream>
using namespace std;
bool checkpalindrome(int);
int main()
{
int n;
cin>>n;
while(n--)
{
	int a,b,count=0;
	cin>>a>>b;
 
	for(int i=a;i<=b;i++)
	{
		bool ans=checkpalindrome(i);
		if(ans==true) count++;
 
	}
	cout<<count<<endl;
}
 
return 0;
}
      
bool checkpalindrome(int i)
{
	int s[10]={0};
	int count=0;
	while(i!=0)
	{
	s[count]=i%10;
	i=i/10;
	count++;
	}
 
	int k=0,l=count-1;
	while(k<=l)
	{
		if(s[k++]!=s[l--])
			return false;
	}
	return true;
 
	
}
