#include <iostream>
#include<stdio.h>
#include<string.h>
using namespace std;

int main()
{
    int n,i,c=0,len,t=882;
    char s[100000];
    while(t--)
    {
        c=0;
    gets(s);
    len=strlen(s)-1;
    while(len>=0)
    {

        if((s[len]=='1' && len!=0) || (s[len]=='0' && len==0))
        {
            for(i=len;i>=0;i--)
            {
                if(s[i]=='0')
                {
                    s[i]='1';
                    break;
                }
                else
                    s[i]='0';
            }
            if(i==-1)
            {
                s[0]='1';
                s[len+1]='0';
                len=len+1;
            }
            c++;
        }
        else
        {
            len=len-1;c++;
        }

    }

    cout<<c-1<<endl;
}
}
