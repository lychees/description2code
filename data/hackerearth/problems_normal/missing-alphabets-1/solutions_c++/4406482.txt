const int MAXN = 1e5 + 3;
char s[51];
char cvt[150];
char rcvt[150];
int n, m;
string words[101];
void solve() {
	int T;
	cin >> T; cin.ignore();
	string s;
	while (T--) {
		getline(cin, s, '\n');
		for (char c = 'a'; c <= 'z'; ++c) {
			cvt[s[c - 'a']] = c;
			rcvt[c] = s[c - 'a'];
		}
		cin >> m; cin.ignore();
		forn(i, m) {
			getline(cin, words[i], '\n');
			forn(j, words[i].length()) {
				words[i][j] = cvt[words[i][j]];
			}
		}

		sort(words, words + m);
		forn(i, m) {
			forn(j, words[i].length()) {
				words[i][j] = rcvt[words[i][j]];
			}
			cout << words[i] << endl;
		}
	}
}
