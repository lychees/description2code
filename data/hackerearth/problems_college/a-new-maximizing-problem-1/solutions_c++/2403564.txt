#include <iostream>
#include <algorithm>
#define d 1000000007;
using namespace std;
void swap(double* a,double* b)
{
	double temp = *a;
	*a = *b;
	*b = temp;
	
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		double x[n];
		for(int i=0;i<n;i++)
		{
			cin>>x[i];
		}
		sort(x,x+n);
		double a[n];
		int k=0;
		if(n%2==0)
		{
			for(int i=0;i<n/2;i++)
			{
				a[k++]=x[i];
				a[k++]=x[i+n/2];
			}
			
		}
		else
		{
			for(int i=0;i<n/2;i++)
			{
				a[k++]=x[i];
				a[k++]=x[i+(n/2)+1];
			}
			a[k]=x[(n/2)];
		}
    	long long asum=0;
    	long long bsum=0;
    	for(int i=0;i<n;i++)
    	{
    	    asum+= a[i]*((i%2==0)?-1:1);
    		bsum+= a[i]*((a[i]>0)?1:-1);
    		asum=asum%d;
    		bsum=bsum%d;
    	}
    	//cout<<endl;
    	//cout<<endl;
    	//cout<<asum<<" "<<bsum<<endl;
    	long long csum = (asum*bsum)%d;
    	cout<<csum<<endl;
    	
    }
    return 0;
}
