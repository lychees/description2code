#include <iostream>
#include<algorithm>
using namespace std;
long long ar[100009];
int main()
{
    int n,q;
    long long l,r,low,high,mid,lans,rans;
    cin>>n;
    for(int i=1;i<=n;i++){
    	cin>>ar[i];
    }
    sort(ar,ar+n+1);
    cin>>q;
    for(int i=1;i<=q;i++){
    	cin>>l>>r;
    	low=0;
    	high=n;
    	while(low<=high){
    		mid=(low+high)/2;
    		if(ar[mid]<l){
    			low=mid+1;
    			lans=mid;
    		}
    		else{
    			high=mid-1;
    		}
    	}
    	low=0;
    	high=n;
    	while(low<=high){
    		mid=(low+high)/2;
    		if(ar[mid]<=r){
    			low=mid+1;
    			rans=mid;
    		}
    		else if(ar[mid]>r){
    			high=mid-1;
    		}
    	}
    	cout<<rans-lans<<endl;
    }
    return 0;
}
