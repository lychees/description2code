#include <bits/stdc++.h>

using namespace std;
int main()
{
		int n, num;
		scanf("%d",&n);
   		queue<int> calling, ideal;
   		for(int i=0; i<n; i++)
   		{ 
   			scanf("%d",&num);;
   			calling.push(num);
   		}
   		for(int i=0; i<n; i++)
   		{
			scanf("%d",&num);
   			ideal.push(num);
   		}
   		
   	
   		int count = 0;
   		while(!ideal.empty())
   		{
   			while(ideal.front()!=calling.front())
   			{
   				calling.push(calling.front());
   				calling.pop();
   				count++;
	   		}
   		
   			count++;
   			ideal.pop();
   			calling.pop();
   				
   		}
   		printf("%d\n",count);;	
   	
}
