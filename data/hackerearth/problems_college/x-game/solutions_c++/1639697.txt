#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
typedef unsigned long long ull;

ull power(ull x, ull y)
{
	if (y == 0)
		return 1;
	ull temp = power(x, y>>1);
	if (temp > mod)
		temp %= mod;
	if (y&1)
		return x * temp * temp;
	else
		return temp * temp;
}
int main() 
{
	ios_base::sync_with_stdio(false);
	
	int t;
	cin >> t;
	while (t--)
	{
		ull n, ans = 0;
		cin >> n;
		for (int i = 0; i < n; ++i)
		{
			ull x;
			cin >> x;
			ans |= x;
		}
		cout << (((power(2, n-1) % mod) * (ans % mod)) % mod) << "\n";
	}
	
	return 0;
}
