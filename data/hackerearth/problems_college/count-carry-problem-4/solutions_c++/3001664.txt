#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        char a[100],b[100];
        cin>>a>>b;
        int c[100],d[100],j=0;
        for(int i=strlen(a)-1;i>=0;i--)
            c[j++]=a[i]-'0';
        j=0;
        for(int i=strlen(b)-1;i>=0;i--)
            d[j++]=b[i]-'0';
        int s=0,s1=0;
        for(int i=0;i<max(strlen(b),strlen(a));i++)
        {
            if(c[i]==NULL)
                c[i]=0;
            if(d[i]==NULL)
                d[i]=0;
            if((c[i]+d[i]+s1)/10>=1)
            {
                //cout<<c[i]<<" "<<d[i]<<endl;
                s++;s1=(c[i]+d[i]+s1)/10;
            }
        }
        if(s==0)
            cout<<"No carry operation\n";
        else if(s==1)
        {
            cout<<"1 carry operation\n";
        }
        else
        {
            cout<<s<<" carry operations\n";
        }
    }
    return 0;
}
