#include<iostream>
using namespace std;

int g=0;
void  checkpossibility(int a[52][52],int n,int i,int j){

if(i==n-1&&j==n-1){
g=5;
}


if(a[i+1][j]==1||a[i][j+1]==1){
if(a[i+1][j]==1&&i<n-1)
{checkpossibility(a,n,i+1,j);}

 if(a[i][j+1]==1&&j<n-1)
{checkpossibility(a,n,i,j+1);
}


}



}



int main()
{

int arra[52][52],n,i,j,k,t;
cin>>t;
while(t--){
g=0;
cin>>n;
for(i=0;i<n;i++)
{for(j=0;j<n;j++)
cin>>arra[i][j];

}
checkpossibility(arra,n,0,0);

if(g==0)
{cout<<"NOT POSSIBLE";
    }
else
    cout<<"POSSIBLE"<<endl;

}



}
