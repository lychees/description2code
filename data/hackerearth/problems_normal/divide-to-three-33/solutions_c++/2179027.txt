#include <iostream>
using namespace std;

int main()
{
	int t , temp , sum=0 ;
	cin>>t;
	for(int i=0; i <t ; i++){
		cin>>temp;
		if(t!=3){
			sum+=temp;
		}
		else{
			if(temp>sum)
				sum=temp;
		}
	}
	if(t == 3)
		cout<<sum;
	else{
		if(sum%3==0)
			cout<<sum/3;
		else
			cout<<(sum/3)+1;
	}
    return 0;
}
