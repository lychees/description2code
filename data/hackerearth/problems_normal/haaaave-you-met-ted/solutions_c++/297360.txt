#include<iostream>
#include<cstdio>
using namespace std;
bool s[100010];
int main()
{
    int TC;
    int n,t,xorr,ans,ans1;
    scanf("%d",&TC);
    while(TC--){
        scanf("%d",&n);
        xorr=0;
        ans=36;
        if(n==1){
            printf("0\n");
            continue;
        }
        for(int i=1;i<=n;i++){
            scanf("%d",&t);
            ans1=0;
            while(t){
                if(t%2==1){
                    ans1++;
                }
                t/=2;
            }
            if(ans>ans1){
                ans=ans1;
            }
        }
        printf("%d\n",ans);
    }

    return 0;
}
