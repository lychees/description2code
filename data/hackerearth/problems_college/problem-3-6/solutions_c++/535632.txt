#include <iostream>
using namespace std;

int main()
{
	int n;
	long long home[100002],away[100002];
	int ch[100002],ca[100002];
	cin>>n;
	for(long long i =0; i<n;i++)
	{
		cin>>home[i]>>away[i];
		ch[i] = n-1;
	}
	for(long long i = 0; i <n; i++)
	{
		for(long long j =0; j<n;j++)
		{
			if(i==j)
			continue;
			if(away[i] == home[j])
				ch[i]++;
			else
				ca[i]++;
		}
	}
	for(long long i = 0;i<n;i++)
	cout<<ch[i]<<" "<<ca[i]<<endl;
	return 0;
}
