#include <iostream>
using namespace std;

int main()
{
 int T;
 cin>>T;
 while(T--){
 	int N,M;
 	cin>>M>>N;
 	if(N==0){
 		cout<<"0"<<endl;
 		continue;
 	}
 	int res=0;
 	if(N%M==0){
 		cout<<(N/M)*10<<endl;
 	}
 	if(N%M!=0){
 	int t=N/M;
 	res+=(t+1)*5;
 	int rest=N-(M-N%M);
 	if(rest%M==0)res+=(rest/M)*5;
 	else res+=((rest/M)+1)*5;
 	cout<<res<<endl;
 }
 }
    return 0;
}
