#include <iostream>
using namespace std;
typedef long int ll; 
int main()
{   int months[13] = {0,31,28,31,30,31,30,31,31,30,31,30,31};
    ll t; 
    cin >> t; 
    while(t--){
    	ll month, year;
    	cin >> month >> year; 
    	bool isLeap;
    	if(year%4==0){
    		if(year%100==0){
    			if(year%400==0){
    				isLeap = true;
    			}else{
    				isLeap = false;
    			}
    		}else{
    			isLeap = true;				
    		}
    	}else{
    		isLeap = false;
    	}
    	
    	// cout << "isLeap " << isLeap << endl; 
    	ll days = 0;
    	ll diff = year - 2000;
    	if(isLeap){
    		months[2] = 29;
    		
    	}else{
    		months[2] = 28;
    	}
    	days = diff*365 + (diff)/4 - diff/100 + diff/400;
    	for(int i =1; i<month; i++){
    		days += months[i];
    	}
    	days += ((!isLeap) && (diff) )?(1):(0);
    	days = days%7; 
    	int num = (7-days)%7 + 2;
    	if(num>7)
    	 num = 1;
    	while(num <= months[month]){
    		cout << num << " ";
    		num+=7;
    	}
    	cout << endl; 
    	
    	 
    	 
    	 
    	
    }
    return 0;
}
