#include <bits/stdc++.h>
using namespace std;

bool pali(string s,int i,int j)
{
    while(i<=j)
    {
        if(s[i]!=s[j])
        {
            return false;
        }
        i++;
        j--;
    }
    return true;
}
int main()
{
    
        string s;
        cin>>s;
        int n=s.size();
        // /int flag=0;
        	long long int sum=0;
            for(int i=0;i<n;i++)
            {
                for(int j=i;j<n;j++)
                {
                    if(pali(s,i,j))
                    {
                        sum++;
                    }
                }
            }
           cout<<sum<<endl;
    return 0;
}
