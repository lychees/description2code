#include <iostream>
#include <string>
#include <cmath>
using namespace std;
long long int res(int x, int y)
{
	if(y==0) return 1;
	long long int temp = res(x,y/2);
	if(y%2==0)
	 return temp*temp;
	else
	return x*temp*temp;
}
int main() {
	int t;
	cin >> t;
	while(t--)
	{
		long long int n;
		cin >> n;
		int pos = floor (log2(n));
		long long int l = n-res(2,pos);
		cout << 2*l+1 << endl;
	}
	return 0;
}
