#include <bits/stdc++.h>
#define c 1000000007
#define ll long long

using namespace std;

int main () {
    ll t, p, e;
    cin >> t;
    while(t--) {
        cin >> p >> e;
        double np = 0, ep = p + (e * p / 100.0);
        ll n = ep;
        while(n != 0) {
            ll r = n % 10;
            np = np * 10 + r;
            n /= 10;
        }
        ll pr = (p - np);
        double pc = pr * 100.0 / p;
        double pl = (pc * ep / 100.0);
        cout << (ll)(pl + ep) << endl;
    }
    return 0;
}
