def findxor(num):
    ret = []
    ret.append(num)
    ret.append(1)
    ret.append(num+1)
    ret.append(0)
    return ret[num % 4]


t = input()
t = int(t)
for i in range(t):
    n = raw_input()
    n = n.split()
    l = int(n[0])
    h = int(n[1])
    print findxor(h) ^ findxor(l-1)
