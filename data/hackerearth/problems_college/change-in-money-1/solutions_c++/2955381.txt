#include <bits/stdc++.h>
using namespace std;

int a[51];

bool vis[251][51];
long long memo[251][51];

long long solve(int n, int m, int pos=0)
{
	if(n==0)
		return 1;
	if(n<0 || pos==m)
		return 0;
	if(vis[n][pos])
		return memo[n][pos];
	else
	{
		long long ans=0;
		for(int i=0;i<=n;i+=a[pos])
			ans+=solve(n-i,m,pos+1);
		vis[n][pos]=1;
		return memo[n][pos]=ans;
	}
	
}

int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=0;i<m;i++)
		cin>>a[i];
	sort(a,a+m,greater<int>());
	cout<<solve(n,m);
    return 0;
}
