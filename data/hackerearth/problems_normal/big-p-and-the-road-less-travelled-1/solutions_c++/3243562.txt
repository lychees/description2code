#include<stdio.h>
#include<vector>
using namespace std;
int n;
vector<int>v[10000];
int sum(int );
int main(){
int x,y,i,j,dp[10000];
scanf("%d",&n);

scanf("%d%d",&x,&y);
while(x!=0 &&y!=0){
    v[x].push_back(y);
    scanf("%d%d",&x,&y);
}
for(i=1;i<=n;i++)
    dp[i]=0;
dp[n]=1;
for(i=n-1;i>0;i--){
    for(j=0;j<v[i].size();j++)
    if(dp[v[i][j]]!=0){
        dp[i]+=dp[v[i][j]];
    }
}

printf("%d",dp[1]);
//printf("%d",sum(1));

}
int sum(int i){
int s=0,x;
if(i==n)
    return 1;
for(x=0;x<v[i].size();x++){
    s+=sum(v[i][x]);
}
return s;

}
