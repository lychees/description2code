#include <iostream>
#include<stdio.h>
#include<stdlib.h>
using namespace std;
int cmp(const void *a,const void *b)
{
return(*(long long int *)a-*(long long int *)b);
}
int main()
{
    long long int t,n,d,a[100001],i,k;
    cin>>t;
    while(t--)
    {
    	k=0;
    	cin>>n>>d;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	qsort(a,n,sizeof(a[0]),cmp);
    	for(i=0;i<n-1;i++)
    	{
    		if(a[i+1]-a[i]<=d)
    		{
    			k++;
    			i++;
    		}
    	}
    	cout<<k<<endl;
    }
    return 0;
}
