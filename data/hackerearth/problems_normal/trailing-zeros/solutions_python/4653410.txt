def trailingZerosInFactorial(n):
    if n < 5:
        return 0
    power = 0
    divisors = []
    while True:
        divisor = 5 ** (power + 1)
        if n < divisor :
            break
        power += 1
        divisors.append(divisor)
    return reduce((lambda  x,y:x+y),map((lambda x: n/x),divisors))


N = int(raw_input())
result = trailingZerosInFactorial(N)
print result
