typedef long long int ll;
typedef pair<int, int > pii;

#define MOD 1000000007
#define pb push_back


void solve()
{
	int n;
	cin>>n;
	vector<int> a(n);
	for(int i=0;i<n;i++)
		cin>>a[i];
	ll buy = 0;
	int shares = 0;
	ll profit = 0;
	for(int i=0;i<n-1;i++) {
		if(a[i] < a[i+1]) {
			shares ++;
			buy += a[i];
		}
		else {
			profit += a[i]*shares - buy;
			shares = 0;
			buy = 0;
		}
		if(i == n-2) {
			profit += a[n-1]*shares - buy;
		}
	}
	cout<<profit<<"\n";
	
}

	while(t--)
	{
		solve();
	}
