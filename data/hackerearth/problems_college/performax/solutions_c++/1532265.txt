#include <iostream>
using namespace std;

int main()
{
	int n;
	cin >>n;
	int d;
	cin >>d;
	int i,a[n];
	for(i=0;i<n;i++){
		cin >>a[i];
	}
	int x=a[0],tima=0;
	for(i=1;i<n;i++){
		if(a[i-1]>=a[i]){
			int tim=a[i-1]-a[i];
			if(tim==0)
				tim=1;
			int temp=tim%d;
			
			tim/=d;
			if(temp){
			
				tim++;
			}
			
			tima+=tim;
		//	cout <<tim<<endl;	
			a[i]+=tim*d;
			while(a[i]<=a[i-1]){
				a[i]+=d;
				tima++;
			}
		}
		
	}
    cout << tima << endl;
    return 0;
}
