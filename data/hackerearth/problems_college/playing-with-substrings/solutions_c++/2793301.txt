 #include<bits/stdc++.h>
using namespace std;
int main()
{
	string str;
	string s;
	cin>>str;
	cin>>s;
	int n =str.length(),i,j,c=0;
	vector<string> nw;
	for(i=1;i<=n;i++)
	{
		for(j=0;j<n-i+1;j++)
		{
			string st = str.substr(j,i);
			nw.push_back(st);
			if(st.find(s) != std::string :: npos)
			{
				c++;
			}
		}
	}
	
	sort(nw.begin(),nw.end());
	
	int count =0;
	for(i=0;i<nw.size()-1;i++)
	{
		if(nw[i]==nw[i+1] && nw[i].find(s) != std::string::npos)
		{
		count++;	
		}
	}
	cout<<c-count<<"\n";
}
