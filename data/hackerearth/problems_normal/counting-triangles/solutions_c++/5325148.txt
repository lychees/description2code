#include<bits/stdc++.h>
#define gc getchar_unlocked
using namespace std;

long long int read() {
  char c = gc();
  while(c<'0' || c>'9') c = gc();
  long long int ret = 0;
  while(c>='0' && c<='9') {
    ret = 10 * ret + c - 48;
    c = gc();
  }
  return ret;
}
int main()
{
	int t;
	scanf("%d",&t);
	map<pair<long long int , pair<long long int ,long long int > > ,long long int  >mp;
	while(t--)
	{
		long long int a[3];
		for(int i=0;i<3;i++)
		a[i] = read();
		sort(a,a+3);
		mp[make_pair(a[0],make_pair(a[1],a[2]))]++;
	}
	int c=0;
	map<pair<long long  int , pair<long long int ,long long int > > ,long long int  >::iterator it;
	for(it=mp.begin();it!=mp.end();it++)
	{
		if(it->second ==1)
		{
			c++;
		}
	}
	printf("%d",c);
}
