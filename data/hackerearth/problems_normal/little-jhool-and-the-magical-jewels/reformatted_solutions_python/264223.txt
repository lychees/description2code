t = input()
while (t > 0):
    line = raw_input()
    l = len(line)
    list1 = [0]*26

    for i in range(0, l):
        list1[ord(line[i])-97] += 1

    min = list1[1]
    if list1[ord('u')-97] < min:
        min = list1[ord('u')-97]
    if list1[ord('r')-97] < min:
        min = list1[ord('r')-97]
    if list1[ord('y')-97] < min:
        min = list1[ord('y')-97]

    print(min)

    t = t-1
