n = int(raw_input())
m = int(raw_input())

boxes = [0] * n
for i in xrange(m):
	l, r = (int(x) for x in raw_input().split())
	boxes[l - 1] += 1
	if r < n:
		boxes[r] -= 1

count = [0] * (10 ** 6 + 1)
count[boxes[0]] = 1
for i in xrange(1, n):
	boxes[i] += boxes[i - 1]
	count[boxes[i]] += 1

for i in xrange(len(count) - 1, 0, - 1):
	count[i - 1] += count[i]

for i in xrange(int(raw_input())):
	x = int(raw_input())
	print count[x]
