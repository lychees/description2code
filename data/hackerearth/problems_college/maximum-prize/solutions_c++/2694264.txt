#include <iostream>
using namespace std;
#define MAX 1000001
int main()
{
    int t,n,k,i;
    int *a,*b;
    long long int sum;
    cin>>t;
    while(t--){
    	cin>>n>>k;
    	a = (int *)malloc(sizeof(int)*(n+1));
    	b = (int *)malloc(sizeof(int)*(n+1));
    	
    	int c[MAX] = {0};
    	    	
    	for(i=1;i<=n;i++){
    		cin>>a[i];
    		b[i]=0;
    	}
    	
		for(i=1;i<=n;i++){
			c[a[i]]++;
		}    	
    	
    	for(i=1;i<MAX;i++){
    		c[i]=c[i]+c[i-1];
    	}
    	
    	for(i=n;i>0;i--){
    		b[c[a[i]]] = a[i];
    		c[a[i]]--;
    	}
    		
    	sum=0;	
		if(n>=k){
	    	for(i=n;i>n-k;i--){
	    		sum+=b[i];
	    	}
		}else{
			for(i=1;i<=n;i++){
				sum+=b[i];
			}
		}
    	cout<<sum<<endl;
    }
    return 0;
}
