#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		int n, a[100001];
		bool vis[100001];
		memset(vis, 0, sizeof(vis));
		scanf("%d", &n);
		for(int i=0; i<n; i++)
			scanf("%d", &a[i]);
		int idx=0;
		bool res=true;
		while(idx<n && a[idx]!=-1)
		{
			vis[idx]=true;
			if(vis[a[idx]])
			{
				res=false;
				break;
			}
			idx=a[idx];
		}
		if(res)
			printf("Win\n");
		else
			printf("Lose\n");
	}
	return 0;
}
