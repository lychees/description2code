#include <iostream>
#include <vector>
#include <cstdio>
#include <map>
#include <string>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,i;
    	scanf("%d",&n);
    	map < int, string > Mymap;
    	for(i=0;i<n;i++)
    	{
    		int a;
    		string s;
    		scanf("%d",&a);
    		cin>>s;
    		Mymap.insert(make_pair(a,s));
    	}
    	
    	scanf("%d",&n);
    	n--;
    	map <int,string> :: iterator it;
    	it = Mymap.begin();
    	while(n--)
    	{
    		it++;
    	
    	}
    	cout<<it->second<<endl;
    	
    	
    	
    }
    
    return 0;
}
