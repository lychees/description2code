#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long t,n,i,count,max;
	cin>>t;
	while(t--)
	{
		cin>>n;
		long long a[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		max=a[n-1];
		count=1;
		for(i=n-2;i>=0;i--)
		{
			if(a[i]>max)
			{
				max=a[i];
				count++;
			}
		}
		cout<<count<<endl;
	}
    return 0;
}
