#include<iostream>
#include<algorithm>
using namespace std;
typedef long long ll;
int main(){

    int n,r;
    int a[100010];
    cin>>n>>r;
    ll t=0,count=0;
    int m;
    for(int i=0; i<n; i++){
         cin>>m;
         m++;
         a[i] = m;
         t+=m;

    }
    if(t%n){
        cout<<"-1"<<"\n";
        return 0;
    }
    else {

        if(r==0){
            cout<<"0"<<"\n";
            return 0;
        }
        else{
            t = t/n;
            sort(a,a+n);
            int i=0;
            int j=n-1;

            while(i<=j){

                ll dw =0,dg=0;
                dw = t - a[i];
                dg = a[j]-t;
                if(dw==dg&&dw!=0){

                    i++,j--;
                    count++;
                }
                else if(dw>dg){
                    a[i]+=dg;
                    count++;
                    j--;
                }
                else if(dw>0){
                    count++;
                    a[j]-=dw;
                    i++;
                }
                else
                    break;
            }

        }

     }
     cout<<count*r<<"\n";
     return 0;
}
