#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	long int sum=0,i=0;
    	while(s[i]!='\0')
    	{
    		if(s[i]=='0')sum+=6;
    		else if(s[i]=='1')sum+=2;
    		else if(s[i]=='2')sum+=5;
    		else if(s[i]=='3')sum+=5;
    		else if(s[i]=='4')sum+=4;
    		else if(s[i]=='5')sum+=5;
    		else if(s[i]=='6')sum+=6;
    		else if(s[i]=='7')sum+=3;
    		else if(s[i]=='8')sum+=7;
    		else
    		sum+=6;
    		i++;
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
