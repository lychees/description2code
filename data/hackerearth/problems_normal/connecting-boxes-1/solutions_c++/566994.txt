using namespace std;
int A[1009][3];
int C[1009];
int main() 
{
	int n,q;
	cin>>n>>q;
	bool ff=false;
	for(int i=0;i<n;i++) C[i]=0;
	for(int i=0;i<q;i++)
	{
		int x1,y1,x,y;
		cin>>x1>>y1;
		x1--;
		y1--;
		
		x=min(x1,y1);
		y=max(x1,y1);
		A[i][0]=x;
		A[i][1]=y;
		A[i][2]=0;
		if(x==y || C[x]==1 || C[y]==1)
		{
			cout<<"NO"<<endl;
		}
		else
		{
			bool f=false;
			for(int j=0;j<i;j++)
			{
				
				if(A[j][2]==1 || ff==false)
				{
					if(ff==false) ff=true;
					if(x<A[j][0] && x<A[j][1] && y<A[j][1] && y>A[j][0])
					{
						f=true;
					}
					if(x>A[j][0] && x<A[j][1] && y>A[j][1] && y>A[j][0])
					{
						f=true;
					}
				}
			}
			if(f==true) cout<<"NO"<<endl;
			else 
			{
			cout<<"YES"<<endl;
			A[i][2]=1;
			C[x]=1;
			C[y]=1;
			}
		}
	}
		    return 0;
}
