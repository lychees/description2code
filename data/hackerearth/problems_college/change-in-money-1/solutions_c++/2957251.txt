#include <bits/stdc++.h>
using namespace std;
#define ll long long

int a[50],n,k;
ll int dp[50][250];

ll int func(int i,int x){
	if (x<0) return 0;
	if (x==0) return 1;
	if (i==k) return 0;
	if (dp[i][x]>0) return dp[i][x];
	for (int j=i;j<k;j++){
		dp[i][x]+=(func(j,x-a[j]));
	}
	return dp[i][x];
}

int main(){
	scanf("%d%d",&n,&k);
	for (int i=0;i<k;i++)
		scanf("%d",&a[i]);
	cout << func(0,n);
	return 0;
}
