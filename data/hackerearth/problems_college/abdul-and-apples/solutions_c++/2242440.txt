#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
#define pii pair<int, int>
#define pll pair<long long, long long>
#define vi  vector<int>
#define pb  push_back
#define mp  make_pair
#define M(a,x) memset(a,x,sizeof(a))
#define sci(x) scanf("%d",&x);
#define scs(x) scanf("%s",x);
#define print(x) printf("%d",x);
#define nl printf("\n")
#define printl(x) printf("%lld",x)
#define F(i,a,n) for(int i=a;i<n;i++)
#define INF 100000000000000000LL
#define LL long long
const int N=105;
LL dp[N][N];
int main() {
    int t;sci(t);
    while(t--) {
        int n;sci(n);
        int m;sci(m);
        M(dp,0);
        F(i,1,n+1) {
            F(j,1,m+1) {
                scanf("%lld",&dp[i][j]);
                dp[i][j]+=dp[i-1][j];
                dp[i][j]+=dp[i][j-1];
                dp[i][j]-=dp[i-1][j-1];
            }
        }
        LL ans=-100000000000000;
        for(int i=1;i<=n;i++) {
            for(int j=1;j<=m;j++) {
                for(int k=i;k<=n;k++) {
                    for(int l=j;l<=m;l++) {
                        if(abs(k-i)==abs(l-j)) {
                            LL temp=dp[k][l];
                            temp-=dp[i-1][l];
                            temp-=dp[k][j-1];
                            temp+=dp[i-1][j-1];
                            ans=max(ans,temp);
                        }
                    }
                }
            }
        }
        printf("%lld\n",ans);
    }
    return 0;
}
