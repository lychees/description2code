#include <bits/stdc++.h>

using namespace std;

int a,l,b,ld[20000],i;

inline int F(int x)
{
    int y = 0,k = 0;
    while(x)
    {
        y = x%10;
        if((y == 4) || (y == 7))
            ++k;
        x /= 10;
    }
    return k;
}

int main()
{
    cin >> a >> l;
    for(i=a;i<a+l;++i)
        ld[i-a] = F(i);
    b = a;
    while(++b)
    {
        if(F(b) == ld[0])
        {
            for(i=b;i<b+l;++i)
                if(F(i) != ld[i-b])
                    break;
            if(i == b+l)
            {
                printf("%d\n",b);
                break;
            }
        }
    }
    return 0;
}
