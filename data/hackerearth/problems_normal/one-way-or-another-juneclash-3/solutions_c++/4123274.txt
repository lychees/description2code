#include <iostream>
using namespace std;

int main()
{
	int t,n;
	long long r,c,x;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cin>>r>>c>>x;
		int odd,even;
		if(x&1)
		{
			if((r+c)&1)
			{
				odd=1;
				even=0;
			}
			else
			{
				even=1;
				odd=0;
			}
		}
		else
		{
			if((r+c)&1)
			{
				odd=0;
				even=1;
			}
			else
			{
				even=0;
				odd=1;
			}
		}
		int ans=1;
		for(int i=1;i<n;i++)
		{
			cin>>r>>c>>x;
			if((r+c)&1)
			{
				if(x&1 && odd==0)
					ans=0;
				else if(!(x&1) && odd==1)
					ans=0;
			}
			else
			{
				if(x&1 && even==0)
					ans=0;
				else if(!(x&1) && even==1)
					ans=0;
			}
		}
		if(ans)
			cout<<"Yes"<<endl;
		else
			cout<<"No"<<endl;
	}
    return 0;
}
