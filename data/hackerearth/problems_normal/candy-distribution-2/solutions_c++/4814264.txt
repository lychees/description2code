#include <iostream>
#define ll long long
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n;
    	cin >> n;
    	int n1= n;
    	ll s = 0;
    	while(n--)
    	{
    		int x;
    		cin >> x;
    		s += x;
    	}
    	cout << s-n1+1 << '\n';
    }
    return 0;
}
