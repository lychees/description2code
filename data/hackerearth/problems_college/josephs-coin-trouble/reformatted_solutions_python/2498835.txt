from itertools import combinations_with_replacement
n = int(raw_input())
lst = []
lst = map(int, raw_input().split())
con = 1
count = 0
while con != n:
    z = [p for p in combinations_with_replacement(lst, con)]

    for j in z:
        if sum(j) == n:
            count += 1
            # print i
    con += 1
print count
