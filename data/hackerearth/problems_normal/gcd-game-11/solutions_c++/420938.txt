#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
using namespace std;
typedef int ll;
bool arjit[1001][1001],chandu[1001][1001];
void initialize()
{   ll i,j,z;//i,j represents bullets a,b respectively
	for(i=1;i<=1000;i++)//initialize
	{
			arjit[i][1]=1;
			arjit[1][i]=0;
			chandu[1][i]=1;
			chandu[i][1]=0;
	}
	for(i=2;i<=1000;i++) // according to conditions
	{
		for(j=2;j<=1000;j++)
		{	z=__gcd(i,j);
			arjit[i][j]=(chandu[i][j-1]==0)||(z>1&&(chandu[i][j/z]==0));//move made by arjit
			chandu[i][j]=(arjit[i-1][j]==0)||(z>1&&(arjit[i/z][j]==0));// move made by chandu
		}
	}
}
int main()
{	
	ll a,b,t;
	initialize();
	cin>>t;
  while(t--)
  {
	cin>>a>>b;
	if(a==1&&b==1)
	 cout<<"Draw\n";
	else if(arjit[a][b]==1)
	 cout<<"Arjit\n";
	else 
	 cout<<"Chandu Don\n";
  }	
	return 0;
}
