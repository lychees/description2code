#include <stdio.h>
#include<string.h>

int main()
{
long int t,len1,len2,i,j,flag,count,temp;
char a[10001],b[10001];
scanf("%ld",&t);
while(t--)
{
flag=temp=0;
scanf("%s%s",&a[0],&b[0]);
len1=strlen(&a[0]);
len2=strlen(&b[0]);
if(len2>len1)
{
printf("NO\n");
}
else
{
for(i=0;i<len2;i++)
{
count=0;
for(j=temp;j<len1;j++)
{
if(b[i]==a[j])
{
count++;
temp=j+1;
break;
}
}
if(count==0)
break;
}
if(count==0)
printf("NO\n");
else
printf("YES\n");
}
}
return 0;
}
