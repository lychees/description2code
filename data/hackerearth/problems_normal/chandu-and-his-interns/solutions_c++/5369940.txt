#include <iostream>
using namespace std;

int main()
{
	int ans;
	long int i,j;
    long int a,b;
    cin>>a;
    while(a--)
    {
    	ans = 0;
    	cin>>b;
    	if(b==1)
    	{
    		cout<<"NO"<<endl;
    		continue;
    	}
    	for(i=1;i*i<=b;i++)
    	{
    		if(b%i==0)
    		{
    			if(i*i==b)
    			ans = ans+1;
    			else
    			ans = ans+2;
    		}
    		if(ans>3)
    		{
    			cout<<"YES"<<endl;
    			break;
    		}
    	}
    	if(ans<4)
    		cout<<"NO"<<endl;
    }
    return 0;
}
