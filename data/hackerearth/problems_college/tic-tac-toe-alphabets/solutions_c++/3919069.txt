#include <iostream>
#include<stdio.h>
#include<string.h>
using namespace std;

int main()
{
    int n;
    char a[10000];
    cin>>n;
    for(int k=1;k<=n;k++)
    {
    cin>>a;
    int l=strlen(a);
    for(int i=0;i<l-1;i++)
    {
    	for(int j=i+1;j<l;j++)
    	if(a[i]>a[j])
    	{
    		char r=a[i];
    		a[i]=a[j];
    		a[j]=r;
    	}
    }
    puts(a);
   }
    return 0;
}
