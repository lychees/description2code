def gcd(*numbers):
    from fractions import gcd
    return reduce(gcd, numbers)
 
def lcm(*numbers):
    def lcm(a, b):
        return (a * b) // gcd(a, b)
    return reduce(lcm, numbers, 1)
 
for i in range(int(raw_input())):
	n,a,b= map(int,raw_input().split())
	if a==0 or b==0:
		print '-1'
	else:
		al = n//a
		bl = n//b
		l = n//lcm(a,b)
		print al+bl-l
