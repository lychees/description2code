#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <climits>
using namespace std;

int main(int argc, char *argv[])
{
    int T, n,K,start, end;
    long long int maximum,minimum,temp,difference;
    scanf("%d",&T);
    while(T--){
          maximum = INT_MIN;
          minimum = INT_MAX;     
          scanf("%d%d",&n,&K);
          int A[n];
          for(int i = 0;i<n;++i) scanf("%d",&A[i]);
          sort(A,A+n);
          start = 0;
          end = n-1;    
          while(start<=end){
                temp = A[start]+A[end];
                maximum = max(maximum,temp);
                minimum = min(minimum,temp);
                ++start;
                --end;
          } 
          difference = maximum - minimum;    
          printf("%lld\n",difference);
          if(difference > K) printf("No more girlfriends!\n");
          else if(difference == K) printf("Lucky chap!\n");
          else                     printf("Chick magnet Jhool!\n");                  
    }
    return 0;
}
