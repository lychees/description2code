#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int main()
{
int i,t,n,counter,j;
cin>>t;
int a[t];
string s;
for( i = 0; i < t; i++ ){
	cin>>s;
	n = s.size();
	//cout<<s<<" "<<s[3]<<" "<<n<<" ";
	int b[n];
	counter = 0;
	for( j = 0; j < n; j++ ){
		b[j] = 1;
	}
	for( j = 0; j < n; j++ ){
		if( s[j] == 'D' ){
			b[j] = 0;			
		}
		else if( s[j] == 'B' ){
			b[j] = 0;
			b[j - 1] = 0;
			b[j - 2] = 0;
		}
		else if( s[j] == 'S' ){
			b[j] = 0;
			b[j - 1] = 0;
			b[j + 1] = 0;
		}
	}
	for( j = 0; j < n; j++ ){
		if(b[j]==1){
			counter++;
		}
	}
	a[i] = counter;
}
for( i = 0; i < t; i++ ){
	cout<<a[i]<<"\n";
}
return 0;
}
