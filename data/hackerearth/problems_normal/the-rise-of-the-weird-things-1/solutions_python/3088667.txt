#print 'Hello World!'
raw_input()
l=map(int,raw_input().split())
od=[]
ev=[]
for i in l:
	if i%2==0:
		ev.append(i)
	else:
		od.append(i)
ev.sort()
od.sort()
print ' '.join(map(str, ev)), sum(ev), ' '.join(map(str, od)), sum(od)
