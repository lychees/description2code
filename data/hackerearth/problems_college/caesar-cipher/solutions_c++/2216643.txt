#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
   { 
     int k;
     cin>>k;
     string str;
     cin>>str;
     int p=str.length();
     for(int i=0;i<p; i++)
     {
     	if(str[i]>=65 && str[i]<=90)
     	{
     	int s=k+str[i]-65;
     	s=s%26;
     		str[i]=65+s;
     	}
     	else
     	{
     		int s=k+str[i]-97;
     		s=s%26;
     		str[i]=97+s;
     	}
     }
     for(int i=0;i<p;i++)
     cout<<str[i];
     cout<<endl;
   }
    return 0;
}
