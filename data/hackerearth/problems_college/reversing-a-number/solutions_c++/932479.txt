#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
    string input;
    int t;
    cin>>t;
    cin.ignore();
    while(t--)
    {
        getline(cin,input);
        if((int)input[0]%2==0)
        {
            printf("EVEN\n");
        }
        else
        {
            printf("ODD\n");
        }
        input.clear();
    }
    return 0;
}
