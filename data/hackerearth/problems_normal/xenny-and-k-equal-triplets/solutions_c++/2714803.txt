#define ll long long
ll int c(ll int n)
{
	if(n<3)
	return 0;
	ll int ans= (n*(n-1))/2;
	return (ans*(n-2))/3;
}  
ll int gcd(ll int a,ll int b)
{
	if(b!=0)
	return gcd(b,a%b);
	return a;
}
