#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    char a[1001];
    scanf("%d",&t);
    getchar();
    while(t--)
    {
       scanf("%s",a);
       int l=strlen(a);
       int flag=0;
       for(int i=1;i<l;i++)
       {
       	if(abs(a[i]-a[i-1])!=abs(a[l-i]-a[l-i-1]))
       	{
       	flag=1;
       	break;
       	}
       }
       if(flag==0)
       printf("Funny\n");
       else
       printf("Not Funny\n");
    }
    return 0;
}
