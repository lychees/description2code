#include <bits/stdc++.h>

#define sf scanf
#define pf printf

#define row 1005
#define col 1005
#define MAX 100005
#define MOD 10000007
#define set(x) memset(x,-1,sizeof(x))

typedef long long ll;
typedef unsigned long long llu;

using namespace std;

int main()
{
    //freopen("in.txt","r",stdin);
    //freopen("out.txt","w",stdout);
    int n,zero,one,q;
    string a,b;
    sf("%d",&n);
    for(int i=1;i<=n;i++){
        cin>>a>>b;
        zero=one=q=0;
        int len=a.length();
        for(int j=0;j<len;j++){
            if(b[j]=='0')zero++;
            if(b[j]=='1')one++;
            if(a[j]=='0')zero--;
            if(a[j]=='1')one--;
            if(a[j]=='?')q++;
        }
        if(one<0){
            pf("Case %d: -1\n",i);
            continue;
        }
        //pf("%d %d %d\n",zero,one,q);
        int cnt=0,zo=0,oz=0;
        for(int j=0;j<len;j++){
            if(a[j]=='?'&&b[j]=='1'){
                if(one>0)a[j]='1',one--,cnt++;
                else if(zero>0)a[j]='0',zero--,cnt++;
            }
            if(a[j]=='?'&&b[j]=='0'){
                if(zero>0)a[j]='0',zero--,cnt++;
                else if(one>0)a[j]='1',one--,cnt++;
            }
            if(a[j]!=b[j]){
                if(a[j]=='0'&&b[j]=='1')zo++;
                if(a[j]=='1'&&b[j]=='0')oz++;
            }
        }
        //cout<<a<<"\n"<<b<<endl;
        //cout<<zo<<endl;
        cnt+=zo;
        pf("Case %d: %d\n",i,cnt);
    }
    return 0;
}
