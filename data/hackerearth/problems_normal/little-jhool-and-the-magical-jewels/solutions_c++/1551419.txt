#include <iostream>
using namespace std;
int min(int x,int y)
{   int minimum=0;
	if(x>y||x==y)
	{
	  return y;
	}
	else{
		return x;
	}
}
int main()
{
    int T,i,r_no,y_no,b_no,u_no,p,q,z;
    char str[100];
    cin>>T;
    while(T--)
    {
    	cin>>str;
    	r_no=0,y_no=0,b_no=0,u_no=0;
    	for(i=0;str[i]!='\0';i++)
    	{
    		if(str[i]==98)  //b
    		{
    			b_no++;
    		}
    		if(str[i]==114)  //r
    		{
    			r_no++;
    		}
    		if(str[i]==117)   //u
    		{
    			u_no++;
    		}
    		if(str[i]==121)
    		{
    			y_no++;
    		}
    	}
    	p=min(b_no,r_no);
    	q=min(u_no,y_no);
    	z=min(p,q);
    	cout<<z<<endl;
    }
    return 0;
}
