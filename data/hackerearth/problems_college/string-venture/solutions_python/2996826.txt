import operator


given_String=raw_input()
x=0
h={i:0 for i in given_String}

for i in given_String:
	x+=ord(i)
	h[i]+=1

x/=len(given_String)
if x%2!=0:
	print(max(h.iteritems(), key=operator.itemgetter(1))[0])
else:
	print(given_String[::-1])
