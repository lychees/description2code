#include <iostream>
using namespace std;
typedef long long ll;
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
	    int n;
	    cin>>n;
	    ll a,sum;
	    for(int i=0;i<n;i++)
	    {
	    	cin>>a;
	    	if(i==0)
	    	sum=a;
	    	else
	    	sum=sum+a;
	    }
	    ll ans = sum-(n-1);
	    cout<<ans<<endl;
	}
    return 0;
}
