#include<bits/stdc++.h>

#define MAX 1000005
#define ll long long int
#define PI 3.14159265359
#define MOD 1000000007
#define SET(x,y) memset(x,y,sizeof(x));

using namespace std;

ll n,k;
ll a[100000],dp[100000];

ll fun(ll n){

	if(n<0) return 0;

	if(dp[n]!=-1) return dp[n];

	dp[n]=max(a[n]+fun(n-k),fun(n-1));

	return dp[n];
}

int main(){

	ll t,i,ans;

	scanf("%lld",&t);

	while(t--){
		scanf("%lld%lld",&n,&k);

		for(i=0;i<n;i++)
			scanf("%lld",&a[i]);

		SET(dp,-1);

		n--;k++;

		ans=fun(n);

		printf("%lld\n",ans);
	}
	return 0;
}
