#include <iostream>
using namespace std;

int main()
{
		int i,j,x,set[100000],N,count=0;
		cin>>N;
		for(i=0;i<N;i++)
		{
				cin>>x;
				if(x>=1&&x<=1000)
				set[i]=x;
		}
		for(i=0;i<N;i++)
		{
			x=set[i];
			for(j=0;j<i;j++)
			{
				if(x==set[j])
				{
					count++;
					break;
				}
			}
			
		}
		cout<<count;
		return 0;
}
