#include <bits/stdc++.h>

using namespace std;

const int N = 1e4 + 10;
const long long inf = 1e18;

int main() {
	int tt;
	scanf("%d", &tt);
	for (int tc = 1; tc <= tt; tc++) {
		map <int, int> mp;
		int n;
		scanf("%d", &n);
		for (int i = 0; i < n; i++) {
			int x, y;
			scanf("%d %d", &x, &y);
			mp[x]++;
			mp[y]--;
		}
		long long cur = 0, res = -inf;
		for (auto it = mp.begin(); it != mp.end(); ++it) {
			cur += it->second;
			res = max(res, cur);
		}
		printf("%lld\n", res);
	}
	return 0;
}
