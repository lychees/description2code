#include <iostream>
#include <math.h>
using namespace std;
#define ll long long

ll pr(ll x)
{	if(x==2 || x==3 || x==5 || x==7)
		return 1;
	if(x<11 || x%2==0)
		return 0;
	ll p=sqrt(x);
	for(ll i=3;i<=p;i+=2)
		if(x%i==0)
			return 0;
	return 1;
}

int main()
{	ll t;
	cin>>t;
	while(t--)
	{	ll n,a,c=0;
		cin>>n;
		while(n--)
		{	cin>>a;
			if(pr(a))
				c++;
		}
		cout<<c<<"\n";
	}
    return 0;
}
