#include<cstdio>
int p[1000000]={};
int contain(int i)
{
 while(i>0)
 {
  int d=i%10;
  if(d==7)
   return 1;
  i=i/10; 
 }
 return 0;
}
void seive()
{
 int i,j;
  for(i=2;i<1000000;i++)
   p[0]=0; 
  for(i=2;i<1000000;i++)
  {
   if(p[i]==1)
    continue;
   for(j=2;i*j<1000000;j++)
   {
     p[i*j]=1;
   }
  }
}

int main()
{
 int t,m,n,i;
 scanf("%d",&t);
 seive();
 while(t--)
 {
  int c=0;
  scanf("%d%d",&m,&n);
  for(i=m;i<=n;i++)
  {
   if(p[i]==1&&contain(i))
    c++;
  }
  if(c==0)
   printf("-1\n");
  else
   printf("%d\n",c); 
 }
return 0;
}
