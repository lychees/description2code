#include <bits/stdc++.h>
using namespace std;

int main()
{
	double a1,a2,a3,b1,b2,b3;
	
	 cin>>a1>>b1>>a2>>b2>>a3>>b3;
    
    //cout << "Hello World!" << endl;
    
    double x1,x2,x3,y1,y2,y3;
    
    double temp=(a1+a2+a3);
  
    x3=temp-2*a1;
    x1=temp-2*a2;
    x2=temp-2*a3;
    
    temp=(b1+b2+b3);
  
    y3=temp-2*b1;
    y1=temp-2*b2;
    y2=temp-2*b3;
    
    pair<double,double> a[3];
    
    a[0].first=x1;
    a[0].second=y1;
    a[1].first=x2;
    a[1].second=y2;
    a[2].first=x3;
    a[2].second=y3;
    
    sort(a,a+3);
    
    for(int i=0;i<3;i++)
      printf("%.4f %.4f\n",a[i].first, a[i].second);
    
    
    return 0;
}
