#include<stdio.h>
#include<string.h>
int main()
{
    int t,i,j,max,l,k,count;
    char s[10001],c;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s",s);
        k=strlen(s);
        max=0;
        l=0;
        for(i=0;i<k;i++)
        {
            c=s[i];
            count=1;
            for(j=i-1;j>=l;j--)
            {
                if(c==s[j])
                {
                    l=j+1;
                    break;
                }
                else count++;
            }
            if(count>max) max=count;
        }
        printf("%d\n",max);
    }
    return 0;
}
