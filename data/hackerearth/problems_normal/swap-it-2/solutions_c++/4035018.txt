#include<bits/stdc++.h>
using namespace std;
#define in(n) scanf("%lld",&n)
int main()
{
    long long int t,n,k,i,m,a,j;
    in(t);
    while(t--)
    {
        in(n);
        in(k);
        vector<long long int> v(n);
        for(i=0;i<n;i++)
            in(v[i]);
        i=0;
        while(k>0 && i<n)
        {
            m=1000000,a=0;
            for(j=i;j<k+1+i && j<n;j++)
            {
                if(v[j]<m)
                    a=j,m=v[j];
            }
            if(a>i)
            {
                for(j=a-1;j>=i;j--)
                    v[j+1]=v[j];
                v[i]=m;
                k=k-(a-i);
            }
            i++;
        }
        for(i=0;i<n;i++)
            cout << v[i] << " ";
        cout << endl;
    }
    return 0;
}
