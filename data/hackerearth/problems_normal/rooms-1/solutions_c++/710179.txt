#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long n;
    	cin>>n;
    	long long a[n],b[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	for(int i=0;i<n;i++)
    	{
    		cin>>b[i];
    		b[i]+=a[i];
    	}
    	sort(a,a+n);
    	sort(b,b+n);
    	long long c=0,i=0,j=0,max=-1;
    	while(i!=n && j!=n)
    	{
    		//cout<<"A"<<a[i]<<"B"<<b[i]<<endl;
    		if(a[i]<b[j])
    		{c++;i++;}
    		else if(a[i]>b[j])
    		{c--;j++;}
    		else
    		{i++;j++;}
    		
    		if(c>max)max=c;
    		//cout<<"C"<<c<<"Max"<<max<<endl;
    	}
    	
    	if(i!=n)
    	{
    		i++;c++;
    	}
    	else if(j!=n)
    	{
    		j++;c--;
    	}
    	if(c>max)max=c;
    	cout<<max<<endl;
    }
    return 0;
}
