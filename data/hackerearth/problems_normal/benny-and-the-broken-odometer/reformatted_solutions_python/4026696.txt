def p(x, y):
    if y == 0:
        return 1
    return x*p(x, y-1)


for _ in range(int(input())):
    n = (raw_input().split())[0]
    ans = 0
    j = 0
    # print n
    for i in range(len(n)-1, -1, -1):
        # print i
        # print n[i]
        if int(n[i]) < 3:
            ans += p(9, j)*int(n[i])
        else:
            ans += p(9, j)*(int(n[i])-1)
        j += 1
    print ans
