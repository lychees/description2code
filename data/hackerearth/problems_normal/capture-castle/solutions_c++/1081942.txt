#include<stdio.h>
#include<iostream>
#include<queue>
using namespace std;
int main()
	{
	queue<int> q;
	int testcases,m,n,rd,cd,td,r,c;
	int row[4]={1,-1,0,0};
	int col[4]={0,0,1,-1};
	scanf("%d",&testcases);
	int mat[1000][1000];
	int cost[1000][1000];
	while(testcases>0)
		{
		scanf("%d %d",&m,&n);
		for(int j=0;j<m;j++)
			for(int i=0;i<n;i++)
				{
				scanf("%d",&mat[j][i]);
				cost[j][i]=999;
				}
		scanf("%d %d %d",&rd,&cd,&td);
		rd--;
		cd--;
		cost[0][0]=mat[0][0];
		q.push(0);
		q.push(0);
		while(!q.empty())
			{
			r=q.front();
			q.pop();
			c=q.front();
			q.pop();
			for(int i=0;i<4;i++)
				{
				if(r+row[i]>=0 && c+col[i]>=0 && cost[r][c]+mat[r+row[i]][c+col[i]]>=0 &&cost[r+row[i]][c+col[i]]>cost[r][c]+mat[r+row[i]][c+col[i]])	
					{
					
						cost[r+row[i]][c+col[i]]=cost[r][c]+mat[r+row[i]][c+col[i]];
						//printf("cost of source is %d & cell [%d][%d] is %d",cost[r][c],r+row[i],c+col[i],cost[r+row[i]][c+col[i]]);
						q.push(r+row[i]);
						q.push(c+col[i]);
					
					}				
				}
			}
		int ans=td-cost[rd][cd];
		if(cost[rd][cd]>td)
			printf("NO \n");
		else
			{
			printf("YES");
			printf("\n");
			printf("%d \n",ans);
			}
		testcases--;
		}
	}
