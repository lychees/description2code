#include <bits/stdc++.h>
using namespace std;

pair<int, int> convert(string s)
{
	pair<int, int> ret;
	ret.first = 1000*s[0] + 100*s[1] + 10*s[3] + s[4];
	ret.second = 1000*s[6] + 100*s[7] + 10*s[9] + s[10];
	return ret;
}

int main()
{
	int n;
	string s;
	pair<int, int> arr[200];
	
	cin >> n;
	for(int i = 0; i < n; ++i) {
		cin >> s;
		arr[i] = convert(s);
	}
	
	sort(arr, arr+n);
	
	bool want = false;
	for(int i = 1; i < n; ++i) {
		if(arr[i].first < arr[i-1].second) {
			want = true;
			break;
		}
	}
	
	cout << (want ? "Will need a moderator!\n": "Who needs a moderator?\n"); 
}
