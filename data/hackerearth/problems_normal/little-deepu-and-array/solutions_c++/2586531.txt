#include <iostream>
using namespace std;

int main()
{
  int t;
	cin >> t;
	long *arr = new long[t];
	int i = 0;
	int ctr = t;
	while (ctr-- > 0)
	{
		long ip;
		cin >> ip;
		arr[i++] = ip;
	}
	int cs;
	cin >> cs;
	while (cs-- > 0)
	{
		long val;
		cin >> val;
		for (int i = 0; i < t; i++)
		{
			if (arr[i] > val)
				arr[i]--;
		}
	}
	for (int i = 0; i < t; i++)
		cout << arr[i] << " ";
}
