char solve(int level, ll x){
 // cerr<<endl<<"Call "<<level<<" "<<x<<endl;
  if(level==0) return s[level][(int)(x)];
  p = csl[level-1];
  //cerr<<p<<" "<<endl;
  if(x<p) return solve(level-1,x);
  else if(x<2*p) return solve(level-1,2*p - x-1);
  else return s[level][(int)(x - 2*p)];
}
