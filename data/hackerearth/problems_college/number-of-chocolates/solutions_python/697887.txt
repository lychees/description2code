n,m=map(int,raw_input().split())
a=[0]+map(int,raw_input().split())
for i in range(1,n+1):
	a[i]+=a[i-1]
for i in range(m):
	q,w=map(int,raw_input().split())
	e,r=map(int,raw_input().split())
	if e<q: e,q=q,e
	if r<w: r,w=w,r
	if r<q: print a[r+1]-a[w]+a[e+1]-a[q]
	elif r<=e and w<=q: print a[e+1]-a[w]
	elif r<=e: print a[e+1]-a[q]
	elif w<=q: print a[r+1]-a[w]
	elif w<=e: print a[r+1]-a[q]
	else: print a[r+1]-a[w]+a[e+1]-a[q]
