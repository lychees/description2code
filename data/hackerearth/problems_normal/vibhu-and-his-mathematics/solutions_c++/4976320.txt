#include <bits/stdc++.h>

using namespace std;

#define MOD 1000000007

int main() {
    int T;
    cin >> T;
    long long dp[1000001];
    dp[1] = 1;
    dp[2] = 2;
    for (int i = 3; i <= 1000000; i++) {
    	dp[i] = (dp[i-1] + (i-1)*dp[i-2]) % MOD;
    }
    for (int t = 0; t < T; t++) {
    	int N;
    	cin >> N;
    	cout << dp[N] << endl;
    }
    return 0;
}
