#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int l=s.length();
    	long long ans=0;
    	for(int i=0;i<l-1;i++)
    	{
    		for(int j=i,k=i+1;j>=0&&k<l;j--,k++)
    		{
    			if(s[j]==s[k])
    			ans+=((k-j+1)*(k-j+1));
    			else
    			break;
    		}
    	}
    	for(int i=0;i<l;i++)
    	{
    		for(int j=i,k=i;j>=0&&k<l;j--,k++)
    		{
    			if(s[j]==s[k])
    			ans+=((k-j+1)*(k-j+1));
    			else
    			break;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
