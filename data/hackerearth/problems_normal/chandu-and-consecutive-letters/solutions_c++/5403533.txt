#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	string s,result="";
    	cin>>s;
    	int len = s.size();
    	result+=s[0];
    	for (int i=1;i<len;i++) {
    		if(s[i]==s[i-1]) {
    			continue;
    		}else {
    			result+=s[i];
    		}
    	}
    	cout<<result<<endl;
    }
    return 0;
}
