//Longest common subsequence
int maxHealth(int *arr,int n)
{
    int i,j;
    int curr_so_far=arr[0],max_so_far=arr[0];
    for(i=1;i<n;i++)
    {
        curr_so_far=max(curr_so_far+arr[i],arr[i]);
        max_so_far=max(curr_so_far,max_so_far);
    }
    return max_so_far;
}

int main()
{
    int i,t,n;
    int *arr=new int[N];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(i=0;i<n;i++)
        scanf("%d",&arr[i]);
        printf("%d\n",maxHealth(arr,n));
    }
    return 0;
}
