void erest()
{
    int i,j;
    kt[1] = true; d[1] = 1;
    for (i=2;i<=lim;i++) if (!kt[i])
    {
        d[i] = i;
        for (j=2*i;j<=lim;j+=i)
        {
            kt[j] = true;
            d[j] = i;
        }
    }
}


int main()
{

            dem++;
            j /= d[j];
        }
        res ^= dem;
