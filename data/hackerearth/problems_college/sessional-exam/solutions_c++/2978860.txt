#include <bits/stdc++.h>
using namespace std;

int main() {
	long long int n,i,j,k,l,m,count=0;
	cin>>n>>m;
	long int a[n][m];
	memset(a,0,sizeof(a));
	for(i=0;i<n;i++)
	{
		string str;
		cin>>str;
		for(j=0;j<m;j++)
		{
			a[i][j]=str[j]-'0';
		}
	}
	
	long long int c[n];
	memset(c,0,sizeof(c));
	
	for(i=0;i<m;i++)
	{
		int max=0;
		for(j=0;j<n;j++)
		{
			if(a[j][i]>max)
			{
				max=a[j][i];
				//l=j;
			}
			
		}
		
		for(j=0;j<n;j++)
		{
			if(a[j][i]==max)
			c[j]++;
		}
		

	}
	
	for(i=0;i<n;i++)
	{
		if(c[i]>0)
		count++;
	}
	
	cout<<count;
}
