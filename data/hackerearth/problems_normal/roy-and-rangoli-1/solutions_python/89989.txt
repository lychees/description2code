def main():
	n=int(raw_input())
	primear=[True for i in range(2*n)]
	primear[0]=primear[1]=False;
	for i in range(2*(n-1)):
		if(primear[i]):
			j=i+i;
			while(j<2*n):
				primear[j]=False;
				j+=i;
	ans=0
	if n==2:
		print "1"
		return
	for i in range(2*(n-1)):
		if(primear[i]):
			if(i<n):
				ans+=(i+1);
				ans%=1000000007
			else:
				opp=i-(n-1)
				ans+=(n-1-(opp))
				ans+=1
				ans%=1000000007

	print ans
main()
