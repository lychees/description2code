#include <bits/stdc++.h>
using namespace std;
bool isone(char x)
{
	switch(x)
	{
		case 'A':
		case 'a':
		case 'B':
		case 'b':
		case 'C':
		case 'c':
			return(true);
			break;
		default:
			return(false);
	};
}

int main()
{
    int t;
    char str[100001];
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	int count=0,j=0;
    	while(j<strlen(str))
    	{
    		int i=0;
    		while((j<strlen(str))&&(isone(str[j])))
    			{
    				i++;
    				j++;
    			}
    		count+=(i*(i+1))/2;	//substrings possible with a string of length i...
    		j++;
    		
    	};
    	cout<<count<<endl;
    	
    }
    return 0;
}
