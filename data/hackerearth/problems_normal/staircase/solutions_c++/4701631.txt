#include<bits/stdc++.h>
#define LL long long
#define MAX  100001
#define MAXM 1000001
#define FI(i,n) for(int i=0;i<n;i++)
#define FIM(i,m,n) for(int i=m;i<n;i++)
#define FLL(i,n) for(LL i=0;i<n;i++)
using namespace std;
LL JACK[MAXM];
LL JILL[MAXM];
LL jackcount(LL n){
JACK[0]=1;
JACK[1]=1;
JACK[2]=2;
for(LL i=3;i<=n;i++)
    JACK[i]=JACK[i-1]+JACK[i-2]+JACK[i-3];
return JACK[n];
}
LL jillcount(LL n){
JILL[0]=1;
JILL[1]=1;
for(LL i=2;i<=n;i++)
    JILL[i]=JILL[i-1]+JILL[i-2];
    return JILL[n];
}
int main(){
LL n;
cin>>n;
if(n<0){
    cout<<"ERROR";
}
else{
cout<<"Jack-"<<jackcount(n)<<endl;
cout<<"Jill-"<<jillcount(n)<<endl;
}
return 0;
}
