#include<iostream>

#define ll long int
using namespace std;
ll numDigits( ll number ) {
    ll digits = 0 ;
    ll num = 1 ;

    for( ll i = 9 ; number > i ; i *= 10 ){ 
		digits += i*num ;
		++num; 
		number -= i ;
	}
    return digits + number*num ;
}

int main(){
	int t;
	cin>>t;
	while(t--){
	long long int n;
	cin>>n;
	
	cout<<numDigits(n)<<endl;
	}
	
	
}
