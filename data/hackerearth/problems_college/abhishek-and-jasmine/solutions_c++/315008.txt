#include <iostream>
using namespace std;
int main()
{
	
	char n;
    int r=0;
	scanf("%d",&n);
	 char a[n];
	 char b[n];
	 char c[n];
	for(int i=0;i<n;i++)
	{
            c[i]=0;
            }
	for(int i=0 ; i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	for(int i=0;i<n-1;i++)
	{
		for(int j=i+1,k=0;j<n-i,k<n-i-1;j++,k++)
		{
			b[k] = a[i]^a[j];
			
			c[i]= b[k] | c[i];
			
		}

		r = r | c[i];
		
	}
	
	
    printf("%d\n",r);
   
    return 0;
}
