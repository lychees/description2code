def isPalin(c):
    if c == "":
        return True
    if c[0] == c[-1]:
        return isPalin(c[1:-1])
    return False

t = int(t)
while t > 0:
    a = raw_input()
    b = a.split(" ")
    count = 0
    for i in range(int(b[0]), int(b[1])+1):
        if isPalin(str(i)):
            count += 1
    print count
