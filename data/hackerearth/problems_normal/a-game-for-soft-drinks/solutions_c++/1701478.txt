#include <iostream>
using namespace std;

void sol(int dp[101][101][101])
{
	int i,j,k;

	for(i=0;i<=100;i++)
	{
		for(j=0;j<=100;j++)
		{
			for(k=0;k<=100;k++)
			{
				if(i==0 && j==0 && k==0)
					dp[i][j][k]=1;

				else if(i>0 && dp[i-1][j][k]==1)
					dp[i][j][k]=0;

				else if(j>0 && dp[i][j-1][k]==1)
					dp[i][j][k]=0;

				else if(k>0 && dp[i][j][k-1]==1)
					dp[i][j][k]=0;

				else if(i>0 && j>0 && dp[i-1][j-1][k]==1)
					dp[i][j][k]=0;

				else if(i>0 && k>0 && dp[i-1][j][k-1]==1)
					dp[i][j][k]=0;

				else if(k>0 && j>0 && dp[i][j-1][k-1]==1)
					dp[i][j][k]=0;

				else if(i>0 && j>0 && k>0 && dp[i-1][j-1][k-1]==1)
					dp[i][j][k]=0;
				else
				 	dp[i][j][k]=1;
			}
		}
	}
}


int main()
{
    int tno,N1,N2,N3;
    int dp[101][101][101];
    cin>>tno;

    sol(dp);

    while(tno--)
    {
    	cin>>N1>>N2>>N3;

    	if(dp[N1][N2][N3]==0)
    		cout<<"Ashima"<<endl;
    	else
    		cout<<"Aishwarya"<<endl;
    }
    return 0;
}
