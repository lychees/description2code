#include<iostream>
 #include <stdio.h>
#include <string.h>
#include <stdlib.h>
#define MOD 1000000007
using namespace std;
int main()
{
	char s[3000000];
	long long int l;
	cin>>s;
	l=strlen(s);
	long long int sum=s[0]-'0';
	long long int temp=s[0]-'0';
	for(int i=1;i<l;i++)
	{
		temp=temp*10+(s[i]-'0')*(i+1);
		temp%=MOD;
		sum=sum+temp;
	}	
	cout<<sum%MOD<<endl;
return 0;

}
