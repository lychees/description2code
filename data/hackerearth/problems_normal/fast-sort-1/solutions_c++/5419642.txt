#include<bits/stdc++.h>

using namespace std;

typedef long long LL;
typedef vector<LL> V;
typedef string S;

#define pb push_back 
#define mp make_pair 

int main()
{
    LL T;
    LL N,M,K;
    LL count,c1,c2;
    LL i,j,k;
    S P,F;
    
    cin>>T;
    
    map < char , int > m;
    vector <LL> C;
    
    while(T--)
    {
        cin>>P;
        cin>>F;
            
        for(i=0;i<P.length();i++)
        {
            m.insert(make_pair(P[i],i));
            C.pb(0);
        }
        for(i=0;i<F.length();i++)
        {
            C[m[F[i]]]++;
        }
        for(i=0;i<P.length();i++)
        {
            while(C[m[P[i]]]>0)
            {
                cout<<P[i];
                C[m[P[i]]]--;
            }
        }
        cout<<endl;
        
        C.clear();m.clear();
    }
    return 0;
}
