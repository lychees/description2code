#include<iostream>
#include<vector>
#define mod 1000000007
using namespace std;
int main()
{
    vector<long long int> v;
    long long int q,x,y,n;
    scanf("%lld%lld%lld",&x,&y,&q);
    v.push_back(1);
    for(int i=1;i<=1000001;i++)
    {
        int temp;
        temp=((v[i-1]%mod*x%mod)-y%mod)%mod;
        //if(temp>1000000007) temp = temp%1000000007;
        v.push_back(temp);
    }
    while(q--)
    {
        scanf("%lld",&n);
        printf("%lld\n",v[n]);
    }
    return 0;
}
