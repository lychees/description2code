#include<iostream>
#include<vector>
#include<queue>
using namespace std;
vector<long int> g[10001];
long int visited[10001];
long int dfs(int x,int size)
{
	visited[x]=1;
	for(int i=0;i<g[x].size();i++)
	{
		if(visited[g[x][i]]==0)
		size=dfs(g[x][i],size+1);
	}
	return size;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,f,s,a,b,i,x,j,c=0;
		cin>>n>>f>>s;
		for(i=0;i<f;i++)
		{
			cin>>a>>b;
			g[a].push_back(b);
		}
		for(j=0;j<=n;j++)
		visited[j]=0;
		for(i=0;i<s;i++)
		{
			cin>>x;
			if(visited[x]==0)
			c+=dfs(x,1);
		}
		cout<<c<<endl;
		for(i=0;i<=n;i++)
		g[i].clear();
	}
}
