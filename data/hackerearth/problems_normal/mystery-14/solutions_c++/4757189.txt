#include <iostream>
using namespace std;

int main()
{
   long long  int arr[1000000],result,n;
    int t;
    cin>>t;
     while(t--)
     {
             result=0;
     	cin>>n;
     	for(int i=0;i<n;i++)
     	{
     		 cin>>arr[i];
     		 if(arr[i]%2==1)
     		 
     		 {
     		 	result+=arr[i];
     		 
     		 	continue;
     		 }
     	}
     
     	cout<<result<<"\n";
     }
    return 0;
}
