#include<iostream>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;cin>>n;
		int a[200];
		int m=1,temp=0,index=0,i,j,l,x;
		a[0]=1;
		for(i=1;i<=n;i++){
			temp=0;index=0;
			for(j=0;j<m;j++){
				x=temp+a[index]*i;
				temp=x/10;
				a[index]=x%10;
				index++;
			}
			while(temp!=0){
				a[index]=temp%10;
				temp=temp/10;
				index++;m++;
			}			
		}
		for(i=index-1;i>=0;i--)
			cout<<a[i];
		cout<<endl;
	}
	return 0;
}
