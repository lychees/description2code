#include <iostream>
using namespace std;

int main()
{
    int  T, count, N;
    cin >> T;
    while(T--)
    {
    	count=0;
    	cin >> N;
    	for(int i=0,p=1;i<32;i++,p=(p<<1))
    	 if(N & p) count++;
    	if(count & 1)
    	 cout << "odd\n";
    	else
    	cout << "even\n";
    }
    return 0;
}
