#include<bits/stdc++.h>
using namespace std;
int palindrome(string str)
{
	string str1 = str;
	reverse(str.begin(),str.end());
	if(str1 == str)
	{
		return 1;
	}
	else
	{
		return 0;
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str;
		cin>>str;
		unsigned long long int pro = 1;
		if(palindrome(str) == 1)
		{
			cout<<"Palindrome"<<"\n";
		}
		else
		{
			for(int i =0;i<str.length();i++)
			{
				pro = pro*(str[i] - 'a' + 1);
			}
			cout<<pro<<"\n";
		}
	}
}
