# include <iostream>

using namespace std;

long long int maximum(long long int a,long long int b){
	if (a > b) return a;
	return b;
}

long long int gcd(long long int a,long long int b){
	long long int t = a+b;
	a = maximum(a,b); b = t - a;
	while(b > 0){
		t = b;
		b = a%b;
		a = t;
	}
	return a;
}

int main(){
	int T;
	scanf("%d",&T);
	for(int kases=0;kases<T;kases++){
		int N,j,pj;
		scanf("%d",&N);
		long long int A[N+2],B[N+2],C[N+2],D[N+2],E[N+2];
		
		char ch;
		for(int i=0;i<N;i++)
			scanf("%Ld%c%Ld",&A[i],&ch,&B[i]);

		C[N-1] = A[N-1]; D[N-1] = B[N-1];E[N] = 0;E[N-1] = N-1;
		for(int i=N-2;i>=0;i--){
			long double n1 = (long double) A[i]/B[i]; 
			long double n2 = (long double) C[i+1]/D[i+1];
			C[i] = A[i];
			D[i] = B[i];
			E[i] = i;

			if (n1<=n2){
				j  = i+1;
				while(n1 <= n2){
					C[i] = C[i] + C[j];
					D[i] = D[i] + D[j];
					j = E[j] + 1;
					if (j >= N) break;
					n1 = (long double) C[i]/D[i]; 
					n2 = (long double) C[j]/D[j];
				}
				E[i] = j - 1;
			}
		}
		//printf("%Ld/%Ld\n",C[0],D[0]);
		for(int i=0;i<N;i++){
			long long int g = gcd(C[i],D[i]);
			//g = 1;
			printf("%Ld/%Ld\n",C[i]/g,D[i]/g);
		}
		printf("\n");
	}
}
