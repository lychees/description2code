import math
question = []
primes = []
t = int(raw_input())
while t > 0:
    n = int(raw_input())
    question.append(n)
    t -= 1

maxi = max(question)
j = 2
while j <= maxi:
    k = 2
    z = 0
    while k <= math.sqrt(j):
        if j % k == 0:
            z = 1
            break
        k += 1
    if z == 0:
        primes.append(j)
    j += 1

l = []
for j in primes:
    for k in primes:
        if k <= j:
            continue
        if k+j > maxi:
            break
        if k+j in primes:
            if k+j not in l:
                l.append(k+j)

for i in question:
    t = [x for x in l if x <= i]
    print ' '.join(map(str, t))
