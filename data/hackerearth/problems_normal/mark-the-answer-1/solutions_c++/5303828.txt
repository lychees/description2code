#include <iostream>
using namespace std;

int main()
{
    int N,X,i;
    cin>>N>>X;
    int A[N];
    for(i=0;i<N;i++)
       cin>>A[i];
    int count=0,flag=1;
    i=0;
    while(i<N && flag<=2){
    	if(A[i] <=X)
    	   count++;
    	else
    	  flag++;
      i++;	  
    }
    cout<<count;
    return 0;
}
