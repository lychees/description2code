#include<cmath>
#include<cstdio>

using namespace std;

int main() {
    long t, n, min=0, max=1000000000, cb, pb;
    double avg;
    scanf("%ld", &t);
    for(; t ; t--) {
        min = 0;
        max = 1000000000;
        scanf("%ld", &n);
        if(n == 1) {
            scanf("%ld", &pb);
            printf("%ld %ld\n", min, max);
            continue;
        }
        else {
            scanf("%ld", &pb);
            n--;
        }
        for(; n ; n--) {
            scanf("%ld", &cb);
            avg = (double)(pb + cb) / 2.0;
            if(pb < cb) {
                if(avg < max)
                    max = (long)floor(avg);
            }
            else if(pb > cb) {
                if(avg > min)
                    min = (long)ceil(avg);
            }
            pb = cb;
        }
        if(min > max)
            printf("-1\n");
        else
            printf("%ld %ld\n", min, max);
    }
    return 0;
}
