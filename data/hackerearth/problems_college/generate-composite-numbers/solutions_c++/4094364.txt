#include <bits/stdc++.h>
using namespace std;
void prime(long long p[])
{
	long long i,j;
	memset(p,0,sizeof(p));
	p[0]=p[1]=-1;
	for(i=2;i<=10005;i++)
	{
		if(p[i]!=-1)
		{
			p[i]=1;
		}
		for(j=i*i;j<=10005;j+=i)
		{
			p[j]=-1;
		}
	}
}
int main()
{
    long long p[10005];
    prime(p);
    long long i,n;
    cin>>n;
    for(i=4;i<=n;i++)
    {
    	if(p[i]==-1)
    	{
    		cout<<i<<" ";
    	}
    }
    return 0;
}
