#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int N,i,j,k;
	vector<pair<long int,long int> > v;
	cin>>N;
	for(i=1;i<=N;i++)
	{
		cin>>j>>k;
		v.push_back(make_pair(j+k,i));
	}
	sort(v.begin(),v.end());
	pair<long int,long int> w;
	for(i=0;i<N;i++)
	{
		w=v[i];
		printf("%ld ",w.second);
	}
}
