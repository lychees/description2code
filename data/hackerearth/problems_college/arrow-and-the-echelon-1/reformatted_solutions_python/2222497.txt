for _ in range(0, int(input())):
    n = int(input())
    flag = False
    if n % 3 == 0:
        print("5"*n)
        flag = True
    elif n % 5 == 0:
        print("3"*n)
        flag = True
    else:
        for i in range(3, n):
            if i % 3 == 0 and (n-i) % 5 == 0:
                print("5"*i+"3"*(n-i))
                flag = True
    if not flag:
        print("-1")
