#include <bits/stdc++.h>
using namespace std;

int main()
{
	string s;
    std::getline(cin, s); 
	int i,len=s.length();
	int arr[256]={0},ch;
	for(i=0;i<len;i++)
	{
		ch=(int)s[i];
		arr[ch]++;
	}
	int max=0;
	char max_index;
    for( i=0;i<256;i++)
	{
		if(arr[i]>max)
		{
			max=arr[i];
			max_index=(char)i;
		}
	}
	cout<<max_index<<" "<<max;
    return 0;
}
