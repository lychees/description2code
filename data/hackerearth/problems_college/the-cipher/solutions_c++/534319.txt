#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
        int i, test_cases;
        scanf("%d",&test_cases);
        while(test_cases > 0)
        {
                int N,arr[10000],i=0;
                scanf("%d",&N);
                while(i<N)
                {
                        scanf("%d",&arr[i]);
                        i = i+1;
                }
                
                for(i=0;i<N;i++)
               	{
               		arr[i]=~arr[i];
               	}
               	
               	for(i=0;i<N;i++)
               		printf("%c",arr[i]);
               		
               	printf("\n");
                
                test_cases--;
        }
        return 0;
}
