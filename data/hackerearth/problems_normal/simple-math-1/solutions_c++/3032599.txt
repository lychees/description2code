#include <iostream>
using namespace std;



const int mod=10000007;
int dp[101][101];
int getVal(int n, int r){
	if(dp[n][r]>=0)return dp[n][r];
	if(n<r)return 0;
	
	if(n==r){
	//	cout<<"value for "<<n<<": 1"<<endl;
	return 1;	
	} 
	if(r==1)return 1;
	long long  su=0;
	for(int i=1;i<n-r+2;i++){
		su+=getVal(n-i,r-1);
	}
su%=mod;
//cout<<"value for"<<n<<" "<<r<<": "<<su<<endl;
dp[n][r]=su;
return su;
}
int main()
{
	
	for(int i=0;i<101;i++)for(int j=0;j<101;j++)dp[i][j]=-1;
	int n,r ; cin>>n>>r;
	int l=getVal(n,r);
    cout << (l?l:-1)<< endl;
}
