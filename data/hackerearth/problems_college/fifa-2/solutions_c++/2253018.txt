#include <bits/stdc++.h>
using namespace std;

int x[112345], y[112345], fr[112345];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++) {
    	scanf("%d %d",&x[i], &y[i]);
    	fr[x[i]]++;
    }
    for(int i=0;i<n;i++) 
    	printf("%d %d\n",n-1 + fr[y[i]], n-1 - fr[y[i]]);
    return 0;
}
