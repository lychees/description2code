#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,n,k,f;
	cin>>n;
	cin.ignore(256,'\n');
	while(n--)
	{
		string s;
		k=0,f=0;
		getline(cin,s,'\n');
		vector<int>cnt(26);
		for(i=0;i<s.size();i++)
		cnt[s[i]-'a']++;
		for(i=0;i<26;i++)
		{
			if(cnt[i]!=0)
			k++;
		}
		if(k>=4)
		f=1;
		else if(k==1)
		{
			if(s.size()>=10)
			f=1;
		}
		else if(k==2)
		{
			if(s.size()>=7)
			f=1;
		}
		else if(k==3)
		{
			if(s.size()>=5)
			f=1;
		}
		if(f==1)
		cout<<"YES"<<endl;
		if(f==0)
		cout<<"NO"<<endl;
	}
}
