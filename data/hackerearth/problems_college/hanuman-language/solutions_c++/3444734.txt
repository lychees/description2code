#include<bits/stdc++.h>
using namespace std;
char s[100005];
void mysubstr(char str[], int low, int high)
{
    printf("%.*s", high-low+1, (str+low));
}
int main()
{
   int t,len,i,j;

   scanf("%d",&t);
 getchar();
   while(t--)
   {
      gets(s);
     
      len=strlen(s);
      for(i=len-1,j=len-1;i>=0;i--)
      {
         if(s[i]==' ')
         {
           mysubstr(s,i+1,j);
           printf(" ");
           i--;
           j=i;
         }
      }
      mysubstr(s,0,j);
      printf("\n");
   }

return 0;
}
