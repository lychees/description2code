#include <bits/stdc++.h>
using namespace std;
long long mod = 1000000007;
long long fact(long long n)
{
    long long f = 1;
    for(long long i = 1;i<=n;i++)
    {
        f*=i;
        f = f%mod;
    }
    return f;
}
long long pow_mod(long long a,long long b)
{
    long long p =1;
    while(b)
    {
        if(1&b)
        {
            p*=a;
            p%=mod;
        }
        a*=a;
        a%=mod;
        b=b/2;
    }
    return p;
}

long long c[101][101];
long long d[101];



int main(   )
{
    d[0] = 1;
    d[1] = 0;
    d[2] = 1;

    long long n,l,r;
    cin>>n>>l>>r;
    for(long long i = 3;i<=n;i++)
    {
        d[i] = (((i-1)%mod)*(d[i-1]%mod + d[i-2]%mod))%mod;
    }
    for(long long i = 0;i<=n;i++)
    {
        c[i][0] = 1;
    }
    for(int i = 0;i<=n;i++)
    {
        for(int j = i+1;j<=n;j++)
        {
            c[i][j] = 0;
        }
    }
    for(long long i = 1;i<=n;i++)
    {
        for(long long j = 1;j<=i;j++)
        {
            c[i][j] = (c[i-1][j]%mod+ c[i-1][j-1]%mod)%mod;
        }
    }
    long long P = 0;
    long long Q = pow_mod(fact(n),mod-2);
    for(long long i = l;i<= r;i++)
    {
        P+=(c[n][i]*d[i])%mod;
        P %=mod;


    }

    cout<<(P*Q)%mod<<endl;
}
