ll i,j;
ll N,M;
ll arr[100005];
bool fun(ll K)
{
	
	i=0;
	ll days=0;
	ll sum=0;
	while(i<N)
	{
		sum+=arr[i];
//	cout<<sum<<"<--sumnow"<<days<<"<--days now"<<endl;
		
		if(sum>K)
		{
			sum=arr[i];
			days++;
		}
		else if(sum==K)
		{
			sum=0;
			days++;
		}
		i++;
	}
	if(sum>0)
	days++;
	
//	cout<<days<<"<--days would take for hours:"<<K<<"<--hours now"<<endl;
	if(days>M)
	return 0;
	else
	return 1;
}
int main()
{
	cin>>N>>M;
	ll sumall=0;
	ll minval=0;
	
	for(i=0;i<N;i++)
	{
		cin>>arr[i];
		sumall+=arr[i];
		minval=max(minval,arr[i]);
	}
	
	ll L=minval,R=sumall;
	ll ans=0;
	
	while(L<=R)
	{
	//	cout<<L<<"<--L now"<<R<<"<--R now"<<endl;
		
		ll mid=L+(R-L)/2;
	//	cout<<mid<<"<--mid now"<<endl;
		ll val=fun(mid);
		if(val==0)
			L=mid+1;
		else
		{
			ans=mid;
			R=mid-1;
		}	
	}
	cout<<ans;	
return 0;
}
