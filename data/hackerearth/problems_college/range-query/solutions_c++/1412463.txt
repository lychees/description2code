#include<bits/stdc++.h>
using namespace std;
int main(){
	int n; cin>>n;
	int a[n+1];
	int arr[n+1][101];
	memset(arr,0,sizeof(arr));
	for(int i=0; i<n; i++) cin>>a[i];
	arr[0][a[0]]=1;
	for(int i=1; i<n; i++){
		for(int j=0; j<=100; j++) arr[i][j]=arr[i-1][j];
		arr[i][a[i]]++;
	}
	int q; cin>>q;
	while(q--){
		int l,r,val;cin>>l>>r>>val;
	//	l--; r--;
		if(l==0) cout<<arr[r][val]<<endl;
		else cout<<arr[r][val]-arr[l-1][val]<<endl;
	}
}
