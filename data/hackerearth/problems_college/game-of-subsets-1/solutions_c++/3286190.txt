#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;

long long int powr (long long int a, long long int b)
{
if (b == 0)
return 1;
long long int x = powr(a, b/2);
if (b % 2 == 0)
return (x*x)%mod;
else
return (((x*x)%mod)*a)%mod;
}

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	long long int n;
   	cin>>n;
   	long long p =(powr(3,n));
   	cout<<p<<"\n";
   }
    return 0;
}
