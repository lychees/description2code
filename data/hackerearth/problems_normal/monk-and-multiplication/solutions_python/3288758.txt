n = int(raw_input())
q = map(int, raw_input().split())

max1 = max2 = max3 = None
ans = [-1] * n
for i in range(n):
	if q[i] > max1:
		max3, max2, max1 = max2, max1, q[i]
	elif q[i] > max2:
		max3, max2 = max2, q[i]
	elif q[i] > max3:
		max3 = q[i]
	if i >= 2:
		ans[i] = max1 * max2 * max3

for e in ans:
	print e
