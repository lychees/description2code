#include<bits/stdc++.h>
#define ll long long int
#define md 1000000009
using namespace std;
ll mulmod(ll a,ll b,ll c)
{
	ll x=0,y=a%md;
	while(b>0)
	{
		if(b%2==1)
		{
			x=(x+y)%md;
		}
		y=(2*y)%md;
		b=b/2;
	}
	return x%md;
	
}
ll power(ll a,ll b,ll c)
{
	ll x=1,y=a;
	while(b>0)
	{
		if(b%2==1)
		{
			x=mulmod(x,y,md);
		}
		y=mulmod(y,y,md);
		b=b/2;
	}
	return x%md;
}
int main()
{
	ll t,n,i,j,ans,temp;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		ans=0;
		if(n==1 || n==0)
		{
			printf("%lld\n",n);
		}
		else
		{
			ans=((power(10,n,md))-(power(8,n,md)));
			if(ans<0)
			{
				ans+=md;
			}
			ans=(ans*power(2,md-2,md))%md;
			printf("%lld\n",ans);
		}
	}
	return 0;
}
