tc = input()
for _ in range(tc):
    N, M = map(int, raw_input().split(" "))
    x1, y1 = map(int, raw_input().split(" "))
    x2, y2 = map(int, raw_input().split(" "))
    p = input()
    X = min(abs(x2 - x1), abs(N - x2) + abs(x1), abs(x2) + abs(N - x1))
    Y = min(abs(y2 - y1), abs(M - y2) + abs(y1), abs(y2) + abs(M - y1))
    cost = (X + Y) * p
    print cost
    if cost > 1000:
        print "Let go of the group."
    else:
        print "You saved the group."
