#include <bits/stdc++.h>
#define ll long long
using namespace std;
void input(int a[],int n)
{
    for(int i=0;i<n;i++)
        cin>>a[i];
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s,sr;
        cin>>s;
        int n=s.length();
        sr=string(s.rbegin(),s.rend());
        bool f=1;
        for(int i=1;i<n;i++)
        {
            if(abs(int(s[i]-s[i-1]))!=abs(int(sr[i]-sr[i-1])))
            {
                f=0;
                break;
            }
        }
        if(f)
            cout<<"Funny"<<endl;
        else
            cout<<"Not Funny"<<endl;
    }
    return 0;
}
