#include <iostream>
#include<stdio.h>
using namespace std;
void sol()
{
	int a[26],i,n;
	char str[1000];
	for(i=0;i<26;i++) a[i]=0;
	cin>>n;
	fflush(stdin);
	for(i=0;i<n;i++)
	{
		fflush(stdin);
		cin>>str;
		a[(int)(str[0]-'a')]++;
	}
	for(i=0;i<26;i++) 
	{
		if(a[i]>1) 
		{
			cout<<"NO"<<endl;
			return ;
		}
	}
	cout<<"YES"<<endl;
}
int main()
{
	int t;
	cin>>t;
	while(t--) sol();
	return 0;
}
