#include <bits/stdc++.h>
using namespace std;
int a[100005];
int main(){
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	int n;
	cin >> n;
	int m = 0;
	for(long long i=0;i<n;i++){
		cin >> a[i];
		m = max(m,a[i]);
	}
	int b= 0 ;
	int l=0;
	for(int i=0;i<n;i++){
		if(a[i]==m){
			int cur = 1;
			while(a[i]==m){
				cur++;
				i++;
			}
			if(cur>l){
				l=cur;
				b = i-1;
			}
		}
	}
	l=b;
	int r=b;
	cout << "1";
	int pr[n+1],sf[n+1];
	pr[0]=0;
	sf[n-1]=0;
	for(int i=1;i<n;i++) pr[i]=max(pr[i-1],a[i-1]);
	for(int i=n-2;i>=0;i--) sf[i]=max(sf[i+1],a[i+1]);
	int cur = a[b];
	for(int i=0;i<n-1;i++){
		//cout << l << " " << r << "\n";
		if(pr[l] > sf[r] || (pr[l]==sf[r] && a[l-1]>a[r+1])){
			l--;
			cur = min(cur,a[l]);
		}
		else{
			r++;
			cur = min(cur,a[r]);
		}
		if(cur >= max(pr[l],sf[r])) cout << "1";
		else cout << "0";
	}
	return 0;
}
