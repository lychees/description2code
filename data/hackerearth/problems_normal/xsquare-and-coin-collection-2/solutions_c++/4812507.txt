#include <bits/stdc++.h>
#define ll long long int
using namespace std;

ll dp[100005];
ll a[100005];
int main()
{
	ll n, k, t;
	cin >> t;
	while (t--) {
		cin >> n >> k;
		for (ll i = 0; i < n; i++) {
			cin >> a[i];
		}
		
		memset(dp, 0, sizeof(0));
		ll mx = INT_MIN;
		if (a[0] <= k) {
			dp[0] = a[0];
		}
		
		mx = max(mx, dp[0]);
		for (ll i = 1; i < n; i++) {
			if (a[i] <= k) {
				dp[i] = dp[i - 1] + a[i];
			} else {
				dp[i] = 0;
			}
			mx = max(mx, dp[i]);
		}
		cout << mx << endl;
	}
	
	return 0;
}
