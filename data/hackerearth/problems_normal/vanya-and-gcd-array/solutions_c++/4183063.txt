#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll dp[501][501][101];
const ll mod = 1e9+7;
ll gcd(ll a,ll b)
{
	if(a==-1)
	return b;
	return (!b)?(a):gcd(b,a%b);
}
ll func(int arr[],int n, int idx, int prev_idx, int g)
{
	
	if(prev_idx!=-1 && dp[idx][prev_idx][g]!=-1)
	return dp[idx][prev_idx][g];
	if(idx==n)
	{
		// cout << "Here, g:"<< g <<"\n";
		if(g==1)
		{
			//cout << "Returning 1\n";
			return 1;
			
		}
		else
		return 0;
	}
	// donot pick up					//pickup
	ll aa = func(arr,n,idx+1,prev_idx,g);
	ll bb = 0;
	bb = ((prev_idx==-1)||(arr[idx]>arr[prev_idx]))?(func(arr,n,idx+1,idx,gcd(g,arr[idx]))):(0);
	// cout << "idx: " << idx << " prev_idx: " << prev_idx << " g: " << g << " aa: " << aa << " bb: " << bb <<"\n";
	if(prev_idx!=-1)
	dp[idx][prev_idx][g] = (aa+bb)%mod;
	return (aa + bb)%mod;
	
}
int main()
{
	memset(dp,-1,sizeof(dp));
    int n;
    cin >> n;
    int arr[n];
    for(int i = 0; i<n; i++)
    	cin >> arr[i];
    cout << func(arr,n,0,-1,-1);
    return 0;
}
