using namespace std;

#define ll long long

ll prime_factors_s(ll n) {
	set<ll> factors;
	ll d = 2;
	while (n > 1) {
		while (!(n % d)) {
			factors.insert(d);
			n /= d;
		}
		++d;
		if (d*d > n) {
			if (n > 1) factors.insert(n);
			break;
		}
	}
	return accumulate(factors.begin(), factors.end(), (ll)0);
}

int main()
{
	ll N, K; cin >> N >> K;
	ll s = 0;
	for (ll i = 1; i <= N; i++) {
		if (prime_factors_s(i) == K) {
			++s;
		}
	}
	cout << s << endl;
	return 0;
}
