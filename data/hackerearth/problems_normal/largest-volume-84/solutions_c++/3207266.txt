#include<iostream>
#include<iomanip>
#include<math.h>
using namespace std;
double pi = 3.1415926535897;
int main()
{
	double r2;
	double s;
	cin >> r2 >> s;
	double r1 = r2;
	long double h = s;
	double temp;
	double vol = pi*r2*r2*h;
	double tempvol = 0.0;
	for (; (r1 - r2) <= s; r1 += 0.000001)
	{
		temp = (double)(s*s) - ((r1 - r2)*(r1 - r2));
		h =(long double) sqrtl(temp);
		//cout << h << endl;
		tempvol = (long double)pi*((r1*r1) + (r2*r2) + (r1*r2))*h;
		tempvol = (double)(tempvol / 3);
		if (tempvol > vol)vol = tempvol;
		//cout << tempvol << endl;
	}


	cout << fixed << setprecision(2) << vol << endl;
	//	int volumemax=
	cin >> r2;
}
