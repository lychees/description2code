vector <ll> g[10005];
ll dp[10005],n;
ll dfs(ll x)
{
	if(x==n)
	{
		return 1;
	}
	if(dp[x]!=-1)
	{
		return dp[x];
	}
	//cout<<x<<endl;
	ll sz = g[x].size();
	ll val = 0;
	F(i,0,sz-1)
	{
		val += dfs(g[x][i]);
	}
	return (dp[x]=val);
}

{
	std::ios::sync_with_stdio(false);
