T = input()
for i in xrange(T):
    A = list(raw_input())
    B = list(raw_input())
    C = Cr = A+B
    C = "".join(C)
    D = Dr = B+A
    D = "".join(D)
    Cr.reverse()
    Cr = "".join(Cr)
    Dr.reverse()
    Dr = "".join(Dr)
    if C == Cr or D == Dr:
        print "YES"
    else:
        print "NO"
