template <class T>
void scan(T& t) {
	t = 0;
	char c;
	c=getchar_unlocked();

	while(c<'0' || c>'9')
		c=getchar_unlocked();

	while(c>='0' && c<='9') {
		t=(t<<3)+(t<<1)+c-'0';
		c=getchar_unlocked();
	}
}

ll y[100005];

int main() {

	int t; scan(t);
	ll n, w, i, temp, sol;


	while(t--) {

		scan(w); scan(n);
		y[0] = 0; sol = 0;
		sol += (n*w);

		for(i = 1; i <= n; i++) {
			scan(temp);
			if(temp == 1) {
				y[i] = 2*i;
			}
			else if(temp == 2) {
				y[i]  = y[i-1]+1;
			}
			sol += (y[i] - y[i-1]);
		}
		cout << sol << endl;
	}





}
