test = int(raw_input())
while(test>0):
	s1 = raw_input()
	s2 = raw_input()
	flag = 0
	for each in s1:
		if(each in s2):
			flag = 1
			break
	if(flag == 1):
		print "Yes"
	else:
		print "No"
	test-=1
