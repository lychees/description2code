import sys
t = int(sys.stdin.readline())
while t > 0:
    s1 = (sys.stdin.readline()).strip()
    s2 = (sys.stdin.readline()).strip()
    if len(s1) == len(s2):
        l = len(s1)
    elif len(s1) < len(s2):
        l = len(s1)
    elif len(s2) < len(s1):
        l = len(s2)
    (i, m) = (0, 0)
    while i < l:
        if s1[i] == s2[i]:
            m += 1
        i += 1
    print m
    t -= 1
