#include <iostream>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
   int t,n,k,cnt1=0,cnt2=0;
   cin>>t;
   for(int i=0;i<t;i++)
   {
   		cin>>n>>k;
   		if(n>60)
   		cout<<"NA"<<endl;
   		else 
   		{
   			
   			
   					for(int j=1;j<=n-2;j++)
   					{cnt1=j*12;}
   					for(int j=0;j<=k;j++)
   					{
   						cnt2=j*2;
   					}
   					cout<<cnt1+cnt2+50<<endl;
   					cnt1=0;
   					cnt2=0;
   				
   		}
   }
}
