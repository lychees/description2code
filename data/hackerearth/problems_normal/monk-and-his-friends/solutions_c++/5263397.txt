#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,m;
    	cin>>n>>m;
    	vector<long long int> v(n+m);
    	//vector<int>v1;
    	
    	for(int i=0;i<n+m;i++)
    	cin>>v[i];
    	set<long long int> s;
    	for(int i=0;i<n;i++)
    	{
    		s.insert(v[i]);
    	}
    	
    	for(int i=n;i<n+m;i++)
    	{
    		if(s.find(v[i])!=s.end())
    		cout<<"YES"<<endl;
    		else
    		cout<<"NO"<<endl;
    		s.insert(v[i]);
    	}
  
  
    	
    
    	
    }
    
    return 0;
}
