using namespace std;

int fun(int x, LL y){
    int ans = 1;
    if(x == 0){
        return 0;
    }
    while(y--){
        ans *= x;
    }
    ans %= 10;
    return ans;
}

int main(){
    int test;
    LL x, y;
    cin>>test;
    while(test--){
        sd(x);
        sd(y);
        x %= 10;
        if(y % 4 == 0){
            y = 4;
        }
        else{
            y %= 4;
        }

        printf("%d\n", fun(x, y));
    }
    return 0;
}
