#include<iostream>
using namespace std;
int gcd(int a,int b)
{
	int r;
while(b!=0)
{
	r=a%b;
	a=b;
	b=r;
}
return a;
}
int prime(int n)
{
	if(n==1)
	{
		cout<<"FALSE\n";
	}
	else{
	
int flag=0;
    for(int i=2;i<n;i++)
       {
		 if(n%i==0)
        {
		cout<<"FALSE\n";
		flag=1;
		break;
	}
    }
    if(flag==0)
    {
    	cout<<"TRUE\n";
	}
}
	
 
}
int main()
{
int t,n,i;
cin>>t;
int* counter=new int[t];
for(int i=0;i<t;i++)
counter[i]=0;
 
int* a=new int[t];
for(i=0;i<t;i++)
    cin>>a[i];
for(i=0;i<t;i++)
{
    n=a[i];
    for(int j=1;j<=n;j++)
   {
    if(gcd(j,n)==1)
        counter[i]++;
}
}
 
for(i=0;i<t;i++)
    {
    prime(counter[i]);
    }
 
return 0;
}
