#include<iostream>
using namespace std;

int main() {



int n;
int c=0;
cin>>n;
int t=2*n;
int a[t];
for (int p=1;p<=t;p++)
    cin>>a[p];
for(int h=2;h<=t;h=h+2)
{ c=0;
    for (int j=1;j<=t;j=j+2)
       {
             if (a[h]==a[j])
               c++;


        }

cout<<(n-1+c)<<" "<<(n-1-c)<<endl;
   

}
return 0;
}
