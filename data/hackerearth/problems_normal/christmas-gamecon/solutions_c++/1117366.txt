#include <iostream>
#include <vector>
using namespace std;
long long int arr[1000000];
int main()
{    
    int T,Test_cases,i,elem,Ans;
    cin>>T;
	int N;
    for(Test_cases=0;Test_cases<T;Test_cases++)
    {
    	cin>>N;
    	for(i=0;i<N;i++)
		{
    		cin>>elem;
			arr[i]=elem%5;
		}
    	Ans=0;
    	for(i=0;i<N;i++)
    		Ans=Ans^arr[i];
    		if(Ans)
    			cout<<"Aniruddha";
    		else 
    			cout<<"Andrew";
    	cout<<endl;
    	
    }
    return 0;
}
