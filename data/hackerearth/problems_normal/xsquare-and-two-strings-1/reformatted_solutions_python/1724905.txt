def main(str1, str2):
    s1len = len(str1)
    s2len = len(str2)
    found = False
    checked_string = {}
    if s2len <= s1len:
        for ch in str2:
            if ch not in checked_string and ch in str1:
                found = True
                break
            else:
                checked_string[ch] = ''
        if found is True:
            print 'Yes'
        else:
            print 'No'
    else:
        for ch in str1:
            if ch not in checked_string and ch in str2:
                found = True
                break
            else:
                checked_string[ch] = ''
        if found is True:
            print 'Yes'
        else:
            print 'No'


if __name__ == '__main__':
    t = int(raw_input())
    for i in xrange(t):
        s1 = raw_input()
        s2 = raw_input()
        main(s1, s2)
