const ll mod = 1000000007;
ull pow_mod(ull base,ull power){
	ull res = 1;
	while(power){
		if(power&1)
			res = (res * base)%mod;
		base = (base * base)%mod;
		power >>= 1;
	}
	return res;
}
ull fact[1000* 1000 + 10];

	}

}
