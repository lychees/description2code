#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int t,sume=0,sumo=0;
    cin>>t;
    while(t--)
    {
    	long long int r,c;
    	cin>>r>>c;
    	if(r<=c)
    	{
    		if(r%2==0)
    		{
    			sume=(r*(2+r))/4;
    		//	cout<<sume<<"*";
    			sumo=(r*r)/4;
    		//	cout<<" "<<sumo<<"#";
    		}
    		else
    		{
    			sume=((r-1)*(1+r))/4;
    			sumo=((r+1)*(r+1))/4;
    		}
    	}
    	else
    	{
    		if(r%2==0 && c%2!=0)
    		{
    			sume=((r-c+1)*c)/2+((c-1)*(1+c))/4;
    		//	cout<<sume<<"*";
    			sumo=((r-c-1)*c)/2+((c+1)*(c+1))/4;
    		//	cout<<" "<<sumo<<"#";
    		}
    		else if(r%2==0 && c%2==0)
    		{
    			sume=((r-c)*c)/2+(c*(2+c))/4;
    			sumo=((r-c)*c)/2+(c*c)/4;
    		}
    		else if(r%2!=0 && c%2!=0)
    		{
    			sume=((r-c)*c)/2+((c-1)*(1+c))/4;
    			sumo=((r-c)*c)/2+((c+1)*(c+1))/4;
    		}
    		else
    		{
    			sume=((r-c-1)*c)/2+(c*(2+c))/4;
    			sumo=((r-c+1)*c)/2+(c*c)/4;
    		}
    	
    	}
    	cout<<sume<<" "<<sumo<<endl;
    	sume=0;sumo=0;
    }
    return 0;
}
