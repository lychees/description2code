#include <iostream>
#include <stdio.h>
#include <string>
using namespace std;

void reverse(string &str, int low, int high) {
	while(low < high) {
		char temp = str[low];
		str[low] = str[high];
		str[high] = temp;
		low++;
		high--;
	}
}

int main()
{
    int T;
    scanf("%d", &T);
    while(T--) {
    	string str;
    	cin >> str;
    	int len = str.length();
    	int i;
    	for(i=len-1; i > 0; i--) {
    		if(str[i-1] < str[i])
    			break;
    	}
    	if(i == 0)
    		printf("no answer\n");
    	else {
    		//cout << i << endl;
    		int large = i;
    		for(int j=i+1; j < len; j++) {
    			if(str[j] > str[i-1] && str[j] < str[large])
    				large = j;
    			//cout << large << endl;
    		}
    		char temp = str[large];
    		str[large] = str[i-1];
    		str[i-1] = temp;
    		reverse(str, i, len-1);
    		cout << str << endl;
    	}
    }
    return 0;
}
