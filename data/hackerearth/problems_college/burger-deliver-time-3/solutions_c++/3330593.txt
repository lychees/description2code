#include<bits/stdc++.h>
using namespace std;

int main() {
	ios_base::sync_with_stdio(0);
	
	int n;
	cin >> n;
	int t, d, s;
	vector<int> arr(n, 0);
	for(int i = 0; i<n; ++i) {
		cin >> t >> d;
		s = t+d;
		arr[i] = s;
	}
	
	int mn = 0, save;
	int i = 0;
	while(i < n) {
		mn = INT_MAX;
		for(int i = 0; i<arr.size(); ++i) {
			if(arr[i] < mn) {
				mn = arr[i];
				save = i;
			}
		}
		cout << save + 1 << " ";
		arr[save] = INT_MAX;
		i++;
	}
	
	
	return 0;
}
