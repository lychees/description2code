#include <bits/stdc++.h>
using namespace std;


int main()
{
	int t;
    cin>>t;
    while(t--)
    {
    	string w;
    	cin>>w;
    	string r;
    	cin>>r;
    	
    	sort(r.begin(),r.end());
    	int x=0;
    	for(int i=0;i<w.size();i++)
    		if(x<r.size() && r[x]<w[i])
    		{
    			w[i]=r[x];
    			x++;
    		}
    			
    	cout<<w<<endl;
    }
    return 0;
}
