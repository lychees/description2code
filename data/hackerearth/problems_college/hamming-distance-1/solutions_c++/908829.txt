#define FOR(i,a,b) for(int i=a;i<b;i++)
#define DRT()     int t;cin>>t;while(t--)
#define gcd(a,b)  __gcd(a,b)
#define MOD       1000000007

int main()
{
      inp_s;
      DRT()
      {
            LL a,b;
            cin>>a>>b;
            vector<int> bin_a,bin_b;
            while(a)
            {
                  bin_a.pb(a%2);
                  a/=2;
            }
            while(b)
            {
                  bin_b.pb(b%2);
                  b/=2;
            }
            int ans = 0;
            FOR(i,0,min(sz(bin_a),sz(bin_b)))
            if(bin_a[i] != bin_b[i]) ans++;
            if(sz(bin_b) > sz(bin_a))
            {
                  FOR(i,sz(bin_a),sz(bin_b)) ans += bin_b[i];
            }
            else
            {
                  FOR(i,sz(bin_b),sz(bin_a)) ans += bin_a[i];
            }
            cout<<ans<<endl;
      }
      return 0;
}
