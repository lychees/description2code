#include <string.h>
#include <stdio.h>
#include<iostream>
using namespace std;
int dec(int p)
{
    int a=1;
	while(p!=97)
	{
		p--;
		a++;
	}
	return a;
}
int main()
{
    char str[1001];
    int s=0;
    scanf("%s",str);
    int l=strlen(str);
    for(int i=0;i<l;i++)
    {
    	int b=str[i];
    	s=s+dec(b);
    }
    cout << s;
    return 0;
}
