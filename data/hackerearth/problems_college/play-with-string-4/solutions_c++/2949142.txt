#include <iostream>
#include<string.h>
#include<stdio.h>
#include<algorithm>
using namespace std;

int main()
{

     char a[100000],b[100000];int n,i,f,j,diff1,diff2;
     cin>>n;
     while(n--)
     {
     	cin>>a;f=1;
     	for(i=strlen(a)-1,j=0;i>=0;--i,j++)
     	{
     		b[j]=a[i];
     	}
     	for(i=1;i<strlen(a);++i)
     	{diff1=a[i]-a[i-1];
     	diff2=b[i]-b[i-1];//cout<<diff1<<" "<<diff2<<"\n";
     		if((abs(diff1)==abs(diff2)))
     		{
     			f=0;
     		}
     		else
     		{
     			f=1;break;
     		}
     	}
     	if(f==0)
     	{
     		cout<<"Funny\n";
     	}
     	else if(f==1)
     	{
     		cout<<"Not Funny\n";
     	}
     }

    return 0;
}
