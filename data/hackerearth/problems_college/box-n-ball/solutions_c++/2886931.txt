#include <iostream>
using namespace std;
#include<stdio.h>
typedef long long LL;

int main()
{
    LL t;
    cin>>t;
    
    while(t--)
    {
    	LL a,b,c;
    	cin>>a>>b>>c;
    	
    	LL t=min(a,min(b,c));
    	
    	long double ans=t/2.0;
    	
    	printf("%.1llf\n",ans);
    }
    return 0;
}
