def main():
    for i in range(input()):
        number = input()
        list1 = map(int, raw_input().split())
        for x in range(number):
            if ((x + 1) != list1[(list1[x] - 1)]):
                print 'not inverse'
                break
        else:
            print 'inverse'


if __name__ == '__main__':
    main()
