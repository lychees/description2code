#include <bits/stdc++.h>
using namespace std;

long long gcd(long long a,long long b)
{
	if(b==0)
	return a;
	else 
	return gcd(b,a%b);
}

int main()
{
	int d;
	scanf("%d",&d);
	while(d--)
	{
		long long x,y,k;
		scanf("%lld%lld%lld",&x,&y,&k);
		long long i,z,count,flag=0;
		z=gcd(x,y);
		count=0;
		for(i=z;i>0;i--)
		{
			if(z%i==0)
			count++;
			if(count==k)
			{
				printf("%lld\n",i);
				flag=1;
				break;
			}
		}
		if(flag==0)
		printf("No crymeth today\n");
	}
	return 0;
}
