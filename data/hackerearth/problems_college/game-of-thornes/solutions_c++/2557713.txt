#include <iostream>
using namespace std;

int main()
{
    string a;
    cin>>a;
    
    int i,c;
    int count[26]={0};
    
    for(i=0;i<a.size();i++){
    	count[a[i]-'a']++;
    }
    
    for(i=0;i<26;i++){
    	if(count[i]%2==1){
    		c++;
    	}
    }
    
    if(c>1){
    	cout<<"NO"<<endl;
    }else{
    	cout<<"YES"<<endl;
    }
    
    return 0;
}
