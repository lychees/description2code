#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

unsigned long long coin_change(int N, int M, vector<int> v)
{
    vector<unsigned long long> table(N+1, 0);
    
    table[0] = 1;
    for (int i=0; i<M; i++) {
        for (int j=v[i]; j<=N; j++) {
            table[j] += table[j-v[i]];
        }
    }
    return table[N];
}

int main() {
    int N, M;
    vector<int> v;
    cin >> N >> M;
    for (int i=0; i<M; i++) {
        int currNum;
        cin >> currNum;
        v.push_back(currNum);
    }
    cout << coin_change(N, M, v) << endl;
    return 0;
}
