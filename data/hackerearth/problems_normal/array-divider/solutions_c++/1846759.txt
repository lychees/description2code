using namespace std;
typedef long long ll;
typedef vector<int> vi;
typedef vector<ll> vl;
typedef vector<vl> vvl;
typedef vector<vi> vvi;
typedef vector<double> vd;
typedef pair<int, int> pii;
typedef pair<double, double> pdd;
typedef vector<pii> vii;
typedef vector<string> vs;

int main() {
    int n,k;
    cin >> n >> k;
    vl a(n), s(n + 1);
    for (int i = 0; i < n; ++i) {
    	cin >> a[i];
    	s[i+1] = s[i] + a[i];
    }
    vl d(n + 1), d2(n + 1);
    for (int i = 0; i < d2.size(); ++i) d2[i] = s[i]*s[i];
    for (int t = 2; t <= k; ++t) {
    	vl nd(d.size(), 1e18);
    	vl nd2(d2.size());
    	for (int i = 0; i < nd.size(); ++i) {
    		for (int j = 0; j <= i; ++j) {
    			ll sum = s[i] - s[j];
    			ll cand = d[j] + sum*sum*(t-1) - 2*sum*s[j] + d2[j];
    			if (nd[i] > cand) {
    				nd[i] = cand;
    				nd2[i] = sum*sum + d2[j];
    			}
    		}
    	}
    	d.swap(nd);
    	d2.swap(nd2);
    }
    cout << d.back() << endl;
    return 0;
}
