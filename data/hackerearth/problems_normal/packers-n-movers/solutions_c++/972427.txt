#include<cstdio>
#include<iostream>
#include<climits>
using namespace std;

int main()
{
    int m,p,i,j,k;
    scanf("%d%d",&m,&p);
    int a[p];
    int dp[p+1][m+1];
    for(i=0;i<p;i++)
        scanf("%d",&a[i]);
    for(i=1;i<p;i++)
    {
        a[i]=a[i-1]+a[i];
    }

    for(i=1;i<p;i++)
        dp[i][1]=a[i-1];

    for(i=1;i<=p;i++)
    {
        for(j=2;j<=m;j++)
        {
            dp[i][j]=INT_MAX;
            for(k=1;k<i;k++)
            {
                dp[i][j]=min(dp[i][j],max(a[i-1]-a[k-1],dp[k][j-1]));
            }
        }
    }

    /*for(i=1;i<=p;i++)
    {
        for(j=1;j<=m;j++)
            cout<<dp[i][j]<<" ";
        cout<<endl;
    }?*/

    printf("%d\n",dp[p][m]);
    return 0;
}
