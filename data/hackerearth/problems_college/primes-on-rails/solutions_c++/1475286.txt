#include <bits/stdc++.h>
using namespace std;

int prime[1000100];
int ans[100010];

void prim()
{
	int k=1000;
	prime[0]=1;
	prime[1]=1;
	int i,j,p,r;
	for(i=2;i<=k;i++)
	{
		j=0;
		for(;j<=1000000;j+=i)
		{
			if(j!=i)
			{
				prime[j]=1;
			}
		}
	}
}

int main()
{
    int h,m,n,s,i,j,k;
    memset(prime,0,sizeof(prime));
    prim();
   // memset(prime,0,sizeof(prime));
    memset(ans,0,sizeof(ans));
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>h>>m>>s;
    	
    	int ti=h*3600 + m*60 + s;
    	
    	if(prime[ti]==0)
    	{
    		for(j=0;j<300;j++)
    		{
    			ans[ti+j]++;
    		}
    	}
    	else if(ti%2==1)
    	{
    		for(j=0;j<180;j++)
    		{
    			ans[ti+j]++;
    		}
    	}
    	else
    	{
    		for(j=0;j<120;j++)
    		{
    			ans[ti+j]++;
    		}
    	}
    }
    
    int ma=0;
    
    for(i=0;i<100010;i++)
    {
    	if(ans[i]>ma)
    	{
    		ma=ans[i];
    	}
    }
    cout<<ma;
    return 0;
}
