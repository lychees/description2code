#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    int N;
    scanf("%d",&N);
    vector<int> A(N);
    for(int i = 0; i < N; i++){
    	scanf("%d",&A[i]);
    }
    sort(A.begin(),A.end(),greater<int>());
    int ans = -1;
    ans = max(ans,max(A[0] - A[1],A[0] - A[2]));
    for(int i = 1; i < N; i++){
    	if(i+2 < N) ans = max(ans,A[i] - A[i+2]);
    	else if(i + 1 < N) ans = max(ans,A[i] - A[i+1]);
    }
    printf("%d",ans);
    return 0;
}
