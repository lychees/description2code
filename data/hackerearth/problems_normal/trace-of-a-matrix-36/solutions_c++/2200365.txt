#include <iostream>
using namespace std;
#define ull unsigned long long

int main()
{
    int tc;
    ull C, R,ans=0;
    cin>>tc;
    while(tc--)
    {
    	ans =0;
    	cin>>C>>R;
    	
    	if(C<R)
    	{
    		ans = (C*(4 + (C-1)*(C+1) + (C-1)*(R+1) ) ) /2;
    	}
    	else
    	{
    		ans = (R*(4 + (R-1)*(C+1) + (R-1)*(R+1) )) /2;
    	}
    	cout<<ans<<endl;
    }
}
