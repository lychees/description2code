#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long r,c;
    	cin>>r>>c;
    	long long ce,co;
    	if(c>=r)
    	{
    		ce=0;co=0;
    		int n=r/2;
    		if(r%2==0)
    		{
    			ce+=(n*(r+2))/2;
    			co+=(n*(1+(r-1)))/2;
    		}
    		else
    		{
    			co+=((n+1)*(r+1))/2;
    			ce+=(n*(2+(r-1)))/2;
    		}
    	}
    	else
    	{
    		ce=0;co=0;
    		int n=c/2;
    		if(c%2==0)
	    	{
	    		ce+=(n*(2+c))/2;
	    		co+=(n*(1+(c-1)))/2;
    		
    		}
    		else
    		{
    			co+=((n+1)*(c+1))/2;
    			ce+=(n*(2+(c-1)))/2;
    		}
    		if(((c+1)%2==r%2)&&(c+1)%2==0)
    		{
    			long long c1=((r-c)*c)-((r-c)/2)*c;
    			ce+=c1;
    			co+=((r-c)*c)-c1;
    		}
    		else
    		if(((c+1)%2==r%2)&&(c+1)%2!=0)
    		{
    			long long c2=((r-c)*c)-((r-c)/2)*c;
    			co+=c2;
    			ce+=((r-c)*c)-c2;
    		}
    		else
    		if((c+1)%2!=r%2)
    		{
    			ce+=((r-c)*c)/2;
    			co+=((r-c)*c)/2;
    		}
    	}
    	cout<<ce<<" "<<co<<endl;
    	}
    	return 0;
    }
