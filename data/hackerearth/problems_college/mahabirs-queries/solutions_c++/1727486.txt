
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int a[4][101][101];
int n, q, x;

void ConverMatrix(){
    for (int k = 1; k < 4; k++)
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                a[k][n - j - 1][i] = a[k - 1][i][j];
}

int main(){
    scanf("%d%d", &n, &q);
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            scanf("%d", &a[0][i][j]);
    ConverMatrix();
    while (q--){
        scanf("%d", &x);
        x %= 360;
        x /= 90;
        if (x) x = 4 - x;
        for (int i = 0; i < n; i++){
            for (int j = 0; j < n - 1; j++)
                printf("%d ", a[x][i][j]);
            printf("%d\n", a[x][i][n - 1]);
        }
        if (q)
            printf("\n");
    }

    return 0;
}
