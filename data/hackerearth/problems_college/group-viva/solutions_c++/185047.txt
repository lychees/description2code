#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <list>
#include <vector>
#include <queue>
#include <cstring>
using namespace std;

int t , n , r , ans ;

int gcd(int a , int b) { return (!b ?a:gcd(b,a%b)) ; }

int main()
{
  scanf("%d",&t) ;
  while(t--)
  {
     scanf("%d%d",&n,&r) ; r-- , ans=0 ;
     for(int i=1 ; i*i<=r ; i++) if(r%i==0)
     {
        ans++ ; int x=r/i ;
        if(i!=x && r%x==0) ans++ ; 
     } 
     ans=n-ans ; 
     
     printf("%d/%d\n",ans/gcd(ans,n),n/gcd(ans,n)) ;   
  }

return 0;
}
