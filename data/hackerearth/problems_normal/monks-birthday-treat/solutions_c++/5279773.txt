#include <iostream>
#include <stack>
#include <cstring>
#include <vector>
using namespace std;
 
#define MAX 100005
#define ll long long

bool visited[1005];
vector <int> adj[1005];

int dfs(int s){
	stack <int> S;
	S.push(s);
	visited[s] = true;
	int friends =1;
	while(!S.empty()){
		int top = S.top();
		S.pop();
		for(int i=0;i<adj[top].size();i++){
			if(visited[adj[top][i]]==false){
				S.push(adj[top][i]);
				visited[adj[top][i]]=true;
				friends++;
			}
		}
	}
	return friends;
}
 
int main()
{
    int t,n,d,a,b,temp;
    int min;
	n=d=a=b=0;
	cin>>n>>d;
	for(int i=0;i<d;i++){
		cin>>a>>b;
		adj[a].push_back(b);
	}
	min = 1005;
	temp =0;
	for(int i=1;i<=n;i++){
		memset(visited,false,sizeof(visited));
		temp = dfs(i);
		if(temp<min)
			min = temp;
	}
	cout<<min<<endl;
}
