#include <iostream>
#include<stdio.h>
#include<string.h>
using namespace std;

int main()
{
  		char s[100];
		cin>>s;
		int d,l,a=0,s1=0;
		l=strlen(s);
		int c= (int)s[0];
		int n;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>s;
			d=(int)s[0];
			if(d==c&& strlen(s)==l)
				s1++;
			else if((d-3)==c&& strlen(s)==l)
				a++;
		}
		cout<<s1<<" "<<a;
}
