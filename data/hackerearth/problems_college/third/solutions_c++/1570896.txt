#include <iostream>
using namespace std;
int max(int a,int b)
{
	if(a>b) return a;
	else return b;
}
int main()
{
	int t,i,n,k;
    cin>>t;
    while(t--)
    {
    	cin>>n;int F[n+1];F[0]=0;
    	cin>>k;F[1]=k;
    	for(i=2;i<=n;i++)
    	{
    		cin>>k;
    		F[i]=max(k+F[i-2],F[i-1]);
    	}
    	cout<<F[n]<<endl;
    }
    return 0;
}
