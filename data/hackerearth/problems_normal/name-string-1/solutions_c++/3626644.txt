#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	unsigned long long int test,a[4],i,j;
	cin >> test;
	
	for(i = 0;i < test;i++)
	{
		for(j = 0;j < 3;j++)
		{
			cin >> a[j];
		}
		sort(a,a+3);
		if(a[2] == a[1])
		{
			cout << a[2] * 2  << endl;
		}
		else if(a[1] == 0)
		{
			cout << "1" <<endl;
		}
		else
		{
			cout << 2 * a[1] + 1 <<endl;
		}
	}
	return 0;
}
