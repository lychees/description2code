#include <iostream>
using namespace std;

int check_palindrome(string str)
{
	int start = 0, end = str.length()-1;
	while(start<end)
	{
		if(str[start]!=str[end])
		return 0;
		
		start++;
		end--;
	}
	return 1;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	string str1,str2;
    	cin>>str1;
    	str2.append(str1);
    	str2.append(str);
    	int count = 0;
    	if(check_palindrome(str2))
    	count++;
    	int n = str.length();
    	for(int i=0; i<n; i++)
    	{
    		string str2;
    		str2.append(str,0,i+1);
    		str2.append(str1);
    		str2.append(str,i+1,n);
    		if(check_palindrome(str2))
    		count++;
    	}
    	cout<<count<<"\n";
    	
    	
    }
    return 0;
}
