#include <algorithm>
#include <bitset>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <ctime>
#include <fstream>
#include <iostream>
#include <map>
#include <queue>
#include <set>
#include <string>
using namespace std;

string p, s, xp[2005], xs[2005], ans = "0", ls;
int n, lenp, lens;
bool boo;
int main() {
  getline(cin, p);
  getline(cin, s);
  scanf("%d", &n);
  lenp = p.size();
  lens = s.size();
  boo = false;
  for (int i = 0; i < lenp; i++)
    xp[i % n] += p[i];
  for (int i = 1; i <= n; i++) {
    for (int j = 0; j < i; j++)
      xs[j].clear();
    for (int j = 0; j < lens; j++)
      xs[j % i] += s[j];
    int now = i - 1;
    ls = "";
    for (int j = n - 1; j >= 0; j--)
      if (now >= 0 && xp[j] == xs[now])
        ls = "1" + ls, now--;
      else
        ls = "0" + ls;
    if (now < 0)
      if (!boo || ls < ans) {
        boo = true;
        ans = ls;
      }
  }
  cout << ans << endl;
  return 0;
}
