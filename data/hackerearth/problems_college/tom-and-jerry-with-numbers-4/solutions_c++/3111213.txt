#include <iostream>
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll buf[1000000][2];



int fun(ll n,ll a,ll b,int pre)
{
	if(n< min(a,b))
	{
		return pre;
	}
	if(n==a || n==b)	return !pre;
	int r1,r2;
	if(n>a)
	{
		if(buf[n-a][!pre]!=-1)	r1= buf[n-a][!pre];
		else r1=fun(n-a,a,b,!pre);
		buf[n-a][!pre]=r1;
		
	}
	if(n>b)
	{
		if(buf[n-b][!pre]!=-1) r2= buf[n-b][!pre];
		else r2=fun(n-b,a,b,!pre);
		buf[n-b][!pre]=r2;
	}
	if(r1==!pre || r2==!pre)
		return !pre;
	else
		return pre;
}



int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	ll n,a,b;
    	cin>>n>>a>>b;
    	memset(buf,-1,sizeof(buf));
    	cout<<fun(n,a,b,0)<<endl;
    }
    return 0;
}
