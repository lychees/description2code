#include <iostream>
using namespace std;

int main()
{
    string str;
    cin>>str;
    int n=str.size(),i,j,k,count=0;
    for(i=0;i<n;i++)
    {
    	for(j=i,k=i;j>=0 && k<n ;j--,k++)
    	{
    		if(str[j]==str[k])
    		count++;
    		else
    		break;
    	}
    }
    for(i=0;i<n;i++)
    {
    	for(j=i,k=i+1;j>=0 && k<n ;j--,k++)
    	{
    		if(str[j]==str[k])
    		count++;
    		else
    		break;
    	}
    }
    cout<<count<<endl;
    return 0;
}
