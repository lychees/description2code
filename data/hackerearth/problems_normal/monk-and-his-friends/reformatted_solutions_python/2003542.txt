try:
    t = int(input())
    while t > 0:
        t -= 1
        n, m = map(int, raw_input().split())
        l = map(int, raw_input().split())
        ln = l[:n]
        lm = l[n:]
        d = {}
        for x in ln:
            if x not in d:
                d[x] = 1
        for x in lm:
            if x not in d:
                print "NO"
                d[x] = 1
            else:
                print "YES"
except BaseException:
    pass
