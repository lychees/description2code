#include <iostream>
#define gc getchar_unlocked
using namespace std;


void scanint(unsigned int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

int main()
{
   unsigned int N,k,ans=0;
   cin>>N;
   for(unsigned int i=0;i<N;i++)
   {	
   	 scanint(k);
   	if(k<300000000)ans++;
   }
   
   cout<<ans;
    return 0;
}
