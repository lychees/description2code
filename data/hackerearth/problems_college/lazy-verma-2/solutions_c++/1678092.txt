#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main() 
{
    ll n,m;
    cin>>n>>m;
    ll a[n],b[m],i,f,s;
    f = 0;
    for(i = 0; i < n; i++) {
        cin>>a[i];
        f += a[i];
    }
    s = 0;
    for(i = 0; i < m; i++) {
        cin>>b[i];
        s += f*b[i];
    }
    cout<<s<<endl;
	return 0;
}
