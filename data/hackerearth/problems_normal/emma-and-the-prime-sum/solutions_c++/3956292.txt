#include<bits/stdc++.h>
using namespace std;

bool prime[1000000]={false};
long long int sum[1000000];

void soe() {
	long long int i,j;
	prime[0]=prime[1]=true;
	for(i=2;i<=1005;i++) {
		if(!prime[i])
		for(j=i*i;j<=1000000;j+=i) {
			prime[j]=true;
		}
	}
}

int main() {
	long long int i,j,n;
	soe();
	sum[0]=0;
	for(i=1;i<=1000000;i++) 
	{	
		if(!prime[i]) {
			sum[i]=i+sum[i-1];
		}
		else
			sum[i]=sum[i-1];
	}

	cin >> n;
	while(n--) {
		cin >> i >> j;
		cout << sum[j]-sum[i-1] << endl;
	}
	
	return 0;
}
