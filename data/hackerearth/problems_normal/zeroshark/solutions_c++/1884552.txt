lli a[100010],b[100010],c[100010],d[100010],dp[100010];


int main()
{
	lli t,i,j,k;
	cin>>t;
	a[1]=1;
	a[2]=1;
	b[1]=1;
	b[2]=2;
	d[2]=0;
	c[2]=1;
	for(i=3;i<=100000;i++)
	{
		a[i]=b[i-1]%mod;
		b[i]=(b[i-1]+b[i-2])%mod;
		c[i]=(b[i-2]+d[i-1])%mod;
		d[i]=(d[i-1]+c[i-1])%mod;
		dp[i]=(c[i]+d[i])%mod;
	}
	while(t--)
	{
		int n;
		cin>>n;
		if(n==1)
			cout<<"0"<<endl;
		else if(n==2)
			cout<<"1"<<endl;
		else if(n==3)
			cout<<"2"<<endl;
		else
		{
			cout<<dp[n]%mod<<endl;
		}
	}
	return 0;
}
