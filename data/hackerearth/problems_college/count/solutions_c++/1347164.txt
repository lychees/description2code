#include <iostream>
using namespace std;

int main()
{	int test;
	scanf("%i",&test);
	while(test-->0)
	{	long long k1,k2;
		long long evencount,oddcount;
		scanf("%llu%llu",&k1,&k2);
		long long even_row = k2/2 * 2 ;
		long long odd_row = k2;
		if(k2%2==0)odd_row=odd_row-1;
		if(k1<=even_row)
		{	long long even=k1/2;
		
			evencount = (long long)even*(even+1);
			
		}
		else if(k1>even_row)
		{	evencount = (long long)(even_row/2)*(even_row/2+1);
			long long diff = (k1-even_row)/2;
			evencount += (long long)diff*k2;
		}
		if(k1<=odd_row)
		{	long long prog=  (k1+1)/2;
			oddcount=  prog*(prog+1) - prog;
		}
		else if(k1>odd_row)
		{	oddcount= (long long)((odd_row+1)/2)*((odd_row+1)/2+1) -  ((odd_row+1)/2);
		
			long long  diff =  (k1 - odd_row)/2;
			oddcount += (long long)diff * k2;
		}
		cout<<evencount<<" "<<oddcount<<endl;
	}
}
