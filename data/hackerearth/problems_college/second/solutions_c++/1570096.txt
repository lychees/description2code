#include<bits/stdc++.h>
using namespace std;
int main() {
	long n,m,ans=0,q=0;
	scanf("%ld %ld",&n,&m);
	long a[n];
	for(long i=0;i<n;i++) {
		scanf("%ld",&a[i]);
	}
	sort(a,a+n);
	for(long i=0;i<n;i++) {
		long x=a[i]/m;
		if(x==(q+1) && a[i]%m==0) {
			q=x;
			ans=a[i];
		}
	}
	printf("%ld",ans);
	return 0;
}
