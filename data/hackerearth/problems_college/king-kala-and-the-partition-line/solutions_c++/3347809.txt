#include <iostream>
using namespace std;

int main()
{
    int n,t;
     float a,b,x,i,c,y,s,ans;
    cin>>t;
    while(t--)
    {
    	long g=0,h=0;
    	cin>>n;
    	
    	cin>>a>>b>>c;
    	for(i=0;i<n;i++)
		{
		cin>>x>>y;
    	s=(x*a)+(y*b)+c;
    	if(s>0)
    		g++;
    	else if(s<0)
    		h++;
    }
		    ans=g>h?g:h;
    		cout<<ans<<"\n";
    }
    
    return 0;
}
