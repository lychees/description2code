#include <iostream>
using namespace std;
bool getprime(int n)
{
	int c=0;
	for(int i=2;i<=n;i++)
	{
		if(n%i==0)
		{
			c++;
			while(n%i==0)
			{
				n/=i;
			}
		}
	}
return c>=3;
}


int main()
{
	int t,n;
	cin>>t;
	
	for(int i=0;i<t;i++)
	{
		int ct=0;
		cin>>n;
		for(int j=30;j<=30000;j++)
		{
			if(getprime(j))
			{
				ct++;
			}
			
			if(ct==n)
			{
				cout<<j<<endl;
				break;
			}
			else
		
				continue;
		}
		
	}
}
