def numofdiv(number):
    temp = []
    a = int(number**0.5)+1
    for i in range(1, a):
        if number % i == 0:
            if int(number/i) != i:
                # add=add+(number/i)
                temp.append(number/i)
            temp.append(i)
        if len(temp) > 3:
            return temp
    return temp


for _ in xrange(input()):

    n = input()
    if len(numofdiv(n)) > 3:
        print "YES"
    else:
        print "NO"
