#include <bits/stdc++.h>
using namespace std;

void tc()
{
	unordered_map<int, int> c;
	int n, v = 0;
	cin >> n;
	for (int i = 0; i < n; i++) {
		int a, b;
		cin >> a >> b;
		c[a]++;
		if (!c[b])
			v++;
		else
			c[b]--;
	}
	cout << v << '\n';
}

int main()
{
	ios::sync_with_stdio(false), cin.tie(nullptr);
	int t;
	cin >> t;
	for (int i = 0; i < t; i++)
		tc();
}
