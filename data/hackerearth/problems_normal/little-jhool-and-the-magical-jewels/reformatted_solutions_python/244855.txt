def rubies(string):
    if "r" not in string or "u" not in string or "b" not in string or "y" not in string:
        return 0
    rloc = string.find("r")
    string = string[:rloc] + string[rloc + 1:]
    uloc = string.find("u")
    string = string[:uloc] + string[uloc + 1:]
    bloc = string.find("b")
    string = string[:bloc] + string[bloc + 1:]
    yloc = string.find("y")
    string = string[:yloc] + string[yloc + 1:]
    return 1 + rubies(string)


for i in range(input()):
    print rubies(raw_input())
