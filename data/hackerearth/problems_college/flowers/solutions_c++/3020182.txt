#include<bits/stdc++.h>
using namespace std;
int main(){
   int n , k;
   cin >> n >> k;
    int a[n+1];
   for(int i = 0 ; i < n ; ++i )
     cin >> a[i];
     sort(a,a+n);
     int ans = 0;
     priority_queue< int , vector<int> , greater<int> > Q;
     int id = n - 1;
     for(int i = 0 ; i < k ; ++i ){
     ans = ans + a[id];
     id--;
     Q.push(1);
     }
     for(int i = id ; i >= 0 ; --i ){
      int val = Q.top() + 1;
      Q.pop();
      ans = ans + val*a[i];
      Q.push(val);
     }
     cout << ans << endl;


}
