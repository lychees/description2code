#include <iostream>
#include <vector>

using namespace std;
int main() {

  string N;
  string S;
  cin >> N >> S;
long long arr[12345];
  long long NC = 0;
  long long NO = 0;
  long long NW = 0;
  for (int i = 0; i < S.size(); i++) {
    if (S[i] == N[0]) {
      NC++;
    } else if (S[i] == N[1]) {
      NO += NC;
    } else if (S[i] == N[2]) {
      NW += NO;
    }
  }
	for(int i=0;i<S.size();i++)
	{
		if(S[i]==N[0])
		arr[0]++;
		else
		{
			for(int j=1;j<N.size();j++)
			{
				if(S[i]==N[j])
				{
					arr[j]+=arr[j-1];
					break;
				}
			}
		}
	}
  cout << arr[N.size()-1] << endl;
  return 0;
}
