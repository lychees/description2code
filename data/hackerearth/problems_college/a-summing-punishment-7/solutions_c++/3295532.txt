#include <iostream>
#include <cstring>
#include <cmath>
using namespace std;
int main()
{
 long int arr[100001],ans[100001],n,q;
 int a,b,i,j;
 cout<<"";
 cin>>n;
 for(i=1;i<=n;i++)
 {
  cin>>arr[i];
 }
 cout<<"";
 cin>>q;
 for(i=1;i<=q;i++)
 {
  cin>>a;
  cin>>b;
  ans[i]=0;
  for(j=a;j<=b;j++)
  {
   ans[i]=ans[i]+arr[j];
  }
 }
 for(i=1;i<=q;i++)
 {
  cout<<ans[i]<<endl;
 }
 return 0;
}
