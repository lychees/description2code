using namespace std;

const int MAXN = 1000006;

const int ALPH = 256;

char s[MAXN];

int p[MAXN], c[MAXN], cnt[MAXN], pn[MAXN], cn[MAXN];
int phi[MAXN], plcp[MAXN], lcp[MAXN];

int main() {
  int k, n;
  scanf("%d %d", &k, &n);

  if (k == 1) {
    printf("%d", n);
    return 0;
  }

  scanf("%s", s);
  s[n++] = '$';

  for (int i = 0; i < n; i++) {
    cnt[s[i]]++;
  }
  for (int i = 1; i < ALPH; i++) {
    cnt[i] += cnt[i - 1];
  }
  for (int i = 0; i < n; i++) {
    p[--cnt[s[i]]] = i;
  }
  c[p[0]] = 0;
  int classes = 1;
  for (int i = 1; i < n; i++) {
    if (s[p[i]] != s[p[i - 1]]) {
      classes++;
    }
    c[p[i]] = classes - 1;
  }

  for (int h = 0; (1 << h) < n; h++) {
    for (int i = 0; i < n; i++) {
      pn[i] = p[i] - (1 << h);
      if (pn[i] < 0) {
        pn[i] += n;
      }
    }
    for (int i = 0; i < classes; i++) {
      cnt[i] = 0;
    }
    for (int i = 0; i < n; i++) {
      cnt[c[pn[i]]]++;
    }
    for (int i = 1; i < classes; i++) {
      cnt[i] += cnt[i - 1];
    }
    for (int i = n - 1; i >= 0; i--) {
      p[--cnt[c[pn[i]]]] = pn[i];
    }
    cn[p[0]] = 0;
    classes = 1;
    for (int i = 1; i < n; i++) {
      const int mid1 = (p[i] + (1 << h)) % n;
      const int mid2 = (p[i - 1] + (1 << h)) % n;
      if (c[p[i]] != c[p[i - 1]] || c[mid1] != c[mid2]) {
        ++classes;
      }
      cn[p[i]] = classes - 1;
    }
    for (int i = 0; i < n; i++) {
      c[i] = cn[i];
    }
  }

  phi[p[0]] = -1;
  for (int i = 1; i < n; i++) {
    phi[p[i]] = p[i - 1];
  }

  for (int i = 0, l = 0; i < n; i++) {
    if (phi[i] < 0) {
      plcp[i] = 0;
      continue;
    }
    while (s[i + l] == s[phi[i] + l]) {
      l++;
    }
    plcp[i] = l;
    l = max(l - 1, 0);
  }
  for (int i = 0; i < n; i++) {
    lcp[i] = plcp[p[i]];
  }

  int ans = 0;

  deque<pair<int, int>> w;
  for (int i = 2; i < n; i++) {
    while (!w.empty() && w.back().first >= lcp[i]) {
      w.pop_back();
    }
    w.push_back({lcp[i], i});

    while (w.front().second <= i - (k - 1)) {
      w.pop_front();
    }

    if (i - 1 >= k - 1) {
      ans = max(ans, w.front().first);
    }
  }

  printf("%d", ans);

  return 0;
}
