#include<iostream>
#include<cstdio>
#define gc getchar_unlocked

void scanint(long long int &x)
{
    register long long int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

int main()
{
	long long int t,n,m,temp1,temp2,temp3;
	scanint(t);
	while(t)
	{
			t--;
			scanint(n);
			scanint(m);
			temp1=n+m-1;
			temp2=temp1-n;
			temp3=temp1-m;
			printf("%lld %lld\n",temp2,temp3);
	}	
	return 0;
}
