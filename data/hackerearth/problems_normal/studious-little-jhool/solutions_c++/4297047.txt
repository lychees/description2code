#include <bits/stdc++.h>
using namespace std;
#define inf INT_MAX
int main(){

	int b,t,n,ans;
	cin >> t;
	while(t--){
		cin >> n;
		ans  = inf;
		for(int a=0;a<=(n/10);a++){
			// cout << "value of a is " << a << endl;
			if((n-10*a)%12==0){
				b = (n-10*a)/12;
				// cout << "value of b is " << b << endl;
				ans  = min(ans,a+b);
			}
		}
		if(ans == inf)
			cout << "-1" << endl;
		else cout << ans << endl;
	}
	return 0;
}
