#include <iostream>
using namespace std;

int main()
{
	int t,i,j;
	long long ans;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>s;
		i=0;
		j=s.size()-1;
		ans=1;
		while(i<j)
		{
			if(s[i]!=s[j])
				break;
			i++;
			j--;
		}
		if(i!=j)
		{
			for(i=0;i<s.size();i++)
				ans*=(s[i]-'a'+1);
			cout<<ans<<endl;
		}
		else
			cout<<"Palindrome"<<endl;
	}
    return 0;
}
