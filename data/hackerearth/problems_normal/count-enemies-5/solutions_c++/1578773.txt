#include <iostream>
#include <string.h>
using namespace std;

int main()
{
   int t;
   char s[100001];
   cin>>t;
   
   while(t--)
   {
   	int arjit=0,enemy=0,survivors=0;
   	cin>>s;
   	for(int i=0;i<strlen(s);i++)
   	{
   		
   		if(s[i]=='X')arjit++;
   		if(s[i]=='O')enemy++;
   		if(s[i]=='*')
   		{
   			if(arjit==0)
   			survivors=survivors+enemy;
   			arjit=0;
   			enemy=0;
   		}
   	}
   	if(arjit==0)
   	survivors+=enemy;
   	cout<<survivors<<endl;
   }
}
