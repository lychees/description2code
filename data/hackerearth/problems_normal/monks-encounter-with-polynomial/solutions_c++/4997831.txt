#include <iostream>
using namespace std;

int main()
{
	int T;
    cin >> T;
    while(T-- > 0){
    	long long int A, B,C, K;
    	cin >> A >> B >> C >> K;
    	if(C >= K){
    		cout << 0 << endl;
    	}else{
    		long long int e = 1;
    		while(A*e*e + B*e +C < K){
    			e  = e << 1;
    		}
    		long long int s = 1;
 			while(e > s+1){
 				long long int m = s + (e-s)/2;
 				if(A*m*m + B*m +C >= K){
 					e = m;
 				}else{
 					s = m;
 				}
 			}
 			cout << e << endl;
    	}
    }
    return 0;
}
