#include <iostream>
using namespace std;

int main()
{
    int diff=10000,t, n, h, hg[1000],k;
    cin>>t;
    for(int i=0;i<t;i++){
    	cin>>n>>h;
    	for(int j=0;j<n;j++){
    		cin>>hg[j];
    		if(hg[j]<=h){
    			k=h-hg[j];
    		} else {
    			k=hg[j]-h;
    		}
    		if(k<diff){
    			diff=k;
    		}
    	}
    	cout<<diff<<endl;
    	diff=10000;
    }
    return 0;
}
