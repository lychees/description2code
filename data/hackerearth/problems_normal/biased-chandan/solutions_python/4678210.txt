n=input()
arr=[0]*n
i=0
while i<n:
	arr[i]=input()
	i+=1

j=0
sum1=0
while j<n:
	if j==0 and arr[j]==0:
		j+=1
	elif j==0 and arr[j]!=0:
		sum1+=arr[j]
		j+=1
	elif j!=0 and arr[j]==0:
		k=j
		while k>=0:
			if arr[k]!=0:
				red=arr[k]
				arr[k]=0
				break
			else:
				k-=1
		sum1=sum1-red
		j+=1
	elif j!=0 and arr[j]!=0:
		sum1+=arr[j]
		j+=1

print sum1
