#include<bits/stdc++.h>
using namespace std ;

int main()
{   int test ;
    cin>>test;
    while(test--)
    {
       char inp[102];
       scanf("%s",&inp);
       int le = strlen(inp);
       for(int i = 0 ; i< le ; i++)
       {
           if( inp[i]+i > 122 )
           {
               inp[i]=((inp[i]+i)%122)+96;
           }
           else
           inp[i]+=i;

       }
       cout<<inp;
       cout<<endl;



    }
    return 0 ;
}
