#include<bits/stdc++.h>
using namespace std;
int main(){
    int N,T;
    cin>>T;
    while(T--){
        cin>>N;
        int temp=5;
        int zeros=0;
        while(temp<=N){
            zeros+=N/temp;
            temp*=5;
        }
        cout<<zeros<<endl;
    }
    return 0;
}
