#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include<string.h>
using namespace std;
float area(int x1, int y1, int x2, int y2, int x3, int y3)
{
   float ans=((x1*(y2-y3) + x2*(y3-y1)+ x3*(y1-y2))/2.0);
   if(ans<0)
   return -(ans);
   else
   return ans;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int x1,y1,x2,y2,x3,y3,x4,y4;
		cin>>x1>>y1>>x2>>y2>>x3>>y3>>x4>>y4;
		float A=area(x1,y1,x2,y2,x3,y3);
		float A1=area(x4,y4,x2,y2,x3,y3);
		float A2=area(x1,y1,x4,y4,x3,y3);
		float A3=area(x1,y1,x2,y2,x4,y4);
		if(A==A1+A2+A3)
		cout<<"INSIDE"<<endl;
		else
		cout<<"OUTSIDE"<<endl;
	}
}
