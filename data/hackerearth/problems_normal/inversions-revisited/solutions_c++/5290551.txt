#include <bits/stdc++.h>
using namespace std;

const int MAX = 1e6;
int BIT[MAX + 1];
int a[2000001];

void update(int k) {
	while (k) {
		BIT[k]++;
		k -= k & -k;
	}
}

int getSum(int k) {
	int ret = 0;
	while (k <= MAX) {
		ret += BIT[k];
		k += k & -k;
	}
	return ret;
}

int main() {
	//freopen("input.txt", "rt", stdin);
	int n = 0; scanf("%d", &n);
	for (int i = 0; i < n; ++i) scanf("%d", a + i);
	int val = 0;
	long long res = 0;
	for (int i = 0; i < n; ++i) {
		scanf("%d", &val);
		res += getSum(val + 1);
		update(a[i]);
	}
	printf("%lld", res);
	return 0;
}
