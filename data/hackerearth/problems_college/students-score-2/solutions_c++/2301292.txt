#include<iostream>

using namespace std;
typedef long long ll;
int main(){
	ll max=0;
	ll n;
	cin>>n;
	vector<ll> v;
	for(ll i=0;i<n;i++){
		ll temp;
		cin>>temp;
		v.push_back(temp);
	}
	ll max_so_far=0;
	for(ll i=0;i<n;i++){
		max_so_far+=v[(int)i];
		if(max_so_far<0){
		//	max_so_far-=v[(int)i];
			
			max_so_far=0;
		}
		if(max<max_so_far){
				max=max_so_far;
			}
	}
	cout<<max<<endl;
}
