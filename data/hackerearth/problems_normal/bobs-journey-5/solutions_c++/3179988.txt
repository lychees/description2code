#include <bits/stdc++.h>
using namespace std;

#define tcase int tc; cin >> tc; while(tc--) 
#define boost ios_base::sync_with_stdio(false);
#define rep(i,n) for(int i=0; i<n; i++)

int main()
{
	string a;
	int t;
	bool flag;
    tcase{
    	map<char,int> hash;
 		cin >> t;
 		flag = true;
 		rep(i,t){
 			cin >> a;
 			if(flag){
 				hash[a[0]]++;
 				if(hash[a[0]]>1) flag=false;
 			}
 		}
 		if(flag) cout << "YES" << endl;
 		else cout << "NO" << endl;
    }
}
