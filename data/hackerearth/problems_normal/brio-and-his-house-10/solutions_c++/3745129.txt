#include <iostream>
#include <iomanip>
#include <math.h>
using namespace std;

int main()
{
	int noofcases;
	cin>>noofcases;
	for(int i=0;i<noofcases;i++)
	{
		double A,B,C;
		double pi = 3.1415;
		cin>>A>>B>>C;
		double temp = A*B*C;
		double temp1=(A+B+C);
		double temp2 =(B+C-A);
		double temp3 =(C+A-B);
		double temp4=(A+B-C);
		double deno= temp1*temp2*temp3*temp4;
		double denom = sqrt(deno);
		double radius = temp/denom;
		double area = pi*radius*radius;
		cout << fixed << setprecision(4) << area << endl;
		
	}
   
    return 0;
}
