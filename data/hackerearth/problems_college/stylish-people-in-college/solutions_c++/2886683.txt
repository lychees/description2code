#include<bits/stdc++.h>
using namespace std;


int main()
{

	int t;
	cin>>t;
	int k=1;
	while(t--)
	{
		int n;
		cin>>n;
		int *arr = new int[n];
		for(int i=0; i<n; i++)
		cin>>arr[i];
		int s1=0, s2=arr[0],tmp;
		for(int i=1; i<n; i++)
		{

            tmp = max(s1, s2);
			s2 = arr[i] + s1;
			s1 = tmp;
		}
      int val = max(s1, s2);
		cout<<"Case "<<k<<": "<<val<<endl;
		k++;
		delete arr;
	}
	return 0;
}
