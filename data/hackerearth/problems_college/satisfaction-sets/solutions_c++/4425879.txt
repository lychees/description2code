#include <iostream>
#include <stdio.h>
#include <vector>
#include <string>
#include <queue>
#include <algorithm>
#include <map>
#include <set>
#include <cassert>
#include <cmath>

#define si(n) scanf("%d",&n)
#define sd(n) scanf("%lf",&n)
#define sl(n) scanf("%lld",&n)
#define ss(str) scanf("%s",str)
#define FOR(i,a,b) for(int i = a; i<b; i++)
#define pb push_back
#define mp make_pair

typedef long long int ll;
typedef unsigned long long int ull;

using namespace std;

int main()
{
    int t;
    si(t);
    int t1 = t;
    int cnt[505] = {0};

    int uni=0,diff = 0;

    int n;
    while(t--)
    {
        set<int> S;
        si(n);
        FOR(i,0,n)
        {
            int tmp;
            si(tmp);
            S.insert(tmp);
        }

        for(set<int>::iterator it = S.begin();it != S.end(); it++)
            cnt[*it]++;
    }

    FOR(i,0,501)
    {
        if(cnt[i] != 0)
        {
            uni+=i;
            if(cnt[i] != t1)
                diff+=i;
        }

    }
    cout<<diff - uni;


    return 0;
}
