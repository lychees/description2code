#include <bits/stdc++.h>
using namespace std;
#define MAX 1000001
#define LL long long
int t[MAX]={0};
LL a[MAX],i=0;
int main()
{
    LL n=0,q=0,x=0,y=0;
    cin>>n>>q;
    for(i=1;i<=n;i++)
    {
        cin>>a[i];
        t[i]=a[i]+t[i-1];
    }
    while(q--)
        {
            cin>>x>>y;
            cout<<(t[y]-t[x-1])<<" "<<(a[y]-t[y-1]+t[x-1])<<endl;
        }
    return 0;
}
