#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		string s1, s2;
		cin >> s1 >> s2;
		int j, n1 = s1.size(), n2 = s2.size();
		int hash1[26] = {0}, hash2[26] = {0};
		for(j = 0; j < n1; j++)
		{
			hash1[s1[j] - 97]++;
		}
		for(j = 0; j < n2; j++)
		{
			hash2[s2[j] - 97]++;
		}
		int flag = 0;
		for(j = 0; j < 26; j++)
		{
			if(hash1[j] > 0 && hash2[j] > 0)
			{
				cout << "Yes\n";
				flag = 1;
				break;
			}
		}
		if(flag == 0)
		{
			cout << "No\n";
		}
	}
}
