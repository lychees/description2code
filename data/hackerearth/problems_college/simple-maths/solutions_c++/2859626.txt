#include<bits/stdc++.h>
using namespace std;
int gcd(long long int a, long long int b)
{
    if (a == 0)
        return b;
    return gcd(b%a, a);
}

int lcm(long long int a, long long int b)
{
    return (a*b)/gcd(a, b);
}
int main()
{
    long long int n,ans;
    long long int t,a,b,c;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld%lld%lld",&n,&a,&b);
        if((a==0 && b==0) || a==0 || b==0)
            printf("-1\n");
        else
        {
        c=lcm(a, b);
        ans=n/a +n/b -n/c;
        printf("%lld\n",ans);
        }
    }
    return 0;
}
