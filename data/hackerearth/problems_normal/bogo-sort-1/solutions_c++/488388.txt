#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
   int n;
   cin>>n;
   vector<long long>nums;
   for(int i=0;i<n;i++)
   {
		long long a;
		cin>>a;
		nums.push_back(a);
   }
   sort(nums.begin(),nums.end());
   for(long long num : nums)
   {
   		cout<<num<<"\n";
   }
}
