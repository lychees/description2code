#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    std::ios::sync_with_stdio(false);
    int n,a;
    long int sum=0;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	cin>>a;
    	sum+=a;
    }
    if(sum/2==n-1)
    cout << "Yes" << endl;
    else
    cout<<"No"<<endl;
    return 0;
}
