def con(a):
  n=len(a);
  st="";
  for i in range(0,n):
  	p=ord(a[i])+i;
  	if(p>122):
  		p=p-122;
  		st+=str(chr(96+p));
  	else:
  		st+=str(chr(p));
  return st;
n=int(raw_input(""));
