inputs = raw_input()
inputs = inputs.split()
inputs = [int(i) for i in inputs]
row = inputs[0]
col = inputs[1]
matrix = []
row_sum = []
col_sum = [0 for i in range(0,col)]
for i in range(0, row):
	row_value = raw_input()
	row_value = row_value.split()
	row_value = [int(val) for val in row_value]
	row_sum.append(sum(row_value))
	for i in range(0, col):
		col_sum[i] = col_sum[i] + row_value[i]
	matrix.append(row_value)

pair_xor = []
for rsum in row_sum:
	for csum in col_sum:
		pair_xor.append(rsum^csum)
print max(pair_xor)
