
T = int(raw_input())
results = []

for t in range(T):
    N = int(raw_input())
    coords_x = []
    coords_y = []
    for n in range(N):
        x, y = map(float, raw_input().split())
        coords_x.append(x)
        coords_y.append(y)
    X = sum(coords_x) / len(coords_x)
    Y = sum(coords_y) / len(coords_y)
    results.append("(%0.2f , %0.2f)" % (round(X, 2), round(Y, 2)))

for result in results:
    print result
