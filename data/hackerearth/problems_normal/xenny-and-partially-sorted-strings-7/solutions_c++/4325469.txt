#include <iostream>
#include <cstring>
#include <algorithm>
#include <string>
#include <map>
#define  sc pair<string,string>
#define boost std::ios::sync_with_stdio(0)
using namespace std;
int main(int argc, char const *argv[])
{
	boost;
	int t,n,k,m;
	std::vector<sc> v;
	string str,str1;
	cin >> t;
	while(t--)
	{
		cin >> n >> k >> m;
		v.clear();
		sc temp;
		while(n--)
		{
			cin >> str;
			str1 = str.substr(0,m);
			temp = make_pair(str,str1);
			v.push_back(temp);
		}
		stable_sort(v.begin(),v.end(),[](const sc &a,const sc &b)->bool{return (a.second < b.second);});
	std::cout << v[k-1].first << std::endl;
	}
	return 0;
}
