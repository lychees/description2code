#define MAXN 100005

int n,q;

ll ans[11][MAXN];

void precalc(){
	//calc ans1
	
	FOR(K,1,10)
	{
		FOR(pos,1,K)
		{
			int L = pos;
			int R = n;

			while(L<=R){
				ans[K][L]+=(ll)ans[K][L-K]+(ll)arr[L];
				L+=K;
			}
		}
	}
}

int main() {
	RESET(ans,0LL);

	int L,R,K;
	scanf("%d%d",&n,&q);
	REP(i,n)
