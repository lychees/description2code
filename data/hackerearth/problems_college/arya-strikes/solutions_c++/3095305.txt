#include<bits/stdc++.h>
using namespace std;
int marked[52][52],maze[52][52];
int result = 0,n;

void dfs(int x,int y)
{
//	cout << x << " " << y << endl;
	if(marked[x][y] == 1)
		return;
		
	if(x == n-1 and y == n-1)
		result = 1;
	
	marked[x][y] = 1;
	if(marked[x+1][y] == 0 and x+1 < n and maze[x+1][y] == 1)
		dfs(x+1,y);
		
	if(marked[x][y+1] == 0 and y+1 < n and maze[x][y+1] == 1)
		dfs(x,y+1);
}

int main()
{
	int t;
	cin >> t;
	
	while(t--)
	{
		cin >> n;
		
		memset(maze,0,sizeof(maze));
		
		for(int i = 0 ; i < n ; ++i)
			for(int j = 0 ; j < n ; ++j)
				cin >> maze[i][j];
			
		result = 0;
		memset(marked,0,sizeof(marked));
		
		dfs(0,0);
		
		if(result == 1)
		{
			cout << "POSSIBLE\n";
		}
		else
		cout << "NOT POSSIBLE\n";
	}
	
	return 0;
}
