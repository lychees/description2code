#include <bits/stdc++.h>
using namespace std;

bool palli(string s)
{
	int l=s.length();
	for(int i=0;i<l;i++)
	if(s[i]!=s[l-1-i])return false;
	
	return true;
	
}
int main() {
	
	int t;
	cin>>t;
	while(t--)
	{
		int c=0;
		string a,b,tmp;
		cin>>a>>b;
		tmp=a;
		int l=a.length();
		for(int i=0;i<=l;i++)
		{
			a.insert(i,b);
			if(palli(a))c++;
			a=tmp;
		}
		cout<<c<<endl;
			
	}
	return 0;
}
