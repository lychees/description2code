#include <iostream>
#include<string.h>
using namespace std;

int main()
{
           int t,n,count;
           int a[109],d[107],i,j;
           cin>>t;
           while(t--)
           {
           	  cin>>n;
           	  count=0;
           	  memset(d,0,sizeof(d));
           	  for(i=0;i<n;i++)
           	  {
           	  	 cin>>a[i];
           	  	 d[a[i]%10]++;
           	  }
           	  for(i=0;i<107;i++)
           	  {
           	  	if(d[i]>1)
           	  	 count+=(d[i]-1);
           	  }
           	   cout<<count<<endl;
           }
    return 0;
}
