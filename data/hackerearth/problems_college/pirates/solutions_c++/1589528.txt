#include <iostream>
#include <cmath>

using namespace std;

int main(void) {
	int n;
	cin >> n;
	int a[n];
	for(int i = 0; i < n; i++)
		cin >> a[i];
	int i = 0;
	int j = 0;
	int m = 0;
	while(i < n) {
		int temp = 1;
		while(j < n-1 && abs(a[j]-a[j+1]) == 1) {
			j++;
			temp++;
		}
		if(temp > m) m = temp;
		if(temp == 1) {
			i = j;
			j = j+1;
		}
		else
			i = j;
	}
	cout << m << endl;
	return 0;
}
