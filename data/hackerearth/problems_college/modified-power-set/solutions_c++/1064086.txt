#include <iostream>
#include<cstdio>
using namespace std;

char str[1000010];

int main()
{
    int t,n,i,j,l,c[256]={0};
    
    scanf("%d",&t);
    
    while(t--)
    {
    	scanf("%s",&str);
    	
    	for(i=0;i<256;i++)
    	{
    		c[i]=0;
    	}
    	
    	l=0;
    	for(i=0;str[i]!='\0';i++)
    	{
    		c[str[i]]++;
    		if(c[str[i]]==1)
    		l++;
    	}
    	
    	printf("%d\n",l*(l+1)/2);
    }
    
    
    return 0;
}
