#define SIZE 200010
const ll mod = 1000000007L;

using namespace std;
typedef pair<ll, ll>pll;
int ar[SIZE];
int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt", "r", stdin);
#endif
	ios_base::sync_with_stdio(0); cin.tie(0);
	int t;
	cin >> t;
	while (t--)
	{
		ll n, m, z;
		cin >> n >> m >> z;
		for (int i = 0; i < n; i++)
		{
			cin >> ar[i];
		}
		ar[n] = 0;
		for (int i = n - 1; i >= 0; i--)
		{
			if (i % 2 == (n - 1) % 2)
			{
				ar[n] = (ar[n] + ar[i] + m) % m;
			}
			else
			{
				ar[n] = (ar[n] - ar[i] + m) % m;
			}
		}
		ll temp = z - n;
		ll cons = (temp) / (n + 1);
		if (cons * (n + 1) != temp)
		{
			cons++;
		}
		temp = z - cons * (n + 1);
		ll ans = 0;
		if (n % 2 == 0 && (cons % 2 != 0))
		{
			ans = (-ar[temp] + m) % m;
		}
		else if (n % 2 == 0 && cons % 2 == 0)
			ans = ar[temp];
		else
		{
			ans = ar[temp];
		}
		cout << ans << endl;
	}

	return 0;
}
