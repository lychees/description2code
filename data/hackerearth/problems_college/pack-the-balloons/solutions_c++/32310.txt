#include <stdio.h>
#include <iostream>

using namespace std;

int main()
{
    int t;
    long long int l,w,h;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%lld %lld %lld",&l,&w,&h);
    	long long int min=l;
    	if(min>w)
    	min=w;
    	if(min>h)
    	min=h;
    	if(min%2==0)
    	printf("%lld.0\n",(min/2));
    	else
    	printf("%lld.5\n",(min/2));

    }
}
