#include <iostream>
using namespace std;

long hcf(long a, long b)
{
	long dend, sor;
	if(a>b)
	{
		dend = a;
		sor = b;
	}
	else
	{
		dend = b;
		sor = a;
	}
	while(dend%sor!=0)
	{
		long temp = dend%sor;
		dend = sor;
		sor = temp;
	}
	return sor;
}

int main()
{
	int t;
	cin>>t;
    while(t--)
    {
    	long a,b,c,d,k;
    	cin>>a>>b>>c>>d>>k;
    	long ay=hcf(a,b);
    	long go=hcf(c,d);
    	long lcm=(ay*go)/hcf(ay,go);
    	k/=lcm;
    	cout<<2*k+1<<endl;
    }
    return 0;
}
