#include <iostream>
#include<stdlib.h>
#include<string>
#include<cstdio>
#include<bits/stdc++.h>
using namespace std;
 
 
int main()
{
int s;
cin>>s;
while(s--)
{
string x,y;
cin>>x;
cin>>y;
int c=0;
int maxi=INT_MAX;
 
 
if(x.size()>y.size())
{
string temp=x;
x=y;
y=temp;
 
}
for(int i=0;i<=(y.size()-x.size());i++)
{
c=0;
for(int j=0;j<x.size();j++)
{
if(x[j]!=y[j+i])
c++;
}
if(c<maxi)
maxi=c;
}
 
cout<<maxi<<endl;
}
 
return 0;
}
