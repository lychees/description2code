#include <iostream>
#include <cstdio>
#define N 1000000
using namespace std;
long long max(long long x, long long y)
{ return (y > x)? y : x; }

long long  maxSubArraySum(int a[], long long size)
{
   long long max_so_far = a[0], i;
   long long  curr_max = a[0];
 
   for (i = 1; i < size; i++)
   {
        curr_max = max(a[i], curr_max+a[i]);
        max_so_far = max(max_so_far, curr_max);
   }
   return max_so_far;
}
 
int arr[N+2];
int main()
{
   long long n;
   int t;
   scanf("%d",&t);
   while(t--)
   {
   	 scanf("%lld",&n);
   	 for(long long i=0;i<n;i++)
   	 scanf("%d",&arr[i]);
   	 
   	 long long max=maxSubArraySum(arr,n);
   	 printf("%lld\n",max);
   	
   }
   return 0;
}
