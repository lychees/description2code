#include <bits/stdc++.h>
using namespace std;
#define ll long long

long double a[1005][1005][3];
int ans[1005];

int main() {
	int t;
	scanf("%d",&t);
	while(t--){
		int n,m;
		scanf("%d %d",&n,&m);
		for(int i=0;i<n;i++){
			for(int j=0;j<m;j++){
				for(int k=0;k<3;k++){
					scanf("%Lf",&a[i][j][k]);
				}
			}
		}
		long double price,temp,x;
		int pos=0;
		for(int i=0;i<n;i++){
			x=1000000005.00;
			for(int j=0;j<m;j++){
				price=1000000000.00;
				temp=1000000000.00;
				for(int k=0;k<3;k++){
					temp=(long double)(price)*(long double)(a[i][j][k]);
					temp=(long double)(temp)/(long double)(100.00);
					price=price-temp;
					//cout<<price<<" ";
				}
				//cout<<endl;
				if(x>price){
					x=price;
					pos=j;
				}
				//cout<<"here!"<<x<<" "<<pos<<endl;
			}
			ans[i]=pos+1;
		}
		for(int i=0;i<n;i++){
			cout<<ans[i]<<" ";
		}
		cout<<endl;
	}
	return 0;
}
