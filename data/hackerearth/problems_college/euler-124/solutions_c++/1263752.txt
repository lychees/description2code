#include<bits/stdc++.h>
using namespace std;
vector<int>v;
int a[100010],k;
vector<pair<int,int> >v1;
void sieve()
{
	long long i,j;
	for(i=2;i<=100000;i++)
	{
		if(a[i]==0)
			for(j=i*i;j<=100000;j+=i)
				a[j]=1;
	}
	for(i=2;i<=100000;i++)
		if(a[i]==0)
		{
			
			v.push_back(i);
		}	
	k=v.size();		
	
}
int main()
{
	sieve();
	int t,i,j,n;
	//cout<<k<<endl;
	for(i=1;i<=100000;i++)
	{
		a[i]=1;
		j=0;
		int m=sqrt(i),p=i;
		while(j<k&&v[j]<=m&&p!=1)
		{
			if(p%v[j]==0)
			{
				a[i]*=v[j];
				while(p%v[j]==0)
					p=p/v[j];
				
			}
			//cout<<i<<" "<<p<<endl;
			j++;
		}
		if(p!=0)
		a[i]*=p;
		//scout<<i<<" "<<a[i]<<endl;
		v1.push_back(make_pair(a[i],i));
	}
	sort(v1.begin(),v1.end());
	scanf("%d",&t);
	//cout<<'{';
	while(t--)
	{
		scanf("%d",&n);
		//cout<<"hi"<<endl;
		printf("%d\n",v1[n-1].second);
		//cout<<v1[n-1].second<<",";
	}
	//cout<<"};";
	return 0;
}
