bool check(string ss)
{
    fori(i,0,ss.size()) {
        if(ss[i] != ss[ss.size()-i-1]) return false;
    }
    return true;
}

V(V(ll)) mat, ans;


int n, m;
int a[1000000];

V(V(ll)) mul(V(V(ll)) v1, V(V(ll)) v2)
{
    V(V(ll)) vv(n*m);
    fori(i,0,n*m){
        vv[i].resize(n*m);
    }
    fori(i,0,n*m) fori(j,0,n*m) fori(k,0,n*m) {
        vv[i][j]=(vv[i][j]+(v1[i][k]*v2[k][j])%MOD)%MOD;
    }
    return vv;
}


V(V(ll)) solve(ll n)
{
    if(n == 1) return mat;
    if(n % 2 == 1) {
        V(V(ll)) vv = solve(n-1);
        vv = mul(mat, vv);
        return vv;
    }
    else {
        V(V(ll)) vv = solve(n/2);
        vv = mul(vv,vv);
        return vv;
    }
}

    S(n), S(m);
    string s;
    fori(i,0,n) {
        fori(j,0,m) {
            I(s);
            map<string, int> mm;
            fori(k,0,s.size()) {
                string ss = "";
                fori(l,k,s.size()) {
                    ss += s[l];
                    if(check(ss)) mm[ss]++;
                }
            }
            a[i*m + j] = mm.size();
        }
    }

    int t;

    S(t);
    mat.resize((n*m));
    fori(i,0,n*m) mat[i].resize(n*m);
    int x, y, xx, yy;
    fori(i,0,t) {
        S(x), S(y), S(xx), S(yy);
        mat[x*m + y][xx*m + yy]++;
        mat[xx*m + yy][x*m + y]++;
    }

    fori(i,0,m*n) mat[i][i] = 1;
    int d;
    S(d);
    ans = solve(d);

    fori(i,0,n*m) {
        ll val = 0;
        fori(j,0,n*m) {
            val += ans[i][j]*a[j];
            val %= MOD;
        }
        if(i && i % m == 0) p(endl);

    }
    return 0;
}
