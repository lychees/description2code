#include <bits/stdc++.h>
using namespace std;
int main(){ 
  long long  int n,m ,i,j,A[1000000],N[1000000],max1,c,min1;
    cin>>n;
    for(i=0;i<n;i++){
        cin>>N[i];
    }
    for(i=0;i<n;i++){
        A[i]=1;
    }
    
    for(i=1;i<n;i++){
        
        for(j=0;j<i;j++){
            if(abs(N[i])>abs(N[j]) && N[i]*N[j]<0){ 
                
               A[i]=max(A[i],(A[j]+1));
                 
            }
        }
    }
    max1=A[0];
    for(i=0;i<n;i++){
        if(A[i]>max1){
            max1=A[i];
        }
    }
     cout<<max1;   
       return 0;

    
}
