from math import sqrt
for _ in range(int(raw_input())):
	a,b,c = map(float,raw_input().split())
	s=(a+b+c)/2;
	area=sqrt(s*(s-a)*(s-b)*(s-c));
	r=(a*b*c)/(4*area);
	plot=3.1415*r*r;
	print "%.4f"%plot
