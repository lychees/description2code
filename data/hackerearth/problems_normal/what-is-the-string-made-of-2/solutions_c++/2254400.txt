#include <iostream>
#include <string.h>
using namespace std;

	for(int i=0;i<strlen(a);i++){
		if(a[i]=='0')
		c+=6;
		else if(a[i]=='1')
		c+=2;
		else if(a[i]=='2')
		c+=5;
		else if(a[i]=='3')
		c+=5;
		else if(a[i]=='4')
		c+=4;
		else if(a[i]=='5')
		c+=5;
		else if(a[i]=='6')
		c+=6;
		else if(a[i]=='7')
		c+=3;
		else if(a[i]=='8')
		c+=7;
		else if(a[i]=='9')
		c+=6;
	}
	cout<<c;
	return 0;
}
