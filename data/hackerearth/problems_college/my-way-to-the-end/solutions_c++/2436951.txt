#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k;
    	cin>>n>>k;
    	int a[n][n];
    	string s;
    	for(int i=0;i<n;i++)
    	{
    		cin>>s;
    		for(int j=0;j<n;j++)
    		{
    			if(s[j]=='1')
    				a[i][j]=1;
    			else
    				a[i][j]=0;
    		}
    	}
    	for(int i=1;i<n;i++)
    	{
    		if(a[0][i]!=0)
    		{
    			if(a[0][i-1]==0)
    				a[0][i]=0;
    			else
    				a[0][i]=a[0][i-1]+1;
    		}
    			
    	}
    	for(int i=1;i<n;i++)
    	{
    		if(a[i][0]!=0)
    		{
    			if(a[i-1][0]==0)
    				a[i][0]=0;
    			else
    				a[i][0]=a[i-1][0]+1;	
    		}
    		
    	}
    	for(int i=1;i<n;i++)
    	{
    		for(int j=1;j<n;j++)
    		{
    			if(a[i][j]!=0)
    			{
    				if(a[i-1][j]==0 && a[i][j-1]==0)
    					a[i][j]=0;
    				else if((a[i-1][j]!=0) && (a[i][j-1]!=0))
    					a[i][j]=min(a[i-1][j],a[i][j-1])+1;
    				else if(a[i-1][j]==0)
    					a[i][j]=a[i][j-1]+1;
    				else
    					a[i][j]=a[i-1][j]+1;
    			}
    		}
    	}
    	
    	if(a[k-1][k-1]==0)
    		cout<<-1<<endl;
    	else
    		cout<<a[k-1][k-1]-1<<endl;
    }
    return 0;
}
