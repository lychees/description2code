#include <stdio.h>

int a[26];
typedef unsigned long long ll;
ll sum = 0;

void paintingCost(ll n)
{
	n = 12 * n;
	ll res = 0;

	ll m = n / 26;

	res = m*sum + m * (m - 1) * 13;

	for (int i = 0; i < n % 26; i++)
	{
		res += a[i] + m;
	}

	printf("%llu\n", res);
}


int main()
{
	int t;
	scanf("%d", &t);

	while (t--)
	{
		ll n;
		scanf("%lld", &n);

		sum = 0;
		for (int i = 0; i < 26; i++)
		{
			scanf("%d", &a[i]);
			sum += a[i];
		}

		paintingCost(n);
	}
}
