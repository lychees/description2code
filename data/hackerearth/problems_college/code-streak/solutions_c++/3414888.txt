#include <iostream>
#include <algorithm>
#include <cmath>

using namespace std;

int main()
{
    int t, n, s, r;
    
    cin>>t;
    while(t--)
    {
    	bool problems[1000001];
    	fill(problems, problems+1000001, false);
    	
    	cin>>n;
    	int count = 0, ans = 0;
    	for(int i = 0; i<n; i++)
    	{
    		cin>>s>>r;
    		if(r == 0)
    		{
    			ans = max(ans, count);
    			count = 0;
    		}
    		else
    		{
    			if(!problems[s])
    				count++;
    			problems[s] = true;
    		}
    		
    	}
    	ans = max(ans, count);
    	cout<<ans<<endl;
    }
    
    
    return 0;
}
