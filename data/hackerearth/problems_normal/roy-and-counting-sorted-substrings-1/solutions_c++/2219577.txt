	int i,j,n;
	long long count;
	char last;
	long long re;
	cin>>T;
	while(T--)
	{
		cin>>n;
		cin>>s;
		count=0;
		last='z'+1;
		re=0;
		for(i=0;i<n;i++)

			if(s[i]<last)
			{
				re+=count*(count+1)/2;

			}

			{
				count++;
			}
			last=s[i];
		}
		re+=count*(count+1)/2;
		cout<<re<<endl;
	}
