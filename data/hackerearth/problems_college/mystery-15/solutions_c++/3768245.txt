#include <bits/stdc++.h>

using namespace std;

inline long long power(long long base,long long exp)
{
	long long ans=1;
    while(exp)
	{
        if(exp%2==1)
			ans=(ans*base);
       	base=(base*base);
       	exp/=2;
    }
    return ans;
}

int main()
{
    std::ios::sync_with_stdio(false);
    long long T, k, a, b, c;
	cin >> T;
	while(T--)
	{
		cin >> k >> a >> b >> c;
		cout << power(a+b+c, k) << endl;
	}
    return 0;
}
