#include <iostream>
#include<map>
using namespace std;

map<int,int>count;

int main()
{
    //cout << "Hello World!" << endl;
    int test;
    cin>>test;
    while(test--){
    	count.clear();
    	int n;
    	cin>>n;
    	int ele,i,max=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>ele;
    		if(count[ele]==0)
    		  count[ele]++;
    		 else
    		  count[ele]--;
    		 if(ele>max)
    		  max=ele;
    	}
    	int ans;
    	for(i=0;i<=max;i++)
    	{
    		if(count[i]!=0)
    		 {
    		 	ans=i;
    		 	break;
    		 }
    	}
    	
    	cout<<ans<<endl;
    }
    return 0;
}
