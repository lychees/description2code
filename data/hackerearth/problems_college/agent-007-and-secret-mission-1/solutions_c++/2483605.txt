#include <iostream>
#include <string.h>
using namespace std;
int main()
{
  	int t;
  	cin>>t;
  	while(t--){
  		string a;
  		cin>>a;
  		int set=0,unset=0,n;
  		cin>>n;
  		for(int i=0;i<a.length();i++){
  			if(a[i]=='1')
  				set++;
  			else
  				unset++;
  		}
  		if(n==1||n==2)
  			cout<<unset;
  		else if(n==3||n==4)
  			cout<<set;
  		else{
	  		if((n-1)%4==0||(n-2)%4==0)
	  			cout<<unset;
	  		else
	  			cout<<set;
  		}
  		cout<<endl;
  	}
    return 0;
}
