#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
const int MAXN=10007;
struct node
{
    ll x,y;
    bool p;
};
bool type(node n1,node n2)
{
    ll dist=abs(n1.x-n2.x)+abs(n1.y-n2.y);
    if(n1.p==n2.p&&(dist&1))
        return false;
    if(n1.p==n2.p&&!(dist&1))
        return true;
    if(n1.p!=n2.p&& dist&1)
        return true;
    return false;
}
int main()
{
    int T;
    scanf("%d",&T);
    while(T--)
    {

    int N;
    scanf("%d",&N);
    vector<node> e,o;
    bool ans=true;
    for(int i=0;i<N;++i)
    {
        node tmp;
        ll tt;
        scanf("%lld %lld %lld",&tmp.x,&tmp.y,&tt);
        if(tt&1)
            tmp.p=true;
        else
            tmp.p=false;
        if(tmp.p)
            o.push_back(tmp);
        else
            e.push_back(tmp);

    }
        for(int i=0;ans&&i<o.size();++i)
        {
            for(int j=0;ans&&j<e.size();++j)
            {
                if(!type(o[i],e[j]))
                    ans=false;

            }
        }
    for(int i=0;ans&&i<o.size();++i)
        for(int j=i+1;ans&&j<o.size();++j)
            if(!type(o[i],o[j]))
                ans=false;

    for(int i=0;ans&&i<e.size();++i)
        for(int j=i+1;ans&&j<e.size();++j)
            if(!type(e[i],e[j]))
                ans=false;
    if(!ans)
        printf("No\n");
    else
        printf("Yes\n");
    }
    return 0;
}
