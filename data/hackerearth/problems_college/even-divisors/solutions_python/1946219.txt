import sys
		
def factors(n):    
    return even_only(list(set(reduce(list.__add__, 
                ([i, n//i] for i in range(1, int(n**0.5) + 1) if n % i == 0)))))
                
def even_only(lst):
    evens = []
    for number in lst:
        if number % 2 == 0:
            evens.append(number)
    return len(evens)
		
for place,line in enumerate(sys.stdin):
	if place:
		print factors(int(line.strip()))
