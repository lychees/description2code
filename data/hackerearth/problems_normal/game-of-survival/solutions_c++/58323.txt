#include <stdio.h>
#include<algorithm>
#include<iostream>
#define MAX 100000
#define FOR(i,n) for(int i=0; i<n; ++i)

using namespace std;


int main()
{
    	int t,n,a[MAX], s[MAX];
	scanf("%d",&t);

	FOR(T,t)
	{
		scanf("%d", &n);
		
		FOR(i,n) scanf("%d", &s[i]);
		FOR(i,n) scanf("%d", &a[i]);

		sort(a,a+n);
		sort(s,s+n);

		bool can_kill = true;

		FOR(i,n)
		{
			if(s[i]<=a[i])
			{
				can_kill = false;
				break;
			}
		}	

		(can_kill ? printf("YES\n") : printf("NO\n"));
	}

return 0;
}
