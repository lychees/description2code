#include <bits/stdc++.h>
using namespace std;

const int MXN = 1e4 + 9;

int tc,N,X;
int A[MXN];
int dp[2][MXN];

int main(){
	scanf("%d",&tc);
	while(tc--){
		memset(dp,0,sizeof(dp));
		long long total = 0;
		scanf("%d %d",&N,&X);
		for(int i = 1;i <= N;++i){
			scanf("%d",&A[i]);
			total += A[i];
		}
		for(int i = 1;i <= N;++i){
			for(int j = 0;j <= X;++j){
				int cur = i & 1, prev = cur ^ 1;
				dp[cur][j] = dp[prev][j];
				if(A[i] <= j)
					dp[cur][j] = max(dp[cur][j],dp[prev][j - A[i]] + A[i]);
			}
		}
		puts((total - dp[N & 1][X] <= X) ? "YES" : "NO");
	}
	return 0;
}
