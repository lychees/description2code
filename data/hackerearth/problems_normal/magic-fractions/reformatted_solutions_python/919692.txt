import math


def isPrime(n):
    i = 2
    while i <= math.sqrt(n):
        if n % i == 0:
            return False
        i += 1
    return True


def magicFraction(n):
    if n == 1:
        return 0
    j, result = 1, 1

    for i in range(3, n + 1):
        if isPrime(i):
            j *= 2
        result += j
    return result
