#include <iostream>
using namespace std;

int main()
{
	int t,i,j,loc =0, temp;
	cin>>t;
	int flag[t];
	int *s = new int[t];
	int *a[t];
	for(i=0; i<t; i++)
	{
		flag[i] = 0;
		cin>>s[i];
		a[i] = new int[s[i]];
		for(j=0; j<s[i]; j++)
		cin>>a[i][j];
	}
/*	
	for(i=0; i<t; i++)
	{
		for(j=0; j<s[i]; j++)
		cout<<a[i][j]<<" ";
		cout<<endl;
	}
*/

	for(i=0; i<t;i++)
	{
		loc = 0;
		while(1)
		{
			if(loc == -1)
			{
				flag[i] = 1;
				break;
			}
			
			temp = a[i][loc];
			a[i][loc] = -2;
			loc = temp;
			
			if(a[i][loc] == -2)
			{
				flag[i] = 0;
				break;
				
			}
		}
	}

	for(i=0; i<t; i++)
	{
		if(flag[i])
		{
			cout<<"Win"<<endl;
		}
		else
		{
			cout<<"Lose"<<endl;
		}
	}

	return 0;
}
