#include<bits/stdc++.h>
using namespace std;

int main() {
   int i,j,cnt,k,l,t,x,y;
   string a,b;
   cin >> t;
   while(t--) {
       cin >> a >> b;
       x=a.size();
       y=b.size();
       
       
       int ans=INT_MAX;

       for(i=0;i<=x-y;i++) {
       	cnt=0;
       	 for(j=0;j<y;j++) {
       	 	if(b[j]!=a[i+j])
       	 		cnt++;
       	 }
       	 ans=min(ans,cnt);
        }

        cout << ans << endl;
   }

  return 0;
}
