#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
#define ll long long

long long dp[1000005][10] = {};
long long ans[1000005] = {};

int main()
{
	int tc;
	cin >> tc;
	for(int i=0;i<10;i++)
	{
		dp[1][i] = 1;
		ans[1] += dp[1][i];
	}
	for(int i=2;i<=1000000;i++)
	{
		dp[i][0] = dp[i-1][1]%MOD;
		dp[i][9] %= MOD;
		dp[i][9] = dp[i-1][8]%MOD;
		dp[i][9] %= MOD;
		ans[i] = dp[i][9]%MOD;
		ans[i] %= MOD;
		for(int j=1;j<9;j++)
		{
			dp[i][j] = dp[i-1][j-1]%MOD + dp[i-1][j+1]%MOD;
			dp[i][j] %= MOD;
			ans[i] += dp[i][j];
			ans[i] %= MOD;
		}
	}
	while(tc--)
	{
		int n;
		cin >> n;
		cout << ans[n] << "\n";
	}
    return 0;
}
