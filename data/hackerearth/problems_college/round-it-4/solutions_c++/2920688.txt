#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--){
   long long int m,n,j,c1,c2;
   	cin>>m>>n;
   	j=m/n;
   
   
   		c1=abs(m-(n*j));
   		c2=abs(m-(n*(j+1)));
   		if(c1==c2)
   		cout<<n*(j+1)<<endl;
   		else{
   		m=c1>c2 ? n*(j+1) : n*j;
   		cout<<m<<endl;}
   		
   	
   }
}
