#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
char a[100000];
int main()
{
    ll t,k;
    cin>>t;
    while(t--)
    {
    	cin>>k;
    	cin>>a;
    	for(ll i=0;a[i]!='\0';i++)
    	{
    		if(islower(a[i]))
    		a[i]='a'+(((a[i])-'a'+(k%26))%26);
    		else
    		a[i]='A'+(((a[i])-'A'+(k%26))%26);
    	}
    	cout<<a<<endl;
    }
    return 0;
}
