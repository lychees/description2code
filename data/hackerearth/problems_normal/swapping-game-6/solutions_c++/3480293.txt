#include <iostream>
#include<stack>
#include<algorithm>
#include<string>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    string s,s1,s2;
    long int k;
    cin>>k;
    cin>>s;
    s2=s;
 
    int l=0;
    stack<char>st;
    bool flag=true;
    while(s!=s2||flag){
    	for(unsigned int i=1;i<s.length();i=i+2)
    	st.push(s.at(i));
    	for(unsigned int i=0;i<s.length();i=i+2)
    	s1.push_back(s.at(i));
    	while(!st.empty()){
    		string p;
    		char c=st.top();
    		p.push_back(c);
    		s1.append(p);
    		st.pop();
    	}
    	s=s1;
    	s1.clear();
    	flag=false;
    	l++;
    }
    s2.clear();
    k=k%l;
   while(k--){
    	for(unsigned int i=1;i<s.length();i=i+2)
    	st.push(s.at(i));
    	for(unsigned int i=0;i<s.length();i=i+2)
    	s1.push_back(s.at(i));
    	while(!st.empty()){
    		string p;
    		char c=st.top();
    		p.push_back(c);
    		s1.append(p);
    		st.pop();
    	}
    	s=s1;
    	s1.clear();
    //	l++;
    }
    
    cout<<s;
    return 0;
}
