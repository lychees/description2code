#include <stdio.h>
int a[1000005];
int p[1000006];
int get(int m)
{
	int ans =0;
	while(m!=1)
	{
		ans++;
		m/=p[m];
	}
	return ans;
}
int main()
{
   // printf("Hello World!\n");
   int n,i;
   int cur;
   int hi = 1000000;
   for(n=2;n<=hi;n++)
   if(p[n]==0)
   {
   	for(cur=n;cur<=hi;cur+=n)
   	p[cur] = n;
   }
   for(i=2;i<=hi;i++)
   a[i]=a[i-1] +get(i);
  // for(i=1;i<=10;i++)
  // printf("%d has %d and also %d\n",i,get(i),p[i]);
   scanf("%d",&n);
   while(n--)
   {
   	scanf("%d",&cur);
   	printf("%d\n",a[cur]);
   }
    return 0;
}
