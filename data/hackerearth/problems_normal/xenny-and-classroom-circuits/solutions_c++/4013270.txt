#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
    	int count1=0,count2=0,N;
    	char ch;
    	cin >> N;
    	for(int i=0;i<2*N;i++){
    		cin >> ch;
    		if(i%2==0){
    			if(ch=='B') count2++;
    			else count1++;
    		}
    		if(i%2==1){
    			if(ch=='G') count2++;
    			else count1++;
    		}
    	}
    	cout << min(count1,count2)/2 << endl;
    }
    return 0;
}
