const int mod = 1000000000 + 7 ;
const int maxn = 100000 + 10 ;
const int inf = 2*1e9 ; 

int n ;
ll A[100] , dp[100][100] ;

ll solve(int l,int r,ll v1,ll v2){
	
	if(r-l+1 < 3)
		return 0 ;
	if(r-l+1 == 3)
		return 1LL * v1 * v2 ;
	ll &ret = dp[l][r] ;
	if(ret != -1)
		return ret ;
	ret = 0 ;
	for(int i=l+1;i<r;i++){
		ret = max(ret,1LL*v1*v2 + solve(l,i,v1,A[i]) + solve(i,r,A[i],v2) ) ;
	}
	return ret ;
}

int main(){
	
	cin >> n ;
	for(int i=1;i<=n;i++){
		cin >> A[i] ;
	}
	memset(dp,-1,sizeof dp) ;
	cout << solve(1,n,A[1],A[n]) << endl ;
	return 0 ; 	
}
