#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,i;
    cin>>n;
    int arr[n];
    for(i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    int s,count = 0;
    cin>>s;
    int sum = 0;
    sort(arr,arr+n);
    reverse(arr,arr+n);
    for(i=0;i<n;i++)
    {
        sum+=arr[i];
        count++;
        if(sum>=s)
        {
            break;
        }
    }
    cout<<count<<endl;
    return 0;
}
