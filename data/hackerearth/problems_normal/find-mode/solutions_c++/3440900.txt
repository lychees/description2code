#include <iostream>
using namespace std;
#include <vector>
#include <algorithm>
#define ll long long

int main()
{	ll t;
    cin>>t;
    while(t--)
    {	ll c=1,n,i,q,m=0;
    	cin>>n;
    	vector<ll> z,x;
    	for(i=0;i<n;++i)
    	{	cin>>q;
    		z.push_back(q);
    	}
    	sort(z.begin(),z.end());
    	for(i=1;i<=n;++i)
    	{	if(z[i]==z[i-1])
    			c++;
    		else
    		{	if(c>m)
    			{	x.clear();
    				x.push_back(z[i-1]);
    				m=c;
    			}
    			else if(c==m)
    				x.push_back(z[i-1]);
    			c=1;
    		}
    	}
    	for(i=x.size()-1;i>=0;--i)
    		cout<<x[i]<<" ";
    	cout<<"\n";
    }
    return 0;
}
