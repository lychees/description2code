#include <iostream>
#include <algorithm>
#include <cmath>

using namespace std;

int main()
{
  	int t, n, k, v;
  	cin>>t;
  	while(t--)
  	{
  		cin>>n>>k;
  		
  		int types[k+1];
  		fill(types, types+k+1, 0);
  		
  		int maxC = 0, ans = 0;
  		for(int i = 0; i<n; i++)
  		{
  			cin>>v;
  			types[v]++;
  			maxC = max(types[v], maxC);
  		}
  		ans = n - maxC;
  		cout<<ans<<endl;
  	}
    return 0;
}
