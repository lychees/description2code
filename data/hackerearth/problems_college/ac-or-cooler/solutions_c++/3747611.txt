#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;

int main()
{
    int N, M;
    scanf("%d %d", &N, &M);
    int i, a[2 * N];
    for(i=0;i<2 * N;++i)
    {
    	scanf("%d", &a[i]);
    }
    sort(a, a + 2 * N);
    int cost = 0;
    for(i=0;i<2*N;++i)
    {
    	cost += a[i];
    	if(cost >= M)
    	{
    		break;
    	}
    	//cout << a[i] << " ";
    }
    printf("%d\n", i);
    return 0;
}
