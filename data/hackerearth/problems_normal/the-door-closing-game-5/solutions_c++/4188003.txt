#include<stdio.h>
int main()
{
    int t;
    char s[200];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s",s);
        int x=0,ans=0,f=0;
        for(int i=0;s[i]!='\0';i++)
        {
            if(s[i]=='_')
                x++;
            if(s[i]=='|')
            {
                f=1;
                ans^=x;
                x=0;
            }
        }
        ans^=x;
        if(ans&&f)
            printf("Alice\n");
        else
            printf("Bob\n");
    }
    return 0;
}
