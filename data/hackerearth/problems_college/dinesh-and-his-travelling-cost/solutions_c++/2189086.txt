#include <bits/stdc++.h>
using namespace std;

#define MAXN 505
vector<pair<int,int> > adj[MAXN];

int main()
{
	int n,u,v,w;
	cin >> n;
	while(n--)
	{
		cin >> u >> v >> w;
		adj[u].push_back(make_pair(v,w));
		adj[v].push_back(make_pair(u,w));
	}
	int s;
	cin >> s;
	priority_queue<pair<int,int>, vector<pair<int,int> >, greater<pair<int,int> > > pq;
	pq.push(make_pair(0,s));
	vector<int> d(MAXN,100000000);
	d[s] = 0;
	while(!pq.empty())
	{
		pair<int,int> p = pq.top();
		pq.pop();
		v = p.second;
		int cost = p.first;
		if(cost>d[v])
			continue;
		for(int i=0;i<adj[v].size();++i)
		{
			int to = adj[v][i].first,
				edge = adj[v][i].second;
			if(d[to]>d[v]+edge)
			{
				d[to] = d[v]+edge;
				pq.push(make_pair(d[to],to));
			}
		}
	}
	cin >> n;
	while(n--)
	{
		cin >> v;
		if(d[v]!=100000000)
			cout << d[v] << endl;
		else
			cout << "NO PATH" << endl;
	}
	return 0;
}
