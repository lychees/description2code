# print 'Hello World!'

n = int(raw_input())
for i in range(n):
    tc = list(raw_input())
    # tc = [x for x in tc if x != ' ']
    ans = 0
    for j in tc:
        ans += ord(j)

    ans /= len(tc)

    if ans % 2 == 0:
        print ''.join(tc)
    else:
        print ''.join(tc[::-1])
