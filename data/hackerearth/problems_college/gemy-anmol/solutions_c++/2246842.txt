int main()
{
    int t;
    scanf("%d",&t);

    {
    	long v[10]={0};
    	long n,m;

    	//long *v = new long[n];
    	//long v[n];

    	{
    		for(int j =0;j<n-i-1;j++)
    		{

    			{
    				long temp = v[j];
    				v[j]= v[j+1];

    		if(sum >= m)

    		}
    		cnt++;
