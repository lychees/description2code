#include<iostream>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int i=4;
    while(i<=n)
    {
        int j=2;
        while(j<i)
        {
            if(i%j==0)
            {
                cout<<i<<endl;

                break;
            }
            else
            j++;

        }i++;
    }
}
