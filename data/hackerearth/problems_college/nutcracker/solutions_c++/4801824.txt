#include <bits/stdc++.h>
using namespace std;

bool match(int n)
{
	bool dp[n+1];
	dp[1]=true;
	dp[2]=false;
	dp[3]=true;
	dp[4]=true;
	dp[5]=false;
	n=n%5;
	if(n==0)
	n=5;
	return dp[n];
}

int main()
{
   	int t,n;
   	cin>>t;
   	while(t--)
   	{
   		cin>>n;
   		if(match(n))
   			cout<<"Yes";
   		else
   			cout<<"No";
   		cout<<endl;
   	}
    return 0;
}
