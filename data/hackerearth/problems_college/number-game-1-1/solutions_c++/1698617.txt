#include <iostream>
using namespace std;

int compute(unsigned long long n)
{
	int op=0;
	if(n==1)
		return 0;
	while(n!=1)
	{
		if(n%2==0)
		{
			n/=2;
			op++;
		}
		else if(n%3==0)
		{
			n/=3;
			op++;
		}
		else
		{
			n=n-1;
			op++;
		}
		if(n<1)
			break;
	}
	return op;
}
	
int main()
{
    int i,t,res=0;
    unsigned long long n;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>n;
    	res=compute(n);
    	cout<<res<<endl;
    }
    return 0;
}
