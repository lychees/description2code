int gcd(int a,int b)
{
   if(b==0) return a;
   else return gcd(b,a%b);
}    

int main()
{  

  int t;
  scanf("%d",&t);
  while(t--)
  {
     int n;
     scanf("%d",&n);
     
     ll ans=0;
     
     for(int i=2;i<=sqrt(n-1)+1;i++)
     {
        for(int j=1;j<i;j++)
        {
          if((gcd(i,j)==1) and ((i-j)%2) and (i*i+j*j)<=n)          
          {
             int k=0;
             ll c=i*i+j*j;
              while ((++k) * c <= n)
                      ans++;
          }
        }
     }
     cout<<ans<<endl;               
   } 
}
