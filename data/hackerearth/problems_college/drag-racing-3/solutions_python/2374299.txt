n = int(raw_input())
indeg = {}

for i in range( (1<<n) - 1 ):

	indeg[b] = (indeg.get(b,0) + 1)
	indeg[a] = indeg.get(a,0)

for k in indeg:
    if indeg[k]==0:
        print k
