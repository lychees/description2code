#include <iostream>
using namespace std;
#define mod 1000000007
int n;
int arr[505];
int dp[505][105];
int gcd(int a,int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
long long int calldp()
{
	for(int i=0;i<n;i++)
	{
		dp[i][arr[i]]++;
		for(int j=0;j<i;j++)
		{
			
			if(arr[j]<arr[i])
			{
				for(int k=1;k<=100;k++)
				{
					int g=gcd(arr[i],k);
					dp[i][g]=((dp[i][g]%mod)+(dp[j][k]%mod))%mod;
				}
			}
		}
	}
	long long int sum=0;
	for(int i=0;i<n;i++)
	{
		sum=sum+dp[i][1];
	}
	sum=sum%mod;
	return sum;
}
int main()
{
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	cin>>arr[i];
    }
    cout<<calldp();
    return 0;
}
