#include <iostream>
#include <cmath>
#include <algorithm>
#define LL long long
#define MOD 1000000007
using namespace std;

int arr[5001];
LL oldseq[5001];
LL newseq[5001];
LL oldsum[5001];
LL newsum[5001];
int main()
{
    int n,a,b,k,l,r;
    cin>>n>>a>>b>>k;
    
    for(int i=1;i<=n;++i)
    {
    	
    	l=max(1,i-abs(i-b)+1);
    	r=min(n,i+abs(i-b)-1);
    	arr[i]=(i==b?0:r-l);
    	oldseq[i]=arr[i];
    	oldsum[i]=oldsum[i-1]+arr[i];
    }
    for(int i=2;i<=k;++i)
    {
    	for(int j=1;j<=n;++j)
    	{
    		l=max(1,j-abs(j-b)+1);
    		r=min(n,j+abs(j-b)-1);
    		if(l==r || j==b)
    			newseq[j]=0;
    		else
    			newseq[j]=(oldsum[r]-oldsum[l-1]-oldseq[j]+MOD+MOD)%MOD;
    		newsum[j]=(newsum[j-1]+newseq[j])%MOD;
    	}
    	for(int j=1;j<=n;++j)
    	{
    		oldseq[j]=newseq[j];
    		oldsum[j]=newsum[j];
    		newseq[j]=0;
    		newsum[j]=0;
    	}
    }
    cout<<oldseq[a]<<endl;
    return 0;
}
