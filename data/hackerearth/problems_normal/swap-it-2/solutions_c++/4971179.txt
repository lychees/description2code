#include<iostream>
#include<algorithm>
using namespace std;

main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,k;
	cin>>n>>k;
	int a[n],i,j;
	for(i=0;i<n;i++)
	cin>>a[i];
	for(i=0;i<n;i++)
	{
		int min=a[i];
		int mind=i;
		for(j=i+1;j<n && j<=i+k;j++)
		{
			if(a[j]<min)
			{
				min=a[j];
				mind=j;
			}
		}
		if(mind!=i)
		{
			int x=mind;
			int key=a[mind];
			x--;
			while(x>=i)
			{
				a[x+1]=a[x];
				x--;
				k--;
			}
			a[i]=key;
		}
		if(k==0)
		break;
	}
	for(i=0;i<n;i++)
	cout<<a[i]<<" ";
	cout<<endl;
	}
}
