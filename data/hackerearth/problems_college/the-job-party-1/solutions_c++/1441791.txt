#include <bits/stdc++.h>
using namespace std;
int main()
{
    long long int t;
    scanf("%lld",&t);
    while(t--) {
    	long long int n,k;
    	scanf("%lld%lld",&n,&k);
    	k++;
    if(k==1)
    printf("0\n");
    else if(k%2==0) {
    		printf("%lld\n",(k/2)*(k-1));
    	}
    	else
    	printf("%lld\n",(k)*((k-1)/2));
    }
    return 0;
}
