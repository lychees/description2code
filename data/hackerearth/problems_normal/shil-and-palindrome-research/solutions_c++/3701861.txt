int bt[100011][26];
int N,Q;
void update(int ind ,int val,int pos){
	while(ind<=N){
		bt[ind][pos]+=val;
		ind+=(ind&-ind);
	}
}
int query(int ind,int pos){
	int ans=0;
	while(ind>0){
		ans+=bt[ind][pos];
		ind-=(ind&-ind);
	}
	return ans;
}

			update(L,1,x-'a');
			update(L,-1,S[L-1]-'a');
			S[L-1]=x;
		}
		else{
			cin >> L >> R;
			int odd=0;
			int cur;
			rep(i,26){
				cur=query(R,i)-query(L-1,i);
