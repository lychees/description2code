#include <bits/stdc++.h>
#define ll long long int

using namespace std;

ll bit[1000001],dp[1000001];
ll maxval;

void update(int x,int val) {
	while (x <= maxval) {
		bit[x] += val;
		x += (x & -x);
	}
}

ll query(int x) {
	ll sum = 0;
	while (x > 0) {
		sum += bit[x];
		x -= (x & -x);
	}

	return sum;
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie();
	memset(bit,0,sizeof(bit));
	memset(dp,0,sizeof(dp));
	cin >> maxval;
	int m;
	cin >> m;

	for (int i = 0; i < m; i++) {
		int x,y;
		cin >> x >> y;
		update(x,1);
		update(y + 1,-1);
	}
	
	for (int i = 1; i <= maxval; i++) {
		int coins = query(i);
		dp[coins] += 1;
	}
	for (int i = m - 1; i >= 0; i--) {
		dp[i] += dp[i+1];
	}
	int q;
	cin >> q;
	while (q--) {
		int n;
		cin >> n;
		cout << dp[n] << endl;
	}

	return 0;
}
