#include<cstdio>
#include<iostream>
#include<cstdio>
using namespace std;
#define mod 1000003
long long f[1000003];
int main()
{
	long long i;
	f[0] = 1;
	for(i=1;i<=1000003;i++)
	{
		f[i]=(i*f[i-1])%mod;
	}
	int t;
	cin>>t;
	while(t--)
	{
	long long n,x;
	cin>>n>>x; 
	x = x%mod;
	if(n>mod)
	n = mod;
       long long k = f[n];
       k = k*x;
       cout<<k%mod<<"\n";
	}
	
}
