def GCD(a, b):
    a = abs(a)
    b = abs(b)
    while a:
        a, b = b % a, a
    return b


def GCD_List(l):
    return reduce(GCD, l)


def prodt(a, b):
    return a*b


def prod_list(l):
    return reduce(prodt, l)


n = int(raw_input())
l = map(int, raw_input().split())
l1 = list(set(l))
