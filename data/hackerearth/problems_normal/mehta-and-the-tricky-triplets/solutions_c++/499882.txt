#include<cstdio>
using namespace std;
long long int arr[1000009];
long long int ncr(long long int n)
{
   if(n<3)
      return 0;
   return n*(n-1)*(n-2)/6;
}
int main()
{
   int n;
   long long int tw=0,th=0,fi=0,se=0,twth=0,twfi=0,twse=0,ttfs=0;
   long long int thfi=0,thse=0,fise=0,twthfi=0,twthse=0,thfise=0,twfise=0;
   scanf("%d",&n);
   for(int i=0;i<n;i++)
   {
      bool two=false,three=false,five=false,seven=false;
      scanf("%lld",&arr[i]);
      long long int temp=arr[i];
      while(temp!=0)
      {
         int dig=temp%10;
         temp/=10;
         if(dig==2)
            two=true;
         if(dig==3)
            three=true;
         if(dig==5)
            five=true;
         if(dig==7)
            seven=true;
      }
      if(two)
         tw++;
      if(three)
         th++;
      if(five)
         fi++;
      if(seven)
         se++;
      if(two&&three)
         twth++;
      if(two&&five)
         twfi++;
      if(two&&seven)
         twse++;
      if(three&&five)
         thfi++;
      if(three&&seven)
         thse++;
      if(five&&seven)
         fise++;
      if(two&&three&&five)
         twthfi++;
      if(two&&five&&seven)
         twfise++;
      if(two&&three&&seven)
         twthse++;
      if(three&&five&&seven)
         thfise++;
      if(two&&three&&five&&seven)
         ttfs++;
   }
   long long int ans=0;
   ans+=ncr(tw);
   ans+=ncr(th);
   ans+=ncr(fi);
   ans+=ncr(se);
   ans-=ncr(twth);
   ans-=ncr(twfi);
   ans-=ncr(twse);
   ans-=ncr(thfi);
   ans-=ncr(thse);
   ans-=ncr(fise);
   ans+=ncr(twthfi);
   ans+=ncr(twthse);
   ans+=ncr(thfise);
   ans+=ncr(twfise);
   ans-=ncr(ttfs);
   printf("%lld\n",ans);
   return 0;
}
