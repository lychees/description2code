#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int T,i,j,k,D,W,ansD,ansW;
	cin>>T;
	while(T--)
	{
		ansD=0;
		ansW=0;
		cin>>D>>W;
		if(D>60)
		{
			cout<<"NA\n";
		}
		else
		{
			if(D<=2)
				ansD=50;
			else
			{
				ansD=(D-2)*12+50;
			}
			ansW=W*2;
			cout<<ansD+ansW<<"\n";
		}
	}
}
