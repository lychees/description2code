#include <bits/stdc++.h>
using namespace std;
int N,con=0;
map<string,int> M1,M2;
int main()
{
	string s,fir,las,firs;
	cin>>N;
	while(N--)
	{
		cin>>s;
		fir=s.substr(0,1);
		las=s.substr(s.length()-1,1);
		fir=fir.insert(1,las);
		sort(s.begin(),s.end());
		if(!M1[fir])
		{
			M1[fir]=1;
			M2[s]=1;
			con++;
		}
		else 
		{
			if(!M2[s])
			{
				M2[s]=1;
				con++;
			}
		}
		s.erase(0,s.length());
		fir.erase(0,fir.length());
		las.erase(0,las.length());
	}
	cout<<con<<endl;
    return 0;
}
