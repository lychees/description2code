#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n,i,j,k;
    string s;
    cin>>n;
    while(n--)
    {
    	int f=0;
    	cin>>s;
    	for(i=0;i<s.length();i++){
    		if(s[i]=='0' || s[i]=='1' || s[i]=='8')
    			continue;
    		else{
    			f=1;
    			cout<<"NO"<<endl;
    			break;
    		}
    	}
    	if(!f){
    		string p=s;
    		reverse(s.begin(),s.end());
    		if(p==s)
    			cout<<"YES"<<endl;
    		else
    			cout<<"NO"<<endl;
    	}
    }
    return 0;
}
