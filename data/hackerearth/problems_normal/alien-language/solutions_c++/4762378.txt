#include <iostream>
using namespace std;

int main()
{
    int t,ans,i;
    int a[150]={0};
    string s1,s2;
    cin>>t;
    
    while(t--)
    {
    	ans = 0;
    	cin>>s1>>s2;
    	for(i = 0 ; i < s1.length() ; i++)
    	{
    	   a[s1[i]]	++;
    	}
    	for(i = 0 ; i < s2.length(); i++)
    	if(a[s2[i]] != 0)
    	{
    		ans = 1;
    		break;
    	}
    	if(ans == 1)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    	for(i = 0 ; i < s1.length(); i++)
    	a[s1[i]]=0;
    }
    
    return 0;
}
