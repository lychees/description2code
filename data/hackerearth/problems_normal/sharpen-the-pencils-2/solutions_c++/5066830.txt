#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >>t;
	while(t--){
		int n;
		cin >> n;
		int ar[n];
		for(int i = 0; i < n; i++){
			cin >> ar[i];
		}
		
		int madhavtime = 0, riyatime = 0, madhav = 0, riya = n-1;
		int madhavcount = 0, riyacount = 0;
		while(madhav <= riya){
			if(madhavtime <= riyatime && madhav == riya){
				madhav++;
				madhavcount++;
			}else if(madhavtime > riyatime && madhav == riya){
				 riya--;
				riyacount++;
			}else if(madhavtime < riyatime){
				madhavtime += ar[madhav++];
				madhavcount++;
			}else if(madhavtime > riyatime){
				riyatime += 2*ar[riya--];
				riyacount++;
			}
			else{
				riyatime += 2*ar[riya]; riya--;
				riyacount++;
				madhavtime += ar[madhav++];
				madhavcount++;
			}
		}
		cout << madhavcount << " " << riyacount << endl;
	}
    
    return 0;
}
