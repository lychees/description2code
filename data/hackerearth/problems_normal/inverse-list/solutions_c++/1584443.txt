#include <iostream>
using namespace std;

int main()
{
    long T, N;
    
    cin>> T;
    while(T--)
    {
    	cin>> N;
    	
    	long A[N+1];
    	for(int i=1; i< N+1; i++)
    	{
    		cin>> A[i];
    	}
    	long B[N+1];
    	for(int i=1; i < N+1; i++)
    	{
    		B[A[i]] = i;
    	}
    	int n = 1;
    	for(int i=1; i < N+1; i++)
    	{
    		if(A[i] != B[i])
    		{
    			n = 0;
    			break;
    		}
    	}
    	
    	if(n==0)
    		cout << "not inverse" << endl;
    	else
    		cout << "inverse" << endl;
    	
    }
    return 0;
}
