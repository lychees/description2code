#include <bits/stdc++.h>
using namespace std;

int visit[100005];
vector<int> X[100005];

void dfs(int i)
{
	visit[i] = 1;
	
	for(int j=0; j<X[i].size(); j++)
		if(!visit[X[i][j]])
			dfs(X[i][j]);
}

int main()
{
    int T;
    scanf("%d", &T);
    
    while(T--)
    {
    	int N, M;
    	scanf("%d %d", &N, &M);
    	
    	for(int i=0; i<N; i++)
    	{
    		visit[i] = 0;
    		X[i].clear();
    	}
    	
    	while(M--)
    	{
    		int x, y;
    		scanf("%d %d", &x, &y);
    		
    		X[x].push_back(y);
    		X[y].push_back(x);
    	}
    	
    	int ans = 0;
    	
    	for(int i=0; i<N; i++)
    	{
    		if(!visit[i])
    		{
    			dfs(i);
    			ans++;
    		}
    	}
    	
    	printf("%d\n", ans);
    }
    
    return 0;
}
