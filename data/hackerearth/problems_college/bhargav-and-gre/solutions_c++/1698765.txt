#include <iostream>
#include <string>
using namespace std;

int main()
{
    string a,b;
    cin>>a;
    int l,l2,N,S=0,A=0;
    cin>>N;
    l=a.size();
    while(N--)
    { 
    	cin>>b;
    	l2=b.size();
    	if(l==l2)
    	{
    		if(a[0]==b[0])
    		S++;
    		if(b[0]-a[0]==3)
    		A++;
    	}
    }
    cout<<S<<" "<<A;
    return 0;
}
