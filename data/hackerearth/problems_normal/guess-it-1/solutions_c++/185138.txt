#define MAX 210
//#define getchar_unlocked() getchar()
typedef long long LL;
using namespace std;
inline int read() //faster input for integer
{
	int n = 0;
	register char c = getchar_unlocked();//register is used due to its repeated use
	while (!('0' <= c && c <= '9'))
 	{
  		c = getchar_unlocked();
 	}
 	while ('0' <= c && c <= '9')
	{
  	n = (n<<3) + (n<<1) +  c - '0';//equivalent to n = n*10 + c - 48;
  	c = getchar_unlocked();
	}
 return n;
}

int main()
{
	int test;
	scanf("%d",&test);
	while(test--)
	{
		char number[MAX];
		int ans=0,i;
		cin>>number;
		for(i=0;i<strlen(number);i++)
		{
			if(number[i]!='_')
			{
				ans+=((number[i]-48));
				ans=ans%9;
			}
		}
		if(ans==0||ans==9)
		cout<<"0 or 9\n";
		else
		{
		cout<<9-ans<<endl;
		}
	}
//	getchar();
	return 0;
}
