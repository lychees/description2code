#include<iostream>
using namespace std;

int main()
{
	int x,y,c,cases=2,z,b;
	cin >> x;
	while(cases--)	
	{	
			c=1;
			while(c<=x)
			{
				y=x-c;
				while(y>0)
				{
					cout << " ";
					y--;
				}
				
				z=1;
				while(z<=c)
				{
				    if(z%2==1)
				     cout << "/";
				    else
				     cout << " ";
			    	z++;
		    	}
		    
				z=1;
				while(z<=c)
				{
				    if(z%2==c%2)
				     cout << "\\";
				    else
				     cout << " ";
				    z++;
			    }
			    cout << endl;
				c++;
			}
			
			c=0;
			while(c<x)
			{
				y=2+c;
				while(y>0)
				{
					cout << " ";
					y--;
				}
				
				z=1;
				while(z<=x-c)
				{
				    if(z%2==1)
				     cout << "\\";
				    else
				     cout << " ";
			    	z++;
		    	}
		    
				z=1;
				while(z<=x-c)
				{
				    if(z%2==(x-c)%2)
				     cout << "/";
				    else
				     cout << " ";
				    z++;
			    }
			    cout << endl;
				c++;
			}
	x++;
	}
}
