n = input()
for i in range(n):
    pat = raw_input()
    li = list(pat)
    st = raw_input()
    stt = list(st)
    for i in range(len(stt)):
        temp = stt[i]
        stt[i] = li.index(temp)
    stt.sort()
    for j in range(len(stt)):
        temp = int(stt[j])
        stt[j] = li[temp]
    print ''.join(stt)
