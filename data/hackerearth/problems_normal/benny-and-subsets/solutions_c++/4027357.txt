#include<bits/stdc++.h>
#define mod 10000007
using namespace std;
static int dp[2][(1<<21)];
int main(){
	int n,x,k;
	scanf("%d %d",&n,&k);
	vector<int> a,b;
        int ma=0;
	for(int i=1;i<=n;i++){
		int x;
		scanf("%d",&x);
        if(x>(1<<10))
        	b.push_back(x);
        else
        	a.push_back(x);
	}
	int f=1;
	dp[!f][0]=1;
	for(int i=1;i<=a.size();i++){
		for(int j=0;j<=(1<<11);j++){
			int new_num=j^a[i-1];
			dp[f][j]=dp[!f][new_num]+dp[!f][j];
			if(dp[f][j]>=mod)
			dp[f][j]-=mod;
		}
		f=!f;
	}
	for(int i=1;i<=b.size();i++){
		for(int j=0;j<(1<<21);j++){
			int new_num=j^b[i-1];
			dp[f][j]=dp[!f][new_num]+dp[!f][j];
			if(dp[f][j]>=mod)
			dp[f][j]-=mod;
		}
		f=!f;
	}
	printf("%d\n",dp[!f][k]);
	return 0;
}
