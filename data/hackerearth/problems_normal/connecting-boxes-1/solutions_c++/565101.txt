#include<cstdio>
#include<iostream>
using namespace std;
#define MAX 1005
int mat[MAX][MAX];
int n;
/// 0 means not possible -1 vacant na dcan be made
void fill()
{
    for(int i =0;i<MAX;i++)
    {
        for(int j=0;j<MAX;j++)
        {
            mat[i][j] = -1;
            if(i==j)
                mat[i][j]=0;
        }
    }
}

void update(int x, int y)
{
    for(int i =1;i<MAX;i++)
    {
        for(int j=1;j<MAX;j++)
        {
            mat[x][i]=0;
            mat[y][i]=0;
            mat[i][x]=0;
            mat[j][y]=0;
            if(i<x)
            {
                if(j>x&&j<y)
                {
                    mat[i][j]=0;
                    mat[j][i]=0;
                }
            }
            else if(i<y)
            {
                if(j>y)
                {
                    mat[i][j]=0;
                    mat[j][i]=0;
                }
            }

        }
    }
}

int main()
{
    fill();
    int q;
    scanf("%d%d",&n,&q);
    while(q--)
    {
        int i,j;
        scanf("%d%d",&i,&j);
        int x = min(i,j);
        int y = max(i,j);
        if(mat[x][y]==-1)
            {
                cout<<"YES\n";
                update(x,y);
            }
        else if(mat[x][y]==0)
            cout<<"NO\n";

    }



    return 0;
}
