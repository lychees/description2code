#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
int dp[200];
char map[200][200];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,m;
    	cin>>n>>m;
    	for(int i=0;i<n;++i)
    	{
    		for(int j=0;j<m;++j) scanf(" %c",&map[i][j]);
    	}
    	memset(dp,-1,sizeof dp);//
    	dp[0]=0;//
    	for(int i=0;i<n;++i)
    	{
    		if(dp[0]>=0)
    		{
    			if(map[i][0]=='T') dp[0]+=150;
    			else if(map[i][0]=='D') dp[0]-=100;
    		}
    		for(int j=1;j<m;++j)
    		{
    			int mx = max(dp[j],dp[j-1]);
    			if(mx>=0)
    			{
    			if(map[i][j]=='T') dp[j]= mx+150;
    			else if(map[i][j]=='D') dp[j]= mx-100;
    			else dp[j] = mx;
    			}
    		}
    	}
    	if(dp[m-1]<0) cout<<-1<<endl;
    	else cout<<dp[m-1]<<endl;
    }
    return 0;
}
