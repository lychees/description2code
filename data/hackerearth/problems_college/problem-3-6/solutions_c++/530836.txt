#include<iostream>
#include<vector>
#include<algorithm>
#include<cstdio>
#include<string>

#define getcx getchar_unlocked

inline void input(int &n)
{
n=0;
int ch=getcx();
while( ch < '0' || ch > '9' ){ ch=getcx();}
while( ch >= '0' && ch <= '9' )
n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
}


using namespace std;

int main(){

	int n;
	cin>>n;
	
	int x,y;	
	vector<pair<int,int> > v;
	for(int i=0;i<n;++i){
		cin>>x>>y;
		v.push_back(make_pair(x,y));
	}

	vector<pair<int,int> > op(n,make_pair(0,0));
	
	for(int i=0;i<n;++i){				//home
		for(int j=0;j<n;++j){			//away	
			if(i!=j){
				if(v[i].first!=v[j].second){
					op[i].first++;
					op[j].second++;
				}
				else{
					op[i].first++;
					op[j].first++;
				}	
			}
		}
	}

	for(int i=0;i<n;++i)
		cout<<op[i].first<<" "<<op[i].second<<"\n";

}
