inline void Solve(){
    string s;
    cin>>s;
    int i, n;
    cin>>n;
    bool a[26];
    memset(a, false, sizeof(a));
    char ch;
    for(i = 0; i < n; i++){
        cin>>ch;
        a[ch - 'a'] = true;
    }
    int ans = 0;
    for(i = 0; i < s.length(); i++){
        if(!a[s[i] - 'a']) ans += (s[i] - 'a') + 1;
    }
    cout<<ans<<endl;
    return;
}

int main(){
    int t = 1;
    cin>>t;
    while(t--){
        Solve();
    }
    return 0;
}
