#include<cstdio>
#include<algorithm>

using namespace std;

//ninary search
int answer(int arr[], int start, int end, int find){
    int middle ;
    while(start < end) {
        middle = (start+end)/2;
        if(arr[middle] == find) {return middle;}
        if(arr[middle] < find){
            start= middle+1;
        }
         if(arr[middle] > find){
            end= middle;
        }
    }
    return start;

}
int main()
{
    int N,Q,choc[1000000];
    int i,t;

    scanf("%d",&N);
    choc[0]=0;
    //Read N numbers , the number of choclates in box i
    for(i=0;i<N;i++)
    {
        scanf("%d",&t);
 
        choc[i] = choc[i]+t;
        if( N > i+1)
	        choc[i+1]=choc[i];
    }



    //Read Q
    scanf("%d",&Q);
 
    //read Q numbers, each for chocalate query
    while(Q--){
       scanf("%d",&t);
 
        int ans = answer(choc,0,N-1,t)+1;
       
       printf("%d\n", ans );
    }
    return 0;
}
