#include <iostream>
#include <queue>
#include <algorithm>
using namespace std;


int main()
{
    long m,n,price=0;
    cin>>m>>n;
    priority_queue<long> x;
    for(long i=0;i<m;i++)
    {
    	long tmp;
    	cin>>tmp;
    	x.push(tmp);
    }
    
  		
  		
   // cout<<"index: "<<index<<"\n";
    for(long j=0;j<n;j++){
    	long tp=x.top();
   		price+=tp;
   		x.pop();
   		x.push(tp-1);
    }
    cout<<price;
    return 0;
}
