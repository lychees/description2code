#include <iostream>
using namespace std;
#include<cmath>
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int p,x,y,u,v;
    	cin>>p>>x>>y>>u>>v;
    	if(p==1)
    	{
    		if(x==u&&y==v)
    		cout<<"Counter-Terrorists Win !\n";
    		else
    		cout<<"Terrorists Win !\n";
    	}
    	if(p==2)
    	{
    		int d=u-x;
    		if(y+d==v||y-d==v)
    		cout<<"Counter-Terrorists Win !\n";
    		else
    		cout<<"Terrorists Win !\n";
    	}
    	else if(p>=3)
    	{
    		int t=abs(u-x)+abs(v-y);
    		if(t%2==0)
    		cout<<"Counter-Terrorists Win !\n";
    		else
    		cout<<"Terrorists Win !\n";
    	}
    }
}
