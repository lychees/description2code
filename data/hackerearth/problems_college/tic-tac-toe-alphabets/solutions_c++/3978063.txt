#include<iostream>
#include<string.h>
using namespace std;

int main()
{
	int case1;
	cin>>case1;
	while(case1--)
	{
	char s[1000];
	cin>>s;
	int length=strlen(s),pos=0;
	while(length--)
	{
		int small=pos;
		for(int k=pos+1;s[k]!='\0';k++)
		{
			if(s[small]>s[k])
				small=k;
		}
		if(small!=pos)
		{
			char temp=s[pos];
			s[pos]=s[small];
			s[small]=temp;
		}
		pos++;
	}
	cout<<s<<endl;
}
return 0;
}
