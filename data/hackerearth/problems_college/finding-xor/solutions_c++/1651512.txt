#include <iostream>
#include<string.h>
using namespace std;
 
int main()
{
    char str1[10000],str2[10000];
    int n,i,l;
    cin>>n;
    
    while(n-->0)
    {
    	i=0;
    
    	cin>>str1>>str2;
    	l=strlen(str1);
    	while(i<l)
    	{
    	if(str1[i]==str2[i])
    	cout<<"0";
    	else
    	cout<<"1";
    	
    		i++;
    	}
    	cout<<"\n";
    }
    return 0;
}
