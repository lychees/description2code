#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int main()
{
    long long a[1000000];
    memset(a,0,sizeof(a));
    for(long i=1;i<1000000;i++)
    {
        if(a[i]==0)
        {
            long long s=0;
            for(long j=i;j<1000000;j+=(i+1))
            {
                if(!a[j])
                {
                    a[j]=1;
                    s++;
                }
            }
            a[i]=s;
        }
    }
    long t;
    cin>>t;
    while(t--)
    {
        long long n;
        cin>>n;
        cout<<a[n-1]<<endl;
    }
    return 0;
}
