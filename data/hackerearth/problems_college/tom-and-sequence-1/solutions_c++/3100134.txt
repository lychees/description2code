#include<bits/stdc++.h>

using namespace std;

int a[10000000];
vector<long long> vec;
vector<long long> ans;
#define MOD 1000000007
void sieve()
{
	a[0] = a[1] = 1;
	
	for(int i = 2; i < 10000000; i++)
	{
		if(a[i] == 0)
		{
			vec.push_back(i);
			
			for(int j = 2; (i*j) < 10000000; j++)
			{
				a[i*j] = 1;
			}
		}
	}
	
	long long prod = 1;
	for(int i = 0; i < (int)(vec.size()); i++)
	{
		prod = (prod * vec[i]) % MOD;
		ans.push_back(prod);
	}
}

int main()
{
	sieve();
	int t; 
	cin >> t;
	while(t--)
	{
		int n; cin >> n;
		cout << ans[n-1] << endl;
	}
	return 0;
}
