#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    while(t--) {
    	int n;
    	scanf("%d", &n);
    	set<int>s;
    	for(int i = 0; i < n; i++) {
    		int te;
    		scanf("%d", &te);
    		s.insert(te);
    	}
    	printf("%d\n", s.size());
    }
    return 0;
}
