#include <iostream>
#include <math.h>

using namespace std;
long long int gcd(long long int a,long long int b)
{
	if(a%b==0)
	return (b);
	else return(gcd(b,a%b));
}

int main()
{
    long long int a,b,n,total,lcm,g;
    int t,i;
    
    cin>>t;
    for(i=0;i<t;i++)
    {
	cin>>n>>a>>b;
	if(a==0||b==0)cout<<"-1"<<endl;
	else
	{
	g=gcd(a,b);
 	lcm=(a*b)/g;
	total=floor(n/a)+floor(n/b)-floor(n/lcm);
	cout<<total<<endl;
	}
	
    }
}
