name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
S = raw_input()
Q = int(raw_input())


while Q > 0:
    yesNo = map(int, raw_input().split())
    yesNo[0] -= 1
    yesNo[1] -= 1
    if S[yesNo[0] % len(S)] == S[yesNo[1] % len(S)]:
        print "Yes"
    else:
        print "No"
    Q = Q-1
