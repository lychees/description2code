int sl[30];
int n;
string s;
int point[N];
int cnt[N];

void solve() {
	for(int i = 0; i < 26; i++) sl[i] = 0;
	cin >> n >> s;
	rep(i, 0, sz(s))
		sl[s[i] - 'a']++;
	int win = 1;
	for(int i = 1; i <= n; i++) {
		point[i] = 0;
		cin >> s;
		cnt[i] = s.size();
		rep(j, 0, sz(s))
			point[i] += sl[s[j] - 'a'];
		if (point[i] > point[win]) {
			win = i;
		} else if (point[i] == point[win]) {
			if (cnt[i] < cnt[win]) {
				win = i;
			}
		}
	}
	cout << win << "\n";
}
