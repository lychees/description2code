#include <iostream>
#include <string>
//#include <ctype>

using namespace std;

int main()
{
	while(1)
	{
    string str;
    cin>>str;
    string a="pragyan";
    int c=0;
    for(int i=0;a[i];++i)
    {
    	if(tolower(str[i])==a[i]) c++;
    	else break;
    	
    }
    
    if(c==a.length())
    {
    cout<<str<<"\n";
    break;
    }
    else cout<<str<<"\n";
	}
    return 0;
}
