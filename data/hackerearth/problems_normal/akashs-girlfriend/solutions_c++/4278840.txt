#include <iostream>
using namespace std;

#define M 1000000009

long long modexp(long long x, long long y){
	long long ans = 1;
	while(y){
		if(y&1){
			ans = (ans*x)%M;
		}
		x = (x*x)%M;
		y >>= 1;
	}
	return ans;
}

int main()
{
    long long t,n;
    cin >> t;
    while(t--){
    	cin >> n;
    	cout << (((modexp(10,n) - modexp(8,n) + M)%M)*modexp(2,M-2))%M << endl;
    }
    return 0;
}
