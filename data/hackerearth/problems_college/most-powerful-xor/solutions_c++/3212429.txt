
#include <iostream>
#include <math.h>
using namespace std ;
int main()
{
      int t ;
      cin >> t ;
      while ( t--)
      {
         long long  int l ;
          cin >> l ;
         long long  int r ;
          cin >> r ;
          int i ;
         long long  int a[100000] ;
        long long   int k = 0 ;
          for ( i = l; i<= r ; i++)
          {
              for (long long  int j = i ; j <= r ; j++)
              {
                  a[k] = i^j ;
                  k = k+1 ;
              }
          }
         long long  int max = 0 ;

         // cout << endl;
          for ( i = 0; i<k ; i++)
          {
              if ( max < a[i])
              {
                  max = a[i] ;
              }
          }

          cout << max <<endl;

      }
      return 0 ;
}
