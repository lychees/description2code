#include<iostream>
#include<set>
using namespace std;
#define mk make_pair
int main()
{
	int i,j,n,k;
	cin>>n>>k;
	int a[n];
	set<pair<int,int> >s;
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0,j=k;i<n-k&&j<n;i++,j++)
	{
		if((abs(a[i]-a[j])%k==0)||((a[i]+a[j])%k==0))
		s.insert(mk(a[i],a[j]));
	}
	cout<<s.size()<<"\n";
	return 0;
}
