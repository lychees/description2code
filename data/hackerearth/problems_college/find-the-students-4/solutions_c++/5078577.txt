#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int n,i,j,max=0;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    {
   cin>>a[i];
    if(a[i]>max)
    {
    	max=a[i];
    	j=i;
    }
    }
    int count=n;
    while(a[j]>0)
    {
    	cout<<count<<endl;
    	for(i=0;i<n;i++)
    	a[i]=a[i]-2;
    	count=0;
    	for(i=0;i<n;i++)
    	{
    		if(a[i]!=0 && a[i]>0)
    		count++;
    	}
    	//if(count==1)
    	//break;
    }
    return 0;
}
