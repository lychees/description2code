#include <bits/stdc++.h>
using namespace std;

long long int fn(long long int a,long long int b,long long int mod) {
	if(b == 0) return 1;
	else if(b == 1) return a;
	if(b&1) return a*fn((a*a)%mod,(b-1)/2,mod)%mod;
	else return fn((a*a)%mod,b/2,mod);
}

int main()
{
    int t;
    cin >> t;
    int MOD = 1000000007;
    while(t--) {
    	int n, k;
    	cin >> n >> k;
    	long long int arr[k][n];
    	for(int i = 0; i < k; i++) {
    		for(int j = 0; j < n; j++) {
    			cin >> arr[i][j];
    		}
    	}
    	long long int maxi = -1;
		long long int mini = 1000000000;
		long long int  x = 1000000000;
		long long int y = -1;
    	for(int i = 1; i < pow(2,n); i++) {
    		long long int mid = 1,count=0;
    		for(int j = 0; j < n; j++) {
    			if(i & (1 << j)) {
    				long long val = 0;
    				for(int m = 0; m < k; m++) {
    					val += arr[m][j];
    					val %= MOD;
    				}
    				mid *= val;
    				mid %= MOD;
    				count++;
    			}
    		}
    		long long int p = fn(count, MOD - 2, MOD);
    		mid *= p;
    		mid %= MOD;
    		if(mid > maxi) {
    			maxi = mid;
    			x = count;
    		} 
			else if(mid == maxi) {
    			if(x > count)
    			x = count;
    		}
    		if(mid < mini) {
    			mini= mid;
    			y = count;
    		} 
			else if(mid == mini) {
    			if(y > count)
    			y = count;
    		}
    	}
    	long long int ans1,ans2;
    	ans1 = maxi^mini;
    	ans2 = x^y;
    	cout << ans1 << " " << ans2 << endl;
    }
    return 0;
}
