#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
    int test,n,k;
    cin>>test;
    while(test--){
    	scanf("%d",&n);
    	int dp[1000000] = {0};
    	for(int i = 0;i < n;++i){
    		scanf("%d",&k);
    		if(i == 0)dp[i + 1] = k;
    		else dp[i + 1] = max(k + dp[i - 1],dp[i]);
    	}
    	printf("%d\n",dp[n]);
    }
    return 0;
}
