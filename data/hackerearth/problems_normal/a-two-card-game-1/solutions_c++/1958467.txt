#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	long long n,i,draw=0,same=1,tot;
    float p;
    cin>>n;
    long A[n];
    for(i=0;i<n;++i)
    	cin>>A[i];
    sort(A,A+n);
    for(i=1;i<n;++i)
    {
    	if(A[i] == A[i-1])
    		same++;
    	else
    	{
    		draw += same*(same-1);
    		same = 1;
    	}
    }
    tot = n*(n-1);
    p = (tot-draw)/(tot*2.0);
    printf("%.6f %.6f",p,p);
    return 0;
}
