#include <iostream>
using namespace std;

#define MAX 10001

typedef long long LL;

LL dp[21];

LL fib(LL X)
{
	if(X==1)
	 return 1;
	if(X==2)
	 return 1;
	return (fib(X-1)%MAX+fib(X-2)%MAX)%MAX; 
}	

void store()
{
	LL i;
  for(i=1;i<=20;i++)
   dp[i]=fib(i)%MAX;
} 

int main()
{
    store();
    int terms;
    scanf("%d",&terms);
    int i;
    for(i=1;i<=terms;i++)
     printf("%d\n",dp[i]);
    return 0;
}
