#include <iostream>
#define ll long long int 
#define mod 1000000007
using namespace std;

ll dp[1001000][11];

int main()
{
	int t;
    cin>>t;
    int i,j;
    for(j=1;j<10;j++)
    	dp[1][j]=1;
    for(int i = 2 ;i<=1000000;i++){
    	for(int j= 0;j<10;j++){
    		if(j+1<10)
    			dp[i][j] = (dp[i][j] + dp[i-1][j+1])%mod;
    		if(j-1>=0)
    			dp[i][j] = (dp[i][j] + dp[i-1][j-1])%mod;
    	}
    }
    while(t--){
    	int n;
    	cin>>n;
    	ll ans = 0;
    	if(n==1)	cout<<"10\n";
    	else
    	{
    		for(int i=0;i<10;i++)
    			ans = (ans+dp[n][i])%mod;
    		cout<<ans<<"\n";
    	}
    }
    return 0;
}
