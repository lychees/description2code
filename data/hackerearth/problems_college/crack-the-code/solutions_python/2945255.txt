for _ in range(int(input())):
    k , l= map(int, raw_input().split())
    a = 0
    b = 1
    if b < 0:
        b = 0
    f = 0
    if k == 1:
        count = 1
    else:
        
        count = 0
    while f <= l:
        if f >= k:
            count = count +1
        f = a + b
        a = b
        b = f
    if count % 2 == 0:
        print 'VALID'
    else:
        print 'INVALID'
