#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int n;
vector< string > v;
int m[30][30];
int in[30];
int out[30];
int used[30];
void Scan()
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		string s;
		cin>>s;
		v.push_back(s);
		for(int j=0;j<s.size();j++)
			used[s[j]-'a']=1;
	}
	for(int i=0;i<26;i++)
		for(int j=0;j<26;j++)
			m[i][j]=0;
}

void Solve(int p,int s,int e)
{
	if(s==e) return;
	vector<int> tmp;
	int prev=s;
	int i=s;
	while(1)
	{
		if(p<v[prev].size()) tmp.push_back(v[prev][p]-'a');
		while(p<v[i].size() && v[i][p]==v[prev][p])
		{
			i++;
			if(i>e) break;
		}
		Solve(p+1,prev,i-1);
		if(i>e) break;
		prev=i;
	}
	for(int j=0;j<tmp.size()-1;j++)
	{
		m[tmp[j]][tmp[j+1]]=1;
	}
	return;
}

void Print()
{
	for(int i=0;i<26;i++)
	{
		int f1=0,f2=0;
		for(int j=0;j<26;j++)
		{
			if(m[j][i]==1) f1++;
			if(m[i][j]==1) f2++;
		}	
		in[i]=f1;
		out[i]=f2;
	}		
	
	while(1)
	{
		string s;
		for(int i=0;i<26;i++)
		{
			if(in[i]==0 && used[i]==1)
			{
				s+=i+'a';
				used[i]=0;
			}
		}
		if(s.size()==0) break;
		cout<<s<<endl;
		for(int i=0;i<s.size();i++)
		{
			int k=s[i]-'a';
			for(int j=0;j<26;j++)
			{
				if(m[k][j]==1) in[j]--;
			}
		}
	}
}
void Out()
{
	Solve(0,0,n-1);
	/*for(int i=0;i<26;i++)

		printf("\n");
	}*/
	Print();
}

int main()
{
    Scan();
    Out();
    return 0;
}
