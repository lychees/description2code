#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n;
		cin >> n;
		vector<int> a(n),b(n);
		for(int i=0;i<n;i++)
		{
			cin >> a[i] >> b[i];
		}
		map<int,int> mymap;
		int cnt=0;
		for(int i=0;i<n;i++)
		{
			mymap[a[i]]+=1;
			if(mymap[b[i]]==0)
			{	
				cnt++;
			}
			else
			{
				mymap[b[i]]-=1;
			}
		}
		cout << cnt << endl;
	}
	return 0;
}
