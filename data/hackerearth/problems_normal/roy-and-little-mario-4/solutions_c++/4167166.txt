#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
const LL mod = 1000000007;
LL dp[100008];
int main()
{
   LL t,n,i;
   dp[0] = 0;
   dp[1] = 1;
   dp[2] = 2;
   dp[3] = 4;
   for(i = 4;i <= 100005;i++)
   dp[i] = (dp[i-1] + dp[i-2] + dp[i-3])%mod;
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	cout << dp[n]<<endl;
   }
}
