    ll dp[2][1005];
    ll black[1005],white[1005];
    ll ans;
     
    int main(){
        //fr; fw;
        int n,m,x,y;
        char ch;
        cin >> n >> m >> x >> y;
        for(int i = 1; i <= n; i++){
            for(int j = 1; j <= m; j++){
                cin >> ch;
                if(ch == '#'){
                    black[j]++;
                    white[j] = n - black[j];
                }
            }
        }
     
        for(int i = 1; i <= m; i++){
            black[i] += black[i-1];
            white[i] += white[i-1];
        }
     
        fill(dp[0], dp[0] + 1001, INT_MAX);
        fill(dp[1], dp[1] + 1001, INT_MAX);
     
        dp[0][0] = 0;
        dp[1][0] = 0;
     
        for(int i = x; i <= m; i++){
            for(int j = i-x; j >= max(i-y, 0); j--){
                dp[0][i] = min(dp[0][i], white[i] - white[j] + dp[1][j]);
                dp[1][i] = min(dp[1][i], black[i] - black[j] + dp[0][j]);
            }
        }
     
        ans = min(dp[0][m], dp[1][m]);
     
        if(ans == INT_MAX)
            cout << "0";
        else
            cout << ans;
        return 0;
    }
