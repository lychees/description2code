#include <cstdio>
#include <iostream>
#include <string>

using namespace std;

const string RESULT[2] = {"ALICE", "BOB"};
const int MAX_ELEMENT = 100111;
bool isPrime[MAX_ELEMENT];
int a[MAX_ELEMENT];
int nPrime;
int prime[MAX_ELEMENT];

void sieve()
{
    for (int i = 0; i < MAX_ELEMENT; i++)
        isPrime[i] = true;

    isPrime[1] = false;
    for (int i = 2; i < MAX_ELEMENT; i++)
        if (isPrime[i])
            for (int j = i; j <= MAX_ELEMENT / i; j++)
                isPrime[i * j] = false;

    nPrime = 0;
    for (int i = 2; i < MAX_ELEMENT; i++)
    {
        if (isPrime[i])
        {
            prime[nPrime] = i;
            nPrime++;
        }
    }
}

int count(int a)
{
    int curr = 0;
    int cnt = 0;
    while (a != 1)
    {
        if (isPrime[a])
        {
            cnt++;
            break;
        }

        while (a % prime[curr] == 0)
        {
            a /= prime[curr];
            cnt++;
        }
        curr++;
    }

    return cnt;
}

int main()
{
    sieve();
    int n;
    scanf("%d", &n);
    for (int i = 0; i < n; i++)
        scanf("%d", &a[i]);

    int nim_sum = count(a[0]);
    for (int i = 1; i < n; i++)
        nim_sum = nim_sum ^ count(a[i]);

    printf("%s", nim_sum == 0 ? "BOB" : "ALICE");

    return 0;
}
