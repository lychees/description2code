#include <iostream>
using namespace std;
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
        int n,k,coun=0;
        cin>>n>>k;
        int a[n];
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
        }
        for(int i=0;i<n;i++)
        {
            if(a[i] <= 0)
                coun++;
        }
        if(k <= coun) cout<<"NO"<<endl;
        else cout<<"YES"<<endl;
    }
}
