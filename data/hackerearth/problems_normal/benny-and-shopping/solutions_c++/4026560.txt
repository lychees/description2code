#include <iostream>
#include<string>
using namespace std;

int main()
{
    int n ;
    string s ;
    cin >> n ;
    getline(cin,s);
    for(int i =0 ;i<n ;i++){
    	getline(cin,s);
    	int k = 0;
    	while(s[k]!='$')k++;
    	string s1 = "$" ;
    	k++;
    	while((s[k]==' '|| ('0' <= s[k] && s[k]<= '9'))&&(k<s.length())){
    		if(s[k]!=' ')s1.push_back(s[k]);
    		k++;
    	}
    	int j = 1 ;
    	while(j<s1.length()&&s1[j]=='0') s1.erase(s1.begin()+1) ;
    	if(j==s1.length()) cout << "$0\n" ;
    	else cout << s1 << "\n" ;
    	
    }
    return 0;
}
