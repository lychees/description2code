#include<bits/stdc++.h>
using namespace std;
#define mp pair<int,int>
#define fst first
#define snd second

int n,m,q,a,b,cnt = 1,pre[100001],low[100001];
bitset<100001> visit, bdg;
vector<mp> ady[100001];

void bridges(int nd,int p){
    visit[nd] = true;
    pre[nd] = low[nd] = cnt++;
    int tam = ady[nd].size();

    for(int i=0;i<tam;i++){
        if(ady[nd][i].fst == p) continue;

        if(!visit[ady[nd][i].fst]){
            bridges(ady[nd][i].fst,nd);
            low[nd] = min(low[nd],low[ady[nd][i].fst]);
            if(pre[nd] < low[ady[nd][i].fst])
                bdg[ady[nd][i].snd] = true;
        }
        else{
            low[nd] = min(low[nd], pre[ady[nd][i].fst]);
        }
    }
}

int main(){
    scanf("%d %d",&n,&m);

    for(int i=1;i<=m;i++){
        scanf("%d %d",&a,&b);
        ady[a].push_back(mp(b,i));
        ady[b].push_back(mp(a,i));
    }

    for(int i=1;i<=n;i++)
        if(!visit[i])   bridges(i,i);

    scanf("%d",&q);
    for(int i=0;i<q;i++){
        scanf("%d",&a);

        if(bdg[a])    printf("Unhappy\n");
        else    printf("Happy\n");
    }
}
