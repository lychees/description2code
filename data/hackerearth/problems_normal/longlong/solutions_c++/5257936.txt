#include <stdio.h>
#include <iostream>
#include <vector>

using namespace std;

typedef long long int lld;

string s;
lld tam;
lld M = 922337203687;
lld B = 26;

lld int_mod(lld a, lld b){
    return (a % b + b) % b;
}

bool rk(lld sz){
    //    DEBUG("sz ", sz);

    lld hp = 0;
    lld E = 1;

    for(lld i = 0; i < sz - 1; i++){
	E = (E*B)%M;
    }

    for(lld i = 0; i < sz; i++){
	hp = int_mod(hp * B + s[i], M);
    }

    vector<lld> v;

    v.push_back(hp);
    for(lld i = sz; i < tam; i++){
	hp = int_mod(hp - int_mod(s[i - sz] * E, M), M);
	hp = int_mod(hp * B, M);
	hp = int_mod(hp + s[i], M);

	v.push_back(hp);
    }

    sort(v.begin(), v.end());

    lld tamv = (lld) v.size();
    
    for(lld i = 0; i < (tamv-1); i++){
	if(v[i] == v[i+1]){
	    v.clear();
	    return true;
	}
    }

    v.clear();

    return false;
}


int bb(int lo, int hi){
    int mid;
    int ans = 0;

    while(lo < hi){
	mid = lo + (hi - lo) / 2;

	if(!rk(mid)){
	    hi = mid;
	}else{
	    lo = mid + 1;
	    if(ans < mid) ans = mid;
	}
    }

    return ans;
}

int main(){

    cin >> s;

    tam = (lld) s.size();

    cout << bb(1, tam) << endl;

}
