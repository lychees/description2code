#include<bits/stdc++.h>
using namespace std;

int arr[26];
int brr[26];

int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */
    int t;
    cin>>t;
    while(t--){
        string a;
        string b;
        cin>>a>>b;
        if(b.size()<a.size()){
            cout<<"NO"<<endl;
            continue;
        }
        for(int i=0;i<26;i++)
            arr[i]=brr[i]=0;
        for(int i=0;i<a.size();i++)
            arr[a[i]-'a']++;
        for(int i=0;i<a.size();i++)
            brr[b[i]-'a']++;
        int i;
        int j=a.size();
        for( i=0;i<26;i++){
            if(arr[i]!=brr[i])
                break;
        }
        if(i==26){
            cout<<"YES"<<endl;
            continue;
        }
        int flg=0;
        while(j<b.size()){
            brr[b[j]-'a']++;
            brr[b[j-a.size()]-'a']--;
            int ii;
            for( ii=0;ii<26;ii++){
                if(arr[ii]!=brr[ii])
                    break;
            }
            if(ii==26){
                    flg=1;
                break;
            }
            j++;
        }
        if(flg)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
}
