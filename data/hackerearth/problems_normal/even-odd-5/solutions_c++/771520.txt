#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
int t;
//cin>>t;
scanf("%d",&t);
while(t--)
{
unsigned long long r,c,odd_ans=0,even_ans=0;
//cin>>r>>c;
scanf("%llu %llu",&r,&c);
if(r<=c)
{
if(r%2)
{
odd_ans=(((r+1)/2)*(1+r))/2;
even_ans=((r/2)*(1+r))/2;
}
else
{
odd_ans=((r/2)*(r))/2;
even_ans=((r/2)*(2+r))/2;
}
}
else
{
if(c%2)
{
odd_ans=(((c+1)/2)*(1+c))/2;
even_ans=((c/2)*(1+c))/2;
if((r-c)%2)
{
even_ans+=(((r-c+1)/2)*c);
odd_ans+=(((r-c)/2)*c);
}
else
{
even_ans+=(((r-c)/2)*c);
odd_ans+=(((r-c)/2)*c);
}
}
else
{
odd_ans=((c/2)*(c))/2;
even_ans=((c/2)*(2+c))/2;
if((r-c)%2)
{
even_ans+=(((r-c)/2)*c);
odd_ans+=(((r-c+1)/2)*c);
}
else
{
even_ans+=(((r-c)/2)*c);
odd_ans+=(((r-c)/2)*c);
}
}
}
printf("%llu %llu\n",even_ans,odd_ans);
//cout<<even_ans<<" "<<odd_ans<<endl;
}
return(0);
}
