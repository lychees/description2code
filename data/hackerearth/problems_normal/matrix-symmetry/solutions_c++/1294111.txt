#include <iostream>
//#include<algorithm>
#include<cstring>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {int n;
    cin>>n;
    string s[n];
    for(int i=0;i<n;i++)
    cin>>s[i];
    bool hf=false;
    bool vf=1;
    int j=0;int k=n-1;
    while(j<k)
    {
        if(s[j]!=s[k])
        break;
        else
        {
            j++;k--;
        }
    }
    if(k<=j)
    hf=1;
    for(int i=0;i<n;i++)
    {
        int p=0;int q=s[i].size()-1;
        while(p<q)
        {
            if(s[i][p]!=s[i][q])
           {
               vf=0;break;
           } 
            else
            {p++;q--;}
        }
    }
 if(vf==1&&hf!=1)
 cout<<"VERTICAL"<<endl;
 else if(vf==1&&hf==1)
 cout<<"BOTH"<<endl;
 else if(vf!=1&&hf==1)
 cout<<"HORIZONTAL"<<endl;
 else
 cout<<"NO"<<endl;
    }

return 0;
}
