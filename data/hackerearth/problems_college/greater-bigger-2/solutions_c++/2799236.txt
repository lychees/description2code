#include <cstdio>
#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main()
{
    int T;
    cin >> T;
    string input,temp;
    while(T--)
    {
        cin >> input;
        temp=input;

        sort(temp.begin(),temp.end(),greater<char>());
        if(input == temp)
            cout << "no answer\n";
        else
        {
            next_permutation(input.begin(),input.end());
            cout << input << endl;
        }
    }
    return 0;
}
