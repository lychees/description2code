vector<int> v[100005];
ll int ans=0;
void recurse(int pos,int val)
{
//	printf("%d\n",pos);
	if(pos!=0)
	{
		ans+=val;
	}
	int n=v[pos].size(),i;
	if(n==0)
		return;
	else
	{
		for(i=0;i<n;i++)
		{
			recurse(v[pos][i],val+1);
		}
	}
}
