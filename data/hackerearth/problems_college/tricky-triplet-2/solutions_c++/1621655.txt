#include <iostream>
using namespace std;

int main()
{int t;int i,j,k,sum=0;
cin>>t;int arr[t];
for(i=0;i<t;i++)
{cin>>arr[i];
    }
    for(i=0;i<t-2;i++)
    {for(j=i+1;j<t-1;j++)
    {for(k=j+1;k<t;k++){
    	if(arr[k]-arr[j]==arr[j]-arr[i])
    	{sum+=1;}
    	}}}
    	cout<<sum<<endl;
    return 0;
}
