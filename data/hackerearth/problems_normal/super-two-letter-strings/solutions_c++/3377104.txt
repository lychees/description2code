#include <iostream>
using namespace std;

const int MOD=1000000007;

long long dp[16384][2];

int calc(int n, int p)
{
    for (int i=0; i<16384; i++) dp[i][0]=dp[i][1]=0;
    dp[1][0]=1;
    for (int i=2; i<=n; i++)
    {
        dp[i][0]=(dp[i-1][0]+dp[i-1][1])%MOD;
        int j=i-1;
        while (j>=1 and i-j<p)
        {
            dp[i][1]=(dp[i][1]+dp[j][0])%MOD;
            j--;
        }
    }
    return (dp[n][0]+dp[n][1])%MOD;
}

int main()
{
    int t, n, p;
    cin>>t;
    while (t-->0)
    {
        cin>>n>>p;
        int ans=calc(n, p);
        cout<<ans<<endl;
    }
    return 0;
}
