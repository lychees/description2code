#include<iostream>
#include<cstring>
using namespace std;

int main()
{
	int T=0;
    cin >>T;
    while(T--){
    		int N=0;
    		cin >>N;
    		unsigned int arr[N];
    		memset(arr,0,sizeof(arr));
    		unsigned long long int sum =0;
    		for(int i =0;i<N;i++){
    			unsigned int element =0;
    			cin >>element;
    			arr[i] =element;
    			sum +=element;
    		}
    		
    		long long int M;
    		cin >>M;
    		M = (M-1) % sum;
    		M +=1;
    		int i=0;
    		for(i=0;i<N;i++){
    			M -=arr[i];
    			if(M<=0)
    				break;
    		}
    		
    		cout <<i+1<<endl;
    }
    return 0;
}
