#include <bits/stdc++.h>

int countOne(string s) {
    int cnt = 0;

    for(int i = 0; i < s.size(); i++) {
        if(s[i] == '1') {
            cnt++;
        }
    }

    return cnt;
}

int countZero(string s) {
    int cnt = 0;

    for(int i = 0; i < s.size(); i++) {
        if(s[i] == '0') {
            cnt++;
        }
    }

    return cnt;
}

int main(){

    string a, b;
    int a0, a1, b0, b1, diff0, diff1;

    int t;
    cin >> t;
    int cas = 1;
    while(t--) {

        int step = 0;

        cin >> a;
        cin >> b;

        a1 = countOne(a);
        a0 = countZero(a);

        b1 = countOne(b);
        b0 = countZero(b);

        //cout <<a0 <<" "<<b0 <<endl;

        if(a1 > b1 || a.size() != b.size()) {
            printf("Case %d: %d\n", cas, -1);
            cas++;
            continue;
        }

        if(a0 > b0) {
            diff0 = a0 - b0;
            //1
            for(int i = 0; i < a.size(); i++) {
                if(b[i] == '1' && a[i] == '0') {
                    a[i] = '1';
                    diff0--;
                    step++;
                 //   cout <<" 1 "<< a <<endl;
                }
                if(diff0 == 0)
                    break;

            }
            // 2
            if(diff0 > 0) {
                for(int i = 0; i < a.size(); i++) {
                    if(b[i] == '0' && a[i] == '0') {
                        a[i] = '1';
                        diff0--;
                        step++;
                     //   cout <<" 2 "<< a <<endl;
                    }
                    if(diff0 == 0)
                        break;
                }
            }

            a0 = b0;
            a1 = a1 + diff0;

        }

        diff0 = (b0 - a0);
        diff1 = (b1 - a1);

        for(int i = 0; i < a.size(); i++) {
            if(a[i] == '?') {

                if(diff0 > 0 && b[i] == '0') {
                    a[i] = '0';
                    diff0--;
                    a0++;
                    step++;
                }

                if(diff1 > 0 && b[i] == '1') {
                    a[i] = '1';
                    diff1--;
                    a1++;
                    step++;
                }

               // cout <<" 3 "<< a <<endl;

            }
        }

        if(diff0 > 0) {
            for(int i = 0; i < a.size(); i++) {
                if(a[i] == '?') {
                    a[i] = '0';
                    diff0--;
                    a0++;
                    step++;

                  //  cout <<" 4 "<< a <<endl;
                }
            }
        }
        if(diff1 > 0) {
            for(int i = 0; i < a.size(); i++) {
                if(a[i] == '?') {
                    a[i] = '1';
                    diff1--;
                    a1++;
                    step++;
                  //  cout <<" 5 "<< a <<endl;
                }
            }
        }

        a1 = countOne(a);
        a0 = countZero(a);

        b1 = countOne(b);
        b0 = countZero(b);

       // cout << " 6 "<< a <<endl;

       // cout <<" diff in '1' "<< b1 - a1 <<" diff in '0' " << b0 - a0 <<endl;


        int mis = 0;

        for(int i = 0; i < a.size(); i++) {
            if(a[i] != b[i])
                mis++;
        }

        step += (mis / 2);

        printf("Case %d: %d\n", cas, step);
        cas++;
        //cout <<step <<endl;
    }
}
