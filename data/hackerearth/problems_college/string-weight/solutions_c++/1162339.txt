#include <iostream>
#include<cstring>
using namespace std;

int main()
{
   int T;cin>>T;
   while(T--){
   	char s[10000];
   	cin>>s;
   	int len = strlen(s);
   	int n;
   	cin>>n;
   	char A[n];
   	for(int i=0;i<n;i++)
   	cin>>A[i];
   	long int wt=0;
   	for(int i=0;i<len;i++)
   	{bool x=true;
   		for(int j=0;j<n;j++)
   			if(s[i]==A[j])
   			{	x=false;
   				break;
   			}
   		if(x==true)wt+=s[i]-96;	
   	}
   	cout<<wt<<endl;
   }
}
