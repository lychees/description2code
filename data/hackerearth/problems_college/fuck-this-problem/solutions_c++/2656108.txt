#include<iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    getline(cin,s);
    int n,j=0;
    cin>>n;
    int l=s.length();
    for(int i=0;i<l;i++)
        {

                cout<<s[i];
                j++;
                if(j==n)
                {
                    cout<<endl;
                    j=0;
                }

        }

    return 0;
}
