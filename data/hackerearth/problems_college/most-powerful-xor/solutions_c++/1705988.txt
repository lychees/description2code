#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;
int main()
{
    int l,r,i,t,maxi,j;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&l,&r);
        maxi=l^r;
        for(i=l;i<=r;i++)
        {
            for(j=i+1;j<=r;j++)
            {
                if((i^j)>maxi)
                    maxi=i^j;
            }
        }
        printf("%d\n",maxi);
    }
    return 0;
}
