#include<bits/stdc++.h>
#define MOD 1000000009
using namespace std;
long long int power(long long base,long long p)
{
    long long int temp;
    if(p==0)return 1;
    temp=power(base,p/2);
    if(p%2==0)
    {
         temp=((temp%MOD)*(temp%MOD))%MOD;
         return temp;
    }
    else
    {
         temp=((base%MOD)*(((temp%MOD)*(temp%MOD))%MOD))%MOD;
         return temp;
    }
}
int main()
{
    long long int t,ans,N;
    scanf("%lld",&t);
    while(t--)
    {
        cin>>N;
        ans=power(26,(N+1)/2);
        cout<<ans<<"\n";

    }
}
