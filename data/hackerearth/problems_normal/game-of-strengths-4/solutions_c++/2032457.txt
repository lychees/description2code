#include <iostream>
#include <algorithm>
#define mod 1000000007
using namespace std;

int main()
{
	int t;
	cin>>t;
	long long a[100001],n;
	while(t--){
		cin>>n;
		long max=0;
		for(long i=0;i<n;i++){
			cin>>a[i];
			if(a[i]>max) max=a[i];
		}
		sort(a,a+n);
		long sum=0,p=n-1;
		for(long i=n-1;i>=0;i--){
			sum+=(a[i]*p);
			p-=2;
		}
		sum%=mod;
		cout<<(sum*max)%mod<<endl;
	}
	return 0;
}
