import sys
import math

T = int(sys.stdin.readline())
for i in xrange(T):
    N = int(sys.stdin.readline())
    line = sys.stdin.readline().split()
    x = int(line[0])
    y = int(line[1])
    dist = math.sqrt(x**2+y**2)
    sides = []
    sides.append(dist)
    for j in xrange(N):
        sides.append(float(sys.stdin.readline()))
    sides.sort()
    s = sum(sides[0:len(sides)-1])
    if s < sides[len(sides)-1]:
        print "Not able"
    else:
        print "Able"
