#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--){
    	string str;
    	cin>>str;
    	int s=str[0]-'a';
    	for(int i=0;i<str.length();i++){
    		str[i]=str[i]-s;
    		if(str[i]<97)
    			str[i]=str[i]+26;
    	}
    	cout<<str<<endl;
    }
    return 0;
}
