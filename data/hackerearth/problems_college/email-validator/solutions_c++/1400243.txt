#include<iostream>
#include<string.h>
 
using namespace std;
int main()
{
	int t,n,k,i,j,l,user,host,at,dot,last,len,flag,alpha;
	char str[100];
	cin>>t;
	
	while(t--)
	{
		user=at=host=dot=last=flag=alpha=0;
		len=strlen(str);
		l=strlen(str);
		cin>>str;
		len=strlen(str);
			for(i=0;i<16;i++)
			{
				if((str[i]>='a'&&str[i]<='z')||(str[i]>='A'&&str[i]<='Z')||(str[i]>='0'&&str[i]<='9')||(str[i]=='_'))
					user++;
				else
					break;
			}
		
			if(str[user]=='@')
				at++;
				
			for(i=user+at;i<16+user+at;i++)
			{
				if((str[i]>='a'&&str[i]<='z')||(str[i]>='A'&&str[i]<='Z'))
				alpha++;
				if((str[i]>='a'&&str[i]<='z')||(str[i]>='A'&&str[i]<='Z')||(str[i]>='0'&&str[i]<='9')||(str[i]=='-'))
					host++;
				else
					break;
			}
			if(str[user+host+at]=='.')
				dot++;
			
			for(i=host+user+at+dot;str[i];i++)
			{
				if(str[i]<'a'||str[i]>'z')
				{
					break;
				}
				else
					last++;
			}
			//cout<<user<<" "<<at<<" "<<host<<" "<<dot<<" "<<last<<" "<<alpha<<" "<<len<<endl;
		if(host<=16&&host>0&&dot==1&&at==1&&last>0&&last<=10&&user<=16&&user>0&&alpha&&(host+last+user+at+dot==len))
			cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
}
