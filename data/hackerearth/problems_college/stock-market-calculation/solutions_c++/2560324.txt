#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int T,i,j,k,A[1000001],N;
	cin>>T;
	while(T--)
	{
		cin>>N;
		for(i=0;i<N;i++)
		scanf("%ld",&A[i]);
		
		long int M[N];
		M[N-1]=A[N-1];
		for(i=N-2;i>=0;i--)
		{
			M[i]=max(M[i+1],A[i]);
		}
		long int sum=0;
		for(i=0;i<N;i++)
		{
			M[i]=M[i]-A[i];
			sum+=M[i];
		}
		cout<<sum<<"\n";
		
		
	}
}
