#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;cin>>n;
    int a[n],i;
    for(i=0;i<n;i++) cin>>a[i];
    int dp[n];
    dp[0]=1;
    int ans=0;
    for(i=1;i<n;i++)
    {
    	dp[i]=dp[i-1];
    	if(a[i]>a[i-1]) dp[i]++;
    	else dp[i]=1;
    }
    for(i=n-2;i>=0;i--)
    {
    	if(a[i]>a[i+1] && dp[i]<=dp[i+1])
    	{
    		dp[i]=dp[i+1]+1;
    	}
    }
    for(i=0;i<n;i++) ans+=dp[i];
    cout<<ans;
    return 0;
}
