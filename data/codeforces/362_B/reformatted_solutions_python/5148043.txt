n, m = map(int, raw_input().split())
if m == 0:
    print "YES"
else:
    num = sorted(map(int, raw_input().split()))
    ans = "YES"
    for i in num:
        if i == 1 or i == n:
            ans = "NO"
    for i in range(m-2):
        if num[i] == num[i+1]-1 and num[i+1] == num[i+2]-1:
            ans = "NO"
    print ans
