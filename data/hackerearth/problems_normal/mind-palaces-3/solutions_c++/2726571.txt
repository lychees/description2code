#include <iostream>
using namespace std;

int main()
{
    int n,m,a[1000][1000],i,j,x,y,t,q;
    cin>>n>>m;
    for(i=0;i<n;i++)
    {
    	for(j=0;j<m;j++)
    		cin>>a[i][j];
    }
    cin>>q;
    while(q--)
    {
    	cin>>t;
    	x=n-1;
    	y=0;
    	while(x>=0 && y<=m-1 && a[x][y]!=t)
    	{
    		if(t>a[x][y])
    			y++;
    		else
    			x--;
    	}
    	if(x<0 || y==m)
    		cout<<"-1 -1"<<endl;
    	else
    		cout<<x<<" "<<y<<endl;
    }
    return 0;
}
