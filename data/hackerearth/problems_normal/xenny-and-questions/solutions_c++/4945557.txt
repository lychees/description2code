#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int ar[100005];

int main()
{
	int t;
	scanf("%d", &t);
	while (t--)
	{
		int n, k, i;
		long long sum = 0LL;
		scanf("%d %d", &n, &k);
		for (i = 0; i<n; i++)
		{
			scanf("%d", &ar[i]);
		}
		sort(ar, ar + n);
		for (i = n - 1; k>0 && i >= 0; i--, k--)
		{
			sum += ar[i];
		}
		printf("%lld\n", sum);
	}
	return 0;
}
