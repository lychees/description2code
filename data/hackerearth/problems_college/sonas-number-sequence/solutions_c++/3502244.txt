#include <iostream>
using namespace std;
 
int main()
{
    //cout << "Hello World!" << endl;
    
    int arr[100],brr[100],j=0 , a,n ;
    signed int sum=0 ;
    cin>>n ;
    
    
    for(int i=0;i<n;i++ )
    { 
    	
    	cin>>a ;
    	if(0<a<=1000000000)
    	{arr[i]=a;}
    	else
    	{i--;}
    }
    
    brr[0]=arr[0] ;
    for(int i=1 ;i<n;i++)
    { 
    	j=0 ;
    	while(j!=i)
    	{
    		sum =sum+brr[j] ;
    		j++;
    	}
    	brr[j]=(i+1)*arr[i]-sum ;
    	sum=0;
    }
    
    
    
    for(int i=0;i<n;i++ )
    {
    	cout<<" "<<brr[i];
    }
    
    
    
    
    
    
    
    
    
    
    
    
    
    
    return 0;
}
