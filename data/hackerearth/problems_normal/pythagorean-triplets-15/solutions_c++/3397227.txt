#include <bits/stdc++.h>
using namespace std;
#define ll long long


int main()
{
  ll n,a,b,c;
  int t;
  scanf("%d", &t);
  ll ans = 0;
  while(t--) {
    scanf("%lld",&n);
    ans = 0;
    for(int i = 2; i < sqrt(n)+1; i++) {
      for(int j = 1; j < i; j++) {
        if ((__gcd(i,j) == 1) && ((i-j) % 2) && ((i*i + j*j) <= n)) {
        int k = 1;
        a = i * i - j * j;
        b = 2 * i * j;
        c = i * i + j * j;
        while(k * c <= n) {
          k++;
          ans++;
        }
      }
      }
    }
    printf("%lld\n", ans);
  }
return 0;
}
