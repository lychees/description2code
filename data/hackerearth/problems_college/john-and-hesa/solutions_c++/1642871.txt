#include<iostream>
#include<cmath>
using namespace std;
typedef long long int ll;
int main()
{
    ll t,x,p;
    cin>>t;
    while(t--)
    {
        cin>>x>>p;
        ll n=0,r=0;
        if(x==1&&p==1)
            cout<<1<<" "<<0<<'\n';
        else
        {
           r=x%p;
           while(r==0)
           {
             n++;
             x=x/p;
             r=x%p;
           }
           if(x==1)
             cout<<n<<" "<<0<<'\n';
           else
             cout<<n<<" "<<x<<'\n';
        }
    }
    return 0;
}
