#include<bits/stdc++.h>

using namespace std;
long long int MOD=1000000007;

int main()
	{
		int t;
		cin>>t;
		while(t--)
			{
				int n;
				cin>>n;
				long long int a[n+1];
				for(int i=0;i<n;i++)
					scanf("%lld",&a[i]);
				sort(a,a+n);
				long long int prod=n-1;
				long long int sum=0;
				for(int i=n-1;i>=0;--i)
					{
						sum=sum+prod*a[i];
						prod-=2;
					}
				sum=sum%MOD;
				long long int res=a[n-1]%MOD;
				res=(res*sum)%MOD;
				cout<<res<<endl;
			}
		return 0;
	}
