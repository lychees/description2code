#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int t;
    cin>>t;
    while(t--)
    {
    	long long int k,i,j;
    	cin>>k;
    	long long int a[k],ba[k],b[k],len;
    	len=0;
    	for(i=0;i<k;i++)
    	cin>>a[i];
    	for(i=0;i<k;i++)
        cin>>ba[i];
        sort(a,a+k);
        sort(ba,ba+k);
        for(i=0;i<k;i++)
        {
        	b[i]=ba[k-i-1];
        	
        }
        for(i=0;i<k;i++)
         {
         	
         		if(a[i]%b[i]==0||b[i]%a[i]==0)
         		len++;
         	
    
         }
       cout<<len<<endl; 
    }
    
    return 0;
}
