#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[1010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	a[0]=1;
	a[1]=1;
	int i;
	for(i=2;i<=1000;i++)
	{
		a[i]=a[i-1]+a[i-2];
	}
	int test;
	cin>>test;
	while(test--)
	{
		int n;
		cin>>n;
		cout<<a[n]<<endl;
	}
	return 0;
}
