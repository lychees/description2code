#include<bits/stdc++.h>

//perfect copy

using namespace std;

int main()
{

    long int t;
    string s1;
    string s2;
    cin >> t;
    int i;
    int counti = 0;
    int sum = 0;
    while(t--)
    {
        sum = 0;
        cin >> s1;
        cin >> s2;
        counti = 0;
        for(i = 0 ; i < s1.length(); i++)
        {
            sum = (s1[i] - '0')+ (s2[i] - '0');
            if(sum >= 10)
            {
                counti++;
            }
        }
        if(counti == 1)
        {
            cout << "1 carry operation" <<endl;
        }
        else if(counti == 0)
        {
            cout << "No carry operation"<<endl;
        }
        else if(counti > 1)
        {
            cout <<counti << " carry operations" <<endl;
        }
    }
    return 0;
}
