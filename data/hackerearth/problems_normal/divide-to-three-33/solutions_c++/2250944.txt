#include <iostream>
using namespace std;
int max(int a,int b)
{
	if(a>=b)
	return a;
	else
	return b;
}
void recur(int arr[],int n,int index,int a,int b,int c,int *min)
{
	if(index==-1)
	{
		int temp=max(max(a,b),c);
		if(temp<*min)
		*min=temp;
		return ;
	}
	a+=arr[index];
	recur(arr,n,index-1,a,b,c,min);
	a-=arr[index];
	b+=arr[index];
	recur(arr,n,index-1,a,b,c,min);
	b-=arr[index];
	c+=arr[index];
	recur(arr,n,index-1,a,b,c,min);
	c-=arr[index];
}
int main()
{
	int n,i,arr[15]={0};
    cin>>n;
    for(i=0;i<n;i++)
    cin>>arr[i];
    int min=12000;
    recur(arr,n,n-1,0,0,0,&min);
    cout<<min;
    return 0;
}
