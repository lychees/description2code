#include <iostream>
using namespace std;

 
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {long int n;
    cin>>n;
    long int a[n];
    long int i,sum=0;
    for(i=0;i<n;i++)
    {cin>>a[i];
    sum=sum+a[i];
    }
    if((sum%n)==0)
    {cout<<"YES"<<endl;
    cout<<sum/n<<" "<<"0"<<endl;
    }
    else
    {cout<<"NO"<<endl;
    cout<<sum/n<<" "<<sum%n<<endl;
    }
    }
    return 0;
}
