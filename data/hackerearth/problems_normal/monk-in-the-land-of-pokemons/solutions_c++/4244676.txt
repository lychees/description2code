#include <iostream>
using namespace std;

int main()
{
    int t,n,x,y,count;
    cin>>t;
    while(t--){
    	count = 0;
    	long a[1000001] = {0};
    	
    	cin>>n;
    	
    	for(int i=1;i<=n;i++){
    		cin>>x>>y;
    		//cout<<x<<" "<<y<<endl;
    		a[x]++;
    		
    		if(a[y] > 0)
    			a[y]--;
    		else
    			count++;
    		
    	}
    	cout<<count<<endl;
    }
    return 0;
}
