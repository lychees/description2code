#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int F, G, S, P, C, lb, s, p, c, ans = 0;
		int foodleft, goldleft;
		scanf("%d%d%d%d%d%d",&F,&G,&S,&P,&C,&lb);

		for( s = 0; s * 100 <= F; s++ )
		{
            for( p = 0; p * 125 + 100 * s <= F &&  p * 50 <= G; p++)
			{
				 foodleft = F - 100 * s - 125 * p;
				 goldleft = G - 50 * p;
				c=min(foodleft / 50, goldleft / 100);
				ans=max(ans, S * s + P * p + C * c);
			}
		}
		
		if(ans > lb)
		printf("%d\n",(ans-lb));
		else
		printf("-1\n");
	}
	
	return 0;
}
