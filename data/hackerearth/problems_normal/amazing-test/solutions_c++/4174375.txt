#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int arr[101];
bool vis[101];
int dp[100001][101];
int n,m;
int solve(int idx,int k){
    if( k > m)return -1e9;
    if(idx == n){
        return 0;
    }
    int &ret = dp[k][idx];
    if(ret!=-1)
        return ret;
    ret = solve(idx+1,k);
    ret = max(ret,solve(idx+1,k+arr[idx])+arr[idx]);
    return ret;
}
void pr(int idx,int k){
    if( k > m)return;
    if(idx == n){
        return;
    }
    int &ret = dp[k][idx];

    int a = solve(idx+1,k);
    if(a == ret){
        pr(idx+1,k);
    }
    else{
        vis[idx] = true;
        pr(idx+1,k+arr[idx]);
    }
}
int main(){
    int t;
    cin >> t;
    while(t--){
        cin >> n >> m;
        for(int i=0;i<n;i++){
            cin >> arr[i];
        }
        memset(dp,-1,sizeof dp);
        memset(vis,false,sizeof vis);
        solve(0,0);
        pr(0,0);
        int k = 0;
        for(int i=0;i<n;i++){
            if(!vis[i])
                k+= arr[i];
        }
        if(k <= m)cout << "YES\n";
        else cout << "NO\n";
    }
    return 0;
}
