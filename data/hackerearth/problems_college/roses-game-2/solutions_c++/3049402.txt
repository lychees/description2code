#include <iostream>
#include <cmath>
using namespace std;

int main()
{
   long int N,t;
   cin>>t;
   while(t--){
   		long int a[1001],flag;
   		cin>>N;
   		for(long int i=0;i<N;i++)
   			cin>>a[i];
   		
   		//logic
   		for(long int i=0;i<N;i++){
   			if(a[i] == -1)
   			{
   				flag =0;
   				//break;
   				continue;
   			}
   			else if(a[a[i]] == i)
   			{
   				flag =1;
   				break;
   			}
   			
   			
   		}
   		if(flag == 1)
   			cout<<"Lose\n";
   		else if(flag == 0)
   			cout<<"Win\n";
   }
   return 0;
}
