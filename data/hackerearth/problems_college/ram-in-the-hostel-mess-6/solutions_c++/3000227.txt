int count(int n)
{
	int a[n],b[n];
	a[0]=1;
	b[0]=1;
	for(int i=1;i<n;i++)
	{
		a[i]=a[i-1]+b[i-1];
		b[i]=a[i-1];
	}
	return a[n-1]+b[n-1];
}

int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	int col;
    	cin>>col;
    	int c=count(col);
    	cout<<c*c<<endl;
    }
    return 0;
}
