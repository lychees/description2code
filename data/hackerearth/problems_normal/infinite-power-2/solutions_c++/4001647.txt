#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        scanf("%d",&n);
        int gt=0,car=0,t=n,mul=1;
        while(t)
        {
            int r=t%10;
            r=r+car;
            if(r&1)
            {
                r=r+1;
                gt=0;
            }
            if(r==10)
            {
                r=0;
                car=1;
            }
            else
             car=0;
            gt=mul*r+gt;
            mul=mul*10;
            t=t/10;
        }
        if(car)
        {
            if(car&1)
                car++;
            gt=mul*car+gt;
        }
        int bor=0,lt=0;
        t=n;
        mul=1;
        while(t)
        {
            mul=mul*10;
            t=t/10;
        }
        mul=mul/10;
        t=n;
       while(mul)
        {
            int r=t/mul;
            if(r&1)
            {
                r--;
                lt=mul*r+lt;
                mul=mul/10;
                break;
            }
            lt=mul*r+lt;
            t=t%mul;
            mul=mul/10;
        }
        while(mul)
        {
            lt=mul*8+lt;
            mul=mul/10;
        }

       if(n==lt)
        printf("Unlimited Power\n");
       else
       {
           int gcd=__gcd(n-lt,gt-n);
           printf("%d/%d\n",(gt-n)/gcd,(n-lt)/gcd);
       }
    }
}
