#include <iostream>
using namespace std;

int main()
{
    long long n,x,cnt;
    
    cin>>n;
    
    while(n>0)
    {
    	cin>>x;
    	cnt=0;
    	for(int i=1;i*i<x;i++)
    	{
    		if(x%i==0)
    		cnt+=2;
    		
    		if(cnt>=4)
    		break;
    	}
    	
    	if(cnt<4)
    	cout<<"NO"<<endl;
    	else cout<<"YES"<<endl;
    	
    	n--;
    }
    return 0;
}
