noOfCases = int(raw_input())


def dicipher(ciphered_text):
    # print ciphered_text
    # print "input -----> " + ciphered_text
    diciphered_text = ciphered_text[0]
    noOfShifts = 1

    for index in xrange(1, len(ciphered_text)):
        diciphered_text += shift_lower_case_letter(
            ciphered_text[index], noOfShifts)
        noOfShifts += 1
    return diciphered_text


def shift_lower_case_letter(letter, noOfShifts):
    ordinality_of_letter = ord(letter)
    if ordinality_of_letter >= 97 or ordinality_of_letter <= 122:
        shifted_result_ord = ordinality_of_letter + noOfShifts

        if shifted_result_ord < 97:
            shifted_result_ord += 26

        elif shifted_result_ord > 122:
            shifted_result_ord -= 26

        return chr(shifted_result_ord)
    return letter


for i in xrange(noOfCases):
    print dicipher(raw_input())
