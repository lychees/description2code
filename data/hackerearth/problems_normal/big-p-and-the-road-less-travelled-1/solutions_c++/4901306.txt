#include <bits/stdc++.h>
using namespace std;
const int MAX = 10000 + 7;
vector  <int > graph[MAX];
bool visited[MAX];
int dp[MAX];
int N;
int coun = 0;
long long dfs ( int v)
{
	 if( v == N)
	 return 1;
	 if(visited[v])
	 return dp[v];
	visited[v] = 1;
	long long ans = 0;
	for(int i = 0; i < (int)graph[v].size(); i++)
	{
		ans += dfs(graph[v][i]);
	}
	
	dp[v] = ans;
	return ans;
	
	
}
int main()
{
	int a, b;
	cin >> N;
	cin >> a >> b;
	while ( a !=0 && b != 0)
	{
		graph[a].push_back(b);
		cin >> a >>b;
	}
   
  cout <<  dfs(1);
  
    return 0;
}
