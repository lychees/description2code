#include<cstdio>


int main()
{

long long int n,m;

scanf("%lld %lld",&n,&m);
long long int c[m];
long long int arr[n];
for(long long int i=0;i<m;i++)
{
scanf("%lld",&c[i]);

}


for(long long int i=0;i<n;i++)
{
scanf("%lld",&arr[i]);

}

long long int curmax=c[arr[0]-1];


for(long long int i=1;i<n;i++)
{
long long int num=c[arr[i]-1];

    while(true)
    {
    bool flag=false;

    long long int k=0;
    curmax++;

    if(curmax%num==0)
    {
        for(long long int j=c[arr[i]+k];(arr[i]+k)<m;j=c[arr[i]+(++k)])
        {

        if(curmax%j==0)

        {
        flag=true;
        break;
        }
    }

    if(!flag)
    break;
    }
    }


}


printf("%lld",curmax);


return 0;

}
