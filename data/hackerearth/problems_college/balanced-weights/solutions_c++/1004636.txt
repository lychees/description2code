#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<algorithm>
using namespace std;
	
long long gcd(long long  a, long long b)
	{
		while(b){
			a %= b;
			b ^= a;
			a ^= b;
			b ^= a;
		}
	return a;
	}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
              long long n,g;
              cin>>n;
              long long* arr=(long long*)malloc (sizeof(long long)*n);
              for(long long i=0;i<n;i++)  scanf("%lld",&arr[i]);
              sort(arr,arr+n);
              g=arr[0];
               for(long long i=1;i<n;i++)
              g= gcd(g,arr[i]);
              cout<<g*n<<endl;
    }
    return 0;
}
