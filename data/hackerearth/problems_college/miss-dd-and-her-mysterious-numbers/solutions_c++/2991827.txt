#include<iostream>
using namespace std;
int main()
{
    int n,t,b,c,i,j;
    cin>>n;
    int a[n+1];
    for(i=1;i<=n;i++)
        a[i]=0;
    for(i=2;i<=n/2;i++)
    {
        if(a[i]==0)
        {
            a[i]=1;
            for(j=2*i;j<=n;j=j+i)
            {
                a[j]++;
            }
        }
    }
    b=0;
    for(i=2;i<=n;i++)
        {
            if(a[i]>b)
                b=a[i];
            if(a[i]==0)
                a[i]=1;
        }
        int as[b+1];
        for(i=0;i<=b;i++)
        {
            as[i]=0;
            for(j=1;j<=n;j++)
            {
                if(a[j]==i)
                    as[i]++;
            }
        }
        cin>>t;
        while(t--)
        {
            cin>>c;
            if(c>b)
                cout<<0<<endl;
            else
            {
                cout<<as[c]<<endl;
            }
        }
        return 0;
}
