#include <iostream>
using namespace std;

int main()
{
	int total;
	int index,moves;
	cin>>total;
	for(int i=0;i<total;i++)
	   {
	   	  cin>> index;
	   	  cin>>moves;
	   	  if(index==0)
	   	  {
	   	  	cout<<moves%2<<endl;	   	  	
	   	  }
	   	 else if(index==2)
	   	 {
	   	 	if(moves==0)
	   	 	{
	   	 		cout<<2<<endl;	   	 		
	   	 	}
	   	 	else
	   	 	cout<<moves%2<<endl;
	   	 }
	   	  else if(index==1)
	   	   {
	   	   	cout<<(moves%2==0?1:0)<<endl;
	   	   }
	   	  
	   	   
	   }
    return 0;
}
