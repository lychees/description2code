#include <iostream>
#include <string>
using namespace std;

int main()
{
    string s;
    while(getline(cin,s)){
    	for(unsigned int i = 0; i < s.size(); i++){
      		if(i==0 && s[i] >= '!' && s[i] <= '/' ) {
    			s[i] = s[i] + 32;
    		}
    		else if ( i==0 && s[i] >= '1' && s[i] <= '@'){
    			s[i] = s[i] + 32;
    		}
    		else if(i==0 && s[i] >= 'a' && s[i] <= 'z'){
    			s[i] = s[i] - 32;
    		}
    		else if(s[i]== ' ' && s[i+1] >= '!' && s[i+1] <= '/' ){
    			s[i+1] = s[i+1] + 32;
    		}
    		else if( s[i] == ' ' && s[i + 1] >= '1' && s[i+1] <= '@'){
    			s[i+1] = s[i+1] +32 ;
    		}
    		else if(s[i]== ' ' && s[i+1] >= 'a' && s[i+1] <= 'z'){
    			s[i+1] = s[i+1]-32;
    		}
    	}
    }
    cout<<s<<"\n";
    return 0;
}
