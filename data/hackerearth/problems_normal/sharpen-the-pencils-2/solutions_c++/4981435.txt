#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   long long int t,n,i,j,k;
   long double m,r;
   cin>>t;
   while(t--){
   	cin>>n;
   	long double arr[n]; 
   	for(i=0;i<n;i++)
   		cin>>arr[i];
   		m=r=k=0;
   	for(i=0,j=n-1;i<=j;){
   		if(m<=r){
   			m+=arr[i]/2;
   			i++;
   			k++;
   		}
   		if(r<m && i<=j)
   			r+=arr[j--];
	}
   			
   	cout<<k<<" "<<n-k<<endl;
   }
    return 0;
}
