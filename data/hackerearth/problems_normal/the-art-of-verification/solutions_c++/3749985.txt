#include <iostream>
#include<string>
using namespace std;

int main()
{
    string url,pwd="",usr="",prof="",role="",key="";
    cin>>url;
    url[url.length()]='\0';
    int point = url.find("username");
	for(int i=point+9;url.substr(i,4)!="&pwd";i++)
		usr=usr+url[i];
	cout<<"username: "<<usr<<endl;		
    point = url.find("pwd");
	
	for(int i=point+4;url.substr(i,8)!="&profile";i++)
		pwd=pwd+url[i];
	
	cout<<"pwd: "<<pwd<<endl;		
    point = url.find("profile");
	for(int i=point+8;url.substr(i,5)!="&role";i++)
		prof=prof+url[i];
	cout<<"profile: "<<prof<<endl;		
    point = url.find("role");
	for(int i=point+5;url.substr(i,4)!="&key";i++)
		role=role+url[i];
	cout<<"role: "<<role<<endl;	
    point = url.find("key");
	for(int i=point+4;url[i]!='\0';i++)
		key=key+url[i];
	cout<<"key: "<<key<<endl;
    return 0;
}
