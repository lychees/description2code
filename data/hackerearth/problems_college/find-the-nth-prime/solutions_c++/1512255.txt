#include <iostream>
#include <cmath>
using namespace std;

int prime(int i)
{
  if(i<2 || (i%2==0 and i!=2))
   return 0;
  else
  {
   for(int j=3;j<=sqrt(i);j++)
   {
    if(i%j==0)
     return 0;
   }
   return i;
  }
}

int main(){
	ios::sync_with_stdio(false);
	int T=0;
	int count=0;
	int ans=0;
	
	cin >> T;
	
	for(int i=2;count<T;i++)
	 {
	  if(prime(i))
	  {
	   count++;
	   ans=prime(i);
	  }
	 }
	 
	cout << ans << endl;
	
	return 0;
}
