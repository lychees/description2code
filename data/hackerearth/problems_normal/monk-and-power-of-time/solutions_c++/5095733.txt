#include<iostream>
#include<queue>

using namespace std;

int main()
{
        queue<int> q;
        int n;
        cin >> n;
        int i=0,t,a[n],b[n],count=0;

        for(i=0;i<n;i++){
                cin >> b[i];
        }
        for(i=0;i<n;i++){
                q.push(b[i]);
        }
        for(i=0;i<n;i++){
                cin >> a[i];
        }
        i=0;
        while(!q.empty()){
                if(q.front()==a[i]){
                        q.pop();
                        i++;
                        count++;
                }
                else{
                        t=q.front();
                        q.pop();
                        q.push(t);
                        count++;
                }
        }

        cout <<count;
}
