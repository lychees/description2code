T = input()
for _ in xrange(T):
	N = input()
	pencils = map(int, raw_input().strip().split())
	mPCount = 0
	rPCount = 0
	mPencil = 0
	rPencil = 0
	mI = 0
	rI = N - 1
	while rI >= mI:
		if mPencil <= rPencil:
			mPencil += pencils[mI]
			mPCount += 1
			mI += 1
		else:
			rPencil += pencils[rI] * 2
			rPCount += 1
			rI -= 1
			
	print mPCount, rPCount
