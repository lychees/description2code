#include <bits/stdc++.h>
using namespace std;
#define rep(i,a,b) for(int i=a;i<=b;i++)
pair<int,int> a[100002];
int x,y,d,n;
long long ans;
int main()
{
    cin>>n;
    rep(i,1,n)
    {
    	cin>>x>>y>>d;
    	a[i] = {x,d};
    }
    sort(a+1,a+n+1);
    int curx = a[1].first,curd = a[1].second;
    ans = curd+1;
    rep(i,2,n)
    {
    	x = a[i].first,y=a[i].second;
    	//totally out 
    	if(x>curx+curd)
    	{
    		ans += y+1;
    		curx = x;curd = y;
    	}
    	// partial overlap
    	else if(x>=curx && x<= curx+curd && x+y>curx+curd)
    	{
    		ans += (x+y-curx-curd);
    		curx = x;curd = y;
    	}
    	// total overlap ki chinta nhi    	
    }
    cout<<ans;
    return 0;
}
