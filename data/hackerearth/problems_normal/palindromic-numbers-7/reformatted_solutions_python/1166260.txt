test_cases = int(raw_input())
for i in xrange(test_cases):
    count = 0
    numbers = raw_input()
    A, B = numbers.split()
    A = int(A)
    B = int(B)
    for num in xrange(A, (B+1)):
        if str(num) == str(num)[::-1]:
            count += 1
    print count
