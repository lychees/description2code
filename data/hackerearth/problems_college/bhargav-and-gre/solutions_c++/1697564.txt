#include <iostream>
using namespace std;

int main()
{
    string W, temp;
    int n,S=0,A=0;
    cin>>W;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	cin>>temp;
    	if(temp.size()==W.size())
    	{
    		if(W[0]==temp[0])
    			S++;
    		else if((W[0]+3)==temp[0])
    			A++;
    	}
    
    }
    	cout<<S<<" "<<A<<endl;
    return 0;
}
