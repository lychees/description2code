for t in range(input()):
 x1,y1,x2,y2,x3,y3=map(float,raw_input().split())
 if (x2-x1)!=0 and (x3-x2)!=0 and (x3-x1)!=0:
  m1=(y2-y1)/float(x2-x1)
  m2=(y3-y2)/float(x3-x2)
  m3=(y3-y1)/float(x3-x1)
  if m1==m2==m3:
   print "YES"
  else:
   print "NO"
 else:
   if x1==x2==x3:
     print "YES"
   else:
     print "NO"
