#include <iostream>
#include<vector>
using namespace std;


int traverse(vector<vector<bool>>& vec, vector<vector<bool>>& col, int i, int j)
{
	if(i< 0 || i>= vec.size() || j <0 || j>=vec.size()|| !vec[i][j] || col[i][j])
		return 0;
		
	col[i][j] = true;
	int a1 = traverse(vec, col, i+1, j);
	int a2 = traverse(vec, col, i, j+1);
	int a3 = traverse(vec, col, i-1, j);
	int a4 = traverse(vec, col, i, j-1);

	return a1+a2+a3+a4+1;
}

int main()
{
    int N=0;
    cin>>N;
	vector<vector<bool>> vec(N, vector<bool>(N, false));
	for(int i =0;i<N;i++)
	{
		for(int j =0;j<N;j++)
		{
			int x =0;
			cin>>x;
			vec[i][j] = x;
			
		}
	}
	
	vector<vector<bool>> col(N, vector<bool>(N, false));
	int res =0;
	for(int i =0;i<N;i++)
	{
		for(int j =0;j<N;j++)
		{
			if(col[i][j] == false && vec[i][j]== true)
			{
				res = max(res, traverse(vec, col, i, j));
			}
		}
		
	}
	
	cout<<res;

    return 0;
}
