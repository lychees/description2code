#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

const int lmt = 10005;
lli dp[lmt][1005];

int main(){
    string p,t;
    cin>>p>>t;

    int n = t.size();
    int m = p.size();

    for(int i=0;i<1005;i++)
        dp[i][0] = 1;

    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            dp[i][j]=dp[i-1][j];
            if(t[i-1]==p[j-1])
                dp[i][j] += dp[i-1][j-1];
        }
    }
    cout<<dp[n][m]<<"\n";
    return 0;
}
