#include<iostream>
using namespace std;
int main()
{
    long int t,n,d,r;
    int rem,q;
    cin>>t;
    while(t!=0)
    {
        cin>>n>>d>>r;
        if(n>d)continue;
        rem=n*10;
        while(r!=0)
        {
            q=rem/d;       
            rem=rem%d;
            r--;
            rem*=10;
        }
        cout<<q<<endl;
        t--;
    }
}
