#include<bits/stdc++.h>
#define SIZE 100007
using namespace std ;
int main()
{
	int t ;
	scanf("%d",&t);
	while(t--)
	{
		long a[SIZE],n,i ;
		scanf("%ld",&n) ;
		a[0] = 1 ;
		a[1] = 2 ;
		for(i=2;i<=n;i++)
		{
			a[i] = a[i-1] + a[i-2] ;
		}
		a[n] = a[n]*a[n] ;
		cout<<a[n]<<endl ;
	}
}
