pad = [".,?!1", "abc2", "def3", "ghi4", "jkl5", "mno6", "pqrs7", "tuv8", "wxyz9", "_0"]

for i in xrange(int(raw_input())):
	string = raw_input().strip()

	time = 0
	index = 0
	for c in string:
		if c in pad[index]:
			time += pad[index].find(c) + 1
		else:
			for i in xrange(len(pad)):
				if c in pad[i]:
					index = i
					break

			time += 1 + pad[index].find(c) + 1

	print time
