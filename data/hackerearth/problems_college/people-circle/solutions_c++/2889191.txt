#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int m,f,k,i,j=0;
        char arr[100001]={'\0'};
        bool visit[100001]={0};
        cin>>m>>f>>k;
        for(i=0;i<f;i++)
        {
            int c=0;
            while(c!=k-1)
            {
                if(visit[j]==0)
                {
                    c++;
                }
                j++;
                if(j==m+f)
                    j=0;
            }
            while(visit[j]==1)
            {
                j++;
                if(j==m+f)
                    j=0;
            }
            visit[j]=1;
            arr[j]='F';
        }
        for(i=0;i<m+f;i++)
     {
         if(arr[i]!='F')
            arr[i]='M';
     }
     arr[i]='\0';
     for(i=0;i<m+f;i++)
        cout<<arr[i];
     cout<<endl;

    }
return 0;
}
