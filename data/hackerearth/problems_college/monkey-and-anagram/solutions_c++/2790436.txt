#include <iostream>
#include <vector>
#include <string>
#define ll long long
using namespace std;
int main()
{
	int t;
	cin>>t;
	string s,k;
	while(t--)
	{
		int i=0;
		int count=0;
		cin>>s>>k;
		int l=s.length();
		int m=k.length();
		int a[26];
		int b[26];
		for(int i=0;i<26;i++)
		{
			a[i]=0;
			b[i]=0;
		}
		for (int i = 0; i < m; ++i)
		{
			/* code */
			if(i<l)
				{a[(s[i]-'a')]++;}
			b[(k[i]-'a')]++;
		}
		for (; i < 26; ++i)
		{
			/* code */
			if(a[i]>b[i])
			{
				break;
			}
			else if(a[i]<b[i])
					count++;
		}
		if(i!=26)
			cout<<-1<<endl;
		else
			cout<<count<<endl;
	}
	return 0;
}
