#include <iostream>
#include <cstring>
#include <cmath>

using namespace std;

bool memo[1000010];
int N, M1, M2;
int MAXI  = 1000010;
void preCalc() {
	memo[0] = false;
	memset(memo, false, sizeof(memo));
	for (int i = 0; i <=N; i++) {
		if (i+M1 <= N) memo[i+M1] |= !memo[i];
		if (i+M2 <= N) memo[i+M2] |= !memo[i];
	}
	
	
}

int main()
{
	int T;
	for (cin >> T; T; T--) {
		cin >> N >> M1 >> M2;
		preCalc();
		cout << (memo[N] ? 1 : 0) <<"\n";
	}
}
