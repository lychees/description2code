input = raw_input
vowels = list("aeiouy")
for _ in range(int(input())):
    s = input()
    nvowels = 0
    for x in vowels:
        nvowels += s.count(x)

    cons = False
    for x in range(2, len(s)):
        if not (s[x] in vowels or s[x - 1] in vowels or s[x - 2] in vowels):
            cons = True
            break
    if cons or nvowels < (len(s) - nvowels):
        print("hard")
    else:
        print("easy")
