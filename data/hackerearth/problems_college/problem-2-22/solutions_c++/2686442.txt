#include<cstdio>
#include<iostream>
#include<cmath>
#include<cstring>
using namespace std;
typedef int w;
typedef long long l;


void solve(){
    w r;
    w x,y;
    cin>>r>>x>>y;
    
    r=r*r;
    w res=0;
    for(int i=1;i<=4;++i){
        w tx,ty; cin>>tx>>ty;
        w hihi=abs(tx-x)*abs(tx-x)+abs(ty-y)*abs(ty-y);
        res=max(res,hihi);
    }
    if(res<r) cout<<"Yes"<<endl;
    else cout<<"No"<<endl;
}

int main(){
    ios_base::sync_with_stdio(false);
   //freopen("in","r",stdin);
    w T;
    cin>>T;
    while(T--) solve();
}
