int BIT[27][MX];
void add(int *BIT,int p,int v) {
    for(int i=p; i<=MX; i+=i&-i)BIT[i]+=v;
}

int read(int *BIT,int p) {
    int ret=0;
    for(int i=p; i>0; i-=i&-i)ret+=BIT[i];
    return ret;
}

char s[MX+2];
int main() {
    int tc,m,n,q,i,j,k,x,y,op;
    //S(tc);
    tc=1;
    while(tc--) {
        S2(n,q);
        getchar();
        gets(s);
        for(i=0; i<n; i++) {
            k=s[i]-'a';
            add(BIT[k],i+1,+1);
        }
        while(q--) {
            S(op);
            if(op==0) {
                char ch;
                scanf("%d %c",&x,&ch);
                k=s[x-1]-'a';
                add( BIT[k],x,-1 );
                s[x-1]=ch;
                k=s[x-1]-'a';
                add( BIT[k],x,+1 );
            } else {
                S2(x,y);
                S(k);
                int f=0,sm=0;
                for(i=0; i<=25; i++) {
                    sm+=( read(BIT[i],y)-read(BIT[i],x-1) );
                    if( sm>=k ) {
                        f=1;
                        break;
                    }
                }
                if(f==0) printf("Out of range\n");
                else printf("%c\n",(char)(i+'a') );
            }
        }
    }
    return 0;
}
