#include <cstdio>
#include <cstring>
#include <cmath>
#define MAX 1007
typedef long long ll;
using namespace std;
 
int is_prime[MAX], primes[MAX*MAX], rem[MAX*MAX];
ll L, R;
void sieve()
{
	for(int i = 0; i<MAX; i++)
		is_prime[i]= 0;
	is_prime[0] = is_prime[1] = 1;
	for(int i = 2; i<MAX; i++)
	{
		if(is_prime[i]==0)
		{
			for(int j=i*2; j<MAX; j+=i)
				is_prime[j] = i;
		}
	}
}
 
void prime_factors()
{
	for(int i=0; i<MAX*MAX; i++)
		primes[i] = 0, rem[i] = i;
		
	for(int i=2; i<MAX; i++)
	{
		if(is_prime[i]==0)
		{for(int j = i; j<MAX*MAX; j+=i)
			{
				while(rem[j]%i==0)rem[j]/=i,primes[j]++;
			}
		}
	}
	for(int j = 2; j<MAX*MAX; j++)
	{
		if(rem[j]>sqrt(j))
			primes[j]++;
		primes[j] += primes[j-1];
	}
}
 
int main()
{
    sieve();
    prime_factors();
    int t, n;
    scanf("%d", &t);
    while(t--)
   {
    	scanf("%d", &n);
    	printf("%d\n", primes[n]);
    }
    return 0;
}
