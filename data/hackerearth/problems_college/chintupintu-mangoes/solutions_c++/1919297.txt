#include <iostream>
using namespace std;

int t;
int x, h, d;

int main()
{
    scanf("%d", &t);
    while (t--) {
    	scanf("%d %d %d", &x, &h, &d);
    	bool res;
    	if (x % 2 == 0) res = false;
    	else {
	    	int sm = h;
    		for (int i = 2; i * i <= h; i++)
    			if (h % i == 0) { sm = i; break; }
    		res = h / sm >= d;
    	}
    	printf("%s\n", res? "Chintu": "Pintu");
    }
    return 0;
}
