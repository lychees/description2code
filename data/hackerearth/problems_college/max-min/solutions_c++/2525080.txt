#include<iostream>
#include<stdlib.h>
using namespace std;
int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}
int main()
{
 long *A,N,K,i,min;
 cin>>N>>K;
  A=(long *)malloc(sizeof(long)*N);
 for(i=0;i<N;i++)
    cin>>A[i];
qsort(A,N,sizeof(long),compare);
  min=A[N-1];
  for(i=0;i<=N-K;i++)
     if(A[i+K-1]-A[i]<min)
        min=A[i+K-1]-A[i];
  cout<<min<<endl;
return 0;
}
