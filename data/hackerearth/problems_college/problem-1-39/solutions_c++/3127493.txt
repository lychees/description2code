#include <iostream>
#include<math.h>
using namespace std;

int main()
{
long long int t,a,b,c;
std::cin>>t;
while(t--)
{ std::cin>>a>>b>>c;
   
    long long x=pow(a,b);
    int q=x%c;
    if(x%c==0)
       std::cout<<x<<endl;
   
  else if(((x-q+c)-x)<(x-(x-q)))
     {  

	    std::cout<<(x-q+c)<<endl;
       }
   else 
    std::cout<<(x-q)<<endl;
     
}

    return 0;
}
