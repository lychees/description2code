#include <bits/stdc++.h>
using namespace std;

int main()
{
    int test,n;
    cin >> test;
    long long int x;
    vector<long long int > vec;
    while(test--)
    {
    	long long int ans=0;
    	cin >> n;
    	for(int i=0;i<n;i++)
    	{
    		cin >> x;
    		vec.push_back(x);
    	}
    	sort(vec.begin(),vec.end());
    	for(int i=vec.size()-1;i>=0;i=i-2)
    	{
    		if(i!=0)
    			ans += vec[i] *vec[i-1] ;
    		else
    			ans += vec[i];
    	}
    	cout << ans<<endl;
    	vec.clear();
    }
    return 0;
}
