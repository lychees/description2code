import math
test_cases = int(raw_input())
for i in range(test_cases):
    number = int(raw_input())
    if number == 3:
        print 3
    elif number == 2:
        print 2
    elif number == 1:
        print 1
    else:
        j = 2
        b = True
        while j <= (int(math.sqrt(number))):
            if number % j == 0:
                print 1
                b = False
                break
            j += 1
        if b:
            print 2
