#include <cstdio>
#include <iostream>
#include <vector>
#define DEBUG if(0)
using namespace std;

int main()
{
    int S,B,tt,tmp;
    cin >> S >> B;
    vector<bool> flag;
    flag.resize(B+1);
    int unique = 0;
    for(int i = 0; i < S; ++i) {
       cin >> tt;
       for(int j = 0; j < tt; ++j)
       {
	 		cin >> tmp;
	  		if(!flag[tmp])  ++unique;
	  		flag[tmp] = true;
       }
    }
    
    DEBUG
    for(int i = 0; i < flag.size(); ++i)
      cout << flag[i] << endl;
    
    ( unique == B ) ? printf("YES\n") : printf("NO\n");
    return 0;
}
