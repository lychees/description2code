#include <iostream>
#include <algorithm>
#include <math.h>
using namespace std;
 
bool checkRAT(int *points){
	int side[3];
	side[0] = pow((points[0]-points[2]),2) + pow((points[1]-points[3]),2);
	side[1] = pow((points[2]-points[4]),2) + pow((points[3]-points[5]),2);
	side[2] = pow((points[4]-points[0]),2) + pow((points[5]-points[1]),2);
	
	sort(side, side + 3);
	
        if (side[2] == side[0] + side[1] && 0.5*side[0]*side[1]!=0) return true;
	else return false;
}
 
 
 
int main()
{
    int T,i;
    bool flag;
    cin>>T;	
    int points[6];
    while(T--){
 
	for(i=0;i<6;i++)
		cin>>points[i];
	
	if(checkRAT(points)){
		cout<<"YES\n";
		continue;
	}
	
	flag = false;
	for(i=0;i<6;i++){
 
		points[i] += 1;
		if(checkRAT(points)){
			flag = true;
			break;
		}
		else{
			points[i] -= 2;
			if(checkRAT(points)){
				flag = true;
				break;
			}
			else points[i] += 1;
		}
	}
	
	if(flag) cout<<"YES\n";
	else cout<<"NO\n";    
   }	
}
