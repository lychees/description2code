#include <iostream>
using namespace std;

int main()
{
	int t,d=0;
    cin>>t;
    while(t--)
    {
    	d=0;
    	int a;
    	cin>>a;
    	while(a!=0)
    	{
    		d=d+a%10;
    		a=a/10;
    		if(a==0)
    		{
       		 if(d>9){
       		 a=d;
       		 d=0;}
       		 else
       		 break;
    		}
         }
        cout<<d<<endl;
    }
}
