#include<iostream>
#include<cstring>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
	long int n,count=1,i;
	cin>>n;
	vector<string> v;
	string  str;
	for(i=0;i<n;i++)
	{
		cin>>str;
		v.push_back(str);
	}
	sort(v.begin(),v.end());
	for(i=0;i<v.size()-1;i++)
	{
		if(v[i]==v[i+1])
		count++;
		else
		{
			cout<<v[i]<<" "<<count<<endl;
			count=1;
		}
	}
	cout<<v[i]<<" "<<count<<endl;
}
