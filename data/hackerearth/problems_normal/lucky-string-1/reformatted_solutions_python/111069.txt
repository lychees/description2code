import sys


def lucky(n):
    i, sumc, sump = 0, 0, 0
    while (sumc < n):
        sump = sumc
        i = i + 1
        sumc = sumc + int(pow(2, i))*i
    lft = n-sump-1
    d = lft / i
    r = lft % i
    l = "{0:{fill}{align}{width}{base}}".format(
        d, fill='0', width=i, align='>', base='b')
    if l[r] == '0':
        return "Hacker"
    else:
        return "Earth"


nt = int(sys.stdin.readline())
for i in xrange(nt):
    n = int(sys.stdin.readline())
    print lucky(n)
