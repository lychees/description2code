#include <iostream>
using namespace std;

long long fib[15001];
long long sumfib[15001];
long long mod = 1000000007;

void pre(){
	fib[0] = 0;
	fib[1] = 0;
	fib[2] = 1;
	sumfib[0] = 0;
	sumfib[1] = 0;
	sumfib[2] = 1;
	for(int i = 3; i<15001; i++){
		fib[i] = (fib[i-1] + fib[i-2])%10000;
		sumfib[i] = (sumfib[i-1]+fib[i])%mod;
	}
}

long long sum_last_4_digits(long long n){
	return (((n/15000)%mod)*(sumfib[15000])+sumfib[n%15000])%mod;
}

void solve(){
	long long l,r;
	cin>>l>>r;
	cout<<(sum_last_4_digits(r) - sum_last_4_digits(l-1) + mod)%mod<<endl;
}

int main() {
	pre();
	int t;
	cin>>t;
	while(t--) solve();
	return 0;
}
