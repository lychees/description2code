#define frn(i,a,b) for(int i=a;i>b;i--)

#define sll(x)  scanf("%lld",&x)
#define pln printf("\n")
#define pi(x) printf("%d",x)
#define pll(x) printf("%lld",x)

typedef long long ll;
ll MOD=1000000007;
ll pw(ll a,ll b){
  ll r=1;
  while(b>0){
     if(b&1){
       r=(r*a)%MOD;
        }
     b=b>>1;
     a=(a*a)%MOD;
  }
return r;
}

int valid(ll x){
   return (x<=1e12&&x>=0);
}

int main(){
 int t;
 si(t);
 while(t--){
    ll dp[22][22];
    char s[22];
    scanf("%s",s);
    int len=strlen(s);

    fr(i,0,len){
       fr(j,0,len){
         dp[i][j]=-1;
       }
    }
//all segments
    fr(ss,0,len){
       dp[ss][ss]=s[ss]-'0';
       if(s[ss]!='0'){
          fr(se,ss+1,len){
            dp[ss][se]=(dp[ss][se-1]*10)+(s[se]-'0');
          }
       }
    }
  ll ans=-1;
    fr(a,1,len){
       fr(b,a+1,len){
          fr(c,b+1,len){
            if(valid(dp[0][a-1]) && valid(dp[a][b-1]) &&valid(dp[b][c-1]) &&valid(dp[c][len-1])){
                ans=max(ans,dp[0][a-1]+dp[a][b-1]+dp[b][c-1]+dp[c][len-1]);
            }
          }
      }
    }

   if(ans<0){
      cout<<"unlucky\n";
   }
   else{
       pll(ans);pln;
   }

 }


return 0;
}
