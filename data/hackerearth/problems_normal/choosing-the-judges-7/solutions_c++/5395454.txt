#include<iostream>
#include<cstring>
#define ll long long
using namespace std;
ll ans[10001];ll X[10001];ll n;
ll DP(int index){
	if(index>=n){
		return 0;
	}
	if(ans[index]>=0){
		return ans[index];
	}
	ll q=0;
	q=max(X[index]+DP(index+2),DP(index+1));
	ans[index]=q;
	return q;
}
int main(){
	ll t;
	cin>>t;ll e;
	for(int y=1;y<=t;y++){
	 	cin>>n;
	 	memset(ans,-1,sizeof ans);
	 	for(int i=0;i<n;i++){
	 		cin>>e;
	 		X[i]=e;
		 }
		 cout<<"Case "<<y<<": "<<DP(0)<<endl;
	}
}
