int main(){

	int T;
	cin >> T;
	
		while(T--) {
			int n, k, cnt = 0;
			cin >> n >> k;
			
			int arr[n];
			
			for(int i = 0; i < n; i++) {
			cin >> arr[i];
			if(arr[i] <= 0){cnt++;}
			}
			
				if(cnt >= k) cout << "NO" << endl;

		}
return 0;
