int fact(int k)
{	return (k*(k-1)*(k-2))/6;	}

int gcd(int a,int b)
{	if(b==0)
	{	return a;
	}
	else
	{	return gcd(b,a%b);
	}
}

{	for(int i = 1; i <= 100 ; i++)

		}
	}
}
