#include <unordered_set>
#include <iostream>
using namespace std;

int main()
{
	int t,n;
	scanf("%d",&t);
	long long int a[100000];
	while(t--){
		scanf("%d",&n);
		for(int i=0;i<n;i++)
			scanf("%lld",&a[i]);
		
		unordered_set<long long int> s(a, a + n);
	    cout << s.size() <<endl;
	}
    	return 0;
}
