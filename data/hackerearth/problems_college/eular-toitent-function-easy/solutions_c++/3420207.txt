#include <stdio.h>
#include<algorithm>

int main()
{
    int t,i,n,count=0,k;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	for(i=1;i<n;i++)
    	{
    		k=std::__gcd(n,i);
    		if(k==1)
    		   count++;
    		
    	}
    	printf("%d\n",count);
    	count=0;
    }
    
    return 0;
}
