#include <iostream>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	string s,p;
    	cin>>s>>p;
    	int i = 0, j = 0;
    	bool flag = false;
    	while(i<s.length() and j<p.length()){
    		if(s[i]==p[j]){
    			i++;
    			j++;
    		}
    		else{
    			i++;
    		}
    		if(j==p.length()){
    			flag=true;
    			break;
    		}
    	}
    	if(flag){
    		cout<<"YES"<<endl;
    	}
    	else{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
