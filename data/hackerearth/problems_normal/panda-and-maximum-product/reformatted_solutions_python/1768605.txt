n = input()

a = map(int, raw_input().split(" "))

max1 = -(10**9)-1
max2 = -(10**9)-1

min1 = 10**9 + 1
min2 = 10**9 + 1
for i in a:
    if i < max1:
        if i > max2:
            max2 = i
    else:
        max2 = max1
        max1 = i

    if i > min1:
        if i < min2:
            min2 = i
    else:
        min2 = min1
        min1 = i

if min1*min2 > max1*max2:
    print min1*min2
else:
    print max1*max2
