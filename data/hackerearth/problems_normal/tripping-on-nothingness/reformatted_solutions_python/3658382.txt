def gcd(a, b):
    if b == 0:
        return a
    else:
        return gcd(b, a % b)


s, q = raw_input().split()
s = int(s)
q = int(q)
lis = [0 for i in range(s+1)]
for i in range(q):
    j = int(raw_input())
    ans = gcd(s, j)
    if lis[ans] == 0:
        print ans
        lis[ans] = 1
    else:
        print -1
