#include <bits/stdc++.h>
using namespace std;
const int MAXN = 200005;
const long long int MOD = 1000000007;
int A[MAXN];
long long int two_pow[MAXN];
int main()
{
	// freopen("input.txt", "r", stdin);
	// precompute powers of two
	two_pow[0] = 1;
	for (int i = 1; i < MAXN; ++i)
	{
		two_pow[i] = (two_pow[i-1]*2)%MOD;
	}
	//
	int n;
	cin>>n;
	for (int i = 1; i <= n; ++i)
	{
		cin>>A[i];
	}
	if(n == 1)
	{
		cout<<A[1]<<endl;
		return 0;
	}
	long long int ans = 0;
	for (int i = 1; i <= n; ++i)
	{
		ans = (ans + ((long long int)A[i])*(i+1))%MOD;
	}
	ans = (ans*two_pow[n-2])%MOD;
	cout<<ans<<endl;
	return 0;
}
