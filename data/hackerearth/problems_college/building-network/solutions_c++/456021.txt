#include<bits/stdc++.h>
using namespace std;

int main()
{

	long long int t,n,sum,i,j,a[1000000],mini=INT_MAX,pos;
	cin>>t;
	while(t--)
	{
	mini=INT_MAX;
	sum=0;
		cin>>n;
		for(i=0;i<n;i++)
			{
			cin>>a[i];
			if(a[i]<mini)
			{
				pos=i;
mini=a[i];
			}
			}
		for(i=0;i<n;i++)
		{
			
			if(i!=pos)
					sum=sum+a[i]+a[pos];
				}
			
		
		cout<<sum<<endl;
	}
	return 0;
}
