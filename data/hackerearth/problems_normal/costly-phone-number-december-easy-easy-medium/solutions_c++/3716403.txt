#include <bits/stdc++.h>
using namespace std;
long int cost[10];
int main()
{
	int t;
	cin>>t;
	while(t--){
    for(int i=0;i<10;i++){
    	cin>>cost[i];
    }
    bool ok=true;
    while(ok){
    	ok=false;
    for(int i=0;i<10;i++){
    	for(int j=0;j<10;j++){
    		if(cost[(i+j)%10]>cost[i]+cost[j]){
    			cost[(i+j)%10]=cost[i]+cost[j];
    			ok=true;
    		}
    	}
    }
    }
    int n;
    cin>>n;
    string s;
    cin>>s;
    long int ans=0;
    for(int i=0;i<n;i++){
    	ans+=cost[s[i]-'0'];
    }
    cout<<ans<<endl;}
    return 0;
}
