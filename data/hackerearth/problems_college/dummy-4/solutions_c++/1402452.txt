#include <iostream>
#include <string>

using namespace std;

int main()
{
	int n;
	unsigned long long int sum=0,tempSum=0,m;
	cin>>n;
	while(n--)
	{
		string str;
		cin>>str;
		for(int i=0;i<str.size();i++)
		{
			sum+=str[i]-'0';
		}
		while(sum>=10)
		{
			m=sum;
			tempSum=0;
			while(m!=0)
			{
				tempSum+=m%10;
				m/=10;
			}
			sum=tempSum;
		}
	}
	cout<<sum<<endl;
	return 0;
}
