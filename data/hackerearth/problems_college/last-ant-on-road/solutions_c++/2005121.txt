#include<iostream>
#include<cmath>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,i,j,count;
        cin>>n;
        int a[n];
        for(i=0;i<n;i++)
        {
            cin>>j;
            a[i]=(i+1)*j;
        }
        count=n;
        i=0;j=n;
        int pos;
        while(count!=0)
        {
            int flag=0;
            //cout<<"count="<<count<<endl;
            for(i=0;i<j;i++)
            {
                if(a[i]>0&&i==n-1)
                {
                    count--;
                 //    cout<<a[i]<<" ";
                     if(flag==0)
                     {
                         pos=a[i];
                         flag=1;
                     }
                     a[i]=0;
                     continue;
                }
                if(a[i]<0&&i==0)
                {
                    count--;
                 //cout<<a[i]<<" ";
                     if(flag==0)
                     {
                         pos=a[i];
                         flag=1;
                     }
                     a[i]=0;
                     continue;
                }
                if(a[i]>0&&a[i+1]==0)
                {
                     int x=a[i];
                    a[i]=a[i+1];
                    a[i+1]=x;
                    i++;
                    continue;
                }
                if(a[i]<0&&a[i-1]==0)
                {
                    int x=a[i];
                    a[i]=a[i-1];
                    a[i-1]=x;
                    i++;
                    continue;
                }
                if((a[i]>0&&a[i+1]<0)||(a[i]<0&&a[i+1]>0))
                {
                    a[i]=a[i]*(-1);
                    a[i+1]=a[i+1]*(-1);
                    i++;
                    continue;
                }
            }
            //cout<<endl;
        }
        cout<<abs(pos)<<endl;
    }
}
