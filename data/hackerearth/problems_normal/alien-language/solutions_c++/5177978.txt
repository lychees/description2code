#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string a, b;
    	cin>>a>>b;
    	bool x[26]={false};
    	for(int i=0; i<a.length(); i++){
    		x[int(a[i])-int('a')]=true;
    	}
    	bool o=false;
    	for(int i=0; i<b.length(); i++){
    		if (x[int(b[i])-int('a')]){
    			o=true;
    			break;
    		}
    	}
    	if(o)cout<<"YES"<<endl;
    	else cout<<"NO" <<endl;
    }
    return 0;
}
