bubbly = 0
for _ in xrange(input()):
    word = list(raw_input())
    stack = []
    for AorB in word:
        if AorB == 'A':
            if stack != [] and stack[-1] == 'A':
                stack.pop()
            else:
                stack.append(AorB)
        if AorB == 'B':
            if stack != [] and stack[-1] == 'B':
                stack.pop()
            else:
                stack.append(AorB)
    if stack == []:
        bubbly += 1
print bubbly
