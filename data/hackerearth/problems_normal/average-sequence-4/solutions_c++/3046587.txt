#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
  int n;
  cin>>n;
  int arr[n+1],temp[n+1];
  for(int i=0;i<n;i++)
    cin>>arr[i];
  temp[0] = arr[0];
  int cc=0;
  for(int i=1;i<n;i++)
  {
    int aa = i+1;
    int bb = aa*arr[i];
    cc = cc + temp[i-1];
    temp[i] = bb-cc;
  }
  for(int i=0;i<n;i++)
    cout<<temp[i]<<" ";
  cout<<endl;
  return 0;
}
