#include <stdio.h>
#include <string.h>

int main(){
	int t,k,n,i;
	double val ;
	scanf("%d",&t);
	while(t--){
        scanf("%d%d",&k,&n);
        if(k > n){
            printf("1.000000000\n");
            continue;
        }
        else {
            val = 1.0 ;
            for(i=0;i<k;i++)
                val = val*((double(n-i))/n);
            val = 1 - val ;
            printf("%.9lf\n",val);
        }
	}
	return 0;
}
