#include <iostream>
#include<vector>
#include<stdio.h>
#include<limits.h>
#include<algorithm>
using namespace std;
int main()
{
	long long int i,j,k,l,m,n,t,count,high,low,mid,prev,count1,c;
	vector<long long int> v;
	scanf("%lld",&t);
	while(t--)
    {

        scanf("%lld%lld",&n,&c);
        for(i=1;i<=n;i++)
        {
            scanf("%lld",&k);
            v.push_back(k);
        }
        sort(v.begin(),v.end());
        low=1;
        high=INT_MAX;
        //cout << high << "\n";
        mid=low+(high-low+1)/2;

        long long int ans=0;
        while(low<high)
        {
            count1=1;
            prev=v[0];
            for(i=1;i<v.size();i++)
            {
                if(v[i]-prev>=mid)
                {
                    count1++;
                    prev=v[i];
                }
            }
            if(count1<c)
                high=mid-1;
            else
            {
                ans=max(ans,mid);
                low=mid;
            }
            mid=low+(high-low+1)/2;
        }
        printf("%lld\n",ans);
        v.clear();
    }
	return 0;
}
