#include <iostream>
#include <cmath>
using namespace std;

int main() {
	int t;
	cin >> t;
	while (t--) {
		int x2, y2, x1, y1;
		double r;
		cin >> x1 >> y1 >> x2 >> y2 >> r;
		double a = ((x2 - x1) * (x2 - x1)) + ((y2 - y1) * (y2 - y1));
		a = sqrt(a);
		double b = fabs(((x2 - x1) * y1) - ((y2 - y1) * x1));
		double ans = b / a;
		if (ans == r) {
			cout << "JUST MISSED" << endl;
		}
		else if (ans > r) {
			cout << "SAFE" << endl;
		}
		else if (ans < r) {
			cout << "REPLANNING" << endl;
		}
	}
	return 0;
}
