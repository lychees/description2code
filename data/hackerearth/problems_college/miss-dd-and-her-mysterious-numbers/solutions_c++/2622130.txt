#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
int isp[1000005],ip[10000005];
int main()
{
	fill_n(isp,1000005,1);
	fill_n(ip,10000005,0);
ll n,k,q,j,count,i;
cin>>n>>q;
for(i=2;i<=n;i++)
{
	if(isp[i])
	{
		for(j=i;j<=n;j+=i)
		{
			ip[j]++;
			isp[j]=0;
		}
	}
}
for(i=0;i<q;i++)
{
	count=0;
	cin>>k;
for(j=k;j<=n;j++)
	if(ip[j]==k)
		count++;
cout<<count<<endl;
}
}
