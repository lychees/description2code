    #include <iostream>
     
    using namespace std;
     
    int main()
    {
        int i, a, b, t;
        cin>>t;
     
        while (t--) {
            cin>>a;
            cin>>b;
            i = 0;
     
            if (a <= 7 && b >= 7) {
                i+=8;
            }
            if (a <= 13 && b >= 13) {
                i+=144;
            }
            cout<<i<<endl;
        }
     
        return 0;
    }
