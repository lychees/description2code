from math import sqrt


def ceil(n):
    if (n % 1 == 0):
        return int(n)
    else:
        return int(n)+1


t = input()
for i in range(t):
    d = input()
    s = ceil((sqrt(1+8*d)-1)/2)
    if (s % 3 == 0):
        print "SL"
    elif (s % 3 == 1):
        print "LB"
    else:
        print "BS"
