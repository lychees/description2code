#include <iostream>
#include <cstdio>

using namespace std;
const int MAX = 1e6 + 5;
char s1[MAX];

struct node
{
    int weight;
    node *ch[26];
};


node *getnode()
{
    node *p = new node;
    p->weight = 0;
    for(int i = 0;i < 26;++i)
        p->ch[i] = NULL;
    return p;
}

void insert(node *root, string s, int w)
{
    node *p = root;
    int idx;
    for(int i = 0;i < s.length();++i)
    {
        idx = s[i] - 'a';
        if(p->ch[idx] == NULL)
        {
            p->ch[idx] = getnode();
        }
        p = p->ch[idx];
        p->weight = max(p->weight, w);
    }
}

int search(node *root, string s)
{
    node *p = root;
    int m = 1e6 + 5, idx;
    for(int i = 0;i < s.length();++i)
    {
        idx = s[i] - 'a';
        if(p->ch[idx] == NULL)
            return -1;
        p = p->ch[idx];
        m = min(m, p->weight);
    }
    return m;
}


int main()
{
    int n, w, q;
    node *root = getnode();
    string s;
    scanf("%d %d", &n, &q);
    for(int i = 0;i < n;++i)
    {
        scanf("%s %d", s1, &w);
        s = s1;
        insert(root, s, w);
    }
    while(q--)
    {
        scanf("%s", s1);
        s = s1;
        printf("%d\n", search(root, s));
    }
    return 0;
}
