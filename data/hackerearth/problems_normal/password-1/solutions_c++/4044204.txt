#include<bits/stdc++.h>
using namespace std;
string make_rev(string s)
{
    string p;
    p=s;
    reverse(p.begin(),p.end());
    return p;
}
void sol()
{
    string str[205];
    int i,n;
    cin>>n;
    for(i=0;i<n;++i) cin>>str[i];
 int j=0;
    for(j=0;j<n;++j)
    {
        str[i++]=make_rev(str[j]);
    }
    sort(str,str+2*n);
    for(i=0;i<2*n;)
    {
        string t=str[i];
        int j=i+1;
        int c=1;
        while(t==str[j]) ++j,++c;
        if(c>1) {
            cout<<t.size()<<' '<<t[t.size()/2]<<endl;
            return;
        }
        ++i;
    }
}
int main()
{
    sol();
    return 0;
}
