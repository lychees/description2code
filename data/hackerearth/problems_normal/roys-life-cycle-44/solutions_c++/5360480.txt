#include <iostream>
#include<string>
using namespace std;

int main()
{
    int n,i,j,l;
    int count,max=0;
    cin>>n;
    string s[n];
    for(int i=0;i<n;i++)
    {
        count=0;
    	cin>>s[i];
    	for(int j=0;j<s[i].length();j++)
    	{
    		switch(s[i][j])
    		{
    			case 'C':
    			{
    				count++;
    				if(count>max)
    				   max=count;
    				break;
    			}
    			default :
    			{
    				
    			    count=0;
    			    break;
    			}
    		}
    	}
    }
    for(i=1;i<n;i++)
    {
    	s[0]=s[0]+s[i];
    }
    int count1=0,max1=0;
    for(i=0;i<s[0].length();i++)
    {
    	switch(s[0][i])
    	{
    		case 'C':
    		{
    			count1++;
    			if(count1>max1)
    			    max1=count1;
    			break;
    		}
    		default:
    		{
    			
    			count1=0;
    		}
    	}
    }
    cout<<max<<" "<<max1;
    return 0;
}
