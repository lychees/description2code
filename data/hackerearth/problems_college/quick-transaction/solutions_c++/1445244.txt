#include <iostream>
using namespace std;
int main()
{
   // printf("Hello World!\n");
   int t;
   cin>>t;
   while(t--)
   {
   	long long cont=0;
   	long long n;
   	cin>>n;
   	while(n>=1000)
   	{
   		n=n-1000;
   		cont++;
   	}
   	while(n>=500)
   	{
   		n=n-500;
   		cont++;
   	}
   	while(n>=100)
   	{
   		n=n-100;
   		cont++;
   	}
   	while(n>=50)
   	{
   		n=n-50;
   		cont++;
   	}
   	while(n>=20)
   	{
   		n=n-20;
   		cont++;
   	}
   	while(n>=10)
   	{
   		n=n-10;
   		cont++;
   	}
   	while(n>=5)
   	{
   		n=n-5;
   		cont++;
   	}
   	while(n>=2)
   	{
   		n=n-2;
   		cont++;
   	}
   	while(n>=1)
   	{
   		n=n-1;
   		cont++;
   	}
   	cout<<cont<<endl;
   }
    return 0;
}
