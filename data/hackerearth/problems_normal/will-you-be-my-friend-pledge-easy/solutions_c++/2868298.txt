#include<bits/stdc++.h>
using namespace std;
int gcd(int a , int b)
{
   if(b==0) return a;
   return gcd(b,a%b);
}
int main(){
	
	int NO,N,IN;
	cin>>NO;
	assert(NO >= 1 && NO <= 1000);
	cin>>N;
	assert(N >= 1 && N <= 1000);
	int count=0;
	for(int i=0;i<N;i++){
	cin>>IN;
	assert(IN >= 1 && IN <= 1000);
	if(gcd(NO,IN)!=1)
	count++;
	}
	cout<<count<<endl;
	return 0;
}
