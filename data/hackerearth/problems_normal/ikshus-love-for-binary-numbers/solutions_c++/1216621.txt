int N,K;
ll dp[100][100];
ll f(int idx , int L)
{
  ll &ret = dp[idx][L];
  if(ret == -1)
  { 
    ret = 0;
    if(idx == N)
    ret = 1;
    else
    {
      if( L+1 < K )
      {
        ret += f(idx+1 , L+1 );
        ret += f(idx+1 , 0);
      }
      else
      {
        ret += f(idx+1 , 0 );
      }
    }
  }
  return ret;
}

int main()
{
  si(N);si(K);
  memset(dp,-1,sizeof(dp));
  ll numo = f(0,0);
  ll deno = 1LL<<N;
  numo = deno - numo;
  ll g = __gcd(numo,deno);
  numo/=g;deno/=g;
  printf("%lld/%lld",numo,deno);
  nl;
}
