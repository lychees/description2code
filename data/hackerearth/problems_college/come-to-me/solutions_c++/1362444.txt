#include <bits/stdc++.h>
using namespace std;
#define int long long
int last[10005]; int dp[10005]; 
main()
{
	int a; cin >> a;
	for (int g=0; g<a; g++)
	{
		memset(last,0,sizeof(last)); 
		memset(dp,0,sizeof(dp)); 
		string b; cin >> b;
		dp[b.length()]=1e9; 
		int ans=0; 
		for (int y=b.length()-1; y>=0; y--)
		{
			int k=dp[y+1]; 
			if (last[b[y]]==0)	
			{
				k=min(k-y, (int)b.length()-y); 
				dp[y]=dp[y+1]; 
			}
			else
			{
				k=min(k-y, last[b[y]]-y); 
				dp[y]=min(dp[y+1], last[b[y]]); 
			}
			ans=max(ans, k); 
			last[b[y]]=y; 
		}
		cout << ans << '\n';
	}
	return 0; 
}
