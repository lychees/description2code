#include <iostream>
#include<algorithm>
using namespace std;
int store_male[1001];
int store_woomen[1001];
bool des(int i,int j){
	return i>j;
}

int main()
{
    int T,N,sum=0;
    cin>>T;
    while(T--){
    	cin>>N;
    	for(int i=0;i<N;i++){
    		cin>>store_male[i];
    	}
    	for(int i=0;i<N;i++){
    		cin>>store_woomen[i];
    	}	
    	sort(store_male,store_male+N,des);
    	sort(store_woomen,store_woomen+N,des);
    		for(int j=0;j<N;j++){
    			sum=sum+(store_male[j]*store_woomen[j]);
    		}
    	cout<<sum<<endl;
    	sum=0;
    	
    }
    return 0;
}
