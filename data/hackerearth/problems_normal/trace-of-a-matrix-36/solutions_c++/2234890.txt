#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T;
    long int R,C,d1,d2,m,s1,s2,s;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%ld %ld",&R,&C);
        m=min(R,C);
        d1=C+1;
        d2=R+1;
        s1=(m*((2)+(m-1)*d1))/2;
        s2=(m*((2)+(m-1)*d2))/2;
        s=s1+s2;
        cout<<s<<"\n";
    }
}
