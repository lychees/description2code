def numberofmoves(arr1,arr2,n):
        lim=len(arr1)
        i=0
        sofar=0
        while (i<lim):
                p1= int(arr1[i])
                p2=int(arr2[i])
                sofar += min(p1-1,n-p1) + min (p2-1,n-p2)
                i+=1
        print sofar
