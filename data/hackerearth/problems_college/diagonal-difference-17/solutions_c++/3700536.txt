#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
 
 
int main() {
    int n = 0, element,diff;
    long long int sumA = 0, sumB = 0;
    cin>>n;
    for(int i = 0; i < n; i++){
        for(int j = 0; j < n; j++){
            cin>>element;
            if(i == j){
                sumA += element;
            }
            if(j == n - 1 - i){
                sumB += element;
            }
        }
    }
    diff=abs(sumA - sumB);
    cout<<diff;
    return 0;
}
