#include <iostream>
using namespace std;
typedef long long ll;
int main()
{
    int t;
    ll a,b;
    cin>>t;
    while(t--){
    	cin>>a>>b;
    	ll cnt = 0;
    	while (a>1 || b>1){
    		if (a>b) {
    			cnt += a/b;
    			a %= b;
    		}
    		else{
    			cnt += b/a;
    			b %= a;
    		}
    		
    	}
    	if (cnt>0) cnt--;
    	cout << cnt << "\n";
    }
    return 0;
}
