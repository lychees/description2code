#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define lli long long int 
#define sc(x) scanf("%lld",&x)
#define pf(x) printf("%lld\n",x)
lli a[1000005],s[1000005];
int main()
{
    lli n,q,l,r,ans,i;
    sc(n);
    for(i=1;i<=n;i++)
    {
    	sc(a[i]);
    }
    s[1]=a[1];
    for(i=2;i<=n;i++)
    {
    	s[i]=s[i-1]+a[i];
    }
    sc(q);
    while(q--)
    {
    	sc(l);
    	sc(r);
    	if(l>1)
    	{
    		ans=s[r]-s[l-1];
    	}
    	else
    	{
    		ans=s[r];
    	}
    	pf(ans);
    }
    return 0;
}
