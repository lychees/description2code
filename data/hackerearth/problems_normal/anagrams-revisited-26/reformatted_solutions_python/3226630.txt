'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
n = int(raw_input())
d = dict()
for i in range(n):
    s = raw_input()
    x = ''.join(sorted(s))
    if not d.has_key(x):
        d[x] = 1
    else:
        d[x] += 1
max = -1
for j in d.values():
    if j > max:
        max = j
print max
