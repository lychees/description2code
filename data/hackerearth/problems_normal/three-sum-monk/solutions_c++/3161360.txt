#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,sum=0;
		cin>>n;
		vector<int>A(n);
		for(int i=0;i<n;i++)
			cin>>A[i];
		sort(A.begin(),A.end());
	/*	for(int i=0;i<n;i++)
		cout<<A[i]<<" ";
		cout<<endl;*/
		for(int a=2;a<=18;a++)
		{
			int Cub=(a*a*a);
			for(int i=0;i<n-2&&Cub-A[i]>=A[i];i++)
			{
				int Cub1=Cub-A[i];
				int j=i+1,k=n-1;
					while(j<k)
					{
						int temp=A[j]+A[k];
						if(temp>Cub1)
							k--;
						else if(temp<Cub1)
							j++;
						else
						{
							
							if(A[j]==A[k])
							{
								sum+=(k-j)*(k-j+1)/2;
								k=j;
							}
							else
							{
								int prev1=A[j],prev2=A[k],cnt1=0,cnt2=0;
								while(A[j]==prev1)
								{
								cnt1++;
								j+=1;
								}
								while(A[k]==prev2)
								{
								cnt2++;
								k-=1;
								}
								sum+=(cnt1*cnt2);
								
							}
							
						}
					}
			}
		}
		double total=(n*(n-1)*(n-2))/6.0;
		double ans=(sum*1.0)/total;
		printf("%0.9lf\n",ans);
	}
}
