ll powmod(ll a,ll b)
{
	if(b==0)return 1;
	ll x=powmod(a,b/2);
	ll y=(x*x)%MOD;
	if(b%2)
		return (a*y)%MOD;
	return y%MOD;
}

ll dp[N][2],act[N],req[N];
bool vis[N];
vi g[N];
void dfs(int u)
{
	vis[u]=true;
	ll zero=0,one=0;
	REP(i,0,sz(g[u]))
	{
		int v=g[u][i];
		if(!vis[v])
		{
			dfs(v);
			zero+=dp[v][0];
		}	one+=dp[v][1];
	}
	ll flag=0;
	if(act[u]==req[u])
		flag=1;
	dp[u][0]=min(zero+flag,one+1+1-flag);
	dp[u][1]=min(zero+1+flag,one+1-flag);
}

	}

		sll(act[i]);
	REP(i,1,n+1)
		sll(req[i]);
	dfs(1);
