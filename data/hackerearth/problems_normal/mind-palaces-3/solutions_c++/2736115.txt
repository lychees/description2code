#include <iostream>
#include <string.h>

using namespace std ;

int **arr ;



int main ()
{
	int n , m ; 
	scanf("%d%d",&n,&m) ;
	
	arr = (int **)malloc(sizeof(int *)*n) ;
	
	
	for(int i = 0 ; i  < n ; i++)
	{
		arr[i] = (int *)malloc(sizeof(int)*m) ;
		memset(arr[i],0,sizeof(int)*m) ;
		
		for(int j = 0 ; j < m ; j++)
		{
			scanf("%d",&arr[i][j]) ;	
		}
	}

	int q ;
	scanf("%d",&q) ;

	while(q--)
	{
		int y ;
		scanf("%d",&y) ;
		
		int i = 0 , j = m-1 ; 
		bool flag = false ;
		while(i<n && j < m)
		{
			if(arr[i][j] < y)
			{
				i++;
			}
			else if(arr[i][j] > y)
			{
				j-- ;
			}
			else
			{
				printf("%d %d\n",i,j) ;
				flag = true ;
				break; 
			}
		}
		
		if(flag == false)
			printf("-1 -1\n") ;
	}

	return 0 ; 
}
