#include<bits/stdc++.h>
using namespace std;


int main(){
    int t,n;
    string s;
    cin >> t;
    while(t--){
    	int A[26]={};
    	cin >> n >> s;
    	for(int i=0;i<s.size();i++){
    		A[s[i]-'a']++;
    	}
    	int M=0;
    	for(int i=0;i<26;i++){
    		M=max(M,A[i]);
    	}
    	
    	if(M%2) cout << "No" << endl;
    	else cout << "Yes" << endl;
    }
    return 0;
}
