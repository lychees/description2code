#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int n;
    cin>>n;
    long long int x;
    vector<int>nums;
    for(int i=0;i<n;i++)
    {
    	int num;
    	cin>>num;
    	nums.push_back(num);
    }
    x = nums[0]|nums[1];
    for(int i=2;i<n;i++)
    {
    	x = x|nums[i];
    }
    cout<<x;
    
    
    return 0;
}
