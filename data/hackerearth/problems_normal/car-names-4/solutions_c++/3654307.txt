#include <iostream>
#include <set>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t)
    {
    	string s;
    	cin>>s;
    	set<char> myset;
    	bool f = true;
    	for(int i=0;i<s.size();i++)
    	{
    		myset.insert(s[i]);
    	}
    	if(myset.size()!=3)
    	{
    		cout<<"Not OK\n";
    	}
    	else
    	{
    		char prevch=' ';
    		int hash[26]={0};
    		for(int i=0; i<s.size();i++)
    		{
    			if(prevch==' ')
    			{
    				prevch = s[i];
    				hash[s[i]-97] = 1;
    			}
    			else if(prevch == s[i])
    			{
    				hash[s[i]-97]++;
    			}
    			else
    			{
    				//character change
    				if(hash[s[i]-97]!=0) //again we encouter character that has already been processed
    				{
    					f=false;
    					break;
    				}
    				else if(hash[s[i]-97] ==0)
    				{
    					hash[s[i]-97]++;
    					prevch = s[i];
    				}
    			}
    		}
    		int x=-1;
    		for(int i=0;f&&i<26;i++)
    		{
    			if(hash[i]!=0 && x==-1)
    				x=hash[i];
    			else if(hash[i]!=0 && x!=hash[i])
    			{
    				f = false;
    				break;
    			}
    		}
    		if(f)
    			cout<<"OK\n";
    		else
    			cout<<"Not OK\n";
    	}
    	t--;
    }
    return 0;
}
