#include <iostream>
#include<algorithm>
#define ll long long int
using namespace std;
long int N,K,cord[1000000],total;
int main()
{
	total=0;
    cin>>N>>K;
    for(long int i=0;i<N;i++)
    {
    	cin>>cord[i];
    }
    sort(cord,cord+N);
    int ans=cord[0];
    for(long int i=0;i<N-1;i++)
    {
    	
    	if((cord[i+1]-cord[i])<=K)
    	{
    	ans=cord[i+1];	
    	}
    	else
    	{
    		break;
    	}
    }
    cout<<ans;
}
