#include <bits/stdc++.h>
using namespace std;

long long search(long long preSum[], long long key, long long start, long long end){
	while(end>=start){
		long long mid = (start+end)/2;
		if(preSum[mid]==key) return mid;
		if(preSum[mid]>key) end = mid-1;
		else if(preSum[mid]<key) start = mid+1;
	}
	return end;
}

int main(){
	int t;
	cin>>t;
	while(t--){
		long long n, k;
		cin>>n>>k;
		long long preSum[n+1];
		for(long long i=0; i<=n; i++){
			preSum[i]=0;	
		}
		for(long long i=1; i<=n; i++){
			long long temp;
			cin>>temp;
			preSum[i] = preSum[i-1]+temp;
		}
		long long ans[n+1];
		memset(ans, 0, sizeof(ans));
		for(long long i=1; i<=n; i++){
			long long h = search(preSum, preSum[i-1]+k, i, n);
			ans[h-i+1]++;
		}
		for(long long i=n-1; i>=1; i--){
			ans[i] += ans[i+1];
		}
		for(long long i=1; i<=n; i++){
			cout<<ans[i]<<" ";
		}
		cout<<endl;
	}
	return 0;
}
