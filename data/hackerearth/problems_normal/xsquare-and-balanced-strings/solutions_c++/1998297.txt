#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
  ll ts;
  cin>>ts;
  ll ans;
  while(ts--)
  {
     string st;
     ans=0; 
     cin>>st;
     ll n=st.length();
     for(int i=0;i<n;i++)
     {
           ans=ans^(1<<st[i]-'a');
     }
   cout<<(ans==0?"1":"-1")<<endl;
  }
}
