mod = 10**9+7


def square(x, y):
    if y == 0:
        return 1
    res = square(x, y/2)
    res *= res
    if y % 2 == 0:
        return res % mod
    else:
        return (x*res) % mod


test = int(raw_input())
for i in range(test):
    n = int(raw_input())
    result = (square(2, n) - 1)
    print(result)
