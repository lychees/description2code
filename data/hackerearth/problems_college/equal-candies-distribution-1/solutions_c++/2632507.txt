#include<bits/stdc++.h>
#define inf 100000000
using namespace std;

int f(int a[],int n){

    int min1=inf;
    for(int i=0;i<n;i++){
        if(a[i]<min1)
            min1=a[i];
    }
    int count1=0,temp;
    for(int i=0;i<n;i++){
        
        temp = a[i]-min1;
        count1+=temp/5;
        temp%=5;
        count1+=temp/2;
        temp%=2;
        count1+=temp;
    }
    return count1;
}

int main(){
	
    int t;
    cin>>t;
    
    while(t--){
        int n;
        cin>>n;
        int a[n];
        for(int i=0;i<n;i++)
            cin>>a[i];
        
        cout<<f(a,n)<<endl;
    }
}
