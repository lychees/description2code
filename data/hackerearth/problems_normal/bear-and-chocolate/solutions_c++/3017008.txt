#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--){
		int n,i,j,ch=0,ct=0,st1=0,st2=0;
		cin>>n;
		char a[n+1][n+1];
		int cr[n],cc[n];
		for(i=0;i<n;i++){
			for(j=0;j<n;j++){
				cin>>a[i][j];
				if(a[i][j]=='#')ch++;
			}
		}
		if(ch%2)cout<<"NO\n";
		else{
			for(i=0;i<n;i++){
				for(j=0;j<n;j++){
					if(a[i][j]=='#')ct++;
				}
				cr[i]=ct;
			}
			ct=0;
			for(i=0;i<n;i++){
				for(j=0;j<n;j++){
					if(a[j][i]=='#')ct++;
				}
				cc[i]=ct;
			}
			for(i=0;i<n;i++){
				if(cc[i]==ch/2){
					st1=1;
					break;
				}
				if(cr[i]==ch/2){
					st1=1;
					break;
				}
			}
			if(st1)cout<<"YES\n";
			else cout<<"NO\n";
		}
		
	}
	return 0;
}
