#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
    	int n;
    	cin>>n;
    	int arr[n];
    	for(int i = 0;i < n; i++)
    	{
    		cin>>arr[i];
    	}
    	
    	sort(arr, arr + n);
    	
    	for(int i = 0; i < n/2; i++)
    	{
    		arr[i] = arr[i] + arr[n-1-i];
    	}
    	
    	sort(arr, arr + n/2);
    	
    	cout<<arr[(n/2)-1] - arr[0]<<endl;
    }
    return 0;
}
