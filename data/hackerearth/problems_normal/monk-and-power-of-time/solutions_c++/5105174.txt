#include <iostream>
#include<vector>
using namespace std;
int main()
{
	int n , i,t, cnt=0;
	cin>>n;
	vector<int> arr , idl;
	int temp;
	for(i=0;i<n;i++)
	{
		cin>>t;
		arr.push_back(t);
	}
	for(i=0;i<n;i++)
	{
		cin>>t;
		idl.push_back(t);
	}
	while(!arr.empty())
	{

		if(*arr.begin()==*idl.begin())
		{
			arr.erase(arr.begin());
			idl.erase(idl.begin());
			cnt++;
		}
		else
		{
			temp = arr[0];
			arr.erase(arr.begin());
			arr.push_back(temp);
			cnt++;

		}
	}
	cout<<cnt;
    return 0;
}
