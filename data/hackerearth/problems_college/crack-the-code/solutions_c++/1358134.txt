#include <iostream>
using namespace std;

int main()
{
	int t,l,r,i;
	cin>>t;
	long long int a[32];
	a[1] = 0;
	a[2] = 1;
	for(i = 3; i <= 30; i++) {
		a[i] = a[i-1]+a[i-2];
	}
	//cout<<a[30]<<endl;
	int c;
	while(t--) {
		c = 0;
		cin>>l>>r;
		for(i = 1; i <= 50; i++) {
			if(a[i] > r) {
				break;
			}
			if(a[i] >= l) {
				c++;
			}
		}
		puts(c%2?"INVALID":"VALID");
	}
    return 0;
}
