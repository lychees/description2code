#include<iostream>

using namespace std;

int main()
{
	int n;
	cin>>n;
	while(n--)
	{
		long long int N,K,num;
		cin>>N>>K;
		long long int ar[K][2];
		for(int j=0;j<K;j++)
		{
				ar[j][1]=0;
				ar[j][0]=j+1;
		}

		for(int i=0;i<N;i++)
		{
			cin>>num;
			for(int j=0;j<K;j++)
			{
				if(num==ar[j][0])
					ar[j][1]++;
			}
		}
		int sum =0,fin = ar[0][1];
		for(int j=0;j<K;j++)
		{
			sum = sum+ ar[j][1];
			if(ar[j][1]>fin)
			{
				fin = ar[j][1];
			}
					
		}	
		cout<<sum-fin<<endl;
		
	}
	return 0;
}
