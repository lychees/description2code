#include <iostream>
using namespace std;

int gcd(long int a, long int b)
{
	if(a<b)
	{
		long int temp = a;
		a = b;
		b = temp;
	}
	long int r = 1;
	while(b>0)
	{
		r = a%b;
		a = b;
		b = r;
	}
	return a;
}

bool isPrime(long int n)
{
	if(n==1)
		return false;
		
	bool flag = true;
	for(long int i=2; i<=n/2; ++i)
	{
		if(n%i == 0)
		{
			flag = false;
			break;
		}
	}
	return flag;
}

bool rbf(long int n)
{
	long int rp_to_n = 1;
	for(long int i = 2; i<=n; ++i)
	{
		if(gcd(n,i) == 1)
			++rp_to_n;
	}
	return isPrime(rp_to_n);
}

int main()
{
    int t;
    cin >> t;
    while(t)
    {
    	long int n;
    	cin>>n;
    	if(rbf(n))
    		cout<<"TRUE\n";
    	else
    		cout<<"FALSE\n";
    	t--;
    }
    return 0;
}
