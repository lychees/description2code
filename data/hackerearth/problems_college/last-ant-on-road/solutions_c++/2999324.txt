#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	int n,i,a[1001];
    	
    	cin>>n;
    	for(i=0;i<n;i++){
    		cin>>a[i];
    	}
    	
    	int l=0,r=n-1;
    	
    	while(l<r){
    		if(a[l]==-1) l++;
    		if(a[r]==1) r--;
    		
    		for(i=l;i<r;i++){
    			if(a[i]==1 && a[i+1]==-1){
    				a[i]=-1;
    				a[i+1]=1;
    			}
    		}
    	}
    	cout<<l+1<<endl;
    }
    return 0;
}
