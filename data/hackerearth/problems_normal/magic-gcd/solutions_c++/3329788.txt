#include <iostream>
using namespace std;

int gcd(int A,int B){
	if(A<0 || B<0)
		return 0;
	if(A==0)
        return B;
    if(B==0)
        return A;
    if(A==B)
        return A;
    if(A>B)
        return( gcd(B,A%B) );
    return (gcd(A,B%A));
	
}

void solve(){
	int n1,n2,k,m=0;
	cin >> n1;cin >> n2;cin >> k;
	for(int t=0;t<=k;t++){
	for(int i=0;i<=t;i++){
		m=max(m,gcd(n1-i,n2-t+i));
		m=max(m,gcd(n1-i,n2+t-i));
		m=max(m,gcd(n1+i,n2-t+i));
		m=max(m,gcd(n1+i,n2+t-i));
	}
	}
	cout << m<<endl;
}
int main()
{
    int t;
    cin >> t;
    for(int i=1;i<=t;i++)
    	solve();
    return 0;
}
