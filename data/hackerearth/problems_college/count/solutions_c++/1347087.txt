    #include<iostream>

    using namespace std;

    int main()
    {
    long t,r,c,count_e=0,count_o=0;
    cin>>t;
    while(t--)
    { count_e=0,count_o=0;
    cin>>r>>c;
    if(c>=r)
    {
    if(r%2==0)
    {
    long n=r/2;
    count_e=(n*(n+1));
    count_o=(n*(2+(n-1)*2))/2;
    }
    else
    {
    long n=(r+1)/2;
    count_o=(n*(2+(n-1)*2))/2;
    n=(r-1)/2;
    count_e=(n*(n+1));
    }
    }
    else if(r>c)
    {
    while(r!=c)
    {
    if(r%2==0)
    count_e+=c;
    else
    count_o+=c;
    r--;
    }
    if(r%2==0)
    {
    long n=r/2;
    count_e+=(n*(n+1));
    count_o+=n*n;
    }
    else
    {
    long n=(r+1)/2;
    count_o+=n*n;
    n=(r-1)/2;
    count_e+=(n*(n+1));
    }
    }
    cout<<count_e<<" "<<count_o<<endl;
    }
    }
