#include <iostream>
#include<vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	vector<int> v(n);
    	long ans=0;
    	for(int i=0;i<n;i++)
    	{cin>>v[i];ans+=v[i];}
    	if(ans%2==0)
    	cout<<"Yes";
    	else
    	{
    		cout<<"No";
    	}
    	cout<<endl;
    	
    }
    
    return 0;
}
