#define pi M_PI
using namespace std;

int n;
int main() {
  scanf("%d", &n);
  if (n < 5) {
    printf("No solution\n");
    return 0;
  }
  double X = 0, Y = 0, deg = 0, tur = 2 * pi / n, L = 100.00;
  for (int i = 1; i < n; i++) {
    X += cos(deg) * L, Y += sin(deg) * L;
    deg += tur;
    printf("%.5lf %.5lf\n", X, Y);
    L += 0.003;
  }
  printf("%.5lf 0.0000000\n", X - Y / tan(deg));
  return 0;
}
