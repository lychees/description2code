#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
int check(long a[],long m,int n,int c)
{
int jump=0,pos=0;
int j;
for(j=1;j<n;)
{
if(a[j]-a[pos]>m)
{if(j==pos+1)
return 0;
jump++;
pos=j-1;
continue;
}
j++;
}
if(jump<c)
return 1;


else
return 0;
}

int main()
{
int t;
scanf("%d",&t);
while(t--)
{
int n,c;
long l=1;
scanf("%d %d",&n,&c);
long a[n];
int i;
for(i=0;i<n;i++)
{
scanf("%ld",&a[i]);
}
long h=a[n-1]-a[0];
long mid,max=-1;
while(h>l)
{mid=(l+h)/2;
if(check(a,mid,n,c))
{
h=mid;
}
else
l=mid+1;
}
printf("%ld\n",h);
}
return 0;
}
