#include<bits/stdc++.h>

using namespace std;
int main() {

    int t;
    long long int n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        int div3 = n%3;
        int i;

        if(div3 == 0)
        {
            for(i =0;i<n;i++)
               cout<<5;
            printf("\n");
        }

        else if(div3 == 2 && n>=5)
        {
            for(i=0;i<n-5;i++)
                cout<<5;
            for(i=0;i<5;i++)
                cout<<3;
            printf("\n");
        }
        else if(div3 == 1 && n>=10)
        {
            for(i=0;i<n-10;i++)
                cout<<5;
            for(i=0;i<10;i++)
                cout<<3;
            printf("\n");
        }
        else if(n%5 == 0)
        {
            for(i=0;i<n;i++)
                cout<<3;
            printf("\n");
        }
        else
            printf("-1\n");

    }
    return 0;
}
