bool check(ll m)
{
	for(ll i=0;i<n;i++)
	{
		a[i]=b[i];
	}
	a[0]=max(1LL,a[0]-m);
	for(ll i=1;i<n;i++)
	{
		if(a[i]>a[i-1])
		{
			a[i]=max(a[i-1]+1,a[i]-m);
		}
		else
		{
			a[i]=min(a[i-1]+1,a[i]+m);
		}
		if(a[i]<=a[i-1])
		{
			return 0;
		}
	}
