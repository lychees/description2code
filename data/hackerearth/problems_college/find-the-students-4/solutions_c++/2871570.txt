#include <iostream>
using namespace std;

int main()
{
	int N,i;

	int X[N],A[N];
	int c=N;
	for(i=0;i<N;i++)
	{
		cin>>X[i];
		A[i]=0;
	}
	//Minimize
	int j=0,k;
	while(c!=0)
	{
		cout<<c<<endl;
	for(i=0;i<N;i++)
	{
		
		X[i]-=2;
		if(X[i]>0)
			A[i]=X[i];
		else if(X[i]==0 || X[i]<0)
			{
				for(k=i;k<N;k++)
					X[k]=X[k+1];
				i--;
				c--;
			}
	}

}
