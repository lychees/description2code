int m;
int n;
long long int a[100001];
bool ok(long long int val){
	if (val == 0LL){
		return true;
	}
	long long int kari = 0;
	int countt = 0;
	for (int i = 0; i < n&&countt<m; i++){
		kari += a[i];
		if (kari >= val){
			countt++;
			kari = 0LL;
		}
	}
	if (countt == m){
		return true;
	}
	return false;
}
int main(){
	int t;
	scanf("%d", &t);
	while (t--){
		scanf("%d%d", &n,&m);
		long long int sum = 0;
		for (int i = 0; i < n; i++){
			scanf("%lld", &a[i]);
			sum += a[i];
		}
		long long int mint = 0;
		long long int maxt = sum;
		while (mint+1LL < maxt){
			if (ok((mint + maxt) / 2LL)){
				mint = (mint + maxt) / 2LL;
			}
			else{
				maxt = (mint + maxt) / 2LL;
				maxt--;
			}
		}
		if (mint != maxt){
			if (ok(maxt)){
				mint = maxt;
			}
		}
		printf("%lld\n", mint);
	}
	return 0;
}
