#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
	ll  n,q;
	vector<ll>p;
	cin>>n>>q;
	ll mini=0;
	for(int i=0;i<n;i++)
	{
		ll a,b;
		cin>>a>>b;
		if(a>b)swap(a,b);
		mini+=a;
		b=b-a;
		p.push_back(b);
	}
	sort(p.begin(),p.end());
	bool flag=true;
	ll val=0;
	for(int i=0;i<n;i+=2)
	{
		if(p[i]!=p[i+1])
		{
			flag=false;
		}
		val+=p[i];
	}
	val+=mini;
	for(int i=0;i<q;i++)
	{
		ll x;
		cin>>x;
		if(x==val&&flag==true)
			cout<<"YES"<<endl;
		else cout<<"NO"<<endl;

	}

	return 0;
}
