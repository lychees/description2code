for j in range(input()):
    N = input()
    data = raw_input()
    chocper = []
    chocper = data.split(" ")
    sum = 0.0
    for i in range(N):
        sum = int(chocper[i])+sum

    avg = sum/N
    avg = format(avg, ".8f")
    print avg
