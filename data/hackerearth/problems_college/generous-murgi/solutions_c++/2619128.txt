#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long int x,y;
		scanf("%ld%ld",&x,&y);
		if(y>x)
		{
			cout<<"0\n";
			continue;
		}
		if(y==x)
		{
			printf("-1\n");
		}
		else if(y==0)
		{
			int count=0;
			int square_root = (int) sqrt(x);
			for (int i = 1; i <= square_root; i++){ 
				if (x % i == 0&&i>y)
    					count++;
    				int timepass=x/i;
    				if (x % timepass == 0&&timepass>y)
    					count++;
			}
			if(y<square_root&&square_root*square_root==x)
			{	count--;}			
			printf("%d\n",count);
		}
		else
		{
			int count=0;
			long int temp=x;
			x-=y;
			int square_root = (int) sqrt(x);
			for (int i = 1; i <= square_root; i++){ 
				if (x % i == 0&& temp%i!=0&&i>y)
    					count++;
    				int timepass=x/i;
    				if (x % timepass == 0&& temp%timepass!=0&&timepass>y)
    					count++;
			}
			if(y<square_root&&square_root*square_root==x)
			{	count--;}			
			printf("%d\n",count);
		}	
	}	
	return 0;
}
