from collections import Counter
# print 'Hello World!'
for i in range(int(raw_input())):
    x = raw_input()
    size = len(x)
    if size >= 10:
        print "YES"
    elif size < 4:
        print "NO"
    else:
        y = len(set(x))
        if size == 4:
            if y == 4:
                print "YES"
            else:
                print "NO"
        elif size == 5:
            if y <= 2:
                print "NO"
            else:
                print "YES"
        elif size == 6:
            if y >= 3:
                print "YES"
            elif y == 2:
                z = Counter(x)
                if 1 in z.values():
                    print "NO"
                else:
                    print "YES"
            else:
                print "NO"
        else:
            if y < 2:
                print "NO"
            else:
                print "YES"
