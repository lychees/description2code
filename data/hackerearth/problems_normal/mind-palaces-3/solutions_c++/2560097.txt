#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    long long n,i,j,m,q;
    long long x,a[1000][1000];
    scanf("%lld%lld",&n,&m);
    for(i=0;i<n;i++)
    {
    	for(j=0;j<m;j++)
    	scanf("%lld",&a[i][j]);
    }
    scanf("%lld",&q);
    int flag=0;
    while(q--)
    {flag=0;
    	scanf("%lld",&x);
    for(i=0;i<n;i++)
    {
    	for(j=0;j<m;j++)
    	{if(x==a[i][j]){printf("%lld %lld\n",i,j);flag=1;break;}}
    if(flag)break;
    }
    if(flag==0)printf("-1 -1\n");
    }
    return 0;
}
