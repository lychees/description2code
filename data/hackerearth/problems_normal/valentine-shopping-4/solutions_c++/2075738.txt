#include <iostream>
using namespace std;

int main() {
int t,n,i ,m,maxm=0,p1,p2,p3,id;

cin>>t;
while(t--)
{
	cin>>n>>m;
	while(n--)
	{maxm=1000000;
		for(i=1;i<=m;i++)
		{   int tot=1000000;
			cin>>p1>>p2>>p3;
			tot=tot-((tot*p1)/100);
			tot=tot-((tot*p2)/100);
			tot=tot-((tot*p3)/100);
			if(tot<maxm)
			{
			maxm=tot;
			id=i;
			}
		}
		cout<<id<<" ";
	}
	cout<<endl;
}


return 0;
}
