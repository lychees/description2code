t = int(raw_input())

for tc in range(t):
	te_in1 = raw_input().split(" ")
	n = int(te_in1[0])
	x = int(te_in1[1])
	te_in2 = raw_input().split(" ")
	arr = []
	for z in te_in2:
		arr.append(int(z))
	cnt = 0
	for j in range(n-1):
		if arr[j] >= arr[j+1]:
			s = arr[j]-arr[j+1]
			temp = s/x
			cnt += (temp+1)
			arr[j+1] = arr[j+1] + (temp+1)*x
	print cnt
