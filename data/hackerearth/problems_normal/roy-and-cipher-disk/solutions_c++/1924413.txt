#include <bits/stdc++.h>
#define ll long long int
#define ALPABHET_COUNT 26
using namespace std;
 
ll ab(ll a){
	return ((a > 0) ? a : ALPABHET_COUNT + a);
}
int main(){
    ll t = 0,len = 0,i = 0,start = 0,dist = 0;
    cin >> t;
    while(t--){
    	string s;
    	cin >> s;
    	len = s.length();
    	start = 0;
    	for(i = 0; i < len; i++) {
         dist = ab(s[i] - 'a' - start);
         cout << ((dist <= (ALPABHET_COUNT / 2)) ? dist : (dist - ALPABHET_COUNT));
         cout << ((i < len - 1) ? " " : "");
         start= s[i] - 'a';
    	}
    	cout << endl;
    }
    return 0;
}
