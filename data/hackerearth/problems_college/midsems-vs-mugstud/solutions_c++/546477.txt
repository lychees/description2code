#include <iostream>
#include<cstdio>
using namespace std;
int prev1[100][100],temp[100][100],curr[100][100],res,ans,t,num,n;
int main()
{
    //freopen("input.txt","r",stdin);
    scanf("%d",&t);
    while(t--)
    {
    	for(int i=0;i<100;i++)
    		for(int j=0;j<100;j++)
    			prev1[i][j]=0;
    	scanf("%d",&n);
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<n;j++)
    		{
    			for(int k=0;k<n;k++)
    			{
    				scanf("%d",&num);
    				temp[j][k]=num;
    				curr[j][k]=num-prev1[j][k];
    			}
    		}
    		for(int j=0;j<n;j++)
    		{
    			for(int k=0;k<n;k++)
    			{
    				ans=curr[j][k];
    				if(k-1>=0)
    					ans=ans-curr[j][k-1];
    				if(j-1>=0)
    					ans=ans-curr[j-1][k];
    				if(j-1>=0&&k-1>=0)
    					ans=ans+curr[j-1][k-1];
    					printf("%d ",ans);
    				prev1[j][k]=temp[j][k];
    			}
    			printf("\n");
    		}
    	}

    }
    return 0;
}
