#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
int n,a,b,num;
cin>>n;
for(int i=0;i<n;i++)
{
cin>>a;
cin>>b;
num=a%b;
if(b%2==0)
{
if(num<b/2)
cout<<(a/b)*b<<"\n";
else if(num>=b/2)
cout<<((a/b)+1)*b<<"\n";
}
else if(b%2==1)
{
if(num<=b/2)
cout<<(a/b)*b<<"\n";
else if(num>b/2)
cout<<((a/b)+1)*b<<"\n";
}

}
return 0;
}
