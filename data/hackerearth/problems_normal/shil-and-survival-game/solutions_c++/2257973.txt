#include <iostream>
using namespace std;

int main()
{
   long long ar[100005];
   int n;
   cin>>n;
   for(int i=0;i<n;i++){
   	long long num;
   	cin>>num;
   	ar[i]=num;
   }
   long long l2r[n];long long r2l[n];
   l2r[0]=ar[0];
   for(int i=1;i<n;i++){
   	l2r[i]=max(l2r[i-1],ar[i]);
   }
   r2l[n-1]=ar[n-1];
   for(int i=n-2;i>=0;i--){
   	r2l[i]=max(r2l[i+1],ar[i]);
   }
   for(int i=0;i<n;i++){
   	if(l2r[i]==ar[i]||r2l[i]==ar[i])
   		cout<<i+1<<" ";
   	}
   
}
