#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int t =0;
	cin>>t;
	while(t--)
	{
		int k;
		long long int counter=0;
		vector <long long int> v;
		long long int D;
		cin>>k;
		cin>>D;
		for(int i =0;i<k;i++)
		{
			long long int temp ;
			cin>>temp;
			v.push_back(temp);
		}

		sort(v.begin() ,v.end());

		// for(int i =0;i<k;i++)
		// {
		// 	cout<<v[i]<<"  ";
		// }
		// cout<<endl;
		for(int i =0; i<k-1;i++)
		{
			if(v[i+1]-v[i]<=D)
			{
				counter++;
				i=i+1;
			}
		}
		cout<<counter<<endl;
		
	}
    return 0;
}
