#include<bits/stdc++.h>
using namespace std;
int dp[100005][26];
char str[100005];
int n;
void update(int index,int val,int pos)
{
	int i;
	for(i=index;i<=n;i+=(i&-i))
	{
		dp[i][pos]+=val;
	}
}
int query(int index,int pos)
{
	int i,sum=0;
	for(i=index;i>0;i-=(i&-i))
	{
		sum=sum+dp[i][pos];
	}
	return sum;
}
int main()
{
	int q,i;
	scanf("%d%d",&n,&q);
	scanf("%s",str);
	for(i=1;i<=n;i++)
	{
		update(i,1,str[i-1]-'a');
	}
	int t,l,r;
	char x;
	while(q--)
	{
		cin>>t;
		if(t==1)
		{
			cin>>l>>x;
			update(l,1,x-'a');
			update(l,-1,str[l-1]-'a');
			str[l-1]=x;
		}
		else if(t==2)
		{
			int count=0,temp;
			cin>>l>>r;
			for(i=1;i<=26;i++)
			{
				temp=query(r,i-1)-query(l-1,i-1);
				if(temp%2==1)
					count++;
			}
			if(count<=1)
				printf("yes\n");
			else
				printf("no\n");
		}
	}
	return 0;
}
