import math
t = int(raw_input())
while t:
	flag=0
	n = int(raw_input())
	arr = [int(i) for i in raw_input().split()]
	item = max(arr)
	if item==0:
		print 'NO'
		t-=1
		continue
	l = int(math.log(item,2))
	for i in range(l+1):
		bm = 1<<i	##i'th bit set
		cm = -1		## all bit set
		for ele in arr:
			if bm&ele:
				if cm==-1:
					cm = ele
				else:
					cm = cm&ele
		if cm == bm:
			flag=1
			break
	if flag:
		print 'YES'
	else:
		print 'NO'
	t-=1
