#include <bits/stdc++.h>
using namespace std;
#define ll long long
//#define cl ios_base::sync_with_stdio(false); cin.tie(NULL)
int main()
{
  //  cl;
    ll n,a[123456];
    cin>>n;
    for (ll i = 0; i <n; ++i)
    {
        cin>>a[i];
    }
    sort(a,a+n);
    ll i=0,j=n-1,sum1=0,sum2=0;
    while(i<j)
    {
        sum1 += abs(a[j] - a[i])+abs(a[j] - a[i+1]);
        i++,j--;
    }
    i=0,j=n-1;
    sort(a,a+n,greater<int>());
    while(i<j)
    {
        sum2 += abs(a[j] - a[i])+abs(a[j] - a[i+1]);
        i++,j--;
    }
    cout<<max(sum1,sum2)<<endl;
    return 0;
}
