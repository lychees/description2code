#include <iostream>
using namespace std;
int c[333];
int main()
{
    string s;
    int i,n;
    int re,x;
    while(1)
    {
    	cin>>s;
    	if(s=="0")
    	break;
    	cin>>x;
    	for(i='A';i<='Z';i++)
    		c[i]=0;
    	re=0;
    	n=s.size();
    	for(i=0;i<n;i++)
    	{
    		if(c[s[i]]==1)
    		{
    			x++;
    		}
    		else if(c[s[i]]==0)
    		{
    			if(x==0)
    			{
    				re++;
    				c[s[i]]=2;
    			}
    			else
    			{
    				x--;
    				c[s[i]]=1;
    			}
    		}
    	}
    	cout<<re<<'\n';
    }
    return 0;
}
