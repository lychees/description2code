#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,carry=0;
char s[1001],s1[1001],n,n1,i,j,c;
cin>>t;
while(t--)
{
    carry=0;
  cin>>s>>s1;
   n=strlen(s);
   n1=strlen(s1);
    if(n<n1)
    {
        for(i=n-1,j=n1-1;i>=0;i--,j--)
        {
          // cout<<s[i]<<s1[j]<<endl;

            if((s[i]-'0')+(s1[j]-'0')>9)
            carry++;
          //  cout<<"carry = "<<carry<<endl;
        }
    }
    else
    {
        for(j=n1-1,i=n-1;j>=0;j--,i--)
        {

        // cout<<s[i]<<s1[j]<<endl;
          //c=(s1[j]-'0')+(s[i]-'0');
         // cout<<c<<endl;
            if((s1[j]-'0')+((s[i]-'0'))>9)
          carry++;
           // cout<<"carry = "<<carry<<endl;
        }
    }
   if(carry==0)
        cout<<"No carry operation"<<endl;
   else if(carry==1)
    cout<<"1 carry operation"<<endl;
   else
    cout<<carry<<" carry operations"<<endl;

}
return 0;
}
