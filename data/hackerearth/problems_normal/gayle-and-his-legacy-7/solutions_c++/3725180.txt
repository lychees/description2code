bool cmp(pair<long long, long long> x, pair<long long, long long> y)
{
    return x.second < y.second;
}

	while(t--) {
        long long int n,k,i,m,c = 0,f = 0;

	    for (i = 0; i < n; i++) {
            scanf("%lld%lld",&(op[i].first),&(op[i].second));
	    }
	    sort(op,op+n,cmp);
       for (i = 0; i < n; i++) {
                if (m <= op[i].second) {
