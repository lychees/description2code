#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int homecolor[100005]={0};
	int awaycolor[100005]={0};
	int mask[100003]={0};
	for(int i=0;i<n;i++)
	{
		int t1,t2;
		cin>>t1>>t2;
		homecolor[i]=t1;
		awaycolor[i]=t2;
		mask[t1]++;
	}
	for(int i=0;i<n;i++)
	{
		int ans=n-1;
		ans=ans+mask[awaycolor[i]];
		cout<<ans<<" "<<(2*(n-1))-ans<<endl;
	}

	exit(0);
}
