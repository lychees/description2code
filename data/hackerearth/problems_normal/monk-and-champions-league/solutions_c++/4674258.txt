#include <iostream>
using namespace std;
long long hap[1000001];
void heapify(int ind){
	int par = ind/2;
	while(par>0){
		if(hap[par]<hap[ind]){
			int t = hap[par];
			hap[par] = hap[ind];
			hap[ind] = t;
			ind = par;
			par = ind/2;
		}else
			break;
		
	}
	
}
void deapify(int nu){
	int par = 1;
	int left = 2*par;
	int right = left+1;
	while(left<=nu&&right<=nu){
		int bgr = left;
		if(hap[bgr]<hap[right])
			bgr = right;
		if(hap[bgr]>hap[par]){
			int t = hap[par];
			hap[par] = hap[bgr];
			hap[bgr] = t;
			par = bgr;
			left = 2*par;
			right = left+1;
		}else{
			break;
		}
		
	}
	if(left<=nu&&hap[left]>hap[par]){
		int t = hap[left];
		hap[left] = hap[par];
		hap[par] = t;
	}
	
}
int main()
{
    int m,n;
    cin>>m>>n;
    int ind = 1;
    long long val;
    for(int i=0;i<m;i++){
    	cin>>val;
    	hap[ind] = val;
    	heapify(ind);
    	ind++;
    	
    }
    ind--;
    long long ans = 0;
    for(int i=0;i<n;i++){
    	//cout<<hap[1]<<"---> "<<hap[ind]<<endl;
    	ans += hap[1];
    	hap[1]--;
    	deapify(ind);
    }
    cout<<ans<<endl;
    return 0;
}
