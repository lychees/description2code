#include<bits/stdc++.h>
using namespace std;
int main()
{
    string str;
    while(getline(cin,str))
    {
    int i,n=str.size();
    for(i=0;i<n;i++)
        if(str[i]=='_' && str[i+1]!='_' && i+1<n)
        str[i+1]-=32;
    if(str[0]!='_')
    str[0]-=32;
    cout<<str<<" ";
    }
}
