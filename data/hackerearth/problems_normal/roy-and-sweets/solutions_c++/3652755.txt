typedef long long ll;
#include <iostream>
#include <stdio.h>
#include <cmath>
#include <algorithm>
#include <vector>
#include <string>
#include <string.h>
#include <map>
#include <queue>
#include <stack>
#include <sstream>
#include <fstream>
#include <stdlib.h>
#include <utility>

using namespace std;
ll dp[1000001];
map<ll,ll> m;
int main()
{
    ll n,maxi,x,y,i,j,q,w,sum;
    scanf("%lld",&n);
    m.clear();
    for(i=0;i<n;i++)
    {
    scanf("%lld %lld",&x,&y);
    m[x]=y;
    }
    memset(dp,0,sizeof(dp));
    maxi=0;
    for(i=1000000;i>0;i--)
    {
        if(m[i]>maxi)
        maxi=m[i];
        dp[i]=maxi;
    }
    sum=0;
    scanf("%lld",&q);
    while(q--)
    {
        scanf("%lld",&w);
        sum+=(dp[w]/w)*100;
    }
    printf("%lld\n",sum);
	return 0;
}
