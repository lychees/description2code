vector<bool> isprime;
vector<int> primes;
void sieve(int n)
{
    isprime.resize(n + 1);
    for (int i = 0; i <= n; i++)
        isprime[i] = 1;
    isprime[1] = 0, isprime[2] = 1;
    for (int i = 2; i * i <= n; i++)
        if (isprime[i])
            for (int j = 2; i * j < n; j++)
                    isprime[i * j] = 0;
    for (int i = 2; i < n; i++)
        if (isprime[i])
            primes.push_back(i);
    return;
}

long long pow(long long n, long long m)
{
    if(m==0) return 1;
    long long x=pow(n,m/2);
    if(m%2==0) return (x*x)%MOD;
    else return (((x*x)%MOD)*n)%MOD;
}
