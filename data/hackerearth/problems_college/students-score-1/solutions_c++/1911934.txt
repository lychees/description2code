#include <iostream>
#include <string>
#include <iomanip>
using namespace std;

int fact(int n){
	return (n==0 || n==1)? 1 : fact(n-1)*n;
}

int main() {



	int a;
	cin>>a;
	int max=0;
	int temp=0;

	int res[a];
	for (int i=0;i<a;i++){
		cin>>res[i];
		temp=temp+res[i];
		if(temp<=0)
			temp=0;
		if(temp>max)
			max=temp;
	}

	cout<<max;
	return 0;

}
