#include <iostream>

using namespace std;

int main()
{
        int test_cases, a;
        cin>>test_cases;
        while(test_cases > 0)
        {
                int X1,X2,Y1,Y2;
                cin>>X1>>Y1>>X2>>Y2;
                
                if(X2==Y1 && X1==Y2)
                a=2;
                else if(X2==Y1 || X1==Y2)
                a=1;
                else
                a=0;
                cout<<"\n"<<a;
                
                test_cases --;
        }
        return 0;
}
