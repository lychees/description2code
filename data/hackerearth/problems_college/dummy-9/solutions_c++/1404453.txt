#include <iostream>
#include<algorithm>

using namespace std;

struct pairs
{
	int x,y;
};

bool myf(pairs p1,pairs p2)
{
	if(p1.x==p2.x)
		return p1.y < p2.y;
	return p1.x < p2.x;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
    	int n;
    	cin>>n;
    	pairs *p;
    	p=new pairs[n];
    	for(int i=0;i<n;i++)
    		cin>>p[i].x;
    	for(int i=0;i<n;i++)
    		cin>>p[i].y;
    	sort(p,p+n,myf);
    	//applying longest incresing subsequence.
    	int *l;
    	l=new int[n];
    	int ans=-1;
    	for(int i=0;i<n;i++)
    	{
    		l[i]=0;
    		for(int j=0;j<=i-1;j++)
    		{
    			if(p[i].y>=p[j].y && l[j]>l[i])
    				l[i]=l[j];
    		}
    		l[i]+=1;
    		if(ans<l[i])
    			ans=l[i];
    	}
    	cout<<ans<<endl;
//    	for(int i=0;i<n;i++)
//    	{
//    		cout<<p[i].x<<" "<<p[i].y<<endl;
//    	}
	}
    return 0;
}
