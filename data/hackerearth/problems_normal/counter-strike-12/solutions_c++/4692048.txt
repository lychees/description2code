#include <iostream>
#include <vector>
#include<math.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t>0)
	{
		int n,m;
		float d;
		cin>>n>>m>>d;
		vector<pair<int,int> > vec1;
		vec1.resize(n);
		vector<pair<int,int> > vec2;
		vec2.resize(m);
		int cnt=0;
		for(int i=0;i<n;i++)
		{
			cin>>vec1[i].first;
			cin>>vec1[i].second;
		}
		for(int i=0;i<m;i++)
		{
			cin>>vec2[i].first;
			cin>>vec2[i].second;
		}
		for(int i=0;i<m;i++)
		{
			
			for(int j=0;j<n;j++)
			{
				if(d>=abs(vec1[j].first-vec2[i].first)+abs(vec1[j].second-vec2[i].second) )
				{
					cnt++;
					break;
				}
			}
		}
		if(cnt>m/2+1)
			cout<<"YES\n";
		else
			cout<<"NO\n";
		t--;	
		
	}
}
