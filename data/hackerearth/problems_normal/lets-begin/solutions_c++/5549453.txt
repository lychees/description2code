#include<bits/stdc++.h>
using namespace std;
long int dp[1000005];
int main()
{
    long int t,n,i;
    cin>>t;
    dp[1]=INT_MAX;
    dp[2]=1;
    dp[3]=1;
    dp[4]=2;
    dp[5]=1;
    dp[6]=2;
    dp[7]=1;
    for(i=8;i<1000005;i++)
    {
        dp[i]=1+min(dp[i-2],min(dp[i-3],min(dp[i-5],dp[i-7])));
    }
    while(t--)
    {
        cin>>n;
        if(n==1)
            cout<<-1<<endl;
        else
            cout<<dp[n]<<endl;

    }
    return 0;
}
