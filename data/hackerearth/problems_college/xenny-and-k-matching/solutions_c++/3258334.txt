#include<bits/stdc++.h>
typedef long long int ll;
using namespace std;
int main()
{
ll i,n,k,a[1000000],cnt=0;
cin>>n>>k;
for(i=0;i<n;i++)
cin>>a[i];
for(i=0;i<n;i++)
{
if(i+k<n)
{ 
if((abs(a[i]+a[i+k]))%k==0 || (abs(a[i]-a[i+k]))%k==0)
{
cnt++;
//cout<<i<<":"<<a[i]<<" "<<i+k<<":"<<a[i+k]<<endl;
}}
}
cout<<cnt<<endl;
return 0;
}
