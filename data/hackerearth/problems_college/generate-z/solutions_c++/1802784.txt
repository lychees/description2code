#include<bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	int n1=n-2;
	int temp=n-2;
	for(int i=0;i<n;i++){
		cout<<'*';
	}
	cout<<endl;
	for(int i=0;i<temp;i++){
		for(int j=0;j<n;j++){
			if(j==n1){
				cout<<'*';
				break;
			}
			else
				cout<<" ";
		}
		cout<<endl;
		n1--;
	}
	for(int i=0;i<n;i++){
		cout<<'*';
	}
	return 0;
}
