#include <bits/stdc++.h>
using namespace std;
int a[200001], x;
int main()
{
    int n, m;
    scanf("%d %d", &n, &m);
    for (int i = 0; i < n + n; ++i) {
        scanf("%d", &a[i]);
    }
    sort(a , a + n + n);
    int ans = 0;
    for (int i = 0; i < n; ++i) {
        if (a[i] > m) {
            break;
        }
        m -= a[i];
        ++ans;
    }
    cout<<ans;
}
