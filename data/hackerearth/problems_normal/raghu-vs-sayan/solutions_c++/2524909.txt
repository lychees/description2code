#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

vector <int> dish;

int main()
{
   long long int t,n,r,s,cal;
    cin>>t;
    while(t--)
    {
    	dish.clear();
    	cin>>r>>s>>n;
    	for(int i=0;i<n;i++)
    	{
    		cin>>cal;
    		dish.push_back(cal);
		}
		sort(dish.begin(),dish.begin()+n);
		int cr=0,cs=0;
		
		for(int j=0;j<n;j++)
		{
			if(r>=dish[j])
			 {
			 	r-=dish[j];
			 	cr++;
			 }
			 if(s>=dish[j])
			 {
			 	s-=dish[j];
			 	cs++;
			 }
		}
		if(cr>cs)
		cout<<"Raghu Won"<<endl;
		else if(cs>cr)
		cout<<"Sayan Won"<<endl;
		else
		cout<<"Tie"<<endl;
	}
	
	return 0;
	}
