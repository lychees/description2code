#include <iostream>
using namespace std;

int main()
{
    int t, n, sam, gangsterCount;
    bool flag;
    cin>>t;
    
    for(int i=0;i<t;i++)
    {
    	cin>>n>>sam;
    	gangsterCount=0;
    	flag=false;
    	long *arr = new long[n];
    	for(int k=0;k<n;k++)
    	{
    		cin>>arr[k];
    		if(arr[k] == sam)
    		{flag = true;}
    		
    		else if(arr[k]>sam)
    		{
    			gangsterCount++;
    		}
    		
    	}
    	if(!flag)
    	{
    		gangsterCount=0;
    		cout<<"NO "<<gangsterCount<<endl;
    	}
    	else 
    	{
    		cout<<"YES "<<gangsterCount<<endl;
    	}
    	
    	
    }
    
    
    
    return 0;
}
