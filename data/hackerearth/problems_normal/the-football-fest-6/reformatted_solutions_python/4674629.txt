t = int(raw_input())
for i in range(t):
    n, init = [int(val) for val in raw_input().rstrip().split()]
    stack = []
    stack.append(init)
    for j in range(n):
        p = raw_input().rstrip().split()
        if p[0] == 'P':
            stack.append(p[1])
        else:
            stack.append(stack[len(stack)-2])
    print "Player", stack.pop()
