#include <bits/stdc++.h>
using namespace std;
long long MOD = 1e9+7;
 
long long power(long long a,long long b)
{
	if(b==0)
		return 1;
	if(b==1)
		return a;
	long long c=power(a,b/2);
	c=(c*c)%MOD;
	if(b%2)
		c=(c*a)%MOD;
	return c;
}

int main()
{
    long long t,n,n1,num;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	n1=n;
    	set <long long> s;
    	while(n--)
    	{
    	   cin>>num;
    	   s.insert(num);
    	}
    	cout<<power(n1,s.size())<<"\n";
    }
    return 0;
}
