#include<bits/stdc++.h>

using namespace std;

int main(){

    int T,N;
    int ori,togg;
    scanf("%d",&T);

    string bits;

    while(T--){

        cin>>bits;
        cin>>N;
        ori=0;togg=0;
        for(int i=0;i<bits.length();i++){

                if(bits[i]=='1')
                    ori++;
                else
                    togg++;
        }
        N=(N*(N+1))/2;
        if(N&1)
            cout<<togg<<endl;
        else
            cout<<ori<<endl;


    }

    return 0;
}
