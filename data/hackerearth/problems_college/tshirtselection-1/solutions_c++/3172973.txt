#include<bits/stdc++.h>
using namespace std;
int main(){

 cin.sync_with_stdio(false);

      int t;
      cin >> t;
      while( t-- ){

         int n;
         cin >> n;
          int color[n+1];
          for( int i = 1 ; i <= n;  ++i )
               cin >> color[i];

               int len = 0;
               int l  = 1 ,  left = 1 , right = 1;
           int en = 1;
           map< int , int > my;
          while(  en  <= n ){
               if( my[color[en]]  == 0 ){
                if((en - l + 1) > len ){
                 len = en - l + 1;
                 left = l;
                 right = en;
                }
               my[color[en]]++;
               en++;
               }
               else {
                my[color[l]]--;
                l++;

               }
          }

          cout << left <<" "<< right << endl;

      }
      return 0;

}
