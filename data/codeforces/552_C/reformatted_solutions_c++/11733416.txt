#include <algorithm>
#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;

int main() {
  int m, w;
  cin >> w >> m;
  while (m) {
    if (!((m - 1) % w))
      m--;
    else if (!((m + 1) % w))
      m++;
    else if (m % w) {
      cout << "NO" << endl;
      return 0;
    }
    m /= w;
  }
  cout << "YES" << endl;
  return 0;
}
