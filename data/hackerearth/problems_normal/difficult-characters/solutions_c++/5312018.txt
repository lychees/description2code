#include <iostream>
#include <queue>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	string s;
   	cin>>s;
   	int n = s.length();
   	int occ[26] = {0};
   	
   	for(int i =0;i<n;i++)
   	{
   		int u = s[i]- 'a';
   		//cout<<u;
   		occ[u]++;
   	}
   	
   for(int i =0;i<n;i++)
   {
   	for(int j =25;j>=0;j--)
   	{
   		char r = 'a' + j;
   		if(occ[j] == i )
   		cout<<r<<' ';
   	}
   }
   cout<<endl;
   }
    return 0;
}
