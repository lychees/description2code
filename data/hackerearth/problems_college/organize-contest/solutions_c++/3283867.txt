#include <iostream>
#include <string>
#include <vector>

using namespace std;

struct node
{
	string name;
	int count;
	int index;
};

int main(){
	int N;

	int max=0;
	string str,result;
	int r;
	vector< node > a;
	cin >> N;
	int M=N;
	while(N--){
		cin >> str >> r;
		if(r<0)r=0;
		int match=-1;
		for(int i=0;i<a.size();i++){
			if(a[i].name==str){
				a[i].count+=r;
				a[i].index=M-N+1;
				match=0;
				if(max<a[i].count){
					max=a[i].count;
					result=a[i].name;
				}
				break;
			}
		}
		if(match==-1){
			node temp;
			temp.name=str;
			temp.index=M-N+1;
			temp.count=r;
			if(max<r){
				max=r;
				result=str;
			}
			a.push_back(temp);
		}


	}
	// for(int i=0;i<a.size();i++){
	// 	cout<<a[i].name<<" "<<a[i].index << " " << a[i].count<<endl;
	// }

	cout<<result;

	return 0;
}
