#include<bits/stdc++.h>
using namespace std;
unsigned long long int ar[100001];
int main()
{
	long long int n,t;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(int i=1;i<=n;i++)
		cin>>ar[i];
		unsigned long long int val=ar[1];
		for(int i=2;i<=n;i++)
		{
			if(ar[i]>val)
			val=ar[i];
			else
			{
				cout<<"NO\n";
				goto sos;
			}
		}
		cout<<"YES\n";
		sos:;
	}
}
