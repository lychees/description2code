#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    
    for(int i=0;i<t;i++)
    {
    	string s;
    	cin >> s;
    	
    	bool b=true;
    	while(b)
    	{
    		int n=s.length();
    		if(n==1)
    		break;
    		
    		for(int j=0;j<s.length();j++)
    		{
    			if(j==0)
    			{
    				if(s.at(0)=='x' && s.at(1)=='o')
    				{
    					s.erase(s.begin()+1);
    				}
    			}
    			else if(j==s.length()-1)
    			{
    				if(s.at(j)=='x' && s.at(j-1)=='o')
    				{
    					s.erase(s.begin()+j-1);
    					j--;
    				}
    			}
    			else
    			{
    				if(s.at(j)=='x' && s.at(j+1)=='o')
    				{
    					s.erase(s.begin()+j+1);
    				}
    				if(s.at(j)=='x' && s.at(j-1)=='o')
    				{
    					s.erase(s.begin()+j-1);
    					j--;
    				}
    			}
    		}
    		if(n==s.length())
    		b=false;
    	}
    	cout << s << endl;
    }
}
