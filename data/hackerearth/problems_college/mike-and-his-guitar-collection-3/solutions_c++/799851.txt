#include <iostream>
#include <vector>
#include <set>
using namespace std;

int Process(vector<int> &vals) {
set<int> s;
int x;
s.insert(0);
int last_sum = 0;
for(int i = vals.size() - 1; i >= 0; --i) {
set<int> temp(s);
for(set<int> ::iterator itr = s.begin(); itr != s.end(); ++itr) {
temp.insert(*itr + vals[i]);
x=*itr + vals[i];
}
s.swap(temp);
}
return s.size();
}

int main()
{
int t;
cin >> t;
while(t--) {
int guitars;
cin >> guitars;
vector<int> vals(guitars);
for(int i = 0; i < vals.size(); ++i)
cin >> vals[i];
cout << Process(vals) << endl;
}
return 0;
}
