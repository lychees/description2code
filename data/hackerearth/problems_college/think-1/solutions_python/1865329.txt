def fast_exponentiation(a , n):
    if n == 0:
        return 1
    if n == 1:
        return a
    t = fast_exponentiation(a , n/2)
    return ( t * t * fast_exponentiation(a , n%2))
for i in xrange(input()):
    num = raw_input()
    num2 = num[::-1]
    e1 = str(fast_exponentiation(int(num) , 2))
    e2 = str(fast_exponentiation(int(num2) , 2))
    e_2 = e2[::-1]
    if num == num2:
        print "No"
    elif e1 == e_2:
        print num2
    else:
        print "No"
