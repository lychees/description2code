#include<iostream>
#include<algorithm>
#include<cmath>
#include<string.h>
#include<vector>
#include<map>
#include<limits.h>
using namespace std;

vector<string>v1;
map<string,int>mymap;
map<string,int>:: iterator it;
vector<pair<int,string> >v;
pair<int,string>apair;

int main()
{
  int n,i,ans=0,temp,x=0,y=0,k=0,j;
  string s,s1;
  cin>>n;
   for(i=0;i<n;i++){cin>>s;v1.push_back(s);mymap[s]++;}
   for(it=mymap.begin();it!=mymap.end();it++)
   {
       apair.first=(*it).second;
       apair.second=(*it).first;
       v.push_back(apair);

   }
   for(i=1;i<v.size();i++)
   {
       for(j=0;j<v.size()-i;j++)
       {
           if(v[j].first<v[j+1].first)
           {
               temp=v[j].first;s1=v[j].second;
               v[j].first=v[j+1].first;v[j].second=v[j+1].second;
               v[j+1].first=temp;v[j+1].second=s1;
           }
           if(v[j].first==v[j+1].first)
           {
               for(k=0;k<v1.size();k++){if(v1[k]==v[j].second)x=k;if(v1[k]==v[j+1].second)y=k;}
               if(x>y)
               {
               temp=v[j].first;s1=v[j].second;
               v[j].first=v[j+1].first;v[j].second=v[j+1].second;
               v[j+1].first=temp;v[j+1].second=s1;
               }
           }
       }
   }
   for(i=0;i<v.size();i++){cout<<v[i].second<<" "<<v[i].first<<endl;}
   return 0;
}
