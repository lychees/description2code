#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
     int n,a,i,j,l=0,k=0,q;
     cin>>n>>a;
     int arr[n],arr1[n];

     for(i=0;i<n;i++)
        {
        arr[i]=i;
        }
        l=n/2;
        q=n;
        for(i=0;i<=l;i++)
        {
            for(j=(--q);j>=l;)
            {
                arr1[k++]=arr[j];
                // cout<<arr1[k-1]<<" ";
             break;
            }
            arr1[k++]=arr[i];
            // cout<<arr1[k-1]<<" ";
            // cout<<endl;
        }
         for(i=0;i<(k-1);i++){
       if(a==arr1[i])
        cout<<i<<endl;
         }

    }
    return 0;
}
