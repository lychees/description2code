#include <iostream>
#include<cmath>
#include<iomanip>
using namespace std;

int main()
{  long long int T,i;
    cin>>T;
    long  double a[T],b[T],c[T],area[i],r[i];
    for(i=0;i<T;i++){
    	cin>>a[i]>>b[i]>>c[i];
    }for(i=0;i<T;i++){
    r[i]=(a[i]*b[i]*c[i])/sqrt((a[i]+b[i]+c[i])*(c[i]+b[i]-a[i])*(a[i]+c[i]-b[i])*(a[i]+b[i]-c[i]));
    area[i]=(3.1415)*r[i]*r[i];
    cout<<setprecision(4)<<fixed<<area[i]<<"\n";
    }

    return 0;
}
