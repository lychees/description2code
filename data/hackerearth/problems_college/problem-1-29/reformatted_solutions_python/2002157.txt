from sys import stdin
from itertools import imap
from collections import deque


def meet(maze, n):
    if maze[0][0]:
        queue = deque([(0, 0)])
    else:
        queue = deque()
    visited = set()
    while queue:
        x, y = queue.popleft()
        if (x, y) in visited:
            continue
        visited.add((x, y))
        if x < n - 1 and maze[x + 1][y]:
            queue.append((x + 1, y))
        if y < n - 1 and maze[x][y + 1]:
            queue.append((x, y + 1))
    return (n - 1, n - 1) in visited


def main():
    dstream = imap(int, stdin.read().split())
    for _ in xrange(next(dstream)):
        n = next(dstream)
        maze = [[next(dstream) for _ in xrange(n)] for _ in xrange(n)]
        if meet(maze, n):
            print 'POSSIBLE'
        else:
            print 'NOT POSSIBLE'


main()
