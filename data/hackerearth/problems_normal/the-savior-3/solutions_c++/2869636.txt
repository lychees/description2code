#include <iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    int n;
    while(t--){
        cin>>n;
        int a[n];
        for(int i=0;i<n;i++){
            cin>>a[i];
        }
        int ans=0;
         for(int i=0;i<n;i++){
             for(int j=i;j<n;j++){
                if(a[i]!=a[j] && (a[i]+a[j])%2==0){
                    ans++;
                }
             }
         }
         cout<<ans<<endl;
    }
    return 0;
}
