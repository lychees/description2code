#include <iostream>
#include<algorithm>
#include<cstdlib>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	long int a0,a1,mod,n,*c,f=0;
    	cin>>a0>>a1>>n>>mod;
    	int a[n];
    	a[0]=a0;
    	a[1]=a1;
    	for(long int i=2;i<n;i++)
    		a[i]=(a[i-1]+a[i-2])%mod;
    	c=(long int*)calloc(sizeof(long int),mod);
    	for(long int i=0;i<n;i++)
    	{
    		c[a[i]]++;
    	}
    	for(long int i=0;i<mod;i++)
    		f+=c[i]*c[i];
    	cout<<f<<endl;
    	t--;
    }
    return 0;
}
