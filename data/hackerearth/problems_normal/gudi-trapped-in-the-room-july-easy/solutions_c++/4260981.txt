
#include <bits/stdc++.h>

using namespace std;

const int maxn = 105;

#define fi first
#define se second
#define pi pair<int, int>

int dx[3] = {1, 1, 0};
int dy[3] = {0, 1, 1};

int t, a, h;
string ans, s;

string sera(string s) {
    string x = s;
    rotate(x.begin(), x.begin() + (x.size()-(h%x.size())), x.end());
    return x;
}

string xhaka(string s) {
    string x = "";
    for (int i = 0; i < s.size(); i++) {
        if (i&1==1) {
            x += to_string((long long int) ((s[i]-'0') + a) % 10);
        }
        else {
            x += s[i];
        }
    }
    return x;
}

void solve() {
    cin >> s;
    ans = s;
    scanf("%d%d", &a, &h);
    queue<string> Q;
    set<string> vis;
    Q.push(s);
    while (!Q.empty()) {
        string u = Q.front(); Q.pop();
        string a = sera(u);
        string b = xhaka(u);
        ans = min(ans, min(u, min(a, b)));
        if (vis.find(a) == vis.end()) {
            Q.push(a);
            vis.insert(a);
        }
        if (vis.find(b) == vis.end()) {
            Q.push(b);
            vis.insert(b);
        }
    }
    cout << ans << endl;
}

int main(void) {
    scanf("%d", &t);
    while (t--) {
        solve();
    }
}
