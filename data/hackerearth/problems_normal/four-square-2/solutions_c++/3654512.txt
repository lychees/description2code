#include<iostream>
#include<vector>
#include<algorithm>
#include<cmath>
using namespace std;
void sol()
{
    long long int n,l,r;
    cin>>n;
    long long int ans=0;
    for(l=1;l<=sqrt(n);l++)
    {
        if(n%l==0)
        {
            if(l%4!=0)
                ans+=l;
            if(l*l!=n)
            {
                if((n/l)%4!=0)
                    ans+=n/l;
            }
        }
    }
    ans*=8;
    cout<<ans<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--) sol();
    return 0;
}
