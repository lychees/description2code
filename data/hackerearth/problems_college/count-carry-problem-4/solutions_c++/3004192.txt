#include <iostream>
#include<cmath>


using namespace std;

int main()
{
	int t;
	cin>>t;

	long long int *a=new long long int[t];
	long long int *b=new long long int[t];
	for(int i=0;i<t;i++)
    {
        cin>>a[i]>>b[i];
    }
    int count=0;

    for(int i=0;i<t;i++)
	{
	    for(int j=0;(max(a[i],b[i]))>0;j++)
        {
            int x=a[i]%10;
            a[i]=a[i]/10;
            int y=b[i]%10;
            b[i]=b[i]/10;

            if((x+y)>=10)
            {
                count++;
            }
            x=0;
            y=0;

        }

        if(count==0)
        {
            cout<<"No carry operation"<<endl;
        }
        else if(count==1)
        {
            cout<<"1 carry operation"<<endl;
        }
        else
        {
            cout<<count<<" carry operations"<<endl;
        }
        count=0;
	}










return 0;
}
