#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;

int main()
{   long int n,q,i=0,l=0,r=0,pos;
    int j=0;
    char a[100001],x;
    scanf("%ld %ld",&n,&q);
    scanf("%s",a);
    for(i=0;i<q;i++)
    {
        scanf("%d ",&j);
        if(j==0)
        {   scanf("%ld %c",&pos,&x);
            a[pos-1]=x;
        }
        else if(j==1)
        {   scanf("%ld %ld",&l,&r);
            l--;
            r--;
            bool flag=0;
            while(l<=r)
            {
                if(a[l++]!=a[r--])
                {
                    flag=1;
                    break;
                }
            }
            if(flag==0)
                cout<<"Yes\n";
            else
                cout<<"No\n";
        }
    }
    return 0;
}
