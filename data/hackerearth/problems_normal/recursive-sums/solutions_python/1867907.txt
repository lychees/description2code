t = int(input())
for i in range(t):
	m = int(input())
	num = []
	for j in range(m):
		num.append(list(map(int, raw_input().split())))
	sum1 = 0
	for i in range(m):
		sum1 += num[i][0]*num[i][1]
	while sum1 >= 10:
		sum2 = 0
		sum1 = list(map(int,list(str(sum1))))
		for i in range(len(sum1)):
			sum2 += sum1[i]
		sum1 = sum2
	print(sum1)
