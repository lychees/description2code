#include <bits/stdc++.h>
using namespace std;
long long int a[1001][1001];
int main()
{
    long long int r,c,key;
    cin>>r>>c;
    for(int i=1;i<=r;i++)
    {
        for(int j=1;j<=c;j++)
        {
            cin>>a[i][j];
        }
    }
    long int q,p; cin>>q; //while(q--)
    for(p=0;p<q;p++)
    {
        cin>>key;
        long int i=1,j=c,flag=0;
        while(i<=r and j>=1)
        {
            if(a[i][j]==key)
            {
                flag=1;
                break;
            }
            else if(a[i][j]>key)
            {
                j=j-1;
            }
            else if(a[i][j]<key)
            {
                i=i+1;
            }
        }
        if(flag==1) cout<<i-1<<" "<<j-1<<endl;
        else cout<<-1<<" "<<-1<<endl;
    }
    return 0;
}
