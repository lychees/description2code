#include<bits/stdc++.h>

typedef vector<int> vi;

#define debug(x) cerr<<#x<<" "<<x<<endl;
#define f first
#define s second
#define mp make_pair
#define pb push_back
#define MOD 10000007
ll dp[105][105]={0};
int main()
{
    int n,k;
    scanf("%d %d",&n,&k);
    for(int i=0;i<=k;i++)
        dp[0][i]=0;
    for(int i=0;i<=n;i++)
        dp[i][0]=1;
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=k;j++)
            {if(i==j)
            dp[i][j]=1;
           else if(j>i)
            dp[i][j]=0;
           else
            dp[i][j]=dp[i-1][j]+dp[i-1][j-1];
             dp[i][j]=dp[i][j]%MOD;
            }

    }
    if(dp[n-1][k-1])
    printf("%lld\n",dp[n-1][k-1]);
    else
        printf("-1\n");
    return 0;
}
