#include <iostream>
#include<string>
using namespace std;
int main()
{
    int t,i;
    cin>>t;
    string s;
    while(t--)
    {
    	cin>>s;
    	int ans=-1;
    	int freq[26];
    	for(i=0;i<26;i++)
    	freq[i]=0;
    	for(i=0;i<s.size();i++)
    	freq[s[i]-'a']++;
    	for(i=0;i<26;i++)
    	ans=max(ans,freq[i]);
    	cout<<ans<<endl;
    	
    }
    return 0;
}
