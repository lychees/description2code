#include <iostream>
using namespace std;

int main()
{
    int t,n,k,i;
    cin >> t;
    while(t--)
    {
    	cin >> n >> k;
    	int a[n];
    	int cnt=0;
    	for(i = 0;i < n;i++)
    	{
    		cin >> a[i];
    		if(a[i] <= 0)
    			cnt++;
    	}
    	
    	if(cnt < k)
    		cout << "YES" << endl;
    	else
    		cout << "NO" << endl;
    }
    return 0;
}
