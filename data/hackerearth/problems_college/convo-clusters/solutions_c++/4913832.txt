#include <iostream>
#include <sstream>
#include <vector>
#include <cctype>
using namespace std;

bool check(char x) {
	if(x=='a'||
	   x=='e'||
       x=='i'|| 
	   x=='o'||
	   x=='u') 
		 return true;
	else	
		return false;
}

int main() {
	
	string str;
	bool con,vow, close;
	int ctr1,ctr2;
	vector<string> vec;
	while(getline(cin,str)) {

		stringstream ss(str);
		string token;
		int ctr;
		con = vow = close = false;
		ctr = 0;
		while(ss >> token) 
			vec.push_back(token);
	
		for(int i=0; i<vec.size() && close==false;i++) {
			for(int j=0; j<vec[i].length()-1; j++) {
					if(check(vec[i][j])==true && check(vec[i][j+1])==true) 
						vow = true;

					if(check(vec[i][j])==false ) {
						if(isalpha(vec[i][j]))
							ctr++;
					}
					if(ctr>=2)
						con = true;
				
					if(con==true && vow==true) {
						close = true;
						break;
					}
			}	
		}
		
		close!=true?
			cout << "NO" << endl:
			cout << "YES" << endl;
			
		con = vow = close = false;
		vec.clear();
	}
	return 0;
}
