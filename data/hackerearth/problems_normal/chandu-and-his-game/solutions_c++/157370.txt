#include <iostream>
using namespace std;
int main()
{
   int  N, K;
   long sum = 0;
   cin>>N;
   cin>>K;
   int xy[2][K] ;
   for(int i = 0 ; i < K ; i++)
        cin>>xy[0][i];
   for(int i = 0 ; i < K ; i++)
        cin>>xy[1][i];
    for(int i = 0 ; i < K ; i++)
        {
            if((xy[0][i] == 1 || xy[0][i] == N )&&(xy[1][i] == 1 || xy[1][i] == N ))
            {
                continue;
            }
            else if((xy[0][i] == 1 || xy[0][i] == N ))
            {
                if((N - xy[1][i]) >= (xy[1][i] - 1))
                  sum += (xy[1][i] - 1);
                  else
                   sum += (N - xy[1][i]);
            }
            else if((xy[1][i] == 1 || xy[1][i] == N ))
            {
                if((N - xy[0][i]) >= (xy[0][i] - 1))
                  sum += (xy[0][i] - 1);
                  else
                   sum += (N - xy[0][i]);
            }
            else
            {
                 if((N - xy[0][i]) >= (xy[0][i] - 1))
                  sum += (xy[0][i] - 1);
                  else
                   sum += (N - xy[0][i]);

                   if((N - xy[1][i]) >= (xy[1][i] - 1))
                  sum += (xy[1][i] - 1);
                  else
                   sum += (N - xy[1][i]);

            }
        }
            cout<<sum<<endl;

   return 0;
}
