#include <bits/stdc++.h>

using namespace std;

#define boost ios::sync_with_stdio(true); cin.tie(0); cout.tie(0);
#define pii pair<int,int>
#define vpii vector<pair<int,int> >
#define vvpii vector<vector<pair<int,int> > >
#define vi vector<int>
#define vii vector<vector<int> >
#define pb push_back 
#define mp make_pair 


int main()
{
	boost;
	int t;
	t=1;
	while(t--)
	{
		int n;
		cin>>n;
		vi a(n);
		int i;
		for(i=0;i<n;i++)
			cin>>a[i];
		int ans=0;
		sort(a.begin(),a.end());
		for(i=0;i<n;i++)
		{
			if(i+2>=n)
				break;
			else
				ans=max(ans,a[i+2]-a[i]);
		}
		cout<<ans<<endl;
	}

}
