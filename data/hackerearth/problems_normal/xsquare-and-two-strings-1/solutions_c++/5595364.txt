#include <iostream>
using namespace std;
int main()
{
    char str1[100000],str2[100000];
    int t,j;
    int alp1[26],alp2[26];
    cin>>t;
    while(t--)
    {
    	for(j=0;j<26;j++)
    	{
    		alp1[j]=0;
    		alp2[j]=0;
    	}
    	cin>>str1;
    	cin>>str2;
    	for(int i=0;str1[i]!='\0';i++)
    	alp1[str1[i]-97]++;
    	for(int i=0;str2[i]!='\0';i++)
    	alp2[str2[i]-97]++;
    	for(j=0;j<26;j++)
    	if(alp1[j]>0&&alp2[j]>0)break;
    	if(j==26)cout<<"No"<<endl;
    	else cout<<"Yes"<<endl;
    }
    return 0;
}
