#include<iostream>
#include<cstdio>
#include<cstring>
int main()
{
          int t;
          scanf("%d",&t);
          while(t--)
          {
          char str[10000];
          scanf("%s",str);
          int n=strlen(str);
          int a,b;
          int i,c=0;
          for(i=0;i<n-1;i++)
          {
                    a=0;
                    b=0;
                    a=str[i+1]-str[i];
                    b=str[n-i-2]-str[n-i-1];
                    if(a==b || a==b*(-1))
                              c=1;
                    else
                    {
                              c=0;
                              break;
                    }
          }
          if(c==1)
                    printf("Funny\n");
          else
                    printf("Not Funny\n");
          }
          return 0;
}
