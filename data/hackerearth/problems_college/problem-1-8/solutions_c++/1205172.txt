#include <iostream>
using namespace std;

int main()
{
   string s;
   int n,i=0,min=-1,min_l;
   cin>>s>>n;
   string s1[n];
   int k = s.length();
   while(n--)
   {
   	cin>>s1[i];
   	int x = s1[i].find(s);
   	if(x==-1||x!=0)
   	{i++;continue;}
   	else
   	{
   		string z = s1[i];
   		z.erase(0,k);
   		int y = z.length();
   		if(min==-1)
   		{
   			min_l=y;
   			min=i;
   		}
   		else
   		{
   			if(y<min_l)
   			min=i;
   		}
   		i++;}
   }
   (min==-1)?cout<<s:cout<<s1[min];
    return 0;
}
