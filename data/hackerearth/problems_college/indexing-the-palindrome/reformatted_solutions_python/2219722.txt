for _ in xrange(input()):
    A = raw_input()
    l, i = len(A), 0
    j, flag = l-1, 1
    while i < l/2:
        if A[i] != A[j]:
            if A[i+1] == A[j]:
                print i
                flag = 0
                break
            if A[i] == A[j-1]:
                print j
                flag = 0
                break
        i += 1
        j -= 1
    if flag:
        print "-1"
