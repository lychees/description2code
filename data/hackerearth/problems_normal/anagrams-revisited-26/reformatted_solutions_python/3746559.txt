n = input()
string_counts = dict()
result, maxi = "", 0
for _ in xrange(n):
    string = raw_input().strip()
    string = ''.join(sorted(string))
    if string in string_counts:
        string_counts[string] += 1
    else:
        string_counts[string] = 1
    if string_counts[string] > maxi:
        maxi = string_counts[string]
        result = string
print maxi
