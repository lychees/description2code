#include <bits/stdc++.h>
#define MOD 1000000000
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n,i,x;
    int sum=0;
    set<int>s;
    unordered_set<int>unset,unset1;
    unordered_set<int>:: iterator it1;
    cin>>t;
    t=t-1;
    int flag=1;
    cin>>n;
    for(i=0;i<n;i++){
    	cin>>x;
    	s.insert(x);
    	unset.insert(x);
    }
    while(t--){
    	cin>>n;
    	if(!flag)
    	unset.clear();
    	else
    	unset1.clear();
    	for(i=0;i<n;i++){
    		cin>>x;
    		s.insert(x);
    		if(flag){
    		if(unset.find(x)!=unset.end())
    		unset1.insert(x);
    	}
    	    else
    	    {
    	    	if(unset1.find(x)!=unset1.end())
    	    	unset.insert(x);
    	    }
    	}
    	flag=!flag;
    }
        if(flag){
        	for(it1=unset.begin();it1!=unset.end();it1++){
        	   sum+=*it1;
        	   sum%=MOD;
        	   //cout<<*it1<<endl;
        	}
        }
    	else
    	{
    	   for(it1=unset1.begin();it1!=unset.end();it1++){
    	   	sum+=*it1;
    	   	sum%=MOD;
    	   }	
    	}
    	cout<<(-sum);
    return 0;
}
