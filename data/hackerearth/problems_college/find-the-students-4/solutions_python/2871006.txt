list = raw_input().split();

list = [int(i) for i in list]

while True:
	minVal = min(x for x in list if x > 0)

	#print("Min Val %d" % (minVal))
	count = 0
	for index in range(0, len(list)):
		if list[index] > 0:
			newVal = list[index] - minVal
			#print("New Value %d" % (newVal))
			list[index] = newVal
			count = count + 1

	#print(list)
	print count

	finish = True
	for x in list:
		if x > 0:
			finish = False

	if finish:
		break
