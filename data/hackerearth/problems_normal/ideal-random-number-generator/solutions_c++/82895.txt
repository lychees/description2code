#define ll long long int

ll  gcd(ll a,ll b){
    if(b==0){
       return a; 
    }
    return gcd(b,a%b);
}
int main(){
    ll a,b,c;
    ll p,q;
    cin >> a >> b >> c;
    if(b>a){
       swap(b,a); 
    }
    if( a>=c and b>=c){
        p=(c*c);
	q=(2*a*b);
    }else if( a>=c and b<c ){
        p=(b*b)+(2*b*(c-b));
	q=2*a*b;
    }else{
        if(a+b>c){
	   p=(2*a*b)-( ( (a+b)-c ) * ( (a+b)-c ) );
	   q=2*a*b;
	}else{
	   p=1;
	   q=1;
	}
    }
    ll g=gcd(p,q);
    cout << p/g << "/" << q/g << endl;
    return 0;
}
