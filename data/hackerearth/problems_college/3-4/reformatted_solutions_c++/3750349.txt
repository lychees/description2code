#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;

int main() {
  int t, i;
  scanf("%d", &t);
  while (t--) {
    int n, k;
    cin >> n >> k;
    int ar1[k + 1];
    int arr[1001] = {0};
    for (i = 1; i <= k; i++) {
      cin >> ar1[i];
      arr[ar1[i]]++;
    }
    for (i = 1; i <= n; i++) {
      if (arr[i] % 2 != 0)
        cout << i << endl;
    }
  }
  return 0;
}
