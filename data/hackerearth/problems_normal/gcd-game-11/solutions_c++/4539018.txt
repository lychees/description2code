#include<bits/stdc++.h>
using namespace std;
int dp[1001][1001][2];
void pre()
{
    int val;
    for(int i=1;i<=1000;i++)
    {
        dp[i][1][0] =0;dp[1][i][0] = 1;
        dp[i][1][1] =0;dp[1][i][1] = 1;
    }

    for(int i=2;i<=1000;i++)
    {
        for(int j=2;j<=1000;j++)
        {
               if(i==j)
                {
                    dp[i][j][0] =0; dp[i][j][1] = 1;
                }
                else
                {
                    val = __gcd(i,j);
                    if(val>1)
                    {
                        if(!dp[i][j/val][1])
                            dp[i][j][0] = dp[i][j/val][1];
                        else if(!dp[i][j-1][1])
                            dp[i][j][0] = dp[i][j-1][1];
                        else
                            dp[i][j][0] = 1;
                        dp[i][j][1] = dp[i-1][j][0] || dp[i/val][j][0];
                    }
                    else
                    {
                        dp[i][j][0] = dp[i][j-1][1];
                        dp[i][j][1] = dp[i-1][j][0];
                    }

                }
         }
     }
}

int main()
{
    pre();
    int t,a,b;
    scanf("%d",&t);
    while(t--)
    {
        cin>>a>>b;
        if(a==b && a==1)
        {
            cout<<"Draw"<<endl;
            continue;
        }

        if(!dp[a][b][0])
            cout<<"Arjit"<<endl;
        else
            cout<<"Chandu Don"<<endl;
     }

     return 0;
}
