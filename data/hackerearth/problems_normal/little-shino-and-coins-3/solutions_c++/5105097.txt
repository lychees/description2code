
#include <bits/stdc++.h>
using namespace std;
typedef long long int lli;

int main()
{
	int k; cin>>k;
	string s; cin>>s; 
	int n = s.length();
	s = " " + s;
	
	map< char , int > l , r;
	int ans = 0;
	int li=1 , ri=1;
	for ( int i=1 ; i<=n ; i++ )
	{
		l[ s[i] ]++;
		r[ s[i] ]++;
		
		while ( l.size() > k )
		{
			char c = s[li];
			l[c]--;
			if ( l[c] == 0 ) l.erase( c );
			li++;
		}
		
		while ( r.size() >= k )
		{
			char c = s[ri];
			r[c]--;
			if ( r[c] == 0 ) r.erase( c );
			ri++;
		}
		
		if ( l.size() < k ) continue;
		ans += ri - li;
	}
	
	cout<<ans<<"\n";
	return 0;
}
