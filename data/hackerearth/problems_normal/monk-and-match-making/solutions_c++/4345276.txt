#include <iostream>
#include <cstdio>
#include <string>

typedef long long ll;

const ll p1 = (int)1e6 + 3;
const ll p2 = 59;

const int NMAX = (int)1e5 + 5;

ll pre[NMAX];

ll pow(ll p, int s, ll mod)
{
	ll res = 1;
	ll it = p;
	while(s){
		if(s & 1)
			res = (res * it) % mod;
		it = (it * it) % mod;
		s >>=1;
	}
	return res;
}

int main()
{
	int q,l1,r1,l2,r2;
	std::string s;
	std::cin >> s;
	
	pre[0] = s[0];
	ll it = p2;
	for(int i = 1; i < (int)s.size(); i++) {
		pre[i] = (pre[i-1] + s[i] * it) % p1;
		it = (it * p2) % p1;
	}

	scanf("%d", &q);
	while(q--){
		scanf("%d %d %d %d", &l1, &r1, &l2, &r2);
		if(r1 - l1 == r2 - l2) {
			l1--; r1--; l2--; r2--;
			ll res1 = l1 == 0 ? pre[r1] : pre[r1] - pre[l1-1];
			ll res2 = l2 == 0 ? pre[r2] : pre[r2] - pre[l2-1];
			res1 = l1 < l2 ? (res1 * pow(p2, l2 - l1, p1)) % p1 : res1;
			res1 = res1 < 0 ? res1 + p1 : res1;
			res2 = l2 < l1 ? (res2 * pow(p2, l1 - l2, p1)) % p1 : res2;
			res2 = res2 < 0 ? res2 + p1 : res2;
			if(res1 == res2)
				printf("Yes\n");
			else
				printf("No\n");

		}
		else
			printf("No\n");
	}

}
