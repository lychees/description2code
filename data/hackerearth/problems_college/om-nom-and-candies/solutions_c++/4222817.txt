#include <bits/stdc++.h>
using namespace std;

int prime[100005];
vector<int>pr[100005];

void pre()
{
	prime[0]=prime[1]=1;
	int i,j;
	for(i=2;i<=100000;i++)
	{
		if(!prime[i])
		{
			for(j=2*i;j<=100000;j+=i)
			{
				prime[j]=1;
			}
		}
	}
}



void SieveOfEratosthenes()
{
    int n=100000;
    for (int p=2; p<=n; p++)
    {
    	pr[p].push_back(1);
        for (int i=p*2; i<=n; i += p)
            pr[i].push_back(p);
    }
}

int main()
{
	pre();
	SieveOfEratosthenes();
    int n;
    cin>>n;
    int a[n+1];
    for(int i=1;i<=n;i++)
    {
    	cin>>a[i];
    }
    int tot[n+1];
    tot[1]=a[1];
    cout<<tot[1]<<" ";
    for(int i=2;i<=n;i++)
    {
    	if(prime[i]==0)
    		tot[i]=a[i]+a[1];
    	else
    	{
    		vector <int>::iterator it=pr[i].begin();
    		int m=-1;
    		for(;it!=pr[i].end();++it)
    		{
    			if(m<tot[*it])
    				m=tot[*it];
    		}
    		tot[i]=a[i]+m;
    	}
    	cout<<tot[i]<<" ";
    }
    return 0;
}
