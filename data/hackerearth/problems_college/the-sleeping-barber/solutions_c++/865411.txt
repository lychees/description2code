#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while (t > 0){
		long long g,m;
		cin >> g >> m;
		long long l[g];
		vector<long long> myvector;
		for (int i = 0; i < g; i++){
			long long ans;
			cin >> ans;
			myvector.push_back(ans);
		}
		sort(myvector.begin(), myvector.end());
		int i = 0;
		while (m > 0 && i < g){
			m -= myvector[i];
			i += 1;
		}
		if (m == 0)
			cout << i << endl;
		else
			cout << i-1 << endl; 
		t--;
	}
    return 0;
}
