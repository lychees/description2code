#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int i,n,count=0;
    	cin>>n;
    	long long int arr[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>arr[i];
    	}
    	for(i=0;i<n-1;i++)
    	{
    		if(arr[i]>=arr[i+1])
    		count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
