#include <iostream>
using namespace std;

int main()
{
    int cases;
    cin>>cases;
    int i = 0;
    int possible = 0;
    for(i;i<cases;i++) {
        int a, b, c;
        cin>>a>>b>>c;
        int d = b*b - 4*a*c;
        if (d>=0) {
            possible++;
        }
    }
    cout<<possible;
    return 0;
}
