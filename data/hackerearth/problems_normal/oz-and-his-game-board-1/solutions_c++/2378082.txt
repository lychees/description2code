#include <stdio.h>
#include <algorithm> 

using namespace std;

void OzGameBoard()
{
	int t;
	scanf("%d", &t);

	while (t--)
	{
		int N, M, A, B, C;
		scanf("%d %d %d %d %d", &N, &M, &A, &B, &C);

		int t = min(N, M);
		long long int x = (long long int)N*A + (long long int)M*B;
		long long int y = (long long int)t*C*2 + (long long int)(N - t)*A + (long long int)(M - t)*B;

		printf("%lld\n", max(x, y));
	}

}


int main()
{
	OzGameBoard();
}
