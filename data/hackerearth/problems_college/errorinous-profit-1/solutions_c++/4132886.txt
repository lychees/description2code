#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		double a,b;
		scanf("%lf %lf",&a,&b);
		double c,d;
		c=a+(b/100.00)*a;
		long long e,f=0;
		e=c;
		while(e!=0)
		{
			int dig=e%10;
			f=f*10+dig;
			e=e/10;
		}
		if(a>=f)
		{
			double x=a-f;
			x=(x*100)/a;
			//cout<<x<<endl;
			x=(x/100.00)*(c);
			c+=x;
			printf("%lld\n",(long long)c);
		}
		else
		{
			double x=f-a;
			x=(x*100)/a;
			x=(x/100.00)*c;
			c-=x;
			printf("%lld\n",(long long)c);
		}
	}
}
