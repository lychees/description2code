using namespace std;

const long long MOD = (1e9) + 7;

int n;
int a[1024][1024];
long long ans,dp[1024][1024];
int sz;
pair <int,int> z[1<<20];

bool cmp(pair <int, int> x, pair <int, int> y) {
	return a[x.first][x.second]>a[y.first][y.second];
}

int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	//freopen("test.txt","r",stdin);
	int i,j,x,p,t;

	cin>>n;
	for(i=1;i<=n;i++) {
		for(j=1;j<=n;j++) {
			cin>>a[i][j];
			z[++sz]=make_pair(i,j);
		}
	}

	sort(z+1,z+1+sz,cmp);

	for(x=1;x<=sz;x++) {
		i=z[x].first;
		j=z[x].second;
		dp[i][j]=1;
		p=i+1;t=j;
		if(p>=1 && p<=n && t>=1 && t<=n) {
			if(a[p][t]>a[i][j]) dp[i][j]+=dp[p][t];
		}
		p=i-1;t=j;
		if(p>=1 && p<=n && t>=1 && t<=n) {
			if(a[p][t]>a[i][j]) dp[i][j]+=dp[p][t];
		}
		p=i;t=j-1;
		if(p>=1 && p<=n && t>=1 && t<=n) {
			if(a[p][t]>a[i][j]) dp[i][j]+=dp[p][t];
		}
		p=i;t=j+1;
		if(p>=1 && p<=n && t>=1 && t<=n) {
			if(a[p][t]>a[i][j]) dp[i][j]+=dp[p][t];
		}
		dp[i][j]%=MOD;
		ans+=dp[i][j];
		ans%=MOD;
	}

	cout<<ans<<endl;

	RETURN:
  //eprintf("Time: %d ms\n", (int)(clock()*1000.0/CLOCKS_PER_SEC));
  return 0;
}
