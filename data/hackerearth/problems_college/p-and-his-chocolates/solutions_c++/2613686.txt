#include <bits/stdc++.h>
#include <stdio.h>    
#include <string.h>
using namespace std;
#define mod 1000000007

#define llu long long unsigned
#define lld long long
#define ld long

lld A[1100][1100];
int main(){
    lld n,i,j,m,k,T,t,p,r;
    for(i=0;i<1100;i++) A[i][0]=A[i][i]=1;
    for(i=1;i<1100;i++){
        for(j=1;j<i;j++){
            A[i][j]=(A[i-1][j]+A[i-1][j-1])%mod;
        }
    }
	cin>>T;
	while(T--){
	    cin>>k;
	    cin>>j;
	    lld ways,sum;
	    ways=1;
	    sum=j;
	    for(i=2;i<=k;i++){
	        cin>>j;
	        sum+=1;
	        ways=(ways*A[j-1+sum-1][sum-1])%mod;
	        sum+=j-1;
	    }
	    cout<<ways<<endl;
	}
    return 0;
}
