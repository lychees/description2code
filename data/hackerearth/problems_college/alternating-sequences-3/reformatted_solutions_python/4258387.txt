noInSeries = input()
series = map(int, raw_input().split())
dp = [0] * 5001
res = 0
for i in range(noInSeries):
    curr = 0
    for j in range(i):
        if abs(
            series[j]) < abs(
            series[i]) and (
            (series[j] > 0 and series[i] < 0) or (
                series[j] < 0 and series[i] > 0)):
            curr = max(curr, dp[j])
    dp[i] = curr + 1
    res = max(res, dp[i])
print res
