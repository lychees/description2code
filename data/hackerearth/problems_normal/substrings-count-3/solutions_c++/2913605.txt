#include<bits/stdc++.h>
#define MOD 1000000007
#define pb push_back 
using namespace std;
int main()
{
int n,i,j,k,idx,idx1,ans,l,q,r;
vector<int> :: iterator sz;
map<string,vector<int> > m;
map<string,vector<int> >:: iterator it;
scanf("%d",&n);
char a[15];
//printf("%d\n",n);
for(i=0;i<n;i++)
{
	scanf("%s",a);
	l=strlen(a);
//	printf("%s",a);
	for(j=0;j<l;j++)
	{
		string temp="";
		for(k=j;k<l;k++)
		{
			temp.pb(a[k]);
			m[temp].pb(i);
			
		}
		
		temp.clear();
		
	}
	
	
}

for(it=m.begin();it!=m.end();it++)
{
	sz=unique((it->second).begin(),(it->second).end());
	(it->second).resize(distance((it->second).begin(),sz));
	
	
}

scanf("%d",&q);
//getchar();
while(q--)
{
	scanf("%d %d %s",&l,&r,a);
	getchar();
	l--;r--;
	idx=upper_bound(m[a].begin(),m[a].end(),r)-m[a].begin();
	idx1=lower_bound(m[a].begin(),m[a].end(),l)-m[a].begin();
	
	ans=idx-idx1;
	
	printf("%d\n",ans);
	
}


}
