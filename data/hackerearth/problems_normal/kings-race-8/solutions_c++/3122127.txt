#include <bits/stdc++.h>
using namespace std;
#define N 1000005

long long int min( long long int a, long long int b )
{
	return ( a < b )? a : b;
}
int main()
{
    long long int i, j, n, k, mins, minval;
    int t;
    for( cin >> t ; t ; t--)
    {
    	scanf( "%lld %lld", &n, &k );
    	vector < pair <long long int,long long int> > a(n);
    	vector <long long int> d(k);
    	for( i = 0 ; i < n ; i++ )
    	{
         	scanf( "%lld", &a[i].first );
         	a[i].second = i;
    	}
    	for( i = 0 ; i < k ; i++ )
    	{
         	scanf( "%lld", &d[i] );
    	}
    	sort( a.begin(), a.end() );
    	j = 0;
    	mins = 0;
    	minval = -1;
    	for( i = 0 ; i < k && j < n ;  )
    	{
    		if( a[j].first >= d[i])
    		{
				mins =  a[j].second;
				minval = a[j].first;
				i++;
			//	cout << mins << " m1\n";
    		//	cout << minval << " mv1\n";
    		}
    		else
				j++;
    	}
    //	cout << mins << " m\n";
    //	cout << minval << " mv\n";
    	for( i = 0 ; i < n ; i++ )
    	{
    		if( a[i].second < mins && minval <= a[i].first )
    		{
				mins =  a[i].second;
				//cout << mins << " m2\n";
    		}
    	}
    	cout << mins << "\n";
    //	cout << minval << "\n";
    }
    return 0;
}
