using namespace std;
const int N=100005;
priority_queue<int> pq;
int n;
int a;
long long ans;
int main(){
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&a);
        pq.push(a);
        ans=-1;
        if(pq.size()>2){
            int m[3];
            ans=1;
            for(int j=0;j<3;j++){
                m[j]=pq.top();
                pq.pop();
                ans*=m[j];
            }
            for(int j=0;j<3;j++){
                pq.push(m[j]); }
            
        }
        printf("%lld\n",ans);
    }
    return 0;
}
