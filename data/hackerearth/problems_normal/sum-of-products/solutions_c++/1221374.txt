#include<iostream>
#include<cmath>
#define ll long long
using namespace std;
ll p=1000000007;
int main()
{
 int T;
 cin>>T;
 while(T--)
 	{
 	ll N;
 	cin>>N;
 	ll s=0;
 	int i;
 	//i*j such that N/i=j
 	//i<=j
 	for(i=1;i*i<=N;i++)//sqrt is not taken because of precision errors
 		
 		{
 		 
 		 s=(s+N/i*i)%p;
 		}
 		
 	for(int j=N/i;j>=1;j--)//i>j such that N/i=j.initial j is i-1
 		{
 		 ll hi=N/j;
 		 ll lo=N/(j+1)+1;
 		 //cout<<"lo="<<lo<<" hi="<<hi<<endl;
 		 s=(s+(j*(lo+hi)*(hi-lo+1)/2)%p)%p;
 		}	
 	cout<<s<<endl;
 	}	

}
