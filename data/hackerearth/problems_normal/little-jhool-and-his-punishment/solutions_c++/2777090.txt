#include <iostream>
using namespace std;
int check(int b,int g,int mark,int bg,int bb,char c)
{
	static int c1=0;
	if(bb>bg&&b==0&&g==0)
	{
		c1=1;
		return(c1);
	}
	/*cout<<b<<" "<<g<<" "<<mark<<" "<<bg<<" "<<bb<<" " <<c<<" "<<c1<<"\n";*/
	if(c=='0')
	{
		c1=0;
		check(b-1,g,1-mark,bg,bb,'b');
		check(b,g-1,1-mark,bg,bb,'g');
	}
	if(mark==0)
	{
		if(c=='g')
		{
			if(g>=1)
			{
				check(b,g-1,1-mark,bg,bb+1,'g');
			}
			else
			{
				if(b>=1)
				check(b-1,g,1-mark,bg+1,bb,'b');
			}
		}
		if(c=='b')
		{
			if(b>=1)
			{
				check(b-1,g,1-mark,bg,bb+1,'b');
			}
			else
			{
				if(g>=1)
				check(b,g-1,1-mark,bg+1,bb,'g');
			}
		}
	}
	if(mark==1)
	{
		if(c=='g')
		{
			if(b>=1)
			{
				check(b-1,g,1-mark,bg+1,bb,'b');
			}
			else
			{
				if(g>=1)
				check(b,g-1,1-mark,bg,bb+1,'g');
			}
		}
		if(c=='b')
		{
			if(g>=1)
			{
				check(b,g-1,1-mark,bg+1,bb,'g');
			}
			else
			{
				if(b>=1)
				check(b-1,g,1-mark,bg,bb+1,'b');
			}
		}
		}
		return(c1);
	}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,a,b;
    	cin>>n>>a>>b;
    	
    	if(check(a,b,0,0,0,'0'))
    	cout<<"Little Jhool wins!"<<"\n";
    	else
    	cout<<"The teacher wins!"<<"\n";
    }
}
