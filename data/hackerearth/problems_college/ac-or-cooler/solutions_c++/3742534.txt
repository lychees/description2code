#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,m;
    cin>>n>>m;
    int a[n],c[n],t[2*n];
    for(int i=0;i<n;i++){
    	cin>>a[i];
    	t[i]=a[i];
    }
    for(int i=0;i<n;i++){
    	cin>>c[i];
    	t[i+n]=c[i];
    }
    sort(t,t+2*n);
    int count=0,i=0;
    while(m>=t[i]&&count<n){
    	m-=t[i];
    	i++;
    	count++;
    }
    cout<<count;
    return 0;
}
