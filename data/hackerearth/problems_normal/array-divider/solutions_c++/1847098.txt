int a[MAXN];

#ifdef LOCAL

  scanf("%d%d", &n, &k);
  forn(i, n) scanf("%d", &a[i]);

  const ll INF = 1e18;
  forn(i, n + 1) forn(j, k + 1) dp[i][j] = INF;

  dp[0][0] = 0;
  forn(i, n) forn(j, k) {
    if  (dp[i][j] >= INF) {
      continue;
    }
    int s = 0;
    for (int z = i; z < n; ++z) {
      s += a[z];
      dp[z + 1][j + 1] = min(dp[z + 1][j + 1], dp[i][j] + s * 1ll * s);  
    }
  } 

  ll S = accumulate(a, a + n, 0);
  cout << (k * 1ll * dp[n][k] - S * S) << endl;
  return 0;
}
