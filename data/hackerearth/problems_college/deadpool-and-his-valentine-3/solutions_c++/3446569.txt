#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,k;
    cin >> k >> n;
    int a[k][n];
    int sw[n];

    for (int j = 1;j <= n;j++)
            sw[j] = -1;

    for (int i = 0;i < k;i++)
    {
        int num;
        cin >> num;
        for (int j = 0;j < num;j++)
        {
            int l;
            cin >> l;
            sw[l] = 1;
        }
    }
    int fa = 0;
    for (int i = 1;i <= n;i++)
    {
        if (sw[i] == -1)
        {
            fa = 1;
            break;
        }
    }

    if (fa == 0)
        cout << "YES" << endl;
    else
        cout << "NO" << endl;

    return 0;
}
