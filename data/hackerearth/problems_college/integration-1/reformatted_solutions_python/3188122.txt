def out(n, p, u, r, s, x):
    for k in range(n):
        if p+r <= u:
            p += r
        elif p-s >= 0:
            p -= s
        else:
            break
    print 'Case #%d:' % (x+1), p


t = input()
for i in range(t):
    n, p, u, r, s = map(int, raw_input().split())
    out(n, p, u, r, s, i)
