#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n,k,i,j,l,r,mid,ans,count1,count2;
    cin>>n>>k;
    int pos[n+1];
    for(i=1;i<=n;i++)
        cin>>pos[i];
    sort(pos+1,pos+n+1);
    ans=pos[n];
    l=1;
    r=pos[n];
    while(l<=r)
    {
        count1=0;
        count2=0;
        mid=(l+r)/2;
        i=1;
        j=1;
        while(1)
        {
            while(pos[j]-pos[i]<=(2*mid)&&j<=n)
                j++;
            i=j;
            count1++;
            if(i==(n+1))
                break;
        }
        //cout<<count1<<endl;
        i=1;
        j=1;
        while(1)
        {
            while(pos[j]-pos[i]<=(2*mid-2)&&j<=n)
                j++;
            i=j;
            count2++;
            if(i==(n+1))
                break;
        }
        //cout<<count2<<endl;
        if(count1<=k&&count2>k)
        {
            ans=mid;
            break;
        }
        else if(count1>k)
            l=mid+1;
        else if(count1<=k&&count2<=k)
            r=mid-1;
    }
    cout<<ans<<endl;
}
