	#include <iostream>
   #include<bits/stdc++.h>
    using namespace std;
    int main(){
    	int t;
    	cin>>t;
    	while(t--){
    		int ka;
    		cin>>ka;
    		string s;
    		set <char> k;
    		vector <char> sa;
    		while(ka--){
    			cin>>s;
    			char res=s[0];
    			k.insert(res);
    			sa.push_back(res);
    		}
    		if(k.size()==sa.size()) cout<<"YES"<<endl;
    		else cout<<"NO"<<endl;
    	}
    	return 0;
    }
