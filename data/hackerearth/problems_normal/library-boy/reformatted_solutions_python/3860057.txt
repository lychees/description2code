n = input('')
i = 0
li = {}
while i < n:
    str1 = raw_input('')
    str1 = list(str1)
    if str1[0] not in li:
        li[str1[0]] = 1
    else:
        li[str1[0]] = li[str1[0]]+1
    i = i+1
c = 0
for i in li.itervalues():
    if i > 10:
        c = c+(i/10)
        if i % 10 != 0:
            c = c+1
    else:
        c = c+1
print c
