#include<iostream>
#include<stdio.h>
using namespace std;


int main()
{
    //freopen("nk.txt","r",stdin);
    int q;
    cin>>q;

    while(q--)
    {
        long long unsigned int n,t;
        cin>>n>>t;

        for(int i=0;i<t;i++)
            {

                if(n%2==0)
                    n-=n/2;
                else
                    n-=(n+1)/2;

                n-=n/4;

            }

        cout<<n<<endl;

    }
}
