#include<stdio.h>
int main()
{
	int T;
	for(scanf("%d",&T);T>0;T--)
	{
		int a[1000],b[1000],i, n;
		scanf("%d", &n);
		for(i=0;i<n;i++)
		{
			scanf("%d", &a[i]);
			b[i]=0;
		}
		i=0;
		b[0]=1;
		while(true)
		{
			if(a[i]==i)
			{
				printf("Lose\n");
				break;
			}
			if(a[i]!=-1)
			{
				i=a[i];
				if(b[i]==0)
					b[i]=1;
				else if(b[i]==1)
				{
					printf("Lose\n");
					break;
				}
			}
			else
			{
				printf("Win\n");
				break;
			}
		}
	}
	return 0;
}
