int stairs(int k)
	{
		if(k == 1 || k == 2 )
			return k;
		else
		 return stairs(k-1) + stairs(k-2);
	}
	


using namespace std;
int main()
{
	int t;
	cin >> t;
	int s[t], res;
	for (int i = 0; i < t; ++i)
	{
		cin >> s[i];
	}

	for (int i = 0; i < t; ++i)
	{
			cout << stairs(s[i]) << endl;
	}

}
