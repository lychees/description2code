#include <iostream>
using namespace std;
long long a[100005];
int main()
{
	int t;
	scanf("%d",&t);
	while(t--){
		int n,x;
		scanf("%d %d",&n,&x);
		for(int i=0;i<n;i++){
			scanf("%lld",&a[i]);
		}
		long long ans=0;
		for(int i=1;i<n;i++){
			if(a[i]<=a[i-1]){
				ans+=(a[i-1]-a[i])/x+1;
				a[i]+=((a[i-1]-a[i])/x+1)*x;
			//	printf("%lld\n",a[i]);
			//	ans+=(a[i-1]-a[i])/x+1;
			//	printf("%lld ans",(a[i-1]-a[i])/x+1);
				
		}
		}
		printf("%lld\n",ans);
	}
    return 0;
}
