#include <bits/stdc++.h>
using namespace std;

int and_set(vector<int>& values)
{
    int result = values[0];
    for(int q=1; q!=values.size(); q++)
        result = result & values[q];
    return result;
}

int main()
{
    int t;
    cin>>t;
    while(t!=0){
        t--;
        int n;
        cin>>n;
        int array[n];
        int max_el;
        cin>>max_el;
        array[0] = max_el;
        for(int q=1; q!=n; q++){
            cin>>array[q];
            if(array[q]>max_el)
                max_el = array[q];
        }
        string message = "NO";
        map<int, vector<int> > table;
        int curr_num = 1;
        while(curr_num<=max_el){
            for(int q=0; q!=n; q++)
                if((array[q] & curr_num)==curr_num)
                    table[curr_num].push_back(array[q]);
            curr_num<<=1;
        }
        for(map<int, vector<int> >::iterator iter=table.begin(); iter!=table.end(); iter++){
            if(and_set(iter->second)==iter->first){
                message = "YES";
                break;
            }
        }
        cout<<message<<endl;
    }
}
