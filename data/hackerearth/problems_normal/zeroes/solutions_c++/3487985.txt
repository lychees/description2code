#include<iostream>
#include<map>
#include<climits>
using namespace std;
#define MAX 1000001
#define ll long long
#define foreach(a,b) for(map<int,int>::iterator b=a.begin();b!=a.end();b++)
int cnt[MAX];
int sieve[MAX];
void precomp(){
	for(int i=2;i*i<=MAX;i++){
		if(sieve[i]==0){
			for(int j=i*i;j<=MAX;j+=i)sieve[j]=i;
		}
	}
	for(int i=2;i<=MAX;i++){
		if(sieve[i]==0)sieve[i]=i;
	}
}
int main()
{
	precomp();
	int n;
	cin>>n;
	int ans=0;
	for(int i=0;i<n;i++){
		int tmp;
		cin>>tmp;
		while(tmp!=1){
			cnt[sieve[tmp]]++;
			tmp/=sieve[tmp];
		}
	}
	int q,x;
	cin>>q;
	while(q--){
		cin>>x;
		map<int,int> mp;
		while(x!=1){
			mp[sieve[x]]++;
			x/=sieve[x];
		}
		ans=INT_MAX;
		foreach(mp,b){
			ans=min(ans,cnt[b->first]/b->second);
		}
		cout<<ans<<'\n';
	}
    return 0;
}
