#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long int k,diff;
    cin>>k;
    long long int arr[k];
    int i;
    for(i=0;i<k;i++)
    {
    	cin>>arr[i];
    }
    sort(arr,arr+k);
    int sum=2;
    int max=0;
    for(i=k-1;i>=0;i--)
    {
        diff=arr[i]+k-i;
    	if(diff>max)
           max=diff;
    }
    max++;
    cout<<max<<endl;
    return 0;
}
