#include <iostream>
using namespace std;
#include<bits/stdc++.h>
long long arr[1000][1000];
long long int solve(int k,int n)
{
	if(k>n)
	{
		return 0;
	}
	else if(k==n)
	{
		return 1;
	}
	else if(arr[k][n])return arr[k][n];
	
	arr[k][n]=solve(k+1,n)+solve(k,n-k);
	return arr[k][n];
}
int main()
{
	long long int t,a;
//    cout << "Hello World!" << endl;
   cin>>t;
   while(t--)
   {
   	cin>>a;
   cout<<solve(2,a)<<endl;
   }
   return 0;
}
