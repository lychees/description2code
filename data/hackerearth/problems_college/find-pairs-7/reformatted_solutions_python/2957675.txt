def main():
    # step1:Input of test cases
    testCase = input()
# step2:Input of each test cases
    for i in range(0, testCase):
        arraysize = input()
        arr = raw_input()
        arr = arr.split()

        arr = map(int, arr)
        count = 0
        for m in range(0, arraysize):
            for n in range(m, arraysize):
                if int(arr[m]) == int(arr[n]):
                    count += 1

        print count


if __name__ == "__main__":
    main()
