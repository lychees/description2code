#include <iostream>
#include <algorithm>
using namespace std;

int main()
{long  t,n,i,m,x,j;
long long sum;
    scanf("%ld",&t);
    while (t--){
    	sum=0;
    	scanf("%ld",&n);
    	long long a[n];
    	for(i=0;i<n;i++){
    		scanf("%ld",&a[i]);
    	}
    sort(a,a+n);
    m=a[(n)/2];
    for (i=0;i<n;i++){
    	sum+=abs(a[i]-m);
    	//printf("%lld",a[i]);
    }
    printf("%lld\n",sum);
    }
    return 0;
}
