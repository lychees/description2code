#include <bits/stdc++.h>
using namespace std;

int main()
{
	string str;
	char key[15],username[15],pwd[15],profile[15],role[15];
	cin>>str;
	int i,j,k,l,a,b,c,d,e,f;
	for(i=0;i<str.length();i++)
		{	//cout<<str.substr(i,3)<<endl<<i<<endl;
			//cout<<str.substr(2,3)<<endl<<i<<endl;
			if((str.substr(i,3)).compare("key")==0)
			{for(j=i+4,a=0;str[j]!='\0';++j,++a)
				{key[a]=str[j];}
			}	
			if((str.substr(i,8)).compare("username")==0)
			{
				for(j=i+9,b=0;(str.substr(j,2).compare("&p")!=0);++j,++b)
				{username[b]=str[j];}
			}
			if((str.substr(i,3)).compare("pwd")==0)
			{for(j=i+4,c=0;(str.substr(j,2).compare("&p")!=0);++j,++c)
				{pwd[c]=str[j];}
			}
			if((str.substr(i,7)).compare("profile")==0)
			{for(j=i+8,d=0;(str.substr(j,2).compare("&r")!=0);++j,++d)
				{profile[d]=str[j];}
			}						
			if((str.substr(i,4)).compare("role")==0)
			{for(j=i+5,e=0;(str.substr(j,2).compare("&k")!=0);++j,++e)
				{role[e]=str[j];}
			}
			l++;			
		}
		cout<<"username: ";
		for(l=0;l<b;l++)
		cout<<username[l];
		cout<<endl;
		cout<<"pwd: ";
		for(l=0;l<c;l++)
		cout<<pwd[l];
		cout<<endl;
		cout<<"profile: ";
		for(l=0;l<d;l++)
		cout<<profile[l];
		cout<<endl;
		cout<<"role: ";
		for(l=0;l<e;l++)
		cout<<role[l];
		cout<<endl;
		cout<<"key: ";
		for(l=0;l<a;l++)
		cout<<key[l];
		cout<<endl;
		return 0;
}
