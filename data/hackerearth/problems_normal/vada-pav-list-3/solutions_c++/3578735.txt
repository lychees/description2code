#include <iostream>
#include <set>
#include <algorithm>
#include <string>

using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	long long int n;
	set<string>names;
	cin >> n;
	string name;
	while (n--)
	{
		
		cin >> name;
		names.insert(name);
	}
	cout << names.size() << endl;
	for (auto it = names.begin(); it != names.end(); it++)
		printf("%s\n",(*it).c_str());
}
