#include <bits/stdc++.h>
using namespace std;

int t,n,p[1000005];

int main(){
	ios::sync_with_stdio(0); cin.tie(0);
	for (int i=2;i<=1000000;i++) p[i]=i;
	for (int i=2;i<=1000000;i++){
		if (p[i]==i){
			for (long long j=1ll*i*i;j<=1000000;j+=i)
			p[j]=i;
		}
	}
	cin >> t;
	while (t--){
		cin >> n; int ok=0;
		for (int i=((n-1)>>1);i;i--){
			if (p[i]==i&&p[n-i]==n-i){
				cout << i << " " << n-i << "\n";
				ok=1; break;
			}
		}
		if (!ok) cout << "-1 -1\n";
	}
}
