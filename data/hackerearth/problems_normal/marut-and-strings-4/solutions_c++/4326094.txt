#include <iostream>

using namespace std;


int main(){

int t;
cin>>t;
if(t<1 || t>10){
        cout<<"Invalid Test"<<endl;
        return 0;
}
while(t--){

int i;
int count_lower = 0 ,count_upper =0 ;
string a;
cin>>a;
int n = a.size();
for(i=0;i<n;i++){
    if(a[i]<='z' && a[i]>='a') count_lower = count_lower + 1;
    else if(a[i]<='Z' && a[i]>='A') count_upper = count_upper + 1;
}
if((count_upper ==0 && count_lower ==0) || (a.size()>100 || a.size()<1)) cout<<"Invalid Input"<<endl;
else{
        if(count_upper < count_lower) cout<<count_upper<<endl;
        else cout<<count_lower<<endl;

}

}

return 0;
}
