#include<stdio.h>
#include<string>
#include<iostream>
using namespace std;
int arr[1000];
int main()
{
    int t;
    cin>>t;

    while(t--)
 {

      string s;
      cin>>s;

    int j=0,count;
    for(int i=0;s[i]!='\0';)
    {
        count=0;
        while((s[i]!='\0')&&(s[i]=='_'))
        {
            count++;
            i++;
        }
        if(count!=0)
        {
        arr[j]=count;
        j++;
        }
        if(s[i]=='\0')
          break;

        i++;



    }

    if(j==1&&arr[j-1]==s.length())
     {

         printf("Bob\n");
         continue;
     }

   // for(int i=0;i<=j;i++)
     //printf("%d\n",arr[i]);
    int xori=0;
    for(int i=0;i<j;i++)
    {

       xori=xori^arr[i];
    }

    if(xori==0)
    printf("Bob\n");

    else
    printf("Alice\n");

  }

}
