#include <iostream>
using namespace std;

int main()
{
    int l;
    int i, j, n;
       
    cin>>l;
    cin>>n;
       
    int arr[n][2];
       
    for(i=0;i<n;i++)
    {
    	for(j=0;j<2;j++)
       	{
       		cin>>arr[i][j];
       	}
    }
	
  	for(i=0;i<n;i++)
    {
    	for(j=0;j<2;j++)
    	{
       		if(arr[i][j] == l && arr[i][j+1] == l )
       		{
       			cout<<"ACCEPTED \n";
       			break;
       		}
       		else if(arr[i][j] == arr[i][j+1] && (arr[i][j] > l && arr[i][j+1] > l) )
       		{
       			cout<<"ACCEPTED \n";
       			break;
       		}
       		else if(arr[i][j] >= l && arr[i][j+1] >= l)
       		{
       			cout<<"CROP IT \n";
       			break;
       		}
       		else if(arr[i][j] <= l || arr[i][j+1] <= l)
       		{
       			cout<<"UPLOAD ANOTHER \n";
       			break;
       		}
       	}
    }
       	
}
