#include<bits/stdc++.h>
using namespace std;
int main()
{
	int test;
	cin>>test;
	while(test--)
	{
		long long n;
		cin>>n;
		if(n &1)
		cout<<"0"<<endl;
		else
		{
			long long cnt=0;
			for(int i=2;i<=sqrt(n);i++)
			{
				if(n % i ==0)
				{
					if((i%2==0) &&((n/i) % 2==0))
					{
						if(i != n/i)
						cnt+=2;
						else
						cnt+=1;
					}
					else if(((i%2==0) &&((n/i)&1))||((i&1) &&((n/i)%2==0)))
					cnt++;
				}
			}
			cout<<cnt+1<<endl;
		}
	}
}
