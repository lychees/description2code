#include<iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
        string s;
        cin>>s;
        int len = s.length();
        int i,j;
        int ans = -1;
        for(i=0,j=len-1;i<=j;i++,j--)
            if(s[i]!=s[j])
            {
                    if(s[i] == s[j-1]){
                    	ans = j;
                    	break;
                    }
                    else if(s[i+1] == s[j]){
                        ans = i;
                        break;
                    }
            }
        cout<<ans<<endl;
    }
}
