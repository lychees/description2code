#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t,count,n,i;
    cin>>t;
    while(t--)
    {
        count=0;
        cin>>n;
        int A[n];
        for(i=0;i<n;i++)
        {
            cin>>A[i];
        }
        sort(A,A+n);
        for(i=0;i<n-1;i++)
        {
            if(A[i]==0)
            continue;
            for(int j=i+1;j<n;j++)
            {
                if(A[j]==0)
                continue;
                if(A[i]==A[j])
                {
                    count++;
                    A[j]=0;
                }
            }
        }
        cout<<n-count<<endl;
    }
}
