#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,s,d,x,y,flag;
    cin>>n;
    while(n--)
    {
    	flag=0;
    	cin>>s>>d;
    	if((s+d)%2==1)
    	flag=1;
    	else
    	{
    		x=(s+d)/2;
    		y=(s-x);
    		if(y<0)
    		flag=1;
    	}
    	if(flag)
    	cout<<"impossible\n";
    	else
    	cout<<x<<" "<<y<<"\n";
    	
    }
    return 0;
}
