#include <bits/stdc++.h>
using namespace std;
#define ll long long
bool prime[1000005];
int ans[1000005];
void primeNumbers()
{
	memset(prime,true,sizeof(prime));
	prime[0] = false;
	prime[1] = false;
	ans[0] = 0;
	ans[1] = 0;
	for( int i = 2 ;i <= 1000 ; i++)
	{
		if ( prime[i] ==  true)
		{
			for ( int j = 2*i ; j<1000001; j += i)
			prime[j] = false;
		}
	}
	int cnt = 0;
	for ( int i = 2; i < 1000001; i ++ )
	{
		
		if ( prime[i] == true)
		cnt ++;
		if(prime[cnt] == true)
		ans[i] = ans[i-1] + 1;
		else
		ans[i] =  ans[i-1];
	}
}
int main()
{
    //cout << "Hello World!" << endl;
    primeNumbers();
   
    int t;
    cin >> t;
    while(t -- )
    {
    	int x,y;
    	cin >> x >> y;
    	int p = ans[y] - ans[x-1];
    	cout << p  << endl;
    }
    return 0;
}
