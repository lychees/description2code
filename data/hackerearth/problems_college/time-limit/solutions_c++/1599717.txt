#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
  long long int n,m,i,p;
   cin>>n>>m;
   vector<long long int> v1,v2;
   vector<long long int>:: iterator itr1,itr2,itr3,itr4;
   for(i=0;i<n;i++)
   {
   	cin>>p;
   	v1.push_back(p);
   }
   for(i=0;i<m;i++)
   {
   	cin>>p;
   	v2.push_back(p);
   }
   itr1=v1.begin();
   itr2=v1.end();
   itr3=v2.begin();
   itr4=v2.end();
   sort(itr1,itr2,greater<int>());
   sort(itr3,itr4);
   if(v1[0]<v2[0])
   {
   	if(v1[0]>2*v1[n-1])
      cout<<v1[0]<<endl;
      else
      cout<<2*v1[n-1]<<endl;
   }
   else
   cout<<"-1"<<endl;
   
}
