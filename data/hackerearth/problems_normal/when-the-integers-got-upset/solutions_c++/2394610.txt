template <typename K, typename V>
V GetWithDef(const std::unordered_map <K,V> & m, const K & key, const V & defval ) {
    auto it = m.find(key);
    return (it == m.end()) ? defval : it->second;
}

template <typename K, typename V>
void SetDef(std::unordered_map <K,V> & m, const K & key, const V & defval ) {
    auto it = m.find(key);
    if (it == m.end()) m[key] = defval;
}

#define DEBUG 1
#define MAXN 12

int N, t;
int A[MAXN + 123];
int P[MAXN + 123];

vector<int> by_ham[13];
int dp[1 << MAXN][MAXN][MAXN] = {};

int solve() {
    char checked[1 << MAXN] = {};
    queue<int> tocheck;

    int ans = 99999999;
    int left = (1 << N) - 1;
    FORN(i, 1 << MAXN) FORN(j, N) FORN(k, N) dp[i][j][k] = ans;
    FORN(j, N) FORN(k, N) dp[left][j][k] = 0;
    FORN(pp, N) FORN(p, N) {
        if (p == pp) continue;
        int mask = left ^ (1 << p) ^ (1 << pp);
        tocheck.push(mask);
        dp[mask][pp][p] = 0;
    }
    while (tocheck.size()) {
        int mask = tocheck.front(); tocheck.pop();
        checked[mask] = 1;
        EPR("check 0x%x\n", mask);
        int ham = __builtin_popcountl(mask);
        FORN(bit, N) {
            if (mask & (1 << bit)) {
                FORN(pp, N) {
                    if (mask & (1 << pp)) continue;
                FORN(p, N) {
                    if (mask & (1 << p)) continue;
                    if (p == pp) continue;
                    int new_mask = mask ^ (1 << bit);

                    int add = (A[bit] ^ A[p] ^ A[pp]) * P[N - ham];
                    if (new_mask) {
                        if (!checked[new_mask]) {
                            tocheck.push(new_mask);
                            EPR("push 0x%x -> 0x%x\n", mask, new_mask);
                            checked[new_mask] = 1;
                        }
                        UPDATE_MIN(dp[new_mask][p][bit], dp[mask][pp][p] + add);
                    }
                    else {
                        UPDATE_MIN(ans, dp[mask][pp][p] + add);
                    }
                }}
            }
        }
    }
    return ans;
}
int main() {
    ios_base::sync_with_stdio(0);
    FORN(i, 1 << 12) {
        int ham = __builtin_popcountl(i);
        by_ham[ham].push_back(i);
    }
    cin >> t;
    FORN(_it, t) {
        cin >> N;
        FORN(i, N) cin >> A[i];
        FORN(i, N) cin >> P[i];
        P[0] = P[1] = 0;
        memset(dp, 0, sizeof(dp));
        cout << solve() << endl;
    }
    return 0;
}
