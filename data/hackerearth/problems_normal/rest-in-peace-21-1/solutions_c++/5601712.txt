#include <iostream>
#include <string>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    
    int t,n,g,m;
    string ch;
    cin>>t;
    for(int j=1;j<=t;j++)
    {
    	g=0;
    	cin>>ch;
    	
    	n=stoi(ch);
    	
    	if(n%21==0)
    	{
    		g=1;
    		//cout<<"Hello"<<endl;
    		cout<<"The streak is broken!"<<endl;
    	}
    	else
    	{
    		for(int i=0;i<ch.size()-1;i++)
    		{
    			if(ch[i]=='2' && ch[i+1]=='1')
    			{
    				g=1;
    				cout<<"The streak is broken!"<<endl;
    				break;
    			}
    		}
    	}
    	
    	if(g!=1)
    	{
    		cout<<"The streak lives still in our heart!"<<endl;
    	}
    	
    }
    return 0;
}
