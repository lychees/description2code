#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,x,r,i,j,smx,n;
    int a[200];
   cin >> t;

   while (t--)
   {
       memset(a,0,sizeof(a));
       smx = 1;

       cin >> n;
       a[199] = 1;

       for (i = 2; i <= n; i++) {
            r = 0;
            for (j = 199; j >= 0; j--) {
                x = a[j] * i + r;
                a[j] = x % 10;
                r = x / 10;
            }
       }
       i = 0;
       while (a[i] == 0) {
        i++;
       }

       while (i < 200) {
        printf("%d",a[i]);
        i++;
       }
       printf("\n");
   }

    return 0;
}
