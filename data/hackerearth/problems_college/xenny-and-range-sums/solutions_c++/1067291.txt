#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    //cout << "Hello World!" << endl;
    int N,Q;
    cin>>N>>Q;
    vector<ll> sum(1000001);
    vector<ll> arr(1000001);
 //   ll sum[N+1];
    //ll arr[N+1];
    ll x;
    sum[0]=0;
    for(int i=1;i<=N;i+=1)
    {
    	scanf("%lld",&x);
    	arr[i]=x;    	
    	sum[i]=sum[i-1]+arr[i];
    }
    while(Q--)
    {
    	int A,B;
    	scanf("%d %d",&A,&B);
    	//cout<<A<<" "<<B<<endl;
    	ll ans1=sum[B]-sum[A-1];
    	ll ans2=arr[B]-(sum[B-1]-sum[A-1]);
    	printf("%lld %lld\n",ans1,ans2);
    }
    
    return 0;
}
