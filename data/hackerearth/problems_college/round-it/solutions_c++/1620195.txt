#include <iostream>
#include <cstdio>
#include <vector>

using namespace std;

typedef long long ll;


int main() {
    int t;
    scanf ("%d", &t);
    for (int tests = 0; tests < t; tests++) {
        int n, b;
        scanf ("%d%d", &n, &b);
        int p = n / b;
        int q = p + 1;
        int val1 = p * b;
        int val2 = q * b;
        if (val2 - n <= n - val1)
            printf ("%d\n", val2);
        else
            printf ("%d\n", val1);
    }
    return 0;
}
