#include <bits/stdc++.h>
#include<algorithm>
#define ll long long
using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    int t,p;
    cin>>t;
    while(t--)
    {
        int c=0;
        cin>>p;
        int n=p;
        int a[1000];
        for(int i=0;i<p;++i)
            cin>>a[i];
        sort(a,a+p);
        //for(int i=0;i<n;++i)
          //  cout<<a[i]<<endl;
        int i,j;
        for(i=0;i<n-1;++i)
        {
            for(j=i+1;j<n;++j)
            {
                if(a[i]!=a[j])
                if((a[i]+a[j])%2==0)
                c++;
            }
        }
        cout<<c<<endl;
    }
    return 0;
}
