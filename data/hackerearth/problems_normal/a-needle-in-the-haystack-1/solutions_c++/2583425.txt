#include <bits/stdc++.h>

using namespace std;

int main()
{
  int t,i,k,c;
  cin>>t;
  while(t--)
  {
  char p[1000],x[100000];
  int q[26]={0},y[26]={0};
  int count=0;
  
  cin>>p>>x;
  int pl=strlen(p);
  for(i=0;i<pl;i++)
  q[x[i]-'a']++;
  int xl=strlen(x);
  for(i=0;i<pl;i++)
  y[p[i]-'a']++;
  for(i=0;i<=xl-pl;i++,c=0)
  {
  for(k=0;k<26;k++)
  if(y[k]==q[k])
  c++;
  if(c==26)
  count++;
  q[x[i]-'a']--;
  q[x[i+pl]-'a']++;
  }
  if(count>0)
  cout<<"YES"<<endl;
  else
  cout<<"NO"<<endl;
  
  	
}

    return 0;
}
