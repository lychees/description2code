#include <iostream>
#include <stdio.h>
#include <vector>
#define DIVI 1000000007

using namespace std;

typedef long long int Lint;

int main() {
	
	vector< vector<Lint> > A( 1005, vector<Lint> ( 1005 ) );
	Lint i, j, t, a, b, c, d, v1, v2;

	for( i = 0;i < 1005;i++ ) {
		A[ i ][ 0 ] = 1;
		A[ 0 ][ i ] = 1;
	}

	for( i = 1;i < 1005;i++ )
		for( j = 1;j < 1005;j++ )
			A[ i ][ j ] = ( ( ( ( A[ i - 1 ][ j ] % DIVI ) + ( A[ i ][ j - 1 ] % DIVI ) ) % DIVI )+ ( A[ i - 1 ][ j - 1 ] % DIVI ) ) % DIVI;

	scanf("%lld", &t );

	while( t-- ) {

		scanf("%lld %lld %lld %lld", &a, &b, &c, &d );
		v1 = c - a;
		v2 = d - b;

		printf("%lld\n", A[ v1 ][ v2 ] ); 
	} 
	return 0;
}
