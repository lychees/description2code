#include <iostream>
#define endl '\n'
using namespace std;

struct node
{
    int parent;
    node *childs[26];

    node()
    {
        parent = 0;
        for(int i = 0; i < 26; i++)
        {
            childs[i] = NULL;
        }
    }
};

node *root, *currNode;
int n, ans = 1;
string s;

void addWord(string word)
{
    if(word.size() == 0) return;

    if(currNode->childs[word[0] - 'a'] != NULL) currNode = currNode->childs[word[0] - 'a'];
    else
    {
        currNode->childs[word[0] - 'a'] = new node();
        currNode = currNode->childs[word[0] - 'a'];
        ans++;
    }

    addWord(&word[1]);
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    root = new node();

    cin>>n;
    for(int i = 0; i < n; i++)
    {
        cin>>s;

        currNode = root;
        addWord(s);
    }

    cout<<ans<<endl;
    return 0;
}
