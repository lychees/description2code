#include<bits/stdc++.h>
using namespace std;

const int N=1234567;
const int INF=1e9;

int a[N];

int main() {
        int n;
        scanf("%d", &n);
        for(int i=0;i<n;i++) {
            scanf("%d", &a[i]);
        }
        int ans=-INF, curr=-INF;
        for(int i=0;i<n;i++) {
            curr=max(a[i], curr+a[i]);
            ans=max(ans, curr);
        }
        printf("%d\n", ans);
}
