#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll t,i,j,k,n,m,l;
int main()
{
    cin>>n;
    priority_queue<int> pq,pr;
    ll maxn=0,maxp=0;pq.push(0);pr.push(0);
    while(n--)
    {
    	cin>>k;
    	if(k<0)
    	{
    		l=abs(k);
    		pr.push(l);
    	}
    	else
    	{
    		pq.push(k);
    	}
    }
    i=pr.top();
    pr.pop();
    maxn=i*pr.top();
    i=pq.top();
    pq.pop();
    maxp=i*pq.top();
    i=max(maxn,maxp);
    cout<<i<<endl;
}
