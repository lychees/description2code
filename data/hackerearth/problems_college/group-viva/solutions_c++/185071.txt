#include <iostream>
#include <cstdio>
using namespace std;

int gcd(int x,int y){
    if(y==0) return x;
    return gcd(y,x%y);
}

int main()
{
    int T;
    cin >> T;
    while(T--){
    int N,R;

    int ans=1;
    int i;
    int F=R-1;
    for(i=2;i*i<=F;i++){
        if((F%i)==0) {
            if((F/i)==i) ans++;
            else ans+=2;
        }
    }
    /*

    }
    return 0;
}
