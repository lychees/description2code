from sys import stdin
t = int(stdin.readline())
for j in range(0, t):
    a = stdin.readline()
    b = list(a)
    if (j != (t-1)):
        b.pop()
    for i in range(0, len(b)):
        c = ord(a[i])+i
        while (c > 122):
            d = c-122
            c = 96+d
        b[i] = chr(c)
    a = ''.join(b)
    print a
