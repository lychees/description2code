#include <iostream>
#include<stdio.h>
using namespace std;
int main() {
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int h,m,s,i,flag=0;
		char ch;
		scanf("%d%c%d%c%d",&h,&ch,&m,&ch,&s);
		//s+=1;
		while(1)
		{
			if(s>59){
				m+=1;
				s=0;
			}
			if(m>59){
				h+=1;
				m=0;
			}
			if(h>23)
			{
				h=0;
				s=0;
				m=0;
			}
			int hash[10]={0};
			int copyh=h,copym=m,copys=s;
			if(copyh==0) hash[0]+=2;
			else if(copyh<=9) hash[0]++;
			while(copyh>0)
			{
				hash[copyh%10]++;
				copyh=copyh/10;
			}
			if(copym==0) hash[0]+=2;
			else if(copym<=9) hash[0]++;
			while(copym>0)
			{
				hash[copym%10]++;
				copym=copym/10;
			}
			if(copys==0) hash[0]+=2;
			else if(copys<=9) hash[0]++;
			while(copys>0)
			{
				hash[copys%10]++;
				copys=copys/10;
			}
			flag=1;
			for(i=0;i<10;i++)
			{
				if(hash[i]>1)
					flag=0;
			}
			if(flag)
			{
				if(h<10)
					printf("0%d:",h);
				else
					printf("%d:",h);
				if(m<10)
					printf("0%d:",m);
				else
					printf("%d:",m);
				if(s<10)
					printf("0%d\n",s);
				else
					printf("%d\n",s);
				
				break;
			}
			//if(flag>120)
			//	break;
			s+=1;
		}
		//printf("%d %d %d\n",h,m,s);
	}
	return 0;
}
