#include <iostream>
using namespace std;

double price (int a, int b, int c);

int
main ()
{
  int tc = 0, N = 0, M = 0;
  double minp = 100,p=0;
  int a=0,b=0,c=0;
  int minpi = -1;
  cin >> tc;
  while (tc--)
    {
      cin >> N;
      cin >> M;
      for (int i = 0; i < N; i++)	//each gift
	{
	  minp = 100;
	  minpi = -1;
	  for (int j = 0; j < M; j++)	//each shop
	    {
	      cin >> a >> b >> c;
	      p = price (a, b, c);
	      if (minp - p > 0.000000001) //p < minp
		{
		  minp = p;
		  minpi = j + 1;
		}
	    }
	  /*if (i == N - 1)
	    cout << minpi;
	  else*/
	    cout << minpi << " ";
	}
      cout << endl;
    }
}

double price (int a, int b, int c)
{
  double p = 100;
  p = 100 - a;
  p = p - (p * b / 100);
  p = p - (p * c / 100);
  return p;
}
