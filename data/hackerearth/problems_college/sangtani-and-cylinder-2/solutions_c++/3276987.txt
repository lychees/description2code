#include <bits/stdc++.h>
using namespace std;

int main()
{
    double t,k,r,v,s,h,pi;
    
    cin >> t;
    pi = 2* acos(0.0);
   // cout <<setprecision(10)<< pi <<endl;
    while(t--)
    {
    	cin >> v;
    	
    	k = v/(2*pi);
    	
    	r = pow(k,1/3.);
    	
    	h = v / (pi * r*r);
    	
    	s= 2*pi*r*(r+h);
    	
    	
    	
    	cout <<fixed<<setprecision(9)<<s <<endl;
    }
}
