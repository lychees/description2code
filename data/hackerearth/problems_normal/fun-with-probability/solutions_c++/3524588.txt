#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef pair<int,int> pii;
#define ff first
#define ss second
double dp[101][501][101];
int vis[101][501][101];
int amr=1;
int n;
double solve(int idx,int e,int d){
    if(e<0||d<0)return 0 ;
    if(idx==n)return 1.0;
    if(d<=0)return 0;
    double &ret = dp[idx][e][d];
    if(vis[idx][e][d]==amr)return ret;
    vis[idx][e][d]=amr;
    ret = 1e-9;
    ret+=solve(idx+1,e-22,d-1)*(1.0/3.0)*0.5;
    ret+=solve(idx+1,e-25,d-1)*(1.0/3.0)*0.5;
    ret+=solve(idx+1,e-30,d-1)*(1.0/3.0)*0.5;
    ret+=solve(idx+1,e-2,d)*(1.0/3.0)*0.5;
    ret+=solve(idx+1,e-5,d)*(1.0/3.0)*0.5;
    ret+=solve(idx+1,e-10,d)*(1.0/3.0)*0.5;
    return ret;
}
int main(){
    int t;
    cin >> t;
    while(t--){
        int e,d;
        cin >> n >> e >> d;
        amr++;
        double x = solve(0,e,d);
        x*=100;
        int y = x;
        cout << y << "\n";
    }
    return 0;
}
