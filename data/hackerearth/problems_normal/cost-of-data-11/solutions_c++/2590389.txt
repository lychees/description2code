			#define ain(A, B, C) assert(IN(A, B, C))
			#define ull unsigned long long int
			#define ll long long int
			#define pii pair<int,int>
			#define MAXX 1000009
			#define fr(a,b,i) for(int i=a;i<b;i++)
			vector<int>G[MAXX];
			ll cnt=1;
			ll trie[MAXX][27];
			bool leaf[MAXX];
			void add(string &s)
			{
				int r=0;
				for(int i=0;i<s.length();i++)
				{
					int id=s[i]-'a';
					if(trie[r][id]==0)
					 trie[r][id]=++cnt;
					r=trie[r][id]; 
				}
				leaf[r]=true;
			}
			int main()
			{
				
		      int t;
		      cin>>t;
		      while(t--)
		      {
		      	string s;
		      	cin>>s;
		      	add(s);
		      }
		      cout<<cnt<<endl;
			   return 0;
			}
