#include<bits/stdc++.h>

using namespace std;

int a,b;
int v[500007],l[500007];
int main()
{
	int to,from,i,j,x;
	queue<int> Q;
	vector<vector<int> > G;
	vector<int> R;
	cin>>a>>b;
	for(i=0;i<a;i++)
	{
		v[i]=0;
		l[i]=-1;
		G.push_back(R);
	}
	for(i=0;i<b;i++)
	{
		cin>>to>>from;
		G[to].push_back(from);
		G[from].push_back(to);
	}
	
	v[0]=1;
	l[0]=0;
	Q.push(0);
	
	while(!Q.empty())
	{
		x=Q.front();
		Q.pop();
		
		for(i=0;i<G[x].size();i++)
		{
			if(v[G[x][i]]==0)
			{
				v[G[x][i]]=1;
				l[G[x][i]]=l[x]+1;
				Q.push(G[x][i]);
			}
		}
	}
	for(i=1;i<a;i++)
		cout<<l[i]<<"\n";
	return 0;
}
