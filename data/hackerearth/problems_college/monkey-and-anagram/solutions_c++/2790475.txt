#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s,r;
		cin>>s>>r;
		int z=r.length()-s.length();
		int a[26]={'\0'},b[26]={'\0'};
		int i=0,j=0;
		while(s[i]!='\0')
		{
			a[s[i]-97]++;
			i++;
		}
		while(r[j]!='\0')
		{
			b[r[j]-97]++;
			j++;
		}
		int count=0;
		for(int k=0;k<26;k++)
		{
			if(a[k]!=b[k])
				count++;
		}
		if(count>z)
			cout<<"-1"<<endl;
		else
			cout<<count<<endl;
		
	}
	return 0;
}
