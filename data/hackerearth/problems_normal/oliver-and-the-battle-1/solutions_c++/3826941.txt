#include<bits/stdc++.h>
using namespace std;
int n,m;
bool arr[1001][1001];
int conncomp;
int cnt;
void find(int x,int y)
{
	arr[x][y]=0;
	conncomp++;
	if(x+1<n && arr[x+1][y])
	find(x+1,y);
	if(y+1 <m &&arr[x][y+1])
	find(x,y+1);
	if(x-1>=0&&arr[x-1][y])
	find(x-1,y);
	if(y-1>=0&&arr[x][y-1])
	find(x,y-1);
	if(x+1<n&&y-1>=0 && arr[x+1][y-1])
	find(x+1,y-1);
	if(x+1<n&&y+1<m&&arr[x+1][y+1])
	find(x+1,y+1);
	if(x-1>=0&&y-1>=0&&arr[x-1][y-1])
	find(x-1,y-1);
	if(x-1>=0&&y+1<m&&arr[x-1][y+1])
	find(x-1,y+1);
	return;
}
int main()
{
	int tc,i,j,max;
	scanf("%d",&tc);
	while(tc--)
	{
		max=0;
		cnt=0;
		conncomp=0;
		scanf("%d%d",&n,&m);
		for(i=0;i<n;i++)
		for(j=0;j<m;j++)
		scanf("%d",&arr[i][j]);
		for(i=0;i<n;i++)
		for(j=0;j<m;j++)
		{
			conncomp=0;
			if(arr[i][j]==1)
			{
				cnt++;
				find(i,j);
				if(conncomp>max)
				max=conncomp;
			}
		}
		printf("%d %d\n",cnt,max);
	}
}
