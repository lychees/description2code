#include<stdio.h>
#include<stdlib.h>
void qsort(long int ll,long int ul,long int a[],long int n)
{     
    int i,j,p,temp,k;
	if(ll<ul)
	{
		i=ll;
		j=ul+1;
		p=ll;
		while(i<j)
		{
		do
		{
			i=i+1;
		}while(a[i]<=a[p]);
		do
		{
			j=j-1;
		}while(a[j]>a[p]);
		if(i<j)
		{
			temp=a[i];
			a[i]=a[j];
			a[j]=temp;
			
		}
	}
			temp=a[p];
			a[p]=a[j];
			a[j]=temp;
	qsort(ll,j-1,a,n);
	qsort(j+1,ul,a,n);
}
}
int main()
{
	long int n,a[1000],i,ll,ul,sum1=0,sum=0;
	scanf("%ld",&n);
	ll=0;ul=n-1;
	for(i=0;i<n;i++)
	{
		scanf("%ld",&a[i]);
	}
	qsort(ll,ul,a,n);
	for(i=0;i<n/2;i++)
	{
		sum=sum+(a[n-1-i]-a[i]);
		sum=sum+(a[n-2-i]-a[i]);
	}
	for(i=0;i<n/2;i++)
	{
	    sum1=sum1+(a[n-1-i]-a[i]);
		sum1=sum1+(a[n-1-i]-a[i+1]);
	}
	if(sum>=sum1)
	{
		printf("%ld\n",sum);
	}
	else
	{
		printf("%ld\n",sum1);
	}
}
