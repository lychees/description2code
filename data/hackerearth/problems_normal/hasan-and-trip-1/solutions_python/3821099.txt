import math
x=[0]*3000
y=[0]*3000
f=[0]*3000
d=[0]*3000
n=int(input())
for i in xrange(n):
	x[i],y[i],f[i]=map(int,raw_input().split())

for i in xrange(n):
	d[i]=-999999

d[0]=f[0]

for i in xrange(1,n):
	for j in xrange(i):
		dx=x[i]-x[j];
		dy=y[i]-y[j];	
		dist=math.sqrt((dx*dx)+(dy*dy));
		d[i]=max(d[i],(f[i]+d[j]-dist));
print "%.6f"%d[n-1]
