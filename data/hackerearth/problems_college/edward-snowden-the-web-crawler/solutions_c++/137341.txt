string foo(string& html, int pos) {
	string link; int n = html.length();
	while(pos < n && html[pos]!='"') link+=html[pos++];
	return link;
}

int main()
{
	string site; cin >> site;
	if(site.find("http")!=string::npos) {
		int sn = site.length();
		site = site.substr(7, sn-7);
	}
	string html, t;
	while(!cin.eof()) { getline(cin, t); html+=t; }
	string link; int i=0;
	while(i < html.length()) {
		//cout << i << endl;
		int pos = html.find("<a href=", i);
		if(pos==string::npos) break;
		pos += 9;
		string link = foo(html, pos);
		if(link.find(site)!=string::npos) cout << link << endl;
		i = pos;
	}
    return 0;
}
