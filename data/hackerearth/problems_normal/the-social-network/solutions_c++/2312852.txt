#include <iostream>
#include <stdio.h>
#include <map>
using namespace std;

int main()
{
    int N;
    scanf("%d", &N);
    map<int, int> m;
    for(int i=1; i<=N; i++) {
    	int f, x;
    	scanf("%d %d", &f, &x);
    	m[f] = 1;
    	for(int j=1; j<=x; j++) {
    		scanf("%d", &f);
    		m[f] = 1;
    	}
    }
    
    printf("%d", m.size() - N);
    return 0;
}
