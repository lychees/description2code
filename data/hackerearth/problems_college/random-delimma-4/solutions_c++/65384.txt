#include<iostream>
#include<cstdlib>
#include<iomanip>
using namespace std;

int main()
{
    long double k,n,T,j;
    long double prob;
    cin>>T;
    for(long double i=0;i<T;i++)
    {
    	prob=1;
    	cin>>k;
    	cin>>n;
    	if(k<n)
    	{
   			for(j=1;j<k;j++)
   			{
   				prob=prob*((long double)(n-j))/n;
   			}
   			prob=1-prob;
   			cout<<fixed<<setprecision(9)<<prob<<endl;
    	}
    	else
    		cout<<fixed<<setprecision(9)<<prob<<endl;
    }
    	return 0;
}
