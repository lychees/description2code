#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int x,y;
    	cin>>x>>y;
    	float a[n+1];
    	float s=0;
    	for(int i=0;i<n;i++) {cin>>a[i];s+=a[i];}
    	a[n]=sqrt(x*x + y*y);
    	s+=a[n];
    	sort(a,a+n+1);
    	s-=a[n];
    	if(s<a[n]) cout<<"Not able\n";
    	else cout<<"Able\n";
    }
    return 0;
}
