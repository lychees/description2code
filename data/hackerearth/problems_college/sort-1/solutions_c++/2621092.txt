#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t,n,i;
    long *a;
    cin>>t;
    while(t--){
    	cin>>n;
    	a = (long *)malloc(sizeof(long)*n);
    	for(i=0;i<n;i++){
    		cin>>a[i];
    	}
    	sort(a,a+n);
    	for(i=0;i<n;i++){
    		cout<<a[i]<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
