#include <iostream>
using namespace std;
class st
{
	public:
		string str;
		void read();
		void op();
		st(){str="\0";}
};
int main()
{
    st x;
    x.read();
    x.op();
}
void st::read()
{
	cin>>str;
}
void st::op()
{
	int n,a,i,j,ev;
	n=str.length();
	for(i=0;i<n;i++)
	{
	 ev=0;
	for(j=i;j<n;j++)
	{
		a=(int)str[j];
		if(a%2==0){ev++;}
	}
	cout<<ev<<" ";
	}
}
