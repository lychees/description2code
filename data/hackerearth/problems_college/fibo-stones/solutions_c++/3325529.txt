long long int series(int long long  *fib,long long int n)
{
	long long int i,sum=2;
	fib[1]=1;
	fib[2]=1;
	
	for(i=3;i<=n;i++)
	{
		fib[i]=fib[i-1]+fib[i-2];
	//	cout<<"fib "<<fib[i];
		sum+=fib[i];
	//	cout<<"sum "<<sum;
	}
	return sum;
	
}
