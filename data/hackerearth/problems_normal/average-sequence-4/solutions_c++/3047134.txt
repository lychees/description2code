#include <iostream>
using namespace std;

int main()
{
    int N,i;
    long long int current,previous,value;
    
    cin>>N;
    cin>>current;
    cout<<current<<" ";
    previous=current;
    
    for(i=2;i<=N;i++){
    	cin>>current;
    	value=current*i-previous*(i-1);
    	previous=current;
    	cout<<value<<" ";
    }
    return 0;
}
