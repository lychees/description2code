#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int arr[n+9];
    for(int i=0;i<n;i++)
    	cin>>arr[i];
    sort(arr,arr+n);
    int ans=0;
    int i=0;
    while(i<n)
    {
    	ans++;
    	int temp=arr[i]+4;
    	while(arr[i]<=temp)
    		i++;
    }
    cout<<ans;
    return 0;
}
