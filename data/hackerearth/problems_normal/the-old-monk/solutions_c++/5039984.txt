#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int t;
	long n;
	
	for (cin >> t; t--;)
	{
		cin >> n;
		
		long long a[n], b[n];
		for (int i = 0; i < n; ++i) cin >> a[i];
		for (int i = 0; i < n; ++i) cin >> b[i];
		
		reverse (b, b + n);
		
		long long res = 0;
		for (int i = 0; i < n; ++i) 
		{
			long long temp = b + n - lower_bound(b, b + n, a[i]) - i - 1;
			res = (temp > res ? temp : res);
		}
		
		cout << res << endl;
	}
	
	return 0;
}
