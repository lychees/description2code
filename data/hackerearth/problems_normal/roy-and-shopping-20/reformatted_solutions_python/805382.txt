n = int(raw_input())

primes = [0]*1000002
primes[0] = 1
primes[1] = 1
idx = 2
while idx < 1000002:
    if primes[idx] == 0:
        test = idx
        while test <= 1000001:
            if primes[test] == 0:
                primes[test] = idx
            test += idx
    idx += 1
for i in range(0, n):
    total = int(raw_input())
    ans = total
    print(total-primes[total])
