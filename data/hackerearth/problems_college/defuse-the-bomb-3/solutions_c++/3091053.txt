#include <iostream>
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;


int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll b;
		string a;
		cin>>a>>b;
		int len=0;
		ll temp=b;
		while(temp!=0)
		{
			temp/=10;
			len++;
		}
		for(ll i=0;i<a.size();i++)
		{
			ll num=0;
			int flag=0;
			for(ll j=i;j<i+len && i+len<=a.size();j++)
			{
				num=num*10+a[j]-'0';
				flag=1;
			}
			if(flag==1) cout<<num*b<<endl;
		}
		
	}
	
    return 0;
}
