#define MAXN 10010
#define LOGMAXN 15
using namespace std;

int N, PARENT[MAXN],LEVEL[MAXN],ANCESTOR[MAXN][LOGMAXN];
vector<int> v[MAXN];

void Preprocess(){
      int i, j;
   
  //we initialize every element in ANCESTOR with -1
      for (i = 0; i < N; i++)
          for (j = 0; 1 << j < N; j++)
              ANCESTOR[i][j] = -1;
   
  //the first ancestor of every node i is PARENT[i]
      for (i = 0; i < N; i++)
          ANCESTOR[i][0] = PARENT[i];
   
  //bottom up dynamic programing
      for (j = 1; 1 << j < N; j++)
         for (i = 0; i < N; i++)
             if (ANCESTOR[i][j - 1] != -1)
                 ANCESTOR[i][j] = ANCESTOR[ANCESTOR[i][j - 1]][j - 1];
}

int LCA(int p, int q){
      int tmp, log, i;
   
  //if p is situated on a higher level than q then we swap them
      if (LEVEL[p] < LEVEL[q])
          tmp = p, p = q, q = tmp;
  
  //we compute the value of [log(LEVEL[p)]
      for (log = 1; 1 << log <= LEVEL[p]; log++);
      log--;
   
  //we find the ancestor of node p situated on the same level
  //with q using the values in ANCESTOR
      for (i = log; i >= 0; i--)
          if (LEVEL[p] - (1 << i) >= LEVEL[q])
              p = ANCESTOR[p][i];
   
      if (p == q)
          return p;
   
  //we compute LCA(p, q) using the values in ANCESTOR
      for (i = log; i >= 0; i--)
          if (ANCESTOR[p][i] != -1 && ANCESTOR[p][i] != ANCESTOR[q][i])
              p = ANCESTOR[p][i], q = ANCESTOR[q][i];
   
      return PARENT[p];
}

void FillLevelsAndParents(int q,int l,int p){
	LEVEL[q]=l;
	PARENT[q]=p;
	for(int i=0;i<v[q].size();i++)
		if(v[q][i]!=p)
			FillLevelsAndParents(v[q][i],l+1,q);
}

int main(){
	int t,i,j,x,y,q,k;
	
	scanf("%d",&t);
	
	while(t--){
		scanf("%d",&N);
		
		for(i=0;i<N;i++)	v[i].clear();
		
		for(i=0;i<(N-1);i++){
			scanf("%d%d",&x,&y);
			x--,y--;						
			v[x].push_back(y);
			v[y].push_back(x);
		}
		
		FillLevelsAndParents(0,0,-1);
		Preprocess();
		
		scanf("%d",&q);
		
		while(q--){
			scanf("%d%d",&x,&y);
			x--,y--;
			k=LCA(x,y);
			
			i=0;
			if(k==x){
				i=1 + 2*( LEVEL[y] - LEVEL[x] -1 );
			}else if(k==y){
				i=2*( LEVEL[x] - LEVEL[y] );
			}else{
				i=2*( LEVEL[x] - LEVEL[k] ) + 1 + 2*( LEVEL[y] - LEVEL[k] -1 );
			}
			printf("%d\n",i);
		}
		
	}
	
	return 0;
}
