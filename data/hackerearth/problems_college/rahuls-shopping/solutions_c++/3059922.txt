#include<iostream>
using namespace std;
int compute(int a[],int n,int area)
{
	double temp=0;
	long int sum=0;
	for(int i=0;i<n;i++)
	{
		if(a[i]>=1 && a[i]<=10)
		{
			int t=a[i],r=0,s=0;
			while(t!=0)
   			{
    			r=t%10;
      			s=s+r;
      			t=t/10;
   			}
   			temp=area*a[i]*s;
			temp=(temp-(temp*25)/100);
			sum+=int(temp);
		}
		else
		{
			if(a[i]>=11 && a[i]<=20)
			{
				int t=a[i],r=0,s=0;
				while(t!=0)
   				{
    				r=t%10;
      				s=s+r;
      				t=t/10;
   				}
   				temp=area*a[i]*s;
				temp=(temp-(temp*30)/100);
				sum+=int(temp);
			}
			else
			{
				int t=a[i],r=0,s=0;
				while(t!=0)
   				{
    				r=t%10;
      				s=s+r;
      				t=t/10;
   				}
   				temp=area*a[i]*s;
				temp=(temp-(temp*20)/100);
				sum+=int(temp);
			}
		}
	}
	return sum;
}
int main()
{
	int n,l,b;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)
		cin>>a[i];
	cin>>l>>b;
	cout<<compute(a,n,l*b)<<endl;
	return 0;
}
