#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t; cin>>t;
    while(t--) {
    	int n, m, a, b, i;
    	cin>>m>>n;
    	vector<pair<int, int> > M, N;
    	for(i = 0; i < m; i++) {
    		scanf("%d%d", &a, &b);
    		M.push_back(make_pair(a, b));
    	}
    	for(i = 0; i < n; i++) {
    		scanf("%d%d", &a, &b);
    		N.push_back(make_pair(a, b));
    	}
    	int G, H;
    	scanf("%d%d", &G, &H);
    	sort(M.begin(), M.begin()+m); sort(N.begin(), N.begin()+n);
    	int g = 0, h = 0;
    	for(i = 0; i < m; i++) {
    		if(M[i].first == N[i].first) g++;
    		if(M[i].second == N[i].second) h++;
    	}
    	if(g < G && h < H) printf(":(\n");
    	else if(g >= G && h < H) printf("Good\n");
    	else if(g >= G && h >= H) printf("Great\n");
    }
    return 0;
}
