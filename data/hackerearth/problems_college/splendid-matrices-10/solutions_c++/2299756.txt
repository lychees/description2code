
#include <iostream>
#include<cmath>
using namespace std;

int arr[1024][1024];


void constructMatrix(int n,int s,int a,int b){
	
	if(n>1){
		
		int c=pow(2,n)/2;
	    int p=pow(2,n-1)*pow(2,n-1);
	    constructMatrix(n-1,s,a,b);
	    constructMatrix(n-1,s+p,a,b+c);
	    constructMatrix(n-1,s+p*2,a+c,b);
	    constructMatrix(n-1,s+p*3,a+c,b+c);
	    
		
	}
	 else if(n==1){
	   
	     arr[a][b]=s;
	     arr[a][b+1]=s+1;
	     arr[a+1][b]=s+2;
	 	 arr[a+1][b+1]=s+3;
	 }
	 
	}
	


int main()
{
  int n;
  cin>>n;
  int N=pow(2,n);
  constructMatrix(n,1,0,0);
  
  for(int i=0;i<N;i++){
   for(int j=0;j<N;j++){
        cout<<arr[i][j]<<" ";
       }
      cout<<endl;
  }
}
