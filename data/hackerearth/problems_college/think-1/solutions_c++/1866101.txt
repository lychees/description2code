#include<iostream>
#include<math.h>

using namespace std;
typedef long long int ll;

ll reverse1(ll a)
{
    ll rev=0;
    while(a>0)
    {
        rev=rev*10+a%10;
        a=a/10;
    }
    return rev;

}
int main()
{
    int t,n,si,i=0,j;
    ll arr[51],temp,tmp1;
    ios::sync_with_stdio(0);
    cin>>t;
    si=t;

    while(t--)
    {
        cin>>n;
        temp=reverse1(n);
		if(temp==n)
		{
			cout<<"No\n";
			continue;
		}
        n=n*n;
        temp=temp*temp;
        tmp1=reverse1(temp);
        if(tmp1==n)
            cout<<sqrt(temp)<<"\n";
        else
            cout<<"No\n";
        i++;
    }


}
