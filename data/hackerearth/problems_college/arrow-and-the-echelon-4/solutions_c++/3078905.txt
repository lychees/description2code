#include <iostream>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    for(int tt=1;tt<=t;tt++)
    {
    	int x;
    	scanf("%d",&x);
    	int n5=-1;
    	for(int i=0;i<=x;i++)
    	{
    		if(i%3==0 && (x-i)%5==0)
    		{
    			n5=i;
    		}
    	}
    	if(n5==-1)
    	{
    		printf("-1\n");
    	}
    	else
    	{
    		for(int i=1;i<=x;i++)
    		{
    			if(i<=n5)
    			{
    				printf("5");
    			}
    			else
    			{
    				printf("3");
    			}
    		}
    		printf("\n");
    	}
    }
}
