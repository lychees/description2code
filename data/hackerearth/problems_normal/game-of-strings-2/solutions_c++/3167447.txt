#include<bits/stdc++.h>
using namespace std;
int kmp1[1000005];
int mxp=0;
string s;
 
void kmp(int pos)
{
	int i,k=0;
	kmp1[0]=0;
	for(i=1;i<s.size();i++)
	{
		while(k && s[i]!=s[k])
			k=kmp1[k-1];
		if(s[i]==s[k])
			k++;
		kmp1[i]=k;
		if(i<=pos) mxp=max(mxp,kmp1[i]);
	}
}
 
int main(){
	int T;
	cin>>T;
	while(T--){
		cin>>s;
		int n=s.length();
		int k;	
		cin>>k;
		mxp=0;
		k--;
		memset(kmp1,0,sizeof(kmp1));
		kmp(k);
		int ans=kmp1[n-1];	
		while( ans>mxp && ans)	ans=kmp1[ans-1];
		if(ans==0){
			cout<<"Puchi is a cheat!\n";
		}
		else{
			for(int i=0;i<ans;i++)	cout<<s[i];
			cout<<"\n";
		}
	}
}
