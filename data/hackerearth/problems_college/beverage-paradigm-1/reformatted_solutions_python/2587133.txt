import sys


def subset_sum(numbers, j, target, partial=[]):
    s = sum(partial)
    k = 1
    # print n
    # check if the partial sum is equals to target
    if s == target:
        print "True"
        sys.exit()
    if s >= target:
        return  # if we reach the number why bother to continue

    for i in range(len(numbers)):
        n = numbers[i]
        remaining = numbers[i+1:]
        # print i,
        if (i == j-1):
            print "False"
        subset_sum(remaining, j, target, partial + [n])
