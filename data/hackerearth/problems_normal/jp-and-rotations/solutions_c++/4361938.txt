#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int main()
{
    int n,m,h,ans=-1,flag,temp=0;
    cin>>n>>m;
    int A[n],B[n],H[m+1];
    char c,C[m+1];
    for(int i=0;i<n;i++)
        cin>>A[i];
    for(int i=0;i<n;i++)
        cin>>B[i];
    for(int i=1;i<=m;i++)
        cin>>C[i]>>H[i];
    for(int i=1;i<=m;i++)
    {
        c=C[i];h=H[i];
        if(c == 'R')
            temp = (temp + h)%n;
        else
        {
            temp = (temp - h)%n;
            if(temp<0)
                temp=temp + n;
        }
        flag=0;
        for(int j=0;j<n;j++)
        {
            if(A[j]!=B[(temp+j)%n])
            {
                flag=1;
                break;
            }
        }
        if(flag==0)
        {
            ans=i;
            break;
        }
    }
    cout<<ans<<endl;
    return 0;
}
