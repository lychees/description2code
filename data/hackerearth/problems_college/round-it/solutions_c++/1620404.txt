#include <bits/stdc++.h>
using namespace std;

int main()
{
	long int t,a,b;
	long int temp,temp1;
	scanf("%ld",&t);
	while(t--)
	{
		temp=temp1=0;
		scanf("%ld%ld",&a,&b);
		if(a<b){
	   printf("%lld\n",(a*b)/__gcd(a,b));continue;}
	   temp=a/b;
	   temp1=(temp+1)*b;
	   temp*=b;
	   if(abs(temp-a)==abs(temp1-a))
	   printf("%ld\n",temp1);
	   else
	   printf("%ld\n",abs(temp-a)>abs(temp1-a)?temp1:temp);
	}
    return 0;
}
