int arr [100000 + 10];
int dp[100000 + 10][2][2];

int rec (int index , int res , int take){

    if(index == n){
        if(res == 0 && take == 1) return 1;
        else return 0;
    }
    int &st = dp[index][res][take];
    if(st != -1) return st;

    int e = 0;
    if(take == 0) e += rec(index + 1 , res , 1);
    if( (res == 0 && arr[index] == 1 ) || (arr[index] == 0 && res == 1) ) res = 1;
    else res = 0;
    if(index == 0) res = arr[index];
    e += rec(index + 1 , res , take);
    return st = e;
}
