#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long int t,n,a[100];
	a[0]=0;
	a[1]=1;
	for (int i=2;i<100;i++) {
		a[i]=a[i-1]+a[i-2];
	}
	scanf("%lld",&t);
	long long int sum=0;
	while(t--) {
		scanf("%lld",&n);
		sum=0;
		for (int i=0;a[i]<=n;i++) {
			if(!(a[i]&1)) sum+=a[i];
		}
		printf("%lld\n",sum);
	}
}
