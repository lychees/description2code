#include<bits/stdc++.h>
using namespace std;

typedef long long int ll;

int main()
{
	ios::sync_with_stdio(0);
	int n,x;
	cin >> n ;
	while(n--)
	{
		cin >> x ;
		int i,flag=0,l=sqrt(x);
		if(l*l==x)
		l--;
		for(i=2;i<=l;i++)
		{
			if(x%i==0)
			{
				flag=1;
				break;
			}
		}
		if(flag==1)
		cout <<"YES\n";
		else
		cout <<"NO\n";
	}
	return 0;
}
