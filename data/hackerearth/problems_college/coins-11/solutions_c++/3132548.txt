#include<stdio.h>

int isp(int n){
	int i;
	for(i = 2; i * i <= n; i++)	if(n % i == 0)	return 0;
	return 1;
}

int main(){
	int t, n; scanf("%d", &t);
	while(t-- && scanf("%d", &n)){
		if(n == 3)	printf("3\n");
			else if(n == 1 || n % 2 == 0)	printf("1\n");
				else	isp(n) ? printf("2\n") : printf("1\n");
	}		
	return 0;
}
