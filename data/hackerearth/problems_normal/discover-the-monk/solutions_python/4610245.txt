def bs(arr,se,l,r):
	if l<=r:
		mid=(l+r)/2
		if se==arr[mid]:
			return "YES"
		elif se<arr[mid]:
			return bs(arr,se,l,mid-1)
		elif se>arr[mid]:
			return bs(arr,se,mid+1,r)
	else:
		return "NO"


n,q=map(int,raw_input().split())
arr=map(int,raw_input().split())
arr=sorted(arr)
while q>0:
	x=input()
	if x>=arr[0] and x<=arr[-1]:
		print bs(arr,x,0,n)
	else:
		print "NO"
	q-=1
