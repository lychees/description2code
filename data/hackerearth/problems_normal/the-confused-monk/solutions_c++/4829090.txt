#include <bits/stdc++.h>
#define M 1000000007;
using namespace std;

long long int power(long long int x, long long int n)
{
    long long int ans=1;
    while(n>0)
    {
        if(n%2==1)
            ans=(ans*x)%M;
        n/=2;
        x=(x*x)%M;
    }
    return ans;
}

int main()
{
    long long int a[100],n;
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>a[i];
    long long int gcd=a[0],mul=1;
    for(int i=0;i<n;i++)
    {
        gcd = __gcd(gcd,a[i]);
        mul=(mul*a[i])%M;
    }
    long long int ans=power(mul,gcd);

    cout<<ans<<endl;
}
