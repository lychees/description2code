#include <bits/stdc++.h>
using namespace std;
#define ll long long int
ll a[100005] = {0};
void add()
{
	int i;
	a[0] = 1;
	a[1] = 1;
	a[2] = 2;
	for(i = 3; i <= 1000; i++) {
		a[i] = a[i-2]+a[i-3];
	}
}
int main()
{
	int n;
	add();
	scanf("%d",&n);
	printf("%lld\n",a[n]);
	return 0;
}
