inline bool EQ(double a,double b) { return fabs(a - b) < 1e-9; }

const int MAXN = 1e5+3;
ll ar[MAXN];
ll br[MAXN];
int main() {
	int t;
	scanf("%d",&t);
	while(t--) {
		int n;
		ll k;
		scanf("%d%lld",&n,&k);
		for(int i=0;i<n;i++)
			scanf("%d",&ar[i]);
		for(int i=0;i<n;i++)
			scanf("%d",&br[i]);
		ll ans=0;
		for(int i=0;i<n;i++) {
			ans=max(ans,(k/ar[i])*br[i]);
		}
		printf("%lld\n",ans);
	}
	return 0;
}
