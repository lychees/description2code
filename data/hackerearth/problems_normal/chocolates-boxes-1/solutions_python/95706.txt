from collections import defaultdict

memory = {}


def n_combinations(n_nrs):
    m = memory.get(n_nrs, None)

    if m is None:
        ret = sum(range(n_nrs))
        memory[n_nrs] = ret
    else:
        ret = m

    return ret

def go(arr, k):
    count = 0
    occurences = {}

    i = 0
    for n in arr:
        if n < k:
            occurences[n] = occurences.get(n, 0) + 1

    while occurences:
        n, n_count = occurences.popitem()
        m = k - n


        if m in occurences:
            count += n_count * occurences[m]
            occurences.pop(m)
        elif n * 2 == k:
            count += n_combinations(n_count)

    print count
