#include <stdio.h>
#include <string.h>

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i;
        char j;
        char a[110][110];
        int f[27][110]={0};
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            scanf("%s",a[i]);
            for(j='a';j<='z';j++)
                if(strchr(a[i],j)!=NULL)
                    f[j-'a'][i]=1;

        }
        int y[27]={0};
        int max=0,k;
        for(i=0;i<26;i++)
            for(k=0;k<n;k++)
                y[i]+=f[i][k];

        char c;
        for(i=0;i<26;i++)
            if(max<y[i])
             {
             	max=y[i];
             	c=i+'a';
             }
        printf("%c\n",c);
    }
return 0;
}
