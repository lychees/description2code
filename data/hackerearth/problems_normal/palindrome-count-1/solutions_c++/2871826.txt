#include <iostream>

bool isPalindrome(string s) {
	for (int i = 0; i < s.size() / 2; i++) {
		if (s[i] != s[s.size() - i - 1]) return false;
	}
	return true;
}

int helper(string s, int begin, int end) {
	int res = 0;
	while(begin >= 0 && end < s.size() && s[begin] == s[end]) {
		res++;
		begin--;
		end++;
	}
	return res;
}
