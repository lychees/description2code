T = int(raw_input())


class Stack:
    def __init__(self):
        self.items = []

    def isEmpty(self):
        return self.items == []

    def push(self, item):
        self.items.append(item)

    def pop(self):
        return self.items.pop()

    def peek(self):
        return self.items[len(self.items)-1]

    def size(self):
        return len(self.items)


count = 0
for _ in range(T):
    stack = Stack()

    lines = raw_input()
    for letter in list(lines):
        if (stack.isEmpty()):
            stack.push(letter)
        elif (stack.peek() == letter):
            stack.pop()
        elif (stack.peek() != letter):
            stack.push(letter)
    if stack.isEmpty():
        count += 1
print count
