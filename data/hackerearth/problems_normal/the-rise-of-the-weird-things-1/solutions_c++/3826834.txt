#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
	int N;
	cin>>N;
	
	multiset<int> zombie;
	multiset<int> vampires;
	ll zombie_sum =0;
	ll vampire_sum =0;
	int  a;
	for(int i=0;i<N;i++)
	{
		cin>>a;
		if(a%2 == 0)
		{
			//it is a zombie
			zombie.insert(a);
			zombie_sum+=a;
		}
		else 
		{
			vampires.insert(a);
			vampire_sum+=a;
		}
	}
	
	for( multiset<int>::iterator it = zombie.begin(); it!= zombie.end() ; it++)
	{
		cout<<*it<<" ";
	}
	cout<<zombie_sum<<" ";
	for( multiset<int>::iterator it = vampires.begin(); it!= vampires.end() ; it++)
	{
		cout<<*it<<" ";
	}
	cout<<vampire_sum;
	
    return 0;
}
