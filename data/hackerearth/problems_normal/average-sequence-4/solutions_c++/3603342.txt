#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
   	long long int len, ite;
   	long long int tmp, cumSum = 0;
   	scanf("%lld", &len);
   	for(ite=1;ite<=len;ite++)
   	{
   		scanf("%lld", &tmp);
   		if(ite==1)
   		{
   			cumSum += tmp;
   			printf("%lld ", tmp);
   		}
   		else
   		{
   			printf("%lld ", ((tmp*ite)-cumSum));
   			cumSum += ((tmp*ite)-cumSum);
   		}
   	}
    return 0;
}
