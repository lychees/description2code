#include <bits/stdc++.h>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	long long t,a,b,c,k;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		cin>>a>>b>>c>>k;
		long long left=-1, right=sqrt(k)+1, mid, eq;
		while(right-left>1)
		{
			mid=(left+right)/2;
			eq=a*mid*mid+b*mid+c;
			if(eq >= k)
			{
				right=mid;
			}
			else
			{
				left=mid;
			}
		}
		cout<<right<<endl;
	}
	return 0;
}
