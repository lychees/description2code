#include<iostream>
using namespace std;
#define ll long long
int num[110]={0},pc=0,pri[110],fac[110][110],invf[110][110],xx[1000],pp[1000],k=0;
ll power(ll a, ll b, ll mod)
{
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
ll euclid(ll a,ll m) 
{
	return power(a,m-2,m);
}
ll luc(ll a,ll b,ll c)
{
	if(a==0 || b==0) return 1;
	else
	{
		ll d1=a%pri[c],d2=b%pri[c];
		if(d1<d2)
			return 0;
		else
			return ((fac[c][d1]*invf[c][d1-d2]*invf[c][d2])*luc(a/pri[c],b/pri[c],c))%pri[c];
	}
}
ll ncr(ll a,ll b, ll c)
{
	k=0;
	for(int i=0;i<pc;i++)
	{
		if(c%pri[i]==0)
		{
			pp[k]=pri[i];
			xx[k]=luc(a,b,i);
			k++;
		}
	}
	ll p, i, prod = 1, sum = 0;
	for (i = 0; i < k; i++) 
	prod *= pp[i];
	for (i = 0; i < k; i++)
	{
		p = prod / pp[i];
		sum = (sum+(xx[i] * euclid(p, pp[i]) * p)%c)%c;
	}
	return sum %c;
}
int main()
{	
	int test;
	for(int i=2;i<=100;i++)
	{
		if(num[i]==0)
		{
			pri[pc]=i;
			pc++;
			for(int j=i*2;j<=100;j+=i)
			num[j]=1;
		}
	}
	for(int i=0;i<pc;i++)
	{
		fac[i][0]=1,invf[i][0]=1;
		for(int j=1;j<=100;j++)
			fac[i][j]=(fac[i][j-1]*j)%pri[i],invf[i][j]=(euclid(j,pri[i])*invf[i][j-1])%pri[i];
	}
	cin>>test;
	while(test--)
	{
		ll a,b,c;
		cin>>a>>b>>c;
		if(a<b)
		cout<<0<<endl;
		else if(c==1)
		cout<<0<<endl;
		else if(b==1)
		cout<<1<<endl;
		else
		{
			a=a-b;
			cout<<ncr(a+b-1,b-1,c)<<endl;
		}
	}
	return 0;
}
