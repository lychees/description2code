#include <bits/stdc++.h>
using namespace std;


int dp[120];

int count_(int n)
{

	
	if(n==0)
	return 0;
	
	else if(n<0)
	return 1e9;
	else if(dp[n]!=-1)return dp[n];
	
	else {
	int res=1+min(count_(n-12),count_(n-10));
	 return dp[n]=res;
	}
	
}
int main()
{
    ios_base::sync_with_stdio(0);
    int t;
    cin>>t;
    memset(dp,-1,sizeof(dp));
    while(t--)
    {
    	int n;
    	cin>>n;
    	int res=count_(n);
    	if(res>n)cout<<"-1"<<endl;
    	else
    	cout<<res<<endl;
    }
    return 0;
}
