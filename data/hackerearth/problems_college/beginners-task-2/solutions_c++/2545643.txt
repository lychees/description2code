#include <iostream>
using namespace std;

int main()
{
    int t,n,i;
    long int sum;
    long int *a;
    cin>>t;
    while(t--){
    	cin>>n;
    	a = (long int *)malloc(sizeof(long int)*n);
    	for(i=0;i<n;i++){
    		cin>>a[i];
    	}
    	sum=0;
    	for(i=0;i<n;i++){
    		sum+=a[i];
    	}
    	sum/=n;
    	cout<<sum<<endl;
    }
    return 0;
}
