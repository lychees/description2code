#include<bits/stdc++.h>

typedef long long ll;
using namespace std;

int main(){
    std::ios::sync_with_stdio(false);
    int t,w,d,l,temp,max=-100;
    cin>>t;
    for(int i=0; i<t; i++){
        cin>>w>>d>>l;
        temp=((3*w)+(d)-(l));
        if(temp>max)max=temp;
    }
    cout<<max<<endl;
    return 0;
}
