#include <iostream>
#include <stdio.h>
#include <math.h>

#define MAXTAB 1005

using namespace std;

int mapa[MAXTAB][MAXTAB];

int main(){
    int n, x, y, raio, res = 0;

    scanf("%d", &n);
    for(int k = 1; k <= n; k++){
        scanf("%d %d %d", &x, &y, &raio);

        int fim_i = x+raio; if(fim_i>1000) fim_i=1000;
        int fim_j = y+raio; if(fim_j>1000) fim_j=1000;

        int i = x-raio; if(i<1) i=1;
        for(; i<=fim_i; i++){
            int j = y-raio; if(j<1) j=1;
            for(; j<=fim_j; j++){
                double dist = sqrt(pow(x-i,2) + pow(y-j,2));
                if(dist <= raio){
                    mapa[i][j]++;
                    if(mapa[i][j]==2) res++;
                }
            }
        }
    }
    printf("%d\n", res);

    return 0;
}
