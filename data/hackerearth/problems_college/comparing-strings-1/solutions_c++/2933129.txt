#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	long long int t,l1,l,i,j,flag;
char a[100000],b[100000];
	cin>>t;
	while(t--)
	{flag=0;
		cin>>a>>b;
		l1=strlen(b);
		
		l=strlen(a);
		for(i=0;i<l;i++)
		{
		for(j=0;j<l1;j++)
		{
		if(a[i]==b[j])
		{
			flag=1;
			break;
		}
		}
		}
		if(flag==1)
		cout<<"YES\n";
		else cout<<"NO\n";
	}
    return 0;
}
