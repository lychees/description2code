#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    for(int k=0;k<n;k++)
    {
        char s[100000];int sum=0,x1=0;
        cin>>s;
        int l=strlen(s);
        for(int i=0;i<l/2;i++)
            sum=sum+s[i];
        for(int i=l/2;i<l;i++)
            x1=x1+s[i];
        if(sum==x1)cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
 
    }
    return 0;
}
