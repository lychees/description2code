#include <iostream>
using namespace std;

int main()
{
	int t,n,index;
	cin>>t;
	while(t--){
		cin>>index>>n;
	if(index==2 && n==0)
		cout<<"2"<<endl;
	else if((index==0 && n%2==0) ||(index==1 && n%2==1)||(index==2 && n%2==0)){
		cout<<"0"<<endl;
	}else
		cout<<"1"<<endl;
	
	}
    //cout << "Hello World!" << endl;
    return 0;
}
