typedef long long LL;
typedef vector<int> vi;

typedef vector <LL> vll;
typedef vector <pair<LL,LL> > vpll;

struct Point{
    int x;
    int y;
};
Point l1,l2,r1,r2;
bool overlap()
{
    if(r1.y<=l2.y || r2.y<=l1.y)
        return false;
    if(r1.x<=l2.x || r2.x<= l1.x)
        return false;
    return true;
}
bool point()
{
    int w,l;
    w = r1.x - l1.x;
    l = r1.y - l1.y;
    Point a[8];
    a[0] = l1;
    a[1] = r1;
    a[2].x = l1.x;
    a[2].y = l1.y + l;
    a[3].x = l1.x + w;
    a[3].y = l1.y;


    w = r2.x - l2.x;
    l = r2.y - l2.y;
    a[4] = l2;
    a[5] = r2;
    a[6].x = l2.x;
    a[6].y = l2.y + l;
    a[7].x = l2.x + w;
    a[7].y = l2.y;
    for(w=0;w<4;w++)
        for(l=4;l<8;l++)
            if(w!=l && a[w].x==a[l].x && a[w].y==a[l].y)
                return 1;
    return 0;
}
bool line()
{
    if(r1.y==l2.y || l1.y==r2.y)
    {
        if(r1.x<=l2.x || r2.x<=l1.x)
            return false;
        return 1;
    }
    if(r1.x==l2.x || l1.x==r2.x)
    {
        if(r1.y<=l2.y || r2.y<=l1.y)
            return false;
        return 1;
    }
    return 0;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        cin>>l1.x>>l1.y>>r1.x>>r1.y;
        cin>>l2.x>>l2.y>>r2.x>>r2.y;
        //Overlap
        if(overlap())
            cout<<"Rectangle\n";
        else if(line())
            cout<<"Line\n";
        else if(point())
            cout<<"Point\n";
        else
            cout<<"Apart\n";
    }
    return 0;
}
