t = input()
if t >= 1 and t <= 100:
    for i in range(0, t):
        s = raw_input()
        S = 0
        if int(s) >= 1 and int(s) <= pow(10, 18):
            S = s.count("4")+s.count("7")
            if str(S).count("4")+str(S).count("7") == len(str(S)):
                print("YES")
            else:
                print("NO")
