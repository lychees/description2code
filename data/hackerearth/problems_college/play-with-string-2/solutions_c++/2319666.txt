#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int n= s.length();
    	string r="";
    	for(int i=n-1;i>=0;i--)
    	{
    		r=r+s[i];
    	}
    	int flag=0;
    	for(int i=1;i<n;i++)
    	{
    		if(abs(s[i]-s[i-1])!=abs(r[i]-r[i-1]))
    		{
    			flag=1;
    			break;
    		}
    	}
    	
    	if(flag==1)
    		cout<<"Not Funny"<<endl;
    	else
    		cout<<"Funny"<<endl;
    }
    
    return 0;
}
