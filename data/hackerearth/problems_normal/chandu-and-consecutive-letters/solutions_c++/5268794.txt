#include <iostream>
using namespace std;

int main()
{
    int count;
    cin >> count;
    for( int i = 0; i < count; i++)
    {
    	string str;
    	cin >> str;
    	cout<<str.at(0);
    	for(int j = 1; j < str.length(); j++)
    	{
    		if(str.at(j) != str.at(j-1))
    		{
    			cout<< str.at(j);
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
