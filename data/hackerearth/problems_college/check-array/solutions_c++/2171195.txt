#include <iostream>
#include<string.h>
using namespace std;
int a[100003];
int main()
{
	int T;
	cin>>T;
	while(T--){
		int n ;
		cin>>n;
		for(int  i=0;i<n;i++){
			cin>>a[i];
		}
		long s1[n],s2[n];
		for(int i =0;i<n;i++)s1[i] =0;
		for(int i = 0;i<n;i++)s2[i] =0;
		s1[0] = a[0];
		for(int i =1;i<n;i++){
			s1[i] = a[i]+s1[i-1];
		
		}
		s2[n-1] = a[n-1]; 
		for(int i = n-2;i>=0;i--){
			s2[i] = a[i]+s2[i+1];
		
		}
		bool k = false;
		for(int i = 1;i<=n-2;i++){
			if(s1[i-1] == s2[i+1]){
				k = true;	
				break;
			}
		}
		if(k){
			cout<<"YES"<<endl;
		}
		else{
			cout<<"NO"<<endl;
		}
	}
    
    return 0;
}
