#include<cmath>
#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	long int n,k,i;
   	cin>>n;
   	for(i=1;i<n;i++)
   	{
   		if(pow(2,i)>n)
   		{
   			k=i;
   			break;
   		}
   	}
   	
   	k--;
   	long int h=n-pow(2,k);
   	cout<<h*2+1<<"\n";
   }
}
