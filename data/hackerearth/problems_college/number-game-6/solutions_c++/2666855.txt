#include<bits/stdc++.h>
#include<iostream>
#define ll long long
using namespace std;
ll arr[1000006],k;
void pre()
{
	ll d=100000000000000000;
	ll j=2,i;
	arr[k++]=0;
	ll tmp=4;
	for(i=3; tmp<=d; i++)
	{
		arr[k++]=tmp;
		tmp=tmp+i*i*j;
		j++;
	}
}
ll scan()
{
    ll n = 0, ch = getchar_unlocked();
    while (ch < '0' || ch > '9')
        ch = getchar_unlocked();
    while (ch >= '0' && ch <= '9') {
        n = (n << 3) + (n << 1) + ch - '0';
        ch = getchar_unlocked();
    }
    return n;
}
int main()
{
	pre();
	ll t,n;
	t=scan();
	while(t--)
	{
		n=scan();
		if(n<4)
		{
			puts("0");
			puts("Draw");
			continue;
		}
		ll *x;
		x=lower_bound(arr,arr+k,n);
		ll ans=x-arr;
		if(arr[ans]!=n)
		ans--;
		cout<<ans<<endl;
		puts((ans&1)?"Dhruv":"Ketan");
	}
	return 0;
}
