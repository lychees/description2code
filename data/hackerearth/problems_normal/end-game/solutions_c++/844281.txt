#include<bits/stdc++.h>
#include <iostream>

using namespace std;

int main()
{
	long long int t,n,a,b,c,d;
	int move;
	cin>>t;
	while(t--)
	{
		cin >> n >> a >> b >> c >> d >> move;
		if(move == 0)
		{
			if(c >= a && abs(d-b) <= (n-a))
			cout<<"Draw"<<endl;
			else
			cout<<"White Wins"<<endl;
		}
		else
		{
			if(c >= a-1 && abs(d-b) <= (n-a+1))
			cout<<"Draw"<<endl;
			else
			cout<<"White Wins"<<endl;
		}
	}
	
    
    return 0;
}
