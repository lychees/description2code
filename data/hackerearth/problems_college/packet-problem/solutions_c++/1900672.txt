#include<iostream>
using namespace std;

int main(){

 long long int arr[90];
 
 arr[0]=0;
 arr[1]=1;
 
 long long int brr[90];
 
 brr[0]=1;
 brr[1]=1;
 
 for(int i=2;i<90;i++){
  arr[i]=arr[i-1]+arr[i-2];
  brr[i]=brr[i-1]+brr[i-2];
 }
 
 int t;
 
 cin>>t;
 
 while(t--){
 
 int m;
 
 cin>>m;
 
 cout<<arr[m-1]<<" "<<brr[m-1]<<"\n"; 

 }

return 0;
}
