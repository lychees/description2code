#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	long long N, M;
    	cin >> N >> M;
    	int s = 0;
    	for(int i = 63; i >= 0; i --) {
    		if( M & (1LL << i) ) {
    			if( s ) cout << " + "; s = 1;
    			cout << "(" << N << "<<" << i << ")";
    		}
    	}
    	cout << endl;
    }
    return 0;
}
