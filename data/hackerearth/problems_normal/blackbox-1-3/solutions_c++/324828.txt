#include <iostream>
#include <cstdio>
#include<cmath>
#include<algorithm>

using namespace std;

#define ll long long

int gcd(ll a,ll b)
{
	if(b==0)
		return a;
	return gcd(b,a%b);
}


int main()
{
    
    int t;
    ll a[3];
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%lld %lld %lld",&a[0],&a[1],&a[2]);
    	sort(a,a+3);
    	if(gcd(a[0],a[1])==1 && gcd(a[1],a[2])==1 && gcd(a[2],a[1])==1)
    	{
    		if((a[2]*a[2])==(a[0]*a[0]+a[1]*a[1]))
    			printf("YES\n");
    		else
    			printf("NO\n");
    	}
    	else
    		printf("NO\n");
    }
    return 0;
}
