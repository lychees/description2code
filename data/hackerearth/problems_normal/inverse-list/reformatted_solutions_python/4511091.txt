t = int(raw_input())

for n in xrange(1, t+1):
    x = raw_input()
    out = [None] * (int(x)+1)
    in_ = raw_input().split()
    in_ = [int(k) for k in in_]
    in_.insert(0, 0)
    for n in in_:
        out[in_[n]] = n
    if out == in_:
        print "inverse"
    else:
        print "not inverse"
