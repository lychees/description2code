#include <iostream>

using namespace std;

int main()
{
	int lowO = 200;
	int n;
	int sum = 0;
	cin >> n;
	int arr[n];

	for (int i = 0; i < n; i++) { 
		cin >> arr[i];
		sum += arr[i];
		if ( arr[i]%2 && arr[i] < lowO ) lowO=arr[i];
	}

	if(!(sum%2)) {
		if(lowO==200) {
			cout << ":(\n";
		}else {
			cout << sum-lowO << endl;
		}
	} else {
		cout << sum << endl;	
	}

	return 0;
}
