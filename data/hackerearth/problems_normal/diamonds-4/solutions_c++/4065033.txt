#include <iostream>
using namespace std;

int main()
{
   int i,t,j,k,n,m;char a[1001][1001];
   cin>>t;
   for(i=0;i<t;i++)
   {
   	cin>>n>>m;
   	for(j=0;j<n;j++)
   	{
   		for(k=0;k<m;k++)
   		{
   			cin>>a[j][k];
   		}
   	}
   	int s=0;
   	for(j=0;j<n-1;j++)
   	{
   		for(k=0;k<m-1;k++)
   		{
   			if(a[j][k]=='/')
   			{
   			if(a[j][k+1]=='\\' && a[j+1][k]=='\\' && (a[j+1][k+1]=='/'))
   			{
   				s+=1;
   			}
   			}
   		}
   	}
   	cout<<s<<"\n";
   }
    return 0;
}
