#include<iostream>
using namespace std;

int main()
{
    int n1,n2,ans;
    string word,str;
    cin>>word>>str;
    n1=word.length();
    n2=str.length();

    int z[n2+1],temp[n2+1];
    for(int i=0;i<=n2;i++)
    z[i]=0;
    for(int i=0;i<=n2;i++)
    temp[i]=-1;

    for(int i=0;i<n1;i++)
    {
        if(i==0)
        {
            for(int j=0;j<n2;j++)
            {
                if(word[i]==str[j])
                {
                    temp[j]=i;
                    z[j]=1;
                }
            }
        }
        else
        {
            ans=0;
            for(int j=0;j<n2;j++)
            {
                if(temp[j]==i-1)
                ans+=z[j];
                else if(word[i]==str[j])
                {
                    temp[j]=i;
                    z[j]=ans;
                }
            }
        }
    }
        ans=0;
        for(int j=0;j<n2;j++)
        {
            if(temp[j]==n1-1)
            ans+=z[j];
        }
        cout<<ans;
    return 0;
}
