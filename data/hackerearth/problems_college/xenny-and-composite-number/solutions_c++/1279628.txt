int prime[100005];
int ans[100005];
int main(){
	int i,j;
	for(i=2;i<=100004;i++){
		if(!prime[i]){
			for(j=2*i;j<=100004;j+=i)
			prime[j]++;
		}
	}
	int tmp=-1;
	for(i=100004;i>=0;i--){
		ans[i]=tmp;
		if(prime[i]!=0)
		tmp=i;
	}
	int t,n;
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		printf("%d\n",ans[n]);
	}
	return 0;
}
