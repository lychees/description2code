#include <iostream>
#include <vector>
#define pb push_back
#define ll long long
using namespace std;

int main() {
	
	vector<ll> vec;
	ll t,n,x,psg;
	cin >> t;
	while(t--) {
		cin >> n;
		vec.pb(n);
	}
	
	cin >> x;
	ll dp = 1;
	while(x--) {
		cin >> psg;
		dp *= psg;
	}
	
	for(ll i=0; i<vec.size(); i++) {
		dp==0?
			vec[i] = 0:  
			vec[i] /= dp;
	}
	
		for(ll i=0; i<vec.size(); i++) 
			cout << vec[i] << ' ';
		cout << endl;

	return 0;
}
