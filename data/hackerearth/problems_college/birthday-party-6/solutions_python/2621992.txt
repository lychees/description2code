MOD = 1000000007

def factorial(num):
	ans = 1
	for i in range(1,num+1):
		ans = ans*i
	
	return ans	

def nPr(n,r):
	ans = 1
	for i in range(r):
		ans = ans*n
		n = n-1
	
	return ans	
			
def noOfWays(G,B):
	if G+1 < B:
		return -1
	else:
		G1 = factorial(G)
		G2 = nPr(G+1,B)
		
		return G1*G2%MOD
		
t = int(raw_input())
for i in range(t):
	G,B = map(int, raw_input().strip().split(' '))
	print noOfWays(G,B)
