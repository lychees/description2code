#include<cstdio>
#define MOD 1000000007
long long POW(long long a,long long n){
    long long ans=1;
    while(n){
        if(n&1)
            ans=(ans*a)%MOD;
        a=(a*a)%MOD;
        n>>=1;
    }
    return ans;
}
int main(){
    int cs;
    scanf("%d",&cs);
    while(cs--){
        long long n;
        scanf("%lld",&n);
        long long ans1=(((n*(n-1))%MOD*(n-1))%MOD*POW(4,MOD-2))%MOD;
        long long ans2=(((n*(n-1))%MOD*(2*n-1))%MOD*POW(6,MOD-2))%MOD;
        printf("%lld %lld\n",ans1,ans2);
    }
    return 0;
}
