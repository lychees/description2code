#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,a,b,c,max1,min1,mid;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d%d",&a,&b,&c);
		max1=max(max(a,b),c);
		min1=min(min(a,b),c);
		if(max1!=a&&min1!=a)
			mid=a;
		else if(max1!=b&&min1!=b)
			mid=b;
		else if(max1!=c&&min1!=c)
			mid=c;
		printf("%d\n",max(mid-min1,max1-mid)-1);
	}
	return 0;
}
