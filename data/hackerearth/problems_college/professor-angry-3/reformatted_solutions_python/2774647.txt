t = int(raw_input())

res_list = []
while t > 0:
    count = 0
    n, k = map(int, raw_input().split())
    stu = map(int, raw_input().split())
    for i in stu:
        if i <= 0:
            count += 1
    if count >= k:
        res_list.append('NO')
    else:
        res_list.append('YES')
    t -= 1
for res in res_list:
    print res
