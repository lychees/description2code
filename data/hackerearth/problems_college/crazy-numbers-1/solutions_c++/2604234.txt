#include <iostream>
using namespace std;


long long calctot(int arr[10][1000000], int n)
{
	long long total=0;
	if(n>1)	{
	for(int i=2;i<=n;i++)
	{for(int d=1;d<=9;d++)
	
	{
		if(d-1 == 0 && i>2)	arr[d][i] = arr[1][i-2];
		else arr[d][i] = arr[d-1][i-1];
		if(d+1<=9)	arr[d][i] = (arr[d][i] + arr[d+1][i-1])%1000000007;
		//cout<<"arr["<<d<<"]"<<"["<<i<<"] = "<<arr[d][i]<<endl;
	}
	
	}
	}
	for(int i=0;i<=9;i++)
	total = (total + arr[i][n])%1000000007;
	return total;
}

int main()
{
    //cout << "Hello World!" << endl;
    int t,n;
    cin>>t;
    //cout<<t<<endl;
    static int arr[10][1000000] = {0};
    //arr[0][1] = 1;
    for(int i=0;i<10;i++)
    arr[i][1] = 1;
    for(int i=0;i<t;i++)
    {
    	cin>>n;
    	//cout<<n<<endl;
    	cout<<calctot(arr,n)<<endl;
    	
    }
    return 0;
}
