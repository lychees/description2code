N,M=map(int,raw_input().split())
c=map(int,raw_input().split())
c.sort()

dp=dict()

def change(n,mi):
 	if(n<0): return 0
 	if n==0: return 1
 	if (n,mi) in dp: return dp[(n,mi)]
 	su=0
 	for co in c:
 		if co>=mi:
 			su+=change(n-co,co)
 	dp[(n,mi)]=su
 	return su
