#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int main(){
   int t,n,g,me;
   scanf("%d",&t);
   while(t--){
    scanf("%d %d",&n,&g);
    int arr[n];
    scanf("%d",&me);
    arr[0]= me;
    for(int i = 1; i < n;i++){
       scanf("%d",&arr[i]);
    }
    sort(arr,arr+n);
    int m = n-1, count = 0;
    while(arr[m]!=me){
          m--;
          count++;
    }
    printf("%d\n",count/g);
   }
}
