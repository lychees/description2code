#include <iostream>
#include <list>
using namespace std;

int main()
{
    // cout << "Hello World!" << endl;
    int t;
    cin >> t;
    while(t--) {
    	int n,k, num;
    	cin >> n >> k;
    	list<int> friends;
    	for(int i = 0; i < n; i++) {
    		cin >> num;
    		friends.push_back(num);
    	}
    	
    	list<int>::iterator it;
    	for(it = friends.begin(); it != friends.end();) {
    		int temp = *friends.begin();
    		if(k == 0) break;
    		int num1 = *it++;
    		int num2 = *it;
			if(num1 >= num2) {
				continue;
			} else {
				it--;
				friends.erase(it);
				if(*it != temp) { 
					it--;
				} else {
					it = friends.begin();
				}
				k--;
			}
    	}
    	
    	while(k--) {
    		friends.pop_back();
    	}
    	
    	for(it = friends.begin(); it != friends.end(); it++) {
    		cout << *it << " ";
    	}
    	cout << endl;
    }
    return 0;
}
