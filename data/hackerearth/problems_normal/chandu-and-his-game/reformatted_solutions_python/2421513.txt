n, k = map(int, raw_input().split())
x = map(int, raw_input().split())
y = map(int, raw_input().split())
ans = 0
for i in range(k):
    s1 = abs(x[i]-1)+abs(y[i]-1)
    s2 = abs(x[i]-n)+abs(y[i]-n)
    s3 = abs(x[i]-1)+abs(y[i]-n)
    s4 = abs(y[i]-1)+abs(x[i]-n)
    ans += min(s1, s2, s3, s4)
print(ans)
