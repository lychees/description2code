#include <iostream>

using namespace std;

int main()
{
	ios_base::sync_with_stdio( false );
	string s;
	int q, i, j, ct, p;
	
	cin >> s;
	
	cin >> q;
	
	while( q-- ) {
		cin >> i >> j;
		ct = 0;
		for( p = i - 1;p < j - 1;p++ ) {
			if( s[ p ] == s[ p + 1 ] )
				ct++;
		}
		
		cout << ct << "\n";
	}
	return 0;
}
