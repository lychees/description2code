#include <bits/stdc++.h>
using namespace std;
int a[10005];
int n,hi = -1;
int solve(int m)
{
	int cur = abs(m-a[0]);
	int i;
	for(i=0;i<n;i++)
	cur = min(cur,abs(m-a[i]));
	return cur;
	
}
int main()
{
   // cout << "Hello World!" << endl;
   
   int i;
   scanf("%d",&n);
   set<int> s;
   for(i=0;i<n;i++){
   scanf("%d",&a[i]);
   s.insert(a[i]);
   }
   int p,q;
   int ans = -1;
   scanf("%d%d",&p,&q);
   s.insert(p);
   s.insert(q);
   set<int> t;
   set<int>::iterator I;
  int all[1005];
  int in =0;
  for(I=s.begin();I!=s.end();I++)
  all[in++] = *I;
  int j;
  for(i=0;i<in;i++)
  for(j=i+1;j<in;j++)
  {
  	int fir,sec;
  	fir = (all[i] + all[j])/2;
  	t.insert(fir);
  	t.insert(fir-1);
  	t.insert(fir+1);
  }
   for(I=t.begin();I!=t.end();I++)
   s.insert(*I);
   for(I=s.begin();I!=s.end();I++)
   {
   	i = *I;
   	if(i>=p && i<=q){
   	int cur = solve(i);
   //	printf("for %d it is %d\n",i,cur);
   	if(cur > hi){
   		hi = cur;
   		ans = i;
   	}
   	}
   }
   printf("%d\n",ans);
   
    return 0;
}
