for i in range(0 ,int(raw_input())):
    inputs = raw_input().split();
    n = int(inputs[0]);
    m =int(inputs[1]);
    x = int(inputs[2]);
    t= (n+1)* x;
    score = sum([int(x) for x in raw_input().split()])
    if(score >= t):
        print "1"
    elif(score+m < t):
        print "Impossible"
    else:
        print (t-score)
