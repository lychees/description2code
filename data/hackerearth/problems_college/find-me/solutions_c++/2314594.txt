#include <iostream>
using namespace std;

int main()
{
    int t,n,p;
    cin>>t;
    while(t--)
    {
    	int flag = 0;
    	cin>>n>>p;
    	int arr[n];
    	for(int i=0; i<n; i++)
    	cin>>arr[i];
    	
    	for(int i=0; i<n; i++)
    	{
    		if(arr[i] == p){
    		flag = 1;
    		break;
    		}
    	}
    	if(flag == 1)
    	{
    		cout<<"YES ";
    		int x = p;
    		int i = -1;
    		for(int j=0; j<n; j++)
    		{
    			if(arr[j]<=x)
    			{
    				i++;
    				int temp = arr[i];
    				arr[i] = arr[j];
    				arr[j] = temp;
    			}
    		}
    		int num = n - (i+1);
    		cout<<num<<"\n";
    	}
    	else
    	cout<<"NO "<<"0"<<"\n";
    }
    
    return 0;
}
