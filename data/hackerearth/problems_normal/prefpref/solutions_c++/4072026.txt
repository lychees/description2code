#include <iostream>
#include <string>
using namespace std;

int main () {
        string a,b;
        cin >> a >> b;
        int count = 0;
        for(int i = 0; i < a.length(); ++i) {
                for(int j = 0; j < b.length(); ++j) {
                        if( a[i] != b[j])
                                continue;
                        else {
                                ++count;
                                ++i;
                        }

                }
                break;
        }
        cout<<count<<endl;
}
