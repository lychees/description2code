#include <iostream>
    #include<string.h>
    using namespace std;
     
    int main()
    {
        int t;
        cin>>t;
        while(t--)
        {
        	char s[100000];
        	cin>>s;
        	long long int n=strlen(s),i,count=0;
        	for(i=1;i<n;i++)
        	{  if(s[i]=='O'&&s[i-1]=='X')
        	     s[i]='X';
        
        	}
        		for(i=n-2;i>=0;i--)
        	{  if(s[i]=='O'&&s[i+1]=='X')
        	     s[i]='X';
        
        	}
        	for(i=0;i<n;i++)
        	{
        		if(s[i]=='O')
        		count++;
        	}
           cout<<count<<endl;
        }
        return 0;
    }
