#include<bits/stdc++.h>
using namespace std;
#define pb push_back 
vector<int>adj[1009];
int vis[1009];
int dist[1009];
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;++i)
	{
		vis[i]=0;
		adj[i].clear();
		dist[i]=0;
	}
	for(int i=1;i<=n-1;++i)
	{
		int u,v;
		cin>>u>>v;
		adj[u].pb(v);
		adj[v].pb(u);	
	}
	queue<int>q;
	q.push(1);
	dist[1]=0;
	while(!q.empty())
	{
		int u=q.front();
		vis[u]=1;
		q.pop();
		for(int i=0;i<(int)adj[u].size();++i)
		{
			if(!vis[adj[u][i]])
			{
				dist[adj[u][i]]=1+dist[u];
				q.push(adj[u][i]);
			}
		}
	}
	int maxidist=*max_element(dist+1,dist+n+1);
	int ans=2*(n-1)-maxidist;
	cout<<ans<<endl;

	return  0;
}
