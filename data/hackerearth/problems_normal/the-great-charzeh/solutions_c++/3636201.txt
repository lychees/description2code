#include <bits/stdc++.h> 

using namespace std;

typedef long long ll; 
typedef pair<int, int> pii;

#define REP(i,n) for(int(i)=0;(i)<(int)(n);(i)++)

int n,m;
int a[310];
int f[610];
int ff[610];
int g[310][310];
int match[310];
int seen[310];

bool bpm(int v) {
	if (seen[v]) return false;
	seen[v]=1;
	for (int i = 0; i < n; i++) {
		if (g[v][i] && (match[i]==-1 || bpm(match[i]))) {			
			match[i]=v;
			return true;
		}
	}
	return false;
}

bool check(int goal) {
	REP(i,n) REP(j,n) {
		g[i][j] = (f[i+a[j]] <= ff[goal]);
	}
	REP(i,n) match[i]=-1;
	//REP(i,n){ REP(j,n) printf("%d ", g[i][j]); printf("\n");}
	REP(i,n) {
		memset(seen,0,sizeof(seen));
		if (!bpm(i)) {
			//printf("check %d fails\n", ff[goal]);
			return false;
		}
	}
	//printf("check %d passes\n", ff[goal]);
	return true;
}

int main() {
	scanf("%d %d", &n, &m);
	REP(i,n) scanf("%d",&a[i]);
	REP(i,n+m-1) {
		scanf("%d",&f[i]);
		ff[i]=f[i];
	}
	sort(ff,ff+(n+m-1));

	int st = 0, ed = n+m-2;
	while (st < ed) {
		int md = (st+ed)/2;
		if (check(md)) ed = md;
		else st = md+1;
	}
	printf("%d\n", ff[st]);
}
