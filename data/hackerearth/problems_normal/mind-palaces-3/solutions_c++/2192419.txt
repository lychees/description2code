#include<iostream>
#include<stdio.h>
using namespace std;
int RBS(int *a,int low,int high,int x)
{
			int mid;
			if(low>high)
					return -1;
			mid=(low+high)/2;
			if(x>a[mid])	RBS(a,mid+1,high,x);
			else if(x<a[mid]) RBS(a,low,mid-1,x);
			else return mid;
}
int main()
{
	int N,M,Q,x,i,j,pos;
	cin>>N>>M;
	int **a=new int*[N];
	for(i=0;i<N;i++)
		a[i]=new int[M];
		for(i=0;i<N;i++)
			for(j=0;j<M;j++)
					scanf("%d",&a[i][j]);
			cin>>Q;
			for(i=0;i<Q;i++)
			{
						scanf("%d",&x);
						for(j=0;j<N;j++)
						{
									pos=RBS(&a[j][0],0,M-1,x);
									if(pos!=-1)	break;
						}
						if(pos==-1)
											cout<<"-1 -1"<<endl;
						else cout<<j<<" "<<pos<<endl;
			}
	return 0;
}
