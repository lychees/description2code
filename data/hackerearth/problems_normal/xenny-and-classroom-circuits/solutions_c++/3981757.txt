#include <iostream>
#include<string>
#include<vector>

using namespace std;
    long long n,t,d1,d2;
    string str1="BG",str2="GB",str;


int main()
{
    cin>>t;
    for(long long i=1; i<1000000; i++)
    { str1 += "BG"; str2+="GB"; }
    while(t--)
    {
        cin>>n;
        cin>>str;
        d1=0; d2=0;
        for(int i=0; i<2*n; i++)
        {
          if(str[i]!=str1[i]) d1++;
          if(str[i]!=str2[i]) d2++;
        }
        //cout<<d1<<' '<<d2<<endl;
        d1/=2; d2/=2;
        cout<<min(d1,d2)<<endl;
    }

}
