#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	vector<string>v[101];
	while(n--)
	{
		string x;
		cin>>x;
		int marks;cin>>marks;
		v[marks].push_back(x);
		sort(v[marks].begin(),v[marks].end());
	}
	for(int i=100;i>=1;i--)
	{
		for(int j=0;j<v[i].size();j++)
		{
			cout<<v[i][j]<<" "<<i<<endl;
		}
	}
	return 0;
}
