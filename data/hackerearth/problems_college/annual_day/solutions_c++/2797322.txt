#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,k;
		cin>>n>>k;
		int a[n],b[n];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		for(int i=0;i<n;i++)
		{
			int ans=0;
			while(a[i]>=1)
			{
				ans=ans+a[i]%2;
				a[i]=a[i]/2;
			}
			b[i]=ans;
		}
		for(int i=0;i<n;i++)
		{
			for(int j=i+1;j<n;j++)
			{
				if(b[j]>b[i])
				{
					int temp;
					temp=b[j];
					b[j]=b[i];
					b[i]=temp;
				}
			}
		}
		int m=0;
		for(int i=0;i<k;i++)
		{
			m=m+b[i];
		}
		cout<<m<<endl;
	}
	return 0;
}
