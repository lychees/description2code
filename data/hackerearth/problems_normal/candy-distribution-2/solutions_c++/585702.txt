#include <bits/stdc++.h>
#define rep(i,s,n) for(long int i =s;i < n; ++i)
#define ll long long
const int MAX = 1e5+10;
using namespace std;

int main()
{
    int t;
    long int n;
    ll minsum = 0;
    ll inp;
    cin >> t;
    rep(i,0,t){
    	minsum =0;
    	cin >> n;
    	rep(i,0,n){
    		cin >> inp;
    		minsum+=inp;
    	}
    	minsum -= n;
    	++minsum;
    	cout << minsum << "\n";
    }
    return 0;
}
