#include <iostream>
#include<cstdio>
using namespace std;

int main()
{
    int t;
    long n,p;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%ld%ld",&n,&p);
    	while(p--)
    	{
    		if((n&1)==0)
    		n=n>>1;
    		else
    		{
    			n=n>>1;
    			n++;
    		}
    		n++;
    		if(n<4)
    		break;
    	}
    	printf("%ld\n",n);
    }
    return 0;
}
