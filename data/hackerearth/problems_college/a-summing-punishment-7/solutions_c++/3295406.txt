#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    int a[n],b[n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    	if(i==0)
    		b[i]=a[i];
    	else
    		b[i]=b[i-1]+a[i];
    }
    int q;
    cin>>q;
    
    while(q--)
    {
    	int l,r;
    	cin>>l>>r;
    	cout<<b[r-1]-b[l-2]<<endl;
    }
    return 0;
}
