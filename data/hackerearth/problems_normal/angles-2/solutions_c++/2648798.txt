#include <iostream>
using namespace std;
int main()
{
    int n,k;
    cin >> n >> k;
    int a[n],b[k];
    for(int i=0;i<n;i++)
    cin >> a[i];
    for(int i=0;i<k;i++)
    cin >> b[i];
    int i,j,m;
    for(i=0;i<k;i++)
    {
        for(j=0;j<n;j++)
        {
    		int x = a[j];
    		for(m=0;m<10000;m++){
    			if(x%360 == b[i])
    			{
    				cout << "YES" << endl; break;
    		    }
    			x= x+a[j];
    		}
    		if(m!=10000) break;
    	}
    	if(j==n) cout << "NO" << endl;
    	
    }
    
    return 0;
}
