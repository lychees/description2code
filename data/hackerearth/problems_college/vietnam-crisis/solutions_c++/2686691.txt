#include <bits/stdc++.h>
using namespace std;
#define gc getchar_unlocked
void scanlli(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
int a[20000000];
int main()
{

    while(1)
