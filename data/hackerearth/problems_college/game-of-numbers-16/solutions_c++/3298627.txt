#include<bits/stdc++.h>

using namespace std;

int check(long long int n)
{
    long int count=0;
    for(int i=1; i<=n; i++)
    {
        for(int j=1; j<i; j++)
        {
            if(j < i && 1 <=j && i <=n)
            {
                if((i*j)%(i+j)==0)
                    count++;
            }
        }
    }
    return count;
}
int main()
{
    long long int t,n,y;
    cin>>t;

    while(t--)
    {
        cin>>n;
        y=check(n);
        cout<<y<<endl;
    }
}
