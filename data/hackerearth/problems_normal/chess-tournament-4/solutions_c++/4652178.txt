#include<bits/stdc++.h>
#define N 1050
static int a[N][N];
using namespace std;
int win(int a[][1050],int na[],int k)
{   int i,j,h,z,t=1;
    for(i=1;i<k;i+=2)
    {   j=na[i+1]; h=na[i];
        if(a[j][h]==1) { na[t]=j; t++;}
        else { na[t]=h;t++;}
    }
    if(t==2) { z=na[1];return z;}
    else return win(a,na,t);

}
int main()
{
    int i,n,na[1050],l,j;
    cin>>n;
    l=pow(2,n);
    for(i=1;i<=l;i++)
        for(j=1;j<i;j++)
            if(i>j) cin>>a[i][j];
     int k=1;
    for(i=1;i<=l;i+=2)
    {  if(a[i+1][i]==1) { na[k]=i+1; k++;}
        else { na[k]=i;k++;}
    }
    int out=win(a,na,k);
    cout<<out;
    return 0;
}
