string stradd(string str1,string str2)
{
    int n1=sz(str1),n2=sz(str2);
    reverse(all(str1));
    reverse(all(str2));
    string zrs(abs(n1-n2),'0');
    int c=0;
    int n=max(n1,n2);
    if (n1>n2) swap(str1,str2);
    str1.append(zrs);
    string ans="";
    int i;
    FOR(i,0,n)
    {
        int temp=(str1[i]-'0')+(str2[i]-'0')+c;
        c=temp/10;
        ans.pb((temp%10)+'0');
    }
    if (c>0) ans.pb(c+'0');
    reverse(all(ans));
    return ans;
}

string strmult2(string str,int m)
{
    int n=sz(str);
    reverse(all(str));
    string ans="";
    int i;
    int c=0;
    FOR(i,0,n)
    {
        int temp=m*(str[i]-'0')+c;
        c=temp/10;
        ans.pb((temp%10)+'0');
    }
    if (c>0) ans.pb(c+'0');
    reverse(all(ans));
    return ans;
}

string strmult(string str1,string str2)
{
    if (str1.compare("0")==0||str2.compare("0")==0) return "0";
    int n1=sz(str1),n2=sz(str2);
    if (n1>n2) swap(str1,str2);
    int i;
    int n=min(n1,n2);
    string ans="";
    string zrs="";
    for(i=n-1;i>=0;i--)
    {
        string tans=strmult2(str2,str1[i]-'0');
        tans.append(zrs);
        ans=stradd(ans,tans);
        zrs.pb('0');
    }
    return ans;
}
