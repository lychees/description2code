#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    
    while(T--)
    {
    	int n,sump=0;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++)
    	  { cin>>a[i];
    	   sump+=a[i];
    	  }
    	  
    	 int mean=(sump/n);
    	 
    	 cout<<mean<<endl;
    	   
    }
   
    return 0;
}
