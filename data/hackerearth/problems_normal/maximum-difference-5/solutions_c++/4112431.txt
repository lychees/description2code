#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void prt_max(int a, int b)
{
	if(a>b)
	cout<<a;
	else
	cout<<b;
}
int main()
{
    //cout << "Hello World!" << endl;
    int n;
    int i;
    long long int m=0;
    int *arr;
    cin>>n;
    arr = (int *)malloc(sizeof(int)*n);
    for(i=0; i<n; i++)
    {
    	cin>>arr[i];
    }
    sort(arr, arr+n);
    vector <int> v1,v2;
    
    i=0;
    int j=n-1;
    while(i<=j)
    {
    	v1.push_back(arr[i]);
    	v2.push_back(arr[j]);
    	
    	if(i!=j)
    	{
    		v1.push_back(arr[j]);
    		v2.push_back(arr[i]);
    	}
    	i++;
    	j--;
    }
    /*for(i=0;i<n;i++)
    {
    	cout<<v1[i]<<" "<<v2[i]<<"\n";
    }*/
    int ans1, ans2;
    ans1=ans2=0;
    for(i=0;i<n-1;i++)
    {
    	ans1=ans1 + abs(v1[i]-v1[i+1]);
    	ans2=ans2 + abs(v2[i]-v2[i+1]);
    }
    prt_max(ans1, ans2);
    return 0;
}
