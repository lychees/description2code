#include<iostream>

using namespace std;
int arr[1000001] = { 0 };
int main() {

	int testCase, elements;
	cin >> testCase;
	while (testCase--)
	{
		cin >> elements;
		for (int i = 0; i < elements; i++)
		{
			cin >> arr[i];
		}
		int visited[10000] = { 0 };
		int i = 0,flag=0;
		while (visited[i]!=-3)
		{
			if (arr[i] == -1)
			{
				flag = 1; 
				break;
			}
			visited[i] = -3;
			i = arr[i];
		}
		if (flag)
			cout << "Win\n";
		else
			cout << "Lose\n";
	}
	return 0;
}
