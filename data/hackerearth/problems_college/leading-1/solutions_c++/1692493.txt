#include <iostream>
using namespace std;
int fact(int N)
{
	if(N==1)
	return 1;
	if(N==0)
	return 1;
	return N*fact(N-1);
}	
int main()
{
    int T;
    cin>>T;
    int N,M;
    while(T--)
    {
    	cin>>N>>M;
    	//int ncr=fact(N)/fact(N-1);
    	//int mcr=fact(M)/fact(M-1);
    	//cout<<((ncr*mcr)%116)<<"\n";
    	int ncr= (fact(M+N-1)) / ((fact(M-1)) * fact(N));
    	cout<<ncr<<"\n";
    }
    return 0;
}
