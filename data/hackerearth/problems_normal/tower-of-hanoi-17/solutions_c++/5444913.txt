#include<bits/stdc++.h>
#define mp make_pair
#define endl '\n'
#define pi pair<int,int>
#define int long long
using namespace std;
vector<pi> v;
int dp[100005];
 main()
{
    int t;
    cin>>t;
    while(t--)
    {
    int n;
    cin>>n;
    v.clear();
    for(int i=0;i<=n;++i)dp[i]=0;
     for(int i=0;i<n;i++)
     {
         int r,h;
         cin>>r>>h;
         v.push_back(mp(r,h));
     }
      sort(v.begin(),v.end());
      v.push_back(v[n-1]);


      for(int i=n-1;i>=0;i--)
      {
          int j=i+1;
          int anss=0;
          for( ;j<n;++j)if(v[j].first>v[i].first&&v[j].second>v[i].second)anss=max(anss,dp[j]);
          dp[i]=v[i].second+anss;
      }
      int ans=0;
      for(int i=0;i<n;++i)ans=max(ans,dp[i]);
      cout<<ans<<endl;
    }
    return 0;
}
