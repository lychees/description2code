#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const int N=1e5+10;

int a[N];
int temp[N];

int main() {
  //freopen("a.cpp","r",stdin);
  int tt;
  scanf("%d",&tt);
  while(tt--) {
    int l,p;
    scanf("%d%d",&l,&p);
    string s;
    cin>>s;
    for(int i=0;i<l;i++) {
      a[i+1]=s[i]-'0';
    }
    for(int i=1;i<=l;i++) {
      temp[i]=a[i];
      if(i>1) {
        temp[i]+=temp[i-1];
      }
    }
    ll total=1ll*l*l;
    ll ans=0;
    for(int i=1;i<=l;i++) {
      int left=i-p;
      int right=i+p;
      if(a[i]==1) {
        int flag=0;
        if(left<1) {
          left=1;
        }
        if(left>=1) {
          ans+=temp[i]-temp[left-1];
          flag++;
        }
        if(right>l) {
          right=l;
        }
        if(right<=l) {
          ans+=temp[right]-temp[i-1];
          flag++;
        }
        if(flag==2) {
          ans-=1;
        }
      }
    }
    ll res=__gcd(ans,total);
    printf("%d/%d\012",ans/res,total/res);
  }
  return 0;
}
