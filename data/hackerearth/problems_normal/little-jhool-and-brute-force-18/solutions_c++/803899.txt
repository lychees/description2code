long long power(int a, int b)
{
    if( b == 0 )
        return 1;
    else
        return a*power(a,b-1);
}

vector<ll>v;

int main()
{
   //freopen("input.txt","r",stdin);
   //freopen("output.txt","w",stdout);
   int t;
   cin>>t;

   while(t--)
   {
       v.clear();
       ll n;
       sll(n);

       ll i,j;
       for(i=1;i*i*i<n;i++)
       {
           for(j=i;j*j*j<n;j++)
           {
               ll x=i*i*i+j*j*j;
               //if(x==91) cout<<i<<" "<<j<<"\n";
               if(x<n) v.push_back(x);
           }
       }

       sort(v.begin(),v.end());

       int flag=0;
       if(n==1 || n==2) {printf("-1\n"); continue; }
       //cout<<v.size();

       for(i=v.size()-1;i>0;i--)
       {
           if(v[i]==v[i-1]) {printf("%lld\n",v[i]);flag=1; break;}
       }

       if(!flag) printf("-1\n");
   }


 return 0;
}
