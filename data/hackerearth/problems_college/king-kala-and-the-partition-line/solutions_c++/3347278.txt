#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    long long int a,b,c,i,j,k,ctr,cnt,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cin>>a>>b>>c;
    	cnt=0;
    	ctr=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>j>>k;
    		if((a*j)+(b*k)+c>0)
    		ctr++;
    		else if((a*j)+(b*k)+c<0)
    		cnt++;
    	}
    	if(ctr>cnt)
    	cout<<ctr<<endl;
    	else
    	cout<<cnt<<endl;
    	
    }
    return 0;
}
