#include <iostream>
using namespace std;
long long int m=1000000007;
long long int modpowIter(long long int a,long long  int b,long long int c) {
        long long int ans=1;
        while(b != 0) {
                if(b%2 == 1)
                        ans=(ans*a)%c;

                a=(a*a)%c;
                b /= 2;
        }
        return ans;
}
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   long long int n;
   	cin >> n;
   	
   	cout << modpowIter(modpowIter(2,n-1,m),m-2,m)<<endl;
   }
    return 0;
}
