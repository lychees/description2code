#include<iostream>
using namespace std;

int main()
{

    int dp[100];

    for(int i=1;i<=7;i++)
        dp[i]=-1;
    dp[6]=1;
    dp[8]=1;

    for(int i=9;i<=100;i++)
    {
            int a=i-8;
            int b=i-6;

            if(dp[a]==-1 && dp[b]==-1)
                dp[i]=-1;
            else if(dp[a]!=-1 && dp[b]!=-1)
                dp[i]=1+min(dp[a],dp[b]);
            else
            {
                if(dp[a]==-1)
                    dp[i]=dp[b]+1;
                else
                    dp[i]=dp[a]+1;
            }
    }

    int t;
    cin>>t;

    while(t--)
    {
        int n;
        cin>>n;

        cout<<dp[n]<<endl;
    }
}
