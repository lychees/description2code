#include <bits/stdc++.h>
typedef long long ll;
#define N 100001
static ll ih[N],fh[N];
using namespace std;
int main()
{	int t;
	ll n,x,y,i;
	cin>>t;
	while(t--){
		cin>>n>>x>>y;
		ll ef=0,mef=0;
		 
		for(i=1;i<=n;i++)
		      cin>>ih[i]>>fh[i];
		sort(ih+1,ih+1+n);
		sort(fh+1,fh+n+1);
		for(i=1;i<=n;i++)
			{ if(ih[i]-fh[i]>0) ef+=abs(ih[i]-fh[i])*y;
			  else ef+=abs(ih[i]-fh[i])*x;
			}
			cout<<ef<<endl;
	}
   // cout << "Hello World!" << endl;
    return 0;
}
