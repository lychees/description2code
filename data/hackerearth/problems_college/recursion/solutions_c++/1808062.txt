#include<bits/stdc++.h>
using namespace std;
void printPermutation(int a[],int b[],int n,int digitno)
{
	int i,j;
	if(digitno==n+1)
	{
		for(i=1;i<=n;i++)
		printf("%d",b[i]);
		printf(" ");
	}
	else
	{
		for(i=1;i<=n;i++)
		{
			if(a[i]==0) // has not been used yet
			{
				a[i]=1;
				b[digitno]=i;
		     	printPermutation(a,b,n,digitno+1);
		     	a[i]=0;
		        
		}
		}
	}
}
int main()
{
	int t,n,i,j;
	scanf("%d",&t);
	while(t--)
	{
		int a[10],b[10];
		memset(a,0,10*sizeof(int));
		scanf("%d",&n);
		printPermutation(a,b,n,1);
		printf("\n");
	    
	}
return 0;
}
