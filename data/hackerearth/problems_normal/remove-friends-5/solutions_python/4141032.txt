tc = int(raw_input())
for t in range(tc):
    n, k = map(int, raw_input().split())
    ary = map(int, raw_input().split())
    tmp = ary[0]
    stack = []
    stack.append(ary[0])
    i=1
    while (k > 0):
        if len(stack) > 0 and stack[len(stack)-1] < ary[i] :
            stack.pop(len(stack)-1)
            k=k-1
        else:
            stack.append(ary[i])
            i+=1

    print " ".join(map(str, stack+ary[i:]))
