using namespace std;
typedef long long ll;
typedef long double ld;

const int N = 1e6+6;

int g[N];

void solve(){
	int n,m;
	scanf("%d%d", &n, &m);
	for(int i=0;i<n;++i) g[i]=0;
	for(int k=0;k<m;++k){
		int i,j;
		scanf("%d%d", &i, &j);
		--i;--j;
		g[i]|=1<<j;
		g[j]|=1<<i;
	}
	int ans = 0;
	for(int ms=1;ms<(1<<n);++ms){
		bool ok = true;
		for(int i=0;i<n;++i) if(ms>>i&1){
			ok&=(ms&g[i])==0;
		}
		if(ok) ++ans;
	}
	printf("%d\n",ans);
}

int main(){
	//freopen("input.txt","r",stdin);// freopen("output.txt","w",stdout);
	
	int t;
	scanf("%d", &t);
	while(t--) solve();
	
	
	return 0;
}
