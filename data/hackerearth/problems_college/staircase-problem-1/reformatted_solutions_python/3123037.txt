def getSteps(n):
    if n == 0:
        return 0
    if (n == 1 or n == 2):
        return 1
    c = 0
    currentStep = 0
    i = 2
    while (currentStep < n):
        c = c+1
        currentStep = currentStep+i
        i = i+1
    return c


t = int(raw_input())
for i in range(0, t):
    n = raw_input()
    print(getSteps(int(n)))
