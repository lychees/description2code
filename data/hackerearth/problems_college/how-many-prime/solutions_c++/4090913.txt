#include <iostream>
#include <algorithm>
using namespace std;
bool isprime(unsigned long long n)
{
	if(n==1)
	return false;
	if((n==2)||(n==3))
	return true;
	for(unsigned long long i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
		return false;
	}
	return true;
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	unsigned long long n;
    	cin>>n;
    	unsigned long long val1;
    	for(val1=n/2;val1>=2;val1--)
    	{
    		if(val1!=n-val1)
    		{
    		if((isprime(val1))&&(isprime(n-val1)))
    		{
    			cout<<val1<<" "<<n-val1<<endl;
    			break;
    		}
    		}
    	}
    	if(val1<2)
    	cout<<-1<<" "<<-1<<endl;
    }
    return 0;
}
