#include<iostream>
#include<algorithm>
using namespace std;
int a[1000000][3],m[1000000][3];
int main()
{
	int t,n;
	cin>>t;
	while(t--){
		cin>>n;
		for(int i=0;i<n;i++){
			for(int j=0;j<3;j++)
			cin>>a[i][j];
		}
		for(int j=0;j<3;j++) m[0][j]=a[0][j];
		for(int i=1;i<n;i++){
			for(int j=0;j<3;j++){
				if(j==0)
				m[i][0]=min(m[i-1][1],m[i-1][2])+a[i][0];
				else if(j==1)
				m[i][1]=min(m[i-1][0],m[i-1][2])+a[i][1];
				else
				m[i][2]=min(m[i-1][0],m[i-1][1])+a[i][2];
			}
		}
		cout<<min(m[n-1][0],min(m[n-1][1],m[n-1][2]))<<endl;
	}
}
