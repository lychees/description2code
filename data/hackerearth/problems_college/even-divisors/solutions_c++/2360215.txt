#include<bits/stdc++.h>
using namespace std;
int main()
{
    int k;
    cin>>k;
    for(int j=0;j<k;j++)
    {
    long long int n;
    cin>>n;
    int c=0;
    if(n%2!=0)cout<<0<<endl;
    else
    {
    for(int i=2;i<=sqrt(n);i++)
    {
    int x=n/i;
        if(n%i==0&&i%2==0&&x%2==0&&i!=x&&n%i==0)c=c+2;
        if(n%i==0&&i%2!=0&&x%2==0&&n%x==0)c=c+1;
        if(n%i==0&&i%2==0&&x%2!=0)c=c+1;
 
        if(n%i==0&&n%2==0&&x%2==0&&i==x&&n%i==0)c=c+1;
 
 
    }
    cout<<c+1<<endl;
    }
    }
    return 0;
}
