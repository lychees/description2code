# T no of transactions in day
T = int(raw_input())
Tworth = list()

# Accepting Ai
x = raw_input()
Tworth = x.split()

Tworth = [int(i) for i in Tworth]

total = 0
arr = list()
for i in Tworth:
    total = total+i
    arr.append(total)

# print "Tworth: "+str(Tworth)

# Q no of queries
# Accepting queries in target list
Q = int(raw_input())
target = list()
for i in range(Q):
    target.append(int(raw_input()))

# Logic
ans = list()
for tx in target:
    low = 0
    high = T-1
    if tx > arr[T-1]:
        ans.append(-1)
        continue
    while low <= high:
        mid = low+(high-low)/2
        # print "low:"+str(low)+" mid:"+str(mid)+" high:"+str(high)+"
        # arr[mid]:"+str(arr[mid])
        if arr[mid] > tx:
            high = mid
        if arr[mid] < tx:
            low = mid+1
        if arr[mid] == tx or (low == mid and mid == high):
            ans.append(mid+1)
            break

for i in ans:
    print i
