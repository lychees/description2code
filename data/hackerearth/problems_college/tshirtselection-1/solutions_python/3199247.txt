from sys import stdin


def longest_color(S):
    occurences = {}
    cur_longest_length = 0
    longest_pos = 0
    cur_start = 0
    cur_length = 0

    for k, ch in enumerate(S):
        last_o = occurences.get(ch, -1)
        if last_o < cur_start:
            cur_length += 1
        else:
            if cur_length > cur_longest_length:
                longest_pos = cur_start
                cur_longest_length = cur_length
            cur_length -= last_o - cur_start
            cur_start = last_o + 1
        occurences[ch] = k

    if cur_length > cur_longest_length:
        longest_pos = cur_start
        cur_longest_length = cur_length

    return longest_pos, cur_longest_length

in_list = [line.rstrip().split() for line in stdin.readlines()]
t = int(in_list.pop(0)[0])

while t:
    in_list.pop(0)
    x, y = longest_color(in_list.pop(0))
    print str(x + 1) + ' ' + str(x + y)
    t -= 1
