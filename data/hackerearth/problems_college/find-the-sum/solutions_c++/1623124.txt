#include<bits/stdc++.h>
using namespace std;
long long f(long long n)
{
    long long c=0,i=0,ans=0;
    while(n>0)
    {
        i++;
        if(n-i<0)
            break;
        n-=i;
        c++;
    }
     ans=(((c+1)/2)*((c+1)/2));
   //  cout<<ans<<endl;
    if(n % 2)
    ans+=(c+1);
    return ans;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long l,r;
        cin>>l>>r;
        cout<<f(r)-f(l-1)<<endl;
    }
    return 0;
}
