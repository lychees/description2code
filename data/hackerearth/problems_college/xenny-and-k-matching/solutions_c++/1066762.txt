#include <iostream>
using namespace std;

int main()
{
	int n,k,i,a,b,s;
    scanf("%d%d",&n,&k);
    int arr[n+1];
    for(i=1;i<=n;i++)
    {
    	scanf("%d",&arr[i]);
    }
    s=0;
   
    for(i=1;i<=n-k;i++)
    {
    	a=arr[i];
    	b=arr[i+k];
    	if((a+b)%k==0||(a-b)%k==0||(b-a)%k==0)
    	s++;
    }
    printf("%d\n",s);
}
