k = input()
s = raw_input().strip()
length = len(s)
if k > length:
    print 0
else:
    counter = 0
    for i in xrange(length-k+1):
        letters = {}
        cnt = 0
        for j in xrange(i, length):
            if letters.has_key(s[j]):
                pass
            else:
                cnt += 1
                letters[s[j]] = 1
            if cnt > k:
                break
            if cnt == k:
                counter += 1

    print counter
