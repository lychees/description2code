#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int l,r;
    	cin>>l>>r;
    	int max=0;
    	for(int i=l;i<=r;i++)
    	{
    		for(int j=i+1;j<=r;j++)
    		{
    			int x = i^j;
    			if(x>max)
    				max=x;
    		}
    	}
    	cout<<max<<endl;
    }
    return 0;
}
