#include<cstdio>
int a, l, i;

int f(int x)
{
  int wynik = 0;
  while(x > 0)
  {
    if (x % 10 == 4 || x % 10 == 7)
    {
      wynik++;
    }
    x /= 10;
  }
  return wynik;
}

bool spr(int b)
{
  for(int i = 0; i < l; i++)
  {
    if (f(a + i) != f(b + i))
    {
      return false;
    }
  }
  return true;
}

int main()
{
  scanf("%d%d", &a, &l);  
  
  for(i = a + 1; ; i++)
  {
    if (spr(i) == true)
    {
      printf("%d\n", i);
      return 0;
    }
  }  
  
return 0;
}
