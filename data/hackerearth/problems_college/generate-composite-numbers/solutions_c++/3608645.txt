#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int n;
    cin>>n;
    int a[n+2];
    memset(a , 0 , sizeof(a));
    for(long long int i = 2 ; i <= n ; i++)
    {
    	if(a[i] == 0)
    	{
    		for(long long int j = i+i ; j <= n ; j += i)
    			a[j] = 1;
    	}
    	if(a[i] == 1) 
    		cout<<i<<" ";
    }
    return 0;
}
