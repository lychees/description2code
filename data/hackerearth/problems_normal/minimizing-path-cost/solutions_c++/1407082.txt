#include <iostream>
#include<string.h>
#include<cstdio>
#include<algorithm>
#include<map>
using namespace std;
string a[101],y,z;
map<string,int> b;
int d[101][101];
int main()
{
    int n,m,l,q,i,j,k;
    cin>>n>>m;
    for(i=0;i<n;++i)
    {
    	cin>>a[i];
    	b[a[i]]=i;
    }
    for(i=0;i<n;++i)
    for(j=0;j<n;++j)
    d[i][j]=10000000;
    while(m--)
    {
    	cin>>y>>z>>k;
    	d[b[y]][b[z]] =k;
    	d[b[z]][b[y]]=k;
    }
    for(k=0;k<n;++k)
    {
    	for(i=0;i<n;++i)
    	{
    		for(j=0;j<n;++j)
    		{
    			if(d[i][k]+d[k][j]<d[i][j])
    			d[i][j]=d[i][k]+d[k][j];
    		}
    	}
    }
    cin>>q;
    while(q--)
    {
    	cin>>y>>z;
    	cout<<d[b[y]][b[z]]<<endl;
    }
    
    return 0;
}
