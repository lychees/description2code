using namespace std;

ll d[10000001];

int main(){
	
	for(int i=1;i<10000001;i++) d[i]=1;
	
	for(int i=2;i<10000001;i++){
		if(d[i]==1){
			for(int j=i;j<10000001;j+=i){
				d[j]*=2;
			}
		}
	}
	
	int n;
	sd(n);
	
	ll ans=0;
	
	while(n--){
		int x;
		sd(x);
		ans+=d[x];
	}
	plld(ans);
}
