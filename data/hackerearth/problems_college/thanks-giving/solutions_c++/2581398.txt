#include <iostream>
#include <math.h>
#include <cmath>
#include <string.h>
#include <algorithm>
#include <functional>
int a[26];
using namespace std;
int main()
{
  string s;
  cin>>s;
  for(int j=0;j<s.length();j++)
    a[s[j]-'a']++;
    int max=a[0],index=0;
  for(int j=0;j<26;j++){
    if(a[j]>max)
    {
        max=a[j];
        index=j;
    }
  }
  char k=index+97;
  int count1=0;
  int count2=0;
  string s1="headrush";
  string s2="cubingclub";
  for(int j=0;j<s1.length();j++)
  {if(k==s1[j])
  {
      count1++;
      break;
  }}
  for(int j=0;j<s2.length();j++)
   {if(k==s2[j])
  {
      count2++;
      break;
  }}
  if(count1==1&&count2==1)
    cout<<"Thanks both"<<endl;
  else if(count1==1)
    cout<<"Thanks head rush"<<endl;
  else if(count2==1)
    cout<<"Thanks cubing club"<<endl;
  else
    cout<<"none"<<endl;
}
