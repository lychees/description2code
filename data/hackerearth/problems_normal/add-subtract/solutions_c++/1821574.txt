#include<stdio.h>
#include<algorithm>
#include<vector>
using namespace std;
int main(){
 int t,n,k,i,j,cost,temp;
 vector<int> a,b;
 scanf("%d",&t);
 while(t--){
   scanf("%d%d",&n,&k);
   for(i=0;i<n;++i){
     scanf("%d",&temp);
     a.push_back(temp);
   }
   cost=-1;
   sort(a.begin(),a.end());
   for(i=0;i<n;++i){
     if(i==0 || a[i]!=a[i-1]){
       b.clear();
       for(j=0;j<n;++j){
          b.push_back(a[i]-a[j]);
          if(b[j]>=0) b[j]*=3;
          else b[j]*=(-5);
       }
       sort(b.begin(),b.end());
       temp=0;
       for(j=0;j<k;++j)
        temp+=b[j];
       if(cost==-1 || cost>temp) cost=temp;
     }
   }
   printf("%d\n",cost);
   a.clear();
 }
 return 0;}
