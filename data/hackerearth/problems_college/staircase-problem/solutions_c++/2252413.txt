#include <iostream>
using namespace std;

int a[100005];

int main()
{
    a[1]=1;
    a[2]=1;
    
    int i=3;
    int count=2;
    while(i<100002)
    {
    	for(int j=i;j<i+i+1 && j<100002;j++)
    		a[j]=count;
    	
    	i+=count+1;
    	count++;
    }
    
	int t;
	
	cin >> t;
	while(t--)
	{
		int n;
		cin >> n;
		cout << a[n] << '\n';
	}
	
    
    return 0;
}
