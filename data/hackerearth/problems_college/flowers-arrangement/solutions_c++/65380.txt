#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	string str;
    	cin >> str;
    	int l = str.size();
    	int i;
    	for(i=0;i<l;) {
    		if(i+2<l && str[i] == 'R' && str[i+1] == 'Y' && str[i+2] == 'Y') i+=3;
    		else if(i+1 < l && str[i] == 'R' && str[i+1] == 'Y') i+=2;
    		else if(i < l && str[i] == 'R') i++;
    		else break;
    	}
    	if(i < l) cout << "NO\n";
    	else cout << "YES\n";
    }
}
