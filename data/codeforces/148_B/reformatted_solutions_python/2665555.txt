import sys
vp = input()
vd = input()
t = input()
f = input()
c = input()

if (vp >= vd):
    print 0

dist = 0
ans = 0
while (dist < c):
    ans += 1
    # Next Encounter
    X = (float(t * vp) / (vd - vp))
