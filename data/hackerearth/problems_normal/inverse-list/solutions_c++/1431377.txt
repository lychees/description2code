#include <iostream>
using namespace std;

int main()
{
   int i,n,a[100],b[1000],t;
   cin>>t;
   while(t--)
   {
   cin>>n;
   for(i=1;i<=n;i++)
   cin>>a[i];
   for(i=1;i<=n;i++)
   b[a[i]]=i;
   for(i=1;i<=n;i++)
   {
   	if(a[i]!=b[i])
   	{
   		cout<<"not inverse"<<endl;
   		break;
   	}
   	else if(i==n && a[i]==a[n])
   	cout<<"inverse"<<endl;
   }
   }
    return 0;
}
