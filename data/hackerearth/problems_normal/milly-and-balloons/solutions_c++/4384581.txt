

    #include <iostream>
    #include <cstdio>
    #include <assert.h>
    #include <cstring>
    using namespace std;
    int main() 
    {
        int t;
        cin >> t;
        while(t--) {
            int n, m, l, r, a, b, arr[1001], cnt, flag;
            scanf("%d%d", &n, &m);
            scanf("%d%d", &a, &b);
            assert(0 <= m && m <= 1000);
            assert(0 <= a && a <= b && b <= m);
            cnt=flag=0;
            memset(arr, 0, sizeof(arr));
            for(int i=a; i<=b; i++) arr[i]++;
            for(int i=0; i<n; i++) {
                scanf("%d%d", &l, &r);
                assert(0 <= l && l <= r && r <= m);
                for(int j=l; j<=r; j++) arr[j]++;
            }
            for(int i=a; i<=b; i++) {
                if(arr[i]==1) cnt++;
            }
            printf("%d\n", cnt);
        }
        return 0;
    }
