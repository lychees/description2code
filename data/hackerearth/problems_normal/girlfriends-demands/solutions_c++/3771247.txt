#include<iostream>
using namespace std;
#include<string.h>
#define max 100000
int main()
{
	char S[max];
	long int Q,i;
	long long int a,b,l,l1,l2;
	cin>>S;
	l= strlen(S);
	cin>>Q;
	for(i=0;i<Q;i++)
	{
		cin>>a>>b;
		l1= a%l;
		l2= b%l;
		if(l1==0)
			l1= l;
		if(l2==0)
			l2= l;
		if(S[ l1-1 ] == S[ l2-1] )
			cout<<"Yes\n";
		else
			cout<<"No\n";
	}
	return 0;
}
