#include <iostream>
using namespace std;
int gcd(int a,int b){
	int max=a>b?a:b;
	int min=a>b?b:a;
	while(max%min!=0){
		int temp=max;
		max=min;
		min=temp%min;
	}
	return min;
}
int main()
{
	int n;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	long count=0;
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			if(j>i){
				if(a[i]>a[j]){
					if(gcd(a[i],a[j])==1)
					count++;
				}
			}
		}
	}
	cout<<count<<endl;
    //cout << "Hello World!" << endl;
    return 0;
}
