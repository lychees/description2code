#include <iostream>
using namespace std;

int main()
{
     int t,n,m,i,j,k,l;
     cin>>t;
     while(t--)
     {
     	int ans=0;
     	cin>>n>>m;
     	char a[n+1][m+1];
     	for(i=0;i<n;i++)
     	 for(j=0;j<m;j++)
     	 cin>>a[i][j];

  for(int i = n - 1; i >= 0; i--){
            for(int j = m - 1; j >= 0; j--){

                if(a[i][j] == '0'){                   // check if this cell equals 0

                    ans++;                              // increase the number of moves

                    for(int k = 0; k <= i; k++){        // flip from this cell to the start point
                        for(int h = 0; h <= j; h++){

                            if(a[k][h] == '1') a[k][h] = '0';       // flip the cell
                            else a[k][h] = '1';
                        }
                    }
                }
            }  
     	  }
     	  cout<<ans<<endl;
     }
    return 0;
}
