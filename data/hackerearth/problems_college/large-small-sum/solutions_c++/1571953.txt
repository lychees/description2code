#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int a[1001],n,i,j,l,r;
        scanf("%d",&n);
        for(i=0;i<n;i++)
            scanf("%d",&a[i]);
        sort(a,a+n);
        scanf("%d %d",&i,&j);
        int ans=a[n-i]+a[j-1];
        printf("%d\n",ans);
    }
}
