#include <bits/stdc++.h>
using namespace std;

map <long long int, long long int> dp;
map <long long int, long long int> :: iterator it;

int main()
{
	long long int t,n,l,r,i,ma,prev;
    cin >> t;
    
    while (t--) {
    	dp.clear();
    	cin >> n;
    	
    	for (i = 1; i <= n; i++) {
    		cin >> l >> r;
    		dp[l]++;
    		dp[r]--;
    	}
    	
    	it = dp.begin();
    	prev = it->second;
    	it++;
    	ma = prev;
    	
    	while (it != dp.end()) {
    		dp[it->first] += prev;
    		prev = dp[it->first];
    		it++;
    		ma = max(ma,prev);
    	}
    	
    	cout <<ma<<endl;
    	
    }
    
    return 0;
}
