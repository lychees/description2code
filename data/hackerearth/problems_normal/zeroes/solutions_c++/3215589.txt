#include <stdio.h>
#include <iostream>
#include <vector>
#include <limits.h>

using namespace std;


int main() {
	
	int n, q, base;
	scanf("%d", &n);

	vector <long long > a;
	a.clear();
	a.resize(n);
	vector <long long > count;

	count.clear();
	count.resize(100005);

	for (int i = 0; i < n; i++)
		scanf("%lld", &a[i]);

	scanf("%d", &q);
	int i = 0;
	while (i < n) {
		long long z = a[i];
		i++;
			while (z % 2 == 0) {
				count[2]++;
				z = z >> 1;
			}

			for (int j = 3; j*j <= z; j = j + 2) {
				while (z % j == 0) {
					count[j]++;
					z = z / j;
				}
			}
			if (z > 2)
				count[z]++;
		
	}

	while (q--) {
		scanf("%d", &base);

		long long m = INT_MAX;
		long c = 0;

			while (base % 2 == 0) {
				base = base >> 1;
				c++;
			}

			if (c != 0)
			m = min (count[2] / c, m);

			for (int j = 3; j*j <= base; j = j + 2) {
				c = 0;
				while (base % j == 0) {
					c++;
					base = base / j;
				}
				if (c != 0)
					m = min (count[j]/c, m);
			}

			if (base > 2)
				m = min (count[base], m);

			printf("%lld\n", m);
		}
	return 0;
	}
