#include <bits/stdc++.h>

using namespace std;

int main()
{
	char s1[100001],s2[100001];
	int t,s,a;
	scanf("%s",s1);
	scanf("%d",&t);
	s=0;
	a=0;
	while(t--){
		scanf("%s",s2);
		
		if (strlen(s1)==strlen(s2) && s1[0]==s2[0]) {
			s++;
		} else if (strlen(s1)==strlen(s2) && s1[0]==(s2[0]-3)) {
			a++;
		}
	}
	printf("%d %d\n",s,a);
	
	return 0;
}
