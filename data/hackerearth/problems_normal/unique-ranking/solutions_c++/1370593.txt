#include<bits/stdc++.h>
using namespace std;
int a[1010][25];
int main(){
		#ifndef ONLINE_JUDGE
		freopen("input.txt","r",stdin);
	#endif
	int n,k,q;
	cin>>n>>k;
	for(int i=0;i<n;i++)
		for(int j=0;j<k;j++)
			cin>>a[i][j];
	cin>>q;
	int row;
	while(q--){
		int k,p;
		cin>>k;
		int temp=0;
		bool flag[1050];
		memset(flag,0,sizeof(flag));
		for(int i=1;i<=k;i++)
		{
			
			cin>>p;
			 row=0;int minn=10000000;
			int ct=1;

			for(int j=0;j<n;j++)
			{
			if(!flag[j]){
				if(a[j][p-1]==minn)
					ct++;
				if(a[j][p-1]<minn)
				{	minn=a[j][p-1]; ct=1;row=j;}
				
			}
			}
			
			if(ct==1)
			{
				for(int l=i+1;l<=k;l++)
					cin>>p;
				break;
			}
			else{
			for(int j=0;j<n;j++){
				if(a[j][p-1]!=minn){flag[j]=1;}
			}
		}
			
		}
		cout<<row+1<<endl;
	}
}
