#include <bits/stdc++.h>
using namespace std;

int main()
{
    int test,n,s,r;
    map<int,int> hmap;
    cin >> test;
    while(test--)
   	{
   		
   		cin >> n;
   		int counter=0,maxi=-1;
   		for(int i=0;i<n;i++)
   		{
   			cin >> s >> r;
   			if(hmap.find(s)==hmap.end())
   			{
   				hmap[s]=r;
   				if(hmap[s]==0)
   				{
   					maxi = max(maxi,counter);
   					counter=0;
   				}
   				else
   				++counter;
   			}
   			else
   			{
   				if(r==1)
   				{
   					if(hmap[s] == 0)
   					{
   						++counter;
   						hmap[s]=1;
   					}
   				}
   				else
   				{
   					maxi = max(maxi,counter);
   					counter=0;
   				}
   			}
   		}
   		if(counter!=0)
   			maxi = max(maxi,counter);
   			cout << maxi <<endl;
   		hmap.clear();
    }
    return 0;
}
