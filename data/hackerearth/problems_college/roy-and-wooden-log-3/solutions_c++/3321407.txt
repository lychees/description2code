#include <iostream>
#define ll long long 
using namespace std;

int main()
{
    ll t,min,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(n==1)
    	cout<<"1 0\n";
    	else
    	{
    		min=n/2-1;
    		if(n%2!=0)
    		min++;
    		if(n%2!=0)
    		n--;
    			cout<<n/2+1<<" "<<min<<endl;
    	}
    }
    return 0;
}
