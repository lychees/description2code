#include <iostream>
#define ll long long int
using namespace std;

ll solve (ll n){
	ll ans=(n*(n+1))/2;
	ans*=2;
	return ans;
}
ll pre[1000005];
int main()
{
    long long int t,n,i;
    cin>>t;
    pre[1]=0;
    pre[2]=2;
    for(i=3;i<1000002;i++){
    	pre[i]=pre[i-1]+solve(i-1);
    }
    while(t--){
    	cin>>n;
    	cout<<pre[n]<<endl;
    }
    return 0;
}
