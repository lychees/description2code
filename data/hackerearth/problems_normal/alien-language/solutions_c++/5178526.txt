#include <iostream>
#include <cassert>
#include <string>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	string s1, s2;
    	cin >> s1 >> s2;
    	
    	assert(s1.length() <= 100000);
    	assert(s2.length() <= 100000);
    	
    	int arr1[26] = {0}, arr2[26] = {0};
    	for(int i = 0; i < s1.length(); i++)
    	{
    		arr1[s1[i] - 'a'] = 1;
    	}
    	for(int i = 0; i < s2.length(); i++)
    	{
    		arr2[s2[i] - 'a'] = 1;
    	}
    	bool flag = false;
    	
    	for(int i = 0; i < 26; i++)
    	{
    		if(arr1[i] && arr2[i])
    		{
    			cout << "YES" << endl;
    			flag = true;
    			break;
    		}
    	}
    	if(flag == false)
    		cout << "NO" << endl;
    
    }
    
    return 0;
}
