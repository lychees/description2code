#include<bits/stdc++.h>
using namespace std;

#define ll long long

int main(){
	ll n; cin >> n;
	ll i; double a[n];
	for(i=0;i<n;i++)
		cin >> a[i];
	ll x,y; cin >> x >> y;
	ll d = x*y;
	ll sum=0;
	for(i=0;i<n;i++){
		ll num = a[i];
		ll sum2 = 0;
		while(num){
			ll rem = num%10;
			sum2+= rem;
			num/=10;
		}
		double cost = sum2*a[i]*d;
		if(a[i]>=1 && a[i]<=10)
			cost = cost*0.75;
		else if(a[i]>=11 && a[i]<=20)
			cost = cost*0.70;
		else
			cost = cost*0.80;
		sum+= int(cost);
	}
	cout << sum << endl;
	return 0;
}
