#include <bits/stdc++.h>
using namespace std;
int x[101],y[101];
int table[101][10001];
int n,p,t;
void napsack(){
	int i,j;
	for(i=0;i<=n;i++){
		for(j=0;j<=p;j++){
			if(j==0||i==0){
				table[i][j]=0;
			}
			else if(y[i-1]<=j){
				table[i][j]=max(x[i-1]+table[i-1][j-y[i-1]],table[i-1][j]);
			}
			else{
				table[i][j]=table[i-1][j];
			}
		}
	}
	for(i=p;i>=0;i--){
		if(table[n][i]!=table[n][p]){
			break;
		}
	}
	cout<<i+1<<" "<<table[n][p]+p<<"\n";
}
int main(){
	cin>>t;
	while(t--){
		cin>>n>>p;
		for(int i=0;i<n;i++){
			cin>>x[i]>>y[i];
			x[i]=x[i]-y[i];
		}
		napsack();
	}
}
