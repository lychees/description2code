tc=int(raw_input())
for case in range(tc):
	a,b=map(int,raw_input().split())
	c=0
	d=10
	count=0
	while a>0 or b>0:
		if (a%d+b%d+c)>=10:
			c=1
			count=count+1
		else:
			c=0
		a=a/10
		b=b/10
	if count==0:
		print "No carry operation"
	elif count==1:
		print "1 carry operation"
	else:
		print count,"carry operations"
