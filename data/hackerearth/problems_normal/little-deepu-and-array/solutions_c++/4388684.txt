#include <bits/stdc++.h>
#include <unordered_set>
#pragma comment(linker, "/STACK:256000000")
using namespace std;
typedef long long ll;
typedef long double ld;
constexpr int MOD = 1000 * 1000 * 1000 + 7;

int n;
vector<int> a;

int main()
{
	ios::sync_with_stdio(false); cin.tie(0);
	cin >> n;
	a.resize(n);
	for (int i = 0; i < n; ++i)
	{
		cin >> a[i];
	}

	int m;
	cin >> m;

	for (int i = 0; i < m; ++i)
	{
		int h;
		cin >> h;
		for (int j = 0; j < n; ++j)
		{
			if (h < a[j]) --a[j];
		}
	}

	for (int i = 0; i < n; ++i)
	{
		cout << a[i] << ((i == n - 1) ? '\n' : ' ');
	}
	return 0;
}
