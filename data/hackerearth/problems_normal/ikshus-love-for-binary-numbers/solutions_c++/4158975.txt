#include<bits/stdc++.h>
#define ll long long
ll gcd(ll a,ll b)
{
    if (b==0)
		return a;
	else
		return gcd(b,a%b);
}
ll power(ll a,ll b)
{
    if(b==0)
        return 1;
    if(b==1)
        return a;
    ll temp=power(a,b/2);
    if(b%2==0)
        return temp*temp;
    else
        return temp*temp*a;
}
ll dp[1000];
using namespace std;
int main()
{
    ll int i,j,k,n;
   cin>>n>>k;
    dp[0]=0;
    for(i=0;i<k;i++)
        dp[i]=0;
    for(i=k;i<=n;i++)
    {
        ll sum=0;
        for(j=1;j<=k;j++)
        {
            if(i-j>=0)
            sum+=dp[i-j];
        }
        dp[i]=sum+power(2,i-k);
    }
    ll ans=gcd(dp[n],power(2,n));
    //cout<<ans<<endl;
    //cout<<ans<<endl;
    //for(i=1;i<=10;i++)
      //  cout<<dp[i]<<" ";
 cout<<dp[n]/ans<<"/"<<power(2,n)/ans<<endl;

}
