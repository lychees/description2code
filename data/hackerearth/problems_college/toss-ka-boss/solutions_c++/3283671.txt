#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str;
		cin>>str;
		long long int i,len = str.length();
		long long int c=1,sum=0;
		for(i=0;i<len;i++)
		{
			if(str[i]=='H')
			{
				sum = sum+c;
				c++;
			}
			else{
				c=1;
			}
		}
		cout<<sum<<"\n";
	}
}
