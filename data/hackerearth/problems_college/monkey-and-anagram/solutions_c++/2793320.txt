#include <iostream>
#include<algorithm>
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<math.h>
using namespace std;
int main() {
	int t;
	scanf("%d",&t);
	while(t--){
		char s[1000],t[1000];
		scanf("%s",s);
		scanf("%s",t);
		int a[26]={0},b[26]={0};
		int i,j,k=0;
		for(i=0;i<strlen(s);i++){
			a[s[i]-'a']++;
		}
		for(i=0;i<strlen(t);i++){
			b[t[i]-'a']++;
		}
		for(i=0;i<26;i++){
			if(a[i]==b[i])
			continue;
			else if(a[i]<b[i]){
				k++;
				continue;
			}
			else if(a[i]>b[i]){
				k=-1;
				break;
			}
		}
		
			printf("%d\n",k);
			
	}
}
