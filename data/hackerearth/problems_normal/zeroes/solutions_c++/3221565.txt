#include<bits/stdc++.h>
using namespace std;
#define ll long long
vector<vector<pair<ll,ll> >  >v(100101);
vector<bool> check(100101);
#define lt 100101
#define pb push_back
int main()
{
	ll i,j,c,val,temp,n,q;
	vector<ll> prm;
	for(i=2;i<lt;i++)
	{
		if(check[i]==false)
		{
			prm.pb(i);
			v[i].pb(make_pair(i,1));
			for(j=2*i;j<lt;j=j+i)
			{
				check[j]=true;
				v[j].push_back(make_pair(i,0));
			}
		}
	}
	for(i=2;i<lt;i++)
	{
		for(j=0;j<v[i].size();j++)
		{
			val=i;
			c=0;
			temp=v[i][j].first;
			while(val%temp==0)
			{
				c++;
				val/=temp;
			}
			v[i][j].second=c;
		}
	}
	vector<ll>::iterator it;
	vector<ll> m(100001);
	scanf("%lld",&n);
	for(i=0;i<n;i++)
	{
		scanf("%lld",&val);
		for(it=prm.begin();(*it)<=val;it++)
		{
			j=*it;
			while(val%j==0)
			{
				m[j]++;
				val/=j;
			}
		}
	}
	ll mini;
	scanf("%lld",&q);
	while(q--)
	{
		mini=1000000;
		scanf("%lld",&val);
		for(j=0;j<v[val].size();j++)
		{
			temp=v[val][j].second;
			n=m[v[val][j].first];
			mini=min(mini,n/temp);
		}
		printf("%lld\n",mini);
	}
	return 0;
}
