#include<iostream>
#include<algorithm>
using namespace std;
#define ll long long
int main()
{
   ll t,n,i;
   cin>>t;
   while(t--)
   {
      cin>>n;
      if(n==1)
       cout<<"1\n";
       else if(n==2)
        cout<<"2\n";
        else
        {
        	ll res;
           for( i=1;;i++)
           {
           	 if(i*i>=n)
           	 {
           	 	res=2*i-1;
           	 	
           	 	break;
           	 }
           } 
           if((i*i -i)<n)
     cout<<res<<"\n";
     else cout<<res-1<<"\n";
   
        }   
  }
   return 0;
}
