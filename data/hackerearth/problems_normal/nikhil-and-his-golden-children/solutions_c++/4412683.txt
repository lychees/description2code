#include <iostream>
using namespace std;
long long int modpowIter(long long int a, long long int b,long long int c) {
        long long int ans=1;
        while(b != 0) {
                if(b%2 == 1)
                        ans=(ans*a)%c;

                a=(a*a)%c;
                b /= 2;
        }
        return ans;
}
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int k,m;
    	cin >> k >> m;
    	long long int sum=0;
    	int i;
    	for(i=1;i<=k;i++)
    	{
    		int a;
    		cin >> a;
    		sum+=a;
    	}
    	m-=sum;
    	long long int a,b,c;
    	a=1;b=1;c=1;
    	int mod=1000000007;
    	for(i=m+k-1;i>=2;i--)
    	{
    		a=(a%mod*i%mod)%mod;
    		if(i<=m)
    		b=(b%mod*i%mod)%mod;
    		if(i<=k-1)
    		c=(c%mod*i%mod)%mod;
    	}
    	long long int ans=(a%mod*modpowIter(b,mod-2,mod)%mod)%mod;
    	ans=(ans%mod*modpowIter(c,mod-2,mod)%mod)%mod;
    	if(m>=0)
    	cout << ans<<endl;
    	else
    	cout << 0 << endl;
    }
    return 0;
}
