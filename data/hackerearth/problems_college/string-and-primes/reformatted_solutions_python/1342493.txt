def isPrime(n):
    for i in range(2, n):
        if n % i == 0:
            return 1
    return 0


n = int(raw_input())
i = 0
str = []
out = []
while i < n:
    str.append(raw_input())
    i += 1


for s in str:
    temp = ""
    for c in s:
        if isPrime(ord(c)) == 1:
            temp += c

    out.append(temp)

for i in out:
    if i:
        print i
    else:
        print "NULL"
