#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int A[10005], n;
ll dp[10005];
ll solve(int a)
{
	if(dp[a] != -1)
		return dp[a];
	if(a == n-2)
	{
		dp[a] = max(A[n-2],A[n-1]);
		return dp[a];
	}
	if(a > n-1)
		return 0;
	if(a == n-1)
		return A[n-1];
	dp[a] = max(A[a] + solve(a+2), A[a+1] + solve(a+3));
	return dp[a];

}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		memset(dp, -1, sizeof dp);
		for (int i = 0; i < n; ++i)
		{
			scanf("%d",&A[i]);

		}
		printf("%lld\n", solve(0));
	}
	return 0;
}
