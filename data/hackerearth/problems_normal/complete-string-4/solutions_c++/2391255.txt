#include<iostream>
#include<string>
using namespace std;
int main()
{
int n,i,j,c;
cin>>n;
while(n--)
{
char s[100];
cin>>s;
c=0;
for(i=97;i<=122;i++)
{
for(j=0;s[j]!='\0';j++)
{
if(s[j]==i)
{
    c++;
    break;
}
}
}
if(c==26)
cout<<"YES"<<endl;
else
cout<<"NO"<<endl;
}
return 0;
}
