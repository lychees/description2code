#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int dp[110][10],ar[3]={1,3,5},k=3;

int main()
{
	ios_base::sync_with_stdio(0);
	int t,a,v1,v2;
	cin>>t;
	while(t--)
	{
		cin>>a;
		if(a<0)
		{
			cout<<"0\n";
			continue;
		}
		for(int i=0;i<k;i++)
			dp[0][i]=1;
		for(int i=1;i<=a;i++)
			for(int j=0;j<k;j++)
			{
				v1=v2=0;
				if(i-ar[j]>=0)
					v1=dp[i-ar[j]][j];
				if(j)
					v2=dp[i][j-1];
				dp[i][j]=v1+v2;
			}
		cout<<dp[a][k-1]<<"\n";
	}
	return 0;
}
