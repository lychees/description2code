#include <iostream>
#include<string.h>
using namespace std;
string s, b;
int main()
{
    int k,i,j,t,f=0,win=0;
    //char b[100000],s[100000];
    cin>>b>>k;
    cin>>t;
    while(t--)
    {
    	f=0;
    	j=0;
    	cin>>s;
    	int l=s.size();
    	for(i=0;i < b.size();i++)
    	{
    		if(s[j]==b[i])
    		{
    			f++;
    			j++;
    			if(j==l)
    			break;
    			
    		}
    	}
    	if(f==l)
    	win++;
    	
    }
    cout<<win;
    if(win>=k)
    cout<<" LUCKY";
    else
    cout<<" UNLUCKY";
    return 0;
}
