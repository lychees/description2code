#include <iostream>
#include <vector>
#include <cstring>
using namespace std;

#define SQR(_x) ((_x)*(_x))
#define MAX 1024
char f[MAX][MAX];

int main()
{
	int n, x, y, r, resp;
	cin >> n;
	resp = 0;
	memset(f, 0, sizeof(f));
	for(int i = 0; i < n; i++) {
		cin >> x >> y >> r;
		for(int j = 1; j <= 1000; j++) {
			for(int k = 1; k <= 1000; k++) {
				// ((x1-x2)^2+(y1-y2)^2) <= r^2
				if (f[j][k] < 2 && SQR(x-j)+SQR(y-k) <= SQR(r)) {
					f[j][k]++;
					if (f[j][k] == 2) resp++;
				}
			}
		}
	}
	cout << resp << endl;
    return 0;
}
