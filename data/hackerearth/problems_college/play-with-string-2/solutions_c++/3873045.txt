#include <iostream>
#include<string.h>
using namespace std;
int main()
{
	int t,temp,i,a,b;
	cin>>t;
	string s;
	while(t--)
	{
		temp=1;
		cin>>s;
		int l1=s.length()-1;
		for(i=0;i<(s.length())/2;i++)
		{
			a=abs(s[i+1]-s[i]);
			b=abs(s[l1-i]-s[l1-i-1]);
			if(a!=b)
			{
				cout<<"Not Funny\n";
				temp=0;
				break;
			}
		}
		if(temp)
		cout<<"Funny\n";
	}
    return 0;
}
