#include <cstdio>
#include <algorithm>
using namespace std;

int t;
int n, k, p;
int a[100003];
int i;

int main()
{
	scanf("%d", &t);

	while(t--)
	{
		scanf("%d%d%d", &n, &k, &p);
		for(i = 0; i < k; ++i)
			scanf("%d", &a[i]);
		sort(a, a + k);
		i = 0;
		while(i < k && p <= n && a[i++] <= p) p++;
		printf("%d\n", p <= n ? p : -1);
	}

	return 0;
}
