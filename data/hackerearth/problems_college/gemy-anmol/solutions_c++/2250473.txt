#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i;
		long long m,sum=0;
		int count=0;
		cin>>n>>m;
		long long *arr=new long long[n];
		for( i=0;i<n;++i)
		cin>>arr[i];
		sort(arr,arr+n);
		for(i=n-1;i>=0;--i)
		{
			sum+=arr[i];
			
			count++;
			if(sum>=m)
			{
				cout<<count<<endl;
				break;
			}
		}
		if(i==-1)
		cout<<"-1"<<endl;
		
	}
	return 0;
}
