int c[1000001];
int main(){
	//freopen("input.in","r",stdin);
	int n,i,j,k,m,q;
	k=1;
	cin>>n;
	c[0]=0;
	for(i=0;i<n;++i){
		cin>>m;
        for(j=k;j<(k+m);++j)

        k+=m;
	}
	cin>>q;
	while(q--){
		cin>>i;
		cout<<c[i]<<endl;
	}
	return 0;
}
