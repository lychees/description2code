#include <bits/stdc++.h>
using namespace std;
bool wayToSort(int i, int j) { return i > j; }
int main() {
	// your code goes here
	long long n, k, i, sum=0, j, c;
	cin>>n>>k;
	int a[n];
	for(i=0; i<n; i++){
		cin>>a[i];
	}
	sort(a, a+n);c=0;
	for(i=n-1; i>=0; ){
		j=k;
		while(j--){
			sum+=(c+1)*a[i--];
		//	cout<<sum<<endl;
			if(i<0)break;
		}
		c++;
	}
	cout<<sum;
	return 0;
}
