""" http://www.hackerearth.com/problem/algorithm/weird-friends/"""

inp = raw_input().split()
l = int(inp[0])
x = int(inp[1])
y = int(inp[2])
count = 0
buffer = 0
while True :
    buffer += x
    if buffer >= l :
        count += 1
        break
    else :
        buffer -= y
        count += 1
print count
