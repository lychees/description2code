#include <bits/stdc++.h>
using namespace std;

#define ll long long int
#define mp(i,j) make_pair(i,j)
#define pb(x) push_back(x)
#define MAXN 10010

ll sp[MAXN];
ll done[MAXN];

int main()
{	
	ll t,n,m,i,j,pre,cnt;

	scanf("%lld", &t);

	while(t--) {
		scanf("%lld", &n);

		for(i=0;i<n;i++) {
			scanf("%lld", &sp[i]);
		}
	
		scanf("%lld", &m);
	
		ll o[m][n];
		ll r[m][n];

		memset(done,0,sizeof(done));		

		for(i=0;i<m;i++) {
			for(j=0;j<n;j++) {
				scanf("%lld", &o[i][j]);
			}
		}
		for(i=0;i<m;i++) {
			for(j=0;j<n;j++) {
				scanf("%lld", &r[i][j]);
			}
		}

		pre = -1;
		cnt = 0;
		while((pre!=cnt) && (cnt<m)) {
			pre = cnt;
			for(i=0;i<m;i++) {
				if(done[i]==1) continue;
				for(j=0;j<n;j++) {	
					if(sp[j]+o[i][j] < r[i][j]) break;
				}
				if(j==n) {
					done[i] = 1;
					cnt++;
					for(j=0;j<n;j++) {	
						sp[j] += o[i][j];
					}
					break;	
				}
			}
		}
		if(cnt>=m) printf("YES\n");
		else printf("NO\n");
	}	
	
	return 0;
}
