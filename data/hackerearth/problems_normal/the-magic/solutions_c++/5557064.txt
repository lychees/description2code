#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    for (int i = 0; i < t; i++){
    	int n;
    	cin >> n;
    	int counter = 0;
    	while(n != 0){
    		
    		n = n & (n-1);
    		if (!(n & 1)){
    			n >>= 1;
    		}
    		++counter;
    	}
    	cout << counter << endl;
    }
    return 0;
}
