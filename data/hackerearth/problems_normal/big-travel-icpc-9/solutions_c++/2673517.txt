#include <stdio.h>
#include <algorithm>
#include <vector>
using namespace std;
#define gc getchar_unlocked
#define N 1000000007

long long int a1[200010]={0},b1[200010]={0};

int fin()
{
	int res=0;
	register char c=gc();
	while(c<'0' || c>'9')
		c=gc();
	while(c>='0' && c<='9')
	{
		res=res*10+(c-48);
		c=gc();
	}
	return res;
}

int main()
{
	int n=fin();
	long int tmp;
	unsigned long long tot=0;
	vector<long long int> a,b;
	for(int i=0;i<n;++i)
	{
		scanf("%ld",&tmp);
		a.push_back(tmp);
		scanf("%ld",&tmp);
		b.push_back(tmp);
	}
	sort(a.begin(),a.end());
	sort(b.begin(),b.end());
	a1[n-1]=a[n-1];
	b1[n-1]=b[n-1];
	for(int i=n-2;i>=0;--i)
	{
		a1[i]=(a1[i+1]+a[i])%N;
		b1[i]=(b1[i+1]+b[i])%N;
	}
	
	for(int i=0;i<n;i++)
	{
		tot+=(a1[i]%N-(n-i)*a[i]%N)%N;
		tot+=(b1[i]%N-(n-i)*b[i]%N)%N;
	}
	printf("%llu\n",tot%N);



	return 0;
}
