while (test > 0):
    test -= 1

    cal = sorted([int(s) for s in raw_input().split()])
    l = len(cal)
    r = 0
    s = 0
    rc = 0
    sc = 0
    while (r < l and rc+cal[r] <= a):
        rc += cal[r]
        r += 1

        sc += cal[s]
        s += 1
    if (r > s):
