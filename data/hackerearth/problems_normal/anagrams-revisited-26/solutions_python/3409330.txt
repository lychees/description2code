from collections import defaultdict


if __name__ == "__main__":
    count_dict = defaultdict(int)
    test_cases = input()
    # print test_cases
    input_set = set()
    max_count = 0

    for _ in xrange(test_cases):
        new_str = raw_input()
        new_str_identifier = ''.join(sorted(new_str))
        # print new_str, "id:", new_str_identifier
        if new_str not in input_set:
            input_set.add(new_str)
            count_dict[new_str_identifier] += 1
        if count_dict[new_str_identifier] > max_count:
            max_count = count_dict[new_str_identifier]
    print max_count
