#include <bits/stdc++.h>
using namespace std;

int main()
{
	std::ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		map<int,int> m;
		int x,y,count=0;
		for(int i =0 ;i<n;++i)
		{
			//cout<<count<<endl;
			cin>>x>>y;
			if(m.find(x)!=m.end())
				m[x]+=1;
		  else 	if(m.find(x)==m.end())
				{m[x]=1;}
			
			if(m.find(y)!=m.end() && m[y]>0)
			{
				m[y]-=1;
			}
			else if(m.find(y)==m.end() || m[y]==0 )
			{
				count++;
				//cout<<x<<" "<<y<<" "<<m[y]<<endl;
				//cout<<count<<endl;
			}
			
			
			
			//cout<<count<<endl;
		}
		
		cout<<count<<endl;
	}
    return 0;
}
