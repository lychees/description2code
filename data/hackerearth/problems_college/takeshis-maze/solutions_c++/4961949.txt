#include<bits/stdc++.h>
using namespace std;
 
typedef long long lld;
lld cost[1001][1001];
lld check(int n)
{
     lld i, j;
     lld tc[n+1][n+1];  
 
     for(i=0;i<n;i++)
      { for(j=0;j<n;j++)
          tc[i][j]=INT_MIN;
      }
 
     
     tc[0][0] = cost[0][0];
          
     
     for (i = 1; i <n; i++)
       { 
          // if(cost[i][0] != -1)
            tc[i][0] = tc[i-1][0] + cost[i][0];
       
        }
     
     for (j = 1; j <n; j++)
      {  
        // if(cost[0][j] != -1)
         tc[0][j] = tc[0][j-1] + cost[0][j];
         
      }
 
     
     for (i = 1; i <n; i++)
      {  for (j = 1; j <n; j++)
          {  
            tc[i][j] = max(tc[i-1][j], tc[i][j-1]) + cost[i][j];
             
         }
      }
     return tc[n-1][n-1];
}
 
 
 
int main()
{
	lld t;
	cin>>t;
 
     while(t--)
     {
         lld n,c;
         cin>>n>>c;
         
        // lld a,b;
         for(lld i=0;i<n;i++)
         {  for(lld j=0;j<n;j++)
              cin>>cost[i][j];
         }
         
        lld ans=check(n);
        if((ans-c)<0)
         cout<<"-1\n";
        else
         cout<<ans-c<<"\n"; 
  
     }
 
 
}
