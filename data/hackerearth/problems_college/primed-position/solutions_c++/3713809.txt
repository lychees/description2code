#include <bits/stdc++.h>
using namespace std;
#define MAX 10000000
#define MOD 1000000007
int a[MAX];
vector <int> prime;
inline int mul(int x, int y){return ((1LL)*x * y)%MOD;}
void seive()
{
	a[0]=a[1]=1;
	
	for(int i=2;i<=MAX;++i)
	{
		if(a[i]==0)
		{
			for(int j=2;j*i<=MAX;++j)
			a[j*i]=1;
		}
	}
	int c=1;
	for(int i=2;i<=MAX;++i)
	{
		if(a[i]==0)
		{
			if(a[c]==0)
			prime.emplace_back(i);
			c++;
		}
	}
}
int main()
{
	seive();
	int t,n,m;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&m);
		printf("%lld\n",mul(prime[n-1],prime[m-1]));
	}
    return 0;
}
