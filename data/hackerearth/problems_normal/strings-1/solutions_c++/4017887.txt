#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	int t,i,flag=0;
	string n,m;
	cin>>t;
	while(t--)
	{
		flag=0;
		cin>>n>>m;
		if(m==n)
		{
			cout<<"YES"<<endl;
		}
		else if(m.size()!=1||n.size()!=1)
		{
			cout<<"NO"<<endl;
		}
		else
		{
			if(m[0]=='2'&&n[0]=='4')
				cout<<"YES"<<endl;
			else if(m[0]=='4'&&n[0]=='2')
				cout<<"YES"<<endl;
			else
				cout<<"NO"<<endl;
		}
	}
    return 0;
}
