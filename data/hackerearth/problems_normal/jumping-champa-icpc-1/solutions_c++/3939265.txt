#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long int t,n,q,i,count1;
	long long int *arr=new long long int[1010];
	cin>>t;
	while(t--)
	{
		cin>>n;
		cin>>q;
		for(i=0;i<n;i++)
		cin>>arr[i];
		sort(arr,arr+n);
		count1=0;
		for(i=1;i<n;i++)
		{
			count1+=arr[i]-arr[i-1];
		}
		count1=count1*q;
		cout<<count1<<endl;
	}
    return 0;
}
