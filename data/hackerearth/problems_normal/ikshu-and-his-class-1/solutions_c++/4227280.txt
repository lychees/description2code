#include <bits/stdc++.h>
using namespace std;
vector<int> adj[100001];
	bool v[100001]={0};
	int ncc[100001]={0};
int n,m;
int sss=0;
void dfs(int s)
{
    int j;
    v[s]=1;
    sss++;
    for(j=0;j<adj[s].size();j++)
    {
        if(v[adj[s][j]]==0)
        dfs(adj[s][j]);
    }
   
}
int main() {
	// your code goes here
	int i,x,y;
	scanf("%d",&n);
	scanf("%d",&m);
long long 	int fact[100001];
fact[0]=1;
	for(i=1;i<=100000;i++)
	{
		fact[i]=fact[i-1]*i;
		fact[i]=fact[i]%1000000007;
	}
	for(i=0;i<m;i++)
	{
	    scanf("%d",&x);
	scanf("%d",&y);
	adj[x].push_back(y);
		adj[y].push_back(x);
		
	}
	int cc=0;
	int ss=0;

	for(i=0;i<n;i++)
	{
	    if(v[i]==0)
	    {
	       
	        
	        dfs(i);
	       ncc[cc]=sss;
	       
	        if(cc>=1)
	        {
	            ncc[cc]=ncc[cc]-ss;
	        }
	         ss=ss+ncc[cc];
	        cc++;
	        //cout<<i<<endl;
	        
	    }
	}
//	cout<<cc<<endl;
	long long int ans=1;
	for(i=0;i<cc;i++)
	{
	    
	        ans=ans*fact[ncc[i]];
	        ans=ans%1000000007;
	    
	}
	cout<<ans<<endl;
	return 0;
}
