using namespace std;
#define MOD 1000000007
map<int,int> ma;
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int i,j,n,m,q;
	int x;
	cin>>n>>m;
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
		{
			cin>>x;
			ma[x]=i*10000+j+1;
		}
	cin>>q;
	while(q--)
	{
		cin>>x;
		n=ma[x];
		if(n==0)
			cout<<"-1 -1\n";
		else
			cout<<n/10000<<' '<<n%10000-1<<'\n';
	}
//	system("pause");
	return 0;
}
