#include<bits/stdc++.h>
#include<time.h>
using namespace std;
#define ll long long int
long long int P[168];
void shieve(){
	  
	  
     long long int i,j,k,count=0;
       
    for(i=2; i<1000; i++){
    	k=0;

    	for(j=2; j<=(int)sqrt(i); j++){
            if(i%j==0){
            	k=1;
            }
    	}
    	if(k!=1){
    		P[count++]=i;

         
    	} }
    
}
   
int main(int argc, char const *argv[])
{
	ll i,j,k,n,m,s,t,l,x,f;
     shieve();
	cin>>t;
	
	for (l=0; l<t; l++)
	{
	   cin>>n;
       f=0;
       x=0;
        s=0;
      for (i=2; i<=sqrt(n); i++)
        {
         if(n%i==0){
         	
            s=1;
           }
        }
       if(s!=1&&n!=2){
       	  cout<<"2^0";
          }
       else if(n==2){
       	cout<<"2^1";
       }   
         else{ 
        	if(n%2!=0)
        	    cout<<"2^0*";
        	
           
           for (i =0; i<168; i++)
          {
         s=0;
         
         if(n%P[i]==0){
             if(x>=1){
             	cout<<"*";
             }
            while(n%P[i]==0){
            	n=n/P[i];
            	s++;
            }
            x++;	
         }
         if(s!=0){
         	cout<<P[i]<<"^"<<s;
         }
 	    }
 	      if(n!=1){
 	      	cout<<"*"<<n<<"^1";
 	      }
        }
       cout<<"\n";
   }
    

    
	return 0;
}
