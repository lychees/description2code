#include <bits/stdc++.h>
using namespace std;

int main()
{
   
   int t;
   int A;
   cin>>t;
   while(t--){
   cin>>A;
    
   vector<vector<int>>vec(A,vector<int>(A));
   int sum = A*A;
   int x = 0;
   int y = 0;
   int step = 0;
   for(int i = 0; i < sum;){
       while(y + step < A){
           i++;
           vec[x][y] = i;
           y++;
       }
       y--;
       x++;
       while(x+step < A){
           i++;
           vec[x][y] = i;
           x++;
       }
       x--;
       y--;
       while(y>= 0+step){
           i++;
           vec[x][y] = i;
           y--;
       }
       step++;
       x--;
       y++;
       while(x >= 0 + step){
           i++;
           vec[x][y] = i;
           x--;
       }
       x++;
       y++;
   }
   for(int i = 0; i < A; i++){
   		for(int j = 0; j < A; j++){
   			cout<<vec[i][j];
   		}
   		cout<<endl;
   }
   cout<<endl;
   }
}
