#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	bool flag=1;
    	cin>>str;
    	int l=str.length();
    	int i=1;
    	while(i<=l/2)
    	{
//    		cout<<str[i-1]<<str[l-i]<<'\n';
    		if(str[i-1]!=str[l-i])
    		{
    			if(str[i-1]==str[l-1-i])
    				cout<<l-i<<'\n';
    			else
    				cout<<i-1<<'\n';
    			flag=0;
    			break;
    			
    		}
    		i++;
    	}
    	if(flag)
    		cout<<"-1"<<'\n';
    }
    return 0;
}
