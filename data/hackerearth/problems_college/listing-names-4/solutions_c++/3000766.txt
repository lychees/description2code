#include<bits/stdc++.h>
using namespace std;
int main()
{
	vector<string>s;
	vector<string>::iterator it;
	long long t,i;
	cin>>t;
	string name;
	for(i=0;i<t;i++)
	{
		cin>>name;
		s.push_back(name);
	}
	sort(s.begin(),s.end());
	//for(it=s.begin();it!=s.end();it++)
	//cout<<*it<<endl;
	long long c=0,j;
	for(i=0;i<t;i++)
	{
		//cout<<i<<endl;
		c=0;
		for(j=i;j<t;j++)
		{
			//cout<<"j="<<j<<endl;
			if(s.at(i)==s.at(j))
			c++;
			else
			{
				cout<<s.at(i)<<" "<<c<<endl;
				i=j-1;
				break;
			}
		}
		if(j==t)
		{
			cout<<s.at(i)<<" "<<c<<endl;
			i=j;
		}
	}
	return 0;
}
