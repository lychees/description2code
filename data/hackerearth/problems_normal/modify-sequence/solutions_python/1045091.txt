n = int(raw_input())
lst = map(int,raw_input().split())
'''
Store N numbers in array/vector arr[].
Set boolean variable possible to true.
Repeat for i =0 to N-1, subtract arr[i] from arr[i+1], if arr[i+1] is negative, set possible to false and break. Set arr[i]=0.
If last element of arr[] is not zero, set possible to false.
If at-least one non zero element exist in arr[], set possible to false.

If possible is true, print YES, else NO
'''
flag = True

for i in range(n - 1):
	lst[i + 1] -= lst[i] 
	if lst[i + 1] < 0: 
		flag = False
		break
	lst[i] = 0

if lst[-1] != 0: flag = False
if sum(lst) != 0: flag = False
if flag == True: print "YES"
else: print "NO"
