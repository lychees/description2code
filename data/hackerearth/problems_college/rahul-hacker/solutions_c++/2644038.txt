#include <iostream>
using namespace std;
int rev(int num)
{
		 int new_num=0;
		 while(num > 0)
    {
            new_num = new_num*10 + (num % 10);
            num = num/10;
    }
    return new_num;
}
int main()
{
    int a,b,c,d,e,t;
    cin>>t;
    for(int i=1;i<=t;i++)
    {
    	cin>>a>>b;
    	c=rev(a);
    	d=rev(b);
    	cout<<rev(c+d)<<endl;
    }
    return 0;
}
