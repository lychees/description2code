#include<bits/stdc++.h>
using namespace std;
int count1[1000001];
int main(){
	int n, k;
	cin >> n >>k;
	vector<int> a(n);
	for(int i=0;i<n;i++) cin >> a[i];
	sort(a.begin(),a.end(),greater<int>());
	long long int sum=0;
	int j,l=0;
	for(int i=0;i<n;){
		for( j=i;j<i+k && j<n;j++){
			sum+=((l+1)*(a[j]));
		}
		i=j;l++;
	}
	cout << sum <<endl;
	return 0;
}
