#include<stdio.h>
#include<iostream>
#include<stdlib.h>
using namespace std;
long long count_carry(long long n1,long long  n2)
{
	int s=0,c=0,count=0;
	while(n1>0||n2>0)
	{
		int unit_digit1=n1%10;
		int unit_digit2=n2%10;
		s=unit_digit1+unit_digit2+c;
		c=s/10;
		if(c>0)
		count++;
		n1=n1/10;
		n2=n2/10;
	}
	return count;
}
int main()
{
	long long t,n1,n2;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&n1,&n2);
		int count=count_carry(n1,n2);
		if(count==0)
		cout<<"No carry operation"<<endl;
		else
		if(count==1)
		cout<<count<<" carry operation"<<endl;
		else
		cout<<count<<" carry operations"<<endl;
	}
}
