#include<cstring>
#include<bitset>
#include<stdio.h>
using namespace std;
int main()
{
    int test;
    scanf("%d",&test);
    unsigned long int n;
    while(test--)
    {
        scanf("%d",&n);
        string str;
        int l,count;
        for(int i=n;i<131072;i++)
        {
            str = bitset<32>(i).to_string();
            count=0;
            for(int c=0;c<31;c++)
            {
                if(str[c]=='1' && str[c+1]=='1')
                {
                    count=1;
                }
            }
            if(count==0)
            {
                n=bitset<32>(str).to_ulong();
                printf("%d \n",n);
                break;
            }
        }
    }
        return 0;
}
