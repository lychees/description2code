#include <bits/stdc++.h>
#define si(n) scanf("%d",&n)
#define sl(n) scanf("%lld",&n)
#define ss(n) scanf("%s",n)
#define fr(i, n) for(int i = 0; i < n; i++)
#define REP(i, a, b) for (int i = int(a); i <= int(b); i++)
#define REN(i, a, b) for (int i = int(a); i >= int(b); i--)
#define ms(i, n) memset(i, n, sizeof(i))
#define INF 1002000000
#define MOD 1000000007
typedef long long LL;
using namespace std;
vector<int> a, p, b, vec, visited;
vector<vector<int> > ans;
void dfs(int curr) {
    if(visited[curr] == 1)
        return;
    visited[curr] = 1;
    vec.push_back(curr);
    dfs(p[curr]);
}
int main ()
{
    //freopen ("input.txt","r",stdin);
    //freopen ("output.txt","w",stdout);
    ios_base::sync_with_stdio(false);
    int n;
    LL t;
    si(n);sl(t);
    a.resize(n);
    b.resize(n);
    p.resize(n);
    visited.resize(n);
    for(int i = 0; i < n; i++) {
        si(a[i]);
    }
    for(int i = 0; i < n; i++) {
        si(p[i]);p[i]--;
    }
    for(int i = 0; i < n; i++) {
        visited[i] = 0;
    }
    for(int i = 0; i < n; i++) {
        vec.clear();
        dfs(i);
        if(vec.size() != 0)
            ans.push_back(vec);
    }
    for(int i = 0; i < ans.size(); i++) {
        LL tmp = t % (LL)ans[i].size();
        for(int j = 0; j < ans[i].size(); j++) {
            b[ans[i][j]] = a[ans[i][(j + tmp) % ans[i].size()]];
        }
    }
    for(int i = 0; i < n; i++) {
        printf("%d ", b[i]);
    }
    return 0;
}
