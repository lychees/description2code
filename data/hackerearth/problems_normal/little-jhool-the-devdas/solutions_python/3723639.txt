n=int(raw_input())
mn=-1
psum=0
numlist=raw_input()
p=numlist.split()
for i in range(n):
	p[i]=int(p[i])
psum=sum(p)
if (psum%2==0):
	for i in range(n):
		if mn==-1 and p[i]%2==1:
			mn=p[i]
		if p[i]%2==1 and mn>p[i]:
			mn=p[i]
if psum%2==1:
	print psum
elif mn==-1:
	print ':('
else:
	print psum-mn
