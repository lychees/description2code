#include<bits/stdc++.h>
using namespace std;
int main()
{
	
	
	int t,k=1;
	scanf("%d",&t);
	while(t--)
	{
		long int dec[3];	
		cin>>dec[0]>>dec[1]>>dec[2];
		printf("Case %d: ",k);
		for(int p=0 ; p<3 ; ++p)
		{
			char hexdecnum[100];
			int i=1, j, temp;
			long int  rem, quot;
			quot=dec[p];
			if(quot==0)
			{
				cout<<"00 ";
				continue;
			}
			else
			{
				while(quot!=0)
				{
					temp=quot%16;
					// to convert integer into character
					if(temp<10)
					{
						temp=temp+48;
					}
					else
					{
						temp=temp+55;
					}
					hexdecnum[i++]=temp;
					quot=quot/16;
				}
				
				if(i==2)
					cout<<"0";
				for(j=i-1; j>0; j--)
				{
					if(hexdecnum[j]>='A' && hexdecnum[j]<='F' )
					{
						hexdecnum[j]=hexdecnum[j]-'A'+'U';
					}
					cout<<hexdecnum[j];
				}
				cout<<" ";
			}
			
		}
		cout<<endl;
		++k;
	}
return 0;
}
