#include <iostream>
using namespace std;

int main()
{
    int t,x,y;
    cin>>t;
    while(t--){
    	cin>>x>>y;
    	if (x<y){
    		cout << "No Number\n";
    		continue;
    	}
    	if (x-y>2){
    		cout << "No Number\n";
    		continue;
    	}
    	if (x&1 && y&1) cout << x+y-1 << "\n";
    	else if (x%2==0 && y%2==0) cout << x+y << "\n";
    	else cout << "No Number\n";
    }
    return 0;
}
