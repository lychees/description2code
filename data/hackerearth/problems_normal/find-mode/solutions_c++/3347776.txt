#include <iostream>
#include<bits/stdc++.h>
#define ll long long
using namespace std;
bool cmp(ll a,ll b)
{
	return a>b;
}
int main()
{
	 
  ll t,n,temp;
    cin>>t;
    while(t--)
	{	 map<ll,ll> m;
       map<ll,ll> ::iterator r;
       ll a[100004]={0};
    cin>>n;;
    while(n--){
    	cin>>temp;
    	 m[temp]+=1;
    }
    ll max=-1,j=0;
   for(r=m.begin();r!=m.end();r++){
   	  if(r->second>max)max=r->second;
   }
    for(r=m.begin();r!=m.end();r++){
   	  if(r->second==max)a[j++]=r->first;
   }
   sort(a,a+j,cmp);
   for(ll i=0;i<j;i++)
    cout<<a[i]<<" ";
    cout<<endl;
}
    return 0;
}
