#include <iostream>
#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    int t;
    cin >>t;
    while(t--){
    	ll n,m,x,sum=0,res;
    	cin >>n >>m >>x;
    	ll a[n];
    	for(ll i=0;i<n;i++){
    		cin >> a[i];
    		sum += a[i];
    	}
    	res = (n+1)*x - sum;
    	if(res <= 1){
    		cout << 1 <<endl;
    	}
    	else if(res > m) {
    		cout <<"Impossible"<<endl;
    	}
    	else{
    		cout <<res <<endl;
    	}
    }
    return 0;
}
