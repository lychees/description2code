#include <iostream>
using namespace std;
int main()
{
    int t;
    int n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int a[n][3];
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i][0]>>a[i][1]>>a[i][2];
    	}
    	int sum[n][3];
    	sum[0][0]=a[0][0];
    	sum[0][1]=a[0][1];
    	sum[0][2]=a[0][2];
    	for(int i=1;i<n;i++)
    	{
    		sum[i][0]=a[i][0]+min(sum[i-1][1],sum[i-1][2]);
    		sum[i][1]=a[i][1]+min(sum[i-1][0],sum[i-1][2]);
    		sum[i][2]=a[i][2]+min(sum[i-1][0],sum[i-1][1]);
    	}
    	int minz=sum[n-1][0];
    	if(sum[n-1][1]<minz&&sum[n-1][1]<sum[n-1][2])
    		minz=sum[n-1][1];
    	else if(sum[n-1][2]<minz&&sum[n-1][2]<sum[n-1][1])
    		minz=sum[n-1][2];
    	cout<<minz<<endl;
    }
    return 0;
}
