#include <bits/stdc++.h>
using namespace std;

int mark[20][20],cell[20][20],c=0,n;

void call(int i,int j)
{
	if(i==n-1 && j==n-1)
	{
		c++;
	return ;
	}
	mark[i][j]=1;
	if((j+1)<n && mark[i][j+1]==0 && cell[i][j+1]==0)
	call(i,j+1);
	if((i+1)<n && mark[i+1][j]==0 && cell[i+1][j]==0)
	call(i+1,j);
	if((j-1)>=0 && mark[i][j-1]==0 && cell[i][j-1]==0)
	call(i,j-1);
	if((i-1)>=0 && mark[i-1][j]==0 && cell[i-1][j]==0)
	call(i-1,j);
	mark[i][j]=0;
	return;
	
}
int main()
{
    int t,i,j;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    	for(j=0;j<n;j++)
    	{
    		cin>>cell[i][j];
    		mark[i][j]=0;
    	}
    	}
    	c=0;
    	call(0,0);
    	cout<<c<<endl;
    }
    return 0;
}
