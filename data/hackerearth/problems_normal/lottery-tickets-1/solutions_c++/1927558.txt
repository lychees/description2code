#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
struct node{
ll s,e;
int wt;

}a[1000];
int dp[1000];
bool co(struct node a,struct node b)
{

    return a.e<b.e;
}
int main()
{
    int t;
    int n;
    cin>>t;long long int maxm;
    while(t--)
    {
      maxm=-1;
        cin>>n;
        for(int i=0;i<n;i++)
        {
            cin>>a[i].s>>a[i].e>>a[i].wt;
        }
        sort(a,a+n,co);
        for(int i=0;i<n;i++)
        {
            dp[i]=a[i].wt;
            for(int j=i-1;j>=0;j--)
            {
                if(a[i].s>=a[j].e && dp[i]<dp[j]+a[i].wt)
                {
                    dp[i]=a[i].wt+dp[j];
                }
            }
        }
        for(int i=0;i<n;i++)
        {
            if(maxm<dp[i])
            maxm=dp[i];
                    }
                    double x=(float)maxm/100.00;
                printf("%.2f\n",x);
    }
    return 0;
}
