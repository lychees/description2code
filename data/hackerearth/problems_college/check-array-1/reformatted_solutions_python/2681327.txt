t = int(raw_input())
while t > 0:
    n = int(raw_input())
    arr = [int(i) for i in raw_input().split()]
    flag = 0
    sFromEnd = sum(arr)
    sFromStart = 0

    for i in arr:
        if (sFromStart == sFromEnd - i):
            flag = 1
            break
        sFromStart += i
        sFromEnd -= i
    if flag == 1:
        print 'YES'
    else:
        print 'NO'
    t -= 1
