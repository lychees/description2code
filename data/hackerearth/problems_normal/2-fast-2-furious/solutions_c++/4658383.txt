#include<iostream>
#include<stdio.h>
using namespace std;

int main()
{
    long long int n;
    cin>>n;
    long long int brian[n];
    long long int dom[n];

    long long int max1 = -1;
    long long int max2 = -1;
    for(int i=0;i<n;i++)
    {
        cin>>dom[i];
        if(i>0)
        {
            long long int m1 = dom[i] - dom[i-1];
            if(m1<0) {
                m1*=-1;
            }
            if(max1<m1){
                max1 = m1;
            }

        }
    }
    for(int i=0;i<n;i++)
    {
        cin>>brian[i];
        if(i>0)
        {
            long long int m1 = brian[i] - brian[i-1];
            if(m1<0) {
                m1*=-1;
            }
            if(max2<m1){
                max2 = m1;
            }

        }
    }
    if(max1<max2){
        cout<<"Brian\n"<<max2;
    }

    if(max1>max2){
        cout<<"Dom\n"<<max1;
    }
    if(max1 == max2){
        cout<<"Tie";
    }

}
