#include<iostream>
#include<stdlib.h>
using namespace std;
int compare(const void *a,const void *b)
{
    return (*(int*)a - *(int*)b );
}
int main()
{
    int t,m,limit,max,min;
    cin>>t;
    while(t--)
    {
        cin>>m;
        int num[m];
        for(int i=0;i<m;++i)
        {
            cin>>num[i];
        }

        if(m==0)
        {
            cout<<"0\n";
        }
        else
        {
            qsort(num,m,sizeof(int),compare);
            limit=m/2;
            max=min=num[0]+num[m-1];
            for(int i=1,j=m-2;i<limit;++i,--j)
            {
                if(max<(num[i]+num[j]))
                        max=num[i]+num[j];
                else if(min>(num[i]+num[j]))
                        min=num[i]+num[j];

            }
            cout<<max-min<<"\n";
        }
    }
    return 0;
}
