#include<bits/stdc++.h>

using namespace std;
int dp[2][100001];
int main()
{
    int t,n,x,i,j,a[101];
    cin>>t;
    while(t--)
    {
        cin>>n>>x;
        int sum=0;
        for(i=1;i<=n;i++){
            cin>>a[i];
          sum+=a[i];}
          if(sum<=x)
            cout<<"YES"<<endl;
          else
          {
              memset(dp,0,sizeof(dp));
              for(i=1;i<=n;i++)
              {
                  for(j=0;j<=x;j++)
                  {
                      if(j>=a[i])
                      dp[i%2][j]=max(dp[(i-1)%2][j],dp[(i-1)%2][j-a[i]]+a[i]);
                      else
                        dp[i%2][j]=dp[(i-1)%2][j];
                      //  cout<<dp[i%2][j]<<" ";
                  }
              }
              if(sum-dp[(n)%2][x]<=x)
                cout<<"YES"<<endl;
              else
                cout<<"NO"<<endl;
          }
    }
    return 0;
}
