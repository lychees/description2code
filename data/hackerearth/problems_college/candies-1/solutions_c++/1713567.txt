#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--)
   {
   	int n;
   	cin>>n;
   	long long int sum=0;
   	for(int i=1;i<=n;i++)
   	{
   		long long int x;
   		cin>>x;
   		sum+=x;
   	}
   	long long int a=sum%n;
   	if(a==0)
   	{
   		cout<<"YES"<<endl;
   		cout<<sum/n<<" "<<"0";
   	}
   	else
   	{
   		cout<<"NO"<<endl;
   		cout<<sum/n<<" "<<a;
   	}
   	cout<<endl;
   }
    return 0;
}
