#include <bits/stdc++.h>
 
using namespace std;
 
bool ok[42][42];
 
int main() {
  int tt;
  scanf("%d", &tt);
  while (tt--) {
    int n, p;
    scanf("%d %d", &n, &p);
    for (int i = 0; i < n; i++) {
      for (int j = 0; j < n; j++) {
        ok[i][j] = true;
      }
    }
    for (int i = 0; i < p; i++) {
      int x, y;
      scanf("%d %d", &x, &y);
      x--; y--;
      ok[x][y] = false;
    }
    int ans = 0;
    for (int t = 1; t < (1 << n); t++) {
      bool good = true;
      for (int i = 0; i < n; i++) {
        if (t & (1 << i)) {
          for (int j = 0; j < n; j++) {
            if (t & (1 << j)) {
              if (!ok[i][j]) {
                good = false;
                break;
              }
            }
          }
          if (!good) {
            break;
          }
        }
      }
      if (good) {
        ans++;
      }
    }
    printf("%d\n", ans);
  }
  return 0;
}
