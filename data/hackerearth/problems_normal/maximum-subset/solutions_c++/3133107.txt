#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007

 
long long modpow(long long x,long long y, long long z)
{
 long long r = 1, i = y, j = x%z;
 while(i)
 {
  if(i & 1) r = r*j%z;
  j = j * j % z;
  i = i/2;        
 }      
 return r;
}

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	int n;
    	cin>>n;
    	long long arr[n];
    	for(int i=0;i<n;i++) {
    		cin>>arr[i];
    	}
    	long long ans  = 0;
    	for(int i=1;i<(1<<n);i++) {
    		vector<long long> newsub;
    		for(int j=0;j<n;j++) {
    			if(i & (1<<j)) {
    				newsub.push_back(arr[j]);
    			}
    		}
    		int newsubsize = newsub.size();
    		if(newsubsize%2==0) {
    			sort(newsub.begin(), newsub.end());
    			long long num = 1;
    			long long den = 1;
    			
    			for(int j=0; j<newsubsize/2;j++) {
    				den *= newsub[j];
    				den%=MOD;
    				num *= newsub[j + (newsubsize/2)];
    				num %= MOD;
    			}
    			
    			den = modpow(den, MOD-2, MOD);
    			long long tempans = (num * den)%MOD;
    			if(tempans > ans)
    				ans = tempans;
    		}
    		
    	}
    	cout<<ans<<endl;
    }
}
