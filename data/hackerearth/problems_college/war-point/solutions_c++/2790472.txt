#include<bits/stdc++.h>
#define mod 1000000007
#define mp make_pair
#define pb push_back
#define ll long long
#define F first

#define scni(t) scanf("%d",&t)
#define scnl(t) scanf("%lld",&t)
#define rep(i,n) for(int i=0;i<n;i++)
#define REP(i,a,b) for(int i=a;i<=b;i++)
#define N 100005
#define mod 1000000007
using namespace std;

int arr[N];

int main(){
	int n;
	ll sum,cur;
	int t;
	scni(t);
	while(t--){
		scni(n);
		sum = 0;
		rep(i,n){
			scni(arr[i]);
			sum += (ll)arr[i];
		} 
		cur=0;
		ll k,minn = mod;
		rep(i,n-1){
			cur += (ll)arr[i];
			k = sum - 2*cur;
			if(k < 0) k = -k;
			minn = min(minn,k);
		}
		printf("%lld\n",minn );
	}
	return 0;
}
