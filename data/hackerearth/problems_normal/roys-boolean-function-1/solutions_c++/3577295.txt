    #include<iostream>
    using namespace std;
    int t;
    long n,i,k,x;
    bool f;
    long gcd(long,long);
    bool prime(long);
    int main()
    {
    	cin>>t;
    	while(1)
    	{
    		if(t==0)
    		break;
    		t--;
    		cin>>n;
    		x=1;
    		for(i=2;i<n;i++)
    		{
    			k=gcd(n,i);
    			if(k==1)
    			x++;
    		}
    		f=prime(x);
    		if(f)
    		cout<<"TRUE"<<endl;
    		else
    		cout<<"FALSE"<<endl;
    	}
    	return 0;
    }
    long gcd(long a,long b)
    {
    	if(b==0)
    	return a;
    	gcd(b,a%b);
    }
    bool prime(long a)
    {
    	if(a==1)
    	return false;
    	for(i=2;i*i<=a;i++)
    	{
    		if(a%i==0)
    		return false;
    	}
    	return true;
    }
