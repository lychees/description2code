#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t){
    	int n,p;
    	cin>> n>> p;
    	int pr[n];
    	int f[n];
    	for(int it=1;it<=n;it++){
    		cin>> pr[it]>> f[it];
    	}
    	
    	vector<vector<int> > dp(n+1,vector<int>(p+1,0));
    	
    	for(int i=1;i<=n;i++){
    		for(int j=1;j<=p;j++){
    			if(j-f[i]>=0)
    				dp[i][j]=max(dp[i-1][j-f[i]]+pr[i],dp[i-1][j]);
    			else
    				dp[i][j]=dp[i-1][j];
    		}
    	}
    	
    	int maxi=0;
    	for(int i=1;i<=p;i++){
    		if(dp[n][maxi]-maxi<dp[n][i]-i){
    			maxi=i;
    		}
    	}
    	
    	cout<<maxi<<" "<<dp[n][maxi]+p-maxi<<endl;
    	
    	t--;
    }
    return 0;
}
