def solution(t):
    a = t[1] - t[0]
    b = t[2] - t[1]
    return a-1 if a > b else b-1


T = int(raw_input())
for i in xrange(T):
    line = raw_input()
    u = line.split()
    t = map(int, u)
    print solution(t)
