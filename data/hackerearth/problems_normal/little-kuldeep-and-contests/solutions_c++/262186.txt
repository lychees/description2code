#include <iostream>
#include<cstdio>
using namespace std;

struct timing
{
	int start;
	int finish;
};
int comp(const void *a,const void *b)
{

	timing *x,*y;
	x=(timing*)a;
	y=(timing*)b;
	if(x->finish<y->finish)
		return -1;
	else if(x->finish>y->finish)
		return 1;
	else 
		return 0;
		
}
int main()
{
	int flag;
    int n,sh,sm,fh,fm;
    timing t[150];
    cin>>n;
    
    for(int i=0;i<n;i++)
    {
    	scanf("%d:%d-%d:%d",&sh,&sm,&fh,&fm);
    	t[i].start=sh*60+sm;
		t[i].finish=fh*60+fm;    	
    }
    flag=0;
    qsort(t,n,sizeof(timing),comp);
    for(int i=0;i<n-1;i++)
    {
    	if(t[i].finish>t[i+1].start)
    	{	
    		flag=1;
    		break;
    	}
	}
	if(flag)
		cout<<"Will need a moderator!";
	else
		cout<<"Who needs a moderator?";
    return 0;
}
