#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int x=0, big=0;
    string y, biggest;
    
    while(cin >> y){
    	for(int i = 0; i < y.length(); i++){
    		x+=y[i] - '0';
    	}
    	
    	if(big <= x){
    		big = x;
    		biggest = y;
    	}
    		
    	x = 0;
    }
    
    cout << biggest << endl;
}
