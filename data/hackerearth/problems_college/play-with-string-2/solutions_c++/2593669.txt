#include<bits/stdc++.h>
using namespace std;
int main()
{
    string s;
    int T;
    cin>>T;
    while(T--)
    {
    cin>>s;
    bool fg=false;
    int i,j;
    for(i=1,j=s.size()-2;i<s.size();i++,j--)
    {
        char ch1=s[i],ch2=s[i-1],ch3=s[j],ch4=s[j+1];
        int t1=ch1,t2=ch2,t3=ch3,t4=ch4;
      //  cout<<t1<<" "<<t2<<" "<<t3<<" "<<t4<<endl;
        t1-=t2;
        t3-=t4;
        if(t1<0)
            t1*=-1;
        if(t3<0)
            t3*=-1;
    if(t1!=t3)
    {
        fg=true;
        break;
    }
    }
    if(fg)
        cout<<"Not Funny"<<endl;
    else
        cout<<"Funny"<<endl;
    }
}
