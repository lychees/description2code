#include <iostream>
using namespace std;
 
long long a[1000010];
 
int main()
{
    int n;
    cin >> n;
    for (int i = 0; i < n; i++) {
    	cin >> a[i];
    }
    long long ans = a[0];
    for (int j = 1; j < n; j++) {
    	ans += (j + 1)*(a[j] - 1) + 1;

    }
    cout << ans<< endl;
    return 0;
}
