#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const double eps = 1e-10;
const int maxn = 505;
const int maxm=maxn*maxn;
const int inf = 0x3fffffff;
const int mod = 1000000007;

ll f[maxn][maxn],A[maxm];
int n,m;

void init()
{
    A[0]=1;
    for(int i=1; i<maxm; i++)
        A[i]=A[i-1]*2%mod;
}

int main()
{
    init();
    scanf("%d%d",&n,&m);
    for(int i=0; i<=n; i++)
    f[i][0]=1;
    
    for(int j=1; j<=n; j++)
        for(int i=1; i<=m; i++)
        {
            for(int k=0; k<j; k++)
            {
                ll tt=(A[j-k]+mod-1)%mod*A[(j-k)*k]%mod;
                f[j][i]=(f[j][i]+f[k][i-1]*tt%mod)%mod;
            }
        }
    printf("%lld\n",f[n][m]);
    return 0;
}
