#include<bits/stdc++.h>
#include<iostream>
using namespace std;
#define fre 	freopen("0.in","r",stdin),freopen("0.out","w",stdout)
#define abs(x) ((x)>0?(x):-(x))
#define M 1000000000007LL
#define lld signed long long int
#define pp pop_back()
#define ps(x) push_back(x)
#define mpa make_pair
#define pii pair<int,int>
#define fi first
#define se second
#define scan(x) scanf("%d",&x)
#define print(x) printf("%d\n",x)
#define scanll(x) scanf("%lld",&x)
#define printll(x) printf("%lld\n",x)
#define boost ios_base::sync_with_stdio(0)
//vector<int> g[2*100000+5];int par[2*100000+5];
int dad[100000+5],rnk[100000+5],lt[100000+5],rt[100000+5];
void call(int n)
{
	for (int i = 1; i <=n; i++)
	{
		rnk[i] = 0;
		dad[i] = i;
		lt[i] = i-1;
		rt[i] = i+1;
	}
}
int root(int v)
{
	if (v == dad[v]) return v;
	return dad[v] = root(dad[v]);
}
void join(int v1, int v2)
{
	v1 = root(v1), v2 = root(v2);
	if (v1 == v2) return;
	if (rnk[v1] < rnk[v2]) swap(v1, v2);
	dad[v2] = v1;
	lt[v1]=min(lt[v1],lt[v2]);
	rt[v1]=max(rt[v1],rt[v2]);
	if (rnk[v1] == rnk[v2]) rnk[v1]++;
}

vector<pair<lld,lld> >V;
lld arr[100000+5];

int main()
{
//	fre;
	int t,n;
	lld x;
	cin>>t;
	while(t--)
	{
		cin>>n;

		arr[n+1]=M;
		arr[0]=M;

		call(n);

		V.clear();
		V.ps(mpa(0,M));

		lld ans=0;
		for(int i=1;i<=n;++i)
		{
			cin>>x;
			V.ps(mpa(x,i));
			arr[i]=x;
		}

		sort(V.begin(),V.end());

		for(int i=1;i<n; ++i)
		{
			int at = root(V[i].se);
			int x = root(lt[at]);
			int y = root(rt[at]);
			
			if(arr[x]<arr[y])
			{
				ans+=arr[x];
				join(x,at);
				arr[root(at)]=arr[x];
			}
			else
			{
				ans+=arr[y];
				join(y,at);
				arr[root(at)]=arr[y];
			}
		}
		cout<<ans<<endl;
		for(int i=0;i<=n+1;++i)
		{
			arr[i]=0;
			rnk[i] = 0;
			dad[i] = 0;
			lt[i] = 0;
			rt[i] = 0;
		}
	}
}
