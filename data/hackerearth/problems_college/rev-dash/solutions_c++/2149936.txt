#include <iostream>
using namespace std;

int main()
{
    string s;
    cin >> s;
    
    for(int i=0;i<s.length()-1;i++)
    {
    	cout << s.at(i) << "-";
    }
    cout << s.at(s.length()-1) << endl;
    
    for(int i=0;i<s.length()-1;i++)
    {
    	cout << s.at(s.length()-1-i) << "-";
    }
    cout << s.at(0) << endl;
}
