#include <bits/stdc++.h>

#define ll long long

using namespace std;

int main() {
    int t, n, itr, flag;

    cin >> t;
    while (t--) {
        cin >> n;
        int a[n];
        for (int i = 0; i < n; i++) cin >> a[i];
        flag = 0;
        itr = 0;
        while(n--) {
            if (a[itr] == -1) {
                flag = 1;
                break;
            } else {
                itr = a[itr];
            }
        }
        if (flag == 1)
            cout << "Win\n";
        else
            cout << "Lose\n";
    }

    return 0;
}
