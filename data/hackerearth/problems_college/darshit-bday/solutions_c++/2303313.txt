#include <iostream>
using namespace std;

int main()
{
    int t,result;
    cin>>t;
    while(t--)
    {long long int a,b,x,y,z,result=0;
    cin>>a>>b;
    cin>>x>>y>>z;
     result=a*x+b*y;
   if(result>((y+z)*a+(y)*b))
     result=(y+z)*a+y*b;
       if(result>(x*a+(x+z)*b))
       result=x*a+(x+z)*b;
     cout<<result<<endl;
    }
    
    return 0;
}
