#include <cstdio>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
	int n,b;
	scanf("%d %d",&n,&b);
	int tmp = b;
	while(tmp<=n)
	{
	    tmp+=b;
	}
	int n1 = tmp-b;
	int n2 = tmp;
	if(n2-n==n-n1)
	{
	    printf("%d\n",n2);
	}
	else if(n2-n<n-n1)
	{
	    printf("%d\n",n2);
	}
	else
	{
	    printf("%d\n",n1);
	}
    }
    return 0;
}
