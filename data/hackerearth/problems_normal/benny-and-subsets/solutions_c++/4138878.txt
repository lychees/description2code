#include <bits/stdc++.h>
#define mod 10000007
#define ll long long 
using namespace std;
vector<int>a, b;
int dp1[1001][(1<<11)+1], dp2[21][(1<<21)+1];

int main()
{
    int n, x;
    scanf("%d %d", &n, &x);
    for(int i=0; i<n; i++){
    	int xx;
    	scanf("%d", &xx);
    	if(xx<(1<<11)){
    		a.push_back(xx);
    	}
    	else{
    		b.push_back(xx);
    	}
    }
    //cout<<"aaa\n";
    	int n1=a.size(), n2=b.size();
    	dp1[0][0]=1;
    	for(int i=1; i<=n1; i++){
    		for(int j=0; j<(1<<11); j++){
    			dp1[i][j]=dp1[i-1][j];
    			if((j^a[i-1])>=0){
    				dp1[i][j]+=dp1[i-1][j^a[i-1]];
    			}
    			dp1[i][j]%=mod;
    		}
    	}
    	dp2[0][0]=1;
    	for(int i=1; i<=n2; i++){
    		for(int j=0; j<(1<<21); j++){
    			dp2[i][j]=dp2[i-1][j];
    			if((j^b[i-1])>=0){
    				dp2[i][j]+=dp2[i-1][j^b[i-1]];
    			}
    			dp2[i][j]%=mod;
    		}
    	}
    	
    	/*for(int i=0; i<n1; i++){
    		for(int j=0; j<4; j++){
    			cout<<dp1[i][j] << " ";
    		}
    		cout<<endl;
    	}*/
    	
    	ll ans=0;
    	
    		for(int i=0; i<(1<<11); i++){
    			ans=ans+(ll)dp1[n1][i]*dp2[n2][x^i]%mod;
    			ans%=mod;
    		}
    	
    	printf("%lld\n", ans);
    return 0;
}
