#include <bits/stdc++.h>
using namespace std;

int main()
{
    vector <int> v;
    int l, r, vsize;
    cin >> l >> r;
    for(int i = 1; i <= l; i++){
    	v.push_back(i);
    }

    int tr = r, erase_idx = 0;
    vsize = v.size();
    int rem = vsize;
    while(vsize > 1){
    	r = tr;
    	if(r > rem) {r = r-rem-1; r %= vsize; erase_idx = r;} // Process r
    	else{
    		erase_idx += r-1; 
    	}
    	//cout << "Erasing " << *(v.begin()+erase_idx) << endl;
    	//cout << "erase_idx " << erase_idx << endl;
    	v.erase(v.begin()+erase_idx);
    	for(int i = 0; i < v.size();i++){
    		//cout << v[i] << " ";
    	}
    	//cout << endl;
    	rem = ((vsize - 1) - erase_idx);
		vsize--;
    }
    cout << v[0] << endl;
    return 0;
}
