#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    string s;
    map<string,int> mp;
    mp.clear();
    map<string,int> ::iterator it;
    while(t--)
    {
        cin>>s;
        it=mp.find(s);
        if(it==mp.end())
        {
            mp[s]=1;
        }
        else
        {
            it->second++;
        }
    }
    int m=0,M=0;
    for(it=mp.begin();it!=mp.end();it++)
    {
        if(it->second>M)
        {
            M=it->second;
            m=M;
        }
        else if(it->second==M)
        {
            m+=M;
        }
    }
    printf("%d\n",m);
    return 0;
}
