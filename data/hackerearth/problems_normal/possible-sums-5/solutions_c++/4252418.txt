#include<bits/stdc++.h>
using namespace std;
#define ll  long long
#define mod 1000000007
ll dp[102][10003];
ll arr[10003];
int main()
{

  int t;
  cin>>t;
  while(t--)
  {
      int n,i,j;
      cin>>n;
      for(i=1;i<=n;i++)
      {
          cin>>arr[i];
      }
      for(i=0;i<=n;i++)
        dp[i][0]=1;
      for(i=1;i<=n;i++)
      {
          for(j=1;j<=100*n;j++)
          {
              if(j-arr[i]>=0)
              dp[i][j]=dp[i-1][j]|dp[i-1][j-arr[i]];
              else
                dp[i][j]=dp[i-1][j];
          }

      }
      ll sum=0;
      for(i=0;i<=100*n;i++)
      {
          if(dp[n][i]!=0)
            sum++;
      }
      cout<<sum<<endl;
  }
}
