#include <bits/stdc++.h>

typedef long long int li;

#define min(a,b) a<=b?a:b
#define max(a,b) a>=b?a:b
#define Fi(i, st, ft) for(i = st;i <= ft; ++i)
#define Fd(i, st, ft) for(i = st;i >= ft; --i)
#define pb(a, b) a.push_back(b)
#define mod 1000000007
#define MAXN 1500000
#define MX 1450000
#define inf 1000000000000000000ll
const int maxBufSize = (10000000);

using namespace std;

li n , m , q;
li bits[5505];

int main()
{
	li i , j , a , b , c , mask , ans;
	scanf("%lld %lld" , &n , &q);
	Fi(i , 1 , n) {
		a = i;
		vector<li> v , temp;
		li chk[15];
		c = 0;
		memset(chk , 0 , sizeof(chk));
		while(a!=0) {
			if(chk[a%10] == 0) {
				chk[a%10] = 1;
				pb(v , a%10);
				c++;
			}
			a = a/10;
		}
		Fi(mask , 1 , (1<<c)-1) {
			b = 0;
			Fi(j , 0 , c-1) {
				if(mask & (1<<j)) {
					b = b | (1<<v[j]);
				}
			}
			bits[b]++;
		}
	}
	while(q--) {
		scanf("%lld" , &m);
		vector<li> v;
		Fi(i , 1 , m) {
			scanf("%lld" , &a);
			pb(v , a);
		}
		ans = 0;
		Fi(mask , 1 , (1<<m)-1) {
			b = 0;
			a = 0;
			Fi(i , 0 , m-1) {
				if(mask & (1<<i)) {
					b++;
					a = a | (1<<v[i]);
				}
			}
			if(b%2 == 1) {
				ans += bits[a];
			} else {
				ans -= bits[a];
			}
		}
		printf("%lld\n" , ans);
	}
	return 0;
}
