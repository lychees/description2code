ll dp[19][1<<19],k,n;
ll fun(ll i,ll mask){
	if(mask==((1<<(n+1))-2))
		return 1;
	if(dp[i][mask]!=-1)
		return dp[i][mask];
	ll sum=0,j;
	//pfn2(i,mask);
	for(j=1;j<=n;j++){
		//pfn1(!((1<<j)&mask));
		if(!((1<<j)&mask)){
			if(abs(3*i-2*j)<=k)
				sum+=fun(j,mask|1<<j);
		}
	}
	return dp[i][mask]=sum;
}
int main(){
 ll t,i,j,ans;
 sf(t);
 while(t--){
	sf(n);sf(k);
	ans=0;
	for(i=1;i<=n;i++){
		for(j=1;j<=1<<(n+2);j++)
			dp[i][j]=-1;
	}
	for(i=1;i<=n;i++){
		ans+=fun(i,1<<i);
	}
	printf("%lld\n",ans);
 }
 return 0;
}
