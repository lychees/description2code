#include<bits/stdc++.h>
using namespace std;
int main()
{
	int test;
	cin>>test;
	while(test--)
	{
		string str;
		cin>>str;
		int cnt1=0,cnt0=0;
		int n=str.length();
		for(int i=0;i<str.length();i++)
		{
			if(str[i]=='0')
			cnt0++;
			else
			cnt1++;
		}
	//	cout<<cnt1<<" "<<cnt0<<endl;
		int ans = n-2*min(cnt0,cnt1);
		cout<<ans<<endl;
	}
}
