from itertools import combinations
T = int(raw_input())
for t in xrange(T):
    raw_input()
    nums = map(int ,raw_input().split())
    S = int(raw_input())
    done = False
    if S == 0 or S in nums:
        done = True
    else:
        for r in xrange(len(nums), 0, -1):
            if done:
                break
            for c in combinations(nums, r):
                if sum(c) == S:
                    done = True
                    break
    print "YES" if done else "NO"
