#include <iostream>
using namespace std;

int main()
{ int t;
cin>>t;
while (t--){
int n,x=0,o=0,c=0;
string s;
cin>>s;
n= s.length();
for(int i=0;i<n;i++)
{
    		if(s[i]=='X') x++;
    		else if(s[i]=='O') o++;
    		else if(s[i]=='*')
    		{
    			if(x==0) c=c+o;
    				x=0;
    				o=0;
    		}
    	}
    	if(x==0) c=c+o;
    	cout<<c<<endl;
    }
    return 0;

    }
