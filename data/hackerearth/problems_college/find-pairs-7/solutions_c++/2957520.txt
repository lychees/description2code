#include <bits/stdc++.h>
using namespace std;
long long a[1000000+2],b[1000000+3];
int main() {
	int t;
	cin>>t;
	while(t--)
	{
	for(int i=0;i<1000000;i++) a[i]=b[i]=0;
	int x;
		int n;
		cin>>n;
		for(int i=0;i<n;i++) {cin>>x;
		if(x>=0) a[x]++;
		else b[-1*x]++;
		}
		long long count=0;
		for(int i=0;i<1000000;i++)
		{
		if(a[i]) {count+=((a[i]*(a[i]+1))/2);}
		if(b[i]) {count+=((b[i]*(b[i]+1))/2);}
		}
		cout<<count<<endl;
	}
	return 0;
}
