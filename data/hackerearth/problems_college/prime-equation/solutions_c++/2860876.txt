#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int prime[10000002];
vector<int>x;
void sieve()
{
	int i,j;
	memset(prime,0,sizeof(prime));
	for(i = 2; i <= 10000000; i++) {
		if(!prime[i]) {
			for(j = 2; j*i <= 10000000; j++) {
				prime[j*i] = 1;
			}
			x.push_back(i);
		}
	}
}
map<int,int>mp;
int arr[10000002];
int main()
{
	int t,n,i,k;
	sieve();
	for(i = 2; i < x.size(); i++) {
		mp[x[i]-x[i-1]]++;
		arr[x[i]-x[i-1]]++;
	}
	scanf("%d",&t);
	while(t--) {
		scanf("%d",&n);
		if(n%2) {
			puts("NO");
			continue;
		}
		bool flag = false;
		for(auto it = mp.begin(); it != mp.end(); it++) {
			if(it->first >= n) {
				break;
			}
			else {
				k = n-it->first;
				if(k == it->first) {
					if(arr[k]) {
						flag = true;
						break;
					}
				}
				else {
					if(arr[k]) {
						flag = true;
						break;
					}
				}
			}
		}
		puts(flag?"YES":"NO");
	}
	return 0;
}
