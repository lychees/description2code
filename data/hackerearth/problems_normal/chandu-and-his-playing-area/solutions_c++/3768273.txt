#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
       ll n;
       cin>>n;
       if(n%4==0||n%4==1)
       {
           ll temp=n/4;
           cout<<temp*temp<<endl;
       }
       else
       {
           ll temp=n/4;
           cout<<temp*(temp+1)<<endl;
       }

   }

}
