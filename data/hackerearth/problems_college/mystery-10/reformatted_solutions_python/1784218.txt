"""
its a problem which uses mapping of character to character in encrypting the input sentence
here it follows a standard keyboard format
"""
mapdict = {
    char: transform for char,
    transform in zip(
        "abcdefghijklmnopqrstuvwxyz",
        "qwertyuiopasdfghjklzxcvbnm")}

for _ in range(input()):
    mstring = raw_input()
    newword = ''
    for char in mstring:
        if mapdict.get(char, "not") != "not":
            newword += mapdict[char]
        else:
            newword += char
    print newword
