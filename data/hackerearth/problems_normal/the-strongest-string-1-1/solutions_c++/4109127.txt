#include<iostream>
#include<cstring>
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
int arr[27];
int main()
{
	long int n;
	cin>>n;
	string str;
	cin>>str;
	int i=0;
	while(i<n)
	{
		int max=-1;
		char maxch='A';
		for(int j=i;j<n;j++)
		{
			if(str[j]>maxch && arr[str[j]-97]==0)
			{
				maxch=str[j];
				max=j;
			}
		}
		if(max==-1)
		break;
		cout<<maxch;
		i=max+1;
		arr[str[max]-97]=1;
	}
}
