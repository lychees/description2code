#include <iostream>
#define max 100001
using namespace std;

int main()
{
	int n,arr[max],t;
	cin>>t;
	while(t--)
	{
		int cnt=0;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>arr[i];
			if(arr[i]%2==0)
			{
				cnt++;
			}
		}
		cout<<cnt<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
