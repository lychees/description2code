using namespace std;
typedef long long ll;
#define mod 1000000007
#define pb push_back

vector<int> v[100002];
int N;
int len = 0;
int visited[100002];

void dfs(int index){
    visited[index] = 1;
    len++;
    for(int i=0;i<v[index].size();i++){
        if(!visited[v[index][i]])
            dfs(v[index][i]);
    }
}

ll fact[100001];

void precalc(){
    fact[0] = 1;
    fact[1] = 1;
    for(int i=2;i<=100000;i++){
        fact[i] = (i*fact[i-1])%mod;
        if(fact[i]>mod)
            fact[i]-=mod;
    }
}

int main(){
    //ios::sync_with_stdio(false);
    #ifdef DEBUG
    freopen("input.txt", "r", stdin);
    #endif // DEBUG

    int M,a,b;
    precalc();
    scanf("%d %d", &N, &M);
    while(M--){
        scanf("%d %d", &a, &b);
        v[a].pb(b);
        v[b].pb(a);
    }
    ll ans=1;
    for(int i=0;i<N;i++){
        if(!visited[i]){
            len = 0;
            dfs(i);
            ans = (ans*fact[len])%mod;
            if(ans>mod)
                ans-=mod;
        }
    }
    printf("%lld\n", ans);

    return 0;
}
