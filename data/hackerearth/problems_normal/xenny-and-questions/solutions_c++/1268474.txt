int readInt () {
	bool minus = false;
	int result = 0;
	char ch;
	ch = getchar();
	while (true) {
		if (ch == '-') break;
		if (ch >= '0' && ch <= '9') break;
		ch = getchar();
	}
	if (ch == '-') minus = true; else result = ch-'0';
	while (true) {
		ch = getchar();
		if (ch < '0' || ch > '9') break;
		result = result*10 + (ch - '0');
	}
	if (minus)
		return -result;
	else
		return result;
}

int main()
{
   // ios_base::sync_with_stdio(0);
    int t,n,k;
    cin>>t;
    while(t--){
        n=readInt(),k=readInt();
        int a[n];
        for(int i=0;i<n;i++)
            a[i]=readInt();
        sort(a,a+n,greater<int>());
        ll sum=0;
        for(int i=0;i<min(k,n);i++)
            sum+=a[i];
        cout<<sum<<endl;
    }
    return 0;
}
