#include<bits/stdc++.h>
using namespace std;

const int N=1e5+5;
int dp[N];

int pow(int a,int b) {
	int ans=1;
	for(int i=1;i<=b;i++) {
		ans*=a;
	}
	return ans;
}
int main() {
  //freopen("a.cpp","r",stdin);
  dp[1]=5;
  int idx;
  int resp=2;
  for(int i=2;i<N;i++,resp*=2) {
    dp[i]=resp*5;
    dp[i]+=dp[i-1];
    if(dp[i]>1e9) {
      idx=i;
      break;
    }
  }
  int n;
  scanf("%d",&n);
  int lo=1,hi=idx;
  int ans;
  while(lo<=hi) {
    int mi=(lo+hi)/2;
    if(dp[mi]>=n&&dp[mi-1]<=n) {
      ans=mi;
      break;
    } else if(dp[mi]<n) {
      lo=mi+1;
    } else {
      hi=mi-1;
    }
  }
  n-=dp[ans-1];
  int res=n/pow(2,ans-1);
  if(n%pow(2,ans-1)) {
    res+=1;
  }
  if(res==1) {
    cout<<"Sheldon";
  } else if(res==2) {
    cout<<"Leonard";
  } else if(res==3) {
    cout<<"Penny";
  } else if(res==4) {
    cout<<"Rajesh";
  } else {
    cout<<"Howard";
  }
  return 0;
}
