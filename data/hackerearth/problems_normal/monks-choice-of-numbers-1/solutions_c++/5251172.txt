#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int compare (int a, int b) {
	return (a > b);
}

int no_of_ones (int temp) {
	int cnt = 0;
	while (temp){
		temp = temp & (temp - 1);
		cnt++;
	}
	
	return cnt;
}

int main()
{
    int t, n, k, ans;
    int arr[1000000];
    scanf ("%d", &t);
    while (t--) {
    	scanf ("%d %d", &n, &k);
    	for (int i = 0; i < n; i++){
    		scanf ("%d", &arr[i]);
    		arr[i] = no_of_ones (arr[i]);
    	}
    	sort (arr, arr + n, compare);
    	ans = 0;
    	for (int i = 0; i < k; i++)
    		ans += arr[i];
    	printf ("%d\n", ans);
    }
    return 0;
}
