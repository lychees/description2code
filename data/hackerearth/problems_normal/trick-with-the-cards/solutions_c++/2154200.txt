#include <iostream>
using namespace std;

#define ll long long int

int main()
{
    ll t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	if(n==0){
    		cout<<0<<endl;
    		continue;
    	}
    	else if(n<=3){
    		cout<<1<<endl;
    		continue;
    	}
    	if(n%3==0)
    		cout<<n/3<<endl;
    	else
    		cout<<n<<endl;
    	
    }
    return 0;
}
