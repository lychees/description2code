import math
o = []
t = input()
for i in range(0, t):
    l = raw_input()
    x1 = int(l.split()[0])
    y1 = int(l.split()[1])
    x2 = int(l.split()[2])
    y2 = int(l.split()[3])

    r = input()

    num = (((x2-x1)*y1) - (x1*(y2-y1)))
    den = math.sqrt((x2-x1)**2+(y2-y1)**2)

    d = num/den

    if d < 0:
        d *= -1

    if d == r:
        o.append("JUST MISSED")
    elif d < r:
        o.append("REPLANNING")
    else:
        o.append("SAFE")

for e in o:
    print e
