#include <iostream>
using namespace std;

int main()
{
    int l;
    scanf("%d", &l);
    string s;
    cin >> s;
    long int sum[100001];
    sum[0] = 0;
    for(int i = 1; i <= l; i++) {
    	sum[i] = sum[i - 1] + s[i - 1] - '0';
    }
    int n;
    scanf("%d", &n);
    for(int i = 0; i < n; i++) {
    	int start, end;
    	scanf("%d %d", &start, &end);
    	if(!((sum[end] - sum[start - 1]) % 9))
    		printf("SAMEER\n");
    	else
    		printf("NO\n");
    }
    return 0;
}
