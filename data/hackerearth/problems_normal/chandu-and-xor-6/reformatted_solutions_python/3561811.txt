'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
n, m = map(int, raw_input().split())
c = [0 for x in xrange(m)]
r = [0 for x in xrange(n)]
maxor = -1
for i in xrange(n):
    temp = map(int, raw_input().split())
    for j in xrange(m):
        c[j] += temp[j]
    r[i] = sum(temp)
for i in xrange(n):
    for j in xrange(m):
        xor = r[i] ^ c[j]
        maxor = max(maxor, xor)
print maxor
