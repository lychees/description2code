#include<bits/stdc++.h>
#define ll long long int
using namespace std;
vector<ll> p;
ll limuplow(ll val,ll low,bool up)
{
	ll h=p.size(),l=low,mid;
	while(l<h)
	{
		mid=(l+h)/2;
		if(up)
		{
			if(p[mid]>val)
			{
				h=mid;
			}
			else
			{
				l=mid+1;
			}
		}
		else
		{
			if(p[mid]<val)
			{
				l=mid+1;
			}
			else
			{
				h=mid;
			}
		}
	}
	if(up)
	{
		return h;
	}
	else
	{
		return l;
	}
}
int main()
{
	ll q,l,r,i,j,ans;
	for(i=1;;i++)
	{
		ll temp=(i*(i+1))/2;
		p.push_back(temp);
		if(temp>1e12)
		{
			break;
		}
	}
	ll len=p.size();
	scanf("%lld",&q);
	while(q--)
	{
		ans=0;
		scanf("%lld%lld",&l,&r);
		for(i=0;i<len && p[i]<=r;i++)
		{
			ans+=limuplow(r-p[i],i,true)-limuplow(l-p[i],i,false);
			//printf("%lld\n",ans);
		}
		printf("%lld\n",ans);
	}
	return 0;
}
