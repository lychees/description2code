#include<iostream>
#include<stdlib.h>
using namespace std;
int main()
{
	int t,c;
	cin>>t;
	if(t>10||t<1)
	exit(0);
	int a[10];
	char x[100000];
	for(int i=0;i<t;i++)
	{
		c=0;
		cin>>x;
		for(int j=0;x[j]!='\0';j++)
		{
			if(x[j]=='0'||x[j]=='6'||x[j]=='9')
			c=c+6;
			if(x[j]=='1')
			c=c+2;
			if(x[j]=='2'||x[j]=='3'||x[j]=='5')
			c=c+5;
			if(x[j]=='7')
			c=c+3;
			if(x[j]=='4')
			c=c+4;
			if(x[j]=='8')
			c=c+7;
		}
		a[i]=c;
	}
	for(int i=0;i<t;i++)
	cout<<a[i]<<endl;
	return 0;
}
