#include <cstring>
#include <algorithm>
#include <iostream>
#include <unordered_map>
#define read(x) scanf("%d",&x)
#define write(x) printf("%d ",x)

using namespace std;
 
int main()
{
    int n,t;
    cin >> t;
    while(t--)
    {
        read(n);
        int arr[n],temp[n];
        for (int i=0; i<n; i++)
            read(arr[i]);
        /*for (int i=0; i<n; i++)
            cout << arr[i];*/
        memcpy(temp, arr, n*sizeof(int));
        sort(temp, temp + n); 
        
        unordered_map<int, int> umap;
        int val = 0;
        for (int i = 0; i < n; i++)
            if(umap.find(temp[i]) == umap.end())
                umap[temp[i]] = val++;
 
        for (int i = 0; i < n; i++)
            write(umap[arr[i]]);
        printf("\n");
    }
    return 0;
}
