#include <iostream>

using namespace std;

typedef long long ll;

long long findNumCom(long long n)
{
	long long val=n/6;
    val-=1;
    ll ans=0;
    ll tp=1ll*val*(val+1)/2;
    tp*=6;
    ans+=tp;
    ans+=(val+1)*5ll;
    ans+=(val+2);
    int ts=n%6;
    int qs=1;
    while(qs<=ts) {
      if(qs==1) {
        ans+=(val+1);
      } else {
        ans+=(val+2);
      }
      qs++;
}

return ans;
}

int main() {
  int q;
  cin>>q;
  while(q--) {
    int n;
    cin>>n;
    
    long long ans=findNumCom(n);
    cout<<ans<<endl;
  }
  return 0;
}
