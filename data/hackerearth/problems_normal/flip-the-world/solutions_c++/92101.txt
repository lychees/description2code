#include <iostream>
using namespace std;
int a[100][100]={0};
int findzerocol(int r, int c)
{
	for(int i=r; i>=0; i--)
		if(a[i][c]==0)
			return i;
	return -1;
}

int findzerorow(int r, int c)
{
	for(int i=c; i>=0; i--)
		if(a[r][i]==0)
			return i;
	return -1;
}

int main()
{
    int t,r,c;
    cin>>t;
    while(t--)
    {
    	cin>>r>>c;
    	char str[100]={0};
    	for(int i=0; i<r; i++)
    		{
    			cin>>str;
    			for(int k=0; k<c; k++)
    				a[i][k] = str[k]-'0';
    		}
    			
    	int ch=0;
    	r--;c--;
    	int rf=r, cf=c, count=0;
    	while(ch!=1)
    	{
    		int col = findzerocol(r,c);
    		if(col==-1)
    		{
    			c--;
    			int row = findzerorow(r,c);	
    			if(row==-1)
    				r--;
    			else
    			{
    				count++;
    				for(int i=0; i<=r; i++)
    				for(int j=0; j<=row; j++)
    					a[i][j] = !(a[i][j]);
    			}
    		}
    		else
    		{
    			count++;
    			for(int i=0; i<=col; i++)
    				for(int j=0; j<=c; j++)
    					a[i][j] = !(a[i][j]);
    		}
    		int check=0;
    		for(int i=0; i<=rf; i++)
    		for(int j=0; j<=cf; j++)
    		if(a[i][j]!=1) {check=1; break;}
    		if(check==0) ch=1;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
