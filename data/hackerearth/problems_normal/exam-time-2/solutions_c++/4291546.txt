#include <iostream>
#define MAX 1000000
using namespace std;
long long phi[MAX + 2], mark[MAX + 2];
//Phi from 1 to n
void sievephi(int n) {
	int i, j;
	for (i = 0; i <= n; i++)phi[i] = i;
	mark[1] = 1;
	for (i = 2; i <= n; i++) {
		if (!mark[i]) {
			for (j = i; j <= n; j += i) {
				mark[j] = 1;
				phi[j] = phi[j] / i * (i - 1);
			}
		}
	}
}

int main()
{
    sievephi(MAX);
    int t, n;
    cin>>t;
    while(t--){
    	cin>>n;
    	cout<<phi[n]<<endl;
    }
}
