#include <iostream>
using namespace std;

#include <cstdio>
#include <cmath>
#include <algorithm>

int main()
{
	int t,n,a[10000],i,sum=0;
    scanf("%d",&t);
    while(t--)
    {
    	sum=0;
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&a[i]);
    	}
    	sort(a,a+n);
    	for(i=1;i<n;i++)
    	{
    		sum+=abs(a[i]-a[i-1]);
    	}
    	printf("%d\n",2*sum);
    }
    
    return 0;
}
