#include<bits/stdc++.h>
#define lli long long int
using namespace std;
int main()
{
        lli c,i,j,k,m,n,t,count,max;
        scanf("%lld",&t);
        while(t--)
        {
                scanf("%lld",&n);
                lli arr[n];

                for(i=0;i<n;i++)
                        scanf("%lld",&arr[i]);

                sort(arr,arr+n);
                c=arr[0];
                count=1;
                max=0;
                for(i=1;i<n;i++)
                {
                        if(c==arr[i])
                                count++;
                        else
                        {
                                c=arr[i];
                                if(count>max)
                                        max=count;
                                count=1;
                        }
                }
                if(count>max)
                        max=count;
                printf("%lld\n",max);
        }
}
