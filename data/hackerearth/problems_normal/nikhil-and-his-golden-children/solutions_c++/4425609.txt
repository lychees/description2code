#include <bits/stdc++.h>

using namespace std;

long long int modinverse(long long int x){
	long long int n=1000000005;
	long long int ans=1;
	while(n){
		if(n&1)
			ans=(ans*x)%1000000007;
		n=n>>1;
		x=(x*x)%1000000007;
	}

	return ans;
}

int main()
{
    int t;
    cin >> t;
    
    while(t--)
    {
    	long long int k,m;
    	
    	cin >> k >> m;
    	
    	int sum=0,b;
    	
    	for(int i=0;i<k;i++)
    	{
    		cin >> b;
    		sum+=b;
    	}
    	
    	m=m-sum;
    	if(m<0){
    		cout<<0<<endl;
    		continue;
    	}
    
    	long long int ans=1;
    	for(int i=m+k-1;i>m;i--){
    		ans=(ans*i)%1000000007;
    	}
    	long long denom=1;
    	for(int i=1;i<=k-1;i++){
    		denom=(denom*i)%1000000007;
    	}
    	
    	ans=(ans*modinverse(denom))%1000000007;
    	cout<<ans<<endl;
 
    }
    return 0;
}
