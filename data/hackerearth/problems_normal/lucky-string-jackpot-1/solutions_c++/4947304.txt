#include <iostream>
using namespace std;

using ull = unsigned long long;

ull sn[46];
ull invans[90];

void prepare() {
	sn[0] = 0;
	for (int i = 1; i < 46; ++i)
		sn[i] = sn[i - 1] + i * (1ull << i);

	for (int i = 0; i < 90; ++i)
		invans[i] = sn[i / 2] + i;
}



int main() {
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	prepare();

	int T;
	cin >> T;
	while (T--) {
		ull n;
		cin >> n;
		int i = 0;
		while (invans[i] <= n)
			++i;
		cout << i - 1 << '\n';
	}
	return 0;
}
