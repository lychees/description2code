#include <iostream>
using namespace std;

int main()
{
	int t,i;
	string ch;
	cin>>t;
	int total;
	int count;
	while(t--){
		cin>>ch;
		total =ch.length();
		count = 0;
		for(i = 4; i < ch.length() ;i++ ){
			if(ch[i] != 'a' && ch[i] != 'i' &&ch[i] != 'o' && ch[i] != 'u' && ch[i] != 'e' ){
			
				count++;
			}
		}
		count+=1;
		cout<<count<<"/"<<total<<endl;
		
	}
    
    return 0;
}
