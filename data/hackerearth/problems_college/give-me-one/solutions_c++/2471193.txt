#include <iostream>
#include <limits.h>
using namespace std;


int arr[20000005];


int main()
{
	arr[0]=0;
	arr[1]=0;
	arr[2]=1;
	arr[3]=1;
	for(int i=4;i<20000002;i++)
	{
		long long int min=999999999;
		if(i%2==0)
		{
			if(min>arr[i/2]+1)min=arr[i/2]+1;
		}
		if(i%3==0)
		{
			if(min>arr[i/3]+1)min=arr[i/3]+1;
		}
		if(min>arr[i-1]+1)min=arr[i-1]+1;
		arr[i]=min;
	}
	
    int T,N;
    cin>>T;
    while(T--)
    {
    	cin>>N;
    	//cout<<"---"<<endl;
    	cout<<arr[N]<<endl;
    }
    return 0;
}
