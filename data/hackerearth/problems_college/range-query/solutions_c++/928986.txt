#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <cassert>
#include <string>
#include <vector>
#include <algorithm>
#include <set>
#include <map>
#include <stack>
#include <queue>
#include <cstdlib>
using namespace std;

int main()
{
    int N;
    cin>>N;
    int a[N];
    vector<int> ia[101];
    for(int i=0;i<N;i++){
            cin>>a[i];
            ia[a[i]].push_back(i);
    }
    int Q,x,y,n,ans;
    cin>>Q;
    for(int i=1;i<=100;i++){sort(ia[i].begin(),ia[i].end());}

    for(int i=0;i<Q;i++){
        cin>>x>>y>>n;
        ans=0;
        for(int j=0;j<ia[n].size();j++){
            if(ia[n][j]>=x && ia[n][j]<=y){ans++;}
            if(ia[n][j]>y) break;
        }
        cout<<ans<<endl;
    }

}
