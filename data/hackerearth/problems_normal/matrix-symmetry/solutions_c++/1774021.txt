#include <iostream>
using namespace std;

int main()
{
	int t,n;
	cin>>t;
	while(t--){
		cin>>n;
		char a[n][n];
		for(int i=0;i<n;i++){
			for(int j=0;j<n;j++){
				cin>>a[i][j];
			}
		}
		
		int flag=0;
		for(int i=0;i<n/2;i++){
			for(int j=0;j<n;j++){
				if(a[i][j]!=a[n-i-1][j]){
					flag=1;
					break;
				}
			}
		}
		bool horz=false;
		if(flag==0) horz=true;
	
		flag=0;
		for(int j=0;j<n/2;j++){
			for(int i=0;i<n;i++){
				if(a[i][j]!=a[i][n-j-1]){
					flag=1;
					break;
				}
			}
		}
		
		bool vert=false;
		if(flag==0) vert=true;
		
		
		if(vert && horz) cout<<"BOTH\n";
		else if (vert) cout<<"VERTICAL\n";
		else if (horz) cout<<"HORIZONTAL\n";
		else cout<<"NO\n";
		
	}
	
	return 0;
}
