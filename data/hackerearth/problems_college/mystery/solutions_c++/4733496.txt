#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   long long int a,b,c,i,t;
   cin>>t;
   for(;t--;)
   {
   	cin>>a>>b>>c;
   	if(c==0)
   	cout<<(a)<<endl;
   	else if(c==1)
   	cout<<(b)<<endl;
   	else
   	{
   	long long int sum=0;
   		for(i=0;i<c-1;i++)
   		{
   			sum=a+b;
   			a=b;
   			b=sum;
   		}
   		cout<<(sum)<<endl;
   	}
   }
    return 0;
}
