#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int p=sqrt(n),c=0;
		for(int i=1;i<=p;i++)
		{
			if(n%i==0)
			{
				int x=i,y=n/i;
				if(x==y)
				{
					if(x%2==0)
					c++;
				}
				else
				{
					if(x%2==0)
					c++;
					if(y%2==0)
					c++;
				}
			}
		}
		cout<<c<<endl;
	}
	return 0;
}
