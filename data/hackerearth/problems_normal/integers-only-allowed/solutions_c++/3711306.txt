    #include <iostream>
    #include<algorithm>
    #include<cmath>
    #include<cstdio>
    #define ll long long
    #define mod 1000000007
    using namespace std;

    class point
    {
    public:
        ll x, y;
        point(ll a=0, ll b=0):x(a),y(b) {}
    };

    ll gcd(ll a, ll b)
    {
        if (b == 0)
           return a;
        return gcd(b, a%b);
    }

    int getCount(point p, point q)
    {

        if (p.x==q.x)
            return abs(p.y - q.y) - 1;

        if (p.y == q.y)
            return abs(p.x-q.x) - 1;

        return gcd(abs(p.x-q.x), abs(p.y-q.y))-1;
    }
    int main() {
    	ll n;
    	cin>>n;
    	point ar[n];
    	scanf("%lld %lld",&ar[0].x,&ar[0].y);
    	ll B=0,A=0;
    	for(ll i=1;i<n;i++)
    	{
    		scanf("%lld%lld",&ar[i].x,&ar[i].y);
    		B+=getCount(ar[i-1],ar[i]);
    		A+=(ar[i-1].x*ar[i].y)-(ar[i-1].y*ar[i].x);
    	}
    	B+=getCount(ar[n-1],ar[0])+n;
    	A+=(ar[n-1].x*ar[0].y)-(ar[n-1].y*ar[0].x);
    	cout<<(abs(A)+2-B)/2;
    	return 0;
    }
