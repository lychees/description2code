#include <iostream>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(true);
    int n,x,y;
    cin>>n;
    while(n--){
    	cin>>x;
    	cin>>y;
    	if(y>x || (x+y)%2!=0){
    		cout<<"impossible"<<endl;
    		continue;
    	}
    	else{
    		cout<<(x+y)/2<<" "<<(x-y)/2<<endl;
    	}
    }
    
    
    return 0;
}
