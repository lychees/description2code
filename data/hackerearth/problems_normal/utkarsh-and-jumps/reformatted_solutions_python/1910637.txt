
def prob(n, p1, p2):
    mem = [0 for i in xrange(n+3)]
    mem[0] = 0.0
    mem[1] = 0.0
    mem[2] = p1
    mem[3] = p2
    for i in xrange(4, n+1):
        mem[i] = p1*mem[i-2]+p2*mem[i-3]
    return mem[n]


inp = [int(i) for i in raw_input().split(" ")]
p1 = inp[1]/100.0
p2 = 1-p1

print "%.6f" % prob(inp[0], p1, p2)
