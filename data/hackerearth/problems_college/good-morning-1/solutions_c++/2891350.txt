#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int l,d;
		cin>>l>>d;
		int i,sum=0,c=1;
		while(l>=0)
		{
			if(c<4)
			{
			sum = sum + 2;
			c++;
			l--;
		}
		else
		{
			
			c=1;
	
			l = l+2;
		}
		}
		if(sum-2>=d)
		{
			cout<<"Yes"<<"\n";
		}
		else
		{
			cout<<"No"<<"\n";
		}
	}
}
