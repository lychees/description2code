#include <iostream>
#include <stdio.h>

using namespace std;

long long fact(long long n){
	
	long long ans = n * (n-1)*(n-2);
	return ans;
	
}

long long gcd(long long a,long long b){
	if(a==0)
		return b;
	if(b==0)
		return a;
	if(a>b)
		return gcd(b,a%b);
	return gcd(a,b%a);
}

int main()
{
	int test;
	cin>>test;
	while(test--){
	    long long n,k;
	    scanf("%lld",&n);
	    scanf("%lld",&k);
	    
	    long long k_count;
	    k_count =0;
	    
	    for(long long i = 0;i<n;i++){
	    	long long num;
	    	scanf("%lld",&num);
	    	if(num==k)
	    		k_count++;
	    }
	    long long numerator = fact(k_count);
	    long long denominator = fact(n);
	    
	    long long factor = gcd(numerator,denominator);
	   // cout<<numerator<<" "<<denominator<<" "<<factor<<endl;
	    
	    numerator = numerator/factor;
	    denominator = denominator/factor;
    
	    cout<<numerator<<"/"<<denominator<<endl;
	    
	}
    
    return 0;
}
