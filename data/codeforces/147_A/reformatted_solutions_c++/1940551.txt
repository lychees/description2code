#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>

using namespace std;
char s[10010];

int main() {
  int i, t = 0;
  gets(s);
  // cout<<s<<endl;
  for (i = 0; s[i]; i++) {
    if (s[i] == ' ')
      continue;
    if (t != 0 && s[i] >= 'a' && s[i] <= 'z' &&
        (s[i - 1] < 'a' || s[i - 1] > 'z'))
      printf(" ");
    printf("%c", s[i]);
    t = 1;
  }
  printf("\n");
}
