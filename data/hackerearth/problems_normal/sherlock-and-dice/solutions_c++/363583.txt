#include<stdio.h>
#include<iostream>
#include<cmath>
#include<string>
#include<cstring>
#include<algorithm>
#include<vector>
#include<map>
#include<utility>
#define PB push_back
#define MP make_pair
#define LL long long int
#define sc(x) scanf("%d",&x)
#define scl(x) scanf("%lld",&x)
using namespace std;

long double dp[51][30001];

int main()
{
	int y,t,i,j,n,m,k,K;
	 long double ans;
	cin>>t;
	while(t--)
	{
		cin>>m>>n>>K;
		for(i=0;i<=m;i++)
		{
			for(j=0;j<=30000;j++)
			dp[i][j] = 0.0;
		}
		
		for(i=1;i<=n;i++)
		{
			dp[1][i] = 1;
		}
		
		for(i=2;i<=m;i++)
		{
			for(j=i-1;j<=n*(i - 1);j++)
			{
				for(k=1;k<=n;k++)
				{
					dp[i][j + k] += dp[i -1][j];
					//cout<<i<<" "<<j+k<<" "<<dp[i][j + k]<<endl;
				}
			}
		}
		
		for(i=1;i<=m;i++)
		{
			for(j=1;j<=n*i;j++)
			{
//				cout<<i<<" "<<j<<" "<<dp[i][j]<<endl;
			}
		}
		//cout<<dp[m][k]<<endl;
		ans =  dp[m][K] * 1.0;
		//cout<<ans<<endl;
		if(ans == 0)
		{
			cout<<"0.000 0"<<endl;
			continue;
		}
		y = 0;
		
		for(i=1;i<=m;i++)
		{
			ans /= (n * 1.0);
			
//			cout<<ans<<endl;
		}
		while(ans < 1)
			{
				ans *= 10.0;
				y++;
			}
		
		printf("%.3Lf %d\n",ans,y);
//		cout<<ans<<" "<<y<<endl;
		
	}
	
	return 0;
}
