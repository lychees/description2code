from sys import stdin
t = int(stdin.readline())
for _ in xrange(t):
	z,n,m = map(int,stdin.readline().split())
	i = 1
	fa = []
	while i*i <z:
		if z%i==0:
			fa.append(i)
			fa.append(z/i)
		i+=1
	if z==i*i:
		fa.append(i)
	ans = 0
	#print "factors of ",z,fa
	for i in fa:
		j = z/i
		if n>=i and m>=j:
			ans+=1
	print ans
