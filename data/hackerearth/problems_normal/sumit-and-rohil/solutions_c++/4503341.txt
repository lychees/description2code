#include<iostream>
#include<string.h>
using namespace std;
int main()
{
int n,count=0;
cin>>n;
string str;
char str1[n+1][101];
int a[n+1];
for(int i=1;i<=n;i++)
{cin>>str;
 for(int j=1;j<=str.size();j++)
 str1[i][j]=str[j-1];
 a[i]=str.size();	
}
bool visited[101]={false};
for(int i=1;i<=n;i++)
{ if(visited[i]==false) 
{ 
int b[26]={0};
for(int j=1;j<=a[i];j++)
b[str1[i][j]-'a']++;
char ch1,ch2;
ch1=str1[i][1],ch2=str1[i][a[i]];
for(int j=i+1;j<=n;j++)
{ int c[27]={0}; 
if(ch1==str1[j][1]&&ch2==str1[j][a[j]]&&visited[j]==false)
{ int flag=1;
for(int k=1;k<=a[j];k++)
c[str1[j][k]-'a']++;
int l;
for( l=1;l<=a[j];l++)
if(b[str1[j][l]-'a']!=c[str1[j][l]-'a'])
break;
if(l<=a[j])
flag=0;
if(flag!=0)
visited[j]=true;
}
}
count++;
visited[i]=true;
}
}
cout<<count<<endl;
return 0;
}
