#include<iostream>
using namespace std;


int main(){
	int n;
	int k,count=0;
	int A[100005];
	scanf("%d %d",&n,&k);
	for(int i=1;i<=n;i++){
		A[i]=i;
	}
	int x,y;
	for(int i=0;i<k;i++){
		scanf("%d %d",&x,&y);
		A[y]=A[x];
	}
	for(int i=1;i<=n;i++){
		if(A[i]==i){
			count++;
		}
	}
	printf("%d\n",count);
	
	return 0;
}
