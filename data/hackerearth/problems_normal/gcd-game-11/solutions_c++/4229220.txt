#include "bits/stdc++.h"
using namespace std;

int dp[1001][1001];

int fun(int a,int b)
{
	if(a==1 and b==1) return -1;
	if(a == 1) return 0;
	if(b == 1) return 1;
	if(dp[a][b] == -1)
	{
		int gcd = __gcd(a,b);
		if(gcd == 1)
		{
			int tmp = fun(b-1,a);
			if(tmp == 0)
			 dp[a][b] = 1;
			 else
			 dp[a][b] = 0; 
		}
		else
		{
			int tmp1=1;
		  if(__gcd(a,b)<=b) tmp1 = fun(b/__gcd(a,b),a);
			int tmp2 = fun(b-1,a);
			if(tmp1 == 0 or tmp2 ==0)
				dp[a][b] = 1;
			 else
			 	dp[a][b] = 0;
		}
	}
   return dp[a][b];
}



int main(int argc, char const *argv[])
{
	//pre();
	memset(dp,-1,sizeof(dp));
	int t; cin >> t;
	while(t--)
	{
		 int a,b; cin >> a >> b;
		 int tmp = fun(a,b);
		 if(tmp == 1)
		 	cout << "Arjit" << endl;
		 else if(tmp == 0)
		 	cout << "Chandu Don" << endl;
		 else
		 	cout << "Draw" << endl;
	}
	return 0;
}
