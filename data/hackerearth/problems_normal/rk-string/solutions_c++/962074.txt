#include<stdio.h>
#include<string.h>
inline long long int nextInd(char str[],long long int i,char ch)
{
	long long int len=strlen(str);
	if(i==len)
	return len;
	char *p=strchr(str+i+1,ch);
	if(p==NULL)
	return len;
	return (p-str);
}
int main()
{
	long long int ntc,i,j,c,nj,k,len,a,b,x,y;
	long long int ans,ans1;
	
	scanf("%lld",&ntc);
	while(ntc--)
	{
		scanf("%lld%lld",&a,&b);
		char str[2000009];
		scanf("%s",&str);
		len=strlen(str);
		ans=ans1=0;
		if(a==0)
		{
			c=0;
			for(i=0;i<len;i++)
			if(str[i]=='K')
			c++;
			else
			{
				ans+=c*(c+1)/2;
				c=0;
			}
			ans+=c*(c+1)/2;
		}
		else
		{
			x=-1;
			y=-1;
			i=nextInd(str,x,'R');
			j=i;
			for(k=1;k<a;k++)
			{
				y=j;
				j=nextInd(str,j,'R');
			}
			while(j!=len)
			{
				nj=nextInd(str,j,'R');
				ans+=(i-x)*(nj-j);
				j=nj;
				x=i;
				i=nextInd(str,i,'R');
			}
		}
		
		if(b==0)
		{
			c=0;
			for(i=0;i<len;i++)
			if(str[i]=='R')
			c++;
			else
			
			{
				ans1+=c*(c+1)/2;
				c=0;
			}
			ans1+=c*(c+1)/2;
		}
		else
		
		{
			x=y=-1;
			i=nextInd(str,x,'K');
			j=i;
			for(k=1;k<b;k++)
			{
				y=j;
				j=nextInd(str,j,'K');
			}
			while(j!=len)
			{
				nj=nextInd(str,j,'K');
				ans1+=(i-x)*(nj-j);
				j=nj;
				x=i;
				i=nextInd(str,i,'K');
			}
		}
		printf("%lld %lld\n",ans,ans1);
	}
}
