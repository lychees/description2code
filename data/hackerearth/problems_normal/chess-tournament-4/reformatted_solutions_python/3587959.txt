n = int(input())
m = 2**n
li = [i for i in range(0, m)]
res = [[]]
for j in range(m-1):
    p = map(int, raw_input().split())
    res.append(p)
while (len(li) != 1):
    l = len(li)
    p = []
    for i in range(0, l/2):
        if (res[li[2*i+1]][li[2*i]] == 0):
            p.append(li[2*i])
        else:
            p.append(li[2*i+1])
    li = p
print(li[0]+1)
