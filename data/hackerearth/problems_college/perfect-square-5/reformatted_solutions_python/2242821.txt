import math
noc = input()
while noc:
    noc -= 1

    ins = input()
    res = int(math.ceil(math.sqrt(ins)))
    if res**2 == ins:
        print "YES"
        continue
    diff1 = abs(res**2 - ins)
    diff2 = abs((res-1)**2 - ins)

    if diff1 < diff2:
        print "+"+str(min(diff1, diff2))
    else:
        print "-"+str(min(diff1, diff2))
