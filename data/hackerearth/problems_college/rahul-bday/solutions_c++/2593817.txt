#include <iostream>
#define LL long long
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	LL b,w;
    	scanf("%lld%lld",&b,&w);
    	LL x,y,z;
    	scanf("%lld%lld%lld",&x,&y,&z);
    	
    	int tmpx,tmpy;
    	
    	tmpx = min(x,y+z);
    	tmpy = min(y,x+z);
    	
    	printf("%lld\n",tmpx*b+tmpy*w);
    }
    return 0;
}
