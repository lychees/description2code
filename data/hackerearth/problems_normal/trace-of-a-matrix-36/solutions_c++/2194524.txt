#include<bits/stdc++.h>
using namespace std;
int main(){


    int t;
    cin >> t;
    while(t--)
    {

        int n , m;
        cin >> n >> m;
        long long term  = min( n , m);
        long long ans1 =  (term * (2 + (term-1)*(n+1)))/2;
        long long ans2 =  (term * (2 + (term-1)*(m+1)))/2;
        cout << ans1 + ans2 << endl;


    }



}
