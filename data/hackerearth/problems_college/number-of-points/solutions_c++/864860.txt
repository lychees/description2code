#include <iostream>
#include<algorithm>
using namespace std;
int gcd(int a,int b)
{
if(a== 0){
return b;
} else return gcd(b%a,a);
}
int main()
{
int t;
cin>>t;
while(t--){
int x1,x2,y1,y2;
cin>>x1>>y1>>x2>>y2;
int a,b;
a=abs(x1-x2);
b=abs(y1-y2);
int ans;
if(a!=0&&b!=0)
ans = gcd(min(a,b),max(a,b))-1;
else {
if(a==0)
    ans=b-1;
else
    ans=a-1;
}
if(ans>=0)
 //printf("%d\n",ans);
 cout<<ans;
else
cout<<"0";
cout<<"\n";
}
return 0;
}
