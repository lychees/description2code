#include<iostream>
#include<algorithm>
#include<stdio.h>
#define INT_MAX 2147483647
using namespace std;
int n,m,q;
int binarySearch(int*a,int start,int end,int value){
    int answer=-1;
    int mid;
    while(start<=end){
        mid=start+(end-start)/2;
        if(a[mid]<=value){
            answer= mid;
            start=mid+1;
        }
        else{
            end=mid-1;
        }
    }
    return answer;
}

int main(){
    scanf("%d %d %d",&n,&m,&q);
    int finalAnswer,temp;
    int*para=new int[n];
    int**a=new int*[n];
    for(int i=0;i<n;i++){
        a[i]=new int[m];
        for(int j=0;j<m;j++){
            scanf("%d",&a[i][j]);
        }
    }
    while(q--){
        finalAnswer=INT_MAX;
        for(int i=0;i<n;i++){
            scanf("%d",&para[i]);
            temp=binarySearch(a[i],0,m-1,para[i])+1;
            // cout<<temp<<endl;
            finalAnswer=min(finalAnswer,temp);
        }
        printf("%d\n",finalAnswer);
    }
}
