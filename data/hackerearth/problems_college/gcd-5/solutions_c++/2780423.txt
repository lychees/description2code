#include <iostream>
using namespace std;

int main()
{
           long int t,n,i,j;
           long long int a[100005];
           cin>>t;
           int f;
           while(t--)
           {
           	  cin>>n;
           	  f=0;
           	  for(i=0;i<n;i++)
           	    cin>>a[i];
           	    for(i=0;i<n-1;i++)
           	    {
           	    	if(a[i]>a[i+1])
           	    	{
           	    		f=1;
           	    		break;
           	    	}
           	    }
           	    if(f==0)
           	    cout<<"YES"<<endl;
           	    else
           	     cout<<"NO"<<endl;
           }
    return 0;
}
