#include<iostream>
using namespace std ;
int main()
{
	unsigned long long  min , sum , x ;
	int t ;
	int n , p ;
	cin >> t ;
	while ( t-- )
	{
		cin >> n ;
		p = n ;
		sum = 0 ;
		while ( n-- )
		{
			cin >> x ;
			sum = sum + x ;
		}
		min = sum / p ;
		cout<<min <<"\n" ;
	}
	return 0 ;
}
