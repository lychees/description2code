#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T=0;
    scanf("%d",&T);
    while(T--)
    {
        int N=0,a,b,c,ans=0;
        scanf("%d",&N);
        for(int i=0;i<N;i++)
        {
            scanf("%d %d %d",&a,&b,&c);
            ans=ans^a^b^c;
        }
        if(ans!=0)
            printf("Stark\n");
        else
            printf("Lannister\n");
    }
}
