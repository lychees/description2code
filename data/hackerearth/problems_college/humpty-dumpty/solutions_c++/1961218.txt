#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t;cin >> t;
    while(t--)
    {
    	int n,flag=0; cin >> n;
    	float x,y; cin >> x >> y;
    	float a[n],d=0,dd;
    	for(int i=0;i<n;i++)
    		cin >> a[i];
    	dd = sqrt(x*x + y*y);	
    	d += dd;
    	for(int i=0;i<n;i++)
    		d += a[i];
    	if(dd>d*1.0/2)
    	{
    		cout << "Not able\n";continue;
    	}
    	for(int i=0;i<n;i++)
    	{
    		if(a[i]>d*1.0/2)
    		{
    			cout << "Not able";
    			flag=1; break;
    		}
    	}
    	if(flag==0)
    		cout << "Able";
    	cout << endl;
    }
    return 0;
}
