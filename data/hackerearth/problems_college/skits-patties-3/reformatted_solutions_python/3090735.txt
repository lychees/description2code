S = set()
for i in xrange(10):
    for j in xrange(10):
        for k in xrange(10):
            S.add(i*6 + j*9 + k*20)


def solve(N):
    return N in S


rr = raw_input
for tc in xrange(int(rr())):
    N = int(rr())
    if N > 50:
        print "True"
    else:
        print["False", "True"][solve(N)]
