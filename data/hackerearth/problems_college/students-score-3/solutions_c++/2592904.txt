#include<stdio.h>
#include<iostream>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<stdlib.h>
#include<vector>
#include<map>
#include<set>

using namespace std;

main()
{
	int x,n,a[10000];
	cin>>n;
	
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	
	for(int i=0;i<n;i++)
	{
		if(a[i]>0)
		{
			x=i;
			break;
		}
	}
	int max=a[x],sum=0;
	
	for(int i=x;i<n;i++)
	{
		sum=0;
		for(int j=i;j<n;j++)
		{
			sum=sum+a[j];
			if(sum>max)
			max=sum;
		}
	
	
	}
	cout<<max<<endl;
}
