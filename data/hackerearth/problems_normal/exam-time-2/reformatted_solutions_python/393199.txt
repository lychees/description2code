import fractions
import math


def totient(n):
    result = n
    root = int(math.sqrt(n))
    for i in range(2, root+1):
        if n % i == 0:
            result = result - result/i
        while (n % i == 0):
            n = n/i
    if (n > 1):
        result = result - result/n
    return result


t = input()
for i in range(1, t+1):
    n = input()
    print(totient(n))
