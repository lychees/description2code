#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main()
{
    int l;
    cin >> l;
    string s;
    cin >> s;
    vector<int> pos[256];
    for(int i=0;i<l;i++) {
    	
    	pos[s[i]].push_back(i);
    }
    
    int q;
    cin >> q;
    while(q--) {
    	
    	char x, y;
    	cin >> x >> y;
    	if(x==y) {
    		
    		cout<< (pos[x].size() * (pos[x].size() -1) )/2<<endl;
    	}
    	else 
    	cout<< pos[x].size() * pos[y].size()<<endl;
    }
    return 0;
}
