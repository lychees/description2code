n,q=map(int,raw_input().split())
a=[0,0]+map(int,raw_input().split())
b=[0,0]+map(int,raw_input().split())
for i in range(4,n+2):
    a[i]+=a[i-2]
    b[i]+=b[i-2]
for i in range(q):
    c,l,r=map(int,raw_input().split())
    if l>r:
        l,r=r,l
    t=l+((r-l)/2)*2
    if t==r:
        r=t-1
    if c==1:
        print a[t+1]-a[l-1]+b[r+1]-b[l]
    else:
        print b[t+1]-b[l-1]+a[r+1]-a[l]
