#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define lim 100000
ll makenum(string s)
{
   ll ans=s[0]-'0';
   for(ll i=1;i<s.length();i++)
      ans=10*ans+(s[i]-'0');
   return ans;
}
int main()
{
   ll t;
  // freopen("input.txt","r",stdin);
   cin>>t;
   string s;
   ll left;
   while(t--)
   {
      cin>>s;
      left=0;
      ll j,len=s.length();
      for(ll i=0;i<s.length();i++)
      {
         j=i+1;
         if(s[i]>'3')
            {left+=pow(10,len-j);
         left+=(s[i]-'1')*(pow(10,len-j)-pow(9,len-j));
      }
      else  left+=(s[i]-'0')*(pow(10,len-j)-pow(9,len-j));


      }
      ll ans=makenum(s);
      cout<<ans-left<<endl;
   }
}
