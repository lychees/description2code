#include <iostream>
#include <stdio.h>
#include <math.h>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;
typedef long long ll;


int main()
{

		int n,max=0,sum=0;
		scanf("%d",&n);
		int arr[n];

		for(int i=0;i<n;i++)
			scanf("%d",&arr[i]);

		for(int i=0;i<n;i++)
		{
			for(int j=i;j<n;j++)
			{
				sum+=arr[j];
				if(sum>max)
					max=sum;
			}
			sum=0;
		}
		if(max<0)
			printf("0\n");
		else
			printf("%d\n",max );
	return 0;
}
