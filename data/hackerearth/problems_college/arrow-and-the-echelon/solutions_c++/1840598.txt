#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        int x;
        cin>>x;
        int powof5=0;
        bool god=0;
        while(powof5<=x)
        {
            if((x-powof5)%3==0 && x-powof5>=0){god=1;break;}
            powof5+=5;
        }
        if(!god){cout<<-1<<endl;continue;}
        string best="";
        for(int j=0;j<x-powof5;j++)best+='5';
        for(int j=0;j<powof5;j++)best+='3';
        cout<<best<<endl;
    }
    return 0;
}
