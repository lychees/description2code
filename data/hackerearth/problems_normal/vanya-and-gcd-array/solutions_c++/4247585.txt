using namespace std;
int n;
int ar[505];
ll dp[505][105]; // state -- last pos,gcd at last pos


int main() 
{
	cin>>n;

	for(int i=0;i<n;i++)
		cin>>ar[i];
	
	ll ans =0;
	
	for(int i=0;i<n;i++){
		dp[i][ar[i]] =1;
		
		for(int j=i-1;j>=0;j--){
			if(ar[j]<ar[i]){
				for(int k=1;k<=100;k++){
					int g = gcd(k,ar[i]);
					dp[i][g]+=dp[j][k];
					
					if(dp[i][g]>=MOD){
						dp[i][g]%=MOD;
					}
				}
			}
		}
	}
	
	for(int i=0;i<n;i++){
		ans+=dp[i][1];
		
		if(ans>=MOD){
			ans%=MOD;
		}
	}
	
	cout<<ans<<endl;

	
  return 0;
}
