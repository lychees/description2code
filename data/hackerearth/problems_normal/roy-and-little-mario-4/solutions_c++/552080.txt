#include <bits/stdc++.h>
#define MAX 100001
#define MOD 1000000007
#define LL long long
int main(void)
{
    LL T=0,n=0;
    int i=0;
    LL dp[MAX]={0};
    scanf("%d",&T);
    dp[0]=1;
    dp[1]=1;
    dp[2]=2;
    for(i=3;i<MAX;i++)
        dp[i]=((dp[i-1]%MOD)+(dp[i-2]%MOD)+(dp[i-3]%MOD))%MOD;
    while(T--)
        {
            scanf("%d",&n);
            printf("%lld\n",dp[n]);
        }
    return 0;
}
