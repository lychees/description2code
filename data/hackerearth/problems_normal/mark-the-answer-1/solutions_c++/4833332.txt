#include <iostream>
using namespace std;

int main()
{
    int N,X,count=0,skip=0;
    int diff[10000000];
    
    cin>>N>>X;
    
    for(int i=0; i<N; i++)
    {
    	cin>>diff[i];
    }
    
    for(int i=0; i<N; i++)
    {
    	if(diff[i]<=X)
    		count++;
    	else if(diff[i]>X && skip==0)
    	{
    		skip=1;
    		continue;
    	}
    	else
    		break;
    }
    cout<<count<<endl;
    return 0;
}
