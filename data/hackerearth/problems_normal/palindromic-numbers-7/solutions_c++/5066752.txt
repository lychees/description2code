 #include<bits/stdc++.h>
using namespace std;
#define ll long long
#define li long int
#define w while
bool fun(ll a)
{
	ll b=a;
	ll sum=0;
	w(a!=0)
	{
		sum=sum*10+a%10;
		a=a/10;
	}
	if(sum==b)
	return true;
	else
	return false;
}
int main()
{
	ll t;
	cin>>t;
	w(t--)
	{
		ll a,b,i,count=0;
		cin>>a>>b;
		for(i=a;i<=b;i++)
		{
			if(fun(i))
			count++;
		}
		cout<<count<<endl;
	}
	return 0;
}
