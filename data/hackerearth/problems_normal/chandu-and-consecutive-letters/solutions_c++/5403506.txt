#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t,l,i,j;
   cin>>t;
   string s;
   while(t--){
   	cin>>s;
   	l = s.size();
   	j=0;
   	string r;
   	r.insert(r.begin(),s[0]);
   	for(i=1;i<l;i++){
   		if(r[j]!=s[i]){
   			j++;
        	r.insert(r.begin()+j,s[i]);
   		}
   	}
   	cout<<r<<endl;
   }
    return 0;
}
