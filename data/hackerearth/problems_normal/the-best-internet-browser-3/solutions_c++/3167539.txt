#include <bits/stdc++.h>

using namespace std;


int main ()
{
    int t;
    cin >> t;
    while (t--)
    {
    string s;
    cin >> s;

    int l = s.length();
    int m = 0;
    for (int i = 0; i < l-4; i++)
    {
        if (s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u')
        {
            ++m;
        }
    }
    cout << l-(m+4) << "/" << l << "\n";
    }
    return 0;
}
