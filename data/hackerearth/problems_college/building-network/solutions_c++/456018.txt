#include <bits/stdc++.h>
using namespace std;

long long R[1000005];

int main()
{
	int t,n,i;
	long long ans;
	
	scanf("%d",&t);
	
	while(t--)
	{
		ans=0;
		
		scanf("%d",&n);
		
		for(i=0;i<n;i++)
		scanf("%lld",&R[i]);
		
		sort(R,R+n);
		
		for(i=1;i<n;i++)
		{
			ans=ans+R[0]+R[i];
		}
		printf("%lld\n",ans);
	}
	return 0;
}
