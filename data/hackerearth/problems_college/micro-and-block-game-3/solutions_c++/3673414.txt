#include <bits/stdc++.h>
 
using namespace std;
 
#define pb push_back
#define mp make_pair
#define ll long long int
 
int myCompare(string X, string Y)
{
    string XY = X.append(Y);
    string YX = Y.append(X);
    return XY.compare(YX) > 0 ? 1: 0;
}
int main()
{
	int t;
	cin>>t;
	vector<string>s(t);
	int i;
	for(i = 0; i < t; i++) {
		cin>>s[i];
	}
	sort(s.begin(),s.end(),myCompare);
	for(i = 0; i < t; i++) {
		cout<<s[i];
	}
	cout<<endl;
	return 0;
}
