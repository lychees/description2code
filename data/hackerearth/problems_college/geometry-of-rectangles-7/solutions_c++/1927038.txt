#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	double x1,y1,x2,y2;
    	double xx1,yy1,xx2,yy2;
    	cin>>x1>>y1>>x2>>y2>>xx1>>yy1>>xx2>>yy2;;
        double e=(x1+x2)/2.0,f=(y1+y2)/2.0,g=(xx1+xx2)/2.0,h=(yy1+yy2)/2.0;
        if((abs(g-e)==abs(x2-e)+abs(xx2-g))&&(abs(h-f)==abs(y1-f)+abs(yy1-h))) cout<<"Point\n";
        else if(((abs(g-e)==abs(x2-e)+abs(xx2-g))&&(abs(h-f)<abs(y1-f)+abs(yy1-h)))||((abs(g-e)<abs(x2-e)+abs(xx2-g))&&(abs(h-f)==abs(y1-f)+abs(yy1-h)))) cout<<"Line\n";
        else if((abs(g-e)<abs(x2-e)+abs(xx2-g))&&(abs(h-f)<abs(y1-f)+abs(yy1-h))) cout<<"Rectangle\n";
        else cout<<"Apart\n";
    }
    return 0;
}
