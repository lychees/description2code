#include <iostream>
#include<string>
#include <stdio.h>
#include <ctype.h>
using namespace std;

int main()
{
   int n;cin>>n;
   while(n--)
   {
   	string s;
   	cin.ignore();
   	getline(cin,s);
   	char *a=new char[s.size()];
   	int j=0;
   	for(int i=0;i<s.size();i++)
   	{
   		if(s[i]=='$')
   		{
   			i++;
   			while(i<s.size()&&!isalpha(s[i]))
   			{
   				if(s[i]==' ')
   				{
   					i++;continue;
   				}
   				if(s[i]>=48&&s[i]<=57)
   				{
   					a[j++]=s[i];
   					i++;
   				}
   			}
   			a[j]='\0';
   			break;
   		}
   		
   	}
   	cout<<'$';
   	int flag=0;
   	for(int i=0;i<j;i++)
   	{
   		if(a[i]=='0'&&flag==0)
   		continue;
   		flag=1;
   		cout<<a[i];
   	}
   	if(flag==0)
   	cout<<'0';
   	cout<<"\n";
   	
   }
}
