t = int(raw_input())
while t > 0:
    t -= 1
    l = []
    n = int(raw_input())
    for i in range(n):
        l.append(int(raw_input()))
    c = 0
    while len(l) > 0:
        l.sort()
        k = len(l)-2
        temp = len(l)-1
        while k >= 0:
            if l[temp] > l[k]:
                del l[temp]
                temp = k
            k -= 1
        del l[temp]
        c += 1
    print c
