#include<bits/stdc++.h>
using namespace std;
int main()
{

    int t,flag,i,j,l;
    char str[110000];
    cin>>t;
    while(t--)
    {
        cin>>str;
        l=strlen(str);
        flag=0;
        for(i=1,j=l-2;i<l&&j>=0;i++,j--)
        {
            if( abs(str[i]-str[i-1])!= abs(str[j]-str[j+1]))
            {
                flag=1;
                break;
            }
        }
        if(flag)
        {
            cout<<"Not Funny\n";
        }
        else
        {
            cout<<"Funny\n";
        }
    }
    return 0;

}
