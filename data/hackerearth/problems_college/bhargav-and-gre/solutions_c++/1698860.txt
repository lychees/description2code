#include <iostream>
#include <cstdio>
#include <string>

using namespace std;

int main()
{
    string s;
    int n;
    
    cin >> s >> n;
    
    int len = s.size();
    int syn = 0, ant = 0;
    
    for (int i = 0; i < n; i++) {
    	string p;
    	cin >> p;
    	
    	int l = p.size();
    	 
    	if (p[0] == s[0] && l == len) {
    		syn++;
    	}
    	if (p[0]-3 == s[0] && l == len) {
    		ant++;
    	}
    }
    
    printf("%d %d", syn,ant);
    
    return 0;
}
