#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long t;
    cin>>t;
    while(t--)
    {
    	long long n,i;
    	cin>>n;
    	string str;
    	cin>>str;
    	for(i=0;i<str.size();i++)
    	{
    		if((str[i]-'A')<n)
    		{
    			cout<<char(26-n+str[i]);
    		}
    		else
    		cout<<char((str[i])-n);
    	}
    	cout<<endl;
    }
    return 0;
}
