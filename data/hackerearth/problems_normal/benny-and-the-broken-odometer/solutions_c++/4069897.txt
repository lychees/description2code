#include <iostream>
using namespace std;
typedef long long int ll;
int main()
{
    ll t;
    cin >> t;
    while(t--) {
    	ll n;
    	cin >> n;
    	ll ans = 0;
    	ll p = 1;
    	ll r;
    	while(n > 0) {
    		r = n%10;
    		if(r > 3) {
    			r--;
    		}
    		ans = r*p + ans;
    		p = p * 9;
    		n/=10;
    	}
    	cout << ans << endl;
    }
    return 0;
}
