#include<cstdio>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int sume=0,sumo=1,cnt=0;
		while(n!=1)
		{
			if(n%2==0)
			{
				sume=sume+n;
				n=n/2;
			}
			else
			{
				sumo=sumo+n;
				n=3*n+1;
			}
			cnt++;
		}
		int x=sume%cnt;
		int y=sumo%cnt;
		if(x<y)
			printf("Odd Rules\n");
		else if(x>y)
			printf("Even Rules\n");
		else
			printf("Tie\n");
	}
	return 0;
}
