import sys

T = input()

while (T > 0):
    n, m = map(int, raw_input().split())
    i = 0
    res = ""
    while (i < n):
        j = 0
        ri = -1
        rv = sys.maxint
        while (j < m):
            d1, d2, d3 = map(int, raw_input().split())
            c = (100-d1)*(100-d2)*(100-d3)
            if (c < rv):
                ri = j
                rv = c
            j = j+1
        res = res + str(ri+1) + " "
        i = i+1
    print(res)
    T = T-1
