#include <iostream>
#include<cmath>
using namespace std;

int main()
{
	int T;
	long long int i,n,flag;
    scanf("%d",&T);
    while(T--)
    {
    	flag=0;
    	scanf("%lld",&n);
    	if(n==0 || n==1)
    	printf("NO\n");
    	else if(n==2 || n==3)
    	printf("YES\n");
    	else
    	{
    		for(i=2 ; i<=sqrt(n) ; i++)
    		{
    			if(n%i==0)
    			{
    				flag=1;
    				break;
    			}
    		}
    		if(flag==0)
    		printf("YES\n");
    		else
    		printf("NO\n");
    	}
    }
    return 0;
}
