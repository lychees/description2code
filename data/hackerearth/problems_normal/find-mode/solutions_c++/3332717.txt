#include <iostream>
#include<map>
#include<vector>
#include<algorithm>
#include<stack>
#include<stdio.h>
using namespace std;
int a[1000005];
vector<int>v;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,Max=-1;
        cin>>n;
        v.resize(n);
        for(int i=0;i<n;i++)
        {
            cin>>v[i];
            a[v[i]]++;
            if(a[v[i]]>Max)
                Max=a[v[i]];
        }
        sort(v.begin(),v.end());
        for(int i=v.size()-1;i>=0;i--)
        {
            if(a[v[i]]==Max)
                cout<<v[i]<<" ";
            a[v[i]]=0;
        }
        cout<<endl;
        v.clear();
    }
}
