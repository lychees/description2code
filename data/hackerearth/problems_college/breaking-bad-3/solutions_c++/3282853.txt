#include <bits/stdc++.h>
#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--) {
		int n;
		cin >> n;

		long int a[n];

		for(int i = 0; i < n; i++) {
			cin >> a[i];
		}

		sort(a, a+n);

		unsigned long long int value = 0;

		for(int i = n-1; i >= 0; i = i - 2) {
			value = value + (a[i] * a[i - 1]);
			//cout << a[i] << "  " << a[i-1] << endl;
		}

		cout << value << endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
