const int MAXN = 1e6 + 3;
char s[MAXN];

void zFunction() {
	int L, R, N;
	N = strlen(s);
	L = R = 0;
	for (int i = 1; i < N; i++) {
		if (i > R) {
			L = R = i;
			while (R < N && s[R - L] == s[R]) R++;
			z[i] = R - L;
			R--;
		}
		else {
			int k = i - L;
			if (z[k] <= R - i)
				z[i] = z[k];
			else {
				L = i;
				while (R < N && s[R - L] == s[R]) R++;
				z[i] = R - L;
				R--;
			}
		}
	}
}
