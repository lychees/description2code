#include <iostream>
using namespace std;

int main()
{
    int n;
    long long int ans = 0,ans1;
    while(cin>>n){
    	int p = n;
    	 int pp = 0;
    	while(p != 0){
    		p = p%10;
    		pp = pp + p;
    		p = p/10;
    		
    	}
    	if(ans <= pp){
    		ans = pp;
    		ans1 = n;
    	}
    }
    cout<<ans1<<endl;
}
