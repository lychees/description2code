#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;
unsigned long long int mod= pow(10,9)+7;
unsigned long int dp[1000001];
void initialize()
{
    fill_n(dp,1000001,-1);
    dp[0]=1;
    dp[1]=1;
}
long long int comb(long long int n)
{
    long long int p,q;
	if(dp[n]==-1)
	{
		if(n==0)
    		return dp[0];
    	else if(n==1)
        	return dp[1];
    	else
        {
            p=(comb(n-1)%mod);
            q=(((n-1)%mod)*(comb(n-2)%mod));
            dp[n]=p+q;
        }
	}
    return dp[n];
}

int main()
{
    int t;
    long long int n,c;
    initialize();
    cin>>t;
    while(t--)
    {
        cin>>n;
        c=(comb(n));
        //cout<<c<<endl;
        c=(c%mod);
        cout<<c<<endl;
    }
    return 0;
}
