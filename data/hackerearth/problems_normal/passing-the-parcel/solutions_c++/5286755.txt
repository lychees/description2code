#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
   int n;
   cin>>n;
   map<int,int> h;
   string s;
   cin>>s;
   int cur=1;
   int x=0;
   int ans;
   int f=0;
   while(1)
   {
   for(int i=0;i<s.length();i++)
   	{
   		if(s[i]=='a')
   		{
   			
   			cur=(cur+1)%n;
   			while(h[cur]!=0)
   				cur=(cur+1)%n;
   			
   		}
   		else
   		{
   			h[cur]=1;
			x++;
			while(h[cur]!=0)
   				cur=(cur+1)%n;
   				if(x==n-1)
   					{
   						ans=cur;
   						if(ans==0)
   							ans=n;
   						f=1;
   						break;
   					}
			
			
   		}
   	}
   	if(f==1)
   		break;
  }
   	cout<<ans<<endl;
   
}
