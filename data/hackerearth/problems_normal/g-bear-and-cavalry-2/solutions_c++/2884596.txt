#include <bits/stdc++.h>
using namespace std;

long long W[50], H[50];

int main(){
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	
	long long T, N, i, f, j, k, x;
	cin >> T;
	while(T--){
		cin >> N;
		for(i=0;i<N;i++){
			cin >> W[i];
		}
		for(i=0;i<N;i++){
			cin >> H[i];
		}
		sort(W+1, W+N);
		sort(H, H+N);
		f=1;
		for(i=0;i<N-1;i++){
			if(H[i]*W[N-i-1]>=H[N-1]*W[0]){
				f=0;
				break;
			}
		}
		if(f) cout << "YES\n";
		else cout << "NO\n";
	}
	return 0;
}
