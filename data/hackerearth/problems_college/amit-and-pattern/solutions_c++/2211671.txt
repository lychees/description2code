#include <iostream>
using namespace std;

int main()
{	int m,n,i,j,k,c=0;
    cin>>m>>n;
    int a[m][n];
    for(i=0;i<m;++i)
    	for(j=0;j<n;++j)
    		cin>>a[i][j];
    for(k=0;k<m+n-1;++k)
    {	for(i=0;i<m;++i)
    		for(j=0;j<n;++j)
    			if(i+j==c)
    				cout<<a[i][j]<<" ";
    	++c;
    	cout<<"\n";
    }
    return 0;
}
