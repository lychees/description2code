#include<iostream>
#include<vector>
using namespace std;
vector<int> v[10001];
int k;
pair<int,int> solve(int x)
{
    pair<int,int> a=make_pair(0,0);
    for(int i=0;i<v[x].size();i++)
    {
        pair<int,int>cur=solve(v[x][i]);
        a.first+=cur.first;
        a.second+=cur.second;
    }
    a.second++;
    if(a.second>=k)
    {
        a.first++;
        a.second=0;
    }
    return a;
}
int main()
{
    int n;
    cin>>k>>n;
    for(int i=2;i<=n;i++)
    {
        int m;
        cin>>m;
        v[m].push_back(i);
    }
    pair<int,int> ans=solve(1);
    cout<<ans.first<<endl;
}
