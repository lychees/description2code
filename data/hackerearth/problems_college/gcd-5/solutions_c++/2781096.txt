#include<iostream>
using namespace std;
main()
{
    int t,n;
    long int a[99999];
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(int i=0;i<n;i++)
            cin>>a[i];
        int i;
        for(i=1;i<n;i++)
        {
            if(a[i-1]<a[i])
                continue;
            else
                break;
        }
        if(i==n)
            cout<<"YES\n";
        else
            cout<<"NO\n";
    }
}
