#include<bits/stdc++.h>
using namespace std;
main()
{
	ios_base::sync_with_stdio(0);cin.tie(0);
	int t,k,j,l;
	cin>>t;
	while(t--)
	{
		cin>>k;
		string s;
		cin>>s;
		for(int i=0;i<s.length();i++)
		{
			if((int)s[i]>=65 && (int)s[i]<=90)
			{
				l=0;
			}
			else if((int)s[i]>=97 && (int)s[i]<=122)
			{
				l=1;
			}
			for(j=1;j<=k;j++)
			{
				if(l==0)
				{
					s[i]++;
					if(((int)s[i])>90)
					{
						
						s[i]='A';
					}
				}
				else if(l==1)
				{
					s[i]++;
					if(((int)s[i])>122)
					{
						
						s[i]='a';
					}
				}
			}
		}
		cout<<s<<endl;
	}
}
