int solve(vector<int>&table,int a,int e,int *l)
{
    int i,val=0;
    table.push_back(a);
    if(a==e)
        val=1;
    for(i=0;i<(*l);i++)
    {
        if(e==table[i]+a)
            val=1;
        table.push_back(table[i]+a);
    }
    *l=(*l)+(*l)+1;
    return val;
}
int func(int a[],vector<int>&table,int e,int *m,int n,int *l)
{
    int i;
    for(i=0;i<(*l);i++)
    {
        if(table[i]==e)
            return 1;
    }
    int val;
    while(1)
    {
        if((*m)>n)
            return 0;
        if(e<a[(*m)-1])
            return 0;
        val=solve(table,a[(*m)-1],e,l);
        (*m)++;
        if(val==1)
            return 1;
    }
}
int main()
{
    int t,p,m,i,j,l=0;
    cin>>t;
    vector<int>table;
    while(t!=0)
    {
    cin>>p;cin>>m;
    table.clear();
    int a[m],b[p];
    for(i=0;i<m;i++)
        cin>>a[i];
    cin>>b[0];
    int e=0,count=0,cu=1;
    l=0;
    for(i=1;i<p;i++)
    {
        cin>>b[i];
        if(b[i]==b[i-1])
            e++;
        else
        {
            e++;
            if(func(a,table,e,&cu,m,&l)==1)
                count++;
            e=0;
        }
    }
    e++;
    if(func(a,table,e,&cu,m,&l)==1)
        count++;
    e=0;
    cout<<count<<endl;
    t--;
    }
    return 0;
}
