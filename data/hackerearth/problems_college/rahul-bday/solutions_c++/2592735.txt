#include <iostream>
using namespace std;
#define ll long long
int main()
{	ll t;
    cin>>t;
    while(t--)
    {	ll a,b;
    	cin>>a>>b;
    	ll p,q,r;
    	cin>>p>>q>>r;
    	ll x,y,z;
    	x=a*p+b*q;
    	y=a*(q+r)+b*q;
    	z=b*(p+r)+a*p;
    	if(x<y&&x<z)
    		cout<<x<<"\n";
    	else if(y<z&&y<x)
    		cout<<y<<"\n";
    	else
    		cout<<z<<"\n";
    }
    return 0;
}
