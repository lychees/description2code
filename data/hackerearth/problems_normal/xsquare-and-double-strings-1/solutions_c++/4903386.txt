#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    string s;
    int *c;
    int i;
    for(int p=0;p<t;++p)
    {
    	c=new int[26];
    	cin >> s;
    	for(int j=0;j<s.size();++j)
    		++c[s[j]-'a'];
    	
    	for(i=0;i<26;++i)
    	{
    	//	cout << "i" << i << "c" << c[i];
    		if(c[i]>=2)
    			break;
    		
    	}
    	if(i==26)
    		cout << "No\n";
    	else
    		cout << "Yes\n";
    }
    return 0;
}
