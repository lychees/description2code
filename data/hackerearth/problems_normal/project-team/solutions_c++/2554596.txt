#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		vector<int>v;
		int n,x;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>x;
			v.push_back(x);
		}
		sort(v.begin(),v.end());
		int s,i=0,j=n-1,m=INT_MIN,M=INT_MAX;
		while(i<j)
		{
			s=v[i]+v[j];
			m=max(s,m);
			M=min(s,M);
			i++,j--;
		}
		cout<<(m-M)<<endl;
	}
}
