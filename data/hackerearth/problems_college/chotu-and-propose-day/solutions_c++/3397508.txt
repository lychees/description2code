#include <iostream>
using namespace std;

int main()
{
    int t;
   	int aa,bb,ab,ba;
   	cin>>t;
   	while(t--)
	{
		cin>>aa>>bb>>ab>>ba;

		for(int i=1; i<=aa/2; ++i)
		{
			cout<<"aa";
		}
		if(ab>0 && ba>0)
		{
			for(int i=1; i<=min(ab,ba); ++i)
			{
				cout<<"ab";
			}	
		}

		if(bb&1 && aa%2==0)
		{
			for(int j=1; j<=bb; ++j)
			{
				cout<<"bb";
			}
		}
		else
		{
			for(int j=1; j<=bb/2; ++j)
			{
				cout<<"bb";
			}
			if(aa&1)
				cout<<"aa";
			for(int j=1; j<=bb/2; ++j)
			{
				cout<<"bb";
			}
		}
		
		if(ab>0 && ba>0)
		{
			for(int i=1; i<=min(ab,ba); ++i)
			{
				cout<<"ba";
			}	
		}
		for(int i=1; i<=aa/2; ++i)
		{
			cout<<"aa";
		}

		cout<<endl;	
	}
    return 0;
}
