bool isPalindrome(string str) {
	int i = 0, j = str.size() - 1;

	while (i < j) {
		if (str[i] != str[j])
			return false;
		++i;
		--j;
	}

	return true;
}

			if (isPalindrome(str)) {

			}
		}

		cout << cnt << endl;
	}
