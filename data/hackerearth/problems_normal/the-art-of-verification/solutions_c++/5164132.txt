#include<iostream>
#include<string>
using namespace std;

int main(){
	string s;
	cin >> s;
	int i, username, pwd, profile, role, key;
	username = s.find("username");
	pwd = s.find("pwd");
	profile = s.find("profile");
	role = s.find("role");
	key = s.find("key");
	
	i = username + 9;
	cout << "username: ";
	while(i < pwd - 1){
		cout << s[i++];
	}
	cout << "\n";
	
	i = pwd + 4;
	cout << "pwd: ";
	while(i < profile - 1){
		cout << s[i++];
	}
	cout << "\n";
	
	i = profile + 8;
	cout << "profile: ";
	while(i < role - 1){
		cout << s[i++];
	}
	cout << "\n";
	
	i = role + 5;
	cout << "role: ";
	while(i < key - 1){
		cout << s[i++];
	}
	cout << "\n";
	
	i = key + 4;
	cout << "key: ";
	while(i < s.length()){
		cout << s[i++];
	}
	cout << "\n";
}
