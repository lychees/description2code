#include <iostream>
using namespace std;

int main()
{
    int N,set[100000],count=0,save;
    //Enter Size
    cin>>N;
    //Enter Elements
    for(int i=0;i<N;++i)
    	cin>>set[i];
    //Processing
    for(int i=0;i<N;++i)
    	if(set[i]!=0)
    		for(int j=i+1;j<N;++j)
    			if(set[j]==set[i])
    			{
    				count++;
    				set[j]=0;
    			}
    cout<<count;
    return 0;
}
