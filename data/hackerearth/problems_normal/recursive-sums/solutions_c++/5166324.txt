#include <iostream>
using namespace std;

int main()
{	long long int i,j,m,t,len[10000],d[10000],n,c,sum=0;
	cin>>t;
	for(j=0;j<t;j++){
		n=0;
		sum=0;
	cin>>m;
	for(i=0;i<m;i++){
		cin>>len[i]>>d[i];
		n=n+len[i]*d[i];
		while(n>9){
			c=n%10;
			n=n/10;
			n=n+c;
		}
	}
		cout<<n<<endl;
	}
    return 0;
}
