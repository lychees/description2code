#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int n,i,j,k=0,a[1000],flag;
    cin>>n;
    for(i=2;k!=n;i++)
    {flag=0;
       for(j=2;j<=sqrt(i);j++)
       {
       	if(i%j==0) 
       	{
       		flag++; break;
       	}
       }
       if(!flag)
       { a[k]=i; k++; } 
    }
    printf("%d",a[n-1]);
    return 0;
}
