#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int par[1000005];
long long int ans;
struct edg
{
	int u;
	int v;
	long long int w;
};
struct edg arr[1000005];
bool cmp(struct edg a,struct edg b)
{
	return a.w < b.w;
}
void ms(int n)
{
	int i;
	for(i=1;i<=n;i++)
	{
		par[i]=i;
	}
}
int fp(int x)
{
	if(x==par[x])
	{
		return x;
	}
	else
	{
		return par[x]=fp(par[x]);
	}
}
void uni(int x,int y,long long int w)
{
	int px,py;
	px=fp(x);
	py=fp(y);
	if(px!=py)
	{
		par[px]=py;
		ans+=w;
	}
}
int main()
{
    int n,m,u,v,i;
    long long int w;
    scanf("%d%d",&n,&m);
    for(i=1;i<=m;i++)
    {
    	scanf("%d%d%lld",&u,&v,&w);
    	arr[i-1].u=u;
    	arr[i-1].v=v;
    	arr[i-1].w=w;
    }
    sort(arr,arr+m,cmp);
    ms(n);
    ans=0;
    for(i=0;i<m;i++)
    {
    	uni(arr[i].u,arr[i].v,arr[i].w);
    }
    printf("%lld\n",ans);
    return 0;
}
