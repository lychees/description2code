N, M = list(map(int, raw_input().strip().split(' ')))
Co = list(map(int, raw_input().strip().split(' ')))
Do = list(map(int, raw_input().strip().split(' ')))
A = []
for i in xrange(N):
    t = Do[i]-1
    if len(A) == 0:
        C = 1
    else:
        C = A[i-1]//Co[t]+1
    while True:
        x = Co[t]*C
        for j in xrange(t+1, M):
            if x % Co[j] == 0:
                C += 1
                break
        else:
            A.append(x)
            break
print A[N-1]
