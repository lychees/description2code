#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,h;
	cin>>t;
	while(t--)
	{
		cin>>n>>h;
		int hg[n];
		for(int i=0;i<n;i++)
			cin>>hg[i];
		sort(hg,hg+n);
		int *v=lower_bound(hg,hg+n,h);
		int val1=abs((h-(*v)));
		int val2=abs(h-(*(v-1)));
		int val3=abs((h-(*(v+1))));
		if(*v==hg[0])
		{
			cout<<min(val1,val3)<<endl;
			continue;
		}
		cout<<min(val1,min(val2,val3))<<endl;
	}
    return 0;
}
