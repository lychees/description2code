#include <iostream>
#include<stdlib.h>
#include<stdio.h>
#include<cmath>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n, k;
    	scanf("%d%d", &n, &k);
    	if(n%(k+1) == 1)
    		printf("Nandu\n");
    	else
    		printf("Rani\n");
    }
    
    return 0;
}
