#include<iostream>
#include<stdio.h>
#include<cstring>

using namespace std;


int isPalindrome(string s,int n)
{
	int i=0,j=n-1;
	while(i<j)
	{
		if(s[i]!=s[j])
			return 0;
		i++;
		j--;
	}

	return 1;
}

int main()
{
	
	string a,b,temp;
	int t,n,count,i;

	cin>>t;
	while(t--)
	{
		cin>>a;
		cin>>b;

		n=a.size();
		count=0;

		for(i=0;i<n;i++)
		{
			temp="";
			temp+=a.substr(0,i)+b+a.substr(i,n);

			// cout<<temp<<" ";
			if(isPalindrome(temp,temp.size()))
				count++;
		}
		temp="";
		temp+=a+b;
		if(isPalindrome(temp,temp.size()))
			count++;

		cout<<count<<"\n";

	}	

	return 0;
}
