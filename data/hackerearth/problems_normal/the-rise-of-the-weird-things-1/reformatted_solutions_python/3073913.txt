n = int(raw_input())
s = map(int, raw_input().split())
a = []
b = []
for i in s:
    if i % 2 == 0:
        a.append(i)
    else:
        b.append(i)
s1 = 0
for j in a:
    s1 = s1+j
a.append(s1)
s2 = 0
for k in b:
    s2 = s2+k
b.append(s2)
a.sort()
b.sort()
print ' '.join(map(str, a)), ' '.join(map(str, b))
