def hamdist(str1, str2):
    diffs = 0
    for i in xrange(len(str1)):
        if str1[i] != str2[i]:
            diffs += 1
    return diffs


t = int(raw_input())
for qq in xrange(t):
    nn, mm = map(int, raw_input().split())
    s1 = bin(nn)[2:]
    s2 = bin(mm)[2:]
    a = '0'*(32-len(s1)) + s1
    b = '0'*(32-len(s2)) + s2
    s1 = a
    s2 = b
    print hamdist(s1, s2)
