#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <vector>
#include <iostream>
#include<string>
#include <algorithm>
using namespace std;
long isp(long n){
    string st=to_string(n);
    long long k=n*n;
    string s=to_string(k);
    int d=st.length();
    int l=s.length();
    string  s1={};
    string s2={};
    if(l%2==0){
        long n1,n2;
        for(int j=0;j<d;j++){
            s1.push_back(s[j]);
        }
         for(int j=d;j<l;j++){
            s2.push_back(s[j]);
        }
       n1=atoi(s1.c_str());
       n2=atoi(s2.c_str());
       long n3=n1+n2;
       if(n3==n){
            return n;
        }
    }
    else{
        for(int j=0;j<d-1;j++){
            s1.push_back(s[j]);
        }
         for(int j=d-1;j<l;j++){
            s2.push_back(s[j]);
        }
        long n1=atoi(s1.c_str());
        long n2=atoi(s2.c_str());
        long n3=n1+n2;
        if(n3==n){
           return n;
        }
    }
    return 0;
 }

int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */  
    int p,q;
    cin>>p>>q;
    int c=0;
    for(long j=p;j<=q;j++){
        long k=isp(j);
        if(k==j){
            cout<<j<<" ";
            c++;
        }
    }
    if(c==0){
        cout<<"INVALID RANGE"<<endl;
    }
    //cout<<endl;
    return 0;
}
