from __future__ import division

N, P = map(int, raw_input().split())
Prob = []
Prob.append(0)
Prob.append(0)
Prob.append(P/100)
Prob.append(1 - Prob[2])
for i in xrange(4, N+1):
    Prob.append(Prob[i-2]*Prob[2] + Prob[i-3]*Prob[3])

print "%.6f" % Prob[N]
