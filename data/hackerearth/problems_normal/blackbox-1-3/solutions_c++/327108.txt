#include<iostream>
using namespace std;

void arrange(int *array1) {
    int mini = 0;
    int index = 0;
    for(int i=0;i<2;++i){
        mini = array1[i];
        index = i;
        for(int j=i+1;j < 3; ++j){
            if(array1[j] < mini){
                index = j;
                mini = array1[j];
            }
        }
        int temp = array1[i];
        array1[i] = array1[index];
        array1[index] = temp;
    }
}

bool triplets(int *array1) {

    if(((array1[0]*array1[0]) + (array1[1]*array1[1])) == (array1[2]*array1[2]))
        return true;
    else
        return false;
}

int calHCF(int b,int a){

        int t;
        while (b != 0) {
            t = b;
            b = a % b;
            a = t;
        }

        return a;
}
bool coprime(int *array1){

    int a = calHCF(array1[0],array1[1]);
    int b = calHCF(array1[1],array1[2]);

    if(a==1 && b==1)
        return true;
    else
        return false;
}
int main()
{

    std::ios_base::sync_with_stdio(false);

    int T;
    cin>>T;
    int array1[3];
    for(int i=0; i< T; ++i) {
        cin>>array1[0]>>array1[1]>>array1[2];
        arrange(array1);
        if(triplets(array1)){
            if(coprime(array1))
                cout<<"YES\n";
            else
                cout<<"NO\n";
        }
        else
            cout<<"NO\n";
    }

    return 0;
}
