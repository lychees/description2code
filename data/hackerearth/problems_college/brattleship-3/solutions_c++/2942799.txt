#include<iostream>
using namespace std;
int main()
{
    int T,R,C,W;
    cin>>T;
    for(int i=1;i<=T;i++){
        cin>>R>>C>>W;
        int x= (C/W);
        if((C%W)!=0)
            int r= (x*R)+W;
        int r= (x*R)+W-1;
        cout<<"Case #"<<i<<": "<<r<<endl;
    }
    return 0;
}
