#include <bits/stdc++.h>

using namespace std;

#define LL long long int 
#define sd(x) scanf("%d", &x)
#define F first
#define S second
#define mp make_pair
#define pb push_back
#define N 2000000
#define pii pair<int,int>

int main(){
	int t;
	LL b, e;
	int a;
	sd(t);
	while(t--){
		scanf("%lld %lld", &b, &e);
		b %= 10;
		a = 1;
		if(e == 0){
			printf("1\n");
		}
		else{
			e -= 1;
			e %= 4;
			e += 1;
			a = 1;
			while(e--){
				a *= b;
			}
			a %= 10;
			printf("%d\n", a);
		}
	}
	return 0;
}
