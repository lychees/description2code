#include<stdio.h>
int main()
{
	long long int t;
	scanf("%lld",&t);
	while(t--)
	{
		long long int x,y,z,a,count=0;
		scanf("%lld",&a);
		for(x=0; x<=a; x++)
		{
			for(y=0; y<=a; y++)
			{
				for(z=0; z<=a; z++)
				{
					if(x+3*y+5*z==a)
						++count;
				}
			}
		}
		printf("%lld\n",count);
	}
	return 0;
}
