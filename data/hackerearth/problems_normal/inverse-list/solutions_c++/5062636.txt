#include <iostream>
using namespace std;

int main()
{   
    int t;
    cin>>t;
    while(t--){
    	int n; cin>>n;
    	int a[n],b[n];
    	for(int i=1; i<=n; i++){
    		cin>>a[i];
    		b[a[i]]=i;
    	}
    	int f=0;
    	for(int i=1; i<=n; i++){
    		if(a[i]!=b[i]){
    		f=1;
    		break;
    		}
    	}
    	if(f==0)
    	cout<<"inverse\n";
    	else
    	cout<<"not inverse\n";
    }
    return 0;
}
