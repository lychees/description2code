#include <iostream>
#include <bits/stdc++.h>
using namespace std;
map <int,int> mp;
map <int,int> :: iterator it;
int a[1000005];
int main()
{
    int n,k,i,mn=0,mi=1000000;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    	mp[a[i]]++;
    }
    for(it=mp.begin();it!=mp.end();it++)
    {
    	mn=max(mn,it->second);
    }
    for(it=mp.begin();it!=mp.end();it++)
    {
    	if(it->second==mn)
    	{
    		mi=min(mi,it->first);
    	}
    }
    cout<<mi<<endl;
    return 0;
}
