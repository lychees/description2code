#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define mp make_pair
int bit[2*100*1000+9][26];
int n;
inline void upd(int at,int by,int w){
    for(;at<=n;at+=at&-at)
        bit[at][w]+=by;
    return;
}
inline int qry(int at,int w){
    int sum=0;
    for(;at>=1;at-=at&-at)
        sum+=bit[at][w];
    return sum;
}
char s[1<<20];
int main(){
    scanf("%s",s+1);
     n=strlen(s+1);
    memset(bit,0,sizeof bit);
    for(int i=1;i<=n;++i){
        int w=s[i]-'a';
        upd(i,1,w);
    }
    int m;
    scanf("%d",&m);
    while(m--){
        int k;char ch;
        scanf("%d %c",&k,&ch);
        int l=1,r=n;
        int mid=0;
        int pos=0;
        /*if(qry(n)<k){
          puts("NO");
          continue;
          }*/
        int w=ch-'a';
        while(l<=r){
            mid=(l+r)>>1;
            if(qry(mid,w)==k ){
                if(qry(mid-1,w)==k-1){
                    pos=mid;
                    goto POR; 
                }
                r=mid-1;
            }
            else if(qry(mid,w)<k){
                l=mid+1;
            }
            else if(qry(mid,w)>k){
                r=mid-1;
            }
        }

POR:
        s[pos]='*';
        upd(pos,-1,w);
        //cout<<pos-1<<"\n";

    }
//    puts("asdf");
    for(int i=1;i<=n;++i){
        if(s[i]!='*')printf("%c",s[i]);
    }
    printf("\n");
    return 0;
}
