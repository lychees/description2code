#include <iostream>
#include <math.h>
using namespace std;

int pali(int n) {
	int n1 = n, sum = 0,l=0,i=0;
	while(n1!=0) {
		l++;
		n1/=10;
	}
	n1 = n;
	while(n1!=0) {
		sum+=((n1%10)*pow(10,l-i-1));
		n1/=10;
		i++;
	}
	if(sum==n) {
		return 1;
	}
	return 0;
}

int main()
{
    int t,a,b;
    cin>>t;
    while(t--) {
    	int c = 0;
    	cin>>a>>b;
    	for(int i = a; i<=b; i++) {
    		c+=pali(i);
    	}
    	cout<<c<<endl;
    }
    return 0;
}
