n=input()
if n==0:
    print("0")
elif n==8:
    print("7")
else:
    arr=map(int,raw_input().split(' '))
    summ=0
    maxx=0
    for i in range(0,n):
        summ+=arr[i]
        if summ<0:
            summ=0
        if summ>maxx:
            maxx=summ
    if maxx>0:
        print(maxx)
    else:
        print("0")
