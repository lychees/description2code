#include <bits/stdc++.h>

using namespace std;

int main()
{
	int p;
	cin>>p;
	int t(p);
	cin.ignore();
	while(t--)
	{
		string s;
		getline(cin,s);
		int count=0,invalid=0;
		int upper=0,lower=0,no_alpha=0;
		if(s.length()>0&&s.length()<=100)
		{
			for(int i=0;i<s.length();i++)
			{
				if(s[i]>='A'&&s[i]<='Z')
					upper++,no_alpha=1;
				else if(s[i]>='a'&&s[i]<='z')
					lower++,no_alpha=1;
			}
			if(no_alpha==0)
				cout<<"Invalid Input"<<endl;
			else
			{
				int count=min(upper,lower);
				cout<<count<<endl;				
			}
		}
		else if(s.empty())
		{	
			cout<<"Invalid Test"<<endl;	
			break;
		}
		else
		{
			cout<<"Invalid Input"<<endl;
			break;
		}
	}
	if(p==0)
		cout<<"Invalid Test"<<endl;
}
