#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
    int t;cin>>t;
    	string s;
    	getline(cin,s);
    while(t--){
    	getline(cin,s);
    	int i,n=s.length(),j=0;
    	for(i=0;i<n;i++){
    		if(s[i]==' '){
    			reverse(s.begin()+j,s.begin()+i);
    			j=i+1;
    		}
    	}
    	reverse(s.begin()+j,s.begin()+i);
    	reverse(s.begin(),s.end());
    	cout<<s<<endl;
    }
    return 0;
}
