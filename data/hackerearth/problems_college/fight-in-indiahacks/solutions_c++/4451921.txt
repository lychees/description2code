#include <iostream>
using namespace std;
char ans[1000000];
int main()
{
	int t,n,m1,m2,i;
    cin >> t;
    while(t--){
    	cin >> n >> m1 >> m2;
    	ans[0] = 0;
    	if(m1 == 0 || m2 == 0){
    		ans[1] = 1;
    	}
    	else{
    		ans[1] = 0;
    	}
    	for(i = 1;i <= n;i++){
    		if(i >= m1 && !ans[i-m1]){
    			ans[i] = 1;
    		}
    		else if(i >= m2 && !ans[i-m2]){
    			ans[i] = 1;
    		}
    		else{
    			ans[i] = 0;
    		}
    	}
    	cout << int(ans[n]) << endl;
    }
    return 0;
}
