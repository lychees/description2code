#include<iostream>
using namespace std;
int main()
{
    int t,n,i,sum,value,k,a[8];
    int b[]={1,2,4,8,16,32,64,128};
cin>>t;
while(t--)
{
    scanf("%d",&n);
    while(n--)
    {
    for(i=0;i<8;i++)a[i]=0;
    scanf("%d",&value);
    k=0;
    sum=0;
    while(k<8)
    {
        if(value%2==0)sum=sum+b[k];
        value=value/2;
        k++;
    }
    cout<<(char)sum;
    }
    cout<<"\n";
}
return(0);
}
