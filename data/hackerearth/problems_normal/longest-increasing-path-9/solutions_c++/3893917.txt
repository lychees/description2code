#include <iostream>
#include <vector>
#include <algorithm>
#include <functional>

using namespace std;
int n,m;
vector<vector<long long>>v(100+5,vector<long long>(100+5,0));
vector<vector<long long>>dp(100+5,vector<long long>(100+5,0));


auto main()->int
{
    int t;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        vector<vector<long long>>vv(100+5,vector<long long>(100+5,0));
        v = vv;
        vector<vector<long long>>dpp(100+5,vector<long long>(100+5,0));
        dp = dpp;
        for(auto i=0;i<n;i++)
        {
            for(auto j=0;j<m;j++)
            {
                cin>>v[i][j];
            }
        }
        dp[0][0] = 1;
        long long ans = 1;
        for(auto i=1;i<m;i++)
        {
            if(v[0][i]>v[0][i-1])
            {
                dp[0][i] = dp[0][i-1] +1;
                if(dp[0][i]>ans) ans = dp[0][i];
            }
            else
            {
                v[0][i] = 1000005;
            }
        }


        for(auto i=1;i<n;i++)
        {
            if(v[i][0]>v[i-1][0])
            {
                dp[i][0] = dp[i-1][0] +1;
                if(dp[i][0]>ans) ans = dp[i][0];
            }
            else{
                v[i][0] = 1000005;
            }
        }

        for(auto i=1;i<n;i++)
        {
            for(auto j=1;j<m;j++)
            {
                    if(v[i][j]>v[i-1][j] && v[i][j]>v[i][j-1])
                    {
                        dp[i][j] = 1+max(dp[i-1][j],dp[i][j-1]);
                        if(dp[i][j]>ans) ans = dp[i][j];
                    }
                    else if(v[i][j]>v[i-1][j])
                    {
                        dp[i][j] = 1 + dp[i-1][j];
                        if(dp[i][j]>ans) ans = dp[i][j];
                    }
                    else if(v[i][j]>v[i][j-1])
                    {
                        dp[i][j] = 1 + dp[i][j-1];
                        if(dp[i][j]>ans) ans = dp[i][j];
                    }
                    else
                    {
                        v[i][j] = 1000005;
                    }
            }
        }
        cout<<ans<<endl;

    }
}
