import sys
sys.setrecursionlimit(1000000)

def f(i):
    if i>=n:
        return 0
    if dp[i]!=-1:
        return dp[i]
    dp[i]=max( f(i+1),f(i+2)+l[i])
    return dp[i]
t=input()
c=1
while t!=0:
    n=input()
    l=map(int,raw_input().split())
    dp=[-1]*(n+1)
    print "Case %d: %d" %(c,f(0))
    t-=1
    c+=1
