#include<bits/stdc++.h>
using namespace std;

const int N=105;
int dp[N][N*N];
int a[N];

void solve(int n) {
    memset(dp,0,sizeof(dp));
    for(int i=0;i<=n;i++) {
        dp[i][0]=1;
    }
    for(int i=1;i<=n;i++) {
        for(int sum=1;sum<N*N;sum++) {
            if(sum>=a[i-1]) {
                dp[i][sum]=dp[i-1][sum-a[i-1]];
            }
            dp[i][sum]=dp[i][sum]||dp[i-1][sum];
        }
    }
    int cnt=0;
    for(int sum=0;sum<N*N;sum++) {
        if(dp[n][sum]) {
            cnt++;
        }
    }
    printf("%d\012",cnt);
    return;
}
int main() {
    int tt;
    cin>>tt;
    while(tt--) {
        int n;
        cin>>n;
        for(int i=0;i<n;i++) {
            cin>>a[i];
        }
        solve(n);
    }
}
