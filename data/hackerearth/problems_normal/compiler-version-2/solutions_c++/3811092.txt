#include<iostream>
using namespace std;
int main()
{
string a;
while (getline(cin, a))
{
	int flag=1;
	for(int i=0;i<a.length();i++)
	{
		if(a[i]=='-'&&a[i+1]=='>'&&flag){ cout<<'.'; i+=1; }
		else if(a[i]=='/'&&a[i+1]=='/'&&flag){ flag=0; cout<<a[i]; }
		else cout<<a[i];
	}
 cout<<endl;
}
    return 0;
}
