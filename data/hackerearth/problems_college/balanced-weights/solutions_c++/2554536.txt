#include <bits/stdc++.h>
#define sll(a)  scanf("%lld",&a)
using namespace std;
template<typename T> T gcd(T a, T b) {if(!b) return a;return gcd(b, a % b);}
int main() {
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		long long int gcd1,b;
		sll(gcd1);
		for(int i=1;i<n;i++)
		   {
		   sll(b);
		    gcd1=gcd(gcd1,b);
		   }
		
		long long int res=gcd1*n*1LL;
		printf("%lld\n",res);
		
	}
	// your code goes here
	return 0;
}
