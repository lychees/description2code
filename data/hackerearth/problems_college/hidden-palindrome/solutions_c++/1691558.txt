#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define F(i,a,b) for(int i = (int)(a); i <= (int)(b); i++)
#define RF(i,a,b) for(int i = (int)(a); i >= (int)(b); i--)
#define MOD 1000000007
int func(char *S, int n)
{
    bool dp[n][n];
    memset(dp, 0, sizeof(dp));
    int MaxL = 1;
    F(i,0,n-1) dp[i][i] = true;
    int start = 0;
    F(i,0,n-2)
    {
        if (S[i] == S[i+1])
        {
            dp[i][i+1] = true;
            start = i;
            MaxL = 2;
        }
    }
    F(k,3,n)
    {
        F(i,0,n-k)
        {
            int j = i + k - 1;
            if (dp[i+1][j-1] && S[i] == S[j])
            {
                dp[i][j] = true;
                if (k > MaxL)
                {
                    start = i;
                    MaxL = k;
                }
            }
        }
    }
    return MaxL; 
}

int main()
