#include <bits/stdc++.h>

#define ll long long

using namespace std;

int main()
{
    int n, i, a[105];
    
    scanf("%d", &n);
    
    for( i = 0 ; i < n ; i++ )
    {
    	scanf("%lld",&a[i]);
    }
    
    sort(a, a+n);
    int b[105], j = 0;
    
    for( i = 1 ; i < n ; i++ )
    {
    	b[j++] = a[i] - a[i - 1];
    }
    
    int g = b[0];
    for( i = 1 ; i < j ; i++ )
    {
    	g = __gcd(g, b[i]);
    }
    
    vector<int> k;
    for( i = 2 ; i <= sqrt(g) ; i++ )
    {
    	if( g % i == 0 )
    	{
    		k.push_back(i);
    		if( g / i != i )
    			k.push_back(g/i);
    	}
    }
    k.push_back(g);
    sort(k.begin(), k.end());
    for( i = 0 ; i < k.size() ; i++ )
    {
    	printf("%d ", k[i]);
    }
    printf("\n");
    return 0;
}
