#include<bits/stdc++.h>
#define mod 1000000007
#define gc getchar
using namespace std;
typedef long long int ll;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
