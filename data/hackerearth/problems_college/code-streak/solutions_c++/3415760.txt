#include <bits/stdc++.h>
using namespace std;
#define ll long long int
unordered_map<ll,bool>visited;
//unordered_map<ll,ll>a;
//unordered_map<ll,ll>b;
void initialise(){
	for(int i=0; i<=1000001; i++){
	 visited[i] = false;
}
}
int main()
{
	ll t , n,i,a,b,ans;
	cin >> t;
	while(t--){
		//a.clear();
		//b.clear();
		visited.clear();
		initialise();
		cin >> n;
		ll count=0;
		ans = 0;
		 while(n--){
		 	cin >> a >> b;
		 	if(b==1 && visited[a]==false)
		 	{
		 		visited[a] = true;
		 		count++;
		 	}
		 	else  if(b==0)
		 	count=0;
		 	
			 ans = max(ans,count);
			// cout << ans << " " << count << endl;
		}
		cout << ans << endl;
	}
    return 0;
}
