#include <bits/stdc++.h>
#include <iostream>
#include<math.h>
#include <algorithm>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--)
	{
	  int n,count=0;
	  cin>>n;
	  int a[n];
	  for(int i=0;i<n;i++)
	  {
	      cin>>a[i];
	  }
	  sort(a,a+n);
	  
	  for(int i=0;i<n-1;i++)
	  {
	      for(int j=i+1;j<n;j++)
	      {
	          if( (a[i]!=a[j]) && (a[i]+a[j])%2==0)
	          {
	              count++;
	          }
	      }
	  }
	  cout<<count<<endl;
	   
	}
	return 0;
}
