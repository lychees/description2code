import re
z=[]
n=int(raw_input())
for i in range(n):
    text=raw_input()
    words = re.findall('\w+', text)
    cwords = {}
    for word in words:
         try:
             cwords[word] += 1
         except KeyError:
             cwords[word] = 1
    z.insert(i,len(cwords))
for i in z:
    print i
