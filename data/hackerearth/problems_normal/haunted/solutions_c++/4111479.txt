#include<bits/stdc++.h>
using namespace std;
stack<int> S;
priority_queue<int>PQ;
int main()
{
	int age,days,ind,i,j,titles,con_tropy,prev_age,prev_value,inpt;
	cin>>days>>inpt;
	map<int,int>group;
	for(ind=0;ind<days;ind++)
		{
		cin>>age;
		if(ind==0)
			{
			cout<<age<<" "<<"1"<<endl;
			S.push(age);
			group[age]++;
			PQ.push(1);
			}
		else
			{	
			group[age]++;
			//cout<<group[age]<<endl;
			//map<int,int>::iterator it,it1;
			
					if(group[age]>PQ.top()){
						S.push(age);
						PQ.push(group[age]);
						
						}
					else if(group[age]==PQ.top())
						{
							if(age>S.top())
								S.push(age);
						}
					cout<<S.top()<<" "<<PQ.top()<<endl;
					
				}
					
				}
		
		
	return 0;
}
