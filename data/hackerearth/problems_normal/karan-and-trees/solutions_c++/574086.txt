#include <iostream>
using namespace std;

int t;unsigned long long a,b;

int main()
{
	cin>>t;
	while(t--)
	{
		scanf("%d",&a);
		scanf("%d",&b);
		
		while(1)
		{
		
		if(a>b)
		{
			if(a%2==0)
			a/=2;
			else if(a%2!=0)
			a=(a-1)/2;
			
			if(a==b)
			{
				
			cout<<a<<endl;
			break;
			}
		}
		else if(b>a)
		{
			if(b%2==0)
			b/=2;
			else if(b%2!=0)
			b=(b-1)/2;
			
			if(a==b)
			{
				
			cout<<b<<endl;
			break;
			}
		}
		
		}
		
	}
    return 0;
}
