#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int v=0;v<t;v++){
    	long long a,b,c;
    	cin>>a>>b>>c;
    	int n=0;
    	while(pow(a*b,n)<c)
    	n++;
    	bool ad=true;
    	if((pow(a*b,n-1)*a)>=c)
    	ad=false;
    	if((pow(a*b,n-1)*b)>=c)
    	ad=false;
    	if(ad)
    	cout<<"Bob"<<"\n";
    	else
    	cout<<"Adam"<<"\n";
    	
    }
    
    
    return 0;
}
