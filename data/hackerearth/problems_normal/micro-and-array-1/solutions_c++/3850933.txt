#include <iostream>
using namespace std;

#define maxn 100001
int a[maxn];

int main()
{
    int n, q, o, x, y, v, i;
    
    cin >> n >> q;
    for(i=1; i<=n; ++i) {
    	cin >> a[i];
    }
    
    while (q--) {
    	cin >> o;
    	
    	if (o) {
    		cin >> v;
    		
    		for (i=1;i<=n;++i) {
    			if (a[i]>=v) {
    				cout << i << endl;
    				break;
    			}
    		}
    		if (i > n) {
    			cout << -1 << endl;
    		}
    	} else {
    		cin >> x >> y;
    		
    		a[x] = y;
    	}
    }
    
    return 0;
}
