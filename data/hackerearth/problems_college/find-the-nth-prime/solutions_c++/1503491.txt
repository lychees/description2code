#include <iostream>
#include <cstdlib>

using namespace std;

bool IsPrime(int);

int main()
{
		int t;
		int n;
		int k;
		int p[1000];
		
		p[0] = 2;
		p[1] = 3;
		
		for(int i=6,k=1; k<1000; i = i+6){
			if(IsPrime(i-1))
				p[++k] = i-1;
			if(IsPrime(i+1))
				p[++k] = i+1;
		}
		
		t = 1;
		
		int *temp = new int[t];
		
		for(int i=0; i<t; i++)
			cin >> temp[i];
		
		for(int i=0; i<t; i++)
			cout << p[temp[i]-1] << endl;
		
		
}
bool IsPrime(int n)
{
	for(int i=2; i*i <= n; i++)
	{
			if(n % i == 0)
				return false;
	}
	return true;
}
