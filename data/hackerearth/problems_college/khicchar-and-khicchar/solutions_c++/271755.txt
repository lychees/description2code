#include <iostream>
#include <string>
using namespace std;

long gcd(long a, long b)
{
	if(a==0)
		return b;
	else
		return gcd(b%a,a);
}

long modu(long a, string b)
{
	long i,d=0;
	for(i=0;i<b.length();i++)
	{
		d=(d*10+(long(int(b[i])-48)))%a;
	}
	return d;
}
	
int main()
{
	long i,n,a,f;
	string b,c,d,e,g,ss;
	cin>>n;
	for(i=0;i<n;i++)
	{
		getline(cin,g);
		getline(cin,e);getline(cin,d);getline(cin,c);getline(cin,b);cin>>a;
		
		f=modu(a,b);
		if(a>f)
			a=gcd(f,a);
		else
			a=gcd(a,f);
			
		f=modu(a,c);
		if(a>f)
			a=gcd(f,a);
		else
			a=gcd(a,f);
		
		f=modu(a,d);
		if(a>f)
			a=gcd(f,a);
		else
			a=gcd(a,f);
		
		f=modu(a,e);
		if(a>f)
			a=gcd(f,a);
		else
			a=gcd(a,f);

		ss.append(to_string(a));
		ss.append("\n");
		b.clear();c.clear();d.clear();e.clear();
	}
	cout<<ss;
    return 0;
}
