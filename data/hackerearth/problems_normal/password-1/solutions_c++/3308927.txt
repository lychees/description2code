#include <iostream>
#include<set>
#include<string.h>
using namespace std;
void reverse(char *s)
{
	for(int i=0,j=strlen(s)-1;i<j;++i,--j)
	{
		char temp = s[i];
		s[i] = s[j];
		s[j] = temp;
	}
}
int main()
{
    int n;
    cin>>n;
    set<string> hs;
    set<string> :: iterator it;
    for(int i=0;i<n;++i)
    {
    	char s[101];
    	cin>>s;
    	reverse(s);
    	it = hs.find(s);
    	if(it != hs.end())
    	{
          int len = strlen(s);
          cout<<len<<" "<<s[len/2];
          break;
    	}
    	reverse(s);
    	hs.insert(s);
    }
    return 0;
}
