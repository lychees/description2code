#include<bits/stdc++.h>
using namespace std;
int main(){		
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		cin>>n;
		int cube[60][60][60];
			int dp[60][60][60]={0};
		for(int i=1;i<=n;i++){
			for(int j=1;j<=n;j++){
				for(int k=1;k<=n;k++){
					int x;
					cin>>x;
					cube[i][j][k]=x;
					dp[i][j][k]=x;
				}
			}
		}
		/*
		for(int i=1;i<=n;i++){
			for(int j=1;j<=n;j++){
				for(int k=1;k<=n;k++){
					cout<<cube[i][j][k]<<" ";
			}
		}
		*/
		
	
		for(int i=n;i>=1;i--){
		for(int j=n;j>=1;j--){
			for(int k=n;k>=1;k--){
				dp[i][j][k]=dp[i-1][j][k]+dp[i][j-1][k]+dp[i][j][k-1]-(dp[i-1][j-1][k]+dp[i][j-1][k-1]+dp[i-1][j][k-1])+dp[i-1][j-1][k-1];
				dp[i][j][k]=cube[i][j][k]-dp[i][j][k];			
			}
		}
	}
	
		for(int i=1;i<=n;i++){
			for(int j=1;j<=n;j++){
				for(int k=1;k<=n;k++){
					cout<<dp[i][j][k]<<" ";
			}
			cout<<"\n";
		}
		
	
	
	}}}
