#include<iostream>
using namespace std;

void hexdiff(int x, int hexa[])
{
    int y,i;
    i=1;
    y=x;
    hexa[0]=0;
    hexa[1]=0;
    do
    {
        hexa[i] = y%16;
        y = y/16;
        i--;

    } while(y>0);

    for (i=0;i<2;i++)
    {
        y = hexa[i];

        if (y==10)
            cout << 'U';
        else if (y==11)
            cout << 'V';
        else if (y==12)
            cout << 'W';
        else if (y==13)
            cout << 'X';
        else if (y==14)
            cout << 'Y';
        else if (y==15)
            cout << 'Z';
        else
            cout << y;
    }
}

int main()
{
    
    int hexa[2]={0}, x,y,i=1,t,r,g,b,ctr=1;

    cin>>t;
    
    while(t--)
    {
        cin>>r>>g>>b;
        cout<<"Case "<<ctr<<":";
        cout<<" ";
        hexdiff(r,hexa);
        cout<<" ";
        hexdiff(g,hexa);
        cout<<" ";
        hexdiff(b,hexa);
        cout<<endl;
        ctr++;
    }

    
    return 0;
}
