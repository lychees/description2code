#include <bits/stdc++.h>
using namespace std;

long long josephus(long long n, long long k)
{
    long long r = 0;
    long long i = 1;
    while (i <= n)
    {
      r = (r + k) % i;
      i+= 1;
     }
    return r+1;
}

int main() {
	int t;
	cin>>t;
	
	while(t--)
	{
		long long n;
		cin>>n;
		
		
			long long ans;
			ans = josephus(n, 2);
			cout<<ans<<endl;
		
	}
	
	return 0;
}
