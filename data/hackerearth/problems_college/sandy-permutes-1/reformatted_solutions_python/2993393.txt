N, K = map(int, raw_input().split())
arr = map(int, raw_input().split())
if K >= N:
    for x in range(N, 0, -1):
        print x,
else:
    swaps_made = 0
    for i, j in enumerate(sorted(arr, reverse=True)):
        if j > arr[i]:
            p = arr.index(j, i)
            arr[i], arr[p] = arr[p], arr[i]
            swaps_made += 1
            if swaps_made >= K:
                break

    print ' '.join(map(str, arr))
