#include <iostream>
using namespace std;

int main()
{
   int T;
   cin>>T;
   for(int i=0; i<T; i++)\
   {
  		int n;
  		cin>>n;
  		int initial[n];
  		int final[n];
  		for(int j=0; j<n;j++)
  		{
  			cin>>initial[j];
  		}
  		for(int j=0; j<n;j++)
  		{
  			cin>>final[j];
  		}
  		int count = 0;
  		for(int j=0; j<n; j++)
  		{
  			if(initial[j] == final[j])
  				continue;
  			for(int k=0; k<n; k++)
  			{
  				if(initial[k] == final[j])
  				{
  					if(k > j)
  					{
	  					for(int x = k; x > j; x--)
	  					{
	  						initial[x]^=initial[x-1]^=initial[x]^=initial[x-1];
	  						count++;
	  					}
	  				}
	  				else
  					{
	  					for(int x = j; x < k-1; x++)
	  					{
	  						initial[x]^=initial[x+1]^=initial[x]^=initial[x+1];
	  						count++;
	  					}
	  				}
	  				break;
  				}
  			}
  		}
  		cout<<count<<endl;
   }
   return 0;
}
