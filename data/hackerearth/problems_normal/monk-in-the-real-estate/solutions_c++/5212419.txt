#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,x,y;
    cin>>t;
    while(t--){
    	int m;
    	cin>>m;
    	int a[10001];
    	memset(a,0,sizeof(a));
    	
    	for(int i=0;i<m;i++)
    	{
    		cin>>x>>y;
    		a[x]=1;
    		a[y]=1;
    	}
    	int c=0;
    	for(int i=0;i<10001;i++)
    	if(a[i]==1) c++;
    	
    	cout<<c<<"\n";
    }
    return 0;
}
