#include <iostream>
using namespace std;
#include <bits/stdc++.h>

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	vector<int> a;
    	for(int i=0; i<n; ++i)
    	{
    		int b;
    		cin>>b;
    		a.push_back(b);
    	}
    	sort(a.begin(),a.end());
    	int c=0;
    	for(int i=0; i<a.size(); ++i)
    	{
    		c++;
    		if((a[i]+1)==a[i+1])
    		i++;
    	}
    	cout<<c<<endl;
    }
    return 0;
}
