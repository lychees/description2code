    #include<iostream>
    #include<algorithm>
    using namespace std;
    int main()
    {
        int t;
        cin>>t;
        while(t--)
        {
            long long n,i;
            cin>>n;
            int s[n];
            for(i=0;i<n;i++)
                cin>>s[i];
     
            sort(s,s+n);
            for(--i;i>=0;i--){
                cout<<s[i]<<" ";
            }
            cout<<endl;
     
        }
        return 0;
    }
