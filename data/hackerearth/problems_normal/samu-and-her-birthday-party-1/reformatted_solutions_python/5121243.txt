for _ in range(int(raw_input())):
    n, k = map(int, raw_input().split())
    a = [int(raw_input(), 2) for __ in range(n)]
    ans = []
    for i in range(2**k):
        c = 0
        for j in a:
            if j & i == 0:
                break
            c += 1
        if c == len(a):
            ans.append(bin(i).count("1"))
    print min(ans)
