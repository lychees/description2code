memo = [0] * 1000001

for i in xrange(2, 1000+1):
    j = 2
    while int(pow(i, j)) <= 1000000:
        idx = int(pow(i, j))
        memo[idx] = 1
        j += 1

memo[1] = 1
for _ in xrange(input()):
    x = input()
    ans = False
    for i in xrange(1, (x >> 1)+1):
        if memo[i] and memo[x-i]:
            ans = True
            break

    if ans:
        print "Yes"
    else:
        print "No"
