#include<bits/stdc++.h>
using namespace std;

int main(){
    int t;
    cin>>t;

    while(t--){
        int n,tmp;
        cin>>n;

        vector<int> a;

        for(int i=0;i<n;i++){
            cin>>tmp;
            a.push_back(tmp);
        }

        vector<int> res(n+1,0);

        for(int i=0;i<n;i++){
            if(i==0){
                res[0]=a[0];
                continue;
            }

            if(i==1)
                res[1] = max(a[0],a[1]);

            else
                res[i] = max(res[i-1],a[i] + res[i-2]);
        }

        cout<<res[n-1]<<endl;
    }

    return 0;
}
