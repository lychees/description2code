#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t,n;
    cin >>t;
    int zero=0,one=0;
    while(t--)
    {
    	cin >>n;
    	int a;
    	zero=0;
    	one=0;
    	for(int i=0;i<n;i++)
    	{
    		cin >>a;
    		if(a==1)
    			one++;
    		else
    			zero++;
    	}
    	if(one%2!=0)
    		cout << one<<endl;
    	else 
    		cout << zero<<endl;
    }
    return 0;
}
