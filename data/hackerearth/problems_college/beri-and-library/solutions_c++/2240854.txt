#include<bits/stdc++.h>
using namespace std;

#define s(n)                        scanf("%d",&n)
#define sc(n)                       scanf("%c",&n)
#define sl(n)                       scanf("%lld",&n)
#define sf(n)                       scanf("%lf",&n)
#define ss(n)                       scanf("%s",n)
#define su(n)                       scanf("%llu",&n)

typedef unsigned long long int ull;


int main()
{
	ull t;
	su(t);
	while (t--)
	{
		ull n;
		int ans = 0;
		su (n);
	//	cout << n << endl;
		while (n > 0)
		{
			if (n & 1)
				ans++;
			n = n >> 1;
		}
		cout << ans << endl;
	}	
}
