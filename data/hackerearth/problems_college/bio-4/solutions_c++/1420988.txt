#include <iostream>

using namespace std;

int main()
{
	int t,ans,i,j,aaa,k;
	string a,b;
	cin>>t;
	while (t--)
	{
		cin>>a>>b;
		ans=a.length();
		for (i=0;i<a.length();i++)
		{
			k=i;
			aaa=0;
			for (j=0;j<b.length() && k<a.length();j++,k++)
			{
				if (a[k]!=b[j]) aaa++;
			}
			if (j==b.length())
			ans=min(ans,aaa);
		}
		cout<<ans<<endl;
		
	}
}
