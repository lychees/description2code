#include <iostream>
using namespace std;


long long int a[1000005];
void pre()
{
	a[1]=1;
	for(long long int i=2;i<1000004;i++)
	{
		if(a[i]!=0)
			continue;
		for(long long int j=i*i;j<1000004;j=j*i)
		{
			
			a[j]=1;
		}
	}
}
int main()
{
	pre();
	int t;
	long long int n,ans;
	cin >>t;
	while(t--)
	{
		cin>> n;
		ans=1;
		for(int i=0;i<n;i++)
		{
			if(a[i]==1&&a[n-i]==1)
			{
				cout <<"Yes"<<endl;
				ans=0;
				break;
			}
		}
		if(ans)
			cout << "No"<<endl;
	}
    
}
