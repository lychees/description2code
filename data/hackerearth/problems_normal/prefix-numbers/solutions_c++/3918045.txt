int pow(int a,int b){
    int ans=1;
    while(b){
        if(b&1)ans=ans*(ll)a%mod;
        a=(ll)a*a%mod;b>>=1;
    }
    return ans;
}

const int M=7*9*4;
const ll mo=1LL<<32;
typedef unsigned int arr[M][M];
arr a,b;
void mul(arr &a,arr &b){
    static arr c;
    memset(c,0,sizeof(c));
    rep(i,0,M)
        rep(j,0,M)
            rep(k,0,M){
                c[i][j]+=1LL*a[i][k]*b[k][j]%mo;
                if(c[i][j]>=mo)c[i][j]-=mo;
            }
    rep(i,0,M)rep(j,0,M)a[i][j]=c[i][j];
}
