#include <bits/stdc++.h>
using namespace std;

long long power(long long base,long long p)
{
	if(p==0)
		return 1;
	long long x;
	x= power(base,p/2);
	if(p%2==1)
		return x*x*base;
	else
		return x*x;
}

map<long long ,long long>m;
map<long long ,long long >::iterator it;
int main()
{
	long long i;

	long long num,t,j,p,c;
	for(i=0;i<63;i++) {
		m[power(2,i)]++;
}
	cin>>t;
	while(t--) {
		c=0;
		cin>>num;
		for(j=0;num-j>=2;){
			for(it=m.begin();it != m.end();it++) {
				p = it->first;
				if(num-j == p)
				{
					j += p;
					c++;
					break;
				}
				if(num-j <= p){
					j += (p/2);

					c++;
					break;
				}
			}
		}
		c += num-j;
		cout << c<<endl;

	}

}
