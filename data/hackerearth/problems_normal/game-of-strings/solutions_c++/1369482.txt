char c[2][5010];
int n,m,dp[5010][5010];

int rec(int i,int j)
{
	if(i==0 || j==0)
		return 0;
	if(dp[i][j]!=-1)
		return dp[i][j];
	if(c[0][i-1]==c[1][j-1] && (c[0][i-1]!='a' && c[0][i-1]!='o' && c[0][i-1]!='i' && c[0][i-1]!='u' && c[0][i-1]!='e' && c[0][i-1]!='A' && c[0][i-1]!='O' && c[0][i-1]!='I' && c[0][i-1]!='U' && c[0][i-1]!='E'))
	{
		//cout<<c[0][j-1]<<'\n';
		dp[i][j]=(1+rec(i-1,j-1));
	}
	else
	{
		dp[i][j]=max(rec(i,j-1),rec(i-1,j));
	}
	return dp[i][j];
}

void scan()
{
	scanf("%s %s",c[0],c[1]);
	n=strlen(c[0]);
	m=strlen(c[1]);
}

void out()
{
	FILL(dp,-1);
	cout<<rec(n,m)<<'\n';
}

int main()
{
	scan();
	out();
	return 0;
}
