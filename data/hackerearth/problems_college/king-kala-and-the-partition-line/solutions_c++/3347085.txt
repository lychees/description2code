#include <bits/stdc++.h>
using namespace std;
 
typedef long long LL;
 
#define PII       pair<LL,LL>
#define all(c)    c.begin(),c.end()
#define sz(c)     (int)c.size()
#define clr(c)    c.clear()
#define pb        push_back
#define mp        make_pair
#define cin(x)	scanf("%d" , &x)
#define MOD		1000000007
#define EPS       1E-10

int main()
{
	int t;
	cin(t);
	while(t--)
	{
		int n;
		cin(n);
		int a,b,c;
		cin(a);
		cin(b);
		cin(c);
		int a1 = 0 , a2 = 0;
		for(int i = 0; i < n; i++)
		{
			int x,y;
			cin(x);
			cin(y);
			int delta = a*x + b*y + c;
			if(delta > 0)
				a2 += 1;
			else if(delta < 0)
				a1 += 1;
		}
		printf("%d\n", max(a1,a2));
	}
	return 0;
}
