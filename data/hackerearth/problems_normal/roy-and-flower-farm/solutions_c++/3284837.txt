#include <bits/stdc++.h>
using namespace std;
int cost[101], sell[101];
int main(){
int test_case,n, a;
scanf("%d", &test_case);
while(test_case--){
	scanf("%d %d", &n, &a);
	for(int i=0; i<n; i++){
		scanf("%d %d", &sell[i], &cost[i]);
	}
	int dp[n+1][a+1];
	for(int j=0; j<=a; j++){
		dp[0][j]=0;
	}
	for(int j=0; j<=n; j++){
		dp[j][0]=0;
	}
	int p1=INT_MAX, p2=-1;
	for(int i=1; i<=n; i++){
		for(int j=0; j<=a; j++){
			dp[i][j] = dp[i-1][j];
			if(j-cost[i-1] >= 0){
				dp[i][j] = max(dp[i-1][j], dp[i-1][j-cost[i-1]]+sell[i-1]);

			}
			if(a-j+dp[i][j]>p2 || (a-j+dp[i][j]==p2 && j<p1)){
				p1=j;
				p2= a-j+dp[i][j];
			}
		}
	}
	cout<<p1<<" "<<p2<<endl;
}

}
