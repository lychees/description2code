    #include <iostream>
    #include<map>
    
    using namespace std;
     
    int main() {
    	int n;
    	cin >> n;
    	string s;
    	long long count = 0;
    	map<string, int> m;
    	for(int i = 0; i < n; ++i) {
    		cin >> s;
    		count += m[s];
    		string rev(s.rbegin(), s.rend());
    		m[rev]++;
    	}

    	cout << count << endl;
	    return 0;
    }
