n = input()
for z in range(n):
    a = raw_input()
    a = list(a)
    c = []
    b = []
    for i in range(len(a)):
        if a[i].isdigit() == 1:
            r = map(int, a[i])
            c = []
            for j in range(r[0]):
                c.extend(b)
            b = c
        else:
            b.append(a[i])
    print ''.join(c)
