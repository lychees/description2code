#include <iostream>
using namespace std;
int parity(int a)
{
	int b;
	b=a^(a>>1);
	b=b^(b>>2);
	b=b^(b>>4);
    b=b^(b>>8);
	b=b^(b>>16);
	return b&1;
}
int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(parity(n)==0)
    	 cout<<"even"<<endl;
    	 else
    	   cout<<"odd"<<endl;
    }
    return 0;
}
