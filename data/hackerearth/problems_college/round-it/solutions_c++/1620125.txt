#include <iostream>
using namespace std;

int main()
{
	int t,n,b,p,res,a;
	cin>>t;
	while(t--)
	{
		cin>>n>>b;
		p=b;
		if(p%2!=0)
		p+=1;
		a=n%b;
		if(a<(p/2))
		{
			a=n/b;
			res=a*b;
		}
		else
		{
			a=n/b;
			res=(a+1)*b;
		}
		cout<<res<<endl;
		 
	}
    //cout << "Hello World!" << endl;
    return 0;
}
