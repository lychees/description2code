#include<bits/stdc++.h>
using namespace std;
char a[1000004];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int l=0,c=0,i;
        long long int s=0;
        string a;
        cin>>a;
        l=a.length();
        for(i=0;i<l;i++)
            {
                if(a[i]=='H')
                    c++;
                else
                    c=0;
                s+=c;
            }
            printf("%lld\n",s);
    }
    return 0;
}
