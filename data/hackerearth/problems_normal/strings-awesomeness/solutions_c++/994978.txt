int main() {
	string s; cin >> s;
	int n = s.length();
	for (int i = 0; i < n; i++)
		if (s[i] >= 'A' && s[i] <= 'Z')
			s[i] += 'a' - 'A';
	vector<bool> a(n);
	for (int i = 0; i + 11 <= n; i++) {
		string t = s.substr(i, 11);
		sort(t.begin(), t.end());
		a[i] = (t == "aaceehhkrrt");
	}
	vector<int> sum(n + 1);
	for (int i = 1; i <= n; i++)
		sum[i] = sum[i - 1] + (a[i - 1] ? 1 : 0);
	int q; cin >> q;
	while (q--) {
		int l, r; scanf("%d%d", &l, &r);
		if (r - l <= 9) printf("0\n");
		else printf("%d\n", sum[r - 10] - sum[l - 1]);
	}
}
