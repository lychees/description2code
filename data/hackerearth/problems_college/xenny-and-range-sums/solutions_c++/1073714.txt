#include <iostream>
using namespace std;

int main()
{
    long int n,q;
    cin>>n>>q;
    long int arr[n];
    long int i;
    for(i=0;i<n;i++)
    {
    	cin>>arr[i];
    }
    while(q--)
    {
    	long int l,r;
    	cin>>l>>r;
    	l--;r--;
    	long long sum=0,diff=0;
    	for(i=l;i<=r;i++)
    	{
    		sum+=arr[i];
    		if(i!=r)
    		 diff+=arr[i];
    	}
    	cout<<sum<<" "<<arr[r]-diff<<endl;
    }
    return 0;
}
