#include <bits/stdc++.h>

using namespace std;

#define true 0
#define false 1

map<int, int> ans;

int prime[10002];

void markPowers(int x) {

    int i = 2;

    while(i*x <= 10001) {
        prime[i*x] = false;
        i++;
    }
}

void getPrime(int limit) {
    int cnt = 2;


    for(long long int i = 2; i <= limit; i++) {
        if(prime[i] == true) {

            cnt++;
       //     cout << i <<endl;
            markPowers(i);

        }
    }

}


int main()
{

    int t;

    getPrime(10001);
    long long x, n;

    prime[1] = false;
    prime[0] = false;

    cin >> t;

    while(t--) {
        cin >> n;
        long long cnt = 0;

        for(int i = 0; i < n; i++) {

            cin >> x;

            if(prime[x] == true) {
                cnt++;

            }
        }
        cout << cnt << endl;
    }


}
