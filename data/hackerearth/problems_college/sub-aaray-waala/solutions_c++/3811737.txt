#include<iostream>
#include<algorithm>
#define m 1000000007
using namespace std;
long long int power(long int a,long long int b)
{
	long long int c;
	if(b==0)
	return 1;
	else if(b==1)
	return a%m;
	else if(b%2==0)
	{
		c=power(a,b/2);
		return (c*c)%m;
	}
	else
	{
		c=power(a,b/2);
		return (((c*c)%m)*a)%m;
	}
}
int main()
{
	int t,n,i;
	long int a[100002];
	cin>>t;
	int p;
	long long int ans;
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a,a+n);
		ans=1;
		p=0;
		for(i=1;i<n;i++)
		{
			if(a[p]==a[i])
			p=i;
			else
			{
				ans++;
				p=i;
			}
		}
		cout<<power(n,ans)<<endl;
	}
	return 0;
}
