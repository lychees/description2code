  short base = 0, c = 0, car;
  int a, b, t, sum;

  while (t) {
    base = (base > t % 10) ? base : t % 10;
    t /= 10;
  }
  t = b;
  while (t) {
    base = (base > t % 10) ? base : t % 10;

  }
  base++;
  car = 0;
  while (a || b || car) {
    car = (a % 10 + b % 10 + car) / base;

    b /= 10;
    c++;
  }

}
