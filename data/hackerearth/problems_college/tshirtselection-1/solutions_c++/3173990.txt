#include<iostream>
#include<set>
#include<cstdio>
#define TAB "\t"

using namespace std;

set<int> s;

void fun(int i,int f,int size,int *arr)
{

	int maxv = 0;
	int l=0;
	int m=0;
	while(1)
	{
		while(f != size && s.find(arr[f]) == s.end())
		{
			s.insert(arr[f]);
			f++;
		}



		if(f - i > maxv)
		{
			l = i+1;
			m = f;
			maxv = f-i;
		}

		if(f == size)
        {
            cout<<l<<" "<<m<<endl;
            return;
        }
        else
        {
            set<int>::iterator it = s.find(arr[f]);
            while(s.find(arr[i]) != it)
            {
                s.erase(s.find(arr[i]));
                i++;
            }

            s.erase(s.find(arr[i]));
            i++;
        }
	}
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int *arr = new int[n+2];

    	for(int i=0;i<n;i++)
    	{
    		cin>>arr[i];
    	}
    	fun(0,0,n,arr);
    	s.clear();

    }
    return 0;
}
