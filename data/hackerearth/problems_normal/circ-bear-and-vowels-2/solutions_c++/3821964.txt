#include <iostream>
#include <string>

using namespace std;

int main() {
	int t;
	cin >> t;

	string str;
	while (t--) {
		cin >> str;

		int len = str.length();
		int con = 0, vow = 0;
		for (int i = 0; i < len; ++i) {
			if (str[i] == 'a' || str[i] == 'e' || str[i] == 'i' || str[i] == 'o' || str[i] == 'u' || str[i] == 'y')
				vow++;
			else
				con++;
		}

		if (con > vow) {
			cout << "hard" << endl;
		}
		else {
			int count = 0;
			bool flag = false;
			for (int i = 0; i < len; ++i) {
				if (str[i] == 'a' || str[i] == 'e' || str[i] == 'i' || str[i] == 'o' || str[i] == 'u' || str[i] == 'y') {
					count = 0;
				}
				else {
					count++;
					if (count == 3) {
						flag = true;
						break;
					}
				}				
			}

			if (flag) {
				cout << "hard" << endl;
			}
			else {
				cout << "easy" << endl;
			}
		}
	}
}
