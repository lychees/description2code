#include <cstdlib>
#include <iostream>

using namespace std;
long T,I,n,k,i,s;
int main(int argc, char *argv[])
{   cin>>T;
    for(I=1;I<=T;I++)
    {   cin>>n;
        k=0;
        if(n==1)
           cout<<1<<endl;
        else
        {   s=0;i=2;k=0;
            while(s<n)
            { s=s+i;
              i++;
              k++;
            }
            cout<<k<<endl;
        }
                        
    } 
    
    return EXIT_SUCCESS;
}
