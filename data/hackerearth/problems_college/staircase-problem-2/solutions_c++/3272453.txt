#include <iostream>
using namespace std;
int d[1000001];
int p[1000001];
int dp(int n)
{
	if(n == 1 || n == 2)
		return 1;
	if(n == 3)
		return 2;
	int i;
	d[1] = 1;
	d[2] = 2;
	p[1] = 1;
	p[2] = 1;
	for(i = 3 ; d[i-1] < n; i++) {
		d[i] = d[i-1]+i;
	}
	return i-2;
}
int main()
{
    int i, n, t;

    cin>>t;
    while(t--) {
    	cin>>n;
    	cout<<dp(n)<<endl;
    }
    return 0;
}
