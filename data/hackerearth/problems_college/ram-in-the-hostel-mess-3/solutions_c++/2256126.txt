#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t,n;
    cin >>t;
    for(int test=0;test<t;test++)
    {
    	int result;
    	cin >>n;
    	if(n==1)
    	result = 2;
    	else
    	{
    		int p=2,ptp=1,element;
    		for(int i=2;i<=n;i++)
    		{
    			element = p+ptp;
    			ptp=p;
    			p=element;
    		}
    		result = element;
    	}
    	cout << result*result << endl;

    }
}
