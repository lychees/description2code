#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str;
		cin>>str;
		int l = str.length();
		int f[26] = {0};
		for(int i=0;i<l;i++)
		{
			f[str[i]-'a']++;
		}
		int c=0;
		for(int i=0;i<26;i++)
		{
			if(f[i]%2!=0)
			{
				c++;
			}
			if(c>1)
			{
				break;
			}
		}
		if(c==0||c == 1)
		{
			cout<<"Yes"<<"\n";
		}
		else
		{
			cout<<"No"<<"\n";
		}
	}
}
