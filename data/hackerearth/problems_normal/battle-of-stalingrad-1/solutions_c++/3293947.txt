#include<bits/stdc++.h>
#include<math.h>
using namespace std;
float area(int x1, int y1, int x2, int y2, int x3, int y3)
{
   return abs((x1*(y2-y3) + x2*(y3-y1)+ x3*(y1-y2))/2.0);
}
int main()
{ 
	 int tc,x1,x2,x3,x4,y1,y2,y3,y4 ;float A,A1,A2,A3 ;
	 cin>>tc ;
	 while(tc--)
	 {
	 	cin>>x1>>y1>>x2>>y2>>x3>>y3>>x4>>y4 ;
	 	 
	 	 A=area(x1,y1,x2,y2,x3,y3) ;
	 	 A1=area(x1,y1,x2,y2,x4,y4) ;
	 	 A2=area(x1,y1,x3,y3,x4,y4);
	 	 A3=area(x3,y3,x2,y2,x4,y4);
	 	 
	 	 if(A==A1+A2+A3)
	 	 {
	 	 	cout<<"INSIDE"<<endl ;
	 	 }
	 	 else
	 	 cout<<"OUTSIDE"<<endl ;
	 }
   
    return 0;
}
