#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int T,N,A[100000],M,sum,sum1;
    cin>>T;
    while(T--)
    {
        cin>>N;
        sum=0;
        sum1=0;
        int i;
        for(i=0;i<N;i++)
        {
            scanf("%lld",&A[i]);
        }

        scanf("%lld",&M);
        long long int j;
        if(M==0)
        {
            cout<<"0\n";
        }
        else
        {
            for(i=0;i<N;i++)
            {
                sum+=A[i];
                if(sum>=M)
                {
                    cout<<i+1<<"\n";
                    break;
                }
            }
            if(sum<M)
            {
                long long int t=M%sum;
                if(t==0)
                    t=sum;

                for(i=0;i<N;i++)
                {
                    sum1+=A[i];
                    if(sum1>=t)
                    {
                        cout<<i+1<<"\n";
                        break;
                    }
                }
            }
        }
    }
}
