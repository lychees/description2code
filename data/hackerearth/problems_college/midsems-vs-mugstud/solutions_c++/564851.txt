#include <iostream>
using namespace std;

int main()
{
    int T;
    int i,j,k,N;
    scanf("%d",&T);
    while(T--)
    	{
    	scanf("%d",&N);
    	int a[51][51][51]={0};
    	int ans[51][51][51];
    	N++;
    	for(i=1;i<N;i++)
    		for(j=1;j<N;j++)
    			for(k=1;k<N;k++)
    				scanf("%d",&a[i][j][k]);
    	for(i=1;i<N;i++)
    		{
    		for(j=1;j<N;j++)
    			{
    			for(k=1;k<N;k++)
    				{
    				ans[i][j][k]=a[i][j][k]-a[i-1][j][k]-a[i][j-1][k]-a[i][j][k-1];
    				ans[i][j][k]+=-a[i-1][j-1][k-1]+a[i-1][j-1][k]+a[i][j-1][k-1]+a[i-1][j][k-1];
    				printf("%d ",ans[i][j][k]);
    				}
    				printf("\n");
    			}
    		}	
    	}
    return 0;
}
