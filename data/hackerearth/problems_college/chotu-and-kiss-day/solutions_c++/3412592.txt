#include<bits/stdc++.h>
using namespace std;
int main()
{
	long i,j,n,k,p;
	scanf("%ld",&n);
	int found,flag;
	long s[n];
	for(i=0;i<n;i++)
		scanf("%ld",&s[i]);
	scanf("%ld",&k);
	sort(s,s+n);
	for(i=0;i<n-1;i++) {
		flag=0;
		for(j=i+1;j<n;j++) {
			p=k-s[i]-s[j];
			if(p==s[i]||p==s[j])
			continue;
			if(binary_search(s,s+n,p)){
				printf("YES\n");
				flag=1;
				break;
			}
		}
		if(flag)
		break;
	}
	if(!flag)
	printf("NO\n");
	return 0;
}
