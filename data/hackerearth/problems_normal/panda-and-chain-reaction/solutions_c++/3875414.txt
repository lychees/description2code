#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000003
#define bitcount __builtin_popcountll
#define ll long long
ll int arr[1000005];
int main()
{
	int i,j,k;
	ll int x,n,ans;
	arr[0]=1;
	for(i=1;i<1000003;i++)
	{
		arr[i]=(arr[i-1]*i)%mod;
	}
	int t;
	sd(t);
	while(t--)
	{
		slld(n);
		slld(x);
		if(n<1000003)
			ans=((x%mod)*arr[n])%mod;
		else
			ans=0;
		printf("%lld\n", ans);
	}
	return 0;
}
