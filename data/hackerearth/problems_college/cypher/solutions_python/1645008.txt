tests = int(raw_input())
for test in range(tests) :
    strings = raw_input().split()
    for string in strings :
        c = 0
        word = []
        for letter in string :
            if ord(letter) + c > 122 :
                word.append(chr(97 +(c - (123 - ord(letter)))))
            else :
                word.append(chr(ord(letter)+c))
            c += 1
        print ''.join(word)
