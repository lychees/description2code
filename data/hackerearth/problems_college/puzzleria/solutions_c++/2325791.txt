#include <iostream>
using namespace std;
long long int fact(int n)
{
  long long int res=1;	
  for(int i=1;i<=n;i++)
  {
  	res=res*i;
  }
  return res;
}
int main()
{
   long long int t,n,res;
   cin>>t;
   while(t--)
   {
   	long long int f1,f2,f3,a,b,res;
   	cin>>a;
   	cin>>b;
   	f1=fact(a+b);
   	f2=fact(7) / ( fact(7-a)*fact(a) );
   	f3=fact(5) / ( fact(5-b)*fact(b) );
   	
   	res=f1*f2*f3;   
   	cout<<res<<endl;
   } 
    return 0;
}
