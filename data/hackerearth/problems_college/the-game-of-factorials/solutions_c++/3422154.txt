#include<iostream>
using namespace std;

void factorial(int n)
{
    int s[10000];
    int i,j,size=1;
    int sum=0,carry=0;
    s[0]=1;

    for(i=1;i<=n;i++)
    {
        carry=0;
        for(j=0;j<size;j++)
        {
            sum = i*s[j] +carry;
            s[j]=  sum%10;
            carry = sum/10;
        }

        while(carry)
        {
            s[size++]=carry%10;
            carry/=10;
        }
    }
    int ans =0;
    for(i=size-1;i>=0;i--)
        ans+=s[i];
    cout<<ans<<endl;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        factorial(n);
    }
    return 0;
}
