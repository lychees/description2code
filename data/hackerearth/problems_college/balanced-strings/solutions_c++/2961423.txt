#include<iostream>
#include<string>
using namespace std;

int main()
{
	string name;
	int arr[26];
	
	int i;
	
    int test;
    cin>>test;
    while (test--)
    {
    	int i;
    	cin>>name;
    	int len=name.size();
    	
    	for (i=0;i<26;i++) arr[i]=0;
    	
    	for (i=0;i<len;i++)
    		arr[name[i]-97]+=1;

    	int flag=0;
    	for (int i = 0; i < 25-1; ++i)
    	{
    		if (arr[i]!=arr[i+1]) {flag=1; break;}
    	}
    	if (flag==0) cout<<"Yes"<<endl;
    	else cout<<"No"<<endl;

    }
}
