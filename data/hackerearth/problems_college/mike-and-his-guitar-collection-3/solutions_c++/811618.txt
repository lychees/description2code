#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,sum,n;
    scanf("%d",&t);
    while(t--) {
    	int c=0;
    	map <int , int > m;
    	scanf("%d",&n);
    	int i,j,a[n];
    	for(i=0;i<n;i++) scanf("%d",&a[i]);
    	for(i=0;i<(1 << n);i++) {
    		sum=0;
    		for(j=0;j<n;j++) {
    			if(i&(1 << j)) {
    				sum+=a[j];
    			}
    		}
    		if(m[sum]==0) {
    			m[sum]++;
    			c++;
    		}
    	}
    	printf("%d\n",c);
    }
    return 0;
}
