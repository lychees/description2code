#include <bits/stdc++.h>
using namespace std;

int main() {
    int head, tail1, tail2 ;
    cin>>head>>tail1>>tail2;
    int n;
    cin>>n;
    
    vector<int> v(n);
    for(int i=0;i<n;i++)
    	cin>>v[i];
    sort(v.begin(),v.end());
    
	vector<int> R;
	R.push_back(min(tail1,tail2));
	
	for(int i=0;i<n/2;i++)
		R.push_back(v[i]);
		
	R.push_back(head);
	
	for(int i=n/2;i<n;i++)
		R.push_back(v[i]);
		
	R.push_back(max(tail1,tail2));
	
	int sum1 = 0, sum2 = 0;
	
	for(int i=1;i<R.size();i++) {
		sum1 += abs(R[i]-R[i-1]);
	}
	
	if(R.size()%2==0) {
		swap(R[n/2],R[(n+1)/2]);
	}
	
	for(int i=1;i<R.size();i++) {
		sum2 += abs(R[i]-R[i-1]);
	}
	
	cout<<min(sum1,sum2)<<endl;
	return 0;
}
