ll maxa(ll a[],ll n)
{
    ll cur_sum=a[0],max_sum=a[0];
    for(ll i=1 ; i<n ; i++)
    {
        cur_sum = max(a[i],cur_sum+a[i]);
        max_sum = max(cur_sum,max_sum);
    }
    return max_sum;
}
