#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int a[n],f=-1;
		for(int i=0;i<n;i++){cin>>a[i];}
		int b=a[0];
		if(b==-1){f=1;}
		for(int i=0;i<n && f==-1;i++){
			if(a[b]==-1){f=1;break;}
			else{b=a[b];}
		}
		if(f==1)
		cout << "Win" << endl;
		else
		cout << "Lose" << endl;
	}
    
    return 0;
}
