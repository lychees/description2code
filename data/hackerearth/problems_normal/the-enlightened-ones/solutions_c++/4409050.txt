#include <bits/stdc++.h>
#include <iostream>
using namespace std;

int check(vector<long long int> arr, int v, int k)
{
	long long int pos;
	pos = arr[0] + v;
	k--;
	for(int i=1; i<arr.size(); i++){
		if(pos + v >= arr[i])
			continue;
		if(k == 0)
			return 0;
		pos = arr[i] + v;
		k--;
	}
	return 1;
}

int main()
{
    int n, k;
    cin >> n >> k;
    long long int x, l, r, mid;
    vector< long long int > arr;
    for(int i=0; i<n; i++){
    	cin >> x;
    	arr.push_back(x);
    }
    sort(arr.begin(), arr.end());
    l = 0; r = 10000000;
    while(l<=r){
    	mid = (l+r)/2;
    	int l1 = check(arr, mid, k);
    	int l2 = check(arr, mid-1, k);
    	if(l1 && !l2)
    		break;
    	if(l1)
    		r = mid-1;
    	else if(!l1)
    		l = mid+1;
    }
    cout << mid << endl;
    return 0;
}
