#include <iostream>
using namespace std;

int ans[10000001],s[10000001],a[10000001];
int smith[10000001];
int n=100000;
inline void readint(int &a)
{
    register int c;
    a = 0;
    do c = getchar_unlocked(); while(c < '0');
    do{
    a = (a << 1) + (a << 3) + c - '0';
    c = getchar_unlocked();
    }while(c >= '0');
}
     
inline void printint(int a)
{
    char s[11];
    int t = -1;
    do{
    s[++t] = a % 10 + '0';
    a /= 10;
    }while(a > 0);
    while(t >= 0)putchar_unlocked(s[t--]);
    putchar_unlocked('\n');
}
int getsum(int a)
{
	int ans=0;
	while (a)
	{
		ans+=a%10;
		a/=10;
	}
	return ans;
}
void init()
{
	for (int i=0;i<n;i++)
	{
		s[i]=getsum(i);		
	}
}
void modified()
{
	int i,j;
	for (i=0;i<n;i++) ans[i]=0;
	for (i=0;i<n;i++) a[i]=1;
	i=2;
	while (i<n)
	{
		if (a[i]==1)
		{
			j=i;
			while (j<n)
			{
				int k=j;
				while (k%i==0)
				{
					ans[j]+=s[i];
					k/=i;
				}
				a[j]=0;
				j+=i;
			}
					
		}
		i+=1;	

	}
}
void finalinit()
{
	int i;
	smith[0]=0;
	for (i=1;i<n;i++)
	{
		smith[i]=smith[i-1];
		if (s[i]==ans[i])
		{
			smith[i]+=1;
		}
	}
}
int main()
{
	init();
	modified();
	finalinit();
	int t,l,r;
	readint(t);
	while (t--)
	{
		readint(l);
		readint(r);
		printint(smith[r]-smith[l-1]);
	}
	return 0;
}
