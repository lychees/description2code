#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        char s[100];
        cin>>s;
        int l;
        l=strlen(s);
        int c=0;
        for(int i=0,j=l-1;i<=j;i++,j--)
        {
            if(s[i]!=s[j])
            {
                c=1;
                break;
            }
        }
        if(c==1)
            cout<<"NO\n";
        else
            cout<<"YES\n";
    }
    return 0;
}
