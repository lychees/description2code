#include <bits/stdc++.h>
using namespace std;

int main() {
	int t,n,a[]={-1,-1,1,1,2,1,2,1,2};
	cin>>t;
	while(t--){
	    cin>>n;
	    int s=0;
	    if(n==1||n==0)
	    {
	       cout<<"-1"<<endl;
	        continue;
	    }
	    else if(n<=8)
	    {
	    	cout<<a[n]<<endl;
	    	continue;
	    }
	    else 
	    {
        s+=n/7;
        n=n%7;
        if(n==1)
        {
        	s+=1;
        }
        else if(n>1)
        {
        	s+=a[n];
        }
        cout<<s<<endl;
	}
	}
	
	return 0;
}
