#include <iostream>
#include <string>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int te,count,flag;
    string s,t;
    int sarr[256],tarr[256];
    cin>>te;
    for(int i=0;i<te;i++)
    {
    	cin>>s;
    	cin>>t;
    	count=0;
    	flag=1;
    	for(int j=0;j<256;j++)
    	{
    		sarr[j] = 0;
    		tarr[j] = 0;
    	}
    	for(int j=0;j<s.size();j++)
    	sarr[s[j]]++;
    	for(int j=0;j<t.size();j++)
    	tarr[t[j]]++;
    	for(int j=0;j<256;j++)
    	{
    		if(sarr[j] > tarr[j])	
    		{
    			cout<<"-1"<<endl;
    			flag=0;
    			break;
    		}
    		else if(tarr[j]!=sarr[j])
    		{
    			count++;
    			
    		}
    	}
    	if(flag)
    	cout<<count<<endl;
    }
    return 0;
}
