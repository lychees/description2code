vowels = ['a', 'i', 'e', 'u', 'o', 'y']

#accepting no of test cases
t = int(raw_input())
strings = []

for i in range(t):
	strings.append(raw_input())

for i in range(t):
	s = list(strings[i])
	count = 0
	consecutive = 0
	for j in s:
		if j in vowels:
			count = count + 1
			consecutive = 0
		else:
			count = count - 1
			consecutive = consecutive + 1
		if consecutive == 3:
			break
	if count >= 0 and consecutive != 3:
		print "easy"
	else :
		print "hard"
