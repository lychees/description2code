#include<bits/stdc++.h>
using namespace std;
#define ll long long

ll a[200010];

int main()
{
    int t,n,i,j;
    ll k;
    cin>>t;
    while(t--)
    {
           ll cnt=0;
           cin>>n>>k;
           for(i=0;i<n;i++)
           {
                   cin>>a[i];
           }

           ll needed=(1ll*(n)*(n+1))/2,notneeded=0;

           i=0;
           while(i<n)
           {
                   if(a[i]>k)
                   {
                           i++;
                           continue;
                   }
                   cnt=0;
                   while(i<n && a[i]<=k)
                   {
                           cnt++;
i++;
                   }
                   notneeded+=(1ll*cnt*(cnt+1))/2;
           }

           cout<<needed-notneeded<<endl;

    }
    return 0;
}
