#include <iostream>
#include<string>
#include<algorithm>
#include<vector>

using namespace std;

string p,s;

int main()
{
    int i,j,k,len;
    cin >> s;
    int n = s.length();
    s +=s;
    for(i=0,j=1;i<n && j<n;){
    	for(k=0;s[i+k]==s[j+k];k++);
    	if(k>=n) break;
    	if(s[i+k]>s[j+k]){
    		j = j+k+1;
    	}
    	else{
    		len = i+k;
    		i = j;
    		j = max(len,j+1);
    	}
    }
    cout << s.substr(i,n) << endl;
}
