#include<iostream>
 #include <stdio.h>
#include <string.h>
#include <stdlib.h>
#define MOD 1000000007
using namespace std;
int p[10000000];
int main()
{
	p[0]=0;p[1]=1;p[2]=1;p[3]=2;
	for(int i=4;i<32;i++)
	{
		p[i]=p[i-1]+p[i-2];
	}
	int t;
	cin>>t;
	while(t--)
	{
	int a,b,cnt=0;
	cin>>a>>b;
	for(int i=0;i<32;i++)
	{
		if(p[i]>=a&&p[i]<=b)
		cnt++;
		else if(p[i]>b)
		break;
	}
	if(cnt%2==0)
	cout<<"VALID"<<endl;
	else
	cout<<"INVALID"<<endl;
	}	

}
