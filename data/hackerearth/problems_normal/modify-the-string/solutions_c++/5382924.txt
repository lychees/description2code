#include <iostream>
#include<string.h>
using namespace std;

int main()
{
   char s[101];
   cin>>s;
   int l=strlen(s);
   for(int i=0;i<l;i++)
   {
   	if((s[i]<=90)&&(s[i]>=65))
   	{
   		s[i]=s[i]-'A';
   		s[i]+='a';
   	}
   	else
   	{
   		s[i]=s[i]-'a';
   		s[i]+='A';
   	}
   }
   cout<<s<<endl;
    return 0;
}
