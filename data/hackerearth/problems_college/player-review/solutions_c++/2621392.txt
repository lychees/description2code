#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
	long long int i,j,n,t,curr,max,ans;
		scanf("%lld",&n);
		long long int num[n+9];
		for(i=0;i<n;i++)
		scanf("%lld",&num[i]);
		curr=num[0];
		max=num[0];
		for(i=1;i<n;i++)
		{	
				if(curr<0)
				curr=num[i];
				else
				curr+=num[i];
				if(curr>max)
				max=curr;
		}
		printf("%lld\n",max);
		
	
	return 0;
}
