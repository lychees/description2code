"""
A single integer T denoting number of cases. T lines follow a string S,consisting of only lowercase latin characters ( 'a' - 'z').
"""


def decipher(ch, step):
    """
    moves ch back step times
    """
    pos = ord(ch)

    if pos + step <= 122:
        pos = pos + step
    else:
        x = ord('z') - ord(ch)
        pos = ord('a') + step - x - 1
    return chr(pos)


T = int(raw_input())

while (T != 0):

    S = raw_input()
    step = 0
    answer = ""
    for letter in S:
        answer += decipher(letter, step)
        step += 1
    print answer
    T -= 1
