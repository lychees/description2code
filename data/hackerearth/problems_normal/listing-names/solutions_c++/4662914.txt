#include <iostream>
#include <string>
#include <map>

using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    
    
    int t;
    string s;
    
    map <string, int> names;
    
    cin >> t;
    
    while (t--)
    {
    	cin >> s;
    	names[s]++;
    }
    
    for (map <string, int>::iterator it = names.begin(); it != names.end(); ++it)
    {
    	cout << it->first << " " << it->second << endl;
    }
    
    return 0;
}
