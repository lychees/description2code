#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define li long int
#define prnt_newline printf("\n")
#define fr0(a,b) for(i=0;i<a;i+=b)
#define fr1(a,b) for(i=1;i<=a;i+=b)
#define set_bits(x) __builtin_popcount(x)
#define sc1ll(x) cin>>x
#define sc2ll(x,y) cin>>x>>y
#define sc3ll(x,y,z) cin>>x>>y>>z
vector<ll> pat_arr(26);
vector<ll> str_arr(26);
//vector<vector<ll> > mat(10001,vector<ll>(10001));
bool check()
{
    for(ll i=0;i<26;i++)
        {
            if(str_arr[i]!=pat_arr[i])
            {
                return false;
            }
        }
        return true;
}
int main()
{
    ll test,i,j,num,maxx,minn,pat_len,len;
    sc1ll(test);
    string pattern,str;
    while(test--)
    {
        bool flag=false;
        cin>>pattern>>str;
        fill(pat_arr.begin(),pat_arr.end(),0);
        fill(str_arr.begin(),str_arr.end(),0);
        pat_len=pattern.length();
        len=str.length();
        for(i=0;i<pat_len;i++)
            pat_arr[pattern[i]-'a']++;
        for(i=0;i<pat_len;i++)
            str_arr[str[i]-'a']++;
        if(check()==true)
            flag=true;
        if(flag==false)
        {
            for(i=pat_len;i<len;i++)
            {
                str_arr[str[i]-'a']++;
                str_arr[str[i-pat_len]-'a']--;
                if(check()==true)
                {
                    flag=true;
                    break;
                }
            }
        }
        if(flag==true)
            printf("YES\n");
        else
            printf("NO\n");
    }
}
