#include <iostream>
#include<algorithm>
#include<string.h>
using namespace std;

int main()
{
    long long int t,i,j,k,l;
    cin>>t;
    while(t--)
    {
        string str;
        cin>>str;
        int x = str.length();
        if(x>=10)
        {
        	cout<<"YES\n";
        }
        else
        {
        	int flag=0;
        	for(i=1;i<str.size();++i)
        	{
        		for(j=i+1;j<str.size();++j)
        		{
        			for(k=j+1;k<str.size();++k)
        			{
        				string str1 = str.substr(0,i);
        				string str2 = str.substr(i,j-i);
        				string str3 = str.substr(j,k-j);
        				string str4 = str.substr(k,str.size()-k);
        				if(str1!=str2 && str1!=str3 && str1!=str4 && str2!=str3 && str2!=str4 && str3!=str4)
        				{
        					flag=1;
        					break;
        				}
        			}
        		}
        	}
        	if(flag==1)
        	cout<<"YES\n";
        	else cout<<"NO\n";
        }
    }
    return 0;
}
