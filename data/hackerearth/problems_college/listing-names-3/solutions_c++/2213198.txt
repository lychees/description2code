#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	string nm[t];
	string t1;
	int i,j;
	cin.ignore();
	for(i=0;i<t;i++)
	{
		getline(cin,nm[i]);
	}
		for(i=0;i<t;i++)
		{
			for(j=0;j<t-1-i;j++)
			{
				if(nm[j]>nm[j+1])
				{
					t1=nm[j];
					nm[j]=nm[j+1];
					nm[j+1]=t1;
				}	
			}
		}
		for(i=0;i<t;i++)
		{
			int flag=1;
			while( i<(t-1))
			{
				if(nm[i]==nm[i+1])
				{
					flag++;
					i++;
					continue;
				}
				else
				break;
			}
			cout<<nm[i]<<" "<<flag<<endl;
}}
