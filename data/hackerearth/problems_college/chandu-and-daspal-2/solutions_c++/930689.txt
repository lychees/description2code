#include <iostream>
using namespace std;

int gcd(int a, int b){
	if(a == b)
		return a;
	else if(a > b)
		return gcd(b, a-b);
	else
		return gcd(a, b-a);
}

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int x, y;
    	string name;
    	cin>>x>>y>>name;
    	if(x==y){
    		cout<<name<<endl;
    		continue;
    	}
    	int temp;
    	int flag = 0;
    	while(x && y){
    		temp = gcd(x, y);
    		x = x - temp;
    		y = y - temp;
    		flag = !flag;
    	}
    	if(flag==1)
    		cout<<name<<endl;
    	else{
    		if(name == "Chandu")
    			cout<<"Daspal"<<endl;
    		else
    			cout<<"Chandu"<<endl;
    	}
    }
    return 0;
}
