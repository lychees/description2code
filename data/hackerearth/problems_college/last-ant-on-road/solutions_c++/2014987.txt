#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,n,x,c1,c2,i;
	scanf("%d",&t);
	while(t--) {
		scanf("%d",&n);
		for(i = 0; i < n; i++) {
			cin >> x;
			if(x == -1) {
				c1++;
			}
		}
		if(c1 >= n-c1) {
			cout << c1 << endl;
		} else {
			cout << c1+1 << endl;
		}
	}
		return 0;
}
