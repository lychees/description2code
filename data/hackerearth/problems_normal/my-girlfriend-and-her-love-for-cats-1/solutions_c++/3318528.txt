#include <iostream>
#include <algorithm>
using namespace std;
 
#define ld long long int
 
 
int main()
{
	int t;
 
	cin >> t;
		int a[t],b[t];
 
		int i,j,r;
		
		for (i = 0; i < t; i++)
		{
			cin >> a[i];
		}
		for (i = 0; i < t; i++)
		{
			cin >> b[i];
		}
		sort(a,a+t);
		sort(b,b+t);
		ld temp = 0;
		
		for (i = 0; i < t; i++)
		{
			
			temp += 1ll * a[i]*b[i];
			
		}
		cout << temp << "\n";
		
	
    
    return 0;
}
