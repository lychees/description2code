n, q = [int(s) for s in raw_input().split()]
arr = sorted([int(s) for s in raw_input().split()])
for _ in xrange(q):
	x = input()
	start = 0
	end = len(arr)-1
	while start <= end:
		mid = start + (end-start)/2
		if arr[mid] == x:
			print 'YES'
			break
		if arr[mid] > x:
			end = mid-1
		else:
			start = mid+1
	else:
		print 'NO'
