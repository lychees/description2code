from itertools import *


def primes(n):
    primfac = []
    d = 2
    while d*d <= n:
        while (n % d) == 0:
            primfac.append(d)  # supposing you want multiple factors repeated
            n //= d
        d += 1
    if n > 1:
        primfac.append(n)
    return primfac


tc = input()
while tc:
    n = input()
    output = primes(n)
# print output
    answer = 0
    for i in range(1, len(output)+1):
        answer = answer + len(list(set(combinations(output, i))))
    print answer
    tc = tc-1
