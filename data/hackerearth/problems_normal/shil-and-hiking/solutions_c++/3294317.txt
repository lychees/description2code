#include<stdio.h>
#include<iostream>
#include<vector>
#include<stack>
#include<queue>
#include<algorithm>
#include<math.h>
#define mod 1000000007//10^9+7
using namespace std;
int main()
{
    long long int h[1003][2],n,p=0,i,j;
	float t,tm;
	scanf("%lld",&n);
	for(i=0;i<n;i++)
	scanf("%lld %lld",&h[i][0],&h[i][1]);
	for(i=0;i<n-1;i++)
	{
		tm=-1*mod;;
		for(j=i+1;j<n;j++)
		{
			t=(float)(h[j][1]-h[i][1])/((float)(h[j][0]-h[i][0]));
			//cout<<t<<" "<<tm<<endl; 
			if(t+0.0001>=tm)
			{
				tm=t;
				p++;
			}
		}
	}
	printf("%lld",p);
    return 0;
}
