#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		int n;
		cin>>n;
		int num[n];
		for(int j=0;j<n;j++)
		{
			cin>>num[j];
		}
		int count=0, k=0;
		while(count<n)
		{
			if(num[k]==-1)
				break;
			k=num[k];
			count++;
		}
		if(count>=n)
			cout<<"Lose"<<"\n";
		else
			cout<<"Win"<<"\n";
	}
    return 0;
}
