int str[31][26];
int freq[26];
int flag;
void solve(int *freq,int ind,int n)
{
	int check=0,cnt=0,temp[26];
	if(ind==n+1||flag==1)
	  return;
	for(int i=0;i<26;i++)
	{
		temp[i]=freq[i]-str[ind][i];
		if(temp[i]==0)
		  cnt++;
	     if(temp[i]<0)
	       check=1;
	}
	if(check==0)
	{
	   if(cnt==26)
	   {
	   	flag=1;
	   	return;
	   }   
	   solve(temp,ind+1,n);	
	}
	solve(freq,ind+1,n);
	  
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
	   int n,i,j,l;
           flag=0;
	   char s[1000];
	   for(i=0;i<31;i++)
	   {
	     for(j=0;j<26;j++)
	     { 
	     	str[i][j]=0;
	        freq[j]=0;
	     }   
	   }  
	   scanf("%d",&n);
	   for(i=1;i<=n;i++)
	   {
	   	   scanf("%s",s);
	   	   l=strlen(s);
	   	   for(j=0;j<l;j++)
	   	     str[i][s[j]-97]++;
	   }
	   scanf("%s",s);
	   l=strlen(s);
	   for(i=0;i<l;i++)
	     freq[s[i]-97]++;
	   solve(freq,1,n);  
	   if(flag==1)
	     printf("YES\n");
	   else
	     printf("NO\n");
	}
	return 0;
}
