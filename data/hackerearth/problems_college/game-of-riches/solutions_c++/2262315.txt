#include<iostream>
using namespace std;
int main()
{
    int T;
    cin>>T;
    for(int i=0;i<T;i++)
    {
        int P,G;
        cin>>P>>G;
        int arr[100];
        //cout<<P<<G;
        for(int i=0;i<P;i++)
        {
            arr[i]=-1000000;
        }
        int total=0;
        cin>>arr[0];
        for(int i=1;i<P;i++)
        {
            cin>>arr[i];
            if(arr[i]>arr[0])
            {
                total++;
            }
            //cout<<total;
        }
        cout<<total/G<<endl;

    }
    return 0;
}
