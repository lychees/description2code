#include<cstdio>
#define MAX 1000000
using namespace std;

long long max(long long a,long long b){
    return a>b?a:b;
}    
long long sum(long long a[],long long n){
    long long max_so_far,max_ending_here;
    max_so_far=max_ending_here=a[0];
    for(long long i=1;i<n;i++){
        max_ending_here=max(a[i],max_ending_here+a[i]);
        max_so_far=max(max_so_far,max_ending_here);
    }
    return max_so_far;    
}    

int main(){
    int t;
    long long n,heal[MAX],i;
    scanf("%d",&t);
    while(t--){
        scanf("%lld",&n);
        for(i=0;i<n;i++){
            scanf("%lld",&heal[i]);
        }    
        printf("%lld\n",sum(heal,n));
    }    
    return 0;
}
