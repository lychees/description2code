#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll dp[1001][1001];
set<ll> cube;

int main()
{
	for(ll i=0;i<=10001;i++) cube.insert(i*i*i);
   	int n,k;
   	scanf("%d%d",&n,&k);
   	for(int i=1;i<=n;i++)
   	{
   		for(int j=1;j<=n;j++)
   		{
   			scanf("%lld",&dp[i][j]);
   			dp[i][j]+=dp[i-1][j]+dp[i][j-1]-dp[i-1][j-1];;
   		}
   	}
   	int c=0;
   	for(int i=k;i<=n;i++)
   	{
   		for(int j=k;j<=n;j++)
   		{
   			ll x=dp[i][j]-dp[i-k][j]-dp[i][j-k]+dp[i-k][j-k];
   			if(cube.find(x)!=cube.end()) c++;
   		}
   	}
   	printf("%d\n",c);
    return 0;
}
