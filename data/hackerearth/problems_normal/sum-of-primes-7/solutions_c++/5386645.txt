#include<bits/stdc++.h>
using namespace std;
long long  primes[1000010];
long long dp[1000010];
void sieve()
{
	long long i,j,k=0;
	for(i=1;i<=1000001;i++)
		primes[i]=1;
	for(i=2;i<=1000001;i++)
	{
		if(primes[i]==1)
		{
			for(j=i;j*i<=1000001;j++)
			{
				primes[j*i]=0;
			}
		}
	}
}
 
int main()
{
	sieve();
	dp[0]=0;
	dp[1]=0;
	long long i;
	for(i=2;i<=1000000;i++)
	{
		if(primes[i]==0)
		{
			dp[i]=dp[i-1];
		}
		else
		{
			dp[i]=i+dp[i-1];
		}
	}
	long long q,left,right;
	cin>>q;
	while(q--)
	{
		long long ans=0;
		cin>>left>>right;
		ans=dp[right]-dp[left-1];
		cout<<ans<<endl;
	}
	return 0;
}
