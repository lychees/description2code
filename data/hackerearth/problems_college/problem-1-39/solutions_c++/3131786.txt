#include <bits/stdc++.h>

using namespace std;

int main()
{
    int tc;
    cin >> tc;
    long int a,b,c,ans;

    while(tc--)
    {
        cin >> a >> b >> c;
        ans = pow(a,b);

        //cout << "ans is : " << ans << endl;
        long int temp = ans / c;
        long int temp1 = (temp * c);
        long int temp2 = (temp+1) * c;
        if (c > ans)
        {
            if (abs(ans) <= abs(c-ans))
                cout << 0 << endl;
            else
                cout << c << endl;
        } else if (ans >= c)
        {
            //cout << ans <<"   " << temp1 <<  "   " << temp2 << endl;
            if (abs(ans - temp1) >= abs(ans - temp2))
                cout << temp2 << endl;
            else
                cout << temp1 << endl;
        }
    }

    return 0;
}
