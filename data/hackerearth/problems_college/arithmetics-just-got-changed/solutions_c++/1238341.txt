#include<iostream>

long long int rev(long long int);
int main()
{
   int t;
   scanf("%d",&t);
   while(t--)
   {
   
   long long int a,b,x,y,z,z1;
  scanf("%lld%lld",&a,&b);
   x=rev(a);
   y=rev(b);
    z=x+y;
   z1=rev(z);
   
  printf("%lld \n",z1);
}}
long long int rev(long long int n)
{
	long long int d,r=0;
	while(n>0)
	{
		d=n%10;
		r=(r*10)+d;
		n=n/10;
	}
	return r;
}
