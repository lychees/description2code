T = input()
names = []
groups = 0

for i in range(T):
    name = str(raw_input())
    names.append(name)

while names:
    temp_name = names.pop(0)
    temp = list()
    temp.append(temp_name)
    matches = [name for name in names if temp_name[0]
               == name[0] and temp_name[-1] == name[-1]]
    for match in matches:
        temp.append(match)
        names.remove(match)

    sort_temp = list()
    for name in temp:
        sort_temp.append(''.join(sorted(name)))
    groups += len(set(sort_temp))

print groups
