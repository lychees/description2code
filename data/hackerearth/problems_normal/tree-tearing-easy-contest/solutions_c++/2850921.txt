#include <bits/stdc++.h>
using namespace std;

int K,N;
vector<int> adj[10013];

pair<int,int> dfs(int x, int pp=0) {
	int cur = 0;
	int sz = 1;
	for (int i: adj[x]) if (i!=pp) {
		auto p = dfs(i,x);
		cur+=p.first;
		sz+=p.second;
	}
	if (sz>=K) cur+=1, sz = 0;
	return make_pair(cur,sz);
}

int main() {
	scanf("%d%d",&K,&N);
	for (int i=2;i<=N;i++) {
		int a;
		scanf("%d",&a);
		adj[a].push_back(i);
		adj[i].push_back(a);
	}
	printf("%d\n",dfs(1).first);
	
    return 0;
}
