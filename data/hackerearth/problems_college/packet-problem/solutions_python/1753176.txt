#Fast Fibonnaci
def fib(n):
    if n <= 2:
        return 1
    k = n/2
    a = fib(k + 1)
    b = fib(k)
    if n % 2 == 1:
        return a*a + b*b
    else:
        return b*(2*a - b)
 
t=input()
while t:
	t-=1
	n=input()
	if n==1:
		print "0 1"
		continue
	print fib(n-1), fib(n)
