#include <iostream>
#include <cstdlib>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,i,j,a[1000001];
    	cin>>n;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	int count=0;
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<n;j++)
    		{
    			if((a[i]==a[j])&&(j>=i))
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
