#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int i,j=-1,n,ans=0;
		int a[500]={0};
		cin>>n;
		for(i=2;i<=n;i++)
		{
			if(n%i == 0)
				j++;
			while(n%i == 0)
			{
				n /= i;
				a[j]++;
			}
		}
		for(i=0;i<=j;i++)
			ans ^= a[i];
		if(ans == 0)
			cout<<"Pappu Bhujia\n";
		else
			cout<<"Printf{}\n";
	}
return 0;
}
