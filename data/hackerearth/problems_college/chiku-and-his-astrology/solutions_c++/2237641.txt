#include <iostream>
#include<stdio.h>
#include<string.h>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
			char ch[1000000];
			scanf("%s",ch);
    int i,zero=0,one=0;
    for(i=0;i<strlen(ch);i++)
    {
    	if(ch[i]=='0')
    	zero++;
    	else if(ch[i]=='1')
    	one++;
    	
    }
    if(one==zero)
    printf("0\n");
    else if(one>zero)
    {
    	printf("%d\n",(strlen(ch)-2*zero));
    }
    else
    printf("%d\n",strlen(ch)-2*one);
	}

    return 0;
}
