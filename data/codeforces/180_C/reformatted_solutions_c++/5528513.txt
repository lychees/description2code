#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <vector>
#define inf 0xfffffff
using namespace std;
int main() {
  string s;
  cin >> s;
  int x = 0, y = 0;
  for (int i = 0; i < s.size(); i++) {
    x = min(x, y) + (s[i] >= 'A' && s[i] <= 'Z');
    y = y + (s[i] >= 'a' && s[i] <= 'z');
  }
  cout << min(x, y) << endl;
  return 0;
}
