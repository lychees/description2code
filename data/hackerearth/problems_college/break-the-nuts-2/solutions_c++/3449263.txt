#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,i,start,end,count = 1;
    cin>>n;
    cin.ignore();
    cin.ignore();
    string str;
    std::getline(std::cin,str);
if(n!=1)
{
    for(i=0;i<str.length();i++)
    {
        if(str[i]=='1')
        {
            start = i;
            break;
        }
    }
    for(i=str.length()-1;i>=0;i--)
    {
        if(str[i]=='1')
        {
            end = i;
            break;
        }
    }
    long long prod = 1;
    count = 1;
    for(i=start+1;i<=end;i++)
    {
        if(str[i]=='0')
        {
            count++;
        }
        if(str[i]=='1')
        {
            if(count!=1)
            {
                prod*=count;
                count = 1;
            }
        }
    }
    cout<<prod<<endl;
}
else
{
    if(str[0]=='0')
    {
        cout<<"0"<<endl;
    }
    if(str[0]=='1')
    {
        cout<<"1"<<endl;
    }
}
    return 0;
}
