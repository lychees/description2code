import sys

T = int(sys.stdin.readline())
while T > 0:
	N = int(sys.stdin.readline())
	nums = map(int, sys.stdin.readline().split())
	S = int(sys.stdin.readline())
	if S == 0:
		print 'YES'
	else:
		def memo(func):
			cache = {}
			def solve_memo(*args):
				if cache.has_key(args) == False:
					cache[args] = func(*args)
				return cache[args]
			return solve_memo
		@memo
		def solve(curr, summation):
			if curr == len(nums):
				return True if summation == 0 else False
			if summation == 0:
				return True
			return solve(curr+1, summation) or solve(curr+1, summation - nums[curr])
		if solve(0, S) == True:
			print 'YES'
		else:
			print 'NO'
	T -= 1
