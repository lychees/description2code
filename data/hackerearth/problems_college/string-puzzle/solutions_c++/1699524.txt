#include <iostream>
using namespace std;
void swap(char *x,char *y)
{
	char temp;
	temp=*x;
	*x=*y;
	*y=temp;
}
void permute(char *a, int i, int n) 
{
   int j; 
   if (i == n)
   {
     cout<<a<<endl;
   }
   else
   {
        for (j = i; j <= n; j++)
       {
          swap((a+i), (a+j));
          permute(a, i+1, n);
          swap((a+i), (a+j));
       }
   }
}  
int main()
{
	int q,l,r;
   char a[1001] ;
   cin>>a>>q;
   while(q--)
   {
   	  cin>>l>>r;
      permute(a, l, r);
      cout<<endl;
   }
   return 0;
}
