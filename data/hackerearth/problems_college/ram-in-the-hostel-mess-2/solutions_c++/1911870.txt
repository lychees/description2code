#include <bits/stdc++.h>
#define ll long long

using namespace std;

ll pos[1000005], neg[1000005];

int main()
{
    int t;

    cin >> t;

    pos[1] = 1;
    neg[1] = 1;
    for (int i = 2; i < 1000005; i++) {
        pos[i] = neg[i-1];
        neg[i] = pos[i-1] + neg[i-1];
    }

    while (t--) {
        int n;

        cin >> n;

        long long t1 = pos[n] + neg[n];

        cout << (t1 * t1) << endl;
    }

    return 0;
}
