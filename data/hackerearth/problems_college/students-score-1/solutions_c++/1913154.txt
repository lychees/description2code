#include <iostream>
#include <cmath>
using namespace std;
 
int main()
{
	int n,a[100001];
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	int max_ending_here=0,max_so_far=0;
	for(int x=0;x<n;x++)
	{
		max_ending_here=max(0,max_ending_here+a[x]);
		max_so_far=max(max_so_far,max_ending_here);
	}
	cout<<max_so_far<<endl;
}
