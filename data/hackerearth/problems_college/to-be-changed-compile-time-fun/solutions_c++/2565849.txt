#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int n,i,j;
    vector<long long int> v;
    vector<long long int>::iterator it,it1;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>j;
        v.push_back(j);
    }
    long long int m,ans=0;
    while(v.size()!=2)
    {
    	m=0;
    	for(it=v.begin()+1;it!=v.end()-1;it++)
    	{
    		if((*(it-1) * *(it+1))>m)
    		{
    			it1=it;
    			m=*(it-1) * *(it+1);
    		}
    	}
    	v.erase(it1);
    	ans+=m;	
	}
    cout<<ans<<"\n";
}
