#include<vector>
#include<unordered_set>
#include<set>
#include<map>
#include<algorithm>
#include<stdlib.h>
#include<iostream>
#include<utility>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		string s;
		cin>>n>>s;
		int score[26]={};
		for(int i=0;i<s.size();i++)
		{
			score[s.at(i)-97]++;
		}
		int maxScore=-1,size=-1,pos=0;
		for(int i=0;i<n;i++)
		{
			int temp=0;
			cin>>s;
			for(int j=0;j<s.size();j++)
			{
				temp+=score[s.at(j)-97];
			}
			if(temp>maxScore)
			{
				pos=i;
				maxScore=temp;
				size=s.size();
			}
			else if(temp==maxScore && s.size()<size)
			{
				size=s.size();
				pos=i;
			}
		}
		printf("%d\n",pos+1);
	}
	return 0;
}
