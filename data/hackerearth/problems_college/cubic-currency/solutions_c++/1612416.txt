#include <bits/stdc++.h>
using namespace std;

long long dp[10001] = {};

void pre(){
	dp[0] = 1;
	for(long long i = 1; i<22; i++){
		long long cur_cub = i*i*i;
		int j;
		for(j = 1; j<10001; j++){
			if(j<cur_cub) continue;
			dp[j] += dp[j-cur_cub];
		}
	}
}

int main()
{
	pre();
    int t,n;
    cin>>t;
    while(t--) cin>>n, cout<<dp[n]<<endl;
    return 0;
}
