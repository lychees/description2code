#include<bits/stdc++.h>
#define MOD 1000003
using namespace std;
int main()
{
	unsigned long long int n,k,a,fact[1000003];
	fact[1]=1;
	for(int i=2;i<1000003;i++)
		fact[i]=((fact[i-1]*i)%MOD)%MOD;
	int t;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		if(n>=1000003){
			cout<<"0\n";
			continue;
		}
		k%=MOD;
		a=((fact[n]*k)%MOD)%MOD;
		cout<<(a%MOD)<<endl;
	}
}
