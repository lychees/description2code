#include <iostream>
#include <limits.h>
using namespace std;

int main(){
    int tc;
    cin >> tc;
    while(tc--){
    	int floors, eggs;
    	cin >> floors >> eggs;
    	int arr[1001][1001];
    	int c=0;
        for(int i=0; i <= floors; i++)
            arr[1][i] = i;
    	
    	for(int e=2; e<= eggs; e++){
    		for(int f=1; f<=floors; f++){
    			arr[e][f] = INT_MAX;
    			for(int k=1;k<=f;k++){
    				c = 1 + max(arr[e-1][k-1],arr[e][f-k]);
    				if(c < arr[e][f])
                        arr[e][f] = c;
    			}
    		}
    	}
    	
    	cout << arr[eggs][floors] << endl;
    }
}
