#include <iostream>
using namespace std;

int main()
{
    int n;
    string s;
    scanf("%d", &n);
    while(n--) {
    	cin >> s;
    	int ans = 0;
    	char ch = '/0';
    	for(int i = 0; i < s.length(); i++) {
    		if(ch != s[i]) {
    			ch = s[i];
    		} else if(ch == s[i]) {
    			ans++;
    		}
    	}
    	cout << ans << endl;
    }
    return 0;
}
