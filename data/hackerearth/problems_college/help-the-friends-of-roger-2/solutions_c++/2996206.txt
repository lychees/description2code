#include<iostream>
using namespace std;
int main()
{
    string s;
    cin>>s;
    int n,i,j,k,l,a,b,m;
    cin>>n;
    while(n--)
    {
        m=0;
        cin>>a>>b;
        for(i=a-1;i<b-1;i++)
        {
            if(s[i]==s[i+1])
            m++;
        }
        cout<<m<<endl;
    }
    return 0;
}
