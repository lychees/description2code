t = input()
A = [1, 2, 5, 10, 20, 50, 100, 500, 1000]
if t >= 1 and t <= 1000:
    for i in range(0, t):
        n = input()
        if n >= 1 and n <= pow(10, 9):
            c = 0
            while (n != 0):
                for j in range(len(A)-1, -1, -1):
                    if A[j] <= n:
                        n = n-A[j]
                        c += 1
                        break
            print(c)
