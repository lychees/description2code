#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll a[26],c[26];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        ll n;
        scanf("%lld",&n);
        for(ll i=0;i<26;i++)
        {
            scanf("%lld",&c[i]);
        }
        ll p=n*12;
        //cout<<"P="<<p<<endl;
        ll freq=(p/26);
        //cout<<"Freq="<<freq<<endl;
        for(ll i=0;i<26;i++)
        {
            a[i]=freq;
        }
        ll x=p%26;
        //cout<<"X="<<x<<endl;
        for(ll i=0;i<x;i++)
        {
            a[i]++;
        }
        ll sum=0;
        for(ll i=0;i<26;i++)
        {
           // cout<<"c[i]="<<c[i]<<" f[i]="<<a[i]<<endl;
            sum+=(c[i]*a[i]+a[i]*(a[i]-1)/2);
        }
        cout<<sum<<"\n";
    }
    return 0;
}
