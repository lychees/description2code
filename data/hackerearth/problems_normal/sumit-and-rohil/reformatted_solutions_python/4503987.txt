n = int(raw_input())
l = []
for i in range(n):
    l.append(raw_input())

for i in range(n):
    s = l[i][1:len(l[i])-1]
    s_list = sorted(s)
    s_sort = "".join(s_list)
    l[i] = l[i].replace(s, s_sort)

l.sort()
l = set(l)
print len(l)
