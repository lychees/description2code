T = input()

def steps(diff):
    r = diff%5
    return diff/5 + r/2 + r%2


while T > 0:
    T -= 1
    input()
    nums = [int(x) for x in raw_input().split(' ')]
    nums.sort()
    accuDiff = 0
    accuSteps = 0
    for i in range(1,len(nums)):
        accuDiff += nums[i] - nums[i-1]
        accuSteps += steps(accuDiff)
    print accuSteps
