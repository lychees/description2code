#include<bits/stdc++.h>
using namespace std;
long int P[1000001];
long int primeFactor(long int n)
{
    long int i,cnt=0;
    while(n%2==0)
    {
        cnt++;
        n/=2;
    }
    for(i=3;i<=sqrt(n);i+=2)
    {
        while(n%i==0)
        {
            cnt++;
            n/=i;
        }
    }
    if(n>2)
        cnt++;
    return cnt;
}
int main()
{
    int T;
    long int A,B,i;
    P[0]=P[1]=0;
    for(i=2;i<=1000000;i++)
    {
        P[i]=primeFactor(i);
    }
    for(i=1;i<=1000000;i++)
    {
        P[i]=P[i]+P[i-1];
    }

    cin>>T;
    while(T--)
    {
        scanf("%ld %ld",&A,&B);
        cout<<P[B]-P[A-1]<<"\n";
    }
}
