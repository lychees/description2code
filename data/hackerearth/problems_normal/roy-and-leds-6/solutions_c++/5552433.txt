#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,r,g,b;
    cin>>t>>r>>g>>b;
    vector<vector<int> > v(3,vector<int>(t,0));
      //vector< vector< float > > item ( 2, std::vector<float> ( 2, 0 ) );

    int count=r,flag=1;
    for(int i=0;i<t;i++)
    {
    	if(flag)
    	{
    		v[0][i]=0;
    		count--;
    		if(count==0)
            {
                count=r;
                flag=0;
            }
    	}
    	else
    	{
    		v[0][i]=1;
    		count--;
    		if(count==0)
            {
                count=r;
                flag=1;
            }
    	}
    }

     count=g,flag=1;
    for(int i=0;i<t;i++)
    {
    	if(flag)
    	{
    		v[1][i]=0;
    		count--;
    		if(count==0)
            {
                count=g;
                flag=0;
            }
    	}
    	else
    	{
    		v[1][i]=1;
    		count--;
    		if(count==0)
            {
                count=g;
                flag=1;
            }
    	}
    }

    count=b,flag=1;
    for(int i=0;i<t;i++)
    {
    	if(flag)
    	{
    		v[2][i]=0;
    		count--;
    		if(count==0)
            {
                count=b;
                flag=0;
            }
    	}
    	else
    	{
    		v[2][i]=1;
    		count--;
    		if(count==0)
            {
                count=b;
                flag=1;
            }
    	}
    }
/*
    cout<<"Vector 0 contains :\n";
    for(int i=0;i<t;i++)
        cout<<v[0][i]<<" ";
    cout<<endl;
*/
    map<int,int> m;
    for(int i=0;i<t;i++)
    {
    	int val = v[0][i]*100+v[1][i]*10+v[2][i];
    	if(m.find(val)==m.end())
    	{
    		m[val]=1;
    	}
    	else
    	{
    		m[val]+=1;
    	}
    }

    cout<<m[100]<<" ";
    cout<<m[10]<<" ";
    cout<<m[1]<<" ";
    cout<<m[110]<<" ";
    cout<<m[11]<<" ";
    cout<<m[101]<<" ";
    cout<<m[111]<<" ";
    cout<<m[0];


    return 0;
}
