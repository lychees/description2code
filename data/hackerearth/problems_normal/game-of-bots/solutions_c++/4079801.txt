#include <bits/stdc++.h>
using namespace std;
int na[1000001],nb[1000001],nc[1000001],nd[1000001];
int main()
{
    int t,i,n,ca,cb;
    string a,b;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	n=a.size();
    	ca=0;
    	cb=0;
    	for(i=0;i<n;i++)
    	{
    		if(i==0)
    		{
    			if(a[i]=='#')
    			{
    				nc[i]=0;
    				na[i]=0;
    				nb[i]=0;
    				nd[i]=0;
    			}
    			else if(a[i]=='A')
    			{
    				nc[i]=1;
    				na[i]=0;
    				nb[i]=0;
    				nd[i]=1;
    				ca++;
    			}
    			else
    			{
    				nc[i]=0;
    				na[i]=1;
    				nb[i]=1;
    				nd[i]=0;
    				cb++;
    			}
    		}
    		else
    		{
    			if(a[i]=='#')
    			{
    				nc[i]=nc[i-1];
    				na[i]=cb;
    				nb[i]=nb[i-1];
    				nd[i]=ca;
    			}
    			else if(a[i]=='A')
    			{
    				nc[i]=nc[i-1]+1;
    				na[i]=cb;
    				nb[i]=nb[i-1];
    				ca++;
    				nd[i]=ca;
    			}
    			else
    			{
    				nc[i]=nc[i-1];
    				na[i]=cb+1;
    				nb[i]=nb[i-1]+1;
    				cb++;
    				nd[i]=ca;
    			}
    		}
    	}
    	int ra=0,rb=0,f=1;
    	for(i=0;i<n;i++)
    	{
    		if(b[i]=='A')
    		{
    			ra++;
    			if(na[i]==rb)
    			continue;
    			f=0;
    		}
    		else if(b[i]=='B')
    		{
    			rb++;
    			if(nd[i]==ra && nb[i]>=rb)
    			continue;
    			f=0;
    		}
    	}
    	if(f==0)
    	cout<<"No"<<endl;
    	else
    	cout<<"Yes"<<endl;
    }
    return 0;
}
