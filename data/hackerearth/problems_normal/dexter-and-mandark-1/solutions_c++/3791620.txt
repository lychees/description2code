#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
long long int dp[1005][1005];
long long n,k;
long long solve(long long i,long long cont1,long long cont2)
{
    if(cont1<(cont2*k))
 return 0;

 if(i>=n)
 {
     if(cont1>cont2)
        return 1;
       
        return 0;

 }
 long long &val=dp[cont1][cont2];
 if(val!=-1)
 return val;
 
 val=0;
 val+=solve(i+1,cont1+1,cont2);
 val=val%mod;
 val+=solve(i+1,cont1,cont2+1);
 val=val%mod;
 return val;

}

int main()
{
  long long t;
  cin>>t;
  while(t--)
  {

       cin>>n>>k;
       memset(dp,-1,sizeof(dp));
   cout<<solve(0,0,0)<<endl;
  }
 }
