#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int count_pair(long int arr[],int low,int up,long int val)
{
    long int i=0;
    while(low<up)
    {
        if(arr[low]*arr[up]>val) up--;
        else
        {
            i+=up-low;
            low=low+1;
        }
    }
    return i;
}
void sol()
{
    long int arr[2001],i,n,k,total=0;
    cin>>n>>k;
    for(i=0;i<n;i++)
         cin>>arr[i];
    sort(arr,arr+n);
    for(i=2;i<n;i++)
    {
        total+=count_pair(arr,0,i-1,k/arr[i]);
    }
    cout<<total<<endl;
}
int main()
{
    sol();
    return 0;
}
