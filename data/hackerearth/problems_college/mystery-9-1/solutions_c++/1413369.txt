#include<bits/stdc++.h>
using namespace std;
char a[10001];
int pal(){
	int l=strlen(a);
	for(int i=0; i<l; i++){
		if(a[i]!=a[l-i-1]) return 0;
	}
	return 1;
}
int main(){
	int t;
	cin>>t;
	while(t--){
		cin>>a;
		if(!pal()) cout<<-1<<endl;
		else{
			int x=a[strlen(a)/2];
			cout<<x<<endl;
		} 
	}
}
