using namespace std;
pair<int, int> a[100007];
int lens[100007];
int main()
{
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);
	int n;
	cin >> n;
	for (int i = 1; i <= n; i++)
	{
		cin >> a[i].second;
		a[i].first = i;
		lens[i] = a[i].second;
	}
	int len = n;
	int pos = 1;
	vector<int>ans;
	for (int i = 1; i <= n; i++)
	{
		int sum = 0;		
		sum = lens[i];
		for (int j = 1; j < sum; j++)
		{
			pos++;
			if (pos > len) pos = 1;
		}
		ans.push_back(a[pos].first);
		for (int j = pos; j <= len; j++)
		{
			a[j] = a[j + 1];
		}
		if (pos == len) pos = 1;
		len--;
	}
	for (int i = 0; i < ans.size();i++)
	{
		printf("%d\n", ans[i]);
	}
}
