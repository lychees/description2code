#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    char c[100000];
    unsigned long long int a,b,n,no;
    cin>>c;
    n=strlen(c);
    cin>>no;
    while(no--)
    {
    	cin>>a>>b;
    	a=a-1;b=b-1;
    	a=(a%n);
    	b=(b%n);
    	if(c[a]==c[b])
    	cout<<"Yes\n";
    	else
    	cout<<"No\n";
    }
    return 0;
}
