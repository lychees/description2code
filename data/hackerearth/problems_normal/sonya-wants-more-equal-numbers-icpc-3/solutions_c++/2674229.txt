int a[105];
vector<int> v;
int main()
{
	int n,i,sum=0,max1=MIN;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
		sum=sum+a[i];
	}
	int c=0;
	int psum,j,k,csum;
	for(psum=1;psum<=sum;psum++)
	{
		c=0;
		j=0;
		k=0;
		csum=a[j];
		while(j<n && k<n)
		{
			if(csum>psum)
			{
				csum=csum-a[j];
				j++;
			}
			else if(csum<psum)
			{
				k++;
				csum=csum+a[k];
			}
			else
			{
				//cout<<"hello";
				c++;
				csum=0;
				k++;
				j=k;
				csum=a[j];
			}
		}
		if(c==max1)
		{
			v.push_back(psum);
		}
		else if(c>max1)
		{
			max1=c;
			v.clear();
			v.push_back(psum);
		}
	}
	cout<<max1<<"\n";
	for(i=0;i<v.size();i++)
	cout<<v[i]<<" ";
	cout<<"\n";
	return 0;
}
