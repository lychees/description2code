#include <iostream>
#include<queue>
using namespace std;

int main()
{
    int t;
    cin>>t;
    priority_queue<long long>pq;
    for(int i=0;i<t;i++)
    {
    	long long n,k,j;
    	cin>>n>>k;
    	for(j=0;j<n;j++)
    	{
    	  long long x;
    	  cin>>x;
    	  long count=0;
    	  while(x)
    	  {
    	  	x=x&(x-1);
    	  	count++;
    	  }
    	  pq.push(count);
    	}
    	long sum=0;
    	for(j=0;j<k;j++)
    	{
    		long x;
    		x=pq.top();
    		sum=sum+x;
    		pq.pop();
    	}
    	while(!pq.empty())
    	pq.pop();
    	cout<<sum<<endl;
    }
    return 0;
}
