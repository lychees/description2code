#include <iostream>
using namespace std;

int main()
{
	int n, m, i, j, index, max, c=0;
    cin>>n>>m;
    int a[n][m], vis[n];
    for(i=0;i<n;i++)
    {
    	vis[i]=0;
    	for(j=0;j<m;j++)
    		cin>>a[i][j];
    }
    for(i=0;i<m;i++)
    {
    	index=0;max=a[0][i];
    	for(j=1;j<n;j++)
    	{
    		if(a[j][i]>max)
    		{
    			max=a[j][i];index=j;
    		}
    		else if(a[j][i]==max)
    		{
    			index=-1;
    		}
    	}
    	if(index!=-1)
    		vis[index]=1;
    }
    for(i=0;i<n;i++)
    	if(vis[i]==1)
    		c++;
    cout<<c<<endl;
    return 0;
}
