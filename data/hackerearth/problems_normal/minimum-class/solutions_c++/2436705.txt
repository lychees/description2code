#include <iostream>
#include<math.h>
using namespace std;

int main()
{
     int t;  
cin>>t;
while(t>0)
  { int d;
    long int N;
    cin>>N;
    long int k;
    k=sqrt(N);
  
    if( N>(k*k) && N<=(k*(k+1)) )
        { d=2*k;
         cout<<d<<endl;
        }
        else if( N==(k*k) )
         { d=2*k-1;
           cout<<d<<endl;
         }  
        else
        { d=2*k+1;
        cout<<d<<endl; }
        t--;
        }
    
    return 0;
}
