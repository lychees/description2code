import sys

def isprime(n):
	for i in xrange(2,n/2+1):
		if n%i == 0:
			return False
	return True
n = int(sys.stdin.readline())

a = map(int, sys.stdin.readline().split())
b = map(int, sys.stdin.readline().split())
c = 0
i,j = 0,0
while i<n-1:
	while i<j+1 and j<n:
		if a[i]>b[j]:
			c+=1
		j+=1
	i+=1

if not isprime(c) :
	print "SIMPLE INVERSION"
else:
	print "MAGIC INVERSION"
