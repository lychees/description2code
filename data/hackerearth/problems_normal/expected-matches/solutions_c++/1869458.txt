#include <stdio.h>
double a[1001][1001];
double dp[1001][1001];
int main()
{
	int t,i,j,n,m;
	scanf ("%d",&t);
	while (t--)
	{
	int n,m;
	scanf ("%d%d",&n,&m);
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			scanf ("%lf",&a[i][j]);
			
		}
	}
	dp[n-1][m-1]=1;
	for(i=m-2;i>=0;i--)
	{
		dp[n-1][i]=1+((a[n-1][i])*(dp[n-1][i+1]));
	}
	for(i=n-2;i>=0;i--)
	{
		dp[i][m-1]=1+((1-a[i][m-1])*(dp[i+1][m-1]));
	}
	for(i=n-2;i>=0;i--)
	{
		for(j=m-2;j>=0;j--)
		{
			dp[i][j]=1+(1-a[i][j])*dp[i+1][j]+(a[i][j]*dp[i][j+1]);
		}
	}
	printf ("%f\n",dp[0][0]);
	}
	
	return 0;
}
