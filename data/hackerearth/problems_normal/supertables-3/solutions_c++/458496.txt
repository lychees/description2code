#include<iostream>
using namespace std;
int main()
{
    long long n,a,a1,b,b1,c,c1,i,j,l,t;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>a>>b>>c;
        if(a<b)
        {
            t=a;
            a=b;
            b=t;
        }
        if(a==b)
            cout<<(a*c);
        else if(a%b==0)
            cout<<(b*c);
        else
        {
            a1=a;
            b1=b;
            while(b1!=0)
            {
                t=a1;
                a1=b1;
                b1=t%b1;
            }
            l=(a*b)/a1;
            t=l/a+l/b-1;
            if(c==t)
                cout<<l;
            else if(c<t)
            {
                for(j=0;j<t;j++)
                {
                    if(((int)(a*j/b)+j)==c)
                    {
                        cout<<(a*j);
                        break;
                    }
                    else if(((int)(b*j/a)+j)==c)
                    {
                        cout<<(b*j);
                        break;
                    }
                }
            }
            else
            {
                c1=c%t;
                for(j=0;j<t;j++)
                {
                    if(((int)(a*j/b)+j)==c1)
                    {
                        cout<<(a*j+(int)(c/t)*l);
                        break;
                    }
                    else if(((int)(b*j/a)+j)==c1)
                    {
                        cout<<(b*j+(int)(c/t)*l);
                        break;
                    }
                }
            }
        }
        cout<<endl;
    }
    return 0;
}
