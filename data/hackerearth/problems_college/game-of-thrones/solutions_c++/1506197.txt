int main()
{
	int T;
	cin>>T;
	long long int N,i,count,x,j;

	while(T--)
	{
		cin>>N;
		count=0;
		vector<long long int> A;
		for(i=0;i<N;i++)
		{
			cin>>x;
			A.push_back(x);
		}
        sort(A.begin(),A.end());
        for(i=0;i<A.size();i++)
        {
        	x=A[i];
        	for(j=i+1;A[j]==x;j++,i++)
        	{
        		count++;
        	}
        }
		cout<<count<<endl;
	}
}
