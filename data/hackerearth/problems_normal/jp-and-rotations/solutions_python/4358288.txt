n,m = map(int,raw_input().split())
initial = map(int,raw_input().split())
target = map(int,raw_input().split())
current_index = 0
target_index = initial.index(target[0])
for i in range(1,m+1):
    direction,shift = raw_input().split()
    shift = int(shift)
    if direction == "R":
        current_index -= shift
        current_index %= n
    else:
        current_index += shift
        current_index %= n
    if current_index == target_index:
        print i
        exit(0)
print -1
