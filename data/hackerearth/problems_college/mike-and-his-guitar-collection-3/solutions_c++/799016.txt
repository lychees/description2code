#include <stdio.h>
bool isSubsetSum(int set[], int n, int sum)
{
    
    bool subset[sum+1][n+1];
 
    for (int i = 0; i <= n; i++)
      subset[0][i] = true;
 
    for (int i = 1; i <= sum; i++)
      subset[i][0] = false;
 
     for (int i = 1; i <= sum; i++)
     {
       for (int j = 1; j <= n; j++)
       {
         subset[i][j] = subset[i][j-1];
         if (i >= set[j-1])
           subset[i][j] = subset[i][j] || subset[i - set[j-1]][j-1];
       }
     }
 
 
     return subset[sum][n];
}
int main()
{
	  int t,n,i;
	  scanf("%d",&t);
	    while(t--)
	    {
            int set[21];
            int sum=0;
            scanf("%d",&n);
              for(i=0;i<n;i++)
                {
                	  scanf("%d",set+i);
                	  sum=sum+set[i];
                }
            int ct=1;
             for(i=1;i<=sum;i++)
             {
     if (isSubsetSum(set, n, i) == true)
         ct++;
             }
              printf("%d\n",ct);
    } 
    return 0;
}
