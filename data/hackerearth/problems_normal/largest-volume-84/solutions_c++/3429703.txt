#include <iostream>
#include <cmath>

// https://www.hackerearth.com/problem/algorithm/moody-numbers/?

constexpr double Pi = 3.1415926535897932384626433832795;

using namespace std;

double vol(double r, double l, double p) {
	const double cp = cos(p);
	const double sp = sin(p);
	return Pi * sp * l * (r * r + l * cp * (r + l * cp / 3.));
}

double dvol(double r, double l, double p) {
	const double cp = cos(p);
	const double c2p = cos(2. * p);
	const double c3p = cos(3. * p);
	return l * l * l * cp / 12. + l * r * r * cp + l * l * r * c2p + l * l * l * c3p / 4.;
}

void run_test() {
	double rd, ls;
	cin >> rd >> ls;
	double l = 0.;
	double r = Pi / 2.;
	while (l + 1e-8 < r) {
		double m = (l + r) / 2.;
		if (dvol(rd, ls, m) > 0.) 
			l = m;
		else
			r = m;
	}
	cout << fixed;
	cout.precision(2);
    cout << vol(rd, ls, l) << '\n';    
}

int main() {
    run_test();
    return 0;
}
