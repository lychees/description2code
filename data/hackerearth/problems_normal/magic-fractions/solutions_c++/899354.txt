#include <math.h>
#include <vector>
#include <iostream>
using namespace std;

int main()
{
    vector<int> sieve(501);
	sieve[0] = -1;
	sieve[1] = -1;
	for(int i=2; i<501; i++)
	{
		if(sieve[i] == 0)
		{
			for(int j=2*i; j<501; j+= i)
				sieve[j] = -1;
		}
	}
	vector<long long int> magicFracNum(501);
	magicFracNum[0] = 0;
	magicFracNum[1] = 0;
	magicFracNum[2] = 1;
	for(int i=3; i<501; i++)
	{
		if(sieve[i] == 0)
			magicFracNum[i] = 2*magicFracNum[i-1];
		else
			magicFracNum[i] = magicFracNum[i-1];
	}
	for(int i=1; i<501; i++)
		magicFracNum[i] = magicFracNum[i] + magicFracNum[i-1];
	int x;
	cin >> x;
	cout<< magicFracNum[x];
}
