ll func(ll x)
{

}

{

ll k ;
sc(t) ;

{
	scanf("%d %d %d %lld",&a,&b,&c,&k) ;

	ll high = 1000000 ; 
	ll mid ; 
	while(1)
	{
		mid = (low+high)/2;
		if(func(mid)>=k)
		{
			high = mid ;
		}
		else
		{
			low = mid+1;
		}
	//	printf("low = %lld hihg = %lld mid = %lld func(mid) = %lld\n",low,high,mid,func(mid));

		break ; 
	}
	mid = (low+high)/2;
	if(func(mid)>=k)
	printf("%lld\n",mid) ; 
	else if(func(mid+1)>=k)
	printf("%lld\n",mid+1) ;
}

}
