#include <iostream>
#include<cmath>
using namespace std;
int isprime(int n)
{
	int i;
	for(i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
		  return 0;
	}
	if(n==1)
	 return 0;
	return 1;
}
int main()
{
    int tc,a,b,i;
    cin>>tc;
    while(tc--)
    {
    	long int sum=0;
    	cin>>a>>b;
    	for(i=a;i<=b;i++)
    	{
    		if(isprime(i))
    		 sum+=i;
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
