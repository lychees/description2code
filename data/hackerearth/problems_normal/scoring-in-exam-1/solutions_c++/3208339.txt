#include<iostream>
#include<cstdlib>
#include<algorithm>

using namespace std;

struct score_t{
    int time;
    int marks;
};
typedef struct score_t score;
bool compare(score a, score b){
    return (a.marks > b.marks);
}

int main(){
    int N, Q;
    cin>>N;
    cin>>Q;
    score q[N];
    for(int i=0;i<N;i++)
        cin>>q[i].time;
    for(int i=0;i<N;i++)
        cin>>q[i].marks;
    int size = sizeof(q)/sizeof(q[0]);
    sort(q, q+size, compare);
    for(int i=0;i<Q;i++){
       // cout<<q[i].time <<"  "<<q[i].marks<<endl;
       int temp, res=0;
       cin>>temp;
       for(int j=0;j<temp; j++)
           res += q[j].time;
       cout<<res<<endl;
    }
    return 0;
}
