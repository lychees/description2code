#include<stdio.h>
#include<math.h>
#include<bits/stdc++.h>
using namespace std;
long long gcd(long long a, long long b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
int main()
{
	vector <long> v;
	long long i, g, j, m, a, b, q, l, r;
	scanf("%lld%lld%lld",&a,&b,&q);
	g=gcd(a,b);
	m=sqrt(g);
	for(i=1;i<m;i++) {
		if(g%i==0) {
			v.push_back(i);
			v.push_back(g/i);
		}
	}
	if(i*i==g)
	v.push_back(i);
	else if(g%i==0){
		v.push_back(i);
		v.push_back(g/i);
	}
	sort(v.begin(),v.end());
	vector <long int>::const_iterator it;
	while(q--) {
		scanf("%lld%lld",&l,&r);
		it=upper_bound(v.begin(),v.end(),r)-1;
		if(*it<l) {
			printf("-1\n");
			continue;
		}
		printf("%ld\n",*it);
	}
	return 0;
}
