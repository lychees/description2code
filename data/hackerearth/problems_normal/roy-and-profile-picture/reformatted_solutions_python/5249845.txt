l = int(raw_input())
n = int(raw_input())
for _ in range(n):
    w, h = map(int, raw_input().split())
    if w < l or h < l:
        print "UPLOAD ANOTHER"
    else:
        if w != h:
            print "CROP IT"
        else:
            print "ACCEPTED"
