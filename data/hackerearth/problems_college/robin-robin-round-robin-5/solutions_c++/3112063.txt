#include <bits/stdc++.h>
using namespace std;
#define ll long long int
struct st
{
	ll value;
	ll time;
	ll index;
};
struct str
{
	ll value;
	ll index;
};
bool cmp(str a, str b){ return a.value < b.value;}
int main()
{
	ll n,i,q,t,in,total=0;
	int c;
	scanf("%lld%lld",&n,&t);
	st s[n];
	str ss[n];
	for(i=0;i<n;i++){
		scanf("%lld",&s[i].value);
		s[i].index = i;
	}
	s[0].time=0;
	for(i=1;i<n;i++)
	{
		total=total+ ((s[i-1].value<=t)?s[i-1].value:t);
		s[i].time=total;
	}
	for(i=0;i<n;i++){
		ss[i].value = (s[i].value-1)/t;
		ss[i].index = s[i].index;
	}
	sort(ss,ss+n,cmp);/*
	for(i=0;i<n;i++)
		cout<<ss[i].index<<" ";
	cout<<endl;
	for(i=0;i<n;i++)
		cout<<ss[i].value<<" ";
		*/
	for(i=0;i<n;i++)
	{
		s[ss[i].index].index =i+1;
	}
	scanf("%lld",&q);
	while(q--)
	{
		scanf("%d%lld",&c,&in);
		if(c==1)
		{
			printf("%lld\n",s[in-1].time);
		}
		else
		{
			printf("%lld\n",s[in-1].index);
		}
	}
	return 0;
}
