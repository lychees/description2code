#include <iostream>
using namespace std;
void initialise(int *h,int n)
{
	while(n--)
		h[n]=0;
}
int check(int from,int to,int *hash)
{
	if(from==to)
		return 0;
	if(hash[from]!=0||hash[to]!=0)
		return 0;
	for(int i=from+1;i<to;i++)
	{
		if((hash[i]!=0)&&(hash[i]>=to||hash[i]<=from))
			return 0;
	}
	return 1;
	
}
int main()
{
    int n,q;
    cin>>n>>q;
    int *hash;
    hash=(int*)malloc(sizeof(int)*(n+1));
    initialise(hash,n+1);
    while(q--)
    {
    	int from,to;
    	cin>>from>>to;
    	int a=from;
    	from=from<to?from:to;
    	to=a>to?a:to;
    	if(check(from,to,hash))
    	{
    		hash[from]=to;
    		hash[to]=from;
    		cout<<"YES"<<endl;
    	}
    	else
    		cout<<"NO\n";
    }
    return 0;
}
