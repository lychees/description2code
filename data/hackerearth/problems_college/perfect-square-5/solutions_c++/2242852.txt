#include <bits/stdc++.h>
using namespace std;
long long int xx[1234567];
int main()
{
	long long int i;
    for(i = 0;i < 1204567;i++)
    	xx[i] = i*i;
    int t;
    long long int kk;
    scanf("%d",&t);
    while(t--) {
    	scanf("%lld",&kk);
    	
    	long long int xx1 = (long long int)sqrt(kk);
    	if(xx1*xx1==kk)
    	printf("YES\n");
    	else if(kk-xx[xx1]>=xx[xx1+1]-kk) {
    		
    		printf("+%lld\n",xx[xx1+1]-kk);
    	}
    	else 
    		printf("-%lld\n",kk-xx[xx1]);
    		
    }
    return 0;
}
