#include <iostream>
using namespace std;

int main()
{
    int n;cin>>n;string s,ss[n];int a[n];int c=0;int p=0;
    for(int i=0;i<n;i++){ss[i]="";a[i]=0;}
    for(int i=0;i<n;i++)
    {cin>>s;p=0;
    for(int j=0;j<c;j++)
    {if(ss[j]==s){a[j]++;p=1;break;}}
    if(p==0){ss[c]=s;a[c]++;c++;}
    }
    
    for(int i=0;i<c-1;i++)
    {for(int j=i+1;j<c;j++)
    {if(a[i]<a[j])
    {swap(a[i],a[j]);swap(ss[i],ss[j]);}
    }
    }
    for(int i=0;i<c;i++)
    cout<<ss[i]<<" "<<a[i]<<endl;
    return 0;
}
