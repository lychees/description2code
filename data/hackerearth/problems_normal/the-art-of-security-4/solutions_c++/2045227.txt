#include <bits/stdc++.h>

using namespace std;

#define ll long long
ll n, a[10];

ll solve (int x, int one, int two) {
	if (x >= n) return 1;
	ll res = 0;
	if (one + a[x] <= two) {
		res += solve (x + 1, one + a[x], two);
	}
	res += solve (x + 1, one, two + a[x]);
	
	return res;
}
