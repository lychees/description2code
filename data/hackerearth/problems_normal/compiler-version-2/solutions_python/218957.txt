import sys
f=sys.stdin
ans=""
for l in f.readlines():
    y=l.find("//")
    s=""
    for i in range(len(l)):
        if i<y or y==-1:
            if l[i]=='-' and l[i+1]=='>':
                s += '.'
            elif l[i]=='>' and l[i-1]=='-':
                pass
            else: s += l[i]
        else: s += l[i]
    ans += s
print ans
