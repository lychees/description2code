#include<bits/stdc++.h>
using namespace std;
#define ll int
ll dp[1005],n,a[1005],b[1005];
ll solve(ll pos)
{
    if(pos >= n)
       return 0;
    if(dp[pos]!=-1)
       return dp[pos];
    ll ans=1;
    for(int i=pos + 1;i<n;i++){
        if(a[i]%a[pos]==0){
            ans = max(ans,solve(i) + 1);
        }
    }
    return dp[pos]=ans;
}
int main()
{
    ll i,c=0,m;
    ios::sync_with_stdio(0);
    cin.tie(0);
    cin>>n;
    memset(dp,-1,sizeof(dp));
    for(i=0;i<n;i++){
        cin>>a[i];
    }
    sort(a,a+n);
    int k = 0;
    for(int i = 0;i < n;++i){
    	k = max(k,solve(i));
    }
    if(k<=1)
    k=-1;
    cout<<k<<"\n";
    return 0;
}
