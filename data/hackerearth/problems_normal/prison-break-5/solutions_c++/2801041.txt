#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int mat[20][20];
int ans=0;
void solve(int x,int y,int n)
{
	//if(mat[n-1][n-1]==1)
	//return ;
	
	if(x < 0 || x >=n || y < 0 || y >=n || (mat[x][y]==1))
	return ;
	if(x==n-1 && y==n-1)
	{
	ans++;
	return ;
	}
	mat[x][y]=1;
	if(x+1 < n )
	{
	solve(x+1,y,n);
	}
	if(x-1>=0)
	{
	solve(x-1,y,n);
	}
	if(y+1 < n)
	solve(x,y+1,n);
	if(y-1 >=0)
	solve(x,y-1,n);
	mat[x][y]=0;
	return;
}
int main()
{
	ios_base ::sync_with_stdio(false);
	int t,n;
	cin >> t;
	while(t--)
	{
		ans=0;
		//memset(mat,0,sizeof(mat));
		cin >> n;
		for(int i=0; i < n; i++)
		{
			for(int j=0; j < n; j++)
			{
				cin >> mat[i][j];
			}
		}
		if(mat[0][0]==1 && mat[n-1][n-1])
		{
			cout <<"0\n";
			continue;
		}
		if(mat[0][0]==1 && mat[n-1][n-1]!=1)
		mat[0][0]=0;
		solve(0,0,n);
		cout << ans <<"\n";
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
