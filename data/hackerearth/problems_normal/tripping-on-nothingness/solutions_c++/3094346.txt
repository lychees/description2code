int fr()
{
    char ch;
    int t=0,flag=0;
    ch=gc();
    while(ch<33)
        ch=gc();
    if(ch=='-')
        {
            ch=gc();
            flag=1;
        }
    while(ch>33)
    {
        t=t*10+ch-'0';
        ch=gc();
    }
    if(flag)
        return -t;
    else
        return t;
}
int gcd(int a,int b)
{
    if(b==0)
        return a;
    return gcd(b,a%b);
}
