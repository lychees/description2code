#include <bits/stdc++.h>

using namespace std;
vector <string> v;
int main()
{
    int t,i,c=1;string name,prev;
    cin>>t;
    while (t--)
    {
        cin>>name;
        v.push_back(name);
    }
    sort(v.begin(),v.end());
    prev=v[0];
    for (i=1;i<v.size();i++)
    {
        if (v[i]!=prev)
        {
            cout<<prev<<" "<<c<<endl;
            prev=v[i];
            c=1;
        }
        else
            c+=1;
    }
    cout<<prev<<" "<<c<<endl;
    return 0;
}
