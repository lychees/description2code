#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char a[101];
    	cin>>a;
    	int l=strlen(a),f=0;
    	for(int i=0;i<l/2;i++)
    	{
    		if(a[i]!=a[l-1-i])
    		{
    			f=1;
    			break;
    		}
    	}
    	if(f==1)
    	cout<<"-1"<<endl;
    	else
    	{
    		int p=a[l/2];
    		cout<<p<<endl;
    	}
    }
    return 0;
}
