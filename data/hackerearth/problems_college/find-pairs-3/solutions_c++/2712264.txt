#include<iostream>
using namespace std;
long long t,n,a[1000005];
int main()
{
cin>>t;
while(t--)
{
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    long long c=0;
    for(int i=0;i<n;i++)
    {
        for(int j=i+1;j<n;j++)
        {
            if(a[i]==a[j])
            {
                c++;
            }
        }
    }
    cout<<c+n<<endl;
}
return 0;
}
