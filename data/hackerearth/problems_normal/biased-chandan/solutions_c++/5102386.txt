#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	cin>>n;
	list<int> s1;
	vector<int> vec(n);
	int sum=0;
	for(int i=0;i<n;i++)
		cin>>vec[i];

	for(int i=0;i<n;i++)
	{
		if(vec[i]!=0)
			s1.push_back(vec[i]);
		else
			s1.pop_back();
	}

	for(auto it=s1.begin();it!=s1.end();++it)
	{
		sum+=*it;
	}

	cout<<sum<<endl;
}
