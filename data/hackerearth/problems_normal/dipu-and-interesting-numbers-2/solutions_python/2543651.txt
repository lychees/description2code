# -*- coding: utf-8 -*-

def calc(x):
    ans = 0    
    L = 1
    R = 10**9
    while(L<=R):
        M = (L+R)>>1
        if(M*M<=x):
            ans = M
            L = M + 1
        else:
            R = M - 1
    return ans

t = int(input())

for i in range(t):
    l,r = map(int,raw_input().split())
    fs = calc(l)
    if(fs*fs!=l):
        fs+=1
    sc = calc(r)
    print sc-fs+1
