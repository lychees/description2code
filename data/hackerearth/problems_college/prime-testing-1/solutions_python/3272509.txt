def isPrime(n):
    if n in [2,3]:
        return True
    if n < 2 or n % 2 ==0 or n % 3==0:
        return False
    i, w = 5, 2
    while i * i <= n:
        if n % i == 0:
            return False
        i, w = i + w,  6 - w
    return True


for _ in xrange(input()):
    toi = input()
    cnt = 0
    while toi:
        toi -= 1
        if isPrime(int(raw_input())):
            cnt +=1
    print cnt
