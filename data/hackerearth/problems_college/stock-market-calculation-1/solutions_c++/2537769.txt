#include <iostream>
using namespace std;

int main()
{
	long long int t,n,a,i,nn;
    scanf("%lld",&t);
    while(t--) {
    	scanf("%lld",&n);
    	scanf("%lld",&a);
    	n-=1;
    	long long sum=a,ans=0;
    	long long int cnt=0;
    	for(i=1;i<=n;i++) {
    		scanf("%lld",&nn);
    		if(a<nn) {
    			cnt++;
    			sum+=nn;
    			a=nn;
    			//printf("pehla  ");
    		}
    		else {
    			sum-=a;
    			//cnt+=1;
    			ans+=((cnt*a)-sum);
    			sum=nn;
    			a=nn;
    			cnt=0;
    			//printf("doosra  ");
    		}
    		//printf("i=%d a=%d nn=%d sum=%d ans=%d cnt=%d\n",i,a,nn,sum,ans,cnt);
    	}
    	sum-=nn;
    	ans+=((cnt*nn)-sum);
    	printf("%lld\n",ans);
    }
    return 0;
}
