#include <iostream>
#include<string>

using namespace std;

int main()
{
	int n, num;
	int days[32];
	cin>>n;
	string* a=new string[n+1];
	for(int i=0; i<=n; i++)
	{
		getline(cin, a[i]);
	}
	
	for(int i=1; i<=31; i++)
	{
		days[i]=0;
	}
	
	for(int i=0; i<=n; i++)
	{
		for(int j=0; j<a[i].length(); j++)
		{
			string str="";
			if(a[i][j]>='1' && a[i][j]<='9')
			{
				while(a[i][j]!=' ' && j<a[i].length())
				{
					str=str+a[i][j];
					j++;
				}
				num=stoi(str);
				if(a[i][0]=='G')
				{
					days[num]=days[num]+2;
				}
				else if(a[i][0]=='B')
				{
					days[num]=days[num]+1;
				}
			}
		}
	}
	
	int max=days[1];
	int f=0;
	
	for(int i=2; i<=31; i++)
	{
		if(days[i]!=0)
		{
			f=2;
			if(days[i]==max)
			{
				f=1;
				break;
			}
			if(days[i]>max)
			{
				max=days[i];
			}
		}
	}
	
	if((max>days[19] && max>days[20]) || f==1 || f==0)
	{
		cout<<"No Date";
	}
	else
	{
		cout<<"Date";
	}
	
	return 0;
}
