#include<iostream>
#include<algorithm>
#include<string.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s;
        cin>>s;
        long long len=0,i,x=0,y=0,z=0;
        len=s.length();
        x=len/2;
        for(i=0;i<x;i++)
        {
            y+=(s[i]-'0');
            z+=(s[x+i]-'0');
        }
        if(y==z){cout<<"YES";}
        else cout<<"NO";
        cout<<endl;

    }
    return 0;

}
