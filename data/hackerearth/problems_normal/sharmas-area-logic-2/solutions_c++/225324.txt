#include<bits/stdc++.h>
using namespace std;

int main()
{
	int test=0;
	cin>>test;
	int n[1005][2]= {0};
	while(test-- )
	{
		int num=0;
		long int minx =90,miny=180,maxx=-90,maxy=-180;
	
		cin>>num;
		for(int i=0;i<num;i++ )
		{
			cin>>n[i][0] >> n[i][1];
			//cout<< n[i][0] << " " << n[i][1] <<endl; 
		}
		
		for(int i=0 ; i<num ; i++ )
		{
			//cout<< "minx " << minx <<endl;
			if( minx>n[i][0] )
			{
				minx=n[i][0];
				//cout<< "minx " << minx <<endl;
			}
			//cout<< "maxx " << maxx <<endl;
			 if( maxx < n[i][0] )
			{
				maxx=n[i][0];
		//		cout<< "maxx " << maxx <<endl;
			}
		//	cout<< "miny " << miny <<endl;
			if( miny > n[i][1] )
			{
				miny=n[i][1];
		//		cout<< "miny " << miny <<endl;
			}
		//	cout<< "maxy " << maxy <<endl;
			if( maxy < n[i][1] )
			{
				maxy=n[i][1];
		//		cout<< "maxy " << maxy <<endl;
			}
		}
		
		long int ans= (maxx-minx) * (maxy-miny);
		if( ans ==0 )
		{
			if(maxx-minx==0 )
			{
				cout<< maxy-miny <<endl; 
			}
			else
			{
				cout<< maxx-minx <<endl;
			}
		}
		else
		{
			cout<< ans  <<endl;
		}
		
	}
}
