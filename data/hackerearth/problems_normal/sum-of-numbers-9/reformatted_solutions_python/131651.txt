from itertools import combinations

for i in xrange(input()):
    num = input()
    numlist = map(int, raw_input().split())
    sumlist = input()
    flag = False
    for j in xrange(num):
        for c in combinations(numlist, j+1):
            if sum(c) == sumlist:
                flag = True
                break
    if flag:
        print "YES"
    else:
        print "NO"
