#include <iostream>
#include <vector>

using namespace std;

int main()
{
    int test;
    cin>>test;
    
    while(test--){
    	int n,m;
    	cin>>n>>m;
    	
    	vector<int> lannisters;
    	lannisters.push_back(1);
    	for(int i=1;i<=n;i++){
    		int x;
    		cin>>x;
    		
    		lannisters.push_back(x);
    	}
    	lannisters.push_back(1);
    	vector<int> results;
    	results.assign(n+2,-1);
    	vector<int> prev;
    	
    	for(int i=0;i<m;i++){
    		if(i>0 && prev==lannisters){
    			break;	
    		}
    		for(int j=1;j<=n;j++){
    			
    			results[j]=lannisters[j-1] & lannisters[j+1];
    		}
    		prev=lannisters;
    		lannisters=results;
    	}
    	
    	for(int i=1;i<=n;i++){
    		cout<<lannisters[i]<<" ";
    	}
    	cout<<endl;
    	
    }
    return 0;
}
