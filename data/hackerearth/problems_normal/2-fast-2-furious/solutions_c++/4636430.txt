#include <iostream>
using namespace std;
#define ll long long

ll mod(ll x)
{	if(x>0)
		return x;
	return -1*x;
}

int main()
{	ll n,i,p=0,m=0,a,b;
    cin>>n>>a;
    for(i=1;i<n;++i)
    {	b=a;
    	cin>>a;
    	if(mod(b-a)>m)
    		m=mod(b-a);
    }
    cin>>a;
    for(i=1;i<n;++i)
    {	b=a;
    	cin>>a;
    	if(mod(b-a)>p)
    		p=mod(b-a);
    }
    if(m>p)
    	cout<<"Dom\n"<<m<<"\n";
    else if(p>m)
    	cout<<"Brian\n"<<p<<"\n";
    else
    	cout<<"Tie\n";
    return 0;
}
