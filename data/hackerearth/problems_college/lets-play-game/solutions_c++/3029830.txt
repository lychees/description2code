#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int test,n,k,a[100009],b[100009],i;
	scanf("%lld",&test);
	while(test--)
	{
		scanf("%lld %lld",&n,&k);
		for(i=0;i<n;i++)scanf("%lld",&a[i]);
		for(i=0;i<n;i++)scanf("%lld",&b[i]);
		long long int xx=0;
		long long int temp;
		for(i=0;i<n;i++)
		{
			temp=k/a[i];
			temp=temp*b[i];
			if(temp>xx)
				xx=temp;
		}
		cout<<xx<<endl;
	}
	return 0;
}
