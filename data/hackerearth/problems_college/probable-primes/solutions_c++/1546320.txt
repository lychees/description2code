#include<iostream>
#include<stdio.h>
#define size 1000002
using namespace std;

long a[size],b[size];
void sieve()
{
    long i,j,count=0;
    for(i=0;i<size;i++)
        a[i]=b[i]=0;

    for(i=2;i<size;i++)
    {
        if(a[i]==0)
        {
        for(j=i*2;j<size;j+=i)
        {
            if(a[j]==0)
                a[j]=i;
        }
        }
    }
    for(i=2;i<size;i++)
    {
        if(a[i]==0)
            b[i]=++count;
        else
            b[i]=count;
    }
}

int main()
{
    sieve();
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long i,l,r;
        float l1,l2,l3;
        scanf("%ld%ld",&l,&r);
        l1=b[r]-b[l-1];

        l2=r-l+1;

        l3=l1/l2;

        printf("%f\n",l3);

    }
}
