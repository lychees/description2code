def complete_str(string):
    letters = set("abcdefghijklmnopqrstuvwxyz")
    string = set(string)
    if string == letters:
        print "YES"
    else:
        print "NO"


tests = input()
for i in range(tests):
    func_input = raw_input()
    complete_str(func_input)
