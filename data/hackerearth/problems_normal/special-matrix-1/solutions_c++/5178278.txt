#include <iostream>
#include<cmath>
using namespace std;
char a;
int main()
{
    int t,n,h,x,y,X,Y,sum;
    cin>>t;
    while(t)
    {cin>>n;
    for(int i=0;i<n;i++)
      for(int j=0;j<n;j++)
       { cin>>a;
          if(a=='*')
          {x=i;
           y=j;
          }
       }
       h= n/2;
       
       X=h-(x);
       Y=h-(y);
       cout<<abs(X)+abs(Y)<<endl;
       t--;
    }
        
    return 0;
}
