#include <iostream>
#include<string.h>
#include<vector>
using namespace std;
#define MAX 1000005
int main()
{
	int t;
	cin>>t;
	while(t--){
	vector<unsigned long long int>v;
	bool arr[MAX];
	v.push_back(1);
	memset(arr,false,(MAX)*sizeof(arr[0]));
	arr[1]=true;
	unsigned long long int x;
	cin>>x;
    for(unsigned long long int i=2;i<x;i++){
    	for(unsigned long long int j=i*i;j<x;j=j*i){
    	v.push_back(j);
    	arr[j]=true;}
    }
    bool val=false;
    for(unsigned long long int i=0;i<v.size();i++){
    	if(arr[x-v[i]]==true){
    		val=true;
    		break;
    	}
    }
    if(val)cout<<"Yes\n";
    else cout<<"No\n";
	}
    return 0;
}
