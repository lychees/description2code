#include<iostream>
#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll fib[1000007];
ll sum[1000007];
void sol()
{
	fib[0]=1;
	fib[1]=1;
	sum[0]=0;
	sum[1]=0;
	for(ll i=2;i<=1000005;i++){
		fib[i]=fib[i-1]+fib[i-2];
		fib[i]=fib[i]%1000000007;
		sum[i]=sum[i-1]+fib[i];
		sum[i]=sum[i]%1000000007;
	}
}
int main()
{ sol();
	ll t,n;
	cin>>t;
	while(t--){
		cin>>n;
		
	 cout<<sum[n+1]<<endl;
	}
	
	return 0;
}
