#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll dp[1000][1000];
ll arr[1000][1000];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
                ll m,n,i,j;

        cin>>m>>n;
        for(i=1;i<=m;i++)
        {
            for(j=1;j<=n;j++)
            {
                cin>>arr[i][j];
            }
        }
                 ll ans=1;

        dp[1][1]=1;
        for(i=2;i<=n;i++)
        {
            if(arr[1][i]>arr[1][i-1]&&dp[1][i-1]!=0)
            {
                dp[1][i]=1+dp[1][i-1];
            }
            else
                dp[1][i]=0;
                ans=max(ans,dp[1][i]);
        }
        for(i=2;i<=m;i++)
        {
            if(arr[i][1]>arr[i-1][1]&&dp[i-1][1]!=0)
            {
                dp[i][1]=1+dp[i-1][1];
            }
            else
                dp[i][1]=0;
                ans=max(ans,dp[i][1]);
        }
        for(i=2;i<=m;i++)
        {
            for(j=2;j<=n;j++)
            {
               int a=0;
                if(arr[i][j]>arr[i-1][j]&&dp[i-1][j]!=0)
                 a=1+dp[i-1][j];
                 int b=0;
                if(arr[i][j]>arr[i][j-1]&&dp[i][j-1]!=0)
                 b=1+dp[i][j-1];
                 dp[i][j]=max(a,b);
                 ans=max(ans,dp[i][j]);

            }
        }
        cout<<ans<<endl;
    }
}
