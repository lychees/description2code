#include <iostream>
using namespace std;

int main() {
	long long int m,n;
	int t;
	cin >> t;
	while(t--){
		cin >> m >> n;
		//m-=1;
		if(n<m/2){
			cout << 2*n+1 << endl;
		}
		else
		{
			m-=n;
			m--;
			cout << m*2 << endl;
		}
	}
	return 0;
}
