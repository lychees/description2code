#include<iostream>
#include<stdio.h>
#include<math.h>
#include<algorithm>
using namespace std;

long long int min(long long int a,long long int b)
{
	if(a<b)
	   return a;
	else
	   return b;   
}

int main()
{
	int t;
	long long int b,w,x,y,z,cost1,cost2,ans;
	scanf("%d ",&t);
	while(t--)
	{
		scanf("%lld %lld",&b,&w);
		scanf("%lld %lld %lld",&x,&y,&z);
		cost1 = min(x,y+z);
		cost2 = min(y,x+z);
		ans = cost1*b + cost2*w;
		printf("%lld\n",ans);
	}
	return 0;
}
