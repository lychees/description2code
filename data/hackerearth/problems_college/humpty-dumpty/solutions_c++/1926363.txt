using namespace std;
typedef long long LL;
typedef vector<int> vi;
typedef vector<pair<int,int> >vpi;
typedef vector <LL> vll;
typedef vector <pair<LL,LL> > vpll;




int main()
{
    ios_base::sync_with_stdio(false);cin.tie(0);
    LL cx,cy,i,n,t,x[50];
    cin>>t;
    while(t--)
    {
        LL sum=0;
        LL suml = 0;
        cin>>n;
        cin>>cx>>cy;
        LL dist2 = cx*cx + cy*cy;
        for(i=0;i<n;i++) {
            cin>>x[i], sum += x[i];
        }
        if(n==1)
        {
            if(x[0]*x[0]==dist2)
                cout<<"Able\n";
            else
                cout<<"Not able\n";
             continue;
        }
        sort(x,x+n);
        for(i=0;i<n-1;i++)
            suml += x[i];
        LL mindiff = x[n-1] - suml;
        sum = sum*sum;
        if(mindiff<0)
            mindiff = 0;
        mindiff *= mindiff;
        if(sum>=dist2 && mindiff<=dist2)
            cout<<"Able\n";
        else
            cout<<"Not able\n";
    }
    return 0;
}
