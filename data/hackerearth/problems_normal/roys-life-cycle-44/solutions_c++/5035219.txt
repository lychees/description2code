#include<iostream>

using namespace std;

int main()
{

    int num;
    cin>>num;
    int count =0,max =0,linecount=0,maxline=0;

    char code[num][1440];

    for(int i = 0;i<num;i++)
    {

        linecount =0;

        for(int j =0;j<1440;j++)
        {
            cin>>code[i][j];

            if(code[i][j] == 'C')
            {
                count++;
                linecount++;
                if(count > max)
                {
                    max = count;
                }

                if(linecount>maxline)
                {
                    maxline = linecount;
                }
            }
            else
            {

                count =0;
                linecount = 0;
            }
        }
    }

    cout<<maxline<< " " <<max<<endl;







}
