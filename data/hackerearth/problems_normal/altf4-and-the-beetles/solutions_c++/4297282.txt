#include<bits/stdc++.h>
using namespace std;
int comp(pair<long long,int> a,pair<long long,int> b)
{
    if(a.first==b.first)
        return a.second<b.second;
    return a.first<b.first;
}
void sol()
{
   vector<pair<long long,int> > v;
   long long n,m,s,i,ans=0,a,b,c;
   cin>>n>>a>>b>>c;
   for(i=0;i<n;++i)
   {
       cin>>m>>s;
       v.push_back(make_pair(m,1));
       v.push_back(make_pair(s,2));
   }
   sort(v.begin(),v.end(),comp);
   long long cnt=n*a;
   ans=cnt;
   for(i=0;i<2*n;++i)
   {
       if(v[i].second==1) cnt+=(b-a);
       else cnt+=(c-b);
       ans=max(ans,cnt);
   }
   cout<<ans<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--) sol();
    return 0;
}
