long long largestSquare(long long n){
	long long x = n;
  	while( true ){
	    long long y = (x + n/x)/2;
	    if( y >= x ) return x;
	    x = y;
  	}
}

int main()
{
	int t;
	scanf("%d",&t);
	for(int i=1; i<=t; i++){
		long long n;
		scanf("%lld",&n);
		printf("%lld\n",n-largestSquare(n));
	}
	return 0;
}
