#include <bits/stdc++.h>
using namespace std;

typedef long long int ll;
const ll mod=1e9+7;

#define fileinput freopen("input.txt","r",stdin)
#define fileoutput freopen("output.txt","w",stdout)
#define T ll t ; cin>>t ; while(t--)

 ll a[2502][2502],ans[2502][2502];


int main()
{
    T{
        ll n,c,i,j;
		
		scanf("%lld %lld",&n,&c);
		for(i=0;i<=n;i++){
			for(j=0;j<=n;j++){
				a[i][j]=0;
				ans[i][j]=0;
			}
		}
		
        for(i=1;i<=n;i++)
            for(j=1;j<=n;j++)
                scanf("%lld",&a[i][j]);
        for(i=1;i<=n;i++){
            for(j=1;j<=n;j++){
                ans[i][j]=a[i][j] + max(ans[i][j-1],ans[i-1][j]);
            }
        }

        ll lft=ans[n][n]-c;
        if(lft>=0){
            printf("%lld\n",lft);
        }
        else
            printf("-1\n");


    }

    return 0;
}
