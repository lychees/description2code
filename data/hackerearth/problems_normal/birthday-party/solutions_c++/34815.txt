#include <iostream>
#include <algorithm>
#include <map>
#include <vector>
using namespace std;

int a[1005];

bool compute(int n,int sum){
	if (sum == 0)
     return true;
   if (n == 0 && sum != 0)
     return false;

   if (a[n-1] > sum)
     return compute (n-1, sum);

   return compute(n-1, sum) || compute (n-1, sum-a[n-1]);
}
int main()
{
	int t;
	cin>>t;
	while(t--){
		map<int,int>mp;
		map<int,int>::iterator it;
		int p,m,k;
		cin>>p>>m;
		int sm=0;
		for(int i=0;i<m;i++) cin>>a[i];
		for(int i=0;i<p;i++){
			cin>>k;
			mp[k]++;
		}
		int ct=0;
		for(it=mp.begin();it!=mp.end();it++){
			if(compute(m,(*it).second)) ct++;
		}
		cout<<ct<<"\n";
	}
    return 0;
}
