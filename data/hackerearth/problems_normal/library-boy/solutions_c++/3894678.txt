#include <iostream>
#include <cstring>

#define ASCII_DIFF 97
#define EACH_T while(t--)
#define FOR(i, a, b) for(i = a; i <= b; i++)
using namespace std;

int main() {
  int t, i, shelves = 0;
  int c[26] = {};
  string s;
  cin >> t;
  EACH_T { cin >> s; c[s[0] - ASCII_DIFF]++; };
  FOR(i, 0, 26) shelves += (int) (c[i]/ 10) + 1 - (c[i] % 10 == 0 ? 1: 0);
  cout << shelves << endl;
  return 0;
}
