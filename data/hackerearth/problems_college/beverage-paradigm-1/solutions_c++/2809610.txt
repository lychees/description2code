#include <iostream>
using namespace std;
 
int main()
{
    long x,a[30000],n;
    cin>>n;
    for(int i=0;i<=n-1;i++)
    cin>>a[i];
    cin>>x;
    for(int i=0;i<=n-1;i++)
    {
        for(int j=i+1;j<=n-1;j++)
        {
        	for(int k=j+1;k<=n-1;k++)
        	{
        		if(a[i]+a[j]+a[k]==x)
        	    {cout<<"True";
        	     return 0;}
        	}
        }
    }
    cout<<"False";
    return 0;
}
