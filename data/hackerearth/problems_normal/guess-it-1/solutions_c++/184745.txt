#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	int t;
	cin>>t;
	char c;
	cin.get(c);
	while(t--)
	{
		char str[1000];
		cin>>str;
		int ans = 0;
		for(int i=0;i<strlen(str);i++)
		{
			if(str[i]=='_')
			continue;
			else
			ans += str[i]-'0';
		}
		if(ans%9==0)
		cout<<"0 or 9"<<endl;
		else
		cout<<9-ans%9<<endl;
	}
    return 0;
}
