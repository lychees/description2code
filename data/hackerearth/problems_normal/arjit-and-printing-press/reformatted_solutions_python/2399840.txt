t = int(raw_input())
for i in xrange(t):
    a = raw_input()
    a = list(a)
    b = raw_input()
    b = sorted(b)
    q = 0
    for j in xrange(len(a)):
        if q < len(b):
            if a[j] > b[q]:
                a[j] = b[q]
                q += 1
        else:
            break
    print ''.join(str(a[k]) for k in xrange(len(a)))
