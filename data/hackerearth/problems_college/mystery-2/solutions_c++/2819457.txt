#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		int a,b,i=0;
		cin>>a>>b;
		int arr[100]={0};
		while(a>0){
			int d=a%b;
			arr[i]=d;i++;
			a=a/b;
		}
		for(i=i-1;i>=0;i--)
		cout<<arr[i];
		cout<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
