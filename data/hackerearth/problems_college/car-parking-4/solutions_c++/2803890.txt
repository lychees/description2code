#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string str;
    	cin>>str;
    	int ans=0;
    	int len=str.length();
    	for(int i=0;i<len;i++){
    		int flag=1;
    		if(str[i]=='B'||str[i]=='D'||str[i]=='S'){
    			//cout<<i;
    			flag=0;
    		}
    		if(i+1<len&&flag==1&&(str[i+1]=='S'||str[i+1]=='B')){
    		//	cout<<i;
    			flag=0;
    		}
			if(i+2<len&&flag==1&&str[i+2]=='B'){
			//	cout<<i;
    			flag=0;
    		}
    		if(i-1>=0&&flag==1&&str[i-1]=='S'){
    		//	cout<<i;
    			flag=0;
    		}
    		ans+=flag;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
