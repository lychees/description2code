cases = input()
cipher = [
    'a',
    'b',
    'c',
    'd',
    'e',
    'f',
    'g',
    'h',
    'i',
    'j',
    'k',
    'l',
    'm',
    'n',
    'o',
    'p',
    'q',
    'r',
    's',
    't',
    'u',
    'v',
    'w',
    'x',
    'y',
    'z'] * 10
while cases > 0:
    s = raw_input()
    ans = ''
    for i in range(len(s)):
        j = cipher.index(s[i]) + i
        ans += cipher[j]
    print ans
    cases -= 1
