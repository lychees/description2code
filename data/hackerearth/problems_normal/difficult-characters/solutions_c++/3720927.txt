#include <bits/stdc++.h>
using namespace std;
struct sort_pred {
    bool operator()(const std::pair<int,int> &left, const std::pair<int,int> &right) {
        return left.first < right.first;
    }
};
int main()
{
	int tc,a[27];
	vector<pair<int,int>> vp;
	string str;
    cin >> tc;
    while(tc--) {
    	cin >> str;
    	memset(a,0,sizeof(a));
    	int length = str.length();
    	for(int i = 0; i < length; i++) {
    		a[str[i] - 'a'] +=1;
    	}
    	for(int i = 25; i >= 0;i--) {
    		vp.push_back(make_pair(a[i],i));
    	}
    	
    	stable_sort(vp.begin(),vp.end(),sort_pred());
    	
    	for(const auto& p : vp) {
    		printf("%c ",p.second+'a');
    	}
    	cout << endl;
  		vp.clear();
    }
    return 0;
}
