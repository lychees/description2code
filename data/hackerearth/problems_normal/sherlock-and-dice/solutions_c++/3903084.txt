using namespace std;
int m,n,k;
double dp[10005][55];
double func(int sum,int times)
{
    if(sum>k)
        return 0;
    if(times>=m)
    {
        if(sum==k)
            return 1;
        else
            return 0;
    }
    if(dp[sum][times]!=-1.0)
        return dp[sum][times];
    double ans=0.0;
    for(int i=1;i<=n;i++)
    {
        ans+=func(sum+i,times+1);
    }

    return dp[sum][times]=ans/n;
}
int main()
{
    int t;
    cin>>t;

    while(t--)
    {
        for(int i=0;i<=10004;i++)
        {
            for(int j=0;j<=54;j++)
            {
                dp[i][j]=-1.0;
            }
        }
        cin>>m>>n>>k;
        double ans=func(0,0);
        int c=0;
        if(ans==0.00)
        {
             printf("%0.3f %d\n",ans,c);
             continue;
        }
        while(ans<1.0)
        {
            ans*=10;
            c++;
        }

        printf("%0.3f %d\n",ans,c);
    }
}
