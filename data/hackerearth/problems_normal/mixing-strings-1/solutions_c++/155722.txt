#include<iostream>
using namespace std;
char s[15][50];
char p[300];
int a[20]={0};
int maxi=2000,n;
#include<string.h>
void recur(int j,int l)
  {int i,k,len,l2,m;
   if(j==n)
     {if(l<maxi) 
        {maxi=l;
        }
     return ;
     }
     
   for(i=0;i<n;i++)
     {if(!a[i])
        {a[i]=1;
         if(j==0)
            {p[0]='\0';
			 strcpy(p,s[i]);
             k=0;
            }
         else
           {l2=strlen(s[i]);
            m=l-1;
            int flag=0,pt,pr;
            for(k=l2-1;k>=0;k--)
               {if(s[i][k]==p[m])
                  {for(pt=k-1,pr=m-1;pt>=0 && pr>=0;pt--,pr--)
                     {if(s[i][pt]!=p[pr])
                        break;
                     }
                    if(pt<0)
                      {flag=1;
                       break;
                      }
                  }
               }
               if(flag)
             strcat(p,(s[i]+k+1));
             else
               strcat(p,(s[i]));
            // cout<<p<<endl;
           }
         recur(j+1,strlen(p));
         p[l]='\0';
         a[i]=0;
        }
     }
  }

int main()
 {int i;
 cin>>n;
 for(i=0;i<n;i++)
   {a[i]=0;

  }
 p[0]='\0';

 }
