inpt = int(raw_input())
inp_spd = raw_input()
spd_lst = inp_spd.split(" ")
inp_cons = raw_input()
cons_lst = inp_cons.split(" ")
fuel = int(raw_input())
m = 0
for j, i in enumerate(cons_lst):
    v = float(float(fuel)/float(i))
    d = float(float(spd_lst[j]) * v)
    if d > m:
        m = d
print '%.3f' % (int(m*1000)/float(1000))
