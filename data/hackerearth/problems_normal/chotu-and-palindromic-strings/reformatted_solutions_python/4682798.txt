n = input()
i = 0
while i < n:
    s1 = raw_input()
    s2 = raw_input()
    i += 1
    p = s1+s2
    if p == p[::-1]:
        print "YES"
        continue
    p = s2 + s1
    if p == p[::-1]:
        print "YES"
        continue
    print "NO"
