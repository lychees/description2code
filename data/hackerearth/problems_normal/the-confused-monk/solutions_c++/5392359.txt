#include <bits/stdc++.h>
using namespace std;
#define ll long long 
#define mod 1000000007;

ll powermod(ll a,ll m)
{
	ll res=1;
	while(m!=0)
	{
		if(m&1) res=(res*a)%mod;
		a=(a*a)%mod;
		m=m/2;
	}
	// res=(res*a)%mod;
	return res;
}

int main()
{
    
	int n;
	cin>>n;
	ll a[n];
	ll gcdval;
	ll mulval=1;
	for(int i=0;i<n;i++) 
	{
		cin>>a[i];
		if(i==0) gcdval=a[i];
		else gcdval=__gcd(gcdval,a[i]);
		mulval=(mulval*a[i])%mod;
	}
	// cout<<mulval<<" "<<gcdval<<endl;
	ll ans=powermod(mulval,gcdval);
	cout<<ans<<endl;
}
