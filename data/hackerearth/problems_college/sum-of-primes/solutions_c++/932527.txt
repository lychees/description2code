#include <bits/stdc++.h>
using namespace std;
int a[1000005];
long long unsigned b[1000005],g;
void sieve() 
{
    int i,j;
    for(i = 2; i < 1000005; i++) {
        if(a[i]) {
          for(j = 2; j*i < 1000005; j++) {
            a[i*j] = 0;
          }
        }
    }
}
int isPrime(long long int n) { 
    long long  int i;
    if(n == 2) return 1;
    if(n%2 == 0) return 0;
    for(i = 3; i*i<=n; i+=2) {
        if(n%i == 0) return 0;
    }
    return 1;
}
int main() 
{
    int i,j;
    for(i = 0; i < 1000002; i++) {
        a[i] = 1;
    }
    a[0] = a[1] = 0;
    sieve();
    b[1]=0;
	for(j=2;j<=1000001;j++) {
		if(a[j-1]) b[j]=b[j-1]+j-1;
		else b[j]=b[j-1]+a[j-1];
	}
	//for(i=1;i<=1;i++) printf("%llu ",b[38]-b[14]);
    int t,a,b1;
    scanf("%d",&t);
    while(t--) {
    	long long int g1;
    	scanf("%d%d",&a,&b1);
    	g1=b[b1+1]-b[a];

    }

}
