#include <iostream>

using namespace std;

int main()
{
	int n, m, i, j, p, color;
	cin >> n >> m;
	int c[m];
	long long int temp;
	int nums[n];
	long long int ans[n];
	bool flag;
	for(i = 0; i < m; i++)
		cin >> c[i];
	for(i = 0; i < n; i++)
		cin >> nums[i];
//	cout << "hola" << endl;
	ans[0] = c[nums[0] - 1];
	for(i = 1; i < n; i++)
	{
		flag = true;
		color = c[nums[i] - 1];
		p = ans[i-1]/color;
		while(flag)
		{
			p++;
			temp = color * p;
//			cout << p << " " << temp << endl;
			flag = false;
			for(j = nums[i]; j < m; j++)
			{
				if(temp % c[j] == 0)
				{
					flag = true;
					break;
				}
			}
		}
//		cout << i << "*" << temp << endl;
		ans[i] = temp;
	}
	cout << ans[n-1] << endl;
	return 0;
}
