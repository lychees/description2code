#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int n,t,ans,i;
    long int a[52];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	sort(a,a+n);
    	int f=0;
    	for(i=0;i<n-1;i+=2)
    	{
    		if(a[i]!=a[i+1])
    		{
    			if(a[i]==a[i+2])
    			cout<<a[i+1]<<endl;
    			else
    			cout<<a[i]<<endl;
    			f=1;
    			break;
    		}
    		
    	}
    	if(f==0)
    	cout<<a[n-1]<<endl;
    }
    return 0;
}
