#include <iostream>
using namespace std;
#include<math.h>
int main()
{
   	int t,n;
   	int i,j;
   	cin>>t;
   	int count[t],m=0;
	while(t--)
   	{
   		cin>>n;
   		string s[n];
   		for(i=0;i<n;i++)
   		cin>>s[i];
   		for(i=0;i<n;i++)
   		{
   			for(j=0;j<n;j++)
   			{
   				if(s[i][j]=='*')
   				{
						   count[m]=abs(i-n/2)+abs(j-n/2);
						   break;
   					
				}
   			}
   			if(s[i][j]=='*')
   			break;
   		}
   		m++;
   	}
   	for(i=0;i<m;i++)
   	cout<<count[i]<<"\n";
    return 0;
}
