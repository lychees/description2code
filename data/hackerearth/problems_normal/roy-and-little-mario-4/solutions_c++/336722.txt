#include <iostream>
using namespace std;
int p=0,q=0,r=0;
unsigned int pl=0, count=0;
unsigned output[3][100000];
unsigned int calculate(int a, int b)
{
    
	
	
	if(b<0){
		return 0;
	}else if(b==0){
		return 1;
	}	
	
	if(a!=0 && output[a-1][b]!=-1 ){
		return output[a-1][b];
	}
	
     pl= calculate(1,b-1)+calculate(2,b-2)+calculate(3,b-3);
     if(a!=0){
     output[a-1][b]=pl%1000000007;
     }
     return pl%1000000007;
}
	
int main()
{
	
	int t;
	cin>>t;
	int i=0;
	int n;
	int input[t];
	
	for(i;i<t;i++){
        cin>>n;
        input[i]=n;
	}
	
	for (int l=0;l<100000;l++){
		output[0][l]=-1;
		output[1][l]=-1;
		output[2][l]=-1;
	}
	
	i=0;
    for(i;i<t;i++){
    	 count=0;
        count=calculate(0,input[i]);
       
        cout<<count;
        cout<<"\n";
	}
   
   
}
