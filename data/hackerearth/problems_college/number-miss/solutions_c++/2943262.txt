#include <iostream>
#include <vector>
#include <algorithm>
#include <map>
using namespace std;

int main()
{
    long long int n,M;
    cin>>n;
    long long int i;
    map<long long int,long long int>m;
    for(i=0;i<n;i++){
        long long int k;
        cin>>k;
        if(m.find(k)!=m.end())
            m[k]++;
        else m[k]=1;
    }
    cin>>M;
    for(i=0;i<M;i++){
        long long int k;
        cin>>k;
        if(m.find(k)!=m.end())
            m[k]--;
        else m[k]=1;
    }
    vector<long long int>v;
    map<long long int,long long int>::iterator it;
    for(it=m.begin();it!=m.end();it++){
        if((it->second)!=0)
            v.push_back(it->first);
    }
    sort(v.begin(),v.end());
    for(int i=0;i<v.size();i++){
        cout<<v[i]<<' ';
    }
    cout<<'\n';
    
    return 0;
}
