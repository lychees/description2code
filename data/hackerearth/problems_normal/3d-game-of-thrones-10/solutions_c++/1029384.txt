#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
	while (t--){
		int n;
		cin >> n;
		int r = 0;
		for (int i = 0; i < n; i++) {
			int x, y, z;
			cin >> x >> y >> z;
			r ^= x;
			r ^= y;
			r ^= z;
		}
		if (r)
		  cout << "Stark\n";
		else
		  cout << "Lannister\n";
	}
}
