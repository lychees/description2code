#include <iostream>
using namespace std;

int main()
{
	int t,count,i,l;
	string s;
	cin>>t;
	while(t--)
	{
	cin>>s;
	int *flag = NULL;
	l=s.length();
	flag = new int [l];
	count =0;
	for(i=0;i<l;i++)
	  {
	  	flag[i] = 1;
	  	if(s[i]=='-')
	  	 count++;
	  }
//	cout<<"count"<<count;
	for(i=0;i<l;i++)
	  {
	  	if(s[i]=='B')
	  	  {
	  	  	if(i>0)
	  	  	  {
	  	  	  	if(s[i-1] == '-' && flag[i-1] ==1)
	  	  	  	  {count--;
	  	  	  	  flag[i-1]=-1;}
	  	  	  	
	  	  	  }
	  	  	if(i>1)
	  	  	 if(s[i-2]=='-' && flag[i-2] ==1)
	  	  	  {count--;
	  	  	  flag[i-2] =-1;}
	  	  }
	   if(s[i]=='S')
	  	  {
	  	  	if(i>0)
	  	  	  {
	  	  	  	if(s[i-1] == '-'&& flag[i-1] ==1)
	  	  	  	 {count--;
	  	  	  	  flag[i-1] =-1;}
	  	  	  	
	  	  	  }
	  	  	if(i<l-1)
	  	  	 if(s[i+1]=='-'&& flag[i+1] ==1)
	  	  	  {count--;
	  	  	   flag[i+1] =-1;}
	  	  }
	  }
	cout << count << endl;
	}
    
    return 0;
}
