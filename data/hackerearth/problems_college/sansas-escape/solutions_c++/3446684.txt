#include <iostream>
using namespace std;

int main()
{
    int t,sum[100005],a[100005],q,l,r;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>a[i];
    	if(i==0)
    		sum[i]=a[i];
    	else
    		sum[i]=sum[i-1]^a[i];
    }
    cin>>q;
    while(q--)
    {
    	cin>>l>>r;
    	if(l==0)
    		cout<<sum[r]<<endl;
    	else
    	cout<<(sum[r] ^ sum[l-1])<<endl;
    }
    return 0;
}
