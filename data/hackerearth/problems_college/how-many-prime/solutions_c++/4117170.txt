#include <bits/stdc++.h>
using namespace std;
bool prime[1000001];
void initialize1()
{

  for(int i=0;i<=1000001;i++)
  prime[i]=true;
  prime[0]=false;
  prime[1]=true;
  for(int i=2;i*i<=1000001;i++)
  {
    if(prime[i])
    {
      for(int j=i*i;j<=1000001;j+=i)
         prime[j]=false;
    }

  }


}
int main()
{
  initialize1();
  int t;
  cin>>t;
  int a;
  while(t--)
  {
    cin>>a;
    int i;

      int bc=-1,vc=-1;
      for( i=(a-1)/2;i>0;i--)
    {
      if(prime[i]&&prime[a-i])
        {
          bc=i;
          vc=a-i;
          break;
        }
    }
    cout<<bc<<" "<<vc<<endl;
  }

  
  return 0;
}
