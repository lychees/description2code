#include <iostream>
using namespace std;

int main()
{
	int t,n;
    cin >> t;
    int count;
    for(int i=0; i<t; i++){
    	cin >> n;
    	count = 0;
    	for(int j=1; j<=n; j++){
    		for(int k=j+1; k<=n; k++){
    			if((j^k) <= n){
    				count = count + 1;
    			}
    		}
    	}
    	cout << count << endl;
    }
    return 0;
}
