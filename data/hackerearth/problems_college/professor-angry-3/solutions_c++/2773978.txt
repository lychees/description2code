#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int a[1000],n,k,count=0;
    	cin>>n>>k;
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	for(int i=0;i<n;i++)
    	if(a[i]<=0)
    	count++;
    	if(count>=k)
    	cout<<"NO"<<endl;
    	else
    	cout<<"YES"<<endl;
    }
    return 0;
}
