#include<bits/stdc++.h>
#define ll long long
#define sc(n) scanf("%d",&n)
#define scl(n) scanf("%lld",&n)
#define prt(n) printf("%d ",n)
#define mod 1000000007
using namespace std;
ll power(ll n,ll p)
{
	if(p==0)
	return 1;
	if(p==1)
	return n;
	ll c=power(n,p/2)%mod;
	c=(c*c)%mod;
	if(p%2)
	c=(c*n)%mod;
	return c;
}
int main()
{
//	freopen("in.txt","r",stdin);
	int n,t,i;
	ll a;
	set<ll> st;
	sc(t);
	while(t--)
	{
		st.clear();
		sc(n);		
		ll ans=0;
		for(i=0;i<n;i++)
		{
			scl(a);
			st.insert(a);
		}
		int p=st.size();
		ans=power(n,p)%mod;
		cout<<ans<<"\n";
	}
return 0;
}
