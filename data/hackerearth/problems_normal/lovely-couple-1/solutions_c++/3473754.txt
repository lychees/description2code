#include <iostream>
#include <vector>
#include <math.h>
using namespace std;
vector<bool> sieve;

int hcf(int a, int b){
	if(b==0){
		return a;
	}else{
		return hcf(b, a%b);
	}
}

void primeSieve(){
	for(long i=0;i<=1000000;i++){
		sieve.push_back(true);
	}
	sieve[0]=false;
	sieve[1]=false;
	for(long i = 2; i * i <= 1000000; ++i) {
         if(sieve[i] == true) {
             for(int j = i * i; j <= 1000000 ;j += i)
                 sieve[j] = false;
        }
    }
}
int main()
{
    primeSieve();
    long t;
    cin>>t;
    while(t--){
    	int a, b;
    	cin>>a>>b;
    	int high = 0, low = 0;
    	if(a>b){
    		low = b, high = a;
    	}else{
    		low = a, high = b;
    	}
    	int h = hcf(high, low);
    	long lcm = (a*b)/h;
    	int c = 0;
    	long n = lcm;
    	bool isDiv = false;
    	while(n%2 == 0){
    		isDiv = true;
    		n = n/2;
    	}
    	if(isDiv){
    		//cout<<"2 is a factor"<<endl;
    		c++;
    	}
    	
    	for(long i=3;i<=n;i+=2){
    		if(sieve[i]==true){
    			//cout<<i<<" is a prime"<<endl;
    			isDiv = false;
    			while(n%i == 0){
    				isDiv = true;
    				n = n/i;
    			}
    			if(isDiv){
    				c++;
    				//cout<<i<<" is a factor"<<endl;
    			}
    		}
    	}
    	//cout<<c<<endl;
    	if(sieve[c]){
    		cout<<"Yes"<<endl;
    	}else{
    		cout<<"No"<<endl;
    	}
    	
    }
    return 0;
}
