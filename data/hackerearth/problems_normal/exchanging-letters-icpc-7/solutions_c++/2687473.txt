#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

long long mod=1000000007LL;

long long factorial(long long n){
    long long ans=1LL;
    for(long long i=1;i<=n;i++)
        ans=(ans*i)%mod;
    return ans;
}

long long power(long long a,long long e){
    long long ans=1LL;
    while(e>0){
        if(e%2==1)
            ans=(ans*a)%mod;
        a=(a*a)%mod;
        e/=2;
    }
    return ans;
}

long long inv(long long n){
    return power(n,mod-2);
}

long long myabs(long long n){
    if(n<0)
        return -n;
    return n;
}

long long c(long long n,long long k){
    long long ans=1LL;
    ans=(ans*factorial(n))%mod;
    ans=(ans*inv(factorial(k)))%mod;
    ans=(ans*inv(factorial(n-k)))%mod;
    return ans;
}

int main(){
    long long n,k,t,temp,sum=0;
    scanf("%lld%lld%lld",&n,&k,&t);
    for(long long i=0;i<k;i++){
        scanf("%lld",&temp);
        sum+=temp;
    }
    temp=myabs(sum-t)+k;
    if(temp%2==1||temp/2>k)
        printf("0\n");
    else
        printf("%lld\n",c(k,temp/2));
}
