#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int T;
	scanf("%d", &T);
	for(int i=0; i<T; i++) {
		int N, K;
		scanf("%d %d\n", &N, &K);
		vector<vector<int> > names(K, vector<int>(N));
		char temp;
		for(int j=0; j<N; j++){
			for(int k=0; k<K; k++) {
				scanf("%c", &temp);
				names[k][j] = temp-'a';
				//cout << names[k][j] << " ";
			}
			scanf("\n");
			//cout << endl;
		}
		for(int k=0; k<K; k++) {
			sort(names[k].begin(), names[k].end());
			cout << (char)(names[k][(N-1)/2]+'a');
		}
		cout << endl;
		// Optimize
		
		// vector<int> solution(K, 0);
	}
    return 0;
}
