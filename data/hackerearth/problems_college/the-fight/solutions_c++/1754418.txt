#include <iostream>
#include<cstring>
using namespace std;
long long b[1000005],a[10000005];
int main()
{
    int n,m,i,j,k;
    
    scanf("%d",&n);
    
    for(i=1;i<=n;i++)
    scanf("%lld",&a[i]);
    
   scanf("%d",&m);
 
   while(m--)
   {
   	scanf("%d%d%d",&i,&j,&k);
   	b[i]+=k;
   	b[j+1] -=k;
   }
   
   for(i=1;i<=n;i++)
   {
   	b[i] = b[i] + b[i-1];
   	printf("%lld\n",b[i]+a[i]);
   }
    
   
    return 0;
}
