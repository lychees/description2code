#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
int t, d;

int getNum(int x) {
	int lo, hi, mid, ans;
	lo = 1; hi = 50000;
	while (lo <= hi) {
		mid = lo + (hi - lo) / 2;
		if ((ll)mid * (mid + 1) < 2LL * x) {
			lo = mid + 1;
		} else {
			hi = mid - 1;
			ans = mid;
		}
	}
	return ans;
}

int main()
{
    scanf("%d", &t);
    while (t--) {
    	scanf("%d", &d);
    	int x = getNum(d);
    	if (x % 3 == 0) printf("SL\n");
    	else if (x % 3 == 1) printf("LB\n");
    	else printf("BS\n");
    }
    return 0;
}
