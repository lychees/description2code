#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int l,r,mx=-1;
    	cin>>l>>r;
    	for(int i=l;i<=r;i++)
    	{
    		for(int j=i;j<=r;j++)
    		{
    			if((i^j)>mx)
    				mx=i^j;
    		}
    	}
    	cout<<mx<<endl;
    }
    return 0;
}
