#include <iostream>
using namespace std;

int main()
{
    int t,n,j;
    cin>>t;
     long long sum,d,e;
    while(t--){
    	sum=0;
    
    cin>>n;
    
    for(j=n-1;j>=0;j--)
    {
      d=d+j;
      
    }
    e=d;
    for(j=n-1;j>=0;j--)
    {
    e=e-j;
    d=d+e;
    }
    sum=d*2;
    d=0;
    cout<<sum<<"\n";
}
return 0;
}
