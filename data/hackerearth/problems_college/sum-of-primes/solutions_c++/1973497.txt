#include <iostream>
#include<string.h>
#include<math.h>
# define N 1000001
 long long int a[N];
using namespace std;

int main()
{
   long long int i,j,k,l,t,m,n,sum,flag; 
    	memset(a,0,N);
   	a[0]=1;
   	a[1]=1;
  
   	for(i=2;i<=sqrt(N);i++)
   	{
   		if(a[i] == 0)
   		{
   	  for(j = i*i ;j<=N;j+=i)
   	  {
   			a[j] = 1;
   	  }
   	}
   	}
  cin>>t;
   while(t--)
   {
   		cin>>m>>n;
   	sum=0;
   for(i=m;i<=n;i++)
   {
   	if(a[i]==0)
   	{
   	//	cout<<i;
   	sum=sum+i;
   }
   }
   flag=0;
   if(sum<N)
   {
   if(a[sum]==1)
   {
   flag=1;
   }
   }
   else
   {
   	
   	for(i=2;i<=sqrt(sum);i++)
   	{
   		if(sum%i==0)
   		{
   			flag=1;
   			break;
   		}
   	}
   }
 // out<<sum;
   if(flag==0)
   cout<<"YES\n";
   else
   cout<<"NO\n";
   }
    return 0;
}
