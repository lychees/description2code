#include <bits/stdc++.h>
#define ll long long
#define MOD 1000000007
using namespace std;

ll pw(ll a, ll b) {
	if(b == 0) return 1LL;
	ll x = pw(a, b/2);
	ll y = (x % MOD) * (x % MOD);
	y %= MOD;
	if(b&1) {
		y = (y * (a % MOD)) % MOD;
	}
	return y % MOD;
}

ll inv(ll x) {
	return pw(x, MOD-2);
}

void solve() {
	int n; cin >> n;
	vector<int> a(n);
	for(int i=0;i<n;i++) cin >> a[i];
	ll ans = 0;
	for(int i=1;i<1<<n;i++) {
		ll pdt = 1;
		ll sm = 0;
		for(int j=0;j<n;j++) {
			if(i & (1<<j)) {
				pdt = ((pdt % MOD) * (a[j] % MOD)) % MOD;
				sm = (sm + a[j]) % MOD;
			}
		}
		ll val = ((pdt % MOD) * (inv(sm) % MOD) ) % MOD;
		ans = max(ans, val);
	}
	cout << ans << "\n";
}

int main() {
	int t; cin >> t;
	for(int qq=1;qq<=t;qq++) {
		cout << "Case #" << qq << ": ";
		solve();
	}
}///
