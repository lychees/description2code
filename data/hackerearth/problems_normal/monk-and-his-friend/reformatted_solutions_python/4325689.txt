def counter(ip):
    op = 0
    while (ip != 1):
        op = op+ip % 2
        ip = ip/2
    return op+1


N = int(input())
for i in range(N):
    IP = raw_input()
    IPP = IP.split()
    ip = [int(IPP[0]), int(IPP[1])]
    xor = ip[0] ^ ip[1]
    if (xor == 0):
        print 0
    else:
        count = counter(xor)
        print count
