#include <iostream>
using namespace std;

int main()
{
    long long int a[100000]={0},f[10]={0};int t,c=0;
    cin>>t;
    for(int i=0;i<t;i++){
    	cin>>a[i];
    }
    for(int i=0;i<t;i++){
    	c++;
    	for(int j=i;j<t;j=j+3){
    		f[i]+=a[j];
    	}
    	
    	if(c==3)break;
    	
    }
    int d=0;
    for(int i=0;i<t+3;i++){
    	d++;

    	cout<<f[i]<<" ";
    	if(d==3)break;
    }
    
    
    return 0;
}
