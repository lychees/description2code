#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int nr,ng;
    	scanf("%d%d",&nr,&ng);
    	int arr[nr];
    	for(int i = 0;i<nr;i++)
    	{
    		scanf("%d",&arr[i]);
    	}
    	int x = arr[0];
    	sort(arr,arr+nr);
    	int g = 0;
    	int my_g = 0;
    	vector<int> v;
    	for(int i = nr-1;i>=0;i--)
    	{
    		if(g >ng)
    		g = 1;
    		else
    		{
    		g = g+1;
    		if(g > ng)
    		g = 1;
    		}
    		if(x == arr[i])
    		{
    			my_g = g;
    			break;
    		}
    	}
    //	cout<<"my_g = "<<my_g<<endl;
    g = 0;
    	for(int i = nr-1;i>=0;i--)
    	{
    		if(g >ng)
    		g = 1;
    		else
    		{
    		g=g+1;
    		if(g > ng)
    		g = 1;
    		}
    		if(g == my_g)
    		v.push_back(arr[i]);
    	}
    	int cnt = 0;
    //	printf("%d\n",v.size());
    	for(int i =0;i<v.size();i++)
    	{
    	//	cout<<v[i]<<endl;
    		if(v[i]>x)
    		cnt++;
    	}
    	printf("%d\n",cnt);
    }
    return 0;
}
