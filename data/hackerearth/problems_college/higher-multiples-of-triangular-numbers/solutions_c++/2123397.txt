#include <iostream>
#include <cmath>
using namespace std;
int values[400];
int divisors[400];
int ind = 0;
int finddivisors(unsigned long num) {
	int no = 0;
	for(unsigned long i = 1; i<=num;i++) {
		if(num%i==0) no++;
	}
	return no;
}
void calculate() {
	values[ind]=1;
	divisors[ind++]=1;
	for(long i=2;i<385;i++) {
		values[ind]=values[ind-1]+i;
		int d = finddivisors(values[ind]);
		divisors[ind++] = d;
		
	}
}
int main()
{
    calculate();
    int t;
    cin>>t;
    int n;
    while(t--) {
    	cin>>n;
    	for(int i=0;i<ind;i++) {
    		if(divisors[i]>n) {
    			cout<<values[i]<<endl;
    			break;
    		}
    	}
    }
    return 0;
}
