#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,count = 0;
    	long int x;
    	cin>>n;
    	for(int i = 0 ; i < n ;i++)
    	{
    		cin>>x;
    		if(x%2 != 0)
    		count++;
    	}
    	if(count %2 == 0)
    	cout<<"Yes";
    	else cout<<"No";
    	cout<<endl;
    }
    return 0;
}
