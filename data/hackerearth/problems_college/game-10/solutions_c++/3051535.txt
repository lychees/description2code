#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,m;
		cin>>n>>m;
		long long int a[n],l=0;
		for(int i=0,j=n-1;j>=i;i++,j--)
		{
			a[l] = j;
			l++;
			a[l] = i;
			l++;
		}
		for(int i=0;i<n;i++)
		{
			if(a[i] ==m)
			{
				cout<<i<<"\n";
				break;
			}
		
		}
	}
}
