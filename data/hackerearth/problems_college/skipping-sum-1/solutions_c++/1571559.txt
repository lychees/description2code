#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define ll long long
#define MAX 100010

ll a[MAX];
ll f[MAX][20];
int n, q, l, r, k;

void Init(){
	memset(f, 0, sizeof(f));
	for (int i = 1; i <= n; i++)
		for (int j = 1; j <= 10; j++)
			if (i - j > 0)
				f[i][j] = a[i] + f[i - j][j];
			else
				f[i][j] = a[i];
}

int main(){
    scanf("%d%d", &n, &q);
    for (int i = 1; i <= n; i++)
    	scanf("%lld", &a[i]);
    Init();
    while (q--){
    	scanf("%d%d%d", &l, &r, &k);
    	r = l + ((r - l) / k) * k;
    	if (l > k)
    		printf("%lld\n", f[r][k] - f[l - k][k]);
    	else
    		printf("%lld\n", f[r][k]);
    }
    
    return 0;
}
