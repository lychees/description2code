#include <iostream>
using namespace std;

int main()
{
	string s1,s2;
	cin>>s1;
	int size1,size2;
	int anto=0,syno=0;

	size1=s1.size();
	int n;
	cin>>n;
	while(n--)
	{
		cin>>s2;
		size2=s2.size();
		if(size1==size2)
		{
		if(s1[0]==s2[0])
		syno=syno+1;
		else
		{
			if((s1[0]+3)==s2[0])
			anto=anto+1;
		}
		}
		
	}
	cout<<syno<<" "<<anto;
    return 0;
}
