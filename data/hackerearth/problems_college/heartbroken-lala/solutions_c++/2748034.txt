#include <cstdio>
#include <cstring>
#include <cstdlib>

typedef long long int ll;

using namespace std;

ll K;

int known[8] = {-1, 0, 0, 1, 0, 0, 1, 1};

ll closestPow(ll no) {
  ll temp = 1;
  while(temp <= no) {
    temp *= 2;
  }
  return (temp / 2);
  
}

int solve(ll no) {
  if(no <= 7) return known[no];
  ll pow = closestPow(no);
  if(pow == no) return 0;
  else return (1 - solve(pow - (no - pow)));
}

void work(int index) {
  scanf("%lld", &K);
  int ans;
  if(K == 1) ans = 0;
  else ans = solve(K);

  //printf("Case #%d: %d\n", index, ans);
  printf("%d\n", ans);
}

int main() {
  int T;
  scanf("%d", &T);
  int i;
  for(i = 1; i <= T; i++) work(i);
  return 0;
}
