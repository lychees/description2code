#include <iostream>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    bool sorted;
    while(t--)
    {
    	cin>>n;
    	long int array[n];
    	sorted=true;
    	cin>>array[0];
    	for(int i=1;i<n;i++)
    	{
    	cin>>array[i];
    	if(array[i]<array[i-1])
    	sorted=false;
    	}
    	if(sorted)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    	
    }
}
