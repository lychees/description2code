#include<bits/stdc++.h>
using namespace std;
struct val
{
    int x;
    int y;
};

int h[1000009];
void solve(val a[],int n)
{
    memset(h,0,sizeof(h));
    int count=0;
    int max=0;
int i;
    for(i=0;i<n;i++)
    {
        if(a[i].y==1)
        {
        if(!h[a[i].x])
        {
            count++;
            if(count>max)
                max=count;
            h[a[i].x]=1;
        }
        }
        else
        {
            count=0;
        }
    }
    cout<<max<<endl;
}
int main()
{
int t;
cin>>t;
while(t--)
{
int n;
cin>>n;
val pi[n];
int i;

for(i=0;i<n;i++)
{
    cin>>pi[i].x>>pi[i].y;
}
solve(pi,n);
}
}
