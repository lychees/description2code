#include <iostream>
using namespace std;

#include <stdio.h>

int main()
{
	int T,N,temp;
	cin>>T;
	while(T--)
	{
	 cin>>N;
	 temp=N;
	 int ans=0;
	 while(temp!=1)
	 {
	 while(temp%3!=0 && temp%2!=0)
	 {
	 	temp--;
	 	ans++;
	 }
	 while(temp%3==0)
	 {
	 	temp=temp/3;
	 	ans++;
	 }
	 while(temp%2==0)
	 {
	 	temp=temp/2;
	 	ans++;
	 }
	 }
	 cout<<ans<<endl;
	 
	}
	return 0;
}
