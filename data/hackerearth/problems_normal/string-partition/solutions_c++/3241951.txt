#include<bits/stdc++.h>
using namespace std;
long long dp[102][102];
char str[105];
long long find(int a, int b) {
	long long ret=0;
	for(int i=a;i<=b;i++) {
		ret*=10;
		ret+=str[i]-48;
	}
	return ret;
}
int main() {
	int t;
	cin>>t;
	while(t--) {
		long long k,z;
		cin>>k>>z;
		scanf("%s",str+1);
		int len=strlen(str+1);
		bool f=false;
		memset(dp,-1,sizeof(dp));
		for(int i=1;i<=len;i++) {
			if(f) {
				dp[1][i]=-1;
				continue;
			}
			long long num=find(1,i);
			if(num>z) {
				f=true;
				dp[1][i]=-1;
			}
			else
				dp[1][i]=num;
		}
		for(int i=2;i<=k;i++) {
			for(int j=1;j<=len;j++) {
				for(int p=j;p>=i;p--) {
					long long num=find(p,j);
					if(num>z)
						break;
					if(dp[i-1][p-1]!=-1)
						dp[i][j]=max(dp[i][j],dp[i-1][p-1]+num);
				}
			}
		}
		cout<<dp[k][len]<<endl;
	}
	return 0;
}
