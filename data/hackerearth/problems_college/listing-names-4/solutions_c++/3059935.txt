#include<bits/stdc++.h>
using namespace std;
set <string> v;
map <string,int> map1;
int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>s;
		v.insert(s);
		map1[s]++;
	}
	set <string> ::iterator it=v.begin();
	while(it!=v.end())
	{
		cout<<*it<<" "<<map1[*it]<<endl;
		it++;
	}
	return 0;
}
