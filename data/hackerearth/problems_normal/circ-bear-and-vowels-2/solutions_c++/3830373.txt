#include <bits/stdc++.h>
using namespace std;

int main()
{
   ios::sync_with_stdio(false);
   cin.tie(0);
   cout.tie(0);
   int t;
   cin>>t;
   while(t--)
   {
   		char a[100];
   		cin>>a;
   		int i;
   		int count1=0,count2=0,flag=0;
   		int c=0;
   		for(i=0;i<strlen(a);i++)
   		{
   			if(a[i]=='a' ||a[i]=='e' ||a[i]=='i' ||a[i]=='o' ||a[i]=='u' ||a[i]=='y' )
   				{
   					count1++;
  					c=0;
   				}
   			else
   				{
   					count2++;
   					c++;
   				}
   				if(c==3)
   				{
   					flag=1;
   					break;
   				}
   		}
   		if(count2>count1)
   			flag=1;
   		if(flag==1)
   			cout<<"hard"<<endl;
  		else
  			cout<<"easy"<<endl;
   }
    return 0;
}
