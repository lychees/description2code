#include<bits/stdc++.h>
using namespace std;
#define ll   long long  int
#define INF 10000
#define mp make_pair
#define mod 1000000007
#define f first
#define pi pair<ll,ll>
#define pb emplace_back
#define s second
map<ll,ll>m;
ll bpow(ll x,ll n){
ll ans=1;
while(n>0){
    if(n&1) ans*=x;
    x*=x;
    x%=mod;
    ans%=mod;
    n/=2;
}
return ans;
}
ll r(ll n){
    if(m.find(n)!=m.end()) return m[n];
    ll h=n/4;
    if(n%4!=0) h++;
    ll t=bpow(2,h);
    t*=r((n+4)/5);
    t%=mod;
    t*=r((n+9)/10);
    t%=mod;
    m[n]=t;
    return t;
}
int main(){
ll t;
cin>>t;
ll n;
m[0]=1;
m[1]=1;
m[2]=1;
m[3]=3;
m[4]=3;
m[5]=15;
m[6]=15;
m[7]=105;
m[8]=105;
m[9]=105;
m[10]=105;
while(t--){
    cin>>n;
    ll h=r(n);
    ll ans=0;
    for(int i=1;i<=5;i++){
        for(int j=1;j<=5;j++){
            if(__gcd(i,j)==1){
                    ans=max(ans,bpow(i*h,j*h));
            }
        }
    }
    cout<<ans<<"\n";
}
}
