#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   if((t<1)||(t>10))
   cout<<"Invalid Test\n";
   else
   {
   while(t)
   {
   	string s;
   	cin>>s;
   	bool c_count=false;int u_count=0,l_count=0;
   	for(int i=0;i<s.length();i++)
   	{
   		if(isalpha(s[i]))
   		  c_count=true;
   		if((s[i]>96)&&(s[i]<123))l_count++;
   		else if((s[i]>64)&&(s[i]<91))u_count++;
   		 
   	}
   	if((!c_count)||(s.length()>100))
    cout<<"Invalid Input\n";
    else
     {
     	cout<<min(l_count,u_count)<<"\n";
     }
   	t--;
   }
   }
    return 0;
}
