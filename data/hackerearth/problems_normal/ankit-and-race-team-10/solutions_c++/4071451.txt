#include<bits/stdc++.h>
using namespace std;
 
#define fr(i,n) for(int i=0;i<n;i++)
#define fr1(i,n) for(int i=1;i<=n;i++)
#define ll long long
#define pii pair<int,int>
#define ff first
#define ss second
#define mp make_pair
#define pb push_back
#define MOD 1000000007
#define MAX 100002
#define INF 100005
ll fact[MAX];
ll inv[MAX];

ll invpow(ll a){
	ll ans=1,b=MOD-2;
	while(b>0){
		if(b%2)ans=(ans*a)%MOD;
		a=(a*a)%MOD;
		b/=2;
	}
	return ans;
}

ll ncr(int n,int r){
	if(r>n||r<0)return 0;
	return (((fact[n]*inv[r])%MOD)*inv[n-r])%MOD;
}

int main(){
	int t,n,k;
	cin>>t;
	while(t--){
		cin>>n>>k;
		memset(fact,0,sizeof(fact));
		memset(inv,0,sizeof(inv));
		int arr[n];
		fr(i,n){
			cin>>arr[i];
		}
		sort(arr,arr+n);
		fact[0]=1;
		fr1(i,n){
			fact[i]=(i*fact[i-1])%MOD;
		}
		fr1(i,n){
			inv[i]=invpow(fact[i]);
		}
		inv[0]=1;
		ll ans=0;
		fr1(i,n-k+1){

		//	cout<<ncr(n-i,k-1)<<endl;
			ans=(ans+(arr[i-1]*ncr(n-i,k-1))%MOD)%MOD;//when ith element is the lowest element-k-1 numbers after the ith have to be selected
		}
		cout<<ans<<endl;
	}
}
