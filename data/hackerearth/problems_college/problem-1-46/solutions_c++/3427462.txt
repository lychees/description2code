#include<bits/stdc++.h>
using namespace std;
int dp[100001][26];
int main(){
	int n;
	cin>>n;
	string s;
	cin>>s;
//	int dp[100001][26];
	memset(dp,0,sizeof(dp));
//	for(int i=0;i<26;i++)dp[0][i]=0;
	
	for(int i=0;i<=n;i++){
		for(int j=0;j<26;j++){
			dp[i+1][j]=dp[i][j]+(s[i]-'a'==j?1:0);
		}
	}
	int q;
	cin>>q;
	int l,r;
	while(q--){
		cin>>l>>r;
		int count=0;
		for(int j=0;j<26;j++){
			if((dp[r][j]-dp[l-1][j])%2){
				count++;
			}
		}
		int range=r-l+1;
		if(range==1){
			cout<<"NO"<<endl;
			continue;
		}
		if(count==0 && range%2==0)cout<<"YES"<<endl;
		else if(count==1 && range%2)cout<<"YES"<<endl;
		else if(count==2 && range%2==0)cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
	return 0;
}
