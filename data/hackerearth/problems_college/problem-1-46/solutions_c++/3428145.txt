#include<bits/stdc++.h>
using namespace std;
int a[100010][26];

int main()
{
	int n,i,j,q,l,r,count;
//	int a[100010][26];
	string st;
	cin>>n;
	cin>>st;
	for(i=0;i<n;i++)
	{
		for(j=0;j<26;j++)
		{
			a[i][j]=0;
		}
	}
	a[0][st[0]-'a']=1;
	for(i=1;i<n;i++)
	{
		int val=st[i]-'a';
	//	cout <<st[i]<<endl;
		for(j=0;j<26;j++)
		{
			a[i][j]=a[i-1][j];
		}
		a[i][val]++;
	}
	cin>>q;
	int len,v;
	for(i=0;i<q;i++)
	{
		cin>>l>>r;
		len=r-l+1;
		count=0;
		l=l-1;
		r=r-1;
		for(j=0;j<26;j++)
		{
			
			if(l==0)
				v=a[r][j];
			else
				v=a[r][j] - a[l-1][j];
			if(v%2)
				count++;
		}
		if(len==1)
			cout<<"NO"<<endl;
		else  if(len%2==0)
		{
			if(count==0)
				cout<<"YES"<<endl;
			else if(count==2)
				cout<<"YES"<<endl;
			else
				cout<<"NO"<<endl;
		}
		else
		{
			if(count==1)
				cout<<"YES"<<endl;
			else
				cout<<"NO"<<endl;
		}
	}
	return 0;
}
