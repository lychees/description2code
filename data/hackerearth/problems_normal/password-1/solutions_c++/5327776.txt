#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n; cin>>n;
    map<string,int> m;
    string str[n];
    for(int i=0;i<n;i++)
    {
    	cin>>str[i];
    	m[str[i]]=1;
    }
    for(int i=0;i<n;i++)
    {
		string tmp=str[i];
		reverse(tmp.begin(),tmp.end());
		if(m[tmp]==1)
		{
			cout<<tmp.length()<<" "<<tmp[tmp.length()/2]<<endl;
			break;
		}
    }
}
