#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define pii pair<int,int>
#define mp make_pair
#define pb push_back
#define si(a)     scanf("%d", &a)
#define sl(a)     scanf("%lld", &a)
#define pi(a)     printf("%d", a)
#define pl(a)     printf("%lld", a)
#define pm(str)   printf("%s\n",str)
#define pn        printf("\n")

int main()
{
	vector<int> x1;
	vector<int> x2;
	vector<int> y1;
	vector<int> y2;

	int X;
	int n;
	scanf("%d %d",&X,&n);

	x1.clear();
	x2.clear();
	y1.clear();
	y2.clear();
	x1.resize(n);
	x2.resize(n);
	y1.resize(n);
	y2.resize(n);
	
	for(int i = 0;i < n;i++){

		scanf("%d %d %d %d",&x1[i],&y1[i],&x2[i],&y2[i]);
		if(y2[i] < y1[i]){
			swap(y1[i],y2[i]);
			swap(x1[i],x2[i]);
		}
	}

	int Y = INT_MAX;
	while(1){
		int ylimit = 0;
		int pos = -1;
		for(int i = 0;i < n;i++){
			if(X >= min(x1[i],x2[i]) && X<= max(x1[i],x2[i]) && Y>=y2[i] && X != x1[i]){
				if(y2[i] > ylimit){
					ylimit = y2[i];
					pos = i;
				}
			}
		}
		if(pos == -1)
			break;
		else{
			X = x1[pos];
			Y = y1[pos];
		}
	}
	
	printf("%d\n",X);

	return 0;
}
