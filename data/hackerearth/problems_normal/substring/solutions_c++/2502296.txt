#include<iostream>
using namespace std;
int a[5];
int main(void){
	int T;
	string s;
	int i,n,x , re;
	a['a']=a['A']=a['B']=a['b']=a['c']=a['C']=1;
	cin>>T;
	while(T--)
	{
		cin>>s;
		s+='z';
		n=s.size();
		x=0;
		re=0;
		for(i=0;i<n;i++)
		{
			if(a[s[i]])
			{
				x++;
			}
			else
			{
				re+=x*(x+1)/2;
				x=0;
			}
		}
		cout<<re<<endl;
	}
	return 0;
}
