#include <iostream>
using namespace std;

int main()
{
	int t, k, n, count;
	cin >> t;
	while (t--)
	{
		cin >> n;
		count = 0;
		for (int i = 1; i <= n; i++)
		{
			for (int j = i + 1; j <= n; j++)
			{
				k = i^j;
				if (k <= n)
				{
					count += 1;
				}
			}
		}
		cout << count << endl;
	}
	return 0;
}
