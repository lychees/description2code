#include <iostream>
using namespace std;

int main()
{
    long t;
    int n;
    cin>>t;
    while(t-->0){
    	cin>>n;
    	if(n<32)cout<<1<<endl;
    	else if(n<60)cout<<2<<endl;
    	else if(n<91)cout<<3<<endl;
    	else if(n<121)cout<<4<<endl;
    	else if(n<152)cout<<5<<endl;
    	else if(n<182)cout<<6<<endl;
    	else if(n<213)cout<<7<<endl;
    	else if(n<244)cout<<8<<endl;
    	else if(n<274)cout<<9<<endl;
    	else if(n<305)cout<<10<<endl;
    	else if(n<335)cout<<11<<endl;
    	else cout<<12<<endl;
    }
    return 0;
}
