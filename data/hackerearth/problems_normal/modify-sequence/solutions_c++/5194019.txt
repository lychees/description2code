#include <iostream>
#include <vector>
using namespace std;

int main()
{
	vector<int>arr;
	int n,j,count=0,val;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>val;
		arr.push_back(val);
		
	}
	int i;
	if(n==1){
		if(arr[0]!=0)
		cout<<"NO";
		else
		cout<<"YES";
	}
	else{
		for(i=0;i<n-1;i++)
			arr[i+1]=arr[i+1]-arr[i];
		if(arr[i]==0)
		cout<<"YES";
		else
		cout<<"NO";
	}
}
