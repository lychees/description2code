int lcs(int a1[], int a2[], int n)
{
	int c[n+1][n+1];

	FOR(i,n)
	{
		c[0][i] = c[i][0] = 0;
	}

	FORS(i,1,n+1)
	{
		FORS(j,1,n+1)
		{
			if(a1[i-1] == a2[j-1])
			{
				c[i][j]=c[i-1][j-1]+1;
			}
			else
			{
				c[i][j] = max(c[i][j-1], c[i-1][j]);
			}

		}
	}

	return c[n][n];
}
