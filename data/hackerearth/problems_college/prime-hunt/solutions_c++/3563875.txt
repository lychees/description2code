#include <bits/stdc++.h>
using namespace std;

bool prime[100000003];
void abc(){
    	memset(prime,true,sizeof(prime));
    	long long int i,p;
    	for (p=2; p*p<=100000003; p++)
    	{
		     if (prime[p] == true)
        	{
        	    for ( i=p*2; i<=100000003; i += p)
        	        prime[i] = false;
        	}
    	}
}

int main()
{
    int t;
    cin>>t;
    abc();
    
    while(t--){
    	long long int n,m,count=0,p,i;
    	
    	cin>>n>>m;
    	
		if(n==1)count--;
    	for (p=n; p<=m; p++)
    	   	if (prime[p])
        		count++;
    	
    	
    	cout<<count<<endl;
    }
    return 0;
}
