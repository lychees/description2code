#include<bits/stdc++.h>
using namespace std;
 
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		string s;
		cin>>s;
		int l = s.length();
		string best = s;
		for(int i = 0;i<l;i++){
			for(int j = 0;j<l;j++){
				string a = s;
			    a.erase(a.begin()+i);
			    a.insert(j,1,s[i]);
			    best = min(best,a);
				
			}
		}
		cout<<best<<endl;
		
	}
}
