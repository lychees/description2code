#include <iostream>
using namespace std;

int main()
{
	int i,flag =0;
	string s;
	cin >> s;
	for(i=0;i<s.size();i++)
	{
		if(s[i] == '?')
		{
			flag = 1;
			continue;
		}
		if(flag == 1)
		{
			if(s[i] == '=' &&(s[i-1] == 'e' || s[i-1] == 'd' || s[i-1] == 'y') )
			{
				cout << ": ";
			}
			else if(s[i] == '&' && (s[i+1] == 'u'|| s[i+1] == 'p'||s[i+1] == 'r' ||s[i+1] == 'k'))
			{
				cout << endl;
			}
			else
			{
				cout <<s[i];
			}
		}
	}
	
    
    return 0;
}
