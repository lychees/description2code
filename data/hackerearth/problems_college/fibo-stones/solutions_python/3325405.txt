def fib_to(n):
    fibs = [0, 1]
    for i in range(2, n+1):
        fibs.append(fibs[-1] + fibs[-2])
    return fibs


allFibs = fib_to(81)
allFibs = allFibs[1:]
n, m = map(int, raw_input().split())
stones = sum(map(int, raw_input().split()))
