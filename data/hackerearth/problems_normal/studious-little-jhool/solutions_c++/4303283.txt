#include<bits/stdc++.h>

using namespace std;

int dp[120];

int check(int n)
{
	if(n==0)
		return 0;
	else if(n<0)
		return 1e9;
	else if(dp[n]!=-1)
		return dp[n];
	else 
	{
		int	res=1+min(check(n-12),check(n-10));
		return dp[n]=res;
	}
	
}

int main()
{
	int t;
	memset(dp,-1, sizeof(dp));
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int res=check(n);
		if(res>n)
		printf("-1\n");
		else
		printf("%d\n",res);
	}
	return 0;
}
