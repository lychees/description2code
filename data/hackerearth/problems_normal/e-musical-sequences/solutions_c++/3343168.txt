#include<bits/stdc++.h>
#define ll long long
using namespace std;
long long int n,m;
ll z;
ll int a[1000005];
ll int answers[1000005];
int main()
{
	int i,j,fl,test;
	scanf("%d",&test);
	while(test--)
	{	long long sum=0;
		ll int size_ans=0;
		scanf("%lld%lld%lld",&n,&m,&z);
		for(i=0;i<n;i++)
		{
			scanf("%lld",&a[i]);
			answers[size_ans++]=a[i]%m;
				if(answers[size_ans-1]<0)
			answers[size_ans-1]+=m;
		}
		fl=1;
		for(i=n-1;i>=0;i--)
		{
			if(fl)
				sum+=a[i];
			else
				sum-=a[i];
			sum=sum%m;
			if(sum<0)
				sum+=m;
			fl=(fl+1)%2;
		}

		answers[size_ans++]=sum%m;
		if(answers[size_ans-1]<0)
			answers[size_ans-1]+=m;
		if(n%2)
		{
			printf("%lld\n",(answers[z%size_ans] ));
			continue;
		}
		
		for(i=0;i<n;i++)
		{
			answers[size_ans++]=((-1)*a[i])%m;
				if(answers[size_ans-1]<0)
			answers[size_ans-1]+=m;
		}
		answers[size_ans++]=((-1)*sum)%m;
		if(answers[size_ans-1]<0)
			answers[size_ans-1]+=m;
		printf("%lld\n",(answers[z%size_ans] ));
	}
	return 0;
}
