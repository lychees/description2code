import math

def nCr(n,r):
    f = math.factorial
    return f(n) / f(r) / f(n-r)

t = int(raw_input())
while t>0:
	a = raw_input().split(' ')
	a = [int(x) for x in a]
	c = nCr(a[1],a[0]-1)
	x = a[1]-a[0]+1
	y = a[0]-1
	
	if(c!=1):
		print c,

	if(x!=0):
		print "x^%d"%x,

	if(y==1):
		print "y"
	elif(y==0):
		print ""
	else:
		print "y^%d"%y
		
	t=t-1
