#include<iostream>
using namespace std;
	int hcfOf(int a, int b){
		while (a != 0 && b != 0) {
	        if (a > b) {
	        	a = a%b;
	        } else {
	        	b = b%a;
	        }
	    }
		if (a == 0)
			return b;
		else
			return a;
	}
	
	int main(){
		int a,b,c;
		cin>>a>>b>>c;
		int numerator,denominator,hcf;
		
		if( a < b){
			b = (a+b) - (a=b);
		}
		if(a+b <= c){
			numerator = 1;
			denominator = 1;
		} else if(a>=c && b>=c){
			numerator = c*c;
			denominator = 2*a*b;
		} else if ( c >= b && c <= a){
			numerator = 2*c-b;
			denominator = 2*a;
		} else {
			numerator = 2*a*(c-a) + (a+b-c)*(c-b+a);
			denominator = 2*a*b;
		}
		hcf = hcfOf(numerator, denominator);
		numerator/=hcf;
		denominator/=hcf;
		cout<<numerator<<"/"<<denominator;
	}
