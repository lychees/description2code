#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int num_step;
    cin >> num_step;
  //  num_step--;
  if (num_step <0){
  	cout<<"ERROR"<<endl;
  }else{
    int dp[2][10000];
    dp[0][0] =0;
    dp[1][0]=0;
    dp[0][1]=1;
    dp[1][1]=1;
    dp[0][2]=2;
    dp[1][2]=2;
    for(int i =3; i<= num_step ; i++){
    	if (i == 3){
    	dp[0][i] = dp[0][i-1]+ dp[0][i-2]+ 1;
    	}else{
    	dp[0][i] = dp[0][i-1]+ dp[0][i-2]+ dp[0][i-3];
    	}
    	dp[1][i] = dp[1][i-1]+ dp[1][i-2];
    }
    std::cout<<"Jack-"<<dp[0][num_step]<<endl;
    std::cout<<"Jill-"<<dp[1][num_step]<<endl;
  }
    return 0;
}
