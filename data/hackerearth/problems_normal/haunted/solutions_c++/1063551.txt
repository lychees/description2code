#include <stdio.h>
#include <map>
using namespace std;

int main()
{
    int n,m,iter,maxTrophy;
    long long winner;
    scanf("%d%d",&n,&m);
    map <long long int , int> countStore;
    long long int ghost[n];
    maxTrophy=-1;
    winner=-1;
    for(iter=0;iter<n;iter++){
    	scanf("%lld",&ghost[iter]);
    	if(countStore[ghost[iter]]!=0)
    	countStore[ghost[iter]]++;
    	else
    	countStore[ghost[iter]]=1;
    	if(countStore[ghost[iter]]>maxTrophy){
    		maxTrophy = countStore[ghost[iter]];
    		winner= ghost[iter];
    	}
    	else if(countStore[ghost[iter]]==maxTrophy && ghost[iter]>winner)
    	winner=ghost[iter];
    	printf("%lld %d\n",winner,maxTrophy);
    }
    return 0;
}
