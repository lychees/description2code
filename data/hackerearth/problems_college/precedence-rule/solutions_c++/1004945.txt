#include<iostream>
using namespace std;
#include<cstdio>
#include<cstring>
int main()
{
int t;
cin>>t;
char a[40];
while(t--)
{
scanf("%s\n",a);
int l=strlen(a);
long long int ans=0;
int k=a[l-1]-48;
if(l==1)
ans=k;
else
{
int m=a[l-3]-48;
if(a[l-2]=='+')
ans=k+m;
else if(a[l-2]=='-')
ans=k-m;
else
ans=k*m;
for(int i=l-4;i>0;i=i-2)
{
	int k=a[i-1]-48;
	if(a[i]=='+')
ans=k+ans;
else if(a[i]=='-')
ans=ans-k;
else
ans=k*ans;
}
}
cout<<ans<<endl;
}
return 0;
}
