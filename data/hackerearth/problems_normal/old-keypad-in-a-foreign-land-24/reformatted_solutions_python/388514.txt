n = input()
freq = map(int, raw_input().split())
freq.sort(reverse=True)
k = input()
key = map(int, raw_input().split())
ans = 0
indx = 1
if (sum(key) < n):
    print -1
else:
    i = 0
    while i < n:
        j = 0
        while j < k and i < n:
            if (key[j] > 0):
                ans += freq[i]*indx
                key[j] -= 1
                i += 1
            j += 1
        indx += 1
    print ans
