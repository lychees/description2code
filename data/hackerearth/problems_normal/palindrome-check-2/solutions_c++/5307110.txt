#include<bits/stdc++.h>
using namespace std;

#define mod 1000000007
#define W while
#define LL long long int


int main()
{
	string str;
	LL i,flag;
	cin>>str;
	flag=0;
	for(i=0;i<str.length()/2;i++)
	{
		if(str[i]!=str[str.length()-i-1])
		{
			flag=1;
			break;
		}	
		
	}
	if(flag==1)
		cout<<"NO"<<endl;
	else
		cout<<"YES"<<endl;
	
	
	return 0;
}
