#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

int arr[105][105];
int cache[105][105];
int n;

int main() {
	// your code goes here
	int t,x;
	scanf("%d",&t);
	
	while(t--){
		scanf("%d %d",&n,&x);
		
		for(int i=1;i<=n;i++)
			for(int j=1;j<=n;j++)
				scanf("%d",&arr[i][j]);
		for(int i=0;i<=n;i++){
			cache[0][i] = 0;
			cache[i][0] = 0;
		}
		int maxx = -1;
		cache[0][0] = 0;
		for(int i=1;i<=n;i++){
			for(int j=1;j<=n;j++){
				cache[i][j] = max(cache[i-1][j],cache[i][j-1])+arr[i][j];
			}
		}
		
		int ans;
		ans = cache[n][n]-x;
		ans = ans>=0 ? ans : -1;
		printf("%d\n",ans);
	}
	
	return 0;
}
