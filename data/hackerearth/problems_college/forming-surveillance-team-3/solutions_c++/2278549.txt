#include <iostream>
#include<set>
#include<cstdio>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	int a,i;
    	set <int> s;
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&a);
    		s.insert(a);
    	}
    	printf("%d\n",s.size());
    }
    return 0;
}
