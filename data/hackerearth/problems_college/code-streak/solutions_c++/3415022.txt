#include<bits/stdc++.h>
#define ll long long 

using namespace std;

int main()
{
    long int t,n,a,b,c,ans;
    int v[1000000]={0};
    scanf("%d",&t);
    while(t--)
    {
    	cin>>n;
    	c=0;
    	ans=0;
    	int v[1000000]={0};
    	while(n--)
    	{
    		cin>>a>>b;
    		
    		if(b==1)
    		{
    			if(v[a]==0)
    			{
    				c++;
    				ans=max(c,ans);
    				v[a]++;
    				
				}
				
			}
			else if(b==0)
				c=0;
		}
		cout<<ans<<endl;
    
    }
    return 0;
}
