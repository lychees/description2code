#include <iostream>
using namespace std;

int gcd(int a, int b)
{
    for (;;) {
        if (a == 0) return b;
        b %= a;
        if (b == 0) return a;
        a %= b;
    }
}

int lcm(int a, int b)
{
	if(a == 0 || b == 0) {
		return 0;
	}
    int temp = gcd(a,b);

    return ((a / temp) * b);
}

int primeFactors(int n) {
	int count = 0;
    while (n%2 == 0) {
        n = n/2;
		count++;
    }

    for (int i = 3; i < n/2; i = i+2) {
		while (n%i == 0) {
            n = n/i;
			count++;
        }
    }
 
    if (n > 2) {
    	count++;
    }
    return count;
}

int distPrimeFactors(int n) {
	int count = 0;
	if(n%2 == 0) {
		count++;
	    while (n%2 == 0) {
	        n = n/2;
	    }
	}

    for (int i = 3; i < n/2; i = i+2) {
    	if(n%i == 0) {
    		count++;
    		while (n%i == 0) {
	            n = n/i;
	        }
    	}
        
    }
 
    if (n > 2) {
    	count++;
    }
    return count;
}


int main()
{
	int t;
	cin >> t;
	int a[t], b[t];
	for(int i = 0; i < t; i++) {
		cin >> a[i] >> b[i];
	}
	
	for(int i = 0; i < t; i++) {
		int l = lcm(a[i], b[i]);
		int fact = distPrimeFactors(l);
		if(primeFactors(fact) == 1) {
			cout << "Yes" << endl;
		} else {
			cout << "No" << endl;
		}
	}
    return 0;
}
