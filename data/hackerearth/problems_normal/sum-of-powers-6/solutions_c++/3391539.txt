#include <iostream>
#include<stdlib.h>
#include<math.h>

using namespace std;
long getlong();
int possbl(long val);
int equaltoval(int i,int j,long val);
int smequaltoval(int i,int j,long val);

int main()
{
	long tstcs=getlong();
	long val=0;
	while(tstcs>0){
		val=getlong();
		if(possbl(val))
		printf("Yes\n");
		else 
		printf("No\n");
		
		--tstcs;
	}
    return 0;
}

int equaltoval(int i,int j,long val){

	long tmpi=i*i,tmpj;
	
	for(;val>tmpi;tmpi=tmpi*i){
		tmpj=j*j;
	//	printf("%d %d\n",tmpi,tmpj);
		for(;val>tmpj;tmpj=tmpj*j){
		if((tmpi+tmpj)==val){
		
			return 1;
		}else if((tmpi+tmpj)>val){
			break;
		}
			
		}
	}
	return 0;
}

int smequaltoval(int i,int j,long val){
	int tmp=i;
	
	if(i==1&j==1){
		if(i+j==val)
		return 1;
		else return 0;
	}
	if(j!=1)
	tmp=j*j;
	for(;tmp<val;tmp=tmp*j){
		if(tmp+1==val){
			return 1;
		}
	}
	
	return 0;
}

int possbl(long val){
	int outrlmt=sqrt(val);
	for(int i=1;i<outrlmt+1;++i){
		for(int j=i;j<outrlmt+1;++j){
			if(j==1||i==1){
				if(smequaltoval(i,j,val)){
					return 1;
				}
			}
			else if(equaltoval(i,j,val)){
				return 1;
			}
			
		}
	}
	return 0;
}


long getlong(){
	char input=getchar_unlocked();
	long val=0;
	
	while(input<'0'||input>'9'){
		input=getchar_unlocked();
	}
	
	while(input>='0'&&input<='9'){
		val=val*10+input-'0';
		input=getchar_unlocked();
	}
	return val;
}
