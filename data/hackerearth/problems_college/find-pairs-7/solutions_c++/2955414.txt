#include <iostream>
using namespace std;
 
int main()
{
    int T,N,j,x;
    cin>>T;
    while(T--)
    {
    	long long int ans=0;
    	int R[1000001]={0};
    	int L[1000001]={0};
    	cin>>N;
    	for(long long int i=0;i<N;i++)
    	{
    		cin>>x;
    		if(x>=0)
    		R[x]++;
    		else
    		L[-x]++;
    	}
    	for(long long int i=0;i<=1000000;i++)
    	{
    		if(R[i]>=1)
    		ans+=((R[i])*(R[i]-1))/2;
    		if(L[i]>=1)
    		ans+=((L[i])*(L[i]-1))/2;
    	}
    	ans+=N;
    	cout<<ans<<endl;
    }
    return 0;
   }
