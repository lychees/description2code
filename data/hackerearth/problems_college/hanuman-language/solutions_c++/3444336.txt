#include <bits/stdc++.h>
using namespace std;

int main()
{
   	int t,m,k,n,i;
   	scanf("%d",&t);
   	getchar();
   	while(t--)
   	{
   		char a[101],b[101];
   		gets(a);
   		k=0;
   		n=strlen(a);
   		for(i=n-1;i>=0;i--)
   		{
   			if(a[i]!=' ')
   			{
   				b[k]=a[i];
   				k++;
   			}
   			else
   			{
   				for(m=k-1;m>=0;m--)
   					printf("%c",b[m]);
   				printf(" ");
   				k=0;
   			}
   		
   		}
   			for(m=k-1;m>=0;m--)
   					printf("%c",b[m]);
   		printf("\n");
   	}
    return 0;
}
