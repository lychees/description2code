long long gcd ( long long a, long long b )
{
    long long c;
    while ( a != 0 )
    {
        c = a;
        a = b%a;
        b = c;
    }
    return b;
}

bool isprime( long long n )
{
    long long i;
    if ( n < 4 ) return n != 0;
    if ( (n % 2) == 0 || (((n+1)%6) != 0 && ((n-1)%6) != 0) ) return 0;
    for ( i= 3; i*i <= n; i += 2 )
        if ( (n % i) == 0 ) return false;
    return true;
}

int calc(int n)
{
    //int a=3,b=4,c=5,count=0;
    std::queue<int> myqueue;
    myqueue.push(3);
    myqueue.push(4);
    myqueue.push(5);

    int count=0;
    while(!myqueue.empty())
    {
        int a=myqueue.front();
        myqueue.pop();
        int b=myqueue.front();
        myqueue.pop();
        int c=myqueue.front();
        myqueue.pop();
        if(c<=n)
        {
          int a2=2*a;
          int b2=2*b;
          int c2=2*c;
          int c3=3*c;
          myqueue.push((-1)*a+b2+c2);
          myqueue.push((-1)*a2+b+c2);
          myqueue.push((-1)*a2+b2+c3);
          myqueue.push(a+b2+c2);
          myqueue.push(a2+b+c2);
          myqueue.push(a2+b2+c3);
          myqueue.push(a-b2+c2);
          myqueue.push(a2-b+c2);
          myqueue.push(a2-b2+c3);
          count+=n/c;
        }


    }
    return count;
}
