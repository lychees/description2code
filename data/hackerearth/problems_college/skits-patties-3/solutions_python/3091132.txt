def pattiesno(n):
	if n<=5:
		return False
	if n>=0 and (n%6==0 or n%9==0 or n%20==0):
		return True
	else:
		return pattiesno(n-6) or pattiesno(n-9) or pattiesno(n-20)
	
for _ in xrange(input()):
	print pattiesno(input())
