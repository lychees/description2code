#include<bits/stdc++.h>
using namespace std;
#define il(x) scanf("%d",&x)
#define in(x) scanf("%lld",&x)
#define mod 1000000007
#define pb push_back
#define mp make_pair
typedef long long int ll;
int main()
{
    ll n,k,d,i,j,ans;
    in(n),in(k),in(d);
    vector<ll> v(n);
    for(i=0;i<n;i++)
        in(v[i]);
    vector<ll> sum;
    j=0,ans=0;
    for(i=0;i<n;i++)
    {
        if(i<k)
        {
            while(i<k)
                j=j+v[i],i++;
            if(j==0)
            {
                ans=-1;
                break;
            }
            else
                sum.pb(j);
        }
        if(i>=k)
        {
            j=j+v[i]-v[i-k];
            if(j==0)
            {
                ans=-1;
                break;
            }
            else
                sum.pb(j);
        }
    }
    if(ans==0)
    {
    for(i=0;i<n-k+1;i++)
    {
        j=sum[i];
        if(j<d)
        {
            i=i+k-1,ans++;
            if(v[i]==0)
                while(v[i]==0)i--;
            i=i+k-1;
            if(v[i]==0)
                while(v[i]==0)i--;
        }
    }
    }
    cout << ans << endl;
    return 0;
}
