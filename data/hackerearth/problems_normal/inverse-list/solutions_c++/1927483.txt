#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int n,i;
    	cin>>n;
    	long int a[n],temp[n];
    	for(i=1;i<=n;i++)
    	{
    		cin>>a[i];
    	}
    	int flag;
    	for(i=1;i<=n;i++)
    	{
    		int k=a[a[i]];
    		temp[k]=a[i];
    	}
    	for(i=1;i<=n;i++)
    	{
    		if(temp[i]==a[i])
    		flag=1;
    		else
    		{
    			flag=0;
    			break;
    		}
    	}
    	if(flag==0)
    	cout<<"not inverse"<<endl;
    	else
    	cout<<"inverse"<<endl;
    }
}
