#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int t;
    cin>>t;
    while(t--)
    {
    	char a[200];
    	long long int i;
    	cin>>a;
    	long long int len=strlen(a);
    	for(i=0;i<len;i++)
    	{
    		if(a[i]-48==1 || a[i]-48==0 || a[i]-48==8)
    		continue;
    		else
    		break;
    	}
    	if(i!=len)
    	cout<<"NO"<<endl;
    	else
    	{
    	for(i=0;i<len/2;i++)
    	{
    		if(a[i]!=a[len-i-1])
    		{
    			cout<<"NO"<<endl;
    			break;
    		}
    	}
    	if(i==len/2)
    	cout<<"YES"<<endl;
    	}
    }
    return 0;
}
