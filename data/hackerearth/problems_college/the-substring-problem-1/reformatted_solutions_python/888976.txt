for i in xrange(input()):
    n = input()
    p = []
    q = [[0, chr(97+j)] for j in range(26)]
    for j in xrange(n):
        p.append(raw_input())
        for k in set(p[j]):
            q[ord(k)-97][0] += 1
    q.sort()
    j = 1
    while j < 26:
        if q[-j-1][0] != q[-j][0]:
            break
        j += 1
    else:
        print q[0][1]
    print q[-j][1]
