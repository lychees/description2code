#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t;
	cin>>t;
	while(t--)
	{
		long long int n;
		cin>>n;
		long long int a[n],i;
		for(i=0;i<n;i++)
			cin>>a[i];
		sort(a,a+n);
		long long int s=0;
		for(i=1;i<n;i++)
			s+=a[0]+a[i];
		cout<<s<<"\n";
	}
	return 0;
}
