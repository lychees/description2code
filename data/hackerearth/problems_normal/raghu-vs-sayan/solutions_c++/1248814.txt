#include<algorithm>
#include <iostream>
using namespace std;

int main()
{
    int t;
    int a,b,n;
    cin>>t;
    while(t--){
    	cin>>a>>b>>n;
    	int i,arr[n];
    	for(i=0;i<n;i++)
    		cin>>arr[i];i=0;
    	sort(arr,arr+n);
    	int sa=0,sb=0;
    	while(sa<=a && i<n){
							if((sa+arr[i])<=a)
									sa+=arr[i];
							else
								break;
							i++;
				}i=0;
				while(sb<=b && i<n){
							if((sb+arr[i])<=b)
									sb+=arr[i];
							else
								break;
							i++;
				}
    	if(sa==sb)
    		cout<<"Tie\n";
    	else if(sa>sb)
    		cout<<"Raghu Won\n";
    	else
    		cout<<"Sayan Won\n";
    		
    }
    
    
    return 0;
}
