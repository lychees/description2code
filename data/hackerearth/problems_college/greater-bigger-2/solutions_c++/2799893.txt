#include <bits/stdc++.h>
using namespace std;

int main() {
long long int t,i,j,k,g,m,n;
cin>>t;
while(t--){
	g=0;
	string str;
	cin>>str;
	k=str.length();
	
	for(i=k-1;i>=0;i--)
	{
		for(j=i-1;j>=0;j--){
			if(str[i]>str[j])
			{
				//cout<<str[i]<<" "<<str[j]<<" ";
				swap(str[i],str[j]);
				
				for(m=j+1;m<k;m++)
				{
					for(n=m+1;n<k;n++)
					{
						if(str[m]>str[n])
						swap(str[m],str[n]);
					}
				}
				
				g=1;
				break;
			}
			
		}
		if(g==1)
		break;
	}
	if(g==0)
	cout<<"no answer\n";
	else
	cout<<str<<"\n";
	
	}
}
