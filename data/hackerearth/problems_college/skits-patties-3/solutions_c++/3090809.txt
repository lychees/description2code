#include<iostream>
#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#define MOD 1000000007
using namespace std;
int check(int a)
{	
	if(a<0)
	return 0;
	if(a==0)
	return 1;
	return (check(a-6)||check(a-9)||check(a-20));
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int N;
		cin>>N;
		if(check(N))
		{
			cout<<"True"<<endl;
		}
		else
		{
			cout<<"False"<<endl;
		}
	}
}
