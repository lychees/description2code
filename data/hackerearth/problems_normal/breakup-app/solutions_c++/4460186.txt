#include <iostream>
#include <string>
#include <cstdlib>
#include <cstdio>
using namespace std;

bool isDigit(char c){
	if(c >= 48 && c < 58){
		return true;
	}
	return false;
}

int main()
{
    int N;
    cin >> N;
    char msg[1000];
    int weight = 1;
    int a[31], max = 0, date;

    for(int i=0; i<31; ++i){
        a[i] = 0;
    }

    while(N--){
        cin.clear();
        cin.ignore();
    	cin.getline(msg, sizeof(msg));

    	if(msg[0] == 'G'){
    		weight = 2;
    	}else weight = 1;

    	for(int i = 3; msg[i] != '\0'; ++i){
    		if(isDigit(msg[i]) && isDigit(msg[i+1])){
				a[atoi(&msg[i])] += weight;
                i++;
    		}else if(isDigit(msg[i]) && !isDigit(msg[i+1])){
    			a[atoi(&msg[i])] += weight;
    		}
    	}
    }

    for(int i = 0; i<=30; ++i){
    	if(a[i] > max){
    		max = a[i];
    		date = i;
    	}
    }

    cout << ((date == 19 || date == 20)?"Date" : "No Date") << endl;

    return 0;
}
