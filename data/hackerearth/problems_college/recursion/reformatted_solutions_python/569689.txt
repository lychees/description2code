from itertools import permutations
tcs = int(input())


def do(x):
    stringx = ''
    for x in range(1, intx+1):
        stringx += str(x)

    perms = [''.join(p) for p in permutations(stringx)]

    final = ''
    for p in perms:
        final += str(p)
        final += ' '

    return final.strip()


for tc in range(tcs):
    intx = int(input())
    print(do(intx))
