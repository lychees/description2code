inline int Read()
{
	register int c=getchar();
	int x=0;
	for(;(c<48 || c>57);c=getchar());
	for(;c>47 && c<58;c=getchar())
		x=(x<<1)+(x<<3)+c-48;
	return x;
}

int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","o",stdout);
	int t=Read();
	while(t--)
	{
		int n=Read();
		if(n<3)
		{
			printf("-1\n");
			continue;
		}
		int a=n-n%3,flag=0;
		if(n%3==0)
		{
			for(int i=0;i<n;i++)
				putchar('5');
			putchar('\n');
			continue;
		}
		while(a>0)
		{
			int rem=n-a;
			if(rem%5==0)
			{
				flag=1;
				for(int i=0;i<a;i++)
					putchar('5');
				for(int i=a;i<n;i++)
					putchar('3');
				putchar('\n');
				a=0;
			}
			else
			{
				a-=3;
				if(a==0 && n%5==0)
				{
					for(int i=0;i<n;i++)
						putchar('3');
					putchar('\n');
					flag=1;
				}
			}
		}
		if(!flag)
			printf("-1\n");
	}
	return 0;
}
