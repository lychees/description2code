#include<bits/stdc++.h>
using namespace std;

const int N=21;
int a[N];

int main() {
  //freopen("a.cpp","r",stdin);
  int n;
  cin>>n;
  for(int i=0;i<n;i++) {
    cin>>a[i];
  }
  int tt;
  cin>>tt;
  vector<pair<long long,long long> > que;
  while(tt--) {
    long long foo,bar;
    cin>>foo>>bar;
    que.push_back(make_pair(foo,bar));
  }
  vector<long long> ans(que.size(),0);
  for(int i=1;i<(1<<n);i++) {
    vector<int> res;
    for(int j=0;j<n;j++) {
      if(i&(1<<j)) {
        res.push_back(a[j]);
      }
    }
    long long pr=1;
    for(int k=0;k<res.size();k++) {
      pr=(pr*res[k])/__gcd(pr,(long long)res[k]);
    }
    for(int k=0;k<que.size();k++) {
      long long temp=que[k].second/pr-(que[k].first-1)/pr;
      if(res.size()&1) {
        ans[k]+=temp;
      } else {
        ans[k]-=temp;
      }
    }
  }
  for(int i=0;i<ans.size();i++) {
    cout<<ans[i]<<endl;
  }
  return 0;
}
