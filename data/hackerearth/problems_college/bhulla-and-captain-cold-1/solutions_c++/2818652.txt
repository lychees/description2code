#include <iostream>
#include <cmath>
#include <cstring>
#include <cstdio>
#include <cstdlib>
 
using namespace std;
 
const int N = 100005;
int a[N][3];
 
void solve() {
    int n; cin>>n;
    for(int i = 0;i < n;i++) {
        scanf("%d %d %d",&a[i][0],&a[i][1],&a[i][2]);
    }
    for(int i = 1;i < n;i++) {
        a[i][0]+=min(a[i - 1][1],a[i - 1][2]);
        a[i][1]+=min(a[i - 1][0],a[i - 1][2]);
        a[i][2]+=min(a[i - 1][1],a[i - 1][0]);
    }
    cout<<min(a[n - 1][0], min(a[n - 1][1],a[n - 1][2]))<<endl;
}
int main() {
    int t; cin>>t;
    while(t--) {
        solve();
    }
}
