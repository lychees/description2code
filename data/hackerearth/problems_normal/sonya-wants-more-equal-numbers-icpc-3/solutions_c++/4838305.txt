#include<bits/stdc++.h>
using namespace std ;
int pref[1005];
int a[1005];
int dp[120][100001];
int n;
vector<int>ans ;
int main(){
	cin >> n ;
	for(int i=1;i<=n;++i) {
		cin>>a[i];
		pref[i]=a[i]+pref[i-1];
	}
	for(int i=0;i<=100000;++i)
	dp[n+1][i]=n+1;
	for(int i=1;i<=100000;++i){
		int ptr = n ;
		for(int j=n;j;--j){
			dp[j][i]=dp[j+1][i] ;
			while(pref[ptr]-pref[j-1]>i) --ptr ;
			if(pref[ptr]-pref[j-1] == i)
			dp[j][i]=ptr;
		}
	}
	
	int mx=0;
	
	for(int i=1;i<=100000;++i){
		int cnt =0 ;
		int ind = 1 ;
		while(ind<=n){
			if(dp[ind][i]!=n+1)
			cnt++;
			ind=dp[ind][i]+1;
		}
		if(cnt>mx){
			mx=cnt;
			ans.clear();
		}
		if(cnt==mx){
			ans.push_back(i);
		}
	}
	
	cout << mx << endl ;
	for(int i=0;i<ans.size();++i)
	cout << ans[i] << " " ;
	
	
}
