#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10;
const int md = 1e9 + 7;

int a[N], c[N];

inline void add(int &a, int b) {
	a += b;
	if (a >= md) {
		a -= md;
	}
}

inline void sub(int &a, int b) {
	a -= b;
	if (a < 0) {
		a += md;
	}
}

inline int mul(int a, int b) {
	return (long long) a * b % md;
}

int main() {
	int tt;
	scanf("%d", &tt);
	while (tt--) {
		int n;
		scanf("%d", &n);
		for (int i = 0; i < n; i++) {
			scanf("%d", &a[i]);
		}
		sort(a, a + n);
		fill(c, c + n + 1, 0);
		for (int i = 0; i < n; i++) {
			add(c[i + 1], c[i]);
			add(c[i + 1], a[i]);
		}
		int sum = 0;
		for (int i = 0; i < n; i++) {
			add(sum, c[n]);
			sub(sum, c[i]);
			int v = a[i];
			v = mul(v, n - i);
			sub(sum, v);
		}
		int res = mul(sum, a[n - 1]);
		printf("%d\n", res);
	}
	return 0;
}
