#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int main()
{
   	int t;
   	scanf("%d",&t);
   	while(t--)
	{
		long int n,q;
		scanf("%ld %ld",&n,&q);
		long int a[n];
		long int i;
		for(i=0;i<n;i++)
		{
			scanf("%ld",&a[i]);
		}
		sort(a,a+n);
		long int s = 0;
		for(i=1;i<n;i++)
		{
			s = s + a[i] - a[i-1];
		}
		printf("%ld\n",s*q);
	}
	return 0;
}
