#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    char str[100];
    int t=2000,i,n,flag;
    while(gets(str)!='\0'){
    	i=0;
    	n=strlen(str);
    	flag=0;
    	for(i=0;i<n;i++){
    		if(flag){
    			cout<<str[i];
    		}
    		else if(str[i]=='/'&&str[i+1]=='/'){
    			cout<<"//";
    			flag=1;
    			i++;
    		}
    		else if(str[i]=='-'&&str[i+1]=='>'){
    			cout<<".";
    			i++;
    		}
    		else
    			cout<<str[i];
    	}
    	cout<<endl;
    }
    return 0;
}
