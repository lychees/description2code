#include <iostream>
using namespace std;
 
int a[2000007];
 
int main()
{  int T,N,m1,m2,i;
    cin>>T;
    a[0]=0;
   
    while(T--)
    {
    	cin>>N>>m1>>m2;
      if(m1==1||m2==1)
       a[1]=1;
       else
       a[1]=0;
      for (int i=1;i<=N;i++)
       {
       	if(i>=m1&&!a[i-m1])
       	   a[i]=1;
       	else if (i>=m2&&!a[i-m2])
       	   a[i]=1;
       	else a[i]=0;
       }
       cout<<a[N]<<"\n";
    }
    return 0;
}
