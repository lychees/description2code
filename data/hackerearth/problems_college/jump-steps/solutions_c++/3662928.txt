#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    long long int k,c,i,j,l,m,n,nn,r;
    while(t--)
    {
    	cin >> n;
    	r=n;
    	l=0;
    	m=1;
    	c=0;
    	while(l!=r)
    	{
    		while(m<r-l)
    		{
    			m=m*2;
    		}
    		c++;
    		if(m>r-l)	m=m/2;
    		l=m+l;
    		m=1;
    	}
    	cout << c << endl;
    }
    return 0;
}
