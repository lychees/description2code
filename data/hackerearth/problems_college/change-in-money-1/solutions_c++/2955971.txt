#include<bits/stdc++.h>
using namespace std;

long long  a[100]={0},m=0,n=0;

long long  count( )
{
	long long i, j, x, y;
	long long table[n+1][m];
	
	for (i=0; i<m; i++)
		table[0][i] = 1;

	for (i = 1; i < n+1; i++)
	{
		for (j = 0; j < m; j++)
		{
			
			x = (i-a[j] >= 0)? table[i - a[j]][j]: 0;

			
			y = (j >= 1)? table[i][j-1]: 0;

		
			table[i][j] = x + y;
		}
	}
	return table[n][m-1];
}

{
	
		
		scanf("%lld %lld ",&n,&m);

}
