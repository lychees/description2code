#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int num;
	//vector items;
	cin>>num;
	string s;
	for (int i=0;i<num+1;i++)
	{
		getline(cin,s);
		size_t f = s.find("$");
		f++;
		bool zero=true;
		//cout<<s<<"  "<<f;
		if (f == 0)
		continue;
		cout<<"$";
		while (f<s.size())
		{
			//cout<<s;
			if (s[f] >= '1' && s[f] <= '9')
			{
				cout<<s[f];
				zero = false;
				//cout<<"ins 1";
			}
			else if (s[f] == '0')
			{
				if (!zero)
				cout<<s[f];
				//cout<<"ins 2";
			}
			else if (s[f] == ' ')
			{
				//cout<<"ins sp";
				//nothing
			}
			else
			{
				//cout<<"ins br";
				break;
			}
			f++;
		}
		if (zero)
		cout<<"0";
		cout<<"\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
