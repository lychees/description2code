#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;
vector<int> a;
int main(){

int t,n,k,p,x;
scanf("%d",&t);
while(t--){
    scanf("%d %d %d",&n,&k,&p);
    a.clear();
    for(int i=0;i<n;i++){
        scanf("%d",&x);
        a.push_back(x-p);
    }
    sort(a.begin(),a.end());
    int flag=0;
    for(int i=0;i<n;i++){
    int x=upper_bound(a.begin(),a.end(),a[i])-lower_bound(a.begin(),a.end(),a[i]-2*p);
   // printf("%d\n",x);
    if(x>=k||x<=-k){flag=1;break;}
    }
    if(flag==1)printf("NO\n");
    else printf("YES\n");

}
return 0;
}
