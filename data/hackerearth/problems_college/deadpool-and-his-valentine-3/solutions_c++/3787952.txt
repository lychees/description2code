#include<bits/stdc++.h>
using namespace std;
#define testcases int t;scanf("%d",&t);while(t--)
typedef long long int lld;
int main(){
	lld k,m,i,j,x,temp;
	scanf("%lld %lld",&k,&m);
	set<lld> bulb;
	for(i=0;i<k;i++){
		scanf("%lld",&x);
		for(j=0;j<x;j++){
			scanf("%lld",&temp);
			bulb.insert(temp);
		}
	}
	if(bulb.size()==m)
		printf("YES");
	else
		printf("NO");
return 0;
}
