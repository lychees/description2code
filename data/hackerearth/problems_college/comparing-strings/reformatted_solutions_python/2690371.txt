t = int(raw_input())
while t:
    a = str(raw_input())
    b = str(raw_input())
    l1 = len(a)
    l2 = len(b)
    f = 0
    if (l1 >= l2):
        for i in a:
            if i in b:
                f = 1
                break
    else:
        for i in b:
            if i in a:
                f = 1
                break
    if f:
        print 'YES'
    else:
        print 'NO'
    t -= 1
