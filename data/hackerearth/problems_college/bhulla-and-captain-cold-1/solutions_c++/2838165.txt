#include<cstdio>
using namespace std;
int A[1000001][3];

int main()
{
	int ntc; scanf("%d",&ntc);
	while(ntc--)
	{
		int N;
		scanf("%d",&N);
		for(int i=0;i<N;i++) scanf("%d %d %d",&A[i][0],&A[i][1],&A[i][2]);
		long long a0,a1,a2;
		a0=A[0][0]; a1=A[0][1]; a2=A[0][2];
		for(int i=1;i<N;i++)
		{
			long long na0,na1,na2;
			if(a0<a1) na2=a0+A[i][2]; else na2=a1+A[i][2];
			
			if(a0<a2) na1=a0+A[i][1]; else na1=a2+A[i][1];
			
			if(a1<a2) na0=a1+A[i][0]; else na0=a2+A[i][0];
			
			a0=na0; a1=na1; a2=na2;
			
		}
		//printf("%lld %lld %lld\n",a0,a1,a2);
		long long ans=a0; if(ans>a1) ans=a1; if(ans>a2) ans=a2; 
		
		printf("%lld\n",ans);
	}
}
