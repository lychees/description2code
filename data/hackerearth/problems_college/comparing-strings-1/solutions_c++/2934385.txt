#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s,p;
    	cin>>s>>p;
    	int l=s.length(),r=p.length(),i=0,ch=0;
    	while(i!=l&&i!=r)
    	{
    		if(s[i]==p[i])
    		{
    			cout<<"YES"<<endl;
    			ch=1;
    			break;
    		}
    		i++;
    	}
    	if(ch==0)
    	cout<<"NO"<<endl;
    }
    return 0;
}
