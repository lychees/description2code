//#define test
ll n, x, y, a, b, c, d;
vector<pair<pair<ll, ll> , pair<ll, ll> > > v;
int chk[1000];

int foo() {
    int ans = -1;
    double h = -1;
    for(int i = 0; i < n; i++) {
        if(chk[i] || v[i].f.f > x || v[i].s.f < x) continue;
        double h1 = v[i].f.s + (v[i].s.s - v[i].f.s) * 1.0 * (x - v[i].f.f) / (v[i].s.f - v[i].f.f);
        if(h1 > y + 1e-9) continue;
        if(h1 > h) {
            h = h1;
            ans = i;
        }
    }
    if(ans >= 0) chk[ans] = 1;
    return ans;
}


int main() {
    std::ios::sync_with_stdio(false);
#ifdef test
	freopen("a.in","rt",stdin);
	freopen("a.out","wt",stdout);
#endif
    cin >> x >> n;
    y = 1e9;
    for(int i = 0; i < n; i++) {
        cin >> a >> b >> c >> d;
        v.pb(mp(mp(a, b), mp(c, d)));
    }
    while(1) {
        int id = foo();
        if(id == -1) break;
        if(v[id].f.s < v[id].s.s) {
            x = v[id].f.f;
            y = v[id].f.s;
        }
        else {
            x = v[id].s.f;
            y = v[id].s.s;
        }
    }
    cout << x << endl;
    return 0;
}
