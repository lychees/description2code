#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include<math.h>
using namespace std;
int main()
{
	   int t;
	   cin>>t;
	   while(t--)
	   {
	   	int n;
	   	int a[10000];
	   	int ans=0;
	   	cin>>n;
	   	  
	   	for(int i=0;i<n;i++)
	   	{
	   		cin>>a[i];
	   		ans=ans^a[i];
	   	}
	   	cout<<ans<<endl;
	   }
}
