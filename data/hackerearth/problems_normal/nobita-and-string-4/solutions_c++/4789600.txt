#include<iostream>
#include<stdio.h>
#include<string>
#include<vector>
#define MAX 100000

using namespace std;

int main()
{
	
	int t;
	cin>>t;
	getchar();
	while(t--)
	{
		string s;
		getline(cin,s);
		//string word[MAX];
		vector<string> word;
		int start=0,k=0,i;
		for(i=0;i<s.length();i++)
		{
			if(s[i]==' ')
			{
				word.push_back(s.substr(start,i-start));
				//cout<<start<<" "<<i<<" "<<s.substr(start,i-start)<<endl;
				start=i+1;
				k++;
			}
		}
		word.push_back(s.substr(start,i-start));
		//cout<<s.substr(start,i);

	
		int left=0,right=k;
		//cout<<word.size();

		while(left<right)
		{
			string temp=word[left];
			word[left]=word[right];
			word[right]=temp;
			left++;
			right--;
		}
	 	for(int j=0;j<=k;j++)
			cout<<word[j]<<" ";
		cout<<endl;

	}
	return 0;
}
