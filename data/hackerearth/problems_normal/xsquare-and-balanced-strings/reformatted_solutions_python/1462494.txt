# print 'Hello World!'

t = int(raw_input())

while t:
    s = raw_input()

    m = [0]*26
    end = 0
    for i in xrange(26):
        m[i] = s.count(chr(i+97))
        if m[i] % 2 != 0:
            end = 1
            break
    if end:
        print -1
    else:
        print 1

    t -= 1
