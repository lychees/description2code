#include<bits/stdc++.h>
#define s(a) scanf("%lld",&a)
#define ss(a) scanf("%s",a)
using namespace std;
typedef long long int ll;
ll n;
ll inp[100005];
bool present[100005][4];
int prime[4]={2,3,5,7};
ll cnt[20];
inline bool contains(ll num,int dig)
{
    while(num)
    {
        if(num%10 == dig)return true;
        num = num/10;
    }
    return false;
}
inline void mapping(int num)
{
    for(int s=num;s!=0;s=(s-1)&num)
          cnt[s]++;
}
int main()
{
      int i,j;
      memset(present,0,sizeof(present));
      memset(cnt,0,sizeof(cnt));
      s(n);
      for(i=0;i<n;i++)
      {
            s(inp[i]);
            for(j=0;j<4;j++)
                if(contains(inp[i],prime[j]))
                      present[i][j]=true;
      }
    long long int res=0,temp2;
    int temp;
    for(i=0;i<n;i++)
    {
          temp=0;
          for(j=0;j<4;j++)
          {
                if(present[i][j])temp=(temp<<1) + 1;
                else temp=temp<<1;
          }
          mapping(temp);
    }
    int mult[5]={0,1,-1,1,-1};
    for(i=1;i<16;i++)
    {
        temp = __builtin_popcount(i);
        temp2 = (cnt[i]*(cnt[i]-1)*(cnt[i]-2))/6;
        res = res+temp2*mult[temp];
    }
    printf("%lld",res);
    return 0;
}
