#include <bits/stdc++.h>
using namespace std;
#define pii pair<int,int>
set <pii> dijk;
int dist[(1<<20)];
int main()
{
	// freopen("input.txt", "r", stdin);
	int n;
	scanf("%d", &n);
	char inp[n+5];
	scanf("%s", inp);
	int x=0;
	memset(dist, -1, sizeof dist);
	for (int i = 0; i < n; ++i)
	{
		if(inp[i]=='W')
			x^=(1<<i);
	}
	dist[0]=0;
	dist[(1<<n)-1]=0;
	dijk.insert(pii(0,0));
	dijk.insert(pii(0,(1<<n)-1));
	while(!dijk.empty())
	{
		pii top=*dijk.begin();
		dijk.erase(top);
		int v=top.second,d=top.first;
		for (int i = 0; i < n-1; ++i)
		{
			int two=0;
			two^=(1<<i);
			two^=(1<<(i+1));
			int v2=(v^two);
			if(dist[v2]==-1 || dist[v2] > dist[v]+1)
			{
				if(dist[v2] != -1)
					dijk.erase(pii(dist[v2],v2));
				dist[v2]=1+dist[v];
				dijk.insert(pii(dist[v2],v2));
			}
			if(i < n-2)
			{
				two^=(1<<(i+2));
				v2=(v^two);
				if(dist[v2]==-1 || dist[v2] > dist[v]+1)
				{
					if(dist[v2] != -1)
						dijk.erase(pii(dist[v2],v2));
					dist[v2]=1+dist[v];
					dijk.insert(pii(dist[v2],v2));
				}
			}
		}
	}
	printf("%d\n", dist[x]);
	return 0;
}
