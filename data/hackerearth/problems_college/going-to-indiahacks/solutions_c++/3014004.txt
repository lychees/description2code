#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

const int lmt = 1009;
lli dp[lmt][lmt];

void pre(){
    dp[0][0] = 1;
    for(int i=1;i<lmt;i++){
        for(int j=1;j<lmt;j++){
            dp[i][j] = dp[i-1][j] + dp[i][j-1] + dp[i-1][j-1];
            dp[i][j] %= mod;
        }
    }
}

int main(){
    pre();
    int t,a,b,c,d;
    scanf("%d",&t);
    while(t--){
        scanf("%d %d %d %d",&a,&b,&c,&d);
        c -= abs(a-1);
        d -= abs(b-1);
        printf("%lld\n",dp[c][d]);
    }
    return 0;
}
