#include<iostream>
using namespace std;
int main()
{
    int N;
    cin>>N;
    while(N--)
    {
        int S;
        cin>>S;
        long long int count=0,i,j;
        for(i=1;i<=S;i++)
        {
            j=i*i;
            while(j<=S)
            {
                count++;
                j= j+i;
            }
        }
        cout<<count<<endl;
    }
    return 0;
}
