#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int rec(int n)
{
	int i;
	for(i=2;i<=n/2;i++)
	{
		if(n%i==0)
		  return 0;
	}
	return 1;
}
int main()
{
    int n;
    cin >> n;
    vector<int> vec;
    int temp;
    for(int i=0;i<n;i++)
    {
    	cin >> temp;
    	vec.push_back(temp);
    }
    sort(vec.begin(), vec.end());
    reverse(vec.begin(),vec.end());
    vector<int>::iterator it;
    for(it=vec.begin();it!=vec.end();it++)
    {
    	if(rec(*it)==1)
    	{
    	   cout << *it << endl;
    	   break;
    	}
    }
    int t = *it;
    int count = 0;
    for(it=vec.begin();it!=vec.end();it++)
    {
    	if(*it==t)
    	{
    	   count++;
    	}
    	else if(*it<t){
    		break;
    	}
    }
    cout << count << endl;
    return 0;
}
