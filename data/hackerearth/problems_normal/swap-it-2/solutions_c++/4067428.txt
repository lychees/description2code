#include<bits/stdc++.h>
using namespace std;
void newarr(int arr[],int j,int idx)
{
    int temp;
    while(idx!=j)
    {
        swap(arr[idx],arr[idx-1]);
        idx--;
    }
}
int main()
{
    ios::sync_with_stdio(false);
    int rem,j,mi,t,n,k,i,idx;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        int arr[n+1];
        for(i=0;i<n;i++)
            cin>>arr[i];
        rem=k;
        for(j=0;j<n;j++){
                mi=INT_MAX;
            for(i=j;i<=rem+j && i<n;i++)
            {
                if(arr[i]<mi)
                {
                    idx=i;
                    mi=arr[i];
                }
            }
            rem-=(idx-j);
            newarr(arr,j,idx);
            if(rem==0)
                break;
        }
        for(i=0;i<n;i++)
            cout<<arr[i]<<" ";
        cout<<"\n";
    }
    return 0;
}
