#include <bits/stdc++.h>
#define INF 10000007

using namespace std;

int inp[100005];
int tree[1000005];

void build(int node, int a, int b) {
    if(a > b) return;

  	if(a == b) {
        tree[node] = inp[a];
		return;
	}

	build(node*2, a, (a+b)/2);
	build(node*2+1, 1+(a+b)/2, b);

	tree[node] = min(tree[node*2], tree[node*2+1]);
}

int query(int node, int a, int b, int i, int j) {

	if(a > b || a > j || b < i) return INF;

	if(a >= i && b <= j)
		return tree[node];

	int q1 = query(node*2, a, (a+b)/2, i, j);
	int q2 = query(1+node*2, 1+(a+b)/2, b, i, j);

	int res = min(q1, q2);

	return res;
}

int main()
{
    int n, qu, i, l, r;

    scanf("%d %d", &n, &qu);

    for (i = 0; i < n; i++)
        scanf("%d", &inp[i]);

    build(1, 0, n-1);

    while (qu--) {
        scanf("%d %d", &l, &r);

        printf("%d\n", query(1, 0, n-1, l, r));
    }

    return 0;
}
