#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1e6+3;
bool dp[MAXN];
long long power(long long a, long long b) {
	long long ret=1;
	while(b) {
		if(b&1)
			ret=ret*a;
		a=a*a;
		if(ret>=MAXN or (a>=MAXN and b!=1))
			return (long long)1e9+7;
		b>>=1;
	}
	return ret;
}		
void pre() {
	for(int i=1;;i++) {
		long long val=power(i,i);
		if(val>=MAXN)
			break;
		dp[val]=true;
	}
}
int main() {
	int t;
	cin>>t;
	memset(dp,false,sizeof(dp));
	pre();
	for(int i=1;i<MAXN;i++) {
		if(!dp[i]) {
			int cc=i;
			while(!dp[i] and cc!=0) {
				int val=cc%10;
				if(i-val*val>=1)
					dp[i]|=dp[i-val*val];
				cc/=10;
			}	
		}
	}
	while(t--) {
		int n;
		scanf("%d",&n);
		printf("%s\n",dp[n]?"Yes":"No");
	}
	return 0;
}
