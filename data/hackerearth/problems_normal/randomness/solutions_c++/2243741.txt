#include <stdio.h>
#include <iostream>
#include <cmath>
using namespace std;
int gcd(int a , int b)
{
   if(b==0) return a;
   a%=b;
   return gcd(b,a);
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int a,b,c;
		cin>>a>>b>>c;
		int m = max(a,b);
		int n = min(a,b);
		a=m, b=n;
		if(c<=a && c<=b){
			c=c*c;
			a=2*a*b;
			int g = gcd(a,c);
			a/=g;
			c/=g;
			cout<<c<<"/"<<a<<endl;
		}
		else if(b<=c && c<=a){
			c=c*c-(c-b)*(c-b);
			a=2*a*b;
			int g = gcd(a,c);
			a/=g;
			c/=g;
			cout<<c<<"/"<<a<<endl;
		}
		else if(a<c && c<b){
			c=c*c-(c-a)*(c-a);
			a=2*a*b;
			int g = gcd(a,c);
			a/=g;
			c/=g;
			cout<<c<<"/"<<a<<endl;
		}
		else if(a<c && b<c && c<(a+b)){
			c=c*c-(c-b)*(c-b)-(c-a)*(c-a);
			a=2*a*b;
			int g = gcd(a,c);
			a/=g;
			c/=g;
			cout<<c<<"/"<<a<<endl;
		}
		else cout<<"1/1"<<endl;
	}
	return 0;
}
