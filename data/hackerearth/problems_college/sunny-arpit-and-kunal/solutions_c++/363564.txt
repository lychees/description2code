#include <iostream>
using namespace std;

int main()
{
	int a[3],i,j,temp,t;
    cin>>t;
    while(t--)
    {
    	cin>>a[0]>>a[1]>>a[2];
       for(i=0;i<2;i++)
       {
       	for(j=i+1;j<3;j++)
       	{
       		if(a[i]>a[j])
       		{
       			temp=a[j];
       			a[j]=a[i];
       			a[i]=temp;
       		}
       	}
       }
       if((a[1]-a[0])>(a[2]-a[1]))
       cout<<a[1]-a[0]-1<<endl;
       else
       cout<<a[2]-a[1]-1<<endl;
        
    }
    return 0;
}
