import sys
begin = True
total = 0
count = 0
mod_val = pow(10, 9) + 7
for line in sys.stdin:
    line = line.strip()
    if begin:
        begin = False
        total = int(line)
        continue
    count += 1
    players = int(line)
    
    if players is 0:
        print '0', '0'
    n = players - 1 
    max_winnings = n * (n + 1) * (2 * n + 1) / 6
    matches = players * (players - 1) / 2
    min_winnings = pow(matches / players, 2) * players
    print min_winnings % mod_val,max_winnings % mod_val
    if count >= total:
        break
