#include <iostream>
using namespace std;

int main()
{

    	cin>>n>>m;
    	dup=n;

    	for(i=0;i<100;i++){

    		while(dup!=0){
    			r=dup%10;
    			totat+=(r*r);
    			dup=dup/10;
    		}
    		dup=totat;
         arr[i]=dup;
    	}

      	if(arr[i]==arr[i+8]){
      		count++;
      	}

      	count=0;
      	if(count==8){
      		pos=i-8+1;
      		break;
      	}
      }
      m=m-pos;
      rem=m%8;
      if(rem==0)
      rem=8;
      for(i=pos;i<pos+8;i++){
      	aa[l++]=arr[i];
      }
    cout<<aa[rem-1]<<endl;
    }
    else{
    	for(i=0;i<m;i++){
    		totat=0;
    		while(dup!=0){

}
