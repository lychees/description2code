#include<iostream>
#include<algorithm>
#include<string>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while (t--)
    {
        int k;
        cin>>k;
        string s;
        cin>>s;
        string ans="";
        char c;
        for(int i=0;i<s.length();i++)
        {

            c=s[i]-k;
            if(c<65)
                c+=26;
            ans.push_back(c);

        }
        for(int i=0;i<ans.length();i++)
            cout<<ans[i];
        cout<<endl;
    }
 return 0;
}
