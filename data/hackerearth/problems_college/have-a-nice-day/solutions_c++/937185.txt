#include <iostream>
using namespace std;

int max(int a[],int n)
{
	long i;
	int max=0;
	
	for(i=0;i<n;i++)
		if(a[i]>max)
			max = a[i];
			
		return max;
}

int count_sort(int a[],int n,int k)
{
	int i;
	int count[k];
	int out[n];
	
	for(i=0;i<k;i++)
		count[i]=0;
	
	for(i=0;i<n;i++)
		count[a[i]]++;
		
	for(i=1;i<k;i++)
		count[i]=count[i-1]+count[i];
		
	for(i=n-1;i>=0;i--)
		out[--count[a[i]]] = a[i];
	for(i=0;i<n;i++)
		a[i]=out[i];
	
}

int main()
{
    int i,j,tno,N,sum,k;
    
    cin>>tno;
    
    for(i=1;i<=tno;i++)
    {
    	cin>>N;
    	
    	int *m = new int[N];
    	int *f = new int[N];
    	
    	for(j=0;j<N;j++)
    		cin>>m[j];
    		
    	for(j=0;j<N;j++)
    		cin>>f[j];
    		
    	
    	k = max(m,N)+1;
    	
    	count_sort(m,N,k);
    	
    	k = max(f,N)+1;
    	
    	count_sort(f,N,k);
    	
    	sum =0;
    	
    	for(j=0;j<N;j++)
    		sum= sum + m[j]*f[j];
    		
    	cout<<sum<<endl;
    }
    return 0;
}
