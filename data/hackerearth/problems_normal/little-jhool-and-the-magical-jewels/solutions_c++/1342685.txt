#include <iostream>
#include<cstring>
using namespace std;

int main()
{
   int t,c1,c2,c3,c4,j;
    cin>>t;
    char str[100];
    while(t--)
    {
    	cin>>str;
    	c1=0;c2=0;c3=0;c4=0;
    	for(j=0;j<strlen(str);j++)
    	{
    	if(str[j]=='r')
    	{
    		c1=c1+1;
    	}
    	else if(str[j]=='u')
    	{
    		c2=c2+1;
    	}
    		else if(str[j]=='b')
    	{
    		c3=c3+1;
    	}
    	else if(str[j]=='y')
    	{
    		c4=c4+1;
    	}
    	}
        if(c1<=c2 && c1<=c3 && c1<=c4)
        {cout<<c1<<endl;}
    	else if(c2<=c1 && c2<=c3 && c2<=c4)
        {cout<<c2<<endl;}
        else if(c3<=c1 && c3<=c2 && c3<=c4)
        {cout<<c3<<endl;}
       else
        {cout<<c4<<endl;}
    }
    return 0;
}
