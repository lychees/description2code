n = input()

mlst = []
sdic = {}
for i in xrange(n):
    # mlst = map(int, raw_input().split())
    mlst = sorted(raw_input().split())
    str1 = ''.join(mlst)
    if sdic.has_key(str1):
        sdic[str1] = False
    else:
        sdic[str1] = True

count = 0
for val in sdic.values():
    if val:
        count += 1

print count
