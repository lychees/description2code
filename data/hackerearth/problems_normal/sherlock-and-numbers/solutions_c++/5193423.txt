int main(){

	int T;
	int n, p, k;
	scanf("%d",&T);
	int x;
	for (int t = 0; t < T; ++t){
		
		scanf("%d%d%d", &n, &k, &p);
	
		for (int i = 0; i < k; ++i){
			scanf("%d",&x);
			if (x <= p)
				p++;
		}
		if (p > n)
			{printf("-1\n");}
		else
			{printf("%d\n", p);}
	}
	return 0;
}
