#include<iostream>
#include<vector>
#include<string>
#include<cmath>
using namespace std;
int main()
{
	int a,b;
	cin>>a;
	string c;
	int count ;
	for(int i=0;i<a;i++)
	{
        count=0;
		cin>>c;
		int q=c.length();
		int ar[q];
		for(int j=0;j<q;j++)
		{
			ar[j]=c[j];
		}
		for(int j=1;j<q;j++)
		{
			if(abs(ar[j]-ar[j-1])==abs(ar[q-1-j]-ar[q-j]))
			{
				count++;
			}
		}
		if(count==q-1)
		cout<<"Funny"<<endl;
		else
		cout<<"Not Funny"<<endl;	
	}
	return 0;
}
