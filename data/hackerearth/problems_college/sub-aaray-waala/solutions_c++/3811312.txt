#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll arr[100100];
map <ll,int> m;
ll M=1e9+7;
ll fexpo (ll a,ll b)
{
	ll res=1;
	while (b)
	{
		if (b%2)
		{
			res*=a;
			if (res>M)
			res%=M;
		}
		a*=a;
		if (a>M)
		a%=M;
		b/=2;
	}
	return res;
}
int main()
{
	ll ans,res,i,j,k,n,t;
	cin>>t;
	while (t--)
	{
	cin>>n;
	for (i=0;i<n;i++)
	cin>>arr[i];
	j=0;
	for (i=0;i<n;i++)
	{
		if (m.find(arr[i])!=m.end())
		continue;
		j++;
		m[arr[i]]=1;
	}
	cout<<fexpo (n,j)<<endl;
	m.clear();
	}
	return 0;
}
