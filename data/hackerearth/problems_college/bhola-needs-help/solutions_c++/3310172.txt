#include<bits/stdc++.h>
using namespace std;
int gcd(int a,int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
vector<int>v;
int main()
{
	int a,b;
	scanf("%d %d",&a,&b);
	int q;
	int c=gcd(a,b);
	int i,j;
	for(i=1;i<=sqrt(c);i++)
	{
		if(c%i==0)
		{
			if(i==(c/i))
			v.push_back(i);
			else
			{
				v.push_back(i);
				v.push_back(c/i);
			}
		}
	}
	sort(v.begin(),v.end());
	scanf("%d",&q);
	while(q--)
	{
		int l,r;
		scanf("%d %d",&l,&r);
		int x,y;
		x=lower_bound(v.begin(),v.end(),l)-v.begin();
		y=upper_bound(v.begin(),v.end(),r)-v.begin();
		y--;
		if(y>=0&&y>=x)
		{
			printf("%d\n",v[y]);
		}
		else
		printf("-1\n");
	}
	return 0;
}
