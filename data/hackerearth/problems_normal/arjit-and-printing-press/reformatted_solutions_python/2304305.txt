t = input()
a = {}

while (t):
    t -= 1
    a = raw_input()
    b = raw_input()
    a = list(a)
    b = sorted(b)
    for i in range(len(a)):
        if b == []:
            break
        if b[0] < a[i]:
            a[i] = b[0]
            b.remove(b[0])
    k = ""
    for i in a:
        k = k+i
    print k
