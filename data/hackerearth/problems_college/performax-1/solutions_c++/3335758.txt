#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,c=0;
	long long d,a,in,r;
	cin>>n>>d;
	vector<long long> v;
	for(int i=0;i<n;i++)
	{
		cin>>a;
		v.push_back(a);
	}
	for(int i=1;i<n;i++)
	{
		if(v[i]<=v[i-1])
		{
			in=(v[i-1]+1)-v[i];
			r=in/d;
			if(in%d!=0)
			r++;
			c+=r;
			v[i]+=(r*d);
		}
	}
	cout<<c<<"\n";
}
