#include<bits/stdc++.h>
using namespace std;
#define gc getchar
#define ll long long
#define v vector
#define pr pair<int,int>
#define sd second
#define ft first
#define pb push_back
int  fr()
{
    char ch;
    int t=0,flag=0;
    ch=gc();
    while(ch<33)
        ch=gc();
    if(ch=='-')
        {
            ch=gc();
            flag=1;
        }
    while(ch>33)
    {
        t=t*10+ch-'0';
        ch=gc();
    }
    if(flag)
        return -t;
    else
        return t;
}
int main()
{
    int i,j,k,l,m,n,t,ans,en;
    int a[10000];
    t=fr();
    while(t--)
    {
        m=fr();
        n=fr();
        for(i=0;i<n;i++)
                a[i]=fr();
            sort(a,a+n);
        ans=0;
        for(i=0,en=n-1;i<n&&i<=en;)
        {
            if(i==en)
                ans++;
            else if(a[i]+a[en]<=m)
            {
                i++;
                ans++;
            }
            else if(a[en]<=m)
                ans++;
            en--;
        }
        printf("%d\n",ans);
    }
    return 0;
}
