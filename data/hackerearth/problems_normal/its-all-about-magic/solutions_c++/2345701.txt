#include <iostream>
using namespace std;

int main()
{
    long long int zeon[2][91]={0};
    zeon[1][0]=1;
    for(int i=1;i<91;i++)
    {
    	zeon[0][i]=zeon[1][i-1]+zeon[0][i-1];
    	zeon[1][i]=zeon[0][i-1];
    }
    int t;
    cin>>t;
    while(t--)
    {
    	int m;
    	cin>>m;
    	cout<<zeon[1][m]<<' '<<zeon[0][m]<<endl;
    }
    return 0;
}
