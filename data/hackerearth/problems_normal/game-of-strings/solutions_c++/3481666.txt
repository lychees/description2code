#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define fi first
#define se second
#define mp make_pair
#define pb push_back

string s1,s2; 
int n1,n2;
char a[]={'a','e','i','o','u'};

bool check(char ch)
{
	for(int i=0;i<5;i++)
		if(a[i]==ch)
			return false;
	return true;
}

int dp[5001][5001];

int lcs(int i,int j)
{
	if(n1-i-1 < 0 || n2-j-1 < 0)
		return 0;
		
	if(dp[n1-i-1][n2-j-1]!=-1)
		return dp[n1-i-1][n2-j-1];
	int result=0;	
	if((s1[n1-i-1]==s2[n2-j-1]) && check(s1[n1-i-1]))
		result = lcs(i+1,j+1)+1;
	else 
		result = max(lcs(i+1,j),lcs(i,j+1));
		
	dp[n1-i-1][n2-j-1]=result;
	return result;
}

int main()
{
	int i,j,k;
	
	memset(dp,-1,sizeof(dp));
	cin>>s1>>s2;
	n1=s1.length();
	n2=s2.length();
	
	int x=lcs(0,0);
	
	cout<<x<<endl;
	
	return 0;
}
