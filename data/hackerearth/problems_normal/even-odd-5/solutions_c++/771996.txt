#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
    	long int r,c;
    	cin >> r >> c;
    	
	
		long int n_odd=0,sum_odd=0;
		long int n_even=0,sum_even=0;
		
		if(r%2==0){
			n_even = (r-2)/2+1;
			n_odd = (r-2)/2+1;
		}else{
			n_even = (r-3)/2+1;
			n_odd = (r-1)/2+1;
		}
//    		cout << n_even << " " << n_odd << endl;
		
		sum_even = n_even*(n_even+1);
		sum_odd = n_odd*n_odd;
		
    	if(r <= c){
    		cout << sum_even << " " << sum_odd << endl;  
    	}else{
    		for(int i=c+1; i<=r; i++){
    			if(i&1)
    				sum_odd -= i-c;
    			else
    				sum_even -= i-c;
    		}
    		cout << sum_even << " " << sum_odd << endl;
    	}
    }
    return 0;
}
