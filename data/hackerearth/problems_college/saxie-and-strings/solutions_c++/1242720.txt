#include <iostream>
#include <string>
using namespace std;

int main()
{
    int T;
    string s;
    cin>>T;
    while(T--)
    {
    	cin>>s;
    	string rs(s.rbegin(), s.rend());
    	
    	if (s.compare(rs) == 0)
    		cout<<"YES"<<endl;
    	else
    		cout<<"NO"<<endl;
    	
    }
    return 0;
}
