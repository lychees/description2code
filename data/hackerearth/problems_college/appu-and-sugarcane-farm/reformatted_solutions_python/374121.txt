n = int(raw_input())
x = [int(i) for i in raw_input().split()]
p = 1
for i in range(n-1):
    if x[i] > x[i+1]:
        a, b = x[i], x[i+1]
    else:
        b, a = x[i], x[i+1]
    if a % b:
        p = 0
        break
    else:
        y = a/b
        if (y & (y-1)) != 0:
            p = 0
            break
if p:
    print "YES"
else:
    print "NO"
