#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	int arr[n],sum[n];
    	for(int i=0;i<n;i++){
    		cin>>arr[i];
    		sum[i]=i>=0?sum[i-1]+arr[i]:arr[i];
    	}
    	int flag=0,su=0;
    	for(int i=n-1;i>=0;i--){    		
    		if(su==sum[i]-arr[i]){
    			flag=1;
    			break;
    		}
    		su+=arr[i];
    	}
    	if(flag)cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
    return 0;
}
