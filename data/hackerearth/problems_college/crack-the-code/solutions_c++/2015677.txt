#include <bits/stdc++.h>
using namespace std;
int main()
{
    long long int fib[100000];
    fib[0]=0;
    fib[1]=1;
    for(int i=2;i<=100000;i++)
    {
		fib[i]=fib[i-1]+fib[i-2];
    }
    int t;
    cin>>t;
    int x,y;
    while(t--)
    {
    	cin>>x>>y;
    	int count=0;
    	for(int i=0;i<=100000;i++)
    	{
    		if(fib[i]>=x && fib[i]<=y)
    		{
    			count++;
    		}
    		if(fib[i]>y)
    		{
    			break;
    		}
    	}
    	if(count%2==0)
    	{
    		cout<<"VALID"<<endl;
    	}
    	else
    	{
    		cout<<"INVALID"<<endl;
    	}
    }
}
