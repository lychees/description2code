#include <iostream>
#include <map>
#include <cstring>

using namespace std;

int main()
{
	int i, j, k, l, t, n, mx, index;
	int a[100001];
	int b[100001];
	int dp[100001];

	cin>>t;

	while (t--)
	{
		cin>>n;
		
		map <int,int> m;
		memset(b, -1, sizeof(b));
		memset(dp, 0, sizeof(dp));
		
		for (i = 0; i < n; i++) {
			cin>>a[i];

			if (m[a[i]] == 0) {
				m[a[i]] = i+1;
			}
			else {
				b[m[a[i]]-1] = i - m[a[i]] + 1;
				m[a[i]] = i+1;
			}
		}
		
/*		for (i = 0; i < n; i++) {
			cout<<b[i]<<" ";
		}
		cout<<endl;
*/		dp[n-1] = 1;
		mx = 1;
		index = n-1;
		for (i = n-2; i >= 0; i--) {
			if (b[i] == -1) {
				dp[i] = dp[i+1]+1;
			}
			else if (dp[i+1] >= b[i]) {
				dp[i] = b[i];
			}
			else {
				dp[i] = dp[i+1]+1;
			}

			if (dp[i] >= mx) {
				mx = dp[i];
				index = i;
			}
		}
/*		for (i = 0; i < n; i++) {
			cout<<dp[i]<<" ";
		}
		cout<<endl;
*/		cout<<index+1<<" "<<index+mx<<endl;
	}

	return 0;
}
