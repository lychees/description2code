#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<algorithm>
using namespace std;
int main(void)
{
    int n,*x,*y,i,tot=0,t;
   	cin >> t;
    while(t--)
    {
        cin >> n;
        x=(int *)malloc(sizeof(int)*n);
        y=(int *)malloc(sizeof(int)*n);
        for(i=0;i<n;i++)
            scanf("%d",(x+i));
        for(i=0;i<n;i++)
            scanf("%d",(y+i));
        sort(x,x+n);
        sort(y,y+n);
        tot=0;
        for(i=0;i<n;i++)
            tot+=x[i]*y[i];
        printf("%d\n",tot);
    }
    return 0;
}
