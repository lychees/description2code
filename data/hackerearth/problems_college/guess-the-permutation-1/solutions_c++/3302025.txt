
#include<bits/stdc++.h>

using namespace std;

#define si(x) scanf("%d",&x)
#define pri(x) printf("%d",x)
#define sll(x) scanf("%lld",&x)
#define pll(x) printf("%lld",x)
#define sstr(s) scanf("%s",s)
#define pstr(s) printf("%s",s)
#define nl printf("\n")
#define ll long long int
#define pb push_back
#define mp make_pair

int a[1001][1001];
int cnt[1001][1001];
int ans[1001];
bool visited[1001];
int n;

int find_min(int num) {
    int pos=-1,mx=-1;
    for(int i=1;i<=n;++i) {
	if(cnt[num][i]>mx && !visited[i]) {
	    pos=i;
	    mx=cnt[num][i];
	}
    }
//    cout<<num<<" "<<pos<<endl;
    visited[pos]=true;
    return pos;
}

int main()
{
    int t;
    si(t);
    while(t--) {
	memset(cnt,0,sizeof(cnt));
	memset(visited,false,sizeof(visited));
	si(n);
	for(int i=1;i<=n;++i) {
	    for(int j=1;j<=n;++j) {
		si(a[i][j]);
		cnt[a[i][j]][i]++;
	    }
	}
//	for(int i=1;i<=5;++i) {
//	    cout<<cnt[1][i]<<endl;
//	}
//	si(t);
	vector< pair<int,int> > v;
	v.clear();
	for(int i=1;i<=n;++i) {
	    ans[i]=find_min(i);
	    v.pb(mp(ans[i],i));
	}
	sort(v.begin(),v.end());
	for(int i=0;i<n;++i) {
	    printf("%d ",v[i].second);
	}
	nl;
    }
    return 0;
}
