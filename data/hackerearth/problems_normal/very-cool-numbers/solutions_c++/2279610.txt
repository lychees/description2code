#include <iostream>
#include<cstring>
using namespace std;
int cool(int x)
{int s=0,ans=0;
	while(x)
	{
	if(x&1)
	{
		if(s==0)
			s++;
		else if(s==2)
			s++;
	}
	else
	{
		if(s==1)
			s++;
		else s=0;
	}
	if(s==3)
	{
		ans++;s=1;}
	x/=2;
	}return ans;
}
int main()
{int t,r,k;
    cin>>t;
    while(t--)
    {int c=0,x=0;
    	cin>>r>>k;
    	int *arr=new int[r+1];
    	memset(arr,0,r+1);
    	for(int i=1;i<=r;i++)
    		arr[i]=cool(i);
    	for(int i=1;i<=r;i++)
    	{
    		if(arr[i]>=k)
    			x++;
    	}cout<<x<<endl;
    }
}
