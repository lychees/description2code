#include <iostream>
using namespace std;

int main()
{
	int n, a[100], b[100], count=0, start=0, end;
	cin>>n;
	end = n-1;
	for(int i=0;i<n;i++){
		cin>>a[i];
		// cout<<a[i]<<" ";
	}
	for(int i=0;i<n;i++){
		cin>>b[i];
		// cout<<b[i]<<" ";
	}
	int i=0, j=0;
	while(i < n){
		if(a[j] == -1){
			j = (j+1)%n;
			continue;
		}
		if(b[i] != a[j]){
			count++;
		} else {
			a[j] = -1;
			i++;
			count++;
		}
		j = (j+1)%n;
	}
	cout<<count;
    return 0;
}
