#include <iostream>
#include <math.h>

using namespace std;

void solve(){
	int n,p;cin >> n >> p;
	for(int i=0;i<p;i++){
		if(n==3){
			break;
		}else{
			if(n%2==0){
                n=(n/2)+1;
			}
			else{
                n=(n/2)+2;
			}
		}
	}
	cout << n << endl;

}



int main()
{
	int t;cin >> t;
	for(int i=0;i<t;i++){
		solve();
	}

}
