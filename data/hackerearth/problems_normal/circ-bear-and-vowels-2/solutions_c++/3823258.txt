#include <iostream>
#include <string>
using namespace std;

bool isHard(string s)
	{
		int cc=0, vc=0, cir=0;
		for(size_t i=0; i<s.size(); ++i)
			{
				if(s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || 
				    s[i] == 'o' || s[i] == 'u' || s[i] == 'y' )
				    {
				    ++vc;
				    cir=0;
				    }
				else
				   {
				   	++cir;
				   	if(cir == 3) return true;
				    ++cc;
				   }
			}
		if(cc > vc) return true;
		return false;
	}

int main()
{
	int C=0;
	string T;
	cin >> C;
	for(int tc=0; tc< C; ++tc)
	{
		cin >> T;
		if(isHard(T)) cout << "hard" << endl;
		else cout << "easy" << endl;
	}
	
    return 0;
}
