#include<iostream>
#include<algorithm>
using namespace std;
int main() { 
	int n; cin>>n;
	long long arr[n];
	for(int i=0;i<n;i++) cin>>arr[i];
	sort(arr,arr+n); 
	long long i=0,j=n-1,sum=0;
	while(i<j){
		sum+= abs(arr[j]-arr[i]); 
		i++; 
		sum+= abs(arr[j]-arr[i]); 
		j--; 
	} 
	long long s=0;
	i=0,j=n-1;
	
while(i<j){
int swap=arr[i];
arr[i]=arr[j];
arr[j]=swap;
i++;
j--;
	} 
	i=0,j=n-1;
		while(i<j){
		s+= abs(arr[j]-arr[i]); 
		i++; 
		s+= abs(arr[j]-arr[i]); 
		j--; 
	} 
	
		cout<<max(sum,s)<<endl; }
