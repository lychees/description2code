#include <iostream>
#include <cmath>
using namespace std;

long long expo(int p, int y){
    if (p==1) return 10%y;
    else if (p==0) return 1;
    
    else{
        long long q = expo(p/2,y);
        q = (q*q)%y;    
        if(p%2==0){
            return q;
        }
        else{
            return ( (10%y)*q)%y;
        }
    } 
}

int main(){
    int t,r;
    int n,d;
    cin>>t;
    for(int i=0;i<t;i++){
        cin>>n>>d>>r;
        long long x = expo(r-1,d);
        x = (x*n)%d;
        
        double z = (double)x/(double)d;
        cout<<(int)(10*z)<<endl;
    }
    return 0;
}
