#include <iostream>
#include <stdio.h>
#include <math.h>
#include <stdlib.h>

using namespace std;

typedef long long int Lint;

int main()
{
	Lint t, n, v, x, val, s, k;
	
	scanf("%lld", &t );
	
	while( t-- ) {
		
		scanf("%lld", &n );
		v = 2*n;
		x = v - 2;
		val = 1 + 4*x;
		s = ( Lint )ceil( sqrt( static_cast<double> ( val ) ) );
		k = (Lint)ceil( static_cast<double>( s - 1 ) / 2.0 );
		printf("%lld\n", k );
	}
	
	return 0;
}
