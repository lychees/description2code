#include <bits/stdc++.h>
#define ll long long int
#define m 10000003
using namespace std;
int main()
{
    ll t, n, i;
    ll a, b, c, d;
    cin>>t;
    while(t--) {
    	cin>>n;
    	a = 0;
    	b = 1;
    	c = 1;
    	d = 7;
    	if(n == 0)
    		cout<<"0\n";
    	else if(n == 1 || n == 2)
    		cout<<"1\n";
    	else {
	    	for(i = 3; i <= n; i++) {
	    		d = a + b + 2 * i;
	    		a = b % m;
	    		b = c % m;
	    		c = d % m;
	    	}
    	cout<<c<<endl;
    	}
    }
    return 0;
}
