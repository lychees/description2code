#include <bits/stdc++.h>
using namespace std;

int main()
{	
	int t; cin>>t;
	while(t--){
		vector<pair<string,int>>ans;
		vector<string>table;
		int n,k,m; cin>>n>>k>>m;
		for(int i=0;i<n;i++){
			string temp; cin>>temp;
			table.push_back(temp);
			ans.push_back({temp.substr(0,m),i});
		}
		sort(ans.begin(),ans.end());
		//cout<<ans[k-1].second<<endl;
		string res=table[ans[k-1].second];
		cout<<res<<endl;
	}
    
    return 0;
}
