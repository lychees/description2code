#include <iostream>
using namespace std;

int main()
{
    int i,n;
    long long int m,j,k,c,a[1000000];
    cin>>n;
    for(i=1;i<=n;i++)
    {
    	c=0;
    	cin>>m;
    	for(j=1;j<=m;j++)
    	cin>>a[j];
    	for(j=1;j<=m;j++)
    	{
    		for(k=j;k<=m;k++)
    		{
    			if(a[j]==a[k])
    			c++;
    		}
    	}
    	cout<<c;
    	cout<<"\n";
    }
    return 0;
}
