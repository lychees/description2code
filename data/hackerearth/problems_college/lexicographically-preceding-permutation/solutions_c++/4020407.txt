using namespace std;

int main()
{
	long int t;
	long int i;
	long int x;
	long int n;
	
	vector<long int> v;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<n;i++)
        {
        cin>>x;
        v.push_back(x);
        }
        if(is_sorted(v.begin(),v.end()))
        {
        for(i=0;i<v.size();i++)
        {
         cout<<v[i]<<" ";
        }
        cout<<endl;
        }
        else{
        prev_permutation(v.begin(),v.end());
        for(i=0;i<v.size();i++)
        {
        cout<<v[i]<<" ";
        }
        cout<<endl;
        }
        v.clear();
    }
    return 0;
}
