#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {   	 
    	long int n,c=0;
    	scanf("%ld",&n);
    	for(long int i=1;i<=n;++i)
    	{
    		for(long int j=1;j<=n;++j)
    		{
    			if(i!=j)
    			{
    				if(((i*j)%(i+j))==0)
    				c++;
    			//	cout<<i<<" "<<j<<"\n";
    			}
    		}
    	}
    	
    	printf("%ld\n",c/2);
    }
    return 0;
}
