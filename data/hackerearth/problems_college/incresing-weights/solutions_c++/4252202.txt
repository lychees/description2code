#include<stdio.h>
#define INF 1000000001
 
int a[1000005];
int longest;
void bS(int x)
{
    int l = 1, r = longest;
    while(l <= r)
    {
        int mid = (l + r)/2;
        if(a[mid] <= x) l = mid+1;
        else r = mid-1;
    }
    a[++r] = x;
    if(r > longest) longest = r;
}
 
int main() 
{
    int i, N, v;
    scanf("%d",&N);
    for(i=0; i<N; i++)   a[i] = INF;
    for(i=1; i<=N; i++)
    {
        scanf("%d",&v);
		v-=i;
        if(v>=0) bS(v);        
    }
    printf("%d\n",N-longest); 
    return 0;
}
