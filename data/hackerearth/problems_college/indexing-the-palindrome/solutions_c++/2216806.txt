#include <bits/stdc++.h>
using namespace std;
bool isPalin(char *a)
{
int l=strlen(a);

for(int i=0;i<l;i++) if(*(a+i)!=*(a+l-i-1)) return 0;
return 1;

	
}
int main()
{
	int t,i,j;
	char s[1000000];
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	int l=strlen(s);
    	if(isPalin(s)) cout<<"-1"<<endl;
    	else {
    	char s1[l];
    	for(i=0;i<l;i++)
    	{
    		for(j=0;j<i;j++) s1[j]=s[j];
    		for(j=i+1;j<l;j++) s1[j-1]=s[j];
    		s1[j-1]='\0';
    		if(isPalin(s1) ) {cout<<i<<endl;break;}
    	}
    	}
    	
    }
    return 0;
}
