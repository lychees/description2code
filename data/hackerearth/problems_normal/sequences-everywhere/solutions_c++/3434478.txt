vector<LL> slow( LL x ) {
    vector<LL> v{x};
    while ( v.back()>2 ) {
        REP1(i,1,100) if ( v.back()%i!=0 ) {
            v.PB(i);
            break;
        }
    }
    // dump(x,v);
    return v;
};

LL lcm[42];
int len[42];
void predo() {
    lcm[0]=1;
    REP1(i,1,41) {
        len[i]=slow(i).size();
        lcm[i]=lcm[i-1]/__gcd(lcm[i-1],(LL)i)*i;
    }
}

LL lasai( LL l, LL r ) {
    LL ans=0;
    for ( LL i=l; i<=r; i++ ) ans+=slow(i).size();
    return ans;
}

LL solve( LL n ) {
    LL ans=0,pre=0;
    for ( int i=41; i>=1; i-- ) {
        LL now=n/lcm[i];
        ans+=(now-pre)*(len[i+1]+1);
        dump(i,now,pre,len[i+1]);
        pre=now;
    }
    if ( n>=2 ) ans-=2;
    ans-=2;
    dump(n,ans);
    return ans;
}

int main() {
    predo();
    LL a,b;
    R(a,b);
    LL ans=solve(b)-solve(a-1);
    W(ans);
    // dump(lasai(a,b));
    return 0;
}
