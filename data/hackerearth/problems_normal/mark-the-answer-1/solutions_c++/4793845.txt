#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int N,X;
    cin>>N>>X;
    int maxCont=0,tempCont=0,countCont=0,difficult=0;
    for(int i=0;i<N;i++){
    	int a;
    	cin>>a;
    	if(a<=X){
    		tempCont++;
    	}
    	else{
    		difficult++;
    		countCont+=tempCont;
    		if(difficult==2){
    			maxCont=max(maxCont,countCont);
    			countCont=tempCont;
    			difficult=0;
    			break;
    		}
    		tempCont=0;
    	}
    }
    if(maxCont==0){
    	maxCont=max(countCont,tempCont);
    }
    cout<<maxCont<<endl;
    
    return 0;
}
