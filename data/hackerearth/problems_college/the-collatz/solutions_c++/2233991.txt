#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,t;
    cin>>t;
    for(int i=0;i<t;i++)
    {

    cin>>n;int e=0,o=0;int c=0;
    while(n>=2)
    {
        if(n%2==0)
        {e=e+n;
            n=n/2;

        }
        else
        {
            o=o+n;
            n=3*n+1;
        }
c++;
    }
    if(n=1)o=o+1;

    if(e%c>o%c)cout<<"Even Rules"<<endl;
    if(e%c<o%c)cout<<"Odd Rules"<<endl;
    if(e%c==o%c)cout<<"Tie"<<endl;
    }

}
