#include <stdio.h>
char str[10000005];
int main()
{
	int i;
	fflush(stdin);
	unsigned long long int ans=0,one=0,zero=0;
	scanf("%s",str);
	for(i=0;str[i]!='\0';i++)
	{
		if(str[i]=='0')
		{
			zero=zero+one;
		}
		else if(str[i]=='1')
		{
			one=one+1;
			
			ans=ans+zero;
		}
	}
	printf("%llu\n",ans);
	

	return 0;
}
