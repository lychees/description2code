#include <bits/stdc++.h>
using namespace std;

#define MAXN 100010
typedef long long ll;
int t, n, r[MAXN], mini, id1, id2, range;
ll sum[MAXN];

int main() {
    scanf("%d", &t);
    while (t--) {
    	scanf("%d", &n);
    	sum[0] = 0;
    	for (int i = 1; i <= n; i++) {
    		scanf("%lld", &sum[i]);
    		sum[i] += sum[i - 1];
    	}
    	for (int i = 1; i <= n; i++) sum[i] %= n;
    	for (int i = 0; i < n; i++) r[i] = 0;
    	r[0] = 0;
    	mini = 1000000000;
    	for (int i = 1; i <= n; i++) {
    		if (r[sum[i]] == 0) {
    			if (sum[i] == 0) range = i - r[sum[i]];
    			else range = n + 10;
    		} else range = i - r[sum[i]];
			if (mini > range) {
				mini = range;
				id1 = r[sum[i]] + 1;
				id2 = i;
			}
			r[sum[i]] = i;
    	}
    	printf("%d %d\n", id1, id2);
    }
    return 0;
}
