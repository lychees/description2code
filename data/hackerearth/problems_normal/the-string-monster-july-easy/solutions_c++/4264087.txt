#include <bits/stdc++.h>
using namespace std;
int n;
char s[35][35];
char t[905];
int freqs[128], freqt[128];
int slen[35], tlen;
bool found;
void recur(int x, int len, int rlen) {
  if (x == n) {
    bool ok = true;
    for (char c = 'a'; c <= 'z'; c++) {
      //printf("%c %d %d\n", c, freqs[c], freqt[c]);
      ok &= freqs[c] == freqt[c];
    }
    found |= ok;
    return ;
  }
  if (len > tlen) {
    return ;
  }
  if (len + rlen < tlen) {
    return ;
  }
  recur(x + 1, len, rlen - slen[x]);
  bool ok = true;
  for (int i = 0; i < slen[x]; i++) {
    ok &= ++freqs[s[x][i]] <= freqt[s[x][i]];
  }
  if (ok) {
    recur(x + 1, len + slen[x], rlen - slen[x]);
  }
  for (int i = 0; i < slen[x]; i++) {
    freqs[s[x][i]]--;
  }
}
int main() {
  int T;
  scanf("%d", &T);
  for (int tt = 0; tt < T; tt++) {
    memset(freqt, 0, sizeof freqt);
    scanf("%d", &n);
    int sum = 0;
    for (int i = 0; i < n; i++) {
      scanf("%s", s[i]);
      slen[i] = strlen(s[i]);
      sum += slen[i];
    }
    scanf("%s", t);
    tlen = strlen(t);
    for (int i = 0; i < tlen; i++) {
      freqt[t[i]]++;
    }
    found = false;
    recur(0, 0, sum);
    printf(found ? "YES\n" : "NO\n");
  }
  return 0;
}
