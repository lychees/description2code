int main()
{
	int t;
	cin>>t;
	string s;
	while(t--)
	{
		cin>>s;
		int ans = 0;
		for(int i=0;i<s.size();++i)
		{
			if(s[i]!='B'and s[i]!='D'and s[i]!='S')
			{
				if(i==0)
				{
					if(s[i+1]!='B' and s[i+2]!='B' and s[i+1]!='S')ans++;
				}
				else if(i==s.size()-1)
				{
					if(s[i-1]!='S')ans++;
				}
				else{
					if(i==s.size()-2){
						if(s[i+1]!='B'and s[i+1]!='S'and s[i-1]!='S')ans++;
					}
					else{
						if(s[i+1]!='B'and s[i+1]!='S'and s[i-1]!='S'and s[i+2]!='B')ans++;
					}
				}
			}
		}
		cout<<ans<<'\n';
	}
    return 0;
}
