#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	int i,j,t,s;
	cin>>t;
	while(t--){
		cin>>s;
		int result=0;
		for(i=1;i<=s;i++)
		for(j=1;j<=s;j++)
		if(i*j<=s&&i!=j)
		result++;
		result/=2;
		for(i=1;i<=s;i++)
		if(i*i<=s)
		result++;
		cout<<result<<endl;		
	}
}
