#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,ans,k,i,j;
    scanf("%d",&n);
    int a[n];
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    ans=0;
    for(i=0;i<n;i++)
    {
        for(j=i+1;j<n;j++)
        {
            if(a[i]>a[j])
            {
                k=__gcd(a[i],a[j]);
                if(k==1)
                    ans++;
            }
        }
    }
    printf("%d\n",ans);
    return 0;
}
