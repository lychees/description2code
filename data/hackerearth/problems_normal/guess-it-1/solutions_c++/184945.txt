#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

int a[200];

int main() {
	int T;	
	scanf("%d",&T);
	
	while(T--){
		string str;
		cin>>str;
		
		int k = str.length();
		
		for(int i = 0; i<k; i++){
			a[i] = str[i] - 48;
		}	
		int sum = 0;
		for(int i = 0; i<k; i++){
			if(a[i]<=9){
				sum +=a[i];
			}
		}	
		//cout<<sum<<endl;
		
		sum %=9;
		sum = 9-sum;
		if(sum==0||sum==9){
			cout<<"0 or 9\n";
		}
		else 
			cout<<sum<<endl;
			
	
	}
	return 0;
}
