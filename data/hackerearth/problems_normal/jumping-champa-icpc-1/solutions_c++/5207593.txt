#include <iostream>
#include <algorithm>
using namespace std ;
bool so(int i,int j) {
	return (i<j);
}
int main()
{
	long long t,i,j,u,v,n,q,sum,ds[2000];
	cin>>t;
	while (t--)
	{
		cin>>n>>q;
		for (i=1;i<=n;i++)
		{
			cin>>ds[i];
		}
		sum=0;
		sort(ds+1,ds+1+n,so);
		for (i=2;i<=n;i++)
		sum+=ds[i]-ds[i-1];
		sum=sum*q;
		cout<<sum<<endl;
	}
}
