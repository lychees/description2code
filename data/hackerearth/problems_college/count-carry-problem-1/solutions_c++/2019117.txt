#include<iostream>
#include<stdlib.h>
#include<stdio.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n1,n2,sum=0,count=0;
		cin>>n1>>n2;
		while(n1||n2)
		{
			sum=0;
			int rem1=n1%10;
			int rem2=n2%10;
			sum=rem1+rem2;
			if(sum>9)
				count++;
			n1/=10;
			n2/=10;
		}
		if(count==0)
			cout<<"No carry operation"<<endl;
		else if(count==1)
			cout<<count<<" carry operation"<<endl;
		else
			cout<<count<<" carry operations"<<endl;
	}
}
