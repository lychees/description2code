//input
void scan(int &n)
{
    int sign = 1;
    n = 0;
    char c = gx();
    while( c < '0' || c > '9' )
    {
        if( c == '-' ) sign = -1;
        c = gx();
    }
    while( c >= '0' && c <= '9' ) n = (n<<3) + (n<<1) + c - '0', c = gx();  n = n * sign;
}
void lscan(ll &n)
{
    int sign = 1;
    n = 0;
    char c = gx();
    while( c < '0' || c > '9' )
    {
        if( c == '-' )
        sign = -1;
        c = gx();
    }
    while( c >= '0' && c <= '9' ) n = (n<<3) + (n<<1) + c - '0', c = gx();  n = n * (ll)(sign);
}
int sscan(char a[])
{
    char c = gx();
    while(c==' ' || c=='\n') c=gx();
    int i=0;
    while(c!='\n')a[i++] = c,c=gx();
    a[i]=0;
    return i;
}
int wscan(char a[])
{
    char c = gx();
    while(c==' ' || c=='\n') c=gx();
    int i=0;
    while(c!='\n' && c!=' ')a[i++] = c,c=gx();
    a[i]=0;
    return i;
}

//output
void print(int n)
{
    if(n<0)
    {
        n=-n;
        px('-');
    }
    int i=10;
    char o[10];
    do{o[--i] = (n%10) + '0'; n/=10;}while(n);
    do{px(o[i]);}while(++i<10);
}
void lprint(ll n)
{
    if(n<0LL)
    {
        n=-n;
        px('-');
    }
    int i=21;

}
void sprint(const char a[])
{

}


ll po(ll a, ll b, ll m)
{
	ll x=1,y=a;
	while(b > 0)
	{
		if(b%2 == 1)
		{
			x=(x*y);
			if(x>m) x%=m;
		}
		y = (y*y);
		if(y>m) y%=m;
		b /= 2;
	}
	return x;
}

const ll MOD = 1000000007LL;
const int siz = 1000005;

int a[3005][2];
int sp[3005];

int main()
{
    int n,i,j,k,x,y,z,ans=0,s=0;
    map<long double, int> smp;
    map<long double, int>::iterator it;
    long double temp;
    long double d[3005];
    scan(n);
    for(i=0;i<n;i++)
    {
        scan(a[i][0]);
        scan(a[i][1]);
        if(!a[i][1])
        {
            d[i] = -1.1111111111;
            s++;
        }
        else
        {
            d[i] = a[i][0] / (long double)a[i][1];
            smp[d[i]]++;
        }
    }
    ans = 0;
    for(j=0;j<n;j++)
    {
        for(k=j+1;k<n;k++)
        {
            x = a[j][0] + a[k][0];
            y = a[j][1] + a[k][1];
            if(!x) if(!y) continue;
            if(!x)
            {
                ans += s;
                if(d[j] == -1.1111111111) --ans;
                if(d[k] == -1.1111111111) --ans;
                continue;
            }
            temp = -y / (long double) x;
            it = smp.find(temp);
            if(it == smp.end()) continue;
            ans += it->second;
            if(d[j] == temp) --ans;
            if(d[k] == temp) --ans;
        }
    }
    cout << ans*2 << endl;
    return 0;
}
