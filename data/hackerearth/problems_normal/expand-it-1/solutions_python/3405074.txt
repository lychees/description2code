from string import ascii_lowercase

input1 = raw_input()
freq = {}
for i in ascii_lowercase:
    freq[i] = 0

cur_character='a'
cur_frequency=0
temp = input1+"z"
for c in temp:
    if c >= 'a' and c <= 'z':
        # print c
        freq[cur_character] += int(cur_frequency)
        cur_character = c
        cur_frequency = 0
    else:
        cur_frequency = 10*cur_frequency + int(c)

T = int(raw_input())
while T > 0:
    k = int(raw_input())
    done = False
    for cr in ascii_lowercase:
        if k - freq[cr] <= 0:
            print cr
            done = True
            break
        else:
            k = k - freq[cr]
    if not done:
    	print '-1'
    T -= 1
