art = []
l = input()
for _ in range(l):
    art.append(raw_input())
art = sorted(art)
arr = []
i = 0

while i < l:
    stack = []
    count = 1
    stack.append(art[i])
    i += 1
    while i < l and art[i] == stack[0]:
        count += 1
        i += 1
    arr.append(count)
arr = sorted(arr, reverse=True)

x = arr[0]
add = x
j = 1
l = len(arr)
while j < l and arr[j] == x:
    add += x
    j += 1
print add
