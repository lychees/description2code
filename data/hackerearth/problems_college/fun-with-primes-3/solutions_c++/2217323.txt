#include <iostream>
using namespace std;

int main()
{
   int t,n;
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	int p=n;
   	int sum=0;
   	int count=0;
   	while(p!=0)
   	{
   		int i=p%10;
   		sum+=i;
   		p=p/10;
   	}
   	for(int j=1;j<sum;j++)
   	{
   		if(sum%j==0)
   		count++;
   	}
   	if(count==1)
   	cout<<"YES";
   	else 
   	cout<<"NO";
   	cout<<endl;
   }
    return 0;
}
