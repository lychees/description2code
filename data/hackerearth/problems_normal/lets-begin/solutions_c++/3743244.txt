#include<bits/stdc++.h>
using namespace std;
int arr[]={2,3,5,7};
long long int dp[1000000];
int main()
{
   long long int n,i,t,j;
       cin>>t;
       while(t--)
       {

           scanf("%lld",&n);
           for(i=1;i<=n;i++)
           {
               dp[i]=INT_MAX;
               for(j=0;j<4;j++)
               {
                   if(i-arr[j]>=0)
                   {
                       dp[i]=min(dp[i],1+dp[i-arr[j]]);
                   }
               }
           }
           dp[1]=-1;
          cout<<dp[n]<<endl;
       }
return 0;

}
