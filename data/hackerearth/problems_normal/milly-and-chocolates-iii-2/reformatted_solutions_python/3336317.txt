def fun(i, n, c, cn, k, lis, b):
    if c == cn:
        if len(lis) >= k:
            return 0
        else:
            return 1
    if i == n:
        return 1
    listu = []
    for j in range(1, len(b[i])):
        if len(b[i][j]) > 0 and b[i][j] not in lis:
            lis.append(b[i][j])
            listu.append(b[i][j])
    if fun(i+1, n, c+1, cn, k, lis, b) == 0:
        return 0
    for j in range(0, len(listu)):
        lis.remove(listu[j])
    return fun(i+1, n, c, cn, k, lis, b)
