#include <bits/stdc++.h>
using namespace std;
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
		long long n,m,k,i,x,count=0,a,b;
		scanf("%lld %lld",&n,&k);
		for(i=0;i<n;i++){
			scanf("%lld",&x);
			if(x==k)
				count++;
		}
		a = (count*(count-1)*(count-2))/6;
		b = (n*(n-1)*(n-2))/6;
		x = __gcd(a,b);
		a = a/x;
		b = b/x;
		printf("%lld/%lld\n",a,b);
    }
    return 0;
}
