#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000000007
#define bitcount __builtin_popcountll
#define ll long long
ll gcd(ll a, ll b)
{
    if(a%b==0)
        return b;
    else
    {
        a=a%b;
        if(b>a)
            swap(a,b);
        return gcd(a,b);
    }
}
int main()
{
	int i,j,k;
	int t;
	ll val,val2,a,b,c,ans;
	sd(t);
	while(t--)
    {
        slld(a);
        slld(b);
        slld(c);
        if(a>b)
            swap(a,b);
        val2=gcd(a,b);
        val=gcd(max(val2,c),min(val2,c));
        ans=a/val;
        ans=(ans*(b/val))%mod;
        ans=(ans*(c/val))%mod;
        printf("%lld %lld\n",val,ans);
    }
	return 0;
}
