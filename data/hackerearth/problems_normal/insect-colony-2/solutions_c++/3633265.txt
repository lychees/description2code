#include<iostream>
#include<math.h>
#include<string.h>
#include<stdlib.h>
using namespace std;
string Xor_Of_Number(string s1,string s2)
{
	int n1,n2,max,min,diff,pointer1=0,pointer2=0,i,a1,a2;
	
	string s="";
	
		n1=s1.length();
		
		n2=s2.length();
		
		max=n1>n2?n1:n2;
		
		min=n1>n2?n2:n1;
		
		diff=max-min;
		
		if(n1>n2)
			pointer2=diff;
			
		else
			pointer1=diff;
			
		for(i=0;i<diff;i++)
		{
				s=s+"1";
		}
		
		for(i=diff;i<max;i++)
		{
				a1=s1.at(i-pointer1)-'0';
				
				a2=s2.at(i-pointer2)-'0';
				
				if(a1+a2==1)
					s=s+"1";
					
				else
					s=s+"0";
					
		}
		
		return s;
}
string BinaryOfNumber(int a)
{
		string s;
		int i=0;
		while(a!=0)
		{
			s+=a%2+'0';
			a=a/2;
		}
		string p="";
		for(i=s.length()-1;i>=0;i--)
			p+=s.at(i);
		return p;	
}
int compare(const void *a,const void *b)
{
	return ( *(int*)a-*(int*)b );
}
int main()
{
		int t,n,p[1000],i;
		
		cin>>t;
		
		while(t-->0)
		{
				cin>>n;
				
				for(i=0;i<n;i++)
					cin>>p[i];
					
				qsort(p,n,sizeof(int),compare);
				string b1=BinaryOfNumber(p[n-1]);
				for(i=n-2;i>=0;i--)
				{
					string b2=BinaryOfNumber(p[i]);
					b1=Xor_Of_Number(b1,b2);
				}
				
				char k=b1.at(b1.length()-1);
				
				if(k=='0')
					cout<<"Yes\n";
				else
					cout<<"No\n";
		}



}
