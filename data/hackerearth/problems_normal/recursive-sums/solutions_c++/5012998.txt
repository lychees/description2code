    #include <iostream>
    using namespace std;
     
    int main(){
        long long t,n,a,b,sum,x;
        cin >> t;
        while(t--){
        	sum=0;
        	cin >> n;
        	for(int i=0;i<n;i++){
        		cin >> a >> b;
        		sum+=b*a;
        		while(sum>9){
        			x=0;
        			while(sum){
        				x+=sum%10;
        				sum/=10;
        			}
        			sum=x;
        		}
        	}
        	cout << sum << endl;
        }
        return 0;
    }
