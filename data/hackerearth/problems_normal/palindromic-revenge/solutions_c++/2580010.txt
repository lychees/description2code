#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	int len = str.length();
    	int start = 0, end = len - 1;
    	int count = 0;
    	while(start < end)
    	{
    		if(str[start] != str[end])
    		{
    			int n = str[start] - str[end];
    			
    			if(n > 0)
    			count += n;
    	        else
    	        {
    	        	int m = -n;
    	        	count += m;
    	        }
    		}
    		start++;
    		end--;
    	}
    	cout<<count<<"\n";
    }
    
    return 0;
}
