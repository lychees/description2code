#define S 500009

vector<bool> prime(S + 1, true);
vector<int> ctr(S + 1, 0);
void seive()
{
	prime[0] = prime[1] = false;
	for(int i = 2; i <= S; i++)
	{
		if(prime[i])
		{
			for(ll j = 2*i; j <= S; j += i)
			{
				ll jj = j;
				int c = 0;
				if (i != 2)
				while(jj % i == 0)
				{
					c++;
					jj /= i;
				}
				ctr[j] = max (ctr[j], c);
				prime[j] = false;
			}
		}
	}
}


ll powe(int x, int n, int m)
{
	if(!n)
		return 1;
	ll ans = powe(x, n/2, m);
	ans = (ans * ans) % m;
	if(n & 1)
		ans  = (ans * x) % m;
	return ans;
}
