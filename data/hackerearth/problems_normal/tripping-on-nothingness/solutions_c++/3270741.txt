#include<bits/stdc++.h>
using namespace std;
# define N 1000000000
bool mark[N];
int gcd(int a,int b)
{
    if(b==0)
        return a;
    return gcd(b,a%b);
}
int main()
{
    int s,val,q,x;
    cin>>s>>q;
    while(q--)
    {
        cin>>x;
        val = gcd(s,x);
        if(!mark[val])
        {
            mark[val]=true;
            cout<<val<<endl;
        }
        else
            cout<<"-1"<<endl;
    }
}
