using namespace std;

int a,b,prime[maxm],primes[maxm];
void read()
{
    cin >> a >> b;
}

void sievee()
{
    for(int i = 2;i<maxm;++i)
    {
        if(prime[i] == 0)
        {
            for(int j = i*i;j<maxm;j+=i)
            {
                if(prime[j] == 0)
                    prime[j] = i;
            }
        }
    }
    int j = 0;
    for(int i = 2;i<maxm;++i)
    {
        if(prime[i] == 0)
            primes[j++] = i;
    }
}

bool is_prime(int n)
{
    int j = 0;
    for(int i = primes[j];i*i<=n;i=primes[++j])
    {
        if(n%i == 0)
            return false;
    }
    return true;
}

void solve()
{
    if(a > 2)
    {
        cout << 0 << endl;
        return;
    }
    int ans = 0;
    for(int i = 3;i<=b;++i)
    {
        if(is_prime(i))
            ans++;
    }
    cout << ans << endl;
}

int main()
{
    ios_base::sync_with_stdio(false);
    sievee();
    int t;
    cin >> t;
    while(t--)
    {
        read();
        solve();
    }
    return 0;
}
