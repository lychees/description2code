#include <bits/stdc++.h>
using namespace std;

typedef long long ll;


const ll mod=1000000007ll;
double eps=1e-8;
const int N=100001;
const int inf=20005;

	
#define sc(x) scanf("%lf",&(x))
#define scl(x) scanf("%lld",&x)
#define fori(n) for(int i=0;i<n;i++)
#define forj(n) for(int j=0;j<n;j++)

struct line{
	int x1;
	int y1;
	int x2;
	int y2;
	int pos;
	int flag=1;
};

int x;
int y=inf;
line a[101];


int find(int n){
	double temp=-1.0;
	int idx=-1;
	for(int i=0;i<n;i++){
		if(x>=a[i].x1 && x<=a[i].x2 && a[i].flag==0){
			double t1=1.0*(a[i].y2-a[i].y1)*(x-a[i].x1);
			t1/=(a[i].x2-a[i].x1);
			t1+=(a[i].y1*1.0);
			if(t1>temp && t1<=(y*1.0)){temp=t1;idx=i;}
		}
	}
	a[idx].flag=1;
	return idx;
}

////************
int main()
{
	//freopen("input.txt","r",stdin);
	int n;
	cin>>x>>n;
	for(int i=0;i<n;i++){
		cin>>a[i].x1>>a[i].y1>>a[i].x2>>a[i].y2;
		a[i].pos=i;
		a[i].flag=0;		
	}	
	for(int i=0;i<n;i++){		
		int idx=find(n);
		if(idx==-1)continue;		
		if(a[idx].y1<a[idx].y2){y=a[idx].y1;x=a[idx].x1;}
		else{y=a[idx].y2;x=a[idx].x2;}		
	}
	cout<<x<<endl;	
	


}
