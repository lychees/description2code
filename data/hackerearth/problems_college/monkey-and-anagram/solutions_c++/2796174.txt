void scanint(int &x)
{
    register int c = gx();
    x = 0;
    for(;(c<48 || c>57);c = gx());
    for(;c>47 && c<58;c = gx())
    {
        x = (x<<1) + (x<<3) + c - 48;
    }
}
int main()
{
	ios_base::sync_with_stdio(0);
	int t,flg,cnt,n1,n2;
	int cnt1[27],cnt2[27];
	string s,tt;
	cin>>t;
	while(t--)
	{
		flg=1;
		for(int i=0;i<27;i++)
		{
			cnt1[i]=cnt2[i]=0;
		}
		cnt=0;
		flg=1;
		cin>>s>>tt;
		n1=s.length();
		n2=tt.length();
		for(int i=0;i<n1;i++) cnt1[s[i]-'a']++;
		for(int i=0;i<n2;i++) cnt2[tt[i]-'a']++;
		
		for(int i=0;i<=25;i++)
		{
			if(cnt1[i]>cnt2[i]) flg=0;
		}
		if(flg==0) cout<<-1<<endl;
		else
		{
			for(int i=0;i<=25;i++)
			{
				if(cnt1[i]<cnt2[i]) cnt++;
			}
			cout<<cnt<<endl;
		}
	}
	return 0;
}
