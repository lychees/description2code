#include <iostream>
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;


int fun(ll n)
{
	if(n==0)	return 1;
	if(n<0)	return 0;
	return fun(n-6)||fun(n-9)||fun(n-20);
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll n;
		cin>>n;
		if(fun(n))	cout<<"True"<<endl;
		else cout<<"False"<<endl;
	}
    return 0;
}
