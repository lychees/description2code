#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        char a[100000];
        cin>>a;
        long int l=strlen(a);
        long int s=0;
        for(long int i=0;i<l/2;i++)
        {
            long int temp=a[(l-1)-i]-a[i];
            if(temp<0)
            temp=(-1*temp);
            s+=temp;
        }
        cout<<s<<endl;
    }
    return 0;
}
