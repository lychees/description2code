#include <bits/stdc++.h>
using namespace std;
vector<int>v,v1;
int fun(int i,int j,int sum)
{
	if(i==0||j==0)
	return sum=0;
	if(v[i]>v[j])
	{
		sum=100+fun(i-1,j-1,sum);
	}
	if(v[i]==v[j])
	{
		sum=max(50+fun(i-1,j-1,sum),fun(i,j-1,sum));
	
	}
	else
	{
			sum=fun(i,j-1,sum);	
	}
	return sum;
	
}
int main()
{
    int n;
    cin>>n;
    int a,i,j;
    for(int i=0;i<n;i++)
    {
    	cin>>a;
    	v.push_back(a);
    }
    for(int i=0;i<n;i++)
    {
    	cin>>a;
    	v1.push_back(a);
    }
    sort(v.begin(),v.end());
    sort(v1.begin(),v1.end());
   cout<<fun(n+1,n+1,0)<<endl; 
}
