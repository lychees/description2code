#include<iostream>
#include<vector>

using namespace std;

long select_coin(vector<long> H, int N, long K)
{
    int i, j;
    long max_stack = 0;
    long sum = 0;

    i = 0;

    while(i<N)
    {
        while(H[i] <= K && i<N)
        {
            sum += H[i];
            i++;
        }

        if(sum > max_stack)
        {
            max_stack = sum;
            sum = 0;
        }
        else
        {
            sum = 0;
        }
        i++;
    }

    return max_stack;
}

int main()
{
    int T, N;
    long K;
    vector<long> H;

    cin>>T;

    int i, j;
    long ele;

    for(i=0;i<T;i++)
    {
        cin>>N;
        cin>>K;
        for(j=0;j<N;j++)
        {
            cin>>ele;
            H.push_back(ele);
        }

        cout<<select_coin(H, N, K)<<endl;
        H.clear();
    }

    return 0;
}
