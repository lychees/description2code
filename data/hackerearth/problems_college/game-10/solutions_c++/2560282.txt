#include<stdio.h>
#include<iostream>

using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long int n,m;
        scanf("%lld %lld",&n,&m);
        long long int i,j,k=0;
        i=n-1;
        j=0;
        while(k<n)
        {
            if(m==i)
            {
                printf("%lld\n",k);
                break;
            }
            i--;
            k++;
            if(m==j)
            {
                printf("%lld\n",k);
                break;
            }
            j++;
            k++;
        }
    }
    return 0;
}
