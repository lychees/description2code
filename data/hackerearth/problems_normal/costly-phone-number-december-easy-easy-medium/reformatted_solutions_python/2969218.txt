def solve(l, n, p):
    b = True
    while b:
        b = False
        for i in xrange(10):
            for j in xrange(10):
                if l[i] + l[j] < l[(i+j) % 10]:
                    b = True
                    l[(i+j) % 10] = l[i] + l[j]

    r = 0
    for x in p:
        r += l[x]

    return r


t = int(raw_input())
for i in xrange(t):
    l = map(int, raw_input().strip().split())
    n = int(raw_input())
    p = map(int, raw_input())
    print solve(l, n, p)
