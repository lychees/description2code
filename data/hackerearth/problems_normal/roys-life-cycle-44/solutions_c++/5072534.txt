#include <iostream>
#include <map>
#include <vector>
#include <algorithm>

using namespace std;

int count(string s) {
    int cnt = 0;
    int ans = 0;
    for(int i=0; i<s.length(); i++ ) {
        if(s[i] == 'C' and cnt == 0) {
            cnt = 1;
        } else if(i>0 and s[i] == 'C' and s[i] == s[i-1]) {
            cnt += 1;
        } else {
            cnt = 0;
        }
        ans = max(ans, cnt);
    }
    return ans;
}

int main() {
    int n; cin >> n;
    string t, s;
    int res = 0;
    for(int i=0; i<n; i++ ) {
        cin >> s; t += s;
        res = max(res, count(s));
    }
    cout << res << " " << count(t) << endl;
    return 0;
}
