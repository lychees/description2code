#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    char a[7],b[7],c[8];
    int t,i,j,k,m,r;
    cin>>t;
    while(t--)
    {k=0;m=0;
        cin>>a>>b;
        i=strlen(a);
        j=strlen(b);
        r=0;
        while(k<i&&m<j)
        {
            c[k]=(r+a[k]-48+b[m]-48)%10;
              r=(r+a[k++]-48+b[m++]-48)/10;
              //cout<<"kam"<<c[k-1]+48<<endl;
              }
         while(k<i)
         {
             c[k]=(r+a[k]-48)%10;
               r=(r+a[k++]-48)/10;


         }
         while(m<j)
         {
             c[k++]=(r+b[m]-48)%10;
               r=(r+b[m++]-48)/10;

         }

            c[k++]=r;

          //c[k++]='/0';
         m=k;
         //cout<<m<<endl;

          while(c[k]==0)k++;
          while(k<m)
          {cout<<char(c[k++]+48);}
