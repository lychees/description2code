k = input("")
arr = [-1]*1000002
freq = [0]*1000002
temp = raw_input().split()
i = 0
while i < k :
	temp[i] = int(temp[i])
	arr[temp[i]] = 1
	freq[temp[i]] += 1
	i += 1
tempMem = 1000000
i = 1
while tempMem >= 1 :
	if(arr[tempMem] != -1) :
		arr[tempMem] = tempMem + i + freq[tempMem] - 1
		i += freq[tempMem]
	tempMem -= 1
max = arr[1]
i = 2
while i <= 1000000 :
	if(arr[i] > max) :
		max = arr[i]
	i += 1
print max + 1
