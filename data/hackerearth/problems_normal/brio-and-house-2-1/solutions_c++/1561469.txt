#include <iostream>
using namespace std;

int main()
{
    int test = 0;

  	cin>>test;
  	while(test--){
  		double x1,y1,x2,y2,r;
  		cin>>x1>>y1>>x2>>y2;
  		cin>>r;
  		double A = x2 - x1;
  		double B = y2 - y1;
  		
  		double C = x1*y2 - y1*x2;
  		double diff = r*r*(A*A + B*B) - C*C;
  		
  		if( diff == 0 )
  			cout<<"JUST MISSED\n";
  		else if(diff > 0)
  			cout<<"REPLANNING\n";
  		else
  			cout<<"SAFE\n";
  	}
    return 0;
}
