#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    char s[100];
    int t,i,n=0,u=0,l=0,len=0;
    scanf("%d",&t);
    if(t>10||t<=0)
     printf("Invalid Test");
    else
    {
     while(t--)
     {
       scanf("%s",&s);
       u=l=n=0;
       len=strlen(s);
       if(len>100)
        printf("\nInvalid Input");
       else{ 
       for(i=0;s[i]!='\0';i++)
       {
	    if(isalpha(s[i]))
	    {
		 if(islower(s[i]))
		  l++;
		 else if(isupper(s[i]))
		  u++;
    	}
	   else
	     n++;
      }
      if(n==len)
      printf("\nInvalid Input");
      if(u==len || l==len)
       printf("\n0"); 
      else if(n!=len)
       {
	    if(u<l)
	     printf("\n%d",u);
	    else
	     printf("\n%d",l);
       }  
       }
     } 
    }  
   return 0;
}
