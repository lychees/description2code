#include<bits/stdc++.h>
using namespace std;
const int MAXN = 100000 + 1;
int Tn, a[MAXN], n;
 
int main () {
	cin >> Tn;
	while (Tn--) {
		cin >> n;
		int x = 0;
		for(int i = 0; i < n; i++) {
			cin >> a[i];
			x ^= a[i];
		}
		int ret = 0;
		for(int i = 0; i < n; i++) 
			if (a[i] == x)
				++ret;
		cout << ret << endl;
	}
    return 0;
}
