#include <cstdlib>
#include <iostream>
#include <math.h>

using namespace std;
long long I,T,n;
double t,p;
int main(int argc, char *argv[])
{   cin>>T;
    for(I=1;I<=T;I++)
    {   cin>>n;
        t=sqrt(n);
        if(t==(long long)t)
            cout<<"YES"<<endl;
        else
        {   p=round(t);
            
            if(n>p*p)
               cout<<"-"<<(long long)(n-p*p)<<endl;
            else
               cout<<"+"<<(long long)(p*p-n)<<endl;    
        }                  
    }
   
    return EXIT_SUCCESS;
}
