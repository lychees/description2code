#include<stdio.h>
#include<set>
using namespace std;
int main()
{
	int n,a,b;
	int arr[100009];
	for(int i=0;i<100009;i++)
		arr[i]=0;
	scanf("%d",&n);
	set<int> s;
	set<int>::iterator it1,it2;
	
	while(n--)
	{
		scanf("%d",&a);
		if(a==1)
		{
			scanf("%d",&b);
			if(arr[b]==0)
			s.insert(b);
			arr[b]++;
			
		}
		else if(a==2)
		{
			scanf("%d",&b);
			if(arr[b]<=0)
			printf("-1\n");
			else if(arr[b]==1)
			{
				s.erase(b);
				arr[b]--;
			}
			else
			arr[b]--;
			
		}
		else if(a==3)
		{
			if(s.empty())
			{printf("-1\n");
			continue;
			}
			it1=s.end();
			it1--;
			printf("%d\n",*it1);
		}
		else
		{
			if(s.empty())
			{printf("-1\n");
			continue;
			}
			it2=s.begin();
			printf("%d\n",*it2);
		}
	}
	return 0;
}
