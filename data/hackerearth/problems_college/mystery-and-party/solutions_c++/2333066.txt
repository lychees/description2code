#include <iostream>
using namespace std;
#include <stdio.h>
#include <algorithm>
int main()
{
	int TC;
	long int n1,n,k;
	long long int *Arr;
	scanf("%d",&TC);
	while(TC>0)
	{
		TC--;
		scanf("%ld%ld",&n,&k);
	 	n1 = n;
	 	Arr = (long long int *)malloc(n*sizeof(long long int));
		while(n>0)
		{
				n--;
			scanf("%lld",&Arr[n]);
		
         }
		sort(Arr,Arr+n1);
		printf("%lld\n",Arr[k-1]);
	}
		

    return 0;
}
