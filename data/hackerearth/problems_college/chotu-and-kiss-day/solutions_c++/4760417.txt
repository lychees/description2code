#include <bits/stdc++.h>
using namespace std;

long long int a[10000];

int main()
{	
	int n;
	int i,j;
	cin>>n;
	long long int k;
	for(i=0;i<n;i++){
		cin>>a[i];
	}
	
	cin>>k;
	sort(a,a+n);
	long long int x;
	int flag;
	for(i=0;i<n-1;i++){
		for(j=i+1;j<n;j++){
			x = k-a[i]-a[j];
			flag=0;
			int up = n-1;
			int low = 0;
			 while (low <= up) {
				int mid = (up + low)/2;
			      if (a[mid] < x)
			         low = mid + 1;    
			      else if (a[mid] == x) {
			         cout<<"YES"<<endl;
			         flag = 1;
			         break;
			      }
			      else{
			         up = mid - 1;
			    }
			}
			if(flag)
			{
				break;
			}
		}
		if(flag)
		{
			break;
		}
		}
		
	if(!flag)
	{
		cout<<"NO"<<endl;
	}
	
   
   
}
