#include<bits/stdc++.h>
#include<string>
using namespace std;
int main()
{
int t,j,i,c;
cin>>t;
cin.ignore (std::numeric_limits<std::streamsize>::max(), '\n');
char s[100001];
while(t>0)
{
    i=0;
    char p[200][100001]={" "};
    fflush(stdin);
    cin.getline(s,100001);
    istringstream iss(s);
    char word[100001];
    while(iss >> word)
    {   c=0;
    	for(j=0;j<i;j++)
    	{
    		if(strcmp(p[j],word)==0)
    		{
    			c=1;
    		   break;
    		}
    	}
    	if(!c)
        strcpy(p[i++],word);
    }
    for(j=0;j<i;j++)
    cout<<p[j]<<" ";
    cout<<endl;
    t=t-1;
}

}
