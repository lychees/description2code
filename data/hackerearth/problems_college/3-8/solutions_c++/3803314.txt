#include <iostream>

using namespace std;

int main()
{
	int t, i, j, k;
	char str[20000], a[2000];
	cin >> t;
	for (; t; --t) {
		cin >> a;
		str[3001] = a[0];
		i = j = 3001;
		for (k = 1; a[k] != '\0'; ++k) {
			if (str[i] <= a[k]) {
				str[--i] = a[k];
			} else {
				str[++j] = a[k];
			}
		}
		
		for (k = i; k <= j; ++k) {
			cout << str[k];
		}
		cout << endl;
	}
	
	return 0;
}
