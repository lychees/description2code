#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,j=2,i,count=1,flag=0,flag1=0;
    	cin>>n;
    	while(j<n)
    	{ flag1=0;
    		for(i=2;i<=j;i++)
    		{
    			if(j%i==0&&n%i==0)
    			{flag1=1; break;}
    		}
    		if(flag1==0)
    		count++;
    		j++;
    		
    	}
    	
    	
    	
    	
    	for(i=count-1;i>1;i--)
    	{
    		
    		if(count%i==0)
    		{
    			flag=1;
    			break;
    		}
    		
    	}
    	
    	if(flag==0&&n>2)
    	cout<<"Correct\n";
    	else
    	cout<<"Incorrect\n";
    }
    return 0;
}
