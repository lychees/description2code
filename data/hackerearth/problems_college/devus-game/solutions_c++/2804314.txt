#include <iostream>
using namespace std;

int main() {
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int a[n],j=0,x=0,k;
		for(int i=0;i<n;i++) cin>>a[i];
		while(j<n)
		{
			if(a[j]==-1) { x=1;break; }
			else if(a[j]==-5) { x=0; break; }
			//j++;
			k=j;
			j=a[j]; a[k]=-5;
		}
		if(x) cout<<"Win\n";
		else cout<<"Lose\n";
	}
	return 0;
}
