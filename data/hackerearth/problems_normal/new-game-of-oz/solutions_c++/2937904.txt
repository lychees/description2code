#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	long int a[n],i,cnt=0;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	sort(a,a+n);
    	for(i=0;i<n;)
    	{
    		if( (i<n-1) &&(abs(a[i+1]-a[i])==1))
    		{
    			cnt++;
    			i=i+2;
    		}
    		else
    		{
    			cnt++;
    			i++;
    		}
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
