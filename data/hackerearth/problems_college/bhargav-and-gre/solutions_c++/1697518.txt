#include <iostream>
#define ll long long
using namespace std;

int main()
{
    ll n = 0, syn = 0, ant = 0;
    string s = "";
    cin >> s;
    cin >> n;
    string stc = "";
    while (n--){
    	cin >> stc;
    	if (stc[0] == s[0] && stc.length() == s.length())
    		syn++;
    	else if (((int(stc[0]) - int(s[0]) == 3) && stc.length() == s.length())){
    		ant++;
    	}
    }
    cout << syn << " " << ant << endl; 
    return 0;
}
