#include <iostream>
using namespace std;

using ull = unsigned long long;
constexpr ull mod = 1000000007;

ull powm(ull x, ull n) {
	ull res = 1;
	while(n) {
		if (n % 2)
			res = (res * x) % mod;
		x = (x * x) % mod;
		n /= 2;
	}
	return res;
}

int main() {
	ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    int T;
    cin >> T;
    while (T--) {
    	ull n;
    	cin >> n;
    	cout << powm(500000004, n - 1) << '\n';
    }
    return 0;
}
