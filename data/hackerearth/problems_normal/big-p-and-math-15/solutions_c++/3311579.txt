#include <iostream>
#include <math.h>
#include <string>
using namespace std;
int find_max(string x){
	for(int i=0;i<x.length();i++){
		if(x.at(i)=='5'){
			x.at(i)='6';
		}
	}
	return atoi(x.c_str());
}
int find_min(string x){
	for(int i=0;i<x.length();i++){
		if(x.at(i) == '6'){
			x.at(i)='5';
		}
	}
	return atoi(x.c_str());
}

int main()
{
    int a,b;
    string temp;
    int temp_max_a,temp_min_a,temp_max_b,temp_min_b;
    long result_max,result_min;
    cin>>a>>b;
    if(a<1 || a>pow(10,6)){return 0;}
    if(b<1 || b>pow(10,6)){return 0;}
    temp = to_string(a);
    temp_max_a = find_max(temp);
    temp_min_a = find_min(temp);
    temp = to_string(b);
    temp_max_b = find_max(temp);
    temp_min_b = find_min(temp);
    result_min = temp_min_a+temp_min_b;
    result_max = temp_max_a+temp_max_b;
    cout<<result_min<<" "<<result_max;
    return 0;
}
