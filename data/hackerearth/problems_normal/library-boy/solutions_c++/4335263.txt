#include <iostream>
using namespace std;

int main()
{
    int n,i,j,k=0,p=0;
    cin>>n;
    char a[n][30];
    for(i=0;i<n;i++)
      cin>>a[i];
    for(i=0;i<n;i++)
    {
    	for(j=0;j<=i;j++)
    	{
    		if(a[i][0]==a[j][0])
    		  k++;
    	}
    	if(k%10==1)
    	  p++;
    	k=0;
    }
    cout<<p<<endl;
    return 0;
}
