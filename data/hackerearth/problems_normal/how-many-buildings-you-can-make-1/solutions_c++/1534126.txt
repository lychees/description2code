#include<iostream>
#include<math.h>
#include<stdlib.h>
#include<stdio.h>
#define MOD 1000000007
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,cnt=0;
		cin>>n;
		for(int i=1;i<=n;i++)
		{
			for(int j=i;j<=n;j++)
			{
				if(i*j<=n)
				cnt++;
				else
				break;
			}
		}
		cout<<cnt<<endl;
	}
return 0;
}
