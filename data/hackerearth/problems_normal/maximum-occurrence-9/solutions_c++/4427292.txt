#include <iostream>
#include<stdio.h>
#include<string.h>
#include<unordered_map>
using namespace std;

int main()
{
    char a[1005];
    gets(a);
    int l=strlen(a);
    unordered_map<char,int> mymap;
    for(int i=0;i<l;i++)
    {
    	if(mymap.find(a[i])==mymap.end())
    	{
    		mymap[a[i]]=1;
    	}
    	else
    		mymap[a[i]]=mymap[a[i]]+1;
    }
    unordered_map<char,int>::iterator it=mymap.begin();
    char fin;
    int count;
    fin=it->first;
    count=it->second;
    for(it=mymap.begin();it!=mymap.end();it++)
    {
    	if(it->second>count)
    	{
    		count=it->second;
    		fin=it->first;
    	}
    	else if(it->second==count && it->first<fin)
    	{
    		fin=it->first;
    		count=it->second;
    	}
    }
    cout<<fin<<" "<<count<<endl;
}
