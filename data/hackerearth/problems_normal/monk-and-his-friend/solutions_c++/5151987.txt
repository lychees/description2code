#include <iostream>
using namespace std;
int countbits(long long a)
{
	int count =0 ;
	while(a){
		a = a&(a-1);
		count ++;
	}
	return count;
}
int main()
{
    //cout << "Hello World!" << endl;
    int tc;
    long long m,n;
    cin>>tc;
    while(tc--){
    	cin >> m >> n;
    	cout << countbits(m^n) << endl;
    }
    return 0;
}
