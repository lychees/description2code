#include<cstdio>
#include<iostream>
#include<cstdio>
#include<cmath>
#include<algorithm>
#include<vector>
using namespace std;
int compare(const void *a ,const void *b)
{
	return (*(int*)b - *(int*)a);
}
int main(int argc , char *argv[])
{
	int n,i;
	cin>>n;
	int a[n],b[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	for(i=0;i<n;i++)
	{
		cin>>b[i];
	}
	qsort(a,n,sizeof(int),compare);
	qsort(b,n,sizeof(int),compare);
	int rem =0;
	for(i=0;i<n-1;i++)
	{
		if(a[i]>b[i+1])
		{
			rem = rem + 100;
		}
		else
		if(a[i]==b[i+1])
		{
			rem = rem + 50;
		}
	}
	
	cout<<rem<<"\n";
}
