#include <bits/stdc++.h>
#define maxii 10005
using namespace std;
#define ll long long

struct point
{
    int x,y;
};
struct point lines[105][2];
int initx;
double gety(int i,double x)
{
    double m=(double)(lines[i][1].y-lines[i][0].y)/(lines[i][1].x-lines[i][0].x);
    double c=(double)lines[i][1].y-m*lines[i][1].x;
    return m*x+c;
}

bool betweenx(int x,int i)
{
  double xx=(x-lines[i][0].x)*(x-lines[i][1].x);
  if(xx<=0)
  return true;
  else
  return false;
}
bool vis[105]={false};
int main()
{
   ios_base::sync_with_stdio(0);
   int n;
   cin>>initx>>n;

   for(int i=0;i<n;i++)
   {
    cin>>lines[i][0].x>>lines[i][0].y>>lines[i][1].x>>lines[i][1].y;
   }
   int currx=initx,curry=100005;
   
   for(int i=0;i<n;i++)
   {
    double temp=-1;
    int in=-1;
    for(int j=0;j<n;j++)
    {
     if(!betweenx(currx,j))
        continue;
     if(vis[j])
     continue;
     double t=gety(j,currx);
    // cout<<"hii"<<t<<" ";
     if(t<=curry)
     {
        
        if(temp<t)
        {
        	temp=t;
        	in=j;
        }
        
     }
     //cout<<temp<<" "<<in<<" ";
    }
    if(in==-1)
        break;
    if(lines[in][0].y<lines[in][1].y)
    { currx=lines[in][0].x;
      curry=lines[in][0].y;
    }
    else
     {
      currx=lines[in][1].x;
      curry=lines[in][1].y;
    
     }
     vis[in]=true;
  // cout<<"hell"<<currx<<" "<<curry<<"\n";
   }

   cout<<(int)currx;


    return 0;
}
