#include<bits/stdc++.h>

using namespace std;
#define ll long long int
#define mq 1000000007
#define pb push_back
#define mk make_pair
#define fi first
#define se second
#define pii pair<ll, ll>

int c[10][26];
char dice[10][6];

int main() {

	int n,k;
	cin>>n>>k;

	for(int i=0; i<n; i++) {
		for (int j=0; j<6; j++) {
			cin>>dice[i][j];
			c[i][dice[i][j]-'a']++;
		}
	}

	string s;
	cin>>s;
	int ways =0;
	
	for(int i=0; i<(1<<n); i++) {

		if(__builtin_popcount(i)==k) {
			vector <int > cubes;
			int index=0;
			int val = i;
			while(val>0) {
				if(val%2==1) {
					cubes.pb(index);
				}
				index++;
				val=val/2;
			}

			sort(cubes.begin(), cubes.end());

			
			do{
				int pways=1;
				for(int j=0; j<k; j++) {
					pways*=c[cubes[j]][s[j]-'a'];
				}
				ways+=pways;
			} while(next_permutation(cubes.begin(), cubes.end()));
		}
	}
	
	cout<<ways;
}
