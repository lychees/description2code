#include<bits/stdc++.h>
using namespace std;
inline bool solve(int n)
{
    int prev=n%2;
    n>>=1;
    while(n)
    {
        if(prev && n%2)
        return false;
        prev=n%2;
        n>>=1;
    }
    return true;
}
int main()
{
    int n,t,lgn;
    cin>>t;
    while(t--)
    {
        cin>>n;
        while(!solve(n))
            n++;
        cout<<n;
        cout<<'\n';
    }
}
