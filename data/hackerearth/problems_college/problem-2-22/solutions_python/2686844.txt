from math import sqrt
t=int(raw_input(''))
for k in range(0,t):
    s=0
    a=map(int,raw_input('').split())
    r=a[0]
    x=a[1]
    y=a[2]
    for i in range(0,4):
        a=map(int,raw_input('').split())
        x1=a[0]
        y1=a[1]
        d=sqrt(((x-x1)**2)+((y-y1)**2))
        if d>r:
            s=1
    if s==0:
        print 'Yes'
    else :
        print 'No'
