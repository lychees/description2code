#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    string str1;
    string str2;
    while(tc--)
    {
    	int pos;    	
    	cin>>str1;
    	cin>>str2;
     	for(int i=0;i<str2.size();i++)
     	{
     	pos=str1.find(str2[i]);
     	if(pos>-1)
     	{
     		break;
     	}
     	}
     	if(pos>-1)
     	{
     	cout<<"YES"<<endl;	
     	}
     	else
     	{
     	cout<<"NO"<<endl;	
     	}
     	
    }
}
