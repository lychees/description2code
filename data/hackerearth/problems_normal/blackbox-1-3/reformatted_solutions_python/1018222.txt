import sys


def gc(a, b):
    if b == 0:
        return a
    else:
        return gc(b, a % b)


t = int(raw_input())
while t > 0:
    t -= 1
    a, b, c = map(int, raw_input().split())
    z = gc(a, b)
    s = gc(z, c)

    if (a**2 + b**2 == c**2 or a**2 + c**2 ==
            b**2 or b**2 + c**2 == a**2) and s == 1:
        print 'YES'
    else:
        print 'NO'
