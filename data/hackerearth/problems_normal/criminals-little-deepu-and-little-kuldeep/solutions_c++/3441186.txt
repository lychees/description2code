#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,tc;
    cin >> tc;

    while(tc--){
    cin >> n;
    long int a[n];
    for (int i = 0;i <n;i++)
        cin >> a[i];

    long int count = 0;
    sort(a,a+n);
    int b[n];
    for (int i = 0;i <n;i++)
        b[i] = -1;
    long int k = 0;

    for (int i = 0;i <n;i++)
    {
        if (b[i] == -1){
        for (int j = i;j < n;j++)
        {
            if (a[i] == a[j] && b[j] == -1){
                k++;
                b[j] = 1;
            } else 
            	break;
        }

        count = max(count,k);
        k = 0;
        }
    }

    cout << count << endl;
    }

    return 0;
}
