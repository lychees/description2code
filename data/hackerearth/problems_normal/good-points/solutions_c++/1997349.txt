#include <iostream>
using namespace std;
 
int main()
{
    int N;
    cin>>N;
    if(N==1 || N==2 || N==4) cout<<"YES"<<endl;
    if(N==1) cout<<"0 0"<<endl;
    else if(N==2)
    {
    	cout<<"0 0"<<endl;
    	cout<<"1 0"<<endl;
    }
    else if(N==4)
    {
    	cout<<"0 0"<<endl<<"0 1"<<endl<<"1 1"<<endl<<"1 0"<<endl;
    }
    else
    cout<<"NO"<<endl;
}
