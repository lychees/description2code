#include <bits/stdc++.h>
using namespace std;
#define nextInt(n) scanf("%d", &n);
#define next(s) scanf("%s", &s);

struct Pair {
    int c,r;
    Pair(int xx, int yy) {
        c = xx;
        r = yy;
    }
    Pair() {}
};

bool cmp(Pair a, Pair b){
    if(a.c == b.c) return b.r < a.r;
    else return a.c < b.c;

}

int main() {
    int n = 0;
    nextInt(n);
    Pair p[n];
    for(int i = 0; i < n; i++) {
        int a = 0, b = 0;
        nextInt(a); nextInt(b);
        p[i] = Pair(a,b);
    }
    sort(p, p+n, cmp);
    for(int i = 0; i < n; i++) {
        printf("%d %d\n", p[i].c, p[i].r);
    }

    return 0;
}
