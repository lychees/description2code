#include<iostream>
using namespace std;
#define max 100000
#define ll long long
ll a[max];
void sieve(ll n)
{
	ll i,j;
	for(i=0;i<=n;i++)
	{
		a[i]=1;
		}	
a[0]=0;
	a[1]=0;

	for(i=2;i*i<=n;i++)
	{
		if(a[i]==1)
		{
			for(j=i*i;j<=n;j+=i)
			{
				a[j]=0;
			}
		}
	
	}
}
ll gcd(ll a,ll b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
void div(ll n)
{
	ll i,s=1;
	for(i=2;i<n;i++)
	{
		if(gcd(n,i)==1)
		{
			s+=1;
		}
	}

	if(a[s]==1)
	{
		cout<<"TRUE"<<endl;
		
	}
	else
	cout<<"FALSE"<<endl;
}
int main()
{
ll t,i,x;
cin>>t;
sieve(max);
for(i=0;i<t;i++)
{
	cin>>x;
	
	div(x);
}
	return 0;
}
