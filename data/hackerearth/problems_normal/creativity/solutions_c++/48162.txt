#include <iostream>
#include <string>
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <vector>
#include <algorithm>
using namespace std;
#define PR(x) cout << #x " = " << x << "\n";


int main(){
	int t;
	cin>>t;
	getchar();
	string s1,s2;
	
	while(t--){
		std::getline (std::cin,s1);
		std::getline (std::cin,s2);
		int flag=0;
		for (int i = 0; i < s2.size() - s1.size()+1; ++i)
		{
			// cout<<"substr is ->"<<s2.substr(i,s1.size())<<endl;
			if(s2.substr(i,s1.size()) == s1){
				flag = 1;
				break;
			}
		}

		if(flag){
			cout<<"possible"<<endl;
		}else{
			cout<<"not possible"<<endl;
		}
	}
	return 0;
}
