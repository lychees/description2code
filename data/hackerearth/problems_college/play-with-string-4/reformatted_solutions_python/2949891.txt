t = int(raw_input())
while t > 0:
    s = raw_input()
    r = ""
    f = 0
    for i in s[::-1]:
        r = r+i
    le = len(r)
    for i in range(1, le):
        if (abs(ord(s[i])-ord(s[i-1]))) != abs((ord(r[i])-ord(r[i-1]))):
            print "Not Funny"
            f = 1
            break
    if i+1 == le and f != 1:
        print "Funny"
    t = t-1
