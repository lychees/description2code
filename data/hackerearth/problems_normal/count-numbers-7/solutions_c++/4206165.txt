#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    int K,T; ll a,b;
    cin >> K >> T;
    vector<ll> A(K);
    for(int i = 0; i < K; i++){
    	cin >> A[i];
    }
    while(T--){
    	ll ans = 0LL;
    	cin >> a >> b;
    	for(int i = 1; i < (1 << K); i++){
	    	ll P = 1LL;
	    	int C = 0;
	    	for(int j = 0; j < K; j++){
	    		if(i&(1 << j)){
	    			C++;
	    			P *= A[j];
	    		}
	    	}
	    	ll n1 = (a - 1)/P;
	    	ll n2 = b/P;
	    	if(C&1) ans += (n2 - n1);
	    	else ans -= (n2 - n1);
	    }
	    cout << ans << "\n";
    }
    return 0;
}
