def issub(s1, s2, m, n):
    if n == 0:
        return True
    if m == 0:
        return False
    if s1[m-1] == s2[n-1]:
        return issub(s1, s2, m-1, n-1)
    else:
        return issub(s1, s2, m-1, n)


s = raw_input()
if issub(s, "love", len(s), 4):
    print "I love you, too!"
else:
    print "Let us breakup!"
