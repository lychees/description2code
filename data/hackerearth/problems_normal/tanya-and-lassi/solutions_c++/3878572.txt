#include <iostream>
#include <cmath>
#include <cstdio>
using namespace std;
typedef long long int64;

int main()
{
	int T;
	int L;
	int64 arr[10001];
	int64 max_prof[10001];
	cin>>T;
	while(T) {
		int L;
		cin>>L;
		for(int i=1;i<=L;i++)
			cin>>arr[i];
		max_prof[1]	= arr[1];
		for(int i=2;i<=L;i++) {
			max_prof[i] = arr[i];
			for(int j=1;j<i;j++) {
				max_prof[i] = max(max_prof[i],arr[j]+max_prof[i-j]);
			}
		}
		cout<<max_prof[L]<<endl;
		T--;
	}
    return 0;
}
