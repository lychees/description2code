#include <iostream>
using namespace std;
long int dp[100001][10];
 
int main()
{
    int n,q,arr[100001],l,r,k;
    cin>>n>>q;
    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    for(int i=0;i<10;i++)
    {
        for(int j=1;j<=10;j++)
        {
        	int k=i;
        	int sum=0;
        	while(k<n){
        		sum+=arr[k];
        		k=k+j;
        	}
        	dp[i][j-1]=sum;
        }
    }
    for(int i=10;i<n;i++) {
    	for(int j=1;j<=10;j++)
    	{
    		dp[i][j-1]=dp[i-j][j-1]-arr[i-j];
    	}
    }
    
    while(q--)
    {
        cin>>l>>r>>k;
        int temp=(r-l)/k;
        int temp1=l+k*temp;
        cout<<dp[l-1][k-1]+arr[temp1-1]-dp[temp1-1][k-1]<<endl;
    }
    return 0;
}
