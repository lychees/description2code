#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    char s[100];
    int a[130],b[130],i,m,n,k=0;
    for(i=97; i<130; i++)
        a[i]=0;
    cin>>m>>n;
    for(int j=0; j<m; j++)
    {
        cin>>s;
        for(i=97; i<130; i++)
            b[i]=0;

        for(i=0; i<(strlen(s)); i++)
            b[s[i]]++;
        for(i=97; i<130; i++)
            if(b[i]>0) a[i]++;
    }
    for(i=97; i<130; i++)
        if(a[i]==m) k++;
    if(k>=n) cout<<":-)";
            else cout<<":-(";
                    return 0;
}
