#include <stdio.h>
#include <ctype.h>
#include <iostream>
int main()
{
    char s[100];
    scanf("%s",s);
    int i,a;
    for(i=0;s[i]!='\0';i++){
    	if(s[i]<='z' && s[i]>='a'){
    		s[i]=toupper(s[i]);
    	}
    	else{
    		s[i]=tolower(s[i]);
    	}
    }
    printf("%s",s);
    return 0;
}
