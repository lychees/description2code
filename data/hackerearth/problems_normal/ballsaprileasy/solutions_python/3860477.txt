layers = int(raw_input())
ballsEachLayer = map(int, raw_input().split())
ballsEachLayer.sort()
res = 0
ballLayer = 0
i = 1
while (ballLayer < layers):
	ballsRecovered = (i * i) - ballsEachLayer[ballLayer]
	if ballsRecovered >= 0:
		res += ballsRecovered
		ballLayer += 1
	else:
		res += (i * i)
	i += 1
print res
