#include <iostream>
using namespace std;

int main()
{
    int t,i,j;
    cin >> t;
    int arr[t];
    for(i = 0; i < t; i++){
    	cin >> arr[i];
    }
    for(i = 0; i < t; i++){
    	if(i == 0){
    		cout << arr[i] << " ";
    	}else{
    		cout << (i+1)*arr[i]-(i)*arr[i-1] << " ";
    	}
    }
    return 0;
}
