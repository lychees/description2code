#include <iostream>
#include <algorithm>
using namespace std;
long long a[100001];
int main()
{
    ios::sync_with_stdio(0);
    int n,t;
    long long k; 
    cin>>t;
    while(t--){
    	cin >> n >> k; 
    	for(int i=0;i<n;i++)
        	cin >> a[i]; 
    	long long sm = 0; 
    	long long best = -1,bestaj;
    	sort(a,a+n);
    	long long mx = 0;
    	for(int j = 0;j<n;j++){
    		mx = a[j];
    		int j0 = j;
    		//while(j0<n && a[j0]==mx) j0++;
    		//if (j0<n) j0--;
    		long long cnt = 0;
    		int freq = 0;
    		//cout << mx << ",,";
    		for(int i = j0;i>=0;i--){
    			if (cnt+(a[j0]-a[i])>k) break;
    			cnt += a[j0]-a[i];
    			freq ++;    			
    		}
    		if (freq>best){
    			best = freq;
    			bestaj = a[j0];
    		}
    	}
    	cout << best << " " << bestaj << "\n";
    }
    return 0;
}
