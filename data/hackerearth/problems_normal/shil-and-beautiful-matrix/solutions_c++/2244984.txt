const int mod = 1000000000 + 7 ;
const int maxn = 1000000 + 10 ;
const int inf = 2*1e9 ; 
int F[maxn] ;

void pre(){
	F[0] = 1 ; 
	for(int i=1;i<maxn;i++) F[i] = 1LL * F[i-1] * i % mod ;
}
void solve(){
	pre() ;
	int n , m ; cin >> n >> m ;
	vector<int> A ;
	for(int i=1;i<=n;i++) for(int j=1;j<=m;j++) A.push_back(i^j) ;
	sort(all(A)) ;
	int i = 0 ;
	n = A.size() ;
	int ans =  1 ; 
	while(i < n){
		int x = A[i] ;
		int c = 0 ;
		while(i < n && x == A[i]){ c ++ ; i ++ ;}
		ans = 1LL * ans * F[c] % mod ; 
	}	
	cout << ans << endl ;
}
int main(){

	int t ; t = 1 ; 
	while(t --) solve() ;
	return 0 ; 
}
