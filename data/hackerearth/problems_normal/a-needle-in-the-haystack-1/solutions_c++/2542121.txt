bool matches(int *a, int *b)
{
	for(int i=0;i<26;i++)
		if(a[i] != b[i])
			return false;
	return true;
}

void solve()
{
	string pat, text;
	cin>>pat>>text;
	int p[26] = {0}, t[26] = {0};
	for(int i=0;i<pat.size();i++)
		p[pat[i]-'a']++, t[text[i]-'a']++;
	
	if(matches(p, t))
		cout<<"YES\n";
	else {
		int prev = 0;
		for(int i=pat.size();i<text.size();i++) {
			t[text[prev]-'a']--;
			t[text[i]-'a']++;
			if(matches(p, t)) {
				cout<<"YES\n";
				return;
			}
			prev++;
		}
		cout<<"NO\n";
	}
}
