#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,k,a[1001],c;
    cin>>t;
    while(t--)
    {
    	c=0;
    	cin>>n>>k;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		if(a[i]<=0)
    		c++;
    	}
    	if(c<k)
    	cout<<"YES\n";
    	else
    	cout<<"NO\n";
    }
    return 0;
}
