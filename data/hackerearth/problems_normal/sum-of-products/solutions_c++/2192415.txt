#include <bits/stdc++.h>
#define _ ios_base::sync_with_stdio(false);cin.tie(0);
using namespace std;
#define pb push_back
#define pob pop_back
#define pf push_front
#define pof pop_front
#define mp make_pair
#define all(a) a.begin(),a.end()
#define bitcnt(x) __builtin_popcountll(x)
#define MOD 1000000007
#define MAXN 100005
#define eps  1e-9
typedef unsigned long long int uint64;
typedef long long int int64;


int64 f(int64 x){
	x*=(x+1);
	x/=2;
	if(x>=MOD)
	x%=MOD;
	return x;
}

int main(){
	int t,n;
	int64 i,x,y,ans=0;
	cin>>t;
	while(t--){
		cin>>n;
		ans=0;
		int root=sqrt(n);
		for(i=1;i*i<=n;i++){
			x=i;
			y=n/i;
			ans+=x*y;
			if(ans>=MOD)
			ans%=MOD;
			int64 add=f(y);
			add-=f(max(n/(i+1),(int64)root));
			if(add<0)
			add+=MOD;
			add*=i;
			if(add>=MOD)
			add%=MOD;
			ans+=add;
			if(ans>=MOD)
			ans%=MOD;
		}
		cout<<ans<<endl;
	}
	return 0;
}
