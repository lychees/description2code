#include <iostream>
using namespace std;

int solution_1(int n) {
	return (n&0x1) ? 0 : 1;
}

int main()
{
    int T;
    cin >> T;
    while(T--) {
    	int i,n;
    	cin >> i >> n;
    	if(i==1) 
    		cout << solution_1(n) << endl;
    	else 
    		cout << (n ? solution_1(n-1) : i) << endl;
    }
}
