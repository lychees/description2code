def bin_search(lst, num, length):
    first = 0
    last = length-1
    found = 0
    while (first <= last and not found):
        mid = (first+last)/2
        if lst[mid] >= num:
            if mid != 0:
                if lst[mid-1] < num:
                    found = 1
                    break
            last = mid - 1
        else:
            first = mid + 1
    return mid+1


numBox = int(raw_input())

    Co.append(total)
