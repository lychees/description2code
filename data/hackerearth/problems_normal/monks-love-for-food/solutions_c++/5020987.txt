#include <iostream>
#include <vector>
using namespace std;

int main()
{
	long Q;
	cin >> Q;
	vector<long> foodCost;
	for(long i = 0; i < Q; i++) {
		int type;
		long cost;
		cin >> type;
		if(type == 1) {
			if(foodCost.size() == 0) {
				cout  << "No Food" << endl;
			}else {
				long costOutput = foodCost.back();
				foodCost.pop_back();
				cout << costOutput << endl;
			}
		}else {
			cin >> cost;
			foodCost.push_back(cost);
		}
	}
	
    return 0;
}
