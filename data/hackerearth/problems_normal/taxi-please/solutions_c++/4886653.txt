    #include<vector>
    #include<algorithm>
    using namespace std;
    bool com(const pair<pair<int,int>,int>&a ,const pair<pair<int,int>,int>&b)
    {
    return(a.first.first < b.first.first);
    }
    int main()
    {
    int n,m;
    cin>>n>>m;
    pair<pair<int,int>,int> p;
    vector<pair<pair<int,int >,int> > v;
    for(int i=0;i<n;i++)
    {
    int x,y;
    cin>>x>>y;
    p.first.first=x;
    p.first.second=y;
    p.second=i;
    v.push_back(p);
    }
    sort(v.begin(),v.end(),com);
    vector<int> cab(m,0);
    vector<int> cust(n);
    for(int i=0;i<n;i++)
    {
    for(int z=0;z<m;z++)
    {
    if(cab[z]-v[i].first.first <=0)
    cab[z]=0;
    }
    int x=min_element(cab.begin(),cab.end())-cab.begin();
    //cust[v[i].second]=x+1;
    if(cab[x]<=v[i].first.first)
    {
    cust[v[i].second]=x+1;
    cab[x]=v[i].first.first+v[i].first.second;
    }
    else
    cust[v[i].second]=-1;
    }
    for(int j=0;j<n;j++)
    cout<<cust[j]<<" ";
    return 0;
    }
