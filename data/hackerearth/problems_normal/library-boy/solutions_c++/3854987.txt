#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int i,count=0;
		int track[26]={0};
	while(n--)
	{
	
		char str[100];
		cin>>str;
		//int len=strlen(str);
		
		track[str[0]-97]++;
		}
	for(i=0;i<26;i++)
	{
		if(track[i]>0&&track[i]<=10)
		count++;
		else
		{
			if(track[i]%10!=0)
			{
				count=count+(track[i]/10)+1;
				
			}
			else
			count=count+track[i]/10;
		}
		
	}
	cout<<count<<"\n";
	return 0;
}
