#include <iostream>
#include<cstdlib>
#include<cstdio>
#include<algorithm>
#include<string>
using namespace std;
int main()
{
	int t;
	cin>>t;
	t++;
	while(t--)
	{
		string s;
		cin>>s;
		int a[26];
		fill(a,a+26,0);
		int i=0;
		while(s[i]!='\0')
		{
			a[int(s[i]-97)]++;
			i++;
		}
		i=0;
		while(s[i]!='\0')
		{
			if(a[int(s[i]-97)])
			{
				cout<<s[i];
				a[int(s[i]-97)]=0;
			}
			i++;
		}
		cout<<endl;
	}
    return 0;
}
