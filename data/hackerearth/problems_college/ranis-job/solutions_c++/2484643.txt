#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int total[90000];
    	int sum =0;
    	for(int i  =0;i<n;i++)
    	{
    		int x;
    		scanf("%d",&x);
    		sum+=x;
    	}
    	for(int i  =0;i<n;i++)
    	scanf("%d",&total[i]);
    	
    	sort(total,total+n);
    	int s = 0,c = 0;;
    	for(int i = n-1;i>=0;i--)
    	{
    		c++;
    		s+=total[i];
    		if(s>=sum)
    		break;
    	}
    	cout<<c<<endl;
    }
    return 0;
}
