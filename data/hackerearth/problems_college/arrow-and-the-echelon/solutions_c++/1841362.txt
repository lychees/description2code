#include <iostream>
using namespace std;

int main()
{
	int t,n,m,i,j;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int k=n%3;
    	for(m=n-k,j=k;m>0;m-=3,j+=3)
    	{
    		if(j%5==0) break;
    	}
    	if(j%5!=0) cout<<-1<<endl;
    	else {
    		for(i=0;i<m;i++) cout<<5;
    		for(i=0;i<n-m;i++) cout<<3;
    		cout<<endl;
    	}
    }
    return 0;
}
