for i in xrange(int(raw_input())):
    s = raw_input()
    found_start_Star = True
    safe_O_count = 0
    local_safe_O_count = 0
    for ch in s:
        # print ch, local_safe_O_count, found_start_Star
        if ch == 'X':
            local_safe_O_count = 0
            found_start_Star = False
        elif ch == 'O':
            if found_start_Star:
                local_safe_O_count += 1
        elif ch == '*':
            safe_O_count += local_safe_O_count
            local_safe_O_count = 0
            found_start_Star = True

    safe_O_count += local_safe_O_count
    print safe_O_count
