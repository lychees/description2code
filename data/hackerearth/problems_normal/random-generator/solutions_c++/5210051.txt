#include <bits/stdc++.h>

using namespace std;

int a[100005];
string str[]={"NO\n","YES\n"};
int main()
{
   
   
    int t;
    cin>>t;
    while(t--)
    {
        int n,k,p;
        
        cin>>n>>k>>p;
        for(int i=1;i<=n;i++)
        {
            cin>>a[i];
        }
        sort(a+1,a+1+n);
        bool ans=true;
        for(int i=1;i<=n && ans;i++)
        {
            int x=a[i];
            int r=upper_bound(a+i,a+n+1,x+2*p)-a;
            
            ans&=(r-i)<k;
        }   
        cout<<str[ans];
    }
    return 0;
}
