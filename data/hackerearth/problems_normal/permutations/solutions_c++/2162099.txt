#define MAX 210
#define MOD 1000000007
int perm[MAX],pos[MAX];
int adjMat[MAX][MAX];
int mem[MAX][MAX][MAX];
int floyd(int i,int j,int k){
	if(k == 0)	return adjMat[i][j];

	int& res = mem[i][j][k];
	if(res != -1)	return res;

	res = floyd(i,j,k-1) || (floyd(i,k,k-1) && floyd(k,j,k-1)); 
	return res;
}
int main()
{
	ios::sync_with_stdio(false);

	string str;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>perm[i];
		pos[perm[i]] = i;
	}

	for(int i=0;i<n;i++){
		cin>>str;
		for(int j=0;j<n;j++)
			adjMat[i][j] = (str[j] == 'Y');
	}

	memset(mem,-1,sizeof mem);
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
			floyd(i,j,n-1);

	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
			adjMat[i][j] = mem[i][j][n-1];
		
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n;j++)
			if(adjMat[pos[perm[i]]][pos[perm[j]]] && perm[j] < perm[i]){
				swap(pos[perm[i]],pos[perm[j]]);
				swap(perm[i],perm[j]);
			}

	for(int i=0;i<n;i++)
		cout<<perm[i]<<" ";
	cout<<endl;

	return 0; 
}
