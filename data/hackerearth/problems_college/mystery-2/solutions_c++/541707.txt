#include <iostream>
using namespace std;


string reverse(string s)
{
	char temp;
	int l = s.length();
	int i;
	for(i=0;i<(l/2);i++)
	{
		temp = s[i];
		s[i] = s[l-i-1];
		s[l-i-1] = temp;
	}
	return s;
}

int main()
{
    int t;
    cin>>t;
    int a,b;
    string s;
    while(t--)
    {
    	s="";
    	cin>>a>>b;
    	
    	while(a!=0)
    	{
    		s+=char((a%b)+48);
    		a/=b;
    	}
    	cout<<reverse(s)<<"\n";
    }
    return 0;
}
