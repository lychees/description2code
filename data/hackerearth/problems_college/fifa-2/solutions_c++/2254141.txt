#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    long long int *hk=new long long int[n+1];
    long long int *ak=new long long int [n+1];
   long long int color[100001]={0};
    for(long long int i=1;i<=n;i++)
    {
      cin>>hk[i]>>ak[i];
       color[hk[i]]++;
    }
    
    
    int h,a,tmp;
    for(long long int i=1;i<=n;i++)
    {
    	 h=n-1; a=n-1;
    	 tmp=color[ak[i]];
    	h+=tmp;
    	a-=tmp;
    	
    	
    	cout<<h<<" "<<a<<endl;
    }
    return 0;
}
