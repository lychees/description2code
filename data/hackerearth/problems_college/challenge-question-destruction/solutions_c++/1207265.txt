int isleft(int pos)
{
	if(pos+7==10)
	return 10;
	else
	return (pos+7)%10;
}
int isright(int pos)
{
	if(pos+3==10)
	return 10;
	else
	return (pos+3)%10;
}
int main()
{
	ios::sync_with_stdio(false);
	int a[11],i,pos;
	int min=1000;
	for(i=1;i<=10;i++)
 {
	cin>>a[i];
	if(min>a[i])
  {	min=a[i];
 pos=i;
  }
 }
 int left,right;
 left=isleft(pos);
 right=isright(pos);
 cout<<a[pos]<<" ";

 
while(1)
 {
 	if(a[left]<a[right])
 	{
 		cout<<a[left]<<" ";
 		left=isleft(left);
 	}
 	else
 	{
 		cout<<a[right]<<" ";
 		right=isright(right);
 	}
 	if(left==right)
 	break;
 }
 
 return 0;
}
