#include<bits/stdc++.h>
using namespace std;

long long int A,B,C,D,K;

int binarySearch(long long int l, long long int r) {
	long long int num,mid;
	while(l<=r) {
		mid=(l+r)/2;
		num=(A*(mid*mid*mid))+(B*(mid*mid))+(C*mid)+(D);
		if(K==num) {
			return mid;
		}
		else if(K<num) {
			r=mid-1;
		}
		else {
			l=mid+1;
		}
	}
	num=(A*(mid*mid*mid))+(B*(mid*mid))+(C*mid)+(D);
	if(num>K) return mid-1;
	return mid;
}

int main() {
	int T;
	scanf("%d",&T);
	while(T--) {
		scanf("%lld %lld %lld %lld %lld",&A,&B,&C,&D,&K);
		printf("%d\n",binarySearch(1,793701));
	}
	return 0;
}
