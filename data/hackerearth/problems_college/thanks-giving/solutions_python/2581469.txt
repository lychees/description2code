from string import ascii_lowercase
d = {}
for c in ascii_lowercase:
    d[c] = 0
    
s=raw_input()

l=list(s)

for x in l:
    if x in d:
        d[x]=d[x]+1
        
maxKey='a'

for x in d:
    if d[x]>d[maxKey]:
        maxKey=x

if (maxKey in "cubing club" and maxKey in "head rush"):
    print "Thanks both"
elif (maxKey in "cubing club"):
    print "Thanks cubing club"
elif (maxKey in "head rush"):
    print "Thanks head rush"
else:
    print 'none'
