#include <cstdlib>
#include <cstring>
#include <iostream>
using namespace std;
int main() {
  char s[100001];
  int i, sum = 0, c = 0;
  cin >> s;
  while (strlen(s) > 1) {
    for (i = 0; i < strlen(s); i++) {
      sum += (s[i] - '0');
    };
    itoa(sum, s, 10);
    sum = 0;
    c++;
  }
  cout << c;
}
