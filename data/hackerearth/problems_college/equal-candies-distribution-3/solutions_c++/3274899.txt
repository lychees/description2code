#include<bits/stdc++.h>
using namespace std;
int a[100000+4];
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		cin>>n;
		for(int i=0;i<n;++i)
			cin>>a[i];
		sort(a,a+n);
		/*for(int i=0;i<n;++i)
			printf("%d ",a[i]);
		printf("\n");*/
		int bas=a[0];
		for(int i=0;i<n;++i)
			a[i]=a[i]-bas;
		/*for(int i=0;i<n;++i)
			printf("%d ",a[i]);
		printf("\n");*/
		int cnt=0;
		for(int i=0;i<n;++i)
		{
			int temp=a[i];
			cnt+=temp/5;
			temp%=5;
			cnt+=temp/2;
			temp%=2;
			cnt+=temp/1;
			temp=temp%1;
		}
		cout<<cnt<<endl;
	}

}
