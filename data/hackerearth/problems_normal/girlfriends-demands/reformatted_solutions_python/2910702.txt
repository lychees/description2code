def w_h_find(text):
    li = len(text)
    for index, i in enumerate(text):
        if i == " ":
            pos = index
    w = int(text[0:pos])
    l = int(text[pos + 1: li])
    return w, l


demand = raw_input()
tc = input()
inp = []
for i in range(tc):
    inp.append(raw_input())
for tex in inp:
    wi, le = w_h_find(tex)
    m = len(demand)
    wi = wi % m
    le = le % m
    if demand[wi - 1] == demand[le - 1]:
        print "Yes"
    else:
        print "No"
