#include<iostream>
#include<string>
using namespace std;
int main()
{
	int test;
	cin>>test;
	while(test--)
	{
		string s;int a,b;
		cin>>s>>a>>b;
		int x=s.length();
		int ans=0;
		for(int i=0;i<x/2;i++)
		{
			if(s[i]=='/' && s[x-i-1]=='/')
			{
				ans+=(a<b?a:b)*2;
			}
			else if(s[i]=='/' )
			{
				ans+=(s[x-i-1]=='a'?a:b);
			}
			else if(s[x-i-1]=='/')
			{
				ans+=(s[i]=='a'?a:b);
			}
			else if(s[i]!=s[x-i-1])
			{
				ans=-1;
				break;
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
