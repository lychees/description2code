#include <iostream>
#include <stdio.h>
#include <algorithm>

#define MAXN 20

using namespace std;

typedef pair<int,int> ii;

ii v[MAXN];

bool cmp(const ii &a, const ii &b){
    return (a.second>b.second);
}

int main(){
    int n;

    scanf("%d", &n);
    for(int i = 0; i < n; i++){
        scanf("%d", &v[i].first);
    }
    for(int i = 0; i < n; i++){
        scanf("%d", &v[i].second);
        v[i].second *= i;
    }
    sort(v,v+n,cmp);
    long long int res = 0;
    for(int i = 0; i < n; i++){
        v[i].second = 0;
        for(int j = 0; j < n; j++){
            res += v[j].first;
            v[j].first -= v[j].second;
        }
    }
    printf("%lld\n", res);

    return 0;
}
