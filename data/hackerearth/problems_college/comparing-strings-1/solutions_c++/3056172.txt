#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	string a,b;
    	cin>>a>>b;
    	
    	int a1[27]={0},b1[27]={0};
    	
    	for(int i=0;i<a.size();i++){
    		a1[a[i]-'a']++;
    	}
    	for(int i=0;i<b.size();i++){
    		b1[b[i]-'a']++;
    	}
    	
    	int flag=0;
    	for(int i=0;i<27;i++){
    		if(a1[i]!=0 && b1[i]!=0){
    			cout<<"YES"<<endl;
    			flag=1;
    			break;
    		}
    	}
    	
    	if(!flag) cout<<"NO"<<endl;
    }
    return 0;
}
