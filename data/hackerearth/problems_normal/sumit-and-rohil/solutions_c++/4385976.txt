#include <iostream>
#include <string>
#include <map>
using namespace std;

int index = 0;
string a[100];

void fill(string s, int n){
	int m,count;
	map<char, int>c1;
	map<char, int>c2;
	c1.clear();
	for(int i = 0; i < n; i++){
		c1[s[i]]++;
	}
	for(int i = 0; i < index; i++){
		m = a[i].length();
		if(s[0] == a[i][0] && s[n-1] == a[i][m-1]){
			c2.clear();
			for(int j = 0; j < m; j++){
				c2[a[i][j]]++;
			}
			count = 0;
			for(char c = 'a'; c <= 'z'; c++){
				if(c1[c] != c2[c]){
					a[index++] = s;
					return;
				}
				count++;
			}
			if(count == 26){
				return;
			}
		}
		else if(i == index-1){
			a[index++] = s;
			return;
		}
	}
}

int main()
{
    int t,n;
    string s;
    cin >> t;
    t--;
    cin >> s;
    a[index++] = s;
    while(t--){
    	cin >> s;
    	n = s.length();
    	fill(s,n);
    }
    cout << index << endl;
    return 0;
}
