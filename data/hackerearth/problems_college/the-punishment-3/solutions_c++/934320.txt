#include<bits/stdc++.h>

using namespace std;
int main()
{
	long n, t;
	cin>>t;
	while(t--)
	{
		long carry, ans[1000000]={0}, k, i, j, first, sec, temp;
		cin>>n>>k;
		first=k;
		sec=n-k;
		temp=min(first, sec);
	
		if(temp == 1)
		for(i=1; i<=n-1; i++)
			ans[i]=1;
		else	
		{
			for(i=1, j=n-1; i<temp; i++, j--)
				ans[i]=ans[j]=i;
			for(i=temp; ans[i]!=temp-1; i++)
				ans[i]=temp;
		}
		carry=0;
		for(i=n-1; i>=1; i--)
		{
			temp=ans[i];
			ans[i]=(ans[i]*81 + carry)%10;
			carry=(temp*81 + carry)/10;	
		}
		ans[0]=carry;	
		if(ans[0])
			cout<<ans[0];
		for(i=1; i<=n-1; i++)
			cout<<ans[i];
		
		cout<<endl;
	}
	return 0;
}
