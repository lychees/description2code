#include<bits/stdc++.h>
using namespace std;
long long hcf;
long long gcd(long long  a,long long b)
	{
   	long long temp;    
   	while(b)
   		{
           temp=a;
           a=b;
           b=temp%b;
   		}
   		return a;
	}

long long search(long long lp,long long up,long long a,long long b,long long n)
  {//cout<<lp<<" "<<up<<endl;
  	long long mp=(lp+up)/2;
  	if(n==mp/a+mp/b-mp/hcf)
  	  {
  	  	if(mp%a==0 || mp%b==0)
  	  	 return mp;
  	  	return search(lp,mp-1,a,b,n);
  	  }
  	if(n<mp/a+mp/b-mp/hcf)
  	return search(lp,mp-1,a,b,n);
  return search(mp+1,up,a,b,n);
  }

int main()
 {
 	int t;
 	scanf("%d",&t);
 	while(t--)
 	  {long long a,b,n;
 	  	scanf("%lld",&a);
 	  	scanf("%lld",&b);
 	  	scanf("%lld",&n);
 	  	hcf=(a*b)/gcd(a,b);
 	  	long long up=max(a,b)*n+1;
 	  	long long lp=1;
 	  	printf("%lld\n",search(lp,up,a,b,n));
 	  //	cout<<search(lp,up,a,b,n)<<endl;
 	  }
 }
