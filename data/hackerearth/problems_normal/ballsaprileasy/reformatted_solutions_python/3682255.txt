from sys import stdin
n = int(stdin.readline())
a = map(int, stdin.readline().strip().split())
ans = 0
a.sort()
cur = 1
for i in a:
    while True:
        if i <= cur*cur:
            break
        ans += cur*cur
        cur += 1
# print i,"used in ",cur
    ans += cur*cur-i
    cur += 1
print ans
