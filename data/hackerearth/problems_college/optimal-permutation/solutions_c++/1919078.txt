#include <cstdio>

int main()
{
	int t, n;
	scanf("%d", &t);
	while (t--)
	{
		scanf("%d", &n);
		if (n == 1) printf("0\n");
		else
		if (n == 2) printf("1\n");
		else
		if (n == 3) printf("3\n");
		else
		if (n == 4) printf("7\n");
		else
		{
			long long ans = n - 1 + 2 * (n - 2);
			int v = n - 4;
			for (int i = 2, lim = n / 2; i < lim; ++i, v -= 2)
				ans += 2 * v;
			if (n & 1)
				ans += v;
			printf("%lld\n", ans);
		}
	}
	return 0;
}
