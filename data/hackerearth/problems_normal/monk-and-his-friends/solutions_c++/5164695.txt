#include<bits/stdc++.h>
using namespace std;
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
  	unsigned long long n,m,i,temp;
  	map<unsigned long long, bool> m1;
  	cin>>n>>m;
  	for(i=0;i<n;i++)
  	{
  		cin>>temp;
  		m1[temp]=true;
  	}
  	map<unsigned long long,bool> :: iterator it;
	for(i=0;i<m;i++)
  	{
  		cin>>temp;
  		it=m1.find(temp);
  		if(it!=m1.end())
  			cout<<"YES\n";
  		else
  		{
			cout<<"NO\n";
			m1[temp]=true;
  		}
  		
  	}
  
		
  }
    return 0;
}
