#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    int t;
    cin>>t;
    int n;
    long int count,ans,i;
    while(t--)
    {
        cin>>n;
        count=1;
        ans=2;
        i=3;
        if(n<=2)
           {
               cout<<"1\n";
               continue;
           }
        else
        {
            while(ans<n)
            {
                ans+=i;
                i++;
                count++;
            }
        }
        cout<<count<<"\n";
    }
    return 0;
}
