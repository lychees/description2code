vector<pair<double, double> > A;

double F(double alpha)
{
    double cs = cos(alpha);
    double sn = sin(alpha);
    double minX = 1e47;
    double maxX = -1e47;
    double minY = 1e47;
    double maxY = -1e47;
    FOR(i,0,A.size())
    {
        double x = A[i].first * sn + A[i].second * cs;
        double y = A[i].first * cs - A[i].second * sn;
        minX = min(minX , x);
        maxX = max(maxX , x);
        minY = min(minY , y);
        maxY = max(maxY , y);
    }
    return max(maxX - minX , maxY - minY);
}

int main() {

	int n;
	cin >> n;

    FOR(i,0,n)
    {
        double x , y;
        cin >> x >> y;
        A.push_back(MP(y , x));
    }

    sort(ALL(A));
    FOR(i,1,A.size())
    {
        A[i].first -= A[0].first;
        A[i].second -= A[0].second;
    }
    A[0] = MP(0,0);
    FOR(i,0,n)
        swap(A[i].first , A[i].second);

    double res = 1e47;

    FOR(it1,0,ITER)
    {
        double L = Pi / ITER * it1;
        double R = Pi / ITER * (it1 + 1);
        FOR(it2 , 0 , MAGIC)
        {
            double X1 = (2 * L + R) / 3;
            double X2 = (L + 2 * R) / 3;
            if (F(X1) < F(X2)) R = X2;
            else L = X1;
        }
        res = min(res , F(L));
    }

    printf("%.4f\n" , res);

    return 0;
}
