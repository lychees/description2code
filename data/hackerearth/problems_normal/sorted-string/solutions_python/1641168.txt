for test in range(input()):
	mystr = raw_input()
	mydict = {}
	for char in mystr:
		if mydict.get(char, -1) == -1:
			mydict[char] = 1
		else:
			mydict[char] += 1
	sortedlist = [(k, v) for k, v in sorted((v, k) for k, v in mydict.items())]
	finallist = [k*v for k, v in sortedlist]	
	print  ''.join(finallist)
