#include<iostream>
#include<algorithm>
#include<vector>

using namespace std;

vector<long long>v;

int main(){

    int t;
    cin >> t;
    while(t--) {
        v.clear();
        long long n,i,dp[15][15],temp;
        cin >> n;
        temp = n;
        while(temp > 0) {
            v.push_back(temp % 10);
            temp /= 10;
        }
        if(n == 0) v.push_back(0);
        reverse(v.begin(),v.end());
        for(i=0;i<v.size();i++) {
            if(i == 0) {
                dp[0][i] = 1;
                if(v[i] > 3) dp[1][i] = v[i] - 1;
                else dp[1][i] = v[i];
            }
            else {
                dp[0][i] = 1;
                if(v[i] > 3) dp[1][i] = v[i] - 1;
                else dp[1][i] = v[i];
                dp[1][i] += (dp[1][i - 1] * 9);
            }
        }
        cout << dp[1][v.size() - 1]  << "\n";
    }
    return 0;






}
