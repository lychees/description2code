from math import pow

def getLastDigit(n):
    return n%10

tc = int(raw_input())

for i in range(tc):
    s = raw_input().strip()
    l = s.split()
    a,b = int(l[0]),int(l[1])
    last_digit = getLastDigit(a)
    fourth_power = b%4
    if fourth_power == 0:
        fourth_power = 4
    ans = int(pow(last_digit,fourth_power))
    print getLastDigit(ans)
