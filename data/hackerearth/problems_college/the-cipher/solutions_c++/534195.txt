#include <iostream>
#include <cstdio>
#include<math.h>
using namespace std;

int main()
{
        int test_cases;
        scanf("%d",&test_cases);
        while(test_cases > 0)
        {
                int N,arr[10009],i=0;
                scanf("%d",&N);
                while(i<N)
                {
                        scanf("%d",&arr[i]);
                        i = i+1;
                }
                int A[10];
                for(i=0;i<N;i++)
                {
                	int j=0,c,k;
                	for(c=7;c>=0;c--)
                	{
                		k = arr[i] >> c;
                          if (k & 1)
                             A[j]=1;
                          else
                             A[j]=0;
                          j++;   
                	}
                	for(c=0;c<8;c++)
                	{
                		if(A[c]==1)
                		 A[c]=0;
                		else if(A[c]==0)
                		 A[c]=1;
                	}
                	int ans=0;
                	for(c=0;c<8;c++)
                	{
                		ans+=A[c]*pow(2,j-c-1);
                	}
             
                	char ch=(char)ans;
                	printf("%c",ch);
                }
                printf("\n");
                test_cases--;
        }
        return 0;
}
