def opr(n):
    x1, r1 = divmod(n, 5)
    x2, r2 = divmod(r1, 2)
    return x1 + x2 + r2


for _ in xrange(int(raw_input())):
    raw_input()
    a = sorted(map(int, raw_input().split()))
    ans = 0
    c = 0
    for i in xrange(1, len(a)):
        c += a[i] - a[i-1]
        ans += opr(c)
    print ans
