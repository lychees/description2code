
#include <iostream>

                j++;
        }

    	}else{*/

                for(ll i=0;i<256;i++) {

                        k++;
                    }
                }
                c = k - 1;
            }
            
            else {
                for(ll i=0;i<256;i++) {

                        k++;

                    if((a[i]%2 != 0) && (a[i] != 1))k++;
                }
                  c = k-1;
