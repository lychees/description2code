#include <iostream>
#include <string>
#include <cstring>
#include <algorithm>

using namespace std;

int cnt[26];
bool cmp(int a, int b)
{
	return (cnt[a] == cnt[b] ? a > b : cnt[a] < cnt[b]);
}

int main()
{
	ios::sync_with_stdio(false);
	int idx[26];
	string s;
	int t;
	
	cin >> t;
	while (t--)
	{
		cin >> s;
		memset(cnt, 0, sizeof(cnt));
		for (int i = 0; i < 26; ++i)
			idx[i] = i;
		for (int i = 0; i < s.length(); ++i)
			cnt[s[i] - 'a']++;
		sort(idx, idx + 26, cmp);
		for (int i = 0; i < 26; ++i)
			cout << char(idx[i] + 'a') << " ";
		cout << endl;
	}
    
    return 0;
}
