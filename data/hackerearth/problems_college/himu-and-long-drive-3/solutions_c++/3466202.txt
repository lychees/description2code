#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    int t;
    ll n;

    cin >> t;
    while (t--) {
        cin >> n;
        if (n <= 2) {
            cout << n << endl;
            continue;
        }
        ll r = log2(n);
        ll large = pow(2, r);
        cout << (large ^ (large - 1)) << endl;
    }
    return 0;
}
