# print 'Hello World!'

from collections import defaultdict

N = int(raw_input())
books = []
for i in range(N):
	books.append(raw_input())
	
shelves = defaultdict(int)
for book in books:
	shelves[book[0]] += 1
	
n_shelves = sum([(v-1)/10+1 for v in shelves.values()])

print n_shelves
