#include <bits/stdc++.h>
using namespace std;



int main(){
	int t;
	scanf("%d",&t);
	int aa=1;
	while(t--){
		char a[105],b[105];
		scanf("%s",a);
		scanf("%s",b);
		int c[3]={};
		bool ans=true;
		int d[2]={};
		int l = strlen(a);
		for(int i=0;i<l;i++){
			if(a[i]=='0')c[0]++;
			if(b[i]=='0')d[0]++;
			if(a[i]=='1')c[1]++;
			if(b[i]=='1')d[1]++;
			if(a[i]=='?')c[2]++;
		}
		int steps=0;
		int s1=0,s2=0;
		int sq1=0,sq0=0,s01=0;
		int n1=d[1]-c[1];
		if(n1>=0){
			if(c[2]>=n1){
				c[2]-=n1;
				s1+=n1;
				sq1+=n1;
				n1=0;
			}
			else {
				
				s1+=c[2];
				sq1+=c[2];
				n1-=c[2];
				c[2]=0;
			}
			if(n1>0){
				if(c[0]>=n1){
					c[0]-=n1;
					s1+=n1;
					s01+=n1;
					n1=0;
				}
				else {
				
					s1+=c[0];
					s01+=c[0];
					n1-=c[0];
						c[0]=0;
				}
			}
			if(c[0]+c[2]==d[0]){
				s2+=c[2];
				sq0+=s2;
				steps=s1+s2;
			}else{
			
				ans=false;
				steps=-1;
			}
			
		}
		else{
			ans=false;
			steps=-1;
		}
		int swap=0;
		if(ans){
			for(int i=0;i<l&&sq1>0;i++){
				if(a[i]=='?'&&b[i]=='1'){
					a[i]='1';
					sq1--;
				}
			}
			for(int i=0;i<l&&sq0>0;i++){
				if(a[i]=='?'&&b[i]=='0'){
					a[i]='0';
					sq0--;
				}
			}
			for(int i=0;i<l&&s01>0;i++){
				if(a[i]=='0'&&b[i]=='1'){
					a[i]='1';
					s01--;
				}
			}
			for(int i=0;i<l&&sq0>0;i++){
				if(a[i]=='?'){
					a[i]='0';
					sq0--;
				}
			}
			for(int i=0;i<l&&sq1>0;i++){
				if(a[i]=='?'){
					a[i]='1';
					sq1--;
				}
			}
			for(int i=0;i<l;i++){
				if(a[i]!=b[i]){
					swap++;
				}
			}
			steps+=swap/2;
			printf("Case %d: %d\n",aa,steps);
		}else{
			printf("Case %d: -1\n",aa);
		}
		aa++;
	}
}
