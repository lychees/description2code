#include <bits/stdc++.h>
using namespace std;

vector <bool> p(10001,true);
void f()
{
	p[0]=p[1]=false;
	for(int i=0;i<p.size();i++)
	{
		if(p[i])
		{
			for(int j=i*i;j<p.size();j+=i)
				p[j]=false;
		}
	}
}

int main()
{
	f();
	int n,k;
	cin>>n>>k;
	int a[n][n],b[n][n];
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
		{
			cin>>a[i][j];
			b[i][j]=0;
		}
	long long mi=1000000,l=0;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			l=0;
			while(!p[a[i][j]]&&a[i][j]<=10000)
			{
				a[i][j]+=k;
				l++;
			}
			if(a[i][j]>10000)
			{
				b[i][j]=10000000;
			}
			else
				b[i][j]=l;
		}
	}
	int x,y;
	for(int i=1;i<n-1;i++)
	{
		for(int j=1;j<n-1;j++)
		{
			if(b[i][j]+b[i-1][j-1]+b[i-1][j+1]+b[i+1][j-1]+b[i+1][j+1]<mi)
			{
				mi=b[i][j]+b[i-1][j-1]+b[i-1][j+1]+b[i+1][j-1]+b[i+1][j+1];
				x=i+1,y=j+1;
			}
		}
	}
	if(mi!=1000000)
		cout<<"yes\n"<<mi<<endl<<x<<" "<<y;
	else
		cout<<"no";
	return 0;
}
