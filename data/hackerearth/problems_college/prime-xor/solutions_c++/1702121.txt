#include<stdio.h>
#include<math.h>
bool arr[1000001]={0};
int isprime(int num)
    {
	int max;
	if(num<=1000000)
	  {
		return arr[num];
	  }
	else
	  {
		max=sqrt(num);
		for(int i=2;i<=max;++i)
		   {
			if(num%i==0)
				return 1;
		   }
		return 0;
  	  }
    }

int countprime(int a,int b)
    {
	int i,count=0;
	for(i=a;i<=b;++i)
	  {
		if(isprime(i)==0)
		  ++count;
	  }
	//printf("%d",count);
	return count;
    }

int main()
    {
	int t,a,b;
	int i,j;
	arr[0]=1;
	arr[1]=1;
	for(i=2;i<=1000000;++i)
	   {
		if(arr[i]==0)
		  {
			for(j=2*i;j<=1000000;j+=i)
				arr[j]=1;
		  }
	   }
	scanf("%d",&t);
	while(t-->0)
	  {
		scanf("%d%d",&a,&b);
		if(a!=1&&a!=2)
		  {
			printf("0\n");
		  }	
		else
		 {
			printf("%d\n",countprime(3,b));
		 }
	  }
    }
