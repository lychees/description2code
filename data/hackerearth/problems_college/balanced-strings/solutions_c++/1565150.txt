#include <iostream>
#include<string.h>

using namespace std;
int a[26]={0};
int main()
{
	
	
	int t,i,j,k,freq;
	char  s[100000];
	
	cin>>t;
	for(i=0;i<t;i++)
	{
		cin>>s;
		
		for(j=0;j<strlen(s);j++)
		{
			a[s[j]-'a']++;
			
			
		}
		k=1;
		freq=a[0];
		for(j=0;j<26;j++)
		{
			if(a[j]!=freq)
			k=0;
			
			a[j]=0;
			
		}
		
		
		if(k)
		 cout<<"Yes"<<endl;
		 else
		cout<<"No"<<endl;
		 
	
	}
   
    return 0;
}
