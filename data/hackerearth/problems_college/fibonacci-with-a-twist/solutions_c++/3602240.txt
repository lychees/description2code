#include <bits/stdc++.h>
#define ll long long
const int M = 1e9+7;
using namespace std;
void matmult(ll a[][2],ll b[][2],ll c[][2],ll M)
{
	ll i,j,k;
	for(i=0;i<2;i++)
	{
		for(j=0;j<2;j++)
		{
			c[i][j]=0;
			for(k=0;k<2;k++)
			{
				c[i][j]=(c[i][j]%M+a[i][k]%M*b[k][j]%M)%M;
			}
		}
	}
}
void matpow(long long Z[][2],ll n,long long ans[][2],ll M)
{
	long long temp[2][2];
	ans[0][0]=1;
	ans[1][0]=0;
	ans[0][1]=0;
	ans[1][1]=1;
	ll i,j;
	while(n>0)
	{
		if(n&1)
		{
			matmult(ans,Z,temp,M);
			for(i=0;i<2;i++)
			for(j=0;j<2;j++)
			ans[i][j]=temp[i][j];
		}
		matmult(Z,Z,temp,M);
		for(i=0;i<2;i++)
		for(j=0;j<2;j++)
		Z[i][j]=temp[i][j];
		n/=2;
	}
	return;
	
}
