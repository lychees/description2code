#include <iostream>
#include<cstdio>

using namespace std;

int main()
{
	int nT;
	scanf("%d",&nT);
	while(nT--)
	{
		int l;
		scanf("%d",&l);
		int A[l];
		for(int i =0;i<l;i++)
		scanf("%d",&A[i]);
		unsigned long long int max_cst =0;
		unsigned long long int dp[l+1];
		for(int i =0;i<=l;i++)
		dp[i]=0;
		dp[0]=0;
		dp[1]=A[0];
		for(int i =2;i<=l;i++)
		{
			for(int j =1;j<=i;j++)
			{
				
					dp[i]=max(dp[i],A[j-1]+dp[i-j]);
					max_cst = max(max_cst,dp[i]);
			}
		}
		printf("%llu\n",max_cst);
		
	}
	
	return 0;
}
