t = int(raw_input().strip())
for _ in xrange(t):
    n = int(raw_input())

    d = -1
    for a in xrange(0, n+1, 3):
        if (n-a) % 5 == 0:
            d = a
    s = ''
    if d != -1:
        for i in xrange(d):
            s += '5'
        for i in xrange(n-d):
            s += '3'
        print s
    else:
        print '-1'
