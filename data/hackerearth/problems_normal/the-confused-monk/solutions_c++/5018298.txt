#include<bits/stdc++.h>
#define ll long long
using namespace std;
const ll mod = 1e9 + 7;
ll ar[55];
ll gcd(ll a,ll b)
{
    if(b == 0)
    return a;
    else
    return gcd(b,a%b);
}
ll mod_expo(ll a,ll b)
{
    ll ans=1;
    while(b!=0)
    {
        if(b%2)
        {
            ans *= a;
            ans %= mod;
        }
        a *= a;
        a %= mod;
        b /= 2;
    }
    return ans;
}
int main()
{
    //freopen("i.txt","r",stdin);
    int n;
    cin>>n;
    ll tot_product =1;
    for(int i=0;i<n;i++)
    {
        cin>>ar[i];
        tot_product  *= ar[i];
        tot_product %= mod;
    }
    ll tot_gcd = ar[0];
    for(int i=1;i<n;i++)
    {
        tot_gcd = gcd(tot_gcd,ar[i]);
    }
    //calculating tot_product raise to the power tot_gcd
    ll ans = mod_expo(tot_product,tot_gcd);
    cout<<ans<<endl;
    return 0;
}
