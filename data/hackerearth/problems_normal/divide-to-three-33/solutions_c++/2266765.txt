#include <iostream>
#include<math.h>
using namespace std;
int main() 
{
	int n,i,a,max=-1,sum=0;
	cin>>n;
	if(n==3)
	{   for(i=0;i<3;i++)
	    {   cin>>a;
	        if(a>max)
	            max=a;
	    }
	    cout<<max;
	    return 0;
	}
	else
	 {  for(i=0;i<n;i++)
	        {   cin>>a;
	            sum+=a;
	        }
	    float sum1 = (float)sum/3;
	    
	    int result=ceil(sum1);
	    cout<<result;
	 }
	return 0;
}
