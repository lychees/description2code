#include<iostream>
#include<iomanip>
#include<algorithm>
#include<stdlib.h>
using namespace std;
int main(void)
{
    int t,n,i;
     int *dishes,raghu,sayan;
    cin >> t;
    while(t--)
    {
        cin >> raghu >> sayan >> n;
        dishes=(int *)malloc(sizeof( int)*n);
        for(i=0;i<n;i++)
        {
            cin >> dishes[i];
        }
        sort(dishes,dishes+n);
        for(i=0;i<n;i++)
        {
        	raghu-=dishes[i];
        	sayan-=dishes[i];
        	if(raghu<0&&sayan<0)
        	{
        		cout << "Tie" << endl;
        		break;
        	}
        	if(raghu < 0)
        	{
        		cout << "Sayan Won" << endl;
        		break;
        	}
        	if(sayan < 0)
        	{
        		cout << "Raghu Won" << endl;
        		break;
        	}
        }
        if(i==n)
        	cout << "Tie" << endl;
    }
    return 0;
}
