#include <iostream>
using namespace std ; 
int main()
{
	
	int t ; 
	cin >> t; 
	while ( t--)
	{
		  int n ;
		  cin >> n  ;  
		  int i ; 
		  char c ; ;
		  string p ; 
		   int a[26] = {0} ; 
		    
		  for ( i = 0; i< n ; i++)
		  {
		  	cin >> p ; 
		  	c = p[0] ; 
		    a[c-'a']++ ; 	
		  }
		  int r = 0; 
		  for ( i = 0; i< 26 ; i++)
		  {
		  	if ( a[i] >=2)
		  	{
		  		r = 1  ; 
		  		break ;
			  }
		  }
		  if( r == 0 )
		  {
		  	cout << "YES" <<endl; 
		  }
		  else
		  {
		  	cout << "NO" <<endl; 
		  }
		  
		
	}
	return 0 ; 
}
