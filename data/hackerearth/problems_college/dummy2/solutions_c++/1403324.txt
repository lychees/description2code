#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		vector<char> v;
		string a; cin>>a; int l=a.size();
		for(int i=0; i<l; i++){
			if(a[i]>'0' && a[i]<='9'){
				int x=a[i]-'0'-1;
				int ff=v.size();
				for(int i=0; i<x; i++){
					for(int j=0; j<ff; j++){
						v.push_back(v[j]);
					}
				}
			}
			else v.push_back(a[i]);
			
		}
		int g=v.size();
		for(int i=0; i<g; i++) cout<<v[i];
		cout<<endl;
	}
}
