#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,a;
    vector <int> evenvec,oddvec;
    long long sumeven=0, sumodd=0;
    cin>>n;
    while (n--)
    {
    	cin>>a;
    	if (a%2==0)
    	{
    	 evenvec.push_back(a);
    	 sumeven+=a;
    	}
    	else
    	{
    	 oddvec.push_back(a);
    	 sumodd+=a;
    	}
    }
    sort(evenvec.begin(),evenvec.end());
    sort(oddvec.begin(),oddvec.end());
    for (int i=0; i<evenvec.size();i++)
    cout<<evenvec[i]<<" ";
    cout<<sumeven<<" ";
    for (int i=0; i<oddvec.size();i++)
    cout<<oddvec[i]<<" ";
    cout<<sumodd;
    
    return 0;
}
