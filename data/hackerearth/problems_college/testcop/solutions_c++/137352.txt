#include<stdio.h>

int gcd(int a,int b)
{
 if(b==0)
  return a;
 else 
  return gcd(b,a%b);
}

int main()
{
 int t,n,*a;
 scanf("%d",&t);
 while(t--)
 {
  scanf("%d",&n);
  int b;
  a=new int[n];
  for(int i=0;i<n;i++)
  {
   scanf("%d",&a[i]);
   if(i==0)
    b=a[i];
   else 
    b=gcd(b,a[i]);
  }
  delete[] a;
  if(b==1)
   printf("YES\n");
  else 
   printf("NO\n");
 }
 return 0;
}
