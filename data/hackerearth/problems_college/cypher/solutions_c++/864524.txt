#include <iostream>
#include <string>

using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--) {
    	string s;
    	cin>>s;
    	
    	for(int i=0;i<s.length();i++) {
    		int a = (int)s[i];
    		a += i;
    		if(a>122) a -= 26;
    		cout<<(char)a;
    	}
    	
    	cout<<endl;
    	
    }
    return 0;
}
