tc=int(input())
while tc>0:
    dic={'a':0,'b':0,'c':0,'d':0,'e':0,'f':0,'g':0,'h':0,'i':0,'j':0,'k':0,'l':0,'m':0,'n':0,'o':0,'p':0,'q':0,'r':0,'s':0,'t':0,'u':0,'v':0,'w':0,'x':0,'y':0,'z':0}
    s=raw_input()
    for i in range(len(s)):
        dic[s[i]]=dic[s[i]]+1
    f=1
    c=dic['a']
    dc=list(dic.values())
    for i in range(1,len(dc)):
        if dc[i]!=c:
            f=0
            break
    if f==1:
        print 'Yes'
    else:
        print 'No'
    tc=tc-1
