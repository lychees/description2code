#include <iostream>
using namespace std;

int main()
{
    long long int n;
   cin >> n;
   long long int team[n][2];
    for(int i=0;i<n;i++){
        for(int j=0;j<2;j++){
        	cin >> team[i][j];
        }   
  }
   
   long long int team_home[n],team_away[n];
       for(int i=0;i<n;i++)
       {
       	   team_home[i]=n-1;
       	   team_away[i]=n-1;
       	  for(int j=0;j<n;j++){
              if(i==j)
                continue;
                else{
                	
                   if(team[j][0]==team[i][1]){
                   	    team_away[i]--;
                   	    team_home[i]++;
                   }
       	  		  
       	  		  
       	  	}  
       	  }
       }
       for(int i=0;i<n;i++){
       	    cout << team_home[i]<< " " << team_away[i]<< "\n";
       }
     
     
    return 0;
}
