#include <iostream>
#include<string>
using namespace std;

int main()
{
    string str,str1;
    long long int n,i,a=0,s=0;
    cin>>str>>n;
    for(i=0;i<n;i++)
    {
    	cin>>str1;
    	if(str1.length()==str.length())
    	{
    		if(str[0]==(str1[0]))
    		s++;
    		if((str[0]+3)==str1[0])
    		a++;
    	}
    }
    cout<<s<<" "<<a;
    return 0;
}
