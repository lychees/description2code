#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int dp[100000];
int func(int n){
	if(n == 0)return 0;
	if(n < 0)return 9999999;
	if(dp[n]!=-1) return dp[n];
	int ans =0;
	ans = 1 + min(func(n-2),min(func(n-3) , min(func(n-5) , func(n-7))));
	return dp[n] = ans;
}

int main()
{
	int t;
	cin >> t;
	while(t--){
		memset(dp,-1,sizeof(dp));
		int n;
		cin >> n;
		int res = func(n);
		cout << res << endl;
	}
    return 0;
}
