#include <iostream>
using namespace std;
int hcf(int a, int h) 
{
    int temp; 
    while(1)
    {
        temp = a%h;
        if(temp==0)
        return h;
   	    a = h;
        h = temp;
    }
};
int main()
{
	int T, A[100000][4], j=0,i=0,a,b,c,flag;
    cin>>T;
    for(i=0;i<T;i++)
    {
    	for(j=0;j<3;j++)
    		cin>>A[i][j];
    }
    for(i=0;i<T;i++)
    {
    	flag=0;
    	a=A[i][0];
    	b=A[i][1];
    	c=A[i][2];
    	if(b<a)
    	{
    		a=a+b;
    		b=a-b;
    		a=a-b;
    		if(c<a)
    		{
    			a=a+c;
    			c=a-c;
    			a=a-c;
    		}
    	}
    	else
    	{
    		if(c<a)
    		{
    			a=a+c;
    			c=a-c;
    			a=a-c;
    		}
    	}
    	flag=hcf(a,b);
    	if(flag==1)
    	{
    		flag=hcf(b,c);
    		if(flag==1)
    			flag=hcf(a,c);
    	}
    	a=A[i][0]*A[i][0];
    	b=A[i][1]*A[i][1];
    	c=A[i][2]*A[i][2];
    	if(flag==1)
    	{
    		if(a+b==c || b+c==a || c+a==b)
    			cout<<"YES"<<endl;
    		else
    			cout<<"NO"<<endl;
    	}
    	else
    		cout<<"NO"<<endl;
    }
    return 0;
}
