#include <bits/stdc++.h>
#define min(X,Y) ((X) < (Y) ? (X) : (Y))
#define max(X,Y) ((X) > (Y) ? (X) : (Y))
using namespace std;
int main()
{
    int test;
    scanf("%d",&test);
    while(test--)
    {
        int Food,Gold,S,P,C,lal,Samurai,Paladin,Champion,temp=-1;
        scanf("%d%d%d%d%d%d",&Food,&Gold,&S,&P,&C,&lal);
        for(Samurai=0;Samurai*100<=Food;Samurai++)
        {
            for(Paladin=0;Paladin*125+100*Samurai<=Food && Paladin*50<=Gold;Paladin++)
            {
                int Food_Left=(Food-100*Samurai-125*Paladin);
                int Gold_Left=(Gold-50*Paladin);
                Champion=min(Food_Left/50,Gold_Left/100);
                temp=max(temp,S*Samurai+P*Paladin+C*Champion);
            }
        }
        if(temp>lal)
        {int res=temp-lal;
        printf("%d\n",res);}
        else
        printf("-1\n");
    }
    return 0;
}
