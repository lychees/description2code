#include <cstdio>
#include <cstring>
using namespace std;
bool ispalindrome(char *s)
{
    int n = strlen(s);
    int i=0,j=n-1;
    while(i<=j)
    {
	if(s[i]!=s[j])
	    return false;
	i++;
	j--;
    }
    return true;
}
int main()
{
    int t;
    scanf("%d",&t);
    char s[51];
    while(t--)
    {
	scanf("%s",s);
	int n = strlen(s);
	int ans = 2*n;
	for(int i=0;i<strlen(s);i++)
	{
	    if(ispalindrome(s+i))
	    {
		ans = n+i;
		break;
	    }
	}
	printf("%d\n",ans);

    }
    return 0;
}
