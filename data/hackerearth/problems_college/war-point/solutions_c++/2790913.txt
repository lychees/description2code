#include<iostream>
#include<math.h>
#include<cstdlib>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--) {
    long long int n,i=0,j=0,small;
    cin>>n;
    long long int arr[n];
        for(i=0;i<n;i++) {
            cin>>arr[i];
        }

        for(i=1;i<n;i++)
        {
            arr[i]=arr[i]+arr[i-1];
        }
        small=abs(arr[0]-(arr[n-1]-arr[0]));
        for(i=1;i<n-1;i++) {
            j=abs(arr[i]-(arr[n-1]-arr[i]));
            if(small>j)
            {
                small=j;
            }
        }
        cout<<small<<endl;
    }
    return 0;
}
