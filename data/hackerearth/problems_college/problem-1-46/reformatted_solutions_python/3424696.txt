from string import ascii_lowercase
A = ord("a")
ad = {}
for c in ascii_lowercase:
    i = ord(c) - A
    ad[c] = 2 ** i
n = int(raw_input())
t = (n + 1) * [0]
s = raw_input().strip()
for i in range(n):
    t[i+1] = t[i] ^ ad[s[i]]
for _ in range(int(raw_input())):
    l, r = map(int, raw_input().split())
    if r - l < 1:
        print "NO"
        continue
    x = bin(t[l-1] ^ t[r]).count("1")
    print "NO" if x > 2 else "YES"
