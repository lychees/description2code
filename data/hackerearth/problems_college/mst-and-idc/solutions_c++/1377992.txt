#include<bits/stdc++.h>

using namespace std;

int main() {
	int t;scanf ("%d",&t);
	while(t--){
		int n,i;
		scanf ("%d",&n);
		int a[n];
		for (i =0; i <n; i++){
			scanf ("%d",&a[i]);
		}
		int sum =0;
		sort(a,a+n);
		for (i =0; i< n-1; i++){
			sum += (a[i+1]-a[i]);
		}
		printf("%d\n",2*sum);
	}
	return 0;
}
