#include<iostream>
using namespace std;
long long int N,*A;
long long int cnt=0;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        cin>>N;
        A=new long long int[N];
        for(int i=0;i<N;i++)
            cin>>A[i];
        for(int j=N-1;j>=0;j--)
        {
            for(int i=0;i<=j;i++)
                if(A[i]==A[j])
                cnt++;
        }
        cout<<cnt<<endl;
        cnt=0;
        delete A;
    }
}
