#include<cstdio>
using namespace std;
inline int ck(int a,int b)
{
	int ns=0;
	for(int i=0;i<10;i++);

	while(!(a==1 && b==1))
	{
		if(a==b) 
		{
			ns=20000000; break;
		}
		if(a<b) b=b-a;
		else a=a-b;
		ns++;
	}
//	sum(4,6);
	return ns;
}
int main()
{//	sum(3,9);
	int ntc; scanf("%d",&ntc);
	//sum(6,2);
	while(ntc--)
	{
		int L; scanf("%d",&L);
		int ans=200000000;
		int ta;
		if(L==1)
		{	//sum(1,6);
			printf("0\n"); continue;
		}
		//sum(9,45);
		for(int i=1;i<L;i++)
		{    //sum(5,2);
			int ta=ck(i,L-i);
		//	sum(7,9);
			if(ta<ans) ans=ta;
		}
		
		printf("%d\n",ans+1);
	}
}
