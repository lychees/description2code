#include <iostream>
#define ll long long int
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    ll t,n,a,b,c,num;
    cin >> t;
    while(t--){
    	cin >> a >> b >> c >> n;
    	num = a;
    	ll arr[n];
    	if(n==1){
    		if(a==0)
    			cout << "YES" << endl;
    		else
    			cout << "NO" << endl;
    	}else{
    		arr[0]=a;
    		arr[1]=b;
    		for(int i=2;i<n;i++){
    			arr[i]=(arr[i-1]*c+arr[i-2])%10;
    		}
    		for(int i=1;i<n;i++){
    			num=(num*10+arr[i])%41;
    		}
    		if(num%41)
    			cout << "NO" << endl;
    		else
    			cout << "YES" << endl;
    	}
    }
    
    return 0;
}
