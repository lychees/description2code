#include <bits/stdc++.h>
using namespace std;
#define ll long long int

int main()
{
	ll n,i,sum=0;
	cin >> n;
	ll arr[n];
	for(i=0;i<n;i++){
		cin >>arr[i];
	}
	sort(arr,arr+n);
	ll pos=0,l=1;
	while(pos<n){
		if(l*l<arr[pos]){
			sum+=l*l;
		}
		else{
			sum+=l*l-arr[pos];
			pos++;
		} 
		l++;
	}
	cout << sum << endl;
    return 0;
}
