N = int(raw_input())
runs = {}

for n in xrange(N):
	c, r = [int(i) for i in raw_input().split()]
	try:
		runs[c].append(r)
	except:
		runs[c] = []
		runs[c].append(r)
		
country_codes = sorted(runs)

for c in country_codes:
	dec_runs = sorted(runs[c], reverse=True)
	for r in dec_runs:
		print str(c) + ' ' + str(r)
