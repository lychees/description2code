#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
   int t,n,numb;
   cin>>t;
   while(t--)
   {
   	map<int,int>mp;
   	int max=0;
   	cin>>n;
   	for(int i=0;i<n;i++)
   	{
   		cin>>numb;
   		mp[numb]++;
   	}
   	for(map<int,int>::reverse_iterator it=mp.rbegin();it!=mp.rend();it++)
   	{
   		if(it->second >max)
   		{
   			max =it->second ;
   		}
   	}
   	//cout<<max<<endl;
   	for(map<int,int>::reverse_iterator it=mp.rbegin();it!=mp.rend();it++)
   	  	{
   	  		if(it->second == max)
             cout<<it->first<<" ";
   	  	}
   	  	cout<<endl;
   }
    return 0;
}
