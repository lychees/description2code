#include<iostream>
using namespace std;

int main(){
	long int t,R,G,B;
	bool ra,ga,ba;
	cin >> t >> R >> G >> B;
	long int r=0,g=0,b=0,y=0,c=0,m=0,w=0,bl=0;
	for(long int i=0;i<t;i++){
		ra=((i/R)%2)?true:false;
		ga=((i/G)%2)?true:false;
		ba=((i/B)%2)?true:false;
		if(ra){
			if(ga){
				if(ba){
					w++;
				}
				else{
					y++;
				}
			}else{
				if(ba){
					m++;
				}else{
					r++;
				}
			}
		}else{
			if(ga){
				if(ba){
					c++;
				}else{
					g++;
				}
			}else{
				if(ba){
					b++;
				}else{
					bl++;
				}
			}
		}
	}
	cout << r << " "  << g << " " << b << " " << y << " " << c << " " << m << " " << w << " " << bl;
	return 0;
}
