#include <bits/stdc++.h>
using namespace std;
long long int a[1000001];
long long int b[1000001];
long long int call(int n)
{
	int i;
	int flag=0;
	for(i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
		{
			flag=1;
			break;
		}
	}
	if(flag==0)
	{
		return 1;
	}
	else
	{
		return (a[i]+a[n/i]);
	}
}
void init()
{
	a[0]=b[0]=0;
	a[1]=b[1]=0;
	a[2]=b[2]=1;
	a[3]=1;
	for(int i=4;i<=1000000;i++)
	{
		a[i]=call(i);
	}
	for(int i=3;i<=1000000;i++)
	{
		b[i]=a[i]+b[i-1];
	}
}
int main()
{
	init();
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int l,r;
    	scanf("%d %d",&l,&r);
    	long long int res=b[r]-b[l-1];
    	printf("%lld\n",res);
    }
    
}
