#include <stdio.h>
#include <iostream>
using namespace std;

int main()
{
    long int T, N, i, sum, mn, j;
    cin >> T;
    while(T--)
    {
        scanf("%ld", &N);
        long int B[N];
        i=0;
        mn=1000; sum=0;
        long int num=N;
        while(N--)
        {
            scanf("%ld", &B[i]);
            sum=sum+B[i];

            if(B[i]<mn)
                mn=B[i];
            i++;
        }
        sum=sum-num;
        mn=mn-1;
        printf("%ld %ld\n",mn,sum);
    }
    
    return 0;
}
