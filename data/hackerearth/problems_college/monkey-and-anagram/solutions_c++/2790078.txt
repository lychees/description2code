#include <iostream>
#include <string.h>
using namespace std;

int main()
{
	string s,t;
	int p,i;
	cin>>p;
	while(p!=0){
   cin>>s>>t;
   int a[2][26];
   memset(a,0,sizeof(a));
   /*if(s.size()>t.size())
     	{
     		cout<<"-1";return 0;
     	}*/
   for(i=0;i<s.size();i++)
   a[0][s[i]-97]++;
   for(i=0;i<t.size();i++)
   a[1][t[i]-97]++;
   int cnt=0;
   for(i=0;i<26;i++)
   {if(a[0][i]>0 || a[1][i]>0)
     {
     	if(a[0][i] >a[1][i])
     	{
     		cnt = -1;break;
     	}
     	if(a[0][i]<a[1][i])
     	{
     		cnt++;
     	}
     }}
     cout<<cnt<<endl;
     p--;};
    return 0;
}
