#include<math.h>

#include<algorithm>

using namespace std;

int main()
{
    long long t,n,w,A[26],B[26],i,c,count=0;
    char a[1000];
    cin>>t;
    for(i=0;i<26;i++)
        {A[i]=0;}
    cin>>a;
    for(i=0;a[i]!='\0';i++)
    {
            c=int(a[i])-int('a');
            A[c]++;
    }
    t--;
    for(w=0;w<t;w++)
    {
        for(i=0;i<26;i++)
        {B[i]=0;}
        cin>>a;
        for(i=0;a[i]!='\0';i++)
        {
            c=int(a[i])-int('a');
            B[c]++;
        }
        for(i=0;i<26;i++)
        {
                if(B[i]<A[i])
                    A[i]=B[i];
        }
    }
    for(i=0;i<26;i++)
        count+=A[i];
    cout<<count;
    return 0;
}
