using namespace std;
ll a[1005];

ll gcd(ll a, ll b)
{
	if(b == 0) return a;
	else return gcd(b, a%b);
}


int main()
{
	int t, n;
	
	ll m;
	for(cin >> t;t--;)
	{
		cin >> n;
		for(int i = 0;i < n;++i)
		{
			cin >> a[i];
		}
		m = a[0];
		for(int i = 0;i < n;++i)
		{
			m = min(a[i], m);
		}
		bool flag = true;
		while(flag)
		{
		flag = false;
		m = a[0];
		for(int i = 0;i < n;++i)
		{
			m = min(a[i], m);
		}
		if(m == 0)
		{
			
			m =1;
			break;
		}
		for(int i = n-1;i >= 0;--i)
		{
				if(a[i] % m == 0) a[i] = m;
				else a[i] %= m;
				if(a[i] != m) flag = true;
		}
		}
		cout << m << endl;
	}
	return 0;
}
