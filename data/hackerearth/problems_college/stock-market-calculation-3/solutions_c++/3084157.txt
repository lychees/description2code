#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int
#define s(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)
#define mod 1000000007
#define get getchar//_unlocked

int mrk[100005];
int a[100005];

int main()
{
    int t, i, n;
    ll ans;
    s(t);
    while (t--) {
        s(n);
        memset(mrk, 0, sizeof(mrk));
        for (i = 0; i < n; ++i)
            s(a[i]);
        ans = 0;
        for (i = n-1; i >= 0; --i) {
            mrk[i] = max(mrk[i+1], a[i]);
            ans += (mrk[i] - a[i]);
        }
        printf("%lld\n", ans);
    }
    return 0;
}
