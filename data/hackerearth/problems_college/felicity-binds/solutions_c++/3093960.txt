#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,k,n,m,p,t ;
    cin >> t ;
    while(t--) {
    	cin >> n ;
    	int a[10001] ;
    	int b[10001] ;
    	for(i=0;i<n;i++) cin >> a[i] ;
    	for(i=0;i<n;i++) cin >> b[i] ;
    	sort(a,a+n) ;
    	sort(b,b+n) ;
    	p = 0 ;
    	for(i=0;i<n;i++) p += a[i]*b[i] ;
    	cout << p << endl ;
    }
    return 0;
}
