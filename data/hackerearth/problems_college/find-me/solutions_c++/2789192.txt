#include <iostream>
using namespace std;

int main()
{
    int t,i,n,j,c,k;
    long long int a[10000],p;
    cin>>t;
    for(i=1;i<=t;i++)
    {
    	c=0;
    	k=0;
    	cin>>n>>p;
    	for(j=1;j<=n;j++)
    	cin>>a[j];
    	for(j=1;j<=n;j++)
    	{
    		if(a[j]==p)
    		goto x;
    		else
    		c++;
    	}
    	if(c==n)
    	cout<<"NO"<<" "<<"0"<<"\n";
    	else{
    	x:
    	for(j=1;j<=n;j++)
    	{
    		if(a[j]>p)
    		k++;
    	}
    	cout<<"YES"<<" "<<k<<"\n";}
    }
    
    return 0;
}
