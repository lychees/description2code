#include <bits/stdc++.h>

using namespace std;

#define icin(x) scanf("%d",&x)
#define lcin(x) scanf("%lld",&x)

typedef long long LL;
typedef vector<int> vi;
typedef vector<vi> vvi; 

int main(){
	int t;
	icin(t);
	while(t--){
		LL n;
		lcin(n);
		LL a=1,b=int(1e9),mid=(a+b)>>1;
		while(1){
			if(a==b)
				break;
			if(mid*mid<=n){
				a=mid+1;
			}
			else{
				b=mid;
			}
			mid=(a+b)>>1;
		}
		a--;
		cout<<(n-a)<<endl;
	}
	return 0;
}
