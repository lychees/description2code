# print 'Hello World!'
n = int(raw_input())
ans = ""
v = set()
l = 0
while (n > 0):
    s = raw_input()
    x = s[::-1]
    if (x in v):
        l = len(s)
        ans = s[(l-1)/2:(l-1)/2+1]
    else:
        v.add(s)
    n = n-1
print l, ans
