#include <iostream>
#include<map>
#include<cmath>
using namespace std;
#define MAX 100
map<int,bool> tab;
map<int, bool> used;
int arr[MAX+1];
int main()
{
	int N;
    cin>>N;
       
	for(int i = 0; i<N; i++)
		cin>>arr[i];
	
	for(int i = 0; i<N; i++)
		tab[arr[i]] = true;
	
	int a_b = 1; bool possible,stop = false;
	while(!stop)
	{
		a_b *= -1;
		possible = false;
	for(int i = 0; i<N && !possible ; i++)
	{
		
		
		for(int j = 0; j <N  ; j++)
		{
			if(i==j) continue;
			if(used.find(arr[i]) != used.end() || used.find(arr[i]) != used.end()) continue;
			
			int val = abs(arr[i] -arr[j]);
			//cout<<val<<endl;
			
			
			if(tab.find(val) == tab.end()){
				//cout<<"arr[i]-"<<arr[i]<<endl;
				used[arr[i]] =true; used[arr[j]] = true; tab[val] = true; arr[N++] = val; 
				possible = true; break;
				}
		}
		
		 
		
	}
	
		if(!possible) {stop = true;}	
	}
	
	if(a_b == 1) cout<<"Alice"<<endl;
	else cout<<"Bob"<<endl;
	
    return 0;
}
