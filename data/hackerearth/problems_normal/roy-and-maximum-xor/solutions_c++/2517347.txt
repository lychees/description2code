

    #include <bits/stdc++.h>
    #define MAX_N 1000000000000000000
    #define endl '\n'
    using namespace std;
    int main()
    {
    	ios_base::sync_with_stdio(false);
    	vector <long long> powers_of_2;
    	long long i = 1;
    	while(i <= MAX_N)
    	{
    		powers_of_2.push_back(i);
    		i = i*2;
    	}
    	int t;
    	scanf(" %d",&t);
    	//cin >> t;
    	while(t--)
    	{
    		long long max_xor, count;
    		long long n;
    		scanf(" %lld",&n);
    		//cin >> n;
    		n--;
    		if(n==0)
    		{
    			printf("0 1\n");
    			//cout << "0 1\n";
    		}
    		else
    		{
    			vector <long long>::iterator it;
    			it = lower_bound(powers_of_2.begin(), powers_of_2.end(), n);
    			int idx = it - powers_of_2.begin();
    			if(powers_of_2[idx] == n) //means given number is a power of 2
    			{
    				max_xor = n^(n-1);
    				count = 2;
    				printf("%lld %lld\n",max_xor, count);
    				
    			}
    			else
    			{
    				idx--;
    				max_xor = powers_of_2[idx] ^ (powers_of_2[idx] - 1);
    				count = 2*(n-powers_of_2[idx]+1);
    				printf("%lld %lld\n",max_xor, count);
    			}
    			//cout << max_xor << ' ' << count << endl;
    		}
    	}
    	return 0;
    }
