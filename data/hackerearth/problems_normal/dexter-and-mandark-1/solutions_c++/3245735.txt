#include<bits/stdc++.h>
using namespace std;
#define ll	long long int
#define MOD	1000000007
ll dp[1005][1005];
int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	assert(t>=1 && t<=10);
	while(t--)
	{
		memset(dp,0,sizeof(dp));
		int n,m;
		cin>>n>>m;
		assert(n>=1 && n<=1000 && m>=1 && m<=n);
		dp[1][1]=1;
		for(int i=1;i<=n;++i)
		{
			for(int j=1;j<=n;++j)
			{
				if(i==1 && j==1)
					continue;
				dp[i][j]=dp[i-1][j-1];
				if(j>=m*(i-j))
					dp[i][j]=(dp[i][j]+dp[i-1][j])%MOD;
			}
		}
		ll ans=0;
		for(int i=1;i<=n;++i)
			ans=(ans+dp[n][i])%MOD;
		if(!(n&1) && m==1)
			ans=(ans-dp[n][n/2]+MOD)%MOD;
		assert(ans>=0 && ans<MOD);
		cout<<ans<<endl;
	}
	return 0;
}
