#include<bits/stdc++.h>
#define gc getchar
using namespace std;
typedef unsigned long long ll;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
ll gcd(ll a,ll b)
{
    if(b==0)
        return a;
    return gcd(b,a%b);
}
int main()
{
    ll t,n,k,c,x,nn,nu,d,i;
    t=read_int();
    while(t--)
    {
        n=read_int();
        k=read_int();
        c=0;
        for(i=0;i<n;i++)
        {
            x=read_int();
            if(x==k)
                c++;
        }
        if(c<3)
            cout<<0<<"/"<<1<<'\n';
        else
        {
            nn=c*(c-1)*(c-2)/6;
            d=n*(n-1)*(n-2)/6;
            ll g=gcd(nn,d);
            cout<<nn/g<<"/"<<d/g<<'\n';
        }
    }
    return 0;
}
