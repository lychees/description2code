s = raw_input()
l = len(s)
assert(1 <= l and l <= 10**5)
q = int(raw_input())
assert(1 <= q and q <= 10**5)
for i in xrange(q):
    	a1, a2, b1, b2 = map(int, raw_input().split(" "))
    	assert(1 <= a1 and a1 <= 10**5)
    	assert(1 <= a2 and a2 <= 10**5)
    	assert(1 <= b1 and b1 <= 10**5)
    	assert(1 <= b2 and b2 <= 10**5)
    	x = s[a1-1:a2]
    	y = s[b1-1:b2]
    	if x == y:
    		print "Yes"
    	else:
    		print "No"
