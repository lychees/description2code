#include<bits/stdc++.h>
using namespace std;
#define s(a) scanf("%lld", &a);
#define ll long long

ll int a[1000006], b[1000006];

int main()
{

    ll int n, i, m, j, k;

    s(n);

    for(i = 0; i < n; i++) {
        s(a[i]);
    }

    s(m);

    while(m--) {
        s(i);
        s(j);
        s(k);

        b[i - 1] += k;
        b[j] += -k;
    }
    ll int cnt = 0;
    for(i = 0; i < n; i++) {
        cnt = cnt + b[i];
        a[i] = a[i] + cnt;
        printf("%lld\n", a[i]);
    }

}
