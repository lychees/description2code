#include<bits/stdc++.h>
using namespace std;
typedef long long lld;

int ispalin(char s[])
{
	int l=strlen(s);
	for(int i=0;i<l;i++)
	{
		
		if(s[i]!=s[l-1-i])
		return 0;
		
		
	}
	return 1;
	
	
	
}
int main()
{
	lld t;
	cin>>t;
	
	while(t--)
    {   lld i,ans=1;
        lld hash[30];
        memset(hash,0,sizeof hash);
    	char s[15];
    	cin>>s;
    	
    	 if(ispalin(s))
    	 cout<<"Palindrome\n";
    	 
    	 else
    	 {
    	 	for(i=0;i<strlen(s);i++)
    	 	{
    	 		hash[s[i]-'a'+1]++;
    	 		
    	 	}
    	 	
    	 	for(i=1;i<=26;i++)
    	 	{
    	 		if(hash[i]!=0)
    	 		ans=ans*(pow(i,hash[i]));
    	 		
    	 	}
    	 	cout<<ans<<"\n";
    	 }
    	
    	
    	
    }
	
	
	
	
	
	
	
	
	
	
	
	
}
