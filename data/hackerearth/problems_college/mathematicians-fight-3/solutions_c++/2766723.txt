#include<bits/stdc++.h>
using namespace std;
 
int main()
{
    long long n,a;
    cin>>n;
    while(n--)
    {
    	cin>>a;
    	long long x=floor(log2(a));
    	long long p=2*(a-pow(2,x))+1;
    //	cout<<p<<endl;
    	printf("%lld\n",p);
    }
    return 0;
}
