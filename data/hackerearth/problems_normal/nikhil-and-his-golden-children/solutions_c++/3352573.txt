#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int
#define s(x) scanf("%d", &x)
#define sd(x) scanf("%lf", &x)
#define sl(x) scanf("%lld", &x)
#define mod 1000000007
#define get getchar_unlocked

ll fac[2000005];

ll power(ll a, ll b)
{
    if (b == 0)
        return 1;
    ll x = power(a, b>>1);
    if (b & 1)
        return (((x*x)%mod)*a)%mod;
    return (x*x)%mod;
}

int main()
{
    int t, i, n, m, r, x;
    ll ans, tmp;
    s(t);
    fac[0] = 1;
    for (i = 1; i <= 2000000; ++i)
        fac[i] = (fac[i-1] * i)%mod;
    while (t--) {
        s(n);
        s(m);
        r = n;
        while (n--) {
            s(x);
            m -= x;
        }
        if (m < 0) {
            printf("0\n");
            continue;
        }
        if (m == 0) {
            printf("1\n");
            continue;
        }
        n = m;
        ans = fac[n+r-1];
        tmp = (fac[n]*fac[r-1])%mod;
        ans = (ans * power(tmp, mod-2))%mod;
        printf("%lld\n", ans);
    }
    return 0;
}
