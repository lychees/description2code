#include <iostream>
using namespace std;

int main()
{
    long t,n1,n2,carry,ans;
    cin>>t;
    while(t--)
    {
    	cin>>n1>>n2;
    	carry=0;ans=0;
    	while(n1>0 and n2>0)
    	{
    		if(n1%10+n2%10+carry>=10)
    		{
    			carry=1;
    			ans++;
    			n1=n1/10;
    			n2=n2/10;
    		}
    		else {
    			carry=0;
    			n1=n1/10;
    			n2=n2/10;
    		}
    	}
    	if(ans==0)cout<<"No carry operation"<<endl;
    	if(ans==1)cout<<"1 carry operation"<<endl;
    	else if(ans>1)
    	cout<<ans<<" carry operations"<<endl;
    }
    return 0;
}
