#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;
int main()
{
	int x;
	cin >> x;
	vector <string> b;
	

	int t=0;

	while(x--)
	{
		string a;
		cin >> a;
		int z=0; 
		if(b.size()==0)
		{
			t++;
		}
		else 
		{
			for(int d=0;d<b.size();d++)
			{
				if(a[0]==b[d][0])
				{
					
					if(a.size()==b[d].size())
					{
						
						if(a[a.size()-1]==b[d][b[d].size()-1])
						{
							
							string c= a;
							string e= b[d];
							sort(c.begin(),c.end());
							sort(e.begin(),e.end());
							
							if(c==e)
							{								
								z++;
							}

						}
					}
				}
		
			}
			if(z==0)
			{
				t++;
			}

		
		}
	
		b.push_back(a);



	}


	cout << t << endl;

}
