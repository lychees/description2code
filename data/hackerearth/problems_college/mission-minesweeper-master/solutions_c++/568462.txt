#include<iostream>
using namespace std;
int n, m;

void nextStage(char game[101][101], int x, int y){
	if(x<0||x>=n||y>=m||y<0||(game[x][y]>='0'&&game[x][y]<='8'))
		return;
	int i, j, count=0;
	for(i=-1;i<2;i++)
		for(j=-1;j<2;j++)
			if(i==0&&j==0)
				continue;
			else if(x+i<n&&x+i>-1&&y+j<m&&y+j>-1&&game[x+i][y+j]=='*')
				count++;
	game[x][y]=count+48;
	if(count==0){
		nextStage(game, x-1, y-1);
		nextStage(game, x-1, y+1);
		nextStage(game, x-1, y);
		
		nextStage(game, x, y-1);
		nextStage(game, x, y+1);
		
		nextStage(game, x+1, y-1);
		nextStage(game, x+1, y+1);
		nextStage(game, x+1, y);
	}
}
int main()
{
    int test_cases;
    cin>>test_cases;
    while(test_cases>0)
    {
        int i,j;
        char game[101][101];
        cin>>n>>m;
        for(i=0;i<n;i++)
            cin>>game[i];
        for(i=0;i<n;i++)
        	for(j=0;j<m;j++)
        		if(game[i][j]=='c'){
        			nextStage(game, i, j);
        			break;
        		}
       	for(i=0;i<n;i++)
       		cout<<game[i]<<endl;
        test_cases--;
    }
    return 0;
}
