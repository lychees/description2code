
N = input()
count = 0
for _ in range(N):
    s = raw_input()
    st = []
    for i in s:
        if i == 'A':
            if st and st[-1] == 'A':
                st.pop()
            else:
                st.append(i)
        else:
            if st and st[-1] == 'B':
                st.pop()
            else:
                st.append(i)
    if not st:
        count += 1

print count
