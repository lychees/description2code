#include <bits/stdc++.h>
using namespace std;

int main()
{
    int m;
    cin>>m;
    int a[m],i,j;
    for(i=0;i<m;i++){
    	cin>>a[i];
    }
    int n,qual[10002],check[10002],count=0;
    cin>>n;
    char c,quality[3000];
    c=getchar();
    //c=getchar();
    while(n>0){
    	for(i=0;i<10002;i++){
    		check[i]=0;
    		qual[i]=0;
        }
    	c=getchar();
    	//cout<<c;
    	i=0;
    	while(c!='\n'&&c!='\0'&&c!=EOF){
    		if(c==' '){
                c=getchar();
                check[qual[i]]++;
                //cout<<qual[i];
                i++;
                continue;
            }
            qual[i]*=10;
    		qual[i]+=(c-'0');
    		c=getchar();
    		//cout<<i;
    	}
        check[qual[i]]++;
        //cout<<qual[i];
        i++;
    	for(j=0;j<m;j++){
    		if(check[a[j]]==0)
    			break;
    	}
    	//cout<<m<<endl;
    	if(j>=m)
    		count++;
    	//cout<<count<<endl;
    	n--;
    }
    cout<<count<<endl;
    //getchar();
    return 0;
}
