#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,t,i,k,j,x,y,a[1001],total;
   	scanf("%d",&t);
   	while(t--){
   		scanf("%d%d",&n,&k);
   		for(i=0;i<n;i++)
   			scanf("%d",&a[i]);
   		for(i=0;i<n;i++){
   			x = __builtin_popcount(a[i]);
   			for(j=i+1;j<n;j++){
   				y = __builtin_popcount(a[j]);
   				if(x<y){
   					x = a[i];
   					a[i] = a[j];
   					a[j] = x;
   					x = __builtin_popcount(a[i]);
   				}
   			}
   		}
   		i=total=0;
   		while(k--){
   			total += __builtin_popcount(a[i++]);
   		}
   		printf("%d\n",total);
  
   	}
    return 0;
}
