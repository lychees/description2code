#include<iostream>
#include<cmath>

using namespace std;

int dp[502][502];

int recurse(int b,int m)
{
    int i,j,k;
    for(i=1;i<=b;i++)
    {
       dp[i][1]=1;
       dp[i][0]=0;
    }
    
    for(i=1;i<=m;i++)
    dp[1][i]=i;
    
    for(i=2;i<=b;i++)
    {
       for(j=2;j<=m;j++)
       {
           dp[i][j]=100001;
           for(k=1;k<=j;k++)
           {
              int tmp = max(dp[i-1][k-1],dp[i][j-k]) ;
              if(dp[i][j] > tmp + 1)
              {
                 dp[i][j] = tmp + 1;
              }
           }
       }
    }
    return dp[b][m];
}

int main()
{
    int t,n,k;
    scanf("%d",&t);
    recurse(500,500);
    while(t--)
    {
              scanf("%d%d",&n,&k);
              printf("%d\n",dp[k][n]);   
    }
    return 0;
}
