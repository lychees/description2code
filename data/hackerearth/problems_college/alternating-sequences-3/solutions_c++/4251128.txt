#include<iostream>
using namespace std;
int abs(int x)
{
	if(x<0)
	return -x;
	else
	return x;
}
int main()
{
	int n;
	cin>>n;
	int A[n],i,j,dp[n],max;
	for(i=0;i<n;i++)
	cin>>A[i];
	int maximum=1;
    dp[0]=1;
    for(i=1;i<n;i++)
    {
    max=0;                
 	for(j=i-1;j>=0;j--)
  	{	
  	  if(abs(A[i])>abs(A[j]) && ((A[i]>0 &&A[j]<0) || (A[i]<0 && A[j]>0)))
	  {
	   if(max<dp[j])
	   max=dp[j];
	  }		
	}
	dp[i]=1+max;
	if(dp[i]>maximum)
	maximum=dp[i];
    }
    cout<<maximum;
	return 0;
}
