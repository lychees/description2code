import sys
DP = [[0 for i in xrange(65)] for i in xrange(65)]
A = [[0 for i in xrange(65)] for i in xrange(65)]
def pre():
    DP[0][0]=1
    DP[0][1]=1
    for i in xrange(1,65):
	DP[i][0]=(DP[i-1][0]+DP[i-1][1])
	DP[i][1]=(DP[i-1][0]+DP[i-1][1])
def solve(n):
	if(n%4 == 0):
		return n
	if(n%4 == 1):
		return 1
	if(n%4 == 2):
		return n+1
	return 0
for _ in xrange(input()):
    pre()
    L,R=map(int,raw_input().split())
