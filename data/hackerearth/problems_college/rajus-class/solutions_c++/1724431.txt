#include <bits/stdc++.h>

using namespace std;

struct srt
{
    int age;
    string name;
};

bool comp(struct srt a, struct srt b)
{
    if (a.age < b.age)
        return 1;

    if (a.age == b.age) {
        if (a.name < b.name)
            return 1;
    }

    return 0;
}
int main()
{
    int t;

    cin >> t;

    while (t--) {
        int n, r, i, j;

        cin >> n;

        struct srt s[n];

        for (i = 0; i < n; i++) {
            cin >> s[i].age >> s[i].name;
        }

        sort(s, s + n, comp);

        cin >> r;

        cout << s[r-1].name << endl;
    }

    return 0;
}
