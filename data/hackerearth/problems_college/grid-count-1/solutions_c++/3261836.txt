#include <bits/stdc++.h>
#define ll long long
using namespace std;

bool isCube(ll n) {
    ll root(round(cbrt(n)));
    return n == root * root * root;
}

int main()
{
	ll n,k,i,j,m,count=0;
	cin>>n>>k;
	vector<vector< ll > > v(n),t(n);
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			cin>>m;
			v[i].push_back(m);
			t[i].push_back(m);
		}
	}
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			if(j==0) continue;
			if(j<k){
				v[i][j]+=v[i][j-1];
			}
			else{
				v[i][j] = v[i][j] + v[i][j-1] - t[i][j-k];
			}
		}
	}
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			t[i][j]=v[i][j];
		}
	}
	/*
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			cout<<v[i][j]<<" ";
		}
	}
	*/
	
	
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
//			cout<<v[i][j]<<" ";
			if(i==0) continue;
			if(i<k){
				v[i][j]+=v[i-1][j];
			}
			else{
				v[i][j] = v[i][j] + v[i-1][j] - t[i-k][j];
			}
		}
	}
	
	
	
	
	for(i=k-1;i<n;i++){
		for(j=k-1;j<n;j++){
			if(isCube(v[i][j])){
				count++;
			}
		}
	}
	
	cout<<count;
	
    return 0;
}
