#include <bits/stdc++.h>

typedef long long int li;

#define Fi(i, st, ft) for(i = st;i <= ft; ++i)
#define Fd(i, st, ft) for(i = st;i >= ft; --i)
#define pb(a, b) a.push_back(b)
#define mod 1000000007
#define MAXN 1500000
#define MX 1450000
#define inf 1000000000000000000ll
const int maxBufSize = (10000000);

using namespace std;

li n , q , l , r;
li arr[10001];
vector<li> v;

int main()
{
	li i , j , a , b , c;
	scanf("%lld %lld %lld %lld" , &n , &l , &r , &q);
	Fi(i , 0 , n-1) {
		scanf("%lld" , &a);
		if(i>=l && i <= r) {
			pb(v , a);
		}
	}
	sort(v.begin() , v.end());
	Fi(i , 1 , q) {
		scanf("%lld" , &b);
		a = (li)(lower_bound(v.begin() , v.end() , b)-v.begin());
		if(v[a] == b) {
			a++;
		}
		if(a == (r-l+1)) {
			printf("-1\n");
		} else {
			printf("%lld\n" , v[a]);
		}
	}
	return 0;
}
