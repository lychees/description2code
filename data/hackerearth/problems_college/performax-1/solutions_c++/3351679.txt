    #include <iostream>
    using namespace std;
     
    int main()
    {
    	long long int n;
    	cin>>n;
    	long long int a;
    	cin>>a;
    	long long int exam[n];
    	for(long long int i=0;i<n;i++)
    	{
    		cin>>exam[i];
    	}
    	long long int tut=0;
    	for(long long int i=1;i<n;i++)
    	{
    		while(exam[i]<=exam[i-1])
    		{
    			exam[i]=exam[i]+a;
    			tut++;
    		}
    	}
    	cout<<tut<<endl;
        
        return 0;
    }
