#include"iostream"
using namespace std;
int main()
{
	string s;
	int t,count;
	cin>>t;
	while(t--)
	{
		count=0;
		cin>>s;
		int l=s.length();	
		for(int i=4;i<l-4;i++)
		{
			if(!(s[i]=='a' || s[i]=='e' || s[i]=='i' || s[i]=='o' || s[i]=='u'))
				count++;
		}
		cout<<count+4<<"/"<<l<<endl;
	}
	
	
}
