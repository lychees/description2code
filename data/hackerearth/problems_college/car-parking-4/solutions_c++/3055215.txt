#include <iostream>
using namespace std;

int main()
{
   	int t;
   	cin>>t;
   	
   	while(t--){
   		string a;
   		cin>>a;
   		int c=0;
   		
   		for(int i=0;i<a.size();i++){
   			if(a[i]=='-'){
   				if(a[i+2]=='B'||a[i+1]=='B'||a[i+1]=='S'||a[i-1]=='S');
   				else c++;
   			}
   		}
   		cout<<c<<endl;
   		
   	}
    return 0;
}
