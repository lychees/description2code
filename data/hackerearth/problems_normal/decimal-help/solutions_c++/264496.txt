#include <iostream>
using namespace std;

int main() {
	int t;
	cin >> t;
	while(t--){
		int n,d,r;
		cin >> n >> d >> r;
		int a,b;
		a=n/d;
		b=n-d*a;
		while(b!=0 && r!=0){
			r--;
			n=b*10;
			a=n/d;
			b=n-d*a;
		}
		if(r==0)
			cout << a << endl;
		else
			cout << 0 << endl;
	}
	return 0;
}
