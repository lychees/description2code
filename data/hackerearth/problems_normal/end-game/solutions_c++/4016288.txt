#include <iostream>
using namespace std;
bool game(int n, int a, int b, int c, int d, int side)
{
	if (side == 0)
	{
		++a;
	}
	int t = ((d > b)? d-b : b-d);
	return (t <= (n-a + 1) && c >= (a-1));
}
int main()
{
	int T;
	cin >> T;
	bool* out = new bool[T];
	for(int i = 0; i < T; ++i)
	{
		int n,a,b,c,d,side;
		cin >> n >> a >> b >> c >> d >> side;
		out[i] = game(n,a,b,c,d,side);
	}
	for(int i = 0; i < T; ++i)
	{
		cout << ((out[i])?"Draw":"White Wins") << endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
