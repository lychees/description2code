#include <bits/stdc++.h>
using namespace std;
#define ll long long 

int main(){

    ll t , l , r , i , j , k ;
	cin >> t;
	double low;
	double high;
	ll lowind;
	ll highind;
	ll tot = 0;
	while(t--){
		tot = 0;
		cin >> l >> r;
		if(l>r){
			swap(l,r);
		}
		low = ceil((-1+sqrt(1+8*l))/2);
		high = ceil((-1 + sqrt(1+8*r))/2);
		lowind = low;
		highind = high;
		if(l==r){
			k =( lowind*(lowind-1)/2)+1;
			if(abs(l-k)%2==0)cout << low<<endl;
			else cout << -low<<endl;
			
			continue;
		}
		k =lowind*(lowind+1)/2;
		
	//	cout << lowind << " " << highind <<endl;
		if((k-l)%2==0){
			tot+=(lowind%2==0)?-lowind:lowind;
		}
		if(l==r){
			cout << tot <<endl;
			continue;
		}

		k =highind*(highind-1)/2+1;

		if((r-k)%2==0&&(k!=r)){
			tot+=highind;
		}
		if(k==r){
			tot+=highind;
		}
		 //cout << tot <<" ";
		for( i = lowind +1 ; i < highind ; i++){
			if(i%2){
				tot+=i;
			}
		}
        cout << tot <<endl;
	}
    return 0;
}
