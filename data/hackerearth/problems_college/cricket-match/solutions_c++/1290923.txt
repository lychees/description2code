#include<bits/stdc++.h>
#define ll long long
#define S(n) scanf("%d",&(n))
#define P(n) printf("%d",(n))
#define SL(n) scanf("%lld",&(n))
#define PL(n) printf("%lld",(n))
#define NL printf("\n")
#define SP printf(" ")
#define Debug cout<<"Hello"

#define N 100000
using namespace std;

vector<int>g[100006];
int c[100006];
int visited[1000006];

void dfs(int u, int color)
{
	
	visited[u]=1;
	c[u] = color;
	int l = g[u].size();
	
	for(int i=0;i<l;i++)
	{
		int v = g[u][i];
		
		if(!visited[v])
		{
			visited[v]=1;
			if(color==1)dfs(v,2);
			else dfs(v,1);
		}
	
	}
	
	
}

int main()
{
	
	//freopen("input.txt","r",stdin);
	
	

	int t;
	
	S(t);
	
	//t = 1;
	
	while(t--)
	{
		
		int n;int m;
		
		S(n);S(m);
		
		for(int i=1;i<=n;i++)g[i].clear();
		
		for(int i=1;i<=m;i++)
		{
			int u,v;
			S(u);S(v);
			g[u].push_back(v);
			g[v].push_back(u);
		}
		
		int flag=1;
		memset(visited,0,sizeof(visited));
		memset(c,0,sizeof(c));
		for(int i=1;i<=n;i++)if(!visited[i])dfs(i, 1);
		
		for(int i=1;i<=n;i++)
		{
			int l = g[i].size();
			for(int j=0;j<l;j++)
			{
				
				if(c[g[i][j]]==c[i])
				{
					flag=0;
				}
				
			}
		}
		
		if(flag==1)
		{
			printf("YES\n");
		}
		else
		{
			printf("NO\n");
		}
		
		
		
	}
		
	
	return 0;
	
}
