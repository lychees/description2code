#include <bits/stdc++.h>
using namespace std;
int count1[26];
int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin >>t ;
   while(t-- ){
   	string ch,ch1;
   	memset(count1,0,sizeof(count1));
   	cin >> ch >> ch1;
   	for(int i=0;i<ch.length();i++){
   		count1[ch[i]-'a']++;
   	}
   	int ok=0;
   	for(int i=0;i<ch1.length();i++){
   		if(count1[ch1[i]-'a']) {
   			ok=1;break;
   		}
   	}
   	if(ok) cout << "YES" <<endl;
   	else cout <<"NO" <<endl;
   }
    return 0;
}
