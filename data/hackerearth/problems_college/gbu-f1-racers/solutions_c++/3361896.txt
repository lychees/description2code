#include<iostream>

using namespace std;

int main()
{
	int d[1005],s[1005],i,t,f,mx,n,k;
	cin >> n >> k;
	for(i=0;i<n;i++)
    {
        cin >> d[i];
    }
	for(i=0;i<n;i++)
    {
        cin >> s[i];
    }
	f=t=mx=0;
	for(i=0;i<n;i++)
    {
        f+=s[i];
        if(i==0||mx<s[i])
        {
            mx=s[i];
        }
        while(f<d[i])
        {
            f+=mx;
            t+=k;
        }
        f-=d[i];
        t+=d[i];
    }
    cout << t;
    return 0;
}
