#include <cstdio>

int capitalAlphabets[26];
int smallAlphabets[26];
int numerals[10];

int markCharacter(char input);
int getSetSize();
int getPowerSets();

int main(){
	int t;
	scanf("%d ",&t);
	for(int i=0; i<t; i++){
		for(int j=0; j<26; j++){
			capitalAlphabets[j]=0;
			smallAlphabets[j]=0;
		}
		for(int j=0; j<10; j++)
			numerals[j]=0;
		while(1){
			char curChar=getchar();
			if(!markCharacter(curChar))
				break;
			//printf("%d*",curChar);
		}
		printf("%d\n",getPowerSets());
		//printf("\n");
		//printf("\nset size is %d\n",getSetSize());
	}

	return 0;
}

int markCharacter(char input){
	if(input>='0'&& input <='9')
		numerals[input-'0']=1;
	else if(input>='a' && input<='z')
		smallAlphabets[input-'a']=1;
	else if(input>='A' && input<='Z')
		capitalAlphabets[input-'A']=1;
	else
		return 0;
	return 1;
}

int getPowerSets(){
	int size=getSetSize();
	return size*(size+1)/2;
}


int getSetSize(){
	int size=0;
	for(int i=0; i<26; i++){
		size+=capitalAlphabets[i];

		/*if(capitalAlphabets[i])
			printf("%c",i+'A');*/

		size+=smallAlphabets[i];

		/*if(smallAlphabets[i])
			printf("%c",i+'a');*/
	}
	for(int i=0; i<10; i++){
		size+=numerals[i];
		
		/*if(numerals[i])
			printf("%c",i+'0');*/

	}
	return size;
}
