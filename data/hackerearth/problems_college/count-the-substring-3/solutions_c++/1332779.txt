#include<iostream>
#include<string>
#include<cstring>
#include<algorithm>
using namespace std;
int dp[2000001][4];
int main(){
	 int t,i,j;
	 cin >> t;
	 while(t--){
 	   string x;
 	   cin >> x;
 	   int sz = x.size();
		memset(dp,0,sizeof dp);	
 	        for(i = 0 ; i < sz ; ++i)
        	 	dp[i][x[i]-'x'+1]=i+1;
        	 dp[sz][1]=sz+1;
        	 dp[sz][2]=sz+1;
        	 dp[sz][3]=sz+1;
	      for(i = sz-1; i >= 0 ; --i)
	       {
       		for(j = 1 ; j <= 3 ; ++j)
       		      if(dp[i][j] == 0)
       		         dp[i][j]=dp[i+1][j];
       		 
       	}
       	
	       long long int ans = 0;
	       for(i = 0 ;i < sz; ++i)
	       {
       		   if(x[i]=='x')
       		   ans = ans + (max(dp[i][2]-i,dp[i][3]-i)-1);
       		   else
       		   if(x[i]=='y')
       		   ans = ans + (max(dp[i][1]-i,dp[i][3]-i)-1);
       		   else
       		   ans = ans + (max(dp[i][1]-i,dp[i][2]-i)-1);
       	}
	         cout << ans << endl;     
	 }
	 
}
