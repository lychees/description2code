#include <iostream>
#include <vector>

using namespace std;


int main(){
	

	int num;

	int counter;
	
	int even_c;

	vector<int> even_num;


	while(cin >> counter)
	{
		even_c = 0;
		for(int i = 0; i < counter; i++)
		{
			cin >> num;
			if (num % 2 == 0)
			{
				even_num.push_back(num);
				even_c++;
			}
				
		}
		if(even_c == 0)
		{
			cout << "None to display";
		}
		else
		{
		for(int j = even_num.size()-1; j >= 0; j--)
		{
			cout << even_num[j] << " ";
			even_num.pop_back();
		}
		}

		cout << endl;
	}



	return 0;
}
