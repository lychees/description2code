#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<string>

using namespace std;

int main()
{
    int t;

    cin>>t;
    while(t--)
    {
        char a[100],b[100],c[100];
        cin>>a>>b;
        int i=0;
        while(a[i])
        {
            if(a[i]==b[i])
                c[i]='0';
            else
                c[i]='1';

            i++;

        }
        c[i]='\0';

        cout<<c<<endl;

    }
return 0;
}
