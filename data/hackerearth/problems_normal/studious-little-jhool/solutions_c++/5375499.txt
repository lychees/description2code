#include <bits/stdc++.h>
#define inf 100000000
using namespace std;
typedef long long int ll;

int f(int a)
{
    if(a == 0)
        return 0;
    else if(a < 0)
        return inf;
    return 1 + min(f(a-12) , f(a-10));
}


int main()
{
	int t;
	cin >> t;
	while(t--)
    {
        int n;
        cin >> n;
        int l = f(n);
        if(l >= inf)
            cout << -1<<"\n";
        else
            cout << l << "\n";
    }

	return 0;
}
