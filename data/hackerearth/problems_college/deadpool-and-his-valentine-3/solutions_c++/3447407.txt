#include <iostream>
#define LLI long long int
using namespace std;

int main()
{
	LLI k,m;
	cin>>k>>m;
	LLI a[m+1];
	a[0]=-1;
	for(LLI i=0;i<k;i++)
	{
		LLI len;
		cin>>len;
		for(LLI j=0;j<len;j++)
		{
			LLI z;
			cin>>z;
			a[z]=1;
		}
	}
	int flag=0;
	for(LLI i=1;i<=m;i++)
	{
		if(a[i]!=1)
		{
			flag=1;
			break;
		}
	}
	if(flag==1)
	{
		cout<<"NO"<<endl;
	}
	else
	
	{
		cout<<"YES"<<endl;
	}
     
    return 0;
}
