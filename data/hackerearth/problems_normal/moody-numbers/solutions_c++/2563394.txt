#include <bits/stdc++.h>
using namespace std ;
#define LL long long
int T,N ;
int digit_sum(LL x){
    int s = 0 ;
    while(x){
        s += x%10 ;
        x /= 10 ;
    }
    return s ;
}
int func(int N){
    return digit_sum(1LL*N*N) ;
}
int main(){
    scanf("%d",&T) ;
	while(T--){
        scanf("%d",&N) ;
        int itr = 5 ;
        int f = 0 ;
        while(itr--){
            f = (N == 4 || N == 1) ;
            if(f) break ;
            N = func(N) ;
        }
        puts(f ? "YES" : "NO") ;
    }
   
}
