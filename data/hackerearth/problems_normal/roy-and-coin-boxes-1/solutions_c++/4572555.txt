#include <bits/stdc++.h>
using namespace std;
#define maxn 1000005
int cnt[maxn],boxcnt[maxn],dp[maxn];
int main(){

	int n,m,l,r,q,x;
	cin >> n >> m;
	for(int i=0;i<=1000000;i++){
		cnt[i]=0;
		boxcnt[i]=0;
		dp[i]=0;
	}
	while(m--){
		cin >> l >> r;
		cnt[l]+=1;
		cnt[r+1]-=1;
	}
	for(int i=1;i<=n;i++){
		cnt[i] += cnt[i-1];
		// cout << cnt[i] << " ";
	}
	for(int i=1;i<=maxn;i++){
		// cout << "cnt value is" << cnt[i] << endl;
		boxcnt[cnt[i]]++;
		// cout << "value of boxcnt[" << cnt[i] << "] is "  << boxcnt[cnt[i]] << endl;
	}
	for(int i=maxn;i>=1;i--)
			dp[i-1] = dp[i] + boxcnt[i-1];

	cin >> q;

	while(q--){
		cin >> x;
		cout << dp[x] << endl;	
	}

	return 0;
}
