#include <bits/stdc++.h>

using namespace std;

const int MAXN = 1000010;

int n, m;
int ar1[2 * MAXN];
int ar2[2 * MAXN];

void success(int x) {
	cout << x << endl;
	exit(0);
}
void fail() {
	cout << "-1" << endl;
	exit(0);
}

int main() {
	cin >> n >> m;
	for (int i = 0; i < n; i++) {
		cin >> ar1[i];
	}
	for (int i = 0; i < n; i++) {
		cin >> ar2[i];
	}
	int beg = -1;
	for (int i = 0; i < n; i++) {
		if (ar1[0] == ar2[i]) {
			if (beg != -1) while (true);
			beg = i;
		}
	}

	int cur = 0;
	for (int i = 0; i < m; i++) {
		char ch;
		int x;
		cin >> ch >> x;
		if (ch == 'L') cur -= x;
		else cur += x;
		cur = (cur % n + n) % n;
		if (cur == beg) {
			success(i + 1);
		}
	}
	fail();
}
