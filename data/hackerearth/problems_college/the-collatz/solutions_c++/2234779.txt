#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	 long long int n,sum1=0,sum=0,count=0;
    	cin>>n;
    	while(n!=1)
    	{
    		if(n%2==0){
    			sum=sum+n;
    			n=n/2;
    		}
    		else {
    			sum1=sum1+n;
    			n=(3*n)+1;
    		}
    		count++;
 
    		
    }
    	sum1++;		
    if((sum%count)>(sum1%count))
    cout<<"Even Rules"<<endl;
    else if((sum%count)==(sum1%count))
    cout<<"Tie"<<endl;
    else
    cout<<"Odd Rules"<<endl;
    }
    return 0;
}
