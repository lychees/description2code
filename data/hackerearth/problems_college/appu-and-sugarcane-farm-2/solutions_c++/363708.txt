#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;

int main()
{
    int n, r, v = 0, pg = 0, ng = 0, pn = 0, nn = 0;
    int *a;
    scanf("%d %d",&n,&r);
    a = new int[n];
    for(int i = 0; i<n; ++i)
    {
    	scanf("%d",&a[i]);
    	a[i] += 1;
    	v += a[i];
    }
    sort(a,a+n);
    v = v / n;
    for(int i = 0; i<n; ++i)
    {
    	if(v - a[i] > 0)
        {
            pg += v - a[i];
            ++pn;
        }
    	else if(v - a[i] < 0)
        {
            ng += v - a[i];
            ++nn;
        }
        else;
    }
    if(pg + ng != 0)
    	printf("-1");
    else
    {
        if(pn >= nn)
            printf("%d",pn*r);
        else
            printf("%d",nn*r);
    }

    return 0;
}
