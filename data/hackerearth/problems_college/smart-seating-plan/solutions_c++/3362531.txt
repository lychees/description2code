#include <bits/stdc++.h>
using namespace std;
void update(vector<int>& bit,int idx,int n,int val)
{
	while(idx<=n)
	{
		bit[idx]+=val;
		idx+=(idx&(-idx));
	}
}
void updateinter(vector<int>& bit,int u,int v,int n)
{
	update(bit,u,n,1);
	update(bit,v+1,n,-1);
}
int sum(vector<int>& bit,int idx)
{
	int ans=0;
	while(idx)
	{
		ans+=bit[idx];
		idx-=(idx&(-idx));
	}
	return ans;
}
int main()
{
    int r,b,n,p,q,s,i;
    scanf("%d",&r);
    vector<int> bit(r+2);
    scanf("%d",&b);
    while(b--)
    {
    	scanf("%d%d",&p,&q);
    	updateinter(bit,p,q,r);
    }
    vector<int> a(r+1);
    for(i=1;i<=r;i++)
    {
    	a[i]=sum(bit,i);
    }
    sort(a.begin(),a.end());
    vector<int>::iterator it;
    scanf("%d",&n);
    while(n--)
    {
    	scanf("%d",&s);
    	it=lower_bound(a.begin(),a.end(),s);
    	printf("%d\n",r+1-(it-a.begin()));
    }
    return 0;
}
