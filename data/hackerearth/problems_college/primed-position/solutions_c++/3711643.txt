#include <bits/stdc++.h>
using namespace std;
#define ll long long 
#define pb push_back
#define MAX 1500001
#define mod 1000000007
bool is[MAX];
vector<int>p;
vector<int>ans;
int main()
{
    memset(is,true,sizeof(is));
    is[1] = false;
    for(ll i=2;i<MAX;i++)
    {
    	if(is[i])
    	{
    		p.push_back(i);
    		for(ll j=i*i;j<MAX;j+=i)
    		{
    			is[j] = false;
    		}
    	}
    }
    for(int i=0;i<(int)p.size();i++)
    {
    	if(is[i+1])
    	{
    		ans.pb(p[i]);
    	}
    }
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,m;
    	scanf("%d%d",&n,&m);
    	long long res = 1;
    	res = (res * (ans[n-1]%mod))%mod;
    	res = (res * (ans[m-1]%mod))%mod;
    	printf("%lld\n",res);
    }
    return 0;
}
