#include <iostream>
#include<limits.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int n,k,x;
   	cin>>n>>k;
   int	count[k+1];
   	for(int i=1;i<=k;i++)
   	count[i]=0;
   	
   	for(int i=0;i<n;i++)
   	{
   	cin>>x;
   	count[x]++;
   	}
   	
   	int max=INT_MIN;
   	for(int i=1;i<=k;i++)
   	if(count[i]>max)
   	max=count[i];
   	
   	cout<<n-max<<"\n";
   	
   	
   }
    return 0;
}
