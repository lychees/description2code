mod = 10**9+7


def gcd(a, b):
    while b != 0:
        a = a % b
        a, b = b, a
    return a


n = int(raw_input())
a = map(int, raw_input().split())
gx = a[0]
for i in a[1:n]:
    gx = gcd(gx, i)
fx = 1
for i in a:
    fx = (fx*i) % mod
ans = 1
while gx > 0:
    if gx & 1:
        ans = (ans*fx) % mod
    fx = fx*fx
    fx = fx % mod
    gx /= 2
print ans
