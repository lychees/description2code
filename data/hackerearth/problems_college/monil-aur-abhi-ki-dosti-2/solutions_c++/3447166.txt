#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	string s,p;
	int c1[26];
	int c2[26];
	while(t--){
		bool flag = true,damnflag = true;
		memset(c1,0,sizeof(c1));
		memset(c2,0,sizeof(c2));
		cin >> p;
		cin >> s;
		int plen = p.length();
		int slen = s.length();
		if(plen>slen) printf("NO\n");
		else{
		for(int i=0;i<plen;i++){
			c1[p[i]-'a']++;
		}
		for(int j=0;j<plen;j++){
				c2[s[j]-'a']++;
		}
		for(int i=0;i+plen<=slen;i++){
			flag = true;
			for(int j=0;j<26;j++){
				if(c2[j]!=c1[j]){flag=false;break;}
			}
			c2[s[i]-'a']--;
			c2[s[i+plen]-'a']++;
			if(flag){printf("YES\n");damnflag=false;break;}
		}
		
		if(damnflag) printf("NO\n");
		}
	}
    return 0;
}
