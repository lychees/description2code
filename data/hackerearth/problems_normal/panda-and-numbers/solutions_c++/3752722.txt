ll power(ll base,ll num)
{
    if(num==0)
        return 1;
    else if(num==1)
        return base;
    else
    {
        ll temp=power(base,num/2);
        if(num%2==0)
            return temp*temp;
        else
            return temp*temp*base;
    }
}
