T = int(raw_input())

for t in range(T):
    S = raw_input()

    maxim, secv, cnt = 0, 0, 0

    for c in S:
        if c in 'aeiouy':
            cnt += 1
            secv = 0
        else:
            secv += 1

        maxim = max(maxim, secv)

    if maxim >= 3 or (len(S) - cnt) > cnt:
        print 'hard'
    else:
        print 'easy'
