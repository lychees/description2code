#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    //cout << "Hello World!" << endl;
    int t,n,a[1000006];
    a[0]=INT_MAX;a[1]=INT_MAX;a[2]=1;a[3]=1;a[5]=1;a[7]=1;a[6]=2;a[4]=2;
    	for(int i=8; i<=1000000; i++)
    	{
    		a[i]=1+min(min(a[i-2],a[i-3]),min(a[i-5],a[i-7]));	
    	}
    cin >> t;
    while(t--)
    {
    	cin >> n;
    	
    	if(a[n]==INT_MAX)
    	cout << "-1\n";
    	else
    	cout << a[n] <<"\n";
    	
    }
    
    
    return 0;
}
