#include<iostream>
#include<vector>
#include<string>
#include<algorithm>
#include<stdio.h>
using namespace::std;
struct person{
    string name;
    int age;
};
bool cond(struct person a,struct person b)
{
    if(a.age<b.age) return true;
    else if(a.age>b.age) return false;
    return a.name<b.name;
}

int main()
{
    
    int t,n,r,a,i;
    vector<struct person> people;
    struct person tmp;
    string s;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	people.clear();
    	for(i=0;i<n;i++)
    	{
    		cin>>a>>s;
    		tmp.name=s;
            tmp.age=a;
    		people.push_back(tmp);
    	}
    	cin>>r;
    	sort(people.begin(),people.end(),cond);
    	cout<<people[r-1].name<<endl;
   	}
   
    
}
