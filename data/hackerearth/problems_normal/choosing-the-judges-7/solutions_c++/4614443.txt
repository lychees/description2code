ll max(ll a, ll b){
    return (a>b?a:b);
}

int main(){
	ll t, n, i;
	cin >> t;
	ll k = t;
	while(t--!=0){
		cin >> n;
		ll a[n+1];
		for(i=1; i<=n; i++)
			cin >> a[i];
		ll dp[n+1];
		dp[0] = 0;
		dp[1] = a[1];
		for(i=2; i<=n; i++){
			dp[i] = max(dp[i-1], a[i]+dp[i-2]);
		}
		cout << "Case " << k-t << ": " << dp[n] << nl;
	}
}
