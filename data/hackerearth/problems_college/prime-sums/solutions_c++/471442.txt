#include <iostream>
using namespace std;

int main()
{
    bool notPrime[4000001]={false};
    long long i,j,n,sum=0;
    
    cin>>n;
    
    notPrime[1]=true;
    
    for(i=2;i<=n;i++)
    {
    	if(!notPrime[i])
    	{
    		for(j=i*2;j<=n;j=j+i)
    		notPrime[j]=true;
    	}
    }
    
    for(i=1;i<n;i++)
    {
    	if(!notPrime[i])
    	sum+=i;
    }
    cout<<sum<<"\n";
    
    return 0;
}
