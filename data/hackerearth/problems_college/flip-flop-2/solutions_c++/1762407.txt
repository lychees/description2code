#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,i,s;
	cin>>t;
	while(t--) {
		string str;
		cin>>str;
		char c;
		s = 0;
		for(i = 0; str[i]; i++) {
			if(!i) {
				c = str[i];
			}
			if(str[i] == c) {
				s++;
			}
			else {
				c = str[i];
			}
		}
		cout<<s-1<<endl;
	}
    return 0;
}
