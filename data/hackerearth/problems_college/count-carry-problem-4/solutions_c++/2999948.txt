#include <iostream>
using namespace std;

int main()
{
    long long int t,n1,n2,carry,count=0;
    cin>>t;
    while(t--)
    {
    	carry=0;
    	count=0;
    	cin>>n1>>n2;
    	while(n1||n2)
    	{
    		carry=(n1%10+n2%10+carry)/10;
    		n1/=10;
    		n2/=10;
    		if(carry)
    		count++;
    	}
    	if(count==1)
    	cout<<count<<" carry operation\n";
    	else
    	if(count)
    	cout<<count<<" carry operations\n";
    	else
    	cout<<"No carry operation\n";
    }
    return 0;
}
