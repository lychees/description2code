typedef pair<int, int> pii;

const double eps = 1e-9;
const double pi = acos(-1.0);

double p[1111][1111];
double dp[1111][1111];
char s[111];
int n,m;

double get(int x, int y) {
	if (x == n || y == m) return (double)x + y;
	double &res = dp[x][y];
	if (res == res) return res;
	return res = get(x,y + 1) * p[x][y] + get(x + 1,y) * (1.0 - p[x][y]);
}

void solve() {
	cin >> n >> m;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			scanf("%lf",&p[i][j]);
		}
	}
	memset(dp,-1,sizeof(dp));
	double res = get(0,0);
	printf("%.6lf\n",res);
}

int main() {
	#ifdef LOCAL
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	#endif

	int T; cin >> T;

	while(T--) solve();
	
	return 0;
}
