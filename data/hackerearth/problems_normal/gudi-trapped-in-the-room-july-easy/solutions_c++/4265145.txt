using namespace std;
set < string > sd;
int add,shift;
int n;
void dfs(string tmp)
{
    sd.insert(tmp);
    string next;
    next=tmp;
    int i;
    rep(i,0,n)
    {
        next[i]=tmp[(i+shift)%n];
    }
    //cout << next << "\n";
    if(sd.find(next)==sd.end())
        dfs(next);
    next=tmp;
    rep(i,0,n)
    {
        if(i%2)
            next[i]=((((next[i]-'0')+add)%10)+'0');
    }
    if(sd.find(next)==sd.end())
        dfs(next);
}
