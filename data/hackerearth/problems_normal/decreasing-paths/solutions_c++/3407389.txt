const int mod = 1e9+7;
const int N = 1e3+10;
const int inf = 2e9;
const double eps = 1e-7;
const long long INF = (long long)1e18;
/* -------Template ends here-------- */ 
int a[N][N];
ll dp[N][N];
int n;
bool check(int x,int y,int val)
{
	if(x>0 && x<=n && y>0 && y<=n && a[x][y] > val)return true;
	return false;
}
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)for(int j=1;j<=n;j++)si(a[i][j]);
	for(int i=1;i<=n;i++)for(int j=1;j<=n;j++)dp[i][j]=1;
	for(int k=99;k>0;k--)
	{
		for(int i=1;i<=n;i++)for(int j=1;j<=n;j++)if(a[i][j]==k)
		{
			ll ret= 0;
			if(check(i-1,j,k))
				ret += dp[i-1][j];
			if(check(i+1,j,k))
				ret += dp[i+1][j];
			if(check(i,j-1,k))
				ret += dp[i][j-1];
			if(check(i,j+1,k))
				ret += dp[i][j+1];
			ret %= mod;
			dp[i][j]+=ret;
			dp[i][j]%=mod;
		}
	}
	ll ans = 0;
	for(int i=1;i<=n;i++)for(int j=1;j<=n;j++)ans = (ans + dp[i][j])%mod;
	cout<<ans;
	return 0;
}
