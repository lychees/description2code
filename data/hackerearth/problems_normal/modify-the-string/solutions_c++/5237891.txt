#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    string s;
    cin>>s;
    if(s.length()<1||s.length()>101) return -1;
    int i=0;
    while(s[i]!='\0'){
    	if(s[i]>64&&s[i]<91) s[i]+=32;
    	else if(s[i]>96&&s[i]<123) s[i]-=32;
    	else return -1;
    	i++;
    }
    cout<<s;
    return 0;
}
