#include <iostream>
#include<bits/stdc++.h>
using namespace std;
struct type
{
	long val;
	long pos;
	bool operator <(const type &t1)const
	{
		return val<t1.val;
	}
};
class comp
{
	public:
    bool operator ()(type t1,type t2)
	{
		return t1.pos<t2.pos;
	}
}c1;
int main()
{
    long n;
    scanf("%ld",&n);
    long arr[n];
    vector<type> a(n);
    long i;
    for(i=0;i<n;i++)
    {
    	scanf("%ld",&arr[i]);
    	a[i].val=arr[i];
    	a[i].pos=i;
    }
    sort(a.begin(),a.end());
    long m;
    scanf("%ld",&m);
    vector<type>::iterator it;
    /* for(it=a.begin();it!=a.end();it++)
        cout<<it->val<<","<<it->pos<<" ";*/
    while(m--)
    {
    	type k;
    	scanf("%ld",&k.val);
    	k.pos=0;
    	it=upper_bound(a.begin(),a.end(),k);
        if(it!=a.end())
        {
            while(it!=a.end())
            {
                it->val=it->val-1;
                it++;
            }
        }
    }
    sort(a.begin(),a.end(),c1);
    for(it=a.begin();it!=a.end();it++)
        cout<<it->val<<" ";
    return 0;
}
