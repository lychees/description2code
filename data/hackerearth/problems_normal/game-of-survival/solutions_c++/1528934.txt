#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,i,d;cin>>t;
    while(t--)
    {   d=0;
        int n; cin>>n;
        int a[n+1],s[n+1];
        for(i=0;i<n;i++)
            cin>>s[i];
        for(i=0;i<n;i++)
            cin>>a[i];
        sort(a+0,a+n);
        sort(s+0,s+n);
        for(i=0;i<n;i++)
        {
            if(a[i]>s[i])
            {d=1;break;}
        }
        if(d==1)
            cout<<"NO"<<endl;
        else
            cout<<"YES"<<endl;
    }
    return 0;
}
