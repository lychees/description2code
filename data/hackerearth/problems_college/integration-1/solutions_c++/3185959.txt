#include <bits/stdc++.h>
using namespace std;
int main(){
	long long int t,n,p,u,r,s,i;
	long long int c = 1;
	scanf("%lld",&t);
	while(t--){
		long long int count;
		scanf("%lld %lld %lld %lld %lld",&n,&p,&u,&r,&s);
		 count = p;
		for(i=1; i<=n; i++){
			if(count+r<=u)
			 count = count+r;
			else if(count-s>=0)
			 count = count-s;
			else
			 break; 	  
		}
		printf("Case #%d: %d\n",c++,count);
	}
	return 0;
}
