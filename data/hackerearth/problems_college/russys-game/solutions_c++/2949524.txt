#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i,flag=0;
		cin>>n;
		int *a=new int[n];
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		//1 -1   -2
		for(i=0;i<n;i++)
		{
			int x=a[i];
			if(a[i]==-1)
			{
				flag=1;
				break;
			}
			if(a[i]==-2)
			{
				flag=2;
				break;
			}
			a[i]=-2;
			i=x;
			i--;
		}
		if(flag==1)
			cout<<"Win\n";
		else
			cout<<"Lose\n";
	}
}
