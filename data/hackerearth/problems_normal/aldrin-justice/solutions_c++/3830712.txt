#include <iostream>
using namespace std;

int main()
{
     int t,bt1,bt2,mt1,mt2;
     cin>>t;
    while(t--)
    {
    	cin>>bt1>>bt2>>mt1>>mt2;
    	if(bt1>bt2)
    	{
    	bt1=bt1+bt2;
    	bt2=bt1-bt2;
    	bt1=bt1-bt2;
    	}
    	if(mt1>mt2)
    	{
    		mt1+=mt2;
    		mt2=mt1-mt2;
    		mt1-=mt2;
    	}
    	if(mt1<bt1)
    	{
    		if((mt2==bt1))
    		cout<<"Point";
    		else if(mt2<bt1)
    		cout<<"Line";
    		else if((bt2==bt1))
    		cout<<"Point";
    		else 
    		cout<<"Nothing";
    	}
    	else if(bt1<mt1)
    	{
    		if((bt2==mt1))
    		cout<<"Point";
    		else if(bt2<mt1)
    		cout<<"Line";
    		else if(mt1==mt2)
    		cout<<"Point";
    		else
    		cout<<"Nothing";
    	}
    	else if(bt1==mt1)
    	{
    		if((mt2==mt1)||(bt1==bt2))
    		cout<<"Point";
    		else
    		cout<<"Nothing";
    	}
    	cout<<endl;
    }
    return 0;
}
