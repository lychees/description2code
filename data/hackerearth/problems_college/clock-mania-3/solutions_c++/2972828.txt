#include <iostream>
using namespace std;
int min(int x,int y) 
{   if(x<y)
        return x;
    else
        return y;
}
 
int calcAngle(double h, double m)
{   if(h==12)
        h=0;
    if(m==60) 
        m=0;    
    int hour=0.5*(h*60+m);
    int minute=6*m;
    int angle=abs(hour-minute);
    angle=min(360-angle,angle);
    return angle;
}

int main()
{   int n;
    cin>>n;
    while(n--)
    {   int a,b;
        cin>>a>>b;
        cout<<calcAngle(a,b)<<"\n";
    }
    return 0;
}
