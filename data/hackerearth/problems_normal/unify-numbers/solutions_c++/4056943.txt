#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T;
    scanf("%d", &T);
    
    while(T--) {
    	int N;
    	scanf("%d", &N);
    	vector<long long> Vec;
    	for(int i = 0; i < N; ++i) {
    		long long Data;
    		scanf("%lld", &Data);
    		Vec.push_back(Data);
    	}
    	
    	sort(Vec.begin(), Vec.end());
    	int sz = Vec.size();
    	int mid = Vec[sz/2];
    	long long ans = 0;
    	for(int i = 0; i < sz; ++i)
    		ans += abs(Vec[i] - mid);
    	printf("%lld\n", ans);
    }
    return 0;
}
