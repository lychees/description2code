#include <bits/stdc++.h>
using namespace std;
int check(string s)
{
	int k=s.length();
	for(int i=0;i<=k/2;i++)
	{
		if(s[i]!=s[k-1-i])
		return 0;
	}
	return 1;
}
int main()
{int t;string s,s1,temp;
cin>>t;
while(t--){
	int sum=0;
	cin>>s>>s1;
	temp=s;
	for(int i=0;i<=s.length();i++)
	{ s.insert(i,s1);
	
		if(check(s))
		sum++;
	s=temp;
	}
	cout<<sum<<endl;
	
}
   
    return 0;
}
