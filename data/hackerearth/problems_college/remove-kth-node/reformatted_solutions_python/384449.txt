num = int(raw_input())
for i in range(num):
    total, step = raw_input().split()
    total = int(total)
    step = int(step)
    nxt = 0
    l = range(1, total+1)
    while len(l) > 1:
        delete = (nxt+step) % total
        l.pop(delete)
        total -= 1
        nxt = delete % total

    print l[0]
