#include<bits/stdc++.h>
using namespace std;

int main()
{
    ios::sync_with_stdio(0);
    int n,k;
    cin>>n>>k;
    int arr[n];
    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    long long ans=0;
    for(int i=0;i<n-k;i++)
    {
        if(((abs(arr[i]+arr[i+k])%k)==0) || ((abs(arr[i]-arr[i+k])%k)==0))
           ans++;
    }
    cout<<ans;
    return 0;
}
