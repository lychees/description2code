# print 'Hello World!'

for _ in range(input()):
    num = input()
    flag = True
    lst = map(int, raw_input().split())
    for i in range(1, num+1):
        if lst[lst[i-1]-1] != i:
            flag = False
            break
    if flag:
        print 'inverse'
    else:
        print 'not inverse'
