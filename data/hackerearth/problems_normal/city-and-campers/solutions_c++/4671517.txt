int a[100005],cap[100005];
int parent(int x)
{
	while(x!=a[x])
	{
		a[x]=a[a[x]];
		x=a[x];
	}
	return x;
}
int main()
{
    
    int n,q,aa,b;
    cin>>n>>q;
    set<pair<int,int> > ms;
    for(int i=1;i<=n;i++)
    {
    	a[i]=i;
    	cap[i]=1;
    	ms.insert({1,i});
    }
    set<pair<int,int> >::iterator it1,it2;
    while(q--)
    {
    	cin>>aa>>b;
    	int p1=parent(aa);
    	int p2=parent(b);
    	if(p1!=p2)
    	{
    		
    			ms.erase({cap[p1],p1});
    			ms.erase({cap[p2],p2});
    			a[p2]=p1;
    			cap[p1]+=cap[p2];
    			ms.insert({cap[p1],p1});
    		
    	}
    
    		it1=ms.begin();
