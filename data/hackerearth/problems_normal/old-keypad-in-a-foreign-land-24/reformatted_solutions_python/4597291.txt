n = int(raw_input())
fre = map(int, raw_input().split())
k = int(raw_input())
key = map(int, raw_input().split())
fre = sorted(fre, reverse=True)
key = sorted(key)
ans, idx = 0, 1
totkey = sum(key)
if totkey < n:
    print -1
    exit(0)
x = 0
for i in xrange(n):
    for j in xrange(k):
        if x < n:
            if key[j] > 0:
                key[j] -= 1
                ans = ans + idx*fre[x]
                x = x + 1
    idx = idx+1
print ans
