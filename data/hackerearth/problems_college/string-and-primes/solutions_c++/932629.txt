#include <iostream>
#include<math.h>
using namespace std;

int isprime(int a)
{ if(a==1)
return 0;
if(a==2)
return 1;
   int c;
   int sq=sqrt(a);
   for ( c = 2 ; c <=sq ; c++ )
   {
      if ( a%c == 0 )
     {return 0;}
   }
      return 1;
}


int main()
{
    int t;
    char c[100000];
    scanf("%d",&t);
    while(t--){
    	scanf("%s",c);
    	char *p;
    	p=c;
    	int flag=0;
    	while(*p){
    	 if(isprime((int)*p)==0)
    	   { 
    	   	cout<<*p; 
    	   flag=1;}
    	   p++;
    	}
    	if(flag==0)
    	cout<<"NULL";
    	cout<<"\n";
    }
    return 0;
}
