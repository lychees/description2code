def monkiness(i, a, b, n):
    l = i
    u = n-1
    pos = i
    while l <= u:
        mid = (l+u)/2
        if b[mid] >= a[i]:
            pos = mid
            l = mid+1
        else:
            u = mid-1
    return pos-i


for _ in xrange(int(raw_input())):
    n = int(raw_input())
    a = map(int, raw_input().split())
    b = map(int, raw_input().split())
    masans = [0]
    for i in range(n):
        masans.append(monkiness(i, a, b, n))
    print max(masans)
