		dic[s[i]]=[];
	dic[s[i]].append(i);
for i in range(m):
	temp=raw_input().split();
	loc=int(temp[0])-1;
	key=temp[1];
	strloc=dic[key][loc];
	del dic[key][loc];
	if not dic[key]:
		del dic[key]
testans=['' for x in range(len(s))]
