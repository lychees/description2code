#include <bits/stdc++.h>
using namespace std;
#define fo(i,n) for(int i=0;i<(n);i++)
#define MOD 1000000007
#define PB push_back
typedef long long ll;

ll N;
vector<ll> v;

inline bool ok (ll i) {
	for (ll j = 2; j*j <= i; j++) {
		ll t = 1;
		while (i%j == 0) i /= j, t *= j;
		if (2*t > N) return 0;
		if (2*i <= N) return 1;
	}
	if (2*i > N) return 0;
	return 1;
}

int main () {
	scanf("%lld", &N);
	ll i = 1 + N/2;
	while (ok(i)) i++;
	printf("%lld\n", i);
	return 0;
}
