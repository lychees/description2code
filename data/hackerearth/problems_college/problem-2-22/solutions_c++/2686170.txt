#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    for(int i=0;i<T;i++){
    	long int r,xc,yc;
    	cin>>r;
    	cin>>xc;
    	cin>>yc;
    	long int rhs=r*r;
    	int j=0;
    	bool flag=true;
    	for(j=0;j<4;j++){
    		long int x,y;
    		cin>>x;
    		cin>>y;
    		long int lhs=(x-xc)*(x-xc)+(y-yc)*(y-yc);
    		//cout<<lhs<<","<<rhs<<endl;
    		if(lhs>rhs){
    			flag=false;
    		}
    	}
    	if(flag==false)
    		cout<<"No"<<endl;
    	else
    		cout<<"Yes"<<endl;
    }
    return 0;
}
