for t in range(input()):
    aliceMap = {}
    bobMap = {}
    alice = raw_input()
    bob = raw_input()
    for c in alice:
        if c != ' ' and aliceMap.get(c) is None:
            aliceMap[c] = 1

        elif c != ' ':
            aliceMap[c] += 1
    for c in bob:
        if c != ' ' and bobMap.get(c) is None:
            bobMap[c] = 1
        elif c != ' ':
            bobMap[c] += 1
    ls = list(aliceMap.iterkeys())
    for key in ls:
        if bobMap.get(key) is None:
            pass
        else:
            if bobMap.get(key) > aliceMap.get(key):
                bobMap[key] -= aliceMap[key]
                del aliceMap[key]
            elif bobMap.get(key) < aliceMap.get(key):
                aliceMap[key] -= bobMap[key]
                del bobMap[key]
            else:
                del bobMap[key]
                del aliceMap[key]

    if aliceMap == {} and bobMap != {}:
        print 'You lose some.'
    elif bobMap == {} and aliceMap != {}:
        print 'You win some.'
    else:
        print 'You draw some.'
