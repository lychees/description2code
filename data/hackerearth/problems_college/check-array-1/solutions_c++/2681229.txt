#include <iostream>
using namespace std;

int main()
{
	int psums[1000003];
	int t;
	cin >> t;
	while(t--){
		for(int i = 0; i < 1000001; i++) psums[i] = 0;
		long long k;
		cin >> k;
		for(int i = 1; i <= k ; i++){
			long long l;
			cin >> l;
			psums[i] = psums[i-1]+l;
		}
		bool answ = false;
		for(int i = 1; i <= k; i++){
			if(psums[i-1] == psums[k]-psums[i]){
				answ = true;
			}
		}
		if(answ){
			cout << "YES" << endl;
		}else cout << "NO" << endl;
	}
	return 0;
}
