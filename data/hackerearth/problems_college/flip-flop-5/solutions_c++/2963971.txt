#include<bits/stdc++.h>
using namespace std;
int main()
{
    long  t;
    cin>>t;
    while(t--)
    {
        string s;
        cin>>s;
        long cou =0;
        for(long i=1;i<s.size();i++)
        {
            if(s[i-1]==s[i])
                cou++;
        }
        cout<<cou<<endl;

    }


    return 0;
}
