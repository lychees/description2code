#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n,k,a[100001],i;
    scanf("%d",&n);
    scanf("%d",&k);
    for(i=0;i<n;i++) {
    	scanf("%d",&a[i]);
    }
    sort(a,a+n);
    int mn=1000000001;
    int x;
    for(i=0;i<(n-k+1);i++)  {
    	x=(a[i+k-1]-a[i]);
    	if(mn>x)
    		mn=x;
    	//printf("%d\n",mn);
    }
    printf("%d\n",mn);
    return 0;
}
