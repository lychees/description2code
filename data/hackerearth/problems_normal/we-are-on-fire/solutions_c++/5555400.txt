#include <bits/stdc++.h>
int n,m;
int a[1005][1005];
int ans=0;
using namespace std;
void calculate(int x,int y)
{
	a[x][y]=2;
	ans--;
	if(x+1<=n && a[x+1][y]==1)
	calculate(x+1,y);
	if(x-1<=n && a[x-1][y]==1)
	calculate(x-1,y);
	if(y+1<=m && a[x][y+1]==1)
	calculate(x,y+1);
	if(y-1<=m && a[x][y-1]==1)
	calculate(x,y-1);
	
}
int main()
{
	int q;
 scanf("%d %d %d",&n,&m,&q);
 int i,j;
 for(i=1;i<=n;i++)
 {
 	for(j=1;j<=m;j++){
 	scanf("%d",&a[i][j]);
 	ans+=a[i][j];
 	}
 }
 while(q--)
 {
 	int x,y;
 	scanf("%d %d",&x,&y);
 	if(a[x][y]==1)
 	calculate(x,y);
 	printf("%d\n",ans);
 }
    return 0;
}
