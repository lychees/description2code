#include <iostream>
#include<string>
using namespace std;

int main()
{
    int t;
    cin>>t;
   string s[t]; 
    for(int i=0;i<t;i++){
    	cin>>s[i];
    }
    float a[t][2];
    for(int i=0;i<t;i++){
    	a[i][0]=s[i][0]*10+s[i][1]+0.01*(s[i][3]*10+s[i][4]);
    	a[i][1]=s[i][6]*10+s[i][7]+0.01*(s[i][9]*10+s[i][10]);
    }
    for(int i=0;i<t-2;i++){
    	for(int j=0;j<t-1-i;j++){
    		if(a[j][0]>a[j+1][0]) {
    		swap(a[j][0],a[j+1][0]);
    		swap(a[j][1],a[j+1][1]);
    		}
    	}
    }
    bool b=true;
    for(int i=0;i<t-1;i++){
   
    		if(a[i][1]>a[i+1][0]) b=false;
    	
    }
    if(b) cout<<"Who needs a moderator?"<<endl;
    else cout<<"Will need a moderator!"<<endl;
    return 0;
}
