'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
s = raw_input()
nword = True
new = ''
for x in s:
    if nword and x != ' ':
        new += x.upper()
        nword = False
    else:
        if x == ' ':
            nword = True
        new += x
print new
