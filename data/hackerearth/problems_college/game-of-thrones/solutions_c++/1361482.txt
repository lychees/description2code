#include<bits/stdc++.h>
#define ll long long
#define ld double
#define f first
#define s second
#define pb push_back
#define M 1000000007
#define MX 1000003
#define INF 10000000000
#define PI 3.141593
#define mmx(a,b) a>b?a:b
#define mmn(a,b) a<b?a:b
using namespace std;
typedef pair<ll,ll> ii;
typedef vector<ll> vv;
ll arr[200005];
int main()
{
   ll t;
   cin>>t;
   while(t--)
   {
       ll n,rep=0,prev=-1,ans=0;
       cin>>n;
       for(ll i=0;i<n;i++)
        cin>>arr[i];
       sort(arr,arr+n);
       prev=arr[0];
       for(ll i=0;i<n;i++)
       {
           if(arr[i]==prev)
            rep++;
           if(arr[i]!=prev||i==n-1)
           {
               ans+=rep-1;
               rep=1;
               prev=arr[i];
           }
       }
       cout<<ans<<"\n";
   }

}
