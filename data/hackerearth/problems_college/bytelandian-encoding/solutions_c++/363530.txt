#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

int main()
{
    int n;
    char str[50];
    scanf("%d",&n);
    while(n--)
    {
    	scanf("%s",str);
    	int len = strlen(str);
    	
    	int i=len/2-1,j;
    	if(len%2)
    	{
    		printf("%c",str[len/2]);
    		j=len/2+1;
    	}
    	else
    	{
    		j=len/2;
    	}
    	while(i>=0)
    	{
    		if(str[i]<str[j])
    			printf("%c%c",str[i],str[j]);
    		else
    			printf("%c%c",str[j],str[i]);

    		i--;j++;
    	}
    	printf("\n");
    }
    return 0;
}
