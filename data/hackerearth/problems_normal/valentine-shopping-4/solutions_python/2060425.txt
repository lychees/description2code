tc = input()
for t in xrange(tc):
	items, shops = raw_input().split(' ')
	bests = ""
	for i in range(int(items)):
		minp = 10000
		best = -1
		for s in range(int(shops)):
			price = 1000	
			x,y,z = map(int,raw_input().split(' '))
		 	price = price*(1.0-x/100.0)*(1.0-y/100.0)*(1.0-z/100.0)
			if price < minp : minp = price; best = s
	
		bests += (' ' if bests else '')+ str(best+1)
	print bests
