#include <bits/stdc++.h>

#define ll long long
#define pb push_back
#define mp make_pair
#define ld long double
#define line pair<ll, ll>

using namespace std;

const int nm = 1000002;

int n;
ll a[nm], sum[nm];

int main() {
#ifdef LOCAL
	freopen("input.txt", "r", stdin);
#endif
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i) {
		scanf("%lld", &a[i]);
		sum[i] = sum[i - 1] + a[i];
	}
	ll res = -1;
	ll smin = 0;
	for (int i = 1; i <= n; ++i) {
		res = max(res, sum[i] - smin);
		smin = min(smin, sum[i]);
	}
	if (res < 0)
		printf("0");
	else
		printf("%lld\n", res);
}
