#include <bits/stdc++.h>
using namespace std;
int p;
bool comp(string a, string b){
	return a.substr(0, p) <= b.substr(0, p);
}

int main()
{
	int t, n, k;
	cin >> t;
	while (t--){
		cin >> n >> k >> p;
		string s[n];
		for (int i = 0; i<n; ++i)
			cin >> s[i];
		stable_sort(s, s + n, comp);
		cout << s[k - 1] << endl;
	}
	return 0;
}
