#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int main()
{
int test;
cin>>test;
getchar();
while(test--)
{
string s;
getline(cin,s);
string temp = "";
for(int i = 0 ;i < s.length() ;i++)
{
if(s[i] == '.' || s[i] == ',' || s[i] == '?' || s[i] == '!' || s[i] == ' ')
	continue;
temp = temp + s[i];
}
//cout<<temp<<endl;
for(int i = 0,j = (temp.length() - 1);i <= j ;i++,j--)

{
//cout << "here";
	if(i == j && temp[i] == '*')
		cout << "*" <<endl;
	else
	if(temp[i] == '*')
		cout << temp[j] << endl;
	else	
	if(temp[j] == '*')
		cout << temp[i] << endl;
		
}
//getchar();
//cout<<s<<endl;
}
return 0;
}
