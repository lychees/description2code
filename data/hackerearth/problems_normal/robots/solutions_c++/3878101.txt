
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main(void)
{
	int t, n, i;
	long p;
	cin>>t;

	while(t--)
	{
		cin>>n;
		vector<int> M(n);
		for(i=0; i<n; ++i)
			cin>>M[i];

		sort(M.begin(), M.end());
		p = 1;
		for(i = 0; i < n; ++i)
		{
			if(M[i]>i)
				p = (p*(M[i]-i))%(1000000007);
			else
			{
				p = 0;
				break;
			}
		}
		cout<<p<<endl;
	}
	return 0;
}
