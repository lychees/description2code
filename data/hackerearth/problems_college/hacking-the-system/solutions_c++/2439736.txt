#include<bits/stdc++.h>
using namespace std;
# define ll long long
int main()
{
    string a,b;
    cin>>a;
    cin>>b;
    int flag=0;
    int i;
    for(i=0;i<a.size();i++)
        if(a[i]!=b[i])
        flag++;
    if(flag<=1)
        cout<<"wins";
    else
        cout<<"loses";
}
