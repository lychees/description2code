#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,a;
    set<int> s;
    scanf("%d", &t);
    while(t--){
        s.clear();
        scanf("%d", &n);
        while(n--){
            scanf("%d",&a);
            s.insert(a);
        }
        set<int>::iterator it2,it1,it;
        it2 = s.end();
        it2--;
        for(it1=s.begin(); ; it1++,it2-- ){
            if(*it1==*it2) cout<<*it1;
            else{
            cout<<*it2<<" "<<*it1<<" ";
            }
            it = it1;
            if(*(++it)==*(it2) || *it1==*it2) break;
        }
        printf("\n");
    }
    return 0;

}
