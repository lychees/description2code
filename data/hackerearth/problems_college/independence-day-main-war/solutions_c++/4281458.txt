    #include<bits/stdc++.h>
    using namespace std;

    int main()
    {
        //freopen("in.txt","r",stdin);
        int n,t;
        scanf("%d",&t);
        while(t--)
        {
            scanf("%d",&n);
            double x[n],y[n],sumx=0.0,sumy=0.0;
            for(int i=0;i<n;i++)
            {
                scanf("%lf %lf",&x[i],&y[i]);
                sumx+=x[i];
                sumy+=y[i];
            }
            sumx=sumx/n;
            sumy=sumy/n;
            printf("(%.2lf , %.2lf)\n",sumx,sumy);
        }
        return 0;
    }
