#include <bits/stdc++.h>

using namespace std;

long long int arr[1000001];

int main()
{
	int n, flag = 0;
	long long int result = -1;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> arr[i];
		
	sort(arr, arr + n);
	if (arr[n - 1] != arr[n - 2])
		result= arr[n - 1];
	else
	{
		for (int i = n - 2; i > 0; i--)
		{
			if (arr[i] != arr[i - 1] && arr[i] != arr[i + 1])
			{
				result = arr[i];
				flag = 1;
				break;
			}	
		}
	}
	if (arr[0] != arr[1])
		result = max(arr[0], result);
	cout << result << endl;
	return 0;
}
