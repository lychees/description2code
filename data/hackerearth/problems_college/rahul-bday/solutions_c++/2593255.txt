#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int b,w,x,y,z,min=0;
    	cin>>b>>w>>x>>y>>z;
    	
    	if(x<y && y>z+x)
    	{
    		min=(x*b)+(w*(x+z));
    		
    	}
    	else if(y<x && x>z+y)
    	{
    		min=(y*w)+(b*(y+z));
    	}
    	else
    	{
    		min=(b*x)+(w*y);
    	}
    	cout<<min<<endl;
    }
    return 0;
}
