#include <bits/stdc++.h>
using namespace std;

int main()
{
   int n,m,i,j;
   long long a[1001],mi,d,sum,ans;
   cin>>n>>m;
   for(int i=0;i<n;i++)
   cin>>a[i];
   sort(a,a+n);
   while(m--)
   {
   	d=1000000001;
   	cin>>mi;
   	i=0;j=n-1;
   	sum=a[i]+a[j];
   	while(i<j)
   	{
   		if(abs(sum-mi)<d)
   		{
   			d=abs(sum-mi);
   			ans=sum;
   		}
   	//	cout<<a[i]<<" "<<a[j]<<" "<<sum<<"\n";
   		if(sum<mi)
   		i++;
   		else if(sum>mi)
   		j--;
   		else
   		break;
   		sum=a[i]+a[j];
   	}
   	cout<<ans<<"\n";
   }
    return 0;
}
