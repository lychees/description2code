t = input()
# output=[]
for i in range(t):
    n = input()
    q = list(map(int, raw_input().split()))
    # k=0
    m = 0
    for k in range(n):
        x = q[k]
        if k == 0:
            tmp = x
            m += 1
        else:
            if tmp > x:
                tmp = x
                m += 1
        k += 1
    print m
