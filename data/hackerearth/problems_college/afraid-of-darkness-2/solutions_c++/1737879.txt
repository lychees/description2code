#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    int n;
    cin>>n;
    int *arr;
    arr=new int[n];
    for(int i=0;i<n;i++)
    {
    	cin>>arr[i];
    }
    
    int days=0;
    for(int i=0;i<n;i++)
    {
    	sort(arr,arr+n);
    	if(arr[n-i-1]>0)
    	{
    		days++;
    		for(int j=0;j<i+1;j++)
    			arr[n-j-1] -= 1;
    	}
    }
    cout<<days<<endl;
    }
    return 0;
}
