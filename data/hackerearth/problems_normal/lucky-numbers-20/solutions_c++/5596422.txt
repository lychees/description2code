#include <iostream>
#include <cstdio>
#include <math.h>
using namespace std;


int main()
{
    int t;
    unsigned long long n;
    scanf("%d", &t);
    while(t--){
        //scanf("%Ld", &n);
        cin >> n;
        unsigned long long  qsum = 0, one = 1;
        int d = floor(log2(n) + 1) ;
        for(int i = 0 ; i< d ; i++){
            for(int j = i+1; j<d; j++){
                unsigned long long t =  ((one << j) + (one << i));

                if(t <= n){
                    
                    qsum = (qsum + t ) % 1000000007;
                }
            }
        }
        //printf("%Ld\n", qsum);
        cout << qsum << endl;

    }
    return 0;

}
