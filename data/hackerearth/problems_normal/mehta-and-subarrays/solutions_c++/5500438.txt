#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pb push_back
ll arr[1000009];
int main()
{
	ll n,cnt=0;
	cin>>n;
	vector<pair<ll,ll> > v;
    v.pb({0,0});
    ll sum=0LL,a;
    //cout<<sum<<" ..."<<"\n";
	bool f=0;
    for(ll i=1;i<=n;i++)
    {
    	cin>>a;
    	if(a>=0) f=1;
    	//cout<<a<<"   ";
    	sum+=a;//cout<<sum<<endl;
    	v.pb({sum,i});
    }
    if(!f)
    {
    	cout<<"-1\n";
    	return 0;
    }
    sort(v.begin(),v.end());
//for(ll i=0;i<v.size();i++)  cout<<v[i].first<<"    "<<v[i].second<<endl;
    ll min_idx=LONG_MAX;
    ll max_len=0;
    for(ll i=0;i<v.size();i++)
    {
    	ll idx=v[i].second;
    	min_idx=min(min_idx,idx);
    	ll diff=idx-min_idx;
    	if(diff>max_len)
    	{
    		max_len=diff;
    		cnt=1;
    	}
    	else if(diff==max_len)
    		cnt++;
    }
    cout<<max_len<<" "<<cnt<<"\n";
    
}
