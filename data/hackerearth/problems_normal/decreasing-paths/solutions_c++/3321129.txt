#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
#define pii pair<int, int>
#define pll pair<long long, long long>
#define vi  vector<int>
#define pb  push_back
#define mp  make_pair
#define ALL(x) x.begin(),x.end()
#define M(a,x) memset(a,x,sizeof(a))
#define sci(x) scanf("%d",&x);
#define scl(x) scanf("%lld",&x);
#define scs(x) scanf("%s",x);
#define print(x) printf("%d",x);
#define nl printf("\n")
#define fr first
#define se second
#define printl(x) printf("%lld",x)
#define F(i,a,n) for(int i=a;i<n;i++)
#define INF 100000000000000000LL
#define LL long long

const int N = 1e3+3;
int a[N][N];
int dp[N][N];
int ans;
vector<pii> v[105];

const int dx[] = {1,-1,0,0};
const int dy[] = {0,0,1,-1};

int main() {
  int n;
  sci(n);
  F(i,1,n+1) {
    F(j,1,n+1) {
      sci(a[i][j])
        v[a[i][j]].pb(mp(i,j));
    }
  }
  for(int i=1;i<=100;i++) {
    for(int j=0;j<v[i].size();j++) {
      int x = v[i][j].fr;
      int y = v[i][j].se;
      dp[x][y]=1;
      for(int k=0;k<4;k++) {
        int nx = x+dx[k];
        int ny = y+dy[k];
        if(a[nx][ny]<a[x][y]) {
          dp[x][y]+=dp[nx][ny];
          if(dp[x][y]>=MOD) dp[x][y]-=MOD;
        }
      }
      ans+=dp[x][y];
      if(ans>=MOD) ans-=MOD;
    }
  }
  cout<<ans<<endl;
}
