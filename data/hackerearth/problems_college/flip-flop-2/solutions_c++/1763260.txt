#include <iostream>
using namespace std;

int main()
{
	ios_base::sync_with_stdio( false );
	int t, n, i, count;
	char first;
	
	cin >> t;
	
	while( t-- ) {
		
		string s;
		cin >> s;
		n = s.length();
		
		first = s[ 0 ];
		count = 0;
		
		for( i = 1;i < n;) {
			
			if( s[ i ] == first ) {
				
				while( ( i < n ) && ( s[ i ] == first ) ) {
					count++;
					i++;
				}
				
				if( i < n ) {
					first = s[ i ];
					i++;
				}
			}
			else {
				
				first = s[ i ];
				i++;
			}
		}
		
		cout << count << endl;
	}
	return 0;
}
