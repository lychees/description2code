#include<iostream>

using namespace std;

int reverseNum(int n)
{
    int rev =0;
    while(n!=0)
    {
        rev = rev*10;
        rev = rev + n%10;
        n = n/10;
    }
    return rev;
}

int main()
{
    ios_base::sync_with_stdio(false);cin.tie(NULL);
    int t,n,ans;
    int temp1,temp2;
    cin>>t;
    while(t-- > 0)
    {
        cin>>n;
        ans = n*n;
        temp1 = reverseNum(n);
        temp2 = reverseNum(temp1*temp1);
        if(temp2 == ans&&n!=temp1)
        {
            cout<<temp1<<"\n";
        }
        else
        {
            cout<<"No\n";
        }
    }
    return 0;
}
