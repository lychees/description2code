

    #include <iostream>
    using namespace std;
     
    int main()
    {
     int test;
     
     cin>>test;
     
     for(int i=0;i<test;i++)
     {
     	int sum=0;

     	int arr[n];
     	for(int j=0;j<n;j++)

     sum+=arr[j];

     }
    }
