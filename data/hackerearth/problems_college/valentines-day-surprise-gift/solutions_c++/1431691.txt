ll mod = 1000000007;
ll pr[]={3,5,7};
ll N;
ll gcd(ll a,ll b){
	if(a<b) swap(a,b);
	if(b==0) {
		if(a==0) return 1;
		return a;

	}
	return gcd(b,a%b);
}
ll mpow(ll a,ll d,ll mod) {
	if(d==0) return 1;
	if(d==1) return a;
	ll ans = mpow(a,d/2,mod);
	ans = (ans*ans)%mod;
	if(d&1) return (ans*a)%mod;
	return ans;
}
ll R(ll n) {
	if(n<=10){
		int i=0;
		ll ans = 1;
		while(i<=2 && pr[i]<=n ){
				ans = ans * pr[i];
				i++;
		}
		return ans;
	}
	ll ans =1;
	ans = mpow(2,ceil(n/4.0),mod);
	ans = ans *R(ceil(n/5.0));
	ans = ans %mod;
	ans = ans*R(ceil(n/10.0));
	ans = ans %mod;
	return ans;

}
void proc() {
	cin>>N;
	ll RN = R(N);
	ll mx = -11111111;
	ll rx,ry;
	//cout<<"RN: "<<RN<<endl;
	for(ll x = RN;x<=RN*5;x+=RN){
		for(ll y=RN;y<=RN*5;y+=RN){
			if(gcd(x,y)!=RN){
				continue;
			}
			ll val = mpow(x,y,mod);
			if(val>mx){
				mx = val;
				rx = x;
				ry = y;
			}
		}
	}
	cout<<mx<<endl;
}
int main() {
	int T;
	cin>>T;
	while(T--) proc();
}
