int main()
{
    fast;
    int t;
    cin>>t;
    while(t--)
    {
        ll x,y,n;
        cin>>x>>y>>n;
        int index = 3;
        ll sum = x+y;
        while(index <= n)
        {
            ll num = abs(x-y);
            sum += num;
            if(num==0)
            {
                int rem = n - index;
                sum = sum + (x+y) * (rem/3);
                if(rem % 3 != 0)
                {
                    sum = sum + x * (rem % 3) ;
                }
                break;
            }
            index++;
            x = y;
            y = num;
        }
        cout<<sum<<endl;
    }

    return 0;
}
