#include<bits/stdc++.h>
using namespace std;
int  main()
{
  int t;
  cin>>t;
  while(t--)
  {
  	int n,k;
  	cin>>n>>k;
  	vector<int> v;
  	for(int i=0;i<n;i++)
  	{
  		int temp;
  		cin>>temp;
  		v.push_back(__builtin_popcount(temp));
  	}
  sort(v.begin(),v.end(),greater<int>());
  long long s=0;
  for(int i=0;i<k;i++)
  {
  	s+=v[i];
  }
  	cout<<s<<"\n";
  	
  	
  }
 	
	
	return 0;
}
