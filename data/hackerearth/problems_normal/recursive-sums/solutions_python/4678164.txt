def count_sum(length,integer,count):
    count += int(length)*int(integer)
    return count
def count_sum1(asum):
    count1 = 0
    for i in asum:
        count1 += int(i)
    return count1
    
    
T = input()
res = ''
for i in range(T):
    M = input()
    add = 0
    count = 0
    for j in range(M):
        length,integer = raw_input().split()
        add += count_sum(length,integer,count)
    add = str(add)
    while len(add) != 1:
        add = count_sum1(add)
        add = str(add)
    res += add + '\n'
        
print res
