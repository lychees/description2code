#include<bits/stdc++.h>
using namespace std;
int main()
{
int t;
cin>>t;
for(int i=0;i<t;i++)
{
int n;
cin>>n;
int a[n],b[n],c=0,x=0;
for(int j=0;j<n;j++)
cin>>a[j];
for(int j=0;j<n;j++)
cin>>b[j];
for(int j=0;j<n-1;j++)
{
for(int k=0;k<n-1-j;k++)
{
if(a[k]<a[k+1])
{
c=a[k];
a[k]=a[k+1];
a[k+1]=c;
}
if(b[k]>b[k+1])
{
c=b[k];
b[k]=b[k+1];
b[k+1]=c;
}
}
}
for(int j=0;j<n;j++)
if(a[j]%b[j]==0||b[j]%a[j]==0)
x++;
cout<<x<<endl;
}
return 0;
}
