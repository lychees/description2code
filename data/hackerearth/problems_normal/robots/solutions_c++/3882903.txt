#include <iostream>
using namespace std;
long long arr[1000];
long long mod = 1000000007;
int main()
{
    int test;
    cin>>test;
    while(test--){
    	int n;
    	cin>>n;
    	for(int i=0;i<n;i++){
    		int num;
    		cin>>num;
    		int k = i-1;
    		while(k>=0&&arr[k]>num){
    			arr[k+1] = arr[k];
    			k--;
    		}
    		arr[k+1] = num;
    		
    	}
    	long long ans = arr[0];
    	long long ct = 1;
    	for(int i=1;i<n;i++){
    		arr[i] = arr[i] - ct;
    		ans = ans*arr[i];
    		ans = ans%mod;
    		ct++;
    		if(arr[i] == 0)
    			break;
    		
    	}
    	cout<<ans<<endl;
    	
    }
    
    
    return 0;
}
