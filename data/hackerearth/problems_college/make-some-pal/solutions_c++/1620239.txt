#include <iostream>
#include <cstdio>
#include <vector>

using namespace std;

typedef long long ll;

bool check (string s, int newLength) {
    int left = 0, right = newLength - 1;
    while (left <= right) {
        if (right >= (int)s.length()) {
            left++;
            right--;
            continue;
        }
        if (s[left] != s[right])
            return false;
        left++;
        right--;
    }
    return true;
}

int main() {
    int t;
    scanf ("%d", &t);
    for (int tests = 0; tests < t; tests++) {
        string s;
        cin >> s;
        for (int newLength = (int)s.length(); newLength <= (int)s.length() * 2; newLength++)
            if (check(s, newLength)) {
                printf ("%d\n", newLength);
                break;
            }
    }
    return 0;
}
