#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t, size;
	char s[100000];
	int visited[100000];
	cin >> t;
	while (t--)
	{
		cin >> s;
//		cout  << "size "<< strlen(s) << endl;
		size = strlen(s);
		for (int i = 0; i < size; i++)
			visited[i] = 0;
		int e = 0, flag = 0;
		for (int i = 0; i < size; i++)
			if (s[i] == 'O')
				e++;
				
		for (int i = 0; i < size; i++)
		{
			if (s[i] == 'X')
			{
				flag = 1;
				for (int j = i; j >= 0 && s[j] != '*'; j--)
				{
					if (s[j] == 'O' && visited[j] != 1)
					{
						visited[j] = 1;
						e--;
					}
				}
			}
			
			if (s[i] == '*')
				flag = 0;
			if (s[i] == 'O' && flag == 1 && visited[i] != 1)
			{
				visited[i] = 1;
				e--;
			}
		}
		cout << e << endl;
	}
	return 0 ;
}
