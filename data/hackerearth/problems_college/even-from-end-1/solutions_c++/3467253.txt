#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    int n, x;

    while (cin >> n) {
        vector<int> v;

        //cin >> n;
        for (int i = 0; i < n; i++) {
            cin >> x;
            if (x < 0) {
                int tmp = -x;
                if (tmp % 2 == 0) {
                    v.push_back(x);
                }
            } else {
                if (x % 2 == 0) {
                    v.push_back(x);
                }
            }

        }
        for (int i = v.size() - 1; i >= 0; i--) {
            cout << v[i] << " ";
        }
        if (v.size() == 0) {
            cout << "None to display";
        }
        cout << endl;
    }

    return 0;
}
