#include <iostream>
using namespace std;

int main()
{
    char str[100000];
    int t;
     cin>>t;
     while(t--)
     {
            int a=0;
     	cin>>str;
     	 for(int i=0;str[i]!=0;i++)
     	 {
           int k=0;
     	      if(i==0)
     	      {
     	          a=str[i]-97;
     	      }
     	     if(i!=0)
     	     {
     	        a=str[i]-str[i-1];
     	     }
             if(a<0&&a>-13)
             {
             	cout<<a<<" ";
                continue;
             }
            if(a==13||a==-13)
            {
              cout<<"13"<<" ";
        	  continue;
             }
           if(a<0){
           	  a=(-1)*a;
           	k++;
             }
            
            if(a>13)
            {
               if(k>0){
                    a=26-a;
                	cout<<a<<" ";
                    continue;}
              else
               a=a-26;
               cout<<a<<" ";
            	
            }
            else
                 cout<<a<<" ";
            	continue;
            
     	 }
        
         cout<<"\n";
     }
    return 0;
}
