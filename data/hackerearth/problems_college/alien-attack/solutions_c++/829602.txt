#include <iostream>
#include<cmath>
#define MOD 1000000007
#define ll long long int
using namespace std;
ll modular_pow(ll base, ll exponent, ll modulus)

{

    ll result = 1;

    while (exponent > 0)

    {

        if (exponent % 2 == 1)

            result = (result * base) % modulus;

        exponent = exponent >> 1;

        base = (base * base) % modulus;

    }

    return result;

}
