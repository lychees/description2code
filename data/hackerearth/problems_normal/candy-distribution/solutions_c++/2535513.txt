#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
unordered_map<ll int,ll int> mp;
ll int mysqr(ll int a)
{
	return (a*a)%mod;
}
ll int mypow(ll int a,ll int n)
{
	if(n==1)
	return a;
	if((n%2)==0)
	return mysqr(mypow(a,n/2));
	return (a*mysqr(mypow(a,n/2)))%mod;
}
ll int myfind(ll int n)
{
	ll int ans=0;
	if(mp.find(n)!=mp.end())
	ans=mp[n];
	else
	{
		ll int x,i;
		x=n-1;
		for(i=1;i*i<=x;i++)
		{
			if((x%i)==0)
			{
				if((x/i)>=2)
				ans=(ans+mypow(myfind(i),x/i))%mod;// x/i groups of i members each
				if(i!=(x/i) && i>=2)				
				ans=(ans+mypow(myfind(x/i),i))%mod;//i groups of x/i memebers each
			}
		}
		mp[n]=ans;
	}
	return ans;
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	mp[1]=1;mp[2]=1;
	ll int n;
	cin>>n;
	myfind(n);
	cout<<mp[n]<<endl;
	return 0;
}
