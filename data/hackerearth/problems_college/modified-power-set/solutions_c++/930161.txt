#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    int t;
    cin >> t;
    while (t--) {
        string s;
        cin >> s;
        int n = s.size();
        int f[256] = {};
        for (int i = 0; i < n; i++)
            f[s[i]]++;
        int distinct = 0;
        for (int i= 0; i < 256; i++)
            if (f[i]) distinct++;
        cout << (distinct * (distinct + 1))/2 << endl;
    }
}
