#include <iostream>
#include <bits/stdc++.h>
#define ll long long int
#define inf 1e18
#define mp make_pair
using namespace std;
 
int main()
{
	int n,q;
	cin >>n>>q;
	ll a[n+10];
	ll f[10010];
	for(int i=1;i<=n;i++){
		cin >>a[i];
	}
	ll mina = a[1];
	for(int i=1;i<=n;i++){
		mina = min(mina,a[i]);
	}
	
	for(int i=1;i<=mina;i++){
		f[i] = inf;
	}
	f[0] = 0;
	priority_queue<pair<ll,ll> > pq;
	pq.push(mp(0,0));
	
	while(!pq.empty()){
		ll dist = -pq.top().first;
		ll ver = pq.top().second;
		pq.pop();
		if(f[ver]<dist){
			continue;
		}
		
		for(int i=1;i<=n;i++){
			ll nxt = (ver+a[i])%mina;
			if(f[nxt] > f[ver] + a[i]){
				f[nxt] = f[ver] + a[i];
				pq.push(mp(-f[nxt],nxt));
			}
		}
	}
	while(q--){
		ll x;
		cin >>x;
		ll ver = x%mina;
		if(f[ver]<=x)
			cout <<"YES"<<endl;
		else
			cout <<"NO"<<endl;
	}
    return 0;
}
