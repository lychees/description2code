#include <iostream>
#include<bits/stdc++.h>


using namespace std;

int main(void)
{
 //   cout << "Hello World!" << endl;
   int ceo_height,coo_height,cto_height,number;
   cin>>ceo_height>>coo_height>>cto_height>>number;
   
   int height[number],i,j;
   for(i=0;i<number;i++)
   {
   	  cin>>height[i];
   }
   
   int A[number+3];
   A[(number+3)/2]=ceo_height;
   if(cto_height>coo_height)
   {
   	A[number+3-1]=cto_height;
   	A[0]=coo_height;
   }
   else
   {
   	A[number+3-1]=coo_height;
   	A[0]=cto_height;
   }
   
   
   
   for(i=0;i<number-1;i++)
   {
   	int minimum=i;
   	  for(j=i+1;j<number;j++)
   	  {
   	  	if(height[minimum]>height[j])
   	  	{
   	  		minimum=j;
   	  	}
   	  }
   	  swap(height[minimum],height[i]);
   }
   
   j=0;
   for(i=1;i<number+3-1;i++)
   {
   	  if(i!=(number+3)/2 && number>j)
   	  {
   	  	A[i]=height[j];
   	  	j++;
   	  }
   }
   int sum=0;
   for(i=0;i<(number+3-1);i++)
   {
   	sum=sum+abs(A[i+1]-A[i]);
   }
   cout<<sum<<endl;
   
    return 0;
}
