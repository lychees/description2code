#include <bits/stdc++.h>
#define si(n) scanf("%d",&n);
#define pi(n) printf("%d\n",n);
#define pl(n) printf("%lld\n",n);
#define sl(n) scanf("%lld",&n);
#define pd(n) printf("%lf\n",n);
#define ss(s) scanf("%s",s);
#define ps(s) printf("%s\n",s);
#define ll long long
#define mod 1000000007
#define pb push_back
#define maxn 100005
ll arr[maxn];
using namespace std;
int main()
{
    arr[0]=0LL;
    arr[1]=0LL;
    arr[2]=1LL;
    arr[3]=2LL;
    for(int i=4;i<maxn;i++)
    {
        arr[i] = (((2*arr[i-1])%mod+arr[i-2])%mod-(2*arr[i-3])%mod + mod -arr[i-4] +mod )%mod;
    }
    int t;
    si(t);
    while(t--)
    {
        int n;
        si(n);
        pl(arr[n]);
    }
}
