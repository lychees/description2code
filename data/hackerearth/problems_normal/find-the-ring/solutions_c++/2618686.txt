int ans1=1,ans2=1;
int func(int pos,int n)
{
	if(n==0)
	return pos;
	if(pos==0)
	return func(1,n-1);
	if(pos==2)
	return func(1,n-1);
	if(pos==1)
	{
		ans1=0.5*ans1;
		ans2*=0.5;
		if(ans1>=ans2)
		return func(0,n-1);
		return func(2,n-1);
	}
}
int main()
{
	int t;
	sd(t);
	while(t--)
	{
		int ind,n;
		sd(ind);sd(n);
		int ans=func(ind,n);
		pd(ans);pn;
	}
    return 0;
}
