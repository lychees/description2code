#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		double a,b,c,area,s,r,ans;
		double pi=3.1415;
		cin>>a>>b>>c;
		
		s=(a+b+c)/2.0;
		
   		area=sqrt(s*(s-a)*(s-b)*(s-c));
   		
   		
   		r = (a*b*c)/(4*area);
   		

   		
   		ans = r*r;
   		ans = pi*ans;
   		
   		std::cout<<std::fixed<<std::setprecision(4)<<ans<<std::endl;
   		
		
	}
}
