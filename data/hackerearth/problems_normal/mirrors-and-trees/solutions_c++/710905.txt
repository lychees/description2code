#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    long long int i,j,indexl,indexr,n,ans,temp,temp1,temp2,al[30],ar[30],il,ir;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld",&n);
        scanf("%lld",&temp);
        ar[0]=temp;
        ir=1;
        il=0;
        indexl=2;indexr=3;
        for(i=2;i<=n;)
        {
            temp1=0;
            temp2=0;
            for(i=indexl;i<=indexr;i++)
            {
                scanf("%lld",&temp);
                if(temp1==0)
                {
                    temp1=temp;
                }
                else
                {
                    if(temp!=0)
                        temp2=temp;
                }

            }
            indexl*=2;
            indexr=2*indexl-1;
            if(temp1!=0 && temp2==0)
                ar[ir++]=temp1;
            else if(temp1!=0 && temp2!=0)
                al[il++]=temp1;
            if(temp2!=0)
                ar[ir++]=temp2;
        }
        for(i=0;i<ir;i++)
            printf("%lld\n",ar[i]);
        for(i=0;i<il;i++)
            printf("%lld\n",al[i]);
        printf("\n");
    }
}
