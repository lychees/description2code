#include <iostream>
#include <vector>

int main() {
	auto N = 0;
    std::cin >> N;
    
    std::vector<long> vec;
    vec.reserve(N);
    while (N--) {
    	auto i = 0l;
    	std::cin >> i;
    	vec.push_back(i);
    }
    
    std::cin >> N;
    while (N--) {
    	auto i = 0l, j = 0l, cnt = 0l;
    	std::cin >> i;
    	auto found = false;
    	for (const auto& itr : vec) {
    		++cnt;
    		j += itr;
    		if (j >= i) {
    			found = true;
    			break;
    		}
    	}
    	std::cout << (found ? cnt : -1) << "\n";
    }
}
