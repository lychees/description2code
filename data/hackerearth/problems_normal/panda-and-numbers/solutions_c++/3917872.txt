#include<bits/stdc++.h>
using namespace std;
long long int arr[1000006],a,b,c,d,t,n,m;
int main()
{
	arr[1]=1;
	arr[4]=1;
	arr[27]=1;
	arr[256]=1;
	arr[3125]=1;
	arr[46656]=1;
	arr[823543]=1;
	for(int i=2;i<=1000000;++i)
	{
		if(arr[i]==1)
		continue;
	
		d=i;
		while(d)
		{
			b=d%10;
			d=d/10;
			if(i-b*b>=0)
			{
				if(arr[i-b*b]==1)
				{
				   arr[i]=1;
				   break;	
				}
			}
		}
		
	}
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		if(arr[n]==1)
		printf("Yes\n");
		else
		printf("No\n");
	}
	return 0;
}
