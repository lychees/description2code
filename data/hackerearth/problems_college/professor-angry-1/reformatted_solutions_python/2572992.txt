'''

'''
t = input()
while t > 0:
    x = raw_input()
    x = x.split()
    k = int(x[1])
    y = raw_input()
    y = y.split()
    j = 0
    for i in y:
        if int(i) <= 0:
            j = j+1
    if j >= k:
        print "NO"
    else:
        print "YES"
    t = t-1
