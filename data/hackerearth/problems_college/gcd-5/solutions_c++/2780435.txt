#include<bits/stdc++.h>
using namespace std;
#define FOR(i,a,b) for(int i=a;i<b;i++)
#define rep(i,n) for(int i=0;i<n;i++)
#define pint(n) printf("%d\n",n)
#define plong(n) printf("%lld\n",n)
#define sint(n) scanf("%d",&n)
#define slong(n) scanf("%lld",&n)
typedef long long int ll;
int main()
{
    int t;
    sint(t);
    while(t--)
    {
        int n;
        sint(n);
        int flag=0;
        ll a[n],b[n];
        rep(i,n)
        {
            slong(a[i]);
            b[i]=a[i];
        }
        sort(a,a+n);
        rep(i,n)
        {
            if(a[i]!=b[i])
            {
                flag=1;
                break;
            }
        }
        if(flag==1)
            printf("NO\n");
        else
            printf("YES\n");
    }
    return 0;
}
