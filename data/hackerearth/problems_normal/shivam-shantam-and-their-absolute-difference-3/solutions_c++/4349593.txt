#include <bits/stdc++.h>
using namespace std;
#define lim 800005
#define offset 400000
#define MOD 1000000007
int dp[2][lim];
int main()
{
    int q,n;
    cin>>n;
    int a[n],b[n];
    for(int i=0;i<n;++i)
    	cin>>a[i];
    for(int i=0;i<n;++i)
    	cin>>b[i];
    for(int i = 0; i < 2; i++)
		for(int j = 0; j < lim; j++)
			dp[i][j] = 0;
    dp[0][offset]=1;
    
    for(int i=0;i<n;++i){
    	for(int j=0;j<lim;++j){
    		if(dp[0][j]){
    			dp[1][j]+=dp[0][j];
    			dp[1][j]%=MOD;
    			
    			dp[1][j+a[i]]+=dp[0][j];
    			dp[1][j+a[i]]%=MOD;
    			
    			dp[1][j-b[i]]+=dp[0][j];
    			dp[1][j-b[i]]%=MOD;
    			
    			dp[1][j+a[i]-b[i]]+=dp[0][j];
    		    dp[1][j+a[i]-b[i]]%=MOD;
    			
    		}
    	}
    	for(int j=0;j<lim;++j){
    		dp[0][j]=dp[1][j];
    		dp[1][j]=0;
    	}
    }
    cin>>q;
    int res=0;
    int l=max(0,offset-q);
    int r=min(lim,offset+q);
    for(int i=l;i<=r;++i)
    	res=(res+dp[0][i])%MOD;
    cout<<res<<endl;
    return 0;
}
