#include <bits/stdc++.h>
using namespace std;

int main()
{


	int n, k;
	cin>>n>>k;
	
	int *arr = new int[n+1];
	for(int i=0; i<n; i++)
	{
      int a ;
		cin>>a;
		arr[i] = a;
		
	}
	if(k>n)k=n;int i=0, j=0;
  while(j<k && i<n)
	{

	  int maxx = arr[i];int track = i;
	  for(int j=i+1; j<n; j++)
	  {

		  if(arr[j]>maxx){maxx = arr[j]; track = j;}

	  }
      if(i==track){i++;continue;}
	  int tmp = arr[i];
	  arr[i] = arr[track];
	  arr[track] = tmp;
      i++;j++;
      
  }
	for(int i=0; i<n; i++)cout<<arr[i]<<" "; 
	return 0;
}
