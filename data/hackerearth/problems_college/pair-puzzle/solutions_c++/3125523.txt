void cal(){
    lol n,k,di,re,ans=0;
    cin>>n>>k;
    di=n/k;
    re=n%k;
    vl a(k,di);
    for(int i=1;i<=re;i++){
        a[i]++;
    }
    for(int i=1;i<=((k-1)/2);i++){
        ans+=(a[i]*a[k-i]);
    }
    ans+=((a[0]*(a[0]-1))/2);
    if(k%2==0){
        k/=2;
        ans+=((a[k]*(a[k]-1))/2);
    }
    cout<<ans<<nl;
}

int main()
{
    ios::sync_with_stdio(false);cin.tie(NULL);cout.tie(NULL);
    int t;
    cin>>t;
    while(t--){
        cal();
    }

}
