n= int(input())
for i in range(n):
    A,B=map(int,raw_input().split())
    result=0
    while A > 0 and B > 0:
            if A > B:
                  result+=A / B
                  A = (A % B)
            else:
                  result+=  B /A
                  B = (B %A)

    print(result-1)
