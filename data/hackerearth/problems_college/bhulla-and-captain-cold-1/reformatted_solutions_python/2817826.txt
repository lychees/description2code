for _ in xrange(T):
    q = int(raw_input())
    l = []
    dp = [[-1 for e in xrange(3)]for w in xrange(q)]
    for e in xrange(q):
        l.append([int(_) for _ in raw_input().split()])
    dp[0] = l[0]
    for e in xrange(1, q):
        dp[e][0] = min(dp[e-1][1], dp[e-1][2])+l[e][0]
        dp[e][1] = min(dp[e-1][0], dp[e-1][2])+l[e][1]
        dp[e][2] = min(dp[e-1][0], dp[e-1][1])+l[e][2]
    print min(dp[q-1])
