#define m0(a) memset(a,0,sizeof(a))
#define M1(a) memset(a,-1,sizeof(a))
#define bits(a) __builtin_popcount(a)
#define N 1000001
using namespace std;

int x[N];
int y[N];
int arr[N];
int main()
{
int t , n;
cin >> t;
while(t--){
    cin >> n;
    for (int i = 0 ; i < n ; i++){
        cin >> arr[i];
    }
    m0(x) ; m0(y);
    ll res = n;
    for (int i = n-1 ; i >= 0 ; i--){
        if (arr[i] >= 0){
            if (x[arr[i]] == 0){x[arr[i]] = 1;}
            else {
                    res += x[arr[i]];
                    x[arr[i]]++;
            }
        }
        else {
            if (y[arr[i]*-1] == 0){y[arr[i]*-1] = 1;}
            else {
                res += y[arr[i]*-1];
                y[arr[i]*-1]++;
            }
        }
    }
    cout << res << "\n";
}


  return 0;
}
