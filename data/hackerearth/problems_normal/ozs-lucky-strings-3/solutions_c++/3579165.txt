#include <iostream>
using namespace std;

int main()
{
    int cases;
    string input;
    cin>>cases;
    while(cases--){
    	cin>>input;
    	int i = 0; 
    	int j = input.size()-1;
    	int ans = 0;
    	while(i<j){
    		if(input[i] == 'R' && input[j] == 'K') {
    			i++;
    			j--;
    			ans +=2;
    		}
    		else if(input[i]=='K') i++;
    		else if(input[j]=='R') j--;
    	}
    	cout<<ans<<endl;
    }
}
