#include <iostream>
using namespace std;
int t;
long int n,i,j;
int main()
{
    cin>>t;
    while(t--)
    {
    	long int c=0;
    	cin>>n;
    	for(i=1;i<=n;i++)
    	{
    		for(j=i;j<=n;j++)
    		{
    			if((i*j)%(i+j)==0&&i!=j)
    			{
    				c++;
    			}
    		}
    	}
    	cout<<c<<endl;
    }
    return 0;
}
