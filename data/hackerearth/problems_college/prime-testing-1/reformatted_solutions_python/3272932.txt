import math
noOfTestCases = int(raw_input())


def checkPrime(no):
    if (no == 0 or no == 1):
        return False
    start = 2
    end = int(math.sqrt(no))
    while (start <= end):
        if (no % start == 0):
            return False
        start += 1
    return True


def count(nos):
    c = 0
    for no in nos:
        if (checkPrime(no)):
            c += 1
    return c
