#include <iostream>
#include <vector>
#include <unordered_set>
using namespace std;

int main()
{
    vector<unsigned int> cycles;
    vector<unsigned int> height;
    unsigned int T;
    cin >> T;
    while (T-- > 0)
    {
        unsigned int N;
        cin >> N;
        vector<unsigned long> code;
        while (N-- > 0)
        {
            unsigned long x;
            cin >> x;
            code.push_back(x);
        }
        int soldiers = 0;
        unordered_set<unsigned long> seen;
        for (int i = 0; i<code.size(); i++)
        {
            if (seen.find(code[i]) == seen.end())
            {
                seen.emplace(code[i]);
            }
            else
            {
                soldiers += 1;
            }
        }
        cout << soldiers << endl;
    }

    return 0;
}
