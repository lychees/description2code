#include <iostream>
#include <string.h>
using namespace std;

int main()
{
    int n,cnt;
    char a[3],b[3];
    cin>>n;
    while(n--)
    {
    	cin>>a;
    	cin>>b;
    	cnt=0;
    	for(int i=0;i<strlen(a);i++)
    	if((a[i]-48)+(b[i]-48)>9)
    	cnt++;
    	if(cnt==0)
    	cout<<"No carry operation\n";
    	if(cnt==1)
    	cout<<cnt<<" carry operation\n";
    	if(cnt>1)
    	cout<<cnt<<" carry operations\n";
    }
    return 0;
}
