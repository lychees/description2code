#include <iostream>
#include <vector>
#include <cmath>
#include <map>
using namespace std;

int main()
{
    long long int p1,p2,t,n,result;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	result=0;
    	for(p1=1;p1<=sqrt(n);p1++)
    	{
    		if(n%p1==0)
    		{
    			if(p1%4!=0)
    				result+=p1;
    			if(p1*p1!=n)
    			{
    				if((n/p1)%4!=0)
    					result+=n/p1;
    			}
    		}
    	}
    	result*=8;
    	cout<<result<<endl;
    }
    return 0;
}
