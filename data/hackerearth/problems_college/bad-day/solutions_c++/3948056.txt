#include <iostream>
using namespace std;
//zone
long long dp[100005];
 
int main()
{
    int t;
    cin >> t;
    
    while(t--) {
    	int n;
    	cin >> n;
    	for(int i = 0; i <= n; i++) dp[i] = 0;
    	dp[0] = 1;
    	for(int i = 1; i <= n; i++) {
    		dp[i] += dp[i-1];
    		if(i > 1) {
    			dp[i] += dp[i-2];
    		}
    	}
    	cout << dp[n] << endl;
    }
    return 0;
}
