#include<bits/stdc++.h>
using namespace std;

int main(){
	
	int t,tt,i;
	string str;
	vector<string> arr;
	
	cin>>t;
	t= t+1;
	
	
	while(t--){
		
		getline(cin,str);
		arr.push_back(str);
	}
	tt = arr.size();
	for(i=0;i<tt;i++){
		
		sort(arr[i].begin(),arr[i].end());
		cout<<arr[i]<<endl;
	}
	return 0;
}
