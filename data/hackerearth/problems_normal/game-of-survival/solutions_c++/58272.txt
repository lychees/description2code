#include<stdlib.h>
#include<stdio.h>
using namespace std;
int compare (const void * a, const void * b)
{
	  return ( *(int*)a - *(int*)b );
}

int main()
{
	int t,n,i;
	int a[100010];
	int b[100010];
	int flag;
	scanf("%d",&t);
	while(t--)
	{
		flag=0;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		for(i=0;i<n;i++)
		{
			scanf("%d",&b[i]);
		}
		  qsort (a, n, sizeof(int), compare);
		    qsort (b, n, sizeof(int), compare);
		    for(i=0;i<n;i++)
		    {
			    if(a[i]<b[i])
			    {
				    flag=1;
				    break;
			    }
		    }
		    if(flag==1)
		    {
			    printf("NO\n");
		    }
		    else
			    printf("YES\n");

	}
	return 0;
}
