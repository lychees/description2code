#include<bits/stdc++.h>
using namespace std;
int main() {
	int t;
	scanf("%d",&t);
	while(t--) {
		long long a,b;
		long long ans=0;
		scanf("%lld%lld",&a,&b);
		long long carry=0;
		while(a!=0||b!=0) {
			long long dig1=a%10;
			long long dig2=b%10;
			long long sum=dig1+dig2+carry;
			if(sum>9)
				ans++,carry=sum/10;
			else
				carry=0;
			a/=10;
			b/=10;
		}
		if(ans==0)
			printf("No carry operation\n");
		else if(ans==1)
			printf("1 carry operation\n");
		else
			printf("%lld carry operations\n",ans);
	}
	return 0;
}
