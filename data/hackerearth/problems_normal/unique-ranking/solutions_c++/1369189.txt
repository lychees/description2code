#include <bits/stdc++.h>

using namespace std;

int N, K, Q;
int A[1001][21];
int B[21];

int main()
{
    scanf("%d%d", &N, &K);
    for(int i=1; i<=N; i++)
        for(int j=1; j<=K; j++)
            scanf("%d", A[i]+j);
    scanf("%d", &Q);
    while(Q--)
    {
        int n;
        scanf("%d", &n);
        for(int i=0; i<n; i++)
            scanf("%d", B+i);
        vector<int> v(n, 0x3f3f3f3f);
        int idx=0;
        for(int i=1; i<=N; i++)
        {
            vector<int> tmp;
            for(int j=0; j<n; j++)
                tmp.push_back(A[i][B[j]]);
            if(tmp<v)
            {
                v=tmp;
                idx=i;
            }
        }
        printf("%d\n", idx);
    }
    return 0;
}
