#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long int n,i,y,count=0;
    cin>>n;
    long int a[n];
    for(i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    y=a[0];
    while(a[n-1]!=0)
    {
      ++count;
      for(i=0;i<n;i++)
      {
      	if(a[i]>=y && a[i]<=y+4)
      	a[i]=0;   
      }
    
      for(i=0;i<n;i++)
      {
      	if(a[i]!=0)
      	{
      	y=a[i];
        break;
      }
    }
    }
    cout<<count<<endl;
    return 0;
}
