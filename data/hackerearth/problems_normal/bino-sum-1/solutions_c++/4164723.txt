ll dp[1001][1001];
ll ncr(int n,int r)
{
    if(r==0 or r==n)
        return dp[n][r]=1;
    if(r<0 or n<0)
        return 0;
    if(dp[n][r]!=-1)
        return dp[n][r];
    else
        return dp[n][r] = (ncr(n-1,r)+ncr(n-1,r-1))%mod;
}
int main()
{   
    memset(dp,-1,sizeof(dp));
    int test;
    cin>>test;
    while(test--)
    {
        int n,p;
        cin>>n>>p;
        ll ans = 0;
        for(int i=0;i<=p;++i)
            ans = (ans+ncr(n,i))%mod;
        cout<<ans<<endl;
    }
}
