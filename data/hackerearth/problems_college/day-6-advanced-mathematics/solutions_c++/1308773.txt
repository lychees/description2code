#include <iostream>
#include <string.h>
#include <stdio.h>
using namespace std;

int main()
{
	int T;
    cin >> T;
    while(T--)
    {
    	char word[100];
    	scanf("%s",word);
    	int len = strlen(word), kiss = 0;
    	for(int i=1;i<len;i++)
    	{
    		if(word[i] == word[i-1])
    		{
    			cout << "SLAP\n";
    			kiss = 1;
    			break;
    		}
    	}
    	if(kiss == 0)
    		cout << "KISS\n";
    }
    return 0;
}
