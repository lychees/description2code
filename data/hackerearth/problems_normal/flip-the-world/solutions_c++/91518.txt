#include <iostream>
# include<cstdio>
# include<cstring>
using namespace std;

int main()
{
	int t=0;
	scanf("%d",&t);
	while(t--)
	{
		int n, m;
		char a[100][100];
		scanf("%d%d",&n,&m);
		for(int i=0;i<n;i++)
				scanf("%s",a[i]);
				
		int count=0;
		int b[50]={0};
		memset(b,0,sizeof(int)*50);
		for(int i=n-1 ;i >=0;i--)
			for( int j = m-1 ;j >= 0 ;j--)
				if(a[i][j] == '0' &&  b[j]%2 == 0)
						{for(int k=j ;k>=0;k--) b[k]++;count++;}
				else if(a[i][j] == '1' && b[j]%2 != 0)
						{for(int k=j ;k>=0;k--) b[k]++;count++;}	
						
		printf("%d\n",count);
						
		
	}
    return 0;
}
