#include <bits/stdc++.h>
#include <iostream>
#include <string>
using namespace std;

int check_(string s){
		int flag=1,j=s.length()-1;
		for(int i=0;i<s.length()/2;++i){
			if(s[i]!=s[j-i])
			flag=0;
		}
		return flag;
}


int main()
{	
	int t;
	string s1,s2,s3,s4;
	for(cin>>t;t>0;--t){
		cin>>s1;
		cin>>s2;
		s3=s1+s2;
		s4=s2+s1;
		if(check_(s3)||check_(s4))
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
	}
    return 0;
}
