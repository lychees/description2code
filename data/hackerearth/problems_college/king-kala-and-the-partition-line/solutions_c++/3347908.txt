#include <iostream>
using namespace std;

int main()
{
  	int t;
  	cin>>t;
  	while(t--)
  	{
  		long long int n;
  		cin>>n;
  		int a,b;
  		long long int c,p,k,s,count=0,count1=0;
  		cin>>a>>b>>c;
  		cin>>p>>k;
  		long long int z=a*p+b*k+c;
  		n=n-1;
  		while(n--)
  		{
  			cin>>p>>k;
  			s=a*p+b*k+c;
  			if((s>0 and z>0) or (s<0 and z<0))
  				count++;
  			else if((s<0 and z>0) or (s>0 and z<0))
  				count1++;
  	
  		}
  		if(count>=count1)
  			cout<<count+1<<endl;
  		else
  			cout<<count1<<endl;
  	}
    return 0;
}
