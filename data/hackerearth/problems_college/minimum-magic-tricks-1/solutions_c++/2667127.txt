#include<bits/stdc++.h>
using namespace std;

const int N=1010;
const int md=1e9+7;

int dp[N];

int powmod(int a,int b) {
  if(b==0) {
    return 1;
  }
  int p=powmod(a,b/2);
  p=1ll*p*p%md;
  if(b&1) {
    p=1ll*p*a%md;
  }
  return p;
}
int add(int& a,int b) {
  a+=b;
  if(a>=md) {
    a-=md;
  }
}
int main() {
  dp[1]=1;
  dp[2]=1;
  for(int i=3;i<N;i++) {
    dp[i]=powmod(2,i-2);
    add(dp[i],dp[i-2]);
  }
  int tt;
  cin>>tt;
  while(tt--) {
    int n;
    cin>>n;
    cout<<dp[n]<<endl;
  }
  return 0;
}
