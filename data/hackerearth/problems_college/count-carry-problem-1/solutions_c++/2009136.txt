#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    long long int t,x,y;
    cin>>t;
    long long int a,b;
    while(t--){
        cin>>a>> b;
        long long int carry=0;
    while((a!=0)&&(b!=0)){
            x=a%10;  
           y=b%10;
        if((x+y)>9)
            carry++;
            a=a/10;
            b=b/10;
        }
        if(carry>1)
                cout<<carry<<" carry operations"<<endl;
        else
            if(carry==1)
             cout<<"1 carry operation"<<endl;
            else 
                cout<<"No carry operation"<<endl;
    }
    return 0;
}
