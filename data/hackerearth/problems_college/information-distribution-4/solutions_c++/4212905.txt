LL power(LL a,LL b,LL m)
{
	LL d=1;
	while(b)
	{
		if(b&1)
			d=(d*a)%m;
		a=(a*a)%m;
		b=b>>1;
	}
	return d%m;
}
LL nCr(LL n,LL r,LL m)
{
	LL dp[102][102];
	clr(dp);
	for(int i=0;i<=n;i++)
	{
		for(int j=0;j<=r && j<=i;j++)
		{
			if(j==0 || j==i)
				dp[i][j]=1;
			else
				dp[i][j]=(dp[i-1][j-1]+dp[i-1][j])%m;
		}
	}
	return dp[n][r]%m;
}
LL lucas(LL n,LL r,LL m)
{
	if(n==0 && r==0)
		return 1;
	LL ni=n%m;
	LL ri=r%m;
	if(ri>ni)
		return 0;
	return lucas((n/m),(r/m),m)*nCr(ni,ri,m);
}
