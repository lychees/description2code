#include <iostream>
#include <algorithm>
#include <cstdio> 

#define MOD 1000000009

using namespace std;

typedef long long int ll;

void multiply(ll F[2][2],ll M[2][2])
{
	ll x =  (F[0][0]*M[0][0] + F[0][1]*M[1][0])%MOD;
  	ll y =  (F[0][0]*M[0][1] + F[0][1]*M[1][1])%MOD;
  	ll z =  (F[1][0]*M[0][0] + F[1][1]*M[1][0])%MOD;
  	ll w =  (F[1][0]*M[0][1] + F[1][1]*M[1][1])%MOD;
 
  	F[0][0] = x;
  	F[0][1] = y;
  	F[1][0] = z;
  	F[1][1] = w;
}

void power(ll F[2][2],ll n)
{
	ll M[2][2]={{1,1},{1,0}};
	if(n==0||n==1)
		return;
	power(F,n/2);
	multiply(F,F);
	if(n&1)
		multiply(F,M);
}

ll fib(ll n)
{
	ll F[2][2]={{1,1},{1,0}};
	if(n==0)
		return 0;
	power(F,n-1);
	return (F[0][0])%MOD;
}


int main()
{
	int t;
	ll n,sum;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld",&n);
		ll t1,t2;
		t1 = fib(n+1);
		t2 = fib(n);
		sum = (t1*t2)%MOD;
		printf("%lld\n",sum);
	}
	return 0;
}
