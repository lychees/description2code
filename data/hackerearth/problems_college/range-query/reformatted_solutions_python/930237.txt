from sys import stdin
from collections import Counter
a = stdin.readline()
b = stdin.readline().split()
t = Counter(b)
le = len(b)
for i in range(0, int(stdin.readline())):
    c = stdin.readline().split()
    l = int(c[0])
    m = int(c[1])
    if (m-l+1) == le:
        print t[c[2]]
    elif l == 0 and m > (le/2):
        print t[c[2]]-b[m+1:le].count(c[2])
    elif l < (le/2) and m == le-1:
        print t[c[2]]-b[0:l].count(c[2])
    else:
        print b[l:m+1].count(c[2])
