#include <bits/stdc++.h>
using namespace std;
bitset <2005> g[2005];
int main(){
	cin.sync_with_stdio(0);
	int n,i,j,x; long long ans=0, cnt;
	cin >> n;
	for(i=1;i<=n;++i) for(j=1;j<=n;++j)
		cin >> x, g[i][j]=x;

	for(i=1;i<=n;++i) for(j=i+1;j<=n;++j)
		cnt = (g[i]&g[j]).count(), ans += cnt*(cnt-1)/2;

	cout << ans/2 << endl;
}
