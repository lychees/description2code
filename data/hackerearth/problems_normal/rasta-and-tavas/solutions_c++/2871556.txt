#include <iostream>
using namespace std;

int main()
{
    int max=1000000,i,j,c=0;
    int n,k;
    cin>>n;
    cin>>k;
    int p[max+1];
    for(i=0;i<=max;i++)
    {
    	p[i]=0;
    }

    for(i=2;i<=n;i++)
    {
    	if(p[i]==0)
    	{
    		for(j=1;i*j<=n;j++)
    		{
    			p[j*i]=p[j*i]+i;
    		}
    	}
    }
    for(i=1;i<=n;i++)
    {
    	if(p[i]==k)
    	{
    		c++;
    	}
    }
    cout<<c;
    return 0;
}
