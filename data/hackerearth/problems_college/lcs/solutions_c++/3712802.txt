ll gcd(ll n1,ll n2){
	if(n1%n2==0)return n2;
	return gcd(n2,n1%n2);
}
ll powmod(ll base,ll exponent)
{
	ll ans=1;
	while(exponent){
		if(exponent&1)ans=(ans*base)%mod;
		base=(base*base)%mod;
		exponent/=2;
	}
	return ans;
}
ll rev[upperlimit+1][10];
ll arr[upperlimit+1][10];
bool dp[upperlimit+1][upperlimit+1];
vector <int> bef[upperlimit+1];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	int i,j,k,t,n,k1;
	cin>>n>>k;
	for(j=1;j<=k;j++){
		for(i=1;i<=n;i++){
			cin>>arr[i][j];
			rev[arr[i][j]][j]=i;
		}
	}
	for(i=1;i<=n;i++)for(j=1;j<=n;j++)if(i!=j){
		bool f=1;
		for(k1=1;k1<=k;k1++){
			if(rev[i][k1]<rev[j][k1])f=0;
		}
		if(f)bef[i].pb(j);
	}
	int answer=0;
	// for(i=1;i<=n;i++)for(j=0;j<bef[i].size();j++)cout<<i<<" "<<bef[i][j]<<endl;

	for(i=2;i<=n;i++){
		for(j=1;j<=n;j++){
			for(k1=0;k1<bef[j].size();k1++){
				dp[i][j]|=dp[i-1][bef[j][k1]];
			}
		}
	}
	for(j=n;j>=1;j--)for(i=1;i<=n;i++)if(dp[j][i])answer=max(answer,j);
	cout<<answer<<endl;

	return 0;
}
