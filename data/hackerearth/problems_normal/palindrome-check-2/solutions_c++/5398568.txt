#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	string st;
	cin >> st;
    int s=0;
    int e=st.length()-1;
    while(s<e)
    {
    	if(st[s]!=st[e])
    	{
    		cout << "NO" << endl;
    		return 0;
    	}
    	else
    	{
    		s++;
    		e--;
    	}
    }
    cout << "YES"<< endl;
    
    return 0;
}
