#include <bits/stdc++.h>

#define pb push_back

using namespace std;

double eps = 1e-9;
const int INF = 1e9+7;
const int MAXN = int (1e6+7);

int n, t, a[100], b[100];

int main () {
    //freopen("database.in", "r", stdin);
    //freopen("database.out", "w", stdout);  
    
    cin >> t;
    for(int w = 0; w < t; w ++) {
    	int ans = 0;
    	cin >> n;
    	for(int i = 0; i < n; i ++) {
    		cin >> a[i];
    		a[i] --;
    	}
    	for(int i = 0; i < n; i ++) {
    		cin >> b[i];
    		ans ^= a[i] - b[i] + 1;
    		ans ^= b[i] - 1;
    	}
    	if(ans)
    		cout << "First" << endl;
    	else
    		cout << "Second" << endl;
    }
    return 0;
}
