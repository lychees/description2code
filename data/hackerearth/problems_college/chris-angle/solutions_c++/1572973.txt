#include<iostream>
using namespace std;
long int hcf(long int a,long int b);
main()
{
	int t;
	long int A,a,b,anstemp;
	cin>>t;
	while(t--)
	{
		cin>>A;
		a=A%360;
		b=180;
		anstemp=hcf(a,b);
		a=a/anstemp,b=b/anstemp;
		if(a==1 && b==1)
		cout<<"$"<<endl;
		else if(a==1)
		cout<<"$/"<<b<<endl;
		else if(b==1)
		cout<<a<<"$"<<endl;
		else
		cout<<a<<"$/"<<b<<endl;
	}
}
long int hcf(long int a,long int b)
{
	if(b!=0)
	return hcf(b,a%b);
	else
	return a;
}
