#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
using namespace std;
typedef long long ll;
const int maxn=1e6+5;
bool isPrime[maxn];
ll dp[maxn];
bool prime(ll val) {
    if(val<2) return false;
    for(ll i=2;i*i<=val;i++) {
        if(val%i==0) return false;
    }
    return true;
}
int main() {
    for(ll i=2;i<maxn;i++) {
        if(!isPrime[i]) {
            for(ll j=i*i;j<maxn;j+=i) isPrime[j]=true;
        }
    }
    dp[1]=0;
    for(int i=2;i<maxn;i++) {
        dp[i]+=dp[i-1];
        if(!isPrime[i]) dp[i]+=i;
    }
    int t;
    scanf("%d",&t);
    for(int i=0;i<t;i++) {
        int m,n;
        scanf("%d%d",&m,&n);
        ll val=dp[n]-dp[m-1];
        if(prime(val)==true) printf("YES\012");
        else printf("NO\012");
    }
    return 0;
}
