#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	string s;
	cin>>s;
	string d=s;
	reverse(s.begin(),s.end());
	if(d==s)
	{cout<<"-1"<<endl;
	continue;
	}
	
	else
	{
		int f;
		for(int i=0;i<d.length();i++)
		{
			
			string a=d.substr(0,i);
			//cout<<"a "<<a<<endl;
			string b=d.substr(i+1,d.length()-i);
			//cout<<"b "<<b<<" efjwe"<<endl;
			string h=a;
			h+=b;
		string p=h;
		reverse(p.begin(),p.end());
		if(p==h)
		{
		f=i;
		break;
		}
			
			
		}
		cout<<f<<endl;
		
		
		
	}
	
	
	}
   // cout << "Hello World!" << endl;
    return 0;
}
