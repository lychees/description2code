ll dp[1001][1001];
ll bpow(ll x,ll n)
{
	ll ans=1;
	while(n>0)
	{
	    if(n&1) ans*=x;
	    x*=x;
	    ans%=mod;
	    x%=mod;
	    n/=2;
	}
	return ans;
}


int main()
{
	ll i,j,n,k,ans=0,tot=0,gcd;
	cin>>n>>k;
	dp[0][0]=1;
	for(i=1;i<=n;i++)
	{
		for(j=0;j<=min(k-1,i-1);j++)
		dp[i][0]+=dp[i-1][j];
		for(j=1;j<=min(k-1,i);j++)
		dp[i][j]+=dp[i-1][j-1];
	}
	for(i=0;i<k;i++)
	ans+=dp[n][i];
	tot=(1LL<<n);
	ans=tot-ans;
	//ans=(ans%mod+mod)%mod;
	gcd=__gcd(ans,tot);
	ans=ans/gcd;
	tot=tot/gcd;
	cout<<ans<<"/"<<tot;
	return 0;
}
