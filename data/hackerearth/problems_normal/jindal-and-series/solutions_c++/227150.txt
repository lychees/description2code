#include <iostream>
#define N 1000000
#define MOD 10000007
using namespace std;
typedef unsigned long long int ull;
ull index[N+2];
int main()
{
    index[0]=1;
    index[1]=1;
    index[2]=1;
    index[3]=2;
    index[4]=4;
    for(ull i=5;i<=N;i++)
    {
    	index[i]=(index[i-1]+index[i-3]+index[i-4])%MOD;
    }
    
    ull t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	ull tmp=index[n];
    	cout<<tmp<<endl;
    }
    return 0;
}
