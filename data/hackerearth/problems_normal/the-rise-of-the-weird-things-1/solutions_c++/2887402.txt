#include <iostream>
#include<bits/stdc++.h>

#define rep(i,a,n) for(int i=a;i<n;i++)

using namespace std;

int main()
{
    int t;
    cin >> t;
    
    int A[t];
    
    rep(i,0,t)
    	cin >> A[i];
    
    int i=0,j=t-1;
    
    while(i < j)
    {
    	if(A[i] % 2 == 1 && A[j] % 2 == 0)
    		swap(A[i],A[j]);
    	else if(A[i] % 2 == 0)
    		i++;
    	else if(A[j] % 2 == 1)
    		j--;
    }
    
    sort(A,A+i);
    
    int cn = 0;
    rep(k,0,i)
    {
    	cout << A[k] << ' ';
    	cn += A[k];
    }
    
    cout << cn << ' ';
    cn = 0;
    
    
    sort(A+i,A+t);
    rep(k,i,t)
    {
    	cout << A[k] << ' ';
    	cn += A[k];
    }
    cout << cn ;
    cout << endl;
    return 0;
}
