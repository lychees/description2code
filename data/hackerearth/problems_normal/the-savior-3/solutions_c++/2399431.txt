#include <iostream>
using namespace std;

int main()
{long int t;
cin>>t;
while(t--)
{ long long int n,i,j,k,c=0,d=0;
  cin>>n;
  long long int a[n];
  for(i=0;i<n;i++) cin>>a[i];
  for(k=0;k<n-1;k++)
  {
  for(j=k+1;j<n;j++)
  { c=0;c=a[k]+a[j];
  if((a[k]!=a[j])&&(c%2==0)) d=d+1;
  }
  }
  cout<<d<<endl;
}
    return 0;
}
