#include <bits/stdc++.h>

using namespace std;

int main()
{

    int t , n , r , d;

    cin >> t;

    while(t--){

        cin >> n >> d >> r;

        int remaindr = n % d;
        int digit;

        while(r--){

            remaindr *= 10;
            digit = remaindr / d;
            remaindr %= d;
        }

        cout << digit << "\n";
    }

    return 0;
}
