#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    long int t,i,n,k,ans;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        int a[n];
        ans=0;
        for(i=0;i<n;i++)
            cin>>a[i];
        for(i=0;i<n;i++)
            if(a[i]==0 && k!=0)
            {
                a[i]=1;
                ans++;
                k--;
            }
            else if(a[i]==1)
                ans++;
        if(k>0)
            if(k%2!=0)
                ans--;
        cout<<ans<<"\n";
    }
}
