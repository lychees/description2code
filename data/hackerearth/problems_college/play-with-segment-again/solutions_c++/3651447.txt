ll power(ll a,ll b)
{
	if(!b)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2);
	temp=(temp*temp);
	if(b&1)
		temp=(temp*a);
	return temp;
}
int main()
    {
    	int i,t,n,m,a,b,val1,val2;
    	ll ans=0;
    	sc("%d",&t);
    	while(t--)
    	{
    		sc("%d",&n);
    		vi v(n);
    		for(i=0;i<n;i++)
    			sc("%d",&(v[i]));
    		sc("%d",&m);
    		set<pp > s;
    		set<pp >::iterator it;
    		for(i=0;i<m;i++)
    		{
    			sc("%d%d",&a,&b);
    			val1=min(a,b);
    			val2=max(a,b);
    			s.insert(mp(val1,val2));
    		}
    		vi dp(n);
    		for(i=1;i<n;i++)
    		{
    			val1=min(v[i],v[i-1]);
    			val2=max(v[i],v[i-1]);
    			it=s.find(mp(val1,val2));
    			if(it!=s.end())
    			{
    				dp[i]=i;
    			}
    			else
    				dp[i]=dp[i-1];
    		}
    		ans=0;
    		for(i=0;i<n;i++)
    		{
    			//cout<<dp[i]<<" ";
    			ans+=dp[i];
    		}
    		pf("%lld\n",ans);
    	}
    return 0;
}
