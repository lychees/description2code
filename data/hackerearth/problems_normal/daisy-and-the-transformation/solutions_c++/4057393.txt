#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t,n,temp,tn;
    string s1,s2;
    cin>>t;
    while(t--){
    	cin>>s1;
    	cin>>s2;
    	cin>>n;
    	tn = n;
    	temp = 0;
    	// while(n--){
    		
    	// }
    	for(int i=0;i<s1.size();i++){
    		if(s1[i]!=s2[i]){
    			// s1[i] = s2[i];
    			temp += 1;
    		}
    	}
    	if(s1 == s2){
    		if(n%2 == 0){
    			cout<<"Yes"<<endl;
    		}
    		else{
    			cout<<"No"<<endl;
    		}
    	}
    	else if(temp == n){
    		cout<<"Yes"<<endl;
    	}
    	else if(temp<n){
    		n = n-temp;
    		if(n%2 == 0){
    			cout<<"Yes"<<endl;
    		}
    		else{
    			cout<<"No"<<endl;
    		}
    	}
    	else{
    		cout<<"No"<<endl;
    	}
    	
    		
    }
    return 0;
}
