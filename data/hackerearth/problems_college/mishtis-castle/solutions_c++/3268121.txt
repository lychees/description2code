#include <iostream>
#include<bits/stdc++.h>
using namespace std;

long long int a[1001][1001],dp[1001][1001];
int main()
{
	ios::sync_with_stdio(false);
	long long int i,j,n,x,c,t;
	cin>>t;
	while(t--)
	{
		cin>>n>>c;
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=n;j++)
			cin>>a[i][j];
		}
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=n;j++)
			{
				dp[i][j]=a[i][j];
				x=max(dp[i-1][j],dp[i][j-1]);
				dp[i][j]+=x;
			}
		}
		if(dp[n][n]<c)
		cout<<"-1"<<endl;
		else
		cout<<dp[n][n]-c<<endl;
	}
	return 0;
}
