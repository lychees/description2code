#include<iostream>
#include<string.h>
#include<cstdio>
using namespace std;
int main()
{
    int test_cases,i,j,l,ind;
    cin>>test_cases;
    getchar();
    while(test_cases>0)
    {
        char str[101],s[101];
        gets(str);
        l = strlen(str);
        i=0,j=0;
        while(i<l)
        {
        	if((str[i]>='a' && str[i]<='z') || str[i] == '*')
        	{
        		s[j] = str[i];
        		j++;
        	}
        	if(str[i] == '*')
        		ind = j-1;
        	i++;
        }       
        s[j]='\0';
        l=strlen(s);        
        	cout<<s[l-ind-1]<<endl;        
        test_cases--;
    }
    return 0;
}
