#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        vector<int> v(n);
        for(int i = 0; i < n; ++i)
            cin>>v[i];
        int a = 0;
        cout<<*min_element(v.begin(),v.end())-1<<" ";
        a = 0;
        for(int i = 0; i < n; ++i)
        {
            v[i] -= 2;
            a += v[i];
        }
        cout<<a+n<<endl;
    }
    return 0;
}
