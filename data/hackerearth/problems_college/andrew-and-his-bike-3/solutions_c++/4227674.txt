#include <bits/stdc++.h>
#define ll long long
#define MY_LONG_MAX (((1ll<<62))|((1ll<<62)-1))
#define MY_LONG_MIN (1ll<<63)
using namespace std;
ll int n;//n junctions
list<ll int> l[100010];
bool visited[100010];
ll int dp[100010];
ll int mydfs(ll int x)
{
	ll int ans=0;
	if(!visited[x])
	{
		visited[x]=1;
		ans=1;
		list<ll int>::iterator it;
		for(it=l[x].begin();it!=l[x].end();it++)
		{
			if(!visited[*it])
			ans+=mydfs(*it);
		}
		dp[x]=ans;
	}
	return ans;
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	memset(visited,0,sizeof(visited));
	memset(dp,0,sizeof(dp));
	cin>>n;
	ll int i,x,y;
	for(i=0;i<(n-1);i++)
	{
		cin>>x>>y;
		x--;y--;//0-based index
		l[x].push_back(y);l[y].push_back(x);
	}
	mydfs(0);
	for(i=0;i<n;i++)
	{
		if(i==0)
		cout<<dp[i]-1<<" ";
		else
		cout<<dp[i]<<" ";
	}
	cout<<endl;
	return 0;
}
