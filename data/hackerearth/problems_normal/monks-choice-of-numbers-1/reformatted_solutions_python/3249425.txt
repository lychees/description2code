import sys
t = int(raw_input())
while (t > 0):
    n, k = map(int, sys.stdin.readline().split(" "))
    a = map(int, sys.stdin.readline().split(" "))

    for i in range(0, n):
        s = 0
        while (a[i] > 0):
            a[i] &= a[i]-1
            s += 1
        a[i] = s
    a = sorted(a, reverse=True)
    s = 0
    for i in range(0, k):
        s += a[i]
    print s
    t -= 1
