#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
ll A,B,S, minRes = -1;
ll memo[17][150];

ll solve(ll prefix, int digits, int sum){
    if (sum < 0) return 0;
    ll maxi = prefix, mini = prefix;
    for (int i = 0; i < digits; i++){
        mini *= 10;
        maxi = maxi * 10 + 9;
    }
    if (maxi < A || mini > B) return 0;
    if (digits == 0){
        if (sum > 0) return 0;
        if (minRes == -1) minRes = prefix;
        return 1;
    }
    bool memoize = (mini >= A && maxi <= B);
    if (memoize && memo[digits][sum] != -1) return memo[digits][sum];

    ll ret = 0;
    for (int dig = 0; dig < 10; dig++)
        ret += solve(prefix * 10 + dig, digits - 1, sum - dig);

    if (memoize) memo[digits][sum] = ret;
    return ret;
}

int main()
{
    cin >> A >> B >> S;
    memset(memo, -1, sizeof memo);
    cout << solve(0, 15, S) << "\n";
    cout << minRes << "\n";
}
