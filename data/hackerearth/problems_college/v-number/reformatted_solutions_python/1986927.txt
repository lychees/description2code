import sys
t = int(sys.stdin.readline())
while t > 0:
    n = [int(x) for x in list((sys.stdin.readline()).strip())]
    (l, h) = (len(n), len(n)/2)
    (i, j) = (0, h)
    (sum_low, sum_up) = (0, 0)
    while (i < h) and (j < l):
        sum_low += n[i]
        sum_up += n[j]

        i += 1
        j += 1
    if sum_low == sum_up:
        print "YES"
    else:
        print "NO"
    t -= 1
