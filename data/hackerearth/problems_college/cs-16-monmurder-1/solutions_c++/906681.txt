#include <iostream>
#include<cmath>
using namespace std;

#define max 1000000

int prime[80000]={2,3};
bool allprime[max]={false};
//jiski chance hai wo jeet jaayega matlab true.
bool health[max+1]={true,true,true,true};
int index=2;


int main()
{
	for(int i=0;i<max;i++)
		allprime[i]=false;
	allprime[2]=true;
	allprime[3]=true;
    for(int i=5;i<max;i++)
    {
    	int cond = sqrt(i);
    	int j;
    	for(j=0;prime[j]<=cond;j++)
    	{
    		if(i%prime[j]==0)
    			break;
    	}
    	if(prime[j]>cond)
    	{
    		prime[index++]=i;
    		allprime[i]=true;
    	}
    }
    
    for(int i=4;i<max;i++)
    {
    	if(allprime[i]==true)
    	{
    		health[i]=true;
    		continue;
    	}
    	health[i]=false;
    	int s=sqrt(i);
    	s++;
    	int j;
	int dec=1;
    	for(j=0;prime[j]<=s;j++)
    	{
		
    		if(i%prime[j]==0)
    		{
    			int val=prime[j];
    			while(val<=i)
    			{
    				if(i%val!=0)
    					break;
    				int temp=i/val;
    				if(temp==1 || (temp>1 && health[temp]==false))
    				{
    					health[i]=true;
    					break;
    				}
    				val*=prime[j];
    			}
			dec*=(val/prime[j]);
    		}
    		if(health[i]==true)
    			break;
		
    	}
	int kot=i/dec;
	if(allprime[kot]==true)
	{
		if(health[i/kot]==0)
			health[i]=true;
	}
    }
    health[max]=false;
    int t;
    cin>>t;
    while(t--)
    {
    	int h;
    	cin>>h;
    	if(health[h]==true)
    		cout<<"Printf{}"<<endl;
    	else
    		cout<<"Pappu Bhujia"<<endl;
    }
    return 0;
}
