#include <iostream>
using namespace std;




int main()
{
	
	string result;
	int n;
	cin>>n;
	int index;
	string a;
	cin>>a;
	int i;
	int val=122;
	for(i=0;i<n;i++){
	if(val<97)
	break;
	else if((i==n-1)&&(val!=a[i])){
		val--;
		i=index;
	}
	
	else if(val==a[i]){
	result=result+a[i];
	val--;
	index=i;
	}
	
}

cout<<result;
	
	
	
	 
    
    return 0;
}
