n=input()
l=[]
for i in range(0,n):
    s=raw_input()
    temp=[]
    length=len(s)
    for j in range(0,length):
        temp.append(s[j])
    temp.sort()
    l.append([s[0],s[length-1],temp])
l.sort()
l.append('*')
i=0
count=0
groups=0
while i<n:
    if l[i]==l[i+1]:
        count+=1
        i+=1
    else:
        count=0
        groups+=1
        i+=1
print groups
