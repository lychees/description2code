#include <iostream>
using namespace std;
#define MAXN 10000001
 
bool DP[MAXN] ;
int T , N ;
int main()
{	DP[0] = 1 ;
    for(int i=1;i<MAXN;i++){
    	DP[i] |= (i-6  >= 0 ? DP[i-6 ] : 0 ) ;
    	DP[i] |= (i-9  >= 0 ? DP[i-9 ] : 0 ) ;
    	DP[i] |= (i-20 >= 0 ? DP[i-20] : 0 ) ;	
    }
    cin >> T ;
    while(T--){
    	cin >> N ;
    	puts(DP[N] ? "True" : "False") ;
    }
    return 0;
}
