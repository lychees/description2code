N = input()
d = {}
anagramCounter = {}
maxAnagramCount = 1
for _ in xrange(N):
	raw = raw_input().strip()
	s = ''.join(sorted(raw))
	
	sumOfChars = 0
	for c in range(len(s)):
		sumOfChars += ord(s[c])
	
	if sumOfChars not in d:
		d[sumOfChars] = [s]
	else:
		shouldAddToList = True
		for word in d[sumOfChars]:
			if word == s:
				shouldAddToList = False
				if s in anagramCounter:
					anagramCounter[s] += 1
				else:
					anagramCounter[s] = 2
					
				if anagramCounter[s] > maxAnagramCount:
					maxAnagramCount = anagramCounter[s]
		if shouldAddToList:
			d[sumOfChars].append(s)
					
					
print maxAnagramCount
