#include<bits/stdc++.h>
using namespace std;
int id(int a[], int t[], int n, int m)
{
	int nrot=0;
	for(int i=0;i<n;i++)
		if (t[i] == a[0])
			nrot = i;
	return nrot;
		
	
}

int main()
{
	bool done = false;
	int n, m;
	int a[100002],  t[100002];
	int x;
	int posindex = 0;
	cin>>n;
	cin>>m;
	int morig = m;
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=0;i<n;i++)
		cin>>t[i];
	int rotreq = id(a,t,n,m);

	while(m--)
	{
		char ch;
		int x;
		cin>>ch>>x;
		if(ch=='R')
		{
			posindex = (posindex + x)%n;
		}
		else{
			posindex = posindex-x;
			if(posindex<0)
			{
				int val = posindex/n;
				val = val-1;
				val = -val;
				posindex = posindex + val*(n);
			}
		}
		if(posindex==rotreq)
		{
			cout<<morig-m<<endl;
			done= true;
			break;
			
			}	
	}
	if(done==false)
	{
		cout<<"-1"<<endl;
	}
	
}
