t = raw_input()
for i in range(int(t)):
    value = raw_input()
    list1 = []
    count = 0
    for j in range(len(value)):
        if (value[j] in list1):
            f = list1.index(value[j])
            del list1[0:f+1]
            list1.append(value[j])
        else:
            list1.append(value[j])
            if (len(list1) > count):
                count = len(list1)

    print count
