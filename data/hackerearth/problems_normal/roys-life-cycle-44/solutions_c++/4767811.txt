#include <iostream>
#include <string>
using namespace std;

int main()
{
    string s;
    int n;
    int carry=0,max=0,maxtot=0,test=0,testtot=0;
    cin>>n;
    while(n--){
    	cin>>s;
    	test=0;
    	for(int i=0;i<s.length();i++){
    		if(s[i]=='C'){
    			test++;
    			testtot++;
    			if(test>max)
    				max=test;
    			if(testtot>maxtot)
    				maxtot=testtot;
    		}
    		else{
    			if(testtot>maxtot)
    				maxtot=testtot;
    			test=testtot=0;
    		}
    	}
    }
    cout<<max<<" "<<maxtot<<endl;
    return 0;
}
