#include <iostream>
using namespace std;
long long fibs[1000001];
int main()
{
    long long t,i,j,n,m=1000000007;
    fibs[0]=1,fibs[1]=2;
    
    for(i=2;i<=1000000;i++)
    
    	fibs[i]=(fibs[i-1]+fibs[i-2])%m;
    	
    	for(i=2;i<=1000000;i++)
    	fibs[i]+=(fibs[i-1])%m;
    
    cin>>t;
    
    for(i=1;i<=t;i++)
    {
    	cin>>n;
    	cout<<fibs[n]%m<<endl;
    }
}
