#include <bits/stdc++.h>
#include <unistd.h>
using namespace std;
#define ll long long
#define ull unsigned long long
#define inf std::numeric_limits<ll>::max();
#define MOD 1000000007

int main(){
	ios::sync_with_stdio(0);
	int t;
	cin>>t;
	for(int z=0;z<t;z++){
		int n;
		string s;
		cin>>n>>s;
		int count[26]={0};
		for(char& c:s){
			count[int(c)-int('a')]+=1;
		}
		int ans=-1,max_count=-1;
		string ans_string;
		for(int i=0;i<n;i++){
			string temp;
			int temp_count=0;
			cin>>temp;
			for(char& c:temp){
				temp_count+=count[int(c)-int('a')];
			}
			if(temp_count>max_count){
				max_count=temp_count;
				ans=i+1;
				ans_string=temp;
			}
			else if(temp_count==max_count && temp.length()<ans_string.length()){
				ans=i+1;
				ans_string=temp;
			}
		}
		cout<<ans<<'\n';
	}
	return 0;
}
