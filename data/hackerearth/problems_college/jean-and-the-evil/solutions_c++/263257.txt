#include<iostream>
using namespace std;
#include<stdio.h>
int MaxContSum(int a[],int n)
{
	int M[1000000];
	int maxsum=0,i;
	if(a[0]>0)
	   M[0]=a[0];
	else
	   M[0]=0;
	for(i=1;i<n;i++)
	{
		if(M[i-1]+a[i] > 0)
		   M[i]=M[i-1]+a[i];
		else
		   M[i]=0;   
	}      
	
	for(i=0;i<n;i++)
	{
		if(M[i]>maxsum)
		    maxsum=M[i];
	}
	return maxsum;
}

int main()
{
  int t,n,a[1000000],i;
  scanf("%d",&t);
  while(t--)
  {
  	 scanf("%d",&n);
  	 for(i=0;i<n;i++)
  	 {
  	 	scanf("%d",&a[i]);
  	 }
  	 int res=MaxContSum(a,n);
     printf("%d\n",res);  	 
  }
  return 0;
}
