int main(){
	char A[1000010],B[1000010];
	int tc;
	scanf("%d",&tc);
	while(scanf("%s",A)!=EOF){
		int t=strlen(A),i,j=0,last_fail=0;
		for(i=t-1;i>=0;i--){
			B[j++]=A[i];
		}
		B[j]='\0';
		int F[1000010];
		F[0]=F[1]=0;
		for(i=2;i<=t;i++){
			j=F[i-1];
			while(1){
				if(B[i-1]==B[j]){
					F[i]=j+1;
					break;
				}
				if(j==0){
					F[i]=0;
					break;
				}
				j=F[j];
			}
		}
		i=0;j=0;
		while(i<t){
			if(A[i]==B[j]){
				i++;
				j++;
			}
			else if(j>0){
				j=0;
			}
			else{
				i++;
			}
		}
		printf("%d\n",strlen(A)+t-j);
	}
	return 0;
}
