#include <bits/stdc++.h>
using namespace std;
long int gcd(long int a,long int b)
{
    if(b==0)
        return a;
    return gcd(b,a%b);
}
int main() {
	long int a[520000];
	long int n,k,i,ans=0,v,j;
	cin>>n;
	cin>>k;
	for(i=0;i<n;i++)
	cin>>a[i];
	for(i=0;i<n-ans;i++)
	{
		if(a[i]>=k)
		{
			v=1;
			
		}
		else
			continue;
		for(j=i+1;j<n;j++)
		{
			
			a[i]=gcd(a[i],a[j]);
			
			if(a[i]>=k)
			v++;
			else
			break;
		}
		ans=max(v,ans);
	}
	cout<<ans<<endl;
	
	
	return 0;
}
