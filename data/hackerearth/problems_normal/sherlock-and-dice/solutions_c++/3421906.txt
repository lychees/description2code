using namespace std;
typedef unsigned long long int ull;
typedef long long int ll;
typedef vector<vector<ll> > mat;
#define N 1000005

int n,m,k;
long double dp[55][10005];

long double solve(int times, int sum){
	if(times == 0){
		if(sum == 0){
			return 1.0;
		} else {
			return 0.0;
		}
	}
	if(dp[times][sum] != -1) return dp[times][sum];
	long double res = 0.0;
	forp(i,1,n){
		if(sum - i < 0) break;
		res  += (long double) 1.0/n*1.0 * solve(times - 1, sum - i); 
	}
	return dp[times][sum] = res;
}


int main(){
	boost;
	int t;
	cin >> t;
	while(t--){
		forp(i,0,54) forp(j,0,10004) dp[i][j] = -1;
		cin >> m >> n >> k;
		/*if(m > k){
			cout << "0.000 0\n";
			continue;
		}*/
		long double ans = solve(m, k); 
		if(ans == 0.000){
			cout << "0.000 0\n";
			continue;
		}
		int y = 0;
		if(ans < 1){
			while(ans < 1){
				ans *= 10.0;
				++y;
			}
		} 
	 	cout << fixed << setprecision(3) << ans << " ";
	 	cout << (int) y << "\n"; 
	}
 return 0;
}
