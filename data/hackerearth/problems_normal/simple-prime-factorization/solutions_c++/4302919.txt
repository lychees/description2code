void prime(int n)
{
	int x=0;
	int count=0;
	while(n%2==0)
	{
		x=1;
		n=n/2;
		count++;
	}
 
	cout<<"2^"<<count;
	long long int i=3;
	while(i*i<=n)
	{
		
		int c=0;
		while(n%i==0)
		{
			n=n/i;
			c++;
		}
		
		
		if(c>0)
		{x=1;
		cout << "*"<<i<<"^"<<c;
		}
		i++;
	}
	if(n>1&&x==1)
	cout << "*"<< n << "^"<<1 ;
	cout << endl;
}
