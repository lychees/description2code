#include <cstdio>
#include <cstdlib>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int ha,mi;
		scanf("%d%d",&ha,&mi);
		
    if(ha == 12)
    {
    	ha = 0;
	}
	if(mi == 00)
	{
		mi = 0;
	}
	int an1 = 0.5*(ha*60 + mi);
	int an2 = 6*mi;
	int angle = abs(an1 - an2);
	angle = min(360 - angle,angle);
	cout<<angle<<"\n";
}
    return 0;
}
