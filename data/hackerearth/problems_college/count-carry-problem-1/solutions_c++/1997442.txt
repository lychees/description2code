#include <iostream>
using namespace std;

int main()
{
	int tc,i;
	cin >> tc;
	int carry=0;
	while (tc--){
		string one,two;
		cin >> one >> two;
		carry = 0;
		for ( i=one.length()-1; i>=0;i-- ){
			int sum = int(one[i]) + int(two[i]) - 96;
			if ( sum >= 10){
				carry += 1;
				one[i-1] += sum -9;
			}
		}
		if(carry == 0)
			cout << "No carry operation" << endl;
		else if( carry == 1)
			cout << carry << " carry operation" << endl;
		else 
			cout << carry << " carry operations" << endl;
	}
    return 0;
}
