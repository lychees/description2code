#include<iostream>
#include<string>
using namespace std;
int main()
{
    int t,i,n,c;
    char ch;
    string str;
    cin >> t;
    while(t--)
    {
        cin >> str;
        n=str.length();
        c=0;
        for(i=0;i<n;i++)
        {
            ch=str[i];
            if(ch=='1')
                c+=2;
            else if(ch=='2')
                c+=5;
            else if(ch=='3')
                c+=5;
            else if(ch=='4')
                c+=4;
            else if(ch=='5')
                c+=5;
            else if(ch=='6')
                c+=6;
            else if(ch=='7')
                c+=3;
            else if(ch=='8')
                c+=7;
            else if(ch=='9')
                c+=6;
            else if(ch=='0')
                c+=6;
        }
        cout << c << endl;
    }
    return 0;
}
