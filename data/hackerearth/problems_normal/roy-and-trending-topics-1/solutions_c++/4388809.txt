#include <bits/stdc++.h>
typedef long long int ll;
using namespace std;
int main()
{
    //cout << "Hello World!" << endl;
    priority_queue<pair<ll,ll>>p;
    map<ll,ll>m;
    ll n,i,id,od,po,ns,d,l,c,s;
    cin>>n;
    for(i=0;i<n;i++) 
    {cin>>id;cin>>od>>po>>l>>c>>s;
      ns=po*50+l*5+c*10+s*20;
      d=ns-od;
      m[id]=ns;
      p.push(make_pair(d,id));
    }
    for(i=0;i<5;i++)
    {
    	cout<<p.top().second<<" "<<m[p.top().second]<<endl;
    	p.pop();
    }
    return 0;
}
