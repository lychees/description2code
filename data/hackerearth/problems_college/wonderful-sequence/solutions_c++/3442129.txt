#include <iostream>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
   long long int t,n,cnt=0;
   cin>>t;
   for(int h=0;h<t;h++)
   {
   	cin>>n;
   	long long int a[n];
   	for(int i=0;i<n;i++)
   	{
   		cin>>a[i];
   	}
   	for(int i=1;i<n;i++)
   	{
   		if(a[i]<=a[i-1])
   		cnt++;
   	}
   	cout<<cnt<<endl;
   	cnt=0;
   }
}
