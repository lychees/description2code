#include <iostream>
#include <stdio.h>
#include <vector>

using namespace std;

int main() {
	
	int n, k, l, i, count, dup;
	vector<int> A( 26, 0 );
	
	cin >> n >> k;
	dup = n;
	
	while( n-- ) {
		
		string s;
		
		cin >> s;
		l = s.length();
		vector<bool> B( 26, true );
		
		for( i = 0;i < l;i++ ) {
			if( B[ s[ i ] - 97 ] ) {
				A[ s[ i ] - 97 ]++;
				B[ s[ i ] - 97 ] = false;
			}
		}
				
	}
	
	count = 0;
	for( i = 0;i < 26;i++ )
		if( A[ i ] >= dup )
			count++;
				
	if( count >= k )
		printf(":-)\n");
	else
		printf(":-(\n");
    return 0;
}
