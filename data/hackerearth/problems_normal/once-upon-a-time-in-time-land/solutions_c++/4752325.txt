#include <iostream>
using namespace std;

int main()
{
   
	int test;
	cin>>test;
	while(test--){
		int n,k;
		cin>>n;
		cin>>k;
		long long int a[n];
		for(int i=0;i<n;i++)
		    cin>>a[i];
	    for(int i=k+1;i<n;i++)
	    {   if(a[i]<0)
	    	  a[i]=0;
	    	  long long int m=0;
	    	for(int j=i-k-1;j>=0 && j>=i-2*k-1; j--)
	    	{
	    	   if(a[j]>m)
	    	   m = a[j];
	    	}
	    	if(m==0)
	    	m=0;
	    	a[i] = a[i]+m;
	    }
	    long long int max=0;
	    for(int i=0;i<n;i++)
	    {   //cout<<a[i]<<" ";
	    	if(max<a[i])
	    	   max = a[i];
	    }    
	        cout<<max<<"\n";
	}
}
