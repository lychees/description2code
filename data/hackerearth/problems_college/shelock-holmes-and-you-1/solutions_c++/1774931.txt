void inline scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

int summ[5001],arr[5001],cum[5001],m,n;
lli dp[5001][5001];

lli func(int i, int j)
{
    //cout<<i<<" "<<j<<endl;
    if(dp[i][j]!=-1) return dp[i][j];
    if(j==0) return 0;
    if(i+m-1>n) return 0;
    if(i>n) return 0;
    lli x=func(i+1,j);
    lli y=func(i+m,j-1)+summ[i];
    dp[i][j]=max(x,max((lli)0,y));
    return dp[i][j];
}

int main()
{
    int k,i;
    scanint(n,m,k);
    cum[0]=0;
    memset(dp,(lli)-1,sizeof(dp));
    for(i=1;i<=n;i++)
    {
        scanint(arr[i]);cum[i]=cum[i-1]+arr[i];//cout<<cum[i]<<" ";
    }//line();
    for(i=1;i+m-1<=n;i++)
    {
        summ[i]=cum[i+m-1]-cum[i-1];//cout<<summ[i]<<" ";
    }//line();
    lli ans=func(1,k);
    cout<<ans<<endl;
}
