#include <bits/stdc++.h>
using namespace std;
int main(){
	int n,k;
	scanf("%d%d",&n,&k);
	vector<long long> a;
	int t;
	for(int i=0;i<n;i++){
		scanf("%d",&t);
		a.push_back(t);
	}
    sort(a.begin(), a.begin()+n);
	long long count=0;
	for(int i=0;i<n;i++){
		for (int j = i+1; j < n; ++j)
		{
            long long f = (long long)k / a[i];
            f = f/a[j];
			vector<long long>::iterator up;
            up = upper_bound(a.begin(), a.end(), f);
            if(up-a.begin()>j)
               count = count + (up - (a.begin()+j) - 1);
		}
	}
	printf("%d\n",count );
}
