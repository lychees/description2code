#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
int main(){
    char s[200001];
    cin>>s;
    int l=strlen(s);
    long sum = 0, mult = 1;
    for (int i = l; i > 0; i--, mult = (mult * 10 + 1)%mod){
        sum = sum%mod + ((s[i - 1] - '0') * (mult%mod) * i)%mod;
        sum%=mod;
    }
    cout<<sum<<endl;
}
