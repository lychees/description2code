#include <iostream>
#include <algorithm>
using namespace std;

int gcd(int a, int b) {
    return b == 0 ? a : gcd(b, a % b);
}

int main()
{
    int a; cin>>a;
    int n; cin>>n;
    int count = 0;
    int curr;
    for(int i=0;i<n;i++)
    {
    	cin>>curr;
    	if(gcd(a,curr)!=1) count+=1;
    }
    cout<<count<<endl;
}
