#define MOD 1000000007
#define pii pair<int,int>
bool isprime[2111111];
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int i,n=2000000;
	long long j,re=0;
	for(i=2;i<=n;i++)
		isprime[i]=true;
	for(i=2;i<=n;i++)
		if(isprime[i])
			for(j=(long long)i*i;j<=n;j+=i)
				isprime[j]=false;
	cin>>n;
	n--;
	for(i=2;i<=n;i++)
	{
		if(isprime[i])
			re+=i+1;
	}
	for(;i<=2*n;i++)
	{
		if(isprime[i])
			re+=2*n-i+1;
	}
	cout<<re%MOD<<endl;
	//system("pause");
	return 0;
}
