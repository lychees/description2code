#include<iostream>
#include<cstdio>
#include<cmath>
#include<vector>
#include<algorithm>
#include<cstring>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int x,y;
		scanf("%d%d",&x,&y);
		if(y==x)
		{
			if(x%2==0)
				printf("%d\n",(2*x));
			else
				printf("%d\n",((2*x)-1));
		}
		else if(y==(x-2))
		{
			if(x%2==0)
				printf("%d\n",((2*x)-2));
			else
				printf("%d\n",((2*x)-3));
		}
		else
			printf("No Number\n");
	}
	return 0;
}
