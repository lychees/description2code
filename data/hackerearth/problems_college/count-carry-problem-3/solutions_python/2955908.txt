cases = int(raw_input())

for case in xrange(cases):
	m, n = [int(i) for i in raw_input().split(' ')]
	carry = 0
	c_count = 0
	
	while m!=0 or n!=0:
		a = m%10
		b = n%10
		if a+b+carry>9:
			c_count += 1
			carry = 1
		else:
			carry = 0
		
		m /= 10
		n /= 10
	
	if c_count==0:
		print "No carry operation"
	elif c_count==1:
		print "1 carry operation"
	else:
		print str(c_count) + " carry operations"
