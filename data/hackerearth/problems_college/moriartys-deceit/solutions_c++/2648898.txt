#include <bits/stdc++.h>
#define ll long long
using namespace std;
int mygcd(int a,int b)
{
	while(1)
	{
		if(a==0)
		return b;
		if(b==0)
		return a;
		if(a>b)
		a=a%b;
		else
		b=b%a;
	}
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int test;
	cin>>test;
	while(test--)
	{
		int n,x;
		cin>>n>>x;
		int i,j;
		for(i=0;i<n;i++)
		{
			cin>>j;
			if(mygcd(j,x)==1)
			{
				cout<<j<<" ";
			}
		}
		cout<<endl;
	}
	return 0;
}
