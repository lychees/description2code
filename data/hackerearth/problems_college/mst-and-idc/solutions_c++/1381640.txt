#include<iostream>
using namespace std;
int main(){
	cout.sync_with_stdio(0);
	cin.tie(0);
	int t,n,i,a,minn,maxx,ans;
	cin>>t;
	while(t--){
		maxx=0;
		minn=101;
		cin>>n;
		for(i=0;i<n;++i){
			cin>>a;
			if(a>maxx)
				maxx=a;
			if(a<minn)
				minn=a;
		}
		ans=2*(maxx-minn);
		cout<<ans<<"\n";
	}
	return 0;
}
