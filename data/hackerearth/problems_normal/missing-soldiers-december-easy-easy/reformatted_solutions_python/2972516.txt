l = []
result = 0
n = input()
for _ in range(n):
    x, y, d = map(int, raw_input().split())
    l.append([x, x+d])
l.sort()

si, sj = l[0][0], l[0][1]
for i in range(1, n):
    if l[i][0] <= sj:
        if l[i][1] <= sj:
            continue
        else:
            sj = l[i][1]
    else:
        result += (sj-si+1)
        si, sj = l[i][0], l[i][1]

result += (sj-si+1)
print result
