#include<iostream>
#include<algorithm>
#include<vector>
#define MP make_pair
#define fi first
#define se second
using namespace std;
vector<pair<int, int> >v;
const int N = 1e6 + 1;
int k, i, t[N];
bool w[N];
string s;

int main()
{
  cin >> s >> k;
  
  for(i = 0; i < s.size(); i++)
  {
    t[s[i]]++;
  }
  
  for(i = 0; i < 256; i++)
  {
    if (t[i] > 0)
    {
      v.push_back(MP(t[i] * (-1), i));
    }
  }
  
  sort(v.begin(), v.end());
  
  while(!v.empty())
  {
    if (v.back().fi * (-1) <= k)
    {
      k += v.back().fi;
      w[v.back().se] = 1;
      v.pop_back();
    }
    else
    {
      break;
    }
  }
  
  cout << v.size() << endl;
  
  for(i = 0; i < s.size(); i++)
  {
    if (w[s[i]] == 0)
    {
      cout << s[i];
    }
  }
  cout << endl;  
  
return 0;
}
