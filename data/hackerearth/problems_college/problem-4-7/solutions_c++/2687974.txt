int rec (int index){

    if(index >= s.size()) return 0;

    int &st = dp[index];
    if(st != -1) return st;

    string w = "";
    w += s[index];
    int e = max(1 , p[ w ]  );

    int r = 0;
    if(index + 1 != s.size()){
        w += s[index + 1];
        r = max(1 , p[ w  ] );
    }

    return st = max( e + rec(index + 1), r + rec(index + 2)  );
}
