for _ in xrange(int(raw_input())):
	n,k=map(int,raw_input().split());
	a=map(int,raw_input().split());	a.sort();	i,j,min_v,max_v=0,len(a)-1,999999,0;
	while i<n/2:
		min_v=min(min_v,a[i]+a[j]);
		max_v=max(max_v,a[i]+a[j]);
		i+=1;	j-=1;
	if max_v-min_v==k:
		print max_v-min_v
		print "Lucky chap!"
	elif max_v-min_v<k:
		print max_v-min_v
		print "Chick magnet Jhool!"
	else:
		print max_v-min_v
