#include <bits/stdc++.h>
using namespace std;
template <typename T>
inline void fi(T *a)
{
    register char c=0;
    while (c<33) c=getchar_unlocked();
    *a=0;
    int tmp = 0;
    while (c>33)
    {
        if ( c == 45 ) tmp = 1;
        else *a=*a*10+c-'0';
        c=getchar_unlocked();
    }
    if ( tmp == 1 ) *a = 0-(*a);
}

int main()
{
    int t,n,i,j;
    fi(&t);
    while(t--)
   { fi(&n);
     vector<vector<int>> shop(n,vector<int>(3));
     for(i=0;i<n;i++)
     { for(j=0;j<3;j++)
        fi(&shop[i][j]); 
     }
      vector<vector<int>> Cost(n,vector<int>(3));
      for(j=0;j<3;j++)
      Cost[0][j]=shop[0][j];
      for(i=1;i<n;i++)
      { Cost[i][0]=min(Cost[i-1][1],Cost[i-1][2])+shop[i][0];
        Cost[i][1]=min(Cost[i-1][0],Cost[i-1][2])+shop[i][1];
        Cost[i][2]=min(Cost[i-1][0],Cost[i-1][1])+shop[i][2];
      }
      
      int ans=min(min(Cost[n-1][0],Cost[n-1][1]),Cost[n-1][2]);
      cout<<ans<<endl;
   }
      return 0;
   }
