#include <bits/stdc++.h>
using namespace std;
#define ll long long int
ll a,b,c,k;
ll solve(ll mid)
{
    return a*mid*mid+b*mid+c;
}
ll ans(ll l , ll h)
{
    while(l<=h)
    {
        ll mid = l+h;
        mid/=2;
        if(solve(mid)>=k)
        {
            h=mid-1;
        }
        else
        {
            l=mid+1;
        }
    }
    return l;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        cin>>a>>b>>c>>k;
        cout<<ans(0,100000)<<endl;
    }
    return 0;
}
