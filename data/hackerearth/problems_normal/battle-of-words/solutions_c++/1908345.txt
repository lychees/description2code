#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;

int main()
{	int t, l1, l2;
	int ct1[26], ct2[26];
	char str1[100003], str2[100003], ch;
	bool flag1,flag2;
	scanf("%d",&t);
	scanf("%c",&ch);
	while ( t-- > 0 )
	{	scanf("%[^\n]s",str1);
		scanf("%c",&ch);
		scanf("%[^\n]s",str2);
		scanf("%c",&ch);
		memset(ct1,0,sizeof(ct1));
		memset(ct2,0,sizeof(ct2)) ;
		l1=strlen(str1);
		l2=strlen(str2);
    		for( int i=0; i < l1; i++ )
			if( str1[i] >='a' && str1[i] <= 'z' )
				ct1[str1[i]-'a']++;
		for( int i=0; i< l2; i++ )
			if( str2[i] >= 'a' && str2[i] <= 'z' )
				ct2[str2[i]-'a']++;
		for( int i=0; i<26; i++ )
		{	int x = min(ct1[i],ct2[i]) ;
			ct1[i]-=x;
			ct2[i]-=x;
		}
		flag1=flag2=0;
		for( int i=0; i<26; i++ )
		{	if ( ct1[i] != 0  )
				flag1=1;
			if( ct2[i] != 0 )
				flag2=1;
		}
		if ( flag1 != 0 && flag2 != 0 )
			cout<<"You draw some."<<endl;
		else if ( flag1 != 0 )
               		cout<<"You win some."<<endl;
		else
			cout<<"You lose some."<<endl;

	}
    	return 0 ;
}
