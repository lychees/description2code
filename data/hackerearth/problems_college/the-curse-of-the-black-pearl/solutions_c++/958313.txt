#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t = 882;
	while(t--)
	{
		char a[33000]={0};
		scanf(" %s",a);
		int len = strlen(a);
		int i = len-1,j;
		int count = 0;
		int carry = 1;
		if(len == 1)
		{
			printf("0\n");
		}
		else
		{
			while(i > 0)
			{
				if(a[i] == '0')
				{
					count++;
	//				printf("1 %d\n",count);
				}
				else if(a[i] == '1')
				{
					count+=2;
	//				printf("2 %d\n",count);
					j = i;
					a[j] = '0';
					a[j-1]++;
					j--;
					while(a[j] > '1' && j >= 0)
					{
						a[j] = '1';
						if(j-1 >= 0)
							a[j-1]++;
						count++;
	//					printf("3 %d\n",count);
						i--;
						j--;
					}
					if(j == -1)
					{
						carry++;
								
					}
				}
				i--;
			}
			printf("%d\n",count );
		}
	}
	return 0;
}
