#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e5 + 9;

int tc;
string s;
int cnt = 0;

int main(){
	cin >> s;

	int len = (int)s.length();

	for(int l = 1;l <= len;++l){
		for(int idx = 0;idx <= len - l;++idx){
			string cur = s.substr(idx,l);
			sort(cur.begin(),cur.end());
			if(cur[0] == cur[(int)cur.length() - 1])
				++cnt;
		}
	}
	
	printf("%d\n",cnt);
	return 0;
}
