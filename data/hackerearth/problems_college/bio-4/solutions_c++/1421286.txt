#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <string>
#include <vector>
#include <map>
#include <memory.h>

using namespace std;

string s1, s2;

int main() {
    ios::sync_with_stdio(0);
    int t; cin >> t;
    while (t --> 0) {
        cin >> s1 >> s2;
        int ans = 1e9;
        for(int add = 0; add + s2.size() - 1 < s1.size(); ++add) {
            int tmp_ans = 0;
            for(int i = 0; i < s2.size(); ++i)
            if (s2[i] != s1[i+add])
                tmp_ans++;
            ans = min(ans, tmp_ans);
        }
        cout << ans << "\n";
    }
    return 0;
}
