#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll dp[1001][1001],mod=1000000007;
int main()
{
	for(int n=0;n<=1000;n++)
	 {
	 	dp[n][0]=1;dp[n][n]=1;
	 	for(int k=1;k<n;k++)
	 	{
	 	  dp[n][k]=(dp[n-1][k-1]%mod+dp[n-1][k]%mod)%mod;	
	 	}
	 }
	 int t;
	 scanf("%d",&t);
	 while(t--)
	{
		ll n,p;
		scanf("%lld%lld",&n,&p);
		ll ans=0;
		for(int i=0;i<=p;i++)
		 ans=(ans%mod +dp[n][i]%mod)%mod;
		 printf("%lld\n",ans);
	}
	return 0;
}
