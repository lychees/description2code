#include <bits/stdc++.h>

using namespace std;

#define fi first
#define se second

int n,m;
vector<pair<int,int> > e[100001];
int seen[100001];
int dis[100001];
int low[100001];
int b[100001];
int tim;

void dfs(int v, int p)
{
	seen[v]=1;
	dis[v]=++tim;
	low[v]=tim;
	for(vector<pair<int,int> >::iterator it=e[v].begin(); it!=e[v].end(); it++)
	{
		if(!seen[it->fi])
		{
			dfs(it->fi,v);
			low[v]=min(low[v],low[it->fi]);
			if(dis[v]<low[it->fi]) b[it->se]=1; 
		}
		else if((it->fi)!=p) low[v]=min(low[v], dis[it->fi]);
	}
}

int main()
{
	scanf("%d%d", &n, &m);
	for(int i=1; i<=m; i++)
	{
		int a,b;
		scanf("%d%d", &a, &b);
		e[a].push_back(make_pair(b,i));
		e[b].push_back(make_pair(a,i));
	}
	for(int i=1; i<=n; i++)
	{
		if(!seen[i]) dfs(i,0);
	}
	int q; cin>>q;
	while(q--)
	{
		int k; cin>>k;
		if(b[k]) cout<<"Unhappy"<<endl;
		else cout<<"Happy"<<endl;
	}
}
