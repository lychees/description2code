#include <iostream>
#include <vector>

using namespace std;

bool flag;
int sum, n;
vector<int>box(100 + 5);
vector<vector<int>>dp(100 + 5, vector<int>(10000 + 5));

void open(int index, int total)
{
	if (index == n)
	{
		if (total == sum / 2) flag = true;
		return;
	}

	if (dp[index][total])return;

	open(index + 1, total + box[index]);
	open(index + 1, total);
	dp[index][total] = true;
}
auto main()->int
{
	int t;
	cin >> t;
	while (t--)
	{
		cin >> n;
		sum = 0;
		flag = false;
		for (auto i = 0; i<n; i++)
		{
			cin >> box[i];
			sum += box[i];
		}
		vector<vector<int>>dpp(100 + 5, vector<int>(10000 + 5, false));
		dp = dpp;

		if (sum % 2 != 0)
		{
			cout << "NO" << endl;
		}
		else
		{
			open(0, 0);
			if (flag)
			{
				cout << "YES" << endl;
			}
			else
			{
				cout << "NO" << endl;
			}
		}
	}
}
