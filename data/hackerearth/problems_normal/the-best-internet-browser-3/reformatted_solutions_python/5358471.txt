tc = raw_input()


def countvowels(string):
    num_vowels = 0
    for char in string:
        if char in "aeiouAEIOU":
            num_vowels = num_vowels+1

    return num_vowels


for i in range(int(tc)):
    web = raw_input()
    i = 0
    total = len(web)
    i = countvowels(web)
    print "%d/%d" % (total-i-3, total)
