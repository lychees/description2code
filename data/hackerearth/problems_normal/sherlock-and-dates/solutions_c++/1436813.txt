#include <bits/stdc++.h>
using namespace std;

#define debug(args...) fprintf(stderr,args)

typedef long long lint;
typedef pair<int,int> pii;
typedef pair<lint,lint> pll;

const int INF = 0x3f3f3f3f;
const lint LINF = 0x3f3f3f3f3f3f3f3fll;

bool before(int d1,int m1,int y1, int d2,int m2,int y2) {
    if(y1 < y2) return 1;
    if(y1 > y2) return 0;
    if(m1 < m2) return 1;
    if(m1 > m2) return 0;
    return d1 <= d2;
}

int main() {
    int t;
    for(scanf("%d",&t);t--;) {
        int d1,m1,y1,d2,m2,y2;
        scanf("%d:%d:%d %d:%d:%d",&d1,&m1,&y1,&d2,&m2,&y2);
        int ans = 0;
        int mm1 = y1 % 100 - 1;
        if(2 <= mm1 && mm1 <= 12) {
            if(before(d1,m1,y1, mm1-1, mm1, y1) &&
               before(mm1-1,mm1,y1, d2,m2,y2))
                ans++;
        }
        if(y1 < y2) {
            int mm2 = y2 % 100 - 1;
            if(2 <= mm2 && mm2 <= 12) {
                if(before(d1,m1,y1, mm2-1, mm2, y2) &&
                   before(mm2-1,mm2,y2, d2,m2,y2))
                    ans++;
            }
        }
        for(int year = y1+1; year < y2; ++year) {
            int month = year % 100 - 1;
            if(2 <= month && month <= 12) ans++;
        }
        printf("%d\n",ans);
    }        
    return 0;
}
