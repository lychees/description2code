#include <iostream>
using namespace std;

int main()
{
	long long int t;
    cin>>t;
    while(t--)
    {
    	long long int n,i,sum=1;
    	cin>>n;
    	if(n==1)
    		cout<<"0"<<endl;
    	else if(n==2)
    		cout<<"2"<<endl;
    	else 
    	{	for(i=2;i<n;i++)
    		sum=sum+(i*(i+1))/2;
    		cout<<2*(sum)<<endl; }
    }
}
