#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
    int T,n;
    cin >> T;
    while(T--)
    {
        cin >> n;
        int arr[n];
        for(int i=0;i<n;i++)
        cin >> arr[i];
        int flag=0,count=0,k=1;
        while(1)
        {
             sort(arr,arr+n);  
            for(int i=1;i<=k;i++)
            {
                if(arr[n-i]>0)
               	arr[n-i]=arr[n-i]-1;
                else
                {
                    flag=1;
                    break;
                }
            }
            count++;
            k++;
            if(flag==1)
            break;
        }
        count--;
        cout << count << endl;
    }
}
