#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s,s1;
    	cin>>s;
    	int i,j;
    	j = s.length();
    	vector <int> vec(26,0);
    	for(i=0;i<j;i++)
    	{
    		if(vec[(int)s[i]-97]==0)
    		{
    			s1.push_back(s[i]);
    			vec[(int)s[i]-97]=1;
    		}
    	}
    	cout<<s1<<endl;
    }
    return 0;
}
