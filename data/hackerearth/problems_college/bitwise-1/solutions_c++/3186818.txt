#include <bits/stdc++.h>
#define lli long long int
#define s(x) scanf("%lld", &x)
#define pb push_back
#define mp make_pair
#define fr first
#define sc second

using namespace std;


int main()
{
    lli i,j,k,n,no,m,p,q,r,ans,s,tt,tcase;

    s(tcase);
    tt = 1;

    while (tt <= tcase) {
    	printf("Case #%lld: ", tt);
    	++tt;

    	s(n);
    	s(k);

    	no = 1;
    	ans = 0;
    	j = 1;

    	while(1) {
    		if (j > k)
    			break;

           // cout << no << " " << n << " " << (lli)(no&n) << endl;
    		if ((lli)(no&n) == 0) {
    			if (j&k) {
             //       cout << no << endl;
    				ans = ans + no;
    				j = j*2;
    			} else {
    				j = j*2;
    			}
    		}
    		no = no*2;
    	}

    	printf("%lld\n", ans);
    }
    return 0;
}
