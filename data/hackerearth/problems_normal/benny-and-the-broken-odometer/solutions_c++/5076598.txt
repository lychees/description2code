int pw[10];
int fn(int n)
{
    int i=0;
    int ans=0;
    while(n)
    {
        int x=n%10;
        if(x>3)
            x--;
        ans+=pw[i]*x;
        i++;
        n/=10;
    }
    return ans;
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int t,n,i;
    pw[0]=1;
    fr(i,1,10)
        pw[i]=pw[i-1]*9;
    cin>>t;
    while(t--)
    {
        cin>>n;
        cout<<fn(n)<<endl;
    }
    return 0;
}
