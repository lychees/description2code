#include <iostream>
using namespace std;
int arr[1000000];
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    arr[0]=arr[1]=1;
    for(int i=2;i<100000;i++)
    {
    	if(arr[i]==0)
    	{
    		for(int j=i+i;j<1000000;j=j+i)
    		arr[j]=1;
    	}
    }
    while(t--)
    {
    	int n,sum=0;
    	cin>>n;
    	while(n)
    	{
    		sum+=n%10;
    		n=n/10;
    	}
    	
if(arr[sum]==0) cout<<"YES"<<endl;
else cout<<"NO"<<endl;
    }
    return 0;
}
