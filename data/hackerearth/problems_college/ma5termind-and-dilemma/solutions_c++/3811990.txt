#include <bits/stdc++.h>
using namespace std;
char s1[100001];
char s2[100001];
char s3[100001];
int main()
{
    int t,l,c,d,i,c1,c2;
    cin>>t;
    while(t--)
    {
    	c=0;
    	c1=0;c2=0;
    	cin>>l;
       cin>>s1;
       for(i=0;i<l;i++)
       if(s1[i]=='1')
       c++;
       d=l-c;
       for(i=0;i<c;i++)
       s2[i]='1';
       for(i=c;i<l;i++)
       s2[i]='0';
       for(i=0;i<d;i++)
       s3[i]='0';
       for(i=d;i<l;i++)
       s3[i]='1';
       for(i=0;i<l;i++)
       if(s1[i]!=s2[i])
       c1++;
       for(i=0;i<l;i++)
       if(s1[i]!=s3[i])
       c2++;
       if(c1>=c2)
       cout<<c2/2<<"\n";
       else
       cout<<c1/2<<"\n";
       
    }
    return 0;
}
