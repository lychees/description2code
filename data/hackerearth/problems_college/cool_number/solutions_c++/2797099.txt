#include <iostream>
using namespace std;

int isCool(int a, int k)
{
	int s1=1,s2=0,s3=0,count=0;
	int temp = a;
	int c;
	while(a)
	{
		c = a%2;
		if(s1 && c)
		{
			
			//cout<<"in state 0 going to 1"<<endl;
			s1 = 0;
			s2 = 1;
		}
		else if(s2 && !c)
		{
			//cout<<"in state 0 going to 1"<<endl;
			s3 = 1;
			s2 = 0;
		}
		else if(s3 && c)
		{
			s2 = 1;
			s3 = 0;
			count++;
		}
		else if(s3 && !c)
		{
			s1 = 1;
			s3 = 0;
		}
		a/=2;
	}
	//cout<<"no. of patterns for "<<temp<<" is "<<count<<endl;
	if(count>=k)	return 1;
	else return 0;
}

int main()
{
    //cout << "Hello World!" << endl;
    int t,r,k;
    cin>>t;
    int count = 0;
    //isCool(9,1);
    for(int i=0;i<t;i++)
    {
    	cin>>r>>k;
    	count = 0;
    	for(int j=1;j<=r;j++)
    	{
    		if(isCool(j,k))	count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
