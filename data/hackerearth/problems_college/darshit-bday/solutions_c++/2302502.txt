#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long t,n,i,m,a,b,c;
	cin>>t;
	while(t--)
    {
        long long cost=0;
        cin>>n>>m;
        cin>>a>>b>>c;
        if(a+c<b)
        {
            cost=n*a+m*(a+c);
        }
        else if(b+c<a)
        {
            cost=m*b+n*(b+c);
        }
        else
        {
            cost=n*a+m*b;
        }
        cout<<cost<<endl;
    }
	return 0;
}
