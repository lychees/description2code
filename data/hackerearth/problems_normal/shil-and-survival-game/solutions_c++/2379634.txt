#include<iostream>
#include<stdlib.h>
using namespace std;
int win(unsigned A[],unsigned left,unsigned right,unsigned val)
{
  unsigned mid=(left+right)/2;
     if(right-left<=0){
      if(A[left]>val)
         return 0;
         return 1;  }
  return (win(A,left,mid,val)&&win(A,mid+1,right,val));
}
int main()
{
   unsigned *A,i,N;
   cin>>N;
   A=(unsigned *)malloc(sizeof(unsigned)*N);
   for(i=0;i<N;i++)
    cin>>A[i];
   for(i=0;i<N;i++){
      if(i==0||i==N-1)
        cout<<i+1<<" ";
      else
        {
         if(win(A,0,i-1,A[i])||win(A,i+1,N-1,A[i]))
           cout<<i+1<<" "; } 
     }
 return 0;
}
