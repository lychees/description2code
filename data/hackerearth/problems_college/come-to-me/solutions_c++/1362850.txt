int main()
{int T;
cin>>T;
while(T--)
{string name;
cin>>name;
    int n=name.size();
    int maximum=0;
    int a[26]={0};
    int j=0;
    
    for(int i=0;i<n;i++)
    {
       if(a[name[i]-'A']==0)
       {
       	  a[name[i]-'A']=1;
       	
       }
       else
       {  
       	 maximum=max(maximum,(i-j));
       	 while(name[i]!=name[j])
       	 { a[name[j]-'A']=0;
       	 	
       	 	j++;
       	 }
       	 j++;
       }}
       maximum=max(maximum,(n-j));
       	 
       cout<<maximum<<endl;
    	
    	
    	
    	
    }
}
