#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    int t;
    long long int n,i,j,k,a[100000],c;
    cin>>t;
    for(i=1;i<=t;i++)
    {
    	c=0;
    	cin>>n;
    	for(j=1;j<=n;j++)
    	cin>>a[j];
    	sort(a+1,a+n+1);
    	
    	for(j=1;j<=n;j++)
    	{
    		
    		
    		if(a[j]==a[j+1])
    		{
    			c++;
    		
    		}
    	}
    	cout<<c<<"\n";
    }
    
    
    return 0;
}
