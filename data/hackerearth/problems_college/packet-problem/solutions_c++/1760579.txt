#include <iostream>
using namespace std;

int main()
{	
	int t,var;
    cin>>t;
    while(t--)
    {	
    	long int zerocount=0,onecount=1,temp=0;
    	cin>>var;
    	while(var--)
    	{
    	temp=zerocount;
    	zerocount+=onecount;
    	onecount=temp;  
    	}
    	cout<<onecount<<" "<<zerocount<<"\n";
    	
    }
    return 0;
}
