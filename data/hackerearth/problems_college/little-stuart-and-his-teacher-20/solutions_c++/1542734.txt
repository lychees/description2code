using namespace std;
lli dp[1000001];

int coPrimes( int n )
{
    int i,ans;
    ans=n;
    for(i=2;i*i<=n;i++)
    {
        if(n%i==0) ans-=ans/i;
        while(n%i==0) n/=i;
    }
    if(n!=1) ans-=ans/n;
    return ans;
}

void precompute( int n)
{
    lli cnt=0;
    dp[1]=0;
    for(int i=2;i<=n;i++)
    {
        int val=coPrimes(i);
        cnt+=(i-1-val);
        dp[i]=cnt;
    }
}

int main()
{
    int t,n;
    precompute(1000000);
    scanf("%d",&t);
    //printf("%d %d %d\n",coPrimes(2),coPrimes(3),coPrimes(4));
    while(t--)
    {
        scanf("%d",&n);
        printf("%lld\n",dp[n]);
    }
    return 0;
}
