#include<bits/stdc++.h>
using namespace std;
#define ll long long 
ll  cnt0=0;
ll  cnt1=0;
ll fib(ll n)
{
	if(n==0)
	{
		cnt0++;
		return 0;
	}
	if(n==1)
	{
		cnt1++;
		return 1;
	}
	return fib(n-1)+fib(n-2);
}
int main()
{

	ll  tc;
	cin>>tc;
	while(tc--)
	{
		ll  n;
		cin>>n;
		cnt0=0;
		cnt1=0;
		fib(n);
		printf("%lld %lld\n",cnt0,cnt1);
	}

	return 0;
}
