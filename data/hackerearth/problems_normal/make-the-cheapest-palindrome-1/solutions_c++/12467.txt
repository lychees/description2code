#include <iostream>
#include <string>

using namespace std;

int main()
{
	int testcase;

	cin >> testcase;

	string strpalindrome;
	int costa, costb, min;
	int sum;

	string::iterator bitr, eitr;
	while (testcase--)
	{
		cin >> strpalindrome >> costa >> costb;
		min = costa < costb ? costa : costb;
		sum = 0;

		bitr = strpalindrome.begin();
		eitr = strpalindrome.end() - 1;

		while (bitr < eitr)
		{
			if (*bitr != *eitr)
			{
				if ((*bitr == '/') && (*eitr == 'a'))
					sum += costa;
				
				else if ((*bitr == '/') && (*eitr == 'b'))
					sum += costb;
				
				else if ((*bitr == 'a') && (*eitr == '/'))
					sum += costa;

				else if ((*bitr == 'b') && (*eitr == '/'))
					sum += costb;
				else
				{
					sum = -1;
					break;
				}
			}
			else if ((*bitr == '/') && (*eitr == '/'))
			{
				sum += min;
				sum += min;
			}

			++bitr; --eitr;
		}

		cout << sum << endl;
	}
}
