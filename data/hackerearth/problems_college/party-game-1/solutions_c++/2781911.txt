#include <iostream>
#include<algorithm>
using namespace std;

int main()
{

    long long t, arr[10005], i,j;
    long long chef, jhool,n;
    cin>>t;
    
    while(t--){
    	cin >> chef >> jhool;
    	cin>>n;
    	
    	for(i=0;i<n;i++)
    		cin>> arr[i];
    	sort(arr, arr+n);
    long long sum1=0, sum2=0,d1=0,d2=0;
    	for(i=0;i<n;i++){
    		if(sum1+arr[i]<=chef){
    			sum1+=arr[i];
    			d1++;
    		}
    		
    	}
    	i=0;
    	while(sum2+arr[i]<=jhool&& i<n)
    	{
    		
    		d2++;
    		sum2+=arr[i];
    		i++;
    	}
    	if(d1>d2)
    		cout<<"Chef Won\n";
    	else if(d2>d1)
    		cout<< "Jhool Won\n";
    	else
    		cout<< "Tie\n";
    	
    }
    return 0;
}
