#include<bits/stdc++.h>
using namespace std;

int main()
{
	int m;
	cin>>m;
	int diag = 0,antidiag = 0;
	for(int i = 0; i < m; i++)
	{
		for(int j =0; j < m; j++)
		{
			int temp;
			cin>>temp;
			if(i==j)
			diag = diag + temp;
			if(i + j == m -1)
			antidiag = antidiag + temp;
		}
	}
//	cout<<diag<<" "<<antidiag;
	int r = abs(diag  - antidiag);
	cout<<r<<endl;
}
