#define ALL(a) a.begin(), a.end()
#define FOR(a) for (int i = 0; i < (a); ++i)
#define D(a) #a << ": " << a << "\n"

int64 gcd(int64 a, int64 b)
{
    if (!b) return a;
    return gcd(b, a % b);
}

int main()
{
    int t;
    cin >> t;
    while (t--)
    {
        int64 a, b, n;
        scanf("%lld %lld %lld", &a, &b, &n);
        int64 start = 0, end = 1LL * 1000 * 1000000000 + 5;
        int64 both = a*b / gcd(a, b);
        while (start <= end)
        {
            int64 mid = (start + end) / 2;
            int64 num = mid / a + mid / b - mid / both;
            if (num == n)
            {
                start = mid;
                break;
            }
            else if (n > num)
                start = mid + 1;
            else
                end = mid - 1;
        }

        start = max(start/a*a, start/b*b);
        printf("%lld\n", start);
    }
    return 0;
}
