	for(i=0;i<101;i++)

		count=0;
		prev=0;
		//cout<<i<<"-> ";
		
		for(std::vector<int>::iterator it=vec[i].begin();it!=vec[i].end();it++)
			{
				//cout<<*it<<" ";
				for(j=prev;j<*it ;j++)
				{
					arr[i][j]=count;
				}
				prev=*it;
				count++;
			}
			for(j=prev;j<N ;j++)
			{
					arr[i][j]=count;
			}
			//cout<<endl;
	}
