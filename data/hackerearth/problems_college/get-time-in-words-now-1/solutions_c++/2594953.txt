int main()
{
	int h,m,t,t1;
	si(h);
	si(m);
	char *a[12]={ "one", "two", "three", "four","five", "six", "seven", "eight", "nine","ten","eleven","tweleve"};
	char *b[9]={"eleven", "tweleve","thirteen","fourteen","fifteen","sixteen","seventeen","eighteen","nineteen"};
	
	if(m==00)
		pf("%s o' clock",a[h-1]);
	else if(m<30 && m!=15)
	{
		t=m%10;
		m/=10;
		t1=m%10;
		if(t1==0)
		{
			if(t==1)
				pf("one minutes past ");
			else
				pf("%s minutes past ",a[t-1]);
		}
		else
		{
			if(t1>=2)
			{
				if(t==0)
					pf("twenty minutes past ");
				else
					pf("twenty %s minutes past ",a[t-1]);
			}
			else
				pf("%s minutes past ",b[t-1]);
		}
		pf("%s",a[h-1]);
	}
	else if(m==30)
		pf("half past %s ",a[h-1]);
	else if(m==45)
		pf("quarter to %s ",a[h]);
	else if(m==15)
		pf("quarter past %s ",a[h-1]);
	else
	{	
			m=60-m;
			t=m%10;
			m/=10;
			t1=m%10;
			if(t1>=2)
			{
				if(t==0)
					pf("twenty minutes to ");
				else
					pf("twenty %s minutes to ",a[t-1]);
			}
			else
				pf("%s minutes to ",b[t-1]);
			pf("%s",a[h]);
		
	}
	

	return 0;
}
