int minmove(char s[]){
    int len = strlen(s);
    int ret=0,l=0,idx=1;
    while(ret+l+1<len && idx<len){
        //cerr<<ret<<" "<<idx<<" "<<l<<"\n";
        int cp=s[ret+l]-s[(idx+l)%len];
        if(!cp) l++;
        else if(cp>0) idx+=l+1,l=0;
        else ret=max(ret+l+1,idx), idx=ret+1, l=0;
    }
    return ret;
}
