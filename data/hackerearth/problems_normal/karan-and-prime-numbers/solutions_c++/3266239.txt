#include<iostream>
#include<cstdlib>
#include<bitset>
#include<vector>
#include<algorithm>
using namespace std;
bitset <10000010> bs;
void sieve(long long int upperbound){
	long long int sieve_size=upperbound+1;
	bs.set();
	bs[0]=bs[1]=0;//0 and 1 are not prime
	for(long long int i=2;i<=sieve_size;i++){
		if(bs[i]){
			for(long long int j=i*i;j<=sieve_size;j+=i)bs[j]=0;
		}
	}
}
int main(void){
	sieve(10000000);
	int tests;
	cin>>tests;
	while(tests--){
		long long int a,b,sum=0;
		cin>>a>>b;
		for(long long int i=a;i<=b;i++)
			if(bs[i]==1)sum+=i;
		cout<<sum<<endl;
	}
	return 0;
}
