#include <iostream>
using namespace std;

int main()
{
	int n,t,posX,posY,i,j,mid,steps;
	string s[51];
	cin>>t;
	while(t--)
	{
		steps = 0;
		cin>>n;
		
		for(i=0;i<n;i++)
			cin>>s[i];
		
		mid = n/2;
		
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				if(s[i][j] == '*')
				{
					posX = i;
					posY = j;
					break;
				}
			}
		}
		
		if(posX>mid)
			posX=posX-mid;
		else
			posX=mid-posX;
		
		
		if(posY>mid)
			posY=posY-mid;
		else
			posY=mid-posY;
		
		steps = posX +posY;
		cout<<steps<<endl;
	}
	
    return 0;
}
