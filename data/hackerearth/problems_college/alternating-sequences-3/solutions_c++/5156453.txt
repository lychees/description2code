{

    		{
    			if(abs(a[i])>abs(a[j]))
    			t[i]=t[j]+1;
    			
    		}

    		{
    			if(abs(a[i])>abs(a[j]))
    			t[i]=t[j]+1;
    			
    		}
    	}
    	
    }
    for(i=0;i<n;i++)
    if(max<t[i])
    max=t[i];

}
