#include<bits/stdc++.h>
using namespace std;
long long int dp[100005];
void getMinSquares()
{
    dp[0] = 0;
    dp[1] = 1;
    dp[2] = 2;
    dp[3] = 3;
    for (int i = 4; i <= 100002; i++)
    {
        dp[i] = i;
        for (long long int x = 1; x <= i; x++) {
            long long int temp = x*x;
            if (temp > i)
                break;
            else dp[i] = min(dp[i], 1+dp[i-temp]);
        }
    }
 

}
 
// Driver program
int main()
{
    getMinSquares();
    int n;
    int k=1;
    scanf("%d",&n);
    
    	printf("%lld\n",dp[n]);
    
    return 0;
}
