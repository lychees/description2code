#include<bits/stdc++.h>
using namespace std;
int A[10000][10000],P[10000][10000];
int main()
{
    int T;
    int N;

    cin>>T;
    while(T--)
    {
        cin>>N;
        for(int i=0;i<N;i++)
        {
            for(int j=0;j<N;j++)
                cin>>A[i][j];
        }
        P[0][0]=1;
        for(int i=1;i<N;i++)
            P[i][0]=P[i-1][0]&&A[i][0];
        for(int j=1;j<N;j++)
            P[0][j]=P[0][j-1]&&A[0][j-1];
        for(int i=1;i<N;i++)
        {
            for(int j=1;j<N;j++)
            {
                P[i][j]=(P[i-1][j]||P[i][j-1])&&A[i][j];
            }
        }
        if(P[N-1][N-1]==1)
            cout<<"POSSIBLE\n";
        else
            cout<<"NOT POSSIBLE\n";

    }

    return 0;
}
