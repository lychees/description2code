#include <cstdio>
#include <cstring>
using namespace std;
int gcd(int a,int b)
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}
int main()
{
    int t,len,l,r;
    char s[100000];
    scanf("%d",&t);
    while(t--)
    {
      scanf("%s",s);
      len=strlen(s);
      l=1;
      int ans=s[0]-48;
      while(l<len)
      {
          int x=s[l]-48;
          ans=gcd(x,ans);
          l++;
      }
      for(int i=0;i<len;i++)
         printf("%d",(s[i]-48)/ans);
       printf("\n");
    }
    return 0;
}
