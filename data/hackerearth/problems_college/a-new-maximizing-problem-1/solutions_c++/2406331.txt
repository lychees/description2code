#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
ll mode(ll val)
{
	while(val<0) val+=mod;
	val%=mod;
	return val;
}
int main()
{
    ll t,n,arr[100004];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	ll ans=0;
    	ll sum=0;
    	for(int i=0;i<n;i++) 
    	{
    		cin>>arr[i];
    		if(arr[i]<0) sum+=-1*arr[i];
    		else sum+=arr[i];
    	}
    	sort(arr,arr+n);
    	for(int i=0;i<n/2;i++) 
    	{
    		ans+=-1*arr[i]+arr[n-1-i];
    	}
    	if(n%2) ans+=-1*arr[(n-1)/2];
    	cout<<mode(mode(ans)*mode(sum))<<endl;
    }
    return 0;
}
