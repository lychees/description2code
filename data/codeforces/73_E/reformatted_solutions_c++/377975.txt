#include <algorithm>
#include <cmath>
#include <fstream>
#include <iostream>
#include <map>
#include <queue>
#include <set>
#include <sstream>
#include <stack>
#include <stdio.h>
#include <string.h>
#include <string>
#include <vector>
using namespace std;
#define li long long
#define rep(i, to) for (li i = 0; i < ((li)to); i++)
#define pb push_back
#define sz(v) ((li)v.size())

int main() {
  int n, x;
#define MAX_N (1 << 21)
  bool ok[MAX_N];
  rep(i, MAX_N) ok[i] = false;
  scanf("%d%d", &n, &x);
  set<int> s;
  rep(i, n) {
    int tmp;
    scanf("%d", &tmp);
    s.insert(tmp);
  }
  if (x == 2) {
    printf("0\n");
    return 0;
  }
  if (s.count(1) > 0) {
    printf("1\n");
    return 0;
  }
  int res = 0;
  for (int i = 2; i < x; i++)
    if (!ok[i]) {
      if (s.count(i) == 0) {
        printf("-1\n");
        return 0;
      }
      res++;
      for (int j = i; j < MAX_N; j += i)
        ok[j] = true;
    }
  printf("%d\n", res);
}
