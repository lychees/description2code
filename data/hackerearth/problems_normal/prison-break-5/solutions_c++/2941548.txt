#include<iostream>
#include<stdlib.h>
#include<vector>
#include<limits.h>
#include<queue>
using namespace std;
typedef long long ll;
ll tc,i,j,count,n;
int adj[20][20];
bool vst[20][20];
void init(int n)
{
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            vst[i][j]=false;
}
void dfs(int i, int j)
{
    if(i==n-1 && j==n-1)
        count++;
    vst[i][j]=true;
    if(i+1<n && adj[i+1][j]==0 && vst[i+1][j]==false)
        dfs(i+1,j);
    if(j+1<n && adj[i][j+1]==0 && vst[i][j+1]==false)
        dfs(i,j+1);
    if(i-1>-1 && adj[i-1][j]==0 && vst[i-1][j]==false)
        dfs(i-1,j);
    if(j-1>-1 && adj[i][j-1]==0 && vst[i][j-1]==false)
        dfs(i,j-1);
    vst[i][j]=false;
}
int main()
{
    cin>>tc;
    while(tc--)
    {
        cin>>n;
        count=0;
        for(i=0;i<n;i++)
            for(j=0;j<n;j++)
                cin>>adj[i][j];
        dfs(0,0);
        cout<<count<<endl;
    }
}
