int ans[1500][1500];
int arr[1500];
int a[1500];
int cnt=1;
int select(int x){
	if(cnt==1){
		a[cnt]=x;
		return x;
	}
	int i;
	for(i=cnt-1;i>=1;i--){
		if(a[i]>x){
			a[i+1]=a[i];
		}
		else{
			break;
		}
	}
	a[i+1]=x;
	int ret=ceil(cnt/2.0);
	return a[ret];
}
int main(){
	int n,q,i,j;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	scanf("%d",&arr[i]);
	for(i=1;i<=n;i++){
	//	cout<<i<<"   ";
		for(j=i;j<=n;j++){
			ans[i][j]=select(arr[j]);
		//	cout<<ans[i][j]<<" ";
			cnt++;
		}
		//cout<<endl;
		cnt=1;
	}
	scanf("%d",&q);
	while(q--){
		scanf("%d%d",&i,&j);
		printf("%d\n",ans[i][j]);
	}
	return 0;
}
