#include <bits/stdc++.h>

using namespace std;
int N;
struct points
{
	int x,y,r;
	void read(){ 
		scanf("%d%d%d",&x,&y,&r);
	}
};

points p[11];

bool taDentro(int x, int y)
{
	int c = 0;
	for(int i = 0 ; i < N; i++)
	{
		int px = p[i].x;
		int py = p[i].y;
		int r = p[i].r;
		if( sqrt( (px-x)*(px-x) + (py-y)*(py-y)) <= r) c++;
		if(c == 2) return true;
			
	}
	return false;
	
}

int main()
{
	
	

	scanf("%d",&N);
	for(int i = 0 ; i < N; i++)
	{
		p[i].read();
	}
	
	int ans = 0;
	for(int i = 1 ; i <= 1000; i++)
		for(int j = 1 ; j <= 1000; j++)
		{
			
			if(taDentro(i,j)) 
				ans++;
			
		}	
	printf("%d\n",ans);
	
		
	
	
	return 0;
}
