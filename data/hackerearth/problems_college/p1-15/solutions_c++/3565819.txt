#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	string a,b;
	cin >>t;
	while(t--)
	{
	
		cin >> a>> b;
		int l1,l2,s=-1,i;
		l1=a.length();
		l2=b.length();
		for(i=0;i<l1;i++)
		{
			if(a[i]==b[0])
			{
				if(l2>=(l1-i)&&a.substr(i,(l1-i))==b.substr(0,(l1-i)))
				{
				s=i;
				break;
				}
			}
		}
		if(s!=-1)
		{
			cout << a.substr(0,s);
			cout<<b<<"\n";
        }
		else
		cout << a << b<<"\n";
	}
	return 0;
}
