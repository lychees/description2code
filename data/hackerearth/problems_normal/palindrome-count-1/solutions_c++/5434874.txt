//constants
const int mxn=1e5+1;

using namespace std;
int main()
{
	int t=1;
	sc1(t);
	while(t--)
	{
		string s;
		cin>>s;
		int count=0;
		for(int i=0;i<s.length();++i)
		{
			int up,low;
			up=i+1;
			low=i-1;
			while(up<s.length() && low>=0 && s[low]==s[up])
			{
				count++;
				up++;
				low--;
			}
			up=i+1;
			low=i;
			while(up<s.length() && low>=0 && s[low]==s[up])
			{
				count++;
				up++;
				low--;
			}

		}	
		cout<<count+s.length()<<endl;	
	}
	return 0;
}
