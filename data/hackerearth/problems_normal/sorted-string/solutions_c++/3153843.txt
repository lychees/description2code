    #include<bits/stdc++.h>
    using namespace std;
    int main()
    {
    int t;
    cin>>t;
    while(t--)
    {
    string str;
    cin>>str;
    pair<char,int>p[100000];
    int i,len = str.length();
    int count[26]={0},max=0;
    for(i=0;i<len;i++)
    {
    count[str[i]-'a']++;
    if(count[str[i]-'a']>max)
    {
    max = count[str[i]-'a'];
    }
    }
    //cout << max << endl;
    int k=0;
    for(i=0;i<26;i++)
    {
    if(count[i]!=0)
    {
    p[k] = make_pair((char)(97+i),count[i]);
    k++;
    }
    }
    sort(p,p+k);
    for(i=1;i<=max;i++)
    {
    for(int j=0;j<k;j++)
    {
    if(p[j].second == i)
    {
    for(int l =1;l<=i;l++)
    {
    cout<<p[j].first;
    }
    }
    }
    }
    cout<<"\n";
    }
    }
