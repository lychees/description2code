#include<bits/stdc++.h>
using namespace std;
main(){
	int k,t,j,i,l;
	cin>>t;
	while(t--)
	{
		j=0;
		char w[10010],r[10010];
		scanf("%s", &w);
		scanf("%s", &r);
		k=strlen(r);
		l=strlen(w);
		sort(r, r + k);
		for(i=0;i<l;i++)
		{
			if((int(w[i])<=int(r[j]))||j>=k)
			cout<<w[i];
			else
			cout<<r[j++];
		}
		cout<<endl;
	}
}
