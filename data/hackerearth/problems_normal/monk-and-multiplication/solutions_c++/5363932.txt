#include<bits/stdc++.h>
using namespace std;
long long arr[100005],n,maxx=-1,maxx2=-1,maxx3=-1;
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>n;
	if(n==1)
	{
		cin>>maxx;
		cout<<-1<<endl;
		return 0;
	}
	if(n==2)
	{
		cin>>maxx>>maxx2;
		cout<<-1<<endl<<-1<<endl;
		return 0;
	}
	cin>>arr[0]>>arr[1]>>arr[2];
	cout<<-1<<endl<<-1<<endl;
	sort(arr,arr+3);
	maxx=arr[2];
	maxx2=arr[1];
	maxx3=arr[0];
	cout<<maxx*maxx2*maxx3<<endl;
	for(int i=3;i<n;i++)
	{
		cin>>arr[i];
		if(maxx<arr[i])
		{
			maxx3=maxx2;
			maxx2=maxx;
			maxx=arr[i];
		}
		else
		if(maxx2<arr[i])
		{
			maxx3=maxx2;
			maxx2=arr[i];
		}
		else
		if(maxx3<arr[i])
		{
			maxx3=arr[i];
		}
		if(maxx==-1 || maxx2==-1 || maxx3==-1)cout<<-1<<endl;
		else cout<<maxx*maxx2*maxx3<<endl;
	}
	return 0;
}
