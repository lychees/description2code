#include<iostream>
#include<iomanip>
#include<cstdio>
 
using namespace std;
 
 
int main(){
    //cout<<setprecision(9);
    int t, i;
    double ans,n, k;
    cin>>t;
    while(t--){
        cin>>k>>n;
        if(n<k){
            cout<<1.000000000<<endl;
            continue;
        }
        ans = 1.0;
        for(i=0; i<k; i++){
            ans = ans * ((n-i)/n);
        }
        ans = 1.0 - ans;
        printf("%.9lf\n",ans);
    }
    return 0;
}
