#include <bits/stdc++.h>
#define MAXN 100007
#define MAXX 100000
using namespace std;

int n;
int a[MAXN];
int b[MAXN], c[MAXN];

int beeps[MAXN], mint[MAXN];
int intensity[MAXN];
int q[MAXN];

void pre() {
	memset(q,0,sizeof(q));
	memset(beeps,0,sizeof(mint));
	memset(mint,0,sizeof(mint));
	memset(intensity,0,sizeof(intensity));
	for(int i = 0 ; i < n ; i++) {
		int x = a[i];
		for(int j = x ; j <= MAXX ; j += x) {
			beeps[j] = 1; // there occurs a beep at time j
			intensity[j]++; // one more beep at time j
		}
	}
	
	for(int i = 1 ; i <= MAXX ; i++) {
		mint[i] = max(mint[i], intensity[i]); // max intensity uptil i
		mint[i] = max(mint[i - 1], mint[i]);
	}
	
	for(int i = 1 ; i <= MAXX ; i++) {
		beeps[i] += beeps[i - 1]; // total number of beeps uptil i
	}
}

int main() {
    cin >> n;
    for(int i = 0 ; i < n ; i++) cin >> a[i];
    int p; cin >> p;
    for(int i = 0 ; i < p ; i++) {
    	int x, y;
    	cin >> b[i] >> c[i];
    }
    
    pre();
    
    int q; cin >> q;
    for(int i = 0 ; i < q ; i++) {
		int x, y; cin >> x >> y;
		int b_lim = b[x - 1];
		int i_lim = c[x - 1];
		
		if(beeps[y] > b_lim || mint[y] > i_lim) cout << "Awake\n";
		else cout << "Asleep\n";
	}
}
