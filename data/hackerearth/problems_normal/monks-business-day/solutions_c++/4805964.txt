#include <bits/stdc++.h>
using namespace std;
const int MAX = 1e5 + 5;
typedef long long ll;
#define lim		100000000
//#pragma warning(disable:4996)
#define pb(a)		push_back(a)
#define pp			pop_back()
#define sz			size()
#define ins(a)		insert(a)
#define pf(a)		printf("%d",a)
#define sf(a)		scanf("%d",&a)
typedef pair<int, int> pii;
typedef pair<ll, ll> pll;
typedef pair<ll, pll> node;
ll dist[MAX];
pair<ll,pll> edges[MAX];

int main()
{
	ios::sync_with_stdio(0);
//	freopen("in", "r", stdin);
	int t;
	cin >> t;
	while (t--) {
		int n, m;
		cin >> n >> m;
		int i, j, c;
		for (int k = 0;k<m;k++) {
			cin >> i >> j >> c;
			edges[k]=make_pair(-c,pll(i,j));
		}
		for (int i = 1;i <= n;i++) {
			dist[i] = lim;
		}
		bool cyc=0;
		dist[1]=0;
		for (int i =1;i<=n;i++) {
			for (int j=0;j<m;j++) {
				int u = edges[j].second.first;
				int v = edges[j].second.second;
				int w = edges[j].first;
				if(dist[u]!=lim&&dist[v]>dist[u]+w)dist[v]=dist[u]+w;
			}
		}
		for(int j=0;j<m;j++){
			int u = edges[j].second.first;
			int v = edges[j].second.second;
			int w = edges[j].first;
			if(dist[u]!=lim&&dist[v]>dist[u]+w)cyc=1;
		}
		if (cyc)cout << "Yes\n";
		else cout << "No\n";
	}
	return 0;
}
