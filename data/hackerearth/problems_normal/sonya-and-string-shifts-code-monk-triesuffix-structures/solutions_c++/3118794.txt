#include<iostream>
using namespace std;
int main()
{
	int len,i,j,ind,ind2;
	cin>>len;
	int dp[len];
	dp[0]=0;
	string s;
	cin>>s;
	for(i=1;i<len;i++)
	{
		    j=0;
		    ind=i+j;ind2=dp[i-1]+j;
			if(ind>=len) ind-=len;
			if(ind2>=len) ind2-=len;
			while(s[ind]==s[ind2])
			{
				j++;
				ind =i+j;
				ind2=dp[i-1]+j;
				if(ind>=len) ind-=len;
				if(ind2>=len) ind2-=len;
			}
			if(s[ind]>s[ind2])
			dp[i]=dp[i-1];
			else
			dp[i]=i;
		
	}
	int qr;
	cin>>qr;
	while(qr--)
	{
		int a;
		cin>>a;
		cout<<dp[a]<<endl;
	}
}
