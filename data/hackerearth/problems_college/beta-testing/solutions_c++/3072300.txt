#include <bits/stdc++.h>
using namespace std;
bool compare(pair< pair<string,int> , int> A, pair< pair<string,int> , int> B) {
	return A.second > B.second or (A.second == B.second and A.first.second < B.first.second);
}
void solve(vector<string> V,int n) {
	map<string,int> M;
	for(int i=0;i<n;i++) {
		M[V[i]] += 1;
	}
	int RC = 0;
	vector< pair< pair<string,int> , int> > R;
	for(int i=0;i<n;i++) {
		if(M[ V[i] ] != 0) {
			R.push_back(make_pair(make_pair(V[i],RC++),M[ V[i] ]));
		}
		M[ V[i] ] = 0;
	}
	sort(R.begin(),R.end(),compare);
	for(int i=0;i<RC;i++) {
		cout<<R[i].first.first<<" "<<R[i].second<<endl;
	}
}
int main()
{
    int n;
    cin>>n;
    vector<string> V(n);
    for(int i=0;i<n;i++)
    	cin>>V[i];
    solve(V,n);
    return 0;
}
