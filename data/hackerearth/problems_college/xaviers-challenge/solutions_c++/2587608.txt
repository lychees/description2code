#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t; cin>>t;
	getchar();
	while(t--){
		string s;
		char r[101]; 
		getline(cin,s);
		int l=s.length(); int sum=0; int i,j;
		for(int k=0;k<l;k++) sum+=s[k];		
		int f=sum/l;
  
		for (i=l-1,j=0;i>=0;i--,j++)
		    r[j]=s[i]; r[j]='\0';
		    
		if(f%2!=0) cout<<r<<endl;
		else cout<<s<<"\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
