#include <bits/stdc++.h>
#define ll long long
#define pb push_back
#define mk make_pair
#define pi pair<int, int>
#define fi first
#define se second
using namespace std;
const int M = 1e9 + 7;
ll dp[505][5050];
ll val[15][15][2];
int main() {
	int n, m, K;
	cin >> n >> m >> K;
	if(m == 1) {
		cout << "0";
		return 0;
	}
	val[0][0][0] = 1;
	for (int i = 1; i <= m; i++) {
		for(int j = 0; j <= m / 2 + 1; j++) {
			val[i][j + 1][1] = (val[i][j + 1][1] + val[i - 1][j][0]) % M;
			val[i][j][0] = (val[i][j][0] + val[i - 1][j][1] + val[i - 1][j][0]) % M;
		}
	}
	dp[0][0] = 1;
	for(int i = 1; i <= n; i++) {
		for(int j = 2; j <= m / 2 + 1; j++) {
			for(int k = 0; k <= 5000; k++) {
				if(k + j > 5000) continue;
				dp[i][k + j] = (dp[i][k + j] + (1LL * dp[i - 1][k] * (val[m][j][0] + val[m][j][1])) % M) % M;
			}
		}
	}
	ll ans = 0;
	for(int i = K; i <= 5000; i++) {
		ans = (ans + dp[n][i]) % M;
	}
	cout << ans;
	return 0;
}
