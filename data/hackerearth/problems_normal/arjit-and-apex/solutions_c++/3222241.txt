#include <bits/stdc++.h>
using namespace std;

struct realm
{
	int a;
	int b;
}current[100010], old[100010];

bool myfunc(struct realm emp1, struct realm emp2)
{
	if(emp1.a < emp2.a)
		return true;
	else if(emp1.a == emp2.a and emp1.b < emp2.b)
		return true;
	else
		return false;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int m, n, g, h, count_g=0, count_h=0, i, j;
		cin >> m >> n;
		
		for(i=0; i<m; i++)
		{
			cin >> current[i].a >> current[i].b;
		}

		for(i=0; i<n; i++)
		{
			cin >> old[i].a >> old[i].b;
		}

		cin >> g >> h;

		sort(current, current+n, myfunc);
		sort(old, old+n, myfunc);

		for(i=0,j=0; i<m,j<n; i++,j++)
		{
			if(current[i].a==old[i].a)
			{
				count_g++;
				if(current[i].b==old[i].b)
					count_h++;
			}
		}

		if(count_g==g)
		{
			if(count_h==h)
				cout << "Great\n";
			else
				cout << "Good\n";
		}
		else
			cout << ":(\n";
	}
	return 0;
}
