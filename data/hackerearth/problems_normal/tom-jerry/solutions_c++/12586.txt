#include <iostream>
using namespace std;
int count =0;
int n, a[100][100];
int valid(int i, int j)
{
	if(i>-1 && i<n && j>-1 && j<n)
		return 1;
	return 0;
}

void pass(int i, int j)
{
	if(a[i][j]==1)
		return;	
	
	if(i==n-1 && j==n-1)
	{
		a[i][j] = 0;
		count ++;
		return;
	}
	
	
	a[i][j]=1;
	if(valid(i+1,j))
		pass(i+1,j);
	if(valid(i,j+1))
		pass(i,j+1);
	if(valid(i-1,j))
		pass(i-1,j);
	if(valid(i,j-1))
		pass(i,j-1);
	a[i][j]=0;
}

int main()
{
    int i,j;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	for(j=0;j<n;j++)
    	{
    		cin>>a[i][j];
    	}
    }
    pass(0,0);
    cout<<count<<endl;
}
