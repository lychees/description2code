#include<bits/stdc++.h>
using namespace std;

const int N=1024;

bool dp[N];
int in[N];

int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int m,n,ans=0;
        scanf("%d%d",&n,&m);
        memset(dp,0,sizeof(dp));
        dp[0]=1;
        while(m--) {
            int a;
            scanf("%d",&a);
            for(int i=n-1;i-a>=0;i--) {
                if(dp[i-a]==true) {
                    dp[i]=true;
                }
            }
        }
        for(int i=0;i<n;i++) {
            scanf("%d",in+i);
        }
        sort(in,in+n);
        int cnt=0;
        for(int i=0;i<=n;i++) {
            if(i>0&&i<n&&in[i-1]==in[i]) {
                cnt++;
            } else {
                if(cnt>0&&dp[cnt]) {
                    ans++;
                }
                cnt=1;
            }
        }
        printf("%d\012",ans);
    }
    return 0;
}
