#include <iostream>
using namespace std ; 
int main()
{
	int n ; 
	cin >> n ; 
	int i ; 
	int a[n] ; 
	for ( i = 0 ; i< n ; i++)
	{
		cin >> a[i] ; 
	}
	int m ; 
	cin >> m ; 
	while ( m--)
	{
		int p ; 
		cin >> p ; 
		for ( i = 0; i< n ; i++)
		{
			if ( a[i]>p )
			{
				a[i]  = a[i] -1 ; 
			}
		}
	}
	for( i = 0; i< n ; i++)
	{
		cout << a[i] << " " ; 
	}
	cout << endl; 
	return 0  ; 
}
