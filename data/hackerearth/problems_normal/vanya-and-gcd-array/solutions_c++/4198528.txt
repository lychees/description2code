#include<iostream>

#include <algorithm>
using namespace std;
#define MOD 1000000007
int greatestcomdivisor(int a, int b);
int main()
{
	int no;
	cin>>no;
	int arr[50002];
	for(int i=0;i<no;i++)
		cin>>arr[i];
	long long dp[no][101];
	for(int i=0;i<=no-1;i++)	{
		for(int j=0;j<=100;j++)
			dp[i][j]=0;
	}
	dp[0][arr[0]]=1;
	long long res = dp[0][1];
	int  temp=-1;
	//for(int i=1;i<no;i++) 
	int i=1;
	while(i<no)
	{
   		for(int j=i-1;j>=0;j--) {
   			if(arr[i]>arr[j]){
   				for(int k=1;k<=100;k++){
   					dp[i][greatestcomdivisor(arr[i],k)] += dp[j][k];
					dp[i][greatestcomdivisor(arr[i],k)] %= MOD;
				   }
			   }
    		}
    	dp[i][arr[i]]++;
		res += dp[i][1];
		res %= MOD;
		i++;
	}
	cout<<res<<endl;
//	getch();
	return 0;
}
int greatestcomdivisor(int a, int b){
	if(a%b==0)
	return b;
	return greatestcomdivisor(b,a%b);
}
