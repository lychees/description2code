#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,flag,n1,n2,i;
    string s1,s2;
    scanf("%d",&t);
    while(t--)
    {
    	cin>>s1>>s2;
    	vector<int> v(26);
    	n1=s1.length();
    	n2=s2.length();
    	if(n1^n2)
    	{
    		flag=false;
    	}
    	else
    	{
    		flag=true;
    		for(i=0;i<n1;i++)
    		{
    			v[s1[i]-97]++;
    			v[s2[i]-97]--;
    		}
    		for(i=0;i<26;i++)
    		{
    			if(v[i])
    			{
    				flag=false;
    				break;
    			}
    		}
    	}
    	if(flag)
    		printf("Yes\n");
    	else
    		printf("No\n");
    }
    return 0;
}
