#define pb				push_back

bool wins(int n, int ct,int t,int type){
	vi v;
	if(type)v.pb(1),t--;
	else v.pb(0),ct--;
	for(int i=1;ct||t;i++){
		if(v.size()==n)break;
		if(i%2){
			if(v.back()==0 && t){
				t--;
				v.pb(1);
			}else if(v.back()==1 && ct){
				ct--;
				v.pb(0);
			}else if(t){
				t--;
				v.pb(1);
			}else{
				ct--;
				v.pb(0);
			}
		}else{
			if(v.back()==0 && ct){
				ct--;
				v.pb(0);
			}else if(v.back()==1 && t){
				t--;
				v.pb(1);
			}else if(t){
				t--;
				v.pb(1);
			}else{
				ct--;
				v.pb(0);
			}
		}
	}
	int n1=0,n2=0;
	FOR(i,1,v.size()){
		if(v[i]==v[i-1])n1++;
		else n2++;
	}
	return n1>n2;
}
