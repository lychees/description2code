#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int dp[2000007];

int main()
{
    ll t, n, m1, m2;

    dp[0] = 0;

    cin >> t;

    while (t--) {
        cin >> n >> m1 >> m2;

        if (m1 == 1 || m2 == 1) {
            dp[1] = 1;
        } else {
            dp[1] = 0;
        }

        for (int i = 1; i <= n; i++) {
            if (i >= m1 && dp[i - m1] == 0) {
                dp[i] = 1;
            } else if (i >= m2 && dp[i - m2] == 0) {
                dp[i] = 1;
            } else {
                dp[i] = 0;
            }
        }
        cout << dp[n] << endl;
    }
    return 0;
}
