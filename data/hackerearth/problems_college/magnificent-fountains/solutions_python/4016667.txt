
from fractions import gcd
T = int(raw_input())
for t in xrange(T):
        N = int(raw_input())
        arr = map(int,raw_input().strip().split())
        lcm = arr[0]
        for i in xrange(1,N):
                lcm = (lcm * arr[i])/gcd(lcm,arr[i])
        lcm=lcm%1000000007
        print lcm
