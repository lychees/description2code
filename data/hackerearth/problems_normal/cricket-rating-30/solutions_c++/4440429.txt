#include <vector>
#include <iostream>
using namespace std;
int main()
{
	int x;
	cin >> x;
	vector <int> a;

	while(x--)
	{
		int q;
		cin >> q;
		a.push_back(q);
	}
	int t=0,t1=0;
	for(int d=0;d<a.size();d++)
	{

		t+=a[d];
		if(t1<t)
		{
			t1=t;
		}

		if(t<0)
		{
			t=0;
		}
	}
	if(t1<t)
	{
		t1=t;
	}
	cout << t1 << endl;

}
