for x in xrange(int(input())):
	word, pwd = raw_input().split()
	word=word.replace(" ", "")
	pwd=pwd.replace(" ", "")
	res = True
	if 2*len(word) == len(pwd):
		for ch in word:
			try:
				pwd=pwd.replace(ch,"",2)
			except:
				res = False
	if res and not pwd:
		print "Possible"
	else:
		print "Impossible"
