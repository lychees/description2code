int computeAngle(int h, int m) {
    if (h == 12) {
        h = 0;
    }
    if (m == 60) {
        m = 0;
    }
 
    int hourAngle = 0.5 * (h*60 + m);
    int minuteAngle = 6*m;
 
    int angle = abs(hourAngle - minuteAngle);
 
    angle = min(360-angle, angle);
 
    return angle;
}
