#include <iostream>
using namespace std;

int main()
{
	int tc;
	int h, m, a;
    cin >> tc;
    while(tc--)
    {
    	cin >> h >> m;
    	a = h * 30 + m / 2 - m * 6;
    	if(a < 0)
    		a = 0 - a;
    	//cout << a << endl;
    	a = a < 360 - a ? a : 360 - a;
    	cout << a << endl;
    }
    return 0;
}
