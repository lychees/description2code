#include <bits/stdc++.h>

using namespace std;

#define DEBUG ON

#ifdef DEBUG
#define dout(...) cout<<__VA_ARGS__
#else
#define dout(...) do_nothing()
#endif

void do_nothing() {}



int main()
{
    int no_tests;
    cin>>no_tests;
    int no_cities;
    char name[11];
    vector<int> city;
    int flag;
    while(no_tests--)
    {
        flag = 0;
        city.clear();
        city.assign(27, 0);

        cin>>no_cities;
        for(int i=1; i<=no_cities; i++)
        {
            cin>>name;
            if( city[ name[0] - 'a' ] == 0)
            {
                city[ name[0] - 'a' ]++;
            }
            else
                flag = 1;
        }
        if(flag == 1)
            cout<<"NO"<<endl;
        else
            cout<<"YES"<<endl;
    }

    return 0;
}
