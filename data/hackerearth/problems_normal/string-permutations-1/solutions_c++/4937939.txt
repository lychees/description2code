#include <iostream>
#include<map>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    map<string,int> mymap;
    while(t--)
    {
    	string str;
    	cin>>str;
    	mymap[str]++;
    }
    map<string,int> ::iterator it=mymap.begin();
    int times=0;
    int mx=0;
    for(it;it!=mymap.end();it++)
    {
    	if(it->second>mx)
    	{
    		mx=it->second;
    		times=1;
    	}
    	else if(it->second==mx)
    	  times++;
    }
    cout<<(mx*times);
    return 0;
}
