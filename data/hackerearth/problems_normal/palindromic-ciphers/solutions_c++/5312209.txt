#include<iostream>
#include<algorithm>
using namespace std;
bool palcheck(string str)
{
	string tmp1 = str;
	reverse(tmp1.begin(),tmp1.end());
	if(tmp1 == str)
	{
		cout<<"Palindrome";
		return true;
	}
	else return false;
}

int main()
{
	string alpha="abcdefghijklmnopqrstuvwxyz",str;
	int t;

}
