#include <iostream>
using namespace std;

int main()
{
	string s;
	cin>>s;
	int len=s.length();
	int i,l=0,o=0,v=0,e=0;
	for(i=0;i<len;i++)
	{
		if(s[i]=='l')
		l++;
	else if(s[i]=='o' && l>0)
		o++;
	else if(s[i]=='v' && l>0 && o>0)
		v++;
		else if(s[i]=='e' && l>0 && o>0 && v>0)
		e++;
		else
		continue;
    }
    if(l>0 && o>0 && v>0 && e>0)
    cout<<"I love you, too!";
    else
    cout<<"Let us breakup!";
    //cout << "Hello World!" << endl;
    return 0;
}
