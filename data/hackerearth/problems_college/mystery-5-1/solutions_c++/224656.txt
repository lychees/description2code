#include<iostream>
#include<cstring>
using namespace std;
int main()
{
	int test,count,max,i,j,k;
	char string[11000];
	int carr[127];
	scanf("%d",&test);
	while(test--)
	{
		for(i=0;i<127;i++)
		carr[i]=0;
		max=0;
		scanf("%s",string);
		i=0;
		while(string[i]!='\0')
		{
			j=string[i];
			carr[j]++;
			i++;
		}
		max=carr[0];
		for(k=0;k<127;k++)
		{
			if(max<carr[k])
			max=carr[k];	
		}
		
		printf("%d\n",max);
	}
	return 0;
}
