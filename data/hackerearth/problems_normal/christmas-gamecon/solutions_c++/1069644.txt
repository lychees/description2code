#include<bits/stdc++.h>
using namespace std;
inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}
int main(){
    int t=scan();
    while(t--){
        int n=scan();
        int xo=0;
        while(n--){
            xo^= (scan()%5);
        }
        if(xo){
            printf("Aniruddha\n");
        }
        else{
            printf("Andrew\n");
        }
    }
}
