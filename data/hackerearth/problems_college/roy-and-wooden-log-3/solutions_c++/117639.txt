#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long n,way,t;
    	scanf("%lld",&n);
    	if(n%2)
    	{
    		t=(n+1)/2;
    	}
    	else
    		t=n/2;
    	t-=1;
    	way=n/2 + 1;
    	printf("%lld %lld\n",way,t);
    }
    return 0;
}
