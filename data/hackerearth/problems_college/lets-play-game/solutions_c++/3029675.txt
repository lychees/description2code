#include<iostream>
#include<algorithm>
#include<string>
using namespace std;

int main()
{
    int t;
    int n,k;
    cin>>t;
    while(t>0)
    {
        cin>>n>>k;
        int a[n],b[n],c[n];
        for(int i=0;i<n;i++)
            cin>>a[i];
        for(int i=0;i<n;i++)
            cin>>b[i];

        for(int i=0;i<n;i++)
            c[i]=(k/a[i])*b[i];

            int elements = sizeof(c) / sizeof(c[0]);
  sort(c, c + elements);

        cout<<c[n-1]<<endl;

    t--;}


}
