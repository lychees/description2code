#include <iostream>
using namespace std;

int main()
{
	int test,a,b;
    cin>>test;
    while(test--)
    {
    	cin>>a>>b;
    	int first = a/b;
    	int second = first+1;
    	
    	if(abs(a-first*b)<abs(second*b-a))
    	{
    		cout<<first*b<<endl;
    	}
    	else
    	{
    		cout<<second*b<<endl;
    	}
    }
    return 0;
}
