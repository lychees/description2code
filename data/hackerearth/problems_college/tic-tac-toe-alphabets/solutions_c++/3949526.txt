#include<iostream>
#include<string.h>
using namespace std;
int main(){
	int m,i,j;
 	char a[1000],t;
 	cin>>m;
 	while(m--){
 		cin>>a;
 		for(i=0;i<strlen(a);i++){
 			for(j=i+1;j<strlen(a);j++){
 				if(a[i]>a[j]){
 					t=a[i];
 					a[i]=a[j];
 					a[j]=t;
 				}
 			}
 		}
 		cout<<a<<endl;
 	}
}
