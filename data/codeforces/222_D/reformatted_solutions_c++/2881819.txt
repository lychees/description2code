#include <algorithm>
#include <cmath>
#include <cstdio>
#include <iostream>
using namespace std;

long ans = 0, a[500000], b[500000], i, n, k, l, r;

int main() {
  cin >> n >> k;
  for (i = 0; i < n; i++)
    cin >> a[i];
  for (i = 0; i < n; i++)
    cin >> b[i];
  sort(a, a + n);
  sort(b, b + n);
  r = n - 1;
  for (l = 0; l < n; l++)
    if (a[l] + b[r] >= k) {
      ans++;
      r--;
    }
  cout << "1 " << ans;
  return 0;
}
