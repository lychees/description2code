
def find(qu, n):

    wro = 0
    for i in xrange(n):

        for j in xrange(i+1, n):
            r = abs(qu[i][0] - qu[j][0])
            c = abs(qu[i][1] - qu[j][1])
            dis = r + c
            # print r, c
            if (qu[i][2] %
                2 == 0 and qu[j][2] %
                2 != 0) or (qu[i][2] %
                            2 != 0 and qu[j][2] %
                            2 == 0):
                if dis % 2 == 0:
                    # print 'asdf', i, j
                    wro = 1
                    break

            else:
                if dis % 2 != 0:
                    # print 'qwe', i, j
                    wro = 1
                    break

        if wro == 0:
            break

    if wro == 1:
        return 'No'

    else:
        return 'Yes'

    que = []
    for j in xrange(n):
        que.append(map(int, raw_input().split()))

    sol.append(find(que, n))
