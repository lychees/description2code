#include <bits/stdc++.h>
using namespace std;

bool alpha(char *str1,char *str2)
{
    int l1=strlen(str1);
    int l2=strlen(str2);
    if(str1[0]==str2[0] && str1[l1-1]==str2[l2-1])
        return true;
    return false;
}

bool anagram(char *str1,char *str2)
{
    int c1[256] = {0};
    int c2[256] = {0};
    int i;
    for (i=0;str1[i] && str2[i];i++)
    {
        c1[str1[i]]++;
        c2[str2[i]]++;
    }
    if (str1[i] || str2[i])
      return false;
    for (i=0;i<256;i++)
        if(c1[i] != c2[i])
            return false;

    return true;
}

int main()
{
    int n,i,j,c=0,z=0;
    cin>>n;
    int x[n];
    for(i=0;i<n;i++)
        x[i]=0;
    char s[n][100];
    for(i=0;i<n;i++)
        cin>>s[i];
    for(i=0;i<n-1;i++)
    {
        if(!x[i])
        {
            for(j=i+1;j<n;j++)
            {
                if(alpha(s[i],s[j]) && anagram(s[i],s[j]))
                {
                    if(x[i]==0)
                    {
                        c++;
                        x[i]=x[j]=c;
                    }
                    else
                    {
                        x[j]=x[i];
                    }
                    z=1;
                }
            }
            if(z!=1)
            {
                c++;
                x[i]=c;
            }
        }
        z=0;
    }
    if(!x[n-1])
    {
        c++;
        x[n-1]=c;
    }
    cout<<c<<endl;
    return 0;
}
