#include<bits/stdc++.h>
using namespace std;

struct trie
{
	trie *a[26];
	trie() { for(int i=0;i<26;++i) a[i]=NULL; }
};
int cnt=1;

trie *insert(string &s,int ind,trie *root)
{
	int cur = s[ind] - 'a';
	if(!root->a[cur]) 
	{
		root->a[cur]=new trie(); cnt++;
	}
	if(ind+1 != s.size() )
	{
		root->a[cur] = insert(s,ind+1,root->a[cur]);
	}
	return root;
}
