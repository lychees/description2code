#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {   map<int,int> m1;
    	int n;
    	cin>>n;
    	 int arr[n];
    	for(int i=0;i<n;i++)
    	{cin>>arr[i];
    	 if(m1[arr[i]])
          m1[arr[i]]++;
          else m1[arr[i]]=1;
        }
        map<int,int>::iterator itr=m1.begin();
        long long int count=0;
        while(itr!=m1.end())
        { //  cout<<itr->second<<" ";
        	count+=(itr->second*(itr->second+1)/2);
        	itr++;
        }
        cout<<count<<endl;
    	
    }
    
    
    return 0;
}
