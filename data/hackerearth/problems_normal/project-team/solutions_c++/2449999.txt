#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,a[1005];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(int i=0;i<n;i++)cin>>a[i];
    	sort(a,a+n);
    	int mx = a[0] + a[n-1];
    	int mn = a[0] + a[n-1];
    	for(int i=0;i<n/2;i++)
    	{
    		mx = max(mx , a[i] + a[n-1-i]);
    		mn = min(mn , a[i] + a[n-1-i]);
		}
		cout<<abs(mx-mn)<<endl;
    }
    return 0;
}
