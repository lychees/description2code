#include <iostream>
using namespace std;
#include<stdio.h>
#include<string.h>

int main()
{
 int t,i,j,k,l;
 scanf("%d",&t);
 char A[51];
 while(t--)
 {
 	cin>>A;
 	l=strlen(A);
 	char K[51];
 	i=l;
 	j=0;
 	while(i>=0)
 	{
 		if(i%2!=0)
 		{
 			K[j]=A[i/2];
 			j++;
 			for(k=i/2;k<l;k++)
 			 A[k]=A[k+1];
 			l--; 
 		}
 		else
 		{
 			if((int)A[i/2]<(int)A[(i/2)-1])
 			{
 			 K[j]=A[i/2];
 			 j++;
 			 for(k=i/2;k<l;k++)
 		      A[k]=A[k+1];
 		     l--;
 			}
 			else
 			{
 			 K[j]=A[(i/2)-1];
 			 j++;
 			 for(k=(i/2)-1;k<l;k++)
 			  A[k]=A[k+1];
 			 l--; 
 			}
 		}
 		i--;
 	}
 	K[j]='\0';
 	cout<<K<<"\n";
 }
 return 0;
}
