#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[20],b[20],ans,len;
ll int n;
void myfind()
{
	ll int i,t,mc=0,my_and=-1ll,my_xor=0ll;
	for(i=0;i<n;i++)
	{
		if(b[i])
		{
			my_and=my_and & a[i];
			my_xor=my_xor ^ a[i];
			mc++;
		}
	}
	t=(my_and|my_xor);
	if(t>ans)
	{
		ans=max(ans,t);
		len=mc;
	}
	else if(t==ans)
	len=max(len,mc);
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		cin>>n;
		ll int i,j;
		ans=0;len=0;
		for(i=0;i<n;i++)
		cin>>a[i];
		memset(b,0,sizeof(b));
		for(i=0;i<(1ll<<n)-1;i++)
		{
			for(j=(n-1);j>=0;j--)
			{
				b[j]=(b[j]+1)%2;
				if(b[j])
				break;
			}
			myfind();
		}
		cout<<ans<<" "<<len<<endl;
	}
	return 0;
}
