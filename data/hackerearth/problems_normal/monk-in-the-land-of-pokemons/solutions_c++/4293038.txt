#include <bits/stdc++.h> 

using namespace std ;

int a[1000001]={0};


int main()
{
	int t;
	cin >> t; 
	int temp;
	int n,count =0;
	int x,y;
	int k;

	while (t--)
	{
		for(k=0;k<1000001;k++)
			a[k]=0;

		cin >> n;

		while ( n-- )
		{
			cin >> x >> y ;
		
			a[x]++;

			if(a[y]==0)
				count ++ ;
			else
				a[y]--;

		
		}


		cout << count << endl ;
		count =0;




	}


	return 0;
}
