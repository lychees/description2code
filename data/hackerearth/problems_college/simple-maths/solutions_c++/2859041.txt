#include <iostream>
using namespace std;

int gcd(int a,int b)
{

return a%b==0?b:gcd(b,a%b);

}


int main()
{
    
    int  te;
    cin>>te;
    
    while(te--)
    {
    	
    	long long int a,b,n;
    	long long com;
    	cin>>n>>a>>b;
    	if(a==0 || b==0)
    		{cout<<-1<<endl;continue;}
    	com=(a*b)/gcd(a,b);
    	
    	cout<<(n/a)+(n/b)-(n/com)<<endl;
    	
    	
    }
    
    
    
    return 0;
}
