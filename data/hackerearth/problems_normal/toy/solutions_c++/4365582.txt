#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
const unsigned int MAX = 100001;
int main()
{
    unsigned int n, temp;
    vector<unsigned int> v;
    cin >> n;
    for(unsigned int i = 0; i < n; ++i){
    	cin >> temp;
    	v.push_back(temp);
    }
    sort(v.begin(), v.end());
    unsigned int ans = 1, current = v[0];
    for(unsigned int i = 1; i < n; ++i){
    	if(v[i] > current + 4){
    		++ans;
    		current = v[i];
    	}
    }
    cout << ans << endl;
    return 0;
}
