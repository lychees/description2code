#include<bits/stdc++.h>
using namespace std;
main(){
	long long int n,i,j,q,d,p=1;
	cin>>n;
	int arr[n];
	for(i=0;i<n;i++)
		cin>>arr[i];
	cin>>q;
	//cout<<INT_MAX<<endl;
	while(q--){
		cin>>d;
		if(p<=INT_MAX)
			p=p*d;
	}
		for(i=0;i<n;i++)
			arr[i]=arr[i]/p;
	for(i=0;i<n;i++){
		cout<<arr[i]<<" ";
	}
}
