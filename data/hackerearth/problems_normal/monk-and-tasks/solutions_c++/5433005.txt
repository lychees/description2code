#include <iostream>
#include <algorithm>
#include <vector>

int countBits(int i) {
  int count = 0;
  while(i) {
    if(i & 1) 
      count++;
    i >>= 1;
  }
  return count;
}

struct CompareByBits {
  bool operator()(int a, int b) {
    return countBits(a) < countBits(b);
  }
};

int main(int argc, char * argv[]) {
  std::vector<int> numbers;
  CompareByBits comparer;
  int T, N;
  std::cin >> T;
  for(int i = 0; i < T; ++i) {
    std::cin >> N;
    numbers.clear();
    for(int j = 0; j < N; ++j) {
      int num;
      std::cin >> num;
      numbers.push_back(num);
    }
    std::stable_sort(numbers.begin(), numbers.end(), comparer);
    for(auto num : numbers) {
      std::cout << num << " ";
    }
    std::cout << std::endl;
  }

  return 0;
}
