
#include <stdio.h>
#include<iostream>
#include<math.h>
#include<stdlib.h>
using namespace std;

int main()
{
    int s[100],i,n,j,count,rem;
   
    cin>>n;
    for(i=0;i<n;i++)
    cin>>s[i];
    for(i=0;i<n;i++)
    {
    	for(j=i+1;j<n;j++)
    	{
    		if(s[i]==s[j])
    		{
    		cout<<"duplicate exists";
    		exit(0);
    		}
    	}
    }
    for(i=0;i<n;i++)
    {count=0;
    	for(j=0;j<n;j++)
    	{
    			rem=s[i]%s[j];
    			if(rem==0)
    			count++;
    			if(count>1)
    			break;
    		
    	}
    	if(count==1)
    	cout<<s[i]<<" ";
    }
    return 0;
}
