#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int t,n,m;
	int a[1000],b[1000],i,c;
	cin>>t;
	
	
	while(t--)
	{
	    cin>>n;
	    cin>>m;
	    
	    
	    for(i=0;i<n;i++)
	    {
	        cin>>a[i];
	    }
	    
	    
	    while(m--)
	    {

	        for(i=0;i<n;i++)
	        {
	            if(i==0)
	            {
	                if(n>=2 && a[1]==1)
	                {
	                    b[i]=1;
	                }
	                else
	                {
	                    b[i]=0;
	                }
	            }
	            else if(i==n-1)
	            {
	                if(a[n-2]==1)
	                {
	                    b[i]=1;
	                }
	                else
	                {
	                    b[i]=0;
	                }
	            }
	            else
	            {
	                if(a[i+1]==1 && a[i-1]==1)
	                {
	                    b[i]=1;
	                }
	                else
	                {
	                    b[i]=0;
	                }
	            }
	        
	    }
	    c=0;
	  
	    	for(i=0;i<n;i++)
	{
	     if(a[i]==b[i])
	            c++;
	    else
	    a[i]=b[i];
	}
	 if(c==n)
	        break;
	}
	for(i=0;i<n;i++)
	{
	    cout<<a[i]<<" ";
	}
	cout<<endl;
	}
	return 0;
}
