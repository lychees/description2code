#include <bits/stdc++.h>
using namespace std;

int main()
{
   int n;
   
   cin>>n;
   
   int a[n+2];
   int dp[n+2];
   int dp1[n+2];
   for(int i = 0; i < n; i++){
   		cin>>a[i];
   		dp[i] = 0;
   		dp1[i] = 0;
   }
   
   sort(a,a+n);
   
   for(int i = 0; i < n; i++){
   		dp[i] = dp[i]+i;
   		for(int j = n-1; j >= i; j--){
   			if((a[j]-3*a[i]) <= 0){
   				break;
   			}
   			dp[i] = dp[i]+1;
   		}
   		
   }
   
   for(int i = n-1; i >= 0; i--){
   		dp1[i] = n-i-1;
   		for(int j = 0; j <= i; j++){
   			if((a[i]-3*a[j]) <= 0){
   				break;
   			}
   			dp1[i] = dp1[i]+1;
   		}
   }
   int ans = INT_MAX;
   
   for(int i = 0; i < n; i++){
   	ans = min(ans,min(dp[i],dp1[i]));
   }
   
   cout<<ans<<endl;
   
   return 0;
}
