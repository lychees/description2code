#include <iostream>
using namespace std;
long long dp[1600][160];
int c[60];
int main(){
//	freopen("in","r",stdin);
	int n,m;
	cin>>n>>m;
	for(int i=0;i<m;i++){
		cin>>c[i];
	}
	for(int i=0;i<=n;i++){
    for(int j=0;j<m;j++){
      if(i == 0)
         dp[i][j] = 1;          
      else if(j==0)
      	if(i%c[j]==0)
       	 	dp[i][j]=1;
     	else 
     		dp[i][j]=0;
      else if(c[j]>i)
         dp[i][j]=dp[i][j-1];
      else 
         dp[i][j] = dp[i-c[j]][j] + dp[i][j-1];
 		}
 	}
 	cout<<dp[n][m-1];
	return 0;
}
