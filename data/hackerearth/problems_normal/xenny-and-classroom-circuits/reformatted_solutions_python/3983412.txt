for i in range(int(raw_input())):
    n = int(raw_input())
    s = raw_input()
    c1 = 0
    c2 = 0
    for j in range(0, (2*n)-1, 2):
        if s[j] == 'B':
            c2 += 1
        else:
            c1 += 1
        if s[j+1] == 'B':
            c1 += 1
        else:
            c2 += 1
    if c1 < c2:
        print c1/2
    else:
        print c2/2
