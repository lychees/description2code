#include <iostream>
using namespace std;

int main()
{
	long int n,q,l,r,x[100000],sum=0;
	int i,j,k;
	cin>>n>>q;
	for(i=0;i<n;i++){
		cin>>x[i];
		sum+=x[i];
	}
	
	long int ans=sum;
	
	for(k=0;k<q;k++){
		cin>>l>>r;
		for(j=0;j<l-1;j++){
			ans-=x[j];
		}
		for(j=i-1;j>=r;j--){
			ans-=x[j];
		}
		cout<<ans<<endl;
		ans=sum;
	}
    return 0;
}
