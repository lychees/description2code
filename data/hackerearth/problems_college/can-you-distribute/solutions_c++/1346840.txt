LL BigMod(LL a,LL b,LL mod = MOD)
{
      if(!b) return 1;
      LL x = BigMod(a,b/2,mod);
      x = (x*x)%mod;
      if(b%2)
            return (x*a)%mod;
      else
            return x;
}

LL modinv(LL a,LL n)
{
      return BigMod(a,n);
}
VLL fact(1000);

LL choose(int n,int r)
{
      if(n < r) return -1;
      LL ans = fact[n];
      LL check = MOD;
      check -= check/941;
      check -= check/10627;
      check--;
      ans = (ans * modinv(fact[r],check))%MOD;
      ans = (ans * modinv(fact[n-r],check))%MOD;
      return ans;
}

int main()
{
      fact[0] = 1;
      FOR(i,1,1000) fact[i] = (fact[i-1]*i)%MOD;

      int n,r;
      cin>>n>>r;

      n -= r;

      cout<<choose(n+r-1,r-1)<<endl;

      return 0;
}
