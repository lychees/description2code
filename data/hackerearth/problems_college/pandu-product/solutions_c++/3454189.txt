#include <bits/stdc++.h>
#define ll long long int
#define MOD 1000000007
#define inf 1000000007
#define pii pair<int, int>

using namespace std;

ll N, x;
ll p[1000008], n[1000008];

int main()
{
    cin >> N;
    memset(p, 0, sizeof(p));
    memset(n, 0, sizeof(n));

    int l1 = 0, l2 = 0;

    for (ll i = 0; i < N; i++) {
        cin >> x;
        if (x >= 0) {
            p[l1++] = x;
        } else {
            n[l2++] = x;
        }
    }
    if (N == 2 && l1 == 1 && l2 == 1) {
        cout << p[0] * n[0] << endl;
        return 0;
    }

    sort(p, p + l1);
    sort(n, n + l2);
    ll mx = p[l1 - 1] * p[l1 - 2] > n[0] * n[1] ? p[l1 - 1] * p[l1 - 2] : n[0] * n[1];

    cout << mx << endl;
    return 0;
}
