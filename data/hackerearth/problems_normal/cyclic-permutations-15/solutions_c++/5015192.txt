#include <bits/stdc++.h>
using namespace std;

string A, B;
int Z[300006];

int main()
{
    int T;
    cin >> T;
    
    while(T--)
    {
    	cin >> A >> B;
    	B = A + "$" + B + B;
    	B[B.size()-1] = '#';
    	
    	memset(Z, 0, sizeof(Z));
    	int L = 0, R = 0, ans = 0;
    	
    	for(int i=1; i<B.size(); i++)
    	{
    		if(i<=R)
    			Z[i] = min(R-i+1, Z[i-L]);
    		
    		for(; i+Z[i]<B.size() && B[Z[i]]==B[i+Z[i]]; Z[i]++);
    		
    		if(i+Z[i]-1>R)
    			L = i, R = i+Z[i]-1;
    		
    		if(Z[i]==A.size())
    			ans++;
    	}
    	
    	cout << ans << endl;
    }
    
    return 0;
}
