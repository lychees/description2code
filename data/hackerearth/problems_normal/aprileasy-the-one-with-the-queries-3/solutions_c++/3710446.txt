#include <bits/stdc++.h>
using namespace std;
const int MAXN = 100005;
int nxt[MAXN][26];
string str1,str2;
int main()
{
	cin>>str1;
	int len1 = str1.length();
	for (int i = 0; i < 26; ++i)
		nxt[len1][i] = len1;
	for (int i = len1-1; i >= 0; --i)
	{
		for (int j = 0; j < 26; ++j)
			nxt[i][j] = nxt[i+1][j];
		nxt[i][str1[i]-'a'] = i;
	}
	int q;
	cin>>q;
	while(q--)
	{
		cin>>str2;
		int pos = 0;
		bool flag = true;
		for (int i = 0; i < str2.length() && flag; ++i)
		{
			if(nxt[pos][str2[i]-'a'] >= len1)
				flag = false;
			else
				pos = nxt[pos][str2[i]-'a']+1;
		}
		if(flag)
			cout<<"Yes\n";
		else
			cout<<"No\n";
	}
	return 0;
}
