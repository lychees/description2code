#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
#include <stdio.h>
#include <assert.h>
#include <queue>
using namespace std;
int n;
string bad[11];
string hh;
int m;
char sol[111];
int r=0;
int main(){
	cin>>n;
	assert(1<=n && n<=10);
	for(int i=0;i<n;i++){
		cin>>bad[i];
		assert(bad[i].length()<=100);
	}
	cin>>m;
	int y=m;
	int l=0;
	while(y--){
		cin>>hh;
		l+=hh.length();
		bool found=false;
		for(int i=0;i<n;i++){
			if(hh==bad[i]){
				found=true;
				break;
			}
		}
		if(!found){
			sol[r++]=(char)(hh[0]-'a'+'A');
		}
	}
	assert(l+m-1<=100);
	string g;
	if(cin>>g)
		assert(false);
	cout<<sol[0];
	assert(r!=0);
	for(int i=1;i<r;i++){ 
		cout<<'.'<<sol[i];
	}
	cout<<endl;
}
