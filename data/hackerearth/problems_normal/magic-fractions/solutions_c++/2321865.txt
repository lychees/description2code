#include<bits/stdc++.h>
using namespace std;
int *pr(long long int limit)
{
    long long int i,j;
    int *primes;
    primes = new int[limit+1];

    for (i=2;i<=limit;i++)
        primes[i]=1;

    for (i=2;i<=limit;i++)
        if (primes[i])
            for (j=i;i*j<limit;j++)
                primes[i*j]=0;
    return primes;
}
int main()
{
    long long int n,i,j;
    int *p=pr(600);
    long long int ans[550];
    ans[0]=ans[1]=0;
    ans[2]=1;
    for(i=3;i<=506;i++)
    {
        if(p[i]==1)
            ans[i]=ans[i-1]*2;
        else ans[i]=ans[i-1];
    }
    cin>>n;
    long long int an=0;
    for(i=0;i<=n;i++)
        an+=ans[i];
    cout<<an;
    return 0;
}
