#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<forward_list>

using namespace std;

forward_list <int> vmen, vwomen;

int main()
{
	int T, N;
	int input;
	unsigned long sum = 0;
	cin>>T;
	while (T--)
	{
	 	//Do something
	 	cin>>N;
	 	for (int i = 0; i < N; i++)
	 	{
	 		cin>>input;
	 		vmen.push_front(input);
	 	}
	 	
	 	for (int i = 0; i < N; i++)
	 	{
	 		cin>>input;
	 		vwomen.push_front(input);
	 	}
	 	
	 	vmen.sort();
	 	vwomen.sort();
	 	//vmen.sort(greater<unsigned>());
		for (auto mit = vmen.begin(), wit = vwomen.begin(); mit != vmen.end(); ++mit, ++wit)
			sum += (*mit * *wit);
			
	 	cout<<sum<<endl;
	 	vmen.clear();
	 	vwomen.clear();
	 	sum = 0;
	}
	return 0;
}
