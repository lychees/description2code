#include <iostream>
using namespace std;

int main()
{
	int T;
	cin >> T;
	for (int it = 0; it < T; it++) {
		int n;
		cin >> n;
		int sum = 0, a, b, c;
		for (int i = 0; i < n; i++) {
			cin >> a >> b >> c;
			sum ^= a ^ b ^ c;
		}
		if (sum)
			cout << "Stark" << endl;
		else
			cout << "Lannister" << endl;
	}
	return 0;
}
