#include <bits/stdc++.h>
using namespace std;
char str[100005];
int main()
{
	int i;
    gets(str);
    if(str[0] <= 'z' && str[0] >= 'a'){
    	str[0] =	str[0] - 32;
    }
    for(i = 1;str[i] != '\0';i++){
    	if(str[i] == ' '){
    		if(str[i+1] != '\0' && str[i+1] != ' '){
    			if(str[i+1] <= 'z' && str[i+1] >= 'a'){
    				str[i+1] -= 32;
    			}
    		}
    	}
    }
    puts(str);
    return 0;
}
