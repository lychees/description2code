#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int i,j,k,n,p,q,r,tt;
    long double aaa;
    
    cin >> p >> q;
    
    cin >> tt;
    
    while (tt--) {
    	cin >> i >> j;
 
		k = j/p - (i-1)/p + j/q - (i-1)/q;
		
		r= p*q;
		
		k = k - j/r + (i-1)/r;
		
		aaa = (long double) k / (long double) (j-i+1);
		
		cout << fixed << setprecision(6) << aaa << endl;
    }
    return 0;
}
