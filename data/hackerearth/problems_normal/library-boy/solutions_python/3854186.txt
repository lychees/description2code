#!/usr/bin/env python

from sys import stdin, stderr

def main():
   N = int(stdin.readline().strip())
   titles = stdin.readlines()
   freq = {}
   for t in titles:
      if t[0] not in freq:
         freq[ t[0] ] = 1
      else:
         freq[ t[0] ] += 1
   res = 0
   for t, v in freq.iteritems():
      res += (v + 9)/10
   print res
   return 0

if __name__ == '__main__': main()
