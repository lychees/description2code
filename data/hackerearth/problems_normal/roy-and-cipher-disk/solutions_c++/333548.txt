#include<iostream>
using namespace std;

int main()
{
    int T,len,i,cpos,rpos,a,b;
    cin>>T;
    while(T--)
    {
        string msg;
        cin>>msg;

        len = msg.length();
        cpos = rpos = a= b = 0;

        for(i=0; i<len; i++)
        {
            rpos = msg[i]-'a';

            if(cpos==rpos)
            {
                cout<<"0"<<" ";
                cpos = rpos;
                continue;
            }
            else if(cpos<rpos)
            {
                a = rpos-cpos;
                b = rpos-cpos - 26;
            }
            else if(rpos<cpos)
            {
                b = rpos - cpos;
                a = rpos - cpos + 26;
            }

            if(a <= (b*-1))
                cout<<a<<" ";
            else
                cout<<b<<" ";

            cpos = rpos;
        }
        cout<<endl;
    }
}
