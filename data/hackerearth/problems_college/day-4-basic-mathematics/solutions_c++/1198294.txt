#include<iostream>
#include<math.h>
using namespace std;
int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
      cin>>n;
       n = ceil((sqrt(1+8*n)-1)/2);
       if(n%3==0)
        {
            cout<<"SL"<<endl;
        }
        else if(n%3==1)
        {
            cout<<"LB"<<endl;
        }
        else if(n%3==2)
        {
            cout<<"BS"<<endl;
        }
    }

 return 0;
}
