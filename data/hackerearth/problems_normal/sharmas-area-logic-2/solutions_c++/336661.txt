#include<iostream>
#include<cmath>
using namespace std;

int main()
{
    int t,n,x,y,maxx,maxy,minx,miny,ans,i;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	minx=100; maxx=-100;
    	miny=200; maxy=-200;
    	for(i=0;i<n;i++)
    	{
    		cin>>x>>y;
    		minx=min(minx,x);
    		maxx=max(maxx,x);
    		miny=min(miny,y);
    		maxy=max(maxy,y);
    	}
    	if(minx==maxx) ans=maxy-miny;
    	else if(miny==maxy) ans=maxx-minx;
    	else
    	{
    		ans=(maxx-minx)*(maxy-miny);
    	}
    	cout<<ans<<endl;
    }
    
    return 0;
}
