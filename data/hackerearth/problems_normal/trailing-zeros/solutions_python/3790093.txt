from math import factorial
#Compute the factorial
#Consicutively divide the value by 10 till reminder is not equal to Zero
#Keep track fo count and pront
 
count = 0
divide = 10
N = int(raw_input()) 

factValue = factorial(N)


reminder = factValue%divide

while (reminder ==0):
    count += 1
    factValue = factValue/10
    reminder = factValue%divide
