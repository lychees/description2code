#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
pair<long long,long long> A[1000005];
	long long B[1000005];
int main()
{
	A[1].first = 1;
	A[1].second = 1;
	B[1] = 2;
	for(int i  = 2;i <= 1000000;i++)
	{
		A[i].first = (A[i-1].first+A[i-1].second)%mod;
		A[i].second = A[i-1].first;
		B[i] = (B[i-1]+A[i].first+A[i].second)%mod;
	}
	int T;
	cin >> T;
	while(T--)
	{
		int N;
		cin >> N;
		cout << B[N] << endl;
	}
	return 0;
}
