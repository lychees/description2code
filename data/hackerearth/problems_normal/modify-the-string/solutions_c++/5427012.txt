#include <bits/stdc++.h>
using namespace std;

int main()
{
    char str[101],i;
    cin>>str;
    int l=strlen(str)-1;
    for(i=0;i<=l;i++)
    {
    	if(str[i]>=64 && str[i]<=90)
    	str[i]=str[i]+32;
    	
    	else if(str[i]>=97 && str[i]<=122)
    	str[i]=str[i]-32;
    	
    }
    cout<<str<<endl;
    return 0;
}
