#include<bits/stdc++.h>
using namespace std;
inline void inp(long long int &n) 
{
    	n=0;
	register long long int ch=getchar_unlocked();
	long long int sign=1;
    	while(ch<'0'||ch>'9')
	{
		if(ch=='-')
			sign=-1; 
		ch=getchar_unlocked();
	}
    	while(ch>='0'&&ch<='9')
            n=(n<<3)+(n<<1)+ch-'0',ch=getchar_unlocked();
    	n=n*sign;
}
long long int gcd(long long int a, long long int b)
{
    while(b) b ^= a ^= b ^= a %= b;
    return a;
}
int main()
{
	long long int t;
	inp(t);
	string str;
	while(t--)
	{
		cin>>str;
		int len=str.length();
		int ar[len];
		for(int i=0;i<len;++i)
			ar[i]=str[i]-'0';
		bool corr=true;
		int ans;
		for(int i=9;i>0;--i)
		{
			corr=true;
			ans=i;
			for(int j=0;j<len;++j)
			{
				if(ar[j]%i==0)
					continue;
				else
				{
					corr=false;
					break;
				}
			}
			if(corr)
				break;
		}
		for(int i=0;i<len;++i)
			printf("%d",ar[i]/ans);
		printf("\n");
	}
	return 0;
}
