#include <iostream>
using namespace std;

#define mod 1000000007

long long int func(long long int a, long long int b){
	long long int x,y;
	x = 1;
	y = 0;
	
	while(b){
		y = b%2;
		b>>=1;
		if(y) x = (x*a)%mod;
		a = (a*a)%mod;
	}
	return x;
}
	
int main()
{
    long long int t,m,i,b;
    
    b=2;
    
    scanf("%lld",&t);
    
    for(i=0;i<t;i++){
    	scanf("%lld",&m);
    	cout<<func(b,m)-1<<"\n";
    }
    
    return 0;
}
