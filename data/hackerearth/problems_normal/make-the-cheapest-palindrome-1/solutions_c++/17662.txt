#include <iostream>
#include<cstring>
using namespace std;
int min(int x,int y)
{
    return x>y?y:x;
}
int main()
{
int t;
cin>>t;
while(t-->0)
{
	string s;
	cin>>s;
	int l = s.length();
	int acost,bcost;
	cin>>acost;
	cin>>bcost;
	int totcost=0;
	char least;
    int mincost = min(acost,bcost);
    if (mincost==acost){least='a';}
    else least='b';
    int i,j;
	for(i=0,j=l-1;i<j;i++,j--)
	{
		if(s[i]=='a' && s[j]=='/'){totcost+=acost;s[j]='a';continue;}
		if(s[i]=='b' && s[j]=='/'){totcost+=bcost;s[j]='b';continue;}
		if(s[j]=='a' && s[i]=='/'){totcost+=acost;s[i]='a';continue;}
		if(s[j]=='b' && s[i]=='/'){totcost+=bcost;s[i]='b';continue;}
		if(s[i]=='/' && s[j]=='/'){totcost+=mincost+mincost;s[i]=least;s[j]=least;continue;}
		
	}
	if(i==j && s[i]=='/'){s[i]=least;totcost+=mincost;}
    for(i=0,j=l-1;i<=j;i++,j--)if(s[i]!=s[j])totcost=-1;
	cout<<totcost<<endl;
}
return 0;
}
