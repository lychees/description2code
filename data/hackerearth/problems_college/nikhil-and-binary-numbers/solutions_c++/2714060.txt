#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(;t>0;t--)
    {
    	int n,x,s=0,s1=0;
    	cin>>n;
    	for(;n>0;n--)
    	{
    		cin>>x;
    		if(x==1)
    		s++;
    		else
    		s1++;
    	}
    	
    	if(s%2==0)
    	cout<<s1<<"\n";
    	else
    	cout<<s<<"\n";
    }
    return 0;
}
