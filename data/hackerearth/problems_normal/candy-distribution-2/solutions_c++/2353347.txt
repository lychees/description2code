#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--){
   	long long int n;
   	cin>>n;
   	long long int ans=0;
   	long long int i;
   	for(i=0;i<n;i++){
   		long long int a;
   		cin>>a;
   		ans=ans+a;
   	}
   	ans++;
   	ans=ans-n;
   	cout<<ans<<endl;
   }
}
