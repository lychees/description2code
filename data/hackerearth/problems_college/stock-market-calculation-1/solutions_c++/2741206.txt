#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,ar[100000],sell[100000],profit=0,i;
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>ar[i];
		}
		for(i=n-1;i>=0;i--)
		{
		if(i==n-1)
         sell[i]=ar[i];
          else
         sell[i]=max(sell[i+1],ar[i]);
        }
        for(int i=0;i<n;++i)
        {
        profit+=(sell[i]-ar[i]);
        }
         cout<<profit<<endl;
          
		
	}
    
}
