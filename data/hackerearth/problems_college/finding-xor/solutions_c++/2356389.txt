#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s1,s2;
    	cin>>s1>>s2;
    	int n =s1.length();
    	for(int i=0;i<n;i++)
    	{
    		int x1 = s1[i]-'0';
    		int x2 = s2[i]-'0';
    		cout<<(x1^x2);
    	}
    	cout<<endl;
    }
    return 0;
}
