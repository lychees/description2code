#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >>t;
    string arr[t];
    for(int i=0;i<t;i++) cin >> arr[i];
    for(int i=0; i<t;i++){
    	string s = arr[i];
    	bool notM = false;
    	for(int j=0; j<s.size() && !notM ;j++){
    		if(s[j] != s[s.size()-j-1]) notM = true;
    		if(s[j] == '2' ||s[j] == '3' ||s[j] == '4' ||s[j] == '5' ||s[j] =='7'){
    			notM=true;
    			break;
    		}
    	}
    	if(notM) cout << "NO";
    	else cout << "YES";
    	cout << endl;
    }
    return 0;
}
