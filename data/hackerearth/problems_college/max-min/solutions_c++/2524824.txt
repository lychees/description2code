#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long n,k,a[100004],minx=1000000004;
    cin >>n >>k;
    for(int i=0;i<n;i++)
     cin >>a[i];
     sort(a,a+n);
    for(int i=0;i<n;i++)
    {
    
    		if((i+k-1)<n)
    		minx=min((a[i+k-1]-a[i]),minx);
    	
    }
    cout << minx <<endl;
    return 0;
}
