#include <iostream>
using namespace std;

int main()
{
	long values[44];
	values[0]=0;
	long dopow = 2;
	for(int i=1;i<44;i++)
	{
		values[i]=values[i-1]+(i*dopow);
		dopow*=2;
	}
	int t;
	cin>>t;
	while(t--)
	{
		long n;
		cin>>n;
		int i=43;
		for(;i>=0;i--)
		{
			if(n>values[i])
			{
				n-=values[i];
				break;
			}
		}
		i++;	//i length character waaleya ch hai
		n--;
		long p=n/i;
		long q=n%i;
		//cout<<i<<" "<<p+1<<" "<<q+1<<endl;
		long divisor=1;
		long c=i-q-1;
		//cout<<c<<endl;
		while(c--)
			divisor*=2;
		//cout<<"divisor "<<divisor<<endl;
		//cout<<"p "<<p<<endl;
		p/=divisor;
		//cout<<"p "<<p<<endl;
		if(p%2==0)
			cout<<"Hacker"<<endl;
		else
			cout<<"Earth"<<endl;
	}
    return 0;
}
