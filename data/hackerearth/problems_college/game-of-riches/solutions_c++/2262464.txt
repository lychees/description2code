#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int nr,ng,i, count = 0;
   	cin>>nr>>ng;
   	int wealth[nr];
   	for(i = 0; i < nr; i++){
   		cin>>wealth[i];
   	}
   	int my = wealth[0];
   	sort(wealth,wealth + nr, greater<int>());
   	for(i = 0; wealth[i] != my; i++);
   	int rem = i%ng;
   	for(i = 0; wealth[i] != my; i++) {
   		if(i%ng == rem) {
   			count++;
   		}
   	}
   	cout<<count<<endl;
   }
    return 0;
}
