#include<bits/stdc++.h>
using namespace std;
int c,n1,n5,n10;
map<int,int> m[151][101][51];
int rec(int buy, int one, int five, int ten){
	if(!buy) return 0;
	int val=9999999;
	if(m[buy][five][ten].find(one)!=m[buy][five][ten].end()){
		return m[buy][five][ten][one];
	}
	if(ten>=1) val=min(val,1+rec(buy-1,one+2,five,ten-1));
	if(five>=2) val=min(val,2+rec(buy-1,one+2,five-2,ten));
	if(five>=1 && one>=3) val=min(val,4+rec(buy-1,one-3,five-1,ten));
	if(one>=8) val=min(val,8+rec(buy-1,one-8,five,ten));
	if(one>=3 && ten>=1) val=min(val,4+rec(buy-1,one-3,five+1,ten-1));
	return m[buy][five][ten][one]=val;
}
int main(){
    int t;
    cin>>t;
    while(t--){
    	for(int i=0; i<151; i++){
    		for(int j=0; j<101; j++){
    			for(int k=0; k<51; k++) m[i][j][j].clear();
    		}
    	}
    	cin>>c>>n1>>n5>>n10;
    	int val=rec(c,n1,n5,n10);
    	if(val>=9999999) cout<<-1<<endl;
    	else cout<<val<<endl;
//    	cout<<rec(c,n1,n5,n10)<<endl;
    }
}
