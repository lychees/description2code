#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

long long num=1000000000000LL;

long long convert(string str){
    long long ans=0;
    int len=str.length();
    for(int i=0;i<len;i++){
        ans=10*ans+(int)str.at(i)-48;
    }
    return ans;
}

int main(){
    int n;
    scanf("%d",&n);
    while(n--){
        long long ans=-1;
        string str;
        cin>>str;
        int len=str.length();
        for(int i=1;i<len;i++){
            for(int j=i+1;j<len;j++){
                for(int k=j+1;k<len;k++){
                    if(str.at(0)=='0'&&i!=1)
                        continue;
                    if(str.at(i)=='0'&&j-i!=1)
                        continue;
                    if(str.at(j)=='0'&&k-j!=1)
                        continue;
                    if(str.at(k)=='0'&&len-k!=1)
                        continue;
                    if(i>13||j-i>13||k-j>13||len-k>13)
                        continue;
                    long long a=convert(str.substr(0,i));
                    long long b=convert(str.substr(i,j-i));
                    long long c=convert(str.substr(j,k-j));
                    long long d=convert(str.substr(k,len-k));
                    if(a>num||b>num||c>num||d>num)
                        continue;
                    long long temp=a+b+c+d;
                    if(temp>ans){
                        ans=temp;
                    }
                }
            }
        }
        if(ans==-1)
            printf("unlucky\n");
        else
            printf("%lld\n",ans);
    }
}
