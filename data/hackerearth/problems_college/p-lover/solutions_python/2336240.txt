def is_prime(x):
    z=int(x**0.5)+3
    for i in range(2,z):
        if x%i==0:
            return False
    return True
def is_palindrome(s):
    n=len(s)
    s1=""
    for i in range(n-1,-1,-1):
        s1+=s[i]
    return s==s1
