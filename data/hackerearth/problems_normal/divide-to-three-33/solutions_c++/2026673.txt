#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{	
	int n;
	scanf("%d",&n);
	int i,j,p=1;
	int b[n],a[n];
	for(i=0;i<n;i++)
	{
		scanf("%d",(b+i));
		p*=3;
		a[i]=0;
	}
	int s[3];
	int mymin=INT_MAX;
	for(i=0;i<p;i++)
	{
		s[0]=0;s[1]=0;s[2]=0;
		for(j=(n-1);j>=0;j--)
		{
			a[j]=(a[j]+1)%3;
			if(a[j])
			break;
		}
		for(j=0;j<n;j++)
		{
			if(a[j]==0)
			{
				s[0]+=b[j];
			}
			else if(a[j]==1)
			{
				s[1]+=b[j];
			}
			else
			{
				s[2]+=b[j];
			}
		}
		sort(s,s+3);
		//~ cout<<s[0]<<" "<<a[1]<<" "<<s[2]<<endl;
		if(s[2]<mymin)
		{
			mymin=s[2];
		}
	}
	printf("%d\n",mymin);
}
