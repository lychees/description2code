#include <iostream>
#include <stdio.h>
#include<algorithm>
using namespace std;

int a[200005];
int main()
{
    int t,n,i,s,ans;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	for(i=0;i<(2*n);i++)
    		scanf("%d",&a[i]);
    	sort(a,a+(2*n));
    	
    	ans=0;
    	for(i=0;i<n;i++)
    	{
    		s=a[i]+a[(2*n)-i-1];
    		if(s>ans)
    			ans=s;
    	}
    	printf("%d\n",ans);
    	
    }
    
    return 0;
}
