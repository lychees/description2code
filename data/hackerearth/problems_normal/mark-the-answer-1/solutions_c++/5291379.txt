#include <iostream>
using namespace std;

int main()
{
   int n, x, temp, flag = 0;
   cin>>n>>x;
   int ans = 0;
   for (int i=0; i<n; i++)
   {
   	cin>>temp;
   	if(flag > 1)
   		break;
   	if(temp > x)
   		{
   			flag += 1;
   		}
   else
   	{
   		ans+= 1;
   	}
   }
   cout<<ans<<endl;
   
    return 0;
}
