#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,i,j,x,y,temp,temp1;
	cin>>t;
	int arr[t];
	int pos[t];
	for(i=0;i<t;i++)
	{
		cin>>x>>y;
		arr[i]=x+y;
		pos[i]=i+1;
	}
	for(i=0;i<t;i++)
	{
		for(j=0;j<t-1-i;j++)
		{
			if(arr[j]>arr[j+1])
			{
				temp=arr[j];
				arr[j]=arr[j+1];
				arr[j+1]=temp;
				temp1=pos[j];
				pos[j]=pos[j+1];
				pos[j+1]=temp1;
			}
		}
	}
	for(i=0;i<t;i++)
	cout<<pos[i]<<" ";
}
