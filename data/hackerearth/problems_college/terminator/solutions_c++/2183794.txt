#include <iostream>
#include <string>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
	{
		string s;
		cin >> s;
		int l = s.length();
		int a[105];
		int i,j,k=0;
		for(i=0;i<l;i++)
		{
			a[i] = 1;
		}
		for(i=0;i<l;i++)
		{
			if(s[i]=='x')
			{
				for(j=i+1;j<l;j++)
				{
					if(s[j]=='*'||s[j]=='x')
					{
						break;
					}
					else if(s[j]=='o')
					{
						a[j] = -1;
					}
				}
				for(j=i-1;j>=0;j--)
				{
					if(s[j]=='*'||s[j]=='x')
					{
						break;
					}
					else if(s[j]=='o')
					{
						a[j] = -1;
					}
				}
			}
		}
		for(i=0;i<l;i++)
		{
			if(a[i]==1)
			{
				printf("%c",s[i]);
			}
		}
		printf("\n");
	}
    return 0;
}
