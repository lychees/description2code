#include <bits/stdc++.h>
using namespace std;

const int MAXN = int(1e6) + 5;

#define rep(i, a, b) for (int i = (a); i < (b); i++)
#define st first
#define id second

typedef pair<int, int> ii;

int n, m;
ii a[MAXN];
int b[MAXN];

int bs(int val) {
    int res(-1);
    int l = 0, h = m-1;
    while (l <= h) {
        int mi = (h-l)/2 + l;
        if (val >= b[mi]) res = mi, l = mi+1;
        else h = mi - 1;
    }

    return res;
}

int main() {
//    freopen("input.inp", "r", stdin);

    int tst; scanf("%d", &tst);
    for (int t = 1; t <= tst; t++) {
        scanf("%d%d", &n, &m);
        rep(i, 0, n) scanf("%d", &a[i].st), a[i].id = i;
        rep(i, 0, m) {
            scanf("%d", &b[i]);
            if (i) b[i] = max(b[i], b[i-1]);
        }

        sort(a, a + n);

        int res(0), mx(-1);
        rep(i, 0, n) {
            int idx = bs(a[i].st);

            if (idx != -1) {
                if (mx < idx) mx = idx, res = a[i].id;
                else if (mx == idx) res = min(res, a[i].id);
            }
        }

        printf("%d\n", res);
    }
}
