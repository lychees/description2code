#include<iostream>
#define MAX 1000005
#define M 1000000007
using namespace std;

long long a[MAX];

void sieve()
{
    int i,j;
    for(i=2;i*i<MAX;i++)
    {
        if(!a[i])
        {

            for(j=i*i;j<=MAX;j+=i)
            {
                if(!a[j])
                a[j]=i;
            }
        }

    }

    for(i=2;i<=MAX;i++)
    {
        a[i]=(a[i]+a[i-1])%M;
    }
}
int main()
{
    int t,l,r;

    cin>>t;
    sieve();
    while(t--)
    {
        cin>>l>>r;

        cout<<(a[r]-a[l-1]+M)%M<<'\n';
    }
    return 0;
}
