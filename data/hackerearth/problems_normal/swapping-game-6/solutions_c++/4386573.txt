#include <iostream>
#include <algorithm>
using namespace std;

int main(){

long long int k,i,j,n;
cin>>k;
string a,d;
cin>>a;
d = a;
n = a.size();
//k = k%(n-1);
for(i=0;i<k;i++){
//cout<<i<<endl;
string b,c;
for(j=0;j<n;j=j+2){
    b = b + a[j];
}
for(j=1;j<n;j=j+2){
    c= c + a[j];
}
reverse(c.begin() , c.end());
a = b + c;
//cout<<a<<endl;
b.clear();
c.clear();
if(a==d){
    k = k%(i+1);
 //   cout<<k<<endl;
    i = -1;
}
else continue;
}

cout<<a<<endl;
return 0;
}
