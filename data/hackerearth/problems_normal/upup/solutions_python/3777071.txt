data = raw_input()
l = len(data)
result = ''
nextWord = True
for i in range(0, l):
    if data[i] == ' ':
        nextWord = True
        result += data[i]
    elif data[i] != ' ' and nextWord:
        result += data[i].upper()
        nextWord = False
    else:
        result += data[i]

print result
