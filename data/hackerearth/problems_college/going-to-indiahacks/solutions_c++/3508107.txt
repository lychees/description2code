#include <bits/stdc++.h>
#define ll  long long int
#define N   1004
#define M   1000000007
using namespace std;

ll dp[1005][1005];

int main()
{
    int t, a, b, c, d;
    cin >> t;

    for (int i = 0; i <= N; i++) {
        dp[i][0] = 1;
        dp[0][i] = 1;
    }
    for (int i = 1; i <= N; i++) {
        for (int j = 1; j <= N; j++) {
            dp[i][j] = (dp[i - 1][j] % M + dp[i][j - 1] % M + dp[i - 1][j - 1] % M) % M;
        }
    }

    while (t--) {
        cin >> a >> b >> c >> d;
        cout << dp[c - a][d - b] << endl;
    }

    return 0;
}
