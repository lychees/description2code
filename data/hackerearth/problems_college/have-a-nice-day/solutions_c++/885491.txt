#include <iostream>
#include<algorithm>
using namespace std;
typedef long long ll;
int main()
{
   int t,i,x;
   cin>>t;
   while(t--)
   {
   	cin>>x;
   	ll s=0;
   	int a[x],b[x];
   	for(i=0;i<x;i++)
   	{
   		cin>>a[i];
   	}
   		for(i=0;i<x;i++)
   	{
   		cin>>b[i];
   	}
   	sort(a,a+x);
   	sort(b,b+x);
   	for(i=0;i<x;i++)
   	{
   		s=s+a[i]*b[i];
   	}
   	cout<<s<<endl;
   }
    return 0;
}
