#include<iostream>
using namespace std;


int present(int n)
{
	int count=0;
 while(n>0)
{
if(n%10==7||n%10==4)
 count++;
n=n/10;
}
return count;
}

int main()
{
  int T,i=0;
 cin>>T;
if(T>=1&&T<=100)
 {
  for(i=0;i<T;i++)
  {
 int N;
    int count=0;
    cin>>N;
 if(N>=1&&N<=250)
 {
   int a[1000],i,j,temp,m,x;
                a[0]=1;
              m=1;
              temp = 0;
              for(i=1;i<=N;i++)
              {
                               for(j=0;j<m;j++)
                               {
                                               x = a[j]*i+temp;
                                               a[j]=x%10;
                                               temp = x/10;
                               }
                               while(temp!=0)
                               { 
                                 a[m]=temp%10;
                                 temp = temp/10;
                                 m++;
                               }
              }
              for(i=m-1;i>=0;i--)
                    count+=present(a[i]);

    cout<<count<<endl;
 }
}
}
return 0;
}
