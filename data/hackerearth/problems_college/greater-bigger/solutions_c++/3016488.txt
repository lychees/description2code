#include <iostream>
#include<string.h>
#include<algorithm>
using namespace std;
void swap(char *a,char *b){
	char t=*a;
	*a=*b;
	*b=t;
}
bool cmp(char a,char b){
	return a<b;
}
int main()
{
    long long int t;
    int len,i,j,flag=0;
    cin>>t;
    char a[100000];
for(;t>0;--t,flag=0){
	cin>>a;
	len=strlen(a);
	for(i=len-2;i>=0;--i)
		if(a[i]<a[i+1]){
			flag=1;
			break;
		}
	if(flag==0){
		cout<<"no answer\n";
		continue;
	}
	for(j=len-1;j>i;--j)
		if(a[j]>a[i])
			break;
	swap(a[i],a[j]);
	
	sort(a+i+1,a+len,cmp);
	
	cout<<a<<endl;
}
    
    return 0;
}
