#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    
    	int a[n];
    	int sum=0;
    	
    for(int i=0;i<n;i++) {cin>>a[i]; sum^=a[i]; }
    int c=0;
    for(int i=0;i<n;i++)
    {
    	if(sum^a[i]==0) c++;
    }
    
    	cout<<c<<endl;
    }
    return 0;
}
