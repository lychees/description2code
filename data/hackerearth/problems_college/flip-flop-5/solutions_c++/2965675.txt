#include <iostream>
#include<string.h>
using namespace std;

int main()
{   int n;
    cin>>n;
    while(n--)
    {   string s;
        cin>>s;
        int l,z=0,i;
        l=s.length();
        for(i=1;i<l;++i)
            if(s[i]==s[i-1])
                ++z;
        cout<<z<<"\n";
    }
    return 0;
}
