#include<bits/stdc++.h>
using namespace std;
long long int dp[50+1][450+1];
long long int answer(int start,int n,int *arr,int val)
{

	if(dp[start][val]==-1)
	{
		dp[start][val]=0;
		int sum=0;
		int i;
		for(i=start;i<n;i++)
		{
			if(sum>=val)
				break;
			sum+=arr[i];
		}

		if(sum>=val)
		{
			 if(i==start)
			 {
				 sum+=arr[i];
				 i++;
			 }
			 for(int j=i;j<n;j++)
			 {

                 dp[start][val]+=answer(j,n,arr,sum);
                 sum+=arr[j];

			 }
			 dp[start][val]+=1;
		}
		else
		{
            dp[start][val]=0;
		}
	}
	//cout<<start<<" "<<val<<" "<<dp[start][val]<<endl;
	return dp[start][val];
}
int main()
{
	int t;
	scanf("%d\n",&t);
	while(t--)
	{
		memset(dp,-1,sizeof(dp));
		string str;
		cin>>str;
		int arr[60];
		for(int i=0;i<str.length();i++)
		{
			arr[i]=str[i]-'0';
		}
		cout<<answer(0,str.length(),arr,0)<<endl;
	}
	return 0;
}
