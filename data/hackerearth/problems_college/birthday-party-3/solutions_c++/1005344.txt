#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
const int md=179424673;
int dp[1005][1005][2];
int main() {
    int t;
    scanf("%d",&t);
    while(t--) {
        int n1,n2,k1,k2;
        scanf("%d%d%d%d",&n1,&n2,&k1,&k2);
        memset(dp,0,sizeof(dp));
        dp[0][0][0]=1;
        dp[0][0][1]=1;
        for(int i=0;i<=n1;i++) {
            for(int j=0;j<=n2;j++) {
                for(int k=1;k<=k1&&k<=i;k++) dp[i][j][0]=(dp[i][j][0]+dp[i-k][j][1])%md;
                for(int k=1;k<=k2&&k<=j;k++) dp[i][j][1]=(dp[i][j][1]+dp[i][j-k][0])%md;
            }
        }
        printf("%d\012",(dp[n1][n2][0]+dp[n1][n2][1])%md);
    }
    return 0;
}
