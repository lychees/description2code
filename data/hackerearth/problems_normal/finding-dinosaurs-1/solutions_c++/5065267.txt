#include <iostream>
#include<vector>
#include<map>
#include<string>
using namespace std;

map<vector<int>,string> mp;

int main()
{
    int n,k,q,i,j,l;
    string s;
    vector<int> v;
    cin>>n>>k>>l>>q;
    while(n--)
    {
    	cin>>s;
    	v.clear();
    	for(i=0;i<k;i++)
    	{
    		scanf("%d",&j);
    		v.push_back(j);
    	}
    	mp[v]=s;
    }
    while(q--)
    {
    	v.clear();
    	for(i=0;i<k;i++)
    	{
    		scanf("%d",&j);
    		v.push_back(j);
    	}
    	if(mp.find(v)!=mp.end())
    		cout<<mp[v]<<endl;
    	else
    		cout<<"You cant fool me :P"<<endl;
    }
    return 0;
}
