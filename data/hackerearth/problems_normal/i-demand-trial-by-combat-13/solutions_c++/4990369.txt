#include <iostream>

using namespace std;

int main()
{
    int testcase;
    cin >> testcase;
    while (testcase--)
    {
        int numpeople, numhour;
        cin >> numpeople >> numhour;
        bool people[numpeople + 5];
        people[0] = 1;
        people[numpeople + 1] = 1;
        for (int i = 1; i <= numpeople; i++)
            cin >> people[i];

        for (int i = 0; i < numhour; i++)
        {
            bool temp = people[0];
            bool change = 0;
            for(int j = 1; j <= numpeople; j++)
            {
                if (temp == 1 && people [j + 1] == 1)
                {
                    change = 1;
                    temp = people[j];
                    people[j] = 1;
                }
                else
                {
                    temp = people[j];
                    if (people[j]) change = 1;
                    people[j] = 0;
                }
            }
            if (!change) break;
        }
        cout << people[1];
        for (int i = 2; i <= numpeople; i++)
            cout << " " << people[i];
        cout << endl;
    }
}
