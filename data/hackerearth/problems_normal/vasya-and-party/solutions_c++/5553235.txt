#include <bits/stdc++.h>
#define mod 1000000007
#define MAX 100005
#define ll long long
#define sc(a) scanf("%d",&a)
#define pf(a) printf("%d", a)
using namespace std;
ll a[MAX]={};
int vis[MAX]={};
vector <int> gr[MAX];
ll y=0,ma=0;
void dfs(int u)
{
	vis[u]=1;
//	cout<<ct<<"\t";
	if(a[u]==ma)
	y++;
	else if(a[u]>ma)
	{ma=a[u];y=1;}
	for(int i=0;i<gr[u].size();i++)
	{//	cout<<u<<" "<<ma<<" "<<y<<"\n";
		int v=gr[u][i];
		if(vis[v]==0)
		dfs(v);
	}
//	cout<<ct<<"\t";
//	return ct;
}
int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=1;i<=n;i++)
	cin>>a[i];
	for(int i=0;i<m;i++)
	{
		int u,v;
		cin>>u>>v;
		gr[u].push_back(v);
		gr[v].push_back(u);
	}
	ll ans=1;
	for(int i=1;i<=n;i++)
	{	y=0;
		if(vis[i]==0)
		{ma=0;
		dfs(i);
		ans=(ans*y)%mod;
		}
	}
	cout<<ans;
}
