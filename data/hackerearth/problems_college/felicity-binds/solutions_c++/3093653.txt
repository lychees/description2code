    #include <iostream>
    #include <algorithm>
    using namespace std;
     
    int main()
    {
        int t,n,i,max;
        int *b;
        int *g;
        cin>>t;
        while(t--){
        	cin>>n;
        	b = (int *)malloc(sizeof(int)*n);
        	g = (int *)malloc(sizeof(int)*n);
        	
        	for(i=0;i<n;i++){
        		cin>>b[i];
        	}
        	
        	for(i=0;i<n;i++){
        		cin>>g[i];
        	}
        	
        	max = 0;
        	sort(b,b+n);
        	sort(g,g+n);
        	for(i=0;i<n;i++){
        		max+=(b[i]*g[i]);
        	}
        	
        	cout<<max<<endl;
        }
        return 0;
    }
