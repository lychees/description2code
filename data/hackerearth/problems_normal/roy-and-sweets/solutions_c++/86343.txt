#include <iostream>
#include <algorithm>
#include <vector>
#include <stdio.h>
using namespace std;

struct sweet
{
    int sweetness;
    int quantity;
};

bool sortByQuantity(const sweet &lhs, const sweet &rhs)
{
    return lhs.quantity > rhs.quantity;
}

int main()
{
    sweet *sv = new sweet[1000008];
    int* guests = new int[1000008];
    int nos, nog;
    scanf("%d", &nos);

    for(int i=0; i<nos; i++)
    {
        scanf("%d %d", &sv[i].sweetness, &sv[i].quantity);
    }

    scanf("%d", &nog);

    for(int i=0; i<nog; i++)
        scanf("%d", &guests[i]);

    sort(sv,sv+nos,sortByQuantity);
    long long  cost = 0;
    for(int i=0; i<nog; i++)
    {
        for(int j=0; j<nos; j++)
        {
            if(sv[j].sweetness >= guests[i])
            {
                cost = cost + 100*(sv[j].quantity/guests[i]);
                break;
                //cout<<"cost : "<<cost<<"  sv[j].quantity "<<sv[j].quantity<<endl;
            }

        }
    }
    delete [] sv;
    delete [] guests;
    printf("%lld\n",cost);

}
