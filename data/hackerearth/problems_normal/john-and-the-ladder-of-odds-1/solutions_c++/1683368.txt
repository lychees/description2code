    if(!b) return a;
    return gcd(b, a % b);
}
template<typename T> T lcm(T a, T b) {
    return a * b / gcd(a, b);

class matrix
{
      public:
      ll a[2][2];
      matrix(){}
      matrix(ll x[2][2])
      {
                for (int i=0;i<2;i++)
                    for (int j=0;j<2;j++)
                        a[i][j]=x[i][j];
      }
      matrix operator *(matrix x)            // Multiplying matrix 
      {
             matrix z;
             for (int i=0;i<2;i++)
                 for (int j=0;j<2;j++)
                 {
                     z.a[i][j]=0;
                     for (int k=0;k<2;k++)
                     {
                         ll temp=a[i][k]*x.a[k][j];
                         temp%=MOD;
                         z.a[i][j]=(z.a[i][j]+temp)%MOD;
                     }
                 }
             return z;
      }
};
matrix solve(matrix M,long long int n)
{
   if (n==1)
      return M;
   else
   {
       matrix R=solve(M,n/2);
       R=R*R;
       if (n%2) R=R*M;
       return R;
   }
}
