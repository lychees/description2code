#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	
    	for(int j=0;j<n;j++)
    	{
    		cin>>a[j];
    	}
    	int k=0;
    	int p=0;
    	while(a[k]!=-1 && p<n)
    	{
    		k=a[k];
    		++p;
    	}
    	
    	if(a[k]==-1)
    	{	cout<<"Win"<<endl;}
   		else
   		{	cout<<"Lose"<<endl;}
    	
    }
    
    
    
    
    return 0;
}
