#include <bits/stdc++.h>

using namespace std;

void prime_factor(int n,int m)
{
	n = pow(2,n)-1;
	m = pow(2,m)-1;
	int div1 = n;
	int div2 = m;
	set<int> s;
	int i,j;
	int a[100000];
	int k = 0;
	//int m = n;
	int n1 = m;
	while (n % 2 == 0) {
		a[k++] = 2;
		n = n/2;
	}
	for (i = 3; i <= sqrt(n); i = i + 2) {
		while (n % i == 0) {
			a[k++] = i;
			n = n/i;
		}
	}
	if (n > 2) {
		a[k++] = n;
	}
	while (n1 % 2 == 0) {
		a[k++] = 2;
		n1 = n1/2;
	}
	for (i = 3; i <= sqrt(n1); i = i + 2) {
		while (n1 % i == 0) {
			a[k++] = i;
			n1 = n1/i;
		}
	}
	if (n1 > 2) {
		a[k++] = n1;
	}
	for (i = 0; i < k; i++) {
		s.insert(a[i]);
	}
	int ans = 0;
	for (set<int> :: iterator i = s.begin(); i != s.end(); i++) {
		if (div1 % *i == 0 && div2 % *i == 0) {
			ans++;
		}
	}
	printf("%d\n",ans);
	s.clear();
}

int main()
{
	int t,a,b,ans;
	scanf("%d",&t);
	while (t --) {
		scanf("%d %d",&a,&b);
		prime_factor(a,b);
	}
	
	return 0;
}
