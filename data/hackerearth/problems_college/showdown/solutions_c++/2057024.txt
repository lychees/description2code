int karate(int a[],int b[],int n)
{
    sort(a,a+n);
    sort(b,b+n);
    int ans=0;
    for(int i=n-1;i>=0;i--)
    {
        for(int j=0;j<n;j++)
        {
            if(a[j]!=0 && a[j]>b[i])
            {
                ans+=100;
                a[j]=0;
                b[i]=0;
                break;
            }
        }
    }
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            if(a[i]!=0 && a[i]==b[j])
            {
                ans+=50;
                a[i]=0;b[j]=0;
            }
        }
    }
    return ans;

}
