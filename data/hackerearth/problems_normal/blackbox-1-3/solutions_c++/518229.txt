#include <iostream>
using namespace std;
long int gcd(long int a, long int b)
{
	if(a>b && a!=0 && b!=0)
	{
		a=a%b;
		if(a==0)
		return b;
		else
		gcd(b,a);
	}
	else if(a<b && a!=0 && b!=0)
	gcd(b,a);
	else return -1;
}
int main()
{
    int n;
    cin>>n;
    int i=0;
    string yes="YES";
    string no="NO";
    while(i<n)
    {
    	long int a,b,c;
    	cin>>a;
    	cin>>b;
    	cin>>c;
    	if(((a*a+b*b==c*c)||(a*a+c*c==b*b)||(b*b+c*c==a*a)) && gcd(a,b)==1 && a>0 && b>0 && c>0)
    	cout<<yes<<endl;
    	else
    	cout<<no<<endl;
    	i++;
    }
    return 0;
}
