#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <fstream>
#include <iostream>
using namespace std;

int c[101];
int main() {
  freopen("input.txt", "r", stdin);
  freopen("output.txt", "w", stdout);
  int n, X;
  scanf("%d%d", &n, &X);
  for (int i = 1; i <= n; i++) {
    int C;
    scanf("%d", &C);
    c[i] = C * (n + 1 - i);
  }
  sort(c + 1, c + n + 1);
  int i;
  for (i = 1; i <= n; i++) {
    if (X - c[i] < 0)
      break;
    else
      X -= c[i];
  }
  printf("%d\n", i - 1);
  return 0;
}
