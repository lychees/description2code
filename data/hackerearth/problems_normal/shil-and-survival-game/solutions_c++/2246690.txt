#include <bits/stdc++.h>
using namespace std;
int arr[100005];
bool stat[100005]={false};
int main()
{	int num;
    scanf("%d",&num);
    priority_queue <int,vector <int> > q1,q2;
    for(int i=0;i<num;i++)
    	scanf("%d",&arr[i]);
    for(int i=0;i<num;i++){
    	q1.push(arr[i]);
    	if(q1.top()==arr[i])
    		stat[i]=true;
    }
    for(int i=num-1;i>=0;i--){
    	q2.push(arr[i]);
    	if(q2.top()==arr[i])
    		stat[i]=true;
    }
    for(int i=0;i<num;i++)
    	if(stat[i])
    		printf("%d ",i+1);
    
    return 0;
}
