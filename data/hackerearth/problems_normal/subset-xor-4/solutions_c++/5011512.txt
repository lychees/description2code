#include <iostream>
using namespace std;

// Returns XOR of all XOR's of given subset
int findXOR(int Set[], int n)
{
    // XOR is 1 only when n is 1, else 0
    if (n == 1)
       return Set[0];
    else
       return 0;
}

// Driver program
int main()
{
    int t;
    int arr[100000];
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	for(int i=0;i<n;i++)
    	scanf("%d",&arr[i]);
    	cout<< findXOR(arr, n)<<endl;
    }
    return 0;
}
