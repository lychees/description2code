#include <iostream>
using namespace std;

void merge(int a1[],int a2[],int n1,int n2)
{
    int i=0,j=0,k=0;
    int aux[n1+n2];
    
    while(i<n1 && j<n2)
    {
    	if(a1[i]>a2[j])
    	{
    		aux[k]=a1[i];
    		i++;
    	}
    	else
    	{
    		aux[k]=a2[j];
    		j++;
    	}
    	k++;
    }
    
    while(i<n1)
    {
    	aux[k]=a1[i];
    	i++;
    	k++;
    }
    
    while(j<n2)
    {
    	aux[k]=a2[j];
    	j++;
    	k++;
    }
    for(int i=0;i<k;i++)
    {
        cout<<aux[i]<<" ";
    }
}

int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
    	int n1,n2;
    	cin>>n1>>n2;
    	int i;
    	
    	int a1[n1],a2[n2];
    	
    	for(i=0;i<n1;i++)
    	{
    		cin>>a1[i];
    	}
    	
    	for(i=0;i<n2;i++)
    	{
    		cin>>a2[i];
    	}
    	
    	merge(a1,a2,n1,n2);
    	cout<<endl;
    }
    
    return 0;
}
