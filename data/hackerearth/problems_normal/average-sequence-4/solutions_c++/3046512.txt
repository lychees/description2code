#include <iostream>
#include "vector"
using namespace std;

int main()
{
    long long length=0,input=0;
    vector<long long> v1,v2,v3;
    cin>>length;
    for (long long i =0; i<length; i++) {
        cin>>input;
        v2.push_back(input);
    }
    v1.push_back(v2[0]);
    
    for (long long i =1,j=2; i<length; i++) {
        v1.push_back((j*v2[i])-((j-1)*v2[i-1]));
        j++;
    }
    
    for (long long i =0; i<length; i++) {
        cout<<v1[i]<<" ";
    }
    return 0;
}
