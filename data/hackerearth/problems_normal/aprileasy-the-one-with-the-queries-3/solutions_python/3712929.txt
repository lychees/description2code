string = raw_input()
times = int(raw_input())
array = []
for i in range(times):
    array.append(raw_input())

for i in range(times):
    init_index = 0
    
    indicate = True
    for char in array[i]:
        init_index = string.find(char, init_index)
        if init_index == -1:
            indicate = False

    if indicate:
        print "Yes"
    else:
        print "No"
