#include <bits/stdc++.h>
using namespace std;

int main()
{
	int s,q;
	cin >> s;
	cin >> q;
	map<int,int> ma;
	while(q--) {
		int a;
		cin >> a;
		int val = __gcd(s,a);
		if(ma[val] == 0) {
			cout << val << "\n";
			ma[val] = 1;
		}
		else
			cout << "-1\n";
	}
//    cout << "Hello World!" << endl;
    return 0;
}
