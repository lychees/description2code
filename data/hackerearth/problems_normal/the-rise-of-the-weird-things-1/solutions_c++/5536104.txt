void insertionSort(int *arr,int size){
	int j,tmp;
	for(int i=1;i<size;i++){
		j=i;
		tmp=arr[i];
		while(j>0 && tmp<arr[j-1]){
			arr[j]=arr[j-1];
			j--;
		}
		arr[j]=tmp;
	}
}

int main()
{
    int n;
    long odd=0,even=0;
    cin >> n;
    int a[n];
    for(int i=0;i<n;i++){
    	cin >> a[i];
    	if(a[i]%2==0){
    		even+=a[i];
    	} else
    		odd+=a[i];
    }
    insertionSort(a,n);
    for(int i=0;i<n;i++){
    	if(a[i]%2==0)
    		cout << a[i] << " ";
    }
    cout << even << " ";
    for(int i=0;i<n;i++){
    	if(a[i]%2!=0)
    		cout << a[i] << " ";
    }
    cout << odd;
    return 0;
}
