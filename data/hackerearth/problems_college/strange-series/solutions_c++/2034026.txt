#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[51];
int main()
{
	a[0]=1;
	ll int i;
	for(i=1;i<51;i++)
	{
		a[i]=a[i-1]*2;
	}
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int n;
		scanf("%lld",&n);
		ll int ans=a[n]-1;
		printf("%lld\n",ans);
	}
	return 0;
}
