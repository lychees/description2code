#include<iostream>
#include<algorithm>
#include<vector>
#include<cmath>
#define MAX 1000000
#define MOD 1000000007
using namespace std;
long long p[1000005];

void calc_prime()
{
  long long i,j;
  for(i=1;i<=MAX;i++) p[i]=1;
  p[1]=0,p[0]=0;
  for(i=2;i*i<=MAX;i++) {
    if(p[i]==1) {
      p[i]=0;
      for(j=i*i;j<=MAX;j+=i) if(p[j]==1) p[j]=i;
    }
  }
  for(i=1;i<=MAX;i++) if(p[i]==1) p[i]=0;
  for(i=1;i<=MAX;i++) {
    p[i]=(p[i]+p[i-1])%MOD;
  }
}

int main()
{
  int t;
  cin>>t;
  calc_prime();
  while(t--) {
    long long l,r,ans;
    cin>>l>>r;
    ans=p[r]-p[l-1];
    if(ans<0) ans=(ans+MOD);
    cout<<ans<<"\n";
  }
//  system("pause");
  return 0;
}
