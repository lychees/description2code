#include <iostream>
#include <algorithm>
#include <stdio.h>
#include <vector>
#include <queue>
#include <string.h>
using namespace std;
#define s(t) scanf("%d",&t)
#define ll long long int
#define mod 1000000007



int minimum = 100000000;

void recur(int * flag, char * a,char * b, int  x, int n, int cost)
{
	int i;
	if( x == n)
	{
		//printf("%s\n",a );
		int diff = cost;
		for( i = 0; i < n ; i++ )
		{
			int tmp =  (int)(a[i] - b[i]);
			if( tmp < 0)
				tmp *= -1;
			diff += tmp;
		}
		// printf("%s diff = %d\n",a,diff);
		
		if( diff < minimum)
			minimum = diff;

		return;
	}
	for( i = x; i < n ; i++ )
	{
		// swap
		char tmp = a[i];
		a[i] = a[x];
		a[x] = tmp;
		flag[x] = flag[i] = 1;

		if( i == x)
			recur(flag, a, b, x + 1, n ,cost);
		else 
			recur(flag, a, b, x + 1, n ,cost + 1);
			
		flag[x] = flag[i] = 0;

		tmp = a[i];
		a[i] = a[x];
		a[x] = tmp;	
	}

}

int main()
{
	int t;
	s(t);
	while(t--)
	{
		minimum = 100000000;
		char a[20], b[20];
		int flag[20], i;

		scanf("%s %s", a , b);
		int n = strlen(a);

		for( i = 0; i < n ; i++)
			flag[i] = 0;
		

		recur(flag, a, b, 0, n, 0); 

		printf("%d\n",minimum );
	}
}
