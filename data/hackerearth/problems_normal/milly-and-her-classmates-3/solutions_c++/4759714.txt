#include <iostream>
using namespace std;

int main()
{
    int T;
    cin >> T;
    for(int j = 0; j < T; ++j)
    {
    	int N;
    	cin >> N;
    	int *A = new int[N+1];
    	for(int i = 1; i <= N; ++i)
    	{
    		cin >> A[i];
    	}
    	int start = -1;
    	int end = -1;
    	bool ife = true;
    	bool switched = false;
    	for(int i = 2; i <= N; ++i)
    	{
    		if (A[i] < A[i-1])
    		{
    			if (switched)
    			{
    				ife = false;
    				break;
    			}else
    			{
    				if (start == -1)
    				{
    					start = i - 1;
    				}
    			}
    		}else
    		{
    			if (start >= 1 && (!switched))
    			{
    				end = i-1;
    				switched = true;
    				if ((A[start] > A[end+1]) || (start > 1 && A[start-1] > A[end]))
    				{
    					ife = false;
    					break;
    				}
    			}
    		}
    	}
    	
    	if (!ife)
    	{
    		cout <<-1<<" "<<-1<< endl;
    	}else
    	{
    		if (switched)
    		{
    			cout << start <<" " << end << endl;
    		}else
    		{
    			if (start == -1)
    			{
    				cout <<-1<<" "<<-1<< endl;
    			}else
    			{
    				if (start > 1 && A[start-1] > A[N])
    				{
    					cout <<-1<<" "<<-1<< endl;
    				}else
    				{
    					cout << start <<" "<<N<<endl;
    				}
    			}
    		}
    	}
    }
    return 0;
}
