#include <iostream>
#include <cmath>
using namespace std;
int gcd(int a,int b) {
	if(a%b==0)
	return b;
	else return gcd(b,a%b);
}
int main() {
	int t;
	cin>>t;
	while(t--) {
		int n,counter=0,a,b,c,i,j,k;
		cin>>n;
		for (i = 2; i <= (sqrt(n) + 1); i++)
     for (j = 1; j < i; j++)
          if ((gcd(i,j) == 1) && ((i-j) % 2) && (i*i + j*j) <= n)
          {
                k = 0;
                a = i * i - j * j;
                b = 2 * i * j;
                c = i * i + j * j;
                while ((++k) * c <= n)
                  counter++;
          }
          cout<<counter<<endl;
	}
	return 0;
}
