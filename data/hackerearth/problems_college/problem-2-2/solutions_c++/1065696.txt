#include<bits/stdc++.h>
using namespace std;
 
int main() 
{
	int temp ,balls,i,k,j,min,max;
	cin>>temp>>balls;
	if(temp==0)
		cout<<balls<<endl;
		if(balls==0 && temp!=0)
		cout<<temp<<endl;
		if(temp!=0 && balls!=0)
		{	max=temp>balls?temp:balls;
			min=temp>=balls?balls:temp;
			min++;
			i=max/min;
			if(max%min!=0)
			i++;
			cout<<i<<endl;
		}	
	return 0;
}
