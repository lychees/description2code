#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int fact(long long int );
int main()
{
          int t; scanf("%d",&t);
          while(t--)
          {
    long long int a,b,c,v,cc,d;
    scanf("%lld %lld",&c,&v);
    a=fact(7)/(fact(7-c)*fact(c));
    b=fact(5)/(fact(5-v)*fact(v));
    cc=a*b;
    d=fact(c+v);
    printf("%lld\n",cc*d);
          }
    return 0;
}

int fact(long long int c)
{
    long long int f=1;
    while(c>0)
    {
        f=f*c;
        c--;
    }
    return f;
}
