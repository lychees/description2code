#include<cstdio>
#include<iostream>
#include<cmath>
#include<cstdlib>
#include<cstring>
using namespace std;
int main()
{
		int n;
		cin>>n;
		int a[n];
		int i,m=0;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			if(m<a[i])
			{
				m = a[i];
			}
		}
		if((m-n)%2==0)
		{
			cout<<"Bob"<<"\n";
		}
		else
		{
			cout<<"Alice"<<"\n";
		}
}
