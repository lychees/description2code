#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll bm(ll b,ll p,ll mod)
{
    ll x=1,y=b;
    while(p>0)
    {
        if(p&1)
            x=(x*y)%mod;
        y=(y*y)%mod;
        p>>=1;
    }
    return x;
}
ll arr[1000009];
int main()
{
    ll n,x,res,t,i,mod=1000003;
    arr[1]=1;
    for(i=2; i<=1000003; i++)
        arr[i]=((arr[i-1])%mod*(i)%mod)%mod;
    cin>>t;
    while(t--)
    {
        cin>>x>>n;
        if(n>=1000003||x>=1000003)
        {
            cout<<0<<endl;
            continue;
        }
        res=(arr[x]*n)%mod;
        cout<<res<<endl;
    }
}
