#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    int m,n;
    cin>>m>>n;
    int a[m];
    for(ll i=0;i<m;i++)
    cin>>a[i];
    ll ans=0;
    priority_queue<ll>pq;
    for(ll i=0;i<m;i++)
    {
    	pq.push(a[i]);
    }
    for(ll i=0;i<n;i++)
    {
    	
    	ll temp=pq.top();
    	ans+=temp;
    	pq.pop();
    	pq.push(temp-1);
    }
    cout<<ans;
    return 0;
}
