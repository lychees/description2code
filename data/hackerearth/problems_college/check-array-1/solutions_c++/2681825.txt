#include <iostream>
using namespace std;

long long int a[1000000];

int main() {
	long long int t;
	cin>>t;
	while(t--){
	long long n,x,flag=0,i; 
	cin>>n;
	a[0]=0;
	for(i=1;i<=n;i++){
		cin>>x;
		a[i]=a[i-1]+x;
	}
	a[n+1]=a[n];
	for(i=1;i<n;i++){
		if(a[i-1]==(a[n+1]-a[i])){
			flag=1;
			break;
			}
	}
	if(flag==1)
		cout<<"YES\n";
	else
		cout<<"NO\n";
	}
	return 0;
}
