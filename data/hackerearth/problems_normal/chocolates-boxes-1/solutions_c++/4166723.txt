#include <iostream>
using namespace std;

int main()
{
    long long tc = 0;
  cin >> tc;
  long long hash[100005];
  while (tc--)
    {
      long long n = 0, k = 0, x = 0, ans = 0, halfk;
      cin >> n;
      for (long long i = 0; i < 100005; i++)
        {
          hash[i] = 0;
        }
      for (long long i = 0; i < n; i++)
        {
          cin >> x;
          hash[x]++;
        }
      cin >> k;
      halfk = (k % 2 == 0 ? k / 2 - 1 : k / 2);
      for (long long i = 1; i <= halfk; i++)
        {
          if (hash[i] == 0)
            continue;
          ans = ans + (hash[i] * hash[k - i]);

        }
      if (k % 2 == 0)
        {
          ans = ans + (hash[halfk + 1] - 1) * (hash[halfk + 1]) / 2;
        }
      cout << ans << endl;
    }


    return 0;
}
