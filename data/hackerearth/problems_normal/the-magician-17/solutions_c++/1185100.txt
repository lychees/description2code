#include <iostream>
using namespace std;

long compute(long n,long k)
{
	if(n%2==0)
	{
		if(k<=(n/2))
			return (2*k-1);
		else
			return 2*compute((n/2),k-(n/2));
	}
	else
	{
		if(k<=(n/2)+1)
			return (2*k-1);
		else
			return 2*(compute((n/2)+1,k-(n/2))-1);
	}
}

int main()
{
    int T;
    long n,k;
    
    cin>>T;
    
    for(int t=0;t<T;t++)
    {
    	cin>>n>>k;

		cout<<compute(n,k)<<endl;
    }
     
    return 0;
}
