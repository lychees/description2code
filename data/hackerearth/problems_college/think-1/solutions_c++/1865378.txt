#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t=0;
    scanf("%d",&t);
    while(t--)
    {
    	int n=0,n1=0,ns=0,ns1=0,a=0,b=0;
    	cin>>n;
    	a=n;
    	ns=n*n;
    	while(n)
    	{
    		int r=n%10;
    		n1=n1*10+r;
    		n/=10;
    	}
    	b=n1;
    	n1=n1*n1;
    	while(n1)
    	{
    		int r=n1%10;
    		ns1=ns1*10+r;
    		n1/=10;
    	}
    	if(ns==ns1 && a!=b)
    	cout<<b<<endl;
    	else
    	cout<<"No"<<endl;
    	
    	
    }
    return 0;
}
