#include <iostream>
using namespace std;
int gcd(int x,int y)
{	if(y<=1)
	return y;
	if(x-((x/y)*y)==0)
	return y;
	else
	return gcd(y,x-((x/y)*y));
}
int main()
{
    int n,k,i,j,flag,c;
    int a[500005],b[500005];
    scanf("%d%d",&n,&k);
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    flag=0;
    for(i=1;i<n;i++)
    {
    	b[i-1]=gcd(max(a[i-1],a[i]),min(a[i-1],a[i]));
    	if(b[i-1]>=k)
    	flag=1;
    }
    if(flag!=1)
    	cout<<"0"<<endl;
    else
    {
    	c=3;
    	
  		for(;c<=n;c++)
  		{	flag=0;
  			for(i=c-1,j=0;i<n && j<=n;i++,j++)
  			{	if(b[j]>=k)
  					b[j]=gcd(max(b[j],a[i]),min(b[j],a[i]));
  				else
  					continue;
  				if(b[j]>=k)
  				flag=1;
  			}
  			if(flag==0)
  			break;
  		}
  		cout<<c-1<<endl;
    }
    return 0;
}
