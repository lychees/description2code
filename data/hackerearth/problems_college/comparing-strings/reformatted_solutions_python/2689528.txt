n = int(raw_input())
for j in range(n):
    s = []
    for k in range(2):
        s.append(raw_input())
    count = 0
    for i in range(len(s[0])):
        for m in range(len(s[1])):
            if s[0][i] == s[1][m]:
                count += 1
    if count == 0:
        print "NO"
    else:
        print "YES"
