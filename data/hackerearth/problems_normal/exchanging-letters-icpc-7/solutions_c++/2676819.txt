#include <bits/stdc++.h>

using namespace std;

#define ll long long

const ll mod = 1000000007;


ll fact[2100], mpow[2100];

ll modpow(ll a, ll b)
{
	ll x=1ll, y=a;
	while(b){
		if(b%2)
			x=(x*y)%mod;
		y=(y*y)%mod;
		b/=2;
	}
	return x;
}

int main()
{
	ll i, j, n, k, t;
	fact[0] = 1ll;
	for(i=1; i<2100; i++){
		fact[i] = (fact[i-1]*i)%mod;
		//mpow[i] = (modpow(fact[i], mod-2));
	}
	ll x, y, z, sum=0, p, q, diff;
	scanf("%lld%lld%lld", &n, &k, &t);
	for(i=0; i<k; i++){
		scanf("%lld", &j);
		sum+=j;
	}
	diff = t-sum;
	//cout << t << " " << sum << " " << diff << "\n";
	ll ans = 0;
	for(i=0; i<=k; i++){
		p = i;
		q = p-diff;
		//cout << p << " " << q << " " << diff << "\n";
		if((p+q)!=k)
			continue;
		x = fact[k];
		y = (fact[i]*fact[k-i])%mod;
		z = modpow(y, mod-2);
		x = (x*z)%mod;
		ans = (ans+x)%mod;
	}
	printf("%lld\n", ans);
	return 0;
}
