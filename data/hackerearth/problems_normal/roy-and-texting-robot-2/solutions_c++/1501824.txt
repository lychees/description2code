#include<bits/stdc++.h>
using namespace std;
int main(){
  int t;
  cin >> t;
  map <char,pair<int,int> > m; 
  m['.'] = make_pair(1,1);
  m[','] = make_pair(2,1);
  m['?'] = make_pair(3,1);
  m['!'] = make_pair(4,1);
  m['1'] = make_pair(5,1);
  m['a'] = make_pair(1,2);
  m['b'] = make_pair(2,2);
  m['c'] = make_pair(3,2);
  m['2'] = make_pair(4,2);
  m['d'] = make_pair(1,3);
  m['e'] = make_pair(2,3);
  m['f'] = make_pair(3,3);
  m['3'] = make_pair(4,3);
  m['g'] = make_pair(1,4);
  m['h'] = make_pair(2,4);
  m['i'] = make_pair(3,4);
  m['4'] = make_pair(4,4);
  m['j'] = make_pair(1,5);
  m['k'] = make_pair(2,5);
  m['l'] = make_pair(3,5);
  m['5'] = make_pair(4,5);
  m['m'] = make_pair(1,6);
  m['n'] = make_pair(2,6);
  m['o'] = make_pair(3,6);
  m['6'] = make_pair(4,6);
  m['p'] = make_pair(1,7);
  m['q'] = make_pair(2,7);
  m['r'] = make_pair(3,7);
  m['s'] = make_pair(4,7);
  m['7'] = make_pair(5,7);
  m['t'] = make_pair(1,8);
  m['u'] = make_pair(2,8);
  m['v'] = make_pair(3,8);
  m['8'] = make_pair(4,8);
  m['w'] = make_pair(1,9);
  m['x'] = make_pair(2,9);
  m['y'] = make_pair(3,9);
  m['z'] = make_pair(4,9);
  m['9'] = make_pair(5,9);
  m['_'] = make_pair(1,0);
  m['0'] = make_pair(2,0);
  while(t--){
	string s;
	cin >> s;
	int prev=1;
	int ans=0;
	for(int i=0;i<s.size();i++){
		pair<int,int> p = m[s[i]];
		int no = p.first;
		int pos = p.second;
		if(pos!=prev)
			ans+=1;
		ans += no;
		prev = pos;
	}
	cout << ans << endl;
  }
  return 0;
}
