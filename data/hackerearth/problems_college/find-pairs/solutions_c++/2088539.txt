#include <cstdlib>
#include <iostream>

using namespace std;
long long i,j,k,n,I,T; 
int main(int argc, char *argv[])
{   cin>>T;
    for(I=1;I<=T;I++)
    {   cin>>n;
        long long a[n];
        for(i=1;i<=n;i++)
           cin>>a[i];
        k=0;
        for(i=1;i<=n;i++)
           for(j=i;j<=n;j++)
              if(a[i]==a[j])
                 k++;
        cout<<k<<endl;         
    }
   
    return EXIT_SUCCESS;
}
