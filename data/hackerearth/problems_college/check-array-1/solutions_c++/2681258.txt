#include<bits/stdc++.h>
using namespace std;
int main(){
    int T,N,counter;
    long long int sum,sum2;
    cin>>T;
    while(T--){
        cin>>N;
        counter=0;
        sum=0;
        vector<int>v(N);
        for(int i=0;i<N;i++){
            cin>>v[i];
            sum+=v[i];
        }
        sum-=v[0];
        sum2=0;
        for(int i=1;i<N;i++){
            if(sum2==sum){
                counter=1;
                break;
            }
            sum-=v[i];
            sum2+=v[i-1];
        }
        if(counter==1)
            cout<<"YES\n";
        else
            cout<<"NO\n";
    }
    return 0;
}
