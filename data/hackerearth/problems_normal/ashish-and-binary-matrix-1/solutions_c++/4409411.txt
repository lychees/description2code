#include <iostream>
#include <set>
using namespace std;

int main()
{
    int T, N, M;
    string str;
    cin >> T;

    while(T--) {
    	cin >> N >> M;
    	set <string> s;

    	for (int i = 0; i < N; i++) {
    		cin >> str;
    		s.insert(str);
    	}
    	if (s.size() == N) {
    		cout << "Yes" << endl;
    	}
    	else {
    		cout << "No" << endl;
    	}
    }
    return 0;
}
