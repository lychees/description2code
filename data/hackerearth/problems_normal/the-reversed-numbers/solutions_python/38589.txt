tst = raw_input()
for i in range(int(tst)):
    num = raw_input()
    num_rev = list(reversed(num))
    revnums = ''.join(num_rev)
    a,b = int(revnums.split(' ')[0]), int(revnums.split(' ')[1])
    s = a+b
    fnum = ''.join(list(reversed(str(s))))
