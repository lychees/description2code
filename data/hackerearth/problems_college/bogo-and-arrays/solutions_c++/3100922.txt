#include<iostream>
using namespace std;
int main(){
    int t,n;
    long long ans,k;
    cin>>t;
    while(t--){
        long long a[200005]={},pos[200005]={};
        cin>>n>>k;
        int i,temp;
        for(i=0;i<n;i++){
            cin>>a[i];
        }
        temp=n;
        for(i=n-1;i>=0;i--){
            if(a[i]>k){
                pos[i]=i;
                temp=i;
            }
            else pos[i]=temp;
        }
        ans=0;
        for(i=0;i<n;i++){
            ans=ans+n-pos[i];
        }
        cout<<ans<<'\n';
    }
    return 0;
}
