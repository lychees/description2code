#include<iostream>

using namespace std;
#define mod 4294967143
int main()
{
    int T;
    cin>>T;

    while(T--)
    {
        int N;
        cin>>N;
        long long r=0;

        for(int i=1;i<11;i++)
        {
            int Ar[270][11]={0};
            Ar[0][i]=1;
            for(int j=1;j<N;j++)
            {
                Ar[j][0]=Ar[j-1][1];
                Ar[j][10]=Ar[j-1][9];
                for(int k=1;k<10;k++)
                    Ar[j][k]=(Ar[j][k]+Ar[j-1][k+1]+Ar[j-1][k-1])%mod;
            }
            if(i==0)
                r=(r+Ar[N-1][1])%mod;
            else if(i==10)
                r=(r+Ar[N-1][9])%mod;
            else r=(r+Ar[N-1][i-1]+Ar[N-1][i+1])%mod;
        }
        cout<<r<<"\n";

    }
    return 0;

}
