a = {0: 0}


def winOrLoose(p, q):
    if a[p] == q:
        return False
    return True


d = 1
for i in xrange(1, 1000007):
    if (a.has_key(i)):
        continue
    else:
        a[i] = i+d
        a[i+d] = i
        d += 1

for i in xrange(input()):
    n1, n2 = ([int(i) for i in raw_input().split()])
    if winOrLoose(n1, n2):
        print 'Play'
    else:
        print "Don't Play"
