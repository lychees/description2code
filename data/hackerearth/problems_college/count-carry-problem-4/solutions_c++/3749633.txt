#include<iostream>
using namespace std;
int main()
{
    long long t,n,m,a,c;
    cin>>t;
    while(t--)
    {
        c=0;
        cin>>n>>m;
        while(n || m)
        {
            a=(n%10)+(m%10);
            if(a>9)
            {
                c++;
            }
            n/=10;
            m/=10;
        }
        if(c==0)
        cout<<"No carry operation";
        else if(c==1)
        cout<<c<<" carry operation";
        else
        cout<<c<<" carry operations";
        cout<<endl;
    }
}
