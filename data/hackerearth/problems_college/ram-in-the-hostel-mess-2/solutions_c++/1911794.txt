lli dp[1000000][2];
int n;
lli func(int i, int prev)
{
    if(i==n+1) return 1;
    if(dp[i][prev]!=-1) return dp[i][prev];
    lli ans=0;
    if(prev==0)
        ans=func(i+1,0)+func(i+1,1);
    else ans=func(i+1,0);
    return dp[i][prev]=ans;
}

int main()
{
    int t;
    sc(t);
    while(t--)
    {
        sc(n);
        for(int i=0;i<=n+1;i++)
            dp[i][0]=dp[i][1]=-1;
        lli ans=func(2,0)+func(2,1);
        ans=pow(ans,2);
        printf("%lld\n",ans);
    }
}
