#include <bits/stdc++.h>
#define ll long long
using namespace std;
bool mycheck(string s,int l)
{
	int i;
	bool flag=1;
	for(i=0;i<l/2;i++)
	{
		if(s.at(i)!=s.at(l-1-i))
		{
			flag=0;
			break;
		}
	}
	return flag;
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int test;
	cin>>test;
	while(test--)
	{
		string s1,s2,x;
		cin>>s1>>s2;
		int i,j,l1,l2;
		l1=s1.length();
		l2=s2.length();
		x=s2+s1;
		int ans=0;
		if(mycheck(x,l1+l2))
		ans++;
		//~ cout<<x<<endl;
		for(i=0;i<l1;i++)
		{
			x=s1.substr(0,i+1)+s2+s1.substr(i+1,l1-(i+1));
			if(mycheck(x,l1+l2))
			{
				//~ cout<<x<<endl;
				ans++;
			}
			//~ cout<<x<<endl;
		}
		cout<<ans<<endl;
	}
	return 0;
}
