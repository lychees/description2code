#include <iostream>
#include<bits/stdc++.h>
#define ll long long int
#define pb push_back
#define s second
#define f first
using namespace std;
typedef pair<int,int > p;
p arr[2500];
int main()
{
arr[1].f=1;
arr[1].s=2;
for(int i=2;i<=2400;i++)
{
 arr[i].f = ((i)*(i+1))/2;
 arr[i].s = arr[i].f+i;
}
int  t;
cin>>t;
int n;
while(t--)
{
cin>>n;
int ctr=1;
while(n>arr[ctr].f)
    ctr++;
if(n==arr[ctr].f)
    cout<<ctr<<endl;
else
    cout<<ctr-1<<endl;
}
}
