#include <iostream>
#include<cmath>
using namespace std;
long long int sum(int a){
	long long int *s=(long long int *)malloc(sizeof(long long int));
	*s=0;
	while(a>0){
		*s+=a%10;
		a/=10;
	}
	return *s;

}
int main()
{
	long long int t,a,b,k;
	cin>>t;
	while(t--){
		long long int res=1;
		cin>>a>>b>>k;
		int num=pow(10,k);
	
		while(b){
			if(b%2==0){
				a=(a*a)%num;
				b/=2;
			}
			else{
				res=(res*a)%num;
				b--;
			}
		}
		cout<<sum(res)<<endl;
	}
	
    return 0;
}
