#include <iostream>
#include <vector>
using namespace std;
#define SIZE 1000
typedef long long LL;

int main()
{

	int n;
	cin >> n;
	
	if(n==1){
		cout << 0 << endl;
		return 0;
	}
	
	vector<bool> isPrime(SIZE, true);
	int p=2;

	while(p<SIZE){
		
		 int _p = 2*p;
		 while(_p<SIZE){
		 	  isPrime[_p]=false;
		 	  _p+=p;
		 }
		 p++;
		 while(p<SIZE && !isPrime[p])p++;
	}
	
	LL sum=1; 
	LL curr=1; // 1 fraction for i=2
	
	for(int i=3; i<=n; i++){
		
		if(isPrime[i]) curr*=2;
		sum+=curr;
		
	}
	cout << sum << endl;
	return 0;
}
