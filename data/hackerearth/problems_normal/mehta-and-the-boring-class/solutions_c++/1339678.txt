#include <iostream>
#include <math.h>
typedef long int li;
#define SIZE 1000000
using namespace std;
int prime_fact[SIZE+1];

void precomputePrimes(int n){
	int count=0,num;
	num=n;
	while(!(n%2)){
		count++;
		n=n/2;
	}
	
	for(int i=3;i<=sqrt(n);i+=2){
		while(!(n%i)){
			count++;
			n=n/i;
		}
	}	
		prime_fact[num]=(n>1)?++count:count;
	
}

int main()
{
    int i,a,b;
    for(i=2;i<=SIZE;i++) precomputePrimes(i);
    for(i=2;i<=SIZE;i++) prime_fact[i]+=prime_fact[i-1];
    int t;
    cin>>t;
    while(t--){
    	scanf("%d %d",&a,&b);
    	printf("%d\n",prime_fact[b]-prime_fact[a-1]);
    }
}
