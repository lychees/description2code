#include <iostream>
#include<algorithm>
using namespace std;

int main() {
	int n,l,r,q;
	int arr[1010];
	cin>>n>>l>>r>>q;
	for(int i=0;i<n;i++)  cin>>arr[i];
    sort(arr+l,arr+r+1);
    while(q--)
    {
    	int x;
    	cin>>x;
    	int flag=0;
    	int i=0;
    	int number;
    	for( i=l;i<=r;i++)
    	{
    		if(arr[i]>x)  { flag=1; number=arr[i]; break; }
    	}
    	if(flag==0)   cout<<"-1"<<endl;
    	else cout<<number<<endl;
    }
    
	// your code goes here
	return 0;
}
