int n;

vector<int> bottom;
vector<int> top;

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		int color, tam;
		scanf("%d %d", &color, &tam);
		if (color == 1) bottom.pb(tam);
		else top.pb(tam); 
	}

	sort(bottom.begin(), bottom.end());
	sort(top.begin(), top.end());

	int can = 0;
	int ans = 0;
	for (int i = 0; i < bottom.size(); i++) {
		while (can < top.size() && top[can]*top[can]*2 <= bottom[i]*bottom[i]) can++;

		if (can < top.size() && top[can]*top[can] < bottom[i]*bottom[i]*2) {
			can++;
			ans++;
		}
	}

	printf("%d\n", ans);
}
