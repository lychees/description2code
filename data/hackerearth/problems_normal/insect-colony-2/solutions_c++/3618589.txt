#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,a;
    	cin>>n;
    	int cnt=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a;
    		if(a%2!=0)
    		{
    			cnt++;
    		}
    	}
    	if(cnt%2!=0)
    	{
    		cout<<"No"<<endl;
    		
    	}
    	else
    	{
    		cout<<"Yes"<<endl;
    	}
    }
    return 0;
}
