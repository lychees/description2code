#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t;
    long long int n,srt,count,fact,i;
    cin>>t;
    while(t--){
    	cin>>n;
    	count=0;
    	if(n%2==0){
    		while(n%2==0){
    			count++;
    			//cout<<2<<" ";
    			n/=2;
    		}
    		srt=sqrt(n);
    		for(i=3;i<=srt;i+=2){
    			fact=0;
    			while(n%i==0){
    				fact++;
    				n/=i;
    				//cout<<i<<" ";
    			}
    			
  				count*=(fact+1);
  				//cout<<"\ncount="<<count<<" ,fact="<<fact<<endl;
    		}
    		if(n>2)
    			count*=2;
    	}
    
    	cout<<count<<endl;
    }
    return 0;
}
