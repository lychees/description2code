//Sharpen the pencils 

#include<iostream>
using namespace std;

int main(){
	int t,i;
	long long n,j;
	
	cin>>t;
	for(i=0;i<t;++i){
		cin>>n;
		double length[n];
		long long x,y,madhav=0,riya=0;
		
		for(j=0;j<n;++j){
			cin>>length[j];
		}
		
		x=0;
		y=n-1;
		
		while(x<y){
			if((length[x]/2.0)<length[y]){
				length[y]-=(length[x]/2.0);
				length[x]=0;
				++madhav;
				
				if(x+1==y){
					++riya;
					break;
				}else{
					++x;
				}			
				
			}else if((length[x]/2.0)>length[y]){
				length[x]-=(length[y]*2.0);
				length[y]=0;
				++riya;
				
				if(y-1==x){
					++madhav;
					break;
				}else{
					--y;
				}
				
			}else{
				length[x]=length[y]=0;
				++madhav;
				++riya;
				
				if(x+1==y-1){
					++madhav;
					break;
				}
				else{
					++x;
					--y;
				}
			}
		}
		
		cout<<madhav<<" "<<riya<<endl;		
	}
	
	return 0;
}
