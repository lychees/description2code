#include<bits/stdc++.h>
using namespace std;
void fastread( long long *a){  //take input by fastread(&n);
char c=0;
while(c<33){
c=getchar_unlocked();}
*a=0;
while(c>33){
*a=*a*10+c-'0';
c=getchar_unlocked();
}
}
#define ll long long 
#define sc(a)  scanf("%lld",&a)
#define p(a)  printf("%lld",a)  
typedef vector<ll> vec ;

int main(){
   ll t,n,x,c,i,y;
   vec v;
   map<ll,bool> mp;
   sc(t);
   while(t--){
   	c=0;
   	sc(n);
   for(i=0;i<n;i++){
   	sc(x);
   	v.push_back(x);
   }
   //sort(v.begin(),v.end());
   for(i=0;i<n;i++){
   	y=v[i];
    if(mp.empty()){
    	mp.insert(pair<ll,bool>(y%10,true));
    }
    else{
    	if(mp.find(y%10)== mp.end())
    	  mp.insert(pair<ll,bool>(y%10,true));
    	else{
    		if(mp[y%10]==true) c++;
    		else mp.insert(pair<ll,bool>(y%10,true));
    	}
    		
    }
   }
   cout<<c<<endl;
   mp.clear();
   v.clear();
   	
   }
}
