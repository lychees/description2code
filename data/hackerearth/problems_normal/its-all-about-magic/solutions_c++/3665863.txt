#include <bits/stdc++.h>
#include <iostream>
using namespace std;
void magic(int m){
	int i;
	long int a[91] = {0};
	a[0] = 0;
	a[1] = 1;
	for(i = 2; i <= 90; i++){
		a[i] = a[i-1] + a[i-2];
	}
	
	cout<<a[m-1]<<" "<<a[m]<<endl;
}

int main()
{
	int t,m;
	cin>>t;
	while(t--){
		cin>>m;
		magic(m);
	}
}
