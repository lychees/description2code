#include <bits/stdc++.h>
using namespace std;
 
 
int main() 
{
    int t,n,k,i;
    cin>>t;
    while(t--)
        {
            cin>>n>>k;
            int a[n],x=0;
        for(i=0;i<n;i++)
            cin>>a[i];
        for(i=0;i<n;i++)
            if(a[i]<=0)x++;
            
            if(x<k) cout<<"YES"<<endl;
            else cout<<"NO"<<endl;
    }
    return 0;
}
