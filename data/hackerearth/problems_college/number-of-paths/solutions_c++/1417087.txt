#include <iostream>
using namespace std;
typedef long long ll;

ll a[1008][1008],dp[1008][1008];

int main()
{
    ll t,i,j,n,c,p,q;
    cin>>t;
    while(t--){
    	cin>>n>>c;
    	for(i=0;i<n;i++)
    		for(j=0;j<n;j++){
    			dp[i][j]=0;
    			a[i][j]=0;
    		}
    	for(i=0;i<c;i++){
    		cin>>p>>q;
    		a[p][q]=1;
    	}
    	int sx=-1,sy=-1;
    	for(j=1;j<n;j++){
    		dp[0][j]=1;
    		if(a[i][j]==1){
    			sx=i;sy=j;
    		}
    	}
    	for(i=1;i<n;i++){
    		
    		if(i>sx && 0>=sy)
    			dp[i][0]=1;
    		if(a[i][0]==1){
    			sx=i;sy=j;
    		}
    		
    		for(j=1;j<n;j++){
    			if(i>sx && j>=sy)
    				dp[i][j]=(dp[i][j]+dp[i-1][j])%1000000007;
    			if(i>=sx && j>sy)
    				dp[i][j]=(dp[i][j]+dp[i][j-1])%1000000007;
    			if(a[i][j]==1){
    				sx=i;sy=j;
    			}
    		}
    	}
    	cout<<dp[n-1][n-1]<<endl;
    }
    return 0;
}
