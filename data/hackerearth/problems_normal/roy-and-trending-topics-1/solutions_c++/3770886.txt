#include <iostream>
#include<algorithm>
#define ll long long
using namespace std;

struct topic
{
	ll id;
	ll z;
	ll p;
	ll l;
	ll c;
	ll s;
	ll zn;
}arr[1000001];

bool compare (topic t1,topic t2) 
{ 
	if((t1.zn-t1.z)>(t2.zn-t2.z))
		return true;
	else if((t1.zn-t1.z)==(t2.zn-t2.z))
		return ((t1.id)>(t2.id));
	else
		return false;
}

int main()
{
    ll N;
    cin>>N;
    for(ll i=0;i<N;i++)
    {
    	cin>>arr[i].id;
    	cin>>arr[i].z;
    	cin>>arr[i].p;
    	cin>>arr[i].l;
    	cin>>arr[i].c;
    	cin>>arr[i].s;
    	arr[i].zn = (arr[i].p*50)+(arr[i].l*5)+(arr[i].c*10)+(arr[i].s*20);
    }
    sort(arr,arr+N,compare);
    for(int i=0;i<5;i++)
    {
    	cout<<arr[i].id<<" "<<arr[i].zn<<endl;
    }
    return 0;
}
