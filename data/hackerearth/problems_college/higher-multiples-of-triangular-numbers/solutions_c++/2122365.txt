#include <iostream>
using namespace std;
 
int main() {
	int t;
	cin>>t;
	while(t--)
	{
	unsigned long long int j,n=1,sum=0,w=1;
	cin>>j;
	while(1)
	{
	unsigned long long int count=0,r=1;
	if(n%2==0)
	{
		while(n%2==0)
		{
			n=n/2;
			count++;
		}
		r=count+1;
	}
	for(unsigned long long int i=3;i*i<=n;i++)
	{
		unsigned long long int e=1;
		while(n%i==0)
		{
			n=n/i;
			e++;
		}
		if(e>1)
		r=r*e;
	}
	if(n>2)
	r=r*2;
	if(r>j)
	break;
	sum+=w;
	w++;
	n=sum;
	}
	cout<<sum<<endl;
	}
	return 0;
}
