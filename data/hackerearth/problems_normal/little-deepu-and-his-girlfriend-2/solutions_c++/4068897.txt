#include <bits/stdc++.h>
using namespace std;
int moves[110];
int iswin[10001];
int main()
{
    int t,num,m,n;
    ios_base::sync_with_stdio(0);
    cin>>t;
    while(t--)
    {
    	cin>>m>>n;
		memset(iswin,0,sizeof(iswin));
    	for(int i=0;i<n;i++)
    	{
    		cin>>moves[i];
    		iswin[moves[i]]=1;
    	}
    	sort(moves,moves+n);
		for(int i=1;i<=m;i++)
		{
			for(int j=0;j<n;j++)
			{
				if(i-moves[j]>=0&&!iswin[i-moves[j]]){
					iswin[i]=1;break;}
				
			}
		}
		if(!iswin[m])
    	cout<<"Kate\n";
    	else
    	cout<<"Little Deepu\n";
    }
    return 0;
}
