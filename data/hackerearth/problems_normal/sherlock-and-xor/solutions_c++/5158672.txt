#include <iostream>
using namespace std;

int main()
{
   unsigned long long int A[100000],i,e,o,n,t;
   cin>>t;
   while(t)
   {
   	o=0; e=0;
   	cin>>n;
   	for(i=0;i<n;i++)
   	 {cin>>A[i];
       if(A[i]%2==0)
        e++;
       else
        o++;
   	 }
   	 cout<<o*e<<endl;
   	t--;
   }
   
    return 0;
}
