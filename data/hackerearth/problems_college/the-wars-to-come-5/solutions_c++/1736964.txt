#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n,i,x,tmp=0,count=0,j,flag=0;
    cin >> n;
    int a[n];
    
    for(i=0 ; i<n ; i++)
    cin >> a[i];
    
    cin >> x;
    
    sort(a,a+n);
    
    for(i=n-1 ; i>=0 ; i--)
    {
    	tmp=a[i];
    	for(j=i-1 ; j>=0 ; j--)
    	{
    		if(tmp >= x || tmp <= x)
    		count++;
    	
    		if(tmp >= x)
    		{
    			flag=1;
    			break;
    		}
    	
    		//if(tmp > x)
    		//tmp-=a[i];
    		
    		tmp+=a[j];
    	}
    	
    	if(flag==1)
    	break;
    	else
    	count=0;
    }
    cout << count << endl;
    return 0;
}
