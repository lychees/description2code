#include <iostream>
#include <vector>
#include <algorithm>
#include <utility>
using namespace std;

int compare( const pair<int,int>& a, const pair<int,int>& b )
{
	return a.first!=b.first ? a.first>b.first : a.second>b.second;
}

int main()
{
    int nrquestions, nrqueries;
    cin >> nrquestions >> nrqueries;
    
    vector<pair<int,int> > questions( nrquestions, pair<int,int>(0,0) );
    for ( int idx=0; idx<nrquestions; idx++ )
    	cin >> questions[idx].second; // Times
    for ( int idx=0; idx<nrquestions; idx++ )
    	cin >> questions[idx].first; // Scores
    	
    sort( questions.begin(), questions.end(), compare );
    
    vector<long long int> requiredtime( nrquestions, -1 );
    requiredtime[0] = questions[0].second;
    for ( int idx=1; idx<nrquestions; idx++ )
    	requiredtime[idx] = requiredtime[idx-1] + questions[idx].second;
    	
    while ( nrqueries-- )
    {
    	int nrsolutions;
    	cin >> nrsolutions;
    	cout << (nrsolutions<=nrquestions ? requiredtime[nrsolutions-1] : requiredtime[nrquestions-1]);
    	cout << "\n";
    }
    
    return 0;
}
