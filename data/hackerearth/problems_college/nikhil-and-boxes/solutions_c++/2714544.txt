#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
 unsigned long t,l,b,h;	
   // cout << "Hello World!" << endl;
    cin>>t;
    while (t--)
    {
    	long double min=0;
    	l=0;
    	b=0;
    	h=0;
    	cin>>l>>b>>h;
   		 {
      	if (l<=b && l<=h)
    	{
     	min = l;
		}
    	else if (b<=l && b<=h)
    	{
    	min = b;
    	}
		else if (h<=l && h<=b)
    	{
    	min = h;
     	}
	    }
		cout<<setprecision(1)<<fixed<<min/2<<endl; 
    }
    return 0;
}
