#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,i,j,n,x,y;
	cin>>t;
	while(t--)
	{ 
		long long int carry=0,c=0;
		cin>>i>>j;
		while(i && j)
		{ 
		   x=i%10;
		   y=j%10;
		   i=i/10;
		   j=j/10;
		   
		   if((x+y+carry)>=10)
		    {    carry=1;
		    	c++;
		    }
		    else
		    carry=0;
		   
		}
		  if(c==0)
		    cout<<"No carry operation\n";
		    else if(c>1)
		          cout<<c<<" carry operations\n";
		          else
		cout<<c<<" carry operation\n";
	}
}
