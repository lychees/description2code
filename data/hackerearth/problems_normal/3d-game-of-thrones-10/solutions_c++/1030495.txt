#include <bits/stdc++.h>
using namespace std;

int g[55][55][55];

int main()
{
  for (int x = 0; x <= 50; x++)
    for (int y = 0; y <= 50; y++)
      for (int z = 0; z <= 50; z++)
      {
        set <int> s;
        for (int i = 1; i <= x; i++) s.insert(g[x - i][y][z]);
        for (int i = 1; i <= y; i++) s.insert(g[x][y - i][z]);
        for (int i = 1; i <= z; i++) s.insert(g[x][y][z - i]);
        while (s.count(g[x][y][z])) g[x][y][z]++;
      }
  
  ios::sync_with_stdio(0);
  int test, n, x, y, z;
  cin >> test;
  while (test--)
  {
    int sumXor = 0;
    cin >> n;
    while (n--)
    {
      cin >> x >> y >> z;
      sumXor ^= g[x][y][z];
    }
    cout << (sumXor ? "Stark" : "Lannister") << '\n';
  }
}
