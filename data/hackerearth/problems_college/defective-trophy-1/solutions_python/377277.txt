import math
t=input()
while t:
	n=input()
	if n==0:
		print '0'
	else:
		ans=math.log(n)/math.log(3)
		ans=math.ceil(ans)
		ans=("%.1f" % ans).replace(".0","")
		print ans
	t=t-1
