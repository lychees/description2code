#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T;
    long long N;
    
    scanf("%d", &T);
    
    while(T--)
    {
    	scanf("%lld", &N);
    	
    	int tmp, count = 0;
    	
    	while(N)
    	{
    		tmp = N%10;
    		N/= 10;
    		
    		if(tmp==4 || tmp==7)
    			count++;
    	}
    	
    	if(count==4 || count==7)
    		printf("YES\n");
    		
    	else
    		printf("NO\n");
    }
    
    return 0;
}
