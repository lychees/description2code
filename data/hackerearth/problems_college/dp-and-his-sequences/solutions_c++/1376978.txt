#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(0);cin.tie(0);
    int n,mini,maxi,ans;cin>>n;
    bool mn=1,mx=1;
    vector<int>a(n),v(n+1);
    for(int i=0;i<n;i++){
    	cin>>a[i];
    }
    v[0]=0;
    for(int i=0;i<n;i++){
    	v[i+1]=a[i]-v[i];
    }
    for(int i=0;i<n;i++){
    	if(i&1){
    		if(mx)maxi=v[i+1]-v[i],mx=0;
    		else maxi=min(maxi,v[i+1]-v[i]);
    	}
    	else{
    		if(mn)mini=v[i]-v[i+1],mn=0;
    		else mini=max(mini,v[i]-v[i+1]);
    	}
    }
    ans=maxi-mini+1;
    if(ans<=0)cout<<"no solution\n";
    else cout<<ans<<endl;
    return 0;
}
