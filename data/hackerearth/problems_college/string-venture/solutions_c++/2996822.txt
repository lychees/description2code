#include <iostream>
#include <string.h>
#include <bits/stdc++.h>

using namespace std;

char max_occurence(char a[])
{
	int count[256];
	for (int j = 0; j < 256; j++)
		count[j] = 0;
	
	int len = strlen(a);
	int max = -1;
	char c;
	for (int j = 0; j < len; j++)
		count[a[j]]++;
	for (int i = 0; i < len; i++)
	{
		if (max < count[a[i]])
		{
			max = count[a[i]];
			c = a[i];
		}
	}
	return c;
}

int main()
{
	char a[100000];

	cin.getline(a, 100000);

	int len = strlen(a);
	int sum = 0;
	for (int i = 0; i < len; i++)
	{
		int b = a[i];
		sum = sum + b;
	}
	int ans = floor((sum / len));
//	cout << ans << endl;
	if (ans % 2 == 0)
	{
		for (int i = len-1; i >= 0; i--)
			cout << a[i];
	}
	else if (ans % 2 == 1)
	{
		char c = max_occurence(a);
		cout << c;
	}
	cout << endl;
	return 0;
}
