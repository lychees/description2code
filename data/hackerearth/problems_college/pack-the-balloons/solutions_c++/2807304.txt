#include <bits/stdc++.h>
using namespace std;
int main()
{
    long l,b,h,min;
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%ld%ld%ld",&l,&b,&h);
    	min=h<(l<b?l:b)?h:(l<b?l:b);
    	if(min%2==0)
    		printf("%ld.0\n",min/2);
    	else printf("%ld.5\n",min/2);
    }
    return 0;
}
