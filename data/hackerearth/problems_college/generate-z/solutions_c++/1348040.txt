#include <iostream>
using namespace std;

int main()
{
    int n, m, i, j;
    cin >> n;
    m=n-2;
    for (i = 0; i < n; ++i)
    	cout << "*";
    cout << "\n";
    
    for (i = m; i > 0; --i)
    {
    	for (j = i; j >= 1; --j)
    	 cout << " ";
    	cout << "*" << "\n";
    }
    
    for (i = 0; i < n; ++i)
    	cout << "*";
    
    return 0;
}
