#include <bits/stdc++.h>

using namespace std;

int prime[1000005];

void sieve()
{
	prime[0] = prime[1] = 0;
	for (int i = 2; i * i <= 1000000; i++) {
		if (prime[i] == 1) {
			for (int j = i + i; j * j <= 1000000; j = i + j) {
				prime[j] = 0;
			}
		}
	}
}

int main()
{
	int t, n;
	for (int i = 0; i <= 1000000; i++) prime[i] = 1;
	sieve();
	scanf("%d", &t);
	while (t--) {
		scanf("%d", &n);
		int cnt = 0;
		while (n--) {
			int x;
			scanf("%d", &x);
			if (prime[x] == 1) {
				cnt++;
			}
		}
		printf("%d\n", cnt);
	}
	
	return 0;
}
