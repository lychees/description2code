alpoints = []
temp = raw_input().split(' ')
for i in range(26):
	alpoints.append(int(temp[i]))

sum1 = 0
sum2 = 0

Q = int(raw_input())
for i in range(Q):
	M = raw_input()
	for j in range(len(M)):
		sum1+=alpoints[ord(M[j]) - 97]
	D = raw_input()
	for j in range(len(D)):
		sum2+=alpoints[ord(D[j]) - 97]

if sum1>sum2:
	print 'Marut'
elif sum1<sum2:
	print 'Devil'
else:
	print 'Draw'
