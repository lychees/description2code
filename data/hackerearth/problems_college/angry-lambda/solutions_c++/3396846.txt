#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */  
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
        {
        int n,k;
        cin>>n>>k;
        int count=0;
        for(int j=0;j<n;j++)
            {
            int l;
            cin>>l;
            if(l<=0)
                count++;
        }
        if(count>=k)
            cout<<"NO"<<endl;
        else
            cout<<"YES"<<endl;
    }
    return 0;
}
