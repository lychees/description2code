#include<bits/stdc++.h>
using namespace std;

int main(){
	int t;
	cin >> t;

	while(t--){
		int n,k;
		cin >> n >> k;
		vector<string> vec;
		for(int i=0;i<n;i++){
			string in;
			cin >> in;
			vec.push_back(in);

		}
		sort(vec.begin(),vec.end());
		string result = "";
		for(int j=0;j<k;j++){
			vector<char> ans;
			for(int i=0;i<n;i++){
				ans.push_back(vec[i][j]);
			}
			sort(ans.begin(),ans.end());
 			int su = 0;
			int mu = 1000000;
			char an;
			for(char i='a';i<='z';i++){
				su = 0;
				for(int k=0;k<n;k++){
					su += abs(i-vec[k][j]);
				}
				if(mu > su){
					mu = su;
					an = i;
				}
				
			}
			result += an; 
		}
		cout << result << endl;
	}
	return 0;
}
