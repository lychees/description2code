#include<stdio.h>
#include<string.h>
#define S 100010

int main()
{
    int i,in,le,tc,flag;
    scanf("%d",&tc);
    char str[S];
    while(tc--)
    {
        flag=0;
        scanf("%s",str);
        le=strlen(str);
        for(i=0; i<le/2; i++){
            if(str[i]!=str[le-1-i]){flag=1;break;}
            }
if(flag==0)printf("-1\n");
else {
    flag=0;
    in=i;
    for(;i<=le/2;i++){
        if(str[i]!=str[le-2-i]){flag=1;break;}
        }
        if(flag==0)printf("%d\n",le-1-in);
    else printf("%d\n",in);

}
        }
    return 0;
}
