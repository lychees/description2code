#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n;
    cin>>t; long x;
    for(;t--;) {
    	cin>>n;
    	if(n==1) {
    		cout<<"1"<<endl;
    		continue;
    	}
    	if(n%2) {
    		x=(n-1)/2;
    		x*=n;
    		x+=(n/2-1);
    		cout<<x<<endl;
    	} else {
    		x=n/2;
    		x*=(n-1);
    		x+=(n/2-1);
    		cout<<x<<endl;
    	}
    }
    return 0;
}
