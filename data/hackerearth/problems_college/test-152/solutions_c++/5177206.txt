#include <iostream>
using namespace std;

int main()
{
    int i,j,k,ara[30],len,t,N,mx;
    char str[100045];
    cin>>t;
    while(t--)
    {
    	for(i=0;i<=29;i++) ara[i]=0;
    	cin>>len;
    	cin>>str;
    	mx=-1;
    	for(i=0;i<len;i++){
    		j=str[i]-'a';
    		ara[j]++;
    		if(ara[j]>mx) mx=ara[j];
    	}
    	if(mx%2==0) cout<<"Yes"<<endl;
    	else cout<<"No"<<endl;

    }
    return 0;
}
