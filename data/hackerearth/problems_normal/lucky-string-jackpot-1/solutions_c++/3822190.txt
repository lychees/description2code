#include <iostream>
using namespace std;

long powerdo(long a)
{
	long result=1;
	while(a--)
		result=result<<1;
	return result;
}

long sum(long n)
{
	return (((n-1)*powerdo(n+1))+2);
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long n;
    	cin>>n;
    	int i;
    	for(i=1;i<45;i++)
    	{
    		if(sum(i)>=n)
    			break;
    	}
    	n-=sum(i-1);
    	if(n>=((2*i)-1))
    		cout<<((2*i)-1);
    	else if(n==((2*i)-2))
    		cout<<n;
    	else
    		cout<<((2*i)-3);
    	cout<<endl;
    }
    return 0;
}
