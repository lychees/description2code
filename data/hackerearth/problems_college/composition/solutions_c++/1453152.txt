#include<iostream>
using namespace std;
#define MOD 1000000007
typedef long long int ll;
ll fibo(ll n);
void power(ll f[2][2],ll n);
void mul(ll f[2][2],ll m[2][2]);
ll fibo(ll n)
{
	ll f[2][2]={{1,1},{1,0}};
	power(f,n-1);
	return f[0][0];
}
void power(ll f[2][2],ll n)
{
	if(n==0||n==1)
	return;
	ll m[2][2]={{1,1},{1,0}};
	power(f,n/2);
	mul(f,f);
	if(n%2!=0)
	mul(f,m);
}
void mul(ll f[2][2],ll m[2][2])
{
	ll a=f[0][0]*m[0][0]+f[0][1]*m[1][0];
	ll b=f[0][0]*m[0][1]+f[0][1]*m[1][1];
	ll c=f[1][0]*m[0][0]+f[1][1]*m[1][0];
	ll d=f[1][0]*m[0][1]+f[1][1]*m[1][1];
	f[0][0]=a%MOD;
	f[0][1]=b%MOD;
	f[1][0]=c%MOD;
	f[1][1]=d%MOD;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll n;
		cin>>n;
		cout<<fibo(n)<<endl;
	}
	return 0;
}
