typedef long long ll;

int n;
string s[55];
ll len[55];
char solve(ll x)
{
    int i;
    for(i=0;i<n;i++)
    {
        if(x+1<=len[i])
        {
            break;
        }
    }
    if(i==0)
    {
        return s[0][x];
    }
    if(x+1<=2*len[i-1])
    {
        return solve(2*len[i-1]-x-1);
    }
    x-=2*len[i-1];
    return s[i][x];
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int t,m,i;
    ll x;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        for(i=0;i<n;i++)
        {
            cin>>s[i];
        }
        len[0]=s[0].size();
        for(i=1;i<n;i++)
        {
            len[i]=2*len[i-1]+s[i].size();
        }
        while(m--)
        {
            cin>>x;
            cout<<solve(x);
        }
        cout<<"\n";
    }
}
