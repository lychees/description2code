#include <bits/stdc++.h>
using namespace std;
int V[100005],A[100005];

int main()
{
	int T,Ind,i,N;
	scanf("%d",&T);
	while(T--)
	{
		memset(V,0,sizeof(V));
		scanf("%d",&N);
		for(i=0;i<N;i++)
			scanf("%d",&A[i]);
		Ind=0;
		while(1)
		{
			if(A[Ind]==-1)
			{
				printf("Win\n");
				break;
			}
			if(V[Ind])
			{
				printf("Lose\n");
				break;
			}
			V[Ind]+=1;	Ind=A[Ind];
		}
	}
return 0;
}
