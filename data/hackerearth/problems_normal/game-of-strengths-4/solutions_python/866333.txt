#Game of Strengths
import sys
import math

t=int(sys.stdin.readline())
while t>0:
	n=int(sys.stdin.readline())
	num=list(sys.stdin.readline().split())
	num=[int(x) for x in num]
	num.sort()
	#print (num)
	s=0
	i=n-1
	while(i>=0):
		s+=num[i]*i-num[i]*(n-i-1)
		i-=1
	#print (int(s))
	s+=1000000007
	s%=1000000007
	s*=num[n-1]
	s%=1000000007
	print (s)

	t-=1
