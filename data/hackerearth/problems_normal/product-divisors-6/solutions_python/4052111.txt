mod = 1000000007
length = long(raw_input())
arr = map(long, raw_input().split())
dict1 = dict()
for num in arr:
	i = 2
	while i*i <= num:
		while num % i == 0:
			dict1[i] = dict1.get(i, 0) + 1
			num /= i
		i += 1
	if num > 1:
		dict1[num] = dict1.get(num, 0) + 1
res = 1
for key, val in dict1.items():
	res = (res %mod * (val + 1)%mod)%mod
print res
