#define max(a,b)  ((a)>(b)?(a):(b))

typedef vector<int> V;

const long long LINF = (long long)1e16 + 5;
int main()
{
	ll n, a[100001], i, j, dp[100001];
	cin >> n;
	for (i = 0; i < n; i++)
	{
		cin >> a[i];
	}
	//memset(dp, 0, sizeof(dp));
	dp[0] = 1;
	for (i = 1; i < n; i++)
	{
		if (a[i] > a[i - 1])
		{
			dp[i] = 1 + dp[i - 1];
		}
		else
		{
			dp[i] = 1;
		}
	}
	for (i = n - 2; i >= 0; i--)
	{
		if (a[i] > a[i + 1] && dp[i] <= dp[i + 1])
		{
			dp[i] = 1 + dp[i + 1];
		}
	}
	ll sum = 0;
	for (i = 0; i < n; i++)
	{
		sum += dp[i];
	}
	cout << sum << endl;
}
