#include <iostream>
using namespace std;
int d2b(int n)
{
	int r=0,p=0,a=n;
	while(a)
	{
		r=a%2;
		if(r&p==1)
		{
		return false;
		}
		p=r;
		a/=2;
	}
	return true;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,j;
    	cin>>n;
    	loop:
    	j=d2b(n);
    	if(j)cout<<n<<endl;
    	else 
    	{
    	n=n+1;
    	goto loop;
    	}
    }
    return 0;
}
