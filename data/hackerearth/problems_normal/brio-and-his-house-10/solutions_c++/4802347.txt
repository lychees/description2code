#include<iostream>
#include<math.h>
#include<iomanip>
using namespace std;
int main()
{
int t;
cin>>t;
while(t>0)
{
double a,b,c,r,area,temp,pi=3.1415;
cin>>a>>b>>c;
temp=sqrt((a+b+c)*(b+c-a)*(c+a-b)*(a+b-c));
r=(a*b*c)/temp;
area=r*r*pi;
cout<<fixed<<setprecision(4)<<area<<endl;;
t--;
}
}
