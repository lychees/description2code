#include <iostream>
#include<map>
using namespace std;

int main()
{
	map<char,long long int> m;
	long long int n,count=0,count2=0;
	cin>>n;string s;
    for(long long int i=0;i<n;i++)
    {
    	cin>>s;
    	count2++;
    	if(m[s[0]]==0)
    	count++;
    	m[s[0]]++;
    	if(m[s[0]]==10)
    	{
    		//count++;
    		m[s[0]]=0;
    	}
    	
    }
    cout<<count<<endl;
    return 0;
}
