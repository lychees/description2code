#define vl vector<ll>
#define pb push_back
#define ll long long int
#define mod 1000000007
#define gcd(a,b) __gcd(a,b)
#define sf(n) scanf("%lld",&(n))
#define pf(n) printf("%lld\n",(n))
#define setbits(n) __builtin_popcount((n))
#define pq priority_queue
using namespace std;
map<ll,ll>m;
ll power(ll a,ll b)
{
    if(b==0)
        return 1;
    else if(b==1)
        return a;
    else if(b%2==0)
    {
        ll x=power(a,b/2)%mod;
        return (x*x)%mod;
    }
    else
        return (a%mod*power(a,b-1)%mod)%mod;
}
vector<ll> divisors(ll n)
{
    vector<ll>v;
    for(ll i=2;i*i<=n;i++)
    {
        if(n%i==0)
        {
            ll x=n/i;
            v.pb(i);
            if(x!=i)
                v.pb(x);
        }
    }
    return v;
}
ll func(ll n)
{
    if(n==1|| n==2)
        return 1;
    if(m.find(n)!=m.end())
        return m[n];
    ll ans=1;
    vector<ll>v=divisors(n-1);
    for(vector<ll>::iterator it=v.begin();it!=v.end();it++)
    {
        ans=(ans%mod+power(func(*it),(n-1)/(*it))%mod)%mod;
    }
    m.insert(mp(n,ans));
    return ans;
}
int main()
{
    ll n;
    cin>>n;
    cout<<func(n)<<"\n";
}
