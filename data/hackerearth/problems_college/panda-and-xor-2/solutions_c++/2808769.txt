#include<bits/stdc++.h>
using namespace std;

unsigned int countSB(unsigned int n)
{
	unsigned int count=0;
	while(n)
	{   n&=(n-1);
		count++;
	}
return count;	
}
int main()
{
    unsigned int t;
    cin>>t;
    while(t--)
    {
    	unsigned int a,b,res;
    	cin>>a>>b;
    	res=a^b;
    	res=countSB(res);
    	
    	if(res%2==0) cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    	
    }
    return 0;
}
