#include <iostream>
#include<cstring>
using namespace std;

int main()
{
   int t,i;
   char s[101],d[101];
   cin>>t;
   while(t--){
     for(i=0;i<101;i++)
     d[i] = '\0';
   	cin>>s;
   	for(i=0;i<strlen(s);i++){
            if(i+s[i]>122)
            d[i] = i + s[i] - 26;
    else
   		d[i] = i + s[i];
   	}
   	cout<<d<<endl;
   }
    return 0;
}
