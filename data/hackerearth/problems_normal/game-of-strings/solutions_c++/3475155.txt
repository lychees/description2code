#include <bits/stdc++.h>
using namespace std;

int arr[5010],temp[5010];

int max(int a,int b){
	return a>b?a:b;
}

int LIS(char s1[],char s2[],int len1,int len2)
{
	int i,j;
	for(i=0;i<=len2;i++)
	{
		arr[i]=0;
		temp[i]=0;
	}
	for(i=1;i<=len1;i++)
	{
		temp[0]=0;
		for(j=1;j<=len2;j++)
		{
			if(s1[i-1]==s2[j-1])
			{
				temp[j]=arr[j-1]+1;
			}
			else
			{
				temp[j]=max(temp[j-1],arr[j]);
			}
		}
		for(j=0;j<=len2;j++)
		{
			arr[j]=temp[j];
		}
	}
	return arr[len2];
}


int main()
{
	char s1[5010],s2[5010];
	int i,j,k=0,len1,len2;
	cin>>s1>>s2;
	for(i=0;s1[i]!='\0';i++)
	{
		if(s1[i]!='a'&&s1[i]!='e'&&s1[i]!='i'&&s1[i]!='o'&&s1[i]!='u')
		{
			s1[k]=s1[i];
			k++;
		}
	}
	s1[k]='\0';
	len1=k;
	k=0;
	for(i=0;s2[i]!='\0';i++)
	{
		if(s2[i]!='a'&&s2[i]!='e'&&s2[i]!='i'&&s2[i]!='o'&&s2[i]!='u')
		{
			s2[k]=s2[i];
			k++;
		}
	}
	s2[k]='\0';
	len2=k;
	//cout<<s1<<" "<<len1<<endl;
	//cout<<s2<<" "<<len2<<endl;
	cout<<LIS(s1,s2,len1,len2)<<endl;
	return 0;
}
