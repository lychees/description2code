# print 'Hello World!'
for i in range(int(raw_input())):
    arr = list(raw_input())
    # print arr
    k = len(arr)/2
    n = len(arr)
    count = 0
    for z in range(k):
        x = ord(arr[z])-ord(arr[n-1-z])
        # print arr[k],arr[n-k]
        # x=m-l
        if x > 0:
            count += x
        else:
            count = count-x
    print count
