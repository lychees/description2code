#include <iostream>
#include <cstring>
#include <cstdlib>
#include <map>
using namespace std;
#include<cmath>

int main()
{
    map <string,int> count;
    int n;
    string a,b;
       
    cin>>n;
    int p = pow(2,n)-1;
    while(p--){
    	cin>>a>>b;
    	count[a]++;
    }
    
    
    
    map<string,int> :: iterator i,max;
    max=count.begin();
    for(i=count.begin();i!=count.end();i++){
    	if((*max).second < (*i).second){
    		max=i;
    	}
    }
    
    cout<<(*max).first<<endl;
    return 0;
}
