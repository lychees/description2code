def upload_status(W, H, L):
    if W < L or H < L:
        return "UPLOAD ANOTHER"
    else:
        if W == H:
            return "ACCEPTED"
        else:
            return "CROP IT"


L = int(raw_input())
N = int(raw_input())
while N != 0:
    (W, H) = [int(alpha) for alpha in raw_input().split(" ")]
    print(upload_status(W, H, L))
    N -= 1
