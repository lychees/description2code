#include <bits/stdc++.h>
using namespace std;

int A[10];

int main()
{
	int N;
	scanf("%d", &N);
	
	int ans = 0, sum = 0;
    
    for(int i=0; i<N; i++)
    {
    	scanf("%lld", &A[i]);
    	
    	sum+= A[i];
    }
    
    sort(A, A+N);
    
    do
    {
    	for(int i=0; i<(1<<N); i++)
    	{
    		bool flag = 0;
    		int one = 0, two = 0;
    		
    		for(int j=0; j<N; j++)
    		{
    			if(i&(1<<j))
    				one+= A[j];
    			else
    				two+= A[j];
    			
    			if(one>two)
    			{
    				flag = 1;
    				break;
    			}
    		}
    		
    		if(!flag)
	    		ans++;
    	}
    	
    }while(next_permutation(A, A+N));
    
    printf("%d %d\n", ans, sum);
    
    if(sum<ans)
    	printf("We will win!\n");
    else
    	printf("Got no way out!\n");
    
    return 0;
}
