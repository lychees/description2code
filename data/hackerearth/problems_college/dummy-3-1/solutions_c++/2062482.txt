#include <iostream>
#include <cassert>
using namespace std;

int main()
{
    int t;
    cin>>t;
    assert(t>=1 && t<=100);
    while(t!=0)
    {
    	long int n,flag=0,i,l=0,m;
    	cin>>n;
    	assert(n>=1 && n<=100000);
    	int a[n];
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	for(i=1;i<=n;i++)
    	{
    		m=a[l];
    		if(a[m-1]==i)
    		{
    			l++;
    		}
    		else
    		flag++;
    	}
    	if(flag==0)
    	cout<<"inverse"<<endl;
    	else
    	cout<<"not inverse"<<endl;
    	t--;
    }
    return 0;
}
