#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

const int N = 100001;
int t, n, a, b, c, sum;

int main(){
    ios_base::sync_with_stdio(false);
    cin >> t;
    while (t--){
        cin >> n;
        sum = 0;
        for(int i = 0; i < n; ++i){
            cin >> a >> b >> c;
            sum ^= (a ^ b ^c);
        }
        cout << (sum ? "Stark\n" : "Lannister\n");
    }    
    return 0;
}
