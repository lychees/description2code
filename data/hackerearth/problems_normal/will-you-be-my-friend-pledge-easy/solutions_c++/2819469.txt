#include <iostream>
using namespace std;

int main()
{
    int n,t,j=2,k=0,c=0;
    cin>>n>>t;
    int b[n/2];
    while(j<=n/2)
    {
    	if(n%j==0)
    	{
    		b[k]=j;
    		k++;
    	}
    	j++;
    }
    int a[t];
    for(int i=0;i<t;i++)
    {
    	cin>>a[i];
    }
    for(int i=0;i<t;i++)
    {
    	for(int j=0;j<k;j++)
    	{
    		if(a[i]%b[j]==0)
    		{
    			c++;
    			break;
    		}
    	}
    }
    cout<<c;
    return 0;
}
