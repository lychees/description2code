#include<bits/stdc++.h>
using namespace std;
int main()
{

    int q;
    int n;
    cin>>n>>q;
    int arr[n];
    memset(arr,0,n);
    int ct[n];
    memset(ct,0,n);
    while(q--)
    {
        int x,a,b;
        cin>>x>>a>>b;
        int count=0;
        if(x==1)
        {
            for(int i=a;i<=b;i++)
            {
                if(arr[i]==1)
                    count++;
            }
            cout<<count<<endl;
        }
        else
        {
            for(int i=a;i<=b;i++)
            {
                if(arr[i]==1)
                    arr[i]=0;
                else
                    arr[i]=1;
            }
        }
    }
}
