t = int(input())

while t:

    a , b = map(int,raw_input().split())

    m , n = a , a
    
    cnt_m , cnt_n = 0 , 0
    
    while m % b != 0:
        m -= 1
        cnt_m += 1

    while n % b != 0:
        n+=1
        cnt_n += 1

    if cnt_m > cnt_n :
        print(n)
    elif cnt_m < cnt_n :
        print(m)
    else :
        print(max(m,n))

    t -= 1
