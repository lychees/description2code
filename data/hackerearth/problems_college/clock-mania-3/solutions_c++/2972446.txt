#include<iostream>
#include<tgmath.h>
#include<stdio.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int a,b,c;
        cin>>a>>b;
        a=a*30+b/2;
        b*=6;
        c=abs(b-a);
        if(c>180)
            c=360-c;
        cout<<c<<endl;
    }
    return 0;
}
