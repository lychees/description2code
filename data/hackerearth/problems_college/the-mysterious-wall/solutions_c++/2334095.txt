#include<bits/stdc++.h>
#define N 110000
using namespace std;
int arr[N][27];
inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}
string ip()
{
    string i="";
    int temp=getchar_unlocked();
    while(temp<'a'||temp>'z')
        temp=getchar_unlocked();
    while(temp>='a'&&temp<='z')
    {
        i+=(char)temp;
        temp=getchar_unlocked();
    }
    return i;
}
int main(){
	int t;
	t=scan();
	while(t--){
		
		string a,b; a=ip(); b=ip();
		int l1=a.size();
		int l2=b.size();
		if(l2>l1){
			printf("NO\n");
			continue;
		}int x[27];
		for(int i=0; i<26; i++){
			arr[0][i]=0;
			x[i]=0;
		} 
		
		for(int i=0; i<l2; i++){
			x[b[i]-'a']++;
		}
		for(int i=1; i<=l1; i++){
			for(int j=0; j<26; j++) arr[i][j]=arr[i-1][j];
			arr[i][a[i-1]-'a']++;
		}
		int f=0;
		for(int i=l2; i<=l1; i++){
			int n[27];
			f=0;
			for(int j=0; j<26; j++) n[j]=arr[i][j]-arr[i-l2][j];
			int flag=0;
			for(int j=0; j<26; j++){
				if(n[j]!=x[j]){
					flag=1;
					break;
				} 
			}	
			if(!flag){
				f=1; break;
			}
		}
		if(f) printf("YES\n");
		else printf("NO\n");
	}
}
