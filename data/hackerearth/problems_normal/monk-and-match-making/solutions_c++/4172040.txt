#include<bits/stdc++.h>
#define b1 15
#define b2 24
#define m1 10000009
#define m2 10000007
#define ll long long
using namespace std;
ll h1[100010],h2[100010],p1[100010],p2[100010];
string s;
void pre()
{
    p1[0]=p2[0]=1;
    for(int i=1;i<100010;i++)
    {
        p1[i]=(b1*p1[i-1])%m1;
        p2[i]=(b2*p2[i-1])%m2;
    }
    for(int i=1;i<=s.length();i++)
    {
        h1[i]=(b1*h1[i-1]+s[i-1]-'a'+1)%m1;
        h2[i]=(b2*h2[i-1]+s[i-1]-'a'+1)%m2;
    }
}
ll gethash1(int l,int r)
{
    ll x=(h1[r]-(h1[l-1]*p1[r-l+1])%m1+m1)%m1;
    return x;
}
ll gethash2(int l,int r)
{
    ll x=(h2[r]-(h2[l-1]*p2[r-l+1])%m2+m2)%m2;
    return x;
}
int main()
{
    cin>>s;
    pre();
    int q;
    cin>>q;
    while(q--)
    {
        int l1,r1,l2,r2;
        cin>>l1>>r1>>l2>>r2;
        if((r1-l1)==(r2-l2))
        {
            ll x1=gethash1(l1,r1);
            ll x2=gethash1(l2,r2);
            ll y1=gethash2(l1,r1);
            ll y2=gethash2(l2,r2);
            if(x1==x2&&y1==y2)
                printf("Yes\n");
            else
                printf("No\n");
        }
        else
            printf("No\n");
    }
    return 0;
}
