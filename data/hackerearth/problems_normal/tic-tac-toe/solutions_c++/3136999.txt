#include <bits/stdc++.h>
#define M 1000000007
#define ll long long int 
using namespace std;
ll modexp(ll a,ll b)
{
	ll ans=1;
	while(b)
	{
		if(b%2)
		{
			ans=(ans*a)%M;
		}
		a=(a*a)%M;
		b/=2;
	}
	return ans;
}
int main()
{
	int t;
	cin>>t;
	ll C=modexp(6,M-2);
	while(t--)
	{
	    ll minx=0,maxx=0;
	    ll n;
	    cin>>n;
	    ll p=(n-1)/2;
	    minx=(((p*p)%M)*n)%M;
	    n=n-1;
	    maxx=(n*(n+1))%M;
	    maxx=(maxx*(2*n+1))%M;
	    maxx=(maxx*C)%M;
	    cout<<minx<<" "<<maxx<<endl;
	}
}
