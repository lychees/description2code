#include<iostream>
using namespace std;
#include<set>
int root(int a[],int i)
{
	while(i!=a[i])
	i=a[i];
	return i;
}
bool find (int a[],int i,int j)
{
int x=root(a,i);
int y=root(a,j);
if(x==y)
return true;
return false;
}
void uni(int a[],int s[],int i,int j)
{
	if((!find(a,i,j)))
{
int x=root(a,i);
int	y=root(a,j);
	if(s[x]>s[y])
	{
		a[y]=x;
		s[x]+=s[y];
	}
	else
   {
	a[x]=y;
	s[y]+=s[x];
   }
}
}
int main()
{
int n,q;
int i,j;
cin>>n;
int a[n+1];
for(int p=1;p<n+1;p++)
a[p]=p;
int s[n+1];
for(int p=1;p<n+1;p++)
s[p]=1;
cin>>q;
while(q--)
{
	cin>>i>>j;
	uni(a,s,i,j);
}
int cnt=0;
for(int i=1;i<n+1;i++)
{
	if(a[i]==i)
	cnt++;
}
cout<<cnt<<endl;
}
