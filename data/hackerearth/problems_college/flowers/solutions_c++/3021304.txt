#include <iostream>
#include<algorithm>
using namespace std;
int a[105];
int upper(int a,int b){
if(a-a/b*b==0)
return a/b;
else return a/b+1;

}
int main()
{
	int n,k;
	scanf("%d %d",&n,&k);
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]);
	}
	sort(a,a+n);
	int ans=0;
	for(int i=n-1;i>=0;i--){
		ans+=upper(n-i,k)*a[i];
	}
	printf("%d\n",ans);
   // cout << "Hello World!" << endl;
    return 0;
}
