#include <iostream>
using namespace std;

int main()
{
    int tes,N,D,R;
    short int ans;
    
    cin>>tes;
    
    while(tes--) {
    
     cin>>N>>D>>R;
     
     for(int i=0; i<R; i++) {
      
       if(N == 0) {
       	ans = 0;
       	break;
       }
         
        if(N<D) {
          N = N*10;
        }
       
        
        ans = N/D;
        
        if(N>=D)
        N = N - ans*D;
     }
     
     cout<<ans<<endl;
     
    }
    
    return 0;
}
