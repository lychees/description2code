#include <bits/stdc++.h>
using namespace std;
int main() {
  int t;
  scanf("%d", &t);
  string s, s1;
  while (t--) {
    cin >> s;
    int l = s.length();
    s1 = s[0];
    for (int i = 1; i < l; i++) {
      if (s[i] >= s1[0])
        s1 = s[i] + s1;
      else
        s1 = s1 + s[i];
    }
    cout << s1 << endl;
  }
  return 0;
}
