#include <bits/stdc++.h>
using namespace std;
#define MAX 1000000
set<int> numbers;
int size;
void preprocess()
{
	numbers.insert(1);
	for(int i=2;i<=sqrt(MAX);i++)
	{
		for (long long int j = i*i; j<=MAX ; j=j*i)
		{
			if(numbers.find(j)==numbers.end())
				numbers.insert(j);
		}
	}
	size=numbers.size();
	
}
bool search(int no)
{
	auto left=numbers.begin();
	auto right=numbers.end();
	int left_id=0;
	int right_id=size-1;
	right--;
	while(left_id<right_id)
	{
		if((*left)+(*right)==no||2*(*left)==no||2*(*right)==no)
			return true;
		else if((*left)+(*right)>no)
			{right--;right_id--;}
		else
			{left++;left_id++;}
	}
	return false;

}
int main()
{
	int tc;
	cin>>tc;
	preprocess();
	while(tc--)
	{	
	int n;
	cin>>n;
	if(search(n))
		cout<<"Yes"<<endl;
	else
		cout<<"No"<<endl;
	}
	return 0;
}
