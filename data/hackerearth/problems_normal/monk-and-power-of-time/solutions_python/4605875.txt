processes = [int(s) for s in raw_input().split()]
ideal = [int(s) for s in raw_input().split()]
p_pointer = 0
p_ideal = 0
time = 0
while p_ideal < len(ideal):
	while processes[p_pointer%len(processes)] is None:
		p_pointer += 1
	if processes[p_pointer%len(processes)] == ideal[p_ideal]:
		p_ideal += 1
		processes[p_pointer%len(processes)] = None
	p_pointer += 1
	time += 1
print time
