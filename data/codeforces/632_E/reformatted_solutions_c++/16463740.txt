#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <ctime>
#include <iostream>
#include <map>
#include <set>
using namespace std;
#define LL long long
int N, K;
int a[1010];
int f[1000010];
int main() {
  scanf("%d%d", &N, &K);
  for (int i = 1; i <= N; i++)
    scanf("%d", &a[i]);
  sort(a + 1, a + N + 1);
  for (int i = N; i > 1; i--)
    a[i] = a[i] - a[1];
  memset(f, 61, sizeof(f));
  f[0] = 0;
  int up = a[N] * K;
  for (int i = 0; i <= up; i++)
    for (int j = 2; j <= N; j++)
      if (i >= a[j])
        f[i] = min(f[i], f[i - a[j]] + 1);
  for (int i = 0; i <= up; i++)
    if (f[i] <= K)
      printf("%d ", i + a[1] * K);
}
