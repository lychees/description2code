#include <iostream>
using namespace std;

int n,t,a[51][51],i,j,flag=0;

void func(int x,int y)
{
	if(x==n-1&&y==n-1)
	{
		flag=1;
		return;
	}
	if(x+1<n&&a[x+1][y]==1)
	{
		func(x+1,y);
	}
	if(y+1<n&&a[x][y+1]==1)
	{
		func(x,y+1);
	}
	return;
}

int main()
{
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<n;j++)
    		cin>>a[i][j];
    	}
    	
    	func(0,0);
    	if(flag==1)
    	cout<<"POSSIBLE"<<endl;
    	else
    	cout<<"NOT POSSIBLE"<<endl;
    	flag=0;
    }
    
    return 0;
}
