#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,x;
		vector<int>v;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>x;
			v.push_back(x);
		}
		int count=0,flag=0;
		for(int i=0;i<n&&count<n;i=x)
		{
			if(v[i]==-1)
			{
				flag=1;
				break;
			}
			x=v[i];
			count++;
		}
		if(flag)	
			cout<<"Win"<<endl;
		else
			cout<<"Lose"<<endl;
	}
}
