#include <bits/stdc++.h>
using namespace std;

int n;
double sv[100010],x[100010],y[100010],f[100010];

int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		
		cin>>x[i]>>y[i]>>f[i];
		
	}
	for(int i=0;i<=n;i++) sv[i]=-1e9;
	sv[1]=f[1];
	for(int i=2;i<=n;i++){
		for(int j=1;j<i;j++){
			double ds=sqrt((x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]));
			sv[i]=max(sv[i],sv[j]-ds);
		}
		sv[i]+=f[i];
	}
	cout<<fixed<<setprecision(6)<<sv[n]<<endl;
	return 0;
}
