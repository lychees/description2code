#include<iostream>
using namespace std;
int a[1001];
void multiply(int y)
{
	int b[1001];
	int c[1001];
	int d[1001];
	int i;
	//cout<<"MUL "<<y<<endl;
	//finding b;
	int carry=0;
	int digit=y%10;
	for(i=1000;i>0;i--)
	{
		b[i]=(a[i]*digit+carry)%10;
		carry=(a[i]*digit+carry)/10;
	}
	//for(int k=991;k<=1000;k++) cout<<b[k];cout<<endl;
	
	//finding c;
	carry=0;
	digit=y/10;
	digit=digit%10;
	c[1000]=0;
	for(i=999;i>0;i--)
	{
		c[i]=(a[i+1]*digit+carry)%10;
		carry=(a[i+1]*digit+carry)/10;
	}
	//for(int k=991;k<=1000;k++) cout<<c[k];cout<<endl;
	
	//finding d
	carry=0;
	digit=y/100;
	digit=digit%10;
	d[1000]=0;
	d[999]=0;
	for(i=998;i>0;i--)
	{
		d[i]=(a[i+2]*digit+carry)%10;
		carry=(a[i+2]*digit+carry)/10;
	}
	//for(int k=991;k<=1000;k++) cout<<d[k];cout<<endl;
	
	//sum
	carry=0;
	for(i=1000;i>0;i--)
	{
		a[i]=b[i]+c[i]+d[i]+carry;
		carry=a[i]/10;
		a[i]%=10;
	}
	
	//for(int k=991;k<=1000;k++) cout<<a[k];cout<<endl;
}

int main()
{
	int i;
	for(i=0;i<=1000;i++) a[i]=0;
	int fact[251];
	fact[1]=0;
	fact[2]=0;
	a[1000]=2;
	for(i=3;i<=250;i++)
	{
		//cout<<a[991]<<a[992]<<a[993]<<a[994]<<a[995]<<a[996]<<a[997]<<a[998]<<a[999]<<a[1000]<<endl;
		multiply(i);
		int count=0;
		for(int j=0;j<=1000;j++) if((a[j]==4)||(a[j]==7)) count++;
		fact[i]=count;
	}
	//cout<<i<<endl;
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<fact[n]<<endl;
	}
}
