#include<bits/stdc++.h>
using namespace std;
int digit(int a)
{
	int count=0;
	while(a>0)
	{
		a=a/10;
		count++;
	}
	return count;
}
int call(int a,int b)
{
	int d=digit(a);
	int mod1,mod2,flag=0;
	for(int i=0;i<d;i++)
	{
		mod1=a%10;
		mod2=b%10;
		if(mod1!=mod2)
		{
			flag++;
		}
		a=a/10;
		b=b/10;
	}
	if(flag==1)
	return 1;
	else if(flag==0)
	return 1;
	else 
	return 0;
}
int main()
{
    int a,b;
    cin>>a>>b;
    int flag=call(a,b);
    if(flag==1)
    {
    	cout<<"wins"<<endl;
    }
    else
    {
    	cout<<"loses"<<endl;
    }
    
}
