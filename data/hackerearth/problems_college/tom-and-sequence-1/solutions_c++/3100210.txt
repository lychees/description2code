#include "bits/stdc++.h"
using namespace std;
vector<int> primes;
int arr[10000002];
int main()
{
	int t;
	cin>>t;
	fill(arr,arr+1000000,0);
	arr[0]=1;
	arr[1]=1;
	//int arr1[3200000];
	for(int i=2;i<=10000000;i++)
	{
		if(arr[i]==0)
		{
			for(int j=i+i;j<=10000000;j+=i)
			{
				arr[j]=1;
			}
		}
	}
	for(int i=0;i<=10000000;i++){
		if(arr[i]==0) primes.push_back(i);
	}
	while(t--){
		int n;
		cin>>n;
		long long int ans=1;
		for(int i=0;i<n;i++){
			ans=(ans*primes[i])%1000000007;
		}
		cout<<ans<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
