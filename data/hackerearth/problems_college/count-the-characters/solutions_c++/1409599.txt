#include <iostream>
#include <algorithm>
#include <cstring>
#define $ while
using namespace std;

int main() {
	int n,i=0,l,c=0,f;
	cin>>n;
	char str[100];
	int al[100][26]={0};
	$(n--)
	{
		cin>>str;
		l=strlen(str);
		for(int j=0;j<l;j++)
		{
			al[i][(str[j])-97]=1;
		}
		i++;
	}
	for(int l=0;l<26;l++)
	{
		for(int j=0;j<i;j++)
		{
			if(al[j][l]==0)
			{
				f=0;
				break;
			}
			else
			f=1;
		}
		if(f==1)
		c++;
	}
	cout<<c;
	return 0;
}
