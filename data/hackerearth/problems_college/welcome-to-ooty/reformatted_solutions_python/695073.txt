while (True):
    try:
        s = raw_input()
        l = len(s)
        f = 0
        b = 0
        for j in range(1, l):
            if (s[l-j] == ' '):
                b = int(s[j:l])
                f = int(s[0:j])
                break
        if (b*f != 0):
            print b*f
    except EOFError:
        break
