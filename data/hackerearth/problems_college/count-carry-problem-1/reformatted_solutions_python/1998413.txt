t = int(raw_input())
for i in range(t):
    ans = 0
    s = 10
    m, n = map(int, raw_input().split())

    for j in range(14):
        n1 = m % 10
        m = m/10
        n2 = n % 10
        n = n/10
        if n1+n2 >= s:
            ans += 1
            s = 9
        else:
            s = 10
    if ans == 0:
        print("No carry operation")
    elif ans == 1:
        print("1 carry operation")
    else:
        print(str(ans)+" carry operations")
