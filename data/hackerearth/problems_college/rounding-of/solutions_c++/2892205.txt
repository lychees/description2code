#include<bits/stdc++.h>

using namespace std;

int main()
{
    int t,n,b;
    cin >> t;

    while(t--)
    {
        cin >> n >> b;

        int r = n%b;
        if(r==0)cout << n<<endl;
        else if(r>=(b+1)/2)
        {
            cout << ((n/b)*b)+b <<endl;
        }
        else cout << ((n/b)*b) <<endl;
    }

}
