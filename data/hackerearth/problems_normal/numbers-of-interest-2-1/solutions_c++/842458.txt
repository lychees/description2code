#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
#define ll long long int
#define ull unsigned long long int
#define scan(x) scanf("%d",&x)
#define scanll(x) scanf("%lld",&x)
#define print(x) printf("%d\n",x)
#define printll(x) printf("%lld\n",x)


#define MAX 1000005

ll x[MAX];
ll f[MAX];
ll y[MAX];
ll g[MAX];

// function to calculate smallest prime factor
void spf(){
	for(ll i=2;i<MAX;i++){
		if(x[i]==0){
			for(ll j=i;j<MAX;j=j+i){
				if (x[j]==0)                // If not done ,largest prime factor gets stored.
					x[j]=i;                 
			}
		}
	}
}

void calc(){
	ll i,j;
	for(i=1;i<=(1000002/2);i++)
    {
        for(j=i+i;j<1000002;j+=i)
        {
            y[j]+=i;
        }
    }
}



int main()
{   
	memset(x,0,sizeof(x));
	memset(f,0,sizeof(f));
	spf();
	calc();	

	for(ll i=2;i<MAX;i++){
		f[i]=f[i-1]+x[i];
	}

	for(ll i=2;i<MAX;i++){
		g[i]=g[i-1]+y[i];
	}


	int t;
    scan(t);
    while(t--)
        {
	        ll n;
	        scanll(n);
	        printll((f[n]+g[n])%n);
        }



	return 0;
}
