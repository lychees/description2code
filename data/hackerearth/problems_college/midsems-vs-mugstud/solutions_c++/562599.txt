#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define S(x) scanf("%lld", &x)

int main()
{
	ll t;
	S(t);
	while(t--) {
		ll n;
		S(n);
		ll b[60][60][60] = {0};
		ll a[60][60][60];
		ll i,j,k;
		for(i = 1;i <= n;i++) {
			for(j = 1;j <= n;j++) {
				for(k = 1;k <= n;k++) {
					S(b[i][j][k]);
					ll val = (b[i - 1][j][k] + b[i][j - 1][k] + b[i][j][k - 1] + b[i - 1][j - 1][k - 1]) - (b[i - 1][j - 1][k] + b[i][j - 1][k - 1] + b[i -1][j][k - 1]);
					cout << b[i][j][k] - val << " ";
				}
				cout << "\n";
			}
		}
	}
}
