#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

vector<int> in, dp;
int N;
int S;

int isOK(int S, int i){
	if(S==0) return 1;
	if(i==N) return 0;
	return isOK(S-in[i], i+1) || isOK(S, i+1); 
}

int sol(){
	cin >> N;
	in.resize(N);
	for(int i=0;i<N;i++) cin >> in[i];
	cin >> S;
	if( isOK(S, 0) )
		cout << "YES" << endl;
	else
		cout << "NO" << endl;
}

int main()
{
    int t= 1;
    cin >> t;
    while(t--) sol();
    return 0;
}
