#include <bits/stdc++.h>
using namespace std;
map<char,int>m;
bool cmp(char a,char b){
	return(m[a]<m[b]);	
}
int main()
{
    int t; cin>>t;
    while(t--){
    	string p,s; cin>>p>>s;
    	m.clear();
    	for(int i=0;i<p.length();i++){
    		m[p[i]]=i;
    	}
    	sort(s.begin(),s.end(),cmp);
    	cout<<s<<endl;
    }
    return 0;
}
