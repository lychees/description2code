#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    int ar[t];
    for(int i=0; i<t; i++)
    {
        cin>>ar[i];
    }
    int q;
    cin>>q;
    while(q!=0)
    {
        int a;
        int sum=0;
        cin>>a;
        for(int i=0; i<t; i++)
        {
            sum=sum+ar[i];
            if(sum>=a)
            {
                cout<<i+1<<"\n";
                break;
            }

        }
        if(sum<a)
        {
            cout<<"-1\n";
        }
        q--;

    }
    return 0;
}
