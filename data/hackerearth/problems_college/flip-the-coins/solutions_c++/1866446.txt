#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    a[i]=0;
    int q=0;
    cin>>q;
    while(q--)
    {
    	int e,b,c;
    	cin>>e>>b>>c;
    	if(e==0)
    	{
    		for(int i=b;i<=c;i++)
    		a[i]=!a[i];
    	}
    	else
    	{
    		int s=0;
    		for(int i=b;i<=c;i++)
    		{
    			if(a[i]==1)
    			s++;
    		}
    		cout<<s<<endl;
    	}
    }
}
