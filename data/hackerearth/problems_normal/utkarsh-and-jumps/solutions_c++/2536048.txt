#include <iostream>
#include <iomanip>
using namespace std;

long n;
double p;

double pr[1000005];

int main()
{
    cin >> n >> p;
    
    p /= 100;
    
    pr[2] = p;
    pr[3] = 1-p;
    
    for( int i = 4; i <= n; i++ )
    	pr[i] = p * pr[i-2] + (1-p) * pr[i-3];
    	
	cout << fixed << setprecision(6) << pr[n];
    
    return 0;
}
