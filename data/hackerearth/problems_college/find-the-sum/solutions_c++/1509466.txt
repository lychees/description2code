#include <stdio.h>
long long a[2000006];
int main()
{
    long long t,l,r,i,n,mid,first,last,left,right,save,ans;
    n=1;
    for(i=1;;)
    {
    	if(i>2000000000000)
    	break;
    	a[n]=i;
    	i+=n;
    	n++;
    }
    //printf("%lld %lld\n",n,a[n-1]);
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld%lld",&l,&r);
    	ans=0;
    	first=1;last=n;
    	while(first<=last)
    	{
    		mid=(first+last)/2;
    		if(a[mid]==l)
    		{
    			save=mid;
    			break;
    		}
    		else if(a[mid]>l)
    		last=mid-1;
    		else if(a[mid]<l)
    		{
    			save=mid;
    			first=mid+1;
    		}
    	}
    	left=save;
    	//printf("%lld ",left);
    	first=1;last=n;
    	while(first<=last)
    	{
    		mid=(first+last)/2;
    		if(a[mid]==r)
    		{
    			save=mid;
    			break;
    		}
    		else if(a[mid]>r)
    		last=mid-1;
    		else if(a[mid]<r)
    		{
    			save=mid;
    			first=mid+1;
    		}
    	}
    	right=save;
    	//printf("%lld\n",right);
    	if(left!=right)
    	{
    		if((l-a[left]+1)%2==0)
    		{
    			if((a[left+1]-l)%2)
    			ans+=-1*left;
    		}
    		else
    		{
    			if((a[left+1]-l)%2)
    			ans+=left;
    		}
    		//printf("%lld ",ans);
    		
    		if((r-a[right]+1)%2) ans+=right;
    		//printf("%lld ",ans);
    		ans+=(right/2)*(right/2)-((left+1)/2)*((left+1)/2);
    	}
    	else
    	{
    		if((l-a[left])%2)
    		{
    			if((r-l+1)%2)
    			ans=-1*left;
    			
    		}
    		else
    		{
    			if((r-l+1)%2)
    			ans=left;
    		}
    	}
    	if(l==r)
    	{
    		if((l-a[left])%2) ans=-1*left;
    		else ans=left;
    	}
    	printf("%lld\n",ans);
    }
    return 0;
}
