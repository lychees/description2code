#include<iostream>
using namespace std;
int main()
{
	int T;
	cin >> T;
	for(int loop = 0; loop < T; loop++)
	{
		int Z = 1;
		int N;
		cin >> N;
		for(int i=2; i<N; i++)
		{
			int gcd = N;
			int temp = i;
			while(temp!=0)
			{
				int r = gcd%temp;
				gcd = temp;
				temp = r;
			}
			if(gcd==1)
				Z++;
		}
		bool b = true;
		int j=2;
		if(Z==1)
		{
			b=false;
		}
		else
		{
			while(j*j<=Z)
			{
				if(Z%j==0)
				{
					b=false;
					}
				j++;
			}
		}
		if(b)
			cout<<"TRUE\n";
		else
			cout<<"FALSE\n";
	}
	return 0;
}
