#include <iostream>
using namespace std;
 
long int score(int a[],long long int n);
 
int main()
{
    long long int n,i;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    
    cout<<score(a,n);
    return 0;
}
 
long int score(int a[],long long int n)
{
	long long int i;
	int q=0,r=0;
	for(i=0;i<n;i++)
	{
		r+=a[i];
		if(r<0)
		{
			r=0;
		}
		if(q<r)
		{
			q=r;
		}
	}
	return q;
	
}
