#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s,p;
    int t;
    cin>>t;
    while(t--)
    {
    	cin>>s>>p;
    	if(s.length()<p.length())
    	cout<<"NO"<<endl;
    	else if(s.length()==p.length())
    	{
    		if(s==p)
    		cout<<"YES"<<endl;
    		else cout<<"NO"<<endl;
    	}
    	else
    	{
    		int i=0,j=0;
    		bool pop=false;
    		int h=p.length(),hh=s.length();
    		while(i<hh)
    		{
    			if(s[i]==p[j]&&j<h)
    			{
    				j++;
    			}
    			i++;
    		}
    		if(j==h)
    		cout<<"YES"<<endl;
    		else cout<<"NO"<<endl;
    	}
    
    }
}
