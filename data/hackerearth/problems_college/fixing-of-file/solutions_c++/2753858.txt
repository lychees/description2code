#include<bits/stdc++.h>
using namespace std;

int main() {
  //freopen("a.cpp","r",stdin);
  int cases=1;
  string str="/home/gcj";
  int tt;
  cin>>tt;
  map<string,bool>mp;
  while(tt--) {
    cout<<"Case #"<<cases<<": ";
    cases++;
    int n,m;
    cin>>n>>m;
    mp.clear();
    string str;
    for(int i=1;i<=n;i++) {
      cin>>str;
      istringstream ss(str);
      string token;
      string temp="";
      while(getline(ss,token,'/')) {
        if(token.size()==0) {
          continue;
        }
        temp+=token+"+";
        mp[temp]=true;
      }
    }
    int cnt=0;
    for(int i=1;i<=m;i++) {
      cin>>str;
      istringstream ss(str);
      string token;
      string temp="";
      while(getline(ss,token,'/')) {
        if(token.size()==0) {
          continue;
        }
        temp+=token+"+";
        if(mp.find(temp)==mp.end()) {
          cnt++;
          mp[temp]=true;
        }
      }
    }
    cout<<cnt<<endl;
  }
  return 0;
}
