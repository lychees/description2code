#include <iostream>
using namespace std;
#include<cmath>
int main() {
int t;
scanf("%d",&t);
while(t--)
{
long long n,z,step,n2,n3,diff,mid;
double n1;
scanf("%lld",&n);
n1=sqrt(n);
z=sqrt(n);
if(n1==z)
{
    step=z+z-1;
    printf("%lld\n",step);
}
else {
    n2=floor(n1);
    n2=n2*n2;
    n3=ceil(n1);
    n3=n3*n3;
    diff=n3-n2;
    mid=diff/2;
    if(n3-mid>n)
        {
            step=2*(sqrt(n3))-2;
             printf("%lld\n",step);
        }
    else
    {
            step=2*(sqrt(n3))-1;
             printf("%lld\n",step);
    }
}

}

return 0;
}
