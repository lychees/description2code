import sys
from collections import deque


n = int(sys.stdin.readline().rstrip())


for i in xrange(0, n):
    flag = 0
    length = int(sys.stdin.readline().rstrip())
    # step 1: add the numbers to the array if less than sum
    array = [int(x) for x in sys.stdin.readline().rstrip().split()]
    subsum = int(sys.stdin.readline().rstrip())
    subSum = array
    for i in xrange(0, length):
        num = array[i]
        if num == subsum:
            flag = 1
    newLength = len(subSum)
    queue = zip(subSum, xrange(0, newLength))
    queue = deque(queue)
    while len(queue) > 0 and flag == 0:
        item = queue.popleft()
        for it in xrange(item[1]+1, newLength):
            sum = subSum[it] + item[0]
            subSum.append(sum)
            queue.append((sum, it))
            if sum == subsum:
                flag = 1

    if flag == 1:
        print 'YES'
    else:
        print 'NO'
