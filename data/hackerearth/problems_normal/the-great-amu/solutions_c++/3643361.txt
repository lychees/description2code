#include<bits/stdc++.h>
using namespace std;
const int mx=321;
vector<int>d[mx],e[mx];
const int mod[3]={2,5,719};
bool ok[mx];

int fexp(int b,int e,int m){
   int ans=1;
   for(;e!=0;e>>=1,b=(b*b)%m)
      if(e&1)ans=ans*b%m;
   return ans;
}
int main(){
//   freopen("data.in","r",stdin);
   int n,q;
   scanf("%d %d",&n,&q);
   for(int i=0;i<n;i++){
      d[i].resize(n+q);
      e[i].resize(n+q);
      for(int j=0;j<n;j++)
         scanf("%d",&d[i][j]);
   }
   for(int i=0;i<q;i++){
      for(int j=0;j<n;j++)
         scanf("%d",&d[j][n+i]);
      ok[i]=true;
   }
   for(int it=0;it<3;it++){
      int M=mod[it];
      for(int i=0;i<n;i++)
         for(int j=0;j<n+q;j++)
            e[i][j]=d[i][j]%M;

      int row=0,col=0;
//      for(int i=0;i<n;i++){for(int j=0;j<n+q;j++)cout<<e[i][j]<<" "; cout<<endl; } cout<<"-----"<<endl;
      while(row<n && col<n){
         int idx=row;
         for(int i=row;i<n;i++)
            if(e[i][col]!=0)idx=i;
         swap(e[idx],e[row]);
//         cout<<"swap!"<<endl;
//         for(int i=0;i<n;i++){for(int j=0;j<n+q;j++)cout<<e[i][j]<<" "; cout<<endl; } cout<<"-----"<<endl;
         if(e[row][col]==0){
            col++;
            continue;
         }
         for(int i=0;i<n;i++)if(i!=row && e[i][col]!=0){
            int c=fexp(e[row][col],M-2,M)*e[i][col]%M;
            c=(M-c)%M;
            for(int j=0;j<n+q;j++){
               e[i][j]=(e[i][j]+e[row][j]*c)%M;
            }
         }
         row++,col++;

//         for(int i=0;i<n;i++){for(int j=0;j<n+q;j++)cout<<e[i][j]<<" "; cout<<endl; } cout<<"-----"<<endl;
      }


//      for(int i=0;i<n;i++){for(int j=0;j<n+q;j++)cout<<e[i][j]<<" "; cout<<endl; } cout<<"-----"<<endl;
      for(int i=0;i<n;i++){
         bool z=true;
         for(int j=0;j<n;j++)
            if(e[i][j]!=0)z=false;
         if(z)for(int j=0;j<q;j++)
            if(e[i][j+n]!=0)ok[j]=false;
      }
   }
   for(int i=0;i<q;i++){
      if(ok[i])puts("Yes");
      else puts("No");
   }
   return 0;
}
