#include <bits/stdc++.h>

using namespace std;

const int N = 1.5e6 + 10;
const long long MAXN = 1e12;

int size;
long long tr[N];

void sieve() {
	int i;
	tr[0] = 0;
	for (i = 1; tr[i - 1] < MAXN; i++) {
		tr[i] = tr[i - 1] + i;
	}
	size = i;
}

long long calc(long long l, long long r) {
	long long res = 0;
	for (int i = 1; tr[i] < r; i++) {
		int lb = lower_bound(tr + i, tr + size, l - tr[i]) - (tr + i);
		int ub = upper_bound(tr + i, tr + size, r - tr[i]) - (tr + i);
		res += ub - lb;
	}
	return res;
}

int main() {
	sieve();
	int t;
	long long l, r;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++) {
		scanf("%lld %lld", &l, &r);
		printf("%lld\n", calc(l, r));
	}
	return 0;
}
