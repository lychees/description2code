int a[100005], ans[100005];
int main()
{
	ios_base::sync_with_stdio(false);
	int n;
	cin >> n;
	for(int i = 1; i <= n; i++) cin >> a[i];
	ans[1] = a[1];
	for(int i = 1; i <= n; i++)
		for(int j = 2; i*j <= n; j++)
			if(ans[i*j] < ans[i] + a[i*j]) ans[i*j] = ans[i] + a[i*j];	
	for(int i = 1; i <= n; i++) cout << ans[i] << " ";
	cout << endl;
	return 0;
}
