#include <bits/stdc++.h>

using namespace std;
#define ll long long 
#define pii pair<int,int>
#define mp make_pair
#define pb push_back
#define si(a) scanf("%d",&a)
#define sl(a) scanf("%lld",&a)
#define pi(a) printf("%d",a)
#define pl(a) printf("%lld",a)
#define pm(str) printf("%s\n",str)
#define pn printf("\n")

int main()
{
	int t;
	si(t);
	while(t--)
	{
		int a0,a1,c,n;
		cin >> a0>>a1>>c>>n;
		vector<int> v;
		v.clear();
		v.resize(n);
		v[0] = a0;
		v[1] = a1;
		if(n==1)
		{
			if(a0==0)
			{
				pm("YES");
			}
			else
			{
				pm("NO");
			}
			continue;
		}
		for(int i=2;i<n;i++)
		{
			v[i] = ( v[i - 1]*c + v[i - 2] );
			v[i] = v[i]%10;
		}
		ll curry = v[0]*10+v[1];
		ll rem=0;
		for(int i=2;i<n;i++)
		{
			rem = curry % 41;
			curry = rem * 10 + v[i];
		}
		if(curry%41==0)
		{
			pm("YES");
		}
		else
		{
			pm("NO");
		}
 	}
 	return 0;
}
