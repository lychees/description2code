#include<cstdio>
#include<math.h>
using namespace std;
long long ar[100003];
int main()
{
	int ntc; scanf("%d",&ntc);
	while(ntc--)
	{
		
		long long N, X; scanf("%lld %lld",&N,&X);
		for(int i=0;i<N;i++) scanf("%lld",&ar[i]);
		long long ans = 0;
	    long long pre = ar[0];
	    for(int i=1;i<N;i++)
	    {
	    	long long tmp = ar[i] - pre - 1;
	    	//printf("tmp is %lld\n",tmp);
	    	if(tmp<0)
	    	{
	    		tmp = -tmp;
	    		long long a = tmp / X;
	    		if(a*X != tmp) a++;
	    	//	printf(" a is %lld\n",a);
	    		ans+=a;
	    		pre = ar[i] + a*X;
	    	}
	    	else pre = ar[i];
	    }
	
	    printf("%lld\n",ans);
	}
}
