#include <bits/stdc++.h>
using namespace std;
long long a[1003][2];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%lld %lld",&a[i][0], &a[i][1]);
    a[n][0] = a[0][0];
    a[n][1] = a[0][1];
    
    long long area = 0;
    for(int i=0;i<n;i++) area += (a[i][0]*a[i+1][1] - a[i+1][0]*a[i][1]);
    area /= 2;
    if(area < 0) area *= -1;
    
    long long b = 0;
    for(int i=0;i<n;i++) b += __gcd(abs(a[i][0]-a[i+1][0]), abs(a[i][1]-a[i+1][1]));;
    
    cout << area + 1 - b/2 << endl;   
    
    return 0;
}
