#include<bits/stdc++.h>
#define ll long long
#define MAX 50000
#define rep(i,a,b) for(ll i=a;i<b;i++)
#define IOS ios::sync_with_stdio(false)
#define vi vector < ll >
#define pb push_back
#define mod 1000000007
using namespace std;

vi a,b;
ll dp[105][10005];

int main()
{
    IOS;
    ll t;
    cin>>t;
    rep(j,0,t)
    {
        a.clear();b.clear();
        ll n,p;
        cin>>n>>p;
        rep(i,0,103) rep(j,0,10003) dp[i][j]=0;
        rep(i,0,n)
        {
            ll x,y;
            cin>>x>>y;
            a.pb(x);b.pb(y);
        }
        ll s=p;
        //rep(i,0,n) cerr<<a[i]<<" "<<b[i]<<endl;
        rep(i,1,n+1)
        {
            rep(k,1,p+1)
            {
                if(b[i-1]<=k)
                   {
                       dp[i][k]=max((a[i-1]+dp[i-1][k-b[i-1]])-b[i-1],dp[i-1][k]);
                       //if(dp[i][k]>0) cerr<<i<<" "<<k<<" "<<dp[i][k]<<endl;
                   }
                else dp[i][k]=dp[i-1][k];
            }
        }
        ll ans1=0,ans2=0;
        rep(i,1,p+1) if(ans2<dp[n][i]) {ans2=dp[n][i];ans1=i;}
        cout<<ans1<<" "<<ans2+p<<endl;
    }
    return 0;
}
