#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;
int main()
{
	int m,i,j,arr[200][200],sum1,sum2;
	cin >> m;
	for(i=0;i<m;i++)
	{
		for(j=0;j<m;j++)
			cin >> arr[i][j];
	}
	sum1 = 0;
	sum2 = 0;
	for(i=0;i<m;i++)
	{
		sum1 += arr[i][i];
		sum2 += arr[m-i-1][i];
	}
	cout << abs(sum1-sum2);
	return 0;
}
