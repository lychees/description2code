#include<iostream>
#include<cstdlib>
using namespace std;

int compare(const void *a,const void *b){
	return (*(int *)a - *(int *)b);
}

int main()
{
    int d,i;
    cin>>d;
    
    for(i=0;i<d;i++){
    	int g,j;
    	long long int m;
    	cin>>g>>m;
    	
    	long long int a[g];
    	
    	for(j=0;j<g;j++){
    		cin>>a[j];
    	}
    	
    	qsort(a,g,sizeof(long long int),compare);
    	
    	int count = 0;
    	long long int sum = 0;
    	
    	j=0;
    	while(sum+a[j]<=m){
    		count++;
    		sum = sum + a[j];
    		j++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
