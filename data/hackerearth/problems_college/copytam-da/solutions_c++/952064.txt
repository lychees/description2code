#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    long long int t,n,a[100000],i,c=0;
    cin>>t;
    while(t--){
            c=0;
        cin>>n;
        for(i=0;i<n;i++){
            cin>>a[i];
        }
        sort(a,a+n);
        for(i=1;i<n;i++){
            if(a[i]!=a[i-1])
                c++;
        }
        cout<<c+1<<endl;
    }
    return 0;
}
