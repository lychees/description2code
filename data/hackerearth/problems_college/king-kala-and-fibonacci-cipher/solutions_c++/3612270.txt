char dp[SIZE];
int dp1[SIZE][26];
void fibo()
{
	int i,x;
	dp[0]='a';
	dp[1]=dp[2]='b';
	forp(i,3,SIZE-1){
		x=(dp[i-1]+dp[i-2]-'a'-'a')%26;
		dp[i]=x+'a';
	}
	return ;
}
bool compare(const pair<ll,ll>&i, const pair<ll,ll>&j)
{
	if(i.ff!=j.ff)
	return i.ff > j.ff;
	return i.ss < j.ss ;
}
