#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll dp[1004],a[1005];
int main()
{
    //cout << "Hello World!" << endl;
    ll n,m,t,i,j;
    cin>>t;
    while(t--){
    	dp[0]=0;
    
    	cin>>m>>n;
    	for(i=0;i<n;i++) cin>>a[i];
    	for(i=1;i<=m;i++){
    		dp[i]=1e9+7;
    		for(j=0;j<n;j++){
    			if(i-a[j]>=0 && dp[i-a[j]]!=1e9+7) dp[i]=min(dp[i-a[j]]+1,dp[i]);
    		}
    }
    if(dp[m]==1e9+7) cout<<-1<<endl;
    else	cout<<dp[m]<<endl;
    	
    	
    	
    }
    return 0;
}
