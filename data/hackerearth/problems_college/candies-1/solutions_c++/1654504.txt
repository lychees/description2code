#include<bits/stdc++.h>
using namespace std;
int main()
{
long long int t;
	cin>>t;
	while(t--)
	{
	long long int n,ans1,ans2,i;
	long long int a[10000];
	cin>>n;
	long long int sum=0;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		sum=sum+a[i];
	}
	ans2=sum%n;
	ans1=sum/n;
	if(ans2==0)
	{
		cout<<"YES"<<'\n';
		cout<<ans1<<' '<<ans2<<'\n';
	}
	else
	{
		cout<<"NO"<<'\n';
		cout<<ans1<<' '<<ans2<<'\n';
		
	}
}
	return 0;
}
