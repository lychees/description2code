template<typename t>
t gi() { t x; cin >> x; return x; }
template<typename t>
t gg( t x ) { cin >> x; return x; }

int main() {
	ios::sync_with_stdio( false );
	int n = gg( 0 );
	int q = gg( 0 );
	vector<ll> a , b;
	rep( i , 0 , n )
		a.push_back( gg( 0 ) );
	rep( i , 0 , n )
		b.push_back( gg( 0 ) );
	vector<ll> t1( n , 0 ) , t2( n , 0 );
	rep( i , 0 , n ) {
		if (i % 2 == 0) {
			t1[ i ] += a[ i ];
			t2[ i ] += b[ i ];
		}
		else {
			t1[ i ] += b[ i ];
			t2[ i ] += a[ i ];
		}
		if (i > 0)t1[ i ] += t1[ i - 1 ];
		if (i > 0)t2[ i ] += t2[ i - 1 ];
	}
	rep( i , 0 , q ) {
		int t , l , r;
		cin >> t >> l >> r;
		--l , --r;
		if (( t == 1 && l % 2 == 0 ) || ( t == 2 && l % 2 ))
			cout << t1[ r ] - ( l == 0 ? 0 : t1[ l - 1 ] );
		else cout << t2[ r ] - ( l == 0 ? 0 : t2[ l - 1 ] );
		cout << "\n";
	}
}
