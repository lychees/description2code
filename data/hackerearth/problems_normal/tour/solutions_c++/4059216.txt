#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    string f,s;
    cin>>n;
    map<string,int> mp;
    int c[n][n];
    for(int i=0;i<n;i++)
    {
        if(i==0)
        {
            cin>>f;
            mp[f]=i;
            continue;
        }
        cin>>s;
        mp[s]=i;
    }
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
        cin>>c[i][j];

    cin>>n;
    long long cost=0;

    for(int i=0;i<n;i++)
    {
        cin>>s;
        cost+=c[mp[f]][mp[s]];
        f=s;
    }

    printf("%lld\n",cost);

}
