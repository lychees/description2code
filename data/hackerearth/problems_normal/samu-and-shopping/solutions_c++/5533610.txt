    #include <bits/stdc++.h>
    using namespace std;
    #define endl '\n';
     
    int main()
    {
          ios_base::sync_with_stdio(false);cin.tie(0);
     
          //freopen("input.in","r",stdin);
         
              int t;cin>>t;while(t--)
              {
                int n;cin>>n;
                int v[3][n];
                for(int i=0;i<n;i++)
                      cin>>v[0][i]>>v[1][i]>>v[2][i];
                int dp[3][n];
                dp[0][0]=v[0][0];
                dp[1][0]=v[1][0];
                dp[2][0]=v[2][0];
                for(int i=1;i<n;i++)
                {
                  for(int j=0;j<3;j++)
                  {
                    int x,y;
                    if(j==0){x=1;y=2;}
                    if(j==1){x=0;y=2;}
                    if(j==2){x=1;y=0;}
                     dp[j][i]=v[j][i]+min(dp[x][i-1],dp[y][i-1]);
                  }
                }
                cout<<min(min(dp[0][n-1],dp[1][n-1]),(dp[2][n-1]))<<endl;
              }           
        return 0;
    }
