#include<bits/stdc++.h>
using namespace std;
int arr[1001][1001];
int sz,n,m;
int visit(int i,int j)
{
	arr[i][j]=2;
	int t=1;
	if(i-1>=1 && arr[i-1][j]==1) t+=visit(i-1,j);
	if(i+1<=n && arr[i+1][j]==1) t+=visit(i+1,j);
	if(i-1>=1 && j-1>=1 && arr[i-1][j-1]==1) t+=visit(i-1,j-1);
	if(i+1<=n && j+1<=m && arr[i+1][j+1]==1) t+=visit(i+1,j+1);
	if(j-1>=1 && arr[i][j-1]==1) t+=visit(i,j-1);
	if(j+1<=m && arr[i][j+1]==1) t+=visit(i,j+1);
	if(i+1<=n && j-1>=1 && arr[i+1][j-1]==1) t+=visit(i+1,j-1);
	if(j+1<=m && i-1>=1 && arr[i-1][j+1]==1) t+=visit(i-1,j+1);
	return t;
}
void func()
{
	int mx=0;
	for(int i=1;i<=n;i++)
	for(int j=1;j<=m;j++)
	{
		if(arr[i][j]==1){
			sz+=1;
			mx=max(mx,visit(i,j));
		}
	}
	cout<<sz<<" "<<mx<<endl;
}
int main()
{
	int t;
	scanf("%d",&t);
	
	while(t--)
	{
		scanf("%d%d",&n,&m);
		for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)scanf("%d",&arr[i][j]);
		sz=0;
		func();
	}
	
	return 0;
}
