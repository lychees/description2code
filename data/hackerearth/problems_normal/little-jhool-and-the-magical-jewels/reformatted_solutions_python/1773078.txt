t = input()
while t:
    t -= 1
    s = raw_input()
    r = 0
    u = 0
    b = 0
    y = 0
    for i in xrange(len(s)):
        if s[i] == 'r':
            r += 1
        if s[i] == 'u':
            u += 1
        if s[i] == 'b':
            b += 1
        if s[i] == 'y':
            y += 1

    print min(r, min(u, min(b, y)))
