#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,x,i,count=0;
		cin>>n;
		int a[1000];
		for(i=0;i<=100;i++)
		{
			a[i]=-1;
		}
		
		for(i=0;i<n;i++)
		{
			 cin>>x;
			 int index=x%10;
			 
			 if(a[index]==-1)
			 {
			 	 a[index]=x;
			 }
			 else
			 count++;
		}
		
		cout<<count<<"\n";
	}
}
