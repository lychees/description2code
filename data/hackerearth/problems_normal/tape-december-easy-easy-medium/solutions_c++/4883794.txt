long countOperations(long *a,long n,long k)
{
	long i=0,count=0,prev=INT_MIN;
	for(i=i+k-1;i<n;i=i+(2*k)-1)
	{
		long o=i;
//		cout<<" i = "<<i<<"  "<<" prev = "<<prev<<"    ";
		while(i>prev && a[i]==0)
		{
			
			i--;
		}
		if(i==prev)
		{
			
			cout<<"-1"<<"\n";
			return count;
		}
		prev=i+k-1;
		count++;
	}
	cout<<count<<"\n";
	return count;
}
