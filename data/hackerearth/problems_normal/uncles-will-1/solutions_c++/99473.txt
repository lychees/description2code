#include <iostream>
#include <cstring>
#include <algorithm>
#define INF 1000000000

using namespace std;

int a[202],n,m,c;
int dp[202][202];

int f(int idx, int sheets)
{
    if ( idx >= c ) return 0;
    if ( idx == c-1 ) {
         if ( sheets + 1 <= n ) return 1;
         return INF;
    }
    int &ans = dp[idx][sheets];
    if ( ans != -1 ) return ans;
    ans = INF;
    for ( int i = idx; i < c; i++ ) {
        if ( sheets + 1 <= n ) {
           ans = min(ans, a[i]-a[idx]+1 + f(i+1,sheets+1));
        }
    }
    return ans;
}
int main()
{
    int t,i;
    cin >> t;
    while ( t-- ) {
          cin >> n >> m >> c;
          for ( i = 0; i < c; i++ ) cin >> a[i];
          sort(a,a+c);
          int ans = INF;
          memset(dp, -1, sizeof(dp));
          for ( i = 0; i < c; i++ ) {
              ans = min(ans, a[i]-a[0]+1 + f(i+1,1));
          }
          if ( c == 1 && n >= 1 ) ans = 1;
          cout << ans << endl;
    }
    return 0;
}
