#include <iostream>
#include <string>
using namespace std;

int main()
{
    string s;
    cin >> s;
    int i=0,j=1,l = s.size();
    int ans =0,count;
    while(i<l)
    {
    	count = 1;
    	while(j<l && s[i]==s[j])
    	{
    		count+=1;
    		j++;
    	}
    	i=j;
    	j++;
    	ans = ans + (count*(count+1))/2;
    }
    cout << ans;
    return 0;
}
