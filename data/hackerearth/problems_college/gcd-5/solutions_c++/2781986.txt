#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n;
	long long int a[100000];
	cin >> t;
	while(t--)
	{
		int f=0;
		cin >> n;
		for(int i=0; i < n; i++)
		cin >> a[i];
		for(int i=0; i < n-1; i++)
		{
			if(a[i] > a[i+1])
			{
				
		     f=1;
			cout << "NO\n";
			break;
			}
		}
		if(f==0)
		cout << "YES\n";
		
	}
	
	
	
	
    //cout << "Hello World!" << endl;
    return 0;
}
