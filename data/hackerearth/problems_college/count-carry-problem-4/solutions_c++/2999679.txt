#include <iostream>
using namespace std;

#define lld long long int

int main()
{
    //cout << "Hello World!" << endl;
    lld n;
    lld a, b, c, count;
    
    cin >> n;
    
    while (n--) {
    	cin >> a >> b;
    	
    	c = 0;	count = 0;
    	while (a > 0 && b > 0) {
    		int la = a % 10;
    		int lb = b % 10;
    		
    		if (la + lb + c >= 10) {
    			c = 1;
    			count++;
    		} else {
    			c= 0;
    		}
    		 
    		a /= 10;
    		b /= 10;
    	}
    	
    	//cout << count << endl;
    	if (count == 0) {
    		cout << "No carry operation\n";
    	} else if (count == 1) {
    		cout << count << " carry operation\n";
    	} else {
    		cout << count << " carry operations\n";
    	}
    	
    }
    
    return 0;
}
