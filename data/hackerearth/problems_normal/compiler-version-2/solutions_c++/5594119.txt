#include <bits/stdc++.h>
using namespace std;

int main()
{	
	string line;
	while(getline(cin, line))
	{
		int l = line.length();
		for(int i=0;i<l;i++)
		{
			if(i<=l-2 && line[i] == '/' && line[i+1] == '/') break;
			if(i<=l-2 && line[i] == '-' && line[i+1] == '>') line.replace(i, 2, ".");
		}
		cout << line << endl;
	}
}
