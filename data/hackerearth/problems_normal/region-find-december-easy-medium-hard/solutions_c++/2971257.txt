#include <bits/stdc++.h>
using namespace std;
vector<int> g[1501];
int w[1501];
long long exactly[1501],greaterThan[1501],mod=1000000007;
int dfs(int u,int p,int k)
{
    for(int i=0;i<g[u].size();++i)
    {
        int v=g[u][i];
        if(v==p)
            continue;
        dfs(v,u,k);
    }
    if(w[u]<k)
    {
        greaterThan[u]=0;
        exactly[u]=0;
    }
    else if(w[u]==k)
    {
        greaterThan[u]=0;
        exactly[u]=1;
        for(int i=0;i<g[u].size();++i)
        {
            int v=g[u][i];
            if(v==p)
                continue;
            exactly[u]*=(1+exactly[v]+greaterThan[v]);
            exactly[u]%=mod;
        }
    }
    else
    {   
        greaterThan[u]=1;
        long long total=1;
        long long justGreater=1;
        for(int i=0;i<g[u].size();++i)
        {
            int v=g[u][i];
            if(v==p)
                continue;
            greaterThan[u]*=(1+greaterThan[v]);
            greaterThan[u]%=mod;
            total*=(1+exactly[v]+greaterThan[v]);
            total%=mod;
        }   
        exactly[u]=(total-greaterThan[u]+mod)%mod;
    }
}
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;++i)
        scanf("%d",&w[i]);
    for(int i=2;i<=n;++i)
    {
        int u,v;
        scanf("%d%d",&u,&v);
        g[u].push_back(v);
        g[v].push_back(u);
    }
    int q;
    scanf("%d",&q);
    while(q--)
    {
        int k;
        scanf("%d",&k);
        dfs(1,0,k);
        long long ans=0;
        for(int i=1;i<=n;++i)
            ans+=exactly[i];
        printf("%lld\n",ans%mod);
    }
}
