#include <iostream>
using namespace std;
int f(long long int arr[100000],int x)
{
	for(int i=2;i<100000;i++)
	{
		if(x%i==0)
		{
			while(x%i==0)
			{
				x=x/i;
				arr[i]++;
			}
		}
	}
	return x;
}
int main() {
     int t;
     cin>>t;
     while(t--)
     {
     	int a,b,c,d;
     	cin>>a>>b>>c>>d;
     	long long int arr[100000]={0};
     	long long int num[100000]={0};
     	a=f(arr,a);
     	c=f(num,c);
     	//cout<<arr[2]<<" "<<num[2]<<endl;
     	for(int i=0;i<100000;i++)
     	{
     		arr[i]=arr[i]*b -num[i]*d;
     	}
     	//cout<<a<<" "<<c<<endl;
     //	cout<<arr[2]<<endl;
     	int flag=0;
     	if(c!=0 && c>1)
     	flag=1;
     	for(int i=0;i<100000;i++)
     	{
     		if(arr[i]<0)
     		{
             
     			flag=1;
     			break;
     		}
     	}
     	if(flag==0)
     	{
     		cout<<"Divisible"<<endl;
     	}
     	else

     }
	return 0;
}
