#include <bits/stdc++.h>
#define ll long long

using namespace std;

int main()
{
    int t;

    cin >> t;

    while (t--) {
        ll n, i, x, sum;

        cin >> n;

        sum = 0;
        for (i = 0; i < n; i++) {
            cin >> x;

            sum += x;
        }

        if (sum % n == 0)
            cout << "YES\n" << (ll)(sum/n) << " 0\n";

        else
            cout << "NO\n" << (ll)(sum/n) << " " << (ll)(sum%n) << endl;
    }

    return 0;
}
