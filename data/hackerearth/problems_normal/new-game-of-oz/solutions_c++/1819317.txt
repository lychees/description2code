#include <bits/stdc++.h>
using namespace std;

int main()
{ 
	int T;
	int N;
	cin>>T;
	while(T--){
		cin>>N;
		vector<int>A(N+1);
		A[N]=20000;
		for(int i=0;i<N;i++)cin>>A[i];
		sort(A.begin(),A.end());
		int ans=0;
		for(int i=0;i<N;i++){
			if(A[i]+1==A[i+1]){
				ans++;i++;
			}
			else ans++;
		}
		cout<<ans<<endl;
	}
    return 0;
}
