#include <iostream>
using namespace std;

int main()
{
	int size =0;
	cin >> size;
	for (int i =0; i< size ; i++)
		cout<<"*";
		cout<<endl;
	for (int j = size-1; j>1; j--){
		for (int i=0; i<(j); i++){
			if (i < j-1)
			cout<<" ";
			else
			cout <<"*"<<endl;
		}
		
	}
	for (int i =0; i< size ; i++)
		cout<<"*";
	
   return 0;
}
