#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,i,j,n,k,l,flag;
    cin>>t;
    while(t--)
    {
    	flag=0;
    	cin>>n;
    		if(n%3==0)
    		{
    			for(i=0;i<n;i++)
    			cout<<"5";
    			cout<<endl;
    			continue;
    		}
    		j=n/3;
    		for(i=j;i>=0;i--)
    		{
    			k=n-(3*i);
    			if(k%5==0)
    			break;
    		}
    		if(((k/5)>=1)&&(k%5==0))
    		{
    			for(i=0;i<n-k;i++)
    			printf("5");
    			for(i=0;i<k;i++)
    			printf("3");
    			cout<<endl;
    		}
    		else if(n%5==0)
    		{
    		for(i=0;i<n;i++)
    		{
    			printf("3");
    		}
    		cout<<endl;
    		}
    		else
    		{
    		printf("-1\n");
    		
    		}
    }
    return 0;
}
