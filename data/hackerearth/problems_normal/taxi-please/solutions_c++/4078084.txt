#include <iostream>
#include <algorithm>
using namespace std;

struct user{    int idx, s, j;};
bool myfunc (struct user u1, struct user u2) { return (u1.s < u2.s); }
int main()
{
    int n, m;           scanf("%d%d", &n, &m);
    struct user u[n];
    int t[m], result[n];
    for(int i = 0 ; i < n ; i++){       u[i].idx = i; scanf("%d%d", &u[i].s, &u[i].j);}
    sort(u, u + n, myfunc);
    for(int k = 0 ; k < m ; k++){       t[k] = -1;}
    t[0] = 0;
    result[u[0].idx] = 1;
    for(int j = 1 ; j < n ; j++){
        int i;
        for(i = 0; i < m; i++){
            if((t[i] == -1) ||((t[i] != -1) && (u[j].s >= u[t[i]].s + u[t[i]].j))){
                t[i] = j;
                result[u[j].idx] = i + 1;
                break;
            }
        }
        if(i == m)      result[u[j].idx] = -1;
    }

    for(int k = 0 ; k < n ; k++)        printf("%d ",result[k]);
    return 0;
}
