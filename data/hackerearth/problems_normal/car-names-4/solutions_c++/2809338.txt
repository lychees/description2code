#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    for(int j=0;j<n;j++)
    {
         char a[10000],d[3];
         cin>>a;
         int c=1,t=0,i=0,b[10000];
         while(a[i]!='\0')
         {
             if(a[i]==a[i+1])
             {
                 c++;
             }
             else
             {
                 d[t]=a[i];
                 b[t]=c;
                 c=1;
                 t++;

             }
             if(t>3)
                break;
             i++;
         }
         if(d[0]==d[2])
         {
              cout<<"Not OK"<<endl;
         }
         else if(t>3||b[0]!=b[1]||b[0]!=b[2]||b[1]!=b[2])
         {
             cout<<"Not OK"<<endl;
         }
         else
            cout<<"OK"<<endl;
    }


    return 0;
}
