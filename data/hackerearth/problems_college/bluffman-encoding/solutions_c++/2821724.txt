#include<iostream>
#include<stdio.h>
#include<string>
using namespace std;

class A
{
    int t, n, i, ar[26], j, tmp, x , y , count;
    char arr[26], temp, ans[95];
    string a;

public:
    void call()
    {
        cin>>t;
        do
        {
            cin>>n;
            for(i=0; i<n; i++)
            {
                cin>>arr[i];
                cin>>ar[i];
            }
            cin>>a;
            for(i=0;i<n;i++)
            {
                for(j=0;j<n-1-i;j++)
                {
                    if(ar[j]>ar[j+1])
                    {
                        tmp = ar[j];
                        ar[j] = ar[j+1];
                        ar[j+1] = tmp;

                        temp = arr[j];
                        arr[j] = arr[j+1];
                        arr[j+1] = temp;
                    }
                }
            }

            x = y = 0;
            do
            {
                count=0;
                while(count<n-1 && x<=a.length()-1)
                {
                    if(a[x] == '1')
                    {
                        count++;
                        x++;
                    }
                    else
                    {
                        x++;
                        break;
                    }
                }

                ans[y] = arr[count];
                y++;
            }while(x<=a.length()-1);
            ans[y] = '\0';
            cout<<ans<<endl;

            t--;
        }while(t);
    }
};

int main()
{
    A obj;
    obj.call();
    return 0;
}
