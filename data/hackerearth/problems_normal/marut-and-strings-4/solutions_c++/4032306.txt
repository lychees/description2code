#include<bits/stdc++.h>

using namespace std;

int main()
	{
		int t,T,i;
		cin >> T;
		if(T<1 || T>10)
			cout << "Invalid Test" <<"\n";
		else{
		for(t=0;t<T;t++)
			{
				string s;
				int COUNT[26]={0};
				int count[26]={0};
				int cap=0;
				int small=0;
				int others=0;
				cin >> s;
				int leng=s.length();
				if(leng<1 || leng >100)
					printf("Invalid Input\n");
				else{
				for(i=0;i<leng;i++)
					{
						int x=s[i];
						if(x>=65 && x<=90)
							{
								int y=s[i]-65;
								COUNT[y]++;
								cap++;
							}
						else if(x>=97 && x<=122)
							{
								int y=s[i]-97;
								count[y]++;
								small++;
							}
						else
							others++;
					}	
				if(small<=cap && others==0 && small!=0 && cap!=0)
					cout << small;
				else if(cap<small && others==0 && small!=0 && cap!=0)
					cout << cap;
				else if((others !=0 ||others==0) &&(cap<=small) && (cap!=0 || small!=0))
					cout << cap;
				else if((others !=0 ||others==0) &&(small<cap) && (cap!=0 || small!=0))
					cout << small;
				
				else
					cout <<"Invalid Input";
				printf("\n");}
			}}
		return 0;
	}
