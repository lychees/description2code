#include<iostream>
#include<vector>
using namespace std;

int main()
{
	int n,q,type,l,r;
	cin>>n>>q;
	vector<long long int> veca(n+1,0);
	vector<long long int> vecb(n+1,0);
	for(int i=1;i<=n;i++)cin>>veca[i];
	for(int i=1;i<=n;i++)cin>>vecb[i];
	for(int i=2;i<=n;i++)veca[i]+=veca[i-2];
	for(int i=2;i<=n;i++)vecb[i]+=vecb[i-2];
	while(q--){
	cin>>type>>l>>r;
	int tempa=l%2;
	int tempb=r%2;
	if(type==1){
	if(tempa==tempb){
	if(l==1)
	cout<<veca[r]+vecb[r-1]-vecb[l-1]<<endl;
	else
    cout<<veca[r]-veca[l-2]+vecb[r-1]-vecb[l-1]<<endl;
	}
    else{
    if(l==1)
    cout<<veca[r-1]+vecb[r]-vecb[l-1]<<endl;
    else
    cout<<veca[r-1]-veca[l-2]+vecb[r]-vecb[l-1]<<endl;
    }
	}
	else{
	if(tempa==tempb){
	if(l==1)
	cout<<vecb[r]+veca[r-1]-veca[l-1]<<endl;
	else
    cout<<vecb[r]-vecb[l-2]+veca[r-1]-veca[l-1]<<endl;
	}
    else{
    if(l==1)
    cout<<vecb[r-1]+veca[r]-veca[l-1]<<endl;
    else
    cout<<vecb[r-1]-vecb[l-2]+veca[r]-veca[l-1]<<endl;
    }
	}
	}
    return 0;
}
