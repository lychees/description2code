#include <iostream>
#include<math.h>
using namespace std;
int gcd(int,int);
int n;
int main()
{
	int a,b,c,k,counter;
	int i,j,t;
	scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	counter=0;
    for (i=2; i <=(sqrt(n-1) + 1); i++)
          for (j = 1; j < i; j++)
              if ((gcd(i,j) == 1) && ((i-j) % 2) && ((i*i + j*j) <= n))
              {
                    k = 0;
                    a = i * i - j * j;
                    b = 2 * i * j;
                    c = i * i + j * j;
                    while ((++k) * c <= n)
                      counter++;
              }
    
    
    printf("%d\n",counter);
    }
    return 0;
}
int gcd(int x,int y)
{
	while(x!=y)
	{
		if(x>y)
		{
			x=x-y;
		}
		else{
			y=y-x;
		}
	}
	return x;
}
