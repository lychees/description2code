#include<stdio.h>
int b[100001] = {0};
int c[100001] = {0};
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,sum=0,q;
	scanf("%d %d",&n,&q);
	int i , j , k;
	for(i=1; i<=n; i++)
	{
		scanf("%d", &b[i]);
		sum = sum + b[i];
		c[i] = sum;
	}
	int  l , r;
	while(q--)
	{
		scanf("%d %d", &l,&r);
		sum = c[r] - c[l-1];
		printf("%d\n",sum);
	}
}
