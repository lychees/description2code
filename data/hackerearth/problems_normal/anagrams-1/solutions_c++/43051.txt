#include <iostream>
#include <string.h>
using namespace std;

string process(char *str1, char * str2)
{
	int count1[26];
	int count2[26];
	int count3[26];
	
	memset(count1, 0, sizeof(int)*26);
	memset(count2, 0, sizeof(int)*26);
	memset(count3, 0, sizeof(int)*26);
	int len1 = strlen(str1);
	int len2 = strlen(str2);
	for(int i =0;i < len1;i++)
	{
		if(str1[i]>='a' && str1[i]<='z')
			count1[str1[i]-'a']++;
		if(str1[i]>='A' && str1[i]<='Z')
			count1[str1[i]-'A']++;
	}
	for(int i =0;i < len2;i++)
	{
		if(str2[i]>='a' && str2[i]<='z')
			count2[str2[i]-'a']++;
		if(str2[i]>='A' && str2[i]<='Z')
			count2[str2[i]-'A']++;
	}
	int newLen1 =0, newLen2 = 0;
	for(int i =0; i < 26; i++)
	{
		newLen1 += count1[i];
		newLen2 += count2[i];
	}
	int count = 0, oddCount = 0;
	for(int i =0; i < 26; i++)
	{
		if(count1[i] == count2[i])
			count++;
		else 
		{
			if(newLen1 == newLen2)
				return "NO LUCK";
			if(newLen1>newLen2 && count1[i] < count2[i])
				return "NO LUCK";
			if(newLen1<newLen2 && count1[i] > count2[i])
				return "NO LUCK";
			count3[i] = count1[i] - count2[i];
			count3[i] = count3[i] > 0 ? count3[i]: -count3[i];
			if(count3[i]%2 != 0)
				oddCount++;
			if(oddCount > 1)
				return "NO LUCK";
		}
	}
	if(count == 26)
		return "YES";
	char finalStr[101];
	memset(finalStr, 0, 101*sizeof(char));
	int end = newLen1 - newLen2;
	end = end > 0 ? end: -end;
	end= end -1;
	int start = 0;
	for(int i = 0; i < 26; i++)
	{
		if(count3[i] >= 1)
		{
			int loop = count3[i]-count3[i]%2;
			for(int j = 0;j < loop;j+=2)
			{
				finalStr[start++] = 'a'+i;
				finalStr[end--] = 'a'+i;
			}
			if(count3[i]%2 != 0)
				finalStr[(start+end)/2] = 'a'+i;
		}
	}
	return finalStr;
}
