bool find_area(int x1,int y1,int x2,int y2,int x3,int y3){
    int area = abs(x1 * (y2 - y3) - y1 * (x2 - x3) + (x2 * y3 - x3 * y2)) ;
    return area ;
}

bool check(int x1,int y1,int x2,int y2,int x3,int y3){

    if(!find_area(x1,y1,x2,y2,x3,y3))
       return 0 ;
    int a = (x2 - x1)*(x2 - x1) + (y2 - y1)*(y2 - y1) ;
    int b = (x3 - x2)*(x3 - x2) + (y3 - y2)*(y3 - y2) ;
    int c = (x3 - x1)*(x3 - x1) + (y3 - y1)*(y3 - y1) ;
    if(a < b)
        swap(a,b) ;
    if(a < c)
        swap(a,c) ;
    return (a == c + b) ;
}

int main(){
	int t ;
	cin >> t ;
	while(t--){
    int x1,y1,x2,y2,x3,y3 ;
    cin >> x1 >> y1 ;
    cin >> x2 >> y2 ;
    cin >> x3 >> y3 ;
    int ans = -1 ;
    if(check(x1,y1,x2,y2,x3,y3)){
        ans = 0 ;
    }else{
        int A[6] ;
        for(int i=0;i<6;i++){
            A[0] = x1 ;
            A[1] = y1 ;
            A[2] = x2 ;
            A[3] = y2 ;
            A[4] = x3 ;
            A[5] = y3 ;
            A[i] -- ;
            if(check(A[0],A[1],A[2],A[3],A[4],A[5]))
                ans = 1 ;
            A[0] = x1 ;
            A[1] = y1 ;
            A[2] = x2 ;
            A[3] = y2 ;
            A[4] = x3 ;
            A[5] = y3 ;
            A[i] ++ ;
            if(check(A[0],A[1],A[2],A[3],A[4],A[5]))
                ans = 1 ;
        }
    }
    puts(ans == -1 ? "NO" : "YES" ) ;
	}
	return 0 ;
}
