a=raw_input()
a=a.split()
rw=int(a[0])
cm=int(a[1])
mls=[]
for i in xrange(rw):
    sls=raw_input()
    sls=sls.split();
    sls=map(int,sls)
    mls.append(sls)
xls=[]
for i in xrange(rw):
    xls.append(sum(mls[i]))
yls=[]
for i in xrange(cm):
    sm=0
    for j in xrange(rw):
        sm=sm+mls[j][i]
    yls.append(sm)
h=xls[0]^yls[0]
for i in xls:
    for j in yls:
        if((i^j)>h):
            h=i^j
print h
