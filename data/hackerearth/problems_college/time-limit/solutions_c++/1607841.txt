#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n,m,x;
	cin>>n>>m;
	int a[100],b[100];
	for( x=0;x<n;x++)
	{
		cin>>a[x];

	}
	for(x=0;x<m;x++)
	{
		cin>>b[x];
	}
	sort(a,a+n);
	sort(b,b+m);
	x=max(2*a[0],a[n-1]);
	x=x<b[0]?x:-1;
	cout<<x;
	return 0;
}
