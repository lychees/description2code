def mult(b):
    sum = 1
    while True:
        if sum == b:
            return 1
        sum *= 2
        if sum > b:
            return sum/2
        elif sum == b:
            return sum


t = input("")
while t:
    a = input("")
    b = a
    count = 0
    sum = 0
    c = 0
    while True:
        if sum == a:
            break
        b -= c
        c = mult(b)
        sum += c
        count += 1
    print(count)
    t -= 1
