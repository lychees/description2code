#include<iostream>
#include<string.h>
#define size 100
using namespace std;
int a[125],c[125];
int main()
{
    int n,i,count=0;
    for(i=0;i<125;i++)
        a[i]=0;
    string b;
    scanf("%d",&n);
    cin>>b;
    for(i=0;i<b.length();i++)
        a[b[i]]++;
    n--;
    while(n--)
    {
        for(i=0;i<125;i++)
            c[i]=0;
        cin>>b;
        for(i=0;i<b.length();i++)
          c[b[i]]++;
        for(i=97;i<=122;i++)
        {
            if(a[i]!=0 && c[i]!=0)
                ;
            else
                a[i]=0;
        }
    }
    for(i=97;i<=122;i++)
    {
        if(a[i]!=0)
            count++;
    }
    printf("%d",count);
    return 0;
}
