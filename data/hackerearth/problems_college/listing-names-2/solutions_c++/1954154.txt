#include <iostream>
#include <map>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
	int t;
	cin >> t;
	map<string, int> mp;
	vector<pair<string, int> > v;
	int idx = 0;
	
	while (t--) {
		string str;
		cin >> str;
		int p;
		if (p = mp[str]) {
			++v[p - 1].second;
		} else {
			v.push_back({str, 1});
			mp[str] = ++idx;
		}	
	}
	sort(v.begin(), v.end());
	
	for (int i = 0; i<v.size(); i++) {
		cout << v[i].first << " " << v[i].second << endl;
	}
	
    return 0;
}
