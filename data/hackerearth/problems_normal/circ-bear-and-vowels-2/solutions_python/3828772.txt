'''

t = input()
for i in range(t):
    cons = 0
    vow = 0
    temp = 0
    isHard = False
    s = raw_input()
    for ch in s:
        if ch in  'aeiouy':
            vow += 1
            temp = 0
        else:
            cons += 1
            temp += 1
            if temp == 3:
                isHard = True
    if isHard or cons > vow:
        print 'hard'
    else:
        print 'easy'
