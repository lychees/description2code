#include <iostream>
#include <string>
#include <stdlib.h> 
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s1,s2,s3,s4;
    	cin>>s1>>s2;
    	int i,l,flag=0;
    	s3 = s1+s2;
    	s4 = s2+s1;
    	l = s3.length();
    	for(i=0;i<l/2;i++)
    	{
    		if(s3[i]==s3[l-i-1] || s4[i]==s4[l-i-1])
    		{
    			flag++;
    		}
    		else break;
    	}
    	if(flag==l/2)
    	cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
    return 0;
}
