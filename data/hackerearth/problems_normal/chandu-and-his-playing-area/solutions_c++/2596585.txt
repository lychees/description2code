#include <iostream>
#include <string.h>
#include <string>
#include <algorithm>

using namespace std;

void test()
{
	long long N, area;
	cin >> N;
	if((N % 2) == 0){
		long long tmp = N / 2;
		if((tmp % 2) == 0){
			area = (tmp / 2) * (tmp / 2);
		}else{
			long long a, b;
			a = tmp / 2;
			b = tmp - a;
			area = a * b;
		}
	}else{
		N--;
		long long tmp = N / 2;
		if((tmp % 2) == 0){
			area = (tmp / 2) * (tmp / 2);
		}else{
			long long a, b;
			a = tmp / 2;
			b = tmp - a;
			area = a * b;
		}
	}
	cout << area << endl;
}

int main()
{
	int T;
	cin >> T;
	while(T--)
		test();
	return 0;
}
