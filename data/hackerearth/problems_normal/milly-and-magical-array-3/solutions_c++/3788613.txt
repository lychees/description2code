#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    unsigned long long int noofcases,N,X,i,j;
    double count=0;
    cin>>noofcases;
    cout<<fixed<<setprecision(0);
    for(i=0;i<noofcases;++i)
    {
    	count=0;
    	cin>>N>>X;
    	double A[N];
    	cin>>A[0];
    	for(j=1;j<N;++j)
    	{
    		cin>>A[j];
    		//while(A[j-1]>=A[j])
    		//{
    		//	A[j]+=X;
    		//	++count;
    		//}
    		if(A[j-1]>=A[j])
    		{
    			long long int temp = A[j-1]-A[j];
    			long long int fact = 1;
    			if(temp%X==0&&temp!=0)
    			{
    				count=count+(temp/X);
    				fact=(temp/X);
    			
    			}else
    			{
    				count=count+(temp/X)+1;
    				fact=(temp/X)+1;
     			}
    			A[j]+=(X*fact);
    		}
    	}
    	cout<<count<<"\n";
    	
    }
    return 0;
}
