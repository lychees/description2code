#include <iostream>
#include <algorithm>
using namespace std;
long long C(long long int n, long long int r) {
    if(r > n / 2) r = n - r; 
    long long ans = 1;
    int i;

    for(i = 1; i <= r; i++) {
        ans *= n - r + i;
        ans /= i;
    }

    return ans%1000000007;
}
int main()
{
	long long int T,N[3];
	cin >>T;
	while(T--){
		cin >> N[0]>>N[1]>>N[2];
		int count[3] = {0};
		for(int i=0;i<3;++i){
			if(N[i]==1){
				count[1]++;
			}
			else if(N[i] == 2){
				count[2]++;
			}
		}
		if(count[1]<2 && count[2]<3){
		sort(N,N+3);
        cout << (C(N[0],1)%1000000007 * C(N[1]-1,1)%1000000007 * C(N[2]-2,1)%1000000007)%1000000007 <<endl;
		}
		else{
			cout << 0 <<endl;
		}
	}
    return 0;
}
