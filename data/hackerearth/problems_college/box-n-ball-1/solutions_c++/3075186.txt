#include <bits/stdc++.h>
using namespace std;
#define gc() getchar_unlocked()
inline long long int read_int()
{
	char c = gc();
    	while(c<'0' || c>'9')
    	c = gc();
    	long long int ret = 0;
    	while(c>='0' && c<='9')
    	{
    		ret = (ret<<3) + (ret<<1) + c - 48;
    		c = gc();
    	}
    	return ret;
}
int main()
{
	long long int t;
	double l,b,h,min;
	t = read_int();
	while(t--)
	{
		cin >> l;
		min = l;
		cin >> b;
		if(b<min)
		{
			min = b;
		}
		cin >> h;
		if(h<min)
		{
			min = h;
		}
		printf("%.1lf\n",min/2);
	}	
	return 0;
}
