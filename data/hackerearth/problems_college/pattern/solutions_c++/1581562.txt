#include <iostream>
using namespace std;

int main () {

	int n;
	cin >> n;

	for (int k = 0; k < n; k++) {
		cout << "*"; }
	cout << endl;

	for (int k = 0; k < n-2; k++) {
		for (int d = 0; d < n-k-2; d++) 
		cout << " ";
		cout << "*" << endl; }
	

	while (n--) {
	cout << "*"; }	

	return 0;
}
