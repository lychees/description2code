#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    long n,k,a[100000],i;
    long long b[100000],max;
    while(t--)
    {
    	cin>>n>>k;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	    a[i]=(k/a[i]);
    	}
    	
    	for(i=0;i<n;i++)
    	cin>>b[i];
    	max=b[0]*a[0];
    	for(i=0;i<n;i++)
    	{	
    		b[i]=b[i]*a[i];
    		if(b[i]>max) max=b[i];
    	}	
    	cout<<max<<endl;
    }
    return 0;
}
