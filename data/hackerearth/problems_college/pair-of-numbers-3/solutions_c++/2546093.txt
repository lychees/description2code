#include <iostream>
using namespace std;

int main()
{
    unsigned long long int n , t ;
    cin>>t;
    while(t--)
    {
    	unsigned long long int cnt = 0;
    	cin>>n;
    	for(int  i = 1 ; i <= n ; i++)
    	{
    		for(int j = i + 1 ; j <= n ; j++)
    		{
    			if( (i * j) % (i + j) == 0)
    				cnt++;
    		}
    	}
    	cout<<cnt<<endl;    	
    }
    return 0;
}
