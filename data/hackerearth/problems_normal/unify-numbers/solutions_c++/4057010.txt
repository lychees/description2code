#include <iostream>
#include <vector>
#include <stdlib.h>
#include <algorithm>
using namespace std;

bool mycomp(long long a, long long b){
	return a <b;
}

int main()
{
	int t=0;
	cin >> t;
	long long n,temp,sum,average,res;
	vector< long long> array;
	while(t--){
		sum=0;res=0;
		array.clear();
		cin >> n;
		for(unsigned long int i=0;i<n;i++){
			cin >> temp;
			//sum += temp;
			array.push_back(temp);
		}
		std::sort(array.begin(),array.end(),mycomp);
		average = array[array.size()/2];
		for(unsigned long int i=0;i<n;i++){
			res += abs(array.at(i) - average);
		}
		cout << res <<endl;
	}
    return 0;
}
