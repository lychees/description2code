#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t, n, arr[1000001], total;
    scanf("%d", &t);
    while (t--) {
    	total = 0;
    	scanf("%d", &n);
    	for (int i = 0; i < n; i++)
    		scanf("%d", &arr[i]);
    	for (int i = 0; i < n; i++) {
    		for (int j = i; j < n; j++) {
    			if (arr[j] == arr[i])
    				total += 1;
    		}
    	}
    	printf("%d\n", total);
    }
    return 0;
}
