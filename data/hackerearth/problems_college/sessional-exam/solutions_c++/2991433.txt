#include <iostream>
using namespace std;

int main()
{
    int n,m,i,j,k;
    cin>>n>>m;
    char s[n][m];int arr[n][m],stud[n],count=0;
    for(i=0;i<n;i++)stud[i]=0;
    for(i=0;i<n;i++)
    {
    	for(j=0;j<m;j++)
    	{
    		cin>>s[i][j];
    		arr[i][j]=int(s[i][j]);
    	}
    }
    int max=0;
    for(j=0;j<m;j++)
    {max=0;
    for(i=0;i<n;i++)
    { if(arr[i][j]>max)
     	{max=arr[i][j];
     	}
    }
    for(k=0;k<n;k++)
    {
    	if(max==arr[k][j])stud[k]++;
    }
    }
    for(i=0;i<n;i++)
    {
    	if(stud[i]>0)count++;
    }
    cout<<count<<endl;
    return 0;
}
