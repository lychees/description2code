#include<bits/stdc++.h>
using namespace std;
const int MAXN=1e5+7;
long long int dp[MAXN][2][2];
const int mod=1e9+7;
int main(){
	dp[1][0][0]=1;
	dp[1][1][0]=1;
	for(int i=2;i<1e5+2;i++){
		dp[i][0][0]=(dp[i-1][1][0])%mod;
		dp[i][0][1]=(dp[i-1][0][0]+dp[i-1][1][1])%mod;
		dp[i][1][0]=(dp[i-1][1][0]+dp[i-1][0][0])%mod;
		dp[i][1][1]=(dp[i-1][1][1]+dp[i-1][0][1])%mod;
	}
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		cout<< (dp[n][1][1]+dp[n][0][1])%mod <<"\n";
	}
}
