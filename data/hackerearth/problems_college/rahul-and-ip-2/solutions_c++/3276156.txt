#include<bits/stdc++.h>
#define sz(x) (int)(x.size())
 
using namespace std;
typedef long long lld;
 
 
int conv(string s)
{
	int ans = 0;
	for(int i = 0; i < sz(s); i++)
	 ans = (ans* 10) + (s[i] - '0');
	return ans;
}
 
int valid(string s)
{
   if(sz(s) > 3) return 0;
   if(s[0] == '0' && sz(s) > 1) return 0;
   int num = conv(s);
   if(num >= 0 && num <= 255) return 1;
   return 0;
}
 
int main()
{
    string s;
    cin >> s;
    vector< pair<int, pair<int, pair<int, int>  > > > vec;
    for(int i = 1; i < sz(s); i++)
    {
       for(int j = i+1; j < sz(s); j++)
       {
       	 for(int k = j+1; k < sz(s); k++)
       	 {
       	    string u, v, w, x;
       	    u = s.substr(0,i);
       	    v = s.substr(i, j-i);
       	    w = s.substr(j, k-j);
       	    x = s.substr(k);
       	    if(valid(u) && valid(v) && valid(w) && valid(x))
       	    {
       	    	vec.push_back(make_pair(conv(u), make_pair(conv(v), make_pair(conv(w), conv(x)))));
       	    }
       	    
       	 }
       }
    }
    sort(vec.begin(), vec.end());
    cout << sz(vec) << endl;
    for(int i = 0; i < sz(vec); i++)
    {
    	cout << vec[i].first << "." << vec[i].second.first << "." << vec[i].second.second.first << "." << vec[i].second.second.second << endl;
    }
	return 0;
}
