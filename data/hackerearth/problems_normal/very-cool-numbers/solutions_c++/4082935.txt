#include<bits/stdc++.h>
using namespace std;
int c[100005];
int main()
{
	for(int i=1;i<=100000;i++)
	{
		for(int j=0;j<20;j++)
		{
			if((i&(1<<j))&&(!(i&(1<<j+1)))&&(i&(1<<j+2)))
			c[i]++;
		}
	}
	int t;
	cin>>t;
	while(t--)
	{
		int r,k;
		cin>>r>>k;
		int t=0;
		for(int i=1;i<=r;i++)
		{
			if(c[i]>=k)
			t++;
		}
		cout<<t<<endl;
	}
}
