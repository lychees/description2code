#include <iostream>
using namespace std;

int main()
{
    int t,i,c,m,f,which;
    string a,b;
    
    cin>>t;
    while(t--){
    	cin>>a>>b;
    	
    	if(a.size()!=b.size()){
    		f=0;	
    		m = (a.size()>b.size())?b.size():a.size();
			which = (a.size()>b.size())?1:2;
    	}else{
    		f=1;
    		m = a.size();
    	}
    	c=0;
    	if(f){
    		for(i=0;i<m;i++){
    			if(a[i]!=b[i]){
    				f=0;
    				c++;
    			}
    		}
    	}else{
    		for(i=0;i<m;i++){
    			if(a[i]!=b[i]){
    				f=0;
    				c++;
    			}
    		}
    		if(which==1){
    			c+=a.size()-m;
    		}else{
    			c+=b.size()-m;
    		}
    	}
    	if(f){
    		cout<<"Exact"<<endl;
    	}else{
    		cout<<"Non Exact "<<c<<endl;
    	}
    }
    
    return 0;
}
