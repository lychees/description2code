#include <iostream>
#include<algorithm>
using namespace std;

int fun(int a[],int st,int end,int k)
{
	if(st>end)
     return -1;
	
	int mid=(st+end)/2;
	if(a[mid]==k)
	return 1;
   else
   if(a[mid]>k)
  return fun(a,st,mid-1,k);
   else
  return fun(a,mid+1,end,k);
}
int main()
{
    //cout << "Hello World!" << endl;
   
   int n,a[100010],i,m,cnt=1,p;
   cin>>n>>m;
   for(i=0;i<n;i++)
   cin>>a[i];
    sort(a,a+n);   
   while(1)
   {
   	 p=fun(a,0,n,m*cnt);
   	 //cout<<p<<" ";
   	 if(p==-1)
   	 {   //if(cnt==1)
   	      //cnt=0;
   	 	break;
   	 }
   	 cnt++;
   	 
   }
   cout<<m*(cnt-1)<<"\n";
   
   return 0;
}
