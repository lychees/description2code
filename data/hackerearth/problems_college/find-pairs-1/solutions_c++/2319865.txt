#include <iostream>
using namespace std;

int main()
{	int t;
	cin>>t;
	while(t--)
	{
		int n,a[10000],count=0;
		cin>>n;
		for(int i=0;i<n;i++)
		cin>>a[i];
		
		
		
		for(int i=0;i<n;i++)
		{
			for(int j=i+1;j<n;j++)
			{
				if(a[j]==a[i])
				count++;
			}
		}
		cout<<count+n<<endl;
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
