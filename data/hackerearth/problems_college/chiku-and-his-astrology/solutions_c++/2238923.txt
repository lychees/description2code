#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s;
    	int count1=0,count0=0;
    	cin>>s;
    	int len = s.length();
    	for(int i=0;i<len;i++){
    		if((int)s[i]-48 == 0)count0++;
    		else count1++;
    	}
    	cout<<abs(count1-count0)<<endl;
    }
    return 0;
}
