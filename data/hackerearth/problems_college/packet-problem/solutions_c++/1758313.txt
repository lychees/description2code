#include <iostream>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    int n;
    unsigned long *z = new unsigned long[tc];
    unsigned long *o = new unsigned long[tc];
    for(int i=0; i<tc; i++) {
    	cin>>n;
    	z[i] = 0;
    	o[i] = 1;
    	for(int j=0; j<n; j++) {
    		unsigned long tmp0 = z[i];
    		z[i] += o[i];
    		o[i] = tmp0;
    	}
    }
    for(int i=0; i<tc; i++)
    	cout<<o[i]<<' '<<z[i]<<endl;
    return 0;
}
