#!/usr/bin/python
n = int(input())
all = [int(x) for x in raw_input().split(' ')]

prime = []

for i in all:
    isprime = 1
    for j in all:
        if i != j and i % j == 0:
            isprime = 0
    if isprime == 1:
        prime.append(i)
primes = ''
for i in prime:
    primes = primes + str(i) + ' '
print(primes)
