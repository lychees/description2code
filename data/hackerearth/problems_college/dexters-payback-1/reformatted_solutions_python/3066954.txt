rr = raw_input


def help(X):
    q, r = X/4, X % 4
    r = min(2, r)
    q *= 2
    nominal = q+r
    rever = X-nominal
    return nominal, rever


def solve(X, Y):
    nx, rx = help(X)
    ny, ry = help(Y)
    return nx*ny + rx*ry


for testcase in xrange(int(rr())):
    N, M = map(int, rr().split())
    print max(solve(N, M), solve(M, N))
