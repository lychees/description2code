#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main()
{
    int n,s=0,k=0,i,j,p,q;
    cin >> n;
    int a[100][100];
    for(i=0;i<n;i++)
    {
       for(j=0;j<n;j++)
       {
          cin >> a[i][j];
       }
    }
    for(i=0;i<n;i++)
    {
       for(j=0;j<n;j++)
       {
          if(i==j)
              {
              s=s+a[i][j];
              
              }
           if(i+j==n-1)
               {
               k=k+a[i][j];
               
           }
       }
    }
    if(k-s>=0)
        {
    cout<<k-s;
    }
    else 
        {
     cout<<s-k;
    }
    return 0;
}
