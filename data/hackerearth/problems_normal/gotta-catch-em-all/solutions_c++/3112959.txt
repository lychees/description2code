#include <iostream>
#include<algorithm>
using namespace std;

bool compare(int a,int b)
{
	return (a>b);
}
int main()
{
    int k;
    cin>>k;
    int arr[k];
    for(int i=0;i<k;i++)
    {
    	cin>>arr[i];
    	arr[i]=arr[i]+1;
    }
    //sort in reverse order
    sort(arr,arr+k,compare);
    
    long count=0, cdays=0;
    for(int i=0;i<k;i++)
    {
    	if(arr[i]>cdays)
    	{
    		count+=arr[i]-cdays;
    		cdays=arr[i]-1;
    	}
    	else
    		cdays--;
    }
    
    cout<<count+1;
    return 0;
}
