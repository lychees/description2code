#include<bits/stdc++.h>
using namespace std;
bool mark[(int)1e7+3];
int dp[(int)1e7+3];
void erasthones()
{
	for(int i=2;i<=(int)1e7;++i)
	{
		if(mark[i]==0)
		{
			for(int j=2*i;j<=(int)1e7;j+=i)
			{
				mark[j]=1;
			}
		}
	}
	for(int i=2;i<=(int)1e7;++i)
	{
		if(mark[i]==0)
			dp[i]=dp[i-1]+1;
		else dp[i]=dp[i-1];
	}
	return ;
}
int main()
{
	int tc;
	cin>>tc;
	memset(dp,0,sizeof dp);
	memset(mark,0,sizeof mark);
	dp[0]=0;
	dp[1]=0;
	erasthones();
	while(tc--)
	{
		int l,r;
		cin>>l>>r;
		cout<<dp[r]-dp[l-1]<<endl;
	
	}
	return 0;
}
