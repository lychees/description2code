#include <iostream>
using namespace std;
#define MAX 500

int mul( int x , int res[] , int res_size )
{
		int carry = 0 ;
		int i;
		int prod ;
		for( i= 0 ; i<res_size ; i++)
		{
			prod = res[i]*x + carry ;
			res[i]=prod%10 ;
			carry = prod/10;
		}
		
		while(carry)
		{
			res[res_size] = carry%10;
			carry = carry/10;
			res_size++;
		}
	
	return res_size;
}
void factorial( int n )
{
	int res[MAX] ;
	int res_size = 1 ;
	res[0] = 1 ;
	int i = 2 ;
	for( i = 2 ; i <= n ; i++ )
	res_size = mul( i , res , res_size );
	
	for( i=res_size-1 ; i>=0 ; i--)
	printf("%d",res[i]);
	
	printf("\n");
}

int main()
{
 	int tc ; 
 	int n ;
 	int i = 0 ;
 	cin>>tc;
 	if( tc <= 100 )
 	{
 		for( i = 0 ; i < tc ; i++)
 		{
 			cin>>n;
 			if( n <= 100)
 			factorial(n); 		
 		}
 	}
    return 0;
}
