def isPrime(val):
    if val==1:
        return False
    if val==2:
        return True
    for subVal in xrange(2,val/2+1):
        if val%subVal==0:
            return False
    return True
            
def findPrimeAbove(num):
    for i in xrange(1,100000):
        if isPrime(i):
            num=num-1
        if num==0:
            return i
        else:
            continue
        
    
def challenge():
    t=raw_input()
    number=findPrimeAbove(num=int(t))
    print "%s" %number

if __name__=="__main__":    
    challenge()
