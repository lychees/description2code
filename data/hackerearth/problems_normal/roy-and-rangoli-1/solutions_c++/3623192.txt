#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pb push_back
#define MOD 1000000007
int prim[2000055];
long long pre[2000055];
void sieve()
{  ll i, c ,d; 
   // memset(pre,0,sizeof(pre));
   // memset(prim,0,sizeof(prim));
   prim[0]=prim[1]=1;
   for(c=2;c*c<=2000005;c++)
   {
      if(prim[c]==0)
      {
         for(d=c*c;d<=2000005;d+=c)
            prim[d]=1;
      }
   }

   for(i=1;i<=2000005;i++)
      pre[i]=((pre[i-1])%MOD+((!prim[i])*(i+1))%MOD)%MOD;

}

int main()
{
   sieve(); 
   ll n, i, j;
   // for(int i=0;i<=10;i++)
   //     cout<<pre[i]<<" ";
   scanf("%lld",&n);
   n--;
   ll ans=0;
   ans+=pre[n];
   //cout<<ans<<endl;
   for(i=0;i<=n;i++)
   {
      if((prim[i+n]==0) and (i+n>n))
          ans=((ans%MOD)+(n-i+1)%MOD)%MOD;
   }
   
   printf("%lld",ans%MOD); 




   return 0;
}
