#include <iostream>

using namespace std;


int main()
{
    int n;
    cin>>n;
    string a,b,c;
    int len1;
    while(n--)
    {
        cin>>a;
        len1=a.length();
        bool flag=false;
        for(int i=0;i<len1/2;i++)
        {
            if(a[i] != a[len1-1-i])
            {
                flag = true;
            }
        }
        if(!flag)
        {
            cout<<"YES"<<endl;
        }
        else
        {
            cout<<"NO"<<endl;
        }
    }
    return 0;
}
