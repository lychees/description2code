#include <iostream>

using namespace std;

int main()
{
    long long int i, j, n, ans, k, l;
    
    string s;
    
    cin>>n;
    while(n--) {
    	s.clear();
    	cin>>s;
    	
    	l = s.length();
    		
    		k = ans = 0;
    	for (i = 0; i < l; i++) {
    		if (s[i] == 'H') {
    			k = k + 1;
    			ans = ans + k;
    		} else {
    			k = 0;
    		}
    	}
    	
    	cout<<ans<<endl;
    }
    return 0;
}
