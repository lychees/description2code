#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
vector<int>prime;
long long ans[500001]={0};
int arr[500001]={0};
void sieve(){
	arr[0] = 1;
	arr[1] = 1;
	for(int i = 4;i<=500000;i+=2){
		arr[i] = 1;
	}
	for(int i = 3;i<=500000;i+=2){
		for(int j = 2*i;j<=500000;j+=i){
			arr[j] = 1;
		}
	}
	for(int i = 2;i<=500000;i++){
		if(!arr[i]){
			//if(i<50)cout<<i<<endl;
			prime.push_back(i);
		}
	}
}
void pre(){
	ans [0] = 0;
	ans[1] = 1;
	ans[2] = 2;
	ans[3] = 6;
	for(int i = 4;i<=500000;i++){
		if(!arr[i]) {
			ans[i] = (ans[i-1]*i)%mod;
			continue;
		}
		int cnt = 0;
		int val;
		int k = i;
		for(int j=0;j<prime.size();j++){
			if(prime[j]*prime[j]>k) break;
			if(k%prime[j]==0){
				cnt++;
				val = prime[j];
			}
			while(k>0&&k%prime[j]==0){
				k = k/prime[j];
			}
		}
		//if(i==6)cout<<val<<endl;
		//if(k==6) cout<<val<<endl;
		if(k>1&&!arr[k])cnt++;
		if(cnt==1){
			ans[i] = (ans[i-1]*val)%mod;
		}
		else{
			ans[i] = ans[i-1];
			//if(i==6)cout<<ans[i-1]<<endl;
		}
	}
}
int main()
{
    sieve();
    //cout<<prime.size()<<endl;
    pre();
    int test;
    scanf("%d",&test);
    while(test--){
    	int n;
    	scanf("%d",&n);
    	cout<<ans[n]<<endl;
    }
    return 0;
}
