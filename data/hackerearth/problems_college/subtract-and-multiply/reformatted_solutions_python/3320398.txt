# print 'Hello World!'
from math import sqrt


def factors(n):
    return [(i, n/i) for i in range(1, int(sqrt(n))+1) if not n % i]


t = input()
while (t):
    d, p = map(int, raw_input().split())
    flag = False
    for i, j in factors(p):
        if j-i == d:
            flag = True
            break
    if (flag):
        print 4 if j != i else 2
    else:
        print 4 if p == 0 else 0
    t = t-1
