#include<bits/stdc++.h>
using namespace std;

typedef long long int ll;

int main(){
	ll n,k,tmp;
	cin>>n>>k;
	ll ar[n];
	for(int i=0;i<n;i++)
	cin>>ar[i];
	sort(ar,ar+n);
	int ctr = 0,mul = 0,ans = 0;
	for(int i=n-1;i>=0;i--){
		if(ctr==0)
		{
			ctr = k;
			mul++;
		}
		ans += (mul*ar[i]);
		ctr--;
	}
	cout<<ans<<endl;
	return 0;
}
