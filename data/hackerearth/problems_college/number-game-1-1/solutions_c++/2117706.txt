#include <iostream>
using namespace std;

int main()
{
  int t;
  cin>>t;
  long long x,count;
  while(t--){
  	count=0;
  	cin>>x;
  	while(x>1){
  		
  		if(x%3==0){
  			x=x/3;
  			count++;
  		}else if(x%2==0){
  			x=x/2;
  			count++;
  		}else{
  			x-=1;
  			count++;
  		}
  	}
  cout<<count<<"\n";
  }
  
    return 0;
}
