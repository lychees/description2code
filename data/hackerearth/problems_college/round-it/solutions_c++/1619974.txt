using namespace std;

int main()
{
	int t,n,b,r,ans,x;
	cin>>t;
	while(t--)
	{
		cin>>n>>b;
		r=n%b;
	   x=b;
	   if(x%2!=0)
	     x+=1;
		if(r < (x/2))
		{r=n/b;
	    ans=r*b;
	   	}
		else
		  {r=n/b;
		  ans=((r+1)*b);
		  }
		  
		cout<<ans<<endl;
	}

    return 0;
}
