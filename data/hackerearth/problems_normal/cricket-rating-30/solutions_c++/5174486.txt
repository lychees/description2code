#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n ,ans = 0, sum = 0, tem1;

    cin >> n;

    for(int i = 0; i < n; i++){

        cin >> tem1;

        if(sum <= 0){               // check if the is negative or = 0

            if(tem1 < 0) continue;      // check if the input number is negative
            sum = tem1;                 // start new try

        } else sum += tem1;             // add new number to the sum

        ans = max(ans , sum);           // maximize the best answer you have in each phase
    }

    cout << ans << "\n";

    return 0;
}
