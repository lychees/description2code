#define Int long long
#define MAX 1000009
int count1[1000009];
Int prime[MAX],P[MAX],top=0;
void sieve()
{
	Int i;
	memset(prime,true,sizeof(bool)*MAX);
	for(i=2;i<=MAX;i+=2)prime[i]=false;
	prime[2]=true;
	for(i=3;i*i<=MAX;i+=2)
	{
		if(prime[i])
		for(Int j=i*i;j<MAX;j+=i)prime[j]=false;
	}
	for(i=2;i<MAX;i++)if(prime[i])P[top++]=i;
}

void factor(Int num)
{
    Int a=num;
    int c=0;
	/*while(a%2==0)
	{
		c++;
		a/=2;
	}*/
	for(Int i=0;P[i] && P[i]*P[i]<=a;i++)
	{
		while(a%P[i]==0)
		{
			c++;
			a=a/P[i];
		}
	}
	if(a>1)c++;
	count1[num]=c+count1[num-1];
}
