#include<iostream>
using namespace std;

int main(){
    int t, j;
    string s;
    cin>>t;
    while(t--){
    	cin>>s;
    	j = 0;
    	for(int i = 0; i < s.length(); i++){
    		if(s[i] != '_'){
    			j += s[i] - '0';
    		}
    	}
    	j %= 9;
    	if(j == 0){
    		cout<<"0 or 9"<<endl;
    	}
    	else{
    		cout<<9 - j<<endl;
    	}
    }
	return 0;
}
