#define MOD 1000000007

int T , N , M[101];

int solve(int N){
	int &ret = M[N] ;
	if(ret != -1)
		return ret ;
	ret = 0 ;
	ret = (1LL * solve(ceil(N/5.0)) * solve(ceil(N/10.0))) % MOD ;
	ret = (1LL * ret * (1LL<<((int)ceil(N/4.0)))) % MOD ;
	 return ret ;
}

LL power(LL X,LL Y){
	LL res = 1 ;
	X %= MOD ;
	while(Y>0){
		if(Y%2)
			res = res * X % MOD ;
		 Y /= 2 ;
		 X = X * X % MOD ;
	}
	return res ;
}
int main(){

	memset(M,-1,sizeof M) ;
	M[0] = M[1] = M[2] = 1 ;
	M[3] = M[4] = 3 ;
	M[5] = M[6] = 15 ;
	M[7] = M[8] = M[9] = M[10] = 105 ;
	sc(T) ;
	while(T--){
		sc(N) ;
		int R = solve(N) ;
		LL ans = 0 ;
		for(int i=1;i<=5;i++){
			for(int j=1;j<=5;j++){
				if(__gcd(i,j) == 1){
					LL X , Y ;
					X = 1LL * R * i ;
					Y = 1LL * R * j ;
					//cout << X << " " << Y << endl ;
					ans = max(ans,power(X,Y)) ;
 				}
			}
		}
		pr(ans) ;
	}
	return 0 ;
}
