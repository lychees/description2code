#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    
    int max=100;
    char *p = new char[max];
    
    char q[5][100] = {"username","pwd","profile","role","key"};
    cin>>p;
    
    int c1=0;
    int c2=0;
    
    int i=0;
    while(p[i]!='\0')
    {
    	if(p[i]=='?'||p[i]=='&')
    	{
    		i++;
    		while(p[i]==q[c1][c2]&&q[c1][c2]!='\0'&&p[i]!='=')
    		{
    			i++;
    			c2++;
    		}
    	//	cout<<c2<<p[i]<<"\n";
    		if(c2==strlen(q[c1]))
    		{
    			cout<<q[c1]<<": ";

    			while(p[i]!='\0')
    			{
    				if(p[i]=='&' && p[i+1]==q[c1+1][0])
    				{
    				   break;	
    				}
    				else
    				cout<<p[i++];
    			}
    			cout<<"\n";
    			c1++;
    			c2=0;
    			i--;
    		}
    		
    		
    	}
    	
    	i++;
    }
    
    
    
    return 0;
}
