#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    int N; ll D,T = 0;
    scanf("%d%lld",&N,&D);
    vector<ll> A(N);
    for(int i = 0; i < N; i++) scanf("%lld",&A[i]);
    for(int i = 1; i < N; i++) {
    	if(A[i] <= A[i-1]) {
    		int F = ceil((A[i-1] - A[i])/D + 0.5);
    		T += F;
    		A[i] = A[i] + F*D;
    	}
    }
    cout << T;
    return 0;
}
