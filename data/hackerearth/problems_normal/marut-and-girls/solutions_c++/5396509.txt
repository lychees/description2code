#include <iostream>
#include<vector>
#include<string.h>
using namespace std;

int main()
{
    int quality[10001]={0};
    int m,i,temp,n,res=0;
    cin>>m;
    for(i=0;i<m;i++)
    {
    	cin>>temp;
    	quality[temp]=1;
    }
    cin>>n;
    for(int l=0;l<n;l++)
    {
    	char girl_quality[9001];
    	gets(girl_quality);
    	//scanf("%[^\n]s",girl_quality);
    //	printf("%s\n",girl_quality);
    	char *sub;
    	int val,count=0;
    	sub = strtok(girl_quality," ");
    	while(sub!=NULL)
    	{
    		val=atoi(sub);
    		if(quality[val])
    			count++;
    		sub = strtok(NULL," ");
    	}
    	if(count>=m)
    	  res++;
    	if(l==n-1 && gets(girl_quality)!=NULL)
    	  l--;
    }
    cout<<res<<endl;
    return 0;
}
