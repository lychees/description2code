#include <iostream>
using namespace std;


int isprime(long long n)
{
	if(n==0||n==1)return 0;
	for(long i=2;i<n;i++)
	{
		if((n%i)==0)return 0;
	}
	return 1;
}

int main()
{
    int t;cin>>t;while(t--)
    {
    	long long n;
    	cin>>n;
    	long long a[n];
    	long cu=0;
    	for(long i=0;i<n;i++)
    	{
    		long long ax;
    		cin>>ax;
    		if(isprime(ax))cu++;
    	}
    	cout<<cu<<endl;
    }
    return 0;
}
