#include<iostream>

using namespace std;

int main()
{
     long long a,b,c;

    cin>>a>>b>>c;
    long long ans=0;
    long long pos=0;
    long long temp=a/(b-c);
   ans=temp;
    pos=temp*(b-c);

    bool flag=1;

    while(flag)
    {
        pos+=c;
        if(pos<a)
            break;
        pos-=b;
        ans--;
    }
    cout<<ans+1;
}
