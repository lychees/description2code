#include <bits/stdc++.h>
using namespace std;
#define li long int
li a[100005];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	li n,x;
    	scanf("%ld %ld",&n,&x);
    	for(int i=0;i<n;i++){
    		scanf("%ld",&a[i]);
    	}
    	li ans=0;
    	for(int i=0;i<n-1;i++){
    		if(a[i]<a[i+1]) continue;
    		li temp=((a[i]-a[i+1])/x)+1;
    		ans+=temp;
    		a[i+1]=a[i+1]+temp*x;
    	}
    	printf("%ld\n",ans);
    }
    return 0;
}
