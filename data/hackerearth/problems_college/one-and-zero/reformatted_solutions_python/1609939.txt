import sys
f = sys.stdin
DP = [0]*(10001)
for i in range(0, 10001):
    DP[i] = [0]*(2)


def pre():
    DP[0][1] = 1
    for i in range(1, 10001):
        DP[i][0] += DP[i-1][1]
        DP[i][1] += DP[i-1][1]+DP[i-1][0]


T = int(f.readline())
pre()
while T:
    N = int(f.readline())
    print DP[N][0] + DP[N][1]
    T -= 1
