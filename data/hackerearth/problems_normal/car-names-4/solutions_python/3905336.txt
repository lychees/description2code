for i in range(input()):
	a=raw_input()
	l=[]
	d=0
	if(a=="carcar"):
		d=1
	for j in range(0,len(a)):
		if(a[j] not in l):
			l.append(a[j])
	if(len(l)!=3):
		print("Not OK")
	else:
		if(d==1):
			print("Not OK")
		elif(a.count(l[0])==a.count(l[1]) and a.count(l[1])==a.count(l[2])):
			print("OK")
		else:
			print("Not OK")
