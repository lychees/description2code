#include <iostream>
#include <string>
using namespace std;

int main()
{
   
   int len = 0;
   cin>>len;
   string s;
   cin>>s;
   long hs[26]={0};
   for(int i=0;i<len;++i)
   	++hs[s[i]-'a'];
   int q;
   cin>>q;
   while(q--)
   {
   		char a,b;
   		cin>>a>>b;
   		if(a==b)
   		{
   				cout<<((hs[a-97])*(hs[a-97]-1))/2<<endl;
   				
   		}
   		else
   		{
   			cout<<hs[a-97]*hs[b-97]<<endl;
   		}
   }
    return 0;
}
