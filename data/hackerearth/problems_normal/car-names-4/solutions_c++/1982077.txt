#include <iostream>
#include <cstring>

using namespace std;

void fun()
{
    char s[500];
    int ch[26];
    int result[26];
    int l = 0;
    int n = 0;

    memset(ch,0,sizeof(ch));
    memset(result,0,sizeof(result));
    cin >> s;
    l = strlen(s);
    for ( int i = 0; i < l; i++) {
        if (ch[s[i]-'a'] == 0) {
            ch[s[i]-'a'] = 1;
            result[s[i]-'a']++;
            while(s[i] == s[i+1] && i+1 < l) {
                result[s[i]-'a']++;
                i++;
            }
        } else {
            cout <<"Not OK\n";
            return;
        }
    }
    l = 0;
    for ( int i = 0; i < 26 ; i++) {
   //         cout << result[i] << " ";
        if (result[i]){
            l++;
            if(!n){
                n = result[i];
            } else {
                if(n != result[i] || l > 3){
                    cout <<"Not OK\n";
                    return;
                }
            }
        }
    }
    if(l == 3)
        cout << "OK"<< endl;
    else 
        cout <<"Not OK\n";
    return;
}

int main()
{
    int t;
    cin >> t;

    while(t--){
        fun();
    }

    return 0;
}
