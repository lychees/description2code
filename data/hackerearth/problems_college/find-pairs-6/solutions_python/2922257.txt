t=input()
while t:
    t -= 1
    n = input()
    l = map(int, raw_input().split())
    l = sorted(l)
    c = 1
    ways = 0
    for i in range(1,n):
        if l[i] == l[i-1]:
            c += 1
        else:
            ways += c*(c+1)/2
            c = 1
    if n>1:
        ways += c*(c+1)/2
    print ways
