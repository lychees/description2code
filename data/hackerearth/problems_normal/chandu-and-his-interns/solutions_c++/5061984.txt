#include <iostream>
using namespace std;

int main()
{
    long long n,i;
    cin>>n;
    long long x[n];
    for(i=0;i<n;i++){
    	cin>>x[i];
    }
    for(i=0;i<n;i++){
    	int count=0;
    	for(long j=2;j*j<=x[i];j++){
    		if((x[i]%j==0)&&(j*j!=x[i])){
    			count = 1;
    			break;
    		}
    	}
    	if(count==1){
    		cout<<"YES"<<endl;
    	}
    	else{
    		cout<<"NO"<<endl;
    	}
    }
    
}
