#include <iostream>
#include<stack>
using namespace std;

int main()
{
    int n,total=0;
    string s;
    cin>>n;
    for(int i=0; i<n; ++i)
    {
    	cin>>s;
    	stack <char> pair;
    	int q;
    	q = s.size();
    	pair.push(s[0]);
    	for(int j=1; j<q; ++j)
    	{
    		if (!pair.empty())
    			if(s[j] == pair.top())
    				pair.pop();
    			else
    				pair.push(s[j]);
    		else
    			pair.push(s[j]);
    				
    	}
    	
    	if(pair.empty())
    		total++;
   
    }
    cout<<total<<endl;
    return 0;
}
