#include <iostream>
#include<map>
using namespace std;

int main()
{
   int t,ans,A,B,N;
   ios::sync_with_stdio(false);cin.tie(0);cout.tie(0);
   cin>>t;
   
   while(t--) {
   	cin>>N;
   	ans = 0;
   	map<int,int>have;
  	for(int i =0;i<N;i++) {
  		cin>>A>>B;if(A==B)continue;
  		if(!(have.count(B)))ans++;
  		else ((have[B]==1)?have.erase(B):have[B]--);
  		have[A] = (have.count(A)?have[A]+1:1);
  	}
  	cout<<ans<<'\n';
   }
    return 0;
}
