#include <iostream>
using namespace std;

bool IsPrime(int number) {
    if (number <= 1) return false; // zero and one are not prime
    int i;
    for (i=2; i*i<=number; i++) {
        if (number % i == 0) return false;
    }
    return true;
}

bool IsPalin(int num){
int n = num;
 int rev = 0;
 while (num > 0)
 {
      int dig = num % 10;
      rev = rev * 10 + dig;
      num = num / 10;
 }
if(n==rev)
return true;
else 
return false;
}

int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	while(1)
    	{
    		if(IsPrime(n)&&IsPalin(n))
    		{
    			cout<<n<<endl;
    			break;
    		}
    		n++;
    	}
    }
    return 0;
}
