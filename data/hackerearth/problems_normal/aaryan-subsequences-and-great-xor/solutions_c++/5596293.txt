#include <iostream>
using namespace std;

int main()
{
	long n;
    cin>>n;
    
    long arr[n];
    for(long i = 0 ; i < n ; i++)
    	cin>>arr[i];
    	
    if(n > 1)
    {
    	long ans = 0;
    	for(long i = 0 ; i < n ; i++)
    		ans |= arr[i];
    		
    	cout<<ans<<endl;
    }
    
    else
    	cout<<(arr[0] | arr[0])<<endl;
    return 0;
}
