#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   int long long s=0;
   int long long l1=0,m1=0;
   
   char s1[t];
   cin>>s1;
   int p;
   cin>>p;
   for(int i=0;i<p;i++){
   	char a1,b1;
   	cin>>a1>>b1;
   	s=0;
   	l1=0;
   	m1=0;
   	for(int j=0;j<t;j++){
   		if(s1[j]==a1)
   		l1=l1+1;
   		else if(s1[j]==b1)
   		m1=m1+1;
    	}
   	if(a1!=b1)
   	s=l1*m1;
   	else 
   	s=l1*(l1-1)/2;
   	cout<<s<<endl;
   }
   
    return 0;
}
