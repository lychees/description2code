#include<iostream>

using namespace std;
long long int countt(int n)
{
    long long int res[n];
    res[0]=res[1]=1;
    for(int i=2;i<n;i++)
    {
        res[i]=0;
        for(int j=1;j<=2&&j<=i;j++)
        {
            res[i]=res[i]+res[i-j];
        }
    }
    return res[n-1];
}
int main()
{
    int T;
    cin>>T;
    int n;
    while(T--)
    {
        cin>>n;
        cout<<countt(n+1)<<"\n";
    }
    return 0;
}
