#include <iostream>
#include <stdlib.h>
using namespace std;
void generateTable(int **table,int x,int y,int size){
	static int val=1;
	if(size==1){
		table[x][y]=val++;
	}
	else{
		generateTable(table,x,y,size/2);
		generateTable(table,x,y+size/2,size/2);
		generateTable(table,x+size/2,y,size/2);
		generateTable(table,x+size/2,y+size/2,size/2);
	}
}
int power(int x,int y){
	while(--y > 0){
		x=2 * x;
	}
	return x;
}
int main()
{
    //cout << "Hello World!" << endl;

    int n;
    cin >> n;
    int N=power(2,n);
    //int **table = (int **) malloc(sizeof(int *)*N);
    int **table = new int*[N];
    //cout << N;
    for(int i=0;i<N;i++){
    	//table[i]=(int *)malloc(sizeof(int) * N);
    	table[i]=new int[N];
    }
    generateTable(table,0,0,N);
    for(int i=0;i<N;i++){
    	for(int j=0;j<N;j++){
    		cout << table[i][j] << " ";
    	}
    	cout << "\n";
    }
    return 0;
    
}
