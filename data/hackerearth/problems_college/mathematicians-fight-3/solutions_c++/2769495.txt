static LL A[1004][504];

void solve()
     {
     int i,j,k;
     
     for (i=0; i<=1001; i++)
         {
         for (j=0; j<=i and j<=502; j++)
             {
             if(!j or i==j) A[i][j]=1;
             else A[i][j]=(A[i-1][j-1]+A[i-1][j])%MOD;
             }
         }

     }
