#include <iostream>
#include <set>
#include <string>
using namespace std;

int main()
{
   set<string> s;
   int t;
   cin >> t;
   while(t--)
   {
	   	int n;
	   	int m;
	   	string str;
	   	cin >> m >> n;
	   	for(int i=0;i<m;i++)
	   	{
	   		cin >> str;
	   		s.insert(str);
	   	}
	   	if(s.size()==m)
	   		cout <<"Yes" << endl;
	   	else
	   		cout << "No" << endl;
	   	s.clear();
   }
    return 0;
}
