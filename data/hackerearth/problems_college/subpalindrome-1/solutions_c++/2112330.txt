#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
#define ll long long
ll cs;
char str[40];
ll M[40][40];
ll solve(ll i, ll j)
{
  ll &res = M[i][j],k,p;
  if (res == -1)
  {
    if (i > j)
      res = 1;
    else
    {
      res = j - i+2;
      for (k = i; k <= j; k++)
        for (p = k + 1; p <= j; p++)
          if (str[k] == str[p])
            res += solve(k + 1, p - 1);
    }
    //printf("%lld %lld %lld\n", i, j, res);
  }
  return res;
}
int main()
{
  scanf("%lld",&cs);
  while (cs--)
  {
    scanf("%s",str);
    memset(M,-1,sizeof M);
    printf("%lld\n",solve(0,strlen(str)-1)-1);
  }
  return 0;
}
