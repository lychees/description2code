#include <iostream>
using namespace std;

int denominations(int x){
	int count=0;
	if(x/1000!=0)
				{
					count+=x/1000;
					x=x%1000;
				}
				if(x/500!=0)
				{
					count+=x/500;
					x=x%500;
				}
				if(x/100!=0)
				{
					count+=x/100;
					x=x%100;
				}
				if(x/50!=0)
				{
					count+=x/50;
					x=x%50;
				}
				while(x/20!=0)
				{
					count+=x/20;
					x=x%20;
				}
				while(x/10!=0)
				{
					count+=x/10;
					x=x%10;
				}
				while(x/5!=0)
				{
					count+=x/5;
					x=x%5;
				}
				while(x/2!=0)
				{
					count+=x/2;
					x=x%2;
				}
				count+=x;

	return count;
}

int main(){
	int a[]={1,2,5,10,20,50,100,500,1000};
	int T,i,X;
	cin>>T;
for(i=0;i<T;i++){
	cin>>X;

	int res=denominations(X);
	cout<<res<<endl;
}
return 0;
}
