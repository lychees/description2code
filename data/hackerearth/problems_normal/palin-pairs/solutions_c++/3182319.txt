#include <bits/stdc++.h>
//zone
//map
//right to left crossover
using namespace std;

int main()
{
	int n;
	cin>>n;
	map<string,int> zone;
	long long counti=0;
	string s;
	while(n--)
	{
		cin>>s;
		counti += zone[s];
		reverse(s.begin(),s.end());
		zone[s]++;
		 
	}
	cout<<counti<<endl;
	return 0;
}
