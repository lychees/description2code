#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin >> n;
	stack<long> s;
	while(n--) {
		int t;
		long cost;
		cin >> t;
		if(t == 1) {
			if(s.empty())
				cout << "No Food\n";
			else {
				cout << s.top() << "\n";
				s.pop();
			}
		} else {
			cin >> cost;
			s.push(cost);
		}
	}
    return 0;
}
