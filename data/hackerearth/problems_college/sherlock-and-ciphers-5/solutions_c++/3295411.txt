#include<bits/stdc++.h>
#include<stdio.h>
using namespace std;
 int i,j,t,fin,l,k;
 char str[201];
int main()
{
    scanf("%d",&t);
    for(k=0;k<t;k++)
    {
    	scanf("\n%[^\n]%*c",&str[0]);
    	l=strlen(&str[0]);
    	fin=sqrt(l);
    	for(i=0;i<fin;i++)
    		for(j=i;j<l;j=j+fin)
    			cout<<str[j];
    	cout<<endl;
    }
    return 0;
}
