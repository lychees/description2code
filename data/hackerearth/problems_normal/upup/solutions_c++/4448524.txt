#include<stdio.h>
#include <iostream>
#include<string.h>
#include<ctype.h>
using namespace std;

int main()
{
    	char a[300005];
    	gets(a);
    	int len=strlen(a);
    	int flag=1;
    	for(int i=0;i<len;i++)
    	{
    		if(a[i]!=' ' && flag==1)
    		{
    			flag=0;
    			char ch=toupper(a[i]);
    			cout<<ch;
    		}
    		else if(a[i]==' ')
    		{
    			flag=1;
    			cout<<a[i];
    		}
    		else 
    			cout<<a[i];
    	}
    	cout<<endl;
    
}
