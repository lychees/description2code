#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int T,N,p1,p2;
    cin>>T;
    while(T--) {
    	cin>>N;
    	vector<pair<int,int> > arr(N);
    	for(int i=0;i<N;i++) {
    		cin>>arr[i].first>>arr[i].second;
    	}
    	sort(arr.begin(),arr.end());
    	vector<int> dp(N);
    	int mmxx=0;
    	for(int i=0;i<N;i++) {
    		int mx=1;
    		for(int j=0;j<i;j++) {
    			if(arr[j].second<arr[i].second && dp[j]>=mx) {
    				mx=dp[j]+1;
    			}
    		}
    		dp[i]=mx;
    		if(mx>mmxx) mmxx=mx;
    	}
    	cout<<mmxx<<endl;
    }
    return 0;
}
