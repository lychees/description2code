#include<bits/stdc++.h>


#define LL long long


using namespace std;

int a[100001];
int sum[100001];
int main()
{
       int n;
       cin>>n;
       sum[0]=0;
       for(int i=1;i<=n;i++)
       {
            cin>>a[i];
            sum[i]=sum[i-1]+a[i];

       }


       int  q;
       cin>>q;
       while(q--)
       {
              int p;
              cin>>p;

               if(sum[n]<p)
               {
                    cout<<-1<<endl;
               }
               else{

              int lo=1;
              int hi=n;
              while(lo<hi)
              {
                  int mid=(lo+hi)/2;

                  if(sum[mid]==p)
                  {
                        lo=mid;
                        break;
                  }

                  if(sum[mid]<p)
                  {
                       lo=mid+1;

                  }
                  else
                  {
                       hi=mid-1;

                  }
              }

              while(sum[lo]<p&&lo<n)
              {
                   lo++;
              }

              cout<<lo<<endl;

               }



       }

       return 0;

}
