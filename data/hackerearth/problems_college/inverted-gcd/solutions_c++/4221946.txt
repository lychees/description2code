#include<bits/stdc++.h>
#define ll long  long
#define fo(i,n) for(i=0;i<n;i++)
#define forr(i,n) for(i=n-1;i>=0;i--)
using namespace std;
ll gcd (ll a,ll b){while(a&&b)a>b?a%=b:b%=a;return a+b;}
int main(){
 ll n;
 cin>>n;
 ll count=0;
 ll a[n];
 ll i, j;
 fo(i,n)
    cin>>a[i];
 fo(i,n){
    for(j=i+1;j<n;j++){
        if(a[i]>a[j]&&gcd(a[i],a[j])==1)
            count++;
    }
 }
 cout<<count<<endl;
 return 0;
}
