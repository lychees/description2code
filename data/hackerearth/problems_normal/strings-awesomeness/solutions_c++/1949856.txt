#include<bits/stdc++.h>
using namespace std;
#define MAX 1000001
#define patlen 11
int d[MAX];
map<char,int> a;
int check()
{
	if(a['a']+a['A']==2&&a['c']+a['C']==1&&a['e']+a['E']==2&&a['h']+a['H']==2&&a['k']+a['K']==1&&a['r']+a['R']==2&&a['t']+a['T']==1)
	return 1;
	else
	return 0;
}
int main()
{
	char str[MAX];
	cin>>str;
	int n,i;
	n=strlen(str);
	for(i=0;i<patlen;i++)
		a[str[i]]++;
	d[0]=check();
	int start=0,length=11;
	while(length<n)
	{
		a[str[start]]--;
		a[str[length]]++;
		start++,length++;
		d[start]=check();
	}
	for(i=1;i<n;i++)
	{
		d[i]=d[i]+d[i-1];
	}
	int q;
	cin>>q;
	while(q--)
	{
		int l,r;
		scanf("%d",&l);
		scanf("%d",&r);
		l--,r--;
		if(l)
			printf( "%d\n" , d[max(r-patlen+1,l-1)] - d[l-1]);
		else
			printf( "%d\n", (max(r-patlen+1,l-1)<0 ? 0 : d[max(r-patlen+1,l-1)]));
	}
	return 0;
}
