n = int(raw_input())
for i in range(n):
    p, x, y, u, v = map(int, raw_input().strip().split(' '))
    x_diff = abs(u-x)
    y_diff = abs(v-y)

    if (x_diff == 0 and y_diff == 0) or (x_diff == y_diff and p > 1):
        print "Counter-Terrorists Win !"
    elif(x_diff+y_diff) % 2 == 0 and p > 2:
        print "Counter-Terrorists Win !"
    else:
        print "Terrorists Win !"
