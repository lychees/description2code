#include <stdio.h>
#include<math.h>
#include<stdlib.h>
using namespace std;

int main()
{
   int n,i;
   int count=1;int max=1;
   scanf("%d",&n);
   int a[n];
   for(i=0;i<n;i++)
   scanf("%d",&a[i]);
   for(i=1;i<n;i++)
   {
   	if(abs(a[i]-a[i-1])!=1)
   	count=1;
   	else 
   	count++;
   	if(count>max)
   	max=count;
   }
   printf("%d\n",max);
    return 0;
}
