#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	int a[2*n];
    	for(int i=0;i<2*n;i++) cin>>a[i];
    	sort(a,a+2*n);
    	int max=0;
    	for(int i=0;i<n;i++){
    		if(max<(a[i]+a[2*n-1-i])) max=(a[i]+a[2*n-1-i]);
    	}
    	cout<<max<<endl;
    }
    return 0;
}
