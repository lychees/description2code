#include <iostream>
#include <stack>
using namespace std;

int main()
{
    int m,sum=0;
    string s;
    stack<char> a;
    cin>>m;
    while(m--) {
    	cin>>s;
    	while(!a.empty()) {
    		a.pop();
    	}
    	for(int i=0;i<s.length();i++) {
    		 if(!a.empty() && a.top()!=s[i]) {
    		 	a.push(s[i]);
    		 }
    		 else if(!a.empty() && a.top()==s[i]) {
    		 	a.pop();
    		 }
    		 else {
    		 	a.push(s[i]);
    		 }
    	}
    	if(a.empty())
    	sum++;
    }
    cout<<sum;
    return 0;
}
