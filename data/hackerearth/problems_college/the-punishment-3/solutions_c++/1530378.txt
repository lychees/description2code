#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,k,a,b,i; cin>>t;
    while(t--)
    {
        cin>>n>>k;
        a=(k>(n-k))?k:n-k;
        b=n-a;
        for(i=1;i<=(b-1);i++)
            cout<<"9";
        cout<<"8";
        for(i=1;i<=(a-b);i++)
            cout<<"9";
        for(i=1;i<=(b-1);i++)
            cout<<"0";
        cout<<"1\n";
    }
    return 0;
}
