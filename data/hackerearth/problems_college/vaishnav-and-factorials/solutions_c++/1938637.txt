#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int no_of_digits=500;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[no_of_digits];
    	for(int i=0;i<no_of_digits;i++)
    		a[i]=0;
    	a[no_of_digits-1]=1;
    	int carry;
    	for(int i=1;i<=n;i++)
    	{
    		carry=0;
    		for(int j=no_of_digits-1;j>=0;j--)
    		{
    			a[j]=a[j]*i+carry;
    			carry = a[j]/10;
    			a[j]=a[j]%10;
    		}
    	}
    	int i=0;
    	while(a[i]==0)
    		i++;
    	long long ans = 0;
    	for(i;i<no_of_digits;i++)
    	{
    	//	cout<<a[i];
    		if(a[i]==4 || a[i]==7)
    			ans ++;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
