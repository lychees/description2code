#include <iostream>
#include <utility>
#include <algorithm>
#include<set>
#include <iomanip>
#include <math.h>
using namespace std;

int main()
{
	pair<float, float> coords[3];
	int T = 3, i = 0;
	while(T--)
	{
		float x,y;
		cin>>x>>y;
		coords[i++] = make_pair(x,y);
	}
	
	float sumX = coords[0].first + coords[1].first + coords[2].first;
	float sumY = coords[0].second + coords[1].second + coords[2].second;
	vector<pair<double, double>> v;
	v.push_back(make_pair(sumX - 2 * coords[0].first, sumY - 2 * coords[0].second));
	v.push_back(make_pair(sumX - 2 * coords[1].first, sumY - 2 * coords[1].second));
	v.push_back(make_pair(sumX - 2 * coords[2].first, sumY - 2 * coords[2].second));
	sort(v.begin(), v.end());
	cout.precision(4);
	cout<<fixed;
	for(int i = 0;i<3;i++)
	{
		cout<<v[i].first;
		cout<<' ';
		cout<<v[i].second<<endl;
	}
	
    return 0;
}
