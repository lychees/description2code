#include<iostream>
#include<cstdio>
#include<cmath>
#include<algorithm>
#include<string>
#include<cstring>
using namespace std;
#define M 100005
#define R 1000000007

int a[100005],b[100005];

bool comp(int i,int j){
    return (i>j);
}

int main(){
    int T,i,j,k,l,n,m,h,ct,ct2;
    int64_t an;
    scanf("%d",&T);
    b[0]=b[1]=1;
    for(i=2;i*i<M;i++)if(b[i]==0){
        for(j=i*i;j<M;j+=i)b[j]=1;
    }

    while(T--){
        scanf("%d%d",&n,&m);
        for(i=0;i<n;i++)scanf("%d",&a[i]);
        sort(a,a+n,comp);
        h=m;l=a[0]; ct2=0;
        for(i=l;i<=h;i++)if(b[i]==0)ct2++;
        an=h-l+1-ct2;
        if(an<=0){ printf("%d\n",0); continue;}
        for(i=1;i<n;i++){
            h=a[i-1]; l=a[i]; ct=0;
            for(j=l;j<h;j++)if(b[j]==0)ct++;
            ct2+=ct;
            k=m-l+1-ct2-i;
            if(k<=0){ printf("%d\n",0); break;}
            an*=k; an%=R;
        }
        if(i!=n)continue;
        printf("%lld\n",an);
    }
    return 0;
}
