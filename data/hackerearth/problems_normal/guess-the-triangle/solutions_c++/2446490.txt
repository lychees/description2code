#include <bits/stdc++.h>
using namespace std;

int main()
{
    double a, b, c, d, e, f, x1, x2, y1, y2, x3, y3;
    cin >> a >> b >> c >> d >> e >> f;
    double x1x2 = 2*a;
    double x1x3 = 2 * c;
    double x2x3 = 2 * e;
    double y1y2 = 2 * b;
    double y1y3 = 2 * d;
    double y2y3 = 2 * f;
    x1 = (x1x2 + x1x3 - x2x3)/2;
    x2 = (x1x2 + x2x3 - x1x3)/2;
    x3 = (x1x3 + x2x3 - x1x2) / 2;
    y1 = (y1y2 + y1y3 - y2y3)/2;
    y2 = (y1y2 + y2y3 - y1y3)/2;
    y3 = (y1y3 + y2y3 - y1y2)/2;
    vector<pair<double, double> > v;
    v.push_back(make_pair(x1, y1));
    v.push_back(make_pair(x2, y2));
    v.push_back(make_pair(x3, y3));
    sort(v.begin(), v.end());
    for(int i = 0; i < 3; i++) {
    	printf("%.4lf %.4lf\n", v[i].first, v[i].second);
    }
    return 0;
}
