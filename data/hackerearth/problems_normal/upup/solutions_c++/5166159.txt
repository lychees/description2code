#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    getline(cin,s);
    int i,l,m,n;
    l=s.length();
    for(i=0;i<l;i++)
    {
    	if(i==0)
    	{
    		n=s[0]-0;
    		if(n>=97&&n<=122)
    		s[0]=s[0]-32;
    	}
    	if(s[i]==' ')
    	{
    		m=s[i+1]-0;
    		if(m>=97&&m<=122)
    		s[i+1]=s[i+1]-32;
    	}
    }
    cout<<s;
    return 0;
}
