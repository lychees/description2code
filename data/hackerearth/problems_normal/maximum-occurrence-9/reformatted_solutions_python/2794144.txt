string = raw_input()
string = string.strip(" ")
counter = 0
letter = ""
i = 0
while string:
    val = string.count(string[i])
    if counter < val:
        counter = val
        letter = string[i]
    elif counter == val:
        if letter > string[i]:
            letter = string[i]
    string = string.strip(string[i])
print("{0} {1}".format(letter, counter))
