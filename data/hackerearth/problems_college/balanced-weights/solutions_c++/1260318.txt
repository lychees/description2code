#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int a[1000001];
#define gu getchar_unlocked

inline int getint()
{
	int n = 0;
	char c = gu();

	while(c >= '0' && c <= '9')
	{
		n = (n << 3) + (n << 1) + c - '0';
		c = gu();
	}
	return n;
}


inline int gcd(int a, int b)
{
	if(b == 0)
		return a;
	return gcd(b, a%b);
}

int main()
{
    int t, n, g;
    t = getint();
    while(t--)
    {
    	n = getint();
    	for(int i = 0; i < n; i++)
    		a[i] = getint();
    	g = a[0];
    	for(int i = 1; i < n; i++)
    		g = gcd(g, a[i]);
    	printf("%lld\n", (ll)g*n);
    }
    return 0;
}
