#include<stdio.h>
#include<algorithm>
using namespace std;

int main()
{
	int t,i,j;
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
		int n;
		scanf("%d",&n);
		int arr1[n+1],arr2[n+2];
		for(j=0;j<n;j++)
		{
			scanf("%d",&arr1[j]);
		}
		for(j=0;j<n;j++)
		{
			scanf("%d",&arr2[j]);
		}
		std:sort(arr1,arr1+n);
		sort(arr2,arr2+n);
		int count=0;
		for(j=0;j<n;j++)
		{
			if(arr1[j]%arr2[n-j-1]==0)
				count++;
				
			else if(arr2[n-j-1]%arr1[j]==0)
				count++;
			
		}
		printf("%d\n",count);
	}
	return 0;
}
