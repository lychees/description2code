#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	int t;
	ll n,i,dig;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld",&n);
		ll cnt=0;
		while(n!=0)
		{
			dig=n%10;
			if(dig==4||dig==7)
			cnt++;
			n=n/10;
		}
		if(cnt==4||cnt==7)
		printf("YES\n");
		else
		printf("NO\n");
	}
	return 0;
}
