#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,km,i,j;
    	cin>>n>>km;
    	j=2*n;
    	
    	for(i=1;i<=j;i++)
    	{
    		
    		if(i%6==0)
    		j+=4;
    	//cout<<j<<" ";
    	}
    	//printf("%d ",i);
    	if(i-1>=km)
    	printf("Yes\n");
    	else
    	printf("No\n");
    }
    return 0;
}
