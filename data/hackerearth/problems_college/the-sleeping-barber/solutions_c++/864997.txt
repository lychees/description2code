#include <iostream>
#include <vector>
#include <algorithm>
#define ll long long
using namespace std;

int main()
{
	int D;
	cin>>D;
	ll G,M;
	while(D--) {
		cin>>G>>M;
		vector<ll> arr(G);
		for(int i=0;i<G;i++) {
			cin>>arr[i];
		}
		sort(arr.begin(),arr.end());
		ll sum=0;
		int j=0; // expected start
		int cnt=0,mx=0;
		for(int i=0;i<G;i++) {
			while(sum+arr[i]>M) {
				sum-=arr[j];
				j++;
				cnt--;
			}
			sum+=arr[i];
			cnt++;
			if(cnt>mx) mx=cnt;
		}
		cout<<mx<<endl;
	}
    return 0;
}
