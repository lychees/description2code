#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <cstring>
using namespace std;
int n, maxx, v, i, x;
int main() {
  scanf("%d", &n);
  maxx = v = 0;
  for (i = 1; i <= n; ++i)
    scanf("%d", &x), v = __gcd(v, x), maxx = max(maxx, x);
  puts((maxx / v - n) & 1 ? "Alice" : "Bob");
}
