#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {int n;
            cin>>n;
            int m;
            cin>>m;
            int a[m];
            for(int i=0;i<m;i++){
                
                cin>>a[i];
                
            }
            long long int dp[n+1][m];
            for(int i=0;i<m;i++){
                dp[0][i]=1;
            }
          sort(a,a+m);
            for(int i=1;i<=n;i++){
                for(int j=0;j<m;j++)
                     { long long int x=0,y=0;
                    if(i>=a[j])
                        x=dp[i-a[j]][j];
       
                    
                    
                    if(j>=1)
                        y=dp[i][j-1];
                      
                      dp[i][j]=x+y;
                }
            }
            cout<<dp[n][m-1];
            
            
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    return 0;
}
