#include <iostream>
using namespace std;

int main()
{	int t;
	long long int n,m,a,b,c,sum,sum1;
	cin>>t;
	while(t--)
	{	sum=0;
		cin>>n>>m>>a>>b>>c;
		 
			sum1=n*a+m*b;
			if(n>m)
			sum=2*m*c+(n-m)*a;
			else
			sum=2*n*c+(m-n)*b;
		
			if(sum1>sum)
				sum=sum1;
		cout<<sum<<"\n";
	}

  //  cout << "Hello World!" << endl;
    return 0;
}
