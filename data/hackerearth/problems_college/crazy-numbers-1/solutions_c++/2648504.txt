#include <iostream>
using namespace std;
long init [10] = {1,1,2,2,2,2,2,2,2,1};
long arr [2][10] = {1,1,2,2,2,2,2,2,2,1};
int main()
{
	int t ;
	int n;
	
    cin>>t;
    while(t--){
    	for(int  i =0 ;i < 10 ;i++) 
    	arr[0][i] =init[i];
    	cin>>n;
    	if(n==0)
    	cout<< 0<<endl;
       else if(n==1)
       cout <<10<<endl;
       else
       {
       	for(int i=0;i<n-2;i++)
       	{
       			
       		for(int k=0 ;k<10;k++)
       		{
       	long sum = 0 ;
       		if(k-1>=0)
       		sum = arr[i%2][k-1]%1000000007;
       		
       		if(k+1<10)
       		sum = (sum+arr[i%2][k+1])%1000000007;
       		
       		arr[(i+1)%2][k] = sum%1000000007; 
       		}
       		
       	}
       long ans = 0;
       for( int i=0;i<10;i++)
       {
       	ans = (ans+arr[n%2][i])%1000000007;
       //	cout <<arr[n%2][i]%1000000007<<endl;
       }
       	cout <<ans<<endl;
       }
    
    }
    
    return 0;
}
