using namespace std;

typedef long long ll;
const int mod = 1000000007;

int main(void) {
	//freopen("input.txt", "r", stdin);
	string sequence;
	cin >> sequence;
	ll ans = 0, cnt = 0,res = 0;
	for (unsigned i = 0; i < sequence.size(); i++) {
		if (sequence[i] == ';') {
			cnt++;
			ans = (ans + res) % mod;
		}
		else {
			res = (2 * res + cnt) % mod;
		}
	}
	cout << ans;
	return 0;
}
