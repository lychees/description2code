using namespace std;

int pr[10001000],ans[10001000];

void prime(int n)
{
    int i,j;
    pr[1]=1;
    for(i=4;i<=n;i+=2)
        pr[i]=2;
    for(i=3;i*i<=n;i+=2)
    {
        if(pr[i]==0)
        {
            for(j=i*i;j<=n;j+=i+i)
            {
                pr[j]=i;
            }
        }
    }
}

int gcd(int a,int b)
{
    if(b==0)
        return a;
    else return gcd(b,a%b);
}

int main()
{
    int t,n,m,i,j,k;

    prime(10000000);

    for(i=2;i<=10000000;i++)
    {
        if(pr[i]==0)
            ans[i]=1;
        else
            ans[i]=ans[i/pr[i]]+1;
    }

    S(t);
    while(t--)
    {
        S(n);  S(m);

        LL lcm=((LL)n) * m;
        lcm=lcm/gcd(n,m);

        P(ans[lcm/n]+ans[lcm/m]);
    }
    return 0;
}
