#include<iostream>
#include<cstring>
using namespace std;
int main()
{
	int tc,z;
	string str1;
	string str2;
	cin>>tc;
	while(tc--)
	{
		cin>>str1>>str2>>z;
		int len1=str1.length();
		int len2=str2.length();
		if(len1!=len2)
		{
			tc=tc+1;
			continue;
		}
		int i=0;
		int c=0;
		while(i<len1)
		{
			if(str1[i]!=str2[i])
			c++;
			
			i++;
		}
		if(c>z)
		cout<<"No"<<endl;
	else
	{
		c=z-c;
		if(c%2==0)
		cout<<"Yes"<<endl;
		else
		cout<<"No"<<endl;
	}
	}
	return 0;
}
