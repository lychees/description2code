int t;
#define MAX 100000
int use[(1<<23)+1];
vector<int> prime;
int main(){
	int t;
	scanf("%d", &t);
	int a, b;
	for (int i = 3; i < (1 << 23) + 1; i++){
		if (use[i] == 0){
			for (int j = i; j < (1 << 23) + 1; j+=i){
				use[j]++;
			}
		}
	}
	while (t--){
		scanf("%d%d", &a, &b);
		int ans = __gcd((1 << a) - 1, (1 << b) - 1);
		printf("%d\n", use[ans]);
	}
	return 0;
}
