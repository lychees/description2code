#include <bits/stdc++.h>

using namespace std;

int M[210][210];

int main()
{
    ios::sync_with_stdio(false);
    int T, n, m;
    char c;
    cin >> T;
    while(T--)
    {
        cin >> n >> m;
        for(int i=0;i<n;++i)
        {
            for(int j=0;j<m;++j)
            {
                cin >> c;
                if(c == 'T')
                    M[i][j] = 150;
                else if(c == 'D')
                    M[i][j] = -100;
                else
                    M[i][j] = 0;
            }
        }
        if(M[0][0] < 0)
        {
            cout << -1 << endl;
            continue;
        }
        for(int i=1;i<n;++i)
        {
            M[i][0] += M[i-1][0];
            if(M[i][0] < 0)
                M[i][0] = -6000000;
        }
        for(int i=1;i<m;++i)
        {
            M[0][i] += M[0][i-1];
            if(M[0][i] < 0)
                M[0][i] = -6000000;
        }
        for(int i=1;i<n;++i)
        {
            for(int j=1;j<m;++j)
            {
                M[i][j] += max(M[i-1][j],M[i][j-1]);
                if(M[i][j] < 0)
                    M[i][j] = -6000000;
            }
        }
        if(M[n-1][m-1] >= 0)
            cout << M[n-1][m-1] << endl;
        else
            cout << -1 << endl;
    }
    return 0;
}
