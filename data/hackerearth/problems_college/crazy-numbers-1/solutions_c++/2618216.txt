#include<stdio.h>
#include <iostream>
#define MOD 1000000007
using namespace std;
long long int dp[1000002][11];

int main()
{
    int i,j,t,n,ans;
    for(i=0;i<=9;i++)
     dp[1][i]=1;
    
    for(i=2;i<=1000000;i++)
    {
    	dp[i][0]=dp[i-1][1];
    	for(j=1;j<=9;j++)
    	{
    		dp[i][j]=(dp[i-1][j-1]+dp[i-1][j+1])%MOD;
    	}
    }
    
    
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	long long int sum=0; 
    	for(i=1;i<=9;i++)
    	{
    		sum=(sum+dp[n][i])%MOD;
    	 //   printf("dp[%d][%d]%lld\t",n,i,dp[n][i]);
    	}
    	
    	if(n==1)
    	sum=sum+1;
    	
    	cout<<sum<<endl;
    }
    
    return 0;
    
    
}
