#include<bits/stdc++.h>
using namespace std;
  int n;
  long long a[100005][3],dp[100005][3];

int main()
{
    for(int i=0;i<100005;i++)
        a[i][0]=a[i][1]=a[i][2]=-1;
    int t;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(int i=0;i<n;i++)
        cin>>a[i][0]>>a[i][1]>>a[i][2];

       for(int i=0;i<3;i++)
       {
           dp[0][i]=a[0][i];
       }
       for(int i=1;i<n;i++)
        {
      for(int j=0;j<3;j++)
       {
          dp[i][j] = min(dp[i-1][(j+2)%3],dp[i-1][(j+4)%3]) + a[i][j];
       }
       }
       cout<<min(dp[n-1][0],min(dp[n-1][1],dp[n-1][2]))<<endl;
    }

    return 0;
}
