#include <iostream>
using namespace std;
int main()
{
    int n,a[100],i,c=0,s,sum=0;
    cin>>n;
    for(i=0;i<n;i++)
        cin>>a[i];
        s=10000;
    for(i=0;i<n;i++)
    {
        if(a[i]%2==1)
        {
            c++;
            if(s>a[i])
                s=a[i];
        }
    }
    if(c==0)
    {
        cout<<":(";
		}
    else if(c%2==0)
    {
        for(i=0;i<n;i++)
            sum+=a[i];
            sum-=s;
            cout<<sum;
    }
    else if(c%2!=0)
    {
       for(i=0;i<n;i++)
            sum+=a[i];

            cout<<sum;
    }
    return 0;
}
