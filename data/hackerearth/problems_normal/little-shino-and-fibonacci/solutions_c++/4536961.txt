#include<stdio.h>
#include<string.h>
#include<iostream>
#define mod 1000000007
using namespace std;
typedef long long ll;
ll fib[15005];
ll fibs[15005];
ll solve(ll n)
{
	ll k=n/15000;
	ll rem=n%15000;
	ll rr=(((k%mod)*fibs[15000])%mod+fibs[rem])%mod;
	return rr;
}
int main()
{
ll i;
fib[1]=0;
fib[2]=1;
fibs[1]=0;
fibs[2]=1;
for(i=3;i<=15000;i++)
fib[i]=(fib[i-1]+fib[i-2])%10000;
for(i=3;i<=15000;i++)
fibs[i]=(fibs[i-1]+fibs[i-2])%10000;
for(i=1;i<=15000;i++)
fibs[i]=(fibs[i]+fibs[i-1])%mod;

ll t;
cin>>t;
while(t--)
{
	ll l,r;
	cin>>l>>r;
	ll ans=(solve(r)+mod-solve(l-1))%mod;
	cout<<ans<<endl;
}
return 0;
}
