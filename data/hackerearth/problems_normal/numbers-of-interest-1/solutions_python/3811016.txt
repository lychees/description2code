test_case = int(raw_input())
test_set = [] #list of x,y,n
number_series_list = []
for t in range(test_case):
    params = raw_input()
    test_set.append([int(n) for n in params.split(" ")])

for t in test_set:
  sublist = []
  for n in range(t[0]):
     sublist.append(t[1])
  length = len(sublist)
  for n in range(t[2]-length):
     from_list = len(sublist)-t[0]
     sublist.append(sum(sublist[from_list:]))
  number_series_list.append(sublist)

for n in number_series_list:
    print n[-1]
