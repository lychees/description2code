#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main(){


    int t;
    cin >> t;
    while(t--){

      ll n , a , b;
      cin >> n >> a >> b;
  if(a==0||b==0){
  	
  	cout << "-1" << endl;
  	continue;
  	
  }
      ll ans = n/a;
      ans = ans + n/b;

      ll lcm = (a*b)/__gcd(a,b);

      ans = ans - n/lcm;

      cout << ans << endl;



    }

}
