#include <iostream>
#include <algorithm>
using namespace std;
int a[100000];
int main()
{
    int t,n,cnt,ans,i;
    cin>>t;
    while(t--){
    	cin>>n;
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	sort(a,a+n);
    	cnt=1;ans=1;
    	for(i=0;i<n-1;i++){
    		if(a[i]==a[i+1])
    			cnt++;
    		else{
    			if (cnt > ans )
    			ans=cnt;
    			cnt=1;
    		}
    	}
    	if (cnt > ans )
    			ans=cnt;
    	cout<<ans<<endl;
    }
}
