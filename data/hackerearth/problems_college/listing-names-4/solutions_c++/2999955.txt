#include <iostream>
#include <algorithm>
#include <vector>
#include <string>

using namespace std;

int  main(){
	vector <string> str;
	vector <string> str2;
	int n,i;
	string chr;
	cin >> n;
	for(i=0;i<n;i++){
		cin >> chr;
		str.push_back(chr);
	}
	sort(str.begin(),str.end());

	str2=str;
	
	int cnt;
	str.erase(unique(str.begin(),str.end()),str.end());
	for(i=0;i<str.size();i++){
		cnt = count(str2.begin(),str2.end(),str[i]);
		cout << str[i] << " " << cnt << endl;
	}

return 0;
}
