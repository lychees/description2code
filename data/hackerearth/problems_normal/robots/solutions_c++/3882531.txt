#include <bits/stdc++.h>

using namespace std;

const long long int mod = 1000000007;

int main()
{
    int t,kase=0;

    cin >>t;
    int a[11111];

    while (t--) {
    	int n;
    	cin >> n;


    	for (int i=0;i<n;i++) cin >> a[i];

    	sort (a,a+n);

    	long long int res = a[0]%mod;

    	for (int i=1;i<n;i++) {res *= (a[i]-i); res%=mod;}

    	cout << res << endl;
    }
    return 0;
}
