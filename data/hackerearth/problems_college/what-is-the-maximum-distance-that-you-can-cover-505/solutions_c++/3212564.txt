#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
   double lim;
   int n;
    cin>>n;
    double a[n];
    double b[n];
    for(int i=0;i<n;i++){
    	cin>>a[i];
    }
    for(int i=0;i<n;i++){
    	cin>>b[i];
    }
    cin>>lim;
    double ratio=a[0]/b[0];
    for(int i=1;i<n;i++){
    	if(a[i]/b[i] > ratio)
    	{
    		ratio = a[i]/b[i];
    	}
    }
    ratio = floor(ratio*lim*1000)/1000;
    printf("%.3lf",ratio);
    return 0;
}
