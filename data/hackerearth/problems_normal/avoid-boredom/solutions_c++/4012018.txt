using namespace std;
typedef long long LL;
typedef unsigned long long ULL;

LL mod=1e9+7;

int n, m;
LL arr[105];
LL dp[105][105];

int main()
{

    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int T;
    cin>>T;



    for(int I=0; I<T; I++)
    {
        cin>>n>>m;

        for(int i=1; i<=n; i++)
            cin>>arr[i];

        for(int i=1; i<=n; i++)
        {
            for(int j=1; j<=n; j++)
                dp[i][j]=-1e18;
        }

        for(int i=1; i<=n-m+1; i++)
        {
            dp[i][1]=0;
        }

        for(int i=2; i<=m; i++)
        {
            for(int j=i; j<=n; j++)
            {
                for(int k=1; k<j; k++)
                {
                    dp[j][i]=max(dp[j][i], dp[k][i-1]+abs(arr[j]-arr[k]) );
                }
            }
        }

        LL ret=-1e18;

        for(int i=1; i<=n; i++)
        {
            ret=max(ret, dp[i][m]);
        }

        cout<<ret<<'\n';

    }



}
