#include <iostream>
using namespace std;
long long int mod=1000000009;
long long int power(long long int  a, long long int  b) { long long int  x=1, y=a; while(b>0) { if(b%2==1) x=(x*y)%mod; y=(y*y)%mod; b=b/2;}     return x; }
int main()
{

	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n;
		cin >> n;
		long long int ans=0;
		if(n==1)
		{
			cout << 1 << endl;
		}
		else
		{
			ans=power(10,n)-power(8,n);
			ans=(ans+mod)%mod;
			ans=(ans*power(2,mod-2))%mod;
			cout << ans << endl;
		}
	}
	return 0;

}
