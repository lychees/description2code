#include <iostream>
#include <cstring>

using namespace std;

int main()
{
    char* str; str = new char;
    cin>>str;
    int length = strlen(str);
    
    int totalEven = 0;
    for(int i = 0; i < length; i++) {
    	if(str[i] == '2' || str[i] == '4' || str[i] == '6' || str[i] == '8') {
    		totalEven++;
    	}
    }

    for(int i = 0; i < length; i++) {
    	cout<<totalEven<<" ";
    	if(str[i] == '2' || str[i] == '4' || str[i] == '6' || str[i] == '8') {
    		totalEven = totalEven - 1;
    	}
    }
    return 0;
    
}
