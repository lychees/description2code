#include<stdio.h>
#include<string.h>
#include<bits/stdc++.h>
#define ll long long
using namespace std;

int ar[1000010];
int har[10000000];
int main()
{
    long long m,n,f;
    int num=1000010;
    //scanf("%d",&num);

    for(m=2; m<=num; m++)
    {
        if(ar[m]==0)
        {
            for(n=m; n<=num; n+=m)
            {
                ar[n]++;
            }

        }
    }
    cin>>n;
    for(int i=2;i<=n;i++)
    {
        har[ar[i]]++;
    }
   ll int q,kl,su=0;
   cin>>q;
   for(int ki=0;ki<q;ki++)
   {
       su=0;
       scanf("%d",&kl);
       if(kl>100000) printf("0\n");
       else
       printf("%d\n",har[kl]);
   }
return 0;
}
