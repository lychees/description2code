#include <iostream>
#include <stack>
using namespace std;


int main()
{
	int total_queries;
	int type, data;
	stack<int> s;
	
	cin >> total_queries;
	
	for(int index=0; index<total_queries;index++)
	{
		cin >> type;
		if(type == 2)
		{
			cin >> data;
			s.push(data);
		}
		else if(type == 1)
		{
				if(!s.empty())
				{
					cout << s.top() << endl;
					s.pop();
				}
				else
				{
					cout << "No Food" <<endl;
				}
		}
		
	}
	
	return 0;
}
