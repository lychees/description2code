#include <iostream>
using namespace std;
#define mod 1000000007
long long dp[100005];

int main()
{
   	int t;
   	scanf("%d",&t);
   	while(t--)
   	{
   		long long n;
    	scanf("%lld",&n);
       	dp[0]=1;dp[1]=1;dp[2]=2;
    	for(int i=3;i<=n;i++)
   	 	{
    		dp[i]=dp[i-1]+dp[i-2]+dp[i-3];
    		dp[i]=dp[i]%mod;
       	}
    	
    	cout<<dp[n]<<endl;
   	}
    return 0;
}
