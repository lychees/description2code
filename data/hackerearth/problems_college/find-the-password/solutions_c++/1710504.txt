const int mod = (int)1e9 + 7;

bool good(ll x) {
	while(x % 2 == 0) x /= 2;
	while(x % 3 == 0) x /= 3;
	while(x % 5 == 0) x /= 5;
	while(x % 7 == 0) x /= 7;
	return x == 1;
}

	ll x; cin >> x;

	if (x == 0) {
		cout << 10 << endl;
		return 0;
	}

     while(!good(x)) {
     	x += mod;
     }

     vector<int> v;

     for (int i = 9; i >= 2; i--) {
     	while(x % i == 0) {
     		 x /= i;
     		 v.pb(i);
     	}
     }

     reverse(all(v));

     if (v.empty()) v.pb(1);

     for (int i = 0; i < sz(v); i++) {
     	cout << v[i];
     }
     cout << endl;

	return 0;
}
