from __future__ import print_function
from math import *


def main():
    t = input()
    for _ in range(t):
        n = input()
        if (n < 4):
            print("1")
            continue
        zeros = 0
        i = 5
        while (i <= n):
            zeros += n//i
            i = i*5
        ans = ceil(log10(2*3.141592653589793*n)/2 +
                   n*log10(n/2.7182818284590452353))
        print(int(ans-zeros))


main()
