#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tc;
    cin >> tc;
    while(tc--)
    {
    	string s;
    	int n;
    	cin >> s >> n;
    	int v = (n*(n+1))/2;
    	if(v % 2 != 0){
    		for(int i = 0; i < s.length(); i++)
    			if(s[i] == '1')
                                s[i] = '0';
                        else
                                s[i] = '1';
    	}
    	int count = 0;
    	for(int i = 0; i < s.length(); i++)
        {
                if(s[i] == '1')
                        count++;
        }
        cout << count << endl;
    }
    return 0;
}
