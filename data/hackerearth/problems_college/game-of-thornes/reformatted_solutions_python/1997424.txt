a = raw_input()
n = len(a)
b = [[], []]
odd = 0
for i in a:
    if (i not in b[1]):
        b[0].append(a.count(i))
        b[1].append(i)
for i in b[0]:
    if (i % 2 == 1):
        odd += 1
if (n % 2 == 0 and odd == 0):
    print "YES"
elif (n % 2 == 1 and odd == 1):
    print "YES"
else:
    print "NO"
