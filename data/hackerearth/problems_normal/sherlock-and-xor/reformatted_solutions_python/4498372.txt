test_cases = int(raw_input())
for test in range(test_cases):
    N = int(raw_input())
    numbers = raw_input()
    num_array = numbers.split(" ")
    even = 0
    odd = 0
    for i in xrange(N):
        if int(num_array[i]) % 2 == 0:
            even += 1
        else:
            odd += 1
    print even*odd
