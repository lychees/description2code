#include<bits/stdc++.h>
using namespace std;
#define int long long
const int MAXN = 5e5+3;
int ar[MAXN][26];
int cnt[MAXN];
char str[MAXN];
#undef int
int main() {
	#define int long long
	scanf("%s",str+1);
	int len=strlen(str+1);
	memset(ar[0],0,sizeof(ar[0]));
	// h2   a2   c   k   e2   r2   t
	for(int i=1;i<=len;i++) {
		if(str[i]>='A' and str[i]<='Z')
			str[i]+=32;
	}
	for(int i=1;i<=len;i++) {
		cnt[i]=0;
		for(int j=0;j<26;j++)
			ar[i][j]=ar[i-1][j];
		ar[i][str[i]-97]++;
		if(i>=11) {
			bool f=true;
			if(ar[i]['a'-97]-ar[i-11]['a'-97]!=2)
				f=false;
			if(ar[i]['h'-97]-ar[i-11]['h'-97]!=2)
				f=false;
			if(ar[i]['c'-97]-ar[i-11]['c'-97]!=1)
				f=false;
			if(ar[i]['k'-97]-ar[i-11]['k'-97]!=1)
				f=false;
			if(ar[i]['e'-97]-ar[i-11]['e'-97]!=2)
				f=false;
			if(ar[i]['r'-97]-ar[i-11]['r'-97]!=2)
				f=false;
			if(ar[i]['t'-97]-ar[i-11]['t'-97]!=1)
				f=false;
			if(f) {
				for(int j=0;j<26;j++) {
					if(j=='a'-97 or j=='h'-97 or j=='c'-97 or j=='k'-97 or j=='e'-97 or j=='r'-97 or j=='t'-97);
					else if(ar[i][j]-ar[i-11][j]!=0)
						f=false;
				}
				if(f)
					cnt[i]=1;		
			}
			else
				cnt[i]=0;
		}
		cnt[i]+=cnt[i-1];
	}
	int q;
	scanf("%lld",&q);
	while(q--) {
		int a,b;
		scanf("%lld%lld",&a,&b);
		if(b-a+1<11)
			printf("0\n");
		else
			printf("%lld\n",cnt[b]-cnt[a+9]);
	}
	return 0;
}
