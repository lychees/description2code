#include <bits/stdc++.h>
using namespace std;

int Y[1003], countx[1003];

int main()
{
    int N;
    scanf("%d", &N);
    
    for(int i=1; i<=N; i++)
    {
    	scanf("%d", &Y[i]);
    	countx[Y[i]]++;
    }
    
    int ans = 0;
    
    for(int i=0; i<=1000; i++)
    {
    	if(!countx[i])
    		continue;
    	
    	for(int j=i; j<=1000; j++)
    	{
    		int tmp_ans;
    		
    		if(!countx[j])
    			continue;
    		
    		if(i!=j)
	    		tmp_ans = countx[i] + countx[j];
	    	else
	    		tmp_ans = countx[i];
    		
    		int k = 1, m = N;
    		
    		while((Y[k]<i || Y[k]>j) && k<=1000)
    			k++;
    		
    		while((Y[m]<i || Y[m]>j) && m>=0)
    			m--;
    		
    		if(k<=1000 && Y[k]!=i && Y[k]!=j)
    			tmp_ans++;
    		if(m>=0 && Y[m]!=i && Y[m]!=j)
    			tmp_ans++;
    		
    		ans = max(ans, tmp_ans);
    	}
    }
    
    printf("%d\n", ans);
    
    return 0;
}
