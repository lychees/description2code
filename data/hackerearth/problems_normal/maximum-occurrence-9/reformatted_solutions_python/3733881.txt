import operator
s = raw_input()
d = dict()
max_occur = 0
max_occur_char_asci = None
for c in s:
    if not d.has_key(c):
        d[c] = 1
    else:
        d[c] += 1
    if max_occur < d[c]:
        max_occur = d[c]

new_dict = dict()
for key, value in d.items():
    if value == max_occur:
        if max_occur_char_asci is None or max_occur_char_asci > ord(key):
            max_occur_char_asci = ord(key)

print chr(max_occur_char_asci), max_occur
