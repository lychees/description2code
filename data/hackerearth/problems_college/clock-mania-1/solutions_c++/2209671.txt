#include<iostream>
#include<stdio.h>
#include<math.h>
#include<stdlib.h>
using namespace std;
int angle(int h,int m)
{
	int errh,errm,exp1,err;
	int mh,hh;
	errh=(float)m/2;
	errm=(m%5)*6;
	mh=(m/5);
	if(mh==0)
	mh=12;
	if(h<=12)
	hh=h;
	else
	hh=h-12;
	err=errh-errm;
	exp1=fabs(hh-mh)*30;
	if(mh<hh)
	{
		if(exp1+err>180)
		return 360-(exp1+err);
		else
		return exp1+err;
	}
	else
	{
	
	if(exp1-err>180)
	return 360-(exp1-err);
	else
	{
		
	
	return fabs(exp1-err);
}
}
}
int main()
{
	
	int T,i;
	cin>>T;
	int *hr,*min;
	float val;
	hr=(int*)malloc(T*sizeof(int));
	min=(int*)malloc(T*sizeof(int));
	for(i=0;i<T;i++)
	{
	cin>>hr[i]>>min[i];	
	}
	for(i=0;i<T;i++)
	{
	val=angle(hr[i],min[i]);
	cout<<val<<"\n";	
	}
	return 0;
}
