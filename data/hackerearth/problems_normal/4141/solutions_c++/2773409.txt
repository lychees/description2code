#include <cstdio>

using namespace std;


int n , k , d , t , c ,sum = 0;



int main()
{


    scanf("%d", &t);

    while(t--){



      scanf("%d%d%d%d", &k, &d , &c, &n);

      sum = k;

      for(int i = 1; i < n; i++){

        sum = ( ( sum * 10 )  + d  ) % 41;
        int val = (d * c + k) % 10;
        k = d;
        d = val;
      }

      if(sum == 0) printf("YES\n");
      else printf("NO\n");
    }

    return 0;
}
