#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d ",&t);
	while(t--) {
		string a;
		getline(cin, a);
		string x[103];
		int k = 0;
		string temp = "";
		int total = 0;
		for(int i = 0; i < a.size(); i++) {
			if(a[i] == ' ') {
				x[k] = temp;
				k++;
				temp = "";
			}
			else if(a[i] >= '0' && a[i] <= '9') {
				total = total + (a[i] - '0');
			} 
			else {
				temp = temp + a[i];
			}
		}
		x[k] = temp;
		k++;
		sort(x,x+k);
		int l = 0;
		for(int i = 0; i < k; i++) {
			int j = i;
			x[l] = x[i];
			l++;
			while(x[j] == x[j+1] && j+1 < k) {
				j++;
			}
			i = j;
		}
		for(int i = 0; i < l; i++) {
			total = total - x[i].size();
			cout << x[i] << " " << abs(total) << " ";
		}
		cout << endl;
	}
	return 0;
}
