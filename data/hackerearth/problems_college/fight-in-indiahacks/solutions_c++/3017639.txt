#include<bits/stdc++.h>
using namespace std;
int arr[1000001]={};
int main()
{
	int t,i;
	scanf("%d",&t);
	while(t--)
	{
		int n,m1,m2;
		scanf("%d%d%d",&n, &m1, &m2);
		arr[5]=5;
		memset(arr,0,n*sizeof(int));
		int x,y;
		for(i=0;i<=n;i++)
		{
			x=-1;
			y=-1;
			if(i-m1>=0)
				x=arr[i-m1];
		
			if(i-m2>=0)
				y=arr[i-m2];
			
			if(x==0 || y==0)
				arr[i]=1;
			else
				arr[i]=0;				
		}
	//	for(i=0;i<=n;i++)
	//		printf("%d ",arr[i]);
		if(arr[n]==1)
			printf("1\n");
		else
			printf("0\n");	
	}
	return 0;
}
