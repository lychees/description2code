t = input()
for i in range(t):
    n = input()
    if n % 2 == 0:
        n = n/2
    else:
        n = n/2+1
    prac = 1
    while n > 0:
        i = 2
        prod = 26
        while i < n:
            prod = (prod*prod) % 1000000009
            i = i*2
        prac = (prod*prac) % 1000000009
        i = i/2
        n = n-i
    print prac
