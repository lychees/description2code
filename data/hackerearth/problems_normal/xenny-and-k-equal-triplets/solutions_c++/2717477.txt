#include<iostream>
using namespace std;

long long int gcd(long long int n, long long int m)
{
	long long int g,rem;
 
	while (n != 0)
	{
		rem = m % n;
		m = n;
		n = rem;
	}
 
	g=m;
 
	return g;
}



long long int a[1000001];
int main()
{
	int t;
	scanf("%d",&t);
	long int n,i,count;
	long long int k,num,num1,den,den1;
	
	while(t--)
	{
	scanf("%ld%lld",&n,&k);
	num=0;den=0;count=0;
	
	for(i=0;i<n;i++)
	{
		scanf("%lld",&a[i]);
		if(a[i]==k)
		count++;
	}
	//cout<<"count"<<count;
	num=count*(count-1)*(count-2);
	den=n*(n-1)*(n-2);
	num1=num/gcd(num,den);
	den1=den/gcd(num,den);
	printf("%lld/%lld\n",num1,den1);	
	}
}
