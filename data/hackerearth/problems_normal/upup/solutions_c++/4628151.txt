#include<iostream>
#include<string>
using namespace std;

int main()
{
    string s;
    int i,l;
    getline(cin,s);
    l=s.length();
    if(s[0]>='a' && s[0]<='z')
    {
    	s[0]=s[0]-32;
    }
    for(i=1;i<l;i++)
  {
    	if(s[i]==' ')
    	{
    		if(s[i+1]>='a' && s[i+1]<='z')
    		{
    		s[i+1]=s[i+1]-32;
    		}
    	}
    }
    cout<<s;
    return 0;
}
