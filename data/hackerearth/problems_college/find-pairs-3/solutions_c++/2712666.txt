#include <bits/stdc++.h>
using namespace std;
#define lli long long int
int a[2000010];

int main()
{
    int t;
    int n,i;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<=2000000;i++)
    		a[i]=0;
    	for(i=0;i<n;i++)
    	{
    		int x;
    		cin>>x;
    		a[x+1000000]++;
    	}
    	lli sum=0;
    	for(i=0;i<=2000000;i++)
    	{
    		lli temp=a[i];
    		temp*=(a[i]+1);
    		temp>>=1;
    		sum+=temp;
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
