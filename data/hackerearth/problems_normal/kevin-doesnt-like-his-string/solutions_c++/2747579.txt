#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,k,n,l,m;
    char s[100001];
    int a[36]={0},b[36]={0};
    scanf("%d%d%s",&n,&k,s);
    if(k==1)
    {
    	printf("%s\n",s);
    	return 0;
    }
    for(i=0;i<n;++i)
    {
    	a[s[i]-'a']++;
    }
    l=0;
    for(i=0;i<n;++i)
    {
    	if(!a[s[i]-'a'])
    	{
    		s[i]='z'+1;
    		for(j=0;j<26;++j)
    		{
    			if(b[j])
    			{
    				b[j]--;
    				s[i]=j+'a';
    				break;
    			}
    			else if(a[j] && l<k)
    			{
	    			a[j]--;
    				l++;
    				s[i]=j+'a';
	    			break;
    			}
    		}
    		continue;
    	}
    	a[s[i]-'a']--;
    	for(j=0;j<(s[i]-'a');++j)
    	{
    		if(b[j] && l<k)
    		{
    			l++;
    			b[j]--;
    			b[s[i]-'a']++;
    			s[i]=j+'a';
    			break;
    		}
    		else if(a[j] && (l+1)<k)
    		{
    			a[j]--;
    			l+=2;
    			b[s[i]-'a']++;
    			s[i]=j+'a';
    			break;
    		}
    	}
    	
    }
    printf("%s\n",s);
    return 0;
}
