#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    char s[100];
    int i,c,f=0;
    cin>>s;
    c=strlen(s);
  if((c>=1)&&(c<=100))
  {
  	for(i=0;i<c;i++)
  	{
  		if(s[i]!=s[c-i-1])
  		{f=1;
  		break;
  		}
  }
  if(f)
  cout<<"NO";
else 
cout<<"YES";
}
    return 0;
}
