#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;
#define max(a,b) a>b?a:b
int main()
{
    long n,i,j=0,k=0,x;
    cin >> n;
    
    long pos[n],neg[n];
    
    
    for(i=0;i<n;i++)
    {
    	cin >> x;
    	if(x>=0)
    	   pos[j++]=x;
    	else
    	   neg[k++]=x;
    }
    if(n==2 && j==1 && k==1)
    {
    	cout << pos[0]*neg[0];
    	return 0;
    }
    sort(pos,pos+j);
    sort(neg,neg+k);
    
    long long ans= pos[j-1]*pos[j-2] > neg[0]*neg[1] ? pos[j-1]*pos[j-2]:neg[0]*neg[1];
    cout << ans;
    return 0;

}
