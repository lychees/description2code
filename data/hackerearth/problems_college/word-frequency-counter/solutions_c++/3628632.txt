#include<bits/stdc++.h>
using namespace std;
struct stu{
	int pos;
	int count;
	string str;
};
int main()
{
	vector<string>vec;
	string str;
	int i,j,k;
	getline(cin,str);
	int l = str.length();
    for(i=0;i<l;i++)
    {
    	if(islower(str[i]))
    	{
    		str[i]= toupper(str[i]);
		}
	}
	string st="";
	for(i=0;i<=l;i++)
	{
		if(str[i]!=' ' && i != l)
		{
			st = st + str[i];
		}
		else
		{
			vec.push_back(st);
			st.clear();
			
		}
	}
	int n = vec.size();
	int count[n];
	for(i=0;i<n;i++)
	{ 
		count[i] = 0;
	}
	int c=0;
	for(i=0;i<n;i++)
	{
		c=0;
		if(count[i]==0)
		{
			c++;
		}
		for(j=i+1;j<n;j++)
		{
			if(vec[i]==vec[j] && count[j]==0)
			{
				count[j]= -1;
				c++;
			}
		}
		if(count[i]!=-1)
		cout<<vec[i]<<" "<<c<<"\n";
		count[i] = -1;
	
	}
}
