#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{	
	vector<int> v1;
	vector<int> v2;
	int t,n,x,y;
	cin>>t;
	while(t--)
	{
		cin>>n>>x>>y;
		for(int i=0;i<n;i++)
		{
			int t1,t2;
			cin>>t1>>t2;
			v1.push_back(t1);
			v2.push_back(t2);
		}
		sort(v1.begin(),v1.end());
		sort(v2.begin(),v2.end());
		int cost=0;
		for(int i=0;i<n;i++)
		{
			if(v1[i]>v2[i])
			{
				cost=cost+((v1[i]-v2[i])*y);
			}
			else if(v1[i]<v2[i])
			{
				cost=cost+((v2[i]-v1[i])*x);
			}
		}
		cout<<cost<<endl;
		v1.clear();
		v2.clear();
	}
}
