#include<bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<int, int> pi;

#define MP make_pair
#define PB push_back

#define S second
#define MOD 1000000007

int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin >> t;
	while(t--)
	{
		int n, i, j;
		cin >> n;
		int a[n], b[n];
		for(i = 0; i < n; i++)
			cin >> a[i];
		sort(a + 1, a + n);
		for(i = 0; i < n; i++)
			cin >> b[i];
		sort(b, b + n);
		reverse(b, b + n);
		int flag = 0;
		for(i = 1; i < n; i++)
		{
			if(a[i] * b[i] >= a[0] * b[0])
			{
				flag = 1;
				break;
			}
		}
		if(flag)
			cout << "NO" << endl;
		else
			cout << "YES" << endl;
	}
	return 0;
}
