#include<iostream>
using namespace std;

int digits(long long int n)
{
  int count_digits=0;
  while(n!=0)
  {
    n=n/10;
    count_digits++;
  }
  return count_digits;
}

unsigned long long int cal(unsigned long long int n)
{
unsigned long long int sum=0;
  while(n!=0)
  {
    sum+=n%10;
    n=n/10;

  }
  return sum;
}

main()
{
  int t;
  cin>>t;
  for(int z=0;z<t;z++)
  {
long long int m;
cin>>m;
unsigned long long int sum=0;
for(int i=0;i<m;i++)
{
unsigned long long int a,b;
  cin>>a>>b;
  sum+=a*b;
while(digits(sum)>1)
{
  sum=cal(sum);
}
}

cout<<sum<<endl;
  }
}
