lps = [0, 1, 2, 4]
mod = 1000000007


def count(n):
    if (len(lps)) < n+1:
        for x in range(len(lps), n+1):
            lps.append((lps[x-1]+lps[x-2]+lps[x-3]) % mod)
        return lps[n]
    else:
        return lps[n]


t = input()
# print t
for _ in range(0, t):
    n = input()
    print count(n)
