cases = int(raw_input())

for case in range(cases):
	string = raw_input()
	if len(string) >= 10:
		print "YES"
		continue
	f = False
	for i in range(1, len(string)):
		for j in range(i + 1, len(string)):
			for k in range(j + 1, len(string)):
				s1 = string[ : i]
				s2 = string[i : j]
				s3 = string[j : k]
				s4 = string[k : ]
				if len(set([s1, s2, s3, s4])) == 4:
					f = True
					break
	if f:
		print "YES"
	else:
		print "NO"
