#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    long int n,q,x,i,j;
    long long int a[100005],b[100005];
    memset(b,0,sizeof(b));
     int f;
      cin>>n;
       for(i=0;i<n;i++)
        {
        	cin>>a[i];
        	b[a[i]]++;
        	
        }
        cin>>q;
        for(i=0;i<q;i++)
        { 
        	cin>>x;
        	if(b[x]>0)
        	cout<<"I Like You"<<endl;
        	else
        	{
        		  f=-1;
        		for(j=0;j<n;j++)
        		{
        			if(b[x-a[j]]>=1)
        			{
        			
        				f=1;
        				cout<<"I Like You"<<endl;
        				break;
        			}
        			else
        			f=0;
        		}
        		if(f==0)
        		cout<<"I Hate You"<<endl;
        	}
        
        }
    return 0;
}
