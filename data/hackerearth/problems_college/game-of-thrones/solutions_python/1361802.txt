def main():
	t = int(input())
	while t>0:
		#print (t)
		n = int(input())
		f = []
		x = raw_input()
		f = [int(y) for y in x.split()]
		#f = x.split( )
		f.sort();
		i = 1
		count=1
		while i<n:
			if(f[i] != f[i-1]):
				count = count+1
			i = i+1
		print (n-count)
		t = t-1
