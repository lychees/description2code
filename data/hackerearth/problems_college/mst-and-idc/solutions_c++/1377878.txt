#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	int T;
	cin>>T;
	while(T--) {
		int n;
		cin>>n;
		int arr[n];
		int mn=10000,mx=0;
		for(int i=0;i<n;i++) {
			cin>>arr[i];
			mn=min(mn,arr[i]);
			mx=max(mx,arr[i]);
		}
		int md=mx-mn;
		cout<<2*md<<endl;
	}
	return 0;
}
