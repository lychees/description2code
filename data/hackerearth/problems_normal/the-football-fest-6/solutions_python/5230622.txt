import sys
from functools import partial

to_ints = partial(map,int)

def readline():
    return sys.stdin.readline().strip()
def read_int():
    return int(readline())
def read_ints(sep=None):
    return to_ints(readline().split(sep))

T = read_int()
for I in xrange(T):
	N, P = read_ints()
	prev = None
	cur = str(P)
	for i in xrange(N):
		p = readline()
		if p[0] == 'P':
			prev = cur
			cur = p[2:]
		else:
			prev, cur = cur, prev
			
	print 'Player', cur
