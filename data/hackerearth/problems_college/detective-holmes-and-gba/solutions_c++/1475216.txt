#include <iostream>
using namespace std;
typedef long long ll;
int main()
{
//    cout << "Hello World!" << endl;
  ll t;
  cin>>t;
  while(t--)
  {
  	ll n,  m, s;
  	cin>>n>>m>>s;
  //	cout<<(n-s+1);
  	if((n-s+1)>m)
  	{
  		cout<<(s+m-1)<<endl;
  	}
  	else
  	{
  		m-=n-s+1;
  		m%=n;
  		cout<<m<<endl;
  	}
  }
  return 0;
}
