#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10;

int a[N];

int main() {
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d", &a[i]);
	}
	sort(a, a + n);
	int cur = 0, cnt = 0;
	while (cur < n) {
		cnt++;
		int next = cur + 1;
		while (next < n && a[next] <= a[cur] + 4) {
			next++;
		}
		cur = next;
	}
	printf("%d\n", cnt);
	return 0;
}
