t = int(raw_input())
while t > 0:
	t = t - 1
	dict = {}
	n = int(raw_input())
	i = 0
	if n == 0:
		print 0
		continue
	max = 1
	while i < n:
		i = i + 1
		cur = int(raw_input())
		if dict:
			if cur in dict:
				dict[cur] = dict[cur] + 1
				if max < dict[cur]:
					max = dict[cur]
			else:
				dict[cur] = 1
		else:
				dict[cur] = 1
	print max
