#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
bool days(ll a[],ll t,ll m,ll n)
{
	ll count=1,i,sum=0;
	for(i=0;i<n;i++)
	{
		
		if(sum+a[i]<=t)
		sum=sum+a[i];
		else
		{
			sum=a[i];
			count++;
		}
	}
	return (count<=m)?true:false;
}


int main()
{
	
	ll n,m,i,l=1,r,mid;
	cin>>n>>m;
	ll a[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		l=max(l,a[i]);
	}
		
	r=1ll*mod*mod;
	
	while(l<r)
	{
		mid=(l+r) /2;
		if(days(a,mid,m,n))
		{
			r=mid;
		}
		else
		l=mid+1;
		
	}
	cout<<l<<endl;
	return 0;
	
}
