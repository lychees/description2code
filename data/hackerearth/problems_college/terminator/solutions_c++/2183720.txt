#include<iostream>
#include<cstdio>
#include<string>
#include<cstring>
#include<set>
using namespace std;
typedef long long int ll;
int main()
{
    ll t,i,j;
    cin>>t;
    string s;
    while(t--)
    {
        cin>>s;
        ll n=s.length();
        for(i=0;i<n;)
        {
            if(s[i]=='x')
            {
                for(j=i;j>=0;j--)
                {
                    if(s[j]=='*')
                        break;
                    else if(s[j]=='o')
                        s[j]='a';
                }
                for(j=i;j<n;j++)
                {
                    if(s[j]=='*')
                        break;
                    else if(s[j]=='o')
                        s[j]='a';
                }
                i=j;
            }
            else
                i++;
        }
        for(i=0;i<n;i++)
        {
            if(s[i]!='a')
                cout<<s[i];
        }
        cout<<'\n';
    }
    return 0;
}
