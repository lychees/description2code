using namespace std;

#define ll long long
#define vt vector
#define mod 1000000007

ll a[100002];

int main()
{
	ll t;
	scanf("%lld", &t);
	while (t--)
	{
		ll n, m, z, nval = 0;
		scanf("%lld %lld %lld", &n, &m, &z);
		memset(a, 0, sizeof(a));
		for (int i = 0; i < n; i++)
		{
			scanf("%lld", &a[i]);
			a[i] %= m;
			if ((n - i) % 2 == 1)
				nval = (nval + a[i]) % m;
			else
				nval = (nval - a[i] + m) % m;
		}
		a[n] = nval;
		ll p = n % 2;
		n++;
		if (p == 1)
		{
			ll y = 1;
			printf("%lld\n", (y * a[z % n] + m) % m);
		}
		else
		{
			ll x = (z) / n, y;
			if (x % 2 == 1)
				y = -1;
			else
				y = 1;
			printf("%lld\n", (y * a[z % n] + m) % m);
		}
	}
	return 0;
}
