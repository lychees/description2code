#include <iostream>
#include<set>
using namespace std;
int len;
bool palin(string s)
{
for	(int i=0;i<len;i++)
{
	if(s[i]!=s[len-1-i])
	return false;
}
return true;
}
int main()
{
	int i,t,c;
	string s1,s2,str;
	
	scanf("%d",&t);
	while(t--)
	{
		c=0;
		cin >>s1>>s2;
		len=s1.length()+s2.length();
		str=s2+s1;
		if(palin(str))
		{
		c++;
		}
		for(i=1;i<=s1.length()-1;i++)
	{
		str=s1.substr(0,i)+s2+s1.substr(i,(s1.length()-i));
	if(palin(str))
		{
		c++;
		}
	}
	str=s1+s2;
	if(palin(str))
		{
		c++;	
		}
		printf("%d\n",c);
	}
    
    return 0;
}
