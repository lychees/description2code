#include <iostream>
using namespace std;

int reverse(int n)
{
	int rev = 0;
	while (n != 0) {
		rev = rev * 10 + n % 10;
		n /= 10;
	}
	return rev;
}
int main()
{
    int t;
    cin >> t;
    while (t--) {
    	int n;
    	cin >> n;
    	int x = reverse(n);
    	int sqn = n * n;
    	int sqm = x * x;
    	int xx = reverse(sqm);
    	if (xx == sqn && x != n) {
    		cout << x << endl;
    	} else {
    		cout << "No" << endl;
    	}
    }
    return 0;
}
