#include <iostream>
using namespace std;
long long int gcd(long long int a , long long int b)
{
   if(b==0) return a;
   a%=b;
   return gcd(b,a);
}

{

    while(t--)
    {
     		cin>>a>>b>>k;

    		q=max(a,b);
    		count=0;
    		for(i=p;i>=1;i--)
    		{
    			if(q%i==0 && p%i==0)
    			{
    				count++;

    					}
    			}
    		}

    }

}
