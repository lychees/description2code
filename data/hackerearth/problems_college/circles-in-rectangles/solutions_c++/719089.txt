#include<iostream>
#define mod 1000000007
using namespace std;

int main()
{
	long long l, b, d,tot;
	int t;
	cin >> t;
	while (t--)
	{
		tot = 0;
		cin >> l >> b >> d;
		if (d > l || d > b)
			cout << "0\n";
		else
		{
			l = l / d;
			b = b / d;
			l = l%mod;
			b = b%mod;
			cout << (l*b%mod) << endl;
		}
	}

	return 0;
}
