#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,q;
    cin>>n>>q;
    int arr[n];
    for(int i=0;i<n;i++){
    	cin>>arr[i];
    }
    while(q--){
    	int x;
    	cin>>x;
    	if(x==1){
    		int val;
    		cin>>val;
    		int out=-1;
    		for(int i=0;i<n;i++){
    			if(arr[i]>=val){
    				cout<<i+1<<endl;
    				out=1;
    				break;
    			}
    		}
    		if(out==-1)			
    			cout<<out<<endl;
    	}
    	else if(x==0){
    		int j,v;
    		cin>>j>>v;
    		arr[j-1]=v;
    	}
    }
    return 0;
}
