#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef vector<int> vi;
typedef pair<int, int> ii;
typedef vector<ii> vii;
typedef vector<vii> graph;

const int MOD = 1e9 + 7;

unordered_map<ll, int> dp;
int sol(ll k,unsigned int sum){
	if (k == 1)
		return  sum == 1 || sum == -1;
	ll p = (k << 32) | sum;
	if (dp.find(p) != dp.end())
		return dp[p];
	int r = 0;
	r += sol(k - 1, sum + 1);
	r %= MOD;
	//r += sol(k - 1, sum);
	//r %= MOD;
	r += sol(k - 1, sum - 1);
	r %= MOD;
	dp[p] = r;
	return r;
}
int main(){
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	int n, k, t, sum;
	cin >> n >> k >> t;
	sum = t;
	vi c(k);
	for (int &i : c)
		cin >> i, sum-=i;
	int ans = sol(k, sum);
	cout << ans;
}
