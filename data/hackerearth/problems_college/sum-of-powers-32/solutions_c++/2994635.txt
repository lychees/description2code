#include <bits/stdc++.h>
using namespace std;

int dp[1000000];
int main()
{
    int t,i,j,pos;
    dp[1]=1;
    for( i=2;i<=1000;i++)
    	for( j=2;pow(i,j)<=1000000;j++)
    	{
    		 pos=pow(i,j);
    		dp[pos]=1;
    	}
	cin >> t;
	while(t--)
    {
    	int n;
    	cin >> n;
    	bool f=false;
    	for(int i=1;i<=n/2;i++)
    	{
    		if(dp[i] && dp[n-i])
    		{
    			f=true;
    			break;
    		}	
    	}
    	if(f)
    		cout << "Yes" << endl;
    	else
    		cout << "No" << endl;
    }
    return 0;
}
