#include <iostream>
#include<cstring>
using namespace std;

#define MAX 1000000

int main()
{
    char str[MAX],stc[MAX],syn,ant;
    long long int S=0,A=0,N;
    cin>>str;
    syn=str[0];
    ant=str[0]+3;
    cin>>N;
    while(N--)
    {
    	cin>>stc;
    	if(stc[0]==syn && strlen(str)==strlen(stc))
    	S++;
    	if(stc[0]==ant && strlen(str)==strlen(stc))
    	A++;
    }
    cout<<S<<" "<<A<<endl;
    return 0;
}
