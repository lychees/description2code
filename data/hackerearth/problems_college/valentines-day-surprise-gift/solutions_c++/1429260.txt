#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const int md=1e9+7;

int dp[1000005];

int pow(int a,int b) {
    if(b==0) {
        return 1;
    }
    int p=pow(a,b/2);
    p=ll(p)*p%md;
    if(b&1) {
        p=ll(p)*a%md;
    }
    return p;
}
int solve(int n) {
    if(n<1000005) {
        return dp[n];
    }
    int res=ll(pow(2,int(ceil(n/4.0))))*solve(int(ceil(n/5.0)))%md;
    res=ll(res)*solve(int(ceil(n/10.0)))%md;
    return res;
}
int main() {
    int tt;
    int primes[]={2,3,5,7};
    dp[0]=1;
    dp[1]=1;
    dp[2]=1;
    dp[3]=3;
    dp[4]=3;
    dp[5]=15;
    dp[6]=15;
    dp[7]=105;
    dp[8]=105;
    dp[9]=105;
    dp[10]=105;
    for(int i=11;i<1000005;i++) {
        dp[i]=ll(pow(2,int(ceil(i/4.0))))*dp[int(ceil(i/5.0))]%md;
        dp[i]=ll(dp[i])*dp[int(ceil(i/10.0))]%md;
    }
    scanf("%d",&tt);
    while(tt--) {
        int n;
        scanf("%d",&n);
        int ans;
        if(n>=1000005) {
            ans=solve(n);
        } else {
            ans=dp[n];
        }
        int result=-1;
        for(int i=1;i<=5;i++) {
            for(int j=1;j<=5;j++) {
                int res1=ll(ans)*i%md;
                int res2=ll(ans)*j%md;
                if(__gcd(res1,res2)==ans) {
                    int pw=pow(res1,res2);
                    if(pw>result) {
                        result=pw;
                    }
                }
            }
        }
        cout<<result<<endl;
    }
    return 0;
}
