for t in xrange(input()):
    n, m = [int(i) for i in raw_input().split()]
    ary = [int(i) for i in raw_input().split()]
    dic = {}
    for i in xrange(n):
        dic[ary[i]] = 1
    for j in xrange(n, n+m):
        k = dic.get(ary[j], 0)
        if k:
            print "YES"
        else:
            print "NO"
            dic[ary[j]] = 1
