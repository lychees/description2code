ll po(ll a,ll b,ll c)
{
	ll ans=1;
	while(b)
	{
		if(b&1)
			ans=(ans*a)%c;
		b=b/2;
		a=(a*a)%c;
	}
	return ans;
}
ll po(ll a,ll b)
{
	ll ans=1;
	while(b)
	{
		if(b&1)
			ans=(ans*a)%M;
		b=b/2;
		a=(a*a)%M;
	}
	return ans;
}
int main()
{
    int t,i,n,k,low_eq,j,low[200],upr[200],eql[200],pos,en,c1,c2;
    cin>>t;
    while(t--)
    {
        int cum[200],a[200];
        cin>>n>>k;
        for(i=0;i<n;i++)
        {
            cin>>a[i];
        }
        sort(a,a+n);
    int lo=0,up=0,eq=0;

    for(j=1;j<=100;j++)
    {
        lo=0,up=0,eq=0;
        for(i=0;i<n;i++)
        {
            if(a[i]<j)
                lo++;
            else if(a[i]>j)
                up++;
            else
                eq++;
        }
        low[j]=lo;
        upr[j]=up;
        eql[j]=eq;
    }
    cum[0]=0;
    for(i=0;i<n;i++)
    {
        cum[i+1]=cum[i]+a[i];
       // chk(cum[i+1]);
    }
    int mini=1e9;
        for(i=1;i<=100;i++)
        {
            for(pos=1;pos<=n-k+1;pos++)
            {
                en=pos+k-1;
                low_eq=low[i]+eql[i];
                c1=0;
                c2=0;
                if(low_eq>=en)
                {
                    c1=3*k*i-3*(cum[en]-cum[pos-1]);
                }
                else if(low_eq<pos)
                {
                    c2=5*(cum[en]-cum[pos-1])-5*k*i;
                }
                else
                {
                    c1=3*((low_eq-pos+1)*i-(cum[low_eq]-cum[pos-1]));
                    c2=5*(cum[en]-cum[low_eq]-(en-low_eq)*i);
                }
                if(c1+c2<mini)
                    mini=c1+c2;
            }
        }
        cout<<mini<<"\n";
    }
	return 0;
}
