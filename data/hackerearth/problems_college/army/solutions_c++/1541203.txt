#include <bits/stdc++.h>
using namespace std;
#define long int64_t

int code[100010];
int n;

void enter () {
	scanf("%d", &n);

	for (int i = 1, j; i <= n; ++i)
		for (j = i; j; j /= 10)
			code[i] |= 1 << (j % 10);
}

void solve () {
	int q, k, i, c;

	scanf("%d", &q);
	while (q--) {
		scanf("%d", &i);

		c = 0;
		while (i--) scanf("%d", &k), c |= 1 << k;

		k = 0;
		for (i = 1; i <= n; ++i)
		if (code[i] & c) ++k;

		printf("%d\n", k);
	}
}

int main () {
	#ifdef isMyPC
		freopen("Army.inp","r",stdin);
	#endif

	enter ();
	solve ();

	return 0;
}
