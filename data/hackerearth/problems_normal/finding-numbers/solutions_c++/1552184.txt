#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t,ways;
    long long int i,z,m,n,x,y;
    cin>>t;
    while(t--)
    {
        ways=0;
        cin>>z;
        cin>>m;
        cin>>n;
        if(m>=n)
        {
            x=n;
            y=m;
        }
        else
        {
            x=m;
            y=n;
        }
        for(i=1;i<=sqrt(z);i++)
        {
            if(z%i==0 && (z/i)<=y)
            {
                ways++;
                if((z/i)<=x && (z/i)!=i)
                {
                    ways++;
                }
            }
        }
        cout<<"\n"<<ways;
    }

    return 0;
}
