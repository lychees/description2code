#include <bits/stdc++.h>

using namespace std;

bool debug = false;

string dfs (int * index, string s, int col, int x, string ans) {
	
	if (col == x)	ans += s[*index];
	if (debug)	cout << s[*index];
	(*index) ++;
	if (s[*index] == '(') {
		if (debug)	cout << s[*index];
		(*index) ++;
	}
	
	if (s[*index] == '.') {
		if (debug)	cout << s[*index];
		(*index) ++;
	} else {
		ans = dfs (index, s, col - 1, x, ans);
	}
	
	if (s[*index] == ')') {
		if (debug)	cout << s[*index];
		(*index) ++;
	}
	
	if (s[*index] == '(') {
		if (debug)	cout << s[*index];
		(*index) ++;
	}
	
	if (s[*index] == '.') {
		if (debug)	cout << s[*index];
		(*index) ++;
	} else {
		ans = dfs (index, s, col + 1, x, ans);
	}
	
	if (s[*index] == ')') {
		if (debug)	cout << s[*index];
		(*index) ++;
	}
	return ans;
}

int main()
{
	int t;
	cin >> t;
	while (t--) {
		int n, i = 0;
		string s, ans = "";
		cin >> n >> s;
		ans = dfs (&i, s, 0, n, ans);
		if (ans.length() == 0)	cout << "Common Gandhijee!\n";
		else {
			sort (ans.begin(), ans.end());
			cout << ans << endl;
		}
	}
	return 0;
}
