#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int t,n;
	cin>>t;
	
	while(t--){
		cin>>n;
		
		int a[n];
		
		for(int i=0;i<n;i++){
			cin>>a[i];	
		}
		
		sort(a,a+n);
		
		int max=0;
		int min=100000000;
		
		for(int i=0;i<n/2;i++){
			int knowledge=a[i]+a[n-1-i];
			
			if(knowledge<min){
				min=knowledge;
			}
			
			if(knowledge>max){
				max=knowledge;
			}
		}
		
		cout<<max-min<<"\n";
	}
    return 0;
}
