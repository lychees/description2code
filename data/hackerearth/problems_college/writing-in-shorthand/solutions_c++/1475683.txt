#include<iostream>
#include<cstdio>
#include<cstring> 

using namespace std;

int main() {
	int n,m,i,j;
	cin>>n>>m;
	char w[m][2][11];
	char s[n][11];
	for(i=0;i<m;i++) {
		cin>>w[i][0]>>w[i][1];
	}
	for(i=0;i<n;i++) {
		cin>>s[i];
	}
	for(i=0;i<n;i++) {
		for(j=0;j<m;j++) {
			if(strcmp(s[i],w[j][0])==0)
				break;
		}
		if(strlen(s[i])>strlen(w[j][1]))
			cout<<w[j][1]<<" ";
		else
			cout<<s[i]<<" ";
	}
	return 0;
}
