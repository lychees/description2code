#include<bits/stdc++.h>
using namespace std;

int main()
{
	int q,i,j,n;
	string s;
	cin>>s;
	cin>>q;
	vector<int> a(s.size()+1),b(s.size()+1);
	map<char,int> check,curr;
	string p="HACKEREARTH";
	for(i=0;i<p.size();i++)
	 {check[p[i]]++;
	 }
	for(i=0;i<11 && i<s.size();i++)
	 {
	 	if(s[i]>='a')
	 	  s[i]-=32;
	 	curr[s[i]]++;
	 }
	if(curr==check)
	  {
	  a[i]=1;
	  b[i-11]=1;
     }
		for(i=11;i<s.size();i++)
	 { 
	   	if(s[i]>='a')
	 	  s[i]-=32;
	 	curr[s[i]]++;
	 	
	 	curr[s[i-11]]--;
	 	if(curr[s[i-11]]==0)
	 	  curr.erase(s[i-11]);
	 	if(curr==check)
	 	  {
	 	  a[i+1]=1;
	 	  b[i+1-11]=1;
	      }
	 }
	for(i=1;i<=s.size();i++)
	  {
	  a[i]+=a[i-1];
	//  cout<<a[i]<<" ";
     }
    
	while(q--)
	 {  int l,r;
	 //	cin>>l>>r;
	 	scanf("%d%d",&l,&r);
	   if(l+10<=r)
	     printf("%d\n",a[r]-a[l+9]);
	 	
	  else
	    printf("0\n");
	 }
}
