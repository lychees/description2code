#include <bits/stdc++.h>
using namespace std;

int main()
{
   string s;
   int T;
   cin >> T;
   
   while (T--)
   {
   	cin >> s;
   	
   	int cou = 0;
   	
   	int i = 0;
   	int j =s.size()-1;
   	
   	while(i < j )
   	{
   		
   	cou += abs(s[i]-s[j]);
   	i++;
   	j--;
   	}
   	cout << cou << endl;
   	
   }
    return 0;
}
