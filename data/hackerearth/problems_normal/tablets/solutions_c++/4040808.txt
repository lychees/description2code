#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	cin >> n;
	int a[n],dp[n];
	for (int i = 0;i < n; i++) {
		cin >> a[i];
	}
	dp[0] = 1;
	for (int i = 1; i < n; i++) {
		if (a[i] > a[i-1]) {
			dp[i] = dp[i-1] + 1;
		} else {
			dp[i] = 1;
		}
	}
	for (int i = n-2; i >= 0; i--) {
		if (a[i] > a[i+1] && dp[i] <= dp[i+1]) {
			dp[i] = dp[i+1] + 1;
		}
	}
	long long tab = 0;
	for (int i = 0; i < n; i++) {
		tab += dp[i];
	}
	cout << tab << endl;
	return 0;
}
