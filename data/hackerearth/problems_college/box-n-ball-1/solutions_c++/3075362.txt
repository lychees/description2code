#include <iostream>
#include <cstdio>
#include<cmath>
using namespace std;
long long int readInt()
{ long long int i=1;
  int cc = getchar_unlocked( );
  for (;(cc < '0' || cc > '9')&&cc!='-';){
		cc = getchar_unlocked( );
	} //ignores char stream other than 0-9.
  long long int ret = 0;
  if(cc=='-'){
    i=-1;
    cc=getchar_unlocked();
  }
  for (;cc >= '0' && cc <= '9';) // if stream of numbers comes, start taking input
  {
    ret = ret * 10 + cc - '0'; // convert each char to integral digit and extend the final number
    cc = getchar_unlocked( );          // input continues
  }
  return (i*ret);                   // return final extended number.
}
int main()
{
    int T=0;
    T=readInt();
    for(int t=0;t<T;t++){
    	long long int L,B,H;
    	L=readInt();
    	B=readInt();
    	H=readInt();
    	long long int max_rad = min(L,min(B,H));
    	printf("%lld",max_rad/2);
	    if(max_rad%2 ==1)
    		printf(".5");
    	else
    		printf(".0");
    	printf("\n");
    
    }
    return 0;
}
