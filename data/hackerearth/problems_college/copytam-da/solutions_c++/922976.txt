#include <iostream>
#include<string>
#include <algorithm>    // std::sort
#include<stdlib.h>
using namespace std;

int main()
{
	int t,n,i=0,c=0;
	cin>>t;
	while(t)
	{
		cin>>n;
		int b[n];
		while(n)
		{
			cin>>b[i];
			++i;
			--n;
		}
		sort(b,b+i);
		i--;
		for(int j=0;j<i;j++)
		{
			if(b[j]!=b[j+1])
			c++;
		}
	//	if(c==0)
	//	cout<<1<<endl;
	//	else
		cout<<c+1<<endl;
		c=0;
		i=0;
		--t;
	}
	return 0;
}
