#include <bits/stdc++.h>
using namespace std;
    
 

int main()
{
    //cout << "Hello World!" << endl;
    int t,i,j,k;
    cin>>t;
    while(t--)
    {
    	string ch;
    	cin>>ch;
    	int len=ch.length();
    	for( i=0;i<len;i++)
    	{
    		for(j=i, k=len-1;j<len;j++,k--)
    		{
    			if(ch[j]!=ch[k])
    			{
    				break;
    			}
    		}
    		if(j>k)
    		{
    			break;
    		}
    	}
    	cout<<len+i<<endl;
     }
    return 0;
}
