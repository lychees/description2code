#include<iostream>
#include<vector>
using namespace std;

int main(){
	int n;
	cin>>n;
	vector<vector<int> > isBetter(1<<(n+2),vector<int>(1<<(n+2)));
	
	for(int i=2;i<=(1<<n);i++)
		for(int j=1;j<i;j++)
			cin>>isBetter[i][j];
			
	vector<int> currentGame;
	vector<int> newGame;
	for(int i=1;i<=1<<n;i++)
		currentGame.push_back(i);
	while(currentGame.size()>1){
		newGame.clear();
		for(int i=0;i<currentGame.size();i+=2){
			int lowerIndex = min(currentGame[i],currentGame[i+1]);
			int upperIndex = max(currentGame[i],currentGame[i+1]);
			if(isBetter[upperIndex][lowerIndex])
				newGame.push_back(upperIndex);
			else
				newGame.push_back(lowerIndex);
		}
		currentGame=newGame;
	}
	cout<<currentGame[0]<<endl;
}
