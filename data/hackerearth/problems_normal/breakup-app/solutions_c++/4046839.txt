#include <bits/stdc++.h>
#include <stdio.h>
#include <iostream>
#include <string>	
using namespace std;

int main()
{
	int lines;
	cin>>lines;
	int arr[31];
	int mul;
	string str;
		
	for(int i=0;i<31;i++)
		arr[i]=0; 
		
	cin.ignore();
	for(int a=0;a<lines;a++)
	{
		str.erase();
		getline(cin,str);
		
		if(str[0]=='G')
			mul=2;	
		else
			mul=1;
		int found=-1,temp;
		for(int i=0;i<str.size();i++)
		{
			found=str.find(" ",found+1);
			
			if(str[found+1]>='0' && str[found+1]<='9')
			{	
				
				char c=str[found+1];
				char *p=&c;
				int n1=atoi(p),n2=0,num;
				c=str[found+2];
				
				if(c>='0' && c<='9')
				{
					n2=atoi(p);
					num=n1*10+n2;
				}
				else 
				{
					num=n1;
				}
	
				
				arr[num]=arr[num]+mul;
			}
			
			if(found==-1)
				break;
		} 
		
	}
	int max=0,date=0;
	for(int i=1;i<31;i++)	
	{
		if(arr[i]==max && max>0)
		{
			date=-1;
			break;
		}
		else if(arr[i]>max)
		{
			max=arr[i];
			date=i;
		}
	}
	
	if(date==19 || date==20)
	{
		cout<<"Date\n";
	}
	else
	{
		cout<<"No Date\n";
	}
	

}
