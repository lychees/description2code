#include <stdio.h>
#include <memory.h>
 
using namespace std;
 
const unsigned int md = 4294967143;
unsigned long long n,step,sum,ans;
unsigned int c00,c01,c02,c03,c04,c11,c12,c13,c22;
unsigned int b00,b01,b02,b03,b04,b11,b12,b13,b22;
unsigned int a[5][5], b[5][5], c[5][5];
unsigned int mat[65][5][5], bit[8][260][5][5];
int tt,i,j,k,pw,st,t,q;
 
int main() {
  for (i=0;i<5;i++)
    for (j=0;j<5;j++)
      if (i == j) a[i][j] = 2; else
      if (i-j == 1 || i-j == -1) a[i][j] = 1;
      else a[i][j] = 0;
  for (i=0;i<5;i++)
    for (j=0;j<5;j++) b[i][j] = a[i][j];
  for (pw=0;pw<64;pw++) {
    for (i=0;i<5;i++)
      for (j=0;j<5;j++) mat[pw][i][j] = b[i][j];
    for (i=0;i<5;i++)
      for (j=0;j<5;j++)
        if (i > j) c[i][j] = c[j][i]; else
        if (i+j > 4) c[i][j] = c[4-j][4-i]; else {
          sum = 0;
          for (k=0;k<5;k++) sum = (sum+(unsigned long long)b[i][k]*b[k][j]) % md;
          c[i][j] = sum;
        }
    for (i=0;i<5;i++)
      for (j=0;j<5;j++) b[i][j] = c[i][j];
  }
  for (st=0;st<8;st++) {
    for (t=0;t<(1<<8);t++) {
      for (i=0;i<5;i++)
        for (j=0;j<5;j++) b[i][j] = (i == j);
      for (q=0;q<8;q++)
        if (t & (1 << q)) {
          for (i=0;i<5;i++)
            for (j=0;j<5;j++)
              if (i > j) c[i][j] = c[j][i]; else
              if (i+j > 4) c[i][j] = c[4-j][4-i]; else {
                sum = 0;
                for (k=0;k<5;k++) sum = (sum+(unsigned long long)b[i][k]*mat[(st << 3)+q][k][j]) % md;
                c[i][j] = sum;
              }
          for (i=0;i<5;i++)
            for (j=0;j<5;j++) b[i][j] = c[i][j];
        }
      for (i=0;i<5;i++)
        for (j=0;j<5;j++) bit[st][t][i][j] = b[i][j];
    }
  }
  scanf("%d",&tt);
  while (tt--) {
    scanf("%llu",&n);
    if (n & 1) {
      puts("0");
      continue;
    }
    n = (n >> 1)-1;
    for (i=0;i<5;i++)
      for (j=0;j<5;j++) b[i][j] = (i == j);
    b00 = 1; b01 = 0; b02 = 0; b03 = 0; b04 = 0;
    b11 = 1; b12 = 0; b13 = 0;
    b22 = 1;
    for (st=0;st<8;st++) {
      t = (n >> (st << 3)) & 255;
      c00 =     ((unsigned long long)b00*bit[st][t][0][0]) % md;
      c00 = (c00+(unsigned long long)b01*bit[st][t][1][0]) % md;
      c00 = (c00+(unsigned long long)b02*bit[st][t][2][0]) % md;
      c00 = (c00+(unsigned long long)b03*bit[st][t][3][0]) % md;
      c00 = (c00+(unsigned long long)b04*bit[st][t][4][0]) % md;
      c01 =     ((unsigned long long)b00*bit[st][t][0][1]) % md;
      c01 = (c01+(unsigned long long)b01*bit[st][t][1][1]) % md;
      c01 = (c01+(unsigned long long)b02*bit[st][t][2][1]) % md;
      c01 = (c01+(unsigned long long)b03*bit[st][t][3][1]) % md;
      c01 = (c01+(unsigned long long)b04*bit[st][t][4][1]) % md;
      c02 =     ((unsigned long long)b00*bit[st][t][0][2]) % md;
      c02 = (c02+(unsigned long long)b01*bit[st][t][1][2]) % md;
      c02 = (c02+(unsigned long long)b02*bit[st][t][2][2]) % md;
      c02 = (c02+(unsigned long long)b03*bit[st][t][3][2]) % md;
      c02 = (c02+(unsigned long long)b04*bit[st][t][4][2]) % md;
      c03 =     ((unsigned long long)b00*bit[st][t][0][3]) % md;
      c03 = (c03+(unsigned long long)b01*bit[st][t][1][3]) % md;
      c03 = (c03+(unsigned long long)b02*bit[st][t][2][3]) % md;
      c03 = (c03+(unsigned long long)b03*bit[st][t][3][3]) % md;
      c03 = (c03+(unsigned long long)b04*bit[st][t][4][3]) % md;
      c04 =     ((unsigned long long)b00*bit[st][t][0][4]) % md;
      c04 = (c04+(unsigned long long)b01*bit[st][t][1][4]) % md;
      c04 = (c04+(unsigned long long)b02*bit[st][t][2][4]) % md;
      c04 = (c04+(unsigned long long)b03*bit[st][t][3][4]) % md;
      c04 = (c04+(unsigned long long)b04*bit[st][t][4][4]) % md;
      c11 =     ((unsigned long long)b01*bit[st][t][0][1]) % md;
      c11 = (c11+(unsigned long long)b11*bit[st][t][1][1]) % md;
      c11 = (c11+(unsigned long long)b12*bit[st][t][2][1]) % md;
      c11 = (c11+(unsigned long long)b13*bit[st][t][3][1]) % md;
      c11 = (c11+(unsigned long long)b03*bit[st][t][4][1]) % md;
      c12 =     ((unsigned long long)b01*bit[st][t][0][2]) % md;
      c12 = (c12+(unsigned long long)b11*bit[st][t][1][2]) % md;
      c12 = (c12+(unsigned long long)b12*bit[st][t][2][2]) % md;
      c12 = (c12+(unsigned long long)b13*bit[st][t][3][2]) % md;
      c12 = (c12+(unsigned long long)b03*bit[st][t][4][2]) % md;
      c13 =     ((unsigned long long)b01*bit[st][t][0][3]) % md;
      c13 = (c13+(unsigned long long)b11*bit[st][t][1][3]) % md;
      c13 = (c13+(unsigned long long)b12*bit[st][t][2][3]) % md;
      c13 = (c13+(unsigned long long)b13*bit[st][t][3][3]) % md;
      c13 = (c13+(unsigned long long)b03*bit[st][t][4][3]) % md;
      c22 =     ((unsigned long long)b02*bit[st][t][0][2]) % md;
      c22 = (c22+(unsigned long long)b12*bit[st][t][1][2]) % md;
      c22 = (c22+(unsigned long long)b22*bit[st][t][2][2]) % md;
      c22 = (c22+(unsigned long long)b12*bit[st][t][3][2]) % md;
      c22 = (c22+(unsigned long long)b02*bit[st][t][4][2]) % md;
      b00 = c00;
      b01 = c01;
      b02 = c02;
      b03 = c03;
      b04 = c04;
      b11 = c11;
      b12 = c12;
      b13 = c13;
      b22 = c22;
    }
    b[0][0] = b00;
    b[0][1] = b01;
    b[0][2] = b02;
    b[0][3] = b03;
    b[0][4] = b04;
    b[1][1] = b11;
    b[1][2] = b12;
    b[1][3] = b13;
    b[2][2] = b22;
    for (i=0;i<5;i++)
      for (j=0;j<5;j++)
        if (i > j) b[i][j] = b[j][i]; else
        if (i+j > 4) b[i][j] = b[4-j][4-i];
    ans = 0;
    for (i=0;i<5;i++)
      for (j=0;j<5;j++) ans += (long long)b[i][j]*a[i][j]*2;
    printf("%llu\n",(ans-b[0][0]+md) % md);
  }
  return 0;
}
