# include<bits/stdc++.h>
using namespace std;


void permute(char *a, int l, int r)
{
   int i;
   if (l == r)
     printf("%s\n", a);
   else
   {
       for (i = l; i <= r; i++)
       {
          swap(a[l], a[i]);
          permute(a, l+1, r);
          swap(a[l], a[i]); //backtrack
       }
   }
}

int main()
{
    char inp[10];
    int n,s,e;
    scanf("%s",inp);
    
    scanf("%d",&n);
    while(n--)
    {
		
        scanf("%d %d",&s,&e);
        permute(inp,s,e);
        cout<<endl;
    }

}
