#include<bits/stdc++.h>
using namespace std;
char str[10000000];
long long int t,n,s,i,j,k;
int main()
{
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%s",str);
    	n=strlen(str);
    	s=0;
    	for(i=0;i<n;i++)
    	{
    		s+=(str[i]-48);
    	}
    	printf("%lld\n",s);
    }
    return 0;
}
