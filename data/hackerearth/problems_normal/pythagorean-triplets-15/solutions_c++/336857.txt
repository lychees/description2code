struct trip
{
	int a;
	int b;
	int c;
};

int compute_pytrip_n(int n)
{
	int cnt = 0;
	stack <struct trip> pyth;
	struct trip t;
	t.a = 3;
	t.b = 4;
	t.c = 5;
	pyth.push(t);
	while(pyth.size() != 0)
	{
		int x, y, z;
		x = pyth.top().a;
		y = pyth.top().b;
		z = pyth.top().c;
		cnt += (n / z);
		//cout << x << " " << y << " " << z << " " << cnt << endl;
		pyth.pop();
		t.a = x - 2 * y + 2 * z;
		t.b = 2 * x - y + 2 * z;
		t.c = 2 * x - 2 * y + 3 * z;
		if(t.c <= n)
			pyth.push(t);
		t.a = x + 2 * y + 2 * z;
		t.b = 2 * x + y + 2 * z;
		t.c = 2 * x + 2 * y + 3 * z;
		if(t.c <= n)
			pyth.push(t);
		t.a = -x + 2 * y + 2 * z;
		t.b = -2 * x + y + 2 * z;
		t.c = -2 * x + 2 * y + 3 * z;
		if(t.c <= n)
			pyth.push(t);
		
	}
	return cnt;
}

int main()
{
	int t, n;
	scan(t);
	while(t--)
	{
		scan(n);
		print(compute_pytrip_n(n));
	}
	return 0;
}
