#include <cstdio>
#include <algorithm>
#include <ctime>
#include <cstdio>
#include <random>
#include <iostream>
#include <cstring>
using namespace std;
const int inf=1111111;
int num[inf];
int org[inf];
int main()
{
	int t,n,a;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		memset(num,0,sizeof(num));
		for (int i=0;i<n;++i)
		{
			scanf("%d",&a);
			num[a]++;
			org[i]=a;
		}
		for (int i=0;i<20;++i)
		for(int bit=0;bit<1<<20;++bit)
			if (bit&(1<<i))
				num[bit]+=num[bit^(1<<i)];

		long long ans=0;
		int all=1<<20;
		all--;
		for (int i=0;i<n;++i)
			ans+=num[all^org[i]];
		printf("%lld\n",ans);
	}
}
