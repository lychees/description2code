vector<int> getfa(int m)
{
	vector<int> res;
	int i;
	for(i=1;i*i<m;i++)
	if(m%i==0){res.push_back(i);res.push_back(m/i);}
	if(i*i==m)res.push_back(i);
	return res;
}
