#include<bits/stdc++.h>
using namespace std;
unsigned long long int power(unsigned long int x, int y)
{
    unsigned long long int temp;
    if( y == 0)
        return 1;
    temp = power(x, y/2);
    if (y%2 == 0)
        return temp*temp;
    else
        return x*temp*temp;
}
int main()
{
	unsigned long long int a,b,c;
	unsigned long long n,i;
	cin>>a>>b>>n;
	for(i=3;i<=n;i++)
	{
		c = power(b,2)+a;
		a=b;
		b=c;
	}
	cout<<c<<"\n";
}
