#include <iostream>
using namespace std;
#include<cmath>
long long power(long long a,long long b)
{
	if(b==0)
	return 1;
	else
	{
		long long ret=power(a,b/2);
		ret*=ret;
		ret%=1000000007;
		if(b%2==1)
		ret*=a;
		ret%=1000000007;
		return ret;
	}
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long n;
    	cin>>n;
    	long long ans=0;
    	int m=n;
    	while(m--)
    	{
    		int a;
    		cin>>a;
    		ans=ans|a;
    	}
    	ans=ans*power(2,n-1);
    	ans%=1000000007;
    	cout<<ans<<endl;
    }
}
