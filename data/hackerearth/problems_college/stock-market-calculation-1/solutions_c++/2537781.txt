#include<bits/stdc++.h>
using namespace std;
int main(){


cin.sync_with_stdio(false);


    int t;
    cin >> t;
    while( t-- ){

        int n;
        cin >> n;

          int ar[n+2];
           for( int i = 1 ; i <= n;  ++i )
              cin >> ar[i];

              int ans  =  0;
              int inc = 0;
              int dec = 0;
              ar[ n  + 1 ]  = 0;
              for( int  i  = 1 ; i <= n  ; ++i ){

                  if( ar[i + 1 ]  <  ar[i] ){

                   ans = ans + ( ar[i ] * inc );

                   ans = ans  -  dec;

                    inc = 0 ;
                    dec = 0;
                  }
                  else
                  {
                   inc++;
                   dec += ar[i];
                  }

              }
              cout << ans << endl;



    }


}
