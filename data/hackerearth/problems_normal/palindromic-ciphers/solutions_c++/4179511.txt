#include <iostream>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int l = s.length();
    	int i;
    	int flag = 0;
    	for(i = 0; i <= l/2; i++)
    	{
    		if(s[i] != s[l-1-i])
    		{
    			flag = 1;
    			break;
    		}
    	}
    	if(flag == 0)
    	{
    		cout<<"Palindrome"<<endl;
    	}
    	else
    	{
    		long long int prod = 1;
    		i = 0;
    		while(s[i] != '\0')
    		{
    			prod *=  (s[i]-96);
    			i++;
    		}
    		cout<<prod<<endl;
    	}
    }
    return 0;
}
