
#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        char ar[50][50];
        int n;
        cin>>n;
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                cin>>ar[i][j];
            }
        }
        int h=1,v=1;
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(ar[i][j]!=ar[n-1-i][j])
                    h=0;
                if(ar[i][j]!=ar[i][n-1-j])
                    v=0;
            }
        }
        if(h&&v)
            cout<<"BOTH";
        else if(v)
            cout<<"VERTICAL";
        else if(h)
            cout<<"HORIZONTAL";
        else
            cout<<"NO";
        cout<<endl;
    }
    return 0;

}
