#include <iostream>
#include <cassert>
using namespace std;
#define MOD 1000003
#define MAXN 100000
#define MAXX 1000000000000000000LL
long long ar[MOD+1];
void init(){
	ar[0]=1;
	for(int i=1;i<=MOD;i++)
		ar[i] = (ar[i-1]*(long long)i)%MOD;
}
int main()
{
	init();
	int T;
	long long int N, A;
	init();
	cin >> T;
	assert(T>0 and T<=MAXN);
	while(T--){
		cin >> N >> A;
		assert(N>0 and N<=MAXX);
		assert(A>0 and A<=MAXX);
		A = A % MOD;
		if(N>MOD)
			N = MOD;
		cout << (ar[N]*A)%MOD << endl;
	}
    return 0;
}
