#include <iostream>
#define MAX 26
using namespace std;

int main()
{
    int c[MAX],t,mc,dc,i;
    string ms,ds;
    
    for(i=0;i<MAX;i++){
    	scanf("%d",&c[i]);
    }
    
    mc=0;
    dc=0;
    cin>>t;
    while(t--){
    	int mcc[MAX]={0};
    	int dcc[MAX]={0};
    	
    	cin>>ms;
    	cin>>ds;
    	
    	for(i=0;i<ms.size();i++){
    		mcc[ms[i]-'a']++;
    	}
    	    	
    	for(i=0;i<ds.size();i++){
    		dcc[ds[i]-'a']++;
    	}
    	
    	for(i=0;i<MAX;i++){
    		if(mcc[i] && dcc[i]){
    			int min = (mcc[i]>dcc[i])?dcc[i]:mcc[i];
    			mcc[i]-=min;
    			dcc[i]-=min;
    		}
    		
    		if(mcc[i]>0){
    			mc+=c[i]*mcc[i];
    		}
    		
    		if(dcc[i]>0){
    			dc+=c[i]*dcc[i];
    		}
    	}
    }
    
    if(mc>dc){
    	cout<<"Marut"<<endl;
    }else if(mc<dc){
    	cout<<"Devil"<<endl;
    }else{
    	cout<<"Draw"<<endl;
    }
    
    
    return 0;
}
