#include<bits/stdc++.h>
using namespace std;
long long a[1000005];
long long m;
int n;
int check(long long x)
{
	int i;
	long long sum=0;
	for(i=0;i<n;i++)
	{
		if(a[i]<=x)
		{
			
		}
		else
		{
			sum+=(a[i]-x);
		}
	}
	if(sum>=m)
	return 1;
	else
	return 0;
}
int main()
{
	scanf("%d %lld",&n,&m);
	int i,j;
	for(i=0;i<n;i++)
	scanf("%lld",&a[i]);
	long long low=0;
	long long high=2000000000;
	long long mid;
	long long ans=-1;
	while(low<=high)
	{
		mid=(low+high)/2;
		if(check(mid)==1)
		{
			low=mid+1;
			ans=max(mid,ans);
		}
		else
		{
			high=mid-1;
		}
	}
	printf("%lld\n",ans);
	return 0;
}
