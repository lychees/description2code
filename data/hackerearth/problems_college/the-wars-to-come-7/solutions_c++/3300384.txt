#include<bits/stdc++.h>
using namespace std;
 
int main()
{
  int a[10000],n;
  scanf("%d",&n);
 int i,st;
 
 for(i=0;i<n;i++)
   scanf("%d",&a[i]);
   scanf("%d",&st);
 
 int sum=0,lr=0,count=0;
 while(sum<st)
 {  lr=0;
    for(i=0;i<n;i++)
      {
      if(a[lr]<a[i])
        lr=i;
        //cout<<" lr "<<lr;
      }
      //cout<<"\n"<<lr<<"\n";
     sum=sum+a[lr];
     //cout<<" ar "<<a[lr];
     //cout<<"\n sum "<<sum<<"\n";
      count++;
    a[lr]=0;
}
printf("%d",count);
return 0;
}
