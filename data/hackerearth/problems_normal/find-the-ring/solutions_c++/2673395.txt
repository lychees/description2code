#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int start,swaps;
    	cin>>start>>swaps;
    	if(swaps == 0){
    		cout<<start<<endl;
    		continue;
    	}
    	if(start == 1 ){
    		if(swaps%2){
    			cout<<"0"<<endl;	
    		}else {
    			cout<<"1"<<endl;
    		}
    	}else {
    		if(swaps%2){
    			cout<<"1"<<endl;
    		}else {
    			cout<<"0"<<endl;
    		}
    	}
    	
    }
    return 0;
}
