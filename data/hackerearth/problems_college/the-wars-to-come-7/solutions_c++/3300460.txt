#include <iostream>
using namespace std;

int main()
{
  int n,a[10000],m,i,j,sum=0,count=0,t;
  cin>>n;
  for(i=0;i<n;i++){
  	cin>>a[i];
  }
  cin>>m;
  for(i=0;i<n-1;i++){
  	for(j=0;j<n-i-1;j++){
  		if(a[j]<a[j+1]){
  	t=a[j];
  		a[j]=a[j+1];
  		a[j+1]=t;
  		}
  	}
  }
  for(i=0;i<n;i++){
  	sum+=a[i];
  	count++;
  	if(sum>m){
  		cout<<count<<'\n';
  		break;
  		
  	}
  }
    return 0;
}
