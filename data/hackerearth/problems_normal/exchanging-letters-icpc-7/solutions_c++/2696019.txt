#include <bits/stdc++.h>
#define ll long long
#define MOD 1000000007
using namespace std;
int k,offset,diff;
ll dp[2005][5005];
ll solve(int i,int sum)
{
    if(i==k)
    {
        if(sum==diff)
            return 1;
        else
            return 0;
    }
    if(dp[i][sum+offset]!=-1)
        return dp[i][sum+offset];
    return dp[i][sum+offset]=(solve(i+1,sum-1)+solve(i+1,sum+1))%MOD;
}
int main()
{
    int n,t;
    scanf("%d%d%d",&n,&k,&t);
    int x,sum=0;
    memset(dp,-1,sizeof(dp));
    offset=2005;
    for(int i=0;i<k;i++)
    {
        scanf("%d",&x);
        sum+=x;
    }
    diff=t-sum;
    if(abs(sum-t)>k)
        printf("0");
    else
        printf("%lld",solve(0,0));
    return 0;
}
