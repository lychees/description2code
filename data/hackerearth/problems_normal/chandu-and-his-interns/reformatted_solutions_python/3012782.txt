def check_prime(num):
    i = 2
    divisors = 2
    while i*i < num:
        if num % i == 0:
            divisors += 2
            if divisors >= 4:
                return "YES"
        i += 1
    if i*i == num:
        divisors += 1
    if divisors < 4:
        return "NO"
    return "YES"


if __name__ == '__main__':
    inputs = int(raw_input())
    for _ in range(0, inputs):
        print check_prime(int(raw_input()))
