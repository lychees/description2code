#include <iostream>
using namespace std;

int main()
{
	int tc,len1,len2;
	string str1,str2;
	cin>>tc;
	while(tc--)
	{
		cin>>str1>>str2;
		len1=str1.length();
		len2=str2.length();
		int arr[26]={0};
		for(int i=0;i<len1;i++)
			arr[str1[i]-'a']+=1;
		int flag=0;
		for(int i=0;i<len2;i++)
		{
			if(arr[str2[i]-'a']>0)
			{
				flag=1;
				break;
			}
		}
		if(flag==1)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
		
	}
    return 0;
}
