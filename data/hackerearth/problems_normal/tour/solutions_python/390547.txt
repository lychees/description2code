a = {}
pos=0
ans=0
n = int(raw_input())
for i in range(n):
	inp = raw_input()
	a[inp] = i
cost = [[0 for i in range(n)]for i in range(n)]
for i in range(n):
	cost[i] = map(int,raw_input().split(' '))
q = int(raw_input())
for i in range(q):
	query = raw_input()
	x = a[query]
	ans = ans + cost[pos][x]
	pos = x
print ans
