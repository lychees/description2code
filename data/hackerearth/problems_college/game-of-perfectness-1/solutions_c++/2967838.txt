#include<algorithm>
#include<vector>

#include <iostream>
using namespace std;

int main()
{
	int j=1,tc;
	cin>>tc;
	while(tc--)
	{ int x;
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>x;
	}
		vector<pair<int,int>>arr;
		for(int i=0;i<n;i++)
		{
			cin>>x;
			arr.push_back(make_pair(-x,i));
		}
		
		sort(arr.begin(),arr.end());
		cout<<"Case #"<<j++<<":";
		for(int i=0;i<n;i++)
		cout<<" "<<arr[i].second;
		
		cout<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
