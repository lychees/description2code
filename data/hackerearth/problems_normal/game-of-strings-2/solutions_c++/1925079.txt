bool issubstring(char *str,int k,int n,int len)
{
	int i;
	for(i=k;i>0 && n>k && len>0;i--,len--)
	{
		if(str[i]!=str[n-1])
		{
			return false;
		}
		n--;
	}
	return true;
}
