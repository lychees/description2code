#include<bits/stdc++.h>
using namespace std;
typedef pair<int, int> pp;
typedef long long ll;
int a[2000004];

int main()
{
	int t;
	scanf("%d", &t);
	while (t--)
	{
		int n;
		scanf("%d", &n);
		for (int i = 0; i<2 * n; i++){
			scanf("%d", &a[i]);
		}
		sort(a, a + 2 * n);
		int ans = INT_MIN;
		int i = 0, j = 2 * n - 1;
		while (i<j)
		{
			if (a[i] + a[j]>ans)
				ans = a[i] + a[j];
			i++;
			j--;
		}
		printf("%d\n", ans);
	}
	return 0;
}
