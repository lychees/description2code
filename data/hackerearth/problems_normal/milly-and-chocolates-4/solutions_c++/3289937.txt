#include<iostream>
#include<stdio.h>

using namespace std;
main()
{
	int t,j;
	
	cin>>t;
	for(j=0;j<t;j++)
	{
		double s=0;
		int n,i,a[500001];
		cin>>n;
		for(i=0;i<n;i++)
		{
		cin>>a[i];
	}
	for(i=0;i<n;i++)
	{	s=s+a[i];	
		
	}
		s=s/n;
		printf("%.8lf",s);
		cout<<endl;
	}
	return 0;
}
