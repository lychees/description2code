#include <cstdio>
#include <algorithm> 
using namespace std;

int n , b[100001][4] ;
long long ans=0 , a[100001] ;

int cnt_bits(int x)
{
   int ans=0 ;
   for(int i=0 ; i<32 ; i++) if( ((x>>i)&1)!=0 ) ans++ ; 
 return ans ;
}

int main()
{
   scanf("%d",&n) ;  
   for(int i=0 ; i<n && scanf("%lld",&a[i]) ; i++) 
   {
      long long tmp=a[i] ; 
      while(tmp)
      {
         int res=tmp%10 ; 
         if(res==2) b[i][0]=1 ;  if(res==3) b[i][1]=1 ;
         if(res==5) b[i][2]=1 ;  if(res==7) b[i][3]=1 ;
         tmp/=10 ;  
      }
   }
    
   for(int i=1 ; i<16 ; i++)
   {
      int sign=1 , cnt=0 ;
      if(cnt_bits(i)%2==0) sign=-1 ; 
      
      for(int j=0 ; j<n ; j++) 
      { 
        int sig=0 ;
        for(int k=0 ; k<4 ; k++) if( ((i>>k)&1)!=0 && !b[j][k]) { sig=1 ; break ; }  
        if(!sig) cnt++ ; 
      }  

      if(cnt>=3) ans += 1LL*sign*(cnt-2)*cnt*(cnt-1)/6 ;
   }

   printf("%lld\n",ans) ;    

 return 0;
}
