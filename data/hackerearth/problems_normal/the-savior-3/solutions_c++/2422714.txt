#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t,n;
   cin>>t;
   while(t--)
   { 
   	int x;
   	cin>>n;
   	 int a[n];
   	for(int i=0;i<n;i++)
   	cin>>a[i];
   	int j,k;
   	int count=0;
   	for(j=0;j<n-1;j++)
   	{
   		for( k=j;k<n-1;k++)
   		{
   		if(a[j]!=a[k+1])
   		{
   	      x=a[j]+a[k+1];
   		if(x%2==0)
   		count++;
   		}}
   	}
   	cout<<count<<endl;
   }
   
   return 0;
}
