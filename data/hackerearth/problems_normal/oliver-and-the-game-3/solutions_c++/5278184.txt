LL MOD = 1000000007;

LL bigmod(LL a, LL b){
    LL x = 1, y = a%MOD;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>MOD) x%=MOD;
        }
        y = (y*y);
        if(y>MOD) y%=MOD;
        b /= 2;
    }
    return x;
}

LL MODINVERSE(LL a){
    return bigmod(a, MOD-2);
}

LL ncrdp[900][1000];
LL NCR(int n, int r)
{
    if(r==1) return n;
    else if(n==r) return 1;
    else
    {
        if(ncrdp[n][r]!=-1) return ncrdp[n][r];
        else
        {
            ncrdp[n][r]=NCR(n-1,r) + NCR(n-1,r-1);
            return ncrdp[n][r];
        }
    }
}

const int MAXN = 105;
int status[(MAXN/32)+10];
vector<int>primelist;

bool check(int n, int pos) { return (bool)(n & (1<<pos)); }
int SET(int n, int pos){ return n=n|(1<<pos);}

void sieve()
{
    int sqrtN=int (sqrt(MAXN));
    status[1>>5]=SET(status[1>>5],1&31);
    for(int j=4; j<=MAXN; j=j+2)
        status[j>>5]=SET(status[j>>5],j&31);
    for(int i=3; i<=sqrtN; i=i+2)
    {
        if(check(status[i>>5],i&31)==0)
        {
            for(int j=i*i; j<=MAXN; j= j + (i<<1))
            {
                status[j>>5]=SET(status[j>>5],j&31);
            }
        }
    }
    primelist.push_back(2);
    for(int i=3; i<=MAXN; i=i+2)
    {
        if(check(status[i>>5],i&31)==0) {
            primelist.push_back(i);
        }
    }
}

vector<int>graph[100005];
bool visited[100005];
int in[100005], out[100005], ti;

void DFS(int u)
{
    in[u] = ++ti;
    visited[u] = true;
    for(int i=0; i<graph[u].size(); i++)
    {
        if(!visited[graph[u][i]]){
            DFS(graph[u][i]);
        }
    }
    out[u] = ++ti;
}

bool flag(int u, int v)
{
    if(in[u]>in[v] && out[u]<out[v]) return true;
    return false;
}

int main()
{
    int n;
    scanf("%d", &n);
    for(int i=1; i<n; i++){
        int u, v;
        scanf("%d %d", &u, &v);
        graph[u].push_back(v);
        graph[v].push_back(u);
    }
    DFS(1);
    int q;
    scanf("%d", &q);
    while(q--)
    {
        int t, u, v;
        scanf("%d %d %d", &t, &u, &v);
        if((!t && flag(v,u)) ||(t && flag(u,v))){
            printf("YES\n");
        }
        else{
            printf("NO\n");
        }
    }
}
