#include <iostream>
using namespace std;

#define M 1000000007

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	unsigned long long int n,k,l;
    	scanf("%llu",&n);
    	n = n%M;
    	k = 1 + 2*(n-1);
    	l = (n%M * k%M)%M;
    	printf("%llu\n",l);
    }
    return 0;
}
