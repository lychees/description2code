#include<bits/stdc++.h>
using namespace std;
int dp[1000050];
int main()
{
	memset(dp,0,sizeof(dp));
	dp[0]=0;
	dp[1]=-1;
	dp[2]=1;
	dp[3]=1;
	dp[4]=2;
	dp[5]=1;
	dp[6]=2;
	dp[7]=1;
	dp[8]=2;
	dp[9]=2;
	dp[10]=2;
	for(int i=11;i<=1000050;i++)
	dp[i]=1+min(dp[i-2],min(dp[i-3],min(dp[i-5],dp[i-7])));
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		printf("%d\n",dp[n]);
	}
	
	return 0;
}
