#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,inp;
    cin>>n;
    vector<int>v;
    vector<int>::iterator it,it1;
    for(int i=0;i<n;i++)
    {
    	cin>>inp;
    	v.push_back(inp);
    }
    int maxi=0;
    long long int sum=0;
    while(n>2)
    {
    	maxi=0;
    	for(it=v.begin()+1;it!=v.end()-1;it++)
    	{
    		if(*(it-1)*(*(it+1))>maxi)
    		{
    			maxi=*(it-1)*(*(it+1));
    			it1=it;
    		}
    	}
    	v.erase(it1);
    	sum+=maxi;
    	n--;
    }
    cout<<sum<<endl;
}
