#include <iostream>
#include <vector>
#include <cstdio>

using namespace std;

int main()
{
    int T,i,m=0,carry=0,N,Ncpy,j,num,lcount=0;
    vector<int> solution;
    fscanf(stdin,"%d",&T);
    //cin.ignore();
    for(i=0;i<T;i++)
    {
        fscanf(stdin,"%d",&N);
        Ncpy=N;
        while(Ncpy>0)
        {
            solution.push_back(Ncpy%10);
            //cout<<Ncpy%10<<endl;
            m++;
            Ncpy=Ncpy/10;
        }
        while(N>1)
        {
            for(j=0;j<m;j++)
            {
                num=solution[j]*(N-1)+carry;
                solution[j]=num%10;
                //cout<<num<<endl;
                carry=num/10;
            }
            while(carry>0)
            {
                solution.push_back(carry%10);
                //cout<<carry%10<<endl;
                carry=carry/10;
                m++;
            }
            N--;
        }
        for(j=solution.size()-1;j>=0;j--)
        {
            if(solution[j]==4 || solution[j]==7)
            {
                lcount++;
            }
        }
        cout<<lcount<<endl;
        solution.clear();
        lcount=carry=0;
        m=0;
    }
    return 0;
}
