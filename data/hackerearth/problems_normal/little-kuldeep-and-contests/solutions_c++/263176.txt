#include<stdio.h>
#include<iostream>
#include<vector>
using namespace std;
int main()
{
	int n;
	int i;
	cin>>n;
	i = n;
	vector<string> times; 
	vector<int> startTimes;
	vector<int> endTimes;
	int day[1440];
	int j=0;
	for(j=0;j<1440;j++)
		day[j]=0;
	while(i--)
	{
		string periods;
		cin>>periods;
		times.push_back(periods);
	}
	vector<string>::iterator it;
	for( it = times.begin();it !=times.end();it++)
	{
		//cout<<*it<<"\n";
		int pos = (*it).find("-");
		//cout<<(*it).substr(0,pos)<<"\n";
		string startAt = (*it).substr(0,pos);
		string endAt  = (*it).substr(pos+1);
		//cout<<"start = "<<startAt<<" endAt "<<endAt<<"\n";
		int colon_inStart  = startAt.find(":");
		int colon_inEnd = endAt.find(":");
		int startHour = atoi((startAt.substr(0,colon_inStart)).c_str());
		int startMin = atoi(startAt.substr(colon_inStart+1).c_str());
		int endHour = atoi(endAt.substr(0,colon_inEnd).c_str());
		int endMin = atoi(endAt.substr(colon_inEnd+1).c_str());
		int startTime = startHour*60+startMin;
		int endTime = endHour*60+endMin;
		for(int iter = startTime; iter < endTime;iter++)
		{
			if(day[iter] ==0)
				day[iter] =1;
			else
			{
				cout<<"Will need a moderator!\n";
				return 0;
			}
		}	

	}
	cout<<"Who needs a moderator?\n";
 	return 0;

}
