#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    int m,k;
    char c;
    vector < int >v[26];
    cin>>s>>m;
    int len=s.length();
    for(int i=0;i<len;i++)
       v[s[i]-'a'].push_back(i);
    for(int i=0;i<m;i++)
    {
    	cin>>k>>c;
    	s[v[c-'a'][k-1]]='A';
        v[c-'a'].erase(v[c-'a'].begin()+(k-1));
    }
    for(int i=0;i<len;i++)
       if(s[i]!='A') cout<<s[i];
       cout<<endl;
    return 0;
}
