#include <iostream>
#include <map>
#include <vector>
using namespace std;
map<long long int,long long int> mp1;
map<long long int,long long int>::iterator it;
vector<vector<long long int> > v1;
int main()
{
    long long int n,p1,p2,result=0,m,p3;
    cin>>n;
    if(n==1)
    {
    	cout<<0;
    	return 0;
    }
    m=n;
    v1.resize(n+1);
    for(p1=0;p1<=n;p1++)
    {
    	v1[p1].resize(n+1,0);
    	v1[p1][0]=1;
    }
    for(p1=1;p1<=n;p1++)
    	for(p2=1;p2<=p1;p2++)
    		v1[p1][p2]=v1[p1-1][p2-1]+v1[p1-1][p2];
    
    mp1.clear();
    
    for(p3=2;p3<=m;p3++)
    {
    //	while(p2)
    //	{
    		n=p3;
    		for(p1=2;p1<=n;p1++)
    			while(n%p1==0)
    			{
    				mp1[p1]++;
    				n/=p1;
    			}
    	//	p2--;
    //	}
    
    	p2=mp1.size();
   	//	cout<<p2<<"\n";
    	for(p1=0;p1<=p2/2;p1++)
    	{
    		result+=v1[p2][p1];
    		if(p2==p1*2)
    		result-=v1[p2][p1]/2;
    	}
	}
    cout<<result;
    return 0;
}
