#include<bits/stdc++.h>
using namespace std;

int names[27][27][27][27][27];

int main()
{
	int count=0,i=0,a,b,c,d,e,j,k,l,m;
	int N;
	char name[10];

	cin >> N;

	while(N--)
	{
        i=0;
        a=0,b=0,c=0,d=0,e=0;
        
        scanf("%s",name);
        int len = strlen(name);
       
		while(i<len)
		{
			if(i==0)
                a = name[i]-'a'+1;
            if(i==1)
            	b = name[i]-'a'+1;
		    if(i==2)
                c = name[i]-'a'+1;
            if(i==3)
            	d = name[i]-'a'+1;
            if(i==4)
                e = name[i]-'a'+1;
	    
	        i++;
		}
 
       // cout << a <<b<<c<<d<<e<<endl;
        if(names[a][b][c][d][e] == 0)
		{
		  names[a][b][c][d][e] =1;
		  count++;
		}  
	}

	printf("%d\n",count);

	for(i=0; i<27; i++)
		for(j=0;j<27;j++)
			for(k=0;k<27;k++)
				for(l=0;l<27;l++)
					for(m=0;m<27;m++)
					{
						if(names[i][j][k][l][m])
						{
							printf("%c",'a'+i-1);

							if(j)
								printf("%c",'a'+j-1);
							if(k)
								printf("%c",'a'+k-1);
							if(l)
								printf("%c",'a'+l-1);
							if(m)
								printf("%c",'a'+m-1);

							cout << endl;
					
						}
					}

return 0;
}
