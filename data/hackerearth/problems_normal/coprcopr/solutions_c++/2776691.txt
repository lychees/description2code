#include<bits/stdc++.h>
using namespace std;
 int seive[10000001];
void seive1() {

 for( int i=1; i<10000001; i++) { seive[i] = i;}

 for( int i=2; i<10000001; i++) {
 	if( seive[i] == i ){
 		for(  int j=i; j < 10000001; j+=i){
 			seive[j] -= seive[j] / i;
		 }
	 }
 }
}

int main()
{
	int t;
	cin >> t;
	vector<long long int> v(10000001);
	v[0] = 0 ;
	v[1] = 1 ; 
	seive1();
	for (  int i=2; i<10000001;i++ ) v[i] = v[i-1] + seive[i]; 
	while ( t -- ) {
		int n; cin >> n;
		cout << v[n] << endl;
	}
	return 0;
}
