#include<iostream>
#include<algorithm>
#include<string.h>
using namespace std;

int pos(char a)
{
	if(a=='.'||a==','||a=='?'||a=='!'||a=='1')
	  return 1;
	  else if(a=='a'||a=='b'||a=='c'||a=='2')
	  return 2;
	  else if(a=='d'||a=='e'||a=='f'||a=='3')
	  return 3;
	  else if(a=='g'||a=='h'||a=='i'||a=='4')
	  return 4;
	  else if(a=='j'||a=='k'||a=='l'||a=='5')
	  return 5;
	  else if(a=='m'||a=='n'||a=='o'||a=='6')
	  return 6;
	  else if(a=='p'||a=='q'||a=='r'||a=='s'||a=='7')
	  return 7;
	  else if(a=='t'||a=='u'||a=='v'||a=='8')
	  return 8;
	  else if(a=='w'||a=='x'||a=='y'||a=='z'||a=='9')
	  return 9;
	  else if(a=='_'||a=='0')
	  return 0;
}

int atpos(char a)
{

	if(a=='.'||a=='a'||a=='d'||a=='g'||a=='j'||a=='m'||a=='p'||a=='t'||a=='w'||a=='_') 
	  return 1;
	  else if(a==','||a=='b'||a=='e'||a=='h'||a=='k'||a=='n'||a=='q'||a=='u'||a=='x'||a=='0')
	  return 2;
	  else if(a=='?'||a=='c'||a=='f'||a=='i'||a=='l'||a=='o'||a=='r'||a=='v'||a=='y')
	  return 3;
	  else if(a=='!'||a=='2'||a=='3'||a=='4'||a=='5'||a=='6'||a=='s'||a=='8'||a=='z')
	  return 4;
	  else if(a=='1'||a=='7'||a=='9')
	  return 5;
	  
}

int main()
{
	int i,j,k;
	int t;
	cin>>t;
	char s[1001];
	for(i=0;i<t;i++)
	{
		cin>>s;
		int ans=0;
		int cur=1,x,y;
		for(j=0;j<strlen(s);j++)
		{
			x=pos(s[j]);
			if(x==cur)
			  ans=ans;
			  else
			  {
			  	cur=x;
			  	ans=ans+1;
			  }
			  y=atpos(s[j]);
			  ans=ans+y;
		}
		cout<<ans<<"\n";
	}
	return 0;
}
