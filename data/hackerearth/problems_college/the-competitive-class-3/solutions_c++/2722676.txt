#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int n,c,i,j;
    cin>>n;
    vector <int> m(n);
    for(i=0;i<n;i++)
    	cin>>m[i];
    for(i=0;i<n;i++)
    {
    	c=0;
    	for(j=0;j<n;j++)
    	{
    		if(m[j] > m[i])
    			c++;
    	}
    	cout<<c+1<<" ";
    }
    return 0;
}
