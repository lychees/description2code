using namespace std;
 
ll arr[1000005];
ll res[1000005];
void sieve()
{
    arr[0]=arr[1]=1;
    ll i,j;
    for(i=2;i<=1000000;i++)
    {
        if(arr[i]==0)
        {
            for(j=2*i;j<=1000000;j+=i)
                arr[j]=1;
        }
    }
    arr[4]=0;
}
 
void pre()
{
    for(ll i=1;i<=1000000;i++)
        arr[i]=arr[i]+arr[i-1];
}
 
int main()
{
    sieve();
    pre();
    ll tc,l,r;
    //for(ll i=0;i<20;i++)
        //printf("%lld ",res[i]);
    sll(tc);
    while(tc--)
    {
        scanf("%lld %lld",&l,&r);
        pll(arr[r]-arr[l-1]);
    }
    return 0;
}
