#include "bits/stdc++.h"
using namespace std;
long long a , b;
long long ans = 0;
int g[10] = {0 , 1 , 2 , 4 , 7 , 9 , 13 , 15 , 18 , 20};
int main(){
    cin >> a >> b;
    if(b < 10){
        cout << g[b] - g[a - 1];
        return 0;
    }
    if(a < 10){
        ans += g[9] - g[a - 1];
        a = 10;
    }
    long long even = (b / 2) - ((a - 1) / 2);
    long long odd = b - a + 1 - even;
    ans += even * 3 + odd * 2;
    long long cnt = 0;
    //multiples of 6 not multiples of 4
    cnt += (b / 6) - ((a - 1) / 6);
    cnt -= (b / 12) - ((a - 1) / 12);
    ans += cnt;
    cnt = 0;
    cnt = (b / 420) - ((a - 1) / 420);
    cnt -= (b / 840) - ((a - 1) / 840);
    ans += cnt;
    cnt = 0;
    cnt += (b / (420 * 858)) - ((a - 1) / (420 * 858));
    cnt -= (b / (420 * 858 * 2)) - ((a - 1) / (420 * 858 * 2));
    ans += cnt;
    cnt = 0;
    long long x = 420LL * 858LL * 200360130LL;
    cnt += (b / x) - ((a - 1) / x);
    cnt -= (b / (x + x)) - ((a - 1) / (x + x));
    ans += cnt;
    cout << ans;
}
