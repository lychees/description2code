#include<iostream>
#include<string>
#include<cctype>
#include<cstdio>

using namespace std;

bool isnotnumberandblank(char a){
	int b=a;
	if((b<48 || b>57) && a!=' ')
	return true;
	else
	return false;
}

int main(){
	int f,n;
	string s,c="";
	cin>>n;
	fflush(stdin);
	n=n+1;
	while(n--){
		f=0;
	getline(cin,s);
	int l=s.length();
	for(int i=0;i<l;i++){
		if(isnotnumberandblank(s[i]))
		f=0;
		if(f==1 && s[i]!='0' && s[i]!=' ')
		f=2;
		if(f==2 && s[i]!=' ' )
		c=c+s[i];
		if(s[i]=='$'){
		f=1;
		c=c+s[i];
	  }
	}
	if(c=="$")
	c=c+'0';
	cout<<c<<endl;
	c="";
	s="";
	fflush(stdin);
}
return 0;
}
