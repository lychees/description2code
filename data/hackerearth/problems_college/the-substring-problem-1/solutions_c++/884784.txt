#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,n;
	string s;
	cin >> t;
	while(t--)
	{
		map<char,int> mp;
		cin >> n;
		while(n--)
		{
			cin >> s;
			sort(s.begin(),s.end());
			s.resize(unique(s.begin(),s.end())-s.begin());
			for(int i=0;i<s.size();++i)
				++mp[s[i]];
		}
		char mx;
		int maxi = INT_MIN;
		for(char ch='a';ch<='z';++ch)
			if(mp[ch]>maxi)
			{
				mx = ch;
				maxi = mp[ch];
			}
		cout << mx << endl;
	}
	return 0;
}
