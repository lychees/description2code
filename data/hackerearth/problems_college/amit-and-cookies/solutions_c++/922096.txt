#include<iostream>
using namespace std;

int main()
{   long long int swag,n,k;
    cin>>n>>k;
    long long int data[n];

    for(int i=0;i<n;i++)
    {
        cin>>data[i];
    }

      for(int j=0;j<n;j++)
      {
          for(int k=j;k<n;k++)
          {
              if(data[j]>data[k])
              {     swag=data[j];
                    data[j]=data[k];
                    data[k]=swag;
                   }
          }
        }
        int g=0;
        int c=0;
    for(int l=0;l<n;l++)
    {   g=g+data[l];
        if(g>k){break;}
        c++;
        //cout<<data[l]<<endl;
    }
    cout<<c<<endl;
    return 0;
}
