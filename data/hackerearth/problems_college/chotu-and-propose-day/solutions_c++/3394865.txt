#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str[2000];
		string str1 = "aa",str2 = "bb",str3 = "ab" ,str4 = "ba";
		int x1,x2,x3,x4;
		cin>>x1>>x2>>x3>>x4;
		int x = min(x3,x4);
		x3 = x;
		x4 = x;
		int k=0;
		if(x1 %2 !=0 && x2 %2 != 0)
		{
			x2 = x2 - 1;
		}
		if(x1%2 == 0 && x2%2 != 0)
		{
			for(int i=0;i<x1/2;i++)
			{
				str[k] = str1;
				k++;
			} 
			for(int i=0;i<x3;i++)
			{
				str[k] = str3;
				k++;
			}
			for(int i=0;i<x2/2+1;i++)
			{
				str[k] = str2;
				k++;
			}
			for(int i=0;i<k;i++)
			{
				cout<<str[i];
			}
			for(int i=k-2;i>=0;i--)
			{
				reverse(str[i].begin(),str[i].end());
				cout<<str[i];
			}
		}
		else if(x1%2==0 && x2%2==0)
		{
			for(int i=0;i<x1/2;i++)
			{
				str[k] = str1;
				k++;
			} 
			for(int i=0;i<x3;i++)
			{
				str[k] = str3;
				k++;
			}
			for(int i=0;i<x2/2;i++)
			{
				str[k] = str2;
				k++;
			}
			for(int i=0;i<k;i++)
			{
				cout<<str[i];
			}
			for(int i=k-1;i>=0;i--)
			{
				reverse(str[i].begin(),str[i].end());
				cout<<str[i];
			}			
		}
		else if(x1%2!=0 && x2%2==0)
		{
			for(int i=0;i<x1/2;i++)
			{
				str[k] = str1;
				k++;
			} 
			for(int i=0;i<x3;i++)
			{
				str[k] = str3;
				k++;
			}
			for(int i=0;i<x2/2;i++)
			{
				str[k] = str2;
				k++;
			}
			str[k] = str1;
			k++;
			for(int i=0;i<k;i++)
			{
				cout<<str[i];
			}
			for(int i=k-2;i>=0;i--)
			{
				reverse(str[i].begin(),str[i].end());
				cout<<str[i];
			}			
		}
		cout<<"\n";
	}
}
