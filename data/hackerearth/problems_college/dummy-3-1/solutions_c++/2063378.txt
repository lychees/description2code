#include<iostream>
using namespace std;
typedef long long ll;
int main()
{
    ll t,n,i;
    cin>>t;
    while(t--)
    {
        cin>>n;
        ll a[n+1],b[n+1];
        for(i=1;i<n+1;i++)
        {
            cin>>a[i];
            b[a[i]]=i;
        }
        for(i=1;i<n+1;i++)
        {
            if(a[i]!=b[i])
                break;
        }
        if(i==n+1)
            cout<<"inverse\n";
        else
            cout<<"not inverse\n";
    }
    return 0;
}
