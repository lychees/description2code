#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
#define pb push_back
using namespace std;

vector<int> a;
vector<int> flag;
int marks[1000][20];
int n,k;

int mina(int x)
{
int i,mi = 1000;
for(i=0;i<n;i++) if(flag[i] == 0) mi = min(mi,marks[i][x]);
return mi;
}

void reduce(int mi,int x)
{
int i;
for(i=0;i<n;i++)
{
if(flag[i] == 0) 
{
if(marks[i][x] != mi) flag[i] = 1;
}
}

}



int main()
{
int i,j,x,q,y,z,mi;
cin>>n>>k;
for(i=0;i<n;i++) for(j=0;j<k;j++) cin>>marks[i][j];
for(i=0;i<n;i++) a.pb(i),flag.pb(0);


cin>>q;
while(q--)
{
cin>>x;

while(x--)
{
cin>>y;
y--;
mi = mina(y);
reduce(mi,y);
}

for(i=0;i<n;i++) if(flag[i] == 0) {cout<<i+1<<"\n";break;}
for(i=0;i<n;i++) flag[i] = 0;
}

return 0;
}
