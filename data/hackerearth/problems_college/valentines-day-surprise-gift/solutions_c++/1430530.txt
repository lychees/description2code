#include <iostream>
#define max(a,b) (a>b?a:b)
#define MOD 1000000007
using namespace std;

int R(int n) {
	if (n <= 2) return 1;
	else if (n <= 4) return 3;
	else if (n <= 6) return 15;
	else if (n <= 10) return 105;
	else if (n <= 12) return 24;
	else if (n <= 16) return 48;
	else if (n <= 20) return 96;
	else if (n <= 24) return 2880;
	else if (n <= 28) return 5760;
	else return 11520;
}

long long gcd(long long x, long long y) {
	if (x == 0) return y;
	if (y == 0) return x;
	if (x > y) {
		return gcd(y, x%y);
	} else {
		return gcd(x, y%x);
	}
}

long long goodPow(long long x, long long y) {
 	long long i, ans = 1;
 	for (i = 0; i < y; ++i) {
 		ans = (ans*x)%MOD;
 	}
 	return ans;
}

long long maxOver(long long x, long long rn) {
	long long i, ans = 1, maxNum = -1, calc = goodPow(x, rn);
	for (i = rn; i <= rn*5; i += rn) {
		ans = (ans*calc)%MOD;
		if (gcd(i, x) == rn) {
			//cout << "gcd of " << i << " and " << x << " = " << rn;
			if (ans > maxNum) {
				maxNum = ans;
				//cout << " CHOSEN" << maxNum;
			}
			//cout << endl;
		}
	}
	return maxNum;
}

int main() {
	int T;
	long long ans = 0, n, i, rn, mo;
	
	cin >> T;
	while ((T--) > 0) {
		cin >> n;
		rn = R(n);
		ans = 0;
		for (i = rn; i <= rn*5; i += rn) {
			mo = maxOver(i, rn);
			if (mo > ans) {
				ans = mo;
				//cout << "ans = " << ans << endl;
			}
		}
		cout << ans << endl;
	}
	return 0;
}
