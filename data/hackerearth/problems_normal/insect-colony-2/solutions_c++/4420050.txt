#include <iostream>
using namespace std;

int main()
{
	int t,a,i,n;
    cin >> t;
    while(t--){
    	cin >> n;
    	long sum = 0;
    	for(i = 0;i < n;i++){
    		cin >> a;
    		sum += a;
    	}
    	if(sum %2 == 1){
    		cout << "No\n";
    	}
    	else{
    		cout << "Yes\n";
    	}
    }
    return 0;
}
