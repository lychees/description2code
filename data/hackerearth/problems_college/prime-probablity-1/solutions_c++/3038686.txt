#include <iostream>
#include <stdio.h>
#include <stdlib.h>
using namespace std;

int main()
{
	unsigned long long a[1000001]={0,1,0,0},i,j=2,l,r,t;
    float ans;
 
    for(i=2;i<1000001;i++)
     for(j=2;j*i<1000001;j++)
      a[i*j]=1;
    for(i=1;i<1000001;i++)
   {
      if(a[i]==1)
       a[i]=a[i-1];
      else
        a[i]=a[i-1]+1;
   }
   cin>>t;
   while(t--)
   {
   	cin>>l>>r;
    ans=r-l+1;
    printf("%.6lf\n",(a[r]-a[l-1])/ans);
   }
    return 0;
}
