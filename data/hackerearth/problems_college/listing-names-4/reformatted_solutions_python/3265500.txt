# print 'Hello World!'
from collections import Counter
n = int(raw_input())
l = []
for i in range(n):
    l.append(raw_input())

s = set(l)
s = sorted(s)
counts = Counter(l)
for letter in s:
    print('%s %d' % (letter, counts[letter]))
