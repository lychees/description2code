#include <iostream>
#include <algorithm>
#include <queue>
#include <vector>

using namespace std;

int main()
{
    long long int t, n, maxR = 0;
    cin >> t;
    while(t--)
    {
    	cin >> n;
    	int arr, dep, x = 0;
    	
    	vector<pair<int, char> > m;
    
    	
    	for(int i = 0; i < n; i++)
    	{
    		cin >> arr;
    		m.push_back(make_pair(arr, 'y'));
    	}
    	
    	for(int i = 0; i < n; i++)
    	{
    		cin >> dep;
    		dep = dep + m[i].first;
    		m.push_back(make_pair(dep, 'x'));
    	}
    	sort(m.begin(), m.end());
    	maxR = 0;
    	int s = m.size();
    	
    	for(int i = 0; i < s; i++)
    	{
    		if(m[i].second == 'y')
    		{
    			x++;
    			if(maxR < x)
    			maxR = x;
    		}
    		else if(m[i].second == 'x')
    			x--;
    	}
    	cout<<maxR<<endl;
    }
    return 0;
}
