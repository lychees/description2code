ll power(ll a,ll b)
{
    ll ret;
    if(b==0)
        return 1;
    if(b==1)
        return (a%MOD);
    ret=(power(a,(b/2)))%MOD;
    ret=((ret%MOD)*(ret%MOD))%MOD;
    if(b&1)
        ret=((ret%MOD)*(a%MOD))%MOD;
    return (ret%MOD);
}
