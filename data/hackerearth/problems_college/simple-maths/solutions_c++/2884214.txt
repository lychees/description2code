#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main()
{
    int t;
    cin >> t;
    while (t--) {
    	ll n,  a, b;
    	cin >> n >> a >> b;
    	if (a == 0 || b == 0) {
    		cout <<"-1"<<endl;
    		continue;
    	}
		cout <<n/a + n/b - n/ ((a*b)/__gcd(a,b)) << endl;    	
    }
    return 0;
}
