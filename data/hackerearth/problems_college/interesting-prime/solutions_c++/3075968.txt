#include <iostream>
#include <cstdlib>

using namespace std;
int main()
{
	
	long long T;
	cin>>T;
	while(T--)
	{
		long long n;
		cin>>n;
		long long i,k=0;
		for(i=1;i*i<=n;i++)
		{
			if(n%i==0)
			{
				if(i==n/i)
					k++;
				else
					k+=2;
			}
		}
	
		cout<<k-1<<endl;
	}
	return 0;
}
