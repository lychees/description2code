#include<bits/stdc++.h>
void f(int b[][210],int i,int x,int n)
{
    if(x>n || x<1)
    {
        return;
    }
    else
    {
        b[i][x]=1;
        if(!b[i][2*x])
        {
            f(b,i,2*x,n);
        }
        if(!b[i][x/2])
        {
            f(b,i,x/2,n);
        }
    }
}
int main()
{
    int a[110][110],b[110][210]={0},n,i,j,k;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        f(b,i,i,n);
    }
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=n;j++)
        {
            scanf("%d",&a[i][j]);
        }
    }
    int c[110][110]={0};
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=n;j++)
        {
            int max=-1;
            for(k=1;k<=n;k++)
            {
                if(b[k][j])
                {
                    if(c[i-1][k]>max)
                    {
                        max=c[i-1][k];
                    }
                }
            }
            c[i][j]=max+a[i][j];
        }
    }
    int max=-1;
    for(j=1;j<=n;j++)
    {
        if(c[n][j]>max)
        {
            max=c[n][j];
        }
    }
    printf("%d\n",max);
    return(0);
}
