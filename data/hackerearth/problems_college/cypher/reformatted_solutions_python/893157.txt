
t = int(raw_input())
if t <= 10000 and t >= 1:
    lists = []
    for i in range(0, t):
        lists.append(raw_input())

    for i in lists:
        j = 0
        s = ''
        for letter in i:
            k = ord(letter) + j
            j += 1
            if k > 122:
                s += chr(k-26)
            else:
                s += chr(k)
        print(s)
