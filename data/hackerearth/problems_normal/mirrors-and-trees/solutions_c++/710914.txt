#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    int arr[262200];
    while(T--)
    {
    	bool visited[262200]={false};
    	int n;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	cin>>arr[i];
    	int k=0,pre=-1,i=0;
    	while(i<n)
    	{
    		
    	
    			while(arr[i]==0&&i>pre)
    			i--;
    			if(i>pre)
    			{
    			cout<<arr[i]<<endl;
    			visited[i]=true;
    			}
    			
    		pre=k;
    		k*=2;
    		k+=2;
    		i=k;
    	}
    	if((n-1)>pre)
    	{
    		i=n;
    			while(arr[i]==0&&i>pre)
    			i--;
    			if(i>pre)
    			{
    			cout<<arr[i]<<endl;
    			visited[i]=true;
    			}
    		
    	}
    	i=0;k=0;
    	int next=1;
    		while(i<n)
    	{
    		
    	
    			while(arr[i]==0&&i<next)
    			i++;
    			if(i<next&&visited[i]==false)
    			{
    			cout<<arr[i]<<endl;
    			visited[i]=true;
    			}
    			
    		next=next*2+1;
    		k*=2;
    		k+=1;
    		i=k;
    	}
    	if((n-1)>i)
    	{
    		i=n;
    			while(arr[i]==0&&i<n)
    			i++;
    			if(i<n&&visited[i]==false)
    			{
    			cout<<arr[i]<<endl;
    			visited[i]=true;
    			}
    		
    	}
    	cout<<endl;
    }
    return 0;
}
