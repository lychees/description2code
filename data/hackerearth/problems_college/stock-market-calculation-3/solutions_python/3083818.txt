def sell(stock):
    Max_index=stock.index(max(stock))
    Max_price=max(stock)
    profit=Max_index*Max_price-sum(stock[0:Max_index])
    if Max_index!=(len(stock)-1):
        stock=stock[(Max_index+1):(len(stock))]
        profit=sell(stock)+profit
    return profit

tc = int(input())

while tc:
	days = int(input())
	prices = map(int, raw_input().split())
	print(sell(prices))
	tc-=1
