#include <stdio.h>
#include <string.h>

int main()
{
	char s[51];
	int t, i, j, n, count;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%s", s);
		n = strlen(s);
		j = n/2;
		count = 0;
		if(n%2 == 1)
		{
			putchar(s[j]);
			count = 1;
			i=j-1;
			j++;
		}
		else
			i = j-1;
		while(count != n)
		{
			if(i >= 0 && j < n)
			{
				if(s[i] > s[j])
				{
					putchar(s[j]);
					putchar(s[i]);
				}
				else
				{
					putchar(s[i]);
					putchar(s[j]);
				}
			}
			i--;
			j++;
			count+=2;
		}
		putchar('\n');
	}
	return 0;
}
