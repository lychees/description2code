#include <iostream>
#include<string>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string st;
		cin>>st;
		int n=st.length();
		long long s=0;
		for(int i=0;i<n;i++)
		{
			for(int j=i,k=i;j>=0&&k<n;j--,k++)
			if(st[j]==st[k])
			s=s+(k-j+1)*(k-j+1);
			else
			break;
		}
		for(int i=0;i<n;i++)
		{
			for(int j=i,k=i+1;j>=0,k<n;j--,k++)
			if(st[j]==st[k])
			s=s+(k-j+1)*(k-j+1);
			else
			break;
		}
		cout<<s<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
