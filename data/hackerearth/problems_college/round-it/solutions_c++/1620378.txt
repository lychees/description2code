#include <iostream>

using namespace std;

int main()
{
    int t,n[60],b[60],i,j,p,q;

    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>n[i]>>b[i];
    }
    for(i=0;i<t;i++)
    {p=q=n[i];
        for(j=0;j<b[i];j++)
        {
            if(p%b[i]==0&&q%b[i]==0)
                {cout<<p<<endl; break;}
            else if(p%b[i]==0)
                {cout<<p<<endl; break;}
            else if(q%b[i]==0)
                {cout<<q<<endl; break;}
            else
            {
                p++;
                q--;
            }
        }
    }
}
