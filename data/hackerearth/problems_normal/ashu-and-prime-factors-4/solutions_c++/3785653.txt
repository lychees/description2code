bool isPrime[MAXN], mark[MAXN];
int cnt[MAXN];
void sieve()
{
	for (int i = 2; i < MAXN; i++)
		isPrime[i] = 1;

	for (int i = 2; i * i < MAXN; i++)
		if (isPrime[i])
			for (int j = i * i; j < MAXN; j += i)
				isPrime[j] = 0;
}
void precalc()
{
	for (int i = 2; i < MAXN; i++)
	{
		if (isPrime[i])
		{
			for (int j = i; j < MAXN; j += i)
			{
				if (!mark[j])
				{
					mark[j] = 1;
					cnt[i]++;
				}
			}
		}
	}
}
int main()
{
	ios_base::sync_with_stdio(false);

	sieve();

	precalc();
	
    int t, x;

	cin >> t;

	while (t--)
	{
		cin >> x;
		
		cout << cnt[x] << "\n";
	}
  	return 0;
}
