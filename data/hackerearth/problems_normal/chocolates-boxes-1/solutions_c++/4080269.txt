#include <bits/stdc++.h>
using namespace std;

int main()
{
    static long int cases,n,k,i,j,count,r,x[15][250];
    scanf("%li",&cases);
    while(cases--)
    {
    	 scanf("%li",&n);
    	for(i=0;i<n;i++)
    	  {
    	  	scanf("%li",&r);
    	  	x[cases][r]++;
    	  }
    	 scanf("%li",&k);
    	count=0; r=0;
    	if(k%2==0)
    	{
    		while(x[cases][k/2]>0)
    		{
    			x[cases][k/2]--;
    			count=count+x[cases][k/2];
    		}
    	}
    	else
    	r=1;
    	for(i=0;i<k/2+r;i++)
    		count=count+x[cases][i]*x[cases][k-i];
    	cout << count << endl;
    }
    return 0;
}
