#include <bits/stdc++.h>
using namespace std;
map<int,vector<int> >m;
map<int,bool>h;
int main()
{
	int t,k,m1,an,u;
	cin>>t;
	u=t;
	while(t--){
		cin>>k;
		cin>>m1;
		h[k]=true;
		while(m1--){
			cin>>an;
			m[k].push_back(an);
		}
	}
    map<int,vector<int> >::iterator it;
    for(it=m.begin();it!=m.end();it++){
    	for(int i=0;i<it->second.size();i++){
    		if(h[it->second[i]]){
    			continue;
    		}else{
    			h[it->second[i]]=true;
    		}
    	}
    }
   cout<<h.size()-u;
   return 0;
}
