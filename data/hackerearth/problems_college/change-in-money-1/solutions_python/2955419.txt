sl = raw_input().split()
N,M = (int(sl[0]),int(sl[1]))
lis = map(int,raw_input().split())
arr = [[0]*M for _ in xrange(N+1)]
for i in xrange(M):
	arr[0][i] = 1
for i in range(1,N+1):
	for j in xrange(M):
		x = arr[i-lis[j]][j] if i-lis[j] >=0 else 0
		y = arr[i][j-1] if j>0 else 0
		arr[i][j] = x+y
print arr[N][M-1]
