[r, c] = map(int, raw_input().split())
m = []
row = []
col = []
for i in xrange(r):
    co = map(int, raw_input().split())
    m += [co]
    row += [sum(co)]
for i in xrange(c):
    temp = 0
    for j in xrange(r):
        temp += m[j][i]
    col += [temp]
re = 0
for i in xrange(r):
    for j in xrange(c):
        temp = row[i] ^ col[j]
        if temp > re:
            re = temp
print re
