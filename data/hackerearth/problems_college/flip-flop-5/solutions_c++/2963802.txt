#include <iostream>
#include <string>
using namespace std;

int main()
{
	
	int n;
	cin>>n;
	while(n--){
		string s;
		cin>>s;
		
		int count = 0;
		
		for(int i = 0; i < s.length() ; i++){
			if(s[i] == 'X'){
				while(s[i+1] == 'X' && i+1 < s.length()){
					count++;
					i++;
				}
			}
			else{
				while(s[i+1] == 'Y' && i+1 < s.length()){
					count++;
					i++;
				}
			}		
			
		}
		cout<<count<<endl;
	}
   // cout << "Hello World!" << endl;
    return 0;
}
