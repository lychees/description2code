#include <iostream>
using namespace std;

int main()
{
    int T, length;
    string str;
    bool bSlap;
    
    cin >> T;
    while(T--)
    {
    	cin >> str;
    	length = str.size() - 1;
    	
    	bSlap = false;
    	for(int i=0; i < length; i++)
    	{
    		if(str[i] == str[i+1])
    		{
    			cout << "SLAP";
    			bSlap = true;
    			break;
    		}
    	}
    	
    	if(!bSlap)
    		cout << "KISS";
    	
    	cout << endl;
    }
    return 0;
}
