import bisect


n, l, r, q = raw_input().split()
n = int(n)
l = int(l)
r = int(r)
q = int(q)
a = list(map(int,raw_input().split()))
queries = list(map(int,raw_input().split()))

a = a[l:r+1]
a = list(sorted(a))


for i in range(0,q):
	index = bisect.bisect_left(a, queries[i])
	if(index >= len(a)):
		print(-1)
	elif (a[index] == queries[i]):
		index += 1
		if(index >= len(a)):
			print(-1)
		else:
			print(a[index])
	else:
		print(a[index])
