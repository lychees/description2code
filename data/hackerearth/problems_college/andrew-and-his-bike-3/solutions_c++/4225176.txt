#include<iostream>
#include<stdio.h>
#include <vector>
using namespace std;
vector<int> graph[100005];
int child[100005]={0},vis[100005]={0};
int dfs(int v)
{
	vis[v]=1;
	for(int i=0;i<graph[v].size();i++)
	{
		if(vis[graph[v][i]]==0)
		child[v]+=dfs(graph[v][i]);
	}
	return 1+child[v];
}
int main()
{
	int n;
	cin >> n;
	for(int i=0;i<n-1;i++)
	{
		int x,y;
		cin >> x >> y;
		graph[x].push_back(y);
		graph[y].push_back(x);
	}
	dfs(1);
	cout << child[1] << ' ';
	for(int i=2;i<=n;i++)
		cout << child[i]+1 << ' ';
		cout << '\n';
	return 0;
}
