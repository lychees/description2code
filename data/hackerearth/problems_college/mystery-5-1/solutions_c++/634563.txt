#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int a[26]={0},max=0;
    	for(int i=0;i<s.length();i++)
    	{
    		a[(int)s[i]-97]++;
    		if(a[(int)s[i]-97] > max)
    			max = a[(int)s[i]-97];
    	}
    	cout<<max<<endl;
    }
    return 0;
}
