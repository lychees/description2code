#include<iostream>
#include<cstdio>
#include<string>
#include<vector>

using namespace std;

int main()
{
    int n,i,j,big=0;
    scanf("%d",&n);
    string x;
    cin>>x;
    n=x.size();
    for(i=0;i<n && x[i]==x[0];i++);
    if(i==n)
    {
        printf("%d\n",n);
        return 0;
    }
    for(i=0;i<n;i++)
    {
        int c=0;
        int prev=i;
        while(x[i]=='w' && i<n)
        {
            i++;
            c++;
        }
        if(i==n)
        {
            c++;
        	for(j=prev-2;j>=0 && x[j]==x[prev-1];j--)
                c++;
            printf("%d\n",(c>big?c:big));
            return 0;
        }
        c++;
        for(j=(i+1)%n;j!=i && (x[j]==x[i] || x[j]=='w');j=(j+1)%n)
            c++;
        for(;j!=i && x[j]!=x[i];j=(j+1)%n)
            c++;
        if(c>big)
            big=c;
    }
    printf("%d\n",big);

    return 0;
}
