#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int main()
{
	int n;
	cin>>n;
	int i,x;
	vector<int>a;
	for(i = 0; i < n; i++) {
		cin>>x;
		a.push_back(x);
	}
	int s = 0;
	while(a.size() > 2) {
		int maxi,p;
		maxi = -1;
		for(i = 1; i < a.size()-1; i++) {
			if(a[i-1]*a[i+1] > maxi) {
				maxi = a[i-1]*a[i+1];
				p = i;
			}
		}
		s += maxi;
		a.erase(a.begin()+p);
	}
	cout<<s<<endl;
	return 0;
}
