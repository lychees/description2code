vector<int> primes;
int vis[1000001];
void solve()
{
	int l,r;
	s(l);
	s(r);
	ll ans=0;
	for(int i=0;i<primes.size()-1;i++)
	{
		int idx,idx1;

		int l1=ceil((double)l/primes[i] -EPS);
		int r1=r/primes[i];
		if(l1 <= primes[i+1])
		{
			idx=i+1;
		}
		else
		{
			idx=lower_bound(primes.begin()+i+1,primes.end(),l1)-primes.begin();
		}
		if(r1 >= primes[primes.size()-1])
		{
			idx1=primes.size();
		}
		else idx1=upper_bound(primes.begin()+i+1,primes.end(),r1)-primes.begin();
		ans += idx1-idx;
	}
	printf("%lld\n",ans);
}
		
int main()
{
	fill(vis,0);
	int pr=2;
	while(1)
	{
		int flag=0;
		primes.pb(pr);
		for(int i=pr*2;i<=1000000;i=i+pr)
		{
			vis[i]=1;
		}
		for(int i=pr+1;i<=1000000;i++)
		{
			if(!vis[i])
			{
				flag=1;
				pr=i;
				break;
			}
		}
		if(flag == 0)
		break;
	}
	
	int t;
	s(t);
	for(int i=0;i<t;i++)
	{
		solve();
	}
}
