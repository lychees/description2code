#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const int N=1e6+10;

int a[N];

int main() {
//  freopen("a.cpp","r",stdin);
	ios::sync_with_stdio(0);
  int tt;
  cin>>tt;
  while(tt--) {
    int n;
    ll x;
    cin>>n>>x;
    for(int i=0;i<n;i++) {
      cin>>a[i];
    }
    int start=0,end=0;
    ll val=a[start];
    while(end<n) {
      while(val>x&&start<n) {
        val-=a[start++];
      }
      while(val<x&&end<n) {
      	end++;
        val+=a[end];
      }
      if(val==x) {
        puts("YES");
        goto hell;
      }
    }
    puts("NO");
    hell:;
  }
  return 0;
}
