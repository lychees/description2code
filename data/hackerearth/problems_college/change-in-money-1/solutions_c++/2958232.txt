#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;
	
int main()
{
	int N, M, a;
	vector<int> A;
	
	cin >> N >> M;
	
	for(int i=0;i<M;i++)
	{
		cin >> a;
		A.push_back(a);
	}
	
	sort(A.begin(), A.end());
	
	long int P[N+1];
	for(int i=0;i<N+1;i++)
		P[i]=0;
	
	P[0]=1;
	
	for(int i=0;i<M;i++)
	{
		for(int j = A[i] ; j<N+1; j++)
			P[j] += P[j - A[i]];
	}
	
	cout << P[N];
	
	return 0;
}
