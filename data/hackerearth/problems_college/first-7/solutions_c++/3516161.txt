#include <bits/stdc++.h>
#define ll long long 
using namespace std;

int main() {
    int t;
    cin >> t;
    while (t--) {
        ll n;
        scanf("%lld", &n);
        ll answer = (n*(n+1)*(2*n+1))/3 + (n*(n-1)*(2*n-1))/3;
        answer -= n*(n+1)/2;
        answer -= n*(n-1)/2;
        answer += n*n*(n-1);
        answer += n*n*(n-1);
        cout.setf(ios::fixed);
        cout << setprecision(6);
        cout << double(answer) / (n*n) << endl;
    }
}
