#include <bits/stdc++.h>
#define ll long long int
#define s(a) scanf("%lld",&a)

using namespace std;

int main()
{
    ll t,a,b,c;
    s(t);
    while(t--) {
        s(a);s(b);s(c);
        if(a%2==0 && b%2==0 && c%2==0) cout<<"Aishwarya\n";
        else cout<<"Ashima\n";
    }
    return 0;
}
