#include <iostream>
using namespace std;

int main()
{
    int t,z,n,i,j,max,count;
    cin>>t;
    for(z=0;z<t;z++)
    {
    	cin>>n;
    	int a[n];
    	
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	max=a[n-1];
    	count=0;
    	
    	for(i=n-2;i>=0;i--)
    	{
    		if(a[i]>max)
    		{
    			max=a[i];
    			count++;
    		}
    	}
    	cout<<count+1<<endl;	
    }
}
