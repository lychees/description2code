#include<bits/stdc++.h>
using namespace std;
char str[100005],temp[100005];
int a[50];
int main()
{
 int t;
 scanf("%d",&t);
 while(t--)
 {
  memset(a,0,sizeof(int)*50);
  scanf("%s",str);
  int n=strlen(str),p;
  a[str[n-1]-'a']=n-1;
  int j,i;
  for( i=n-2;i>=0;i--)
  {
    j=str[i]-'a'+1;
    p=j-1;
    //cout<<j<<endl;
    while(a[j]==0&&j<26)
     j++;
    if(j<26)
    {

     swap(str[i],str[a[j]]);
      break;
    }
    a[p]=i;
    //cout<<j<<" "<<a[j-1]<<endl;
  }
  if(i<0)
  {
   printf("no answer\n");
   continue;
  }
  j=0;
  //cout<<i<<endl;
  for(int k=i+1;k<n;k++)
   temp[j++]=str[k];
   temp[j]='\0';
   //cout<<j<<endl;
   sort(temp,temp+j);

  for(int k=0;k<=i;k++)
   printf("%c",str[k]);
   printf("%s",temp);
   printf("\n");
 }
 return 0;
}
