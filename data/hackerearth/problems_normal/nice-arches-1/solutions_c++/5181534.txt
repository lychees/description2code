#include <iostream>
#include <string>
#include <stack>

using namespace std;

int main(int argc, char **argv) {
	int n;
	cin>>n;
	string s;
	int count=0;
	for (int i = 0; i < n; ++i) {
		cin>>s;
		stack<char> stack;
		for (int j = 0; j < s.size(); ++j) {
			if(!stack.empty()){
				if(stack.top()==s[j]){
					stack.pop();
				}else{
					stack.push(s[j]);
				}
			}else{
				stack.push(s[j]);
			}
		}
		if(stack.empty()){
			count++;
		}
	}
	cout<<count<<endl;
}
