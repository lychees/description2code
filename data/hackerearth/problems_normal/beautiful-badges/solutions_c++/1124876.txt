using namespace std;

double tt = sqrt(2)/2;

bool ok(int a,int b){
	if(a < b) swap(a,b);
	if(((double)a)*tt < b && ((double)b)*tt < a) return 1;
	return 0;
}

main(){

		int n;
		scanf("%d",&n);

		vector<int> a,b;

		for(int i=0;i<n;i++){
			int t,p;
			scanf("%d%d",&t,&p);
			if(t == 1) a.pb(p);
			else b.pb(p);
		}

		sort(a.begin(),a.end());
		sort(b.begin(),b.end());

		n = a.size();
		int m = b.size();

		for(int i=0;i<n;i++) debug("%d ",a[i]);
		debug("\n");
		for(int i=0;i<m;i++) debug("%d ",b[i]);
		debug("\n");

		int p1 = 0, p2 = 0;

		int ans = 0;

		while(p1 < n && p2 < m){

			if(ok(a[p1],b[p2])){
				ans++, p1++, p2++;
				continue;
			}

			if(a[p1] < b[p2])
				p1++;
			else
				p2++;

		}

		printf("%d\n",ans);

}
