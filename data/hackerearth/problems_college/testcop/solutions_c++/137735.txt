#include<iostream>
#include<stdio.h>
using namespace std;
int gcd(int a,int b)
{
    int t;
    while(a)
    {
        t = a;
        a = b%a;
        b = t;
    }
    return b;
}
int main()
{
int a[101],ans; // N denotes length of array
int i,t,n;
scanf("%d",&t);
while(t--)
{
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
ans = a[0];
for(i=1;i<n;i++)
{
    ans = gcd(ans,a[i]); //calls the gcd() that we just created a while ago
}
//printf("%d\n",ans);
if(ans==1)
{
	printf("YES\n");
}
else
{
	printf("NO\n");
}
}
return 0;
}
