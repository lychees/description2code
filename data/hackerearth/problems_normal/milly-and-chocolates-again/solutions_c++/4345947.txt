#include <iostream>
using namespace std;

long binary_search(long *arr, long element,long n){
	
	long low = 1;
	long high = n-1;
	
	while(low <= high){
		
		long mid = low + (high-low)/2;
		
		if(arr[mid-1] < element && arr[mid] >= element){
			
			return mid;
			
		}
		
		if(arr[mid] < element){
			
			low = mid + 1;
		}
		
		else {
			
			high = mid - 1;
		}
		
	}
	
}


int main()
{
    
    long n,q;
    
    cin>>n;
    
    long arr[n+1];
    arr[0]=0;

    for(long i = 1; i<=n;i++){
    	cin>>arr[i];
    	arr[i] += arr[i-1];
    }
    cin>>q;
    for(long j=0;j<q;j++){
    	long ele;
    	cin>>ele;
    	cout<<binary_search(arr,ele,n+1)<<endl;
    }
    return 0;
}
