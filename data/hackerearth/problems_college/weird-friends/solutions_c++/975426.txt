# include <iostream>
# include <stdio.h>

using namespace std;

int main()
{
    long long L,X,Y,temp;
    scanf("%lld%lld%lld",&L,&X,&Y);
    int c=0;
    temp=X;
    while(1)
    {
        if(X>=L)
        {
            c++;
            break;
        }

        else
        {
            c++;
            X=temp+X-Y;
        }
    }
    printf("%d\n",c);
    return 0;
}
