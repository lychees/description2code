#include <bits/stdc++.h>
using namespace std;
bool visited[10050];

void initialize()
{
	for(int i=0;i<10050;i++)
		visited[i]=0;
}
int main()
{
    // cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	initialize();
    	int e;
    	cin>>e;
    	int x,y,a=0;
    	for(int i=0;i<e;i++)
    	{
    		cin>>x>>y;
    		if(visited[x]==0)
    		{
    			visited[x]=1;
    			a++;
    		}
    		if(visited[y]==0)
    		{
    			visited[y]=1;
    			a++;
    		}
    	}
    	cout<<a<<endl;
    }
    return 0;
}
