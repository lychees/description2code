    #include <iostream>
    using namespace std;
    typedef long long ll;
    int main()
    {
       ll k;
       cin>>k;
       ll n,rem,count=0,m=0;
       cin>>n;
       for(ll i=1;i<=n;i++){
       	ll s=i;
       	rem=0;
       	while(s!=0){
       		rem=s%10;
       		if(rem==4||rem==7){
       			count++;
       		}
       		s/=10;
       }
       if(count>k){
       	count=0;
       	m++;
       }
       }
       
       
       cout<<m;
        return 0;
    }
