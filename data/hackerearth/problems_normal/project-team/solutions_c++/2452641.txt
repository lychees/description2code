#include <iostream>
#include<algorithm>
using namespace std;
void fun(){
   int n,i;
   long int temp,min=999999999,max=0;
   cin>>n;
   int a[n];
   for(i=0;i<n;i++){
   	cin>>a[i];
   }
   sort(a,a+n);
   for(i=0;i<n/2;i++){
   	temp=a[i]+a[n-i-1];
   	if(temp<min)
   	min=temp;
   
   if(temp>max)
   max=temp;
   }
   cout<<max-min<<endl;
}

int main()
{
    int t;
    cin>>t;
    while(t--){
    	fun();
    }
    return 0;
}
