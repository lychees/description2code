LL dp[1000][3][3];
int n;
LL rec(int pos,int pu,int pd) {
  if(pos==n)
    return 0;
  if(dp[pos][pu][pd]!=-1)
    return dp[pos][pu][pd];
  LL ans=0;
  if(pu==0 && pd==0) {
    ans+=1+rec(pos+1,1,0);
    ans+=1+rec(pos+1,0,1);
    ans+=1+rec(pos+1,1,1);
    ans+=rec(pos+1,0,0);
  }
  if(pu==0 && pd==1) {
    ans+=1+rec(pos+1,1,0);
    ans+=rec(pos+1,0,0);
  }
  if(pu==1 && pd==0) {
    ans+=1+rec(pos+1,0,1);
    ans+=rec(pos+1,0,0);
  }
  if(pu==1 && pd==1) {
    ans+=rec(pos+1,0,0);
  }
  return dp[pos][pu][pd]=ans;
}
