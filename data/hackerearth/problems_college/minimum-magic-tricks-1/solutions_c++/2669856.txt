#include<bits/stdc++.h>
using namespace std;

#define mod 1000000007
long long dp[1001];//
long long powmod(long long n)
{
	if(n == 0)
		return 1;
	long long temp = powmod(n/2);
	if(n%2==0)
		return (temp%mod*temp%mod)%mod;
	return (2*(temp%mod)*(temp%mod))%mod;
}

long long solve(long long h)
{
	if(h<2)
		return h;
	if(dp[h]!=-1)
		return dp[h];
	return dp[h] = (powmod(h-2)%mod + solve(h-2)%mod)%mod;
}

int main()
{
	memset(dp,-1,sizeof(dp));
	long long t,h;
	cin>>t;
	while(t--)
	{
		cin>>h;
		cout<<solve(h)<<"\n";
	}
	return 0;
}
