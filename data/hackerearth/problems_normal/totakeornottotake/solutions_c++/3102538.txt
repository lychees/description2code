#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int x,i=1,b;
        cin>>b;
        long long mx[55],mn[55];
        mx[0]=mn[0]=1;
        for(int i=1;i<=b;i++)
        {
        	char c;
        	cin>>c;
        	if(c=='N')
        	{
        		mx[i] = max(mx[i-1], -mn[i-1]);
   				mn[i] = min(mn[i-1], -mx[i-1]);
        	}
        	else
        	{
        		cin>>x;
        		if(c=='+')
        		{
        			mx[i]=mx[i-1]+x;
        			mn[i]=mn[i-1];
        		}
        		if(c=='-')
        		{
        			mn[i]=mn[i-1]-x;
        			mx[i]=mx[i-1];
        		}
        		if(c=='*')
        		{
        			mx[i] = max(mx[i-1], mx[i-1]*x);
   					mn[i] = min(mn[i-1], mn[i-1]*x);
        		}
        		if(c=='/')
        		{
        			mx[i] = max(mx[i-1], mx[i-1]/x);
   					mn[i] = min(mn[i-1], mn[i-1]/x);
        		}
        	}

        }

        printf("%lld\n",mx[b]);
    }
    return 0;
}
