#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    {
        long long n,i;
        cin>>n;
        int arr[n];
        for(i=0;i<n;i++)
        {
            cin>>arr[i];
        }
        int mx=0,ans=0;
        for(i=0;i<n;i++)
        {
            mx=max(0,mx+arr[i]);
            if(ans<mx)
                ans=mx;
        }
        cout<<ans<<endl;

    }
    return 0;
}
