#include <iostream>
using namespace std;

int main()
{
    int t;
	cin>>t;
	if(t<1||t>10)
	{
		cout<<"Invalid Test";
		return 0;
	}
	while(t--) 
	{
		string s;
		cin>>s;
		int len=s.length();
		if(len<1||len>100)
		{
			cout<<"Invalid Input"<<endl;
			continue;
		}
		int res=0,i;
		int l=0,u=0;		
		for(i=0;i<len;i++) 
		{
			if(s[i]>='a' && s[i]<='z')
				++l;
			else if(s[i]>='A' && s[i]<='Z')
				++u;
		}
		if(l==0&&u==0)
		{
			cout<<"Invalid Input"<<endl;
			continue;
		}
		res=min(l,u);
		cout<<res<<endl;
	}

    return 0;
}
