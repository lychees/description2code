#include<bits/stdc++.h>
using namespace std;
int main(){
    string str;
    getline(cin,str);
    int N;
    cin>>N;
    int i=0,counter=0;
    for(i=0;i<str.length();){
        if(counter==N){
        	cout<<endl;
            counter=0;
            continue;
        }
        else{
            cout<<str[i];
            i++;
            counter++;
        }
    }
    return 0;
}
