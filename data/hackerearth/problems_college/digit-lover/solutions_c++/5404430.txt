#include <iostream>
#include <cstdio>

int main()
{
	int T, value;
	
	scanf("%d", &T);
	
	if(T<=1 or T>100) return 0;
	
	while(T != 0)
	{
		scanf("%d", &value);
		
		if(value <= 1 or value > 100000) return 0;
		
		int sum=0, reminder;
		
		while(value != 0 or sum/10 != 0)
		{
			reminder = value%10;
			sum+= reminder;
			value = value/10;
			
			if(value == 0 && sum/10 != 0)
				{
				value = sum;
				sum = 0;
				}
		}
		
		printf("%d\n", sum);
	T--;
	}
}
