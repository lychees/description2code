#include <iostream>
#include <algorithm>
#include <set>
using namespace std;

int main() {
	
	set <string> sub;
	string str;
	cin >> str;

	for(int i=0; i<str.length(); i++) {
		for(int j=i; j<str.length(); j++) {
			string cur = str.substr(i,j-i+1);

			sub.insert(cur);
		}
	}

	cout << sub.size() << endl;
	
	return 0;
}
