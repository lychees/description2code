#include <bits/stdc++.h>
using namespace std;
bool isPal(int start,int end,string arr)
{
	string str=arr;
	if(end<str.length()-1)
	{
		reverse(str.begin()+start,str.begin()+end+1);
	}
	else
	{
		reverse(str.begin()+start,str.begin()+str.length());
	}
	if(arr==str)
		return true;
	return false;	
}
int main()
{
	string str;
	long long ans=0;
	cin>>str;
	ans+=str.length();
	for(int i=0;i<str.length();i++)
	{
		for(int j=i+1;j<str.length();j++)
		{
			if(str[i]==str[j])
			{
				if(isPal(i,j,str))
					ans+=1;
			}
		}
	}
	cout<<ans<<endl;
    return 0;
}
