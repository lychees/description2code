#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
  	
int flag=0;
   string t,s;
   
   while(getline(cin,s)){
   	flag=0;
   	t="";
      long long n;
       n=s.size();
       long long i=0;
     for(i=0;i<n;i++){
   	if((s[i]=='/')&&(s[i+1]=='/')){
   		flag=1;
   		t=t+"//";
   		i++;
   	   	continue;}
   	   	
   	   	
   	   	if((s[i]=='-')&&(s[i+1]=='>')){
   	   		if(flag==0)
   	   		{t=t+'.';
   	   		i++;}
   	   		
   	   		else{
   	   			t=t+s[i];
   	   			continue;}
   	   		
   	   	}
   	
   	
   	else {
   		t=t+s[i];
   		continue;}}
   	
     cout<<t;
     cout<<endl;
   	
   }
    
    return 0;
}
