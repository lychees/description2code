#include<iostream>
using namespace std;
long long int abso(long long int a) {
	if(a>0) {
		return a;
	} else {
		return -a;
	}
}
int main() {
	int t;
	cin>>t;
	int i;
	for(i=0;i<t;i++) {
		long long int n,a,b,c,d,move;
		cin>>n>>a>>b>>c>>d>>move;
		if(move==1) {
			c = (c+1 <= n) ? (c+1) : c;
			if(b>d) {
				d = d+1;
			} else if (b<d) {
				d = d-1;
			}
		}
		if((a<=c) && (abso(b-d) <= (n-a))) {
			cout<<"Draw"<<endl;
		} else {
			cout<<"White Wins"<<endl;
		}
	}
	return 0;
}
