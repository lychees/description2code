#include<iostream>
#include<cmath>
using namespace std;

int main(){
	int r, c, n, steps, t_cases;
	char ch;
	cin >> t_cases;
	while(t_cases--){
		cin >> n;
		for(r = 0; r < n; r++){
			for(c = 0; c < n; c++){
				cin >> ch;
				if(ch == '*'){
					steps = abs(n/2 - r) + abs(n/2 - c);
				}
			}
		}
		cout << steps << "\n";
	}
}
