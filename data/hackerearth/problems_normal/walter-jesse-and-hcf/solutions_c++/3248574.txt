#include <iostream>
#include <math.h>
#include <algorithm>
using namespace std;

long long int gcd(unsigned long long int a,unsigned long long int b){
	if(b==0)return a;
	else return gcd(b,a%b);
}

int main()
{
	int d,i;
    cin >> d;
    for(i=0;i<d;i++){
    	unsigned long long int x,y,k;
    	cin >> x >> y >> k;
    	unsigned long long int gc = gcd(x,y);
    	int s = (int)sqrt(gc);
    	int num[s*2];
    	int a=0;
    	
    	for(int j=s;j>=1;j--){
    		if(gc%j==0){
    			if(j*j != gc){
    				num[a] = j;
    				num[a+1] = (gc/j);
    				a+=2;
    			}
    			else{
    				num[a] = j;
    				a++;
    			}
    			 
    		}
    		
    	}
    	
    	if(a<k){
    		cout << "No crymeth today" << endl;
    	}
    	else{
    		sort(num,num+a);
    		cout << num[a-k] << endl;
    	}
    	
    }
    return 0;
}
