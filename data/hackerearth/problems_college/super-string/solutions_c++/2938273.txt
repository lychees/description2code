#include<iostream>
#include<algorithm>
using namespace std;
main()
{
    int i,j,k,l,t,cnt,a,f=0;
    string c;
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>c;
        for(j=0;j<c.length();j++)
        {
            l=count(c.begin(),c.end(),c[j]);
            cnt=toascii(c[j]);

            if(cnt==90)
            {

                a=91-cnt;

            }
            else if(cnt<90)
            {
                a=91-cnt;
            }
            if(a==l)
            {
                f=1;
            }
            else
            {
                f=0;
                cout<<"No"<<endl;
                goto q;
            }
            a=0;
            cnt=0;

        }
        cout<<"Yes"<<endl;
        q:;
    }
}
