#include <iostream>
#include <algorithm>
#include <stdio.h>
#define ll long long int
using namespace std;

ll arr[100000];

int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	ll size, pos;
    	scanf("%lld %lld",&size,&pos);
    	
    	for(ll i = 0 ; i < size ; i++){
    		scanf("%lld",&arr[i]);
    	}
    	
    	sort(arr, arr+size);
    	printf("%lld\n",arr[pos-1]);
    }
    return 0;
}
