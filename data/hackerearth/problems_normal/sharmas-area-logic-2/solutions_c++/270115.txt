#include <iostream>
using namespace std;
#include <stdio.h>

int main()
{
    int t,i,x,y;
    cin>>t;
    while(t--){
    	int n, minx=1000,maxx=-1000,miny=1000,maxy=-1000;;
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>x>>y;
    		if(x<minx)
    		minx=x;
    		if(x>maxx)
    		maxx=x;
    		if(y<miny)
    		miny=y;
    		if(y>maxy)
    		maxy=y;
    	}
    	if(maxx==minx){
    		printf("%d\n",maxy-miny);
    		}
    	else if(maxy==miny)
    	printf("%d\n",maxx-minx);
    	else
    	printf("%d\n",(maxx-minx)*(maxy-miny));
    }
    return 0;
}
