#include<bits/stdc++.h>
using namespace std;

int main(){int tc,i,t;
string a,b;
cin>>tc;
while(tc--){
    cin>>a;
    cin>>b;
    int ans=0;
    t=min(a.length(),b.length());

    
        for(i=0;i<t;i++){
            if(a[i]==b[i]){
                ans++;
            }
        }
    cout<<ans<<endl;
    a.clear();
    b.clear();

    }



return 0;}
