#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    if(t>10 || t<=0)
    	cout<<"Invalid Test"<<endl;
    else
    {
    while(t--)
    {
    	string s;
    	cin>>s;
    	int flag=1;
    	int count1=0,count2=0,count3=0,count4=0;
    	if(s.size()>100)
    		cout<<"Invalid Input"<<endl;
    	else
    	{
    	for(int i=0;i<s.size();i++)
    	{
    		if(isalpha(s[i]))
    			if(islower(s[i]))
    				count1++;
    			else
    				count2++;
    		else if(!isdigit(s[i]))
    			count4++;
    		else
    			count3++;
    	}
    	if((count4+count3)==s.size())
    		cout<<"Invalid Input"<<endl;
    	else if(count3==s.size())
    		cout<<"Invalid Input"<<endl;
    	else
    		cout<<min(count1,count2)<<endl;
    	}
    }
    }
    return 0;
}
