typedef long long ll;
#include <iostream>
#include <stdio.h>
#include <cmath>
#include <algorithm>
#include <vector>
#include <string>
#include <string.h>
#include <map>
#include <queue>
#include <stack>
#include <sstream>
#include <fstream>
#include <stdlib.h>
#include <utility>

using namespace std;
map<ll,ll> f;
int main()
{
    ll n,m,i,flag,x,y,j;
    scanf("%lld %lld",&n,&m);
    for(i=1;i<=m;i++)
    f[i]=0;
    for(i=0;i<n;i++)
    {
        scanf("%lld",&x);
        for(j=0;j<x;j++)
        {
            scanf("%lld",&y);
            f[y]=1;
        }
    }
    flag=0;
    for(i=1;i<=m;i++)
    {
        if(f[i]==0)
        {
            flag=1;
            break;
        }
    }
    if(flag)
    printf("NO\n");
    else
    printf("YES\n");
	return 0;
}
