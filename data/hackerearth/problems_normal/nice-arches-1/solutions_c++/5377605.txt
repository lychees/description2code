#include<bits/stdc++.h>
using namespace std;
int main()
{
	int m;
	scanf("%d",&m);
	string str[105];
	vector<char> v;
	long long int i,j,k,count=0;
	for(i=0;i<m;i++)
		cin >> str[i];
	for(i=0;i<m;i++)
	{
		k=0;
		v.push_back(str[i][0]);
		for(j=1;j<str[i].size();j++)
		{
			if(str[i][j]!=v[k])
			{
				v.push_back(str[i][j]);
				k++;
			}
			else
			{
				v.pop_back();
				k--;
			}
		}
		//printf("%lld\n",v.size());
		if(v.size()==0)
			count++;
		v.clear();
	}
	printf("%lld\n",count);
	return 0;
}
