#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;

long int findIt(long int arr[],long int size)
{
	long int ans = 0,curr = 0,i;
	for(i = 1;i<size;i++)
	{
		curr = ((curr%mod) + ((abs(arr[i]-arr[i-1]))*i)%mod)%mod;
		ans = ((ans%mod)+(curr%mod))%mod;
	}
	return ans;
}
int main()
{
	long int size,i,ans = 0,j;
	cin >> size;
	long int x[size],y[size];
	for(i = 0 ;i <size ;i++)
	{
		cin >> x[i] >> y[i];
	}
	sort(x,x+size);
	sort(y,y+size);
	cout << (findIt(x,size)+findIt(y,size))%mod;
	return 0;
}
