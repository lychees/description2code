#include <iostream>
#include <cmath>

using namespace std;

int main()
{
    int n;
    cin>>n;
    int b[n];
    int a[n];
    int m1[n];
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
        b[i]=0;
    }
    m1[0]=a[0];
    b[0]=1;
    b[n-1]=1;
    for(int i=1;i<n-1;i++)
    {
        if(m1[i-1]<a[i])
        {
            b[i]=1;
            m1[i]=a[i];
        }
        else
            m1[i]=m1[i-1];
    }
    m1[n-1]=a[n-1];
    for(int i=n-2;i>0;i--)
    {
        if(m1[i+1]<a[i])
        {
            b[i]=1;
            m1[i]=a[i];
        }
        else
        {
            m1[i]=m1[i+1];
        }

    }
    for(int i=0;i<n;i++)
    {
        if(b[i]==1)
            cout<<(i+1)<<" ";
    }
    cout<<endl;
}
