#include <iostream>
using namespace std;
int n,a[100],mini=1000000,maxi=0;

int check(int val){
int i,tmp=0,count=0,j=0;
for(i=0;i<n;i++){
    
    tmp+=a[i];
    if(tmp==val){count++;tmp=0;j=i+1;}
    if(tmp>val){
        while(tmp>val){
            tmp-=a[j];
            j++;
        }
    if(tmp==val){count++;tmp=0;j=i+1;}  
    }

    

}
return count;
    
    
}
int main()
{
   
   cin>>n;
   int ans[100005];
   int ma=0;
   for(int i=0;i<n;i++){
       cin>>a[i];
       if(a[i]<mini)mini=a[i];
       maxi+=a[i];
   }
   for(int i=mini;i<=maxi;i++){
       int tmp=check(i);
    ans[i]=tmp;
    
    if(tmp>ma)ma=tmp;
   }
   printf("%d\n",ma);
   for(int i=mini;i<=maxi;i++){
       if(ans[i]==ma)printf("%d ",i);
   }
    return 0;
}
