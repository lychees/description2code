#include<bits/stdc++.h>
using namespace std;

int main()
{
	string s;
	int t,count=0,n,j;
	scanf("%d",&t);
	while(t--)
	{
		cin>>s;
		n=s.length();
		j=n-1;
		count=0;
		for(int i=0; i<n; ++i,j--)
		{
			if(abs(s[i+1]-s[i])==abs(s[j]-s[j-1]))
			{
				count++;
			}
			else
			{
				break;
			}
		
		}
		if(count==(n-1))
			printf("Funny\n");
		else
			printf("Not Funny\n");
	}
	
	return 0;
}
