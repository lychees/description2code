#include <iostream>
using namespace std;
int c=0,c1=0;
int fibo(int n)
{
	//int c=0,c1=0;
	if(n==0)
	{
		c++;
		
	}
	else if(n==1)
	{
		c1++;
		//return 1;
	}
	else
	{
		return fibo(n-1)+fibo(n-2);
	}
	//cout<<c<<" "<<c1<<"\n";
	return 0;
	//return 0;
}

int main()
{
   int t;
   cin>>t;
   int n;
   while(t--)
   {
   	cin>>n;
   	fibo(n);
   	cout<<c<<" "<<c1<<"\n";
   	c=0,c1=0;
   }
    return 0;
}
