ll bSearch(ll arr[],ll lo,ll hi,ll key)
{
  ll mid;
  while(lo<hi)
    {
      mid=(lo+hi)>>1;
      if(arr[mid]==key)return mid;
      else if(arr[mid]<key)lo=mid+1;
      else if(arr[mid]>key)hi=mid-1;
    }
  if(arr[lo]>key)return lo-1;
  else return lo;
}
