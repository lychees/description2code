#include <bits/stdc++.h>
using namespace std;
 
int main()
{
	char str[100010];
    cin>>str;
     int T;
    cin>>T;
    int i;
    for(i=0;i<T;i++)
    {
    	int start1, end1, start2, end2;
    	cin>>start1>>end1>>start2>>end2;
    	start1--;
    	end1--;
    	
    	start2--;
    	end2--;
    	if(memcmp(str+start1, str+start2, end1-start1+1)==0 && (end1-start1+1)==(end2-start2+1))
    	
    		cout<<"Yes"<<endl;
    	else
    		cout<<"No"<<endl;
    }
    
    return 0;
}
