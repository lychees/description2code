#include <iostream>
using namespace std;

int main()
{
    int n,i,t,sum=0;
    cin>>n>>t;
    
    int ar[n+1];
    ar[0]=0;
    for(i=1;i<=n;i++){
    	int k;
    	cin>>k;
    	ar[i]=ar[i-1]+k;

    }
        for(;t>0;--t){
    	int a,b;
    	cin>>a>>b;
    	sum=ar[b]-ar[a-1];
    	cout<<sum<<endl;
    }
    return 0;
}
