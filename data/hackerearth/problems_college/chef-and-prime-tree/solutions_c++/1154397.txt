bool prime[5000005];

void make_prime() {
    memset(prime,true,sizeof(prime));
    prime[1] = false;

    int l = sqrt(5000005);
    int i,j;
    For(i,2,l) {
        if(prime[i]) {
            for(j = i*i; j < 5000004; j += i) {
                prime[j] = false;
            }
        }
    }
}
