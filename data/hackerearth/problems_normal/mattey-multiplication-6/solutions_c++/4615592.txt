#include <bits/stdc++.h>
#include <cmath>
#include <cstdio>
using namespace std;
vector<int> res;

int main()
{
  int t;
  scanf("%d" , &t);
  while(t--)
  {
      long long n , m;
      scanf("%lld %lld" , &n , &m);
      res.clear();
      int bit = 0;
      while ( m != 0 ){
      	
      	if ( m % 2 != 0){
      		res.push_back(bit);
      	}
      	m /= 2;
      	bit++;
      	
      }
      printf("(%lld<<%d)", n, res[res.size() - 1]);

      for(int i = res.size() - 2; i >= 0; i--){       

            printf(" + (%lld<<%d)", n, res[i]);
        }
        printf("\n");
      
  }
    return 0;
}
