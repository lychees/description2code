#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios::sync_with_stdio( false );
    cin.tie( NULL );
    
    for( int n; cin>>n; ) {
    	vector<int>num(n);
    	for( int i=0; i < n; i++){
    		cin>>num[i];
    	}
    	sort( num.begin(), num.end() );
    	
    	for( int i=0; i < n; i++){
    		cout<<num[i]-i<<" ";
    	}
    }
    
    return 0;
}
