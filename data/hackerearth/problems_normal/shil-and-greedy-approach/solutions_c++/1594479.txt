string s ;
int T , N ,DP[1 << 20];

int solve(){
    memset(DP,-1,sizeof DP) ;
    queue<int> Q ;
    int num = 0 ;
    for(int i=0;i<N;i++){
        if(s[i] == 'W'){
            num |= (1 << i) ;
        }
    }

    DP[num] = 0 ;
    Q.push(num) ;

    while(!Q.empty()){
        num = Q.front() ;
        Q.pop() ;
        if(num == 0 || (num == (1 << N)-1)){
            return DP[num] ;
        }
        int two , three , s , e;
        two = 3 ;
        s = 0 ;
        e = 1 ;
        while(e < N){
            int nx = num ^ two ;
            if(DP[nx] == -1){
                DP[nx] = DP[num] + 1 ;
                Q.push(nx) ;
            }
            two ^= (1 << s) ;
            e ++ ;
            two ^= (1 << e) ;
            s ++ ;
        }
        three = 7 ;
        s = 0 ;
        e = 2 ;
        while(e < N){
            int nx = num ^ three ;
            if(DP[nx] == -1){
                DP[nx] = DP[num] + 1 ;
                Q.push(nx) ;
            }
            three ^= (1 << s) ;
            e ++ ;
            three ^= (1 << e) ;
            s ++ ;
        }
    }
    return -1 ;
}
int main(){

    T=1 ;
    while(T--){
    	cin >> N ;
        cin >> s ;
        cout << solve() << endl ;
    }
    return 0 ;
}
