#define pi 3.14159265358979323846

using namespace std;
typedef long long ll;
typedef long double ldb;
typedef double db;
double rad,s,l=0,r=90,l1,r1,call,calr;
double cal(double ang){
	double R=s*cos(ang)+rad;
	double h=s*sin(ang);
	return pi*h*(rad*rad+R*R+R*rad)/3;
}
int main()
{
	
	cin>>rad>>s;
	for(int i=1;i<1000;i++)
	{
	l1=l+(r-l)/3;
	r1=r-(r-l)/3;
	call=cal(l1*pi/180);
	calr=cal(r1*pi/180);
	if(call<calr)
		l=l1;
	else
		r=r1;
	}
	printf("%.2lf\n",cal((l)*pi/180));
	
	return 0;
}
