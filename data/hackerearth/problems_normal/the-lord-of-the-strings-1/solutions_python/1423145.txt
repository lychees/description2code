def remove_duplicates(input_string):
	output_list = []
	for char in input_string:
		if char not in output_list:
			output_list.append(char)
	print ''.join(char for char in output_list)


if __name__ == '__main__':
	cases = int(raw_input().strip())
	start = 1
	while start <= cases:
		input_string = raw_input().strip()
		remove_duplicates(input_string)
		start = start + 1
