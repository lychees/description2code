#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    
    int a[n],i,j=0;
    for(i=0;i<n;i++)
    cin>>a[i];
    
    sort(a,a+n);
    
    j=0;
    while(j<n)
    {
    	//printf("%d ",j);
    for(i=j;i<=n;i++)
    {
    //	printf("yes %d    ",i);
    	
    	if(i>=n||(a[i]!=a[j]))
    	{
    		
    		printf("%d\n",n-j);
    		j=i;
    		break;
    	}
    	
    }
    
    }
    return 0;
}
