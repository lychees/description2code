#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int x=floor(sqrt(n));
    	int mad=n-(x*x);
    //	cout<<mad<<endl;
    	int mya=mad/x;
    	if(mad%x!=0)
    		mya++;
    	long long int ans=2*x-1;
    	cout<<ans+mya<<endl;
    }
    return 0;
}
