#include <bits/stdc++.h>
#define lli long long int
#define s(x) scanf("%lld", &x)
#define mp make_pair
#define pb push_back
#define fr first
#define sc second
#define MAXX 100005
#define MOD 1000000007

using namespace std;

lli dpp[MAXX][10][2],a,d;

lli func(lli a, lli b, lli c)
{
	lli i,j,k,ans,temp;
	
	if (a < 0)
		return 0;
		
	if (a == 0)
		return c;
	
//	cout << a << " " << b << " " << dpp[a][b] << endl;
	if (dpp[a][b][c] != -1)
		return dpp[a][b][c];

	ans = 0;
	
	if (b == 0) {
		for (i = 1; i <= 9; ++i) {
			ans = ans + func(a-i, i,0);
			
			if (ans >= MOD)
				ans = ans % MOD;
		}
	} else {
		for (i = b; i <= min(b+d, (lli)9); ++i) {
			ans = ans + func(a-i, i, (c||(i == (b+d) ? 1 : 0)));
			
			if (ans >= MOD)
				ans = ans % MOD;
		}
	}
	
	//cout << a << " " <<b <<  " " << ans << endl; 
	dpp[a][b][c] = ans;
	
	return ans;
}


int main()
{
    lli tcase,n,i,j,k,tt;
    
    s(tcase);
    tt = 1;
    
    while (tcase--) {
    	printf("Case #%lld: ", tt);
    	++tt;
    	
    	s(n);
    	s(d);
    	
    	memset(dpp, -1, sizeof(dpp));
    	
    	k = func(n,0,0);
    	
    	if ((n <= 9) && (d == 0))
    		++k;
    	printf("%lld\n", k);
    }
    return 0;
}
