#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	int T, N, K, A[1000001], B[1000001];

	cin >> T;

	while (T--)
	{
		cin >> N >> K;
		
		for (int i = 0; i < N; i++)
			cin >> A[i];

		sort(A, A+N);

		for (int i = 0; i < N/2; i++)
			B[i] = A[i] + A[N-i-1];

		int ans = *max_element(B, B+N/2) - *min_element(B, B+N/2);;
		
		cout << ans << endl;
		
		if (ans > K)
			cout << "No more girlfriends!";
		else if (ans < K)
			cout << "Chick magnet Jhool!";
 		else
			cout << "Lucky chap!";
		cout << endl;
	}


	return 0;

}
