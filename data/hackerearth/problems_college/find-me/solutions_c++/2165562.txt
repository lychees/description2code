#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int n,p,cnt=0,flag=0;
    	cin>>n>>p;
    	long int a[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		if(a[i]>=p)
    		cnt++;
    		if(a[i]==p)
    		flag=1;
    	}
    	if(flag==0)
    	cout<<"NO 0"<<"\n";
    	else
    	cout<<"YES "<<cnt-1<<"\n";
    }
}
