#include<bits/stdc++.h>
#define ll long long
using namespace std;
long long func(ll int a,ll int b,ll int c,ll int d,ll int m)
{
	long long val = a*pow(m,3)+b*pow(m,2)+c*pow(m,1) + d;
	return val;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll int a,b,c,d;
		unsigned ll int k;
		cin>>a>>b>>c>>d>>k;
		ll int l = 0,r = 1000000,mid;
		ll int min = 0;
		while(l<=r)
		{
			 mid = (l+r)/2;
			if(func(a,b,c,d,mid)<=k)
			{
				min = mid;
				l = mid+1;
			}
			else
			{
				r = mid-1;
			}
		}
	cout<<min<<"\n";	
	}
}
