#include<bits/stdc++.h>
using namespace std;

#define LL long long
#define pb push_back

int aa[100001];

int main()
{
	for(int i=0; i<=100000; ++i)
		aa[i] = 1;

	vector<int > primes;

	for(int i=2; i*i<=100000; ++i)
	{
		if(aa[i])
		{
			for(int j=2*i; j<=100000; j+=i)
			{
				aa[j] = 0;
			}
		}
	}

	for(int i=2; i<=100000; ++i)
	{
		if(aa[i])
			primes.pb(i);
	}

	int n;
	cin >> n;

	int a[n+1];

	for(int i=1; i<=n; ++i)
		cin >> a[i];


	for(int i=1; i<=n; ++i)
	{
		if(i==1)
			cout << a[i] << " ";

		else if(i==2)
		{
			cout << a[2] + a[1] << " ";
		}

		else if(i==3)
		{
			cout << a[3] + a[1] << " ";
		}

		else
		{
			LL sum = 0, max = 0, j=0;
			vector<int> fac, vt;

			int nn = i;
			while(nn!=1)
			{
				if(nn%primes[j]==0)
					{
						vt.pb(primes[j]);
						nn /= primes[j];
					}

				else
					++j;
			}


		
			sort(vt.begin(), vt.end());
			do
			{
					sum = 0;
				fac.clear();
				fac.pb(1);
				
				for(int k=0; k<vt.size(); ++k)
				{
					LL last = fac.size()-1;
					fac.pb(fac[last]*vt[k]);
				}

				for(int k = 0; k<fac.size(); ++k)
					sum += a[fac[k]];

				if(sum>max)
					max = sum;

			}
			while(next_permutation(vt.begin(), vt.end()));

			cout << max << " ";
		}
	}



	return 0;
}
