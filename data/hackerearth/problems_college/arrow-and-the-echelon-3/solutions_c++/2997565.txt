#include<cstdio>
using namespace std;
int a[100001];
void comp()
{
	int x,j,y,fg;
	for(int i=1;i<=100000;i++)
	{
		x=i/3;
		fg=0;
		for(j=x;j>=0;j--)
		{
			y=i-j*3;
			if(y%5==0)
			{
				fg=1;
				break;
			}
		}
		if(fg==1)
			a[i]=j*3;
		else
			a[i]=-1;
	}
}
int main()
{
	comp();
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int x=a[n];
		if(x==-1)
			printf("-1\n");
		else
		{
			for(int i=1;i<=x;i++)
				printf("5");
			for(int i=1;i<=n-x;i++)
				printf("3");
			printf("\n");
		}
	}
	return 0;
}
