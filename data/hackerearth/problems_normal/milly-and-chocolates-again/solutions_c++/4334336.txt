#include <iostream>
#include <map>
using namespace std;

int main()
{
	int N,Q,C, tmp;
	int I[100000];
	map<int, int> index;
	cin >> N;
	tmp=1;
	for (int i=0; i<N; i++)
	{
		cin >> C;
		for (int j = tmp; j < C+tmp; j++)
        {
            index[j] = i+1;
        }
        tmp = C + tmp;
	}
	cin >> Q;
	for (int i=0; i<Q; i++)
	{
		cin >> I[i];
		cout << index[I[i]]<<endl;
	}
	
	return 0;
}
