#include <iostream>
using namespace std;
#include <bits/stdc++.h>
#define LL long long
const int maxS = 1000;
double Comb[maxS+1][maxS+1];
void binomial() {
	memset(Comb, 0.0,sizeof(Comb));
	for(int i = 0; i <= maxS; i++)
	{	
		Comb[i][0] = Comb[i][i] = 1.0;
		for(int j = 1; j <i; j++)
			Comb[i][j] = Comb[i-1][j] + Comb[i-1][j-1];
	}
}
int main()
{
    binomial();
    int t,n,s,m,k;
    cin >> t;
    while(t--) {
    	cin >> s >> n >> m >> k;
    	double den = Comb[s][n];
    	double num = 0.0;
    	for(int i = k;i<= min(m,n);i++) {
    		num += Comb[m][i]*Comb[s-m][n-i];
    	}
    	printf("%.6lf\n",num/den);
    }
    return 0;
}
