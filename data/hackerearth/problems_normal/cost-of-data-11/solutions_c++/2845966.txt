#include <cstdio>

struct trie {
	trie *next[26];
};

int nodes = 0;

inline trie *newNode() {
	nodes++;
	trie *t = new trie;
	for (int i = 0; i < 26; i++) t->next[i] = NULL;
	return t;
}

inline void insert(trie *tr, char *word) {
	for (int i = 0; word[i]; i++) {
		short cur = word[i]-'a';
		if (tr->next[cur] == NULL) tr->next[cur] = newNode();
		tr = tr->next[cur];
	}
}

int main() {
    char word[35];
    int n;
    trie *tr = newNode();
    
    scanf("%d ", &n);
    
    while (n--) {
    	gets(word);
    	insert(tr, word);
    }
    
    printf("%d\n", nodes);
    
    return 0;
}
