#include<iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int x[n+10];
        int y[n+10];
        for(int i=0; i<n; i++)
        {
            cin>>x[i]>>y[i];
        }
        int maxx=x[0];
        int minx=x[0];
        int maxy=y[0];
        int miny=y[0];
        for(int i=0; i<n; i++)
        {
            if(minx>x[i])
            minx=x[i];
            if(maxx<x[i])
            maxx=x[i];
            if(miny>y[i])
            miny=y[i];
            if(maxy<y[i])
            maxy=y[i];
        }
        int temp1=maxx-minx;
        int temp2=maxy-miny;
        if(temp1==0)
        {
        	cout<<temp2<<endl;
        }
        else if(temp2==0)
        {
        	cout<<temp1<<endl;
        }
        else
        {
        int temp=temp1*temp2;
        cout<<temp<<endl;
        }
    }
    return 0;
}
