#include <iostream>
using namespace std;
#define M 1000000007

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int len;
    	cin>>len;
    	
    	string s;
        cin>>s;
        
       long long rem=0;
       for(int i=0;i<len;i++)
       {
           int c=(int)(s[i]-'0');
           rem=(rem*10 +c)%M;
       }
        
        printf("%lld\n",rem);
    }
    return 0;
}
