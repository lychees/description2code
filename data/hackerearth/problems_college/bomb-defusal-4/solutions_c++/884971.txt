#include<iostream>
#include<cmath>
#define INF 1<<30
using namespace std;
main() {
    int t;
    cin>>t;
    while(t--) {
        int p,x,y,u,v;
        cin>>p>>x>>y>>u>>v;
        int a,b,tm;
        a=u-x;
        b=v-y;
        if(!a&&!b)
            tm=0;
        else if(fabs(a)==fabs(b))
            tm=1;
        else if((int)(fabs(a-b))%2==0)
            tm=2;
        else
            tm=INF;
        if(tm<p)
            cout<<"Counter-Terrorists Win !"<<endl;
        else
            cout<<"Terrorists Win !"<<endl;
    }
    return 0;
}
