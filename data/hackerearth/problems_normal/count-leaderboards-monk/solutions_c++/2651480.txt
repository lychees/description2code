#include <bits/stdc++.h>
#define MOD 1000000007
typedef long long ll;

ll fac[2000];
ll dp[1100][1100];
ll dp1[1100];

using namespace std;
ll num, k;

ll  C(ll n, ll r) {
	if (n < 0 || r < 0)
		return 0;
	if (r == 0 || n == r)
		return 1;
	if (dp[n][r] != -1)
		return dp[n][r];
	ll ans = ( C(n-1, r) + C(n-1, r-1) ) % MOD;
	dp[n][r] = ans;
	return ans;
}

ll f(ll n) {
	if (n == 0)
		return 1;
	if (n == 1)
		return 1;
	if (dp1[n] != -1)
		return dp1[n];
	ll ans = 0;
	for (ll i = 0; i < n; i++)
		ans = (ans + ( (f(i)  * fac[n-i])%MOD * C(n-1, n-i-1) ) % MOD)%MOD;
	dp1[n] = ans;
	return ans;
}
int main() {
	memset(dp ,-1, sizeof(dp));
	memset(dp1, -1, sizeof(dp1));
	
	fac[0] = 1;
	for (int i = 1; i <= 1100; i++)
		fac[i] = (fac[i-1] * i ) % MOD;
	
	scanf ("%lld%lld",&num, &k);
	ll a1 = f(num);
	//cout <<a1 << endl;
	ll ans = (a1 * C(num,k)) % MOD;
	printf ("%lld\n", ans);
	return 0;
}
