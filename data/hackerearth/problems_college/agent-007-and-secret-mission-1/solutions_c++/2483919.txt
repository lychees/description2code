#include <iostream>
#include<string.h>
void calc();
void calc2();
using namespace std;
int i,n,s,sum,test,j;	
char str[100000];
int day;
int main() 
{

	
	cin>>test;
	for(j=0;j<test;j++)
	{
	cin>>str;
	n=strlen(str);
	cin>>day;
	s=(day)*(day+1);
	sum=s/2;
	//printf("%d",sum);
	if(sum%2==0)
	{
    calc2();
	}
	else
	{
	   calc();
	}
	
	}
	
	

    
	return 0;
}

	void calc()
{
    int count0=0;
for(i=0;i<n;i++)
    {
        if(str[i]=='0')
        {
            count0++;
        }
        
    }
    cout<<count0<<"\n";
}

void calc2()
{
    int count1=0;
for(i=0;i<n;i++)
    {
        if(str[i]=='1')
        {
            count1++;
        }
        
    }
    cout<<count1<<"\n";
}
