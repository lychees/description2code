#include "bits/stdc++.h"
using namespace std;
int dp[5001];
#define ll  long long
int main(int argc, char const *argv[])
{
    int n;
    cin >> n;
    ll a[n+1];
    for(int i =0; i < n; i++){
        cin >> a[i];
        dp[i] = 1;
    }
    for(int i = 0; i < n; i++){
        for(int j = i+1; j < n; j++){
        if(abs(a[j]) > abs(a[i])){
            if(a[j]*a[i] < 0){
                dp[j] = max(dp[i]  + 1, dp[j]);
            }
        }
    }

    }
    int max1 = 0;
    for(int i  =0; i  < n; i++)max1=max(max1, dp[i]);
    cout << max1;
    return 0;
}
