def LongestVacation(N,shirt):
    maximum =0
    start =1
    end =1
    colorset=set()
    localStart=1
    for i in range(0,N):
        if shirt[i] not in colorset:
            colorset.add(shirt[i])
        else:
            if (i+1)-localStart>maximum:
                maximum = i+1-localStart
                end =i
                start = localStart
            index = localStart-1+shirt[localStart-1:i].index(shirt[i])          
            colorset = set(shirt[index+1:i+1])
            localStart =index+2
    if (N-localStart+1>maximum):
        end = N
        start = localStart
    print '{0} {1}'.format(start,end)

T = int(raw_input())
for _ in range(0,T):
    N = int(raw_input())
    listofShirtColor = list(map(int,raw_input().split()))
    LongestVacation(N,listofShirtColor)
