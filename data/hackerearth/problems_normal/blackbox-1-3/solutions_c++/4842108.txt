#include <iostream>
using namespace std;

int gcd(int a, int b)
{
    if (a == 0) 
       return b;
     
    return gcd(b%a, a);
}

long long pythog(long long a,long long b)
{
	return (a*a) + (b*b);
}
int main()
{	
	bool pytho = false;
	long long t,a,b,c;
	cin>>t;
	
	while(t--){
		pytho = false;
		cin>>a>>b>>c;
		
		if( a*a == pythog(b,c))
			pytho = true;
	
		if( b*b == pythog(c,a))
			pytho = true;
	
		if( c*c == pythog(a,b))
			pytho = true;
	

		if(pytho && gcd(gcd(a,b),c) == 1 )
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
	
    return 0;
}
