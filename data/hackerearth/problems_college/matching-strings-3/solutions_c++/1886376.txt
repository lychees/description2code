typedef long long int ll;

string a, b;
int t;
int main(){
//    ios_base::sync_with_stdio(0);cin.tie(0);
//    freopen("data.in","r",stdin);
    cin >> t;
    while(t--){
        cin >> a >> b;
        if(a.size() < b.size() )swap(a, b);
        int sza = a.size();
        int szb = b.size();
        int ans = 0;
        for(int i = 0; i <= sza - szb; i++){
            int c = 0;
            for(int j = 0; j < min(sza, szb); j++){
                if(a[j] == b[j]) c++;
            }
            ans = max(ans, c);
        }
        cout << ans << endl;
    }
    return 0;
}
