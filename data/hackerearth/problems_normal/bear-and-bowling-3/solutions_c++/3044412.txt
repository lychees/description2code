#include<bits/stdc++.h>
using namespace std;
long long mod=1e9+7;
long long pwr(long long a,long long b)
{
	if(b==0)return 1;
	long long temp=pwr(a,b/2);
	temp=(temp*temp)%mod;
	if(b%2)temp=(temp*a)%mod;
	return temp;
}
main()
{
	int n,i,a[200020];
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	scanf("%d",&a[i]);
	if(n==1)
	{
		printf("%lld\n",(a[1])%mod);
	}
	else
	{
	long long sum=0,fac=pwr(2,n-2);
	for(i=1;i<=n;i++)
	sum=(sum+(1ll*a[i]*(i+1))%mod)%mod;
	sum=(sum*fac)%mod;
	printf("%lld\n",sum);
	}
}
