#include <iostream>
#include<stdio.h>
#include<string>
using namespace std;

int ascii(char ch)
{
	int a;
	a = int(ch);
	a=a-32;
	//cout<<char(a)<<endl;
	return a;
}

int main()
{
    string s;
    getline(cin,s);
    int i,j,k,a;
    j = s.length();
    if(s[0]>='a' && s[0]<='z')
     {
         a=ascii(s[0]);
        s[0]=char(a);

     }
    for(i=0;i<j-1;i++)
    {

    	if(s[i]==' ' && s[i+1]!=' ')
    	{
			if(s[i+1]>='a' && s[i+1]<='z')
			{
					a = ascii(s[i+1]);
            		s[i+1] = char(a);
			}
    	
    	}

    }
    cout<<s<<endl;
    return 0;
}
