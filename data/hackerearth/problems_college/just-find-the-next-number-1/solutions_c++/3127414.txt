#include <bits/stdc++.h>

using namespace std;

bool is_ans(int num)
{
	int pre_bit=num&1;
	num=num>>1;
	while(num>0)
	{
		int curr_bit=num&1;
		if(curr_bit==1 && pre_bit==1)
			return false;
		num=num>>1;
		pre_bit=curr_bit;
	}
	return true;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	if(!is_ans(n))
    	{
    		for(int i=n+1;;i++)
    		{
    			if(is_ans(i))
    			{
    				cout<<i<<"\n";
    				break;
    			}
    		}
    	}
    	else
    	{
    		cout<<n<<"\n";
    	}
    }
    
    return 0;
}
