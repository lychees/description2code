#include <iostream>
using namespace std;

int main()
{
	int t,n;
	cin>>t;
	
	while(t--){
		cin>>n;
		
		int a[n];
		int profit=0;
		for(int i=0;i<n;i++){
			cin>>a[i];
		}
		
		int maxidx=0;
		
		while(true){
			if(maxidx==n)
				break;
			
			int start=maxidx;
			int max=a[maxidx];
			
			for(int i=maxidx;i<n;i++){
				if(a[i]>max){
					max=a[i];
					maxidx=i;
				}
			}
			
			if(start==maxidx)
				break;
				
			int buy=0;
			
			for(int i=start;i<maxidx;i++){
				buy=buy+a[i];	
			}
			
			profit+=(a[maxidx]*(maxidx-start))-buy;
			
			maxidx++;
		}
		
		cout<<profit<<"\n";
	}
    return 0;
}
