#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,d;
    	cin>>n>>d;
    	if(n>=d)
    	cout<<"Yes"<<endl;
    	else
    	{
    		int i=0;
    		while(n>0)
    		{
    			n--;
    			d-=2;
    			i++;
    			if(i==3)
    			{
    				n+=2;
    				i=0;
    			}
    		}
   			if(d<=0)
   			cout<<"Yes"<<endl;
   			else
   			cout<<"No"<<endl;
    	}
    }
    return 0;
}
