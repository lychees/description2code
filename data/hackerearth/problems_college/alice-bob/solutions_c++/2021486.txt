const double esp = 1e-5;

#define N 110

int gcd(int a, int b)
{
	return (a == 0) ? b : gcd(b % a, a);
}

int main(int argc, char *argv[])
{
	int n, num;
	while (scanf("%d", &n) != EOF) {
		int mx = -INF;
		int tmp = 0;
		for (int i = 0; i < n; ++i) {
			scanf("%d", &num);
			tmp = gcd(tmp, num);
			mx = max(mx, num);
		}
		if ((mx / tmp - n) % 2 == 1) {
			printf("Alice\n");
		} else {
			printf("Bob\n");
		}
	}

	return 0;
}
