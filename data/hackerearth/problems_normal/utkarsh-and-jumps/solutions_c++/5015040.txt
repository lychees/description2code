#include <iostream>
using namespace std;

int main()
{
  long long int n,i;
  int p;
  double p1,p2;
   
   
   cin>>n>>p;
   p1=(double)p/100;
   p2=1-p1;
  double a[n+1];
   a[0]=0;
   a[1]=0;
   a[2]=p1;
   a[3]=p2;
   for(i=4;i<=n;i++)
   {
   	a[i]=a[i-2]*p1+p2*a[i-3];
   }
   
   printf("%.6f",a[n]);
   
    return 0;
}
