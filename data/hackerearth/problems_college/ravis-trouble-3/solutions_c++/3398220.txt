#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int

int main()
{
    ll n, t, x, i, k, cnt, f, sl, pl;
    cin >> t;
    string s, p;
    while (t--) {
        cin >> s;
        cin >> p;
        sl = s.length();
        pl = p.length();
        if(sl < pl) {
            cout << "NO\n";
            continue;
        }
        if((sl == pl) && s != p) {
            cout << "NO\n";
            continue;
        }
        f = 0;
        ll j = 0;
        for(i = 0; i < sl; i++) {
            if(s[i] == p[j]) {
                j++;
            }
            if(j == pl) {
                f = 1;
                break;
            }
        }
        if(!f) {
            cout << "NO\n";
        } else {
            cout << "YES\n";
        }
    }

    return 0;
}
