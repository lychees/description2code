#include <iostream>
using namespace std;
unsigned nChoosek( unsigned n, unsigned k )
    {
        if (k > n) return 0;
        if (k * 2 > n) /*return*/ k = n-k;  //remove the commented section
        if (k == 0) return 1;
     
        int result = n;
        for( int i = 2; i <= k; ++i ) {
            result *= (n-i+1);
            result /= i;
        }
        return result;
    }
     
    
     
    int main(void)
    {
    	long t,c,v,res,fact,i;
        cin >> t;
        
        while(t--)
        {
        fact=1;
        cin >>c>>v;
        
        res= nChoosek(7, c)*nChoosek(5, v);
        
       for (int i = 1; i <= c+v; i++)
     {
         fact = fact * i;
     }
        cout <<res*fact <<endl;
      }
        return 0;
    }
