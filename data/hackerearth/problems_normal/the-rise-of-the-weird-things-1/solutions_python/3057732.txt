evenAry = []
oddAry = []
raw_input();
numAry = map(int, raw_input().split())
for num in numAry:
	if num % 2 == 0:
		evenAry.append(num)
	else:
		oddAry.append(num)
evenAry.sort()
oddAry.sort()
print ' '.join(map(str, evenAry)), sum(evenAry), ' '.join(map(str, oddAry)), sum(oddAry)
