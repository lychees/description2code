count = int(raw_input())
numBubbly = 0

for i in range(count):
	word = raw_input()
	unmatched = []
	for char in word:
		if len(unmatched)>0 and unmatched[-1]==char:
			unmatched.pop()
		else:
			unmatched.append(char)
	if len(unmatched) == 0:
		numBubbly+=1

print numBubbly
