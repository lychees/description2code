#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<set>
using namespace std;
int a[30];
int dp[40];
int main() {
    a[0]=1;
    for(int i=1;i<30;i++) a[i]=a[i-1]*2;
    string str;
    set<char>s;
    dp[0]=0;
    dp[1]=1;
    for(int i=2;i<=40;i++) {
        dp[i]=2*dp[i-1]+1;
        dp[i]-=dp[i-2];
    }
    int t;
    scanf("%d",&t);
    while(t--) {
        cin>>str;
        for(int i=0;i<str.size();i++) s.insert(str[i]);
        int ans=0;
        int l=s.size();
        cout<<dp[l]<<endl;
        s.clear();
    }
    return 0;
}
