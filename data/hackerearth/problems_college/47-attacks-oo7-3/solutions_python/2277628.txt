t=int(input())
while(t!=0):
        ans=1
        n=int(input())
        if(n>120):
                print("UNABLE TO FIRE")
        else:
                for i in range(1,n+1):
                        if(i%2==1):
                                ans=ans*3
                        else:
                                ans=ans+2
                print(ans)
        t=t-1
