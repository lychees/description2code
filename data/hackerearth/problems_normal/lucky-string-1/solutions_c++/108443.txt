#include<stdio.h>
#include<iostream>
#include<map>
using namespace std;
long long n;
char num[100000001];
long long aa[64];
long long power()
{
    long long res=0,a=1,b=2;
    while(res+a*b<n)
    {
        res+=a*b;
        b<<=1;
        a++;
    }
    n-=res;
    return a;
}
void find_ser(int i,long long a,long long o)
{

    if(a==1)
    {
        if(o==1)num[i]='4';
        else num[i]='5';
        num[++i]='\0';
        return;
    }


    if(aa[a-1]>=o)
    {
        num[i]='4';
        find_ser(i+1,a-1,o);
    }
    else
    {
        num[i]='5';
        find_ser(i+1,a-1,o-aa[a-1]);
    }
    //printf("%c ",num[i-1]);
}

int main()
{
    int t,i;
    long long temp,s;
    scanf("%d",&t);
    aa[0]=1;
    for(i=1;i<64;i++)
        aa[i]=aa[i-1]<<1;

    while(t--)
    {
        scanf("%lld",&n);
        temp=power();
        s=n/temp+(n%temp==0?0:1);

        find_ser(0,temp,s);
       // printf("%s\n",num);
        n-=(s-1)*temp;

        if(num[n-1]=='4')
            printf("Hacker\n");
        else
            printf("Earth\n");
    }
    return 0;
}
