#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */ 
    int n;
    vector <int> p;
    cin>>n;
    for(int i=0;i<n;i++)
        {
        int temp;
        cin>>temp;
        p.push_back(temp);
    }
    vector <int>::iterator it;
    vector <int>::iterator it1;
    int tot=0;
    while(n>=3)
        {
        int max=-1;
        for(it=p.begin()+1;it!=p.end()-1;it++)
            {
            if((*(it-1))*(*(it+1))>max)
               {
                max=(*(it-1))*(*(it+1));
                it1=it;
            } 
        }
        tot=tot+max;
        p.erase(it1);
        n--;
    }
    cout<<tot<<"\n";
    return 0;
}
