#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int len,i,j,a_cost,b_cost,min;
    	long int p = 0;
    	cin>>a_cost;
    	cin>>b_cost;
    	if(a_cost>b_cost)
    	{
    		min = b_cost;
    	}
    	else
    	{
    		min = a_cost;
    	}
    	len = s.length();
    	i = 0;
    	j = len - 1;
    	while(i<len/2)
    	{
    		if((s[i]-s[j])==1 || (s[j]-s[i])==1)
    		{
    			cout<<"-1"<<endl;
    			break;
    		}
    		else if(s[i]=='/' && s[j]=='/')
    		{
    			p = p + 2*min;
    		}
    		else if(s[i]=='/')
    		{
    			if(s[j]=='b')
    			p = p+b_cost;
    			else
    			p = p+a_cost;
    		}
    		else if(s[j]=='/')
    		{
    			if(s[i]=='a')
    			p =p+a_cost;
    			else
    			p = p+b_cost;
    		}
    		else{
    			
    		}
    		i++;
    		j--;
    	}
    	if(i==len/2)
    	cout<<p<<endl;
    	
    }
}
