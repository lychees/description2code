#include <iostream>
#include <string>
#include <algorithm>
#include <vector>
#include <map>

using namespace std;
struct stringp{
	string s,sm;	
};
map<string, int> pos;

bool operator <(stringp s1, stringp s2){
	if(s1.sm < s2.sm)return true;
	if(s1.sm == s2.sm){
		if(pos[s1.s] < pos[s2.s])return true;
		else return false;
	}
	return false;
}

int main()
{	ios::sync_with_stdio(false);
	int t,n,k,m;
	cin >> t;
	string s;
	vector<stringp> strV;
	stringp temp;
	while(t--){
		cin >> n >> k >> m;
		strV.clear();
		pos.clear();
		for(int i=0; i<n; i++){
			cin >> s;
			pos[s] = i;
			temp.s = s;
			temp.sm = s.substr(0,m);
			strV.push_back(temp);
		}
		sort(strV.begin(), strV.end());
		cout << strV[k-1].s << "\n";
	}
    
    return 0;
}
