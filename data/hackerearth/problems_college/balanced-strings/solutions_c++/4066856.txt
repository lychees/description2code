#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string s;
    while(t--)
        {
    	cin>>s;
    	int count[256]={0};
    	for(int i=0;i<s.length();i++){
    		count[s[i]-'a']++;
    	}
    	sort(count,count+26);
    	if(count[0]==count[25]) cout<<"Yes\n";
    	else cout<<"No\n";

    }
    return 0;
}
