#include<bits/stdc++.h>
using namespace std;
#define MAX 10000000
int primes[MAX];
bool lp[MAX*10+5];
int pr;
void siever(void)
{
	int i,j;
	pr = 0;
	for(i=2;i<MAX*10+5;++i)
	{
		if(lp[i] == false)
		{
			primes[pr++] = i;
			for(j=i;j<MAX*10+5;j+=i)
			{
				lp[j] = true;
			}
		}
	}
}
int main()
{
    //cout << "Hello World!" << endl;
    siever();
    //printf("done\n");
    int t;
    cin>>t;
    while(t--)
    {
    	int l,r;
    	cin>>l>>r;
    	cout<<upper_bound(primes,primes+pr,r) - lower_bound(primes,primes+pr,l)<<endl;
    }
    return 0;
}
