#include <iostream>
using namespace std;

int main()
{
	int t,n,m1,m2,result,temp;
	t=n=m1=m2=result=temp=0;
	cin >> t;
	for(int i=0 ;i < t;i++)
	{
		cin >> n;
		cin >> m1;
		cin >> m2;
		temp= n%(m1+m2);
		if(temp==0)
		{
			cout << '0' << endl;
			continue;
		}
		else
		{
			if(temp<m1)
				cout << '0' << endl;
			else if(temp <m2)
			{
				if(m1>0&&(((int)(temp/m1))%2) == 0)
				{
					cout << '0' << endl;
				}
				else
					cout << '1' <<endl;
				
			}
			else
				cout << '1' <<endl;
		}
	}
    
    return 0;
}
