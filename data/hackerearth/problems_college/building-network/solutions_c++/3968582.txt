#include<iostream>
#include<cstdlib>
#include<cstdio>
using namespace std;
 
int main()
{
int n;
cin>>n;
while(n--)
{
int k;
cin>>k;
int a[k];
int i=0;
long long int sum=0;
long long int min;
//cout<<min<<"\n";
for(i=0;i<k;i++)
{
 
cin>>a[i];
sum+=a[i];
if(i==0)
min=a[i];
else if(min>a[i])
min=a[i];
}
long long int z=(k-2)*min;
sum+=z;
cout<<sum<<"\n";
}
return 0;
}
