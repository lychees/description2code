#include<iostream>
#include<stdio.h>

using namespace std ;

int main()
{
    int k ;
    int m ;
    int x ;
    int index ;
    int y[111]={0};
    scanf("%d %d", &k, &m);
    for (int p=0 ; p<k ; p++)
    {
        scanf("%d", &x);
        for (int q=0 ; q<x ; q++)
        {
            scanf("%d", &index);
            y[index] = 1 ;
        }
    }
    x=0 ;
    for (int i=1 ; i<=m ; i++)
    {
        if (y[i]==0)
            x=1;
    }
    printf("%s\n", (x==1)?"NO":"YES");
}
