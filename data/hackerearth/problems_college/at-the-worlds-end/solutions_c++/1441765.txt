#include <iostream>
using namespace std;

int main()
{
	int A[500][500];
    int minReserve[500][500];
    
    int T, M, N, i,j;
    
    scanf("%d\n",&T);

        for (int t = 0; t < T; t++) {
        	scanf("%d %d\n",&N, &M);
            
            for (int i = 0; i < N; i++) {
                for (int j = 0; j < M; j++) scanf("%d",&A[i][j]);
                scanf("\n");
            }

            minReserve[N-1][M-1] = 1;

            for (int j = M-2; j >= 0; j--) {
            	int t = minReserve[N-1][j+1] - A[N-1][j];
                minReserve[N-1][j] = (t>1) ? t : 1;
            }

            for (int i = N-2; i >=0; i--) {
            	int t = minReserve[i+1][M-1] - A[i][M-1];
            	minReserve[i][M-1] = (t>1) ? t : 1;
            	
                for (int j = M-2; j >=0; j--) {
                    t = (minReserve[i+1][j] > minReserve[i][j+1]) ?  
                    			minReserve[i][j+1] : minReserve[i+1][j];
                    t -= A[i][j];
                    minReserve[i][j] = (t>1) ? t : 1;
                }
            }

            printf("%d\n",minReserve[0][0]);
        }
        
    return 0;
}
