#include <iostream>
using namespace std;

int main()
{
    long t,a,b;
    long hash[11],res=0;
    cin>>t;
    for(long i=0;i<t;i++)
    {
    	cin>>a;
    	res=0;
    	for(long j=0;j<10;j++)
    	hash[j]=0;
    	for(long j=0;j<a;j++)
    	{
    		cin>>b;
    		hash[b%10]++;
    	}
    	for(long j=0;j<10;j++)
    	if(hash[j]>1)
    	res=res+hash[j]-1;
    	cout<<res<<endl;
    }
    return 0;
}
