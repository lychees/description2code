#include<bits/stdc++.h>
using namespace std;
#define l1 long long int
int main()
{
	ios::sync_with_stdio(false);
	l1 i,j,n,m,ans=0,k;
	cin>>n>>m;
	priority_queue<l1>p;
	for(i=1;i<=n;i++)
	{
		cin>>k;
		p.push(k);
	}
	for(i=1;i<=m;i++)
	{
		k=p.top();
		ans+=p.top();
		p.pop();
		p.push(k-1);
	}
	cout<<ans;
	
	
}
