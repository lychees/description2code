#include <iostream>
#include<iomanip>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	double a[10000][2],x=0,y=0;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	for(int j=0;j<2;j++)
    	{
    		cin>>a[i][j];
    		if(j==0)
    		x+=a[i][j];
    		else
    		y+=a[i][j];
    	
    	}
    	
    	cout<<"("<<fixed<<setprecision(2)<<x/n<<" , "<<fixed<<setprecision(2)<<(y/n)<<")"<<endl;
    	
    	
    }
    return 0;
}
