#include <bits/stdc++.h>
using namespace std;
int min(int a, int b)
{
	return (a>=b)?b:a;
}
int main()
{
   
   int t;
   cin>>t;
   
    while(t--)
   {
     int n,k,maxf=0;
     cin>>n>>k;
     
     int count[101]={0};
     int *a = new int[n];
     for(int i=0;i<n;i++)
      { 
      	cin>>a[i];
      	count[a[i]]++;
        if(count[a[i]]>maxf)
           maxf=count[a[i]];
      }
   
     if(maxf>=k)
     { cout<<0<<endl;
       continue;
     }
     
     int cal[100][100]={0};
     
     for(int i=0;i<n;i++) 
      {  
      	for(int j=0;j<n;j++)
      	   {
      	   	 cal[i][j]=(a[i]>a[j])?(a[i]-a[j])*3:(a[j]-a[i])*5;
           }
      }
   
  for(int i=0;i<n;i++)   
    sort(cal[i],cal[i]+n);
  
  long int msum=INT_MAX;
  for(int i=0;i<n;i++)
  {
  	 int c=count[a[i]];
  	 long int sum=0;
  	 for(int j=0;j<n;j++)
  	  {
  	  	if(cal[i][j]!=0)
  	    { sum+=cal[i][j];
  	      c++;
  	    }
       if(c>=k)
        break;
      }
     msum=min(sum,msum); 
  }
   
   cout<<msum<<endl;
 }   
   //cout << "Hello World!" << endl;
    return 0;
}
