#include <iostream>
#include<map>
using namespace std;
int main()
{
	int n,i,j;
	cin>>n;
	int ar[n+2];
	map<int,int> a;
	a.clear();
	for(i=0;i<n;i++)
	{
		int temp;
		cin>>temp;
		ar[i]=temp;
	}
	int ma=0;
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			int temp=ar[i]-ar[j];
			if(temp<0)
				temp=ar[j]-ar[i];
			a[temp]++;
			if(a[temp]>ma)
				ma=a[temp];
		}
	}
	if(ma%2)
	cout<<"Alice"<<endl;
	else
	cout<<"Bob"<<endl;
    return 0;
}
