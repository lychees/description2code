#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    double speed[n],consumed[n],fuel;
    double dist,min=INT_MAX,temp;
    for(int i=0;i<n;i++)
        cin>>speed[i];
    for(int i=0;i<n;i++)
        cin>>consumed[i];
    cin>>fuel;
    for(int i=0;i<n;i++)
    {
        temp = consumed[i]/speed[i];
        min = min<temp?min:temp;
    }
    //cout<<"\nMax="<<min<<endl;
    cout.setf(ios::fixed);
    cout.precision(3);
    dist=(int)((fuel/min)*1000);
    dist= dist/1000;
    //dist=fuel/min;
    cout<<dist;
    return 0;
}
