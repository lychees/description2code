#include <bits/stdc++.h>
   
using namespace std;

int main() {
    	int n;
    	scanf("%d",&n);
    	int a[n];
    	for(int i=0;i<n;i++) {
    		scanf("%d",&a[i]);
    	}
    	int sum=0; int maxx=0;
    	for(int i=0;i<n;i++) {
    		sum=sum+a[i];
    		if(maxx<sum)
    			maxx=sum;
    		if(sum<0)
    			sum=0;
    	}
    	printf("%d\n",maxx);
    	return 0;
    }
