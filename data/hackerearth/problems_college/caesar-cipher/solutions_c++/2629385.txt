#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int t,k;
    string s;
    scanf("%d",&t);
    while(t--){
    	scanf("%d",&k);
    	cin>>s;
    	char c;
    	int temp;
    	for(int i=0;i<s.size();i++){
    		k=k%26;
    		temp=(int)(s[i])+k;
    		if((int)(s[i])<=(int)('Z')){
    			if(temp<=(int)('Z')){
    				c=char((int)(s[i])+k);
    				printf("%c",c);
    			}
    			else{
    				temp=temp-(int)('Z');
    				c=char((int)('A')+temp-1);
    				printf("%c",c);
    			}
    		}
    		else if((int)(s[i])>=(int)('a')){
    			if(temp<=(int)('z')){
    				c=char((int)(s[i])+k);
    				printf("%c",c);
    			}
    			else{
    				temp=temp-(int)('z');
    				c=char((int)('a')+temp-1);
    				printf("%c",c);
    			}
    		}
    	}
    	printf("\n");
    }
    return 0;
}
