#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,k;
    	cin>>n>>k;
    	long long int a[n+1];
    	long long int maxx[10010]={0};
    	long long int dp[n+1];
    	dp[0]=0;
    	for(long long int i=1;i<=n;i++)
    	{
    		cin>>a[i];
    		dp[i]=max(dp[i-1],maxx[max(i-k-1,(long long int)0)]+a[i]);
    		maxx[i]=max(maxx[i],dp[i]);
    	}
    	cout<<dp[n]<<endl;

    }
    return 0;
}
