long long ab(long long a,long long b)
{
	if(a>b)return a-b;
	return b-a;
}
int main()
{
	vector<long long> arr1,arr2,arr3,arr4;
	long long n;
	long long sum=0;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		long long w;
		cin>>w;
		arr1.push_back(w);
		sum+=w;
		arr4.push_back(w);
	}
	long long average=sum/n;
	for(int i=0;i<n;i++)
	{
		arr1[i]=arr1[i]-average;//Normalizing values
	}

	long long g=0,first=arr1[0],median,ans=0;
	for(int i=0;i<n;i++)
	{
		g+=arr1[i];
		arr3.push_back(g-first);
		arr2.push_back(g);
	}
	sort(arr4.begin(),arr4.end());
	sort(arr2.begin(),arr2.end());
	//Deciding Median
	if(n&1)
	{
		median=arr2[n/2];
	}
	else
	{
		median=(arr2[n/2]+arr2[n/2-1])/2LL;
	}
	for(int i=0;i<n;i++)
	{
		ans=ans+ab(arr2[i],median);
	}
	cout<<ans<<endl;
	return 0;
}
