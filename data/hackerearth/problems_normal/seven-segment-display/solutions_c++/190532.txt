#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;

typedef long long ll;
char s[1000000];
int main()
{
	int hashmap[11]={6,2,5,5,4,5,6,3,7,6};
	int t;
	cin>>t;
	while(t--)
	{
		cin>>s;
		int l = strlen(s);
		long long ans=0;
		int x;
		for(int i=0;i<l;i++)
		{
			x=s[i]-'0';
			ans+=hashmap[x];

		}
            cout<<ans<<endl;
    }
	return 0;
}
