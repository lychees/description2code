#include <iostream>
#include <algorithm>
using namespace std;
bool cmp(int i, int j)
{
	return i>j;
}
int main()
{
    int A[25001], i, t, c, n, flag;
    scanf("%d", &t);
    while(t--)
    {
    	scanf("%d", &n);
    	for(i=0;i<n;i++)
    	scanf("%d", &A[i]);
    	c=0;flag=1;
    	while(c<n)
    	{
    		sort(A, A+n, cmp);
    		for(i=0;i<=c;i++)
    		{A[i]--;
    		if(A[i]<0)
    		{flag=0;break;}
    		}
    		if(flag==0)
    		break;
    		c++;
    	}
    	printf("%d\n", c);
    }
    return 0;
}
