#include <iostream>
using namespace std;

int main()
{
	int t,i;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	int tps,cts,ts;
    	cin>>tps;
    	cin>>cts;
    	cin>>ts;
    	int flag=0;
    	if(cts-ts>1||ts-cts>1)
    	{
    		flag=1;
    	}
    	if(flag==0)
    		cout<<"Chotu\n";
    	else
    		cout<<"Motu\n";
    }
    return 0;
}
