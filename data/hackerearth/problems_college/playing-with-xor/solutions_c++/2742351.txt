#include<bits/stdc++.h>
using namespace std;

int b[1050]={0};
void markMultiples(bool arr[], int a, int n)
{
    int i = 2, num;
    while ( (num = i*a) <= n )
    {
        arr[ num-1 ] = 1; // minus 1 because index starts from 0.
        ++i;
    }
}
 

void SieveOfEratosthenes(int n)
{
    
    if (n >= 2)
    {
        
        bool arr[n];
        memset(arr, 0, sizeof(arr));
 
        
        for (int i=1; i<n; ++i)
        {
            if ( arr[i] == 0 )
            {
                b[i+1]=1;
                markMultiples(arr, i+1, n);
            }
        }
    }
}

    int t;
