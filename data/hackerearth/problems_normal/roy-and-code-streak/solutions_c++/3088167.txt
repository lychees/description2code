#include <bits/stdc++.h>
using namespace std;
int a[1000006];
int main()
{
	int t,n,s,r;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int maxi=0;
		int cur=0;
		memset(a,0,sizeof a);
		for(int i=1;i<=n;i++)
		{
			scanf("%d%d",&s,&r);
			if(r==0)
			{
				cur=0;
				continue;
			}
			else
			{
				if(a[s]==1)
				continue;
				else
				{
					cur++;
					maxi=max(maxi,cur);
					a[s]=1;
				}
			}
		}
		cout<<maxi<<"\n";
		
	}
	return 0;
}
