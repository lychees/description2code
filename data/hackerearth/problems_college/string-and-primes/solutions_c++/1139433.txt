#include<iostream>
#include<string.h>
using namespace std;
long long int prime(char);
int main()
{
	long long int t;
	cin>>t;
	for(long long int k=1;k<=t;k++)
	{
		
string s,m="";
cin>>s;
long long int l=s.length();
char ch;
long long int i=0,c1=0;
while(i<l)
{
	ch=s.at(i);
	long long int y=prime(ch);
	//cout<<y<<endl;
	if(y==0)
	m=m+ch;
	else
	c1++;
i++;
}
//cout<<c1;
if(c1==l)
cout<<"NULL"<<endl;
else
cout<<m<<endl;
}}
long long int prime(char ch)
{
	long long int i=1,c=0;
	while(i<=ch)
	{
		if(ch%i==0)
		c++;
		i++;
	}
	if(c==2)
	{
		return ch;
	}
	else 
	return 0;

}
