ll ar[100005],br[100005],cr[100005],ans;
int main()
{
	ll t,n,x,y,l,r,w,z,a,b,q;
	inll(t);
	while(t--)
	{
		in2ll(n,q);
		for(int i=1;i<=n;i++)
			inll(ar[i]);
		br[n+1]=0;
		for(int i=n;i>=1;i--)
			br[i]=(br[i+1]+ar[i])%MOD;
		cr[n+1]=0;
		for(int i=n;i>=1;i--)
		{
			cr[i]=(cr[i+1]+br[i])%MOD;
			//printf("%lld ",cr[i]);
		}
		//printf("\n");
		while(q--)
		{
			in2ll(w,x);
			in2ll(y,z);
			l=x+y;
			r=x+z;
			ans=((w+y)*(br[l]-br[r+1]+MOD))%MOD;
			//printf("ans=%lld\n",ans);
			//printf("%lld %lld %lld %lld\n",cr[l+1],(r-l),cr[r+1],(r-l)*cr[r+1]);

			printf("%lld\n",ans);
		}
	}
	return 0;
}
