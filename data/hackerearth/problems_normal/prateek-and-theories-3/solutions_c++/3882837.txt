#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int n;
   	cin>>n;
   	int start[n],fini[n];
   	int i;
   	for(i=0;i<n;i++)
   	{
   		cin>>start[i]>>fini[i];
   	}
   	sort(start,start+n);
   	sort(fini,fini+n);
   	int count=0;
   	int maxi=0;
   	int j=0;
   	i=0;
   	
   	while(i<n&&j<n)
   	{
   		//cout<<i<<j<<endl;
   		if(start[i]<fini[j])
   		{
   			count++;
   			if(count>maxi)
   			maxi=count;
   			i++;
   		}
   		else {
   			count--;
   			j++;
   		}
   	}
   	
   	cout<<maxi<<endl;
   	
   }
    return 0;
}
