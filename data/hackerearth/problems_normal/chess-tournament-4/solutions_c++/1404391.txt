#include <iostream>
#include<math.h>
using namespace std;

int main()
{
   	int n;
	cin>>n;
	int total_row = pow(2,n);
	//cout<<"total_row" <<total_row<<endl;
	total_row = total_row;
	bool dp[total_row+1][ total_row+1];
	for(int i=2;i<=total_row;i++)
		for(int j=1;j<=i-1;j++)
		{
		cin>>dp[i][j]; 
		//cout<<"dp["<<i<<"]["<<j<<"]="<<dp[i][j]<<endl;
		}
	int no_of_round = n;
	
	int winner[total_row];
	for(int i=1;i<=total_row;i++)
	  		winner[i]=i;
	for(int round = 1;round<=no_of_round;round++)
		{ int numplayer = pow(2,round-1);
			for(int i=1, kk=1;i<(total_row/numplayer);i=i+2)
			{
			int pl1= winner[i];
			int pl2= winner[i+1];
			
			if(pl1>pl2)
			{
			if(dp[pl1][pl2]==0) winner[kk]=pl2;
			else winner[kk]=pl1;
			}
			else 
			{
			if(dp[pl2][pl1]==0) winner[kk]=pl1;
			else winner[kk]=pl2;		
			}
			//cout<<"pl1 ="<<pl1<<"pl2 =  "<<pl2<<"winner"<<winner[kk]<<endl;
		
			kk++;	
		}
	}
	cout<<winner[1]<<endl;
	
    return 0;
}
