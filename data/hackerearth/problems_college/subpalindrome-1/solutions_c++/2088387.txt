ll rec(int l, int r){
	if(r <= l)
		return 0;
	if(cache[l][r] != -1)
		return cache[l][r];

	rep(i,l,r){
		rep(j,i,r){
			if(s[i] == s[j]){
				ret += rec(i+1, j)+1;
			}
		}
	}
	cache[l][r]=ret;
	return ret;
}
