#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdio>
#include <cstring>

using namespace std;

int main()
{
	int count;
	int t;
	scanf("%d",&t);
	while(t--){
	    count=0;
	  char p[500]={};
	 char q[500]={};
		cin>>p;
		cin>>q;
		int n=strlen(p);
		for(int i=0;i<n;++i)
		{
		      if(p[i]==q[i])
		      ++count;
		      else 
		      continue;
		}
		 cout << count << endl;
	}
   
    return 0;
}
