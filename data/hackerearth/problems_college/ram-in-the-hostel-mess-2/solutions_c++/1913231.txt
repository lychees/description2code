#include <iostream>
using namespace std;
long long func(long long i, int prev, long long n)
{
	if(n==1 && i==1)
		return 2;
	if(i==1)
		return func(i+1,0,n)+func(i+1,1,n);
	if(i==n)
	{
		if(prev==0)
			return 2;
		else
			return 1;
	}
	if(prev==1)	//1 means P
	{
		if(n-i<2)
		{
			return 2;
		}
		return func(i+1,0,n);	//0 means i+1 is N
	}
	else
	{
		return func(i+1,0,n)+func(i+1,1,n);
	}
}
int main()
{
	//cout<<func(3,0,5)<<" "<<func(3,1,5)<<endl;
    //cout << "Hello World!" << endl;
    long long t;
    cin>>t;
    while(t--)
    {
    	long long n;
    	cin>>n;
    	long long ans = func(1,0,n);
    	cout<<ans*ans<<endl;
    }
    return 0;
}
