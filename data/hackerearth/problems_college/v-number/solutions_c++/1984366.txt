#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string s;
    while(t--)
    {
    	cin>>s;
    	long long sumf=0,sume=0;
    	for(int i=0 ; i<s.length()/2 ; ++i)
    	{
    		sumf+=s[i]-'0';
    		sume+=s[s.length()-i-1]-'0';
    	}
    	if(sumf==sume)
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    return 0;
}
