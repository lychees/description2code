#include<iostream>
#include<vector>
using namespace std;
const int MAX = 2000000;
int dg[107];
bool check(int n)
{
	int i = 0;
	while (n > 0)
	{
		i++;
		dg[i] = n % 10;
		n /= 10;
	}
	int l = 1, r = i;
	while (l <= r)
	{
		if (dg[l] != dg[r]) return false;
		l++;
		r--;
	}
	return true;
}
int main()
{
	vector<bool>prime(MAX + 3, true);
	for (int i = 2; i <= MAX; i++)
	{
		if (prime[i])
		{
			for (int j = 2 * i; j <= MAX; j += i)
			{
				prime[j] = false;
			}
		}
	}
	int n;
	cin >> n;
	for (int i = 1; i <= n; i++)
	{
		int x;
		cin >> x;
		for (int i = x;; i++)
		{
			if (prime[i] && check(i))
			{
				cout << i << endl;
				break;
			}
		}
	}
}
