#include <bits/stdc++.h>
using namespace std;
bool palin(string s){
	string t=s;
	reverse(t.begin(),t.end());
	return s==t;
}
int main()
{
    string s; cin>>s;
    int n=s.size(),ans=0;
    for(int i=0;i<n;i++){
    	for(int j=i+1;j<=n;j++){
    		if(palin(s.substr(i,j-i))) ans++;
    	}
    } cout<<ans<<endl;
    return 0;
}
