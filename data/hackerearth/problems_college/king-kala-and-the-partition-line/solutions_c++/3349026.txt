#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a,b,c;
    	cin>>a>>b>>c;
    	int x,y,s1=0,s2=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>x>>y;
    		if(a*x+b*y+c>0)
    		s1++;
    		else
    		if(a*x+b*y+c<0)
    		s2++;
    	}
    	if(s1>s2)
    	cout<<s1<<endl;
    	else
    	cout<<s2<<endl;
    }
    return 0;
}
