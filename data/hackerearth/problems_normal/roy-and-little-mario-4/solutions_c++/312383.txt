#include<iostream>
using namespace std;


int main()
{
	int t,n,i,mod=1000000007;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int A[n+1];
		A[0]=1;
		for(i=1;i<=n;i++)
		{
			A[i]=A[i-1];
			if(i>1) {
				A[i]=(A[i]+A[i-2])%mod;
			}
			if(i>2) {
				A[i]=(A[i]+A[i-3])%mod;
			}
		}
		cout<<A[n]<<endl;
	}
	return 0;
}
