#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	unsigned long long int n,m;
   	cin>>n>>m;
   	unsigned long long int d=n/m;unsigned long long int md=n%m;
   	unsigned long long int s=((m-1)*m*d)/2+(md*(md+1)/2);
   	cout<<s<<endl;
   }
    return 0;
}
