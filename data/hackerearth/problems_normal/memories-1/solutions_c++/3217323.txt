{

	boost;
	ll t,n,p,a,b,max,ans,i,k,j;
	bool vis[300][300];
	bool is;
	cin>>t;
	while(t--)
	{
		mem(vis, false);
		cin>>n>>p;
		while(p--)
		{
			cin>>a>>b;
			vis[a][b]=vis[b][a]=true;
		}
		max=(1<<n);
		ans=0;
		rep(i,1,max-1)
		{
			is=true;
			rep(j,0,n-1)
			{
				rep(k,0,n-1)
				{
					if(((1<<j)&i)&&((1<<k)&i))
					{
						if(vis[j+1][k+1])
						{
							is=false;
							break;
						}
					}
				}
				if(!is)
				break;
			}
			if(is)
			ans++;
		}
		cout<<ans<<endl;
	}
	return 0;
}
