#include<iostream>
using namespace std;
int main(){
	int n;
	cin>>n;
	string s[103];int h;
	for(int i=0;i<n;i++)cin>>s[i];
	int k=0;int l=0;
	while(k<n&&l==0){
		 h=k+1;
		while(h<n&&l==0){int y=0;
			int w=0;int r=s[h].size()-1;
		if(s[k].size()==s[h].size()){
			if(s[k][w]!=s[h][r])y=1;
			while(s[k][w]==s[h][r]&&w<s[k].size()&&r>=0&&y==0){
				w=w+1;r=r-1;if(s[k][w]!=s[h][r]&&w<s[k].size()&&r>0)y=1;
			}
		}
		else y=1;if(y==0){
		l=1;}h=h+1;
	}k=k+1;
	}
	int g=s[h-1].size();
	g=g/2;
	cout<<s[h-1].size()<<" "<<s[h-1][g]<<endl;
}
