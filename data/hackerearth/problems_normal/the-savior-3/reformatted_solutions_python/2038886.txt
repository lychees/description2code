t = input()
for _ in xrange(t):
    l = [0]*(10**5+1)
    n = input()
    odd, even = 0, 0
    num_list = map(int, raw_input().split())
    for element in num_list:
        l[element] += 1
        if element & 1:
            odd += 1
        else:
            even += 1
    ans = (n*(n-1))/2
    for element in l:
        ans -= ((element*(element-1))/2)
    ans -= (even*odd)
    if ans < 0:
        print 0
    else:
        print ans
