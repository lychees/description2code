using namespace std;
int t;
pair<double, double> p;
int main(){
	scanf("%d", &t);
	while (t--){
		scanf("%lf%lf", &p.first, &p.second);
		int n;
		scanf("%d", &n);
		vector<double> vv;
		for (int i = 0; i < n; i++){
			double a, b;
			scanf("%lf%lf", &a, &b);
			vv.push_back(atan2(b - p.second, a - p.first));
		}
		sort(vv.begin(), vv.end());
		if (vv.size() == 1){
			puts("0.000000");
			continue;
		}
		double maxt = vv[0] + 2 * M_PI - vv[vv.size() - 1];
		for (int i = 0; i < vv.size() - 1; i++){
			maxt = max(maxt, vv[i + 1] - vv[i]);
		}
		maxt = maxt*180.0 / M_PI;
		maxt *= 100.0;
		maxt /= 360.0;
		maxt = 100.0 - maxt;
		printf("%.6f\n", maxt);
	}
	return 0;
}
