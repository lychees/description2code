t = input()
while t > 0:
    t -= 1
    b, w = map(int, raw_input().split())
    x, y, z = map(int, raw_input().split())
    cb1 = b*x
    cb2 = y*b+z*b
    cw1 = w*y
    cw2 = w*(z+x)
    print min(cb1, cb2)+min(cw1, cw2)
