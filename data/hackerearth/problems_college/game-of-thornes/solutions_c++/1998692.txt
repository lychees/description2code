#include<bits/stdc++.h>
using namespace std;
int cnt[26];
int main()
{
	ios::sync_with_stdio(0);
	string s;
	cin>>s;
	for(int i=0;i<s.length();i++)
	{
		cnt[s[i]-97]++;
	}
	int c=0;
	for(int i=0;i<26;i++)
	{
		if(cnt[i]%2==1)
		c++;
	}
	
	if(c==0||c==1)
	cout<<"YES"<<endl;
	else
	cout<<"NO"<<endl;
	return 0;
}
