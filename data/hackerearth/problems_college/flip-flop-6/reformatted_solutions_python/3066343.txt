global de
de = 0


def out(s):
    n = len(s)
    s = list(s)
    l = n
    i = 0
    while (i < l-1):
        if s[i] == s[i+1]:
            del s[i+1]
            global de
            de += 1
            l -= 1
            continue
        else:
            i += 1
    print n-len(s)


t = input()
for i in xrange(t):
    s = raw_input()
    out(s)
    global de
    de = 0
