#include <bits/stdc++.h>
using namespace std;

void solve() {
	long long n,x,i;
	cin >> n;
	cin >> x;
	long long val[n + 5];
	long long coun = 0;
	for(i = 0;i < n;i++) {
		cin >> val[i];
		if(i > 0 && val[i] <= val[i- 1]) {
			long long rem = (val[i - 1] - val[i]) / x;
			rem++;
			coun += rem;
			val[i] = val[i] + (rem*x); 
		}
	}
	cout << coun << "\n";
}

int main() {
	long long t;
	cin >> t;
	for(int i = 0;i < t;i++) {
		solve();
	}
}
