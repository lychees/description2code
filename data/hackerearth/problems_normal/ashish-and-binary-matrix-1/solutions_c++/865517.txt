#include<bits/stdc++.h>
#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   scanf("%d",&t);
   set<string> s;
   while(t--)
   {
   	string str;
   	int n,m;
   	scanf("%d %d",&n,&m);
   	for(int i = 0 ; i < n ; i++)
   	{
   		cin>>str;
   		s.insert(str);
   	}
   	if(s.size()==n)
			printf("Yes\n");
		else
			printf("No\n");
		s.clear();
   	
   	
   }
    return 0;
}
