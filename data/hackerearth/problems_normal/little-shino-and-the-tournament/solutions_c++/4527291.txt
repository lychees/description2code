#include<stdio.h>
#include<string.h>
#include<iostream>
#include<algorithm>
#include<math.h>
#include<map>
#include<bits/stdc++.h>
#include<stdlib.h>
#include<vector>
using namespace std;
int cnt[100006];
int level[100005];
int main()
{
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	int n,q;
	cin>>n>>q;
	int i;
		vector<pair<int,int> > v;
	v.clear();
	for(i=0;i<n;i++)
	{
		int x;
		cin>>x;
		v.push_back(make_pair(x,i));
	}
	while(v.size()>1)
	{
		vector<pair<int,int> > v2;
		for(i=0;i<v.size();i=i+2)
		{
			if(i+1==v.size())
			{
				v2.push_back(v[i]);
				continue;
			}
			cnt[v[i].second]++;
			cnt[v[i+1].second]++;
			if(v[i].first>v[i+1].first)
			v2.push_back(v[i]);
			else
			v2.push_back(v[i+1]);
		}
		v=v2;
	}
	while(q--)
	{
		int y;
		cin>>y;
		--y;
		cout<< cnt[y]<<"\n";
	}
	return 0;
}
