#include<bits/stdc++.h>
using namespace std;
int main(){

    int t;
    cin >> t;
    while(t--){

         string x;
         cin >> x;

          int a = 0;
          int b = 0;

          for(int i = 0 ; i < x.size() ; ++i ){
             if(x[i]=='1')
              {
               a++;
               b--;
              }
              else
              {
              a--;
              b++;
              }
          }
              cout << abs(a) << endl;          
  
    }

}
