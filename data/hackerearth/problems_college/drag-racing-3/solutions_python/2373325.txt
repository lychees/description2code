winnerSet = set()
loserSet = set()

# Classifies winners and losers into different two sets
def laps(lines):
    while lines:
        race = raw_input()
        result = race.split(' ')
        winnerSet.add(result[0])
        loserSet.add(result[1])
        lines -= 1

# Identifies a winner.
def results():
    for loser in loserSet:
        if loser in winnerSet:
            winnerSet.remove(loser)    
            
n = input()
lines = 2**n-1           
laps(lines)
results()
print winnerSet.pop()
