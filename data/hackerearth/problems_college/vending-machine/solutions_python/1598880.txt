#print 'Hello World!'
import math
n = int(raw_input())
#row equivalent to number of consecutive persons
i = int(math.log(math.ceil(n*1.0/5),2))
index= int(math.ceil((n-(2**i -1)*5 )/(2**i*1.0)))

if index == 1:
	print "Sheldon"
elif index == 2:
	print "Leonard"
elif index == 3:
	print "Penny"
elif index == 4:
	print "Rajesh"
elif index == 5:
	print "Howard"
