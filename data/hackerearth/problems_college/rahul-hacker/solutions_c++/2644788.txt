#include<bits/stdc++.h>
using namespace std;
void reverse_num(int &a){
    int temp=a,n=0,num=0;
    while(temp>0){
       n=temp%10;
       temp/=10;
       num=num*10+n;
    }
    a=num;
}
int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        int a,b;
        scanf("%d%d",&a,&b);
        reverse_num(a);
        reverse_num(b);
        int ans=a+b;
        reverse_num(ans);
        printf("%d\n",ans);
    }
    return 0;
}
