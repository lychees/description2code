#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main()
{
    string s;
    cin >> s;
    vector<int> v(26, 0);
    int o=0, k=0, l=0;
    for(int i=0; i<s.length(); i++)
    {
    	char x=s.at(i);
    	v[int(x)-97]++;
    }
    for(int i=0; i<26; i++)
    {
    	//cout << v[i]<<"	"<<k<<endl;
    	if(v[i]%2!=0 && l==0){o=i;l++;}
    	else if(v[i]%2!=0 && l==1){k=2;break;}
    	v[i]=v[i]/2;
    }
    if(k==2){cout <<-1;}
    else{
    for(int i=0; i<26; i++)
    {
    	k=v[i];
    	//cout << k<<endl;
    	while(k>0){cout << char(i+97);k--;}
    }
    if(l==1){cout << char(o+97);k--;}
    for(int i=25; i>=0; i--)
    {
    	k=v[i];
    	while(k>0){cout << char(i+97);k--;}
    }}
    return 0;
}
