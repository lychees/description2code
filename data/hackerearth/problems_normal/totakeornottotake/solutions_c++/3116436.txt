#include <iostream>
#include <climits>
#include <cmath>

//https://www.hackerearth.com/problem/algorithm/totakeornottotake/description/

using namespace std;

int main()
{
	int t, n, i, j;
	long long int min, max;
	char c;
	cin>>t;
	while(t--)
	{
		min = max = 1;
		cin >> n;
		for(i=0;i<n;i++)
		{
			cin >> c;
			if( c=='N' )
			{
				if(abs(min) <= max)
					min = -1 * max;
				else
					max = -1 * min;
			}
			else if(c == '+')
			{
				cin >> j;
				max += j;
			}
			else if(c == '-')
			{
				cin >> j;
				min -= j;
			}
			else if(c == '*')
			{
				cin >> j;
				max *= j;
				min *= j;
			}
			else if(c == '/')
			{
				cin >> j;
				if(min == 1)
					min = min/j;
			}
			else
				cout << "invalid" << endl; 
		}
		cout << max << endl;
	}
	return 0;
}
