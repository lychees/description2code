#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

const int N = 100005;
int n, k, p;
long long a[N];

int main() {
    int ntest;
    cin >> ntest;
    while (ntest--) {
    	cin >> n >> k >> p;
    	for (int i = 0; i < n; ++i) cin >> a[i];
    	sort(a, a + n);
    	bool ok = false;
    	for (int i = 0; i < n; ++i) {
    		int ind = upper_bound(a, a + n, a[i] + 2 * p) - a;
    		if (ind - i >= k) ok = true;
    	}
    	if (ok) cout << "NO\n";
    	else cout << "YES\n";
    }
    return 0;
}
