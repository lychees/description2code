#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long int n,m,arr[1003],i,j,q,near;
    cin>>n>>m;
    for(i=0;i<n;i++)
     cin>>arr[i];
     sort(arr,arr+n);
    while(m--)
    {
    	i=0;j=n-1;
    	cin>>q;
    	near=arr[i]+arr[j];
    	while(i<n&&j>=0&&i<j)
    	{
    		if(arr[i]+arr[j]==q)
    		{
    			near=q;
    			break;
    		}
    		else if(arr[i]+arr[j]<q)
    		{
    			if(abs(q-near)>abs(arr[i]+arr[j]-q))
    			  near=arr[i]+arr[j];
    			i++;
    		}
    		else
    		{
    			if(abs(q-near)>abs(arr[i]+arr[j]-q))
    			  near=arr[i]+arr[j];
    			j--;
    		}
    	}
    	cout<<near<<"\n";
    }
    return 0;
}
