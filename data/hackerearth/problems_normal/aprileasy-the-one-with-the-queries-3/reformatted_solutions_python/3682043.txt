def is_subsequence(x, y):
    for i in x:
        if len(y) != 0 and i in y:
            y = y[y.index(i)+1:]
        else:
            return False
    return True


s = raw_input()
n = int(raw_input())
while (n > 0):
    n -= 1
    ss = raw_input()
    if is_subsequence(ss, s):
        print "Yes"
    else:
        print "No"
