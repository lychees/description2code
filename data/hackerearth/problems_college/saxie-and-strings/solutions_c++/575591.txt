#include <iostream>
#include <string.h>
using namespace std;
int main()
{
	int t,l;
	char s[100];
	cin>>t;
	while(t--)
	{
		cin>>s;
		l=strlen(s);
		bool flag=1;
		for(int i=0;i<(l/2);i++)
		{
			if(s[i]!=s[l-i-1])
				flag=0;
		}
		if(flag)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
    return 0;
}
