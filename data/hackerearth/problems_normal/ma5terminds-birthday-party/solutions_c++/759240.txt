#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
int vis[1000001]; int dp[1000001];
int main()
{vis[1]=1;
    for (int g=2;g*g<=1000000; g++)
    {
        if (vis[g]) continue;
        for (int y=g*g; y<=1000000; y+=g)
        {
            vis[y]=1;
        }
    }
    for (int g=1;g<=1000000; g++)
    {
        if (g==4)
        {
                dp[g]=dp[g-1]; continue;
        }
        if (vis[g])
        {
            dp[g]=dp[g-1]+1;
        }
        else dp[g]=dp[g-1];
    }
    int a;scanf("%d", &a);
    for (int g=0;g<a; g++)
    {
        int b,c;scanf("%d %d", &b, &c); printf("%d\n", dp[c]-dp[b-1]);
    }
    return 0;
}
