#include <iostream>
#include<string.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
  int n=0,max=0,min=0;cin>>n;
  string a;
  cin>>a;
  n=a.length();
  for(int i=0;i<n;i++)
  {
  	if(int(a[i])==119)
  	{
  		max+=2;
  	}
  	else max++;
  }
  int count=0;
  for(int i=0;i<n;i++)
  {
  	count=0;
  	if(((int)a[i]==118)||((int)a[i]==119))
  	{
  		while(((int)a[i]==118)||((int)a[i]==119))
  		{
  			if(a[i]=='v') 
  			 count++;
  		   else	if(a[i]=='w')
  		   count=count+2;
  		   if(i>=n)
  		   break;
  		   else
  		    i++;
  		}
  		if(count==1)
  		  min++;
  		else
  		 {
  		 	if(count%2==0)
  		 	  min+=count/2;
  		 	else
  		 	  min+=(count-1)/2+1;
  		 }
  		 i--;
  	}
  	else
  	 min++;
  }
  
   cout<<min<<" "<<max<<endl;
  return 0;
}
