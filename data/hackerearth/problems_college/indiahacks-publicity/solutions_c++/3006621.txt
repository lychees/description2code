#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int visited[100010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	memset(visited,0,sizeof(visited));
	ll int test,t;
	cin>>test;
	for(t=1;t<=test;t++)
	{
		ll int n,r;//n friends & r friendships
		cin>>n>>r;
		list<ll int> l[n];
		ll int i,a,b;
		for(i=0;i<r;i++)
		{
			cin>>a>>b;
			l[a].push_back(b);l[b].push_back(a);
		}
		ll int mc=0,temp1;
		for(i=0;i<n;i++)
		{
			if(visited[i]!=t)
			{
				mc++;
				stack<int> s;
				s.push(i);
				while(!s.empty())
				{
					temp1=s.top();
					s.pop();
					if(visited[temp1]!=t)
					{
						visited[temp1]=t;
						list<ll int>::iterator it;
						for(it=l[temp1].begin();it!=l[temp1].end();it++)
						{
							if(visited[*it]!=t)
							{
								s.push(*it);
							}
						}
					}
				}
			}
		}
		cout<<mc<<endl;
	}
	return 0;
}
