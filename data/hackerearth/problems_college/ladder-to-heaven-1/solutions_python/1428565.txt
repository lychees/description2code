di = {}
def calc(N) :
    sm = 0
    a = 0
    b = 1
    i = 0
    while(i <= N) :
        if i % 2 == 0 :
            sm += i
        i = a + b
        a = b
        b = i        
    return sm
T = int(raw_input())
for i in range(T) :
    N = int(raw_input())
    if di.has_key(str(N)) :
        result = di[str(N)]
    else :
        result = calc(N)
    di[str(N)] = result
    print result
