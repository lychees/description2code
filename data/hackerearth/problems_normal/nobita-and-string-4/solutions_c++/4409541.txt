#include <iostream>
#include <vector>
using namespace std;

void splitbyspace(string str, vector <string> &splittedarr) {

	string temp = "";
	for (int i = 0; i < str.size(); i++) {
		if (str[i] == ' ') {
			splittedarr.push_back(temp);
			temp = "";
		}
		else {
			temp += str[i];
		}
	}
	splittedarr.push_back(temp);
}
int main()
{
    int T;
    cin >> T;
    string str;
    vector <string> splittedarr;
    getline(cin, str);
    while(T--) {
    	getline(cin, str);
    	splitbyspace(str, splittedarr);
    	for (int i = splittedarr.size() - 1; i >= 0; i--) {
    		cout << splittedarr[i] << " ";
    	}
    	cout << endl;
    	splittedarr.clear();
    }
    return 0;
}
