m = int(raw_input())
l = raw_input().split()
n = int(raw_input())
r = 0
for i in range(0, n):
    q = raw_input().split()
    s = 0
    for x in l:
        try:
            q.index(x)
        except BaseException:
            break
        else:
            s += 1
    if s == m:
        r += 1
print r
