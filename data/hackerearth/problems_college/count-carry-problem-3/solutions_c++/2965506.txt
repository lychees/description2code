#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n1,n2,a1,a2,flag=0;
    	cin>>n1>>n2;
    	if(n1>n2)
    	{while(n1!=0)
    	{
    		a1=n1%10;
    		a2=n2%10;
    		n1=n1/10;
    		n2=n2/10;
    		if(a1+a2>=10)
    		{flag++;;n1=n1+1;}
    		
    	}
    		
    	}
    	else
    	{	while(n2!=0)
    		{a1=n1%10;
    		a2=n2%10;
    		n1=n1/10;
    		n2=n2/10;
    		if(a1+a2>=10)
    		{flag++;;n2=n2+1;}
    		}
    	}
    	if(flag==0)cout<<"No carry operation";
    	else if(flag==1)cout<<"1 carry operation";
    	else cout<<flag<<" carry operations";
    	cout<<endl;
    }
    return 0;
}
