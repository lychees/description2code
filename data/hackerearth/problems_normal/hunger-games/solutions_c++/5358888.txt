#include <bits/stdc++.h>
using namespace std;

typedef long long LL;
#define FOR(a,b,c) for(int a = b; a <= c; a++)
#define RFOR(a,b,c) for(int a = b; a >= c ; a--)
#define vector<long long int> v
#define pb push_back
#define all(v) v.begin(), v.end()

int main() {
	int n;
	cin >> n;
	int *a = new int[n + 1];
	FOR(i, 1, n) {
		cin >> a[i];
	}
	sort(a + 1, a + n + 1);
	int ans = 0;
	FOR(i, 1, n) {
		if(i + 2 > n)
			break;
		else {
			ans = max(ans, a[i + 2] - a[i]);
		}
	}
	cout << ans;
	return 0;
}
