#include <iostream>
using namespace std;

int main()
{	int t;
	cin>>t;
	while(t--)
	{
		char str[100];
		bool hash[27];
		for(int i=1;i<=26;i++)
			hash[i]=false;
		int n;
		bool flag=true;
		cin>>n;
		while(n--)
		{   
			cin>>str;
			if(hash[str[0]-96])
				flag=false;
			else
				hash[str[0]-96]=true;
		}
		if(flag)
			cout<<"YES\n";
		else
			cout<<"NO\n";
		
	}
    
    return 0;
}
