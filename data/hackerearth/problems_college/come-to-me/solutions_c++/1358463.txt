#include <iostream>
#include <map>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s;
    	cin>>s;
    	int l =s.length();
    	int max = 0;
    	int length=0,flag=0;
    	for(int i=0;i<l;i++){
    		int m[26]={0};
    		m[s[i]-'A']=1;
    		length =0;
    		int j;
    		for(j=i+1;j<l;j++){
    			if(m[s[j]-'A'] == 0){
    				m[s[j]-'A'] = 1;
    				length++;
    			}
    			else{
    				if(max<length){
    					max=length;
    				}
    				break;
    			}
    		}
    		if(j == l && max<length){
    			max = length;
    		}
    	}
    	cout<<max+1<<endl;
    }
    return 0;
}
