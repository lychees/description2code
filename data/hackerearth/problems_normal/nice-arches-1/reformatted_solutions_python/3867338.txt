n = int(raw_input())
lst = [raw_input() for x in range(n)]

count = 0
for i in range(n):
    mystr = lst[i]
    newlst = list()
    for mychar in mystr:
        if len(newlst) == 0 or newlst[len(newlst)-1] != mychar:
            newlst.append(mychar)
        else:
            newlst.pop()
    if len(newlst) == 0:
        count = count+1
print count
