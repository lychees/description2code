#include <bits/stdc++.h>
#define MAX 10001
#define MOD 1000000007
using namespace std;
typedef long long int lli;
typedef unsigned long long ulli;
int M, N, Index, Total = 0;
bool Arr[MAX];
int main()
{
    //freopen("1.txt", "r", stdin);
    //freopen("2.txt", "w", stdout);
    cin >> M;
    memset(Arr, false, sizeof Arr);
    for(int i = 0; i < M; i++)
    {
        cin >> Index;
        if(!Arr[Index])
            Arr[Index] = true;
    }
    cin >> N;
    getchar();
    while(N--)
    {
        char str[1000001];
        gets(str);
        int i = 0, Num = 0, Count = 0;
        bool Done[100001];
        memset(Done, false, sizeof Done);
        //cout << str << endl;
        while(str[i] != '\0')
        {
            char ch = str[i];
            if(ch == ' ')
            {
                if(!Done[Num])
                {
                    if(Arr[Num])
                        ++Count;
                    Done[Num] = true;
                }
                Num = 0;
            }
            else
            {
                Num = Num * 10 + (ch - '0');
            }
            ++i;
        }
        if(Num)
        {
            if(!Done[Num])
            {
                if(Arr[Num])
                    ++Count;
                Done[Num] = true;
            }
        }
        if(Count == M)
            ++Total;
    }
    cout << Total << endl;
    return 0;
}
