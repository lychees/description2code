#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n, k;
    cin>>n>>k;
    long int a[n];
    
    for(int i=0; i<n; i++){
    	cin>>a[i];
    }
    
    sort(a, a+n);
    int flag = 0;
    for(int i=1; i<n; i++){
    	
    	if( (a[i] - a[i-1]) > k ){
    		cout<<a[i-1];
    		flag = 1;
    		break;
    	}
    }
    if(flag == 0){
    	cout<<a[n-1];
    }
    
    return 0;
}
