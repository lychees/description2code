#include<bits/stdc++.h>
using namespace std;
int main()
{
    int test,i,j,num;
    cin>>test;
    while(test--)
    {
        string str1;
        cin>>num;
        cin>>str1;
        i=0;
        while(str1[i])
        {
            j=str1[i]-'A';
            j=j-num;
            if(j<0)
                j='Z'+j+1;
            else
                j=str1[i]-num;
            str1[i]=j;
            i++;
        }
        cout<<str1<<endl;
    }
}
