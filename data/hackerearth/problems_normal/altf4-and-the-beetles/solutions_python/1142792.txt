t = int(raw_input())
while (t):
	points = []
	n, a, b, c = map(int, raw_input().split())
	p1 = p2 = p3 = 0
	while (n):
		x, y = map(int, raw_input().split())
		points.append((x, 0))
		points.append((y, 1))
		p1 += a
		n -= 1
	res = p1
	points.sort()
	for i in points:
		pos = i[0]
		state = i[1]
		if (state == 0):
			p1 -= a
			p2 += b
		elif (state == 1):
			p2 -= b
			p3 += c
		res = max(res, p1+p2+p3)
	print(res)
	t -= 1
