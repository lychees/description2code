#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	string s;
	while (t > 0){
		cin >> s;
		map<char,int> mymap; 
		int count = 0;
		for (int i = 0; i < s.length(); i++){
			if (mymap.find(s[i]) == mymap.end()){
				mymap[s[i]] = i;
				count++;
			}
		} 
		long long int ans = (count*(count+1))/2; 
		cout << ans << endl;
		t--;
	}
    return 0;
}
