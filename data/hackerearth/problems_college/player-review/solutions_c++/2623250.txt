#include <iostream>
using namespace std;

int main()
{
    long long int n,i,sum=0,partialsum=0,a[1000000];
    scanf("%lld",&n);
    for(i=0;i<n;i++)
    {
    	scanf("%lld",&a[i]);
    }
    for(i=0;i<n;i++)
    {
    	partialsum = partialsum+a[i];
    	if(partialsum<0)
    	partialsum=0;
    	if(partialsum>sum)
    	sum=partialsum;
    }
    printf("%lld",sum);
    return 0;
}
