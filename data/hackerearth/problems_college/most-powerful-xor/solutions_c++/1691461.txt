#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	int l,r;
    	cin>>l>>r;
    	int ans=0;
    	for(int j=l;j<=r;j++)
    	{
    		for(int k=j;k<=r;k++)
    		{
    			int temp=j^k;
    			ans = (temp>ans?temp:ans);
    		}
    	}
    	cout<<ans<<endl;
    	
    }
    return 0;
}
