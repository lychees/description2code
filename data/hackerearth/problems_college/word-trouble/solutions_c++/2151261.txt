#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

int main()
{
	int t;
	cin >> t;
	
	while (t--) {
		char a[50], b[50];
		scanf("%s%s", a, b);
		
		if (strcmp(a, b) == 0) {
			puts("Exact");
		} else {
			int i = 0;
			int p = 0;
			int n = min(strlen(a), strlen(b));
			while (i < n) {
				if (a[i] == b[i]) 
				 ++p;
				++i;
			}
			printf("Non Exact %d\n", max(strlen(a), strlen(b)) - p);
		}
	}
    return 0;
}
