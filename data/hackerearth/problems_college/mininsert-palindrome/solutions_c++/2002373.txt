#include <iostream>
#include <string>
#include <cstring>

using namespace std;

int vis[50][50];
int dp[50][50];
string str;

int get (int st, int dr) {
	if (dr <= st)
		return 0;
	
	if (vis[st][dr])
		return dp[st][dr];
	vis[st][dr] = 1;
	
	if (str[st] == str[dr])
		dp[st][dr] = get(st + 1, dr - 1);
	else
		dp[st][dr] = 1 + min(get(st + 1, dr), get(st, dr - 1));

	return dp[st][dr];
}

int main()
{
	int t = 0;
	cin >> t;
	
	while (t--) {
		memset(vis, 0, sizeof(vis));
		memset(dp, 0, sizeof(vis));
		
		int n = 0;
		cin >> n;
		
		cin >> str;
		cout << get(0, str.size() - 1) << '\n';
	}
	
    return 0;
}
