#include<iostream> 
using namespace std;

int main(){
    
int x,y,t,i,j,result;
	cin>>t;
	for(int l=0;l<t;l++)
        {
		cin>>x>>y; result=0; 
                int a[1002][1002]={0};
                for(int i=0;i<x;i++)
                {
                    for(int j=0;j<y;j++)
                    {
                        if(a[i][j]==0)
                        { 
                            result++;
                            if(i+2<x && a[i+2][j]==0)
                            {
                                a[i+2][j]=1; 
                            }
                            if(j+2<y && a[i][j+2]==0)
                            {
                                 a[i][j+2]=1; 
                            }
                            
                        }
                    }
                }
                cout<<result<<endl;
		  
	} 
}
