#include <bits/stdc++.h>

using namespace std;
double PI = (2*acos(0.0));
int main()
{
	int t;
	double v,sol, r, h, ans;
	scanf("%d",&t);
	while(t--) {
		scanf("%lf",&v);
		sol = v/((double)2*PI);
	
		r = pow(sol, (double)1/(double)3);
		h = v/(r*r*PI);
		
		ans = (2*PI*r*(r+h));
		printf("%0.9lf\n",ans);
	}
	return 0;
}
