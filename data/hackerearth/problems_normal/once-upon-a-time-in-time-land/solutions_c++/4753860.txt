#include <iostream>
#define minus -1000000000
using namespace std;

int main()
{
	int t;
	long i,k,n;
	cin>>t;
	for(;t--;)
	{
		cin>>n>>k;
		long a[n],dp[n];
		for(i=1;i<=n;++i)
		{
			cin>>a[i];
		}
		dp[1]=a[1]>0?a[1]:0;
		for(i=2;i<=k+1;++i)
		{
			dp[i]=dp[i-1]>a[i]?dp[i-1]:a[i];
		}
		for(i=k+2;i<=n;++i)
		{
			dp[i]=max(dp[i-1],dp[i-k-1]+a[i]);
		}
		cout<<dp[n]<<endl;
	}
    return 0;
}
