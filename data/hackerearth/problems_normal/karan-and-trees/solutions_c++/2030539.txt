#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int a,b;
    	cin>>a>>b;
    	while(a>=1&&b>=1)
    	{
    		if(a==b)
    		{
    			cout<<a<<endl;
    			break;
    		}	
    		else if(a>b)
    		{
    			if(a%2==0)
    			{
    				a=a/2;
    			}
    			else
    			{
    				a=(a-1)/2;
    			}
    		}
    		else
    		{
    			if(b%2==0)
    			{
    				b=b/2;
    			}
    			else
    			{
    				b=(b-1)/2;
    			}
    		}
    	}
    }
}
