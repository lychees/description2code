#include <bits/stdc++.h>
using namespace std;

int n;

double x[3007];
double y[3007];
double f[3007];

double inf=10000000.0;

double dp[3007];

int main()
{
    inf*=inf;
    scanf("%d", &n);
    for (int i=1; i<=n; i++)
    {
        scanf("%lf%lf%lf", &x[i], &y[i], &f[i]);
        dp[i]=-inf;
    }
    dp[1]=f[1];
    for (int i=2; i<=n; i++)
    {
        for (int j=1; j<i; j++)
        {
            dp[i]=max(dp[i], dp[j]+f[i]-sqrt((x[j]-x[i])*(x[j]-x[i])+(y[j]-y[i])*(y[j]-y[i])));
        }
    }
    printf("%.6lf\n", dp[n]);
    return 0;
}
