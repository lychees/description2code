#include <iostream>
#include<stdio.h>
#include<vector>
using namespace std;
vector<int> v[100005];
int mark[100005];
void dfs(int cur)
{
	int i;
	if(mark[cur]==1)
		return;
	mark[cur]=1;
	for(i=0;i<v[cur].size();i++)
	{
		if(mark[v[cur][i]]==0)
			dfs(v[cur][i]);
	}
}
int main()
{
    int t,n,e,a,b,i,c;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    	{
    		mark[i]=0;
    		v[i].clear();
    	}
    	scanf("%d",&e);
    	while(e--)
    	{
    		scanf("%d %d",&a,&b);
    		v[a].push_back(b);
    		v[b].push_back(a);
    	}
    	c=0;
    	for(i=0;i<n;i++)
    	{
    		if(mark[i]==0)
    		{
    			c++;
    			dfs(i);
    		}
    	}
    	cout<<c<<endl;
    }
    return 0;
}
