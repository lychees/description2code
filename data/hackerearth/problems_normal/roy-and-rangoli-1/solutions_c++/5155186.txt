#include <bits/stdc++.h>
using namespace std;
struct _ { ios_base::Init i; _() { cin.sync_with_stdio(0); cin.tie(0);cout.tie(0); } } _;

typedef long long int ll; 
vector<long long int>prime;
void sieve(long long int MAX)
{
    bool status[MAX+7];
    for(ll i=2; i<=MAX; i++)
    status[i]=true;
    for(ll i=4; i<=MAX; i+=2)
    status[i]=false;
    ll sqrtn=sqrt(MAX);
    for(ll i=3; i<=sqrtn; i+=2)
    if(status[i])
    for(ll j=i*i; j<=MAX; j+=(i+i))
    status[j]=false;
    for(ll i=2; i<=MAX; i++)
    if(status[i])
    prime.push_back(i);
}
#define rep(i,n) for(ll i=0;i<n;i++)
int main()
{
    ll n;
    cin>>n;
    n--;
    sieve(n+n+10);
    ll sum=0;

    ll temp=2*n;
    temp++;

    rep(i,prime.size())
    {
        if(prime[i]<=n+n)
        {
            ll g=prime[i];
            if(g>n)
            sum=sum+(temp-g);
            else
            sum=sum+g+1;
            sum=sum%1000000007;
        }
    }
    cout<<sum%1000000007<<endl;
}
