#include <iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	char s1[2006],s2[2006];
	while(t--)
	{
		long int aa,bb,ab,ba;
		long int i,j,k;
		i=j=0;
		cin>>aa>>bb>>ab>>ba;
		while(aa>=2)
		{
			s1[i++]='a';
			s1[i++]='a';
			s2[j++]='a';
			s2[j++]='a';
			aa-=2;
		}
		while((ab>0) && (ba>0))
		{
			s1[i++]='a';
			s1[i++]='b';
			s2[j++]='a';
			s2[j++]='b';
			ab-=1;
			ba-=1;
		}
		while(bb>=2)
		{
			s1[i++]='b';
			s1[i++]='b';
			s2[j++]='b';
			s2[j++]='b';
			bb-=2;
		}
		if(aa==1)
		{
			s1[i++]='a';
			s2[j++]='a';
		}
		else if(bb==1)
		{
			s1[i++]='b';
			s2[j++]='b';
		}
		for(k=0;k<i;k++)
			cout<<s1[k];
		for(i=(j-1);i>=0;i--)
			cout<<s2[i];
		cout<<endl;
	}
	return 0;
}
