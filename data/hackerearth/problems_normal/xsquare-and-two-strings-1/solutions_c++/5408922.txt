#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		string s1,s2;
		cin>>s1>>s2;
		int a[27];
		memset(a,0,sizeof(a));
		bool fl=false;
		for(int i=0;i<s1.length();i++)a[s1[i]-'a']++;
		for(int i=0;i<s2.length();i++){
			if(a[s2[i]-'a']>0){
				fl=true;
				break;
			}
		}
		cout<<(fl?"Yes":"No")<<endl;
		
	}
	return 0;
}
