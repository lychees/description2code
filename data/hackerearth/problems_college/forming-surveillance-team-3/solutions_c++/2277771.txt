#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[100010];
int main()
{
	memset(a,0,sizeof(a));
	int t,test;
	scanf("%d",&test);
	for(t=1;t<=test;t++)//Will use t for BFS
	{
		int n;
		scanf("%d",&n);
		int ans=0;
		int i,j;
		for(i=0;i<n;i++)
		{
			scanf("%d",&j);
			if(a[j]!=t)
			{
				ans++;
			}
			a[j]=t;
		}
		printf("%d\n",ans);
	}
	return 0;
}
