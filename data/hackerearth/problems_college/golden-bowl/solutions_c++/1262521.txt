#define rep(i,a,b) for(int i=a; i<b; i++)
#define sc(x) scanf("%d", &x)
#define lint long long
using namespace std;
lint GCD(lint u, lint v) {
    while ( v != 0) {
        lint r = u % v;
        u = v;
        v = r;
    }
    return u;
}
int main(){
    lint x;
    cin>>x;
    lint lo=1;
    lint hi=x;
    lint mid;
    while(lo<=hi){
        mid=(lo+hi)/2;
        lint g1=GCD(mid,x);
        lint g2=GCD(mid-1,x-1);
        lint n1=mid/g1;
        lint n2=(mid-1)/g2;
        lint d1 = x/g1;
        lint d2 = (x-1)/g2;
        g1 = GCD(n1,d2);
        n1/=g1;
        d2/=g1;
        g2 = GCD(n2,d1);
        n2/=g2;
        d1/=g2;
        double t1 = n1;
        double t2 = n2;
        t1/=d1;
        t2/=d2;
        if(t1*t2==0.5)break;
        t1=n1;
        t2=n2;
        t1/=d2;
        t2/=d1;
        if(t1*t2==0.5)break;
        if(t1*t2<0.5)lo=mid+1;
        else hi=mid-1;
    }
    cout<<mid<<endl;
}
