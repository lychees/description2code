T = int(raw_input())


def isPal(pal):
    for i in range(len(pal)-1):
        temp = pal[i:len(pal)]
        if (temp == temp[::-1]):
            return i
    return -1


for i in range(T):
    pal = raw_input()
    index = isPal(pal)
    if (index == -1):
        print len(pal)+len(pal)-1
    else:
        print len(pal)+index
