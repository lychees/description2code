#include<stdio.h>
#include<string.h>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int n,count=0;
    char a[128][128],ch1,ch2;int i,j;
    bzero(a,16384*sizeof(char));
    string array[100];
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
    	int temp=1;
    	cin>>array[i];
    	ch1=array[i][0];
    	ch2=array[i][array[i].length()-1];
    	if(a[ch1][ch2]==0)
    	{
    		a[ch1][ch2]=1;
    	}
    	else
    	{
    		sort(array[i].begin(),array[i].end());
    		for(j=0;j<i;j++)
    		{
    			
    			sort(array[j].begin(),array[j].end());
    			{
    				if (array[j].compare(array[i]) == 0)
    				temp=0;
    			}
    		}
    	}
    	count+=temp;
    }
   	printf("%d",count);
    return 0;
}
