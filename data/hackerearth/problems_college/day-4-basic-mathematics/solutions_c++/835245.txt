#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;

int main()
{
    int t,x;
    long long int n,ans;
    scanf("%d",&t);
    while(t--)
    {
        ans = 0;
        scanf("%lld",&n);
        double det = sqrt(1+8*n);
        det = det -1;
        det = det/2;
        ans = ceil(det);
        if(ans%3==0)
        {
            printf("SL\n");
        }
        else if(ans%3==1)
        {
            printf("LB\n");
        }
        else if(ans%3==2)
        {
            printf("BS\n");
        }
    }

    return 0;

}
