#include <bits/stdc++.h>
 
using namespace std;
 
int tests,ar[1<<20],res,n,c;
int mov[100][100];
int m2[11][11][11];
 
int main(){
	ios_base::sync_with_stdio(0);
	//cin.tie(0);
	
	for (int i=0;i<=41;i++)	
		for (int j=0;j<=10;j++)
			mov[i][j]=i*10+j,
			mov[i][j]%=41;
	
	for (int i=0;i<10;i++)
		for (int j=0;j<10;j++)
			for (int q=0;q<10;q++)
				m2[i][j][q]=(i*j+q)%10;
			
	cin>>tests;
	
	for (;tests;--tests)
	{
		cin>>ar[0]>>ar[1]>>c>>n;
		int res=0;
		res=ar[0]*10+ar[1];
		res%=41;
		
		if (n==1)
			res=ar[0];
			
		for (int i=2;i<n;i++)
		{
			ar[i]=m2[ar[i-1]][c][ar[i-2]];
			res=mov[res][ar[i]];
		}
		if (res)
			puts("NO");
		else
			puts("YES");
	}
	
	return 0;
}
