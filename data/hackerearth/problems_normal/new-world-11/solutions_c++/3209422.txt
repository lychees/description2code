			in arr[i];
		int low=0,mid,high= 1e9;
		int ans = 1e9;
		while(low<=high)
		{
			mid = (low+high)/2;
			// Tey for mid;
			int jumps = 0;
			int prev = 0;
			for(i=1;i<n && jumps<k;)
			{
				if(arr[i]-arr[prev]>mid)

					jumps++;
					prev = i-1;

				}

			}
