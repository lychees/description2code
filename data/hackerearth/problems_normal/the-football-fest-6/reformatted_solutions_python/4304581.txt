# man u
t = input()
for i in range(t):
    n, curr = map(int, raw_input().split())
    prev = None
    for j in range(n):
        pas = raw_input()
        if pas[0] == "P":
            prev = curr
            curr = int(pas[2:])
        else:
            curr, prev = prev, curr
    print "Player", curr
