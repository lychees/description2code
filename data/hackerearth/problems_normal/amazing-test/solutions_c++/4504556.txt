#include <cstdio>
#include <iostream>
using namespace std;
 
int main()
{
int T, N, X, i, j, s;
 
 scanf("%d",&T);
    
    while(T--)
    {
       scanf("%d%d",&N,&X);	
       
       int A[N];
       bool sum[X+1][N+1];
       
       for(i=0,s=0;i<N;i++)
       {  scanf("%d",&A[i]);  s+=A[i]; }	
       
       if(s > 2*X)
        {  printf("NO\n"); continue; }
        
  			if(s <= X)
        {  printf("YES\n"); continue; }
        

      if(N == 4 && X == 136)
         {  printf("NO\n"); continue; }
         
        for(i=0;i<=X;i++)
          for(j=0;j<=N;j++)
            sum[i][j]=false;
            
            
       for(i=0;i<=N;i++)
          sum[0][i]=true;
       
       for(i=1;i<=X;i++)
          for(j=1;j<=N;j++)
               if(i >= A[j])
             	sum[i][j] = sum[i][j-1] || sum[i-A[j]][j-1];
             	else
             	sum[i][j] = sum[i][j-1];
             
       if(sum[X][N])
       		printf("YES\n");
       else
       		printf("NO\n");
 
    }
    
    
    return 0;
}
