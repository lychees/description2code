n = input()
for _ in range(n):
    string = raw_input()
    revstring = string[::-1]
    count = 0
    done = False
    if(revstring!=string):
        for i in range(len(string)):
            if(ord(string[i])>ord(revstring[i]) and not done):
                count+= (ord(string[i])-ord(revstring[i]))
                str1 = list(string)
                str1[i] = revstring[i]
                string = ''.join(str1)
                if(revstring==string):
                    done = True
                    break

    print count
