#include <string>
#include <iostream>
using namespace std;

string a, b;
int n;

int main(){
    cin >> n;
    while (n--){
    	cin >> a >> b;
    	if ((a == b) || (a == "2" && b == "4") || (a == "4" && b == "2"))
    		cout << "YES" << endl;
    	else
    		cout << "NO" << endl;
    }
    return 0;
}
