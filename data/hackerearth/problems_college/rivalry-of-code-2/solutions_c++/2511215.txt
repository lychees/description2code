using namespace std;
int min(int i,int j)
{
	return (i<j?i:j);
}
int main()
{
   	int t;
   	cin>>t;
   	while(t--)
   	{
   		string str1,str2;
   		cin>>str1>>str2;

   		int len2=str2.size();
