
#include <algorithm>
#include <cstdio>
#include <cstdlib>


#include<iostream>

#define gc getchar_unlocked

using namespace std;

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
int main()
{
    while(1)
    {
            int n;
            cin>>n;
            if(n==0)
                return 0;
            else{
                int* ar = new int[20000000];
                for(int i=0;i<n;i++)
                    scanint(ar[i]);
                    //ar[0]=read_int();
                    //scanf("%d",&ar[i]);
                sort(ar, ar + n);
                for (size_t i = 0; i != n; i++)
                    printf("%d ",ar[i]);
                delete[] ar;
                cout<<endl;
    }
}
}
