using namespace std;
typedef long long int ll;



int main()
{
	ll n,a;
	vector<ll> arr;
	queue<ll> que;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a;
		que.push(a);
	}
	for(int i=0;i<n;i++)
	{
		cin>>a;
		arr.push_back(a);
	}
	int ans = 0;
	for(int i=0;i<n;i++)
	{
		while(que.front() != arr[i])
		{
			ans++;
			int t = que.front();
			que.pop();
			que.push(t);
		}
		que.pop();
		ans++;
	}
	cout<<ans<<"\n";
}
