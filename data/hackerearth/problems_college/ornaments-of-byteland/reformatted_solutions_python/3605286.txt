noOfOrn = int(raw_input())

m = {}
count = 0
for c in range(1, noOfOrn + 1):
    inp = raw_input()
    count += 1
    for k in inp:
        if (m.has_key(k) == False):
            m[k] = []
        if (count not in m[k]):
            m[k].append(count)
c = 0
for key in m.keys():
    if (len(m[key]) == noOfOrn):
        c += 1
print c
