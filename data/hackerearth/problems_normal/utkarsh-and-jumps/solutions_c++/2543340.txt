#include <bits/stdc++.h>
using namespace std;
#define ll long long
double dp[1000009];
int main()
{
    //cout << "Hello World!" << endl;
    ll n,p;
    scanf("%lld %lld",&n,&p);
    dp[1]=0;
    dp[0]=1;
    dp[2]=p/100.0;
	dp[3]=1 - p/100.0;
	for(int i=4;i<=n;++i)
	{
		dp[i]=(p/100.0)*dp[i-2] + (1-p/100.0)*dp[i-3];
	}
	printf("%0.6f",dp[n]);
    
    return 0;
}
