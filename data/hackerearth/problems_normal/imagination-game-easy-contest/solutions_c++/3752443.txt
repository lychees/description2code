#include <iostream>
using namespace std;
int a[17];
int main()
{
    int t,i,j,x,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int ans=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	for(i=0;i<n;i++)
    	{
    		cin>>x;
    		ans^=x-1;
    		ans^=a[i]-x;
    		
    	}
    	
    	if(ans==0)
    	cout<<"Second\n";
    	else
    	cout<<"First\n";
    }
    return 0;
}
