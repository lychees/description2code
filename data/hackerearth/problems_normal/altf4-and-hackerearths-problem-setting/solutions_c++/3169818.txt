#include<bits/stdc++.h>
using namespace std;
const int MAXN = 2002;
#define ff first
#define ss second
pair<int,int> ar[MAXN];
long long dp[MAXN];
int n,k;
int pre[MAXN];
int main() {
	int t;
	cin>>t;
	while(t--) {
		cin>>n>>k;
		pre[0]=0;
		for(int i=1;i<=n;i++)
			scanf("%d",&ar[i].ff);
		for(int i=1;i<=n;i++)
			scanf("%d",&ar[i].ss);
		sort(ar+1,ar+n+1);
		for(int i=1;i<=n;i++)
			pre[i]=pre[i-1]+ar[i].ss;
		for(int i=1;i<=n;i++) {
			dp[i]=1e17;
			for(int j=i-1;j>=0;j--) {
				long long val=dp[j]+1LL*(0LL+pre[i-1]-pre[j]+k+ar[i].ss)*ar[i].ff;
				dp[i]=min(dp[i],val);
			}
		}
		cout<<dp[n]<<endl;
	}
	return 0;
}
