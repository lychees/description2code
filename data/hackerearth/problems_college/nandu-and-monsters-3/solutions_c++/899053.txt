#include<iostream>
#include<math.h>
using namespace std;
int main()
{
	int n,q;
	cin>>n>>q;
	int i,j,y,x;
	int d[100001]={0};
	char c;
	for(i=0;i<n;i++)
	{
		cin>>c;
		d[i+1]=(c-48);
	}


	while(q--)
	{ int flag=0;
		
		cin>>x>>y;
		if(d[x]==1 || d[y]==1)
		{cout<<"No"<<endl;
		flag=1;
		continue;
		}
		for(i=0;x-pow(2,i)>=0;i++);
		for(j=0;y-pow(2,j)>=0;j++);
	 if(j>i)
	 {
	 	for(;i!=j;j--)
	 	{y=y/2;
	 	if(d[y]==1)
	 	{
	 		cout<<"No"<<endl;
	 		flag=1;
	 		break;
	 	}
	 	else if(x==y)
	 	{cout<<"Yes"<<endl;
	 	flag=1;
	 	break;}
	 }
	 }
	 else if(i>j)
	 {
	  for(;i!=j;i--)
	 	{x=x/2;
	 	if(d[x]==1)
	 	{
	 		cout<<"No"<<endl;
	 		flag=1;
	 		break;
	 	}
	 	else if(x==y)
	 	{
	 	{cout<<"Yes"<<endl;
	 	flag=1;
	 	break;}
	 }
	}
	 }
	 
	if(flag==0)
	{
		while(1)
		{
			if(d[x]==1 || d[y]==1 ||x==0 ||y==0)
			{cout<<"No"<<endl;
			break;}
			else if(x==y)
			{cout<<"Yes"<<endl;
			break;}
			x=x/2;
			y=y/2;
			
		}
	}
	}
	
	return 0;
}
