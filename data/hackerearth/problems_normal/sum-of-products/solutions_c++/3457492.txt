#include <iostream>
#define c 1000000007

using namespace std;

int main () {
    long long t, n, x, y, sum, lt, flag;
    cin >> t;
    while (t--) {
        cin >> n;
        sum = n;
        lt = n;
        flag = 1;

        for (x = 2; x <= n / x; x++) {
            y = n / x;
            if (y == x) flag = 0;
            sum += x * y + (x - 1) * (lt - y) * (y + 1 + lt) / 2;
            lt = y;
            sum = sum % c;
        }
        if (n == 1) sum = 1;
        else if (n == 2) sum = 4;
        else if (n == 3) sum = 8;
        else if (flag == 1) sum += (x - 1) * (x + lt) * (lt - x + 1) / 2;

        cout << sum % c << endl;
    }

    return 0;
}
