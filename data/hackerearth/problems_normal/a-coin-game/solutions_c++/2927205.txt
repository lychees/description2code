#include<bits/stdc++.h>

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 100000
#define inv2 500000004
const int lmt = 1000005;
int pos[lmt];

void pre(){
    int d=1;
    pos[0] = 0;
    for(int i=1;i<lmt;i++){
        if(pos[i]) continue;
        pos[i] = i+d;
        if(i+d<lmt) pos[i+d]=i;
        d++;
    }
}

int main(){
    pre();
    int t,a,b;
    scanf("%d",&t);
    while(t--){
        scanf("%d %d",&a,&b);
        if(pos[a]==b) printf("Don't Play\n");
        else printf("Play\n");
    }
    return 0;
}
