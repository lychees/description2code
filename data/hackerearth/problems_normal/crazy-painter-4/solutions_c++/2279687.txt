#include <iostream>
using namespace std;

long long getSumOfArray(int C[], int n) {
	long long sum = 0;
	for(int i= 0; i < n; i++) {
		sum += C[i];
	}
	return sum;
}

long long printCost(long long N, int C[]) {
	if(N == 1 || N == 2) {
		return getSumOfArray(C, 12*N);
	}
	
	long long sum = getSumOfArray(C, 26);
	long long d = (N*6)/13;
	long long r = ((N*6)%13)*2;
	sum = sum + (sum*(d-1)) + (13*d*(d-1))+ getSumOfArray(C, r) + r*d;
	
	return sum;
}

int main()
{
   int T;
   
   cin>>T;
   while(T--) {
   	long long N;
   	cin>>N;
   	int C[26] = {0};
   	for(int i = 0; i < 26; i++) {
   		cin>>C[i];
   	}
   	cout<<printCost(N, C)<<endl;
   }
   
    return 0;
}
