#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
ll c[200],d[200];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        ll n,q,i,x,y;
        vector< pair<ll,ll> > p;
        scanf("%lld%lld",&n,&q);
        for(i=0;i<n;i++)
        {
            scanf("%lld%lld",&x,&y);
            p.push_back(make_pair(x,y));
        }
        sort(p.begin(),p.end());
        ll k=-1;
        k++;
        c[k]=p[0].first;
        d[k]=p[0].second;
        for(i=1;i<n;i++)
        {
            if(p[i].first>d[k])
            {
                k++;
                c[k]=p[i].first;
                d[k]=p[i].second;
            }
            else
            {
                d[k]=max(d[k],p[i].second);
            }
        }
        while(q--)
        {
          scanf("%lld",&x);
          ll cnt=0;
          for(i=0;i<=k;i++)
          {
              ll k1=d[i]-c[i]+1;
              if(k1+cnt>=x)
              {
                  ll bakar=x-cnt;
                  printf("%lld\n",c[i]+bakar-1);
                  goto in;
              }
              else
            cnt+=k1;
          }
          printf("-1\n");
          in:;
        }
    }
}
