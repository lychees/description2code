import math
n = int(raw_input())
x, y, f = [], [], []
for i in range(n):
	inp = map(float, raw_input().split())
	x.append(inp[0])
	y.append(inp[1])
	f.append(inp[2])
dp = [-10000000.0] * 3000
dp[0] = f[0]
for i in range(1, n):
	for j in range(i):
		dp[i] = max(dp[i], dp[j] - (math.sqrt((x[i]-x[j])*(x[i]-x[j]) + (y[i]-y[j])*(y[i]-y[j]))))
	dp[i] += f[i]
print "%6f"%dp[n-1]
