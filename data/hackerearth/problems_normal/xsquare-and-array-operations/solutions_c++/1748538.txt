#include <bits/stdc++.h>
using namespace std;
#define ll long long int
#define pb push_back
#define f first
#define s second
#define mod 1000000007
#define inf 100000

#define pi pair<ll,ll>
#define pii pair<ll,pi>
#define f first
#define s second

#define get(t) scanf("%lld",&t)
#define rep(i,n) for(int i=0;i<n;i++)
ll a[500011];
int main(){
	ll t,n;
	get(t);
	ll maxn ;
	while(t--){
		get(n);
		vector<pi >v;
		maxn = n+1;
		rep(i,n)
			get(a[i]) , v.pb({a[i],i});
		sort(v.begin(),v.end());
		set<ll>s;
		rep(i,n)
			s.insert(i);
		s.insert(-maxn);
		s.insert(maxn);
		pi x;
		ll ans=0;
		int l,r;
		ll mi=maxn;
		for(int i=0;i<v.size()-1;i++){
			x=v[i];
			mi=1e10;
			auto it=s.upper_bound(x.s);
			if( *it!=maxn ) mi=min(mi,a[*it] );
			if(it!=s.begin()) --it;
			if(it!=s.begin()) --it;
			if( *it!=maxn  and *it!=-maxn and *it!=x.s) mi=min(mi,a[*it] );
			s.erase(x.s);
			ans+=mi;
		}
		cout<<ans<<"\n";
	}
}
