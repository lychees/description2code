mod = 1000003
a = []
a.append(1)
for x in xrange(1, 1000004):
    a.append(x*a[x-1])
    a[x] = a[x] % mod
t = input()

while (t > 0):
    n, x = [int(y) for y in raw_input().split()]
    x = x % mod
    if (n > mod):
        print("0")

    else:
        cal = a[n]*x
        cal = cal % mod
        print(cal)
    t = t-1
