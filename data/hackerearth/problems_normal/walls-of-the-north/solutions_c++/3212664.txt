#include <bits/stdc++.h>
using namespace std;



int t,n,ans;
int a[100005];
int h[100005];
int main(){
  std::ios_base::sync_with_stdio(false);
  cin >> t;
  for(int iii=0;iii<t;iii++){
      cin >> n;
      cin >> h[0];
      for(int i=1;i<n;i++){
	  cin >> h[i];
	  if(h[i] == h[i-1]){
	      a[i] = 0;
	  } else if(h[i] > h[i-1]){
	      a[i] = 1;
	  } else{
	      a[i] = -1;
	  }
      }
      for(int i=1;i<n;i++){
	  if(a[i] == 0) continue;
	  for(int j=i-1;j>0 && a[j] == 0;j--){
	      a[j] = (a[j+1] == 1)?-1:1;
	  }
	  for(int j=i+1;j<n && a[j] == 0;j++){
	      a[j] = (a[j-1] == 1)?-1:1;
	  }
      }
      if(1 < n && a[1] == 0){
	  a[1] = 1;
	  for(int j=2;j<n;j++){
	      a[j] = (a[j-1] == 1)?-1:1;
	  }
      }
      int ans = 1,last=0;
      for(int i=1;i<n;i++){
	  if(a[i] != last) ans ++;
	  last = a[i];
      }
      cout << ans << endl;
      
  }


  return 0;
}
