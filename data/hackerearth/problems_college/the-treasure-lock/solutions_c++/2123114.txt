#include<iostream>
#include<algorithm>
using namespace std;
#include<stdio.h>
#include<stdlib.h>
#include<string.h>



int main()
{
    int t,n;
    cin>>t;
    while(t--){
            cin>>n;
            if(n==0)
                cout<<-1<<"\n";
            if(n%3==0&&n>0)
            {
                for(int i=0;i<n;i++)
                    cout<<5;
            cout<<"\n";
            }
            else if(n%5==0&&n>0)
            {
                for(int i=0;i<n;i++)
                    cout<<3;
                    cout<<"\n";
            }
            else if(n>0)
            {
                int f=0,k;
                for(int i=0;i<=n;i+=5)
                {
                    if((n-i)%3==0)
                    {
                        k=i;
                        f=1;
                        break;
                    }

                }
                if(f==1)
                {
                    for(int i=0;i<(n-k);i++)
                        cout<<5;
                        for(int i=0;i<k;i++)
                            cout<<3;
                        cout<<"\n";

                }
                else
                    cout<<-1<<"\n";
        }

    }

}
