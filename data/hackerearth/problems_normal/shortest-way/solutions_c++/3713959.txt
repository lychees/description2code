#include <iostream>
#include <algorithm>
#define ll long long
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        ll a,b;
        cin >> a >> b;
        ll step =0;
        while(a>1 || b > 1)
        {
            if(a>b) 
            {
            	step+= a/b;
            	a%=b;
            	}
            else 
            {
            	step+= b/a;
            	b%=a;
            	}
           
        }
        if(step) step--;
        cout << step << endl;
    }
    return 0;
}
