#include <iostream>
using namespace std;
#define SL(x, n) ((x)<<(n))
int main() {
	int test, cs = 1, k, i;
	long long l, p, c, x;
	scanf("%d", &test);
	while(test--) {
		scanf("%ld%ld%ld", &l, &p, &c);
		k = i = 0; x = l;
		while(x < p) {
			x *= c;
			k++;
		}
		while(SL(1, i) < k) i++;
		printf("Case #%d: %d\n", cs++, i);
	}
	return 0;
}
