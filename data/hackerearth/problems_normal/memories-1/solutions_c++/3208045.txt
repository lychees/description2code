int scan()
{ int noRead=0;
    char p= getchar();
    for(;p<33;)
    { p=getchar(); }
    while(p>32)
        {
            noRead = (noRead << 3) + (noRead << 1) + (p - '0');
            p=getchar();
        }
    return noRead;
}

long long int lscan()
{ long long int noRead=0;
    char p= getchar();

}

using namespace std;

int b1[501];
int b2[501];

int main(void){

    int t = scan();
    while(t--) {
        int n = scan();
        int p = scan();
        for(int i = 0; i < p; ++i) {
            b1[i] = scan() - 1;
            b2[i] = scan() - 1;
        }
        int maxx = (1<<n)-1;
        int ans = 0;
        for(int i = 1; i <= maxx; ++i) {
            bool flag = true;
            for(int j = 0; j < p; ++j) {
                if((i&(1<<b1[j])) && (i&(1<<b2[j]))) {
                    flag = false;
                    break;
                }
            }
            if(flag)    ans++;
        }
        printf("%d\n", ans);
    }

    return 0;

}
