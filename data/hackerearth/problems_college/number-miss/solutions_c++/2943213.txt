#include "bits/stdc++.h"
using namespace std;
typedef long long int ll;
int main()
{ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    ll n,m,i,a[10001]={0},b[10001]={0},x;
    cin>>n;
    while(n--)
    {cin>>x;
    	a[x]++;
    }
    cin>>m;x=0;
    while(m--)
    {
    	cin>>x;
    	b[x]++;
    }
    for(i=0;i<10001;i++)
    {
    	if(a[i]!=b[i]) cout<<i<<" ";
    	
    }
    return 0;
}
