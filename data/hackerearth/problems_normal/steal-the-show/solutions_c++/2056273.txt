#include<cmath>
#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int ceo,cto,coo,n,k=0;
    cin>>ceo>>coo>>cto>>n;
    int a[n+3],b[n],i;
    a[(n+3)/2]=ceo;
    if(cto>coo) { a[n+2]=cto; a[0]=coo; }
    else { a[n+2]=coo; a[0]=cto; }
    for(i=0;i<n;i++) cin>>b[i];
    sort(b,b+n);
    for(i=1;i<n+2;i++)
    {
    	if(i==((n+3)/2)) continue;
    	else 
    	{
    		a[i]=b[k];
    		k++;
    	}
    }
    long sum=0;
    for(i=1;i<n+3;i++) sum+=(abs(a[i]-a[i-1]));
    cout<<sum<<endl;
    return 0;
}
