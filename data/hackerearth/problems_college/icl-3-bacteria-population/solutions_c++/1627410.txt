#include <iostream>
typedef long long       ll;
using namespace std;

int main()
{

	
    ll n;
    cin>>n;
    if(n==0 || n==1)
    	cout<<"1";
    else if(n==2)
    	cout<<"2";
    else
    {
    	ll x=1,ans;
    	ll y=1,z=2;
    	for(ll i=3;i<=n;i++)
    	{
    		ans=x+y;
    		x=y;
    		y=z;
    		z=ans;
    	}
    	cout<<ans;
    }
	
}
