#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    int t[26][100];
    scanf("%d",&n);
    string s[n];
    for(int i=0;i<n;i++)
        cin>>s[i];
    int c=0;
    for(int i=0;i<26;i++)
    {
        int l=0;
        for(int j=0;j<n;j++)
        {
            for(int k=0;k<s[j].length();k++)
                if(s[j][k]==i+97)
                {
                    l++;
                    break;
                }
        }
        if(l==n)
            c++;
    }
    printf("%d\n",c);
    return 0;
}
