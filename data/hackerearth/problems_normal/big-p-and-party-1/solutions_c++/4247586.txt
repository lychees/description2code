#include<bits/stdc++.h>
using namespace std;
#define MAX 1006
vector<int> vt[MAX];
int vis[MAX];
void bfs(void)
{
	int i;
	queue<int> q;
	int cur;
	memset(vis,-1,sizeof(vis));
	q.push(0);
	vis[0]=0;
	while(!q.empty())
	{
		cur = q.front();
		q.pop();
		for(i=0;i<vt[cur].size();++i)
		{
			int g = vt[cur][i];
			if(vis[g] == -1)
			{
				q.push(g);
				vis[g]=vis[cur]+1;
			}
		}
	}
}
int main()
{
    //cout << "Hello World!" << endl;
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int n;
    int i;
    int u;
    int v;
    int m;
    cin>>n>>m;
    for(i=0;i<m;++i)
    {
    	cin>>u>>v;
    	vt[u].push_back(v);
    	vt[v].push_back(u);
    }
    bfs();
    for(i=1;i<n;++i)
    	cout<<vis[i]<<endl;
    
    return 0;
}
