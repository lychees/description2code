#include "bits/stdc++.h"
using namespace std;
vector<int> v[100001];
int son[100001];

int visited[100001];
int sons(int node){
    int x = 0;
    for(int i = 0; i < v[node].size(); i++){
        if(visited[v[node][i]] == 0){
            visited[v[node][i]]=1;
        x += sons(v[node][i])+1;}
    }
    return son[node] = x;
}
int main(int argc, char const *argv[])
{
    int n;
    cin >> n;int tt = n;
    memset(son, -1, sizeof son);
    while(tt--){
        int x, y;cin >> x >> y;
        v[x].push_back(y);
        
        v[y].push_back(x);
    }
    visited[1]=1;
    sons(1);
    // for(int i = 1; i <= n; i++)
        // {cout << son[i] <<  " " ;}
    for(int i=1;i<=n;i++)if(son[i]!=n-1)cout << son[i]+1 << " ";else cout << son[i] << " ";

    return 0;
}
