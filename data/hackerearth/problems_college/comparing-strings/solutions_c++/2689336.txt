#include<bits/stdc++.h>
using namespace std;
char str1[100000],str2[100000];
int main()
{
    int t,n,m,i,j,k,a[126],b[126];
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%s",str1);
    	scanf("%s",str2);
    	for(i=97;i<123;i++)
    	{
    		a[i]=0;
    		b[i]=0;
    	}
    	for(i=0;i<strlen(str1);i++)
    	{
    		++a[str1[i]];
    	}
    	for(i=0;i<strlen(str2);i++)
    	{
    		++b[str2[i]];
    	}
    	for(i=97;i<123;i++)
    	{
    		if((a[i]>0)&&(b[i]>0))
    		break;
    	}
    	if(i==123)
    	printf("NO\n");
    	else
    	printf("YES\n");
    }
    return 0;
}
