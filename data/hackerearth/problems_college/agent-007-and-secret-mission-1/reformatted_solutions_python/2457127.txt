T = input()


def determine(s, l):
    count = 0
    for i in range(l):
        if (s[i] == '1'):
            count = count + 1
    return count


while (T > 0):
    s = raw_input()
    tl = len(s)
    l1 = determine(s, tl)

    n = input()
    summation = n*(n+1)/2
    if (summation % 2 == 0):
        print(l1)
    else:
        print(tl - l1)
    T = T-1
