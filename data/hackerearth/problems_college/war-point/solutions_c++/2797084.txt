#include<iostream>
#include <stdio.h>
#include<stdlib.h>
#define INT_MAX 2147483647

using namespace std;
int main()
{
    long long int t,n,m,j,i,l,k,s,ans;
    long long int a[100010],b[100010];
    //t=readInt();
    scanf("%lld",&t);
    while(t--){
    	scanf("%lld",&n);
    	l=0;
    	for(i=0;i<100010;i++){b[i]=0;a[i]=0;}
    	for(i=0;i<n;i++){
    		scanf("%lld",&a[i]);
    		l=l+a[i];
    	}
    	b[0]=a[0];
    	for(i=1;i<n;i++){
    		b[i]=b[i-1]+a[i];
    	}
    	ans=INT_MAX;
    	for(i=0;i<n;i++){
    		j=b[i];
    		m=(l-b[i]);
    		if(abs(m-j)<ans){
    			ans=abs(m-j);
    		}
    	}
    	printf("%lld\n",ans);
    }
    return 0;
}
