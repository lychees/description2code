#include <bits/stdc++.h>

typedef long long int li;

#define min(a,b) a<=b?a:b
#define max(a,b) a>=b?a:b
#define Fi(i, st, ft) for(i = st;i <= ft; ++i)
#define Fd(i, st, ft) for(i = st;i >= ft; --i)
#define pb(a, b) a.push_back(b)
#define mod 1000000007
#define MAXN 1500000
#define MX 1450000
#define inf 1000000000000000000ll
const int maxBufSize = (10000000);

using namespace std;

li t , n , m , rf , bf , rr , br;
li crr[1005][1005];

void calc(li a , li k)
{
    li i, j , d;	
    // Caculate value of Binomial Coefficient in bottom up manner
    for (i = 0; i <= a; i++)
    {
    	d = min(i , k);
        for (j = 0; j <= d; j++)
        {
            if(i == 0) {
            	crr[i][j] = 1;
            }
            // Base Cases
            if (j == 0 || j == i)
                crr[i][j] = 1;
 
            // Calculate value using previosly stored values
            else
                crr[i][j] = (crr[i-1][j-1] + crr[i-1][j])%mod;
        }
    }
}

int main()
{
	li i , j , a , b , c , d , mn , ans , temp;
	
	calc(1001 , 1001);
	//cout << crr[4][2] << "\n";
	scanf("%lld" , &t);
	while(t--) {
		rf = 6 , bf = 6 , rr = 18 , br = 18;
		scanf("%lld %lld" , &n , &m);
		rf *= n , bf *= n , rr *= n , br *= n;
		//cout << rf << " " << bf << " " << rr << " " << br << "\n";
		ans = 0;
		if((rf+bf) < m/2) {
			ans = 0;
		} else if(m&1) {
			ans = 0;
		} else {
			Fi(i , 0 , m/2) {
				if(rf >= i && bf >= m/2-i && rr >= m/2-i && br >= i) {
					a = crr[rf][i];
					b = crr[bf][m/2-i];
					c = crr[rr][m/2-i];
					d = crr[br][i];
					//cout << i << " " << m-i << " " << a << " " << b << " " << c << " " << d << "\n";
					temp = ((a*b)%mod * ((c*d)%mod))%mod;
					ans = (ans + temp)%mod;
					//cout << i << " " << temp << "\n";
				}
			}
		}
		printf("%lld\n" , ans);
	}
	return 0;
}
