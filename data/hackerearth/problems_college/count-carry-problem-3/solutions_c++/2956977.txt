#include <bits/stdc++.h>
using namespace std;

int main() {
	long long int t,n,i,j,k,carry,count;
	cin>>t;
	while(t--){
		carry=0,count=0;
		string str,str1;
		cin>>str>>str1;
		i=str.length();
		j=str1.length();
		
		k=min(i,j);
		while(k--)
		{
			long int y=(str[--i]-'0')+(str1[--j]-'0')+carry;
			if(y>9)
			{
				count++;
				carry=y%10;
			}
		}
	//	cout<<count<<"\n";
	
	if(count==0)
	cout<<"No carry operation\n";
	else if(count==1)
	cout<<"1 carry operation\n";
	else
	cout<<count<<" carry operations\n";
		
	}
}
