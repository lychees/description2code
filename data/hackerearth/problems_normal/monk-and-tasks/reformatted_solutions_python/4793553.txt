t = int(raw_input())

for i in range(t):
    n = int(raw_input())
    arr = map(int, raw_input().strip().split(' '))
    tasks = [None]*n

    for j in range(n):
        temp = arr[j]
        count = 0
        while temp:
            temp = temp & (temp-1)
            count += 1
        tasks[j] = count

    at = zip(tasks, arr)
    at.sort(key=lambda t: t[0])

    for j in range(n-1):
        print at[j][1],
    print at[n-1][1]
