#include<math.h>
#include<stdio.h>
#include<iostream>
using namespace std;





#define s(n)                        scanf("%d",&n)


int ispalindrome(int n){
	int m=n;
	int s=0,c;
	while(m){
		c=m%10;
		s=s*10+c;
		m/=10;
	}
	if(s==n)
		return 1;

	return 0;
}

int main(){
int t;
cin>>t;
while(t--){
	int n;
	s(n);
	int i=n,j;
	while(1){
		if(ispalindrome(i)){
				int root=sqrt(i);
				j=2;
				int flag=0;
				for(j=2;j<=root;j++)
					if(i%j==0)
						flag++;
				if(!flag)
				{	printf("%d\n",i);
					break;
					}

		}



		i++;
	}
}
return 0;
}
