#include <stdio.h>
#include <stdlib.h>

int comp(const void *a, const void *b)
{
	return (*((int *)b) - *((int *)a));
}

int main()
{
    int t, n, i, start, end;
    scanf("%d", &t);
    
    for ( ; t; t--) {
    	scanf("%d", &n);
    	int rose[n];
    	
    	for (i = 0; i < n; i++) {
    		scanf("%d", &rose[i]);
    	}
    	
    	qsort(rose, n, sizeof(int), comp);
    	
    	start = 0; end = n - 1;
    	while (start <= end) {
    		printf("%d ", rose[start]);
    		while ((start < end) && (rose[start] == rose[start + 1])) {
    			start++;
    		}
    		start++;
    		if (start < end) {
    			printf("%d ", rose[end]);
    		}
    		while ((start < end) && (rose[end - 1]) == rose[end]) {
    			end--;
    		}
    		end--;
    	}
    	printf("\n");
    }
    
    return 0;
}
