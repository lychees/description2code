#include <iostream>
using namespace std;

int fact(int q)
{
    int f=1,i;
    for(i=1;i<=q;i++)
    {
        f=f*i;
    }
    return f;
}

int main()
{
    int n,x[100],y[100],a[100],b[100],u[100],d[100],i[100],t[100],p[100],r[100],k[100],j;
    cin>>n;
    for(j=0;j<n;j++)
    {
        cin>>x[j]>>y[j];
        a[j]=7-x[j];
        b[j]=5-y[j];
        u[j]=fact(x[j]);
        d[j]=fact(a[j]);
        i[j]=fact(y[j]);
        t[j]=fact(b[j]);
        p[j]=fact(7)/(u[j]*d[j]);
        r[j]=fact(5)/(i[j]*t[j]);
        k[j]=p[j]*r[j]*fact(x[j]+y[j]);
    }

    for(j=0;j<n;j++)
    {
        cout<<k[j]<<endl;
    }
return 0;
}
