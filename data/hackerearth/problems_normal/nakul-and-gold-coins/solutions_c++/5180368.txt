#include <bits/stdc++.h>
using namespace std;
int a[1000001];
vector<int> pr;
vector<long long int> ans;
int main()
{
	for(int i=2;i<=1000;i++)
		if(!a[i])
			for(int j=i*i;j<=1000000;j+=i)
				a[j]=1;
	for(int i=2;i<=1000000;i++)
		if(!a[i])
			pr.push_back(i);
	int sz=pr.size();
	for(int i=0;i<sz;i++)
		for(int j=i+1;pr[j]<=(100000000/pr[i]) && j<sz;j++)
			ans.push_back((long long int)(pr[i])*pr[j]);
	sort(ans.begin(),ans.end());
	int x,y,t;
	cin >> t;
	while(t--)
	{
		cin >> x >> y;
		cout <<  (upper_bound(ans.begin(),ans.end(),y)-lower_bound(ans.begin(),ans.end(),x)) << endl;
	}
    return 0;
}
