def fun(n):
	if n==0:return 0
	elif prime(n):return 1
	else:
		x=find(n)
		n-=x
		return (1+fun(n))
def find(x):
	if prime(x):return x
	else:return(find(x-1))
def prime(x):
	i=2
	while i*i<=x:
		if x%i==0:return False
		i+=1
	return True
