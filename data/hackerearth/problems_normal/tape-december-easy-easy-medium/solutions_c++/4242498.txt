long long
sumk (long long i, long long k)
{
  long long sum = 0;
  for (int n = i; (n < i + k) && (n < 100005); n++)
    {
      if (arr[n] != 0)		// && nonzero<i)
	sum = sum + arr[n];
    }
  return sum;
}

long long
setk (long long i, long long k)
{
  int c = k;
//4 5 6 
//1 2 3
  while (arr[i] == 0 && c > 1)
    {
      i--;
      c--;
    }
  if (arr[i] == 0)
    return -1;
  for (int n = i; (n < i + k) && (n < 100005); n++)
    {
      if (arr[n] != 0)
	arr[n] = 100007;
    }
}
