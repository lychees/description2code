#include <bits/stdc++.h>
using namespace std;
int dp[1001][1001][2];
int g_c_d(int a, int b){
    if (a == 0)
        return b;
    return g_c_d(b%a, a);
}
int main(){
	int a,b,t,i,j,k,gcd;
	for(i=2;i<=1000;i++){
		dp[1][i][0]=0;
		dp[1][i][1]=1;
	}
	for(i=2;i<=1000;i++){
		dp[i][1][0]=1;
		dp[i][1][1]=0;
	}
	for(i=2;i<=1000;i++){
		for(j=2;j<=1000;j++){
			gcd=g_c_d(i,j);
			for(k=0;k<2;k++){
				if(gcd>1){
					if(k==0){
						dp[i][j][k]=!(dp[i][j-1][1])||!(dp[i][j/gcd][1]);
					}
					else if(k==1){
						dp[i][j][k]=!(dp[i-1][j][0])||!(dp[i/gcd][j][0]);
					}
				}
				else if(gcd==1){
					if(k==0){
						dp[i][j][k]=!(dp[i][j-1][1]);
					}
					else if(k==1){
						dp[i][j][k]=!(dp[i-1][j][0]);
					}
				}

			}
		}
	}
	/*for(i=1;i<=10;i++){
		for(j=1;j<=10;j++){
			cout<<i<<" "<<j<<" "<<dp[i][j][0]<<" "<<dp[i][j][1]<<"\n";
		}
	}*/
	scanf("%d",&t);
	while(t--){
		scanf("%d%d",&a,&b);
		if(a==1&&b==1){
			cout<<"Draw\n";
		}
		else if(dp[a][b][0]){
			cout<<"Arjit\n";
		}
		else{
			cout<<"Chandu Don\n";
		}
	}
	return 0;
}
