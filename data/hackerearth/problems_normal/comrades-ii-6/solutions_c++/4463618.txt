#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int heightof[100001]={0};
vector<int> adj[100001];

	void dfs(int height,int element)
	{
		heightof[element]=height;
		vector< int> ::iterator it=adj[element].begin();
	while(it!=adj[element].end())
		{
		dfs(height+1,*it);
		it++;
		}
		return ;
	}

int main()
{
    //cout << "Hello World!" << endl;
    
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,a,root;
    	cin>>n;
    	for(int i=0;i<=n;i++)
    	{
    		adj[i].clear();
    	}
    	for(int i=1;i<=n;i++)
    	{
    		cin>>a;
    		if(a==0)
    		{
    			root=i;
    			//adj[a]=0;
    		}
    		else
    		{
    			adj[a].push_back(i);
    		}
    	}
    	dfs(0,root);
    long long 	int sumofheight=0;
    	for(int i=1;i<=n;i++)
    	{
    		sumofheight+=(long long) heightof[i];
    	}
    	long long int total=n*(n-1)/2;
    	cout<<sumofheight<<" "<<total-sumofheight<<endl;
    }
    return 0;
}
