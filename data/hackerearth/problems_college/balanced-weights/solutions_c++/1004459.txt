#include <bits/stdc++.h>
using namespace std;

int64_t gcd(int64_t a,int64_t b)
{
	int64_t t;
	if(a==1 || b==1)	return 1;
	if(b>a)
	{
		t=a;
		a=b;
		b=t;
	}
	if(b==0)	return a;
	return gcd(b,a%b);
}

int64_t a[1000000];

int main()
{
    int64_t T;
    scanf("%lld",&T);
    while(T--)
    {
    	int64_t n;
    	scanf("%lld",&n);
    	//cout<<n<<endl;
    	for(int64_t i=0;i<n;i++)	scanf("%lld",&a[i]);
    	int64_t g=gcd(a[0],a[1]);
    	bool done=0;
    	for(int64_t i=2;i<n;i++)
    	{
    		if(g==1)	{printf("%lld\n",n);done=1;break;}
    		else
    		{
    			g=gcd(g,a[i]);
    		}
    	}
    	if(!done)
    	{
    		printf("%lld\n",g*n);
    	}
    }
    return 0;
}
