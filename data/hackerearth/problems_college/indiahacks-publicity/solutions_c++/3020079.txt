#include<bits/stdc++.h>
using namespace std;
int vis[100005];
vector<int>vec[100005];

void dfs(int i)
  {
      vis[i]=1;
      int n=vec[i].size();
      int j;
      for(j=0;j<n;j++)
      {
          if(vis[vec[i][j]]!=1)
              dfs(vec[i][j]);
      }
      return ;
  }
int main()
{
    int test;
    scanf("%d",&test);
    while(test--)
    {
        int n,m,a,b,i;
        scanf("%d%d",&n,&m);
        for(i=0;i<=n;i++)
          vis[i]=-1;
        for(i=0;i<=n;i++)
        vec[i].clear();
       for(i=0;i<m;i++)
       {
           scanf("%d%d",&a,&b);
           vec[a].push_back(b);
           vec[b].push_back(a);
       }
       int count=0;
       for(i=0;i<n;i++)
       {
           if(vis[i]!=1)
           {
           	 count++;
              
               dfs(i);
           }
       }
       printf("%d\n",count);
    }
    return 0;
}
