long long power (long long n){
	long long ctr = 0 ;
	while (n != 1) {
		ctr ++;
		n /= 2;
}

	return ctr;
}
long long int mul(long long int a)
{
	if(a==0)
		return 1;
	if(a==1)
		return 2;
	long long int x=mul(a/2);
	if(a%2==0)
		return x*x;
	else
		return x*x*2;
}
