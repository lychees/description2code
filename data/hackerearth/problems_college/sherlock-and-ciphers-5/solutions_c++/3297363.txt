#include <iostream>
#include<string.h>
#include<math.h>
#include<stdio.h>
using namespace std;

int main()
{  int t;
    cin>>t;
    t++;
    while(t--)
    { 
    	char a[500],b[500];
    	
    	int k=0,i=0,l,s,j;
    	gets(a);
    	l=strlen(a);
    	s=sqrt(l);
    
    	for(i=0;i<l;i++)
    	{ 
    		for(j=i;j<l;j=j+s)
    		{   if(k>=l)
    		    break;
    			b[k]=a[j];
    			k++;
    			
    		}
    		
    	}
    	
    	for(i=0;i<k;i++)
        cout<<b[i];
    
    	cout<<"\n";
    	
    }
    return 0;
}
