#include<iostream>
#include<cmath>
#define ull unsigned long long int
using namespace std;
int main()
{
	ull cycle,test,a,b,i,j,k,ans,rem,mod;
	scanf("%llu",&test);
	while(test--)
	{
		scanf("%llu %llu",&a,&b);
		rem=a%10;
		cycle=1;
		if(rem==5) ans=5;
		else if(rem==0) ans=0;
		else if(rem==1) ans=1;
		else if(rem==6) ans=6;
		else if((rem==2) || (rem==3) || (rem==7) || (rem==8)) cycle=4;
		else if((rem==4) || (rem==9))cycle=2;
		mod=b%cycle;
		if(rem==2)
		{
			if(mod==1) ans=2;
			else if(mod==2) ans=4;
			else if(mod==3) ans=8;
			else if(mod==0) ans=6;
		}
		else if(rem==3)
		{
			if(mod==1) ans=3;
			else if(mod==2) ans=9;
			else if(mod==3) ans=7;
			else if(mod==0) ans=1;
		}
		else if(rem==7)
		{
			if(mod==1) ans=7;
			else if(mod==2) ans=9;
			else if(mod==3) ans=3;
			else if(mod==0) ans=1;
		}
		else if(rem==8)
		{
			if(mod==1) ans=8;
			else if(mod==2) ans=4;
			else if(mod==3) ans=2;
			else if(mod==0) ans=6;
		}
		else if(rem==4)
		{
			if(mod==1) ans=4;
			else if(mod==0) ans=6;
			
		}
		else if(rem==9)
		{
			if(mod==1) ans=9;
			else if(mod==0) ans=1;
			
		}
		printf("%llu\n",ans);
	}
	return 0;
}
