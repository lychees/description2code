#include <bits/stdc++.h>

typedef long long int li;

#define Fi(i, st, ft) for(i = st;i <= ft; ++i)
#define Fd(i, st, ft) for(i = st;i >= ft; --i)
#define pb(a, b) a.push_back(b)
#define mod 1000000007
#define MAXN 1500000
#define MX 1450000
#define inf 1000000000000000000ll
const int maxBufSize = (10000000);

using namespace std;

li t , n , l , r;
li arr[105];

li GCD(li a , li b)
{
	
	li c;
	if(a < b) {
		c = a;
		a = b;
		b = c;
	}
	if(b == 0) {
		return a;
	}
	return GCD(b , a%b);
}

int main()
{
	li i , j , a , b , c;
	scanf("%lld" , &t);
	while(t--) {
		scanf("%lld" , &n);
		map<li,li> M;
		set<li> s;
		set<li>::iterator it;
		Fi(i , 1 , n) {
			scanf("%lld" , &arr[i]);
		}
		Fi(i , 1 , n) {
			for(it = s.begin(); it != s.end(); ++it) {
				a = *it;
				b = GCD(a , arr[i]);
				if(M.find(b) != M.end()) {
					M[b] = M[a]+M[b];
				} else {
					M[b] = M[a];
					s.insert(b);
				}
			}
			if(M.find(arr[i]) != M.end()) {
				M[arr[i]] += 1;
			} else {
				M[arr[i]] = 1;
				s.insert(arr[i]);
			}
		}
		//cout << M[1] << " " << M[2] << "\n";
		scanf("%lld %lld" , &l , &r);
		li ans = 0;
		for(it = s.begin(); it != s.end(); ++it) {
			a = *it;
			if(a >= l && a <= r) {
				ans = ans+M[a];
			}
		}
		printf("%lld\n" , ans);
	}
	return 0;
}
