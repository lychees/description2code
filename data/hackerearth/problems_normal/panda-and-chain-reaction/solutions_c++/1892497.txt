#include<bits/stdc++.h>
using namespace std;
#define p 1000003
long long int a[10000000]; 
void precompute()
{
	long long int i;
	a[0]=1;
	for(i=1;i<=p;i++)
	{
		a[i]=(a[i-1]*i)%p;
	}
}
int main()
{
	precompute();
	long long t,n,x;
	scanf("%lld",&t);
	while(t--)
	{
		cin>>n>>x;
		x%=p;
		if(n>p)
			n=p;
		cout<<(a[n]*x)%p<<endl;
	}
	return 0;
}
