using namespace std;

int main() {
  int t;
  scanf("%d", &t);
  long long n, ans;
  while (t--) {
    scanf("%lld", &n);
    if (n == 1)
      ans = 0;
    else if (n == 2)
      ans = 1;
    else {
      ans = (n * (n - 1));
      ans /= 2;
    }
    printf("%lld\n", ans);
  }
  return 0;
}
