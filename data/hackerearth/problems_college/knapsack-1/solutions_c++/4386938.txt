#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,sum;
    	cin>>n>>sum;
    	vector<int> v;
    	for(int i=0;i<n;i++)
    	{
    		int c; cin>>c;
    		v.push_back(c);
    	}
    	int ans[sum+1];
    	fill_n(&ans[0],sum+1,0);
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<=sum;j++)
    		{
    			if(j>=v[i])
    			{
    				ans[j]=max(ans[j-v[i]]+v[i],ans[j]);
    			}
    		}
    	}
    	//for(int i=1;i<=sum;i++)
    	//	cout<<ans[i]<<" ";
    	//cout<<endl;
    	cout<<ans[sum]<<endl;
    }
    return 0;
}
