#include<bits/stdc++.h>
#include <stdio.h>
#include<string.h>
using namespace std;
char s[100001];
int cum[100000] = {0};
int main()
{
    long long t,p,l,i,total=0,sum=0,j=0,n=0,d=0;
    scanf("%lld",&t);
    while(t--)
      {
                   long long total=0,sum=0,j=0,fin = 0,ctr=0;
    	scanf("%lld %lld",&n,&p);
    	cin>>s;
    	l =n;
    	fin = l*l;
    	if(s[0]=='0')
    	  {
    	    ctr=0;
    	    cum[0]=0;
    	  }

    	else
    	 {
    	   ctr+=1;
    	   cum[0]=1;

    	 }
    	for(i=1;i<l;i++)
    	{
    		if(s[i]=='1')
    	     {
    	        ctr++;
    	        cum[i]=ctr;

    	     }

    	   else
    	     {
    	        cum[i]=cum[i-1];

    	     }

    	}

    	/*for(i=0; i<l; i++)
    	  cout<<cum[i]<<" ";
    	  cout<<endl;*/

    	for(i=0;i<l;i++)
    	  {

                    if(s[i]=='0')
                      continue;
                    j = i+p;
                    if((i+p) > (l-1))
                         j = l-1;

    	sum += (cum[j] - cum[i])*2 + 1;


    	}
    	n = sum/__gcd(sum,fin);

    	d = fin/__gcd(sum,fin);
    printf("%lld/%lld\n",n,d);
    }
    return 0;
}
