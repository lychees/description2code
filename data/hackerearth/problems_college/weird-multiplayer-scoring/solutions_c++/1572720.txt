#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int n,m,val,i;
    cin>>n>>m;
    string s,ans;
    map<string ,long long int > mp;
    long long int mval=0;
    for(i=0;i<m;i++)
    {
    	cin>>s>>val;
    	mp[s]+=val;
    	if(mp[s]>mval)
    	{
    		mval=mp[s];
    		ans=s;
    	}
    }
   
    cout<<ans<<endl;
    return 0;
}
