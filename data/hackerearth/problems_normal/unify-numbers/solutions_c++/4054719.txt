#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
	long long int sum=0;
	long long int n,med1,med2,res;
	cin>>n;
	long long int arr[n];
	for(int i=0;i<n;i++)
		cin>>arr[i];
	sort(arr,arr+n);
	if(n%2==0){
		med1=n/2;
		med2=(n/2)-1;
		res=(arr[med1]+arr[med2])/2;
		for(int i=0;i<n;i++){
			sum=sum+abs(arr[i]-res);
		}
		cout<<sum<<endl;
	}
	else{
		med1=n/2;
		res=arr[med1];
		for(int i=0;i<n;i++){
			sum=sum+abs(arr[i]-res);
		}
		cout<<sum<<endl;
	}
	}
    return 0;
}
