#include<bits/stdc++.h>
using namespace std;
int n,m;
int mat[1002][1002];
int v[1002][1002];

int f(int x,int y)
{
	v[x][y]=1;
	int c=1;
	queue<pair<int,int> >q;
	q.push(make_pair(x,y));
	while(!q.empty())
	{
		x=q.front().first;
		y=q.front().second;
		q.pop();
		for(int i=-1 ; i<=1 ;i++)
		{
			for(int j=-1;j<=1;j++)
			{
				if(!v[x+i][y+j] && mat[x+i][y+j])
				{
					c++;
					q.push(make_pair(x+i,y+j));
					v[x+i][y+j]=1;
				}
			}
		}
	}
	return c;
}

int main()
{
	int t,i,j;cin>>t;
	while(t--)
	{
		cin>>n>>m;
		memset(mat,0,sizeof(mat));
		memset(v,0,sizeof(v));
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=m;j++)
			{
				cin>>mat[i][j];
			}
		}
		int ans=0,c=0;
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=m;j++)
			{
				if(!v[i][j] && mat[i][j])
				{
					ans=max(f(i,j),ans);
					c++;
				}
			}
		}
		cout<<c<<" "<<ans<<endl;
	}
	return 0;
}
