#include <iostream>
using namespace std;

int main()
{
   int t,n,d,d1,dis;
   
   cin>>t;
  
   while(t--)
   {
   	   cin>>n>>d;
   	   d1=0;
  	   dis=0;
   	   for(int i=1;i<=n;i++)
   	   {
   	   	d1=d1+2;
   	   	dis=dis+2;
   	   	
   	   	
   	   	if(d1%6==0){
   	   		d1=0;
   	   		n=n+2;
   	   	}
   	   }
   	  if(dis>=d){
   	  	cout<<"Yes"<<endl;
   	  }
   	  else{
   	  	cout<<"No"<<endl;
   	  }
   }
   return 0;
}
