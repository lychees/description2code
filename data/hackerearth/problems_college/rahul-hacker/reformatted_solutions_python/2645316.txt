def digit(num):

    no = map(str, str(num))
    l = int(len(str(num)))/2
    for i in range(l):

        a = no[i]
        no[i] = no[-1-i]
        no[-1-i] = a
    z = int(''.join(no))

    return z


for k in range(input()):
    x, y = map(int, raw_input().split(" "))
    s = int(digit(x))+int(digit(y))
    ans = digit(s)
    print ans
