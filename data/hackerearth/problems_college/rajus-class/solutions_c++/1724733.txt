#include <iostream>
#include<algorithm>
#include<string>
using namespace std;

int main()
{
    pair<int,string> p[105];
    long long int test,n,a,i;
    cin>>test;
    while(test--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>p[i].first;
    		cin>>p[i].second;
    	}
    	sort(p,p+n);
    	cin>>a;
    	cout<<p[a-1].second<<endl;
    }
    return 0;
}
