using namespace std;
const LL mm = 100000000000000000ll;
LL ar[100005];
LL sum[100005];

int main(){
    LL n,p,res=0ll;
    LL h;
    cin>>n>>h;
    LL hh=h,mx=-1000000009;
    for(int i=1;i<=n;i++){
        cin>>ar[i];
        sum[i]=sum[i-1]+ar[i];
        if(sum[i]>mx){
            mx=sum[i];
            p=i;
        }
        hh-=i;
        if(res==0 && hh<=sum[i]){
            res=i;
        }
    }
    if(res) cout<<res<<endl;
    else{
        LL lo=0ll,hi=3000000000ll,pos;
        while(lo<=hi){
            LL mid=(lo+hi)/2ll;
            LL d=mid*n+p;
            LL gg=mm/d;
            if(d>3000000000LL){
                hi=mid-1ll;
                continue;
            }
            LL dd;
            if(d&1ll) dd=((d+1ll)/2ll)*d;
            else dd = (d/2ll)*(d+1ll);
            LL height=h-dd;
            LL now=sum[n]*mid+sum[p];
            if(height<=now){
         //       cout<<"dfkjv "<<lo<<" "<<mid<<" "<<hi<<" "<<height<<" "<<now<<endl;
          //      cout<<"dfkjv "<<lo<<" "<<mid<<" "<<hi<<" "<<height<<" "<<now<<endl;
                pos=mid;
                hi=mid-1ll;
            }
            else{
        //        cout<<"IN "<<lo<<" "<<mid<<" "<<hi<<" "<<height<<" "<<now<<endl;
                lo=mid+1ll;
            }
        }
     //   cout<<"look "<<pos<<endl;
        LL pre = pos*n+p-n;
        LL now = sum[n]*(pos-1ll)+sum[p];
        LL height = h-(pre*(pre+1ll))/2ll;
        pre++;
        for(int i=p+1;i<=n;i++){
            height-=pre;
            now+=ar[i];
            if(height<=now){
                cout<<pre<<endl;
                return 0;
            }
            pre++;
        }
        for(int i=1;i<=p;i++){
            height-=pre;
            now+=ar[i];
            if(height<=now){
                cout<<pre<<endl;
                return 0;
            }
            pre++;
        }
    }
    return 0;
}
