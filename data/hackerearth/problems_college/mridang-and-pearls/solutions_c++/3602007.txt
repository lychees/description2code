#include <iostream>
#include<map>
using namespace std;

int main()
{
    int n,flag=0,flag1=0,p=0,i=0,j=0;
    int a[100000],b[100000];
    long arr[100010];
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
    	scanf("%ld",&arr[i]);
    	for(j=flag;j<i;j++)
    	{
    		if(arr[i]==arr[j]){
    			flag1++;
    			a[p]=flag;
    			b[p]=i;
    			flag=i+1;
    			p++;
    			break;
    		}
    	}
    }
    if(b[p-1]!=(n-1))
    b[p-1]=n-1;
    if(flag>0)
    cout<<flag1<<endl;
    else
    cout<<-1<<endl;
    for(i=0;i<p;i++){
    	cout<<a[i]+1<<" "<<b[i]+1<<endl;
    }
    return 0;
}
