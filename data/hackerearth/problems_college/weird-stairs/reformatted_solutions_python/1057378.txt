for o in range(input()):
    w, n = map(int, raw_input().split())
    val = map(int, raw_input().split())
    ans = n*w
    arr = [0] * (n+1)
    # print val
    for i in range(1, n+1):
        # print "entered "
        a = val[i-1]
        # print a
        if a == 1:
            height = 2 * i
        elif a == 2:
            height = arr[i-1] + 1
        arr[i] = height
        ans += arr[i] - arr[i-1]
    print ans
