#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,n,k,num,val,c=0;
    scanf("%d%d",&n,&k);
    map<int,int> m;
    map<int,int>::iterator it;
    for(i=0;i<n;i++)
    {
    	scanf("%d",&num);
    	m[num]++;
    }
    for(it=m.begin();it!=m.end();it++)
    {
    	if(c<(it->second))
    	{
    		c=it->second;
    		val=it->first;
    	}
    }
    printf("%d",val);
    return 0;
}
