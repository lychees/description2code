#include <iostream>
using namespace std;
int max(int a,int b)
{
	 return a>b ? a :b;
}
int lcs(string x,string y,int n)
{
	int l[n+1][n+1];
	int k=0;
	int i,j;
	for(i=0;i<=n;i++)
	{
		for(j=0;j<=n;j++)
		{
			if(i==0 || j==0)
			  l[i][j]=0;
			else if(x[i-1]==y[j-1])
			{
				//s[k ++]=x[i-1];
				l[i][j]=l[i-1][j-1]+1;
			}  
			else
			   l[i][j]=max(l[i-1][j],l[i][j-1]);
		}
	}
	return l[n][n];
}
int main()
{
   int t,n;
   string s1,s2;
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	cin>>s1>>s2;
   	cout<<lcs(s1,s2,n)<<endl;
   }
    return 0;
}
