#include "bits/stdc++.h"
#define endl '\n' 
#define mod 1000000007
#define pb push_back
#define mp make_pair
#define FOR(i , start , end) for(int i = int(start) ; i <= int(end) ; i++)
#define REP(i , end , start) for(int i = int(end) ; i >= int(start) ; i--)
typedef long long ll ;
using namespace std ;
const int N = 1e5 + 5 ;
int n , a[N] ;
ll dp[N] ;
int main()
{
	ios_base::sync_with_stdio(false) ; cin.tie(0) ; cout.tie(0) ;
	cin >> n ;
	FOR(i , 1 , n) cin >> a[i] , dp[i] = ll(a[i]) ;
	FOR(i , 1 , n)
		for(int j = 2 ; j * i <= n ; j++)
			dp[j*i] = max(dp[j*i] , ll(a[i*j]) + dp[i]) ;
	FOR(i , 1 , n) cout << dp[i] << ' ' ;
	cout << endl ;
	return 0 ;
}
