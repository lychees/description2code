def remadd(call,time,n,j):
	temp=call[j]
	for i in range(j+1,n):
		call[i-1]=call[i]
	time+=1
	call[n-1]=temp
	return call,time

n=int(raw_input())
call=map(int,raw_input().strip().split())
idea=map(int,raw_input().strip().split())
time=0
j=0
for i in range(n):
	if call[i]==idea[j]:
		j+=1
		time+=1
		#print call, time, "SAME"
	else:
		while call[i]!=idea[j]:
			call,time=remadd(call,time,n,j)
			#print call, time, "DURING ADJ"
		j+=1
		time+=1
		#print call, time,"AFTER ADJ"
