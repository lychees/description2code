#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
	long long int n,k,m;
	cin >> n >> k >>m;
	long long int p[n];
	int i;int j;
	long long int sum;
	int set=0;long long int xmin=0;
	for(i=0;i<n;i++)
	{
		sum=0;
		cin >> p[i];
	}
	int flag=0;
	for(i=0;i<n;i++)
	{
		for(j=0;j<p[i];j++)
	{
	long long int x;cin >> x;sum+=x;
	}
	sum+=(p[i]-1)*m+(p[i])*k;
	if(flag==0)
	{
		xmin=sum;
		set=i+1;
		flag=1;
	}
	else
	if(sum<xmin)
	{
		set=i+1;
		xmin=sum;
	}
	sum=0;
	}
	cout << set << " "<< xmin << endl;
	}
    return 0;
}
