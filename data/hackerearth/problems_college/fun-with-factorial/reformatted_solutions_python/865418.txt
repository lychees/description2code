t = int(raw_input())
while (t > 0):
    n = int(raw_input())
    total = 1
    for i in range(1, n+1):
        total *= i
        total %= 100000
    s = str(total)
    tt = len(s)
    if (tt == 1):
        print "0000"+s
    elif (tt == 2):
        print "000"+s
    elif (tt == 3):
        print "00"+s
    elif (tt == 4):
        print "0"+s
    else:
        print s
    t -= 1
