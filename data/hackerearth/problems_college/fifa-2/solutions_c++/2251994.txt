#include <bits/stdc++.h>
using namespace std;

int n;
int x[100005];
int y[100005];

map <int, int> X;
map <int, int> Y;

int main() {
	scanf("%d", &n);
	for(int i = 1; i <= n; i++) {
		scanf("%d %d", &x[i], &y[i]);
		X[x[i]]++;
	}
	for(int i = 1; i <= n; i++) {
		int a = n - 1, b = n - 1 - X[y[i]];
		a += n - 1 - b;
		printf("%d %d\n", a, b);
		
	}
	
}
