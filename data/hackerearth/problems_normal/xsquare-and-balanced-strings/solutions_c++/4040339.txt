#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    long t;
    cin >> t;
    while(t--) {
    	char str[100001];
    	long alph[26], len;
    	int flag = 1;
    	cin >> str;
    	for(int i = 0; i < 26; i++)
    		alph[i] = 0;
    	len = strlen(str);
    	for(int i = 0; i < len; i++) {
    		int index = str[i] - 97;
    		alph[index] += 1;
    	}
    	for(int i = 0; i < 26; i++) {
    		if(alph[i] % 2 != 0) {
    			flag = -1;
    			break;
    		}
    	}
    	cout << flag << "\n";
    }
    return 0;
}
