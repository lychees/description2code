#include <iostream>
#include <stdio.h>
#include <queue>

using namespace std;

int main()
{
	queue <int> c,p;
	int i,n,t,temp;
	
	cin >> n ;
	
	for(i=0;i<n;i++)
	{
		cin >> t ;
		c.push(t);
	}
	
	for(i=0;i<n;i++)
	{
		cin >> t ;
		p.push(t);
	}
	
    t = 0 ;
	
	while(!c.empty())
	{
		if(c.front() == p.front())
		{
			c.pop();
			p.pop();
			t++;
		}
	
		else
		{
			temp = c.front();
			c.pop();
			c.push(temp);
			t++;
		}
	}
	
	cout << t << endl ;
	
	
	return 0;
}
