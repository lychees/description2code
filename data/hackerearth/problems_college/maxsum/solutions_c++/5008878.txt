#include <iostream>
#include <vector>
#include <cstring>

using namespace std;

long dp[10001];
long A[10001];

long solve(long n)
{
	if(n < 0)
		return 0;
		
	long &ret = dp[n];
	
	if(ret != -1)
		return ret;
		
	ret = max(A[n] + solve(n-2), solve(n-1));
	
	return ret;
}

int main()
{
    long T;
    cin >> T;
    
    for(long t = 1; t <= T; t++)
    {
    	long N;
    	cin >> N;
    	
    	for(long i = 0; i < N; i++)
    		cin >> A[i];
    		
    	memset(dp, -1, sizeof(dp));
    	cout << solve(N-1) << endl;
    }
    
    return 0;
}
