#include<bits/stdc++.h>
using namespace std;
bool sum[10005][105];
int a[105];
int main()
{
    int t,i,j,k,l,m,n,o,p;
    ios::sync_with_stdio(0);
    cin>>t;
    while(t--)
    {
        memset(sum,0,sizeof(sum));p=0;
        sum[0][0]=1;
        cin>>n;
        for(i=1;i<=n;i++)
            cin>>a[i],sum[0][i]=1;
        for(i=0;i<=n*100;i++)
        {
            for(j=1;j<=n;j++)
            {
                sum[i][j]=sum[i][j-1];
                if(i>=a[j])
                    sum[i][j]=sum[i][j]||sum[i-a[j]][j-1];
            }
        }
        for(i=0;i<=n*100;i++)
            if(sum[i][n]==1)
                p++;//,cout<<i<<" ";
        cout<<p<<endl;
    }
}
