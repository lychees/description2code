#include <iostream>
#define ll long long int

using namespace std;

int main()
{
    ll t;
    cin >> t;
    
    while (t--) {
    	ll n;
    	cin >> n;
    	if (n == 0) {
    		cout << "0\n";
    	}
    	else if (n < 3) {
    		cout << "1" << endl;
    	}
    	
    	else if (n % 3 == 0) {
    		cout << n / 3 << endl;
    	}
    	else if (n % 3 == 1) {
    		cout << n << endl;
    	}
    	else {
    		cout << n << endl;
    	}
    }
    return 0;
}
