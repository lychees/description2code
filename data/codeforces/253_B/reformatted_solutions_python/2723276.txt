from bisect import bisect_right
import sys
sys.stdin = open('input.txt', 'r')
sys.stdout = open('output.txt', 'w')
n = int(input())
c = sorted(map(int, raw_input().split()))
if c[-1] <= c[0]*2:
    print(0)
    exit()
ans = 10**19
for a in range(n):
    b = bisect_right(c, 2*c[a])
    ans = min(ans, n-(b-a))
print(ans)
