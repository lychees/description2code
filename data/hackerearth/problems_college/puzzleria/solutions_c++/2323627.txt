#include <iostream>
#include<math.h>
using namespace std;

long long fact(long long x){
	if(x<=1)
		return 1;
	return x*fact(x-1);
}

int main()
{
	int t;
	cin>>t;
	while(t--){
    	long long x,y;
    	cin>>x>>y;
    	long long p1=fact(7)/fact(7-x)/fact(x),p2=fact(5)/fact(5-y)/fact(y),f=fact(x+y);
    	cout<<p1*p2*f<<endl;
    }
    return 0;
}
