#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main(){
    int T ;
    cin>>T ;
    while(T--){
        ll l , r , k ;
        cin >> l >> r >> k ;
        ll ret = 0 , u = l ;
        while(u <= r){
            u = u*k + 1 ;
            ret++ ;
        }
        cout << ret << endl ;
    }
}
