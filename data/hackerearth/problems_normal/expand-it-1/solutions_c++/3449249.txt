#include <iostream>
#define ll long long
using namespace std;

int main()
{	string s;
	cin>>s;
	ll int h[26]={0};
	char x;
	for(ll int i=0; s[i]!='\0';)
	{	x=s[i];
		i++;
		ll int num=0;
		while(s[i] >='0' && s[i]<='9')
		{
			num=num*10+(s[i]-'0')%10;
			i++;
		}
		h[x-'a']+=num;

    }
    ll int n;
    cin>>n;
    ll int in;
    for(ll int i=0; i<n; i++)
    {   cin>>in;
        if(h[0] > in)
            cout<<char(0+'a')<<endl;
        else
        {   ll sum=0;
        	for(ll int k=0; k<26; k++)
        	{	sum=sum+h[k];
        		
        	}
        	if(sum+1 >= in)
        	{
        		ll int k=0;
        
            	while(h[k] < in)
            	{
                	in=in-h[k];
                	k++;
            	}
            	cout<<char(k+'a')<<endl;
        	}
        	else
        		cout<<-1<<endl;

        }



    }

    return 0;
}
