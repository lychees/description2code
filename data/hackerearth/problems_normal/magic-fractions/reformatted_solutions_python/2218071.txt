i = 0
arr = []
while i <= 500:
    arr.append(True)
    i += 1

i = 2
while i <= 500:
    if arr[i]:
        j = 2*i
        while j <= 500:
            arr[j] = False
            j += i
    i += 1

p = []
p.append(1)
i = 1
while i <= 500:
    p.append(2*p[i-1])
    i += 1

ans = 0

s = input()
n = int(s)

cnt = 0

if n is 1:
    print("0")
else:
    i = 2
    while i <= n:
        if arr[i]:
            cnt += 1
        ans += p[cnt-1]
        i += 1
    print(ans)
