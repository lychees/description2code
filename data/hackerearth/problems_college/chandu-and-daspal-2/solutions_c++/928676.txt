using namespace std;
int findGcd(int x,int y)
{
    if(y==0)
       return x;
    return findGcd(y,x%y);
}       
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int x,y;
        string name1,name2;
        cin>>x>>y>>name1;
        if(name1.compare("Chandu")!=0)
             name2="Chandu";
        else
            name2="Daspal";
        int step=0;
        int gcd;
        while(x!=0 && y!=0)
        {
             gcd=findGcd(x,y);
             x-=gcd;
             y-=gcd;
             step+=1;
        }
        if(step%2==0)
            cout<<name2<<endl;
         else
            cout<<name1<<endl;   	
     }
   return 0;
}
