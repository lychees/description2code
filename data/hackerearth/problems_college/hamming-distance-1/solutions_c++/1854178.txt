#include <iostream>
#include <bitset>
using namespace std;

int main()
{
	int t;
	cin>>t;
	long long a,b;
	while(t--){
    	cin>>a>>b;
    	std::bitset<32> bin_a(a);
    	std::bitset<32> bin_b(b);
    	long cnt=0;
    	for(int i=0;i<32;i++){
    		if(bin_a[i]!=bin_b[i])
    			cnt++;
    	}
    	
    	cout<<cnt<<endl;
	}
    return 0;
}
