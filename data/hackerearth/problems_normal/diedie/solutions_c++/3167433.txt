#include<bits/stdc++.h>
using namespace std;
const long long int MOD=1E9 +7;

long long int power(long long int num,long long int value)
{
	long long int ans=1;
	while(value)
	{
		if(value%2==1)
		{
			ans*=num;
			ans%=MOD;
		}
		num=num*num;
		num%=MOD;

		value/=2;
	}
return ans;
}
int main()
{
	long long int n,test,i;
	cin>>test;
	for(i=0;i<test;i++)
	{
		cin>>n;

		long long int raise2=power(2,n-1);
		
		long long int ans=power(raise2, 1E9+5);
	cout<<ans<<endl;
	}

	return 0;
}
