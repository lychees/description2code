#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int i,n,x,y,ini[100000],final[100000],cost=0;
    	cin>>n>>x>>y;
    	for(i = 0 ; i < n ; i++)
    	cin>>ini[i]>>final[i];
    	
    	sort(ini+0,ini+n);
    	sort(final+0,final+n);
    	
    	for(i = 0 ; i < n ; i++)
    	{
    		if(final[i] > ini[i])
    		cost = cost + ((final[i]-ini[i])*x);
    		
    		else if(final[i] < ini[i])
    		cost = cost + ((ini[i]-final[i])*y);
    		
    		else
    		cost = cost;
    	}
    	cout<<cost<<endl;
    	
    	for(i = 0 ; i < n ; i++)
    	{
    		final[i] = 0;
    		ini[i] = 0;
    	}
    }
    return 0;
}
