#include <iostream>
#include <cmath>
#define nqw 5005
using namespace std;


int a[nqw+5],c[nqw+5];

void seive(){
    for (int i=2; i<=sqrt(nqw) ;i++) {
        if (a[i*i]==0) {
            a[i*i]=i*i;
            for (int j=2*(i*i); j<=nqw; j=j+i*i) {
                if(a[j]==0){
                    a[j]=i*i;
                }
            }
        }
    }
}

typedef long long ll;

int main() {
    seive();
    int n,b;
    scanf("%d",&n);
    for (int i=0; i<n; i++) {
        scanf("%d",&b);
        while(a[b]!=0){
            b=b/a[b];
        }
        c[b]++;
    }
    double ans=0.0;
    for (int i=1; i<5005; i++) {
        if(c[i]!=0){
            ans+=(double)c[i]*(c[i]-1);
        }
    }
    printf("%.6lf",ans/((double)n*(n-1)));
    return 0;
}
