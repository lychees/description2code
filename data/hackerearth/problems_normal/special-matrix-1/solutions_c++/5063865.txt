#include "bits/stdc++.h"
using namespace std;
long int abs(long int a,long int b)
{
	if( a > b )
		return a-b;
	return b-a;
}
int main()
{
	long int tc;
	cin>>tc;
	while(tc--){
	long int n;
	cin >> n;
	
	long int starx,stary;
	for(long int i=0;i<n;i++)
	{
		for(long int j=0;j<n;j++)
		{
			char a;
			cin >> a;
			if(a == '*')
			{
				starx=i;
				stary=j;
			}
		}
	}
	cout << abs(starx,n/2) + abs(stary,n/2) << endl;
	}return 0;
}
