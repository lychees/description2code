#include<string>
#include<stdio.h>
using namespace std;
int main()
{
	 int t;
	 cin>>t;
	 for(int k=0;k<=t;k++)
	 {
	 	fflush(stdin);
	 	char s[100000];
	 	char p[100000];
	 	int pi=0;
	 	//cin>>s;
	 	gets(s);
	 	for(int i=0;s[i]!='\0';i++)
	 	{
	 		if(s[i]>=97&&s[i]<=122||s[i]=='*')
	 		{
	 			p[pi++]=s[i];
	 		}
	 	}
	 	//cout<<p<<" "<<pi<<endl;
	 	int i=0;
	 	int j=pi-1;
	 	while(i<j)
	 	{
	 		if(p[i]=='*')
	 		{	
	 			cout<<p[j]<<endl;
	 			break;
	 		}
	 		else if(p[j]=='*')
	 		{	
	 			cout<<p[i]<<endl;
	 			break;
	 		}
	 		else
	 		{
	 			i++;
	 			j--;
	 		}
	 	}
	 }
}
