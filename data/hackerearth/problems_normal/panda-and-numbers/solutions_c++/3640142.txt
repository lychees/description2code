#include<iostream>
#include<algorithm>

using namespace std;

int const N = 1e6 + 41;
int was[N], ans[N], test, n;

void build(){
	for(int i=1;i<=7;++i){
		int cur = 1;
		for(int j=1;j<=i;++j){
			cur *= i;
		}
		was[cur] = ans[cur] = 1;
	}
}

int get(int v){
	if(v <= 0){
		return 0;
	}
	if(was[v]){
		return ans[v];
	}
	was[v] = 1;
	int ret = 0;
	for(int i=1;i<=v;i*=10){
		int dig = v / i % 10;
		ret += get(v - dig * dig);
	}
	if(ret > 0){
		ret = 1;
	}
	return ans[v] = ret;
}

int main(){
	build();
	scanf("%d",&test);
	for(int t=0;t<test;++t){
		scanf("%d",&n);
		if(get(n)){
			printf("Yes");
		}else{
			printf("No");
		}
		printf("\n");
	}
	
	
	
	
	
	return 0;
}
