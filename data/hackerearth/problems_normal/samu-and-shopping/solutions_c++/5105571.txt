#include <bits/stdc++.h>
using namespace std;

int a[100010][3];
int ans[100010][3];

int main()
{
    //cout << "Hello World!" << endl;
    int t, i, j, n;
    cin>>t;
    while(t--) {
    	cin>>n;
    	for(i=0; i<n; ++i)
    		for(j=0; j<3; ++j)	cin>>a[i][j];
    	for(i=0; i<3; ++i)	ans[0][i]=a[0][i];
    	for(i=1; i<n; ++i) {
    		ans[i][0]=min(ans[i-1][1]+a[i][0], ans[i-1][2]+a[i][0]);
    		ans[i][1]=min(ans[i-1][0]+a[i][1], ans[i-1][2]+a[i][1]);
    		ans[i][2]=min(ans[i-1][0]+a[i][2], ans[i-1][1]+a[i][2]);
    	}
    	cout<<min(ans[n-1][0], min(ans[n-1][1], ans[n-1][2]))<<endl;
    }
    return 0;
}
