#include<bits/stdc++.h>
using namespace std;
#define MAX 5010

long long dp[MAX],arr[MAX];

int main()
{
    int N;
    cin>>N;
    for(int i=0;i<N;i++)
    {
        cin>>arr[i];
        dp[i]=1;
    }

    int sign1,sign2;
    int max_val=1;

    for(int i=1;i<N;i++)
    {
        sign1=arr[i]>0?1:-1;

        for(int j=i-1;j>=0;j--)
        {
            sign2=arr[j]>0?1:-1;
            if(sign1!=sign2 && (abs(arr[i])>abs(arr[j])))
                dp[i]=max(dp[i],1+dp[j]);

        }
        if(dp[i]>max_val)
            max_val=dp[i];
    }
    cout<<max_val;
    return 0;
}
