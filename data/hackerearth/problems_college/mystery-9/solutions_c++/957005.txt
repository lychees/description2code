#include <iostream>
using namespace std;
int month(int d)
{
	if(d<=31)
	   return 1;
	else if(d>=32&&d<=59)
	  return 2;
	 
	  else if(d>=60&&d<=90)
	  return 3;
	  else if(d>=91&&d<=120)
	  return 4;
	  else if(d>=121&&d<=151)
	  return 5;
	  else if(d>=152&&d<=181)
	  return 6;
	  else if(d>=182&&d<=212)
	  return 7;
	  else if(d>=213&&d<=243)
	  return 8;
	  else if(d>=244 && d<=273)
	  return 9;
	  else if(d>=274 &&d<=304)
	  return 10;
	  else if(d>=305 &&d<=334)
	  return 11;
	  else
	  return 12;
}
int main()
{
    long long int t;
    cin >> t;
    while(t--)
    {
    	int n;
    	cin >> n;
    	int d=month(n);
    	cout << d <<"\n";
    }
    return 0;
    
}
