#include <iostream>
using namespace std ; 
int main()
{
	int n ; 
	cin >> n ; 
	string s ; 
	cin >>s ; 
	int q = 0 ;
	for ( int j = 0 ; j <s.size() ; j++)
	{
	
	int a[26] = {0} ;
	int p = 0; 
	 
	for (int  i = j; i< s.size() ; i++)
	{
		a[s[i]-'a']++ ; 
		if ( a[s[i]-'a']==1)
		{
		 p++ ; 
		 	
		}
		if( p == n )
		{
		q++ ; 	
		}
	}
}
	cout << q <<endl; 
	return 0 ; 

}
