#include <iostream>
#define ll long long
using namespace std;

int main()
{
    int t;
    cin>>t;
    ll n;
   
    ll a[100000];
    while(t--)
    {
    	 ll s=0;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		s+=a[i];
    	}
      cout<<s-n+1<<endl;
    	
    }
   
    return 0;
}
