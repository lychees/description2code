#include <iostream>
#include <bits/stdc++.h>
using namespace std;
long long int dp[1001][1001];
long long int solve(int n,int k){
    if(k>n)
        return 0;
    if(n==k)
        return 1;
    if(n<2*k)
        return 1;
    if(dp[n][k]!=-1)
        return dp[n][k];
    return dp[n][k]=solve(n-k,k)+solve(n,k+1);
}
int main()
{
   int t;
   cin>>t;
   while(t--){
        memset (dp,-1,sizeof(dp));

    int n,k;
    cin>>n>>k;
    cout<<solve(n,k)<<endl;;

   }


    return 0;
}
