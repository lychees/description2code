    #include <iostream>
    #include <bits/stdc++.h>
    using namespace std;
    set <string> s;
    int main()
    {
    string w;
    char ch[10005];
    int t,l,i;
    cin>>t;
    while(t--)
    {
    w="";
    scanf(" %[^\n]s",ch);
    l=strlen(ch);
    for(i=0;i<l;i++)
    {
    if(ch[i]==' ')
    {
    // cout<<w<<" "<<s.size()<<endl;
    s.insert(w);
    w="";
    }
    else
    {
    w+=ch[i];
    }
    }
    if(w!="")
    {
    s.insert(w);
    }
    cout<<s.size()<<endl;
    s.clear();
    }
    return 0;
    }
