#include<iostream>
using namespace std;
int main()
{
	int N,S[100000],i,j,k;
	cin>>N;
	for(i=0;i<N;i++)
	{
		cin>>S[i];
	}
	
	for(j=0;j<N;j++)
	{
		int count=0;
		for(k=0;k<N;k++)
		{
			if(S[j]%S[k]==0)
			{
				count++;
			}
		}
		if(count==1)
	
		cout<<S[j]<<" ";
	}
	return 0;
}
