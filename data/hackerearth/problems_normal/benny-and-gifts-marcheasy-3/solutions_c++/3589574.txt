#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int a[100005][2],n=0;
   	int x=0,y=0;
   	a[0][0]=0;
   	a[0][1]=0;
   	n++;
   	char s[100002];
   	cin>>s;
   	int count=0;
   	for(int i=0;s[i]!='\0';i++)
   	{
   		if(s[i]=='R')
   		{
   			y++;
   		}
   		if(s[i]=='L')
   		{
   			y--;
   		}
   		if(s[i]=='U')
   		{
   			x--;
   		}
   		if(s[i]=='D')
   		{
   			x++;
   		}
   		a[n][0]=x;
   		a[n][1]=y;
   		n++;
   		for(int j=0;j<n-1;j++)
   		{
   			if(a[j][0]==x && a[j][1]==y)
   			{
   				if(n>0)
   				n--;
   				count++;
   				break;
   			}
   			if(j>40000)
   			break;
   		}
   	}
   	cout<<count<<endl;
    return 0;
}
