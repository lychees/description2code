from itertools import product
t = int(input())
for z in range(t):
    rep = int(input())
    l = list(product('01', repeat=rep))
    num = len(l)
    for i in l:
        s = i
        for j in range(len(s)-1):
            if (s[j] == '1' and s[j] == s[j+1]):
                num = num - 1
                break
    num = num * num
    print(num)
