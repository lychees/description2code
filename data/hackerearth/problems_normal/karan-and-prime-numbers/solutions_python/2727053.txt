import math
for t in xrange(int(raw_input())):
    m,n=map(int,raw_input().split())
    s=0
    prime=[1]*(n+1)
    prime[0]=0
    prime[1]=0
    for i in xrange(2,int(math.sqrt(n)+1)):
        if prime[i]:
            for j in xrange(i*i,n+1,i):
                prime[j]=0

    for p in xrange(m,n+1):
        if prime[p]:
            s+=p
            
    print s
