#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

const int MAXN = 100005;

ll sum;
ll a[MAXN];
ll n, H;
ll res = 2e9 + 10;

ll bs(int past, ll cur) {
    if (cur >= H) {
        return 0;
    }
    ll l = 0, r = (int(2e9 + 10) / n) + 1, mid;
    while (l + 1 < r) {
        mid = (l + r) >> 1;
        ll val = cur + sum * mid;
        ll day = mid * n + past;
        ll rem = H - day * (day + 1) / 2;
        if (val >= rem) r = mid;
        else l = mid;
    }
    for (ll i = l; i <= r; i++) {
        ll val = cur + sum * i;
        ll day = i * n + past;
        ll rem = H - day * (day + 1) / 2;
        if (val >= rem) return i;
    }
    return -1;
}

void brute() {
    ll cur = 0;
    for (int day = 0, now = 0; ; day++) {
        cur += a[now++];
        if (now == n) now = 0;
        H -= (day + 1);
        if (cur >= H) {
            cout << day + 1 << '\n';
            return;
        }
    }
}
