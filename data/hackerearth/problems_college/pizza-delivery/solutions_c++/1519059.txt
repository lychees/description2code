#include <iostream>
using namespace std;

#define MAX 1000

int n,s,g,d;
int i,j;

long int route[101][101];
long int dist[101][101];

void find()
{
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		 dist[i][j]= route[i][j];
	}
	
	for(int k=0;k<n;k++)
	{
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			 {
			 	if(dist[i][j]> dist[i][k]+ dist[k][j])
			 	 dist[i][j]= (dist[i][k]+ dist[k][j]);
			 }
		}
	}
}


int main() {
	// your code goes here
	cin>>n;
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		 cin>>route[i][j];
	}
	find();
	int m;
	cin>>m;
	while(m--){
	cin>>s>>g>>d;
	long int p= dist[s][g]+ dist[g][d];
	cout<<p<<" "<<p- dist[s][d]<<endl;
	}
	return 0;
}
