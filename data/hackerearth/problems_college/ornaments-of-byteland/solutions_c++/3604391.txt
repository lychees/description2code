#include <iostream>
#include <cstring>
using namespace std;
char A[1001];
int A_map[26];
int A_count[26];
int main()
{
    int N,A_len,SP_count=0;
    cin>>N;
    for(int i=0;i<26;i++)
	{
		A_count[i]=0;
	}
    for(int j=0;j<N;j++)
    {
    	for(int i=0;i<26;i++)
        {
            A_map[i]=0;
        }
    	cin>>A;
    	A_len=strlen(A);
    	for(int i=0;i<A_len;i++)
    	{
    		A_map[A[i]-97]=1;
    	}
    	for(int i=0;i<26;i++)
        {
            if(A_map[i]==1)
            {
                    A_count[i]++;
            }
        }
    }
    for(int i=0;i<26;i++)
    {
    	if(A_count[i]==N)
    	{
    		SP_count++;
    	}
    }
    cout<<SP_count;
    return 0;
}
