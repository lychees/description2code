#include<stdio.h>
#include<iostream>
#include<vector>
#include<string.h>

using namespace std;

long int hsh[100][10000];
int main()
{


    memset(hsh,-1,sizeof(hsh));

    int n, i, x, y, val;
    scanf("%d",&n);

    int a[n];

    int flag[101]={0};

    for(i=0; i<n; i++)
    {
        scanf("%d",&a[i]);

        hsh[a[i]][flag[a[i]]]=i;

        flag[a[i]]++;
    }

    long int q;
    scanf("%ld",&q);

    while(q--)
    {
        scanf("%d %d %d",&x,&y,&val);

        int c=0, f=0;

        while(hsh[val][f]!=-1)
        {
            if( x <= hsh[val][f] && hsh[val][f] <= y )
            {   c++;
                //printf("c=%d x=%d\n",c,x);
            }

            //printf("hsh[%d][%d]=%d c=%d \n",val,f,hsh[val][f],c);
            f++;
        }

        printf("%d\n",c);

    }

    return 0;
}
