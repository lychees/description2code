#include <iostream>
using namespace std;

int main()
{
int t;
cin>>t;
for(int i=0;i<t;i++){
	int n,m;
	cin>>n>>m;
	int arr[n];
	for(int j=0;j<n;j++){
		cin>>arr[j];
	}
	int prev;
	int p[n];
	for(int l=0;l<n;l++){
		p[l] = 0;
	}
	for(int j=0;j<m;j++){
		for(int k=0;k<n;k++){
			if(k==0)
			{
				prev=arr[k];
				if(arr[k+1] == 1){
					arr[k]=1;
				}
				else{
						arr[k] = 0;
				}
			}
			else if(k==n-1){
			
				if(prev == 1){
					arr[k]=1;
				}
				else{
						arr[k] = 0;
				}
			}
			else{
				if(prev == 1&&arr[k+1] == 1){
					prev = arr[k];
					arr[k] = 1;
				}
				else{
					prev =arr[k];
				arr[k]=0;
				
					
				}
			}
			
		}
		int l;
		for( l=0;l<n;l++){
			if(arr[l] != p[l])
			{
				break;
			}
		}
		if(l==n)
		break;
		for( l=0;l<n;l++){
			p[l] = arr[l];
		}
		
	}
	for(int j=0;j<n;j++){
		cout<<arr[j]<<" ";
	}
	cout<<"\n";
}
    return 0;
}
