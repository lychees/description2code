#include<stdio.h>
#include<limits.h>
#include<algorithm>
long long int a[100005];
using namespace std;
int main()
{
	long long int t, i, max, sum, n,j;
	scanf("%lld", &t);
	while (t--)
	{
		max = INT_MIN;
		scanf("%lld", &n);
		for (i = 0; i < (2 * n); i++)
		{
			scanf("%lld", &a[i]);
		}
		sort(a, a + (2 * n));
		j = (2 * n) - 1;
		for (i = 0; i < n; i++)
		{
			sum = a[i] + a[j];
			j--;
			if (sum > max)
				max = sum;
		}
		printf("%lld\n", max);
	}
	getchar();
	getchar();
	return 0;
}
