#include <iostream>
using namespace std;
typedef long long int ll;
ll f[1001001];

ll divd(ll a, ll b, ll mod){
	ll ans = 1;
	while(b > 0){
		if(b % 2 == 1){
			ans = (ans * a) % mod;
		}
		a = (a * a) % mod;
		b = b / 2;
	}
	return ans;
}

ll cal(ll n, ll r, ll mod){
	ll ans = f[n];
	ans *= divd(f[r], mod - 2, mod);
	ans %= mod;
	ans *= divd(f[n-r], mod-2, mod);
	ans %= mod;
	return ans;
}

int main() {
	ll t, i;
	f[0] = 1;
	ll mod = 1000000007;
	for(i = 1; i < 1001001; i++){
		f[i] = (i * f[i-1]) % mod;
	}
	cin >> t;
	while(t--){
		ll n, m;
		cin >> n >> m;
		ll a;
		a = n * n;
		a = a - 2;
		a = a + m;
		//cout << a << "___" << m << "__"<< f[a] <<  endl; 
		ll ans = cal(a, m, mod);
		cout << ans << endl;
	}
	return 0;
}
