#include <iostream>
using namespace std;

int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		long int a,b;
		cin>>a>>b;
		int c=0;
		int count=0;
		while(a||b)
		{
			int d=a%10+b%10+c;
			if(d>9){
			count++;c=1;}
			a=a/10;
			b=b/10;
		}
		if(count==1)
		cout<<count<<" carry operation"<<endl;
		else if(count>0)
		cout<<count<<" carry operations"<<endl;
		
		else cout<<"No carry operation"<<endl;
		c=0;
		count=0;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
