using namespace std;

#include <bits/stdc++.h>
#define ll long long int
#define s(a) scanf("%lld",&a)

ll mod=1000000009;

map<ll,ll>m;

void multiply(ll f[2][2],ll m[2][2])
{
	ll a,b,c,d;
	a=f[0][0]*m[0][0]+f[0][1]*m[1][0];
	b=f[0][0]*m[0][1]+f[0][1]*m[1][1];
	c=f[1][0]*m[0][0]+f[1][1]*m[1][0];
	d=f[1][0]*m[0][1]+f[1][1]*m[1][1];
	f[0][0]=a%mod;
	f[0][1]=b%mod;
	f[1][0]=c%mod;
	f[1][1]=d%mod;
}

void power(ll f[2][2],ll n)
{
	if(n==0||n==1) return;
	ll m[2][2]={{1,1},{1,0}};
	power(f,n/2);
	multiply(f,f);
	if(n%2==1) {
		multiply(f,m);
	}
}

void fibo(ll n)
{
	ll f[2][2]={{1,1},{1,0}};

	if(n>0)
		power(f,n-1);
	
	printf("%lld\n", (f[0][0]*f[0][1])%mod);
}

int main()
{
	ll t,x,y,k,i;
	//double p=1.6180339;
	t=0;
	s(t);
	while(t--) {
		s(x);
		i = 0;
		fibo(x+1);
	}
	return 0;
}
