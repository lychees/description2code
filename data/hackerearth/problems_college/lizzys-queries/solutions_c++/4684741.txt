#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	char str[100005];
	int n,q,l,r,i,f,w,j,p;
	char x;
	scanf("%d%d",&n,&q);
	scanf("%s",str);
	while(q--)
	{
		scanf("%d",&w);
		if(w==0)
		{
			scanf("%d %c",&p,&x);
			str[p-1]=x;
		}
		else
		{
			scanf("%d%d",&l,&r);
			f=1;
			for(i=l-1,j=r-1;i<=j;i++,j--)
			{
				if(str[i]!=str[j])
				{
					f=0;
					break;
				}
			}
			if(f)
			{
				printf("Yes\n");
			}
			else
			{
				printf("No\n");
			}
		}
	}
    return 0;
}
