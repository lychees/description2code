#include <iostream>
using namespace std;
double d[32],f[32],an[32];
int main()
{
    int t,n,i,ind;
    double den,max,ans;
    cin >> t;
    while(t--){
    	cin >> n;
    	for(i = 0;i < n;i++){
    		cin >> d[i];
    	}
		for(i = 0;i < n;i++){
			cin >> f[i];
		}
		for(i = 0;i < n;i++){
			an[i] = d[i]*f[i];
		}
		den = 0;
		for(i = 0;i < n;i++){
			den = den + an[i];
		}
		max = -1;
		for(i = 0;i < n;i++){
			if(an[i] > max){
				max = an[i];
				ind = i;
			}
		}
		ans = max/den;
		printf("%d %.3lf\n",ind,ans);
    }
    return 0;
}
