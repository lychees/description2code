#include<iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long int n,k,m;
		cin>>n>>k>>m;
		int p[n];
		for(int i=1;i<=n;i++)
		cin>>p[i];
		
		long long int time=0;
		long long int minn;
		long int ind;
		long long int pt;
		
		for(int i=1;i<=n;i++)
		{
			time=0;
		     long long w = (p[i] - 2) + 1;
			if(w > 0 )
			time = (w * m);
			time+=p[i]*k;
			for(int j=1;j<=p[i];j++)
			{
				cin>>pt;
				time=time+pt;
			}
			if(i==1)
			{
				ind=1;
				minn=time;
			}
			else if(minn>time)
			{
				ind=i;
				minn=time;
			}
		}
		cout<<ind<<" "<<minn<<"\n";
	}
}
