#include <iostream>
using namespace std;

int main()
{int t,l,i,j,n;
float p;
string s;
  cin>>t;
  while(t--)
  {
  	cin>>s;
  	l=s.length();
  	p=(float(l))/2;
  	j=0;n=0;
  	for(i=0;i<l;i++)
  	{
  		if(s[i]=='a' || s[i]=='e' || s[i]=='i' || s[i]=='o' || s[i]=='u' || s[i]=='y')
  		{
  			j++;n=0;
  		}else{n++;if(n==3){cout<<"hard\n";goto lab;}}
  		
  	}
  	if(j<p){cout<<"hard\n";}else{cout<<"easy\n";}
  	
  lab:	
  ;}
    return 0;
}
