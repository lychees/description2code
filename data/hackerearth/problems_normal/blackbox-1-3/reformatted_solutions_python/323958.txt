from fractions import gcd
t = int(raw_input())
a = [0]*3
for i in range(t):
    n = raw_input().split()
    a[0] = (long)(n[0])
    a[1] = (long)(n[1])
    a[2] = (long)(n[2])
    a.sort()
    # print a[0] ,a[1] ,a[2]
    if a[0]+a[1] <= a[2]:
        print "NO"
        continue
    if gcd(a[0], a[1]) == 1 and gcd(a[2], a[1]) == 1 and gcd(a[0], a[2]) == 1:
        if (a[0]*a[0]+a[1]*a[1]) == (a[2]*a[2]):
            print "YES"
        else:
            print "NO"

    else:
        print "NO"
