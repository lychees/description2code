#include<iostream>
#include<vector>
#include<map>
#include<string>
#include<algorithm>
#include<math.h>
#include<functional>
#include<list>
#include<queue>

using namespace std;

int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		char str[105];
		scanf("%s",str);
		int occur[]={0,0,0,0};
		int i=0;
		while(str[i]){
			if(str[i]=='r')
				occur[0]++;
			else if(str[i]=='u')
				occur[1]++;
			else if(str[i]=='b')
				occur[2]++;
			else if(str[i]=='y')
				occur[3]++;
			i++;
		}
		int result = occur[0];
		if(result>occur[1])
			result = occur[1];
		if(result>occur[2])
			result = occur[2];
		if(result > occur[3])
			result = occur[3];
		printf("%d\n",result);

	}
	return 0;
}
