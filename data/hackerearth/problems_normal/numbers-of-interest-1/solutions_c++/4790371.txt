#include <iostream>
using namespace std;

int main()
{
    int cases;
    cin>>cases;
    
    long int x, y, n,l,k;
    long int series[1000];
    	
    
    for(int i=0; i<cases; i++)
    {
    
    	cin>>x>>y>>n;
    	
    	for(k=0; k<n; k++){
    	
    	if(k<x)
    	{
    		series[k] = y;
    	} 
    	
    	else if(k>=x)
    	{
    		series[k] = 0;
    		for(l = k-x; l<k; l++)
    		{
    			series[k]+=series[l];
    		}
    	}
    	}
    
    	cout<<series[n-1]<<endl;
    }
    
    
    
    return 0;
}
