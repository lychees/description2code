#include <iostream>
using namespace std;

int main()
{
	long long t,n,count,maxm,temp, temp2 ;
	cin>>t;
	for(long long i = 0 ; i< t;i++){
		cin>>n;
		if(n==1LL){
		maxm = 0;
		count = 1;
		}
		else{
			maxm = n-1;
			int x = 1;
			while(x<=64){
				maxm=maxm|maxm>>x;
				x<<=1;
			}
			temp  =maxm>>1;
			temp2 = 1LL;
			while(temp!=0){
				temp2 = temp2<<1;
				temp>>=1;
			}
			count = (n-temp2)<<1;
		}
		cout<<maxm<<" "<<count<<endl;
	}
return 0;
}
