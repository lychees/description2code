#include<bits/stdc++.h>
using namespace std;

main(){
 ios_base::sync_with_stdio(0);
 long long t,x,y;
 cin>>t;
 for(int i=0;i<t;i++){
  int a,b,count=0,c=0;
  cin>>x>>y;
  while(x!=0){
    a=x%10;b=y%10;
    if(a+b+c>9){
     count++;
     c=1;
    }
    else{
     c=0;
    }
    x=x/10;y=y/10;
  }
  if(count==0){cout<<"No carry operation"<<endl;}
  else if(count==1){cout<<count<<" carry operation"<<endl;}
  else {cout<<count<<" carry operations"<<endl;}
 }
}
