#include <iostream>
typedef long long int uli;
using namespace std;

int main()
{
    int t,n;
    uli r,c,v;
    scanf("%d",&t);
    while(t--){
    	scanf("%d",&n);
    	scanf("%lld %lld %lld",&r,&c,&v);
    	uli d=((r-1)+(c-1))%2;
    	int color=(v%2)^d;
    	bool ok=true;
    	for(int i=1;i<n;i++){
    		scanf("%lld %lld %lld",&r,&c,&v);
    		d=((r-1)+(c-1))%2;
    		int col=color^d;
    		if(col!=v%2)ok=false;
    	}
    	if(ok)puts("Yes");
    	else puts("No");
    }
    return 0;
}
