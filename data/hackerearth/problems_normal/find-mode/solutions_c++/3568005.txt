#include <bits/stdc++.h>
using namespace std;
int dp[1000001];
int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n,i,a;
    	cin>>n;
    	memset(dp,0,sizeof(dp));
    	for(i=0;i<n;i++) {
    		cin>>a;
    		dp[a]++;
    	}
    	int ans=INT_MIN;
    	for(i=0;i<1000001;i++) {
    		if(dp[i]>ans) {
    			ans=dp[i];
    		}
    	}
    	for(i=1000000;i>=0;i--){
    		if(dp[i]==ans){
    			cout<<i<<" ";
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
