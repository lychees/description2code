int scan()
{
    int n=0;
    char ch=getc_unlocked(stdin);
 
    /*while (ch<'0'||ch>'9')

    while (ch>='0'&&ch<='9')
    {
        n=n*10+ch-'0';

    }
 
    return n;
}
 

void print(int n)
{
    int i=0;
    char num[20];
    
    do
    {
        num[i++] = n%10+'0';
        n/=10;
    }while(n);
    
    i--;
    
    while(i>=0)
    {
        putc_unlocked(num[i--],stdout);
    }
    
    putc_unlocked('\n',stdout);
}
