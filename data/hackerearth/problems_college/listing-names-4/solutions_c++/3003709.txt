#include <iostream>
#include <string>
#include <cstring>
#include <math.h>
using namespace std;
int main()
{
    long long int t,c,i,j;string s[10000],temp;
    cin>>t;
    for(i=0;i<t;i++)
        cin>>s[i];
    for(i=0;i<t-1;i++)
    {
        for(j=0;j<t-i-1;j++)
        {
            if(s[j]>s[j+1])
            {
                temp=s[j];
                s[j]=s[j+1];
                s[j+1]=temp;
            }
        }
    }
    c=1;
    for(i=1;i<t;i++)
    {
        if(s[i]==s[i-1])
            c++;
        else
        {
            cout<<s[i-1]<<" "<<c<<"\n";
            c=1;
        }
    }
    cout<<s[t-1]<<" "<<c<<"\n";
return 0;
}
