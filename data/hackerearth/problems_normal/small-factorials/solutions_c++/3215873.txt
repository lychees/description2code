#include <iostream>
using namespace std;
#define MAX 500

int multiply(int res[],int m,int s)
{
	int c=0,i=0;
	for(i=0;i<s;i++)
	{
		int r=(m*res[i]+c);
		res[i]=r%10;
		c=r/10;
	}
	while(c!=0)
	{
		res[i++]=c%10;
		c=c/10;
	}
	return i;
}

void factorial(int n)
{
	int res[MAX];
	res[0]=1;
	int s=1;
	for(int i=2;i<=n;i++)
		s=multiply(res,i,s);
		
	for(int i=s-1;i>=0;i--)
		cout<<res[i];
		cout<<endl;
}

int main()
{
    int t;
    cin>>t;
    int n;
    while(t--)
    {
    	cin>>n;
    	factorial(n);
    }
    return 0;
}
