#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n;
    	cin>>n;
    	if(n<3)
    	cout<<-1<<endl;
    	else
    	{
    		if(n%3==0 && n%5!=0)
    		{
    			for(int i=0;i<n;i++)
    			cout<<"5";
    			cout<<endl;
    		}
    		else if(n%5==0 && n%3==0)
    		{
    			for(int i=0;i<n;i++)
    			cout<<"5";
    			cout<<endl;
    		}
    		else if(n%5==0 && n%3!=0)
    		{
    			for(int i=0;i<n;i++)
    				cout<<"3";
    				cout<<endl;
    		}
    		else
    		{
    			int j=5;
    			int n1=n;
    			int n2=n-j;
    			int flag=0;
    			while(n2>0)
    			{
    			    if((n2)%3==0)
    			      {
    			      	flag=1;
    			      	for(int i=0;i<n2;i++)
    			      	cout<<5;
    			      	//cout<<endl;
    			      	for(int i=0;i<j;i++)
    			      	cout<<3;
    			      	cout<<endl;
    			      }
    			      if(flag==1)
    			      break;
    			 j=j+5;
    			 n2=n2-j;
    			}
    			int j1=3;
    			int n3=n-j1;
    			if(flag==0)
    			{
    				while(n3>0)
    				{
    					if(n3%5==0)
    					{
    						flag=1;
    						for(int i=0;i<j;i++)
    						cout<<5;
    						for(int i=0;i<n3;i++)
    						cout<<3;
    						cout<<endl;
    					}
    					j=j+3;
    					n3=n3-j;
    					if(flag==1)
    					break;
    				}
    			}
    			if(flag==0)
    			cout<<"-1"<<endl;
    		}
    	}

    }

    return 0;
}
