void reverse(char arr[],int l)
{
	int i,j;
	for(int i =0,j=l-1;i<l/2;i++,j--)
	{
		char t;
		t = arr[i];
		arr[i] = arr[j];
		arr[j] = t;
	}
	cout<<arr<<endl;
}
