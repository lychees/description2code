#include <iostream>
#include <cstring>
#include <cstdio>
#include <cmath>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	char s[105];
	while(t--)
	{
		double sum = 0;		int len ;
		getchar();
		gets(s);
		len = strlen(s) ;
		
		for(int i = 0 ; i < len ; i++){
			sum += s[i] ;
		}
		
		int f = floor(sum/len) ;
		
		if(f&1)
		{
			for(int i = len ; i > 0 ; i--)
				cout << s[i-1] ;
				
			cout << "\n" ;
		}
		else
			cout << s << "\n";
	}

	return 0;
}
