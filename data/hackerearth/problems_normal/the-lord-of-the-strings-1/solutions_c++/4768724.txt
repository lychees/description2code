#include<iostream>
#include<stdio.h>
#include<string.h>
#include<map>

using namespace std;

int main()
{

	int t;
	scanf("%d",&t);
	
	while(t--)
	{
		char s1[100],s[100];
		scanf("%s",&s);

		int len=strlen(s),itr=0,count=0;
		
		map<char,int> hchar;

		for(int i=0;i<len;i++)
		{

			if(hchar[s[i]]==0)
			{
				hchar[s[i]]=1;
				s1[itr++]=s[i];
				count++;
			}

		}
		

		printf("%.*s\n",count,s1);
		hchar.erase(hchar.begin(),hchar.end());
	}

	return 0;
}
