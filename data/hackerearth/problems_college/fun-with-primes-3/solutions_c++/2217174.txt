#include<bits/stdc++.h>
using namespace std;
int p[1000002];
void fun()
{
	int num,j;
	p[0]=1;
	p[1]=1;
	for(int i=2;i<100002;i++)
	{
		 if(p[i]==0)
		 {
		 	j=2;
		 	while((num=i*j)<1000002)
		 	{
		 		p[num]=1;
		 		j++;
		 	}
		 }
	}
}
int main()
{
	fun();
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int x=n,sum=0;
		while(x)
		{
			sum+=x%10;
			x/=10;
		}
		if(p[sum]==1)
			cout<<"NO"<<endl;
		else
			cout<<"YES"<<endl;
	}
}
