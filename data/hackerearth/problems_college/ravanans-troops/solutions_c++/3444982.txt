#include<bits/stdc++.h>
using namespace std;

int dp[1000015];

void solve(int n)
{
    int i, j, tmp, tmp1;
    dp[0] = 0;
    dp[1] = 1;
    dp[2] = 2;
    dp[3] = 3;

    for(i = 4; i <= n; i++) {
        tmp1 = i;
        for(j = 1; j <= i; j++) {
            tmp = j * j;
            if(tmp > i) break;
            tmp1 = min(tmp1, dp[i - tmp]);
        }
        dp[i] = tmp1 + 1; 
    }
}

int main()
{
    solve(1000000);
    int t, n;
    scanf("%d", &t);
    while(t--) {
        scanf("%d", &n);
        printf("%d\n", dp[n]);
    }

    return 0;
}
