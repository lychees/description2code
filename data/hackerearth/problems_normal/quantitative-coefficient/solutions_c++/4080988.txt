int par[10003];

int fp(int x)
{
	if(x != par[x])
		par[x] = fp(par[x]);
		
	return par[x];
}

ll ans;

void join(int x, int y, int c)
{
	x = fp(x);
	y = fp(y);
	
	if(x == y)
		return;
	par[x] = y;
	ans = (ans*(ll)c) % modul;
}


int main()
{
	int t,n,m,i,a,b,c;
	
	cin >> t;
	
	while(t--) {
		cin >> n >> m;
		
		vector<pair<int, pii > >edge;
		For(i, 0, m) {
			sc(a), sc(b), sc(c);
			
			edge.pb(mp(c, mp(a, b)));
		} 
		
		sort(edge.begin(), edge.end());
		
		For(i, 1, n+1)
			par[i] = i;
		
		ans = 1;	
		For(i, 0, m) {
			join(edge[i].se.fi, edge[i].se.se, edge[i].fi);
		}
		
		cout << ans << endl;
		
		
	}
	return 0;
}
