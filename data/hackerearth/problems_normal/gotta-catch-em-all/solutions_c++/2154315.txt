#include <iostream>
#include<algorithm>
using namespace std;

int wayToSort(int i, int j)
{
	return i>j;
}



int main()
{
	int arr[100004],k,i;
cin>>k;
int val;
for(i=0;i<k;i++)
	cin>>arr[i];
	
sort(arr,arr+k,wayToSort);

int max= 0 + arr[0] +2;

for(i=1;i<k;i++)
{
	val= i+ arr[i] +2;
	if(val>max)
		max =val;
}
cout<<max;
    
    
    return 0;
}
