#include<bits/stdc++.h>
using namespace std;
#define MAX 1000006
int arr[MAX];
bool dig_sum(int n)
{
    int i,j,r,flag=0;
    while(n!=0)
    {
        r=n%10;
        if(r==7)
            flag=1;
        n=n/10;
    }
    if(flag==1)
        return true;
    else
        return false;
}

bool prime(int n)
{
    int i;
    if(n==2)
        return true;
    if(n%2==0||n==1||n==2)
        return false;
    for(i=3;i<=sqrt(n);i=i+2)
    {
        if(n%i==0)
            return false;
    }
    return true;
}

int main()
{
    memset(arr, 0, sizeof(arr));
    for(int i=1;i<MAX;i++)
    {
        if(!prime(i) && dig_sum(i))
            arr[i] = 1;
    }
    int i,j,k,n,m,c,t;
    scanf("%d",&t);
    while(t--){
        c=0;
    scanf("%d%d",&m,&n);
    for(i=m;i<=n;i++)
    {
        if(arr[i] == 1) {
            c++;
        }
    }
    if(c==0)
        printf("-1\n");
    else
        printf("%d\n",c);
    }
    return 0;
}
