import sys

if __name__ == '__main__':
    for code in sys.stdin:
        try:
            comment = code[code.index("//"):]
            working_str = code[:code.index("//")]
        except ValueError:
            working_str = code
            comment = ""

        working_str = working_str.replace("->", ".")
        working_str += comment
        sys.stdout.write("%s" % (working_str))
