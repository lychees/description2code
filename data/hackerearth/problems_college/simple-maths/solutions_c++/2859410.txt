#include<bits/stdc++.h>
using namespace std;

long long int mod_pow(long long int a, long long int b,long long int mod)
{
	long long int ans=1;
	while(b)
	{
		int r=b%2;
	
		if(r!=0)
		ans=(ans*a)%mod;
		
		a=(a*a)%mod;
		
		b/=2;
	}	
	return ans;
}

long long gcdf(long long a,long long b)
{
	if(a==0)
	return b;
	return gcdf(b%a,a);	
}

long long min1(long long a,long long b)
{
	return a>b?b:a; 	
}


int main() {
int t;
scanf("%d",&t);

for(;t>0;t--)
{
long long int a,b,n,i=1,c;
scanf("%lld %lld %lld",&n,&a,&b);

if(a==0 || b==0)
printf("-1\n");
else
{
c=n/a+n/b;

if(a<b)
{
	long long int tmp=a;
	a=b;
	b=tmp;
}
for(i=a;i<=n;i+=a)
{
	if(i%b==0)
	c--;
}

printf("%lld\n",c);
	
}
	
}	
return 0;
}
