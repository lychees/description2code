#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int k,len,i,temp=97,x=0,y=0,temp1=65;
        string s;
        char a[28],b[28];
        cin>>k;
        cin>>s;
        len=s.length();
        for(i=0;i<26;i++){a[i+1]=temp;temp++;}
        for(i=0;i<26;i++){b[i+1]=temp1;temp1++;}
        for(i=0;i<len;i++)
        {
            if(s[i]>=97){
            y=s[i]-'a';
            y++;
            y+=k;
            if(y>=26){if(y%26==0)y=26;else y=y%26;}
            cout<<a[y];}
            else
            {
                y=s[i]-'A';
                y++;
                y+=k;
                if(y>=26){if(y%26==0)y=26;else y=y%26;}
                cout<<b[y];
            }
        }
        cout<<endl;
    }
    return 0;
}
