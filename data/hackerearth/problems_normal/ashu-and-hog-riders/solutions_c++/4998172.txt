#include<bits/stdc++.h>
using namespace std;
#define LL long long
#define S 1000005
#define W while
#define M 1000000007
#define str string
#define pi 3.14159265
LL mini=INT_MAX;
LL maxi=INT_MIN;
LL visited[1001][1001];
int n;
LL a[1001][1001];
LL bfs(LL i,LL j)
{
	LL ma=0;
	queue<pair<LL,LL> > q;
	q.push(make_pair(i,j));
	visited[i][j]=1;
	W(!q.empty())
	{
		i=q.front().first;
		j=q.front().second;
		q.pop();
		ma++;
		if(i+1<n&&visited[i+1][j]==0&&a[i+1][j]!=0)
		{
			visited[i+1][j]+=(visited[i][j]+1);
			q.push(make_pair(i+1,j));
		}
		if(i-1>-1&&visited[i-1][j]==0&&a[i-1][j]!=0)
		{
			visited[i-1][j]+=(visited[i][j]+1);
			q.push(make_pair(i-1,j));
		}
		
		if(j+1<n&&visited[i][j+1]==0&&a[i][j+1]!=0)
		{
			visited[i][j+1]+=(1+visited[i][j]);
			q.push(make_pair(i,j+1));
		}
		
		if(j-1>-1&&visited[i][j-1]==0&&a[i][j-1]!=0)
		{
			visited[i][j-1]+=(visited[i][j]+1);
			q.push(make_pair(i,j-1));
		}
	}
	return ma;
}
int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
			cin>>a[i][j];
	}
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			if(visited[i][j]==0&&a[i][j])
				maxi=max(maxi,bfs(i,j));
		}
	}
	cout<<maxi;
	return 0;
}
