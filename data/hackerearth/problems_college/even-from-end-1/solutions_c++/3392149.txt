#include <iostream>
using namespace std;

int main()
{
    int n;
    int a[100009];
    
    while(cin>>n){
    	
    
    	for(int i = 0; i < n; i++){
    		cin>>a[i];
    	}
    	int p = 0;
    	for(int j = n-1; j >= 0; j--){
    		
    		if(a[j]%2 == 0){
    			cout<<a[j]<<" ";
    			p++;
    		}
    	}
    	if(p == 0){
    		cout<<"None to display";
    	}
    	cout<<endl;
    
    }
}
