"""
program to determine win or loose in the word game
"""

def game():
	"""
	inputs two dictcs for words and its count and for one player its +ve and other its -ve
	prints the winner or looser or a draw based on the remaining keys in dict
	"""
	for _ in range(input()):
		mydicta = {}
		mydictb = {}
		mystra = [alfa for alfa in raw_input() if alfa != ' ']
		mystrb = [alfa for alfa in raw_input() if alfa != ' ']
		for val in mystra:
			if mydicta.get(val, None) == None:
				mydicta[val] = 1
			else:
				mydicta[val] += 1
		for val in mystrb:
			if mydictb.get(val, None) == None:
				mydictb[val] = 1
			else:
				mydictb[val] += 1
		keyset = set((mydicta.keys()) + (mydictb.keys()))
		for key in keyset:
			if mydicta.get(key, None) != None and mydictb.get(key, None) != None:
				if mydicta[key] > mydictb[key]:
					del mydictb[key]
				elif mydicta[key] == mydictb[key]:
					del mydicta[key]
					del mydictb[key]
				else:
					del mydicta[key]
		aval = len(mydicta.keys())
		bval = len(mydictb.keys())
		if aval > 0 and bval == 0:
			print "You win some."
		elif bval > 0 and aval == 0:
			print "You lose some."
		else:
			print "You draw some."

if __name__ == '__main__':
	game()
