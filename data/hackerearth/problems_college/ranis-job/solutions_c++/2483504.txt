#include<bits/stdc++.h>
using namespace std;

bool mcmp(int a,int b){
	return a > b;
}
int main()
{
	int t;
	int n;
	int sm=0;
	vector<int>A,B;
	cin >> t;
	while(t--){
		sm = 0;
		cin >> n;
		A.resize(n);
		B.resize(n);
		for(int i=0;i < n;i++){
			cin >> A[i];
			sm += A[i];
		}
		for(int i=0;i < n;i++){
			cin >> B[i];
		}
		sort(B.begin(),B.end(),mcmp);
		//cout << B[0];
		for(int i=0;i < n;i++){
			if(sm <= B[i]){
				cout << i + 1 << endl;
				break;
			}else{
				sm = (sm-B[i]);
			}
		}
		A.clear();
		B.clear();
	}
	return 0;
	
}
