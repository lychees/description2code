#include<iostream>
#include<math.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	for(int k=0;k<t;++k){
	string s;
	cin>>s;
	int hash[26]={0};
	for(int i=0;i<s.length();++i)
	hash[s[i]-'a']++;



	int x =0;
	for(int i=0;i<26;++i){
		if(x<hash[i]){
		x = hash[i];
		
	}
	}
int	j=0;
	while(j<=x){
		for(int i=25;i>=0;--i){
			
			if(hash[i]==j){
				cout<<(char)(i+97)<<" ";
			}
			
			
		}
		j++;
	}
	cout<<endl;
}
	return 0;
}
