#include <iostream>
#include <stdio.h>
#include <vector>
#include <utility>
#include <map>
using namespace std;



int main()
{
	map <long long int, int> mp;
	map <long long int, int> :: iterator it;
	int N;
	cin >> N;
	while(N--)
	{
		long long int x;
		cin >> x;
		++mp[x];
	}
	long long int ans = -1;
	it = mp.begin();
	while(it != mp.end())
	{
		if(it->second == 1 && it->first > ans)
		{
			ans = it->first;
		}
		++it;
	}
	cout << ans << endl;
	return 0;
}
