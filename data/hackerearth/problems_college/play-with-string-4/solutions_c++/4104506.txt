#include <iostream>
#include <string.h>
#include <math.h>
using namespace std;

int main()
{	int t,i,j,c,a,b;
	char s[100001],r[100001];
	cin>>t;
	while(t--){
	cin>>s;c=0;
	for(j=0,i=strlen(s)-1;i>=0;i--,j++)
	r[j]=s[i];
	for(i=1;i<strlen(s);i++){
		a=s[i]-s[i-1];b=r[i]-r[i-1];
		if(fabs(a)!=fabs(b))
		{c=1;break;}
	}
	if(c==0)
	cout<<"Funny\n";
	else
	cout<<"Not Funny\n";
	}
    return 0;
}
