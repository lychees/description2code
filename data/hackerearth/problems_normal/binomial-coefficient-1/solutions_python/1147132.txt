import sys
f = sys.stdin
mod = 1000000007

def inv(x,y):
	res = 1
	while y:
		if y%2:
			res = res*x%mod
		y/=2
		x=x*x%mod
	return res
	
a,b,n,k = [int(x) for x in f.readline().split()]
num = 1
deno = 1
for i in range(1,n+1):
	num = num*i%mod

for i in range(1,k+1):
	deno = deno*i%mod
	num = num*a%mod
for i in range(1,n-k+1):
	deno = deno*i%mod
	num=num*b%mod

print num*inv(deno,mod-2)%mod
