#include <bits/stdc++.h>
using namespace std;

vector<pair<int, pair<long, long>>> v;

int main() {
  ios::sync_with_stdio(false);
  int n, q;
  cin >> n >> q;
  for (int i = 0; i < n; i++) {
    int x;
    long s, c;
    cin >> x >> s >> c;
    v.push_back(make_pair(x, make_pair(s, c)));
  }
  sort(v.begin(), v.end(), cmp);
  return 0;
}
