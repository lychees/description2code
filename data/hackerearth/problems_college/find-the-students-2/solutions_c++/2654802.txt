#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int ar[1000],arr[1000][2],x,k;
    cin>>ar[0];
    arr[0][0]=ar[0];
    arr[0][1]=1;
    x=0;
    for(int i=1;i<n;++i)
    {
    	k=0;
        cin>>ar[i];
    	for(int j=0;j<=x;++j)
    	{
    		if(ar[i]==arr[j][0])
    		{
    			++arr[j][1];
    			++k;
    		}
    	}
    	if(k==0)
    	{
    		++x;
    		arr[x][0]=ar[i];
    		arr[x][1]=1;
    	}
    }
    for(int i=0;i<=x;++i)
    {
    	for(int j=i;j<=x;++j)
    	if(arr[i][0]>arr[j][0])
    	{
    		arr[i][0]=arr[i][0]+arr[j][0];
    		arr[j][0]=arr[i][0]-arr[j][0];
    		arr[i][0]=arr[i][0]-arr[j][0];
    		arr[i][1]=arr[i][1]+arr[j][1];
    		arr[j][1]=arr[i][1]-arr[j][1];
    		arr[i][1]=arr[i][1]-arr[j][1];
        }
    }
    cout<<n<<endl;
    for(int i=0;i<x;++i)
    {
    	cout<<n-arr[i][1]<<endl;
    	n=n-arr[i][1];
    }
    return 0;
}
