#include <bits/stdc++.h>
using namespace std;
int dp[501][101][101] , mod = 1e9+7 , a[501];
int n;
int gcd(int x , int y){
	if(x == 0)return y;
	if(y == 0)return x;
	return gcd(y%x,x);
}
int solve(int x , int g ,int last){
	if(x == n){
		if(g == 1)return 1;
		return 0;
	}
	int &ret = dp[x][g][last];
	if(ret != -1)return ret;
	ret = solve(x+1,g,last);
	if(a[x] > last)ret += solve(x+1,gcd(g,a[x]),a[x]);
	if(ret >= mod)ret -= mod;
	return ret;
}
int main()
{
    cin >> n;
    for(int i = 0 ; i < n ; i++)cin >> a[i];
    memset(dp,-1,sizeof dp);
    cout << solve(0,0,0);
    return 0;
}
