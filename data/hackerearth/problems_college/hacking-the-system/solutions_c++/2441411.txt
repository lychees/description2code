#include<iostream>
#include<stdlib.h>
#include<math.h>
#include<string.h>
using namespace std;
int main(){
	char a[1000000];
	char b[1000000];
	cin>>a>>b;
	
	long long int n=strlen(a),c=0,i;
	for(i=0;i<n;i++){
		if(a[i]!=b[i]){
			c++;
		}
	}
	if(c>1){
		cout<<"loses";
	}
	else{
		cout<<"wins";
	}
	return 0;
}
