


t = int(raw_input())

n = 0
k = 0
dp = [[-1 for j in range(201)] for i in range(201)]

def solve(n, m):
	
	if n == 0: 
		return 1
	if m == 0: 
		return 0
	if n < 0 or n < k or m < k: 
		return 0
	if dp[n][m] == -1:
		dp[n][m] = solve(n, m - 1) + solve(n - m, m)
	return dp[n][m]

while t:
	t-=1

	n, k = map(int, raw_input().strip().split(" "))

	for i in range(201):
		for j in range(201):
			dp[i][j] = -1
	
	print solve(n, n)
