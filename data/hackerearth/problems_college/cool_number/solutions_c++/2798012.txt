#include <iostream>
#include<algorithm>
using namespace std;

int check_very_cool(int n)
{
	int m = 0;
	int arr[100];
	while(n>0)
	{
		arr[m++] = n%2;
		n = n/2;
	}
	reverse(arr,arr+m);
	int arr1[]={1,0,1};
	int count=0;
	for(int i=0; i<=m-3; i++)
	{
		int j = 0;
		while(j<3 && arr[i+j]==arr1[j])
		j++;
		if(j==3)
		count++;
	}
	return count;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k;
    	cin>>n>>k;
    	int count = 0;
    	for(int i=1; i<=n; i++)
    	{
    		int m = check_very_cool(i);
    		if(m>=k)
    		count++;
    	}
    	cout<<count<<"\n";
    }
    return 0;
}
