#include<iostream>
#include<algorithm>
using namespace std;
inline void scanint(int *a)
{
	register char c=0;
	while(c<33)
	c=getchar();
	*a=0;
	while(c>33)
	{
		*a=*a*10+c-'0';
		c=getchar();
	}
}
int main()
{
	int t,i,h,n,x,y;
	cin>>t;
	for(h=0;h<t;h++)
	{
		scanint(&n);
		scanint(&x);
		scanint(&y);
		int A[n],B[n],cost=0;
		for(i=0;i<n;i++)
		{
		  scanint(&A[i]);
		  scanint(&B[i]);
		}
		sort(A,A+n);
		sort(B,B+n);
		for(i=0;i<n;i++)
		{
		  if(A[i]>B[i])
		  cost+=y*(A[i]-B[i]);
		  else if(A[i]<B[i])
		  cost+=x*(B[i]-A[i]);
		}
		printf("%d \n",cost);
	}
	return 0;
}
