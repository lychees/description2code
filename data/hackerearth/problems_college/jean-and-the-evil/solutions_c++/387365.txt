#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

int main()
{
    int t;
    long int n,sum,tmp;
    scanf("%d",&t);
    while(t--)
    {
    	sum=0;
    	tmp=0;
    	scanf("%ld",&n);
    	int health[n];
    	for(int i=0;i<n;i++)
    		scanf("%d",&health[i]);
    	for(int i=0;i<n;i++)
    	{
    		tmp=tmp+health[i];
    		if(tmp<0)
    			tmp=0;
    		if(sum<tmp)
    			sum=tmp;
    	}
    	if(sum==0)
    	{
    		sort(health,health+1);
    		sum=health[n-1];
    		printf("%ld\n",sum);
    	}
    	else
    		printf("%ld\n",sum);
    }
    return 0;
}
