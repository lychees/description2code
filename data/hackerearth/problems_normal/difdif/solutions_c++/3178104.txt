#include <iostream>
#include <string>
#include <set>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
    set<string> myset;
    string s;
    long long int diff = 0;
    cin>>s;
    diff = 1;
    for (int i=1; i<s.size(); i++) {
    	for (int j=0; j<s.size()-i+1; j++) {
    		string temp = s.substr(j, i);
    		//cout<<temp<<"\n";
    		if (myset.find(temp) == myset.end()) {
    			myset.insert(temp);
    			diff++;
    		} 
    	}
    }
    cout<<diff;
    return 0;
}
