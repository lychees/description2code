#include <bits/stdc++.h>
using namespace std;
int arr[25][25];
char str[25];
void flip_bit(int x, int y)
{
	int i,j;
	for(i=1;i<=x;i++)
	for(j=1;j<=y;j++)
	{
		if(arr[i][j]==1)
		arr[i][j]=0;
		else
		arr[i][j]=1;
	}
}
int main()
{
    int t,n,m,i,j,count1;
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	cin>>m;
    	for(i=1;i<=n;i++)
    	{
    	scanf("%s",str);
    	for(j=1;j<=m;j++)
    	arr[i][j]=str[j-1]-48;
    	}
    	count1=0;
    	for(i=n;i>0;i--)
    	for(j=m;j>0;j--)
    	{
//    		if(arr[i][j]==1 && count1%2!=0)
//    		count1++;
    		if(arr[i][j]==0)
    		{
    		count1++;
    		flip_bit(i,j);
    		}
    	}
    	cout<<count1<<endl;
    }
    return 0;
}
