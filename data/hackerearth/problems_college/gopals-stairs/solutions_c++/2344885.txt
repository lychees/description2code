#include <iostream>
using namespace std;
long long int dp[100010];

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		
		int n;
		cin>>n;
		dp[1]=1;
		dp[2]=2;
		dp[3]=4;
		for(int i=4;i<=n;i++)
		{
			dp[i]=dp[i-1]+dp[i-2]+dp[i-3];
		dp[i]=dp[i]%1000000007;
		}
		cout<<dp[n]<<endl;
	}
 //   cout << "Hello World!" << endl;
    return 0;
}
