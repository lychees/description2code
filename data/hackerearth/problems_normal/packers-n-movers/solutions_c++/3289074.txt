#include <bits/stdc++.h>
using namespace std;
int main(){
int m, p;
cin>>m>>p;
int a[p+1];
int s[p+1];
fill(s, s+p+1, 0);
s[0]=0;
for(int i=1; i<=p; i++){
	cin>>a[i];
	s[i] = s[i-1]+a[i];
}
int dp[p+1][p+1][m+1];

for(int i=1; i<=p; i++){
	for(int j=i; j<=p; j++){
		dp[i][j][1]=(s[j]-s[i-1]);
	}
}
for(int c=2; c<=m; c++){
	for(int l=1; l<=p; l++){
		for(int i=1; i<=(p-l+1); i++){
			int j=i+l-1;
			dp[i][j][c] = s[j]-s[i-1];
			for(int k=i; k<j; k++){
				dp[i][j][c] = min(dp[i][j][c], max(dp[i][k][c-1],  (s[j]-s[k])));

			}
		}
	}
}

cout<<dp[1][p][m]<<endl;
}
