K = int(raw_input())
data = raw_input()
length = len(data)
ans = 0
for i in range(length-(K-1)):
    distinct = []
    count = 0
    for j in range(i, length):
        if data[j] not in distinct:
            distinct.append(data[j])
            count += 1

        if count == K:
            ans += 1
        elif count > K:
            break
print ans
