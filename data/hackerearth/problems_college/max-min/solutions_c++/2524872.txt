#include<bits/stdc++.h>
using namespace std;

long long int n,k,a[100004],i,mx,mi,diff,op=INT_MAX;

int main()
{
    cin>>n;
    cin>>k;
    for(i=0;i<n;i++)
    cin>>a[i];
    
sort(a,a+n);
    for(i=0;i<=n-k;i++)
    {
        mx=a[i+k-1];
        mi=a[i];
        diff=mx-mi;
        
        if(diff<op)
        op=diff;
        
        }

}
