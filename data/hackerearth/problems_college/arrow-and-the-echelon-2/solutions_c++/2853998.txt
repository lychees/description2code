#include <iostream>
using namespace std;

int main()
{
	int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	if(n==0 || n==1 || n==2)
    		printf("%d",-1);
    	else
    		{
    			if(n%5==0)
    			{
    				for(int i=0;i<n;i++)
    					printf("%d",3);
    			}
    			else if(n%3==0)
    			{
    				for(int i=0;i<n;i++)
    					printf("%d",5);
    			}
    			else
    			{
    				if(n%3==1)
    					printf("%d",-1);
    				else
    				{
    					for(int i=0;i<n-5;i++)
    						printf("%d",5);
    					for(int i=0;i<5;i++)
							printf("%d",3);
    				}
    			}
    		}
    		printf("\n");
    }
    return 0;
}
