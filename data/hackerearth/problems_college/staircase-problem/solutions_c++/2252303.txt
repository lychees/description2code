#include <iostream>

using std::cin;
using std::cout;
using std::endl;

long long int x(long long int n) {
    long long int sum = 0, prevSum, flag;

    for (long long int increment = 2; ; ++increment) {
        prevSum = sum;
        sum = sum + increment;
        if (n > prevSum && n <= sum) {
            flag = increment-1;
            break;
        }
    }
    return flag;
}

int main() {
    long long int t, n;
    cin>>t;

    while(t--) {
        cin>>n;
        cout<<x(n)<<endl;
    }

    return 0;
}
