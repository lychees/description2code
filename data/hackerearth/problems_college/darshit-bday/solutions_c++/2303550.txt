#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   long long int n,b,w,x,y,z;
   cin>>n;
   while(n!=0)
   {
   	cin>>b>>w;
   	cin>>x>>y>>z;
   	if(x>y)
   	{
   		
   		if(y+z>x)
   		cout<<(x*b+y*w)<<"\n";
   		else
   		cout<<((y+z)*b)+(y*w)<<"\n";
   	}
   	else
   	{
   		
   		if(x+z>y)
   		cout<<(x*b+y*w)<<"\n";
   		else
   		cout<<((x+z)*w)+(x*b)<<"\n";
   	}
   	n--;
   }
    return 0;
}
