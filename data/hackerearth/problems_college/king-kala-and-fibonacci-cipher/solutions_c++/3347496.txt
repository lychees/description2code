#include <bits/stdc++.h>
using namespace std;
static int m[100001][26];
bool cmp(pair<int,int> a,pair<int,int> b){
	if(a.first>b.first)
	   return true;
	if(a.first==b.first&&a.second<b.second)
	    return true;
	return false;
}
int main()
{
    int a=0,b=1;
    for(int i=0;i<26;i++){
    	m[0][i]=m[1][i]=0;
    }
    m[0][0]++;
    m[1][1]++;
    m[1][0]++;
    for(int i=2;i<=100000;i++){
    	int c=a+b;
    	c%=26;
    	a%=26;
    	b%=26;
    	a=b;
    	b=c;
    	for(int j=0;j<26;j++)
    	   m[i][j]=m[i-1][j];
    	m[i][b]++;
    }
    int t;
    scanf("%d",&t);
    while(t--){
    	int l,r;
    	scanf("%d %d",&l,&r);
        pair<int,int> res[26];
        int ans[26];
        for(int i=0;i<26;i++){
        //	printf("freq=%d\n",m[r][i]-((l>0)?m[l-1][i]:0));
           res[i].first=m[r][i]-((l>0)?m[l-1][i]:0),res[i].second=i;}
        sort(res,res+26,cmp);
        for(int i=0;i<26;i++){
        	//printf("%c\n",res[i].second+'a');
        	ans[i]=res[i].second;
        }
        char m1[10001];
        scanf("%s",m1);
        for(int i=0;m1[i]!='\0';i++){
        	printf("%c",ans[m1[i]-'a']+'a');
        }
        printf("\n");
    }
    return 0;
}
