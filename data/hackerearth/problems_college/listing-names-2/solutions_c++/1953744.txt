#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long t;
    cin>>t;
    string a[t];
    long long i;
    for(i=0;i<t;i++)
    {
    	cin>>a[i];
    	
    }
    sort(a,a+t);
    long long n=0;
    for(i=0;i<t-1;i++)
    {
    	n=1;
    	while(i<t-1 && a[i]==a[i+1])
    	{
    		n++;
    		i++;
    	}
    	cout<<a[i]<<" "<<n<<endl;
    }
    if(a[t-2]!=a[t-1])
    	cout<<a[t-1]<<" 1"<<endl;
    return 0;
}
