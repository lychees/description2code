#include <iostream>
using namespace std;
long long  int z[1000000]={0},i,c,a,b;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	a=0;
    	b=1;
   for(i=0;i<n;i++)
    	{
    		z[i]=a;
    		c=a+b;
    	
    	a=b;
    	b=c;
    	}
    	for(i=n-1;i>=0;i--)
    	cout<<z[i]<<" ";
    	cout<<endl;
    	
    }
    return 0;
}
