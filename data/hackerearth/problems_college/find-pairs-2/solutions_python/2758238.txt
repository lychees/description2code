tc= int (raw_input())
for tc in range(tc):
	n=int(raw_input())
	count=0
	x=raw_input().split(' ')
	a=map(int, x)
	for i in range(n):
		for k in range(i+1,n):
			if a[i]==a[k]:
				count+=1
	print(count+n)
