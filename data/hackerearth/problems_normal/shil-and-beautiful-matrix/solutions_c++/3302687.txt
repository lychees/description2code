typedef long long ll;
#include <iostream>
#include <stdio.h>
#include <cmath>
#include <algorithm>
#include <vector>
#include <string>
#include <string.h>
#include <map>
#include <queue>
#include <stack>
#include <sstream>
#include <fstream>
#include <stdlib.h>
#include <utility>

using namespace std;
ll fac[1000007],f[1000007]={0},mod=1000000007;
int main()
{
	ll n,m,i,j,val,maxi,res;
	scanf("%lld %lld",&n,&m);
	fac[0]=fac[1]=1;
	for(i=2;i<=n*m;i++)
	fac[i]=(i*fac[i-1])%mod;
	maxi=0;
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			val=(i^j);
			f[val]++;
			maxi=max(maxi,val);
		}
	}
	res=1;
	for(i=0;i<=maxi;i++)
	res=(res*fac[f[i]])%mod;
	printf("%lld\n",res);
	return 0;
}
