long long max(long long x, long long y){
	if(x>=y) return x;
	else return y;
}
long long ch(vector<long long  >&v){
	
	int sz= v.size();
	long long xx=0;
	long long ans=0;
	for(int u=0;u<sz;u++){
		if(!u){
			xx=v[u];
			ans=v[u];
		}
		else{
			xx=xx^v[u];
			ans&=v[u];
		}}
		return (ans | xx);
//	}
	
}
