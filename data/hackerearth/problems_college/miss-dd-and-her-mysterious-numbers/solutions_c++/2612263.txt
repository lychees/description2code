#include <iostream>
#include <stdio.h>
#include <math.h>
using namespace std;
long long temp[1111111];
long long qu[1111111];
long long final[1111111];
bool isPrime(long long n){
	if(n==1)return false;
	if(n==2)return true;
	for(long long i=2;i*i<=n;i++){
		if(n%i==0)return false;
		}
		return true;}
int main(){
	long long n,q;
	scanf("%lld %lld",&n,&q);
	for(long long i=2;i<=n;i++){
	temp[i]=-1;
	qu[i]=0;
	final[i]=0;
	}
	for(long long i=2;i<=n;i++){
	if(temp[i]==-1){
	temp[i]=1;
	long long k=2;
	while(k*i<=n){
	temp[k*i]=0;
	k++;
	}
	}
	}
	for(long long i=2;i<=n;i++){
	if(temp[i]==1){
	qu[i]=1;
	long long k=2;
	while(i*k<=n){
	qu[k*i]+=1;
	k++;
	}}}
	final[1]=0;
	for(long long i=2;i<=n;i++){
	//printf("%lld ",qu[i]);
	final[qu[i]]++;
	}
	while(q--){
	long long k;
	scanf("%lld",&k);
	if(k>n){
	printf("%lld\n",0);
	}
	else{
	printf("%lld\n",final[k]);
	}
	}
	}
