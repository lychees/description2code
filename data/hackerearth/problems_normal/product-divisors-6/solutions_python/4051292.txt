from math import sqrt	
A,Val,v=[1]*1000050,[0]*1000050,[2];
sq=int(sqrt(1000050));
for i in xrange(2,sq+1):
	if A[i]:
		for j in xrange(i*i,1000050,2*i):
			A[j]=0;

for i in xrange(3,1000050,2):
	if A[i]:	v.append(i);

N=input();	Z=map(int,raw_input().split());
for n in Z:
	sq=int(sqrt(n));	i=0;
	while v[i]<=sq and n>1:
		ct=0;
		while n%v[i]==0:
			n/=v[i];	ct+=1;
		Val[v[i]]+=ct;
		i+=1;
	if n>1:	Val[n]+=1;
res=1;	s=len(v);
for i in xrange(s):	res=(res*(Val[v[i]]+1))%1000000007;
print res;
