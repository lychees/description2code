#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long n,m;
    cin>>n>>m;
    long long a[n];
    for(int i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    long long p,q=0,ans;
    for(int i=0;i<n;i++)
    {
    	if(a[i]%m==0)
    	{
    		p=a[i]/m;
    		if((p-q)==1)
    		{
    			ans=a[i];
    			q=p;
    		}
    	}
    }
    cout<<ans;
    return 0;
}
