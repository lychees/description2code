#include <iostream>
using namespace std;

int n,m,a[1000010],l;
string str;

long long f(char ch,int x)
{
	int y=0;
	for(int i=0;i<l;i++) if(str[i] == ch) a[y++] = i;
	long long an = 0;
	if(x) {
		for(int i=0;i+x-1<y;i++) {
			long long t1,t2;
			t1 = a[i];
			if(i-1>=0) t1 -= a[i-1]; else t1++;
			t2 = 0-a[i+x-1];
			if(i+x<y) t2 += a[i+x]; else t2 += l;
			an += t1*t2;
		}
	} else {
		for(int i=0;i<y;i++) {
			long long t = a[i];
			if(i-1>=0) t -= a[i-1]; else t++;
			t--;
			an += (t*(t+1))/2;
		}
		long long t = l;
		if(y-1>=0) t -= a[y-1]+1;
		an += (t*(t+1))/2;
	}
	return an;
}

int main()
{
	int t;
	cin >> t;
	while(t--) {
		cin >> m >> n;
		cin >> str;
		l = str.size();
		cout << f('R',m) << " " << f('K',n) << endl;
	}
	return 0;
}
