#include <iostream>
#include <cmath>
using namespace std;
long long int xpowm(long long int base,long long int e,long long int m)
{
	if(e==1)
	{
		return base%m;
	}
	if(e==0)
	{
		return 1;
	}
	if(e%2==0)
	{
		int temp=xpowm(base,e/2,m);
		return ((temp%m)*(temp%m))%m;
	}
	else
	{
			return (((base)%m)*xpowm(base,e-1,m))%m;
	}
	
}
long long int sumit(long long int x)
{
	long long int sum=0;
	while(x)
	{
		sum=sum+(x%10);
		x/=10;
	}
	return sum;
}


int main()
{
   int t;
   cin >>t;
  while(t--)
  {
  	long long int a,b,k;
  	cin>>a>>b>>k;
  	long long int h=xpowm(a,b,(pow(10,k)));
    cout<<sumit(h)<<endl;
  }
    return 0;
}
