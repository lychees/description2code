#include <iostream>
#include<algorithm>
#include<cmath>
using namespace std;

int main()
{
    long long a[60],b[3600];
    int i,j,k;
    a[0]=1;
for( i=1;i<60;i++)
{
a[i]=a[i-1]*2;
}

for( i=0;i<3600;i++)
    b[i]=2000000000000000000;


for(i=0,k=0;i<60;i++)
{
    for(j=i+1;j<60;j++,k++)
    {
        b[k]=a[i]+a[j];
    }
}
sort(b,b+3600);
int t;
long long n;
cin>>t;
while(t--)
{
 long long ans=0;
 cin>>n;
 for(i=0;i<3600;i++)
 {
     if(b[i]>n)break;
     ans+=b[i];
     ans%=1000000007;
 }
 cout<<ans<<endl;
}
    return 0;
}
