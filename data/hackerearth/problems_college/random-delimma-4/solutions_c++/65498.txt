#include <stdio.h>

double compute_prob(double n, double k)
{
        double prob = 1.0;
        int j;
        for(j = 1; j < k; j++)
        {
                prob *= ((n-j)/n);
        }
        return (1 - prob);
 }

int main()
{
        int ncases = 0;
        scanf("%d", &ncases);

        while (ncases)
        {
                int n, k;
                scanf("%d %d", &k, &n);
                printf("%.9lf\n", compute_prob((double)n, (double)k));
                ncases--;
        }
        return 0;
}
