#include<bits/stdc++.h>
#define si(x) scanf("%d",&x)
#define sii(x,y) scanf("%d%d",&x,&y)
#define sl(x) scanf("%ld",&x)
#define sll(x,y) scanf("%ld%ld",&x,&y)
using namespace std;
typedef long long ll;
int main(){
	int n,m;
	sii(n,m);
	map<string, string> mp;
	while(m--){
		string str1,str2;
		cin>>str1>>str2;
		if(str1.length()>str2.length()){
			mp[str1]=str2;
			mp[str2]=str2;
		}
		else{
			mp[str1]=mp[str2]=str1;
		}
	}
	string str;
	for(int i=0;i<n;i++){
		cin>>str;
		cout<<mp[str]<<" ";
	}
	
}
