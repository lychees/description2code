ll power(ll a,ll b)
{
	if(!b)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2);
	temp=(temp*temp);
	if(b&1)
		temp=(temp*a);
	return temp;
}
int main()
{
    int i,l,r,t,j,lt=10000000;
    float tot,fav;
    vector<bool> v(lt+1);
    vector<int> dp(lt+1);
    for(i=2;i<=lt;i++)
    {
		if(!v[i])
		{
			for(j=2*i;j<=lt;j=j+i)
				v[j]=true;
		}
	}
	for(i=2;i<=lt;i++)
	{
		dp[i]=dp[i-1];
		if(!v[i])
		{
			dp[i]++;
		}
	}
	v[0]=v[1]=true;
    scanf("%d",&t);
    while(t--)
    {
		scanf("%d%d",&l,&r);
		fav=dp[r]-dp[l];
		if(!v[l])
			fav++;
		tot=r-l+1;
		printf("%.6f\n",fav/tot);
	}
    return 0;
}
