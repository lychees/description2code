#include <bits/stdc++.h>
#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t , n , s ;
    cin >> t ;
    while(t --){
    	cin >> n ;
    	cin >> s ;
    	vector<int> A(n+1) , DP(min(20000,s)+1);
    	for(int i=1;i<=n;i++){
    		cin >> A[i] ;
    	}
    	bool ok = false ;
    	DP[0] = 1 ;
    	int sum = 0 ;
    	if(s >= 20000) ok = true ;
    	for(int i=1;i<=n;i++){
    		for(int j=min(20000,s);j>=0;j--){
    			if(DP[j] && j+A[i] <= min(20000,s)){
    				DP[j+A[i]] = true ;
    			}
			}
			sum += A[i] ;
    	}
    	for(int i=0;i<=min(s,20000);i++) if(DP[i] && sum-i <= s) ok = true ;
    	puts(ok ? "YES" : "NO") ;
    }
    return 0;
}
