#include <iostream>
using namespace std;

int main()
{
    int n,q,a[100005],l,r,i,s[100005];
    string ss;
    cin>>n>>ss;
    l=ss.size();
    for(i=0;i<l;i++){
    	a[i+1]=ss[i]-'0';
    }
    s[1]=a[1];
    for(i=2;i<=n;i++)
    {
    	s[i]=s[i-1]+a[i];
    }
    cin>>q;
    while(q--)
    {
    	cin>>l>>r;
    	if(l==1)
    	{
    		i=s[r]%9;
    	}
    	else
    	{
    		i=(s[r]-s[l-1])%9;
    	}
    	if(i==0)
    	{
    		cout<<"SAMEER"<<endl;
    	}
    	else
    	{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
