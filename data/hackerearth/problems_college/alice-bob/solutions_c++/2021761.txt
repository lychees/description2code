#include <iostream>
#include <cstdio>
#include <algorithm>
using  namespace  std;

int  num [105];

int  gcd ( int  a,  int  b)
{
    return   b ? gcd (b, a% b): a;
}

int  main (void)
{
    int  N,j, temp = 0;
    scanf ( "%d",&N);
    for  ( int  i = 0; i <N; i ++)
    {
        scanf ( "%d",&num[i]);
        temp = max (temp, num[i]);
    }
    int  MAXN = temp;
    for  ( int  i = 0; i <N - 1; i ++)
        for  ( int  i = j + 1; j <N; j ++)
        {
            int  k = gcd (num[i], num[j]);
            temp = gcd (temp, k);
        }
    int  movetimes = MAXN / temp - N;
    puts( movetimes & 1 ?  "Alice"  :  "Bob" );
    return  0;
}
