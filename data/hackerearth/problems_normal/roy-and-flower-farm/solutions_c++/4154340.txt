int x[101],y[101];
int n;
int dp[101][10001];
int func(int index,int paisa)
{
    if(dp[index][paisa]!=-1)
        return dp[index][paisa];
    if(index>=n)
        return 0;
    else if(paisa>=y[index])
        return dp[index][paisa]=max(func(index+1,paisa),x[index]+func(index+1,paisa-y[index]));
    else
        return dp[index][paisa]=func(index+1,paisa);
}
int main()
{
    int test;
    cin>>test;
    while(test--)
    {
        memset(dp,-1,sizeof(dp));
        int paisa;
        cin>>n>>paisa;
        for(int i=0;i<n;++i)
            cin>>x[i]>>y[i];
        ll ans,ans1;
        ans = INT_MAX;
        ans1 = INT_MIN;
        for(ll i=0;i<=paisa;++i)
            if(ans1<func(0,i)-i+paisa)
            {
                ans1 = func(0,i)-i+paisa;
                ans = i;
            }
            else if(ans1==func(0,i)+paisa-i)
            {
                ans = min(ans,i);
            }
    
        cout<<ans<<" "<<ans1<<endl;
        
    }
}
