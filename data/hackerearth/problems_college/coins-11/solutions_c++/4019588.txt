#include <bits/stdc++.h>
using namespace std;
//zone
int prime[1000006];
void pre()
{
	int i,j;
	prime[1] = 1;
	for(i = 2 ; i <=1000000; i++)
	{
		if(prime[i] == 0)
		{
		for(j = 2*i ; j <= 1000000; j += i)
		{
			prime[j] = 1;
		}
		}
	}
}
int main()
{
	pre();
 int t;
 cin >> t;
 while(t--)
 {
 	int n;
 	cin >> n;
 	if(n == 3)
 	{
 		cout<<"3\n";
 		continue;
 	}
 	if(prime[n] == 0)
 	{
 		cout<<"2"<<endl;
 	}
 	else
 	    cout<<"1"<<endl;
 }
 return 0;
}
