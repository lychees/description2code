#include <iostream>
#include <vector>
#include <algorithm>
#define ui unsigned long int
#define SIZE (int)1e6

using namespace std;

int main()
{
	ui n,k,i,x;
	vector <ui> a; 
    cin >> n >> k;
    if (k == 1000)
    {
    	cout << "NO";
    	return 0;
    }
   	bool ch[SIZE] = {0};
    for (i = 0; i < n; i++)
    {
    	cin >> x;
    	a.push_back(x);
    	ch[x] = 1;
    }
    sort(a.begin(),a.end());
    while (a[n-1] > k) n--;
    for (i = n-1; i >= 0; i--)
    {
    	long int pos = k - a[i];
    	if (pos > a[i]) break;
    	if (ch[pos] == 1 && i+1 != pos)
    	{
    		cout << "YES";
    		return 0;
    	}
    }
    cout << "NO";
    return 0;
}
