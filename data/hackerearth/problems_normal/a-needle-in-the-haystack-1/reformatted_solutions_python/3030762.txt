def needle():
    str1 = raw_input()
    str2 = raw_input()

    if (str1 in str2[::-1]) or (str1 in str2) or (str1[::-1] in str2):
        print("YES")
    else:
        print("NO")

    return


def main():
    t = int(raw_input())

    for i in range(t):
        needle()


if __name__ == '__main__':
    main()
