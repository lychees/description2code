#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long n,i,sum=0;
	scanf("%lld",&n);
	long long a[n];
	long long b[n];
	for(i=0;i<n;i++)
		scanf("%lld",&a[i]);
	for(i=0;i<n;i++)
		scanf("%lld",&b[i]);
	sort(a,a+n);
	sort(b,b+n);
	for(i=0;i<n;i++)
		sum=sum+(a[i]*b[i]);
	printf("%lld",sum);
	return 0;
}
