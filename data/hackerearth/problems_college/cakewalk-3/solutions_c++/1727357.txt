#include <iostream>
using namespace std;

int main()
{
    int numtest;
    int num, start = 1;
    string prev,current;
    cin>>numtest;
    while(numtest--){
    	if(start){
    		cin>>current;
    		prev = current;
    		start = 0;
    		continue;
    	}
    	else{
    		cin>>num;
    		cin>>current;
    	}
    	prev = prev.substr(0,num) + current;
    	
    }
    cout<<prev;
    return 0;
}
