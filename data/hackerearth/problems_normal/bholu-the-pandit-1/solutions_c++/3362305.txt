#include <iostream>
using namespace std;
#define ll long long 
#define mod 1000000007
ll gcd(ll a,ll b){
	if(b==0)return a;
	else gcd(b,a%b);
}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	ll a,b,c;
    	cin>>a>>b>>c;
    	ll len=gcd(a,gcd(b,c));
    	a/=len;
    	b/=len;
    	c/=len;
    	ll ans=(((a*b)%mod)*(c%mod))%mod;
    	cout<<len<<" "<<ans<<endl;
    }
    return 0;
}
