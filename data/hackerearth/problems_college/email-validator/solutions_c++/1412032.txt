#include <iostream>
#include<stdio.h>
#include<string.h>
#include<math.h>
using namespace std;
int fun(char s[])
{
	int i,j=0,flag=0;
	int len=strlen(s);
	for(i=0;i<len;i++)
	{
		if((s[i]>='A'&&s[i]<='Z')||(s[i]>='0'&&s[i]<='9')||(s[i]>='a'&&s[i]<='z')||s[i]=='_')
		continue;
		else if(s[i]=='@')
		break;
		else  return 0;
	}
	if(i>16||i==0||i>=len)
	return 0;
	j=0;
	i++;
	for(;i<len;i++,j++)
	{
		if((s[i]>='A'&&s[i]<='Z')||(s[i]>='a'&&s[i]<='z'))
		{
			flag=1;
		continue;
		
		}
		else if((s[i]>='0'&&s[i]<='9')||s[i]=='-')
		continue;
		else if(s[i]=='.')
		break;
		else return 0;
	}
	if(j>16||flag==0||j==0||i>=len)
	return 0;
	
	j=0;
	i++;
	for(;i<len;i++,j++)
	{
		if((s[i]>='a'&&s[i]<='z'))
		continue;
		else
		return 0;
	}
	if(j>10)
	return 0;
	return 1;
	
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char s[100000];
	   	cin>>s;
	   	if(fun(s))
	   	cout<<"YES"<<endl;
	   	else
	   	cout<<"NO"<<endl;  
	}
	 

}
