long long arr[1000010];
int main()
{
	int t,i;
	long long diff=1;
    for(i=1;i<1000010;i++)
    {
    	if(arr[i]==0)
    	{
	    	arr[i]=i+diff;
	    	if(i+diff<1000010)
	    	arr[i+diff]=i;
	    	diff++;
	    }
    }
	cin>>t;
    while(t--)
    {
    	long long n1,n2;
    	cin>>n1>>n2;
    	if(n1>n2)
    	{
	    	long long temp=n1;
	    	n1=n2;
	    	n2=temp;
	    }
		if(arr[n1]==n2)
		{
			cout<<"Don't Play"<<endl;
		}    	
		else
		{
			cout<<"Play"<<endl;
		}
    }
    return 0;
}
