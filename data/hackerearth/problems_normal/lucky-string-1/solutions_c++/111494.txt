#include<stdio.h>
#include<math.h>
int main()
{
    long long int n,temp,x;
    long long int temp2,temp3;
    long int i,j,t;
    scanf("%ld",&t);
    for(j=0;j<t;j++)
    {
        temp=0;
        scanf("%lld",&n);
        //n=j+1;
        for(i=0; ;i++)
        {
        	temp+=pow(2,i)*i;
        	if(temp>=n)

        //printf("Ans will  %lld digit of %ld's nos \n",x,i);
        temp2=x%i;

        {
        	temp=x/i;
        	temp2=i;
        }
        else
        {
        	temp=x/i+1;
        }
        //printf("ans is %lldth digit of %ld digit's %lld no\n",temp2,i,temp);
      	temp2=i-temp2+1;
       	//printf("ans is %lld(temp2)th digit from right of %ld digit's %lld(temp) no\n",temp2,i,temp);
      	temp3=temp2-1;
        temp3=pow(2,temp3);
       // printf("Ans will same for the continous %lld no's \n",temp3);
        if(temp<temp3)
        {
        	//printf("4\n");
        	printf("Hacker\n");
        }
        else if(temp%temp3==0)
        {
        	if((temp/temp3)%2==0)
        	{
        		//printf("5\n");
        		printf("Earth\n");
        	}
        	else
        	{
        		//printf("4\n");
        		printf("Hacker\n");
        	}
        }
        else
        {
        	if((temp/temp3)%2==0)
        	{
        		//printf("4\n");
        		printf("Hacker\n");
        	}
        	else
        	{
        		//printf("5\n");
        		printf("Earth\n");
        	}
        }
    }
}
