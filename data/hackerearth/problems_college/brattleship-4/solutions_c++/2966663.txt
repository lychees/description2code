#include <bits/stdc++.h>
using namespace std;
 
 
int main()
{
    int t,count=1;
    cin>>t;
    while(t--){
    int r,c,w;
    cin>>r>>c>>w;
    int ans=0;
    ans+=c/w*r;
    if(c%w==0)ans+=w-1;
    else ans +=w;
    cout<<"Case #"<<count<<": ";
    cout<<ans<<endl;
    
    count++;
    }
    return 0;
}
