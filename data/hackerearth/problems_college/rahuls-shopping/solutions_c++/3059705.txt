#include<bits/stdc++.h>
main()
{
	int n,i,l,b;
	long long sum,ans,fans=0;
	scanf("%d",&n);
	int a[n];
	for(i=0;i<n;i++)
	scanf("%d",&a[i]);
	scanf("%d%d",&l,&b);
	for(i=0;i<n;i++)
	{
		sum=a[i]*(a[i]%9)*l*b;
		if(a[i]>=1&&a[i]<=10)
		ans=sum-sum*0.25;
		else if(a[i]>=11&&a[i]<=20)
		ans=sum-sum*0.30;
		else
		ans=sum-sum*0.20;
		fans=fans+ans;
	}
	printf("%lld",fans);
}
