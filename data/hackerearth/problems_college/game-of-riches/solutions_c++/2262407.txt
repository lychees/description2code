#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<vector>
using namespace std;

int main()
{
    int t,n,g;
    cin>>t;
    int a[101];
    vector<int> group[101];
    while(t--)
    {
        cin>>n>>g;
        int self;

        for(int i=0;i<n;i++)
        cin>>a[i];

        self = a[0];

        sort(a,a+n);

        int j=1,ans;
        for(int i=n-1;i>=0;i--)
        {
            group[j++].push_back(a[i]);

            if(a[i] == self)
            {
                ans = j-1;
                break;
            }
            if(j == g+1)
                j = 1;

        }

        cout<<group[ans].size()-1<<endl;

        for(int i=0;i<=g;i++)
        {
            group[i].erase(group[i].begin(),group[i].begin()+group[i].size());
        }

    }
    return 0;
}
