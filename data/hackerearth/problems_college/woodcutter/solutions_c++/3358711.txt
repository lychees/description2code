#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
    long long int n,i,j,k,l,m,h,s=0;
    cin >> n >> m;
    long long int a[n];
    for(i=0;i<n;i++)
	{
		cin >> a[i];
		s=s+a[i];
	}
	if(s<m)
	{
		cout << -1 << endl;
	}
	else if(s==m)
	{
		cout << 0 << endl;
	}
	else
	{
	    sort(a,a+n);
	    h=a[n-1];i=1;j=n-1;
	    while(m>0&&i<=n&&j>0)
	    {
	    	if((a[j]-a[j-1])*i<=m)
	    	{
	    		h=a[j-1];
	    		m=m-(a[j]-a[j-1])*i;
	    		j--;i++;
			}
			else
			{
				m=m-i;
				h--;
			}
		}
		if(m>n||m>0)
		{
			h=h-m/n;
			if(m>0)
				h=h-1;
		}
		cout << h << endl;
	}
    return 0;
}
