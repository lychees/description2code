#include<iostream>
#include<vector>
#include<algorithm>
#include<utility>
#define ll long long
#define h(i) disks[i].first
#define r(i) disks[i].second
using namespace std;
int main(){
	int t;cin>>t;int n;ll a,b;
	while(t--){
		cin>>n;
		vector< pair<ll,ll> > disks;
		for(int i=0;i<n;i++){
			cin>>a>>b;
			disks.push_back(make_pair(b,a));
		}
		sort(disks.rbegin(),disks.rend());
		ll dp[n]; ll maxi=0;
		for(int i=0;i<n;i++){
			dp[i]=h(i);
			for(int j=i-1;j>=0;j--){
				if(h(i)+dp[j]>dp[i] && h(j)>h(i) && r(j)>r(i)){
					dp[i]=dp[j]+h(i);
				}
			}
			if(maxi<dp[i]){maxi=dp[i];
			}
		}
		cout<<maxi<<endl;
	}
}
