#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int n,ta,m,td,i,k=0,cnt=0;
   	cin>>n;
   	vector< pair<int,int> > a;
   	for(i=0;i<n;i++)
   	{
   		cin>>ta>>td;
   		a.push_back(make_pair(ta,1));
   		a.push_back(make_pair(td,0));
   	}
   	sort(a.begin(),a.end());
  	for(i=0;i<2*n;)
   	{
   		m=a[i].first;
   		if(a[i].second==1)
   		cnt++;
   		else cnt--;
   		i++;
   		while(i<2*n && m==a[i].first)
   		{
   		if(a[i].second==1)
   		cnt++;
   		else cnt--;
   		i++;
   		}
   		k=max(k,cnt);
   	}
   	cout<<k<<"\n";
   }
   
   
   return 0;
}
