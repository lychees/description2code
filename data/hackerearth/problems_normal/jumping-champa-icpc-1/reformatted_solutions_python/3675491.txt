def champa():
    N, Q = map(int, raw_input().strip().split(' '))
    H = sorted(map(int, raw_input().strip().split(' ')))
    if N < 2:
        return 0
    else:
        ls = []
        for i in range(N - 1):
            ls.append(H[i + 1] - H[i])
        return sum(ls) * Q


for i in xrange(int(raw_input().strip())):
    print champa()
