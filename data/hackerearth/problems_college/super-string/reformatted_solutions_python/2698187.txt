for i in range(int(raw_input())):
    ar = []
    for i in range(27):
        ar.append(0)
    s = raw_input()
    for i in range(len(s)):
        ar[26 - (ord(s[i])-ord('A'))] += 1
    flag = 0

    for i in range(27):
        if ar[i] != i and ar[i] != 0:
            flag = 1
            break
    if flag == 0:
        print "Yes"
    else:
        print "No"
