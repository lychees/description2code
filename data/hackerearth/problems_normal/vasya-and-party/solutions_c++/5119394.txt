long long int maxx;
long long int count;
void dfs2(vector< vector<int> > &graph,vector<int> &vis1,vector<int> &vis2,int curr,vector<int> &level)
{
	vis2[curr]=1;
	if(level[curr-1]==maxx)
	count++;
	for(int i=0;i<graph[curr].size();i++)
	{
		if(vis2[graph[curr][i]]==0)
		dfs2(graph,vis1,vis2,graph[curr][i],level);
	}
}
void dfs(vector< vector<int> > &graph,vector<int> &vis1,vector<int> &vis2,int curr,vector<int> &level)
{
	//cout<<"curr "<<curr;
	vis1[curr]=1;
	if(level[curr-1]>maxx)
	maxx=level[curr-1];
	for(int i=0;i<graph[curr].size();i++)
	{
		if(vis1[graph[curr][i]]==0)
		dfs(graph,vis1,vis2,graph[curr][i],level);
	}
}
