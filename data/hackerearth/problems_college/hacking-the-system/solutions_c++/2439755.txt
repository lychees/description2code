#include <iostream>
#include <string>
using namespace std;

int main()
{
    string a , b;
    cin >> a >> b;
    int l = a.size();
    int i,temp = 0;
    for(i=0;i<l;i++)
    {
    	if(a[i]!=b[i])
    	temp = temp + 1;
    }
    if(temp==1 || temp==0)
    cout << "wins";
    else 
    cout << "loses";
    return 0;
}
