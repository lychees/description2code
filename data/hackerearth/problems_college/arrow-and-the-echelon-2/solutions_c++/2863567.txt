#include <iostream>
using namespace std;

long mixer(long num)
{
    num--;
    long left = 0;
    if(!(num%3))        //divisible
    {
        left = 1;
    }
    else
    {
        num--;
        left = 2;
    }

    while(num>=3)
    {
        if(!(left%5))       // left is divisible by 5
        {
            return num;
        }

        else
        {
            num = num-3;
            left = left+3;
        }
    }

    return -1;
}

int main()
{
    long loop;
    cin>>loop;

    //long *arr = new long[loop];

   /* for(long i=0;i<loop;i++)
    {
        cin>>arr[i];
    }*/

    for(long i=0;i<loop;i++)
    {
     //   long num = arr[i];
        long num;
        cin>>num;
        if(!(num%3))
        {
            while(num)
            {
                cout<<"5";
                num--;
            }
        }

        long check = mixer(num);
        if( check != -1)
        {
            num = num-check;
            while(check)
            {
                cout<<"5";
                check--;
            }

            while(num)
            {
                cout<<"3";
                num--;
            }
        }

        else if(!(num%5))
        {
            while(num)
            {
                cout<<"3";
                num--;
            }
        }

        else
        {
            cout<<"-1";
        }

     cout<<"\n";
    }
    return 0;
}
