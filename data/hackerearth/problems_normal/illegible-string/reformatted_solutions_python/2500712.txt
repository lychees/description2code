tc = input()
data = raw_input()
other = ""
max1 = 0
for char in data:
    if char == 'w':
        max1 += 2
        other += 'vv'
    elif char == 'v':
        max1 += 1
        other += 'v'
    else:
        other += char
        max1 += 1

i = 1
min1 = max1
while i < max1:
    if other[i-1] == "v" and other[i] == "v":
        min1 -= 1
        i += 2
    else:
        i += 1

print min1, max1
