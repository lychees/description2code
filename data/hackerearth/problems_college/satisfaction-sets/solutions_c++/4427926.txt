#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin >> n;
	int a[505] = {};
	long long ans = 0;
	for(int i=0;i<n;i++) {
		int m;
		cin >> m;
		set<int>s;
		for(int j=0;j<m;j++) {
			int x;
			cin >> x;
			if(s.find(x) != s.end()) continue;
			s.insert(x);
			a[x]++;
			if(a[x] == n) ans -= x;
			// ans += 1000000000;
			ans %= 1000000000;
		}
	}
	cout << ans << "\n";
    return 0;
}
