#include<iostream>
#include<stdlib.h>
#include<string>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		int flag=0;
		int n=s.length();
		for(int i=0,j=n-1;i<n-1&&j>=1;i++,j--)
		{
			if(abs(s[i]-s[i+1])!=abs(s[j]-s[j-1]))
			{
				flag=1;
				break;
			}
		}
		if(flag==0)
		cout<<"Funny"<<endl;
		else
		cout<<"Not Funny"<<endl;
	}
}
