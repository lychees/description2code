#include <bits/stdc++.h>
using namespace std;
long long dp1[10000];
long long dp2[10000];
long long recurse1(int level,int n)
{
	if(level == n)
	return 1;
	else if(level>n)
	return 0;
	if(dp1[level] != -1)
	return dp1[level];
	else
	{
		long long x,y,z;
		x = recurse1(level+1,n);
		y = recurse1(level+2,n);
		z = recurse1(level+3,n);
		dp1[level] = x+y+z;
		return dp1[level];
	}
}
long long recurse2(int level,int n)
{
	if(level == n)
	return 1;
	else if(level>n)
	return 0;
	if(dp2[level] != -1)
	return dp2[level];
	else
	{
		long long x,y;
		x = recurse2(level+1,n);
		y = recurse2(level+2,n);
		
		dp2[level] = x+y;
		return dp2[level];
	}
}
int main()
{
    int n,i;
    cin>>n;
    if(n<0)
    cout<<"ERROR"<<endl;
    else
    {
    for(i=0;i<=n;i++)
    {
    	dp1[i] = -1;
    	dp2[i] = -1;
    }
    
    long long ans1 = recurse1(0,n);
    long long ans2 = recurse2(0,n);
    cout<<"Jack-"<<ans1<<endl;
    cout<<"Jill-"<<ans2<<endl;
}
}
