import sys


def findMinCost(Ai, index, K):
    minCost = 0
    cost = []
    for (idx, item) in enumerate(Ai):
        if idx != index:
            c = 0
            if item > Ai[index]:
                c = (item-Ai[index]) * 5
            else:
                c = (Ai[index]-item) * 3
            cost.append(c)
    for costs in sorted(cost):
        if K > 0:
            minCost = minCost + costs
            K = K - 1
    return minCost


T = int(raw_input())
for i in xrange(T):
    N, K = raw_input().split()
    N = int(N)
    K = int(K)
    Ai = map(int, raw_input().split())
    minCost = sys.maxint
    for (index, item) in enumerate(Ai):
        tempCost = findMinCost(Ai, index, K-1)
        if tempCost < minCost:
            minCost = tempCost
