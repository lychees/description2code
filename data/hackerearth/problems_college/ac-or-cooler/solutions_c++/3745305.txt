#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    long n,m;
    cin>>n>>m;
    long a[2*n];
    for(long i=0;i<2*n;i++)
        cin>>a[i];
    sort(a,a+(2*n));
    long ans=0,i=0;
    while(ans<m)
    {
        ans+=a[i];
        i++;
    }
    cout<<i-1<<endl;
}
