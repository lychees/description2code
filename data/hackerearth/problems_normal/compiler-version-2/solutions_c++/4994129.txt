#include <iostream>
#include <string>
using namespace std;

int main()
{
	int length;
	string line;

	while (getline(cin, line))
	{
		length = line.length();
		for (int i = 0; i < length - 1; ++i)
		{
			if (line[i] == '/' && line[i + 1] == '/')
				break;
			if (line[i] == '-' && line[i + 1] == '>')
			{
				line.replace(i, 2, ".");
				--i;
				--length;
			}
		}
		cout << line << "\n";
	}

	return 0;
}
