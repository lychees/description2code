#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n, m, q,i,j,l1;
    cin>>n>>m>>q;
    l1 = q;
    int x[n],l;
    int a[n][m];
   	
   	for(i = 0; i < n; i++)
   		for(j = 0; j < m; j++)
   		cin>>a[i][j];
   	
   	while(q--)
   	{
   		l=m-1;
   		for(j = 0; j < n; j++)
   		cin>>x[j];
   		for(i = 0; i < n ; i++)
   		{
   			while(x[i]<a[i][l])
   			l--;
   		}
   		cout<<l+1<<endl;
   	}
    return 0;
}
