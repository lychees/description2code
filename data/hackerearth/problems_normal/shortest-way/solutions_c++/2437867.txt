#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	long long int a,b;
    	cin>>a>>b;
    	long long int ans =0;
    	while(a>0 && b>0) {
    		if(a>b) {
    			ans += a/b;
    			a = a%b;
    		} else {
    			ans+= b/a;
    			b = b%a;
    		}
    	}
    	cout<<ans-1<<endl;
    }
    return 0;
}
