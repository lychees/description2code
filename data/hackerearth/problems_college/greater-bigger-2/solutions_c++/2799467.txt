#include<bits/stdc++.h>
using namespace std;
void sor(char s[],int st,int ed)
{
    if(st<ed)
    {
        char c=s[(st+ed)/2];
        int i=st;
        int j=ed;
        while(1)
        {
            while(1)
            {
            if(s[i]>=c)
            {
                break;
            }
            i++;
            }
            while(1)
            {
                if(s[j]<=c)
                    break;
                j--;
            }
        if(i>j)
            {
                sor(s,st,j);
                sor(s,i,ed);
                break;
            }
            else if(i<j)
            {
                if(s[i]!=s[j])
                    {
                     char temp=s[i];
                     s[i]=s[j];
                     s[j]=temp;
                    }
                    else //i==j
                    {
                        j--;
                    }
            }
            else
            {
                sor(s,st,j-1);
                sor(s,j+1,ed);
                break;
            }
        }
    }
}

int main()
{
    int t,i,l,j;
    char s[200],s1[200];
    cin>>t;
    while(t--)
    {
        cin>>s;
        l=strlen(s);
        for(i=l-2;i>=0;i--)
        {
            if(s[i]<s[i+1])
            {
                for(j=l-1;j>i;j--)
                {
                    if(s[j]>s[i])
                    {
                        char p=s[j];
                        s[j]=s[i];
                        s[i]=p;
                        break;
                    }
                }
                sor(s,i+1,l-1);
                break;
            }
        }
        if(i>=0)
            cout<<s<<endl;
        else
            cout<<"no answer\n";
    }
    return 0;
}
