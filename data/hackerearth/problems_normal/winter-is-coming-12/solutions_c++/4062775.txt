typedef long long ll;
ll MOD=1000000007;
ll pw(ll a,ll b){
  ll r=1;
  while(b>0){
     if(b&1){
       r=(r*a)%MOD;
        }
     b=b>>1;
     a=(a*a)%MOD;
  }
return r;
}

int main(){
   int t;
   sll(t);
   while(t--){
       int n,f=0,as,ae;
       si(n);
       as=0;
       ae=n+1;
       ll in,a[n+1];
       fr(i,0,n){
         sll(in);
          a[i]=(in)%n;
          if(!f&&!a[i]){
            as=i;
            ae=i;
            f=1;
          }
       }
     if(!f){
            int l=0,r=0;
            ll sum=0;
           while(r<n){
              sum+=a[r];
              if(sum==n){
                 if(r-l+1<ae-as+1){
                    ae=r;
                    as=l;
                 }
              }
              while(sum>n&&l<=r){
                sum-=a[l];
                l++;
                  if(sum==n){
                     if(r-l+1<ae-as+1){
                       ae=r;
                       as=l;
                     }
                 }
              }
            r++;

          }
