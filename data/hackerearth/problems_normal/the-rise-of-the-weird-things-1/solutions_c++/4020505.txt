#include <bits/stdc++.h>
using namespace std;

int main()
{
	int z[1000], v[1000], i, n, z1 = 0, v1 = 0, c1 = 0, c2 = 0;
	cin >> n;
	for (i = 0; i < n; ++i) {
		int a;
		cin >> a;
		if (a % 2) {
			v[c2++] = a;
			v1 += a;
		} else {
			z[c1++] = a;
			z1 += a;
		}
	}
	sort(z, z + c1);
	sort(v, v + c2);
	for (i = 0; i < c1; ++i)
		cout << z[i] << ' ';
	cout << z1 << ' ';
	for (i = 0; i < c2; ++i)
		cout << v[i] << ' ';
	cout << v1 << endl;
    return 0;
}
