#include <iostream>
using namespace std;
long long mod=1000000007;
long long pow(long long n)
{
	if(n==0)
	return 1;
	long long temp=pow(n/2)%mod;
	temp=(temp*temp)%mod;
	if(n&1)
	return (temp*2)%mod;
	else
	return  temp;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long n;
    	cin>>n;
    	long long ans=pow(n);
    	cout<<ans-1<<endl;
    }
    return 0;
}
