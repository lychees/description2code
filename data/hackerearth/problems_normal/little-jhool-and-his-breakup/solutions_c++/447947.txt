#include <iostream>
#include <string>
#include <cmath>
using namespace std;

int main()
{
    int l, i, j, k, m, n;
    string s;
    cin >>s;
    char a[]={'l','o','v','e'};
    l=s.length();
    j=0;
    k=0;
    m=0;
    n=0;
    for(int i=0; i<l; i++)
    {
        if(s[i]==a[0] && j==0)
        {
            j=j+1;
        }
        if(s[i]==a[1] && j==1 && k==0)
        {
            k=k+1;
        }
        if(s[i]==a[2] && m==0 && k==1)
        {
            m=m+1;
        }
        if(s[i]==a[3] && n==0 && m==1)
        {
            n=n+1;
        }
    }
    if(n==1)
    {
        cout <<"I love you, too!"<<endl;
    }
    else{cout <<"Let us breakup!"<<endl;}
    return 0;
}
