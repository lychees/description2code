import math
n = int(raw_input())
ans = 0  # (n*(n+1)*(2*n+1))/6
arr = sorted(map(int, raw_input().split()))
i = 0
elem = arr[0]
val = int(math.sqrt(elem))
if val * val != elem:
    ans += (val*(val+1)*(2*val+1))/6
    init = val+1
else:
    init = val
    if init != 1:
        val -= 1
        ans += (val*(val+1)*(2*val+1))/6
for elem in arr:
    while init * init < elem:
        ans += init*init
        init += 1
    ans += (init * init - elem)
    init += 1
print ans
