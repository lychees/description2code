#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long t,i,l,n,m,c,x,y;
    cin>>t;
    for(i=1;i<=t;i++){
    	cin>>n>>m>>c;
    	cout<<"Case #"<<i<<": ";
    	x=n;y=0;
    	while(x<m){
    		x*=c;
    		y++;
    	}
    	if(y<=1)
    		cout<<0<<endl;
    	else if((y & (y-1)) == 0)
    		cout<<(long long)log2(y)<<endl;
    	else
    		cout<<(long long)log2(y)+1<<endl;
    }
    return 0;
}
