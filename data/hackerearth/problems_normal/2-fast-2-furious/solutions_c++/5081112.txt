#include <iostream>
#include <math.h>
using namespace std;
int main()
{
	long long int ta,tb,n,temp1=0,temp2=0,b,a;
	cin>>n;
	cin>>ta;

	for ( int i=0;i<n-1;i++){
		cin>>a;
		if (temp1<abs(ta-a)){
			temp1=abs(ta-a);
		}
		ta=a;
	}
		cin>>tb;
	for ( int i=0;i<n-1;i++){
		cin>>b;
		if (temp2<abs(tb-b)){
			temp2=abs(tb-b);
		}
		tb=b;
	}
	if(temp1>temp2){
		cout<<"Dom"<<endl;
		cout<<temp1;
	}
	else{
		cout<<"Brian"<<endl;
		cout<<temp2;
	}
    return 0;
}
