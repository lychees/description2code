#include <iostream>
using namespace std;
 
int main()
{
 
   
    	long long int a,b,c,n,i,j,z,d,t;
    
    cin>>t;
    while(t--)
    {
    	d=0;
    	cin>>n;
    	
    	if(n%6==0 || n%9==0 || n%20==0)
    	{
    		cout<<"True\n";
    		continue;
    	}
    	
    	for(i=n/20;i>=0;i--)
    	{
    		for(j=n/9;j>=0;j--)
    		{
    			for(z=n/6;z>=0;z--)
    			{
    				if(z*6+j*9+i*20==n)
    				{
    					cout<<"True\n";
    					d=1;
    					break;
    				}
    			}
    			if(d==1)
    			break;
    		}
    			if(d==1)
    			break;
    	}
    	
    			if(d==0)
    	    	cout<<"False\n";
    	
    }
     return 0;
}
