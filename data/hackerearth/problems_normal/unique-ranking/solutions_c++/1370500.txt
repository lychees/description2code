#include <bits/stdc++.h>
/**
 *
 * @author bassem
 */
using namespace std;
int main(void){
    int n,k;
    scanf("%d%d",&n,&k);
    int arr[n][k];
    for (int i = 0; i < n; ++i)
    {
        for (int j = 0; j < k; ++j)
        {
            scanf("%d",&arr[i][j]);
        }
    }
    int q;
    scanf("%d",&q);
    while(q--){
        int m;
        scanf("%d",&m);
        int x;
        std::vector<int> id;
        scanf("%d",&x);            
        int mini = 101;
        for (int i = 0; i < n; ++i)
        {
            mini = min(arr[i][x-1],mini);
        }
        for (int i = 0; i < n; ++i)
        {
            if (mini==arr[i][x-1])
            {
                id.push_back(i);
            }
        }     
        for (int i = 1; i < m; ++i)
        {
            scanf("%d",&x);
            mini = 101;
            vector<pair<int,int > > hist;
            for (int i = 0; i < id.size(); ++i)
            {
                mini = min(arr[id[i]][x-1],mini);
                hist.push_back(make_pair(arr[id[i]][x-1],id[i]));
            }
            id.clear();
            sort(hist.begin(),hist.end());
            for (int i = 0; i < hist.size(); ++i)
            {
            	if (hist[i].first==mini)
                {
                    id.push_back(hist[i].second);
                }else if(hist[i].second>mini){
                	break;
                }
            }
        }
        sort(id.begin(),id.end());
        printf("%d\n", id[0]+1);
    }
}
