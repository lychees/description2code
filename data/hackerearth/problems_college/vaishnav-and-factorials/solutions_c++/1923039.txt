#include<iostream>
using namespace std;

#define MAX 500

int mul(int x, int res[], int re);

void fact(int n)
{
    int res[MAX],c;
    res[0] = 1;
    int re = 1;
    for (int x=2; x<=n; x++)
        re= mul(x, res, re);
    for (int i=re-1; i>=0; i--)
        if(res[i]==4||res[i]==7)
        	++c;
    cout<<c<<"\n";
}

int mul(int x, int res[], int re)
{
    int c = 0; 
    for (int i=0; i<re; i++)
    {   int prod = res[i] * x + c;
        res[i] = prod % 10; 
        c  = prod/10;   
    }
    while (c)
    {
        res[re] = c%10;
        c = c/10;
        re++;
    }
    return re;
}

int main()
{	long long int n,x;
	cin>>n;
	while(n--)
	{	cin>>x;
    	fact(x);
	}
    return 0;
}
