#include <iostream>
using namespace std;
long long maxx(long long x ,  long long y)
{
	if(x > y)
	return x ; 
	return y ; 
}
int main()
{
   int t;  
   scanf("%d",&t) ; 
   while(t--)
   {
   	long long x,y; 
   	scanf("%lld %lld",&x,&y) ; 
   	if(x == (y-1))
   	printf("%lld\n",x&y) ; 
   	else
   	{
   		long long ans = maxx(y&(y-1) , (y-1)&(y-2)) ; 
   		printf("%lld\n",ans) ; 
   	}
   }
    return 0;
}
