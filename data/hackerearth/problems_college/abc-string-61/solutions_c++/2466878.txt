#define M 1000000007
#define LL long long
long long modpow(LL base, LL exp)
{
  base %= M;
  LL result = 1;
  while (exp > 0) {
    if (exp & 1) result = (result * base) % M;
    base = (base * base) % M;
    exp >>= 1;
  }
  return result;
}
int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        LL f = modpow(3,n-1);
        //cout<<f<<endl;
        LL s = modpow(2,n);
        //cout<<s<<endl;
        f = f-s+1;
        if(f<0) f+=M;
       // cout<<f<<endl;
        f*=3;
       // cout<<f<<endl;
        f%=M;
        cout<<f<<endl;
    }
    return 0;
}
