#include <iostream>
#include <stdio.h>
using namespace std;
long long arr[1000001];
long long GCD(long long a , long long b){
   if(b == 0)
       return a;
    return GCD(b , a % b);
}
int main()
{
    int t ; 
    scanf("%d",&t);
    while(t--){
    	int n ;
    	scanf("%d",&n);
    	for(int i = 0 ; i < n ; i++){
    		int x;
    		scanf("%lld",&x);
    		arr[i] = x;
    	}
    	long long gcd = arr[0];
    	for(int i = 1 ; i < n ; i++){
    		gcd = GCD(gcd,arr[i]);
    	}
    	printf("%lld\n",gcd * n);
    }
    return 0;
}
