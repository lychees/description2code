main()
{
	
	int n;
	
	si(n);
	
	char s[n+1];
	
	scanf("%s",s);
	
	int count = 0;
	
	rep(i,n)
	{
		
		if(s[i] == 'B')
		   count += (1<<i);
		
	}
	
	ll dp[(1<<20) + 5];
	
	mset(dp,0);
	
	//ll ans = 10000000000000;
	
	queue<int> pq;
	
	pq.push(count);
	
	while(not pq.empty())
	{
		
		count = pq.front();
		
		pq.pop();
		
		if(count == 0 || count == (1<<n) - 1)
		{
		  
		    printf("%d",dp[count]);
		    return 0;
		}
		
		for(int i = 0; i <= n - 2;i++)
		{
			
			int x = count ^ (1<<i) ^ (1<<(i+1));
			
			if(dp[x] == 0)
			{
				
				dp[x] = dp[count] + 1;
				pq.push(x);
				
			}
			
			if(i <= n -3)
			{
				
				x = count ^ (1<<i) ^ (1<<(i+1)) ^ (1<<(i+2));
				
				if(dp[x] == 0)
				{
					
					dp[x] = dp[count] + 1;
					pq.push(x);
					
				}
				
			}
			
		}
		
	}
	
}
