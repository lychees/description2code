ll isPalin(char arr[])
{
    ll i,len;
    len=strlen(arr);
    for(i=0;i<=len/2;i++)
    {
        if(arr[i]!=arr[len-1-i])
            return 1;
    }
    return 0;
}

int main()
{
	ll tc,test1,test2;
	char str1[10001],str2[10001],str3[100001],str4[100001];
	scanf("%lld",&tc);
	while(tc--)
	{
		scanf("%s",str1);
		scanf("%s",str2);
		strcpy(str3,str1);
		strcpy(str4,str2);
		strcat(str3,str4);

		    printf("YES\n");
		else

	}
	return 0;	
	   
}
