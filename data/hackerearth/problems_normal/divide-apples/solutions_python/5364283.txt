N=input()
s=raw_input().split(' ')
l=[]
count=0
middle_one=0
z=0
p=0
negative=[0]*N
sss=[0]*N
for i in range(N):
	l.append(int(s[i]))
	count+=int(s[i])
tot=count/N
for i in range(N):
	negative[i]=l[i]-tot
for i in range(N):
	sss[i]=p
	p+=negative[i]
sss.sort()
if N%2==0:
	middle_one=(sss[(N-1)/2]+sss[N/2])/2
else:
	middle_one=sss[N/2]
for i in range(N):
	z+=abs(sss[i]-middle_one)
print z
