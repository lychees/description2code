#include <cstdio>

using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    while (t--)
    {
        char str[1000];  
        scanf("%s", str);
        int sum = 0;
        for (int i=0 ; str[i] ; i++)
        {
           if (str[i] != '_') 
               sum = sum + str[i] - '0';
        }    
           
        if ((sum % 9) == 0)
            printf("0 or 9\n");
        else
            printf("%d\n", 9 - sum%9);
    } 
    return 0;    
}
