    #define pc putchar

using namespace std;
ll read_Long()
{
  char c = gc();
  while(c<'0' || c>'9')
  c = gc();
  ll ret = 0;
  while(c>='0' && c<='9')
  {
    ret = 10 * ret + c - 48;
    c = gc();
  }
  return ret;
}
void write_Long(ll n)
{
    ll N = n, rev, count = 0;
    rev = N;
    if (N == 0) { pc('0'); pc('\n'); return ;}
    while ((rev % 10) == 0) { count++; rev /= 10;}
    rev = 0;
    while (N != 0) { rev = (rev<<3) + (rev<<1) + N % 10; N /= 10;}
    while (rev != 0) { pc(rev % 10 + '0'); rev /= 10;}
    while (count--) pc('0');
    pc('\n');
}
ll i;
char str[2000018];
main()
{
    ll t=read_Long();
    while(t--)
    {
        ll c=0;
        ll d=0;
        ll N=read_Long();
        for(i=0;i<2*N;i++)
        cin>>str[i];
        for(i=0;i<2*N;i=i+2)
        {
            if(str[i]!='B')
            c++;
            else
            d++;
        }
        if(c<d)
        write_Long(c);
        else
        write_Long(d);
        for(i=0;i<2*N;i++)
        str[i]='\0';
    }
}
