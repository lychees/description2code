from sets import Set
T = int(raw_input())
for t in xrange(T):
    [a, b] = [int(x) for x in str(raw_input()).split(" ")]
    grid = range(a)
    for i in xrange(a):
        grid[i] = int(raw_input(), 2)
    if len(Set(grid)) == a:
        print "Yes"
    else:
        print "No"
