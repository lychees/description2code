#include <iostream>
using namespace std;

int main() {
	int t, n, i, j, sum, count;
	cin >> t;

	while (t--) {
		cin >> n;
		int power[n];

		i = 0;
		while (i < n) {
			cin >> power[i];
			i++;
		}

		i = count = 0;
		while (i < n) {
			j = i + 1;
			while (j < n) {
				sum = power[i] + power[j];
				if ( (sum % 3) == 0)
					count++;
				j++;
			}
			i++;
		}
		cout << count << endl;
	}
	return 0;
}
