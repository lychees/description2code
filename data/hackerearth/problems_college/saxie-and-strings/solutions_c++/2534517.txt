#include <iostream>
using namespace std;

int main()
{
    int t,i,j;
    cin>>t;
    while(t--){
    	string str;
    	cin>>str;
    	int flag = 1;
    	i = 0;
    	j = str.size()-1;
    	while(i<j){
    		if(str[i]!=str[j]){
    			flag=0;
    			break;
    		}
    		i++;
    		j--;
    	}
    	if(flag==1){
    		cout<<"YES"<<endl;
    	}else{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
