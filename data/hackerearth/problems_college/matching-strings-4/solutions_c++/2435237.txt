#include <iostream>
#include<map>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int count = 0;
    	string str1, str2;
    	cin>>str1>>str2;
    	int n1 = str1.length(), n2 = str2.length();
    	int min = n1<n2?n1:n2;
    	for(int i=0; i<min; i++)
    	{
    		if(str1[i] == str2[i])
    		count++;
    	}
    	cout<<count<<"\n";
    }
    return 0;
}
