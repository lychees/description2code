#include <iostream>
#include <string>
#include <cmath>

using namespace std;

int main(){

	int t,swaps;
	int swap1,swap2;
	cin >>t ;
	int count_u,count_s;
	string str;

	while(t--){
		cin>>str;
		swaps=0;
		count_u=count_s=0;
		swap1=swap2=0;
		for(int i=0;i<str.size();i++){
			if(str[i]=='U')count_u++;
			if(str[i]=='S')count_s++;
		}

		if(abs(count_s - count_u)>1){
			swaps=-1;
		}

		else if(abs(count_s - count_u)==1 && str.size()%2==0){
			swaps=-1;
		}
		else {
			//begin S
			
				if(str.size()%2==0){
					for(int i=0;i<str.size();i++){
						if(i%2==0){
							if(str[i]=='S')swap1++;
							if(str[i]=='U')swap2++;
						}
						else{
							if(str[i]=='U')swap1++;
							if(str[i]=='S')swap2++;
						}
						swaps=min(swap1,swap2);
						swaps/=2;
					}
				}
				else{
					if(count_s>count_u){

						for(int i=0;i<str.size();i++){
							if(i%2==0){
								if(str[i]=='U')swap1++;
							}
							else{
								if(str[i]=='S')swap1++;
							}
							swaps=swap1;
							swaps/=2;
						}
					}
					else{
						for(int i=0;i<str.size();i++){
							if(i%2==0){
								if(str[i]=='S')swap1++;
							}
							else{
								if(str[i]=='U')swap1++;
							}
							swaps=swap1;
							swaps/=2;
						}
					}
				}
				
			}
			
		
		

		
		cout<<swaps<<endl;
	}

	return 0;
}
