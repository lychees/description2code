#define si(x) scanf("%d",&x)

using namespace std;

typedef long long ll;
int main(){
	int t;
	cin>>t;
	while(t--){
		string str;
		cin>>str;
		int count=0;
		int n=str.length();
		for(int i=0;i<n;i++){
			if(str[i]=='O'){
				int temp_count=0;
				while((i)<n&&str[i]!='X'){
					if(str[i]=='*'){
						count+=temp_count;
						temp_count=0;
					}
					else{
						temp_count++;
					}
					i++;
				}
				if(i>=n){
					count+=temp_count;
					break;
				}
				else{
					i--;
				}
			}
			else if(str[i]=='X'){
				while(i<n&&str[i]!='*'){
					i++;
				}
			}
		}
		cout<<count<<endl;
	}
}
