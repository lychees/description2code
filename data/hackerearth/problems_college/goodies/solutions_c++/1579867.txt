#include <iostream>
#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<math.h>
using namespace std;
int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,a[100000];
		cin>>n;
		 
		for(int i=0;i<n;i++)
		cin>>a[i];
		qsort(a,n,sizeof(int),compare);
		for(int i=0;i<n;i++)
		{
			for(int j=i+1;j<n;j++)
			{
				if(a[i]<a[j])
				{
					a[i]=-1;
					break;
				}
			}
		}
		int cnt=0;
		for(int i=0;i<n;i++)
		{
			if(a[i]!=-1)
			cnt++;
		}
		cout<<cnt<<endl;
	}
}
