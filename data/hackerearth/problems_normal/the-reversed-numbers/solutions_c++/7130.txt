#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;

typedef vector<int> vi;

vi get_digits(int n){
  vi dig;
  while(n){
    dig.push_back(n%10);
    n=n/10;
  }
  return dig;
}

int get_num(vi dig){
  int num = 0;
  for(int i=0; i<dig.size(); i++){
    num = 10*num + dig[i];
  }
  return num;
}

int main(){
  int t, num1, num2;
  vi dig;
  scanf("%d", &t);
  while(t--){
    scanf("%d %d", &num1, &num2);
    int n = get_num(get_digits(num1)) + get_num(get_digits(num2));
    cout<<get_num(get_digits(n))<<endl;
  }
}
