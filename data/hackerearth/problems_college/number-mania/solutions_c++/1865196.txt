#include <iostream>
#include <map>
using namespace std;

int dp[102] = {0};
map <int, string> m;
int res[501];
int multiply(int x, int res_sz)
{
	int c = 0;
	for (int i = 0; i < res_sz; i++) {
		int prod = res[i] * x + c;
		res[i] = prod % 10;
		c = prod / 10;
	}
	while (c) {
		res[res_sz] = c % 10;
		c /= 10;
		res_sz++;
	}
	return res_sz;
}
void factorial(int n)
{
	res[0] = 1;
	int res_size = 1;
	m[1] = "1";
	for (int x = 2; x <= n; x++) {
		res_size = multiply(x, res_size);
		string tt = "";
		for (int j = res_size - 1; j >= 0; j--) {
			tt += (char) (res[j] + '0');
		}
		m[x] = tt;
	}
}

int main()
{
	dp[0] = 1;
	dp[1] = 1;
    for (int i = 2; i < 102; i++) {
    	for (int j = 2 * i; j < 102; j += i) {
    		dp[j] = 1;
    	}
    }
    int n;
    cin >> n;
    factorial(n);
    for (int i = 1; i <= n; i++) {
    	if (dp[i] == 0) {
    		cout << m[i] << endl;
    	}
    }
    if (n == 1) {
    	cout << endl;
    }
    
    return 0;
}
