#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;
long long int max(long long int a,long long int b)
{
	return a>b?a:b;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,a,b,c,p,q,i;
    	cin>>n>>a>>b>>c;
    	vector<pair<long long,long long> > v;
    	for(i=0;i<n;i++)
    	{
    		cin>>p>>q;
    		v.push_back(make_pair(p,1));
    		v.push_back(make_pair(q,2));
    	}
    	sort(v.begin(),v.end());
    	long long int ans=n*a,cmax=n*a;
    	for(i=0;i<v.size();i++)
    	{
    		if(v[i].second==1)
    		cmax+=b-a;
    		else
    		cmax+=c-b;
    		ans=max(ans,cmax);
    	}
    	cout<<ans<<endl;
    	v.clear();
    }
}
