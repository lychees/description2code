#include <iostream>
using namespace std;

int main()
{
    int t,a,b;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	int m=0,n=0;
    	while(a)
    	{
    		m=m*10+a%10;
    		a=a/10;
    	}
    	while(b)
    	{
    		n=n*10+b%10;
    		b=b/10;
    	}
    	m=m+n;
    	n=0;
    	while(m)
    	{
    		n=n*10+m%10;
    		m=m/10;
    	}
    	cout<<n<<endl;
    }
    return 0;
}
