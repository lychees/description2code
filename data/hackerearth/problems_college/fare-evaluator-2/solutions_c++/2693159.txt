#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,i,j,d,wt,fare;
	cin>>t;
	for(i=0;i<t;i++)
	{
		fare=0;
		cin>>d>>wt;
		if(d>60)
		{
			cout<<"NA"<<endl;
			continue;
		}
		if(d<=2)
		{
			fare=50+2*wt;
			cout<<fare<<endl;
			continue;
		}
		fare=12*(d-2)+2*wt+50;
		cout<<fare<<endl;
		
	}
	return 0;
}
