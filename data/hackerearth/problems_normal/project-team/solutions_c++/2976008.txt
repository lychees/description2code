#include <iostream>
#include <algorithm>
#include <cstdio>
#define max 1005
#define lli long long int
using namespace std;

int main() {
	lli test , size;
	lli array[max];

	scanf("%lld",&test);

	while(test--){

		scanf("%lld",&size);
		for(lli i=0;i<size;++i){
			scanf("%lld",&array[i]);
		}

		sort(array,array+size);

		lli maximum = 0 , minimum=9999;

		for(lli i=0 , j=size-1 ; i<j ; ++i , --j){
			if((array[i] + array[j]) > maximum)
				maximum =array[i] + array[j];
			if((array[i] + array[j]) < minimum)
				minimum = array[i] + array[j];
		}

		printf("%lld\n",maximum - minimum);
	}
	return 0;
}
