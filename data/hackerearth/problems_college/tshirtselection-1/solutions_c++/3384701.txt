#include <iostream>
using namespace std;
#include<bits/stdc++.h>
void eval()
{
	int x,n;
	cin>>n;
	map<int,int> mp;
	int sz=0,start=0;
	int currentStart = 0;
	for(int i=0;i<n;i++)
	 {
	 	cin>>x;
	 	if(mp.find(x)!=mp.end())
			{currentStart = max(currentStart, mp[x]+1);	
			}
		mp[x]=i;
		if(i-currentStart+1>sz)
		 {
		 	sz=i-currentStart+1;
		 	start = currentStart;
		 }
 	 }
 	 cout<<start+1<<" "<<start+sz<<endl;
	
	
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	 {
	 	eval();
	 }
}
