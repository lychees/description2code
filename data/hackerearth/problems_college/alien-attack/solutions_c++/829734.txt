#include <bits/stdc++.h>
#define mod	1000000007
int modulo(int a,int b,int c){
    long long x=1,y=a; // long long is taken to avoid overflow of intermediate results
    while(b > 0){
        if(b%2 == 1){
            x=(x*y)%c;
        }
        y = (y*y)%c; // squaring the base
        b /= 2;
    }
    return x%c;
}
int main()
{
    int t,n;
    long long int ans;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        n--;
        ans=modulo(2,n,mod)%mod;
        ans=(ans*3)%mod;

        if(n%2==0)
        {
            if(ans>1)
            ans=(ans-1)%mod;
            else
                ans=mod-1;
        }
        else
        ans=(ans+1)%mod;
        printf("%lld\n",ans);
    }
    return 0;
}
