{
	int t;
	cin >> t;
	while(t--){
		long long l, r;
		cin >> l >> r;
		long long l1 = (ll)floor(sqrt(l))+1;
		ll r1 = (ll)floor(sqrt(r));
		bool rc = false, lc = false;
		if(r > (ll)999999998000000001 && r < (ll)1000000000000000000) r1 = (ll)999999999, rc = true;
		if(l > (ll)999999998000000001 && l < (ll)1000000000000000000) l1 = (ll)999999999, lc = true;
		//cout << rc << " " << lc << endl;
		if((ll)floor((ll)sqrt(l))*(ll)floor((ll)sqrt(l)) == l && !lc) l1 = (ll)sqrt(l);
		if((ll)floor((ll)sqrt(r))*(ll)floor((ll)sqrt(r)) == r && !rc) r1 = (ll)sqrt(r);
		//cout << l1 << " " << r1 << endl;
		cout << (ll)r1-(ll)l1+1 << endl;
	}
}
