    #include<stdio.h>
    #include<string.h>
    int min(int a,int b){
    if(a<b)
    	return a;
    return b;
     
    }
    int main(){
    	int t;
    	scanf("%d",&t);
    	while(t--){
    		char c[10001];
    		int ac,bc;
    		scanf("%s",c);
    		scanf("%d%d",&ac,&bc);
    		long long int sum=0;
    		int flag=0;
    		int l=strlen(c);
    		int i=0,j=l-1;
    		while(i<j){
    			if(c[i]==c[j] && c[i]!='/'){
    				i++;
    				j--;
    			}
    			else if((c[i]=='a' && c[j]=='/')||(c[i]=='/' && c[j]=='a')) {
    				sum+=ac;
    				flag=1;
    				i++;j--;
    			}
    			else if((c[i]=='b' && c[j]=='/')||(c[i]=='/' && c[j]=='b')) {
    				sum+=bc;
    				flag=1;
    				i++;
    				j--;
    				
    			}
    			else if((c[i]=='/' && c[j]=='/')){
    				sum+=(2*min(ac,bc));
    				flag=1;
    				i++;j--;
    			}
    			else if((c[i]!=c[j]))
    			{
    				flag=0;
    				break;
    			}
    			
    			else{
    				i++;j--;
    			}
    				
    		
    		
    		}
    		if(flag)
    			printf("%lld\n",sum);
    		else
    			printf("-1\n");
    			
    		
    		
    	}
     
    return 0;
    }
