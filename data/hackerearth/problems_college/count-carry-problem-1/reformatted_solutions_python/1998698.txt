t = input()
while t:
    t -= 1
    a, b = map(str, raw_input().split())
    c = 0
    for x in range(0, len(a)):
        if (int(a[x])+int(b[x])) > 9:
            c += 1
    if c == 0:
        print 'No carry operation'
    elif c == 1:
        print str(c)+' carry operation'
    else:
        print str(c)+' carry operations'
