#define MAX(a,b) (a>b?a:b)
#define MIN(a,b) (a<b?a:b)
#define FOR(i,n) for(i=0;i<n;i++)
#define FORJAS(i,j,n) for(i=j;i<n;i++)
#define FORS(i,n) for(i=1;i<=n;i++)
#define FORJS(i,j,n) for(i=j;i<=n;i++)
#define LEN 1000005


int main(){
	
	long long i,j;

	long long x,y,cnt;
	int t;	
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld %lld",&x,&y);	
		cnt=0;
		vector<bool> mark(y-x+1,true);
		
		for(i=2;i*i<=y;i++)	
		{
			long long gap=i*i;
			long long k=x+(gap-x%gap)%gap;
			for( j=k;j<=y;j+=gap)
				mark[(j-x)]=false;
		}
		
		printf("%lld\n",(count(mark.begin(),mark.end(),true)));
		
	} 
	return 0;
}
