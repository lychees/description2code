#include <cstdio>
#include <iostream>
#include <cassert>
using namespace std;
#define MAXN 100000
int ar[MAXN+1];
int val(int i){
	int state=0;
	int ans=0;
	while(i){
		if(i&1){
			if(state==1)
				state = 1;
			else if(state==2)
				state = 3;
			else if(state==0)
				state = 1;
		}else{
			if(state==1)
				state = 2;
			else
				state = 0;
		}
		if(state==3){
			ans++;
			state=1;
		}
		i = i/2;
	}
	return ans;
}
void init(){
	for(int  i=1;i<=MAXN;i++){
		ar[i] = val(i);
	}
}
int main(){
	int T, R, K;
	init();
	cin >> T;
	assert(T>0 and T<=100);
	while(T--){
		cin >> R >> K;
		assert(R>0 and R<=100000);
		assert(K>0 and K<=100);
		int ans = 0;
		for(int i=1;i<=R;i++)
			if (ar[i]>=K)
				ans++;
		cout << ans << endl;
	}
}
