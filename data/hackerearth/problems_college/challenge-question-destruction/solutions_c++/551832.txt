#include <iostream>
#include<algorithm>
using namespace std;

int area[11],temp[11],pos,mini=100;
int main()
{
    for(int i=1;i<=10;i++)
    {
    	cin>>area[i];
    }
   	temp[0]=area[1];
   	temp[1]=area[4];
   	temp[2]=area[7];
   	temp[3]=area[10];
   	temp[4]=area[3];
   	temp[5]=area[6];
   	temp[6]=area[9];
   	temp[7]=area[2];
   	temp[8]=area[5];
   	temp[9]=area[8];
   	for(int i=0;i<10;i++)
   	{
   		if(mini>temp[i])
   			{mini=temp[i],pos=i;}
   	}
   	int lidx=(pos-1);
   	int ridx=(pos+1)%10;
   	if((pos-1)<0)
   		{
   			lidx=9;
   		}
   	cout<<temp[pos]<<" ";
   	for(int i=1;i<9;i++)
   	{
		if(temp[lidx]<temp[ridx])
		{
			cout<<temp[lidx]<<" ";
			if(lidx==0)
				lidx=9;
			else
				lidx=lidx-1;
		}
		else
		{
				cout<<temp[ridx]<<" ";
				ridx=(ridx+1)%10;
		}

   	}
    return 0;
}
