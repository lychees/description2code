#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int t;
	int n;
	int i;
	cin >> t;
	int sum =0;
	while(t--){
		sum = 0;
		cin >> n;	
		int a1[n],a2[n];
		for (i =0; i <n; i++)
			cin >> a1[i];
		for (i =0; i <n; i++)
			cin >> a2[i];
			
		sort(a1,a1+n);
	    sort(a2,a2+n);

	for (i =0; i < n; i++ )		
		sum = sum + a1[i]*a2[i];
	cout << sum << endl;
	}
    return 0;
}
