#include <iostream>
using namespace std;
typedef long long int ll;
ll calc(ll a, ll b, ll mod) {
	ll ans = 1;
	while(b > 0) {
		if(b%2 == 1) {
			ans = (ans * a)%mod;
		}
		a  = (a * a)%mod;
		b/=2;
	}
	return ans;
}
int main()
{
    int t;
    cin >> t;
    ll mod = 1000000007;
    while(t--) {
    	ll n;
    	cin >> n;
    	ll a[n];
    	ll b[n];
    	ll i, k;
    	ll ans = 1;
    	for(i = 0; i < n; i++) {
    		cin >> a[i];
    	}
    	for(i = 0; i < n; i++) {
    		cin >> b[i];
    	}
    	for(i = 0; i < n; i++) {
    		k = calc(a[i], b[i] + 1, mod);
    		k = (k - a[i] + mod)%mod;
    		ans = (ans * k) % mod;
    		ans = ans * calc(a[i] - 1, mod - 2, mod);
    		ans = ans%mod;
    	}
    	cout << ans << endl;
    }
    return 0;
}
