#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin >> t;
    string s;
    vector<string> words;
    while( t > 0 )
    {
        cin >> s;
        words.push_back(s);
        t--;
    }
    for(int i=0; i<words.size(); i++)
    {
        for(int j=i+1; j<words.size(); j++)
        {
            if( words[i].length() == words[j].length() )
            {
                int k=0;
                int len = words[i].length();
                int l=len-1;
                while( (k<len) && (words[i][k] == words[j][l]) )
                {
                    k++;
                    l--;
                }
                if(k == len)
                {
                    cout << len << " " << words[i][(len-1)/2] << endl;
                    i = words.size()+5;
                    break;
                }
            }
        }
    }
}
