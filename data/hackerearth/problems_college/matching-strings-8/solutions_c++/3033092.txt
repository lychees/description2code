#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	string a,b;
    	cin>>a>>b;
    
 		int m = min(a.length(), b.length());
    	int count = 0;
    	
    	for(int i = 0; i<m; i++){
    		if(a[i] == b[i]){
    			count++;
    		}
    	}
    	cout<<count<<endl;
    	
    }
    return 0;
}
