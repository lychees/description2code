#include<bits/stdc++.h>
using namespace std;
int n,m,x,y,a[1005]={0},b[1005]={0};
int sh[1005]={0},dot[1005]={0};
int xsh(int i)
{
    int sum=0;
    for(int j=1;j<x&&i>=0;--i,++j)sum+=b[i];
    if(i<0)return sum;
    int mn=1e7;
    for(int j=x;j<=y&&i>=0;++j,--i)
    {
        sum+=b[i];
        if(i==0)mn=min(mn,sum);
        else
        mn=min(mn,sum+dot[i-1]);
    }
    return mn;
}
int main()
{
    cin>>n>>m>>x>>y;
    char c;
    for(int i=0;i<n;++i)
    {
        scanf("\n");
        for(int j=0;j<m;++j)
        {
            scanf("%c",&c);
            if(c=='#')++a[j];
            else ++b[j];
        }
    }
    dot[0]=a[0];
    sh[0]=b[0];
    /*for(int i=0;i<m;++i)cout<<a[i]<<" "<<b[i]<<endl;

    cout<<endl<<endl;*/
    for(int i=0;i<x-1;++i)sh[i]=dot[i]=1e7;
            for(int i=x-1;i<m;++i)
    {
        sh[i]=xsh(i);
        swap(a,b);
        swap(dot,sh);
        sh[i]=xsh(i);
        swap(a,b);
        swap(dot,sh);
       //cout<<sh[i]<<" "<<dot[i]<<endl;
    }
    cout<<min(sh[m-1],dot[m-1]);


}
