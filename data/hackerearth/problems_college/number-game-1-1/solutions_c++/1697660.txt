#include <iostream>
using namespace std;

int main()
{
    int t,c;
    cin>>t;
    while(t--)
    {
    	c=0;
    	long long int i;
    	cin>>i;
    	while(i!=1)
    	{
    		if(i%3==0)
    		{
    			i/=3;
    			c++;
    		}
    		else if(i%2==0)
    		{
    			i/=2;
    			c++;
    		}
    		else
    		{
    			--i;
    		    c++;
       	 	}
       	 	
    	}
    	cout<<c<<endl;
    	
    }
    return 0;
}
