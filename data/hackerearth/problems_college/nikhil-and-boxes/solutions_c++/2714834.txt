#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	long long l,b,h,small;
	cin>>t;
	while(t--)
	{
		cin>>l;
		cin>>b;
		cin>>h;
		if(l<h)
		{
			if(l<b)
			small=l;
			else
			small=b;
		}
		else
		{
			if(h<b)
			small=h;
			else
			small=b;
			
		}
		printf("%.1f",(double)small/2);
		cout<<'\n';
	}
}
