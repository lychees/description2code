#include <bits/stdc++.h>
#define mod 1000000009
using namespace std;

long long mat_power(long long n) {
	long long ans = 0;
	long long A[2][2]  = {1,1,1,0};
	long long B[2][1] = {1,0};
	long long result[2][2] = {1,0,0,1};
	int multiply[2][2];
	while(n) {
		if(n&1) {
			for(int i=0;i<2;i++) {
				for(int j=0;j<2;j++) {
					multiply[i][j] = 0;
					for(int k=0;k<2;k++) {
						multiply[i][j] = ( multiply[i][j] + (result[i][k]* A[k][j]) )%mod;
 					}
				}
			}
			
			for(int i=0;i<2;i++) {
				for(int j=0;j<2;j++) {
					result[i][j] = multiply[i][j];
				}
			}
		}
		
		
		n>>=1;
		
		for(int i=0;i<2;i++) {
			for(int j=0;j<2;j++) {
				multiply[i][j] = 0;
				for(int k=0;k<2;k++) {
					multiply[i][j] = ( multiply[i][j] + (A[i][k]*A[k][j]) )%mod;
				}
			}
		}
		
		for(int i=0;i<2;i++) {
			for(int j=0;j<2;j++) {
				A[i][j] = multiply[i][j];
			}
		}
	}
	long long ans1= ((result[0][0]*B[0][0])%mod+(result[0][1]*B[1][0])%mod)%mod;
	long long ans2= ((result[1][0]*B[0][0])%mod+(result[1][1]*B[1][0])%mod)%mod;
	ans = (ans1*ans2)%mod;
	return ans;
}
int main()
{
	int t;
	scanf("%d",&t);
	
	while(t--) {
		long long n;
		long long ans;
		scanf("%lld",&n);
		
		
		ans = mat_power(n);
		printf("%lld\n",ans);
	}
}
