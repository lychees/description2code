#include <iostream>
using namespace std;

int main()
{
	int t,i,no;
	bool flag;
	string n;
	cin>>t;
	while(t--)
	{
		no=0;
		flag=true;
		cin>>n;
		for(i=0;i<(n.length()-1);i++)
		{
			no = no*10 + (n[i] -'0');
			if(n[i]=='2' && n[i+1]=='1')
			{
				flag=false;
				break;
				cout<<"The streak is broken!"<<endl;
			}
		}	
		no = no*10 + (n[i] -'0');
		if(flag==true && no%21!=0)
		{
			cout<<"The streak lives still in our heart!"<<endl;
		}
		else
		{
			cout<<"The streak is broken!"<<endl;
		}
	}
	return 0;
}
