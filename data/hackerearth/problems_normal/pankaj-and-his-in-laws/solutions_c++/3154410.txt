#include<bits/stdc++.h>
using namespace std;
const int MAXN = 102;
int dp[MAXN];
int ar[MAXN];
int main() {
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>ar[i];
	dp[0]=0;
	int maxi=-1;
	for(int i=1;i<=n;i++) {
		dp[i]=1;
		for(int j=i-1;j>=1;j--) {
			if(ar[j]<ar[i])
				dp[i]=max(dp[i],dp[j]+1);
		}
		maxi=max(maxi,dp[i]);
	}
	vector<int> ans;
	while(maxi!=0) {
		ans.push_back(maxi%2);
		maxi/=2;
	}
	reverse(ans.begin(),ans.end());
	for(int i=0;i<(int)ans.size();i++)
		printf("%d",ans[i]);
	printf("\n");
	return 0;
}
