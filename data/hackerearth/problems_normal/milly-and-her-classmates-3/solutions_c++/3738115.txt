

#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>

using namespace std;

int main() {
    int T, N, temp;
    cin >> T;
    while(T--) {
        cin >> N;
        vector<int> height;
        for(int i = 0; i < N; i++) {
            scanf("%d", &temp);
            height.push_back(temp);
        }
        if(is_sorted(height.begin(), height.end()))
            cout << "-1" << " " << "-1" << endl;
        else if(is_sorted(height.rbegin(), height.rend()))
            cout << "1" << " " << N << endl;
        else {
            int min = height[0], start = 0, end = 0;
            for(int i = 1; i < N; i++) {
                if(height[i] > height[i - 1] && min == height[0]) {
                    
                }
                else if(min == height[0] && height[i] < height[i - 1]) {
                    min = height[i];
                    start = i - 1;
                    end = i;
                }
                else if(height[i] < height[i - 1]) {
                    min = height[i];
                    end = i;
                }
                else
                    break;
            }
            vector<int> vec = height;
            reverse(vec.begin() + start, vec.begin() + end + 1);
            if(is_sorted(vec.begin(), vec.end()))
                cout << start + 1 << " " << end + 1 << endl;
            else
                cout << "-1" << " " << "-1" << endl;
        }
    }
    return 0;
}
