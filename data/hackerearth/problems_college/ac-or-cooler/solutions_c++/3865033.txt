#include<bits/stdc++.h>
using namespace std;
#define mx (int(1e6)+9)
int arr[mx];
int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=1;i<=n;++i)
	{
		cin>>arr[i];
	}
	for(int i=n+1;i<=2*n;++i)
	{
		cin>>arr[i];
	}
	sort(arr+1,arr+2*n+1);
	int sum=0;
	int cnt=0;
	int i=1;
	while(sum<=m)
	{
		sum+=arr[i];	
		i+=1;
		cnt++;
	}
	cout<<min(n,cnt-1)<<endl;
	return 0;
}
