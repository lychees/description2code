#include <algorithm>
#include <cstdint>
#include <iostream>

inline void check(int b_min, int b_max, int m_min, int m_max)
{
    if(b_max < m_min || m_max < b_min)
    {
        std::cout << "Line";
    }
    else if (b_max == m_min || m_max == b_min ||
            (b_max == b_min && b_max <= m_max && b_max >= m_min) ||
	          (m_max == m_min && m_max <= b_max && m_max >= b_min))
    {
        std::cout << "Point";
    }
    else
    {
        std::cout << "Nothing";
    }
    std::cout << std::endl;
}

int main()
{
    unsigned T;
    std::cin >> T;
    int bt1, bt2, mt1, mt2;
    for (unsigned i = 0; i < T; ++i)
    {
        std::cin >> bt1 >> bt2 >> mt1 >> mt2;
        auto bminmax = std::minmax(bt1, bt2);
        auto mminmax = std::minmax(mt1, mt2);
        check(bminmax.first, bminmax.second, mminmax.first, mminmax.second);
    }
}
