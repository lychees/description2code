m = input()
boy = [0] * 10003
temp = map(int, raw_input().split())
for i in range(m):
    boy[temp[i]] = 1
n = input()
ans = 0
for i in range(n):
    cnt = 0
    girl = map(int, raw_input().split())
    l = len(girl)
    
    for j in range(l):
        if boy[girl[j]] == 1:
            cnt += 1

    if cnt == m:
        ans += 1
print ans
