#include <bits/stdc++.h>

using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    int T, n, k, *A, one;
    cin >> T;
    while(T--)
    {
        cin >> n >> k;
        A = new int[n];
        one = 0;
        for(int i=0;i<n;++i)
        {
            cin >> A[i];
            if(A[i])
                one++;
        }
        if(n-one >= k)
            cout << one+k << endl;
        else
            cout << n-((k-n+one)%2) << endl;
    }
    return 0;
}
