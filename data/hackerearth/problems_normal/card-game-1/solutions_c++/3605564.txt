#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	char name[100];
	long long int t,n,a[100000],b[100000],i,j,k;
	cin>>t;
	while(t>0)
	{
		t--;
		int flag=0;
		cin>>n;
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0;i<n;i++)
		cin>>b[i];
		cin>>name;
		for(i=0;i<n;i++)
			{
				if(a[i]!=b[i])
				{
					flag++;
				}
			}
		if(strcmp(name,"Devu")!=0)
		{
			if(n==2)
			{
				if(flag==2)
				cout<<"Devu\n";
				continue;
			}
			cout<<"Churu\n";
			continue;
		}
		else
		{
			
			
			if(flag==2)
			cout<<"Devu\n";
			else
			cout<<"Churu\n";
		}
		
	}
    return 0;
}
