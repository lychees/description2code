#include<bits/stdc++.h>
using namespace std;
vector<int>v[100005];
int vis[100005];
void dfs(int ind)
{
	vis[ind]=1;
	int i;
	for(i=0;i<v[ind].size();i++)
	{
		if(!vis[v[ind][i]])
		{
			dfs(v[ind][i]);
		}
	}
}
int main()
{
	int n;
	int e;
	scanf("%d %d",&n,&e);
	int i,j;
	for(i=0;i<e;i++)
	{
		int a,b;
		scanf("%d %d",&a,&b);
		v[a].push_back(b);
		v[b].push_back(a);
	}
	int co=0;
	for(i=1;i<=n;i++)
	{
		if(!vis[i])
		{
			dfs(i);
			co++;
		}
	}
	printf("%d\n",co-1);
}
