#include<iostream>
#include<limits.h>
using namespace std;
int n;
int arr[100][100];
int maxCoins(int *,int *);
int max(int x, int y)
{
   if (x > y)
      return x;
   else
      return y;
}
int maxCoins(int p,int q){
	

	int t[p][q];
	t[0][0]=arr[0][0];
	for(int i=1;i<p;i++)
	t[i][0]=t[i-1][0]+arr[i][0];
	for(int j=1;j<q;j++)
	t[0][j]=t[0][j-1]+arr[0][j];
		if(p<0||q<0)
	return 0;
	else if(p==0&&q==0){
		return arr[p][q];
	}
	else
	for(int i=1;i<p;i++){
	for(int j=1;j<q;j++){
	t[i][j]=max(t[i-1][j],t[i][j-1])+arr[i][j];
	}
	}
//	cout<<p<<q<<"\n";
	return t[p-1][q-1];
}

int main(){
	int t,c,n;
	scanf("%d",&t);
	while(t--){
		scanf("%d%d",&n,&c);
		
		for(int i=0;i<n;i++){
			for(int j=0;j<n;j++){
				scanf("%d",&arr[i][j]);
			}
		}
	//	cout<<n;
	int w;
	w=maxCoins(n,n);
//	cout<<"\n"<<w<<"\n";
	if(w>=c)
		 printf("%d\n",w-c );
		 else
		printf("-1\n");
	}
}
