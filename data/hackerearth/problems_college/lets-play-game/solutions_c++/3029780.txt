#include <bits/stdc++.h>
typedef long long          ll;
using namespace std;


int main(){
    ll test,n,k;
    cin>>test;
    ll time[100000], profit[100000];
    ll max,val;
    for(ll i=0;i<test;++i){
        cin>>n>>k;
        max=0,val;
        for(ll j=0;j<n;++j){
            cin>>time[j];
        }
        for(ll j=0;j<n;++j){
            cin>>profit[j];
        }
        for(ll j=0;j<n;++j){
            val=(k/time[j])*profit[j];
            if(val>max) max=val;
        }
        cout<<max<<endl;

    }
}
