#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
ll prime[1000001] = {0};
ll a[1000001] = {0};
void sieve()
{
	int i,j;
	for(i = 2; i <= 1000000; i++) {
		if(!prime[i]) {
			a[i] = 0;
			for(j = 2; j*i <= 1000000; j++) {
				prime[i*j] = 1;
				if(!a[i*j]) {
					a[i*j] = i;
				}
			}
		}
	}
	a[1] = 0;
	for(i = 2; i <= 1000000; i++) {
		a[i] = (a[i-1]+a[i])%mod;
	}
}
int main()
{
	ll t,l,r;
	cin>>t;
	sieve();
	while(t--) {
		cin>>l>>r;
		l = (a[r]-a[l-1]+mod)%mod;
		cout<<l<<endl;
	}
    return 0;
}
