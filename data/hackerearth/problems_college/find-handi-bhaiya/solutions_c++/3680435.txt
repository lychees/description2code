#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
   long long int t,i,a,b,c;
   cin>>t;
  // assert(t>=1&&t<=1000000);
   for(i=1;i<=t;i++)
   {
   	cin>>a>>b>>c;
  // 	assert(a>0&&b>0&&c>0);
   	if((a+b>c)&&(b+c>a)&&(c+a>b))
   	 {
   	 	cout<<"YES. ";
   	 	if(a==b&&b==c)
   	 	cout<<"It is a equilateral triangle."<<"\n";
   	 	else if((a==b)||(a==c)||(b==c))
   	 	cout<<"It is a isosceles triangle."<<"\n";
   	 	else 
   	 	cout<<"It is a scalene triangle."<<"\n";
   	 	
   	 }
   	 else
   	 cout<<"NO"<<"\n";
   }
    return 0;
}
