#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	map<string,int> m;
	int n;
	cin>>n;
	int i;
	string s;
	for(i=0;i<n;i++)
	{
		cin>>s;
		m[s]++;
	}
	map<string,int>::iterator it;
	for(it=m.begin();it!=m.end();it++)
	{
		cout<<it->first<<" "<<it->second<<endl;
	}
	return 0;
}
