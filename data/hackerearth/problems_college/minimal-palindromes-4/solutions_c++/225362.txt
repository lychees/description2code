#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;
int gcd(int a,int b)
{
int t;
while(a)
{
t = a;
a = b%a;
b = t;
}
return b;
}
int main()
{
    int t;
    scanf(" %d",&t);
    while(t--)
    {
        char s[10011];
        scanf(" %s",s);
        int len = strlen(s);
        int a[10011];
        int i;
        for(i=0;i<len;i++)
        {
            a[i] = s[i] - '0';
        }
        int ans = a[0];
        for(i=1;i<len;i++)
        {
            ans = gcd(ans,a[i]);
        }
        for(i=0;i<len;i++)
        {
            a[i] = a[i]/ans;
        }
        for(i=0;i<len;i++)
        {
            printf("%d",a[i]);
        }
        printf("\n");
        }
    return 0;
}
