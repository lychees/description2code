def calc(n):
	if n == 0:
		print 0
	elif n == 2:
		print 2
	elif n == 3:
		print 3
	else:
		x=sqrt(n)
		if n == int(x) * int(x):
			print  int(x)+(int(x)-1)
		else:
			ans = int(x)+int(x)
			if n<=(int(x)*int(x)+int(x)):
				print ans
			else:
				print ans+1
