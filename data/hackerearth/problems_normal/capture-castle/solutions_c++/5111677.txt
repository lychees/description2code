#include <bits/stdc++.h>
using namespace std;
#define scan(x) scanf("%d",&x)
#define print(x) printf("%d\n",x)
#define MOD 1000000007
#define scanl(x) scanf("%lld",&x)
#define printl(x) printf("%lld",x)
int arr[1001][1001];
int cost[1001][1001];
int xx[4]={0,0,1,-1};
int yy[4]={1,-1,0,0};

bool check(int x,int y,int n,int m)
{
	if(x>=0 && x<n && y>=0 && y<m) return true;
	return false;
}

void findmin(int n,int m,int x,int y)
{
	for(int k=0;k<4;k++)
	{
		int i=x+xx[k];
		int j=y+yy[k];
		if(!check(i,j,n,m)) continue;
		if(cost[i][j]>cost[x][y]+arr[i][j])
		{
			cost[i][j]=cost[x][y]+arr[i][j];
			findmin(n,m,i,j);
		}
	}
}


int main()
{
	int t;
	scan(t);
	while(t--)
	{
		int n,m;
		scan(n);
		scan(m);
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<m;j++)
			{
				cost[i][j]=1e9;
				scan(arr[i][j]);
			}
		}
		cost[0][0]=arr[0][0];
		int x,y,t;
		scan(x);
		scan(y);
		scan(t);
		x--;
		y--;
		findmin(n,m,0,0);
		int ans=cost[x][y];
		if(ans<t) 
		{
			printf("YES\n");
			print(t-ans);
		}
		else printf("NO\n");
	}
    return 0;
}
