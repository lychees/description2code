#include <iostream>
using namespace std;

int main()
{
   int T;
   cin>>T;
   while(T--){
       int N,K,count=0;
       cin>>N>>K;
       for(int i=0; i<N; i++){
           int a;
           cin>>a;
           if(a>0){
               count++;
           }
       }
       if(N-count<K){
           cout<<"YES"<<endl;
       }else{
           cout<<"NO"<<endl;
       }
   }
    return 0;
}
