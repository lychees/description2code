#include "cstdio"
#define gc getchar_unlocked
using namespace std;
void scanint(long long &x)
{
    register long long c = gc();
    x = 0;
    unsigned long long neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
long long n,x;
int main()
{
	scanint (n);
	while (n--) scanint (x);
	printf ("1\n");

	return 0;
}
