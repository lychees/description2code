#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,d,wt,cst;
    cin>>t;
    while(t--)
    {
    	cst=50;
    	cin>>d>>wt;
    	if(d<=2){
    	cout<<cst+wt*2<<endl;
    
    	}
    
    else if((d>2 && d<=60) &&(wt>=0 && wt<=250))
    	{
    		d=d-2;
    		d=12*d;
    		wt=wt*2;
    		cst=cst+d+wt;
    		cout<<cst<<endl;
    		
    		
    	}
    	else if(d>=61)
    	cout<<"NA"<<endl;
    }
    
    return 0;
}
