#include <bits/stdc++.h>
using namespace std;
typedef unsigned long long uint_64;
typedef vector<int> vec_int;

int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(false);
	int row, col;
	uint_64 i, j, temp, temp_sum, result; 
	vec_int row_sum, col_sum;
	cin>>row>>col;
	row_sum.resize(row,0);
	col_sum.resize(col,0);
	for (i=0; i<row; i++)
	{
		temp_sum=0;
		for(j=0; j<col; j++)
		{
			cin>>temp;
			temp_sum += temp;
			col_sum[j] += temp;
		}
		row_sum[i] = temp_sum;
	}
	result = 0;
	for(i=0; i<row; i++)
	{
		for (j = 0; j < col; j++)
		{
			temp = row_sum[i] ^ col_sum[j];
			if ( temp > result)
				result = temp;
		}
	}
	cout<<result;
	return 0;
}
