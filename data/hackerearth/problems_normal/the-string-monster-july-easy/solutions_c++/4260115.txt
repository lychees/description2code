int T,N;
string S[33];
string R;
int target[26];
set<vector<int> > Set;

int main()
{
#ifndef ONLINE_JUDGE
    freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
    debug = 1;
#endif
    srand (time(NULL));

    cin>>T;
    while(T--){
        cin>>N;
        for(int i=0;i<N;i++)
            cin>>S[i];
        cin>>R;

        vector<string> v1,v2;
        for(int i=0;i<N/2;i++)
            v1.pb(S[i]);
        for(int i=N/2;i<N;i++)
            v2.pb(S[i]);
        Set.clear();
        for(int i=0;i<26;i++)
            target[i] = 0;
        for(char c:R)
            target[c-'a']++;

        for(int mask=0;mask<(1<<v2.size());mask++){
            vector<int> t(26,0);
            for(int i=0;i<v2.size();i++){
                if(bit(mask,i)){
                    for(char c:v2[i])
                        t[c-'a']++;
                }
            }
            Set.insert(t);
        }
        bool fff = false;
        for(int mask=0;mask<(1<<v1.size());mask++){
            vector<int> t(26,0);
            for(int i=0;i<v1.size();i++){
                if(bit(mask,i)){
                    for(char c:v1[i])
                        t[c-'a']++;
                }
            }
            for(int i=0;i<26;i++)
                t[i] = target[i] - t[i];
            if(Set.find(t)!=Set.end())
                fff = true;
        }
        if(fff)cout<<"YES\n";
        else cout<<"NO\n";
    }

    return 0;
}
