#include <iostream>
using namespace std;
class count
{
long int n[100000],t,ans[100000],i;
public:
void getdata()
{
	cin>>t;
	for(i=0;i<t;i++)
	{
		cin>>n[i];
	}
}
void compute()
{
	for(i=0;i<t;i++)
	{
		ans[i]=8*n[i]+6*(n[i]-1);
		cout<<ans[i]<<endl;
	}
}
};
int main()
{
	count c;
	c.getdata();
	c.compute();
    return 0;
}
