#include <iostream>
using namespace std;

int main()
{
    int t, n;
    cin>>t;
    while(t--){
    	cin>>n;
    	if(n<=31) cout<<"1"<<endl;
    	else if(n<=59) cout<<"2"<<endl;
    	else if(n<=90) cout<<"3"<<endl;
    	else if(n<=120) cout<<"4"<<endl;
    	else if(n<=151) cout<<"5"<<endl;
    	else if(n<=181) cout<<"6"<<endl;
    	else if(n<=212) cout<<"7"<<endl;
    	else if(n<=243) cout<<"8"<<endl;
    	else if(n<=273) cout<<"9"<<endl;
    	else if(n<=304) cout<<"10"<<endl;
    	else if(n<=334) cout<<"11"<<endl;
    	else if(n<=365) cout<<"12"<<endl;
    }
    return 0;
}
