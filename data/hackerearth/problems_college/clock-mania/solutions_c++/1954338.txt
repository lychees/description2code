#include <iostream>
using namespace std;
template< class T > T _abs(T n) { return (n < 0 ? -n : n); }
template< class T > T _min(T a, T b) { return (a < b ? a : b); }
int main()
{
   // cout << "Hello World!" << endl;
   int h,m,t,d;
   cin>>t;
   while(t--)
   {
   	cin>>h>>m;
   	if(h==12)h=0;
   	d=_abs(6*m-((60*h+m)/2));
   	
   	cout<<_min(d,360-d)<<endl;
   }
   return 0;
}
