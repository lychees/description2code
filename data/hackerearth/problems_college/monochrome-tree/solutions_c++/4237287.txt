using namespace std;
vll v[100005];
ll a[100005],b[100005];
int dp[100005][5];
ll dfs(ll i , ll status ,ll p){
    if(dp[i][status] != -1)
        return dp[i][status];
    ll t1 = 0;
    ll t2 = 0;
    ll j,flag = 0;
    for(j=0;j<v[i].size();j++){
        if(v[i][j] != p){
            flag = 1;
            t1 += dfs(v[i][j],status,i);
            t2 += dfs(v[i][j],-1*status,i);
        }
    }
    ll ans;
    if(flag == 0){
        if(status == 1){
            if(a[i] == b[i]){
                dp[i][status] = 0;
                return 0;
            }
            else{
                dp[i][status] = 1;
                return 1;
            }
        }
        else{
            if(a[i] != b[i]){
                dp[i][status] = 0;
                return 0;
            }
            else{
                dp[i][status] = 1;
                return 1;
            }
        }
    }
    if(status == 1){
        if(a[i] == b[i]){
            ans = min(t1,2+t2);
            dp[i][status] = ans;
            return ans;
        }
        else{
            ans = min(1+t1,1+t2);
            dp[i][status] = ans;
            return ans;
        }
    }
    else{
        if(a[i] !=   b[i]){
            ans = min(t1,2+t2);
            dp[i][status] = ans;
            return ans;
        }
        else{
            ans = min(1+t1,1+t2);
            dp[i][status] = ans;
            return ans;
        }
    }
}
int main(){
    ll n,i,x,y;
    s(n);
    memset(dp,-1,sizeof(dp));
    f(i,0,n-1){
        s(x),s(y);
        v[x].pb(y);
        v[y].pb(x);
    }
    f(i,1,n+1)
        s(a[i]);
    f(i,1,n+1)
        s(b[i]);
    pl(dfs(1,1,1));

    return 0;
}
