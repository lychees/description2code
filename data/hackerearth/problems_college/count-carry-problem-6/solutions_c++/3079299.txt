#include<iostream>
using namespace std;
int main()
{
int t,cnt;
cin>>t;
while(t--)
{
long int a,b;
cin>>a>>b;
int c[2]={0},sum=0;
cnt=0;
while(a!=0||b!=0)
{
sum=a%10+b%10+c[1];
a/=10;
b/=10;
sum/=10;
c[1]=sum%10;

if(c[1]>0)
cnt++;
}

if(cnt==0)
cout<<"No carry operation"<<endl;
else if(cnt==1)
cout<<"1 carry operation"<<endl;
else
cout<<cnt<<" carry operations"<<endl;
}
return 0;
}
