t = int(raw_input())
while t > 0:
    s = raw_input()
    s1 = list(s)
    i = 1
    while i < len(s1):
        ch = s1[i]
        a = ord(ch)
        a = a+i
        if a > 122:
            a = a-26

        ch = chr(a)
        s1[i] = ch
        i = i+1
    s = ''.join(s1)
    print(s)
    t = t-1
