#include<cstdio>
#define ll long long int
#define M 1000000007
 
using namespace std;
 
ll pw(ll a, ll b){
  ll r;
  if(b==0) return 1;
  r = pw(a,b/2);
  r = (r*r)%M;
  if(b%2) r = (r*a)%M;
  return r;
}
 
int main()
{
    ll n,ans,i,t,k,den;
    scanf("%lld",&t);
    while(t--)
    {
    ans=1,den=1;
    scanf("%lld%lld",&n,&k);
    ll v,sum=0;
    for(i=0;i<n;i++)
    {
    	scanf("%lld",&v);
    	sum=sum+v;
    }
    v=n-1;
    if(k>=sum)
    {
    sum=k-sum+v;
      for(i=1;i<=v;i++)
        {
        ans=(ans*sum)%M;
        sum--;
        }
        for(i=1;i<=v;i++)
        den=(den*i)%M;
        ans=(ans*pw(den,M-2))%M;
        printf("%lld\n",ans);
    }
    else
    printf("0\n");
   } 
    return 0;
}
