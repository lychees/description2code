#include <iostream>
using namespace std;

int main()
{
    int tc,n,i,temp,j,count;
    cin>>tc;
    
    while(tc--)
    {
    	count = 0;
    	cin>> n;
        int b[n],g[n];
    	
    	for(i=0;i<n;i++)
    		cin>>g[i];
    		
        for(i=0;i<n;i++)
            cin>>b[i];
            
        for(i=0;i<n;i++)
        {
          for(j=i+1;j<n;j++)
          {
          	 if(b[j] > b[i])
          	 {
          	 	temp = b[j];
          	 	b[j] = b[i];
          	 	b[i] = temp;
          	 }
          	 
          	 if(g[i] > g[j])
          	 {
          	 	temp = g[i];
          	 	g[i] = g[j];
          	 	g[j] = temp;
          	 }
          }
        }
        
        for(i=0;i<n;i++)
        {
        	if(b[i]%g[i] == 0 || g[i]%b[i] == 0)
        	count++;
        }
    	
    	cout<<count<<endl;
   }
    return 0;
}
