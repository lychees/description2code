using namespace std;


long long gcd(long long a, long long b){
	if(b==0) return a;
	return gcd(b, a%b);
}

void solve(){
	long long p,q;
	cin >> p >> q;
	cout << (gcd(p,q)==1?"Is a Co-Prime":"Not a Co-Prime") << endl;
}


int main(){
	int T;
	cin >> T;
	for(int t=0; t<T; t++){
		solve();
	}
	return 0;
}
