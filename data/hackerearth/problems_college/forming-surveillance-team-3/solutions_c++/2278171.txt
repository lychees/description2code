#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    bool ar[100001];
    while(t--)
    {
        int n;
        cin>>n;
        for(int i=1;i<100001;i++)
            ar[i]=false;
            int cnt=0;
        for(int i=0;i<n;i++)
        {
            int a;
            cin>>a;
            if(ar[a]==true)
                cnt++;
            else
                ar[a]=true;
        }
        cout<<n-cnt<<endl;

    }
    return 0;
}
