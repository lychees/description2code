#include<bits/stdc++.h>
using namespace std ;
#define MOD 1000000000
set < int > st ;
map < int , int > mp ;
vector < int > v[501] ;
int main () {
    int n , m , temp ;
    scanf ( "%d" , &n ) ;
    int i = 0 ;
    int x = n ;
    while ( x-- ) {
        i++ ;
        scanf ( "%d" , &m ) ;
        for ( int i = 0 ; i < m ; i++ ) {
            scanf ( "%d" , &temp ) ;
            st.insert ( temp ) ;
            if ( mp[temp] == 0 ) {
                mp[temp] = 1 ;
                v[temp].push_back ( i ) ;
            }
        }
        mp.clear() ;
    }
    set < int > :: iterator it ;
    int sum = 0 ;
    for ( it = st.begin() ; it != st.end() ; it++ ) {
        sum += (*it) ;
    }
    int sum_diff = 0 ;
    for ( int i = 0 ; i <= 500 ; i++ ) {
        if ( v[i].size() != 0 && v[i].size() != n )
            sum_diff += i ;
    }
    int ans = sum_diff - sum ;
    printf ( "%d\n" , ans ) ;
}
