#include <iostream>
#define ll long long int
using namespace std;

int main()
{
    ll t,n,sum,rest,i,mid;
    cin>>t;
    while(t--)
    {
    	sum=0;
    	cin>>n;
    	ll arr[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>arr[i];
    		sum+=arr[i];
    	}
    	cin>>rest;
    	rest%=sum;
    	if(rest==0)
    	{
    		for(i=n-1;i>=0;i--)
    		{
    			if(arr[i]!=0)
    			{	
    				mid=i;
    				break;
    			}
    		}
    	}
    	else
    	{
    		for(i=0;i<n;i++)
    		{
    			rest-=arr[i];
    			if(rest<=0)
    			{
    				mid=i;
    				break;
    			}
    		}
    	}
    	cout<<mid+1<<endl;
    }
    return 0;
}
