#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int tc;long long n;
    cin>>tc;
    while(tc--)
    {
    	cin>>n;
    	long long s = sqrt(n);
    	long long ans = s*s;
    	if(ans == n)
    	{
    		cout<<"YES"<<endl;
    	}
    	else{
    	long long a=s;
    	char c;
    	int mini=INT_MAX;
    	long long b=a+1;
    	a=a*a;
    	b=b*b;
    	if(a>n) {
    		mini = a-n;
    		c='+';
    		
    	}
    	 if(a<n)
    	{
    		if(mini>n-a)
    		{
    			mini = n-a;
    			c = '-';
    		}
    	}
    	if(b>n) {
    		if(mini>b-n){
    		mini = b-n;
    		c='+';}
    		
    	}
    	if(b<n)
    	{
    		if(mini>n-b)
    		{
    			mini = n-b;
    			c = '-';
    		}
    	}
    	cout<<c<<mini<<endl;
    	}
    }
    return 0;
}
