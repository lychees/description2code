#include <bits/stdc++.h>

using namespace std;

int main(){

    

    long T;
    cin>>T;

    assert(T>=1 && T<=25);
    while(T--){

        float P, E;
        cin>>P>>E;

        assert(P>=10 && P<=10000);
        assert(E>= -70 && E<=70);

        float errored_price = (P + (P * E * 0.01));
        float actual_price = 0 ;
        long temp = errored_price ;
        while(temp > 0){
            actual_price = (actual_price * 10) + (temp % 10);
            temp /= 10;
        }

        float profit_lose = (float)(((P - actual_price) / (float)P) * 100);
        float new_price = (float)(errored_price + (errored_price * profit_lose * 0.01));

       // cout<<"Errored : "<<errored_price<<" , actual :"<<actual_price<<" , profit :"<<profit_lose<<endl;

        cout<<(long)(new_price)<<endl;
    }

    return 0;
}
