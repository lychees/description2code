#include <iostream>
using namespace std;

int main()
{
   int n;
   cin>>n;
   int a[n];
   int i;
   for(i=0;i<n;i++)
   {
   	cin>>a[i];
   }
   int sum=0;
   int max_sum=0;
   for(i=0;i<n;i++)
   {
   	sum=sum+a[i];
   	if(sum<0)
   	{
   		sum=0;
   	}
   	if(sum>max_sum)
   	{
   		max_sum=sum;
   	}
   }
 
   cout<<max_sum;
   return 0;
}
