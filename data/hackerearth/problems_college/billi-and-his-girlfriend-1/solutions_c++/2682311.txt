#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll power(ll x, ll y,ll k)
{
    ll temp;
    if( y == 0)
        return 1;
    else
    {
    	temp=power(x,y/2,k);
    	temp=temp*temp;
    	temp=temp%k;
    	if(y%2)
    	{
    		temp=temp*x;
    		temp=temp%k;
    	}
    	return temp;
    }
}
int main()
{
    ll i,j,k,a,b,maxx,minn,test,num,total;
    cin>>test;
    while(test--)
    {
        cin>>a>>b>>k;
        total=0;
        i=k;
        j=pow(10,k);
        num=power(a,b,j);
        while(num)
        {
            total+=(num%10);
            num=num/10;
        }
        cout<<total<<endl;
    }
}
