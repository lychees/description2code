#define SORTF(x) sort(x.begin(),x.end(),comp);
#define SORT(x) sort(x.begin(),x.end())

#define S(n) scanf("%lld",&n)

struct node {
	double a;
	ll b;
};

ll comp(const node &a, const node &b) {
	if(a.a < b.a) {
		return 1;
	}
	else if(a.a == b.a) {
		if(a.b < b.b) {
			return 1;
		}
	}
	return 0;
}

int main() {
	ll n,m;
	S(n);
	S(m);
	if( m == 0 || n == 0) {
		return 0;
	}
	ll i,j;
	vector<node> v;
	for(i = 1;i <= m;i++) {
		node te;
		te.a = abs((1.0*((m + 1) / 2.00) - i));
		te.b = i;
		v.pb(te);
	}
	SORTF(v);
//	cout << "\n";
	for(i = 0;i < n;i++) {
		cout << v[i % m].b << "\n";
	}
}
