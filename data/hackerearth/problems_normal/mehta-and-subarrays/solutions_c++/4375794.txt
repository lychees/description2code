int main()
{
    //optimize
    ll i,j,k,n;
    cin>>n;
    ll a[n];
    rep(i,0,n) cin>>a[i];
    ll minseen=INT_MAX,len=0,cnt=0;
    pll dp[n+1];
    dp[0]={0,-1};
    rep(i,0,n)
    {
        dp[i+1].ff=dp[i].ff+a[i];
        dp[i+1].ss=i;
    }
    sort(dp,dp+n+1);
    rep(i,0,n+1)
    {
        ll t=dp[i].ss-minseen;
        if(t>len)
        {
            len=t;
            cnt=1;
        }
        else if(t==len)
            cnt++;
        else
        {
            minseen=min(minseen, dp[i].ss);
        }
    }
    if(len==0) cout<<"-1";
    else cout<<len<<" "<<cnt;
    return 0;
}
