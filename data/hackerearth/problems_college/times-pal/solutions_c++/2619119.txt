bool isPal(string &temp) {
	for(int i=0;i<temp.size()/2;i++) {
		if(temp[i]!=temp[temp.size()-1-i])
			return false;
	}
	return true;
}
int main() {
	int t,i,c;
	sc(t);
	while(t--) {
		c=0;
		string S1,S2;
		cin>>S1>>S2;
		for(i=0;i<S1.size();i++) {
			string temp=S1;
			temp.insert(i,S2);
			//cout<<temp<<endl;
			if(isPal(temp))
				c++;
		}
		string temp=S1;
		temp.insert(i,S2);
		if(isPal(temp))
			c++;
		printf("%d\n",c);
	}
	return 0;
	
}
