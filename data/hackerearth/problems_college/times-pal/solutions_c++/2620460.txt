#include<iostream>
#include<vector>
#include<string>
using namespace std;




bool IsPalindrome(const string& str)
{
    if (str.empty())
        return false;

    int i = 0;                // first characters
    int j = str.length() - 1; // last character

    while (i < j)
    {
        if (str[i] != str[j])
        {
            return false;
        }
        i++;
        j--;
    }
    return true;
}


int main()
{
    int T;
    cin>>T;
    while(T--)
    {
            string a,b;
            cin>>a>>b;
            int j=0;
            int count =0;
            for(int i=0;i<=a.size();++i)
            {
                string k=a;
                k.insert(j,b);
                if (IsPalindrome(k))
                    count ++;
                j++;
            }
            cout<<count<<endl;

    }
    return 0;
}
