#include<bits/stdc++.h>
using namespace std;
long long int gcd(long long int a,long long int b)
{
	if(b==0)
	return a;
	else
	return (gcd(b,a%b));
}

long long int phi (long long int n) {
	long long int result = n;
	long long int i;
	for (i=2; i*i<=n; ++i)
		if (n % i == 0) {
			while (n % i == 0)
				n /= i;
			result -= result / i;
		}
	if (n > 1)
		result -= result / n;
	return result;
}
