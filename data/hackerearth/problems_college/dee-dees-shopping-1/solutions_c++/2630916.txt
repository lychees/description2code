#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while (t --) {
		int n;
		scanf("%d",&n);
		int f = n / 8;
		int r = n % 8;
		while (r % 6 != 0 && f >= 0) {
			r = n - (--f) * 8;
		}
		if (r % 6 == 0 && f >= 0) {
			printf("%d\n",r/6 + f);
		} else {
			printf("-1\n");
		}
	}
	
	return 0;
}
