#include <iostream>
#include <cstring>
using namespace std;
int moves[][2] = {0,1, 1,0, 1,1};
int MOD = 1000000007;
int A, B, C, D;
int memo[1010][1010];
bool isValid(int x, int y) {
	if (x < 0 || y < 0 || y >= 1004 || x >= 1004) return false;
	return true;
}
int solve() {
	memset(memo, 0, sizeof(memo));
	memo[0][0] = 1;
	for (int i = 0; i < 1004; i++) {
		for (int j = 0; j < 1004; j++) {
			for (int k = 0; k < 3; k++) {
				int dx = i + moves[k][0], dy = j + moves[k][1];
				if (isValid(dx, dy)) {
					memo[dx][dy] += memo[i][j] %MOD;
					memo[dx][dy] %= MOD;
				}
			}
		}
	}
}
int main()
{
	memset(memo, -1, sizeof(memo));
	solve();
	int T;
	for (cin >> T; T; T--) {
		cin >> A >> B >> C >> D;
		cout << memo[C-A][D-B] <<"\n";
	}
}
