#include<bits/stdc++.h>
#define ll long long
#define N 1000000007
using namespace std;
int main()
{
    ll  t,n;
    ll a[1000001];
        a[0]=0,a[1]=1,a[2]=2;
        for(ll i=3;i<=1000000;i++)
        {
            a[i]=a[i-1]+(a[i-2]*(i-1));
            if(a[i]>=N)a[i]%=N;
        }
        cin>>t;
    while(t--)
    {
        cin>>n;
        cout<<a[n]<<endl;
    }
}
