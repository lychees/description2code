#include<iostream>
#include<stdio.h>
using namespace std;

unsigned long long int catalon(long long int n)
{
	unsigned long long int arr[n+1];
	arr[0]=arr[1]=1;
	for(int i=2;i<=n;i++)
	{
		
		arr[i]=0;
		for(int j=0;j<i;j++)
			arr[i]+=arr[j]*arr[i-j-1];
	}
	return arr[n];
}
int main()
{
	int t;
	scanf("%d",&t);
	long long int n;
	while(t--)
	{
		scanf("%lld",&n);
		unsigned long long int var=catalon(n);
		printf("%u\n",var);
	}
	return 0;
}
