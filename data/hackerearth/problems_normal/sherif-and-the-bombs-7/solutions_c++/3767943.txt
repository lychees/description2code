#include<bits/stdc++.h>
using namespace std;
int res(int a,int b, int x, int y,int r)
{
    if((a-x)*(a-x)+(b-y)*(b-y)<=r*r)
        return 1;
    else
        return 0;
}
int main()
{
    int n;
    cin >> n;
    int a[n+1][3];
    for(int i=0;i<n;i++)
    {
        cin >> a[i][0] >> a[i][1] >> a[i][2];
    }
    int ans=0;
    for(int x=1;x<=1000;x++)
    {
        for(int y=1;y<=1000;y++)
        {
            int temp=0;
            for(int i=0;i<n;i++)
            {
                temp=temp+res(x,y,a[i][0],a[i][1],a[i][2]);
                if(temp>=2)
                {
                    ans++;
                    break;
                }
            }
        }
    }
    cout << ans << endl;
    return 0;
}
