#include <iostream>
#include<string>
#include<algorithm>
using namespace std;

int main()
{
	int n,count=0,len,temp;
	char start,end;
	string name,string[100];
	int a[128][128]={0};
	
	int k=0,i;
	
	cin >> n;

	while(n--)
	{
		cin >> name;
		string[k] = name;
		len = name.length();
		start = name[0];
		end = name[len-1];
		temp = 1;
		
		if(!a[start][end])
		{
			a[start][end] = 1;
	
		}
		else
		{
			for(i=1; i <= k; i++)
			{
			
				sort(string[i-1].begin(),string[i-1].end());
				sort(name.begin(),name.end());
				if(string[i-1] == name)
				{
					temp=0;
					break;
				}
				
			}
			
		}
		if(temp)
		{
			count++;
		}
		k++;
		
	}
	cout << count;
    return 0;
}
