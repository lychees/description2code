#include <bits/stdc++.h>
using namespace std;
#define MAX 1000002
bool isprime[MAX];
int count_prime[MAX];
void seive()
{
	isprime[1]=true;
	for(int i=2;i*i<MAX;i++)
	{
		if(!isprime[i])
		{
			for(int j=i*i;j<MAX;j+=i)
			{
				isprime[j]=true;
			}
		}
	}
	for(int i=1;i<MAX;i++)
	{
		count_prime[i]=count_prime[i-1];
		if(!isprime[i])
			count_prime[i]++;
	}
}	
int main()
{
	seive();
    int t; cin >> t;
    while(t--)
    {
    	int l,r; cin >> l>> r;
    	float ans = (count_prime[r]-count_prime[l-1])*1.0/(r-l+1);
    	cout << fixed << setprecision(6) << ans << endl;
    }
    return 0;
}
