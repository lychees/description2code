#include<stdio.h>
#include<bits/stdc++.h>
#define ll long long
#define INF 8000000001
long int end, a[1000002];
void bS(long int x)
{
    long int l=1, r=end;
    while(l<=r)
    {
        long int mid = (l + r)/2;
        if(a[mid] <= x) l = mid+1;
        else r = mid-1;
    }
    a[++r] = x;
    if(r>end) end = r;
}
int main() 
{
  long int i, v, N,t;
  scanf("%ld",&t);
  while(t--)
  {
  	end=0;
    scanf("%ld",&N);
    for(i=0; i<N; i++) 
	  a[i] = INF;
    for(i=1; i<=N; i++)
    {
        scanf("%ld",&v);
		v-=i;
        if(v>=0)
		{
			//printf("%ld\n",v);
			 bS(v);        
		}
    
	  }
  printf("%ld\n",N-end);
  }
return 0;
}
