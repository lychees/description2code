#include<iostream>
#include<string.h>
#include<stdio.h>
using namespace std;
int main()
{
	char str[1000];
	char c;
	int i;
	int count[300];
	scanf(" %[^\n]s",str);//\n is the delimiter
	for(i=0;i<300;i++)
	{
		count[i]=0;
	}
	for(i=0;i<strlen(str);i++)
	{
		count[(int)str[i]]++;
	//	cout<<count[(int)str[i]]<<"A"<<endl;
	}
	int max=0;
	for(i=0;i<280;i++)
	{
		if(count[i]>max)
		{
			c=char(i);
			max=count[i];
			//cout<<max<<" "<<(int)str[i]<<endl;
		}
		//if(count[(int)str[i]]!=0)
	//	cout<<"Boom "<<(int)str[i]<<endl;
	}
	cout<<c<<" "<<max<<endl;
}
