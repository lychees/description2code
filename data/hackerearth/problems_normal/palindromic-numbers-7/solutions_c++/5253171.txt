#include<iostream>
using namespace std;
int main(){
    unsigned long int t,sum = 0;
    cin>>t;
    while (t--) {
        int count = 0;
        unsigned long int a,b;
        cin>>a>>b;
        for(unsigned long int i=a;i<=b;i++){
            sum=0;
            unsigned long int temp;
            temp = i;
            while (temp>0) {
                int r;
                r = temp%10;
                temp = temp/10;
                sum = sum*10 + r;
            }
            if(sum == i){
                count++;
            }
        }
        cout<<count<<endl;
    }
}
