#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int arr[20000002];

inline void scanint(int *x)

{

	register char c = getchar_unlocked();

	*x = 0;

	for(; (c<48)||(c>57);c = getchar_unlocked());

	for(; (c>47)&&(c<58);c = getchar_unlocked())

		*x = (int)((((*x)<<1) + ((*x)<<3)) + c - 48);

}


inline void printint(int n)

{

	if(n == 0)

	{

		putchar_unlocked('0');

		putchar_unlocked(' ');

	}

	else if(n == -1)

	{

		putchar_unlocked('-');

		putchar_unlocked('1');

		putchar_unlocked(' ');

	}

	else

	{

		char buf[11];

		buf[10] = ' ';

		int i = 9;

		while(n)

		{

			buf[i--] = n % 10 + '0';

			n /= 10;

		}

		while(buf[i] != ' ')

			putchar_unlocked(buf[++i]);

	}

}


int main()
{
    long n;
    while(1){
    	scanf("%ld",&n);
    	if(!n)
    		break;
    	for(long i=0;i<n;i++)
    		scanint(&arr[i]);
    	sort(arr,arr+n);
    	for(long i=0;i<n;i++)
    		printint(arr[i]);
    	printf("\n");
    }
    return 0;
}
