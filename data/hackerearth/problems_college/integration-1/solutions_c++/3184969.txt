#include <iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	int x=1;
	while(t--)
	{
		long long int n,p,u,r,s;
		cin>>n>>p>>u>>r>>s;
		long long int ans=p;
		while(n--)
		{
			if(ans+r<=u)ans=ans+r;
			else if(ans-s>=0) ans=ans-s;
		}
		printf("Case #%d: %lld\n",x,ans);
		x++;
	}
    return 0;
}
