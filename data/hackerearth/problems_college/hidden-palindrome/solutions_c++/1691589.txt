#include<bits/stdc++.h>
using namespace std;
int longestPalSubstr(char *str)
{
    int maxLength = 1;  
 
    int start = 0;
    int len = strlen(str);
 
    int low, high;
 
    
    for (int i = 1; i < len; ++i)
    {
       
        low = i - 1;
        high = i;
        while (low >= 0 && high < len && str[low] == str[high])
        {
            if (high - low + 1 > maxLength)
            {
                start = low;
                maxLength = high - low + 1;
            }
            --low;
            ++high;
        }
 
        
        low = i - 1;
        high = i + 1;
        while (low >= 0 && high < len && str[low] == str[high])
        {
            if (high - low + 1 > maxLength)
            {
                start = low;
                maxLength = high - low + 1;
            }
            --low;
            ++high;
        }
    }
 
    return maxLength;
}
 

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char str[21];
    	cin>>str;
    	int p=longestPalSubstr(str);
    	printf("%d\n",p);
    }
    
    return 0;
}
