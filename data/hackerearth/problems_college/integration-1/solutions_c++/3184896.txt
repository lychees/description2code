#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
	int t;
	cin>>t;
	int no=1;
	while(t--){
		ll n,p,u,r,s;
		scanf("%lld%lld%lld%lld%lld",&n,&p,&u,&r,&s);
		while(n--){
			if(p+r<=u){
				p+=r;
			}else if(p-s>=0){
				p-=s;
			}
		}
		cout<<"Case #"<<no<<": ";
		cout<<p<<endl;
		no++;
	}
	return 0;
}
