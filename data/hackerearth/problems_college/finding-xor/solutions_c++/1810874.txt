#include <iostream>
#include <string>
using namespace std;

char _xor(char a, char b)
{
	if((a == '0') && (b == '0'))
		return '0';
	else if((a == '0') && (b == '1'))
		return '1';
	else if((a == '1') && (b == '0'))
		return '1';
	else
		return '0';
}
int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		string num1, num2;
		cin>>num1>>num2;
		for(int i = 0; i < num1.size(); i++)
		{
			cout<<_xor(num1[i], num2[i]);
		}
		cout<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
