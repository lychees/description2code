#include <iostream>

using namespace std;

int main()
{
    int tc;
    cin>>tc;
    string s[tc];
    bool vovel(char);
    for(int i=0;i<tc;i++)
    {
        cin>>s[i];
    }
    int con,vov,len;
    bool b,f;
    for(int i=0;i<tc;i++)
    {
        con=vov=0;
        len=s[i].length();
        for(int j=0;j<len;j++)
        {
            b=vovel(s[i][j]);
            if(b)
            {
                vov+=1;
            }
            else
            {
                con+=1;
            }
        }
        f=false;
        for(int j=0;j<len-2;j++)
        {
            if(!vovel(s[i][j])&&!vovel(s[i][j+1])&&!vovel(s[i][j+2]))
                f=true;
        }
        if(con>vov||f)
            cout<<"hard\n";
        else
            cout<<"easy\n";
    }
    return 0;
}
bool vovel(char c)
{
    if(c=='a'||c=='e'||c=='i'||c=='o'||c=='u'||c=='y')
        return true;
    else
        return false;
}
