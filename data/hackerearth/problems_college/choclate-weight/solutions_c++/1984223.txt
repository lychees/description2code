#define sd(x) scanf("%d", &x)
#define yes puts("I Like You")
#define no puts("I Hate You")

using namespace std;

int main()
{
#ifdef _MSC_VER
	freopen("input.txt", "r", stdin);
	//	freopen("output.txt", "w", stdout);
#endif
	int n;
	cin >> n;
	int m = n*(n - 1) / 2 + n;

	int arr[m];
	for (int i = 0; i < n; i++) {
		sd(arr[i]);
	}

	int idx = n;
	for (int i = 0; i < n; i++) {
		for (int j = i + 1; j < n; j++) {
			arr[idx++] = arr[i] + arr[j];
		}
	}
	sort(arr, arr + idx);

	int q;
	cin >> q;

	while (q--) {
		int p;
		sd(p);

		if (binary_search(arr, arr + idx, p)) {
			yes;
		}
		else {
			no;
		}
	}


	return 0;
}
