#include<iostream>
#include<string>
using namespace std;
int main()
{
	int test;
	cin>>test;
	string s;
	while(test--)
	{
		cin>>s;
		int n=s.length();
		int l,u;
		l=u=0;
		for(int i=0;i<n/2;i++)
		{
			l+=s[i]-'0';
			u+=s[n-i-1]-'0';
		}
		if(l==u)
		cout<<"YES\n";
		else
		cout<<"NO\n";
	}
}
