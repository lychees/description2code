#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
  int t;
  cin>>t;
  
  while(t--)
  {
  	
   int n;
   cin>>n;
   int a[n];
   
   for(int i=0;i<n;i++)
   	scanf("%d",&a[i]);
   	
   	
   int maxi=a[0];
   int curr=a[0];
   for(int i=1;i<n;i++)
    {
    	curr=max(a[i],curr+a[i]);
    	maxi=max(curr,maxi);	
    	
    }
   	
  	printf("%d\n",maxi);
  	
  	
  	
  }
}
