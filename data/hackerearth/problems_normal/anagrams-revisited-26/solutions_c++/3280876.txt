#include <iostream>
#include <string>
#include <map>
#include <utility>
#include <vector>

using namespace std;

vector<int> strv(string & s) {
	vector<int> v(26, 0);
	for(size_t i = 0; i < s.size(); i++)
		v[s[i] - 'a']++;
	return v;
}

int main(void) {
	int n;
	cin >> n;
	typedef map<vector<int>, int> mvi;
	mvi m;
	while(n--) {
		string s;
		cin >> s;
		vector<int> v = strv(s);
		if(m.find(v) == m.end())
			m.insert(make_pair(v, 1));
		else
			m[v]++;
	}
	
	int mx = 0;
	for(mvi::iterator it = m.begin(); it != m.end(); it++)
		mx = max(mx, (*it).second);
	cout << mx << endl;
}
