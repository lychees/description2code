#include<iostream>
using namespace std;



int calc(int n)
{
int cnt=0;

for(int i=1;i<=n;i++)
if(n%i==0)
cnt++;

return cnt;

}



int main()
{
int i,j;
int t;
cin>>t;
while(t--)
{
int a,n;
cin>>n;

for(i=2;i<10000;i++)
{
a=i*(i+1)/2;
int count=calc(a);

if(count>n)
break;

}

cout<<a<<"\n";

}



}
