using namespace std;
bool isvowel(char x){
    return (x=='a'||x=='u'||x=='e'||x=='o'||x=='i');
}
int L[5001][5001];
int LCSLength(string X,string Y)
{
    int m = X.length();
    int n = Y.length();
    
    for(int i=0; i<=m; i++)
    {
        for(int j=0; j<=n; j++)
        {
            if(i==0 || j==0)
                L[i][j] = 0;
            else if(tolower(X[i-1])==tolower(Y[j-1]) && !isvowel(tolower(X[i-1])))
                L[i][j] = L[i-1][j-1]+1;
            else
                L[i][j] = max(L[i-1][j],L[i][j-1]);
        }
    }
    return L[m][n];
}
int main(void){
    string x,y;
    cin>>x>>y;
    cout<<LCSLength(x,y)<<endl;
}
