#include <iostream>
#include <cmath>
#include <string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    cin.ignore();
    while(t>0)
    {
    	int len,nlen,j=0,i=0,ss;
    	string a,newa;
    	getline(cin,a);
    	
    	len=a.length();
    	ss=len;
    	
    	nlen=round(sqrt(len));
    	
    		while(len>0)
    		{
    			while(i<ss)
    			{
    				newa+=a[i];
    				i=i+nlen;
    			
    				len--;
    			}
    			
    			j++;
    			i=j;
    		    
    		}
    		
    
    		cout<<newa;
    
    	cout<<"\n";
    	
    	
    	t--;
    	
    }
    return 0;
}
