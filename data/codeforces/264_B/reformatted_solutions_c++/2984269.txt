#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <map>
#include <queue>
#include <set>
#include <vector>
using namespace std;
#define N 100010

int d[N], b[N];

int main() {
  int i, j, n, m, a = 0;
  scanf("%d", &n);
  for (i = 1; i <= n; i++) {
    scanf("%d", &m);
    for (j = 1; j * j <= m; j++)
      if (m % j == 0) {
        if (j != 1)
          d[i] = max(d[i], d[b[j]]);
        d[i] = max(d[i], d[b[m / j]]);
        b[j] = b[m / j] = i;
      }
    d[i] += 1;
    a = max(a, d[i]);
  }
  printf("%d\n", a);
  return 0;
}
