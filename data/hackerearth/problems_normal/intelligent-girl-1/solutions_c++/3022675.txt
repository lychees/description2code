#include<iostream>
#include<string>
using namespace std;
int main()
{
        string str;
        int i;
        int len;
        cin>>str;
        len=str.length();
        len--;
        int arr[10000]={0};
        if((str[len]-48)%2==0)
                arr[0]++;
        for(i=len-1;i>=0;i--)
        {
                if((str[i]-48)%2==0)
                        arr[len-i]++;
                arr[len-i]+=arr[len-i-1];
        }
        for(i=len;i>=0;i--)
                cout<<arr[i]<<" ";
        cout<<endl;
        return 0;
}
