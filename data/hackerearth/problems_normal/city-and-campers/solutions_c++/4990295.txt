int n,q,freq[N],dsu[N],asize[N],maxi=1,mini=1,flag=0,ans;

int root(int i)
{
       while(i!=dsu[i])
       {
              dsu[i] = dsu[dsu[i]];
              i = dsu[i];
       }
       return i;
}

void join(int a,int b)
{
       a=root(a);
       b=root(b);

       if(a==b || flag)return;

       freq[asize[a]]--;
       freq[asize[b]]--;

       if(asize[a]<asize[b])
       {
              dsu[a]=b;
              asize[b]+=asize[a];
              freq[asize[b]]++;
              if(asize[b]>maxi)
                     maxi=asize[b];
              if(asize[b]==n)
              {
                     flag=1;
                     maxi=mini=0;
              }
       }
       else
       {
              dsu[b]=a;
              asize[a]+=asize[b];
              freq[asize[a]]++;
              if(asize[a]>maxi)
                     maxi=asize[a];
              if(asize[a]==n)
              {
                     flag=1;
                     maxi=mini=0;
              }
       }

       if( freq[mini]==0 )
       {
              while(freq[mini]!=0)mini++;
       }
}

void init()
{
  for(int i=1;i<=n;i++)
  {
         dsu[i]=i;
         asize[i]=1;
  }

  freq[1]=n;

}
int main(void)
{
  cin>>n>>q;
  init();
  while(q--)
  {
         int i,j;
         scan(i);scan(j);
         join(i,j);
         printf("%d\n",maxi-mini);
  }
  return 0;
}
