#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    char str[10001];
    cin>>str;
    int n=strlen(str);
    int even[n];
    int count=0;
    for(int i=n-1;i>=0;i--)
    {
    	if((str[i]-'0')%2==0)
    	{
    		count++;
    	}
    	even[i]=count;
    }
    for(int i=0;i<n;i++)
    {
    	cout<<even[i]<<" ";
    }
    return 0;
}
