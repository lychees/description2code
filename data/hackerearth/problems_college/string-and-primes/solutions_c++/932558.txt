#define MAX 3000

int p[MAX];

void seiv(){
	MS1(p);
	long int i, j;
	p[0] = p[1] = 0;
	int lim = sqrt(MAX);
	for (i = 2; i <= lim; i++){
		if (p[i]){
			for (j = i*i; j<MAX; j += i){
				if (p[j])
					p[j] = 0;
			}
		}
	}
}
