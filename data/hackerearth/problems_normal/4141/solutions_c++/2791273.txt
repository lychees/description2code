#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define push_back pb
#define sci(x) scanf("%d",&x)
#define scl(x) scanf("%lld",&x)

int arr[50001];
int main(){
	int t;
	int m=41;
	int n,c;
	sci(t);
	while(t-->0){
	sci(arr[0]);sci(arr[1]);
	sci(c);sci(n);
	int rem=arr[0]%m;
	if(n>1)
	rem=(rem*10+arr[1])%m;
	for(int i=2;i<n;i++){
		arr[i]=( arr[i - 1]*c + arr[i - 2] ) % 10;
		rem=(rem*10+arr[i])%m;
	}
	if(rem==0)
	cout<<"YES"<<endl;
	else
	cout<<"NO"<<endl;
	}
	return 0;
}
