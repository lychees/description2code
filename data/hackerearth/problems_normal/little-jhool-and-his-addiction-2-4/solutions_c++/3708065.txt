# include <iostream>
# include <algorithm>
using namespace std;
int a[1000005],b[1000005];
int main ()
{
    int n,t,i,j,k,lo,hi,ans;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        for(i=0;i<n;++i)
        {
            cin>>a[i];
        }
        sort(a,a+n);
        i=0;
        j=n-1;
        while(i<j)
        {
            b[i] = a[i]+a[j];
            i++;
            j--;
        }
        j=i;
        lo=b[0];
        hi=b[0];
        for(i=1;i<j;++i)
        {
            if(b[i]>hi)hi=b[i];
            if(b[i]<lo)lo=b[i];
        }
        ans = abs(hi-lo);
        if(ans==k)cout<<ans<<"\nLucky chap!\n";
        else if(ans>k)cout<<ans<<"\nNo more girlfriends!\n";
        else cout<<ans<<"\nChick magnet Jhool!\n";
    }
    return 0;
}
