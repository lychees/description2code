#define X first
#define Y second

using namespace std;

typedef pair<int, int> pii;

bool in[1000][1000];

int main() {
  int n, m, k;
  scanf("%d%d%d", &n, &m, &k);
  pii pt[1000];
  for (int i = 0; i < n; i++)
    scanf("%d%d", &pt[i].X, &pt[i].Y);
  for (int i = 0; i < m; i++) {
    int r, cx, cy;
    scanf("%d%d%d", &r, &cx, &cy);
    for (int j = 0; j < n; j++)
      in[i][j] = (1LL * (cx - pt[j].X) * (cx - pt[j].X) +
                      1LL * (cy - pt[j].Y) * (cy - pt[j].Y) <
                  1LL * r * r);
  }

  while (k--) {
    int a, b, cnt = 0;
    scanf("%d%d", &a, &b);
    a--;
    b--;
    for (int i = 0; i < m; i++)
      if (in[i][a] ^ in[i][b])
        cnt++;
    printf("%d\n", cnt);
  }

  return 0;
}
