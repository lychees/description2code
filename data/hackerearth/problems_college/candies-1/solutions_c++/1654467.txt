#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	long long arr[n],sum=0;
    	for(int i=0;i<n;i++)
    	{
    		scanf("%lld",&arr[i]);
    		sum+=arr[i];
    	}
    	if(sum%n==0)
    	printf("YES \n%lld 0\n",sum/n);
    	else
    	printf("NO \n%lld %lld\n",sum/n,sum%n);
    }
    return 0;
}
