#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,tc;
    cin >> tc;
    
    while(tc--){
    cin >> n;
    char a[n];
    cin >> a;
    int dis = 0,min1 = 999999;
    for (int i = 0;i < n;i++)
    {
        char d = a[i];
        for (int j = i+1;j < n;j++)
        {
            if (a[i] == a[j])
            {
                dis = j - i;
                break;
            }
        }
        
        if (min1 > dis && dis > 0)
            min1 = dis;
    }
    
    if (min1 <= 0  || min1 > n)
        cout << -1 << endl;
    else 
        cout << min1 << endl;
    }
    
    return 0;
}
