#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int a[2000001],b[2000001];
int isPrime(int n){
	int i,k=n/2;
	for(i=2;i<=k;i++){
		if(n%i==0)
		return 0;
	}
	return 1;}
	

int main()
{
    //cout << "Hello World!" << endl;
    int n,i,j,c=0;
    cin>>n;
    for(i=0;i<n;i++)cin>>a[i];
    for(i=0;i<n;i++)cin>>b[i];
    for(i=0;i<n-1;i++){
    
    if(a[i]<a[i+1]&&a[i]>b[i])
    c++;
    }
    //if(a[i]>a[i-1]&&a[i]>b[i])
    c++;
    if(isPrime(c))
    cout<<"MAGIC INVERSION";
    else
    cout<<"SIMPLE INVERSION";
    
    return 0;
}
