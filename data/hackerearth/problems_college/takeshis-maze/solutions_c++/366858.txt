using namespace std;
#include<iostream>
int ar[101][101];
int kr[101][101];
int recur(int,int);
int n;
int main()
{
    int t; cin>>t;
    while(t>0)
    {
        int c; cin>>n; cin>>c;
        int i=0;
        while(i<n)
        {
            int j=0;
            while(j<n)
            {
                kr[i][j]=-1;
                cin>>ar[i][j];
                j++;
            }
            i++;
        }
        int h=recur(0,0);
        if(h-c<0)
            cout<<"-1"<<endl;
        else cout<<h-c<<endl;
        t--;
    }
}
int recur(int i,int j)
{
    if(i==n-1&&j==n-1)
        return 0;
    if(kr[i][j]==-1)
    {
        int a=0,b=0;
        if(i+1<n)
            a=recur(i+1,j);
        if(j+1<n)
            b=recur(i,j+1);
        kr[i][j]=ar[i][j]+max(a,b);
    }
    return kr[i][j];
}
