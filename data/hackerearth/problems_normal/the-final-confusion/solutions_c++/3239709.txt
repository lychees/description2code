#include <iostream>
using namespace std;


long long arr[40][40];
long long f(int x, int y){
	if(arr[x][y] != -1) return arr[x][y];
	if(x == 0) {
		arr[x][y] = 1; 
		return 1;
	}
	if(y == 0) {
		arr[x][y] = f(x-1, y+1);
		return arr[x][y];
	}
	arr[x][y] = f(x-1, y+1) + f(x, y-1);
	return arr[x][y];
}

int main()
{
	for(int i=0; i<40; i++){
		for(int j=0; j<40; j++){
			arr[i][j] = -1;
		}
	}
	int t, n;
	cin>>t;
	while(t--){
		cin>>n;
		cout<<f(n,0)<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
