    l = map(int, raw_input().split())
    l = sorted(l)
    dmax = 0
    p = []
    if len(l) > 2:
        for x in xrange(0, n/2):
            d = l[x]+l[n-1-x]
            p.append(d)
        dmax = max(p)-min(p)
    print dmax
