using namespace std;


#define LL long long int
#define sc1(x) scanf("%d", &x)
#define sc2(x, y) scanf("%d%d", &x, &y)
#define sc3(x, y, z) scanf("%d%d%d", &x, &y, &z)
#define pr1(x) printf("%0.6Lf\n", x)
#define FOR(i, n) for(int i=0;i<n;i++)
#define pp pair<int, int>

#define MAX 1001

bool isSet[MAX][MAX];
long double mat[MAX][MAX];


long double func(int r, int b) {
	if(r <= 0) return 1.0;

	if(isSet[r][b]) return mat[r][b];

	isSet[r][b] = true;
	if(b) {
		long double result = func(r, b-1);
		for(int i=0;i<=4;i++) {
			result += func(r-i, b-1);
		}
		result += func(r-6, b-1);

		return mat[r][b] = result/7.0;
	}
	return mat[r][b] = 0.0;
}


int main() {

	int t, r, b;

	sc1(t);
	while(t--) {

		sc2(r, b);

		pr1(func(r, b));
	}

	return 0;
}
