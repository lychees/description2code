#include <iostream>
#include<stdio.h>
#include<algorithm>
#include<string>
#include<vector>
using namespace std;

vector<string> v;

int check_palin(string a,string b)
{
	int i,j,l=a.length(),f=0;
	for(i=0,j=l-1;i<l/2;i++,j--)
	{
		if(a[i]!=b[j])
		{
			f=1;
			break;
		}
	}
	return f;
}

int main()
{
	int n,i,l,j;
	string s;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>s;
		v.push_back(s);
	}
	
	sort(v.begin(),v.end());
	
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
//		cout<<v[i][0]<<endl;
			if(check_palin(v[i],v[j])==0)
			{
				l=v[i].length();
				cout<<l<<" "<<v[i][l/2]<<endl;
				break;
			}
		}
	}
	
    return 0;
}
