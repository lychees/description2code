num = int(raw_input())
for i in range(num):
    lng = int(raw_input())
    temp = raw_input().split()
    l = []
    for e in temp:
        l.append(int(e))
    l.sort()
    maxi = l[-1]
    sums = 0
    count = (lng-1)*-1
    for e in l:
        sums += (e*count)
        count += 2
    print(maxi*sums) % 1000000007
