t = int(raw_input())

for i in range(t):
	n = int(raw_input())
	a = map(int, raw_input().split(' ' ))
	b = map(int, raw_input().split(' ' ))
	
	dist = 0
	for j in range(n):
		low = 0
		high = n -1 
		while low < high:
			mid = (low + high) >> 1
	#		print low,high,mid,a[mid],b[j]
			if a[mid] > b[j]:
				low = mid + 1
			else:
				high = mid
		#print 'Final',i,high,a[high],b[j]
		if a[high] <= b[j] and j > high and (j - high) > dist:
			dist = j - high
	print dist
