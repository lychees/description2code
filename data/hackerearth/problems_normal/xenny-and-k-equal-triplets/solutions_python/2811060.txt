def gcd(a,b):
	if b==0:
		return a
	else:
		return gcd(b, a%b)

t=input()
while t:
	t-=1
	n,k=map(int, raw_input().split())
	l=map(int, raw_input().split())
	c=l.count(k)
	d=n*(n-1)*(n-2)
	if c<3:
	##	s='0/'+str(d)
		s='0/1'
	else:
		n=c*(c-1)*(c-2)
		g=gcd(n,d)
		n=n/g
		d=d/g
		s=str(n)+'/'+str(d)
