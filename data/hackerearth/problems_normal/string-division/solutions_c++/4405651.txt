#include <iostream>
using namespace std;

void cleararray(int alphabets[]) {
	for (int i = 0; i < 26; i++) {
		alphabets[i] = 0;
	}
}
int main() {

	int N;
	cin >> N;
	string str;
	int alphabets[26], count;

	while(N--) {

		count = 0;
		long prod = 1;
		cin >> str;
		cleararray(alphabets);

		for (int i = 0; i < str.size(); i++) {
			
			if(alphabets[str[i] - 'a'] == 0) {	
				count++;
			}
			alphabets[str[i] - 'a']++;
		}
		if (count == 2) {
			for (int i = 0; i < 26; i++) {
				if (alphabets[i] != 0) {
					prod *= alphabets[i];
				}
			}
		}
		if (count >= 4) {
			cout << "YES" << endl;
		}
		else if (count == 3 && str.size() >= 5) {
			cout << "YES" << endl;
		}
		else if (count == 1 && str.size() >= 10) {
			cout << "YES" << endl;
		}
		else if (count == 2 && (str.size() >= 7 || (str.size() == 6 && prod >= 9)) ) {
			cout << "YES" << endl;
		}
		else {
			cout << "NO" << endl;
		}
	}
	return 0;
}
