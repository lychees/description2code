#include<iostream>
#define mod 1000000007
using namespace std;
long long int a[1004][1004];
int main()
{
	long long int t,m,r,i,j,s;
	cin>>t;
	while(t--)
	{
		cin>>r>>m;
		for(i=0;i<=r;i++)
		{
			for(j=0;j<=r;j++)
			{
				if(i>=j*m)
				{
					if(i==0 && j==0)
					a[i][j]=1;
					else if(i==0)
					a[i][j]=a[i][j-1]%mod;
					else if(j==0)
					a[i][j]=a[i-1][j]%mod;
					else
					a[i][j]=(a[i-1][j]%mod+a[i][j-1]%mod)%mod;
				}
			}
		}
		s=0;
		for(i=r,j=0;i>=0;i--,j++)
		{
			if(m==1 && i==j)
				continue;
			s=(s%mod+a[i][j]%mod)%mod;
		}
		cout<<s<<endl;
		
		for(i=0;i<=r;i++)
		{
			for(j=0;j<=r;j++)
			a[i][j]=0;
		}
		
	}
	return 0;
}
