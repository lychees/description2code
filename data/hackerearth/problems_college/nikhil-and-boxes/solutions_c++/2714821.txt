#include <iostream>
#include<cstdio>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long double l,b,h;
    	cin>>l>>b>>h;
    	if(l<=b && l<=h)
    	printf("%.1llf\n",l/2.0);
    	else if(b<=l && b<=h)
    	printf("%.1llf\n",b/2.0);
    	else
    	printf("%.1llf\n",h/2.0);
    	
    }
    return 0;
}
