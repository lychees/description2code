#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t,n;
    char str[10001];
    cin>>t;
    while(t--)
    {
    	int weight=0,weight1=0;
    	cin>>str;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		int count=0;
    		char c;
    		cin>>c;
    		for(int i=0;i<strlen(str);i++)
    		{
    			if(c == str[i])
    				count++;
    		}
    		weight+=((int)c - (int)'a' + 1)*count;
    		count=0;
    	}
    	for(int i=0;i<strlen(str);i++)
    	{
    		weight1+=(int)str[i] - (int)'a' + 1;
    	}
    	cout<<weight1-weight<<endl;
     }
    return 0;
}
