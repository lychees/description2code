a=raw_input()

b=raw_input()

n=len(a)
count=0
for i in range(0,n):
	if(a[i]!=b[i]):
		
		if(int(a[i])==(int(b[i])+1) or int(b[i])==(int(a[i])+1)):
			
			count=count+1
		if(int(a[i])>(int(b[i])+1) or int(b[i])>(int(a[i])+1)):
			
			count=count+1
		
			
if(count>1):
	print("loses")
else:
	print("wins")
