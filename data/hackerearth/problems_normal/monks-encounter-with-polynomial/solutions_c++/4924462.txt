#include <iostream>
using namespace std;

void binarysearch(long A[], long a, long b, long c, long k);

int main()
{
    long A[100001];
    for(long i=0;i<100001;i++){
    	A[i]=i;
    }
    long t,a,b,c,k;
    cin>>t;
    for(long i=0;i<t;i++){
    	cin>>a>>b>>c>>k;
    	binarysearch(A,a,b,c,k);
    }
    return 0;
}

void binarysearch(long A[], long a, long b, long c, long k){
	long left=-1,right=(100001-1),mid;
	while(right-left>1){
		mid = left + (right-left)/2;
		if((a*A[mid]*A[mid]+b*A[mid]+c)>=k)
		right=mid;
		else
		left=mid;
	}
	cout<<A[right]<<endl;
}
