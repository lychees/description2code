nos, queries = raw_input().split(" ")
nos, queries = int(nos), int(queries)

timesOfQuestions = raw_input().split(" ")
marksOfQuestions = raw_input().split(" ")

itimesOfQuestions = []
imarksOfQuestions = []

marksToTimeMapping = {}

for time in timesOfQuestions:
	itimesOfQuestions.append(int(time))
	
count = 0

for mark in marksOfQuestions:
	imarksOfQuestions.append(int(mark))
	marksToTimeMapping[int(mark)] = itimesOfQuestions[count]
	count += 1
	
sortedMarks = reversed(sorted(imarksOfQuestions))

sumOfTimes = []
sum1 = 0

for sm in sortedMarks:
	sum1 += marksToTimeMapping[sm]
	sumOfTimes.append(sum1)
	
for q in range(1,queries + 1):
	query = int(raw_input())
	print sumOfTimes[query- 1]
