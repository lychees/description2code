#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n;
    scanf("%d",&t);
    while(t--){
    	scanf("%d",&n);
    	int a[n],freq[1000001]={0};
    	for(int i=0;i<n;i++){
    		scanf("%d",&a[i]);
    		freq[a[i]]++;
    	}
    	int max=0;
    	for(int i=0;i<n;i++){
    		if(max<freq[a[i]]) max=freq[a[i]];
    	}
    	for(int i=1000000;i>=0;i--){
    		if(max==freq[i]) printf("%d ",i);
    	}
    	printf("\n");
    }
    return 0;
}
