#include<bits/stdc++.h>
using namespace std;
 
typedef long long ll;
const int N=1e6+6;
 
int val[N];
ll pre[N];
 
void preprocess() {
    for(int i=0;i<N;i++) {
        int temp=i;
        int ans=0;
        while(temp) {
            ans+=(temp/5);
            temp/=5;
        }
        val[i]=ans;
    }
    for(int i=1;i<N;i++) {
        pre[i]=pre[i-1]+val[i];
    }
    return;
}
int main() {
    preprocess();
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int a,b;
        scanf("%d%d",&a,&b);
        printf("%lld\012",pre[b]-pre[a-1]);
    }
    return 0;
}
