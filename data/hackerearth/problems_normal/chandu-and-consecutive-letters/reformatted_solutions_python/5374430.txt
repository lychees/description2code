def main():
    try:
        T = int(raw_input())
    except BaseException:
        return 0
    while T:
        S = raw_input()
        str_len = len(S)
        newS = [S[i] for i in range(0, str_len-1) if not S[i] == S[i+1]]
        newS += S[str_len-1]
        print(''.join(newS))
        T -= 1


if __name__ == "__main__":
    main()
