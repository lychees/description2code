#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin >> t;
    while ( t-- ) {
        string s;
        cin >> s;
        string s1 = s;
        reverse(s1.begin(),s1.end());
        if ( s1 != s ) {
            cout << -1 << endl;
            continue;
        }
        else {
        	cout << (int)s[s.size()/2] << endl;
        }
    }
    return 0;
}
