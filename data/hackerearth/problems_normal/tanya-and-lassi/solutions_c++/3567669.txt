#include<limits.h>
#include<cstdlib>
#include<iostream>

using namespace std;

long long int max(long long int a,long long  int b) { return (a > b)? a : b;}


long long int max_profit( int price[], int n)
{
   long long int profit[n+1];
   profit[0] = 0;
   int i, j;


   for (i = 1; i<=n; i++)
   {
      long long  int max_profit = INT_MIN;
       for (j = 0; j < i; j++)
         max_profit = max(max_profit,(long long )( price[j] + profit[i-j-1]));
       profit[i] = max_profit;
   }

   return profit[n];
}

int main()
{
    int T;
    cin>>T;
    for(int i =0;i<T; i++){
        int n;
        cin>>n;
        int price[n];
        for(int j=0;j<n; j++)
            cin>>price[j];
       cout<< (long long )max_profit(price, n)<<endl;

    }
    return 0;
}
