#include <iostream>
using namespace std;

int main()
{
    int N,L,W[10000],H[10000];
    cin>>L>>N;
    for(int i=0;i<N;i++)
    {
    	cin>>W[i]>>H[i];
    	if(W[i]<L||H[i]<L)
    	cout<<"UPLOAD ANOTHER"<<endl;
    	else if(W[i]==H[i])
    	cout<<"ACCEPTED"<<endl;
    	else
    	cout<<"CROP IT"<<endl;
    }
    return 0;
}
