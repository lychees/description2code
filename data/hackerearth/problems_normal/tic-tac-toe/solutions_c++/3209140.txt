#include <iostream>
#include <cstdio>
using namespace std;
const int MOD = 1e9  + 7;
long long int PowerMod(long long int a, long long int n)
{
	long long int ans = 1;
	while(n)
	{
		if(n&1)
		{
			ans *= a;
			ans %= MOD;
		}
		a *= a;
		a %= MOD;
		n = n>>1;
	}
	return ans;
}
int main()
{
    int t;
    scanf("%d", &t);
    long long int n;
    long long int minx, maxx;
    long long int powmo4 = PowerMod(4, MOD -2);
    long long int powmo6 = PowerMod(6, MOD-2);
    while(t--)
    {
    	scanf("%lld", &n);
    	minx = ((((((n)*(n-1))%MOD)*(n-1))%MOD)*powmo4)%MOD;
    	maxx = ((((((n-1)*n)%MOD)*(2*n - 1))%MOD)*powmo6)%MOD;
    	printf("%lld %lld \n", minx, maxx);
    }
    return 0;
}
