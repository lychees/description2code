#include <iostream>
#include<map>
using namespace std;

int main()
{
    int t,n,p,x,y,i,j,k;
    cin>>t;
    pair<int,int> ans[t];
    for(i=0;i<t;i++)
    {
    	cin>>n>>p;
    	int a=0,b=0;
    	pair<int,int> arr[n],dp[n+1][p+1];
    	for(j=0;j<n;j++)
    	{
    		cin>>x>>y;
    		arr[j].first=x; arr[j].second=y;
    	}
    	for(j=0;j<=n;j++)
    	{
    		for(k=0;k<=p;k++)
    		{
    			if(j==0)
    			{
    				dp[j][k].first = k;
    				dp[j][k].second = 0;
    			}
    			else if(k==0)
    			{
    				dp[j][k].first = 0;
    				dp[j][k].second=0;
    			}
    			else if(arr[j-1].second<=k)
    			{
    				if(arr[j-1].first + dp[j-1][k-arr[j-1].second].first>dp[j-1][k].first)
    				{
    					dp[j][k].first = arr[j-1].first + dp[j-1][k-arr[j-1].second].first;
    					dp[j][k].second = arr[j-1].second + dp[j-1][k-arr[j-1].second].second;
    				}
    				else if(arr[j-1].first + dp[j-1][k-arr[j-1].second].first==dp[j-1][k].first && arr[j-1].second + dp[j-1][k-arr[j-1].second].second<dp[j-1][k].second)
    				{
    					dp[j][k].first = arr[j-1].first + dp[j-1][k-arr[j-1].second].first;
    					dp[j][k].second = arr[j-1].second + dp[j-1][k-arr[j-1].second].second;
    				}
    				
    				else
    				{
    					dp[j][k].first = dp[j-1][k].first;
    					dp[j][k].second = dp[j-1][k].second;
    				}
    				
    			}
    			else
    			{
    				dp[j][k].first = dp[j-1][k].first;
    				dp[j][k].second = dp[j-1][k].second;
    			}
    		}
    	}
    	ans[i].first = dp[n][p].first; ans[i].second=dp[n][p].second;
    }
    for(i=0;i<t;i++)
    cout<<ans[i].second<<" "<<ans[i].first<<endl;
    
    return 0;
}
