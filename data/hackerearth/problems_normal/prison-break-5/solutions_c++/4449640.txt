#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pii pair<ll,ll>
#define pll pair<ll,ll>
#define mp make_pair
#define ff first
#define ss second
#define pb push_back
#define pf pop_front
#define IOS ios::sync_with_stdio(false)
ll t , n , a[100][100];
int dx[] = {1,0,-1,0};
int dy[] = {0,1,0,-1};
int vis[105][105];
bool check(int x , int y){
    if((x >= 1 && x <= n) && (y <=n && y >= 1)  && (a[x][y]==0) &&!vis[x][y])
        return true;
    else return false;
}
ll solve(ll x , ll y){
    if(x==n && y==n && a[x][y]==0)return 1;
    ll sum = 0;
    vis[x][y] = 1;
    for(int i = 0 ; i < 4 ; i++){
        int nx = x + dx[i];
        int ny = y + dy[i];
        if(check(nx,ny)){
            sum+=solve(nx,ny);
            vis[nx][ny] = 0;
        }

    }
    return sum;
}
int main(){

	ll i , j , k;
    IOS;
    cin.tie(NULL);
    cin >> t;
    while(t--){
        cin >> n;
        memset(a,0,sizeof(a));
        memset(vis,0,sizeof(vis));
        for(i = 1 ; i <= n ; i++){
            for(j = 1 ; j <= n ; j++)
                cin >> a[i][j];
        }
        cout << solve(1,1) << endl;

    }
	return 0;
}
