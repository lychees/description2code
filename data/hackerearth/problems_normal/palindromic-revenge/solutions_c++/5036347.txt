#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int n,i,j,count;
   string s;
   cin>>n;
   while(n--)
   {
   		count=0;
   		cin>>s;
   		i=0;
   		j=s.size()-1;
   		while(i<j)
   			count+=abs(s[i++]-s[j--]);
   		cout<<count<<endl;
   }
    return 0;
}
