#include <iostream>
using namespace std;


void allchar(string a){
	int n=a.size();
	bool p[26];
	
	for(int i=0 ; i<26; i++){
		p[i]=0;
	}
	
	for(int i =0 ; i<n; i++){
		p[a[i]-97]=1;
	}
	
	bool f=1;
	
	for(int i=0 ; i<26 && f;i++ ){
		if(!p[i])f=0;
	}
	
	if(f){
		cout<<"YES"<<endl;
	}
	else{
		cout<<"NO"<<endl;
	}
}

int main()
{
	int n;
	cin>>n;
	string a[n];
	
	for(int i=0 ;i<n ;i++){
		cin>>a[i];
	}
	
 	for(int i=0 ; i<n ;i++){
 		allchar(a[i]);
 	}
}
