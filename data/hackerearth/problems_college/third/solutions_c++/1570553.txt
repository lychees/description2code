#include <iostream>
#include <stdio.h>
#include <vector>

using namespace std;

int main()
{
	int t, n, i, tmp, inc, exc;
	
	scanf("%d", &t );
	
	while( t-- ) {
		
		scanf("%d", &n );
		vector<int> A( n );
		
		for( i = 0;i < n;i++ )
			scanf("%d", &A[ i ] );
		
		if( n == 0 )
			printf("0\n");
		else {
			
			inc = A[ 0 ];
			exc = 0;
			
			for( i = 1;i < n;i++ ) {
				
				tmp = max( inc, exc );
				inc = exc + A[ i ];
				exc = tmp;
			}
			
			printf("%d\n", max( inc, exc ) );
		}
	}
	
	return 0;
}
