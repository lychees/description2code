#include<cstdio>
#include<iostream>
#include<cstring>
#define max 1999999
#define mod 1000000007
#define gc getchar_unlocked
#define pu putchar_unlocked
using namespace std;
int p[max],pc[max],ans[1000001];
inline void sieve(){
	memset(p,1,sizeof(p));p[0]=0;p[1]=0;p[2]=1;pc[0]=0;pc[1]=0;
	for(int i=2;i<=max;i++){
		pc[i]=pc[i-1];
		if(p[i]){
			pc[i]++;
			for(int j=2*i;j<=max;j+=i){
				p[j]=0;
			}
		}
	}

}
inline void input(int &x){ //fast io
   register int c=gc();
   x=0;int neg=0;
   for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
void out(int x)
{
  if(x<0)
    {
      pu('-');
      x=-x;
    }

      data[len++]=x%10;
      x/=10;
    }
  if(!len)data[len++]=0;
  while(len--)
    pu(data[len]+48);
  pu('\n');
}
int main(){
	sieve();ans[0]=0;ans[1]=0;
	for(int i=2;i<=1000000;i++){
        ans[i]=(ans[i-1]%mod+(2*(pc[2*(i-1)-1]-pc[i-2]))%mod)%mod;
        if(p[2*(i-1)])ans[i]++;
	}
	int n;input(n);
	out(ans[n]);
	return 0;
}
