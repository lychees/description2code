#include<iostream>
#include<algorithm>
#include<string.h>
using namespace std;

int main()
{
int arr[1000000];
memset(arr,0,sizeof arr);
int i,j;
int t,n;
for(i=2;i*i< 1000000;i++)
    {
        if(arr[i]==0){
        for(j=i*i;j<1000000;j+=i)
        {
            if(arr[j]==0)
            arr[j]=1;
        }
        }
    }
    
    cin>>t;
    while(t--)
    {
        cin>>n;
        n++;
        while(arr[n]==0)
            n++;
        cout<<n<<endl;
        
    }
return 0;
}
