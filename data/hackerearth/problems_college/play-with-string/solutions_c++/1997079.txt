#include<bits/stdc++.h>

using namespace std;

string reverse(string s)
{
	int i=0;
	int j = s.length()-1;
	while(i<j){
		char t = s[i];
		s[i]=s[j];
		s[j]=t;
		i++;
		j--;
	}
	return s;
}

int main()
{
	int T;
	string s;
	string r;
	cin>>T;
	while(T--){
		cin>>s;
		r = reverse(s);
		int fl=0;
		//cout<<r<<endl;
		for(int i=1;i<s.length();i++){
			if(abs(s[i]-s[i-1]) != abs(r[i]-r[i-1])){
				fl=1;
				break;
			}
		}
		if(fl==1){
			cout<<"Not Funny\n";
		}
		else cout<<"Funny\n";
	}
	return 0;
}
