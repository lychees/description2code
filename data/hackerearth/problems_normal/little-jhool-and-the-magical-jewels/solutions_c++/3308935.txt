#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,l,i,f,minn;
    string s;
    scanf("%d",&t);
    while(t--){
    	cin>>s;
    	l=s.size();
    	int a[26];
    	memset(a,0,sizeof(a));
    	for(i=0;i<l;i++){
    		int k = s[i]-'a';
    		a[k]++;
    	}
    	/*for(i=0;i<26;i++){
    		//cout<<s[i]<<" "<<a[s[i]]<<" ";
    		cout<<a[i]<<" ";
    	}
    	cout<<endl;*/
    	minn=INT_MAX;
    	f=0;
    	if(a['r'-'a']>0 && f==0){
    		if(a['r'-'a']<minn){
    			minn=a['r'-'a'];
    		}
    	}
    	else if(a['r'-'a']==0){
    		f=1;
    	}
    	if(a['u'-'a']>0 && f==0){
    		if(a['u'-'a']<minn){
    			minn=a['u'-'a'];
    		}
    	}
    	else if(a['u'-'a']==0 || f==1){
    		f=1;
    	}
    	if(a['b'-'a']>0 && f==0){
    		if(a['b'-'a']<minn){
    			minn=a['b'-'a'];
    		}
    	}
    	else if(a['b'-'a']==0 || f==1){
    		f=1;
    	}
    	if(a['y'-'a']>0 && f==0){
    		if(a['y'-'a']<minn){
    			minn=a['y'-'a'];
    		}
    	}
    	else if(a['y'-'a']==0 || f==1){
    		f=1;
    	}
    	if(f==0) printf("%d\n",minn);
    	else if(f==1) printf("0\n");
    }
    return 0;
}
