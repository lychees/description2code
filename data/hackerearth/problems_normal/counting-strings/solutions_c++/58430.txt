#include <stdio.h>
#include<algorithm>
#include<iostream>
#include<cstring>

#define MAX 1000000
#define FOR(i,n) for(int i=0; i<n; ++i)

using namespace std;


int main()
{
	int t;
	char s[MAX];

	scanf("%d", &t);

	FOR(T,t)
	{
		scanf("%s", s);
		
		int m = strlen(s), lasti=m;
		long long cnt=0;
		
		for(int i=m-1;i>=0;--i)
		{
			if(s[i]=='a' || s[i]=='z')
			{
				lasti = i;
			}

			cnt += m-lasti;
		}		

		printf("%lld\n",cnt);
	}

	return 0;
}
