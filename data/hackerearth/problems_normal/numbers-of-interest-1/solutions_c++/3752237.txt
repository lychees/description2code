#include <iostream>
#include <list>
using namespace std;

int main()
{
    long cases,x,y,n,answer,head;
    cin>>cases;
    while(cases--){
    	cin>>x>>y>>n;
    	if(n<=x){
    		cout<<y<<endl;
    		continue;
    	}
    	list<long> ans;
    	for(int i=0;i<x;i++) ans.push_back(y);
    	n=n-x-1;
    	ans.push_back(x*y);
    	answer=x*y;
    	for(int i=0;i<n;i++){
    		head=ans.front();
    		ans.pop_front();
    		answer*=2;
    		answer-=head;
    		ans.push_back(answer);
    	}
    	cout<<ans.back()<<endl;
    }
    return 0;
}
