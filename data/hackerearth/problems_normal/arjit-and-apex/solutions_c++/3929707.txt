#include <iostream>
#include <bits/stdc++.h>
using namespace std;

struct MyComparator
{
    const vector<int> & value_vector;
 
    MyComparator(const vector<int> & val_vec):
        value_vector(val_vec) {}
 
    bool operator()(int i1, int i2)
    {
        return value_vector[i1] < value_vector[i2];
    }
};

int main()
{
    int t;
    cin >> t;
    for(int i=0;i<t;i++)
    {
    	int m,n;
    	cin >> m >> n;
    	vector<int> A1;
    	vector<int> A2;
    	vector<int> B1;
    	vector<int> B2;
    	vector<int> X;
    	vector<int> Y;
    	
    	for(int j=0;j<m;j++)
    	{
    		int a,b;
    		cin >> a >> b;
    		A1.push_back(a);
    		A2.push_back(b);
    		X.push_back(j);
    	}
    	for(int j=0;j<n;j++)
    	{
    		int a,b;
    		cin >> a >> b;
    		B1.push_back(a);
    		B2.push_back(b);
    		Y.push_back(j);
    	}
    	
    	int g,h,good=0,great=0;
    	cin >> g >> h;
    	
    	sort(X.begin(),X.end(),MyComparator(A1));
    	sort(Y.begin(),Y.end(),MyComparator(B1));
    	
   		int k=0;
    	for(int j=0;j<n && k<m;j++)
    	{
    		if(B1[Y[j]]==A1[X[k]])
    		{
    			good++;
    			if(B2[Y[j]]==A2[X[k]])
    				great++;
    		}
    		else if(B1[Y[j]]>A1[X[k]])
    		{
    			k++;
    			j--;
    		}
    	}
    	
    	cout << ":(" << endl;
    }
}
