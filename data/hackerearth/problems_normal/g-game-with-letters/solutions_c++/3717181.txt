#include <iostream>
#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t, i, j, k, l, n;
	string a, b, tmp;
	int c[26];
	int score[10001];
	cin>>t;

	while (t--) {
		cin>>n;
		cin>>a;
		memset (c, 0 ,sizeof(c));
		memset (score, 0, sizeof (score));
		for (i = 0; i < a.length(); i++) {
			c[a[i]-'a']++;
		}
		l = INT_MIN;	
		for (i = 0; i < n; i++) {
			cin>>b;
			for (j = 0; j < b.length(); j++) {
				score[i] += c[b[j]-'a'];
			}
			if (score[i] > l) {
				k = i;
				tmp = b;
				l = score[i];
			}
			else if (score[i] == l && tmp.length() > b.length()) {
				k = i;
				tmp = b;
			}
		}

		cout<<k+1<<endl;
	}
	return 0;
}
