#include<bits/stdc++.h>
#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    int n;
    scanf("%d",&n);
    int a[26],counter=0;
    for(int i=0;i<26;i++)
            a[i]=0;
    while(n--)
    {
        char s[100];
        scanf("%s",s);
        int temp=s[0];
        a[temp-97]++;

    }

    for(int i=0;i<26;i++)
    {
        if(a[i]>0&&a[i]<=10)
            counter++;
        if(a[i]>0&&a[i]>10&&a[i]%10==0)
            counter+=a[i]/10;
        if(a[i]>0&&a[i]>10&&a[i]%10!=0)
            counter+=a[i]/10+1;


    }

      printf("%d\n",counter);

    return 0;

}
