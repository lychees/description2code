#!usr/bin/python3

import math


def isprime(k):
    root = int(math.sqrt(k))
    for j in range(2, root + 1):
        if k % j == 0:
            return False
        j = j - 1

    return True


fact = 1
n = input()
for i in range(2, n+1):
    fact = fact * i
    if isprime(i):
        print fact
    i = i - 1
