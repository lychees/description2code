bool isprime(int n)
{
	if (n<2) return false;
    for (int i=2; i<=sqrt(n); i++)
    {
    	if (n%i==0)
    	{
    		return false;
    	}
    }
    return true;
}

int main() {
	int t; scan(t);

		for (int i=0; i<n; i++)
		{
			scan(arr[i]);
		}
		set<int> st; set<int> :: iterator itr;
		st.insert(0);
		for (int i=0; i<n; i++)
		{
			for (itr=st.begin(); itr!=st.end(); ++itr)
			{
				int res=arr[i]^(*itr);
				st.insert(res);
			}
			st.insert(arr[i]);
		}
		int cnt=0;
		for (itr=st.begin(); itr!=st.end(); ++itr)
		{
			if (isprime(*itr))
			{
				cnt+=1;
			}
		}
		cout<<cnt<<endl;
