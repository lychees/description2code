#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int p,x;
    	cin>>x>>p;
    	int count=0;
    	while(x%p==0)
    	{
    		count++;
    		x=x/p;
    		if(x==p)
    		{
    			count++;
    			x=0;
    			break;
    		}
    	}
    	cout<<count<<" "<<x<<endl;
    }
    return 0;
}
