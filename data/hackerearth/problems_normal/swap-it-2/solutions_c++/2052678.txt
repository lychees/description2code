#include <iostream>
using namespace std;

int main(){
//	freopen("input1.txt","r",stdin);
//	freopen("output.txt","w",stdout);
	int t,n,i,k;
	cin>>t;
	while(t--){
		cin>>n>>k;
		int a[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0;i<n;i++){
			int pos=i;
			for(int j=i+1;j<n;j++){
				if((j-i)>k)
				break;
				if(a[j]<=a[pos])
				pos=j;
			}
			for(int j=pos;j>i;j--)
			a[j]=(a[j]+a[j-1])-(a[j-1]=a[j]);
			k-=(pos-i);
		}
		for(i=0;i<n;i++)
		cout<<a[i]<<" ";
		cout<<endl;
	}
//	fclose(stdout);
	return 0;
}
