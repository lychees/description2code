#include <iostream>
using namespace std;

#define MAX 505
#define alpha 1000000007

long long arr[MAX][MAX],n,h, ans[MAX][MAX];

int main() {
	
	for(h=1; h<MAX; h++) {
		arr[0][h]=1;
		arr[1][h]=1;
	}
	
	for(int n=1; n<MAX; n++) {
		arr[n][0]=0;
	}
	
	for(int n=2; n<MAX; n++) {
		for(int h=2;h<MAX;h++) {
			if(h>n) {
				arr[n][h]=arr[n][h-1];
				continue;
			}
			arr[n][h]=0;
			for(int i=1;i<=n;i++) {
				long long temp = (arr[i-1][h-1]*arr[n-i][h-1])%alpha;
				arr[n][h] = (arr[n][h]+temp)%alpha;
			}
		}
	}
	
	for(int n=2; n<MAX; n++) {
		for(int h=2;h<MAX;h++) {
			if(h>n) {
				ans[n][h]=ans[n][h-1];
				continue;
			}
			ans[n][h]=0;
			for(int i=1;i<=n;i++) {
				long long temp = (arr[i-1][h-1]*arr[n-i][h-1])%alpha;
				temp = temp*i;
				ans[n][h] = (ans[n][h]+temp)%alpha;
			}
		}
	}
	
	cin>>n>>h;
	cout<<(ans[n][h]%alpha)<<"\n";
	return 0;
}
