import re
for _ in xrange(input()):
    s = raw_input()
    s = s.split()
    w = []
    tot = 0
    for item in s:
        t_s = ''
        for ch in item:
            if ord(ch) >= ord('0') and ord(ch) <= ord('9'):
                tot += int(ch)
            else:
                t_s += ch
        w.append(t_s)

    w.sort()
    ans = ''
    for item in w:
        tot -= len(item)
        ans += item+" "+str(abs(tot))+" "
    print ans
