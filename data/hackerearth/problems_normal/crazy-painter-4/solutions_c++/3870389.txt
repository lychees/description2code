#include <iostream>
using namespace std;

int main()
{
    long int *arr=new long int[27];
    long long int t,n,i,total,rounds,remainings,cost,x;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=1;i<=26;i++)
    	cin>>arr[i];
    	total=n*12;
    	rounds=total/26;
    	remainings=total%26;
    	cost=0;
    	for(i=1;i<=26;i++)
    	{
    		x=arr[i]+rounds-1;
    		cost+=(x*(x+1))/2;
    		x=arr[i]-1;
    		cost-=(x*(x+1))/2;
    		arr[i]+=rounds;
    	}
    	for(i=1;i<=remainings;i++)
    	{
    		cost+=arr[i];
    	}
    	cout<<cost<<endl;
    }
    return 0;
}
