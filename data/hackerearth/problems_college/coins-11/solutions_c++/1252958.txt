int prime(long n)
{
	int flag=1;
	if(n==3)
	return 3;
	else if(n==1)
	return 1;
	else
	{
		for(long j=2;j*j<=n;j++)
		{
			if(n%j==0)
			{
				flag=0;
			}
		}
	}
	if(flag) return(2);
	else if(!flag) return(1);
}
