#include <iostream>
using namespace std;

int main()
{
	int t,n,i,j,row[1001],col[1001],total,curr;
	char c;
	bool flag;
    cin>>t;
    while (t--)
    {
    	cin>>n;
    	total=0;
    	for (i=0;i<n;i++) row[i]=col[i]=0;
    	for (i=0;i<n;i++)
    	{
    		for (j=0;j<n;j++)
    		{
    			cin>>c;
    			if (c=='#')
    			{
    				row[i]+=1;
    				col[j]+=1;
    				total+=1;
    			}
    		}
    	}
    	flag=false;
    	if (total%2==0)
    	{
	    	curr=0;
	    	for (i=0;i<n;i++)
	    	{
	    		if (curr==total/2)
	    		{
	    			flag=true;
	    		}
	    		curr+=row[i];
	    	}
			if (curr==total/2)
			{
				flag=true;
			}
			
			curr=0;
	    	for (i=0;i<n;i++)
	    	{
	    		if (curr==total/2)
	    		{
	    			flag=true;
	    		}
	    		curr+=col[i];
	    	}
			if (curr==total/2)
			{
				flag=true;
			}
    		
    	}
    	if (flag) cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
    }
    return 0;
}
