#include <bits/stdc++.h>
#define ll long long int
#define gt(i) scanf("%lld", &i)

using namespace std;

int main()
{
	ll n, i, j, k;
	map <string, ll> mp;
	map <string, ll> :: iterator pos;
	gt(n);
	string a;
	for(i = 0; i < n; i++) { 
		cin >> a;
		sort(a. begin(), a.end());
		mp[a]++;
	}
	ll maxv = 0;
	for(pos = mp.begin(); pos != mp.end(); pos++) {
		if(maxv < pos->second)
			maxv = pos->second;
	}
	cout << maxv << endl;
	return 0;
	
}
