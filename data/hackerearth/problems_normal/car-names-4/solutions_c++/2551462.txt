#include <iostream>
using namespace std;

int main()
{
    string s;
    int test;
    cin>>test;
    while(test--)
    {
    	cin>>s;
    	long long int i;
    	int flag=0;
    	char letters[5];
    	letters[0]=s[0];
    	int j=0;
    	long long int ctr=0;
    	long long int rem[3];
    	for(i=0; s[i]!='\0'; i++)
    	{
    		//cout<<s[i];
    		if(letters[j]!=s[i])
    		{
    			//cout<<endl;
    			rem[j]=ctr;
    			//cout<<ctr<<endl;
    			ctr=0;
    			j++;
    			if(j>2)
    			{
    				flag=1;
    				break;
    			}
    			letters[j]=s[i];
    		}
    		ctr++;
    	}
    	if(s[i]=='\0')
    		rem[j]=ctr;
    		
    	//cout<<"rem: "<<rem[0]<<" "<<rem[1]<<" "<<rem[2]<<endl;
    	if(flag==1)
    	{
    		cout<<"Not OK"<<endl;
    		continue;
    	}
    	//cout<<"flag:"<<flag;

    	if(rem[0]!=rem[1] || rem[1]!=rem[2] || rem[2]!=rem[0])
    		cout<<"Not OK"<<endl;
  		else if(letters[0]==letters[1] || letters[2]==letters[1] || letters[0]==letters[2])
    		cout<<"Not OK"<<endl;
    	else
    		cout<<"OK"<<endl;
    }
    return 0;
}
