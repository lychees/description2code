# include<iostream>
# include<cstdio>
# include<algorithm>
# include<cstring>
# include<map>
# include<cmath>
using namespace std;
int main()
{
    int TC = 0;
    cin >> TC;
    while (TC --)
    {
        long long int ID;
        cin >> ID;
        long long int mat[30];
        long long int totalSum = 0;
        for( int i = 0;i< 26;i++)
        {
            cin >> mat[i];
            totalSum = totalSum + mat[i];
        }
	
        long long int points =  10 * ID + 2 * ID;
        long long int cost = 0;
        long long int uses = points / 26;
        int partial_used = points % 26 ;

        if ( uses >= 1)
        cost = uses * totalSum + 13 * (uses -1) * (uses);
		
        for( int i = 0 ;i < partial_used ; i++)
            cost = cost + mat[i] + uses;

        cout << cost << endl;       
    }
}
