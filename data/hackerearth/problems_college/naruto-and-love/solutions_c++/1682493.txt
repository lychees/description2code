
#include <cstdio>
#include <algorithm>
#include <cmath>
using namespace std;
int T;
int N, M;
int pos[150000 + 1];
int dish[150000 + 1];
bool test(int v) {
    int j = 1;
    int i = 1;
    while(i <= N && j <= M) {
        if(abs(dish[j] - pos[i]) <= v) {
            ++i;
            ++j;
        }
        else {
            if(pos[i] + v < dish[j])
                return false;
            ++j;
            if(j - i > M - N)
                return false;
        }
    }
    if(i > N)
        return true;
    return false;
}
int main()
{
    scanf("%d", &T);
    while(T--) {
        scanf("%d%d", &N, &M);
        for(int i = 1; i <= N; ++i)
            scanf("%d", pos + i);
        for(int i = 1; i <= M; ++i)
            scanf("%d", dish + i);
        sort(pos + 1, pos + 1 + N);
        sort(dish + 1, dish + 1 + M);
        long long l = 0, h = 2000000000 + 10, mid;
        while(l != h) {
            mid = (l + h)/2;
            if(test(mid))
                h = mid;
            else
                l = mid + 1;
        }
        printf("%lld\n", l);
    }
    return 0;
}
