#include <iostream>
using namespace std;

int main()
{
    int a,b,i=0;
    cin>>a>>b;
    int n[b-a-2],j=0;
    for(i=a+1;i<b;i++)
    {
    	if((i%3==0&&i%5==0)||(i%3!=0&&i%5!=0))
    	{
    	n[j]=i;
    	j++;
    	}
  
    }
    cout<<n[0];
    for(i=1;i<j;i++)
    {
    	cout<<","<<n[i];
    }
    return 0;
}
