#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int arr[26];
    int T, myst=0, len, t;
    string s;
    cin>>T;
    for(int i=0; i<T; i++){
    	for(int j=0;j<26;j++){
    		arr[j] = 0;
    	}
    	myst = 0;
    	cin>>s;
    	len = s.length();
    	for(int j=0; j<len; j++){
    		t = (int)s[j] - 97;
    		arr[t] = arr[t] + 1;
    		if(arr[t] > myst){
    			myst = arr[t];
    		}
    	}
    	cout<<myst<<endl;
    }
    return 0;
}
