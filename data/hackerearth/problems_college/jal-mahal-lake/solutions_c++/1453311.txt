#include <bits/stdc++.h>
using namespace std;
typedef struct fun{
	long long pos;
	long long len;
};
int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--)
   {
   	long long n,i,j,temp;
   	cin>>n;
   	fun f[n];
   	for(i=0;i<n;i++)
   	cin>>f[i].pos;
   	for(i=0;i<n;i++)
   	cin>>f[i].len;
   for(i=0;i<n;i++)
   {
   	for(j=i+1;j<n;j++)
   	{
   		if(f[i].pos>f[j].pos)
   		{
   			temp=f[i].pos;
   			f[i].pos=f[j].pos;
   			f[j].pos=temp;

   			temp=f[i].len;
   			f[i].len=f[j].len;
   			f[j].len=temp;
   		}
   	}
   }
   
   long long s=0;
   for(i=0;i<n;i++)
   {
   	f[i].len=f[i].len+f[i].pos;
   	
   }
   long long end=n;
   for(i=1;i<n;i++)
   {
   	long long x=f[i].pos-f[i-1].len;
   	s+=min(x*i,x*(end-i));
   }
   cout<<s<<endl;
   }
    return 0;
}
