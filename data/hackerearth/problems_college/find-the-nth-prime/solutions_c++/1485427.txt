#include <iostream>
#include <vector>
#include <math.h> 
#include <algorithm>
using namespace std;

int prime(int x)
{
	int i;
	for(i=2;i*i<=x;i++)
			if(x%i==0)
				return 0;

	return 1;
}
	
	
	


int main()
{
	int t,i=3,count=1;
	cin>>t;
	if(t==1)
		i=2;
	else
	while(i<100000000)
	{
		if(prime(i))
		{
			count++;
			if (count==t)
				break;
		}
		i=i+2;
	}
cout<<i<<endl;
return 0;

}
