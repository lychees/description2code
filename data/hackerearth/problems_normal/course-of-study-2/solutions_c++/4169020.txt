#include <iostream>
#include <map>
using namespace std;

using ull = unsigned long long;
constexpr ull mod = 1000000007ull;

ull powm(ull a, ull n) {
	ull res = 1;
	while (n) {
		if (n % 2)
			res = (res * a) % mod;
		a = (a * a) % mod;
		n /= 2;
	}
	return res;
}

ull inversem(ull a) {
	return powm(a, mod - 2);
}

int main() {
	ull num = 1;
	ull den = 1;
	int n;
	cin >> n;
	ull sc = 0;
	map<int, int> q;
	for (int i = 0; i < n; ++i) {
		int m;
		cin >> m;
		ull t = 0;
		for (int j = 0; j < m; ++j) {
			int x;
			cin >> x;
			num = (num * ++q[x]) % mod;
			num = (num * ++sc) % mod;
			den = (den * ++t) % mod;
		}
		q.clear();
	}
	num = (num * inversem(den)) % mod;
	cout << num << '\n';
    return 0;
}
