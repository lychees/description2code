#include <iostream>

using namespace std;

int main()
{
        int test_cases,c;
        
        cin>>test_cases;
        while(test_cases > 0)
        {
              int   c=0;
                int X1,X2,Y1,Y2;
                cin>>X1>>Y1>>X2>>Y2;
                
                if(Y1==X2)
                c++;
                if(Y2==X1)
                c++;
                
                cout<<c<<endl;
                test_cases --;
        }
        return 0;
}
