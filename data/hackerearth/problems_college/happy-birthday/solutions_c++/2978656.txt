#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
    int n,sum = 0;
    scanf("%d",&n);
    int arr[n],i;
    for(i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
        sum+=arr[i];
    }
     int ans = sum/4;
    if(sum%4==0)
    {
        printf("%d",ans);
    }
    else
    {
        printf("%d\n",ans+1);
    }



    return 0;
}
