#include <iostream>
#include<cmath>
using namespace std;
int reverse(int a)
{
	int x,r=0;
	while(a!=0)
	{
		x=a%10;
		r=r*10+x;
		a=a/10;
	}
	return r;
}
int main()
{
   long long t,n,p,g,k,y;
   cin>>t;
   while(t--)
   {
   		cin>>n;
   		y=n*n;
   		p=reverse(n);
   		if(p!=n)
   		{
   		k=p*p;
   		g=reverse(k);
   		if(g==y)
   		cout<<p<<endl;
   		else
   		cout<<"No\n";
   		}
   		else
   		cout<<"No\n";
   }
    return 0;
}
