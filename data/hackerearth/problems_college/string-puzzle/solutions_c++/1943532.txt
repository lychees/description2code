#include <iostream>
using namespace std;
void swap (char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}
void permute(char *a, int i, int n) 
{
   int j; 
   if (i == n)
     printf("%s\n", a);
   else
   {
        for (j = i; j <= n; j++)
       {
          swap((a+i), (a+j));
          permute(a, i+1, n);
          swap((a+i), (a+j)); //backtrack
       }
   }
} 
int main()
{
    char s[100];
    cin>>s;
    int t,start,end;
    cin>>t;
    while(t--){
    	
    	cin>>start>>end;
    	permute(s,start,end);
    	cout<<endl;
    }
    
}
