#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define ll long long
#define mod 1000000007
#define bitcount    __builtin_popcountll
#define pb push_back
vector<int> q[100005];
int a[100005];
int dfs(int cur,int par)
{
	int j=0;
	for(int i=0;i<q[cur].size();i++)
	{
		if(q[cur][i]==par)
			continue;
		j+=dfs(q[cur][i],cur);
	}
	a[cur]=j;
	return j+1;
}
int main()
{
    //freopen("in.txt","r",stdin);
    //freopen("out.txt","w",stdout);
    int n,i,j,k;
    sd(n);
    for(i=0;i<n-1;i++)
    {
    	sd(j);
    	sd(k);
    	q[j].pb(k);
    	q[k].pb(j);
    }
    dfs(1,0);
    printf("%d ",a[1]);
    for(i=2;i<=n;i++)
    	printf("%d ",a[i]+1);
    return 0;
}
