#include<cstdio>
#include<vector>
const int N = 2001;
int n, i, gl, x, wynik;
std::vector<int>pocz, v[N];

void dfs(int x)
{
  gl++;
  if (gl > wynik){wynik = gl;}
  for(int i = 0; i < v[x].size(); i++)
  {
    dfs(v[x][i]);
  }
  gl--;
}

int main()
{
  scanf("%d", &n);
  
  for(i = 1; i <= n; i++)
  {
    scanf("%d", &x);
    if (x != -1)
    {
      v[x].push_back(i);
    }
    else
    {
      pocz.push_back(i);
    }
  }
  
  gl = 0;
  wynik = 0;
  for(i = 0; i < pocz.size(); i++)
  {
    dfs(pocz[i]);
  }
  
  printf("%d\n", wynik);
  
return 0;
}
