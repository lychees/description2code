#include <iostream>
#include<string>
#include<sstream>
using namespace std;
int pallen( string s3)
{
    string r;
    long int len;
    for(len = (s3.size())-1; len>=0 ;r+=s3[len], len--);
	return (s3==r)?1:0;
}

int main()
{
    string s1,s2;
    int t;
    cin>>t;
    while(t--)
    {
    	cin>>s1>>s2;
    	if(pallen(s1+s2)||pallen(s2+s1)) cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
    return 0;
}
