#include<bits/stdc++.h>
using namespace std;
#define ull long long int
int main()
{
    ios::sync_with_stdio(false);
    ull n,t,i,m,j;
    cin>>t;
    while(t--)
    {
        cin>>m>>n;
        ull arr[n+3];
        for(i=0;i<n;i++)
            cin>>arr[i];
        sort(arr,arr+n);
        bool dp[m+2];
        memset(dp,false,sizeof(dp));
        for(i=0;i<n;i++)
            dp[arr[i]]=true;
        for(i=1;i<=m;i++)
        {
            if(!dp[i])
            {
                for(j=0;j<n;j++)
                {
                    if(i-arr[j]>=0 && !dp[i-arr[j]]){
                        dp[i]=true;
                        break;
                    }
                }
            }
        }
        if(dp[m])
            cout<<"Little Deepu\n";
        else
            cout<<"Kate\n";
    }
    return 0;
}
