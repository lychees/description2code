
def min(arg1, agr2, arg3):
    if arg1 < agr2 and arg1 < arg3:
        return arg1
    elif agr2 < arg3:
        return agr2
    else:
        return arg3


t = int(raw_input())
while t > 0:
    l, b, h = raw_input().split()
    ans = min(long(l), long(b), long(h))/2.0
    print "%.1f" % ans
    t -= 1
