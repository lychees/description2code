#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t-->0)
    {
    	long long int num=0;
    	cin>>num;
    	if(sqrt(num)==floor(sqrt(num)))
    	{
    		cout<<"YES"<<endl;	
    	}
    	else
    	{
    		unsigned long long int p,q,a,b,sq;
    		sq=sqrt(num);
    		
    		a=sq*sq;
    		b=(sq+1)*(sq+1);
    		
    		p=num-a;
    		q=b-num;
    		
    		if(p<q)
    		{
    			cout<<"-"<<p<<endl;
    		}
    		else
    		{
    			cout<<"+"<<q<<endl;
    		}
    	}
    }
    return 0;
}
