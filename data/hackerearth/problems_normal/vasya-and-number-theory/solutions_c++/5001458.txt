#include <bits/stdc++.h>
using namespace std;

const int MXN = 1e3 + 9;

int maxSize = 1;
int dp[MXN];
int A[MXN];

int main (){
    int N; cin >> N;
    for(int i = 1;i <= N;++i)
        cin >> A[i];
    sort(A + 1, A + N + 1);
    for(int i = 1;i <= N;++i){
        dp[i] = 1;
        for(int j = 1;j < i;++j){
            if(A[i] % A[j] == 0){
                // Share a common prime which divides all of them
                dp[i] = max(dp[i],dp[j] + 1);
            }
        }
        maxSize = max(maxSize,dp[i]);
    }
    cout << ((maxSize < 2) ? -1 : maxSize) << "\n";
    return 0;
}
