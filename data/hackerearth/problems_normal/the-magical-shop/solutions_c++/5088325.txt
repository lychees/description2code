#include <iostream>
using namespace std;

int main()
{
    long long int n,m;
    unsigned long long int ans=0;
    cin>>n>>m;
    string s;
    cin>>s;
    long long int len=s.size();
    for(long long int i=0;i<len;i++){
    	if(s[i]=='1'){
    		ans=(ans+n)%m;
    	}
    	
    	n=(n*n)%m;
    		
    	
    
    }
    cout<<ans<<endl;
   
    
    
    return 0;
}
