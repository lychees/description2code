from collections import Counter


def sgn(x):
    if x < 0:
        return -1
    if x == 0:
        return 0
    return 1


rr = raw_input
N = int(rr())


for i in xrange(N):
    A = Counter(rr())
    B = Counter(rr())
    C = [sgn(A[i]-B[i]) for i in 'abcdefghijklmnopqrstuvwxyz']
    if 1 not in C and -1 in C:
        print "You lose some."
    elif -1 not in C and 1 in C:
        print "You win some."
    else:
        print "You draw some."
