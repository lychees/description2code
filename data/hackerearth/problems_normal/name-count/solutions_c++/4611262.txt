#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;

long long modulo(long long a,long long b,long long c)
{
	long long x=1,y=a;
	while(b>0)
	{
		if(b%2==1)
		{
			x=(x*y)%c;
		}
		y=(y*y)%c;
		b/=2;
	}
	return x%c;
}

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long n,k;
		scanf("%lld%lld",&n,&k);
		long long i,j,p,q,ans;
		j=k-n;
		ans=1;
		for(i=0;i<n;i++)
		{
			ans=ans*(k-i)%mod;
			ans%=mod;
		}
		printf("%lld\n",ans);  
	}
	
	return 0;
}
