#include <bits/stdc++.h>
using namespace std;
 
int main()
{
	int t; cin >> t; 
	while(t--)
	{
		string s; cin >> s;
		bool flag=0;
		if(s.length()>9)
		{
			cout << "YES" << endl;
			flag=1;
		} 
		else
		{
			int l = s.length();
			bool flag = false;
			for(int i=1; i<l;i++)
			{
				for(int j=i+1;j<l;j++)
				{
					for(int k=j+1;k<l;k++)
					{
		                    string a = s.substr(0, i);
		                    string b = s.substr(i, j-i);
		                    string c = s.substr(j, k-j);
		                    string d = s.substr(k, s.length()-k);
		                    if(a!=b && a!=c && a!=d && b!= c && b!=d && c!= d )
		                    {
		                    	cout << "YES" <<endl;
		                    	flag = true;
		                    	break;
		                    }
						}
						if(flag) break;
					}
					if(flag) break;
				}
			if(!flag) cout << "NO" << endl;
			}
		}
	}
