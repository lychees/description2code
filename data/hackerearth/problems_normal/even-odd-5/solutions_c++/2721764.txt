#include<iostream>
using namespace std;
main()
{
 long long int r,c,eve,odd,z,t;
cin>>t;
for(int z=0;z<t;z++)
{eve=0;
odd=0;
cin>>r>>c;
if(r<=c)
{
if(r%2==0)
{
odd=(r*r)/4;
eve=(r*(2+r))/4;
}
else
{
odd=(((r+1))*(r+1))/4;
r=(r-1)/2;
eve=r*(r+1);
}
}
else{
long long int a;
a=c;
if(a%2==0)
{
odd=(a*a)/4;
eve=(a*(a+2))/4;
}
else
{
odd=((a+1)*(a+1))/4;
a=(a-1)/2;
eve=a*(a+1);
}
int b=r-c;
if(c%2==0)
{
if(b%2==0){
    odd=odd+(b*c)/2;
    eve=eve+(b*c)/2;}
else{
 odd=odd+((b+1)*c)/2;
 eve=eve+((b-1)*c)/2;}
 }
 else{
if(b%2==0){
    odd=odd+(b*c)/2;
    eve=eve+(b*c)/2;}
    else{
 eve=eve+((b+1)*c)/2;
 odd=odd+((b-1)*c)/2;}
}
}
cout<<eve<<" "<<odd<<endl;
}
}
