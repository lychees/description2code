#include<iostream>
using namespace std;
#define M 1000000007
typedef long long ll;
ll ar[1026][1026];
int main()
{
	ll tc;
	for(ll i=0;i<1026;i++)
		for(ll j=0;j<1026;j++)
			ar[i][j]=0;
	ar[0][0]=1;
	for(ll i=0;i<1025;i++)
		for(ll j=0;j<1025;j++)
		{
			ar[i+1][j]=(ar[i+1][j]+ar[i][j])%M;
			ar[i][j+1]=(ar[i][j+1]+ar[i][j])%M;
			ar[i+1][j+1]=(ar[i+1][j+1]+ar[i][j])%M;
		}
		
	for(ll i=1;i<1026;i++)
	{
		ar[i][1025]++;
		ar[1025][i]++;
	}
	
	/*for(int i=0;i<1026;i++)
	{
		for(int j=0;j<2;j++)
			cout<<ar[i][j]<<" ";
		cout<<endl;
	}*/
	//ar[0][0]=0;
	cin>>tc;
	while(tc--)
	{
		ll a,b,c,d;
		cin>>a>>b>>c>>d;
		cout<<ar[c-a][d-b]<<endl;
	}
	return 0;
}
