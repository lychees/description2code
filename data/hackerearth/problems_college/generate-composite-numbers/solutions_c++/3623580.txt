#include<iostream>
#include<cstdio>
#include<cmath>
#define MAX 10010
using namespace std;
bool is_prime[10010];
void sieve(void)
{
    int i,j;
    is_prime[0]=is_prime[1]=false;
    int sq=sqrt(10010);
    for(i=4;i<10010;i+=2)
    {
        is_prime[i]=true;
    }
    for(i=3;i<sq;i++)
    {
        if(is_prime[i]==false)
        {
//            for(j=i+i;j<=MAX;j+=2*i)
            for(j = i+i; j <= MAX; j += i)
            {
                is_prime[j]=true;
            }
        }
    }
}
int main()
{
    sieve();
    int n,b;
    scanf("%d",&n);
    for(int i=0;i<=n;i++)
    {
        if(is_prime[i]==true)
        {
            if(n==i) printf("%d\n",i);
            else printf("%d ",i);
        }
    }
    //scanf("%d",&b);
    return 0;
}
