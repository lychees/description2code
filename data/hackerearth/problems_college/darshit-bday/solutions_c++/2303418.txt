#include <iostream>
using namespace std;

int main()
{
    long long int t,b,w,x,y,z;
    cin>>t;
    while(t!=0)
    {
    	cin>>b;
    	cin>>w;
    	cin>>x>>y>>z;
    	long long int res=x*b+y*w;
    	//cout<<res;
    	if((x*b+z*w+x*w)<res)
    	  res=x*b+z*w+x*w;
    	if((y*w+z*b+y*b)<res)
    	  res=y*w+z*b+y*b;
    	cout<<res<<endl;
    	--t;
    }
    return 0;
}
