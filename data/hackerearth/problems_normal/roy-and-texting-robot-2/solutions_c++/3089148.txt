#include<iostream>
#include<algorithm>
#include<string.h>
#include<map>
using namespace std;

map<char,pair<int,int> >mp;
int dig[10],l[30],key[30],curr_idx=1;
string s;

void calc() {
    int cnt,idx=0;
    for(int i=0;i<=9;i++) {
        if(i==1 || i==7 || i==9) dig[i]=5;
        else if(i==0) dig[i]=2;
        else dig[i]=4;
    }
    for(int i=2;i<=9;i++) {
        cnt=3;
        if(i==7 || i==9) cnt=4;
        for(int j=1;j<=cnt;j++) {
            key[idx]=i;
            l[idx++]=j;
        }
    }
    mp['_']=make_pair(0,1);
    mp['.']=make_pair(1,1);
    mp[',']=make_pair(1,2);
    mp['?']=make_pair(1,3);
    mp['!']=make_pair(1,4);
}

int main()
{
    int t;
    cin>>t;
    calc();
    while(t--) {
        int i,j,n,m,temp,ans=0;
        curr_idx=1;
        cin>>s;
        for(i=0;i<s.length();i++) {
            if(s[i]-'0'>=0 && s[i]-'0'<=9) {
                if(curr_idx!=s[i]-'0') ans++,curr_idx=s[i]-'0';
                ans+=dig[s[i]-'0'];
            } else if(s[i]-'a'>=0 && s[i]-'a'<26) {
                if(curr_idx!=key[s[i]-'a']) ans++,curr_idx=key[s[i]-'a'];
                ans+=l[s[i]-'a'];
            } else {
                if(curr_idx!=mp[s[i]].first) ans++,curr_idx=mp[s[i]].first;
                ans+=mp[s[i]].second;
            }
        }
        cout<<ans<<"\n";
    }
    return 0;
}
