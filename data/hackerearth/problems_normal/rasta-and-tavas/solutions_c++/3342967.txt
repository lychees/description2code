#include<bits/stdc++.h>
#define _ ios_base::sync_with_stdio(false);cin.tie(0);
const int maxn = 1E6+10;
using namespace std;

int ar[maxn];
int main()
{
    int n,k;long long int sum=0;
    cin>>n>>k;
    for(int i=2;i<=n;i++)
    {
        if(ar[i]==0)
        {
            for(int j=i;j<=n;j+=i)
            {
                ar[j]+=i;
            }
        }
    }
    for(int i=1;i<=n;i++)
    {
        if(ar[i]==k)
            sum++;
    }
    cout<<sum<<'\n';
    return 0;
}
