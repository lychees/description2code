#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    long long x,f,sum;
    cin>>t;
    while(t--)
    {
    	scanf("%lld",&x);
    	sum=0;
    	while(x){
    	if(x%2==1)
    	f=x/2+1;
    	else
    	f=x/2;
    	sum+=f*f;
    	x-=f;
    	
    }
    printf("%lld\n",sum);
    }
    return 0;
}
