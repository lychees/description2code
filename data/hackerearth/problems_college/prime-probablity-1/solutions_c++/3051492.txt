#include<stdio.h>
 
long long primes[1000001]={0},sum[1000001];
 
int main(void)
{
	long long t,i,j,count;
	scanf("%lld",&t);
	primes[0]=0;
	primes[1]=0;
		for( i=2;i<=1000000;i++)
	primes[i]=1;
	for(i=2;i<1001;i++)
	  {
	  	 if(primes[i]==1)
	  	   for(j=2;i*j<1000001;j++)
	  	     primes[i*j]=0;
	  }
	sum[0]=0;
	sum[1]=0;
	for(i=2;i<1000001;i++)
	  {
	  	
	  	 sum[i]=sum[i-1]+primes[i];
	    
	  }
	while(t--)
	  {
	  	scanf("%lld",&i);
	  	scanf("%lld",&j);
	  	count=0;
	  	if(primes[i]==1)
	  	  count=1;
	  	count=count+sum[j]-sum[i];
	  	printf("%.6f\n",(float)count/(j-i+1));
	  }
	return 0;  
}
