#include <iostream>
using namespace std;
int arrSell [1001];
int arrCost [1001];
int arrProfit[1001];

int arrDP[10001][1001];
int main()
{
    int T ; 
    cin >> T;
    while(T--){
    	int N,total;
    	cin >> N;
    	cin >> total;
    	
    	for (int i = 0; i<N;i++){
    		cin >> arrSell[i];
    		cin >> arrCost[i];
    		arrProfit[i] = arrSell[i] - arrCost[i];
    	}
    	for(int i = 0 ; i < total+1; i++){
    		for(int j = 0; j<=N; j++){
    			if (i==0 || j ==0){
    				arrDP[i][j] =0;
    			} else if (arrCost[j-1]<=i) {
    				if (arrProfit[j-1]+ arrDP[i-arrCost[j-1]][j-1] > arrDP[i][j-1]){
    					arrDP[i][j] = arrProfit[j-1]+ arrDP[i-arrCost[j-1]][j-1];
    			
    				}else {
    					arrDP[i][j] = arrDP[i][j-1];
    				
    				}
    			}else{
    				arrDP[i][j] = arrDP[i][j-1];
    			}
    		}
    	}
    	int ans = arrDP[total][N];
    	int i = 0;
    	for ( i = total ; i >= 0; i--){
    		 
    		if (arrDP[i][N] < ans) 
    			break;
    		ans =  arrDP[i][N];
    		
    	}
    	cout <<i+1 <<' '<< (total + arrDP[total][N])<<endl;
    }
    
    return 0;
}
