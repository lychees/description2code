#include<iostream>
using namespace std;
int gcd(int a,int b)
{
    if(b==0)return a;
    else
    return gcd(b,a%b);
    }
int main()
{
    int t,a,b,c;
    scanf("%d",&t);
    while(t--)
    {
              scanf("%d%d%d",&a,&b,&c);
              if(gcd(a,b)==1 && gcd(b,c)==1 && gcd(a,c)==1 && ((a*a==b*b+c*c)||(b*b==a*a+c*c)||(c*c==a*a+b*b)))
              printf("YES\n");
              else
              printf("NO\n");
              }
                 return 0;
    }
