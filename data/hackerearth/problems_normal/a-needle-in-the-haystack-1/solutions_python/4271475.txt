n=int(raw_input().strip(" "))
while(n>0):
	n-=1
	s=raw_input() 
	st=raw_input() 
	temp=s 
	if s in st: 
		print "YES"
		continue
	cnt=0
	flag=0
	for i in range(0,len(st)): 
		if st[i] in s: 
			cnt+=1
			s=s.replace(st[i],'',1) 
		else:
			cnt=0 
			s=temp	
		if cnt==len(temp):
			print "YES"
			flag=1
			break
	if(flag==0):
		print "NO"
