#include <iostream>
#include <string>

using namespace std;

int main()
{	
	int t;
	string s;
	cin>>t;
	getline(cin,s);
	for(int it=1;it<=t;it++){
		string str;
		getline(cin,str);
		int sum=0,len=str.length();
		for(int i=0;i<len;i++)
			sum+=(int)str[i];
		if((sum/len)%2==0)
			cout<<str<<endl;
		else{
			for(int i=len-1;i>=0;i--)
				cout<<str[i];
			cout<<endl;
		}
	}
}
