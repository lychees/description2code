const int MOD=1e9+7;

int main() {

    int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        int a[n];
        for (int i = 0; i < n; ++i) {
            cin>>a[i];
        }
        vector<ll> dp(MAXN);
        dp[0]=0ll;
        for(int i=1;i<=n;i++){
            if(i==1)
                dp[i]=a[0];
            else
                dp[i]=max(dp[i-1],dp[i-2]+a[i-1]);
        }
        cout<<dp[n]<<"\n";
    }
