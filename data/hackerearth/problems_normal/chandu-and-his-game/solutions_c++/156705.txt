#include<stdio.h>
#include<set>
using namespace std;
int main(){
	int n,k;
	scanf("%d%d",&n,&k);
	int x[k],y[k],i;
	long long int sum = 0;
	set<int> moves;
	for(i=0;i<k;i++){
		scanf("%d",&x[i]);
	}
	for(i=0;i<k;i++){
		scanf("%d",&y[i]);
	}
	for(i=0;i<k;i++){
		if((x[i]==1&&y[i]==1)||(x[i]==1&&y[i]==n)||(x[i]==n&&y[i]==n)||(x[i]==n&&y[i]==1))
			continue;
		moves.insert(x[i]+y[i]-2);
		moves.insert(x[i]-1+n-y[i]);
		moves.insert(n-x[i]+y[i]-1);
		moves.insert(n-x[i]+n-y[i]);
		sum = sum + *moves.begin();
		moves.clear();
	}
	printf("%lld",sum);
	return 0;
}
