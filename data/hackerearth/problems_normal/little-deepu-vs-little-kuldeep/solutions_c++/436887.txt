#define ll long long int

using namespace std;

ll MOD = 1000000007;

ll fact[2000002];
ll infact[2000002];

long long inv(long long a,long long b)
{
    long long t,q,b0 = b;
    long long x0 = 0, x1 = 1;
    if (b == 1) return 1;
    while (a > 1) {
        q = a/b;
        t = b;
        b = a%b;
        a = t;
        t = x0;
        x0 = x1 - q*x0;
        x1 = t;
    }
    if (x1 < 0) x1 += b0;
    return x1;
}

int main()
{
        int i;
        fact[0] = 1;
        infact[0] = 1;
        for (i = 1; i <= 2000000; i++) {
            fact[i] = (i*fact[i-1])%MOD;
            infact[i] = (inv(i,MOD)*infact[i-1])%MOD;
        }
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        cout << (((fact[2*n]*infact[n])%MOD)*infact[n  + 1])%MOD << endl;
    }
    return 0;
}
