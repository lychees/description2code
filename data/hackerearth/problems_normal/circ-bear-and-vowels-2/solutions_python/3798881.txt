t=int(raw_input())
for i in range(t):
    s=raw_input()
    v=0
    c0=0
    c=0
    for a in s:
        if (a=='a' or a=='i' or a=='o' or a=='u' or a=='e' or a=='y'):
            v+=1
            c0=0
        else:
            c+=1
            c0+=1
            if c0==3:
                break
    if(c0==3):
        print("hard")
    elif(c>v):
        print("hard")
    else:
        print("easy")
