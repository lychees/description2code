#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;

int main()
{
    int e,o,t;cin>>e>>o>>t;
    int n;cin>>n;
    int a[n+4],i,ans=0;
    for(i=1;i<=n;i++){
    	cin>>a[i];
    }
    sort(a+1,a+n+1);
    //for(i=1;i<=n;i++)	cout<<a[i]<<" ";
    for(i=2;i<=n-1;i++){
    	ans+=abs(a[i]-a[i-1]);
    	//cout<<ans<<" ";
    }
    if(o>t)	swap(o,t);
    ans+=abs(a[1]-o);
    ans+=abs(a[n]-t);
    int mid1;
    if(n%2)		mid1=(n+2)/2;
    else		mid1=(n+2)/2;
    if(n%2==0)
    	ans+=abs(e-a[mid1]);
    else
    	ans+=abs(min(e-a[mid1+1],e-a[mid1-1]));
    ans+=abs(e-a[mid1]);
    cout<<ans<<endl;
    return 0;
}
