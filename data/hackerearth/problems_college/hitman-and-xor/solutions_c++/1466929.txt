#include <iostream>
#include <stdio.h>

using namespace std;

typedef long long int Lint;

int main()
{
	Lint t, n, i, oc, ec, v;
	
	scanf("%lld", &t );
	
	while( t-- ) {
			
		scanf("%lld", &n );
		oc = ec = 0;
		
		for( i = 0;i < n;i++ ) {
			scanf("%lld", &v );
			if( v & 0x1 )
				oc++;
			else
				ec++;
		}
		
		printf("%lld\n", oc*ec );
		
	}
	return 0;
}
