#include <bits/stdc++.h>
using namespace std;
int main()
{
    int T;
    cin>>T;
    while(T--){
       int a,b;
       cin>>a>>b;

       if(b==0)
        cout<<a<<endl;
      else if(a==0){
        if(b%2==0)
            cout<<"0"<<endl;
        else
            cout<<"1"<<endl;
       }
       else if(a==1){
        if(b%2==0)
            cout<<"1"<<endl;
        else
            cout<<"0"<<endl;

       }
       else{
            if(b%2==0)
            cout<<"0"<<endl;
        else
            cout<<"1"<<endl;
       }
    }
    return 0;
}
