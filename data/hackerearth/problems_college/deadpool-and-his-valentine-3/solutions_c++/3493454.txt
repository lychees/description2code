#include <iostream>
using namespace std;

int main()
{
    int k, m, n, t;
    cin >> k >> m;
    int states[100];
    for (int i = 0; i < 100; i++) {
        states[i] = 0;
    }
    while (k--) {
        cin >> n;
        for (int i = 0; i < n; i++) {
            cin >> t;
            states[t - 1]++;
        }
    }
    int state = 1;
    for (int i = 0; i < m; i++) {
        if (states[i] == 0) {
            state = 0;
            break;
        }
    }
    if (state == 0) {
        cout << "NO";
    }
    else {
        cout << "YES";
    }
    return 0;
}
