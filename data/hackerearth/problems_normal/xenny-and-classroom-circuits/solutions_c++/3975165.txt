#include <bits/stdc++.h>
using namespace std;
char a[2000000];
int main() {
	// your code goes here
	long t,n,e=0,o=0;
	cin>>t;
	
for(int j=0;j<t;j++)
{
	cin>>n;
	n=2*n;
	e=0;
	o=0;
	for(long i=0;i<n;i++)
	{
		cin>>a[i];
		if(i%2==0 && a[i]=='G')
		e++;
		else if(i%2!=0 && a[i]=='G')
		o++;
	}
	if(o>e)
	cout<<e<<endl;
	else
	cout<<o<<endl;
}
	return 0;
}
