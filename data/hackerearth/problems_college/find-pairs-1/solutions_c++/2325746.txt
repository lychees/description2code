#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long int n;
        cin>>n;
        long int a[1000005];
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
        }
        long int c=0;
        for(int i=0;i<n;i++)
        {
            for(int j=i;j<n;j++)
            {
                if(j>=i&&a[i]==a[j])c++;
            }
        }
        cout<<c<<endl;
    }
    return 0;
}
