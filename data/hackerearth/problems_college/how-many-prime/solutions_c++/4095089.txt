#include <bits/stdc++.h>
using namespace std;
int cprime(int n)
{
	int ff=0;
	for(int i=2;i<=sqrt(n)&&ff==0;i++)
	{
		if(n%i==0)
		ff=1;
	}
	if(ff==0)
	return 1;
	else
	return 0;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int f=0;
    	for(int i=(n/2);i>=2&&f==0;i--)
    	{
    		if(i!=(n-i))
    		{if(cprime(i)==1)
    		{
    			if(cprime(n-i)==1)
    			{
    				cout<<i<<' '<<n-i<<endl;
    				f=1;
    			}
    		}}
    	}
    	if(f==0)
    	cout<<"-1 -1"<<endl;
    }
    return 0;
}
