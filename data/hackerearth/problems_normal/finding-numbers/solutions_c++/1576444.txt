#include <iostream>
#include<math.h>
 
using namespace std;

int main()
{
	
	long long int i,j,t,z,m,n,c;
	
	scanf("%lld",&t);
	while(t--)
	{
		
		scanf("%lld",&z);
		scanf("%lld",&m);
		scanf("%lld",&n);
		c=0;
		for(j=1;j<=sqrt(z);j++)
		{
			if(z%j==0)
			{
				if(j<=m && (z/j)<=n)
				c++;
				
				if(j<=n && (z/j)<=m)
				c++;
				
				
				if(j==(z/j))
				c--;
				
				
			}
			
			
			
		}
		printf("%lld\n",c);
		
		
	}

	
	
    return 0;
}
