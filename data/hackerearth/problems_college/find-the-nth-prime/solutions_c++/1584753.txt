#include <iostream>
#include<cmath>
using namespace std;
int p(int n)
{
	int i,flag=0;
	for(i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
		{
	    flag=1;
		break;
        }
	}
	if(flag==1)
	return false;
   return true;
}
int main()
{
   int n,count=1,i=2;
   cin>>n;

 while(count!=n)
   {
   	i++;
   	if(p(i))
   	count++;
   	}

   cout<<i;
    return 0;
}
