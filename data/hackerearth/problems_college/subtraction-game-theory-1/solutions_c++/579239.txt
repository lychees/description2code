#include<bits/stdc++.h>
using namespace std;
#define scan(x) scanf("%d",&x)
int main(){

	int t;
	scan(t);
	while(t--){
		int n,k;
		scan(n),scan(k);
		if(n==1&&k==1){
			
				printf("Nandu\n");
				continue;}
		if((k+2)>n){
			printf("Rani\n");
		}else{
			n-=(k+2);
			n%=(k+1);
			if(!n){
				printf("Nandu\n");
			}else{
				printf("Rani\n");
			}
		}
	}
	return 0;
}
