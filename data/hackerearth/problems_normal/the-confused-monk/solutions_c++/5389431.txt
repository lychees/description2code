#include<bits/stdc++.h>
using namespace std;

#define m 1000000007

long long gcd(long long a,long long b)
{
    if(b==0)
        return a;

    else
        return gcd(b,a%b);
}


int main()
{
    long long n;
    cin>>n;

    long long a[n],i,j,k,g,f=1,ans;

    for(i=0; i<n; i++)
    {
        cin>>a[i];
        f=((f%m)*(a[i]%m))%m;
    }

    g=a[0];

    for(i=0; i<n-1; i++)
    {
        g=gcd(a[i],a[i+1]);

        if(g==1)
            break;

    }


    ans=1;

    for(i=0; i<g; i++)
    {
        ans=((ans%m)*(f%m))%m;
    }

cout<<ans<<"\n";
    return 0;
}
