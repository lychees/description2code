t = int(raw_input())
for _ in range(t):
    n = int(raw_input())
    dp = [[0 for x in range(3)] for x in range(n)]
    a, b, c = map(int, raw_input().split())
    dp[0][0] = a
    dp[0][1] = b
    dp[0][2] = c
    for i in range(1, n):
        a, b, c = map(int, raw_input().split())
        dp[i][0] = min(dp[i-1][1], dp[i-1][2])+a
        dp[i][1] = min(dp[i-1][0], dp[i-1][2])+b
        dp[i][2] = min(dp[i-1][0], dp[i-1][1])+c
    # print dp
    print(min(dp[n-1][0], dp[n-1][1], dp[n-1][2]))
