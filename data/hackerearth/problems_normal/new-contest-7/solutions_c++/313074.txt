using namespace std;
long double a[1009][1009];
bool done[1009][1009];
int s,n,m,k;
long double solve(int sd,int md) {
	if(done[sd][md])
		return a[sd][md];
	
	done[sd][md]=1;
	
	if(sd==n) {
		if(md>=k)
			return a[sd][md] = 1;
		else
			return a[sd][md] = 0;
	}
	if(md==k){
		return a[sd][md] = 1;
	}
	
	long double ans=0;
	ans = (long double)(m-md)/(long double)(s-sd) *(solve(sd+1,md+1));
	ans += (long double)((s-sd) - (m-md))/(long double)(s-sd) * (solve(sd+1,md));
	return a[sd][md]=ans;
	
}

	S(t);
	while(t--)
	{
		CLR(done);
		S(s);S(n);S(m);S(k);
		//cout<<s<<n<<m<<k<<endl;
		cout.precision(6);
		cout.setf( std::ios::fixed, std:: ios::floatfield );
		cout<<solve(0,0)<<endl;

	}
	return 0;
}
