#include<bits/stdc++.h>
using namespace std;
long long gcd(long long x,long long y)
{
	if(y==0)
	return x;
	else
	return gcd(y,x%y);
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long x,y,n;
		scanf("%lld %lld %lld",&x,&y,&n);
		long long answ=0;
		answ+=x+y;
		
		long long ans1=0;
		int ind=-1;
		long long j,g;
		g=gcd(x,y);
		for(int i=3;i<=n;i++)
		{
			j=abs(x-y);
			if(x==y)
			{
				
				answ-=(x+y);
				ans1+=x+y;
				ind=i-3;
				//n+=2;
				break;
			}
			answ+=j;
			x=y;
			y=j;
		}
	
		if(ind==-1)
		printf("%lld\n",answ);
		else
		{
			n-=(ind);
			long long n1,n2;
			n1=n/3;
			n2=n%3;
			ans1=(ans1*n1);
			if(n2==0)
			{
				
			}
			else if(n2==1)
			{
				ans1+=(g);
			}
			else if(n2==2)
			{
				ans1+=(2*g);
			}
			answ+=ans1;
			printf("%lld\n",answ);
		}
	}
	return 0;
}
