#include <iostream>
#include <math.h>
#include <algorithm>
using namespace std;

int main(void)
{
    int T, a, b, ans;
    int rsum[6] = {0, 1, 2, 6, 6 , 3};
    cin>>T;
    while(T--)
    {
        cin>>a>>b;
        ans = 0;
        for(int i = a; i <= b; i++)
        {
            if(i < 6)
                ans += rsum[i];
            else
                ans += 9;
        }
        cout<<ans<<endl;
    }
    return 0;
}
