def indexes(a):
	return "abcdefghijklmnopqrstuvwxyz".index(a)


input_string=list(raw_input())
m=int(raw_input())
alpha_indexing=[[] for i in range(0,26)]
i=0
for k in input_string:
	alpha_indexing[indexes(k)].append(i)
	i+=1

while m>0:
	i,sep,alpha=raw_input().partition(' ')
	input_string[alpha_indexing[indexes(alpha)][int(i)-1]]='*'
	alpha_indexing[indexes(alpha)].pop(int(i)-1)
	m-=1
output=""
for i in input_string:
	if i != '*':
		output+=i
		
print output
