#include<bits/stdc++.h>
using namespace std;

#define mod 1000000007
typedef set<string> ss;
typedef vector<int> vs;
typedef map<int,char> msi;
typedef pair<long long int,long long int> pa;
typedef long long int ll;

priority_queue<pa> q;
ll dist[505],n,i,a,b,w,u,Q,x;
vector<pa> v[505];
bool vis[505];
int main()
{
	//freopen("i.in", "r", stdin);
  	//freopen("o.out", "w", stdout);
  	for(i=0;i<=500;i++)
  	dist[i]=INT_MAX;
  	cin>>n;
  	for(i=0;i<n;i++)
  	{
  		cin>>a>>b>>w;
  		v[a].push_back(make_pair(b,w));
  		v[b].push_back(make_pair(a,w));
  	}
  	cin>>u;
  	q.push(make_pair(0,u));
  	dist[u]=0;
  	while(!q.empty())
  	{
  		u=q.top().second;
  		q.pop();
  		if(vis[u])
  		continue;
  		vis[u]=1;
  		int s=v[u].size();
  		for(i=0;i<s;i++)
  		{
  			b=v[u][i].first; w=v[u][i].second;
  			if(dist[b]>dist[u]+w)
  			{
  				dist[b]=dist[u]+w;
  				q.push(make_pair(-dist[b],b));
  			}
  		}
  	}
  	cin>>Q;
  	while(Q--)
  	{
  		cin>>x;
  		if(dist[x]==INT_MAX)
  		cout<<"NO PATH\n";
  		else
  		cout<<dist[x]<<"\n";
  	}
	return 0;
}
