#include<bits/stdc++.h>

using namespace std;
int main(){
	string s;
	cin >> s;
	int count = 0;
	if(s[0]!='_'){
		s[0] -= 32; 
	}
	for(int i = 0 ; i < s.length();i++){
		if(s[i] == '_' && i+1!=s.length()){
			if(s[i+1]!='_')
			s[i+1] -= 32;
		}
	}
	cout << s;
	//cout << cos(0);
	return 0;
}
