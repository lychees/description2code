#include <iostream>
#include <stdio.h>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int n,q,m,x,y;
    scanf("%d",&n);
    vector<int> ar[10];
    for(int i=1;i<=n;i++){
    	x=i;
    	while(x!=0){
            y=x%10;
    		if(binary_search(ar[y].begin(),ar[y].end(),i)==false) ar[y].push_back(i);
    		x/=10;
    	}
    }
    vector<int>::iterator it;
    /*for(int i=0;i<10;i++){
    	cout<<i<<"\t";
        for(it=ar[i].begin();it!=ar[i].end();it++){
            cout<<*it<< " ";
        }
        cout<<endl;
    }*/
    int t,cnt;
    scanf("%d",&q);
    while(q--){
    	scanf("%d",&m);
    	cnt=0;
    	bool se[n];
    	for(int i=0;i<n;i++) se[i]=false;
    	while(m--){
            scanf("%d",&t);
            //cout<<t<<" ";
            for(it=ar[t].begin();it!=ar[t].end();it++){
                if(se[(*it)-1]==false) cnt++;
                 se[(*it)-1]=true;
            }
    	}
    	printf("%d\n",cnt);
    //	for(int i=0;i<n;i++) cout<<se[i]
    }
    return 0;
}
