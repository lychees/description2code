#include<stdio.h>
int main()
{
	int  x,y,ntc;
	scanf("%d",&ntc);
	while(ntc--)
	{
		
		scanf("%d%d",&x,&y);
		if(x!=y&&x-2!=y)
		{
			printf("No Number\n");
		}
		else
		{
			if(x==y)
			{
				if(x%2==1)
				printf("%d\n",2*x-1);
				else
				printf("%d\n",2*x);
			}
			else
			{
				if(x%2==0)
				printf("%d\n",x+y);
				else
				printf("%d\n",x+y-1);
			}
		}
	}
}
