#include <bits/stdc++.h>
#define pb push_back
#define sort(v) sort(v.begin(),v.end())

using namespace std;

int main() {
	long int t,n,x,sev=0,sod=0;
	vector<long int>od;
	vector<long int>ev;
	
	cin>>t;
	while(t--)
	{
		cin>>n;
		if(n&1){
		od.pb(n);
		sod=sod+n;
		}
		
		else
		{
			ev.pb(n);
			sev=sev+n;
		}
	}
	sort(od);
	sort(ev);
	for(x=0;x<ev.size();x++)
	cout<<ev[x]<<" ";
	
	cout<<sev<<" ";
	
	for(x=0;x<od.size();x++)
	cout<<od[x]<<" ";
	
	cout<<sod<<" ";
	
	
	// your code goes here
	return 0;
}
