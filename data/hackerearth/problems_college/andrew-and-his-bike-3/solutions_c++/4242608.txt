#include <iterator>
#include <stdio.h>
#include <list>
#include <vector>
#include <iostream>
#define pb push_back
using namespace std;
int count[100001]={0};
int ans[100001]={0};
int vis[100001]={0};
list<int> adj[100001];
int func(int i)
{
	if(vis[i])
		return 0;
	vis[i]=1;
	if(ans[i]!=0)
		return ans[i];
	if(count[i]==1&&i!=1)
	{
		ans[i]=1;
		return 1;
	}
	ans[i]=1;
	list<int>::iterator it;
	for(it=adj[i].begin();it!=adj[i].end();it++)
	{
		if(vis[*it])
			;
		else if(ans[*it]!=0)
			ans[i]+=ans[*it];
		else 
			ans[i]+=func(*it);
	}	
	return ans[i];
}
int main()
{
	int n,i,j,a,b;
	scanf("%d",&n);
	for(i=1;i<n;i++)
	{
		scanf("%d%d",&a,&b);
		adj[a].pb(b);
		adj[b].pb(a);
		count[a]++;
		count[b]++;
	}
	func(1);
	//	for(i=1;i<=n;i++)
	//		printf("%d ",count[i]);
	ans[1]--;
	for(i=1;i<=n;i++)
		printf("%d ",ans[i]);
	return 0;
}
