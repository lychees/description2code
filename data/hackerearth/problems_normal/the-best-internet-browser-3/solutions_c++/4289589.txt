#include<iostream>
#include<string.h>
using namespace std;
int main(){
 int test,Count,i;
    cin>>test;
    while(test--)
    {
        string S;
        cin>>S;
        Count=0;
        for(i=4;i<S.size()-4;i++)
        {
            if(S[i]!='a' && S[i]!='i' && S[i]!='o' && S[i]!='u' &&S[i]!='e')
                Count++;
        }
        cout<<Count+4<<"/"<<S.size()<<endl;
    }
    return 0;

}
