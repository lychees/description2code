#include<bits/stdc++.h>
using namespace std;
int dp[1007][1007];
int main()
{
    int t;
    cin>>t;

    while(t--)
    {
        string s;
        cin>>s;
        for(int i=0;i<s.size();++i)
        {
            dp[i][i]=1;
        }
        for(int len=2;len<=s.size();++len)
        {
            for(int i=0;i+len-1<s.size();++i)
            {
               int j=i+len-1;
               dp[i][j]= dp[i][ j-1] + dp[i+1][ j] - dp[i+1][j-1] + (s[i] == s[j])*(dp[i+1][ j-1]+1);

            }
        }
        cout<<dp[0][s.size()-1]<<endl;
        if(t)
        {
            memset(dp,0,sizeof(dp));
        }
    }
    return 0;
}
