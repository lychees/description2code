#include <iostream>
using namespace std;



int main()
{
    long long int t,n,sum,i,j,c,k;
    cin>>t;
    while(t--)
    {
    	sum=0;
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>c;
    		sum=sum+c;
    	}
    	if(sum%n==0)
    	printf("YES\n%lld 0\n",sum/n);
    	else
    	{
    		k=sum/n;
    		printf("NO\n");
    		printf("%lld %lld\n",k,sum-(k*n));
    	}
    }
}
