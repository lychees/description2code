#include <bits/stdc++.h>
using namespace std;
 int main ()
 {
     long long tc, n, m;
     cin>>tc;
    vector< pair < long long , bool > >v;
    set<long long>s;
     while(tc)
     {
         cin>>n>>m;
         long long a[n+m];
         v.clear();
         s.clear();
         for (long long i=0; i<n+m; i++)
         {
             scanf("%lld", &a[i]);
             v.push_back(make_pair(a[i], 0));
         }
         for (long long i=0; i<n; i++)
            s.insert(v[i].first);
        for (long long i=n; i<n+m; i++)
         {
             if (s.count(v[i].first)!=0)
                v[i-n].second=true;
             else
             {
                 v[i-n].second=false;
                 s.insert(v[i].first);
             }
         }
         for (long long i=0; i<m ; i++)
         {
             if (v[i].second)
                printf("YES\n");

             else
                printf("NO\n");
         }

    //     for (int i=0; i<n+m; i++)
        // {
        //     cout<<v[i].first<<" "<<v[i].second<<" "<<endl;
      //   }

         tc--;
     }
 }
