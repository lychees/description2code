#include <iostream>
using namespace std;

int main()
{	int t;
	cin>>t;
	while(t--){
		long x,y,a=0,b=1,c,s=0;
		cin>>x>>y;
		int cnt=0;
		
		
		for ( c = 0 ; ; c++ )
   {
      if ( c <= 1 )
         s = c;
      else
      {
         s = a + b;
         a = b;
         b = s;
      }
      if(s>=x && s<=y){cnt++;}
      if(s>y){break;}
   }
		
		if(cnt%2==0)
		cout<<"VALID"<<endl;
		else
		cout<<"INVALID"<<endl;
		
		
	}
    return 0;
}
