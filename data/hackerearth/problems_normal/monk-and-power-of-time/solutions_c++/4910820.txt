#include <bits/stdc++.h>
using namespace std;

int main()
{
   int i, n, num;
   cin>>n;
   		queue<int> calling, required;
   		for(i=0; i<n; i++)
   		{
   			cin>>num;
   			calling.push(num);
   		}
   		for(i=0; i<n; i++)
   		{
   			cin>>num;
   				required.push(num);
   		}
   		
   	
   		int count = 0;
   		while(!required.empty())
   		{
   			while(required.front()!=calling.front())
   			{
   				calling.push(calling.front());
   	//			printf("removed %d\n",calling.front());
   				calling.pop();
   				count++;
   			}
   		
   			count++;
   	//		printf("found %d\n",required.front());
   			required.pop();
   			calling.pop();
   			
   		}
   		cout<<count<<endl;	
   	
}
