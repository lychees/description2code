# print 'Hello World!'
import math

n = int(raw_input())
i = 0
while i < n:
    tc = float(raw_input())

    ans = math.floor(math.sqrt(tc*2) + 0.5)

    if ans % 3 == 0:
        print "SL"
    elif ans % 3 == 1:
        print "LB"
    elif ans % 3 == 2:
        print "BS"

    i += 1
