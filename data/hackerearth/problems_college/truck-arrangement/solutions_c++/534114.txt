#include <iostream>
#include <string.h>
#include <stdio.h>
#include <algorithm>

using namespace std;

//#include<stdio.h>

int main()
{
	int test;
	cin>>test;
	while(test--)
	{
		int n;
		scanf("%d",&n);
		int arr[n+10];
		int arr1[n+10];
		int index[100001]={0};
		for(int i=0; i<n; i++)
		{
			cin>>arr[i];
			arr1[i]=arr[i];
			index[arr[i]]++;
		}
		sort(arr1, arr1+n);
		int indexs[100001]={0};
		int indexe[100001]={0};

		int maxcount=0;
		int flag=0;
		int s=0;
		int e=0;
		for(int i=0; i<n; i++)
		{
			if(i>0)
			if(arr1[i]!=arr1[i-1])
			{
				indexs[arr1[i-1]]=s;
				indexe[arr1[i-1]]=e;
				s=i;
			}
			e=i;
		}
		indexs[arr1[n-1]]=s;
		indexe[arr1[n-1]]=e;
		int maxdist=0;
		int dist=0;
		for(int i=0; i<n; i++)
		{
			if(i<indexs[arr[i]])
			{
				dist=indexs[arr[i]]-i;
			}
			else
			{
				if(i>indexe[arr[i]])
				{
					dist=i-indexe[arr[i]];
				}
				else
					dist=0;
			}
			//cout<<dist<<endl;
			if(dist>maxdist)
				maxdist=dist;
		}
		printf("%d\n",maxdist);
	}
	return 0;
}
