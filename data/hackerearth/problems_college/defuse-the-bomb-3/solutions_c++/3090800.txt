#include <iostream> 
using namespace std;
 
int main()
{
     string x,n;
    int no_testcases,d,i,s,k,l,j;
    cin>>no_testcases;
    for(d=0;d<no_testcases;d++)
    { 
    	cin>>x>>n; l=n.length()-1;
    	s=0;j=l;k=0;
    
	for(i=0;i<=l;i++)
    	{
    	
        k=k+((n[i]-'0'));
        
        k=k*10; 
    	
        }
        i=0;
        k=k/10;
 
    	while(true)
    	{
            
    	while(l>=0 && x[i])
    	{
    	s=s+((x[i]-'0'));
        --l;
        s=s*10;
        ++i;
    	}
    	s=s/10; 
    	 
    	cout<<s*k<<endl;
        if(!x[i])
    	break; 
    	l=j;
    	s=0; 
    	l=j;
    	
    	i=i-l;
    	}
    }
  
}
