#include <iostream>
#include<cstring>
#include<set>
using namespace std;

int main()
{
    string s;int t; long long int n;
    
    cin>>t;
    while(t--)
    {set<char>s1;
    	cin>>s;
    	n=s.length();
    	for(int i=0;i<n;i++)
    	s1.insert(s[i]);
    	n=s1.size();
    	cout<<(n*(n+1))/2<<"\n";
    }
    return 0;
}
