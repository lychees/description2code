t = int(raw_input())
for qq in xrange(t):
    ss = raw_input()
    s = ''
    for char in ss:
        if char.isalpha() or char == '*':
            s += char
    i = 0
    j = len(s)-1
    while i <= j:
        if s[i] == s[j]:
            i += 1
            j -= 1
            continue
        else:
            if s[i] == '*':
                ans = s[j]
            else:
                ans = s[i]
            break
    print ans
