#include <iostream>
#include <vector>
#include <fstream>
#include <string>
#include <math.h>
#include <cstdlib>
#include <stdio.h>     
#include <stdlib.h> 
using namespace std;

int main()
{

   string q;
   int max=-999;
   string thenum;
   int cur;
   
   int mult=1;
   int start=0;
   
   while(cin>>q)
   {
   	mult=1;
   	start=0;
   	cur=0;
   	
   	if (q[0]=='-')
   	{
   		mult=-1;
   		start=1;
   	}
   	
	   for (int i=start;i<q.size();i++)
	   {
	   	if (q[i]!='.')
	   	cur=cur+q[i]-48;
	   }
	   
	   if (cur>max)
	   {
	   	//cout<<"cur = "<<cur<<" max = "<<max<<endl;
	   	max=cur;
	   thenum=q;
	   }
	   
   }
   cout<<thenum<<endl;
  
   
   return 0;
}
