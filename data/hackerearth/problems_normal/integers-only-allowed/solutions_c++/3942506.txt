#include <bits/stdc++.h>
using namespace std;
#define ll long long
int gcd(int a,int b)
{
	a=abs(a);
	b=abs(b);
	while(a&&b)
        a>b?a-=b:b-=a;
	return a+b;
}
int main()
{
    int n;
    cin>>n;
    int x[n+1],y[n+1];
    for(int i=0;i<n;i++)
        cin>>x[i]>>y[i];
    x[n]=x[0];
    y[n]=y[0];
    int border=0;
    ll area=0;
    for(int i=0;i<n;i++)
        area+=1ll*x[i+1]*y[i]-1ll*y[i+1]*x[i];
    for(int i=0;i<n;i++)
        border+=__gcd(abs(y[i+1]-y[i]),abs(x[i+1]-x[i]));

    cout<<(area+2-border)/2<<endl;
    return 0;
}
