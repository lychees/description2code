#include <bits/stdc++.h>
using namespace std;
   
	
int main()
{   
    
    long long int a[22], dp[10000]={0};
    for(int i=1;i<22;i++)
       a[i]=pow(i,3);
   
   
    
    dp[0]=1;
        
    for(int i=1;i<=21;i++)
      for(int j=a[i];j<10001;j++)
           dp[j]+=dp[j-a[i]];    
     int t;
    cin>>t; 
    while(t--)
    {
    	long long int n;
    	cin>>n;
    	cout<<dp[n]<<endl;
    	  	
    }
    return 0;
}
