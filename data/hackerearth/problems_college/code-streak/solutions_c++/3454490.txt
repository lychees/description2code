#include <bits/stdc++.h>
#define ll long long int
#define MOD 1000000007
#define inf 1000000007
#define pii pair<int, int>

using namespace std;

int f[1000005];
vector<pii> v;

int main()
{
    int t, n;

    cin >> t;
    while (t--) {
        cin >> n;
        v.clear();
        memset(f, 0, sizeof(f));

        for (int i = 0; i < n; i++) {
            int x, y;
            cin >> x >> y;
            if (y == 1) {
            f[x] = 1;
            }
            v.push_back(make_pair(x, y));
        }

        int i = 0, j = 0, cnt = 0;
        int mx = INT_MIN;

        while (i < n) {
            if (v[j].second == 1) {
                if (f[v[j].first] == 1) {
                    cnt++;
                    f[v[j].first] = 0;
                }

                j++;
            } else {
                if (cnt > mx) {
                    mx = cnt;
                }
                i++;
                j = i;
                cnt = 0;
            }
        }
        cout << mx << endl;
    }
    return 0;
}
