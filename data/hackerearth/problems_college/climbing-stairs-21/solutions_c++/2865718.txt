// CP template
#include <bits/stdc++.h>

using namespace std;

long long fib(int n){
	long long f[1005];
	f[0]=1; f[1]=1;
	for(int i=2; i<=n; i++){
		f[i]= f[i-2] + f[i-1];
	}
	return f[n];
}

int main()
{
	int t;
	scanf("%d",&t);
	for(int i=1;i<=t; i++){
		int n;
		scanf("%d",&n);
		printf("%lld\n",fib(n));
	}
	return 0;
}
