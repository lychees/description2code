#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

int main () {
	int t; cin >> t; while (t--) {
		long n , cnt = 0; cin >> n;

		vector < long > v;
		v.push_back(1);
		if (!(n % 2))
			cnt++;

		for(long i = 2; i <= sqrt(n); i++){
			if(n%i==0){
				v.push_back(i);
				if( i != sqrt(n))
					v.push_back(n/i);
			}
		}

		for (int i = 0; i < v.size(); i++)
			if (!(v[i] % 2))
				cnt++;

		cout << cnt << endl;

	}
	return 0;
}
