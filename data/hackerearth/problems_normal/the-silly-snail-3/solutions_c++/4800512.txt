#include<bits/stdc++.h>
using namespace std;
pair<int,int> adj[100010];
bool visited[100010];
void dfs(int s)
{

    visited[s]=1;
    cout<<s<<" ";
     if(!visited[adj[s].first])
        dfs(adj[s].first);
             if(!visited[adj[s].second])
         dfs(adj[s].second);

}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        for(int i=0;i<100010;i++)
        {
            visited[i]=0;
        }
        visited[0]=1;

        int m;
        cin>>m;
        for(int i=0;i<m;i++)
        {
            int a,l,r;
            cin>>a>>l>>r;
            adj[a]=(make_pair(l,r));
        }
        dfs(1);
        cout<<endl;
        memset(adj,0,sizeof(adj));
    }
}
