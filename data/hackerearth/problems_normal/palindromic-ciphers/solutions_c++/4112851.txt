#include <iostream>
using namespace std;

int main()
{
   string s;
   int t,i,j,f;
   long long prod;
   cin>>t;
   while(t--)
   {
   	f=0;
   	prod=1;
   	cin>>s;
   	for(i=0,j=s.size()-1;i<j;i++,j--)
   	{
   		if(s[i]!=s[j])
   		{
   			f=1;
   			break;
   		}
   	}
   	if(f)
   	{
   		for(i=0;i<s.size();i++)
   		{
   			prod*=(s[i]-96);
   		}
   		cout<<prod<<endl;
   	}
   	else
   	{
   		cout<<"Palindrome"<<endl;
   	}
   }
    return 0;
}
