#include <iostream>
#include<string>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,i,flag=0,c=0;
    char a[200]={0},s[100];
    cin>>t;
    while(t--)
    {  
    	flag=0;
    	c=0;
    	for(i=97;i<122;i++)
    	a[i]=0;
    	cin>>s;
    	for(i=0;s[i]!='\0';i++)
    	{
    		a[s[i]]++;
    	}
    	for(i=97;i<122;i++)
    	{
    		if((a[i]%2!=0)&&(c==0))
    		c=1;
    		else if(a[i]%2==0)
    		continue;
    		else 
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==1)
    	cout<<"No"<<"\n";
    	else
    	cout<<"Yes"<<"\n";
    }
    return 0;
}
