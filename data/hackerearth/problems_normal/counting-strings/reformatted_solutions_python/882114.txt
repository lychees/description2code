#! /usr/bin/python

t = int(raw_input())
for x in xrange(t):
    m = raw_input()
    l = len(m)
    c = 0
    r = 0
    for i in xrange(l):
        if (m[i] == 'a' or m[i] == 'z'):
            r += (c * (c + 1)) / 2
            c = 0
        else:
            c += 1

    r += (c * (c + 1))/2
    total = (l * (l + 1))/2
    ans = total - r
    print ans
