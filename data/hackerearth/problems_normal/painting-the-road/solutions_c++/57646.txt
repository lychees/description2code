int dp [3000];
int l[3000],r[3000],c[3000];
int main()
{
      int T,t,i,j,n,e,t1,t2,t3,pos,mini;
      si(t);
      rep(T,t)
      {
           si(n); si(e);
           rep(i,e)
	   {
	     si(l[i]); si(r[i]); si(c[i]);
	   }
           fill(dp,-1);
           dp[0]=0;
           FOR(i,1,n+1)
	   {
	      pos=-1; mini=mod;
	      for(j=0; j<e; j++)
	      {
	           if(r[j]==i)
		   {
		        if(dp[l[j]]>=0 && dp[l[j]] + c[j]<mini)
			{
			   pos=j;
			   mini=dp[l[j]]+c[j];
			}
		         
		   }
	      }
	      if(pos!=-1)
	      {
	        for(t1=l[pos]+1 ;t1<= r[pos]; t1++)
		{
		   if(dp[t1]==-1) dp[t1]=mini;
		   else if(dp[t1]>mini) dp[t1]=mini;
		}
	      }
	   }
        pil(dp[n]);
      }
      return 0;
}
