#include<iostream>
#include<cmath>
using namespace std;
void binary(int);
int main()
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		binary(n);
	}
	return 0;
}
void binary(int n)
{
	int i=0,j=0,k,flag=0,m=n;
	while(pow(2,i)<=n)
	i++;
	int *arr;
	arr=new int [i];
	for(k=0;k<i;k++)
	arr[k]=0;
	while(n>0)
	{
		arr[j]=n%2;
		n=n/2;
		j++;
	}
	for(k=j-1;k>0;k--)
	{
		if(arr[k]==1&&arr[k-1]==1)
		{
			flag=1;
			break;
		}
	}
	if(flag==0)
	cout<<m<<endl;
	else
	binary(m+1);
}
