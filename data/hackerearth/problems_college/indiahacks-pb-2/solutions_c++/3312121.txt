void revrese(char s[], char s1[])
{
    int i = 0;
    int j = 0;
    int len = strlen(s);

    for (i = len - 1; i >= 0; i--)
        s1[j++] = s[i];
    s1[j] = '\0';
}
int main()
{
    int n, i;

    char s[n][100];

    for (i = 0; i < n; i++)
        scanf("%s", s[i]);

    for (i = 0; i < n; i++) {
        int len = strlen(s[i]);
        char s1[len + 1];

        revrese(s[i], s1);
        int flag = 0;

                flag = 1;

            }
        }

    }
