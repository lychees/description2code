#include <iostream>
using namespace std;

int main()
{
	int kmax = 0, lmax = 0, temp, i, j, test, n, flag;
	int arr[12];
    cin>> test;
    while(test--)
    {
    	cin>> n;
    	if(n == 1)
    	{
    		cout<< "1" << endl;
    		continue;
    	}
    	for(i=flag=1;i <= 9;i++)
    		arr[i] = i;
    	while(flag)
    	{
    		for(i=1;i <= n;i++)
    			cout<< arr[i];
    		cout<< " ";
    		//kmax
			for(i=1, kmax=-1;i < n; i++)
				if(arr[i] < arr[i+1])
					kmax = i;
			if(kmax == -1)
				flag = 0;
			if(kmax >= 0)
			{
				//lmax
				for(i=0;i <= n; i++)
					if(arr[kmax] < arr[i])
						lmax = i;
	
				//Swaping
				temp = arr[kmax];
				arr[kmax] = arr[lmax];
				arr[lmax] = temp;
				
				//Reversing
				
				i = kmax + 1;
				j = n;
				while(i < j)
				{
					temp = arr[i];
					arr[i++] = arr[j];
					arr[j--] = temp;
				}
			}
    	}
    	cout<< endl;
    }
    return 0;
}
