
__mem_cache = {}


def ser(n):
    if n in __mem_cache:
        return __mem_cache[n]
    elif n == 1 or n == 2:
        __mem_cache[n] = n
        return __mem_cache[n]
    else:
        if n % 2 == 0:
            __mem_cache[n] = ser(n-2) + ser(n-1)
        else:
            __mem_cache[n] = ser(n-2) * ser(n-1)
        return __mem_cache[n]


t = int(input())
while t > 0:
    n = int(input())
    print(ser(n))
    t = t-1
