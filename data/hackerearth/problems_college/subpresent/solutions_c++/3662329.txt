int sub(char a[], char b[], int l1, int l2)
{
        int i,j;
        i=0,j=0;

        char ch,t;
        int flag=1;
        for(i=0 ; i<l1 ;i++ )
        {
            //cout<<'a';

            for(; j < l2;j++)
            {
                //cout<<'l';
                ch=b[j];
                if(t!=ch)
                {
                    flag=1;
                    //cout<<"hi";
                }

                else{

                    //cout<<ch<<", ";
                    j2=j+1;
                    break;
                }
            //flag=1;
            }

                return 0;
            flag=1;
        }

            return 1;
}
int main()
