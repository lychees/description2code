#include <iostream>
#include<map>
#include<iterator>
#include<string.h>
using namespace std;
 
 
int main()
{
 
   int t,i;
   char s[1001];
   cin>>t;
   while(t--)
   {   int a=0;
       int count=0;
       map<char,int> m;
       map<char,int> :: iterator it;
       cin>>s;
       for(i=0;s[i]!='\0';i++)
       {
           {
             m[s[i]]++;
           }
       }
       for(it=m.begin();it!=m.end();++it)
       {
           if(((it->second)%2)!=0)
            {
			count++;
            a=1;
			}
       }
       if(a)
       cout<<count-1<<"\n";
       else cout<<count<<"\n";
   }
    return 0;
}
