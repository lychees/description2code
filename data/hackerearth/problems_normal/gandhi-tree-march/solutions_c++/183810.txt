#include <iostream>
using namespace std;

struct st
{
	char id;
	int no;
	st *left;
	st *right;
};

int s,f,p;
char seq[10000];

void create(char str[],st* ptr,int lr)
{
	if(str[s]=='.'){if(!lr)ptr->left=NULL;else ptr->right=NULL;s++;return;}
	else if(lr==0)
	{
		ptr->left=new st;
		ptr->left->id=str[s];
		ptr->left->no=ptr->no-1;if(ptr->no-1==f){seq[p++]=str[s];}
		s+=2;
		create(str,ptr->left,0);
		create(str,ptr->left,1);s++;
	}
	else if(lr==1)
	{
		ptr->right=new st;
		ptr->right->id=str[s];
		ptr->right->no=ptr->no+1;if(ptr->no+1==f){seq[p++]=str[s];}
		s+=2;
		create(str,ptr->right,0);
		create(str,ptr->right,1);s++;
	}
}

void fre(st* nod)
{
	st *nl=NULL,*nr=NULL;
	delete nod;
	if(nod->left)nl=nod->left;
	if(nod->right)nr=nod->right;
	fre(nl);fre(nr);
}

void sort(char zxc[])
{
	for(int i=0;i<p;i++)
	{
		for(int j=0;j<p-i-1;j++)if(zxc[j]>zxc[j+1]){char temp=zxc[j];zxc[j]=zxc[j+1];zxc[j+1]=temp;}
	}
	zxc[p]='\0';
}

int main()
{
    int n,k;
    char str[10000];
    cin>>n;
    st* star;
    for(int i=0;i<n;i++)
    {
    	cin>>k;
    	cin>>str;
	s=p=0;
	f=k;
	star=new st;
	star->id=str[0];
	star->no=0;
	if(k==0){seq[p++]=str[0];}
	s+=2;
	create(str,star,0);
	create(str,star,1);
	sort(seq);
	if(!p)cout<<"Common Gandhijee!";
	else cout<<seq;
	cout<<endl;
	//fre(star);
    }
}
