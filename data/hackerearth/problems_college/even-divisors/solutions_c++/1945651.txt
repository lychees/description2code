#include <iostream>
#define ll long long
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	ll n;
    	cin>>n;
    	ll ans=0;
    	for(ll i=1;i*i<=n;i++)
    	{
    		if(n%i==0)
    		{
    			if((n/i)%2==0 && i*i!=n)
    			ans++;
    			if(i%2==0)
    			ans++;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
