#include<iostream>
#include<cstdio>
#include<cstring>

using namespace std;

char s[30001];

void check(int i){
    if('a' <= s[i] && s[i] <= 'z'){
        s[i] -= 'a' - 'A';
    }
}

int main(){
    gets(s);
    for(int i=0;i<strlen(s);i++){
        if(i == 0) check(i);
        else if(s[i-1] == ' ') check(i);
    }
    cout << s << endl;
}
