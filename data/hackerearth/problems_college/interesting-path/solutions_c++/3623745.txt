#include <iostream>
#include <bits/stdc++.h>
using namespace std ; 
int main()
{
	int t ; 
	cin >> t ; 
	while ( t--)
	{
		int n ; 
		cin >> n ; 
		int a[100000] ; 
		int i ; 
		for ( i = 0; i< n ; i++)
		{
			cin >> a[i] ; 
		}
		int b[100000] ; 
		b[0] = 0 ;
		int j ;  
		for ( i = 1 ; i< n ; i++)
		{
			b[i] = INT_MAX ; 
			for ( j = 0 ; j < i ; j++ )
			{
				if ( j + a[j]  >= i && b[j] != INT_MAX )
				{
				    
				b[i] = min( b[i] , b[j]+1) ; 
				 
			    }
			}
		}
		
		cout << b[n-1] <<  endl; 
	}
}
