'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
t = input()
for _ in range(t):
    n, b = map(int, raw_input().split())
    h = n//b
    x = h*b
    y = (h+1)*b
    if (n-x < y-n):
        print x
    else:
        print y
