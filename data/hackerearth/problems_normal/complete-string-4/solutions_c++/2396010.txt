#include<iostream>
#include<string>
#include<array>
using namespace std;

int main() {
        string str;
        int howmany,flag=0;
        cin>>howmany;
        while(howmany--) {
        		array<int,25> count={};
                cin>>str;
                for(string::iterator it=str.begin();it!=str.end();++it){
                        count[int(*it)-'a']++;
                }
                flag=1;
                for( int& x : count) {
                        if(x==0) {
                                cout<<"NO"<<endl;
                                flag=0;
                                break;
                        }
                }
                if(flag) cout<<"YES"<<endl;
        }
        return 0;
}
