#include <bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	string s;
	getline(cin,s);
	for(int i=0;i<n;i++){
		getline(cin,s);
		// cout<<s<<"asfasdfasdf"<<endl;
		int sum=0;
		string curr_string="";
		vector<string> v;
		for(int i=0;i<s.length();i++){
			if(s[i]==' '){
				// cout<<"if\n";
				v.push_back(curr_string);
				curr_string="";
			}
			else if((int)s[i]<48 || (int)s[i]>57){
				// cout<<"elseh\n";
				curr_string+=s[i];
			}
			else{
				// cout<<(int)s[i]<<" Else\n";
				sum+= (int)s[i]-'0';
			}
		}
		v.push_back(curr_string);
		sort(v.begin(),v.end());
		for(int i=0;i<v.size();i++){
			sum-=v[i].length();
			cout<<v[i]<<" "<<abs(sum)<<" ";
		}
		cout<<endl;
	}
}
