#include<bits/stdc++.h>
using namespace std;
#define ll long long
map<string,int> mp;
string s[1001];
ll a[1001][1001];
int main()
{
    ll n;
    scanf("%lld",&n);
    mp.clear();
    for(ll i=1;i<=n;i++)
    {
        cin>>s[i];
        mp[s[i]]=i;
    }
    for(ll i=1;i<=n;i++)
    {
        for(ll j=1;j<=n;j++)
        {
            scanf("%lld",&a[i][j]);
            //printf("%lld ",a[i][j]);
        }
        //aprintf("\n");
    }
    ll si=1,di=1,p,ans=0;
    scanf("%lld",&p);
    string str;
    for(int i=0;i<p;i++)
    {
        fflush(stdin);
        cin>>str;
        //cout<<"Destination="<<str<<endl;
        si=di;
        di=mp[str];
        //cout<<"Di:"<<di<<endl;
        ans+=a[si][di];
    }
    printf("%lld\n",ans);
    return 0;
}
