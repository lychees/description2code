#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>

using namespace std;

int main()
{
	int T;
	cin >> T;

	for(;T--;)
	{
		int N,M;
		cin >> N >> M;
		int x1,y1,x2,y2;
		cin >> x1 >> y1 >> x2 >> y2;
		int p;
		cin >> p;

		int X = min(abs(x1-x2), N - abs(x1-x2));
		int Y = min(abs(y1-y2), M - abs(y1-y2));

		int P = p*(X+Y);

		cout << P << "\n";
		if(P > 1000)
			cout << "Let go of the group.\n";
		else
			cout << "You saved the group.\n";
	}
	return 0;
}
