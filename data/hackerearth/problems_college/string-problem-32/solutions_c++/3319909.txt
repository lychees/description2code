#include<iostream>
#include<cstdio>
#include<memory.h>
#include<algorithm>
#include<string>
#define s(n) scanf("%d",&n)
using namespace std;
#define lint long long int
char str[100001];
int main()
{
    int t,i;
    s(t);
    for(int j=1;j<=t;j++)
    {
        scanf("%s",str);
        int len=strlen(str);
        char temp=str[len-1];
        int beg,end;
        int mark=0;
        end=len-1;
        int ans=0;
        for(i=end-1;i>=0;i--)
        {
            if(str[i]==temp)
            {
                beg=i;
                mark=i;
                while(str[beg]==str[end])
                {
                    beg--;
                    end--;
                }
                if(beg==-1)
                ans=max(ans,len-end-1);
                end=len-1;
            }
 
 
 
        }
        if(ans!=0)printf("Case %d: %d\n",j,ans);
        else
        printf("Case %d: -1\n",j);
    }
 
    return 0;
}
