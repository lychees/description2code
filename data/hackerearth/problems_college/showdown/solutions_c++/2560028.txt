#include<bits/stdc++.h>
using namespace std;
vector<int>v1;
vector<int>v2;
int n;
bool fun1(int i,int j)
{
	return (i>j);
}
int fun(int i,int j,int sum)
{
	if(i==n||j==n)
	{
		return 0;
	}
	else if(v1[i]>v2[j])
	{
		sum=100+fun(i+1,j+1,sum);
	}
	else if(v1[i]==v2[j])
	{
		sum=max(50+fun(i+1,j+1,sum),fun(i,j+1,sum));
	}
	else
	{
		sum=0+fun(i,j+1,sum);
	}
	return sum;
}
int main()
{
    cin>>n;
    v1.resize(n);
    v2.resize(n);
    for(int i=0;i<n;i++)
    {
    	cin>>v1[i];
    }
    for(int i=0;i<n;i++)
    {
    	cin>>v2[i];
    }
    sort(v1.begin(),v1.end(),fun1);
    sort(v2.begin(),v2.end(),fun1);
    cout<<fun(0,0,0)<<endl;
}
