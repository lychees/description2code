#include<cmath>
#include <iostream>
#include<algorithm>
using namespace std;
 
int main()
{
	unsigned long long int t;
	cin>>t;
	
	while(t--){
		unsigned long long int z,m,n;
		cin>>z>>m>>n;
		unsigned long long int mi=min(m,n);
		unsigned long long int c=0;
		if(n==mi){
			n=m;
			m=mi;
		}
		unsigned long long int sq=sqrt(z);
		for(unsigned long long int i=1;i<=sq;i++){
			unsigned long long int s=z/i;
			if(s*i==z){
				
				if(s<=m){
					c++;
				}
				if(s<=n&&s!=i){
					c++;
				}
			}
		}
		cout<<c<<endl;
	}
//	return c;
  //  cout << "Hello World!" << endl;
    return 0;
}
