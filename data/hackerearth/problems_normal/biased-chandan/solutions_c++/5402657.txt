#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin >> n;
    stack<int> s;
    int sum=0;
    for(int i=0;i<n;i++)
    {
        int d;
        cin >> d;
        if(s.empty() && d==0)
        {

            //continue;
        }
        else if(d==0)
        {
            sum=sum-s.top();
            s.pop();

        }
        else{
            s.push(d);
            sum=sum+d;
        }




    }
    cout << sum << endl;

}
