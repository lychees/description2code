#include <iostream>
#include<cmath>

using namespace std;
long long int min(long long int a,long long int b)
{
	return a<b?a:b;
}

int main()
{
	int trace;
	cin>>trace;
	while(trace--){
	long long int row,col,di,p,sor=0,des=0;
	cin>>row>>col;
	p=min(row,col);
	di=row+1;
	sor=(p*(2+(p-1)*(di)))/2;
	di=col+1;
	des=(p*(2+(p-1)*(di)))/2;
	
    cout <<sor+des<< endl;
}
    return 0;
}
