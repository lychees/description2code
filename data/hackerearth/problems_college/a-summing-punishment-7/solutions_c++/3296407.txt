#include <iostream>
#include <vector>

using namespace std;

typedef long long int Lint;

int main()
{
	ios_base::sync_with_stdio( false );
	Lint t, n, i, q, a, b;
		cin >> n;
		vector<Lint> A( n ), B( n );
		for( i = 0;i < n;i++ )
			cin >> A[ i ];
		B[ 0 ] = A[ 0 ];
		for( i = 1;i < n;i++ )
			B[ i ] = B[ i - 1 ] + A[ i ];
		cin >> q;
		while( q-- ) {
			cin >> a >> b;
			a -= 1;
			b -= 1;
			cout << B[ b ] - B[ a ] + A[ a ] << "\n";
		}
    return 0;
}
