#include <iostream>
#include <string.h>
using namespace std;

int main() {
	// your code goes here
	char n[201];
	int t;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int len=strlen(n);
		int sum=0;
		for(int i=0;i<len;i++)
		{
			
			if(n[i]!='_')
			sum+=(n[i]-48);
		}
		
		if((sum)%9==0)
		cout<<"0 or 9"<<endl;
		else cout<<9-(sum%9)<<endl;
	}
	return 0;
}
