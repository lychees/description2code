#define N 202000
//vector<pair<int, pii> >v;

int len,pos[N],ind[N],tmp[N],gap,lcp[N],lcppos[N][30];
char s[N];




bool cmp(int i,int j)
{
    if(pos[i]!=pos[j])     return (pos[i]<pos[j]);
    i+=gap;
    j+=gap;
    return ((i<len) && (j<len))?(pos[i]<pos[j]):(i>j);
}


void sa()
{
    int i;
    f(i,0,len-1)    ind[i]=i,pos[i]=s[i];
    for(gap=1;;gap*=2)
    {
        sort(ind,ind+len,cmp);
        fs(i,0,len-1)           tmp[i+1]=tmp[i]+cmp(ind[i],ind[i+1]);
        f(i,0,len-1)            pos[ind[i]]=tmp[i];
        if(tmp[len-1]==len-1)   break;
    }
}

void lcp_()
{
    for(int i=0,k=0;i<len;++i) if(pos[i]!=len-1)
    {
        for(int j=ind[pos[i]+1];s[i+k]==s[j+k];)      ++k;
        lcp[pos[i]+1]=k;
        if(k)   --k;
    }
}

void lcp_2()
{
    int i,j;
    f(i,0,len-1)        lcppos[i][0]=lcp[i];
    for(i=1;(1<<i)<=len;++i)
    {
        for(j=0;j+(1<<i)-1<len;++j)
        {
            if(lcppos[j][i-1]<=lcppos[j+(1<<(i-1))][i-1])
                lcppos[j][i]=lcppos[j][i-1];
            else
                lcppos[j][i]=lcppos[j+(1<<(i-1))][i-1];
        }
    }
}
int solve(int a, int b)
{
    int idx=0;
    if(a>b)     swap(a,b);
    if(a==b)    return len-ind[a];
    while((1<<idx)<=(b-a)) ++idx;
    return min(lcppos[a+1][idx-1],lcppos[b-(1<<(idx-1))+1][idx-1]);
}

int main()
{
    int i,q,x;
    is(s);
    len=strlen(s);
    reverse(s,s+len);
    sa();
    lcp_();
    lcp_2();
    i(q);
    while(q--)
    {
        i(x);
        x=len-x;
        o(solve(pos[x],pos[0]));
        on;
    }
    return 0;
}
