#include <iostream>
using namespace std;
typedef long long ll;
int main()
{
    int T; cin >> T;
    while(T--)
    {
    	int n; cin >> n; // No of non-empty cells.
    	ll r,c,x; cin >> r >> c >> x;
    	bool parity = ((r+c)%2 == x%2);
    	bool ans = true;
    	for(int i=1; i<n; i++)
    	{
    		cin >> r >> c >> x;
    		if(parity!=((r+c)%2==x%2))
    			ans = false;
    	}
    	cout << (ans?"Yes":"No") <<endl;
    }
    return 0;
}
