#include <iostream>
#include <cctype>
#include <cstring>
#include <string>
using namespace std;

int main()
{
    string s;
    long q;
    long long k;
    long long table[26];
    cin >> s;
    cin >> q;
    int n = s.size();
    memset(table, 0, sizeof(table));

    for (int i=0; i<n; )
    {
        if (isalpha(s[i]))
        {
            int sum = 0;
            int j;
            for(j=i+1; s[j]>='0'&&s[j]<='9'; j++)
            {
                sum = sum*10 + (s[j]-'0');
            }
            table[s[i]-'a'] += sum;
            i = j;
        }
    }

    while(q--)
    {
        cin >> k;
        for (int i=0; i<26 ; i++)
        {
            k = k - table[i];
            if (k <= 0)
            {
                int res = 'a' + i;
                printf("%c\n",res);
                break;
            }
        }
        if (k>0)
        	cout <<"-1\n";
    }
   return 0;
}
