#include <bits/stdc++.h>
#include <stdlib.h>
using namespace std;

long long int arr[10000000];

long long int fac(long long int a,long long int b)
{
	if(a>b)
	{
		long long int temp=a;
		a=b;
		b=temp;
	}
	long long int j=1;
	arr[0]=1;
	for(long long int i=2;i*i<=a;i++)
	{
		if(a%i==0)
		{
			if(b%i==0)
			{
				arr[j]=i;
				j++;
			}
			if(b%(a/i)==0 && i*i!=a)
			{
				arr[j]=(a/i);
				j++;
			}
		}
	}
	if(b%a==0)
	{
		arr[j]=a;
		j++;
	}
	return j;
}

int main()
{
	long long int a,b,l,r,q,n;
	scanf("%lld%lld",&a,&b);
	scanf("%lld",&q);
	n=fac(a,b);
	sort(arr,arr+n);
	while(q--)
	{
		scanf("%lld%lld",&l,&r);
		if(l>r)
		{
			long long int temp=l;
			l=r;
			r=temp;
		}
		long long int *ans1 = upper_bound (arr, arr+n, r);
		long long int *ans2 = lower_bound (arr, arr+n, l);
		if(ans1!=ans2)
		{
			ans1--;
			printf("%lld\n",*ans1);
		}
		else
		{
			printf("-1\n");
		}
	}
	return 0;
}
