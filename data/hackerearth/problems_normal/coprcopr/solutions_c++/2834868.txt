#include<bits/stdc++.h>

using namespace std;

#define ll long long int
#define print(a) cout<<a
#define T ll t; cin>>t ; while(t--)
#define fin freopen("input.txt","r",stdin)
#define fout freopen("op.txt","w",stdout)
#define fastio ios_base::sync_with_stdio(false); cin.tie(0)
#define debug(s,n) cout<<">> "<<s<<" : "<<n<<endl
#define pf(n) cout<<n<<endl;

ll sz=1e7;
ll phi[10000005];
ll ans[10000005];
ll prime[10000005];

void preprocess() {

    for(ll i=2;i*i<=sz;i++) {
        if(prime[i]==0) {
            for(ll j=i*i;j<=sz;j+=i) {
                prime[j]=i;
            }
        }
    }

    phi[0]=0;
    phi[1]=1;
    phi[2]=1;

    for(ll i=2;i<=sz;i++) {
        if(prime[i]==0) {
            phi[i]=i-1;
        }
        else {
            ll q=prime[i];
            ll val=1;
            ll x=i;

            while(x%q==0) {
                x=x/q;
                val=val*q;
            }

            phi[i]=phi[x]*val*(q-1)/q;
        }
    }

    ans[0]=0;
    for(ll i=1;i<=sz;i++)
        ans[i]+=phi[i] + ans[i-1];
}

int main(){

    preprocess();

    T {
        ll n;
        cin>>n;

        pf(ans[n]);
    }

    return 0;
}
