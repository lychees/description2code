    #include <iostream>
    #include<bits/stdc++.h>
    using namespace std;
     
    int main()
    {
        int l;
        char str[100005];
        cin>>str;
        l=strlen(str);
        if(str[0]!='_')
        str[0]=toupper(str[0]);
        for(int i=1;i<l;i++)
        {
        	if(str[i-1]=='_')
        str[i]=toupper(str[i]);
        }
        cout<<str;
        return 0;
    }
