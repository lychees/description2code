int BIT[16][100005];
int64 arr[100005];
int dis_cou(int64 x){
	int a[10]={0};
	while(x>0){
		a[x%10]++;
		x=x/10;
	}
	int ret=0;
	for(int i=0;i<10;i++){
		if(a[i])
		ret++;
	}
	return ret;
}

void modify(int dig,int idx,int inc){
	while(idx<=100005){
		BIT[dig][idx]+=inc;
		idx+=(idx&(-idx));
	}
}

int query(int dig,int idx){
	int ret=0;
	while(idx){
		ret+=BIT[dig][idx];
		idx-=(idx&(-idx));
	}
	return ret;
}

	for(i=1;i<=n;i++){
		cin>>arr[i];
		int x=dis_cou(arr[i]);

	}
