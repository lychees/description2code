#include <iostream>
#include <string>
#include <cstring>
#include <cmath>
using namespace std;

int main()
{
    int n,i,j,T,k,t;
    cin>>T;
    for(t=0;t<T;t++){
        cin>>n;
        k=(int)(sqrt(n));
        j=pow(k,2)+k;
        if(n==pow(k,2)) cout<<(2*k-1);
        else if(n<=j) cout<<(2*k);
        else if(n>j) cout<<(2*k+1);
        else;
        cout<<"\n";
    }
    return 0;
}
