#include <stdio.h>
#include <vector>
#include <algorithm>
#define pb push_back

using namespace std;

vector <int> y;
vector <int> b;

int main(void) {
    int n;
    int cor, lado;
    int i, j;
    int r = 0;

    scanf(" %d", &n);

    while(n--) {
        scanf(" %d %d", &cor, &lado);
        if (cor == 1) {
            b.pb(lado);
        } else {
            y.pb(lado);
        }
    }
    sort(y.begin(), y.end());
    sort(b.begin(), b.end());
    reverse(y.begin(), y.end());
    reverse(b.begin(), b.end());

    i = j = 0;
    while(i < (int)y.size() && j < (int)b.size()) {
        //printf("y = %d, b = %d, l1 = %d, l2 = %d\n", y[i], b[j],  2 * y[i] * y[i], b[j] * b[j]);
        if (2 * y[i] * y[i] > b[j] * b[j] && 2 * b[j] * b[j] > y[i] * y[i]) {
            r++;
            i++;
            j++;
        } else if (2 * y[i] * y[i] <= b[j] * b[j]){
            j++;
        } else {
            i++;
        }
    }
    printf("%d\n", r);

    return 0;
}
