arr = int(raw_input())
for n in xrange(0, arr):
    str_arr = [int(x) for x in raw_input().split()]
    if len(str_arr) == 2:
        r = str_arr[0]
        n = str_arr[1]
        if n >= r:
            res = 1
            for i in xrange(n - r + 1, n + 1):
                res = (res*i) % (1000000007)

            print res
