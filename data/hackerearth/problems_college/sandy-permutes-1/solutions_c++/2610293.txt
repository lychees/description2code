    #include <bits/stdc++.h>
    using namespace std;
    int main()
    {
        long long int n,k,arr[100001],i,j,temp,max,c=0;
        cin>>n>>k;
        for(i=0;i<n;i++)
        	cin>>arr[i];
        if(k>=n)
        {
        	for(i=n;i>0;i--)
        		cout<<i<<" ";
        }
        else
        {
        for(i=0;i<n-1;i++)
        {
        	max=i;
        	for(j=i+1;j<n;j++)
        	{
        		if(arr[j]>arr[max])
        		max=j;
        	}
        	if(i!=max)
        	{
        		temp=arr[i];
        		arr[i]=arr[max];
        		arr[max]=temp;
        		c++;
        		if(c==k)
        		break;
        		
        	}
        	
        }
        for(i=0;i<n;i++)
        cout<<arr[i]<<" ";
        return 0;
    }
    }
