#include <iostream>
#include <stdio.h>
using namespace std;

int main() {
long long int t,n;
scanf("%lld",&t);
while(t--)
{
	scanf("%lld",&n);
long long int i,j,c=0;
for(i=3;i<=n;i++)
{
	for(j=i+1;j<=n;j++)
	{
		if((i*j)%(i+j)==0)
		{
			
			c++;
		}
	}
}
printf("%lld\n",c);
}


return 0;
}
