#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--){
    	long long int a,b;
    	cin>>a>>b;
    	long long int c=0;
    	while(!(a==1 && b==1)){
    		if(a==1 || b==1){
    			if(a==1){
    				c = c + (b-1);
    				b = 1;
    			}
    			else{
    				c = c + (a-1);
    				a = 1;
    			}
    		}
    		else if(a<b){
    			long long int t = b/a;
    			c = c + t;
    			b = b - (t*a);
    		}
    		else{
    			long long int t = a/b;
    			c = c + t;
    			a = a - (t*b);
    		}
    	}
    	cout<<c<<endl;
    }
}
