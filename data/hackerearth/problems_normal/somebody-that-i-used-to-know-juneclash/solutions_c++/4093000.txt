#include <bits/stdc++.h>

using namespace std;

int N, M;
long long A[301][301];
long long B[301][301];
long long C[301][301];

int main()
{
    memset(A, 0x3f, sizeof A);
    memset(B, 0x3f, sizeof B);
    memset(C, 0x3f, sizeof C);
    scanf("%d%d", &N, &M);
    int a, b, c;
    char d;
    for(int i=0; i<M; i++)
    {
        scanf("%d%d%d %c", &a, &b, &c, &d);
        if(d=='(')
            B[a][b]=c;
        else
            C[a][b]=c;
    }
    for(int i=1; i<=N; i++)
        A[i][i]=0;
    for(int t=0; t<10; t++)
    {
        for(int k=1; k<=N; k++)
        {
            for(int i=1; i<=N; i++)
            {
                for(int j=1; j<=N; j++)
                {
                    A[i][j]=min(A[i][j], A[i][k]+A[k][j]);
                    A[i][j]=min(A[i][j], B[i][k]+C[k][j]);
                    B[i][j]=min(B[i][j], B[i][k]+A[k][j]);
                    B[i][j]=min(B[i][j], A[i][k]+B[k][j]);
                    C[i][j]=min(C[i][j], C[i][k]+A[k][j]);
                    C[i][j]=min(C[i][j], A[i][k]+C[k][j]);
                }
            }
        }
    }
    for(int i=1; i<=N; i++)
        for(int j=1; j<=N; j++)
            printf("%lld%c", A[i][j]==0x3f3f3f3f3f3f3f3fLL?-1:A[i][j], " \n"[j==N]);
    return 0;
}
