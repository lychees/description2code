#include<iostream>
using namespace std;
int main()
{
 int t;
 cin>>t;
 while(t--)
 {
      long long int n,x,count;
      cin>>n;
      x=0;count=0;
      while(n>0)
      {
        x=n%10;
        if(x==4||x==7)
        {
         count++;
        }
        n=n/10;
      }
      if(count==4||count==7)
      {
      cout<<"YES\n";
      }
      else
      {
      cout<<"NO\n";
      }
   }
}
