#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	long long int ans = 1;
	scanf("%d",&n);

	int a[100];

	int i=0;
	for(;i<n;i++)
	{
		scanf("%d",&a[i]);
	}

	for(i=0;i<n;i++)
	{
		if(a[i] ==1)
		{
			break ;
		}
	}

	int j = n-1 ;
	for(;j>i;j--)
	{
		if(a[j] == 1)
			break ;
	}
    if(i>j) ans=0;
	//printf("%d   %d  ",i,j);
	int count =1;
	for(;i<=j;i++)
	{
		if(a[i]==1)
        {
            ans = ans*(count);
            count =1;
        }
        else
        	count++;
    }
	cout<<ans;

return 0 ;
}
