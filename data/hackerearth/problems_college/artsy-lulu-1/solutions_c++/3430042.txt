#include <iostream>
#include <string>
#include <sstream>
#include <cctype>
#include <vector>
using namespace std;

int main() {

	char c;
	int n;
	
	while(cin >> c >> n) {
	//cout << int(c) + n<< endl;
		if((tolower(int(c))+n)>123 || n == 1)
			cout << "INVALID" << endl;
		else {
			for(int i=0; i<n; i++) {
				for(int j=0; j<i+1; j++) 
				cout << char(int(c)+i) << ' ';
			cout << endl;
			}
		}
		cout << endl;
	}

	return 0;
}
