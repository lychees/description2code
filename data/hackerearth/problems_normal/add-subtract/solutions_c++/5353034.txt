#include <bits/stdc++.h>
using namespace std;

typedef long long LL;
#define FOR(a,b,c) for(int a = b; a <= c; a++)
#define RFOR(a,b,c) for(int a = b; a >= c ; a--)
#define pb push_back
#define all(v) v.begin(),v.end()

int main() {
	LL t, n, k;
	cin >> t;
	while(t--) {
		cin >> n >> k;
		LL *a = new LL[n + 1];
		FOR(i, 1, n) {
			cin >> a[i];
		}
		vector<int> mincost;
		FOR(i, 1, 100) {
			vector<int> v;
			int cost = 0;
			FOR(j, 1, n) {
				cost = 0;
				if(i - a[j] >= 0) {
					cost += 3 * (i - a[j]);
				}
				else {
					cost += 5 * (a[j] - i);
				}
				v.pb(cost);
			}
			sort(all(v));
			cost = 0;
			FOR(j, 0, k - 1) {
				//cout << v[j] << " -vj" << endl;
				cost += v[j];
			}
		//	cout << endl;
			//cout << cost << "-mincost" << endl;
			mincost.pb(cost);
		}
		cout << *min_element(all(mincost)) << endl;
	}
	return 0;
}
