#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <set>
#include <sstream>
#include <string>
using namespace std;
int main() {
  string a;
  int sum = 0;
  cin >> a;
  for (int i = a.size() - 1; i > 0; i--) {
    if (a[i] == '0' && i) {
      sum += a[i] - '0' + (a[i - 1] - '0') * 10;
      i--;
    } else
      sum += a[i] - '0';
  }
  cout << sum + 1 << endl;
  return 0;
}
