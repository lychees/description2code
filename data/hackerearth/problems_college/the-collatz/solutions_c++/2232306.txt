#include <iostream>
using namespace std;

int steps;
int evenSum,oddSum;

void resolve(int n) {
	if(n==1) {
		if((evenSum%steps)>(oddSum%steps)) cout<<"Even Rules\n";
		else if((evenSum%steps)<(oddSum%steps)) cout<<"Odd Rules\n";
		else cout<<"Tie\n";
	}
	else if(n%2==0) {
		evenSum+=n;
		n=n/2;
		steps++;
		resolve(n);
	}
	else {
		oddSum+=n;
		n=(3*n)+1;
		steps++;
		resolve(n);
	}
}

int main()
{
    int t,n;
    cin>>t;
    while(t>0) {
    	cin>>n;
    	steps=0;
    	evenSum=0;
    	oddSum=1;
    	resolve(n);
    	t--;
    }
    return 0;
}
