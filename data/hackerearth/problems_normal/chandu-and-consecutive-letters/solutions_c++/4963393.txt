#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	string a;
    	cin>>a;
    	int len = a.size();
    	
    	for(int i=0;i<len;i++) {
    		int count = 0;
    			if(a[i]==a[i+1]) {
    				for(int j=i+1;j<len;j++) {
    					if(a[i]==a[j] ) {
    						count++;
    					}
    					else {
    						break;
    					}
    				}
    			}
    		for(int j=i;j<len;j++) {
    			a[j] = a[j+count];
    		}
    		len = len - count;
    	}
    	for(int i=0;i<len;i++) {
    		printf("%c",a[i]);
    	}
    	printf("\n");
    }
    return 0;
}
