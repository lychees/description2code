#include<bits/stdc++.h>
#define ll long long
using namespace std;

ll  countFact(ll n, ll p)
{
	ll k=0;
	while (n>0)
	{
		k+=n/p;
		n/=p;
	}
	return k;
}


long long pow(ll a, ll b, ll MOD)
{
	long long x=1,y=a; 
	while(b > 0)
	{
		if(b%2 == 1)
		{
			x=(x*y);
			if(x>MOD) x%=MOD;
		}
		y = (y*y);
		if(y>MOD) y%=MOD; 
		b /= 2;
	}
	return x;
}

long long C(ll n, ll r, ll MOD)
{
	long long res = 1;
	vector<bool> isPrime(n+1,1);
	for (ll i=2; i<=n; i++)
		if (isPrime[i])
		{
			for (ll j=2*i; j<=n; j+=i)
				isPrime[j]=0;
			ll k = countFact(n,i) - countFact(r,i) - countFact(n-r,i);  
			res = (res * pow(i, k, MOD)) % MOD;
		}
	return res;
}
