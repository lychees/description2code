#include<bits/stdc++.h>
using namespace std;

int main()
{
    std::ios::sync_with_stdio(0);
    int t;
    cin>>t;
    while(t--)
    {
    	string c,r;
    	cin>>c>>r;
    	sort(r.begin(),r.end());
    	for(int i=0,j=0;i<c.size();i++)
    	{
    		if(j>=r.size())
    		break;
    		else
    		if(c[i]>r[j])
    		{
    			c[i]=r[j];
    			j++;
    		}
    	}
    	cout<<c<<endl;
    }
   return 0; 
}
