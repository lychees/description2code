#include<bits/stdc++.h>
using namespace std;
long long int t,n,i,j,k,l,s,a,b;
char str[200005];
int main()
{
    cin>>t;
    while(t--)
    {
    	scanf("%lld",&n);
    	scanf("%s",str);
    	a=0;b=0;
    	for(i=0;i<n;i++)
    	{
    		if(str[i]=='0')
    		a++;
    	}
    	b=n-a;
    	j=0;
    	for(i=0;i<a;i++)
    	{
    		if(str[i]=='0')
    		j++;
    	}
    	s=a-j;
    	k=0;
    	
    	for(i=0;i<b;i++)
    	{
    		if(str[i]=='1')
    		k++;
    	}
    	s=min(s,b-k);
    	cout<<s<<endl;
    }
    return 0;
}
