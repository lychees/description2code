#include <iostream>
using namespace std;

long long int mod=1e9+7;

int main()
{
    long long a,b,c,d,e,f,q,ans,n;
    cin>>a>>b>>c>>d>>e>>f;
    cin>>q;
    while(q--)
    {
    	cin>>n;
    	ans=0;
    	if(n==0) {cout<<'0'<<'\n'; continue;}
    	ans=(d*(n-1))%mod;
    	ans++;
    	ans=(ans*f)%mod;
    	cout<<ans<<'\n';
    	
    }
    
    return 0;
}
