
#include <iostream>
#include <algorithm>

using namespace std;

int test,ma,n,m,cc,my;
int a[100],b[100];

int main()
{
   cin >> test;

   while(test--)
   {
       ma=0;

       cin >> n >> m;

       for(int i=0;i<n;i++)
            cin >> a[i];

       my=a[0];

       sort(a,a+n);

       for(int i=0;i<n;i++)
       {
           b[i]=a[n-1-i];
       }

       for(int i=0;i<n;i++)
       {
           if(my==b[i])
           {
               cc=i%m;
               break;
           }
       }

       for(;cc<n;cc+=m)
       {
           if(b[cc]>my)
            ma++;
       }

       cout << ma << "\n";

   }

}
