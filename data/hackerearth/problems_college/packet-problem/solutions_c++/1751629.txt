#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll par[100100];

int main() {
	ll dp[200][4];
	dp[0][0] = 0;
	dp[0][1] = 1;
	for(ll i = 1;i <= 100;i++) {
		dp[i][0] = dp[i - 1][0] +  dp[i - 1][1];
		dp[i][1] = dp[i - 1][0];

	} 
	ll t;
	cin >> t;
	while(t--) {
		ll m;
		cin >> m;
		cout << dp[m][1] <<  " "<< dp[m][0] << "\n";
	}
}
