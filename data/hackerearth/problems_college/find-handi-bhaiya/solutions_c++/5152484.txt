#include<bits/stdc++.h>
using namespace std;
int main(){
    int tc;cin>>tc;
    while(tc--){
        vector<int>a(3);
        for(int i=0;i<3;++i)cin>>a[i];
        sort(a.begin(),a.end());
        if(a[0]+a[1]<=a[2])puts("NO");
        else{
            map<int,int>mpp;
            for(int i=0;i<3;++i)mpp[a[i]]++;
            if(mpp.size()==1)puts("YES. It is a equilateral triangle.");
            else if(mpp.size()==2)puts("YES. It is a isosceles triangle.");
            else puts("YES. It is a scalene triangle.");
        }
    }
    return 0;}
