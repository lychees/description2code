#include <iostream>
using namespace std;

int main()
{
    int n,i,j,count=0;
    cin>>n;
    string str;
    int a[n][26];
    for(i=0;i<n;i++)
        for(j=0;j<26;j++)
              a[i][j]=0;
    for(i=0;i<n;i++)
    {
    	cin>>str;
    	j=0;
    	while(str[j]!='\0')
    	{
    		a[i][str[j]-'a']++;
    		j++;
    	}
    }
    for(i=0;i<26;i++)
    {
         for(j=0;j<n;j++)
         {
         	if(a[j][i]<=0)
         	     break;
         }
         if(j==n)
             count++;
    }
    cout<<count;
    return 0;
}
