#include <cstdio>
#include <iostream>
#include<cmath>
using namespace std;

#define MAX 1000010
#define ll long long
const ll oo = 1000000;

int a[MAX], b[MAX], f[MAX];
int n;

void set(int idx){
	while (idx <= oo){
		f[idx]++;
		idx += idx & (-idx);
	}
}

ll get(int idx){
	ll res = 0;
	
	while (idx > 0){
		res += f[idx];
		idx -= idx & (-idx);
	}
	
	return res;
}

int main(){
	
	
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
		scanf("%d", &a[i]);
	for (int i = 0; i < n; i++)
		scanf("%d", &b[i]);
	
	ll res = 0;
	set(b[n - 1]);
	for (int i = n - 2; i >= 0; i--){
		res += get(a[i] - 1);
		set(b[i]);
	}
	
    int flag=0;
    for(int k=2;k<=sqrt(res);k++)
    {
    	if(res%k==0)
    	{
    		flag++;
    		break;
    	}
    }
    
    if(flag==0)
    cout<<"MAGIC INVERSION"<<endl;
    else
    cout<<"SIMPLE INVERSION"<<endl;
    
    
    
    
    return 0;
}
