#include<bits/stdc++.h>
using namespace std;
int freq[109];
int main()
{
int k,m;
cin>>k>>m;
	while(k--)
	{
		int xi,temp;
		cin>>xi;
		for(int i=1;i<=xi;++i)
		{
			cin>>temp;
			++freq[temp];
		}
	}
	bool flag=true;
	for(int i=1;i<=m;++i)
	{
		if(freq[i]==0)
		{
			flag=false;
			break;
		}
	}
	(flag==false)?puts("NO"):puts("YES");
	return 0;
}
