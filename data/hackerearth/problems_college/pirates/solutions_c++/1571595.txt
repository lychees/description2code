#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n,i,j,ans,k;
	scanf("%d",&n);
	int a[n];
	for (i = 0; i < n; i++) {
		scanf("%d",&a[i]);
	}
	ans = 0, k = 1;
	for (i = 1; i < n; i++) {
		if (abs(a[i] - a[i-1]) == 1) {
			k++;
			if (k > ans) {
				ans = k;
			}
		} else {
			k = 1;
		}
	}
	if (k > ans) {
		ans = k;
	}
	printf("%d\n",ans);
	
	return 0;
}
