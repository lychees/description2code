#include <bits/stdc++.h>
 
#define LLI long long int
#define LD long double
#define PB push_back
#define MP make_pair
#define FORi(i, a, b) for(int i = a; i < b ; ++i)
#define FORd(i, a, b) for(int i = a; i > b ; --i)
 
using namespace std;
 
const int mod = 1000000000;
 
int main(){
	int n;
	cin >> n;
	bool v[n];
	FORi(i,0,n){
		cin >> v[i];
	}
	int l=0;
	FORi(i,0,n)
		if (!v[i])
			l++;
		else
			break;
	LLI tmp=1,ans=1;
	FORi(i,l,n){
		if (v[i]){
			ans *= tmp;
			tmp = 1;
		}
		else{
			tmp++;
		}
	}
	if (l==n)
		cout << 0;
	else
		cout << ans;
    return 0;
}
