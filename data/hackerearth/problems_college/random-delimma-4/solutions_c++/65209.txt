#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
	int t;
    double ans;
    int n,k;
    cin>>t;

    while(t--){
    	ans=1.0;
    	cin>>k>>n;
    	if(k>n)ans=1.0;
    	else
	    {
	    	for(int i=0;i<k;i++)
      		ans *=(n-i)*1.0/n;
	    }
      	ans = 1.0-ans;
      	printf("%.9f\n",ans);
    }
    return 0;
}
