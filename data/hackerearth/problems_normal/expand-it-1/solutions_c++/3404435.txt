#include <iostream>
#include <string>

using namespace std;
long long int arr[30];
	
int main()
{
	string s;
	cin>>s;
	int len = s.size();
	
	string cur=""; 
	int ind=s[0];
	for(int i=0;i<len;i++)
	{
		if(s[i]>='0'&&s[i]<='9')
			cur=cur+s[i];
		
		else if(cur.size()>0)
		{
			arr[ind-'a'] = arr[ind - 'a'] + std::stoi(cur);
			ind = s[i];
			cur="";
		}
	}
	if(cur.size()>0)
		arr[ind-'a'] = arr[ind-'a'] + std::stoi(cur);
		
//	for(int i=0;i<30;i++)
//		cout<<" "<<arr[i];
	int q;
	cin>>q;
	for(int i=0;i<q;i++)
	{
		long long int cur;
		cin>>cur;
		long long int j=0;
		bool flag=false;
		int index=0;
		while(j<=cur&&index<29)
		{
			j = j+arr[index];
			if(j>=cur)
			{
				char out = index + 'a';
				cout<<out<<endl;
				flag=true;
				break;
			}
			index++;
		}
		if(!flag)
			cout<<"-1"<<endl;
	}
    return 0;
}
