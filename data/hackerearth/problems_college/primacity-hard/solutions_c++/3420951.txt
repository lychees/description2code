#include<stdio.h>
#include<vector>
#include<stdlib.h>
#include<algorithm>
using namespace std;
#define max 10000001
int main()
{	long t;
	long long *ar;
	ar=(long long*)malloc(sizeof(long long)*max);
	long long a,b,i,j,k;
	for(i=2; i<max; i++)
	{	if(ar[i]==0)
		{	ar[i]=1;
			for(j=2*i; j<max; j=j+i)
				ar[j]=ar[j]+1;
		}
	}

	vector<long long>v[10];
	for(i=2;i<max;i++)
		v[ar[i]].push_back(i);

	vector<long long>::iterator low;
	vector<long long>::iterator up;
	
	scanf("%ld",&t);
	while(t--)
	{	long long count;
		scanf("%lld %lld %lld",&a,&b,&k);
		if(k<=8)
		{
		low=lower_bound(v[k].begin(),v[k].end(),a);
		up=upper_bound(v[k].begin(),v[k].end(),b);
		count=up-low;
		printf("%lld\n",count);
		}
		else
		printf("0\n");
	}
	return 0;
}
