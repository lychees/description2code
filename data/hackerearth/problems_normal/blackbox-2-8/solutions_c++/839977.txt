#include<stdio.h>
#include<vector>

using namespace std;

vector<int> str1(200000);

void Store(vector<int>& str,int n,int *index)
{	
	while(n)
	{
		str[(*index)++]=n%10;
		n/=10;
	}
}

void Multiply(vector<int>& str,int n,int *index)
{
	int i=0,val,carry=0;
	
	for(i=0;i<*index;i++)
	{
		val=n*str[i] + carry;
		str[i]=val%10;
		carry=val/10;
	}
	
	Store(str,carry,index);
}

int main()
{
	int t,n,i,val;
	
	scanf("%d",&t);
	
	int sum1[1001],j;
	
	str1[0]=1;
	int index=1,sum=0;
	
	for(i=1;i<=1000;i++)
	{
		sum=0;
	
		Multiply(str1,i,&index);
		
		for(j=index-1;j>=0;j--)
		{
			sum+=str1[j];
		}
		
		sum1[i]=sum;
	}
	
	
	while(t--)
	{
		scanf("%d",&n);
		
		if(n!=0)
		printf("%d\n",sum1[n]);
		
		else
		printf("%d\n",1);
	}
	
	return 0;
}
