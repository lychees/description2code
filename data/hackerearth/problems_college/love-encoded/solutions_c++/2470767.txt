#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
	scanf("%d",&t);
	while(t--)
		{
            int k;cin>>k;
            string s;cin>>s;
            for(int i=0;i<s.length();i++)
            {
                printf("%c",((s[i]-65-k + 26)%26+65));
            }
		cout<<"\n";
		}
	return 0;
}
