#include <bits/stdc++.h>
using namespace std;

int main()
{
    string cs,dig,ans;
    cin>>cs;
    int64_t freq[26]={0},q,k;
    for(int64_t i=0;i<(int64_t)cs.size();i++)
    {
    	int64_t j=i;
    	dig="";
    	while(((j+1)<(int64_t)cs.size())&&((int)cs[j+1]>=47)&&(int)cs[j+1]<=57)
    	{
    		dig+=cs[j+1];
    		j++;
    	}
    	stringstream ss;
    	ss<<dig;
    	int64_t frq;
    	ss>>frq;
    	freq[(int)cs[i]-97]+=frq;
    	i=j;
    }
    cin>>q;
    while(q--)
    {
    	cin>>k;
    	ans="-1";
    	for(int64_t i=0;i<26;i++)
    	{
    		k-=freq[i];
    		if(k<=0)
    		{
    			ans="";
    			ans+=(char)(97+i);
    			break;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
