#include <iostream>
using namespace std;

int main()
{
	long n;
	cin>>n;
	long a[1000001];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	long q,fq=1,temp;
	cin>>q;
	for(int i=0;i<q;i++)
	{
		cin>>temp;
		fq*=temp;
	}
	for(int i=0;i<n;i++)
		if(fq>0)
		cout<<(a[i]/fq)<<" ";
		else
		cout<<(0)<<" ";
		
   // cout << "Hello World!" << endl;
    return 0;
}
