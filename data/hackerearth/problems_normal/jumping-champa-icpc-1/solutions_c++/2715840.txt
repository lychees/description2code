#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int testc, no_city,qual, temp;
        
    cin>>testc;
    
    while(testc != 0)
    {
    	cin>> no_city>>qual;
    	long height[1000];
    
    	long long sum=0;
    	
    	for(int i=0; i<no_city; i++)
    	{
    		cin>>height[i];    			
    	}
    	
    	sort(height, height+no_city);
    	
    	
    	if(no_city >= 1)
    	{	for(int i=0; i< no_city-1 ; i++)
    		{
    			sum+=(height[i+1]-height[i]) * qual;
    		}
    	}
    	    	
    	cout<<sum<<"\n";
    	
    	testc--;
    	
    	
    }
    	
    
    
    
    
    return 0;
}
