#include <iostream>
#include <map>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	map<int,int>m;
    	int c =0;
    	for(int i = 0;i<n;i++)
    	{
    		int x;
    		scanf("%d",&x);
    		if(m.count(x) == 0)
    		{
    			c++;
    			m[x] = 1;
    		}
    		
    	}
    	printf("%d\n",c);
    	
    }
    return 0;
}
