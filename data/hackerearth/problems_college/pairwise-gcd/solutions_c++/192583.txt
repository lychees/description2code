#include <iostream>
#include <stdio.h>
#include <math.h>
#define MAX 400001
using namespace std;

long long phiarray[MAX];
long long ans[MAX];
void phi();
void calculate();
int main()
{
    phi();
    calculate();
    long long n,t;
    scanf("%lld",&t);
    while(t--)
    {
    scanf("%lld",&n);
    {
        if(n==0)
        break;
        printf("%lld\n",ans[n]);
    }
    }
    return 0;
}
void phi()
{
	phiarray[1] = 1;
	for(int i=2; i<=MAX; i++)
	{
	    if(phiarray[i]==0)
	    {
	        phiarray[i]=i-1;
			for(int j=i+i; j<=MAX; j=j+i)
			{
				if(phiarray[j]==0)   phiarray[j] = j;
                phiarray[j]= phiarray[j]/i * (i-1);
			}
	    }
	}
}

void calculate()
{
    for(int i=1;i<MAX;i++)
    {
        for(int j=i+i;j<MAX;j=j+i)
        {
            ans[j]+=(i*phiarray[j/i]);
        }
    }
    for(int i=2;i<MAX;i++)
        ans[i]+=ans[i-1];
}
