#include <iostream>
#include <string>
using namespace std;
int main(){
	ios::sync_with_stdio(false);
	string key="";
	
	cin >> key;
	int keyLen = key.length();
	
	int T=0;
	
	cin >> T;
	
	int syn=0;
	int ant=0;
	
	while(T--)
	{
	 string in="";
	 cin >> in;
	 int inLen = in.length();
	 
	 if(inLen != keyLen)
	  continue;

	 if(in[0] == key[0])
	  syn++;
	 else if(key[0] == (char)((int)in[0]-3))
	  ant++;
	}
	
	cout << syn << " " << ant << endl;
	
	return 0;
}
