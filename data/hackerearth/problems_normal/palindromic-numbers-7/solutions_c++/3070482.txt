#include<iostream>
using namespace std;
int main()

{ long long int tc;
cin>>tc;
for(long long int i=0;i<tc;i++)
{long long int lower,upper,temporary=0;
cin>>lower>>upper;
for (long long int j=lower;j<=upper;j++)
 {
            long long int temp=0,num,r=0,rev=0;

            num=j;
            while(num>0)
            {
               num=num/10;
               temp++;
            }

            num=j;
            for(long long int i=0;i<temp;i++)
            {  r=num%10;
               rev=rev*10+r;
               num=num/10;

            }
            if(rev==j)
            temporary++;
            }
cout<<temporary<<endl;
}
   return 0;
}
