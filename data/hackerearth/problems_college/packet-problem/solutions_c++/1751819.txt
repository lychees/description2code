#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int t,a,b,c,m;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld",&m);
    	a=0;
    	b=1;
    	while(m>1)
    	{
    		c=a+b;
    		a=b;
    		b=c;
    		m--;
    	}
    	printf("%lld %lld\n",a,b);
    }
    return 0;
}
