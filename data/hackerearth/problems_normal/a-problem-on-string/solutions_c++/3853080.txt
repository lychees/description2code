#include <iostream>
#include <string>
#include <array>
using ULL = unsigned long long int;

int main()
{
	auto n = 0ULL;
	std::cin >> n;
	auto str = std::string();
	std::cin >> str;
	auto freq = std::array<ULL, 26>();
	for (ULL i = 0; i < n; ++i)
	{
		++freq[str[i] - 'a'];
	}
	auto q = 0;
	std::cin >> q;
	while (q--)
	{
		char a, b;
		std::cin >> a >> b;
		if (a == b)
		{
			std::cout << (freq[a - 'a'] * (freq[a - 'a'] - 1))/2;
		}
		else
		{
			std::cout << freq[a - 'a'] * freq[b - 'a'];
		}
		std::cout << std::endl;
	}
    return 0;
}
