#include <iostream>
#include <vector>
using namespace std;
typedef long long int lld;
const int lim = 10001;
const int MOD = 1000000009;
vector<vector<lld> > lookup;
void preprocess(){
	lookup = vector<vector<lld> >(10 , vector<lld>(lim , 0));
	for(int i=0;i<10;++i){
		lookup[i][1]=1;
	}
	for(int i=2;i<lim;++i){
		lookup[0][i] = lookup[0][i-1] + lookup[8][i-1];
		lookup[1][i] = lookup[1][i-1] + lookup[2][i-1] + lookup[4][i-1];
		lookup[2][i] = lookup[2][i-1] + lookup[1][i-1] + lookup[5][i-1] + lookup[3][i-1];
		lookup[3][i] = lookup[3][i-1] + lookup[2][i-1] + lookup[6][i-1];
		lookup[4][i] = lookup[4][i-1] + lookup[1][i-1] + lookup[5][i-1] + lookup[7][i-1];
		lookup[5][i] = lookup[5][i-1] + lookup[4][i-1] + lookup[2][i-1] + lookup[6][i-1] + lookup[8][i-1];
		lookup[6][i] = lookup[6][i-1] + lookup[5][i-1] + lookup[3][i-1] + lookup[9][i-1];
		lookup[7][i] = lookup[7][i-1] + lookup[4][i-1] + lookup[8][i-1];
		lookup[8][i] = lookup[8][i-1] + lookup[7][i-1] + lookup[5][i-1] + lookup[9][i-1] + lookup[0][i-1];
		lookup[9][i] = lookup[9][i-1] + lookup[8][i-1] + lookup[6][i-1];
		
		for(int j=0;j<10;++j){
			lookup[j][i] = lookup[j][i]%MOD;
		}
	}
}
int main()
{
    int t , n;
    scanf("%d",&t);
    
    preprocess();
    lld sum;
    while(t--){
    	scanf("%d",&n);
    	sum=0;
    	for(int i=0;i<10;++i){
    		sum += lookup[i][n];
    		sum  = sum%MOD;
    	}
    	printf("%lld\n",sum);
    }
    
    return 0;
}
