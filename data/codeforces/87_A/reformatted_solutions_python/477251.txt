#!/usr/bin/python
import sys


def gcd(a, b):
    while b > 0:
        a, b = b, a % b
    return a


def read_ints():
    return map(int, sys.stdin.readline().split())


def read_line():
    return sys.stdin.readline().strip()


a, b = read_ints()
g = gcd(a, b)
x, y = b/g, a/g
if a > b:
    x += 1
else:
    y += 1

if x > y:
    print 'Dasha'
elif x < y:
    print 'Masha'
else:
    print 'Equal'
