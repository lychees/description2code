    #include <iostream>
    #include <cmath>
    using namespace std;
    int stepno(int n)
    {
    	int x;
    	if(n==0)
    	return 0;
    	else
    	{
    	
    	x=(int)((sqrt(1+8*n)-1)/2);
    	if(n>1)
    	x--;
    	if((n-((x*(x+1))/2))==0)
    	return x;
    	else
    	return x+1;
    }
    }
    int main()
    {
    	int t, n;
    	cin >> t;
    	while (t--)
    	{
    		cin >> n;
    		cout << stepno(n)<<endl;
    	}
    	cin.get();
    	cin.get();
    	return 0;
    }
