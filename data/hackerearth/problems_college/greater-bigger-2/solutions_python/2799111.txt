import itertools
t=int(raw_input())
while t:
    w=raw_input()
    a=[''.join(i) for i in itertools.permutations(w)]
    a.sort()
    n=a.index(w)
    if(n+1!=len(a)):
        if(a[n+1]!=w):
            print a[n+1]
        else:
            print 'no answer'
    else:
        print 'no answer'
    t=t-1
