for T in range(int(raw_input())):
    sol = set()
    cnt = 0
    maxcnt = 0
    for N in range(int(raw_input())):
        S, R = map(int, raw_input().split())

        if R == 1 and S not in sol:
            cnt += 1
        if R == 1:
            sol.add(S)
        if R == 0:
            maxcnt = max(cnt, maxcnt)
            cnt = 0
    maxcnt = max(cnt, maxcnt)
    print(maxcnt)
