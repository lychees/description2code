#include <iostream>
using namespace std;

#define ll long long
const int  M = 1e9 + 7;

ll  largenumber(){

	string srr; cin >> srr;
	ll lnum = 0;
	for (int i = 0; i<srr.size(); i++){

		lnum = ((lnum * 10) % (M - 1) + (srr[i] - '0')) % (M - 1);


	}
	//cout<<lnum<<endl;
	return lnum;
}
ll mypow(ll a, ll b) {
	if (b == 0) return 1;
	if (b % 2 == 0)
	{
		ll c = mypow(a, b / 2);
		return (c * c) % M;
	}
	return (a * mypow(a, b - 1)) % M;
}
int main()
{
	ll int a, b;
	//string b;
	cin >> a;
	b = largenumber();


	b = b%M;
	cout << mypow(a, b) << endl;
	return 0;
}
