#include <stdio.h>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;
#define s(t) scanf("%d",&t)
#define ll long long int
#define mod 1000000007
int A[1009][1009];
ll S[1009];	

int  check(ll x)
{
	ll mid, low = 1, high = 1000000;
	while(low <= high )
	{
		mid = (low + high)/2;
		ll product = mid * mid * mid;
		
		if(x == product)
			return 1;
		else if(x < product)
			high = mid-1;
		else
			low = mid + 1;
	}
	return 0;
}
int main()
{
	int n, k, i, j, count = 0;
	s(n); s(k);	
	for(i=1 ; i<=n; i++)
	{
		for(j=1; j<=n; j++)
		{
			s(A[i][j]);
		}
	}

	for(j=1; j<=n; j++)
	{
		S[j] = 0;
	}

	for(i=1 ; i<k; i++) // noted
	{
		for(j=1; j<=n; j++)
		{
			S[j] += A[i][j];
		}
	}

	for(i=k; i<=n; i++)
	{
		for(j=1; j<=n; j++)
		{
			S[j] += A[i][j];

			//printf("%lld ", S[j]);
		}
		//printf("\n");
		ll left_sum = 0;
		for(j=1; j<k; j++)
		{
			left_sum += S[j];	
		}
		
		for(j=k; j<=n; j++)
		{
			left_sum += S[j];
			count += check(left_sum);
			//printf("%lld ", left_sum);
			left_sum -= S[j-k+1];	
		}
		
		//printf("\n");
		for(j=1; j<=n; j++)
		{
			S[j] -= A[i-k+1][j];	
		}
	}
	printf("%d", count);	
	return 0;
}
