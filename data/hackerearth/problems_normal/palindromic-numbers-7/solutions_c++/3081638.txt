int palin(int n)
{
	int temp = n;
	int r,sum=0;
	while(temp)
	{
		r = temp%10;
		temp = temp/10;
		sum = sum*10 + r;
	}
	if(sum==n)
		return 1;
	else 
		return 0;
}
