#include <iostream>
#include <cstdio>
using namespace std;
const int MOD=1000000007;
const int MAX=100005;
int v[MAX];

int main()
{
    int T,i;
    cin >> T;
    v[0]=1;
    v[1]=1;
    v[2]=2;
    v[3]=4;
    for(i=4;i<MAX;i++)
    	v[i]=((v[i-1]+v[i-2])%MOD+(v[i-3])%MOD)%MOD;
    while(T--){
    	int N;
    	cin >> N;
    	printf("%d\n",v[N]);
    }
    return 0;
}
