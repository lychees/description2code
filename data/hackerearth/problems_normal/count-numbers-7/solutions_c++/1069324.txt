#include <iostream>
using namespace std;

int main()
{
    int64_t k,t,i,m[20],n[1050],h[1050]={0};
    cin>>k>>t;
    for(i=0;i<k;i++)
     {
     	cin>>m[i];
     }
    n[0]=m[0];
    int z=1;
    int l=1;
    h[0]=0;
    for(i=1;i<k;i++)
     {  int j=0;
       n[l]=m[i];
       h[l]=0;
       l++;
     	while(j<z)
     	 {  
     	 	if(n[j]>1000000000000000LL)
     	 	 {
     	 	 	continue;
     	 	 }
     	 	 if(h[j]==0)
     	 	  h[l]=1;
     	 	 else h[l]=0;
     	 	 n[l]=n[j]*m[i];
     	 	l++;
     	 	j++;
     	 }
     	 z=l;
     }
    while(t--)
     {
     	int64_t a,b,p=0;
     	cin>>a>>b;
     	for(i=0;i<l;i++)
     	 {
     	 	
     	 	 int64_t x=a/n[i];
     	 	if(x*n[i]!=a)
     	 	  x++;
     	 	  if(x*n[i]>b)
     	 	    continue;
     	 	int64_t ans=b-x*n[i];
     	 	ans=ans/n[i];
     	 	ans++;
     	   if(h[i]==0)
     	    p=p+ans;
     	   else p=p-ans;
     	 	
     	 }
     	 
     	 cout<<p<<"\n";
     }
}
