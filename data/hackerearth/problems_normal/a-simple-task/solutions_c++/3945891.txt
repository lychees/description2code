#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>

#define ll long long
#define ull unsigned long long
#define iloop(i,j,n) for(int i=j;i<n;i++)
#define dloop(i,j,n) for(int i=j;i>=n;i--)
#define all(v) v.begin(),v.end()
using namespace std;

int n;

int check(vector<int> &a,int x)
{
	if(x==1)
		return 0;
	int modval=a[0]%x;
	iloop(i,0,n)
	{
			// cout << a[i] << ' ' << x << endl;
		if(a[i]%x!=modval)
		{
			// cout << a[i] << ' ' << x << endl;
			return 0;
		}
	}
	return 1;
}

int main()
{
	cin >> n;
	vector<int> a(n);
	iloop(i,0,n)
	{
		cin >> a[i];
	}
	sort(all(a));
	vector<int> diff(n);
	iloop(i,1,n)
	{
		diff[i]=a[i]-a[i-1];
	}
	sort(all(diff));
	int d=diff[1];
	vector<int> main;
	// cout << d << endl;
	iloop(i,1,sqrt(d)+1)
	{
		// cout  << i << endl;
		if(d%i!=0)
			continue;
		int ans=i,ans1=d/i;
		if(check(a,ans)==1)
			main.push_back(ans);
		if(check(a,ans1)==1 && ans1!=ans)
			main.push_back(ans1);
	}
	sort(all(main));
	for(auto i=main.begin();i!=main.end();i++)
	{
		cout << *i << " ";
	}
	cout << endl;
}
