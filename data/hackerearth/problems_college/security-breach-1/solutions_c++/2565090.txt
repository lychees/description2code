#include<iostream>
using namespace std;
#include<string.h>
int main()
{
	char a[1000000];
	int i,j,n;
	cin>>n;
	for(j=0;j<n;j++)
	{
		cin>>a;
		for(i=0;a[i]!='\0';i++)
		{
			if(a[i]=='2')
			cout<<"cde";
			else if(a[i]=='6')
			cout<<"b3";
			else if(a[i]=='7')
			cout<<"acf";
			else if(a[i]=='9')
			cout<<"c6a";
			else
			continue;
		}
		cout<<"\n";
	}
}
