#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
	ll t, n, i, c, m;
    cin>>t;
    while(t--)
    {
    	m=0;c=0;
    	cin>>n;
    	ll a[n+1][2];
    	for(i=1;i<=n;i++)
    		cin>>a[i][0]>>a[i][1];
    	ll vis[1000000];
    	memset(vis,0,sizeof(vis));
    	for(i=1;i<=n;i++)
    	{
    		if(a[i][1]==1 && vis[a[i][0]]==0)
    		{
    			vis[a[i][0]]=1;
    			c++;
    		}
    		else if(a[i][1]==0)
    		{
    			if(c>m)
    				m=c;
    			c=0;
    		}
    	}
    	if(c>m)
    		m=c;
    	cout<<m<<endl;
    }
    return 0;
}
