#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    
    long long int t , i , j , k , count , x , y , n ,ans;
    cin>>t ;
    while(t--)
    {
    	 ans = 0 , count = 0 ;
    	 cin>>n ;
    	 int a[n+1] , d[n+1] ;
    	 for(i = 0 ; i < n ; i++)
    	 cin>>a[i] ;
    	 
    	 for(i = 0 ; i < n ; i++)
    	 {
    	 	cin>>d[i] ;
    	 	d[i] += a[i] ;
    	 }
    	 sort(a , a+ n) ;
    	 sort(d , d+n) ;
    	 i = 0 ; j = 0 ; 
    	 while(i < n)
    	  {
    	  	 
    	  	if(a[i] < d[j])
    	  	 {
    	  	 	count++ ;
    	  	 	i++ ;
    	  	 	
    	  	 }
    	  	 
    	  else if(d[j] <= a[i] )
    	  	 {
    	  	 	 j++ ;
    	  	 	 count--;
    	  	 }
    	  	   if(count > ans)
    	  		ans = count ;
    	 
    	  }
    	  
    	  cout<<ans<<endl;
    	  
    }
    
    return 0;
}
