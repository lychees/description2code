#include <iostream>
#include <string>
#include <algorithm>
#include <map>
using namespace std;

int main()
{
int t;
cin>>t;
while(t--)
{
   bool isPossible =false;
    int n,l;
    cin>>n>>l;
    vector < pair<int,int> > a;
        int x, y;
    for(int i=0; i < n  ;i++)
    {
    
        cin>>x>>y;
        a.push_back(make_pair(x,y));
    }
    
 
    sort(a.begin(),a.end());

    for ( int i=0 ;  i < n ;i++)
    {
        int ok=1;
        int r = a[i].second;

        for(int j=i;j<n ;j++)
        {
            if (a[i].first+l >= a[j].second )
            {
                if (a[j].first> r)
                {
                    ok=0;
                }
                else
                {
				
                    r=max(r,a[j].second);
                }

            }
        }

            if(r!=a[i].first+l)
                ok=0;
           
		    isPossible |= ok;
        }
    

    if (isPossible)
        cout<<"Yes"<<endl;
    else
        cout<<"No"<<endl;
    }
    return 0;

}
