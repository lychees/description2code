#include <iostream>
#include<stdio.h>
#include<stdlib.h>
#include<math.h>
using namespace std;

int main()
{
    int t,p,x,y,u,v;
    scanf("%d",&t);
    
    while(t--) {
    	scanf("%d %d %d %d %d",&p,&x,&y,&u,&v);
    	x=(x>=0)?x:-x;
    	y=(y>=0)?y:-y;
    	u=(u>=0)?u:-u;
    	v=(v>=0)?v:-v;
    	
    	if(x==u && y==v) {
    		if(p>0)
    		printf("Counter-Terrorists Win !\n");
    		else
    		printf("Terrorists Win !\n");
    	}
    	else if((x+y)%2==(u+v)%2) {
    		if(abs(x-u) == abs(y-v)) {
    			if(p>1) {
    				printf("Counter-Terrorists Win !\n");
    			} else 
    			printf("Terrorists Win !\n");
    		} else {
    			if(p>2) 
    			printf("Counter-Terrorists Win !\n");
    			else
    			printf("Terrorists Win !\n");
    		}
    	} else 
    		printf("Terrorists Win !\n");
    	
    }
    
    
    return 0;
}
