a = int(raw_input())
for i in range(a):
    n = int(raw_input())
    # print 'n is %d' % n
    flag = 'N'
    l = n/4

    # print 'l is %d' % l
    x = (n-2)/4
    b = x+1

    # print 'x is %d ' % x
    # print 'printing %s' % (2*x+2*b)
    if 2*x+2*b <= n and x*b > l*l:
        flag = 'Y'
        # print 'flag is set'

    if flag == 'Y':
        print x*b
    else:
        print l*l
