#include<iostream>
using namespace std;
int main(){
	string s;
	cin>>s;
	int i,count;
	for(i=0,count=0;i<s.length();i++){
		if(count==0&&s[i]=='l')
		++count;
		else if(count==1&&s[i]=='o')
		++count;
		else if(count==2&&s[i]=='v')
		++count;
		else if(count==3&&s[i]=='e')
		++count;
	}
	if(count==4)
	cout<<"I love you, too!";
	else
	cout<<"Let us breakup!";
	return 0;
}
