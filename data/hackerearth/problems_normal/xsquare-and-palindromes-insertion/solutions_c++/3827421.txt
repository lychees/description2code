#include<bits/stdc++.h>
using namespace std;


int main()
{
      int t;
      cin>>t;
      while(t--)
      {
            string s;
            cin>>s;
            map<char,int>mp;
            for(int i=0;i<s.length();i++)
            {
                  mp[s[i]]++;
            }

            int ans=0;

            bool f=false;
            char c;
            for(char i='a';i<='z';i++)
            {
                 if(mp[i]%2!=0)
                 {
                       c=i;
                       f=true;
                       break;
                 }
            }

        //cout<<c<<endl;
        for(char i='a';i<='z';i++)
        {

               if(mp[i]%2!=0&&i!=c)
               {
                     ans++;
               }
        }

        cout<<ans<<endl;
}
return 0;

}
