for _ in range(int(raw_input())):
    m, n = map(int, raw_input().split())
    ma, na = [], []
    for __ in xrange(m):
        ma.append(raw_input())
    for __ in xrange(n):
        na.append(raw_input())
    g, h = map(int, raw_input().split())
    if len(set(ma) & set(na)) >= h:
        print "Great"
    elif len(set([int(i.split()[0]) for i in ma]) & set([int(i.split()[0]) for i in na])) >= g:
        print "Good"
    else:
        print ":("
