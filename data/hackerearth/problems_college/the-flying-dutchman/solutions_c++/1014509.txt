#include<bits/stdc++.h>
using namespace std;

int main()
{
int t,n,i,j;
scanf("%d",&t);
while(t--)
 {scanf("%d",&n);
  vector<long long> a(n+1),sum(n+1),soln(n+1);
  for(i=0;i<n;i++)
   scanf("%d",&a[i]);
 for(i=n-1;i>=0;i--)
  {sum[i]=a[i]+sum[i+1];
  }
  soln[n-1]=a[n-1];
  for(i=n-2;i>=0;i--)
   {soln[i]=max(a[i]+sum[i+1]-soln[i+1],a[i]+a[i+1]+sum[i+2]-soln[i+2]);
   }
   if(soln[0]*2==sum[0])
     cout<<"DRAW"<<endl;
    else if(soln[0]*2<sum[0])
     {cout<<"WIN"<<endl;
     }
     else
     cout<<"LOSE"<<endl;

 }

}
