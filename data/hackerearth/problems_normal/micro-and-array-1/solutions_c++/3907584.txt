#include <iostream>
#include <vector>
using namespace std;

int main()
{
  int n, q, input1, position;
  vector<long int> arr;
  cin >> n >> q;
  for(int i = 0; i < n; i++) {
    cin >> input1;
    arr.push_back(input1);
  }
  for(int i = 0; i < q; i++) {
    int operation, first_int, second_int;
    cin >> operation;
    if (operation == 1) {
      position = -1;
      cin >> first_int;
      for (vector<long int>::iterator it = arr.begin() ; it != arr.end(); ++it) {
          if (*it >= first_int) {
            position = it - arr.begin() + 1;
            break;
          }
      }
      cout << position << endl;
    } else {
      cin >> first_int >> second_int;
      arr[first_int - 1] = second_int;
    }
  }
  return 0;
}
