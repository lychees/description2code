#include<iostream>
#include<cstdio>
#define N 21
using namespace std;

int min(int a,int b)
{
    if(a<b)
    return a;
    return b;
}

int min(int a,int b,int c)
{
    if(a<=b && a<=c)
    return a;
    if(b<=c && b<=a)
    return b;
    if(c<=a && c<=b)
    return c;
}

int main()
{
    int i=0,t,j,n,house[N][3];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(i=0;i<n;i++)
        scanf("%d%d%d",&house[i][0],&house[i][1],&house[i][2]);
        int cost[n][3];
        for(i=0;i<3;i++)
        cost[0][i]=house[0][i];
        for(i=1;i<n;i++)
        {
            cost[i][0]=min(cost[i-1][1],cost[i-1][2])+house[i][0];
            cost[i][1]=min(cost[i-1][2],cost[i-1][0])+house[i][1];
            cost[i][2]=min(cost[i-1][1],cost[i-1][0])+house[i][2];
        }
        printf("%d\n",min(cost[n-1][0],cost[n-1][1],cost[n-1][2]));
    }
    return 0;
}
