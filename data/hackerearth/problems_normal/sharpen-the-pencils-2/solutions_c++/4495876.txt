#include <iostream>
using namespace std;

int main()
{
    int t,n,a[10000000];
    cin>>t;
    for(int i=0;i<t;i++){
    	cin>>n;
    	for(int j=0;j<n;j++){
    		cin>>a[j];
    	}
    	int count1=0,count2=0,sum1=0,sum2=0;
    	int x=0;
    	int y=n-1;
    	while(x<y){
    		if(sum1<sum2){
    			sum1+=a[x];
    			x++;
    			count1++;
    		}
    		else if(sum2<sum1){
    			sum2+=(2*a[y]);
    			y--;
    			count2++;
    		}
    		else{
    			sum2+=(2*a[y]);
    			y--;
    			count2++;
    			sum1+=a[x];
    			x++;
    			count1++;
    		}
    	}
    	if(x==y && sum1<=sum2){
    		count1++;
    	}
    	if(x==y && sum1>=sum2){
    		count2++;
    	}
    	cout<<count1<<" "<<count2<<"\n";
    	
    }
    return 0;
}
