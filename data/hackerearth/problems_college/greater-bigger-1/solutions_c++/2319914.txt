#include <iostream>
#include <algorithm>
#include <ctype.h>
#include <string.h>
using namespace std;
 
int main()
{
    int t, i, n, j, pos;
    char w[1000];
    char s;
    bool flag;
    scanf("%d", &t);
    while(t--)
    {
    	flag=0;
    	cin>>w;
    	n=strlen(w);
    	for(i=n-1;i>0;i--)
    	{
    		if(w[i]>w[i-1])
    		{pos=i-1;flag=1;break;}
    		flag=0;
    	}
    	if(flag==0)
    	{printf("no answer\n");continue;}
    	//sorting
    	sort(w+pos+1, w+n);
    	for(i=pos+1;i<n;i++)
    	{
    		if(w[i]>w[pos])
    		{
    				s=w[pos];
    				w[pos]=w[i];
    				w[i]=s;
    				break;
    		}
    	}
    	cout<<w<<"\n";
    }
    return 0;

}
