#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k,m,i;
    	cin>>n>>k>>m;
    	vector <string> arr;
    	string s;
    	for(i=0;i<n;i++)
    	{
    		
    		cin>>s;
    		arr.push_back(s);
    		
    	}
    	sort(arr.begin(),arr.end());
    	
    	cout<<arr[k-1]<<endl;
    }
    return 0;
}
