#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;
int dp[10001][11];
int func(int x, int n, int &p, int c){
	if(x==n){
		return 1;
	}
	if(dp[x][c]!=-1){
		return dp[x][c];
	}
	int count_=0;
	if((c+1) < p){
		count_ = (count_ + func(x+1, n, p, c+1))%mod;
	}
	count_ = (count_ + func(x+1, n, p, 0))%mod;
	dp[x][c]=count_;
	return count_;
}
int main()
{
    int t, n, p;
    cin>>t;
    while(t--){
    	memset(dp, -1, sizeof(dp));
    	cin>>n>>p;
    	cout<<func(0, n-1, p, 0)<<endl;
    }
    return 0;
}
