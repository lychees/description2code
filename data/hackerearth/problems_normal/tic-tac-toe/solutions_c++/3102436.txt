#include <bits/stdc++.h>
using namespace std;

long long mod = 1000000007;
int pow_mod(long long a,long long b)
{
    long long res = 1;
    while(b)
    {
        if(b&1)
        {
            res = (res*a)%mod;
        }
            a = (a*a)%mod;
            b>>=1;


    }
    return res;
}

{
    long long t;
    cin>>t;


    while(t--)
    {
        long long n;
        cin>>n;
        long long minx = (n-1)/(2);
         minx = (minx*minx)%mod;
        minx = (n*(minx))%mod;


        long long maxx = (n*(n-1))%mod;
        maxx = maxx*((2*n)-1);
        maxx = ((maxx)%mod*(pow_mod(6,mod-2))%mod)%mod;



        cout<<minx<<" "<<maxx<<endl;

    }
}
