#include<bits/stdc++.h>
using namespace std;
int prime[10005];
int n,k,matrix[1005][1005];
int dx1[]={0,-1,-1,1,1};
int dy1[]={0,-1,1,-1,1};
bool inbound(int a,int b)
{
    if(a<=n-1 && a>=2 && b>=2 && b<=n)
        return true;
    else
        return false;
}
void sieve()
{
    fill(prime,prime+10005,0);
    prime[0]=1;
    prime[1]=1;
    for(int i=2;i*i<10005;i++)
    {
        if(prime[i]==0){
        for(int j=2*i;j<10005;j=j+i)
        {
            prime[j]=1;
        }
        }
    }
}
int main()
{
    sieve();

    cin>>n>>k;
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=n;j++)
            cin>>matrix[i][j];
    }
    int l=0,m=0,max1=10005,max2=0,max3=0;
     for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=n;j++)
        {
            l=0;
            m=0;
            if(inbound(i,j))
            {
               for(int z=0;z<5;z++)
               {
                   if(prime[matrix[i+dx1[z]][j+dy1[z]]]==1)
                   {
                     int v=  matrix[i+dx1[z]][j+dy1[z]];
                      while(v<=10000)
                      {
                         if(prime[v]==0){
                            l++;
                            break;}
                         else{
                            v=v+k;
                            m++;}
                      }
                   }
                   else
                    l++;
               }
            }
            if(l==5){
                if(m<max1)
                {
                    max1=m;
                    max2=i;
                    max3=j;
                }
            }
        }

    }
    if(max2==0 && max3==0)
        cout<<"no"<<endl;
    else
    {
        cout<<"yes"<<endl;
        cout<<max1<<endl;
        cout<<max2<<" "<<max3<<endl;
    }
    return 0;
}
