test=int(raw_input())
for i in range(test):
    l=list(map(int,raw_input().split()))
    a=l[0]
    b=l[1]
    c=l[2]
    k=l[3]
    j=0
    p=j
    while True:
        if k<=(a*j*j+b*j+c):
            break
        p=j
        if j==0:
            j=1
        else:
            j=j*2
    while p!=j:
        m=(p+j)//2
        if k<=(a*m*m+b*m+c):
            j=m
        else:
            p=m+1

    print(p)
