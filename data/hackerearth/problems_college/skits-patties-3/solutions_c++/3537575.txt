#include<bits/stdc++.h>
using namespace std;
int a[int(1e7)+3];
int main()
{
	//memset();
	a[6]=a[20]=a[9]=1;
	for(int i=6;i<=int(1e7);++i)
		a[i]+=a[i-6];
	for(int i=9;i<=int(1e7);++i)
		a[i]+=a[i-9];
	for(int i=20;i<=int(1e7);++i)
		a[i]+=a[i-20];
	int tc;

	cin>>tc;
	while(tc--)
	{
		int temp;
		cin>>temp;
		(a[temp])?puts("True"):puts("False");
	
	}
	return 0;
}
