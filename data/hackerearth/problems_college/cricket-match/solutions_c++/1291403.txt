#include<bits/stdc++.h>
using namespace std;
typedef vector<int> vi;
vi a(400006);
int find(int s)
{
    if(a[s]==s)
        return s;
    return a[s]=find(a[s]);
}
void init(int n)
{
    for(int i=0;i<=2*n;i++)
    {
        a[i]=i;
    }
}
int main()
{
    int t,n,aa,b,m;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        init(n);
        int f=0;
        while(m--)
        {
            cin>>aa>>b;
            int l=aa+n,k=b+n;
            if((find(aa)==find(b))||(find(k)==find(l)))
                f=1;
            else
            {
                a[find(b)]=find(l);
                a[find(aa)]=find(k);
            }
        }
        if(f==1)
            cout<<"NO\n";
        else cout<<"YES\n";
    }
    return 0;
}
