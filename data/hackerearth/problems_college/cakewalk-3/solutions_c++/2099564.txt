#include <iostream>
#include<string>
using namespace std;
using std::string;
int main()
{
    int n,a[1000],i,j,k;
    string s[1000],temp[1000]={""};
    cin >>n;
    cin>>s[0];
    for(i=1;i<n;i++)
    {
    	cin>>a[i-1];
    	cin>>s[i];
    }
    temp[0]=s[0];
    //cout<<temp[0]<<endl;
    for(i=1;i<n;i++)
    {
    	for(j=0;j<a[i-1];j++)
    	{
    		temp[i]+=temp[i-1][j];
    		//cout << temp[i][k-1]<<" ";
    	}
    	temp[i]+=s[i];
    	//cout << temp[i]<<endl;
    }
    cout<<temp[n-1]<<endl;
    return 0;
}
