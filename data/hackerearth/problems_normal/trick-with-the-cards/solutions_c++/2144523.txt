#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	long long int n;
    	long long int ans;
    	cin>>n;
    	if(n<=3)
    	{
    		if(n==0)
    		{
    			ans = 0;
    		}
    		else
    		{
    			ans = 1;
    		}
    	}
    	else if(n%3==0)
    	{
    		ans = (n/3);
    	}
    	else
    	{
    		//ans = ( (n + n + n -3)/3 ) + 1;
    		ans = n;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
