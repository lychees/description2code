using namespace std;
int main()
{
    /*freopen("1in.txt", "r", stdin);
    freopen("1out.txt", "w", stdout);*/
    int t; cin >> t;
    int b,i,val;
    char oper;
    ll minarr, maxarr;
    while(t--)
    {
        cin >> b;
        //fill(minarr,minarr+60,0);
        //fill(maxarr,maxarr+60,0);
        minarr = 1;
        maxarr = 1;
        iter(i,1,b+1,1)
        {
            cin >> oper;
            if(oper == 'N')
            {
                ll ntemp = -minarr;
                ll xtemp = -maxarr;
                minarr = min(minarr,xtemp);
                maxarr = max(maxarr,ntemp);
            }
            else
            {
                cin >> val;
                switch(oper)
                {
                    case '+':
                        maxarr = max(maxarr,(maxarr+val));
                        minarr = min(minarr,minarr+val);
                        break;
                    case '-':
                        maxarr = max(maxarr,maxarr-val);
                        minarr = min(minarr,minarr-val);
                        break;
                    case '*':
                        maxarr = max(maxarr,maxarr*val);
                        minarr = min(minarr,minarr*val);
                        break;
                    case '/':
                        maxarr = max(maxarr,maxarr/val);
                        minarr = min(minarr,minarr/val);
                        break;
                }
            }
        }
        printf("%lld\n",maxarr);
    }
    return 0;
}
