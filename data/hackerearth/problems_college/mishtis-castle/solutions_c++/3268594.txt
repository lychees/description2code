long long int a[250][250];
int t, n;

int main()
{
    int c, x;

    cin >> t;

    while(t--) {
        cin >> n >> c;
        ans = 0;
        memset(a, 0, sizeof(a));

        for(int i = 1; i <= n; i++) {
            for(int j = 1; j <= n; j++) {
                scanf("%d", &x);

                a[i][j] = x;
            }
        }

        for(int i = 1; i <= n; i++) {
            for(int j = 1; j <= n; j++) {
                a[i][j] += (max(a[i-1][j], a[i][j-1]));
            }
        }

        ans = a[n][n];

        ans -= c;

        if(ans >= 0)
            cout << ans <<endl;
        else
            cout << -1 <<endl;


    }
}
