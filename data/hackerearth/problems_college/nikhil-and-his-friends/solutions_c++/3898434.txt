#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int  main(int argc, char const *argv[])
{
	int T;
	cin>>T;
	while(T--){
		ll n,sum;
		cin>>n>>sum;
		ll a[n+1],i;
		for (i = 1; i <= n; ++i)
		{
			cin>>a[i];
		}
		ll res=0,j;
		a[0]=0;
		int flag=0;
		res=a[1];
		for(i=1,j=i+1;i<=n;i++){
			res-=a[i-1];
			if(sum==res){
				flag=1;
				break;
			}
			for(j;j<=n && res<sum;j++){
				res+=a[j];
				if(sum==res){
				flag=1;
				break;
			}
			}
		}
		//cout<<res<<endl;
		if(flag==1)
			cout<<"YES\n";
		else
			cout<<"NO\n";
	}
	return 0;
}
