#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	char str1[100000];
	cin>>t;
	cin.getline(str1,1000);
	while(t--)
	{
		char str[100000];
		
		
		cin.getline(str,10000);
		int l=strlen(str);
		int s=sqrt(l);
		char mat[s][s];
		int k=0;
		for(int i=0;i<s;i++)
		{
			for(int j=0;j<s;j++)
			{
				mat[j][i]=str[k++];
			}
		}
		for(int i=0;i<s;i++)
		{
			for(int j=0;j<s;j++)
			{
				cout<<mat[i][j];
			}
		}
		cout<<endl;
	}
}
