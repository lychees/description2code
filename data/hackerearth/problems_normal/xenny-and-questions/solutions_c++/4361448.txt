#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main(){
	ios::sync_with_stdio(false);
	ll t;cin>>t;
	while(t--){
		ll n, k;cin>>n>>k;
		ll a[100100]={0};
		for(int i=0; i<n; i++){
			int temp;cin>>temp;
			a[temp]++;
		}
		ll i=100000, sum=0;
		while(k && i){
			if(a[i]){
				sum+=i;
				a[i]--;
				k--;
			}
			else 
				i--;
		}
		cout<<sum<<endl;
	}
	return 0;
}
