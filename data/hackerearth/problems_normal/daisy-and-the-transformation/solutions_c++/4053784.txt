#include <iostream>
using namespace std;

int main()
{
    int tc; cin>>tc;
    while(tc--){
    	getchar();
    	string s1,s2; int n,cnt=0;
    	cin>>s1; cin>>s2; cin>>n;
    	int l1=s1.length(),l2=s2.length();
    	if(l1!=l2) { cout<<"No\n"; continue; }
    	else {
    		for(int i=0;i<l1;i++){
    			if(s1[i]!=s2[i]) cnt++;
    		}
    		if(cnt==n) { cout<<"Yes\n"; continue; }
    		else if(cnt>n) { cout<<"No\n"; continue; }
    		else {
    			if((n-cnt)%2==0) cout<<"Yes\n";
    			else cout<<"No\n";
    		}
    	}
    }
    return 0;
}
