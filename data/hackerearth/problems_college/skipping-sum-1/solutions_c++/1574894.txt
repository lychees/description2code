#include <bits/stdc++.h>
using namespace std;

#define SIZE 100010

int a[SIZE];
long long psum[SIZE][11];

void preprocess(int n) {
	for(int k = 1; k <= 10; k++) 
		for(int i = 1; i <= n; i++) 
			psum[i][k] = a[i] + ((i-k>0) ? psum[i-k][k] : 0);
}

int main(){
	int n, q, l, r, k;    
	cin >> n >> q;
	for(int i = 1; i <= n; i++)	cin >> a[i];
	preprocess(n);
	for(int i = 1; i <= q; i++) {
		cin >> l >> r >> k;
		cout << psum[l+((r-l)/k)*k][k] - psum[l][k] + a[l] << endl;
	}
    return 0;
}
