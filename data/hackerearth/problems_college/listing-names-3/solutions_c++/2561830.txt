#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    map<string,int>mymap;
    map<string,int>::iterator it;
    int i;
    for(i=0;i<n;i++)
    {
    	string str;
    	cin>>str;
    	mymap[str]++;
    }
    for(it=mymap.begin();it!=mymap.end();it++)
    {
    	cout<<(*it).first<<" "<<(*it).second<<endl;
    }
}
