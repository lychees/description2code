#define INT long long
INT arr[1000056];

void markMultiples(INT arr[],INT a,INT n)
{
    INT i=a,num;
    while((num=i*a)<=n)
    {
        arr[num-1]=i;
        ++i;
    }
}
void sieve(INT n)
{
    if(n>=2)
    {
        memset(arr,0,sizeof(arr));
        for(INT i=1;i<n;++i)
        {
            if(arr[i]==0)
            {
                arr[i]=1;
                if(i<=sqrt(n)+1)
                {
                    markMultiples(arr,i+1,n);
                }
            }
            else
            {
                arr[i]=1+arr[arr[i]-1];
            }
        }
        for(INT i=1;i<n;i++)
        {
            arr[i]+=arr[i-1];
        }
    }
}
