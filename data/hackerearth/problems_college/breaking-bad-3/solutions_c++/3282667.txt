#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	vector<ll> v;
    	int i;
    	ll x;
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&x);
    		v.push_back(x);
    	}
    	ll ans = 0;
    	sort(v.begin(),v.end());
    	reverse(v.begin(),v.end());
    	for(i=0;i<n/2;i++)
    	ans+= v[2*i] * v[2*i+1];
    	cout<<ans<<endl;
    }
    return 0;
}
