#include<bits/stdc++.h>
using namespace std;
int main()
{long long int n,c,a,i,r,ans;
vector<int>v;
   cin>>n;
   c=n;
   while(c--)
   {
       cin>>a;
       v.push_back(a);
       
   }
   sort(v.begin(),v.end());
   ans=max(v[n-1]-v[n-2],v[n-1]-v[n-3]);
   for(i=0;i<n-2;i++)
   {
       r=v[i+2]-v[i];
       ans=max(ans,r);
   }
   cout<<ans<<endl;
   return 0;
}
