#include<iostream>
using namespace std;
int main()
{
	int t; cin>>t;
	while(t--)
	{
		int n; cin>>n;
		int a=0,b=0;
		while(n!=0)
		{
			if(n%2==0)
			{
			 a=a+1;	
			}
			else
			{
				b=b+1;
			}
			n=n/2;
		}
		if(b%2==0)
	    cout<<"even"<<endl;
	    else
	    cout<<"odd"<<endl;
	}
	return 0;
}
