#include <iostream>
using namespace std;
int sum(int n)
{
	int sm=0;
	while(n>0)
	{
		sm+=n%10;
		n=n/10;
	}
	return sm;
}
int main()
{
    int t=100000,n,max=0,ans;
    while(t--)
    {
    	n=1;
    	cin>>n;
    	if(sum(n)>max) {
    		max=sum(n);
    		ans=n;
    	}
    }
    cout<<ans;
    return 0;
}
