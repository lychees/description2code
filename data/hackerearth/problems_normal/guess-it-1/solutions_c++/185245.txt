#include <iostream> 
using namespace std; 
int main() 
{ 
int sum,n,i,j; 
string num[100]; 
cin>>n; 
for(i=0;i<n;i++) 
{ 
cin>>num[i]; 
} 
for(i=0;i<n;i++) 
{ 
sum=-47; 
for(j=0;num[i][j]!='\0';j++) 
sum+=num[i][j]-48; 
sum=sum%9; 
sum=9-sum; 
if(sum==9) 
cout<<"0 or 9"; 
else cout<<sum; 
cout<<endl; 
} 
return 0; 
}
