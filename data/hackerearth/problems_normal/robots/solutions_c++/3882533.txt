#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	vector<int>v;
    	map<int,int>m;
    	for(int i=0;i<n;i++){
    		int val;
    		cin>>val;
    		v.push_back(val);
    		m[val]++;
    	}
    	sort(v.begin(),v.end());
    	long long int ans=1;
    	for(int i=0;i<v.size();i++)
    		ans=((ans*(v[i]-i))%1000000007);
    	cout<<ans<<endl;
    }
    return 0;
}
