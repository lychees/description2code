#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	for(int x=0; x<t;x++){
		unsigned int n;
		cin>>n;
		vector<unsigned int> v(n);
		unsigned int l=0,p=0;
		for(unsigned int i=0;i<n;i++){
			cin>>v[i];
		}
		for(unsigned int i=1;i<(1<<n);i++){
			unsigned int a=0,b=~0u,c=0;
			for(unsigned int j=0;j<=n;j++)
				if((1<<j)&i){
					c++;
					a=a^v[j];
					b=b&v[j];
				}
				if((a|b)>p){
					p=a|b;
					l=c;
				}else if((a|b)==p){	l=max(l,c);	}
			}
			
		
		cout<<p<<" "<<l<<endl;
	}
	
    return 0;
}
