#include <iostream>
using namespace std;

int main()
{
	long long int t,n,i,sum,a[1005];
    scanf("%lld",&t);
    while(t--)
    {
    	sum=0;
    	scanf("%lld",&n);
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&a[i]);
    		sum+=a[i];
    	}
    	if(sum%n==0)
    	{
    		printf("YES\n");
    		printf("%lld 0\n",sum/n);
    	}
    	else
    	{
    		printf("NO\n");
    		printf("%lld %lld\n",sum/n,sum%n);
    	}
    }
    return 0;
}
