#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
     while(t--)
     {  int flag=1;
     	unsigned long long int n,k;
     	cin>>n;
     	if(n<=0)
        {cout<<"NO\n";
     	 continue;
         }
     	k=sqrt(n);
     	for(long long int i=2;i<=k;i++)
     	   {  if((n%i)==0)
     	    { cout<<"NO\n";
     	      flag=0;
     	      break;
     	    }
     	   } 
     	if(flag==1)
     	  cout<<"YES\n";
     }
    return 0;
}
