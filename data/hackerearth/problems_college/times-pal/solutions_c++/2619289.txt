#include <bits/stdc++.h>
using namespace std;
bool ispal(string s)
{
	int flag=0;
	long long l,i,j;
	l=s.length();
	j=l-1;
	for(i=0;i<l/2;i++)
	{
		if(s[i]!=s[j--])
		{
			flag=1;
			break;
		}
	}
	if(flag)
	return false;
	else
	return true;
}
int main()
{
    int t,c,l;
    string a,b,o;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	
    	c=0;
    	l=a.length();
    	for(int i=0;i<l;i++)
    	{
    		o=a;
    		o.insert(i,b);
    		if(ispal(o))
    		c++;
    	}
    	o=a;
    	o+=b;
    	if(ispal(o))
    	c++;
    	cout<<c<<"\n";
    }
    return 0;
}
