#include<bits/stdc++.h>
using namespace std;
int N,K;
int parent[100005];
int main()
{scanf("%d%d",&N,&K);
 int i;
 for (i=1;i<=N;i++)
  parent[i]=i;
 for (i=0;i<K;i++)
  {int u,v;
   scanf("%d%d",&u,&v);
   parent[v]=u;
  }
  int count=0;
  for (i=1;i<=N;i++)
   {if(parent[i]==i)
     count++;
   }
   printf("%d",count);
}
