#include <iostream>
using namespace std;
bool b[1000001],a;
int main()
{
	int t,n,i,j;
	long k;
	cin>>t;
	for(i=0;i<1000001;i++)
	b[i]=0;
	for(i=2;i<=1000;i++)
	{
		k=i;
		for(;;)
		{
		k=k*i;
		if(k>1000000)
		break;
		b[k]=1;
		}
	}
	b[0]=b[1]=1;
	for(i=0;i<t;i++)
	{
		cin>>n;
		a=0;
		for(j=1;j<=n/2;j++)
		{
			if(b[j]==1 && b[n-j]==1)
			{
				cout<<"Yes"<<endl;
				a=1;
				break;
			}
		}
		if(a==0)
		cout<<"No"<<endl;
	}
}
