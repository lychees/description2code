#include<bits/stdc++.h>
using namespace std;
int main()
{
	int test;int t;int i;int a,b;
	cin>>test;int n,k;
	for(t=0;t<test;t++)
	{
		cin>>n>>k;
		if (k>9*9)
		{
			cout<<"-1\n";
		}
		else
		{
			int j;
		int count1=0;int flag=0;
		for(i=1;i<=9;i++)
		{
			for(j=0;j<=9;j++)
			{
				if (i!=j)
				count1++;
				if (count1==k)
				{
					 a=i;
					 b=j;
					flag=1;
					break;}
			}
			if (flag==1)
			break;
		}
		i=0;
		while(i<n)
		{
			if (i<n)
			cout<<a;
			i++;
			if (i<n)
			cout<<b;
			i++;
		}
		cout<<"\n";
		}
	}//test
	
	return 0;
}
