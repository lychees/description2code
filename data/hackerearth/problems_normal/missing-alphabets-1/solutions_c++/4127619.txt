#include <bits/stdc++.h>
using namespace std;
map<char,int> m;
bool comp(string a,string b)
{
	int f,i,l;
	if(a.size()<=b.size())
		l=a.size();
	else
		l=b.size();
	for(i=0;i<l;i++)
	{
		if(m[a[i]]<m[b[i]])
			return true;
		if(m[a[i]]>m[b[i]])
			return false;
	}
	if(l==a.size())
	{
		return true;
	}
	else
	{
		return false;
	}
}
int main()
{
    int t,i;
    string s;
    cin>>t;
    while(t--)
    {
    	
    	cin>>s;
    	for(i=0;i<s.size();i++)
    	{
    		m[s[i]]=i+1;
    	}
    	int n;
    	vector<string> v;
    	vector<string>::iterator it;
    	cin>>n;
    	while(n--)
    	{
    		cin>>s;
    		v.push_back(s);
    	}
    	sort(v.begin(),v.end(),comp);
    	for(it=v.begin();it!=v.end();it++)
    	{
    		cout<<*it<<endl;
    	}
    	m.erase(m.begin(),m.end());
    }
    return 0;
}
