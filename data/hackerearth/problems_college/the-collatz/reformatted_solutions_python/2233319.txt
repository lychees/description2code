for tc in range(int(input())):
    n = input()
    esum, osum, count = 0, 1, 0
    while n != 1:
        if n % 2 == 0:
            esum += n
            n = n//2
        else:
            osum += n
            n = 3*n+1
        count += 1
    if esum % count > osum % count:
        print('Even Rules')
    elif esum % count < osum % count:
        print('Odd Rules')
    else:
        print('Tie')
    # print(esum,osum,count)
