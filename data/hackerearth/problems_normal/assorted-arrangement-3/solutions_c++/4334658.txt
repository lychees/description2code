int c[55];

int main()
{
	ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    #ifndef LOCAL_SYS
        //freopen("in.txt","r",stdin);

    int n,m,i,j,z,k,ans=0;
    scanf("%d%d",&n,&m);
    for(i=1;i<=m;i++)
    	scanf("%d",&c[i]);
    for(i=1;i<=n;i++)
    {
    	scanf("%d",&a[i]);
    	j=ans/c[a[i]]+1;
    	j*=c[a[i]];
    	//continue;
    	while(true)
    	{
    		z=1;
			for(k=(a[i]+1);k<=m;k++)
				if(j%c[k]==0)
				{
					z=0;
					break;
				}
			if(z)
			{
				a[i]=j;
				//printf("%d\n",j);
				ans=max(ans,j);
				break;
			}
    		j+=c[a[i]];
    	}
    }
    printf("%d\n",ans);
    return 0;
}
