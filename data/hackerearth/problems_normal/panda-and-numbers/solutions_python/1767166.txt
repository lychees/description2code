rr=raw_input
panda = set([k**k for k in xrange(1,8)])
dp = {}
for i in xrange(10**6+1):
    dp[i] = None
for i in panda: dp[i]=True

def dpp(x):
    if dp[x] is not None: return dp[x]
    for d in str(x):
        cand = x - int(d)**2
        if cand > 0 and dp[cand]: return True
    return False

for i in xrange(10**6+1):
    z = dpp(i)
    dp[i]=z
for tc in xrange(int(rr())):
    N = int(rr())
    if dp[N]: print "Yes"
    else: print "No"
