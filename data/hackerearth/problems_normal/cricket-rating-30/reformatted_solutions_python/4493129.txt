n = input()
if n == 0:
    print 0
else:
    arr = list(map(int, raw_input().split()))
    # kadane's algorithm

    def max_subarray(arr):
        mhere = mtill = 0
        for x in arr:
            mhere = max(0, mhere + x)
            mtill = max(mtill, mhere)
        return mtill

    print max_subarray(arr)
