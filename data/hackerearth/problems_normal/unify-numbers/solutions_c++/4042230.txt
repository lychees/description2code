#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin >> t;
    
    while (t--)
    {
    	int n;
    	cin >> n;
    	vector<long long> a;
    	a.clear();
    	for (int i = 0; i < n; i++)
    	{
    		long long j;
    		cin >> j;
    		a.push_back(j);
    	}
    	sort(a.begin(), a.end());
    	long long median = a[n/2];
    	long long sum = 0;
    	for (int i = 0; i < n; i++)
    	{
    		sum += abs(a[i] - median);
    	}
    	cout << sum << endl;
    }
    return 0;
}
