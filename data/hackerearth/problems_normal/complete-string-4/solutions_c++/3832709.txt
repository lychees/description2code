#include <iostream>
#include <string.h>
using namespace std;

int main()
{
	int n,i,j,status=0;
	char str[10][100];
	int check[10][26];
	cin>>n;
	for(i=0;i<n;i++)
	{
		for(j=0;j<26;j++)
		{
			check[i][j]=0;
		}
	}
	cin.ignore();
	for(i=0;i<n;i++)
	{
		status=0;
		cin.getline(str[i],100);
		for(j=0;j<strlen(str[i]);j++)
		{
			//cout<<str[i][j];
			check[i][str[i][j]-97]=1;
		}
		//cout<<endl;
		for(j=0;j<26;j++)
		{
			if(check[i][j]==0)
			{
				status=1;
			}
		}
		if(status==0)
		{
			cout<<"YES"<<endl;
		}
		else
		{
			cout<<"NO"<<endl;
		}
	}
	return 0;
}
