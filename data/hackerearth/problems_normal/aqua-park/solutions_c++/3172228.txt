#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,a;
	scanf("%d%d",&a,&n);
	vector<int> x1(n),x2(n),y1(n),y2(n);
	
	for(int i=0;i<n;i++){
		scanf("%d%d%d%d",&x1[i],&y1[i],&x2[i],&y2[i]);
	}
	int cury = 1000000, curx = a;
	int maxy = -1000000, x=0,y=0;
	for(int j=0;j<n;j++){
		maxy = -10000000;
	for(int i=0;i<n;i++){
		
		if(curx>=x1[i] && curx<=x2[i]){
			float temp = (float)(y2[i]-y1[i])/(float)(x2[i]-x1[i]) * (curx-x1[i]) + y1[i];
			if(cury>temp && maxy<temp) {
				maxy=temp;
				if(y2[i]>y1[i]){
					x = x1[i];
					y = y1[i];
				}
				else{
					x = x2[i];
					y = y2[i];
				}
			}
		}
	}
	cury = y;
	curx = x; 
	//cout << cury << " " << curx << " " << maxy << endl;
}


    cout << curx << endl;
    return 0;
}
