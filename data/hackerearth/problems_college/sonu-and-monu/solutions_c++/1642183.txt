{
   n=0;
   T ch=getcx();int sign=1;
   while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}

   while(  ch >= '0' && ch <= '9' )
           n = (n<<3)+(n<<1) + ch-'0', ch=getcx();

const int maxn = 100005;

LL num, den, ans, L, R, a[maxn];

LL gcd(LL a, LL b) {
    while(b) {
        a %= b;
        swap(a, b);
    }
    return a;
}


int main() {

    //freopen("test.in","r",stdin);
    //freopen("test.out","w",stdout);

    int T, n, p;
    string s;

    cin >> T;
    while(T--) {
        cin >> n >> p >> s;

        rep(i, n + 1) a[i] = 0;
        rbe(i, 1, n) a[i] = a[i - 1] + (s[i - 1] == '1');

        num = 0;
        rbe(i, 1, n) {
            if(s[i - 1] == '0') continue;
            L = max(1, i - p);
            R = min(i + p, n);
            num += (a[R] - a[L - 1]);
        }

        if(num == 0) {
            printf("0/1\n");
            continue;
        }

        den = 1LL * n * n;
        ans = gcd(num, den);
        num /= ans;
        den /= ans;

        printf("%lld/%lld\n", num, den);
    }


    return 0;
}
