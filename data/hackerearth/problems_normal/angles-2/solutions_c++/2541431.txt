#include <iostream>
#include<queue>
using namespace std;

int main()
{
    int hash[360],n,k;
    for(int i=0;i<360;i++)
    {
    	hash[i]=0;
    }
    
    queue<int>q;
    
    cin>>n>>k;
    int array[n],angle[k];
    
    for(int i=0;i<n;i++)
    {
    	cin>>array[i];
    	q.push(array[i]);
    }
    
    for(int i=0;i<k;i++)
    {
    	cin>>angle[i];
    }
    
    while(q.empty()==0)
    {
    	if(hash[q.front()]==0)
    	 {
    	 	int x=q.front();
    	 	hash[x]=1;
    	 	for(int i=0;i<n;i++)
    	 	{
    	 		q.push((x+array[i])%360);
    	 	}
    	 	q.pop();
    	 }
    	else q.pop(); 
    }
    
    for(int i=0;i<k;i++)
    {
    	if(hash[angle[i]]==0)
    	 cout<<"NO"<<endl;
    	else 
    	 cout<<"YES"<<endl; 
    	 
    }
    
    
    
    return 0;
}
