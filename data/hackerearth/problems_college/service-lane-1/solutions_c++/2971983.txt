#include <iostream>
using namespace std;

int main()
{
    int l, t;
    scanf("%d %d", &l, &t);
    
    int fw[l];
    for(int z = 0; z < l; z++){
    	scanf("%d", &fw[z]);
    }
    
    int en, ex;
    for(int z = 0; z < t; z++){
    	scanf("%d %d", &en, &ex);
    	int min = 3;
    	for(int i = en; i <= ex; i++){
    		if(fw[i] < min){
    			min = fw[i];
    		}
    	}
    	printf("%d\n", min);
    }
    
}
