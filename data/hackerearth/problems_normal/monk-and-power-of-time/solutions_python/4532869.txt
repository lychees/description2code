calling_order = map(int, raw_input().split(" "))
pref_order = map(int, raw_input().split(" "))
cost = 0
while pref_order:
	if(pref_order[0] == calling_order[0]):
		pref_order = pref_order[1:]
		calling_order = calling_order[1:]
		cost += 1
	else:
		calling_order = calling_order[1:] + calling_order[:1]
		cost += 1
print cost
