#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int
#define s(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)
#define sd(x) scanf("%lf", &x)
#define mod 1000000007
#define get getchar_unlocked

int a[1000005];

int main()
{
    int t, n, j, ans, p, q, i;
    s(t);
    while (t--) {
        s(n);
        map <int, int> m;
        map <int, int> :: iterator it;
        j = 0;
        ans = 0;
        for (i = 1; i <= n; ++i) {
            s(a[i]);
            ++m[a[i]];
            p = m.begin()->first;
            q = m.rbegin()->first;
            while (q-p > 1) {
                ++j;
                it = m.find(a[j]);
                --it->second;
                if (!it->second)
                    m.erase(it);
                p = m.begin()->first;
                q = m.rbegin()->first;
            }
            ans = max(ans, i-j);
        }
        printf("%d\n", ans);
    }
    return 0;
}
