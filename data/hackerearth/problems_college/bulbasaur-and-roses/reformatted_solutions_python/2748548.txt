def generateSeries():
    series = [1, 2, 2, 4, 8, 12, 96, 108, 10368, 10476, 108615168, 108625644]
    nexs = True
    for x in range(11, 21):
        if nexs:
            series.append(series[x] * series[x-1])
            nexs = False
        else:
            series.append(series[x] + series[x-1])
            nexs = True
    return series


serie = generateSeries()
noc = input()
while noc:
    noc -= 1
    print serie[input()-1]
