#include <bits/stdc++.h>
using namespace std;
int rev(string s)
{
    int i,flag=0;
    for(i=0;i<s.length()/2;i++)
    {
        if(s[i]!=s[s.length()-1-i])
        {flag=1;break;}
    }
    return flag;
}
int main()
{
    int t,flag; cin>>t; string s;
    while(t--)
    {
        cin>>s;
        flag=rev(s);
        if(flag==0)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    return 0;
}
