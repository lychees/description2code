#include <bits/stdc++.h>
using namespace std;

#define MAX 1010

int cnt[MAX], L[MAX], R[MAX], cum[MAX][MAX];

int main()
{
	int n, y, x, i, j;
	cin >> n;
	for(x = 1; x <= n; x++)
	{
		scanf("%d", &y);
		y++;
		cum[x][y]++;
		cnt[y]++;
		if(L[y] == 0)
			L[y] = x;
		R[y] = x;
	}
	for(x = 1; x < MAX; x++)
		for(y = 1; y < MAX; y++)
			cum[x][y] += cum[x][y-1];
	for(y = 1; y < MAX; y++)
		for(x = 1; x < MAX; x++)
			cum[x][y] += cum[x-1][y];
	int res = 0;
	for(i = 1; i < MAX; i++)
	{
		for(j = i+1; j < MAX; j++)
		{
			int tem = cnt[i] + cnt[j];
			if(cnt[i] == 0 || cnt[j] == 0)
			{
				res = max(res, tem);
				continue;
			}
			int l = min(L[i], L[j]);
			int r = max(R[i], R[j]);
			tem += cum[MAX-1][j] - cum[r][j] - cum[MAX-1][i] + cum[r][i] > 0;
			tem += cum[l-1][j] - cum[l-1][i] > 0;
			res = max(res, tem);
		}
	}
	cout << res;
    return 0;
}
