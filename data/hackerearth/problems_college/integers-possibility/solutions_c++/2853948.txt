#include<bits/stdc++.h>// try to get 10000000000000000000000000000000000000000000000
using namespace std;
typedef long long ll;
typedef vector< vector<ll> > matrix;
const int k=2;
const int MOD=1000000007;
matrix multiply(matrix a,matrix b){
    matrix res(k+1, vector<ll>(k+1));
	for(int i=1;i<=k;i++){
		for(int j=1;j<=k;j++){
			for(int s=1;s<=k;s++){
				res[i][j]=(res[i][j]+a[i][s]*b[s][j])%MOD;
			}
		}
	}
	return res;
}
matrix power(matrix a,ll p){
	if(p==1)
		return a;
	if(p%2==0){
		matrix x=power(a,p/2);
		return multiply(x,x);
	}
	return multiply(a,power(a,p-1));
}
int fibo(ll n){
	if(n==1)
	return 1;
	matrix t(k+1,vector<ll>(k+1));
	t[1][1]=0;
	t[1][2]=1;
	t[2][1]=1;
	t[2][2]=1;
	t=power(t,n-1);
	return (t[1][1]+t[1][2])%MOD;
}
int main(){
	int t;
	//cin>>t;
	scanf("%d",&t);
	while(t--){
	ll n;
	scanf("%lld",&n);
	cout<<fibo(n)<<endl;
	}
}
