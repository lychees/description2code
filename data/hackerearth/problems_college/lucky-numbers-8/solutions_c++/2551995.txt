#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[1000010],dp[1000010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n;
		cin>>n;
		ll int i;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		dp[n-1]=-1;
		ll int x=a[n-1];
		for(i=n-2;i>=0;i--)
		{
			dp[i]=x;
			x=max(x,a[i]);
		}
		ll int mc=0;
		for(i=0;i<n;i++)
		{
			if(a[i]>dp[i])
			mc++;
		}
		cout<<mc<<endl;
	}
	return 0;
}
