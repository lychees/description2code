#include <iostream>
#include <vector>
using namespace std;

const int MOD = 1e9+7;

int countcombinations(int n, int p){
	vector<vector<int> > ways(n,vector<int>(p));
	long long prevsum=0;
	for(int i=0;i<n;i++){
		for(int j=0;j<p;j++){
			if(i==0&&j==0) ways[i][j]=1;
			else if (i==0) ways[i][j]=0;
			else{
				if(j==0){
					for(int k=0;k<p;k++){
						ways[i][j]+=ways[i-1][k];
						ways[i][j]%=MOD;
					}
				}
				else ways[i][j]  = ways[i-1][j-1];
			}
		}
	}
	int total=0;
	for(int i=0;i<p;i++){
		total+=ways[n-1][i];
		total%=MOD;
	}
	return total;
}

int main()
{
	int tc; cin>>tc;
	while(tc--){
		int n,p; cin>>n>>p;
		cout<<countcombinations(n,p)<<endl;
	}
    return 0;
}
