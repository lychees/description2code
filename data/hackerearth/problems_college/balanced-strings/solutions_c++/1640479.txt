#include <bits/stdc++.h>
using namespace std;
int arr[26];
int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--)
   {
   	string str;
   	cin>>str;
   	int k=str.length();
   	for(int i=0;i<26;i++)
   	arr[i]=0;
   	for(int i=0;i<k;i++)
   	arr[str[i]-97]++;
   	int flag=0;
   	for(int i=0;i<25;i++)
   	{
   		if(arr[i]==0||arr[i]!=arr[i+1])
   		{
   			flag=1;
   			break;
   		}
   	}
   	if(flag)
   	cout<<"No"<<endl;
   	else
   	cout<<"Yes"<<endl;
   }
   
    return 0;
}
