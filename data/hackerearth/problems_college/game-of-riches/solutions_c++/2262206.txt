typedef long long int ll;
const int mx = 5010;
using namespace std;
int t, n, m, v[mx];
int ans[mx], x;

int main(){
    //freopen("data.in","r",stdin);
    scanf("%d", &t);
    while(t--){
        memset(ans, 0, sizeof ans);
        scanf("%d%d%d", &n, &m, &x);
        n--;
        for(int i = 0; i < n; i++){
            scanf("%d", &v[i]);
        }
        sort(v, v + n, greater <int> ());
        int r = mx;
        for(int i = 0; i < m; i++){
            for(int j = i; j < n; j += m){
                if(x < v[j])++ans[i];
            }
            r = min (r, ans[i]);
        //cout << ans[i] <<" ";
        }
        printf("%d\n", r);
    }
    return 0;
}
