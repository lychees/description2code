#define pb(x) push_back(x)

#define N		30

typedef long long ll;
typedef unsigned long long ull;

const int MOD=1000000007;
 
using namespace std;

set<char>myset;

int main(){
	ll t,i,j,c,l;
	#ifndef ONLINE_JUDGE
		freopen("inpu.txt","r",stdin);
	#endif
	string s;
	cin>>t;
	while(t--){
		cin>>s;
		l=s.length();
		ll maxx=-1;
		for(i=l-1;i>=0;i--){
			myset.clear();
			c=1;
			myset.insert(s[i]);
			for(j=i-1;j>=0;j--){
				if(myset.find(s[j])!=myset.end())	break;
				else{
					c++;
					myset.insert(s[j]);
				}
			}
			maxx=max(maxx,c);
		}
		cout<<maxx<<endl;
	}
		
	return 0;
}
