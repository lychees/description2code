#include <iostream>
using namespace std;

int main()
{
    int t, n,j, x,i;
    cin>>t;
    for(i=0;i<t;i++){
    	cin>>n;
    	int a[n], sum=0;
    	for(j=0;j<n&&cin>>x;j++){
    		a[j]=x;
    		sum+=x;
    	}
    	sum=sum/(n-1);
    	for(j=0;j<n;j++){
    		cout<<(sum-a[j])<<' ';
    	}
    	cout<<endl;
    	
    }
    return 0;
}
