#include <iostream>
#include <cstdlib>
using namespace std;
#define MOD 1000000007

int comparator(const void *a, const void *b){
	int arg1 = *static_cast<const int*>(a);
	int arg2 = *static_cast<const int*>(b);
	if(arg1<arg2)
		return -1;
	if(arg1>arg2)
		return 1;
	return 0;

}
int main(){
	int testcases;
	cin>>testcases;
	while(testcases--){
		int noOfRobots;
		cin>>noOfRobots;
		int *max = new int[noOfRobots];
		for(int i=0; i<noOfRobots; i++)
			cin>>max[i];
		qsort(max, noOfRobots, sizeof(*max), comparator);
		long long int answer = 1;
		for(int i=0; i<noOfRobots; i++){
			answer = (answer * (max[i]-i))%MOD;

		}
		cout<<answer<<"\n";
	}

	return 0;
}
