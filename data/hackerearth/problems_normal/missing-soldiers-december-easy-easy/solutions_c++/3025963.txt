#include <bits/stdc++.h>
using namespace std;
pair<int,int> p[100005];
int main() {
	// your code goes here
	int n,i,a,b,d;
	cin>>n;
	
	for(i=0;i<n;i++)
	{
		cin>>a>>b>>d;
		p[i].first=a;
		p[i].second=a+d;
	}
	sort(p,p+n);
	int ans=p[0].second-p[0].first+1;
	
	for(i=1;i<n;i++)
	{
		 if(p[i].first>p[i-1].second)
		 {
		 	ans+=p[i].second-p[i].first+1;
		 }
		 else
		 {
		 	if(p[i].second>p[i-1].second)
		 	{
		 		ans+=p[i].second-p[i-1].second;
		 	}
		 	else
		 	{
		 		p[i].second=p[i-1].second;
		 	}
		 }
	}
	cout<<ans;
	
	return 0;
}
