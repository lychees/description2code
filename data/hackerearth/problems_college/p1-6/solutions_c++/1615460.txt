{ 
	n=0;    int ch=getcx();int sign=1;   
	while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();} 
	while(  ch >= '0' && ch <= '9' )    
	n = (n<<3)+(n<<1) + ch-'0', ch=getcx();

}

LL l[1000005]={0},r[1000005]={0};
int main()
{
	LL i,j,k,n,m;
	string s;
	cin>>s;
	for(i=0;i<s.length();i++)
	{
		if(i==0)
		{
			if(s[i]=='1')l[i]=1;
			else l[i]=0;
		}
		else
		{
			if(s[i]=='1')l[i]=l[i-1]+1;
			else l[i]=l[i-1];
		}
	}
	for(i=s.length()-1;i>=0;i--)
	{
		if(i==s.length()-1)
		{
			if(s[i]=='1')r[i]=1;
			else r[i]=0;
		}
		else
		{
			if(s[i]=='1')r[i]=r[i+1]+1;
			else r[i]=r[i+1];
		}
	}
	
	LL ans=0;
	for(i=1;i<s.length()-1;i++)
	{
		if(s[i]=='0')
		{
			ans=ans+(l[i-1]*r[i+1]);
		}
	}
	printf("%lld\n",ans);
	return 0;
}
