#include <bits/stdc++.h>
using namespace std;

typedef long long ll ;

int main()
{
    ll t;
    cin >> t;
    while(t--){
    	string st;
    	cin >> st;
    	int flag = 0;
    	for(ll i = 0; i < st.size(); i++){
    		if(flag == 0){
    		for(ll j = 0; j < st.size(); j++){
    			if((st[i] == st[j])&&(i != j)){
    				flag = 1;
    				break ;
    			}
    		}
    	}
    }
    if(flag == 0)
    cout << "No" << endl ;
    else
    cout << "Yes" << endl ;
   }
    return 0;
}
