#include <bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cin>>n;
    int cal[n+1];
    int id[n+1];
    int c=0;
    queue<int>q;
    for(int i=0;i<n;i++){
    	cin>>cal[i];
    	q.push(cal[i]);
	}
	for(int i=0;i<n;i++){
		cin>>id[i];
	}
	int i=0;
	while(!q.empty()){
		if(q.front()==id[i]){
			c++;
			q.pop();
			i++;
		}
		else{
			c++;
			int x=q.front();
			q.pop();
			q.push(x);
		}
	}
	
	
	/*for(int i=0;i<n;i++){
		cout<<cal[i]<<" ";
	}
	*/
	cout<<c<<endl;
	
    return 0;
}
