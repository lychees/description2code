#include<bits/stdc++.h>
using namespace std;
#define in(x) scanf("%d",&x)
#define il(x) scanf("%lld",&x)
#define mod 1000000007
typedef long long int ll;
int main()
{
    int n,k,i,j,x,q,m,a,b,y,z;
    in(n),in(k);
    vector<vector<pair<int,int> > > v(k+1,vector<pair<int,int> >() );
    vector<vector<int> > r(n+1,vector<int> (k+1));
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=k;j++)
        {
            in(x);
            v[j].push_back(make_pair(x,i));
            r[i][j]=x;
        }
    }
    for(j=1;j<=k;j++)
        sort(v[j].begin(),v[j].end());
    /*for(i=1;i<=k;i++)
    {
        for(j=0;j<n;j++)
            cout << v[i][j].second << " ";
        cout << endl;
    }*/
    in(q);
    vector<int> ans;
    for(i=0;i<q;i++)
    {
        in(m);
        vector<int> o;
        o.resize(m);
        for(j=0;j<m;j++)
            in(o[j]);
        vector<int> e;
        e.resize(0);
        a=v[o[0]][0].first;
        b=v[o[0]][0].second;
        e.push_back(b),z=1,j=1;
        while(v[o[0]][z].first==a && z<n)
            e.push_back(v[o[0]][z].second),z++;
        while(e.size()>1 && j<m)
        {
            vector<pair<int,int> > p;
            y=e.size(),p.resize(0);
            for(x=0;x<y;x++)
                p.push_back(make_pair(r[e[x]][o[j]],e[x]));
            sort(p.begin(),p.end());
            e.resize(0);
            a=p[0].first,b=p[0].second;
            e.push_back(b),z=1;
            while(p[z].first==a && z<y)
                e.push_back(p[z].second),z++;
            j++;
        }
        ans.push_back(e[0]);
    }
    for(i=0;i<q;i++)
        cout << ans[i] << endl;
    return 0;
}
