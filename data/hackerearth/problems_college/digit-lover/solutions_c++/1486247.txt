#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<math.h>
#include<queue>
#include<string>
#include<cstring>
#include<cstdio>
#define inp(a) scanf("%ld",&a)
#define out(a) printf("%ld\n",a)
using namespace std;

int main()
{
	long t,n,s=0;
	
	inp(t);
	while(t--)
	{s=0;
		inp(n);
		while(n!=0)
		{  
			s+=(n%10);   
			n=n/10;
			if(n==0){        if(s<10)   break;
   				n=s; s=0;}
		}
		out(s);
	}
	return 0;
}
