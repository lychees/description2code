import fractions
res = []
def primes_sieve1(limit):
    limitn = limit+1
    primes = dict()
    for i in range(2, limitn):
        primes[i] = True
    for i in primes:
        factors = range(i, limitn, i)
        for f in factors[1:]:
            primes[f] = False
    for i in primes:
        if primes[i] == True:
            res.append(i)
(primes_sieve1(10**6))
t = int(input())
for i in range(t):
    num = []
    n = int(input())
    for j in range(1, n+1):
        num.append(j)
    resgcd = []
    for k in num:
        if fractions.gcd(n, k) == 1:
            resgcd.append(k)
    if len(resgcd) in res:
        print("Correct")
    else:
        print("Incorrect")
