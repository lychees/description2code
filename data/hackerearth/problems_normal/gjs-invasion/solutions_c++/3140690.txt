#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
uli f[17][1<<17];
uli g[17][1<<17];
uli pre[17];
int main(){
   int t,n;
   scanf("%d",&t);
   while(t--){
      scanf("%d",&n);
      memset(f,0ll,sizeof f);
      memset(g,0ll,sizeof g);
      g[n][(1<<n)-1]=1;
      for(int i=n-1;i>=0;i--){
         for(int b=0;b<(1<<n);b++){
            if(__builtin_popcount(b)!=i)continue;
            int x=0;
            for(int j=0;j<n;j++){
               pre[j]=x;
               if((b&(1<<j))==0)x++;
            }
            for(int j=0;j<n;j++)if((b&(1<<j))==0 && j!=i){
               g[i][b]+=g[i+1][b^(1<<j)];
               f[i][b]+=f[i+1][b^(1<<j)]+g[i+1][b^(1<<j)]*pre[j];
            }
         }
      }
      printf("%lld\n",f[0][0]);
   }
   return 0;
}
