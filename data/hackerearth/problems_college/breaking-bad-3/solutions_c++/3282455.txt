#include<bits/stdc++.h>
using namespace std;
long long a[200005];
int main()
{
    long long t;
    scanf("%lld",&t);
    while(t--)
    {
        long long i,n;
        long long s=0;
        scanf("%lld",&n);
        for(i=0;i<n;i++)
            scanf("%lld",&a[i]);
        sort(a,a+n);
        for(i=n-1;i>=0;i-=2)
            {
                s+=a[i]*a[i-1];
            }
            printf("%lld\n",s);
    }
    return 0;
}
