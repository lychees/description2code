#include<bits/stdc++.h>

using namespace std;
#define mod 1000000007
vector <long long int> mat[100002];
bool done[100002]={0};
long long fact[100002];
long long counter=1;


void dfs(long long i)
{
	done[i]=true;
	for(int j=0;j<mat[i].size();j++)
	{
		if(!done[mat[i][j]])
		{
			counter++;
			dfs(mat[i][j]);
		}
	}
}


int main()
{
		int n,k,x,y,i;
		long long answer=1;
		cin>>n>>k;
		for(int i=0;i<k;i++)
		{
			cin>>x>>y;
			mat[x].push_back(y);
			mat[y].push_back(x);
		}
		
		fact[0]=0;
		fact[1]=1;
		fact[2]=2;
		i=3;
		while(i<100002)
		{
			fact[i]=i*fact[i-1]%mod;
			i++;
		}
		
		for(int i=0;i<n;i++)
		{
			if(!done[i]&&!mat[i].empty())
				dfs(i);
			answer=(answer*fact[counter])%mod;
			counter=1;
		}
		cout<<answer;
		return 0;
}
