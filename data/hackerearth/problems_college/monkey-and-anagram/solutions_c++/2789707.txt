#define MAX 1000
#define ll long long
#define slld(t) scanf("%lld",&t)
#define sd(t) scanf("%d",&t)
#define pd(t) printf("%d\n",t)
#define plld(t) printf("%lld\n",t)
#define pcc pair<char,char>
#define pii pair<int,int>
#define pll pair<ll,ll>

#define mp(a,b) make_pair(a,b)

using namespace std;

char s1[1001],s2[1001];

int main(){
	int t;
	sd(t);
	while(t--){
		int l1,l2;
		scanf("%s",s1);
		scanf("%s",s2);
		l1=strlen(s1);l2=strlen(s2);
		
		int mark[150],mark2[150];
		
		for(int i='a';i<='z';i++) {
			mark[i]=0;mark2[i]=0;
		}
		
		for(int i=0;i<l1;i++){
			mark[s1[i]]++;
		}
		
		for(int i=0;i<l2;i++){
			mark2[s2[i]]++;
		}
		
		bool flag=0;
		int ans=0;
		for(int i='a';i<='z';i++){
			if(mark2[i]<mark[i]){
				flag=1;
				break;
			}
			else{
				if(mark2[i]!=mark[i]) ans++;
			}
		}
		
		if(flag){
			printf("-1\n");
		}
		else{
			pd(ans);
		}
	}
}
