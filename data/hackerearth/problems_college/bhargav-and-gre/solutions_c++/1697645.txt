#include <iostream>
using namespace std;

int main()
{
	int n, m, k, t1 = 0, t2 = 0;
	string s, t;
	
    cin >> s;
    m = s.length();
    
    cin >> n;
    for(int i = 0;i < n;++i) {
    	cin >> t;
    	k = t.length();
    	if(m != k) continue;
    	if(s[0] == t[0]) t1++;
    	else
    	if(t[0] == (s[0] + 3)) t2++;
    }
    
    cout << t1 << " " << t2 << endl;
    return 0;
}
