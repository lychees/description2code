#include <iostream>
using namespace std;

int fact(int n)
{	
	int fac = 1;
	for(int i=1; i<= n;i++)
	{
		fac = fac *i;
	}
	return fac;
}

int main()
{
    int i,t,n,j,v,c;
    cin >>t;
    int ans =0;
    
    while(t--)
    {
    	ans = 1;
    	cin >> c >> v;
    	
    	for(i =0; i< c;i++){
    		ans = ans * (7-i);
    	}
    	
    	ans = ans/ fact(c);
    	
    	for(i =0; i<v; i++)
    	{
    		ans = ans *(5-i);
    	}
    	
    	ans = ans/ fact(v);
    	
    	ans = ans * fact(v+c);
    	
    	cout << ans <<endl;
    }
}
