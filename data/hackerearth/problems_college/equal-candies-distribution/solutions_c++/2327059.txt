typedef long long int ll;
const int mx = 1e5 + 10;
using namespace std;



int t, n, mn, v[mx], ans;

int main(){
//    ios_base::sync_with_stdio(0);
//    cin.tie(0);
    //freopen("data.in","r",stdin);
    scanf("%d", &t);
    while(t--){
        scanf("%d", &n);
        mn = ans = 1e9;
        for(int i = 0; i < n; i++){
            scanf("%d", &v[i]);
            mn = min(mn, v[i]);
        }
        for(int i = 0; i < 5; i++){
            int r = 0;
            for(int j = 0; j < n; j++){
                int x = v[j] - (mn - i);
                r += x / 5 +(x % 5) / 2 + (x % 5) % 2;
            }
            ans = min(ans, r);
        }
        printf("%lld\n", ans);
    }
    return 0;
}
