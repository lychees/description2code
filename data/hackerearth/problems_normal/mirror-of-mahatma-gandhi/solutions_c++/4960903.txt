#include <iostream>
using namespace std;
 
int main()
{
    int t;
    cin >> t;
    while(t--){
    	string x;
    	cin >> x;
        int len = x.length(),i;
        for(i=0;i<len;i++){
        	if(x[i]!='0' && x[i]!='1' && x[i]!='8') { break; }
        	
        }
        if(i!=len){
        	cout << "NO\n";
        	continue;
        }
        int left = 0,right = len-1;
        while(right >= left){
        	if(x[left] != x[right]) break;
        	left++;
        	right--;
        }
        if(left>=right) cout << "YES\n";
        else cout << "NO\n";
    }
    return 0;
}
