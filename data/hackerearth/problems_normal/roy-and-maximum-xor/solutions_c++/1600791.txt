int gcd(int a, int b)
{
    while(b) b ^= a ^= b ^= a %= b;
    return a;
}

int main()
{
    ll t,n,c,res,i;
    string str;
    cin>>t;
    while(t--)
    {
        cin>>n;
        if(n==1)
        {
            cout<<"0 1\n";
            continue;
        }
        if(n==2)
        {
            cout<<"1 2\n";
            continue;
        }
        ll val = n;
        c = 0;
        while(val)
        {
            val >>= 1;
            c++;
        }
        res = 1;
        for(ll i = 0;i<c;++i)
        {
            res = res*2;
        }
        val = res>>1;
        val = 2*(n-val);
        res--;
        if(n && (!(n&(n-1))))
        {
            cout<<n-1<<" "<<n<<"\n";
        }
        else
            cout<<res<<" "<<val<<"\n";
    }
    return 0;
}
