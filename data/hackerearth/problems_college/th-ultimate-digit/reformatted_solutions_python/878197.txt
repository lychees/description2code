tests = input()
while tests > 0:
    a, b = map(int, raw_input().split(' '))
    pattern = [None]
    for val in range(1, b):
        inputval = int(str(a**val)[-1])
        if pattern[0] is None:
            pattern[0] = inputval
        elif pattern[0] != inputval:
            pattern.append(inputval)
        else:
            break
    print pattern[b % len(pattern) - 1]
    tests -= 1
