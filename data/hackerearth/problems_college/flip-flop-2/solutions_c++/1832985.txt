#include <iostream>
using namespace std;

int main()
{
    short t;
    char str[100001];
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	char ch=str[0];
    	int count=0;
    	for(unsigned long i=1;str[i]!=NULL;++i)
    	{
    		if(str[i]!=ch)
    		{
    			ch=str[i];
    		}
    		else
    		{
    			count++;
    			
    		}
    	}
    	cout<<count<<endl;
    	
    }
    return 0;
}
