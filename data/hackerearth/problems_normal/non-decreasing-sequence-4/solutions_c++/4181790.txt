#include<bits/stdc++.h>
using namespace std;

int main()
	{
		int T;
		cin>>T;
		while(T--)
			{
				int N;
				cin>>N;
				int ne[N];
				for(int i=0;i<N;i++)
					ne[i]=0;
				ne[0]=-1;
				int temp,top;
				top=0;
				for(int i=0;i<N;i++){
					cin>>temp;
					if(temp>ne[top])
						ne[++top]=temp;
					else
						{
							int low=1;
							int high=top;
							while(low<=high)
								{
									int mid=(low+high)/2;
									if(temp>ne[mid])
										{
											low=mid+1;
										}
									else
                    								{
                      									 high = mid -  1 ;
                    								}
								}
							ne[low]=temp;
						}
				}
				int length;
				cin>>length;
				if(length>top)
					cout<<"-1"<<endl;
				else
					cout<<ne[length]<<endl;
			}
		return 0;
	}
