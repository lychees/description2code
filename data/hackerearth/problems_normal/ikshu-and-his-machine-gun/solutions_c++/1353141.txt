#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    vector<long> v;
    long n,e;
    cin >> n;
    long r = n;
    while(r--)
    {
    	cin >> e;
    	v.push_back(e);
    }
    sort(v.begin(),v.end());
    for(int i =0;i<n;i++)
    {
    	cout << v[i]-i << " ";
    }
    return 0;
}
