#include<bits/stdc++.h>
using namespace std;
int n,k,arr[100005];
int dp[100005];
int func(int pos)
{
    if(pos>n)
        return 0;
        else if(dp[pos]!=-1)
            return dp[pos];
    else
        return dp[pos]=max(arr[pos]+func(pos+k+1),func(pos+1));
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        memset(dp,-1,sizeof(dp));
        cin>>n>>k;
        for(int i=1;i<=n;i++)
            cin>>arr[i];
        cout<<func(1)<<endl;
    }
}
