#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#define ll long long
using namespace std;


int main() {
    ll n,k,x,y=0,temp;  
    cin>>n>>k;
    ll a[1000000]={0};
    for(ll i=0;i<n;i++)
    {
    	cin>>x;
    	a[x]++;
    }
    for(ll i=1;i<=k;i++)
    {
                    if(a[i]>y)
                    {
                        y=a[i];
                        temp=i;
                        
                    }  
                    else if(a[i]==y && i<temp)
                    {
                        y=a[i];
                        temp=i;
                        
                    }                  	
    }
    cout<<temp;
    return 0;
}
