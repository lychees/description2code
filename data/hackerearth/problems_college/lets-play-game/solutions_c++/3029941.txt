#include <bits/stdc++.h>
using namespace std;

#define ll long long int
#define mp(i,j) make_pair(i,j)
#define pb(x) push_back(x)
#define MAXN 100010

ll a[MAXN];
ll b[MAXN];

int main()
{
	ll i,n,k,t,x;
	ll ans;
	
	scanf("%lld", &t);
	
	while(t--) {
		scanf("%lld%lld", &n,&k);
		
		for(i=0;i<n;i++) {
			scanf("%lld", &a[i]);
		}
		for(i=0;i<n;i++) {
			scanf("%lld", &b[i]);
		}
		ans = MAXN;
		ans = - (ans * ans * ans);
		for(i=0;i<n;i++) {
			x = k/a[i];
			ans=max(ans , x*b[i]);
		}
		printf("%lld\n", ans);
	}
	
	return 0;
}
