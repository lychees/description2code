#include<bits/stdc++.h>
#define MOD 1000000007

using namespace std;

long long int arr[100010];

void init()
{
	arr[0]=0,arr[1]=1,arr[2]=2,arr[3]=4;
	for(int i=4;i<=100000;i++)
	{
		arr[i]=(arr[i-1]%MOD+arr[i-2]%MOD+arr[i-3]%MOD)%MOD;
	}
}

int main()
{
	int n,t;
	init();
	scanf("%d",&t);
	for(int i=0;i<t;i++)
	{
		scanf("%d",&n);
	    printf("%lld\n",arr[n]);
	}
}
