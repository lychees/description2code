#include <iostream>
using namespace std;
typedef long long ll;

ll gcd(ll u, ll v) {
	return (v != 0)?gcd(v, u%v):u;
}

int main()
{
   int N;
   cin >> N;
   for(int i = 0; i < N; i++)
   {
   		ll p;
   		ll q;
   		cin >> p >> q;
   		if(gcd(p,q) == 1)
   		{
   			cout << "Is a Co-Prime" << "\n";
   		}
   		else
   		{
   			cout << "Not a Co-Prime" << "\n";
   		}
   }
}
