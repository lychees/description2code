#include <iostream>
using namespace std;

int main()
{
    long *a;
    long n,i,m,j,x;
    cin>>n;
    a = new long[n];
    for(i=0;i<n;i++)
       	cin>>a[i];
    cin>>m;
    for(i=0;i<m;i++)
    {
    	cin>>x;
    	for(j=0;j<n;j++)
    	{
    		if(a[j]>x)
    			a[j]--;
    	}
    }
    for(i=0;i<n;i++)
    	cout<<a[i]<<" ";
}
