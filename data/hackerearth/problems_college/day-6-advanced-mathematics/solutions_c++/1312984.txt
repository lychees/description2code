#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t,i,j,flag;
    char str[100];
    cin>>t;
    for(i=1;i<=t;i++)
    {
    	cin>>str;
        flag=1;
    	for(j=0;j<strlen(str)-1;j++)
    	{
    	if(str[j]==str[j+1])
    	{
    		flag=0;break;
    	}
    }
    if(flag==0)
    {cout<<"SLAP"<<endl;}
    else
    {cout<<"KISS"<<endl;}
    }
    return 0;
}
