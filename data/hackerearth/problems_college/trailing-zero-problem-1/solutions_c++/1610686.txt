#include <iostream>
#include<math.h>
using namespace std;
int64_t trailingZeroes(int64_t n) {
int64_t divisor = n;
int64_t noOfFives =0;

while(divisor) {
noOfFives = noOfFives+ (divisor/5);
divisor = divisor/5;
}
return noOfFives;
}

int main()
{
    int64_t t,n,a,tz;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	a=floor((log(2*M_PI*n)/2 + n*(log(n) - 1) + 1/(12*n))/log(10)) + 1;
    	tz=trailingZeroes(n);
    	cout<<a-tz<<'\n';
    }
    return 0;
}
