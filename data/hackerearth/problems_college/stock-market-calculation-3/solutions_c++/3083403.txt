#include<iostream>
using namespace std;
#define ll long long//ascending
ll b1[100000],b2[100000],c1[100000],c2[100000],c3[100000];
void merge(ll *a1,ll *a2,ll s1,ll s2,ll f1,ll f2)
{//cout<<s1<<" "<<s2<<" "<<f1<<" "<<f2<<" before ";
	//copying so that store during copying
	for(ll i=s1;i<=f2;i++)
	{//cout<<a1[i]<<" "<<a2[i]<<endl;
		b1[i]=a1[i];
		b2[i]=a2[i];
	}
	ll i=s1,j=s2,k=s1;//cout<<"after "<<endl;
	while(i<=f1&&j<=f2)
	{
		if(b1[i]<b1[j])
		{//cout<<"if "<<i<<" "<<j<<" "<<k<<" ";
			a1[k]=b1[i];
			a2[k]=b2[i];//cout<<a1[k]<<" "<<a2[k]<<endl;
			i++;
		}
		else if(b1[i]==b1[j])
		{
			if(b2[i]<b2[j])
			{//cout<<"elseif_if "<<i<<" "<<j<<" "<<k<<" ";
				a1[k]=b1[i];
				a2[k]=b2[i];//cout<<a1[k]<<" "<<a2[k]<<endl;
				i++;
			}
			else
			{//cout<<"else_if "<<i<<" "<<j<<" "<<k<<" ";
			    a1[k]=b1[j];
				a2[k]=b2[j];//cout<<a1[k]<<" "<<a2[k]<<endl;
				j++;	
			}
		}
		else
		{//cout<<"else_if "<<i<<" "<<j<<" "<<k<<" ";
			a1[k]=b1[j];
			a2[k]=b2[j];//cout<<a1[k]<<" "<<a2[k]<<endl;
			j++;
		}
		k++;
	}//cout<<"after_for_loop "<<endl;
	if(i<=f1)
	{//cout<<"if ";
		while(i<=f1)
		{
			a1[k]=b1[i];
			a2[k]=b2[i];//cout<<i<<" "<<j<<" "<<k<<" "<<a1[k]<<" "<<a2[k]<<endl;
			k++;
			i++;
		}
	}
	else
	{//cout<<"else ";
		while(j<=f2)
		{
			a1[k]=b1[j];
			a2[k]=b2[j];//cout<<i<<" "<<j<<" "<<k<<" "<<a1[k]<<" "<<a2[k]<<endl;
			k++;
			j++;
		}
	}
}
void mergeSort(ll *a1,ll *a2,ll s,ll f)
{
	ll mid=s+(f-s)/2;
	if(f-s>0)
	{
		mergeSort(a1,a2,s,mid);
		mergeSort(a1,a2,mid+1,f);	
		merge(a1,a2,s,mid+1,mid,f);
	}
}
int main()
{//b,c,->allready declare
	ll t,ans;
	cin>>t;
	for(ll i=0;i<t;i++)
	{
		ans=0;
		ll n;
		cin>>n;
		for(ll j=0;j<n;j++)
		{
			cin>>c1[j];
			c3[j]=c1[j];
			c2[j]=j;
		}
		mergeSort(c1,c2,0,n-1);
		//cout<<" after merge ";
//		for(ll j=0;j<n;j++)
//		{
//			cout<<c1[j]<<" "<<c2[j]<<endl;
//		}
		ll j=n-1,k=0;
		while(j>-1)
		{//cout<<"j_while "<<j<<" "<<c2[j]<<" "<<k<<" "<<ans<<" "<<endl;
			while(k<=c2[j])
			{
				if(c1[j]>c3[k])
				{//cout<<"inside_if ";
				ans+=c1[j]-c3[k];
				}//cout<<" while "<<k<<" "<<c1[j]<<" "<<c1[k]<<endl;
				k++;
			}
			j--;
		}
		cout<<ans<<endl;
	}
}
