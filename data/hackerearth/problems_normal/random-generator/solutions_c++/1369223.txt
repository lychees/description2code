#include <cstdio>

#define MAX 100050

int main()
{
    int t;
    scanf("%d", &t);
    while (t--) {
        long long a[MAX];
        int n, k;
        long long p;
        scanf("%d%d%lld", &n, &k, &p);
        for (int i = 0; i < n; i++) {
            scanf("%lld", a + i);
        }
        sort(a, a+n);
        /*for (int i = 0; i < n; i++) {
            cout << a[i] << " ";
        }
        cout << endl;*/
        bool flag = 1;
        for (int i = 0; i < n; i++) {
            int count = (upper_bound(a, a+n, a[i]+2*p) - (a + i));
            if (count >= k) {
                flag = 0;
            }
        }
        if (!flag) {
            printf("NO\n");
        } else {
            printf("YES\n");
        }
    }
}
