#include <bits/stdc++.h>
using namespace std;
long long int dp[10009],arr[10009];
int n,k;
long long int recur(int idx) {
	if(idx >= n) {
		return 0;
	}
	if(dp[idx]!= -1) {
		return dp[idx];
	}
	long long int ans = max(recur(idx+1), arr[idx]+recur(idx+k+1));
	return dp[idx] = ans;
}
int main()
{
	int t;
	cin >> t;
	while(t--) {
		memset(dp,-1,sizeof dp);
		cin >> n >> k;
		for(int i =0 ; i < n; i++) {
			cin >>arr[i];
		}
		long long int ans = recur(0);
		cout << ans << endl;
	}
    
    return 0;
}
