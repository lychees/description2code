def check(s):
    cnt = 0
    for i in range(len(s)/2):
        if s[i] == s[-i-1]:
            cnt += 1
    if cnt == len(s)/2:
        return True
    else:
        return False


t = int(raw_input())
for i in range(t):
    s1 = raw_input()
    s2 = raw_input()
    a = check(s1+s2)
    b = check(s2+s1)
    if a or b:
        print 'YES'
    else:
        print 'NO'
