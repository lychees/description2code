long double d,tim,x,y,z,vx,vy,vz,p1,p2,p3,q1,q2,q3;
int i,n;
int main()
{
	//freopen("i.in", "r", stdin);
  	//freopen("o.out", "w", stdout);
	ios_base::sync_with_stdio(false);
	cin.tie(0);
  	int t,c=1;
	cin>>t;
	while(t--)
	{
		cout<<"Case #"<<c++<<": ";
	    cin>>n;
	    p1=p2=p3=q1=q2=q3=0;
	    for(i=0;i<n;i++)
        {
            cin>>x>>y>>z>>vx>>vy>>vz;
            p1+=x; p2+=y; p3+=z;
            q1+=vx; q2+=vy; q3+=vz;
        }
        tim=-(p1*q1+p2*q2+p3*q3)/(q1*q1+q2*q2+q3*q3);

            tim=0;

        cout<<fixed<<setprecision(8)<<d<<" "<<tim<<"\n";
	}
	return 0;
}
