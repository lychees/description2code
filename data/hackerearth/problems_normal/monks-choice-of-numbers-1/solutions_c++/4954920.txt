#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,k,ans=0;
	cin>>t;
	while(t--)
	{ans=0;
		cin>>n>>k;
		int a[n],ct=0;

		for(int i=0;i<n;i++)
		{
			cin>>a[i];
			ct=0;
			while(a[i])
			{
				a[i]=a[i]&(a[i]-1);
				ct++;
				//cout<<"err1111"<<endl;
			}

			a[i]=ct;
			//cout<<"err"<<endl;
		}

		sort(a,a+n);

		for(int i=n-1;k>0&&i>=0;i--,k--)
			ans+=a[i];

		cout<<ans<<endl;
	}
	return 0;
}
