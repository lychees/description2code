import sys

for __ in range(input()) :
	n , lists = input(),map(int,sys.stdin.readline().split())
	pre_sum ,tot= [0]*(n+1) , sum(lists)
	for i in range(n) :
		pre_sum[i] = pre_sum[i-1] + lists[i]
	found = False
	for i in range(1,n) :
		if pre_sum[i-1] == (tot-pre_sum[i]) :
			found = True
			break
	if found :
		print "YES"
	else :
		print "NO"
