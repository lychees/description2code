#include<bits/stdc++.h>
using namespace std;
#define ll int
ll dp[1005],n,a[1005],b[1005];
ll solve(ll pos)
{
    if(dp[pos]!=-1)
    return dp[pos];
    if(pos==n-1)
    {
        return 0;
    }
    ll ans,i,temp=0;
    for(i=pos+1;i<n;i++)
    {
        if(a[i]%a[pos]==0)
        {
            ans=solve(i)+1;
            temp=max(temp,ans);
        }
    }
    return dp[pos]=temp;
}
int main()
{
    ll i,c=0,k=0,m;
    ios::sync_with_stdio(0);
    cin.tie(0);
    cin>>n;
    m=n;
    memset(dp,-1,sizeof(dp));
    for(i=0;i<n;i++)
    {
        cin>>a[i];
    }
    sort(a,a+n);
    for(i=0;i<n;i++)
    k=max(k,solve(i)+1);
    if(k==1)
    k=-1;
    cout<<k<<"\n";
    return 0;
}
