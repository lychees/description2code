#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long n;
    cin>>n;
    long long a[n];
    for(long long i=0;i<n;i++)
        cin>>a[i];
    long long result[n];
    for(long long i=0;i<n;i++)
        result[i]=0;
    result[0]=a[0];
    for(long long i=1;i<n;i++)
    {
       /// if(a[i]!=1)
            result[i]=(a[i]-1)*(i) + a[i];
    }
    long long ans=0;
    for(long long i=0;i<n;i++)
        ans+=result[i];
    cout<<ans;
    return 0;
}
