#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,m,i;
    cin>>n>>m;
    string s1,s2;
    map<string,string> mp;
    for(i=0;i<m;i++)
    {
    	cin>>s1>>s2;
    	mp[s1]=s2;
    }
    for(i=0;i<n;i++)
    {
    	cin>>s1;
    	if((s1.length())<=(mp[s1].length()))
    		cout<<s1<<" ";
    	else
    		cout<<mp[s1]<<" ";
    }
    return 0;
}
