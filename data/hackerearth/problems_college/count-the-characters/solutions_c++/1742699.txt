#include<bits/stdc++.h>
using namespace std;
int main()
{
    int hash[101][26]={0}; int total=0,flag=0,i,j; 
    string name ; char ch ;
    int N ;// total no of names ;
    scanf("%d",&N);
    for(int i=0;i<N;i++){
        // create hash array for all names ;
        cin>>name ;
        for(int j=0;j<name.size();j++){
            ch=name[j];
            hash[i][ch-'a']=1;  
        }
    }
    for(int i=0;i<26;i++){
            flag=0;
            for(j=0;j<N;j++)
                if(hash[j][i]!=1){
                    flag++; break;
                }
            if(flag==0){
	                total++;
	            }    
    }
    printf("%d",total);
    return(0);
}
