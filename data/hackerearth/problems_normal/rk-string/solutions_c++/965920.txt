#include <iostream>
#include <string>

using namespace std;

long long k1, k2, cnt1[1000001], cnt2[1000001], sum1, sum2, res1, res2, t;
string s;

int main()
{
    cin>>t;
    while (t--){
    	for (int i=0; i<=1000000; i++){
    		cnt1[i]=0;cnt2[i]=0;
    	}
    	sum1=0;sum2=0;res1=0;res2=0;
    	cin>>k1>>k2;
    	cin>>s;
    	cnt1[0]=1;
    	cnt2[0]=1;
    	for (long long i=0; i<s.size(); i++){
    		if (s[i]=='R')
    			sum1++;
			if (sum1>=k1)
				res1+=cnt1[sum1-k1];
			cnt1[sum1]++;
			
			if (s[i]=='K')
				sum2++;
			if (sum2>=k2)
				res2+=cnt2[sum2-k2];
			cnt2[sum2]++;
    	}
    	cout<<res1<<" "<<res2<<"\n";
    }
    return 0;
}
