#include <iostream>
using namespace std;
#define mod 1000000007
int arr[2000002]={0};
//int hold[2000002]={0};
void genratep()
{
	arr[0]=1;
	arr[1]=1;
	int i=2;
	while(i<1000001)
	{
		
		if(arr[i]==0)
		{
			int j=i+i;
			while(j<2000002)
			{
				arr[j]=1;
				j+=i;
			}
		}
		++i;
	}
}
int main()
{

	genratep();
	int n;
	cin>>n;
	int mid = n-1;
	long count = 0;
	if(arr[mid]==0)
	{
		count = count + n;
	}
	//cout<<count<<endl;
	if(count>=mod)
			   count%=mod;
	int x=1;
	for(int i=0;i<n-1;++i)
	{
		if(arr[i]==0)
		{
			count+=x;
			if(count>=mod)//count = 0;
			   count%=mod;
			
		}
		++x;	
	}
	//cout<<count<<endl;
	x=n-1;
	for(int i=n;i<2*n-1;++i)
	{
		if(arr[i]==0)
		{
			count+=x;
				if(count>=mod)
			   	count%=mod;
		}
		--x;
	}
	cout<<count<<endl;
    return 0;
}
