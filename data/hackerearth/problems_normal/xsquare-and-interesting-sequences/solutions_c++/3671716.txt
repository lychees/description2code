#include <bits/stdc++.h>
using namespace std;
typedef unsigned long long int ulli;
typedef long long int lli;
#define pb push_back
#define ft first
#define se second
#define mp make_pair

lli mod = 1000000009;

lli power(lli a, lli b){
	if(b == 0) return 1;
	if(a == 0) return 0;
	lli res = 1;
	res = power(a, b / 2);
	if(b % 2){
		res = (res * res) % mod;
		return (res * a) % mod;
	}
	else return (res * res) % mod;
}

int main(){
	int t;
	cin >> t;
	while(t--){
		lli n;
		cin >> n;
		lli res = 1;
		if(n % 2)
			res = (res * 26) % mod;
		res = (res * power(26, n / 2)) % mod;
		cout << res << endl;
	}	
}
