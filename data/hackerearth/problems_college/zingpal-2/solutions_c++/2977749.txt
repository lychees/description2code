#include <bits/stdc++.h>
using namespace std;
int a[1000][1000];
int main()
{
    int n,m,i,j,k,x,y;
    cin>>n>>m;
    for(i=0;i<m;i++){
    	cin>>x>>y;
    	a[x][y]=1;a[y][x]=1;
    }
    int c=0;
    for(k=1;k<=n;k++){
    	for(i=1;i<=n;i++){
    		for(j=1;j<=n;j++){
    			if(a[i][k]&&a[k][j]&&a[i][j])
    				c++;
    		}
    	}
    }
    if(c==6)
    	cout<<"FHTAGN!";
    else
    	cout<<"NO";
    cout<<endl;
    return 0;
}
