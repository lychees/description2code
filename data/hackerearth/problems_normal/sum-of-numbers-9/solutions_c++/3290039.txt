#include <iostream>
using namespace std;

bool function1(long long int *A, int i, int n, long long int c_sum,long long sum);

int main()
{
	int t=0,i=0,n=0;
	long long int A[15],sum=0;
	cin>>t;
	
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)cin>>A[i];
		cin>>sum;
		if(function1(A,-1,n,0,sum))
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
		
		
	}
    
    return 0;
}

bool function1(long long int *A, int i, int n, long long int c_sum,long long sum)
{
	if(c_sum==sum)return true;
	
	if(i+1==n)return false;
	
	bool result=false;
	
	for(int j=i+1;j<n;j++)
	{
		result=function1(A,j,n,c_sum+A[j],sum);
		if(result)break;
	}
	
	return result;
}
