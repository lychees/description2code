#include <bits/stdc++.h>
using namespace std;
#define pb push_back
#define ll long long int
#define INF 10000000000000000
#define mod 1000000007
ll gcd (ll a,ll b)
{
	if(b==0)
		return a;
	return gcd(b,a%b);
}
ll modular(ll a,ll b)
{
	ll ans=1;ll c=mod;
	while(b!=0)
	{
		if(b%2==1)
		{
			ans*=a;
			ans%=c;
		}
		a=a*a;
		a%=c;
		b/=2;
	}
	return ans;
}

int main()
{
	int n;
	cin>>n;
	int a[n];
	ll f=1;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		f=((f%mod)*(a[i]%mod))%mod;
	}
	ll gc=a[0];
	for(int i=1;i<n;i++)
	{
		gc=gcd(gc,a[i]);
		gc%=mod;
	}
	ll ans=modular(f,gc);
	
	cout<<ans%mod<<endl;
}
