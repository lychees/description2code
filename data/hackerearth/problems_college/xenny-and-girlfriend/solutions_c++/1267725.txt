#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;

vector<int> v[10005];

int deg[10005],visited[10005];

void dfs(int cur)
{
	int i;
	visited[cur]=1;
	for(i=0;i<v[cur].size();i++)
	{
		if(visited[v[cur][i]]==-1)
			dfs(v[cur][i]);
	}
}

int main()
{
	int t,n,m,x,y,i,f,g,c;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&m);
		for(i=1;i<=n;i++)
		{
			v[i].clear();
			visited[i]=-1;
			deg[i]=0;
		}
		while(m--)
		{
			scanf("%d %d",&x,&y);
			v[x].push_back(y);
			v[y].push_back(x);
			
			deg[x]++;
			deg[y]++;
		}
		
		i=1;
		while(deg[i]==0)
			i++;
		
		dfs(i);
		
/*		for(i=1;i<=n;i++)
			printf("%d ",deg[i]);
		printf("\n");

		for(i=1;i<=n;i++)
			printf("%d ",visited[i]);
		printf("\n");
*/				
		f=0;
		c=0;
		for(i=1;i<=n;i++)
		{
			if(deg[i]%2==1)
				c++;
		}
		if(c==0 || c==2)
			f=1;
		g=0;
		for(i=1;i<=n;i++)
		{
			if(deg[i]!=0 && visited[i]==-1)
			{
				g=1;
				break;
			}
		}
		
		if(g==0 && f==1)
		{
			printf("Yes\n");
		}
		else
		{
			printf("No\n");
		}
	}
	
}
