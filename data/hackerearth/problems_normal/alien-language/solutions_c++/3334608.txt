#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,t,jadu[26],golu[26];
	cin>>t;
	while(t--)
	{
		memset(jadu,0,sizeof(jadu));
		memset(golu,0,sizeof(golu));
		string jd,gl;
		cin>>gl>>jd;
		for(i=0;i<gl.length();i++)
		golu[gl[i]-'a']=1;
		for(i=0;i<jd.length();i++)
		jadu[jd[i]-'a']=1;
		int ans=0;
		for(i=0;i<26;i++)
		if(jadu[i]&&golu[i])
		{
			ans=1;
			break;
		}
		if(ans)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
	}
	return(0);
}
