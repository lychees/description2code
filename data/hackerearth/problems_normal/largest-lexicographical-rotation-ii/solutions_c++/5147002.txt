#include<bits/stdc++.h>
using namespace std;
void find_lexico(string &s)
{
    string temp=s+s;
    int i,n,j,l;
    n=s.length();
    i=l=0;
    j=1;
     while(i+l<n)
     {
         if(temp[i+l]==temp[j+l]) ++l;
         else
         {
             if(temp[i+l]<temp[j+l])
             {
                 i=max(i+l,j);
                 l=0;
                 j=i+1;
             }
             else
             {
                 j=j+l+1;
                 l=0;
             }
         }
     }
     cout<<temp.substr(i,n)<<endl;
}
int main()
{
    string s;
    cin>>s;
    find_lexico(s);
    return 0;
}
