#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   long long a,b,count;
   while(t--){
   	count = 0;
   	cin>>a>>b;
   	while((a!=1 && b!=1)){
   		if(a>b) {count += (a/b); a%=b;}
   		else {count += (b/a);  b%=a;}
   	}
   	if(a>1) count += (a-1);
   	else count += (b-1);
   	cout<<count<<endl;
   }
}
