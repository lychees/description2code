from sys import stdin

inp = stdin.read().strip().split('\n')

seq = [[[] for y in range(3)] for x in range(50)]

for x in range(1, 50):
    for y in range(3):
        for i in range(x):
            seq[x][y].append(y)
        for i in range(50):
            seq[x][y].append(sum(seq[x][y][i:i + x]))

t = int(inp[0])
for i in range(t):
    x, y, n = map(int, inp[1 + i].split(' '))
    print seq[x][y][n - 1]
