#include<cstdio>
#include<iostream>
using namespace std;


long long a[40];
int main(){
    a[0] =1 ;
    a[2] = 6;
    int x,n,t;
    for(x=3;x<=20;x++){
        n = x - 3;
        a[n+3] = ((108*n+72+36*n*n)*a[n]+(24*n*n+96*n+96)*a[n+1]+(n*n+5*n+6)*a[n+2])/(6*n+9+n*n);
    }
    cin >> t;
    while(t--){
        cin >> x;
        cout<<a[x]<<endl;
    }
    return 0;
}
