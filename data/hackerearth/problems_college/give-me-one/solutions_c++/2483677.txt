#include<bits/stdc++.h>
using namespace std;
#define ll long long
vector<ll> dp(20000002);
int main()
{

    ll test,i,j,total,num,m,n;
    dp[0]=0,dp[1]=0;
    for(i=2;i<=20000002;i++)
    {
        if(i%6==0)
            dp[i]=1+min(dp[i/3],min(dp[i/2],dp[i-1]));
        else if(i%3==0)
            dp[i]=1+min(dp[i/3],dp[i-1]);
        else if(i%2==0)
            dp[i]=1+min(dp[i/2],dp[i-1]);
        else
            dp[i]=1+dp[i-1];
    }
    cin>>test;
    while(test--)
    {
        cin>>n;
        cout<<dp[n]<<endl;
    }
}
