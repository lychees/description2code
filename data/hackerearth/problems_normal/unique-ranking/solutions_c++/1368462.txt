const int MAX_N = 1000 + 1;
const int MAX_K = 20 + 1;

int N, K;
int A[MAX_N][MAX_K];
int used[MAX_N];

int main(){
  int Q, t, k;
  
  scanf("%d%d", &N, &K);
  for(int i = 0; i < N; i++)
    for(int j = 0; j < K; j++)
      scanf("%d", &A[i][j]);
  scanf("%d", &Q);
  while(Q--){
    scanf("%d", &t);
    memset(used, 0, sizeof used);
    while(t--){
      scanf("%d", &k);
      k --;
      int p = 111;
      for(int i = 0; i < N; i++)
	if(!used[i])
	  p = min(p, A[i][k]);
      for(int i = 0; i < N; i++)
	if(A[i][k] > p)
	  used[i] |= 1;
    }
    for(int i = 0; i < N; i++){
      if(!used[i]){
	printf("%d\n", i + 1);
	break;
      }
    }
  }
  return 0;
}
