#include<iostream>
#include<set>
using namespace std;
typedef long long ll;
int main()
{
    ll t,n,x,i;
    cin>>t;
    while(t--)
    {
        cin>>n;
        set<ll> s;
        for(i=0;i<n;i++)
        {
            cin>>x;
            s.insert(x);
        }
        cout<<s.size()<<'\n';
    }
    return 0;
}
