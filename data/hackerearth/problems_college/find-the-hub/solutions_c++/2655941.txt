#include <bits/stdc++.h>
using namespace std;
bool isprime(long long n)
{
	int flag=0;
	long long i;
	for(i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
		{
			flag=1;
			break;
		}
	}
	if(flag)
	return false;
	else
	return true;
}
int main()
{
    long long n,a[100001],x,r=0,i,j;
    cin>>n;
    for(i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    for(i=n-1;i>=0;i--)
    {
    	if(isprime(a[i]))
    	{
    		x=a[i];
    		j=i;	
    		while(a[j]==x)
    		{
    			r++;
    			j--;
    		}
    		break;
    	}
    }
    cout<<x<<'\n'<<r<<"\n";
    return 0;
}
