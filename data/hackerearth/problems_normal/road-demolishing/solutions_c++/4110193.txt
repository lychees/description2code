#include<bits/stdc++.h>
using namespace std;
typedef long long lld;


int main()
{
	lld t;
	cin>>t;
	
	while(t--)
	{  lld n,q;
	   cin>>n>>q;
	   
	   lld edges=n*(n-1)/2;
	   q=q-1;
	   lld ans=n*n-(n%q)*((n+q-1)/q)*((n+q-1)/q);
	   ans=ans-(q-n%q)*(n/q)*(n/q);
	  	
	   cout<<edges-(ans)/2<<"\n"; 
		
	
		
		
		
	}
	
	
	
}
