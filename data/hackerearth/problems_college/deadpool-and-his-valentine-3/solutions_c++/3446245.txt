#include <iostream>
#include<math.h>
using namespace std;

int main() {
	long long int m,k;
	cin>>k;
	cin>>m;
	bool f = false;
	long long int a[100] = {0};
	while(k--)
	{
		int temp1,temp2;
		cin>>temp1;
		for(int i = 0;i<temp1;i++)
		{
			cin>>temp2;
			a[temp2]++;
		}
	}
	for(int i = 1;i<=m;i++)
	{
		if(a[i] == 0)
			f = true;
	}
	if(f)
		cout<<"NO"<<endl;
	else 
		cout<<"YES"<<endl;
	return 0;
}
