#include <stdio.h>
#include<iostream>
#include<cstring>
#define gc getchar_unlocked
using namespace std;
typedef long long ll;
void scanint(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
void scanlong(long long &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
int cool(int n){
	char first;
    	char second;
		if(n&1){
    		first='1';
    	}
    	else{
    		first='0';
    	}
    	n=n>>1;
    	if(n&1){
    		second='1';
    	}
    	else{
    		second='0';
    	}
    	n=n>>1;
    	int count=0;
    	while(n!=0){
    		char third;
    		if(n&1)
    		third='1';
    		else
    		third='0';
    		n=n>>1;
    		if((first=='1')&(second=='0')&(third=='1')){
    			count++;
    		}
    		first=second;
    		second=third;
    	}
    	return count;
}
int main(void) {
    int t;
    scanint(t);
    while(t--)
    {
    	int n,k;
    	scanint(n);
    	scanint(k);
    	int count=0;
    	for(int i=1;i<=n;i++){
    		if(cool(i)>=k){
    			count++;
    		}
    	}
    
    	cout<<count<<endl;
    }
	return 0;
}
