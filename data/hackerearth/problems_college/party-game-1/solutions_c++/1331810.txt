#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int main(){
	int t;
	cin >> t;
	while(t--)
	{
		int x,y,n;
		cin >> x >> y >> n;
		int i;
		int a[n+1];
		for(i = 0 ; i < n ;++i)
			cin >> a[i];
		sort(a,a+n);
		int c1=0;
		int c2=0;
		for(i = 0 ; i  < n; ++i)
		{
		    if(a[i] <= x)
			{
				x-=a[i];
				c1++;
			}	
			if(a[i]<=y)
			{
				y-=a[i];
				c2++;
			}
		}
		if(c1<c2)
		cout<<"Jhool Won"<<endl;
		else
		if(c1==c2)
		cout<<"Tie"<<endl;
		else
		cout<<"Chef Won"<<endl;
	}
	return 0;
}
