#include <bits/stdc++.h>
using namespace std;

bool isprime[10000006] = {};
int a[500] = {};
int mx = 0;
void seive()
{
	vector<int>v;
	long long i,j,MAX;
	MAX=10000006;
	isprime[0]=1;
	isprime[1]=1;
	for(i=2;i<MAX;i++)
	{
		if(isprime[i]==0)
		{
			if(v.size() > 0)
			{
				a[(i - v.back())]++;
			}
			if(i > 2)
				v.push_back(i);
			j=i*i;
			for(;j<MAX;j+=i)
				isprime[j]=1;
		}
	}
}
bool pos[10000006] = {};
int main()
{
	seive();
	int tc;
	cin >> tc;
	
	for(int i=0;i<200;i++)
	{
		for(int j=i;j<200;j++)
		{
			if(a[i] > 0 && a[j] > 0) pos[i+j] = 1;
		}
	}
	while(tc--)
	{
		int n;
		cin >> n;
		if(!pos[n]) cout << "NO\n";
		else cout << "YES\n";
	}
    return 0;
}
