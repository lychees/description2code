#include<iostream>
#include<string>
using namespace std;
int main()
{int a[200],b,max=0;
 char c;
 string s;
 for(int i=0;i<200;i++)
  a[i]=0;
 getline(cin,s);
 for(int i=0;i<s.length();i++)
  {b=s.at(i);
   a[b-32]++;
  }
 for(int i=0;i<200;i++)
  {if(a[i]>max)
    {max=a[i];
     b=i;
    }
  }
  b+=32;
  c=b;
  cout<<c<<" "<<max<<endl;
  return 0;
}
