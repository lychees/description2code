#include<iostream>
#include<algorithm>
#include<string.h>
using namespace std;

struct store
{
	int array[100000]={0};
};

int main()
{
	struct store *result=new store[100];	
	int i,j,max,count=0; 
	int freq[1000000]={0};
	int t,n;
	cin>>t;
	for(i=0;i<t;i++)
	{
		cin>>n;
		int *a=new int[n];
		for(j=0;j<n;j++)
		{
			cin>>a[j];
		
		}
		for(j=0;j<n;j++)
		{
			freq[a[j]]+=1;	
		}	
		max=*max_element(freq,freq+1000000);
		
		for(j=0;j<1000000;j++)
		{
			if(freq[j]==max)
			{
				result[i].array[j]=j;
				
			}
		}
		sort(result[i].array,result[i].array+100000);
		memset(freq,0,sizeof(freq));
		
	}	
	for(j=0;j<t;j++)
	{
		for(i=100000-1;i>=0;i--)
		{
	
			if(result[j].array[i]>0)
			{
				cout<<result[j].array[i]<<" ";
			}
			else
				break;
		}
			cout<<endl;
	}
}
