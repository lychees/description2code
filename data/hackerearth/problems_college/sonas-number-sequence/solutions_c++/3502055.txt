typedef long long ll;

const int maxn = 100010;
ll arr[maxn];

int main() {
	int n; cin >> n;
	for (int i = 0; i < n; i ++) cin >> arr[i];
	ll sum = arr[0];
	cout << arr[0] << ' ';
	for (int i = 1; i < n; i ++) {
		ll term = arr[i] * (i + 1) - sum;
		cout << term << ' ';
		sum += term;
	}
	
	cout << "\n";
	
	return 0;
}
