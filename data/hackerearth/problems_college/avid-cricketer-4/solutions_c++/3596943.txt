#include <iostream>
using namespace std;

int main()
{
	int t, r, b;
	double *cur, *prev, x[1010], y[1010], ans;
	cin >> t;
	while (t--) {
		cin >> r >> b;
		for (int i = -1; ++i < 1010; y[i] = 0);
		x[0] = 0;
		y[0] = 1;
		cur = x;
		prev = y;
		ans = prev[r];
		for (int i = 0; i < b; i++) {
			for (int i = -1; ++i < 1010; cur[i] = 0);
			for (int j = 0; j < r; j++) {
				cur[j] += prev[j] * 2 / 7;
				cur[j + 1] += prev[j] / 7;
				cur[j + 2] += prev[j] / 7;
				cur[j + 3] += prev[j] / 7;
				cur[j + 4] += prev[j] / 7;
				cur[j + 6] += prev[j] / 7;
			}
			ans += cur[r] + cur[r + 1] + cur[r + 2] + cur[r + 3] + cur[r + 4] + cur[r + 5];
			double *t = cur;
			cur = prev;
			prev = t;
		}
		printf("%.6f\n", ans);
	}
    return 0;
}
