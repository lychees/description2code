#include <iostream>
#include <stdio.h>
#include <vector>
#include <string>

using namespace std;

typedef long long int Lint;

int main()
{
	Lint t, n, v, count, i;
	
	cin >> t;
	
	while( t-- ) {
		
		string s;
		
		cin >> s;
		n = s.length();
		
		vector<bool> A( 128, true );
		count = 0;
		
		for( i = 0;i < n;i++ ) {
			
			if( A[ s[ i ] ] ) {
				count++;
				A[ s[ i ] ] = false;
			}
		}
		
		v = ( ( count + 1 ) * count ) / 2;
		
		printf("%lld\n", v );
		
	}
	
	return 0;
}
