#include <bits/stdc++.h>
using namespace std;
int main(){
	int t;
	long long x,y,z,i,ans=0;
	cin>>t;
	while(t--){
		cin>>z>>x>>y;
		ans=0;
		for(i=1;i*i<z;i++){
			if(z%i==0){
				if(i<=x&&z<=y*i)
					ans++;
				if(z<=x*i&&i<=y)
					ans++;
			}
		}
		if(i*i==z){
			if(i<=x&&i<=y)
				ans++;
		}
		cout<<ans<<"\n";
		
	}
	return 0;
}
