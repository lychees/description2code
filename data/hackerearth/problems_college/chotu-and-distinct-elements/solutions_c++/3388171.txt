#include <iostream>
#include <set>
using namespace std;

int main()
{
 	int t, n, a;
 	cin>>t;
 	while(t--)
 	{
 		set <int> st;
 		set <int> :: iterator it1, it2;
 		cin>>n;
		for (int i=0;i<n;i++){
			cin>>a;
			st.insert(a);
		}
		int flag=0;
		
		for(it1=st.begin(), it2=--st.end(); *it1<=*it2 && flag!=1; it1++, it2--)
		{
			if (*it1==*it2){
				cout<<*it1<<' ';
				flag=1;
			}
			else{
				cout<<*it2<<' ';
				cout<<*it1<<' ' ;
			}
		}
		cout<<endl;
 	}
 	return 0;
}
