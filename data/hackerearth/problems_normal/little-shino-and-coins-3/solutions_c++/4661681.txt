#include<bits/stdc++.h>
using namespace std;

int main(){
	int k;
	string s;
	cin>>k;
	cin>>s;
	int ans=0;
	int n=s.length();
	for(int i=0;i<n;i++)
	{
		int freq[30]={0};
		int diff=0;
		for(int j=i;j<n;j++)
		{
			freq[s[j]-'a']++;
			if(freq[s[j]-'a']==1)
			{
				diff++;
			}
			if(diff==k){
				ans++;
			}
		}
	}
	cout<<ans<<"\n";
	return 0;
}
