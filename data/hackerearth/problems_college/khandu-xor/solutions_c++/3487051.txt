#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	char temp;
	cin>>t;
	while(t--){
		string s1, s2, t1, t2;
		cin>>s1>>s2;
		t1 = s1 + s2;
		t2 = s2 + s1;
		s1 = t1;
		s2 = t2;
		for(unsigned int i=0, j=s1.length()-1; i<s1.length()/2; i++, j--)
		{
			temp = s1[i];
			s1[i] = s1[j];
			s1[j] = temp;
		}
		for(unsigned int i=0, j=s2.length()-1; i<s2.length()/2; i++, j--)
		{
			temp = s2[i];
			s2[i] = s2[j];
			s2[j] = temp;
		}
		if(s1 == t1 || s2 == t2) cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
	return 0;
}
