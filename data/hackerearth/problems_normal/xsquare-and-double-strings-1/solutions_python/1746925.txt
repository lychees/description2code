t= int(raw_input())

for i in range(t):
	flag=0
	st= raw_input()
	length= len(st)
	for j in range(length):
		for k in range(j+1,length):
			if st[j]==st[k]:
				print "Yes"
				flag=1
				break
		if flag==1:
			break
		
	else:
		print "No"
