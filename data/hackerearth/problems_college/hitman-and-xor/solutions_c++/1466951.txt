#include <iostream>
using namespace std;

int main()
{
   /// cout << "Hello World!" << endl;
   int t;
   long n;
   cin>>t;
   while(t--)
   {
   	long long k;
   	cin>>n;
   	long e=0,o=0;
   	for(long i=1;i<=n;i++)
   	{
   		cin>>k;
   		if(k%2)
   		o+=1;
   		else
   		e+=1;
   	}
   	cout<<o*e<<endl;
   }
    return 0;
}
