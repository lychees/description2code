def checkEqual(iterator):
    return len(set(iterator)) ==1
def test(string):
    count = 0
    for i in xrange(len(string)):
        for j in xrange(i,len(string)):
            #print string[i:j+1]
            if checkEqual(string[i:j+1]):
                count += 1
    return count
