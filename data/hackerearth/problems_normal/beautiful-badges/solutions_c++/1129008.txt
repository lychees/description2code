#include <iostream>
#include <fstream>
#include <vector>
#include <algorithm>

using namespace std;

#define MAX_D 1000

int main()
{
    //freopen("badges.in", "r", stdin);
    int n;
    cin >> n;
    int b[MAX_D+1]={0}, y[MAX_D+1]={0};

    int ci,di;
    for(int i=0; i<n; i++) {
        cin >> ci >> di;
        if(ci<2) b[di]++;
        else y[di]++;
    }

//    for(int db=1; db<=1000; db++) if(b[db]) cout << db << "x" << b[db] << " " ;
//    cout << endl;
//    for(int dy=1; dy<=1000; dy++) if(y[dy]) cout << dy << "x" << y[dy] << " " ;
//    cout << endl;

    int dbsqr, dysqr;
    int cnt = 0;

    for(int db=1; db<=1000; db++) if(b[db]) {
        dbsqr = db*db;

        for(int dy=1; dy<=1000; dy++) if(y[dy]) {
            dysqr = dy*dy;

            if((2*dbsqr>dysqr) && (2*dysqr>dbsqr)) {
                int num = min(b[db],y[dy]);
                cnt += num;
                b[db]-=num;
                y[dy]-=num;
//                cout << "blue-" << db << ":" << "yellow-" << dy << " " << b[db] << "x" << y[dy] << endl;
                if(b[db]<=0) break;
            }
        }


    }

    cout << cnt << endl;
    return 0;
}
