#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int dp[1002][1002];
int a[1002],X,n;
int go(int x, int y)
{
	if(x < 0 || y == n) {
		return INT_MIN;
	}
	if(x == 0) {
		return 0;
	}
	if(dp[x][y] != -1) {
		return dp[x][y];
	}
    dp[x][y] = max(go(x,y+1),go(x-y-1,y)+a[y]);
    return dp[x][y];
}
int main()
{
	int t,i;
	cin>>t;
	while(t--) {
		cin>>n>>X;
		memset(dp,-1,sizeof(dp));
		for(i = 0; i < n; i++) {
			cin>>a[i];
			a[i] -= X;
		}
		cout<<go(n,0)<<endl;
	}
	return 0;
}
