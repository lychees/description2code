
for _ in range(input()):
    string = raw_input().strip()
    i = 0
    j = len(string)-1
    ans = 0
    while (i < j):
        if (string[i] == 'R' and string[j] == 'K'):
            ans += 2
            i += 1
            j -= 1
        elif (string[i] == 'K'):
            i += 1
        elif (string[j] == 'R'):
            j -= 1
    print ans
