#include<bits/stdc++.h>
using namespace std;

int arr[100005];

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n, sum=0;
		cin>>n;
		for(int i=0; i<n; i++)
        {
	        scanf("%d", &arr[i]);
	        sum += arr[i];
        }
        int suml, sumr;
        suml = sumr = 0;
        bool flag = false;
        
        if(n == 1)
        {
        	cout<<"YES\n";
        	continue;
        }
        
        for(int i=1; i<n; i++)
        {
        	suml += arr[i-1];
        	sumr = sum - suml - arr[i];
        	if(suml == sumr)
        	{
        		flag = true;
        		break;
        	}
        }
        
        if(flag == true)
            cout<<"YES\n";
        else
            cout<<"NO\n";
	}
}
