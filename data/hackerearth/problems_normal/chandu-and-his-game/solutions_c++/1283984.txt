#include<iostream>

using namespace std;


int main(){

    long N;
    cin>>N;
    long k;
    cin>>k;
	int x[k];
	int y[k];
    for(int i = 0; i < k; i++){
	int temp;
	cin>>temp;
	x[i] = temp;
    }
    for(int i = 0; i < k; i++){
	int temp;
	cin>>temp;
	y[i] = temp;
    }



    long xstep = 0;
    long ystep = 0;

    for(int i = 0; i < k; i++){

	if(x[i] <= (N/2))
		xstep = xstep + x[i] - 1;
	else
		xstep = xstep + (N - x[i]);

	if(y[i] <= (N/2))
		ystep = ystep + y[i] - 1;
	else
		ystep = ystep + (N - y[i]);
    }

    cout<< xstep+ystep << endl;

}
