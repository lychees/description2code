#include<bits/stdc++.h>
using namespace std;
int main()
{
	map<int,int> m;
	int t,n,i,x,y;
	cin>>t;
	while(t--)
	{
		m.clear();
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>x>>y;
			m[x]++;
			m[y]--;
		}
		map<int,int>::iterator a;
		int res = 0,ans=-1;
		for(a=m.begin();a!=m.end();a++)
		{
			res += a->second;
			ans = max(ans,res);
		}
		cout<<ans<<endl;
	}
}
