
#include <iostream>
#define MOD 1000000007

typedef unsigned long long ull;

using namespace std;

ull add(int n) {
 ull addition[n+1];

  addition[0] = 1;
  addition[1] = 0;
 
  for (int i = 2; i <= n; i++)
  {
      addition[i] = (((addition[i-1] % MOD + addition[i-2] % MOD) % MOD) * ((i-1) % MOD)) % MOD;
  }
 
  return addition[n];

}

ull d(int len) {
 if (len == 0)
    return 0;
 else if (len == 1)
    return 1;
 else
 return add(len);
}
