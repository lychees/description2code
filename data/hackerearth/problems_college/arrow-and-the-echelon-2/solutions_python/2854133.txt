t=input()
for i in range(t):
	n=input()
	assert n>0
	r=n%3
	if r==0: print "5"*n
	elif r==1: 
		if n<10: print -1
		else: print "5"*(n-10) +"3"*10
	else: 
		if n<5: print -1
		else: print "5"*(n-5) +"3"*5
