#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	long long int N,i,j,start,total=1,num;
    vector<long long int>V;
    cin>>N;
    for(i=0;i<N;i++)
    {
        cin>>num;
        V.push_back(num);
    }
    sort(V.begin(),V.end());
    start=V[0];
    for(i=0;i<N;i++)
    {
        if(V[i]-start>4)
        {
            start=V[i];
            total++;
        }
    }
    cout<<total;
    return 0;
}
