		#include<bits/stdc++.h>
		using namespace std;
		
		int gcd(int a,int b)
		{
			if(b==0)return a;
			gcd(b,a%b);
		}
		int main()
		{
			int t;
			cin>>t;
			while(t--)
			{
				int x,y;
				cin>>x>>y;
				string str1,str2;
				cin>>str1;
				if(str1 == "Chandu")
				str2="Daspal";
				else
				str2= "Chandu";
				int count=0;
				int k;
				while(x != 0 && y!=0)
				{
					k = gcd(x,y);
				//	cout<<"gcd ==<<"<<k<<endl;
					x = x - k;
					y=  y-k;
					count++;
				}
				if(count%2!=0)
					cout<<str1<<endl;
				else
					cout<<str2<<endl;
			}
		}
