#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define N 100009

int a[N];
vector <int> primes;
 
void sieve() {
	int i,j;
	for (i = 2; i <= N; i++) {
		if (!a[i]) {
			for (j = i * 2; j <= N; j += i) {
				a[j] = 1;
			}
			primes.push_back(i);
		}
	}
}
 
int divisors (int n) {
	int i,cnt,res;
	res = 1;
	for (i = 0; primes[i] * primes[i] <= n; i++) {
		cnt = 0;
		while (n % primes[i] == 0) {
			n = n / primes[i];
			cnt += 1;
		}
		res = res * (cnt + 1);
	}
	if (n != 1) {
		res = res * 2;
	}
	return res;
}

int main()
{
	sieve();
    int n;
    long int x;
    cin>>n;
    while(n--){
    	cin>>x;
    	if(divisors(x)<4){
    		cout<<"NO\n";
    	}
    	else{
    		cout<<"YES\n";
    	}
    }
    return 0;
}
