#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    int i,n,t,j,maxi;
    scanf("%d",&t);
    while(t--)
    {
    	maxi=0;
    	scanf("%d",&n);
    	vector<int> v(n+1),dp(n+1,1);
    	for(i=1;i<=n;i++)
    	{
    		scanf("%d",&(v[i]));
    		if(v[i]>0)
    			maxi=1;
    	}
    	for(i=2;i<=n;i++)
    	{
    		for(j=1;j<=i;j++)
    		{
    			if(v[i]>v[j])
    			{
    				if((v[i]>0)&&(v[j]>0))
    				{
    					if(dp[i]<dp[j]+1)
    					{
    						dp[i]=dp[j]+1;
    						maxi=max(maxi,dp[i]);
    					}
    				}
    			}
    		}
    	}
    	printf("%d\n",n-maxi);
    }
    return 0;
}
