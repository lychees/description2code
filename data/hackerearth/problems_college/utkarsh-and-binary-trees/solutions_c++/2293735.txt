#include <iostream>
using namespace std;
#define ll long long
#define M 1000000007
int main()
{
    int n,h;
    cin>>n>>h;
    long long dp[n][h+1];
    
    for(int i=0;i<=h;i++)
    	dp[0][i]=0;
    for(int j=0;j<n;j++)
    	dp[j][0]=0;
    dp[0][0]=1;
    for(int i=1;i<n;i++)
    {
    	for(int j=1;j<=h;j++)
    	{
    		if(j>i)
    		{
    			dp[i][j]=dp[i][j-1];
    			continue;
    		}
    		ll sum=0;
    		for(int k=1;k<=i;k++)
    		{
    			if(k==1 || k==i)
    				sum+= (dp[i-1][j-1]);
    			else
    				sum+= (dp[k-1][j-1]*dp[i-k][j-1])%M;
    			sum%=M;
    			//cout<<"sum "<<sum<<" ";
    		}
    		dp[i][j]=sum;
    		//cout<<"dp["<<i<<"]["<<j<<"] "<<dp[i][j]<<" ";
    	}
    	//cout<<endl;
    }
    ll ans=0;
    for(int i=1;i<=n;i++)
    {
    	if(i==1 || i==n)
    		ans += (i*dp[n-1][h-1])%M;
    	else
    		ans+=(i*((dp[i-1][h-1]*dp[n-i][h-1])%M))%M;
    	ans%=M;
    	//cout<<ans<<endl;
    }
    cout<<ans<<endl;
    return 0;
}
