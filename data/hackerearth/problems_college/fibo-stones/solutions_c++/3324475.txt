#include<iostream>
using namespace std;

int main()
{
    long long int n1=1,n2=1,n,m,temp,sum=2;
    cin>>n>>m;
    for(int i=2;i<n;i++)
    {   temp=n2;
        n2=n1+n2;
        n1=temp;
        sum+=n2;
    }
    for(int i=0;i<m;i++)
    {   cin>>temp;
        sum-=temp;
    }
    cout<<sum<<endl;
}
