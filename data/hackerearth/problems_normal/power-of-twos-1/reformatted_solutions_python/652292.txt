import math

total_lines = int(raw_input())
try:
    while total_lines > 0:
        n = int(raw_input())
        n_squareroot = int(math.sqrt(n))
        sum_result = 0.0
        for i in range(1, n_squareroot + 1):
            sum_result += n / i

        answer = 2 * sum_result - n_squareroot * n_squareroot - n
        print(int(answer))
except BaseException:
    pass
