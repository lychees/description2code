#include<iostream>
using namespace std;
int main()
{
	int t;
	
	cin>>t;
	while(t--)
	{
		int n,k,i,c=0;
		cin>>n>>k;
		int a[n];
		for( i=0;i<n;i++)
		{
			cin>>a[i];
			if(a[i]<=0)
			{
				c++;
			}
		}
		if(c>=k)
		{
			cout<<"NO\n";
		}
		else
		{
			cout<<"YES\n";
		}
		
	}
	return 0;
}
