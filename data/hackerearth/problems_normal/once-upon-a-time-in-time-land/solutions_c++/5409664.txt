#include <iostream>
using namespace std;
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
        int n,k,i;
        cin>>n>>k;
        int a[n];
        for(i=0;i<n;i++)
            cin>>a[i];
        long long dp[n];
        for(i=0;i<n;i++)
            dp[i]=0;
        if(a[0]>0) dp[0] = a[0];
        for(i=1;i<=k;i++)
        {
            if(a[i]<0) dp[i]=dp[i-1];
            else if(a[i]<dp[i-1]) dp[i] = dp[i-1];
            else dp[i]=a[i];
        }
        for(i=k+1;i<n;i++)
        {
            if(a[i]<=0) dp[i]=dp[i-1];
            else
            {
                dp[i]=a[i]+dp[i-k-1];
                for(int j=i-k;j<i;j++)
                    if(dp[i]<dp[j]) dp[i]=dp[j];
            }
        }
        cout<<dp[n-1]<<endl;
    }
}
