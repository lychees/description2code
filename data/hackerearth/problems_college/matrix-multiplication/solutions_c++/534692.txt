#include <iostream>
#include <cstdio>
#include <cmath>
#include <string>
using namespace std;

int main() {
	int t;
	cin >> t;
	while (t--) {
		int x1, x2, y1, y2;
		cin >> x1 >> x2 >> y1 >> y2;
		if (x1 == y2 && x2 == y1)
			printf("2\n");
		else if (x2 == y1)
			printf("1\n");
		else if (x1 == y2)
			printf("1\n");
		else
			printf("0\n");
	}
}
