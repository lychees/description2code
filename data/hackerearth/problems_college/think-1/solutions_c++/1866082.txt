ll reverse(ll n){
	ll n1=0;
	while(n!=0){
			n1 = n1*10 + n%10;
			n /= 10;
		}
	return n1;
}
ll get_int(ll *i) {
	register ll a, sign;
	*i = 0;
	sign = 0;
	
	while((a = getchar_unlocked()) < 33);
	if(a == '-') {
		sign = 1;
	} else {
		*i = a - '0';
	}
	while((a = getchar_unlocked()) > 33) {
		*i = (*i << 1) + (*i << 3) + a - '0';
 	}
 	return sign?-*i:*i;
}
