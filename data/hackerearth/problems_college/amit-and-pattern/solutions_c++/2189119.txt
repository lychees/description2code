#include<cstdio>
#include<iostream>
#include<cmath>
using namespace std;
int  main()
{
	int n,m;
	cin>>n>>m;
	int a[n+1][m+1];
	int i,j,k;
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			cin>>a[i][j];
		}
	}
	for(k=2;k<=n+m;k++)
	{
	   for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
		   if((i+j)==k)
		   {
		   	cout<<a[i][j]<<" ";
		   	break;
		   }
        	}
	}
	cout<<"\n";
	}
}
