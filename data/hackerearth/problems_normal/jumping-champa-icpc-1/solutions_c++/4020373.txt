#include <bits/stdc++.h>
using namespace std;
int main()
{
   int t;
    cin>>t;
    while(t--)
    {
       long long int n,q,a[1009];
       cin>>n>>q;
       for (int i = 0; i < n; ++i)
       {
           cin>>a[i];
       }
       sort(a,a+n);
       long long int cost = 0;
       for (int i = n-1; i >0 ; i--)
       {
          // cout<<q*(a[i]-a[i-1]);
           cost = cost+(q*(a[i]-a[i-1]));


       }
       

       cout<<cost<<endl;

    }
}
