#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
static char in[1000001],f[1000001],ti[1000001],tf[1000001];
static int ai[100000],bi[100000],af[100000],bf[100000];
int main()
{
    int i,j,t,k,l,m,n,o,flag;
    cin>>t;
    while(t--)
    {
        k=m=0;
        scanf("%s %s",in,f);
        l=strlen(in);
        k=0;
        n=o=0;
        for(i=0;i<l;i++)
        {
            if(in[i]=='A')
                ai[n++]=i;
            if(in[i]=='B')
                bi[o++]=i;
        }
        n=o=0;
        for(i=0;i<l;i++)
        {
            if(f[i]=='A')
                af[n++]=i;
            if(f[i]=='B')
                bf[o++]=i;
        }
        for(i=0;i<l;i++)
        {
            if(in[i]!='#')
                ti[k++]=in[i];
            if(f[i]!='#')
                tf[m++]=f[i];
        }
        ti[k]='\0';
        tf[m]='\0';

        if(strcmp(ti,tf)==0)
        {
            flag=0;
            for(i=0;i<n;i++)
                if(ai[i]<af[i])
            {
                cout<<"No\n";
                flag=1;
                break;
            }
            for(i=0;i<o;i++)
                if(bi[i]>bf[i])
            {
                cout<<"No\n";
                flag=1;
                break;
            }
            if(flag==0)
            cout<<"Yes\n";
        }
        else
            cout<<"No\n";
    }
    return 0;
}
