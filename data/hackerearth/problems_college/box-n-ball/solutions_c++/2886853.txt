#include <cstdlib>
#include <iostream>
#include <algorithm>

using namespace std;
double a,b,c;
long T;
int main(int argc, char *argv[])
{   cin>>T;
    while(T)
    {  cin>>a>>b>>c;
       cout.setf(ios::fixed,ios::floatfield);
       cout.precision(1); 
       cout<<min(min(a,b),c)/2<<endl;
       
       T--;        
    }
    
    return EXIT_SUCCESS;
}
