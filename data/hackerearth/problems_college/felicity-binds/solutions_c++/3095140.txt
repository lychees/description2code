 #include<bits/stdc++.h>
using namespace std;
 int main(){
        int i,j,t,c,n;
        int arr[10001];
        int brr[10001];
        cin>>t;
        while(t--)
        {
c=0;
        scanf("%d", &n);
           for(i = 0;i <n;i++)
            {
            cin>>arr[i];
            }
            for(int j=0;j<n;j++)
            {
            cin>>brr[j];

            }
            sort(arr,arr+n);
            sort(brr,brr+n);
            for(int k=0;k<n;k++)
            {
            c+=arr[k]*brr[k];
            }
            cout<<c<<endl;
            }
 }
