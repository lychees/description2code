#include <bits/stdc++.h>
using namespace std;

long long int gcd(long long int a, long long int b) {
	while(a && b)
		if(a > b) a %= b;
		else b %= a;
	return (a | b);
}

int main() {
	int test; scanf("%d", &test);
	long long int a, b, c;
	while(test--) {
		scanf("%lld%lld%lld", &a, &b, &c);
		if(gcd(a, b) == 1LL && gcd(b, c) == 1LL) {
			if(b >= a && b >= c && b * b == a * a + c * c) printf("YES\n");
			else if(c >= a && c >= b && c * c == a * a + b * b) printf("YES\n");
			else if(a >= b && a >= c && a * a == b * b + c * c) printf("YES\n");
			else printf("NO\n");
		}
		else printf("NO\n");
	}
    return 0;
}
