#include <bits/stdc++.h>
using namespace std;
int first[100004],second[100005];
int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=0;i<n;i++){

		cin>>first[i];
	}
	for(int i=0;i<n;i++){

		cin>>second[i];
	}

	int head=first[0];
	int i=0;
	while(second[i]!=head)
		i++;

	int x;
	char c;
	int req=i;
	int lptr=0;
	bool flag=false;
	int rptr=n-1;
	int j=0;
	while(j<m)
	{
		cin>>c>>x;
		if(req==lptr){
			flag=true;
			break;
		}
		if(c=='R')
		{
			lptr=(lptr+x)%n;
			rptr=n-lptr-1;
		}
		else
		{
			rptr=(rptr+x)%n;
			lptr=n-rptr-1;
		}
		j++;
	}
	if(flag==true){
		cout<<j<<endl;
	}
	else
	{
		cout<<"-1"<<endl;
	}
	return 0;
}
