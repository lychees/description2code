num = int(raw_input())

for i in xrange(num):
    s1 = raw_input()
    s2 = raw_input()

    sp = s1 + s2
    sr = s2 + s1
    print 'YES' if sp == sp[::-1] or sr == sr[::-1] else 'NO'
