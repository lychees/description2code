#include <bits/stdc++.h>
using namespace std;

int main()
{
	int  num,a[100000],i,min=1000000000,k;
	cin>>num;
    cin>>k;
    for(i=0;i<num;++i)
    cin>>a[i];
    sort(a,a+num);
    for(i=num-1;i>=0&&(i-k+1)>=0;--i)
    {
    	if(a[i]-a[i-k+1]<min)
    	{
    		min=a[i]-a[i-k+1];
    	}
    }
    cout<<min<<endl;
    
    return 0;
}
