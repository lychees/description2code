ll lcm(ll a, ll b) { return a * (b / gcd(a, b)); }
ll power(ll a,ll b) {
  ll ans = 1;  
  while(b > 0){
    if(b & 1)
      ans = ((ans % mod) *(a % mod)) % mod;
    a=((a % mod)*(a % mod)) % mod;
    b >>= 1;
  }
  return ans;
}

int n, t, arr[100010];

int main() { 
  
  int i, j, k;

  cin>>t;

  while(t--) { 

    cin>>n;

    rep(i, 0, n) 
      cin>>arr[i];
    
    sort(arr, arr+n);

    i = n-1;
    j = n-2;
    k = n-3;
    
    bool flag = false;

    while(i >= 0 && j >= 0 && k >= 0) {
      
      if(arr[j] + arr[k] > arr[i]) {
	flag = true;
	cout<<arr[k]<<" "<<arr[j]<<" "<<arr[i]<<endl;
	break;
      }
      i--;
      j--;
      k--;
    }
    if(!flag)
      cout<<"-1\n";

  }
  return 0;
}
