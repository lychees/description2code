#include<iostream>
using namespace std;
int find_new(int );
int main()
{
	int n;
	cin>>n;
	if(n>=3)
	{
	int a=find_new(n)+find_new(n-1)+find_new(n-2);
	cout<<a<<endl;
    }
	else
	{
		if(n==0||n==1)
		cout<<"1"<<endl;
		if(n==2)
		cout<<"2"<<endl;
	}
	return 0;
}
int find_new(int n)
{
	if(n==0||n==2)
	return 1;
	if(n==1)
	return 0;
	return ((find_new(n-2))+find_new(n-3));
}
