#include <bits/stdc++.h>
using namespace std;

char S1[10000];
char S2[10000];

int main()
{
    int T;
    scanf("%d", &T);
    
    while(T--) {
    	int N;
    	scanf("%s%s%d", S1, S2, &N);
    	int len = strlen(S1);
    	for(int i = 0; i < len; ++i) {
    		if(N < 0 || S1[i] == S2[i]) continue;
    		if(N == 0){ N = -1; continue;}
    		N--;
    	}
    	if(N < 0 || (N&1)) {
    		puts("No");
    	} else {
    		puts("Yes");
    	}
    	
    }
    return 0;
}
