#include <iostream>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    for (int i=1;i<=tc;i++)
    {
    	unsigned long long a,b,c;
    	cin>>a;
    	cin>>b;
    	
    	int ap=a+1;
    	c=a & ap;
    	
    	for(int j=(ap+1);j<=b;j++)
    	{
    		c= c & j;
    	}
    	cout<<c<<"\n";
    	
    }
    return 0;
}
