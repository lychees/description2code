#include <iostream>
#include<algorithm>
#include<string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string w,r;
    	int i,j;
    	cin>>w;
    	cin>>r;
    	sort(r.begin(),r.end());
    	for(i=0,j=0;i<w.length();i++)
    	{
    		if(w[i]>r[j]&&r[j]!='\0')//replace
    		{
    			w[i]=r[j++];
    		}
    	}
    	cout<<w<<endl;
    }
    return 0;
}
