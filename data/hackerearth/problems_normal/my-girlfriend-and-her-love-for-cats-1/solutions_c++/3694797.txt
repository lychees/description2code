#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    long long int n;
    cin >> n;
    
    vector<long long int> s(n);
    vector<long long int> c(n);
    
    for (long long int i = 0; i < n; i++) {
    	cin >> s[i]; // strength of ith cat
    }
    
    for (long long int i = 0; i < n; i++) {
    	cin >> c[i]; // calorie value of ith packet
    }
    
    sort(s.begin(), s.end());
    sort(c.begin(), c.end());
    
    long long int ans = 0;
    for (long long int i = 0; i < n; i++) {
    	ans += s[i] * c[i];
    }
    
    cout << ans << endl;
    return 0;
}
