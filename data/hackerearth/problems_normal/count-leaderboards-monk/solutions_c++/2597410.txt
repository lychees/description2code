#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
long long dp[2005][2005];
long long fact[1005];
long long nCr(long long n, long long r);
void calc_factorial();
int main()
{
    int n,k;
    scanf("%d%d",&n,&k);
    long long q=nCr(n,k)%mod;
    long long p=0,ans;
    long long dp2[n+1];
    int i=1,j;
    calc_factorial();
    dp2[0]=1;
    dp2[1]=1;
    for(i=1;i<n;i++)
    {
         p=0;
         for(j=0;j<=i;j++)
         {
            long long tmp;
            tmp=(((fact[j+1]*nCr(i,j))%mod)*dp2[i-j])%mod;
            p=(p+tmp)%mod;
         }
         dp2[i+1]=p;
    }
    ans=(dp2[n]*q)%mod;
    cout<<ans;
    return 0;
}
long long nCr(long long n, long long r)
{
    if(n==r)
        return dp[n][r] = 1;
    if(r==0)
        return dp[n][r] = 1;
    if(r==1)
        return dp[n][r] = n;

    if(dp[n][r])
        return dp[n][r];

    return dp[n][r] = ((nCr(n-1,r)%mod + nCr(n-1,r-1)%mod)%mod);
}
void calc_factorial()
{
    long long tmp=1;
    long i;
    fact[0]=1;
    for(i=1;i<=1000;i++)
    {
        tmp=(tmp*i)%mod;
        fact[i]=tmp;
    }
    return;
}
