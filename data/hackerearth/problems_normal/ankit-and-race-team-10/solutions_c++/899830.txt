#include <stdio.h>
#include<iostream>
#include<algorithm>
#include<cstring>
#define MOD 1000000007
using namespace std;


int pwr(int n, int p)
{
if(p==0) return 1;	

 long long int temp= pwr(n,p/2);	
 if(p%2==0)
  return ( temp * temp )%MOD;  
 else
   return ( (  (temp*temp) %MOD )* n )% MOD; 
 	
}

int main()
{
    
    int q;
    
    int count=1;
    
    cin>>q;
    
    int n , k;
    while(count<=q)
    {
    	
    	cin>>n;
    	cin>>k;
    	long long int a[n+1];
    	long long int c[n+1];
    	
    	memset(c,0, (n+1)* sizeof(long long));
    	
    	for(int i=1;i<=n;i++)
    	cin>>a[i];a[0]=0;
    	
    	sort(a+1,a+n+1);
    	
    	c[n-k+1]=1;
    	
    	for(int i=n; i>=1; i--)
		{
		   if(n-i>k-1)
	       c[i]= (((n-i) *(long long) pwr( n-i-k+1 , MOD-2 ))%MOD * c[i+1])%MOD ;   		
		}
		
    	long long int sum=0;
    	for(int i=1;i<=n;i++)
		{  
           sum = (sum+((a[i]*c[i])%MOD))%MOD;  
		}    	
    	
    	cout<<sum<<endl;
    	count++;
    	
    }
    
    
    
    
    
}
