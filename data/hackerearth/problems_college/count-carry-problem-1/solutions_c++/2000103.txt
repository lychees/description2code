#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ll t,i,ans;
    ll c[20],d[20];
    cin>>t;
    while(t--)
    {
    	ll a,b; ans=0;
    	cin>>a>>b;
    	memset(c,0,sizeof(c));
    	memset(d,0,sizeof(d)); i=0;
    	while(a)
    	{
    		c[i++]=a%10;
    		a/=10;
    	}
    	ll p=i;
    	i=0;
    	while(b)
    	{
    		d[i++]=b%10;
    		b/=10;
    	}
    	p=max(p,i); ll carry=0;
    	for(i=0;i<=p+2;i++)
    	{
    		ll k=(c[i]+d[i]+carry);
    		if(k>=10) { carry=1; ans++;}
    		else carry=0;
    		//cout<<carry<<" "<<ans<<" "<<c[i]<<" "<<d[i]<<endl;
    	}
    	if(!ans) cout<<"No carry operation\n";
    	else if(ans==1) cout<<1<<" carry operation\n";
    	else cout<<ans<<" carry operations\n";
    }
    return 0;
}
