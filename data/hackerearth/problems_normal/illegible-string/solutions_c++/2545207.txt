#include <bits/stdc++.h>
using namespace std;

int main(){

	long long n, vcount = 0, wcount = 0;
	long long maxi = 0, mini = 0;
	string str;
	bool flag = false;

	scanf("%lld", &n);
    cin >> str;
    str += "a";

    for(int i=0; i<str.size(); i++){
        if(str[i] == 'v'){
            vcount++;
            flag = true;
        }
        else if(str[i] == 'w'){
            wcount++;
            flag = true;
        }
        else if(flag){
            flag = false;
            mini = mini + vcount / 2;
            maxi = maxi + wcount;
            vcount = 0;
            wcount = 0;
        }
    }

	printf("%lld %lld", n - mini, n + maxi);

    return 0;
}
