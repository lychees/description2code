#include<iostream>
#include<cstring>
#include<ctype.h>
#include<string>
using namespace std;
int main()
{
	int k=0;
	int t;
	cin>>t;
	t++;
	while(t--)
	{
     k=0;
	string a;
	string stri[10005];
	getline(cin,a);
	string str="";
	for(int i=0;i<a.length();i++)
	{
	if(a[i]!=' ')
	{
		char temp=a[i];
		str+=temp;
	//	cout<<str<<" ";
	}
	else
	{
		stri[k]=str;
		k++;
		str="";
	}
	}
	stri[k]=str;
	for(int i=0;i<=k;i++)
	{
		int f=-1;
	for(int j=0;j<i;j++)
	{
	if(stri[i]==stri[j])
	{
		f=1;
		break;
	}
	}
	if(f==-1)cout<<stri[i]<<" ";
	}
	cout<<endl;
	}
	return 0;
}
