#include <iostream>
#include <cmath>
using namespace std;
 
int main()
{
    int T;
    cin>>T;
    for(int i=0;i<T;i++)
    {
    	int H,M;
    	cin>>H>>M;
    	int A=fabs((30*H)+(M/2)-(6*M));
    	if(A<360-A)
    	cout<<A<<endl;
    	else
    	cout<<360-A<<endl;
    }
    return 0;
}
