#include<iostream>
using namespace std;
int max(int a,int b)
{
    if(a>b)
        return a;
    else
        return b;
}

int knapsack(int w[],int v[],int n,int M)
{
    int a[n+1][M+1],i,j;
    for(i=0;i<=n;i++)
        for(j=0;j<=M;j++)
    {
        if(i==0 || j==0)
            a[i][j]=0;
        else if(w[i-1]<=j)
        {
            a[i][j]=max(v[i-1] + a[i-1][j-w[i-1]],  a[i-1][j]);
        }
        else
            a[i][j]=a[i-1][j];
    }
return a[n][M];
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,M,i,ans;
        cin>>n>>M;
        int w[n],v[n];
        for(i=0;i<n;i++)
            cin>>w[i];
             for(i=0;i<n;i++)
            cin>>v[i];
        ans=knapsack(w,v,n,M);
        cout<<ans<<endl;
    }
return 0;
}
