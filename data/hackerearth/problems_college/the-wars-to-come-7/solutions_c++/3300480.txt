#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,i,x,l=0;
    cin >> t;
    vector <int> v(t);
    for (i = 0; i < t; i++) {
    	cin >> v[i];
    }
    cin >> x;
    sort(v.begin(), v.end());
    reverse(v.begin(), v.end());
    i = 0;
    while (l < x) {
    	l = l + v[i];
    	i++;
    }
    cout << i << endl;
    return 0;
}
