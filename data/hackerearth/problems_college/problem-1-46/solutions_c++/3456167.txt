#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll  freq[100000+3][26];
int main()
{
	ll  n;
	cin>>n;
	string str;
	cin>>str;
	bool flag;
	memset(freq,0,sizeof freq );
	//processing the given information
	for(int i=0;i<n;++i)
	{
		if(i!=0)
		{
			for(int j=0;j<26;++j)
			{
				freq[i][j]=freq[i-1][j];
			}
		}
		++freq[i][str[i]-'a'];
	}
	ll  q;
	cin>>q;
	while(q--)
	{
		int l,r;
		cin>>l>>r;
		--l,--r;
		int oddity=0;
		int diff=0;
		 flag=true;
		if((r-l+1)<=1)flag=false;
		else
		{
			for(int i=0;i<26;++i)
			{
				diff=freq[r][i]-(l==0?0:freq[l-1][i]);
				if(diff&1)++oddity;
			}
			if(oddity>2)flag=false;
		}
		(flag==false)?puts("NO"):puts("YES");
	}
	return 0;
}
