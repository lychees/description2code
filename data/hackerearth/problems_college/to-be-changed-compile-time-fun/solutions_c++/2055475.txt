	LL T,n,i,j,k,p,q,x,y,z;
	
	
	cin>>n;
	VL V(n);
	REP (i,n) cin>>V[i];
	LL ans=0;
	while (V.size()>2)
		{
		LL p=V[0]*V[2],id=1;
		for (i=2; i<V.size()-1; i++)
			{
			if (V[i-1]*V[i+1]>p)
				{
				p=V[i-1]*V[i+1];
				id=i;
				}
			}
		ans+=p;
		V.erase(V.begin()+id);
		}
	cout<<ans<<"\n";
	return 0;
