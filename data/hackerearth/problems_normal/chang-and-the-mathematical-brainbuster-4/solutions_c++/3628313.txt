#include <iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int min(int a,int b)
{
    if(a>b)
		return b;
	return a;
}
int main() {
	// your code goes here
	int N,i;
	scanf("%d",&N);
	int a[N];
	for(i=0;i<N;i++)
		scanf("%d",&a[i]);
	sort(a,a+N);
	int Min=1000000,flag=0,j;
	for(i=N-1;i>=0;i--)
	{
		flag=0;
		for(j=0;j<i;j++)
		{
			if(a[i]-3*a[j]<=0){
				flag=1;
				break;
			}
		}
		if(flag)
			Min=min(Min,N-(i-j+1));
	}
	printf("%d\n",Min);
	return 0;
}
