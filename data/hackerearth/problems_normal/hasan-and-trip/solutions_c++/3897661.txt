#define trace1(arr,size) for(long long i=0;i<size;i++) cout<<arr[i]<<" "; cout<<endl;
#define trace2(arr,row,col) for(long long i=0;i<row;i++){ for(long long j=0;j<col;j++) cout<<arr[i][j]<<" "; cout<<endl; }

unsigned long long w,l,tot=0,prev_tot;

int main(){
    ios_base::sync_with_stdio(false);
    //freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
    int n;
    cin>>n;
    long x[n],y[n],f[n];
    double dp[n];
    for(int i=0;i<n;i++){
        cin>>x[i]>>y[i]>>f[i];
        dp[i]=-999999;
    }
    dp[0]=0;
    for(int i=0;i<n;++i){
        dp[i]+=f[i];
        for(int j=i+1;j<n;++j){
            dp[j]=max(dp[j],dp[i]-sqrt(pow(x[j]-x[i],2)+pow(y[j]-y[i],2)));
        }
    }
    cout<<fixed<<setprecision(6)<<dp[n-1];
}
