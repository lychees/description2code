#include <iostream>
using namespace std;

int main()
{
    long sum[1000005] = {};
    for(int i=0; i<1000005; i++) sum[i]=0;
    for(int i=1; i<1000005; i++){
    	for(int j=2*i; j<1000005; j+=i){
    		sum[j]++;
    	}
    }
    
    for(int i=2; i<1000005; i++){
    	if (sum[i]==0) cout << "EEE" <<endl;
    	if (sum[i]==1){
    		//
    	} else{
    		sum[i]=0;
    	}
    	sum[i]=sum[i-1]+(sum[i]==1?i: 0);
    }
    
   
    
    int t;
    cin>>t;
    while(t-->0){
    	int s, e;
    	cin>>s>>e;
    	if (s-1>=0)
    		cout << (sum[e]-sum[s-1])<<endl;
    	else
    		cout << sum[e] << endl;
    }
}
