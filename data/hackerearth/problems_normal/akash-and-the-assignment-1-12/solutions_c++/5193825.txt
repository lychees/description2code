#include <iostream>
#include <algorithm>
#include <map>
using namespace std;
struct bi
{
	map<char,int> m;
	bi()
	{
		for(int i=0;i<26;i++)
		{
			m.insert(make_pair((i+'a'),0));
		}
	}
};

void make(string s, bi * data)
{
	for(int i=1;i<=s.size();i++)
	{
		data[i].m=data[i-1].m;
		map<char,int>::iterator rt=data[i].m.find(s[i-1]);
		if(rt!=data[i].m.end())
		{
			rt->second++;
		}
	}
}

int main()
{
    int n,q;
    cin>>n>>q;
    string s;
    cin>>s;
    bi *data=new bi[(int)s.size()+1];
    make(s,data);
    cin.ignore();
    while(q--)
    {
    	int l,r,k;
    	cin>>l>>r>>k;
    	map<char,int> m;
    	map<char,int>::iterator rt=data[r].m.begin();
    	map<char,int>::iterator lt=data[l-1].m.begin();
    	while(rt!=data[r].m.end())
    	{
    		m.insert(make_pair(rt->first,rt->second-lt->second));
    		rt++,lt++;
    	}
    	rt=m.begin();
    	while(rt!=m.end())
    	{
    	//	cout<<rt->first<<" "<<rt->second<<endl;
    		k-=rt->second;
    		if(k<=0)
    		{
    			cout<<rt->first<<endl;
    			break;
    		}
    		rt++;
    	}
    	if(k>0)cout<<"Out of range"<<endl;
    }
    return 0;
}
