#include <bits/stdc++.h>
#include <iomanip>
#include <cstdlib>

using namespace std;

int hcf(int x, int y)
{
	int i, temp;
	
	while(y)
	{
		temp = x%y;
		x = y;
		y = temp;
	}
	return x;
}

int main()
{
	int i=0,j,k,n,count;
	int t;
	cin >> t;
 	
	while(t--)
	{
		count =0;
		cin >> n;
		
		if(n==1)
		{
			cout << "Incorrect" << endl;
			continue;
		}
		
		if(n==2)
		{
			cout << "Incorrect" << endl;
			continue;
		}
		
		for(i = 2; i < n;i++)
		{
			k = hcf(i,n);
			
			if(k == 1)
				count++;
		}
		
		count +=1;
		//cout << count << endl;
		j =0;
	
		for(i =2; i *i <=count; i++)
		{
			if(count % i == 0)
			{
				j = 1;
				cout << "Incorrect" << endl;
				break;
			}
		}
		
		if(j==0)
		cout << "Correct" <<endl;
	}
}
