#include <iostream>
#include <limits.h>
using namespace std;

int ass=INT_MAX,val;

void calc(int i,int j,int k,int p,int arr[],int n)
{
	if(p==n)
	{
		val=max(max(i,j),k);
		if(val<ass)
		ass=val;
	}
	else
	{
		calc(i+arr[p],j,k,p+1,arr,n);
		calc(i,j+arr[p],k,p+1,arr,n);
		calc(i,j,k+arr[p],p+1,arr,n);
	}
}

int main()
{
	int n,i;
	cin>>n;
	int arr[n];
	for(i=0;i<n;i++)
	{
		cin>>arr[i];
	}
	calc(0,0,0,0,arr,n);
	cout<<ass;
	return 0;
}
