#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;cin>>t;
	if(t < 1 or t > 10) {
		cout<<"Invalid Test"<<endl;
		return 0 ;
	}
	while(t--) {
		string s;cin>>s;
		if(s.size() < 1 or s.size() > 100) {
			cout<<"Invalid Input"<<endl;
			continue;
		}
		int lCount = 0, uCount = 0;
		for(int i = 0; i <s.size(); i++) {
			if(s[i] >= 'A' and s[i] <= 'Z') {
				uCount++;
			} else if(s[i] >= 'a' and s[i] <= 'z') {
				lCount++;
			}
		}
		if(lCount == 0 and uCount == 0) {
			cout<<"Invalid Input"<<endl;
		} else if((lCount == 0 and uCount > 0) or (lCount > 0 and uCount == 0)) {
			cout<<"0"<<endl;
		} else {
			if(lCount < uCount) {
				cout<<lCount<<endl;
			} else {
				cout<<uCount<<endl;
			}
		}
	}
    return 0;
}
