#include <stdio.h>
#include <iostream>
#include <algorithm>

using namespace std;
typedef long long LL;

int a, b, c;
int values[5];

int main()
{
  int t;

  cin >> t;
  while(t--)
  {      
    cin >> a >> b >> c;    
    
    // Sort values.
    values[0] = a; values[1] = b; values[2] = c;
    sort(values, values + 3);
    a = values[0]; b = values[1]; c = values[2];

    int moves = 0;
    while(!(a + 1 == b && b + 1 == c))
    {
      if(c - b > b - a)
      {
        ++moves;
        int aPosition = c - 1;
        a = b; b = aPosition;
      }
      else
      {
        ++moves;
        int cPosition = a + 1;
        c = b; b = cPosition;
      }
    }

    cout << moves << endl;
  }

  return 0;
}
