def isPalindrome(str, str_len, set_len):
    set1 = set(str)
    D = {}
    if set_len == 2 and set_len == str_len:
        return 1
    for each in set1:
        D[each] = str.count(each)
    bOdd = 0
# 	print D
    for key in D:
        if D[key] % 2 != 0:
            bOdd = bOdd + 1
# 	print bOdd
    if bOdd >= 2:
        return bOdd-1
    return 0


def minimal_palindrome(str):
    set1 = set(str)
    length = len(str)
    set_len = len(set1)
    return isPalindrome(str, length, set_len)


cases = int(raw_input())
answers = []
for case in range(cases):
    answers.append(minimal_palindrome(raw_input()))

    print ans
