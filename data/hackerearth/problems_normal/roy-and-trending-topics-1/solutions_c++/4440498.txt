#include <iostream>
#include <algorithm>
using namespace std;

class Topic{
public:
	long long id;
	long long change;
	long long score;
};

bool comp(Topic a, Topic b){
	if(a.change == b.change){
		return a.id > b.id;
	}
	return a.change > b.change;
}

int main()
{
    long long n,temp,inc;
    cin >> n;
    Topic t[n];
    for(int i = 0; i < n; i++){
    	cin >> temp;
    	t[i].id = temp;
    	cin >> temp;
    	cin >> inc;
    	t[i].change = 50*inc;
    	cin >> inc;
    	t[i].change += 5*inc;
    	cin >> inc;
    	t[i].change += 10*inc;
    	cin >> inc;
    	t[i].change += 20*inc;
    	t[i].score = t[i].change;
    	t[i].change -= temp;
    }
    sort(t, t+n, comp);
    for(int i = 0; i < 5; i++){
    	cout << t[i].id << " " << t[i].score << endl;
    }
    return 0;
}
