LL mpow(LL a, LL n)

LL arr[1009][1009]={};
void pre()
{
    arr[0][0]=1ll;
    for(int i=0; i<=1000; i++)
    {
	arr[i][0]=arr[i][i]=1ll;
	for(int j=1; j<i; j++)
	    arr[i][j]=(arr[i-1][j]+arr[i-1][j-1])%MOD;
    }
}
int main()
{
    pre();
    int t;
    cin >> t;
    while(t--)
    {
	LL n,m,i,ans=0;
	cin >> n >> m;
	for(i=0; i<=m; i++)
	    ans=(ans+arr[n][i])%MOD;
	cout << ans << endl;
    }
    return 0;
}
