#include<bits/stdc++.h>
using namespace std;
#define in(n) scanf("%d",&n)
void update(vector<int> &bit,int M,int j)
{
    for(;j<M;j+=j&(-j))
        bit[j]++;
    return ;
}
int que(vector<int> &bit,int j)
{
    int m=0;
    for(;j>0;j-=j&(-j))
        m+=bit[j];
    return m;
}
int main()
{
    int n,x,i,y,M,cnt;
    in(n);
    vector<pair<int,int> > v;
    for(i=1;i<=n;i++)
    {
        in(x);
        v.push_back(make_pair(x,i));
    }
    sort(v.begin(),v.end());
    vector<int> ar(n+1,0);
    cnt=1,y=v[0].second,ar[y]=1;
    for(i=1;i<n;i++)
    {
        if(v[i].first>v[i-1].first)cnt++;
        y=v[i].second,ar[y]=cnt;
    }
    M=cnt+1;
    vector<int> bit(M,0);
    for(i=1;i<=n;i++)
    {
        y=que(bit,ar[i]-1);
        update(bit,M,ar[i]);
        printf("%d ",y);
    }
    cout << endl;
}
