#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    for(int i=0; i<t; i++){
    	string s1, s2;
    	bool found = false;
    	cin >> s1;
    	cin >> s2;
    	for(auto ch:s1){
    		if(s2.find(ch) != string::npos){
    			found = true;
    			break;
    		}
    	}
    	if(found){
    		cout << "YES" << endl;
    	}
    	else{
    		cout << "NO" << endl;
    	}
    }
    return 0;
}
