#include <bits/stdc++.h>
#define ll long long
using namespace std;
pair<ll int,ll int> p[10100];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n,m;
		cin>>n>>m;
		ll int i;
		for(i=0;i<n;i++)
		{
			cin>>p[i].second>>p[i].first;
		}
		sort(p,p+n);
		for(i=0;i<n;i++)
		{
			if(p[i].first<=m)
			{
				m+=p[i].second;
			}
			else
			break;
		}
		if(i==n)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
	}
	return 0;
}
