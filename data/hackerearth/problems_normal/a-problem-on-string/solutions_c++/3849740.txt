#include <iostream>
using namespace std;

char str[1000006];
int count[1000006];
long long ans[26][26];

int main()
{
	int n,q;
	char a,b;
	cin>>n;
	cin>>str;
	for (int i=0;i<n;i++) {
		count[str[i]-'a']++;
	}
	for (int i=0;i<n;i++) {
		count[str[i]-'a']--;
		for (int j=0;j<26;j++) {
			ans[str[i]-'a'][j] += count[j];
		}
	}
	cin>>q;
	while(q--) {
		cin>>a>>b;
		long long fans;
		if (a != b) {
			fans = ans[a-'a'][b-'a'] + ans[b-'a'][a-'a'];	
		} else {
			fans = ans[a-'a'][b-'a'];
		}
		
		cout<<fans<<endl;
	}
    return 0;
}
