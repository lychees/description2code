#include <iostream>
#include <string.h>
using namespace std;
int main()
{
	int tt;
	string n;
	cin>>tt;
	for(int qq=1;qq<=tt;qq++){
		cin>>n;
        string result = n; 
        if (n.length() > 0) { 
            int l = n.length() - 1; 
            int s = 0; 
            int e = l; 
            while ( e > s) { 
                if (n[s] == n[e]) { 
                    ++s; --e; 
                } else { 
                    s -= (l - e - 1); 
                    e = l; 
                } 
            } 
            string start = n.substr(0, s - (l - e)); 
            for(int i=start.length()-1;i>=0;i--)
            result+=start[i]; 
        } 
        cout<<result.length()<<"\n"; 
    } 
    return 0;
}
