#include <iostream>
using namespace std;

int a[10][10];

int main()
{
	int t, i, j, n, m;
	
	a[1][0] = 1;

	a[2][1] = 1;

	a[3][3] = 1;

	a[4][4] = 3;
	a[4][5] = 6;
	a[4][6] = 1;
	
	cin>>t;
	
	long long int p = 1000000007;
	
	while(t--)
	{
		cin>>n>>m;
		
		long long int ans; 
		
		if(n < 5)
		{
			cout<<a[n][m]<<"\n";
		}
		
		else
		{
			if(m == 2*n - 4 || m == 2*n - 3)
			{
				ans = n;
				ans = ans*(n-1);
				ans = ans/2;
				ans = ans%p;
				
				cout<<ans<<"\n";
			}
			
			else
			{
				cout<<"0\n";
			}
		}
	}
}
