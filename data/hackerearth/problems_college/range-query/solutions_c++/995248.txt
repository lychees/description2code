#include <iostream>
#include <vector>
using namespace std;

int ans[10005][105];

int main()
{
    int N, X, Y;
    cin>>N;
    int a[N+1], val;
    for(int i=0; i<N; i++)
    	cin>>a[i];
    for(int i=0; i<10001; i++)
    {
    	for(int j=0; j<101; j++)
    	{
    		if(i)
    			ans[i][j]=ans[i-1][j];
    		else
    		    ans[i][j]=0;
    	}
    	if(i<N)
    	ans[i][a[i]]++;
    }
    long Q;
    cin>>Q;
    while(Q--){
    	cin>>X>>Y>>val;
    	if(X)
    	cout<<ans[Y][val]-ans[X-1][val]<<endl;
    	else
    	cout<<ans[Y][val]<<endl;
    }
    return 0;
}
