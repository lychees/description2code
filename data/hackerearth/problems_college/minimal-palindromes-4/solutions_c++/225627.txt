#include <iostream>
#include <cstdio>
#include <string.h>

using namespace std;

int main()
{
    int t, i, j, len, no;
    char str[10001];
    cin>>t;
    while (t--)
    {
        cin>>str;
        i=0;
        len=strlen(str);
        no=str[0]-48;
        while (no>0)
        {
            if (i==len)
            {
                for (i=0; i<len; i++)
                {
                    cout<<(str[i]-48)/no;
                }
                cout<<endl;
                break;
            }
            if (((str[i]-48)%no)==0)
            {
                i++;
            }
            else
            {
                no--;
                i=0;
            }
        }
    }
    return 0;
}
