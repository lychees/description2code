#include<iostream>
#include<stdlib.h>
#include<cstdio>
using namespace std;
int main()
{
	int n=0;
	long long int max_length=0;
	scanf("%d",&n);
	long long int a[50005]={0};
	long long int dp[50005]={0};
	for(int i=0;i<n;i++)
	{
		scanf("%lld",a+i);
		dp[i]=1;
		for(int j=0;j<i;j++)
		{
			if(((a[i]<0 && a[j] >0 && abs(a[i]) >a[j]) || (a[i]>0 && a[j] <0 && a[i] >abs(a[j]))) && (1+dp[j] > dp[i]))
			{
				dp[i]=1+dp[j];
				}
			
			}
			if(dp[i] > max_length)
			max_length=dp[i];
		}
		printf("%lld",max_length);
		return 0;
	}
