#include <bits/stdc++.h>
using namespace std;

int main()
{
	vector<pair<int,string>> vt;
	int t;
	cin>>t;
	int n;
	string str;
	for(int i=0;i<t;i++)
	{
		cin>>str >>n;
		vt.push_back(make_pair(-n,str));
	}
    sort(vt.begin(),vt.end());
    //vector<pair<int,string>>::iterator it;
    for(int it=0;it<vt.size();it++)
    {
    	cout<<vt[it].second<<" "<<-vt[it].first<<endl;
    }
    return 0;
}
