#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
          int t;
          scanf("%d",&t);
          while(t--)
          {
                    long long int n,rem=0,sum=0,c=0,i;
                    scanf("%lld",&n);
                    while(n>0)
                    {
                          rem=n%10;
                          sum=sum+rem;
                          n/=10;
                    }
                   for(i=2;i<=sum;i++)
                   {
                   if(sum%i==0)
                   {
                   c++;
                   }
                   }
                   if(c==1)
                    printf("YES\n");
                   else
                    printf("NO\n");
          }
          return 0;
}
