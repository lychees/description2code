#include <iostream>
#define MAX 100001
using namespace std;
int main()
{
    int i,n,t,*a,max;
    cin>>t;
    while(t--){
    	max=0;
    a=new int[MAX];
    cin>>n;
    while(n--){
    cin>>i;
    a[i]++;
    }
    for(i=0;i<MAX;i++){
    	if(a[i]>max)
    	max=a[i];
    }
    cout<<max<<endl;
    }
    return 0;
}
