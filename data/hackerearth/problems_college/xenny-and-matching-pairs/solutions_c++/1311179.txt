#include <bits/stdc++.h>
#include <set>
using namespace std;
int main() {
    int t,n,i,a;
    scanf("%d",&t);
    while(t--)
    {
    	set<int> s;
    set<int>:: iterator it;
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&a);
    		s.insert(a);
    		it=s.find(a);
    		++it;
    		if(it!=s.end())
    		  s.erase(it);	
    	}
    	printf("%d\n",s.size());
    	//cout<<s.size()<<endl;
    	//s.erase(s.begin(),s.end());
    }
             
    return 0;
}
