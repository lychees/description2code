using namespace std;

typedef pair<int,int>pii;
typedef long long LL;

int main()
{
	int t;
	scan(t);
	while(t--)
	{
		LL n;
		scanf("%lld",&n);
		LL i=(LL)sqrt(n)-82;
		if(i<=0)
			i=1;
		int f=0;
		while(i<=(LL)sqrt(n))
		{
			LL temp=i;
			LL b=0;
			while(temp>0)
			{
				b+=temp%10;
				temp/=10;
			}
			if(i*i+(b*i)==n)
			{
				cout<<i<<endl;
				f=1;
				break;
			}
			i++;
		}
		if(!f)
		{
			cout<<"-1\n";
		}
	}
}
