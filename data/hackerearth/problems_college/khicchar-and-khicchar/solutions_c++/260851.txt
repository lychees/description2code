#include<iostream>
using namespace std;
#include<stdlib.h>
#include<string.h>
int gcd(int m,int n)
{
	if(n==0||n==1)
	 return m;
	else
	 return gcd(n,m%n);
}
int main()
{
	char a[255];
	char b[255];
	char c[255];
	char d[255];
    int e;
	int t;
	cin>>t;
	while(t--)
	{
		cin>>a;
		cin>>b;
		cin>>c;
		cin>>d;
		cin>>e;
		int rem=0;
		for(int i=0;i<strlen(a);i++)
		{
			int x=rem*10+a[i]-48;
			rem=x%e;
		}
		int k=gcd(e,rem);
		rem=0;
		e=k;
		for(int i=0;i<strlen(b);i++)
		{
			int x=rem*10+b[i]-48;
			rem=x%e;
		}
	    k=gcd(e,rem);
		rem=0;
		e=k;
		for(int i=0;i<strlen(c);i++)
		{
			int x=rem*10+c[i]-48;
			rem=x%e;
		}
		k=gcd(e,rem);
		rem=0;
		e=k;
		for(int i=0;i<strlen(d);i++)
		{
			int x=rem*10+d[i]-48;
			rem=x%e;
		}
		 k=gcd(e,rem);
		 cout<<k<<"\n";
	}
	return 0;
}
