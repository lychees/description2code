#include <iostream>
using namespace std ; 
#include <set>
int main()
{
	string s ; 
	cin >> s ; 
	set < string > st ; 
	int n = s.length() ; 
	for ( int i = 0 ; i< n ; i++)
	{
		for ( int j = i ;j <n ; j++)
		{
			st.insert(s.substr(i , j-i+1)) ; 
 		}
	}
	cout << st.size() << endl; 
	return 0 ; 
}
