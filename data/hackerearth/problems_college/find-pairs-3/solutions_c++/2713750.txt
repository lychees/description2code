#include <iostream>
using namespace std;

int main()
{
    int T,n,count;
    cin>>T;
		while(T>0){
			cin>>n;
			int *a = new int[10^6];
			for(int i=0; i<n; i++)
				cin>>a[i];
			 count = 0;
			for(int i=0; i<n; i++)
				for(int k=i;k<n;k++){
					if(a[i] == a[k]){
						//System.out.println(a[i] + "  " + a[k]);
						count++;
					}
				}
			cout<<(count)<<endl;
			 T--;
		}
    return 0;
}
