#include <cstdio>
#include <cstdlib>
#include <iostream>
#define mod 1000000007

using namespace std;

long long exp(long long x, long long n){

	if(n == 0)
		return 1;
	if(n == 1)
		return x;
	else if(n%2 == 0)
		return exp((x*x)%mod, n/2);
	else
		return (x*exp((x*x)%mod, (n-1)/2))%mod;
}

int main(){

	int t;
	scanf("%d", &t);

	while(t--){

		long long n, ans = 1;
		scanf("%lld", &n);

		if(n%2 == 0){

			ans = (3*exp(2, n-1))%mod;
			ans = (ans + 1)%mod;
		}
		else{
			
			ans = (3*exp(2,n-1))%mod;
			ans = (ans - 1)%mod;
		}

		printf("%lld\n", ans);
	}

	return 0;
}
