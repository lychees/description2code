def gcd(a, b):
    if a > b:
        a, b = b, a
    while b % a:
        tmp = b % a
        b = a
        a = tmp
    return a


t = int(raw_input())
while t:
    m = map(int, raw_input().split())
    if gcd(m[0], m[1]) == 1:
        print 'Is a Co-Prime'
    else:
        print 'Not a Co-Prime'
    t -= 1
