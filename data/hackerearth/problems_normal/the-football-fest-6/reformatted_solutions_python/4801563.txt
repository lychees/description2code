for _ in range(input()):
    pos = []
    n, ID = map(int, raw_input().split())
    pos.append(ID)
    for i in range(n):
        p = map(str, raw_input().split())
        if p[0] == "P":
            pos.append(p[1])
        elif p[0] == "B":
            pos.append(pos[-2])
    print "Player "+str(pos[-1])
