#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n,k;
	cin>>n>>k;
	vector<pair<int,int>> myvector;
	int x,y;
	for(int i=0;i<k;i++)
	{
		cin>>x>>y;
		make_pair(x,y);
		myvector.push_back(make_pair(x,y));
	}
	vector<pair<int,int>>:: iterator it;
	long long count=0;
	for(it=myvector.begin();it!=myvector.end();it++)
	{
		//cout<<(*it).first<<" "<<(*it).second<<endl;
		int first = abs((*it).first-1) + abs((*it).second-1);
		int second = abs((*it).first-1) + abs((*it).second-n);
		int third = abs((*it).first-n) + abs((*it).second-1);
		int fourth = abs((*it).first-n) + abs((*it).second-n);
		//cout<<first<<second<<third<<fourth<<endl;
		count += min(first,min(second,min(third,fourth)));
	}
	cout<<count<<endl;
	return 0;
}
