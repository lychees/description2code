#include <iostream>
#include<stdio.h>
#include<vector>
using namespace std;

int main()
{
    long int temp,i,n,k,c=0,a[1000001],d[2000001]={0};
    scanf("%ld%ld",&n,&k);
    for(i=0;i<n;i++)
    {
    	scanf("%ld",&a[i]);
    }
    for(i=0;i<n;i++)
    {
    	temp=k-a[i];
    	if(temp>=0&&d[temp]==1)
    	{
    		c=1;
    		break;
    	}
    	else if(temp>0)
    	{
    		d[a[i]]=1;
    	}
    }
    if(c==1)
    printf("YES");
    else
    printf("NO");
    return 0;
}
