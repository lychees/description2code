def solve(tc):
    strx = '0123456789'
    a = []
    b = []
    for i in tc:
        if i in strx:
            a.append(i)
        else:
            b.append(i)
    a.reverse()
    b.reverse()
    curr = a[0]
    for i in range(1, len(a)):
        curr = curr + b[i-1] + a[i]
        curr = str(eval(curr))
    return (curr)


for i in range(int(input())):
    print(solve(raw_input()))
