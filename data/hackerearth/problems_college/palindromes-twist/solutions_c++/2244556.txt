#include<iostream>
#include<cmath>
using namespace std;
int main(){
	int t ;
	cin>>t;
	int a,b ;
	string s ;

	while(t--){
	 cin>>s;
	 int count=0 ;
	 cin>>a>>b ;
	 int length = s.size()-1 ;
	 int l = 0, r = s.size()-1 ;
	 while(l <= length/2){
	 	if(s[l]==s[r]){
	 		if(s[l]=='/'){
	 			count += 2*min(a,b) ;
	 		}
	 		l ++ ;
	 		r -- ;
	 	}
	 	else{
	 		double x = s[l] - s[r];
	 		if(abs(x) ==1){
	 			count = -1 ;
	 			break;
	 		}
	 		else{
	 			if(s[l] == 'a' || s[r]=='a'){
	 				count += a ;
	 			}
	 			else{
	 				count += b ;

	 			}

	 		}
	 			l++ ;
	 			r-- ;
	 	}
	 }
	 cout<<count<<endl;
	}
}
