t = int(raw_input())
while t>0:
	str = raw_input().split(' ')
	bt1 = int(str[0])
	bt2 = int(str[1])
	mt1 = int(str[2])
	mt2 = int(str[3])
	ans = ""
	if bt2<bt1:
		temp = bt2
		bt2 = bt1
		bt1 = temp
	if mt2<mt1:
		temp = mt2
		mt2 = mt1
		mt1 = temp
	if bt2<=mt1-1 or mt2<=bt1-1:
		ans = "Line"
	elif bt2==mt1 or mt2==bt1 or bt1==bt2 or mt1==mt2:
		ans = "Point"
	else:
		ans = "Nothing"
	t-=1
	print ans
