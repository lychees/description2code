kases = int(raw_input())
for kase in range(kases):
    n = int(raw_input())
    output = 'YES'
    swords = []
    animals = []

    for i in range(n):
        swords.append(int(raw_input()))

    for i in range(n):
        animals.append(int(raw_input()))

    swords.sort()
    animals.sort()

    swd_len = len(swords)
    anl_len = len(animals)

    for i in range(n):
        if i < swd_len and i < anl_len and int(swords[i]) <= int(animals[i]):
            output = 'NO'
            break

    print output
