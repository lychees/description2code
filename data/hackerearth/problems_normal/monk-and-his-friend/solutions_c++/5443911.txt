#include<bits/stdc++.h>
using namespace std;
int main()
{
    unsigned int t,k;
    unsigned long long int m,p,r;
    scanf("%u",&t);
    while(t--)
    {   k=0;
        scanf("%llu %llu",&m,&p);
        r=m^p;
        while(r)
        {
            if(r%2) k++;
            r=r/2;
        }
        printf("%u\n",k);
    }
    return 0;
}
