#include <iostream>
using namespace std;

int main()
{
	int n,k=0;
	cin>>n;
	int a[n],b[n];
	for(int i=0;i<n;i++)
	cin>>a[i];
	
	for(int i=0;i<n;i++)
	   {
	   	int c=0;
	   	for(int j=0;j<n;j++)
	   	   {
	   	   	if(a[i]%a[j]==0)
	   	   	  c++;
	   	   	 if (c==2)
	   	   	 break;
	   	   }
	   	if(c==1)
	   	b[k++]=a[i];
	 }
	 for(int i=0;i<k;i++)
	 cout<<b[i]<<" ";
	 
	 return 0;
}
