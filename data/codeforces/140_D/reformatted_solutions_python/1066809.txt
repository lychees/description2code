noProb = int(raw_input())
time = sorted(map(int, raw_input().split(' ')))

timeUsed = 10
penalty = 0
probs = 0

for i in time:
    timeUsed += i
    if timeUsed > 720:
        break
    elif timeUsed > 360:
        penalty += timeUsed-360
        probs += 1
    else:
        probs += 1
print("%d %d") % (probs, penalty)
