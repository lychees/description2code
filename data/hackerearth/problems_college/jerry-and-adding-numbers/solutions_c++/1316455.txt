#include <bits/stdc++.h>
using namespace std ;
 
#define sc(x) scanf("%d",&x)
#define pr(x) printf("%d ",x) 
#define prn(x) printf("%d\n",x)
#define MAXN 100005
#define LL long long int 
 
LL P[MAXN],A[MAXN];
int N,T,K,freq,ans;
 
 
bool check(int mid,int idx){
 
	LL X = P[mid]-P[idx+1] ;
	LL Y = (1LL * A[idx] * (idx-mid+1)) ;
	if(Y-X <= K)
		return 1 ;
	return 0 ;
}
void search(int idx){
 
	int low,high,mid ;
	low = 1 ;
	high = idx ;
	while(low<=high){
		mid = (low+high)/2 ;
		bool f = check(mid,idx) ;
		if(f && (mid == 1 || check(mid-1,idx) == 0)){
			break ;
		}else if(f){
			high = mid-1 ;
		}else{
			low = mid+1 ;
		}
	}
	if((idx-mid+1) > freq){
		freq = idx-mid+1 ;
		ans = A[idx] ; 
	}	
}
 
int main(){
 
	sc(T) ;
	while(T--){
		sc(N) ;
		sc(K) ;
		freq = ans = 0 ;
		for(int i=1;i<=N;i++){
			scanf("%lld",&A[i]) ;
		}
		sort(A+1,A+1+N) ;
		P[N+1] = 0 ;
		for(int i=N;i>=1;i--){
			P[i] = A[i] ;
			P[i] += P[i+1] ;
		}
 
		for(int i=1;i<=N;i++){
			search(i) ;	
		}
		printf("%d %d\n",freq,ans) ;
	}
	return 0 ;
}
