#include <iostream>
using namespace std;

int main()
{
    int coin[15],n=0,q=0,i,p=0,a=0,b=0,num=0;
    cin>>n>>q;
    for(i=0;i<15;++i)
     coin[i]=0;
    for(i=0;i<10;i++)
    {
     p=2;
     cin>>p;
     if(p==0)
     { cin>>a>>b;
     	while(a<=b)
       {
       	if(coin[a]==0) 
       	 coin[a]=1;
       	else if(coin[a]==1)
       	 coin[a]=0;
       	++a;
       }
     }
      else if(p==1)
      { cin>>a>>b;
      	while(a<=b)
        {
         if(coin[a]==1) num++;
         a++;
        }
        cout<<num<<"\n";
        num=0;
      }
      
    }
    
    return 0;
}
