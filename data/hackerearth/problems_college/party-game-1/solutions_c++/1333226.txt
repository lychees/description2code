#include<bits/stdc++.h>
using namespace std;
int main()
{
long long int t,c,j,n,i;
cin>>t;
while(t--)
{
long long int f=0,f1=0;
long long int sum=0,sum1=0;
cin>>c>>j>>n;
long long int a[n+2];
for(i=0;i<n;i++){
cin>>a[i];
}
sort(a,a+n);
for(i=0;i<n;i++)
{
sum=sum+a[i];
sum1=sum1+a[i];
if(sum<=c)
f++;
if(sum1<=j)
f1++;
}
if(f>f1)
cout<<"Chef Won"<<endl;
else if(f1>f)
cout<<"Jhool Won"<<endl;
else if(f==f1)
cout<<"Tie"<<endl;
}
}
