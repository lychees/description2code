using namespace std;

struct node
{
	char value;
	node *edges[26];
};

node *newnode(char key)
{
	node *temp=(node *)malloc(sizeof(node));
	temp->value=key;
	for(int i=0;i<26;i++)
	{
		temp->edges[i]=NULL;
	}
	return temp;
}

void insert(node *root,char *s)
{
	int level;
	int index;
	int length=strlen(s);
	node *ptr=root;

	for(level=0;level<length;level++)
	{
		index=s[level]-'a';

		if(ptr->edges[index]==NULL)
		{
			ptr->edges[index]=newnode(s[level]);
		}

		ptr=ptr->edges[index];
	}
}

int ans;

int countt(node *root)
{
	int i;
	for(i=0;i<26;i++)
	{
		if(root->edges[i]!=NULL)
		{
			ans++;
			countt(root->edges[i]);
		}
	}

}

int main()
{
	int i,j,t;
	int n;
	char s[35];
	sfd(n);
	node *root=newnode(' ');

	ffor(i,0,n)
	{
		sfs(s);
		insert(root,s);
	}

	ans=0;
	countt(root);
	printf("%d\n",ans+1);
	return 0;
}
