N, M = map(int, raw_input().split())
party = [[] for _ in range(N)]
for i in range(M):
    p, q = map(int, raw_input().split())
    party[p-1].append(q-1)
mi = N+1
for j in range(N):
    comming = [0]*N
    check = [j]
    comming[j] = 1
    inv = 1
    while check:
        c = check.pop()
        for e in party[c]:
            if not comming[e]:
                comming[e] = 1
                check.append(e)
                inv += 1
    mi = min(mi, inv)

print mi
