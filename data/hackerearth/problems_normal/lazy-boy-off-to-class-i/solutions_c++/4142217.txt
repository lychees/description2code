#include<iostream>
#include<vector>
#include<cmath>
#include<cstdlib>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long int cls,x,y,total=0;
		vector<long long> v;
		for(int i=0;i<7;i++)
		{
			cin>>cls;
			total=total+cls;
			for(int j=0;j<cls;j++)
			{
				cin>>x>>y;
				v.push_back(abs(y-x));
			}
		}
		sort(v.begin(),v.end());
		int a=ceil((total*3)/4.0);
		long long int ans=24*60*60*7;
		for(int i=0;i<a;i++)
		ans=ans-v[i];
		cout<<ans<<endl;
	}
}
