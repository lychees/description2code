#include <iostream>
#include<cmath>
#define LL unsigned long
using namespace std;
int main() {
	int t;
	cin>>t;
	while(t--)
	{
		LL x,y,z,i,ans=0;
		cin>>z>>x>>y;
		for(i=1;i<=sqrt(z);i++)
		{
			if(z%i==0)
			{
				if(z/i<=x&&i<=y) ans++;
				if(z/i<=y&&i<=x) ans++;
				if(i*i==z) ans--;
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
