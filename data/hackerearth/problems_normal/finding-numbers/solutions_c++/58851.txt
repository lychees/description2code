using namespace std;

int main(){
	long long int x,y,z,i,m,n,T,ans;
	cin >> T;
	while(T--){
		cin >> z >> m >> n;
		ans = 0;
		for(i=1;i*i<=z;i++){
			if(z%i == 0){
				x = i;
				y = z/i;
				if(x <= m && y <= n)
					ans++;
				if(x <= n && y <= m && x != y)
					ans++;
			}
		}
		cout << ans << endl;
	}
	return 0;
}
