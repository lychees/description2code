#include<cstdio>
using namespace std;
int gcd(int x,int y)
{
	if(x%y==0)
		return y;
	else
		return gcd(y,x%y);
} 
int main()
{
	int x;
	scanf("%d",&x);
	int n;
	scanf("%d",&n);
	int cnt=0;
	for(int i=0;i<n;i++)
	{
		int y;
		scanf("%d",&y);
		int gc=gcd(x,y);
		if(gc!=1)
			cnt++;
	}
	printf("%d\n",cnt);
	return 0;
}
