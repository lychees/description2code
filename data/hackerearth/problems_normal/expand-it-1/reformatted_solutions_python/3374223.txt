import re

S = str(raw_input())

pairs = re.findall(r'([a-z])([0-9]+)', S)

alphabets_count = [0] * 26
for pair in pairs:
    alphabets_count[ord(pair[0])-97] += int(pair[1])

for Q_I in range(int(raw_input())):
    K = int(raw_input())
    res = -1
    for i in range(26):
        if K <= alphabets_count[i]:
            res = chr(i+97)
            break
        else:
            K -= alphabets_count[i]
    print res
