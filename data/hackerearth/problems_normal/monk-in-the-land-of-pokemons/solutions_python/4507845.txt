for i in xrange(int(raw_input().strip())):
  d = {}
  count = 0
  for j in xrange(int(raw_input().strip())):
     f, p = map(int, raw_input().split(' '))
     if p == f:
       continue
     d[f] = d.get(f,0)+1
     rf = d.get(p, 0)
     if rf == 0:
        count += 1
     else:
        d[p] = rf -1
     
  print count
