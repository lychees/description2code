#include <iostream>
using namespace std;
int main()
{
    int l=0,u=0,temp;
    cin>>l>>u;
    if(l>u)
    {
        temp=l;
        l=u;
        u=temp;
    }
    for(int i=l+1;i<u;i++)
    {
        if((i%3!=0 && i%5!=0) || (i%3==0 && i%5==0))
    	{
    		  if(i!=u-1)
    		  {
    		    cout<<i<<",";
    		  }
    		  else
    		  {
    		  	cout<<i;
    		  }
    	}
    }
    return 0;
}
