#include <iostream>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long n,sum=0;
    	scanf("%ld",&n);
    	do
    	{
    		sum=0;
    		while(n!=0)
    		{
    			sum+=n%10;
    			n/=10;
    		}
    		n=sum;
    	}
    	while((sum/10)!=0);
    	printf("%d\n",sum);
    }
    return 0;
}
