#include<bits/stdc++.h>

using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	int n;
	cin>>n;
	int arr[n+1];
	for(int i=0;i<n;i++)
		cin>>arr[i];
	int l,b;
	float cost=0;
	cin>>l>>b;
	for(int i=0;i<n;i++)
	{
		int digit=0;
		int temp=arr[i];
		while(temp)
		{
			digit+=temp%10;
			temp/=10;
		}
		if(arr[i]>=1 && arr[i]<=10)
			cost+=floor(float(l*b*arr[i]*digit*75)/100);
		else if(arr[i]>=11 && arr[i]<=20)
			cost+=floor(float(l*b*arr[i]*digit*70)/100);
		else
			cost+=floor(float(l*b*arr[i]*digit*80)/100);
	}
	cout<<cost<<endl;
	return 0;
}
