#include<iostream>
#include<cmath>
#include<cstring>
#include<string>
#include<algorithm>
#include<vector>
#include<map>
#include<bitset>
#include<stack>
#include<stdio.h>
#include<queue>
using namespace std;
int main(){
	int t , freq[30], max ;
	int n;
	string str;
	cin >> t;
	while(t--){
		max = 0;
		cin >> n >> str;
		memset(freq, 0, sizeof freq);
		for(int i = 0; i < n; i++){
			freq[str.at(i) - 'a'] += 1;
		}
		for(int i = 0; i < 27; i++){
			if(freq[i] > max)max = freq[i];
		}
		if(max%2 == 0)cout << "Yes" << endl;
		else cout << "No" << endl;
	}
	return 0;
}
