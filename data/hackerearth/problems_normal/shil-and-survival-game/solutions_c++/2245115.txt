#define MAX 100002
int maxt[MAX];
int maxt2[MAX];
int n;
int b[MAX];

{

    for(int i=0;i<n;i++){
    	int a;
    	scanf("%d",&a);
    	maxt[i]=maxt2[i]=b[i]=a;
    }
    for(int i=1;i<n;i++){
    	maxt[i]=max(maxt[i],maxt[i-1]);
    }
    for(int i=n-1;i>=0;i--){
    	maxt2[i]=max(maxt2[i],maxt2[i+1]);
    }
    for(int i=0;i<n;i++){
    	if(i==0){
    		cout<<1;
    		continue;
    	}
    
    	if(i+1==n){
    		printf(" %d\n",i+1);
    		continue;

    		continue;
    	}
    		if(i){
    		printf(" ");
    	}
    	cout<<i+1;
    }
    return 0;
}
