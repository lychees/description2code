mod=10**9+7
def gcd(a,b):
	while b!=0:
		c=b
		b=a%b
		a=c
	return a
n=int(raw_input())
l=map(int,raw_input().split(' '))
if n==1:
	print (l[0]**l[0])%mod
	exit()
x=l[0]*l[1]
y=gcd(l[0],l[1])
for i in range(2,len(l)):
	y=gcd(y,l[i])
	x*=l[i]
ans=1
while y!=0:
	if y%2==1:
		ans=ans*x
		ans=ans%mod
	x=x*x
	ans=ans%mod
	y/=2
print ans
