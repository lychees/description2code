#include <iostream>
#include <cmath>

using namespace std;

bool isPrime(int n){
	//int sqrtn=(int)sqrt(n);
	for(int i=2;i<=n/2;i++){
		if(n%i==0)
			return false;
	}
	return true;
}

int main(){
	int num;
	cin>>num;
	long int count=0;
	long int prev=1;
	if(num==0 || num==1)
		cout<<count<<endl;
	else{
		for(int i=2;i<=num;i++){
			if(isPrime(i)){
				count+=prev;
				prev=prev*2;
			}
			else{
				count+=prev/2;
			}
		}
		cout<<count<<endl;
	}
	return 0;
}
