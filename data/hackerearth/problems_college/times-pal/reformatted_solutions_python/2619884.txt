def is_palindrome(a):
    if a == a[::-1]:
        return True
    else:
        return False


noc = input()
while noc:
    noc -= 1
    a = raw_input()
    b = raw_input()
    r = 0
    for i in range(len(a)+1):
        t = a[:i]+b+a[i:]

        if is_palindrome(t):
            r += 1
    print r
