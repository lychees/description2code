#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,n,i,j,ans;
	long long a[100001],d;
	scanf("%d",&t);
	while (t --) {
		scanf("%d %lld",&n,&d);
		for (i = 0; i < n; i++) {
			scanf("%d",&a[i]);
		}
		ans = 0;
		for (i = 0; i < n; i++) {
			for (j = i+1; j < n; j++) {
				if (abs(a[i]-a[j]) <= d && a[i] != 0 && a[j] != 0) {
					a[i] = 0;
					a[j] = 0;
					ans++;
					break;
				}
			}
		}
		printf("%d\n",ans);
	}
	
	return 0;
}
