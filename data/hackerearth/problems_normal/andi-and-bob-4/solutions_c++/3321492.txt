#include<iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		long long int arr[n+1];
		arr[0]=0;
		for(int i=1;i<=n;i++)
		 cin>>arr[i];
		
		for(int i=1;i<=n;i++)
		 arr[i]+=arr[i-1];
		 
		int no_odd=0,no_even=1;
		long long int total_odd_sum=0;
		for(int i=1;i<=n;i++)
		{
			if(arr[i]%2){
				no_odd++;
				total_odd_sum+=no_even;
			}else{
				no_even++;
				total_odd_sum+=no_odd;
			}
		}  
		long long int total = 1LL * n * (n+1)/2;
		long long int total_even_sum = total-total_odd_sum;
		total_odd_sum>=total_even_sum?cout<<"BOB"<<endl:cout<<"ANDI"<<endl; 
	}
}
