#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tc; cin>>tc;
    while(tc--)
    {
    	int n,xx;
    	cin>>n>>xx;
    	string str[n];
    	map<string,int> m;
    	for(int i=0;i<n;i++)
    	{
    		cin>>str[i];
    		m[str[i]]++;
    	}
    	bool flag=true;
    	map<string,int>::iterator i;
    	for(i=m.begin();i!=m.end();i++)
    	{
    		if(i->second>1)
    		{
    			flag=false;
    		}
    		
    	}
    	if(flag) cout<<"Yes"<<endl;
    	else cout<<"No"<<endl;
    }
}
