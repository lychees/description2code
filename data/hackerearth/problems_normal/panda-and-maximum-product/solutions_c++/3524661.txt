#include<cstdio>

long long max_value(long long a, long long b){
    if(a > b) return a;
    return b;
}

int main(){
    int n;
    scanf("%d", &n);
    long long maxx1 = -2000000000;
    long long maxx2 = -2000000000;
    long long minn1 = 2000000000;
    long long minn2 = 2000000000;
    int a;
    long long ans = 0;
    for(int i=0;i<n;i++){
        scanf("%d", &a);
        if(maxx1 < a){
            maxx2 = maxx1;
            maxx1 = a;
        }
        else if(maxx2 < a){
            maxx2 = a;
        }
        if(minn1 > a){
            minn2 = minn1;
            minn1 = a;
        }
        else if(minn2 > a){
            minn2 = a;
        }
    }
    printf("%lld\n", max_value(maxx1*maxx2, minn1*minn2));
}
