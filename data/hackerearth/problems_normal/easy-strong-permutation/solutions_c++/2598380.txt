#include <iostream>
#include<algorithm>
using namespace std;
long long arr[1000000];
long long brr[1000000];
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>arr[i];
	}
	sort(arr,arr+n);
	int l=0,r=n-1,pos=0;
	while(l<=r)
	{
		brr[pos]=arr[r];
		r--;
		pos++;
		if(l<r)
		{
			brr[pos]=arr[l];
			l++,pos++;
		}
	}
	long long ans=0;
	for(int i=0;i<n;i++)
	{
		int x=i;
		int y=(i+1)%n;
		ans+=abs(brr[x]-brr[y]);
//		cout<<x<<"-"<<y<<endl;
	}
	cout<<ans;
    return 0;
}
