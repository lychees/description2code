#include <bits/stdc++.h>
#define s(T) scanf("%d",&T)
#define us(T) scanf("%u",&T)
#define uls(T) scanf("%lu",&T)
#define SIZE 65535
#define mx 1005
#define mod 1000000009
using namespace std;
typedef long long ll;

int main(){
   int T,p,x,y,u,v,i,j,k;
   s(T);
   while(T--){
    s(p);
    s(x);
    s(y);
    s(u);
    s(v);
    if(p==1){
        if(x!=u && y!=v)
            printf("Terrorists Win !\n");
        else
            printf("Counter-Terrorists Win !\n");
    }
    else{
        bool flag=false;
        i=abs(x-u);
        j=abs(y-v);
        if(((i&1)&&(j&1))||(!(i&1)&&!(j&1)))
            flag=true;
        if(flag){
            if(i==j)
                k=1;
            else
                k=2;
        }
        if(flag && k<p)
            printf("Counter-Terrorists Win !\n");
        else
            printf("Terrorists Win !\n");
    }
   }
return 0;
}
