import heapq
n=input()
heap=[]
listi=[int(i) for i in raw_input().split()]
for i in xrange(n):
	heapq.heappush(heap,-listi[i])
	if len(heap)<3:
		print "-1"
		continue
	else:
		x=-(heapq.heappop(heap))
		y=-(heapq.heappop(heap))
		z=-(heapq.heappop(heap))
		heapq.heappush(heap,-x)
		heapq.heappush(heap,-y)
		heapq.heappush(heap,-z)
		x=x*y*z
		print x
