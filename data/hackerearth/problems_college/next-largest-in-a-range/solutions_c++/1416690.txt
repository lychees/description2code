#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n,L,R,q,i,j,k,ans;
	scanf("%d %d %d %d",&n,&L,&R,&q);
	int a[n],f;
	int b[n];
	for (i = 0; i < n; i++) {
		scanf("%d",&a[i]);
	}
	k = 0;
	for (i = L; i <= R; i++) {
		b[k++] = a[i];
	}
	sort(b,b+k);
	k = 0;
	for (i = L; i <= R; i++) {
		a[i] = b[k++];
	}
	/*sort(a+L,a+R);
	for (i = 0; i < n; i++) {
		printf("%d ",a[i]);
	}*/
	while (q --) {
		int x;
		scanf("%d",&x);
		ans = a[L];
		f = 0;
		for (i = L; i <= R; i++) {
			if (x < a[i]) {
				f = 1;
				ans = a[i];
				break;
			}
		}
		if (f == 1) {
			printf("%d\n",ans);
		} else {
			printf("-1\n");
		}
	}
	
	return 0;
}
