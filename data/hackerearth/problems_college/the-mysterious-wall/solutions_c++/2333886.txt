#include<bits/stdc++.h>
using namespace std;

#define PB push_back
#define MP make_pair
#define F first
#define S second
//#define LET(it,a) __typeof(a) it(a)
#define LET(it,container) __typeof(container.begin()) it(container.begin())
#define ITER(it,container) for(typeof(container.begin()) it=container.begin();it!=container.end();it++)
#define LL long long
const int MOD = 1e9 + 7;
const int INF = INT_MAX;

char s1[100005],s2[100005];
int pre[100005][26], tmp[26];

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s %s",s1,s2);
		if(strlen(s2) > strlen(s1))
		{
			puts("NO");
			continue;
		}
		for(int j=0;j<26;j++)
			pre[0][j] = tmp[j] = 0;
		pre[0][s1[0]-'a'] = 1;
		for(int i=1;i<strlen(s1);i++)
		{
			for(int j=0;j<26;j++)
				pre[i][j] = pre[i-1][j];
			pre[i][s1[i]-'a']++;
		}
		for(int i=0;i<strlen(s2);i++)
			tmp[s2[i]-'a']++;
		int w = strlen(s2);
		int f, here;
		for(int i=0;i<=strlen(s1)-w;i++)
		{
			f = 1;
			for(int j=0;j<26;j++)
			{
				if(i == 0) here = pre[i+w-1][j];
				else here = pre[i+w-1][j] - pre[i-1][j];
				if(here != tmp[j])
				{
					f = 0;
					break;
				}
			}
			if(f == 1)
			{
				puts("YES");
				break;
			}
		}	
		if(!f) puts("NO");
	}

	return 0;
}
