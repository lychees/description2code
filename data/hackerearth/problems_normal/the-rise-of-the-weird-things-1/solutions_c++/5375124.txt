#include<bits/stdc++.h>
using namespace std;

int main()
{
    std::ios::sync_with_stdio(0);
    int n;
    cin>>n;
    int A[n+1];
    vector<int>v1;
    for(int i=0;i<n;i++)
    cin>>A[i];
    sort(A,A+n);
    int sum=0;
    for(int i=0;i<n;i++)
    {
    	if(A[i]%2==0)
    	{
    		v1.push_back(A[i]);
    		sum+=A[i];
    	}
    }
    v1.push_back(sum);
    sum=0;
    for(int i=0;i<n;i++)
    {
    	if(A[i]%2==0)
    	{
    		continue ;
    	}
    	else
    	{
    		v1.push_back(A[i]);
    		sum+=A[i];
    	}
    }
    v1.push_back(sum);
    for(auto i=0;i<v1.size();i++)
    cout<<v1[i]<< " ";
    cout<<endl;
    return 0;
}
