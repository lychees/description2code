bool find(int arr[100],int i,int length,int val)
{
	for(int j = 0; j <= i-1; j++)
		if(val % arr[j] == 0) return false;
	for(int j = i+1; j < length; j++)
		if( val % arr[j] == 0) return false;
	return true;
}
