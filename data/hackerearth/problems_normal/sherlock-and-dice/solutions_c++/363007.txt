#include <iostream>
#include <cstdlib>
#include <cstdio>

using namespace std;

#define MOD 1000000007

#define LL long long int

LL comb[104][104];

long double dp[52][2502];

void ncr(LL n){
    LL i, j;

    for(i = 0; i <= n; i++){
        for(j = 0; j <= n; j++){
            comb[i][j] = 0;
        }
    }

    for(j = 0; j <= n; j++){
        comb[0][j] = 0;
    }

    for(i = 0; i <= n; i++){
        comb[i][0] = 1;
    }

    for(i = 1; i <= n; i++){
        for(j = 1; j <= i; j++){
            comb[i][j] = (comb[i - 1][j] + comb[i - 1][j - 1])%MOD;
            comb[i][j] %= MOD;
        }
    }
    return;
}

inline void Solve(){
    int n, m, k, i, j, index;
    cin>>m>>n>>k;
    if(k > 2500){
        cout<<"0.000 0"<<endl;
        return;
    }

    for(i = 0; i <= m; i++){
        for(j = 0; j <= k; j++){
            dp[i][j] = 0;
        }
    }

    for(i = 1; i <= n; i++){
        dp[1][i] = 1;
    }

    for(i = 1; i <= m; i++){
        for(j = i; j <= k; j++){
            for(index = 1; index <= n; index++){
                if(j < index){
                    continue;
                }
                dp[i][j] += dp[i - 1][j - index];
            }
        }
    }

    long double ans =  dp[m][k];
    for(i = 0; i < m; i++){
        ans = (ans / (n + 0.0));
    }
    //cout<<ans<<endl;
  	if(ans == 0){
  		cout<<"0.000 0"<<endl;
  		return;
  	}
  	
  	if(ans == 1){
  		cout<<"1.000 0"<<endl;
  		return;
  	}
  	
    int count = 0;
    for(i = 0; ; i++){
        if(ans > 1){
            break;
        }
        ans = ans * (10.0);
        count++;
    }
    printf("%0.3Lf %d\n", ans, count);
    return;
}

int main(){
    int t;
    cin>>t;
    while(t--){
        Solve();
    }
}
