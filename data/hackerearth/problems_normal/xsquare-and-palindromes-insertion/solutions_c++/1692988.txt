#define uli int
uli char_freq[26];

using namespace std;
void count_char_freq(string);
void count_to_balance(int);

int main()
{
    uli t;
    cin>>t;
    string inp_str;
    while(t--)
    {
        cin>>inp_str;
        count_char_freq(inp_str);
        count_to_balance(inp_str.length());
    }
    return 0;
}

void count_char_freq(string str)
{
    uli i;
    memset(char_freq,0,sizeof(char_freq));
    for(i=0;i<str.length();i++)
    {
    char_freq[str[i]-'a']++;
    }
}

void count_to_balance(int x)
{
    int req=0;
    int i;
    switch(x%2)
    {
    //Odd length
    case 1:
        for(i=0;i<26;i++)
            if(char_freq[i]%2==1)
                req++;
    if(req>=1) req--;
    break;

    //Even length
    case 0:
        for(i=0;i<26;i++)
            if(char_freq[i]%2==1)
                req++;
    if(req>1)   req--;
    break;
    default:
        cout<<"Should not be here!!";
        break;
    }
    cout<<req<<"\n";
}
