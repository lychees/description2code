#include <iostream>
#include <string.h>
using namespace std;
     
int main()
{	
	int t;
    cin>>t;
    for(int j=0;j<t;j++)
    { 
    	string a;
    	int c=0;
    	string b;
    	cin>>a;
    	cin>>b;
    	int l=a.length();
    	for(int i=0;i<=l;i++)
    	{
    		string	s=a;
    		s.insert(i,b);
    		if (s == string(s.rbegin(), s.rend())) 
    		{
    			c++;
    		}
    	}
    	cout<<c<<"\n";
    }
    return 0;
}
