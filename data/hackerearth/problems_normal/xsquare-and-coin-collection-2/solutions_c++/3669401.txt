#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll arr[1000000];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	ll sum=0,ans=0;
    	//CLEAR(arr);
    	int n,k;
    	cin>>n>>k;
    	for(int i=0;i<n;i++)cin>>arr[i];
    	for(int i=0;i<n;i++)
    	{
    		if(arr[i]>k)sum=0;
    		else
    		{
    			sum+=arr[i];
    			if(sum>ans)ans=sum;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
