#include <iostream>
#include<string.h>
#include<stdio.h>
#include<stdlib.h>
using namespace std;
int arr[1007];
int lis[1007];
string convert(int num)
{
	string ans;
	while(num)
	{
		int temp=num%2;
		//if(temp==1)
		char c =temp+'0';
		ans= c +ans;
		num=num/2;
	}
	return ans;
}
// lis[i] represents length of longest subsequence ending at index i
int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    int max=-1;
    for(int i=0;i<n;i++)
    cin>>arr[i];
    for(int j=0;j<n;j++)
    lis[j]=1;
    //memset(lis,1,sizeof(lis));
    for(int i=1;i<n;i++)
    for(int j=0;j<i;j++)
    {
    	if(arr[j]<arr[i] && lis[i]<lis[j]+1)
    	{
    		lis[i]=lis[j]+1;
    		if(max<lis[i])
    		max=lis[i];
    	}
    }
    if(max==-1)
    cout<<0;
    else
    cout<<convert(max);
    return 0;
}
