#include <iostream>
#include <map>

using namespace std;

int main()
{
	ios_base::sync_with_stdio( false );
	
	int n, i;
	map<string, int> A;
	map<string, int> :: iterator it;
	
	cin >> n;
	
	for( i = 0;i < n;i++ ) {
		string s;
		cin >> s;
		A[ s ]++;
	}
	
	for( it = A.begin();it != A.end();it++ )
		cout << it->first << " " << it->second << endl;
		
	return 0;
}
