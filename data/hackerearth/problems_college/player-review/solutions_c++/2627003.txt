#include<iostream>
using namespace std;
int main()
{
		int n;
		cin>>n;
		int a[n];
		int i,j;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		int max=0;
		for(i=0;i<n;i++)
		{
		int sum=0;
			for(j=i;j<n;j++)
			{
				sum=a[j]+sum;
				if(sum>max)
				{
					max=sum;
				}
			}
		}
		if(max==0)
		{
			cout<<"0"<<"\n";
		}
		else
	   {
	   	cout<<max<<"\n";
	   }
}
