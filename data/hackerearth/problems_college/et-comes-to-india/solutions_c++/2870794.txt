#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
    int t,n;
    int flag;
    ll int k,val;
    scanf("%d",&t);
    while(t--)
    {
        flag=0;
        scanf("%d%lld",&n,&k);
        while(n!=1)
        {
            val=(ll)1<<(n-1);
            if(k==val)
            {
                if(flag)
                    printf("C\n");
                else
                    printf("G\n");
                break;
            }
            else if(k>=(val*2))
            {
                printf("-1\n");
                break;
            }
            else if(k>val)
            {
                k=val*2-k;
                flag=!flag;
            }
            n--;
        }
        if(k>1 && n==1)
            printf("-1\n");
        else if(n==1)
        {
            if(flag)
                printf("C\n");
            else
                printf("G\n");
        }
    }
    return 0;
}
