#include <iostream>
#include<string>
using namespace std;

int main()
{
    string s;
    int i,t;
    cin>>t;
    while(t--)
    {
    cin>>s;
    int cnt=0;
    for(i=0;i<s.length();i++)
    {
    	if(s[i]=='a'||s[i]=='A'||s[i]=='i'||s[i]=='I'||s[i]=='e'||s[i]=='E'||s[i]=='o'||s[i]=='O'||s[i]=='u'||s[i]=='U')
    	{
    		cout<<s[i];
    		cnt++;
    	}
    }
    if(cnt==0)
      cout<<"No";
      cout<<"\n";
    }
    return 0;
}
