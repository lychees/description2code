#include <stdio.h>
using namespace std;

long long int num = 1000000007;

int main(){
	int t;
	scanf("%d",&t);
	for(int i=0;i<t;i++){
		long long int n;
		scanf("%lld",&n);
		long long int table1[n+1],table2[n+1],res;
		table1[1]=0; table1[2]=1; 
		table2[0]=1; table2[1]=2;
		if(n==1){ res = table1[1]; }
		if(n==2){ res = table1[2]; }
		if(n>2){
			for(int i=2;i<=n;i++){
				table2[i] = (table2[i-1]%num + table2[i-2]%num) %num;
			}
			for(int i=3;i<=n;i++){
				table1[i]= (table1[i-1]%num + table1[i-2]%num)%num;
				table1[i]= (table1[i]%num + table2[i-3]%num)%num;
			}
		}
		res = table1[n];
		printf("%lld\n",res);
	}
}
