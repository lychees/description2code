// LUCIFER <3 SLS <3

#include <bits/stdc++.h>

using namespace std;

long long int l[1000111];
long long int r[1000111];

int main()
{
    string s;
    cin >> s;
    long long int i, ans = 0;
    l[0] = (s[0] == '1' ? 1 : 0);
    for (i = 1; i < s.size(); i++)
        l[i] = l[i-1] + (s[i] == '1' ? 1 : 0);
    r[s.size()-1] = (s[s.size()-1] == '1' ? 1 : 0);
    for (i = s.size()-2; i >= 0; i--)
        r[i] = r[i+1] + (s[i] == '1' ? 1 : 0);
    for (i = 0; i < s.size(); i++) {
        if (s[i] == '0')
            ans += l[i]*r[i];
    }
    cout << ans << endl;
}
