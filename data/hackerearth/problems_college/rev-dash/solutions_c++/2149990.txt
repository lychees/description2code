#include<iostream>
#include<algorithm>
using namespace std;
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#define m 1000000007


int main()
{
	int i;
	char a[10000];
	scanf("%s",&a);
	for(i=0;i<strlen(a);i++)

	if(i<(strlen(a)-1))
	cout<<"-";
	}
	i--;
	cout<<"\n";
	for(;i>=0;i--)
	{cout<<a[i];
	if(i!=0)
	cout<<"-";
	}
}
