#include<iostream>
#include<string.h>

using namespace std;

int main(){
	
	char s[1004];
	cin.getline(s, 1005);
	
	int A[3000];
	int max = 0;
	
	int n = strlen(s);
	
	char ace;
	
	for(int i=0;i<n;i++){
		
		A[s[i]]++;
		
		if(A[s[i]]>max){
			
			max = A[s[i]];
			ace = s[i];
		}
		else if (A[s[i]] == max && ace > s[i]) {
			ace = s[i];
		}
	
		
	}
	
	cout << ace << " " << max;
	
}
