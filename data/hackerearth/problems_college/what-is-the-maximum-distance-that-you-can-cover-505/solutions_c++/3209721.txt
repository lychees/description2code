#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int N;
    cin>>N;
    double speed[N];
    double fuel[N];
    int i;
    for(i=0;i<N;i++)
    {
    	cin>>speed[i];
    }
    
    for(i=0;i<N;i++)
    {
    	cin>>fuel[i];
    }
    
    double fuelPerKm[N];
    
    for(i=0;i<N;i++)
    {
    	fuelPerKm[i]=fuel[i]/speed[i];
    }
    
    double totalFuel;
    cin>>totalFuel;
    
    double maxDistance=totalFuel/(*(min_element(fuelPerKm,fuelPerKm+N)));
    
    int answer=maxDistance*1000;
    
    maxDistance = (double)answer/1000;
    
    printf("%.3f\n",maxDistance);
    return 0;
}
