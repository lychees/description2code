#include <iostream>
using namespace std;

int main()
{
    int a[8];
    a[0] = 0;
    a[2] = 1;
    a[3] = 1;
    a[5] = 1;
    a[4] = 2;
    a[6] = 2;
    a[7] = 1;
    int t;
    scanf("%d",&t);
    while (t--)
    {
    	int n;
    	scanf("%d",&n);
    	if (n == 1) {
    		cout << "-1" << endl;
    		continue;
    	}
    	int ans;
    	if (n % 7 != 1)
    	{
    	//	int j = n%7;
    	//	cout << j << endl;
    		ans = (n/7)+a[n%7];
    		cout << ans << endl; 
    		continue;
    	}
    	int k = n/7;
    	k--;
    	ans = k;
    	n = n-(k*7);
    	if (n%5 != 1)
    	{
    		ans += (n/5)+a[n%5];
    		cout << ans << endl;
    		continue;
    	}
    	k = n/5;
    	k--;
    	ans += k;
    	n = n-(k*5);
    	if (n%3 != 1)
    	{
    		ans += (n/3)+a[n%3];
    		cout << ans << endl;
    		continue;
    	}
    	k = n/3;
    	k--;
    	ans += k;
    	n = n-(k*3);
    	if (n%2 != 1)
    	{
    		ans += (n/2)+a[n%2];
    		cout << ans << endl;
    		continue;
    	}
    	cout << "-1" << endl;
    }
    return 0;
}
