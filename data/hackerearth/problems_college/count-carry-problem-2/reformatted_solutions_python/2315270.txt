t = int(raw_input())
while t != 0:
    c = 0
    n = raw_input().split()
    a = n[0][::-1]
    b = n[1][::-1]
    for i, j in map(None, a, b):
        if i is None:
            i = 0
        if j is None:
            j = 0
        if (int(i) + int(j) + c) > 9:
            c = c + 1

    if c == 0:
        print "No carry operation"
    elif c == 1:
        print "1 carry operation"
    else:
        print "%s carry operations" % c

    t = t - 1
