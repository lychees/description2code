#include<bits/stdc++.h>
using namespace std;
#define ll long long 

int main()
{
    int t; cin>>t;
    cin.clear(); cin.ignore();
    while(t--)
    {
        string str; getline(cin, str);
        if(str.length()==0) { cout<<"\n"; continue; }
        int temp=0;
        for(int i=0; i<str.length(); i++)
        {
            temp+=((int)str[i]);
        }
   
        int f=temp/(str.length());
        if(f%2==0)
        {
             cout<<str;
        }
        else
        {
            for(int i=str.length()-1; i>=0; i--)
                cout<<str[i];
        }
        cout<<"\n";
    }
    
    return 0;
}
