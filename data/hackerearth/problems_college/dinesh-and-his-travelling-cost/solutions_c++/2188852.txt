#include<cstdio>
#include<queue>
#define MP make_pair
#define fi first
#define se second
const int N = 1001, inf = 1000000;
using namespace std;
priority_queue<pair<int, int> >q;
vector<pair<int, int> >v[N];
int n, a, b, c, i, odl, start, dp[N];

int main()
{
  scanf("%d", &n);
  
  while(n--)
  {
    scanf("%d%d%d", &a, &b, &c);
    v[a].push_back(MP(b, c));
    v[b].push_back(MP(a, c));    
  }
  
  for(i = 0; i < 501; i++)
  {
    dp[i] = inf;
  }
  
  scanf("%d", &start);
  dp[start] = 0;
  q.push(MP(0, start));
  
  while(!q.empty())
  {
    a = q.top().se;
    odl = q.top().fi * (-1);
    q.pop();
    
    if (dp[a] == odl)
    {
      for(i = 0; i < v[a].size(); i++)
      {
	b = v[a][i].fi;
	c = v[a][i].se;
	if (dp[b] > dp[a] + c)
	{
	  dp[b] = dp[a] + c;
	  q.push(MP(dp[b] * (-1), b));
	}
      }
    }
  }
  
  scanf("%d", &n);
  
  while(n--)
  {
    scanf("%d", &a);
    if (dp[a] != inf)
    {
      printf("%d\n", dp[a]);
    }
    else
    {
      printf("NO PATH\n");
    }
  }  
  
return 0;
}
