#include <stdio.h>
#include <iostream>

#define MAX 100111

using namespace std;

typedef long long unsigned llu;

int main(){
    llu n, q;
    llu a[MAX], b[MAX], l[MAX], r[MAX];

    scanf("%llu %llu", &n, &q);

    for(llu i = 0; i < n; i++){
	scanf("%llu", &a[i]);
    }

    for(llu i = 0; i < n; i++){
	scanf("%llu", &b[i]);
    }

    l[0] = a[0];
    for(llu i = 1; i < n; i++){
	if(i % 2 == 0){
	    l[i] = l[i-1] + a[i];
	}else{
	    l[i] = l[i-1] + b[i];
	}
    }

    r[0] = b[0];
    for(llu i = 1; i < n; i++){
	if(i % 2 == 0){
	    r[i] = r[i-1] + b[i];
	}else{
	    r[i] = r[i-1] + a[i];
	}
    }

    llu op, x, y, ans;
    for(llu i = 0; i < q; i++){
	scanf("%llu %llu %llu", &op, &x, &y);
	x--; y--;

	if(op == 1){
	    if(x % 2 == 1){
		ans = (r[y] - r[x-1]);
	    }else{
		if(x>1)
		    ans = l[y] - l[x-1];
		else
		    ans = l[y];
	    }
	    
	    printf("%llu\n", ans);
	}else{
	    if(x % 2){
		ans = (l[y] - l[x-1]);
	    }else{
		if(x>1)
		    ans = r[y] - r[x-1];
		else
		    ans = r[y];
	    }
	    printf("%llu\n", ans);
	}
    }
    
    
    return 0;
}
