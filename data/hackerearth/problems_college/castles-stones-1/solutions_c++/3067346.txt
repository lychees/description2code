#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
        int n,m;
        cin>>m>>n;
        int ar[n];
        for(int i=0;i<n;){
            cin>>ar[i];
            i++;
            if(ar[i-1]>m){
                i--;
                n--;
            }
        }
        sort(ar,ar+n);
        int ans = 0,done =0;
        for(int i=n-1;i>=0;i--){
            done = 0;
            if(ar[i] == 0){
                continue;
            }
            for(int j=i;j>=0;j--){
                if(ar[j] == 0){
                    continue;
                }
                if(ar[i]+ar[j] <= m && i!=j){
                    ar[i] = 0;
                    ar[j] = 0;
                    ans++;
                    done = 1;
                    break;
                }
            }
            if(done == 0){
                ans++;
            }
        }
        cout<<ans<<"\n";
    }
    return 0;
}
