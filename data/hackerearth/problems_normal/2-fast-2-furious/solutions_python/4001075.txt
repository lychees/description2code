n = int(raw_input())
arr1 = map(int, raw_input().split())
arr2 = map(int, raw_input().split())
maxd1 = -1
maxd2 = -1
for i in range(n-1):
	diff = abs(arr1[i] - arr1[i+1])
	if maxd1 < diff:
		maxd1 = diff
for i in range(n-1):
	diff = abs(arr2[i] - arr2[i+1])
	if maxd2 < diff:
		maxd2 = diff
if maxd1 > maxd2:
	print "Dom"
	print maxd1
elif maxd1 < maxd2:
	print "Brian"
	print maxd2
else:
	print "Tie"
	print maxd1
