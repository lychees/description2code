#include <cstdio>
#include <algorithm>

using namespace std;

int main()
{
	int t;
    scanf("%d", &t);
    while (t--)
    {
    	int a, b, c, s;
    	scanf("%d %d %d %d %d", &a, &b, &c, &s, &s);
    	if (c - s < a) printf("STAY IN LONDON\n"); else
    	printf("%d\n", min(b, c - s));
    }
    return 0;
}
