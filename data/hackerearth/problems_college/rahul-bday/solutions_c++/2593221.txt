#include <bits/stdc++.h>

using namespace std;

int main()
{int t;
cin>>t;
while(t--){
   long long  int b,w,x,y,z,s,s1,s2;
    cin>>b>>w;
    cin>>x>>y>>z;
    s=b*x+w*y;
    s1=((b+w)*(x))+(w*z);
    s2=((b+w)*(y))+(b*z);
    long long int temp;
    temp=min(min(s,s1),s2);
    cout<<temp<<endl;
}

    return 0;
}
