alpha = "abcdefghijklmnopqrstuvwxyz"
for t in range(int(raw_input())):
    N = int(raw_input())
    arr = [""]*N
    for n in range(N):
        arr[n] = str(raw_input())
    c = [0]*26
    for i in range(26):
        for n in range(N):
            if alpha[i] in arr[n]:
                c[i] += 1
    print alpha[c.index(max(c))]
