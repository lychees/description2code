#include <cstdlib>
#define pii pair<int,int>
#define pi 3.1415
using namespace std;
typedef long long ll;
typedef long double ld;
typedef unsigned long ul;

int main(){

	ios_base::sync_with_stdio(false);
	
	int t;
	ld a,b,c,ans;

	cin>>t;

	while(t--){

		cin>>a>>b>>c;

		ans = pi;

		ans *= (a*b*c*a*b*c);
		ans /= (a+b+c);
		ans /= (b+c-a);
		ans /= (a+b-c);
		ans /= (a+c-b);

		printf("%.4Lf\n",ans);

	}

	return 0;

}
