#include<bits/stdc++.h>
using namespace std;
#define maxn 1000009
int box[maxn];
int main()
{
    int n;cin>>n;int m;cin>>m;
    int s[n+1];
    int e[n+1];
    memset(s,0,sizeof(e));
    memset(e,0,sizeof(s));
    memset(box,0,sizeof(box));
    while(m--)
    {
        int a,b;
        cin>>a>>b;
        s[a]++;
        e[b]++;
    }
    for(int i=2;i<=n;i++)
    {
        s[i]+=s[i-1];
        e[i]+=e[i-1];
    }
    for(int i=2;i<=n;i++)
    {
        s[i]=s[i]-e[i-1];
    }
    for(int i=1;i<=n;i++)
        box[s[i]]++;
    for(int i=1;i<=1000005;i++)
        box[i] +=box[i-1];
    int q;cin>>q;
    while(q--)
    {
        int h;cin>>h;
        cout<<n-box[h-1]<<"\n";
    }


}
