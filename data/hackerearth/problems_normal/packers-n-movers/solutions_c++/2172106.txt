#include<bits/stdc++.h>
using namespace std;
long long k,n,arr[100005],sum[100005],dp[1000][1000],i;
long long go(long long i,long long k){
	long long val=100000005,mx=0;
	if(k==0 && i==n) return 0;
	if(i!=n && k==0) return 10000006;
	if(i==n) return 0;
	if(dp[i][k]!=-1) return dp[i][k];
	for(long long j=i+1;j<=n;j++){
		mx=max(sum[j]-sum[i],go(j,k-1));
		val=min(val,mx);
	}
	//cout<<val<<" "<<i<<" "<<k<<endl;
	return dp[i][k]=val;
}
int main(){
	cin>>k>>n;
	memset(dp,-1,sizeof dp);
	sum[0]=0;
	for(i=0;i<n;i++){
		cin>>arr[i];
		sum[i+1]=sum[i]+arr[i];
	}
	printf("%lld",go(0,k));
	return 0;
}
