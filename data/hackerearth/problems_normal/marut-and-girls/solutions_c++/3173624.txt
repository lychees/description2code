    assert(m >= 1 && m <= 100);
    int q[1000001];
    for(i = 1; i <= m; i++) {
        cin >> x;
        assert(x >= 1 && x <= 10000);
        q[x] = 1;
    }

    cin >> n;

    cin >> ws;

    for(i = 0; i < n; i++) {
        int gq[1000001] = {0};
        char ch;
        char str[1000001];
        gets(str);
        j = num = cnt = 0;
        while (str[j] != '\0') {
            ch = str[j];
            if(ch == ' ') {
                gq[num] = 1;
                if(q[num] == 1) {
                    cnt++;
                }
                num = 0;
            } else {
                num = num * 10 + (str[j] - '0');
            }
            j++;
        }
        if(num != 0) {
            gq[num] = 1;
            if(q[num] == 1) {
                cnt++;
            }
        }
        if(cnt == m) {
            ans++;
        }
    }
    cout << ans << endl;

    return 0;
}
