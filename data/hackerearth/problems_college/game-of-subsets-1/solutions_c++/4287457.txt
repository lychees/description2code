#include<bits/stdc++.h>
using namespace std;
#define m  1000000007
long long power( long long  a, long long  b )
{
	long long temp;
    if(b==0)
        return 1;
    if(b==1)
        return a%m;
    temp=power(a,b/2);
    if(b%2==0)
        return ((temp%m)*(temp%m))%m;
    else
        return ((a)*((temp%m)*(temp%m))%m)%m;
}
 
int main ()
{
	long long  t, n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		n=(power(3,n))%m;
		cout<<n<<endl;
	}
	return 0;
}
