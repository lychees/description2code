#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int main() {
    int x,y,flag=0,c=1,i;
    scanf("%d\t%d\n",&x,&y);
    int a[y];
    while(c<=x)
    {
   for(i=0;i<y;i++)
   {
       scanf("%d\t",&a[i]);
   }
   sort(a,a+y);
   if(a[0]>flag)
   flag=a[0];
   c++;
    }
    printf("%d",flag);
	return 0;
}
