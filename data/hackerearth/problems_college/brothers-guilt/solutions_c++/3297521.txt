#include <iostream>
using namespace std;
int lcs[101][101];
int main()
{
    //cout << "Hello World!" << endl;
    string b1,b2,b3;
    int i,j;
    cin>>b1>>b2;
    cin>>b3;
    int l1=b1.size();
    int l2=b2.size();
    for(i=0;i<=l1;i++){
    	lcs[i][0]=0;
    }
    for(i=0;i<=l2;i++){
    	lcs[0][i]=0;
    }
    for(i=1;i<=l1;i++){
    	for(j=1;j<=l2;j++){
    		if(b1[i-1]==b2[j-1]){
    			lcs[i][j]=lcs[i-1][j-1]+1;
    		}
    		else{
    			lcs[i][j]=max(lcs[i-1][j],lcs[i][j-1]);
    		}
    	}
    }
    if(lcs[l1][l2]<b3.size()){
    	cout<<lcs[l1][l2];//cout<<"sdds";
    }
    else{
	    int sublen=lcs[l1][l2];
	    int temp=sublen;
	    char sub[sublen];
	    for(i=l1,j=l2;i>0,j>0;){
	    	if(b1[i-1]==b2[j-1]){
	    		sub[--temp]=b1[i-1];
	    		i--;
	    		j--;
	    	}
	    	else if(lcs[i][j]==lcs[i-1][j]){
	    		i--;
	    	}
	    	else{
	    		j--;
	    	}
	    }//cout<<sub;
	    int c=0;
	    int l3=b3.size();
	    for(i=0;i<sublen;i++){
	    	int k=i;
	    	for(j=0;j<l3;j++){
	    		if(sub[k]==b3[j]){
	    			k++;
	    		}
	    		else{
	    			break;
	    		}
	    	}
	    	if(j==l3){
	    		c++;
	    	}
	  
	    }
	    cout<<sublen-c;
	}
    return 0;
}
