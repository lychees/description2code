#include <iostream>
long long int n[10007];
long long int total[10007];
using namespace std;
int main()
{
    int t;
    cin >> t;
    int N, M, i, j;
    while(t--)
    {
        int mainflag = 1;
        cin >> N;
        for(i=1;i<=N;i++)
            cin >> n[i];
        cin >> M;
        for(i=1;i<=N;i++)
            total[i]=0;
        int o[N+1][M+1], r[N+1][M+1];
        for(j=1;j<=M;j++)
        {
            o[0][j] = 1;
            for(i=1;i<=N;i++)
            {
                cin >> o[i][j];
                total[i] += o[i][j];
            }
        }
        for(j=1;j<=M;j++)
        {
            for(i=1;i<=N;i++)
            {
                cin >> r[i][j];
            }
        }
        int count = M;
        while(count!=0)
        {
            int flag = count;
            for(j=1;j<=M;j++)
            {
                if(o[0][j]==1)
                {
                    int flag2 = 1;
                    for(i=1;i<=N;i++)
                    {
                        if((n[i] + o[i][j]) < r[i][j])
                        {
                            flag2 = 0;
                            break;
                        }
                    }
                    if(flag2!=0)
                    {
                        for(i=1;i<=N;i++)
                        {
                            total[i] -= o[i][j];
                            n[i] += o[i][j];
                        }
                        count--; 
                        
                        o[0][j]=0;
                    }
                }
                //cout << "count=" << count << "\n";
            }
 
            if(flag==count)
            {
                cout << "NO\n";
                mainflag = 0;
                break;
            }
        }
        if(mainflag!=0)
            cout << "YES\n";
    }
    return 0;
}
