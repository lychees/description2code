#include<cstdio>
#include<iostream>
#include<vector>
#define mod 1000000007
#define ll long long
using namespace std;
long long powermod(ll a,ll b,ll MOD)
{
    long long x=1,y=a;
    while(b > 0)
    {
        if(b%2 == 1)
        {
            x=(x*y);
            if(x>MOD) x%=MOD;
        }
        y = (y*y);
        if(y>MOD) y%=MOD;
        b /= 2;
    }
    return x;
}
int main() {
    int T,N;
    cin >> T;
    for(int t =0; t < T; t++) {
    	cin >> N;
    	vector<int> V(N);
    	int vmin =1000, vmax =0;
    	for(int i =0; i < N; i++) {
    		cin >> V[i];
    		vmin =min(vmin,V[i]);
    		vmax =max(vmax,V[i]);}
    	cout << (vmax-vmin)*2 << "\n";}
    return 0;}
