#include <iostream>
using namespace std;

int main()
{
    
    int t;
    cin>>t;
    int val[t];
    for(int i=0;i<t;i++)
    cin>>val[i];
    for(int i=1;i<t;i++)
    val[i]+=val[i-1];
    int low,high,mid;
    int q;
    cin>>q;
    int S;
    while(q--)
    {
    	cin>>S;
    	if(S>val[t-1])
    	{
    		cout<<"-1"<<endl;
    		continue;
    	}
    	low=0;
    	high=t-1;
    	while(low<=high)
    	{
    		mid=(low+high)/2;
    		if(val[mid]>S)
    		{ 
    		    high=mid-1;	
    		}
    		else if(val[mid]<S)
    		{
    			low=mid+1;
    		}
    		else
    		break;
    		
    	}
    	if(val[mid]==S)
    	cout<<mid+1<<endl;
    	else if(val[high]>S && val[high]>val[low])
    	cout<<high+1<<endl;
    	else
    	cout<<low+1<<endl;
    	
    	
    }
    
    
    
    
    
    return 0;
}
