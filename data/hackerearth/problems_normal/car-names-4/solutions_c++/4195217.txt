#include "bits/stdc++.h"
using namespace std;

#define MOD 1000000007
#define F first
#define S second
#define rev(v) reverse(v.begin(),v.end())
#define ll long long int
#define sz(v) (int)v.size()
#define sort1(v) sort(v.begin(),v.end(),less<long int>())
#define sort2(v) sort(v.begin(),v.end(),greater<long int>())
#define sort3(ARR,n) sort(ARR,ARR+n,less<long int>())
#define sort4(ARR,n) sort(ARR,ARR+n,greater<long int>())
#define pb push_back
#define mp make_pair
#define ins insert
#define pii pair
#define mt(v,x) memset(v,x,sizeof(v))
#define all(v) v.begin(),v.end()
#define alll(arr,n) arr,arr+n
#define rep(x,a,n,b) for(x=a;x<=n;x+=b)
#define rept(x,a,n,b) for(x=a;x>=n;x-=b)

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    int t,l,k,x,y;
    set<char>st;
    string s,str;
    
    cin>>t;
    
    while(t--)
    {
    	cin>>s;
    	l=sz(s);
    	st.clear();
    	
    	rep(x,0,l-1,1)
    	st.ins(s[x]);
    	
    	if((l%3 != 0)||(sz(st)<3))
    	{
    		cout<<"Not OK\n";
    		continue;
    	}
    	
    	k=l/3;
    	
    	rep(x,0,l-1,k)
    	{
    		str=s.substr(x,k);
    		st.clear();
    		
    		rep(y,0,str.size()-1,1)
    		st.ins(str[y]);
    		
    		if(st.size() != 1)
    		break;
    	}
    	
    	if(x == l)
    	cout<<"OK\n";
    	
    	else
    	cout<<"Not OK\n";
    }
    
    return 0;
    
    
}
