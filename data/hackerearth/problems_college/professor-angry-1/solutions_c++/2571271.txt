#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n,i,c=0,x,k;
    	cin>>n>>k;
    	
    	for(i=0;i<n;i++)
    	{cin>>x;
    	if(x<=0) c+=1;}
    	if(c>=k) cout<<"NO\n";
    	else cout<<"YES\n";
    }
    return 0;
}
