import sys

import math
import heapq
import itertools as it
import fractions
import re
import bisect
import collections as coll

n, s = map(int, raw_input().split())
ft = [map(int, raw_input().split()) for i in xrange(n)]
ft = sorted(ft, key=lambda x: x[1], reverse=True)

ans = 0
for f, t in ft:
    ans += s - f
    s = f
    if ans < t:
        ans = t
ans += s
print ans
