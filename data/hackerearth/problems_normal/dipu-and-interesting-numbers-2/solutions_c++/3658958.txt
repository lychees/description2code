#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll pSq(ll x)
{
    ll low = 1;
    ll high = 1e9 + 1;
    ll mid;

    if (x == 0)
		return 0;

    while (low <= high) {
		mid = (low + high) / 2;
		ll u = mid * mid;
		ll v = (mid + 1) * (mid + 1);
		if (u <= x) {
			if (v > x) {
				return mid;
			} else {
				low = mid + 1;
			}
		} else {
			high = mid - 1;
		}
    }
}

int main()
{
	ll t, l, r;
	cin >> t;

	while (t--) {
		cin >> l >> r;
		cout << pSq(r) - pSq(l-1) << endl;
	}

	return 0;
}
