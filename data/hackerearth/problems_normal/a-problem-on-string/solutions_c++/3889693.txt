#include <iostream>
using namespace std;

#define ALPHA_SIZE 26

int main()
{
    int n;
    int queryCount, index;
    char inpCh, firstCh, secondCh;
    long long int alphaCount[ALPHA_SIZE] = {0};
    
    cin>>n;
    
    while( n-- )
    {
    	cin>>inpCh;
    	alphaCount[ inpCh - 'a' ] += 1;
    }
    
    cin>>queryCount;
    
    while( queryCount-- )
    {
    	cin>>firstCh>>secondCh;
    	if( firstCh != secondCh )
    		cout<< alphaCount[ firstCh - 'a' ] * alphaCount[ secondCh - 'a' ] <<endl;
    	else
    		cout<< ( alphaCount[ firstCh - 'a' ] * ( alphaCount[ firstCh - 'a' ] - 1 ) )/2<<endl;
    }
    return 0;
}
