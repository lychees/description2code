from math import factorial

def main():
    T = int(raw_input())
    while T>0:
        M,N = map(int,raw_input().strip().split(" "))
        if N-M >=2:
        	print -1
        else:
        	spaces = M+1
        	result = factorial(M)*factorial(N)*((factorial(spaces))/(factorial(N)*factorial(spaces-N)))
        	print result%1000000007
        T=T-1



main()
