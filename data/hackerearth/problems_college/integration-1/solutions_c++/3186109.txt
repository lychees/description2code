#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	for(int j=1;j<=t;j++)
	{
		long long int n,p,u,r,s;
		cin>>n>>p>>u>>r>>s;
		int sum = p;
		for(int i=1;i<=n;i++)
		{
			if(sum + r <= u)
			{
				sum = sum + r;
			}
			else if((sum-s)>=0 && sum-s<=u)
			{
				sum = sum - s;
			}
		}
		cout<<"Case #"<<j<<": "<<sum<<"\n";
	}
}
