#include <iostream>

using namespace std;

int mini(int a,int b)
{
    return a<b? a : b;
}
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
       long long a,b;
       cin>>a>>b;
       int na=0;
       int nb=0;
       long long da=a;
       long long db=b;
       while(da>0)
       {
           da=da/10;
           na++;
       }
       while(db>0)
       {
           db=db/10;
           nb++;
       }
       int n=mini(na,nb);
       int ca,cb,c;
       int cou=0;
       while(n--)
       {
           ca=a%10;
           a=a/10;
           cb=b%10;
           b=b/10;
           c=ca+cb;
           if(c>9)
            cou++;
       }
       if(cou==0)
        cout<<"No carry operation"<<endl;
       else if(cou!=1)
        cout<<cou<<" "<<"carry operations"<<endl;
        else
        cout<<"1 carry operation"<<endl;
   }
}
