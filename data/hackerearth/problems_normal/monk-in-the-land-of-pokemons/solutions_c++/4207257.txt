#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	vector<int> food;
    	vector<int> pok(1000000,0);
    	
    	int item=0;
    	for(int i=0;i<n;i++)
    	{
    		int x,y;
    		cin>>x>>y;
    		food.push_back(x);
    		++pok[x];
    		if( pok[y])
    		{
    			--pok[y];
    		}
    		else
    		{
    			++item;
    		//	food.push_back(y);
    		}
    	}
    	cout<<item<<endl;
    }
    return 0;
}
