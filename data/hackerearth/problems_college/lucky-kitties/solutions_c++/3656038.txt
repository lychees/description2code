#include <iostream>
using namespace std;

int main()
{
  int k,n,c,m,j;
  m=0;
  c=0;
  cin>>k;
  cin>>n;
  for(int i=1;i<=n;i++)
  {
  	j=i;
  	for(;j>0;)
  	{
  		if(j%10==4||j%10==7)
  			c++;
  			j/=10;
  			
  	}
  	if(c>k)
  	{
  		c=0;
  		m++;
  	}
  	
  }
  cout<<m;
  return 0;
}
