#include <iostream>
#include <algorithm>
#include <vector>
#include <cmath>
#include <iterator>
using namespace std;

bool isprime (const int& n, vector<int>& v) 
{
  double root = sqrt (n);
  vector<int>::iterator it = v.begin();
  while ( it != v.end() && *it <= root ) 
  {
    if (!( n % *it)) return 0;
    it++;
  }
  return 1;
}

int main() 
{
  long long t, count=0;
  cin>>t;
  vector<int> arr;

  for(int i=2; i<=sqrt(100000000); i++)
    if(isprime(i, arr)) 
        arr.push_back( i );

  while(t--) 
  {
    count = 0;
    long long m, n;
    cin >> m;
    cin >> n;
    if (m<2)
        m = 2;

    for(int i=m; i<=n; i++)
    {
        if(isprime(i, arr)) 
           count++;
    }
    cout<<count<<endl;
  }
  return 0;
}
