#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long D;
		cin>>D;
		int temp=1,sum=0;
		while(sum<D)
		{
			sum+=temp;
			temp+=1;
		}
		if((temp-1)%3==0)
			cout<<"SL"<<endl;
		else if((temp-1)%3==1)
			cout<<"LB"<<endl;
		else
			cout<<"BS"<<endl;
	}
   
    return 0;
}
