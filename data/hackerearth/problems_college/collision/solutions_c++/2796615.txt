#include<iostream>
using namespace std;
int main()
{
	int T,i,j,N,X,a[107],count=0,k;
	cin>>T;
	for(i=0;i<107;i++)
	{
		a[i]=0;
	}
	for(i=0;i<T;i++)
	{	
		count =0;
		cin>>N;
		for(j=0;j<N;j++)
		{
		cin>>X;
		if(a[X%10]==1)
		count++;
		else
		a[X%10]=1;
		}
		for(k=0;k<107;k++)
		{
		a[k]=0;
		}
	cout<<"\n"<<count;
	}	
	return 0;
}
