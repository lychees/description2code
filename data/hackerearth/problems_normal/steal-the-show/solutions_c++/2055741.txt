ll sl(){ 
  int c =gc();
  while((c<'0' || c>'9') && c!='-') {
    c = gc();
  }
  bool sign = 0;
  if(c=='-') {
    sign = 1;
    c = gc();
  }
  ll ret = 0;
  while(c>='0' && c<='9') {
    ret = 10 * ret + c - 48;
    c = gc();
  }
  if(sign) return -ret;
  return ret;
}

int ss(char *s) {
  int n = 0;
  int c = gc();
  while(!((c>='a' && c<='z') || (c>='A' && c<='Z'))) c = gc();
  while(((c>='a' && c<='z') || (c>='A' && c<='Z'))) {
    s[n] = (char)(c);
    n++;
    c = gc();
  }
  return n;
}

int a[25];
int b[25];


int main() {
  srand((ui)time(0));
  ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
  int h1, h2, h3;
  cin>>h1>>h2>>h3;
  int n;
  cin>>n;
  for(int i=0; i<n; i++) {
    cin>>a[i];
  }
  int res = 1e9;
  for(int step=0; step<100000; step++) {
    random_shuffle(a,a+n);
    if(rand()%2) { b[0] = h2; b[n+2] = h3; }
    else { b[0] = h3; b[n+2] = h2; }
    int ps = (n+3)/2;
    if((n+3)%2==0 && rand()%2) ps--;
    b[ps] = h1;
    int cur = 0;
    int d = 1;
    for(int i=0; i<n; i++,d++) {
      if(d==ps) d++;
      b[d] = a[i];
    }
    for(int i=1; i<n+3; i++) {
      cur+=abs(b[i]-b[i-1]);
    }
    res= min(res,cur);
  }
  cout<<res<<endl;
  
  
  
  return 0;
