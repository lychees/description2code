#include<bits/stdc++.h>
using namespace std;
int main()
{
    long int s[100005],i,n,t;
    s[0] = 0;
    for(i=1;i<100001;i++)
    s[i] = s[i-1] + i*(i-1);
    double ans[100005];
    for(i=1;i<=100000;i++)
    {
        ans[i] = 2*s[i] - i*(i-1) + i*i*(i-1);
        ans[i]*=2;
        ans[i]/=(i*i);
        ans[i]++;
    }
    cin>>t;
    while(t--)
    {
        scanf("%ld",&n);
        printf("%.6lf\n",ans[n]);
    }
    return 0;
}
