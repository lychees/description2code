#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int n,k;
   scanf("%d%d",&n,&k);
   vector<int> v;
   int i,x;
   for(i=0;i<n;i++)
   {
   	scanf("%d",&x);
   	v.push_back(x);
   }
   sort(v.begin(),v.end());
   int ma = v[n-1] - v[0];
   for(i=k-1;i<n;i++)
   if(v[i] - v[i+1-k] < ma)ma = v[i]-v[i+1-k];
   printf("%d\n",ma);
    return 0;
}
