#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	long long N, M;
    	cin>>N;
    	if(N&(N-1))
    	{
    		M = pow(2, ceil(log2(N)));
    		cout<<M-1<<" "<<N - (M - N)<<endl;
    	}
    	else
    	{
    		cout<<N-1<<" "<<N<<endl;
    	}
    }
    return 0;
}
