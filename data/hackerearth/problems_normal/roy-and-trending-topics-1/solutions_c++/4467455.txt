int compare(pair<li, pair<li, li> > a, pair<li, pair<li, li> > b)//id tot gain
{
    if(a.second.second == b.second.second)
        return a.first < b.first;
    return a.second.second > b.second.second;
}
int main()
{
    /*freopen("1in.txt", "r", stdin);

    priority_queue< pair<li, pair<li, li> > ,vector<pair <li, pair<li,li> > >, decltype(&compare)> arr(&compare);
    li id,z,p,l,c,s;
    li n; sli(n);
    while(n--)
    {
        sli(id);sli(z);sli(p);sli(l);sli(c);sli(s);
        li newz = p*50 + l*5 + c*10 + s*20;
        arr.push(make_pair(id,make_pair(newz,z-newz)));
    }
    n = 5;
    while(n-- && !arr.empty())
    {
        auto temp = arr.top();
        arr.pop();
        printf("%ld %ld\n",temp.first,temp.second.first);
    }
    return 0;
}
