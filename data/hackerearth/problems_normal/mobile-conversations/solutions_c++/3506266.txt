#define MOD 1000000007
struct data
{
	int a,b;
	int ying;
}d[111111];
bool cmp(data d1,data d2)
{
	return d1.ying>d2.ying||((d1.ying==d2.ying)&&d1.a<d2.a);
}
bool cmp2(data d1,data d2)
{
	return d1.b>d2.b;
}
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int i,n;
	long long ll=1,rr=1e18,mm,t;
	long long re;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>d[i].a>>d[i].b;
		if(d[i].b>d[i].a)
			d[i].ying=1;
		else
			d[i].ying=0;
	}
	sort(d,d+n,cmp);
	for(i=0;i<n;i++)
		if(d[i].ying==0)
			break;
	sort(d+i,d+n,cmp2);
	while(ll<=rr)
	{
		mm=(ll+rr)/2;
		t=mm;
		for(i=0;i<n;i++)
		{
			if(t<d[i].a)
				break;
			t-=d[i].a;
			t+=d[i].b;
		}
		if(i==n)
		{
			re=mm;
			rr=mm-1;
		}
		else
		{
			ll=mm+1;
		}
	}
	cout<<re<<endl;
//	system("pause");

}
