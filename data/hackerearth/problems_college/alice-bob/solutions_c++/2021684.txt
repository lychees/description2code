#include <bits/stdc++.h>
using namespace std;

int main()
{
	int N;

	cin >> N;
	int n = N;

	int s[110];
	memset(s, 0, 110);

	queue<int> q;

	while(n--)
	{
		int t;
		cin >> t;
		q.push(t);
		s[t]++;
	}

	bool player = true;  //true for 'Alice' and false for 'Bob'.
	bool gameOver = false;

	while(!gameOver)
	{
		int a = q.front(); //first element
		int count = 0;

		q.pop();
		q.push(a);
		count++;

		while(s[abs(a - q.front())] > 0)
		{
			int x = q.front();
			q.pop();
			q.push(x);
			count++;

			if(count >= q.size())
			{
				gameOver = true;
				break;
			}
		}

		if(gameOver)
			break;

		s[abs(a - q.front())]++;
		q.push(abs(a - q.front()));

		player = !player; //change the player.

	}

	if(player)
		cout << "Alice"; //print the reverse.
	else
		cout << "Bob";

	return 0;
}
