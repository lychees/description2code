#include <iostream>
using namespace std;

string fun(string s)
{
    int l = s.length();
    int i;
    for(i = 0; i < l/2; i++)
    swap(s[i],s[l-1-i]);
        return s;
}

int main()
{
    int n;
    cin>>n;
    string s[n];
    int i;
    for(i = 0; i < n; i++)
    {
        cin>>s[i];
    }
    int flag = 0;
    for(i = 0; i < n; i++)
    {
        string p = fun(s[i]);
        for(int j = 0 ; j < n; j++)
        {
            if(p == s[j])
            {
                int l = s[j].length();
                cout<<l<<" "<<s[j][l/2]<<endl;
                flag = 1;
                break;
            }
        }
        if(flag == 1)
            break;
    }
    return 0;
}
