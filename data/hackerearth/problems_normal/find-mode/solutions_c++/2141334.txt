#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
    int T,N;
    scanf("%d",&T);
    while(T--)
    {
    	scanf("%d",&N);
    	vector <pair <int,int>> v;
    	int freq[1000001]={0},X;
    	while(N--)                   // Loop to count element frequency
    	{
    		scanf("%d",&X);
    		freq[X]++;
    	}
    	int i=0,size;
    	while(i<1000001)
    	{
    		if(freq[i]!=0)
    		v.push_back(make_pair(freq[i],i));
    		i++;
    	}
    	sort(v.begin(),v.end());     // a pair is sorted as per first element.
    	reverse(v.begin(),v.end());  //reverse order to retrieve largest element.
    	i=0;
    	size=v.size();
    	while(v[i].first==v[0].first&&i<size)
    	 cout<<v[i++].second<<" ";
    	cout<<endl;
    }
    return 0;
}
