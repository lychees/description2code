#include <bits/stdc++.h>
#define pb push_back
using namespace std;

int main() {
	// your code goes here
	int n, r;
	cin >> n >> r;
	vector<int> v, v1;
	int ch[n];
	memset(ch, 0, sizeof ch);
	for(int i = 1; i <= n; i++){
		v.pb(i);
	}
	int pos = 0, loop = 0;
	while(1) {
		int k = v.size();
		if(k == 1) break;
		int mid = r % k;
		if(mid == 0){
			mid += k;
			mid--;
		}else {
			mid--;
		}
		v1.clear();
		for(int i = mid + 1; i < v.size(); i++){
			v1.pb(v[i]);
		}
		for(int i = 0; i <= mid - 1; i++){
			v1.pb(v[i]);
		}
		v.clear();
		for(int i = 0; i < v1.size(); i++){
			v.pb(v1[i]);
		}
	}
	cout << v[0];
	return 0;
}
