#include <bits/stdc++.h>
using namespace std;
int result=0;
int k;
 
struct node
{
	int data;
	struct node *next;
};
typedef struct node node;
 
node *A[10005];
int visited[10005];
 
void adj_list(int vi,int vj)
{
	node *temp;
	temp=(node *)malloc(sizeof(node));
	temp->data=vj;
	temp->next=NULL;
	if(A[vi]==NULL)
		A[vi]=temp;
	else
	{
		temp->next=A[vi];
		A[vi]=temp;
	}
	
}
 
int dfs(int v)
{
	int Count;
	node *curr;
	visited[v]=1;
	Count=1;
 
	//cout<<"v: "<<v<<endl;
	curr=A[v];
	while(curr!=NULL)
	{
		if(visited[curr->data]==0)
			Count+=dfs(curr->data);
 
		
		curr=curr->next;
	}
	if(Count>=k)
	{
		result++;
		Count=0;
	}
	return Count;
 
}
 
int main()
{
	int n, i, v, u;
	cin>>k;
	cin>>n;
	//Count=0;
	memset(visited, 0, 4*10005);
	for(i=2;i<=n;i++)
	{
		cin>>u;
		adj_list(u, i);
	}
	dfs(1);
	cout<<result<<endl;
	return 0;
}
