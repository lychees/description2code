#include <cstdio>
#include <iostream>
using namespace std;
#define gc getchar_unlocked
int inpt()
{
	int n=0;
	char c=gc();
	while(c<'0'||c>'9')
	c=gc();
	while(c>='0'&&c<='9')
	{n=n*10 + (c-'0');
	 c=gc();}
	return n;
}
long long int inpl()
{
	long long int n=0;
	char c=gc();
	while(c<'0'||c>'9')
	c=gc();
	while(c>='0'&&c<='9')
	{n=n*10 + (c-'0');
	 c=gc();}
	 return n;
}
int main()
{
    int t;
    long long int n, i, sum ;
    t=inpt();
    while(t--)
    {
    	n=inpl();
    	sum=0;
    	for(i=1;i*i<=n;i++)
    	{
    		if(n%i==0)
    		{
    			if(i%4!=0)
    			sum+=i;
    			if((n/i)%4!=0&&(n/i)!=i)
    			sum+=n/i;
    		}
    	}
    	
    	printf("%lld\n", 8*sum);
   
    }
}
