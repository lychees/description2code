#include <iostream>
using namespace std;

int main()
{
    int t,n,d,ans;
    cin>>t;
    while(t--)
    {
    	cin>>n>>d;
    	while(n!=0)
    	{
    		if(n>=3)
    		{
    			n=n-1;
    			d=d-6;
    		}
    		else if(n<3)
    		{
    			d=d-(n*2);
    			n=0;
    		}
    		if(d<=0)
    			break;
    		
    	}
    	if(n<=0 && d>0)
    		cout<<"No\n";
    	else cout<<"Yes\n";
    }
}
