#include<iostream>
using namespace std;
struct node 
{
	long id;
	node *next;
};
node *f,*b;
node * ins(node *,long,char);
void change();
int main()
{
	
	int t;
	cin>>t;
	while(t!=0)
	{	node *start=NULL;
		long n,no;
		char ch='B';
		cin>>n>>no;
		start = ins(start,no,ch);
		while(n!=0)
		{
			cin>>ch;
			if(ch=='P')
			{
				cin>>no;
				start = ins(start,no,ch);
			}
			else
			{
				change();
			}
			n--;
		}
		cout<<"Player "<<f->id<<endl;
		t--;
	}
	return 0;
}

node * ins(node *start,long no,char ch)
{
	node *ptr;
	ptr=new node;
	ptr->id=no;
	if(start==NULL)
	{
		start=ptr;
		f=start;
		b=start;
		start->next=NULL;
	}
	else 
	{
		ptr->next=start;
		start=ptr;
		b=f;
		f=ptr;
	}
	return start;
}
void change()
{
	node *temp;
	temp=b;
	b=f;
	f=temp;
}
