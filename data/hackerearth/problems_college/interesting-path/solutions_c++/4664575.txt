#include <bits/stdc++.h>
using namespace std;
int a[1000];
int jump[1000];

int main()
{
    int t,n,i,j;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	for(i=0;i<n;i++)
    		jump[i] = INT_MAX;
    	jump[0] = 0;
    	for(i=1;i<n;i++)
    	{
    		for(j=0;j<i;j++)
    		{
    			if(i<=(j+a[j]))
    				jump[i] = min(jump[j]+1,jump[i]);
    		}
    	}
    	cout<<jump[n-1]<<endl;
    }
    return 0;
}
