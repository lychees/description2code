#include <iostream>
using namespace std;

int p[100000];

void sieve()
{
   
    for(int i=2;i<10007;i++)
    {
    	if(p[i]==0)
    	{
    		int j=2*i;
    		while(j<10007)
    		{
    			p[j]=1;
    			j=j+i;
    		}
    	}
    }
}
int main()
{
	sieve();
    int n;
    cin>>n;
    for(int i=4;i<=n;i++)
    {
    	if(p[i]==1)
    	cout<<i<<endl;
    }
    return 0;
}
