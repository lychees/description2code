#include <stdio.h>
using namespace std;
int main()
{
	int testCases;
	scanf("%d", &testCases);
	for(int c=0;c<testCases;c++)
	{
		int scores[3];
		scanf("%d%d%d", &scores[0], &scores[1], &scores[2]);
		int total = 0; 
		for(int i=0;i<3;i++)
		{
			if(scores[i] == 9)
			{
				total += (10 - (3 * i));
				break;
			}
			else if(scores[i] < 7 && scores[i] > 0)
				if(i != 2)
					total += 1;
				else 
					total = 0;
			else if(scores[i] > 6 && scores[i] < 9)
				if (i != 2)
					total += 2;
				else 
					total = 0;
			else if(scores[i] == 0)
				if(i != 2)
					total += 0;
				else 
					total = 0;
		}
		printf("%d\n", total);
	}
}
