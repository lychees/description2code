#include <iostream>
using namespace std;

int main()
{
	int side;
	cin >> side;
	int n;
	cin >> n;
	for (int i = 0; i < n; i++){
		int a, b;
		cin >> a >> b;
		if (a < side || b < side) cout << "UPLOAD ANOTHER" << endl;
		else{
			if (a == b) cout << "ACCEPTED" << endl;
			else cout << "CROP IT" << endl;
		}
	}
    return 0;
}
