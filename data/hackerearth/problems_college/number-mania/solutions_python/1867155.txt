	if x==2:
		return 1
	if x%2==0:
		return 0
	i=3
	while i*i<=x :
		if x%i==0 :
			return 0;
		i=i+1;
	return 1;

def factorial(x):
	if x==1 :
		return 1
	else :
		return x*(factorial(x-1))
