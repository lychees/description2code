def prime(x):
    i = 2
    while i*i <= x:
        if x % i == 0:
            return False
        i += 1
    return True


for _ in range(input()):
    n = input()
    if prime(n) == False:
        print 'NO'
        continue
    else:
        rev = 0
        while n:
            x = n % 10
            rev = rev+x
            n = n/10
    # print rev
        if prime(rev):
            print 'YES'
        else:
            print 'NO'
