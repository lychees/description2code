#include <iostream>
#include<cmath>
using namespace std;

unsigned long long mod =1e9+7;

long long powbase(long long num, long long base)
{
	if(base==1||num==0)
	return 1;
	
	else if(num%2==0)
		return ((powbase(num/2, base)%mod)*(powbase(num/2,base))%mod);
	else 
		return (base*powbase(num-1,base))%mod;
}
		
	
	

int main()
{	
	int t,n;
 	
	cin>>t;
	for(int i=0;i<t;i++)
	{
		cin>>n;
	cout<<(powbase(n,4)-powbase(n,2)+mod)%mod<<endl;
	
	}
	

}
