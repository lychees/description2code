#include <iostream>
#include<cstdio>
#include<cstdlib>
using namespace std;

int main()
{
    long long int a, b, n, j, k, sum;
    scanf("%lld%lld%lld", &n, &a, &b);
    n--;
    j=n/a;
    sum=(j*(j+1)*a)/2;
    k=n/b;
    sum+=(k*(k+1)*b)/2;
    //printf("%d", sum);
    k=j/b;
    sum-=(k*(k+1)*(a*b))/2;
    printf("%lld\n", sum);
    return 0;
}
