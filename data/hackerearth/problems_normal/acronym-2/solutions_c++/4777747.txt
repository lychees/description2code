#include <iostream>
#include <set>
using namespace std;

int main()
{
	set<string>word;
	int n, i, k, j, q=0;
	string final, h, g;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>g;
    	word.insert(g);
    }
    cin>>k;
    for(j=0;j<k;j++)
    {
    	cin>>g;
    	set<string>::iterator it=word.find(g);
    	if(it==word.end())
    	{
    		final[q]=toupper(g[0]);
    		q++;
	    	final[q]='.';
	    	q++;
    	}
    }
    for(int r=0;r<q-1;r++)
    {
    	cout<<final[r];
    }
    return 0;
}
