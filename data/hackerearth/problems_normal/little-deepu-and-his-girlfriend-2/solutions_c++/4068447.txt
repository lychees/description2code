#include <iostream>

using namespace std;

int main()
{
	int t;

	cin >> t;

	while(t--)
    {
        long int m;

        int n;

        cin >> m >> n;

        long int arr[n];

        for(int i=0;i<n;i++)
        {
            cin >> arr[i];
        }

        bool dp[m+1];

        for(long int i=0;i<=m;i++)
        {
            dp[i]=false;
        }

        for(int i=0;i<n;i++)
        {
            dp[arr[i]]=true;
        }

        for(long int i=1;i<=m;i++)
        {
            bool wins = false;

            for(int j=0;j<n;j++)
            {
                if(i>=arr[j] && !dp[i-arr[j]])
                {
                    wins=true;

                    break;
                }
            }

            dp[i]=wins;
        }

        if(dp[m])
        {
            cout << "Little Deepu" << endl;
        }
        else
        {
            cout << "Kate" << endl;
        }
    }

    return 0;
}
