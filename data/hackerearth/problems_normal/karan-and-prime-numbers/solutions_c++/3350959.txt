#include<iostream>
#include<math.h>
using namespace std;
int k;
int prime1[1000000];
void prime(int b)
{
  bool arr[1000000];
  arr[0]=true;
  arr[1]=true;
  //prime1[range];

 for(int i=2;i<1000000;i++)
 {
   arr[i]=false;
  }
  // int k=floor(sqrt(b));
  int range=floor(sqrt((double)b));
 for(int i=0;i<=floor(sqrt((double)range));i++)
 {
   if(arr[i]==false)
   {
     for(int j=i*i;j<=range;j+=i)
     {
       arr[j]=true;
     }
   }
 }
  k=0;
  for(int i=0;i<=range;i++)
  {
   if(arr[i]==false)
   {
    prime1[k]=i;
    k++;
   }
  }
 /* for(int i=0;i<k;i++)
  {
   cout<<prime1[i]<<endl;
  }*/
}

 int main()
 {
   int t;
   int sum;
   bool primeNow[1000000];
   int a,b;

   cin>>t;
   for(int i=0;i<t;i++)
   {
   	   sum=0;
       cin>>a>>b;
       prime(b);
     for(int l=0;l<1000000;l++)
        primeNow[l]=false;

     for(int p=0;p<k;p++)
     {
       int m=a/prime1[p];
           m=m*prime1[p];
       for(int j=m;j<=b;j+=prime1[p])
        {
          if(j<a)
            continue;
          else
            primeNow[j-a]=true;
        }
     }
     cout<<endl;

     for(int j=0;j<k;j++)
     {
         if(prime1[j]>=a && prime1[j]<=b)
           sum+=prime1[j];
     }
     for(int j=0;j<b-a+1;++j)
     {
         if(primeNow[j]==false && (j+a)!=1)
            sum+=j+a;
     }
     cout<<sum;
   }

   return 0;
 }
