#include <bits/stdc++.h>
#include<stdlib.h>

using namespace std;

int main()
{
    string S;
    cin>>S;
    int q,l;
    cin>>q;
    l=S.size();
    long long a,b;
    while(q--){
    	scanf("%lld%lld", &a, &b);
    	a--;
    	b--;
    	a=a%l;
    	b=b%l;
    	if(S[a] == S[b])
    		cout<<"Yes"<<endl;
    	else
    		cout<<"No"<<endl;
    }
    return 0;
}
