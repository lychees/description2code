#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,i,j;
    long int sum;
    cin>>t;
    while(t--)
    {
    	
    	cin>>n;
    	long int a[n];
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	
    	cin>>i>>j;
    	sort(a,a+n);
    	
    	sum=a[j-1]+a[n-i];
    	cout<<sum<<endl;
    }
    return 0;
}
