#include <iostream>
using namespace std;

int main()
{
    unsigned long long int t, x, k, y, count, j, flag;
    cin >> t;
    for(j=1; j<=t; j++) {
    	cin >> x >> k;
    	y = 1;
    	flag = count = 0;
    	while(flag == 0) {
    		if((x+y) == (x xor y)) {
    			count++;
    		}
    		if(count == k) {
    			cout << "Case #" << j << ": " << y << endl;
    			flag = 1;
    		}
    		y++;
    	}
    } 
    return 0;
}
