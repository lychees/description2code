#include<bits/stdc++.h>
using namespace std;
#define ll long long
map<int,ll> prime(int n)
{
	map<int,ll> r;
	ll cnt=0;
	while(!(n&1))
	{
		n/=2;
		cnt++;
	}
	r[2]=cnt;          
	for(int i=3;i<=sqrt(n);i+=2)
	{
		cnt=0;      
		if(n%i==0)
		{
			while(n%i==0)
			{
				cnt++;
				n/=i;             
			}          
		}  
		r[i]=cnt;      
	}
	if(n>1)
	{
		r[n]=1;       
	}        
	return r;    
}
int main()
{
  	//freopen("in.txt","r",stdin);
  	//freopen("out.txt","w",stdout); 
  	ll t;
  	scanf("%lld",&t);
  	while(t--)
	{
		ll a,b,c,d,flag=1;
		scanf("%lld %lld %lld %lld",&a,&b,&c,&d);
		map<int,ll> x,y;
		x=prime(a);
		y=prime(c);
		map<int,ll>::iterator p;
		for(p=y.begin();p!=y.end();p++)
		{
			if((p->second)*d>x[p->first]*b)
			{
				flag=0; 
				break;                
			}                               
		}       
		if(!flag)
		{
			printf("Not divisible\n");         
		}   
		else
		{
			printf("Divisible\n");    
		}
	}  	
  	return 0; 
}
