#include<stdio.h>
#include<string.h>
#include<map>
using namespace std;
int main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		char s[1010];
		scanf("%s", s);
		map<char,int> m;
		char aa[27]="abcdefghijklmnopqrstuvwxyz";
		int i,j, cc[26];
		for(i=0;i<26;i++)
		{
			m[aa[i]]=i;
			cc[i]=0;
		}
		i=0;
		while(i<strlen(s))
		{
			cc[m[s[i]]]++;
			i++;
		}
		int count=0;
		for(i=0;i<26;i++)
		{
			if(cc[i]%2==1)
				count++;
		}
		if(count!=0)
			count--;
		printf("%d\n", count);
	}
}
