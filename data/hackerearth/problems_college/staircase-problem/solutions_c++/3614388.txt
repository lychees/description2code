#include<iostream>
#include<cstdlib>
#include<cmath>
using namespace std;
int cal(int n)
{
    int sq=(int)sqrt(n);
    while((sq*(sq+1)/2)<=n)
        sq++;
    return sq-1;
}
int main()
{

    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        int n;
        cin>>n;
        cout<<cal(n)<<endl;
    }
    return 0;
}
