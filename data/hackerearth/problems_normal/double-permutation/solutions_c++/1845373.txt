int tree[200000+5];
//vector<int> g[2*100000+5];int par[2*100000+5];
int read(int idx){
    int sum = 0;
    while (idx > 0){
        sum += tree[idx];
        idx -= (idx & -idx);
    }
    return sum;
}
#define MaxVal 200000
void update(int idx ,int val){
    while (idx <= MaxVal){
        tree[idx] += val;
        idx += (idx & -idx);
    }
}

lld pow(lld base, lld exponent,lld modulus)
{
    lld result = 1;
    while (exponent > 0)
    {
        if (exponent % 2 == 1)
            result = (result * base) % modulus;
        exponent = exponent >> 1;
        base = (base * base) % modulus;
    }
    return result;
}

int a[200000+5];
int done[200000+5];
int main()
{
//	fre;
	int x,N;
	cin>>N;
	lld ans=0;
	int t=0;
	for(int i=1;i<=2*N;++i)
	{
		scan(a[i]);
	}
	for(int i=2*N;i>=1;i--)
	{
		int x = a[i];
		if(!done[x])
		{
			ans += read(x-1) * pow(2,N-2,MOD);
			ans += read(x + N - 1) * pow(2,N-2,MOD);
			done[x] = 1;
		}
		else
		{
			ans += read(x-1) * pow(2,N-2,MOD);
			ans += (read(x + N - 1) -1)* pow(2,N-2,MOD) + pow(2,N-1,MOD);
		}
		ans %= MOD;
		update(x,1);
		update(x+N,1);
	}
	cout<<ans;
}
