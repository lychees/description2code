		rdi(x); rdi(y);
		v.pb(mp(mp(x,y),i));
	}

		x=v[i].F.F; y=v[i].S;
		f=0;
		rep(j,1,m+1) {
			if(!b[j]) {
				f=1;
				b[j]=x;
				c[j]=v[i].F.S;
