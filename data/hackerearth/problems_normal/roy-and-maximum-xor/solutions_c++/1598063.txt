#include <iostream>
using namespace std;

int main()
{
	int T;
	unsigned long long N, tempN;
	cin >> T;
	while (T--) {
		cin >> N;
		N--;
		int bitz = 0;
		tempN = N;
		while (tempN > 0) {
			bitz++;
			tempN >>= 1;
		}
		int ansBitz = bitz;
		tempN = 1L << (bitz - 1);
		tempN >>= 1;
		bitz -= 2;
		unsigned long long res = 1;
		while (tempN > 0) {
			if (N & tempN) {
				res += (1L << bitz);
			}
			tempN >>= 1;
			bitz -= 1;
		}
		unsigned long long az = 1;
		az <<= ansBitz;
		az -= 1;
		res <<= 1;
		if (N == 0) { res = 1; }
		cout << az << " " << res << "\n";
	}
    return 0;
}
