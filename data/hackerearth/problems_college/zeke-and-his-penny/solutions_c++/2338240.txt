#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--)
   {
   	long long int arr[3],num=1;
   	scanf("%lld%lld%lld",&arr[0],&arr[1],&arr[2]);
   	sort(arr,arr+3);
   	for(int i=0;i<3;i++)
   	{
   		num=(num*(arr[i]-i))%mod;
   	}
   	cout<<num<<endl;
   }
    return 0;
}
