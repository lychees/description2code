#include <bits/stdc++.h>
using namespace std;

int a[2000][2000];

int main()
{
    int n;
    cin >> n;
    int m = (1 << n);
    
    for(int i = 0; i < m; ++i) {
    	for(int j = 0; j < i; ++j) {
    		cin >> a[i][j];
    		a[j][i] = 1 - a[i][j];
    	}
    }
    
    
    vector<int> cur, nxt;
    for(int i = 0; i < m; ++i) nxt.push_back(i);
    
    for(int i = 0; i < n; ++i) {
    	cur = nxt;
    	nxt.clear();
    	for(int i = 0; i < cur.size(); i += 2) {
    		if(a[cur[i]][cur[i + 1]]) nxt.push_back(cur[i]);
    		else nxt.push_back(cur[i + 1]);
    	}
    }
    
    cout << nxt[0] + 1 << endl;
    
    return 0;
}
