#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    int t;
    char *str;
    bool flag=0;
    str=new char[1];
    cin>>t;
    while(t--)
    {
        cin>>str;
        for(int i=0;i<strlen(str)-1;i++)
        {
            if(str[i]==str[i+1])
            {
                flag=1;
                break;
            }
        }
        if(!flag)
            cout<<"KISS"<<endl;
        else
            cout<<"SLAP"<<endl;
        flag=0;
    }
    return 0;
}
