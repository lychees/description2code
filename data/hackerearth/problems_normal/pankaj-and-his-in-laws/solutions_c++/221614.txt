#include <iostream>
#include <cstdio>
#include <cstring>
#include <climits>
#define N 101
using namespace std;

void print(int n)
{
    if(n>0)
    {
        int t=n%2;
        print(n/2);
        printf("%d",t);
    }
}

void longIncSubsequence(int arr[],int n)
{
    int i=0,j,maxi=0;
    int table[N];
    memset(table,1,N);

    for(j=0;j<n;j++)
    table[j]=1;

    for(i=1;i<n;i++)
    for(j=0;j<i;j++)
    if(arr[i]>arr[j] && table[i]<table[j]+1)
    table[i]=table[j]+1;

    for(j=0;j<n;j++)
    if(table[j]>maxi)
    maxi=table[j];
    print(maxi);
}

int main()
{
    int i=0,t,arr[N];
    scanf("%d",&t);
    for(;i<t;i++)
    scanf("%d",&arr[i]);
    longIncSubsequence(arr,t);
    return 0;
}
