#include<bits/stdc++.h>
using namespace std;
int main()
{
int t;
cin>>t;
while(t--)
{
long long int b,w,x,y,z;

cin>>x>>y>>z;
unsigned long long int s1,s2,s3;
s1=(b*x)+(w*y);
s2=(b*y)+(w*y)+(b*z);
s3=(b*x)+(w*x)+(w*z);
if(x<=z&&y<=z)cout<<s1<<endl;
if(x>z&&y<=z){
if(s1<s2)cout<<s1<<endl;
else cout<<s2<<endl;
}
if(x<=z&&y>z)
{
if(s1<s3)cout<<s1<<endl;
else cout<<s3<<endl;


}
if(x>z&&y>z&&x>y)
{
if(s1<s2)cout<<s1<<endl;
else cout<<s2<<endl;

}
if(x>z&&y>z&&x<y)
{
if(s1<s3)cout<<s1<<endl;
else cout<<s3<<endl;

}


}

}
