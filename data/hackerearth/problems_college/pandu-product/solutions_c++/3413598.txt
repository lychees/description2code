#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    ll i,N,Ele1,Ele2,Max,Max1,Max2;
    scanf("%lld",&N);
    vector<ll> Arr(N);
    for(i=0;i<N;i++)
    {
        scanf("%lld",&Arr[i]);
    }
    sort(Arr.rbegin(),Arr.rend());
    Max1 = Arr[0]*Arr[1];
    Max2 = Arr[N-1]*Arr[N-2];

    if(Max1>Max2)
        Max = Max1;
    else
        Max = Max2;
    printf("%lld",Max);

    return 0;
}
