bool check(int c,int a[],int n)
{
    int i,prev=0;
    rep(i,0,n)
    {
        if(a[i]>prev)
        {
            prev=max(prev+1,a[i]-c);
        }
        else
        {
            if(a[i]+c<=prev) return false;
            prev=min(prev+1,a[i]+c);
        }
    }
    return true;
}
