#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string a,b;
		cin>>a>>b;
		int n;
		cin>>n;
		int m=0;
		if(a.length()!=b.length())
			cout<<"No"<<endl;
		else
		{
			for(int i=0;i<a.length();i++)
			{
				if(a[i]!=b[i])
					m++;
					
			}
			if(m>n)
			cout<<"No"<<endl;
			else
			{
				int e=n-m;
				if(e%2==0)
					cout<<"Yes"<<endl;
				else
					cout<<"No"<<endl
;				
			}
		}
	}
	
	
	return 0;
}
