#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,k,n,t,m;
    cin>>n;
    vector <int> v(n);
    for(i=0;i<n;i++)
    {
        cin>>v[i];
    }
    cin>>m;
    k=0;
    sort(v.rbegin(),v.rend());
    /*for(i=0;i<n;i++)

    }*/
    for(i=0;i<n;i++)
    {
        k+=v[i];

        {
            //printf("%d\n",k);
            cout<<i+1<<endl;
            break;
        }
    }
    return 0;
}
