#include <bits/stdc++.h>
using namespace std;
int cpink[24];
int corange[24];
int cyellow[24];
int arr[24][4];
int n;
int func(int cIndex, int lIndex);
int main()
{
	int tc;
	scanf("%d", &tc);
	while (tc--) {
		scanf("%d", &n);
		int k;
		for (k = 1 ; k <= n ; k++) {
			scanf("%d %d %d", cpink + k, corange + k, cyellow + k);
		}
		memset(arr, -1, sizeof(arr));
		int mini;
		mini = min(cpink[1] + func(2, 1), corange[1] + func(2, 2));
		mini = min(mini, cyellow[1] + func(2, 3));
		printf("%d\n", mini);
	}
	
	return 0;
}

int func(int cIndex, int lIndex) {
	if (cIndex > n) {
		return 0;
	}
	if (arr[cIndex][lIndex] == -1) {
		if (lIndex == 1) {
			arr[cIndex][lIndex] = min(corange[cIndex] + func(cIndex + 1, 2), cyellow[cIndex] + func(cIndex + 1, 3));
		} else if (lIndex == 2) {
			arr[cIndex][lIndex] = min(cpink[cIndex] + func(cIndex + 1, 1), cyellow[cIndex] + func(cIndex + 1, 3));
		} else if (lIndex == 3) {
			arr[cIndex][lIndex] = min(cpink[cIndex] + func(cIndex + 1, 1), corange[cIndex] + func(cIndex + 1, 2));
		}
	}
	return arr[cIndex][lIndex];
}
