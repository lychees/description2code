#include<string.h>
#include <iostream>
#include <stack>
using namespace std;

int main()
{

int m,l,t=0,i,j;
cin>>m;
for(i=0;i<m;i++)
{
   std::stack<char> stack;
   char arr[1000000];
   cin>>arr;
   l=strlen(arr);
   stack.push(arr[0]);
   for(j=1;j<l;j++)
   {
     if((!stack.empty())&&arr[j]==stack.top())
     {
       stack.pop();
     }
     else
     {
      stack.push(arr[j]);
     }
   }
   
if(stack.empty()){t++;}

}

cout<<t;
}
