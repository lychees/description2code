#include <iostream>
using namespace std;

void init()
{
int n;
cin>>n;
int a,b;
for(b=0;5*b<=n;b++)
if((n-5*b)%3==0)
break;
a = (n-5*b);
b = b*5;

if(b<=n)
{
for(int i=0;i<a;i++)
cout<<5;
for(int i=0;i<b;i++)
cout<<3;
}
else
cout<<-1;

cout<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    init();
    return 0;
}
