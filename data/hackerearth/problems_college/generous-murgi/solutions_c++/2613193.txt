#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int x,y;
        scanf("%d%d",&x,&y);
         if(x==y)
        {
        printf("-1\n");
        continue;
        }
        vector<int> x_mm;
        vector<int> y_mm;
        for(int i=1;i<=sqrt(x);i++)
        {
            if(x%i==0)
            {
                x_mm.push_back(i);
                x_mm.push_back(x/i);
            }
        }
        for(int i=1;i<=sqrt(abs(x-y));i++)
        {
            if(abs(x-y)%i==0)
            {
                y_mm.push_back(i);
                y_mm.push_back(abs(x-y)/i);
            }
        }
        sort(x_mm.begin(),x_mm.end());
        sort(y_mm.begin(),y_mm.end());
        x_mm.erase( unique( x_mm.begin(), x_mm.end() ), x_mm.end() );
        y_mm.erase( unique( y_mm.begin(), y_mm.end() ), y_mm.end() );
        if(y==0)
        {
        printf("%d\n",x_mm.size());
        continue;
        }
        int ctr = 0;
        vector<int>::iterator up;
        for(int i=0;i<y_mm.size();i++)
        {
            up=upper_bound(x_mm.begin(),x_mm.end(),y_mm[i]);
            int index=up-x_mm.begin()-1;
            if((up==x_mm.end()||x_mm[index]!=y_mm[i])&&(x%y_mm[i]==y))
                ctr++;
        }
        printf("%d\n",ctr);
    }
    return 0;
}
