#include <iostream>
#include <stdio.h>
using namespace std;

int binSearch(int ar[], int m, int query)
{
	int beg = 0;
	int end = m - 1;
	int mid = (beg + end)/2;
	while(beg <= end)
	{
		if(query == ar[mid])
			return mid;
		else if (query > ar[mid])
			beg = mid + 1;
		else
			end = mid - 1;
		mid = (beg + end)/2;
	}
	return -1;
}

int main()
{
    int m, n;
    cin>>n>>m;
    int a[n][m];
    for(int i = 0; i < n; i++)
    {
    	for(int j = 0; j < m; j++)
    		scanf("%d", &a[i][j]);
    }
    int q, query, j;
    cin>>q;
    while(q > 0)
    {
    	int flag = 0;
    	cin>>query;
    	for(int i = 0; i < n; i++)
    	{
    		j = binSearch(a[i], m, query);
    		if(j >= 0)
    		{
    			flag = 1;
    			printf("%d %d\n", i, j);
    			break;
    		}
    	}
    	if(flag == 0)
    		printf("-1 -1\n");
    	q--;
    }
    return 0;
}
