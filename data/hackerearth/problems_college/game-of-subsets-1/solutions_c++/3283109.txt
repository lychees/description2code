#include<bits/stdc++.h>

using namespace std;
#define ll long long

const int mod = 1000000007;
ll f(ll x,ll y)
{
    ll res = 1ll;
    while(y)
    {
        if(y%2)res = (res*x)%mod;
        x = (x*x)%mod;
        y/=2;
    }
    return res;
}
int main()
{
    ll t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        cout<<f(3,n)<<"\n";
    }
    return 0;
}
