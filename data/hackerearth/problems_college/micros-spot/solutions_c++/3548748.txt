#include<bits/stdc++.h>
#define ll long long
#define mp make_pair
using namespace std;
int a[1000][1000];
int b[1000][1000];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,m;
        scanf("%d%d",&n,&m);
        int row[n];
        int col[m];
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                scanf("%d",&a[i][j]);
            }
        }
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                scanf("%d",&b[i][j]);
            }
        }
        for(int i=0;i<n;i++)
        {
        	int mx=0;
            for(int j=0;j<m;j++)
            {
                mx=max(mx,b[i][j]);
            }
            row[i]=mx;
        }
        for(int i=0;i<m;i++)
        {
            int mx=0;
            for(int j=0;j<n;j++)
            {
                mx=max(mx,a[j][i]);
            }
            col[i]=mx;
        }
        int ans=0;
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                if(a[i][j]==col[j] && b[i][j]==row[i]) ans++;
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
