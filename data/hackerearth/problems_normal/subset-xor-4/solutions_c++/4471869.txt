#include <iostream>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--) {
        cin>>n;
        int a[n];
        for(int i = 0;i < n;++i) {
            cin>>a[i];
        }
        if(n == 1) {
            cout<<a[0]<<"\n";
        }
        else {
            cout<<0<<"\n";
        }
    }
    return 0;
}
