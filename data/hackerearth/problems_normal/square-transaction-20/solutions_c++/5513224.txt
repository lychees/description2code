#include <bits/stdc++.h>
using namespace std;
 
 typedef long long int ll;
int main()
{
    ll t,q,sum=0,num,target,low,mid,high,res,flag;
    cin>>t;
    ll arr[t];
    for(ll i=0;i<t;i++)
    {
    	cin>>num;
    	sum+=num;
    	arr[i]=sum;
    }
      cin>>q;
      while(q--)
      {
      	cin>>target;
      	low=0;high=t-1;flag=0;res=t+1;
      	while(low<=high)
      	{
      		mid=(low+high)/2;
      		if(arr[mid]==target)
      		{
      			res=mid+1;
      			break;
      		}
      		else if(arr[mid]<target)
      		low=mid+1;
      		   else
      		   {
      		   	if(res>mid)
      		   	res=mid+1;
      		   high=mid-1;
      		   }
      	}
      	if(res==t+1)
      	cout<<-1<<endl;
      	else
      	cout<<res<<endl;
      }
    return 0;
}
