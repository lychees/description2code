#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t)
	{
		int r,c,sum1=0,sum2=0;
		cin>>r>>c;
		for(int i=1;i<=r;i++)
		{
			if(i%2!=0)
			{
				if(i<=c)
				{
					sum1+=i;
				}
				else
				{
					sum1+=c;
				}
			}
			else
			{
				if(i<=c)
				{
					sum2+=i;
				}
				else
				{
					sum2+=c;
				}
			}
		}
		cout<<sum2<<" "<<sum1<<endl;
		--t;
	}
return 0;
}
