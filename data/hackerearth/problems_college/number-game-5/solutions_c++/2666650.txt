#include <bits/stdc++.h>
using namespace std;

vector<long long> dp;

int main() {
	int t;
	cin >> t;
	long long temp = 1;
	long long cnt = 1;
    temp = cnt * pow (cnt+1, 2);
	while (temp > 0 && temp < pow(10, 18)) {
	    dp.push_back (temp);
	    cnt++;
	    temp += cnt * pow (cnt+1, 2);    
	}
	int size = dp.size();
	while (t--) {
	    long long n;
	    scanf ("%lld \n", &n);
	    //cout << n << endl;
        if (dp[0] > n) {
	        cout << "0" << endl;
	        cout << "Draw" << endl;
	        continue;
	    }
	    int index = 0, l = 0, r = size - 1;
	    while (l < r) {
	        int mid = (l+r)/2;
	        if (dp[mid] == n) {
	            index = mid;
	            break;
	        } else if (dp[mid] < n) {
	            if (mid+1 < size) {
	                if (dp[mid+1] > n) {
	                    index = mid;
	                    break;
	                } else {
	                    l = mid+1;
	                }
	            }
	        } else {
	            if (mid-1 >= 0) {
	                if (dp[mid-1] <= n) {
	                    index = mid-1;
	                    break;
	                } else {
	                    r = mid-1;
	                }
	            }
	        }
	    }
	    index++;
        if (index % 2 == 1) {
            printf ("%d \nDhruv\n", index);
        } else {
			printf ("%d \nKetan\n", index);
        } 
	}
	return 0;
}
