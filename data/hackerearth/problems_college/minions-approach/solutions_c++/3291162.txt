#include <cstdio>
#include <cassert>
#include <cmath>
#include <algorithm>
using namespace std;
#define MAXN 100000
#define MAXT 10
bool ar[MAXN+1];
void sieve(){
    for(int i=0;i<MAXN+1;i++)
        ar[i] = true;
    ar[1] = false;   
    for(int i=2;i<MAXN+1;i++){
        if(ar[i])
            for(int j=2*i; j<=MAXN;j+=i)
                ar[j]=false;
    }
    return;
}
void solve(){
    int N;
    scanf("%d", &N);
    assert(N>0);
    assert(N<=MAXN);
    
    int cnt = 0;
    if(ar[N])
        cnt = N-1;
    else{
        for(int i=1;i<=N;i++)
            if(__gcd(i, N)==1)
                cnt+=1;
    }
    if(ar[cnt])
        printf("Correct\n");
    else
        printf("Incorrect\n");
}
int main(){
    sieve();
    int T;
    scanf("%d", &T);
    assert(T>0);
    assert(T<=MAXT);
    while(T--)
        solve();
    return 0;
}
