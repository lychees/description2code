#include <iostream>
#include<string.h>
#include<math.h>
using namespace std;

int main()
{
   unsigned int t,j,l,i,flag;
   char s[100],r[100];
   cin>>t;
    while(t--)
    { flag=0;
      cin>>s;
      l=strlen(s);
      for(i=(l-1),j=0;i>=0,j<l;i--,j++)
      r[j]=s[i];
      for(i=1;i<l;i++)
       {
       	if(fabs(s[i]-s[i-1])!=fabs(r[i]-r[i-1]))
       	{
       	flag=1;
       	break;
       	}
       }
       if(flag==1)
       cout<<"Not Funny"<<endl;
       else
       cout<<"Funny"<<endl;
       
    }
return 0;
}
