#include<bits/stdc++.h>


using namespace std;

int main(){
	bool M[1000001];
	vector<int> A;
 
 	long long int n;
 		scanf("%lld",&n);
 		A.resize(n);
 			for(int i=0;i<n;i++) 
 				scanf("%d",&A[i]);
 			memset(M,0,sizeof(M));
 			M[0]=true;
 
 			for(int i=0;i<n;i++){
   				for(int j=100000-A[i];j>=0;j--){
     				if(M[j]){
       					M[j+A[i]]=true;
     				}
   				}
 			}
 
 	int tc;
 	scanf("%d",&tc);
 	while(tc--){
   		int s=0;
   		scanf("%lld",&s);
   		if(M[s]){
   			printf("Yes\n");
   		}
   		else{
   			printf("No\n");
   		}
 	}
 return 0;
}
