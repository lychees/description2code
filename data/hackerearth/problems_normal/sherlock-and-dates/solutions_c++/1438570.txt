using namespace std;

bool leap(int y){
	if(y%4!= 0) return false;
	if(y%100 != 0) return true;
	if(y%400 != 0) return false;
	return true;
}

int day(int m, int y){
	if(m==2){
		if(leap(y)){
			return 29;
		}
	}
	int days[] = {31,28,31,30,31,30,31,31,30,31,30,31};
	return days[m-1];
}

bool comp(int d, int m, int y, int D, int M, int Y){
	if(y<Y) return true;
	if(y>Y) return false;
	if(m<M) return true;
	if(m>M) return false;
	if(d<=D) return true;
	return false;
}


int main(){
	int T;
	cin >> T;
	while(T--){
		int d,m,y;
		scanf("%d:%d:%d", &d,&m,&y);


		int D,M,Y;
		scanf("%d:%d:%d", &D,&M,&Y);

		int ans = 0;
		for(int i=y; i<=Y; i++){
			int MM = (i-1)%100;
			if(MM<=0 || MM > 12) continue;
			int DD = MM-1;
			if(DD<=0 || DD > day(MM,i)) continue;
			if(comp(d,m,y, DD,MM,i) && comp(DD,MM,i, D,M,Y)) ans++;
		}

		cout << ans << endl;
	}
	return 0;
}
