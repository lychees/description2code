#!/usr/bin/env python
# coding: utf-8

import math

line = raw_input().split()

n = int(line[0])
k = int(line[1])

answer = n

for i in range(2*n + 1):
    for j in range(1, int(math.sqrt(i)) + 1):
        if i % j == 0:
            if answer is None:
                answer = abs(n - i) + min(abs(k - j), abs(k - i // j))
            else:
                answer = min(answer, abs(n - i) +
                             min(abs(k - j), abs(k - i // j)))

print(answer)
