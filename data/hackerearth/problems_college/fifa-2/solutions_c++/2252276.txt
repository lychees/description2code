#include <iostream>
using namespace std;

int main()
{
	long long  x,y,a[100001]={0},b[100001],n,m,i;
	cin>>n;
    for(i=1;i<=n;i++){
    	cin>>m;
    	a[m]++;
    	cin>>b[i];
    }
    for(i=1;i<=n;i++){
    	x=y=n;
    	x=x+a[b[i]];
    	y=y-a[b[i]];
    	cout<<x-1<<" "<<y-1<<"\n";
    }
    return 0;
}
