#include <iostream>
using namespace std;

int n,maxWeight;
int wt[1005],value[1005];
int dp[1005][1005];

int solve(int i,int sum)
{
	if(i<0 || sum<=0)
		return 0;
	if(dp[i][sum]!=-1)
		return dp[i][sum];
	int ans=0;
	if(wt[i]>sum)
		ans=solve(i-1,sum);
	else
		ans=max(solve(i-1,sum),solve(i-1,sum-wt[i])+value[i]);
	dp[i][sum]=ans;
	return dp[i][sum];
}

int main()
{
    int t,i,j;
    cin>>t;
    while(t--)
    {
    	cin>>n>>maxWeight;
    	for(i=0;i<n;i++)
    		cin>>wt[i];
    	for(i=0;i<n;i++)
    		cin>>value[i];
    	for(i=0;i<1005;i++)
    		for(j=0;j<1005;j++)
    			dp[i][j]=-1;
    	int ans=solve(n-1,maxWeight);
    	cout<<ans<<endl;
    }
    return 0;
}
