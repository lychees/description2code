#include<iostream>
#include<vector>
using namespace std;
int main(){
	   int t;
	   cin>>t;
	   int arr[22];
	   for(int i=1;i<=21;i++){
	   	  arr[i]=i*i*i;
	   }
	   long long dp[10001]={0};
	   dp[0]=1;
	   for(int i=1;i<=21;i++){
	   	  for(int j=0;j<=10000;j++){
	   	  	  if(j>=arr[i]){
	   	  	  	  dp[j]+=dp[j-arr[i]];
	   	  	  }
	   	  }
	   }
	   while(t--){
	   	  int n;
	   	  cin>>n;
	   	  cout<<dp[n]<<endl;
	   }
	}
