#include <iostream>
#include <vector>
#include<algorithm>
using namespace std;

int main()
{
    int t,n;
    vector<int> v;
    
    cin>>t;
    
    while(t--){
    	cin>>n;
    	v.clear();
    	int temp;
    	for(int i=0;i<n;i++)
    		cin>>temp,v.push_back(temp);
    	
    	sort(v.begin(),v.end());
    	
    	for(int i=1;i<n;i++)
    		v[i]=v[i]-v[0];
    	v[0]=0;
    	
    	int t,count,fcount=0;
    	
    	for(int i=1;i<n;i++){
    		int temp=v[i];
    		
    		count=temp/5;
    		temp=temp%5;
    		
    		count=count+temp/2;
    		temp=temp%2;
    		
    		count=count+temp;
    		
    		fcount+=count;
    	}
    	
    	cout<<fcount<<endl;
    }
    return 0;
}
