template <class T> inline T bigmod(T p,T e,T M){
    LL ret = 1;
    for(; e > 0; e >>= 1){
        if(e & 1) ret = (ret * p) % M;
        p = (p * p) % M;
    } return (T)ret;
}
template <class T> inline T gcd(T a,T b){if(b==0)return a;return gcd(b,a%b);}
template <class T> inline T modinverse(T a,T M){return bigmod(a,M-2,M);}

set < pair<int,int> > S[100001];
int pre[100001];
int Find(int x){
    if(x == pre[x]) return x;
    return pre[x] = Find(pre[x]);
}
pair < int, pair < int, int > > pp[100001];
LL res[100001];
int main(){
    ios_base::sync_with_stdio(0); cin.tie(0);
    int n; cin >> n;
    REP(i, n-1) cin >> pp[i].yy.xx >> pp[i].yy.yy >> pp[i].xx;
    sort(pp, pp + n-1);

    int m; cin >> m;
    REP(i, m){
        int sz; cin >> sz;
        REP(j, sz){
            int x; cin >> x;
            S[x].insert(mp(i, 1));
        }
    }

    FOR(i, 1, n+1) pre[i] = i;

    for(int i = n-2; i >= 0; i--){
        int px = Find(pp[i].yy.xx), py = Find(pp[i].yy.yy);
        if(S[px].size() > S[py].size()) swap(px, py);
        for(auto &v: S[px]){
            set<pair<int,int> >::iterator it = S[py].lower_bound(mp(v.xx, 0));
            if(it == S[py].end()) {
                S[py].insert(v);
                continue;
            }
            if((*it).xx != v.xx) {
                S[py].insert(v);
                continue;
            }
            res[v.xx] += (*it).yy * (LL)v.yy * (LL)pp[i].xx;
            int val = (*it).yy + v.yy;
            S[py].erase(it);
            S[py].insert(mp(v.xx, val));
        }
        S[px].clear();
        pre[px] = py;
    }

    REP(i, m) cout << res[i] << endl;
}
