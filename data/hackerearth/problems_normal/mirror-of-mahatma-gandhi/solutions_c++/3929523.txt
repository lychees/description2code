#include <bits/stdc++.h>
using namespace std;

int ispali(char s[])
{
	int l = strlen(s);
	for(int i=0,j=l-1;i<=j;i++,j--)
	{
		if(s[i]!=s[j])return 0;
	}
	return 1;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char s[1000];
    	cin>>s;
    	int l = strlen(s);
    	int f=0;
    	
    	for(int i=0;i<l;i++)
    	{
    		if(s[i]=='1'||s[i]=='0'||s[i]=='8')continue;
    		//{
    			f=1;
    			break;
    		//}
    	}
    	if(!f && ispali(s))cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
    return 0;
}
