def nextperm(s):
    w=list(s)
    l=len(w)-1
    ind=-1
    for i in range(l):
        if w[i+1]>w[i]:
            ind=i
    if ind==-1:
        print('no answer')
        return
    for j in range(ind+1,l+1):
        if w[j]>w[ind]:
            m=j
    #swap
    k=w[ind]
    w[ind]=w[m]
    w[m]=k
    opt=w[:ind+1]+sorted(w[ind+1:])
    print(''.join(opt))
    return

for t in range(int(input())):
	n=raw_input()
	nextperm(n)
