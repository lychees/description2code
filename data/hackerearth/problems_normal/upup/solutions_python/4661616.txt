#print 'Hello World!'
import re

inputStr = raw_input()
inputStrArr = re.split(r'(\s+)', inputStr)

for j, elem in enumerate(inputStrArr):
	if elem:
		elemArr = list(elem)
		elemArr[0] = elemArr[0].upper()
		#print elemArr
		inputStrArr[j] = "".join(elemArr)

print "".join(inputStrArr)
