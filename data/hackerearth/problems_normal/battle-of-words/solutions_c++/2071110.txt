#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
	int test;
	scanf("%d\n",&test);
	while(test--)
	{
		char s1[100010],s2[100010];
		fgets(s1,sizeof(s1),stdin);
		fgets(s2,sizeof(s2),stdin);
		//~ scanf("%s%s",s1,s2);
		int h1[26],h2[26];
		memset(h1,0,sizeof(h1));
		memset(h2,0,sizeof(h2));
		int i;
		for(i=0;s1[i]!='\0';i++)
		{
			if(s1[i]>='a' && s1[i]<='z')
			{
				h1[s1[i]-'a']++;
			}
		}
		for(i=0;s2[i]!='\0';i++)
		{
			if(s2[i]>='a' && s2[i]<='z')
			{
				h2[s2[i]-'a']++;
			}
		}
		int alice=0,me=0;
		for(i=0;i<26;i++)
		{
			if(h1[i]>h2[i])
			{
				alice=1;
			}
			else if(h2[i]>h1[i])
			{
				me=1;
			}
		}
		if(alice==me)
		{
			printf("You draw some.\n");
		}
		else if(alice==1)
		{
			printf("You win some.\n");
		}
		else
		{
			printf("You lose some.\n");
		}
	}
	return 0;
}
