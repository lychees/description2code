
#include <iostream>
#include <queue>
#include <stdio.h>
using namespace std;


class num{
	public :
	int val;
	
	};
	
bool operator<(const num &a,const num &b){  
              return a.val>b.val; 
		 }	
int main(){
	
	int t;
	cin>>t;
	int n;
	
	
	int yes;
	for(int m=0;m<t;m++){
			  cin>>n;
			  priority_queue<num> q;
			  num x;
			  
			  while(!q.empty())
						q.pop();
			  yes=1;
			  for(int j=0;j<n;j++){
				  scanf("%d",&x.val);
				  q.push(x);
			  }
			   
			  int factor=0;              /*Bullet factor*/
			  for(int j=1;j<=n;j++){
				   
                   int last=q.top().val;
                   factor++;
				   if((last-factor)<0)
				      {
						  yes=0;
						  break;
				   }
				   else{
					   q.pop();
					   }
				   
				   if(j%6==0) 
				      factor++;
				      
				      
				  }
	       if(yes)
					cout<<"Rick now go and save Carl and Judas"<<endl;
			else{
					cout<<"Goodbye Rick"<<endl;			 
					cout<<n-q.size()<<endl; 
			}	  
				  
	}
	
	return 0;
}
