#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    fflush(stdin);
    long long int a,b,c,d,s,i;
    while(t-- > 0)
    {
    	cin >> a >> b >> c >> d >> s;
    	if(a+s > c){cout << "STAY IN LONDON\n";continue;}
    	if(b+s < c){cout << b << endl;continue;}
		
    	for(i=b;i>=a;i--)
    	{if(i+s == c)cout << i << endl;}
    	
    }
    return 0;
}
