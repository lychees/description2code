#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,x;
    	cin>>n>>x;
    	int a[n];
    	int s = 0;
    	for(int i=0;i<n;i++){cin>>a[i];s+=a[i];}
    	int b[s+1];
    	for(int i=0;i<=s;i++)b[i]=0;
    	b[0]=1;
    	for(int i=0;i<n;i++)
    	 for(int j=s;j>=a[i];j--)
    	   b[j]=max(b[j],b[j-a[i]]);
    	int ans = x+1;
    	for(int i=0;i<=s;i++)
    	   if(b[i])
    	    ans = min(ans,max(i,s-i));
    	if(ans<=x)cout<<"YES\n";
    	else
    	cout<<"NO\n";
    }
    return 0;
}
