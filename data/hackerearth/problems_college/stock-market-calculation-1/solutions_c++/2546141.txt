#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    int t;
    
    int n;
    scanf("%d", &t);
    while(t > 0) {
    	scanf("%d", &n);
    	int arr[n];
    	for(int i=0; i<n; i++) {
    		scanf("%d", &arr[i]);
    	}
    	long long cost = 0;
    	int maxPrice = arr[n-1];
    	int numShare = 0;
    	long long profit = 0;
    	for(int i=n-2; i>=0; i--) {
    		if(maxPrice < arr[i]) {
    			profit += (maxPrice * numShare -cost);
    			maxPrice = arr[i];
    			cost = 0;
    			numShare = 0;
    			
    		} else {
    			cost += arr[i];
    			numShare += 1;
    		}
    		
    	} 
    	if(cost != 0) {
    		profit += maxPrice*numShare - cost;
    	}
    	printf("%lld\n", profit);
    	t--;
    }
    return 0;
}
