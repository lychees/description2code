#include <cstdio>
#include <string>
#include <cstring>
#include <iostream>
using namespace std;
int main(){
string S;
int Q, slength;
long num1, num2;

cin >> S;
slength = S.size();
scanf("%d", &Q);


for(int i=0;i<Q;i++)
{
scanf("%lld %lld", &num1, &num2);
num1--;
num2--;
num1 = num1%slength;
num2 = num2%slength;

if (S[num1]==S[num2])
printf("Yes\n");
else
printf("No\n");
}
return 0;
}
