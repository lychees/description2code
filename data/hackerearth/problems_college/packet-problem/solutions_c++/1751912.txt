#include <iostream>
using namespace std;

int main()
{

    long long no_0=0, no_1=1;
    cin>>t;
    while(t--) {
    	cin>>m;
    	no_0 = 0, no_1 = 1;
    	long long temp_0 = 0 , temp_1 = 0;
    	while(m--) {
    		temp_0 =0, temp_1 = 0;
    		temp_0 = no_0+no_1;
    		temp_1 = no_0;
    		no_0 = temp_0;
    		no_1 = temp_1;
       	}
       	cout<<no_1<<" "<<no_0<<endl;
    }

}
