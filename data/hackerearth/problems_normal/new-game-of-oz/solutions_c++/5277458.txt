#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
    int n;
    cin>>n;
    vector<int> ar;
    int x;
    for(int i=0;i<n;i++)
    {
    	cin>>x;
    //	cout<<x;
    	ar.push_back(x);
    }
    sort(ar.begin(),ar.end());
    int cc=0,i=0;
    while(i<n)
    {
    	int a1=ar[i]+1;
    	int a2=ar[i+1];
    //	cout<<a1<<" "<<a2<<" ";
    	if(a1==a2)
    	{
    		i++;
    	}
    	i++;
    	cc++;
    }
    cout<<cc<<endl;
	}
    return 0;
}
