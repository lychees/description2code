#include <iostream>
#include<bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	ll n,ch=0,l=0;
    	cin>>n;
    	int a[n];
      for(int i=0;i<n;i++)
       {
       	cin>>a[i];
       }
       sort(a,a+n);
       for(int i=0;i<n-1;i++)
       {  
       	  ll p=0;
       	  p=a[i+1]-a[i] + l;
       	  l=p;
       	  while(p>0)
       	  {
       	    if(p/5>=1)
       	     {
       	  	  ch=ch+p/5;
       	  	   p=p%5;
       	     }
       	     else if(p/2>=1)
       	     {
       	  	  ch=ch+p/2;
       	  	   p=p%2;
       	     }
       	     else
       	     {
       	     	p=0;
       	     	ch=ch+1;
       	     }
       	  }
        } 
       cout<<ch<<endl;
    }
    return 0;
}
