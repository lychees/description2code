#include <iostream>
#include <string>
#include <algorithm>
#include <vector>
#include <functional>
#include <map>
using namespace std;

int main()
{
	int t, n;
	cin >> t;
	while (t--) {
		cin >> n;
		vector<int> vn(n);
		for (auto i = 0; i < n; ++i)
			cin >> vn[i];

		int count = 0;
		for (auto i = 0; i < n; ++i) {
			for (auto j = i + 1; j < n; ++j) {
				if (vn[i] != vn[j] && (vn[i] + vn[j]) % 2 == 0)
					count++;
			}
		}
		cout << count << endl;
	}

	return 0;
}
