#include <iostream>
using namespace std;
uint64_t n,sweet[1000001][4],g,member[1000001],a,b;
    uint64_t i,j,k,l,max_q=0,max1=0,max2,count=0,p,q,max_p=0,mid;
    uint64_t sum=0,end_v,arr[1000001]; 


int main()
{
    
    cin>>n;
    for(i=1;i<=n;i++)
    {
    	cin>>a>>b;
    	sweet[a][1]=a;
    	sweet[a][2]=b;
    	sweet[a][3]=b;
    	if(a>max1)
    	{
    		max1=a;
    	}
    
    }
    for(i=max1;i>1;i--)
    {
    	if(sweet[i-1][3]<sweet[i][3])
    	{
    		sweet[i-1][3]=sweet[i][3];
    	}
    	
    }
  
    cin>>g;
    sum=0;
    for(i=1;i<=g;i++)
    {
    	cin>>member[i];
    	max_q=sweet[member[i]][3];
    	sum=sum+100*(max_q/member[i]);
    }
    
    cout<<sum<<"\n";
}
