#include<bits/stdc++.h>
using namespace std;
#define ll long long int
bool is_good(ll num)
{
    ll first_bit=num&1,second_bit;
    num>>=1;
    while(num>0)
    {
        second_bit=num&1;
        if((second_bit==1)&&(first_bit==1))
            return false;
        first_bit=second_bit;
        num=num>>1;
    }
    return true;
}
int main()
{
    ll num,k,ans,rem,total,flag,test;
    scanf("%lld",&test);
    while(test--)
    {
        scanf("%lld",&num);
        while(!is_good(num))
          num++;
        cout<<num<<endl;
    }
}
