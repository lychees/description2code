#include<bits/stdc++.h>
using namespace std;
int fun(int a,int b)
{
	return (a>b);
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	int b[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		b[i]=a[i];
    	}
    	sort(a,a+n);
    	sort(b,b+n,fun);
    	for(int i=0;i<n;i++)
    	{
    		a[i]=a[i]+b[i];
    	}
    	int mini=*std::min_element(a,a+n);
    	int maxi=*std::max_element(a,a+n);
    	cout<<maxi-mini<<endl;
    	
    }
}
