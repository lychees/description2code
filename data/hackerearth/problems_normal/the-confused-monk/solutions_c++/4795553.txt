#include<iostream>

using namespace std;

long long GCD(long long a,long long b){
	if(b == 0 )
	{
		return a;
	}
	else
		GCD(b,a%b);
	}
	
int main(){
	
	int A[1000],i,N;
	long long result,product=1,ans=1,MonkQuotient = 1000000007;
	
	cin>>N;
	for(i=0;i<N;i++){
		cin>>A[i];
	}
	
	for(i =0;i<N;i++){
		product = (product*A[i])%MonkQuotient;
	}
	if(N == 1){
		result = GCD(A[0],A[0]);
	}
	else{
		result = A[0];
		for(i =1;i<N;i++){
			result = GCD(result,A[i]);
		}
	}
	
	while(result != 0){
		if(result%2 == 1){
			ans = (ans * product)%MonkQuotient;
		}
		product = (product * product)%MonkQuotient;
		result = result/2;
	}
	cout<<ans;
}
