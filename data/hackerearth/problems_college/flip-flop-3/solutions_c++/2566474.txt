#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	int len=str.size();
    	int count=0;
    	for(int i=1;i<len;i++)
    	{
    		if(str[i]==str[i-1])
    		{
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
}
