#include <iostream>
using namespace std;

int main()
{
    int a, b, k = 1, ansmin = 0, ansmax = 0, a1, a2, x, y;
    cin >> a >> b;
    while(a or b)
	{
		x = a % 10; a /= 10;
		y = b % 10; b /= 10;
		if(x == 6) a1 = 5;
		else a1 = x;
		if(y == 6) a2 = 5;
		else a2 = y;
		ansmin = ansmin + (a1+a2)*k;
		
		if(x == 5) a1 = 6;
		else a1 = x;
		if(y == 5) a2 = 6;
		else a2 = y;
		ansmax = ansmax + (a1+a2)*k;
		k *= 10;
	}
	
	cout << ansmin << ' ' << ansmax << endl;
    return 0;
}
