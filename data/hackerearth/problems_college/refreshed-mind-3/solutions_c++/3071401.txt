#include <iostream>
using namespace std;
 
unsigned long int getTotalWorstCaseClicks(unsigned long int a[],unsigned long int n)
{
	unsigned long int clicks = a[0];
	unsigned long int correctlyAnswered = 1;
	
	for (unsigned long int i = 1; i < n; i++)
	{
		clicks += ( 1 + (a[i] - 1)*(1+correctlyAnswered));
		correctlyAnswered++;
	}
	return clicks;
	
}
 
int main()
{
    unsigned long int numOfQuestions;
    
    cin >> numOfQuestions;
    
    unsigned long int ai[numOfQuestions];
    
    for (unsigned long int i = 0; i < numOfQuestions; i++)
    	cin >> ai[i];
    	
    unsigned long int result = getTotalWorstCaseClicks(ai, numOfQuestions);
    
    cout << result;
    
    return 0;
}
