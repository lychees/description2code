#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
	ll t, a, b;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld %lld", &a, &b);
		ll p=a%b;
		if (p==(float)b/2)
		printf("%lld\n", (a+p));
		else if (p<(float)b/2)
		printf("%lld\n", (a-p));
		else
		printf("%lld\n", (a+(b-p)));
	}
	return 0;
}
