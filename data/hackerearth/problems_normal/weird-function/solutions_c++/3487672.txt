#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#define mod 1000000007
#define ll long long
using namespace std;
ll ans[10001];
ll f(ll a,ll b){
    if(b==1)
        return a%mod;
    else{
        ll h=f(a,b/2)%mod;
        h=(h*h)%mod;
        if(b&1)
            return(a*h)%mod;
        else
            return h;
    }
}
void precompute(){
    ll i,j;
    
    for(i=1;i<=10000;i++){
        for(j=i;j<=10000;j=j+i){
            ans[j]=(ans[j]+f(i,j))%mod;
        }
    }
    for(i=1;i<=10000;i++){
        ans[i]=(ans[i]+ans[i-1])%mod;
    }
}



int main() {
   ll t,a,b;
    precompute();
    cin>>t;
    while(t--){
        cin>>a>>b;
        cout<<(ans[b]-ans[a-1]+mod)%mod<<"\n";
    }
    return 0;
}
