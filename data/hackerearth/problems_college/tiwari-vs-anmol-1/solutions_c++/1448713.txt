#include<iostream>
using namespace std;
#include<cmath>
int arr[1000001];
int main()
{
   for(int i=2;i<=sqrt(1000000);i++)
        if(arr[i]==0)
        for(int j=2;j<1000000/i;j++)
        if(arr[i*j]==0)
        arr[i*j]=i;
        for(int i=2;i<1000001;i++)
            arr[i]+=arr[i-1];
        int task;
        cin>>task;
        while(task--)
        {
            int l,k;
        cin>>l>>k;
            cout<<arr[k]-arr[l-1]<<endl;
        }
}
