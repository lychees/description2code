#include<iostream>
#include<algorithm>
using namespace std;
void show(int *ar,int n)
{
	cout<<"Array=";
	for(int i=0;i<n;i++)
	cout<<ar[i]<<" ";
	cout<<"\n";
}
int abc(int n)
{
	if(n<0)
	return (-1)*n;
	else return n;
}
int main()
  {
  	int n,r,count=0,N=0,pos=0,neg=0;
  	cin>>n>>r;
  	int ar[n];
  	for(int i=0;i<n;i++)
  	{
  		cin>>ar[i];
  		N=N+ar[i];
  	}
  	N/=n;
  	for(int i=0;i<n;i++)
  	{
  		ar[i]=N-ar[i];
  		if(ar[i]>0)
  		{
  		pos+=ar[i];}
  		else
  		neg+=ar[i];
  	}
  	if(pos!=abc(neg))
  	{
  	cout<<-1;
  	return 0;}
  	sort(ar,ar+n);
  	//show(ar,n);
  	for(;ar[n-1]!=0;sort(ar,ar+n))
  	{
  		ar[n-1]=ar[n-1]-abc(ar[0]);
  		ar[0]=0;
  		count++;
  	}
  	cout<<r*count;
   return 0;
  }
