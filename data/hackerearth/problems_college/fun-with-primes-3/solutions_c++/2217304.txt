#include<bits/stdc++.h>
int primes[100001];
using namespace std;
int main()
{
    int test;
    scanf("%d",&test);
    for(int i=2;i<=100000;i++)
        primes[i]=1;
    primes[0]=0;
    primes[1]=0;
    for(int i=2;i<=sqrt(100000);i++)
    {
        if(primes[i]==1)
        {
            for(int j=2;i*j<=100000;j++)
                primes[i*j]=0;
        }
    }
    int sum,A,B;
    long int num;
    while(test--)
    {
        scanf("%ld",&num);
        sum=0;
        while(num)
        {
            sum+=num%10;
            num=num/10;
        }
        if(primes[sum]==1)
            printf("YES\n");
        else
            printf("NO\n");
    }
}
