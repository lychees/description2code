#include <iostream>
#include <cmath>
using namespace std;
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
        int a,b,Max,Min;
        cin>>a>>b;
        Max = abs((a+b)/2);
        Min =  abs((a-b)/2);
        if(a == Max+Min) cout<<Max<<" "<<Min<<endl;
        else cout<<"impossible"<<endl;
    }

}
