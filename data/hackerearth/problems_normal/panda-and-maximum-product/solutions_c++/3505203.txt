#include<bits/stdc++.h>
using namespace std;
int main ()
{
	set<long long int >s;
	set<long long int >::iterator it;
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		long long int x;
		cin>>x;
		s.insert(x);
	}
	it=s.begin();
	long long int a=*it;
	long long int b=*++it;
	it=s.end();
	long long int c=*--it;
	long long int d=*--it;
	cout<<max(a*b,c*d)<<endl;
}
