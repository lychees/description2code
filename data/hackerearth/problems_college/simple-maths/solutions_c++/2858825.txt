#include<bits/stdc++.h>
using namespace std;
# define ll long long
int main()
{
	int t;
    ll n,a,b;
    cin>>t;
    while(t--)
    {
        cin>>n>>a>>b;
        if(a==0 || b==0)
            cout<<-1<<endl;
        else
        {
        ll lcm=(a*b)/__gcd(a,b);
        cout<<(n/a)+(n/b)-(n/lcm)<<endl;
        }
    }
}
