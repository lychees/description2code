#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define PI 3.14159265
int main()
{
	int t;
	double a,b,c,d,aa,bb,cc;
	double ans;
	cin>>t;
	while(t--)
	{
		ans=0;
		cin>>a>>b>>c>>d;
		aa=b-a;
		bb=d-c;
		double mini=min(aa,bb);
		cc=c-b;
		double l=cc/2;
		l=l/mini;
		if(cc>=(aa+bb))
		{
			printf ("%lf\n",ans);
		}
		else{
			ans = acos (l);
			l=acos(-1);
		ans=ans/l;
		printf ("%lf\n",ans);
		
		}
		
		
			
	}
		
	return 0;
}
