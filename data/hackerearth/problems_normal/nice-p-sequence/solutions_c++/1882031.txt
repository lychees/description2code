void inv  (ll a, ll b, ll &x, ll &y)
{
    if (a==0)
    {
        x=0; y=1; return; 
    }
    inv((b%a), a, x, y);
    int z=x; int zz=y;
    x=zz-(b/a)*z;
    y=z;
}
