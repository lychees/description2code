#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

long long h[100002],aw[100002];

struct block
{
	long long x,y;
};

int main() 
{
	long long n,i,j,cnt=0;
	cin>>n;
	block bl[200002];
	for(i=0;i<n;i++)
	{
		cin>>bl[i].x>>bl[i].y;
		h[bl[i].x]++;aw[bl[i].y]++;
	}
	for(i=0;i<n;i++)
	{
		cout<<n+(h[bl[i].y]-1)<<" "<<n-(h[bl[i].y])-1<<"\n";
	}
	return 0;
}
