#include<bits/stdc++.h>
using namespace std;
map<char,int> m,m1;
void init()
{
    m['.']=m['a']=m['d']=m['g']=m['j']=m['m']=m['p']=m['t']=m['w']=m['_']=1;
    m[',']=m['b']=m['e']=m['h']=m['k']=m['n']=m['q']=m['u']=m['x']=2;
    m['?']=m['c']=m['f']=m['i']=m['l']=m['o']=m['r']=m['v']=m['y']=3;
    m['!']=m['s']=m['z']=4;
    m['1']=m['7']=m['9']=5;
    m['2']=m['3']=m['4']=m['5']=m['6']=m['8']=4;
    m['0']=2;
    m1['.']=m1[',']=m1['?']=m1['!']=1;
    m1['a']=m1['b']=m1['c']=2;
    m1['d']=m1['e']=m1['f']=3;
    m1['g']=m1['h']=m1['i']=4;
    m1['j']=m1['k']=m1['l']=5;
    m1['m']=m1['n']=m1['o']=6;
    m1['p']=m1['q']=m1['r']=m1['s']=7;
    m1['t']=m1['u']=m1['v']=8;
    m1['w']=m1['x']=m1['y']=m1['z']=9;
    m1['_']=10;
    m1['1']=1;m1['2']=2;m1['3']=3;m1['4']=4;m1['5']=5;m1['6']=6;m1['7']=7;m1['8']=8;m1['9']=9;m1['0']=10;


}
int main()
{
    int T,i,j,k,curr,tm,tmp;
    string str;
    init();
    cin>>T;

    while(T--)
    {
        cin>>str;
        curr=1;
        tm=0;
        i=0;
        while(i<str.length())
        {
            if(m1[str[i]]==curr)
                tmp=0;
            else
            {
                tmp=1;
                curr=m1[str[i]];
            }

            tm+=m[str[i]]+tmp;
            i++;
        }
        cout<<tm<<"\n";

    }
}
