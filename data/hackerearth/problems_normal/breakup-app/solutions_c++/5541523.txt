#include<iostream>
#include<cstdlib>
#include<vector>
#include<map>
#include<cstring>
#include<bits/stdc++.h>


#define ll long long

using namespace std;

int main(void)
{
	ll Number,Gval=0,Bval=0;
	cin>>Number;
	
	while(Number--)
	{
		string str;
		getline(cin,str);
		//cin>>str;
		
		for(ll i=0;i<str.size();i++)
		{
			
		if(str[0]=='G')
		{
				if(str[i]=='1' && str[i+1]=='9' || (str[i]=='2' && str[i+1]=='0'))
				{
				   	Gval=Gval+2;
				}
				else if(str[i]<=39&&str[i]>=30&&str[i+1]<=39&&str[i+1]>=30)
				{
					Bval=Bval+2;
				}
			
		}
		  if(str[0]=='M')
		  {
			
				if(str[i]=='1' && str[i+1]=='9' || (str[i]=='2' && str[i+1]=='0'))
				{
				   	Gval=Gval+1;
				}
				else if(str[i]<=39&&str[i]>=30&&str[i+1]<=39&&str[i+1]>=30)
				{
					Bval=Bval+1;
				}
		   }
		}
		
	}
	if(Gval>Bval)
	{
		cout<<"Date"<<endl;
	}
	else
	{
		cout<<"No Date"<<endl;
	}
	return 0;
}
