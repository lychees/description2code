#include <iostream>
using namespace std;

void patternA(int a)
{
	for(int i=0;i<a;i++)
	{
		for(int j=0;j<(2*a);j++)
		{
			if((j-i)>a)
				break;
			if((i+j)%2!=a%2 && (i+j)>=a-1 && j<a)
				cout<<"/";
			else if((j-i)%2==a%2 && (j-i)<=a && j>=a)
				cout<<"\\";
			else
				cout<<" ";
		}
		cout<<endl;
	}
}

void patternV(int a)
{
	for(int i=0;i<a;i++)
	{
		cout<<"  ";
		for(int j=0;j<(2*a);j++)
		{
			if((i+j)>(2*a)-1)
				break;
			if((j-i)%2==0 && (j-i)>=0 && j<a)
				cout<<"\\";
			else if((i+j)%2==1 && (i+j)<(2*a) && j>=a)
				cout<<"/";
			else
				cout<<" ";
		}
		cout<<endl;
	}
}

int main()
{
	int n;
	cin>>n;
	patternA(n);
	patternV(n);
	patternA(n+1);
    patternV(n+1);
    return 0;
}
