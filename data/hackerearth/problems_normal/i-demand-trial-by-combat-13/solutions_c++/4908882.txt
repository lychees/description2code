#include <iostream>
using namespace std;

int main()
{
    int t,n,m,count=0;
    bool prev,temp;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	bool a[n];
    	count=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		if(a[i]==0)
    			count++;
    	}	
    	
    		
    	while(m--)
    	{
    		if(count==n)
    			break;
    		count=0;	
    		prev=a[0];
    		if(a[1]==1)
    		{
    			a[0]=1;
    		
    		}
    		else
    		{
    			a[0]=0;
    			count++;
    		}
    		for(int i=1;i<n-1;i++)
    		{
    			temp=a[i];
    			a[i]=prev&a[i+1];
    			prev=temp;
    			if(a[i]==0)
	    		{
	    			count++;
	    		}

	    	}
    		
    		if(prev==1)
    		{
    			a[n-1]=1;
    		
    		}
    		else
    		{
    			a[n-1]=0;
    			count++;
    		}
    			
    	}
    	for(int i=0;i<n;i++)
    		cout<<a[i]<<" ";
    	cout<<endl;
    	
    }
    return 0;
}
