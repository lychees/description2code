#include <bits/stdc++.h>
using namespace std;

int main()
{
	int c, p, n;
	cin >> c >> p >> n;

	vector<int> previq(c + 5), num(c + 5);
	vector<int> newstudentiq(p + 5);
	vector<int> oldstudentiq(n + 5);

	set<pair<int, int>> q;

	for (int i = 1; i <= n; i++)
	{
		cin >> oldstudentiq[i];
		previq[i] = oldstudentiq[i];
		num[i] = 1;
	}

	for (int i = 1; i <= c; i++)
	{
		int z = num[i] * previq[i];
		q.insert(make_pair(z, i));
	}

	for (int i = 1; i <= p; i++)
	{
		cin >> newstudentiq[i]; 
		pair<int, int> cpair = *(q.begin());
		int course = cpair.second;
		num[course]++;
		int z;
		if (num[course] == 1)
		{
			z = newstudentiq[i];
		}
		else
		{
			z = (newstudentiq[i] + previq[course]) * num[course];
		}
		q.erase(cpair);
		q.insert(make_pair(z, course));
		previq[course] = newstudentiq[i];
		cout << course << " ";
	}
	system("pause");
}
