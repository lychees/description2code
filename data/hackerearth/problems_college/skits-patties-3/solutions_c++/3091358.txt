#include<iostream>
using namespace std;
int main()
{
    int to;
    cin>>to;
    while(to>0)
    {
        int n,x,a,b,f=0;
        cin>>n;
        x=a=b=n;
        while(x>0)
        {
            if((x%9==0)||(x%20==0)||(x%6==0))
            {
                f=1;
                break;
            }
            x=x-6;
        }
        if(f!=1)
        {
        while(a>0)
        {
            if((a%9==0)||(a%20==0)||(a%6==0))
            {
                f=2;
                break;
            }
            a=a-9;
        }
        }
        if(f!=1&&f!=2)
        {
        while(b>0)
        {
            if((b%9==0)||(b%20==0)||(b%6==0))
            {
                f=3;
                break;
            }
            b=b-20;
        }
        }
        if(f==1||f==2||f==3)
            cout<<"True\n";
        else
            cout<<"False\n";
        to--;
    }
    return 0;
}
