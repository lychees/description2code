#include <iostream>
#include <algorithm>
#include <set>

using namespace std;

int main() {
	ios::sync_with_stdio(false);
	long t,s,p,c,f,g,pwr,i,j,k,mx=-1;
	cin>>t;
	while(t--)
	{
		mx=-1;
		cin>>f>>g>>s>>p>>c>>pwr;
		for(i=0;i<=f/100;i++)
		for(j=0;j<=min(f/125,g/50);j++)
		for(k=0;k<=min(f/50,g/100);k++)
		{
			if(((100*i+125*j+50*k)<=f)&&((50*j+100*k)<=g))
			mx=max(mx,s*i+p*j+c*k);
		}
		if(pwr>=mx)
		cout<<"-1\n";
		else
		cout<<mx-pwr<<"\n";
	}
	
	
	return 0;
}
