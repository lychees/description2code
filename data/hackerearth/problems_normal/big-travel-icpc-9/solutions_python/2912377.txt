tx = 0
ty = 0
s = 0
x = []
y = []
mod = 1000000007

n = int(raw_input())

for i in range(n):
    a, b = map(int, raw_input().split())
    x.append(a)
    y.append(b)
    
list.sort(x)
list.sort(y)    

for j in range(n):
    tx += ((x[j] - x[j - 1]) * j) % mod
    ty += ((y[j] - y[j - 1]) * j) % mod
    s += tx
    s += ty

print(s % mod)
