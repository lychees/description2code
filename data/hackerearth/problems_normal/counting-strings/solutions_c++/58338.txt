#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int t,len;
char s[1000001];
long long cnt=0,val;
int main()
{
	scanf("%d",&t);
	while(t--)
	{
		cnt=0;
		scanf("%s",s);
		len=strlen(s);
		if(s[len-1]=='a' || s[len-1]=='z')
		{
			cnt=1;
			val=1;
		}
		else
		{
			cnt=0;
			val=0;
		}
		for(int i=1;i<len;i++)
		{
			if(s[len-i-1]=='a' || s[len-i-1]=='z')
			{
				cnt=cnt+i+1;
				val=i+1;
				
			}
			else
				cnt=cnt+val;
		}
		cout<<cnt<<endl;
	}
return 0;
}
