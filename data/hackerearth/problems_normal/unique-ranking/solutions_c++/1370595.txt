#include<iostream>
#include<cstdio>
#include<climits>

using namespace std;

int a[1000][20];
int main()
{
    int n,k,i,j,q,x,l;
    cin>>n>>k;
    for(i=0;i<n;i++)
        for(j=0;j<k;j++)
            cin>>a[i][j];
    cin>>q;
    while(q--)
    {
        cin>>x;
        int sub[x];
        for(i=0;i<x;i++)
            cin>>sub[i];
        bool mark[n];
        for(i=0;i<n;mark[i++]=true);
        for(l=0;l<x;l++)
        {
            j=sub[l]-1;
            int small=INT_MAX;
            for(i=0;i<n;i++)
                if(mark[i])
                    small=min(a[i][j],small);
            for(i=0;i<n;i++)
            {
                if(a[i][j]==small && mark[i])
                    mark[i]=true;
                else
                    mark[i]=false;
            }
        }
        for(i=0;i<n && !mark[i];i++);
        cout<<i+1<<endl;
    }

    return 0;
}
