typedef long long LL;

#define MOD 1000000007
#define MAX 500001
#define pb push_back
#define yes puts("YES")
#define no puts("NO")
#define sd(x) scanf("%d", &x)
#define PI 3.14159265

using namespace std;

LL ans[500001];

int main() {
	for (int i=1; i<MAX; i++) ans[i]=i;
	ans[1]=1;
	for (int i=2; i<MAX; i++) {
		if (ans[i]!=1) {
			for (int j=2*i; j<MAX; j+=i) {
				ans[j]/=ans[i];
			}
			ans[i]=(LL)ans[i]*ans[i-1]%MOD;
		} else ans[i]=ans[i-1];
	}
	
	
	int t;
	cin >> t;
	while (t--) {
		int n;
		sd(n);
		printf("%d\n", ans[n]);
	}
	
	
	return 0;
}
