for i in range(0, T):
    N = int(raw_input())
    count = []
    for j in range(0, N):
        count.append(int(raw_input()))
    count.sort()
    currVal = count[0]
    currCount = 1
    maxCount = 0
    for j in range(1, N):
        if (currVal == count[j]):
            currCount += 1
        else:
            if currCount > maxCount:
                maxCount = currCount
            currCount = 1
            currVal = count[j]
    if currCount > maxCount:
        maxCount = currCount
