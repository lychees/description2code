#include <iostream>
using namespace std;


int main()
{
	int t,a[100],b[100],c[100];
    cin>>t;
    for(int i=0;i<t;i++)
   {
   	cin>>a[i];
   }
	for(int k=0;k<t;k++)
	{
		b[k]=a[k];
		int j=0;
		while(b[k]!=0)
		{
	 	//cout<<b[k]<<endl;
			c[j]=b[k]%2;
			b[k]=b[k]/2;
		
            //cout<<c[j]<<endl;
			j++;
			//cout<<j<<endl;
			if(j>=2)
			{ 
				if(c[j-2]==1 && c[j-1]==1)
				{
					j=0;
					a[k]=a[k]+1;
					b[k]=a[k];
					//cout<<"inside this\n";
				//	goto label;
				}			
		
			}
			
		}
		cout<<a[k]<<endl;
	}
}
