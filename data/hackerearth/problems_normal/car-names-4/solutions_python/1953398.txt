cases = int(raw_input())

for case in range(cases):
	name = raw_input()
	if len(name) % 3 == 0:
		x = []
		for i in name:
			c = name.count(i)
			if (i, c) not in x:
				x.append((i, c))
		if len(x) == 3:
			for i in x:
				c = i[1]
				ind = name.find(i[0])
				if name[ind : ind + c] != i[0] * i[1]:
					print 'Not OK'
					break
			else:
				print'OK'
		else:
			print 'Not OK'
	else:
		print 'Not OK'
