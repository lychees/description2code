#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
typedef pair<int,int> ii;
bool cmp(ii a ,ii b)
{
    if(a.first!=b.first)
        return a.first>b.first;
    return a.second>b.second;
}
int chars[100005];
long long sum;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        sum=0;
        for(int i=0;i<n;i++)
            {
                int a,b;
                scanf("%d%d",&a,&b);
                chars[i]=a+b;
                sum-=b;
            }
        sort(chars,chars+n);

        sum+=chars[n-1]+chars[n-2];
        printf("%lld\n",sum);
    }
}
