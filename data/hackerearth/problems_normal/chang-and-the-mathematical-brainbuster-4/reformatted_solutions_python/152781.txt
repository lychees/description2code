n = int(raw_input())
s = raw_input()
a = sorted([int(x) for x in s.split()])
l = len(a)
mx = 1000000
for i in range(l-1, 0, -1):
    for j in range(0, i+1):
        if a[i]-(3*a[j]) <= 0:
            c = j+l-i-1
            break
        if j+l-i-1 > mx:
            break
    if c < mx:
        mx = c
print mx
