#include <iostream>
#include <cstdlib>
#include <cstdio>
using namespace std;

int compare(const void* x, const void* y){
	return  *(int *)x - *(int *)y;
}

int main()
{
	int t;
	double pd, pwl;		// pd - prob of draw; 	pwl - prob of winning losing
	cin >> t;
	int a[t];
	for(int i=0; i<t; i++){
		cin >> a[i];
	}
	qsort(a, t, sizeof(int), compare);
	double ctr = 1.0;
	for(int i=1; i<t; i++){
		if(a[i]==a[i-1])	ctr+=1.0;
		else {
			pd += (ctr/t)*((ctr-1)/(t-1));
			ctr = 1.0; 
		}
	}
	pwl = (1-pd)/2;
	printf("%1.6f %1.6f", pwl, pwl);
    return 0;
}
