#include <iostream>
#include <math.h>
using namespace std;

void doJob(long A, long B, long M, long& count)
{
	if(A>B || B<A)
	{
		return;
	}
	
	long mid=(B+A)/2;
	if(mid%M==0)
	{
		count++;
	}
	doJob(A, mid-1, M, count);
	doJob(mid+1, B, M, count);
	
}

long doJob(long A, long B, long M)
{
	long first;
	if(A%M==0)
	{
		first=A/M;
	}
	else
	{
		first=ceil(double(A)/double(M));
	}
	
	long second;
	if(B%M==0)
	{
		second=B/M;
	}
	else
	{
		second=ceil(double(B)/double(M))-1;
	}
	
	// cout<<first<<" "<<second<<endl;
	if(first>=second)
	{
		return 0;
	}
	
	return second-first+1;
}

void Read(long numTests)
{
	long A, B, M;
	while(numTests>0)
	{
		cin>>A>>B>>M;
		// cout<<A<<" "<<B<<" "<<M<<endl;
		// long Count=0;
		// doJob(A, B, M, Count);
		// cout<<Count<<endl;
		cout<<doJob(A, B, M)<<endl;
		numTests--;
	}
}

int main()
{
    // cout << "Hello World!" << endl;
    long numTests;
    cin>>numTests;
    Read(numTests);
    return 0;
}
