#include <bits/stdc++.h>
using namespace std;
int gcd(int a,int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}

int main() {
	// your code goes here
	int t,n;
	int a[1002];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(int i=0;i<n;i++)
		 scanf("%d",&a[i]);
		 int s=a[0];
		 for(int i=1;i<n;i++)
		 {
		 	s=gcd(s,a[i]);
		 }
		 cout<<s<<endl;
	}
	return 0;
}
