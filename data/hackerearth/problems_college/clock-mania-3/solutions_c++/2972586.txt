#include <iostream>
using namespace std;

int main()
{
	int h,m,t;
	cin >> t;
	while(t--)
	{
	cin >> h >> m ;
	 if (h == 12) h = 0;
    if (m == 60) m = 0;
    
    int hour_angle = 0.5 * (h*60 + m);
    int minute_angle = 6*m;
    int angle = abs(hour_angle - minute_angle);
    if(angle<=180)
    cout << angle << endl;
    else
    cout << 360-angle << endl;
	}
return 0;
}
