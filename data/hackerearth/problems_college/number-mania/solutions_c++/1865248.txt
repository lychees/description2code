int a[123456];
void fact(int n)
{
    int i,m,temp,x,j;
     a[0]=1;
        m=1;
        i=1;
        if(n==1||n==0)
            printf("1\n");
        else
        {
            while(i<=n)
            {
                temp=0;
                for(int j=0;j<m;j++)
                {
                    x=a[j]*i+temp;
                    a[j]=x%10;
                    temp=x/10;

                }
                while(temp!=0)
                {
                    a[m++]=temp%10;
                    temp/=10;
                }
                i++;
            }

            for(int j=m-1;j>=0;j--)
                printf("%d",a[j]);
            printf("\n");
        }
}
