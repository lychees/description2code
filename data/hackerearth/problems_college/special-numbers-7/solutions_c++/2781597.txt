#include <iostream>
using namespace std;
int isPrime(int n){
	int flag=1;
	for(int i=2;i<n;i++){
		if(n%i==0)
		 flag=0;
	}
	return flag;
}
int isFactor(int num,int fac){
	if(num%fac==0)
	 return 1;
	return 0;
}
int LuckyNumber(int count){
	int num=30;
	int number=30;
	while(count>0){
		int c=0;
	
		for(int i=2;i<num;i++){
			if(isFactor(num,i)&&isPrime(i))
			c++;
			
		}
		if(c>=3){
			count--;
			number=num;
		}
		num++;
	}
	return number;
}
