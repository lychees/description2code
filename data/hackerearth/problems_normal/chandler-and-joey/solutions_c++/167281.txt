#include <iostream>
#include<cstdlib>

using namespace std;
unsigned long long int* arr;

 int gcd(int a, int b)
 {
      if (b==0) return a;
      return gcd(b,a%b);
 }

unsigned long long int getMaximum( unsigned long long int arr[] , int numberOfElements)
{
      unsigned long long int maximum = arr[0];
      for( int i = 1; i < numberOfElements;i++)
      {
            if(maximum < arr[i])
                  maximum = arr[i];
      }

      return maximum;
}
int main()
{

                  p = gcd(arr[i] , p);
