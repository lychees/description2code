#include<stdio.h>
#define m 10000003
int main(){
	long long int t, n, a, b, c, d, i; scanf("%lld", &t);
	while(t-- && scanf("%lld", &n)){
		a = 0, b = 1, c = 1;
		if(n == 0)	printf("0\n");
			else if(n == 1 || n == 2)	printf("1\n");
				else{
					for(i = 3; i <= n; i++)	d = a + b +(i * 2), a = b % m, b = c % m, c = d % m; 
					printf("%lld\n", d % m);
				}
	}		
	return 0;
}
