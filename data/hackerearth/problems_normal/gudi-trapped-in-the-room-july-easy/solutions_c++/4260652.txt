#include <iostream>
#include <vector>
#include <string>
#include <queue>
#include <utility>
#include <map>
using namespace std;
string rev(int h, string s)
{
	for (int i = 0; i < h; i++)
	{
		char c1 = s[0], c;
		for (int j = 0; j < s.size()-1; j++)
		{
			c = s[j+1];
			s[j+1] = c1;
			c1 = c;
		}
		s[0] = c1;
	}
	return s;
}
string add(int a, string n)
{
	for (int i = 1; i < n.size(); i += 2)
	{
		int temp = int(n[i]);
		int temp1 = (temp+a-48)%10;
		temp = temp1 + 48;
		n[i] = char(temp);
	}
	return n;
}
int main()
{
    int t, a, h;
    string s, ans;
    cin>>t;
    for (int i = 0; i < t; i++)
    {
    	map<string, int> m;
    	cin>>s;
    	cin>>a>>h;
    	queue<string> q;
    	q.push(s);
    	ans = s;
    	m[s] = 1;
    	//cout<<rev(h, s)<<endl;
    	//cout<<add(a, s)<<endl;
    	while (!q.empty())
    	{
    		string temp = q.front(); q.pop();
    		if (temp < ans)
    			ans = temp;
    		string temp1 = temp;
    		string temp2 = rev(h, temp), temp3 = add(a,temp1);
    		if (m.count(temp2) < 1)
    		{
    			q.push(temp2);
    			m[temp2] = 1;
    		}
    		if (m.count(temp3) < 1)
    		{
    			q.push(temp3);
    			m[temp3] =1;
    		}
    	}
    	cout<<ans<<endl;
    }
}
