typedef long long int llt;

{
	ios::sync_with_stdio(false);
	//f_in("input.txt");
	int n;
	pair<llt , pair<llt,llt> > p;
	cin>>n;
	vector<llt> v(3);
	map< pair< llt , pair<llt,llt> > , int > mp;
	map< pair< llt , pair<llt,llt> > , int >::iterator it ;
	for(int i = 0 ; i < n  ; i++)
	{
		cin>>v[0]>>v[1]>>v[2];
		sort(v.begin(),v.end());

		p.first = v[0];
		p.second.first = v[1];
		p.second.second = v[2];
		it = mp.find(p);
		if(it == mp.end())
			mp[p]=1;
		else
			it->second++;
	}
	int count = 0 ;
	for(it = mp.begin() ; it != mp.end() ; it++ )
	{
		//cout<<(it->second)<<endl;
		count += (it->second == 1 ? 1 : 0);
	}
	cout<<count;
	return 0;
}
