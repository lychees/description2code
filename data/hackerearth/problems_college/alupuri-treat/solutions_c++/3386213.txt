#include<bits/stdc++.h>

using namespace std;


int main(){
   int a[100][100],n,k,l,m=0,i,j,sum=0;
	cin>>n;
	for( i=0;i<n;i++)
	{
		for( j=0;j<n;j++)
		{
			cin>>a[i][j];
		}
	}

	
	for (i = 0; i < n; ++i)
        {
            sum = sum + a[i][i];
            m = m + a[i][n - i - 1];
        }

	k=m-sum;
	if(k<0)
	k=k*(-1);
	cout<<k;
    return 0;
}
