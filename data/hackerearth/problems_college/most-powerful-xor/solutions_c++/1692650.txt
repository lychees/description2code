#include <iostream>
using namespace std;

int main()
{
	int t,l,r,i,j,cnt;
	cin>>t;
	while(t--)
	{
		cin>>l>>r;
		int max=0;
		for(i=l;i<=r;i++)
		{
			for(j=l;j<=r;j++)
			{
			   cnt=i^j;
			   if(cnt>max)
			   {
			 	  max=cnt;
			   }
			}
		}
		cout<<max<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
