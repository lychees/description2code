template<class T>
ostream& operator << (ostream& os, vector<T> vec){
	for(int i=0; i<vec.size(); i++){
		os << vec[i] << " ";
	}
	return os;
}

#define MOD 1000000007
int main(){
	string s;
	cin >> s;

	vector<long long> dp(3,0);

	for(int i=0; i<s.size(); i++){
		auto dp_ = dp;
		if(s[i]==';'){
			dp_[0] += 1;
			dp_[2] += dp[1];
		}else{
			dp_[1] += dp[1] + dp[0];
		}
		swap(dp, dp_);
		for(int j=0; j<3; j++){
			dp[j] %= MOD;
		}

		//cerr << dp << endl;
	}

	cout << dp[2] << endl;
	return 0;
}
