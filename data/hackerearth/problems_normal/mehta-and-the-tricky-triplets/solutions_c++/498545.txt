#include<stdio.h>
#include<string>
#include<iostream>
using namespace std;
int main()
{
int n;
scanf("%d",&n);
int two[100009]={0},three[100009]={0},five[100009]={0},seven[100009]={0};
int i,j;
string str[n];
 for(i=0;i<n;i++)
 {cin>>str[i];
   for(j=0;j<str[i].size();j++)
     if(str[i][j]==50)
	  two[i]=1;
	  else if(str[i][j]==51)
	  three[i]=1;
	  else if(str[i][j]==53)
	  five[i]=1;
	  else if(str[i][j]==55)
	  seven[i]=1;
}
long long tw=0,th=0,fv=0,sev=0;
long long twth=0,twfv=0,twsv=0,thfv=0,thsv=0,fvsv=0;
long long twthfv=0,twthsv=0,twfvsv=0,thfvsv=0,twthfvsv=0;
for(i=0;i<n;i++)
  {
  tw+=two[i];
  th+=three[i];
  fv+=five[i];
  sev+=seven[i];
  if(two[i]&&three[i])twth++;
  if(two[i]&&five[i])twfv++;
  if(two[i]&&seven[i])twsv++;
  if(three[i]&&five[i])thfv++;
  if(seven[i]&&three[i])thsv++;
  if(seven[i]&&five[i])fvsv++;
  if(two[i]&&seven[i]&&five[i])twfvsv++;
  if(two[i]&&three[i]&&five[i])twthfv++;
  if(two[i]&&seven[i]&&three[i])twthsv++;
  if(three[i]&&seven[i]&&five[i])thfvsv++;
  if(two[i]&&three[i]&&seven[i]&&five[i])twthfvsv++;
  }
long long ans=(tw*(tw-1)*(tw-2))/6+(th*(th-1)*(th-2))/6+(fv*(fv-1)*(fv-2))/6+(sev*(sev-1)*(sev-2))/6;;
  ans-=(twth*(twth-1)*(twth-2))/6+(twfv*(twfv-1)*(twfv-2))/6+(twsv*(twsv-1)*(twsv-2))/6;
  ans-=(thfv*(thfv-1)*(thfv-2))/6+(thsv*(thsv-1)*(thsv-2))/6+(fvsv*(fvsv-1)*(fvsv-2))/6;
  ans+=(twthfv*(twthfv-1)*(twthfv-2))/6+(twthsv*(twthsv-1)*(twthsv-2))/6+(twfvsv*(twfvsv-1)*(twfvsv-2))/6+(thfvsv*(thfvsv-1)*(thfvsv-2))/6;
  ans-=(twthfvsv*(twthfvsv-1)*(twthfvsv-2))/6;

  printf("%lld\n",ans);
  return 0;}
