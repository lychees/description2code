#include<iostream>
#include<vector>
#include<stdio.h>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
       int N;
       scanf("%d",&N);
       vector<int> dist;
       long long int total=0;
       for(int i=0;i<N;i++)
          {
              long long int temp;
              scanf("%lld",&temp);
              total=total+temp;
              dist.push_back(temp);
          }

     long long int reach;
     scanf("%lld",&reach);

     long long int goal=reach%total;
     if(goal==0)
     {
        goal=total;
     }
     int i=0;
     long long int sum=0;
     bool flag=true;
     for(;i<N && flag;i++)
     {
         sum=sum+dist[i];
         if(sum >= goal)
         {
             flag=false;
         }
     }
     printf("%d\n",i);
    }
    return 0;
}
