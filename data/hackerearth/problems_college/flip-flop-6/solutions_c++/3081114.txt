#include <iostream>
using namespace std;

int main()
{
    int t,countt=0;
    string s;
    cin>>t;
    for(int i=0;i<t;i++)
    { cin>>s;
      for(int j=0;j<s.size();j++)
      { if(s[j]==s[j+1])
          {s.erase(s.begin()+j);
          countt++;
          j--;
          } }
      cout<<countt<<endl; 
      countt=0;
    }
    return 0;
}
