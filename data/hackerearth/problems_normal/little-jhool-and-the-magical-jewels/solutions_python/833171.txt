#Little Jhool and the magical jewels!
import sys

t=int(sys.stdin.readline())

while t>0:
	letter={'r':0,'u':0,'b':0,'y':0}
	s=sys.stdin.readline()
	letter['r']=s.count('r')
	letter['u']=s.count('u')
	letter['b']=s.count('b')
	letter['y']=s.count('y')
	print (min(letter[key] for key in letter))
	t-=1
