#include <stdio.h>
#include<iostream>
int main()
{
    int t,n,sum=0,temp;
    scanf("%d",&t);
    
    while(t--)
    {
    	scanf("%d",&n);
    	sum=0;
    	while(n--)
    	{
    	scanf("%d",&temp);
    	sum+=temp;
    	}
    	if(sum%2==0)
    	printf("YES\n");
    	else
    	printf("NO\n");
    }
    return 0;
}
