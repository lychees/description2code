#include<bits/stdc++.h>
using namespace std;
int main()
{
int rev[2000],a[2000],b[2000],c[2000];
int i,j,k,l,flag,cntint,cntfra;
int rem,num1,num3;
float rem1,dno,num2,num4;
char s[2000];
cntint=cntfra=0;
flag=l=0;
rem=0;
scanf("%s",&s);
for(i=0,j=0,k=0;i<strlen(s);i++)
{
if(s[i]=='.')
{
flag=1;
}
else if(flag==0)
a[k++]=s[i]-48;
else if(flag==1)
b[j++]=s[i]-48;
}
cntint=k;
cntfra=j;
for(j=0,i=cntint-1;j<cntint;j++,i--)
{
rem = rem + (a[j] * pow(2,i));
}
for(k=0,i=1;k<cntfra;k++,i++)
{
rem1 = rem1 + (b[k] / pow(2,i));
}
rem1 = rem + rem1;
dno=rem1;
num1 = (int)dno;
num2 = dno - num1;
i=0;
while(num1!=0)
{
rem = num1 % 8;
rev[i] = rem;
num1 = num1 / 8;
i++;
}
j=0;
while(num2!=0.0)
{
num2 =num2 * 8;
num3 = (int) num2;
num4 = num2 - num3;
num2 = num4;
c[j] = num3;
j++;
if(j==4)
{
break;
}
}
l=i;
for(i=l-1;i>=0;i--)
{
printf("%d",rev[i]);
}
for( k=0;k<j;k++)
{
if(k==0)
printf(".");
printf("%d",c[k]);
}
printf(" ");
for(int i=l-1;i>=0;i--)
{
if(rev[i]==0)
cout<<" ";
if(rev[i]==1)
cout<<"A";
if(rev[i]==2)
cout<<"B";
if(rev[i]==3)
cout<<"C";
if(rev[i]==4)
cout<<"D";
if(rev[i]==5)
cout<<"E";
if(rev[i]==6)
cout<<"F";
if(rev[i]==7)
cout<<"G";
}
for(int i=0;i<j;i++)
{
if(i==0)
printf(" ");
if(c[i]==0)
cout<<" ";
if(c[i]==1)
cout<<"A";
if(c[i]==2)
cout<<"B";
if(c[i]==3)
cout<<"C";
if(c[i]==4)
cout<<"D";
if(c[i]==5)
cout<<"E";
if(c[i]==6)
cout<<"F";
if(c[i]==7)
cout<<"G";
}
}
