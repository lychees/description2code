#include<bits/stdc++.h>
using namespace std;

void zFunction(char *S, int *Z){
    int L, R, N;
    N = strlen(S);
    L = R = 0;
    for (int i = 1; i < N; i++){
        if (i > R){
            L = R = i;
            while (R < N && S[R - L] == S[R]) R++;
            Z[i] = R - L;
            R--;
        } else {
            int k = i - L;
            if (Z[k] < R - i + 1) Z[i] = Z[k];
            else {
                L = i;
                while (R < N && S[R - L] == S[R]) R++;
                Z[i] = R - L;
                R--;
            }
        }
    }
}
const int maxn = 1e5+5;
char a[maxn],b[maxn],s[3*maxn];
int T,z[3*maxn];
int main() {
 //   freopen("input.txt","r",stdin);
    scanf("%d",&T);
    while(T--){
        scanf("%s%s",a,b);
        int n = strlen(a);
        for(int i = 0; i < n; ++i)
            s[i] = a[i];
        for(int i = n; i < 2*n; ++i)
            s[i] = b[i-n];
        for(int i = 2*n; i < 3*n; ++i)
            s[i] = b[i-2*n];
        zFunction(s,z);
        int res = 0;
        for(int i = n; i < 2*n; ++i)
            if(z[i] >= n) ++res;
        printf("%d\n",res);
    }
}
