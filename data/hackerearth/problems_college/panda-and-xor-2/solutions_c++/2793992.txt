#include <iostream>
using namespace std;

void init()
{
	long int a,b;
	cin>>a>>b;
	long int x = a^b;
	long int c = 0;
	while(x)
	{
		if(x&1) // lsb == 1
		c++;
		x = x>>1;
	}
	if(c&1)
	cout<<"NO\n";
	else
	cout<<"YES\n";
}
	

int main()
{
    int t;
    cin>>t;
    while(t--)
    init();
    return 0;
}
