#include <iostream>
#include <stack>
#include<cstring>
#include<assert.h>

using namespace std;

int main()
{
 int lines,count =0;
 char str[100001];
 
 cin >> lines;
 while(lines--)
  {
  	stack<char> mystack;
  	cin >>str;
  	int l=strlen(str);
  	assert(l>=2 && l <=100001);
  	 register int i;
  	  for(i=0;i<l;i++)
  	   { 
  	   	 if(!mystack.empty())
  	   	  {
  	   	  	 if(mystack.top()==str[i])
  	   	   	  mystack.pop();
  	   	   	 else
  	   	   	  mystack.push(str[i]);
  	   	  }
  	  
  	       else
  	   	   mystack.push(str[i]);
  	   	}
      if(mystack.empty())
       count++;
  }
  
 cout<<count<<endl;
 
 
    return 0;
}
