#include <iostream>
#include <math.h>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
	int n;
	cin>>n;
	int c=0;
		int a[n];
		for(int i=0;i<n;i++)
		{cin>>a[i];
		if(a[i]>=0)
		c=1;
		
		}
		if(c==0)
		{
		cout<<"-1"<<endl;
	//	return 0;
		}
		else
		{
			long long int sum=0;
		 vector<pair<long long int,int>> v;
		v.push_back(make_pair(0,1));
		for(int i=1;i<=n;i++)
		{
			int num;
		//	cin>>num;
			sum+=a[i-1];
			v.push_back(make_pair(sum,i+1));
		
		}
		sort(v.begin(),v.end());
		long long int count=0,len=0,ans=0;
		int minn=v[0].second;
		for(int i=1;i<=n;i++)
		{
			int val=max(0,v[i].second-minn);
			if(val>len)
			{
			//	ans=p;
				len=val;
				count=1;
				
			}
			else if(len==val)
			count++;
			minn=min(minn,v[i].second);
		}
		cout<<len<<" "<<count<<endl;
		}
	return 0;
}
