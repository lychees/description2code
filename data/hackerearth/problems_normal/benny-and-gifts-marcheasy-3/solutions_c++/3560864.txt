#include <bits/stdc++.h>

using namespace std;

int main() {
	set<pair<int, int> > points;
	pair<int, int> pos = make_pair(0, 0);
	int n_slips = 0;

	string path;
	cin >> path;

	points.insert(pos);

	for (int i = 0; i < path.length(); i++) {
		switch(path[i]) {
			case 'L':
				pos.second -= 1;
				break;

			case 'R':
				pos.second += 1;
				break;

			case 'U':
				pos.first -= 1;
				break;

			case 'D':
				pos.first += 1;
				break;

			default:
				break;
		}
		if (points.find(pos) != points.end()) {
			n_slips += 1;
		}
		else {
			points.insert(pos);
		}
	}

	cout << n_slips << endl;
	return 0;
}
