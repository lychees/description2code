#include <iostream>
#include <cstdio>
#include <cstdlib>
using namespace std;

typedef long long ll;

const ll MOD = (ll)1e9 + 7;
ll add(ll x, ll y)
{
	x += y;
	if (x >= MOD) return x - MOD;
	return x;
}
ll mult(ll x, ll y)
{
	return (x * y) % MOD;
}

const int N = 500500;
int n, k, m;
ll a[N];
ll b[20];

int main()
{
	scanf("%d%d%d", &n, &k, &m);
	for (int i = 0; i < k; i++)
		a[i] = 1;
	while(n--)
	{
		int x;
		scanf("%d", &x);
		x--;
		a[x] = mult(a[x], 2);
	}
	for (int i = 0; i < k; i++)
		a[i]--;
	b[0] = 1;
	for (int i = 0; i < k; i++)
		for (int j = m - 1; j >= 0; j--)
			b[j + 1] = add(b[j + 1], mult(b[j], a[i]));
	ll ans = 0;
	for (int i = 0; i <= m; i++)
		ans = add(ans, b[i]);
	printf("%lld\n", ans);

	return 0;
}
