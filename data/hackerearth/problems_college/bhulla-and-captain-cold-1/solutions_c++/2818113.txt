#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		int m[3],n1[3];
		n1[0]=n1[1]=n1[2]=0;
		int a,b,c;
		for(int i=0;i<n;i++){
			scanf("%d %d %d",&a,&b,&c);
			m[0]=a+min(n1[1],n1[2]);
			m[1]=b+min(n1[0],n1[2]);
			m[2]=c+min(n1[0],n1[1]);
			n1[0]=m[0];n1[1]=m[1];n1[2]=m[2];
		}
		printf("%d\n",min(m[0],min(m[1],m[2])));
	}
   return 0;	
}
