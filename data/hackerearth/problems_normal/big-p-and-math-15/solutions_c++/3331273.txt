#include<iostream>
using namespace std;
int main(){
	long long int a[2],n,i=0,aa,ln[2],hn[2];
	cin>>a[0]>>a[1];
	int c[20];
for(int u=0;u<2;u++){
		ln[u]=0;
		hn[u]=0;
		i=0;
	n=a[u];
	while(n){
		c[i]=n%10;
		n=n/10; 
		i++;
	}
	for(int j=i-1;j>=0;j--){
			if(c[j]==5){
			hn[u]=10*hn[u]+6;
		}else{
		hn[u]=10*hn[u]+c[j];}
	}
	for(int j=i-1;j>=0;j--){
			if(c[j]==6){
			ln[u]=10*ln[u]+5;
		}else{
		ln[u]=10*ln[u]+c[j];}
	}

}
cout<<ln[0]+ln[1]<<" "<<hn[0]+hn[1]<<endl;
}
