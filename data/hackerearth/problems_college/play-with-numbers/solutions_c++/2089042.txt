#include<cstdio>
long long max(long long a,long long b)
{
	return a>b?a:b;
}
long long min(long long a,long long b)
{
	return a<b?a:b;
}
int main()
{
	long long n,m,z,t,m1,n1;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld%lld",&z,&m,&n);
		long long i,ans=0;
		m1=min(m,n);
		n1=max(m,n);
		for(i=1;i*i<=z;i++)
		{
			if(z%i==0&&i<=m1)
			{
				long long temp=z/i;
				if(i!=temp)
				   {
				   	if(temp<=n1)
				   	ans++;
				   	if(i<=n1&&temp<=m1)
				   	ans++;
				   }
				   else if((i<=n1)&&(temp<=m1)&&(i==temp))
				   {
				     ans+=1;
				   }
			    
			}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
