#define slld(x) scanf("%lld",&x);

typedef pair<int,int> PII;

#define N 1000001

int ar[N],cnt[N];
LL sum[N],pwr[N];
LL mod = 1000000007;

void pre()
{
    pwr[0] = 0;
    for(int i=1;i<N;i++)
    {
        pwr[i] = pwr[i-1]*2+1;
        if(pwr[i] >= mod)pwr[i] = pwr[i] - mod;
    }
}

LL power(LL a, LL n,LL m)
{
    LL ans = 1;
    while(n>0)
    {
        if(n&1)ans=(ans*a)%m;
        n>>=1;
        a=(a*a)%m;
    }
    return ans;
}

void init()
{
    for(int i=0;i<N;i++)
    {
        cnt[i] = 0;
        sum[i] = 0;
        ar[i] = 0;
    }
}

void solve()
{
    init();
    int n,mx=0;
    sd(n);
    for(int i=0;i<n;i++)
    {
        sd(ar[i]);
        cnt[ar[i]]++;
        mx = max(mx,ar[i]);
    }
    for(int i=1;i<=mx;i++)
    {
        for(int j=i;j<=mx;j+=i)
        {
            sum[i] += cnt[j];
        }
        //if(sum[i]>0)cout<<"i "<<i<<" "<<sum[i]<<endl;
        sum[i] = pwr[sum[i]];
        if(sum[i]<0)sum[i] = sum[i] + mod;
        //if(sum[i]>0)cout<<"i "<<i<<" "<<sum[i]<<endl;
    }
    LL ans = 1;
    for(int i=mx;i>0;i--)
    {
        for(int j=2*i;j<=mx;j+=i)
        {
            sum[i] = sum[i] - sum[j];
            if(sum[i]<0)sum[i] = sum[i] + mod;
        }
        //ans*=power(i,sum[i],mod);
        //if(ans>=mod)ans%=mod;
            //if(sum[i]>0)cout<<"i "<<i<<" "<<sum[i]<<" "<<ans<<endl;
    }
    for(int i=0;i<n;i++)
    {
        printf("%lld ",sum[ar[i]]);
    }
    printf("\n");
    /*if(ans<0)ans+=mod;
    ans%=mod;
    printf("%d\n",ans);*/
}

int main()
{
    pre();
