bool binarysearch(long long lower,long long upper,long long arr[],long long key)
{
	long long mid=lower+(upper-lower)/2;
	if(lower>upper) return false;
	else
	{
		if(arr[mid]>key)
		{
			upper=mid-1;
			binarysearch(lower,upper,arr,key);
		}
		else if(arr[mid]<key)
		{
			lower=mid+1;
			binarysearch(lower,upper,arr,key);
		}
		else if(arr[mid]==key)
		{
			return true ;
		}
	}
	
}
