#include <iostream>
using namespace std;

int main()
{
    int C, X, Y, count=0, sum=0;
    scanf("%d %d %d", &C, &X, &Y);
    //if(X>C)
    //count=1;
    if(X>Y)
    {
    	while(sum<C)
    	{
    		sum= sum + X;
    		count+=1;
    		if(sum>=C)
    		break;
    		else
    		sum=sum - Y;
    	}
    }
    else
    {
    	while(sum<C)
    	{
    		sum= sum + Y;
    		count+=1;
    		if(sum>=C)
    		break;
    		else
    		sum=sum - X;
    	}
    }
    cout<<count;
    return 0;
}
