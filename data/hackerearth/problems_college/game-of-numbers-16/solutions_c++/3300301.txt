#include <bits/stdc++.h>
using namespace std ;
int main()
{

 int  t;
 cin >> t ;
 while ( t--)
 {
    int n ;
    cin >> n ;
    int i ;
    int j ;
    int r = 0;
    for ( i = 1; i<= n ; i++)
    {
      for ( j = i+1; j <= n ; j++)
      {
         if ( ((i*j)%(i+j) == 0)  )

         {
            r = r+1 ;

         }
      }

    }
    cout << r <<endl;

 }
 return 0 ;

}
