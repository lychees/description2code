#include<bits/stdc++.h>
using namespace std;
#define ll long long

ll a[5000010];
ll sum[1000005];

int main() {
	a[0] = 0;
	a[1] = 1;
	a[2] = 1;
	ll mod = 1000000007;
	sum[0] = 1;
	sum[1] = 1;
	sum[2] = 2;
	for(ll i = 3;i <= 1000003;i++) {
		a[i] = a[i - 2] + a[i - 1];
		a[i] %= mod;
		sum[i] =(mod +  a[i] - 1) % mod;
	//	cout << i << " " << a[i] << " " << sum[i] << "\n";
	}
	ll t;
	scanf("%lld",&t);
	ll i;
	while(t--) {
		ll n;
		scanf("%lld",&n);
		cout << sum[n + 2] << "\n";
	}

}
