#include<bits/stdc++.h>
using namespace std;
char s[1000006];
int n,K,z[1000006];
vector<int>v;
bool check(int len)
{
    for(int i=1;i+len-1<=K;i++)
        if(z[i]>=len)
            return true;
    return false;
}
int main()
{
    #ifndef ONLINE_JUDGE
    freopen("inp.txt","r",stdin);
    #endif // ONLINE_JUDGE
    int t;
    scanf("%d",&t);
    while(t--)
    {
        v.clear();
        scanf("%s%d",s,&K);
        K--;
        for(n=0;s[n];n++);
        memset(z,0,sizeof(z));
        for(int i=1,l=0,r=0;i<n;i++)
        {
            if(i<=r)
                z[i]=min(z[i-l],r-i+1);
            while(i+z[i]<n and s[z[i]]==s[i+z[i]])z[i]++;
            if(i+z[i]-1>r)
                r=i+z[i]-1,l=i;
            if(i+z[i]==n)
                v.push_back(z[i]);
        }
        sort(v.begin(),v.end());
        int l=0,r=v.size()-1,ans=-1,mid;
        while(l<=r)
        {
            mid=(l+r)/2;
            if(check(v[mid]))
                ans=v[mid],l=mid+1;
            else r=mid-1;
        }
        if(ans<=0)printf("Puchi is a cheat!\n");
        else{for(int i=0;i<ans;i++)cout<<s[i];printf("\n");}
    }
    return 0;
}
