#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {

    char a[10],b[10];
    cin>>a>>b;
    int len1,len2;
    len1=strlen(a);
    len2=strlen(b);
       int i=0,j=0;
       int flag=1;
        while(a[j]!='\0'and b[j]!='\0')
        {
            if(a[j]!=b[j])
            {
            	flag=0;
            	i++;
            }
           
            j++;
        }
        if(len1!=len2)
        flag=0;
        if(len2>len1)
            i+=len2-len1;
        else
        {
            i+=len1-len2;
        }
            if(flag==1)
            cout<<"Exact";
            else
            cout<<"Non Exact "<<i;
    cout<<endl;
    }
    return 0;
}
