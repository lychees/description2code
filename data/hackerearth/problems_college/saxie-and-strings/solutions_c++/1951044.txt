#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T;
    string S;
    cin>>T;
    while(T--)
    {
        cin>>S;
        int l=S.length();
        string str="";
        int i=l-1;
        while(i>=0)
        {
            str+=S.at(i);
            i--;
        }
        if(S==str)
            cout<<"YES\n";
        else
            cout<<"NO\n";
        S.clear();
        str.clear();
    }
}
