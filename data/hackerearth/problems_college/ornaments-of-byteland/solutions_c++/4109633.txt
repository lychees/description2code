#include <iostream>
using namespace std;

int main()
{
    int a[26] = {0};
    int flag[26] = {0};
    int t,count=0,i,n;
    string s;
    cin>>t;
    n = t;
    while(t--)
    {
    	cin>>s;
    	for(i=0;i<s.length();i++)
    	{
    		if(!flag[s[i]-'a'])
    		{
    			a[s[i]-'a']++;
    			flag[s[i]-'a'] = 1;
    		}
    	}
    	for(i=0;i<26;i++)
    		flag[i]  = 0;
    }
    for(i=0;i<26;i++)
    {
    	//cout<<a[i];
    	if(a[i]==n)
			count++;
    }
	cout<<count<<endl;
    return 0;
}
