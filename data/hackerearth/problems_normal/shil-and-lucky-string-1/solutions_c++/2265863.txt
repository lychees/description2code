#define FNAME ""

const int MAXN = 1e5 + 5;

char s[MAXN];
int cnt1[MAXN], cnt2[MAXN];

int main() 
{
#ifdef LOCAL    
    freopen(FNAME".in", "r", stdin);
    freopen(FNAME".out", "w", stdout); 
#endif    
    int n;
    scanf("%d\n", &n);
    gets(s);
    forn (i, n / 2)
        cnt1[s[i] - 'a']++;
    forn (i, n / 2)
        cnt2[s[i + n / 2] - 'a']++;
    int good = 0;
    forn (i, 26)
        good += min(cnt1[i], cnt2[i]);
    int ans = n / 2 - good;
    sort(s, s + n / 2);
    sort(s + n / 2, s + n);
    int cnt = 0, it = n / 2;
    forn (i, n / 2)
    {
        while (it < n && s[it] <= s[i])
            it++;
        if (it != n)
            cnt++, it++;
    }
    ans = min(ans, max(n / 2 - cnt, cnt1[25] + cnt2[0]));
    cnt = 0, it = 0;
    forn (i, n / 2)
    {
        while (it < n / 2 && s[it] <= s[i + n / 2])
            it++;
        if (it != n / 2)
            cnt++, it++;
    }
    ans = min(ans, max(n / 2 - cnt, cnt1[0] + cnt2[25]));
    printf("%d\n", ans);

    return 0;
}
