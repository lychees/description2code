#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>
using namespace std;

int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        string str;
        cin>>str;
        bool flag=true;

        for(int i=0; i<str.length(); i++){
            if(str[i]!='R' && str[i]!='Y'){
                flag=false;
                break;
            }
        }

        if(!flag){
            printf("NO\n");
            continue;
        }

        if(str[0]=='Y'){
            printf("NO\n");
            continue;
        }

        size_t found=str.find("YYY");
        if (found!=std::string::npos)
            flag=false;

        if(flag) printf("YES\n");
        else     printf("NO\n");
    }
    return 0;
}
