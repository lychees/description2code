#include <bits/stdc++.h>
using namespace std;

typedef long long LL;

#define PII       pair<int,int>
#define all(c)    c.begin(),c.end()
#define sz(c)     (int)c.size()
#define clr(c)    c.clear()
#define pb        push_back
#define mp        make_pair
#define cin(x)    scanf("%d",&x)
#define MOD       1000000007
#define EPS       1E-10

int val[1000000];

int rankE[1000000];
int wait[1000000];

vector<PII> A;

int main()
{
	int n , m;
	cin(n);
	cin(m);
	for(int i = 1; i <= n; i++)
		cin(val[i]);

	int t = 0;
	for(int i = 1; i <= n; i++)
	{
		rankE[i] = wait[i] = 0;
		A.pb(mp((val[i] + m - 1)/m , i));
		wait[i] = t;
		t += min(val[i] , m);
	}
	sort(all(A));

	for(int i = 0; i < sz(A); i++)
	{
		int idx = A[i].second;
		int val = A[i].first;
		rankE[idx] = i + 1;
	}

	int q;
	cin(q);
	while(q--)
	{
		int a,b;
		cin(a);
		cin(b);
		if(a == 1)
			printf("%d\n", wait[b]);
		else
			printf("%d\n", rankE[b]);
	}
	return 0;
}
