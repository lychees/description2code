#include <iostream>
using namespace std;
 
int main()
{
	long n,i,x=0,y=0,xcount=0,ycount=0;
	char a[1000000];
	cin>>n;
	{
		for(i=0;i<n*1440;i++)
		{
			cin>>a[i];
			if(i%1440==0)
			{
				if(xcount>x)
				x=xcount;
				xcount=0;
			}
			if(a[i]=='C')
				{
					xcount++;
					ycount++;
				}
			else if(a[i]=='S' || a[i]=='E')
			{
					if(xcount>x)
					{
						x=xcount;
					}
					if(ycount>y)
					{
						y=ycount;
					}
					ycount=0;
					xcount=0;
			}
			
		}
		if(xcount>x)
		x=xcount;
		if(ycount>x)
		y=ycount;
		cout<<x<<" "<<y;
	}
    return 0;
}
