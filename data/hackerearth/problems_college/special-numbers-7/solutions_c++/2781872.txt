#include<bits/stdc++.h>
using namespace std;

long long int mod_pow(long long int a, long long int b,long long int mod)
{
	long long int ans=1;
	while(b)
	{
		int r=b%2;
	
		if(r!=0)
		ans=(ans*a)%mod;
		
		a=(a*a)%mod;
		
		b/=2;
	}	
	return ans;
}

long long gcdf(long long a,long long b)
{
	if(a==0)
	return b;
	return gcdf(b%a,a);	
}

long long min1(long long a,long long b)
{
	return a>b?b:a; 	
}

int main()
{
	long long int t;
	cin>>t;
	
	int a[1000],b=0;
	for(int i=2;i<=2600;i++)
	{
		int c=0;
		int j=2,v=i;
		while(v && j<=v)
		{
			if(v%j==0)
			c++;
			
			while(v%j==0)
			{
				v/=j;
			}
			
			j++;
			
			if(c>=3)
			break;
		}
		
		if(c>=3){
			a[b]=i;
			b++;
		}
	}
	
	for(;t>0;t--)
	{
		int n;
		cin>>n;
		cout<<a[n-1]<<"\n";
	}
	return 0;
}
