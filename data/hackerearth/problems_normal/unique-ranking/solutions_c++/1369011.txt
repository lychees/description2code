#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int arr[1001][21];

int main(){
int n,k,q,x,sub;
bool st[1001];
scanf("%d %d",&n,&k);
for(int i=1;i<=n;i++){
    for(int j=1;j<=k;j++){
        scanf("%d",&arr[i][j]);
    }
}
scanf("%d",&q);
while(q--){
    scanf("%d",&x);
    for(int i=1;i<=1000;i++)st[i]=true;
    for(int i=0;i<x;i++){
        scanf("%d",&sub);
        int mi=101;
        for(int j=1;j<=n;j++)if(arr[j][sub]<mi&&st[j]==true)mi=arr[j][sub];
        for(int j=1;j<=n;j++)if(arr[j][sub]!=mi){st[j]=false;}
    }
    int i=1;
    for(;i<=n;i++)if(st[i]==true)break;
    printf("%d\n",i);

}
return 0;
}
