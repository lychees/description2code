#include <iostream>
#include <iomanip>
using namespace std;

double num_ways(long int n,double p1,double p2){
	double p[n+1];
	p[0]=0;
	p[1]=0;
	p[2]=p1;
	p[3]=p2;
	
	long int i=0;
	
	for(i=4;i<=n;i++){
		p[i]=p1*p[i-2]+p2*p[i-3];
	}
	
	
	return p[n];
}


int main()
{
	
    long int n;
    int p;
    
    cin>>n>>p;
    
    double p1=.01*p;
    
    double p2=(100-p)*.01;
    
    double ways=num_ways(n,p1,p2);
    
    std::cout << std::setprecision(6)<< std::fixed;
    cout<<ways;
    
    return 0;
}
