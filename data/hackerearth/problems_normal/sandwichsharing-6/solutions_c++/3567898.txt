#include <iostream>
using namespace std;

class Solution{
	public:
	int findFinalSides(int N, int P){
        if(P>0){
            if(N == 3){
                 return 3;
            }
    		else if(N%2 == 0){
                   N = (N/2)+1;
                   P--;
                   return findFinalSides(N,P);
            }
            else{
                 N = ((N+1)/2)+1;
                 P--;
                 return findFinalSides(N,P);
            }
      }
      else{
           return N;
	}
}
};

int main()
{
	int T;
	cin>>T;
	for(int i=0;i<T;i++){
        int N,P;    
		cin>>N>>P;
		Solution obj;
		int noOfSides = obj.findFinalSides(N,P);
		cout<<noOfSides<<"\n";
	}
    return 0;
}
