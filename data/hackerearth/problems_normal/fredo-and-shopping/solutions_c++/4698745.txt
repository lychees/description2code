#include <bits/stdc++.h>
using namespace std;
unordered_map<int,int> mp;
int arr[1000005];
int brr[1000005];
int bit[1000005];

long long getsum(int sum)
{
	long long ans=0;
	while(sum!=0)
	{
		ans+=bit[sum];
		sum-=((sum)&(-sum));
	}
	return ans;
}

int update(int k)
{
	while(k<1000005)
	{
		bit[k]+=1;
		k+=(k&(-k));
	}
}
int main()
{
	//ios::sync_with_stdio(false);
	int n;
	cin>>n;
	for(int i=0;i<1000005;i++)
	bit[i]=0;
	for(int i=0;i<n;i++)
	{
		scanf("%d",&arr[i]);
		brr[i]=arr[i];
	}
    sort(brr,brr+n);
    int cnt=1;
    mp[brr[0]]=cnt;
    for(int i=1;i<n;i++)
    {
    	if(brr[i]==brr[i-1])
    	continue;
    	else
    	{
    		cnt++;
    		mp[brr[i]]=cnt;
    		
    	}
    	
    }
    int x;
    for(int i=0;i<n;i++)
    {
    
    	int k =mp[arr[i]];
    	cout<<getsum(k-1)<<" ";
    	
    	update(k);
    }
    return 0;
}
