#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	int a[1000001];
    	for(int i=0;i<1000001;i++)a[i]=0;
    	int c=0;
    	for(int i=0;i<n;i++){
    		int x,y;
    		cin>>x>>y;
    	a[x]++;
    	if(a[y]==0)
    	c++;
    	else
    	a[y]--;
    		
    	}
    	cout<<c<<endl;
    }
    
    return 0;
}
