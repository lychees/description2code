#include <iostream>
using namespace std;
int count(int b[],int m,int n)
{
	if(n==0)
	return 1;
	if(n<0)
	return 0;
	if(m<=0 && n>=1)
	return 0;
	return count(b,m-1,n)+count(b,m,n-b[m-1]);
}
int main()
{
	int t,a,i;
	int b[]={1,3,5};
	cin>>t;
	while(t--)
	{
		cin>>a;
		cout<<count(b,3,a)<<"\n";
		
	}
	return 0;
    cout << "Hello World!" << endl;
    return 0;
}
