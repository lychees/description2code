#include <iostream>
#include<math.h>
#include<iomanip>
using namespace std;

int main()
{
    long int a,b,p;
    long double ar,v;
    cin>>a>>b>>p;
    ar=(long double)a*(long double)b*(long double)p/100;
    v=sqrt(ar*(long double)b/(long double)a);
    std::cout << std::fixed ;
    std::cout << std::setprecision(2) << b-v;
    return 0;
}
