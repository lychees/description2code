#include <bits/stdc++.h>
using namespace std;

int main()
{
 	int tc;
 	cin>>tc;
 	long int arr[100000];
 	while(tc--)
 	{
 		long int n,max=1,count=1;
 		cin>>n;
 		for(long int i=0;i<n;i++)
 		{
 			cin>>arr[i];
 		}
 		sort(arr,arr+n);
 		for(long int i=0;i<n-1;i++)
 		{
 			if(arr[i]==arr[i+1])
 			{
 			count++;
 			if(count>max)
 			{
 				max=count;
 			}
 			}
 			else
 			{
 				count=1; 			
 			}
 		}
 		cout<<max<<endl;
 	}
}
