#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int tc;
    cin>>tc;
    while(tc--)
    {
    	int n,k;
    	cin>>n>>k;
    	int c=0;
    	for(int i=0;i<n;i++)
    	{
    		int a;
    		cin>>a;
    		if(a<=0) c++;
    	}
    	if(c>=k) cout<<"NO"<<endl;
    	else cout<<"YES"<<endl;
    }
    return 0;
}
