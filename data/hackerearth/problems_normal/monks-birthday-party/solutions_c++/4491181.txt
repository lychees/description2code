using namespace std;
int main()
{
  std::ios::sync_with_stdio(false);
  int t;
  cin>>t;
  while(t--)
  {
    int i,j,n,m;
    cin>>n;
    assert(1<=n && n<=100000);
    set<string>s;
    rep(i,n)
    {
      string str;
      cin>>str;
      assert(1<=str.size() && str.size()<=100000);
      s.insert(str);
    }

   }

}
