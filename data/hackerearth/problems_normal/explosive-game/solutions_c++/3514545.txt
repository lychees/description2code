#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n, q;
    cin>>n>>q;
    long long s = 0;
    vector<int> a;
    int l, r;
    
    for( int i = 0; i < n; i++ ){
    	cin>>l>>r;
    	if( l > r )	swap( l, r );
    	s += l;
    	a.push_back( r - l );
    }
    
    sort( a.begin(), a.end() );
    
    for( int i = 0; i < a.size(); i += 2 ){
    	if( a[i] != a[i+1] ){
    		s = -1;
    		break;
    	}
    	s += a[i];
    }
    
    for( int i = 0; i < q; i++ ){
    	long long x;
    	cin>>x;
    	if( x == s )	cout<<"YES"<<endl;
    	else	cout<<"NO"<<endl;
    	
    }
    
    return 0;
}
