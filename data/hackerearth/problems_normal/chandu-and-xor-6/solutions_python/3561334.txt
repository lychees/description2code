m, n = map(int, raw_input().split())
lis = []
rows = []
cols = []
xor = []
for i in range(m):
    numbers = map(int, raw_input().split())
    rows.append(sum(numbers))
    lis.append(numbers)
    
#print lis, rows


add  =0
for i in range(n):
    for j in range(m):
       add += lis[j][i]
    cols.append(add)
    add = 0

#print cols

for i in range(m):
    for j in range(n):
        xor.append(rows[i] ^ cols[j])

print max(xor)
