const int N = 1000000;
int prec[N];

ll f0(ll x) {
	if (x <= 2) return 1;
	for (int i = 2; ; ++i) if (x % i) {
		return 1 + f0(i);
	}
}

ll f(ll x) {
	if (x < N) return prec[x];
	for (int i = 2; ; ++i) if (x % i) {
		return 1 + f(i);
	}
}

ll res(ll x) {
	ll sum = 0;
	ll mul = 1;
	for (int i = 2; x; ++i) {
		ll rem = i / __gcd((ll)i, mul);
		sum += (1 + f(i))*(x - x/rem - (i-1) / mul);
		for (ll t = mul; t < i; t += mul) sum += f(t);
		x /= rem;
		mul *= rem;
	}
	return sum;
}

int main() {
	for (int i = 1; i < N; ++i) prec[i] = f0(i);
    ll a, b;
    cin >> a >> b;
    //cout << res(1) << ' ' << res(6) << endl;
    cout << res(b) - res(a-1) << endl;
    return 0;
}
