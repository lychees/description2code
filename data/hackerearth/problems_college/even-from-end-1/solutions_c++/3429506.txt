#include <iostream>
#include <cstdlib>
#include <vector>
#define ll long long

using namespace std;

int main() {
	
	ll t;
	ll temp;
	bool flag = false;
	vector<ll> vec;
	while(cin >> t) {
	for(ll i=0; i<t; i++) {
		cin >> temp;
		vec.push_back(temp);
	}
	
	for(ll i=vec.size()-1; i>=0; i--) {
		if(vec[i]%2==0) {
			cout << vec[i] << ' ';
			flag = true;
		}
	}
	if(flag == true)
		cout << endl;
	else
		cout << "None to display" << endl;
		
	vec.clear();
	flag = false;
	}
	
	return 0;
}
