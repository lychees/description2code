#include <iostream>
#include<string>
using namespace std;
string s;
int f(int i,int j)
{
while(i<j)
{if(s[i]!=s[j])
  break;
  i++;
  j--;
}
if(s[i]==s[j])
return 1;
else
return 0;
}

int main()
{long long pro=1;
    int n,count[26],y;
    cin>>n;
    for(int i=0;i<26;i++)
    count[i]=i+1;
    for(int i=0;i<n;i++)
    {pro=1;
    	cin>>s;
    y=s.length();
    if(f(0,y-1))
    cout<<"Palindrome"<<endl;
    else
    { for(int j=0;j<s.length();j++)
       pro*=count[s[j]-'a'];
       cout<<pro<<endl;
    }
    }
    return 0;
}
