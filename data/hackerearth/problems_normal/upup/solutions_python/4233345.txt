'''

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
import sys

for line in sys.stdin:
	ll = list(line)
	up = True
	for i in range(0, len(ll)):
		if ll[i] != ' ' and up:
			ll[i] = ll[i].upper()
			up = False
		
		if ll[i] == ' ':
			up = True
		
	sys.stdout.write(''.join(ll))
