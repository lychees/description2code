#define MAX 100000001

typedef pair<int,int> pr;

bool isprime[1000002];
int  prime[1000000];

vector<int> arr;

int main()
{
	int t,i,mx,cnt,j,ans,l,r;
	
	
	mx = 1000000;
	memset(isprime,true,sizeof(isprime));
	for(i = 2;i*i <= mx;i++)
	{
		if(isprime[i])
		for(j  = i*i;j <= mx;j = j+i)
		{
			isprime[j] = false;		
				
		}
	}
	cnt = 0;
	for(i = 2;i <= mx;i++)
	{
		if(isprime[i])
		{
			prime[cnt++] = i; 
		}
	}
	

	for(i = 0;i < cnt;i++)
	{
		for(j  =  i+1;(j  < cnt) &&   ((((ll)prime[i])*prime[j]) < MAX);j++)	
		{
			
			arr.pb(prime[i]*prime[j]);		
		}	
	}

	sort(arr.begin(),arr.end());

	scanf("%d",&t);
	vector<int>::iterator  it1,it2;
	
	while(t--)
	{
		scanf("%d %d",&l,&r);
		it1 = lower_bound(arr.begin(),arr.end(),l);
		it2 = upper_bound(arr.begin(),arr.end(),r);
	//	cout<<*it2<<endl;
	//	cout<<*it1<<endl;
	
		printf("%d\n",it2-it1);
	}
	return 0;
}
