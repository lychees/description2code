#include <iostream>
#include <algorithm>
using namespace std;
int a[26];
bool ccomp(char c,char c1)
{
	if(a[c-'a']<a[c1-'a'])
	return true;
	else
	if(a[c-'a']==a[c1-'a'])
	{
		if((c-'a')>(c1-'a'))
			return true;
		else
			return false;
	}
	else
	return false;
}
int main()
{
    int t,i,k;
   scanf("%d",&t);
   char s[1000002];
   char s1[26];
   char s2[26];
   while(t--)
   {
   	scanf("%s",s);
   	
   	for(i=25;i>=0;i--)
   	{
   		s1[25-i]='a'+i;
   	}
   	i=k=0;
   	while(s[i]!='\0')
   	{
   		a[s[i]-'a']++;
   		i++;
   	}
   	for(i=25;i>=0;i--)
   	{
   		if(a[i]>0)
   		{
   			s2[k++]='a'+i;
   		}
   	}
   	sort(s2,s2+k,ccomp);
   	s2[k]='\0';
   	//printf("%s\n",s2);
   	for(i=25;i>=0;i--)
   	{
   		if(a[i]==0)
   		printf("%c ",s1[25-i]);
   		a[i]=0;
   	}
   	for(i=0;i<k;i++)
   	printf("%c ",s2[i]);
   	printf("\n");
   }
    return 0;
}
