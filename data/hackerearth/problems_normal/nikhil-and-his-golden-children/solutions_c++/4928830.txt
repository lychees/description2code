#include <bits/stdc++.h>

using namespace std;

#define ll long long int

ll f[100005];
ll mod=1000000007;

void pre()
{
    ll i;
    f[0]=1;
    for(i=1;i<100005;i++)
        f[i]=(f[i-1]*i)%mod;
}

ll modexp(ll a,ll b)
{
    ll res=1;
    while(b)
    {
        if( b%2 )
            res=(res*a)%mod;
        a=(a*a)%mod;
        b=b/2;
    }
    return res;
}

ll nCr(ll n,ll r)
{
    if( n<r )
        return 0;
    ll ans=(f[n]*modexp(f[r],mod-2))%mod;
    ans=(ans*modexp(f[n-r],mod-2))%mod;
    return ans;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    ll t,x,k,m,i;

    pre();

    cin>>t;
    while(t--)
    {
        cin>>k>>m;
        for(i=0;i<k;i++)
        {
            cin>>x;
            m-=x;
        }
        cout<<nCr(m+k-1,k-1)<<"\n";
    }

    return 0;
}
