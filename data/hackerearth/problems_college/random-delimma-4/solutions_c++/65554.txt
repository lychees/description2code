typedef long long ll;
const ll	MOD = 1000000007LL;
const double PI = 2*acos(0);

int main(void)
{
	std::ios_base::sync_with_stdio(false);

	int t;
	double n, k;
	cin >> t;

	while (t--)
	{
		cin >> k >> n;
		if(k > n)
			cout << fixed << setprecision(9) << 1.0 << endl;
		else{
			double ans = 1.0;
			for(int i=0; i<=k-1; ++i)
				ans *= (n-i)/n;
			ans = 1-ans;
			cout << fixed << setprecision(9) << ans << endl;
		}

	}

	return 0;
}
