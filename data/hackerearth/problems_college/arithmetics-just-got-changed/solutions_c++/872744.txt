#include <iostream>
#include<stdio.h>
using namespace std;

 
int Reverse(int n)
{
	int retVal = 0;
	while(n)
	{
		int r=n%10;
		retVal = retVal*10 + r;	
		n=n/10;
	}
	return retVal;
}
 
int main()
{
	int n;
	cin>>n;
	int p,q,temp;
	while ( n!=0 )
	{	
		scanf("%d%d", &p, &q);
		int a=Reverse(p);
		int b=Reverse(q);
		int d=a+b;
		temp = Reverse(d);
		printf("%d\n", temp);
		n--;
	}
	return 0;
}
