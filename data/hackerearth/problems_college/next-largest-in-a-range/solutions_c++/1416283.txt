#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n,l,r,q;
	cin>>n>>l>>r>>q;
	int a[n];
	for(int i=0;i<n;i++)
	cin>>a[i];
	while(q--)
	{
		int x,p=0,no=0,c=0;
		cin>>x;
		if(l==r)
		{
			if(x<a[l])
			cout<<a[l]<<endl;
			else
			cout<<"-1"<<endl;
		}
		else
		{
		
		int b[(r-l)+1];
		for(int j=l,k=0;j<=r;j++,k++)
		{
			b[k]=a[j];
			c++;
		}
		sort(b,b+c+1);
		for(int m=0;m<c;m++)
		{
			if(b[m]>x)
			{
				p=1;
				no=b[m];
				break;
			}
		}
		if(p==1)
		cout<<no<<endl;
		else
		cout<<"-1"<<endl;
		}
	}
	return 0;
}
