using namespace std;
struct n1
{
    int a,b,c;
} arr[100005];
int n;
void build(int node)
{
    if(node>n/2)
    {
        arr[node].b=0;
        arr[node].c=0;
    }
    else
    {
    build(2*node);
    build(2*node+1);
    arr[node].b=max(arr[2*node].b,arr[2*node].c)+arr[2*node].a;
    arr[node].c=max(arr[2*node+1].b,arr[2*node+1].c)+arr[2*node+1].a;
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(int i=1;i<=n;i++)
        {
            cin>>arr[i].a;
        }
        build(1);
        int ans=INT_MIN;
        for(int i=1;i<=n;i++)
            ans=max(ans,arr[i].a+arr[i].b+arr[i].c);
        cout<<ans<<endl;
    }
}
