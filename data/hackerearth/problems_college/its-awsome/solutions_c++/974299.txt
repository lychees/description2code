#include<iostream>
#include<string.h>
using namespace std ;
int main()
{ int flag;
    int T;

    cin>>T;
    while(T--)
    {
        string name;
        cin>>name;
        flag=1;
        int counter=0;
//cout<<flag<<endl;
        for(int i=0;i<name.size();i++)
        {
            if(name[i]-'0'!=4&&name[i]-'0'!=7)
                {flag=0;
                counter++;

                }
        }




        if(name.size()-counter==4||name.size()-counter==7)
            cout<<"YES\n";
            else
                cout<<"NO\n";

    }
}
