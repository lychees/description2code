t = int(raw_input())
n = []
for i in xrange(t):
	m = int(raw_input())
	n.append(m)
	
def search(x):
	arr = []
	while x>0:
		arr.append(x%10)
		x = x/10
	for i in xrange(len(arr)-1):
		if arr[i]==1 and arr[i+1]==2:
			return True
	return False

for i in xrange(t):
	if n[i]%21==0:
		print "The streak is broken!"
	elif search(n[i]):
		print "The streak is broken!"
	else:
		print "The streak lives still in our heart!"
