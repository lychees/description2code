#include<iostream>
#include<cstdio>
#include<math.h>
using namespace std;
int main(){
	int t;
	cin >>t;
	long long int total=0, i, j;
	for(i=0; i<32; i++)
    		total += (1<<i);
	while(t--){
		long long int n;
		cin >> n;
		long long int a[n];
		for(i=0; i<n; i++){
			cin >> a[i];
		}
		bool flag = false;	
		if(n==1){
			if((a[0] != 0) && (!(a[0]&(a[0]-1)) == 1)){
				cout << "YES\n";
			}else{
				cout << "NO\n";
			}
			continue;
		}
		long long int ans;
		for(i=0; i<32; i++){
			ans = total;
			for(j=0; j<n; j++){
				if(a[j]&(1<<i)){
					ans = ans & a[j];
				}
			}
			if((ans != 0) && (!(ans&(ans-1))==1)){
				flag = true;
				break;
			}
		}
		if(flag){
			printf("YES\n");
		}else{
			printf("NO\n");
		}
	}
	return 0;
}
