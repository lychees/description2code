#include<bits/stdc++.h>
using namespace std;
 
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        int a,b;
        cin >> a >> b;
        int flag = 0;
        while(1)
        {
            if(a % 2 == 0 && b % 2 == 0)
            {
                a = a/2;
                b = a/2;
                flag++ ;
                break;
            }
            if(a % 2 == 0 && b % 2 != 0)
            {
                a = a/2;
                b = a/2;
                flag++;
                break;
            }
            if(a % 2 !=0 && b % 2 == 0)
            {
                a = b/2;
                b = b/2;
                flag++;
                break;
            }
            if(a % 2 != 0 && b % 2 != 0)
            {
                break;
            }
        }
        if(flag % 2 == 0)
        {
            cout << "Nilesh" << endl;
        }
        else
        {
            cout << "Shekhar" << endl;
        }
    }
}
