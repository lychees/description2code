#include<iostream>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)
	cin>>a[i];
	for(int i=0;i<n;i++)
	{
		int c=1;
	  for(int j=0;j<n;j++)
	  {
	  	if(j!=i)
	  	{
	  		if((a[i]%a[j])==0)
	  		{
	  			c=0;
	  		break;
			  }
		  }
	  }
	  if(c==1)
	  cout<<a[i]<<" ";
	}
	return 0;
}
