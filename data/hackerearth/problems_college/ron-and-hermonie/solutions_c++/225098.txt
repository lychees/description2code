int G(int num , long long  power){
	if(num == 1)
	     return 1;
	if(num == 2){
		int arr[] = {2 , 4 , 8 , 6};
		int idx = power % 4;
		return arr[idx];
	}
	if(num == 3){
		int arr[] = {3 , 9 , 7 , 1};
		int idx = power % 4;
		return arr[idx];
	}
	if(num == 4){
		int arr[] = {4,6};
		int idx = power % 2;
		return arr[idx];
	}
	if(num == 5){
		
		return 5;
	}
	if(num == 6){
		
		return 6;
	}
	if(num == 7){
		int arr[] = {7 , 9 , 3 , 1 };
		int idx = power % 4;
		return arr[idx];
	}

		int arr[] = {8 , 4 , 2 , 6};
		int idx = power % 4;
		return arr[idx];
	}
	if(num == 9){
		int arr[] = {9,1};
		int idx = power % 2 ;
		return arr[idx];
	}
	return 0;
}

int main()
{
    int t ;
