#include <iostream>
#include<algorithm>
#include<string>
using namespace std;
int m,n,x,y;
string a[101];
int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	for(int i=0;i<n;i++)
	{
		string copy(a[i]);
		reverse(copy.begin(),copy.end());
		for(int j=i;j<n;j++)
		{
			if(copy==a[j])
			{
				cout<<copy.size()<<" "<<copy[(copy.size()-1)/2];
				return 0;
			}
		}
	}
    return 0;
}
