#define SIZE 32000
bool prime[SIZE+1]={false};
int pr[SIZE],npr=0;

void sieve()
{
	int i,j,it,a;
	int sq = sqrt(SIZE);
	for(i=3;i<=sq;i+=2)
	{
		if(prime[i]==false) 
		{
			a=2*i;
			for(j=i*i;j<=SIZE;j+=a) prime[j]=true;
		}
	}

	pr[++npr]=2;
	for(i=3;i<=SIZE;i+=2)
	{
		if(prime[i]==false) pr[++npr]=i;
	}
	pr[npr+1] = 1e9 + 1e3;
}

int inv(LL b,LL a){
	LL m = a,oo=a;
	int p=1;
	
	while ( p<=npr){
		if ( a % pr[p] ==0){
			m /=pr[p];
			m *= (pr[p]-1);
			while ( a % pr[p]==0 ) a/=pr[p];
		}
		p++;
		if ( pr[p] > a) break;
	}
	if ( a>1){
		m /= a;
		m *= a-1;
	}
	return powmod ( b, m-1, oo);
}
