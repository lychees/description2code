#include <iostream>
#include <string.h>
using namespace std;
int check(char str[200])
{
	int length=strlen(str);
	for(int i=0;i<length/2;i++)
	{
		if(str[i]!=str[length-1-i])
		{
			return 0;
		}
	}
	return 1;
}
int main()
{
	int T;
	cin>>T;
	int ans[10];
	for(int i=0;i<T;i++)
	{
		char str1[100],str2[100],str3[200];
		int count=0;
		cin>>str1>>str2;
		for(int j=0;j<=strlen(str1);j++)
		{
			int m;
			for(m=0;m<j;m++)
			{
				str3[m]=str1[m];
			}
			for(int k=0;k<strlen(str2);k++)
			{
				str3[m++]=str2[k];
			}
			for(int k=j;k<strlen(str1);k++)
			{
				str3[m++]=str1[k];
			}
			str3[m]='\0';
		//	cout<<str3<<"\n";
			if(check(str3))
				count++;
		}
		ans[i]=count;
	}
	for(int i=0;i<T;i++)
		cout<<ans[i]<<"\n";
    return 0;
}
