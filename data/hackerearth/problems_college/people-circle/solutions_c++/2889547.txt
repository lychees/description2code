#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int m,f,k;
		
		cin>>m>>f>>k;
		int a[1000]={0};
		int pos=-1,i;
		for(i=0;i<f;i++)
		{
		
			int cnt = k;
			while(cnt>0)
			{
					pos = (pos+1)%(m+f) ;
				if(a[pos]==0)
				{
					cnt--;
				}
			}
			a[pos]=1;
		}
		for(i=0;i<m+f;i++)
		{
			if(a[i]==0)
			{
				cout<<'M';
			}
			else
			{
				cout<<'F';
			}
		}
		cout<<"\n";
	}
	}
