#include <iostream>
#include<algorithm>
using namespace std;
int main()
{
	long long int n,k,i,c,l,r,flag=0;
	cin>>n;
	long long int a[n];
	for(i=0;i<n;i++)
		cin>>a[i];
	cin>>k;
	sort(a,a+n);
	for(i=0;i<n;i++)
	{
		c=k-a[i];
		l=0;
		r=n-1;
		while(l<r)
		{
			if(l==i)
				l++;
			if(r==i)
				r--;
			if(a[l]+a[r]==c)
			{
				flag=1;
				break;
			}
			else if(a[l]+a[r]<c)
				l++;
			else
				r--;
		}
		if(flag==1)
			break;
	}
	if(flag==1)
		cout<<"YES"<<endl;
	else
		cout<<"NO"<<endl;
	return 0;
}
