using namespace std;
#include<iostream>
#include<string.h>
int main()
{
	int t,i,k,max=0,l,j;
	long n;
	cin>>t;
	for(l=0;l<t;l++)
	{
		cin>>n;
		int x[n],ans=0,a=0;
		for(i=0;i<n;i++)
			cin>>x[i];
		for(k=0;k<n;k++)
		{
			if(x[k]!=-1)
			{
				a=x[k];
				ans++;
				for(j=0;j<n;j++)
				{
					if(a==x[j])
						x[j]=-1;
				}
			}
		}
		cout<<ans<<endl;
	}
}
