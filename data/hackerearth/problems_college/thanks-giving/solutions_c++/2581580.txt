#include <iostream>
#include <stdio.h>
#include <string>
using namespace std;

int main()
{
   // cout << "Hello world!" << endl;
   string s;
    cin >> s;
    int a[26]={0};
    for(int i=0;i<s.size();i++){
        a[s[i]-97]++;
    }
    int max=0;
    char c;
    for(int i=0;i<26;i++){
        if(a[i]>max){
            max = a[i];
           // printf("%d\n",a[i]);
            c=i+97;
        }
    }
//cout << max;
  // cout << c;

    string s1 = "head rush";
    string s2 = "cubing club";
    int flag=0;
    for(int i=0;i<s1.size();i++){
        if(s1[i]==c){
            flag=1;
            break;
        }
    }
    for(int i=0;i<s2.size();i++){
        if(s2[i]==c){
            if(flag==1){
                flag = 3;
            }
            else{
                flag=2;
            }
            break;
        }
    }
    if(flag==1){
        printf("Thanks head rush");
    }
    else if(flag==2){
        printf("Thanks cubing club");
    }
    else if(flag==3){
        printf("Thanks both");
    }
    else{
        printf("none");
    }
    return 0;
}
