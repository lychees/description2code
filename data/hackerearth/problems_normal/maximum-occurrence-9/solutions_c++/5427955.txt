#include <iostream>
#include <map>
using namespace std;

int main()
{
    string s;
    getline(cin , s);
    std::map<char, int>str;
    int n = s.size();
    int max = 0;
    char maxc;
    for(int i=0;i<n;i++)
    {
    	if(str.find(s[i])!=str.end())
    		str.find(s[i])->second++;
    	else
    		str.insert(make_pair(s[i] , 1));
    	if(str.find(s[i])->second >= max)
    		{
    			if(max == str.find(s[i])->second)
    			{
    				if((int)maxc > (int)s[i])
    					maxc = s[i];
    			}
    			else
    				maxc = s[i];
    			max = str.find(s[i])->second;
    		}
    }
    cout<<maxc<<" "<<max;
    return 0;
}
