from sys import stdin
n, m = map(int, stdin.readline().split())
tot = 0
cur = n
ans = 0
while cur:
    tot += cur/m
    cur /= m
for i in xrange(n+1):
    fir = n-i
    sec = i
    u = 0
    v = 0
    while fir:
        u += fir/m
        fir /= m
    while sec:
        v += sec/m
        sec /= m
    # print i,n-i,u,v
    if u+v >= tot:
        ans += 1
print ans
