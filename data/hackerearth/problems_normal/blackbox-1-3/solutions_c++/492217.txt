#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{

    long t;
    scanf("%ld",&t);
    while(t--)
    {
        long a,b,c;
        scanf("%ld%ld%ld",&a,&b,&c);
        long long sa,sb,sc;
        sa=a*a;
        sb=b*b;
        sc=c*c;
        //printf("%lld %lld %lld\n",sa,sb,sc);
        int py=0,ans=0;
        if(sa==sb+sc)
            py=1;
        else if (sb==sa+sc)
            py=1;
        else if(sc==sa+sb)
            py=1;
        if(py==1)
        {
            if(__gcd(a,b)==1)
                {
                    if(__gcd(b,c)==1)
                    {
                        if (__gcd(c,a)==1)
                            ans=1;
                    }
                }
             if(ans==1)
                    printf("YES\n");
                    else
                        printf("NO\n");
        }
    else
        printf("NO\n");

    }
    return 0;
}
