#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
	int T;
	long long int sum,cot,co,N,i,val;
	cin>>T;
	while(T--)
	{
		vector<long int> V;
		sum=0,cot=0,co=0;
		cin>>N;
		for(i=0;i<N;i++)
		{
			cin>>val;
			V.push_back(val);
		}
		std::sort(V.begin(), V.end(), std::greater<long int>());
		for(i=1;i<N;i++)
		{
			sum+=(V[0]-V[i]);
			sum%=1000000007;
			
		}
		cot=sum;
		for(i=1;i<(N-1);i++)
		{
			val=V[i-1]-V[i];
			co=(cot%1000000007-(val*(N-i))%1000000007)%1000000007;
			co=(1000000007+co)%1000000007;
			sum+=co;
			sum%=1000000007;
			cot=co;
		}
		cout<<(sum*V[0])%1000000007<<"\n";
		V.clear();
	}
}
