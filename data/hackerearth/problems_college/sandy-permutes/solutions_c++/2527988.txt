#include <iostream>
using namespace std;

long long int findmax(long long int a[],long long int start,long long int end){
	long long int max=0;
	long long int pos=0;
	for(int i=start;i<end;i++){
		if(max<=a[i]){
			max=a[i];
			pos=i;
		}
	}
	return pos;
}

bool issorted(long long int a[],int n){
	for(int i=1;i<n;i++){
		if(a[i]>a[i-1])
			return false;
	}
	return true;
}

bool greatest(long long int a[],long long int pos,long long int end){
	long long int max=0;
	//long long int pos=0;
	for(int i=pos;i<end;i++){
		if(max<=a[i]){
			max=a[i];
			//pos=i;
		}
	}
	if(max==a[pos])
		return true;
	return false;
}
