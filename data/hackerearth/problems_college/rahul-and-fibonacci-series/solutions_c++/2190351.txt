#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define sc(a) scanf("%lld",&a)
#define  pr(a) printf("%lld\n",a)

long long power(long long a,long long b)
{
long long res=1;
 while(b>0)
 {
 if(b&1)
  res=(res*a)%mod;
  a=(a*a)%mod;
  b>>=1;
 }
 return res;
}
long long nonfibo(long long n)
{
	 	long long a=1,b=2,c=3;
	 	while(n>0)
	 	  {
		   	a=b;
		   	b=c;
			c=a+b;
	 	  	n-=c-b-1;
	 	  }
	 	n+=c-b-1;
	 return (b+n);
}

void fibo(long long int n,long long ans[])
{
	long long a,b,c,d;
    if(n == 0)
    {
        ans[0] = 0;
        ans[1] = 1;
        return;
    }
    fibo((n/2),ans);
    a = ans[0];             
    b = ans[1];             
    c = 2*b - a;
    if(c < 0)
        c += mod;
    c = (a * c) % mod;     
    d = (a*a + b*b) % mod; 
    if(n%2 == 0)
    {
        ans[0] = c;
        ans[1] = d;
    }
    else
    {
        ans[0] = d;
        ans[1] = c+d;
    }
}

int main()
{
   // cout << "Hello World!" << endl;
   long long t;
   sc(t);
   while(t--)
   {
   	long long n;
   	sc(n);
   	long long  ans[2]={0};
   	fibo(n,ans);
   	long long b=ans[0];
   	long long a=nonfibo(n);
   	pr(power(a,b));
   }
    return 0;
}
