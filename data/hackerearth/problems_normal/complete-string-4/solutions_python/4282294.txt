#Panagram
t=input()
while t:
    c=0
    s=raw_input().lower()
    alpha=[]
    alphabet =['a','b','c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','v','w','x','y','z']
    for i in s:
        if(i in alphabet):
            c+=1
            ch=i
            alpha.append(ch)
            del alphabet[alphabet.index(ch)]
    if(c==26):
        print "YES"
    else:
        print "NO"
    t=t-1
    alphabet=alpha+alphabet
