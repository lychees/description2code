#include <bits/stdc++.h>
using namespace std;

int main()
{
    vector<int> v;
    int n,a,temp[1001];
    cin>>n;
    for(int i =0;i<n;i++){
    	cin>>a;
    	v.push_back(a);
    }
    sort(begin(v),end(v));
    for(int i =0;i<n;i++){
    	int t = i/2;
    	temp[t]=v[i];
    	i++;
    	if(i>=n)break;
    	temp[n-1-t]=v[i];
    }
    int max =0;
    for(int i =0;i<n-1;i++){
    	if(max<fabs(temp[i]-temp[i+1]))
    		max = fabs(temp[i]-temp[i+1]);
    }
    if(max<fabs(temp[n-1]-temp[0]))
    		max = fabs(temp[n-1]-temp[0]);
    cout<<max;
    	
    
    
    return 0;
}
