#include <iostream>

using namespace std;
int main()
{
	int a;
	cin>>a;
	
	for (int i=0;i<a;i++){
		for (int j=0;j<a-i-1;j++){
		cout<<" ";
		}
	cout<<"/";
	if (i%2!=0){
		for (int k=0;k<(i/2);k++){
			cout<<" "<<"/";
		}	
		cout<<"  ";
		for (int l=(i/2);l<i;l++){
			cout<<"\\"<<" ";
		}
		cout<<endl;
	}
	else{
		for (int k=0;k<i/2;k++){
			cout<<" "<<"/";
		}
		for (int l=i/2;l<i;l++){
			cout<<"\\"<<" ";
		}		
		cout<<"\\"<<endl;
	}	
	}
	
	
	for (int i=a-1;i>=0;i--){
		for (int j=0;j<a-i+1;j++){
			cout<<" ";
		}
		cout<<"\\";
	if (i%2!=0){
		for (int k=0;k<(i/2);k++){
			cout<<" "<<"\\";
		}	
		cout<<"  ";
		for (int l=(i/2);l<i;l++){
			cout<<"/"<<" ";
		}
		cout<<endl;
	}
	else{
		for (int k=0;k<i/2;k++){
			cout<<" "<<"\\";
		}
		for (int l=i/2;l<i;l++){
			cout<<"/"<<" ";
		}		
		cout<<"/"<<endl;
	}
	
}
for (int i=0;i<a+1;i++){
		for (int j=0;j<a-i;j++){
		cout<<" ";
		}
	cout<<"/";
	if (i%2!=0){
		for (int k=0;k<(i/2);k++){
			cout<<" "<<"/";
		}	
		cout<<"  ";
		for (int l=(i/2);l<i;l++){
			cout<<"\\"<<" ";
		}
		cout<<endl;
	}
	else{
		for (int k=0;k<i/2;k++){
			cout<<" "<<"/";
		}
		for (int l=i/2;l<i;l++){
			cout<<"\\"<<" ";
		}		
		cout<<"\\"<<endl;
	}	
	}
	
	
	for (int i=a;i>=0;i--){
		for (int j=0;j<a-i+2;j++){
			cout<<" ";
		}
		cout<<"\\";
	if (i%2!=0){
		for (int k=0;k<(i/2);k++){
			cout<<" "<<"\\";
		}	
		cout<<"  ";
		for (int l=(i/2);l<i;l++){
			cout<<"/"<<" ";
		}
		cout<<endl;
	}
	else{
		for (int k=0;k<i/2;k++){
			cout<<" "<<"\\";
		}
		for (int l=i/2;l<i;l++){
			cout<<"/"<<" ";
		}		
		cout<<"/"<<endl;
	}
	
}

}
