string = raw_input()

start = 0
upperLimit = len(string) - 1

modifiedString = ""

while (start <= upperLimit):
    ch = string[start]

    if (start == 0 and ch.isalpha()):
        modifiedString += ch.upper()
    elif (start == 0 and ch == ' '):
        modifiedString += ch
    elif (string[start - 1] == ' '):
        modifiedString += ch.upper()
    else:
        modifiedString += ch
    start += 1
print modifiedString
