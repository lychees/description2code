#include <iostream>
#include<algorithm>
#include<cstring>
using namespace std;

int main()
{
    long int t,i,j=0,k;
    cin>>t;
    i=t;
    string s[t];
    for(i=0;i<t;i++)
    {
    	cin>>s[i];
    }
    sort(s,s+t);
    for(i=0;i<t;i++)
    {
    	k=1;
    	for(j=i+1;j<t;j++)
    	{
    		if(s[i]==s[j])
    		{
    			k++;
    		}
    		else
    		{
    			cout<<s[i]<<" "<<k<<endl;
    			i=j-1;
    			break;
    		}
    	}
    	if(j==t)
    	{
    		cout<<s[i]<<" "<<k<<endl;
    		i=j;
    	}
    	
    }
    return 0;
}
