import math

T = int(raw_input())
while T>0:
	t0,t1,t2,v1,v2,d = map(int,raw_input().split())
	T1 = int(math.ceil(t1 + (float(d)/v1)*60))
	T2 = int(math.ceil(t2 + (float(d)/v2)*60))
	if(t0>t1 and t0>t2):
		print '-1'
	elif(t0<=t1 and t0>t2):
		print T1
	elif(t0<=t2 and t0>t1):
		print T2
	else:
		print min(T1,T2)
	T = T-1
