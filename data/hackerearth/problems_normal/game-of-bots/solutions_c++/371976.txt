using namespace std;
int A[1000009];
int B[1000009];
int main() 
{
	int t;
	cin>>t;
	while(t--)
	{
		string s1,s2;
		cin>>s1>>s2;
		int x=s1.length();
		int y=s2.length();
		if(x!=y)
		{
			cout<<"No"<<endl;
		}
		else
		{
			string a="",b="";
			int p=0,q=0,a1=0,b1=0;
			for(int i=0;i<x;i++)
			{
				if(s1[i]=='#')
				{
					p++;
				}
				else
				{
					a+=s1[i];
					A[a1]=i;
					a1++;
				}
				if(s2[i]=='#')
				{
					q++;
				}
				else
				{
					b+=s2[i];
					B[b1]=i;
					b1++;
				}
			}
			if(p==q && a.compare(b)==0)
			{
				bool f1=false;
				
				for(int i=0;i<a.length();i++)
				{
					if(a[i]=='A' && A[i]<B[i])
					{
						f1=true;
						break;
						
					}
					else if(a[i]=='B' && A[i]>B[i])
					{
						f1=true;
						break;
					
					}
				}
				if(f1==true)
				{
					cout<<"No"<<endl;
				}
				else cout<<"Yes"<<endl;
			}
			else
			{
				cout<<"No"<<endl;
			}
		}
		
		
	}
   
    return 0;
}
