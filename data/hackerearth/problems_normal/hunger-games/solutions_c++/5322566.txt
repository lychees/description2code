#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,max,i,j;
    scanf("%d",&n);
    int a[n];
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    sort(a,a+n);
    max=0;
    if(n%2!=0)
    {
    	if(a[1]-a[0]>max)
    	max=a[1]-a[0];
    	if(a[2]-a[0]>max)
    	max=a[2]-a[0];
    	for(j=1;j<n-2;j++)
    	{
    		if(a[j+2]-a[j]>max)
    		max=a[j+2]-a[j];
    	}
    	if(a[n-1]>a[n-2])
    	max=a[n-1]-a[n-2];
    }
    else
    {
    	if(a[1]-a[0]>max)
    	max=a[1]-a[0];
    	if(a[2]-a[0]>max)
    	max=a[2]-a[0];
    	for(j=1;j<n-3;j++)
    	{
    		if(a[j+2]-a[j]>max)
    		max=a[j+2]-a[j];
    	}
    	if(a[n-1]-a[n-2]>max)
    	max=a[n-1]-a[n-2];
    	if(a[n-1]-a[n-3]>max)
    	max=a[n-1]-a[n-3];
    }
    printf("%d",max);
    return 0;
}
