#include<bits/stdc++.h>
using namespace std;
int main()
{
	char a[102],ans;
	int l,i,j;
	gets(a);
	l=strlen(a);
	int s=0,max=INT_MIN;
	int fl[256]={0};
	for(i=0;i<l;i++)
	{
		s=s+a[i];
		fl[a[i]]++;
		if(fl[a[i]]>max)
		{
			max=fl[a[i]];
			ans=a[i];
		}
	}
	//cout<<l<<endl;
	//cout<<s/l;
	int x=s/l;
	if(x%2==1)
	{
		cout<<ans<<endl;
	}
	else
	{
		i=0;
		j=l-1;
		char temp;
		while (i < j) 
		{
			temp = a[i];
			a[i] = a[j];
			a[j] = temp;
			i++;
			j--;
		}
		cout<<a<<endl;
	}
	return 0;
}
