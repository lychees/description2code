#include <iostream>
using namespace std;
int ar[704980];
int main()
{
	int i ,j;
	for(i=0;i<704980;i++)
		ar[i]=0;
	for(i=1;i<90;i++)
	{
		for(j=i;j<90;j++)
		{
			int x = i*i*i;
			int y = j*j*j;
			if((x+y)<704980)
				ar[x+y]++;
		}
	}
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int fl=0;
		for(i=n;i>0;i--)
		{
			if(ar[i]>1)
			{
				fl=1;
				cout<<i<<endl;
				break;
			}
		}
		if(fl==0)
		cout<<-1<<endl;
	}
    return 0;
}
