#include <iostream>
#include <cstring>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
    char s[500009];
    scanf("%s",&s);
    string str(s);
    int prefix[500009]={0};
    int n=strlen(s);
    char temp[12]="hackerearth";
    temp[11]='\0';
    string req(temp);
    sort(req.begin(),req.end());
    for(int i=0;i+11<=n;i++)
    {
    char tmp[12];
    for(int j=i;j<i+11;j++)
    {
    if(str[j]<='z' && str[j]>='a')
    tmp[j-i]=str[j];
    else
    tmp[j-i]=char(str[j]-'A'+'a');
    }
    tmp[11]='\0';
    string sr(tmp);
    sort(sr.begin(),sr.end());
    if(sr.compare(req)!=0)
    {
    if(i==0)
    prefix[i]=0;
    else
    prefix[i]=prefix[i-1];
    }
    else
    {
    if(i==0)
    prefix[i]=1;
    else
    prefix[i]=prefix[i-1]+1;
    }
    }
    for(int i=n-10;i<n;i++)
    prefix[i]=prefix[i-1];
    int q;
    scanf("%d",&q);
    while(q--)
    {
    int l,r;
    scanf("%d%d",&l,&r);
    l--;r--;
    int up=0,down=0;
    if(r-10>=0)
    up=prefix[r-10];
    if(l-1>=0)
    down=prefix[l-1];
    if(up==0 || up<down)
    printf("%d\n",0);
    else
    printf("%d\n",up-down);
    }
    return 0;
}
