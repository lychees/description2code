#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,n,t,l,len;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int num[n+9],stk[n+9];
		for(i=0;i<n;i++)
		scanf("%d",&num[i]);
		scanf("%d",&len);
		l=1;
		stk[0]=num[0];
		for(i=1;i<n;i++)
		{
			if(num[i]<stk[0])
			stk[0]=num[i];
			else if(num[i]>stk[l-1])
			{
				if(l<=len)
				{
					stk[l]=num[i];
					l++;
				}
			}
			else if(num[i]<stk[l-1])
			{
				int it=lower_bound(stk,stk+l,num[i])-stk;
				stk[it]=num[i];
				
			}
			
		}
		//for(i=0;i<l;i++)
		//cout<<stk[i]<<endl;
		if(l>=len)
		printf("%d\n",stk[len-1]);
		else
		printf("-1\n");
		
	}
	return 0;
}
