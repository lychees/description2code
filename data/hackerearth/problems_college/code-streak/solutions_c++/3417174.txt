#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,n,b,flag,max,i;
	long long int s;
	long long int tmp[1000000];
	cin>>t;
	while(t--)
	{
		cin>>n;
		flag=0;
		max=0;
		for(i=0;i<1000000;i++)
			tmp[i]=0;
		for(i=0;i<n;i++)
		{
			cin>>s;
			cin>>b;
//			tmp[s]=b;
			if(b==1) {
			if(tmp[s]!=1) {
//				tmp[s]=b;
				flag++;
				if(flag>max)
				max=flag;
				}
				tmp[s]=b;
			}
			else if(b==0) {
				 if(tmp[s]!=0 || tmp[s]!=1)
				{
				flag=0;
				}
			}
			
		}
		cout<<max<<endl;
	}
	return 0;
}
