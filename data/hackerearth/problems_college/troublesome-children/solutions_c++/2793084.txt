vector<int> arr;
int mark[40] = {0};

int k;

int getPos(int len , int now)
{
	int i = 0;
	while(i < len)
	{
		while(mark[now])
		{
			now = (now + 1) % k;
		}
		i += 1;
		if(i == len) return now;
		now = (now + 1) % k;
	}
	return -1;
}

int main()
{
	cin >> k;
	int curr = 0;
	for(int i = 0; i < k; i++)
	{
		while(mark[curr])
		{
			curr = (curr + 1) % k;
		}
		int x;
		cin >> x;
		int pos = getPos(x , curr);
		mark[pos] = 1;
		arr.pb(pos);
		curr = (pos + 1) % k;
	}
	FOR(i,0,sz(arr)) cout << arr[i] + 1 << endl;
	return 0;
}
