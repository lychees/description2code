#include <iostream>
#include<algorithm>
#include<vector>
#include<stdio.h>
using namespace std;

struct match{
    int st;
    int et;
    double prob;
};	

bool cmp(const match &a, const match &b){
	return (a.st < b.st);
}

double max(double a, double b){
	if(a > b){
		return a;
	}
	return b;
}

double lotteryExpectation(vector<match> &v){
	if(v.size() == 0){
		return 0.00;
	}
	sort(v.begin(), v.end(), cmp);
	int sz = v.size();
	vector<double> value(v.size(), 0.00);
	value[sz-1] = v[sz - 1].prob;
	for(int i = v.size()-2; i >=0; i--){
		int j = i+1;
		while(j < sz){
			if(v[j].st >= v[i].et){
				break;
			}
			j++;
		}
		value[i] = max(v[i].prob + value[j], value[i+1]);
	}
	return value[0];
}

int main()
{
    int t;
    cin>>t;
    //cout<<t;
    for(int i = 0; i < t; i++){
    	int n;
    	cin>>n;
    	vector<match> v;
    	for(int j = 0; j < n; j++){
    		int st, et, c;
    		cin>>st;
    		cin>>et;
    		cin>>c;
    	
    		match m;
    		m.st = st;
    		m.et = et;
    		m.prob = (double)c/100;
    		v.push_back(m);
    	}
    	//cout << setprecision (2) <<lotteryExpectation(v);
    	printf("%.2f", lotteryExpectation(v));
    	//printf("%.2f\n", diff); cout << setprecision (2) << diff << endl;
    	cout<<"\n";
    }
    return 0;
}
