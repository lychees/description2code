#include<bits/stdc++.h>

using namespace std;

int main(){
    int t,i,k = 0;
    cin >> t;
    long long sum = 0, temp;
    long long arr[t];
    for(i = 0; i < t; i++){
        cin >> arr[i];
    }
    cin >> temp;
    sort(arr,arr+t);
    for(i = t-1; i >= 0; i--){
        sum += arr[i];
        k++;
        if(sum >= temp){
            cout << k << endl;
            break;
        }
    }

}
