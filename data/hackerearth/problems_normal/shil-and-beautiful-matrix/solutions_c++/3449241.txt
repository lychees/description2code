# include <iostream>
# include <cmath>
# include <queue>
# include <vector>
# include <stdio.h>
# include <set>
using namespace std;
typedef long long ll;

ll fact[1002];
int freq[1025];

int main()
{
    ll n,m,mod = 1000000007;
    cin>>n>>m;
    
    fact[0] = 1;
    for (ll i=1;i<=1000;i++)   fact[i] = (fact[i-1]*i) % mod ;
    
    fill(freq,freq+1025,0); 
    
    for (ll i=1;i<=n;i++)
        for (int j=1;j<=m;j++)
            freq[i^j]++;
    
    ll res = 1;
    for (int i=0;i<1025;i++)
        if (freq[i] > 0)
            res = (res * fact[freq[i]]) % mod;
    
    cout<<res<<endl;
    return 0;
}
