#include <bits/stdc++.h>
using namespace std ;

int T,N,M[26],ans;
string str ;

int main(){

	scanf("%d",&T) ;
	while(T--){
		cin >> str ;
		N = str.length() ;
		memset(M,0,sizeof M) ;
		int s , e ;
		ans = s = e = 0 ;
		while(e<N){
			while( e < N && M[str[e]-'A'] == 0){
				M[str[e]-'A'] = 1 ;
				e ++ ;
			}
			ans = max(ans,e-s) ;
			M[str[s]-'A'] -- ;
			s ++ ;
		}
		printf("%d\n",ans) ;
	}
	return 0 ;
}
