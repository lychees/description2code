#include<iostream>
#include<string>
using namespace std;

int main(){
    int t;
    long long int n, sum, a, b, c;
    cin>>t;
    while(t--){
        cin>>n;
        sum=0;
        a=0;
        b=1;
        c=0;
        for(;(a+b)<=n;){
            c=a+b;
            a=b;
            b=c;
            if(c%2==0)
                sum+=c;
        }
        cout<<sum<<endl;
    }
    return 0;
}
