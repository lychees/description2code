#include <iostream>
using namespace std;


int find_hcf(int a,int b)
{
	if(a%b==0)
	{
		return b;
	}
	else
	{
		return find_hcf(b,a%b);
	}
}

int main()
{
    int val,ang,n,k,temp=360;
    cin>>n;
    cin>>k;
    for(int i=0;i<n;i++)
    {
    	cin >> ang ;
    	temp = find_hcf(temp,ang);
    }
    for(int i=0;i<k;i++)
    {
    	cin >> val;
    	if(val % temp )
    	{
    		cout<<"NO"<<endl;
    	}
    	else
    	{
    		cout<<"YES"<<endl;
    	}
    }
    return 0;
}
