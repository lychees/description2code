#include <iostream>
using namespace std;
#define ll long long
#include <vector>

int main()
{	ll t;
    cin>>t;
    while(t--)
    {	ll a,b,q,c=0,k=0,i;
    	cin>>a>>b;
    	vector<ll> z;
    	for(i=0;i<a;++i)
    	{	cin>>q;
    		z.push_back(q);
    	}
    	for(i=0;i<a;++i)
    		if(z[i]<=0)
    		{	++c;
    			if(c>=b)
    			{	k=1;
    				cout<<"NO\n";
    				break;
    			}
    		}
    	if(!k)
    		cout<<"YES\n";
    }
    return 0;
}
