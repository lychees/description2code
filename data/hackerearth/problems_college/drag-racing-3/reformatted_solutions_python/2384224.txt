import sys
data = sys.stdin.readlines()
N = data[0]
N = int(N)
winner = {}
for i in xrange(1, len(data)):
    a, b = data[i].split()
    if a in winner:
        winner[a] = winner[a]+1
    else:
        winner[a] = 1
for j in winner:
    if winner[j] == N:
        print j
