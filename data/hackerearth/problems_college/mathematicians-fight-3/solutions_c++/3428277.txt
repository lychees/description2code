#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    while(n--)
    {
        long x,i;
        long long p;
        cin>>x;
        for(i=0;;i++)
        {
            p = pow(2,i);
            if(p>x)
            {
                p = pow(2,i-1);
                break;
            }
        }
        long long ans = x-p;
        cout<<2*ans+1<<endl;
    }
    return 0;
}
