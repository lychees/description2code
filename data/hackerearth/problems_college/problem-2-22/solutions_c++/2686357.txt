#include <bits/stdc++.h>
using namespace std;
int main()
{	int t;
	scanf("%d",&t);
	for(int j=0;j<t;j++)
	{
	double r,a,b;
//	cin>>r>>a>>b;
	scanf("%lf",&r);
	scanf("%lf",&a);
	scanf("%lf",&b);
	double x,y;
	int c=0;

	for(int i=0;i<4;i++)
	{	
		scanf("%lf",&x);
		scanf("%lf",&y);
//		x=pow(x-a,2);
//		y=pow(x-a,2);
		if(pow(x-a,2)+pow(y-b,2)-r*r<=0)
		c++;
	}
	if(c==4	)
	printf("Yes\n");
	else
	printf("No\n");
	}
}
