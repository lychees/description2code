n = input()
a = list(map(int, raw_input().split()))

cnt = [0 for i in xrange(100500)]
for i in xrange(2, 100500):
    cur, j = i, 2
    while j * j <= cur:
        while cur % j == 0:
            cnt[i] += 1
            cur //= j
        j += 1
    if cur > 1:
        cnt[i] += 1

ans = 0
for x in a:
    ans ^= cnt[x]

print 'BOB' if ans == 0 else 'ALICE'
