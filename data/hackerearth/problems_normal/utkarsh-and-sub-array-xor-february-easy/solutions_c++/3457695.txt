ll mod= 1e9 + 7;

ll n,k;
map <ll,ll> dad;

ll m_p(ll base, ll exponent)
{  ll modulus=mod;
    ll result = 1;
    while (exponent > 0)
    {
        if (exponent % 2 == 1)
            result = (result * base) % modulus;
        exponent = exponent >> 1;
        base = (base * base) % modulus;
    }
    return result;
}




ll root(ll v)
{
    if (dad.find(v)==dad.end() || v == dad[v]) return v;
    return dad[v] = root(dad[v]);
}



int main()
{   io;
  
cin>>n>>k;
ll ans=m_p(2,n);
while(k--){
    ll x,y;
    cin>>x>>y;
    x--;
    ll tmp1=root(x),tmp2=root(y);
    if(tmp1!=tmp2){
        dad[tmp2]=tmp1;
        ans*=(mod+1)/2;
        ans%=mod;
    }
    cout<<ans<<endl;
    
}

    return 0;

}
