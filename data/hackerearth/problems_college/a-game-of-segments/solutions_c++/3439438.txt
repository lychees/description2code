ll dp[55][500][500];
ll solve(ll in,ll curr,ll pre)
{
    if(in==n && curr >= pre)
    {
        return 1;
    }
    if(in==n)
    {
        return 0;
    }
    if(dp[in][curr][pre]!=-1) return dp[in][curr][pre];
    ll ret = 0;
    if(in==0)
    {
        dp[in][curr][pre] = solve(in+1,a[in],0);
        return dp[in][curr][pre];
    }
    ret += solve(in+1,curr+a[in],pre);
    if(curr>=pre)
    {
        ret += solve(in+1,a[in],curr);
    }
    dp[in][curr][pre] = ret;
    return ret;
}
