#include <bits/stdc++.h>
using namespace std;

bool cmp(const pair<int,int> &left, const pair<int,int> &right){
	if(left.first < right.first) return true;
	if(left.first == right.first && left.second < right.second) return true;
	return false;
}
int main()
{
	int T;
	cin >> T;
	while(T--){
		int m,n,g,h;
		scanf("%d%d",&m,&n);
	//	vector<int> u(m+1),v(m+1),w(n+1),x(n+1);
		vector<pair<int,int> > novel(m);
		vector<pair<int,int> > old(n);
		int a,b;
		for(int i=0;i<m;i++){
			scanf("%d%d",&a,&b);
			novel[i] = make_pair(a,b);
		}
		for(int i=0;i<n;i++){
			scanf("%d%d",&a,&b);
			old[i] = make_pair(a,b);
		}
		sort(novel.begin(),novel.end(),cmp);
	//	for(int i=0;i<m;i++){
	// /		cout << novel[i].first << " " << novel[i].second << endl;
		//}
		sort(old.begin(),old.end(),cmp);
		scanf("%d%d",&g,&h);
		int i=0,k=0;
		int good=0,great=0;
		while(i<m && k<n){
			while(novel[i].first > old[k].first){
				k++;
			}
			if(novel[i].first == old[k].first){
				if(novel[i].second == old[k].second){
					great++;
				}
				else good++;
				k++;
			}
			i++;
		}
		if(great>=h and good>=g) printf("Great\n");
		else if(good>=g) printf("Good\n");
		else printf(":(\n");
	}
   // cout << "Hello World!" << endl;
    return 0;
}
