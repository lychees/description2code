#include <cstdio>
#include <algorithm>

#define NMAX 600000
#define _scan_d(var) scanf("%d", &var)

int min_cnt[NMAX];

int main(int argc, char *argv[]) {
    int i, j, min;

    for (i = 1; i*i < NMAX; ++i)
        min_cnt[i*i] = 1;

    for (i = 2; i < NMAX; ++i) {
        if (min_cnt[i]) continue;

        min = 1 + min_cnt[i-1];
        for (j = 2; j*j < i; ++j) {
            if (min == 2) break;

            min_cnt[i] = 1 + min_cnt[i - j*j];
            if (min_cnt[i] < min)
                min = min_cnt[i];
        }
        min_cnt[i] = min;
    }

    std::_scan_d(i);
    while (i--) {
        std::_scan_d(j);
        std::printf("%d\n", min_cnt[j]);
    }

    return 0;
}
