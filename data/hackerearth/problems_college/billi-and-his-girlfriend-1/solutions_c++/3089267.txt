#include <bits/stdc++.h>
using namespace std;
#define ll long long 
ll pow1(ll a,ll b,int mod)
{
	ll ans=1;
	while(b)
	{
		if(b&1)ans=(ans*a)%mod;
		a = (a*a)%mod;
		b/=2;
	}
	return ans;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	ll a,b;
    	int k;
    	cin>>a>>b>>k;
    	int mod = 1;
    	while(k--)mod*=10;
    	ll temp = pow1(a,b,mod);
    	ll sum = 0;
    	while(temp){sum=sum+temp%10;temp/=10;}
    	cout<<sum<<endl;
    }
    return 0;
}
