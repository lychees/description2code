#include <bits/stdc++.h>

typedef long long ll;
#define MOD 1000000007
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define MAXN 100005

using namespace std;
ll correct[MAXN][2];
ll incorrect[MAXN][2];

int main()
{
    ios_base::sync_with_stdio(0);cin.tie(0);
    int t,i,n;
    incorrect[1][0]=1;
    incorrect[1][1]=1;
    for(i=2;i<=MAXN;i++)
    {
        correct[i][0]=(incorrect[i-1][0]+correct[i-1][1])%MOD;
        incorrect[i][0]=incorrect[i-1][1]%MOD;
        incorrect[i][1]=(incorrect[i-1][0]+incorrect[i-1][1])%MOD;
        correct[i][1]=(correct[i-1][0]+correct[i-1][1])%MOD;
    }
    cin>>t;
    while(t--)
    {
        cin>>n;
        cout<<(correct[n][0]+correct[n][1])%MOD<<'\n';
    }
    return 0;
}
