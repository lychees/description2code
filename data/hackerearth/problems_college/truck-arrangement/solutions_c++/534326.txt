#include <bits/stdc++.h>

using namespace std;

struct pack
{
    int data;
    int index;
};

bool comp(struct pack a1, struct pack a2)
{
    if(a1.data!=a2.data)
    {
        return a1.data<a2.data;
    }
    else
    {
        return a1.index<a2.index;
    }
}

int main()
{
    int t;
    scanf("%d",&t);
    //cin.ignore();

    while(t--)
    {
        int n;
        cin>>n;
        struct pack arr[n];

        for(int i=0;i<n;i++)
        {
            cin>>arr[i].data;
            arr[i].index = i;
        }

        sort(arr, arr+n, comp);

        int ans = -1;

        for(int i=0;i<n;i++)
        {
            //cout<<arr[i].data<<" "<<arr[i].index<<endl;
            ans = max(ans, abs(arr[i].index - i));
        }


        cout<<ans<<endl;
    }
    return 0;
}
