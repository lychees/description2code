ll modulo(ll a,ll b) {
	ll c = 1e9 + 7;
    long long x=1,y=a;
    while(b > 0){
        if(b%2 == 1){
            x=(x*y)%c;
        }
        y = (y*y)%c;
        b /= 2;
    }
    return x%c;
}

ll palin(ll n){

    ll m=n;
    ll a = 0;
    while (m > 0) {
        a = (m % 10 )+ a * 10;
        m = m / 10;
    }
    if(n == a)
    	return 1;
    return 0;
}

int main() {
	ll i;
	ll a,b;
	S(a);
	S(b);
	set<ll> s;
	for(i = 1;i <= 10000;i++) {
		ll sum = i*i;;
		for(ll j = i + 1 ;j <= 10000;j++) {
			sum  += (j*j);
			if(sum > b) {
				break;
			}
			if(palin(sum) == 1 && sum >= a && sum <= b) {
				s.insert(sum);
			} 
		}
	}
	ll ans = 0;
	set<ll>::iterator it;
	for(it = s.begin();it != s.end();it++) {
		ans += (*it);
	} 
	cout << ans << "\n";
}
