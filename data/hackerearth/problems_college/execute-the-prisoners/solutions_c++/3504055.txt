#include<iostream>
#include<cstdio>
#include<bitset>
using namespace std;
typedef long long ll;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		scanf("%d",&n);
		bitset<16> foo(n);
		if(foo.count()==1){
			cout<<"1\n";
		}
		else{
			int temp=2;
			int prev=2;
			while(temp<n){
				prev=temp;
				temp=temp*2;
			}
			printf("%d\n",(n-prev)*2+1);
		}
	}
}
