#include <iostream>
using namespace std;
int a[101][101],n,cnt,b[101][101];
bool visited[101][101];
void dfs(int x,int y)
{
	visited[x][y]=true;
	if(x==n-1 && y == n-1)
	{
		cnt++;
		visited[x][y]=false;
		return;
	}
	//cout<<x<<' '<<y<<'\n';
	if(x>0 && !visited[x-1][y]&&!a[x-1][y])
	{
		dfs(x-1,y);
		visited[x-1][y]=false;
	}
	
	//cout<<'1';
	if(x<n-1 && !visited[x+1][y]&& !a[x+1][y])
	{
		dfs(x+1,y);
		visited[x+1][y]=false;
	}
	
	//cout<<'2';
	if(y>0 && !visited[x][y-1]&& !a[x][y-1] )
	{
			dfs(x,y-1);
			visited[x][y-1]=false;
	}

	//cout<<'3';
	if(y<n-1 && !visited[x][y+1]&& !a[x][y+1])
	{
			dfs(x,y+1);
			visited[x][y+1]=false;
	}

	//cout<<'4';
	
}
int main()
{
	int i,j;
	
	cin>>n;
	
	for(i=0;i<n;i++)
	for(j=0;j<n;j++)
	{
		cin>>a[i][j];
		visited[i][j]=false;
	}
	
	cnt =  0;
	dfs(0,0);
	cout<<cnt;
    
    return 0;
}
