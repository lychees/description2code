#include <iostream>
#define nt() int t; cin>>t; while(t--)
using namespace std;

int main()
{
    nt()
    {
    	long long int s, i, temp;
    	cin>>s;
    	for(i=0;i<2;i++)
    	{
    		cin>>temp;
    		if(temp<s)
    		s=temp;
    	}
    	if(s%2==0)
    	cout<<s/2<<".0"<<endl;
    	else
    	cout<<s/2<<".5"<<endl;
    }
return 0;
}
