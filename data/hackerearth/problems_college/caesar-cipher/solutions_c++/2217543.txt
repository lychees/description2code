

#include <iostream>
#include <string>


int main()
{
	std::ios_base::sync_with_stdio(false);
	int t;
	std::cin>>t;
	while(t--)
	{
		int n;
		std::string s;
		std::cin>>n>>s;
		n=n%26;

		for(int i=0 ; i<s.size() ; ++i)
		{
			
			if( s[i]+n > 90 && s[i]<=90 && s[i]>=65)
				s[i]=char( (s[i]+n)-26);
			else if( s[i]+n > 122 && s[i]>=97 && s[i]<=122 )
				s[i]=char( (s[i]+n)-26);
			else
				s[i]=s[i]+n;
		}
		std::cout<<s<<std::endl;
	}
}
