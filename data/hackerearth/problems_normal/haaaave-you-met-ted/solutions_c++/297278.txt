#include <stdio.h>
#include <string.h>
#include <iostream>
#include <algorithm>

using namespace std;
typedef long long LL;

int n;
LL a[100005];

// Convert to binary
LL countOnes(LL n1)
{
  LL ones = 0;

  while(n1)
  {
    ones += n1 & 1;
    n1 /= 2;
  }

  return ones;
}
//

int main() 
{
  int t;

  scanf("%d", &t);
  while(t--)
  {
    scanf("%d", &n);
    for(int i = 0; i < n; ++i) scanf("%lld", &a[i]);

    LL ans = 1000000000;
    sort(a, a + n);
    for(int i = 1; i < n; ++i)
    { 
      ans = min(ans, countOnes(a[0] ^ a[i]));
    }

    for(int i = n - 2; i >= 0; --i)
    { 
      ans = min(ans, countOnes(a[n - 1] ^ a[i]));
    }

    if(n == 1) printf("0\n");
    else printf("%lld\n", ans);
  }
  
  return 0;
}
