#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int t;
    long long int c,x,y,a,b;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>a>>b>>c;
    	x=pow(a,b);
    	y=x/c;
    	if(x-(y*c)<((y+1)*c)-(x))
    		cout<<y*c<<endl;
    	else
    		cout<<((y+1)*c)<<endl;
    }
    return 0;
}
