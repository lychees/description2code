#include <iostream>
#include<algorithm>
#include<iomanip>
using namespace std;
struct S
{
	double x,y;
};
bool compare(S a, S b)
{
	if(a.x>b.x)
	{
		return 0;
	}
	else if(a.x==b.x)
	{
		if(a.y>b.y)
		{
			return 0;
		}
		else
		{
			return 1;
		}
	}
	else
	{
		return 1;
	}
}
int main()
{
    double x[3],y[3],sx=0,sy=0;
    int i;
    S ans[3];
    for(i=0;i<3;i++)
    {
    	cin>>x[i]>>y[i];
    	sx+=x[i],sy+=y[i];
    }
    for(i=0;i<3;i++)
    {
    	ans[i].x=sx-2*x[i];
    	ans[i].y=sy-2*y[i];
    }
    sort(ans,ans+3,compare);
    for(i=0;i<3;i++)
    {
    	cout<<fixed<<setprecision(4)<<ans[i].x<<' '<<ans[i].y<<endl;
    }
}
