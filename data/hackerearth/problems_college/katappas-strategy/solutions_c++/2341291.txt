#include<iostream>
#include<string>
using namespace std;

int gcd(int p, int q) {
        if (p < q) {
            int tem = p;
            p = q;
            q = tem;
        }
        if (q == 0) {
            return p;
        }
        return gcd(q, p % q);

}
int main() {
        int t,n,arr[100000][2],last[100000];
        double brr[100000];
        bool ans;
        char ccccc;
        string s="";
        cin>>t;
        for (int i = 0; i < t; i++) {
            cin>>n;
            for (int j = 0; j < n; j++) {
                cin>>arr[j][0]>>ccccc>>arr[j][1];
          // cout<<arr[j][0]<<'/'<<arr[j][1]<<endl;
                
                }

			brr[n - 1] = ((double) arr[n - 1][0]) / arr[n - 1][1];
            double x;
            last[n - 1] = n - 1;
            //cout<<"For";
            for (int j = n - 2; j >= 0; j--) {
                x = ((double) arr[j][0]) / arr[j][1];
                last[j] = j;
                int k = j + 1;

                while (k < n && x<brr[k]) {
                    arr[j][0] += arr[k][0];
                    arr[j][1] += arr[k][1];
                    last[j] = last[k];
                    k = last[j] + 1;
                    x = ((double) arr[j][0]) / arr[j][1];
                }
                brr[j] = ((double) arr[j][0]) / arr[j][1];

            }
            int p;
            
            for (int j = 0; j < n; j++) {
                p = gcd(arr[j][0], arr[j][1]);
                arr[j][0] /= p;
                arr[j][1] /= p;
                
                
                s+=to_string(arr[j][0])+"/"+to_string(arr[j][1])+"\n";
            }
            s+="\n";
        }

    }
