#include <iostream>
using namespace std;

int main()
{
    long long int t,n,m;
    cin >> t;
    while(t--)
    {
    	cin >> n >> m;
    	while(1)
    	{
    		while(n<m)
    		{
    			m/=2;
    		}
    		while(m<n)
    		{
    			n/=2;
    		}
    		if(m==n)
    			break;
    	}
    	cout << m << endl;
    }
    return 0;
}
