#include<bits/stdc++.h>
using namespace std;
long long mulmod(long long a,long long b,long long c){
    long long x = 0,y=a%c;
    while(b > 0){
        if(b%2 == 1){
            x = (x+y)%c;
        }
        y = (y*2)%c;
        b /= 2;
    }
    return x%c;
}
int main(){
	int t;
	cin>>t;
	while(t--){
		long long int N,M,D;
		cin>>N>>M>>D;
		N=N/D;
		M=M/D;
		cout<<mulmod(N,M,1000000007)<<"\n";
	}
}
