#include <iostream>
using namespace std;

int main()
{
	int t, n;
	cin>>t;
	while(t--) {
		cin>>n;
		int array[n];
		int visit[n];
		for(int i=0;i<n;i++) {
			cin>>array[i];
			visit[i] = 0;
		}		
		int x = 0;
		while(visit[x] == 0 && array[x] != -1) {
			visit[x] = 1;
			x = array[x];
		}
		if(array[x] == -1) cout<<"Win\n";
		else cout<<"Lose\n";
	}
    return 0;
}
