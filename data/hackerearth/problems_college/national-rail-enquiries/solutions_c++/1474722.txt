#include <stdio.h>
long long fact(long);
long long gcd(long,long);
long long gcd(long a,long b)
{
return (b == 0 ? a : gcd(b, a % b));
}
long long fact(long n)
{
if(n==0 || n==1)
return 1;
else
return n*fact(n-1);
}
int main()
{
long long t,j=0;
scanf("%lld",&t);
while(t--)
{
long long n,res;
scanf("%lld",&n);
if(n>110)
res=0;
else
res=fact(n-1)%110;
if(res>0 && gcd(res,n)==1)
printf("Case %d:Yes\n",++j);
else
printf("Case %d:No\n",++j);
}
return 0;
}
