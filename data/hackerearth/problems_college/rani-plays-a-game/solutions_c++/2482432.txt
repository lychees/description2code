#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,k;
        scanf("%d%d",&n,&k);
        if((n%(k+1))==1)
        {
            printf("Nandu\n");
        }
        else
        {
            printf("Rani\n");
        }
    }
return 0;}
