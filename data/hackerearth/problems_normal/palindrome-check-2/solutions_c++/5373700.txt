#include<iostream>
using namespace std;
#include<string.h>
int main()
{
	char str[100];
	cin>>str;
	int i,flag=0;
	i=strlen(str);
	if(i>=1 && i<=100)
	{
		for(int j=0;j<i;j++)
		{
			if(str[j]==str[i-(j+1)])
			{
				flag=1;
			}
			else
			{
				flag=0;
				break;
			}
		}
		if(flag==1)
		{
			cout<<"YES";
		}
		else
		{
			cout<<"NO";
		}
	}
}
