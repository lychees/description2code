#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
    int N, P;
    cin >> N >> P;
    double p = P/ 100.0;
    
    double out[1000005];
    out[0] = 1;
    out[1] = 0;
    out[2] = p;
    
    for(int i = 3; i <= N; i++)
    {
    	out[i] = p*out[i-2] +(1-p)*out[i-3];
    }
    printf("%.6f\n",out[N]);
    
    return 0;
}
