#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long i,n,t,c,flag;
    scanf("%lld",&t);
    while(t--)
    {
    	flag=0;c=0;
    	scanf("%lld",&n);
    	while(n%2==0)
    	{
    		c+=1;
    		n/=2;
    		if(c>1)
    			break;
    	}
    	if(c>1)
    	{
    		cout<<"No\n";continue;
    	}
    	for(i=3;i<=sqrt(n);i+=2)
    	{
    		c=0;
    		while(n%i==0)
    		{
    			c+=1;
    			n/=i;
    			if(c>1)
    				{flag=1;break;}
    		}
    		if(c>1)
    		{flag=1;break;}
    	}
    	if(flag)
    		cout<<"No\n";
    	else
    		cout<<"Yes\n";
    }
    return 0;
}
