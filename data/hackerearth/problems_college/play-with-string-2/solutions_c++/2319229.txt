#include<cstdio>
#include<iostream>
#include<cstring>
#include<cstdlib>
#include<algorithm>

int main()
{
	int t;

	{
		char str[10000];
		char s[10000];
		cin>>str;
		int len = strlen(str);
         int i,k=0,j;
         for(i=0;i<len;i++)
         {
          s[k]= str[len-i-1];
          k++;
		 }
		// cout<<s<<"\n";
		 s[k]='\0';
		 int flag=0;
		 for(i=0,j=0;i<len-1,j<k-1;i++,j++)
		 {
		 	int p = abs(str[i+1]-str[i]);
		 	int q = abs(s[j+1]-s[j]);
		 	//cout<<p<<" "<<q<<"\n";
		 	if(p!=q)
		 	{
		 		flag =1;
		 		break;
			 }
		 }

		 }

		 }
	}
}
