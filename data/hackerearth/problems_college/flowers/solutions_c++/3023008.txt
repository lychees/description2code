#include<stdio.h>
#include <stdlib.h>
int cmpfunc (const void * a, const void * b)
{
   return ( *(int*)a - *(int*)b );
}
int main(){
   long long int i, N, K,j=0,l=1;
   long long int C[102];
   long long int result=0;
   scanf("%lld %lld", &N, &K);
   for(i=0; i<N; i++){
      scanf("%lld", &(C[i]));
   }
   qsort(C,N,sizeof(long long int),cmpfunc);
    for(i=N-1;i>=0;i--){
            result+=C[i]*l;
            j++;
        if(j==K){
            j=0;
            l++;
        }
    }
   printf("%lld\n", result);
}
