from bisect import bisect_left
def binary_search(a, x, lo=0, hi=None):
	hi = hi if hi is not None else len(a)
	pos = bisect_left(a, x, lo, hi)
	return ((pos + 1) if pos < hi and pos >= 0 else -1)
	
tc = int(raw_input())
lo = 0
hi = None
s = raw_input()
A = map(int, s.split())
q = int(raw_input())
el = []
count = 0
for i in range(tc):
	count = count + A[i]
	el.append(count)
for i in range(0, q):
	n = int(raw_input())
	p = binary_search(el, n, lo, hi)
