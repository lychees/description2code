#include<bits/stdc++.h>
using namespace std;
void sol()
{
    long long s,e,x,d;
    int i,n;
    long long left,right,ans;
    vector<pair<long long ,long long > > v;
    cin>>n>>s>>e;
    for(i=0;i<n;++i)
    {
        cin>>x>>d;
        left=x-d;
        right=x+d;
        if(right<=s||left>=e) continue;
        left=max(s,left);
        right=min(right,e);
        v.push_back(make_pair(left,right));
    }
    if(v.size()==0)
    {
        cout<<e-s<<endl;
        return ;
    }
    sort(v.begin(),v.end());
    left=v[0].first;
    right=v[0].second;
    ans=max(0ll,left-s);
    for(i=1;i<v.size();++i)
    {
        if(v[i].second>right)
        {
            ans=ans+max(0ll,v[i].first-right);
            right=v[i].second;
        }
    }
    ans=ans+max(0ll,e-right);
    cout<<ans<<endl;

}
int main()
{
    sol();
    return 0;
}
