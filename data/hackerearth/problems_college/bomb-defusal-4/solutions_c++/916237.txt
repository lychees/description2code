#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    for(int i=0;i<t;i++)
    {
    	int p, x, y, u, v;
    	cin>>p>> x>> y>> u>> v;
   
   		int a=abs(x-u);
    	int b=abs(y-v);
    	
    	int c=2;
    	if(a==b)
    		c=1;
    		
   		if(a==0 && b==0)
    		cout<<"Counter-Terrorists Win !\n";
		else if(abs(x-y)%2!=abs(u-v)%2)
			cout<<"Terrorists Win !\n";
		else if(c<p)
    		cout<<"Counter-Terrorists Win !\n";
    	else
    		cout<<"Terrorists Win !\n";
    }
    return 0;
}
