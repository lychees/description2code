#include<bits/stdc++.h>
using namespace std;
//Beauty is in relaxed hardwork :)
int main()
{
    int t,l=1;
    cin>>t;
    while(t--)
    {
    	int n,i;
    	cin>>n;
    	long long ans=0,a,b;
    	vector<long long >v(n);
    	for(i=0;i<n;i++)
    	cin>>v[i];
        if(n==0){cout<<"Case "<<l<<": 0\n";l++;continue;}
        if(n==1){cout<<"Case "<<l<<": "<<v[0]<<"\n";l++;continue;}
    	a=v[0];ans=max(v[0],v[1]);
    	for(i=2;i<n;i++)
    	{
    		b=ans;
    		ans=max(ans,a+v[i]);
    		a=b;
    	}
    	cout<<"Case "<<l<<": "<<ans<<endl;
    	l++;
    }
    return 0;
}
