#include <iostream>
using namespace std;

int main()
{
    int tcase;
    cin>>tcase;
    while(tcase--)
    {
    	int n;
    	cin>>n;
    	int x,y,sum=0,temp=n;
    	while(1)
    	{
    		x=temp/2;
    		y=temp%2;
    		sum+=x;
    		if(x==0)
    		{
    			break;
    		}
    		else
    		{
    			temp=x+y;
    		}
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
