# print 'Hello World!'
s = raw_input().strip()
h = {}
for e in s:
    if ord(e) in h:
        h[ord(e)] += 1
    else:
        h[ord(e)] = 1
m = [0, 0]
for e in h:
    if h[e] > m[0]:
        m = [h[e], e]
    elif h[e] == m[0] and e < m[1]:
        m = [h[e], e]
print chr(m[1]), m[0]
