#include <iostream>
using namespace std;
#define ill long long
int main()
{
    int t,m,n,i,j,k,l;
    cin>>t;
    
    while(t--)
    {
    	cin>>n>>m;
    	
    	int a[n+9];
    	int low,high;
    	high=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		high=max(high,a[i]);
    	}
    	
    	low=0;
    	ill ans=0;
    	while(low<=high)
    	{
    		int mid=(low)+(high-low)/2;
    		
    		int ct=1;
    		ill dis=0;
    		ill prev=0;
    	
    		for(i=1;i<n;i++)
    		{
    			if(i)
    			dis+=a[i]-a[i-1];
    			else
    			dis+=a[i];
    			
    			if(dis>=mid)
    			{
    				ct++;
    				dis=0;
    			}
    		}
    		//cout<<mid<<" "<<ct<<"\n";
    		if(ct>=m)
    		{
    			low=mid+1;
    			ans=mid;
    		}
    		else
    		{
    			high=mid-1;
    		}
    	}
    	cout<<ans<<"\n";
    	
    }
    
}
