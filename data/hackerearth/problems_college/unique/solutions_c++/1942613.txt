#include <bits/stdc++.h>
using namespace std;
map<long int,int>mp;
int main()
{
    long long int a[100003];
    int t,n;

    scanf("%d",&t);
    while(t--){
    mp.clear();
    scanf("%d",&n);
    int j,i;

    int count1 = 0;
    for(i = 0; i < n;i++){
        scanf("%lld",&a[i]);
        if(mp[a[i]] == 0){
            count1++;
            mp[a[i]] = 1;
        }
    }
    printf("%d\n",count1);

    }

}
