#include<stdio.h>
#include<iostream>
#include<map>
using namespace std;
map<long int,long int> mm;
map<long int,long int>::iterator it;
int main()
{
long int i,j,k,l,m,n,max_val,max_ind;
scanf("%ld",&n);
scanf("%ld",&m);
max_val=0;
max_ind=-1;
for(i=0;i<n;i++)
{
scanf("%ld",&k);
it=mm.find(k);
if(!(it==mm.end()))
{
l=it->second;
mm.erase(it);
}
else
{
l=0;
}
l++;
if(l>max_val)
{
max_val=l;
max_ind=k;
}
else if(l==max_val)
{
if(k>max_ind)
{
max_val=l;
max_ind=k;
}
}
mm.insert(pair<long int,long int>(k,l));
printf("%ld %ld\n",max_ind,max_val);
}
return 0;
}
