#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char a[10005];
    	cin>>a;
    	int len = strlen(a);
    	char freq[27]={0};
    	for(int i=0; i<len; i++)
    		freq[a[i]-'a']++;
    	int max = -1;
    	for(int i=0; i<26; i++)
    		if(freq[i]>max)
    			max = freq[i];
    	cout<<max<<endl;
    }
    return 0;
}
