t = raw_input()
start_day_lst = []
end_day_lst = []

input_lst = raw_input().split()

input_lst = map(lambda a : int(a) , input_lst)

input_lst.sort()
input_lst.reverse()


for i in range(int(t)):

    start_day_lst.append(i+1)
    end_day_lst.append((i+1) + input_lst[i])

end_day = reduce(lambda a , b : a if a>b else b , end_day_lst)

print end_day + 1
