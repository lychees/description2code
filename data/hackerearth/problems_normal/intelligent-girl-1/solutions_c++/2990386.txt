#include <iostream>
using namespace std;

int main()
{	int count =0;
  string s;
  cin>>s;
  int l = s.length();
  int array[l];
  for(int i=0;i<l;i++){
  	array[i] = int(s[i])-int('0');
  }
  for(int i=0;i<l;i++){
  	for(int j=i;j<l;j++){
  		if(array[j]%2==0)count ++;
  	}
  	cout<<count<<" ";
  	count =0 ;
  }
}
