#include <iostream>
#include <sstream>
#include <string>
#include <stdio.h>
#include <vector>
#include <stack>
using namespace std;
long long int toint(const string &s) { stringstream ss; ss << s; long long int x; ss >> x; return x; }
long long int n ;

int main(){

while(scanf("%lld",&n)!=EOF){
getchar();
//cout<<n<<endl;
long long int i,j;
int flag  = 1;
int taken[n];
string p , num ;
for(i=0;i<n;i++) taken[i]=0;
vector<long long int> arr;
getline(cin,p);
if(p.size()==0){
        cout<<endl;
        continue;
}
stringstream s1(p);
while(s1>>num){
    arr.push_back(toint(num));
    if(toint(num)>n){
        //cout<<toint(num)<<" "<<n<<endl;
        flag = 0;
        break;
    }
}
p.clear();
if(flag==0){
        //cout<<"blah"<<endl;
        cout<<-1<<endl;
        continue;
}
long long int l = arr.size();
long long int c[l];
long long int k[l];
for(i=0;i<l;i++){
    c[i]=0;
    k[i]=0;
}
long long int q = 1, m;
stack<long long int> storer;
for(i=0;i<l;i++){
//cout<<"$$"<<arr[i]<<endl;
if(taken[arr[i]]==1){
        flag = 0;
        break;
}
if(arr[i]>=q){
for(j=q;j<=arr[i];j++){

if(taken[j]==1) continue;
else{
        storer.push(j);
        c[i]++;
}

}
if(!storer.empty()){
//c[i] = storer.size() - q + 1;
q = storer.top() + 1;
storer.pop();
k[i] = 1;
taken[arr[i]] = 1;
}

}
else{
c[i]=0;
while(!storer.empty() && storer.top() >= arr[i]){
    k[i]++;
    m=storer.top();
    taken[m]=1;
    storer.pop();
}
if(storer.empty() && m!=arr[i]){
        flag = 0;
        break;
}

}
//cout<<c[i]<<endl;
}
string a;
a.clear();
if (flag==0) cout<<-1<<endl;
else if(flag!=0){
for(i=0;i<l;i++){

for(j=0;j<c[i];j++){
a = a+"C";
}
for(j=0;j<k[i];j++){
    a= a+"K";
}

}
cout<<a<<endl;
}

}
return 0;
}
