typedef pair<int, int > pii;

#define MOD 1000000007
#define pb push_back


void solve()
{
	int n, k;
	cin>>n>>k;
	vector<int> a(n);
	priority_queue<pii> pq;
	for(int i=0;i<n;i++) {
		cin>>a[i];
		pq.push(pii(a[i], i));
	}
	if(k >= n) {
		sort(a.begin(),a.end());
		reverse(a.begin(), a.end());
		for(int i=0;i<n;i++) {
			cout<<a[i]<<" ";
		}
		cout<<"\n";
		return;
	}
	int i = 0;
	while(k > 0) {
		if(i == n)
			break;
		if(i >= pq.top().second) {
			pq.pop();
		}
		else {
			if(a[i] >= pq.top().first)
				i++;
			else {
				pii p = pq.top();
				pq.pop();
				int t = a[i];
				a[i] = p.first;
				a[p.second] = t;
				pq.push(pii(t, p.second));
				k--;
			}
		}
	}
	for(int i=0;i<n;i++) {
		cout<<a[i]<<" ";
	}
	cout<<"\n";
}

int main()
{
	std::ios::sync_with_stdio(false);
//	freopen ("C:\\Users\\Ritesh\\Desktop\\Coding\\Codeforces\\input.txt","r",stdin);
	int t;
	//cin>>t;
	t = 1;
	while(t--)
	{
		solve();
	}
	return 0;
}
