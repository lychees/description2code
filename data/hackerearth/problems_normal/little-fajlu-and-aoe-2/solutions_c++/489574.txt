#include<bits/stdc++.h>
using namespace std;
#define maxn 40005
#define F(i,p,n) for(i=p;i<n;i++)
#define Ss(x) scanf("%s",x)
#define S(x) scanf("%d",&x)
#define P(x) printf("%d\n",x)
#define Ps(x) printf("%d ",x)
#define LL long long int
main()
{
    int i,j,k,m,n,t,food,gold,lalpow,sum,pal,champ,s,p,c,food_left,gold_left;
    /*cp[0][0]=100;
    cp[0][1]=125;
    cp[0][2]=50;
    cp[1][0]=0;
    cp[1][1]=50;
    cp[1][2]=100;*/
    //printf("%d\n",cp[0][0]);
    S(t);
    while(t--)
    {
        S(food);
        S(gold);
        S(s);
        S(p);
        S(c);

        S(lalpow);
        int maxnum=-1;

        for(sum=0;sum*100<=food;sum++)
        {
            for(pal=0;sum*100+pal*125<=food&&pal*50<=gold;pal++)
            {
                food_left=food-(sum*100+pal*125);
                gold_left=gold-(pal*50);
                champ=min(food_left/50,gold_left/100);
                maxnum=max(maxnum,s*sum+p*pal+c*champ);
            }
        }


        if(lalpow>=maxnum)
            printf("-1\n");
        else
            P(maxnum-lalpow);
    }
}
