#include <iostream>
#include<algorithm>
using namespace std;

struct Topic
{
	long long id;
	long long oldZ;
	long long p,l,c,s;
	long long newZ;
	long long diff;
};
typedef struct Topic tp;

tp arr[1000000];

bool compare(const tp &a,const tp &b)
{
	if(a.diff>b.diff)
	return true;
	
	else if(a.diff==b.diff && a.id>b.id)
	return true;
	
	else
	return false;
}

int main()
{
    int n,i;
    
    cin>>n;
    
    for(i=0;i<n;i++)
    cin>>arr[i].id>>arr[i].oldZ>>arr[i].p>>arr[i].l>>arr[i].c>>arr[i].s;
    
    for(i=0;i<n;i++)
    {
    	arr[i].newZ=arr[i].p*50+arr[i].l*5+arr[i].c*10+arr[i].s*20;
    	arr[i].diff=arr[i].newZ-arr[i].oldZ;
    }
    
    sort(arr,arr+n,compare);
    
    for(i=0;i<5 && i<n;i++)
    cout<<arr[i].id<<" "<<arr[i].newZ<<"\n";
    
    return 0;
}
