#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
	ll ans=0,count;
	int t,n,i;
	string s;
	cin>>t;
	while(t--)
	{	count=1;ans=0;
		cin>>n;
		cin>>s;
		for(i=0;i<n;i++)
		{
			if(i>0 && s[i]>=s[i-1])count++;
			else count=1;
			ans+=count;
		}
		cout<<ans<<endl;
	}
    return 0;
}
