ll fact[SIZE];
void f()
{
	fact[0]=1;
	for(int i=1;i<SIZE;i++){
		fact[i]=fact[i-1]*i;
		fact[i]%=MOD;
	}
	return ;
}
inline long long mod_pow(long long x,long long y)
{
	long long res = 1;
	while(y)
	{
		if(y&1) 
		{
			res = (res*x)%MOD;
		}
		x = (x*x)%MOD;
		y >>= 1;
	}
	return res%MOD;
}
inline long long mod_div(long long a,long long b)
{
	long long ans = (a*mod_pow(b,MOD-2))%MOD; 
	return ans;
}
