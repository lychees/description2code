#include <cstdio>
#include <algorithm>
using namespace std;

int t;
int n, k, p;
int a[100003];
int i;

int main()
{
	scanf("%d", &t);

	while(t--)
	{
		scanf("%d%d%d", &n, &k, &p);
		for(i = 0; i < k; ++i)
			scanf("%d", &a[i]);
		sort(a, a + k);
		i = 0;
		//printf("p before is %d\n",p);
		//printf("i before is %d\n",i);
		while(i < k && p <= n && a[i++] <= p)
        {
            p++;
            //printf("i is %d\n",i);
            //printf("k is %d\n",k);
            //printf("p is %d\n",p);
            //printf("a[i] is %d\n",a[i]);
        }
		printf("%d\n", p <= n ? p : -1);
	}

	return 0;
}
