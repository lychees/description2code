#include <iostream>
using namespace std;

int main()
{
	int n,q;
	cin >> n >> q;
	int* l= new int[n];
	for (int i = 0; i < n; i++){
		cin >> l[i];
	}
	int i,j;
	while (q > 0){
		cin >> i >> j;
		long long int sum = 0;
		long long int dif = 2*l[j-1];
		for (int k= i-1; k <j; k++){
			sum += l[k];
			dif -= l[k];
		}
		cout << sum << " " << dif << endl;
		q--;
	}
    return 0;
}
