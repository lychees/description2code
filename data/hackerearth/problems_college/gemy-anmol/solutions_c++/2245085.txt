#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		long long int m;
		scanf("%d %lld",&n,&m);
		long long int a[n];
		for(int i=0;i<n;i++)
			scanf("%lld",&a[i]);
		sort(a,a+n);
		long long int sum=0;
		int j,cnt=0;
		for(j=n-1;j>=0;j--)
		{
			++cnt;
			sum=sum+a[j];
			if(sum>=m)
				break;
		}
		if(sum>=m)
			printf("%d\n",n-j);
		else
			printf("-1\n");
	}
	return 0;
}
