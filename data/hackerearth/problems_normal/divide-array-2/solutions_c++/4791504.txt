#include<iostream>
using namespace std;
int main()
{
	long int n, q, d, arr[n], f=1;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>arr[i];
	cin>>q;
	for(int j=0;j<q;j++)
	{
		cin>>d;
		f=f*d;
	}
	for(int i=0;i<n;i++)
	{
		if(f!=0)
			arr[i]=arr[i]/f;
		else
			arr[i]=0;
	}
	for(int i=0;i<n;i++)
		cout<<arr[i]<<" ";
	return 0;
}
