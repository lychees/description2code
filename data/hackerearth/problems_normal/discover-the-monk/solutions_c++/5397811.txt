void search(int x, int a[] ,int n){

	
	int k=0;
	int low=0;
	int high=(n-1);
	int mid;
	while(high>=low){
		 	mid=(high+low)/2;
		if(x>a[mid]){
			low=mid+1;
		
		}
			if(x<a[mid]){
			high=mid-1;
		
		}
		if(a[mid]==x){
			cout<<"YES"<<endl;
				k=1;
			break;
	
		}
		
	}
	if(k==0){
		cout<<"NO"<<endl;
	}
	
		
		
	}
