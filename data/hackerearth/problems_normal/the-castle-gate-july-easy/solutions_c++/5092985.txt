#include<iostream>
#include<map>
#include<vector>
#include<algorithm>
#define ll long long
using namespace std;
int main() {
  ll T,n,i,ans;
  cin>>T;
  while(T--) {
    cin>>n;
    ans = 0;
    for(int i=1;i<=n;i++) {
      for(int j=i+1;j<=n;j++) {
        if( (i^j) <= n) {
          ans++;
        }
      }
    }
    cout<<ans<<endl;
  }
}
