#include<iostream>
using namespace std;
typedef long long ll;
int lBound(ll *A,int sz,ll x) {
    int top=-1,bot=sz-1,mid;
    while(bot-top>1) {
        mid=(top+bot)/2;
        if (A[mid]<=x) bot=mid;
        else top=mid;
    }
    //cout<<"bb: "<<A[bot]<<endl;
    if (x>A[0]) return 0;
    else return bot;
}

int main() {
    ios::sync_with_stdio(false);
    int i,j,tc,N,maxim;
    ll *A,*B;
    cin>>tc;
    while(tc--) {
        cin>>N;
        A=new ll[N];
        B=new ll[N];
        for(i=0; i<N; i++) cin>>A[i];
        for(i=0; i<N; i++) cin>>B[i];
        maxim=-1;
  
            for(j=N-1; j>=0; j--) {
                i=lBound(A,j+1,B[j]);
                //cout<<j<<' '<<i<<endl;
                maxim=max(maxim,j-i);
            }
        cout<<maxim<<endl;
    }
    return 0;
}
