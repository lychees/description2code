#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long int n,i;
    cin>>n;
    long long int a[100002];
    for(i=0;i<n;i++){
    	cin>>a[i];
    }
    sort(a,a+n);
    long long int e=a[0];
    long long int f=a[1];
    long long int y=e*f;
    long long int m=a[n-1];
    long long int k=a[n-2];
    long long int x = m*k;
    long long int z=max(x,y);
    cout<<z;
    return 0;
}
