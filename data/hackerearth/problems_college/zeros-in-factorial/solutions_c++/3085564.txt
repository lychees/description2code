#include <iostream>
#include <vector>
using namespace std;
 
int main()
{
	vector<int> twos(1000001);
	vector<int> fives(1000001);
	for (int i = 2; i <= 1000000; ++i) {
		int cp = i;
		while (cp % 2 == 0) {
			twos[i]++;
			cp /= 2;
		}
		while (cp % 5 == 0) {
			fives[i]++;
			cp /= 5;	
		}
		twos[i] += twos[i-1];
		fives[i] += fives[i-1];
	}
	vector<int> zeroes(1000001);
	vector<long long> accum(1000001);
	for (int i = 1; i <= 1000000; ++i) {
		zeroes[i] = min(fives[i], twos[i]);
		accum[i] = zeroes[i] + accum[i-1];
	}
	int T;
	cin >> T;
	for (int i = 0; i < T; ++i) {
		int a, b;
		cin >> a >> b;
		cout << accum[b] - accum[a-1] << endl;
	}
}
