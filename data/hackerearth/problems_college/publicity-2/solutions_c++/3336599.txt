#include <iostream>
#include<vector>
#include<string.h>
using namespace std;
vector<int>v[100002];
bool visited[100023];
 
void dfs(int i)
{
	visited[i]=1;
	for(int j=0;j<v[i].size();j++){
		if(!visited[v[i][j]])dfs(v[i][j]);
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--){
		int n,e;
		cin>>n>>e;
		
		memset(v,0,sizeof(v));
		memset(visited,0,sizeof(visited));
		while(e--){
			int a,b;
			cin>>a>>b;
			v[a].push_back(b);
			v[b].push_back(a);
		}
		int ans=0;
		for(int i=0;i<n;i++){
			if(!visited[i])
			{
				ans++;
				dfs(i);
			}
		}
		cout<<ans<<endl;
		
	}
    
    return 0;
}
