#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,i,x;
    char a[2000],b[2000],d[2000];
    
    cin>>n;
    getchar();
    while(n--)
    {
    	gets(a);
    	
    	for(i=0;i<strlen(a);i++)
    	{
    		if(a[i]==' ')
    		{
    			x=i;
    			break;
    		}
    	}
    	
    	for(i=0;i<strlen(a);i++)
    	{
    		if(i<x)
    		b[i]=a[i];
    		
    		else if(i>x)
    		d[i-x-1]=a[i];
    	}
    	
    	sort(b,b+strlen(b));
    	
    	sort(d,d+strlen(d));
    	
    	//puts(b);
    	
    //	puts(d);
    	
    	if(!strcmp(d,b))
    	cout<<"Yes"<<endl;
    	
    	else 
    	cout<<"No"<<endl;
    }
    
    return 0;
}
