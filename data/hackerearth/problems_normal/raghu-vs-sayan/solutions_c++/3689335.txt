

    #include<bits/stdc++.h>
    #define ull unsigned long long int
    using namespace std;
    int main()
    {
        int t;
        cin>>t;
        while(t--)
        {
            ull a,b,n,i;
            cin>>a>>b>>n;
            ull k[n];
            for(i=0;i<n;i++)
                cin>>k[i];
            sort(k,k+n);
            ull sum=0,r=0,s=0;
            for(i=0;i<n;i++)
            {
                sum+=k[i];
                if(sum<=a)
                    r++;
                if(sum<=b)
                    s++;
            }
            if(r>s)
                cout<<"Raghu Won"<<endl;
            else if(r==s)
                cout<<"Tie"<<endl;
            else
                cout<<"Sayan Won"<<endl;
        }
    }
