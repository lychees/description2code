#include <iostream>
#include <math.h>
using namespace std;

int main()
{    

	int lst[1000001] = {0};

	for(int i = 27; i < 1000001; i++)
	{
		for(int j = 2; j < (int)sqrt(i) + 1; j++)
			{
				if(i % j == 0)
				{
					int k = i;
					while(k)
    				{
    					if(k % 10 == 7)
    					{
    						lst[i] = 1;  
    						break;
    					}
    					k /= 10;
    				}
    				break;
				}
			}
	}	

int tc;
cin >> tc;
while(tc--)
{
	int x, y, count;
	cin >> x >> y;
	
	if(x < 27) x = 27;
	
	count = 0;
	
	for(int i = x; i <= y; i++)
		if(lst[i] == 1) count++;
		
	if (count == 0) cout << -1 << endl;
	else cout << count << endl;
}	

return 0;
}
