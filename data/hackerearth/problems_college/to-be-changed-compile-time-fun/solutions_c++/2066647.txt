#include<iostream>
using namespace std;


int main()
{int l,n,j,temp,maxmum,score=0,index;
int arr[100];

cin>>l;

for(int i=0;i<l;i++)
    cin>>arr[i];
n=l;


for(int i=1;i<l-1;i++)
{
    maxmum=arr[0]*arr[2];
    index=1;
    for(j=1;j<n-1;j++)
    {
    temp=arr[j-1]*arr[j+1];
 if(temp>maxmum)
 {maxmum=temp;
 index=j;
 }
    }

    score+=maxmum;
    //cout<<maxmum;
    //cout<<index;
    n=n-1;
      for(int k=index;k<n;k++)
       arr[k]=arr[k+1];



}



cout<<score;
return 0;
}
