#include <bits/stdc++.h>
using namespace std;
int main()
{
float x1,y1,x2,y2,x3,y3,a[3],b[3],ans=0;
int i,j;
cin>>x1>>y1;
cin>>x2>>y2;
cin>>x3>>y3;
/* l1+l2=2*x1;
l2+l3=2*x2;
l3+l1=2*x3;
m1+m2=2*y1;
m2+m3=2*y2;
m3+m1=2*y3; */
a[0]=(x1-x2+x3);
a[1]=(x1+x2-x3);
a[2]=(-x1+x2+x3);
b[0]=(y1-y2+y3);
b[1]=(y1+y2-y3);
b[2]=(-y1+y2+y3);


for(i=0;i<3;i++)
{
for(j=i+1;j<3;j++)
{
if(a[i]>a[j])
{
ans=a[j];
a[j]=a[i];
a[i]=ans;
ans=b[j];
b[j]=b[i];
b[i]=ans;

}
}
}
cout<<setprecision(4)<<fixed<<a[0]<<" "<<b[0]<<endl;
cout<<setprecision(4)<<fixed<<a[1]<<" "<<b[1]<<endl;
cout<<setprecision(4)<<fixed<<a[2]<<" "<<b[2]<<endl;






return 0;
}
