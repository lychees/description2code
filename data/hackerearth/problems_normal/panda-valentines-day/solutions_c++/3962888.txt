#include <iostream>
#include <cstdio>
using namespace std;
#define N 1000000
#define ll long long
int arr[1000005];
ll ans[1000005];

int factor(int n) {
	int ans = 0;
	while ( n > 1 ) {
		ans++;
		n /= arr[n];
	}
	return ans;
}

int main() {

	int i, j, t, n;

    arr[1] = 0;
	for ( i = 2 ; i <= N ; i++ ) {
		if ( !arr[i] ) {
			arr[i] = i;
			for ( j = i*2 ; j <= N ; j += i )
				arr[j] = i; ;
		}
	}
     
    ans[0] = 0; 
	for ( i = 2 ; i <= N ; i++ )
		ans[i] = ans[i-1] + factor(i);

	scanf("%d", &t);
	while ( t-- ) {
		scanf("%d", &n);
        printf("%lld\n", ans[n]); 
	} 
	return 0;

}
