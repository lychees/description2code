#include <iostream>
using namespace std;
long int 
gcd ( long int a, long int b )
{
  int c;
  while ( a != 0 ) {
     c = a; a = b%a;  b = c;
  }
  return b;
}

int main()
{
   long int t,n,i,ans;
   cin>>t;
   while(t--)
   {
   	 cin>>n;
   	 long int a[n];
   	 for(i=0;i<n;i++)
   	 cin>>a[i];
   	 if(n==1&&a[0]==1)
   	 {
   	 	cout<<"YES"<<endl;
   	 	continue;
   	 }
   	  ans=gcd(a[0],a[1]);
   	  for(i=2;i<n;i++)
   	  {
   	  	ans=gcd(ans,a[i]);
   	  }
   	  if(ans==1)
   	  cout<<"YES"<<endl;
   	  else
   	  cout<<"NO"<<endl;
   }
    return 0;
}
