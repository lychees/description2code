ll power(ll a,ll n,ll m)
{
	if(n==0)
		return 1;
	else if(n==1)
		return a;
	else if(n%2==0)
		return power((a*a)%m,n/2,m)%m;
	return (a * (power((a*a)%m,(n-1)/2,m))%m)%m;
}
 
int mul_inv(int a, int b)
{
	int origb=b;
	int x=0;
	int y=1;
	int u=1;
	int v=0;
    while(a != 0)
    {
        int q=b/a;
        int r=b%a;
        if(r<0) r+=a;
        int m=x-u*q;
        int n=y-v*q;
        b=a;a=r;x=u;y=v;u=m;v=n;
    }
    if(x<0) x+=origb;
    return x%origb;
}
