#include<iostream>
#include<algorithm>

using namespace std;

int findMaxLitDays(int * candles, int  n) {
    if(candles == NULL || n <= 0) {
        return 0;
    }

    int start = 0;
    int maxDays = 0;
    bool done = false;

    while(maxDays < n) {
        sort(candles, candles + n, greater<int>());

        while(candles[start] <= 0) {
            start++;
        }

        for(int itr = start; itr <= start + maxDays; ++itr) {
            if(candles[itr] > 0) {
                candles[itr]--;
                //cout << candles[itr]<< " ";
            } else {
                done=true;
            }
        }
        if(done){
            break;
        }
        maxDays++;
    }
    return maxDays;
}

int main()
{
   int t;
    cin >> t;

    while(t--) {
   	int n;
	cin >>n;
	int candles[n];
	for(int i=0;i<n;++i) {
		cin >>candles[i];
	}
    cout << findMaxLitDays(candles, n) << endl;
    }
    return 0;
}
