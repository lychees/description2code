    //cout<<b[i]<<endl;
	for(i=0;i<j-1;++i)

int find(int b[],int k,int j)
{
	int i,flag=0,in;
	for(i=0;i<j;++i)
	{
		if(b[i]==k)
		{
			flag=1;
			in=i;
			break;
		}
	}
	if(flag==1)
	return in;
	else
	return -1;
}
