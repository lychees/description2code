#include <bits/stdc++.h>
#define LIM 1000000
using namespace std;
long long dp[1000005];
void pre()
{
	int pp = (int)sqrt(LIM);
	for ( int i = 1; i <= pp; i++ ) {
		for ( int j = i; j <= LIM; j += i ) {
			dp[j]++;
			if ( j/i > pp && j/i != i ) dp[j]++;
		}
	}
	dp[1] = 0;
	for ( int i = 2; i <= 1000000; i++ ) dp[i] = dp[i-1] + (dp[i]-1);
	return;
}
int main()
{
	pre();
	int t,x;
	cin >> t;
	while ( t-- ) {
		cin >> x;
		cout << dp[x] << endl;
	}
	return 0;
}
