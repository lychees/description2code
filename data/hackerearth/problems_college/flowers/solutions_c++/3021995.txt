#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define f(i,a,b) for(i=a;i<b;i++)
#define si(a) scanf("%d",&a);
#define slli(a) scanf("%lld",&a);
#define MOD 1000000007
#define MAX 100005
#define pb(a) push_back(a)
lli v[105];
int main()
{
	lli i,j,k,m,n,t,p,turn,sum;
	slli(n);
	slli(k);

	f(i,0,n)
	{
		slli(m);
		v[i]=m;
	}
	sort(v,v+n);
	turn=1;
	sum=0;
	while(1)
	{
		for(p=n-1;p>=n-k;p--)
		{
			if(p<0)
				break;
			sum+=turn*v[p];
		}
                n=p+1;
		if(p<0)
			break;
		turn++;
	}
	printf("%lld\n",sum);
}
