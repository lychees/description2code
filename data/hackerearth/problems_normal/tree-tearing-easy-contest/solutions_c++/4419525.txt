#include <bits/stdc++.h>
using namespace std;

int visited[10005];
int node[10005];
int ans,k,n,x;

int dfs(int u,vector<int> adj[]){
	visited[u]=1;
	int nodes=1;
	for(int i=0;i<adj[u].size();i++)
		if(visited[adj[u][i]]==0)
			nodes+=dfs(adj[u][i],adj);
	if(nodes>=k){
		ans++;
		nodes=0;
	}
	return nodes;
}

int main(){
    cin>>k;
    	cin>>n;
    	memset(visited,0,sizeof(visited));
    	memset(node,0,sizeof(node));
    	vector<int> adj[n+1];
    	for(int i=2;i<=n;i++){
    		cin>>x;
    		adj[i].push_back(x);
    		adj[x].push_back(i);
    	}
    	dfs(1,adj);
    	cout<<ans<<endl;
    return 0;
}
