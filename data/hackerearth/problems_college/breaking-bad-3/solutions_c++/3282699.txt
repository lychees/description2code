#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    long long int n,m,t,i,sum=0;
    scanf("%d",&t);
    while(t--)
    {
    	sum=0;
    	scanf("%lld",&n);
    	long long int a[n];
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&a[i]);
    	}
    	sort(a,a+n);
    	for(i=n-1;i>=0;i-=2)
    	{
    		m=a[i]*a[i-1];
    		sum=sum+m;
    	}
    	printf("%lld\n",sum);
    }
    return 0;
}
