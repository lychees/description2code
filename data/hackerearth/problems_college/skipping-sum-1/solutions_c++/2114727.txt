#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[100010],dp[100010][11];//As  1 ? K ? 10
int main()
{
	ll int n,q;
	scanf("%lld%lld",&n,&q);
	ll int l,r,k;
	ll int i,j,prev,v;
	for(i=0;i<n;i++)
	{
		scanf("%lld",(a+i));
	}
	//Let precalculate the prefix sum
	for(i=0;i<n;i++)
	{
		for(k=0;k<=10;k++)
		{
			prev=(i-k);
			if(prev>=0)
			{
				v=dp[prev][k];
			}
			else
			{
				v=0;
			}
			dp[i][k]=v+a[i];
		}
	}
	while(q--)//Answering the queries
	{
		scanf("%lld%lld%lld",&l,&r,&k);
		l--;r--;//Converting to 0-based index;
		i=(r-l)/k;
		v=dp[l+(i*k)][k] -dp[l][k]+a[l];
		printf("%lld\n",v);
	}
	return 0;
}
