#include <iostream>
using namespace std;

int main()
{
    int t,n,k,cnt;
    cin>>t;
    while(t--){
    	cin>>n>>k;
    	int a[n];
    	cnt = 0;
    	for(int i = 0 ;i < n; i++){
    	  cin>>a[i];
    	  if(a[i]<=0)
    	  cnt++;
    }
    	 	if(cnt>=k)
    	 	cout<<"NO"<<endl;
    	 	else
    	 	cout<<"YES"<<endl;
    	 
    }
    return 0;
}
