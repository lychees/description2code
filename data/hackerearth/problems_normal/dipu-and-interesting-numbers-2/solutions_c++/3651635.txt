#include<iostream>
#include <cmath>

using namespace std;

int
main ()
{
  long long tc = 0;
  cin>>tc;
  while (tc--)
    {
      long long l = 0; long double r = 0;
      cin >> l >> r;

      long long lsq = (long long) sqrt (l);

      if (lsq * lsq != l)
        lsq++;

      long long rsq = (long long) floor(sqrt (r));

      cout << rsq - lsq + 1 << endl;
    }
}
