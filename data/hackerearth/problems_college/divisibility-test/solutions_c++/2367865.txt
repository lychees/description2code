#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	string s;
    	cin >> s;
    	int di;
    	cin >> di;
    	int l,r,i,j;
    	cin >> l >> r;
    	long long int count=0,sum;
    	for(i=l-1;i<r;i++)
    	{
    		sum = s[i]-48;
    		if(sum%di==0)
    		{
    			count+=1;
    			sum = 0;
    		}
    		else
    		sum = sum%di;
    		for(j=i+1;j<r;j++)
    		{
    			sum = 10*sum+(s[j]-48);
    			if(sum%di==0)
    			{
    				count+=1;
    				sum =0;
    			}
    			else
    			{
    				sum = sum%di;
    			}
    		}
    	}
    	cout << count << endl;
    }
    return 0;
}
