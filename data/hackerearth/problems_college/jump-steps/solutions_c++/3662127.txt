#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define size 60

int main()
{
	int test;
	
	ll arr[size];
	arr[0] = 1;
	arr[1] = 2;
	for(int i = 2; i<size; i++)
		arr[i] = arr[i-1] * 2;
	cin >> test;
	while(test--) {
		ll n, steps=0;
		int j=size-1;
		cin >> n;
		while(n>0 && j>=0) {
			if(arr[j] <= n) {
				n-=arr[j];
				j--;
			//	cout << "n = , j = " << n << j  << endl;
				steps++;
			}
			else
				j--;
		}
		cout << steps << endl;
	}
	return 0;
}
