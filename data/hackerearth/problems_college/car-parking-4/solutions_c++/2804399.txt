#include <iostream>
#include<string>
using namespace std;

int main()
{
    
    string loc;
    int i,count,T,n;
    cin>>T;
    count=0;
    while(T--)
    {
    	cin>>loc;
    	for(i=0;i<loc.length();i++)
    	{
    		if(loc.at(i)=='B')
    		{
    			if(i-1>=0 && loc.at(i-1)=='-')
    			loc[i-1]='*';
    			if(i-2>=0 && loc.at(i-2)=='-')
    			loc[i-2]='*';
    		}
    		if(loc.at(i)=='S')
    		{
    			if(i-1>=0 && loc.at(i-1)=='-')
    			loc[i-1]='*';
    			if(i+1<loc.length() && loc.at(i+1)=='-')
    			loc[i+1]='*';
    		}
    		
    	}
    	for(i=0;i<loc.length();i++)
    	{
    		if(loc.at(i)=='-')
    		count++;
    	}
    	cout<<count<<endl;
    	count=0;
    }
    return 0;
}
