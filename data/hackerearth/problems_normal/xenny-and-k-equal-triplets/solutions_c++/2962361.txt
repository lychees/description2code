#include<bits/stdc++.h>
using namespace std;
long long int fact(long long int n)
{
	long long fact=1;
	for(int i=2;i<n+1;i++)
		fact*=i;
	return fact;
}
long long int combination(long long int n,long long int r)
{
	if(n<=2)
		return 0;
	else
	{
		long long int tmp = n * (n - 1);
		tmp /= 2;
		tmp *= (n - 2);
		tmp /= 3;
		return tmp;
	}
}
int main()
{
        int t;
        scanf("%d",&t);
        while(t--)
        {
                long long int n,k,x;
                vector<int>v;
                long long int cnt=0;        
                scanf("%lld%lld",&n,&k);
                for(int i=0;i<n;i++)
                {
                        scanf("%lld",&x);
                        v.push_back(x);
                        if(x==k)
                                cnt++;
                }
                
                long long int poss=combination(cnt,3);
                long long int total=combination(n,3);
                
                long long int g=__gcd(poss,total);
                
                cout<<poss/g<<"/"<<total/g<<endl;
        }
}
