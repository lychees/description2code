#include<bits/stdc++.h>
using namespace std;
int n,a[1300][1300],b[1300][1300];
int res1[1<<15],dist[1<<15],used[1<<15];
int bp;
int solve(int v1,int v2)
{
for (int i=1;i<=n;i++)
	dist[i]=1e9;
dist[v1]=0;
for (int i=1;i<=n;i++)
	used[i]=0;
for (int iter=1;iter<=n;iter++)
{
	int bst=1e9;
	for (int j=1;j<=n;j++)
	{
		if (dist[j]<bst&&used[j]==0)
			bst=dist[j],
			bp=j;
	}
	used[bp]=1;
	for (int j=1;j<=n;j++)
		dist[j]=min(dist[j],a[bp][j]+dist[bp]);
}
for (int i=1;i<=n;i++)
	res1[i]=dist[i],
	dist[i]=1e9;
dist[v2]=0;
for (int i=1;i<=n;i++)
	used[i]=0;
for (int iter=1;iter<=n;iter++)
{
	int bst=1e9;
	for (int j=1;j<=n;j++)
	{
		if (dist[j]<bst&&used[j]==0)
			bst=dist[j],
			bp=j;
	}
	used[bp]=1;
	for (int j=1;j<=n;j++)
		dist[j]=min(dist[j],b[j][bp]+dist[bp]);
}
int res=1e9;
for (int i=1;i<=n;i++)
	if (i!=v1&&i!=v2)
		res=min(res,dist[i]+res1[i]);
return res;
}
int main(){
ios_base::sync_with_stdio(0);
//cin.tie(0);
cin>>n;
for (int i=1;i<=n;i++)
	for (int j=1;j<=n;j++)
		cin>>a[i][j];
for (int i=1;i<=n;i++)
	for (int j=1;j<=n;j++)
		cin>>b[i][j];
	
int v1,v2;
cin>>v1>>v2;
int ans=solve(v1,v2);
for (int i=1;i<=n;i++)
	for (int j=1;j<=n;j++)
		swap(a[i][j],b[i][j]);
ans=min(ans,solve(v1,v2));
cout<<ans;
//cin.get();cin.get();
return 0;}
