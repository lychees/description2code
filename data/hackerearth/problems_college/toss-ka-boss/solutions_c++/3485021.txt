#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	char ch[1000000];
   	cin>>ch;
   	long long int i = 0;
   	long long int count = 0,sum = 0;
   	while(ch[i]!= '\0')
   	{
   		if(ch[i] == 'H')
   		{
   			count++;
   		}
   		else
   		{
   			count = 0;
   		}
   		sum = sum+count;
   		i++;
   	}
   	cout<<sum<<endl;
   }
    return 0;
}
