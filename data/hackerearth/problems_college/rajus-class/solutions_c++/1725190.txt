#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
       int t;
       cin >> t;
       while(t--)
       {

             int n;
             cin >> n;
             int x;
             string y;
             vector<pair<int,string > > vec;
             while(n--)
             {
                 cin >> x;
                 cin >> y;
                 vec.push_back(make_pair(x,y));
             }
             sort(vec.begin(),vec.end());
             int r;
             cin >> r;
             cout << vec[r-1].second << endl;
       }
        return 0;
}
