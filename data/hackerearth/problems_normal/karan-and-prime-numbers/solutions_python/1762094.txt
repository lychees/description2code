def prime_gen(ma):
  c=range(3, ma+1, 2)
  i=0
  while c[i]<(ma+1)**0.5:
    j=i+1
    while j<len(c):
      if c[j] % c[i] == 0: del c[j]
      j+=1
    i+=1
  return [2]+c

tc = int(raw_input())
max = 0
queries = []
for i in range(tc):
	l,r = map(int,raw_input().split())
	queries.append( (l,r))
	if r>max:
		max=r

primes_list = prime_gen(max)

for l,r in queries:
	s = 0
	for el in primes_list:
		if el>=l and el<=r:
			s+=el
	print s
