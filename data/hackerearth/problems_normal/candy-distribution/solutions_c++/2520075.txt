#include<iostream>
#include<algorithm>
#include<cmath>
#include<unordered_map>
#include<cstring>
#define ll long long int
#define mod 1000000007
using namespace std;
ll dp[10000001];
ll power(ll a, ll b)
{
   ll ans=1;
   while(b)
   {
   	if(b&1)
   	ans=(ans*a)%mod;
   	a=(a*a)%mod;
   	b/=2;
   }
   return ans;
}
unordered_map<ll,ll> mp;
ll fun(ll n)
{
    if(n == 1 || n == 2)
        return 1;
    if(mp.find(n)!=mp.end())
        return mp[n];
    mp[n] =1;
    n--;
    for(ll i=2; i*i<=n; i++){
        if(n % i == 0){
            mp[n+1] += power(fun(i), n/i);
                mp[n+1] %= mod;
                if(i*i!=n)
        {
        	 mp[n+1] += power(fun(n/i), i);
                mp[n+1] %= mod;
        }
        }
        
    }
    return mp[n+1];
}

int main()
{
    ll n;
    //memset(dp, -1, sizeof(dp));
    cin>>n;
     cout<<fun(n);
}
