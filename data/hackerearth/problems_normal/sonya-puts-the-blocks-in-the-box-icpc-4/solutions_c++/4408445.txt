#include <bits/stdc++.h>
using namespace std;
#define ll long long 
ll dp[102][1002];
ll best[1005];
int main()
{
    int m,n,sz;
    cin>>n>>m;
    for(int i=1;i<=n;++i){
    	cin>>sz;
    	ll s[sz+1];
    	s[0]=0;
    	for(int j=1;j<=sz;++j){
    		cin>>s[j];
    		s[j]+=s[j-1];
    	}
    	
    	for(int j=1;j<=m;++j)
    		best[j]=-1e15;
    	best[0]=0;	
    	for(int j=1;j<=sz;++j){
    		for(int len=1;len<=m && j+len-1<=sz;++len){
    			best[len]=max(best[len],s[j+len-1]-s[j-1]);
    		}
    	}
    	for(int j=0;j<=m;++j){
    		for(int len=0;len<=sz && j+len<=m ;++len){
    			dp[i][j+len]=max(dp[i][j+len],dp[i-1][j]+best[len]);
    		}
    	}
    }
    cout<<dp[n][m]<<endl;
    return 0;
}
