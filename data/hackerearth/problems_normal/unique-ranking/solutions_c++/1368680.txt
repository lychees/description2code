#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007

int number[1005][21];

int main()
{
	int n,k,q,m,x;
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;++i)
		for(int j=1;j<=k;++j)
			scanf("%d",&number[i][j]);
	scanf("%d",&q);
	while(q--)
	{
		bool mark[n+1];
		for(int i=1;i<=n;++i)
			mark[i] = true;
		scanf("%d",&m);
		while(m--)
		{
			scanf("%d",&x);
			int mini = INT_MAX;
			for(int i=1;i<=n;++i)
				if(mark[i])
					mini = min(mini,number[i][x]);
			for(int i=1;i<=n;++i)
				if(mark[i])
					if(number[i][x]!=mini)
						mark[i] = false;
		}
		for(int i=1;i<=n;++i)
			if(mark[i])
			{
				printf("%d\n",i);
				break;
			}
	}
	return 0;
}
