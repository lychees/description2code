#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll dp[501];
bool is_prime(int n){
	if(n<=1){
		return false;
	}
	if(n==2){
		return true;
	}
	if(n%2==0){
		return false;
	}
	int ix=3;
	int lim=sqrt(n)+1;
	while(ix<=lim){
		if(n%ix==0){
			return false;
		}
		ix+=2;
	}
	return true;
}
void init(){
	int i=0;
	dp[0]=0;
	dp[1]=0;
	dp[2]=1;
	for(int i=3;i<=500;i++){
		if(is_prime(i)){
			dp[i]=2*dp[i-1];
		}
		else{
			dp[i]=dp[i-1];
		}
	}
}
int main(){
	init();
	int n;
	cin>>n;
	ll sum=0;
	for(int i=0;i<=n;i++){
		sum+=dp[i];
	}
	cout<<sum;
}
