#include<iostream>
#include<vector>
#include<set>
using namespace std;

vector<int> v(100001);

int root(int i){
	while( v[i] != i){
		i = v[i];
	}
	return i;
}

void uni(int A, int B){
	int rootA = root(A);
	int rootB = root(B);
	v[rootB] = rootA;
}

int main(){
	int N, K;
	cin>>N>>K;
	for(int i = 0 ; i < 100001 ; i++ )
		v[i] = -1;
	for(int i = 1 ; i <= N ; i++ ){
		v[i] = i;
	}
	while(K--){
		int i, j;
		cin>>i>>j;
		uni(i, j);
	}
	set<int> s;
	for(int i = 1 ; i <= N ; i++){
		s.insert( root(i) );
	}
	cout<<s.size();
	return 0;
}
