#include <iostream>
#include<bits/stdc++.h>
#define size 1000002
using namespace std;
long long int prime[size];
int main()
{
	long long int i,k,j;
      for(i=2;i<=size;i++)
	      prime[i]=1; 
     for(i=2;i<=sqrt(size);i++)
	      {
	          k=1;
	          if(prime[i]==1)
	          {
	              for(j=i*i;j<=size;)
	                {prime[j]=0;j=i*i+k*i;k++;}
	                
	          }
	      }
	      long long int sum=0;
	   for(i=2;i<size;i++)
	     {
	     	if(prime[i]==1)
	     	  sum+=i;
	     	 
	     	 prime[i]=sum;
	     }
	  // for(i=2;i<20;i++)
	   // cout<<prime[i]<<" ";  
	    long long int n;
	    cin>>n;
	    while(n--){
	    	long long int l,r;
	    	cin>>l>>r;
	        cout<<prime[r]-prime[l-1]<<endl;
	    }
    return 0;
}
