#include <iostream>
#include<math.h>
using namespace std;

int main()
{
     int i, j, n, z, five = 0, three = 0;
     cin>>n;
     while(n--)
     {
         a:
         cin>>z;
         if(z <= 2)
         {
             cout<<"-1"<<endl;
             n--;
             goto a;
         }

         for(i = 0; i<=sqrt(z); i++)
         {
                for(j = 0; j<=(z/2); j++)
                 {
                    if(z == (3*(i) + j*5))
                    {
                    five = 3 * i;
                    three = j*5;
                    break;
                    }

            }
            if(z == five + three)
            {
                break;
            }
         }
         for(i = 0; i<five;i++)
         {
             cout<<"5";
         }
         for(i = 0; i<three;i++)
         {
             cout<<"3";
         }
         cout<<endl;

     }
    return 0;
}
