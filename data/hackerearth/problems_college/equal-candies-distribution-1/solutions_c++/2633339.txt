#include<bits/stdc++.h>
using namespace std;

int main()
{
	int T,N,a;
	cin>>T;
	int res=0,temp=0;
	while(T--)
	{
			int res=0,temp=0;
			cin>>N;
		vector<int> vec;
		for(int i=0;i<N;i++)
		{
			cin>>a;
			vec.push_back(a);
		}
		sort(vec.begin(),vec.end());

		for(int i=1;i<N;i++)
		{
			int diff=vec[i]-vec[i-1];
			temp=temp+diff;
			res=res+diff/5;
			diff=diff%5;
			res=res+diff/2;
			diff=diff%2;
			res=res+diff;
			if(i!=N-1)
			vec[i+1]=vec[i+1]+temp;
		}
	
				cout<<res<<endl;
	}
     return 0;
}
