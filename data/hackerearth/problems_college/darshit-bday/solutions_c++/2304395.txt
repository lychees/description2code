#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	long long int  a,b,p,q,r,sum=0,k;
   	cin>>a>>b;
   	cin>>p>>q>>r;
   	if(p>q && p>q+r)
   	{
   
			k=q+r;
   			sum=k*a;
   	}
   	else
   			sum=p*a;
   	
   	if(q>p &&  q>p+r)
   		{
   			k=p+r;
   			sum=sum+(k*b);
   			
   		}
   		else
   			sum=sum+(q*b);
   	
   	if(p==q)
   	cout<<(a+b)*p<<endl;
   	else
   	cout<<sum<<endl;
 
   	
   	
   }
    return 0;
}
