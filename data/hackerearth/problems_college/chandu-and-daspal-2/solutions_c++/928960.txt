#include<iostream>
using namespace std;
void swap(int *a,int *b)
{
	int t=*a;
	*a=*b;
	*b=t;
}
int gcd(int n,int m)
	{ if(m<=n && n%m == 0)
	   return m;
	  if(n < m)
	   return gcd(m,n);
	  else
	   return gcd(m,n%m);
	 }
int solve(int x,int y)
{
	int l=0;
	
	while(x!=0 && y!=0)
	{
		int g=gcd(x,y);
		//cout<<"g="<<g<<"\n";
		x-=g;
		y-=g;
		//cout<<x<<y<<"\n";
		l++;
		
	}
	return l;
}
int main()
{
	int t,x,y;
	char s[10];
	cin>>t;
	for(int i=0;i<t;i++)
	{
		cin>>x>>y;
		cin>>s;
		int l=solve(x,y);
		if(l%2!=0)
		{
			cout<<s<<"\n";
		}
		else
		{
			if(s[0]=='C')
			 cout<<"Daspal\n";
			else
			 cout<<"Chandu\n";
		}
	}
	system("pause");
}
