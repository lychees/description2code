    #include <cstdlib>
    #include <iostream>
    #include <math.h>
    
    using namespace std;
    long long n,N,a,b,m,k,I,T,x;
    long long pow1, pow2; 
    int main(int argc, char *argv[])
    {   cin>>T;
        N=1000;
        for(I=1;I<=T;I++)
        { cin>>x;
          if(x==1)
             cout<<"No"<<endl;
          else
          if(x==2)
             cout<<"Yes"<<endl;
          else
          {
            k=0;
            for(a=1;a<=N;a++)
            {  pow1=a;
               for(m=2;m<=N;m++)
               { pow1*=a;
                 if(pow1>=x)
                    break;
                 else
                 {
                    for(b=1;b<=N;b++)
                    {  pow2=b;
                       for(n=2;n<=N;n++)
                       {  pow2*=b;
                          if( pow2>=x)
                             break;
                          else
                          {   
                             if(pow1+pow2==x)
                             {  k=1;
                                 goto mm;
                             }
                          }
                       }
                    }
                 }
               }
            }
            mm:
             if(k==1)
                cout<<"Yes"<<endl;
             else
                cout<<"No"<<endl;
          }  
        }
        
        return EXIT_SUCCESS;
    }
