bool dp[10002][10002];
int a[10002];
int main()
{
    int n;
    cin >> n;
    int i;
    int maxi = 0;
    for(i = 0 ;i < n ;i++)
    {
        cin >> a[i];
    }
    sort(a,a+n);
    int j = n - 1;
    int ans = 0;
    for(i = 0 ; i < n ; i++)
    {
       for(j = n - 1; j >= 0 ; j--)
       {
           ans = a[j] - 3*a[i];
           if(ans <= 0)
           {
              if(maxi < j - i)
              {
                  maxi = j - i;
              }
           }

       }
    }
    maxi++;
    cout<<n-maxi<<endl;
    return 0;
}
