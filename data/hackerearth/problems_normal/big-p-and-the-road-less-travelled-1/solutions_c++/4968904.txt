#include<bits/stdc++.h>
using namespace std;

vector<int> g[10001];

int dp[10001];
int main()
{
	int x,n,y,ans=0;
	cin>>n;
	while(1)
	{
		cin>>x>>y;
		if(x==0 && y==0)
		break;
		g[x].push_back(y);
	}
	
	dp[1]=1;
	for(int i=1;i<=n-1;i++)
	{
		int x=g[i].size();
		
		
		for(int j=0;j<x;j++)
		{
			int temp=g[i][j];
			//cout<<"temp="<<temp<<"\n";
			if(dp[i]>=1 && temp!=n)
			dp[temp]=dp[temp]+dp[i];
			//if(i==1)
			//dp[temp]-=1;
		}
	}
	
	//for(int i=1;i<=n;i++)
	//cout<<"dp="<<dp[i]<<"\n";
	for(int i=1;i<=n-1;i++)
	{
		int x=g[i].size();
		int flag=0;
		for(int j=0;j<x;j++)
		{
			if(g[i][j]==n)
			{
				flag=1;
				break;
			}
		}
		if(flag==1)
		{
			//cout<<"jhi\n";
			if(i==1)
			ans+=(dp[i]);
			else
			if(dp[i]>0)
			{
			ans+=(dp[i]);
		    }
		}
	}
	cout<<ans<<"\n";
	return 0;
}
