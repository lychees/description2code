#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int prime[1500002];
vector<int>arr;
vector<int>brr;
void sieve()
{
	int i,j;
	prime[1] = prime[0] = 1;
	for(i = 2; i <= 1500000; i++) {
		if(!prime[i]) {
			for(j = 2; j*i <= 1500000; j++) {
				prime[j*i] = 1;
			}
			arr.pb(i);
		}
	}
	int c = 1;
	for(i = 2; i <= 1500000; i++) {
		if(!prime[i]) {
			if(!prime[c]) {
				brr.pb(i);
			}
			c++;
		}
	}
}
int main() 
{
	sieve();
	int t,n,m;
	scanf("%d",&t);
	while(t--) {
		scanf("%d%d",&n,&m);
		ll x = brr[n-1];
		x = x*brr[m-1]%mod;
		int y = x;
		printf("%d\n",y);
	}
	return 0;
}
