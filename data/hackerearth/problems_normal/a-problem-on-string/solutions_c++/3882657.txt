#include <bits/stdc++.h>

int main()
{
    //cout << "Hello World!" << endl;
    long long int i,j,k,m,n,t,q;
    char x,y;
    string s;
    cin>>n;
    cin>>s;
    long long int a[30]={0};
    for(i=0;i<n;i++)
    {
    	a[s[i]-'a']++;
    }
    cin>>q;
    while(q--)
    {
    	cin>>x>>y;
    	if(x==y)
        {
                if(a[x-'a']==1) k=0;
                else k=(a[x-'a']*(a[x-'a']-1))/2;
        }
        else
    	k=a[x-'a']*a[y-'a'];
    	//if(k==1) k=0;
    	cout<<k<<endl;
    }

}
