#include <bits/stdc++.h>
using namespace std;
#define endl '\n';
int dp[4][100005],v[4][100005];
int main()
{
      ios_base::sync_with_stdio(false);cin.tie(0);
      //freopen("input.in","r",stdin);
          int t;cin>>t;while(t--)
          {
            int n;cin>>n;
            for(int i=1;i<=n;i++)
                  cin>>v[1][i]>>v[2][i]>>v[3][i];
            for(int i=1;i<=n;i++)
            {
              for(int j=1;j<=3;j++)
              {
                int x,y;
                if(j==1){x=2;y=3;}
                if(j==2){x=1;y=3;}
                if(j==3){x=1;y=2;}
                 dp[j][i]=v[j][i]+min(dp[x][i-1],dp[y][i-1]);
              }
            }
            cout<<min(min(dp[1][n],dp[2][n]),(dp[3][n]))<<endl;
          }           
    return 0;
}
