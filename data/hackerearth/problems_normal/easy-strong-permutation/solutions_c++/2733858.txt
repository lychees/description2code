#include <bits/stdc++.h>
#define lli long long

using namespace std;


int main()
{
	int n;
	vector <lli> v1,v;
	lli x;
	
	cin >> n;
	
	for ( int i = 0; i < n; i++ ) {
		cin >> x;
		v.push_back(x);
	}
	sort(v.begin(),v.end());
	for ( int i = 0; i < n; i++ ) {
		if ( i > n-i-1 ) break;
		v1.push_back(v[i]);
		if ( i != n-i-1 ) v1.push_back(v[n-i-1]);
	}
	lli ans = 0;
	for ( int i = 0; i < n; i++ ) ans += abs(v1[i]-v1[(i+1)%n]);
	cout << ans << endl;
	return 0;
}
