#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
 
bool comp(pair<ll,pair<ll,ll> > x,pair<ll,pair<ll,ll> > y){
 if(x.first!=y.first)return x.first<y.first;   
 if(x.second.first!=y.second.first)return x.second.first>y.second.first;
 return x.second.second<y.second.second;
 
}
 
int main()
{ ios_base::sync_with_stdio(0); cin.tie(NULL);
 
 
 
ll n,d;
cin>>n>>d;
pair<ll,pair<ll,ll> > a[n];
for(int i=0;i<n;i++){
    int p,q,r;
    cin>>p>>q>>r;
    a[i]=make_pair(p,make_pair(r,q));
}
sort(a,a+n,comp);
ll sp,tot=0,cost=0,tim;
sp=a[0].second.first,tim=a[0].first;cost+=a[0].second.second;
for(ll i=1;i<n;i++){
    if(a[i].first==tim)continue;
    tot+=(a[i].first-tim)*sp;
    if(tot>=d)break;
    tim=a[i].first;
    if(a[i].second.first>sp){sp=a[i].second.first;cost+=a[i].second.second;}
//    cout<<cost<<endl;
    
    
}
 
cout<<cost;
 
return 0;
}
