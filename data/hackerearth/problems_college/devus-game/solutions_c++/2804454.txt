#include<bits/stdc++.h>

using namespace std;

const int N=1234567;

int mark[N], a[N];

int main() {
    int test;
    scanf("%d", &test);
    while(test--) {
        int n;
        scanf("%d", &n);
        for(int i=0;i<n;i++) {
            scanf("%d", a+i);
        }
        bool ok=true;
        int dest=0;
        while(dest!=-1) {
            mark[dest]=1;
            dest=a[dest];
            if(mark[dest]) {
                ok=false;
                break;
            }
        }
        puts(ok?"Win":"Lose");
        memset(mark, 0, sizeof(mark));
    }
}
