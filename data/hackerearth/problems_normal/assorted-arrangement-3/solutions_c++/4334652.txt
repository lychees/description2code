#include <iostream>
#define MAX1 50
#define MAX2 100000

using namespace std;

int n, m;
int col[MAX1];
int copy_col[MAX1];
int list[MAX2];
int pre;

void UpdateCol( int ind)
{
    while(col[ind] < pre+1)
        col[ind] += copy_col[ind];

    for(int i=ind+1; i<m; )
    {
        if( col[ind]%copy_col[i] == 0)
        {    
            col[ind] += copy_col[ind];
            i = ind+1;
            continue;
        }
        i++;
    }        
}


void Algo()
{
	int col_ind;
    col_ind = list[0] - 1;

    list[0] = col[list[0]-1];
    pre = list[0];
    UpdateCol(col_ind);

    for(int i=1; i<n; i++)
    {
        if(col[list[i]-1] < list[i-1])
            UpdateCol(list[i]-1);

        col_ind = list[i] - 1;
        list[i] = col[list[i]-1];                
        pre = list[i];
        UpdateCol(col_ind);        
    }

    cout << list[n-1];
}

void Input()
{

    cin >> n >> m;
    for(int i=0; i<m ; i++)
    {
        cin >> col[i];
        copy_col[i] = col[i];
    }

    for(int i=0; i<n; i++)
        cin >> list[i];
}

int main()
{
    Input(); 
    Algo();

    return 0;
}
