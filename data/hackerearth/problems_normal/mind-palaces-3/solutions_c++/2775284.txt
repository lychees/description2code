#include <bits/stdc++.h>

#define LL long long int

using namespace std;

vector<vector<LL> > v;

int main()
{
    int q,n,m;
    scanf("%d %d",&n,&m);
    v.resize(n);
    for(int i=0;i<n;++i)
    {
        for(int j=0;j<m;++j)
        {
            LL e;
            scanf("%lld",&e);
            v[i].push_back(e);
        }
    }
    scanf("%d",&q);
    while(q--)
    {
        LL x;
        scanf("%lld",&x);
        int flag = 0;
        for(int i=0;i<v.size();++i)
        {
            if (binary_search(v[i].begin(),v[i].end(),x))
            {
                for(int j=0;j<v[i].size();++j)
                {
                    if (v[i][j] == x)
                    {
                        printf("%d %d\n",i,j);
                        flag = 1;
                        break;
                    }
                }
            }
            if (flag)
                break;
        }
        if (!flag)
        {
            printf("-1 -1\n");
        }
    }
    return 0;
}
