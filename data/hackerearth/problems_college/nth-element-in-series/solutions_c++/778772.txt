#define MOD 1000000007
int b[5][5]={{18,18,19,22,9},{9,9,9,10,4},{4,5,5,5,2},{2,2,3,3,1},{1,1,1,2,1}};
#define MATSIZE 5
void mult(int c[][MATSIZE], int a[][MATSIZE])
{
		ll b[MATSIZE][MATSIZE];
		int i,j,k;
		for(i=0;i < MATSIZE;i++)
			for(j=0;j < MATSIZE;j++){
				b[i][j]=0;
				for(k=0;k < MATSIZE;k++)
					b[i][j]=(b[i][j]+(ll)a[i][k]*c[k][j])%MOD;
			}
		for(i=0;i < MATSIZE;i++)
			for(j=0;j < MATSIZE;j++)
				c[i][j]=b[i][j];
}
void pow(int c[][5], int n){
	int i,j,res[5][5];
	f(i,5)
		f(j,5){
			if(i==j)
				res[i][j]=1;
			else
				res[i][j]=0;
			c[i][j]=b[i][j];
		}
	while(n){
		if(n&1)
			mult(res,c);
		n>>=1;
		mult(c,c);
	}
	for(i=0;i < MATSIZE;i++)
		for(j=0;j < MATSIZE;j++)
			c[i][j]=res[i][j];
}
