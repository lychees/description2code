from collections import Counter
m, n = map(int, raw_input().split())
r = Counter(map(int, raw_input().split()))
c = sorted(r.keys(), reverse=1)
for i in range(c[0]-1, 0, -1):
    r[i] += r[i+1]
ans = 0
for i in range(c[0], 0, -1):
    x = min(r[i], n)
    ans += x * i
    n -= x
    if n == 0:
        break
print ans
