ll nth_fib(ll x)
{
	ll a=0,b=1;
	if(x==1) return 1;
	for(long i=2;i<=x;i++)
	{
		ll temp=(a+b);
		a=b;
		b=temp;
	}
	return b;
}	


int main()
{
	ios::sync_with_stdio(0);
	int t;cin>>t;
	while(t--)
	{
		ll n,i,sum=0;cin>>n;
		for(i=3;nth_fib(i)<n;i+=3)sum+=nth_fib(i);
		cout<<sum<<"\n";
	}
}
