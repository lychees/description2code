#include <iostream>
using namespace std;

int main()
{
    long long int t,n,a,b,i;
    bool flag;
    cin>>t;
    while(t--)
    {
        flag=0;
        cin>>n;
        cin>>a;
        for(i=1;i<n;i++)
        {
            cin>>b;
            if(b<a)
                flag=1;
            a=b;
        }
        if(flag==1)
            cout<<"NO"<<endl;
        else
            cout<<"YES"<<endl;
    }
    return 0;
}
