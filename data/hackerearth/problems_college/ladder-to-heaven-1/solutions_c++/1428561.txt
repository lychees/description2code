#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
typedef long long ll;

ll a[10008],s[2][10008];

ll find(ll n){
	ll i;
	n++;
	for(i=0;i<31;i++)
		if(n<=s[1][i])
			break;
		
	return s[0][i-1];
}
		

int main()
{
    ll i,p,t,n;
    a[0]=0;
    a[1]=1;
    a[2]=2;
    
    p=1;
    s[0][0]=2;
    s[1][0]=2;
    for(i=3;i<90;i++){
    	a[i]=a[i-1]+a[i-2];
    	if(i%3==2){
    		s[0][p]=s[0][p-1]+a[i];
    		s[1][p]=a[i];
    		p++;
    	}
    }
    cin>>t;
    while(t--){
    	cin>>n;
    	cout<<find(n)<<endl;
    }
    return 0;
}
