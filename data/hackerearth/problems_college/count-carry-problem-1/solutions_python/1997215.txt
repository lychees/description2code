t=int(input())
for tc in range(t):
	s1,s2=raw_input().split()
	l1,l2=len(s1),len(s2)
	if l1<=l2:
		x,y=s1,s2
	else:
		x,y=s2,s1
	count=0
	l=len(y)
	for i,j in enumerate(x[::-1]):
		if int(y[l-1-i])+int(j)>9:
			count+=1
	if count==0:
		print 'No carry operation'
	elif count==1:
		print "1 carry operation"
	else:
		print str(count)+" carry operations"
