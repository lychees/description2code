#include <algorithm>
#include <cmath>
#include <iostream>
using namespace std;
int main() {
  int l = 0, r = 0, n, f = 0, a, b;
  cin >> n;
  int i;
  for (i = 0; i < n; i++) {
    cin >> a >> b;
    if (a % 2 != b % 2)
      f = 1;
    l += a;
    r += b;
  }
  if (l % 2 == 0 && r % 2 == 0)
    cout << 0 << endl;
  else if (l % 2 == 1 && r % 2 == 1 && f == 1)
    cout << 1 << endl;
  else
    cout << -1 << endl;
}
