#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,time,val,val1,n,x,i;
    ios_base::sync_with_stdio(false);
    cin.tie(false);
    cin>>t;
    while(t--)
    {
        cin>>n>>x;
        priority_queue<int>a;
        for(i=0;i<n;i++)
        {
            cin>>time;
            a.push(time);
        }
        time=0;
        while(1)
        {
            val=a.top();
            a.pop();
            if(a.empty()){
                time+=val;
                break;}
            val1=a.top();
            a.pop();
            time+=val1;
            val-=val1;
            a.push(val);
        }
        if(time<=x)
            cout<<"YES\n";
        else
            cout<<"NO\n";
    }
    return 0;
}
