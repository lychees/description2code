#include <bits/stdc++.h>
using namespace std;

int main()
{
	int rows, cols, num, max = 0, temp;
	cin >> rows >> cols;
	int rowsArray[rows], colsArray[cols];
	
	for(int i = 0; i < rows; i++) {
		rowsArray[i] = 0;
	}
	for(int i = 0; i < cols; i++) {
		colsArray[i] = 0;
	}
	
	for(int i = 0; i < rows; i++) {
		for(int j = 0; j < cols; j++) {
			cin >> num;
			rowsArray[i] += num;
			colsArray[j] += num;
		}
	}
	
	for(int i = 0; i < rows; i++) {
		for(int j = 0; j < cols; j++) {
			temp = rowsArray[i] ^ colsArray[j];
			//cout << "row: " << i+1 << " " << rowsArray[i] << " ; "<< " col: " << j+1 << " " << colsArray[j] << " xor: "<< temp << endl;
			if(temp > max)
				max = temp;	
		}
	}
	
	cout << max;
	
	
    return 0;
}
