#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	vector< pair<int,int> > time;
    	int a[n],b[n];
    	for(int i=0;i<n;i++)
    	 scanf("%d",&a[i]);
    	
    	for(int i=0;i<n;i++)
    	 scanf("%d",&b[i]);
    	
    	for(int i=0;i<n;i++)
    	    time.push_back(make_pair(a[i],1));
    	 
    	for(int i=0;i<n;i++)
    	    time.push_back(make_pair(a[i]+b[i],-1));
    	  
    	sort(time.begin(),time.end());
    	int l=time.size();
    	int dp[l];
    	
    	dp[0]=time[0].second;
    	for(int i=1;i<l;i++)
    	  dp[i]=dp[i-1]+time[i].second;
    	  
    	int ans=0;
    	
    	for(int i=0;i<l;i++)
    	  if(dp[i]>ans)
    	    ans=dp[i];
    	
    	
    	printf("%d\n",ans);
    	 
    }
    return 0;
}
