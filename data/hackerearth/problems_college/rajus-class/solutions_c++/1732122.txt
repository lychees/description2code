#include<bits/stdc++.h>
using namespace std;

typedef struct sats
{
int age;
char name[1001];
}sats;

int compare(const void *a , const void *b)
 {
       int x,y;
       x =( (sats *)a)->age;
       y =( (sats *)b)->age;
      // cout<<x<<"   "<<y<<"\n";
       return x-y;
 }

int main()
{
int t; cin>>t; while(t--)
{
int n,roll,i;
cin>>n;
sats c[n];
for(i=0;i<n;i++)
 {
       cin>>c[i].age>>c[i].name;
 }
 qsort(c,n,sizeof(sats),compare);
cin>>roll;
cout<<c[roll-1].name<<"\n";
}
return 0;
}
