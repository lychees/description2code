#include <iostream>

using namespace std;


void sorting(int *left,int *right,int *a,int nl,int nr)
{
    int i=0,j=0,k=0;

	while(i<nl && j<nr)
	{
		if(left[i] < right[j])
		{
			a[k] = left[i];
			i++;
		}
		else
		{
			a[k] = right[j];
			j++;
		} 
		k++;

	}

	while(i<nl)
	{
		a[k] = left[i];
		i++;
		k++;
	}
	while(j<nr)
	{
		a[k] = right[j];
		j++;
		k++;
	}
}
void sort(int *a,int n)
{
	int i;
	int left[n/2];
	int right[n-n/2];
	if(n<2)
		return ;

	for(i=0;i<n/2;i++)
	{
		left[i] = a[i];
	}
	for(i=0;i<(n-n/2);i++)
	{
		right[i] = a[n/2+i];
	}

	sort(left,n/2);
	sort(right,(n-n/2));
	sorting(left,right,a,n/2,n-n/2);

}


int main()
{

}
