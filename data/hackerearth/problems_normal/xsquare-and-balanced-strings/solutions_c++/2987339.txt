#include <iostream>
#include<string>
using namespace std;

int main()
{
int tc;
cin>>tc;
while(tc--)
{
	string s;
	cin>>s;
	int arr[26];
	for(int i=0;i<26;i++)
	arr[i]=0;
	
	for(int i=0;i<s.size();i++)
	arr[s[i]-'a']++;
	int f=1;
	for(int i=0;i<26;i++)
	{
		if(arr[i]%2!=0)
		{
			f=0;
			break;
		}
	}
	if(f)
cout<<"1"<<endl;
else
cout<<"-1"<<endl;
}


	
    //cout << "Hello World!" << endl;
    return 0;
}
