#include <bits/stdc++.h>
using namespace std;

typedef long long int LL;


int main()
{
	LL N,num,q;
	cin>>N;
	vector<LL> V;
	vector<LL>::iterator low,up;
	for(LL i=0;i<N;i++){
		cin>>num;
		V.push_back(num);
	}
	sort(V.begin(), V.end());
	cin>>q;
	while(q--)
	{
		LL L,R;
		cin>>L>>R;
		low=lower_bound (V.begin(), V.end(), L); 
  		up= upper_bound (V.begin(), V.end(), R);
  		cout<<((up-V.begin())-(low-V.begin()))<<endl; 
	}
	return 0;
}
