#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t,i,count,j,k,l,flag,a;
    cin>>t;
    int digits[10000];
    string s;
    
    for(i=0;i<t;i++)
    {
    	count = 0;
    	cin>>s;
    	
    	l = s.length();
    	
    	for(j=0;j<l;j++)digits[count++] = int(s[j]-48);
    	
    	
    	a = digits[0];
    	
    	for(j=1;j<l;j++)
    	{
    		if(digits[j]<l && digits[j]>0)a = digits[j];
    	}
    	
    	
    	for(j=a;j>=1;j--)
    	{
    		flag = 1;
    		for(k=0;k<count;k++)
    		{
    			
    			if(digits[k] % j !=0)
    			{
    				flag = 0;
    				break;
    			}
    			
    		}
    		
    		if(flag==1)break;
    		
    	}
    	
    	
    	for(k=0;k<l;k++)cout<<digits[k]/j;
    	
    	cout<<"\n";
    }
    return 0;
}
