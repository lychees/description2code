#include <iostream>
#include <algorithm>

using namespace std;

{
    int t;
    long long a,b,ans1,ans2;

          cin >> a >> b;
          ans2 = a-1;
          ans1 = b-1;
          cout << ans1 << " " << ans2 << endl;
    }
    
    return 0;
}
