#include <iostream>
using namespace std;

int main()
{
    long long t;
    cin>>t;
    while(t--)
    {
    	long long n,cont=0;
    	cin>>n;
    	while(n>0)
    	{
    		if(n&1)
    		cont++;
    		n=n/2;
    	}
       if(cont%2)
       cout<<"odd"<<endl;
       else
       cout<<"even"<<endl;
    }
    return 0;
}
