import sys


def f(n):
    if n <= 1:
        return 1
    else:
        return n*f(n-1)


n1, n2 = map(int, sys.stdin.readline().split())
if (n1 >= n2):
    print(f(n1-1)/(f(n1-n2)*f(n2-1))) % 10000007
else:
    print -1
