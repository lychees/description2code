def rec_sum(n):
    if n<10:
        return n
    else:
        return n%10+rec_sum(n/10)
        
t=int(raw_input())
for _ in range(t):
    m=int(raw_input())
    ans=0
    for i in range(m):
        l,d=map(int,raw_input().split())
        ans+=(d*l)
        ans=rec_sum(ans)
    print rec_sum(ans)
