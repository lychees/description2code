#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
int main()
{
    int t,n,a[1000000];
    cin >> t;
    a[0]=0;
    a[1]=1;
    a[2]=2;
    a[3]=3;
    for(int i=4; i<=1000000;i++)
    a[i]=((a[i-2]%mod) + (a[i-1]%mod))%mod;
    while(t--)
    {
    	cin >> n;
    	cout << a[n] <<"\n";	
    }
    
    
    
    
    return 0;
}
