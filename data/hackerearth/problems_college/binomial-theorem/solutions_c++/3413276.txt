#include <iostream>
using namespace std;

int Combination(int n, int r)
{
    if(n == r || r == 0)
        return 1;
    else
    {
        return (Combination(n-1,r) + Combination(n-1, r-1));
    }
}

int main()
{
  int M;
  cin >> M;
 
  int k = 0, n = 0;
  for(int i = 0; i < M; i++){
    cin >> k >> n;
    unsigned long long int coeffx = 0, coeffy = 0, maincoeff = 0;


    coeffx = n - (k - 1);
    coeffy = n - coeffx;
    maincoeff = Combination(n,k-1);
    
    if(maincoeff > 1)
    {
      cout << maincoeff << " ";
      if(coeffx > 1)
        cout << "x^" << coeffx << " ";
      else if(coeffx == 1)
        cout << "x^1 ";
      else cout << "";
      
      if(coeffy > 1)
        cout << "y^" << coeffy << " ";
      else if(coeffy == 1)
        cout << "y ";
      else cout << "";
    }
    else if(maincoeff == 1)
    {
      if(coeffx > 1)
        cout << "x^" << coeffx << " ";
      else if(coeffx == 1)
        cout << "x^1 ";
      else cout << "";
      
      if(coeffy > 1)
        cout << "y^" << coeffy << " ";
      else if(coeffy == 1)
        cout << "y ";
      else cout << "";
    }
    else cout << "";
    cout << endl;
  }
  return 0;
}
