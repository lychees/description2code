#include <bits/stdc++.h>
using namespace std;
#define LI long int
#define LLI long long int 

LI N,M;
LLI arry1[50005];
LLI arry2[50005];
LLI merged[250005];

void merge()
{
	LI p=0,q=0,k=0;

	for(LI k=0;k<N+M;k++)
	{
		if(p>N-1)
			merged[k] = arry2[q++]; //if p reached to the end

		else if(q>M-1)
			merged[k] = arry1[p++];	// if q reached to the end

		else if(arry1[p] >= arry2[q])
			merged[k] = arry1[p++];	// otherwise compare the shit!

		else
			merged[k] = arry2[q++];
	}
}

int main()
{
	int testc;
	scanf("%d",&testc);

	for(int i=0;i<testc;i++)
	{
		scanf("%ld %ld",&N,&M);
		for(LLI j=0;j<N;j++)
			scanf("%lld",&arry1[j]);
		for(LLI j=0;j<M;j++)
			scanf("%lld",&arry2[j]);
		merge();
		for(int j=0;j<N+M;j++)
		{
			printf("%lld ",merged[j]);
		}
		printf("\n");
	}
}
