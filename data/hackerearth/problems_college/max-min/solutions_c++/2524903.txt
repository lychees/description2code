
#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<limits.h>

using namespace std;

#define ll long long int

int main()
{
    ll i,n,k,j,diff;
    scanf("%lld",&n);
    scanf("%lld",&k);

    ll a[n];

    for(i=0;i<n;i++)
        scanf("%lld",&a[i]);

    sort(a,a+n);

    i=0;
    j=k-1;
    diff=INT_MAX;

    while(j<n)
    {
        if(diff>(a[j]-a[i]))
            diff=(a[j]-a[i]);
        i++;
        j++;
    }
    printf("%lld\n",diff);
    return 0;


}
