#include <bits/stdc++.h>
using namespace std;
struct node{
	int first;
	int scnd;
};
int cmp(node a,node b)
{
	if(a.scnd<=b.scnd) return false;
	else return true;
}

 int count(long long arr[],long long n, long long k)
 {
      long long num=0;
      node c[1000];
      for(long long i=0;i<n;i++)
      {
      	long long temp=arr[i];
      	num=0;
      	while(temp)
      	{
      		temp=temp&(temp-1);
      		num++;
      	}
      	 c[i].first=i;
      c[i].scnd=num;
      }
     sort(c,c+n,cmp);
     long long sum=0;
     for(long long i=0;i<k;i++)
     sum=sum + c[i].scnd;
     return sum;
 }
int main()
{
    long long t;

    {
    	long long n,k;
    	cin>>n>>k;
    	long long arr[n];
    	for(long long i=0;i<n;i++)
        cin>>arr[i];
