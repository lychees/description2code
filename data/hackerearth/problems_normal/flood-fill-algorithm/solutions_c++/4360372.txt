#include <iostream>
#include <vector>

using namespace std;
int n,m;
int x1,y1,x2,y2;
vector<vector<int>>mat(12,vector<int>(12));
vector<vector<int>>vis(12,vector<int>(12));

void dfs(int x, int y)
{
    vis[x][y] = 1;
    if(!vis[x-1][y] && mat[x][y]<mat[x-1][y])
    {
        dfs(x-1,y);
    }
    if(!vis[x+1][y] && mat[x][y]<mat[x+1][y])
    {
        dfs(x+1,y);
    }
    
    if(!vis[x][y-1] && mat[x][y]<mat[x][y-1])
    {
        dfs(x,y-1);
    }
    
    if(!vis[x][y+1] && mat[x][y]<mat[x][y+1])
    {
        dfs(x,y+1);
    }
}

auto main()->int
{
    int t;
    cin>>t;
    while(t--)
    {
        vector<vector<int>>m_mat(12,vector<int>(12));
        vector<vector<int>>v_vis(12,vector<int>(12));
        mat = m_mat;
        vis = v_vis;
        cin>>n>>m>>x1>>y1>>x2>>y2;
        for(auto i=1;i<=n;i++)
        {
            for(auto j=1;j<=m;j++)
            {
                cin>>mat[i][j];
            }
        }
        
        dfs(x1,y1);
        
        if(vis[x2][y2])
        {
            cout<<"YES"<<endl;
        }
        else
        {
            cout<<"NO"<<endl;
        }
    }
}
