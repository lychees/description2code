#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int x, n;
    	cin >> x >> n;
    	if(n <= x)
    		cout << (x - n) << endl;
    	else {
    		int count = 0;
    		while(n > x) {
    			if(n & 1) {
    				n++;
    				count++;
    			} else {
    				n /= 2;
    				count++;
    			}
    		}
    		if(n < x) count += x - n;
    		cout << count << endl;
    	}
    }
    return 0;
}
