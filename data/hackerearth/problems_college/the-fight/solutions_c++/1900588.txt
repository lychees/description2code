
#include <bits/stdc++.h>
using namespace std;

#define ll long long int


ll arr[1000100];
ll b[1000100] = {0};

int main() {
	ll n,m,k;
	scanf("%lld",&n);
	ll i,j;
	
	for(i = 1;i <= n;i++) {
		scanf("%lld",&arr[i]);
	}
scanf("%lld",&m);
	while(m--) {
		ll l,r;
		scanf("%lld",&l);
		scanf("%lld",&r);
		scanf("%lld",&k);
		b[l] += k;
		b[r + 1] -= k;
	}
	ll val = 0;
	for(i = 1;i <= n;i++) {
		val += b[i];
		cout << arr[i] + val << "\n";
	}
		
}
