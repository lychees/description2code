typedef long long int ll;
const int mx = 1e6 + 10;

int t, n, v[mx];

int main(){
//    ios_base::sync_with_stdio(0);
//    cin.tie(0);
//    freopen("data.in","r",stdin);
    scanf("%d", &t);
    while(t--){
        scanf("%d", &n);
        for(int i = 0 ; i < n; i++){
            scanf("%d", &v[i]);
        }
        int m = -1;
        int c = 0;
        for(int i = n - 1; i >= 0; i--){
            if(v[i] > m)c++;
            m = max(m, v[i]);
        }
        printf("%d\n", c);

    }
    return 0;
}
