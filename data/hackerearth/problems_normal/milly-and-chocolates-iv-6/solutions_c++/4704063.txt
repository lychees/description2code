#include <iostream>
#include <limits.h>
#include <vector>
using namespace std;

int main()
{
	int t=0;
	cin>>t;
	while(t--)
	{
		int n,m,k;
		cin>>n>>k>>m;
		vector<int> numPieces(n,0);
		for(int i=0;i<n;i++)
		{
			cin>>numPieces[i];
		}
		vector<vector<int>> times;
		for(int i=0;i<n;i++)
		{
			int num = numPieces[i];
			vector<int> t;
			for(int j=0;j<num;j++)
			{
				int a =0;
				cin>>a;
				t.push_back(a);
			}
			times.push_back(t);
		}
		long long int totMin = LLONG_MAX;
		int index = 0;
		for(int i=0;i<n;i++)
		{
			long long int total = 0;
			int num = numPieces[i];
			total += k*num + m*(num-1);
			for(int j=0;j<num;j++)
			{
				total+=times[i][j];
			}
			
			if(totMin>total)
			{
				totMin = total;
				index = i;
			}
		}
		cout<<index+1<<" "<<totMin<<endl;
	}
    // cout << "Hello World!" << endl;
    return 0;
}
