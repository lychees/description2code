#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;

const int mx=2e3+10;
char s[mx][mx];
uli hsh[mx][mx];

uli dig(char ch){
   if(ch=='_')return 0;
   return 1;
}
int main(){
   int n,m;
   scanf("%d %d",&n,&m);
   if(n<32 || m<32){
      puts("0");
      return 0;
   }
   for(int i=0;i<n;i++)
      scanf("%s",s[i]);   
   for(int i=0;i<n;i++){  
      uli h=0;
      for(int j=0;j<32;j++){
         h=h*2ll+dig(s[i][j]);
      }
      hsh[i][0]=h;
      for(int j=1;j+32<=m;j++){
         hsh[i][j]=hsh[i][j-1];
         if(hsh[i][j]&(1ll<<31))hsh[i][j]^=(1ll<<31);
         hsh[i][j]<<=1;
         hsh[i][j]+=dig(s[i][j+31]);
      }
   }
   string s="_#_#_#_#_#_#_#_#_#_#_#_#_#_#_#_#";
   string t="#_#_#_#_#_#_#_#_#_#_#_#_#_#_#_#_";
   uli x=0,y=0;
   for(int i=0;i<32;i++)x=(x*2ll+dig(s[i]));
   for(int i=0;i<32;i++)y=(y*2ll+dig(t[i]));

   int cnt=0;
   for(int i=0;i<n;i++){  
      for(int j=0;j+32<=m;j++){
         bool ok=true;
         for(int k=0;k<32 && ok;k++){
            if(k%2==0 && hsh[i+k][j]!=x)ok=false;
            if(k%2==1 && hsh[i+k][j]!=y)ok=false;
         }
         if(ok)cnt++;
      }
   }
   printf("%d\n",cnt);
   return 0;
}
