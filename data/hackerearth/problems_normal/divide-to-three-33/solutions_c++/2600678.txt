#include <bits/stdc++.h>
using namespace std;

int n,ans=1500;
vector<int> a(13);
void fun(int i,int j,int k, int c)
{
	if(c==n)
		{
			int v1=max(i,max(j,k));
			ans=min(ans,v1);
			return;
		}
	fun(i+a[c],j,k,c+1);	
	fun(i,j+a[c],k,c+1);
	fun(i,j,k+a[c],c+1);
	
	
}

int main()
{
	scanf("%d",&n);
    for(int l=0;l<n;l++)
    	scanf("%d",&a[l]);
    fun(0,0,0,0);
    cout<< ans;
    return 0;
}
