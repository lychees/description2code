def search(min, max, target):
    if trans[min] >= target:
        return min
    elif trans[max] == target:
        return max
    elif trans[max] < target:
        return -2
    else:
        mid = int(float((min+max)/2+(min+max) % 2))
        if trans[mid] == target:
            return mid
        elif trans[mid] > target:
            if max == mid:
                return mid
            else:
                max = mid
                return search(min, max, target)
        elif trans[mid] < target:
            min = mid
            return search(min, max, target)


_transactions = int(raw_input())
_values = map(int, raw_input().split())
trans = []
total = 0
for value in _values:
    total += value
    trans.append(total)
_query = int(raw_input())
for query in xrange(_query):
    target = int(raw_input())
    min = 0
    max = _transactions-1
    trans_num = search(min, max, target)
