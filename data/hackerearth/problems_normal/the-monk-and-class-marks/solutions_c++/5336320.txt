#include<bits/stdc++.h>
using namespace std;
struct st_marks
{
	
	string name;
	int marks;
};
bool compare(st_marks s1,st_marks s2)
{
	if(s1.marks==s2.marks)
		return (s1.name<s2.name);
	return (s1.marks>s2.marks);
}
int main()
{
	int n;
	cin>>n;
	st_marks a[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i].name>>a[i].marks;
	}
	sort(a,a+n,compare);
	for(int i=0;i<n;i++)
	{
		cout<<a[i].name<<" "<<a[i].marks<<endl;
	}
	return 0;
}
