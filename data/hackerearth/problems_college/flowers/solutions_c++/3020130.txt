#include<bits/stdc++.h>
#include<iostream>
using namespace std;
int main()
{
    int n,k;
    cin>>n>>k;
    int arr[n],i,sum=0;
    for(i=0;i<n;i++)
        cin>>arr[i];
    std::sort(arr, arr+n, std::greater<int>());
    for(i=0;i<n;i++)
    {
        int z=(i+k)/k;
        sum=sum+z*arr[i];
    }
    cout<<sum<<endl;
}
