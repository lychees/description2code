#include<iostream>
using namespace std;
int pr[1000001]={0};
int main()
{
    int i,j;
	for(i = 2 ; i <= 1000000 ; ++i ){
		  if(pr[i]==0)
		for(j = 2*i ; j <= 1000000 ; j=j+i){
			  {
  				if(pr[j] == 0)
  				 pr[j]=i;
  			}
		}
	}
	
		for(i = 1 ; i <= 1000000 ; ++i )
		  pr[i]+=pr[i-1];
		  int t;
		  cin >> t;
		  while(t--)
		  {
  			int l , r;
  			cin >> l >> r;
  			cout <<pr[r]-pr[l-1] << endl;
   		}	
   		return 0;
}
