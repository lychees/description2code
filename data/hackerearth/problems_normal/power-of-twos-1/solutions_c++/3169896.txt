#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1e6+3;
long long dp[MAXN];
int main() {
	int t;
	cin>>t;
	for(int i=1;i<MAXN;i++)
		dp[i]=1;
	dp[1]=0;
	for(int i=2;i<MAXN;i++) {
		for(int j=2;j*i<MAXN;j++)
			dp[j*i]++;
	}
	for(int i=2;i<MAXN;i++)
		dp[i]+=dp[i-1];
	while(t--) {
		int n;
		scanf("%d",&n);
		printf("%lld\n",dp[n]);
	}
	return 0;
}
