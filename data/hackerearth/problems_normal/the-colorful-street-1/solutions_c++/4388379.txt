ll t;

struct node
{
    ll a;
    ll b;
    ll c;
};


ll dp[21][4];
node arr[21];

ll solve(ll currind,ll colour)
{
    if(currind==-1){
        
        return 0;
    }
    
    if(dp[currind][colour]!=-1){
        return dp[currind][colour];
    }
    ll ans  =  INT_MAX;
    
    if(colour==1){
        ans  = min(ans,arr[currind].b+solve(currind-1, 2));
        ans  = min(ans,arr[currind].c+solve(currind-1, 3));
        return dp[currind][colour] =  ans;
    }
    else if (colour==2)
    {
        
        
        ans  = min(ans,arr[currind].c+solve(currind-1, 3));
        ans  = min(ans,arr[currind].a+solve(currind-1, 1));
        return dp[currind][colour] =  ans;
    }
    else if(colour==3){
        
        ans  = min(ans,arr[currind].a+solve(currind-1, 1));
        ans  = min(ans,arr[currind].b+solve(currind-1, 2));
        
        return dp[currind][colour] =  ans;
    }
    else{
        
        
        ans  = min(ans,arr[currind].a + solve(currind-1, 1));
        ans = min(ans,arr[currind].b+solve(currind-1, 2));
        ans = min(ans,arr[currind].c+solve(currind-1, 3));
        return dp[currind][colour] =  ans;
        
    }
    
}


int main()
{
    ios::sync_with_stdio(false);

        loop(i, n){
            cin>>arr[i].a>>arr[i].b>>arr[i].c;

    }

}
