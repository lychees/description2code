#include <iostream>
using namespace std;
int main()
{
   int n,pos,max;
    cin>>n;
    int t[1000000],p[1000000];
    for(int i=0;i<n;i++)
    {
      cin>>t[i]>>p[i];     
    }
    for(int i=0;i<n;i++)
    {
        if(i==0)
        { 
            max=t[i];
            pos=i;
        }    
        if(t[i]>max)
        {   
            max=t[i];
            pos=i;
        }    
    }
    if(p[pos]>0)
    cout<<"Left Corner";
    else if(p[pos]<0)
    cout<<"Right Corner";
    else
    cout<<"Either Corner";
    return 0;

}
