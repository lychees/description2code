#include<iostream>
#include<stdio.h>
#include<math.h>
#include<string.h>
using namespace std;
int main() 
{
	long long i,j,t,p,l,k;
	cin>>t;
	getchar();
	while(t--)
	{
		char s[1000];
		cin.getline(s,1000);
		l=strlen(s);
		p=int(sqrt(l));
		char a[p+1][p+1];
		k=0;
		for(i=0;i<p;i++)
		{
			for(j=0;j<p;j++)
			{
				a[i][j]=s[k];
				k++;
			}
		}
		for(i=0;i<p;i++)
		for(j=0;j<p;j++)
			cout<<a[j][i];
		cout<<"\n";
		
	}
	return 0;
}
