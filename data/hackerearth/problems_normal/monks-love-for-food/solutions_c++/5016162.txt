#include <iostream>
#include <stack>
using namespace std;

int main()
{
    int nQueries;
    cin >> nQueries;
    
    stack<int> foodStack;
    for (int i = 0; i < nQueries; i++) {
    	int queryType;
    	cin >> queryType;
    	if (queryType == 1) {
    		if (foodStack.empty()) cout << "No Food\n";
    		else {
    			auto price = foodStack.top();
    			foodStack.pop();
    			cout << price << endl;
    		}
    	}
    	else {
    		int price;
    		cin >> price;
    		foodStack.push(price);
    	}
    }
    
    return 0;
}
