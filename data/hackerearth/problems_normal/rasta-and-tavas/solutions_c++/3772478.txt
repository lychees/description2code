#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int n,k;
    cin>>n>>k;
    int count=0;
    for(int z=1;z<=n;z++)
    {
    	int i=z;
    	int temp=0;
    	int flag=0;
    	while(i%2==0)
    	{
    		if(flag==0)
    		{
    		temp+=2;
    		flag=1;
    		}
    		i=i/2;
    	}
    //	cout<<i<<":"<<endl;
    //	cout<<"temp at 2:"<<temp<<endl;
    	for(int j=3;j<=sqrt(i);j+=2)
    	{
    		flag=0;
    		while(i%j==0)
    		{
    			if(flag==0)
    			{
    				flag=1;
    				temp+=j;
    			}
    			i=i/j;
    		}
    	//	cout<<"temp at "<<j<<" : "<<temp<<endl;
    	}
    	if(i>2)
    		temp+=i;
    	if(temp==k)
    	count++;
    	
    }
    
    cout<<count;
    
    return 0;
}
