#include<iostream>
#include<cstring>

using namespace std;

int main()

{
    int test;
    cin>>test;
    while(test--)
    {
        char s[100];
        cin>>s;
        int len = strlen(s);
        int temp = 0;
        for(int i =0;i<len;i++)
        {
            int curr = temp;
            int alpha = s[i]-97;
            int dista =0 , distc = 0;
            while(1)
            {
                if(temp == alpha)
                    break;
                else
                {
                    distc++;
                    temp = (temp+1)%26;
                }
            }
            temp = curr;
            while (1)
            {
                if(temp == alpha)
                    break;
                else
                {
                    dista++;
                    temp = (temp-1+26)%26;
                }
            }

            if(dista >= distc)
                cout<<distc<<" ";
            else if(distc > dista)
                cout<< -1 *dista<<" ";

            temp = alpha;
        }

        cout<<endl;
    }
}
