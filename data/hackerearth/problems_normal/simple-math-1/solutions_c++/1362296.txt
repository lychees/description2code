
#include <iostream>
#define m 10000007
using namespace std;
int mat[100][100]={0};
 
int binomial_coeff(int n,int r)
{
	
if(r==0 || n==r) return 1;
else if(n==r+1 || r==1 ) return n;
else if(mat[n][r]) return mat[n][r];
else return mat[n][r]=(binomial_coeff(n-1,r-1)%m+binomial_coeff(n-1,r)%m)%m;
	
}
 
int main()
{
   int n,r;
   cin>>n>>r;
   if(n>=r)
   cout<<binomial_coeff(n-1,r-1);
    else 
    cout<<-1;
    return 0;
}
