

    #include<iostream>
    #include<string>
    using namespace std;
     
    string word[60]={"one", "two", "three", "four", "five", "six", "seven", "eight", "nine", "ten", "eleven", "twelve", "thirteen", "fourteen", "fifteen", "sixteen", "seventeen", "eighteen", "nineteen", "twenty", "twenty one", "twenty two", "twenty three", "twenty four", "twenty five", "twenty six", "twenty seven", "twenty eight", "twenty nine", "thirty"};
        
    int main()
    {
        int H,M;
        
        cin>>H>>M;
        if(M==0)
            cout<<word[H-1]<<" o' clock"<<endl;
     
        else if(M==15)
            cout<<"quarter past "<<word[H-1]<<endl;
     
        else if(M>0 && M<30)
            cout<<word[M-1]<<" minutes past "<<word[H-1]<<endl;
     
        else if(M==30)
            cout<<"half past "<<word[H-1]<<endl;
     
        else if(M==45)
            cout<<"quarter to "<<word[H%12]<<endl;
     
        else
            cout<<word[59-M]<<" minutes to "<<word[H%12]<<endl;
     
        return 0;
    }
