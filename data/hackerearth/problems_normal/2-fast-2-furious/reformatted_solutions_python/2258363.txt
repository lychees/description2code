def check(li=[]):
    diff = 0
    ll = len(li)
    for m in range(ll - 1):
        r = abs(li[m] - li[m + 1])
        if r > diff:
            diff = r
    return diff


_ = input()

gcc = raw_input().strip().split(' ')
gcc1 = [int(m) for m in gcc]
gcc2 = check(gcc1)
gcc = raw_input().strip().split(' ')
gcc1 = [int(m) for m in gcc]
gcc3 = check(gcc1)
