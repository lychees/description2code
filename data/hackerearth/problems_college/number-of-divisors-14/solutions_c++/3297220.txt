#include <bits/stdc++.h>
using namespace std;

bool prime[1000000];
 
void SieveOfEratosthenes()
{
    memset(prime, true, sizeof(prime));
 	int n=1000000;
    for (int p=2; p*p<=n; p++)
    {
        if (prime[p] == true)
        {
            for (int i=p*2; i<=n; i += p)
                prime[i] = false;
        }
    }
}

int main()
{
	SieveOfEratosthenes();
	int test;
	scanf("%d",&test);
	while(test--)
	{
		long long int N,i,Divisors=1;
		scanf("%lld",&N);
		for(int i=2;N!=1;i++)
		{
			int g=0;
			if(prime[i])
			{
				while(N%i==0)
				{
					N/=i;
					g++;
				}
				Divisors*=(g+1);
			}
		}
		printf("%lld\n",Divisors);
	}
	return 0;
}
