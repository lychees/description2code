#include <iostream>
using namespace std;

int main()
{
         int t,h,m,ha,ma,angle;
          cin>>t;
          while(t--)
          { 
          	cin>>h>>m;
          	  if(h==12)
          	   h=0;
          	   if(m==60)
          	   m=0;
          	   ha=0.5*(h*60+m);
          	   ma=6*m;
          	   angle=abs(ha-ma);
          	   angle=min(360-angle,angle);
          	   cout<<angle<<endl;
          }
          	   
          	
    return 0;
}
