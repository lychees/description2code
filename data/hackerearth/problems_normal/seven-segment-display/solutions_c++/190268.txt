#include<iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		int len=s.size();
		int arr[]={6,2,5,5,4,5,6,3,7,6};
		int count=0;
		for(int i=0;i<len;++i)
		{
			int a=s[i]-'0';
			count+=arr[a];
			//cout<<a<<" takes "<<arr[a]<<" sticks\n";
		}
		cout<<count<<endl;
	}
}
