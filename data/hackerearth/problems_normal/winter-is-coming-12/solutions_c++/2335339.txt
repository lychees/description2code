#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int arr[100005];
int cf[100005];
int main()
{
    int test;
    unordered_map <int, int> mp; 
    cin >> test;
    while(test--){
    	int n;
    	cin >> n;
    	mp.clear();
    	int sum = 0;
    	for(int i = 0; i < n; i++){
    		cin >> arr[i];
    		sum = sum + arr[i];
    		cf[i] = sum%n;
    	}
    	int min = INT_MAX;
    	int l,r;
    	for(int i = 0; i < n; i++){
    		if(cf[i] == 0){
    			min = i;
    			l = 1;
    			r = i+1;
    			break;
    		}
    	}
    		//cout << l << " " << r << endl;
    	for(int i = 0; i < n; i++){
    		if(mp.find(cf[i]) == mp.end()){
    			mp[cf[i]] = i;
    		}else{
    			int temp = i - mp[cf[i]];
    			if(temp < min){
    				min = temp;
    				l = mp[cf[i]] +2;
    				r = i +1;
    			}
    			mp[cf[i]] = i;
    		}
    	}
    	cout << l << " " << r << endl;
    }
    return 0;
}
