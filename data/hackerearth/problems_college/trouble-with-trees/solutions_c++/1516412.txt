#include <iostream>
using namespace std;

unsigned long int catalan[1002];

#define MAX 1000
#define mod 1000000007

void init()
{
	catalan[0]=catalan[1]=1;
	
	int i,j;
	for(i=2;i<=MAX;i++)
	{
		catalan[i]=0;
		for(j=1;j<=i;j++)
		 {
		 	catalan[i]+= catalan[j-1]*catalan[i-j];
		 	catalan[i]%=mod;
		 }
	}
}

int main() {
	// your code goes here
	init();
	int test,n;
	cin>>test;
	while(test--){
		cin>>n;
		cout<<catalan[n]<<endl;
	}
	return 0;
}
