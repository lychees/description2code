#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   	int t,n,count,i;
   	int *a;
   	cin>>t;
   	while(t--)
   	{
   		count=0;	
   		cin>>n;
   		a=new int[n];
   		for(i=0;i<n;i++)
   			cin>>a[i];
   		sort(a,a+n);
   		reverse(a,a+n);
   		for(i=0;i<n-1;i++)
   			count+=a[i]-a[i+1];
   		cout<<count*2<<endl;	
   		
   	}
    return 0;
}
