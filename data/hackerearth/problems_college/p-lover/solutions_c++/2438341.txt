#include<bits/stdc++.h>
using namespace std;
int isprime(long int n)
{
	if(n==1)
		return 0;
	 if(n==2)
		return 1;
		if(n%2==0)
			return 0;
		for(long int i=3;i<=sqrt(n);i=i+2)
		{
			if(n%i==0)
			{
				return 0;
			}
		}
	return 1;
}
long int revers(long int n)
{
	long int mod,rev=0;
	while(n>0)
	{
		mod=n%10;
		rev=rev*10+mod;
		n=n/10;
	}
	return rev;
}
int ispalindrom(long int n)
{
	long int rev=revers(n);
	if(rev==n)
	{
		return 1;
	}
	else
	{
		return 0;
	}
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int x;
    	scanf("%d",&x);
    	for(long int i=x;;i++)
    	{
    		int flag1=isprime(i);
    		int flag2=ispalindrom(i);
    		if(flag1==1 && flag2==1)
    		{
    			printf("%d\n",i);
    			break;
    		}
    	}
    }
}
