#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    for(int lolo=0;lolo<t;lolo++){
    	string s1,s2;
    	cin >> s1 >> s2;
    	sort(s2.begin(),s2.end());
    	int i=0,j=0;
    	while(i<s1.size() && j<s2.size()){
    		if(s2[j]<s1[i]){
    			s1[i]=s2[j];
    			i++;
    			j++;
    		}else i++;
    	}
    	cout << s1 <<endl;
    }
    return 0;
}
