#include <iostream>
#include <vector>
using namespace std;

bool isPrime(int n)
{
	for(int i=2;i*i<=n;i++)
	 if(n%i==0)
	  return false;
	return true;
}

int main()
{
	vector<int>p;
	for(int i=2;i<=10000;i++)
	 if(isPrime(i))
	  p.push_back(i);
	int ans[10001][5];
	for(int i=10000;i>-1;i--)
	 for(int j=0;j<5;j++)ans[i][j]=100001;
	for(int j=0;j<p.size();j++)
	 ans[p[j]][0]=ans[p[j]][1]=ans[p[j]][2]=ans[p[j]][3]=ans[p[j]][4]=0;
	for(int i=10000;i>=0;i--)
	 for(int k=0;k<5 && i+k+1<10001;k++)
	  {
	  	ans[i][k] = min(ans[i][k],1+ans[i+k+1][k]);
	  }
    int n,k;
    cin>>n>>k;
    int a[n][n];
    for(int i=0;i<n;i++)
     for(int j=0;j<n;j++)
      cin>>a[i][j];
    int minv = -1,x0=0,y0=0;
    for(int i=1;i<n-1;i++)
     for(int j=1;j<n-1;j++) 
      {
      	int x1 = ans[a[i-1][j-1]][k-1];
      	int x2 = ans[a[i][j]][k-1];
      	int x3 = ans[a[i-1][j+1]][k-1];
      	int x4 = ans[a[i+1][j-1]][k-1];
      	int x5 = ans[a[i+1][j+1]][k-1];
      	if(x1>-1 && x2>-1 && x3>-1 && x4>-1 && x5>-1)
      	 {
      	 	if(minv==-1|| minv>x1+x2+x3+x4+x5)
      	 	 {minv = x1+x2+x3+x4+x5;
      	 	  x0=i;y0=j;
      	 	 }
      	 }
      }
    if(minv == -1)
     cout<<"no\n";
    else
    cout<<"yes\n"<<minv<<'\n'<<x0+1<<' '<<y0+1;
    return 0;
}
