#include <bits/stdc++.h>
using namespace std;
#define For(i,a,b) for(int i=a;i<b;i++)
#define mp make_pair
#define pb push_back
#define mod 1000000007

long long ans,dp[1111][1111],t,n,q,a,b;
pair<int,int> lis[111];

int main( ){
    //freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
    //ios::sync_with_stdio(0);
    cin>>t;
    For(i,0,1111) dp[i][0]=dp[0][i]=1;
    For(i,1,1111) For(j,1,1111) dp[i][j]=(dp[i][j-1]+dp[i-1][j])%mod;
    while(t--)
    {
        cin>>n>>q;
        For(i,0,q)
        {
            cin>>a>>b;
            lis[i]=mp(a,b);
        }
        lis[q]=mp(0,0);
        lis[q+1]=mp(n-1,n-1);
        sort(lis,lis+q+2);
        ans=1;
        For(i,0,1+q)
        {
            a=lis[i+1].first-lis[i].first;
            b=lis[i+1].second-lis[i].second;
            ans*=dp[a][b];
            ans%=mod;
        }
        cout<<ans<<endl;
    }
    return 0;
}
