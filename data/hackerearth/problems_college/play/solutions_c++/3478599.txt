#include<stdio.h>
#include<algorithm>
#include<bits/stdc++.h>

using namespace std;

int main() {

int s[550][550],t,n,k,a[550],i,j,l;
	scanf("%d",&t);
	while(t--) {		
		scanf("%d%d",&n,&k);
		for(i=0;i<n;i++) {
			scanf("%d",&a[i]);
		}
		sort(a,a+n);
		for(i=0;i<=n;i++) {
			for(j=0;j<=n;j++) {
				s[j][i]=INT_MAX;
			}
		}
		for(i=0;i<n;i++) {
			s[i][0]=0;
			s[i][1]=(n-i)*a[i];
		}
		 int min = INT_MAX;

		if(k==1) {
			for(i=0;i<n;i++) {
				int temp = a[i]*(n-i);
				if(temp<min) min = temp;				
			}
		}
		else {
		for(i=2;i<=k;i++) { 
			for(j=0;j<n;j++) { 
			if(j>=i-1) {
				for(l=i-2;l<j;l++) { 
					 int temp = s[l][i-1]+(n-j)*(a[j]-a[l]);

					if(temp<s[j][i])   s[j][i] =temp;
				}
			}				
			}
		}		
		for(i=k-1;i<n;i++) {
			if(min>s[i][k]) min = s[i][k];			
		}
	}
		printf("%d\n",min);
	}
}
