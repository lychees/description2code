#include <iostream>
using namespace std;

int search(int array[], int lower, int upper, int element)
{
	int mid = (lower + upper)/2;
	int id;
	
    if( array[mid] < element && mid >= lower && mid <= upper)
	{
		id = search(array, mid+1, upper, element);
		return id;
	}
	if(array[mid] > element && mid >=lower && mid <= upper)
	{
		id = search(array, lower, mid-1, element);
		return id;
	}
	if(lower == upper)
		return lower;
		
	if(array[mid] < element)
		return mid+1;
	return mid;

}

int main()
{
    int no_boxes,a,no_cases,k,cnt=0;
    scanf("%d",&no_boxes);
    
    int arr[no_boxes+1];
    arr[0] = 0;
    
    for(int i = 1 ; i <= no_boxes ; i++)
    {
    	scanf("%d",&a);
        arr[i] = arr[i-1]+a;
    }
    
    scanf("%d",&no_cases);
    
    for(int j = 0; j < no_cases ; j++)
    {
    	scanf("%d",&a);
    	cnt = search(arr,1,no_boxes+1,a);
    	printf("%d\n",cnt);
    }
    
    return 0;
}
