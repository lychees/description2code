#include <iostream>
using namespace std;
int turn=0;
void function(int a, int b, int c){
	turn+=1;
	turn%=2;
	if(a==0 && b==0 && c==0)
		return;
	else{
		if(a%2==0)
			function(a-1, b, c);
		else if(b%2==0)
			function(a, b-1, c);
		else if(c%2==0)
			function(a, b, c-1);
	}
	return ;
}

void solve(){
	int a, b, c;
	cin>>a>>b>>c;
	//function(a, b, c);
	if(a%2 ==0 && b%2==0 && c%2==0)
		cout<<"Aishwarya\n";
	else cout<<"Ashima\n";
	return ;
}

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    	solve();
    return 0;
}
