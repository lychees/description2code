#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long int t, n;
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	
    	long long int temp = sqrt(n);
    	temp = temp*temp;
    	if(temp!=n)
    	{
	  		long long int a = sqrt(n);
	  		long long int b = (a+1)*(a+1);
	  		a = a*a;
	  		
	  		
	  		if(n>=(b-sqrt(b)+1))
	  		{
	  			cout<<2*sqrt(b) - 1<<endl;
	  		}
	  		else
	  		{
	  			cout<<2*sqrt(a)<<endl;
	  		}
    	}
    	else
    		cout<<2*sqrt(n) - 1<<endl;
    }
    return 0;
}
