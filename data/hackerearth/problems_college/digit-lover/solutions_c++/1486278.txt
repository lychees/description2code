using namespace std;
ll arr[100001];

int getCount(ll n){
	ll count =0;
	if(n==0){
		count=1;
	}
	else{

	while(n>0){
		n= n/10;
		count++;
	}

	}
	return count;
}
int getSum(ll n){
	ll sum =0; 
	while(n>0){
		ll rem = n%10;
		sum = sum + rem;
		n = n/10;
	}
	return sum;
}
int main(){
	ll t,n;
	cin>>t;
	while(t-->0){
		cin>>n;
		while(getCount(n)!=1){
			 n = getSum(n);
		}
		cout<<n<<"\n";
	} 
	return 0;
}
