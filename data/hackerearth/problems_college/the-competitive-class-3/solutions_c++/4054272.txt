#include<iostream>
using namespace std;
int cnt[1009];
int cumsum[1009];
int a[1009];
int main()
{
    int n;
    cin>>n;
    for(int i=1;i<=n;++i)
    {
        cin>>a[i];
        cnt[a[i]]++;
    }
    for(int i=1;i<=1000;++i)
    {
        cumsum[i]=cnt[i]+cumsum[i-1];
    }
    for(int i=1;i<=n;++i)
    {
        int rnk=1+cumsum[1000]-cumsum[a[i]];
        cout<<rnk<<" ";
    }
    cout<<endl;
return 0;
}
