#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int main()
{
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		vector<int> g(n,0);
		vector<int> b(n,0);
		for(int i=0;i<n;i++){
			cin>>g[i];
		}
		for(int i=0;i<n;i++){
			cin>>b[i];
		}
		sort(g.begin(),g.end());
		sort(b.begin(),b.end(),greater<int>());
		int c=0;
		for(int i=0;i<n;i++){
			if(g[i]%b[i]==0 || b[i]%g[i]==0){
				c++;
			}
		}
		cout<<c<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
