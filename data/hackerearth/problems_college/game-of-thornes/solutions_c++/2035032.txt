#include <bits/stdc++.h>
using namespace std;

int main()
{
    string str;
    cin>>str;
    int a[26]={0};
    int i;
    for(i=0;i<str.size();i++)
    {
    	a[str[i]-97]++;
    }
    int count=0;
    for(i=0;i<26;i++)
    {
    	if(a[i]%2==1)
    	{
    		count++;
    	}
    }
    if(count>=2)
    cout<<"NO"<<endl;
    else
    cout<<"YES"<<endl;
}
