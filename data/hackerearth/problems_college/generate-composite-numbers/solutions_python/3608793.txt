def get_primes(n):
    numbers = set(range(n, 1, -1))
    primes = [1]
    while numbers:
        p = numbers.pop()
        primes.append(p)
        numbers.difference_update(set(range(p*2, n+1, p)))
    return primes
a=get_primes(10000)
n=input()
b=[]
for i in range(1,n+1):
    if i not in a:
        b.append(i)
print ' '.join(map(str,b))
