#include<iostream>
#include<stdlib.h>
#include<string.h>
#include<math.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int cnt=0,cnt1=0;
		char s[1000];
		cin>>s;
		for(int i=0;i<strlen(s);i++)
		{
			if(s[i]=='A'||s[i]=='D'||s[i]=='O'||s[i]=='P'||s[i]=='R'||s[i]=='Q')
			cnt++;
			else if(s[i]=='B')
			cnt+=2;
			char m=(char)(s[i]+32);
  	   		 if(m=='a' || m=='d' || m=='o' || m=='p' || m=='q' || m=='b' || m=='g' || m=='e')
    			{cnt1=cnt1+1;}
		}
		if(cnt-cnt1==-5)
		cout<<"-3"<<endl;
		else
		cout<<(cnt-cnt1)<<endl;
	}
 
 
}
