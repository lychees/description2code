#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	char s[101];
    	int hash[26]={0};
    	while(n--)
    	{
    		cin>>s;
    		bool present[26]={false};
    		int l=strlen(s);
    		for(int i=0;i<l;i++)
    		{
    			if(present[s[i]-'a']==false)
    			{
    				hash[s[i]-'a']++;
    				present[s[i]-'a']=true;
    			}	
    		}
    	}
    	int max=hash[0];
    	int max_index=0;
    	for(int i=1;i<26;i++)
    	{
    	   if(hash[i]>max)
    	   {
    	   	  max=hash[i];
    	   	  max_index=i;
    	   }
    	}
    	cout<<char(max_index+'a')<<endl;
    }
    return 0;
}
