#include<iostream>
#include<algorithm>
using namespace std;
struct time
{
    int start,end;
}krr[101];
bool compare(struct time a,struct time b)
{
      if(a.start!= b.end)
       return (a.start<b.start);
      else
       return (a.end<b.end);
}
int main()
{
    int n;
    cin>>n;
    char str[15],arr[5],brr[5];
    int i,j,start,end;
    for(i=0;i<n;i++)
    {
        cin>>str;
        int k=0;
        for(j=0;j<=4;j++)
        {
            if(j!=2)
            {
                arr[k]=str[j];
                k++;
            }
        }
        k=0;
        for(j=6;str[j]!='\0';j++)
        {
            if(j!=8)
            {
                brr[k]=str[j];
                k++;
            }
        }
        krr[i].start=atoi(arr);
        krr[i].end=atoi(brr);
       // cout<<krr[i].start<<"  "<<krr[i].end<<endl;
    }
    sort(krr,krr+n,compare);
    //for(i=0;i<n;i++)
      //  cout<<krr[i].start<<"     "<<krr[i].end<<endl;
    for(i=0;i<n-1;i++)
    {
        if(krr[i].start==krr[i+1].start)
        {
            cout<<"Will need a moderator!"<<endl;
            break;
        }
        if(krr[i].end>krr[i+1].start)
        {
            cout<<"Will need a moderator!"<<endl;
            break;
        }
    }
    if(i==n-1)
        cout<<"Who needs a moderator?"<<endl;
    return 0;
}
