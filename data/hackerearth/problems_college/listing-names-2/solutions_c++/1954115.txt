#include<iostream>
#include<map>
using namespace std;
string s;
int t;
map<string,int>name;
int main()
{
    cin>>t;
    while(t--)
    {
        cin>>s;
        name[s]++;
    }
    map<string,int>::iterator it;
    for(it=name.begin();it!=name.end();it++)
    {
        cout<<it->first<<" "<<it->second<<endl;
    }
}
