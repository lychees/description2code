#include<bits/stdc++.h>
using namespace std;
#define tcase(t) cin>>t; while(t--)
int main()
{
    int n,k,v,r,ans;
    tcase(n)
    {
        ans=0;
        cin>>k>>v>>r;
        int a = max(k,v);
        int b= max(min(k,v),r);
        if(a==0&&b==0)ans=0;
        else if(a==0||b==0)ans=1;
        else if(a==b)ans=a+b;
        else ans= min(a,b)*2+1;
        cout<<ans<<endl;
    }
    return 0;
}
