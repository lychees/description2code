def func(N):
	j = set([1])
	cur = 0
 
	for i in range(1, N+1):
		if i in j:
			k.append(cur)
			continue
 
		j.update( range(i, N+1, i) )
		cur += i
		k.append(cur)
 
k = []
func(10**6)
 
T = input()
for i in range(T):
	l, r = map(int, raw_input().strip().split())
	l -= 1
	r -= 1
 
	if l == 0:
		print k[r]
	else:
		print k[r] - k[l - 1]
