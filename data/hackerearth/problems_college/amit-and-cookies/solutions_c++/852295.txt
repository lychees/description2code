#include <iostream>
#include <vector>
#include <algorithm>

#define ll long long

using namespace std;

int main()
{
	ll n = 0, k = 0, tc = 0, sum = 0, ctr = 0;
	ll *cp;
	vector<ll> cookies;
	
	while (cin >> n){
		cin >> k;
		while (n--){
			cin >> tc;
			cookies.push_back(tc);
		}
		sort(cookies.begin(),cookies.end());
		cp = &cookies[0];
		
		for (ll i = 0; i < cookies.size(); i++){
		
			//cout << cp[i] << " " << sum << endl;
			sum+=cp[i];
			if (sum>=k)
				break;
			else
				ctr++;
		}
		
		cout << ctr << endl;
		
		cookies.clear();
		sum = 0; ctr = 0;
	}
	
	
	
    return 0;
}
