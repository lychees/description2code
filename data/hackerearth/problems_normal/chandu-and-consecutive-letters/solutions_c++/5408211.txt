#include <iostream>
#include<string.h>
using namespace std;
void delrep(string s){
int i,l=s.size();
cout<<s[0];
for(i=0;i<l-1;i++){
	if(s[i]==s[i+1])
	continue;
	cout<<s[i+1];
}
return;
}
int main()
{
   int n,i;
   string str[100];
   cin>>n;
   for(i=0;i<n;i++){
   	cin>>str[i];
   	delrep(str[i]);
   	cout<<endl;
   }
}
