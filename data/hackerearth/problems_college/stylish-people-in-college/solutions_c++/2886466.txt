#include <iostream>
using namespace std;

int main()
{
	int t,n;
	cin>>t;
	for(int j=1;j<=t;j++){
		cin>>n;
		
		int a[n];
		
		for(int i=0;i<n;i++){
			cin>>a[i];
			
		}
		
		int including=a[0];
		int excluding=0;
		
		int newex;
		
		for(int i=1;i<n;i++){
			newex = (including >excluding)?including:excluding;
			
			including=excluding+a[i];
			excluding=newex;
			
		}
		
		int ans= (including>excluding)?including:excluding;
		
		cout<<"Case "<<j<<": "<<ans<<"\n";
		
	}
    return 0;
}
