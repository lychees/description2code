#include <iostream>
#include<bits/stdc++.h>
using namespace std;


typedef long long int lli;


int main()
{
	int t;
	 cin>>t;
	 while(t--)
	 {
	 	
	 
	lli n;
//	  cin>>n;
	  lli a,b,c;
	  cin>>a>>b>>n;
	  lli sum=a+b;
	  
	  for(int i=3;i<=n;i++)
	   {
	   	  c=abs(a-b);
	   	    	  // cout<<a<<" "<<b<<" "<<c;//
	   	 
	   	 
	  
	 
	   	  if(b==c)
	   	   {
	   	   
	   	   	 lli rem=n-i+1;
	   	   	 	// cout<<"rem"<<rem<<endl;
	   	   	 sum+=2*b*(rem/3);
	   	   	 if(rem%3==1 || rem%3==2) sum+=b;
	   	   	 
	   	   	 break;
			  }
			   a=b;
	   	  b=c;
	   	  //cout<<sum<<endl;
	   	   	  sum+=c;
	   	   	  //cout<< "sum "<<sum<<endl;
	   	   	  
	   }
	   cout<<sum<<endl;
	 }
 
 
}
