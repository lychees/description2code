#include<bits/stdc++.h>
using namespace std;
int dp[100001],i;
int main(){
dp[0]=1000000;
dp[1]=1000000;
dp[2]=1;
dp[3]=1;
dp[4]=2;
dp[5]=1;
dp[6]=2;
dp[7]=1;
for(int i=8;i<=100000;i++){
dp[i] = min(min(dp[i-2]+1,dp[i-3]+1),min(dp[i-5]+1,dp[i-7]+1));
    }
int t,n;
scanf("%d",&t);
while(t--){
    scanf("%d",&n);
    if(n==0|n==1){
        printf("-1\n");
        continue;
    }
    printf("%d\n",dp[n]);
}
return 0;
}
