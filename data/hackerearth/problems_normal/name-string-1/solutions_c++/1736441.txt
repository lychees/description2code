#include <iostream>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--)
	{

	int a,b,c,maxsec,max,result;
	cin>>a>>b>>c;
	if(a>b)
	{
		if(b>c)
		{
			maxsec=b;
			max=a;
		}
		else if(a>c)
			{maxsec=c;
			max=a;
			}
		else
			{
			maxsec=a;
			max=c;
			}
	}
	else
	{
		if(a>c)
		{
			maxsec=a;
			max=b;
		}
		else if(b>c)
			{
			max=b;
			maxsec=c;
			}
		else
			{
			maxsec=b;
			max=c;
			}
	}

		result=(maxsec*2+1);
		if(max==maxsec)
				{
					result--;
				}



		cout<<result<<endl;
	}


	return 0;
}
