#include <iostream>

using namespace std;

int main()
{
        int test_cases;
        cin>>test_cases;
        while(test_cases > 0)
        {
                int X1,X2,Y1,Y2,ans=0;
                cin>>X1>>Y1>>X2>>Y2;
                
                if(X2==Y1) ans++;
                if(X1==Y2) ans++;
                cout<<ans<<endl;
                test_cases --;
        }
        
        return 0;
}
