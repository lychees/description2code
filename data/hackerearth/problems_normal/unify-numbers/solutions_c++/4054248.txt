#include<iostream>
#include <stdlib.h>
using namespace std;
long long int A[100001];
int compare (const void * a, const void * b)
{
  return ( *(long long int*)a - *(long long int*)b );
}
int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        int i;
        for(i=0;i<n;++i)
        {
            cin>>A[i];
        }
        long long int m1=0,m2=0;
        long long int ans1=0,ans2=0;
        qsort(A, n, sizeof(long long int), compare);
        if(n%2)
        {
            m1=A[n/2];
            for(i=0;i<n;++i)
            {
                ans1+=abs(A[i]-m1);
            }
            cout<<ans1<<'\n';
        }
        else
        {
            m1=A[n/2-1];
            m2=A[n/2];
            for(i=0;i<n;++i)
            {
                ans1+=abs(A[i]-m1);
                ans2+=abs(A[i]-m2);
            }
            if(ans1>ans2)
                cout<<ans2<<'\n';
            else
                cout<<ans1<<'\n';
    }

        }
        
}
