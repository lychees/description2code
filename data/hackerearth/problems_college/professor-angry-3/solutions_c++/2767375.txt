#include <iostream>
using namespace std;

int main()
{
    int t,r[10],n[10],k[10],a[10][1000];
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	int s=0;
    	cin>>n[i]>>k[i];
    	for(int j=0;j<n[i];j++)
    	{
    		cin>>a[i][j];
    		if(a[i][j]<=0)
    		s++;
    	}
    	if(s<k[i])
    	r[i]=1;
    	else r[i]=0;
    	
    }
    
    for(int i=0;i<t;i++)
    {
    	if(r[i]==1)
    	cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
    return 0;
}
