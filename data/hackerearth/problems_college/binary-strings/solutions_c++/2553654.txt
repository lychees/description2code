#include <iostream>
using namespace std;
int MOD=1000000007;
int fib[100000];
void fibo()
{
    fib[0]=1;
    fib[1]=1;
    fib[2]=1;
    for(int i=3;i<100000;i++)
    {
        fib[i]=(fib[i-1]%MOD+fib[i-2]%MOD)%MOD;
    }
}
int main()
{
    fibo();
    int t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        cout<<fib[n+1]<<endl;
    }
    return 0;
}
