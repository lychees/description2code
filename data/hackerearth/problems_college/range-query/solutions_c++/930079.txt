#include<cstdio>
int main()
{
 int n,q,a[10010],val,memo[105][10010],i,j,x,y,count;
 scanf("%d",&n);
 for(i=0;i<n;++i)
  scanf("%d",&a[i]);
  
 for(i=0;i<=100;++i)
 {
  count=0;
  for(j=0;j<n;++j)
  {
   if(a[j]==i)
    ++count;
   memo[i][j]=count;
  }
 }
 scanf("%d",&q);
 for(i=0;i<q;++i)
 {
  scanf("%d%d%d",&x,&y,&val);
  if(x)
   printf("%d\n",memo[val][y]-memo[val][x-1]);
  else
   printf("%d\n",memo[val][y]);
 }
}
