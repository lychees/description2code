#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    long long int t,j,n,m;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&n);
    	m=1;
    	if(n==0)
    	{printf("NO");
    	printf("\n");}
    	if(n>0)
    	{
    		for(j=2;j<=(sqrt(n));j++)
    	{
    		if(n%j==0)
    		{m=0;break;}
    		else
    		{m=1;}
    	}
    if(m==0)
    {printf("NO");
    printf("\n");}
    else
    {printf("YES");
    printf("\n");}
    }}
    return 0;
}
