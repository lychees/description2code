#include<bits/stdc++.h>
using namespace std;
bool is_prime(int n){
	if(n<=1){
		return false;
	}
	if(n==2){
		return true;
	}
	if(n%2==0){
		return false;
	}
	int div=3;
	int limit=sqrt(n)+1;
	while(div<=limit){
		if(n%div==0){
			return false;
		}
		div+=2;
	}
	return true;
}
bool is_palindrome(int n){
	vector<int> v;
	while(n!=0){
		v.push_back(n%10);
		n/=10;
	}
	int r=v.size()-1;
	int l=0;
	while(l<r){
		if(v[l]!=v[r])
			return false;
		l++;r--;
	}
	return true;
}
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		for(int i=n;;i++){
			if(is_prime(i)&&is_palindrome(i)){
				cout<<i<<endl;
				break;
			}
		}
	}
}
