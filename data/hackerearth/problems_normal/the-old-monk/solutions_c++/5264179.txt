#include<iostream>
using namespace std;
int main()
{
	long long t,i,j,n,a[100005],b[100005];
	cin>>t;
	int l,h,m,p;
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		for(i=0;i<n;i++)
		{
			cin>>b[i];
		}
		int ans=0;
		for(i=0;i<n;i++)
        {
            l=0;h=n-1;
            p=-1;
            while(l<=h)
            {
                m=(l+h)/2;
                if(b[m]>=a[i])
                {
                    p=m;
                    l=m+1;
                }
                else
                {
                    h=m-1;
                }
            }
            if(ans<p-i)
                ans=p-i;
        }

		cout<<ans<<endl;

	}
}
