#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,h,a,min,diff;
    cin>>t;
    while(t--){
        min = 1000000;
        scanf("%d %d", &n, &h);
        while(n--){
            cin>>a;
            diff = abs(h-a);
            if(diff<min){
                min = diff;
            }

        }
        cout<<min<<"\n";
    }
    return 0;
}
