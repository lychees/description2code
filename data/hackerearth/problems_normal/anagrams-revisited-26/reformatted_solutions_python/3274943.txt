N = int(raw_input())
strings = [raw_input() for x in xrange(0, N)]
maxCount = 0
resultDict = {}
sortedList = [''.join(sorted(string)) for string in strings]
for string in sortedList:
    count = resultDict.get(string, 0)
    resultDict[string] = count+1
    if (maxCount < count+1):
        maxCount = count+1
print maxCount
