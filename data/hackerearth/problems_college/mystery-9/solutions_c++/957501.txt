#include <iostream>
using namespace std;

int main() {
	long int t;
	int i;
	cin>>t;
	while(t--)
	{
		cin>>i;
		if(i<=31)
		cout<<"1";
		else if(i<=59)
		cout<<"2";
		else if(i<=90)
		cout<<"3";
		else if(i<=120)
		cout<<"4";
		else if(i<=151)
		cout<<"5";
		else if(i<=181)
		cout<<"6";
		else if(i<=212)
		cout<<"7";
		else if(i<=243)
		cout<<"8";
		else if(i<=273)
		cout<<"9";
		else if(i<=304)
		cout<<"10";
		else if(i<=334)
		cout<<"11";
		else
		cout<<"12";
		cout<<endl;
	}
	return 0;
}
