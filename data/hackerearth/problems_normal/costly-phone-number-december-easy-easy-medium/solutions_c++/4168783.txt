#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,t,c,l,ans,p,q,r,v,u;
    int cd[10];
    int j,f;
    string s;
    cin>>t;
    while(t--)
    {
    	for(i=0;i<=9;i++)
    	{
    		cin>>cd[i];
    	}
    	cin>>l>>s;
    	while(1)
    	{
    		r=0;
			for(i=0;i<=9;i++)
			{
				for(j=0;j<=9;j++)
				{
					f=(i+j)%10;
					if(cd[i]+cd[j]<cd[f])
					{
						r=1;
						cd[f]=cd[i]+cd[j];
					}
				}
			}
			if(!r)
			{
				break;
			}
    	}
    	ans=0;
    	for(i=0;i<l;i++)
    	{
    		//cout<<s[i]-'0'<<" "<<mc[s[i]-'0']<<" ";
    		ans+=cd[s[i]-'0'];
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
