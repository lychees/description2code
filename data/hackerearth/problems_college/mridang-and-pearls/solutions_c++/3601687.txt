int A[1<<21];
vector < pii > V;
int dp[1<<21];
map < int ,int > m;
set < int > S;
int main()
{
    int n,start,flag;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        scanf("%d",&A[i]);
    for(int i=n;i>=1;i--)
    {
        dp[i]=dp[i+1];
        if(m[A[i]]!=0)
            dp[i]=-1;
        m[A[i]]++;
    }
    start=1,flag=0;
    for(int i=1;i<=n;i++)
    {
        if(S.find(A[i])!=S.end() )
        {
                    V.pb(mp(start,i));
                    start=i+1;
                    S.clear();
        }
        else
            S.insert(A[i]);
    }
    if(V.size()==0)
    {
        cout<<-1;
        return 0;
    }

    if(V.back().s!=n)
        V[V.size()-1].s=n;
    printf("%d\n",V.size());
    for(int i=0;i<V.size();i++)
        printf("%d %d\n",V[i].f,V[i].s);
