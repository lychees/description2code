#include <iostream>
#include <string>
using namespace std;

int main()
{
    int i, t, len;
    string a;
    cin>>t;
    
    while(t--)
    {
    	bool flag=true;
    	cin>>a;
    	len = a.length();
    	
    	for(i=0; i<len/2; i++)
    	{
    		if(a[i]!='1'&&a[i]!='0'&&a[i]!='8')
    		{
    			flag=false;
    			break;
    		}
    		if(a[i]!=a[len-1-i])
    		{
    			flag=false;
    			break;
    		}
    	}
    	if(a[i]!='1'&&a[i]!='0'&&a[i]!='8')
    	{
    		flag=false;
    	}
    	if(flag==false)
    		cout<<"NO\n";
    	else
    		cout<<"YES\n";
    }
    return 0;
}
