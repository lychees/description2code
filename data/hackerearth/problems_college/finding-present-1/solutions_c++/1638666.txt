#include <bits/stdc++.h>
using namespace std;
#define pb push_back
int main()
{
	std::ios_base::sync_with_stdio (false);
    int p,q;
    cin>>p;
    vector<int64_t>pp,qq;
    set<int64_t>rr;
    for(int i=0;i<p;i++)
    {
        int64_t x;
        cin>>x;
        pp.pb(x);
    }
    cin>>q;
    for(int i=0;i<q;i++)
    {
        int64_t x;
        cin>>x;
        qq.pb(x);
    }
    sort(qq.begin(),qq.end());
    sort(pp.begin(),pp.end());


    int en=0;
    for(int i=0;i<qq.size();i++)
    {
        if(pp[en]==qq[i])
        {
            en++;
            continue;
        }
        else
        {
            rr.insert(qq[i]);
        }
    }
    for(set<int64_t>::iterator itr=rr.begin(); itr!=rr.end();++itr)
        cout<<*itr<<" ";

     return 0;
}
