#include <iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,i,ans,temp;
    	cin>>n;
    	int a[n];
    	for(i=0;i<n;i++)
    	    cin>>a[i];  
    	    
    	if(a[0]==0) ans=0;
    	else if(a[0]==-1) ans=1;
    	else
    	{
    		i=0;
    		while(i<n)
    		{
    		if(a[i]!=-1 && a[i]!=n) {temp=a[i]; a[i]=n; i=temp;}
    		else if(a[i]==n) {ans=0;break;}
    		else if(a[i]==-1) {ans=1;break;}
    		}
    	}
    	if(ans) cout<<"Win"<<endl;
    	else cout<<"Lose"<<endl;
    }
    return 0;
}
