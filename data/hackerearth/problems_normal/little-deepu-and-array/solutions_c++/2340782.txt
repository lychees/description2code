#include <iostream>
using namespace std;

int main()
{
	int n,i;
    cin>>n;
    int arr[n];
    for(i=0;i<n;i++)
    {
    	cin>>arr[i];
    }
    int m;
    cin>>m;
    int ma[m];
    for(i=0;i<m;i++)
    {
    	cin>>ma[i];
    }
    for(i=0;i<m;i++)
    {
    	for(int k=0;k<n;k++)
    	{
    		if(ma[i]<arr[k])
    		{
    			arr[k]--;
    		}
    	}
    }
    
    for(i=0;i<n;i++)
    {
    	cout<<arr[i]<<" ";
    }
    return 0;
}
