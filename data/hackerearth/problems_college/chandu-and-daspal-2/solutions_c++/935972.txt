#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,x,y,i,j,ans,ans1;
	char s[100000];
	scanf("%d",&t);
	while (t --) {
		scanf("%d %d %s",&x,&y,s);
		ans1 = 0;
		while (1) {
			ans1++;
			ans = __gcd(x,y);
			x = x-ans;
			y = y-ans;
			if (x == 0 || y == 0) {
				if (ans1 % 2 == 1 && strcmp(s,"Chandu") == 0) {
					printf("Chandu\n");
				}
				if (ans1 % 2 == 0 && strcmp(s,"Chandu") == 0) {
					printf("Daspal\n");
				}
				if (ans1 % 2 == 1 && strcmp(s,"Daspal") == 0) {
					printf("Daspal\n");
				}
				if (ans1 % 2 == 0 && strcmp(s,"Daspal") == 0) {
					printf("Chandu\n");
				}
				break;
			}
		}
	}
}
