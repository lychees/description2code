import math
store = {0: 0}


def coins(n):
    if n in store:
        a = store[n]
    else:
        if n > 0:
            a = max(n, coins(n/2)+coins(n/3)+coins(n/4))
            store[n] = a
        else:
            a = n

    return int(a)


while True:
    try:
        print coins(int(raw_input()))
    except BaseException:
        break
