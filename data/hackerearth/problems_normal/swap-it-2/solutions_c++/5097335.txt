#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long mini,t,i,j,n,k,start;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		long long a[n+1];
		for(i=1;i<=n;i++)
			cin>>a[i];
		for(i=01;i<n;i++)
		{
			start=i;
			mini=a[i];
			for(j=i+1;j<=n;j++)
			{
				if((j-i)>k)
				{
					break;
				}
				if(a[j]<mini)
				{
					mini=a[j];
					start=j;
				}
			}
			for(j=start; j>i; j--)
            {
                long long temp;
                temp=a[j];
                a[j]=a[j-1];
                a[j-1]=temp;
            }
            k=k-(start-i);
		}
		for(i=1;i<=n;i++)
    	{
            cout<<a[i]<<" ";
        }
        cout<<endl;
	}
	return 0;
}
