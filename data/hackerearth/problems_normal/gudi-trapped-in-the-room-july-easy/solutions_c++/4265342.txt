#include <iostream>
#include <set>
#include <string>
using namespace std;



int A,H;
set<string> myset;



void bfs(string str){
	myset.insert(str);
	string newStr = str;
	for (int i=0;i < str.size();i++){
		newStr[(i+H)%(str.size())] = str[i];
	}
	
	if (myset.find(newStr) == myset.end()){
		bfs(newStr);
	}
	
	
	newStr = str;
	for (int i=0;i < str.size();i++){
		if (i%2){
			newStr[i] = '0' + (str[i] - '0' + A)%10;
		}
	}
	
	if (myset.find(newStr) == myset.end()){
		bfs(newStr);
	}
	
	
}

int main() {
	int testcases;
	cin >> testcases;
	
	
	while (testcases--){
		string str;
		A = H = 0;
		cin >> str;
		cin >> A >> H;
		myset.clear();
		bfs(str);
		cout << *(myset.begin());
		if (testcases > 0){
			cout << endl;
		}
	}
	
	
	
}
