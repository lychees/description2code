#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int tc;
   cin>>tc;
   while(tc--)
   {
   	int k;
   	string s;
   	cin>>k>>s;
   	int t = k;
   	int n = s.size();
   	for(int i=0;i<n;++i)
   	{
   		k=t;
   		int v = s[i];
   	for(int j=1;j<=k;j++)
   	{
   		if(v==65)
   			v=90;
   		else	v--;
   	}
   		char c = (char)v;
   		cout<<c;
   	}
   	cout<<endl;
   }
    return 0;
}
