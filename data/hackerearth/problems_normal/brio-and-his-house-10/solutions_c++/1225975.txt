#include <iostream>
#include<cstdio>
#include<cmath>
using namespace std;

int main()
{
    int t;
    
    double a,b,c,s,r,d,area;
    
    scanf("%d",&t);
    
    while(t--)
    {
    scanf("%lf%lf%lf",&a,&b,&c);
    
    d=(a+b+c)*(b+c-a)*(c+a-b)*(a+b-c);
    
    d=sqrt(d);
    r=(a*b*c)/d;
    
    area=3.1415*r*r;
    
    printf("%.4lf\n",area);
    }
    return 0;
}
