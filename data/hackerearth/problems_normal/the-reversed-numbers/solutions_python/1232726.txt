t = input()
o = []
for i in range(0, t):
	n = raw_input().partition(' ')
	m = list(n[0])
	p = list(n[2])
	m.reverse()
	p.reverse()
	s1 = m[0]
	for j in range(1, len(m)):
		s1 = s1 + m[j]
	s2 = p[0]
	for j in range(1, len(p)):
		s2 = s2 + p[j]
	g = int(s1) + int(s2)
	while (g % 10 == 0):
		g = g / 10
	l = list(str(g))
	l.reverse()
	s3 = l[0]
	for j in range(1, len(l)):
		s3 = s3 + l[j]
	o.append(s3)
for i in o:
	print i
