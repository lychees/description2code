#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007

ll fast_exp(ll base, ll exp)
{
    ll res=1;
    while(exp>0)
    {
       if(exp%2==1) res=(res*base)%mod;
       base=(base*base)%mod;
       exp/=2;
    }
    return res;
}

int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	ll n,arr[22];
    	cin>>n;
    	for(int i=0;i<n;i++) cin>>arr[i];
    	ll ans=0;
    	for(int i=0;i<(1<<n);i++)
    	{
    		vector< ll > v;
    		for(int j=0;j<n;j++)
    		{
    			if(i&(1<<j)) v.push_back(arr[j]);
    		}
    		if((v.size()%2)||v.empty()) continue;
    		sort(v.begin(),v.end());
    		ll k=v.size();
    		ll p=1,m=(k+1)/2;
    		for(int j=0;j<m;j++) p=(p*v[j])%mod;
    		p=fast_exp(p,mod-2);
    		for(int j=m;j<k;j++) p=(p*v[j])%mod;
    		ans=max(p,ans);
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
