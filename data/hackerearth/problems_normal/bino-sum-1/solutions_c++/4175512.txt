#include<iostream>

using namespace std;

#define mod 1000000007

long long int dp[1001][1001];

void init(){
    for(int i=1;i<=1000;i++){
        for(int j=0;j<=i;j++){
            if(j == 0) dp[i][j] = 1;
            else if(j == i) dp[i][j] = 1;
            else{
                dp[i][j] = (dp[i-1][j] + dp[i-1][j-1] ) %mod ;
            }
        }
    }
}

int main(){
    init();
    int t;
    cin >> t;
    while(t--){
        int n, p;
        cin >> n >> p;
        long long int sum = 0;
        for(int i=0;i<=p;i++){
            sum += dp[n][i];
            sum %= mod;
        }
        cout << sum << endl;
    }
    return 0;
}
