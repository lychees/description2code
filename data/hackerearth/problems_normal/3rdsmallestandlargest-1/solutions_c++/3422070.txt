#define MOD 1000000007
string s;
void swap(string& s,int i,int j)
{
	char c=s[i];
	s[i]=s[j];
	s[j]=c;
}
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int T;
	int i,n;
	string t;
	cin>>T;
	while(T--)
	{
		cin>>s;
		n=s.size();
		sort(s.begin(),s.end());
		if(n<=2||s[0]==s[n-1])
			cout<<"Not possible!"<<endl;
		else
		{
			t=s;
			for(i=n-2;i>=0;i--)
				if(s[i]!=s[n-1])
					break;
			if(i==n-2)
			{
				if(s[i]==s[i-1])
					swap(s,i-1,i+1);
				else
					swap(s,i-1,i);
			}
			else
			{
				swap(s,i,i+2);
			}
			cout<<s<<' ';
			s=t;
			reverse(s.begin(),s.end());
			for(i=n-2;i>=0;i--)
				if(s[i]!=s[n-1])
					break;
			if(i==n-2)
			{
				if(s[i]==s[i-1])
					swap(s,i-1,i+1);
				else
					swap(s,i-1,i);
			}
			else
			{
				swap(s,i,i+2);
			}
			cout<<s<<endl;
		}
	}
//	system("pause");
	return 0;
}
