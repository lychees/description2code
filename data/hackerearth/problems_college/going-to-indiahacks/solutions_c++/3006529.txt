#define LL long long
const int MOD = 1e9 + 7;
const int INF = 1e9 + 9;
const int MX = 1e5 + 5;

LL inp[1005][1005];

int main()
{
	int t;
	cin >> t;
	for(int j=0;j<1005;j++)
		inp[0][j] = 1;
	for(int i=0;i<1005;i++)
		inp[i][0] = 1;
	for(int i=1;i<1005;i++)
	{
		for(int j=1;j<1005;j++)
		{
			inp[i][j] = ((inp[i-1][j] + inp[i][j-1])%MOD + inp[i-1][j-1])%MOD;
		}
	}
	int a,b,c,d;
	while(t--)
	{
		cin >> a >> b >> c >> d;
		c = c - a;
		d = d - b;
		cout << inp[c][d] << endl;
	}

	return 0;
}
