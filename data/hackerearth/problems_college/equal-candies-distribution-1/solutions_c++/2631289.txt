#include <iostream>
#include<algorithm>
using namespace std;
 
int main()
{ long long int t;
cin>>t;
while(t--)
{ long long int n;int s=0,c=0;
  cin>>n;
  int a[n],b[n];
  for(int i=0;i<n;i++) cin>>a[i];
  sort(a,a+n);
for(int j=0;j<n;j++) b[j]=a[j]-a[0];
for(int k=0;k<n;k++)
{ s=b[k];
  c=c+s/5;
  s=s%5;
   c=c+s/2;
  s=s%2;
   c=c+s/1;
}
cout<<c<<endl;
}
    return 0;
}
