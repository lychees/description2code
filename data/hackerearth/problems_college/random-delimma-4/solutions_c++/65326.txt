#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        
            long long n,k;
            cin>>k>>n;
            if(k==1)cout<<"0.000000000"<<endl;
            else if(k>n )cout<<"1.000000000"<<endl;
            //else if(k==1
            else
            {
                double give_ans=1;
                double temp=1;
                for(int i=1;i<=k;i++)
                {
                    temp*=(double)(n-k+i)/(double)n;
                }
                give_ans-=temp;
                
                printf("%0.9lf\n",give_ans);
            }
        }
    }
