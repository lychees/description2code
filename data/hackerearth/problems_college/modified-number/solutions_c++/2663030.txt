#include <iostream>
using namespace std;

long a[]={1, 9, 45, 55, 99, 297, 703, 999, 2223, 2728, 4879, 4950, 5050, 5292, 7272, 7777, 9999, 17344, 22222, 38962, 77778, 82656, 95121, 99999};
int main()
{
	long p,q,i,cnt=0;
	cin>>p>>q;
	for(i=0;a[i]<100000;i++)
		if(a[i]>=p&&a[i]<=q){
			cnt=1;
			cout<<a[i]<<" ";
		}
	if(cnt==0)
	cout<<"INVALID RANGE";
    return 0;
}
