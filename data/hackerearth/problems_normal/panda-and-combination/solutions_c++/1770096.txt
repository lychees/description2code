#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    long long n,m,mul,ans;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld %lld",&n,&m);
        if(m==1)printf("%lld\n",n);
        else
        {
            ans=0;
            mul=1;
            while(135)
            {
                if(log10(mul)+log10(m)+1>15)break;
                if(mul*m>n)break;
                mul*=m;
            }
            while(mul)
            {
                ans+=n/mul;
                n%=mul;
                mul/=m;
            }
            printf("%lld\n",ans);
        }
    }
}
