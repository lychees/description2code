#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	int a,b,d,p,i;
    	long long ans=0;
    	cin>>d>>p;
    	if(d<0)
    	{
    		cout<<ans<<endl;
    		continue;
    	}
    	if(p==0)
    	
    	{if(d==0) cout<<"1\n";else cout<<"4\n";}
    	if(p>0)
    	{
        for(i=1;i*i<=p;i++)
        {
        	if(p%i==0)
        	{
        		a=i;
        		b=p/i;
        		if(abs(a-b)==d)
        		{
        			if(a==b)
        			ans+=2;
        			else
        			ans+=4;
        		}
        	}
        }
        cout<<ans<<endl;
    	}
     if(p<0)
     {
     	p=abs(p);
     	for(i=1;i*i<=p;i++)
     	{
     		if(p%i==0)
     		{
     			a=i;b=p/i;
     			if(abs(a+b)==d)
     			{
     				if(a==b)ans+=2;
     				else ans+=4;
     			}
     		}
     	}
     	cout<<ans<<endl;
     }
    }
    return 0;
}
