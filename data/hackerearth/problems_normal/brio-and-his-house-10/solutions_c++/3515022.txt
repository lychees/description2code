#include <iostream>
#include <math.h>
#include <iomanip>
#define pi 3.1415
using namespace std;

int main()
{

	int t,n;
	double a,b,c,m,l,s,a1,r;
	
	cin >> t;
	getchar();
	
	while(t--)
	{
	cin >> a >> b >> c;
	
	s = (a+b+c)/2;
	
	a1 = sqrt(s*(s-a)*(s-b)*(s-c));
	
	r = a*b*c/(4*a1);
	
	a1 = pi *r*r;
	
	cout  <<setprecision(4) <<fixed <<a1 << endl;
	}
	
    return 0;
}
