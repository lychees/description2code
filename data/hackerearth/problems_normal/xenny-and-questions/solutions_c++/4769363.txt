#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int n,k;
   	scanf("%d %d",&n,&k);
   	int arr[n];
   	for(int i=0;i<n;i++) scanf("%d",&arr[i]);
   	sort(arr,arr+n);
   	long long int ans=0;
   	for(int i=n-1;i>=0 && i>n-1-k;i--)
   	{
   		ans+=arr[i];
   	}
   	printf("%lld\n",ans);
   }
    return 0;
}
