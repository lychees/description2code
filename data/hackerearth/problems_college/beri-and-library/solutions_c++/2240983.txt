#include<bits/stdc++.h>
using namespace std;
int countSetBits(long long int n)
{
  int count = 0;
  while(n)
  {
    count += n & 1;
    n >>= 1;
  }
  return count;
}
 
int main()
{
	long long int t;
	scanf("%lld",&t);
	while(t--)
	{
	
    long long int n;
    scanf("%lld",&n);
    printf("%d\n", countSetBits(n));
    getchar();

}
return 0;
}
