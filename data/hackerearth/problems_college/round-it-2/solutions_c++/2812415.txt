#include<bits/stdc++.h>
using namespace std;

int main()
{
    int i,t,n,num,q,rem;
    cin>>t;
    while(t--)
    {
        cin>>num>>n;
        q=num/n;
        rem=num%n;
        i=n-rem;
        if(rem <(float)n/2 && num != (float)n/2)
            cout<<q*n<<endl;
        else
        {
            if(num == n/2 )
                cout<<2*num<<endl;
            else
            {
                if(rem<=n/2 && n!=2)
                    cout<<num - rem<<endl;
                else
                    cout<<num+i<<endl;
            }
        }
    }
    return 0;
}
