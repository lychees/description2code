#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
	int t;
	cin>>t;
	while(t--){
		string a,b;
		cin>>a>>b;
		ll num=0,mod=0;
		for(int i=0;i<b.length();i++){
			mod=(mod*10 + (b[i]-'0'));
		}
		for(int i=0;i<a.length();i++){
			num=(num*10+(a[i]-'0'))%mod;
		}
		cout<<num%mod<<endl;
	}	
	return 0;
}
