#include <iostream>
#include <algorithm>
using namespace std;

int n, a[100001];

int main()
{
    cin >> n;
    for(int i = 0; i < n; ++i)
    	cin >> a[i];
    sort(a, a+n);
    int ans = 0, prev = -1;
    for(int i = 0; i < n; ++i) {
    	if(a[i] > prev) {
    		prev = a[i]+4;
    		ans++;
    	}
    }
    cout << ans;
    return 0;
}
