#include <stdio.h>
#define max 10000001

int a[max] = {0};
int sum[max] = {0};

void sieve ()
{
	int i;
	for (i = 2; i < max ; i++)
	{
		if (a[i] == 0)
		{
			a[i] = 1;
			int j;
			for (j = 2*i ; j < max ; j += i)
			{
				int count = 0;
				int t = j;
				while (t % i == 0)
				{
					count++;
					t = t/i;
				}
				a[j] += count;
			}
			sum[i] = sum[i-1] + a[i];
		}
		else
		{
			sum[i] = sum[i-1] + a[i];
		}
	}
}

int main ()
{
	sieve ();
	int t;
	scanf ("%d",&t);
	while (t--)
	{
		int n;
		scanf ("%d",&n);
		printf ("%d\n",sum[n]);
	}
	return 0;
}
