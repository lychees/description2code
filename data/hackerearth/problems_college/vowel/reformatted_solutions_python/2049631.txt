import sys
for tc in range(input()):
    s = raw_input()
    x = []
    for j in range(len(s)):
        if s[j] == 'a' or s[j] == 'e' or s[j] == 'i' or s[j] == 'o' or s[j] == 'u' or s[
                j] == 'A' or s[j] == 'E' or s[j] == 'I' or s[j] == 'O' or s[j] == 'U':
            x.append(s[j])
    i = 0
    if len(x) == 0:
        print "No"
        continue
    while i < len(x):
        sys.stdout.write(x[i])
        i = i+1
