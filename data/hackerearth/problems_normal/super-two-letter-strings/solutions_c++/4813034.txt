#include<iostream>
using namespace std;
#define mod 1000000007
long long dp[10001];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,p;
		cin >> n >> p;
		dp[0]=1;
		for(int i=1;i<=n;i++)dp[i]=0;
		for(int i=1;i<=n;i++)
		{
			for(int j=i-1;j>=0 && j>=i-p;j--)
			{
				dp[i]=dp[i]+dp[j];
				dp[i]=dp[i]%mod;
			}
		}
		cout<<dp[n]<<endl;
	}
	return 0;
}
