using namespace std;
#define MOD 1000000007
typedef long long LL;
typedef pair<int, int> ii;
bool comp (LL a, LL b) {
	return a>b;
}
int main(){
    ios_base::sync_with_stdio(false);
    int T;
    cin >> T;
    for (int i = 1; i <= T; i++) {
    	int N;
    	cin >> N;
    	vector<LL> A(N), B(N);
    	for (int j = 0; j < N; j++)
    		cin >> A[j];
    	for (int j = 0; j < N; j++)
    		cin >> B[j];
    	
    	LL ans = 0;
    	for (int j = 0; j < N; j++) {
    		LL b = B[j];
    		auto it = lower_bound(A.begin(), A.end(), b, comp);
    		int i = it-A.begin();
    		if (*it <= b && i <= j) {
    			if (j-i > ans)
    				ans = j-i;
    		}
    	}
    	cout << ans << endl;
    }
    return 0;
}
