#include <iostream>
#include <map>

using namespace std;


int main(){
	

	int num;

	int max_sum = 0;
	int sum;
	int max_num;
	int curr_num;

	while (cin >> num)
	{
		curr_num = num;
		sum = 0;

		while(num > 0)
		{
			sum = sum + num% 10;
			num = num/10;
		}

		if (sum > max_sum)
		{
			max_sum = sum;
			max_num = curr_num;
		}

	}

	cout << max_num << endl;



	return 0;
}
