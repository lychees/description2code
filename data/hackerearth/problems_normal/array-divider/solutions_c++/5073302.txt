#include <bits/stdc++.h>
using namespace std;

long int sum[505];
int n,k,a[505];
long long int dp[305][505];

long int getSum(int i,int j)
{
    return sum[j] - sum[i-1];
}

int main()
{
    cin>>n>>k;
    for(int i=1;i<=n;i++)
        cin>>a[i];
    sum[0] = 0;
    for(int i=1;i<=n;i++)
        sum[i] = sum[i-1] + a[i];

    for(int i=1;i<=n;i++)
        dp[1][i] = (k-1)*sum[i]*sum[i];

    for(int i=2;i<=n;i++)
        for(int p=2;p<=i&&p<=k;p++)
        {
            dp[p][i] = 1e18;
            for(int j=p;j<=i;j++)
                dp[p][i] = min(dp[p][i],dp[p-1][j-1] + (k-1)*getSum(j,i)*getSum(j,i) - 2*getSum(j,i)*sum[j-1]);
        }
    cout<<dp[k][n]<<endl;
    return 0;
}
