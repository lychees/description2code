#include <bits/stdc++.h>
using namespace std;
#define pb push_back
bool isprime[5005];
int arr[5005];
int dp[5005];
vector<int> primes;
int main()
{
    int t,n;
    cin>>n;
    for(int i=1;i<=n;++i){
    	cin>>arr[i];
    	arr[i]+=arr[i-1];
    }
    memset(isprime,true,sizeof(isprime));
    for(int i=2;i<=5000;++i){
    	if(isprime[i]){
    		primes.pb(i);
    		for(int j=i*i;j<=5005;j+=i){
    			isprime[j]=false;
    		}
    	}
    }
    dp[0]=dp[1]=0;
    for(int i=2;i<=n;++i){
    	dp[i]=dp[i-1];
    	if(isprime[i]){
    		dp[i]=arr[i];
    		continue;
    	}
    	for(int j=0;j<primes.size() && (i-primes[j])>0;++j){
    		dp[i]=max(dp[i],dp[i-primes[j]-1]+arr[i]-arr[i-primes[j]]);
    	}
    }
    cout<<dp[n]<<endl;
    return 0;
}
