#include <iostream>

using namespace std;

int main()
{
	long sum, count = 0;
	long N, K, D;
	cin >> N >> K >> D;

	long * list = new long[N];

	for (long n = 0; n < N; n++)
	{
		cin >> list[n];
	}
	long lastzeroindex;

	for (long n = 0; n <= N - K; n++)
	{
		sum = 0;
		lastzeroindex = -1;
		
		for (long k = n; k < min(n + K, N); k++)
		{
			sum += list[k];
			if (list[k] != 0)
			{
				lastzeroindex = k;
			}
		}
		if (sum == 0)
		{
			count = -1;
			break;
		}
		if (sum < D)
		{
			count++;
			
			n = lastzeroindex + K - 1;
		}
	}
	cout << count << endl;
	return 0;
}
