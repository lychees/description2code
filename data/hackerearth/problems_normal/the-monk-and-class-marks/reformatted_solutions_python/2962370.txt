
noofstudents = int(raw_input())
studenthash = {}


def processRecord(record):
    (name, marks) = record.split()
    marks = int(marks)
    if marks in studenthash:
        studenthash[marks].append(name)
    else:
        studenthash[marks] = [name]


count = 0
while count < noofstudents:
    record = raw_input()
    processRecord(record)
    count += 1

markslist = studenthash.keys()
markslist.sort(reverse=True)
for marks in markslist:
    names = studenthash[marks]
    if len(names) > 1:
        names.sort()
    for name in names:
        print("{} {}".format(name, marks))
