ll i,j,k,n,t,a[100002],c;

ll fun(ll val){
	ll cnt=1;
	ll pos=a[0];
	for(i=1;i<n;i++){
		if(a[i]-pos>=val){
			cnt++;
			if(cnt==c) return 1;
			pos=a[i];
		}
	}
	return 0;
}

int main(){
	inpll(t);
	while(t--){
		inpll(n); inpll(c);
		for(i=0;i<n;i++) inpll(a[i]);
		sort(a,a+n);
		ll low=1,high=a[n-1];
		while(low<high){
			//printf("low,high %d %d\n",low,high);
			ll mid=(low+high)/2;
			if(fun(mid)) low=mid+1;
			else high=mid;
		}
		cout<<low-1<<'\n';
	}
	return 0;
}
