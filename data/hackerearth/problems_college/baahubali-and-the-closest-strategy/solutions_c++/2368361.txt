#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll printClosest(ll a[], ll n, ll x)
{
    ll res_l, res_r;  // To store indexes of result pair
 
    // Initialize left and right indexes and difference between
    // pair sum and x
    ll l = 0, r = n-1, diff = INT_MAX;
 
    // While there are elements between l and r
    while (r > l)
    {
       // Check if this pair is closer than the closest pair so far
       if (abs(a[l] + a[r] - x) < diff)
       {
           res_l = l;
           res_r = r;
           diff = abs(a[l] + a[r] - x);
       }
 
       // If this pair has more sum, move to smaller values.
       if (a[l] + a[r] > x)
           r--;
       else // Move to larger values
           l++;
    }
 
    return (a[res_r]+a[res_l]);
}
int main() {
    ll n,m,i,x,a[1001];
scanf("%lld%lld",&n,&m);
for(i=0;i<n;i++)
{
    scanf("%lld",&a[i]);
}
sort(a,a+n);
while(m--)
{
    scanf("%lld",&x);
   printf("%lld\n",printClosest(a,n,x));
}
return 0;
}
