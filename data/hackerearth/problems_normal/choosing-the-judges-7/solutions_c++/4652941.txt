#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t,i,n,x=1;
   cin>>t;
   while(x<=t){
   	long long int sol[10000]={0};
   	cin>>n;
   	int arr[n];
   	for(i=0;i<n;i++)
   		cin>>arr[i];
   		sol[0]=arr[0];
   		sol[1]=arr[0]>arr[1]?arr[0]:arr[1];
   		for(i=2;i<n;i++){
   			sol[i]=(sol[i-2]+arr[i])>sol[i-1]?(sol[i-2]+arr[i]):sol[i-1];
   		}
   		printf("Case %d: %lld\n",x,sol[n-1]);
   		x++;
   }
   
    return 0;
}
