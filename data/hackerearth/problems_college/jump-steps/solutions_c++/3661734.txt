#include <iostream>
using namespace std;

int main()
{
    int t,f;
    long long int n,ans;
    cin>>t;
    while(t--)
    {
    	ans=0;
    	cin>>n;
    	if(n==0)
    	{
    		cout<<ans<<endl;
    		break;
    	}
    	while(n>1)
    	{
    		f=n%2;
    		if(f==1)
    		ans++;
    		n=n/2;
    	}
    	
    	ans++;
    	cout<<ans<<endl;
    	
    }
    return 0;
}
