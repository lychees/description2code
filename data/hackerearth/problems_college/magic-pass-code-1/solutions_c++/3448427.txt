    	listl.push_back(num);
    	
    	int sum=0;
    	for (int x=0;x<num.length();x++)

    	   char c=num[x];

    	   sum=sum+tmp;

    	suml.push_back(sum);
    }
    for (int x=0;x<listl.size();x++)

    	    if (suml[x]<suml[y]) {swap(suml[x],suml[y]); swap(listl[x],listl[y]);}
    	}
    }
