#include <iostream>
using namespace std;
int time_period[1001];
int beep_value[1001];
int intensity_value[1001];
int intensity_arr[100001];
int beep_arr[100001];
int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(false);
	int n, p, q, temp;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>temp, time_period[temp]++;// cin>>time_period[i];

	cin>>p;
	for(int i=1;i<=p;i++)
		cin>>beep_value[i]>>intensity_value[i];

	// for(int i=0;i<n;i++)
	// 	for(int j=time_period[i];j<=10000;j+=time_period[i])
	// 		intensity_arr[j]++;
	for(int i=1;i<=1000;i++)
	{
		if(time_period[i])
		{
			for(int j=i;j<=100000;j+=i)
			{
				intensity_arr[j]+=time_period[i];
			}
		}
	}

	for(int i=1;i<=100000;i++)
		beep_arr[i]=!!intensity_arr[i];

	for(int i=1;i<=100000;i++)
		beep_arr[i]+=beep_arr[i-1];

	for(int i=1;i<=100000;i++)
		intensity_arr[i]=max(intensity_arr[i], intensity_arr[i-1]);

	cin>>q;
	while(q--)
	{
		int in, t;
		cin>>in>>t;
		//cout<<intensity_arr[t] <<"   "<<intensity_value[in]<<"\n";
		if((intensity_arr[t]>intensity_value[in])  ||  (beep_arr[t]>beep_value[in]))
			cout<<"Awake\n";
		else
			cout<<"Asleep\n";
	}
	return 0;
}
