#include <iostream>
#include <cmath>

using namespace std;

int main()
{
  int tc;
  int n;
  int i,j;
  
  cin >> tc;
  while (tc--)
	{
	  cin >> n;

	  bool res = false;
	  for (i=2; i*i < n && !res; i++)
		{
		  // int v = i*i;
		  for (int k=i*i; k<n && !res; k*=i)
			{
			  
			  if (n-k==1)
				{
				  res = true;
				  break;
				}
		  
			  for (j=i; j*j <= n-k && !res; j++)
				{
				  int l;
				  for (l=j*j; l<n-k; l*=j);
				  if (l == n-k)
					{
					  // cout << i << " " << j << endl; 
					  res = true;
					  break;
					}
				}
			}
		}
	  if (res || n==2)
		cout << "Yes" << endl;
	  else
		cout << "No" << endl;
	  
	}

  return 0;
}
