	int t;
	cin>>t;
	while(t--)
	{
		long long int x,y,n,i;
		cin>>x>>y>>n;
		long long int sum=x+y,temp;
		for(i=3;i<=n;i++)
		{
			if(x>y)
			{
				temp=x-y;
				sum+=x-y;
				x=y;
				y=temp;
			}
			else if(y>x)
			{
			    temp=y-x;
				sum+=temp;
				x=y;
				y=temp;
			}
			else
			{
				temp=0;
				sum+=temp;
				x=y;
				y=temp;
			   
			   temp=(n-i)/3;
			   sum+=(temp)*2*x;
			   temp=(n-i)%3;
			   sum+=temp*x;
			   
			   break;
			   
			}
		}
		cout<<sum<<endl;
	}

}
