#include<iostream>
#include<stdio.h>

using namespace std;

long long int dp[1000001];

int main()
{
	dp[0]=dp[1]=0;
	for(int i=2;i<1000001;i++){
		if(dp[i]>0) continue ;
		for(int j=i;j<1000001;j+=i){
			int k=j;
			while(k%i==0){
					dp[j]++;
					k /=i;
			}			
		}		
	}
	
	//cumulative sum 
	for(int i=3;i<1000001;i++){
		dp[i] +=dp[i-1];
	}
	
	int T=0;
	scanf("%d",&T);
	
	while(T--){
		int a =0;
		int b=0;
		scanf("%d",&a);
		scanf("%d",&b);
		printf("%lld\n",dp[b]-dp[a-1]);			
	}
    return 0;
}
