#include <iostream>
using namespace std;

int fun(string s1)
{
	int len1=s1.length(),i;
	for(i=0;i<len1/2;i++)
	{
		if(s1[i]!=s1[len1-i-1])
		break;
	}
	if(i==(len1/2))
	return 1;
	return 0;
}
int main()
{
	int ans;
   string str1,str2,tmp;
  int t;
  cin>>t;
  while(t--)
  {
  	ans=0;
  cin >>str1>>str2;
   tmp=str2+str1;
   //cout<<tmp<<endl;
   ans+=fun(tmp);
   for(int i=0;i<str1.length()-1;i++)
   {
   	tmp=str1.substr(0,(i+1))+str2+str1.substr(i+1,(str1.length()-(i+1)));
  // cout<<tmp<<endl;
   ans+=fun(tmp);
   }
   tmp=str1+str2;
   //cout<<tmp<<endl;
   ans+=fun(tmp);
   printf("%d\n",ans);
  }
    return 0;
}
