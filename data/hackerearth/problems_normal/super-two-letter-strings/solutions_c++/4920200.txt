#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll dp[10005][2],n,p;
ll rec(ll idx,int x)
{
    ll ans=0;
    if(idx>n)
        return 0;

        if(idx==n)
            return 1;
        if(dp[idx][x]!=-1)
            return dp[idx][x];
    ans=rec(idx+1,0);
    if(x==0)
    {
         for(int i=1;i<p;i++)
        {
            if(i!=p)
            ans=(ans+rec(idx+i,1))%1000000007;
        }
    }

    return dp[idx][x]=ans;
}
int main()
{
    ll t,i,ans;
    cin>>t;
    while(t--)
    {
        cin>>n>>p;
        memset(dp,-1,sizeof dp);
        ans=rec(1,0);
        cout<<ans<<"\n";
    }
    return 0;
}
