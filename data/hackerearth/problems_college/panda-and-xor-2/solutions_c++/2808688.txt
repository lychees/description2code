#include <iostream>
#include<cmath>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
    long t;
    cin>>t;
    while(t--)
    {
    	long a,b,ans;
    	cin>>a>>b;
    	ans=a^b;
    	if(__builtin_popcountl(ans)%2==0)
    	{
    		cout<<"YES"<<"\n";
    	}
    	else
    	{
    		cout<<"NO"<<"\n";
    	}
    }
    return 0;
}
