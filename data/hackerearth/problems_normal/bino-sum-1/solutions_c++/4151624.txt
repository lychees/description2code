#include "bits/stdc++.h"
using namespace std;

#define mod 1000000007

int dp[1001][1001];
int n,p;
void filldp()
{
   dp[0][0] = 1;
   for (int i = 1; i < 1001; i += 1)
   {
   	 for (int j = 0; j < i+1; j += 1)
   	 {
   	 	 if(i == j) { dp[i][j] = 1; continue;}
   	 	 if(j == 0) { dp[i][j] = 1; continue;}
   	 	 dp[i][j] = dp[i-1][j-1] + dp[i-1][j];
   	 	 if(dp[i][j] > mod) dp[i][j]  -= mod;
   	 }
   }  
}

int cal()
{
  int ans = 0;
  for(int j = 0 ; j <= p; j++)
   {
     ans+=dp[n][j];
     if(ans>mod)
      ans = ans-mod;
   }
  return ans;
}

int main (int argc, char const* argv[])
{
	ios_base :: sync_with_stdio(false);
	filldp();
	int t; cin >> t;
	while (t--)
	{
		cin >> n >> p;
		printf("%d\n",cal());
	}
	return 0;
}
