t = input()

for i in xrange(t):
	A, B, N = map(int, raw_input().split())
	calories = map(int, raw_input().split())
	calories.sort()

	sayan = 0
	raghu = 0

	for r in calories:
		if r > A:
			break
		else:
			A = A - r
			raghu += 1

	for r in calories:
		if r > B:
			break
		else:
			B = B - r
			sayan += 1

	if sayan > raghu:
		print "Sayan Won"
	elif raghu > sayan:
		print "Raghu Won"
	else:
		print "Tie"
