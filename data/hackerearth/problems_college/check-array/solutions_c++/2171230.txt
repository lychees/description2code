#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t, n, i, x;
    long long unsigned int s1, s2;
    cin>>t;
    while(t--) {
    	vector <int> a;
    	s1=0;
    	s2=0;
    	cin>>n;
    	for(i=0;i<n;i++) {
    		cin>>x;
    		a.push_back(x);
    		s1+=x;
    	}
    	s2=a[0];
    	for(i=1;i<n;i++) {
    		s2+=a[i-1];
    		s1-=a[i];
    		if(s1==s2) {
    			cout<<"YES";
    			break;
    		}
    	}
    	if(i==n)
    		cout<<"NO";
    	cout<<endl;
    	a.clear();
    }
    return 0;
}
