#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int k;
	cin>>k;
	
	string s;
	cin>>s;
	
	set<char> set1;
	long int ans=0;
	
	long int len=s.length();
	for(long int i=0;i<len;i++)
	{
		set1.clear();
		for(long int j=i;j<len;j++)
		{
			set1.insert(s[j]);
			long int size=set1.size();
			if(size == k)
				ans++;
		}
	}
	cout << ans << endl;
	return 0;
}
