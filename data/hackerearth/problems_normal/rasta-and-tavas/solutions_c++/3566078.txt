#include<stdio.h>
#include<math.h>
long int sum_prime_facts(long int n){
        long int sum=0;
        if(n%2==0)
                sum+=2;
        while (n%2 == 0)
        {
                n = n/2;
        }
    for (int i = 3; i <= sqrt(n); i = i+2)
    {
        if(n%i==0)
                sum+=i;
        while (n%i == 0)
        {
            n = n/i;
        }
    }
    if (n > 2)
        sum+=n;
        return sum;
}
int main(){
        long int n, k, a, count=0;
        scanf("%ld %ld", &n, &k);
        for(long int i =2; i<=n; i++){
                if(sum_prime_facts(i)==k)
                        count++;
        }
        printf("%ld\n", count);
}
