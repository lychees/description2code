for _ in xrange(input()):
    k = input()
    s = raw_input()
    cnt = 0
    for i in xrange(len(s)):
        sm = 0
        for j in xrange(i, len(s)):
            sm += ord(s[j])-96
            if sm == k:
                cnt += 1
                break
            if sm > k:
                break

    print cnt
