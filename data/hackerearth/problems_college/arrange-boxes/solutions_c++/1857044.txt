#include<bits/stdc++.h>
using namespace std ;
int main()
{
    int i,j,k,n,m,p,t ;
    int swap ;
    int a[100001] ;
    int b[100001] ;
    cin >> t ;
    while(t--) {
        swap = 0 ;
        cin >> n ;
        for(i=0;i<n;i++) cin >> a[i] ;
        for(i=0;i<n;i++) cin >> b[i] ;
        for(i=0;i<n;i++) {
            if(a[i] == b[i]) continue ;
            else {
                for(j=i+1;j<n;j++) {
                    if(a[j] == b[i]) break ;
                }
                for(;j>i;j--) {
                    a[j] ^= a[j-1] ^= a[j] ^= a[j-1] ;
                    swap++ ;
                }
            }
        }
        cout << swap << endl ;
    }
    return 0 ;
}
