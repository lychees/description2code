#include <iostream>
using namespace std;

int main()
{
    int t;
    long int n, m;
    long int a, b;
    int k;
    
    cin>>t;
    while(t--){
    	
    	cin>>n>>m;
    	long int jump = 0;
    	cin>>k;
    	long int x = 1;
    	long int y = 1;
    	while(k--){
    		cin>>a>>b;
    		
    		while( (x+a) <= n && (x+a) > 0 && (y+b) <= m && (y+b) > 0 ){
    			jump++;
    			x += a;
    			y += b;
    		}
    	}
    	
    	cout<<jump<<endl;
    }
    return 0;
}
