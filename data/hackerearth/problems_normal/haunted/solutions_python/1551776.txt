import sys

def fun():
	n, m = [int(x) for x in sys.stdin.readline().rstrip().split(" ")]
	a = [int(x) for x in sys.stdin.readline().rstrip().split(" ")]
	count = { }
	
	maxm=a[0]
	
	for i in xrange(0,n):
		if a[i] not in count:
			count[a[i]]=1
		else:			
			count[a[i]]+=1
		
		if (count[a[i]]>count[maxm]) or (count[a[i]]==count[maxm] and a[i]>maxm):
			maxm=a[i]
		print "{0} {1}".format(maxm, count[maxm])
if __name__ == "__main__":
    fun()
