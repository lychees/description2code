#include<bits/stdc++.h>
using namespace std;
struct code{
	int marks;
	int index;
	int index1;
};
int cmpfunc(struct code a,struct code b)
{
	if(a.marks > b.marks)
	{
		return 1;
	}
	return 0;
}
int cmpfunc1(struct code p,struct code q)
{
	if(p.index1 < q.index1)
	{
		return 1;
	}
	return 0;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		struct code a[n];
		for(int i=0;i<n;i++)
		{
			
			cin>>a[i].marks;
			a[i].index1 = i;
		}
		sort(a,a+n,cmpfunc);
		a[0].index = 1;
		for(int i=1;i<n;i++)
		{
			if(a[i].marks == a[i-1].marks)
			{
				a[i].index = a[i-1].index;
			}
			else
			{
				a[i].index = i+1;
			}
		}
		sort(a,a+n,cmpfunc1);
		for(int i=0;i<n;i++)
		{
			cout<<a[i].index<<" ";
		}
		cout<<"\n";
		
	}
		
}
