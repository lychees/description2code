n=int(raw_input())
for i in range(n):
	count=0
	a=int(raw_input())
	while(a>=1):
		if(a==1):
			break
		if(a%3==0):
			a=a/3
			count=count+1
		elif(a%2==0):
			a=a/2
			count=count+1
		else:
			a=a-1
			count=count+1
	print(count)
