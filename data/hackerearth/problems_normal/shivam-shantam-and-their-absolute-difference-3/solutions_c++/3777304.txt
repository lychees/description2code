#include<bits/stdc++.h>
using namespace std;
int dp[2][800001];
int a[201],b[201];
#define mod 1000000007
int n,k;
int fun(int idx,int diff)
{
	if(diff<-400000 || diff>400000)
	return 0;
	if(idx==n)
	{ if(abs(diff)<=k)return 1;
        return 0;}
	
	int &ret=dp[idx&1][diff+400000];
	if(ret!=-1)
	return ret;
	
	ret=0;
	ret+=fun(idx+1,diff);
	//if(ret>=mod)
	//ret%=mod;
	ret+=fun(idx+1,diff+a[idx]);
	//if(ret>=mod)
	ret%=mod;
	ret+=fun(idx+1,diff-b[idx]);
	//if(ret>=mod)
	ret%=mod;
	ret+=fun(idx+1,diff+a[idx]-b[idx]);
	//if(ret>=mod)
	ret%=mod;
	return ret;
}
int main()
{
	int i,j,m;
	
	scanf("%d",&n);
	
	for(i=0;i<n;i++)
	scanf("%d",&a[i]);
	
	for(i=0;i<n;i++)
	scanf("%d",&b[i]);
	
	scanf("%d",&k);
	
	for(i=n;i>=0;i--)
	{
		memset(dp[i&1],-1,sizeof(dp[i&1]));
		for(j=-400000;j<=400000;j++)
		fun(i,j);
	}
	
	printf("%d",fun(0,0));
}
