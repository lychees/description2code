#include<bits/stdc++.h>
using namespace std;
int inc = 0;
int level( int n ,int times){
   if( n <= times ){
    return n;
   }
   inc++;
   n = n - times;
   level( n , 2*times);
}
int main(){
   int n;
   cin >> n;
   int val =  level( n , 5);
    int pw = ( 1 << inc );
    string x[] = {"Sheldon","Leonard","Penny","Rajesh","Howard"};
  if(val  >= 1 && val <= pw )
  cout << x[0] << endl;
  else
    if(val  > pw && val <= 2*pw )
  cout << x[1] << endl;
  else
  if(val  > 2*pw && val <= 3*pw )
  cout << x[2] << endl;
  else
  if(val  > 3*pw && val <= 4*pw )
  cout << x[3] << endl;
  else
  cout << x[4] << endl;


}
