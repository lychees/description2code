from sys import stdin
for i in xrange(int(stdin.readline())):
    st = ''
    for j in stdin.readline():
        if j in 'abc':
            st += '2'
        elif j in 'def':
            st += '3'
        elif j in 'ghi':
            st += '4'
        elif j in 'jkl':
            st += '5'
        elif j in 'mno':
            st += '6'
        elif j in 'pqrs':
            st += '7'
        elif j in 'tuv':
            st += '8'
        elif j in 'wxyz':
            st += '9'
    print st
