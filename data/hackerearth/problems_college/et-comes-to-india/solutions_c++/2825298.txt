#include<bits/stdc++.h>
#include<iostream>
using namespace std;
#define MOD 1000000007
#define lld signed long long int
double A[1000000+5];
lld len[1000000+5];
int rec(lld K){
	int it = 0;
	for(int i=0;i<=60;++i){
		if(len[i]>=K){
			it = i;
			break;
		}
	}
	if(len[it-1]+1==K)
		return 0;
	lld dist = len[it]-K+1;
	return rec(dist)^1;
}
int solve(){
	lld K;
	lld n;
	cin>> n >> K;
	len[0]=0;
	for(int i=1;i<=64;++i){
		len[i] = 2*len[i-1]+1;
	}
	if(len[n] < K) return -1;
	else return rec(K);
}
int main()
{
	int T;
	cin>>T;

	while(T--){
		int val = solve();
		char ans;
		if(val == -1) cout << -1 << "\n";
		else
		{
			if(val == 0) ans = 'G';
			else ans = 'C';
			printf("%c\n", ans);
		}
	}
}
