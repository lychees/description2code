#include <iostream>
using namespace std;

int main()
{
  int N,M;
  cin>>N>>M;
  int C[M];
  for(int i=0;i<M;i++)
   cin>>C[i];
   
   long long int T[N+1][M];
 long long int i=0,j,x,y;
 for(i=0;i<M;i++)
  {
  	T[0][i]=1;
  }
  for(i=1;i<=N;i++)
   {
   	for(j=0;j<M;j++)
   	 {
   	 	 x = (i-C[j] >= 0)? T[i - C[j]][j]: 0;
 
         
            y = (j >= 1)? T[i][j-1]: 0;
 
            
            T[i][j] = x + y;
   	 }
   }
   cout<<T[N][M-1]<<endl;
    return 0;
}
