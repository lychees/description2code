from collections import Counter as mset
for _ in range(input()):
	astr = raw_input()
	bstr = raw_input()
	a = mset("".join(astr.split()))
	b = mset("".join(bstr.split()))
	c = a & b
	la = len(a - c)
	lb = len(b - c)
	
	if (la ==0 and lb ==0) or (la > 0 and lb > 0):
		print "You draw some."
	elif la==0 and lb > 0:
		print "You lose some."
	else:
		print "You win some."
