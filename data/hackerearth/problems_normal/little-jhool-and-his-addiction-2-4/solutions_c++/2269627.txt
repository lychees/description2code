#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll int a[1000010]; 
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int n,k;
		scanf("%lld%lld",&n,&k);
		ll int i,j,t,mymax,mymin,ans;
		for(i=0;i<n;i++)
		{
			scanf("%lld",(a+i));
		}
		sort(a,a+n);
		j=0;
		mymax=INT_MIN;
		mymin=INT_MAX;
		for(i=0;i<n/2;i++)
		{
			t=a[i]+a[n-1-i];
			j++;
			mymax=max(mymax,t);
			mymin=min(mymin,t);
		}
		if(j==1)
		{
			ans=0;
		}
		else
		{
			ans=mymax-mymin;
		}
		printf("%lld\n",ans);
		if(ans==k)
		{
			printf("Lucky chap!\n");
		}
		else if(ans<k)
		{
			printf("Chick magnet Jhool!\n");
		}
		else
		{
			printf("No more girlfriends!\n");
		}
	}
	return 0;
}
