#include <bits/stdc++.h>
using namespace std;
int dp[10][1<<10];
int freq[10][26] = {0};
int n,k;
int mod = 1000000007;
string s;
int solve(int pos,int mask)
{

    if(pos == k)
        return 1;
    if(dp[pos][mask] != -1)
        return dp[pos][mask];
    int ans = 0;
    for(int i = 0;i< n;i++)
    {
        if(!(mask&(1<<i)))
        {
            int c = 0;

                if(freq[i][int(s[pos])-97] > 0)
                    c+=freq[i][int(s[pos])-97];

            ans+=(c*solve(pos+1,mask|(1<<i)))%mod;
            ans%=mod;


        }
    }
    return dp[pos][mask] = ans;

}
int main()
{
    memset(dp,-1,sizeof(dp));

    cin>>n>>k;
    char c;

    for(int i = 0;i< n;i++)
    {
        for(int j = 0;j< 6;j++)
        {
            cin>>c;
            freq[i][int(c)-97]++;
        }
    }
    cin>>s;
    cout<<solve(0,0)<<endl;
}
