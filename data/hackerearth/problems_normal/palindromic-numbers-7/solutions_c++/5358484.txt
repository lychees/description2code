#include <iostream>
using namespace std;

int main()
{
    int t;
    long long int num,rev=0,num1,rem,num2,i,count=0;
    cin>>t;
    while(t>=1){
    cin>>num>>num1;
    count=0;
    for(i=num;i<=num1;++i){
    	num2=i;
    	rev=0;
    while(num2>0){
    	rem=num2%10;
    	rev=10*rev+rem;
    	num2=num2/10;
    	}
    //	cout<<rev<<endl;
    	if(i==rev){
    		count+=1;
    	}
    	
    	
    }
    	cout<<count<<endl;
    	--t;
    }
    return 0;
}
