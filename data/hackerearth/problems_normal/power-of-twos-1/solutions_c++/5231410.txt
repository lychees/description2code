#include<bits/stdc++.h>
using namespace std;

long long int dp[1000001],dp1[1000001];
bool visit[1000001];

void solve1()
{
   memset(visit,0,sizeof(visit));
   memset(dp1,0,sizeof(dp1));
   
   dp1[0] = 1;
   
   for(int i = 2;i <= 1000000; i++)
   {
   	  // if(!visit[i])
   	   for(int j = i;j <= 1000000;j += i)
   	   {
   	   	
   	       visit[j] = 1;
   	       dp1[j]++;
		  	
	   }
	   
   }
   	
}

void solve()
{
	
	memset(dp,0,sizeof(dp));
	solve1();
	
	dp[1] = 0;
	dp[2] = 1;
	
	for(int i = 3 ; i <= 1000000 ; i++)
	{
		
	    dp[i] = dp1[i] + dp[i-1] ;
	
	}
	
}

main()
{
	
	int t;
	 
	scanf("%d",&t);
	
	solve();
	
	while(t--)
	{
		
		int n;
		 
		scanf("%d",&n);
		
		printf("%lld\n",dp[n]);
	}
}
