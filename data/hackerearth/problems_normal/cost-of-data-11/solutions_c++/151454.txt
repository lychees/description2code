struct tnode
{
    tnode* child[30];
};
tnode Trie;

int updateTrie(char word[203])
{
    int c = 0;
    tnode* ptr = &Trie;
    tnode* tmp;
    for(int i = 0; word[i] != '\0'; i ++)
    {
	tmp = ptr->child[word[i]-'a'];
	if(tmp == NULL)
	{
	    tmp = (tnode*) malloc (sizeof(tnode));
	    for(int k = 0; k < 30; k ++)
		tmp->child[k] = NULL;
	    ptr->child[word[i]-'a'] = tmp;
	    c ++;
	}
	ptr = tmp;
    }
    return c;
}
char str[200];
int main()
{
    for(int i = 0; i < 26; i ++)
	Trie.child[i] = NULL;
    int n;
    scanf("%d", &n);
    int ans = 1;
    while(n--)
    {
	scanf("%s", str);
	ans += updateTrie(str);
    }
    cout << ans << endl;
    return 0;
}
