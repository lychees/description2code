#include <iostream>
using namespace std;

int main()
{
    long long int n, q, i, j, ans, sum[1000005], k, a, b;
    
    cin>>n;
    
    sum[0] = 0;
    for (i = 1; i <= n; i++) {
    	cin>>k;
    	sum[i] = sum[i - 1] + k;
    }
    
    cin>>q;
    
    while(q--) {
    	cin>>a>>b;
    	ans = sum[b] - sum[a - 1];
    	cout<<ans<<endl;
    }
    return 0;
}
