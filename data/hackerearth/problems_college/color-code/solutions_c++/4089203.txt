int powr (int a, int b)
{
    if (b == 0)
        return 1;
    int x = powr(a, b/2);
    if (b % 2 == 0)
        return (x*x);
    else
        return (((x*x))*a);
}

void tohex(int x) {
    string hex = "0123456789UVWXYZ";
    int i;
    vector <int> v;
    while(x > 0) {
        v.pb(x%16);
        x/=16;
    }
    while(v.size() < 2) v.pb(0);
    reverse(v.begin(),v.end());
    for(i = 0 ; i < v.size(); i++) {
        cout<<hex[v[i]];
    }
}

int main()
