#include<iostream>
#include<string>
#include<vector>
#include <iterator>
#include <algorithm>
using namespace std;

int main(){
    string s;
    string vec ("aeiou");
    int n;
    cin>>n;
    cin>>s;
    string::iterator it;
    for(it=s.begin();it<s.end();it++)
        if(*it=='a'||*it=='e'||*it=='i'||*it=='o'||*it=='u')
            vec.erase(std::remove(vec.begin(), vec.end(), *it), vec.end());
    if(vec.empty())
        cout<<"YES";
    else
        cout<<"NO";
    return 0;

}
