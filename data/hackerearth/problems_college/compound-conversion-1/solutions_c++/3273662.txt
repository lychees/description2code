#include<bits/stdc++.h>
using namespace std;
int main()
{
	int p,q,cas,i,j,k,m,n,t,l1,l2,l,count,ans,count1,count2;
	char str1[105],str2[105];
	cin>>n;
	cas=1;
	while(n--)
	{
		p=0;
		q=0;
		scanf(" %s",str1);
		scanf(" %s",str2);
		l=strlen(str1);
		count=0;
		for(i=0;i<l;i++)
		{
			if(str1[i]=='?' && str2[i]=='0')
			p++;
			else if(str1[i]=='?' && str2[i]=='1')
			q++;
	     
		}
		count1=0;
		count2=0;
		for(i=0;i<l;i++)
		{
			if(str1[i]=='0' && str2[i]=='1')
			count1++;
			if(str1[i]=='1' && str2[i]=='0')
			count2++;
		}
		if(count1==count2)
		count=p+q+count1;
		else if(count2>count1)
		{
			if((count2-count1)>q)
			count=-1;
			else
			count=p+q+count2;
		}
		else
		count=p+q+count1;
		printf("Case %d: %d\n",cas,count);
		cas++;
	}
}
