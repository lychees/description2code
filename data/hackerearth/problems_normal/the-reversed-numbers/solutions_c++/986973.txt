#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
int main()
{
 ios_base::sync_with_stdio(false);
 cin.tie(false);
 int t;
 string a_t,b_t,final;
 cin>>t;
 while(t--)
    {
    	cin >>a_t >>b_t;
    	reverse(begin(a_t),end(a_t));
    	reverse(begin(b_t),end(b_t));
        final = to_string(stoull(a_t) + stoull(b_t));
        reverse(begin(final),end(final));
        cout<<stoull(final)<<"\n";
    }
}
