for _ in xrange(input()):
    num = input()

    if num == 1:
        print -1
    elif num % 7 == 0:
        print num/7
    else:
        rem = num % 7
        if rem == 2 or rem == 3 or rem == 5:
            print num/7 + 1
        elif rem == 4 or rem == 6:
            print num/7 + 2
        else:
            print num/7 + 1
