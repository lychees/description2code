#include <iostream>
#include <math.h>
using namespace std;

int main()
{
	bool primes[4000000];
    long long int n,i,sum=0;
    cin>>n;
	for(int o=2;o<n;o++)
	{
   primes[o]=true;
	}
	for(int m=2;m*m<n;m++)
	{	
	if(primes[m]==true)
 	for(int j=m*m;j<n;j+=m)
        primes[j]=false;
	}
    for(i=2;i<n;i++)
    if(primes[i]==true)
    sum+=i;
    cout<<sum;
    return 0;
}
