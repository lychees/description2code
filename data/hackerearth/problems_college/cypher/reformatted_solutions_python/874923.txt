import string
t = input()
while t:
    t = t-1
    ip = raw_input()
    l = len(ip)
    lst = list(string.ascii_lowercase)
    l2 = len(lst)
    newlist = []
    for i in range(l):
        for j in range(l2):
            if ip[i] == lst[j]:
                a = (j+i) % 26
                newlist.append(lst[a])
    re = ""
    for x in newlist:
        re = re+x
    print re
