#include <iostream>
#include <vector>
#include <algorithm>
#include <vector>

using namespace std;

int main(){

	int n,min,total,res;
	cin>>n;
	vector<int> strength(n);

	for(int i=0;i<n;i++)
		cin>>strength[i];
	cin>>min;

	sort(strength.begin(),strength.end());

	total=0;
	res=0;
	for(int i=n-1;i>=0;i--){
		if(total<min){total+=strength[i];res++;}
		else
			break;
	}
	cout<<res;

	return 0;
}
