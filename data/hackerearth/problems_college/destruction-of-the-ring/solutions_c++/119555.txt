#define MAX 2000004
#define MOD 1000000007
#define ll  long long int
 
 
///cout<<(double)(clock() - tStart)/CLOCKS_PER_SEC<<endl;
///clock_t tStart = clock();
 
using namespace std;
ll fact[MAX],ifact[MAX];
 
 
ll power(ll n,int m)
{
   if(m==0) return 1;
   ll x=power(n,m/2);
   if(m%2==0)
        return (x*x)%MOD;
   else
        return (((x*x)%MOD)*n)%MOD;
}
 
void preProcess()
{
  fact[0]=1;
  ifact[0]=1;
  int i;
  for(i=1;i<MAX;i++)
  {
      fact[i]=fact[i-1]*i;
      fact[i]%=MOD;
  }
  ifact[2000000] = power(fact[2000000],MOD-2);
  for(i=2000000;i>1;i--)
     ifact[i-1] = ifact[i]*i%MOD;
}
 
ll comb(int a,int b)
{
return (((fact[a]*ifact[b])%MOD)*ifact[a-b])%MOD;
}
 
 
int main()
{
    preProcess();
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int m,n;
        scanf("%d%d",&n,&m);
        n--;
        m--;
        ll ans=0;
        ans=comb(n+m,m);
       // printf("%lld\n",ans);
        ans%=MOD;
        printf("%lld\n",ans);
     }
return 0;
}
