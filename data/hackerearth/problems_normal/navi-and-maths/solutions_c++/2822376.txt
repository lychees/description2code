#include <iostream>
#include<stdio.h>
#define ll long long int
using namespace std;
ll mod=1e9+7;
ll power(ll a,ll b,ll c){
	ll res=1,x=a;
	while(b>0){
		if(b%2){
			res=(res*x)%c;
		}
		x=(x*x)%c;
		b/=2;
	}
	return res%c;
}
int main()
{
	int t,n;
	scanf("%d",&t);
	int ct=0;
	while(t--){
	scanf("%d",&n);
		ll a[20];
		ll ans=0;
		for(int i=0;i<n;i++) scanf("%lld",&a[i]);
		for(int i=0;i<(1<<n);i++){
				int cnt=0;
			for(int j=0;j<n;j++){
				if(i & (1<<j)){
					cnt++;
				}
			}
			if(cnt<2) continue;
			ll num=1,sum=0;
			for(int j=0;j<n;j++){
				if(i & (1<<j)){
					num*=(a[j]);
					if(num>=mod) num%=mod;
					sum+=a[j];
				}
			}
			ll term=(num*power(sum,mod-2LL,mod))%mod;
			//cout<<num<<" "<<sum<<" "<<term<<endl;
			ans=max(ans,term);
		}
		printf("Case #%d: ",++ct);
		printf("%lld\n",ans);
	}
	return 0;
}
