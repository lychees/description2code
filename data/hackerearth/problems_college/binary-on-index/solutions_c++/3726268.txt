#include <iostream>
using namespace std;

int main()
{
    int i,j,n,m;
    
    cin>>n;
    
    int a[n];
    for(i=0;i<n;++i)
    cin>>a[i];
    
    cin>>m;
    int b[m];
    
    for(j=0;j<m;j++)
    cin>>b[j];
    
    for(j=0;j<m;j++)
    {int lb=0,ub=n-1,mid,pos=-1;
    
    
    while(lb<=ub)
    {
		mid=(lb+ub)/2;
    	 
    	 if(a[mid]==b[j])
     		{pos=mid;
     		break;}
    	else if(a[mid]>b[j])
    			ub=mid-1;
    	else if(a[mid]<b[j])
    			lb=mid+1;
    	
    }
    cout<<pos<<endl;
    
    }
    return 0;
}
