#include<bits/stdc++.h>
using namespace std;

long long int mod_pow(long long int a, long long int b,long long int mod)
{
	long long int ans=1;
	while(b)
	{
		int r=b%2;
	
		if(r!=0)
		ans=(ans*a)%mod;
		
		a=(a*a)%mod;
		
		b/=2;
	}	
	return ans;
}

long long gcdf(long long a,long long b)
{
	if(a==0)
	return b;
	return gcdf(b%a,a);	
}

long long min1(long long a,long long b)
{
	return a>b?b:a; 	
}

long long max1(long long a,long long b)
{
	return a<b?b:a; 	
}

long long int fact(long long int n)
{
	long long int ans=n,mod=1000000007;
	for(int i=1;i<=n-1;i++)
	{
		ans=(ans*i)%mod;
	}
	
	return ans;
}

struct node{
	string s;
	int val;
};

bool comp(struct node less,struct node more)
{
	return less.val<more.val;	
}

int main() {
int n,m;
    cin>>n>>m;
    
    int a[m];
    
    for(int i=0;i<m;i++)
      cin>>a[i];
      
      long long int ar[m+1][n+1];
      for(int i=0;i<=n;i++){
        ar[m][i]=0;
      }
      
      for(int i=0;i<=m;i++){
        ar[i][0]=1;
      }
      
      for(int i=m-1;i>=0;i--){
        for(int j=1;j<=n;j++){
          if(j>=a[i])
            ar[i][j]=ar[i][j-a[i]]+ar[i+1][j];
            else
              ar[i][j]=ar[i+1][j];
            }
          }
          
          cout<<ar[0][n]<<"\n";
          return 0;
}
