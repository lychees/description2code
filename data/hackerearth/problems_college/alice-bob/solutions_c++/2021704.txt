#include <bits/stdc++.h>

using namespace std;

int gcd(int u, int v) {
	return (v != 0)?gcd(v, u%v):u;
}

int main(){
	int n, a[205];
	bool flag = false;

	cin >> n;
	for(int i=0; i<n; i++){	cin >> a[i]; if(a[i]%2)	flag = true; }

	if (!flag) for(int i=0; i<n; i++)	a[i] /= 2;

	int max = a[0];
	for(int i=0; i<n; i++)	if(a[i] > max)	max = a[i];
	
	int play_turns = max - n;

	if(play_turns%2)	cout << "Alice\n";
	else	cout << "Bob\n";
}
