#include<iostream>
#include<stack>
#include<cstring>
#include<vector>
#include<algorithm>
using namespace std;
int arr[1000001][2];
void print(int n)
{
    cout<<n<<' ';
    if(arr[n][0]!=0)
        print(arr[n][0]);
    if(arr[n][1]!=0)
        print(arr[n][1]);
}
int main()
{
    int t;
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin>>t;
    int n,x,y,z;
    while(t--)
    {
        cin>>n;
        for(int i=0;i<n;i++)
        {
            cin>>x>>y>>z;
            arr[x][0]=y;
            arr[x][1]=z;
        }
        print(1);
        cout<<endl;
        memset(arr,0,sizeof arr);
    }

}
