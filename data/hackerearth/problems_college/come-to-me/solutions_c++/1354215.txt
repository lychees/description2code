#include <iostream>
#include <set>
#include <map>
#include <string>
#include <cstring>
using namespace std;

int main()
{
	int ts,cnt,mx,v[100001];
	string st;
    cin>>ts;
    while(ts--)
    {
    	cin>>st;
    	
    	mx=0;
    	for(int i=0;i<st.size();i++)
    	{
    		cnt=0;
    		map<char ,int>mp;
    		for(int j=i;j<st.size();j++)
    		{
    			if(mp[st[j]-'A']==1)
    			{
    				//cnt=0;
    				break;
    			}
    			cnt++;
    			mx=max(cnt,mx);
    			mp[st[j]-'A']=1;
    		}
    	}
    	cout<<mx<<endl;
    }
    return 0;
}
