    i1 = raw_input()
    i2 = raw_input()
    if i1 in i2 or i2 in i1:
        print "YES"
    else:
        flag = False
        for j in i2:
            if j in i1:
                flag = True
                break
        if flag:
            print "YES"
        else:
            print "NO"
