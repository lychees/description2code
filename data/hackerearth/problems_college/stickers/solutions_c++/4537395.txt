#include <iostream>

using namespace std;

int main()
{
    
    // get the number of test cases T
    int T;
    cin >> T;
    
    // process all the T test cases
    while(T > 0) {
    	
    	int num_aud, tot_num=0;;
    	cin >> num_aud;
    	
    	for(int sticker_num = num_aud; sticker_num >= 1; sticker_num--)
    	{
        	tot_num += sticker_num;
    	}
    	
    	cout << tot_num << endl;

    	T--;
    }
    
    
    return 0;
}
