    #include<bits/stdc++.h>
    using namespace std;
    #define ll long long int
    #define sc1ll(x) scanf("%lld",&x)
    vector<ll> arr(10001,0);
    int main()
    {
        ll test,a,b,i,j,minn,maxx,num;
        sc1ll(num);
        for(i=0;i<num;i++)
            sc1ll(arr[i]);
        sort(arr.begin(),arr.begin()+num);
        minn=num;
        for(ll i=0;i<num;i++)
        {
            for(ll j=i+1;j<num;j++)
            {
                if((arr[j]-3*arr[i])<=0)
                    minn=min(minn,num-(j-i+1));
            }
        }
        cout<<minn;
    }
