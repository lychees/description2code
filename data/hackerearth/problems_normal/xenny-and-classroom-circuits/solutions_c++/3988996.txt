#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,m;
	scanf("%d",&n);
	while(n--){
		string given,boy="",girl="";
		cin>>m>>given;
		m*=2;
		for(int i=0;i<m;++i){
			if(i%2==0){
				boy.push_back('B');
				girl.push_back('G');
			}
			else{
				boy.push_back('G');
				girl.push_back('B');
			}
		}
		int ham1=0,ham2=0;
		for(int i=0;i<m;++i){
			if(boy[i]!=given[i]){
				ham1++;
			}
			if(girl[i]!=given[i]){
				ham2++;
			}
		}
		int ans=min(ham1,ham2);
		printf("%d\n",ans/2);
		
	}
	return 0;
}
