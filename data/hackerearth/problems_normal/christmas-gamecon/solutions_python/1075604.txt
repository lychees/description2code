for i in range(int(raw_input())):
	n = int(raw_input())
	lst = map(int,raw_input().split())
	lst =  map(lambda x: x%5, lst)
	answer = 0
	
	for j in range(n):
		answer ^= lst[j]
		
	if(answer): print "Aniruddha"
	else: print "Andrew"
