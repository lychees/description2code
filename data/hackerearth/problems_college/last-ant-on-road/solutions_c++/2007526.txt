using namespace std;
typedef pair<ll,ll> ii;
typedef pair<ii,ii> iii;
typedef vector<ll> vv;
ll arr[1005];
int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		ll n,fl=-2,ans;
		cin>>n;
		for(ll i=0;i<n;i++)
			cin>>arr[i];
		for(ll i=0;i<n;i++)
		{
			fl=-2;
			for(ll j=0;j<n-1;j++)
			{
				if(fl!=j-1&&arr[j]==1&&arr[j+1]==-1)
					arr[j]=-1,arr[j+1]=1,fl=j;
			}
			if(fl==-2)
			{
				if(arr[0]==1)
				{
					ans=0;
					fl=-3;
				}
				if(arr[n-1]==-1)
				{
					ans=n-1;
					fl==-3;
				}
				for(ll j=0;j<n-1;j++)
				{
					if(arr[j]!=arr[j+1])
					{
						fl=-3;
						ans=j;
						break;

					}
				}
			}
			if(fl==-3)
				break;
		}
		cout<<ans+1<<"\n";
	}
}
