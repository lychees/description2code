#include<iostream>
#include<cstdio>
#include<cstring>
int main()
{
	int t,length,i;
	char str[100];
	scanf("%d",&t);
	while(t-- >0)
	{
		scanf("%s",&str);
		length = strlen(str);
		int num[length];
		num[0]=str[0]-97;
		if(num[0]>13)
		num[0]=num[0]-26;
		printf("%d ",num[0]);
		for(i=1;i<length;i++)
		{
			num[i]=str[i]-str[i-1];
			if(num[i]< -12)
			num[i]=26+num[i];
			if(num[i]>13)
			num[i]=num[i]-26;
			printf("%d ",num[i]);
		}
		printf("\n");
	}
}
