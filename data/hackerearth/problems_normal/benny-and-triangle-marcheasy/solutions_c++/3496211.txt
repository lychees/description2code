#include<iostream>
#include<algorithm>
#include<set>
#include<string>

using namespace std;

double a, b, p;

double getArea(double b, double a){
	return b * a / 2.0;
}

double getAns(){
	double l = 0.0, r = b + 1e-18, area = getArea(b, a), maxiArea = getArea(b, a) * p / 100.0;
	for(int i=0;i<4141;++i){
		double m = (l + r) / 2.0;
		if(getArea(m, m * (a / b)) > maxiArea){
			r = m;
		}else{
			l = m;
		}
	}
	return b - l;
}

int main(){
#ifdef _DEBUG
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
#endif

	cin >> a >> b >> p;
	printf("%.2lf\n",getAns());

	return 0;
}
