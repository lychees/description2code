#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll mod =1000000007;
ll fact[2000001];
ll pwr(ll x,ll y)
{
	ll result=1;
	while(y>0){
		if(y&1)result=(result*x)%mod;
		x=(x*x)%mod;
		y=y>>1;
	}
	return result;
}
void pre()
{
	fact[0]=1;fact[1]=1;
	for(ll i=2;i<2000001;i++){
		fact[i]=(i*fact[i-1])%mod;
	}
}
ll catalan(ll n)
{
		return ((((fact[2*n]*pwr(fact[n],mod-2))%mod)*pwr(fact[n+1],mod-2))%mod)%mod;
}
int main() {
	// your code goes here
	pre();
	int t;
	ll n;
	scanf("%d",&t);
	while(t--){
		scanf("%llu",&n);
		ll ans =catalan(n);
		printf("%llu\n",ans);
	}
	return 0;
}
