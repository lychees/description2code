#include<bits/stdc++.h>
using namespace std;
int main(){
	int n;
	scanf("%d",&n);
	long long max1,max2,min1,min2;
	max1=max2=-INT_MAX;
	min1=min2=INT_MAX;
	long long x;
	for(int i=0;i<n;i++){
		scanf("%lld",&x);
		if(max1<x){
			max2=max1;
			max1=x;
		} else
		if(max2<x){
			max2=x;
		}
		if(min1>x){
			min2=min1;
			min1=x;
		} else
		if(min2>x){
			min2=x;
		}
	}
	printf("%lld\n",max(min1*min2,max1*max2));
	return 0;
}
