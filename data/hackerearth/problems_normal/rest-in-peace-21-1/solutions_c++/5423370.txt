#include <iostream>
using namespace std;

int has21(int num){
	int val = 0;
	while(num/10!=0){
		int rem = num%10;
		num = num / 10;
		if(rem == 1){
			rem = num%10;
			if(rem==2){
				val = 1;
				break;
			}
		}
	}
	return val;
}

int main()
{
    int t; cin >> t;
    
    while(t--){
    	int num ; cin >> num;
    	
    	if(num%21==0)
    		cout << "The streak is broken!" << endl;
    	else if(has21(num))
    		cout << "The streak is broken!" << endl;
    	else
    		cout << "The streak lives still in our heart!" << endl;
    	
    }
    return 0;
}
