#include<bits/stdc++.h>
using namespace std;
unsigned long int k=0;
const unsigned long int m=1000001;
map<unsigned long int, unsigned long int>mp;
unsigned long int p=1;
unsigned long int kk=0;

void cal(){
for(unsigned long int i=2;i<=m;i++)
{
          unsigned long int mx=p^i;
          p=i;
           mx=max(mx,i^(unsigned long int)1);
           kk=max(kk,mx);
           mp[i]=kk;

}

}
unsigned int cal(unsigned int n)
{
  unsigned count = 0;

  /* First n in the below condition is for the case where n is 0*/
  if (n && !(n&(n-1)))
    return n;

  while( n != 0)
  {
    n  >>= 1;
    count += 1;
  }

  return 1<<count;
}


int main()
{
    int t;
    cin>>t;
    while(t--)
    {
           unsigned  long int n;
           cin>>n;
         if(n==1)
         {
              cout<<0<<endl;
         }
         else{
                  if((n&(n-1))!=0)
                  {
                         unsigned long int k=cal(n);
                          int p=k*2-1;
                          cout<<p/2<<endl;
                  }
                  else
                  {
                       cout<<n*2-1<<endl;
                  }
         }
    }

}
