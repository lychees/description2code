#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
#include <math.h>
using namespace std;

#define MAX 1000001
int main()
{
    int t,sieve[MAX];
    long long int  x,y,dif,i,j,ans,start;
    for(cin>>t;t;t--)
    {
        cin>>x>>y;
        for(i=0;i<=y-x;i++)
        	sieve[i]=0;
        for (i = 2; i * i <= y; i++) 
        {
            dif = i * i;
            for(j = x ;  j % dif != 0 ; j+=(dif-j%dif))
            {}
            start = j;
            //start = x + dif - ( x % dif);
            //cout<<"S "<<start<<"\n";
            for (j = start; j <= y; j += dif)
                sieve[j - x] = 1;
        }
        ans=0;
        for(i=0;i<=y-x;i++)
        	if(sieve[i]==0)
        	{
        		//cout<<"i "<<x+i<<"\n";
        		ans++;
        	}
        cout<<ans<<"\n";
    }
    return 0;
}
