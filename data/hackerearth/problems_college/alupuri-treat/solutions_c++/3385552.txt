#include <iostream>
#include<cstdlib>
#include<cmath>
#define ll long long
using namespace std;

int main()
{
    ll m;
    cin>>m;
    ll a[m][m];
    ll s1=0,s2=0;
    for(ll i=0;i<m;i++)
    {
    	for(ll j=0;j<m;j++)
    	{
    		cin>>a[i][j];
    		if(i==j)
    		{
    			s1+=a[i][j];
    		}

    			if(i+j==m-1)
    			{
    				s2+=a[i][j];
    			}
    		
    	}
    
    }
    cout<<abs(s1-s2);
    return 0;
}
