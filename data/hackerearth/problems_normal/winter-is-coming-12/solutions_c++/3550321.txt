#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	
	while(t--)
	{
		int n,l,r,x;
		cin >> n;
		
		long long *A = new long long[n];
		long long *I = new long long[n];
		
		for(int i=0;i<n;i++)
		{
			cin >> A[i];
			I[i] = -1;
		}
		
		int min = n+1;
		long long sum = 0;
		
		for(int i=0;i<n;i++)
		{
			sum += A[i];
			sum %= n;

			if(I[sum]==-1 && sum==0)
				x=i+1;
			else if(I[sum]==-1 && sum!=0)
				x=n+1;
			else
				x=i-I[sum];
			
			if(x<min)
			{
				min = x;
				l=I[sum]+1;
				r=i;
			}
			
			I[sum] = i;
		}
		
		cout << l+1 << " " << r+1 << endl;
	}
}
