#include <iostream>

using namespace std;

int main()
{
    int T,status;
    long int N,f,t,i;
    cin >> T;
    while (T--)
    {
        cin >> N;
        status=0;
        f=(N/3)*3;
        t=N-f;
        while (f>=0&&t<=N)
        {
            if (t%5==0)
            {
                for (i=0; i<f; ++i)
                    cout << 5;
                for (i=0; i<t; ++i)
                    cout << 3;
                cout << endl;
                status=1;
                break;
            }
            else
            {
                f-=3;
                t+=3;
            }
        }
        if (!status)
                cout << "-1" << endl;
    }
    return 0;
}
