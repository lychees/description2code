t = int(raw_input())

for _ in range(t):
    text = raw_input()
    m = int(raw_input())
    missed = raw_input().split()
    weight = 0
    for i in range(len(text)):
        if text[i] in missed:
            continue
        else:
            weight += ord(text[i]) - 96
    print(weight)
