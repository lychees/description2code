#include <iostream>
#include <cstring>
using namespace std;
struct obj{
	long long value;
	int index;
};
int main (){
	ios::sync_with_stdio(false);
	long long tc,n,k,ans,pos,temp,sum[10010];
	obj arr[10010];
	cin>>tc;
	while(tc--){
		memset(arr,0,sizeof(arr));
		memset(sum,0,sizeof(sum));
		ans=pos=0;
		cin>>n>>k;
		for(int i=1;i<=n;i++){
			cin>>temp;
			if(temp>0){
				arr[++pos].value=temp;
				arr[pos].index=i;
				sum[pos]=arr[pos].value;	
			}
		}
		for(int i=1;i<=pos;i++){
			for(int j=i-1;j>0;j--){
				if(arr[j].index<=arr[i].index-k-1){
					sum[i]=max(sum[j]+arr[i].value,sum[i]);
				}
			}
			ans=max(ans,sum[i]);
		}
		cout<<ans<<endl;
	}
	return 0;
}
