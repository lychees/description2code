#include<bits/stdc++.h>
using namespace std;
#define debug1(x,y) cerr<<#x<<" "<<x<<#y<<" "<<y<<endl;
vector<int> A,B;
int main()
{
    int n,l,r,q,x,z;
    scanf("%d %d %d %d",&n,&l,&r,&q);
    for(int i=0;i<n;i++)
    {
        scanf("%d",&z);
        A.push_back(z);
        B.push_back(z);
    }
    sort(A.begin()+l,A.begin()+r+1);
    while(q--)
    {  vector<int> :: iterator it;
        scanf("%d",&x);
        it=upper_bound(A.begin()+l,A.begin()+r+1,x);
        if(*it==A[r+1])
            cout<<-1<<endl;
        else
        cout<<*it<<endl;
    }
    return 0;
}
