#include<iostream>
using namespace std;
main(){
	int t;
	cin >> t;
	while(t--){
  	int a[50][50],i,j,n;
  	cin >> n;
  	for(i=0;i<n;i++){
    	for(j=0;j<n;j++)
     	cin >> a[i][j];
  	}
	i=0;j=0;
  	while(i!=n-1 || j!=n-1){
    if(a[i+1][j]==1)
    i++;
    else if(a[i][j+1]==1)
    j++;
    else{
      if(a[i-1][j]==1){
      	a[i][j]=0;
      	i--;
    }
    else if(a[i][j-1]==1){
      a[i][j]=0;
      j--;
    }
    else
    break;
    }
  	}
    if(i==n-1 &j==n-1)
    cout <<"POSSIBLE"<<endl;
    else
    cout <<"NOT POSSIBLE"<<endl;
}
}
