ll power(ll a,ll b)
{
	if(!b)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2);
	temp=(temp*temp);
	if(b&1)
		temp=(temp*a);
	return temp;
}
int L[100][100];
int palin(string& s,int n)
{
	int i,j,l,ans=1;
	for(i=0;i<n;i++)
		L[i][i]=1;
	for (l=2; l<=n; l++)
    {
        for (i=0; i<n-l+1; i++)
        {
            j = i+l-1;
            if (s[i] == s[j] && l == 2)
               L[i][j] = 2;
            else if (s[i] == s[j])
               L[i][j] = L[i+1][j-1] + 2;
            else
               L[i][j] = max(L[i][j-1], L[i+1][j]);
               ans=max(ans,L[i][j]);
        }
    }
    return ans;
}
int main()
    {
    	int i,t,n,ans;
    	sc("%d",&t);
    	string s;
    	while(t--)
    	{
    		cin>>s;
    		n=s.length();
    		ans=1;
    		string k;
    		for(i=n-2;i>=0;i--)
    		{
    			k=s;
    		
    			reverse(k.begin()+i,k.end());
    				//cout<<k<<"\n";
    			ans=max(ans,palin(k,n));
    		}
    		pf("%d\n",ans);
    	}
    return 0;
}
