#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define lli long long int
lli f[10005];
lli arr[10000005];
vector <lli> v;
void sieve()
{
	lli i,j,aa,b,c,w,p,q,r;
	for(i=2;i<=1e7;i++)
	{
		if(arr[i]==0)
		{
			v.push_back(i);
			for(j=i*i;j<=1e7;j+=i)
			{
				arr[j]++;
			}
		}
	}
	j=1;
	for(i=3;i<10000;i+=2,j++)
	{
		aa=4*j*j-2*j+1;
		b=4*j*j+1;
		c=4*j*j+2*j+1;
		if(aa>=1e7)
		{
			for(w=2;w*w<=aa;w++)
			{
				if(aa%w==0)
				{
					p=0;
					break;
				}
			}
			if(w*w>aa)
			{
				p=1;
			}
		}
		else
		{
			p=!arr[aa];
		}
		if(b>=1e7)
		{
			for(w=2;w*w<=b;w++)
			{
				if(b%w==0)
				{
					q=0;
					break;
				}
			}
			if(w*w>b)
			{
				q=1;
			}
		}
		else
		{
			q=!arr[b];
		}
		if(c>=1e7)
		{
			for(w=2;w*w<=c;w++)
			{
				if(c%w==0)
				{
					r=0;
					break;
				}
			}
			if(w*w>c)
			{
				r=1;
			}
		}
		else
		{
			r=!arr[c];
		}
		f[i]+=f[i-2]+p+q+r;
	}
}
int main()
{
    sieve();
    lli t,n,x;
    double ans;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld",&n);
 		x=2*n-1;
    	ans=((100.0*f[n])/(x*1.0));
    	printf("%.6lf\n",ans);
    }
    return 0;
}
