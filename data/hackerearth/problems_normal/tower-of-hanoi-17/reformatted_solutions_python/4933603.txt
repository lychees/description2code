t = int(raw_input())
for i in xrange(t):
    a = []
    n = int(raw_input())
    for i in xrange(n):
        arr = map(int, raw_input().split())
        a.append([arr[0], arr[1]])
    a = sorted(a)
    h = [0 for i in range(n)]
    h[0] = a[0][1]
    for i in range(1, n):
        h[i] = a[i][1]
        for j in range(i):
            if a[i][0] > a[j][0] and a[i][1] > a[j][1]:
                h[i] = max(h[i], h[j] + a[i][1])
    print max(h)
