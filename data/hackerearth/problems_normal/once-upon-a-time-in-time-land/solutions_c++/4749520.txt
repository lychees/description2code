#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    scanf("%d",&t);
   while(t--)
   {
   	 int n,k;
    scanf("%d %d",&n,&k);
    ll ar[n+1];
    for(int i=1;i<=n;i++)
    scanf("%lld",&ar[i]);
    
    ll dp[n+1];
    memset(dp,0,sizeof(dp));
    
    
    for(int i=1;i<=k;i++)
    {
    	dp[i]=max(ar[i],dp[i-1]);
    }
    
    for(int i=k+1;i<=n;i++)
    {
    	dp[i]=max(dp[i-1],dp[i-k-1]+ar[i]);
    	
    	
    }
    printf("%lld\n",dp[n]);
   }
    return 0;
}
