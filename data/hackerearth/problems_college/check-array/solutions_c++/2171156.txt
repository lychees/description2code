#include <iostream>
using namespace std;

int main() {
	int a[100005]={0},n,i,j,b[100005]={0},t,c[100005];
	cin>>t;
	while(t--)
	{int flag=0;
		cin>>n;
		for(i=0;i<n;i++)
		cin>>a[i];
		if(n!=1)
		{
		b[0]=a[0];
		c[n-1]=a[n-1];
		for(i=1,j=n-2;i<n;i++,j--)
		{
			b[i]=b[i-1]+a[i];
			c[j]=c[j+1]+a[j];
		}
		for(i=1;i<n-1;i++)
		if(b[i-1]==c[i+1])
		{
			flag=1;
			break;
		}
		}
		if(flag==1 || n==1)
		cout<<"YES\n";
		else
		cout<<"NO\n";
		
		
		
		
		
	}
	
	return 0;
}
