#include <iostream>
#include<unordered_map>
#include<algorithm>
using namespace std;

int main()
{
    unordered_map<int,int> M;
    unordered_map<int,int>::iterator it;
    
    int n,m;
    cin>>n;
    
    int res[105];
   
    for(int i=0;i<n;i++)
     {
     	int v;
     	cin>>v;
     	it=M.find(v);
     	if(it==M.end())
     	 {
     	 	M[v]=1;
     	 }
     	else
     	 {
     	 	int val=M[v];
     	 	val++;
     	 	M[v]=val;
     	 }
     }
     cin>>m;
     int k=0;
     int ar[m];
      for(int i=0;i<m;i++)
         cin>>ar[i];
     for(int i=0;i<m;i++)
      {
      	it=M.find(ar[i]);
      	if(it==M.end())
      	 res[k++]=ar[i];
      	else
      	 {
      	 	int val=M[ar[i]];
      	 	if(val<=0)
      	 	 res[k++]=ar[i];
      	 	else
      	 	 {
      	 	 	int z=M[ar[i]];
      	 	 	z--;
      	 	 	M[ar[i]]=z;
      	 	 }
      	 }
      }
    sort(res,res+k);
   
    int prev=-999;
    for(int i=0;i<k;i++)
     {
     	
     	if(res[i]!=prev)
     	  {
     	  	cout<<res[i]<<" ";
     	  	prev=res[i];
     	  }
     	
     	 
     	
     }
}
