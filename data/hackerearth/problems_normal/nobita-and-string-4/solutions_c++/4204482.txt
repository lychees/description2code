#include <bits/stdc++.h>
using namespace std;

int main()
{
   	map<int,string>mp;
   	int t,i;
   	scanf("%d ",&t);
   	while(t--){
   		int count=0;
   		char ch[100000];
   		map<int,string> mp;
   		gets(ch);
   		for(i=0;ch[i]!='\0';i++)
   		{
   			if(ch[i]==' ')
   				count++;
   			else
   				mp[count]+=ch[i];
   		}
   		for(i=0;i<=count/2;i++)
   		{
   			string v;
   			v=mp[i];
   			mp[i]=mp[count-i];
   			mp[count-i]=v;
   		}
   		for(i=0;i<=count;i++)
   		{
   			cout<<mp[i]<<' ';
   		}
   		printf("\n");
   	}
    return 0;
}
