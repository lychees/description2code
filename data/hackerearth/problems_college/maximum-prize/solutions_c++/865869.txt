#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main (){ 
	ll t;
	cin>>t;
	while(t--){ 
		ll n,k;
		cin>>n>>k;
		vector<ll>ar;
		for(ll i=0;i<n;i++)
		{ 
		ll y;cin>>y;ar.push_back(y);
		}
		
		
		sort(ar.begin(),ar.end(),greater<ll>());
		ll sum=0,i=0;
		
		if(k==0)
		cout<<"0\n";
		else{
		for(int i=0;i<min(n,k);i++)
		{ 
			sum+=ar[i];
		}
	
		cout<<sum<<"\n";
		}
	}return 0;
}
