#include<iostream>
#include<cmath>

using namespace std;

int main()
{
    double in,n;
	int T;
	cin>>T;
	for(int i=0;i<T;i++)
	{
    cin>>in;
    n=sqrt(in);
    n=round(n);
    
    n=n*n;
    int diff;
    diff=n-in;
    if(diff==0)
        cout<<"YES"<<endl;
    else if(diff<0)
    {
    	 cout<<diff<<endl;
    }else{
    	cout<<"+"<<diff<<endl;
    }
    	
  
    	}
   	}
