#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int testcases,print;
    cin>>testcases;
    cin.ignore();
    
    vector<int> a(26);
    string b;
    for(int i=0;i<testcases;i++)
    {
    	print=1;
    	getline(cin,b);
    	for(int j=0;j<b.size();j++)
    	{	
    		//cout<<b[j]<<endl;
    		a[b[j]-'a']++;
    	}
  
    	for(int j=0;j<26;j++)
    	{
    		if(a[j]%2==1)
    		{
    			print=-1;
    			break;
    		}
    	}
    	cout<<print<<endl;
    	for(int j=0;j<26;j++)
    	{
    		a[j]=0;
    	}
    }
    return 0;
}
