LL chk(char* a,char *b){
    LL rt =0;
    int sl=2*n;
    //change pk1 to pk2
    int p1=0,p2=0;
    int lstg=0,lstb=0;
    while(p1<sl && p2<sl){
        while(p1<sl && p2<sl && a[p1]==b[p2]){
            if(a[p1]=='G')lstg=p1;else lstb=p1;
            ++p1,++p2;
        }
        if(p1>=sl || p2>=sl)break;
        //now a[p1]!=b[p2]; // swap
        int t1;
        if(b[p2]=='G')t1=lstg+1; else t1=lstb+1;
        while(t1<sl && (a[t1]!=b[p2] || a[t1]==b[t1]))++t1;
        //printf("%s %s\n",a,b),printf("p1:%d p2:%d t1:%d\n",p1,p2,t1);
        if(t1>=sl)break;
        rt += 1;
        swap(a[p1],a[t1]);
        ++p1,++p2;
        //must have by condition
    }
    return rt;
}
