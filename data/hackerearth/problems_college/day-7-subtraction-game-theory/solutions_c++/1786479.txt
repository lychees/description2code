#include<bits/stdc++.h>
using namespace std;
long long int ar[1000000];

int main()
{

  int p, q;
  long int t;
  scanf("%d%d%ld",&p,&q,&t);
  ar[0]=p-q;
  
  for(int i=1;i<1000000;i++)
  {
   ar[i]=((ar[i-1]%1000000007)*(p%1000000007))%1000000007-q;
  } 
   while(t--)
    {long int n;
   scanf("%ld",&n);
   	printf("%lld\n",ar[n-1]);
   }
}
