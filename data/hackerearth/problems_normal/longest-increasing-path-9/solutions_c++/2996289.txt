#include<bits/stdc++.h>
#define ll long long
using namespace std;

int dp[105][105], A[303][305];
int t, n, m;

int solve(int i, int j){
	if(i>n || j>m) return 0;
	if(dp[i][j]!=-1)
		return dp[i][j];
	dp[i][j]=0;
	int f=0, g=0;
	if(i<n){
		if(A[i+1][j]>A[i][j]){
			dp[i][j]=solve(i+1, j);
			f=1;
		}
		
	}
	if(j<m){
		if(A[i][j+1]>A[i][j]){
			dp[i][j]=max(dp[i][j], solve(i, j+1));
			g=1;
		}
		
	}
	
	return dp[i][j]=dp[i][j]+1;
}

int main(){
	
	scanf("%d", &t);
	while(t--){
		scanf("%d %d", &n, &m);
		for(int i=1; i<=n; i++){
			for(int j=1; j<=m; j++){
				scanf("%d", &A[i][j]);
			}
		}
		memset(dp, -1, sizeof(dp));
		int ans=solve(1,1);
		
		printf("%d\n", ans);
	}
}
