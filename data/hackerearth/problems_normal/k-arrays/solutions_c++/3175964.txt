#include<bits/stdc++.h>
using namespace std;

long long mod = 1000000007;

int main()
{
	long long mod_inv[17];
	mod_inv[1] = 1;
	for(int i=2;i<=16;i++)
	{
		mod_inv[i]=(-(mod/i)*mod_inv[mod%i]) % mod + mod;
	}
	long long t,n,k,ans,len,val,max,max_len,min,min_len;
	cin>>t;
	while(t--)
	{
		max = INT_MIN;
		min = INT_MAX;
		max_len = INT_MAX;
		min_len = INT_MIN;
		cin>>n>>k;
		long long v[k][n];
		for(int i=0;i<k;i++)
			for(int j=0;j<n;j++)
				cin>>v[i][j];

		vector<long long> sum;
		for(int i=0;i<n;i++)
		{
			ans = 0;
			for(int j=0;j<k;j++)
			{
				ans+=v[j][i];
				ans%=mod;
			}
			sum.push_back(ans);
		}

		for(int i=1;i<pow(2,n);i++)
		{
			val = 1;
			len = 0;
			for(int j=0;j<n;j++)
				if(i&1<<j)
					{
						len++;
						val*=sum[j];
						val%=mod;
					}
			val*=mod_inv[len];
			val%=mod;
			if(val>=max)
			{
				if(val>max)
				{
					max = val;
					max_len = len;
				}
				if(val == max && len<max_len)
				{
					max_len = len;
				}
			}

			if(val<=min)
			{
				if(val<min)
				{
					min = val;
					min_len = len;
				}
				if(val == min && len>min_len)
				{
					max_len = len;
				}
			}
		}

		cout<<(max^min)<<" "<<(max_len^min_len)<<"\n";

	}
}
