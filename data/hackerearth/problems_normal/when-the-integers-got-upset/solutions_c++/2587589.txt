int dp[1<<12][13][13];

int isset(int i,int j)
{
	if( (i | (1<<j)) == i)
		return 1;
	else
		return 0;
}
int main()
{
	int T;

	SI(T);
	
	int p[13],a[13];
	while(T--)
	{
		REP(i,4096)

				}

		int n;

		SI(n);

		REP(i,n)
			SI(a[i]);

			SI(p[i]);

		REP(i,1<<n)
		{
			REP(j,n)

				{
					if( isset(i,j) == 0 || isset(i,k) == 0 || (j==k))

						}

			}
		}

		int mi = INT_MAX;
		REP(i,n)
