#include <iostream>
#include <string>
#include <stdlib.h>
#include <vector>
using namespace std;

string func(string s)
{
	string s1,str;
    
	int i=0,k=0;
	char j = s[i];
	
    while(s[i]!='\0')
    {
    	if(j<=s[i])
        j = s[i];
    	i++;
    }
    i=0;
    while(s[i]!='\0')
    {
    	if(s[i]==j)
    	{
    		k=i;
    		break;
    	}
    	i++;
    }
    //cout<<k<<" "<<s.length()<<" "<<j<<endl;
    if(s.empty())
    return s;
    else if(k==s.length()-1){
    str+=j;
    }
    else
    {
        for(i=k+1;i<s.length();i++)
        {
        	if(s[i]!=j)
        	s1.push_back(s[i]);
        }
        str=j;
        str+=func(s1);
    }
    return str;
}
