#include <bits/stdc++.h>
using namespace std;
map <string,int> mp;
map <string,int> :: iterator it;
int main()
{
    int n;
    string s;
    int ans=0;
    cin>>n;
    while(n--)
    {
    	cin>>s;
    	mp[s]++;
    }
    for(it=mp.begin();it!=mp.end();it++)
    {
    	ans=max(ans,it->second);
    }
    int sum=0;
    for(it=mp.begin();it!=mp.end();it++)
    {
    	if(it->second==ans)
    	{
    		sum+=ans;
    	}
    }
    cout<<sum<<endl;
    return 0;
}
