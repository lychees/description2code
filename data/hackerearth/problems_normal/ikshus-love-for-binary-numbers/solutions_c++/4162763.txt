#include <iostream>
#include <array>
#include <algorithm>
using namespace std;

using ull = unsigned long long;

ull gcd(ull a, ull b) {
	while (b > 0) {
		ull m = a % b;
		a = b;
		b = m;
	}
	return a;
}

void cancel(ull& a, ull& b) {
	ull d = gcd(a, b);
	a /= d;
	b /= d;
}
