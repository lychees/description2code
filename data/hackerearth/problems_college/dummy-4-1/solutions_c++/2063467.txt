#include<bits/stdc++.h>
using namespace std;

#define lli long long
const int lmt = 105;
int grid[lmt][lmt];
bool vis[lmt][lmt];
int n;

int dx[]={0,1,0,-1};
int dy[]={1,0,-1,0};

lli solve(int r,int c){
    if(vis[r][c] || grid[r][c]==1) return 0;
    if(r==n-1 && c==n-1) return 1LL;
    vis[r][c] = true;
    lli ans = 0;
    int nr,nc;
    for(int i=0;i<4;i++){
        nr = r+dx[i];
        nc = c+dy[i];
        if(nr>=0 && nc>=0 && nr<n && nc<n)
            ans += solve(nr,nc);
    }
    vis[r][c]=false;
    return ans;
}

int main() {
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++)
            scanf("%d",&grid[i][j]);
    }
    lli ans = solve(0,0);
    printf("%lld\n",ans);
    return 0;
}
