#define MOD 1000003

string s;
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int T;
	int i,j,n;
	long long re=0;
	cin>>T;
	while(T--)
	{
		cin>>s;
		n=s.size();
		re=0;
		for(i=0;i<n;i++)
		{
			for(j=0;i-j>=0&&i+j<n;j++)
			{
				if(s[i-j]==s[i+j])
					re+=(2*j+1)*(2*j+1);
				else
					break;
			}
			for(j=0;i-j>=0&&i+j+1<n;j++)
			{
				if(s[i-j]==s[i+j+1])
					re+=(2*j+2)*(2*j+2);
				else
					break;
			}
		}
		cout<<re<<'\n';
	}
//	system("pause");
	return 0;
}
