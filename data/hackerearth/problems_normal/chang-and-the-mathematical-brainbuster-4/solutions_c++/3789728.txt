#include <iostream>
#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
    ll n,a[10003],ans=0;
    cin>>n;
    for(ll i=0;i<n;i++){
        cin>>a[i];
    }
    sort(a,a+n);
    for(ll i=0;i<n;i++){
        ll len=0;
        for(ll j=n-1;j>=i;j--){
            if(a[j]-3*a[i]<=0){
                len=j-i+1;
                break;
            }
        }
        ans=max(ans,len);
    }
    cout<<n-ans<<endl;
    return 0;
}
