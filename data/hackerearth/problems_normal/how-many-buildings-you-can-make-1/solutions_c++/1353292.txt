#include <iostream>
using namespace std;

int main()
{
    int t = 0 ;
    cin>>t;
    int s = 0 , count = 0 ;
    for(int i = 0 ; i < t ; i++)
    {
    	cin>>s;
    	count = 0 ;
    	for(int j = 1 ; j <= s ; j++)
    	{
    		for(int k = j ; k <= s ; k++)
    		{
    			if( k * j <= s)
    			count++;
    			else
    			break;

    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
