#include <bits/stdc++.h>
using namespace std;
float rev(float t){
	int i,j,k,l;
     vector<int>p;
      i = t;
     while(1){
     	if(i == 0)
     	break;
	int 	z= i%10;
		
		p.push_back(z);
		i = i/10;
	}
	int s = p.size();
	int n = 0;
	s--;
	i= 0;
	while(i!=(p.size())){
		n = n + p[i] * (int)pow(10,s);
		i++;
		s--;
	}
	return n;
}
int main()
{
  int  n ;
  cin>>n; 
  float p,e;
  while(n--){
  	cin>>p>>e;
  	float  t = ((p + p*(e/100.00)));
  //	cout<<t<<endl;
  	float r = rev(t);
  	//cout<<r<<endl;
  	float q = p;
  	float w = r;
  	float x = (q - w)/q;
  	//cout<<x<<endl;
  	float g=  t;
  	int  ans = t + (x*g);
  	cout<<ans<<endl;
  }
  
  return 0;
}
