#include <iostream>
#include<string>
#include<unordered_map>
#include<map>
#include<vector>
using namespace std;
vector<string>v;
map<string,int>mor;
struct u
{
	string a;
};
u t;
int n,h,k,x;
void brute(u t)
{
	if(mor[t.a])
	{
		return ;
	}
	mor[t.a]++;
	string a,b;
	a=t.a;
	b=t.a;
	int x=h;
	while(x--)
	{
		a.insert(a.begin(),a.back());
		a.pop_back();
	}
	for(int i=0;i<b.length();i++)
	if(i&1)
	b[i]=(b[i]-'0'+k)%10+'0';
	t.a=a;
	brute(t);
	t.a=b;
	brute(t);
}
int main()
{
	int t;

	{
		v.clear();
		
		string a,b;
		u tr;
		cin>>a>>k>>h;
		tr.a=a;
		brute(tr);

}
