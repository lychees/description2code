template <typename T, typename U> 
inline void chkmax(T& x, U y) {
	if (x < y) x = y;
}

template <typename T, typename U>
inline void chkmin(T& x, U y) {
	if (y < x) x = y;
}
const int MAXN = 500005;
int l[MAXN], a[MAXN], g[MAXN], n;

int main() {
	
	int ncase;
	for (scanf("%d", &ncase); ncase--;) {
		scanf("%d", &n);
		for (int i = 0; i < n; i++) scanf("%d", a + i);
		g[n] = n, l[n] = n;
		for (int i = n - 1; i >= 0; i--) {
			int nxt = i + 1;
			while (nxt < n && a[nxt] <= a[i]) nxt = g[nxt];
			g[i] = nxt;
			nxt = i + 1;
			while (nxt < n && a[nxt] >= a[i]) nxt = l[nxt];
			l[i] = nxt;
		}
		int ans = 0;
		for (int i = 0; i < n; i++) {
			int r = g[i];
			chkmin(r, l[l[i]]);
			chkmax(ans, r - i);
			r = l[i];
			chkmin(r, g[g[i]]);
			chkmax(ans, r - i);
		}
		printf("%d\n", ans);
	}
	return 0;
}
