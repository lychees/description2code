#include <iostream>
using namespace std;

#define ll long long
#define mod 1000000007
int main()
{
	ll int t;
    cin >> t;
    ll int dp[1000000+5];
    dp[0]=1;
    dp[1]=1;
    for(int i=2;i<1000000+1;i++){
    	dp[i]=((i-1)*(dp[i-2]));
    	dp[i]=((dp[i])+(dp[i-1]));
    	dp[i]%=mod;
    }
    while(t--){
    	ll int n;
    	cin >> n;
    	cout << dp[n] << endl;
    }
   // cout << "Hello World!" << endl;
    return 0;
}
