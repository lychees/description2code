import math
t = int(raw_input())
i = 1
while (t):
    a, b = map(int, raw_input().split())
    s = ""
    k = len(str(b))
    n = 1
    count = 0
    while (str(b) != s):
        y = n*math.log(a, 10)
        m = y-int(y)
        s = str(pow(10, m)*pow(10, k-1))[0:k]

        count += 1
        n += 1

    print "Case %d:" % i, count
    i += 1
    t -= 1
