    #include <iostream>
    using namespace std;
    bool isPali(long int n)
    {
    	long int rev=0;
    	long int num=n;
    	int digit;
    	while(n!=0)
    	{
    		digit=n%10;
    		rev=rev*10+digit;
    		n=n/10;
    	}
    	if(rev==num)
    		return true;
    	else
    		return false;
    }
     
    int main()
    {
        int t;
        long int a, b, cnt, i;
        cin>>t;
        
        while(t--)
        {
        	cnt=0;
        	cin>>a>>b;
        	for(i=a; i<=b; i++)
        	{
        		if(isPali(i))
        			cnt++;
        	}
        	cout<<cnt<<endl;
        }
        return 0;
    }
