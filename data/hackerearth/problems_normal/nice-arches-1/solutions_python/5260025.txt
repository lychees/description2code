#ABBA
def isBubbly(word):
    stack = []
    for elem in word:
    	if len(stack) > 0:
    		top = stack.pop()
    		if(top == elem):
    			continue
    		else:
    			stack.append(top)
    			stack.append(elem)
    	else:
    		stack.append(elem)
    if len(stack) > 0:
    	return False
    else:
    	return True

        word = raw_input()
