#include <iostream>
using namespace std;

int t , n , temp , i;
int dp[150];

int main()
{
    cin>>t;
    
  	for(i = 0 ; i < 120 ; i++)
  		dp[i] = -1;
  		
  	dp[10] = 1;
  	dp[12] = 1;
  	
  	for(i = 20 ; i < 120 ; i++)
  	{
  		if(dp[i-10]==-1 && dp[i-12]==-1)
  			continue;
  		else if(dp[i-10]!=-1 && dp[i-12]!=-1)	
  			dp[i] = 1 + min(dp[i-10] , dp[i-12]);
  		else
  			dp[i] = 1 + max(dp[i-10] , dp[i-12]);
  	}

   while(t--)
    {
    	cin>>n;
    	cout<<dp[n]<<endl;
    }
    return 0;
}
