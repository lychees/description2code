import sys


def main():

	t  = int(raw_input())

	for i in xrange(t):
		r,xc,yc = map(int,raw_input().split())

		flag = True
		for j in xrange(4):
			x,y = map(int,raw_input().split())

			if ((xc-x)**2)+((yc-y)**2)-(r**2) > 0:
				flag = False
		if flag:
			print "Yes"
		else:
			print "No"

main()
