#include <bits/stdc++.h>
using namespace std;
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int n;
   	cin>>n;
   	vector<int>vec;
   	vector<int>::iterator i;
   	for(int it=1;it<=n;it++)
   	{
   		int x;
   		cin>>x;
   		vec.push_back(x);
   	}
   	if(!is_sorted(vec.begin(),vec.end()))
   	prev_permutation(vec.begin(),vec.end());
   	for(i=vec.begin();i!=vec.end();i++)
   	cout<<*i<<" ";
   	cout<<endl;
   }
    return 0;
}
