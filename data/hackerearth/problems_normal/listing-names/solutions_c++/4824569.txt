#include <iostream>

int compare(char *s,char *v)
{
//	cout<<*s<<" ffsdaff   "<<*v<<"       ";
 while(*s && *s==*v)
  {
 	
  	s++,v++;
  }
//  cout<<*s<<" ffff   "<<*v<<"\n";
  
 return (*s-*v);
}
