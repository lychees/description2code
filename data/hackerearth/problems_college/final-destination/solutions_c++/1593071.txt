#include <iostream>
#include <queue>

using namespace std;

int main(){
    int tC, a, b, m, d, p, s, temp, ans;
    cin >> tC;
    while(tC--){
        cin >> a >> b >> m >> d >> p >> s;
        int stepsTo [1000009] = {0};
        queue <int> q;
        stepsTo[a] = 1;
        q.push(a);
        ans = 0;
        int u;
        while(!q.empty()){
            u = q.front();
            q.pop();
            if(u == b){
                ans = stepsTo[u];
                break;
            }
            temp = u * m;
            if(temp >= 2 && temp  <= 1000000 && !stepsTo[temp]){
                stepsTo[temp] = stepsTo[u] + 1;
                q.push(temp);
            }
            temp = u / d;
            if(temp >= 2 && temp  <= 1000000 && !stepsTo[temp]){
                stepsTo[temp] = stepsTo[u] + 1;
                q.push(temp);
            }
            temp = u + p;
            if(temp >= 2 && temp  <= 1000000 && !stepsTo[temp]){
                stepsTo[temp] = stepsTo[u] + 1;
                q.push(temp);
            }
            temp = u - s;
            if(temp >= 2 && temp  <= 1000000 && !stepsTo[temp]){
                stepsTo[temp] = stepsTo[u] + 1;
                q.push(temp);
            }
        }
        cout << ans - 1 << endl;
    }
}
