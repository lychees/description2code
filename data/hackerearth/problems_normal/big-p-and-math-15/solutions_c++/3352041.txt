#include <iostream>
#include<string.h>
#include<math.h>

using namespace std;

int main()
{
	long int a,b;
	cin >> a >> b;
	int cnta=0,cntb=0;
	long int max=0,min=0;
	while(a!=0 || b!=0)
	{
		int rema,remb;
		rema=a%10;
		remb=b%10;
		if(rema==5 && remb==5)
		{
			max=max+(6*pow(10,cnta))+(6*pow(10,cntb));
		}
		else if(rema==5)
		{
			max=max+(6*pow(10,cnta))+(remb*pow(10,cntb));
		}
		else if(remb==5)
		{
			max=max+(rema*pow(10,cnta))+(6*pow(10,cntb));
		}
		else
		{
			max=max+(rema*pow(10,cnta))+(remb*pow(10,cntb));
		}
		if(rema==6 && remb==6)
		{
			min=min+(5*pow(10,cnta))+(5*pow(10,cntb));
		}
		else if(rema==6)
		{
			min=min+(5*pow(10,cnta))+(remb*pow(10,cntb));
		}
		else if(remb==6)
		{
			min=min+(rema*pow(10,cnta))+(5*pow(10,cntb));
		}
		else
		{
			min=min+(rema*pow(10,cnta))+(remb*pow(10,cntb));
		}
		cnta++;
		cntb++;
		a=a/10;
		b=b/10;
	}
	cout << min <<" "<< max << endl;
    return 0;
}
