#include <bits/stdc++.h>
using namespace std;

int main()
{
	int i,n,temp;
	vector<int> arr;
	vector<int> v;
	vector<int> w;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>temp;
    	arr.push_back(temp);
    }
    sort(arr.begin(),arr.end());
    int start=0,end=arr.size()-1,flag=0;
    while(start<=end)
    {
    	if(flag==0)
    	{
    		v.push_back(arr[start]);
    		start++;
    		flag=1;
    	}
    	else
    	{
    		v.push_back(arr[end]);
    		end--;
    		flag=0;
    	}
    	
    }
    start=0,end=arr.size()-1,flag=1;
    while(start<=end)
    {
    	if(flag==0)
    	{
    		w.push_back(arr[start]);
    		start++;
    		flag=1;
    	}
    	else
    	{
    		w.push_back(arr[end]);
    		end--;
    		flag=0;
    	}
    	
    }
    int sum1=0,sum2=0;
    for(i=0;i<v.size()-1;i++)
    {
    	sum1=sum1+abs(v[i]-v[i+1]);
    	sum2=sum2+abs(w[i]-w[i+1]);
    }	
    cout<<max(sum1,sum2);	
    return 0;
}
