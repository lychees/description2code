#include <iostream>
#include <vector>

using namespace std;

int N, K;
int numSets;
vector <int> pset;

void initSet()
{
    pset.assign(N+1, 0);
    for (int i = 0; i <= N; i++)
        pset[i] = i;
    numSets = N;
}

int findSet(int i)
{
    return (pset[i] == i) ? (i) : (pset[i] = findSet(pset[i]));
}

void unionSet(int i, int j)
{
    if (findSet(i) != findSet(j))
    {
        if (i < j)
            pset[i] = j;
        else
            pset[j] = i;

        --numSets;
    }
}

int main()
{
    cin >> N >> K;

    initSet();

    for (int i = 0; i < K; i++)
    {
        int u, v;
        cin >> u >> v;
        unionSet(u, v);
    }

    cout << numSets << endl;



    return 0;
}
