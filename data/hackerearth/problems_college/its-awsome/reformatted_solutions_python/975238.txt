def awesome(t):
    for g in range(len(t)):
        if (t[g] != '4' and t[g] != '7'):
            return False
    return True


x = int(input(""))
for g in range(x):
    y = raw_input("")
    t = 0
    for g in range(len(y)):
        if (y[g] == '4' or y[g] == '7'):
            t += 1
    if (awesome(str(t))):
        print "YES"
    else:
        print "NO"
