#include <iostream>
#include <vector>
#include <fstream>
#include <string>
#include <math.h>
#include <cstdlib>
#include <stdio.h>     
#include <stdlib.h> 

using namespace std;

main()
{
	
	char x;
	int n;
	while(cin>>x>>n)
	{
	//	cout<<x<<" "<<n<<endl;
		if (n<2)
			cout<<"INVALID"<<endl;
		else if (x+n-1>='A'&&x+n-1<='Z'||x+n-1<='z'&&x+n-1>='a')
		{
			for (int i=0;i<n;i++)
			{
				for (int j=0;j<=i;j++)
				{
					cout<<x<<" ";
					
				}
				x++;
				cout<<endl;
			}
		
		}
		else cout<<"INVALID"<<endl;
		
		cout<<endl;
		
	}
	
	return 0;
	

}
