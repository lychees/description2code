#include <iostream>
#include<bits/stdc++.h>
using namespace std;
bool ispalin(char *str,int n)
{
	int i;
	for(i=0;i<n;i++)
	{
		if(str[i]!=str[n-i-1])
			return false;
	}
	return true;
}
int main()
{
    int t;
    char str[101];
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	if(ispalin(str,strlen(str)))
    	{
    		cout<<(int)str[strlen(str)/2]<<endl;
    	}
    	else
    		cout<<"-1"<<endl;
    }
    return 0;
}
