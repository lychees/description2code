#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    unsigned long long int a[n],i,s=0,l,m=0,k=0;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    	s+=a[i];
    }
    while(s>0)
    {
    	k+=s%10;
    	s=s/10;
    }
    k=k%9;
    for(l=0;l<n;l++)
    {
      while(a[l]>0)
    {
    	m+=a[l]%10;
    	a[l]=a[l]/10;
    }	
    }
    m=m%9;
    cout<<0;
}
