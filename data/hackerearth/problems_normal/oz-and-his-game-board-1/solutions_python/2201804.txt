diff=0
def ABS(a):
	if(a<0):return -a
	return a
	
def MIN(one,two):
	if(one < two):		
		return one
	return two
	
test=input()
for t in range(0,test):
	N,M,A,B,C=map(int,raw_input().split())
	first=N*A + M*B
	second= MIN(N,M)
	if(N>M):
		diff=A *ABS(N-M)
	else: diff=B * ABS(N-M)
	second=2*C*second + diff
	if(first>second):print first
	else :print second
