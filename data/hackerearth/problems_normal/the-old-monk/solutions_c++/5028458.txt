#include <iostream>
using namespace std;

int oldmonk(long long int b[],long long ele,int i,int p,int q){
	//cout<<p<<" "<<q<<" "<<b[p]<<endl;
	if(p==q)return(p-i);
	int mid=(p+q)/2;
	//cout<<b[mid]<<endl;
	if(b[mid]>=ele){
		if(b[mid+1]<ele)
		return mid-i;
		else
		return oldmonk(b,ele,i,mid+1,q);
	}
	return oldmonk(b,ele,i,p,mid);
	
}
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		long long int a[n],b[n];
		for(int i =0;i<n;i++)
		{
			cin>>a[i];
		}
		for(int i=0;i<n;i++)
		cin>>b[i];
		int monk=0,m;
		for(int i=0;i<n;i++){
			m=oldmonk(b,a[i],i,0,n-1);
			//cout<<"*"<<m<<endl;
			if(monk<m)
			monk=m;
		}
		cout<<monk<<endl;
	}
    return 0;
}
