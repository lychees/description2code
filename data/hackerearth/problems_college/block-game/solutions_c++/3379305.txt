#include <iostream>
using namespace std;

long long int f(long long int *h,long long int *cost,long long int k,long long int n,long long int tempans)
{
	if(k==n)
		return tempans;
	long long int x=0;
	for(long long int i=0;i<n;i++)
	{
		x+=(abs(h[k]-h[i])*cost[i]);
	}
	if(x<tempans)
		tempans=x;
	f(h,cost,k+1,n,tempans);
}

int main()
{
    long long int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	long long int h[n],cost[n];
    	for(long long int i=0;i<n;i++)
    	{
    		cin>>h[i];
    	}
    	for(long long int i=0;i<n;i++)
    		cin>>cost[i];
    	long long int ans=f(h,cost,0,n,1000000000000000000);
    	cout<<ans<<endl;
    }
    return 0;
}
