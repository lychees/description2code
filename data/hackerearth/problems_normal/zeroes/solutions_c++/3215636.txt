#include<iostream>
#include<math.h>
using namespace std;
int fac[100005]={0};
int fact(int n)
{
    int x=sqrt(n)+1,i;
    for(i=2;i<x;i++)
    {
        while(n%i==0)
        {
            n/=i;
            fac[i]++;
        }
    }
    if(n!=1)
        fac[n]++;
}
int main()
{
    long long n;
    cin>>n;
    long long a[n],s=1,i;
    for(i=0;i<n;i++)
    {
        cin>>a[i];
        fact(a[i]);
    }
   // for(i=0;i<1000;i++)
     //   cout<<fac[i]<<" ";
    int q;

    {
        long long k,minn=100120;
        cin>>k;
        long long x=sqrt(k)+1,c;
    for(i=2;i<=x;i++)
    {
        c=0;
        if(k%i==0)
        {
            while(k%i==0)
            {
                k/=i;
                c++;
            }
            minn=min(minn,fac[i]/c);
        }
       // cout<<minn<<" ";
    }

    }

}
