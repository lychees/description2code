#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    
    while(t--)
	{
		long long n,st,en;
		cin>>n;map<int,int> mp;
		for(int i=0;i<n;i++)
		{
			cin>>st>>en;
			if(mp.find(st)==mp.end())
			 mp[st]=1;
			 else
			  mp[st]=mp[st]+1;
			  
			if(mp.find(en)==mp.end())
			  mp[en]=-1;
			  else
			  mp[en]=mp[en]-1;
			
			
		}
		long long mx=INT_MIN;
		long long c=0;
		map<int,int>::iterator it=mp.begin();
		while(it!=mp.end())
		{
			c=c+(*it).second;
			mx=max(mx,c);
			it++;
		}
		
		cout<<mx<<endl;
		
	}
	
	return 0;
}
