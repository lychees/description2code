#include<cstring>
#include<cstdio>
#include<cmath>
#include<stack>

#include<vector>
using namespace std;
int main()
{
	int t,i;
	cin>>t;
	for(i=1;i<=t;i++)
	{
		int r,c,w,ans;
		scanf("%d%d%d",&r,&c,&w);
    	ans = r * c/w + w;
    	if(c%w==0)
    	ans--;
    	printf("Case #%d: %d\n",i,ans);
	}
}
