#include <bits/stdc++.h>
using namespace std;
bool set1(int n)
{
	int count=0;
	while(n){
	n = n & (n-1);
	count++;
	}
	return count&1;
}

int main()
{
    int t; cin >> t;
    while(t--)
    {
    	long long c,a,b; cin >> a >> b;
    	c = a ^ b;
    	cout << (!set1(c)?"YES":"NO") << endl;
    }
    return 0;
}
