#include<bits/stdc++.h>
using namespace std;
#define gc getchar
#define ll long long
#define v vector
#define pr pair<int,int>
#define sd second
#define ft first
#define pb push_back
ll  fr()
{
    char ch;
    ll t=0,flag=0;
    ch=gc();
    while(ch<33)
        ch=gc();
    if(ch=='-')
        {
            ch=gc();
            flag=1;
        }
    while(ch>33)
    {
        t=t*10+ch-'0';
        ch=gc();
    }
    if(flag)
        return -t;
    else
        return t;
}
ll compute(ll n) {
if(n==0) return 0;
ll k = (n+1)/2;
return k*k + compute(n/2);
}
int main()
{
    ll  i,j,k,l,m,n,t,ans,en;
    t=fr();
    while(t--)
    {
        n=fr();
        ans=compute(n);
        printf("%lld\n",ans);
    }
    return 0;
}
