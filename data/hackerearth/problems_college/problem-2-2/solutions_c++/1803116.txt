#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
	ll red,white;
	cin>>red>>white;
	ll max,min;
	max=std::max(red,white);
	min=std::min(red,white);
	if(red==0&&white!=0){
		cout<<white<<endl;
	}
	else if(white==0&&red!=0){
		cout<<red<<endl;
	}
	else{
		min++;
		if(max%min==0){
			cout<<(max/min)<<endl;
		}
		else{
			cout<<(max/min)+1<<endl;
		}
	}
}
