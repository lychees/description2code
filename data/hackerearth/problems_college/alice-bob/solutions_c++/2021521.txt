#include <iostream>
using namespace std;
#define ll long long
ll gcd(ll a,ll b)
{
	if(b==0)return a;
	else return gcd(b,a%b);
}
int main()
{
	ll n,a,H,Mx,a1,LO,i;
    cin>>n;
    cin>>a;
    H=a;
    Mx=a;
    for(i=1;i<n;i++)
    {
    	cin>>a1;
    	H=gcd(a1,H);
    	Mx=max(Mx,a1);
    }
    LO=(Mx/H)-n;
    if(LO%2)cout<<"Alice\n";//the left over number is odd, alice makes last move
    else cout<<"Bob\n";//the left over number is even, bob makes last move
    //cout << "Hello World!" << endl;
    return 0;
}
