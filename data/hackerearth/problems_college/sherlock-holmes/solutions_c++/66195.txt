#include<iostream>
using namespace std;
int n,m;
double g[501][501],val[501][501];
int a[501];
double mod(double x,double y)
{
    x-=y;
    if(x<0) return -x;
    return x;
}
double value(int i,int j)
{
    if(i==j || i+1 == j) return 0;
    if(i>j) return 0;
    double x,y,d,ans=0;
    x=a[i];y=a[j];
    d=(y-x)/(j-i);
    ans=0;
    for(int k=1;k+i<j;k++)
    ans+=mod(a[i+k],a[i]+d*k);    
    return ans;
}
double find(int last,int j)
{
    if(j==0)
    {
        double ans=val[last][n-1];
        return ans;
    }
    else 
    {
        double ans=g[last][j-1];
        for(int i=last+1;i<=n-2;i++)
        {
            ans=min(ans,val[last][i]+g[i][j-1]);    
        }    
        return ans;
    }
}
int main()
{
    int test;
    cin>>test;
    while(test--)
    {
    scanf("%d%d",&n,&m);

    for(int i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }

    for(int j=0;j<=m;j++)
    g[n-1][j]=0;
    for(int i=0;i<=n-1;i++)
    for(int j=0;j<=n-1;j++) val[i][j]=0;
    for(int i=n-2;i>=0;i--)
    {
        for(int j=i+1;j<=n-1;j++)
        {
            val[i][j]=value(i,j);
        }
    }   

    for(int i=n-2;i>=0;i--)
    {
        for(int j=0;j<=m;j++)
        {
        g[i][j]=find(i,j);
        }
    }
    double ans=find(0,m-2);
    printf("%.4lf\n",(ans/(n)));
    }
    return 0;
}
