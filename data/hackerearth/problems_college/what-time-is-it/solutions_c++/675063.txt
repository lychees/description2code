#include <iostream>
using namespace std;
#include<math.h>
#include<stdio.h>
float min(float a, float b)
{
	if(a<b)
	 return a;
	return b; 
}

int main()
{
 float h,m,ah,am,ans;
 char ch;
 scanf("%f%c%f",&h,&ch,&m);
 ah=0.5*(h*60+m);
 am=6*m;
 if(ah>am)
  ans=ah-am;
 else
  ans=am-ah;
 ans=min(ans,360-ans);
 printf("%f",ans);
 return 0;
}
