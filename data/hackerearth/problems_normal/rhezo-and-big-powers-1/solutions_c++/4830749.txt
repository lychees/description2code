#include <bits/stdc++.h>
using namespace std;
#define pb push_back
#define ll long long int
#define INF 10000000000000000
#define mod 1000000007
#define phi 1000000006
ll modular(ll a,ll b,ll c)      //(a^b)%c
{
	ll ans=1;
	while(b!=0)
	{
		if(b%2==1)
		{
			ans*=a;
			ans%=c;
		}
		a=a*a;
		a%=c;
		b/=2;
	}
	return ans;
}
int main()
{
	ll a;
	string b;
	cin>>a;
	cin>>b;
	ll len=b.length();
	ll ans=0;
	for(int i=0;i<len;i++)
	{
		 ans=(ans*10+(ll)(b[i]-'0'))%phi;
	}
	ll an=modular(a,ans,mod);
	cout<<an<<endl;
}
