#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int carry=0,count=0;
    	char a[100],b[100];
    	cin>>a>>b;
    	int k=strlen(a);
    	int k1=strlen(b);
    	if(k1<k)
    	k=k1;
    	for(int i=0;i<k;i++)
    	{
    		int m=a[i]-'0';
    		int n=b[i]-'0';
    		int sum=m+n+carry;
    		if(sum>9)
    		{
    			count++;
    			carry=sum%10;
    		}
    		
    	}
    if(count==0)
    cout<<"No carry operation"<<"\n";
     else if(count==1)
    cout<<count<<" carry operation"<<"\n";
    else
    cout<<count<<" carry operations"<<"\n";
    }
}
