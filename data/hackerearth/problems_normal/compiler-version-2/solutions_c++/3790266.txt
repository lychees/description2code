#include <bits/stdc++.h>
using namespace std;

int main()
{
	string s;
	while(getline(cin,s))
	{string v="";bool c=false;
		for(int i=0;i<s.size();i++)
		{
			if((i<s.size()-1)&&s[i]=='/'&&s[i+1]=='/')
				c=true;
			if((i<s.size()-1)&&(s[i]=='-'&&s[i+1]=='>')&&!c)
			{
				v+=".";i++;}
			else
				v+=s[i];
		}cout<<v<<endl;
	}
    return 0;
}
