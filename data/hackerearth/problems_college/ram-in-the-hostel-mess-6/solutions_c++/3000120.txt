#include <iostream>
#define ll long long
using namespace std;

ll fib(ll n)
{
	if(n==0) return 0;
	if(n==1) return 2;
	if(n==2) return 3;
	return fib(n-1)+fib(n-2);
}
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	cout<<fib(n)*fib(n)<<endl;
    	
    }
    
    return 0;
}
