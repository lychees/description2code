#include<bits/stdc++.h>
using namespace std;
void generate()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int count0=0;
		int count1=0;
		string s;
		cin>>s;
		for(int i=0;s[i]!='\0';i++)
		{
			if(s[i]-48)
				count1++;
			else
				count0++;
		}
		int ans0=0;
		int ans1=0;
		int temp0=0;
		int temp1=0;
		int i=0;
		while(i<count0)
		{
			if(s[i]-48)
				ans0++;
			i++;
		}
		i=0;
		while(i<count1)
		{
			if(!(s[i]-48))
				ans1++;
			i++;
		}
		cout<<min(ans0,ans1)<<endl;
	}
}
int main()
{
	generate();
	return 0;
}
