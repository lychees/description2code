ll a[100002],shuruSe[100001],endSe[100001];
     
int main()
{

	ll sum1,sum2;
	int t,n,i,ok;
	sc(t);
	while(t--)
	{
		ok = 0;
		sum1 = sum2 = 0;
		a[0] = 0;
		sc(n);
		for(i=1;i<=n;i++)
		{
			sc(a[i]);
			if(i!=1)
				shuruSe[i] =a[i] + shuruSe[i-1]; 
			else shuruSe[i] = a[i];
		}
		for(i=n;i>=1;i--)
		{
			if(i!=n)
				endSe[i] =a[i] + endSe[i+1]; 
			else endSe[i] = a[i];
		}
	
		for(i=1;i<=n;i++)

			sum1 = shuruSe[i-1];
			sum2 = endSe[i+1];

				ok = 1;
