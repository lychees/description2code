n = int(raw_input())
entries = raw_input().split()
hallOfFame = {}

for i in entries:
    try:
        hallOfFame[i] += 1
    except BaseException:
        hallOfFame[i] = 1

finalStanding = []
for key in hallOfFame:
    finalStanding.append([key, hallOfFame[key]])

for k, v in sorted(finalStanding, key=lambda x: x[1], reverse=True):
    print k, v
