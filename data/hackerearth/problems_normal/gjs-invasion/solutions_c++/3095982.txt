#define FOR(i,l,r) for(auto i=(l);i<=(r);i++)
#define REP(i,l,r) for(auto i=(l);i<(r);i++)
#define FORD(i,l,r) for(auto i=(l);i>=(r);i--)
#define REPD(i,l,r) for(auto i=(l);i>(r);i--)
#define ENDL printf("\n")
#define debug 1

typedef long long ll;
typedef pair<int,int> ii;

const int inf=1e9;
const int MOD=1e9+7;
const int N=16;

ll f[1<<N],way[1<<N];
ll solve(int n){
    memset(f,0,sizeof(f));
    memset(way,0,sizeof(way));
    way[0]=1;
    REP(i,0,1<<n){
        int cur=n-__builtin_popcount(i)-1;
        for(int j=0,c=0;j<N;j++) if (i&(1<<j)) c++;
        else if (j!=cur) {
            f[i^(1<<j)]+=f[i]+way[i]*c;
            way[i^(1<<j)]+=way[i];
        }
    }
    return f[(1<<n)-1];
}
int main(){
    int test,n;
    cin>>test;
    while (test--){
        cin>>n;
        cout<<solve(n)<<'\n';
    }
}
