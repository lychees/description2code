#include<cstdio>

#define ll long long

ll count;


void calc(ll n)
{
    if(n==0)
        return;
    if(n>=1000)
    {
        count=count+(n/1000);
        n=n%1000;
        calc(n);
        return;
    }
    if(n>=500&&n<1000)
    {
        count++;
        n=n-500;
        calc(n);
        return;
    }
    if(n>=100&&n<500)
    {
        count=count+(n/100);
        n=n%100;
        calc(n);
        return;
    }
    if(n>=50&&n<100)
    {
        count++;
        n=n-50;
        calc(n);
        return;
    }
    if(n>=20&&n<50)
    {
        count=count+(n/20);
        n=n%20;
        calc(n);
        return;
    }
    if(n>=10&&n<20)
    {
        count++;
        n=n-10;
        calc(n);
        return;
    }
    if(n>=5&&n<10)
    {
        count++;
        n=n-5;
        calc(n);
        return;
    }
    if(n>=2&&n<5)
    {
        count=count+(n/2);
        n=n%2;
        calc(n);
        return;
    }
    if(n==1)
    {
        count++;
        n=n-1;
        calc(n);
        return;
    }

}

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        ll n;
        scanf("%lld",&n);
        count=0;
        calc(n);
        printf("%lld\n",count);
    }
    return 0;
}
