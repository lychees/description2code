#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int test,n,p,u,r,s;
	scanf("%lld",&test);
	long long int i=1;
	while(test--)
	{
		scanf("%lld %lld %lld %lld %lld",&n,&p,&u,&r,&s);
		while(n>0)
		{	
	
			while((p+r)<=u&&n>0)
			{
				p=p+r;
				n--;
			}
			if(p>=s&&n>0)
			{
				p=p-s;
				n--;
			}
			else if(p<s&&(p+r)>u)
			{	
				break;
			}
		}
		printf("Case #%lld: %lld\n",i,p);
		i++;
	}
	return 0;
}
