#include<cstdio>
#include<stack>
#include<cstring>
using namespace std;

int main(){
	char str[32810], c;
	stack<char> st;
	for(int i=0;i<882;i++){
		scanf("%s", str);
		int len = strlen(str);
//		printf("%s: %d\n", str, len);
		if(strcmp(str, "1") == 0)
			printf("0\n");
		else{
			for(int j=0; j < len; j++)
				st.push(str[j]);
			int cnt=0;
			while(!st.empty()){
//				printf("here\n");
				if(st.top() == '0'){
//					printf("found 0\n");
					cnt++;
//					printf("cnt = %d\n", cnt);
					st.pop();
				}
				else{
//					printf("found 1\n");
					if(st.size() == 1)
						break;
					cnt++;
					while(!st.empty() && st.top() == '1'){
						cnt++;
//						printf("cnt = %d\n", cnt);
						st.pop();
					}
					if(!st.empty()){
						st.pop();
						st.push('1');
					}
				}
			}
			printf("%d\n", cnt);
		}
		while(!st.empty())
			st.pop();
	}
	return 0;
}
