#include<iostream>
using namespace std;
#include<bits/stdc++.h>
typedef long long LL;
int main()
{ 
    int t,i;
	cin>>t;
	string s;
	while(t--)
	{
		cin>>s;
		LL ans=s[s.length()-1]-'0';
		int temp;
		for(i=s.length()-3;i>=0;i=i-2)
		{
			temp=s[i]-'0';
			if(s[i+1] == '*')
			   ans=ans*temp;
			else if(s[i+1] == '+')
			   ans=ans+temp;
			else if(s[i+1] == '-')
			   ans=ans-temp;
		}
		cout<<ans<<endl;
	}  
	return 0;
}
