#include <iostream>
#include<cstdio>
using namespace std;
int a[105][5005];
int b[105];
int main()
{
	int i,j,n,m,q;
	scanf("%d %d %d",&n,&m,&q);
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			scanf("%d",&a[i][j]);
		}
	}
	while(q--)
	{
		for(i=0;i<n;i++)
		{
			scanf("%d",&b[i]);
		}
		j=m-1;
		for(i=0;i<n;i++)
		{
			while((j<m)&&(a[i][j]>b[i]))
			{
				j--;
			}
		}
		printf("%d\n",j+1);
	}	
    return 0;
}
