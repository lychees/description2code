#include<iostream>
#include<string>
#include<vector>

using namespace std;

int main()
{
    int n ,i,j;
    string test_string="",compare_string = "";
    vector <string> v_test_string;
    cin >> n;
    for( i = 0 ; i < n ; i++)
    {
        cin >> test_string;
        v_test_string.push_back(test_string);

    }

    for( i = 0 ; i < v_test_string.size() ;i++)
    {
        for(j = v_test_string[i].length()-1 ; j >=0;j--){
            compare_string.push_back(v_test_string[i][j]);

        }

        if(v_test_string[i] == compare_string){
            cout<<"YES"<<endl;
        }
        else{
            cout<<"NO"<<endl;
        }
        compare_string ="";

    }





    return 0;
}
