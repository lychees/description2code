#include <stdio.h>
#include <stack>
#include <queue>
using namespace std;

int main()
{
    int n,_time=0,temp,i;
    queue<int> st;
    queue<int> qe;
    scanf("%d",&n);
    for(i=0;i<n;++i)
    {
    	scanf("%d",&temp);
    	qe.push(temp);
    }
    for(i=0;i<n;++i)
    {
    	scanf("%d",&temp);
    	st.push(temp);
    }
    while(!qe.empty())
    {
    	if(st.front()==qe.front())
    	{
    		st.pop();
    		qe.pop();
 			++_time;
    	}
    	else
    	{
    		temp = qe.front();
    		qe.pop();
    		qe.push(temp);
    		++_time;
    	}
    }
    printf("%d",_time);
    return 0;
}
