#include <iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;

#define LL long long 

LL m,n;

LL gcd(LL a, LL b){
	LL s=0,old_s=1;
	LL t=1,old_t=0;
	LL r=b,old_r=a;
	while(r!=0){
		LL quotient = old_r/r;
		LL temp = old_r;
		old_r = r;
		r=temp-quotient*r;
		temp=old_s;
		old_s=s;
		s=temp-quotient*s;
		temp=old_t;
		old_t=t;
		t=temp-quotient*t;
	}
	m=old_s,n=old_t;
	return old_r;
}



int main()
{
	int t;
	scanf("%d",&t);
	while(t--) {
		LL h,u,d;
		scanf("%lld%lld%lld",&h,&u,&d);
		LL g=gcd(u,d);
		if(h%g!=0){
			printf("-1\n");
			continue;
		}
		m=m*(h/g);
		n=n*(h/g);
		n=n*(-1);
		u=u/g;
		d=d/g;
		LL temp=max(abs(m)/d,abs(n)/u)+1;
		m=m+temp*d;
		n+=temp*u;
		temp = min(m/d,n/u);
		printf("%lld\n",m+n-temp*(u+d));
	}
    
    return 0;
}
