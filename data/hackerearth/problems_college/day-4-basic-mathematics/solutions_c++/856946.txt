#include <iostream>
using namespace std;

int main()
{
    unsigned int  t;
    cin>>t;
    for(int x=1;x<=t;x++)
    {
    	int n;
    	cin>>n;
    	int sum=0;
    	int i=0;
    	while(sum<n)
    	{
    		i++;
    		sum+=i;
    	}
    	if(i%3==0)
    	cout<<"SL"<<endl;
    	if(i%3==1)
    	cout<<"LB"<<endl;
   		if(i%3==2)
    	cout<<"BS"<<endl;
    }
    return 0;
}
