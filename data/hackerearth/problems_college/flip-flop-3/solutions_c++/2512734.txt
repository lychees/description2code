#include <iostream>
#include <string>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int test;
    cin>>test;
    while(test--){
    	int len,i,cnt;
    	string str;
    	cin>>str;
    	len=str.size();
    	cnt=0;
    	for(i=0;i<len-1;i++){
    		if(str[i]==str[i+1])
    		cnt++;
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
