n = raw_input()
n = int(n)

cnt = 2
prime = 0
ans = 0
for i in range(2, n+1):
    l = i
    for j in range(2, cnt):
        while (l % j == 0):
            l = l/j
    for j in range(cnt, i+1):
        if (l % j == 0):
            cnt = j+1
            prime = prime + 1
        while (l % j == 0):
            l = l/j
    ans = ans + ((2**(prime))/2)
print ans
