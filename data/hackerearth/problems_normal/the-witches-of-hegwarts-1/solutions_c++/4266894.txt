#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pii pair<int,int>
#define pll pair<ll,ll>
#define mp make_pair
#define ff first
#define ss second
#define pb push_back
#define pf pop_front
#define IOS ios::sync_with_stdio(false)
int t , n ;
int moves[100000];
int main(){

	int i , j , k ;
    cin >> t;
    while(t--){
        cin >> n;
        queue <pii> q;
        map <int,int> moves;
        q.push(mp(n,0));
        moves[n] = 0;
        while(!q.empty()){
            pii el = q.front();
            q.pop();
            if(el.ff==1){
                cout << el.ss << endl;
                break;
            }
            if(moves.find(el.ff-1)==moves.end()){
                q.push(mp(el.ff-1,el.ss+1));
                moves[el.ff-1] = el.ss+1;
            }    
            if(moves.find(el.ff/2)==moves.end())
            if(el.ff%2==0){
                q.push(mp(el.ff/2,el.ss+1));
                moves[el.ff/2] = el.ss+1;
            }
            if(moves.find(el.ff/3)==moves.end())
            if(el.ff%3==0){
                q.push(mp(el.ff/3,el.ss+1));
                moves[el.ff/3] = el.ss+1;
            }     
        }
    }
	return 0;
}
