#include <bits/stdc++.h>
using namespace std;

#define ll long long int
ll a[10000];
int main()
{
    int i,j,k,t,x;
    ll n;
    cin>>n;
    for(i=1;i<=n;i++)
    {
    	cin>>x;
    	a[x]++;
    }
    if(n==1)
    {
    	cout<<0<<endl;
    	return 0;
    }
    ll ans=0;
    int p;
    for(i=1;i*i<=25000000LL;i++)
    {
    	p=i*i;
    	for(j=1;j*j<=p;j++)
    	{
    		if(p%j==0 && (j<=5000 && p/j <=5000))
    		{
    			if(a[j]&&a[p/j])
    			{
    			if(j!=p/j)
    			ans+=a[j]*a[p/j];
    			else
    			{
    				ans+=(a[j]*(a[j]-1))/2;
    			}
    			}
    			
    			
    		}
    	}
    }
    
    ll temp=(n*(n-1))/2;
    double dd=(double)ans/temp;

    printf("%.6lf",dd);
    return 0;
}
