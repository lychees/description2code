#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[1000010],visited[1000010];
bool mydfs(ll int x)
{
	if(x==-1)
	return 1;
	if(visited[x])
	return 0;
	else
	{
		visited[x]=1;
		mydfs(a[x]);
	}
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n;
		cin>>n;
		ll int i;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			visited[i]=0;
		}
		bool ans=mydfs(0);
		if(ans)
		cout<<"Win"<<endl;
		else
		cout<<"Lose"<<endl;
	}
	return 0;
}
