#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
	int t;
	scanf("%d ",&t);
	while(t--)
	{
		char s[100000];
		int a[100000];
		gets(s);
		int l=strlen(s);
		for(int i=1;i<l;i++)
		{
			if(s[i]>=s[i-1])
				a[i]=s[i]-s[i-1];
			else
				a[i]=s[i-1]-s[i];
		}
		int fg=0;
		for(int i=1;i<=(l-1)/2;i++)
		{
			if(a[i]!=a[l-i])
			{
				fg=1;
				break;
			}
		}
		if(fg==1)
			printf("Not Funny\n");
		else
			printf("Funny\n");
	}
	return 0;
}
