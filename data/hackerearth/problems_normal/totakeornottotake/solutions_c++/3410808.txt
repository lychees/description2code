int main()
{
    int t;
    sd(t);
    while(t--)
    {
    	ll pe = 1;
    	ll ne = 1;
    	int b;
    	sd(b);
    	char curr;
    	int num;
    	range(b)
    	{
    		cin>>curr;
    		if(curr=='*') {sd(num);pe*=num;if(ne!=1) ne*=num;}
    		else if(curr=='/') {sd(num); if(ne==1) ne=ne/num;}
    		else if(curr=='+') {sd(num); pe+=num;}
    		else if(curr=='-') {sd(num); ne-=num;}
    		else 
    		{
    			ll val=max(abs(pe),abs(ne));
    			pe=val;
    			ne=val*(-1);
    		}
    	}
    	cout<<pe<<endl;
    	
    }
    return 0;
}
