#include<bits/stdc++.h>
using namespace std;
 
 
int main(int argc, char const *argv[])
{
  
  int T; cin>>T;
  while(T--){
 
  string str1,str2; int n; cin>>str1>>str2; cin>>n;
  
  if(str1.length() != str2.length())
    cout<<"No";
  
  else {
    int counter=0;
    for (int i = 0; i < str1.size(); ++i)
    {
        if(str1[i] != str2[i]) counter++;
    }
    if((n-counter)>=0&&(n-counter)%2==0 )
      cout<<"Yes";
    else
      cout<<"No";
  }
 
  cout<<endl;
}
  return 0;
}
