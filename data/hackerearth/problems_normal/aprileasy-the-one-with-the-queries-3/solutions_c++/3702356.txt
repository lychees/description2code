//learn to make table
//usefull string matching algo
#include<bits/stdc++.h>
using namespace std;
long long int table[100010][26];
int main()
{

	string str;
	cin>>str;

	//create table 
	//long long int table[100000][26];
	long long int len=str.length();
	for(long long int i=0;i<26;i++)
	{
		table[str.length()][i]=str.length();
	}

	for(long long int i=str.length()-1;i>=0;i--)
	{

		for(long long int j=0;j<26;j++)
		{
			table[i][j]=table[i+1][j];
		}
		table[i][str[i]-'a']=i;
	}

	long long int q;

	cin>>q;

	for(long long int i=0;i<q;i++)
	{
		string seq;
		cin>>seq;
		long long int pos=0,flag=1;

		for(long long int j=0;j<seq.length();j++)
		{
			if(table[pos][seq[j]-'a']==len)
			{
				flag=0;
				break;
			}
			else
			{
				pos=table[pos][seq[j]-'a']+1;
			}
		}
		if(flag==0)
			cout<<"No"<<endl;
		else
			cout<<"Yes"<<endl;
	}



	return 0;
}
