T=int(raw_input())
for t in range(0,T):
	n=int(raw_input())
	a=map(int,raw_input().split())
	d=int(raw_input())
	
	if d==0:
		print 1
		continue
	
	i=0
	s=0
	while i < n:
		if a[i]>0:
			s = s+a[i]
		i=i+1
	d=d%s

	if d == 0:
		d=s
		
	i=0
	s=0
	while i < n:
		if a[i]>0:
			s = s+a[i]
			if(s>=d):
				break
		i=i+1
	print i+1
