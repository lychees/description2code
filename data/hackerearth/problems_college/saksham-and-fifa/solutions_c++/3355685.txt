#include <iostream>
using namespace std;
#define ll long long

int main()
{	ll n,a=0,b;
    cin>>n;
    while(n--)
    {	ll p,q;
    	cin>>p>>q;
    	if(p>a)
    	{	a=p;
    		b=q;
    	}
    }
    if(b<0)
    	cout<<"Right Corner\n";
    else if(b>0)
    	cout<<"Left Corner\n";
    else
    	cout<<"Either Corner\n";
    return 0;
}
