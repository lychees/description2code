    #include<stdio.h>
    #include<stdlib.h>
     
    int main(void)
    {
     int n,m,i,j,k,flag=0,ele1,ele2;
     scanf("%d%d",&n,&m);
     
     long arr[n];
     
     for(i=0;i<n;i++)
      scanf("%ld",&arr[i]);
     
     for(i=0;i<n;i++)
        {
         ele1=arr[i];
         for(j=i+1;j<n;j++)
         {
            ele2=arr[j];
            for(k=j+1;k<n;k++)
                {
                  if((ele1+ele2+arr[k])==m)
                      {
                        printf("YES\n");
                        flag=1;
                        break;
                      }
                 }
               if(flag==1)
                 break;
          }
     
         if(flag==1)
          break;
     
        }
     
     if(flag==0)
       printf("NO\n");
     
     return 0;
     
    }
