#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll int a[3]={1,3,5};
	ll int dp[110][4];
	ll int i,j;
	for(j=0;j<=3;j++)
	{
		dp[0][j]=1;
	}
	ll int x,y;
	for(i=1;i<110;i++)
	{
		for(j=0;j<=3;j++)
		{
			if(j==0)
			{
				dp[i][j]=0;
			}
			else
			{
				x=dp[i][j-1];
				if((i-a[j-1])>=0)
				{
					y=dp[i-a[j-1]][j];
				}
				else
				{
					y=0;
				}
				//~ if(i==1 && j==1)
				//~ cout<<x<<" "<<y<<endl;
				dp[i][j]=x+y;
			}
		}
	}
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int n;
		scanf("%lld",&n);
		printf("%lld\n",dp[n][3]);
	}
	return 0;
}
