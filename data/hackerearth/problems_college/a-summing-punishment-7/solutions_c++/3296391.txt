#include <iostream>
using namespace std;
#define lim 100000
int a[lim+1];
long long sum[lim+1];
int main()
{
    int n;
    cin>>n;
     cin>>a[1];
     sum[1]=a[1];
    for(int i=2;i<=n;i++)
    {
    	cin>>a[i];
    	sum[i]=sum[i-1]+a[i];
    	//out<<sum[i]<<" "<<i<<"  p"<<endl;
    }
    int q,l,r;
    cin>>q;
    while(q--)
    {
    	cin>>l>>r;
    	cout<<sum[r]-sum[l]+a[l]<<endl;
    }
}
