#include<bits/stdc++.h>
using namespace std;
int dp[3001];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,x,y;
		cin>>n>>x;
		int a[n];
		for(int i=0;i<n;i++)
			cin>>a[i];
		for(int i=0;i<=x;i++)
			dp[i]=0;
		for(int i=1;i<=x;i++)
		{
			for(int j=0;j<n;j++)
			{
				y=a[j];
				if(y<=i)
					dp[i]=max(dp[i],y+dp[i-y]);
			}
		}
		cout<<"\n"<<dp[x];
	}
	return 0;
}
