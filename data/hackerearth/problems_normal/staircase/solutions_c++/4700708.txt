#include <iostream>
#include <cstdlib>
using namespace std;

int main()
{
	int n;
    cin>>n;
    if(n<1){
    	cout<<"ERROR";
    	exit(0);
    }
    int DP[n];
    int DPJI[n];
    DP[0]=DP[1]=1;
    DP[2]=2;
    for(int i=0;i<3;i++)
    	DPJI[i]=DP[i];
    for(int i=3;i<=n;i++){
    	DP[i]=DP[i-1]+DP[i-2]+DP[i-3];
    	DPJI[i]=DPJI[i-1]+DPJI[i-2];
    }
    cout<<"Jack-"<<DP[n]<<endl;	
    cout<<"Jill-"<<DPJI[n];	
}
