
no_of_test_cases = int(raw_input())

for _ in range(no_of_test_cases):
    inp = raw_input().split()
    N,M,A,B,C = int(inp[0]),int(inp[1]),int(inp[2]),int(inp[3]),int(inp[4])
    inplace = N*A+M*B;
    outplace = min(N,M)*C*2
    if (N>M):
        outplace+=(N-M)*A
    else:
        outplace+=(M-N)*B
    print max(inplace,outplace)
