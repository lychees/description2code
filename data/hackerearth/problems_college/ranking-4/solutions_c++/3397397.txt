#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int

int main()
{
    ll n, t, x, i;
    cin >> t;
    while (t--) {
        cin >> n;
        vector<ll> v;
        vector<ll> v1;
        for(i = 0; i < n; i++) {
            cin >> x;
            v.pb(x);
            v1.pb(x);
        }
        sort(v1.begin(), v1.end());
        /*for(i = 0; i < n; i++) {
            cout << v1[i] << " ";
        }
        cout << endl;*/
        for(i = 0; i < n; i++) {
            cout << n - (upper_bound(v1.begin(), v1.end(), v[i]) - v1.begin()) + 1 << " ";
        }
        cout << endl;
    }

    return 0;
}
