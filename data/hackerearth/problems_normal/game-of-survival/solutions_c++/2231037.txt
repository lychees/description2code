#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int t,n;
	cin>>t;
	while(t--){
		cin>>n;
		int k[n],a[n];
		for(int i=0;i<n;i++)
			cin>>k[i];
		for(int i=0;i<n;i++)
			cin>>a[i];
		sort(a,a+n);
		sort(k,k+n);
		int i,flag=0;
		for(i=0;i<n;i++){
			if(a[i]>=k[i]){
				cout<<"NO"<<endl;
				flag=1;
				break;
			}
		}
		if(flag==0)
			cout<<"YES"<<endl;
	}
	return 0;
}
