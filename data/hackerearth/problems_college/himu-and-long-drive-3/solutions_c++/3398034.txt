#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll power(ll x, ll y)
{
    if( y == 0)
        return 1;
    else if (y%2 == 0)
        return power(x, y/2)*power(x, y/2);
    else
        return x*power(x, y/2)*power(x, y/2);
 
}

int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		ll n;
		cin>>n;
		ll cnt=log2(n);
		//cout<<"Log2="<<cnt<<endl;
		printf("%lld\n",power(2,cnt+1)-1);
		//cout<<pow(2,cnt+1)-1<<endl;
	}
	return 0;
}
