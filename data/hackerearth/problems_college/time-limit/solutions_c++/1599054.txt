#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m;
	cin>>n>>m;
	int a[n],b[m],i;
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0;i<m;i++)
		cin>>b[i];
	sort(a,a+n);
	sort(b,b+m);
	if(a[n-1]>=2*a[0])
	{
		if(a[n-1]<b[0])
		{
			cout<<a[n-1]<<endl;
		}
		else
		{
			cout<<'-1'<<endl;
		}
	}
	else if(a[n-1]<2*a[0])
	{
		if((2*a[0])<b[0])
		{
			cout<<2*a[0]<<endl;
		}
		else
		{
			cout<<'-1'<<endl;
		}
	}
	return 0;
}
