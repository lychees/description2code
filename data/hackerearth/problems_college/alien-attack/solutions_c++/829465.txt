using namespace std;
inline void fi(int &x) {
	register int c = getchar_unlocked();x = 0;
	for(; c<48 || c>57; c = getchar_unlocked());
	for(; c>47 && c<58 ; c = getchar_unlocked()) x = (x<<1) + (x<<3) + c - 48;
}
typedef unsigned long long uint64;

uint64 PowMod(uint64 x, uint64 e, uint64 mod)
{
  uint64 res;

  if (e == 0)
  {
    res = 1;
  }
  else if (e == 1)
  {
    res = x;
  }
  else
  {
    res = PowMod(x, e / 2, mod);
    res = res * res % mod;
    if (e % 2)
      res = res * x % mod;
  }

  return res;
}

long int m=1000000007;
int main() {
	int t,n;
	long long int ans;
	fi(t);
	while (t--) {
		fi(n);
		ans=3*PowMod(2,n-1,m);
		if(n&1) ans--;
		else ans++;
		printf("%lld\n",ans%m);	
	}
	return 0;
}
