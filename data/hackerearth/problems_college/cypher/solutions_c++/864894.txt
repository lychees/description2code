#include <bits/stdc++.h>
using namespace std;

int main()
{
	char luv[28];
	for(int i=97;i<123;i++)
	luv[i-97]=i;
	int t;
	scanf("%d",&t);
	while(t--)
	{
		char str[105];
		scanf("%s",str);
		int len=strlen(str);
		for(int i=0;i<len;i++)
		{
			int j=(str[i]+i-97)%26;
			cout<<luv[j];
		}
		cout<<endl;
	}
    return 0;
}
