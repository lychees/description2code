#include<iostream>

using namespace std;

int main(){
	
	int num;
	
	cin>>num;
	
	int arr[num+1];
	arr[0]=0;
	int sum=0;
	
	for(int i=1;i<=num;i++){
		int j;
		cin>>j;
		sum=sum+j;
		arr[i]=sum;
	}
	
	int q;
	cin>>q;
	for(int i=0;i<q;i++){
		
		int m,n;
		cin>>m>>n;
		int ans=arr[n]-arr[m-1];
		cout<<ans<<"\n";
	}
}
