#include <bits/stdc++.h>
using namespace std;

int main()
{
   	int t , cnt ;
   	long long num , n;
   	cin >> t;
   	while(t--)
   	{
   		cnt = 0;
   		cin >> num;
   		n = sqrt(num);
   		for(int i = 1 ; i <= n ; ++i)
   		{
   			if(num % i == 0)
   			{
   				if((num / i != i) && ((num / i) % 2 == 0)) ++cnt;	
   				if(i % 2 == 0) ++cnt;
   			}
   		}
   		cout << cnt << endl;
   	}
    return 0;
}
