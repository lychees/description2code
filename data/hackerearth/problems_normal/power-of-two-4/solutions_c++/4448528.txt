#include <iostream>
using namespace std;
int bit[205][32],arr[205];
int main()
{
	int p[32];
	p[31]=1;
	for(int i=30;i>=0;i--)
	p[i]=p[i+1]*2;
   int t,n;
   cin>>t;
   while(t--)
   {
   	int flag=0;
   	cin>>n;
   	for(int i=0;i<n;i++)
   	{
   		cin>>arr[i];
   		int x=arr[i];
   		for(int j=31;j>=0;j--)
   		{
   			bit[i][j]=x%2;
   			x/=2;
   			
   		}
   	}
   	for(int i=0;i<32;i++)
   	{
   		int ans=-1;
   		for(int j=0;j<n;j++)
   		{
   			if(bit[j][i]==1)
   			{
   				if(ans==-1)
   				ans=arr[j];
   				else
   				ans&=arr[j];
   			}
   			
   		}
   		if(ans==p[i])
   		flag=1;
   	}
   	if(flag)
   	cout<<"YES\n";
   	else
   	
   	cout<<"NO\n";
   }
    
    return 0;
}
