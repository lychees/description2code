#include<bits/stdc++.h>
using namespace std;
int f[26];
int ind[26];
char a[10010];
int main(){
	int test,c,n,l;
	
	
	cin>>test;
	while(test--){
		cin>>a;
		n=0;
		c=0;
		l=strlen(a);
		memset(f,0,sizeof(f));
		
		for(int i=0;i<l;i++){
			if(f[a[i]-'A']){
				if(c>=i-ind[a[i]-'A'])
				c=i-ind[a[i]-'A'];
			else 
				c++;
				ind[a[i]-'A']=i;
				
			}
			else
			{
				c++;
				f[a[i]-'A']=1;
				ind[a[i]-'A']=i;
			}
			if(c>n)
				n=c;
		}
		cout<<n<<endl;
	}
	
	return 0;
}
