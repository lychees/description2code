#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	string s;
	int n,i,j,count;
	cin>>n;

	 for(i=0;i<n;i++)
	 {
	 	count=0;
	 	cin>>s;
	 	for(j=0;j<s.size();j++)
	 	{
	 		if(s[j]==s[j+1])
	     	count++;
	 		else
	 		continue;
	 	}
	 	cout<<count<<endl;
	 }
	
	return 0;
}
