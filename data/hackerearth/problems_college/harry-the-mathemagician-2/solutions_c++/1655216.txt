#include <iostream>
#include <assert.h>
using namespace std;
#define ll long long int

int main()
{
    ll t, n, r, f;
    scanf("%lld", &t);

    while(t--) {
    	f = 0;
        cin >> n >> r;
       
      
        
        if (r > n) {
        	cout << "odd\n";
        	f = 1;
        	continue;
        }

        while (n) {
        	int x = n % 2;
        	int y = r % 2;
        	
        	if (x == 0 && y == 1) {
        		cout << "even\n";
        		f = 1;
        		break;
        	}
        	
        	n >>= 1;
        	r >>= 1;
        }
        
        if (f == 0) cout << "odd\n";
    }

    return 0;
}
