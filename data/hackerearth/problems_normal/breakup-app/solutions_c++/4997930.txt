#include<iostream>
#include<sstream>
#include<vector>
#include<string>
#include<algorithm>
using namespace std;
#define IOS ios_base::sync_with_stdio(false);cin.tie(0)
typedef long long LL;
int main() {
  IOS;
  int n;
  cin >> n;
  cin.ignore();
  string message;
  vector<int> scores(31,0);
  for (int i = 0; i < n; i++) {
    message.clear();
    getline(cin, message);
    message.push_back(' ');
    int base = 1;
    if (message[0] == 'G') {
      base = 2;
    }
    for (int j = 3; j < message.size(); j++) {
      if (message[j] - '0' <= 9 &&
        message[j] - '0' >= 0) {
        int num = message[j] - '0';
        for (int k = j + 1; k < message.size(); k++) {
          if (message[k] == ' ') {
            j = k;
            break;
          }
          num = num * 10 + message[k] - '0';
        }
        scores[num] += base;
      }
    }
  }
    int max = -1;
    int maxCount = 0;
    int maxIndex = -1;
    for (int i = 1; i < scores.size(); i++) {
      if (scores[i]>max) {
        maxCount = 1;
        max = scores[i];
        maxIndex = i;
      }
      else if (scores[i] == max) {
        maxCount++;
      }
    }

    if (maxCount > 1) {
      cout << "No Date" << endl;
    }
    else {
      if (maxIndex == 19 || maxIndex == 20) {
        cout << "Date" << endl;
      }
      else {
        cout << "No Date" << endl;
      }
    }

  return 0;
}
