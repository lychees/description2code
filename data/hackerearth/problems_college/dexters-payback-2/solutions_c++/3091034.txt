#include<bits/stdc++.h>
#include <stdio.h>
#include<stdlib.h>
 
int main()
{
    int t,i,j=0;
    
    int c,r,answer=0;
    scanf("%d",&t);
    while(t--)
    {
        answer=0;
        j=0;
        scanf("%d%d",&c,&r);
        int arr[r][c];
        for(i=0;i<r;i++)
        {
            for(j=0;j<c;j++)
            {
                arr[i][j]=0;
                if(j-2>=0){
                if(arr[i][j-2]==1)
                    continue;
                }
                if(i-2>=0){
                	if(arr[i-2][j]==1)
                    continue;
                }
             
                    arr[i][j]=1;
                    answer++;
            
            }
        }
 
        printf("%d\n",answer);
    }
 
    return 0;
}
