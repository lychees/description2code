def cal(p):
    a = [1, 2, 3, 5, 8]
    while (a[len(a)-1] < p):
        b = a[len(a)-1]+a[len(a)-2]
        if b <= p:
            a = a+[b]
        else:
            break
    c = 0
    for i in range(0, len(a)):
        if a[i] % 2 == 0:
            c = c+a[i]
    return c


k = input()
for i in range(0, k):
    p = input()
    print cal(p)
