#include<iostream>
#include<cstdio>
#include<cstdlib>
using namespace std;

#define GC getchar_unlocked
 
long long inp()
{
    long long ip = getchar_unlocked(), ret = 0, flag = 1;
    for(; ip < '0' || ip > '9'; ip = getchar_unlocked())
    {
        if(ip == '-')
        {
            flag = -1;
            ip = getchar_unlocked();
            break;
        }
    }

    for(; ip >= '0'&& ip <= '9'; ip = getchar_unlocked())
        ret = ret * 10 + ip - '0';
    return flag * ret;
}
int main()
{
	long int t,n,i,k,m,l;
	char ch;
	t=inp();
	while(t--)
	{
		n=inp();
		long int a[n];
		for(i=0;i<n;i++)
		{
			a[i]=inp();
			if(a[i]>127)
			{
				ch=255-a[i];
				cout<<ch;
			}
			else 
			{
				ch=a[i];
				cout<<ch;
			}
		}
		cout<<endl;
	}
	return 0;
}
