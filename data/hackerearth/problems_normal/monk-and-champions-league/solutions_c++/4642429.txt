#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{ int m,n;
  cin>>m>>n;
  priority_queue<int>pq;
  for(int i=1;i<=m;i++)
  {
  int num;
  cin>>num;
  pq.push(num);
  }
  int count=0;
  for(int i=1;i<=n;i++)
  {
  int tmp=pq.top();
  count+=tmp;
  pq.pop();
  pq.push(tmp-1);
  }
  
  cout<<count;
    return 0;
}
