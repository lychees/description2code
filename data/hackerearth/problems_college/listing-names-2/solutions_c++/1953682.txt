#include <iostream>
#include <map>
#include <vector>
#include <iterator>

using namespace std;

int main () {
	int n; cin >> n;
	map <string , int> m;
	map <string , int>::iterator that;
	for (int i = 0; i < n; i++) {
		string s; cin >> s;
		m[s]++;
	}
	for (that = m.begin(); that != m.end(); that++) {
		cout << that->first << " " << that->second << endl;
	}
	return 0;
}
