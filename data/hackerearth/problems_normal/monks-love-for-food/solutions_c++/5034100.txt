#include <iostream>
using namespace std;

int main()
{
    int stack[10000000];
    int top = -1;
    int tc;
    cin>>tc;
    while(tc--){
    	int a,cost;
    	cin>>a;
    	if(a==1){
    		if(top<0)
    			cout<<"No Food"<<endl;
    		else{
    			cout<<stack[top]<<endl;
    			top--;
    		}
    	}
    	else{
    		cin>>cost;
    		top++;
    		stack[top]=cost;
    	}
    	
    }
    return 0;
}
