typedef long long LL;

#define MOD 1000000007
#define MAX 1000
#define sd(x) scanf("%d", &x)

using namespace std;

bool prime[MAX];

int main() {
	int s = sqrt(MAX);
	
	for (int i=3; i<=s; i+=2) {
		if (prime[i]==false) {
			int k=2*i;
			for (int j=i*i; j<MAX; j+=k) prime[j]=true;
		}
	}
	
	int n;
	cin >> n;
	
	while (n--) {
		int a;
		int ans=0;
		scanf("%d", &a);
		if (a%2==0) {
			int cnt=1;
			a/=2;
			while (a%2==0){++cnt;a/=2;}
			ans^=cnt;
		}
		
		for (int i=3; i<1000 && a!=1; i+=2) {
			if (prime[i]==false && a%i==0) {
				int cnt=1;
				a/=i;
				while (a%i==0) {++cnt;a/=i;}
				ans^=cnt;
			}
		}
		if (a!=1)ans^=1;
		if (ans==0) {
			puts("Pappu Bhujia");
		} else puts("Printf{}");
	}
	
	
	
	return 0;
}
