#define pus(a,n)                    for(int i=0;i<n;i++)    printf("%d ",a[i])

int main()
{
ios::sync_with_stdio(0);
cin.tie(0);
int t;
cin>>t;
while(t--)
{
	int x,a,b,s=0;
	cin>>x;
	for(int i=0;i<=x;i++)
	{
		for(int j=0;j<=x;j++)
		{
			if((pow(i,j)+pow(j,i))==x && abs(i-j)>=s)
			{
				a=i;
				b=j;
				s=abs(a-b);
			}
		}
	}
	cout<<a<<" " <<b<<"\n";
	
}


return 0;
}
