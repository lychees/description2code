#include<bits/stdc++.h>
using namespace std;
#define ll long long 
ll int dp[1002];
int main()
{
    dp[0]=1;
    dp[1]=1;
    dp[2]=2;
    dp[3]=5;
    int i;
    for(i=4;i<=100;i++)
    {
        dp[i]=(((2*(2*i-1)))*dp[i-1])/(i+1);
    }
   int t;
   cin>>t;
   while(t--)
   {
       int n;
       cin>>n;
       cout<<dp[n]<<endl;
   }

}
