
#include <iostream>
using namespace std;
   long long int arr[30];
   void preprocess(string ss){
   	for(int i=0;i<ss.size();i++){
   		   arr[ss[i]-'a']++;
   	}
   	
   	
   }
   long long int dp[30][30];
int main()
{
    int n;
    cin>>n;
     string ss;
    	cin>>ss;
    	preprocess(ss);
    	long long int brr[30];
    	for(int i=0;i<26;i++){
    		brr[i]=arr[i];
    		
    	}
       for(int i=0;i<ss.size();i++){
       	    for(int j=0;j<26;j++){
       	    	   if(arr[j]>0)
       	    	    dp[ss[i]-'a'][j]+=arr[j];
       	    }
       	    arr[ss[i]-'a']--;
       	
       	
       }
    	int q;
    	cin>>q;
    	while(q--){
    		char a,b;
    		cin>>a>>b;
    		if(a!=b)
    		cout<<dp[a-'a'][b-'a']+dp[b-'a'][a-'a']<<endl;
    		else
    		cout<<dp[a-'a'][a-'a']-brr[a-'a']<<endl;
    		
    	}
    }
