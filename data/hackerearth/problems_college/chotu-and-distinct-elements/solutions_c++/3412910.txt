#include<bits/stdc++.h>
using namespace std;
main()
{
	int t;
	scanf("%d",&t);
	while(t>0)
	{
		long long int n,i;
		scanf("%lld",&n);
		long long int a[n],b[n];
		for(i=0;i<n;i++)
		{
			scanf("%lld",&a[i]);
		}
		sort(a,a+n);
		b[0]=a[0];
		int k=1;
		for(i=1;i<n;i++)
		{
			if(a[i]!=a[i-1])
			b[k++]=a[i];
		}
		if(k==1)
		{
		printf("%lld\n",b[0]);
		}
		else{
			int m=k;
			if(k%2!=0)
			{
				m=(k+1);
			}
		for(i=0;i<m/2;i++)
		{	if(b[k-(i+1)]==b[i])
			{
			printf("%lld",b[i]);
			break;}
			printf("%lld %lld ",b[k-(i+1)],b[i]);
		}
		printf("\n");
	}
	t--;}
}
