#include <iostream>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int i=0;
    while(s[i]!='\0')
    {
    	if(s[i]=='u'&& s[i+1]=='s' && s[i+2]=='e' && s[i+3]=='r' && s[i+4]=='n' && s[i+5]=='a' && s[i+6]=='m' && s[i+7]=='e' && s[i+8]=='=')
    	{
    		i=i+9;
    		cout<<"username: ";
    		while(s[i]!='&')
    		{
    			cout<<s[i];
    			i++;
    		}
    		cout<<"\n";
    		i=i+5;
    	
    		cout<<"pwd: ";
    		while(s[i]!='&' || s[i+1]!='p')
    		{
    			cout<<s[i];
    			i++;
    		}
    		cout<<"\n";
    		i=i+9;
    	
    		cout<<"profile: ";
    		while(s[i]!='&' || s[i+1]!='r')
    		{
    			cout<<s[i];
    			i++;
    		}
    		cout<<"\n";
    		i=i+6;
    	
    		cout<<"role: ";
    		while(s[i]!='&' || s[i+1]!='k')
    		{
    			cout<<s[i];
    			i++;
    		}
    		cout<<"\n";
    		i=i+5;
    	
    		cout<<"key: ";
    		while(s[i]!='\0')
    		{
    			cout<<s[i];
    			i++;
    		}
    	}
    	if(s[i]=='\0')
    	break;
    	i++;
    }
    return 0;
}
