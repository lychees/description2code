#include <iostream>
#include<algorithm>
#include<string>
using namespace std;
bool ispalindrome(string a)
{
//	cout<<a<<" ";
	string h;
	h=a;
	reverse(a.begin(),a.end());
	//cout<<a<<endl;
	if(h==a)
	return true;
	return false;
}
int main()
{
   int n;
   cin>>n;
   while(n--)
   {
   	string s,t,tem;int count=0;
   	cin>>s>>t;
   	
   	for(int i=0;i<=s.length();i++)
   	{
   		tem=s;
   		tem.insert(i,t);
   		if(ispalindrome(tem))
   		count++;
   	}
   	cout<<count<<endl;
   }
    return 0;
}
