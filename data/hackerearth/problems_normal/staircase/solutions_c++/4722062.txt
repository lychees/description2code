#include <vector>

int main()
{
    int n;

	if(n<0)
		cout<<"ERROR\n";

	{

	    vec.resize(n+1);
	    vec[0]=0;
	    vec[1]=1;
	    vec[2]=2;
	    vec[3]=4;
	    for(int i=4;i<=n;i++)
		{
			vec[i]=vec[i-1]+vec[i-2]+vec[i-3];
		}
		vector<long long int> vec2;
		vec2.resize(n+1);
		vec2[0]=0;
		vec2[1]=1;
		vec2[2]=2;
		for(int i=3;i<=n;i++)
		{
			vec2[i]=vec2[i-1]+vec2[i-2];
		}
			cout<<"Jack-"<<vec[n]<<endl;
			cout<<"Jill-"<<vec2[n]<<endl;
	}	
}
