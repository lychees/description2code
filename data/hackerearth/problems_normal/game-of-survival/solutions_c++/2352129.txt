#include <iostream>
using namespace std;

void quicksort(int *a,int low, int high)
{
	int i,j,tmp,pivot;
	if(low<high)
	{
		pivot=low;
		i=low;
		j=high;
		while(i<j)
		{
			while(a[i]<=a[pivot]&&i<=high)
			{
				i++;
			}
			while(a[j]>a[pivot]&&j>=low)
			{
				j--;
			}
			
			if(i<j)
			{
				tmp=a[i];
				a[i]=a[j];
				a[j]=tmp;
			}
		}
		
		tmp=a[j];
		a[j]=a[pivot];
		a[pivot]=tmp;
		quicksort(a,low,j-1);
		quicksort(a,j+1,high);
	}
}
