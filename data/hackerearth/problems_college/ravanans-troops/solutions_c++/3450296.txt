#include <bits/stdc++.h>

using namespace std;
int dp[1000005];

void fun(int num)
{
    dp[0] = 0;
    dp[1] = 1;
    dp[2] = 2;
    dp[3] = 3;
    dp[4] = 2;

    for (int i = 5; i<= num;i++)
    {
        int temp = i;
        for (int j = 1;j < i;j++)
        {
            if (pow(j,2) > i)
                break;
            else
                temp = min(temp,dp[i-(j*j)]);
        }

        dp[i] = temp+1;
    }
}

int main()
{
    fun(600000);
    
    int tc;
    scanf("%d",&tc);

    while(tc--)
    {
        int num;
        scanf("%d",&num);
        printf("%d\n",dp[num]);
    }

    return 0;
}
