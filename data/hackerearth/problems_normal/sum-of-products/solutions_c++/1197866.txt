#include <iostream>
#include <math.h>
#define ll long long
#define mod 1000000007
using namespace std;

int main()
{
    ll t,n;
    
    ios_base::sync_with_stdio(false);
    
    
    
    cin>>t;
    while(t--)
    {
    cin>>n;
    
    ll ans=0;
    
    ll st=1;
    ll s1=1,s2;
    while(st<((ll)sqrt(n)))
    {
    	
    	
    	ll le= n/(st+1) + 1;
    	ll ri= n/(st);
    	
    	ll to= n/(n/st);
    	ll no= ri-le+1;
    	
    	ll val= (no*(le+ri))/2;
    	
    	//if(n==10)cout<<st<<" "<<val<<" "<<to<<endl;
    	val%=mod;
    	
    	
    	ans=(ans+ (to*val)%mod)%mod;
    	
    	st++;
    	
    	
    }
    
    ll lim= n/(ll)(sqrt(n));
    for(int i=1;i<=lim;i++)
    {
    	ans=( ans+ (i*(n/i))%mod)%mod;
    }
    
    cout<<ans;
    if(t>0)cout<<"\n";
    }
    
    
    return 0;
}
