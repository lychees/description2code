def seive():
    p = [0] * 1000001
    for i in range(2, 1000001):
        if not p[i]:
            j = i
            while j < 1000001:
                p[j] += i
                j += i

    return p


prim = seive()
n, k = map(int, raw_input().split())
cnt = 0
for i in range(1, n+1):
    if prim[i] == k:
        cnt += 1
print cnt
