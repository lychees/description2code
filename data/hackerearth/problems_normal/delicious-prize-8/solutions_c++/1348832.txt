#include <iostream>
#include <vector>
#include <set>
#include <algorithm>
using namespace std;

int main()
{
    
    int t,n;
    long long k;
    scanf("%d",&t);
    while(t--){
    	scanf("%d%lld",&n,&k);
    	multiset<long long> s;
    	vector<long long> v;
    	long long x = 2;
    	while(x*x<=k){
    		if (k%x==0){
    			//v.push_back(x);
    			while(k%x==0){
    				v.push_back(x);
    				k /= x;
    			}
    		}
    		if (x==2) x++;
    		else x+=2;
    	}
    	if (k!=1) v.push_back(k);
    	sort(v.rbegin(),v.rend());
    	for(int i=0;i<n;i++) s.insert(1);
    	for(int i=0;i<v.size();i++){
    		auto it = s.begin();
    		s.erase(s.begin());
    		s.insert(v[i]*(*it));    		    		
    	}
    	long long ans  =0;
    	for(auto it=s.begin();it!=s.end();it++) ans += (*it);
    	printf("%lld\n",ans);
    }
    return 0;
}
