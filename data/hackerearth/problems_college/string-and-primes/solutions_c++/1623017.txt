bool isprime(int n)
{  bool flag=true;
	for(int i=2;i<=n/2;i++)
	{
		if(n%i==0)
		flag=false;
	}
	return flag;
}
char *afteredit(char str[])
{
	int l=strlen(str);
	int k=0;
	for(int i=0;str[i];i++)
	{
		int t=str[i];
		if(!isprime(t))
		str[k++]=str[i];
	}
	str[k]='\0';
	if(k==0)
	return"NULL";
	return str;
}
