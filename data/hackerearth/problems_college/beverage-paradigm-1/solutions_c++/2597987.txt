#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	int n, target, state = 0;
	cin >> n;
	int arr [n];
	for ( int i = 0; i < n; i++ )
		cin >> arr [i];
	cin >> target;
	sort(arr, arr + n);
	int i = 0 , l, r;
	for (; i < n - 2;)
	{
		l = i + 1; 
        r = n - 1; 
        while (l < r)
        {
            if( arr[i] + arr[l] + arr[r] == target)
            {
                state = 1;
                break;
            }
            else if ( arr[i] + arr[l] + arr[r] < target )
                l++;
            else 
                r--;
        }
        if ( state == 1 )
        	break;
        i++;
	}
	if ( state == 0 )
		cout << "False" << endl;
	else 
		cout << "True" << endl;
    return 0;
}
