#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

int main()
{
    long long i, t, n;
    cin >> t;
    
    while (t--) {
    	cin >> n;
    	i = 0;
    	while (n >= 0) {
    		n -= 30;
    		i++;
    	}
    	
    	cout << fixed << setprecision(0) << pow(2, i) << endl;
    	
    }
    return 0;
}
