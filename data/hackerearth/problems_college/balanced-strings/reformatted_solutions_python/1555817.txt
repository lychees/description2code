for _ in xrange(input()):
    string = raw_input()
    count = {}
    for ele in string:
        if count.has_key(ele):
            count[ele] += 1
        else:
            count[ele] = 1

    keys = count.keys()
    if len(keys) != 26:
        print "No"
    else:
        init = count[keys[0]]
        flag = 0
        for key in keys:
            if count[key] != init:
                print "No"
                flag = 1
                break

        if flag == 0:
            print "Yes"
