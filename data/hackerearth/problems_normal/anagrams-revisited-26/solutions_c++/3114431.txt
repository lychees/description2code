#include <bits/stdc++.h>
using namespace std;

const int maxn = 1e5+5;
int main(){
	ios_base::sync_with_stdio(false);
	int T;
	cin >> T ;
	string s ;
	map<string,int>m;
	while(T--){
		cin >> s ;
		sort(s.begin(),s.end());
		m[s]+=1;
	}
	int mx = -1 ;
	for(map<string,int>:: iterator it = m.begin();it!=m.end();++it){
		if( it->second > mx)
		mx = it->second ;
	}
	cout << mx << endl; 
}
