int dpa(int n)
{
	if(n<0)
		return 0;
	if(n==0)
	{
		return 1;
	}
	int arr[] = {1,2,3};
	int count = 0;
	for(int i=0;i<3;i++)
	{
		count+=dpa(n-arr[i]);
	}
	return count;
}
int dpi(int n)
{
	if(n<0)
		return 0;
	if(n==0)
	{
		return 1;
	}
	int arr[] = {1,2};
	int count = 0;
	for(int i=0;i<2;i++)
	{
		count+=dpi(n-arr[i]);
	}
	return count;
}
