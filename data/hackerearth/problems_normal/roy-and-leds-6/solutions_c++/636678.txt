#include <iostream>
using namespace std;
int op[9];

int main() {
	int t,r,g,b;
	scanf("%d %d %d %d",&t,&r,&g,&b);
	
	int str=0,stg=0,stb=0,i;
	int sr=r,sg=g,sb=b;
	
	
	
	for(i=0;i<t;i++)
	{
		if(str==0 && stg==0 && stb==0)op[8]++;
		
		else if(str==0 && stg==0 && stb==1)op[3]++;
		
			else if(str==0 && stg==1 && stb==0)op[2]++;
				else if(str==0 && stg==1 && stb==1)op[5]++;
					else if(str==1 && stg==0 && stb==0)op[1]++;
						else if(str==1 && stg==0 && stb==1)op[6]++;
							else if(str==1 && stg==1 && stb==0)op[4]++;
								else op[7]++;
								
		sr--; sg--; sb--;
		
		if(sr==0){sr=r; str= (str+1)&1;}
		if(sg==0){sg=g; stg= (stg+1)&1;}
		if(sb==0){sb=b; stb= (stb+1)&1;}
	
		
	}
	
	printf("%d %d %d %d %d %d %d %d",op[1],op[2],op[3],op[4],op[5],op[6],op[7],op[8]);
	
	
	
	return 0;
}
