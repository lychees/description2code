#include<bits/stdc++.h>
#define ll long long
using namespace std;
vector< ll > v[11];
void create(ll num , int len ){
  if(len == 10){
  v[len].push_back(num);
  return;
  }
  v[len].push_back(num);
  for(int i = 0 ; i <= 8 ; i += 2 )
   create( num * 10 + i , len + 1);
}
int Count(ll Num){
  int Ct = 0;
  while(Num){
   Ct++;
   Num /= 10;
  }
  return Ct;
}
int main(){
     v[1].push_back(0);
    for(int i = 2 ; i <= 8 ; i += 2 )
     create( i , 1  );
      int t;
      scanf("%d",&t);
      while( t-- ){
         ll N;
         scanf("%lld",&N);
         int Dig = Count(N);
         int idx = lower_bound( v[Dig].begin() , v[Dig].end() , N ) - v[Dig].begin();
         if(idx == v[Dig].size()){
         idx = 0;
         Dig++;
         }
         if(v[Dig][idx] == N )
         printf("Unlimited Power\n");
         else {
         ll up = - N + v[Dig][idx];
         ll down;
         if(idx)
         down =  N - v[Dig][idx-1];
         else
         down =  N - v[Dig-1][v[Dig-1].size()-1];
         ll gc = __gcd(up,down);
         up /= gc;
         down /= gc;
         printf("%lld/%lld\n",up,down);
         }
      }
    return 0;
}
