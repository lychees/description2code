#include<bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	string str;
	cin>>str;
	bool visited[27]={false};
	int max_char=-1;
	int start=0;
	while(start<n){
		int max_index=-1;
		  max_char=0;
		for(int i=start;i<n;i++){
			if(str[i]>max_char&&!visited[str[i]-'a']){
				max_char=str[i];
				max_index=i;
			}
		}
		if(max_index==-1){
			break;
		}
		printf("%c",max_char);
		start=max_index;
		visited[max_char-'a']=true;
	}
	
}
