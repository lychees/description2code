#include <iostream>
#define ll long long
using namespace std;

int main()
{
	ll t, c, q, i, j, multi, sum;
    cin>>t;
    while(t--)
    {
    	c=0;
    	cin>>q;
    	for(i=1;i<=q;i++)
    	{
    		for(j=1;j<i;j++)
    		{
    			multi=i*j;
    			sum=i+j;
    			if(multi%sum==0)
    				c++;
    		}
    	}
    	cout<<c<<endl;
    }
    return 0;
}
