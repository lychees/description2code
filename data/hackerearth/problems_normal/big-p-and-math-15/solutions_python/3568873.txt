ab = raw_input().split(' ')
a_values = []
b_values = []
sums = []
a_values.append(int(ab[0]))
b_values.append(int(ab[1]))
a_values.append(int(ab[0].replace('5', '6')))
a_values.append(int(ab[0].replace('6', '5')))
b_values.append(int(ab[1].replace('5', '6')))
b_values.append(int(ab[1].replace('6', '5')))

i = 0
while i < len(a_values):
	j = 0
	while j < len(b_values):
		sums.append(a_values[i] + b_values[j])
		j += 1
	i += 1

print str(min(sums)) + " " + str(max(sums))
