#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

const int max_k = 20;

ll cnt[max_k];

int main()
{
	ll t, i, j, n, k, maxi, a;

	scanf("%lld", &t);

	while(t--)
	{
		memset(cnt, 0, sizeof(cnt));
		maxi = 0;

		scanf("%lld %lld", &n, &k);

		for(i=0;i<n;++i)
		{
			scanf("%lld", &a);
			++cnt[a];
			maxi = max(maxi, cnt[a]);
		}

		printf("%lld\n", n-maxi);

	}	

	return 0;
}
