'''

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
for i in range(int(raw_input())):
    A, B, C, K = map(int, raw_input().split())
    low, high = 0, 100000
    answer = 0
    while low <= high:
        mid = (low + high) / 2
        poly = A * mid ** 2 + B * mid + C
        if poly >= K:
            answer = mid
            high = mid - 1
        else:
            low = mid + 1
    print answer
