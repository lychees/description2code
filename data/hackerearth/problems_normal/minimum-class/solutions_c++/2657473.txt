#include<iostream>
#include<cmath>
using namespace std;

int main()
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int s=n-1;
		int no=1;
		int pre=1;
		while(s!=0)
		{
			if((((pre+1)*(pre+2))/2)<=s)
			{
				pre=pre+1;
				no=no+1;
				s=s-pre;
			}
			else if((((pre+1)*pre)/2)<=s)
			{
				no=no+1;
				s=s-pre;
			}
			else
			{
				no=no+1;
				pre=pre-1;
				s=s-pre;
			}
		}
		cout<<no<<endl;
	}
	
}
