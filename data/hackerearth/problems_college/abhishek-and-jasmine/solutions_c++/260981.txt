#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	static long long arr[1000010];
	int count[70],answer[70];
	for (int i=0;i<70;i++)
	{
		answer[i]=count[i]=0;
	}
	cin>>n;
	for (int i=0;i<n;i++)
	{
		cin>>arr[i];
		long long x = arr[i];
		int pos = 0;
		while(x!=0)
		{
			if((x&1) == 1)
				{
					//cout<<"sdf "<<pos<<endl;
					count[pos]++;
				}
			x=x>>1;
			pos++;
		}
	}
	for (int i=0;i<n-1;i++)
	{
		long long x = arr[i];
		int pos = 0;
		for (pos=0;pos<65;pos++)
		{
			if(((x&1) == 1))
				count[pos]--;
			if(((x&1) == 1) && count[pos]<(n-i-1))
			{
				answer[pos] = 1;
			}
			if(((x&1) == 0) && count[pos]>0)
				answer[pos] = 1;
			x=x>>1;
		}
	}
	long long ans = 0;
	for (int i=70;i>=0;i--)
	{
		ans = ((ans<<1) | answer[i]);
	}
	cout<<ans<<endl;
	return 0;
}
