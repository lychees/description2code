#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
   int t,n,k;
   long long count;
   cin>>t;
   while(t--)
   {cin>>n>>k;
   count =0;
   int a[n];
   for(int i=0;i<n;i++)
   cin>>a[i];
   sort(a,a+n);
   if(k>n)
  { for(int i=n-1;i>=0;i--)
   count += a[i];
  }
   else
   {for(int i=n-1;i>=(n-k);i--)
   count += a[i];
   }

   cout<<count<<"\n";
   }

    return 0;
}
