#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define d1(a)cout<<#a<<": "<<a<<"\n";
#define d2(a,b)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<"\n";
#define d3(a,b,c)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<" , "<<#c<<": "<<c<<"\n";
#define d4(a,b,c,d)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<" , "<<#c<<": "<<c<<" , "<<#d<<": "<<d<<"\n";

ll mod = 1e9 +7;

ll a[55];

ll powmod(ll a, ll b){
	ll ans = 1;
	while(b!=0){
		if(b%2 == 1){
			ans = (ans*a)%mod;
		}
		a = (a*a)%mod;
		b = b/2;
	}
	return ans;
}

ll gcd(ll a,ll b){
	if(b == 0){
		return a;
	}
	else{
		return gcd(b, a%b);
	}
}

int main() {
	int n;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	ll d = a[0];
	for(int i=1;i<n;i++){
		d = gcd(d, a[i]);
	}
	ll ans = 1;
	for(int i=0;i<n;i++){
		ll x = powmod(a[i], d);
		ans = (ans*x)%mod;
		if(ans < 0){
			ans += mod;
		}
	}
	cout<<ans;
	return 0;
}
