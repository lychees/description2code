#include<iostream>
#include<set>
using namespace std;
int root(int [],int);
void uni(int [],int [],int,int,int);
bool find(int [],int,int);
int main()
{
	int n,m,i;
	cin>>n>>m;
	set<int>str;
	int a[n+1],s[n+1];
	for(i=1;i<=n;i++)
	{
		a[i]=i;
		s[i]=1;
	}
	while(m--)
	{
		int p,b;
		cin>>p>>b;
		uni(a,s,p,b,n);
	}
	for(i=1;i<=n;i++)
	str.insert(root(a,i));
	cout<<str.size()<<"\n";
	return 0;
}
int root(int a[],int i)
{
	while(i!=a[i])
	i=a[i];
	return i;
}
bool find(int a[],int i,int j)
{
	int x,y;
x=root(a,i);
 y=root(a,j);
	if(x==y)
	{
		return true;
	}
	return false;
}
void uni(int a[],int s[],int i,int j,int n)
{
	int x,y;
	if(!find(a,i,j))
	{
	 x=root(a,i);
	 y=root(a,j);
		if(s[x]>s[y])
		{
			a[y]=x;
			s[x]+=s[y];
		}
	
	else
	{
		a[x]=y;
		s[y]+=s[x];
	}
	}
}
