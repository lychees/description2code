#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		int flag=0;
		int l=s.length();
		for(int i=l-1;i>0;i--)
		{
		     if(s[i-1]>=s[i])
			 continue;
			 else
			 {
			 	flag=1;
			 	int j=l-1;
			 	while(j>=i&&s[j]<=s[i-1])
			 	j--;
			 	//sw9ap j&i-1;
			 	char m=s[j];s[j]=s[i-1];s[i-1]=m;
			 	//cout<<s<<"\n";break;
			 	string sub1=s.substr(0,i);
			 	string sub2=s.substr(i,l-i);
			 	sort(sub2.begin(),sub2.end());
			 	s="";
			 	s.append(sub1);
			 	s.append(sub2);
			 	break;
				 }	
		}
		if(flag==1)cout<<s<<endl;
		else cout<<"no answer\n";
	}
}
