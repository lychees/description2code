#include <iostream>
#define MOD 10000000007;
#include <map>
#include <string>
using namespace std;

int main()
{
	std::map<char, int> _map;
	int t;
	cin >> t;
	while(t--){
		_map.clear();
		std::string str1, str2;
		cin >> str1;
		cin >> str2;
		int length = str1.length();
		int i = 0;
		while(i<length){
			// cout << str1[i];
			_map[str1[i]]++;
			i++;
		}
		length = str2.length();
		i = 0;
		bool flag = false;
		while(i<length){
			if(_map[str2[i]] > 0){
				cout << "Yes" << endl;
				flag = true;
				break;
			}
			i++;
		}
		if(flag == false){
			cout << "No" << endl;
		}
	}
    return 0;
}
