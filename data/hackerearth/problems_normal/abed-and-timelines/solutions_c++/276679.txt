#include<iostream>
using namespace std;
long long min(long long a,long long b){
    return a<b?a:b;
}    
int main(){
    int tc;
    cin>>tc;
    while(tc--){
        long long n,m,x1,x2,y1,y2,p,row,col,check;
        cin>>n>>m>>x1>>y1>>x2>>y2>>p;
        x1++;y1++,x2++;y2++;
        if(x1>x2){
        row=min((n-x1+x2),(x1-x2));
        }
        else{
        row=min((n-x2+x1),(x2-x1));
        }        
        if(y1>y2){
        col=min((m-y1+y2),(y1-y2));
        }
        else{
        col=min((m-y2+y1),(y2-y1));
        }        
        //cout<<row<<"\t"<<col<<endl;
        check=(row+col)*p;
        cout<<check<<endl;
        if(check<=1000){
            cout<<"You saved the group."<<endl;
        }    
        else{
            cout<<"Let go of the group."<<endl;
        }    
    }    
    return 0;
}
