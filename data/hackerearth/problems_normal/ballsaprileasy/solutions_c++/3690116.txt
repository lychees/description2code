lli gcd(lli a,lli b) {
	if (b == 0) return a;
	return(a % b == 0 ? b : gcd(b, a % b)); 
}
int main() {
	int n;
	cin >> n;
	lli a[n];
	rep(i, 0, n) {
		cin >> a[i];
	}
	sort(a,a+n);
	lli ct = 1;
	int d = 0;
	lli ans = 0;
	while (d < n) {
		lli val = ct*ct;
		if (val < a[d]) {
			ans += val;
		}
		else {
			ans += (val-a[d]);
			d++;
		}
		ct++;
	}
	cout << ans << endl;
	return 0;
}
