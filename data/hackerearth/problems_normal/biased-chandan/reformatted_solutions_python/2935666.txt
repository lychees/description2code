def main():
    n = int(raw_input())
    stack = []
    i = 0
    while (i < n):
        rate = int(raw_input())
        if rate:
            stack.append(rate)
        else:
            if len(stack) > 0:
                stack.pop()
        i += 1
    print sum(stack)


main()
