#include <bits/stdc++.h>
using namespace std;

int A[20][20];

int main()
{
    int T;
    scanf("%d", &T);
    
    while(T--)
    {
    	int N, M;
    	scanf("%d %d", &N, &M);
    	
    	memset(A, 0, sizeof(A));
    	
    	while(M--)
    	{
    		int x, y;
    		scanf("%d %d", &x, &y);
    		
    		x--, y--;
    		A[x][y] = A[y][x] = 1;
    	}
    	
    	int ans = 0;
    	
    	for(int i=0; i<(1<<N); i++)
    	{
    		vector<int> X;
    		
    		for(int j=0; j<N; j++)
    		{
    			if(i&(1<<j))
    				X.push_back(j);
    		}
    		
    		if(X.size())
    		{
    			int flag = 0;
    			
    			for(int k=0; k<X.size(); k++)
    				for(int l=k+1; l<X.size(); l++)
    					if(A[X[k]][X[l]])
    						flag = 1;
    						
    			if(!flag)
    				ans++;
    		}
    	}
    	
    	printf("%d\n", ans);
    }
    
    return 0;
}
