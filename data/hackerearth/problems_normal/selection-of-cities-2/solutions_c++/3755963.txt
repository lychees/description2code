#include <bits/stdc++.h>
using namespace std;
#define FOR(i,a,b) for(i=a;i<b;++i)
#define rep(i,n) FOR(i,0,n)
#define MOD 1000000007
typedef long long lln;
int min(int a,int b,int c){ return min(a,min(b,c));}
int max(int a,int b,int c){ return max(a,max(b,c));}

lln Power(lln a,lln b){
   a = a%MOD;
   if(b==0) return 1;
   if(b==1) return a;
   lln Ans=1;
   while(b!=0){
        if(b%2)
            Ans = (Ans*a)%MOD;

        a = (a*a)%MOD;
        b/=2;
   }
   return Ans;
}


int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    lln t,n;
    cin>>t;
    while(t--){
        cin>>n;
        cout<<Power(2,n)-1<<'\n';
    }
    return 0;
}
