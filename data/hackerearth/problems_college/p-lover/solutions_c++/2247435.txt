#include <bits/stdc++.h>
using namespace std;
bool sym(long long int x)
{
	long long int ans=0,u=x;
	while(x)
	{
		ans=(ans*10+x%10);
		x/=10;
	}
	if(u==ans)
	return true;
	return false;
}
bool prime(long long int x)
{
	for(int i=2;i<=sqrt(x);i++)
	  if(x%i==0)
	   return false;
	return true;
}
int main()
