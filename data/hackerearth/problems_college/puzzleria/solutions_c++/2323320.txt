#include<bits/stdc++.h>
using namespace std;
long long fact(int n)
{
  if (n == 0)
    return 1;
  else
    return(n * fact(n-1));
}
int main()
{
int t;
long long int ans;
cin>>t;
while(t--)
{
    int m,n;
cin>>m>>n;
ans=(fact(7)*fact(5)*fact(m+n))/(fact(7-m)*fact(5-n)*fact(m)*fact(n));
cout<<ans<<endl;
}
}
