#include <iostream>
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	ll a,b;
    	cin>>a>>b;
    	ll h=max(a,b);
    	ll w=min(a,b);
    	ll ini=0,count=0;
    //	cout<<h<<w<<endl;
    	for(ll i=0;i<w;i+=2)
    	{
    		for(ll j=ini;j<h;j+=2)
    		{
	    		if(j+2<=h && i<=w-2)	{count+=4;j+=1;}
	    		else if(j+2<=h) {count+=2;j++;}
	    		else if(j+1<=h && i<=w-2)	count+=2;
	    		else if(j+1<=h)	count+=1;
	    		j++;
    		}
    		if(ini==0)	ini+=2;
    		else ini-=2;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
