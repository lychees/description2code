#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    int t;
    cin>>t;
    for(;t--;)
    {
    	long long int u;
    	scanf("%lld",&u);
    	long long int a,b,c,i,j,k,flag=0;
    	a=u/6;
    	b=u/9;
    	c=u/20;
    	for(i=0;i<a;i++)
    	{
    		for(j=0;j<b;j++)
    		{
    			for(k=0;k<c;k++)
    			{
    				if(6*i+9*j+20*k==u)
    				{
    					cout<<("True\n");
    					flag=1;
    					break;
    				}
    			}
    			if(flag==1)
    			break;
    		}
    		if(flag==1)
    		break;
    	}
    	if(flag==0)
    	cout<<("False\n");
    }
    return 0;
}
