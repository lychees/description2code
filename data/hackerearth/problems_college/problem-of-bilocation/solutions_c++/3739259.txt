#include<iostream>
using namespace std;
int main()
{
    int a,b,c,d,s,t;
    cin>>t;
    while(t--)
    {
       cin>>a>>b>>c>>d>>s;
       int k=c-s;
       if(k<a)
            cout<<"STAY IN LONDON"<<endl;
        else if(k>b)
            cout<<b<<endl;
        else
            cout<<k<<endl;
    }

}
