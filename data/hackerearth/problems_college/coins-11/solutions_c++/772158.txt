#include <iostream>
using namespace std;

int main()
{
    bool a[100002]={0};
    int i,j;
    for(i=4;i<=100002;i+=2)
    a[i]=1;
    for(i=3;i<320;i++)
    {	if(a[i]==1)
    	continue;
    	for(j=i;j*i<100002;j++)
    	a[i*j]=1;
    }
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	if(a[n]==1 || n<2)
    	printf("1\n");
    	else
    	{
    		if(n==3)
    		printf("3\n");
    		else
    		printf("2\n");
    	}
    }
    return 0;
}
