#define INF 2000000000 // 2 * 10^9
#define INFLL 1000000000000000000ll  // 10^18
#define mod 1000000007

#define N 212345

int rs[100],ppw[100],isp[100];

LL fn(LL n)
{
    if(n<100)
    {
        LL frs = 0;
        for(LL i=1;i<=n;++i)
            frs += rs[i];
        return frs;
    }
    memset(ppw,0,sizeof(ppw));
    LL a=1;
    LL tmp;
    LL frs = 2,pi,j;
    for(LL i=2;i<100;++i)
    {
        for(j=2;j<=i;++j)
        {
            if(isp[j]==1 && i%j==0)
            {
                pi = i;
                while(pi%j==0)
                {
                    pi/=j;
                }
                break;
            }
        }
        if(pi==1)
        {
            tmp = (n/a) - (n/(a*j));
            //cout  << " i: " << i << " ymp: " << tmp << endl;
            if(a <= i)
            {
                tmp -= ((i)/a - (i/(a*j)));
            }
            frs += (tmp*(rs[i]+1));
            //cout <<" i: " << i <<  " tmp: " << tmp << " frs: " << frs << endl;
            a*=j;
        }
        if(a>INFLL) break;
    }
    return frs;
}

void solve()
{
    LL i,j;
    for(i=2;i<100;++i)
    {
        isp[i] = 1;
        for(j=2;j<i;++j)
        {
            if(i%j==0)
            {
                isp[i] = 0;
                break;
            }
        }
    }
    rs[0] = 0;
    rs[1] = 1;
    rs[2] = 1;
    for(i=3;i<100;++i)
    {
        rs[i] = 1;
        for(j=2;j<i;++j)
        {
            if(i%j!=0)
            {
                rs[i] += rs[j];
                break;
            }
        }
    }
    LL a,b;
    slld(a);
    slld(b);
    //cout << " fn("<<b<<"): " << fn(b) << endl;
    printf("%lld\n",fn(b)-fn(a-1));
}

int main()
{
    int t = 1;
    //scanf("%d",&t);
    while(t--)
    {
        solve();
    }
}
