void fun()
{
	

    int sum;
	sum = 0;
	int n;
	scanf("%d", &n);
	
vector<int>remain(n);
vector<int>capacity(n);
vector<int>cum(n);
	//remain.resize(n);
	//capacity.resize(n);
	//cum.resize(n);
	for(int i = 0; i < n; i++) {
		//scanf("%d", &remain[i]);
		cin>>remain[i];
		sum += remain[i];
	}
	for(int i = 0; i < n; i++) {
		//scanf("%d", &capacity[i]);
		cin>>capacity[i];
	}
	sort(capacity.begin(), capacity.end());
	reverse(capacity.begin(), capacity.end());
	cum[0] = capacity[0];
	for(int i = 1; i < n; i++) {
		cum[i] = cum[i - 1] + capacity[i];
	}
	for(int i = 0; i < n; i++) {
		if(cum[i] >= sum) {
			printf("%d\n", i + 1);
			
			return;
		}
	}
	
}

int main()
{
	int t;
	scanf("%d", &t);
	while(t--) {
		fun();
	}
	return 0;
}
