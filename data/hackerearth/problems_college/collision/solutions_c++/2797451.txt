#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a;
		int b[10];
		for(int i=0;i<10;i++)
		{
			b[i]=0;
		}
		for(int i=0;i<n;i++)
		{
			cin>>a;
			b[a%10]++;
		}
		int ans=0;
		for(int i=0;i<10;i++)
		{
			if(b[i]>1)
			{
				ans=ans+(b[i]-1);
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
