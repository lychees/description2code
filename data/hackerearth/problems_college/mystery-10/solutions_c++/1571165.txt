#include<bits/stdc++.h>
using namespace std;
string ans={"qwertyuiopasdfghjklzxcvbnm"};
int main()
{
    int t;
    scanf("%d\n",&t);
    while(t--)
    {
    	int i,n;
    	char s[1000];
    	gets(s);
    	n=strlen(s);
    	for(i=0;i<n;i++)
    	{
    		if(s[i]==' ')
    			cout<<" ";
    		else
    			cout<<ans[s[i]-'a'];
    	}
    	cout<<endl;
    }
    return 0;
}
