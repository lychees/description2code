N = int(raw_input())
val = [[0]*(2**N+1) for i in range(2**N+1)]
for i in range(1, 2**N ):
	nums = [int(j) for j in raw_input().split()]
	for j in range(len(nums)):
		val[i+1][j+1] = nums[j]
	
#print val

cur = []
prev = [i for i in range(1, 2**N+1)]

for i in range(N, 0, -1):
	l = len(prev)
	for j in range(0, l, 2):
		if val[prev[j+1]][prev[j]] == 1:
			cur.append(prev[j+1])
		else:
			cur.append(prev[j])
		#print j, prev[j], prev[j+1]
	prev = cur
	cur = []
