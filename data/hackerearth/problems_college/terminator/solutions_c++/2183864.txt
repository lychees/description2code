#include <iostream>
#include <stack>
#include <algorithm>
#include <cstring>
using namespace std;

int main()
{
	int tc,i;
	char str[100];
	//get the number of testcases
	cin>>tc;
	while(tc--)
	{
		memset(str,'\0',sizeof(str));
		stack<char>stck;
		stack<char>stck2;
		int len=0;
		//get the string
		cin>>str;
		len=strlen(str);
		for(i=0;i<len;i++)
		{
			switch(str[i])
			{
				case '*':stck.push('*');
						break;
				case 'x':while(!stck.empty()&&stck.top()=='o')
							stck.pop();
						stck.push('x');
						break;
				case 'o':if(stck.empty())
							{stck.push('o');
							break;
							}
							
						else if(!stck.empty()&&stck.top()!='x')
						stck.push('o');
						break;
				default:cout<<str[i];			
			}
		}
		
		//print the content of stack in reverse order
		while(!stck.empty())
		{
			stck2.push(stck.top());
			stck.pop();
		}
		while(!stck2.empty())
		{
			cout<<stck2.top();
			stck2.pop();
		}
		cout<<endl;
}

return 0;
}
