int main()
{
	int t;
	scan(t);
	wl(t)
	{
		string s1,s2;
		cin>>s1>>s2;
		int n = s1.size();
		int m = s2.size();
		int i,j,car=0,ans=0;
		for(i=n-1, j=m-1;i>=0 && j>=0;)
		{
			int x = s1[i]-'0';
			int y = s2[i]-'0';
			if(x+y+car>9)
			{
				car = 1;
				ans++;
			}
			else
				car = 0;
			i--;
			j--;
		}
		if(ans==0)
			printf("No carry operation\n");
		else if(ans==1)
			printf("%d carry operation\n",ans);
		else
			printf("%d carry operations\n",ans);
	}	
	return 0;
}
