using namespace std;
bool isMoody[163];
bool eval[163];
bool moodCheck(ull i)
{
    if(eval[i]) return isMoody[i];
    else
    {
        ll n = i*i;
        ll tot = 0;
        while(n > 0)
        {
            tot += n%10;
            n/=10;
        }
        bool ans = moodCheck(tot);
        if(i < 163)
        {
            eval[i] = true;
            isMoody[i] = ans;
        }
        return ans;
    }
}

    isMoody[0] = false;

    eval[9] = true;
    //eval[13] = true;
    eval[16] = true;
    int i = 0;
    iter(i,0,163,1) isMoody[i] = moodCheck(i);
}
int main()
{
