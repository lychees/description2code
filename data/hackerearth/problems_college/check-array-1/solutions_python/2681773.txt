noc = input()
while noc:
    noc -= 1
    n = input()
    ar= map(int, raw_input().split())
    br= ar[:]
    for x in range(1, n-1):
        br[x]+= br[x-1]
    cr = ar[:]
    for x in range(n-2, -1, -1):
        cr[x] += cr[x+1]
    ans = "NO"    
    for x in range(1, n-1):
        if br[x-1] == cr[x+1]:
            ans = "YES"
            break
    print ans
