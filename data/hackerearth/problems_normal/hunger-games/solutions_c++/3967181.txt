#include<bits/stdc++.h>
#define ll long long 
using namespace std;
int main()
{
	ll n;
	cin>>n;
	ll a[n+1];
	memset(a,0,sizeof(a));
	for(int i=0;i<n;i++)
		cin>>a[i];
	sort(a,a+n);
	ll ans=0;
	for(int i=0;i<n-2;i++)
	{
		ans=max(ans,abs(a[i]-a[i+1]));
		ans=max(ans,abs(a[i]-a[i+2]));
	}
	cout<<ans<<endl;
	return 0;
}
