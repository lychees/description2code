#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    ios::sync_with_stdio(false);
    int n,x,y;
    cin>>t;
    while(t--)
    {
        cin>>n>>x>>y;
        int ar[n],arr[n];
        for(int i=0;i<n;i++)
        {
            cin>>ar[i]>>arr[i];
        }
        sort(ar,ar+n);
        sort(arr,arr+n);
        int sum = 0;
        for(int i=0;i<n;i++)
        {
            if(arr[i]>ar[i])
                sum+=x*(arr[i]-ar[i]);
            else
                sum+=y*(ar[i]-arr[i]);
        }
        cout<<sum<<endl;

    }

}
