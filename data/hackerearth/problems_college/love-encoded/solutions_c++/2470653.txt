#include <iostream>
#include <cstdio>
#include <string>
using namespace std;

int main()
{
	int t,n,i,k;
	string a;
	cin>>t;
	while(t--)
	{
		cin>>n;
		n=n%26;
		cin>>a;
		for(i=0;i<a.length();i++)
		{
			k=int(a[i]);
			k-=n;
			if(k<65)
			{
				a[i]=a[i]+26-n;

			}
			else
			{
				a[i]-=n;
			}
			
		}
		cout<<a<<endl;
	}
	return 0;
}
