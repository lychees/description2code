#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	int a[100050];
	int las[100050];
	
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	las[0]=1;
	int max=0;
	for(int i=1;i<n;i++)
	{
		 max=0;
		for(int j=0;j<i;j++)
		{
			if(abs(a[i])>abs(a[j]))
			{
				if((a[j]>0&&a[i]<0)||(a[i]>0&&a[j]<0))
				{
					if(las[j]>=max)
					max=las[j];
				}
			}
		}
		las[i]=max+1;
	}
	max=0;
	for(int i=0;i<n;i++)
	{
		if(las[i]>=max)
		max=las[i];
	}
	printf("%d\n",max);
	return 0;
}
