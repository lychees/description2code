using namespace std;
bool operator < (const pair<int, int>&i, const pair<int, int>&j)

{

    return i.first < j.first;

}
int main(){
    int n,a,b;
    vector<pair<int,int>> v;

    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>a>>b;
         v.push_back(make_pair(a+b,i));
    }
    sort(v.begin(),v.end());

   for(int i=0;i<n;i++)
   {
    cout<<v[i].second<<" ";
   }
    return 0;
}
