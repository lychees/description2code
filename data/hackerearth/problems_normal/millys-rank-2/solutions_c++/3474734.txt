#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int T;
    long long N,X,Y;
    cin>>T;
    while(T) {
    	cin>>N>>X>>Y;
    	long long d1 = X+1;
    	long long d2 = N-Y;
    	if(d1<=d2) {
    		cout<<min(Y+1,N)<<endl;
    	}
    	else {
    		cout<<N-X<<endl;
    	}
    	T--;
    }
    return 0;
}
