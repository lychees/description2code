#include <iostream>

using namespace std;

int main()
{
    int t,i;
    cin>>t;
    for(i=0;i<t;i++)
    {
        int d,wl;
        cin>>d>>wl;
        if(d>60)
        {
            cout<<"NA\n";
        }
        else
        {
        if(d<=2)
        {
            cout<<(50+2*wl)<<"\n";
        }
        else
        {
            cout<<(50+(d-2)*12+2*wl)<<"\n";
        }
        }
    }
}
