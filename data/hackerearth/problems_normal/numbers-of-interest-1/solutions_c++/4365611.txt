#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,x,y,n;
    cin>>t;
    while(t--){
    	cin>>x>>y>>n;
    	if(n<=x){
    		cout<<y<<endl;
    		continue;
    	}
    	int rank = x+1;
    	long long int arr[x] ;
    	long long int sum = x*y;
    	for(int i=0;i<x;i++) arr[i]=y;
    	int i=0;
    	long long int next;
    	while(rank<=n){
    		 next = sum;
    		 sum=sum+next-arr[i%x];
    		 arr[i%x]=next;
    		 i++;
    		 rank++;
    	}
    	cout<<next<<endl;
    }
    return 0;
}
