#include <iostream>
using namespace std;

int main()
{
	long t;
    cin>>t;
    
    while(t--)
    {
    	long n;
    	cin>>n;
    	
    	long arr[n];
    	for(long i = 0 ; i < n ; i++)
    		cin>>arr[i];
    		
    	if(n > 1)
    		cout<<"0\n";
    		
    	else
    		cout<<arr[0]<<endl;
    }
    return 0;
}
