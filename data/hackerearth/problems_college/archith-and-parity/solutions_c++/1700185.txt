#include <iostream>
using namespace std;

int main()
{
    int c,n,i,ar[100],r;
    long int k;
    cin>>n;
    for(i=0;i<n;i++)
    {cin>>k;
    r=0;
    c=0;
    while(k)
    {
    r=k%2;	
    if(r==1)
    {
    	c++;
    }
    k=k/2;
    }
    ar[i]=c;
    }
    for(i=0;i<n;i++)
    {
    	if(ar[i]%2==0)
    	cout<<"even\n";
    	else
    	cout<<"odd\n";
    }
    return 0;
}
