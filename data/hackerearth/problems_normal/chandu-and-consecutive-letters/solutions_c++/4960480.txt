#include <iostream>
using namespace std;

int main()
{
    int t=0;
    cin>>t;
    while(t>0){
    	string s;
    	cin>>s;
    	string goodString="";
    	goodString=goodString+s[0];

    	//cout<<goodString<<endl;
    	int l=s.length();
    	int k=0;
    	
    	for(int i=1;i<l;i++){
    		//cout<<s[i]<<" ";
    		//cout<<goodString<<endl;
    		if(s[i]!=goodString[k]){
    			goodString=goodString+s[i];
    			k++;
    		}
    	}
    	cout<<goodString<<endl;
    	t--;
    }
    return 0;
}
