//0-1 Knapsack

#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	int i,j,n,maxWt;
	cin>>n;
	cin>>maxWt;
	vector<int> wt(n+1),value(n+1);
	for(i=1;i<=n;i++)
	 {
	 	cin>>wt[i];
	 }
	 for(i=1;i<=n;i++)
	 {
	 	cin>>value[i];
	 }
	vector<vector<int> > knapsack(n+1,vector<int>(maxWt+1));
	for(i=1;i<=n;i++)
	 {
	 	for(j=1;j<=maxWt;j++)
	 	 {
	 	 	if(j-wt[i]>=0)
	 	 	  {
	 	 	  	knapsack[i][j]=max(value[i]+knapsack[i-1][j-wt[i]],knapsack[i-1][j]);
	 	 	  }
	 	 	else
	 	 	  {
	 	 	  	knapsack[i][j]=knapsack[i-1][j];
	 	 	  }
	 	 }
	 }
	 cout<<knapsack[n][maxWt]<<endl;
}
}
