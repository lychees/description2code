int main()
{
	int n,i;
	cin>>n;
	vector<int> v(n);
	for(i=0;i<n;i++)
	{
		cin>>v[i];
	}
	vector<int>::iterator it,t;
	int ans=0;
   int  maxi=0;
    while(n>2)
    {
    	maxi=0;
        for(it=v.begin()+1;it<v.end()-1;it++)
        {
          if(*(it-1)*(*(it+1))>maxi)
          {
              maxi=(*(it-1))*(*(it+1));
              t=it;

          }
    }
    ans+=maxi;
    v.erase(t);
    n--;
    }
	cout<<ans<<"\n";
	return 0;
	
}
