#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int a[n];
		int i=0;
		long long int sum=0;
		long long int min;
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			sum+=a[i];
			if(i==0)
			min=a[i];
			else if(min>a[i])
			min=a[i];
			
		}
		
		sum+=(n-2)*min;
		printf("%lld\n",sum);
		
	}
	return 0;
	
}
