#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    for(int i=4;i<=n;i++)
    {
    	if(i%2==0)
    	{
    		cout<<i<<" ";
    	}
    	else
    	{
    		for(int j=3;j<=n/2;j=j+2)
    		{
    			if(i%j==0 && i!=j)
    			{
    				cout<<i<<" ";
    				j = n/2+1;
    				//break;
    			}
    			//	cout<<i<<" ";
    			
    		}
    	}
    }
    return 0;
}
