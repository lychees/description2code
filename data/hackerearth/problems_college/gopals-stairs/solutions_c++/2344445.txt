#include <stdio.h>


void findways(long long int n)
{
	long long int arr[n+1];
	long long int i,result;
	arr[1]=1,arr[2]=2;
	arr[3]=4;
	if(n==1)
  {	printf("1\n");
     return;
  }
	if(n==2)
	{printf("2\n");
	 return;
	}
	 if(n==3)
	 {
	 printf("4\n");
	 return;
	 }
	for(i=4;i<=n;i++)
	{
		arr[i]=arr[i-1]%1000000007+arr[i-2]%1000000007+arr[i-3]%1000000007;
	}
	result=arr[n];
	result=result%1000000007;
	printf("%lld\n",result);
	
}


int main()
{ long long int n,t;
  scanf("%lld",&t);
  while(t--)
  {
  	scanf("%lld",&n);
  	findways(n);
  	
  	
  }
    
    return 0;
}
