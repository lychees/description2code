
/*  https://www.hackerearth.com/problem/algorithm/snack-ruiner/   */

#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
        int i, n, index = 0;
        float d[30], f[30], div = 0, ans = 0;
        cin >> n;
        for(i=0; i<n;i++)
            cin >> d[i];
        for(i=0; i<n;i++)
            cin >> f[i];
        for(i=0; i<n; i++)
            div += d[i]*f[i];
        ans = 0.000;
        for(i=0; i<n; i++){
            if(d[i]*f[i] > ans)
                index = i;
            ans = max(ans, d[i]*f[i]);
        }
        cout << index << " " << setprecision(3) << fixed << ans/div << endl;
    }
    return 0;
}
