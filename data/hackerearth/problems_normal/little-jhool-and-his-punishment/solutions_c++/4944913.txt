#include <iostream>
#include <algorithm>
#include <cstdlib>

using namespace std;

int main()
{
    long T;
    cin >> T;
    
    for(long t = 0; t < T; t++)
    {
    	long n, b, g;
    	cin >> n >> b >> g;
    	
    	long x = min(b, g);
    	long y = max(b, g);
    	
    	if(x < y - 1)
    		cout << "Little Jhool wins!" << endl;
    	else
    		cout << "The teacher wins!" << endl;
    }
    return 0;
}
