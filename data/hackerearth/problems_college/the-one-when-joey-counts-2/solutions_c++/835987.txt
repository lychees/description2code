#include<iostream>
using namespace std;

int main(){
	long t,a,i;
	cin>>t;
	while(t--){
		i=1;
		cin>>a;
		while(a>3*i){
			a=a-3*i;
			i*=3;
		}
		while(i>0){
			if(a==0)
				cout<<4;
			else if(a<=i)
				cout<<2;
			else if(a<=i*2)
				cout<<3;
			else
				cout<<4;
			a%=i;
			i/=3;
		}
		cout<<endl;
		
	}
}
