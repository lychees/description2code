#define per(i,a,n) for (int i=n-1;i>=a;i--)

const ll mod=1000000007;

// head

const int N=2010000;
int _,n,m,f[N],sz[N],top,st[N];
char s[N];
int find(int x) { top=0; while (f[x]!=x) st[top++]=x,x=f[x]; rep(i,0,top) f[st[i]]=x; return x;}
int query(int x,int y) { if (x>=0&&y>=0&&x<n&&y<m) return s[x*m+y]; else return 0;}
void add(int x,int y) { f[find(x)]=find(y);}
int main() {
	for (scanf("%d",&_);_;_--) {
		scanf("%d%d",&n,&m);
		rep(i,0,n) scanf("%s",s+i*m);
		rep(i,0,(n+1)*(m+1)) f[i]=i,sz[i]=0;
		rep(i,0,n) rep(j,0,m+1)
			if (query(i,j)!=query(i,j-1)) add(i*(m+1)+j,(i+1)*(m+1)+j);
		rep(i,0,n+1) rep(j,0,m)
			if (query(i,j)!=query(i-1,j)) add(i*(m+1)+j,i*(m+1)+j+1);
		int cnt=0;
		rep(i,0,(n+1)*(m+1)) sz[find(i)]++;
		rep(i,0,(n+1)*(m+1)) if (sz[i]>=2) cnt++;
		puts(cnt>=2?"NO":"YES");
	}
}
