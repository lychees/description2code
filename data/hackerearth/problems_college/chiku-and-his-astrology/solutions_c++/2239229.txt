#include <bits/stdc++.h>
using namespace std;

int main(){
	int t;cin>>t;
	while(t--){
		char a[200005];
		cin>>a; int l=strlen(a);
		int c1=0,c2=0;
		for(int i=0;i<l;i++){
			if(a[i]=='0') c2++;
			else c1++;
		}
		if(c1>c2) cout<<c1-c2<<endl;
		else cout<<c2-c1<<endl;
    }
    return 0;
}
