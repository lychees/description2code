#include<iostream>
#include<bits/stdc++.h>

using namespace std;
int v[2000000];
int main()
	{
		int t;
		double fi=((1+sqrt(5))/2);
		for(int i=1;i<=1000000;++i)
			v[i]=-1;
		for(int i=1;i<=1000000;++i)
			v[(int)(fi*(double)i)] = (int)(fi*fi*i);
		cin>>t;
		while(t--)
			{
				
				int a,b;
				scanf("%d %d",&a,&b);
				if(v[a]==b)
					cout<<"Chandu"<<endl;
				else
					cout<<"Chandni"<<endl;
			}
		return 0;
	}
