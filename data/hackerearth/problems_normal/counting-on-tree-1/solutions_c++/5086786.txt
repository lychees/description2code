#include<bits/stdc++.h>
using namespace std;;
#define f first
#define s second
#define ll long long
#define mp make_pair
#define mod 1000000007LL
#define pii pair<int,int>
#define pb push_back
vector<int>adj[200011];
int a[200011],k,sz[200011];
ll dp[200011][110],temp[2][110];
void dfs(int u,int p)
{
    sz[u]=1;
    for(int i=0; i<adj[u].size(); i++)
    {
        int k1=adj[u][i];
        if(k1==p)
            continue;
        dfs(k1,u);
        sz[u]+=sz[k1];
    }
    int c=1;
    for(int i=0; i<=min(k,sz[u]); i++)
        temp[0][i]=temp[1][i]=0;
    temp[0][a[u]]=1;
    for(int i=0;i<adj[u].size();i++)
    {
        int k1=adj[u][i];
        if(k1==p)
            continue;
        for(int j=0; j<=k; j++)
            temp[c][j]=temp[!c][j];
        for(int j=0; j<=min(k,sz[k1]); j++)
        {
            for(int l=j; l<=min(k,sz[u]); l++)
                temp[c][l]=(temp[c][l]+temp[!c][l-j]*dp[k1][j])%mod;
        }
        c=!c;
    }
    for(int i=0; i<=min(k,sz[u]); i++)
        dp[u][i]=temp[!c][i];
//    cout<<u<<endl;
//    for(int i=0; i<=k; i++)
//        cout<<dp[u][i]<<" ";
//    cout<<endl;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int i,j,n;
        scanf("%d %d",&n,&k);
        for(i=0;i<=n;i++)
        {
            for(j=0;j<=k;j++)
                dp[i][j]=0LL;
            adj[i].clear();
        }
        for(i=1; i<=n; i++)
            scanf("%d",&a[i]);
        for(i=1; i<n; i++)
        {
            int u,v;
            scanf("%d %d",&u,&v);
            adj[u].pb(v);
            adj[v].pb(u);
        }
        dfs(1,0);
        ll ans=0LL;
        for(int i=1; i<=n; i++)
            ans=(ans+dp[i][k])%mod;
        printf("%lld\n",ans);
    }
    return 0;
}
