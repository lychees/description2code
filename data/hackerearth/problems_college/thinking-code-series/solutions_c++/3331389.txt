#include <iostream>
using namespace std;

int main()
{
    int t,l,p,c,count,d=1,i;
    cin>>t;
    while(t--)
    {
    	cin>>l>>p>>c;
    	count=0;
    	
    	while(l<p)
    	{
    		l*=c;
    		count++;
    	}
    	
    	i=0;
    	while((1<<i)<count)
    		i++;
    	cout<<"Case #"<<(d++)<<": "<<i<<"\n";
    }
    return 0;
}
