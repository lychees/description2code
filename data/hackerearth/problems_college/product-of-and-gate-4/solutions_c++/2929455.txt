#include <iostream>
using namespace std;

int main()
{
    long long int t ;
    cin>>t;
    while(t--)
    {
    	long long int a , b;
    	cin>>a>>b;
    	long long x = a^b;
    	long long s = x>>1;

    	while (s) 
    	{
        	x = x|s;
        	s >>= 1;
    	}
    	
    	cout<<(a&b&~x)<<endl;
    }
    return 0;
}
