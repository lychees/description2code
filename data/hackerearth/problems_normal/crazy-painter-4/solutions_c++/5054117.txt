#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n;
    	cin>>n;
    	long long int c[26],sum=0;
    	for(int i=0;i<26;i++)
    	{
    		cin>>c[i];
    		sum+=c[i];
    	}
    	n*=12;
    	long long int rem,quot;
    	quot=n/26;
    	rem=n%26;
    	long long int ans=0;
    	ans+=(quot)*(sum +(quot-1)*13);
    	for(int i=0;i<rem;i++)
    	{
    		ans+=c[i]+quot;
    	}
    	cout<<ans<<"\n";
    }
    return 0;
}
