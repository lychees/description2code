#include <bits/stdc++.h>
using namespace std;
#define lli long long int
int main(){
    int n;
    cin>>n;
    lli a[n],ans=0;
    for(int i=0;i<n;i++)cin>>a[i];
    sort(a,a+n);
    if(n&1){
    	for(int i=n-1;i>n/2;i--){
	    	ans+=a[i];
	    }
	    for(int i=0;i<n/2;i++){
	    	ans-=a[i];
	    }
	    cout<<(ans+ans);
    }
    else{
    	for(int i=n-1;i>n/2-1;i--){
	    	ans+=a[i];
	    }
	    for(int i=0;i<n/2;i++){
	    	ans-=a[i];
	    }
	    cout<<(ans+ans);    	
    }
    return 0;
}
