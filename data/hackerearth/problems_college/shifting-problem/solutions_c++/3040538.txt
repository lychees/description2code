#include<iostream>
#include<stdlib.h>
using namespace std;
long long int fib(long long int n)
{
 long long  int a = 1, b = 2, c, i;
  if( n == 1){
  
    return a;
}
    if(n == 2){
    	return b;
	}
  for (i = 3; i <= n; i++)
  {
     c = a + b;
     a = b;
     b = c;
  }
  return b;
}
int main(){
	long long int t;
	cin>>t;
	while(t--){
		long long int n;
		cin>>n;
		long long int ans=fib(n);
		cout<<ans<<"\n";
	}
	return 0;
}
