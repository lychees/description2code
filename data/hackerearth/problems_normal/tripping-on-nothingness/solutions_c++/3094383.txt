void scanint(ll &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}


ll binary_gcd(ll u, ll v)
{
  ll shl = 0;

  while ( u && v && u!=v ) {
    bool eu = !(u & 1);
    bool ev = !(v & 1);

    if ( eu && ev ) {
      ++shl;
      u >>= 1;
      v >>= 1;
    }
    else if ( eu && !ev ) u >>= 1;
    else if ( !eu && ev ) v >>= 1;
    else if ( u>=v ) u = (u-v)>>1;
    else {
      ll tmp = u;
      u = (v-u)>>1;
      v = tmp;
    }
  }

  return !u? v<<shl : u<<shl;
}

int main()
{
    ll s,q,a,gcd;
    set<ll> b;
    b.clear();
    scanint(s);
    scanint(q);
    while(q--){
    	scanint(a);
    	gcd = binary_gcd(s,a);
    	if(b.find(gcd) == b.end()) {
    		printf("%lld\n",gcd);
    		b.insert(gcd);
    	}	
    	else {
    		cout<<-1<<endl;	
    	}
    }
    
    return 0;
}
