#include <iostream>
using namespace std;

int main()
{
    int h,m;
    char nums[21][10]={"","one","two","three","four","five","six","seven","eight","nine","ten",    "eleven","twelve","thirteen","fourteen","fifteen","sixteen","seventeen","eighteen","nineteen", "twenty"};
    
    cin>>h>>m;
    
    if(m==0)
    	cout<<nums[h]<<" o' clock";
    else if(m==30)
    	cout<<"half past "<<nums[h];    	
    else if(m<30)
    {
    	if(m==1)
    		cout<<nums[1]<<" minute past "<<nums[h];
    	else if(m==15)
    		cout<<"quarter past "<<nums[h];
    	else if(m<=20)
    		cout<<nums[m]<<" minutes past "<<nums[h];
    	else
    	{
    		int m1=m%10;
    		cout<<nums[20]<<' '<<nums[m1]<<" minutes past "<<nums[h];
    	}
    }
    else
    {
    	m=60-m;
    	if(m==1)
    		cout<<nums[1]<<" minute to "<<nums[h+1];
    	else if(m==15)
    		cout<<"quarter to "<<nums[h+1];
    	else if(m<=20)
    		cout<<nums[m]<<" minutes to "<<nums[h+1];
    	else
    	{
    		int m1=m%10;
    		cout<<nums[20]<<' '<<nums[m1]<<" minutes to "<<nums[h+1];
    	}
    }
    return 0;
}
