#include <iostream>
using namespace std;

int main()
{
     int t;
   cin>>t;
   while(t--)
   {
   		int n,m;
   		cin>>n;
   		cin>>m;
   		int x1,y1,x2,y2,cost;
   		cin>>x1;cin>>y1;cin>>x2;cin>>y2;cin>>cost;
   		int r,c;
   		r=abs(x2-x1);
   		if(r>(n-r))
   		r=(n-r);
   		
   		c=abs(y2-y1);
   		if(c>(m-c))
   		c=(m-c);
   		int tc = (r+c)*cost;
   		cout<<tc<<endl;
   		if(tc<=1000)
   		cout<<"You saved the group."<<endl;
   		else
   		cout<<"Let go of the group."<<endl;
   }
}
