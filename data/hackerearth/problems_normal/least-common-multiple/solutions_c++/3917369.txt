#include <bits/stdc++.h>
#define ll long long
#define s 31622778
using namespace std;
bool p[s];
ll int a[1951959];
ll int myfind(ll int x)
{
	ll int i,t=x,mc=0;
	for(i=0;(a[i]*a[i])<=x;i++)
	{
		if((t%a[i])==0)
		{
			mc++;
			if(mc>1)
			break;
			while((t%a[i])==0)
			{
				t/=a[i];
			}
		}
	}
	if(t!=1)
	mc++;
	return mc;
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	//generate primes till s
	ll int i,j;
	p[0]=1;p[1]=1;
	for(i=2;(i*i)<=s;i++)
	{
		if(p[i])
		continue;
		for(j=2;(j*i)<=s;j++)
		{
			p[j*i]=1;
		}
	}
	j=0;
	for(i=0;i<s;i++)
	{
		if(!p[i])
		{
			a[j++]=i;
		}
	}
	//~ for(i=0;i<10;i++)
	//~ cout<<a[i]<<" ";
	//~ cout<<endl;
	//~ cout<<j<<endl;
	ll int n;
	cin>>n;
	ll int ans,mc;
	if(n==1)
	ans=1;
	else
	{
		for(i=(n/2) +1;;i++)
		{
			mc=myfind(i);
			if(mc==1)
			{
				ans=i;
				break;
			}
		}
	}
	cout<<ans<<endl;
	return 0;
}
