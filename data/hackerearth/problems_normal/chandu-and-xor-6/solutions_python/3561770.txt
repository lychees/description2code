n, m = map(int, raw_input().split())
mat = []
row = []
for i in range(n):
	l = map(int, raw_input().strip().split())
	mat.append(l)
	su = 0
	for x in l:
		su += x
	row.append(su)
col = []
for r in range(m):
	su = 0
	for i in range(n):
		su+=mat[i][r]
	col.append(su)
max = 0
for r in row:
	for c in col:
		t = r^c
		if t > max:
			max = t
print max

#The Solution is O(n*m)
