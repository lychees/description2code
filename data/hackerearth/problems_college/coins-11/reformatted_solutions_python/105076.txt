import math


def isPrime(n):
    if n == 1:
        return False
    if n % 2 == 0:
        return False
    else:
        for i in range(3, int(math.sqrt(n))+1, 2):
            if n % i == 0:
                return False
    return True


t = input()
for i in range(t):
    n = input()
    if n == 3:
        print 3
    elif isPrime(n):
        print 2
    else:
        print 1
