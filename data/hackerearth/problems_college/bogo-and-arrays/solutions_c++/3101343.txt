#include <iostream>
using namespace std;

int main()
{
    long int t,n,k,a[200000]={0},i,j;
    long long int ans=0;
    int flag=0;
    cin>>t;
    while(t--)
    {ans=0;
    	cin>>n>>k;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	for(i=0;i<n;i++)
    	{flag=0;
    		for(j=i;j<n;j++)
    		{   
    			if(a[j]>k)
    			{
    				ans+=n-j;
    				flag=1;
    			}
    			if(flag==1)
    			break;
    		}
    	}
    	cout<<ans<<"\n";
    }
    return 0;
}
