#include <iostream>
#include <stdio.h>
#include <string>

using namespace std;

bool isPal( string s ) {
	
	int len = s.length(), i;
	
	for( i = 0;i < len/2;i++ )
		if( s[ i ] != s[ len - i - 1 ] )
			return false;
	
	return true;
}

int main()
{
	int t, n, i, val;
	
	
	cin >> t;
	
	while( t-- ) {
		
		string s, s1;
		cin >> s;
		
		n = s.length();
		
		for( i = 0;i < n;i++ ) {
			
			s1 = s.substr( i, n - i );
			if( isPal( s1 ) ) {
				val = i + n;
				break;
			}
		}
		
		printf("%d\n", val );
	}
	return 0;
}
