import sys
n = int(raw_input())
i = n
while i > 0:
    j = 0
    if (i == n or i == 1):
        while j < n:
            sys.stdout.write("*")
            j = j + 1
        print
    else:
        k = i
        while j < k-1:
            print(''),
            j = j + 1
        print('*')
    i = i - 1
