typedef long long int ll;

template<class T1> void deb(T1 e1) {
    cout << e1 << endl;
}

template<class T1,class T2> void deb(T1 e1, T2 e2) {
    cout << e1 << " " << e2 << endl;
}

int n,m,mx,c;
vector<int> G[MAX],temp;
bool visited[MAX]={false};
int a[MAX];

void dfs(int i) {
    visited[i]=true;
    mx = max(mx,a[i]);
    temp.pb(i);
    for(v(it)=G[i].begin();it!=G[i].end();it++)
        if(!visited[*it])
            dfs(*it);
}

int main() {
    ios_base::sync_with_stdio(false);
    int u,v;
    ll ans=1;
    cin>>n>>m;
    REP(1,n)
        cin>>a[i];
    while(m--) {
        cin>>u>>v;
        G[u].pb(v);
        G[v].pb(u);
    }
    REP(1,n) {
        if(!visited[i]) {
            mx=-1,c=0;;
            dfs(i);
            for(v(it)=temp.begin();it!=temp.end();it++){
                if(mx==a[*it])
                    c++;
            }
            ans = (ans*c)%MOD;
            temp.clear();
        }
    }
    deb(ans);
    return 0;
}
