#include <iostream>
#include<cmath>
using namespace std;
 
int main() 
{
	
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		long int a[n],i,r=0,l=0;
		cin>>a[0];
		r=a[0];
		for(i=1;i<n;i++)
		{
			cin>>a[i];
			l+=a[i];
		}
		long int j,mini;
		mini=abs(l-r);
		for(i=1;i<=n-1;i++)
		{
			r+=a[i];
			l-=a[i];
			j=abs(l-r);
			if(j<mini)
				mini=j;
		}
		cout<<mini<<endl;
	}
	return 0;
}
