#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <iterator>
#define ll long long int
#define pb(x) push_back(x)
#define mp(x,y) make_pair(x,y)
#define m 1000000007
using namespace std;
int primes[2000005];

void sieve(ll n)
{
	ll i,j;
	for(i=0;i<=n;i++)
	primes[i]=1;
	
	primes[0]=0;
	primes[1]=0;
	
	for(i=2;i<=sqrt(n);i++)
	{
		if(primes[i])
		{
			for(j=i*i;j<=n;j+=i)
			{
				primes[j]=0;
			}
		}
	}
//	cout<<"end"<<endl;
}

int main()
{
sieve(2000005);
	ll n,i=0,j=0,ans=0,sum;
	cin>>n;
	sum=2*(n-1);
	while(i<=sum)
	{
	//	cout<<"i= "<<i<<"   ";
		if(i<n)
		j++;
		else
		j--;
	//	cout<<"j= "<<j<<endl;
		if(primes[i])
		ans=((ans%m)+(j%m))%m;
		i++;
	}
	cout<<ans%m<<endl;
   return 0;
}
