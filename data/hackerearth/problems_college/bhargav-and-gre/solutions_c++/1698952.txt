#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	char ch[10000];
	char s[10000];
	cin>>ch;
	cin>>n;
	int c=0,k=0;
	int len=strlen(ch);
	for(int i=0;i<n;i++)
	{
		cin>>s;
		
		int l=strlen(s);
		if(len==l)
		{
			if(ch[0]==s[0])
			c++;
			else
			{
				int p=ch[0];
				p=p+3;
				char d=(char)p;
				if(d==s[0])
				k++;
			}
		}
	}
	cout<<c<<" "<<k<<endl;
	return 0;
	}
