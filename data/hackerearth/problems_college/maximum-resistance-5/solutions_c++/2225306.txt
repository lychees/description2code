#include<bits/stdc++.h>
using namespace std;
int main()
{
	int test; cin>>test;
	while(test--)
	{
		string s; cin>>s;
		int count=0,x,a;
		vector<int>v;
		stack<int>st;
		
		for(int i = s.size()-1;i>=0;i--)
		{
			if(s[i]=='X')
			{
				st.push(1);
				cin>>x;
				v.push_back(x);
				count++;
			}
			else if(s[i]=='A')
			{
				a = st.top();
				st.pop();
				a += st.top();
				st.pop();
				st.push(a);
			}
			else
			{
				a = st.top();
				st.pop();
				a = max(a,st.top());
				st.pop();
				st.push(a);
			}
		}
		sort(v.begin(),v.end(),greater<int>());
		int ans = 0;
		for(int i=0;i<st.top();i++)
		{
			ans += v[i];
		}
		cout<<ans<<endl;
	}
}
