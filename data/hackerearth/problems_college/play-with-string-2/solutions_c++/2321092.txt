#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--) {
		int flag = 0;
		string s, r;
		cin >> r;
		s = r;
		reverse(r.begin(), r.end());
		for (int i = 1; i < s.length(); i++) {
			if (abs(((int)s[i]) - ((int) s[i-1])) != abs(((int)r[i]) - ((int) r[i-1]))) {
				flag = 1;
				break;
			}
		}
		if (flag == 0)
			cout << "Funny" << endl;
		else
			cout << "Not Funny" << endl;
	}
    return 0;
}
