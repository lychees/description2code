#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int x,y,a[n],b[n];
    
    for(int i=0;i<n;i++)
    {   
    	cin>>x>>y;
        a[i]=x+y;
        b[i]=a[i];    
    }
    
    sort(a,a+n);
    
    int c[n];
    for(int j=0;j<n;j++)
    {
      for(int k=0;k<n;k++)
    	{
    		if(a[j]==b[k])
    		{
    		b[k]=-1;
    		c[j]=k+1;
    		break;
    		}
    	}
 
    }for(int l=0;l<n;l++)
      cout<<c[l]<<" ";
  return 0;

}
