#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int points[n][2],minX=91,minY=181,maxX=-91,maxY=-181;
    		for(int i=0;i<n;i++)
    		for(int j=0;j<2;j++)
    		cin>>points[i][j];
    		
    		for(int i=0;i<n;i++)
    		{
    			if(minX>points[i][0])
    			minX=points[i][0];
    			if(maxX<points[i][0])
    			maxX=points[i][0];
    			
    		    if(minY>points[i][1])
    			minY=points[i][1];
    			if(maxY<points[i][1])
    			maxY=points[i][1];
    		}
    	    
    	    int area = (maxX-minX)*(maxY-minY);
    	    if(area==0)
    	    {
    	    	if(maxX==minX)
    	        cout<<(maxY-minY)<<endl;
    	        else if(maxY==minY)
    	        cout<<(maxX-minX)<<endl;
    	    }
    	    else
    	    cout<<area<<endl;
    }
    return 0;
}
