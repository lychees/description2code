typedef long long int ll;
bool status[1100002];
struct st
{
    ll a,b;

};

bool cmp(st n,st m)
{
    if(n.a<m.a) return true;
    else if(n.a>m.a) return false;
    else
    {
        if(n.b>=m.b) return true;
        else return false;
    }
}

int main()
{
    ll n;
    cin>>n;
    st ar[n+1];
    int i;
    ll a,b;
    for(i=0;i<n;i++)
    {
        scanf("%lld%lld",&a,&b);
        ar[i].a=a;
        ar[i].b=b;

    }
    sort(ar,ar+n,cmp);
    for(i=0;i<n;i++)
    {
        pf("%lld %lld\n",ar[i].a,ar[i].b);
    }
}
