#include <bits/stdc++.h>
using namespace std;
long long t,n,k,m,p[1024],Tij,ind[16],mint[16],curr[1024];
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>t;
	for(int q=0;q<t;q++)
	{
		cin>>n>>k>>m;
		for(int i=0;i<n;i++)
		{
			cin>>p[i];
			curr[i]=p[i]*k+(p[i]-1)*m;
		}
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<p[i];j++)
			{
				cin>>Tij;
				curr[i]+=Tij;
			}
			if(i==0) 
			{
				mint[q]=curr[i];
				ind[q]=i+1;
			}
			else
			{
				if(mint[q]>curr[i])
				{
					mint[q]=curr[i];
					ind[q]=i+1;
				}
			}
		}
	}
	for(int i=0;i<t;i++)
	{
		cout<<ind[i]<<" "<<mint[i]<<endl;
	}
	return 0;
}
