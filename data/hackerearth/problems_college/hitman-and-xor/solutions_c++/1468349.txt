#include <iostream>
using namespace std;

int main()
{
long long int t,n,arr[100001];
cin>>t;
while(t--)
{
	long long int cnt=0,tmp=0;
	cin>>n;
	for(long long int i=0;i<n;i++)
	{
		cin>>arr[i];
		if((arr[i]%2)==0)
		{cnt++;}
		if(arr[i]%2!=0)
		{tmp++;}
	}
	
	cout<<cnt*tmp<<endl;
}

    return 0;
}
