#include <iostream>
#include<stdio.h>
using namespace std;
#include<cmath>

int main()
{
	long long t;
	scanf("%lld",&t);
	while(t--)
	{
		long long n,k;
		scanf("%lld %lld",&n,&k);
	
		double total=((double)(n-1)/(k+1))-1;
		
		
		
		if(floor(total)==total)
		printf("Nandu\n");
		else
		printf("Rani\n");
	}
    
    return 0;
}
