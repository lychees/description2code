#include <iostream>
#include <cmath>
#define pi 3.141592653589793
#define ll long long
using namespace std;
int main(){
	int test;
	cin>>test;
	while(test--){
		ll n;
		cin>>n;
		ll count=0,i=5,temp=n;
		
		for (ll i=5; temp/i>=1; i *= 5)
          count += temp/i;

		ll numberOfDigits=floor(( (log(2*pi*n))/2 +n*(log(n)-1) )/log(10))+1;
		
		cout<<numberOfDigits-count<<endl;

	}
}
