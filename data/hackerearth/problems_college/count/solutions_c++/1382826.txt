#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {

        int r,c,a[100000][100000],i,j;
        int ct=0,cnt=0;
        cin>>r>>c;
        if(r<=c)
        {
           for(i=1;i<=r;i++)
           {
              for(j=1;j<=i;j++)
              {
                  if(i<=c)
                  {
                       a[i][j]=i;
                   if(a[i][j]%2==0)
                   {
                       ct++;
                   }
                   else
                   {
                       cnt++;
                   }
                  }


              }
           }
        }
        else {
            for(i=1;i<=r;i++)
           {
               for(j=1;j<=i;j++)
               {
                  if(j<=c)
                  {
                     a[i][j]=i;
                     if(a[i][j]%2==0)
                     {
                        ct++;
                     }
                     else
                     {
                        cnt++;
                     }
                  }
              }
           }
        }
        cout<<ct<<" "<<cnt<<endl;
    }
    return 0;
}
