#include<cstdio>
using namespace std;
int a[1000001]={0};
void compute()
{
	a[1]=0;
	for(int i=2;i*i<=1000000;i++)
		if(a[i]==0)
			for(int j=2*i;j<=1000000;j+=i)
				a[j]+=1;
}
int main()
{
	compute();
	int n,q;
	scanf("%d %d",&n,&q);
	int b[11]={0},x;
	for(int i=2;i<=n;i++)
	{
		x=a[i];
		if(x==0)
			b[1]++;
		else
			b[x]++;
	}
	while(q--)
	{
		int ch;
		scanf("%d",&ch);
		if(ch>10)
			printf("0\n");
		else
			printf("%d\n",b[ch]);
	}
	return 0;
}
