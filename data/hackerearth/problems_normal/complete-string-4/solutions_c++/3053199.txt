#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	string S;
    	cin>>S;
    	int i;
    	if(S.length()<26)
    	cout<<"NO"<<endl;
    	else
    	{
    		int no=0;
    		char c='a';
    		while(c<='z'){
    			for(i=0;i<=S.length();i++)
    			{
    				if(S[i]==c){
    					no++;
    					break;
    				}
    			}
    			c++;
    		}
    		if(no==26)
    		{
    			cout<<"YES"<<endl;
    		}
    		else 
    		{
    			cout<<"NO"<<endl;
    		}
    	}
    }
}
