#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t,m,n;
   unsigned long int tm;
   cin>>t;
   while(t--)
   {
   	 cin>>m>>n;
   	 tm=0;
   	 if(!n)
   	 {  cout<<tm<<endl;continue;}
   	 while(n-m>=m)
   	 {
   	 	tm+=10;
   	 	n-=m;
   	 }
   	 if(n<=m)
   	  tm+=10;
   	 else
   	 {
   	 	if((m-(n-m))>=(n-m))
   	 	  tm+=15;
   	 	else
   	 	  tm+=20;
   	 }
   	 cout<<tm<<endl;
   }
    return 0;
}
