long int gcd(long int a,long int b)
{
    if(b==0)
        return a;
    else
        return (gcd(b,a%b));
}
int main()
{
	int t,N;
	cin>>t;
	while(t--){
		cin>>N;
		long s[N];
		cin>>s[0];
		long g=s[0];
		for(long i=1;i<N;i++){
			cin>>s[i];
			g=gcd(g,s[i]);
		}
		cout<<g<<endl;
	}
	return 0;
}
