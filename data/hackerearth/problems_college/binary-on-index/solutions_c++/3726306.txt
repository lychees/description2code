#include <iostream>
using namespace std;
int main()
{long long n,m,a[100000],c,pos; int low,high,mid;
 cin>>n;
 for(int i=0;i<n;i++){
 	cin>>a[i];
 }
 cin>>m;
 for(int j=0;j<m;j++){cin>>c;pos=-1;high=n-1;low=0;
 while(low<=high){
        
 	           mid=(low+high)/2;
 	           if(a[mid]==c){
               pos = mid;
               break;}
               if(a[mid]>c){
               	high = mid-1;
               }
               if(a[mid]<c){
               	low = mid+1;
               }
 	}
 cout<<pos<<"\n";
 }
    return 0;
}
