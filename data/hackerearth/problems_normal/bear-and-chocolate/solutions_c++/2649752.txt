#include <bits/stdc++.h>
using namespace std;

char A[1005][1005];

int main()
{
   	int t,n;
   	scanf("%d",&t);
   	while(t--)
   	{
   		scanf("%d",&n);
   		int col[n],row[n];
   		for(int i=0; i<n; i++)
   			row[i]=col[i]=0;
   		for(int i=0; i<n; i++)
   		{
   			for(int j=0; j<n; j++)
   				cin>>A[i][j];
   		}
   		int x=0;
   		for(int i=0; i<n; i++)
   		{
   			x=0;
   			for(int j=0; j<n; j++)
   			{
   				if(A[i][j]=='#')
   				{
   					col[j]++;
   					x++;
   				}
   				
   			}
   			row[i]=x;
   			if(i>=1)
   				row[i]+=row[i-1];
   		}
   		for(int i=1; i<n; i++)
   			col[i]+=col[i-1];
   		int f=0;
   		//for(int i = 0; i<n; i++)
   		//	cout<<row[i]<<" ";
   		for(int i=0; i<n; i++)
   		{
   			if(row[i]==row[n-1]-row[i] || col[i]==col[n-1]-col[i])
   			{
   				printf("YES\n");
   				f=1;
   			}
   		}
   		if(!f)
   			printf("NO\n");
   	}
    return 0;
}
