#include <iostream>
using namespace std;

int main()
{
	int t,m,n,cnt,i,j;
    cin>>t;
    while(t--)
    {
    	cin>>m>>n;
    	char s[m][n];
    	for(i=0;i<m;i++) 
    	for(j=0;j<n;j++) cin>>s[i][j];
    	cnt=0;
    	for(i=1;i<m;i++)
    	{
    		for(j=1;j<n;j++)
    		{
    			if(s[i-1][j-1]==47&&s[i-1][j]==92&&s[i][j-1]==92&&s[i][j]==47) cnt++;
    		}
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
