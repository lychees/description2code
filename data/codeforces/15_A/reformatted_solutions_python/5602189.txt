#!/usr/bin/env python

import time
import sys
import io
import re
import math
start = time.clock()
ans = 2
(n, t) = map(int, raw_input().split())
a = sorted([map(float, raw_input().split()) for _ in range(n)])
while True:
    if len(a) < 2:
        break
    else:
        for i in range(1, n):
            if t > ((a[i][0]-a[i][1]/2)-(a[i-1][0]+a[i-1][1]/2)):
                pass
            if t == ((a[i][0]-a[i][1]/2)-(a[i-1][0]+a[i-1][1]/2)):
                ans += 1
            if t < ((a[i][0]-a[i][1]/2)-(a[i-1][0]+a[i-1][1]/2)):
                ans += 2
        break
print ans
