
#include <iostream>
#include<math.h>
using namespace std;
 
int main()
{
    int t, d, i, I, m, n, a[1000];
    int cst, mincost, op;
    bool del=false, ins=false;
    cin>>t;
    int j=1;
    while(t){
    	del=false, ins=false;
    	mincost=0;
    	cin>>d>>I>>m>>n;
//    	if(d<i)
//    	{	op=d;
//    		del=true;
//    	}
//    	else
//    	{	op=i;
//    		ins=true;
//    	}
    	for(i=0; i<n; i++)
    		cin>>a[i];
		
		i=0;
		while(i<n-1)
		{
			if(abs(a[i+1]-a[i])> m){
			cst=abs( a[i+1]-(a[i]+m));
		//	cout<<"cost : "<<cst<<endl;
		//	cout<<"min : "<<mincost<<endl;
 
			if(cst<d && cst <I){
				mincost +=cst;
				a[i+1]=a[i]+m;
				i++;
		//		cout<<"i :"<<i<<endl;
			}
			else if(d<I){
					mincost += d;
					a[i+1]=a[i];
					i++;
			}
			else{
				mincost += I;
				if(a[i+1]>a[i])
				a[i] = a[i]+m;
				else a[i]=a[i]-m;
			}
			}
			else i++;
			
		}
		
		cout<<"Case #"<<j<<": "<<mincost<<endl;
		t--;
		j++;
    }
    return 0;
}
