#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007
#define ll long long int
#define l long int

int main()
{
    ll q, c, co;
    vector<ll> cost;
    scanf("%lld",&q);
    while(q--){
    	scanf("%lld",&c);
    	if(c==1){
    		if(cost.empty())
    			printf("No Food\n");
    		else{
    			printf("%lld\n",cost.back());
    			cost.pop_back();
    		}
    	}
    	else{
    		scanf("%lld",&co);
    		cost.push_back(co);
    	}
    }
    return 0;
}
