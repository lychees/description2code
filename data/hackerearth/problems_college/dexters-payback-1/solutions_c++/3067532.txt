#include<iostream> 
using namespace std;

int main(){
    
int array[1003][1003], x,y,t,result;
	cin>>t;
	for(int l=0;l<t;l++)
        {
		cin>>x>>y;
		 result = 0;
		for (int i = 0; i < x; i++){
			for (int j = 0; j < y; j++){
				array[i][j] = 0;
                                if (j - 2 >= 0){
					if (array[i][j - 2]){
						continue;
					}
				}
				if (i - 2 >= 0){
					if (array[i - 2][j]){
						continue;
					}
				}
				
				array[i][j] = 1;
				result++;
			}
		}
		cout << result << endl;
	} 
}
