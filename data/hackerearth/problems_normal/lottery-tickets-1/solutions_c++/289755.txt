#include<iostream>
#include<iomanip>

using namespace std;

int main()
{
    int testcases;
    cin>>testcases;
    while(testcases>0)
    {
        int N;
        cin>>N;
        long long S[N],T[N],P[N],f[N];
        for(int i=0;i<N;i++)
        {
            cin>>S[i];
            cin>>T[i];
            cin>>P[i];
        }
        for (int i=0;i<N;i++)
            for (int j=i+1;j<N;j++)
                if (T[j]<T[i])
                {
                    swap(S[i],S[j]);
                    swap(T[i],T[j]);
                    swap(P[i],P[j]);
                }
        int result=0;
        for(int i=0;i<N;i++)
        {
            f[i]=P[i];
            for(int j=0;j<i;j++)
            {
                if(S[i]>=T[j]&&f[j]+P[i]>f[i])
                {
                    f[i]=f[j]+P[i];
                }
            }
            if(f[i]>result)
            {
                result=f[i];
            }
        }
        cout<<fixed;
        cout<<setprecision(2)<<(double)result/100<<"\n";
        testcases--;
    }
}
