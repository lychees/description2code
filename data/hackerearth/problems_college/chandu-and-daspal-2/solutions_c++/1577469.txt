#include <iostream>
using namespace std;
int gcd(int a,int b)
{
	if(b==0)
	return a;
	else
	return(gcd(b,a%b));
}
int main()
{
    //cout << "Hello World!" << endl;
    long long t;
    cin>>t;
    while(t--)
    {
    	long long a,b;
    	 	int cont=0;
    	string str;
    	cin>>a>>b;
    	cin>>str;
    	if(a%b==0||b%a==0)
    	{
    		if(str.compare("Chandu")==0)
    		cout<<"Chandu"<<endl;
    		else
    		cout<<"Daspal"<<endl;
    	}
   
    	else
    	{
    		while(a&&b)
    		{
    			int temp=gcd(a,b);
    			a-=temp;
    			b-=temp;
    			cont=cont^1;
    		}
    
    	if(cont==0&&str.compare("Chandu")==0)
    	cout<<"Daspal"<<endl;
    	else if(cont==1&&str.compare("Chandu")==0)
    	cout<<"Chandu"<<endl;
    	else if(cont==1&&str.compare("Daspal")==0)
    	cout<<"Daspal"<<endl;
    	else
    	cout<<"Chandu"<<endl;
    	}
    	
    }
    return 0;
}
