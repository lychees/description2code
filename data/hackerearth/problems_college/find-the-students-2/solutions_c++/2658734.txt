#include <iostream>
#include<algorithm>
#include<stdlib.h>
using namespace std;


int main()
{
    //cout << "Hello World!" << endl;
    int n,k,i,ctr=0,min;
    scanf("%d",&n);
    int a[n];
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    sort(a,a+n);
    //int j=0;
    for(k=0;k<n;k++)
    {
    	ctr=0;
    	if(a[k]<=0)
    	{
    		continue;
    	}
    	//printf("%d\t",a[k]);
    	min=a[k];
    	//printf("%d ",k);
    	for(i=k;i<n;i++)
    	{
    		
    		
    		a[i]=a[i]-min;
    		//printf("%d\t",a[i]);
    		ctr++;	
	    }
	
	    printf("%d\n",ctr);
    
    }
    
    
    
    
    return 0;
}
