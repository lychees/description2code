#include<iostream>
#include<algorithm>
#include<map>
#include<vector>
using namespace std;
bool compare (const pair<int,int>&i ,const pair<int,int>&j){
	if(i.first==j.first){
		return i.second>j.second;
	}
   else{
   return i.first<j.first;
   }

}

int main(){
	int t,i;
	cin>>t;
	string str;
	while(t--){
	cin>>str;
	vector<int> arr(26,0);
	for(i=0;i<str.length();i++){
	arr[str[i]-'a']=arr[str[i]-'a']+1;	
	}	
    
    for(i=25;i>=0;i--){
    if(arr[i]==0){
    cout<<char('a'+i)<<" ";	
    }	}	
	
	
	vector< pair<int,int> > brr;
	for(i=0;i<26;i++){
	if(arr[i]!=0)
	brr.push_back(std::make_pair(arr[i],i));	
	}
	
	sort(brr.begin(),brr.end(),compare);
	
	for(i=0;i<brr.size();i++){
	cout<<char('a'+brr[i].second)<<" ";	
	}
	
	cout<<endl;
	
}
	
	return 0;
}
