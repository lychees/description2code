#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
#define MAX 100005
vector<int> vt[MAX];
void preorder(int cur)
{
	cout<<cur<<" ";
	if(vt[cur].size() && vt[cur][0] != 0)
	{
		preorder(vt[cur][0]);
	}
	if(vt[cur].size() > 1 && vt[cur][1] != 0)
	{
		preorder(vt[cur][1]);
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int i;
		for(i=0;i<MAX;++i)
			vt[i].clear();
		int n;
		cin>>n;
		int p,lc,rc;
		for(i=0;i<n;++i)
		{
			cin>>p>>lc>>rc;
			vt[p].push_back(lc);
			vt[p].push_back(rc);
		}
		preorder(1);
		cout<<endl;
	}
    return 0;
}
