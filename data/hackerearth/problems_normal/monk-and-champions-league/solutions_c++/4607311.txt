#include <iostream>
using namespace std;
#include<queue>

int main()
{
    priority_queue<int> seats;
    int m,n;
    cin>>m>>n;
    while(m--)
    {
    	int x;
    	cin>>x;
    	seats.push(x);
    }
    int maxRow,revenue=0;
    while(n--)
    {
    	maxRow=seats.top();
    	seats.pop();
    	seats.push(maxRow-1);
    	revenue+=maxRow;
    }
    cout<<revenue<<endl;
    return 0;
}
