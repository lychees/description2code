#include <bits/stdc++.h>
#define rep(i,n) for(int i =0 ;i<n;i++)
#define recrep(i,n) for(int i =n ;i>=0;i--)
#define scl(l) scanf("%lld",&l)
#define prn(s) printf("%s\n",s)
#define prl(l) printf("%lld ",l)
#define ll long long
using namespace std;

int main()
{
    ll N,T;
    scl(T);
    while(T--)
    {
        scl(N);
        ll A[N],B[N];
        rep(i,N)
            scl(A[i]);
        rep(i,N)
            scl(B[i]);
        ll monk =0,j=0;
        rep(i,N)
        {
            while(B[j]>= A[i] &&j<N)
            {
                monk = max(j-i,monk);
                j++;
            }
        }
        prl(monk);
        prn("");

    }
    return 0;
}
