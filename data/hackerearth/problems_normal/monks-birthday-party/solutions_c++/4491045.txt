#include <iostream>
#include <cstdio>
#include <set>
using namespace std;

int main(int argc, char const *argv[])
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		string temp;
		set <string> s;
    	set <string>::iterator it;
		for (int i = 0; i < n; ++i)
		{
			cin>> temp;
			s.insert(temp);
		}
		for (it = s.begin(); it != s.end(); it++)
		{
			cout<< *it  << endl;
		}
	}
	return 0;
}
