#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<vector>
#include<map>
#include<queue>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
int maze[300][300];
int infi=-100000000;
char x;

int main()
	{
	int i,j,k;
	int T;
	int m,n;
	ios_base::sync_with_stdio(false);
	int ans[202][202];
	cin>>T;
	while(T--)
	{
	cin>>n>>m;
	for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			cin>>x;
			if(x=='T')
				maze[i][j]=150;
			else if(x=='D')
				maze[i][j]=-100;
			else if(x=='L' && i==n && j==m)
				maze[i][j]=0;
			else return 1;
		}
	}
	for(i=0;i<=n;i++)
		ans[i][0]=infi;
	for(i=0;i<=m;i++)
		ans[0][i]=infi;
	ans[0][1]=0;
	
	for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			ans[i][j]=maze[i][j]+max(ans[i-1][j],ans[i][j-1]);
			if(ans[i][j]<0)
				ans[i][j]=infi;
		}
	}
	i=n;j=m;
	if(ans[i][j]<0)
		ans[i][j]=-1;
	printf("%d\n",ans[i][j]);
	
	}
	return 0;
	}
