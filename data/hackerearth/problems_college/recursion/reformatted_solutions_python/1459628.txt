from itertools import permutations

tc = int(input())
while tc > 0:
    n = int(input())
    string = ""
    for i in range(1, n+1):
        string = string+str(i)
    perms = [''.join(p) for p in permutations(string)]
    perms = sorted([int(x) for x in perms])
    for a in perms:
        print a,
    print ""
    tc = tc-1
