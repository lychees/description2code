test=int(raw_input())
for i in range(test):
	Food,Gold,S,P,C,lal=map(int,raw_input().split(' '))
	sam,pal,champ,faj=0,0,0,-1
	while sam*100<=Food:
		pal=0
		while ((pal*125+sam*100)<=Food) and (pal*50<=Gold):
			Food_Left=Food-100*sam-125*pal
			Gold_Left=Gold-50*pal
			champ=min(Food_Left/50,Gold_Left/100)
			faj=max(faj,S*sam+P*pal+C*champ)
			pal+=1
		sam+=1
	if faj>lal:
		print faj-lal
	else:
		print '-1'
