	{
	scanf("%d %d %lld",&N,&M,&A);
	long long ans=0;
	for(i=1;i*i<=A;i++)
		{
		if(A%i!=0)
			continue;
		long long w=A/i;
		long long h=i;
	//	cout<<w<<h<<"l"<<endl;
		if(N>=h && M>=w)
			{
			ans+=(N-h)*(M-w);
			}
		h=A/i;
		w=i;
		if(h==w)
			continue;
		if(N>=h && M>=w)
			{
			ans+=(N-h)*(M-w);
			}
		}
	printf("%lld\n",ans);
	}
return 0;
}
