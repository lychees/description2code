#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long int t,n,ans,temp=0,i,j;
    cin>>t;
    while(t--){
    	cin>>n;
    	ans=temp=0;
    	long long int a[n],b[n];
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	for(i=0;i<n;i++)
    		cin>>b[i];
    	for(i=0;i<n;i++){
    		for(j=i;j<n;j++){
    			if(b[j]<a[i])
    				break;
    			temp=j-i;
    			if(temp>ans)
    				ans=temp;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
