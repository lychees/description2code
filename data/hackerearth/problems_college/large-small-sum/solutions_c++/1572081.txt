#include <cstdio>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int a[10005],i,n,x,y;
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    	scanf("%d",&a[i]);
    	sort(&a[0],&a[n]);
    	scanf("%d%d",&x,&y);
    	printf("%d\n",a[y-1]+a[n-x]);
    }
    return 0;
}
