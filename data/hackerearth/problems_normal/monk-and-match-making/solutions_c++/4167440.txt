#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

#define B1 33
#define M1 1000000009
#define B2 163
#define M2 1000000007

string s;
ll h1[100005];
ll h2[100005];
ll p1[100005];
ll p2[100005];



void init(){
	int n=s.length(),i;
	p1[0]=p2[0]=1;
	for(int i=1;i<=n;i++){
		p1[i]=(B1*p1[i-1])%M1;
		p2[i]=(B2*p2[i-1])%M2;
	}
	for(int i=1;i<=n;i++){
		h1[i]=(B1*h1[i-1]+s[i-1]-'a'+1)%M1;
		h2[i]=(B2*h2[i-1]+s[i-1]-'a'+1)%M2;
	}
}

ll getHash1(int l, int r){
	int len=r-l+1;
	ll ret = (h1[r]-(h1[l-1]*p1[len])%M1+M1)%M1;
	return ret;
}

ll getHash2(int l, int r){
	int len=r-l+1;
	ll ret = (h2[r]-(h2[l-1]*p2[len])%M2+M2)%M2;
	return ret;
}

int main(){
	cin>>s;
	int q;
	cin>>q;
	init();
    while(q--){
        int x1,y1,x2,y2;
        scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
        if(y1 - x1 == y2 - x2)
        {
            pair < ll , ll > H1(getHash1(x1,y1),getHash2(x1,y1));
            pair < ll , ll > H2(getHash1(x2,y2),getHash2(x2,y2));

            if(H1 == H2){
                printf("Yes\n");
            }
            else{
                printf("No\n");
            }
        }
        else{
            printf("No\n");
        }
    }
}
