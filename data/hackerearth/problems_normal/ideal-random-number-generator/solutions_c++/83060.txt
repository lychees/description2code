#include<cstdio>
#include<iostream>
#include<cstring>
#include<cmath>
using namespace std;
#define max 100001
long long int gcd(long long int  a, long long int b);

int main()
{
		long long  a,b,c;
	 	long long  ans1,ans2,temp1,temp2;
        cin>>a>>b>>c;
		if(a<b)	
		{
			temp1=a;
			a=b;
			b=temp1;
		}
		if(c>=a+b)
			{
				ans1=1;
				ans2=1;
		    }
		else if(c<=b)
		{	
			ans1=(c*c);
			ans2=(2*a*b);
		}
		else if(c>=a)
		{
			 temp2=c-b;
			ans1=2*b*temp2+2*a*c-a*a-2*c*temp2+temp2*temp2;
			ans2=2*(a*b);
		}
		else
		{
			temp2=c-b;
			ans1=2*temp2+b; ans2=(2*a);
		}
		long long int ans = gcd (ans1,ans2);
		printf("%lld/%lld\n",ans1/ans,ans2/ans);
	
	return 0;
}
long long int gcd(long long int  a, long long int b)
{
	
	while(a !=b)
	{
		if(a>b)
			return gcd(a-b,b);
		else
			return gcd (a,b-a);
	}
	return a;
}
