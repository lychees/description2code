#include <bits/stdc++.h>
using namespace std;
int noOfMarbles(long int *a,long int n)
{
	sort(a,a+n);
	long int i=0,k=0,count=0,check=1;
	while(i<n)
	{
		k=a[i];
		check=1;
		while(a[i]<=(k+4))
		{
			i++;
			check=0;
		}
		count++;
		if(check)
		{i++;}
	}
	cout<<count<<"\n";
}
int main()
{
	long int n;
	cin>>n;
	long int a[n],i=0;
	while(i<n)
	{
		cin>>a[i];
		i++;
	}
	noOfMarbles(a,n);
}
