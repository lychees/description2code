#include <iostream>
using namespace std;

int main()
{
    long T, N, K, num, result, current;
    cin >> T;

    while(T--) {

    	result = 0; current = 0;
    	cin >> N >> K;

    	for (long n = 0; n < N; n++) {
    		cin >> num;
    		if (num <= K) {
    			current += num;
    			
    		}
    		else {
    			if (current > result) {
    			   result = current;
    			}
    			current = 0;
    		}
    	}
    	cout << max(result,current) << endl;
    }
    return 0;
}
