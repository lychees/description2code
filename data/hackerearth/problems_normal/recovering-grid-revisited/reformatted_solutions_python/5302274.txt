import math
modulo = (10**9)+7
fa = [1]
for i in xrange(1, 2000000):
    fa.append((fa[i-1]*i) % modulo)


def combination(n, r):
    numerator = fa[n]
    denominator = (fa[r]*fa[n-r]) % modulo
    den = pow(denominator, modulo-2, modulo)
    return (numerator*den) % modulo


t = input()
for i in range(t):
    s = raw_input().split(' ')
    n = int(s[0])
    m = int(s[1])
    t = m+n*n-2
    answer = combination(t, m)
    if n == 1:
        answer = 0
    print answer
