#include<stdio.h>
#define min(a, b) (((a) <= (b)) ? (a) : (b))
int main(){
	long long int t, r, c, e, o;
	scanf("%lld", &t);
	while(t-- && scanf("%lld%lld", &r, &c)){
		e = o = 0;
		do{
			if(r % 2 == 0)	e += min(r, c); 
				else	o += min(r, c);
		}while(r--); printf("%lld %lld\n", e, o);
	}return 0;}
