#include <iostream>
using namespace std;

const int Maxn = 20;

int fac[Maxn];
int t;

int main()
{
    fac[0] = 1;
    for (int i = 1; i < Maxn; i++)
    	fac[i] = i * fac[i - 1] % 10;
   	scanf("%d", &t);
   	while (t--) {
   		int a, b; scanf("%d %d", &a, &b);
   		int res = 0;
   		for (int i = a; i <= b; i++)
   			res = (res + fac[i]) % 10;
   		printf("%d\n", res);
   	}
    return 0;
}
