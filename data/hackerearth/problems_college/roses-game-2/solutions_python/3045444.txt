for _ in xrange(input()):
    l=input()
    arr=map(int,raw_input().split())
    if arr.count(-1)==0 or arr[0]==0:
        print "Lose"
    else:
        idx=0
        localIndex=[]
        while True:
            if arr[idx]==-1:
                print "Win"
                break
            else:
                if idx in localIndex:
                    print "Lose"
                    break
                
                localIndex.append(idx)
                idx=arr[idx]
