def fun():
	n=input()
	l=raw_input().split()
	k={}
	for j in l:
		if k.has_key(j):
			k[j]=k[j]+1
		else:
			k[j]=1
	max=0
	for j in k:
		if max<k[j]:
			max=k[j]
	l=[]
	for j in k:
		if max==k[j]:
			l.append(int(j))
	l.sort()
	s=""
	for j in xrange(len(l)-1,-1,-1):
		s=s+str(l[j])+" "
	s.strip()
	print(s)
a=input()
for i in range(a):
	fun()
