#include<bits/stdc++.h>
using namespace std;

long int arr[1000006];

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        long int a, b, n, mod, temp;
        cin>>a>>b>>n>>mod;
        for(int i=0; i<1000005; i++)
            arr[i] = 0;
        arr[a]++;
        arr[b]++;
        for(int i=2; i<n; i++)
        {
            temp = ((a+b)%mod);
            arr[temp]++;

            if(i!=2)
                b=a;
            a=temp;
        }
        long long int ans = 0;
        for(int i=0; i<1000005; i++)
            if(arr[i]!=0)
                ans += arr[i]*arr[i];
        cout<<ans<<endl;
    }
    return 0;
    
}
