ll fact[102];
const int mod = (1e9)+7;

ll modulo(int a,int b=mod-2,int c=mod){
    long long x=1,y=a; 
    while(b > 0){
        if(b%2 == 1){
            x=(x*y)%c;
        }
        y = (y*y)%c; 
        b /= 2;
    }
    return x%c;
}

ll ncr(int n,int r){
    ll num = fact[n];
    ll den = (fact[r]*fact[n-r])%mod;
    ll ans = (num * modulo(den))%mod;
    return ans;
}
