#include <iostream>
using namespace std;

int main()
{
    int test;
    cin>>test;
    for(int i=0;i<test;i++)
    {
    	string s;
    	cin>>s;
    	int len=s.length();
    	int count=1;
    	int j=0;
    	while(j<len-1)
    	{
    		if(s[j]!=s[j+1])
    		 count++;
    		j++;
    	}
    	cout<<len-count<<"\n";
    }
    return 0;
}
