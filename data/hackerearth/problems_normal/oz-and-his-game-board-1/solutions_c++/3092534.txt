#include <iostream>
using namespace std;
int main()
{
      unsigned int t;
      long long n,m,a,b,c,temp;
    cin>>t;
   for(;t>0;t--)
    {
    
    	cin>>n>>m>>a>>b>>c;
    	if(m>n)
    	temp=2*n*c+(m-n)*b;
    	else
    	temp=2*m*c+(n-m)*a;
    	if(m*b+n*a > temp)
    	cout<<m*b+n*a<<"\n";
    	else
    	cout<<temp<<"\n";
	}
	return 0;
}
