int common(string a,string b){
	int len1 = a.size();
	int len2 = b.size();
	int i,j;int max=0;
	int mat;
	for(i=0;i<len1;i++){
		int k=i;
		mat = 0;
		for(j=0;j<len2;j++){
			if(k>=len1){
				if(mat>max)max=mat;break;
			}
			if(a[k]==b[j]){
			mat++;
			k++;
			}else{
				break;
			}
		}if(j>=len2){
			if(mat>max)max=mat;break;
		}
	}
	return max;
}
