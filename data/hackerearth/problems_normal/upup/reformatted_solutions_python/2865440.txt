S = raw_input()
T = []
PrevIsSpace = True

for i, l in enumerate(S):
    if PrevIsSpace and l >= 'a' and l <= 'z':
        l = chr(ord(l)-ord('a') + ord('A'))
    PrevIsSpace = (l == ' ')
    T.append(l)


print "".join(T)
