#include<bits/stdc++.h>
using namespace std;
#define FOR(i,N) for(int i=0;i<(N);i++)
const int maxn = 1000001;
bool  arr[maxn] = {false};
vector<int> sol;
void siv(){
	for(int i=2;i<=sqrt(maxn);i++){
		if(arr[i] == false){
			for(int j=i*i;j<maxn;j +=i)
				arr[j] = true;
		}
	}
}

int main(){
	siv();
	for(int i=2;i<maxn;i++){
		if(arr[i] == false){
			sol.push_back(i);
		}
	}
	
	int t; cin>>t;
	while(t--){
		int n; cin>>n;
		int count;
		int m = n;
		for(int i=0;n >1 && sol[i] < m ;i++){
			count = 0;
			
			while(n>1 && n%sol[i] == 0){
					n = n/sol[i];
					//cout<<n<<endl;
					count++;		
			}
			
			if(i>=1 && count >0) cout<<"*"; 
			//cout<<n<<endl;
			if(sol[i] == 2) cout<<sol[i]<<"^"<<count;
			else if(count>0){
				cout<<sol[i]<<"^"<<count;
			}
		}
		cout<<endl;
	}
	
}
