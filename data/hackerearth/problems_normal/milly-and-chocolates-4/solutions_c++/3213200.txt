#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int t,n,i;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	double res=0.0;
    	long sum=0;
    	int a[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		sum+=a[i];
    	}
    	res=(sum*1.0/(n*100*1.0))*100;
    	printf("%.8f\n",res);
    }
    return 0;
}
