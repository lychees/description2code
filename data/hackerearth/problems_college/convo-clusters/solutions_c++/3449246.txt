#include<iostream>
#include<fstream>
#include<sstream>
#include<vector>
#include<cmath>
#include<iomanip>
#include<algorithm>
using namespace std;
main()
{
    string str,vow = "aeiou",conso="bcdfghjklmnpqrstvwxyz";
    int end = 0;
    while(getline(cin,str))
    {
        if(end > 0) cout<<endl; end++;
        int vctr=0,cctr=0;
        for(int x=0; x<str.size(); x++)
        {
            if(str[x] == ' ') {vctr = 0; continue;}
            if(count(vow.begin(),vow.end(),str[x]) > 0) vctr ++;
            else
            {
                if(vctr > 1)
                {
                    if(count(conso.begin(),conso.end(),str[x]) > 0) 
                    {
                        vctr = 2; 
                        break;
                    }
                }
                vctr = 0;
            }
        }
        for(int x=0; x<str.size(); x++)
        {
            if(str[x] == ' ') {cctr = 0; continue;}
            if(count(vow.begin(),vow.end(),str[x]) == 0) cctr ++;
            else cctr = 0;
            if(cctr == 2) break;
        }
        if(cctr == 2 && vctr == 2) cout<<"YES";
        else cout<<"NO";
    }
}
