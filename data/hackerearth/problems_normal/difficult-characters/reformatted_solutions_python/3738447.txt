from string import ascii_lowercase as alphabet

num_test_cases = int(raw_input())
for _ in xrange(num_test_cases):
    input_string = raw_input()
    result_dict = {letter: 0 for letter in alphabet}

    for char in input_string:
        result_dict[char] += 1

    sorted_dict = sorted(result_dict.iteritems(),
                         key=lambda (key, value): (-value, key),
                         reverse=True)

    sorted_key_list = [key for key, value in sorted_dict]

    print ' '.join(sorted_key_list)
