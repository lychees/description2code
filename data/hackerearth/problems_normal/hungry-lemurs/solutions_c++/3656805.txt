#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int modi,i,j,k,t,n,m,count,maxi,ans=INT_MAX;
	cin>>k>>n;
	//k-langur, n- kele
	
	//pehle kele khata ke dekhte hai
	for(i=n;i>=0;i--)
	{
		if(i!=0)
			modi=(k%i);
		else 
			modi=0;
		maxi=(n-i)+min(modi,i-modi);
		if(maxi<ans)
			ans=maxi;
		//~ cout<<i<<" "<<ans<<endl;
	}
	for(i=n;i<=2*n;i++)
	{
		modi=(k%i);
		maxi=(i-n)+min(modi,i-modi);
		if(maxi<ans)
			ans=maxi;
		//~ cout<<i<<" "<<ans<<endl;
	}
	cout<<ans;
	return 0;
}
