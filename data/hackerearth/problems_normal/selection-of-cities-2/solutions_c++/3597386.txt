ll modexp(ll n)
{
ll res,a,b;
  res=1;
  a=2;
  b=n;
  while(b>0)
  {
    if(b%2)
    {
      res*=a;
      res%=MOD;
    }
    b>>=1;
    a*=a;
    a%=MOD;

  }
  return res%MOD;
}
