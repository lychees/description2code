#include<iostream>
#include<algorithm>
using namespace std;

int a[10004];

int main()
{
	int t,n;
	cin>>t;
	while(t--){
		cin>>n;
		for(int i=0;i<n;i++) cin>>a[i];
		sort(a, a+n);
		int ans=0;
		int temp=a[0];
		for(int i=0;i<n;i++)a[i]-=temp;
	
	for(int i=0; i<n; i++)
		{
			if(a[i]/5 >0){
				ans+=(a[i]/5);
				a[i]%=5;
			}
			 if(a[i]/2 >0){
				ans+=(a[i]/2);
				a[i]%=2;
			}
	     	if(a[i]) ans++;
		}
		cout<<ans<<endl;
	}
	return 0;
}
