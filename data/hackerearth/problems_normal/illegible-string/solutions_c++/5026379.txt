#include <iostream>
using namespace std;

int main(){
    string s;
    int n,a=0,b=0,c=0,sum=0;
    cin >> n >> s;
    for(int i=0;i<s.size();i++){
    	if(s[i]=='v') a++;
    	else if(s[i]=='w') b++;
    	else c++;
    }
    
    int cnt=0;
    for(int i=0;i<s.size();i++){
    	if(s[i]=='v'){
    		cnt++;
    	}
    	else if(s[i]=='w');
    	else{
    		sum+=cnt/2;
    		if(cnt%2) sum++;
    		cnt=0;
    	}
    }
    if(cnt>0){
    	sum+=cnt/2;
    	if(cnt%2) sum++;
    }
    cout << c+sum+b << " " << c+a+b*2 << endl;
    
    return 0;
}
