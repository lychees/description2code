#include <iostream>
using namespace std;

#define ll long long
ll solve(ll m, ll n) 
{
    ll x = 0;
    while ((m|x) < n)
       x = (x<<1) | 1;

    return m & ~x;
}
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	ll a,b;
    	scanf("%lld%lld",&a,&b);
    	printf("%lld\n",solve(a,b));
    }
    return 0;
}
