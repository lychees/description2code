#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    
    while(t--)
    {
    	long long n;
    	cin>>n;
    	long long a[n],i;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	sort(a,a+n);
    	long long ans=0;
    	for(i=0;i<n;i++)
    	{
    		long long j=i+1;
    		long long count=0;
    		while(j<n && a[j]==a[i])
    			j++;
    		count+=j-i;
    		ans+=count*(count-1)/2;
    		i=j-1;
    	}
    	cout<<ans+n<<endl;
    }
    return 0;
}
