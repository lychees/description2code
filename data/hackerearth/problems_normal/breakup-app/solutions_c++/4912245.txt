#include <iostream>
#include<string>
using namespace std;

int main()
{
	int c[32];
	int n;
	cin>>n;
	string s[n];
	getline(cin,s[0]);
	for(int i=0;i<n;i++)
	getline(cin,s[i]);
	for(int i=0;i<32;i++)
	c[i]=0;
	int x;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<s[i].size();j++)
		{
			x=0;
		//	cout<<"index of s["<<i<<"]"<<j<<"\n";
			if(s[i][j+1]>='1'&&s[i][j+1]<='9')
			{
				x=s[i][j]-'0';
				if(j<s[i].size()-1)
				if(s[i][j]>='0'&&s[i][j]<='9')
				x=(x*10)+(s[i][j+1]-'0');
			//	cout<<x<<" "<<s[i][j]<<"\n";
				if(x<=31)
				{
				c[x]++;
				if(s[i][0]=='G')
				c[x]++;
				}
			}
		}
	}
	int m=c[0];
	int mi=0;
		for(int i=1;i<32;i++)
		{
			if(c[i]>m)
			{
				m=c[i];
				mi=i;
			}
		}
		if(mi==19||mi==20)
		cout<<"Date\n";
		else
		cout<<"No Date\n";

    return 0;
}
