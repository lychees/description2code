#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
ll int fact[4004];
ll int ifact[4004];
ll power(ll a,ll b)
{
    if(b==0)
        return 1;
    if(b==1)
        return a;
    ll temp=power(a,b/2);
    if(b%2==0)
        return (temp%mod*temp%mod)%mod;
    return ((a%mod*temp%mod)%mod*temp%mod)%mod;
}
int main()
{
    int i;
    fact[0]=1;
    for(i=1;i<=2002;i++)
    {
        fact[i]=(i%mod*fact[i-1]%mod)%mod;
    }
    ifact[0]=1;
    for(i=1;i<=2002;i++)
    {
        ifact[i]=power(fact[i],mod-2)%mod;
    }
    ll sum=0;
    ll n,k,t;
    cin>>n>>k>>t;
    sum=t;
    ll num;
    for(int i=1;i<=k;i++)
    {cin>>num;
    sum=sum-(num-1);
    }
    if(sum<0||sum%2!=0)
    {
        cout<<"0"<<endl;
    }

    else
    {
        if(k<sum/2)
        cout<<"0"<<endl;
       else
       {
       	if(sum>=0)
        cout<<((fact[k]%mod*ifact[sum/2]%mod)%mod*ifact[k-sum/2]%mod)%mod<<endl;

       }
    }



}
