#include <iostream>
using namespace std;

int main()
{
    int t,n,g,a,m,x,y,z;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	x=y=z=0;
    	cin>>x>>y>>z;
    	for(int i=1;i<n;i++)
    	{
    		cin>>g>>a>>m;
    		g+=min(y,z);
    		a+=min(x,z);
    		m+=min(x,y);
    		x=g;
    		y=a;
    		z=m;
    		
    	}
    	cout<<min(x,min(y,z))<<endl;
    }
    return 0;
}
