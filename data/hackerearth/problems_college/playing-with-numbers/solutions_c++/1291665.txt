#include <iostream>
#include <algorithm>
 #include <stdio.h> 
 #include <stdlib.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    long long n;
    int op = 0;
    while(t--){
    	scanf("%lld",&n);
    	int num = __builtin_popcountll(n);
    	if (op==0 && num&1){
    		cout << "Sherlock\n";
    		continue;    		
    	}
    	if (op==0 && num%2==0){
    		cout << "Watson\n";
    		op^=1;
    		continue;    		
    	}
    	if (op==1 && num&1){
    		cout << "Watson\n";
    		continue;    		
    	}
    	if (op==1 && num%2==0){
    		cout << "Sherlock\n";
    		op^=1;
    		continue;    		
    	}
    }
    return 0;
}
