#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
const int MAX = 1e6 + 10;
ll pr[MAX], p[MAX];
void pre() {
	for (int i = 1; i < MAX; i++) {
		pr[i] = i;
	}
	for (int i = 2; i<MAX; i++) {
		if (pr[i] == i) {
			for (int j = i; j < MAX; j += i)
			{
				pr[j] -= pr[j] / i;
			}
		}
	}
	p[1] = 1;
	for (int i = 2; i < MAX; i++)
		p[i] = p[i - 1] + pr[i];
	// for (int i = 2; i < 10; i++)
	// 	cout << p[i] << " ";
	// cout << endl;
}
int main() {
	int n;
	ll su;
	//freopen("../input.txt", "r", stdin);
	pre();
	scanf("%d", &n);
//	cout << n << endl;
	for (int i = 0; i < n; i++) {
		scanf("%lld", &su);
		printf("%lld\n", (((su * (su - 1)) / 2) + 1  - p[su]));
	}
	return 0;
}
