#include <bits/stdc++.h>
using namespace std;
int main ()
{
	int tc;
	cin >> tc;
   int ab[2005];
	while (tc--)
	{
			int x1,x2,y1,y2,a1,b1,c1,d1,i;
	    int a, b, c, d;
		cin >> a1 >> b1 >> c1 >> d1;
		a=min(a1,b1)+1000;
		b=max(a1,b1)+1000;
		c=min(c1,d1)+1000;
		d=max(c1,d1)+1000;
	    for(i=0;i<2005;i++)
	    ab[i]=0;
	    for(i=a;i<=b;i++)
	    ab[i]++;
	    for(i=c;i<=d;i++)
	    ab[i]++;
	    
	    int count=0;
	    for(i=0;i<2005;i++)
	    if(ab[i]>1)
	    count++;
	    
		if(count==0)
		cout << "Line" << endl;
		else if(count==1)
		cout <<"Point"<< endl;
		else
		cout<<"Nothing"<<endl;
	}
	return 0;
}
