#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int N,Q;
    
    vector <int> sum;
    scanf("%d",&N);
    for(int i=0;i<N;i++)
    {
    	int num;
    	scanf("%d",&num);
    	
    	if(i==0)
    	sum.push_back(num);
    	else
    	sum.push_back(num+sum[i-1]);
    	
    }
    
    int n=sum.size();
    
    scanf("%d",&Q);
    
    for(int i=0;i<Q;i++)
    {
    	
    	int question;
    	scanf("%d",&question);
    	
    	int low=0,high=n-1,flag=0;
        
        while(low<=high)
        {
    	
    	  int mid=low+(high-low)/2;
           
          if(sum[mid]==question)
           {
           	printf("%d\n",mid+1);
           	flag=1;
           	break;
           }
          else if(sum[mid]<question)
          {
          	low=mid+1;
          }
          else if(sum[mid]>question)
          {
          	high=mid-1;
          }
          
          
        
       }
       
       if(flag==0)
          {
          	cout<<low+1<<endl;
          }
    
   }
  return 0;
}
