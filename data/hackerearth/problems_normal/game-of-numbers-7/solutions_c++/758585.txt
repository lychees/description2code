#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
    int t;
    long long x,i,y,j;
    cin>>t;
    while(t--)
    {
        bool b[1000001]={0};
        cin>>x>>y;
        for(i=2;i*i<1000000001;i++)
        {
            long long gap=i*i;
            long long add=x + (gap - x % gap) % gap;
            for(j=add;j<=y;j+=gap)
            {
                b[j-x]=1;
            }
        }
        int ans=0;
        for(j=x;j<=y;j++)
        {
            if(b[j-x]==0)
                ans++;
        }
        cout<<ans<<endl;
    }
    return 0;
}
