#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string p;
    getline(cin,p);
    while(t--)
    {
    	string r;
    	string s;
    	getline(cin,r);
    	getline(cin,s);
    	int a[26];
    	int c=0,d=0;
    	for(int i=0;i<26;i++)
    	a[i]=0;
    	for(int j=0;j<r.size();j++)
    	{
    		if(r[j]-' '==0)
    		{}
    		else
    		a[r[j]-'a']++;
    	}
    	for(int k=0;k<s.size();k++)
    	{
    		if(s[k]-' '==0)
    		{}
    		else
    		a[s[k]-'a']--;
    	}
    	for(int i=0;i<26;i++)
    	{
    		
    		if(a[i]>0)
    		c++;
    		if(a[i]<0)
    		d++;
    	}
    	if(c>0&&d<=0)
    	cout<<"You win some."<<endl;
    	if(d>0&c<=0)
    	cout<<"You lose some."<<endl;
    	if((c==0&&d==0)||(c>0&&d>0))
    	cout<<"You draw some."<<endl;
    }
    return 0;
}
