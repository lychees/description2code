#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long n;
        cin>>n;
        long long int a;
        cin>>a;
        if(a<n/2)
        {
            cout<<a*2+1<<endl;
        }
        else
        {
            cout<<((n-1)-a)*2<<endl;
        }
    }
    return 0;
}
