#include <iostream>

using namespace std;

int main() {
	long long int flag1=0, i,j, n,sum=0;
	cin >> n;
	long long int *a = new long long int[n];
	for(i=0; i<n; i++) {
		cin >> a[i];
		if(a[i]==i)
			sum++;
	}
	if(sum==n){
		cout<<n-2<<endl;
		return 0;
	}
	else {
	i=0;
	while(i<n){
		while(a[i]==i)
			i++;
		for(j=i+1;j<n;j++){
			while(a[j]==j)
				j++;
			if(a[i]==j && a[j]==i){
				cout<<sum+2<<endl;
				return 0;
			}
		}
		i++;
	}
	cout<<sum+1<<endl;
	}
	return 0;
}
