#include <bits/stdc++.h>
using namespace std;

int main()
{   int t,n,k,a[1001],temp[1001],i,j;
	cin>>t;
	while(t--){
		cin>>n>>k;
		for(i=0;i<n;i++) cin>>a[i];
		long long int msum=100000000;
		for(i=0;i<n;i++){
			memset(temp,0,sizeof(temp));
			int r=0,sum=0;
			for(j=0;j<n;j++){
				if(j!=i) {
					if(a[j]-a[i]>0) temp[r]=5*(a[j]-a[i]);
				     else temp[r]=3*(a[i]-a[j]);
				     //cout<<"::"<<temp[r]<<"::";
				     r++;
				}
			}
			sort(temp,temp+r);
			for(j=0;j<k-1;j++) sum+=temp[j];
			//cout<<sum<<endl;
			if(sum<msum) msum=sum;
		}
	
		cout <<msum<< endl;
	}
    
    return 0;
}
