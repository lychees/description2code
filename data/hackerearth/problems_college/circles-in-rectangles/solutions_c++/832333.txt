#include<bits/stdc++.h>
#define m 1000000007
#define ll long long int
#define gc getchar_unlocked

void scanint(ll &x)
{
    register ll c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

          ll t;

           scanint(t);
