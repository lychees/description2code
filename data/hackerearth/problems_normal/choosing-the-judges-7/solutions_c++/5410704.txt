#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int j=1;j<=t;j++)
    {
    	int n;
    	cin>>n;
    	long long a[n];
    	for(int i=0;i<n;i++)
    	 cin>>a[i];
    	long long b[n+1];
    	b[0] = 0;
    	b[1] = a[0];
    	for(int i=2;i<=n;i++)
    	 {b[i] = max(b[i-2]+a[i-1],b[i-1]); }
    	cout<<"Case "<<j<<": "<<b[n]<<endl;
    }
    return 0;
}
