#include<bits/stdc++.h>
using namespace std;
long long int gcd(long long int a,long long int b){
	if(b==0) return a;
	else return gcd(b,a%b);	
}
long long int swap(long long int &a,long long int &b){
	long long int temp=b;
	b=a;
	a=temp;
}
int main(){
	long long int t;
	scanf("%lld",&t);
	while(t--){
		long long int x,y,turn=1,temp;
		scanf("%lld %lld",&x,&y);
		string s;
		cin>>s;
		if(x<y) swap(x,y);
		while(1){
			temp=gcd(x,y);
			x-=temp;
			y-=temp;
			if(x==0 || y==0) break;
			if(turn==1) turn=2;
			else turn=1;
		}
		if(turn==1)
	    cout<<s<<endl;
		else{
			if(s.compare("Chandu")==0)
			cout<<"Daspal"<<endl;
			else
			cout<<"Chandu"<<endl;
		}
	}
	return 0;
	}
