#include<bits/stdc++.h>
#include <stdio.h>
#include <stdlib.h>
using namespace std;

char fun(int code){
char a;
if(code==10)
	a='U';
else if(code==11)
	a='V';
else if(code==12)
	a='W';
else if(code==13)
	a='X';
else if(code==14)
	a='Y';
else if(code==15)
	a='Z';
else{
	 a='0'+code;
	}	
	return a;
}

int main(){
int T;
cin>>T;
int r,g,b;
int div,mod;
int i=1;
while(T--){
cin>>r>>g>>b;
div=r/16;
mod=r%16;

cout<<"Case "<<i<<": "<<fun(div)<<fun(mod)<<" ";
div=g/16;
mod=g%16;
cout<<fun(div)<<fun(mod)<<" ";
div=b/16;
mod=b%16;
cout<<fun(div)<<fun(mod)<<endl;
i++;
}
return 0;
}
