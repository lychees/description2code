#include<iostream>
#include<queue>
using namespace std;
int main()
{
	long long n,a;
	priority_queue<long long>pq;
	cin>>n;
	for(long i=0;i<2;i++)
	{
		cin>>a;
		pq.push(a);
		cout<<"-1"<<"\n";
	}
	for(long i=2;i<n;i++)
	{
		long long pdt=1;
		cin>>a;
		pq.push(a);
		long long first=pq.top();
		pq.pop();
		long long second=pq.top();
		pq.pop();
		long long third=pq.top();
		pq.pop();
		pdt=first*second*third;
		pq.push(first);
		pq.push(second);
		pq.push(third);
		cout<<pdt<<"\n";
		
	}
	return 0;
	
}
