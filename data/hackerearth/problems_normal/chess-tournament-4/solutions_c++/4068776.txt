#include <iostream>
using namespace std;
int main()
{	int n,A[1024][1024],win[1024],l,m;
	cin>>n;
	int k=1;
	for(int i=0;i<n;++i)
	  k*=2;
	for(int i=0;i<k-1;++i)
	{
		for(int j=0;j<=i;++j)
			cin>>A[i][j];
	}
	
		for(int j=0;j<k;++j)
			win[j]=j+1;
			
	//win[0]=1;win[1]=2;
	for(;k>1;k=k/2)
	{	l=0;m=0;
		for(int j=win[l]-1,x=win[l+1]-2;l<k;l+=2)
		{	
			j=win[l]-1;x=win[l+1]-2;
			if(A[x][j]==0)
				win[m]=win[l];
			else
				win[m]=win[l+1];
				++m;
		//	cout<<win[m-1]<<"\n";
		}
	/*	k=k/2;
		if(k<=1)
			break;*/
			
		
	}
	cout<<win[0];
   // printf("Hello World!\n");
    return 0;
}
