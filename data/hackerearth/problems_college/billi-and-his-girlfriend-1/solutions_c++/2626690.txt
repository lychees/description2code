#include<cstdio>
#define ull unsigned long long
using namespace std;
ull compute(ull a,ull b,ull k)
{
	ull an=1,m=1;
	for(int i=0;i<k;i++)
		m=m*10;
	while(b!=0)
	{
		if(b%2==1)
			an=((an%m)*(a%m))%m;
		a=((a%m)*(a%m))%m;
		b=b/2;
	}
	return an;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		ull a,b,k;
		scanf("%llu %llu %llu",&a,&b,&k);
		ull an=compute(a,b,k);
		ull sum=0;
		while(an!=0)
		{
			sum=sum+(an%10);
			an=an/10;
		}
		printf("%llu\n",sum);
	}
	return 0;
}
