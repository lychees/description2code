#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;

int main()
{
	int a[200];
	int str,n,i,b;
	cin>>b;
	while(b--)
	{
		cin>>n;
		a[0]=1;
		int m=1;
		int carry=0;
		for(i=1; i<=n; i++)
		{
			for(int j=0; j<m; j++)
			{
				str=(a[j]*i)+carry;
				a[j]=str%10;
				carry=str/10;
			}
			while(carry>0)
			{
				a[m]=carry%10;
				carry=carry/10;
				m++;
			}
		}
		for(int i=m-1; i>=0; i--)
		{
			cout<<a[i];
		}
		cout<<"\n";
	}
	
}
