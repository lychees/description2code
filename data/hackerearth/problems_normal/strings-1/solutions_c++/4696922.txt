#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t>0)
	{
		string s1,s2;
		cin>>s1>>s2;
		if(s1.size()==1 && s2.size()==1)
		{
			if((s1.at(0)=='2' && s2.at(0)=='4')||(s1.at(0)=='4' && s2.at(0)=='2')||(s1==s2))
				cout<<"YES\n";
			else
				cout<<"NO\n";
		}
		else if(s1==s2)
		{
			cout<<"YES\n";
		}
		else
			cout<<"NO\n";
		t--;
	}
}
