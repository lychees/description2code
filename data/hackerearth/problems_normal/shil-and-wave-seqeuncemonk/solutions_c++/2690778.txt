using namespace std;
 
typedef unsigned long long int ull;
typedef long long int ll;
typedef vector<vector<ll> > mat;

int n,a[100005];ll m=1e9+7,bit[2][300005];

ll f(int i,int idx)
{
	ll s=0;
	while(i>0)
	{
		s+=bit[idx][i];
		s%=m;
		i-=(i&(-i));
	}
	return s;
}

void upd(int i,int idx,ll v)
{
	while(i<=100000)
	{
		bit[idx][i]+=v;
		bit[idx][i]%=m;
		i+=i&(-i);
	}
}


int main()
{ios_base::sync_with_stdio(false);
cin>>n;
rep(i,1,n)
cin>>a[i];
upd(a[1],0,1);
upd(a[1],1,1);
//ol(f(1,0));on;
rep(i,2,n)
{
	
	ll a1=f(100000,0)-f(a[i],0);
	//ol(a1);os;
	upd(a[i],1,a1);
	a1=f(a[i]-1,1);
	upd(a[i],0,a1);
	//ol(a1);on;
	upd(a[i],0,1);
    upd(a[i],1,1);
}
ll ans=(f(100000,0)+f(100000,1))%m;
ol((ans-2*n+m)%m);
return 0;
}
