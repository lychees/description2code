#include <cstdio>
#include <cmath>
#include <cstring>

int calc(int x){
	int i = 2;
	int res = 0;
	
	for (; x % i && i <= sqrt(x); i++);
	if (i > sqrt(x))
		return 1;
	
	while (x % i == 0){
		x /= i;
		res++;
	}
	
	for (i++; i <= x; i++)
		if (x % i == 0){
			while (x % i == 0){
				res++;
				x /= i;
			}
		}
		
	return res;
}

int main(){
	int n, x, res = 0;
	scanf("%d", &n);
	
	scanf("%d", &x);
	if (x != 1)
		res = calc(x);
	for (int i = 1; i < n; i++){
		scanf("%d", &x);
		if (x != 1)
			res ^= calc(x);
	}
	
	printf("%s", res ? "ALICE" : "BOB");
	
	return 0;
}
