#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	int t,i;
	long long int n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		long long int a[n];
		for(i=0;i<n;i++)
		  cin>>a[i];
		sort(a, a + n );
		long long int b[n/2];
		int j=0;
		for(i=0;i<n;)
		{
			b[j]=a[i]*a[i+1];
			j++;
			i=i+2;
		}
		long long int sum=0;
		for(i=0;i<n/2;i++)
		   sum=sum+b[i];
		cout<<sum<<"\n";
	}
    return 0;
}
