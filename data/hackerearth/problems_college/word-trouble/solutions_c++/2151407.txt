#include <bits/stdc++.h>
using namespace std;
//map<char,int> my;
int main()
{
  int t,i;
  string a,b;
  cin>>t;
  
  while(t--)
    {int c=0;
   // my.clear();
    	cin>>a>>b;
    	int j=0;
    	i=0;
    while(i<a.size() && j<b.size())
    {
    	if(a[i]!=b[i])
    	 		c++;
    		i++;
    		j++;
    		
    }
    if(i<a.size())
    c+=a.size()-i;
    
    if(j<b.size())
    c+=b.size()-j;
    if(c==0)
    cout<<"Exact"<<endl;
    else
    cout<<"Non Exact "<<c<<endl;
    	 // cout<<c<<endl;
    }
  
    return 0;
}
