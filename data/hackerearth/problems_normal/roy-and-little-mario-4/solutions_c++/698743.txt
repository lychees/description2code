#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
long long int a[100005];
int main()
{
    int t,i;
    a[0]=a[1]=0;
    a[2]=1;
    for(i=3;i<100004;i++)
    {
    	a[i]=(a[i-1]+a[i-2]+a[i-3])%MOD;
    	
    }
    scanf("%d",&t);
    while(t--)
    {
    	int  n;
    	scanf("%d",&n);
    	printf("%lld\n",a[n+2]);
    }
	return 0;
}
