#include <iostream>
#include<string.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	string s[n];
	for(int i=0;i<n;++i)
  cin>>s[i];
  for(int i=0;i<n;++i){
  	for(int j=i+1;j<n;++j){
  		if(s[i].size()!=s[j].size())
  		continue;
  		bool flag=false;
  		for(int k=0,h=s[i].size()-1;k<s[j].size();++k,--h)
  		{
  			if(s[i][k]!=s[j][h]){
  				flag=true;
  				break;
  			}
  		}
  		if(!flag)
  		cout<<s[i].size()<<" "<<s[i][s[i].size()/2];
  	}
  }
   
 
}
