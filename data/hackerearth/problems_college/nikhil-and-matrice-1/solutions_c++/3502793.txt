int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int i, k = 0, l = 0,n,m,c=1,nn;
		scanf("%d",&n);
		m=n;
		nn=n;
 		int a[n][n];
   
 
    while (k < m && l < n)
    {
        for (i = l; i < n; ++i)
        {
            a[k][i]=c++;;
        }
        k++;
 
        for (i = k; i < m; ++i)
        {
            a[i][n-1]=c++;
        }
        n--;
 
        if ( k < m)
        {
            for (i = n-1; i >= l; --i)
            {
                a[m-1][i]=c++;
            }
            m--;
        }
 
        if (l < n)
        {
            for (i = m-1; i >= k; --i)
            {
                a[i][l]=c++;
            }
            l++;    
        }        
    }
    for(i=0;i<nn;i++)
    {
    	for(int j=0;j<nn;j++)
    		printf("%d",a[i][j]);
    	printf("\n");
	}
	printf("\n");
	}
	return 0;
}
