t = []
for _ in range(input()):
	t1,t2 = raw_input().split('-')
	h1,m1 = map(int,t1.split(':'))
	h2,m2 = map(int,t2.split(':'))
	s = h1*60 + m1
	e = h2*60 + m2
	t.append((s,e))
t.sort()
flag = True
for i in range(0,len(t)-1):
	if t[i][1] > t[i+1][0]:
		flag = False
		break
if flag:
	print "Who needs a moderator?"
else:
	print "Will need a moderator!"
