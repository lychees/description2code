#include<bits/stdc++.h> 
#include<cassert>
using namespace std;
char buffer[4000040];
 
string Find(string s) {
  int n = s.length();
  s += s;
  int i, j, k, l;
  for (i = 0, j = 1; j < n; ) {
    for (k = 0; k < n && s[i + k] == s[j + k]; ++k);
    if (k >= n) break;
    if (s[i + k] > s[j + k]) {
      j += k + 1;
    } else {
      l = i + k;
      i = j;
      j = std::max(l, j) + 1;
    }
  }
  return s.substr(i, n);
}
 
int main() {
  int t = 1;
  while (t--) {
    assert(scanf(" %s", buffer) == 1);
    printf("%s\n", Find(buffer).c_str());
  }
  return 0;
}
