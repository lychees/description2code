#include <bits/stdc++.h>
using namespace std;

#define LL long long

LL a[112345];

inline void Solve(){
	LL n, k, p, i, j;
	cin>>n>>k>>p;
	for(i = 1; i <= n; i++){
		cin>>a[i];
	}
	sort(a + 1, a + n + 1);
	for(i = 1; i <= n; i++){
		j = i + k - 1;
		if(j > n){
			break;
		}
		if(a[j] - a[i] <= 2 * p){
			cout<<"NO"<<endl;
			return;
		}
	}
	cout<<"YES"<<endl;
	return;
}

int main(){
	int t;
	cin>>t;
	while(t--){
		Solve();
	}
	return 0;
}
