#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int T;
    cin >> T;
    for(int h=0; h<T; h++){
    	long long int a, b, c, d;
    	long long int k;
    	cin >> a >> b >> c >> d >> k;
    	int r = cbrt(k/a);
    	long long int f = a*r*r*r + b*r*r + c*r + d;
    	while(f > k && r!=0){
    		r--;
    		f = a*r*r*r + b*r*r + c*r + d;
    //	cout<<"f "<<f<<" \n ";
    	}
    	//cout<<"f "<<f<<"  ";
    	if(k != d)
    	cout<<r<<"\n";
    	else
    	cout<<"0\n";
    }
    return 0;
}
