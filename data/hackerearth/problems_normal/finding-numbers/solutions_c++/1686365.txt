#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
    int T;
    cin >> T;
    while(T--)
    {
    	long long int z,n,m,i,j,count;
    	cin >> z >> m >> n;
    	count=0;
    	for(i=1 ; i<=sqrt(z) ; i++)
    	{
    		j=z/i;
    		if(i*j==z)
    		{
    			if(j<=m)
    			count++;
    			if(j<=n && j!=i)
    			count++;
    		}
    	}
    	cout << count << endl;
    }
    return 0;
}
