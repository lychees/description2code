#include <iostream>
#include<cmath>
#include <iomanip>
using namespace std;
const double pi = 3.1416;
double radius_of(int x)
{
	double n;
	n = x/pi;
	return sqrt(n/4);
}
int main() {
	int t;
	cin>>t;
	while(t--)
	{
		int n,i,x;
		double sum = 0;
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>x;
			sum+=radius_of(x);
		}
		cout<<setprecision(4)<<fixed<<4*pi*sum*sum<<endl;
	}
	return 0;
}
