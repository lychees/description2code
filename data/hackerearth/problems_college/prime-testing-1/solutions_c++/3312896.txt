#include<bits/stdc++.h>
using namespace std;
int a[1000005]={};
int b[1000003];
int main()
{
	int i,j,n,t,c;
	a[1]=1,a[0]=1;
	for(i=2;i<=1003;i++)
			{
				if(a[i]==0)
				{
					for(j=i*i;j<=10002;j+=i)
					{
						a[j]=1;
				
					}	
				}
			}

	scanf("%d",&t);
		while(t--)
		{
			scanf("%d",&n);
			int c=0;
				for(i=0;i<n;i++)
			{
				scanf("%d",&b[i]);
			if(a[b[i]]==0)
	 		c++;
			}		
		
	printf("%d\n",c);
}
	return 0;

}
