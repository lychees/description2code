#include <iostream>
using namespace std;
typedef long long int LL;
LL N;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	scanf("%lld",&N);
    	if(N == 0){
    		puts("0");
    	}else if(N<3){
    		puts("1");
    	}else if(N%3 == 0){
    		// for 3*k numbers
    		printf("%lld\n",N/3);
    	}else{
    		// for 3*k + 1 numbers
    		// ans = n/3 + 1 + n/3 + n/3
