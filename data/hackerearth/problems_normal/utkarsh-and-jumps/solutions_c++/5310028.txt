#include <bits/stdc++.h>
using namespace std;
double s,f;double val[1000005];int done[1000005];
double find(int st,int end){
    if(done[st]!=-1)
        return val[st];
	if(st>end)
		return 0;
	if(st==end)
		return 1;
	val[st]= s*find(st+2,end)+f*find(st+3,end);
	done[st]=1;
    return val[st];
}

int main()
{
    //cout << "Hello World!" << endl;
    memset(done,-1,sizeof(done));
    //cout<<done[0];
    int n,p;
    cin>>n>>p;
    s=p/100.0;
    f=1-s;
    //cout<<s<<" "<<f;
    double ans=find(0,n);
    cout.setf(ios::fixed);

    cout<<setprecision(6)<<ans;
    //printf("%.6f",ans);
    return 0;
}
