#include <iostream>
#include<string>
using namespace std;
typedef long long int ll;
int main()
{
    string s;
    int k;
    cin>>s;
    cin>>k;
    int t;
    cin>>t;
    ll c  = 0;
    ll l = s.size();
    while (t--) {
    	
    	string str;
    	cin>>str;
    	ll l2 = str.size();
    	ll j = 0;
    	for(ll i = 0; i < l ; i++ )
    	{
    		if(str[j] == s[i]) {
    			j++;
    			if(j == l2)
    				break;
    		}
    	}
    	if(j  == l2)
    		c++;
    	
    }
    if(c >= k) {
    	cout<<c<<" "<<"LUCKY"<<endl;
    } else {
    	cout<<c<<" UNLUCKY"<<endl;
    }

    return 0;
}
