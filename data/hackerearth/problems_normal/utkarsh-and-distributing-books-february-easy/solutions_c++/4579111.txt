#include <iostream>
using namespace std;

int main()
{
    int n,a[1000],t,i;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	int min=999,max=0;
    	
    	for(i=0;i<n;i++)
    	{
          if(a[i]<min)
          min=a[i];
    	}
    	
    	for(i=0;i<n;i++)
    	{
    		max=max+(a[i]-1);
    	}
    	
    	cout<<min-1<<" "<<max;
    	cout<<endl;
    }
    return 0;
}
