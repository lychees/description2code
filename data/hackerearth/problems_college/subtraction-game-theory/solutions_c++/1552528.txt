#include<iostream>
using namespace std;
int sod(int n)
{
	int c=0;
	while(n!=0)
	{
		n%=10;
		c+=n;
		n/=10;
	}
	return c;
}
int main()
{
	int n,sum=0,sum1=0;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];		
	}
	for(int i=0;i<n;i++)
	{
		sum+=a[i];
	}
	for(int i=0;i<n;i++)
	{
		sum1+=sod(a[i]);
	}
	sum=sod(sum);
	sum%=9;
	sum1%=9;
	cout<<"0"<<endl;
	return 0;
}
