#include <iostream>
using namespace std;
long long int reverse(long long int  number)
{
	long long int answer=0;
	while(number)
	{
		answer=answer*10+number%10;
		number/=10;
	}
	return answer;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int a ,b,result;
    	cin>>a>>b;
    	if(a==b)
    	result =2*reverse(a);
    	else
    	result=reverse(a)+reverse(b);
    	cout<<reverse(result)<<endl;
    }
    return 0;
}
