#include<bits/stdc++.h>
using namespace std;
int bin(int *arr, int l, int r, int ele)
{
	int m;
	int flag=-1;
	while(l<=r)
	{
		//printf("l=%d r=%d m=%d\n",l,r,m);
		m=(l+r)/2;
		if(arr[m]==ele)
		{
			flag=m;
			break;
		}
		else if(arr[m]>ele)
		r=m-1;
		else l=m+1;
	}
	return flag;
}
int main()
{
	int n,tmp;
	cin>>n;
	int arr[n];
	for(int i=0;i<n;++i)
	{
		cin>>arr[i];
	}
	long int i,j,k,diff,flag,ck;
	cin>>k;
	sort(arr,arr+n);
	flag=0;
	for(i=0;i<n;++i)
	{
		for(j=i+1;j<n;++j)
		{
			diff=k-(arr[i]+arr[j]);
			ck=bin(arr,0,n-1,diff);
			if(ck!=-1 && ck!=i && ck!=j)
			{
				flag=1;
				break;
			}
		}
		
	}
	if(flag!=1 ) cout<<"NO"<<endl;
	else if(flag==1) cout<<"YES"<<endl;
//	for(i=0;i<n;++i)
//	printf("%d ",arr[i]);
	return 0;

}
