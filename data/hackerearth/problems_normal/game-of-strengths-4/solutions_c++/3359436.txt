#include <iostream>
#include<algorithm>
#define ll long long int
using namespace std;

int main()
{
    ll t,d=1000000007;
    cin>>t;
    while(t--){
    	ll n;
    	cin>>n;
    	ll a[n],b[n-1];
    	for(ll i=0;i<n;i++){
    		cin>>a[i];
    	}
    	sort(a,a+n);
    	ll sum1=0;
    	for(ll i=0;i<n-1;i++){
    		b[i]=(a[i+1]%d-a[0]%d)%d;
    		sum1=((sum1%d)+(b[i]%d))%d;
    		//cout<<b[i]<<endl;
    	}
    	ll x=b[0],sum=sum1;
    	for(int i=1;i<n-1;i++)
    	{
    		sum=(sum%d+(sum1-x-(n-1-i)*b[i-1])%d)%d;
    		//cout<<sum<<endl;
    		x=(b[i]%d+x%d)%d;
    	}
    	cout<<(sum*a[n-1])%1000000007<<endl;
    }
}
