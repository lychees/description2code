#include <iostream>
using namespace std;

int main()
{
    long long int a[42],i,test,c,sum[41];
    a[0]=0;a[1]=1;
    sum[0]=0;
    sum[1]=1;
    for(i=2;i<41;i++)
    {
    	a[i]=a[i-1]+a[i-2];
    	sum[i]=sum[i-1]+a[i];
    }
    cin>>test;
    while(test--)
    {
    	cin>>c;
    	cout<<sum[c-1]+1<<endl;
    }
    return 0;
}
