#include<bits/stdc++.h>
using namespace std;
bool isPrime(long long n) 
{
    if(n<=3)
        return n>1;
	else if(n%2==0||n%3==0)
        return false;
	else
	{
        for(long long i=5;i*i<=n;i+=6)
		{
            if(n%i==0||n%(i+2)==0)
                return false;
        }
        return true;
    }
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n;
		scanf("%lld",&n);
		if(isPrime(n))
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
