def fact(n):
	index=0
	#a=[0 for i in xrange(200)]
	a=[]
	a.append(1)
	while(n>=2):
		temp=0
		i=0
		while(i<=index):
			temp=a[i]*n+temp
			a[i]=temp%10
			temp=temp/10
			i+=1
		while temp >0:
			a.append(temp%10)
			temp/=10
			index+=1
		n-=1
	number=''.join(str(e) for e in a)
	print number[::-1]
