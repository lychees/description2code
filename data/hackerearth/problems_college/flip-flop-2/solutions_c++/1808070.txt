#include<bits/stdc++.h>
using namespace std;

int nits(string s  , int l)
  {
       int i,d=0;
       for(i=0;i<l;i++)
          {
             // cout<<i<<" "<<l<<"\n";
             // cout<<s[i]<<" "<<s[i+1]<<"\n";
              if(s[i]==s[i+1]) d++;
              else continue;
          }
          return d;
  }
