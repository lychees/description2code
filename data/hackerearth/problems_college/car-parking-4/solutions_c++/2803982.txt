#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <functional>
#include <algorithm>

using namespace std;
int nlist[100005];
int main()
{
	int tc;
	cin >> tc;
	string s;
	int slen,count,i;
	while(tc--)
	{
		count = 0;
		memset(nlist,0,100005);
		cin >> s;
		slen = s.length();
		
		for(int i=0;i<slen;i++)
		{
			if(s[i] == '-')
			{
				if(nlist[i] > -1)
				{nlist[i] = 1;}
				
			}
			else if(s[i] == 'D')
			{
				nlist[i] = 0;
			}
			else if(s[i] == 'S')
			{
				if((i-1) >= 0)
				{
					nlist[i-1] = 0;
				}
				nlist[i] = 0;
				nlist[i+1] =-1;
			}
			else 
			{
				if(i-2>=0)
				{
					nlist[i-2] = 0;
				}
				nlist[i-1] = 0;
				nlist[i] =0;
			}
		}
	
		for(int j =0 ; j<slen ;j++)
		{
			if(nlist[j] > 0)
			{
				count = count + nlist[j];
			}
		}
		
		cout << count << endl;
	}
	return 0;
}
