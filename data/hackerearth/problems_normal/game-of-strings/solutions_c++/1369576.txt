int lcs[5005][5005];
int LCS(char a[5005],char b[5005],int l1,int l2){
	int i,j;
	for(i=0;i<=l1;i++)lcs[i][0]=0;
	for(i=0;i<=l2;i++)lcs[0][i]=0;
	for(i=1;i<=l1;i++){
		for(j=1;j<=l2;j++){
			if(a[i-1]==b[j-1]){
				lcs[i][j]=lcs[i-1][j-1]+1;
			}
			else{
				lcs[i][j]=max(lcs[i-1][j],lcs[i][j-1]);
			}
		}
	}
	return lcs[l1][l2];
}
