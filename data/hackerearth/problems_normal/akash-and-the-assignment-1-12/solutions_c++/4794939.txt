
#include<iostream>
#include<vector>
#include <unordered_map>
#include<string>
using namespace std;
int main() {
  int n, q;
  cin >> n >> q;
  long long int l, r, k;
  string s;
  cin >> s;
  unordered_map<int, unordered_map<char, int>> record;

  for (int i = 0; i < s.size(); i++) {
    for (char j = 'a'; j <= 'z'; j++) {
      record[i][j] += record[i - 1][j];
    }
    record[i][s[i]]++;
  }
  while (q--) {
    cin >> l >> r >> k;

    if (r - l + 1 < k) {
      cout << "Out of range" << endl;
      continue;
    }

    int count = 0;
    for (char i = 'a'; i <= 'z'; i++) {
      count += (record[r - 1][i] - record[l - 2][i]);
      if (count >= k) {
        cout << i << endl;
        break;
      }
    }
  }


  return 0;

}
