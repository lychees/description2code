#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	long long n;
    	cin>>n;
    	long long a[n],i,j;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	long long ans = n;
    	for(i=0;i<n;i++)
    	{
    		for(j=i+1;j<n;j++)
    		{
    			if(a[i]==a[j])
    				ans++;
    		}
    	}
    	cout<<ans<<endl;
    }
    
    return 0;
}
