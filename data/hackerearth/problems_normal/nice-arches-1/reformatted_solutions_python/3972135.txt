import threading
M = int(raw_input())

count = 0


def parser(word):
    global count
    while word and (word.find('AA') != -1 or word.find('BB') != -1):
        word = word.replace('AA', '')
        word = word.replace('BB', '')
    if not word:
        count += 1


q = []
for i in range(M):
    word = raw_input()
    t = threading.Thread(target=parser, args=(word,))
    t.start()
    q.append(t)

for j in q:
    j.join()

print count
