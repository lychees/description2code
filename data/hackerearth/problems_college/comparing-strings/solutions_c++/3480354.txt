#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	char ch[100000],bh[100000];
   	cin>>ch>>bh;
   	int i = 0;
   	int flag = 0;
   	while(ch[i] != '\0')
   	{int j = 0;
   		while(bh[j] != '\0')
   		{
   			if(ch[i] == bh[j])
   			{
   				flag = 1;
   				cout<<"YES"<<endl;
   				break;
   			}
   			j++;
   		}
   		if(flag == 1)
   		{
   			break;
   		}
   		i++;
   	}
   	if(flag == 0)
   	{
   		cout<<"NO\n";
   	}
   }
    return 0;
}
