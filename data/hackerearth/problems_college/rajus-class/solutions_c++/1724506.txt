#include <iostream>
#include <map>
#include <algorithm>
using namespace std;

struct node {
	int age;
	string name;
};
bool comp(node a, node b)
{
	if (a.age < b.age) return true;
	else if (a.age == b.age) {
		if (a.name <= b.name) return true;
		return false;
	}
	return false;
}
int main()
{
    int t;
    cin >> t;
    while (t--) {
    	int n;
    	cin >> n;
    	node p[n];
    	for (int i = 0; i < n; i++) {
    		cin >> p[i].age >> p[i].name;
    	}
    	sort(p, p + n, comp);
    	map <int, string> m;
    	for (int i = 0; i < n; i++) {
    		m[i+1] = p[i].name;
    	}
    	int r;
    	cin >> r;
    	cout << m[r] << endl;
    }
    return 0;
}
