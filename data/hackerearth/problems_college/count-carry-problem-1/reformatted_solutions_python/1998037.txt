t = int(input())
for tc in range(t):
    a, b = [list(i) for i in raw_input().split()]
    if len(a) < len(b):
        x, y = a, b
    else:
        y, x = a, b
    count = 0
    for i, j in enumerate(x):
        if int(j)+int(y[i]) > 9:
            count = count+1
    if count == 0:
        print('No carry operation')
    elif count == 1:
        print('1 carry operation')
    else:
        print(str(count)+' carry operations')
