#include<bits/stdc++.h>
using namespace std;
int dp[1005][1005];
int func(int i,int w,int arr1[],int arr2[],int n)
{
    if(i>=n)
        return 0;
    if(dp[i][w]!=-1)
        return dp[i][w];
    if(w-arr1[i]>=0)
        return dp[i][w]=max(arr2[i]+func(i+1,w-arr1[i],arr1,arr2,n),func(i+1,w,arr1,arr2,n));
    else
        return dp[i][w]=func(i+1,w,arr1,arr2,n);
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        memset(dp,-1,sizeof(dp));
        int n,w;
        cin>>n>>w;
        int arr1[n];
        int arr2[n];
        for(int i=0;i<n;i++)
            cin>>arr1[i];
        for(int i=0;i<n;i++)
            cin>>arr2[i];
        cout<<func(0,w,arr1,arr2,n)<<endl;
    }
}
