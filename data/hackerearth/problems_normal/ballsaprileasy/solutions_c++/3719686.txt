#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long n;
    std::ios::sync_with_stdio(false);
    cin>>n;
    long long a[n];
    long long i,j;
    for(i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    j=1;
    long long balls=0;
    for(i=0;i<n;i++)
    {
        while(a[i]>j*j)
        {
            balls+=j*j;
            j++;
        }
        balls+=(j*j-a[i]);
        j++;
    }
    cout<<balls;
    return 0;
}
