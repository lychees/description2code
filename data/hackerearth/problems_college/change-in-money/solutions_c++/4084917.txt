#include<bits/stdc++.h>
using namespace std;
long int f(long int A[],long int N,long int n)
{
	long int T[N+1],i,j;
	memset(T,0,sizeof(T));
	T[0]=1;
	for(i=0;i<n;i++)
	{
		for(j=A[i];j<=N;j++)
		{
			T[j]=T[j]+T[j-A[i]];
		}
	}
	return T[N];
}
 int main()
 
 {
	long int N,K,i,j,A[100001];
	cin>>N>>K;
	for(i=0;i<K;i++)
		cin>>A[i];
    	cout<<f(A,N,K);
}
