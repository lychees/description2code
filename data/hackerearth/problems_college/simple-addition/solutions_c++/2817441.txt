#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
     {
     	int l;
     	cin>>l;
     	int x=l/2,y=l-l/2,flag=0,ans=10000000;
     	while(y<l-1)
     	 {
     	 	int u=y,v=x;
     	 	int cnt=1;
     	 	while(u!=1||v!=1)
     	 	 {
     	 	 	if(u==v)
     	 	 	 break;
     	 	 	int m=u,n=v;
     	 	 	u=max(m-n,n);
     	 	 	v=min(m-n,n);
     	 	 	cnt++;
     	 	 }
     	 	 if(u==1&&v==1)
	     	 {	  
	     	 	  flag=1;
	     	 	  ans=min(ans,cnt);
	     	 }
     	 	 y++;
     	 	 x--;
     	 }
     	 cout<<min(ans,l-1)<<"\n";
     }
}
