#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <cstdlib>
#include <cstring>
#include <cmath>

#define REALLYLARGE 1000000007
using namespace std;

vector<int> primes;
bool HOLDER[2200000];
void sieve()
{
    memset(HOLDER, false, sizeof HOLDER);

    for(int i = 2; i <= 1449; i++)
    {
	if(HOLDER[i])
	    continue;
	for(int j = 2*i; j <= 2100000; j += i)
	    HOLDER[j] = 1;
    }
    for(int i = 2; i <= 2100000; i ++)
	if(!HOLDER[i]) primes.push_back(i);
}


int main()
{
    sieve();
    int t, a, b;
    cin >> t;
    while(t--)
    {
	cin >> a >> b;
	int mx = 0;
	int ans = 0;
	for(int n = a; n <= b; n ++)
	{
	    int tmp = n;
	    int div = 1;
	    int ps = 0;
	    while(primes[ps]*primes[ps] <= tmp)
	    {
		int i = primes[ps++];
		int cnt = 0;
		while(tmp%i == 0)
		{
		    cnt ++;
		    tmp /= i;
		}
		div *= (cnt + 1);
	    }
	    if(tmp != 1) div *= 2;
	    if(div > mx)
	    {
		mx = div;
		ans = 1;
	    }
	    else if(div == mx)
		ans ++;
	}
	cout << ans+1 << endl;
    }
    return 0;
}
