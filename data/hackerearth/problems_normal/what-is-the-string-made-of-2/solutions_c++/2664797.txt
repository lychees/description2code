#include <iostream>
#include <unordered_map>
using namespace std;

int main()
{
    unordered_map<int,int> f;
    f[1] = 2;
    f[2] = 5;
    f[3] = 5;
    f[4] = 4;
    f[5] = 5;
    f[6] = 6;
    f[7] = 3;
    f[8] = 7;
    f[9] = 6;
    f[0] = 6;
    string str;
    cin>>str;
    int cnt = 0;
    for(auto c:str) 
    	cnt += f[c-'0'];
    cout << cnt << endl;
    	
    return 0;
}
