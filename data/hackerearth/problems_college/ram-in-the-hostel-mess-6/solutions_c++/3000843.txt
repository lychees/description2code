#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int a[100000];
    while(t--)
    {
       int n;
       cin>>n;
       a[0] = 1;
       a[1] = 2;
       a[2] = 3;
       for(int i=3;i<=n;i++)
       a[i] = a[i-1] + a[i-2];
       cout<<a[n]*a[n]<<endl;
    }
    return 0;
}
