#include<bits/stdc++.h>
using namespace std;

int main(){
	int t;
	cin >> t;
	while(t--){
		int A[26]={},B[26]={};
		string s,s2;
		bool ok=false,ok2=false;
		cin.ignore();
		getline(cin, s);
		getline(cin, s2);
		
		for(int i=0;i<s.size();i++){
			if(s[i]!=' '){
				A[s[i]-'a']++;
			}
		}
		for(int i=0;i<s2.size();i++){
			if(s2[i]!=' '){
				B[s2[i]-'a']++;
			}
		}
		for(int i=0;i<26;i++){
			if(A[i]>B[i]) ok=true;
			else if(A[i]<B[i]) ok2=true;
		}
		
		if(ok&&ok2) cout << "You draw some." << endl;
		else if(ok) cout << "You win some." << endl;
		else if(ok2) cout << "You lose some." << endl;
	}
	return 0;
}
