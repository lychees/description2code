#include <bits/stdc++.h>

using namespace std;

long long int coin_count(int a[], int m, int n)
{
	long long int table[n+1];
	memset(table,0,sizeof(table));
	table[0] = 1;
	for (int i = 0; i < m; i++) {
		for (int j = a[i]; j <= n; j++) {
			table[j] += table[j-a[i]];
		}
	}
	
	return table[n];
}

int main()
{
	int m, n, i, j;
	long long ans;
	scanf("%d",&n);
	scanf("%d",&m);
	int a[m];
	for (i = 0; i < m; i++) {
		scanf("%d",&a[i]);
	}
	ans = coin_count(a, m, n);
	printf("%lld\n",ans);
	
	return 0;
}
