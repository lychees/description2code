#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int T;
    cin >> T;
    for(int i=0; i<T; i++){
    	int N;
    	cin >> N;
    	uint64_t res=1;
    	vector<int> vm(N);
    	for(int j=0; j<N; j++){
    		cin >> vm[j];
    	}
    	sort(vm.begin(), vm.end());
    	for(int j=0; j<N; j++){
    		res = (res*(vm[j]-j))%1000000007;
    		if(res<=0){
    			res=0;
    			break;
    		}
    	}
		
    	cout << res << endl;
    }
    return 0;
}
