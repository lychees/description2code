#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n, k;
    	cin>>n>>k;
    	int A[1005];
    	for(int i = 0; i < n; ++i)
    		cin>>A[i];
    	for(int i = 0; i < n && k; ++i)
    	{
    		int min = i;
    		for(int j = i+1; j < i+k+1 && j < n; ++j)
    		{
    			if(A[j] < A[min])
    				min = j;
    		}
    		int j;
    		int a = A[min];
    		for(j = min; j > i && k; --j)
    		{
    			A[j] = A[j - 1];
    			--k;
    		}
    		A[j] = a;
    	}
    	for(int i = 0; i < n; ++i)
    		cout<<A[i]<<" ";
    	cout<<endl;
    }
    return 0;
}
