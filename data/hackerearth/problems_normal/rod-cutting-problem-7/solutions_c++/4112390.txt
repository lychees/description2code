#include <iostream>
using namespace std;

#define ll long long

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	ll N;
    	cin>>N;
    	ll temp = 3;
    	ll count = 0;
    	while(temp <= N)
    	{
    		count++;
    		temp = temp*2 + 1;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
