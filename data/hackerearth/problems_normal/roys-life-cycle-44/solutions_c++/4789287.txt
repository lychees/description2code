	scanf("%d", &n);
	int day_so_far = 0;
	int n_so_far = 0;
	int preday = 0;

	while(n--) {
		scanf("%s", str);
		int buffer = 0;
		
		forall(i, 0, strlen(str)) {
			if (str[i] == 'C') {
				buffer += 1;
			} else {
				day_so_far = maX(day_so_far, buffer);
				n_so_far = maX(n_so_far, buffer + preday);

				preday = 0;
			}
		}
		day_so_far = maX(day_so_far, buffer);
		n_so_far = maX(n_so_far, buffer + preday);
