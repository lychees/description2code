#include <iostream>
#include <string.h>
using namespace std;

int main()
{
    char name[100];
    cin>>name;
    char cap_name[100];
    int val = int(name[0]);
    strcpy(cap_name, name);
    if((int( name[0]) > 96) && (int (name[0] < 123)))
    cap_name[0] = char( (val - 32) );
    
    for(int i=0; name[i] != '\0'; i++){
		if( (name[i] == '_') && (int( name[i+1]) > 96) && (int (name[i+1] < 123)) ){
			val = int ( name[i+1] );
			cap_name[i+1] = char( (val-32) );
		}
    }
    
    cout<<cap_name;
    return 0;
}
