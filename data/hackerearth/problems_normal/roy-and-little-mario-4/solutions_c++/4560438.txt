#include <iostream>
using namespace std;
#define mod 1000000007
int main()
{
    int t,n,maxm,i;
    int a[100001];
    cin>>t;
    a[1]=1;
    a[2]=2;
    a[3]=4;
    maxm=3;
    while(t--){
    	
    	cin>>n;
    	if(n<=maxm){
    		cout<<a[n]<<"\n";
    		continue;
    	}
    	for(i=maxm+1;i<=n;i++){
    		a[i]=((a[i-1]+a[i-2])%mod+a[i-3])%mod;
    	}
    	maxm=n;
    	cout<<a[n]<<"\n";
    	
    }
    return 0;
}
