#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int i, sum=0;
    string a;
    cin>>a;
    for(i=0;i<a.length();i++)
    {
    	if(a[i]=='1')
    		sum+=2;
    	if(a[i]=='2'||a[i]=='3'||a[i]=='5')
    		sum+=5;
    	if(a[i]=='4')
    		sum+=4;
    	if(a[i]=='6'||a[i]=='9'||a[i]=='0')
    		sum+=6;
    	if(a[i]=='7')
    		sum+=3;
    	if(a[i]=='8')
    		sum+=7;
    	}
    cout<<sum;
    return 0;
}
