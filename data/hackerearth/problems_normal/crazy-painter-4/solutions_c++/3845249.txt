#include <iostream>
using namespace std;

int main()
{
   long int t, a[26], i, j, w, m;
   long long int sum, tot, cost, n;
   
   cin>>t;
   
   for (i=0; i<t; i++)
   {
   		cost=0;
  		cin>>n;
  		sum=0;
  		for (j=0; j<26; j++)
  		{
  			cin>>a[j];
  			sum+=a[j];
  		}
  		
  		tot=12*n;
  		w=tot/26;
  		m=tot%26;
  
  			cost=w*(sum)+w*(w-1)*13;

  		
  		for (j=0; j<m; j++)
  		{
  			cost+=a[j];
  		}
  		
  		cost+=w*m;
  		
  		cout<<cost<<endl;
  		
   }
   
   return 0;
   
}
