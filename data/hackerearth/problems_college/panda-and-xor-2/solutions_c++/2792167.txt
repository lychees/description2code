#include <bits/stdc++.h>
using namespace std;
#define aa long long
int main()
{
    ios_base::sync_with_stdio(false);
    aa int tc;
    cin>>tc;
    while(tc--)
    {
        aa int a,b,ans;
        cin>>a>>b;
        ans = a^b;
        bitset<64> hmm(ans);
        if((hmm.count())&1) cout<<"NO"<<endl;
        else cout<<"YES"<<endl;
    }
}
