#include <iostream>
#include <cmath>
#include <stdio.h>
using namespace std;
const long long MAX = pow(10,6);

int main(){
	int a,b;
	int t;	
	bool isPrime[MAX+1];
	isPrime[0] = false;
	isPrime[1] = false;
	for(long i=2;i<=MAX;i++)
	    isPrime[i] = true;
	for(long i=2;i*i<=MAX;i++){
		if(isPrime[i] == true){
			for(long j=i;i*j<=MAX;j++)
			    isPrime[i*j] = false;
		}
	}
	int count[1000000];
	int primes = 0;
	for(int i=0;i<=MAX;i++){
		if(isPrime[i])
				primes++;
		count[i] = primes;
	}
	
	float result;
	cin>>t;
	while(t--){
		cin>>a>>b;
		result = ((float)(count[b]-count[a-1]))/((float)(b-a+1));
		printf("%1.6f\n",result);
	}
	return 0;
}
