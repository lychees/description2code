#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin >> t;
    long long int a,b,c,d;
    while(t--){
		cin >> a;
		b = sqrt(a);
		if(b*b == a){
			cout << "YES\n"; 
		}
		else{
			c = a - b*b;
			d = (b+1)*(b+1) - a;
			if(c<=d){
				cout << "-" << c << endl;
			}
			else{
				cout << "+" <<d <<endl;
			}
		}
    }
    return 0;
}
