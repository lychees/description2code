#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define ll long long
#define lim 10001

ll n,i,s,m;
ll A[lim],M[lim];

int main()
{
  cin>>n;
  for(i=0;i<n;i++)
  {
    cin>>A[i];
    s+=A[i];
  }
  s/=n;
  for(i=0;i+1<n;i++)
  {
    M[i]=A[i]-s;
    A[i+1]+=M[i];
  }
  sort(M,M+n);
  m=M[n/2];
  for(i=s=0;i<n;i++)
    s+=abs(M[i]-m);
  cout<<s<<"\n";
  return 0;
}
