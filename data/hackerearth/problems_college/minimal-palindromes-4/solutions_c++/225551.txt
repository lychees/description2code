int GCD(int a, int b)
{
	int r = 1;
	if(a < b)
	{
		int temp;
		temp = a;
		a = b;
		b = temp;
	}
	while(1)
	{
		r = a % b;
		if(r == 0)
		break;
		a = b;
		b = r;
	}
	return b;
	
}
