#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t>0)
    {
    	long int t1,t2,m,a=1,b=0;
    	cin>>m;
    	for(long int i=m;i>0;i--)
    	{
    		t1=b;
    		t2=a+b;
    		a=t1;
    		b=t2;
    	}
    	cout<<a<<" "<<b<<"\n";
    	t--;
    }
    return 0;
}
