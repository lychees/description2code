	for(i=0;i<n;i++){
		cin>>num;
		b.push_back(num);	
	}
	a.push_back(b[0]);
	
	for(i=1;i<n;i++){
		sum=0;
		for(j=0;j<i;j++){

		}

	}

		cout<<a[i]<<" ";
	}
	
    return 0;
