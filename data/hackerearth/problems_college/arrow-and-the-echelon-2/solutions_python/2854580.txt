def decent():
	from itertools import product
	for _ in xrange(input()):
		k = int(raw_input())
		a = list(product(range(3,6,2), repeat=k))
		
		flag = 0
		for i in range(len(a)-1,-1,-1):
			three = 0
			five = 0
			
			for x in a[i]:
				if x == 3:
					three += 1
				else:
					five += 1
			
			if three%5 == 0 and five%3 == 0:
				str1 = ''.join(str(e) for e in a[i])
				print str1
				flag = 1
				break
			
		if flag == 0:
			print '-1'
		

if __name__ == "__main__":
	decent()
