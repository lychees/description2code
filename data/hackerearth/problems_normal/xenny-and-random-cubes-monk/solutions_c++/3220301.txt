#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
#include <cstring>
#define ll long long int
const int MOD = 1e9+7;
using namespace std;
int freq[15][30];
int cubes[20];
int main()
{
    int N, K;
    scanf("%d %d", &N, &K);
    memset(freq, 0, sizeof(freq));
    char ch;
    string s;
    for(int i = 0;i<N;i++)
    {
    	for(int j = 0;j<6;j++)
    	{
    		scanf(" %c", &ch);
    		freq[i][ch-'a']++;
    	}
    }
    cin>>s;
    ll ans = 0;
    for(int mask = 1;mask<(1<<N); mask++)
    {
    	if(__builtin_popcount(mask) == K)
    	{
    		vector<int> cubes;
    		int val = mask;
    		int index = 0;
    		while(val)
    		{
    			if(val&1)
    			{
    				cubes.push_back(index);
    			}
    			val = val>>1;
    			index++;
    		}
    		sort(cubes.begin(), cubes.end());
    		do
    		{
    			ll ways = 1;
    			for(int i = 0;i<K;i++)
    			{
    				int chu = s[i] - 'a';
    				int cube = cubes[i];
    				if(freq[cube][chu] > 0)
    				{
    					ways = ways*freq[cube][chu];
    					ways %= MOD;
    				}
    				else
    				{
    					ways = 0;
    				}
    			}
    			ans += ways;
    			ans %= MOD;
    		}while(next_permutation(cubes.begin(),cubes.end()));
    		
    	}
    }
    printf("%lld", ans);
    
    return 0;
}
