#include <bits/stdc++.h>
#define ll long int
using namespace std;
int main()
{
	ll t, n, i, pre, in, old;
	scanf("%ld",&t);
	while(t--)
	{
		pre = in = old = 0;
		scanf("%ld",&n);
		int a[n];
   		for(i=0;i<n;i++)
	    	scanf("%d",&a[i]);
		for(i=0;i<n;i++)
		{
			pre = in;
			in = max(pre, old+a[i]);
			old = pre;
		}
		printf("%ld\n", in);
	}
	return 0;
}
