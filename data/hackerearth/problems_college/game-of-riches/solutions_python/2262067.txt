T = int(raw_input().strip())
while T>0:
	line = raw_input().strip()
	Peoples = int(line.split(" ")[0])
	Groups = int(line.split(" ")[1])
	List = map(int, raw_input().strip().split(" "))
	MyW = List[0]
	List.sort(reverse=True)
	MyG = List.index(MyW)
	JumpS = MyG%Groups
	count = 0
	for i in range(JumpS, Peoples-1, Groups):
		if List[i] > MyW:
			count = count+1
	print count
	T = T-1
