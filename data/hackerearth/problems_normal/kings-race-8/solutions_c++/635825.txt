int64 fn(int64 vl){
int64 lo,mid,hi,i;
lo = 0,hi = m+1;
while(lo<hi-1){
mid=(lo+hi)/2;
if(vl>=b[mid])lo=mid;
else hi = mid-1;
}
if(vl>=b[hi])lo=hi;
return lo;
for(i=0;i<=m+1;i++)
if(b[i]<=vl)continue;
else break;
return i-1;
}
