#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll arr[1000100],rem[1000100];
int main()
{
	ll t;
	scanf("%lld",&t);
	while(t--)
	{
		ll n;
		scanf("%lld",&n);
		ll i;
		for(i=1; i<=n; i++)
		{
			scanf("%lld",&arr[i]);
		}
		for(i=0; i<=n; i++)
		{
			rem[i]=0;
		}
		ll sum=0,length=INT_MAX,maxlength=INT_MAX,left,right;
		for(i=1; i<=n; i++)
		{
			sum=sum+arr[i];
			sum=sum%n;
			if(rem[sum]==0)
			{
				if(sum==0)
				{
					length=i;
				}
				else
				{
					length=INT_MAX;
				}
			}
			else
			{
				length=i-rem[sum];
			}
			if(length<maxlength)
			{
				maxlength=length;
				left=rem[sum]+1;
				right=i;
			}
			rem[sum]=i;
		}
		printf("%lld %lld\n",left,right);
	}
	return 0;
}
