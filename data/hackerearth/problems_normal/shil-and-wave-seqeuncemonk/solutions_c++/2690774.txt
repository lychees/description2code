int N,M=1e5;
int a[100010];
lli segtreeinc[400040];
lli segtreedec[400040];

void update(lli segtree[],int s,int e,int x,lli y,int idx)
{
    if(s==e)
        segtree[idx]=(segtree[idx]+y)%MOD;
    else
    {
        int mid=(s+e)>>1;
        if(x<=mid)
            update(segtree,s,mid,x,y,2*idx+1);
        else
            update(segtree,mid+1,e,x,y,2*idx+2);
        segtree[idx]=(segtree[2*idx+1]+segtree[2*idx+2])%MOD;
    }
}

lli query(lli segtree[],int s,int e,int x,int y,int idx)
{
    if(s==x && e==y)
        return segtree[idx];
    int mid=(s+e)>>1;
    if(y<=mid)
        return query(segtree,s,mid,x,y,2*idx+1);
    if(x>mid)
        return query(segtree,mid+1,e,x,y,2*idx+2);
    return (query(segtree,s,mid,x,mid,2*idx+1)+query(segtree,mid+1,e,mid+1,y,2*idx+2))%MOD;
}

        ans=(ans+x+y)%MOD;
