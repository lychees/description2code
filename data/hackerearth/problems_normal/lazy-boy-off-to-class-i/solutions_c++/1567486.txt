#include<cstdio>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
int main()
{
	long long int t,n,i,count,l,r;
	scanf("%lld",&t);
	while(t--)
	{
		vector<long long int> v;
		long long int sum=0;
       for(i=0;i<7;i++)
	   {
	   	scanf("%lld",&n);
	   	sum+=n;
	   	while(n--)
	   	{
	   		scanf("%lld%lld",&l,&r);
	   		v.push_back(r-l);
	   	}
	   }		
	   sort(v.begin(),v.end());
	   long long int per;
	   if(sum%4==0)
	   per=(3*sum)/4;
	   else
	   {
	   	 per=((3*sum)/4)+1;
	   }
	   long long int time=0;
	   for(i=0;i<per;i++)
	   {
	   	time+=v[i];
	   }
	   printf("%lld\n",(7*24*3600)-time);
	}
}
