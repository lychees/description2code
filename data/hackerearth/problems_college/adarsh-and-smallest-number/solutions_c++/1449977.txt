#include <bits/stdc++.h>

using namespace std;

#define ll long long int
#define s(a) scanf("%lld",&a)
#define pb push_back
#define mp make_pair
#define fr first
#define sc second
#define clr(a) memset(a,-1,sizeof(a))

vector<pair<ll,ll> > v[500001];
bitset<500001>visit;
ll mod=1000000007;
ll powers[500001]={0};
ll ans[500001]={0};

ll dow(ll a,ll b)
{
    ll tot=0;
    while(a%b==0) {
        tot++;
        a/=b;
    }
    return tot;
}

ll power(ll a,ll b)
{
    if(b==0) return 1;
    if(b==1) return a;
    ll k=power(a,b/2);
    if(b%2==0) return (k*k)%mod;
    else return (((k*k)%mod)*a)%mod;
}

void sieve()
{
    ll i,j;
    for(i=2;i<=500000;i++) {
        if(visit[i]==1) continue;
        for(j=i;j<=500000;j+=i) {
            visit[j]=1;
            v[j].push_back(mp(i,dow(j,i)));
        }
    }
    ans[1]=1;
    for(i=2;i<=500000;i++) {
        for(j=0;j<v[i].size();j++) {
            ans[i]=ans[i-1];
            if(v[i][j].second > powers[v[i][j].first]) {
                ans[i]=ans[i]*power(v[i][j].first,(v[i][j].second > powers[v[i][j].first]));
                ans[i]%=mod;
                powers[v[i][j].first]=v[i][j].second;
            }
        }
    }
}

int main()
{
    sieve();
    //freopen("in.txt","r",stdin);
    ll t,n,i;
    s(t);
    //t=0;
    //for(i=0;i<v[18].size();i++) cout<<v[18][i].fr<<" "<<v[18][i].sc<<endl;
    //cout<<endl;
    while(t--) {
        s(n);
        cout<<ans[n]<<endl;
    }
    return 0;
}
