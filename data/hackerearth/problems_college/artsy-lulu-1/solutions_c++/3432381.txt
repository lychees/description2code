	int a,end=0;
	while(cin>>let>>a)
	{
		int ctr;
		for(int x=0; x<alpha.size(); x++)
		{
			if(tolower(let) == alpha[x]) {ctr = x; break;}
		}
		if(end > 0) cout<<endl<<endl; end++;
		if(a==1) cout<<"INVALID";
		else if(ctr+a > 26) cout<<"INVALID";
		else
		{
			for(int x=0; x<a; x++)
			{
				if(x>0) cout<<endl;
				for(int y=0; y<x+1; y++)
				{
					if(y>0) cout<<' ';
					if(isupper(let)) cout<<big[ctr];
					else cout<<alpha[ctr];
				}
				ctr++;
			}
		}
	}
}
