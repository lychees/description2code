import math
n = int(raw_input())
while n:
    a = int(raw_input())
    arev = str(a)[::-1]
    if str(a) != arev and a%10 != 0:
        sq = a*a
        sqrev = int(str(sq)[::-1])
        x =  int(math.sqrt(sqrev))
        if x * x == sqrev:
            print (x)
        else:
            print ('No')
    else:
        print ('No')
    n-=1
