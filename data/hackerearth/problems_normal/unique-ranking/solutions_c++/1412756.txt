#include <bits/stdc++.h>
using namespace std;


int main(){
	int n,k;
	scanf("%d%d",&n,&k);
	int a[1000][20];
	for(int i=0;i<n;i++){
		for(int j=0;j<k;j++) scanf("%d",&a[i][j]);
	}
	int q;
	scanf("%d",&q);
	while(q--){
		vector<int> myvector;
		int p;
		scanf("%d",&p);
		int b[p];int x;
		for(int i=0;i<p;i++) {scanf("%d",&x);b[i]=x-1;}
		int y=0;int j=0;
		for(int i=0;i<n;i++) myvector.push_back(i);
		while(j<p){
			x=y;
			y=myvector.size();
			int min=100,count=0;
			for(int i=x;i<y;i++) if(a[myvector[i]][b[j]]<min) min=a[myvector[i]][b[j]];
			for(int i=x;i<y;i++) if(a[myvector[i]][b[j]]==min) {myvector.push_back(myvector[i]);count++;}
			if(count==1) break;			
			j++;
		}
		printf("%d\n",myvector[y]+1);
	}
}
