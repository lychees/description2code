t = input()
q = []
o = []
for i in range(0, t):
    l = raw_input()
    x = int(l.split()[0])
    y = int(l.split()[1])
    n = int(l.split()[2])

    q = [y for w in range(0, x)]
    for j in range(x, n):
        s = sum([q[z] for z in range(j-x, j)])
        q.append(s)
    o.append(q.pop())

for i in o:
    print i
