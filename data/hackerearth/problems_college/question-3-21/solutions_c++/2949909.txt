#include<iostream>
#include<math.h>
using namespace std;

bool Prime(int a)
{
    if (a == 0 || a == 1)
        return false;
    for (int i = 2; i <= sqrt(a); i++)
        if (a%i == 0)
            return false;
    return true;

}

int main()
{
    int i, j, n, t, count = 0, x;
    int ans[100];
    cin >> n >> t;
    for (i = 0; i < t; i++)
    {
        count = 0;
        cin >> x;
        for (j = 1; j <= x; j++)
            if (Prime(j))
                count++;
        ans[i] = count;
    }
    for (i = 0; i < t; i++)
    {
        cout << ans[i] << endl;
    }

}
