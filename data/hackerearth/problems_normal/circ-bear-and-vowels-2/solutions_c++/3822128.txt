#include <iostream>
#include <string.h>
using namespace std;


string ishardoreasy(string s){
	int length = s.length();
	int countVowe = 0;
	int countConti = 0;
	int lastIncrease = -2;
	for(int i = 0; i < length; i++){
		if(s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u' || s[i] == 'y'){
			countVowe++;
			lastIncrease = -2;
			countConti = 0;
		} else {
			if(lastIncrease ==  i - 1){
				if(countConti == 2)
					return "hard";
				countConti++;
				lastIncrease = i;
			} else if(lastIncrease == -2){
				lastIncrease = i;
				countConti++;
			} 
		}
	}
	if(countVowe < (length - countVowe)){
		return "hard";
	} else {
		return "easy";
	}
}

int main()
{
	int no_test;
	cin >> no_test;

	string s;
	while(no_test--){
		cin >> s;
		cout << ishardoreasy(s) << endl;
	}
    return 0;
}
