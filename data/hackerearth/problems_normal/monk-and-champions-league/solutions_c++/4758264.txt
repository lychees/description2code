#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long m,n,sum=0;
    long long x;
    priority_queue<long long> pq;
    scanf("%lld%lld",&m,&n);
    for(int i=0;i<m;i++)
    {
        scanf("%lld",&x);
        pq.push(x);
    }
    while(n--)
    {
        sum+=pq.top();
        x=pq.top();
        pq.pop();
        x=x-1;
        pq.push(x);
    }
    printf("%lld\n",sum);

}
