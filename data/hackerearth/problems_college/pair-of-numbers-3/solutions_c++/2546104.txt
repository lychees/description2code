#include<bits/stdc++.h>
using namespace std;

int main()
{
	//ios_base::sync_with_stdio(0);
    int t,count;
    long int a,sum,pro,i,j;
    cin>>t;
    while(t--)
    {        
        cin>>a;
        count=0;
        for(i=1; i<=a; i++)
        {
            for(j=i+1; j<=a; j++)
             {
                 sum=i+j;
                 pro=i*j;
                 if(pro%sum==0)
                    count++;
             }
        }
        cout<<count<<endl;
    }
    return 0;
}
