set<string> S;

int main()
{
    int t;
    scanf("%d",&t);
    int ans = 1;
    while(t--)
    {

        scanf("%s",str);
        string s = str;
        int l = s.length();
        int flag = 0;
        for(int i=l,k=0;i>=1;i--,k++)
        {
            string st = s.substr(0,i);
            if(S.find(st) != S.end())
            {
                flag = 1;
                for(int j=i+1;j<=l;j++)
                {
                    ans++;
                    S.insert(s.substr(0,j));
                }
                break;
            }
        }
        if(!flag)
        {
            ans += l;
            for(int i=l;i>=1;i--)
                S.insert(s.substr(0,i));
        }

    }
    printf("%d",ans);
return 0;
}
