#include <iostream>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--){
        cin>>n;
        int index;
        int a[200];
        int i;
        for(i=0;i<200;i++){
            a[i]=0;
            
        }
    	a[0] = 1;
    	int loop = n;
    	int m=1;
    	for( i=1;i<=loop;i++){
    		
    		int index=0,carry=0,x;
    		while(m--){
    			x = a[index] * n + carry;
    			a[index] = x%10;
    			carry = x/10;
    			index++;
    			
    		}
    		while(carry>0){
    			a[index] = carry%10;
    			carry /= 10;
    			index++;
    		}
    		
    			n--;
    			m = index+1;
    		
    	}
    	int p=199;
    	while(a[p]==0){
    		p--;
    	}
    	for(int prin = p;prin>=0;prin --){
    		cout<<a[prin];
    	}
    	cout<<endl;
    }
    return 0;
}
