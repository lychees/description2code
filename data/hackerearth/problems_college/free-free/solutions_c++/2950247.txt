#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <iostream>
using namespace std;
int main() {
    int t;
    string a;
    int f;
    cin>>t;
    while(t--)
    {cin>>a;
     f=0;
     int b[256]={0};
     int l=a.length();
     for(int i=0;i<l;i++)
       b[a[i]]++;
     for(int i=0;i<256;i++)
       f+=(b[i]%2)+(b[i]/2);
     cout<<f<<endl;
    }
    return 0;
}
