using namespace std;

int n;
int a[10000];
int ans;
vector<int> r[10000];

void add(int x,int y)
{
	r[x].pb(y);
}
int dfs(int v)
{
	int tk=0;
	for (int i=0;i<sz(r[v]);i++)
	{
		tk=max(tk,dfs(r[v][i]));
	}
	return tk+1;
}
int main()
{
	
	 
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		if (a[i]!=-1) 
			add(a[i],i);
	}
	ans=0;
	for (int i=1;i<=n;i++)
	if (a[i]==-1){
		ans=max(ans,dfs(i));
	}
	cout<<ans<<endl;
	return 0;
}
