#include <iostream>
#include <string>
using namespace std;

int main()
{
    int N,i,j,count;
    string s;
    cin>>N;
    
    while(N--){
    	cin>>s;
    	count = 0;
    	j=0;
    	for(i=1;s[i]!='\0';i++){
    		while(s[i] == s[j]){
    			i++;
    			count++;
    		}
    		j=i;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
