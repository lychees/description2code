#include <iostream>
#include <vector>
using namespace std;
 
int main()
{
    int n,num;
    int count;
    cin>>n;
    for(int i=0;i<n;i++)
    {	
    	//cout<<"this";
    	cin>>num;
    	count = 0;
    	for(int j = 1;j<=num;j++)
    	{
    		for(int k=j+1;k<=num;k++)
    		{
    			if( (j^k) <= num)
    			{
    			count++;
    			}
    		}
    	}
    	cout<<count<<endl;
    }
    
    return 0;
}
