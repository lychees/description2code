#include <bits/stdc++.h>
using namespace std;
long long int gcd(long long int a , long long int b)
{
	if (a == 0)
        return b;
    return gcd(b%a, a);
}

int main()
{
    int test,n;
    cin >> test;
    while(test--)
    {
    	cin >> n;
    	int counter = 1;
    	for(int i=2;i<=n-1;i++)
    	{
    		if(gcd(i,n)==1)
    		 	counter++;
    	}
    	
    	if(counter == 1)
    		cout <<"Incorrect"<<endl;
    	else if(counter == 2 || counter ==3)
    		cout << "Correct"<<endl;
    	else
    	{
    		int val = sqrt(counter),flag=0;
    		for(int i=2;i<=val;i++)
    		{
    			if(counter%i==0)
    				flag=1;
    		}
    		if(flag)
    			cout <<"Incorrect"<<endl;
    		else
    			cout << "Correct"<<endl;
    	}
    }
    return 0;
}
