

    #include <bits/stdc++.h>
    using namespace std;
     
    int main()
    {
        int t;
        cin>>t;
        while(t--)
        {
        	int i,j,count=0;
        	long long int n;
        	cin>>n;
        	long int arr[n];
        	for(i=0;i<n;i++)
        	cin>>arr[i];
        	sort(arr,arr+n);
        	for(i=0;i<n;i++)
        	{
        		for(j=i;j<n;j++)
        		{
        		if(arr[i]==arr[j])
        		{
        		count++;
        		
        		}
        
        		}
        	}
        	cout<<count<<endl;
        	
        }
        return 0;
    }
