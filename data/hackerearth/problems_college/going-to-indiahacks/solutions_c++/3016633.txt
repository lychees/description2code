typedef long long LL;

#define LIM 1100
#define MOD 1000000007

int dp[LIM][LIM];

int main()
{
	//freopen("in.txt","r",stdin);
	//freopen("out.txt","w",stdout);
	int i,j,k;


	for(i=0;i<LIM;i++)
	{
		dp[i][0]=1;
		dp[0][i]=1;
	}

	for(i=1;i<LIM;i++)
	for(j=1;j<LIM;j++)
		dp[i][j]=( (dp[i-1][j]+dp[i][j-1])%MOD + dp[i-1][j-1])%MOD;

	int T,x1,x2,y1,y2;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%d %d %d %d",&x1,&y1,&x2,&y2);
		printf("%d\n",dp[x2-x1][y2-y1]);
	}
	return 0;
}
