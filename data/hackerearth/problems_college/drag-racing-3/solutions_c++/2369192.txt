#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	n=pow(2,n)-1;
	
	map<string,int> m;
	vector<string> s;
	
	while(n--)
	{
		string a,b;
		cin>>a>>b;
		
		m.insert(make_pair(a,0));
		m.insert(make_pair(b,0));
		s.push_back(b);
		
	}
	
	vector<string>::iterator i;
	for(i=s.begin();i!=s.end();i++)
	m[*i]=-1;
	
	map<string,int>::iterator it;
	for(it=m.begin();it!=m.end();it++)
	{
		if((*it).second ==0 )
			cout<<(*it).first;
	}
}
