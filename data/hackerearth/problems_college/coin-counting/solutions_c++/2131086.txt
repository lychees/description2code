#include <iostream>
using namespace std;
int count(int a[],int m,int n){
    if(n==0) return 1;
    if(n<0) return 0;
    if(m<=0&&n>0) return 0;
    else return count(a,m-1,n)+count(a,m,n-a[m-1]);
    
}

int main() {
	// your code goes here
	int a[3] ={1,3,5};
	int t;
	cin>>t;
	while(t--){
	    int number;
	    cin>>number;
	int res = count(a,3,number);
	cout<<res<<endl;
	}
	return 0;
}
