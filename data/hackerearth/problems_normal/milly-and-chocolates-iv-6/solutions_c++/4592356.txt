#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		long long i,j,n,k,m,ti,po,x;
		cin>>n>>k>>m;
		long long p[n];
		for(i=0;i<n;i++)
			cin>>p[i];
		for(i=0;i<n;i++){
			long long a;
			x = 0;
			for(j=0;j<p[i];j++){
				x += k;
				cin>>a;
				x += a;
				if(j != p[i]-1)
					x += m;
			}
			if(i==0){
				ti = x;
				po = 1;
			}
			if(ti > x){
				ti = x;
				po = i+1;
			}
		}
		cout<<po<<" "<<ti<<endl;
	}
    return 0;
}
