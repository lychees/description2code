#include<iostream>
using namespace std;
int min(int a,int b)
{
	return (a<b?a:b);
}
int main()
{
	int n;
	cin>>n;
	if(n>10 || n<1)	
		cout<<"Invalid Test\n";
	else
	{
	for(int i=0;i<n;i++)
	{int lo=0,up=0;
		string s;
		cin>>s;
		for(int i=0;s[i];i++)
		{
			if(s[i]>=97 && s[i]<=122)
			lo++;
			else if(s[i]>=65 && s[i]<=91)
			up++;
		}
		if(lo==0 && up==0  || s.length()>100 || s.length()<1 )
		cout<<"Invalid Input"<<"\n";
		else
		cout<<min(lo,up)<<"\n";
	}
	}
}
