#include <iostream>
#include <algorithm>
#define ll long long int

using namespace std;

int main()
{
    int test ;
    
    cin >> test ;
    
    ll a , b ;
    
    while(test--)
    {
    	ll res = 0 ;
    	
    	cin >> a >> b ;
    	
      while(a > 0 && b > 0)
      {   
      	 if(a > b)
      	 {
      	 	res = res + a / b ;
      	 	a = a % b ;
      	 }
      	 
      	 else
      	 {
      	 	res = res + b / a ;
      	 	b = b % a ;
      	 }
      }
    	
    if(res > 0) res-- ;	
      
      cout << res << endl ;
    }
    
    return 0;
}
