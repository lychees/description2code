#include <iostream>
#include<string>
#include<algorithm>
using namespace std;


int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	bool flag1=true;
    	bool flag2=true;
    	int sl=s.length();
    	for(int i=0;i<sl;i++)
    	{
  if(s[i]=='2' ||s[i]=='3' ||s[i]=='4' ||s[i]=='5' ||s[i]=='6' ||s[i]=='7' ||s[i]=='9') flag1=false;
    	}
    	if(flag1)
    	{
    		for(int i=0,j=sl-1;i<j;i++,j--)
    		{
    			if(s[i]!=s[j]) flag2=false;
    		}
    	}
    	if(flag1 && flag2)
    	{
    		cout<<"YES"<<endl;
    	}
    	else cout<<"NO"<<endl;
    }
    return 0;
}
