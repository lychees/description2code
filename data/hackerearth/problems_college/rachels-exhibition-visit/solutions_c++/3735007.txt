#include <iostream>
#include <string>
#include <cmath>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int n,k,l,r,i,j,z,c=0;
   	cin>>n>>k>>l>>r;
   	string s1,s2;
   	cin>>s1>>s2;
   	for(i=0;i<k;i++)
   	{
   		for(j=0;j<n;j++)
   		{if(s1[j]==s2[i]){s1[j]=10;}
   			
   		}
   	}
   	for(i=0;i<n;i++)
   	     {
   	     	z=0;
   	     	for(j=i;j<n;j++)
   	     	{
   	     		if(s1[j]==10){z++;if(z>r){break;}}
   	     	if(z>=l && z<=r){c++;}	
   	     	}
   	     }
   	cout<<c<<endl;
   }
    return 0;
}
