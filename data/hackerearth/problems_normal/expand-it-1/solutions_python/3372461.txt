countArr = {}
str = raw_input()
length = len(str)

#count distinct characters
st = []
currChar = ''
num = 0
flag = False
for i in range(length):

    if str[i].isalpha():

        #update previous count
        if(flag):
            if currChar not in countArr:
                countArr[currChar] = 0
                st.append(currChar)
                
            countArr[currChar] = countArr[currChar] + num
                
                
        #start new count
        currChar = str[i]
        num = 0
        flag = True
    else:
        num = num*10+int(str[i])


#### update last letter count
if currChar not in countArr:
    countArr[currChar] = 0
    st.append(currChar)

countArr[currChar] = countArr[currChar] + num
st.sort()


T = int(raw_input())
for i in range(T):
    K = int(raw_input())
    ans = '-1'
    for letter in st:
        if K <= countArr[letter]:
            ans = letter
            break

        K = K - countArr[letter]

    print ans
