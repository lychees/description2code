#include<iostream>
#include<cmath>
#define ll long long
using namespace std;
int main()
{
    ll t,a,b,c,n;
    cin>>t;
    while(t--)
    {cin>>a>>b>>c;
   ll maxx=pow(10,c);
    a=a%maxx;
    ll bb=b,ans=1,fin=1;
    while(b>0)
    {
        ll cc=b%2;
        if (b==bb)
            ans=a;
        else
            ans=(ans*ans)%maxx;
        if(cc==1)
            fin=(fin*ans)%maxx;
        b=b/2;
    }
    ans=0;
    while(fin>0)
    {
        ans=ans+fin%10;
        fin=fin/10;
    }
    cout<<ans<<endl;
    }
    return 0;
}
