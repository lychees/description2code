#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int A[100][100],n,sum1=0,sum2=0;
    cin>>n;
    for(int i=0;i<n;i++)
    for(int j=0;j<n;j++)
    cin>>A[i][j];
    for(int i=0;i<n;i++)
    {
    	sum1=sum1+A[i][i];
    	int j=n-1-i;
    	sum2=sum2+A[i][j];
    	
    }
    cout<<abs(sum1-sum2);
    return 0;
}
