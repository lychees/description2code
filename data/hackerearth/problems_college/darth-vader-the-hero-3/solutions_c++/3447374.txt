#include<iostream>

using namespace std;

#define max(a,b) a>b?a:b
#define min(a,b) a<b?a:b

void scan(int &x); //IMPORTANT: for positive only
void travel(int,int);

 bool a[3000][3000];
 bool visited[3000][3000];
 long tlen=0;

int main(void)
{
  int i,t,n,m,x,j;

  cin >> t;

  while(t--)
  {
     scan(n);
     scan(m);
     memset(a,false,sizeof(a));
     memset(visited,false,sizeof(visited));

     for(i=0;i<n;i++)
        for(j=0;j<m;j++)
           {
               scan(x);
               if(x)
                 a[i][j]=true;
               else
                 a[i][j]=false;
           }

     long ans=0,temp=0,total=0;
     for(i=0;i<n;i++)
     {
         for(j=0;j<m;j++)
         {
             tlen=0;
             if(!visited[i][j] && a[i][j])
             {
               //  cout << "visiting: " << i << " " << j << endl;
                 travel(i,j);
                 temp = tlen;
                 total++;
                 if(temp>=ans)
                    ans = temp;
             }
            // cout << endl;
         }
     }

     cout << total << " " << ans << endl;
  }

  return 0;
}


void travel(int i,int j)
{
    tlen++;
    visited[i][j] = true;
  //  cout << "index: " << (i*j)+j << endl;
  //  cout << i <<" " << j << endl;


    if(a[i+1][j] && !visited[i+1][j])
        travel(i+1,j);
    if(a[i+1][j-1] && !visited[i+1][j-1])
        travel(i+1,j-1);
    if(a[i+1][j+1] && !visited[i+1][j+1])
         travel(i+1,j+1);

    if(a[i-1][j] && !visited[i-1][j])
        travel(i-1,j);
    if(a[i-1][j+1] && !visited[i-1][j+1])
        travel(i-1,j+1);
    if(a[i-1][j-1] && !visited[i-1][j-1])
         travel(i-1,j-1);

    if(a[i][j-1] && !visited[i][j-1])
       travel(i,j-1);
    if(a[i][j+1] && !visited[i][j+1])
        travel(i,j+1);
}
