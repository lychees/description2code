# doesn't need heap(nlogn), linear sufficient O(n)
n=int(raw_input())
a=map(int,raw_input().split())
x=0
y=0
z=0
for i in range(n):
    if a[i]>=x:
        x,y,z=a[i],x,y
    elif a[i]>=y:
        y,z=a[i],y
    elif a[i]>=z:
        z=a[i]
    if i<2:
        print -1
    else:
        print x*y*z
