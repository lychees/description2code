#include <cstdio>  
int main()  
{
long long a[303];  
long long int n,i,j,t;  
  scanf("%lld",&t);
    for (int i=1;i<=300;i++)  a[i]=i*i*i;    
    while (t--)  
    {
    	scanf("%lld",&n);
	  long long d[100004]={0};      
        d[0]=1;  
        for (i=1;i<=300;i++)  
        {  
            for (j=0;j<=n-a[i];j++)  
            {  
                if (d[j]) 
				d[j+a[i]]+=d[j];  
            }  
        }  
        printf("%lld\n",d[n]);
        
    }  
    return 0;  
}
