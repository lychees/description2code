    #include <bits/stdc++.h>
    using namespace std;
     
    int main()
	{
        string str;
        cin>>str;
        int even=0;
        for(int i=0; i<str.length(); i++)
		{
        	if((str[i]-'0')%2==0)
			 even++;
        }
        for(int i=0; i<str.length(); i++)
		{
        	cout<<even<<" ";
        	if((str[i]-'0')%2==0)
			even--;
        }
        return 0;
    }
