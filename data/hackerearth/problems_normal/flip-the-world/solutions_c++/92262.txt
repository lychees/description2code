#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
#include <string>
#include <cstdlib>
#include <cstring>
using namespace std;
int arr[100][100];
void flip(int a,int b){
	for(int i=0;i<=a;i++){
		for(int j=0;j<=b;j++){
			if(arr[i][j] == 0)
				arr[i][j] = 1;
			else 
				arr[i][j] = 0;
		}
	}
}
void print(int a,int b){
	    int i,j;
	    for(i=0;i<=a;i++){
			for(j=0;j<=b;j++){
				cout<<arr[i][j]<<" ";
			}
			//cout<<endl;
		}
}
int main(){
	int n,m,t;
	char str[100];
	cin>>t;
	while(t--){
		cin>>n>>m;
		n--;m--;
		int count=0;
		for(int i=0;i<=n;i++){
			scanf("%s",str);
			for(int j=0;j<=m;j++){
				arr[i][j] = str[j]-48;
			}
		}
		//print(n,m);cout<<endl;
		for(int i=n;i>=0;i--){
			for(int j=m;j>=0;j--){
				if(arr[i][j] == 0){
					flip(i,j);
					count++;
				}
			}
		}
		//print(n,m);cout<<endl;
		cout<<count<<endl;
    }
return 0;
}
