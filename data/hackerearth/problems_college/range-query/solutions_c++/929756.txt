#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	scanf("%d",&n);
	int a[10010] = {0};
	int i = 0;
	while(i < n)
		scanf("%d",&a[i++]);
	i = 0;
	int q;
	scanf("%d",&q);
	int dp[101][10010] = {0};
	int size[101] = {0};
	int j = 0;
	while(i < n)
	{
		dp[a[i]][size[a[i]]] = i;
		size[a[i]] += 1;
	//	printf("%d %d %d bug1\n", a[i],size[a[i]],dp[a[i]][size[a[i]]]);
		i++;
	}
/*	i = 1;
	while(i < 5)
	{	printf("%d ", size[a[i]]);
		i++;}
	printf("\n");
*/	i = 0;
	while(i < q)
	{
		int x,y,val;
		scanf("%d %d %d",&x,&y,&val);
		int count = 0;
		j = 0;
		while(j < size[val])
		{
			if(dp[val][j] >= x && dp[val][j] <= y)
			{
//				printf("%d bug\n",dp[a[val]][j]);
				count++;
			}
			j++;
		}
		printf("%d\n",count);
		i++;
	}
	return 0;
}
