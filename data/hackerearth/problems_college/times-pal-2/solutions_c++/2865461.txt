#include <bits/stdc++.h>
using namespace std;
bool palin(string s1)
{
	int i,l2,l1=s1.length();
	l2=l1;
	l2/=2;
	for(i=0;i<l2;i++) {if(s1[i]!=s1[l1-i-1]) return false;}
	return true;
}
int main()
{
    int t,i;
    cin>>t;
    while(t--)
    {
    	string str1,str2,str3;
    	cin>>str1;
    	cin>>str2;
    	//cout<<str1<<" "<<str2<<endl;
    	int l1=str1.length(),count=0;
    	str3=str1;
    	for(i=0;i<=l1;i++)
    	{
    		str1.insert(i,str2);
    		//cout<<str1<<endl;
    		if(palin(str1)) count++;

    	}
    	printf("%d\n",count);
    }

}
