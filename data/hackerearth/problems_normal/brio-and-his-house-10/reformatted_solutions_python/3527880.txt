import math
t = int(raw_input())
for i in range(t):
    a = [float(f) for f in raw_input().split()]
    abc = a[0]*a[1]*a[2]
    mul = (a[0]+a[1]+a[2])*(-a[0]+a[1]+a[2])*(a[0]-a[1]+a[2])*(a[0]+a[1]-a[2])
    mulsq = math.sqrt(mul)
    r = abc/mulsq
    x = round(3.1415*r*r, 4)
    x = str(x)
    n = x.split('.')
    n1 = n[1]
    lent = 4 - len(n1)
    if lent > 0:
        n2 = n1 + '0'*lent
        n3 = n[0]+'.'+n2
    else:
        n3 = n[0]+'.'+n1
    print(n3)
