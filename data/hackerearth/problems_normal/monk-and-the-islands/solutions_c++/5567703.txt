#include <bits/stdc++.h>
using namespace std;

int main()
{
    int test;
    scanf("%d",&test);
    while(test--){
    	int n,m;
    	scanf("%d%d",&n,&m);
    	vector<bool> vis(n+1,false);
    	vector<int> dist(n+1,0);
    	vector<int> v[n+1];
    
    	for(int i=0;i<m;i++){
    		int a,b;
    		scanf("%d%d",&a,&b);
    		v[a].push_back(b);
    		v[b].push_back(a);
    	}
    	vis[1]=true;
    
    	queue<int> q;
    	q.push(1);
    	while(!q.empty()){
    		int node=q.front();
    		q.pop();
    		for(int i=0;i<v[node].size();i++){
    			if(!vis[v[node][i]]){
    				vis[v[node][i]]=true;
    				dist[v[node][i]]=dist[node]+1;
    				q.push(v[node][i]);
    			}
    		}
    	}
    	printf("%d\n",dist[n]);
    }
    return 0;
}
