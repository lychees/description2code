ll p2(ll x) {

}

int main() {
    ll n0;
    cin >> n0;
    ll a = n0, b = n0 - 1;
    if (a % 2 == 0) swap(a, b);
    b /= 2;
    for (ll x = 1; x * x <= b; ++x) if (b % x == 0) {
    	ll y = b / x;
    	if (y > 1e9) continue;
    	if (a == p2(y) - p2(x)) {
    		ll k1 = p2(y) + p2(x);
    		cout << (k1 + 1) / 2 << endl;

    	}
    }

}
