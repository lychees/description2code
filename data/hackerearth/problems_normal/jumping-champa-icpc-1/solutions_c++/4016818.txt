#include <iostream>
#include<stdio.h>
#include<algorithm>
#include<cmath>
using namespace std;

int main()
{
    long long int n,k,i,a[1000],q,t,sum=0;
    scanf("%lld",&t);
    while(t--){
    	sum=0;
    	scanf("%lld %lld",&n,&q);
    	
    	for(i=0;i<n;i++){
    		scanf("%lld",&a[i]);
    	}
    	std::sort(a,a+n);
    	for(i=0;i<n-1;i++){
    		sum=sum+abs(a[i]-a[i+1]);
    	}
    	printf("%lld\n",(sum*q));
    }
    return 0;
}
