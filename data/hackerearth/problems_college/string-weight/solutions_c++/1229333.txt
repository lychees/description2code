#include<bits/stdc++.h>
using namespace std;

bool hashTable[26];

int main() {
	int tt;
	cin>>tt;
	while(tt--) {
		string str;
		memset(hashTable,0,sizeof(hashTable));
		cin>>str;
		int len=str.size();
		int n;
		cin>>n;
		char ch;
		for(int i=0;i<n;i++) {
			cin>>ch;
			hashTable[ch-'a']=true;
		}
		int cost=0;
		for(int i=0;i<len;i++) {
			if(hashTable[str[i]-'a']==false) {
				cost+=(str[i]-'a')+1;
			}
		}
		cout<<cost<<endl;
	}
	return 0;
}
