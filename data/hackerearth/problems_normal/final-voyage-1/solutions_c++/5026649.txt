#include <bits/stdc++.h>
using namespace std;
#define ll long long int
ll wt[1005],val[1005];
int knap(int n,int w)
{
	int i,j;
    ll dp[n+1][w+1]; 
	for(i=0;i<=n;i++)
	dp[i][0]=0;
	
	for(i=0;i<=w;i++)
	dp[0][i]=0;
	
	for(i=1;i<=n;i++)
	{
		for(j=0;j<=w;j++)
		{
			dp[i][j]=dp[i-1][j];
			
			if(j>=wt[i])
			{
				dp[i][j]=max(dp[i][j],dp[i-1][j-wt[i]]+val[i]);
			}
		}
	}
	return dp[n][w];
}
int main()
{
	int t;
	ll n,w,i,j;
    cin>>t;
    while(t--)
    {
    	cin>>n>>w;
    	for(i=1;i<=n;i++)
    	{
    		cin>>wt[i];
    	}
    	for(i=1;i<=n;i++)
    	{
    		cin>>val[i];
    	}
    	cout<<knap(n,w)<<endl;
    }
    return 0;
}
