#include <bits/stdc++.h>
#include<math.h>

using namespace std;
#define ll long long int
#define S(n) scanf("%d",&n)


int main()
{
    ll T,M;
    ll dp[200][4];
    dp[0][0]=0;
    dp[0][1]=1;
    for(int i=1;i<=100;i++)
    {
    	dp[i][0]=dp[i-1][0]+dp[i-1][1];
    	dp[i][1]=dp[i-1][0];
    }
    S(T);
    for(int i=0;i<T;i++)
    {
        S(M);
        cout << dp[M][1] << " " << dp[M][0] <<"\n";
        
    }
    return 0;
}
