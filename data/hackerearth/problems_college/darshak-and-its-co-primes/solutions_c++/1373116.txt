#include <bits/stdc++.h>
using namespace std;
int gcd(long long p,long long q)
{
	long long i=2;
	while(i<=min(p,q))
	{
		if((p%i==0)&&(q%i==0))
		return 0;
	}
	return 1;
}
int main()
{
    //cout << "Hello World!" << endl;
    long long t;
    cin>>t;
    while(t--)
    {
    	long long p,q,r;
    	cin>>p>>q;
    	while(p%q!=0)
        {
            r=p%q;
            p=q;
            q=r;
        }
        if(q==1)
    	cout<<"Is a Co-Prime"<<endl;
    	else
    	cout<<"Not a Co-Prime"<<endl;
    }
    return 0;
}
