#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
	vector<ll> v(7);
    ll i,t,n,m;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld%lld",&n,&m);
    	for(i=0;i<n;i++)
    		scanf("%lld",&(v[i]));
    	sort(v.begin(),v.begin()+n);
    	for(i=n-1;i>=0;i--)
    	{
    		m-=v[i];
    		if(m<=0)
    			break;
    	}
    	if(m<=0)
    	{
    		printf("%lld\n",n-i);
    	}
    	else
    		printf("-1\n");
    }
    return 0;
}
