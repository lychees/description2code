int sum;

bool dp[100006][101];

int main()
{
	int n;
    cin>>n;
    int a[n],i,j,k;
    sum=0;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    	sum+=a[i];
    }	
    
    //subset_sum(a);
    
    memset(dp,0,sizeof(dp));
	
	for(i=0;i<=n;i++)
		dp[0][i]=1;
	
	for(i=1;i<=sum;i++)
		dp[i][0]=0;
		
	for(i=1;i<=sum;i++)
	{
		for(j=1;j<=n;j++)
		{
			int v=a[j-1];
			dp[i][j]=dp[i][j-1];
			if(i>=v)
				dp[i][j]=dp[i][j-1] || dp[i-v][j-1];
		}
	}
    
    /*for(i=0;i<=n;i++)
    {
    	for(j=0;j<=sum;j++)
    		cout<<dp[i][j];
    	cout<<endl;
    }*/
    
    int q,t;
    cin>>q;
    
    while(q--)
    {
    	cin>>t;
    	if(dp[t][n]==1)
    		cout<<"Yes"<<endl;
    	else
    		cout<<"No"<<endl;
    }
    
    return 0;
}
