#include <iostream>
using namespace std;

int main()
{
    int t,n,a,b;
    cin>>t;
    while(t--)
    {
    	long int r,c,i;
    	cin>>r>>c;i=1;
    	a=0;b=0;
    	if(r<=c)
    	{
    	if(r%2==1)
    	{
    	n=(r+1)/2;
    	b=n*n;
    	a=n*(n-1);
    	
    	}
    	else
    	{
    		n=r/2;
    		b=n*n;
    		a=n*(n+1);
    		
    	}
    	}
    	else
    	{
    	while(i<=r)
    	{
    		if(i<c)
    		if(i%2==0)
    		a=a+i++;
    		else
    		b=b+i++;
    		else
    		{
    			if(i%2==0)
    			a=a+c;
    			else
    			b=b+c;
    			i++;
    		}
    	}
    	}
    	cout<<a<<" "<<b<<endl;
    	
    }
    return 0;
}
