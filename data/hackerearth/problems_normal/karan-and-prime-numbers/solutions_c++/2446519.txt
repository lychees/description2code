#include <iostream>
using namespace std;

bool isprime(int n) {
	if (n < 2) return false;
	
	for (int i = 2; i*i <= n; i++) {
		if (n%i == 0) return false;
	}
	
	return true;
}

int main()
{
    int t;
    cin >> t;
    
    long long sum = 0;
    long long sp[1000001];
    
    for (int i = 0; i <= 1000000; i++) {
    	if (isprime(i)) sum += i;
    	sp[i] = sum;
    }
    
    while (t-- > 0) {
    	int a, b;
    	cin >> a >> b;
    	
    	cout << (sp[b] - sp[a-1]) << endl;
    	
    }
    return 0;
}
