using namespace std;
#include<iostream>
#include<math.h>
int main()
{
	int b=0,c=0,con=0,j,t,k;
	char a[10000];
	cin>>t;
	for(j=0;j<t;j++)
	{
		cin>>a;
		con=0;
		for(int i=0;a[i]!='\0';i++)
		{
			c=0;
			int b=a[i];
			if(b%2==0)
			{
				cout<<a[i];
				con=1;
			}
			else
			{
				for(k=3;k<b;k=k+2)
				{
					if(b%k==0)
					{
						c=1;
						break;
					}
				}
				if(c==1)
				{
					cout<<a[i];
					con=1;
				}
			}
		}
		if(con==0)
			cout<<"NULL";
		cout<<endl;
	}
}
