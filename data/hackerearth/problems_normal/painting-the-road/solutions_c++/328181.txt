#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
int dp[258];
int arr[257][514][2];
	while(t--){
	int l,k;
	cin>>l>>k;
	int count[257]={0};
	for(int i=0;i<k;i++){
		int a,b,c;
		cin>>a>>b>>c;
		int len=count[a]++;
		
		arr[a][len][0]=b;
		arr[a][len][1]=c;
	}
	dp[l]=0;
	for(int i=l-1;i>=0;i--){
		
		if(count[i]==0){
			dp[i]=-1;
			continue;
		}
		int len=count[i];
		dp[i]=INT_MAX;
		for(int j=0;j<len;j++){
			int end=arr[i][j][0],currcost=arr[i][j][1];int mini=INT_MAX;
			for(int k=i+1;k<=end;k++){
				if(dp[k]==-1)
					continue;
				mini=min(mini,dp[k]);
			}
			if(mini!=INT_MAX)
			dp[i]=min(dp[i],currcost+mini);
		}
		
		if(dp[i]==INT_MAX)
			dp[i]=-1;
	}
	cout<<dp[0]<<endl;
	}
	return 0;
}
