int n, a[100100], res;

int cal(int x){
    int i = 2;
    int res = 0;
    while (i <= sqrt(x)){
        while (x % i == 0){
            x = x / i;
            res++;
        }
        i++;
    }
    if (x > 1) res++;
    return res;
}

int main()
{
    ios::sync_with_stdio(false);
    //freopen("input.txt", "r", stdin);
    cin >> n;
    FOR(i,1,n) cin >> a[i];
    res = 0;
    FOR(i,1,n) res = res ^ cal(a[i]);
    if (res == 0) cout << "BOB";
    else cout << "ALICE";
    return 0;
}
