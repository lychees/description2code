#include <iostream>
#include <cstdio>
using namespace std;
#define LL long long
#define M 1000000007
LL inver[1000006];
LL cat[1000006];
LL pow(LL n,LL k)
{
	if(k==0)return 1;
	if(k==1)return (n%M);
	LL v=pow(n,(k/2));
	if(k%2)
	return (((v*v)%M)*n)%M ;
	
	return (v*v)%M;
}
LL inv(LL n)
{
	return pow(n,M-2) ;
}
void pre()
{
	inver[1]=1;
	for(int i=2;i<=1000002;++i)
	{
		inver[i]=inv(i);
	}
}

void pre2()
{
	cat[1]=1;
	cat[2]=2;
	for(int i=3;i<=1000001;++i)
	{
		cat[i] = (((cat[i-1]*2)%M)*(2*i-1))%M ;
		cat[i]= (inver[i+1]*cat[i])%M ;
	}
}
int main() {
	pre();pre2();
	int t;
	cin>>t;
	LL n;
	while(t--)
	{
		cin>>n;
		cout<<cat[n]<<endl;
	}
		
	return 0;
}
