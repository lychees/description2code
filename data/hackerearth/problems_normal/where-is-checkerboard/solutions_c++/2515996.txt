#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main() {
	// your code goes here
	
	int n,m,x,flag=1;
	cin>>n>>m;
	
/*	if( n>1450 || m >1450)
	{
		cout<<0;
		return 0;
	}*/
	
	vector<int> f(m,0);
	vector<vector<int>> a(n,f);
//	int a[n][m];
	string ch;
	for( int i=0 ; i<n ; i++)
	{
		cin>>ch;
		for ( int j=0 ; j< ch.length() ; j++)
		{
			if( ch[j]=='_')
			a[i][j]=1;
			else
			a[i][j]=0;
		}
	}

	for( int i=0 ; i<n ; i++)
	{
		for ( int j=0 ; j< m-31 ; j++)
		{
			if( j+31 <= m)
			{
			flag=1;
			if( a[i][j]==1)
				x=0;
			else
				x=1;
				for(int k=j+1; k<m && k <n && k<j+32 ;k++)
				{
					if( a[i][k]!=x)
					{
						flag=0;
						break;
					}
					if(x==1)
					x=0;
					else
					x=1;
				}
			if(flag==1)
			{
				if( a[i][j]==1)
				{
					a[i][j]=2;
				}
				else
				{
					a[i][j]=3;
				}
			}
			}
			
		}
	}
	int cnt=0;
	for( int i=0 ; i < n-31 ; i++)
	{
		for( int j=0 ; j< m ; j ++)
		{
			if( a[i][j] == 2 && i+31 <= n)
			{
				flag=1;
				x=3;
				int p=1;
				for(int k=i+1; k<m && k<n && k<i+32 ;k++)
				{
					p++;
					if( a[k][j]!=x)
					{
						flag=0;
						break;
					}
					if(x==3)
					x=2;
					else
					x=3;
				}
				if(flag==1)
				{
					cnt++;
				}
			}
		}
	}
	cout<<cnt;
	return 0;
}
