#include <bits/stdc++.h>
using namespace std;
#define ll long long

map<int , vector<ll> > mp;

int main()
{
    int T;
    cin>>T;
    while(T--){
    	
    	mp.clear();
    	ll n,x,temp;
    	cin>>n;
    	
    	for(int i=0;i<n;i++){
    		cin>>temp;
    		x=temp;
    		int cnt=0;
    		while(x){
    			x= x & (x-1);
    			cnt++;
    		}
    		//cout<<temp<<" ";
    		mp[cnt].push_back(temp);
    	}
    	//cout<<endl;
    	
    	map<int , vector<ll> > :: iterator it;
    	for(it=mp.begin();it!=mp.end();it++){
    		for(int i=0;i<mp[it->first].size();i++){
    			cout<<mp[it->first][i]<<" ";
    		}
    	}
    	
    	cout<<endl;
    	
    }
}
