def solve(i, j, a, count):
    if i > j:
        return count
    if a[i] != a[j]:
        return min(solve(i+1, j, a, count+1), solve(i, j-1, a, count+1))
    else:
        return solve(i+1, j-1, a, count)


t = int(raw_input())
for i in xrange(t):
    n = int(raw_input())
    a = raw_input()
    print(solve(0, n-1, a, 0))
