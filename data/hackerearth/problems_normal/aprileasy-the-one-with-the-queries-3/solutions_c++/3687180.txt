#include <bits/stdc++.h>

using namespace std;
#define MAX 100005
int arr[MAX][26];

int main() {
   string str1, str2;
   int len1, i, j, len2, query;
   
   cin >> str1;
   len1 = str1.size();
   
   for(i = 0 ; i < 26 ; i++)
      arr[len1][i] = len1;

   for(i = len1-1 ; i >= 0 ; i--) {
      for(j = 0 ; j < 26 ; j++)
         arr[i][j] = arr[i+1][j];
      arr[i][str1[i] - 'a'] = i;
    }
  
    cin >> query;
    while(query--) {
       cin >> str2;
       int len2 = str2.size(), pos = 0;
       bool f = true;
       
       for(i = 0 ; i < len2 && f ; i++) {
          if(arr[pos][str2[i] - 'a'] >= len1)
             f = false;
          else
             pos = arr[pos][str2[i] - 'a'] + 1; 
       }
       
       if(f)
          cout << "Yes";
       else
          cout << "No";

       cout << "\n";  
    }
    return 0;
}
