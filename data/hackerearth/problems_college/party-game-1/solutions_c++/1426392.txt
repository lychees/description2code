#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int x,y,n,c1=0,c2=0;
    	cin>>x>>y>>n;
    	int a[n];
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	sort(a,a+n);
    	for(int i=0;i<n;i++)
    	{
    		if(x-a[i]>=0)
    		{	
    			c1++;
    			x=x-a[i];
    		}
    		if(y-a[i]>=0)
    		{	
    			c2++;
    			y=y-a[i];
    		}
    	}
    	
    	if(c1>c2)
    	cout<<"Chef Won"<<endl;
    	else
    	if(c1<c2)
    	cout<<"Jhool Won"<<endl;
    	else
    	cout<<"Tie"<<endl;
    	
    }
    return 0;
}
