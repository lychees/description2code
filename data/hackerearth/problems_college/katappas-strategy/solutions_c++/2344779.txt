#include<cstdlib>
#include<cstdio>
#define LIM 100002

long long gcd(long long a,long long b)
{
    if(b==0)
        return a;
    else
        gcd(b,a%b);
}

void solve(int n)
{
    long long a[LIM],b[LIM],asum,bsum,range[LIM],pos;
    char str[20],*ch;

    for(int i=0;i<n;i++)
    {
        scanf("%s",&str);

        for(ch=str; (*ch)!='/'; ch++);
        *ch = '\0';
        a[i] = atoi(str);
        b[i] = atoi(ch+1);
        range[i] = i;
    }

    for(int i=n-2; i>=0; i--)
    {
        pos = i+1;
        asum = a[i]+a[pos];
        bsum = b[i]+b[pos];

        while(pos<n && (((double)asum)/((double)bsum)) > (((double)a[i])/((double)b[i])))
        {
            a[i] = asum;
            b[i] = bsum;

            range[i] = range[pos];
            pos = range[pos]+1;
            if(pos<n)
            {
                asum += a[pos];
                bsum += b[pos];
            }
        }
    }

    long long g;
    for(int i=0;i<n;i++)
    {
        g=gcd(b[i],a[i]);
        printf("%lld/%lld\n",a[i]/g,b[i]/g);
    }
    printf("\n");
}

int main()
{
    long long cases,n;

    scanf("%lld",&cases);
    while(cases--)
    {
        scanf("%lld",&n);
        solve(n);
    }
    return 0;
}
