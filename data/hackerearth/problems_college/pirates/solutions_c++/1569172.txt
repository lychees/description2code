#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0; i<n; i++)
    	cin>>a[i];
    int max = -1, max_till_now = -1;
    int i = 1;
    while(i<n)
    {
    	int cnt = 1;
    	while(i<n && abs(a[i]-a[i-1])==1)
    	{
    		cnt++;
    		i++;
    	}
    	if(max_till_now<cnt)
    	{
    		max_till_now  = cnt;
    	}
    	i++;
    }
    cout<<max_till_now<<endl;
    return 0;
}
