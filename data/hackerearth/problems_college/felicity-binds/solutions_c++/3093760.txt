#include <stdio.h>
int main()
{
    long long int t;
    scanf("%lld",&t);
    while(t--)
    {
    	long long int n,m,i,j,sum=0,temp1,temp2;
    	scanf("%lld",&n);
    	long long int a[n],b[n];
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&a[i]);
    	}
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&b[i]);
    	}
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<n-1-i;j++)
    		{
    			if(a[j]>a[j+1])
    			{
    				temp1=a[j];
    				a[j]=a[j+1];
    				a[j+1]=temp1;
    			}
    		}
    	}
    		for(i=0;i<n;i++)
    	{
    		for(j=0;j<n-1-i;j++)
    		{
    			if(b[j]>b[j+1])
    			{
    				temp2=b[j];
    				b[j]=b[j+1];
    				b[j+1]=temp2;
    			}
    		}
    	}
    	for(i=0;i<n;i++)
    	{
    		sum=sum+(a[i]*b[i]);
    	}
    	printf("%lld\n",sum);
    }
    return 0;
}
