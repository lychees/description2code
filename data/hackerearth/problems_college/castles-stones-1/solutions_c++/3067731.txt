

#include <iostream>
#include <algorithm>
#define lli long long int

using namespace std;

int main()
{

    lli test, n, m, a[1000] ;

    lli i, j, ans ;

    cin >> test ;

    while(test--)
    {

        ans = 0 ;

        cin >> m >> n ;

        for(i=0;i<n;i++)
        {
            cin >> a[i] ;
        }

        sort(a,a+n) ;

        i = 0 ;
        j = n-1 ;

        while(i<j)
        {

            if(a[i]+a[j]<=m)
            {

                a[i] = -1 ;
                a[j] = -1 ;

                i++ ;
                j-- ;

                ans++ ;

            }
            else
            {
                j-- ;
            }

        }

        for(i=0;i<n;i++)
        {
            if(a[i]!=-1 && a[i]<=m)
                ans++ ;
        }

        cout << ans << "\n" ;

    }

}
