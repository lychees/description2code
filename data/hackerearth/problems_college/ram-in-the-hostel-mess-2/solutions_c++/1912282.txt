#include<iostream>
#include<stdio.h>
#include<math.h>
#include<algorithm>
using namespace std;
#define lli long long int

lli factorial(lli n) {
   lli c;
   lli result = 1;
 
   for (c = 1; c <= n; c++)
      result = result*c;
 
   return result;
}

lli find_ncr(lli n, lli r) {
   lli result;
 
   result = factorial(n)/(factorial(r)*factorial(n-r));
 
   return result;
}


int main()
{
    lli t,n,i,ans;
    cin>>t;
    while(t--)
    {
              cin>>n;
              
              ans=0;
              
              for(i=0;i<=(n+1)/2;i++)
              ans+=find_ncr(n-i+1,i);
              
              cout<<ans*ans<<endl;
              
    }
    
    return 0;
}
