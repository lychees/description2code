#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;
		cin>>m>>n;
		int a[n];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a,a+n);
		int i;
		for( i=n-1;i>=0;i--)
		{
			if(a[i]<=m)
			{
				break;
			}
		}
		int c=0,k=0;
		for(int j = i;j>=0,k<=j;j--)
		{
			if(a[j] == m)
			{
				c++;
			}
			else
			{
				if(a[j] + a[k] <=m)
				{
					c++;
					k++;
				}
			}
		}
		cout<<c<<"\n";

	}
}
