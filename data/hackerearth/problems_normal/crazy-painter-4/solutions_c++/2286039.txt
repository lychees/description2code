#include <iostream>
using namespace std;

int main()
{
	long long i,j,k,l,m,n,t;
	cin>>t;
	while(t--){
		cin>>n;
		long long c[30];
		long long s=0;
		for(i=0;i<30;i++) c[i]=0;
		for(i=1;i<=26;i++) cin>>c[i];
		m=12*n;
		long long d=0;
		for(i=1;i<=26;i++){
			if(m>=i){
				d =1+ (m-i)/26;
				if(d%2==0 && d>0){
					long long tmp = c[i]+d-1+c[i];
					s+=tmp*d/2;
				}
				else if(d%2==1){
					if(d==1){
						s+=c[i];
					}
					else{
						long long tmp=(c[i]+d-1+c[i]);
						s+=(tmp*(d/2));
						s+=tmp/2;
					}
				}
			}
			//cout<<d<<" "<<s<<"\n"; 
		}
		cout<<s<<"\n";
	}
    return 0;
}
