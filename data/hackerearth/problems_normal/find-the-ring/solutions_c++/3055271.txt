#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long int t,n,i;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld %lld",&i,&n);
		if(n==0) { printf("%lld\n",i); continue; }
		
		if(i==0)
		{
			if(n%2==0) printf("0\n"); 
			else printf("1\n");
		}
		else if(i==1)
		{
			if(n%2==0) printf("1\n");
			else printf("0\n");
		}
		else
		{
			if(n%2==0) printf("0\n");
			else printf("1\n");
		}
		
		
	}
	
}
