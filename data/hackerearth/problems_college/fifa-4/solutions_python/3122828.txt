n = int(raw_input())
home = [0]*n
away = [0]*n
for i in range(n):
    t = raw_input().strip().split(' ')
    home[i], away[i] = int(t[0]), int(t[1])
memo = dict.fromkeys(home,0)
for i in home:
    memo[i] += 1

for i in range(n):
    t = memo.get(away[i], 0)
    y = n-1-t
    x = n-1+t
    print x, y
