#include <bits/stdc++.h>

using namespace std;
#define MAXN 105
int N,M,T,grid[MAXN+5][MAXN+5],dp[MAXN+5][MAXN+5];

int main()
{
    scanf("%d",&T);
    while (T--){
        int ans = 0;
        scanf("%d%d",&N,&M);
        for (int i=1;i<=N;i++)
            for (int j=1;j<=M;j++)
                scanf("%d",&grid[i][j]);

        for (int i=0;i<MAXN;i++)
            for (int j=0;j<MAXN;j++)
                dp[i][j]=-1;

        dp[1][1]=1;
        for (int i=1;i<=N;i++)
        for (int j=1;j<=M;j++){
            if (grid[i-1][j] < grid[i][j] && dp[i-1][j]!=-1)
                dp[i][j] = max(dp[i][j],dp[i-1][j]+1);
            if (grid[i][j-1] < grid[i][j] && dp[i][j-1]!=-1)
                dp[i][j] = max(dp[i][j],dp[i][j-1]+1);
            ans = max(ans,dp[i][j]);
        }
        printf("%d\n",ans);
        
    }
}
