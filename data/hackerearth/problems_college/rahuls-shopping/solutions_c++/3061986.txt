#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;
int main()
{
	float ans=0;
	int n;
	cin>>n;
	float a[n];
	int plank[n];
	for(int i=0;i<n;i++)
	{
		int x,dig=0;
		cin>>x;
		plank[i]=x;
		int r=x;
		while(r)
		{
			int temp=r%10;
			dig+=temp;
			r/=10;
		}
		a[i]=dig*x;
	}
	int len,wid;
	cin>>len>>wid;
	int lw=len*wid;
	for(int i=0;i<n;i++)
	{
		a[i]*=lw;
		if(plank[i]<10)
			a[i]*=0.75;
		else if(plank[i]<20)
			a[i]*=0.70;
		else
			a[i]*=0.80;
		a[i]=floor(a[i]);
		ans+=a[i];
	}
	cout<<ans<<endl;	
}
