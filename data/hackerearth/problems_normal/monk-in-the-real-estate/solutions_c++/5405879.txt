#include <iostream>
#include<set>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--){
    	int E, x, y;
    	cin>>E;
    	set<int> s;
    	for(int i = 0 ; i < E ; i++ ){
    		cin>>x>>y;
    		s.insert(x);
    		s.insert(y);
    	}
    	cout<<s.size()<<endl;
    }
    return 0;
}
