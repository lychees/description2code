#include <cstdio>
#include <utility>
#include <algorithm>
#include <vector>

using namespace std;

typedef pair<double, double> pdd;

vector<pdd> arr;

bool cmp(pdd const &v1, pdd const &v2){return v1.first < v2.first ||
					(v1.first == v2.first && v1.second < v2.second);}

int main()
{
	double x1,y1,x2,y2,x3,y3;
	scanf("%lf %lf %lf %lf %lf %lf", &x1, &y1, &x2, &y2, &x3, &y3);
	arr.push_back(make_pair(x1+x2-x3,y1+y2-y3));
	arr.push_back(make_pair(x1-x2+x3,y1-y2+y3));
	arr.push_back(make_pair(-x1+x2+x3,-y1+y2+y3));

	sort(arr.begin(), arr.end(), cmp);
	for(int i = 0; i < arr.size(); i++)
		printf("%.4lf %.4lf\n", arr[i].first, arr[i].second);
}
