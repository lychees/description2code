#include <iostream>
#include<iomanip>


using namespace std;
#define ll long long



int main()
{
	ll int t;
	cin>>t;



    while(t--)
    {
    	long double l,b,h;
    	l=0,b=0,h=0;
    	cin>>l>>b>>h;
    	long double ans1=min(l,b);
    	double ans=(double)min(ans1,h)/2;

    	cout<<fixed<<setprecision(1)<<ans<<endl;


    }


    return 0;
}
