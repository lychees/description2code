#include<bits/stdc++.h>
using namespace std;

int hashTable[26];

int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        string str;
        cin>>str;
        memset(hashTable,-1,sizeof(hashTable));
        int len=str.length();
        hashTable[str[0]-'A']=0;
        int maxLen=1;
        int cnt=1;
        int lastIndex=0;
        for(int i=1;i<len;i++) {
            if(hashTable[str[i]-'A']==-1) {
                cnt++;
            } else {
                int j;
                for(j=lastIndex;j<=hashTable[str[i]-'A'];j++) {
                    if(hashTable[str[j]-'A']==j) {
                        hashTable[str[j]-'A']=-1;
                    }
                }
                lastIndex=j;
                cnt=i-lastIndex+1;
            }
            hashTable[str[i]-'A']=i;
            maxLen=max(maxLen,cnt);
        }
        printf("%d\012",maxLen);
    }
    return 0;
}
