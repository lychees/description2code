numberOfTestCases = int(raw_input())

for counter in range(numberOfTestCases):
    temp, temp1 = raw_input().split()
    ring = int(temp)
    ringUpdated = int(temp) % 2
    swaps = int(temp1)
    print(
        ring if swaps == 0 else (
            ringUpdated if swaps %
            2 == 0 else (
                1 if ringUpdated == 0 else 0)))
