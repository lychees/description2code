#include<iostream>
#include<cstring>
using namespace std;
int main()
{
 int t;
 cin>>t;
 while(t--)
 { int c=0,acost,bcost,flag=0;
  char a[100001];
  cin>>a;
  int l=strlen(a);
  cin>>acost>>bcost;
  l--;
  int i,j;
  i=l/2; j=l/2+1;
  while(i>=0)
  {  if(a[i]!=a[j] && a[i]!='/' &&a[j]!='/')
     {cout<<-1<<endl; flag++; break; }
     else if(a[i]!=a[j] && a[i]=='/')
           a[j]=='b'?c+=bcost:c+=acost;
     else if(a[i]!=a[j] && a[j]=='/')
          a[i]=='b'?c+=bcost:c+=acost;
     else if(a[i]==a[j] && a[i]=='/' &&a[j]=='/')
                 acost>bcost?c+=2*bcost:c+=2*acost;
      
        --i;
        ++j;     
  }
    if(flag!=0)
     continue;
     else
    cout<<c<<endl;      
 }
 
 return 0;
    
    
}
