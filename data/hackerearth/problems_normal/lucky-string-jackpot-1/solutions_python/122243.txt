def magic(k):
    return ((k-1)*(2**(k+1)) + 2)

def main(n):
    k = 0
    while magic(k+1) < n:
        k += 1
    diff = n - magic(k)
    if diff >= 2*k+1:
        print 2*k+1
    elif diff <= 2*k-1:
        print 2*k-1
    else:
        print diff

if __name__ == '__main__':
    numTest = int(raw_input())
    for t in range(numTest):
        main(int(raw_input()))
