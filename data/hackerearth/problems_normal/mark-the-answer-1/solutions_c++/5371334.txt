#include <stdio.h>

int main() {
	int n, x, i;
	scanf("%d%d",&n,&x);
	int arr[n];
	for(i=0;i<n;i++)
		scanf("%d",&arr[i]);
	int done = 0, ret = 0;
	for(i=0;i<n;i++) {
		if(arr[i] <= x)
			ret++;
		else {
			if(!done)
				done = 1;
			else
				break;
		}
	}
	printf("%d\n",ret);
	return 0;
}
