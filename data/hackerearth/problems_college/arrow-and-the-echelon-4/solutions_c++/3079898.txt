#include <iostream>
#include <cstdio>
#include <string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string  s[t];
    for(int m=0;m<t;m++)
    {
        s[m]="";
    }
    for(int m=0;m<t;m++)
    {
        long long n;
        cin>>n;
        if((n>=5 && n!=7))
        {
            if(n%3==0)
            {
                for(int i=1;i<=n;i++)
                {
                    s[m]=s[m].append("5");
                }
            }
            else if(n%3==1)
            {
                int x=(n-10);
                for(int i=1;i<=x;i++)
                {
                    s[m]=s[m].append("5");
                }
                for(int j=1;j<=10;j++)
                {
                    s[m]=s[m].append("3");
                }
            }
            else
            {
                int x=(n-5);
                for(int i=1;i<=x;i++)
                {
                    s[m]=s[m].append("5");
                }
                for(int j=1;j<=5;j++)
                {
                    s[m]=s[m].append("3");
                }
            }
        }
        else if(n==3)
        {
            s[m]="555";
        }
        else
        {
            s[m]="-1";
        }
    }
    for(int m=0;m<t;m++)
    {
        cout<<s[m]<<endl;
    }
    return 0;
}
