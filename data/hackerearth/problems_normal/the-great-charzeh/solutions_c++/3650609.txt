#include <bits/stdc++.h>
using namespace std;

#define ll long long
const int N = 605;

int a[N], f[N], match[N];
vector<int> adj[N];
bool visited[N];
int n;

bool augment(int l){
    if(visited[l])  return 0;
    visited[l] = 1;
    for(auto r : adj[l]){
        if(match[r] == -1 or augment(match[r])){
            match[r] = l;
            return 1;
        }
    }
    return 0;
}

bool valid(int x){

    for(int i = 0; i < 2*n; i++){
        adj[i].clear();
        match[i] = -1;
    }

    for(int i = 0; i < n; i++){
        for(int j = 0; j < n; j++){
            int val = f[j + a[i]];
            if(val > x) continue;
            adj[i].push_back(n + j);
        }
    }

    int ans = 0;

    for(int i = 0; i < n; i++){
        for(int j = 0; j < 2*n; j++)    visited[j] = 0;
        ans += augment(i);
    }

    return (ans == n);
}

int main(){
    int m;
    cin>>n>>m;

    for(int i = 0; i < n; i++)  cin>>a[i];
    for(int i = 0; i <= n + m - 2; i++)  cin>>f[i];

    int lo = 1, hi = 1e9;
    while(lo < hi){
        int mid = (lo + hi)/2;
        if(valid(mid))    hi = mid;
        else lo = mid + 1;
    }
    cout<<lo;
}
