#include <cstdio>
using namespace std;
int main()
{
int t;
 scanf("%d",&t);
 while(t--)
 {
 int n;
 scanf("%d", &n);
 int a[n];
 int i;
for(i=0; i<n; i++)
  scanf("%d", &a[i]);
 
 int l = 0, r = 0;

 for(i=0; i<n; i++)
  if(a[i] == -1)
   l++;
  else
   break;
 for(i=n-1; i>=0; i--)
  if(a[i] == 1)
   r++;
  else
   break;
  int leftgoingants = 0;
 for(i=l; i <= n-r-1; i++){
  if(a[i] == -1){
   leftgoingants ++;
  }
 }
 if(l >= r)
  printf("%d\n", l + leftgoingants);
else
  printf("%d\n", l + leftgoingants + 1);
 
 }
}
