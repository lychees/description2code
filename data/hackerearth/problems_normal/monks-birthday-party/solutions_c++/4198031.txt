#include <iostream>
#include <algorithm>
#include <set>
#include <string>

using namespace std;

#define ll long long int

set<string> S;

int main() {
	int T;
	cin >> T;
	while (T--) {
		S.clear();
		int n;
		cin >> n;
		for (int i = 0; i < n; i++) {
			string x;
			cin >> x;
			S.insert(x);
		}
		for (auto a : S) {
			cout << a << endl;
		}
	}

}
