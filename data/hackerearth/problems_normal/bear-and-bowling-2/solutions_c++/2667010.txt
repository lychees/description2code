#include <bits/stdc++.h>
using namespace std;

#define F(i, x) for(long long i=0; i<x; i++)
#define F1(i, x) for(long long int i=1; i<x; i++)
#define ll long long
#define p pair<ll, ll>
#define mp(x, y) make_pair(x, y)

ll a[100003], max_l[100003], max_r[100003];
int main()
{
	ll n, l=0, r=0, m=0, cmax=0, c=0;
	scanf("%lld", &n);
	F(i, n)
	{
		scanf("%lld", &a[i]);
		m=max(m, a[i]);
	}
	max_l[0]=a[0];
	max_r[n-1]=a[n-1];
	if(a[0]==m)
	c++;
	F1(i, n)
	{
		if(a[i]==m)
		{
			c++;
		}
		else
		{
			if(c>cmax)
			{
				cmax=c;
				r=i-1;
				l=r-cmax+1;
			}
			c=0;
		}
		//cout<<c<<"\n";
		max_l[i]=max(max_l[i-1], a[i]);
		max_r[n-1-i]=max(max_r[n-i], a[n-1-i]);
	}
	if(c>cmax)
		{
			cmax=c;
			c=0;
			r=n-1;
			l=r-cmax+1;
		}
	F(i, cmax)
	{
		printf("1");
	}
	while(r-l!=n-1)
	{
		if(l>0 && r<n-1)
		{
			if(a[l-1]>a[r+1])
			l--;
			else if(a[l-1]<a[r+1])
			r++;
			else
			{
				if(max_l[l-1]>max_r[r+1])
				l--;
				else
				r++;
			}
		}
		else if(l>0)
		l--;
		else if(r<n-1)
		r++;
		m=min(a[l], m);
		m=min(a[r], m);
		
		if(l>0 && r<n-1)
		{
			if(m>=max_l[l-1] && m>=max_r[r+1])
			printf("1");
			else printf("0");
		}
		else if(l>0)
		{
			if(m>=max_l[l-1])
			printf("1");
			else printf("0");
		}
		else if(r<n-1)
		{
			if(m>=max_r[r+1])
			printf("1");
			else printf("0");
		}
		else printf("1");
	}
	return 0;
}
