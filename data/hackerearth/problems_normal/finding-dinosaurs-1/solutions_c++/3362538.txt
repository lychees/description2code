#include <map>
#include <iostream>
using namespace std;

int main() {
    int N, K, L, Q;
    map<string, string> Dinosaur;
    cin >> N >> K >> L >> Q;
    
    string s, key = "";
    int tuple[K];
    // Take inputs for dinosaurs
    for (int i = 0; i < N; i++) {
        cin >> s;
        
        for (int j = 0; j < K; j++) {
            cin >> tuple[j];
            key = key + " " + to_string(tuple[j]);
        }
        Dinosaur[key] = s;
        key = "";
    }
    
    // find dinosaur from tuples
    key = "";
    for (int i = 0; i < Q; i++) {
        for (int j = 0; j < K; j++) {
            cin >> tuple[j];
            key = key + " " + to_string(tuple[j]);
        }
        
        if (Dinosaur.find(key) == Dinosaur.end()) {
            cout << "You cant fool me :P" << endl;
        } else {
            cout << Dinosaur[key]<<endl;
        }
        key = "";
    }
    
    return 0;
}
