#include <iostream>
using namespace std;

int main()
{
    int n,m,q,a;
    int arr[1002][1002];
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;i++)
    for(int j=0;j<m;j++)
    scanf("%d",&arr[i][j]);
    scanf("%d",&q);
    while(q--)
    {
    	scanf("%d",&a);
    	int i=0;
    	int j=m-1;
    	while(i<n&&j>=0)
    	{
    		if(arr[i][j]==a)
    		break;
    		else if(a>arr[i][j])
    			i++;
    		else j--;
    	}
    	if(i<n&&j>=0)
    	cout<<i<<" "<<j<<"\n";
    	else
    	cout<<-1<<" "<<-1<<"\n";
    }
    return 0;
}
