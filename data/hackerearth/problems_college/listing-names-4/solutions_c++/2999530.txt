#include <iostream>
#include <map>
using namespace std;
 
int main()
{
	int t;
	string s;
	map <string,int> m;
	cin>>t;
	while(t--){
		cin>>s;
		m[s]++;
	}
	
	map<string,int> :: iterator it;
	
	for(it=m.begin();it!=m.end();it++){
		cout<<it->first<<" "<<it->second<<endl;
	}
	
	return 0;
}
