#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    string s1,s2;
    vector<char> a;
    while(t--)
    {
    	cin>>s1>>s2;
    	int len=s1.length();
    	a.resize(len);
    	for(int i=0;i<len;i++)
    	{
    		if(s1[i]==s2[i])
    		{
    			a[i]='0';
    		}
    		else
    		{
    			a[i]='1';
    		}
    	}
    	for(int i=0;i<len;i++)
    	{
    		cout<<a[i];
    	}
    	cout<<endl;
    	a.clear();
    }
}
