#include<bits/stdc++.h>
using namespace std; 
#define mod 10000007
int main()
{
	long long n,series[1000002],i,test;
	series[1]=1; series[2]=1;series[3]=2;series[4]=4;
    for(i=5;i<=1000000;i++)
    {
        series[i]=((series[i-1]%mod+series[i-3]%mod)%mod+series[i-4]%mod)%mod;
    }
    cin>>test;
    while(test--)
    {
        cin>>n;
        cout<<series[n]<<endl;
    }
 
}
