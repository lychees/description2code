#include <iostream>
#include<algorithm>
#include<vector>
#include<cstring>
using namespace std;

int main()
{
    long long t;
    cin>>t;
    while(t--)
    {
    	long long n,m,flag=1;
    	cin>>n>>m;
    	string res;
    	vector <string> a;
    	for(long long i=0;i<n;++i)
    	{
    		cin>>res;
    		if(i==0)
    		  a.push_back(res);
    		  else
    		  {
    		if(std::find(a.begin(),a.end(),res)!=a.end())
    		{
    			flag=0;
    			
    		}
    		else
    		a.push_back(res);}
    	}
    	if(flag==1)
    	  cout<<"\nYes";
    	  else
    	  cout<<"\nNo";
    }
}
