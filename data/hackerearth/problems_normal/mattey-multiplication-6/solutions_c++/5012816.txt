        #include <iostream>
        #include <cstdio>
        #include <string>
        using namespace std;
         
        typedef long long UL;
        int main()
        {
        	ios::sync_with_stdio(false);
            int t, i, j, k = 0;
            string s = "";
            cin >> t;
            for(i = 0; i <= t; ++i) {
            	UL n, m;
            	cin >> n >> m;
            	s = "";
            	k = 0;
            	for(; m; m >>= 1, ++k) {
            		if (m & 1) {
            			s = "(" + to_string(n) + "<<" + to_string(k) + ") + " + s;
            		}
            	}
            	cout << s.substr(0, s.size() - 2) << "\n";
            }
            return 0;
        }
