#include <bits/stdc++.h>
typedef long long ll;
using namespace std;

ll findMin(ll *a, ll n) {
	ll i;
	ll maxi = 10000007;
	for (i = 0; i < n; i++) {
		if (a[i] < maxi && a[i] != 0) {
			maxi = a[i];
		}
	}
	return maxi;
}

int main()
{
    ll n, i, j;
    cin>>n;
    ll a[n];
    int f[1001];
    for (i = 0; i <= 1000; i++) {
    	f[i] = 0;
    }
    ll d = 0;
    for (i = 0; i < n; i++) {
    	cin>>a[i];
    	if (f[a[i]] == 0) {
    		d += 1;
    		f[a[i]] += 1;
    	}
    }ll ctr = 0;
    for (i = 0; i < d; i++) {
    	int m = findMin(a, n);
    	for (j = 0; j < n; j++) {
    		if ((a[j] - m) >= 0) {
    			a[j] -= m;
    			ctr += 1;
    		}
    	}
    	cout<<ctr<<endl;
    	ctr = 0;
    }
    return 0;
}
