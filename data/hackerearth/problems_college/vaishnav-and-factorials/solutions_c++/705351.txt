#include <iostream>
using namespace std;

int main()
{
    int t;
        scanf("%d",&t);
        int i;
        while(t--) {
            int  N;
            scanf("%d", &N);
            int arr[2000];
            arr[0]=1;
            int length=1,j,temp,carry=0;
            int count=0;
            for(i=2;i<=N;i++) {
                for(j=0;j<length;j++) {
                    temp=carry+arr[j]*i;
                    carry=temp/10;
                    arr[j]=temp%10;
                }
                while(carry) {
                    arr[j]=carry%10;
                    carry=carry/10;
                    j++;
                }
                length=j;
            }
            for(int i=length-1;i>=0;i--){
                if(arr[i]==4 || arr[i]==7)
                {
                	count++;
                }
                //printf("%d",arr[i]);
            }
            
            printf("%d\n",count);
        }
    return 0;
}
