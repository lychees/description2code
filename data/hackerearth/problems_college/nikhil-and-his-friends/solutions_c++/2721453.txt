#include<bits/stdc++.h>
using namespace std;
int main(){
long t;
 long long N,co,q=0;
cin>>t;
while(t--)
{
    q=0;
    cin>>N>>co;
    int a[N];
    for(int i=0;i<N;i++)
        cin>>a[i];
        for(int i=0;i<N;i++){
                  int s=0;
        for(int j=i;j<N;j++){
                s=s+a[j];
            if(s==co)
            {
                cout<<"YES"<<endl;
                 q=1;
            }
            if(s>=co) break;
        }
        if(q==1)
            break;
        }
        if(q==0)
            cout<<"NO"<<endl;
}
}
