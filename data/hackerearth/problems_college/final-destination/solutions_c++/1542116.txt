#include <cstdio>
#include <cassert>
#include <queue>

#define MAX	1000000


int main(void)
{
	int t;
	int x;
	int y;
	int z;
	int m, d, a, s;
	int best[1 + MAX];
	std::queue<int> q;

	for(scanf("%d", &t); t --;)
	{
		for(m = MAX + 1; m --;)
			best[m] = -1;

		while(!q.empty())
			q.pop();

		scanf("%d %d %d %d %d %d", &x, &y, &m, &d, &a, &s);

		best[x] = 0;
		q.push(x);

		while(best[y] == -1 && !q.empty())
		{
			x = q.front();
			q.pop();

			z = x + a;
			if(z >= 2 && z <= MAX && best[z] == -1)
			{
				best[z] = best[x] + 1;
				q.push(z);
			}


			z = x - s;
			if(z >= 2 && z <= MAX && best[z] == -1)
			{
				best[z] = best[x] + 1;
				q.push(z);
			}


			z = x * m;
			if(z >= 2 && z <= MAX && best[z] == -1)
			{
				best[z] = best[x] + 1;
				q.push(z);
			}


			z = x / d;
			if(z >= 2 && z <= MAX && best[z] == -1)
			{
				best[z] = best[x] + 1;
				q.push(z);
			}
		}

		printf("%d\n", best[y]);
	}

	return 0;
}
