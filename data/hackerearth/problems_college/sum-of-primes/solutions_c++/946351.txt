#include<stdio.h>
#include<math.h>
#define M 1000001
 long long int p[M];
int check(long long sum)
{
    long long i;
    if(sum%2==0)
        return false;
    for( i=3;i*i<=sum;i+=2)
        {
            if(sum%i==0)
             return false;
        }
    return true;

}
int main()
{
    long long int a,b,i,j;
    int f,q;
    for(i=0;i<=M;i++)
        p[i]=1;
    p[0]=p[1]=0;

    for(i=2;i*i<=M;i++)
    if(p[i])
    for(j=i*i;j<M;j+=i)
    p[j]=0;
int k;


    j=0;
    for(i=2;i<M;i++)
    {
        if(p[i])
        {
           p[i]+=i-1+j;
           j=p[i];
        }

        p[i]= j;
    }

      scanf("%d",&q);

      while(q--)
      {
          scanf("%lld%lld",&a,&b);

          j=p[b]-p[a-1];
          if(j<=M)
          {
              if(p[j]>p[j-1])
                f=1;
              else
                f=0;
          }else
          f=check(j);

          if(f)
          printf("YES\n");
          else
          printf("NO\n");

      }

    return 0;
}
