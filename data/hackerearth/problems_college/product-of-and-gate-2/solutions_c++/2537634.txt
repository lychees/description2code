#include <bits/stdc++.h>
using namespace std;
long long And(long long a, long long b) {
    long long x = a^b;
    long long s = x>>1;

    while (s) {
        x = x|s;
        s >>= 1;
    }
    return a&b&~x;
}

int main() {	

	int t;
	scanf("%d",&t);
	while(t--){
		long long  a,b;
		cin >> a >> b;
		cout << And(a,b) << endl;
	}
 	return 0;
}
