#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll arr[100100],f[100100][3],barr[100100],func[100100];
int main()
{
	ll i,j,k,t,n,m,q;
	cin>>t;
	while (t--)
	{
		cin>>n>>m;
		for (i=0;i<m;i++)
		cin>>f[i][0]>>f[i][1]>>f[i][2];
		for (i=0;i<n;i++)
		barr[i]=0;
		for (i=0;i<m;i++)
		arr[i]=0;
		cin>>q;
		while (q--)
		{
			cin>>j>>k;
			arr[j-1]++;
			arr[k]--;
		}
		ll sum=0;
		for (i=0;i<m;i++)
		{
			sum+=arr[i];
			func[i]=sum;
		}
		for (i=0;i<m;i++)
		{
			barr[f[i][0]-1]+=f[i][2]*func[i];
			barr[f[i][1]]-=f[i][2]*func[i];
		}
		sum=0;
		for (i=0;i<n;i++)
		{
			sum+=barr[i];
			cout<<sum<<" ";
		}
		cout<<endl;
	}
	return 0;
}
