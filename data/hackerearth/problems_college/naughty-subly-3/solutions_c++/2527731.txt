#include <iostream>
#include<cstring>
using namespace std;

int main()
{
	cout.sync_with_stdio(false);
	cin.sync_with_stdio(false);
	long n;
	cin>>n;
	if(n<1 || n>100000) return 1;
	static int set[1001], x, count;
	
	for(long i=0;i<n;i++) 
	{
		cin>>x;
		if(x<1 || x>1000) 
			return 1;
		else 
		{
			++set[x];
		    if (set[x] >1)count++;        
		}	
	
	}
    cout <<count;
    return 0;
}
