#define MOD 1000000007
long long gcd(long long x,long long y)
{
	if(y==0)
		return x;
	return gcd(y,x%y);
}
vector<int> a,b;
int main(void){
	//ifstream cin("test.txt");
	std::ios::sync_with_stdio(false);cin.tie(0);
	int T;
	long long x,y,k;
	int i,m;
	cin>>T;
	while(T--)
	{
		cin>>x>>y>>k;
		x=gcd(x,y);
		a.clear();
		b.clear();
		m=sqrt(x);
		for(i=1;i<m;i++)
		{
			if(x%i==0)
			{
				a.push_back(i);
				b.push_back(x/i);
			}
		}
		if(x%m==0)
		{
			a.push_back(m);
			if(x!=(long long)m*m)
				b.push_back(x/m);
		}
		if(a.size()+b.size()<k)
			cout<<"No crymeth today\n";
		else if(b.size()>=k)
			cout<<b[k-1]<<'\n';
		else
		{
			k-=b.size();
			cout<<a[a.size()-k]<<'\n';
		}
	}
//	system("pause");
	return 0;
}
