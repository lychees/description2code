#include <bits/stdc++.h>
using namespace std;
#define FOR(i,a,n) for(int i=a;i<n;i++) //for loop
#define si(x) scanf("%d",&x) // scan int
#define sll(x) scanf("%lld",&x) // scan ll
#define ss(x) scanf("%s",x) // scan string
#define pi(x) printf("%d\n",x) // print int
#define pll(x) printf("%lld\n",x) // print ll
#define ps(x) printf("%s\n",ch) // print string
#define INF 2000000000 // infinity
#define ll long long
#define mp make_pair
#define pb push_back
#define fr first
#define se second
int main()
{
	int n;
	si(n);
	map<string,int>m;
	vector<pair<string,int> > v;
	FOR(i,0,n)
	{
	    string s;
	    cin>>s;
	    int p;
	    si(p);
	    v.pb(mp(s,p));
	    m[s]+=p;
	}
	map<string,int>::iterator it;
	string ans;
	int MAX = -9999999;
	for(it=m.begin();it!=m.end();it++)
    {
        if(it->se > MAX)
            MAX = it->se;
    }
    map<string,int> m2;
    FOR(i,0,v.size())
    {
        m2[v[i].fr] += v[i].se;
        if(m2[v[i].fr]==MAX)
        {
            ans = v[i].fr;
            break;
        }
    }
    cout<<ans<<endl;
	return 0;
}
