#include <iostream>
using namespace std;

int main()
{
     int t;
    cin>>t;
    long  x[t],y[t],count=0;
    long k[t];
    for(int i=0;i<t;i++)
    cin>>x[i]>>k[i];
    for(int i=0;i<t;i++)
    { count=0;
       y[i]=0;
      while(count<=k[i])
      {  if((x[i]+y[i])==(x[i]^y[i]))
      	count++;
      	y[i]++;
      }
      cout<<"Case #"<<i+1<<": "<<y[i]-1<<endl;	
    }
    
    return 0;
}
