#include <iostream>
using namespace std;

int main()
{
    int testCases = 0;
    long rows, cols, min, natural, rowSum, colSum;
    cin >> testCases;
    
    for (int i = 0; i < testCases; i++)
    {
    	cin >> rows >> cols;
    	if (rows > cols)
    		min = cols;
    	else
    		min = rows;
    	natural = (min * (min + 1)) / 2;
    	min = min - 1;
    	rowSum = (min * ((2 * rows) + (min - 1) * rows)) / 2;
    	colSum = (min * ((2 * cols) + (min - 1) * cols)) / 2;
    	cout << natural * 2 + rowSum + colSum << endl;
    }
    
    return 0;
}
