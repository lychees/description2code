#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	int T;
	cin>>T;
	string str;
	int *alpha = new int[27]; 
	while(T--){
		cin>>str;
		int len = str.length();
		
		for(int i=0;i<26;i++)
		alpha[i] = 0;
		
		for(int i=0;i<len;i++)
		alpha[str[i]-'a']++;
		
		int odd = 0;
		for(int i=0;i<26;i++){
			if(alpha[i]%2==1)
			odd++;
		}
		if(odd > 1)
		cout<<"No"<<endl;
		else
		cout<<"Yes"<<endl;
	}
    return 0;
}
