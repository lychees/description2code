#include<iostream>
#include<cmath>
using namespace std;
#define ll long long 
int main()
{
	int test;
	cin>>test;
	while(test--)
	{
		ll x;
		cin>>x;
		ll root=sqrt(x);
		ll a=(root+1)*(root+1)-x;
		ll b=x-root*root;
		if(a==0 || b==0)
		{
			cout<<"YES\n";
		}
		else if(a<b)
		{
			cout<<"+"<<a<<endl;
		}
		else
		cout<<"-"<<b<<endl;
	}
	return 0;
}
