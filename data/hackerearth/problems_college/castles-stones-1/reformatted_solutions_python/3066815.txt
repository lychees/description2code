for _ in xrange(input()):
    m = input()

    t = map(int, raw_input().split())

    s = t[0]

    a = t[1:]

    a.sort(reverse=1)

    ans = 0

    posT = 0

    while a[posT] > m:

        posT += 1

    endT = s-1

    while posT <= endT:
        ans += 1
        if a[endT]+a[posT] <= m:
            endT -= 1
        posT += 1
    print ans
