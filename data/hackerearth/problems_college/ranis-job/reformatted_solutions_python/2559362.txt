t = int(raw_input())


def chk(s, z, q):
    if (s <= 0):
        return (q)
    else:
        m = max(z)
        z.remove(m)
        s = s-m
        q += 1
        return (chk(s, z, q))


while (t):
    n = int(raw_input())
    l = list(map(lambda x: int(x), raw_input().split()))
    s = sum(l)
    z = list(map(lambda x: int(x), raw_input().split()))
    print(chk(s, z, 0))
    t -= 1
