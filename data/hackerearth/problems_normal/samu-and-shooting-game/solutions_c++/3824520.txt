#include <stdio.h>
#include <math.h>
#include <string.h>
#include <memory.h>
#include <algorithm>
using namespace std;

#define ep 1e-6
#define N 1011
#define min(a, b) ((a) < (b) ? (a) : (b))
#define max(a, b) ((a) > (b) ? (a) : (b))

int n, w;
int x, y;
double p, q;
double ans[N][N];

int main(){
    //freopen("a.in", "r", stdin);
    int Tc;
    for(scanf("%d", &Tc); Tc --;){
        scanf("%d %d %d %d %lf %lf", &x, &y, &n, &w, &p, &q);
        p /= 100, q /= 100;

        memset(ans, 0, sizeof(ans));
        ans[0][0] = 1.0;
        for(int i = 1; i < N; i ++) {
            ans[i][0] = 1;
            for(int j = 1; j < N; j ++){
                if(j < x) ans[i][j] = max(ans[i][j], p + ans[i - 1][j] * (1 - p));
                if(j >= x) ans[i][j] = max(ans[i][j], ans[i - 1][j - x] * p + ans[i - 1][j] * (1 - p));
                if(j < y) ans[i][j] = max(ans[i][j], q + ans[i - 1][j] * (1 - q));
                if(j >= y) ans[i][j] = max(ans[i][j], ans[i - 1][j - y] * q + ans[i - 1][j] * (1 - q));
            }
        }

        printf("%lf\n", ans[n][w] * 100);
    }
}
