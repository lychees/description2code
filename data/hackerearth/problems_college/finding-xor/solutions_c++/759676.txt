#include <iostream>
#include <sstream>

using namespace std;

/*
 * 
 */
int main() {
    string a,b,r[100];
    
    int i,j,t;
    cin>>t;
    for(i=0;i<t;i++){
        stringstream str;
        cin>>a>>b;
        int l=a.length();
        for(j=0;j<l;j++)
            if(a[j]==b[j])
                str<<0;
            else
                str<<1;
        r[i]=str.str();
    }
    for(i=0;i<t;i++)
        cout<<r[i]<<'\n';
    return 0;
}
