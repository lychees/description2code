typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
typedef pair<ll,ll> ii;


const int M=1e9+7;
ll n,fct[100],cmb[100][100],res;
char s[100];




int main()
{
	std::ios::sync_with_stdio(0);
	cin.tie(0);
	cin>>n>>s;
	if(n==1)
	{
		cout<<"1\n";
		return 0;
	}
	fct[0]=1;
	for(int i=1;i<=n;i++)fct[i]=(fct[i-1]*i)%M;
	for(int i=1;i<=n;i++)
	{
		cmb[i][0]=cmb[i][i]=1;
		for(int j=1;j<i;j++)cmb[i][j]=(cmb[i-1][j]+cmb[i-1][j-1])%M;
	}
	res=0;
	for(int i=1;i<=n;i++)
	{
		if(i&1)res=(res+fct[n-i]*cmb[n][i])%M;
		else res=(res-fct[n-i]*cmb[n][i])%M;
	}
	res=(fct[n]-res)%M;
	if(res<0)res+=M;
	cout<<res<<"\n";
	return 0;
}
