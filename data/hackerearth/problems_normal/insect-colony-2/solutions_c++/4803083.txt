#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int sum=0;
    	while(n--)
    	{
    		int temp;
    		cin>>temp;
    		sum+=temp;
    	}
    	if(sum%2==0) cout<<"Yes\n";
    	else cout<<"No\n";
    }
    return 0;
}
