#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define pii pair<int,int>
#define mp make_pair
#define pb push_back
#define si(a)     scanf("%d", &a)
#define sl(a)     scanf("%lld", &a)
#define pi(a)     printf("%d", a)
#define pl(a)     printf("%lld", a)
#define pm(str)   printf("%s\n",str)
#define pn        printf("\n")

int main(){
	
	map <int,int> v,v1;
	int t;
	si(t);
	while(t--){
		int m,n;
		scanf("%d %d",&m,&n);
		
		int x,y;
		for(int i = 1;i <= m;i++){
			scanf("%d %d",&x,&y);
			v[x] = 1;
			v1[y] = 1;
		}
		int count = 0;
		int count1 = 0;
		for(int i = 1;i <= n;i++){
			scanf("%d %d",&x,&y);
			if(v[x]){
				count++;
			}
			if(v[x] && v1[y]){
				count1++;
			}
		}
		int h,g;
		scanf("%d %d",&g,&h);
		if(count == g && count1 == h){
			pm("Great");
		}else if(count == g){
			pm("Good");
		}else {
			pm(":(");
		}

	}
	return 0;
}
