for i in range(int(raw_input())):
    line = raw_input().split(" ")
    n, m = int(line[0]), int(line[1])
    x = y = 1
    count = 0
    for j in range(int(raw_input())):
        line = raw_input().split(" ")
        a, b = int(line[0]), int(line[1])
        while True:
            x += a
            y += b
            if x <= n and x >= 1 and y <= m and y >= 1:
                count += 1
            else:
                x -= a
                y -= b
                break
    print count
