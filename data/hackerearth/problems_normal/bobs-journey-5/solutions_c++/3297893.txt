#include <iostream>
#include<set>
#include<string>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		set<char> st;
		for(int i=0;i<n;i++)
		{
			string a;
			cin>>a;
			st.insert(a.at(0));
		}
		if(n==st.size())
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
