#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    int t;
    cin>>t;
    double pi = 3.1415;
    while(t--)
    {
    	double a,b,c;
    	cin>>a>>b>>c;
    	std::cout << std::fixed;
    	std::cout << std::setprecision(4);
    	double ans = pi;
    	ans*=a*a;
    	ans/=c+a-b;
    	ans*=b*b;
    	ans/=b+c-a;
    	ans*=c*c;
    	ans/=a+b-c;
    	ans/=a+b+c;
		std::cout << ans<<endl;
    }
    return 0;
}
