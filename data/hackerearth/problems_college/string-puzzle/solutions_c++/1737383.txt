#include <iostream>
using namespace std;

void swap (char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}
void permute(char *a, int i, int n) 
{
   int j; 
   if (i == n)
     printf("%s\n", a);
   else
   {
        for (j = i; j <= n; j++)
       {
          swap((a+i), (a+j));
          permute(a, i+1, n);
          swap((a+i), (a+j)); //backtrack
       }
   }
} 
int main()
{
    char string[100];
    cin>>string;
    int q,s,e;
    cin>>q;
    for(int i=1;i<=q;i++)
    {
    	cin>>s>>e;
    	permute(string,s,e);
    	cout<<endl;
    }
    return 0;
}
