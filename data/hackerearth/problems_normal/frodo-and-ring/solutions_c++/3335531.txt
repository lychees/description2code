using namespace std;
typedef unsigned long long int ull;
typedef long long int ll;
typedef vector<vector<ll> > mat;
#define N 1000005
#define INF 1e9

ll a[205][205],dp[205][205];

int main(){
	boost;
	int t;
	cin >> t;
	while(t--){
		int n,m;
		cin >> n >> m;
		forp(i,1,n){
			forp(j,1,m){
				char c;
				cin >> c;
				if(c == 'T'){
					a[i][j] = 150;
				} else if(c == 'D'){
					a[i][j] = -100;
				} else{
					a[i][j] = 0;
				}
			}
		}
		forp(i,1,n){
			forp(j,1,m){
				if(i == 1) {
					dp[i][j] = a[i][j] + dp[i][j-1];
				} else if(j == 1){
					dp[i][j] = a[i][j] + dp[i-1][j];
				} else {
					dp[i][j] = a[i][j] + max(dp[i-1][j], dp[i][j-1]);
				}
				if(dp[i][j] < 0){
					dp[i][j] = -INF;
				}
			}
		}
		cout << (dp[n][m] >= 0 ? dp[n][m] : -1) << "\n";
	}
  return 0;
}
