#include<iostream>
#include<string>

using namespace std;

int main()
{
	int count[26],counter=0;
	for (int i=0; i<26; i++)
		count[i] = 0;
	string str;
	cin >> str;
	for (int i=0; i<str.size(); i++)
		count[(int)str[i] - 97]++;
	if (str.size()&1)
	{
		for (int i=0; i<26; i++)
		{
			if (count[i]&1)
				counter++;
		}
		if (counter == 1)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
	else
	{
		for (int i=0; i<26; i++)
		{
			if (count[i]&1)
			{
				counter++;
				break;
			}
		}
		if (counter)
			cout << "NO" << endl;
		else
			cout << "YES" << endl;
	}
}
