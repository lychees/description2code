#include <bits/stdc++.h>
using namespace std;
long long a,mod=1000000006,ans,ans2,len,k,i,l;
char b[1000000];
long long power(long long a,long long b){
	long long x=a,y=1;
	while(b>0){
		if(b%2==1){
			y=y*x;
			y=y%mod;
			
		}
		x=x*x;
		x=x%mod;
		b=b/2;
	}
	return y;
}
long long po(long long a,long long b){
	long long x=a,y=1,mod = 1000000007;
	while(b>0){
		if(b%2==1){
			y=y*x;
			y=y%mod;
			
		}
		x=x*x;
		x=x%mod;
		b=b/2;
	}
	return y;
}
int main()
{
	scanf("%lld",&a);
	scanf("%s",&b);
	len=strlen(b);
	ans=0;
	k=0;
	for(i=len-1;i>=0;i--){
		l=(b[i]-48);
		l=l*power(10,k);
		l=l%mod;
		k++;
		ans=ans+l;
		ans=ans%mod;
	}
	ans2=po(a,ans);
	ans2=ans2%mod;
	cout<<ans2<<endl;
    return 0;
}
