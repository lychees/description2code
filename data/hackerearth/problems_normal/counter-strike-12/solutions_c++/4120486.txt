#include <iostream>
#include <vector>
#include <algorithm>
#define endl '\n'

using namespace std;

int main() {
	ios::sync_with_stdio(false);
	long t, n, m, x, y, d;

	cin >> t;
	while (t--) {
		vector < pair <long, long> > locations;
		vector < pair <long, long> > targets;

		cin >> n >> m >> d;

		for (int i = 0; i < n; ++i) {
			cin >> x >> y;
			locations.push_back(make_pair(x, y));
		}

		for (int i = 0; i < m; ++i) {
			cin >> x >> y;
			targets.push_back(make_pair(x, y));
		}

		long req = m / 2;
		long count = 0;
		bool ok = false;

		for (vector < pair <long, long> >::iterator loc = locations.begin(); loc != locations.end(); ++loc) {
			for (vector < pair <long, long> >::iterator tar = targets.begin(); tar != targets.end(); ++tar) {
				long dist = abs(loc->first - tar->first) + abs(loc->second - tar->second);
				if (dist <= d) {
					++count;
					break;
				}
			}
			if (count > req) {
				ok = true;
				break;
			}
		}
		// cerr << count << " " << m << endl;
		if (ok) {
			cout << "YES" << endl;
		} else {
			cout << "NO" << endl;
		}
	}
}
