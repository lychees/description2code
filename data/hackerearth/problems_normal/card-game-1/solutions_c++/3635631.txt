#include<iostream>
using namespace std;

int
main ()
{
  int tc = 0, N=0;
  cin >> tc;
   int arr1[100000];
   int arr2[100000];

  while (tc--)
    {
      cin >> N;
      for (int i = 0; i < N; i++)
	{
	  cin >> arr1[i];
	}
      for (int i = 0; i < N; i++)
	{
	  cin >> arr2[i];
	}
      char firstMove[10];
      cin>>firstMove;
  
      int ndiff = 0;
      for (int i = 0; i < N; i++)
	{
	  if (arr1[i] != arr2[i])
	    {
	      ndiff++;
	    }
	  if (ndiff > 2)
	    break;
	}

      switch (ndiff)
	{
	case 0:
	  cout << "Should not happen 0" << endl;
	  break;

	case 1:
	  cout << "Should not happen 1" << endl;
	  break;

	case 2:
	  if (firstMove[0] == 'C' && N !=2)
	    cout << "Churu" << endl;
	  else
	    cout << "Devu" << endl;
	  break;

	case 3:
	  cout << "Churu" << endl;
	  break;

	default:
	  cout << "Should not happen default" << endl;
	  break;

	}
    }
}
