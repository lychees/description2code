cases = int(raw_input())

for case in range(cases):
    A, B, C, K = map(int, raw_input().split())
    lb = 0
    ub = 10 ** 5
    ans = -1
    while lb <= ub:
        mid = (lb + ub) / 2
        val = A * mid ** 2 + B * mid + C
        if val >= K:
            ans = mid
            ub = mid - 1
        else:
            lb = mid + 1
    print ans
