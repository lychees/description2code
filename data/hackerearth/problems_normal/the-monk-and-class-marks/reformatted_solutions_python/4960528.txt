marks_dict = {}
for _ in range(int(raw_input())):
    n, m = raw_input().split()
    m = int(m)
    if m in marks_dict:
        marks_dict[m].append(n)
    else:
        marks_dict[m] = []
        marks_dict[m].append(n)

marks = sorted(marks_dict.keys(), reverse=True)
for num in marks:
    names = sorted(marks_dict[num])
    for val in names:
        print val+" "+str(num)
