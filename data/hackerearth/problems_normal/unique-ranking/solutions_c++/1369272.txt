#include<bits/stdc++.h>
#define ll long long
#define ld double
#define f first
#define s second
#define pb push_back
#define M 1000000007
#define MX 1000001
#define INF 10000000000
#define PI 3.141593
#define mmx(a,b) a>b?a:b
#define mmn(a,b) a<b?a:b
using namespace std;
typedef pair<ll,ll> ii;
typedef vector<ll> vv;
vector<ll> students[1005];
vector<ll> arr;
ll k;
int main()
{
    ios_base::sync_with_stdio(false);
    ll n;
    cin>>n>>k;
    for(ll i=0;i<n;i++)
    {
        for(ll j=0;j<k;j++)
        {
            ll a;
            cin>>a;
            students[i].pb(a);
        }
    }
    ll q;
    cin>>q;
    while(q--)
    {
        ll qn;
        vv v;
        arr.clear();
        cin>>qn;
        for(ll i=0;i<qn;i++)
            {
                ll a;
                cin>>a;
                arr.pb(a);
            }
       ll st=arr[0],mn=MX;
       for(ll i=0;i<n;i++)
       {
           if(mn>students[i][st-1])
           {
               mn=students[i][st-1];
               v.clear();
               v.pb(i);
           }
           else if(mn==students[i][st-1])
            v.pb(i);
       }
       for(ll i=1;i<arr.size();i++)
       {
           if(v.size()==1)
            break;
           vv t;
           ll in=arr[i]-1;
           mn=MX;
           for(ll j=0;j<v.size();j++)
           {
              ll k=v[j];
              if(mn>students[k][in])
              {
               mn=students[k][in];
               t.clear();
               t.pb(k);
              }
             else if(mn==students[k][in])
               t.pb(k);
           }
           v.clear();
           for(ll j=0;j<t.size();j++)
            v.pb(t[j]);
       }
       sort(v.begin(),v.end());
       cout<<v[0]+1<<"\n";
    }
}
