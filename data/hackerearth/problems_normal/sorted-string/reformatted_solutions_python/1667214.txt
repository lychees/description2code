t = int(input())

while (t):
    t -= 1
    s = raw_input()
    a = list(s)
    b = set(a)
    c = []
    s2 = ""
    for i in range(1, 101):
        if (len(s2) == len(s)):
            break
        for j in b:
            if (a.count(j) == i):
                c.append(i*j)
        c = sorted(c)
        s2 += ''.join(c)
        c = []
    print(s2)
