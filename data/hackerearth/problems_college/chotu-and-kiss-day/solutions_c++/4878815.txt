#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int i, j , k ,n, m, sum,p=0;
	scanf("%lld", &n);
	long long int a[n];
	for(i=0; i<n; i++)
		scanf("%lld", &a[i]);
	sort(a,a+n);
	scanf("%lld", &m);
	for(i=0; i<n-1; i++)
	{
		sum = m - a[i];
		for(j=i+1, k=n-1; j!=k; )
		{
			if(sum == a[j]+a[k])
			{
				printf("YES\n");
				p++;
				break;
			}
			if(sum < a[j]+a[k])
				k--;
			else
				j++;
		}
		if(p > 0)
		break;
	}
	if(p==0)
	printf("NO\n");
	return 0;
}
