#include<iostream>
using namespace std;
int main()
{
	int n;
	cin>>n;
	long long int A[n];
	for(int i=0;i<n;i++)
	cin>>A[i];
	
	long long int sum1=0;
	long long int sum2=0;
	long long int sum3=0;
	for(int m=0;m<n;m+=3)
	{
		sum1=sum1+A[m];
    }
	for(int m=1;m<n;m+=3)
	{
		sum2=sum2+A[m];
    }
    for(int m=2;m<n;m+=3)
	{
		sum3=sum3+A[m];
    }


	cout<<sum1<<" "<<sum2<<" "<<sum3;
	return 0;
}
