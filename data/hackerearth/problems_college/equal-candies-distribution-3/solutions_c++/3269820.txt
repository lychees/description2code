#include<bits/stdc++.h>
using namespace std;
int main()
{
	int test,n,i,a[100009],ans;
	scanf("%d",&test);
	while(test--)
	{
		ans=0;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		sort(a,a+n);
		int x=0;
		for(i=0;i<n-1;i++)
		{
			if((a[i]!=(a[i+1]+x)))
			{
				ans=ans+(a[i+1]+x-a[i])/5+((a[i+1]+x-a[i])%5)/2+(((a[i+1]+x-a[i])%5)%2);
			//	cout<<ans<<endl;
				x=a[i+1]+x-a[i];
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
