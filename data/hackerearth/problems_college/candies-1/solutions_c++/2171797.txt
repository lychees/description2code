#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int n,i;
    	cin>>n;
    	long long int a[n],sum=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		sum+=a[i];
    		
    	}
    	
    	if(sum%n==0)
    	{
    	cout<<"YES"<<"\n";
    	cout<<sum/n<<" 0"<<"\n";
    	}
    	else
    	{
    		cout<<"NO"<<"\n";
    		cout<<sum/n<<" "<<sum%n<<"\n";
    	}
    }
}
