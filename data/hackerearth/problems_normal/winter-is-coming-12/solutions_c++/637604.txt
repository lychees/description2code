#include<iostream>
#include<cstdio>
#include<set>
using namespace std;

int a[100010],b[100010];

int main()
{
int t,n,i,j,p,m,min,x,y;
scanf("%d",&t);

b[0]=0;

while(t--)
{
scanf("%d",&n);
min=n+1;

for(i=0;i<=n;i++)
{
b[i]=-1;
}

b[0]=0;
a[0]=0;

for(i=1;i<=n;i++)
{
scanf("%d",&m);
a[i]=(a[i-1]+m)%n;
if(b[a[i]]!=-1)
{
p=i-b[a[i]]-1;
if(p<min)
{
min=p;
x=b[a[i]]+1;
y=i;
}
}

b[a[i]]=i;
}

printf("%d %d\n",x,y);
}

return 0;
}
