#include <iostream>
#include<string.h>
using namespace std;
char alice[100005];
char bob[100005];
		
int main()
{
	int t;
	cin>>t;
	char q[100]
    ;
	gets(q);
	while(t--)
	{
		gets(alice);
		gets(bob);
		
		int la=strlen(alice);
		int lb=strlen(bob);
		int a[26]={0},b[26]={0},suma=0,sumb=0;
		
		for(int i=0;i<la;i++)
		{
			if(alice[i]!=' ')
			a[alice[i]-'a']++;
		}
		
		for(int i=0;i<lb;i++)
		{
			if(bob[i]!=' ')
			b[bob[i]-'a']++;
		}
		
		for(int i=0;i<26;i++)
		{
                while(a[i]>0&&b[i]>0)
                {
                a[i]--;
                b[i]--;
                }
                }
                int f=0,x=0;
		for(int i=0;i<26;i++)
        {
                if(a[i]!=0)
                f=1;
                if(b[i]!=0)
                x=1;
                }	
	
		if(f==1&&x==0)
		cout<<"You win some."<<endl;
		else if(f==0&&x==1)
		cout<<"You lose some."<<endl;
		else
		cout<<"You draw some."<<endl;
	
}

	return 0;
}
