#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
#define LL long long int
#define MOD 1000000007
int main()
{
    int test,ArraySize,i;
    cin>>test;
    while(test--)
    {
        scanf("%d",&ArraySize);
        LL a[ArraySize],Diff=0,element,SumOfElements=0,ans=0,Size=ArraySize-1,Max;
        for(i=0;i<ArraySize;i++)
            scanf("%lld",&a[i]);
        sort(a,a+ArraySize);
        Max=a[ArraySize-1];
        for(i=0;i<ArraySize;i++)
        {
            //scanf("%lld",&a[i]);
            if(i==0) element=a[0];
            else Diff+=(a[i]-element);
           // printf("%lld ",Diff);
        }
        Size--;
        ans=(ans+Diff*Max)%MOD;
        for(i=1;i<ArraySize;i++)
        {
        	SumOfElements+=(a[i]-element);
            ans=(ans+((Diff-SumOfElements-Size*(a[i]-element))*Max)%MOD)%MOD;
            Size--;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
