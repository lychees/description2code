#include <iostream>
using namespace std;

int main()
{
    int temp,inc,i,j,t,max,min,a[100],n;
    cin>>t;
    for(inc=0;inc<t;inc++)
    {
    	cin>>n;
    	
    	for(j=0;j<n;j++)
    	cin>>a[j];
    	 
    	for(i=1;i<n;i++)
        {
    	   temp=a[i];
    	   for(j=i-1;j>=0;j--)
    	   {
    		   if(temp<a[j])
    		{
    			a[j+1]=a[j];
    			a[j]=temp;
    		}
    		   else break;
    	    }
        }
        
        for(i=0;i<n/2;i++)
        a[i]=a[i]+a[n-1-i];
        
        max=min=a[0];
        for(i=0;i<n/2;i++)
        {
        	if(a[i]>max) max=a[i];
        	if(a[i]<min) min=a[i];
        }
        cout<<(max-min)<<endl;
    }
    return 0;
}
