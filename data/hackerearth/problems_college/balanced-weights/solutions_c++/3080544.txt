#include<bits/stdc++.h>
using namespace std;

long int arr[1000001];

long int gcd(long int a, long int b)
{
    while(true)
    {
        if(a>b)
            a %= b;
        else
            b %= a;
        if(a == 0)
            return b;
        if(b == 0)
            return a;
    }
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        for(int i=0; i<n; i++)
            scanf("%ld", &arr[i]);

        long long int temp = arr[0];
        for(int i=1; i<n; i++)
            temp = gcd(temp, arr[i]);
            
        temp *= n;
        cout<<temp<<endl;
    }
    return 0;
}
