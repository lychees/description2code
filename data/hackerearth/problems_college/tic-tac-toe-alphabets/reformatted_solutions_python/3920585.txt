noOfTestCases = int(raw_input())


def InOrderPlacing(string):
    characterArray = []
    for i in range(1, 27):
        characterArray.append(0)
    for c in string:
        characterArray[ord(c) - 65] += 1
    index = 0
    s = ""
    for x in characterArray:
        if (x > 0):
            s += chr(65+index) * x
        index += 1
    print s


for caseNo in range(1, noOfTestCases + 1):
    InOrderPlacing(raw_input())
