#include <iostream>
using namespace std;

void sort( long long int a[],long long int b[],long long int n){
	for(long long int i=0;i<n-1;i++){
		for(long long int j=0;j<n-1;j++){
			if(a[j]<a[j+1]){
				int t=a[j];
				a[j]=a[j+1];
				a[j+1]=t;
				t=b[j];
				b[j]=b[j+1];
				b[j+1]=t;
			}
		}
	}
}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	long long int  i,j,n;
    	cin>>n;
    	long long int r[n],h[n],dp[n];
    	for(i=0;i<n;i++)
    		cin>>r[i]>>h[i];
    	sort(h,r,n);
      /*	
    	for(i=0;i<n;i++)
    		cout<<r[i]<<" "<<h[i]<<endl;
     */	
    	for(i=0;i<n;i++)
    		dp[i]=h[i];
    	
    	for(i=1;i<n;i++){
    		for(j=0;j<i;j++){
    			if( r[j]>r[i] && h[j]>h[i]){
    				
    				dp[i] =(dp[j]+h[i] > dp[i])? dp[j]+h[i] : dp[i];
    				//cout<<h[i]<<" "<<h[j]<<" "<<dp[i]<<endl;
    			}
    			
    		}
    	}
    	j=0;
    	for(i=0;i<n;i++)
    		if(j<dp[i])
    			j=dp[i];
    	
    	cout<<j<<endl;
    	
    }
    return 0;
}
