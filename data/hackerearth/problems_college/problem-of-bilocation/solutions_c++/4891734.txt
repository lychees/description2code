#include <bits/stdc++.h>
using namespace std;

int main()
{
	int T; cin >> T;

	while(T--)
	{
		int a, b, c, d, s;
		cin >> a >> b >> c >> d >> s;
		
		if((c - s) > b)
		cout << b << endl;
		else if((c - s) >= a && (c - s) <= b)
			cout << (c - s) << endl;
		else
			cout << "STAY IN LONDON" << endl;
	}
	return 0;
}
