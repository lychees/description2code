#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>

using namespace std;

int main()
{
    int t ;
    scanf("%d", &t);
    long dp[10001] = {0};
    while (t--) {
    	int n, L, m;
    	scanf("%d",&L);
    	
    	memset(dp, 0, sizeof(dp));
    	for (int j = 1 ; j <= L ; ++j) {
    		
    		scanf("%ld",&dp[j]);
    	}
    	
    	for (int  k = 2; k <=L ; ++k) {
    		for (int  i =0 ; i < k ; ++i) {
    			//printf("dp[%d] = %d dp[%d] = %d dp[%d] = %d \n", k , dp[k],i,dp[i],k-i,dp[k-i]);
    			dp[k] = max (dp[k], dp[i] + dp[k-i]);
    			
    		}
    	}
    	printf("%ld\n",dp[L]);
    }
    
    
    
    return 0;
}
