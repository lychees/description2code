typedef long long ll;
#include <iostream>
#include <stdio.h>
#include <cmath>
#include <algorithm>
#include <vector>
#include <string>
#include <string.h>
#include <map>
#include <queue>
#include <stack>
#include <sstream>
#include <fstream>
#include <stdlib.h>
#include <utility>

using namespace std;
ll mod=1000000007;
ll dp[1007][1007];
int main()
{
    ll t,n,m,i,sum,j;
    scanf("%lld",&t);
    while(t--)
    {
    	memset(dp,0,sizeof(dp));
        scanf("%lld %lld",&n,&m);
        dp[0][0]=1;
       	for(i=1;i<=n;i++)
       	{
       		dp[0][i]=0;
       		dp[i][0]=1;
       	}
       	for(i=1;i<=n;i++)
       	{
       		for(j=1;j<=n;j++)
       		{
       			if(i<j*m)
       			continue;
       			dp[i][j]=(dp[i-1][j]+dp[i][j-1])%mod;
       		}
       	}
       	sum=0;
       	for(i=0;i<=n && n-i>=i*m;i++)
       	{
       		if(n-i>i)
       		sum+=dp[n-i][i];
       		sum%=mod;
       	}
        printf("%lld\n",sum);
    }
	return 0;
}
