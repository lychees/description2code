#include <bits/stdc++.h>
using namespace std;
int min(int x, int y) { return (x < y)? x: y; }
 
int calcAngle(double h, double m)
{
    if (h == 12) 
       h = 0;
    if (m == 60)
       m = 0;
 
   
    int ho= 0.5 * (h*60 + m);
    int mi= 6*m;
 
    int angle = abs(ho- mi);
 
   
    angle = min(360-angle, angle);
 
    return angle;
}
 
int main()
{
    int t;
    double h,m;
    cin>>t;
    while(t--)
    {
    	cin>>h>>m;
    	cout<<calcAngle(h,m)<<endl;
    	
    }
    
    return 0;
}
