#include <iostream>
#define MAX 500
using namespace std;

int main()
{
    int test,num,i,m,j,temp,x;
    int arr[MAX];
    cin>>test;
    while(test--)
    {
        cin>>num;
        arr[0]=1;
        m=1;
        temp=0;
        for(i=2;i<=num;i++)
        {
            for(j=0;j<m;j++)
            {
                x=arr[j]*i+temp;
                arr[j]=x%10;
                temp=x/10;
            }
            if(temp>0)
            {
                while(temp!=0)
                {
                    arr[j++]=temp%10;
                    temp=temp/10;
                    m=m+1;
                }
            }
        }
        for(i=m-1;i>=0;i--)
            cout<<arr[i];
        cout<<endl;
    }
    return 0;
}
