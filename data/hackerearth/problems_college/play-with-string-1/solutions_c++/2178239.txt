#include <iostream>
#include<string>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t=0;
	cin>>t;
	while(t--)
	{
		int flag=0,i=0;
       string a;
       cin>>a;
       int len=a.length();
       int *x=new int[len];
       int *y=new int[len];
       for( i=0;i<len;i++)
       {
       	  x[i]=a[i]-'a';
       	  //cout<<x[i]<<" ";
       	  y[len-1-i]=x[i];
       }
       //cout<<endl;
       for(i=0;i<len/2;i++)
       {
       	//cout<<y[i]<<" ";
       	if(abs(x[i+1]-x[i])==abs(y[i+1]-y[i]))
       	{
       		flag=1;
       	}
       	else
       	{
       		flag=0;
       		break;
       	}
       }
       if(flag)cout<<"Funny"<<endl;
       else cout<<"Not Funny"<<endl;
    }
}
