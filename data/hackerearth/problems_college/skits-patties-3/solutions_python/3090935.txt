def check(n):
	l=[6,9,20]
	if n<6:
		return False
	elif n%6==0 or n%9==0 or n%20==0:
		return True
	else:
		i=0
		j=0
		while(i<len(l) and j<len(l)):
			if(l[i]+l[j]<=n):
				if(l[i]+l[j]==n):
					return True
				else:
					l.append(l[i]+l[j])
			if len(l)-1==j:
				i+=1
				j=i-1
			j+=1
		return False
	
x=input()
for i in range(x):
	y=input()
	print check(y)
