#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	int p;
    	long int N,M,xi,yi,xf,yf,moves=0,cost;
    	cin>>N>>M>>xi>>yi>>xf>>yf>>p;
    	moves=min(abs(xi-xf),N-abs(xi-xf))+min(abs(yi-yf),M-abs(yi-yf));
    	cost=moves*p;
    	cout<<cost<<endl;
    	if(cost>1000)
    	{
    		cout<<"Let go of the group."<<endl;
    	}
    	else
    	{
    		cout<<"You saved the group."<<endl;
    	}
    	
    }
}
