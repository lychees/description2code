    #include <stdio.h>
     
    int arr[1000001];
    char YES[] = {'Y', 'e', 's', '\0'};
    char NO[] = {'N', 'o', '\0'};
    int main()
    {
    	arr[1] = 1;
    	for(int i=2; i*i<=1000001;i++)
    		for(int j=i*i;j<1000001;j*=i)
    			arr[j] = 1;
    	int T, n;
    	scanf("%d", &T);
    	while(T-->0) {
    		scanf("%d", &n);
    		char *ans = NO;
    		for(int i=1;i<n;i++)
    			if(arr[i]==1 && arr[n-i]==1) {
    				ans = YES;
    				break;
    			}
    		printf("%s\n", ans);
    	}
     
    	return 0;
    }
