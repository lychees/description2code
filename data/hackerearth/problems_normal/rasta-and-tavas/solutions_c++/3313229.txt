#define MOD 1000000007
bool isprime[1111111];
int a[1111111];
int main()
{
	std::ios::sync_with_stdio(false);cin.tie(0);
	int i;
	long long j;
	int n,k;
	cin>>n>>k;
	for(i=2;i<=n;i++)
		isprime[i]=true;
	for(i=2;i<=n;i++)
		if(isprime[i])
		{
			a[i]+=i;
			for(j=i+i;j<=n;j+=i)
			{
				a[j]+=i;
				isprime[j]=false;
			}
		}
	int re=0;
	for(i=1;i<=n;i++)
		if(a[i]==k)
			re++;
	cout<<re<<endl;
//	system("pause");
	return 0;
}
