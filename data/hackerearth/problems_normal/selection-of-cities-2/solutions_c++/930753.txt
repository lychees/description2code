#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define mod 1000000007
ll power(ll a,ll b){
  ll res = 1;
  while(b){
  	if(b&1)
  	  res= (res*a)%mod;
  	a = (a*a)%mod;
  	b >>= 1;
  }
  return res;
}
int main(){
	int t;
	ll n;
	cin >> t;
	while(t--){
		cin >> n;
		ll ans = power(2,n);
		ans = ((ans-1)+mod)%mod;
		cout << ans << endl;
	}
	return 0;
}
