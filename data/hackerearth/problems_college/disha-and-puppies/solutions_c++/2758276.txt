int sb(int sum,int zz)
{
	for (int i = 0; i <= n; i++)
        dp[0][i]=1;

	for (int i = 1; i <= sum; i++)
        dp[i][0]=0;

	for (int i = 1; i <= sum; i++){
        for (int j = 1; j <= n; j++){
            dp[i][j] = dp[i][j-1];
            if (i >= a[j-1])
                dp[i][j] = dp[i][j] || dp[i - a[j-1]][j-1];
        }
	}
	if(dp[zz][n]==1)
        return zz;
    int k1=zz,k2=zz;
    while(1){
        if(k1<sum)
            k1++;
        if(k2>0)
            k2--;
        if(dp[k1][n]==1)
            return k1;
        if(dp[k2][n]==1)
            return k2;
    }
}
