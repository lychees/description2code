struct node
{
    ll x;
    ll y;
    ll z;
} arr[1000005];
ll Arr[1000005],siz[1000005];
ll root (ll i)
{
    while(Arr[ i ] != i)
    {
        Arr[ i ] = Arr[ Arr[ i ] ] ;
i = Arr[ i ];
    }
return i;
}
void unio(ll A,ll B)
{
    ll root_A = root(A);
    ll root_B = root(B);
    if(siz[root_A] < siz[root_B ])
    {
Arr[ root_A ] = Arr[root_B];
siz[root_B] += siz[root_A];
}
    else
    {
Arr[ root_B ] = Arr[root_A];
siz[root_A] += siz[root_B];
}
}
bool cmp(node sk,node vk)
{
return sk.z<vk.z;
}
