#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 100000000
 
int main()
{
     std::ios::sync_with_stdio(false);
     int t; cin>>t;
     while(t--)
     {
         string str; cin>>str;
         for(int i=0; i<str.length(); i++)
         {
             if(str[i]=='x')
             {
                 for(int j=i-1; j>=0; j-- )
                 {
                     if(str[j]=='*' || str[j]=='x' || str[j]=='a') break;
                     str[j]='a';
                 }
                 for(int j=i+1; j<str.length(); j++)
                 {
                     if(str[j]=='*' || str[j]=='x' || str[j]=='a') break;
                     str[j]='a';
                 }    
             }
         }
         
         for(int i=0; i<str.length(); i++)
             if(str[i]!='a') cout<<str[i];
         cout<<"\n";    
     }
     return 0;
}
