void sieve(){
	ll i,j,k=0;
	a[0]=1;a[1]=1;
	p[1]=0;

		a[j]=1;
		p[j]=2;
	}
	for(i=3;i<sqrt(1000005);i+=2){
		if(a[i]==0){
			p[i]=0;
			for(j=i*i;j<1000005;j+=2*i){
				a[j]=1;
				if(p[j]==0)
				     p[j]=i;
			}
		}
	}
	for(i=3,j=1;i<(1000005);i+=2){
		if(a[i]==0){
			p[i]=0;
		}
    }
    
	for(i=2;i<1000005;i++){
		p[i]+=p[i-1];
		p[i]=p[i]%MOD;
	}
}
