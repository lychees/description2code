#include<bits/stdc++.h>
using namespace std;
#define ll long long int
pair<ll,ll> p[2000100];
int main()
{
	ll t;
	ll n,x;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld %lld",&n,&x);
		ll i,j;	
		for(i=0; i<n; i++)
		{
			scanf("%lld %lld",&(p[i].first),&(p[i].second));
		}
		sort(p,p+n);
		ll flag=0;
		for(i=0; i<n; i++)
		{
			ll last=p[i].second;
			
			for(j=i+1; j<n; j++)
			{
				if(p[j].first>=p[i].first && p[j].second<=p[i].first+x)
				{
					last=max(last,p[j].second);
				}
				if(last==p[i].first+x)
				{
					flag=1;
					break;
				}
			}
		}
		if(flag==0)
		{
			printf("No\n");
		}
		else
		{
			printf("Yes\n");
		}
		
	}
    	return 0;
	
	
}
