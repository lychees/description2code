#include<bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	while(n--){	
	int t,j,i;
	cin>>t;
	vector <int> A(t),B(t);
	for(i=0;i<t;i++){
		cin>>A[i];
	}
	
	for(i=0;i<t;i++)
	cin>>B[i];
	int count1 = 0,prev_i=0,k;
	i=0;
	for(j=0;j<t;j++){
		if(A[j]!=B[j]){
			k = find(A.begin(),A.end(),B[j]) - A.begin();
		
		while(k>j){
			swap(A[k],A[k-1]);
			count1++;
			k--;
		}
		
		}
		
	}
	cout<<count1<<endl;
}
	return 0;
}
