def minodd(lst, n):
    p = 101
    for i in range(n):
        if lst[i] % 2 != 0:
            if lst[i] < p:
                p = lst[i]
    if p == 101:
        p = 0
    return p


n = input()
lst = map(int, raw_input().split())
s = sum(lst)
if s % 2 != 0:
    print s
else:
    p = minodd(lst, n)
    if p == 0:
        print ":("
    else:
        print s-p
