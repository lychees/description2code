#include<bits/stdc++.h>
using namespace std;


bool F(int n){
	stringstream ss;
	ss << n;
	string s = ss.str(),s2;
	s2=s;
	reverse(s2.begin(),s2.end());
	if(s==s2) return true;
	else return false;
	
}


int main(){
	int n,a,b;
	cin >> n;
	for(int i=0;i<n;i++){
		int cnt=0;
		cin >> a >> b;
		for(int j=a;j<=b;j++){
			if(F(j))cnt++;
		}
		cout << cnt << endl;
	}
	return 0;
}
