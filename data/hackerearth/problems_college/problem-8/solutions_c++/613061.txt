#include <iostream>
using namespace std;

int main()
{
	int n, cnt=0, i;
	int arr[3];
	arr[0] = arr[1] = 1;
    cin>> n;
    for(i=2;cnt < n-2;cnt++,i++)
    {
    	arr[i%3] = arr[(i-1)%3] + arr[(i-2)%3];
    	cout<< arr[(i-2)%3] << endl;
    }
    cout<< arr[(i-2)%3] << endl << arr[(i-1)%3];
    return 0;
}
