#include <stdio.h>
#include<iostream>
#include<iomanip>
using namespace std;

int main()
{
   int n,i;
   long int tot_fuel;
   double temp=0.0,t;
   scanf("%d",&n);
   int speed[n],fuel[n];
   for(i=0;i<n;i++)
       scanf("%d", &speed[i]);
   for(i=0;i<n;i++)
       scanf("%d", &fuel[i]);
   scanf("%ld", &tot_fuel);
   
   for(i=0;i<n;i++)
   {
   	t=(double)tot_fuel/fuel[i]*speed[i];
   	if(temp<t)
   	temp=t;
   }
   double p=temp*1000;
   int s=temp*1000;
   if((p-s)>0.5)
   p=p-0.5;
   p=p/1000;
   std::cout << std::fixed << std::setprecision(3) <<p;
   return 0;
}
