#include <bits/stdc++.h>

using namespace std;
const int N = 1234567;
long long sum[N+5];
long long sum2[N+5];
long long sum3[N+5];
const long long mod = 1e9 + 7;
int main(int argc, char const* argv[])
{
  long long n, k;
  cin>>n>>k;
  for(long long i = 1; i <= n; i++){
    long long x;
    cin>>x;
    sum[i] =  (sum[i-1] +(((i*i)%mod)*x)%mod)%mod;
    sum2[i] = (sum2[i-1] + (i*x)%mod)%mod;
    sum3[i] = (sum3[i-1] + x)%mod;
  }
  for(long long i = 1; i <= n - k + 1; i++){
    long long ans = 0;
     ans = (sum[k+i-1] - sum[i-1] + mod)%mod;
     ans =  (ans - (2ll*(i-1ll)*(sum2[k+i-1] - sum2[i-1] + mod))%mod + mod)%mod;
     ans +=  (i-1ll)*(i-1ll)%mod*(sum3[k+i -1] - sum3[i-1] + mod)%mod;
     ans %= mod;
     cout<<ans<<" ";
  }
  cout<<endl;
  return 0;
}
