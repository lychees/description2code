#include <bits/stdc++.h>
using namespace std;

int j = 0;
long long A[10000];

void divisor(long long x)
{
	long long i, tmp;
	j = 0;
	
	for(i=1; i<=sqrt(x); i++)
	{
	
		if(x%i==0)
			A[j++] = i;
	}
	
	long long k = j-1;
	
	while(1)
	{
		
		if(x/A[k]==A[k])
		{
			k--;
			
			if(k==-1)
				break;
			
			continue;
		}
		
		else
			A[j++]=x/A[k];
			
		k--;
		
		if(k==-1)
			break;
	}
	
	j--;
	
}

int same(long long a, long long b)
{
	int a1[12], b1[12], k, l;
	
	k = 0, l = 0;

	while(a)
	{
		a1[k++] = a%10;
		a/=10;
	}
	
	while(b)
	{
		b1[l++] = b%10;
		b/=10;
	}
	
	for(int i=0; i<k; i++)
		for(int m=0; m<l; m++)
		{
			if(a1[i]==b1[m])
				return 1;
		}
		
	return 0;
}

int main()
{
    int T;
    long long x;
    
    scanf("%d", &T);
    
    while(T--)
    {
    	scanf("%lld", &x);
    	
    	divisor(x);
    	
    	int count = 0;
    	
    	for(int i=0; i<=j; i++)
    		if(same(x, A[i]))
    			count++;
    			
    	printf("%d\n",count);
    }
    
    return 0;
}
