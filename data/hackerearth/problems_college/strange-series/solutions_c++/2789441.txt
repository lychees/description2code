#include <iostream>
using namespace std;
long long int fun(int n)
{
long	int i,j,table[n+1][n+1];
	long long int r=0;
	for(i=0;i<=n;i++)
	{
		for(j=0;j<=i;j++)
		{
			if(i==0)
			table[i][j]=0;
			else
			{
			if(j==0 || j==i)
			table[i][j]=1;
			else
			{
			table[i][j]=table[i-1][j]+table[i-1][j-1];	
			}
			}
		}
	}
	for(i=1;i<=n;i++)
	{
		r=r+table[n][i];
	}
	return r;
}



int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	cout<<fun(n)<<endl;
    }
    return 0;
}
