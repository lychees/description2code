#include <iostream>
using namespace std;

int main()
{
	int n, i, x, c, j;
    cin>>n;
    int a[4]={0};
    for(i=0;i<n;i++)
    {
    	cin>>x;
    	a[x-1]++;
    }
    c=a[3];
    i=0;j=2;
    while(a[i]!=0&&a[j]!=0)
    {
    	c++;a[i]--;a[j]--;
    }
    if(a[i]==0&&a[j]!=0)
    {
    	c+=a[2];
    	c+=a[1]/2;
    	if(a[1]%2!=0 && a[1]!=0)
    		c++;
    }
    else if(a[j]==0&&a[i]!=0)
    {
    	while(a[1]!=0&&a[0]>=2)
    	{
    		c++;a[1]--;a[0]-=2;
    	}
    	if(a[1]!=0&&a[0]==0)
    	{
    		c+=a[1]/2;
	    	if(a[1]%2!=0 && a[1]!=0)
	    		c++;
    	}
    	else if(a[0]!=0&&a[1]==0)
    	{
    		c+=a[0]/4;
    		if(a[0]%4!=0 && a[0]!=0)
    			c++;
    	}
    	else
   			c+=1+(a[1]/2);
    }
    else
    {
    	c+=a[1]/2;
    	if(a[1]%2!=0 && a[1]!=0)
    		c++;
    }
    cout<<c<<endl;
    return 0;
}
