#include <iostream>
#include <string>
#define ll long long
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    
    ll n,sum=0,a,b,l,val;
    cin>>n;
    ll cost[n],temp, check[n];
    for(ll j=0;j<n;++j){
    	cin>>cost[j];
    	check[j]=cost[j];
    	sum=0;
    	temp=cost[j];
    	while(temp>0){
    		sum+=temp%10;
    		temp/=10;
    	}
    	cost[j]*=sum;
    	
    }
    cin>>a>>b;
    l=a*b;
    sum=0;
    for(ll j=0;j<n;++j){
    //	sum+=l*cost[j];
    val=l*cost[j];
    if(check[j]>=1&&check[j]<=10) val=val-val*0.25;
    if(check[j]>=11&&check[j]<=20) val=val-val*0.30;
    if(check[j]>=21&&check[j]<=50) val=val-val*0.20;
    sum+=val;
    
    }
    cout<<sum;
    return 0;
}
