

#include <iostream>
#include <vector>
#include <algorithm>



long convert(std::string &  s)
{
	long ans=0;
	if(s.size()==1)
		return s[0]-'0';
	int i=0;
	for( ; i<s.size()/2 ; ++i)
	{
		ans+=(s[i]-'0')+(s[s.size()-1-i]-'0');
	}
	if(s.size()%2)
		ans+=s[i]-'0';
	return ans;
}

int main()
{
	std::ios_base::sync_with_stdio(false);

	int t;
	std::cin>>t;
	while(t--)
	{
		std::string n;
		std::cin>>n;
		long num=convert(n);

		if(num==1){
			std::cout<<"NO\n";
				continue;
		}
		if(num==2 || num==3)
		{
			std::cout<<"YES\n";
			continue;
		}

		bool yes=true;
		for(int i=2 ; i<=num/2 ; ++i)
		{
			if(!(num%i)){
				yes=false;
				break;
			}
		}
		if(yes)
			std::cout<<"YES\n";
		else
			std::cout<<"NO\n";

	}
}
