for _ in range(input()):
    n = input()
    a = raw_input()
    b = 'B'
    g = 'G'
    c = d = 0
    for i in range(n*2):
        # print i
        if (i % 2 == 0 and a[i] == b):
            # print i
            c += 1
        if (i % 2 and a[i] == g):
            c += 1
    for i in range(n*2):
        if (i % 2 == 0 and a[i] == g):
            d += 1
        if (i % 2 and a[i] == b):
            d += 1
    print min(d//2, c//2)
