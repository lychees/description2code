#include <iostream>
using namespace std;

int main()
{
     long long tc,n,p,q,b;
     float c;
     cin>>tc;
     while(tc--)
     {
     	cin>>n>>b;
     	p=n%b;
     	c=b*.5;
     	if(p<c)
     	{
     		cout<<(n/b)*b<<"\n";
     	}
     	if(p>=c)
     	{ q=((n/b)*b)+b;
     		cout<<q<<"\n";
     	}
     }
    return 0;
}
