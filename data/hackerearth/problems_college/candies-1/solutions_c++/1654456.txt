#include<iostream>
using namespace std;
int main()
{
   long long int t,n,i,a,sum;
   cin>>t;
   while(t--)
   { sum=0;
   	cin>>n;
   	for(i=0;i<n;i++)
   	{
   		cin>>a;
   		sum+=a;
   	}
   	if(sum%n==0)
   	cout<<"YES\n";
   	else
   	cout<<"NO\n";
   	cout<<sum/n<<" "<<sum%n<<endl;
   }
}
