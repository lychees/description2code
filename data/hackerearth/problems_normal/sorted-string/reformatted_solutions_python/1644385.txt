from operator import itemgetter, attrgetter, methodcaller
tc = int(raw_input())
while tc > 0:
    string = raw_input()
    newstr = ''.join(sorted(string))
    op = []
    for i in range(len(newstr)):
        a = newstr.count(newstr[i])
        ha = [newstr[i], a]
        if ha not in op:
            op.append(ha)
        i = i+a-1
    op = sorted(op, key=itemgetter(1))
    final = ''
    for i in op:
        final = final+i[0]*i[1]
    print final
    tc -= 1
