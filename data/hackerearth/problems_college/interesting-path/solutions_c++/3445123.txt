#include <iostream>
#include<algorithm>
using namespace std;
long long int minJumpsRequired(int arr[], int n)
{
  long long   int mjumps[10001];
    mjumps[0]=0;
 
    if (n == 0 || arr[0] == 0)
        return 100005;
 
  for (long long int i = 1; i < n; i++)
    {
        mjumps[i] = 100005;
        for (long long int j = 0; j < i; j++)
        {
            if (i <= j + arr[j] && mjumps[j] !=100005)
            {
                 mjumps[i] = min(mjumps[i], mjumps[j] + 1);
                 break;
            }
        }
    }
    return mjumps[n-1];
}
 
int main()
{
    int t=0;
    scanf("%d",&t);
    while(t>0)
    {
    	
    	int n=0;
    	scanf("%d",&n);
    	int arr[10001];
    	for(int i=0;i<n;i++)
    	scanf("%d",&arr[i]);
    	printf("%lld\n",minJumpsRequired(arr,n));
    	
    	
    	t--;
    }
    return 0;
}
