#include <bits/stdc++.h>
using namespace std;

int main()
{
	string s;
	map<int,int> m;
	getline(cin,s);
	for(int i=0;i<s.length();i++)
	{
		m[s[i]]++;
	}
	map<int,int>::iterator it;
	int max=0;
	char c;
	for(it=m.begin();it!=m.end();it++)
	{
		//cout<<it->first<<" "<<it->second<<endl;
		if(it->second > max)
		{
			max = it->second;
			c = it->first;
		}
	}
	cout<<c<<" "<<max<<endl;
    return 0;
}
