#include <iostream>
using namespace std;

int fact(int n)
{
	if(n==0) return 1;
	if(n==1) return 1;
   
   return n*fact(n-1);
}
int ans(int n,int m)
{
	return (fact(n+(m-1))/((fact (n))*(fact(m-1))));
}
int main()
{
   int k,n,m,h=0,x;
   cin>>k;
   while(h<k)
   {
   	cin>>n>>m;
   	x=ans(n,m);
   	cout<<x<<endl;
   	h++;
   	}
   
}
