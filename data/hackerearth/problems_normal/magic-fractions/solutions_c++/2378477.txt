#include <iostream>
using namespace std;

#define SIZE 1000

int main()
{
    int n;
    cin >> n;
    if(n==1) {
		cout << 0 << endl;
		return 0;
	}
    bool isprime[SIZE+1];
    for(int i=1; i<=SIZE; i++) isprime[i] = true;
    int p = 2;
    while(p<=SIZE) {
    	int p_ = 2*p;
    	while(p_<=n) {
			isprime[p_] = false;
			p_ += p;
		}
		p++;
    	// find next prime
    	while(p<=SIZE && !isprime[p]) p++;
    }
    
    long long curr = 1;		// for i = 2
    long long sum = 1;
    for(int i=3; i<=n; i++) {
    	if(isprime[i]) curr *= 2;
    	sum += curr;
    }
    cout << sum << endl;
    return 0;
}
