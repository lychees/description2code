#include <bits/stdc++.h>
using namespace std;
int solve(vector<int> V,int n) {
	if(n<=2)
		return 0;
	int res = 0;
	for(int i=1;i<n-1;i++) {
		vector<int> R;
		for(int j=0;j<n;j++) {
			if(i == j)
				continue;
			R.push_back(V[j]);
		}
		res = max(res,V[i-1]*V[i+1]+solve(R,n-1));
	}
	return res;
}
int main()
{
    int n;
    cin>>n;
    vector<int> V(n);
    for(int i=0;i<n;i++) {
    	cin>>V[i];
    }
    cout<<solve(V,n)<<endl;
    return 0;
}
