#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   int n,h,m,min,i;
   cin>>t;
   for(;t--;)
   {
   	min=100000;
   	cin>>n>>h;
   	for(i=0;i<n;i++)
   	{
   		cin>>m;
   		if(abs(h-m)<min)
   		min=abs(h-m);
   	}
   	cout<<(min)<<endl;
   }
    return 0;
}
