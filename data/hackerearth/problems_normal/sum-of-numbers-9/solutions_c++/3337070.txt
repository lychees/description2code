#include <iostream>
using namespace std;

int main()
{
    int t,n,flag;
    long int a[1000000],i,j,s,sum;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	cin>>s;
    	long int p = 1<<n;
    	flag=0;
    	for(i=0;i<p;i++)
    	{
    		sum = 0;
    		for(long int k=0;k<n;k++)
    		if(i&1<<k) {
    				sum+=a[k]; 
    				if(sum==s){cout<<"YES"<<endl; flag=1;
    				break;}
    			}
    		if(flag==1) break;
    	}
    	if(i==p) cout<<"NO"<<endl;
    }
}
