#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,cnt=1;
	cin>>n;
	string s[n+1];
	for(int i=0 ; i<n ; ++i)
		cin>>s[i];
	sort(s,s+n);
	for(int i=0 ; i<n-1 ; ++i)
	{
		if(s[i]==s[i+1])
		{
			cnt++;
		}	
		else
		{
			cout<<s[i]<<" "<<cnt<<endl;
			cnt=1;
		}
	}
	cout<<s[n-1]<<" "<<cnt<<endl;
}
