#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
char S[1000005];

int main()
{
    
    int K,N;
    ll ans = 0LL;
    map<int,int> M;
    map<int,int>::iterator it;
    scanf("%d",&N);
    scanf("%s",S);
    int PrefixSum = 0;
    M[0] = 1;
    int i;
    for(i=0;i<=N-1;i++)
    {
        PrefixSum = PrefixSum + (S[i] == 'S' ? 2 : -1);
        it = M.find(PrefixSum);
        if(it != M.end())
            ans += (it->second);
        M[PrefixSum]++;
    }
    printf("%lld",ans);
    return 0;
}
