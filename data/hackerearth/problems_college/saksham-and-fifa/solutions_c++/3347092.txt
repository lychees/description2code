#include <bits/stdc++.h>
#define MAX 1000000 

using namespace std;

struct node
{
	int time;
	int position;
};

int main()
{
	int i,n;
	node values[MAX+5];
	cin>>n;

	for(i=0;i<n;i++)
	{
		cin>>values[i].time>>values[i].position;
	}

	int max_time=INT_MIN;
	int pos;
	for(i=0;i<n;i++)
	{
		if(values[i].time > max_time)
		{
			max_time = values[i].time;
			pos = values[i].position;
		}
	}

	if(pos < 0)
	{
		cout<<"Right Corner\n";
	}
	else if(pos > 0)
	{
		cout<<"Left Corner\n";
	}
	else
	{
		cout<<"Either Corner\n";
	}

}
