lines=[e.replace('\n','') for e in sys.stdin.readlines()]

def function(s,n):
	if len(s)>0:
		p=[int(e) for e in s.split(' ')]
	else:
		print ''
		return 0

	stack=[0]
	prev=0
	st=''
	for e in p:
		while stack[-1]<e:
			st+='C'
			prev+=1
			stack.append(prev)
		while stack[-1]>=e:
			i=stack.pop()
			st+='K'
		if i!=e:
