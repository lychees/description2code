#include <iostream>
#include <string>
using namespace std;

int main()
{
    int T;
    cin >> T;
    while(T--)
    {
        string str;
        cin >> str;
        int arr[26] = {0};
        int length = str.length();
        int index;
        int large = 0;
        for(int x = 0; x < length; x++)
        {
            index = (int(str[x])-97);
            arr[index]++;
            if(arr[index] > large)
                large = arr[index];
        }
        cout << large << endl;
    }
}
