template <typename T, typename U> 
inline void chkmax(T& x, U y) {
	if (x < y) x = y;
}

template <typename T, typename U>
inline void chkmin(T& x, U y) {
	if (y < x) x = y;
}

const int MAXN = 105;
int n;
char s[MAXN];
int dp[MAXN][MAXN];

int main() {
	
	int ncase;
	for (scanf("%d", &ncase); ncase--; ) {
		scanf("%s", s);
		int n = strlen(s);
		int ret = 1;
		for (int i = 0; i < n; i++) {
			reverse(s + i, s + n);
			for (int i = 0; i < n; i++) dp[i][i] = 1;
			for (int L = 2; L <= n; L++) {
				for (int i = 0; i <= n - L; i++) {
					int j = i + L - 1;
					dp[i][j] = max(dp[i + 1][j], dp[i][j - 1]);
					if (s[i] == s[j]) dp[i][j] = dp[i + 1][j - 1] + 2;
				}
			}
			chkmax(ret, dp[0][n - 1]);
			reverse(s + i, s + n);
		}
		printf("%d\n", ret);
		
	}

	return 0;
}
