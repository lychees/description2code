n, k = map(int, raw_input().strip().split())

array = [0 for _ in range(k)]
array.append(1)

powers = [2 ** i for i in range(n+1)]

for i in range(k+1, n+1):
	array.append(powers[i - k - 1] - array[i - k - 1] + 2 * array[i-1])

num = array[-1]
denom = powers[-1]

while num % 2 == 0:
	num = int(num / 2)
	denom = int(denom / 2)

print "%d/%d" % (num, denom)
