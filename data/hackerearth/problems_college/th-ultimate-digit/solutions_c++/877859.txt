#include <iostream>
#include <vector>
#define ll long long
#define pb push_back
using namespace std;

vector<vector<int> >  cyc(10);
void PR() {
	for(int i=0;i<10;i++) {
		//cout<<endl<<i<<": ";
		int num=i;
		cyc[i].pb(num);
		//cout<<num<<" ";
		while(1) {
			num*=i;
			num%=10;
			if(cyc[i][0]==num) break;
			cyc[i].pb(num);
			//cout<<num<<" ";
		}
	}
}

int main()
{
	PR();
    int T;
    ll a, b;
    string A;
    cin>>T;
    while(T--) {
    	cin>>A>>b;
    	//while(a>9) {
    		a=A[A.length()-1]-'0';
    	//}
    	int ind=(b-1)%cyc[a].size();
    	cout<<(b>0?cyc[a][ind]:1)<<endl;
    }
    return 0;
}
