#include <iostream>
using namespace std;
#include <algorithm>
 
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long int n, i, j, c=0;
    	scanf("%lld",&n);
    	long long int a[n];
    	for(i=0; i<n; ++i)
    	{
    		scanf("%lld",&a[i]);
    	}
    	sort(a,a+n); c=n;
    	for(i=0; i<n; ++i)
    	{
    		for(j=i+1; j<n; ++j)
    		{
    			if(a[i]==a[j])
    			c++;
    			else
    			break;
    		}
    	}
    	printf("%lld\n",c);
    }
    return 0;
}
