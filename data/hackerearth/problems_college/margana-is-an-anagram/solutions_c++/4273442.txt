#include <bits/stdc++.h>
using namespace std;

int main()
{
	int T;
	cin>>T;
	while(T--){
		string str1,str2;
    	cin>>str1>>str2;
    	if(str1.size()!=str2.size()){
    		cout<<"No";
    	}
    	else{
    		int count[26];
	    	memset(count,0,sizeof(count));
	    	for(int i=0;i<str1.size();i++){
	    		count[(str1[i]-'a')]++;
	    	}
	    	for(int i=0;i<str2.size();i++){
	    		count[(str2[i]-'a')]--;
	    	}
	    	int flag=0;
	    	for(int i=0;i<26;i++){
	    		if(count[i]!=0){
	    			flag=1;
	    			break;
	    		}
	    	}
	    	if(flag==1){
	    		cout<<"No";
	    	}
	    	else{
	    		cout<<"Yes";
	    	}
	    }
	    cout<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
