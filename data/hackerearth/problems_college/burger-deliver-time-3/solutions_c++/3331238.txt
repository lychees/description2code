#include <bits/stdc++.h>
#define mp make_pair
#define pb push_back
using namespace std;

int main()
{
    int n;
    cin>>n;
    vector<pair<int,int> > vec;
    int t,d,i;
    for(i=1;i<=n;i++) {
    	cin>>t>>d;
    	vec.pb(mp(t+d,i));
    }
    sort(vec.begin(),vec.end());
    for(i=0;i<n;i++) {
    	cout<<vec[i].second<<" ";
    }
    return 0;
}
