#include <bits/stdc++.h>

using namespace std;

#define s(x) scanf("%d", &x)

int main()
{
    int t, mx, i, l, r;
    cin >> t;
    string s;
    while (t--) {
        cin >> s;
        mx = INT_MAX;
        for (i = s.size()/2; i < s.size(); i++) {
            l = i-1;
            r = i;
            while (r < s.size() && s[l] == s[r]) {
                l--;
                r++;
            }
            if (r == s.size()) {
                cout << 2*i << endl;
                goto l1;
            }
            l = i-1;
            r = i+1;
            while (r < s.size() && s[l] == s[r]) {
                l--;
                r++;
            }
            if (r == s.size()) {
                cout << 2*i+1 << endl;
                goto l1;
            }
        }
        l1: ;
    }
    return 0;
}
