long long power(int a, int b)
{
    if( b == 0 )
        return a;
    else
        return a*power(a,b-1);
}

int arr[5009];
int p[400009];
int n[400009];

int main()
{
   //freopen("input.txt","r",stdin);
   //freopen("output.txt","w",stdout);
   int no;
   cin>>no;
   int i,j;
   for(i=0;i<no;i++) cin>>arr[i];
   mem(p); mem(n);

   int cnt=0;

   if(arr[0]*2>=0) p[arr[0]*2]=1; else n[-arr[0]*2]=1;

   for(i=1;i<no;i++)
   {
       for(j=0;j<i;j++)
       {
           if(arr[i]-arr[j]>=0 && p[arr[i]-arr[j]]) {cnt++; break;}
           else if(arr[i]-arr[j]<0 && n[-arr[i]+arr[j]]) {cnt++; break;}


       }
       for(j=0;j<=i;j++)
       {
          if(arr[i]+arr[j]>=0) p[arr[i]+arr[j]]=1;
          else n[-arr[i]-arr[j]]=1;
       }
   }
   cout<<cnt;


 return 0;
}
