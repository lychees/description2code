for __ in xrange(input()):
    n = input()
    rh = []
    for i in xrange(n):
        r, h = map(int, raw_input().strip().split(' '))
        rh += [[r, h]]
    rh.sort()
    dp = [0 for i in xrange(n)]
    dp[0] = rh[0][1]
    for i in xrange(1, n):
        dp[i] = rh[i][1]
        for j in xrange(i):
            if rh[i][1] > rh[j][1] and rh[i][0] > rh[j][0]:
                dp[i] = max(dp[i], dp[j] + rh[i][1])
    print max(dp)
