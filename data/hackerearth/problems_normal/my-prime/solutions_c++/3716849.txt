#include<iostream>
using namespace std;

int isprime(int,int[],int);
int main()
{
	int n;
	cin>>n;
	int a[n];
	int i;
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0;i<n;i++)
	{
		int p=isprime(a[i],a,n);
		if(p==1)
			cout<<a[i]<<" ";
	}
}
int isprime(int num,int a[],int n)
{
	int f=1;
	for(int i=0;i<n;i++)
	{
		if(a[i]==num)
			continue;
		if(num%a[i]==0)
		{
			f=0;
			break;
		}
	}
	return f;
}
