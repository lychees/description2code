#include <bits/stdc++.h>
using namespace std;
int main()
{
	string inp;
    int num;
    long long int i,sz,cnt;
    scanf("%d",&num);
    while(num--)
    {
    	cin>>inp;
    	sz=inp.size();
    	for(cnt=0,i=0;i<sz;)
    	{
    		if(inp[i]==inp[i+1] && i+1<sz)
    		{
    		while(inp[i]==inp[i+1])
    		{
    			i++;
    			cnt++;
    		}
    		}
    		else
    		i++;
    	}
    	printf("%lld\n",cnt);
    }
    return 0;
}
