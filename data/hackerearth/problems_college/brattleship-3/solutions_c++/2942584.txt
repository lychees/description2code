#include <iostream>
#include<cmath>
using namespace std;
 
int main()
{
    int T,m=1;
    cin>>T;
    while(T--){
    	int R,C,W;
    	cin>>R>>C>>W;
    
    	int d = (C-W)+1,f=0;
    	
    	if(C==W)
    		f = C;
    	else if(d==C)
    		f = C * R ;
    	else
    		f = d ;
    	
    	
    	cout<<"Case #"<<m<<": "<<f<<endl;
    	m++;
    }
    return 0;
}
