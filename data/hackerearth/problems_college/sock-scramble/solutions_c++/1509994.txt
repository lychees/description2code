#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    int n;
    cin>>n;
    vector<int> a;
    for(int i=0;i<n;i++)
    {
        int g;
        cin>>g;
        a.push_back(g);
    }
   int x = a[0];
   for (int i = 1; i<a.size() ; ++i)
   {
      x = x ^ a[i];
   }
   cout << x<<endl;
    }
   return 0;
}
