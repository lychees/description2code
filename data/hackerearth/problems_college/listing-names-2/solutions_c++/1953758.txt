#include <bits/stdc++.h>
using namespace std;

int main()
{
	map<string, int>m;
    int t;
    cin >> t;
    while(t--) {
    	string str;
    	cin >> str;
    	m[str]++;
    }
    map<string, int> :: iterator it;
    for(it = m.begin(); it != m.end(); ++it) {
    	cout << it->first << " " << it->second << endl;
    }
    return 0;
}
