for i in range(input()):
    c = int(raw_input())
    p = {}
    fin = 0
    a = map(int, raw_input().split(' '))
    for i in xrange(len(a)):
        t = a[i]
        if (str(t) in p):
            fin = fin+p[str(t)]
            p[str(t)] = p[str(t)]+1
        else:
            p[str(t)] = 1
    fin = fin+c
    print fin
