ull p[25],res[10],cases,n,m,c;



int main()
{
	std::ios::sync_with_stdio(0);
	cin.tie(0);
	p[0]=1;
	for(int i=1;i<=19;i++)p[i]=10*p[i-1];
	res[0]=1;
	res[1]=1;
	res[2]=2;
	res[3]=720;
	res[4]=1;
	c=0;
	for(int i=1;i<=24;i++)
	{
		res[4]=(res[4]*i)%p[16];
		while(res[4]%10==0)
		{
			c++;
			res[4]/=10;
		}
	}
	res[4]%=p[19-c];
	res[4]*=p[c];
	cin>>cases;
	for(int kase=1;kase<=cases;kase++)
	{
		cin>>n>>m;
		if(n>4)cout<<"0\n";
		else cout<<res[n]%p[m]<<"\n";
	}
	return 0;
}
