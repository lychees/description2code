#include<bits/stdc++.h>
#define pii pair<long long,long long>
#define fi first
#define se second
#define mp make_pair
#define vpi vector< pii >
#define pb push_back
#define ll long long int
#define mod 1000000007
#define gcd(a,b) __gcd(a,b)
#define sf(n) scanf("%lld",&(n))
#define pf(n) printf("%lld\n",(n))
#define setbits(n) __builtin_popcount((n))
#define pq priority_queue
using namespace std;
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
        ll n;
        cin>>n;
        pii arr[n];ll flag=0,index=0;
        int ans_x=-1,ans_y=-1,ans_range=INT_MAX,sum=0;
        arr[0]=mp(0,0);
        for(ll i=1;i<=n;i++)
            {
                int a;
                cin>>a;
                sum=(sum+a)%n;
                arr[i]=mp(sum,i);
            }
        sort(arr,arr+n+1);
        for(int i=0;i<=n;i++)
        {
            if(i+1<=n && arr[i].fi==arr[i+1].fi)
            {
                int range=abs(arr[i].se-arr[i+1].se);
                int x=min(arr[i].se,arr[i+1].se);
                int y=max(arr[i].se,arr[i+1].se);
                x++;
                if(range<ans_range||(range==ans_range && x<ans_x))
                {
                    ans_range=range;
                    ans_x=x;
                    ans_y=y;
                }
            }
        }
        cout<<ans_x<<" "<<ans_y<<"\n";
}
}
