#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t,i;
   cin>>t;
   while(t--)
   {
   	long long int a,b,n,sum=0,r=0,s=0;
   	cin>>a>>b>>n;
   	long long int cal[n];
   	for(i=0;i<n;i++)
   	cin>>cal[i];
   	sort(cal,cal+n);
   	for(i=0;i<n;i++)
   	{
   		sum+=cal[i];
   		if(sum<=a)
   		r++;
   		if(sum<=b)
   		s++;
   	}
   	if(r>s)
   	cout<<"Raghu Won\n";
   	else if(s>r)
   	cout<<"Sayan Won\n";
   	else
   	cout<<"Tie\n";
   }
    return 0;
}
