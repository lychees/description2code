#include <bits/stdc++.h>

using namespace std;


pair<long long,long long> tReq[100005];

long long t[100005];
int main()
{
    long long n,q,k,i,j;
    
    cin>>n>>q;
    
    for(i=1;i<=n;i++)
    {
    	cin>>tReq[i].second;
    }
    
    for(i=1;i<=n;i++)
    {
    	cin>>tReq[i].first;
    }
    
    sort(tReq+1,tReq+n+1);
    
    j=2;
    t[1]=tReq[n].second;
    for(i=n-1;i>=1;i--)
    {
    	t[j]=t[j-1]+tReq[i].second;
    	j++;
    }
    
    while(q--)
    {
    	cin>>k;
    	
    	cout<<t[k]<<endl;
    }
    
}
