#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		vector < char > arr;
		 string a;
		 cin>>a;
		 int l  = a.length();
		 for (int i = 0; i < l; ++i)
		 {
		 	arr.push_back(a[i]);
		 }
		sort(arr.begin(),arr.end());
		for (int i = 0; i < l; ++i)
		{
			cout<<arr[i];
		}
		cout<<endl;
	}
}
