using namespace std;
int gcd(int a,int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
int main()
{
	int t,n,x=0,y;
	cin>>t;
	while(t--)
	{
		cin>>n;
		x=0;
		for(int i=0;i<n;i++)
		{
			cin>>y;
			x=gcd(y,x);
		}
		if(x==1)
		cout<<"YES\n";
		else
		cout<<"NO\n";
	}
	return 0;
}
