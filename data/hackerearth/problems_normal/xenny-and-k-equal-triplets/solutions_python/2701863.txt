def ans(a,b):
	for i in range(2,a):
		while a%i==0 and b%i==0:
			a,b = a/i,b/i
	print str(int(a))+'/'+str(int(b))

def ncr(n,r):
	if r> n/2:
		r = n-r
	ans = 1
	for i in range(1,r+1):
		ans *= (n - r + i)
		ans /= i
	return ans

for _ in range(input()):
	n,k = map(int,raw_input().split())
	lst = map(int,raw_input().split())
	count = lst.count(k)
	base = ncr(n,3)
	if count == 0:
		print '0/1'
		continue
	elif count == n:
		print '1'
	else:
		top = ncr(count,3)
		ans(top,base)
