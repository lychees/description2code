using namespace std;
typedef long long ll;

int gcd(int a, int b)
{
    while(b) b ^= a ^= b ^= a %= b;
    return a;
}

ll max(ll a, ll b)
{
    return (a>b)?a:b;
}

int main()
{
    int arr[26],t,val,flag;
    string str;
    cin>>t;
    while(t--)
    {
        flag=0;
        cin>>str;
        int len = str.length(),i;
        memset(arr,0,sizeof(arr));
        for(i=0;i<len;++i)
        {
            arr[str[i]-'a']++;
        }
        val = arr[0];
        for(i=0;i<26;++i)
        {
            if(arr[i]!=val)
            {
                flag=1;
            }
        }
        if(flag)
            cout<<"No\n";
        else
            cout<<"Yes\n";
    }
    return 0;
}
