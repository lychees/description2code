#include <iostream>
#include <cstdio>
#include <string>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		string s;
		cin>>s;
		int n=s.size();
		bool a[n];
		for (int i = 0; i < n; ++i)
		{
			a[i]=true;
		}
		for (int i = 0; i < n; ++i)
		{
			if(s[i]=='B')
			{
				a[i]=false;
				if(i-1>=0)
					a[i-1]=false;
				if(i-2>=0)
					a[i-2]=false;
			}
			else if(s[i]=='D')
			{
				a[i]=false;
			}
			else if(s[i]=='S')
			{
				a[i]=false;
				if(i-1>=0)
					a[i-1]=false;
				if(i+1<n)
					a[i+1]=false;
			}
		}
		int count=0;
		for (int i = 0; i < n; ++i)
		{
			if(a[i]==true)
			{
				count++;
			}
		}
		printf("%d\n",count );
	}
    return 0;
}
