    #include <iostream>
    #include<cstring>
    using namespace std;
    
    long long int dp[300][300];
    long long int numberOfTeams(int a,int b){
    	if(dp[a][b] > 0)
    		return dp[a][b];
    	if(a<b)
    		return 0;
    	long long int result = 1;
    	for(int i=b;i<a;i++)
    		result+=numberOfTeams(a-i,i);
    	dp[a][b] = result;
		return result;
    }
    
   int main(){
   	
   	int t;
   	cin>>t;
   	memset(dp,-1,sizeof(dp));
   	
   	while(t--){
   		int N,K;
   		cin>>N>>K;
   		long long int result = numberOfTeams(N,K);
   		cout<<result<<endl;
   	}
    return 0;
    }
