#include <iostream>
#define MAX 1000000
using namespace std;

int main()
{
    long long int T,N,A[MAX],i,j,count;
    cin>>T;
    while(T--)
    {
     count=0;
     cin>>N;
     for(i=0;i<N;i++)
     {
      cin>>A[i];
     }
     for(i=0;i<N;i++)
     {
      for(j=i;j<N;j++)
      {
       if(j>=i && A[i]==A[j])
       count++;
      }
     }
     cout<<count<<"\n";
    }
    return 0;
}
