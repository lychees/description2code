#include <stdio.h>
#include <iostream>
using namespace std;
int main() {
    unsigned int digits[10000];
    int first=0, last=0, carry, x, sum, limit=100000, temp;
    digits[0] = 1;
    int S[1001];
    S[0]=1;
    S[1]=1;
    for(int n=2; n <= 1000; n++) {
        carry = 0;
        //first=last=0;
        for(x=first; x <= last; x++) {
            carry = digits[x]*n + carry;
            digits[x] = carry%limit;
            if(x == first && !(carry%limit)) first++;
            carry /= limit;
        }
        if(carry)
            digits[++last] = carry;
   		 sum=0;
    	for(int i=first; i<= last; i++){
        	temp=digits[i];
        	for(;temp;temp/=10)
            	sum+=(temp%10);
    	}
    	S[n]=sum;
    	//cout<<last<<"\n";
    }
    int T,N;
    cin>>T;
    for(;T>0;T--){
    	scanf("%d",&N);
    	printf("%d\n",S[N]);
    }
}
