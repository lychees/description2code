#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,i,j,ans,ma;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        long int a[n+1],b[n+1];
        for(i=0;i<n;i++)
        {
            scanf("%ld",&a[i]);
        }
        //printf("hi");
        for(i=0;i<n;i++)
        {
            scanf("%ld",&b[i]);
            b[i]+=a[i];
        }
        //printf("hi");
        sort(a,a+n);
        sort(b,b+n);
        ans=0;
        ma=0;
        i=0;j=0;
        while(i<n)
        {
            //printf("%d %d\n",i,j);
            while(a[i]<b[j] && i<n)
                i++,ans++;
            ma=max(ans,ma);
            if(i>=n)
                break;
            if(a[i]==b[j])
                i++,j++;
            while(a[i]>b[j] && j<n)
                j++,ans--;
        }
        printf("%d\n",ma);

    }
}
