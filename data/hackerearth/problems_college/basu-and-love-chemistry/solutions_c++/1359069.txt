#include<iostream>
#include<string>
#include<cmath>
#define max 1000000007
using namespace std;
int main()
{
	string a;
	long long int i,sum=0,num,a1[200005],b[200005],n;
	cin >>a;
	a1[0]=1;
	n=a.length();
	for(i=1;i<n;i++)
    {
    	a1[i-1]%=max;
    	a1[i]=(a1[i-1]*10+1);
    	a1[i]%=max;
    }
	for(i=0;i<n;i++)
	{
		b[i]=b[i]%max;
		b[i]=(a1[i]*(n-i));
		b[i]%=max;
	}
	for(int i=0;i<n;i++)
	{
		sum+=(a[n-i-1]-'0')*b[i];
		sum%=max;
	}
	cout << sum <<endl;
	return 0;
}
