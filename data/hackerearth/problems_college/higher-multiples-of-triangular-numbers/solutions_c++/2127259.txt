#include <cstdlib>
#include <iostream>

using namespace std;
int i,n,t,a[1001],b[1001],j;
int main(int argc, char *argv[])
{   a[1]=1;b[1]=1;
    for(i=2;i<=1000;i++)
    {  a[i]=a[i-1]+i;
       for(j=2;j<=a[i]/2;j++)
          if(a[i]%j==0)
            b[i]++;
       b[i]+=2;
          
     // cout<<a[i]<<" "<<b[i]<<endl;
    }
    cin>>t;
    for(i=1;i<=t;i++)
    {   cin>>n;
        for(j=1;j<=1000;j++)
           if(b[j]>n)
           {   cout<<a[j]<<endl;
               break;
           }
                     
    }
    
    return EXIT_SUCCESS;
}
