const int  N=1e6+10;
ll n,k,s[N],res,a[N];
bool p[N];


int main()
{
	std::ios::sync_with_stdio(0);
	cin.tie(0);
	cin>>n>>k;
	for(int i=1;i<=n;i++)
	{
		p[i]=1;
		s[i]=0;
		a[i]=i;
	}
	p[1]=0;
	res=0;
	for(int i=2;i<=n;i++)
	{
		if(i*i<=n)
		{
			if(p[i])
			{
				for(int j=i;j<=n;j+=i)
				{
					p[j]=0;
					s[j]+=i;
					while(a[j]%i==0)a[j]/=i;
				}
			}
		}
		if(a[i]>1)s[i]+=a[i];
		if(s[i]==k)res++;
	}
	cout<<res<<"\n";
	return 0;
}
