#include<iostream>
#include<string.h>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    for(int i=0;i<tc;i++)
    {
    	int count=0,alen=0,blen=0,alencounter=0,blencounter=0;
    	string a;
    	string b;
    	cin>>a;
    	cin>>b;
    	alen=a.length();
    	blen=b.length();
    	while(alencounter<=(alen-1) && blencounter<=(blen-1))
    	{
    		if(a.at(alencounter)==b.at(alencounter))
    		{
    			count++;
    		}alencounter++;blencounter++;
    	}cout<<count<<endl;
    }
    return 0;
}
