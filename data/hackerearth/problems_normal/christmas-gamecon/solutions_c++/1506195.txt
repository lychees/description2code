#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    long long n,i,ch;
    long arr[100005];
    while(t--)
    {
    	ch=0;
    	cin>>n;
    	for(i=0;i<n;i++)
    	     cin>>arr[i];
    	for(i=0;i<n;i++)
    	     arr[i]=arr[i]%5;
    	for(i=0;i<n;i++)
    	 ch=arr[i]^ch;
    	 if(ch)
    	  cout<<"Aniruddha"<<endl;
    	  else
    	  cout<<"Andrew"<<endl;
    }
    return 0;
}
