	#include<bits/stdc++.h>
	using namespace std;
	#define fr(i,n) for(int i=0;i<n;i++)
	#define fr1(i,n) for(int i=1;i<=n;i++)
	#define ll long long
	#define pii pair<int,int>
	#define ff first
	#define ss second
	#define mp make_pair
	#define pb push_back
	#define MOD 1000000007
	#define MAX 100002
	#define INF 100005
	vector<pii> v;
	ll fact[101];
	ll inv1[101];
	ll inv(ll t){
		ll ans=1,b=MOD-2;
		while(b>0){
			if(b%2)ans=(t*ans)%MOD;
			t=(t*t)%MOD;
			b/=2;
		}
		return ans;
	}

	ll ncr(int n,int r){
		if(r>n||r<0)return 0;
		return(((fact[n]*inv1[r])%MOD)*inv1[n-r])%MOD;
	}

	int main(){
		int n,l,r;
		cin>>n>>l>>r;
		fact[0]=1;
		fact[1]=1;
		ll dp[101];
		dp[0]=1;
		dp[1]=0;
		for(int i=2;i<=n;i++){
			fact[i]=(i*fact[i-1])%MOD;
			dp[i]=((i-1)*(dp[i-1]+dp[i-2]))%MOD;
		}
		for(int i=1;i<=n;i++){
			inv1[i]=inv(fact[i]);
		}
		inv1[0]=1;
		ll ans=0;
		for(int i=l;i<=r;i++){
			ans=(ans+(ncr(n,i)*dp[i])%MOD)%MOD;
			//cout<<dp[i]<<endl;
		}
		cout<<(ans*inv1[n])%MOD;
	}
