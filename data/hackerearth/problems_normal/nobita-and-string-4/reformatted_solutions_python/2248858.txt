'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
t = input()
while (t > 0):
    s = map(str, raw_input().split(" "))
    l = len(s)
    k = {}
    c = 0
    c_l = l-1
    while c_l >= 0:
        k[c] = s[c_l]
        c += 1
        c_l -= 1
    rev = ""
    for i in range(l):
        rev += k[i]
        rev += ' '
        l -= 1
    t = t-1
    print rev
