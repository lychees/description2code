#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t,a,b,c,d,e,f;
   cin>>t;
   while(t--)
   {
   	cin>>a>>b;
   	c=a;
   	d=a;
   	while(c%b!=0)
   	{
   		c++;
   	}
   	while(d%b!=0)
   	d--;
   	
   	e=abs(a-c);
   	f=abs(a-d);
   	if(e==f)
   	cout<<max(c,d)<<endl;
   	else
   	{
   		if(e>f)
   		cout<<d<<endl;
   		else
   		cout<<c<<endl;
   	}
   	
   }
    return 0;
}
