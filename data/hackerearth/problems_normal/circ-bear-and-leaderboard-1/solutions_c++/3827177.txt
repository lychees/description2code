#include <bits/stdc++.h>
#define mp make_pair
#define pb push_back
#define pii pair<int,int>
#define LL long long
#define st first
#define nd second
using namespace std;
LL f[1000005],f2[1000005],po[500005],ans,n,q;
set<LL> s;
vector<LL> v;
vector<pii> vv(500005);
void add(LL x,LL val){
	for(;x<=1000004;x+=x&-x)
		f[x]+=val;
}
void add2(LL x,LL val){
	for(;x<=1000004;x+=x&-x)
		f2[x]+=val;
}
LL que(LL x){
	LL ret=0;
	for(;x>0;x-=x&-x)
		ret+=f[x];
	return ret;
}
LL que2(LL x){
	LL ret=0;
	for(;x>0;x-=x&-x)
		ret+=f2[x];
	return ret;
}
int bs(LL x){
	return lower_bound(v.begin(),v.end(),x)-v.begin();
}
int main(){
	scanf("%lld %lld",&n,&q);
	s.insert(-1);
	s.insert(0);
	for(int i=0;i<q;++i){
		scanf("%d %d",&vv[i].st,&vv[i].nd);
		po[vv[i].st]+=vv[i].nd;
		s.insert(po[vv[i].st]);
		s.insert(po[vv[i].st]-1);
	}
	memset(po,0,sizeof po);
	for(set<LL>::iterator it=s.begin();it!=s.end();++it)
		v.pb(*it);
	add(bs(0),n);
	for(int i=0;i<q;++i){
		ans-=(n-que(bs(po[vv[i].st]))+1)*po[vv[i].st];
		//cout<<(n-que(bs(po[vv[i].st]))+1)<<" "<<po[vv[i].st]<<endl;
		ans-=que2(bs(po[vv[i].st]-1));
		add2(bs(po[vv[i].st]),-po[vv[i].st]);
		add(bs(po[vv[i].st]),-1);
		po[vv[i].st]+=vv[i].nd;
		ans+=que2(bs(po[vv[i].st]-1));
		//cout<<que2(bs(po[vv[i].st]-1))<<" "<<po[vv[i].st]<<endl;
		add2(bs(po[vv[i].st]),po[vv[i].st]);
		add(bs(po[vv[i].st]),1);
		ans+=(n-que(bs(po[vv[i].st]))+1)*po[vv[i].st];
		//cout<<(n-que(bs(po[vv[i].st]))+1)<<" "<<po[vv[i].st]<<endl;
		printf("%lld\n",ans);
	}
}
