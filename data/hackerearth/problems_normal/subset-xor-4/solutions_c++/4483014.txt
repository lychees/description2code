#include <cstdio>
#include <vector>
using namespace std;

int main()
{

    int t, n;

    scanf("%d", &t);
    while(t--)
    {
    	scanf("%d", &n);
    	vector<int> read(n);
    	for(int i = 0; i < n; ++i)
    		scanf("%d", &read[i]);

    	if(n > 1)
    		printf("0\n");
    	else
    		printf("%d\n", read[0]);
    }

    return 0;
}
