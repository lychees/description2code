#include <iostream>
#include <map>
#include <set>
#include <algorithm>
#define MAX 1000001
using namespace std;

int main()
{
    int t,n,max,a,i;
    
    scanf("%d",&t);
    while(t--){
    	int m[MAX]={0};
    	scanf("%d",&n);
    	while(n--){
    		scanf("%d",&a);
    		m[a]++;
    	}
    	
    	max=m[0];
    	for(i=1;i<MAX;i++){
    		if(max<m[i]){
    			max = m[i];
    		}
    	}
    	
    	for(i=MAX-1;i>=0;i--){
    		if(max==m[i]){
    			printf("%d ",i);
    		}
    	}
    	
    	printf("\n");
    }
    
    
    return 0;
}
