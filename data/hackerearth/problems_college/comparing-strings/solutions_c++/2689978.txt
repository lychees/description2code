#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   
   while(t--)
   {
   	string s1,s2;
   	int c=0;
   	cin>>s1>>s2;
   	for(int i=0;i<s1[i] && s2[i];i++)
   	{
   		if(s1[i]==s2[i])
   		{
   			c=1;
   			break;
   		}
   		
   		
   	}
   	
   	if(c==1)
cout<<"YES"<<endl;  
else
   		cout<<"NO"<<endl;
   }
    return 0;
}
