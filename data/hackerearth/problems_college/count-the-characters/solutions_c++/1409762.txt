#include<cstring>
#include <iostream>
#include<string>
using namespace std;

int main()
{
    int n,i,a[27][2],j;
    string s;
    cin>>n;
    memset(a,0,sizeof a);
    for(j=1;j<=n;j++)
    {
    	cin>>s;
    	for(i=0;i<s.length();i++)
    	{
    		if(a[s[i]-'a'][1]==0)
    		{	
    			a[s[i]-'a'][0]++;
    			a[s[i]-'a'][1]=1;
    		
    		}
    	}
    	for(i=0;i<26;i++)
    		a[i][1]=0;
    }
    int ans=0;
    for(i=0;i<26;i++)
    	if(a[i][0]==n)
    		{ans++;}
    cout<<ans;
}
