#include<stdio.h>
#include<string.h>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,n,t,l,m,s,tst=0;
	scanf("%d",&t);
	while(t--)
	{
		tst++;
		char x[1000]={};
		char y[1000]={};
		scanf("%s%s",x,y);
		l=strlen(x);
		m=strlen(y);
		s=0;
		int ans=0;
		if(l!=m)
		printf("Case %d: -1\n",tst);
		else
		{
			for(i=0;i<l;i++)
			{
				if(x[i]==y[i])
				continue;
				if(x[i]=='0')
				{
					int flg=0;
					for(j=i+1; j<l; j++)
					{
						if(x[j]=='1' && y[j]=='0')
						{
							char ch;
							ch=x[i];
							x[i]=x[j];
							x[j]=ch;
							ans++;
							flg=1;
							break;
						}
					}
					for(j=i-1; j>=0; j--)
					{
						if(flg)
						break;
						if(x[j]=='1' && y[j]=='0')
						{
							char ch;
							ch=x[i];
							x[i]=x[j];
							x[j]=ch;
							ans++;
							flg=1;
							break;
						}
					}
				}
			}
			//printf("%s\n",x);
			int ct=0,ct1=0,ct2=0;
			for(i=0; i<l; i++)
			{
				if(x[i]==y[i])
				continue;
				if(x[i]=='1')
				{
					ct++;
				}
				if(x[i]=='?')
				{
					if(y[i]=='1')
					ct1++;
					else
					ct2++;
				}
			}
			int ct3=0;
			for(i=0; i<l; i++)
			{
				if(x[i]==y[i])
				continue;
				if(x[i]=='0' && y[i]=='1')
				ct3++;
			}
			if(ct<=ct1 || ct==0)
			{
				if(ct==0)
				ans=ans+ct1+ct2+ct3;
				else
				ans=ans+ct*2+ct2+ct1-ct+ct3;
				printf("Case %d: %d\n",tst,ans);
			}
			else
			{
					printf("Case %d: -1\n",tst);
				//puts("-1");
			}
		}
		
	}
}
