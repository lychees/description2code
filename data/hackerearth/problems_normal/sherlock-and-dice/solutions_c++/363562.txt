using namespace std;

#define ll long long

#define mpa make_pair

int N;
int K;

double dp[51][10001];

double rec(int M, int sum)
{
	if(sum < 0)
		return 0;
	if(M == 0) {
		//cout<<"sum => " << sum <<"\n";
		return double (sum == 0);
	}
	
	if(dp[M][sum] != -1.0)
		return dp[M][sum];
	
	double ans = 0;
	for(int i = 1; i <= N; ++i) {
		if(sum - i >= 0)
			ans += (rec(M - 1, sum - i));
	}
	ans /= (double)N;
	return dp[M][sum] = ans;
}

void solve()
{
	int M;
	scanf("%d%d%d", &M, &N, &K);
	
	if(K < M || K > N*M){
	 	printf("0.000 0\n");
		return;
	}
	
	double chaye = rec(M, K);
	int cnt = 0;
	while(chaye < 1 && chaye > 0){chaye *= 10; cnt++;}
	printf("%.3lf %d\n",chaye,cnt);
}

int main()
{
	
	int T;
	scanf("%d", &T);
	
	while(T--) {
		for(int i = 0; i < 51; ++i) {
		for(int j = 0; j <= 10001; ++j)
			dp[i][j] = -1.0;
	}
		solve();
	}
	
	return 0;
}
