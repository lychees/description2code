#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
	int t;
	scanf("%d ",&t);
	while(t--)
	{
		char str[200001];
		gets(str);
		int l=strlen(str);
		int i=0,j;
		for(int k=0;k<l;k++)
		{
			if(i>0 && str[k]=='1' && str[i-1]=='0')
				--i;
			else
			{
				str[i]=str[k];
				++i;
			}
		}
		j=i-1;
		for(int k=i-1;k>=0;k--)
		{
			if(j<i-1 && str[k]=='1' && str[j+1]=='0')
				++j;
			else
			{
				str[j]=str[k];
				--j;
			}
		}
		//printf("%d\n",j);
		printf("%d\n",i-j-1);
	}
}
