#include <iostream>

using namespace std;

int find_lowest_odd(int *arr, int length) {
        int min = 190;
        for(int i=0;i<length;i++) {
                if((arr[i]%2 != 0) && (min > arr[i])) {
                        min = arr[i];
                }
        }
        if(min == 190)
        return 0;
        return min;
}

int main() {
        int n , a[10000], sum = 0, l;
        cin >> n;
        for(int i=0; i<n; i++) {
                cin >> a[i];
                sum += a[i];
        }
        if(sum%2 != 0)
                cout << sum << endl;
        else {
                /* find lowest odd */
                l = find_lowest_odd(a, n);
                if(l==0)
                        cout << ":(" << endl;
                else
                        cout << sum-l << endl;
        }
        return 0;
}
