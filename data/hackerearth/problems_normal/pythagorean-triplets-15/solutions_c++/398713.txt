#include<iostream>
#include<map>

long gcd(long a, long b)
{
if(b==0)
  return a;
else
  return gcd(b,a%b);  
     
}


using namespace std;
int main()
{
    long t,n;
    cin>>t;
    while(t--)
    {
    cin>>n;
    long long cnt=0;
    long lim=n;
    
    if(n>10000)
      lim=n/100;  
    if(n>50000)
      lim=n/200;  
    if(n>100000)
      lim=n/300;  
    
      
     for(long i=1;i<=lim;i++)
      {
      for(long j=1;j<i ;j++)
       {
         long c=i*i+j*j;    
         if( (c<=n)&&gcd(i,j)==1&&(i%2!=1||j%2!=1))      
           {
 //          cout<<(i*i-j*j)<<"-"<<(2*i*j)<<"-"<<(i*i+j*j)<<"::"<<i<<"-"<<j<<endl;
                                      
             cnt=cnt+n/c;}                           
                      
       }
      }
     cout<<cnt<<endl;              
     }
     return 0;
}
