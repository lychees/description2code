#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    int t,i,Arr[55][55],p,count,j;
    char str[55];
    cin>>t;
    while(t--)
    {
        for(i=0;i<55;i++)
        for(j=0;j<55;j++)
        Arr[i][j]=0;
        cin>>str;
        p=strlen(str);
        for(i=p-1;i>=0;i--)
        {
            count=1;
            Arr[i][i]=1;
            if(i+1!=p){
            if(str[i]==str[i+1])
            {
                count++;
                Arr[i][i+1]=3;// in case aab-->a,a,aa
            }
            else
            Arr[i][i+1]=2;
            }
            for(j=i+2;j<p;j++)
            {
                if(str[i]==str[j])
                {
                    count++;
                    count+=Arr[i+1][j-1];
                }
                Arr[i][j]=Arr[i+1][j]+count;
            }
        }
            cout<<Arr[0][p-1]<<endl;
}
        return 0;
}
