#include <iostream>
using namespace std;

int main()
{
    int T,i,N,K,j,ctr;
    int a[1000];
    scanf("%d\n",&T);
    for(i=1;i<=T;i++)
    { ctr=0;
    	scanf("%d ",&N);
      scanf("%d\n",&K);
      for(j=0;j<N;j++)
      cin>>a[j];
      for(j=0;j<N;j++)
      {
      	if(ctr==K)
      	break;
      else	if(a[j]<=0)
      	ctr++;
      }
     if(ctr>=K)	
     printf("\nNO");
     else 
     printf("\nYES");
    	
    }
    
    
    return 0;
}
