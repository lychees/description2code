#include <iostream>
#include <bits/stdc++.h>
using namespace std;

long long int A,B,C,K;
long long int func(long long int x)
{
	long long int val;
	val = A*x*x + B*x + C;
	return val;
}

int main()
{
    long long int ans=100001,T,i,l,r,mid,p;
    cin >> T;
    while(T--)
    {
    	l = 0;
    	r = 100001;
    	cin >> A >> B >> C >> K;
    	while(l < r)
    	{
    		mid = l + (r - l)/2;
    		p = func(mid);
    		if(p < K)
    		{
    			l = mid + 1;
    		}
    		else
    		{
    			r = mid;
    			ans = mid;
    		}
    		
    	}
    	cout << ans << endl;
    }
    return 0;
}
