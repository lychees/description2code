n, q = map(int, raw_input().split())
arr = map(int, raw_input().split())
for i in range(0, q):
    s = map(int, raw_input().split())
    if s[0] == 0:
        arr[s[1]-1] = s[2]
    else:
        j = 0
        while j < n and arr[j] < s[1]:
            j += 1
        if j == n:
            print -1
        else:
            print j+1
