#include <iostream>
#include <cstdio>
using namespace std;
int t,l,r,i;
char str[1000100];
int sum[1000100];
long long res;
int main()
{
	scanf("%d",&t);
	while(t--){
		scanf("%s",str);
		sum[0]=0;
		r=0;
		for(i=1;str[i]!='$';i++){
			if(str[i]>='a'&&str[i]<='z')
				sum[i]=sum[i-1]+(str[i]-'a')+1;
			else
				sum[i]=sum[i-1]+(str[i]-'A')+1;
			r++;
		}
		l=1;
		res=100;
		while(l<=r){
			res+=(sum[r]-sum[l-1]+100);
			if(l==r)
			{
				res+=100;
				break;
			}
			res+=(sum[r]-sum[r-1]);
			if(str[l]==str[r])
			{	
				r--;
				res+=(sum[r]-sum[l]);
				l++;
				res+=100;
			}
			else
				break;
		}
		printf("%lld\n",res);		
	}
    return 0;
}
