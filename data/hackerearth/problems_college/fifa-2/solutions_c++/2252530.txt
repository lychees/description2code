#include <cstdio>  
#include <cstring>  
int const MAX = 1e5 + 5;  
int x[MAX], y[MAX], z[MAX];  
  
int main()  
{  
    int N;  
    scanf("%d", &N);  
    memset(z, 0, sizeof(z));  
    for(int i = 0; i < N; i++)  
    {  
        scanf("%d %d", &x[i], &y[i]);  
        z[x[i]] ++;  
    }  
    for(int i = 0; i < N; i++)  
        printf("%d %d\n",  N - 1 + z[y[i]], N - 1 - z[y[i]]);  
}
