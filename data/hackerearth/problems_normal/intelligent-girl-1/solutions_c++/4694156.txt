#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   	string s;
   	int dp[100000],i;
   	getline(cin,s);
   	memset(dp,0,sizeof(dp));
   	if((s[s.length()-1]-48)%2==0)
   		dp[s.length()-1]=1;
   	for(int i=s.length()-2;i>=0;i--){
   		if((s[i]-48)%2==0)
   			dp[i]=dp[i+1]+1;
   		else
   			dp[i]=dp[i+1];
   	}
   	for(i=0;i<s.length();i++)
   		cout<<dp[i]<<" ";
   	
   
   
    return 0;
}
