#include<bits/stdc++.h>
using namespace std ;

int main()
{   int test ;
    cin>>test;
    while(test--)
    {
       int no , maxi;
       cin>>no>>maxi;
       long long gr[no];
       for(int i = 0 ; i<no;i++)
        cin>>gr[i];
       sort(gr,gr+no);
       int ser = 0 ,countr=0;
       for(int i = 0 ; i<no;i++)
       {
           ser+=gr[i];
           if(ser<=maxi)
            countr++;
           else
           {

               break;
           }

       }

     cout<<countr<<endl;

    }
    return 0 ;
}
