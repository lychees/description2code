#include <iostream>
#include <string.h>
using namespace std;
int main()
{
    int t=0;
    cin>>t;
    while(t--)
    {
        char s[21]={'\0'};
        cin>>s;
        int len = strlen(s);
        int i = len;
        int ans =-1;
        for(i;i>0;--i)
        {
            ans =-1;
            int b =0;
            int e = b + i -1;
            while(e<len)
            {
                int j = b;
                int l = e;
                bool btrue = true;
                while(btrue && j<=l)
                {
                    if(s[j]!=s[l])btrue =false;
                    ++j;
                    --l;
                }
                if(btrue)
                {
                    ans = i;
                    break;
                }
                    
                ++b;
                ++e;// = begin + i -1;
            }
            if(ans!=-1)
                break;
        }
        cout<<ans<<endl;
    }
    return 0;
    
}
