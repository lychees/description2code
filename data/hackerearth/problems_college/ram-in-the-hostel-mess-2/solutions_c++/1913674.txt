#include <iostream>
#include <limits>
#include <vector>

using namespace std;

size_t arr[100000]={0};

size_t nways(size_t n)
{
  if(n==1)
    return 2;
  if(n==2)
    return 3;
  if(arr[n])
    return arr[n];
  return (arr[n] = nways(n-1) + nways(n-2));
}

int main()
{
  int t;
  cin>>t;
  size_t n,nwa;
  while(t--)
  {
    cin>>n;
    nwa=nways(n);
    cout<<nwa*nwa<<"\n";
  }
  return 0;
}
