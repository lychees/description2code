#include <iostream>
#include <algorithm>

#define LL long long int

using namespace std;

int c[53], dp[53];
int n,m;
LL ans = 0;

int findLeastColour(int d);

int main(){
	cin>> n>> m;
	
	for(int i = 1; i <= m; i++)
		cin>>c[i];
		
	for(int i = 0; i < n; i++)
	{
		int d;
		cin >> d;
		findLeastColour(d);
	}
	
	cout<<ans<<endl;
	return 0;
}

int findLeastColour(int d){
	int cur = c[d];
	int k = dp[d];
	while(cur * (++k) <= ans);
	dp[d] = k;
	ans = cur * k;
	
	while(1){
		bool flag = true;
		for(int i = d + 1; i <= m; i++)
		{
			if(ans % c[i] == 0){
				flag = false;
				ans += cur;
				break;
			}
		}
		if(flag) break;
	}
	//cout<<ans<<endl;
	return 0;
}
