#include <iostream>
using namespace std;
#include <string>

int check(string str)
{
	
	
	
	int len=str.size();
	int i=0;
	for(i=0;i<len;i++)
	{
		if(str[i]=='@')
		break;
		if(i>16)
		return 0;
		
		if(str[i]>='a'&&str[i]<='z')
		continue;
		else if(str[i]>='A'&&str[i]<='Z')
		continue;
		else if(str[i]>='0'&&str[i]<='9')
		continue;
		else if(str[i]>='_')
		continue;
		else 
		return 0;
		
	}
	if(i==0)
	return 0;
	int k=0;
	int flag=0;
	for(i++;i<len;i++)
	{
		if(str[i]=='.')
		break;
		k++;
		if(k>16)
		return 0;
		
		if(str[i]>='a'&&str[i]<='z')
	{	flag=1;
		continue;
	}
		else if(str[i]>='A'&&str[i]<='Z')
	{
		flag=1;
		continue;
	}
		else if(str[i]>='0'&&str[i]<='9')
		continue;
		else if(str[i]=='-')
		continue;
		else 
		return 0;
		
		
	}
	if(k==0)
	return 0;
	if(flag==0)
	return 0;
	k=0;
	for(i++;i<len;i++)
	{
		k++;
		if(k>10)
		return 0;
		if(str[i]>='a'&&str[i]<='z')
		continue;
		else 
		return 0;
				
		
	}
	if(k==0)
	return 0;
	return 1;
	
	
	
}
int main()
{
    int num=0;
    cin>>num;
    
    for(int i=0;i<num;i++)
    {
    	string str;
    	cin>>str;
    	if(check(str))
    	{
    		cout<<"YES"<<endl;
    		
    	}
    	else
    	{
    		cout<<"NO"<<endl;
    	}
    }
}
