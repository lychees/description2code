#include<iostream>
#include<cmath>
using namespace std;

int main()
{
  long int a[1000000],b[1000000],n,bk,bd,ak,ad;
 int tc,i;


 
 ak=0;bk=0;

for(i=0;i<n;i++)
 { 
  
    cin>>a[i];
     if(i>0)
   {
    ad=a[i]-a[i-1]; 
   if(ak<abs(ad))
    ak=abs(ad);
   }
 } 
  
for(i=0;i<n;i++)
 { 
   
   cin>>b[i];
   if(i>0)
   {
    bd=b[i]-b[i-1];

}
