#include <bits/stdc++.h>
#define aa long long
using namespace std;
int main()
{
    aa int tc;
    cin>>tc;
    while(tc--)
    {
        aa int n,m,coun=0,sum=0,ans=0;
        cin>>n>>m;
        aa int a[n];
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
            ans += a[i];
        }
        sort(a,a+n,greater<int>()); // sorted in decreasing order
        int i=0;
        
           while(sum<m && i<n){
           	sum += a[i];
           	i=i+1;
           	coun++;
           }
            if(ans < m) cout<<-1<<endl;
            else
        cout<<coun<<endl;
    }
}
