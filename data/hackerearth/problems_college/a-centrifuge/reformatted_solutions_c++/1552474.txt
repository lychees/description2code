#include <algorithm>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <string>
#include <vector>
using namespace std;

typedef long long LL;
typedef vector<int> VI;

#define REP(i, n) for (int i = 0, i##_len = (n); i < i##_len; ++i)
#define EACH(i, c)                                                             \
  for (__typeof((c).begin()) i = (c).begin(), i##_end = (c).end();             \
       i != i##_end; ++i)

int T, M, N;
int C[2];
int main() {
  scanf("%d", &T);
  REP($, T) {
    scanf("%d%d", &M, &N);
    C[0] = N;
    C[1] = 0;
    int ans = 0;
    while (C[0] | C[1]) {
      int g = min(M, C[0]), h = min(M - g, C[1]);
      C[0] -= g;
      C[1] -= h;
      C[1] += g;
      ans++;
    }
    printf("%d\n", ans * 5);
  }
  return 0;
}
