n, m = map(int, raw_input().split())
mat = []
row = []
column = []
for i in range(0, n):
    num = map(int, raw_input().split())
    mat.append(num)
    row.append(sum(num))
for j in range(0, m):
    num = 0
    for i in range(0, n):
        num += mat[i][j]
    column.append(num)
maxim = 0
for i in range(0, n):
    for j in range(0, m):
        num = row[i] ^ column[j]
        if num > maxim:
            maxim = num
print maxim
