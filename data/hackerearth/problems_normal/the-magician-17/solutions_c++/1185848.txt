#include <bits/stdc++.h>

using namespace std;

int solve(int n, int k) {
    if (n == 1) return 1;
    if (k <= n / 2) return(k + k - 1);
    int x = solve(n - (n / 2), k - (n / 2));
    if (n % 2 == 0) return(x + x);
    else if (x == 1) return n;
    else return(x + x - 2);
}

int main()
{
    int test;
    cin >> test;
    while (test --) {
        int n, k;
        cin >> n >> k;
        cout << solve(n, k) << endl;
    }
}
