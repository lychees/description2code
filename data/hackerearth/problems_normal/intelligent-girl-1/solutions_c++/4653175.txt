#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   char str[100000];
   int arr[100000]={0},i,n,x;
   cin>>str;
   n=strlen(str);
   for(i=0;i<n;i++){
   		if(str[i]%2==0)
   			arr[i+1]=arr[i]+1;
   		else
   			arr[i+1]=arr[i];
   }
   x=arr[n];
   for(i=0;i<n;i++){
   	cout<<x<<" ";
   	if(str[i]%2==0)
   		x--;
   	
   	
   }
   
   	
    return 0;
}
