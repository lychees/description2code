#include <bits/stdc++.h>
using namespace std;
     
int main()
{
    int t;
    long long i,p,sum;
   	string a;
    cin>>t;
    assert(t<=10);
    while(t--)
    { 
    	p=0;
       	sum=0;
        i=0;
        cin>>a;
        int len=a.length();
        assert(len<=100000);
        while(i<len)
        {
        	while(i<len&&(a[i]=='A'||a[i]=='B'||a[i]=='C'||a[i]=='a'||a[i]=='b'||a[i]=='c'))
        	{
        		p++;
        		i++;
        	}
        	sum+=p*(p+1)/2;
        	p=0;
        	i++;
        }
        cout<<sum<<"\n";
    }
    return 0;
}
