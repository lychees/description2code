#include <bits/stdc++.h>

using namespace std;

int solve(int n, int k)
{
    if(k*2<n)
        return k*2;
    int x=(n+1)/2;
    if(n&1)
        return (solve(n-x, k-x)+1)%(n-x)*2+1;
    return solve(n-x, k-x)*2+1;
}

int main()
{
    int T;
    scanf("%d", &T);
    while(T--)
    {
        int N, K;
        scanf("%d%d", &N, &K);
        printf("%d\n", solve(N, K-1)+1);
    }
    return 0;
}
