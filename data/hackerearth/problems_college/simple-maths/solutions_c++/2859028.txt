#include<iostream>
using namespace std;
typedef long long ll;
ll gcd(ll a,ll b)
{
    if(b==0)
		return a;
	if(a>b)
		return gcd(b, a%b);
	return gcd(a, b%a);
}
ll lcm(ll a,ll b)
{
	ll num;
	num=gcd(a,b);
	return (a*b)/num;
}
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		ll n,a,b,cnt=0;
		cin>>n>>a>>b;
		if(a==0 || b==0)
			cnt=-1;
		else
		{
			if(a==0)
				cnt+=(n/b);
			else if(b==0)
				cnt+=(n/a);
			else
			{
				cnt+=((n/a)+(n/b));
				cnt-=(n/lcm(a,b));	
			}
		}
		cout<<cnt<<endl;
	}
	return 0;
}
