n = input()
s = []
for i in range(n):
    temp = raw_input()
    s.append(list(temp))
s2 = []
for i in range(len(s)):
    temp = []
    for j in range(1, len(s[i])+1):
        temp.append(s[i][-j])
    s2.append(''.join(temp))
for i in range(len(s)):
    s[i] = ''.join(s[i])
a = set(s).intersection(set(s2))
b = (a.pop())
print len(b), b[len(b)/2]
