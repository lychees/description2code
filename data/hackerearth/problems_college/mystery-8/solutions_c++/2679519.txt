#include <stdio.h>
#include <string.h>
#include <ctype.h>


int main(){
	int t, i, count = 0;
	scanf("%d", &t);
	char array[t][101], *ptrstart = NULL, *ptrlast = NULL;
	
	for(i = 0; i < t; i++)	scanf(" %[^\n]s", array[i]);	
	
	for(i = 0; i < t; i++){
		ptrstart = array[i];
		ptrlast = ptrstart + (strlen(array[i]) - 1);
		while(*ptrstart != '*' || *ptrlast != '*'){
			if(ispunct(*ptrstart) && *ptrstart != '*')	ptrstart++;
			if(ispunct(*ptrlast) && *ptrlast != '*')	ptrlast--;
			if(isalpha(*ptrstart) && isalpha(*ptrlast) && *ptrstart != '*' && *ptrlast != '*')ptrstart++, ptrlast--;
			
			if(isspace(*ptrstart))	ptrstart++;
			if(isspace(*ptrlast))	ptrlast--;
			
			if(isalpha(*ptrstart) && *ptrlast == '*'){
				printf("%c\n", *ptrstart);
				break;
			}
			if(*ptrstart == '*' && isalpha(*ptrlast)){
				printf("%c\n", *ptrlast);
				break;
			}
		}
		
	}
	return 0;
}
