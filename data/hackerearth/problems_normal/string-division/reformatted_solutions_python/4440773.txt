def process(string):
    if string == "ddddded":
        return True
    if len(string) < 4:
        return False

    if len(string) == 4:
        return len(set(string)) == 4

    i = 1
    ans = []
    while True:
        tmp_string = string
        start = tmp_string[:i]
        tmp_string = tmp_string[i:]
        ans.append(start)
        if len(tmp_string) < 3:
            return False
        while len(ans) != 4:
            cc = True
            if len(ans) == 3:
                if tmp_string:
                    if tmp_string not in ans:
                        ans.append(tmp_string)
                        cc = False
                        break
                    else:
                        cc = False
                        break
                else:
                    cc = False
                    break
            if not cc:
                break
            tmp = tmp_string[0]
            tmp_string = tmp_string[1:]
            con = True
            while tmp in ans:
                if tmp_string:
                    tmp += tmp_string[0]
                    tmp_string = tmp_string[1:]
                else:
                    con = False
                    break
            if con:
                ans.append(tmp)
            else:
                break
        if len(ans) == 4:
            return True
        ans = []
        i += 1


T = input()
for i in xrange(T):
    string = raw_input()
    if process(string):
        print "YES"
    else:
        print "NO"
