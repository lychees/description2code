#include <iostream>
using namespace std;

int main()
{
	int t, n, i, j, count;
    cin>>t;
    while(t--){
    	cin>>n;
    	int arr[n];
    	for(i=0; i<n; i++){
    		cin>>arr[i];
    	}
    	count = n;
    	for(i=0; i<n-1; i++){
    		for(j=i+1; j<n; j++){
    			if(arr[i] == arr[j])
    				count++;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
