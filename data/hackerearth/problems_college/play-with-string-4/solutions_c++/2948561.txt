#include <iostream>
#include <string.h> 
using namespace std;
 
int main()
{
    //cout << "Hello World!" << endl;
  int t;
  cin>>t;
  string a;
  int len;
  while(t--)
  {
  	cin>>a;
  	int flag=0;
    	len =a.length();
  	for(int i=0;i<len-2;i++)
  	{
  	    if(abs((a[i+1]-a[i]))!=abs(a[len-(i+1)]-a[len-(i+2)]))
  	    {
  	    	
  	        
  	    	flag=-1;
  	    	break;
  	    }
    }
    if(flag==-1)
    {
    	cout<<"Not Funny"<<endl;
    }
    else
    {
    	cout<<"Funny"<<endl;
    }
  }
    
    return 0;
}
