#include <bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	string s[n];
	for(int i=0;i<n;i++) cin>>s[i];
	vector<int>v[30];
	int coun[30],occ[30];
	memset(coun,0,sizeof(coun));
	memset(occ,0,sizeof(occ));
	for(int i=0;i<n;i++){
		for(int j=0;j<s[i].length();j++){
			occ[s[i][j]-'a']=1;
			
		}
	}
	
	for(int i=0;i<n;i++){
		for(int j=i+1;j<n;j++){
			string p=s[i],q=s[j];
			for(int k=0;k<min(p.length(),q.length());k++){
				if(p[k]!=q[k]){
					v[p[k]-'a'].push_back(q[k]-'a');
					++coun[q[k]-'a'];					
					break;
				}
			}
		}
	}
	vector<int>p,q;
	vector<vector<int> >T;
	for(int i=0;i<26;i++){
		if(coun[i]==0 && occ[i]==1)p.push_back(i);
	}
	while(!p.empty()){//cout<<"in";
		q.swap(p);
		T.push_back(vector<int>());
		while(!q.empty()){
			int ind=q.back();
			q.pop_back();
			T.back().push_back(ind);
		//	cout<<ind<<" "<<v[ind].size()<<endl;
			for(int i=0;i<v[ind].size();i++){
				if(--coun[v[ind][i]]==0){
					p.push_back(v[ind][i]);
				//	cout<<"in";
				}
				
			}
		}
	}
	for(int i=0;i<T.size();i++){
		sort(T[i].begin(),T[i].end());
		for(int j=0;j<T[i].size();j++)cout<<(char)(T[i][j]+'a');
		cout<<endl;
	}
	return 0;
}
