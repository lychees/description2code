#include <iostream>
using namespace std;
#include <string.h>
int main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		int n,hasht[300],max=0;
		memset(hasht,0,sizeof(hasht));
		cin>>n;
		string s;
		cin>>s;
		for(int i=0;i<s.length();i++)
		{
			hasht[s[i]]++;
		}
			for(int i=0;i<s.length();i++)
		{
			if(hasht[s[i]]>max)
			max=hasht[s[i]];
		}
		if(max%2==0) cout<<"Yes"<<endl;
		if(max%2!=0) cout<<"No"<<endl;
		
	}
  //  cout << "Hello World!" << endl;
    return 0;
}
