#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    for(int q=1;q<=n;q++)
    cout<<"*";
    cout<<endl;
    for(int i=n-2;i>=1;i--)
    {
    	for(int j=1;j<=i;j++)
    	{
    		cout<<' ';
    		
    	}
    	cout<<"*"<<endl;
    }
    for(int q=1;q<=n;q++)
    cout<<"*";
    return 0;
}
