#include<bits/stdc++.h>
using namespace std;
const string target = "101";


int main() {




                string s;
                cin>>s;
                long int n = s.size();
                long int k = target.size();
                vector<vector<long int> > dp(n+1, vector<long int>(k+1));
                dp[0][0] = 1;
                const int mod = 10000;
                assert(k == 3);
                 for(long int i=0;i<n;i++ ) for(long int j=0;j< k+1;j++) {
                        dp[i+1][j] = (dp[i+1][j]+dp[i][j]);
                        if (j < k && s[i] == target[j])
                                dp[i+1][j+1] = (dp[i+1][j+1]+dp[i][j]);
                }
                printf("%ld\n", dp[n][k]);


        exit(0);
}
