#include <iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while( t-- )
	{
		unsigned long long w, n;
		cin>>w>>n;
		unsigned long long A[n+1];
		A[0] = 0;
		unsigned long long sum = 0;
		for( unsigned long long i = 1; i <= n; i++ )
		{
			unsigned long long x, ans = 0;
			cin>>x;
			if( x == 1 )
			{
				A[i] = 2*i;
				ans = w + A[i] - A[i-1];
			}
			else if( x == 2 )
			{
				A[i] = A[i-1] + 1;
				ans = w + A[i] - A[i-1];
			}
			sum += ans;
		}
		cout<<sum<<endl;
	}
}
