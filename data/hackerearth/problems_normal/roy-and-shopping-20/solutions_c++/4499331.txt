#include<bits/stdc++.h>
using namespace std;
bool prime[1000006];
int arr[1000006];
int main()
{
	int t,r,n,i,j,p=0,ans;
	prime[0]=prime[1]=1;
	for(i=2;i*i<=1000006;i++)
	{
		if(!prime[i])
		{
			j=2;
			while(j*i<=1000006)
			{
				prime[j*i]=1;
				j++;
			}
			arr[p++]=i;
		}
	}
	for(i=j;i<=1000006;i++)
	{
		if(!prime[i])
		{
			arr[p++]=i;
		}
	}
	scanf("%d",&n);
	while(n--)
	{
		scanf("%d",&r);
		if(!prime[r])
		{
			printf("0\n");
			continue;
		}
		ans=0;
		for(i=0;i<p;i++)
		{
			if(r%arr[i]==0)
			{
				ans=r-arr[i];
				//flag=1;
				break;
			}
		}
		printf("%d\n",ans);
	}
}
