#include <bits/stdc++.h>
using namespace std;
int a[1000001]={0};
int powerprime(int n)
{
	int count=0;
	while(n%2==0)
	{
		count++;
		n=n/2;
	}
	for(int i=3;i<=sqrt(n);i+=2)
	{
		while(n%i==0)
		{
			count++;
			n=n/i;
		}
	}
	if(n>2)
	{
		count++;
	}
	return count;
}
void init()
{
	a[0]=0;
	a[1]=0;
	int i;
	for(i=2;i<1000001;i++)
	{
		a[i]=powerprime(i);
	}
	for(int i=1;i<1000001;i++)
	{
	  	a[i]+=a[i-1];
	}
}
int main()
{
    init();
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int sum=0;

		printf("%d\n",a[r]-a[l-1]);
	}
}
