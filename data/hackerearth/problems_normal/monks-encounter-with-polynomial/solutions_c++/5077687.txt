typedef long long int ll;
typedef unsigned long long ull;

template<class T1> void deb(T1 e1) {
    cout << e1 << endl;
}

template<class T1,class T2> void deb(T1 e1, T2 e2) {
    cout << e1 << " " << e2 << endl;
}

int a,b,c,k;

ull value(ull i) {
    return (a*i*i)+(b*i)+c;
}

int main() {
    ios_base::sync_with_stdio(false);
    ull hi,lo,mid;
    t() {
        cin>>a>>b>>c>>k;
        lo=0,hi=10000000005;
        while(lo<hi) {
            mid=(lo+hi)>>1;
            if(value(mid)>=k)
                hi=mid;
            else
                lo=mid+1;
        }
        deb(lo);
    }
    return 0;
}
