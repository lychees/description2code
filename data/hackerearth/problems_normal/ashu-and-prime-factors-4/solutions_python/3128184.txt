visited = [-1]*1000001
count = [0]*1000001

for i in range(2,1000001):
    if visited[i] != -1:
        continue
    count[i]+= 1
    for j in range(2*i, 1000001, i):
        if visited[j] == -1:
            count[i]+= 1
        visited[j] = 1

if __name__ == '__main__':
    t = int(raw_input())
    while t > 0 :
        t-= 1
        i = int(raw_input())
        print (count[i])
