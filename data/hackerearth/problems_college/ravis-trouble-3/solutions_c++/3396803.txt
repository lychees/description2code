#include <iostream>
using namespace std;

int main()
{
	int t, i;
	string s, p;
    cin>>t;
    while(t--)
    {
    	cin>>s>>p;
    	if(s.length()<p.length())
    		cout<<"NO"<<endl;
    	else
    	{
    		int j=0, c=0;
    		for(i=0;i<s.length()&&j<p.length();i++)
    		{
    			if(s[i]!=p[j])
    			{
    				c++;
    			}
    			else
    				j++;
    		}
    		if(j!=p.length())
    			cout<<"NO"<<endl;
    		else
    			cout<<"YES"<<endl;
    	}
    }
    return 0;
}
