#include<iostream>
using namespace std;
int max(int x,int y)
{
    return x>y?x:y;
}
int min(int x,int y)
{
    return x<y?x:y;
}
int main()
{
    int T,F,G,S,P,C,POW,sam,pal,champ,temp;
    cin>>T;
    while(T--)
    {
        temp=-1;
        cin>>F>>G>>S>>P>>C>>POW;
        for(sam=0;sam*100<=F;sam++)
        {
            for(pal=0;pal*125+100*sam<=F && pal*50<=G;pal++)
            {
                int flft=F-sam*100-pal*125;
                int glft=G-pal*50;
                champ=min(flft/50,glft/100);
                temp=max(temp,S*sam+P*pal+C*champ);
            }
        }
        if(POW<temp)
            cout<<temp-POW<<endl;
        else
            cout<<"-1"<<endl;
    }
    return 0;
}
