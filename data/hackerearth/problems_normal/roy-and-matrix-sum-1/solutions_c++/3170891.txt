#include <bits/stdc++.h>

using namespace std;

int main()
{
    int tc;
    cin >> tc;

    while(tc--){
    long int n;
    cin >> n;
    unsigned long long int sum = 0;

    for (int i = 0;i<= n;i++)
    {
        sum = sum + (i * (n - i));
    }

    cout << 2 * sum << endl;
    }
    return 0;
}
