#include <iostream>
using namespace std;

int main()
{
    int n,m;
    cin >> n >> m;
    
    int A[n][m];
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<m;j++)
    	{
    		cin >> A[i][j];
    	}
    }
    
    for(int k=0;k<n+m-1;k++)
    {
    	for(int i=0;i<=k;i++)
    	{
    		int j=k-i;
    		if(j<m && i<n)
    		cout << A[i][j] << " ";
    	}
    	cout << endl;
    }
}
