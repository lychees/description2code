#include<bits/stdc++.h>
using namespace std;
# define INF 1400
pair<int,pair<int,pair<int,string> > >   V[1001];
int main()
{
    cin.sync_with_stdio(false);
    string a,b,c;
    int n,i,ind=0,ia,ib,ic;
    cin>>n;
    map <string,int> mp;
    map<string,bool> mp1;
    for(i=0;i<n;i++)
    {
        cin>>a>>b>>c;
        if(mp1[a])
            ia=mp[a];
        else
        {
            mp1[a]=true;
            mp[a]=ind++;
            V[mp[a]]=make_pair(INF,make_pair(INF,make_pair(INF,a)));
            ia=mp[a];
        }
        if(mp1[b])
            ib=mp[b];
        else
        {
            mp1[b]=true;
            mp[b]=ind++;
            V[mp[b]]=make_pair(INF,make_pair(INF,make_pair(INF,b)));
            ib=mp[b];
        }
        if(mp1[c])
            ic=mp[c];
        else
        {
            mp1[c]=true;
            mp[c]=ind++;
            V[mp[c]]=make_pair(INF,make_pair(INF,make_pair(INF,c)));
            ic=mp[c];
        }
        V[ib].second.first--;
        V[ia].first--;
        V[ic].second.second.first--;
    }
    sort(V,V+ind);
    for(i=0;i<ind;i++)
        cout<<V[i].second.second.second<<" "<<1400-V[i].first<<" "<<1400-V[i].second.first<<" "<<1400-V[i].second.second.first<<endl;
    return 0;
}
