#include<iostream>
#include<cmath>
using namespace std;
#define ull unsigned long long
int main()
{
	int test;
	cin>>test;
	while(test--)
	{
		ull a,b,n;
		ull ans=0;
		cin>>n>>a>>b;
		ull x=(ull)sqrt(n);
		for(ull i=1;i<=x;i++)
		{
			if(n%i==0)
			{
				if(n/i!=i)
				{
					if(a>=n/i && b>=i)
					ans++;
					if(a>=i && b>=n/i)
					ans++;
				}
				else
				{
					if(a>=i && b>=i)
					ans++;
				}
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
