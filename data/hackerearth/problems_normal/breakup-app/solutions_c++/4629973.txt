#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin >> n;
	map < string , int > mp;
	n = n + 1;
	while (n--) {
		string line, word;
		getline(cin , line);
		//cout << line << endl;
		int flag = 1;
		if (line[0] == 'G')
			flag = 0;
		istringstream iss(line, istringstream::in);
		while( iss >> word ) {
				if (word[0] >= 48 && word[0] <= 57) {
					if (flag)
						mp[word] ++;
				   else
				   mp[word] += 2;
				}
		 }
	}
		 
		 int ans = -1;
		 string fin;
		 map <string, int > :: iterator it;
		 for (it = mp.begin(); it != mp.end(); ++it ) {
		 	if(it->second >= ans) {
		 		ans = it->second;
		 		fin = it->first;
		 	}
		 }
		 if (fin == "19" || fin == "20") {
		 	cout << "Date" << endl;
		 } else {
		 	cout<< "No Date" << endl;
		 }
	
    		return 0;
}
