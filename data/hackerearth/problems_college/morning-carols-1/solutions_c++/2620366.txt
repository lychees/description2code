lld modpow(lld a,lld n,lld temp){lld res=1;while(n>0){if(n&1)res=(res*a)%temp;a=(a*a)%temp;n>>=1;}return res%temp;} 
int res;
void recur (vector<lld> &a, int cur, int cnt, int mask, int n) {
  if (mask == (1<<n)-1) {
    res = min(res, cnt);
  }
  if (cur >= a.sz)
    return;
  recur(a, cur+1, cnt, mask, n);
  recur(a, cur+1, cnt+1, mask|a[cur], n);
  return;
}
int main()
{
  int t;
  cin>>t;
  while (t--) {
    int n, i, j;
    cin>>n;
    vector<lld> a;
    string s;
    rep (i, n) {
      cin>>s;
      if (a.sz == 0)
        a = vector<lld>(s.sz);
      rep (j, s.sz)
        if (s[j] == 'Y')
          a[j] += 1<<i;
    }
    res = INT_MAX;
    recur (a, 0, 0, 0, n);
    cout<<res<<"\n";
  }

	return 0;
