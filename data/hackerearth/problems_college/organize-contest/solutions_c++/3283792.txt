#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i, j, n, t, k;
    cin>>n;
    string x, s;
    k = 0;
    map < string, int > m;
    for(i = 0 ; i < n; i++) {
    	cin>>x;
    	cin>>t;
    	m[x] += t;
    	if(m[x] > k) {
    		k = m[x];
    		s = x;
	 }
    }
    cout<<s<<endl;
    return 0;
}
