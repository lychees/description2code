#include <bits/stdc++.h>
using namespace std;

int main() 
{
	int n, m, i, j, k;
	cin>>n>>m;
	map<string,int>mp;
	string s, d;
	int c;
	for(i=0;i<n;i++)
	{
		cin>>s;
		mp[s]=i;
	}
	int a[n+1][n+1];
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			if(i==j)
				a[i][j]=0;
			else
				a[i][j]=INT_MAX;
		}
	}
	for(i=0;i<m;i++)
	{
		cin>>s>>d;
		cin>>c;
		a[mp[s]][mp[d]]=c;
		a[mp[d]][mp[s]]=c;
	}
	for(k=0;k<n;k++)
	{
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				if(a[i][k]!=INT_MAX && a[k][j]!=INT_MAX)
					a[i][j]=min(a[i][j], a[i][k]+a[k][j]);
			}
		}
	}
	int q;
	cin>>q;
	while(q--)
	{
		cin>>s>>d;
		cout<<a[mp[s]][mp[d]]<<endl;
	}
	return 0;
}
