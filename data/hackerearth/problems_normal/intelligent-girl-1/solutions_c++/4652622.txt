#include <iostream>
#include <string>

using namespace std;

int main()
{
	string S;
	int ans[100001];
	
	cin >> S;
	
	for (int i = 0; i < 100001; i++)
		ans[i] = 0;
	
	for (int i = S.length()-1; i >= 0; i--)
	{
		if ((S[i]-'0') % 2)
			ans[i] = ans[i+1];
		else
			ans[i] = 1 + ans[i+1];
	}
	
	for (int i = 0; i < S.length(); i++)
		cout << ans[i] << " ";
		
	cout << endl;
	
    return 0;
}
