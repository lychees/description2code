#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int T, N;
    unsigned long long range = 1000000007;
    cin >> T;
    for(int i = 0; i < T; i++)
    {

        cin >> N;

        unsigned long long Max[N] ;

        for(int j = 0 ; j < N; j++)
        {
            cin >> Max[j];
        }

        sort(Max, Max + N);
        bool isIsponssible = false;
        for(int i = 0; i < N; i++)
        {
            if(Max[i] < i+1)
            {
                cout << "0\n";
                isIsponssible = true;
                break;
            }
        }
        if(isIsponssible){
        continue;
        }
        unsigned long long ans = 1;
        for(int i = 0; i < N; i++){

            unsigned long long temp = Max[i] - i;

            ans = (temp*ans)%range;

        }

        cout << ans << "\n";



    }


}
