#include<iostream>
#include<algorithm>
#include<list>
#include<iterator>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    while(n>0)
    {
    int min=a[0];
    for(int i=1;i<n;i++)
    {
        if(a[i]<min && a[i]!=0)
        {
            min=a[i];
        }
    }
    //cout<<min;
    cout<<n<<"\n";
    for(int i=0;i<n;i++)
    {
        a[i]=a[i]-min;
        if(a[i]<=0)
        {
            n--;
        }
    }
    }

}
