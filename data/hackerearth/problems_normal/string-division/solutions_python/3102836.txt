def solve():
	s = str(raw_input())
	n = len(s)
	
	if n >= 10:
		return "YES"
	
	for i in range(n):
		for j in range(i + 1, n):
			for k in range(j + 1, n):
				for t in range(k + 1, n):
					a = s[0:i + 1]
					b = s[i + 1:j + 1]
					c = s[j + 1:k + 1]
					d = s[k + 1: n]
					if a != b and a != c and a != d and b != c and c != d and b != d:
						return "YES"
	
	return "NO"

t = int(raw_input())
while (t):
	t -= 1
	print solve()
