#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
	int p, q;
	scanf("%d",&p);
	int a[12345] = {0},b[12345] = {0};
	for(int i = 0;i < p;++i){
		int f;
		scanf("%d",&f);
		a[f]++;
	}
	scanf("%d",&q);
	for(int i = 0;i < q;++i){
		int f;
		scanf("%d",&f);
		b[f]++;
	}
	for(int i = 1;i <= 10000;++i){
		if(a[i] != b[i])printf("%d ",i);
	}
    return 0;
}
