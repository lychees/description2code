l = raw_input()
l = list(l)
d = {}
max = 0
for x in l:
    if x not in d:
        d[x] = 1
    else:
        d[x] += 1
    if max < d[x]:
        max = d[x]
l1 = []
for x in d:
    if d[x] == max:
        l1.append(x)
ans = min(l1)
l2 = list("headrush")
l3 = list("cubingclub")
if ans in l2 and ans in l3:
    print "Thanks both"
elif ans in l2:
    print "Thanks head rush"
elif ans in l3:
    print "Thanks cubing club"
else:
    print "none"
