#include<iostream>
#include<string.h>
#include<stdio.h>
using namespace std;
int main()
{
    int T,flag=0,c=0;
    bool Store[80]={0};
    char S[50];
    cin>>T;
    for(int t=0;t<T;t++)
    {
        scanf("%s",&S);
        int l=strlen(S);
        flag=0;
        c=0;
        //cout<<S[l-1]<<"--\n";
        for(int i=0;i<l;i++)
        {
            switch(S[i])
            {
                case 'a':
                case 'e':
                case 'i':
                case 'o':
                case 'u':
                case 'y':
                    flag++;
                    c=0;
                    break;
                default:
                    flag--;
                    c++;
                    break;
            }
            if(c==3)
            {	//cout<<i<<"--"<<S[i]<<"\n";
            flag=-1;
                break;}
        }
        if(flag>=0)
        Store[t]=1;
    }
    for(int t=0;t<T;t++)
    if(Store[t]==1)
    cout<<"easy\n";
    else
    cout<<"hard\n";
    return 0;
}
