#include <iostream>
using namespace std;

int main()
{
    int t,n,sumeven,sumodd,count,oddrem,evenrem;
    cin>>t;
    while(t--){
    	cin>>n;
    	sumeven=0;
    	sumodd=1;
    	count=0;
    	while(n!=1){
    	if(n%2==0)
    	{
    		
    		sumeven=sumeven+n;
    		n=n/2;
    		count++;
    	}
    	else
    	{
    		
    		sumodd=sumodd+(n);
    		n=3*n+1;
    		count++;
    	}
    //	cout<<n<<" ";
    	}
    //	cout<<" "<<endl;
    //	cout<<count<<endl;
    	evenrem=(sumeven%count);
    //	cout<<evenrem<<endl;
    	oddrem=(sumodd%count);
    //	cout<<oddrem<<endl;
    	if(oddrem>evenrem)
    	cout<<"Odd Rules"<<endl;
    	else if(oddrem<evenrem)
    	cout<<"Even Rules"<<endl;
    	else
    	cout<<"Tie"<<endl;
    	
    }
    return 0;
}
