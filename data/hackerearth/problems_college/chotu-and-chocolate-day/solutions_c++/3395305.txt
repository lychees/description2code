#include<bits/stdc++.h>
using namespace std;
int main()
{
	
	int t;
	cin>>t;
	while(t--)
	{
	int n,k;
	cin>>n>>k;
	int c;
	int i,j;
	int a[n+1][n+1];
	if(k==1)
	{
		c=1;
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=n;j++)
			{
				a[i][j]=c;
				c++;
			}
		}
	}
	else
	{
		c=1;
		for(i=1;i<=n;i++)
		{
			for(j=1;j<k;j++)
			{
				a[i][j]=c;
				c++;
			}
		}
		for(i=1;i<=n;i++)
		{
			for(j=k;j<=n;j++)
			{
				a[i][j]=c;
				c++;
			}
		}
	}
	long long sum=0;
	for(i=1;i<=n;i++)
	{
		sum = sum + a[i][k];
	}
	cout<<sum<<"\n";
}
}
