#include <stdio.h>
#include <stdlib.h>
#define max  1000000007
int main()
{
    int t,a,b,c,d,i,j,k;
    long long int w[1005][1005];
    for (i=1;i<=1004;i++)
    {
        w[1][i]=1;
        w[i][1]=1;
    }
    for (i=2;i<=1004;i++)
        for (j=2;j<=1004;j++)
            w[i][j]=(w[i-1][j]+w[i][j-1]+w[i-1][j-1])%max;
    scanf("%d",&t);
    for (k=0;k<t;k++)
    {
        scanf("%d%d%d%d",&a,&b,&c,&d);
        printf("%lld\n",w[c-a+1][d-b+1]);
    }
    return 0;
}
