n = input()
planks = map(int,raw_input().split(' '))
area = reduce(lambda x,y: x*y, map(int,raw_input().split(' ')))


cost = 0
for i in planks:
    c = i*reduce(lambda x,y: x+y, map(int,list(str(i))))*area
    if i in range(1,11):
        c -= 0.25*c
    elif i in range(11,21):
        c -= 0.3*c
    else:
        c -= 0.2*c
    c = int(c)
    cost += c

print cost
