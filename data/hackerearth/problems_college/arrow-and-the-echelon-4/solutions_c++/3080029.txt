#include <iostream>
#include<string>
using namespace std;
#include<vector>



int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int x;
        cin>>x;
        if(x<=5)
        {
            if(x==5)
            {
            for(int i=1;i<=5;i++)
            {
                cout<<3;
            }

            }
            else if(x==3)
            {
                for(int i=1;i<=3;i++)
            {
                cout<<5;
            }
            }
            else
            cout<<"-1";

        }
        else
        {
            if(x%3==0)
            {
                for(int i=1;i<=x;i++)
            {
                cout<<5;
            }
            }
            else if(x%3==1)
            {


            		for(int i=1;i<=x-10;i++)
            {
                cout<<5;
            }

            for(int i=x-10+1;i<=x;i++)
            {
                cout<<3;
            }





            }
            else if(x%3==2)
            {

            for(int i=1;i<=x-5;i++)
            {
                cout<<5;
            }
            for(int i=1;i<=5;i++)
            {
                cout<<3;
            }

            }




        }
    cout<<endl;

    }


    return 0;
}
