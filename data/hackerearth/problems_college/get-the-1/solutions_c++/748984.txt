#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int get(int n){
    int r,s=0;
    while(n){
        r= n%10;
        s += r*r;
        n=n/10;
    }
    return s;
}

int main(){
    int t,n;
    vector<int> v;
    cin>>t;
    for(int i=0;i<t;i++){
        cin>>n;
        while(n){
            n = get(n);
            if(n==1 || binary_search(v.begin(), v.end(), n) || n==19)
                break;
            v.push_back(n);
        }
        if(n==1)
            cout<<"Case #"<<i+1<<":YES\n";
        else
            cout<<"Case #"<<i+1<<":NO\n";
    }

    return 0;
}
