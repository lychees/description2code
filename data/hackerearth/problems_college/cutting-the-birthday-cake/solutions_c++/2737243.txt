#include <iostream>
#include <math.h>
using namespace std;


bool possible(int R, int A, int B){
    
    	double sq;
    	
    	sq= sqrt(A*A + B*B);
    	
    	if(R*2 < sq)
    		return false;
    	
    	return true;
    	
    }


int main()
{
    int T=0,t=0;
    
    cin>>T;
    
    
    
    while(t<T){
    	t++;
    	int R=0,A=0,B=0;
    	cin>>R>>A>>B;
    	
    	if(possible(R,A,B))
    		cout<<"ALICE\n";
    	else
    		cout<<"EQUAL\n";
    	
    }
    
    
    
    return 0;
}
