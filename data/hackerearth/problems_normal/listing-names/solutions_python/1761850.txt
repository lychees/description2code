import sys
ip=[]
for line in sys.stdin:
    ip.append(line)
for i in range(0,len(ip)):
    ip[i]=ip[i].rstrip('\n')
string_list=ip[1:len(ip)]
temp=[]
dict_count={}
for word in string_list:
    if word in temp:
        count=count+1
    else:
        temp.append(word)
        count=1
    dict_count[word]=count
