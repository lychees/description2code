#include<bits/stdc++.h>
using namespace std;
#define in(x) scanf("%d",&x)
int main()
{
    int t,j,x,z,a,b,i;
    long long int mx,mn;
    in(t);
    while(t--)
    {
        in(j);
        vector<int> n,p;
        z=0;
        for(i=0;i<j;i++)
        {
            in(x);
            if(x>0)
                p.push_back(x);
            else if(x<0)
                n.push_back(x);
            else if(x==0)
                z=1;
        }
        sort(p.begin(),p.end());
        reverse(p.begin(),p.end());
        sort(n.begin(),n.end());
        a=p.size(),b=n.size();
        if(a!=0 && b!=0)
        {
            mx=1;
            for(i=0;i<a;i++)
                mx=mx*p[i];
            mn=mx;
            if(b%2==0)
            {
                for(i=0;i<b;i++)
                    mx=mx*n[i],mn=mn*n[i];
                mn=mn/n[b-1];
            }
            else if(b%2==1)
            {
                for(i=0;i<b;i++)
                    mx=mx*n[i],mn=mn*n[i];
                mx=mx/n[b-1];
            }
        }
        else if(a!=0 && b==0)
        {
            mx=1;
            for(i=0;i<a;i++)
                mx=mx*p[i];
            if(z==0)
                mn=p[a-1];
            else if(z==1)
                mn=0;
        }
        else if(a==0 && b!=0)
        {
            mn=1,mx=1;
            if(b==1 && z==0)
                mx=n[0];
            if(b==1 && z==1)
                mx=0;
            if(b%2==0)
            {
                for(i=0;i<b;i++)
                    mx=mx*n[i],mn=mn*n[i];
                mn=mn/n[b-1];
            }
            else if(b%2==1)
            {
                for(i=0;i<b;i++)
                    mx=mx*n[i],mn=mn*n[i];
                mx=mx/n[b-1];
            }
        }
        else if(a==0 && b==0)
        {
            mx=0,mn=0;
        }
        cout << mx << " " << mn << endl;
    }
    return 0;
}
