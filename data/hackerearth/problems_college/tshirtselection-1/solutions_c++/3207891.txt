#include <iostream>

#include <bits/stdc++.h>
using namespace std ;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
	vector<int > v(1000001,0);
	map<int,int> m;	
	int n;
	scanf("%d",&n);
	int i,j,start=0,end=0;
	int max1=0;
	for(i=0;i<=n-1;i++) 
         scanf("%d",&v[i]);
	for(i=0,j=0;j<n;j++)
	{
		m[v[j]]++;
		if(m[v[j]]>1)
		{
			while(m[v[j]]>1 && i<=j)
			{
				m[v[i]]--;
				i++;
			}
		}
		if(max1<j-i+1)
		{
			start = i;
			end = j;
			max1 = j-i+1;
		}
	}
	cout << start+1 << " " << end+1 <<  endl;
}
return 0;
}
