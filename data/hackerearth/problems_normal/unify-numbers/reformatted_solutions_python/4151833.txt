import math

t = int(raw_input())
while t:
    t = t-1
    n = int(raw_input())
    a = sorted(map(int, raw_input().split()))
    m = 0
    x = n/2
    if n % 2 == 0:
        m = int(math.ceil((a[x-1]+a[x])/float(2)))
    else:
        m = a[x]
    cost = 0
    for i in a:
        cost += abs(m-i)
    print cost
