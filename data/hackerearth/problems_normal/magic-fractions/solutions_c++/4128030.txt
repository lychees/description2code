ll ans[1000];
vector<int>prime;
bool vis[1000];
vector<int>::iterator it;
void solve()
{
    prime.push_back(2);
    for(int i=2;i<=504;i+=2)
        vis[i]=true;
    for(int i=3;i<=504;i+=2)
    {
        if(!vis[i])
        {
            prime.push_back(i);
            vis[i]=true;
            for(int j=i;i*j<=504;j+=2)
            vis[i*j]=true;
        }
    }
    for(int i=2;i<=504;i++)
    {
        it=lower_bound(prime.begin(),prime.end(),i);        
        ans[i]=(it-prime.begin())+1;
         if(!binary_search(prime.begin(),prime.end(),i)) 
            ans[i]--;
    }
    for(int i=2;i<=504;i++)
    {
        ans[i]=powmod(2,ans[i]-1);
    }
    for(int i=2;i<=504;i++)
        ans[i]=ans[i-1]+ans[i];
}
