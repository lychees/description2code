#include<iostream>
using namespace std;
int main()
{
    long int n,i;
    int count=0,max=0;
    cin>>n;
    long int a[n];
    //cin>>a[0];
    if (n==1)
    cout<<n;
    else
    {for(long int i=0;i<n;i++)
    {
        cin>>a[i];
        if((a[i]-a[i-1]==1 )|| (a[i-1]-a[i]==1))
            {count++;
            if(count>max)
            max=count;
            }
        else
            count=0;
    }
    cout<<max+1;
    }
}
