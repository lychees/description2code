from __future__ import print_function

t = input()

for i in range(t):
    s = raw_input().split('.')
    old_count = 8
    new_count = 4
    vowels = ['a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U']
    k = len(filter(lambda x: x not in vowels, s[1]))
    print(new_count+k, end='')
    print('/', end='')
    print(old_count+len(s[1]))
