n = input()
for i in range(n):
    s = raw_input()
    s1 = raw_input()
    sl = set(list(s))
    s1l = set(list(s1))
    s2 = set(sl & s1l)
    if len(s2) > 0:
        print "YES"
    else:
        print "NO"
