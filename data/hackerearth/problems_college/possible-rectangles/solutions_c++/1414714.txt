#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

int main()
{
    int test,n,m;long long area;
    scanf("%d",&test);
    while(test--){
    	scanf("%d%d%lld",&n,&m,&area);
    	long long res = 0;
        for (long long i = 1; i*i <= area ; i++) {
	        long long d = area / i;
	        if (d*i != area)continue;
	        if (i == d && (i < n && i < m))res += (long)(n - i)*(m - i);
	        else{ 
	        	if(d < n && m > i)res += (long long)(n - d)*(m - i);
	        	if(d < m && i < n)res += (long long)(n - i)*(m - d);
	        
	        }
	    }
        printf("%lld\n",res);
    }
    return 0;
}
