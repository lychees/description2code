c = 0
for i in text:
    if c == 0:
        if i == 'l':
            c += 1
    elif c == 1:
        if i == 'o':
            c += 1
    elif c == 2:
        if i == 'v':
            c += 1
    elif c == 3:
        if i == 'e':
            c += 1
    else:
        break
if c == 4:
