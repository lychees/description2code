#include<bits/stdc++.h>
using namespace std;
int rev(int n)
{
	int val=0;
	while(n)
	{
		val = val*10 + n%10;
		n /= 10;
	}
	return val;
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	long double p,e;
    	cin>>p>>e;
    	long double np,ep;
    	ep = p*e/100+p;
    	np = rev(ep);
    	long double pr;
    	pr = (p-np)/p;
    	long double ans;
    	ans = ep + ep*pr;
    	int as = ans;
    	cout<<as<<endl;
    }
    return 0;
}
