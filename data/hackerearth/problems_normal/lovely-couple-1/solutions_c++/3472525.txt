#include <bits/stdc++.h>
using namespace std;
bool check(int x){
	
	set<int> s;
	while(x%2==0)
	{
		x=x/2;
		
		s.insert(2);
	}
	
	for(int i=3;i<=sqrt(x);i=i+2)
	{
		
		while(x%i==0)
		{
			x=x/i;
			s.insert(i);
		}
	
	}
	if(x>2)
	s.insert(x);
	
	
	int num=s.size();
	
	bool flag=true;
	
	for(int i=2;i<=sqrt(num);i++)
	{
		if(num%i==0)
		{
			flag=false;
			break;
		}
	}
	
	if(flag && num!=1)
	return true;
	else
	return false;
	
}

int gcd(int p,int q)
{
	if(q==0)
	return p;
	else
	return gcd(q,p%q);
}
int main()
{
    int t,a,b;
    cin>>t;
    
    while(t--){
    	
    	cin>>a>>b;
    	
    	int lcm=(a*b)/gcd(max(a,b),min(a,b));
    	
    	if(check(lcm))
    	 cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
    }
    
    return 0;
}
