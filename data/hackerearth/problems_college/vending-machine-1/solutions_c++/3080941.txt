#include <iostream>
using namespace std;
int counter;
int sum(int counter){
    int m=5,ma=1;
    for(int i=0;i<counter;i++){
        ma=ma*2;
        m=m+5*ma;
    }
    return m;
}
int xn(int counter){
    int y=1;
    for(int i=0;i<counter;++i){
        y=y*2;
    }
    return y;
}
int main()
{
    int n;
    cin>>n;
    int x=5,y=1,res=0;
    if(n<6){
        res=n;
    }
    else
    {
        counter=0;
        while(n>sum(counter)){
            counter++;
        }
        counter=counter-1;
        n=n-sum(counter);
        res=(n-1)/xn(counter+1)+1;

    }
    if(res==1) cout<<"Sheldon"<<endl;
    if(res==2) cout<<"Leonard"<<endl;
    if(res==3) cout<<"Penny"<<endl;
    if(res==4) cout<<"Rajesh"<<endl;
    if(res==5) cout<<"Howard"<<endl;
    return 0;
}
