Mem={}
def Find(N):
	if N<=1:
		return 0
	if N in Mem:
		return Mem[N]
	a1,a2,a3=N//2,N//3,N//4
	max1=Find(a1)
	max2=Find(a2)
	max3=Find(a3)
	if N<=max1+max2+max3:
		Mem.update({N:max1+max2+max3})
		return Mem[N]
	else:
		Mem.update({N:N})
		return N

def solve():
	try:
		while True:
			N=int(raw_input())
			ans=Find(N)
			print ans
	except:
		return
solve()
