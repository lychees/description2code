#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int mat[1010][1010],r[1010],c[1010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int n,m;
	cin>>n>>m;
	ll int i,j,t;
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		cin>>mat[i][j];
	}
	for(i=0;i<n;i++)
	{
		t=0;
		for(j=0;j<m;j++)
		t+=mat[i][j];
		r[i]=t;
	}
	for(j=0;j<m;j++)
	{
		t=0;
		for(i=0;i<n;i++)
		{
			t+=mat[i][j];
		}
		c[j]=t;
	}
	ll int ans=0;
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			ans=max(ans,r[i]^c[j]);
		}
	}
	cout<<ans<<endl;
	return 0;
}
