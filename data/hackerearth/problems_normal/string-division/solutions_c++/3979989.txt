#include<bits/stdc++.h>
using namespace std;
int a[26];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        memset(a,0,sizeof(a));
        string s;
        cin>>s;
        int u=0;
        for(int i=0;i<s.length();i++)
        {
            if(a[s[i]-97]==0)
            {
                u++;
            }
            a[s[i]-97]++;
        }
        if(s.length()<4)
            printf("NO\n");
        else if(s.length()==4)
        {
            if(u==4)
                printf("YES\n");
            else
                printf("NO\n");
        }
        else if(s.length()==5)
        {
            if(u>=3)
                printf("YES\n");
            else
                printf("NO\n");
        }
        else if(s.length()==6)
        {
            if(u<2)
                printf("NO\n");
            else if(u==2){
            bool flag=false;
            for(int i=0;i<26;i++)
            {
                if(a[i]==4)
                {
                    flag=true;
                    break;
                }
            }
            if(flag)
                printf("YES\n");
            else
                printf("NO\n");
            }
            else
                printf("YES\n");
        }
        else if(s.length()<10)
        {
            if(u>=2)
                printf("YES\n");
            else
                printf("NO\n");
        }
        else
        {
            printf("YES\n");
        }
    }
    return 0;
}
