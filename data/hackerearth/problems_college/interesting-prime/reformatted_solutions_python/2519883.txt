# print 'Hello World!'
t = int(raw_input())
for i in range(t):
    c = 0
    x = int(raw_input())
    for j in xrange(2, int(x ** 0.5)+1):
        if (x % j == 0):
            if (j*j == x):
                c += 1
            else:
                c += 2
    if (x == 1):
        print("0")
    else:
        print(c+1)
