#include <iostream>
#include <string>
#include <algorithm>
#include <vector>
using namespace std;

int main() {
	int t;
	 std::vector<string> pin;
	 string p;
	int n;
	cin>>t;
	while(t--) {
		cin>>n;
		std::size_t found = std::string::npos;
		pin.clear();
		
		for(int i=0;i<n;i++) {
			cin>>p;
			pin.push_back(p);
		}
		
		std::sort(pin.begin(), pin.end());
		string s = pin[0];
		for (int i = 1; i < pin.size(); ++i)  {
			found = pin[i].find(s);
			if (found == 0) {
				//cout<<s<<" "<<pin[i]<<"\n";
				break;
			}
				
			s = pin[i];
		}
		
		if (found == 0)
		cout<<"NO\n";
		else
		cout<<"YES\n";
     }
	}
