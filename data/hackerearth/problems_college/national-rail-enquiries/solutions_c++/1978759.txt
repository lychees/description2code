#include <iostream>
using namespace std;
long long gcd(long long a, long long  b){
	return (b==0 ?a:gcd(b,a%b));
}
long long fact(long long n){
	if(n==0 || n==1)
		return 1;
	else
	 return (n)*(fact(n-1));
	
}
 
int main()
{
	long t, j=0;
	cin>>t;
	while(t--){
  	long long n,deno;
  	cin>>n;
  	if(n>=110)
  		deno =0;
  	else
  	deno = fact(n-1)%110;
  	if(deno>0 && gcd(deno,n)==1)
  	printf("Case %d:Yes\n",++j);
  	else
  	printf("Case %d:No\n",++j);
  	
    
	}
    return 0;
}
