#include <iostream>
#include <math.h>
using namespace std;

bool prime(unsigned long long int a)
{
    unsigned long long int i;
    if(a==1)
        return false;
    for(i=2;i<=sqrt(a);i++)
    {
        if(a%i==0 )
        return false;
    }
     return true;
}

int main()
{
	unsigned long long int n;
	cin>>n;
	while(n--)
	{
		unsigned long long int x;
		cin>>x;
		if(prime(x)==true)
		cout<<"NO"<<endl;
		else if(prime(sqrt(x))==true)
		{
			unsigned long long int k = sqrt(x);
			if(x==k*k)
			cout<<"NO"<<endl;
			else cout<<"YES"<<endl;
		}
		else if(x==1)
		cout<<"NO"<<endl;
		else cout<<"YES"<<endl;
	}
    return 0;
}
