#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>

using namespace std;

int main()
{
	int ar[100005], n, i, j, c=0;
	bool mark[100005];
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>ar[i];
	}
	sort(ar,ar+n);
	memset(mark, true, sizeof(mark));
	for(i=0;i<n;i++)
	{
		if(mark[i])
		{
			c++;
			for(j=i;ar[j]<=(ar[i]+4);j++)
			{
				mark[j]=false;
			}
		}
	}
    cout << c << endl;
    return 0;
}
