N = int(raw_input())

dictionary = []

dictionary.append([0,str(raw_input())])

for _ in range(N-1):
	(num,tail) = str(raw_input()).split()
	dictionary.append([int(num),tail])

reqd = dictionary[N-1][0]
tail = dictionary[N-1][1]

i = N-2
while (reqd > 0):
	num = dictionary[i][0]
	if reqd > num:
		tail = dictionary[i][1][:(reqd-num)] + tail
		reqd = num
	i -= 1

print tail
