#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

typedef struct
{
	float start;
	float end;
} Duration;

vector <Duration> schedule;

bool compare(Duration a, Duration b)
{
	return a.start < b.start;
}

int main()
{
	int n;
	char in[100];
	Duration slot;
	cin >> n;
	
	for (int i = 0; i < n; i++)
	{
		cin >> in;
		
		in[2] = 0;
		in[5] = 0;
		in[8] = 0;
		
		slot.start = atoi(in);
		slot.start += atoi(in + 3)*1.0/100;
		slot.end = atoi(in + 6);
		slot.end += atoi(in + 9)*1.0/100;
		
		if (slot.end == 0)
			slot.end = 24;
			
		//cout << "start = " << slot.start << " end = " << slot.end << endl;

		schedule.push_back(slot);
	}
	
	sort(schedule.begin(), schedule.end(), compare);
	/*cout << "Final list:" << endl;
	for (int i = 0; i < schedule.size(); i++)
		cout << "start = " << schedule[i].start << " end = " << schedule[i].end << endl;*/
		
	for (int i = 0; i < schedule.size() - 1; i++)
	{
		if (schedule[i].end > schedule[i + 1].start)
		{
			cout << "Will need a moderator!" << endl;
			return 0;
		}
	}
	
	cout << "Who needs a moderator?" << endl;
	return 0;
}
