#include <bits/stdc++.h>
    using namespace std;
     
    int main()
    {
    	int t=0,n,m,l,h,arr[100001];
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		cin>>arr[i];
    	}
    	cin>>m;
    	while(m--)
    	{
    		cin>>l>>h;
    		t=0;
    		for(int j=l-1;j<h;j++)
    		{
    			t+=arr[j];
    		}
    		cout<<t<<endl;
    	}
    	
    }
