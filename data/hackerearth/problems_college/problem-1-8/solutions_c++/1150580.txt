#include <iostream>
#include <string>
using namespace std;

struct node
{
	int val;
	struct node *children[26];
};

struct node * newNode()
{
	struct node* temp = new struct node();
	temp->val =0;
	
	for(int i=0;i<25;i++)
		temp->children[i]=NULL;
		
	return temp;
};

void insert(struct node* root,string s)
{
	int i,index;
	struct node *p;
	
	p=root;
	
	for(i=0;i<s.length();i++)
	{
		index = s[i]-'a';
		
		if(p->children[index]==NULL)
		{
			p->children[index]= newNode();
		}
		p=p->children[index];
	}
	
	p->val =1;
}

int main()
{
    int i,index,N;
    string s,input;
    struct node*p;
    
    struct node *root  = newNode();
    
    cin>>s;
    
    cin>>N;
    
    for(i=0;i<N;i++)
    {
    	cin>>input;
    	insert(root,input);
    }
   	
   	string out=s;
   	
   	p=root;
   	for(i=0;i<s.length();i++)
   	{
   		index = s[i]-'a';
   		if(p->children[index]==NULL)
   			break;
   		p=p->children[index];
   	}
   	
   	if(i!=s.length())
   		cout<<out<<endl;
   	else
   	{
   		while(p->val!=1)
   		{
   			for(i=0;i<26;i++)
   				if(p->children[i]!=NULL)
   					break;
   			
   			p=p->children[i];
   			out = out+ (char)(i+97);		
   		}
   		
   		cout<<out<<endl;
   		
   		
   	}
    
    return 0;
}
