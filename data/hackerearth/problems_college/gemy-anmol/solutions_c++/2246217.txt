#include<iostream>
#include<algorithm>
using namespace std;
#define ull unsigned long long
ull arr[10000];
int main()
{
	int test;
	cin>>test;
	while(test--)
	{
		ull n,req;
		cin>>n>>req;
		ull ans=-1;
		ull tot=0;
		for(int i=0;i<n;i++)
		{
			cin>>arr[i];
			tot+=arr[i];
		}
		if(tot<req)
		cout<<-1<<endl;
		else
		{
			sort(arr,arr+n,greater<int>());
			ans=0,tot=0;
			for(int i=0;i<n;i++)
			{
				if(tot>=req)
				break;
				ans++;
				tot+=arr[i];
				if(tot>=req)
				break;
			}
			cout<<ans<<endl;
		}
	}
	return 0;
}
