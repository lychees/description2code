#include <bits/stdc++.h>
#define ll unsigned long long int
using namespace std;
int main()
{
	ll t;
	ll s,d;
	ll a,b;
	cin>>t;
	
	while(t--)
	{
		cin>>s>>d;
		if(d>s)
		{
			cout<<"impossible\n";
			continue;
		}
		a=(s+d)/2;
		b=(s-d)/2;
		
		if(a+b!=s)
			cout<<"impossible\n";
		else if(a>=b)
			cout<<a<<" "<<b<<"\n";
		else
			cout<<b<<" "<<a<<"\n";
			
	}
	
	return 0;
}
