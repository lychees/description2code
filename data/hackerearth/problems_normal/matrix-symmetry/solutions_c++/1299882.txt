#include <iostream>
using namespace std;

int main()
{ int h=0,v=0,flag=0,t,i,j,n=0;
char a[100][100];
    cin>>t;
    while(t--)
    {h=v=flag=0;
        cin>>n;
        for(i=0;i<n;i++){
        	for(j=0;j<n;j++)cin>>a[i][j];
        }
        for(i=0;i<n/2;i++)
        {
            j=0;
            while(j<n)
            {
                if(a[i][j]!=a[n-i-1][j])
                {
                	
                    flag=1;
                    break;
                }
                j++;
            }
            if(flag==1)break;
        }
        if(flag==1)h=0;
        else h=1;
        flag=0;
        for(j=0;j<n/2;j++)
        {
            i=0;
            while(i<n)
            {
                if(a[i][j]!=a[i][n-j-1])
                {
                    flag=1;
                    break;
                }
                i++;
            }
            if(flag==1)break;
        }
        if(flag==1)v=0;
        else v=1;
        
        if(h==0 && v==0)cout<<"NO";
        else if(h== 1 && v==1)cout<<"BOTH";
        else if(h == 1)cout<<"HORIZONTAL";
        else cout<<"VERTICAL";
        cout<<"\n";
    }
    return 0;
}
