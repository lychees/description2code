#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	
	while (t--) {
		string s1, s2;
		cin >> s1 >> s2;
		
		if (s1.length() == s2.length()) {
			int c = 0;
			
			for (int i = 0; i < s1.length(); i++) {
				c += (s1[i] != s2[i]);
			}
			
			cout << c << "\n";
		} else if (s1.length() < s2.length()) {
			int d = 50005;
			
			for (int i = 0; i <= s2.length() - s1.length(); i++) {
				int c = 0;
				
				for (int j = 0; j < s1.length(); j++) {
					c += (s1[j] != s2[i + j]);
				}
				
				d = min(c, d);
			}
			
			cout << d << "\n";
		} else {
			int d = 50005;
			
			for (int i = 0; i <= s1.length() - s2.length(); i++) {
				int c = 0;
				
				for (int j = 0; j < s2.length(); j++) {
					c += (s2[j] != s1[i + j]);
				}
				
				d = min(c, d);
			}
			
			cout << d << "\n";
		}
	}
	
    return 0;
}
