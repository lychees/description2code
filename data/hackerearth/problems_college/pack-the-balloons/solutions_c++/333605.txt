using namespace std;
#include<iostream>
int main()
{
    int t;
    cin>>t;
    unsigned long long a,b,c,mn;
    while(t>0)
    {
        cin>>a;
        cin>>b;
        cin>>c;
        mn=min(a,b);
        //cout<<mn<<endl;
        mn=min(mn,c);
        //cout<<mn<<endl;
        if(mn%2==0)
            cout<<mn/2<<".0"<<endl;
        else cout<<mn/2<<".5"<<endl;
        t--;
    }
}
