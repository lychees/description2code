n = int(input())
while n > 0:
    s = raw_input()
    if len(s) <= 3:
        print s
    else:
        s = s[::-1]
        t = s[:3]
        t = t+','
        c = 0
        for i in range(3, len(s)):
            t = t+s[i]
            c = c+1
            if c == 2:
                t = t+','
                c = 0
        if len(s) % 2 != 0:
            t = t[:-1]
        print t[::-1]
    n = n-1
