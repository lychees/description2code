#include <bits/stdc++.h>
using namespace std;

#define pb push_back
#define mp make_pair
#define ff first
#define ss second

typedef long long ll;
typedef pair<int,int> PII;

const ll MOD = 1000000007;

int modinv(ll a)
{
  int b=MOD,t,q,x=0,y=1;
  while(a>1)
    {
      q=a/b;
      t=b,b=a%b,a=t;
      t=x,x=y-q*x,y=t;
    };
  if(y<0)
    y+=MOD;
  return y;
};


int main()
{
  ios::sync_with_stdio(false);
  int N,L,R,i;
  cin >> N >> L >> R;

  ll F[101],D[101],C[N+1];
  F[0]=1;//factorial
  for(i=1;i<=100;i++)
    {
      F[i] = i*F[i-1];
      F[i]%=MOD;
    }

  D[0]=1,D[1]=0;//derangements.
  for(i=2;i<=100;i++)
    {
      D[i] = D[i-1]+D[i-2];
      D[i]%=MOD;
      D[i] *= (i-1);
      D[i]%=MOD;
    };

  C[0] = 1;
  for(i=1;i<=N;i++)
    {
      C[i] = C[i-1]*(N-i+1);
      C[i]%=MOD;
      C[i] *= modinv(i);
      C[i]%=MOD;
    };

  ll P = 0,Q = modinv(F[N]);
  for(i=L;i<=R;i++)
    {
      P += D[i]*C[i];
      P%=MOD;
    }

  cout << (P*Q)%MOD << endl;
  return 0;
}
