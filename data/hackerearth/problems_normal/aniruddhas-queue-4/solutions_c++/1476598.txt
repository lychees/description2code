#include <iostream>
#include <algorithm>
#include <set>
#include <string>
#include <vector>
using namespace std;
#define LL long long 
#define MAX 100005
LL arr[MAX];
int main()
{
	int t;
	cin >> t;
	while (t--) {
		int n;
		cin >> n;
		for (int i = 0; i < n; i++) {
			cin >> arr[i];
		}
		LL sum = 0;
		LL dest;
		bool flag = false;
		cin >> dest;
		int i;
		for (i = 0; i < n; i++) {
			sum += arr[i];
			if (sum >= dest) {
				flag = true;
				break;
			}
		}
		if (flag) {
			printf("%d\n", i + 1);
		}
		else {
			dest = dest % sum;
			sum = 0;
			for (i = 0; i < n; i++) {
				sum += arr[i];
				if (sum >= dest) {
					break;
				}
			}
			if (dest != 0) {
				printf("%d\n", i + 1);
			}
			else {
				long long ans = 0;
				for (i = 0; i < n; i++) {
					if (arr[i] > 0) {
						ans = i;
					}
				}
				printf("%d\n", ans + 1);
			}
		}
	}
	return 0;
}
