#include<iostream>
#include<map>
#include<algorithm>
#include<cstdio>
#include<cstring>
#define LL long long
using namespace std;
map<string,LL> MyMap;
char Ans[1000001]="";
int main()
{
    int n,m;
    scanf(" %d %d",&n,&m);
    int i;
    char name[1000001];
    LL pt;
    LL mx=-1;
    for(i=0;i<m;i++)
    {
        scanf(" %s %lld",name,&pt);
        if(MyMap[name]+pt>mx)
        {
            mx=MyMap[name]+pt;
            strcpy(Ans,name);
        }
        MyMap[name]+=pt;
    }
    // We ROCK :P
    // AkankShaival
    printf("%s\n",Ans);
    return 0;
}
