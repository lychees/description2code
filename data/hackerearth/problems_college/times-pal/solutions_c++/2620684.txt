bool palin(string a)
{
    int i,j,k,f=1;
    int n=a.size();
    for(i=0;i<n/2;i++)
    {
        if(a[i]!=a[n-i-1])
        {
            f=0;
            break;
        }
    }
    if(f==0)
        return false;
    else
        return true;
}

    int t,i,j,n;

            c="";
            c=a.substr(0,i)+b+a.substr(i,n);
            //cout<<c<<endl;

    }

}
