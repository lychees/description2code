#include <bits/stdc++.h>
using namespace std;
string a,b;
int carr()
{
	reverse(a.begin(),a.end());
	reverse(b.begin(),b.end());
	int cnt=0,crr=0;
	int i;
	for( i=0;i<a.size()&&i<b.size();i++){
		int t=a[i]+b[i]-'0'-'0';
		t+=crr;
		crr=t/10;
		if(crr>0)
		cnt++;
	}
	while(i<a.size()){
		int t=a[i]-'0'+crr;
		crr=t/10;
		if(crr>0);
			cnt++;
		i++;
	}
	
	while(i<b.size()){
		int t=b[i]-'0'+crr;
		crr=t/10;
		if(crr>0);
			cnt++;
		i++;
	}
	return cnt;
}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	cin>>a>>b;
    	int lll=carr();
    	if(lll==0){
    		cout<<"No ";
    	}
    	else
    		cout<<lll<<" ";
    	if(lll<2)
    	cout<<"carry operation"<<endl;
    	else
    	    	cout<<"carry operations"<<endl;

    }
    return 0;
}
