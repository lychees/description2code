
f = []
x = 1
f.append(x)
f.append(x)
for i in range(2, 55):
    x = x*i
    f.append(x)


for _ in range(input()):
    m, n = [int(i) for i in raw_input().strip().split()]

    if n > m+1:
        print -1
        continue

    res = f[m+1]
    res1 = f[m]

    print(res1*res/(f[m+1-n])) % 1000000007
