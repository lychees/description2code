#include<iostream>
#include <stdio.h>
#include <stdlib.h>
using namespace std ;
int comp2(const void *a, const void *b)
{
	return *(int *)a - *(int *)b;
}

int comp(const void *pa, const void *pb)
{
	int *a = (int *) pa;
	int *b = (int *) pb;

	return a[1] - b[1];
}
int gcd (int a,  int b)
{
	return b == 0 ? a : gcd (b, a % b);
}

int main()
{
	int t ;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int ar[n];
		int i , j , k , l , m ;
		for(i=0;i<n;i++)
		{
			scanf("%d",&ar[i]);
		}
		int p,q ;
		scanf("%d%d",&p,&q);
			int hash[n][2];
		j=0;
		for (i = 0; i < n; ++i)
		 {
			if (gcd(p, ar[i]) != 1) 
			{
				hash[j][0] = ar[i];
				hash[j][1] = abs(p - ar[i]);
				j++;
			}
		}

		if (j < q) 
		{
			printf("-1\n");
		}
		 else 
		{
			qsort(hash, j, sizeof(hash[0]), comp);
			int hash2[q];
			for (i = 0; i < q; ++i)
			 {
				hash2[i] = hash[i][0];
			}
			qsort(hash2, q, sizeof(hash2[0]), comp2);
			for (i = 0; i < q; ++i) 
			{
				printf("%d ", hash2[i]);
			}
			printf("\n");
		}
	}
	return 0 ;
}
