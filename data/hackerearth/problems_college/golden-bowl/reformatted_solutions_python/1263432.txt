def f(val, t):
    return val*(val-1)-(t*(t-1))/2


def solve(t):
    lo = 0
    hi = t
    while lo <= hi:
        mid = (lo+hi)/2
        if (f(mid, t) < 0):
            lo = mid+1
        elif (f(mid, t) == 0):
            print(mid)
            return
        else:
            hi = mid-1


t = int(raw_input())
solve(t)
