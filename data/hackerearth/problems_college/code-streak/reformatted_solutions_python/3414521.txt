import sys

for __ in range(input()):
    n = input()
    has, ans, temp = {}, 0, 0
    for _ in range(n):
        a, b = map(int, sys.stdin.readline().split())
        if has.has_key(a):
            if b == 0:
                ans = max(ans, temp)
                temp = 0
        else:
            if b == 1:
                has[a] = b
                temp += 1
            else:
                ans = max(ans, temp)
                temp = 0
    print max(ans, temp)
