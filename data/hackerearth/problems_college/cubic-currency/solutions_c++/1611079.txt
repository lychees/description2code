#include <stdio.h>
long long int A[10005][215];
int main()
{
	int i,x,T,N,j;
	for(i=0;i<=213;i++)
	A[0][i]=1;
	
	for(i=1;i<=10000;i++)
	{
		for(j=1;j<=213;j++)
		{
			x=j*j*j;
			if(x<=i)
			{
				A[i][j]+=A[i-x][j];
				
			}
			A[i][j]+=A[i][j-1];
			
		}
	}
	scanf("%d",&T);
	while(T--)
	{
		scanf("%d",&N);
		printf("%lld\n",A[N][213]);
	}
	return 0;
	
}
