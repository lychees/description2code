#include <iostream>
#define ll long long
using namespace std;
int main()
{
	ll t, i, j, k, a1, x, co, y, a2, b1, b2, c, tp;
    cin>>t;
    for(i=1;i<=t;i++)
    {
    	c=0;
    	cin>>a1>>a2>>b1>>b2;
    	for(j=a1;j<=a2;j++)
    	{
    		for(k=b1;k<=b2;k++)
    		{
    			if(k!=j)
    			{
    				co=0;
    				x=k;y=j;
    				if(x<y)
    				{
    					tp=x;x=y;y=tp;
    				}
    				while((x/y)==1)
    				{
    					x=x%y;
    					if(x<y)
	    				{
	    					tp=x;x=y;y=tp;
	    				}
    					co++;
    				}
    				if(co%2==0) c++;
    			}
    		}
    	}
    	cout<<"Case #"<<i<<": "<<c<<endl;
    }
    return 0;
}
