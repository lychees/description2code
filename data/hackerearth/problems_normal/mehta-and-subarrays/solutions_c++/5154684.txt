#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int n;
vector<ll> a;

ll query(int l, int r) {
	return a[r] - (l ? a[l - 1] : 0);
}

int cnt(int len) {
	int ans = 0;
	for (int i = 0; i < n - len + 1; ++i) {
		ans += (query(i, i + len - 1) >= 0);
	}
	return ans;
}

int main() {
	scanf("%d", &n);
	a.resize(n);
	for (int i = 0; i < n; ++i) scanf("%lld", &a[i]);
	partial_sum(a.begin(), a.end(), a.begin());
	
	
	int l = 1;
	int r = n;
	
	while (l < r) {
		int mid = l + (r - l + 1 >> 1);
		if (cnt(mid)) l = mid;
		else r = mid - 1;
	}
	int t = cnt(l);
	if (t) cout << l << ' ' << cnt(l) << endl;
	else cout << -1 << endl;
    
    return 0;
}
