#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int a,b;
		scanf("%lld %lld",&a,&b);
		long long int op=0;
		while(1)
		{
			if(a==1)
			{
				op+=(b-a);
				break;
			}
			if(b==1)
			{
				op+=(a-b);
				break;
			}
	
			if(a>b)
			{
				 long long int fact=a/b;
				 a=a-fact*b;
				 op+=fact;
			}
			else
			{
				long long int fact=b/a;
				b=b-fact*a;
				op+=fact;
			}
		
		}
		cout<<op<<endl;
		
	}
	return 0;
}
