#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,x,y;
    cin>>t;
    int array[10];
    while(t--)
    {
    	cin>>n;
    	
    	memset(array,0,sizeof(array));
    	y=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>x;
    		x=x%10;
    		if(array[x]==0)
    		array[x]=1;
    		else
    		y++;
    	}
    	cout<<y<<endl;
    }
}
