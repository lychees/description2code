#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int n, process, proc, pasos, j;
    bool encontre;
    cin >> n;
    vector<int> process1;
    for(int i = 1; i <= n; i++){
    	cin >> process;
        process1.push_back(process);
    }
    for(int i = 1; i <= n ; i++){
    	cin >> process;
    	j = 0;
    	encontre = false;
    	while(j < n && !encontre){
    		proc = process1.at(0);
    		process1.erase(process1.begin());
    		if(proc == process){
    		    encontre = true;
    		    pasos ++;
    		}else{
    			process1.push_back(proc);
    			pasos++;
    		} 
    		j++;    
    	}
    }
    
    
    cout << pasos<< endl;
    
    return 0;
}
