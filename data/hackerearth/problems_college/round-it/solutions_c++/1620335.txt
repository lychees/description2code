#include <bits/stdc++.h>
using namespace std;
typedef long long ui;
int main()
{
	ui t;
	cin>>t;
	while(t--){
		ui n,b;
		cin>>n>>b;
		ui lrem = n%b;
		ui rrem = b-lrem;
		ui ans = n;
		if(lrem == 0){
			ans = n;
		}
		else if(lrem < rrem){
			ans = ans - lrem;
		}
		else {
			ans = ans + rrem;
		}
		cout<<ans<<endl;
	}
    return 0;
}
