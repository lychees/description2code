n = input()
a = map(int, raw_input().split())
if (n == 1):
    print 1
else:
    i = 1
    ans = 0
    tans = 1
    current = a[0]
    while i < len(a):
        # print i,current,tans,ans
        if (abs(a[i]-current) == 1):
            current = a[i]
            tans += 1
        else:
            ans = max(ans, tans)
            current = a[i]
            tans = 1
        i += 1
    ans = max(ans, tans)
    print ans
