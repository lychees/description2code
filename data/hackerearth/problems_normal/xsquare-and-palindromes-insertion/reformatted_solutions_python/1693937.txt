a = input()
for i in xrange(a):
    c = [0 for k in xrange(26)]
    b = raw_input()
    for j in xrange(len(b)):
        d = ord(b[j]) - 97
        c[d] = c[d] + 1

    e = 0
    for i in c:
        if (i % 2 == 1):
            e = e+1

    if (e > 1):
        print e-1
    else:
        print 0
