#include <iostream>
using namespace std;
#include<cmath>
int main()
{
	
	int t,i,n,l;
	
	cin>>t;
	int a[100002];
	while(t--){
		
		cin>>n;
		
		l=0;
		if(floor(sqrt(n))==sqrt(n)) a[l++]=sqrt(n);
		if(n!=1)a[l++]=n;
		
		for(i=2;i<sqrt(n);++i) if(n%i==0){
			a[l++]=i;
			a[l++]=n/i;
		}
		if(n!=1)a[l++]=1;
		int d[10];
		for(i=0;i<10;++i) d[i]=0;
		while(n){
			d[n%10]=1;
			n/=10;
		}
		int count=0,flag;
		for(i=0;i<l;++i){
			flag=0;
			while(a[i]){
				if(d[(a[i]%10)]==1) {flag=1;break;}
				a[i]/=10;
			}
			if(flag==1) count++;
		}
		cout<<count<<"\n";	
	}
	
    return 0;
}
