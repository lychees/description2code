#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
   int n;
   scanf("%d",&n);
   vector<long long>a(n);
   for (int i = 0; i < n; i++)
   {
   		scanf("%lld", &a[i]);
   }
   sort(a.begin(), a.end());
   long long ans = max(a[0] * a[1], a[n - 2] * a[n - 1]);
   printf("%lld\n", ans);
}
