using namespace std;
ll dp[1000][1000],n;
ll compute(ll in ,ll c)
{
    if(dp[in][c]!=-1)
        return dp[in][c];
    if(in==n)
    {
        if(c==(n/2-1))
            dp[in][c] = 1;
        else
            dp[in][c]= 0;
    }
    else
    {
        if((c+1)>(in/2))
            dp[in][c]=compute(in+1,c);
        else
            dp[in][c]=compute(in+1,c)+compute(in+1,c+1);
    }
    return dp[in][c];
}
int main()
{
    ll t;
    ios::sync_with_stdio(false);
    cin >> t;
    while(t--)
    {
        memset(dp,-1,sizeof(dp));
        cin >> n;
        n = n*2;
        cout << compute(1,0) << endl;
    }

}
