#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long n,m;
    cin>>n>>m;
    long a[n],c[n];
    for(long i=0;i<n;i++)
    cin>>a[i];
    for(long j=0;j<n;j++)
    cin>>c[j];
    sort(a,a+n);
    sort(c,c+n);
    long q=0,w=0,z=0,sum=0;
    while((q<n&&w<n)&&(z<n&&sum<m))
    {
    	if(a[q]<=c[w])
    	{
    		sum=sum+a[q];
    		if(sum<m){
    		q++;z++;}
    	}
    	else
    	{
    		sum=sum+c[w];
    		if(sum<m){w++;z++;}
    	}
    }
    while(w<n&&(z<n&&sum<m))
    {
    	sum=sum+c[w];
    	if(sum<m){w++;z++;}
    }
    while(q<n&&(z<n&&sum<m))
    {
    	sum=sum+a[q];if(sum<m){q++;z++;}
    }
   cout<<z<<endl;
    
    return 0;
}
