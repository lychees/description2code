
#include <bits/stdc++.h>

{

      for(j=1; j <=n  || j<=i; j++)

        {
            hcf=j;
        }
      }
      if ( hcf == 1)
      {
       r = r+1 ;
      }


    }
   // cout <<r <<endl;

      }

   cout << "Correct" <<endl;

  }

}
