#include <iostream>
#include<stdio.h>
using namespace std;
 
int main()
{
    long long int t,count;
    cin>>t;
    while(t--)
    {
    	char a[1000000];
    	count=0;
    	scanf("%s",a);
    	for(int i=1;a[i]!='\0';i++)
    	{
    		if(a[i-1]==a[i])
    		count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
