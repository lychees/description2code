#include <bits/stdc++.h>
using namespace std;

int main()
{
    unsigned long long i,t,a[3];
    cin>>t;
    while(t--){
    	cin>>a[0]>>a[1]>>a[2];
    	sort(a,a+3);
    	i=pow(2,32);
    	if((a[0]>=i && a[1]>i)||(a[0]>=i && a[1]>i) || (a[0]>=i && a[1]>=i)){
    		cout<<"No\n";
    		continue;
    	}
    	if(a[0]==0 ||a[1]==0 || a[2]==0){
    		cout<<"NO\n";
    		continue;
    	}
    	if((a[0]+a[1])>a[2]){
    		if(a[0]==a[1] && a[1]==a[2])
    			cout<<"YES. It is a equilateral triangle.\n";
    		else if(a[0]==a[1] || a[1]==a[2])
    			cout<<"YES. It is a isosceles triangle.\n";
    		else
    			cout<<"YES. It is a scalene triangle.\n";
    	}
    	else
    		cout<<"NO\n";
    }
    return 0;
}
