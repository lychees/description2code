#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
 long long int power(long long int a,long long int b,long long int m)
{
  long long int c=1;

 while(b>0)
 {
   if(b%2==1)
    c=(c*a)%m;
   b=b>>1;
    a=(a*a)%m;
  }
  return c;
}

	long long int t;
