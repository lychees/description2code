
#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;

int64_t twoP(int n)
{
	int64_t ans=1;
	for(int i=0;i<n;i++)
	ans=ans<<1;

	return ans;
}
int main()
{
		int t;
		scanf("%d",&t);
		while(t--)
		{
			int64_t n,potu,i=1,c=1; cin>>n;
			while(1)
			{
				potu=i*twoP(i);
				if(n>potu ) {n=n-potu; c++;} //cout<<" n= "<<n<<" c= "<<c<<"\n";}
				else break;
				i++;
			}
			int64_t bval,ind;
			bval=ceil(n*1.0/c) -1;
			ind=n%c;
			//cout<<"bval = "<<bval<<" ind= "<<ind<<"\n";

			bool ans;
			if(ind==0)
            {
                //cout<<"if\n";
                ans= bval&1;
            }
            else
            {
                //cout<<"else\n";
                for(int i=0;i<(c-ind);i++) bval=bval>>1;
                ans=bval&1;
            }
			if(ans) cout<<"Earth\n";
			else cout<<"Hacker\n";
		}

		return 0;
}
