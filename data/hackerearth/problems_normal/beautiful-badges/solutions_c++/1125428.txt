#include <cstdio>
#include <algorithm>
#include <vector>
#include <cmath>
using namespace std;

vector <int> A, B;

bool verifica_left(int a, int b) {
    return (B[b]*B[b]*2 <= A[a]*A[a]);
}

bool verifica_right(int a, int b) {
    return (A[a]*A[a]*2 > B[b]*B[b]);
}

int main() {
    int N, c, d;
    scanf("%d",&N);
    for (int i=1; i<=N; i++) {
        scanf("%d %d",&c,&d);
        if (c == 1) A.push_back(d);
        else B.push_back(d);
    }
    sort(A.begin(),A.end());
    sort(B.begin(),B.end());
    int tam = A.size(), pos = 0, resp = 0, tam2 = B.size();
    for (int i=0; i<tam; i++) {
        while (pos < tam2 && verifica_left(i,pos)) pos++;
        if (pos == tam2) break;
        if (verifica_right(i,pos)) {
            resp++;
            pos++;
        }
    }
    printf("%d\n",resp);
    return 0;
}
