int main()
{
  int n;
  si(n);
  ll num,f=0;
  map<ll,ll>m;
  map<ll,ll>::iterator it;
  for(int i=0;i<n;i++)
  {
    sl(num);
    if(num==1 && n==1 )
    {
      f=2;
      break;
    }
    for(ll j=2;j*j<=num;j++)
    {
      while(num%j==0)
      {
        num/=j;
        m[j]++;
      }
      if(num==1)break;
    }
    if(num!=1)
    m[num]++;
  }
  ll ans=1;
  for(it=m.begin();it!=m.end();it++)
  {
    ll ff=it->first;
    ll ss=it->second;
    if(ff%4==3)
    {
      if(ss%2)
      {
        f=1;
        break;
      }
    }
    else if( ff%4==1 )
    {
      ans=(ans*(ss+1))%MOD;
    }
  }
  if(f==2)
  {
    ans=6;
  }
  else if(f==1)
  ans=0;
  else
  {
    ans=(4*ans)%MOD;
  }
  pl(ans);nl;
  
}
