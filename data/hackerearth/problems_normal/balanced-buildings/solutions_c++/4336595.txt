using namespace std;


#define LL long long int
#define sc1(x) scanf("%d", &x)
#define sc2(x, y) scanf("%d%d", &x, &y)
#define sc3(x, y, z) scanf("%d%d%d", &x, &y, &z)
#define pr1(x) printf("%lld\n", x)
#define FOR(i, a, n) for(int i=a;i<n;i++)
#define pp pair<int, int>
#define pb push_back 

#define MAX 2501
#define MOD 1000000007LL


LL S, M, P;
int arr[MAX];
LL mat[MAX][MAX];


LL change(LL a, LL b) {
	if(a < b) {
		return -P*(b-a);
	} else {
		return -M*(a-b);
	}
}


LL func(int n) {

	FOR(j, 0, n) {
		mat[0][j] = change(arr[0], arr[j]);
	}

	FOR(i, 1, n) {
		LL value = mat[i-1][0];
		FOR(j, 1, n) value = max(value, mat[i-1][j]);

		FOR(j, 0, n) {
			mat[i][j] = change(arr[i], arr[j]) + 
								max(value, S + mat[i-1][j]);
		}
	}

	LL result = 0;
	FOR(j, 0, n) result = max(result, mat[n-1][j]);

	return result;
}



int main() {

	int n;

	cin>>n>>S>>M>>P;
	FOR(i, 0, n) cin>>arr[i];

	pr1(func(n));

	return 0;
}
