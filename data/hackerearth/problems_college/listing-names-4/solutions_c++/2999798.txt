#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
	int n,count=1;
	cin>>n;
	string s[n];
	for(int i=0;i<n;i++)
	cin>>s[i];
	n=sizeof(s)/sizeof(s[0]);
	sort(s,s+n);
	for(int i=0;i<n;i++)
	{
		if(s[i]==s[i+1])
		count+=1;
		else
		{
			cout<<s[i]<<" "<<count<<"\n";
			count=1;
		}
	}
    return 0;
}
