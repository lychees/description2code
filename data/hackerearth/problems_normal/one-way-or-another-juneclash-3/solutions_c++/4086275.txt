#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pll pair<ll,ll>
#define X first
#define Y second
#define mp(a,b) make_pair(a,b)
vector<pair<pll,int> > bef,af;

ll dx[]={0,0,1,-1};
ll dy[]={1,-1,0,0};
const ll INF=1e18;	
bool valid(ll x,ll y)
{
	return (x>=0&&x<=INF)&&(y>=0&&y<=INF);
}	

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int t;
	cin>>t;
	while(t--)
	{
		bef.clear();
		af.clear();
		int n;
		cin>>n;
		for(int i=1;i<=n;i++)
		{
			ll x,y,c;
			cin>>x>>y>>c;
			c%=2LL;
			bef.push_back(mp(mp(x,y),c));
			
		}

		bool scheme1=true;//even 0
		bool scheme2=true;//even 1
		for(int i=0;i<bef.size();i++)
		{
			ll x,y,c;
			x=bef[i].X.X,y=bef[i].X.Y,c=bef[i].Y;
		//	cout<<x<<" "<<y<<" "<<c<<endl;
			if((x+y)%2LL==0)
			{
				if(c==1LL)
					scheme1=false;
				if(c==0)
					scheme2=false;
			}
			else 
			{
				if(c==1LL)
					scheme2=false;
				if(c==0)
					scheme1=false;
			}
		}
		bool posi=scheme1||scheme2;
		if(posi)
			cout<<"Yes"<<endl;
		else cout<<"No"<<endl;
	}
	return 0;
}
