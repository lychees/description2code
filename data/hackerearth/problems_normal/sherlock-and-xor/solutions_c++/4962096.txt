#include <bits/stdc++.h>
using namespace std;

typedef long long int ll;

int main()
{
    ll t,n,a,ce,co,ctr,i;
    
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	
    	ctr=0;
    	ce=0;
    	co=0;
    	
    	for(i=0;i<n;i++)
    	{
    		cin>>a;
    		
    		if(a%2==0)
    		{
    			ctr+=(co);
    			ce++;
    		}
    		else 
    		{
    			ctr+=(ce);
    			co++;
    		}
    	}
    	
    	cout<<ctr<<endl;
    }
    
    return 0;
}
