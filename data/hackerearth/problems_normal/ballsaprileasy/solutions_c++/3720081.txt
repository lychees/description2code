#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
    long int a,b,i,j,k,n;
    vector<long int> v;
    cin >> n;
    k = n;
    while(k--)
    {
    	cin >> a ;
    	v.push_back(a);
    }
    
    sort(v.begin(),v.end());
    k = 1;
    a = 0; //ans
    for(auto it = v.begin();it!=v.end();it++)
    {
    	j = k*k;
    	while(*it > j)
    	{
    		a += j;
    		k++;
    		j = k*k;
    	}
    	
    	a += j-*it ;
    	k++;
    	
    }
    cout << a;
    return 0;
}
