#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,i,len;
    string s;
    scanf("%d",&t);
    int *b;
    while(t--){
    	cin>>s;
    	len=s.size();
    	b=(int *)calloc(len+1,sizeof(int));
    	if(s[0]=='D' || s[0]=='B'){
    		b[0]=1;
    	}
    	else if(s[0]=='S'){
    		b[0]=b[1]=1;
    	}
    	for(i=1;i<len;i++){
    		if(s[i]=='D'){
    			b[i]=1;
    		}
    		else if(s[i]=='B'){
    			b[i]=b[i-1]=1;
    			if(i-2>=0)	b[i-2]=1;
    		}
    		else if(s[i]=='S'){
    			b[i]=b[i-1]=b[i+1]=1;
    		}
    	}
    	int ctr=0;
    	for(i=0;i<len;i++){
    	//	cout<<b[i];
    		if(b[i]==0)
    			ctr++;
    	}
    	//cout<<endl;
    	printf("%d\n",ctr);
    	free(b);
    }
    return 0;
}
