#include <iostream>
using namespace std;

typedef long long LL;
const int MAXN = 1e5 + 10;
const LL MOD = 1e9 + 7;

LL dp[MAXN];

int main(){
	dp[0] = 1;
	dp[1] = 1;
	dp[2] = 2;
	for(int i = 3;i < MAXN;++i){
		dp[i] = (dp[i - 1] + dp[i - 2] + dp[i - 3]) % MOD;
	}
	int t;
	cin >> t;
	while(t--){
		int N;
		cin >> N;
		cout << dp[N] << "\n";
	}
    return 0;
}
