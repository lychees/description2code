for tc in xrange(input()):
    x = raw_input()
    danger = False
    for i in xrange(len(x)):
        for j in xrange(i+1, len(x)+1):
            if (x.count(x[i:j]) >= 2):
                danger = True
                break
        if (danger):
            break
    if (danger):
        print "YES"
    else:
        print "NO"
