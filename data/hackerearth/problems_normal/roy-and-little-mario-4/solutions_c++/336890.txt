#include<iostream>
using namespace std;
#define mod 1000000007
long long t[100001];
void trib()
{
	t[0]=1;
	t[1]=1;
	t[2]=2;
	for(int i=3;i<=100000;i++)
	{
		t[i]=t[i-1]+t[i-2]+t[i-3];
		t[i]=t[i]%mod;
	}
}
int main()
{
	trib();
	int g,n;
	cin>>g;
	while(g--)
	{
		cin>>n;
		cout<<t[n]<<endl;
	}
	
	return 0;
}
