#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int arr[105][105];
int main()
{
   int n,m;
   cin>>n>>m;
   for(int i=1;i<=n;i++)
   {
       for(int j=1;j<=m;j++)
        cin>>arr[i][j];
   }
   for(int i=1;i<=n;i++)
   {
   	for(int j=1;j<=m;j++)
   	{ int x = i;int y=j;
      if(arr[i][j]!=10000)
      {
      	while(x!=n+1&&y!=0)
      	{cout<<arr[x][y]<<" ";
      	arr[x][y] = 10000;
      	 x++;y--;
      	}cout<<"\n";
      }
   	}
   }

    return 0;
}
