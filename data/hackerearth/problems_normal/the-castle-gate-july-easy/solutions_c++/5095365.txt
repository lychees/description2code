#include <iostream>
using namespace std;

int main()
{
    int t,n,i,j,k,count=0,p;
    cin>>t;
    for(k=0;k<t;k++)
    {
    	cin>>n;
    	for(i=1;i<n;i++)
    	{
    		for(j=i+1;j<=n;j++)
    		{
    			if((i^j)<=n)
    			{
    				count++;
    			}
    			
    		}
    	}
    	cout<<count<<endl;
    	count=0;
    }
    return 0;
}
