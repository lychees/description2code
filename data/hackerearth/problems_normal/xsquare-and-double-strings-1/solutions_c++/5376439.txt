#include <iostream>
#include<functional>

#include <algorithm>
#include <string>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
 int t;
 cin>>t;
 string s;
 while(t--){
 	int a[26]={0};
 	cin>>s;
 	int l=s.size();
 	for(int i=0;i<l;i++){
 		int cn=s[i];
 		a[cn-97]++;
 	}
 	sort(a+0,a+26,greater<int>());
 	if(a[0]>=2){cout<<"Yes"<<endl;}
 	else{cout<<"No"<<endl;}
 }
 return 0;
}
