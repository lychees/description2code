#include <iostream>
#include <bits/stdc++.h>
using namespace std;
set <int> s;
set <int> :: reverse_iterator it;
int arr[100005];
int main()
{
    int t,n,i,x,j;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>x;
    		s.insert(x);
    	}
    	i=0;
    	for(it=s.rbegin();it!=s.rend();it++)
    	{
    		arr[i++]=*it;
    	}
    	n=i;
    	for(i=0,j=n-1;i<j;i++,j--)
    	{
    		cout<<arr[i]<<" "<<arr[j]<<" ";
    	}
    	if(i==j)
    	{
    		cout<<arr[i];
    	}
    	cout<<endl;
    	s.clear();
    }
    return 0;
}
