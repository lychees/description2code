factorial = []
factorial.append(1)
for i in range(1, 52):
    factorial.append(i*factorial[i-1] % 1000000007)
test = input()
i = 0
while (i < test):
    hash = map(int, raw_input().split())
    girls = hash[0]
    boys = hash[1]
    i = i+1
    if (boys-girls <= 1):
        print(factorial[girls+1]*pow(factorial[boys]*factorial[girls-boys+1],
              1000000007-2, 1000000007)*factorial[girls]*factorial[boys]) % 1000000007
    else:
        print -1
