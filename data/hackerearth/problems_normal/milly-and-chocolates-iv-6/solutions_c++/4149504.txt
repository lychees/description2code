#include <bits/stdc++.h>
#define ll long long int

using namespace std ;

int main()
{
	int T ;
	
	cin >> T ;
	
	while(T--)
	{
		ll N , K , M , ans = 1e18 , mx = 0 ;
		
		cin >> N >> K >> M ;
		
		int P[N] , ind ;
		
		for(int i = 0 ; i < N ; ++i)
			cin >> P[i] ;
			
		for(int i = 0 ; i < N ; ++i)
		{
			mx = 0 ;
			
			for(int j = 0 ; j < P[i] ; ++j)
			{
				ll x ;
				
				cin >> x ;
				
				if(j != 0)
					mx = mx + x + K + M ;
					
				else
					mx = mx + x + K ;
				
				// she will take Ti,j to eat, M sec to wait after eating and K sec to break one peice
				
			}
			
			if(ans > mx)
			{
				ans = mx ;
				ind = i + 1 ;
			}
		}
		
		cout << ind << ' ' << ans << '\n' ;	
	}
	
	return 0 ;
}
