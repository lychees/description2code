#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long int n,s,sum=0;
    cin>>n>>s;
    long long int a[n],k=0;
    int flag=0;
    for(int i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    for(int i=0;i<n;i++)
    {
    	if(a[i]>=s || i==n-1)
    	{
    		k=i;
    		break;
    	}
    }
    int size=k-1;
    for(int i=0;i<=size;i++)
    {
    	if(a[k]+a[i]==s)
    	{
    		flag=1;
    		break;
    	}
    }
    if(flag==0)
   	cout<<"NO\n";
	else
    cout<<"YES\n";
    return 0;
}
