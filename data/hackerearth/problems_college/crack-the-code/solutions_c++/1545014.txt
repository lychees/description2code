#include <iostream>
using namespace std;
void fibo(long long x,long long y)
{
	long long s,a=0,b=1,i,cont=0;
	for(i=1;i<=100000;i++)
	{
		if(b>=x&&b<=y)
		cont++;
		if(b>y)
		break;
		s=a+b;
		a=b;b=s;
		
	}
	if(cont%2)
	cout<<"INVALID"<<endl;
	else
	cout<<"VALID"<<endl;
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    long long x,y;
    cin>>x>>y;
    fibo(x,y);
    }
 
    return 0;
}
