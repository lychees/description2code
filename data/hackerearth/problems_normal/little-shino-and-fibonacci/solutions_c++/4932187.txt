#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
#define MAX 100005
ll fib[15005];
ll sum[15005];
ll modmul(ll a, ll b)
{
	ll res = 0;
	while(b)
	{
		if(b&1) res = (res+a)%MOD;
		a=(a+a)%MOD;
		b/=2;
	}
	return res;
}
int main()
{
		int i;
		fib[1]=0;
		fib[2]=1;
		sum[1]=0;
		sum[2]=1;
		for(i=3;i<15005;++i)
		{
				fib[i] = (fib[i-1]+fib[i-2])%10000;
				sum[i] = (sum[i-1]+fib[i])%MOD;
		}
		int t;
		sum[0]=0;
		scanf("%d",&t);
		while(t--)
		{
				ll l,r,a,ar,b,br;
				scanf("%lld %lld",&l,&r);
				l--;
				a = l/15000;
				ar = l%15000;
				b = r/15000;
				br = r%15000;
				a = modmul(a,sum[15000]);
				b = modmul(b,sum[15000]);
				a=(a+sum[ar])%MOD;
				b=(b+sum[br])%MOD;
				ll ans = b-a;
				ans%=MOD;
				if(ans<0) ans+=MOD;
				printf("%lld\n",ans);


		}
		return 0;
}
