inline ll power(ll a, ll b, ll m) {
	ll r = 1;
	while(b) {
		if(b & 1) r = r * a % m;
		a = (a * a)% m;
		b >>= 1;
	}
	return r;
}
inline ll power(ll a, ll b) {
	 ll r = 1;
	while(b) {
		if(b & 1) r = r * a ;
		a = a * a;
		b >>= 1;
	}
	return r;
}

#define Size 1000006
int a[Size];
int main(){
	input();
	int t,n;
	s(t);
	while(t--){
		s(n)
		f(i,0,n) s(a[i]);
		sort(a,a+n);
		int i = 1;
		ll counter=1;
		ll ans = 0;
		
		while(i<n){
			if(a[i] == a[i-1]) counter++;
			else{
				
				ans += (counter*(counter+1))/2;
				counter = 1;
			}
			i++;
		}
		ans += (counter*(counter+1))/2;
		pl(ans)
	}
