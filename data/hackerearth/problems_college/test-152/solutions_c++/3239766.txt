#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin >> t;
    while(t--) {
    	int n;
    	cin >> n;
    	string s;
    	cin >> s;
    	int a[26];
    	for(int i = 0; i < 26; i++) a[i] = 0;
		for(int i = 0; i < n; i++) {
			a[s[i] - 'a']++;
		}
		int ans = a[0];
		for(int i = 1; i < 26; i++) {
			if(a[i] > ans) ans = a[i];
		}
		if(ans & 1) cout << "No" << endl;
		else cout << "Yes" << endl;
    }
    return 0;
}
