#include <iostream>
using namespace std;

int main()
{
	string word;
	int t;
	cin>>t;
	while(t--)
	{
	cin >> word;
	const char* a= word.c_str();
	int vou =0, constant=0, overallconstant=0;
	while(*a!='\0')
	{
		if((*a =='a') ||(*a =='e') ||(*a =='i') ||(*a =='o') ||(*a =='u') ||(*a =='y') )
		{
			vou++;
			if(constant < 3)
			
			constant = 0;
		}
		else
		{
			constant++;
			overallconstant++;
			
		}
		a++;
	}
	if(overallconstant > vou || constant >= 3)
	cout << "hard\n";
	else
	cout<<"easy\n";
	}
    return 0;
}
