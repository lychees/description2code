for case in range(input()):
    raghu, sayan, dishes = map(int, raw_input().split())
    kcal = sorted(map(int, raw_input().split()))
    r, s = 0, 0
    for dish in xrange(dishes):
        dish_kcal = kcal[dish]
        if (raghu >= dish_kcal):
            raghu -= dish_kcal
            r += 1
        else:
            break

    for dish in xrange(dishes):
        dish_kcal = kcal[dish]
        if (sayan >= dish_kcal):
            sayan -= dish_kcal
            s += 1
        else:
            break

    print 'Raghu Won' if r > s else 'Sayan Won' if s > r else 'Tie'
