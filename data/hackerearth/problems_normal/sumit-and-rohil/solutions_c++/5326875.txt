#include <bits/stdc++.h>
using namespace std;

string str[111];

bool check(string a,string b)
{
	int na=a.length();
	int nb=b.length();
	if(a[0]!=b[0]) return false;
	if(a[na-1]!=b[nb-1]) return false;
	sort(a.begin(),a.end());
	sort(b.begin(),b.end());
	if(a!=b) return false;
	return true;
}


int main()
{
    int tc; 
    cin>>tc;
    
    int ans=tc;
    for(int i=0;i<tc;i++)
    {
    	cin>>str[i];
    	for(int j=0;j<i;j++)
    	{
    		if(check(str[i],str[j]))
    		{
    			ans--;
    			break;
    		}
    	}
    }
    cout<<ans<<endl;
}
