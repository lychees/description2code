#include<iostream>
#include<stdio.h>
using namespace std;
long lcm(long a,long b);
long gcd(long a,long b);
int main()
{
	int t=0;
	long n,a,b;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%ld%ld%ld",&n,&a,&b);
		if(a==0 || b==0)
			printf("-1\n");
		else
			printf("%ld\n",n/a+n/b-n/lcm(a,b));
	}
	return 0;
}
long lcm(long a,long b)
{
	return (a*b)/gcd(a,b);
}
long gcd(long a,long b)
{
	if(a%b==0)
		return b;
	return gcd(b,a%b);
}
