#include <iostream>
using namespace std;

int main()
{
	int T,N,p=0;
    cin >> T;
    while (T--)
    {
        cin >> N;
        int x = 0,max= 0;
        int arr[10010]= {0,};
        while (x<N)
        {
                cin >> p;
                arr[p]++;
                if (arr[p] > max)
                	max = arr[p];
                x++;
        }
        cout <<max<<"\n";
    }

    
    return 0;
}
