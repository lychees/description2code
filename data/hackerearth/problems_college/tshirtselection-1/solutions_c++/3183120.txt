#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[100010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n;
		cin>>n;
		unordered_map<ll int,ll int> mp;
		ll int t,i,j,start,end,ans;
		j=0,t=0,start=0,end=0,ans=0;
		for(i=0;i<n;i++)
		cin>>a[i];
		mp[a[0]]++;t=1;
		for(i=1;i<n;i++)
		{
			if(mp[a[i]]==0)
			{
				mp[a[i]]++;
				t++;
			}
			else
			{
				while(mp[a[i]]!=0)
				{
					mp[a[j]]--;
					if(mp[a[j]]==0)
					t--;
					j++;
				}
				mp[a[i]]++;
				t++;
			}
			if(t>ans)
			{
				ans=t;
				start=j;
				end=i;
			}
			//~ cout<<i<<" "<<j<<" "<<t<<" "<<ans<<endl;
		}
		cout<<(start+1)<<" "<<(end+1)<<endl;
	}
	return 0;
}
