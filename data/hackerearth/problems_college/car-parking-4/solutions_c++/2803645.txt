#include <bits/stdc++.h>
#define ll long long int
#define size 100005
using namespace std;

int main() 
{
	ios::sync_with_stdio(false);
	ll t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		ll f[size];
		memset(f,0,sizeof(f));
		for(int i=0;i<s.length();i++)
		{
			if(s[i]=='B')
			{
				f[i]=1;
				if(i>0)
					f[i-1]=1;
				if(i>1)
					f[i-2]=1;
			}
			else if(s[i]=='D')
			{
				f[i]=1;
			}
			else if(s[i]=='S')
			{
				f[i]=1;
				if(i>0)
					f[i-1]=1;
				if(i<s.length()-1)
					f[i+1]=1;
			}
		}
		ll cnt=0;
		for(int i=0;i<s.length();i++)
		{
			if(f[i]==0)
			{
				cnt++;
			}
		}
		cout<<cnt<<endl;
	}
	return 0;
}
