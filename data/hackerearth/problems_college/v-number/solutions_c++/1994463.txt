#include<iostream>
#include<string.h>
#include<math.h>
#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char num[100]={0};
		int k=0,j;
		cin>>num;
		k=strlen(num);
		int half=k/2;
		int firsthalfsum=0;
		int secondhalfsum=0;
		for(j=0;j<half;j++)
		{
			firsthalfsum+=num[j]-48;
		}
		for(j=half;j<k;j++)
		{
			secondhalfsum+=num[j]-48;
		}
		if(firsthalfsum==secondhalfsum)
		{
			cout<<"YES"<<'\n';
		}
		else
		{
			cout<<"NO"<<'\n';
		}

	}
	return 0;
}
