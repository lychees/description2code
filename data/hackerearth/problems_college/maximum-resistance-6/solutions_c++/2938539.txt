#include <bits/stdc++.h>

using namespace std;

int main(void) {
	int t;
	cin >> t;
	while (t--) {
		string s;
		cin >> s;
		int n = s.size();
		
		stack<int> st;
		int cnt = 0;
		
		for (int i = n - 1; i >= 0; --i) {
			if (s[i] == 'X') {
				++cnt;
				st.push(1);
			}
			else {
				int a = st.top();
				st.pop();
				int b = st.top();
				st.pop();
				
				if (s[i] == 'A') st.push(a + b); 	
				else st.push(max(a, b));
			}
		}
		
		vector<int> a(n);
		for (int i = 0; i < cnt; ++i) {
			cin >> a[i];
		}
		
		sort(a.rbegin(), a.rend());
		cout << accumulate(a.begin(), a.begin() + st.top(), 0) << '\n';
	}
}
