from collections import deque

N = int(raw_input())
q = deque(map(int,raw_input().split()))
q1 = deque(map(int,raw_input().split()))
time = 0
while len(q) != 0 and len(q1) != 0:
    if q[0] != q1[0]:
         q.append(q.popleft())
         time += 1
    else:
        q.popleft()
        q1.popleft()
        time += 1

print time
