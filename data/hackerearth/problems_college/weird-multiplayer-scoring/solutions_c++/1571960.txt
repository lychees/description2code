#include<bits/stdc++.h>
using namespace std;
int main()
{
   long long int m,n,val,ans;
   cin>>m>>n;
   map<string,long long int>memo;
   string str,inp;
   cin>>inp;
   cin>>val;
   memo[inp]=val;
   str=inp;
   ans=val;
   for(int i=1;i<n;i++)
   {
      cin>>inp;
      cin>>val;
      if(memo[inp]==0)
          memo[inp]=val;
      else
         memo[inp]+=val;
      if(memo[inp]>ans)
      {
          ans=memo[inp];
          str=inp;
      }
   }
   cout<<str;
   return 0;
}
