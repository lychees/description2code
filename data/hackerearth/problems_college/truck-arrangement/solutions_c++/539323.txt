#include <bits/stdc++.h>
using namespace std;
int main(){
int t,x,n;
scanf("%d",&t);
while(t--){
 scanf("%d",&n);
 vector<pair<int,int> > v,vv;
 for(int i=0;i<n;i++){
    scanf("%d",&x);
    v.push_back(make_pair(x,i));
 }
 for(int i=0;i<n;i++)
    vv.push_back(v[i]);
 sort(vv.begin(),vv.end());
 vector<pair<int,int> > :: iterator it,xx;
 int mini = (-1)*INT_MAX;
 int k = 0;
 for(it = v.begin();it!=v.end();it++){
  xx = lower_bound(vv.begin(),vv.end(),*it);
  int a = xx-vv.begin()-k;
  k++;
  //printf("%d ",abs(a));
  mini = max(mini,abs(a));
 }
 printf("%d\n",mini);
}
return 0;
}
