#include <iostream>
#include <cstring>
#define mod 1000000007
using namespace std;
typedef long long int ll;

int main()
{
	string S;
	cin>>S;
	ll prev=0,total=0,curr;
	prev=(S[0]-'0');
	total=prev;
	prev=(prev*10)%mod;
	for(int i=1;i<S.length();++i)
	{
		prev=(prev+(i+1)*(S[i]-'0'))%mod;
		total=(total+prev)%mod;
		prev=(prev*10)%mod;
	}
	cout<<total<<endl;
    //cout << "Hello World!" << endl;
    return 0;
}
