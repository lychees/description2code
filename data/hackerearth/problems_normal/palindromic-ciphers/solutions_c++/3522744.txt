#include <iostream>
#include <cstring>
using namespace std;

bool isPalin(char s[],int l){
	int i=0;
	while(i<=l){
		l--;
		if(s[i]!=s[l])
			return false;
		i++;
	}
	return true;
}

void prodJc(char s[],int l){
	long long p=1;
	for(int i=0;i<l;i++){
		p*=(s[i]-96);
	}
	cout<<p<<endl;
}

int main()
{
	bool isPalin(char[],int);
	void prodJc(char [],int);
	int t=0;
	char s[10];
	cin>>t;
	while(t>0){
		t--;
		s[0]='\0';
		cin>>s;
		int l=strlen(s);
		if(isPalin(s,l)==true){
			cout<<"Palindrome"<<endl;
		}
		else{
			prodJc(s,l);
		}
	}
    return 0;
}
