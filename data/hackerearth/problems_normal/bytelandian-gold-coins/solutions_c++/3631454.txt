#include<bits/stdc++.h>
using namespace std;
#define ll long long
map<ll,ll> mp;
ll findSol(ll x)
{
    if(x==0)
        return 0;
    if(mp[x]!=0)
        return mp[x];
    ll ans = findSol(x/2) + findSol(x/3) + findSol(x/4);
    mp[x] =max(ans,x);
    return mp[x];

}
int main()
{
    ll n;
    while(scanf("%lld",&n)!=EOF)
    {

        cout<<findSol(n)<<endl;

    }
}
