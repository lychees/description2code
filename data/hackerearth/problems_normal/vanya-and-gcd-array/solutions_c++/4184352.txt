#include <iostream>
#include <bits/stdc++.h>

using namespace std;
#define LL long long int

#define MOD 1000000007
#define MAXN 505

int arr [MAXN];


LL mem [MAXN][MAXN][105];
LL dp(int idx, int lst, int g){
	
	if(idx < 0 ){
		return (g==1);
	}
	
	if(mem[idx][lst][g] > -1) return mem[idx][lst][g];
	
	LL ret = dp(idx-1, lst, g);
	
	if(arr[idx] < arr[lst]){
		ret = (ret + dp(idx-1, idx, __gcd(g, arr[idx])));
	}
	
	
	return mem[idx][lst][g] = ret;
}


int main()
{
    int n;
    
    cin>>n;
    
    for(int i=0;i<n;i++){
    	cin>>arr[i];
    }
    
    LL res = 0;
    
    memset(mem, -1, sizeof(mem));
    
    
    for(int i=0;i<n;i++){
    	res = (res + dp(i-1, i, arr[i]))%MOD;
    }
    
    cout<<res<<"\n";
    
    return 0;
}
