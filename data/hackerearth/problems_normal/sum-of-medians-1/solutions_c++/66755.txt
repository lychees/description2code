#include <iostream>
#include <set>

int main()
{
    long n, v, sum;
    std::set<long>		s;
    std::set<long>::iterator	m;
    std::cin >> n >> v;
    sum = v;
    s.insert(v<<32);
    m = s.begin();
    for (int i = 1; i < n; i++) {
	std::cin >> v;
	v = (v << 32) + i;
	s.insert(v);
	if (v < *m) {
	    if (i&1) m--;
	} else if (!(i&1))
	    m++;
	sum += *m >> 32; }
    std::cout << (sum % 100000) << std::endl;
    return 0;
}
