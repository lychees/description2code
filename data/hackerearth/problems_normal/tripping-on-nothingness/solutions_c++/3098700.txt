#include<bits/stdc++.h>
using namespace std;
#define lli long long int

lli nothingness(lli S,lli x)
{
    if(x==0)
    	return S;
    return nothingness(x,S%x);
}
int main()
{
    lli x,S,Q,n;
    //lli A[100001];
    map<lli ,int> mapp;
    cin>>S>>Q;
    while(Q--)
    {
        n=0;
        cin>>x;
        if(S>x)
        	n=nothingness(S,x);
        else
        	n=nothingness(x,S);

        if(mapp.find(n)==mapp.end())
            cout<<n;
        else
            cout<<"-1";
        mapp[n]=1;
        cout<<"\n";
    }
    return 0;

}
