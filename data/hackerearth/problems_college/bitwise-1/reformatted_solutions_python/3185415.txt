t = int(input())
k = 1

for _ in range(0, t):
    x, n = map(int, raw_input().split())
    count = 0
    ans = 0
    for i in range(1, 10000000):
        if (x+i == x ^ i):
            count += 1
        if (count == n):
            ans = i
            break

    print("Case #"+str(k)+": "+str(ans))
    k = k + 1
