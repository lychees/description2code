#include<iostream>
using namespace std;

long long int gcd(long long int a, long long int b){
	while(b != 0){
		int temp = a;
		a = b;
		b = temp%b;
		gcd(a, b);
		
	}
	return a;
}

long long int foo(long long int n){
	 long long int ret = 0;
	 int i;
  	for(i = 1; i <= n; i++){
    	if(gcd(n, i) == 1)
           ret += 1;
   	}
  	return ret;
}

long long int SpecialSum(long long int n){
	long long int ret=0;
	long long int i;
	for(i = 1; i <= n; i++){
      if(n%i == 0){
         ret += foo(i);
      }
 	}
 	return ret;
}
