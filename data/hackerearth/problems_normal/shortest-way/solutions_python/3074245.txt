T=int(raw_input())
for i in range(T):
	pair=[int(i) for i in raw_input().split()]
	k=0
	while pair!=[1,1] and pair!=[1,0] and pair!=[0,1]:
		if pair[0]>pair[1]:
			k+=int(pair[0]/pair[1])
			pair[0]%=pair[1]
		else:
			k+=int(pair[1]/pair[0])
			pair[1]%=pair[0]
	if pair==[1,1]:
		print 0
	else:
		print k-1
