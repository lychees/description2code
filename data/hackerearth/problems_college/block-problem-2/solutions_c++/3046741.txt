    {
        ll &val=dp1[i];
        if(val!=-1)
            return val;
        if(i==0)
        {
            val=1;
            return val;
        }
     
        if(i<0)
        return 0;
        val=(((f(i-1)+f(i-2))%mod+f(i-3))%mod+f(i-4))%mod;

    }
    ll po(int x, int y)
    {
        ll temp;
        if( y == 0)
            return 1;
        temp = po(x, y/2);
        if (y%2 == 0)
            return (temp*temp)%mod;
        else
            return ((x*temp)%mod*temp)%mod;
    }
    ll fun(int i)
        {

            return val;
         if(i==1)

             val=1;

         ll temp=0;
         for(int j=1;j<i;j++)
         {
             temp=(temp+(g[i-j]*fun(j))%mod)%mod;
         }
         return (g[i]-temp+mod)%mod;
        }
