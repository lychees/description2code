#include <iostream>
#include <algorithm>
using namespace std;
#define ll long long

int main()
{
    int t; cin>>t;
    while(t--)
    {
    	int n; cin>>n;
    	string s; cin>>s;
    	ll r=1,l=0;
    	ll ans = 0;
    	while(true)
    	{
    		while(s[r]>=s[r-1]&&r<n) r++;
    		ans+=(r-l)*(r-l+1)/2;
    		if(r==n) break;
    		else {l=r;r++;}
    	}
    	cout<<ans<<endl;
    }
}
