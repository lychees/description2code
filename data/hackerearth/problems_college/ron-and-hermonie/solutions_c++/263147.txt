#include<iostream>
using namespace std;
#include<stdio.h>
int main()
{
  int t,res;	
  long long int b,e,i,z;
  scanf("%d",&t);
  while(t--)
  {
  	scanf("%lld %lld",&b,&e);
  	if(e==0)
  	   res=1;
  	else
	{
	   e=e%4;
	   if(e==0)
	     e=4;
	   b=b%10;
	   z=1;
	   for(i=1;i<=e;i++)
	   {
	   	  z=z*b;
	   	  z=z%10;
	   } 	
	   res=z;
	}   
	printf("%d\n",res);
  }
  return 0;
}
