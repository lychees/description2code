#include <stdio.h>
#include <vector>

using namespace std;

int n,m,q,countries;
bool planet[1000][1000];
vector<int> res;

int dfs(int i, int j)
{
    //printf("i %d j %d\n",i,j);
    if(i < 0 || j < 0 || i == n || j == m) return 0;

    if(planet[i][j] == 0) return 0;

    planet[i][j] = false;
    return 1 + dfs(i+1,j) + dfs(i-1,j) + dfs(i,j+1) + dfs(i,j-1);

}

int main()
{
    scanf("%d %d %d",&n,&m,&q);
    //printf("n %d m %d\n",n,m);

    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            int c;
            scanf("%d",&c);
            if(c == 1){countries++;}
            planet[i][j] = c;
        }
    }

    for(int i=0;i<q;i++){
        int x,y;
        scanf("%d %d",&x,&y);
        x--;y--;
        countries -= dfs(x,y);
        res.push_back(countries);
    }
    for(int i=0;i<q;i++){
        printf("%d\n",res[i]);
    }
}
