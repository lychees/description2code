#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
#include<string>
#include<cstring>
using namespace std;

#define REP(i,n) for(int i=0, i##_len=(n); i<i##_len; ++i)
#define EACH(i,c) for(__typeof((c).begin()) i=(c).begin(),i##_end=(c).end();i!=i##_end;++i)

int F(int n) {
    if (n<10) return n;
    int r = 0;
    while (n) {
	r += n%10;
	n /= 10;
    }
    return F(r);
}

int T, N;
int main() {
    scanf("%d", &T);
    REP ($, T) {
	scanf("%d", &N);
	printf("%d\n", F(N));
    }
    return 0;
}
