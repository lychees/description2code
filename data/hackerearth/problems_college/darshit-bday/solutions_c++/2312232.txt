#include <iostream>
using namespace std;

int main()
{
    int t;
     long long int b,w,x,y,z,cost;
    cin>>t;
    while(t--)
    {
    	cost=0;
    	cin>>b>>w;
    	cin>>x>>y>>z;
    	if(x<y)
    	{
    		cost=x*b;
    		if(y<(x+z))
    		{
    			cost+=w*y;
    		}
    		else
    		cost+=(x+z)*w;
    	}
    	else
    	{
    		cost=w*y;
    		if(x<(y+z))
    		cost+=x*b;
    		else
    		cost+=(y+z)*b;
    	}
    	cout<<cost<<endl;
    }
}
