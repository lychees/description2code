#include<iostream>
using namespace std;
#define ll long long int
ll dp[20000004];
int main()	{
  	int i;
    ll t;
    ll n;
    ll ans;
    dp[1] = 0;
    for(i=2; i <= 20000003; i++) {
        dp[i] = dp[i-1] + 1;
        if (i%2 == 0) {
            dp[i] = min(1 + dp[i/2], dp[i]);
        }
        if (i%3 == 0) {
            dp[i] = min(1 + dp[i/3], dp[i]);
        }
    }
    scanf("%lld",&t);
   	while(t--)	{
    	scanf("%lld",&n);
    	ans=dp[n];
   		printf("%lld\n",ans);
   	} 
    return 0;
}
