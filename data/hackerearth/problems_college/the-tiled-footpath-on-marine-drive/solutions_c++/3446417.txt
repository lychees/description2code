#include <cstring>
#include <iostream>
#include <cstdlib>
#include <math.h>
#define ll long long
using namespace std;
int hu;
void multiply(ll F[2][2], ll M[2][2])
{
    ll x =  F[0][0] * M[0][0] + F[0][1] * M[1][0];
    ll y =  F[0][0] * M[0][1] + F[0][1] * M[1][1];
    ll z =  F[1][0] * M[0][0] + F[1][1] * M[1][0];
    ll w =  F[1][0] * M[0][1] + F[1][1] * M[1][1];
    F[0][0] = x%hu;
    F[0][1] = y%hu;
    F[1][0] = z%hu;
    F[1][1] = w%hu;
}

void power(ll F[2][2], int n)
{
    if (n == 0 || n == 1)
        return;
    ll M[2][2] = {{1,1},{1,0}};
    power(F, n / 2);
    multiply(F, F);
    if (n % 2 != 0)
        multiply(F, M);
}
ll fibo_matrix(ll n)
{
    ll F[2][2] = {{1,1},{1,0}};
    if (n == 0)
        return 0;
    power(F, n - 1);
    return F[0][0];
}
main(){
	int t;
	cin >> t;
	for(long int i=0;i<t;i++){
    long long int n,k,l;
	cin>>n>>k;
	hu=pow(2,k);
	l=fibo_matrix(n+1);
	cout<<l%hu<<endl;
	}
}
