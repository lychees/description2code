#include <cstdio>
#define max 101
using namespace std;

int calculate(char *s){
	int count = 0;
	for(int i=0; *(s+i)!='\0';i++){
		switch(*(s+i)){
			case '0':
				count += 6;
				break;
			case '1':
				count += 2;
				break;
			case '2':
				count += 5;
				break;
			case '3':
				count += 5;
				break;
			case '4':
				count += 4;
				break;
			case '5':
				count += 5;
				break;
			case '6':
				count += 6;
				break;
			case '7':
				count += 3;
				break;
			case '8':
				count += 7;
				break;
			case '9':
				count += 6;
				break;
			default:
                break;
		}
		
	}
	return count;
}

int main()
{
    char string[max];
    fgets (string, max, stdin);
    printf("%d",calculate(string));
    return 0;
}
