#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long int n,m,count=0;
	cin>>n;
	long long int arr[n];
	for(int i=0;i<n;i++)
	scanf("%lld",&arr[i]);
	scanf("%lld",&m);
	sort(arr,arr+n);
	long long int sum=0;
	for(int i=n-1;i>=0;i--)
	{
		sum+=arr[i];
		count++;
		if(sum>=m)
		break;
	}
	printf("%lld",count);
    return 0;
}
