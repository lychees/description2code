#include <iostream>
using namespace std;
typedef long long int ll;

ll modpow(ll x, ll y , ll m)
{
	ll ans=1;
	ll a=x;
	while(y)
	{
		if(y&1)
			ans = (ans*a) % m;
		a = (a*a) %m;
		y>>=1;
	}
	return ans;
}


int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int a,b,k;
    	cin >> a >> b >> k ;
    	int m=1;
    	while(k--)
    		m*=10;
		int x = modpow(a,b,m);
		int sum=0;
		while(x)
		{
			sum += x%10;
			x/=10;
		}
		cout << sum << '\n';
	}
    return 0;
}
