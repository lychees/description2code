#include<bits/stdc++.h>
using namespace std;
vector<int>z;
int main()
{ int n,lol,i,x,t;
  cin>>t;
  while(t--)
  {cin>>n;
  z.resize(n);
  for(i=0;i<n;i++)
  {

    cin>>z[i];
  }
  sort(z.begin(),z.end());
  lol=1;
  for(i=0;i<n;i++)
  {
    if(i==0)
    {
      x=z[i];
    }
    else if(z[i]!=x)
    {
      lol++;
      x=z[i];
    }
  }

  cout<<lol<<"\n";
  }



return 0;
}
