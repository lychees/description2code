#include <iostream>
#define pi pair<int,int>
#include<vector>
#include<algorithm>
#include<cstring>
using namespace std;
vector<pi> v;
int main()
{
   int n;
   cin>>n;
   bool flag=0;
   string str;
   for(int i=0;i<n;i++)
   {
   	cin>>str;
   	int h1=((str[0]-'0')*10)+str[1]-'0';
   	int m1=((str[3]-'0')*10)+str[4]-'0';
   	
   	int h2=((str[6]-'0')*10)+str[7]-'0';
   	int m2=((str[9]-'0')*10)+str[10]-'0';
   //	cout<<h1<<m1<<h2<<m2<<endl;
   	v.push_back(pi(h1*60+m1,h2*60+m2));
   }
   sort(v.begin(),v.end());
   for(int i=1;i<v.size();i++)
   {
   	if(v[i].first<v[i-1].second)
   	{
   		flag=1;
   		break;
   	}
   }
   if(flag)
   cout<<"Will need a moderator!";
   else
   cout<<"Who needs a moderator?";
}
