T = int(raw_input())

days ={}

def findNum(data):
	multiplier = 1
	if data[0] == 'G':
		multiplier = 2
	
	lst = [int(s) for s in data.split() if s.isdigit()]
	for i in lst:
		if i not in days:
			days[i] = 1 * multiplier
		else:
			days[i] += 1*multiplier

for i in range(T):
	data = raw_input()
	findNum(data)

max_key = 0
max_value = 0
for i in days:
	if max_value < days[i]:
		max_key = i
		max_value = days[i]

if max_key == 19 or max_key == 20:
	print "Date"
else:
	print "No Date"
