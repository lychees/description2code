#include <iostream>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    long t;
    cin>>t;
    while(t--)
    {
    	long n,i,j,temp=0;
    	cin>>n;
    	long a[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	i=0;
    	for(j=0;j<n;j++)
    	{
    		if(a[i]==-1)
    		{
    			cout<<"Win"<<"\n";
    			break;
    		}
    		else if(a[i]==i || a[i]==-2)
    		{
    			cout<<"Lose"<<"\n";
    			break;
    		}
    		else 
    		{
    			temp=a[i];
    			a[i]=-2;
    			i=temp;
    		}
    		
    		
    	}
    }
    return 0;
}
