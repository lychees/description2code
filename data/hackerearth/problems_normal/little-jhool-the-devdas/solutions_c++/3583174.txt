#include<bits/stdc++.h>
using namespace std;
main()
{
    int n,i,d=0,j=0;
    cin>>n;
    int a[n],b[n];
    for(i=0;i<n;i++)
    {
        cin>>a[i];
        d=d+a[i];
    }
    if(d%2!=0)
    {
        cout<<d;
    }

    else if(d%2==0)
    {
        for(i=0;i<n;i++)
        {
            if(a[i]%2!=0)
            {

                b[j]=d-a[i];
                j++;
            }
        }
        if(j==0)
        {
            cout<<":(";
        }
        else{
        sort(b,b+j);
        cout<<b[j-1];
        }


    }



}
