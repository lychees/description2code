#include<bits/stdc++.h>
using namespace std;
long long int fun(long long int n)
{
	long long int sum=0,i=5;
	for(i=5;(n/i)>=1;i=i*5)
	{
		sum=sum+(n/i);
	}
	return sum;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
     long long int i,k,j,n,ans;
     cin>>n;
     double s=0.0;
     for(i=1;i<=n;i++)
     {
     	s=s+log10(i);
     }
     ans=int(s)+1;
     cout<<ans-fun(n)<<endl;
 }
     return 0;
     
 		
}
