bool is_prime(int n){
	if(n<=1)
		return false;
	if(n==2)
		return true;
	if(n%2==0){
		return false;
	}
	int i=3;
	int lim=sqrt(n)+1;
	while(i<=lim){
		if(n%i==0)
			return false;
		i+=2;
	}
	return true;
}
