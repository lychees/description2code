#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,n;
	cin>>t;
	n = t;
	t++;
	while(t--)
	{
		string s,s1;
		char ans,ch;
		fflush(stdin);
		getline(cin,s);
		for(int i =0; i< s.size(); i++)
		{
			if(s[i] != '.' && s[i] != ',' && s[i] != '?' && s[i] != '!'&& s[i] != ' ')
			{
				s1 = s1 + s[i];
			}
		}
		int flag = 0;
		if(s1.size()%2 == 0)
		{
			for(int i = 0; i < s1.size()/2; i++)
			{
				if(s1[i] == '*')
				{
					s1[i] = s1[s1.size()-1-i];
					ans = s1[s1.size()-1-i];
					flag = 1;
				}
			}
			if(flag == 0)
			{
				for(int i = s1.size()/2; i < s1.size(); i++)
				{
					if(s1[i] == '*')
					{
						s1[i] = s1[s1.size()-1-i];
						flag = 1;
						ans = s1[s1.size()-1-i];
					}
				}
			}
		} 
		else {
			for(int i = 0; i < s1.size()/2-1; i++)
			{
				if(s1[i] == '*')
				{
					s1[i] = s1[s1.size()-1-i];
					flag = 1;
					ans = s1[s1.size()-1-i];
				}
			}
			if(flag == 0)
			{
				for(int i = s1.size()/2+1; i < s1.size(); i++)
				{
					if(s1[i] == '*')
					{
						s1[i] = s1[s1.size()-1-i];
						flag = 1;
						ans = s1[s1.size()-1-i];
					}
				}
			}
		}
		if(n != t)
		cout<<ans<<endl;
	}
	return 0;
}
