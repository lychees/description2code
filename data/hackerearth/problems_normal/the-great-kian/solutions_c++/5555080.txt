#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
        int a[100009];
        long long int s1=0,s2=0,s3=0;
        cin>>n;
        for(int i=0;i<n;i++)
            cin>>a[i];
        for(int i=0;i<n;i=i+3)
            s1=s1+a[i];
        for(int i=1;i<n;i=i+3)
            s2=s2+a[i];
        for(int i=2;i<n;i=i+3)
            s3=s3+a[i];
        cout<<s1<<" "<<s2<<" "<<s3;
}
