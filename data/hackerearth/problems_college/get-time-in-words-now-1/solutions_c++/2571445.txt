#include<stdio.h>
#include<iostream>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<stdlib.h>
#include<vector>
#include<map>
#include<set>

using namespace std;

main()
{
	int h,m;
	cin>>h>>m;
	
	map<int,string>ma;
	ma[0]="twelve";
	ma[1]="one";
	ma[2]="two";
	ma[3]="three";
	ma[4]="four";
	ma[5]="five";
	ma[6]="six";
	ma[7]="seven";
	ma[8]="eight";
	ma[9]="nine";
	ma[10]="ten";
	ma[11]="eleven";
	ma[12]="twelve";
	ma[13]="thirteen";
	ma[14]="fourteen";
	ma[15]="fifteen";
	ma[16]="sixteen";
	ma[17]="seventeen";
	ma[18]="eighteen";
	ma[19]="nineteen";
	ma[20]="twenty";
	ma[30]="thirty";
	ma[40]="forty";
	ma[50]="fifty";
	

	
	if(m==0)
	cout<<ma[h]<<" o' clock";
	else if(m==30)
	cout<<"half past "<<ma[h];
	else if(m==45)
	cout<<"quarter to "<<ma[(h+1)%12];
	else if(m==15)
	cout<<"quarter past "<<ma[h];
	else if(m<30)
	{
		if(m>20)
		{
		int ans=m%10;
		m=m/10;
		int ans1=m%10;
		cout<<ma[ans1*10]<<" "<<ma[ans]<<" minutes past "<<ma[h];
		}	
		else
		{
		if(m>1)
		cout<<ma[m]<<" minutes past "<<ma[h];
		else
		cout<<ma[m]<<" minute past "<<ma[h];
		}
	}
	else if(m>30)
	{
		if((60-m)>20)
		{
			m=60-m;
		int ans=m%10;
		m=m/10;
		int ans1=m%10;
		cout<<ma[ans1*10]<<" "<<ma[ans]<<" minutes to "<<ma[(h+1)%12];
		}
		else
		{
			if(60-m>1)
			cout<<ma[60-m]<<" minutes to "<<ma[(h+1)%12];
			else
			cout<<ma[60-m]<<" minute to "<<ma[(h+1)%12];
		}
	}
}
