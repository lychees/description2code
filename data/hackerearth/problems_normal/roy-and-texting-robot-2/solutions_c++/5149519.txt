using namespace std;

	keypad[1][0]='.';
	keypad[1][1]=',';
	keypad[1][2]='?';
	keypad[1][3]='!';
	keypad[1][4]='1';
	keypad[2][0]='a';
	keypad[2][1]='b';
	keypad[2][2]='c';
	keypad[2][3]='2';
	keypad[3][0]='d';
	keypad[3][1]='e';
	keypad[3][2]='f';
	keypad[3][3]='3';
	keypad[4][0]='g';
	keypad[4][1]='h';
	keypad[4][2]='i';
	keypad[4][3]='4';
	keypad[5][0]='j';
	keypad[5][1]='k';

}
int search(char keypad[10][5], int r, int c, char x, int& key){
	for(int i=0;i<r;i++){
		for(int j=0;j<c;j++){
			if(x==keypad[i][j]){
				key=i;
				return j+1;
			}
		}
	}
	return -1;
}
int main()
{
    int t;
    cin>>t;
    char keypad[10][5]={-1};
    set(keypad);
    while(t--){
    	string s;
    	cin>>s;
    	int l=s.size();
    	int k=1;
    	int timeTotal=0;
    	int key=0;
    	for(int i=0;i<l;i++){
    		int number=search(keypad,10,5,s[i],key);
    		timeTotal+=number;
    		if(k!=key)
    		timeTotal++;
    		k=key;
    	}
    	cout<<timeTotal<<endl;
    }
    return 0;
}
