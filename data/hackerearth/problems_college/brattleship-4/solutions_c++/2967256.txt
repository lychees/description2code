#include <iostream>
using namespace std;
#include<cstdio>
int ceil(int a,int b){
	int q=a/b,r=a%b;
	return r?q+1:q;
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    scanf("%d",&t);
    for(int i=1;i<=t;i++){
    	int r,c,w;
    	scanf("%d %d %d",&r,&c,&w);
    	printf("Case #%d: %d\n",i,r*(ceil(c,w)+w-1));
    }
    return 0;
}
