#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll a[1000006][3],dp[1000006][3],n,t;
int main()
{
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		for(ll i=0;i<n;i++)
			scanf("%lld %lld %lld",&a[i][0],&a[i][1],&a[i][2]);
		dp[0][0]=a[0][0];
		dp[0][1]=a[0][1];
		dp[0][2]=a[0][2];
		for(ll i=1;i<n;i++)
		{
			dp[i][0]=a[i][0] + min(dp[i-1][1],dp[i-1][2]);
			dp[i][1]=a[i][1] + min(dp[i-1][2],dp[i-1][0]);
			dp[i][2]=a[i][2] + min(dp[i-1][0],dp[i-1][1]);
		}
		printf("%lld\n",min(dp[n-1][0],min(dp[n-1][1],dp[n-1][2])));
	}
	return 0;
}
