#include<iostream>
#include<stdio.h>
using namespace std;
int getcount(int a,int b)
{
    int i,count=0;
    for(i=1;i*i<=a;i++)
    {
        if(a%i==0)
        {
            if(i>b)
            {
                if(i*i==a)
                    count++;
                else
                count+=2;
            }
            else
            {
                if(a/i>b)
                {
                    count++;
                }
            }
        }

    }
    return count;
}
int main()
{
    int t,x,y;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&x,&y);
        if(x==y)
        {
            printf("%d\n",-1);
        }
        else
        {
            printf("%d\n",getcount(x-y,y));
        }

    }
}
