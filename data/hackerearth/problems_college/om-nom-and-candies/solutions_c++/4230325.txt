#include <iostream>
using namespace std;
int data[100001];
int newData[100001];

int maxVal(int a, int b)
{
	if(a<b)
		return b;
	else
		return a;
}

int main()
{
	int N;
	cin>>N;
	cin>>data[1];
	newData[1] = data[1];
	for(int i=2;i<=N;i++)
	{
		cin>>data[i];
		newData[i] = data[i] + data[1];
		//data[i] += data[1];
	}
	
	for(int i=2;i<=(N/2);i++)
	{
		int k=2;
		int prod=i*k;
		k++;
		while(prod<=N)
		{
			newData[prod] = maxVal(newData[prod], data[prod]+newData[i]);
			prod = i*k;
			k++;
		}
	}
	
	for(int i=1;i<=N;i++)
		cout<<newData[i]<<" ";
    return 0;
}
