#include <iostream>
#include <algorithm>

using namespace std;

void canKill(int a[],int b[],int n)
{
	sort(a,a+n);
	sort(b,b+n);
	for(int i = 0;i < n;i++)
		if(a[i] <= b[i])
		{
			cout << "NO\n";
			return;
		}
	cout << "YES\n";
	return;
}

int main()
{
	int t,i,n;
	cin >> t;
	while(t--)
	{
		cin >> n;
		int a[n],b[n];
		for(i = 0;i < n;i++)
			cin >> a[i];
		for(i = 0;i < n;i++)
			cin >> b[i];
		canKill(a,b,n);
	}
	return 0;
}
