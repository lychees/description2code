#include<bits/stdc++.h>
using namespace std;
int main () {

    int t,n,h,mini = 10000;
    cin>>t;
    int a[1000];
    while ( t-- ) {
        mini = 10000;
        cin >> n >> h;
        for ( int i =0 ; i < n ; i ++ ) {
            cin >> a[i];
            if ( abs((a[i] - h)) < mini )
                mini = abs(a[i] - h);
        }
        cout << mini <<endl ;
    }
    return 0;
}
