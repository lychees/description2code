#include <iostream>
using namespace std;

int main()
{
    int total, result, bits, index, array[1000];
    long long  n, m;
    cin >> total;
    while(total--) {
    	cin >> n >> m;
    	bits = 0;
    	index = 0;
    	while(m) {
    		if (m%2) {
    			array[index++] = bits;
    		}
    		m/=2;
    		bits++;
    	}
    	for (bits = index - 1;bits > 0; bits--){
    		cout << "(" << n << "<<" << array[bits] << ")" << " + ";
    	}
    	cout << "(" << n << "<<" << array[0] << ")" << "\n";
    }
}
