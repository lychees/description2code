#include <iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main()
{
    int r,n,t,w,ans=0,x;
    vector<int> v;
    cin>>t;
    while(t--)
    {
    	ans=0;
    	cin>>n>>r;
    	int a[n],b[n];
    	for(int i=0;i<n;i++)
    	{
    		w=a[0];
    			cin>>a[i];
    	}
    	sort(a,a+n);
    	for(int i=n-1,j=0;i>=0;i--,j++)
    		b[j]=a[i];
    	
		for(int i=0;i<n;i++)
			if(b[i]==w)
			{
					x=i%r;
					break;
			}
		
    	while(x<n)
    	{
    	v.push_back(b[x]);
    	x+=r;
    	}
    	
    	sort(v.begin(),v.end());
    	for(int i=v.size()-1;i>=0;i--)
    	{
    			if(v[i]>w)
    			ans++;
    			//cout<<v[i]<<endl;
    	}
    	cout<<ans<<endl;
    	v.clear();
    	
    }
}
