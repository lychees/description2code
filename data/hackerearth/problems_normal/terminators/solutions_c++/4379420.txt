#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,a,b;
    cin>>t;
    while(t--){
    	cin>>n;
    	pair<int,int> arr[n];
    	for(int i=0;i<n;++i){
    		cin>>a>>b;
    		arr[i]=make_pair(a,b);
    	}
    	sort(arr,arr+n);
    	int dp[n+1],res=1;
    	dp[0]=1;
    	for(int i=1;i<n;++i){
    		dp[i]=1;
    		for(int j=i-1;j>=0;--j){
    			if(arr[i].second>arr[j].second)
    				dp[i]=max(dp[i],dp[j]+1);
    		}
    		res=max(res,dp[i]);
    	}
    	cout<<res<<endl;
    }
    return 0;
}
