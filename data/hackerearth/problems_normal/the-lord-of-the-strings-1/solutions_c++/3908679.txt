#include <bits/stdc++.h>
using namespace std;

int main()
{
	char input[31];
	char output[31];
	int visited[26];
	int t,i,k;
	cin>>t;
	while(t--)
	{
		cin>>input;
		k=0;
		for(i=0;i<26;i++)
			visited[i]=0;
		for(i=0;i<strlen(input);i++)
		{
			if(visited[input[i]-'a']==0)
			{
			output[k++]=input[i];
			visited[input[i]-'a']=1;
			}
		}
		output[k]='\0';
		cout<<output<<endl;
	}
}
