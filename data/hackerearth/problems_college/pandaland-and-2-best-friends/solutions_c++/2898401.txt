#include <iostream>
#include <math.h>
using namespace std;
long long cakemax(long long nrzile){
	long long ped;
	ped = nrzile/2;
	return ped*(ped+1) + (nrzile%2)*(ped+1);	
}
long long calc(long long nrbomb){
	long long inf;
	long long sup,aux;
	inf = 0; 
	sup = 100000;
	while(inf != sup){
		aux = (inf+sup)/2;		
		if( cakemax(aux) >= nrbomb)
			sup = aux;
		else
			inf = aux +1;
	}
	return inf;
}
int main()
{

    while(t){

    	scanf("%lld",&n);
