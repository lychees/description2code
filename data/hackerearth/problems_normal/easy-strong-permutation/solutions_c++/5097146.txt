#include<bits/stdc++.h>
using namespace std;
int main()
{
	
	long long i,ans=0,n,j,k;
	cin>>n;
	long long a[n],temp[n];
	for(i=0;i<n;i++)
		cin>>a[i];
	if(n==1)
	{
		cout<<a[0]<<endl;
		return 0;
	}	
	sort(a,a+n);
	long  long x=(n+1)/2;
	i=0;
	j=n-1;
	k=0;
	while(x--)
	{
		if(i==j)
		{
			temp[k++]=a[i++];
			break;
		}
		temp[k++]=a[i++];
		temp[k++]=a[j--];
	}
	//for(i=0;i<n;i++)
	//	cout<<temp[i]<<endl;
	ans=abs(temp[n-1]-temp[0]);
	//cout<<ans<<endl;
	for(i=1;i<n;i++)
	{
	
		ans=ans+abs(temp[i]-temp[i-1]);
	//	cout<<ans<<endl;
	}
	cout<<ans<<endl;
	
	return 0;
}
