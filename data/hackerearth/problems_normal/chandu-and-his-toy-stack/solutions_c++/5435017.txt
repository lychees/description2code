#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t; cin >> t;
	while(t--)
	{
		int n, x, y; cin >> n >> x >> y;
		vector<int> v1; vector<int> v2;
		int res = 0;
		for(int i=0;i<n;i++)
		{
			int a, b; cin >> a >> b;
			v1.push_back(a);
			v2.push_back(b);
		}
		
		sort(v1.begin(), v1.end());
		sort(v2.begin(), v2.end());
		
		for(int i=0;i<n;i++)
		{
			if(v1[i] > v2[i])
			{
				res += (y*abs(v1[i]-v2[i]));
			}
			else if(v1[i] < v2[i]) res += (x*abs(v1[i]-v2[i]));
		}
		
		cout << res << endl;
	}
}
