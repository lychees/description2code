#include <iostream>
#include <vector>
#include <string>
using namespace std;
typedef vector<int> vi;
#define pb push_back
#define forn(i, n) for (int i = 0; i < n; ++i)

vi fib;

void init()
{
    fib.pb(1); fib.pb(1);
    while (fib.back() < 1e5 + 100)
        fib.push_back(fib[(int)fib.size() - 1] + fib[(int)fib.size() - 2]);
}

string solve()
{
    int a, b;
    cin >> a >> b;
    int cnt = 0;
    forn(i, fib.size())
        cnt += a <= fib[i] && fib[i] <= b;
    return cnt & 1 ? "INVALID" : "VALID";
}
 
int main()
{
    init();
    int t; cin >> t;
    while (t --> 0)
        cout << solve() << endl;
 
    return 0;
}
