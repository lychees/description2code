#include "bits/stdc++.h"
using namespace std;
const int N = 2e3 + 3;
const int MAX = 1e6 + 6;
int t;
int n , l;
pair < int , int > arr[N];
bool ok;
int sum[MAX];
int main(){
    scanf("%d" , &t);
    while(t--){
        int mx = 0;
        scanf("%d %d" , &n , &l);
        for(int i = 1 ; i <= n ; ++i){
            scanf("%d %d" , &arr[i].first , &arr[i].second);
            mx = max(mx , arr[i].second);
        }
        ok = 0;
        sort(arr + 1 , arr + 1 + n);
        for(int i = 1 ; i + l - 1 <= mx ; ++i){
            int x = i;
            int y = i + l;
            for(int j = 1 ; j <= n ; ++j){
                if(arr[j].first >= x && arr[j].second <= y){
                    ++sum[arr[j].first];
                    --sum[arr[j].second];
                }
            }
            for(int j = x ; j <= y ; ++j){
                sum[j] += sum[j - 1];
            }
            for(int j = x ; j <= y ; ++j){
                sum[j] = min(sum[j] , 1);
            }
            bool res = 1;
            for(int j = x ; j < y ; ++j){
                if(!sum[j]){
                    res = 0;
                    break;
                }
            }
            for(int j = x ; j <= y ; ++j){
                sum[j] = 0;
            }
            if(res){
                ok = 1;
                break;
            }
        }
        if(ok){
            printf("Yes\n");
        }
        else{
            printf("No\n");
        }
    }
}
