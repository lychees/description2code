int primes[1000001],fl[1000001];
void findPrimes()
{
    int i,j,lmt=1000000,sq=sqrt(lmt);
    for(i=3;i<=sq;i+=2)
    {
        if(fl[i]) continue;
        for(j=i*i;j<=lmt;j+=2*i)
        {
            fl[j]++;
        }
    }
    primes[0]=2;
    int k=0;
    for(i=3;i<lmt;i+=2)
    {
        if(!fl[i]) primes[k++]=i;
    }
    //for(i=0;i<50;i++) printf("%d\n",primes[i]);
    //printf("%d\n",k);
}
