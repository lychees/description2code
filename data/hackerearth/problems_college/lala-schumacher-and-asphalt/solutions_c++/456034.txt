
#include<stdio.h>
#include<iostream>
using namespace std;
#define gc getchar

inline int gi()
{
int n=0;
int si=1;
int c=gc();
while(c<48||c>57)
{
    if(c=='-')
    si=-1;
c=gc();
}
while(c>47&&c<58)
{n=(n<<1)+(n<<3)+c-48;
c=gc();
}
if(si==-1)
    return -1*n;
return n;
}
int main()
{
    int t,n,i;
   t=gi();
    while(t--)
    {
        n=gi();
        int a[n];
        int b[n];
        int fx,fy,k,l;
        long long qx,qy,wx,wy,s=1000000000000;
        for(i=0;i<n;i++)
            a[i]=gi();
                for(i=0;i<n;i++)
            b[i]=gi();

            fx=gi();fy=gi();l=gi();k=gi();

            for(i=0;i<n;i++)
            {qx=a[i];
            qy=b[i];
            wx=a[i]-fx;
            wy=b[i]-fy;
            if(qx<0)
                qx=-1*qx;
            if(qy<0)
                qy*=-1;
            if(wx<0)
                wx*=-1;
            if(wy<0)
                wy*=-1;

        if(s>(qx+qy)*l+(wx+wy)*k)
            s=(qx+qy)*l+(wx+wy)*k;

            }
            if(fx<0)
                fx=-1*fx;
            if(fy<0)
                fy*=-1;
            if(s>l*(fx+fy))
                s=l*(fx+fy);
            cout<<s<<'\n';
    }
    return 0;

}
