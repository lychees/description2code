#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,ans,l,e,i;
    pair<double,int>p[1005];
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	
    	for(i=1;i<=n;i++)
    	{
    		cin>>e;
    		
    		p[i]=make_pair(i*1.0,e);
    	}
    	
    	e = n;
    	l=1;
    	
    	while(l + 1 < e )
    	{
    		for(i=l;i<=e;i++)
    		{
    			p[i].first = p[i].first + 0.5*p[i].second;
    			
    			if(i>l)
    			if(p[i].first == p[i-1].first)
    			{
    			p[i].second = - p[i].second;
    				p[i-1].second = - p[i-1].second;
    			}
    			
    			if(i<e)
    			if(p[i].first == p[i+1].first)
    			{
    				p[i].second = - p[i].second;
    				p[i+1].second = - p[i+1].second;
    			}
    			
    			
    		}
    			if(p[l].first < 0.0)
    				l++;
    			
    			
    			if(p[e].first > n+1)
    				e--;
    			
    			
    	}
    	
    	if(p[l].second == p[e].second)
    	{
    		if(p[l].second ==1)
    		ans = l;
    		else
    		ans =e;
    	}else
    	{
    
    			if(p[l].first <= n + 1.0 - p[e].first )
    			ans = l;
    			else
    			ans = e;
    			
    		
    	}
    	
    	cout<<ans<<'\n';
    }
    
    return 0;
}
