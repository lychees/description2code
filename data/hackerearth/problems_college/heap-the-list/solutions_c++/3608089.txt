#include<iostream>
#include<algorithm>
using namespace std;
main()
{
    int i,a[1000],min,cnt=0;
    for(i=0;i<100;i++)
    {
        cin>>a[i];

        	if(a[i]==0)
            {
                cnt++;
            }
            else if(a[i]!=0)
            {
                cnt=0;
            }

    }
   for(i=0;i<100-cnt;i++)
    {
        if(i==0)
        min=a[i];
        else if(a[i]<min)
        {
            min=a[i];
        }
    }

 cout<<min<<endl;

}
