#include <iostream>
using namespace std;

int main()
{
    string a,b;
    cin>>a>>b;
    
    if(a.size()==b.size()){
    	int f=0,c=0;
    	for(long int i=0;i<a.size();i++){
    		if(a[i]!=b[i])c++;
    		if(c>1){
    			f=1;
    			break;
    		}
    	}
    	if(f==0)cout<<"wins"<<endl;
    	else cout<<"loses"<<endl;
    }
    else cout<<"loses"<<endl;
    return 0;
}
