#include<iostream>
#include<bits/stdc++.h>
#include<string.h>

using namespace std;

int main()
{
    string s;
    cin>>s;
    for(int i = s.length()-1; i >= 0; i--)
        s[i] = s[i] ^ (0xf0fffff0 | (0xffffffff & 0x0f00000f));
    for(int i = 0; i < s.length(); i++)
        s[i] = s[i] ^ (0xf0ff0ff0 | (0xffffffff & 0x0f00f00f));
    for(int i = s.length()-1; i >= 0; i--)
        s[i] = s[i] ^ (0xf0ff00f0 | (0xffffffff & 0x0f00ff0f));

    for(int i = 0; i < s.length(); i++)
        cout<<int(s[i])<<" ";
    cout<<endl;
    return 0;
}
