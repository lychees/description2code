#include <bits/stdc++.h>
#define MOD 1000000007
#define ll long long
#define LL long long
using namespace std;

LL powerMod(LL a, LL p, LL mod=MOD){
    LL ret = 1;
    a %= mod;
    while(p > 0){
        if(p & 1){
            ret *= a;
            ret %= mod;
        }
        a *= a;
        a %= mod;
        p >>= 1;
    }
    return ret;
}

int main()
{
	int tc;
	cin >> tc;
	while(tc--)
	{
		int n;
		cin >> n;
		ll a[20];
		for(int i=0;i<n;i++) cin >> a[i];
		ll mx = 0;
		ll finans = 0;
		for(int i=1;i< 1 << n;i++)
		{
			vector<ll>here;
			if(__builtin_popcount(i) & 1) continue;
			
			for(int j=0;j<n;j++)
			{
				if(i & 1 << j)
				{
					here.push_back(a[j]);
				}
			}
			sort(here.rbegin(), here.rend());
			ll hereans = 1;
			for(int j=0;j<here.size();j++)
			{
				if(j < here.size()/2)
				{
					hereans *= here[j];
					hereans %= MOD;
				}
				else
				{
					hereans *= powerMod(here[j], MOD-2);
					hereans %= MOD;
				}
			}
			finans = max(finans, hereans);
		}
		cout << finans << "\n";
	}
    return 0;
}
