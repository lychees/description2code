#include<bits/stdc++.h>
using namespace std;
#define ll	long long int
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		ll ar[n],cnt=0;
		for(int i=0;i<n;++i)
		{
			scanf("%lld",&ar[i]);
			cnt+=ar[i];
		}
		ll m;
		scanf("%lld",&m);
		m%=cnt;
		int ans;
		if(m==0)
		{
			for(int i=n-1;i>=0;--i)
			{
				if(ar[i]!=0)
				{
					ans=i;
					break;
				}
			}
		}
		else
		{
			for(int i=0;i<n;++i)
			{
				m-=ar[i];
				if(m<=0)
				{
					ans=i;
					break;
				}
			}
		}
		printf("%d\n",ans+1);
	}
	return 0;
}
