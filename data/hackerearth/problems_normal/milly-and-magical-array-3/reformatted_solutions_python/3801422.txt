import math


def Milly():
    n, x = map(int, raw_input().strip().split(' '))
    arr = map(float, raw_input().strip().split(' '))
    count = 0
    for i in xrange(n - 1):
        if arr[i] == arr[i + 1]:
            count += 1
            arr[i + 1] += x
        elif arr[i] > arr[i + 1]:
            temp = int(math.ceil((arr[i] - arr[i + 1]) / x))
            count += temp
            arr[i + 1] += (temp * x)
    return count


for i in xrange(int(raw_input().strip())):
    print Milly()
