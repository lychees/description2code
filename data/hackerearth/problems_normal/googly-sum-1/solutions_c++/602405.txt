#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<vector>
#include<map>
using namespace std;
map<int,int> m;
int hashh[400007]={0};
int main()
{

    m.clear();
    int n,a;
    scanf("%d",&n);
    vector<int> v;
    for(int i=0;i<n;i++)
    {
        scanf("%d",&a);
        v.push_back(a);
    }
    int mini=200000;
    int ans=0;
    hashh[v[0]*2+mini]=1;
    for(int i=1;i<n;i++)
    {
        for(int j=i-1;j>=0;j--)
        {
                if(hashh[v[i]-v[j]+mini]){
                ans++;
                break;}
        }
        hashh[v[i]*2+mini]=1;
        for(int j=i-1;j>=0;j--)
                hashh[v[i]+v[j]+mini]=1;
    }
    printf("%d\n",ans);
    return 0;
}
