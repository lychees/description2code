#include<iostream>
using namespace std;
int main()
{
int t;
scanf("%d",&t);
while(t--)
{
int n,i,flag=1;
scanf("%d",&n);
int a[n+1];
for(i=1;i<=n;i++)
{
scanf("%d",&a[i]);

}
for(i=1;i<=n;i++)
{
if(i!=a[a[i]])
{
flag=0;
break;
}

}
if(flag)
printf("inverse\n");
else
printf("not inverse\n");
}
return 0;
}
