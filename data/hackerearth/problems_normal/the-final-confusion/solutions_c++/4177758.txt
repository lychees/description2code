#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll nCr(int n, int r){
	ll result = 1;
	for(int i = 0; i < r; i++){
		result *= (n - i);
		result /= (i + 1);
	}
	return result;
}
int main()
{
    int T,N;
    scanf("%d",&T);
    while(T--){
    	scanf("%d",&N);
    	ll ans = nCr(2*N,N);
    	ans /= (N + 1);
    	cout << ans << "\n";
    }
    return 0;
}
