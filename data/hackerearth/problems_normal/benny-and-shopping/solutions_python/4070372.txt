import re

pattern = r'\$[\s\d]+'

num_of_input = input()
inputs = []
for i in range(num_of_input):
    inputs.append(raw_input())


for sample_input in inputs:
    try:
        price_with_symbol = re.search(pattern, sample_input).group(0)
        price_with_symbol =  price_with_symbol.replace(' ','')
        price = re.search('\d+', price_with_symbol).group(0)
        print "$%s" % int(price)
    except:
        next
