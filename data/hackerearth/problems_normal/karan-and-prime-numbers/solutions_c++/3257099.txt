#include <iostream>
#include<bits/stdc++.h>
//#include<stdio.h>
//#include<cstdlib>
using namespace std;

void prime(int a,int b)
{
	long int sum=0;
	bool prime[b+1];
	memset(prime,true,sizeof(prime));
	for(int i=2;i*i<=b;i++)
	{
		if(prime[i]==true)
		{
		for(int j=2*i;j<=b;j+=i)
		prime[j]=false;
		}
	}
	for(int i=2;i<=b;i++)
	{
		if(prime[i]==true&&i>=a)
		sum+=i;
	}
	cout<<sum<<endl;
//	return sum;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int a,b;
    	cin>>a>>b;
    	prime(a,b);
    }
    return 0;
}
