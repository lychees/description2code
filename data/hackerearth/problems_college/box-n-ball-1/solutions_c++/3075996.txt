#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define min(a,b) a>b?b:a
int main()
{
    lli T,L,B,H;
    cin>>T;
    while(T--)
    {
        cin>>L>>B>>H;
        long double min_val = min(L,B);
        min_val=min(min_val,H);
        //cout<<min_val<<endl;
        cout<<fixed;
        cout<<setprecision(1)<<min_val/2<<endl;
    }

    return 0;

}
