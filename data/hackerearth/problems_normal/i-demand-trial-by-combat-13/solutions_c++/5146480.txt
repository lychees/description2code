#include <iostream>
using namespace std;

int main() {

	long long int t,n,m,i,j,k,c;
	cin>>t;
	while(t--)
	{
	
		cin>>n>>m;
		long long int a[n];
		for(i=0;i<n;i++)
			cin>>a[i];
		for(i=0;i<m;i++)
		{
			long long int t[n];
			t[0]=(a[1]==1) ? 1 : 0;
			for(j=1;j<n-1;j++)
			{
				if(a[j-1]==1 && a[j+1]==1)
					t[j]=1;
				else
				{
					t[j]=0;
				}
			}
			t[n-1]=(a[n-2]==1) ? 1 : 0;
			c=0;
			for(k=0;k<n;k++){
				a[k]=t[k];
				if(a[k]==1)
					c=1;
				}
			if(c==0)
				break;

		}
		for(k=0;k<n;k++)
		{
			if(c==0 && m>0)
				cout<<"0"<<" ";
			else
				cout<<a[k]<<" ";
		}
		cout<<endl;
	}

	return 0;
}
