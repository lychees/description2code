#include<iostream>
#include<climits>
#define ull long long int
using namespace std;

ull max(ull x, ull y)
{ return (y > x)? y : x; }
 
ull maxSubArraySum(int a[], ull size)
{
   ull max_so_far = a[0], i;
   ull curr_max = a[0];
 //printf("%lld %lld\n",max_so_far,curr_max);
   for (i = 1; i < size; i++)
   {
   		//printf("before %lld %lld\n",curr_max,max_so_far);
        curr_max = max(a[i], curr_max+a[i]);
        max_so_far = max(max_so_far, curr_max);
        //printf("after %lld %lld\n",curr_max,max_so_far);
   }
   return max_so_far;
}
