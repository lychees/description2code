def noOfString(string, noOfDays):
	sumOfDays = (noOfDays * (noOfDays + 1)) / 2 
	countZero = 0
	countOne = 0
	for i in string:
		if(i == "0"):
			countZero += 1
		else:
			countOne += 1
	if(sumOfDays % 2 == 0):
		return countOne
	else:
		return countZero
