bool isprime[200];
int p[20],c[20];
bool prime(int n){
	for(int i=2;i*i<=n;i++){
		if(n%i==0) return false;
	}
	return true;
}
void pre(){
	isprime[0]=isprime[1]=false;
	isprime[2]=true;
	fr(i,3,200){
		if(prime(i)) isprime[i]=true;
		else isprime[i]=false;
	}
}
int main(){
	pre();
	int t,n,pp,pc,x,y;
	si(t);
	while(t--){
		si(n);
		f(i,n) si(c[i]);
		f(i,n) si(p[i]);
		pp=0;pc=0;
		f(i,n){
			x=c[i]; y=p[i];
			if(isprime[x] && !isprime[y]){
				if(x>y) pc+=x-y;
				else pp+=y-x;
			}
			else if(!isprime[x] && isprime[y]){
				if(x>y) pc+=x-y;
				else pp+=y-x;
			}
			else if(!isprime[x] && !isprime[y]) pc+=x;
			else if(isprime[x] && isprime[y]) pp+=y;
		}
		if(pp>pc) printf("The Primes\n");
		else if(pc>pp) printf("The Composites\n");
		else printf("Tie\n");
	}
	return 0;
}
