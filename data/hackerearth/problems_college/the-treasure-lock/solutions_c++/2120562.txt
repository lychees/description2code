	    			n=n-5;
	    			count++;
	    		}
	    	}
	    	if(n>0)
	    	{
	    		int cc=n/3;
	    		for(int i=0;i<cc*3;i++)
	    			cout<<"5";

	    		 	cout<<"3";
	    		
	    	}

    	}
    	
    }

}
