#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    long int n,i,j;
    scanf("%d",&t);
    j=1;
    while(t--)
    {
    	scanf("%ld",&n);
    	long long int a[n],s[n],ans=0,sum=0;
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&a[i]);
    		sum+=a[i];
    	}
    	if(n==1)
    	{
    		printf("Case %ld: %lld\n",j,sum);
    		j++;
    		continue;
    	}
    	if(n==2)
    	{
    		if(a[0]<a[1])
    		ans=a[1];
    		else
    		ans=a[0];
    		printf("Case %ld: %lld\n",j,ans);
    		j++;
    		continue;
    	}
    	else
    	{
    		s[0]=a[0];
    		if(a[0]>a[1])
    			s[1]=a[0];
    		else
  				s[1]=a[1];  		
  			for(i=2;i<n;i++)
  			{
  				if(s[i-1]>s[i-2]+a[i])
  				s[i]=s[i-1];
  				else
  				s[i]=s[i-2]+a[i];
  			}
    	}
    	   	
    	printf("Case %ld: %lld\n",j,s[n-1]);
    	j++;
    }
    
    
    
    
    return 0;
}
