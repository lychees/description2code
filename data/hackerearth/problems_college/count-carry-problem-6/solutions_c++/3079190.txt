#include <bits/stdc++.h>
#define ll long long
#define INF 1000000000
#define M 2147483648
#define ei else if
#define MAX 50000000
#define fori(l,r,x) for (int i=l;i<=r;i+=x)
#define forj(l,r,x) for (int j=l;j<=r;j+=x)
using namespace std;

int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		ll int a,b,c=0,x=0;
		scanf("%lld%lld",&a,&b);
		while(a!=0&&b!=0){
			ll int d1 = a%10;
			ll int d2 = b%10;
			if (d1+d2+c>9){x++;c=1;}
			a/=10;
			b/=10;
		}
		if (x==0) printf("No carry operation\n");
		else if (x==1) printf("1 carry operation\n");
		else printf("%d carry operations\n",x);
	}
	return 0;
}
