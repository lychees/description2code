#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int a,b;
        cin>>a>>b;
        int x=a/b;
        int y=x+1;
        if(a==x*b||a==y*b) cout<<a<<endl;
        else if(2*a==((x*b)+(y*b))) cout<<y*b<<endl;
        else if((y*b-a)<(a-x*b)) cout<<y*b<<endl;
        else cout<<x*b<<endl; 
    }
    return 0;
}
