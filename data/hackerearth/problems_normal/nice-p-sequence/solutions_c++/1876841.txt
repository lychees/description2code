double EPS = 1e-10;

ll gcd(ll a, ll b) {
	if (b == 0) return abs(a);
	else return gcd(b, a % b);
}

ll extgcd(ll a, ll b, ll& x, ll& y) {
	if (b == 0) {
		x = (a >= 0) ? 1 : -1;
		y = 0;
		return abs(a);
	}
	else {
		ll res = extgcd(b, a % b, y, x);
		y -= (a / b) * x;
		return res;
	}
}

ll mod_inverse(ll a, ll m) {
	ll x, y;
	extgcd(a, m, x, y);
	return (m + x % m) % m;
}

int main() {
	int T; cin >> T;
	while (T--) {
		ll a, p, n; scanf("%lld%lld%lld", &a, &p, &n);
		if (gcd(a, p) != 1) {
			printf("-1\n");
			continue;
		}
		ll b = mod_inverse(a, p);
		if (n % 2 == 0) printf("%lld\n", (a + b) * (n / 2));
		else printf("%lld\n", (a + b) * (n / 2) + a);
	}
}
