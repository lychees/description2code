#include<bits/stdc++.h>
#include<algorithm>
#include<cstdio>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int a,b,n;
    	cin>>a>>b>>n;
    	int i,arr[n];
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&arr[i]);
    	}
    	sort(arr,arr+n);
    	int max1=0,max2=0,sum=0;
    	i=0;
    	while(sum<=a)
    	{
    		sum=sum+arr[i];
    		max1++;
    		i++;
    	}
    	i=0;sum=0;
    	while(sum<=b)
    	{
    		sum=sum+arr[i];
    		max2++;
    		i++;
    	}
    	if(max1>max2)
    	{
    		printf("Raghu Won\n");
    	}
    	else if(max2>max1)
    	{
    		printf("Sayan Won\n");
    	}
    	else
    	printf("Tie\n");
    }
    return 0;
}
