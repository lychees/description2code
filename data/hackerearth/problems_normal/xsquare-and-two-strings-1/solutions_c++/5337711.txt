#include <iostream>
#include <vector>

using namespace std;

bool isPossible(const string & s, const string & t)
{
	vector<int> countS(26, 0), countT(26, 0);
	
	for(int i = 0, length = s.size(); i < length; ++i) {
		countS[s[i] - 'a'] += 1;
	}
	
	for(int i = 0, length = t.size(); i < length; ++i) {
		countT[t[i] - 'a'] += 1;
	}
	
	// for (int i = 0; i < 26; ++i) {
	// 	cout << countS[i] << " " << countT[i] << "\n";
	// }
	for(int i = 0, length = 26; i < length; ++i) {
		if (countS[i] > 0 && countT[i] > 0) {
			return true;
		}
	}
	
	return false;
}

int main()
{
	int testCases = 0;
	string S1, S2;
	cin >> testCases;
	while(testCases--){
		cin >> S1 >> S2;
		if(isPossible(S1, S2)) {
			cout << "Yes" << "\n";
		} else {
			cout << "No" << "\n";
		}
	}
}
