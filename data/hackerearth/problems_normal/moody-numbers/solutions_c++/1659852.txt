typedef unsigned long long int uint64;
typedef long long int int64;
 
int vis[200];
int counter=1;
int sum_dig(int64 x){
	int ret=0;
	while(x){
		ret+=x%10;
		x/=10;
	}
	return ret;
}
 
int go(int64 val){
	if((val==1)||(val==4)){
		return 1;
	}
	int64 sqr=val*val;
	int x=sum_dig(sqr);
	if(vis[x]==counter)
	return 0;
	vis[x]=counter;
	return go(x);
}
 
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int64 n;
		scanf("%lld",&n);
		counter++;
		if(go(n))
		printf("YES\n");
		else
		printf("NO\n");
	}
	return 0;
}
