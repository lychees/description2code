#include <iostream>
using namespace std;

int main()
{
	int T,i=0,c;
    string a,b="",t="";
    cin>>T;
    while(T--)
    {
    	cin>>a;
    	b="";t="";i=0;
    	while(a[i]!='\0')
    	{
    		if(isalpha(a[i]))
				b+=a[i];
			else
			{
				t=b+"";b="";
    			c=(int)(a[i]-'0');
				while(c-->0)
					b+=t;
			}
			i++;
    	}
    	cout<<b<<endl;
    }
    return 0;
}
