#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <iterator>
#define ll long long int
#define pb(x) push_back(x)
#define mp(x,y) make_pair(x,y)
using namespace std;

int main()
{
	ll n,i,a,b,x,s,e,ans=0;
	cin>>n>>s>>e;
	vector<pair<ll,ll> >v;
	for(i=0;i<n;i++)
	{
		cin>>a>>b;
		v.pb(mp(a-b,a+b));
	}
//	cin>>s>>e;
   sort(v.begin(),v.end());
   
 /*  for(i=0;i<n;i++)
	{
		
		cout<<v[i].first<<"  "<<v[i].second<<endl;
	} */
   
   x=s;
   if(v[0].first>s&&v[0].first<e)
   {
   ans=ans+v[0].first-s;
   if(x>v[0].second)
   x=min(x,e);
   else
   x=min(v[0].second,e);}
   
   for(i=1;i<n;i++)
   {
   	if(v[i].first>=e)
   	break;
   	if(v[i].first>x)
   	ans=ans+v[i].first-x;
   	
   	if(x>v[i].second)
   	x=min(x,e);
   	else
   	x=min(v[i].second,e);
   }
   
   cout<<ans+e-x;
   return 0;
}
