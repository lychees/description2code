#include <bits/stdc++.h>
using namespace std;
#define pb push_back
#define lli long long int
const lli N=1e10;
vector<lli>v;
void pre(){
	v.pb(3);
	v.pb(5);
	lli x=3,i=0;
	while(x<N){
		lli tt=v[i]*10+3;
		v.pb(tt);
		tt=v[i]*10+5;
		v.pb(tt);
		i++;
		x=v[v.size()-1];
	}
}
int main(){
	pre();
	//sort(v.begin(),v.end());
	int t;
	cin>>t;
	while(t--){
		int a,b;
		cin>>a>>b;
		int it1=(lower_bound(v.begin(),v.end(),a-1)-v.begin());
		int it2=(upper_bound(v.begin(),v.end(),b)-v.begin());
		cout<<(it2-it1)<<"\n";
	}
  
    return 0;
}
