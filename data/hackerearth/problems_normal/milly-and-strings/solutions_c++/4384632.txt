

    #include <iostream>
    #include <stdio.h>
    #include <stdlib.h>
    #include <map>
    #include <assert.h>
    #include <string>
    #include <cstring>
    using namespace std;
    int dp[100001];
    int main()
    {
    	int t;
    	scanf("%d", &t);
    	while(t--) {
    		int q;
    		string s;
    		cin >> s;
    		int n = s.length();
    		memset(dp, 0, sizeof(dp));
    		assert(3 <= n && n <= 100000);
    		scanf("%d", &q);
    		assert(1 <= q && q <= 100000);
    		if (s[0] == s[1] && s[1] == s[2]) {
    			dp[1] = 1;
     		}
     		for (int i = 1; i < n - 2; i++) {
                if (s[i] == s[i + 1] && s[i + 1] == s[i + 2]) {
                    dp[i + 1] = dp[i] + 1;
                } else {
                    dp[i + 1] = dp[i];
                }
            }
            dp[0] = 0;
            dp[n-1] = dp[n-2];
            dp[n] = dp[n-2];
    		while(q--) {
    			int l, r;
    			scanf("%d%d", &l, &r);
    			assert(1 <= l && l <= r && r <= n - 2);
    			printf("%d\n", dp[r] - dp[l-1]);
    		}
    	}
    	return 0;
    }
