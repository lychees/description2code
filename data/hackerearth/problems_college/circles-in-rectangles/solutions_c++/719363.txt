#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	long long int l, b, d, A=0, AC=0;
	int t;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%lld%lld%lld", &l, &b, &d);
		A=A+(l/d);
		AC=AC+(b/d);
		A=((A%1000000007)*(AC%1000000007))%1000000007;
		printf("%lld\n", A);
		A=0;
		AC=0;
	}
	return 0;
}
