#include <iostream>
#include<string.h>

using namespace std;
bool isSubString(string b , string s) {
	int i  = 0;
	int c = 0;
	int lb = b.length();
	int ls = s.length();
	int f = 0;
	for(i = 0; i < lb ; i ++) {
		if(b[i] == s[c] && c < ls) {
			c++;
			f = 1;
		}
	}
	if( c == ls && f == 1)
		return true;
	return false;
	
}
int main()
{
    //cout << "Hello World!" << endl;
    int  k , i , t;
    i = 0;
    string b , s ;
    cin>>b>>k;
    cin>>t;
    while(t--) {
    	cin>>s;
    	if(isSubString(b , s))
    		i++;
    }
    cout<<i<<" ";
    if(i >= k)
    	cout<<"LUCKY\n";
    else cout<<"UNLUCKY\n";
    return 0;
}
