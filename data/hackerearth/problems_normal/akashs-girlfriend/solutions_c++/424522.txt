#include<bits/stdc++.h>
#define MOD 1000000009
using namespace std;
long long int expon(long long int base,long long int  exp)
{
	long long int res=1;
	while(exp>0)
	{
		if(exp%2==1)
		{
			res=(res*base)%MOD;
			
		}
		
		base=(base*base)%MOD;
		exp/=2;
	}
	return res%MOD;
}
long long int abson(long long int w)
{
	if(w<0)
	return MOD + w;
	return w;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n;
		cin>>n;
		long long int ans;
		ans=((expon(10,n)-expon(8,n))*expon(2, MOD-2))%MOD;
		cout<<abson(ans)<<endl;
	}
	return 0;
}
