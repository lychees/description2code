#include <iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--) {
		string str;
		cin>>str;
		int flag=0,len=str.length();
		for(int i=0,j=len-1;i<j;i++,j--) {
			if(str[i]!=str[j]){flag=1;break;}
		}
		if(flag) {
			cout<<"-1\n";
		} else {
			char ch=str[len/2];
			cout<<(int)ch<<endl;
		}
	}
	return 0;
}
