void calc(int indx, int par)
{
    subSize[indx]=a[indx];
    int i;
    int prev=-1;
    for(i=0;i<v[indx].size();i++)
    {
        if(v[indx][i]==par)
        {
            continue;
        }
        calc(v[indx][i],indx);
        subSize[indx]+=subSize[v[indx][i]];
    }
}

void dp(int indx, int par)
{
    int i,j;
    for(i=0;i<v[indx].size();i++)
    {
        if(v[indx][i]==par)
        {
            continue;
        }
        dp(v[indx][i],indx);
    }
    subSize[indx]=a[indx];
    memo[indx][a[indx]]=1;
    for(i=0;i<v[indx].size();i++)
    {
        if(v[indx][i]==par)
        {
            continue;
        }
        int x=v[indx][i];
        /*
        if(indx==0)
        {
            cout<<memo[x][0]<<endl;
        }
        */
        {
            CLR(tmp);
            int i,j;
            for(i=0;i<=k && i<=subSize[indx];i++)
            {
                for(j=0;j<=k-i && j<=subSize[x];j++)
                {
                    tmp[i+j]=(tmp[i+j]+((lld)memo[indx][i]*memo[x][j])%mod)%mod;
                    /*
                    if(indx==0)
                    {
                        cout<<x<<" ("<<i<<","<<j<<") "<<memo[indx][i]<<" "<<memo[x][j]<<endl;
                    }
                    */
                }
            }
            for(i=0;i<=k && i<=subSize[indx]+subSize[x];i++)
            {
                memo[indx][i]=(memo[indx][i]+tmp[i])%mod;
            }
            subSize[indx]+=subSize[x];
        }
    }
}
