#include<stdio.h>
#include<iostream>
#include<string.h>
#include<stdlib.h>
using namespace std;

int main()
{
	char s[10000];
	long long t;
	cin>>t;
	while(t--){
		cin>>s;
		int st,en;
		st=1;
		en=strlen(s)-1;
		int n=en+1;
		while(st<n&&en>0){
			//cout<<s[st]<<s[st-1]<<s[en-1]<<s[en]<<abs(s[st]-s[st-1])<<abs(s[en-1]-s[en])<<endl;
			if(abs(s[st]-s[st-1])!=abs(s[en-1]-s[en]))
				break;
			st++;
			en--;
		}
		
		if(st==n)
			cout<<"Funny"<<endl;
		else
			cout<<"Not Funny"<<endl;
	}
}
