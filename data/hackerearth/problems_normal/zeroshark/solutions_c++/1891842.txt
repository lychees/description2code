#include <bits/stdc++.h>
#define max 100001
#define mod 1000000007
using namespace std;
long long a[max],b[max];
int main(){
	int t,n,i;
	cin >> t;
	a[1]=b[1]=0;
	a[2]=b[2]=1;
	for(i=3;i<max;i++){
		a[i] = (a[i-1]+a[i-2])%mod;
	}
	for(i=3;i<max;i++){
		b[i] = (b[i-1]+b[i-2]+a[i])%mod;
	}
	while(t--){
		cin >> n;
		cout << b[n] << endl;
	}
	return 0;
}
