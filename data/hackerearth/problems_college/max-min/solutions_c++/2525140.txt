#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
int n,k;
cin>>n;
cin>>k;
int arr[n];
for(int i=0;i<n;i++)
cin>>arr[i];
sort(arr,arr+n);

int min=arr[k-1]-arr[0];
for(int i=k;i<n;i++)
{
	if(arr[i]-arr[i-k+1]< min)
	min=arr[i]-arr[i-k+1];
}
cout<<min<<endl;

    return 0;
}
