#include <iostream>
#include <cmath>

using namespace std;

int main()
{
	int T, L, R, K;
	cin >> T;
	
	while(T--) {
		int output = 1;
		cin >> L >> R >> K;
		int temp = L;

		while(1) {
			temp = temp * K + 1;
			if (temp <= R) {
				output++;
			}
			else {
				break;
			}
		}
		cout << output<< endl;
		
	}
    return 0;
}
