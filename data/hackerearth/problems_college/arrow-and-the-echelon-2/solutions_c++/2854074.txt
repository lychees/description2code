#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int x,five,three;
        cin>>x;
        five=x;
        three=0;
        while(five>=0)
        {
             if(five<3)
                break;
            if(five%3==0&&three%5==0)
            {
                break;
            }
            else if(five>=5)
            {
                five-=5;
                three+=5;
            }


        }
        if(five%3!=0)
            cout<<-1<<endl;
        else
        {
            for(int i=0;i<five;i++)
                cout<<5;
            for(int i=0;i<three;i++)
                cout<<3;
            cout<<endl;
        }
    }
    return 0;
}
