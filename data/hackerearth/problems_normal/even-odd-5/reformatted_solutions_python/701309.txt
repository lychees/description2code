import sys

data = sys.stdin.readlines()

numTrials = data.pop(0)
matrices = [[int(numString) for numString in row.rstrip().split(" ")]
            for row in data]

for matrix in matrices:
    rows, columns = matrix[0], matrix[1]

    if rows > columns:
        square = columns
        extraRows = rows - columns
        evenSquareRows = square//2
        oddSquareRows = square//2 + square % 2

        squareEvens = evenSquareRows*(evenSquareRows+1)
        squareOdds = (oddSquareRows-1)*(oddSquareRows+1)+1

        startsEven = ((square + 1) % 2 == 0)
        endsEven = (rows % 2 == 0)

        if startsEven:
            if endsEven:
                evenExtraRows = extraRows//2+1
                oddExtraRows = extraRows//2
            else:
                evenExtraRows = extraRows//2
                oddExtraRows = extraRows//2
        else:
            if endsEven:
                evenExtraRows = extraRows//2
                oddExtraRows = extraRows//2
            else:
                evenExtraRows = extraRows//2
                oddExtraRows = extraRows//2+1

        extraEvens = square * evenExtraRows
        extraOdds = square * oddExtraRows

        evens = squareEvens + extraEvens
        odds = squareOdds + extraOdds

    else:
        evenRows = rows//2
        oddRows = rows//2 + rows % 2

        evens = evenRows*(evenRows+1)
        odds = (oddRows-1)*(oddRows+1)+1

    sys.stdout.write(str(evens) + " " + str(odds) + "\n")
