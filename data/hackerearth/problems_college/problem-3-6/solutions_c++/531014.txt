#include <iostream>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
	int n;
	cin>>n;
	int x[n], y[n], a[n], b[n];
	for(int i=0;i<n;i++)
	{
		cin>>x[i]>>y[i];
	}
	for(int i=0;i<n;i++){
		a[i]=n-1;
		b[i]=n-1;
	}
	for(int i=0;i<n;i++){
		for(int j=0;j<n&&j!=i;j++){
			if(x[i]==y[j]){
				a[j]++;
				b[j]--;
			}
            if(x[j]==y[i]){
                a[i]++;
                b[i]--;
            }
		}
	}
	for(int i=0;i<n;i++){
		cout<<a[i]<<" "<<b[i]<<endl;
	}
    return 0;
}
