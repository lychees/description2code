#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;


bool my_comp(pair<int, char> p1, pair<int, char> p2)
{
	if(p1.first < p2.first || (p1.first == p2.first && p1.second > p2.second))
	{
		return true;
	}
	return false;
}


int main()
{
	int t, i;
	string str;
    cin>>t;
    while(t--)
    {
    	vector<pair<int, char> > v(26);
    	//cout<<v.size();
    	for(i=0; i<26; i++)
    	{
    		v[i].second = char(i+97);
    		v[i].first = 0;
    	}
    	cin>>str;
    	for(i=0; i<str.length(); i++)
    	{
    		v[str[i]-97].first++;
    	}
    	sort(v.begin(), v.end(), my_comp);
    	//cout<<v[0].second;
    	for(i=0; i<26; i++)
    	{
    		cout<<v[i].second<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
