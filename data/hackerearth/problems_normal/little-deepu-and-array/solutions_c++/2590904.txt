#include <iostream>
using namespace std;

int main()
{
    int n;
    long unsigned int a[100000];
    cin>>n;
    for(int i=0;i<n;i++)
    	cin>>a[i];
    int m;
    cin>>m;
    while(m)
    {
    	int num;
    	cin>>num;
    	for(int i=0;i<n;i++)
    	{
    	if(a[i]>num)
    		a[i]=a[i]-1;
    	}
    	m--;
    }
    for(int i=0;i<n;i++)
    	cout<<a[i]<<" ";
    
    return 0;
}
