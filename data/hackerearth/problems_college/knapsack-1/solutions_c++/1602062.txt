#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--){
		int n,k;
		cin>>n>>k;
		int dp[2002],a[n],i,j;
		memset(dp,0,sizeof(dp));
		for(i=0;i<n;i++){
			cin>>a[i];
		}
		dp[0] = 1;
		for(i=0;i<n;i++){
			for(j=a[i];j<=2000;j+=a[i]){
				dp[j] = 1;
				
			}
		}
		for(i=1;i<=2000;i++){
			for(j=0;j<n;j++){
				if(i >= a[j]){
					dp[i] = (dp[i] || dp[i-a[j]]);
				}
			}
		}
		for(i=k;i>=0;i--){
			if(dp[i] == 1){
				break;
			}
		}
		printf("%d\n",i);
	}
	return 0;
}
