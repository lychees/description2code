#include <bits/stdc++.h>
using namespace std;
int n, m;
double pro[1008][1008], dp[1008][1008];
int main() {
	// your code goes here
	int t;
	cin >> t;
	while(t--){
		
		cin >> n >> m;
		for(int i = 1; i <= n; i++) {
			for(int j = 1; j <= m; j++){
				scanf("%lf", &pro[i][j]);
				dp[i][j] = 0.0;
			}
		}
		for (int i=1;i<=n+1;i++)
	 	for (int j=1;j<=m+1;j++)
	  		dp[i][j]=0;
		dp[1][1] = 1.0;
		for(int i = 1; i <= n; i++){
			for(int j = 1; j <= m; j++){
				if (dp[i][j] < 1e-20) continue;
				dp[i + 1][j] += dp[i][j] * (1.0 - pro[i][j]);
				dp[i][j + 1] += dp[i][j] * pro[i][j];
			}
		}
		double ans = 0.0;
		for(int i = 1; i <= n; i++){
			ans += (m + i - 1) * dp[i][m + 1];
		}
		for(int i = 1; i <= m; i++){
			ans += (n + i - 1) * dp[n + 1][i];
		}
		cout.precision(6);
     	cout << fixed << ans << endl;
	}
	return 0;
}
