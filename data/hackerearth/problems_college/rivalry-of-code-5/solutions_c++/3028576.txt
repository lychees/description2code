#include <iostream>
#include <string>
using namespace std;

void swapstring(string &s1, string &s2){
	string temp = s1;
	s1 = s2;
	s2 = temp;
}

int main()
{
	int n, l1, l2, ans, chng;
	string s1, s2;
	cin >> n;
	for(int i =0; i < n ; i++){
		ans = 6000;
		chng = 0;
    	cin >> s1 >> s2;
		l1 = s1.length();
		l2 = s2.length();
		if(l1>l2){
			swapstring(s1, s2);
			l1 = s1.length();
			l2 = s2.length();
		}
		for(int j =0; j <= l2-l1; j++){
			chng = 0;
			for(int k=0; k < l1; k++){
				if(s1[k]!=s2[j+k])
					chng++;
			}
			ans = (ans > chng) ? chng : ans;
		}
		cout <<ans << endl;
	}
    return 0;
}
