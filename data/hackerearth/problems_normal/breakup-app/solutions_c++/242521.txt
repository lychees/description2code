#include <iostream>
using namespace std;

char str[1005];
int cnt[50];

int main()
{
	int n, x, w,max;

	cin >> n;
	getchar();
	while (n--)
	{
		cin.getline(str, 1005);

		if (str[0] == 'G')
			w = 2;
		else
			w = 1;

		x = 0;

		for (int i = 0; str[i]; i++)
		{
			if ((str[i] >= '0') && (str[i] <= '9'))
			{
				x = (x * 10) + (str[i] - 48);

			}
			else
			{

				cnt[x] += w;
				x = 0;
			}

		}

		cnt[x] += w;

	}

	max = 1;
	for (int i = 2; i<31; i++)
	if (cnt[i]>cnt[max])
		max = i;

	if ((max == 19) || (max == 20))
		cout << "Date" << endl;
	else
		cout << "No Date" << endl;



	return 0;
}
