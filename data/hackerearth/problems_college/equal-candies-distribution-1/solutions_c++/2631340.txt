#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--) {
    	int n;
    	cin>>n;
    	int a[n], b[n], s, c=0;

    	for(int i=0; i<n; i++)
    		cin>>a[i];
    	sort(a, a+n);
    	for(int i=0; i<n; i++)
    		b[i] = a[i] - a[0];

    	for(int i=0; i<n; i++) {
    		s = b[i];
    		c = c + s/5;
    		s = s%5;
    		c = c + s/2;
    		s = s%2;
    		c = c + s/1;
    	}
    	cout<<c<<endl;
    }

}
