{
	int n,q;
	sc(n);
	Fore(i,1,n) sc(p[i].F, p[i].S);

	//reverse(p+1,p+n+1);
	sc(q);

		sc(a[i]);
	}
	sort(a,a+q);
	lli ans=0;
	int maxx=0,j=n;
	rFore(i,q-1,0){
		while(j>=1 and p[j].F>=a[i]) maxx=max(maxx, p[j].S), j--;

	}
