#include <iostream>
typedef long long ll;
using namespace std;

int main()
{
    int n;
    cin>>n;
    ll B[n];
    
    for(int i=0;i<n;i++)
    {
    	cin>>B[i];
    }
    ll sum=0;
    ll A[n];
    for(int i=0;i<n;i++)
    {
    	A[i]=(B[i]*(i+1)-sum);
    	sum+=A[i];
    }
    
    for(int i=0;i<n;i++)
    cout<<A[i]<<" ";
    return 0;
}
