#include<iostream>
using namespace std;
#include<map>
int main()
{
int t;
cin>>t;
while(t--)
{
	int n,mp;
	long long int tmp;
	cin>>n>>mp;
	map<long long int,int>m;
	for(int i=0;i<n;i++)
	{
		cin>>tmp;
		m[tmp]=1;
	}
	for(int i=0;i<mp;i++)
	{
		cin>>tmp;
		if(m[tmp]==1)
		cout<<"YES"<<endl;
		else
		{cout<<"NO"<<endl;
		 m[tmp]=1;
		}
	}
	
}
}
