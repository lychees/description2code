#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin >> t;
	
	while(t--)
	{
		int n,k;
		cin >> n >> k;
		
		long long a[n],b[n];
		for(int i = 0 ; i < n ; ++i)	cin >> a[i];
		for(int i = 0 ; i < n ; ++i) 	cin >> b[i];
		
		long long ans = 0;
		for(int i = 0 ; i < n ; ++i)
		{
			long long netpr = b[i]*(k/a[i]);
			
			ans = max(ans,netpr);
		}
		
		cout << ans << endl;
	}
	
	return 0;
}
