#include <bits/stdc++.h>

using namespace std;
int prime[10010];

void prime_()
{
    int i,j;
    for(i=0; i<=10010; i++)
    {
        prime[i] = 1;
    }
    prime[0] = 1;
    prime[1] = 1;
    for(i=2; i<sqrt(10010); i++)
    {
        if(prime[i])
        {
            for(j=i*i; j<=10010; j += i)
            {
                prime[j] = 0;
            }
        }
    }

}


int main()
{
    prime_();
     int i, j, n, k = 0;

    cin >> n;
    for(i=1; i<=n; i++)
    {
        if(prime[i] == 0)
        {
            if(k > 0) cout << " ";
            cout << i;
            k++;
        }
    }

}
