#include <iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t,n,i,s,f,a[200000];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<2*n;i++)
    	cin>>a[i];
    	sort(a,a+2*n);
    	s=0;
    	for( i=0;i<2*n;i++)
    	     
    {	      f=a[i]+a[2*n-i-1];
           if(s<=f)
              s=f;
    }
    cout<<s<<endl;
}
    return 0;
}
