#include<iostream>
using namespace std;
int main(){
	
	int t;
	cin>>t;long x,y;
	for(long i=0;i<t;i++){
		
		cin>>x>>y;
		long l=0;int h=0;
		while(x!=0||y!=0){
			int u=x%10;int v=y%10;
			int w=u+v+h;
			if(w>9){l=l+1;h=(w)/10;
			}
			else{h=0;
			}
			x=x/10;y=y/10;
		}
		if(l!=0){
			
			if(l==1)cout<<"1 carry operation"<<endl;
			else cout<<l<<" carry operations"<<endl;
		}
		else cout<<"No carry operation"<<endl;
	}
}
