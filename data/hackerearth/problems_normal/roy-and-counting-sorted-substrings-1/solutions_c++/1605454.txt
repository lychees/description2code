#include <iostream>
#include <cstdio>
using namespace std;

#define arr_size  1000003

long long int fun(long long  x)
{
	return (x*(x+1))>>1;

}

int main()
{
	int t;
	cin>>t;
	long long res=0;	
	char arr[arr_size];
	
	while(t--)
	{
		int n;
		cin>>n; res=0;
		scanf("%s",arr);
		
		int count=1;
		for(int i=0;i<n-1;i++)
		{
			if(arr[i]<=arr[i+1])
			count++;
			else
			{
				res+=fun(count);
				count=1;
			}
		}
		if(n>=2 and arr[n-1]>arr[n-1])
		res++;
		if(arr[n-1]<=arr[n-1])
		res+=fun(count);
		cout<<res<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
