#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define F(i, a, b) for (int i = (int)(a); i <= (int)(b); i++)
#define RF(i, a, b) for (int i = (int)(a); i >= (int)(b); i--)
int main() {
#ifndef ONLINE_JUDGE
  freopen("input.txt", "r", stdin);
  freopen("output.txt", "w", stdout);
#endif
  ll T, M, N;
  scanf("%lld", &T);
  while (T--) {
    scanf("%lld%lld", &M, &N);
    if (N == 0) {
      printf("0\n");
    } else if (N <= M) {
      printf("10\n");
    } else {
      ll rounds = 0;
      ll remaining = N;
      rounds = N / M;
      if (N % M) {
        rounds++;
        remaining -= (M - (N % M));
      }
      if (remaining) {
        rounds += (remaining / M);
        if (remaining % M)
          rounds++;
      }
      printf("%lld\n", rounds * 5);
    }
  }
  return 0;
}
