#define pii pair<long long,long long>
#define inp(x) scanf("%d",&x)
using namespace std;
typedef long long ll;

int gcd(int a, int b)
{
    while(b) b ^= a ^= b ^= a %= b;
    return a;
}

int main()
{
    ll t,sum[101],rev[101],n,i;
    pii pos[100];

    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<n;++i)
        {
            cin>>pos[i].first;
        }

        for(i=0;i<n;++i)
        {
            cin>>pos[i].second;
            pos[i].second += pos[i].first;
        }

        sort(pos,pos+n);
        ll tem = pos[0].second;
        memset(sum,0,sizeof(sum));
        for(i=1;i<n;++i)
        {
            sum[i] = i*(pos[i].first-tem);
            //cout<<sum[i]<<" ";
            sum[i] += sum[i-1];
            tem = pos[i].second;
            //cout<<sum[i]<<"\n";
        }

        tem = pos[n-1].first;
        //cout<<"\n";
        memset(rev,0,sizeof(rev));
        ll ctr = 1;
        for(i=n-2;i>=0;--i)
        {
            rev[i+1] = ctr*(tem - pos[i].second);
            //cout<<rev[i+1]<<" ";
            rev[i+1] += rev[i+2];
            tem = pos[i].first;
            //cout<<rev[i+1]<<"\n";
            ctr++;
        }

        ll tot = rev[1];
        for(i=1;i<n;++i)
        {
            tot = min(tot,sum[i]+rev[i+1]);
            //cout<<sum[i]<<" "<<rev[i]<<"\n";
        }

        cout<<tot<<"\n";


    }
    return 0;
}
