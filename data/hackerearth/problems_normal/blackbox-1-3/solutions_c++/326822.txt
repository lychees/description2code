#include <iostream>
#include<stdlib.h>
using namespace std;

int cmp(const void*x,const void*y)
{
	return *(long long int *)x - *(long long int*)y;	
}

long long int gcd(long long int a, long long int b){

	long long int temp;

	while(b){

		temp=b;
		b=a%b;
		a=temp;
	}
	return a;
}


int comp(long long int a,long long int b,long long int c)
{

	if((gcd(a,b)==1)&&(gcd(b,c)==1)&&(gcd(a,c)==1))
		return 1;
	return 0;
	
}
