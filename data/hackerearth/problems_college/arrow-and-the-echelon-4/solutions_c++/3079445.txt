#include <iostream>
using namespace std;

int main()
{
		int t;
     	cin>>t;
        
        while(t-->0)
        {
        	int m = 0;
            int x;
            cin>>x;
            int temp=x/3;
            for(int i=3*temp;i>=0;i-=3)
            {
                 m=(x-i)%5;
                if(m==0)
                {
                   for(int j=0;j<i;j++)
                        cout<<"5";
                   for(int k=0;k<(x-i);k++)
                        cout<<"3";
                   break;
                }
            }
            if(m!=0)
               cout<<"-1";
            cout<<endl;
        }return 0;
}
