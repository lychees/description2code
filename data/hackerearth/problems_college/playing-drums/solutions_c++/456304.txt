bool check(int arr[], int n,int m, int mi)
{
	ll cur = 0;
	m--;
	for(int i = 1; i < n && m; i++)
	{
		cur += arr[i];
		if(cur > mi)
		{
			cur = 0;//arr[i];
			m--;
		}
		else if(cur == mi)
		{
			m--;
			cur = 0;
		}
	//	cout<<cur<<" "<<m<<endl;
	}
	if(m >0)
	return false;
	return true;
}
int main()
{
	int t, i, n, x, y, m;
	inp(t);
	while(t--)
	{
		inp2(n, m);
		int arr[n], last = 0;
		
		rep(i, n)
		{
			inp(x);
			arr[i] = x - last;
			last = x;
		}
		int ans = 0;
		int lo = 0, hi = x+1, mi;
        while(lo<hi)
        {
		
            mi=lo+(hi-lo+1)/2;
		//printf("\nmi=%d %d %d\n ",mi,lo, hi);
            if(check(arr, n, m, mi))
            {
		//printf("Y\n");
                ans=mi;
                lo=mi;
            }
            else
			{
				//printf("N\n");
                hi=mi-1;
			}
        }
        if(check(arr, n, m, lo))
        {
        	ans = max(ans, lo);
        }
        if(check(arr, n, m, ans+1))
        {
        	ans++;// = max(ans, lo);
        }
		out(ans);
	}
	return 0;
}
