#include<iostream>
using namespace std;
int main()
{
   int n,c,m,total=0,wrap=0,t;
   cin>>t;
   while(t--)
   {
   cin>>n>>c>>m;
   total=n/c;
   n=0;
   c=0;
   wrap=total;
   while(wrap>=m)
   {
   	total++;
   	wrap-=m;
   	wrap++;
   }
   cout<<total<<endl;
   total=0;
   wrap=0;
}
	return 0;
}
