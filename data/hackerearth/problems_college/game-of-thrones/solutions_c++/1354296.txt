#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,i,j,ans,n;
	set<int> s;
	scanf("%d",&t);
	while (t --) {
		scanf("%d",&n);
		for (i = 0; i < n; i++) {
			scanf("%d",&j);
		    s.insert(j);
		}
		ans = n - s.size();
	    printf("%d\n",ans);
	    s.clear();
	}
	
	return 0;
}
