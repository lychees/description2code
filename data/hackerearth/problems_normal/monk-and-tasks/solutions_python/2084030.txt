
tc = int(raw_input())
while(tc>0):
	tc-=1
	raw_input()
	data, tasks={}, raw_input().split(" ")
	tasks=[int(x) for x in tasks]
	for x in tasks:
		b=bin(x)[2:].count('1')
		if(data.has_key(b)) :
			data[b].append(x)
		else:
			data[b]=[x]
	ke, ans=sorted(data), ""
	for x in ke:
		for y in data[x]:
			ans+=" "+str(y)	
	print ans.strip()
