#include<stdio.h>
#include<string.h>
#include<algorithm>

using namespace std;

int main()
{
    int t;
    long long k,x,m;
    char s[100];

    scanf("%d",&t);

    while(t--)
    {
        scanf("%lld",&k);

        int l = 1;
        x = 1<<l;
        while(x < k)
        {
			k -= x;
			l++;
            x = 1LL<<l;
        }
        k--;

        int i=0;
        while(k)
        {
            if(k&1)
            s[i++]='7';
            else
            s[i++]='4';

            k=k>>1;
            l--;

        }

        while(l--)
        s[i++]='4';
        s[i]='\0';
        reverse(s,s+strlen(s));

        printf("%s\n",s);

    }
    return 0;
}
