#include<iostream>
#include<stdio.h>
#define MOD 1000000007
using namespace std;
int main()
{
    int t;
    cin>>t;
    int n;
    long long arr[100000];
    arr[0]=1;
    arr[1]=1;
    arr[2]=2;

    while(t--)
    {
        cin>>n;
        for(int i=3;i<=n;i++)
            arr[i]=(arr[i-2]+arr[i-1]+arr[i-3])%MOD;
        cout<<arr[n]<<endl;

    }
    return 0;
}
