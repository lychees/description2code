#include<bits/stdc++.h>
#define MOD 1000000007
using namespace std;
long int A[100001];
long int cal(long int n)
{
	if(n==1) return 1;
	if(n==2) return 2;
	if(n==3) return 4;
	if(A[n]!=0) return A[n];
	return A[n]=(cal(n-1)+cal(n-2)+cal(n-3))%MOD;
}
int main()
{
	long int T,i,j,k;
	cin>>T;
	A[1]=1;
	A[2]=2;
	A[3]=4;
	for(i=4;i<=100000;i++)
	{
		cal(i);
	}
	while(T--)
	{
		cin>>j;
		cout<<A[j]<<"\n";
	}
}
