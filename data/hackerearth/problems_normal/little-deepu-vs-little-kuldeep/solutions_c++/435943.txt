using namespace std;

#include <iostream>
#include <cstdio>
#define ll long long int
#define s(a) scanf("%lld",&a)

ll fact[10000009];
ll mod=1000000007;

ll expo(ll x,ll y)
{
	ll result;
        if(y==1)
               return (x%mod);
        if(y==2)
	        return (x*x)%mod;
        if (y%2==1)
        {
	        return (x*expo(x,y-1))%mod;
        }
        else
        {
               result = expo(x,y/2)%mod;
               return (result*result)%mod;
	}
}

ll comb(ll x,ll y)
{
	ll res,den=(fact[y]*fact[x-y])%mod;
	res=expo(den,mod-2);
	return ((fact[x]*res)%mod);
}


int main()
{
    ll t,n,k,i;
    fact[0]=1;
    fact[1]=1;
    for(i=2;i<=10000000;i++) fact[i]=(fact[i-1]*i)%mod;
    s(t);
    while(t--) {
        s(n);
        k=comb(2*n,n)-comb(2*n,n+1);
        if(k<0) k+=mod;
        cout<<k<<endl;
    }
    return 0;
}
