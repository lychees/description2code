#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int arr[100005];
int main()
{
    int n,t;
    cin>>n>>t;
    int a,b,ans;
    for(int i=0; i<n; i++)
        cin>>arr[i];
    while(t--)
    {
        cin>>a>>b;
        ans=arr[a];
        for(int i=a+1; i<=b; i++)
        {
            if(arr[i]<ans)
                ans = arr[i];
        }
        cout<<ans<<endl;
    }
return 0;
}
