#include <bits/stdc++.h>

#define MAXN 2015
#define H 32

using namespace std;

int N, M;
int koupepi;
char S[MAXN][MAXN];
int len[MAXN][MAXN];

int main()
{
	ios_base::sync_with_stdio(false);
    cin >> N >> M;
    
    for (int i=0; i<N; i++)
    	for (int j=0; j<M; j++)
    		cin >> S[i][j];
    
    for (int i=0; i<M; i++)
    	len[N-1][i] = 1;
    
    for (int i=N-2; i>=0; i--) {
    	len[i][M-1] = 1;
    	for (int j=M-2; j>=0; j--) {
    		int x = min(len[i+1][j], len[i][j+1]);
    		if (S[i][j] != S[i+1][j] && S[i][j] != S[i][j+1]) {
    			len[i][j] = x;
    			if (S[i][j] == S[i+x][j+x]) len[i][j] ++;
    		}
    		else len[i][j] = 1;
    	}
    }
    
    for (int i=0; i<N; i++)
    	for (int j=0; j<M; j++) {
    		if (len[i][j] >= H && S[i][j] == '_') koupepi += 1;
    	}
    		
   	cout << koupepi << "\n";
    return 0;
}
