#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n;
		cin>>n;
		ll int i,j;
		ll int t=0;
		for(i=0;i<n;i++)
		{
			cin>>j;
			j=j%5; //Subtraction game
			t^=j;
		}
		if(t)
		{
			cout<<"Aniruddha\n";
		}
		else
		{
			cout<<"Andrew\n";
		}
	}
	return 0;
}
