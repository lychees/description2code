#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
    char a[105];
    cin>>a;
    int n;
    cin>>n;
    while(n--)
    {
        int b,c,e=0;
        cin>>b>>c;
        b--;c--;
        for(int i=b;i<c;i++)
            if(a[i]==a[i+1])
                e++;
        cout<<e<<endl;
    }
    return 0;
}
