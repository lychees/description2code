#include<bits/stdc++.h>
using namespace std;
int mult(string str,int p,int m)
{
	int n = pow(10,p);
	int num=0;
	for(int i=0;i<str.length(),p!=0;i++)
	{
		num = num + (int(str[i])-48)*pow(10,p);
		p--;
	}
	return num*m;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str;
		cin>>str;
		int n;
		cin>>n;
		int m = n;
		int p=0;
		while(n!=0)
		{
			p++;
			n = n/10;
		}
		int l = str.length();
		for(long long int j = 0;j<l-p+1;j++)
		{
			cout<<mult(str.substr(j,p),p,m)/10<<"\n";
		}
	}
	
}
