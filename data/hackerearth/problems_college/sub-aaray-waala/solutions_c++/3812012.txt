#include <bits/stdc++.h>
using namespace std;
#define M 1000000007
#define ll long long
ll mat(ll  a,ll b){
	if(b==0) return 1;
	if(b==1) return a;
	if(b%2){
		ll x=mat(a,b-1);
		return (a*x)%M;
	}
	else{
		ll x=mat(a,b/2);
		return (x*x)%M;
	}
}
int main()
{
    ll t,n,x;
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cin>>t;
    while(t--){
    	cin>>n;
    	set<ll > s;
    	for(int i=0;i<n;i++){
    		cin>>x;
    		s.insert(x);
    		
    	}
    	cout<<mat(n,s.size())%M<<'\n';
    	
    }
    return 0;
}
