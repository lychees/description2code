#include <iostream>
#include <cstring>
using namespace std;

const int mod = 1e9 + 7;
int temp=0,ans=0,x=0;

int main()
{
    //freopen("stdin.input","r",stdin);
    string s;cin>>s;
    for(auto &c:s)
    {
        if(c=='_')
        {
            temp = (2*temp + x) % mod;
        }
        else
        {
            ans = (ans + temp) % mod;
            x++;
        }
    }
    cout<<ans;
    return 0;
}
