#include <iostream>
using namespace std;



int main()
{
    int t, x, y, Cherry, i;
    cin >> t;
    
    while (t--) {
    	cin >> x >> y;
    	Cherry = 0;
    	i = 1;
    	
    	do {
            Cherry = i * (x + y);
            i++;
        } while (Cherry % x);
    	
    	cout << Cherry / x << endl;
    	
    	
    }
}
