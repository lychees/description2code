#include <cstdio>
#include<algorithm>
using namespace std;
#define ll long long
ll arr[100];
void prec()
{
	ll i,j,k;
	arr[1]=1;
	for(i=2; i<20; i++)
	{
		arr[i]=i*arr[i-1];
	}
}
int main()
{
	prec();
    ll t,n,a,b,i;
    scanf("%lld",&t);
    while(t--)
    {
    	ll sum=0;
    	scanf("%lld %lld",&a,&b);
    	for(i=a; i<=b; i++)
    	{
    		sum=sum+arr[i];
    	}
    	ll ans=sum%10;
    	printf("%lld\n",ans);
    }
    return 0;
}
