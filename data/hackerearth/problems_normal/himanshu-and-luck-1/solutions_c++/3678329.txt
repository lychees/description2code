#include<iostream>
#include<cmath>
using namespace std;

long long reverse(long long num) {
	long long revnum = 0;
    while(num) {
        revnum = revnum*10 + (num%10);
        num/=10;
    }
    return revnum;
}

long long calc(long long val) {
	long long twodig = val;
    long long twopow = 1;
    while(twodig) {
        twopow *= 2;
        twodig/=10;
    }
    long long result = twopow - 2;
	long long revnum = reverse(val);
	while(revnum) {
			long long modval = revnum%10;
            if(modval > 7) {
                result += twopow;
            }
            else if(modval > 4) {
                result += (twopow/2);
            }
            if(modval != 4 && modval != 7) break;
			revnum/=10;
            twopow/=2;
    }
    return result;
}

int main() {
	int t;
	cin>>t;
	while(t--) {
		long long l,r;
		cin>>l>>r;
		cout<<calc(r+1) - calc(l)<<endl;
	}

}
