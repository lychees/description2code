#include <iostream>
#include<string.h>
#include<cstdio>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   
   int t;
   scanf("%d",&t);
   while(t--)
   {
   	int k;
   	scanf("%d",&k);
   	int i,j;
   	char ch[100],s[100];
   	scanf("%s",ch);
   	for(i=0;i<strlen(ch);i++)
   	{
   		int n=(int)ch[i];
   		n=n-k;
   		if(n>=65)
   		s[i]=(char)n;
   		else
   		{
   			int e=(65-n-1)%26;
   			e=90-e;
   			s[i]=(char)e;
   		}
   	}
   	s[i]='\0';
   	printf("%s\n",s);
   }
    return 0;
}
