#include <iostream>
using namespace std;

int main()
{
    int i,T,N;
    cin>>T;
    while(T--)
    {
        long int X,sum=0;
        cin>>N;
        for(i=0;i<N;i++)
        {
            cin>>X;
            sum+=X;
        }
        if(sum%N==0)
        {
            cout<<"\nYES\n"<<(sum/N)<<" "<<(sum%N);
        }
        else
        {
            cout<<"\nNO\n"<<(sum/N)<<" "<<(sum%N);
        }
    }
    return 0;
}
