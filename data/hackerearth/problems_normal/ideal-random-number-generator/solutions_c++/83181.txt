ll gcd(ll a, ll b)
{
    if(b == 0)
        return a;
    return gcd(b,a%b);
}
int main()
{
    ll a,b,c;
    scanf("%lld %lld %lld",&a,&b,&c);
    if(a < b)
        swap(a,b);
    if(c >= a+b)
    {
        printf("1/1");
        return 0;
    }
    ll tot;
    ll gc;
    if(c <= b)
    {
        tot = c*c;
        gc = gcd(tot,a*b*2);
        printf("%lld/%lld\n",tot/gc,(a*b*2)/gc);
        return 0;
    }
    if(c > b && c <= a)
    {
        ll diff = c - b;
        tot = (b*b) + diff*b*2;
        gc = gcd(tot,a*b*2);
        printf("%lld/%lld\n",tot/gc,(a*b*2)/gc);
        return 0;
    }
    ll diff = c - b;
    tot = b*diff*2;
    ll diff2 = c-a;
    ll diff3 = a-diff;
    tot += (diff2*diff3*2) + ((b-diff2)*diff3);
    gc = gcd(tot,a*b*2);
    printf("%lld/%lld\n",tot/gc,(a*b*2)/gc);

return 0;
}
