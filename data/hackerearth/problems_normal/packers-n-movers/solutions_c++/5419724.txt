#include<bits/stdc++.h>
using namespace std;
int a[260];
int main()
{
    int i,m,n;
    scanf("%d %d",&m,&n);
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    int low=0,high=2500000,s=0,mid,cnt;
    while(low<high)
    {
        mid=(low+high)/2;
       // break;
        s=0,cnt=0,i=0;
        while(i<n)
        {
            if(s+a[i]>mid)
            {
                cnt++;
                s=0;
            }
            else if(s+a[i]==mid)
            {
                cnt++;
                i++;
                s=0;
            }
            else
            {
                s=s+a[i];
                i++;
            }
            if(cnt>m)
                break;
        }
        if(s!=0)
            cnt++;
        if(cnt>m)
            low=mid+1;
        else if(cnt<=m)
            high=mid;
        if(low==high)
        {
            mid=(low+high)/2;
            break;
        }
    }
    printf("%d\n",mid);
}
