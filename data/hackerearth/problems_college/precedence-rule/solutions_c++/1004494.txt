#include<iostream>
#include<string>
#include<stack>
#include<vector>
using namespace std;

int main() {
	int t;
	cin >> t;
	while (t--) {
		string input;
		cin >> input;
		if (input.length() < 3)
			cout << input[0] << endl;
		else {
			stack<char> operators;
			stack<long long int> numbers;
			for (int i = 0; i < input.length(); i++) {
				char ch = input[i];
				if (ch == '+' || ch == '-' || ch == '*') {
					operators.push(ch);
				} else {
					numbers.push(ch - '0');
				}
			}
			int len=input.length();

//			vector<char> operators;
//			vector<int> numbers;
//			for (int i = 0; i < input.length(); i++) {
//				char ch = input[i];
//				if (ch == '+' || ch == '-' || ch == '*') {
//					operators.push_back(ch);
//				} else {
//					numbers.push_back(ch - '0');
//				}
//			}
//			for (int i = 0; i < operators.size(); i++) {
//				cout << operators[i] << " ";
//			}
//			cout << endl;
//			for (int i = 0; i < numbers.size(); i++) {
//				cout << numbers[i] << " ";
//			}
//			cout << endl;

			long long int a, b,i=0,numop=(len-1)/2;
			while (i<numop) {
				i++;
				char ch = operators.top();
				operators.pop();
				a = numbers.top();
				numbers.pop();
				b = numbers.top();
				numbers.pop();
				if (ch == '-') {
					numbers.push(a - b);
				} else if (ch == '*') {
					numbers.push(a * b);
				} else if(ch=='+'){
					numbers.push(a + b);
				}
			}
			cout << numbers.top() << endl;
		}
	}
	return 0;
}
