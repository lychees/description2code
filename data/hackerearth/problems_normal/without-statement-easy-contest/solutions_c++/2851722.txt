#include <bits/stdc++.h>
using namespace std;
#define ll long long

map<ll,ll> memo,vec;


ll solve(ll x,int i){
	//if(memo[x]!=0) return memo[x];
	ll y = x;
     ll e=0;
     while(x!=0){
     	e=e+(x%10)*(x%10);
     	x=x/10;
     }
     memo[y]=i;
     vec[i]=y;
     return e;
}

int main()
{
    int t;
    cin>>t;
    while(t--){
        memo.clear();
        vec.clear();
    	ll n,m;
    	scanf("%lld %lld",&n,&m);
    	ll ans = n;
    	bool flg= false;
    	for(int i=1;i<=m;i+=1){
    		ans = solve(ans,i-1);
    		if(memo[ans]!=0){
    			 int p = i - memo[ans];
                    m -= i;
                    m %= p;
                    ans = vec[memo[ans] + m];
                cout<<ans<<endl;
                flg=true;
                break;
    		}

    	}
    	if(!flg)cout<<ans<<endl;
    }

    return 0;
}
