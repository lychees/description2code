#include <bits/stdc++.h>
using namespace std;

int main () {
	int tc;
	scanf("%d",&tc);
	while (tc--) {
		string s, p;
		cin >> s;
		cin >> p;
		int ans = 0;
		for (int i=0; i<s.size(); i++) {
			if ( s[i] == p [ans] )
				ans++;
			if ( ans == p.size() )
				break;
		}
		if ( ans == p.size() )
			puts("YES");
		else
			puts("NO");
	}
	return 0;
}
