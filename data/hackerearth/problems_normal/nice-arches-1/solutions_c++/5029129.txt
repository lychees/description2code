#include <bits/stdc++.h>
using namespace std;

int main()
{
    
    string c;
    vector<char> v;
    int n,count=0;
    cin>>n;
    while(n--)
    {
    	cin>>c;
    	v.clear();
    	for(int i=0;i<c.size();i++){
    		if(v.empty())
    		v.push_back(c[i]);
    		else if(v.back()==c[i])
    		v.pop_back();
    		else
    		v.push_back(c[i]);
    		
    	}
    if(v.empty())
    count++;
    }
    cout<<count;
    return 0;
}
