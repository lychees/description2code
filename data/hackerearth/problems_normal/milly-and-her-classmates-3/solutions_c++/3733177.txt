#include<bits/stdc++.h>
using namespace std;

int main(){
	int tc;cin>>tc;
	while(tc--){
		int n;cin>>n;
		long int arr[n];
		for(int i=0;i<n;i++) cin>>arr[i];
		int count=0;
		for(int i=1;i<n;i++){
			if(arr[i]>=arr[i-1]) continue;
			if(arr[i]<arr[i-1] ){
				while(i<n-1 && arr[i]>=arr[i+1]) i++;
				count++;
			}
		}
		//cout<<count<<endl;
		if(count==1){
			int i;
			for(i=0;i<n-1;i++) if(arr[i]>arr[i+1]) break;
			int k=i;
			i++;
			while(i<n-1 && arr[i]>=arr[i+1]) i++;
			i;
			//cout<<k<<" "<<i<<endl;
			if(k==0) cout<<k+1<<" "<<i+1<<endl;
			else if(arr[k-1]<=arr[i]) cout<<k+1<<" "<<i+1<<endl;
			else cout<<"-1 -1\n";
		}
		else cout<<"-1 -1\n";
	}
}
