#include<bits/stdc++.h>
using namespace std;
#define l1 long long int
bool comp(pair<l1,l1> a,pair<l1,l1> b)
{
	return a.first<b.first;
}
int main()
{
	l1 i,n,l,m,t,q,j,k,f,s;
	cin>>t;
	while(t--)
	{
	pair< l1,l1>y;
	vector<pair<l1,l1> > vec;
	vec.clear();
	cin>>n>>q;
	for(i=1;i<=n;i++)
	{
		cin>>l>>m;
		y=make_pair(l,m);
		vec.push_back(y);
	}
	l1 sum[1000]={0};
	
	sort(vec.begin(),vec.end(),comp);
	sum[1]=vec[0].first;
	sum[2]=vec[0].second;
	j=3;
	for(i=1;i<vec.size();i++)
	{
		l=vec[i].first;
		m=vec[i].second;
		if(l>=sum[j-2]&&l<=sum[j-1])
		{
			if(m>sum[j-1])
			sum[j-1]=m;
		}
		else if(l>sum[j-1])
		{
			sum[j++]=0;
			sum[j++]=l;
			sum[j++]=m;
		}
	}
	sum[j++]=0;
	//
	while(q--)
		{
			cin>>k;
			f=0;
		for(i=1;i<j;i=i+3)
		{
			f=0;
			s=sum[i+1]-sum[i]+1;
			if(s>=k)
			{
				cout<<sum[i]+k-1<<endl;
				f=1;
				break;
			}
			k=k-s;
		}
		if(f==0)
		cout<<"-1"<<endl;
		}
	
}
	return 0;
}
