#include <iostream>
#include <cstring>
using namespace std;
bool A[51][51];
int n;
bool move(int i, int j)
{
	bool a=0,b=0;
	if(i==n-1&&j==n-1)
	return 1;
	if(i<n-1&&A[i+1][j]==1)
	a=move(i+1, j);
	if(j<n-1&&A[i][j+1]==1)
	b=move(i, j+1);
	return a|b;
}
int main()
{
    int t, i, j;bool ans;
    scanf("%d", &t);
    while(t--)
    {
    	scanf("%d", &n);
    	for(i=0;i<n;i++)
    	for(j=0;j<n;j++)
    	cin>>A[i][j];
    	ans=move(0, 0);
    	if(ans)
    	printf("POSSIBLE\n");
    	else
    	printf("NOT POSSIBLE\n");
    }
    return 0;
}
