#include <iostream>
#include<queue>
#include<map>
using namespace std;
map<long long,long long> mp;
priority_queue<pair<long long,long long>> q;
pair<long long,long long> it;
long long n,m,count[10000000]={0},a,y;
int main()
{map<long long,long long>::iterator ij;
    cin>>n>>m;
    while(n--)
    {cin>>a;
      ij=mp.find(a);
      if(ij!=mp.end())
      {y=ij->second;
      
       y++;
      	mp.erase(ij);
      	mp.insert(pair<long long,long long>(a,y));
      }
      else
     { mp.insert(pair<long long,long long>(a,1));
      y=1;
     }
      q.push(make_pair(y,a));
      it=q.top();
      cout<<it.second<<" "<<it.first<<endl;
    }
   
   
    return 0;
}
