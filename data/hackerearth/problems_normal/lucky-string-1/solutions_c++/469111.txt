#include <iostream>
using namespace std;

int main()
{
	int t;
	long long n,x,num,pos,rem,ans;
	scanf("%d",&t);
	while(t--){
		scanf("%lld",&n);
    	long long int i=1,k=0,j=2;
    	while (n > (j*i+k)){
    		k += (j*i);
    		j = j<<1;
    		i++;
    	}
    	x = n-k-1;
    	num = x/i;
    	pos = x%i;
    	rem = num>>(i-1-pos);
    	ans = rem%2;
    	if(ans)
    		printf("Earth\n");
    	else
    		printf("Hacker\n");
    	
    	
	}
    return 0;
}
