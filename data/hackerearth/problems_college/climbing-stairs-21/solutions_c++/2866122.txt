#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	int i,n;
    	long int a,b,s=0;
       	cin>>n;
    
    a=1;b=2;
    if(n==1)
    cout<<1<<endl;
    else if(n==2)
    cout<<2<<endl;
    else
    {
    	for(i=2;i<n;i++)
    	{
    		s=a+b;
    		a=b;
    	b=s;
    		
    	}
    cout<<s;
    	cout<<endl;
    }
    	t--;
    }
    return 0;
}
