def getNumbers(s):
    l = [0]*26
    for v in s:
        l[ord(v)-ord('a')] += 1
    return tuple(l)


def makeKey(s):
    return (s[0], s[-1], getNumbers(s))


n = input()
names = dict()
for i in range(n):
    s = raw_input()
    t = makeKey(s)
    if t in names:
        names[t] += 1
    else:
        names[t] = 1

print len(names.keys())
