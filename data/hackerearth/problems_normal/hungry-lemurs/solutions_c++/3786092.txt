#include <iostream>
#include<limits.h>
#include<math.h>
using namespace std;
int min(int a,int b)
{
	return a>b?b:a;
}


int main()
{
    int n,k;
    cin>>n>>k;
    int result=INT_MAX;
   for(int i=1;i<2*k;i++)
   {
   	result=min(result,abs(k-i)+min(n%i,i-n%i));
   }
   cout<<result<<endl;
    return 0;
}
