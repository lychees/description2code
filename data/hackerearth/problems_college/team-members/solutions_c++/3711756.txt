typedef long long ll;

const int inf=1e9;
const int MOD=1e9+7;
const int N=15,M=1e3+10;
 
ll f[1<<N][M];
int mod,k,g[N];
string s;
ll solve(){
    int n=s.length();
    f[0][0]=1;
    REP(i,0,n)
        REP(j,0,i) if (s[j]==s[i]) g[i]|=1<<j;
    REP(pha,0,1<<n){
        for(int i=0;i<n;i++) if (!(pha&(1<<i))){
            if ((pha&g[i])!=g[i]) continue;
            int nx=pha|(1<<i),v=s[i]-'0';
            REP(j,0,mod) if (f[pha][j])
                f[nx][(10*j+v)%mod]+=f[pha][j];
        }
    }
    return f[(1<<n)-1][k];
}
