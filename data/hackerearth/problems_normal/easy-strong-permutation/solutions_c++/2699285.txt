#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n,i;
    long long int *a;
    long long int sum;
    
    scanf("%d",&n);
    a = (long long int *)malloc(sizeof(long long int)*n);
    for(i=0;i<n;i++){
    	scanf("%lld",&a[i]);
    }
    
    sort(a,a+n);
    
    for(i=0;i<n/2;i++){
    	sum-=2*a[i];
    	sum+=2*a[n-i-1];
    }
    cout<<sum<<endl;
    
    return 0;
}
