#include<bits/stdc++.h>
using namespace std;
int c,dp[1000][200][200];

int coke(int n1,int n5,int n10,int count )
{
     if(count==c)
        return 0;
int    ans=INT_MAX;
    if(dp[n1][n5][n10]!=-1)
        return dp[n1][n5][n10];

    if(n10>=1)
    ans=min(ans, coke(n1+2,n5,n10-1,count+1)+1);
    if(n5>=2)
        ans=min(ans,coke(n1+2,n5-2,n10,count+1)+2);
    if(n1>=8)
       ans=min(ans, coke(n1-8,n5,n10,count+1)+8);
    if(n5>=1 && n1>=3)
        ans=min(ans,coke(n1-3,n5-1,n10,count+1)+4);
    if(n1>=3 && n10>=1)
        ans=min(ans,coke(n1-3,n5+1,n10-1,count+1)+4);
    dp[n1][n5][n10]=ans;
    return dp[n1][n5][n10];
}
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
      memset(dp,-1,sizeof(dp));
     int n1,n5,n10;
     cin>>c>>n1>>n5>>n10;
     if(c*8>(n1*1+n5*5+n10*10))
        cout<<"-1"<<endl;
     else
        cout<<coke(n1,n5,n10,0)<<endl;

  }

}
