#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    unsigned long long int t,a,b,c,prev,total,tmp,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	prev=total=0;
    	a=2;
    	b=3;
    	while(total<n)
    	{
    		c=a+b;
    		tmp=c-b-1;
    		prev=total;
    		total+=tmp;
    		a=b;b=c;
    	}
    	tmp=n-prev;
    	cout<<a+tmp<<endl;
    }
    
    return 0;
}
