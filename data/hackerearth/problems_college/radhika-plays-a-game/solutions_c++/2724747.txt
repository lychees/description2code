#include <iostream>
#include<string.h>
#include<stdlib.h>
using namespace std;

int main()
{

    int T;

    cin>>T;

    while(T--!=0){

        int N,K;

        cin>>N>>K;

        int min = 1;
        int max = K;

        float val = (N-1)%(min+max);

        if(val==0.0){
            cout<<"Nandu"<<endl;
        }else{
            cout<<"Rani"<<endl;
        }

    }

    return 0;
}
