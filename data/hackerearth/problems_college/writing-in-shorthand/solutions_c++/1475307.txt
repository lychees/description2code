#include<map>
#include<string>
#include<iostream>
using namespace std;
long long len(char s[])
{   long long i;
	for(i=0;s[i]!='\0';i++);
	return i;
}
int main()
{

	char s1[1000],s2[1000],me[1000];
	long long mes,d;
    map<string, string> m;
    cin>>mes>>d;
    for(int i=0;i<d;i++)
    {
      	cin>>s1>>s2;
      	m.insert(pair<string, string>(s1, s2));
    }
    for(int i=0;i<mes;i++)
    {
    	cin>>me;
    	if(len(me)<=m[me].length())
    	 cout<<me<<" ";
    	 else
    	 cout<<m[me]<<" ";
    }

   return 0;
}
