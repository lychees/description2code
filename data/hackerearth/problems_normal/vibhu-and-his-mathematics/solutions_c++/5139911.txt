#include <iostream>
#include <bits/stdc++.h>
#define MAXN 1000001
#define MOD 1000000007
long long int F[MAXN]={0};
using namespace std;

int main()
{
	
   		F[0]=1,F[1]=1;
   		for(int i=2;i<=MAXN;i++)
   		{
   			F[i]=F[i-1]+(i-1)*F[i-2];
   			F[i]%=MOD;
   		}
   	long long int T,N;
   	cin>>T;
   	while(T--)
   	{
   		cin>>N;
   		printf("%d\n",F[N]);
   	}
    return 0;
}
