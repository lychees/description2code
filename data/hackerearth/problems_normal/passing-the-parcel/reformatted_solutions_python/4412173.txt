t = int(raw_input())
s = raw_input()
a = []
for i in range(1, t+1):
    a.append(i)
p = 0
while (len(a) > 1):
    for i in s:

        if i == 'a':
            if len(a) != 0:
                p = (p+1) % (len(a))

        else:
            a.remove(a[p])
            if len(a) != 0:
                p = p % len(a)
            if len(a) == 1:
                break
    if len(a) == 1:
        break
print a[0]
