#include <iostream>

using namespace std;

bool pow_of_two(long long int num)
{

    return (num)&&(!(num&(num-1)));
}

int main()

{

    long long int  m,test,i,j,num,maxx,minn,ans,total;

    cin>>test;
    while(test--)
    {
        cin>>num>>m;
        if((m-num)==1)
            cout<<(num&m)<<endl;
        else if(pow_of_two(m))
            cout<<(m-2)<<endl;
        else
        {
            if(m%2)
                cout<<(m-1)<<endl;
            else
                cout<<(m-2)<<endl;
        }
    }
}
