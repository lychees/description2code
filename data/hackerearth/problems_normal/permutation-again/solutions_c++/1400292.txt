#include <iostream>
using namespace std;
typedef unsigned long long int llui;
int main()
{
    int T;
    llui N,ans=0;
    cin>>T;
    while(T--){
    	cin>>N;
    	if(N==1)
    		ans=1;
    	else	
    	{	
    		if(N&1)
    			ans=((N/2)*(N+1))-1;
    		else
    			ans=((N/2)*N)-1;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
