#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    if(t>10 || t==0 )
    {
    	cout<<"Invalid Test"<<endl;
    }
    else
    {
    	while(t--)
    	{
    		string str;
    		cin>>str;
    		if(str.length()>100)
    		{
    			cout<<"Invalid Input"<<endl;
    			continue;
    		}
    		int count1=0;
    		int count2=0;
    		for(int i=0;i<str.length();i++)
    		{
    			if(str[i]>='a' && str[i]<='z')
    				count1++;
    			else if(str[i]>='A' && str[i]<='Z')
    				count2++;
    			
    		}
    		if(count1==0 && count2==0)
    			cout<<"Invalid Input"<<endl;
    		else
    			cout<<min(count1,count2)<<endl;
    		
    	}
    	
    }
    return 0;
}
