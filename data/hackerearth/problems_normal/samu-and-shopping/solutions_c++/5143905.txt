#include <bits/stdc++.h>
using namespace std;

vector<vector<int>> spec;

vector<vector<int>> dp;
int n;

int findmin(int i,int j){
	int minmum = INT_MAX;
	for(int t=0;t<3;t++){
		if(t!=j){
			minmum = min(minmum,dp[i][t]); 
		}
	}
	return minmum;
}

int getmin(){
	int minmum = INT_MAX;
	int i = n-1;
	for(int t=0;t<3;t++){
			minmum = min(minmum,dp[i][t]); 
	}
	return minmum;
}

int main() {
	int t;
	cin>>t;
	while(t--){
		
		spec.clear();
		dp.clear();
		
		cin>>n;
		dp.resize(n,vector<int>(3,INT_MAX));
		int temp;
		vector<int> row;
		for(int i=0;i<n;i++){
			cin>>temp;
			row.push_back(temp);
			cin>>temp;
			row.push_back(temp);
			cin>>temp;
			row.push_back(temp);
			spec.push_back(row);
			row.clear();
		}
		dp[0][0] = spec[0][0];
		dp[0][1] = spec[0][1];
		dp[0][2] = spec[0][2];
		
		for(int i=1;i<n;i++){
			for(int j=0;j<3;j++){
				dp[i][j] = findmin(i-1,j) + spec[i][j];
			}
		}
		cout<<getmin()<<endl;
	}
	return 0;
}
