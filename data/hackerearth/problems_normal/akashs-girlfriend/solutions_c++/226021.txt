#define MOD 1000000009
typedef long long ll;
long long mpow(long long a, long long b)
{
	long long ans = 1;
	while(b > 0)
	{
		if(b&1)
			ans = (ans * a) % MOD;
		a = (a*a) % MOD;
		b /= 2;
	}
	return ans % MOD;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        ll n;
        scanf("%lld",&n);
        n--;
        ll ans = ((5*mpow(10,n))%MOD - (4*mpow(8,n))%MOD + MOD)%MOD;
        printf("%lld\n",ans);
    }
return 0;
}
