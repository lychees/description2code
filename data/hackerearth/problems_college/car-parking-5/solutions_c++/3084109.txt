#include<bits/stdc++.h>

using namespace std;


int main()
{
    int t,l,i;
    string s;
    vector<char> v;
    cin>>t;
    while(t--)
    {
        cin>>s;
        l=s.size();

        for(int i=0;i<l;i++)
        {
            v.push_back('Y');
        }


        for(int i=0;i<l;i++)
        {

            if(s[i]=='B')
            {
                if(i-2>=0)
                {
                    v[i-1]='N';
                    v[i-2]='N';
                }
                else if(i-1>=0)
                {
                    v[i-1]='N';
                }
                v[i]='N';
            }
            else if(s[i]=='D')
            {
                v[i]='N';
            }
            else if(s[i]=='S')
            {
                if(i-1>=0)
                    v[i-1]='N';
                if(i+1<=l-1)
                    v[i+1]='N';

                v[i]='N';

            }

        }
        int sum=0;
        for(int i=0;i<l;i++)
        {
            if(v[i]=='Y')
                sum++;
        }

        cout<<sum<<endl;

        v.clear();
    }


    return 0;
}
