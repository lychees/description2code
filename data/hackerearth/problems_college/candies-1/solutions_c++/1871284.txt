#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	long n;
	while(t--){
		cin>>n;
		long a[n],sum=0;
		for(long i=0;i<n;i++) a[i]=0;
		for(long i=0;i<n;i++){
			cin>>a[i];
			sum+=a[i];
		}
		
		
		if(sum%n==0){
			cout<<"YES"<<endl;
		}else cout<<"NO"<<endl;
		
		cout<< sum/n <<" " << sum-n*(sum/n)<<endl;
		
	}
	return 0;
}
