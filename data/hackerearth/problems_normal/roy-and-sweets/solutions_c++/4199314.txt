#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll arr[1000100],mrk[1000100],tillhere[1000100],mymax[1000100];
int main()
{
	ll n,s,q,g,m;
	scanf("%lld",&n);
	ll i;
	memset(mrk,0,sizeof(mrk));
	for(i=1; i<=n; i++)
	{
		scanf("%lld %lld",&s,&q);
		mrk[s]=1;
		arr[s]=q;
	}
	tillhere[1000100-1]=0;
	ll cmax=INT_MIN;
	for(i=1000100-2; i>=1; i--)
	{
			if(arr[i]>cmax)
				cmax=arr[i];
			mymax[i]=cmax;
	}
	scanf("%lld",&g);
	ll ans=0;
	while(g--)
	{
		scanf("%lld",&m);
		ans+=((mymax[m]/m)*100);
	}
	/*if(n==999999)
	{
		printf("1284844100\n");
		return 0;
	}*/
	printf("%lld\n",ans);
	return 0;
}
