#include <bits/stdc++.h>
using namespace std;
void countPencil(double *a,long long int n)
{
	long long int i=0,j=n-1;
	long long int mad=1;
	long long int ruhi=1;
	while(i<=j)
	{
		if(i==j)
		{
			mad++;
			break;
		}
		if(i<j)
		{
			double p=a[i];
			p=p/2;
			double k=a[j];
			double diff=k-p;
			if(diff>0)
			{
				mad++;
				a[j]=diff;
			}
			else if(diff==0)
				{
				j--;
			}
			else
			{
				
				diff=-diff;
				j=j-1;
				while(j>i)
				{
					ruhi++;
					diff=a[j]-diff;
					if(diff < 0)
					{
						diff=-diff;
						j--;
					}
					else
					{
						a[j]=diff;
						break;
					}
				}
			}
			i++;
		}
	}
	if(i==j)
	{
		i=i;
		cout<<i<<" ";
		cout<<n-j;
	}
	else
	{
		cout<<i<<" "<<n-j-1;
	}
//	cout<<"i = "<<i<<" i-1 = "<<i-1<<" j = "<<j<<" j-1 =   "<<j-1<<"\n";
//	cout<<mad<<" " ;
//	cout<<ruhi;
}
int main() 
{
	long long int t;
	cin>>t;
	while(t--)
	{
		long int n;
		cin>>n;
		long long int i=0;
		double a[n];
		while(i<n)
		{
			cin>>a[i];
			i++;
		}
		countPencil(a,n);
		cout<<endl;
	}
}
