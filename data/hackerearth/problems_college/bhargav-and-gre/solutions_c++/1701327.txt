#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<cstdlib>
#include<algorithm>
using namespace std;
int main() {
	int i,j,N;
	long long int anto=0, syn=0;
	string ref;
	string curnt[100000];
	cin>>ref;
	cin>>N;
	for(i=0;i<N;i++){
		cin>>curnt[i];
	}
	for(i=0;i<N;i++){
		if(ref.length() == curnt[i].length()){
			if(ref[0] == curnt[i][0])
				syn += 1;
			if(abs(ref[0]-curnt[i][0]) == 3)
				anto += 1;
		}
	}
	cout<<syn<<" "<<anto;
	return 0;
}
