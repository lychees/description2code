#include <stdio.h>
#include <vector>

using namespace std;

int v[100005] = {0};
int f[100005];

vector <int> g[100005];

int count (int node)
{
	if (v[node] == 0)
	{
	v[node] = 1;
	int sum = 0;
	for (int i = 0 ; i < g[node].size() ; i++)
	{
		sum += count (g[node][i]);
	}
	return f[node] =  sum+1;
	}
	return 0;
}

int main ()
{
	int n;
	scanf ("%d",&n);
	for (int i = 0 ; i < n-1 ; i++) {
		int x,y;
		scanf ("%d %d",&x,&y);
		g[x].push_back (y);
		g[y].push_back (x);
	}

	f[1] = count (1);
	printf ("%d ",f[1]-1);
	for (int i = 2 ; i <= n ; i++) printf ("%d ",f[i]);
	printf ("\n");

	return 0;
}
