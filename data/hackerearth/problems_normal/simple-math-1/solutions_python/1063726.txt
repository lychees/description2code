import math
n,m=map(int,raw_input().split())
if(n>=m):
	n=n-1
	m=m-1
	a=math.factorial(n)
	b=math.factorial(m)
	c=math.factorial(n-m)
	ans=a/(b*c)
	print ans%10000007
else:
	print -1
