#include<bits/stdc++.h>
#define  den 1000000001
typedef unsigned long long int lu;
using namespace std;
int main()
{
	lu t,n,i,ans,gans;
	scanf("%llu",&t);
	while(t--)
	{
		scanf("%llu",&n);
		ans=(n*(n-1))/2;
		gans=ans;
		for(i=n-1;i>=1;i--)
		{
			gans=gans-i;
			ans=ans+gans;
		}
		ans=ans*2;
		printf("%llu\n",ans);
	}
	return 0;
}
