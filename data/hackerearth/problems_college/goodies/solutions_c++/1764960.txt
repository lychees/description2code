#include <iostream>
#include <algorithm>
using namespace std;
int main ()
{
	long T,no,a[100000];
	cin>>T;
	while (T--)
	{
		cin>>no;
		long count=1;
		for (long i=0;i<no;i++)
			cin>>a[i];
		sort (a,&a[no]);
		for (long j=no-1;j>=0;j--)
		{
			if (a[j]==a[j-1])
				count++;
			else
				break;
		}
		cout<<count<<"\n";
	}
}
