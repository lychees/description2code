#include<bits/stdc++.h>
using namespace std;
int main()
{
   int t;
   std::ios::sync_with_stdio(false);
   cin>>t;
   while(t--)
   {
       long long n;
       cin>>n;
       string s;
       cin>>s;
       long long c1=0,i;
       for(i=0;i<n;i++)
       {
           if(s[i]=='1')
              c1++;
       }
      //  cout<<c1<<"\n";
       long long k=0,swaps=0;
       for(i=0;i<n;i++){
          if(k==c1)
              break;
          if(s[i]=='0')
            swaps++;
          k++;
       }
        k=0;
        long long swapr=0;
       for(i=n-1;i>=0;i--)
       {
          if(k==c1)
              break;
          if(s[i]=='0')
            swapr++;
          k++;
       }
       cout<<min(swaps,swapr)<<"\n";
   }
   return 0;
}
