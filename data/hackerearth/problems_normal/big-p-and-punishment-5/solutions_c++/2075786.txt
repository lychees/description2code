#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n,f,s; cin>>n>>f>>s;
		vector<int> v[n+1];
		for(int i=0; i<f; i++){
			int x,y; cin>>x>>y; v[x].push_back(y);
		}
		int ans=0;
		int visited[n+1]; memset(visited,0,sizeof visited);
		for(int i=0; i<s; i++){
			int x; cin>>x;
			if(!visited[x]){
				visited[x]=1; ans++;
				queue<int> q;
				q.push(x);
				while(!q.empty()){
					int node=q.front();
					q.pop();
					for(int j=0; j<v[node].size(); j++){
						if(!visited[v[node][j]]){
							visited[v[node][j]]=1; ans++;
							q.push(v[node][j]);
						}
					}
				}
			}
		}
		cout<<ans<<endl;
	}
}
