using namespace std;
int dp[100005];
int get(int x)
{
    string s("");
    while(x)
    {
        s=(char) ((x%2)+48 ) + s;
        x>>=1;
    }
    int ct=0;
    int to=s.size()-2;
    for(int i=0;i<to;i++)
    {
        if(s[i]=='1' && s[i+1]=='0' && s[i+2]=='1') ct++;
    }
    return ct;
}
int main()
{
    REPU(i,1,100001)
        dp[i]=get(i);
    int t; si(t);
    while(t--)
    {
        int r; int k; si(r); si(k);
        int ans=0;
        REP(i,r+1)
            if(dp[i]>=k) ans++;
        cout<<ans<<endl;
    }
	return 0;
}
