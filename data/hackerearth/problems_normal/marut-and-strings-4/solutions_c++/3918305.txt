#include<cstdio>
#include<iostream>
#include<algorithm>
#include<string>

#define ll long long

using namespace std;
int main()
{
	int t;
	bool chk;
	cin>>t;
	string s;
	if(t < 1 || t > 10) cout<<"Invalid Test";
	else
		{
		  while (t--)
	    	{
			  cin>>s;
			  if(s.length() > 100 || s.length() == 0)
		        cout<<"Invalid Input";
		      else
		      {
		    	string :: iterator it = s.begin();
		    	
		    		it = s.begin();
                    ll lower_cnt,upper_cnt;
                    lower_cnt = upper_cnt = 0;
                    while(it != s.end())
                    {
                    	
                    	if(*it >= 'a' && *it <= 'z') lower_cnt++;
                    	else if(*it >= 'A' && *it <= 'Z') upper_cnt++;
                    	else ;
                    	++it;
                    }
                    //cout<<lower_cnt<<" "<<upper_cnt<<"\n";
                    if(lower_cnt == 0 && upper_cnt == 0) cout<<"Invalid Input";
                    else cout<<min(lower_cnt,upper_cnt);
		       }
		       cout<<"\n";
		  }
		}
		cout<<"\n";
	return 0;
}
