const double pi = 3.14159265358979323846264338327950288419716939937510;
using namespace std;
const int maxx=1000005;
bool isprime[maxx]={0};
int A[maxx],cnt[maxx]={0};
int util[maxx]={0};
vector<vector<int> > pf (maxx);     

void sieve()
{
    int i,j;
    int p,a;
    isprime[1]=1;
    for(i=2;i<maxx;i++)
    {
        if(!isprime[i])
        {
            pf[i].pb(i);
            util[i]+=cnt[i]; 
            for(j=2;i*j<maxx;j++)
            {
                isprime[i*j]=1;
                pf[i*j].pb(i);
                p=j;
                a=1;
                while(p%i==0)
                {
                    p/=i;
                    ++a;
                }
                util[i]+=a*cnt[i*j]; //if(i==2&&j<10) cout<<a<<" "<<i<<" "<<j<<endl;
            }
        }
    }
}
 
int main()
{
    int X,Q,N;
    cin>>N; 
    _for(i,0,N)
    {
        cin>>A[i]; 
        ++cnt[A[i]];
    } 
    sieve(); 
    cin>>Q; //cout<<pf[49].sz<<endl;
    int temp,ans; 
    while(Q--)
    {
        cin>>X;
        temp=X;
        ans=100000007;
           
        _for(i,0,pf[X].sz)
        {
            int a=0,p=X;
            while(p%pf[X][i]==0)
            {
                ++a;
                p/=pf[X][i];
            }
            
            ans=min(ans,util[pf[X][i]]/a);
        }
        cout<<ans<<endl;
    }
    return 0;
}
