    #include <iostream>
    using namespace std;
    long long Ae[1000];
    int main()
    {
    	long long ie,Te,ne;
        cin>>Te;
        while(Te--){
        	cin>>ne;
        	Ae[0]=1;
        	Ae[1]=2;
        	for(ie=2;ie<=ne;ie++) 
        		Ae[ie]=Ae[ie-1]+Ae[ie-2];
        	cout<<Ae[ne]*Ae[ne]<<endl;
        }
        return 0;
    }
