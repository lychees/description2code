#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    int n,i,j;
    cin>>n;
    vector<int> v(n);
    for(i=0;i<n;i++)
    cin>>v[i];
    for(i=0;i<n;i++)
    {
        int temp=0;
        for(j=0;j<n;j++)
        {
            if(v[j]>v[i])
            temp++;
        }
        cout<<temp+1;
        if(i<n-1) cout<<" ";
    }
    return 0;
}
