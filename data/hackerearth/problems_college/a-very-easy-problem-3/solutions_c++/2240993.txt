#include <bits/stdc++.h>
using namespace std;

#define INF 10007

typedef long long ll;
typedef unsigned long long ull;
typedef vector<int> vi;
typedef vector<ll> vll;
typedef vector<vector<int> > vvi;
typedef pair<int,int> ii;
typedef vector<pair<int,int> > vii;
typedef vector<vector<pair<int,int> > > vvii;

#define all(x) x.begin(), x.end()
#define tr(x,it) for(auto it = x.begin();it!=x.end();++it)
#define ntr(x,it) for(auto it = x.rbegin();it!=x.rend();++it)
#define sz(a) int((a).size()) 
#define pb push_back 
#define MP make_pair
#define fpresent(c,x) ((c).find(x) != (c).end())  // set,map
#define present(c,x) (find(all(c),x) != (c).end())  //vector
#define F first
#define S second
#define FOR(i,a,b) for(int i=a;i<=b;++i)
#define NFOR(i,a,b) for(int i=a;i>=b;--i)




map<ll,int> s1,s2;
map<ll,int>temp1;
int main()
{
    ios_base::sync_with_stdio(false);cin.tie(0);cout.tie(0);
    int T=1;
    cin>>T;
    FOR(Tc,1,T)
    {
        int n;
        cin>>n;
        vll a1,a2;
      	vi c1,c2;
      	int x;
        FOR(i,0,n-1)
        {
        	cin>>x;
        	if(i&1)a1.pb(x);
        	else a2.pb(x);
        }
        FOR(i,0,n-1)
        {
        	cin>>x;
        	if(i&1)c1.pb(x);
        	else c2.pb(x);
        }
        s1.clear();
        s2.clear();
        
        int n1=a1.size();
        int n2=a2.size();
        
        s1[0]=s2[0]=1;
        FOR(i,0,n1-1)
        {
        	temp1.clear();
        	FOR(j,1,c1[i])
        	tr(s1,it)temp1[it->F+j*a1[i]]+=it->S;
        	tr(temp1,it)s1[it->F]+=it->S;
        }
        
        FOR(i,0,n2-1)
        {
        	temp1.clear();
        	FOR(j,1,c2[i])
        	tr(s2,it)temp1[it->F+j*a2[i]]+=it->S;
        	tr(temp1,it)s2[it->F]+=it->S;
        }
        ll V;cin>>V;
        int ans=0;
        
        tr(s1,it)ans+=(it->S)*s2[V-it->F],ans%=INF;
        
       
        cout<<ans<<"\n";
    }
	return 0;
}
