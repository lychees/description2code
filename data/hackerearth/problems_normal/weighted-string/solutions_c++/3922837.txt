#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
long long int tc,n,i,c=0,k,j;
cin>>tc;
while(tc--)
{
	cin>>k;
	string s;
	cin>>s;
	n=s.size();
	long long int a[1000000]={0};
	a[0]=s[0]-'a'+1;
	
	for(i=1;i<n;i++)
	{
		a[i]=a[i-1]+(s[i]-'a')+1;
	}
	c=0;
	int l;
	for(l=0;l<n;l++)
	{
		if(a[l]==k)
		c++;
       for(j=l+1;j<n;j++)
       {
       	  if(a[j]-a[l]==k)
       	  c++;
       	  
       	  if(a[j]-a[l]>k)
       	  break;
       }	
	}
	cout<<c<<"\n";
	s.clear();
}
    return 0;
}
