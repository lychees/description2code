#include <bits/stdc++.h>
using namespace std;


int main()
{
    //freopen("file1.txt","r",stdin);
    int t;
    cin>>t;
    while(t--)
    {
    	int m,n;
    	cin>>m>>n;
    	int arr[n+1];
    	int dp[10009];
		memset(dp,0,sizeof dp);
    	for(int i=1;i<=n;i++)
    	{
    		cin>>arr[i];
    		
    	}
		
		dp[0] = 0;
		for(int i=1;i<=m;i++)
		{
			dp[i] = 0;
			for(int j=1;j<=n;j++)
			{
				if(i>=arr[j])
					if(dp[i-arr[j]]==0)
					{
						dp[i] = 1;
						break;
					}
					
			}
		}
		if(dp[m])
    		cout<<"Little Deepu";
    	else cout<<"Kate";
    	cout<<endl;
    }
}
