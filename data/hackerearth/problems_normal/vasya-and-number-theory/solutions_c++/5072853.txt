#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n], dp[n];
    for(int i=0; i<n; i++)
    	cin>>a[i];
    for(int i=0; i<n; i++)
    	dp[i]=1;
    sort(a, a+n);
    for(int i=1; i<n; i++)
    {
    	for(int j=0; j<i; j++)
    	{
    		if(a[i]%a[j]==0)
    			dp[i]=max(dp[i], dp[j]+1);
    	}
    	//cout<<dp[i]<<"\n";
    }
    int res=0;
    for(int i=0; i<n; i++)
    	res = max(res, dp[i]);
    // for(int i=0; i<n; i++)
    // 	cout<<dp[i]<<" ";
    if(res>1)
    	cout<<res<<"\n";
    else
    	cout<<"-1\n";

    return 0;
}
