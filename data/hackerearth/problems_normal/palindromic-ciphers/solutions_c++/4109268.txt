#include <iostream>
#include<string>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	string input;
   	int check=1;
   	cin>>input;
   	int len=input.length();
   	int mid=len/2;
   	for(int i=0;i<=mid;i++)
   	{
   		if(input[i]!=input[len-1-i]){
   			check=0;
   			break;
   		}
   	}
   	if(check==1)cout<<"Palindrome"<<endl;
   	else{
   		long long ans=1;
   		for(int i=0;i<len;i++)
   		{
   			ans=ans*(input[i]-'a'+1);
   		}
   		cout<<ans<<endl;
   	}
   }
    return 0;
}
