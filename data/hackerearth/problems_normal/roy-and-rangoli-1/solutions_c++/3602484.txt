#include <iostream>
#include <bits/stdc++.h>
#define MOD 1000000007

using namespace std;

int seive[10000000]={0};

int main()
{
    long int n,i,j;
    scanf("%ld",&n);
    long int x=2*(n-1);
    seive[0]=1;
    seive[1]=1;
    for(i=2;i*i<=x;i++)
    {
        if(seive[i]==0)
        {
            for(j=2*i;j<=x;j+=i)
            {
                seive[j]=1;
            }
        }
    }
    long long int c=0;
    long int d;
    for(i=0;i<=x;i++)
    {
        if(seive[i]==0)
        {
            d=abs(n-1-i);
            d=n-d;
            c=((c%MOD)+(d%MOD))%MOD;
        }
    }
    printf("%lld",c);
    return 0;
}
