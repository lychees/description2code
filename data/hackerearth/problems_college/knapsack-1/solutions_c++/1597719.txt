void knapsack(int maxw,int nitem){
	int i,j,k;
	for(i=0;i<=maxw;i++){
		knpsk[0][i]=0;
	}
	for(i=0;i<=nitem;i++){
		knpsk[i][0]=0;
	}
	for(i=1;i<=nitem;i++){
		for(j=0;j<=maxw;j++){
			knpsk[i][j]=knpsk[i-1][j];
			k=1;
			while((j-(k*a[i]))>=0){
				knpsk[i][j]=max(knpsk[i][j],knpsk[i-1][j-(k*a[i])]+(k*a[i]));
				k++;
			}
		}
	}
}
