short IsPrime( long long P)
{
	if(P==2 || P==3 || P==5)
		return 0;
	else if (P==0 || P==1 || P%2==0 || P%3==0 || P%5==0)
		return 1;
	else 
	{
		 long long i;

		for (i=1;i<=( long long)(ceil(sqrt(P)/6));i++)
		{
			if (i*6+1==P || i*6-1==P)
				return 0;
			else if(P%(6*i+1)==0 || P%(6*i-1)==0)
				return 1;
		}
		return 0;
	}
}
