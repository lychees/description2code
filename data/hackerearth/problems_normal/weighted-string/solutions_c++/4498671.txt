#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,n,t,k,l,start,end,count;
	scanf("%d",&t);
	while(t--)
	{
		char s[1000009];
		scanf("%d",&k);
		scanf("%s",s);
		n=strlen(s);
		int num[n+9];
		num[0]=0,count=0;
		start=0,end=1;
		for(i=1;i<=n;i++)
			num[i]=num[i-1]+s[i-1]-'a'+1;
			while(end<=n)
			{
				if(num[end]-num[start]==k)
				{
					count++;
					start++;
					end++;
				}
				else if(num[end]-num[start]<k)
				end++;
				else
				start++;
			}
			printf("%d\n",count);
	}
	return 0;
}
