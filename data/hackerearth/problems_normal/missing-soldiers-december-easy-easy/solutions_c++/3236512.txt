#define MOD 1000000009
struct data
{
	int x;
	int type;
}d[222222];
bool cmp(data d1,data d2)
{
	return d1.x<d2.x||(d1.x==d2.x&&d1.type<d2.type);
}
int main()
{
	std::ios::sync_with_stdio(false);cin.tie(0);
	int i,n;
	int re=0;
	int x,y,l;
	int count=1;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>x>>y>>l;
		d[i*2].x=x;
		d[i*2].type=1;
		d[i*2+1].x=x+l;
		d[i*2+1].type=2;
	}
	n*=2;
	sort(d,d+n,cmp);
	for(i=1;i<n;i++)
	{
		if(d[i].type==2)
		{
			count--;
			re+=d[i].x-d[i-1].x;
			if(count==0)
				re++;
		}
		else
		{
			count++;
			if(count>1)
				re+=d[i].x-d[i-1].x;
		}
	}
	cout<<re<<endl;
//	system("pause");
	return 0;
}
