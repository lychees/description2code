#include <bits/stdc++.h>
using namespace std;

typedef long long LL;

#define PII       pair<int,int>
#define all(c)    c.begin(),c.end()
#define sz(c)     (int)c.size()
#define clr(c)    c.clear()
#define pb        push_back
#define mp        make_pair
#define cin(x)    scanf("%d",&x)
#define MOD		1000000007
#define EPS		1E-10

void solve()
{
	int x1,y1,x2,y2;
	cin >> x1 >> y1 >> x2 >> y2;
	double minD = INT_MAX;
	int r;
	cin >> r;
	r *= r;
	if(x1 == x2)
		minD = x1 * x1;
	else if(y1 == y2)
		minD = y1 * y1;
	else
	{
		int a = y2 - y1;
		int b = x2 - x1;
		int c = a * x1 - b * y1;
		if(c < 0)
			c = -c;
		minD = (1.0 * c * c) / (a * a + b * b);
	}
	double val = fabs(minD - r);
	if(val < EPS)
		printf("JUST MISSED\n");
	else if(minD < r)
		printf("REPLANNING\n");
	else
		printf("SAFE\n");

}

int main()
{
	int t;
	cin >> t;
	while(t--)
		solve();
	return 0;
}
