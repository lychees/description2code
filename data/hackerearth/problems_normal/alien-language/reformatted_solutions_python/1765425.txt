from collections import Counter
for x in range(input()):
    src = Counter([x for x in raw_input()])
    tgt = Counter([x for x in raw_input()])
    found = False
    for a in src:
        if tgt[a] > 0:
            found = True
            break
    print("YES" if found else "NO")
