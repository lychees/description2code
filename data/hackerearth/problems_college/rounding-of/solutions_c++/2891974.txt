#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

int main(){
    int t;
    lli n,b;
    scanf("%d",&t);
    while(t--){
        scanf("%lld %lld",&n,&b);
        if(n%b==0){
            printf("%lld\n",n);
            continue;
        }
        lli lw = (n/b)*b;
        lli hg = (n/b+1)*b;
        lli up = hg-n;
        lli dw = n-lw;
        if(up<=dw)
            printf("%lld\n",hg);
        else
            printf("%lld\n",lw);
    }
    return 0;
}
