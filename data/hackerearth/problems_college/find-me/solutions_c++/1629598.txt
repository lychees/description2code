#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
	int t,n,p,a[10005],i;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&p);
		int pres=0,cnt=0;
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			if(a[i]==p)pres=1;
		}
		if(pres==0)
		{
			cout<<"NO 0\n";
		}
		else
		{
			//a.sort(0,n-1);
			for(i=0;i<n;i++)
			if(a[i]>p)cnt++;
			cout<<"YES "<<cnt<<endl;
		}
	}
	return 0;
}
