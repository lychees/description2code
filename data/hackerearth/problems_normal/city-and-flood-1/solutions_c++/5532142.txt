#include<iostream>
#include<bits/stdc++.h>

using namespace std;

int arr[100000],size[100000];

int root1(int arr[],int i)
{
        while(i != arr[i])
        {
                i=arr[i];
        }
        return i;
}

void union1(int arr[],int size[],int i,int j)
{
        int ri=root1(arr,i);
        int rj=root1(arr,j);
        if(size[i] <= size[j])
        {
                arr[ri]=arr[rj];
                size[rj]+=size[ri];
        }
        else
        {
                arr[rj]=arr[ri];
                size[ri]+=size[rj];
        }
}

int main()
{
	int i,j,n,k,l;
	cin>>n;
//	int arr[n],size[n];
	for(i=0;i<n;i++)
	{
		arr[i]=i;
		size[i]=1;
	}
	cin>>k;
	while(k--)
	{
		cin>>i>>j;
		union1(arr,size,i,j);
	}

	set<int> s;
	for(l=0;l<n;l++)
		s.insert(root1(arr,arr[l]));
	cout<<s.size()<<endl;
	return 0;
}
