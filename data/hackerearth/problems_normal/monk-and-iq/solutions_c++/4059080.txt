#include<bits/stdc++.h>
using namespace std;
vector<pair<int,int> > v;
typedef pair<int,int> P;
priority_queue<P,vector<P>,greater<P > > pq;
pair<int,int> pr;
int p,n,c,f[100000],en[100000],iq1[100000],iq2[100000],no[100000];
int main()
{
    int i,j,in,mn,l,r,m;
    scanf("%d %d %d",&c,&p,&n);
    for(i=0;i<n;i++)
    {
        no[i]=1;
        scanf("%d",en+i);
        iq2[i]=en[i];
    }
    for(i=0;i<p;i++)
        scanf("%d",f+i);
    for(i=0;i<c;i++)
        pq.push({en[i],i});
    for(i=0;i<p;i++)
    {
        pr=pq.top();
        pq.pop();
        mn=pr.second;
        printf("%d ",mn+1);
        no[mn]++;
        iq1[mn]=iq2[mn];
        iq2[mn]=f[i];
        pr.first=no[mn]*(iq1[mn]+iq2[mn]);
        pq.push(pr);
    }
}
