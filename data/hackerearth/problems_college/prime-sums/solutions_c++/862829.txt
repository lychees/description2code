#include <bits/stdc++.h>

using namespace std;
//int i,j;
bool isprime[4000000];

int main()
{
	int i,j;
	for (i = 2; i <= 4000000; i++) {
			isprime[i] = true;
	}
	for (i = 2; i*i <= 4000000; i++) {
		if (isprime[i]) {
			for (j = i*i; j <= 4000000; j+=i) {
				isprime[j] = false;
			}
		}
	}
	int n;
	unsigned long long ans;
	ans = 0;
	scanf("%d",&n);
	for (i = 2; i < n; i++) {
		if (isprime[i]) {
			ans +=i;
		}
	}
	printf("%llu\n",ans);
	
	return 0;
}
