#include <iostream>
using namespace std;
int sod(int num)
{
	int digits=0;
	while(num>0)
	{
		digits++;
		num=num/10;
	}
	return digits;
}
int main()
{
    long int t,n,count=1;
    cin>>t;
    while(t--)
    {
    	int sum=0;
    	cin>>n;
    	while(sod(n)>=2)
    	{
    		while(n>0)
    		{
    			int rem=n%10;
    			sum+=rem*rem;
    			n=n/10;
    		}
    		n=sum;
    		sum=0;
    	}
    	if(n == 1)
    		cout<<"Case #"<<count<<":YES"<<endl;
    	else
    		cout<<"Case #"<<count<<":NO"<<endl;
    	count++;
    }
    return 0;
}
