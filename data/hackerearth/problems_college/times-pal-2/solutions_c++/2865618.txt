#include <iostream>
#include <cstdio>
#include <string>
using namespace std;

int isPalin(string str)
{
	
	for (long int i=0,j=str.size()-1; i < j ;i++,j--)
	{
		if(str[i] != str[j])
			return 0;
	}	
	return 1;
}

int main() {
	
	int t;
	scanf("%d",&t);
	while(t--)
	{
		string s1,s2;
		cin>>s1;
		cin>>s2;
		
		long int count = 0;
		
		for(long int i=0;i<=s1.size();i++)
		{
			s1.insert(i,s2);
			if(isPalin(s1))
			{
				count++;
			}
			s1.erase(i,s2.size());
			
		}
		printf("%ld\n",count);
	}
	return 0;
}
