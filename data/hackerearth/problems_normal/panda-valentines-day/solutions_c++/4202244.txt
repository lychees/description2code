#include <iostream>
#include <vector>
using namespace std;
 
vector<unsigned> ans(1000001, 0);
 
void prepare() {
	//vector<unsigned> primes;
	//primes.reserve(300);
	//vector<int> sieve(1000, 0);
	for (unsigned j = 2; j <= 1000000; j += 2) {
		unsigned tmp = j;
		while (tmp % 2 == 0) {
			++ans[j];
			tmp /= 2;
		}
	}
	
	for (unsigned i = 3; i <= 1000000; i += 2)
		if (ans[i] == 0) {
			for (unsigned j = i; j <= 1000000; j += i) {
				unsigned tmp = j;
				while (tmp % i == 0) {
					++ans[j];
					tmp /= i;
				}
			}
		}
	
	for (unsigned i = 2; i <= 1000000; ++i) 
		ans[i] += ans[i - 1];
}
 
int main() {
	ios_base::sync_with_stdio(0);
    cin.tie(0);
	
	prepare();
	int t;
	cin >> t;
	while (t--) {
		unsigned x;
		cin >> x;
		cout << ans[x] << '\n';
	}
    return 0;
}
