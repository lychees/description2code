#include <iostream>
#define sc(n) scanf("%ld",&n)
#define pr(n) printf("%ld\n",n)
using namespace std;

int main()
{
    long int n,q,g1,p1,g2,p2,i=0;
    sc(n);
    sc(q);
    long int b[n];
    sc(b[0]);
    i=1;
  
    for(;i<n;i++)
    {
    	sc(b[i]);
    	b[i]=b[i-1]+b[i];
    }
    while(q--)
    {
    	
    	sc(g1);
    	sc(p1);
    	sc(g2);
    	sc(p2);
        	if(g1 < p1)
		{
			if(p1 <= g2)
			{
				if(p2 < g2)
				{
					if(g1 > 0)
						pr(b[g2]-b[g1-1]);
					else
						pr(b[g2]);
				}
				else
				{
					if(g1 > 0)
						pr(b[p2]-b[g1-1]);
					else
						pr(b[p2]);
				}
			}
			else
			{
				if(g1 > 0)
					pr((b[g2]-b[g1-1]) +(b[p2]-b[p1-1]));
				else
					pr(b[g2] + (b[p2]-b[p1-1]));
			}
		}
		else if(g1 == p1)
		{
			if(p1 < g2)
			{
				if(g2 < p2)
				{
					if(p1 > 0)
						pr(b[p2]-b[p1-1]);
					else
						pr(b[p2]);
				}
				else
				{
					if(g1 > 0)
						pr(b[g2]-b[g1-1]);
					else
						pr(b[g2]);
				}
			}
			else if(p1 == g2)
			{
				if(g2 == p2)
				{
					if(g2 > 0)
						pr(b[g2]-b[g2-1]);
					else
						pr(b[g2]);
				}
				else
				{
					if(p1 > 0)
						pr(b[p2] - b[p1-1]);
					else
						pr(b[p2]);
				}
			}
		}
		else if(p1 < g1)
		{
			if(g1 <= p2)
			{
				if(g2 < p2)
				{
					if(p1 > 0)
						pr(b[p2]-b[p1-1]);
					else
						pr(b[p2]);
				}
				else
				{
					if(p1 > 0)
						pr(b[g2] - b[p1-1]);
					else
						pr(b[g2]);
				}
			}
			else
			{
				if(p1 > 0)
					pr((b[p2]-b[p1-1])+(b[g2]-b[g1-1]) );
				else
					pr( b[p2] + (b[g2]-b[g1-1]));
			}
		}
    	
    }
    return 0;
}
