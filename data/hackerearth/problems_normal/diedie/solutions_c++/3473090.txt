#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define MOD 1000000007
ll modpow(ll a,ll b,ll m)
{
	ll res=1;
	while(b)
	{
		if(b%2)
			res=(res*a)%m;
		a=(a*a)%m;
		b>>=1;
	}
	return res;
}
int main()
{
	ll t,n,ans;
	cin>>t;
	ans=modpow(2,MOD-2,MOD);
      while(t--)
      {
      	cin>>n;
      	cout<<modpow(ans,n-1,MOD)<<endl;
      }
      return 0;

}
