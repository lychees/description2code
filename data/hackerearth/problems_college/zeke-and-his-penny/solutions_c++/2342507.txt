#include <iostream>
#include <stdio.h>
#include <algorithm>
#define MOD 1000000007
using namespace std;

int main()
{
    int te;
    scanf("%d",&te);
    while(te--) {
    	long long int a[3];
    	int i;
    	for(i=0;i<3;i++)
    		scanf("%lld",&a[i]);
    	sort(a,a+3);
    	long long int ans=1;
    	for(i=0;i<3;i++)
    		ans=(ans*((a[i]-i)%MOD))%MOD;
    	printf("%lld\n",ans);
    }
    return 0;
}
