
#include<bits/stdc++.h>
#define ll long long 
using namespace std;
ll int arr[105][105];
int main()
{
    ll int t,i,j,count1=0,count2=0,ans;
    scanf("%lld",&t);
    
    for(i=0; i<t; i++)
    {
        for(j=0; j<t; j++)
        {
            cin>>arr[i][j];
        }
        count1 += arr[i][i];
        count2 += arr[i][t - i - 1];
    }
    ans=abs(count1-count2);
    cout<<ans<<endl;
    
    return 0;
}
