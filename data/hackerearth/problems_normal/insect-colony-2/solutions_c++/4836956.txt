#include <iostream>

using namespace std;

int main()
{
	int i, j, tc, n, a;
	long long int temp;
	cin >> tc;
	for(i = 0; i < tc; i++)
	{
		cin >> n;
		a = 0;
		for(j = 0; j < n; j++)
		{
			cin >> temp;
			if(temp % 2)
				if(a == 1)
					a = 0;
				else
					a = 1;
		}
		if(a == 1)
			cout << "No" << endl;
		else
			cout << "Yes" << endl;
	}
	return 0;
}
