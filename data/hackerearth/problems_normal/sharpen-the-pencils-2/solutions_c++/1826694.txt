#include <iostream>
using namespace std;

int main()
{
	double *a;
    int i, j, t, n, m, r, pm, pr, *b;
    cin>>t;
    for(i=0; i<t; i++)
    {
    	pm=0;pr=0;
    	cin>>n;
    	a=(double *)malloc(sizeof(double)*n);
    	b=(int *)malloc(sizeof(int)*n);
    	for(j=0; j<n; j++)
    	{
    		cin>>a[j];
    		b[j]=0;
    	}
    	//cout<<"Rth pen="<<n-1;
    	m=0;
    	r=n-1;
    	while(m<r)
    	{
    		//cout<<"\nM="<<a[m]<<" R="<<a[r];
    		a[m]/=2;
    		b[m]=1;
    		b[r]=2;
    		if(a[m]>a[r])
    		{
    			pr++;
    			//cout<<"\nPR inc="<<pr;
    			a[m]*=2;
    			a[m]-=(a[r]*2);
    			r--;
    		}
    		else if(a[m]<a[r])
    		{
    			pm++;
    			//cout<<"\nPM inc="<<pm;
    			a[r]-=a[m];
    			m++;
    		}
    		else
    		{
    			m++;
    			r--;
    			pm++;
    			pr++;
    		}
    	}
    	if(m==r)
    	{
    		if(b[m]==0 )
    		{
    			pm++;
    		}
    		else if(b[m]==2)
    		{
    			pr++;
    			//cout<<"kjjsa";
    		}
    		else
    		{
    			pm++;
    			//cout<<"h3\n m="<<m<<" r="<<r;
    		}
    	}
    	
    	cout<<pm<<" "<<pr<<"\n";
    	free(a);
    	free(b);
    }
    return 0;
}
