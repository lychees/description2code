#include <iostream>
using namespace std;

int main()
{
	int t,nn,i,n;
    cin>>t;
    while(t--) {
    	long long sum=0;
    	int a[1000005]={0};
    	int b[1000005]={0};
    	cin>>n;
    	for(i=0;i<n;i++) {
    		cin>>nn;
    		if(nn<0) {
    			nn=(-1*nn);
    			b[nn]++;
    		}
    		else if(nn>=0)
    			a[nn]++;
    	}
    	for(i=0;i<=1000001;i++) {
    		sum+=(a[i]*(a[i]-1)/2);
    	}
    	for(i=0;i<=1000001;i++) {
    		sum+=(b[i]*(b[i]-1)/2);
    	}
    	sum+=n;
    	printf("%lld\n",sum);
    }
    return 0;
}
