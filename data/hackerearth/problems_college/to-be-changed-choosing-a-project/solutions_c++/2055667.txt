int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(NULL);
    LL n,r,i,l,x,flag=0;
    set <LL> s;
    set <LL>::iterator it,it1;
    cin >>n>>r;
    for(i=1;i<=n;i++)
    {
        s.insert(i);
    }
    it=s.begin();
    l=s.size();
    while(l!=1)
    {
        x=r%l;
        if(x==0)
            x=l;
        flag=0;
        while(x!=0)
        {
            if(flag==0)
            {
                flag=1;
                x--;
                continue;
            }
            if(it==(--s.end()))
                it=s.begin();
            else
                it++;
            x--;
        }
        it1=it;
        if(it1==(--(s.end())))
        {
            it=s.begin();
        }
        else
            it++;
        s.erase(it1);
        l--;
    }
    cout <<*(s.begin())<<"\n";
    return 0;
}
