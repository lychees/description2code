#include<cstdio>
#include<cstring>
#define MOD 1000000007
 char s[100000+11];
 
long long POW(long long a,long long n){
    long long ans=1;
    while(n){
        if(n&1){
            ans=(ans*a)%MOD;
        }
        a=(a*a)%MOD;
        n>>=1;
    }
    return ans;
}

long long solve(){
    long long  ans=0;
    for(int i=0;s[i]!='\0';i++){
        ans=((ans*10)%(MOD-1)+(s[i]-'0'))%(MOD-1);
    }
    return ans;
}
int main(){
    long long  a;
    scanf("%lld%s",&a,s);
    long long n=solve();
    n%=MOD;
    long long ans=POW(a,n);
    printf("%lld\n",ans);
    return 0;
}
