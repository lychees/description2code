#include<bits/stdc++.h>
using namespace std;

long long gcd ( long long a, long long b )
{
   if ( b == 0 )
      return a;
   else return gcd ( b, a%b );
}

int main()
{
   int t;
   long long int l, b, h;
   cin >> t;
   while (t--)
   {
      cin >> l >> b >> h;
      long long g = gcd ( l, b );
      long long a = gcd ( g, h );
      cout << a << " " << ( ( ( (l/a) * (b/a) ) % 1000000007) * (h/a) ) % 1000000007 << endl;
   }
   return 0;
}
