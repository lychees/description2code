#include<bits/stdc++.h>
using namespace std;
long long int str[1000000]={0};
long long int cumstr[1000000]={0};
long long int findzero(int x)
{
	long long int div=5;long long int c=0;
	while(x/div)
	{
		c+=(x/div);
		x/=div;
		
	}
	return c;
}
void init()
{
	int i;long long int c=0;
	for(i=1;i<=1000000;i++)
	{
		str[i]=findzero(i);
		c+=str[i];
		cumstr[i]+=c;
	}
}
int main()
{
	int t,a,b,i;long long int count;
	cin >> t;
	init();
	while(t--)
	{
		count=0;
		cin >> a >> b;
		count= cumstr[b] - cumstr[a] + str[a];
		cout << count << endl;
	}
	return 0;
}
