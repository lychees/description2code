{
	if((r1-l1)==(r2-l2) && s[l1]==s[l2] && s[r1]==s[r2])
	{
		/*for(int i=l1,j=l2;i<r1;i++,j++)
		{

				return false;
		}*/
		return true;
	}

}

int main()
{
    char s[100001];
    cin>>s;
    int cum=0;
    int l=strlen(s);
   	int a[l];
    for(int i=0;i<l;i++)
    {
    	cum+=(int)(s[i]-'a');
    	a[i]=cum;
    }
    int q;
    cin>>q;
    while(q--)
    {
    	int l1,r1;
    	int l2,r2;
    	cin>>l1>>r1;
    	cin>>l2>>r2;
    	l1--;r1--;
    	l2--;r2--;
    	int x,y;
    	if(l1==0)
    		x=a[r1];
    	else
    		x=a[r1]-a[l1-1];
    	if(l2==0)
    		y=a[r2];
    	else 
    		y=a[r2]-a[l2-1];
    	//cout<<x<<" "<<y<<" ";
    	if(x==y)
    	{
    		if(match(s,l1,r1,l2,r2))
    			cout<<"Yes\n";
    		else 
    			cout<<"No\n";
    	}
    	else

    }
    return 0;
}
