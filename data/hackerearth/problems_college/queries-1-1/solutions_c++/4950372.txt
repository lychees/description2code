void quicksort(long long int arr[],long long int l,long long int h){
	long long int p,j;
	long long int temp;
	long long int i=l-1;
	if(l<h){
	for(j=l;j<h;j++){
		if(arr[j]<=arr[h]){
			i++;
			temp=arr[i];
			arr[i]=arr[j];
			arr[j]=temp;
		}
	}temp=arr[i+1];
	arr[i+1]=arr[h];
	arr[h]=temp;
	p=i+1;
	quicksort(arr,l,p-1);
	quicksort(arr,p+1,h);
	
}
