#include <iostream>
using namespace std;

int main()
{
    string s,temp="";
    cin>>s;
    int n;
    cin>>n;
    string words[n];
    int i;
    for(i=0;i<n;i++)
    {
    	cin>>words[i];
    }
    int boo=0;
    for(i=0;i<n;i++)
    	if(words[i].substr(0,s.length())==s)
    	{
    		temp=words[i];
    		boo=1;
    		break;
    	}
    if(!boo)
    {
    	cout<<s;
    	return 0;
    }
    for(i=0;i<n;i++)
    {
    	if(words[i].substr(0,s.length())==s)
    	{
    		if(words[i].length()<temp.length())
    				temp=words[i];
    	}
    	
    }
    cout<<temp;
    return 0;
}
