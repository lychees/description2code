#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
   int n;
   cin>>n;
   vector<int> v(n);
   for(int i=0;i<n;i++)
   cin>>v[i];
   sort(v.begin(),v.end());
   vector<int> dp(n,1);
   int final_max=-1;
   for(int i=0;i<n;i++)
   {
   		for(int j=i-1;j>=0;j--)
   		{
   			if(v[i]%v[j]==0)
   			{
   				dp[i]=max(dp[i],dp[j]+1);
   				if(final_max<dp[i])
   				final_max=dp[i];
   			}
   		}
   }
   cout<<final_max<<endl;
}
