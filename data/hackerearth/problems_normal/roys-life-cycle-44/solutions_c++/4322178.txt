#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    string m = "";
    int x = 0;
    while(n--){
    	string s;
    	cin>>s;
    	int cnt = 0,temp = 0;
    	
    	for(int i = 0;i<s.length();i++){
    		if(s[i] == 'C'){
    			cnt++;
    		} else {
    			temp = max(temp , cnt);
    			cnt = 0;
    		}
    	}
    	if(s[s.length() - 1] == 'C'){
    		temp = max(temp , cnt);
    	}
    	x = max(x , temp);
    	m += s;
    }
    int cnt = 0,temp = 0;
    int y = 0;
    for(int i = 0;i<m.length();i++){
    	
    		if(m[i] == 'C'){
    			cnt++;
    		} else {
    			temp = max(temp , cnt);
    			cnt = 0;
    		}
    	
    	if(m[m.length() - 1] == 'C'){
    		temp = max(temp , cnt);
    	}
    	y = max(y , temp);
    }
    cout<<x<<" "<<y;
    return 0;
}
