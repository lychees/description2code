#include <iostream>
using namespace std;

int main()
{
    int t,i,l;
    cin>>t;
    string s;
    while(t--)
    {
    	cin>>s;
    	l = s.length();
    	for(i=0;i<l;i++)
    	{
    		if(s[i]+i > 122)
    		{
    			cout<<char(s[i]+i-26);
    		}
    		else
    		{
    			cout<<char(s[i]+i);
    		}
    		
    	}
    	cout<<"\n";
    }
    return 0;
}
