#include <bits/stdc++.h>
using namespace std;
#define ll long long int 

int main()
{
    int a, b, c;
    ll area = 0;
    cin >> a >> b >> c;
	
	if(a<b)
		swap(a,b);
	
	if(c >= a+b)
		area = 2*a*b;
	else if(c >=a && c >=b)
		area = 2*(c-b)*b + 2*(c-a)*(a-(c-b)) + (b-(c-a))*(a-(c-b));
	else if (c <= a && c >= b)
		area = 2*(c-b)*b + b*b;
	else
		area = c*c;
	
	ll total_area = 2*a*b;
	
	int gcd = __gcd(total_area, area);
	
	cout << area/gcd << "/" << total_area/gcd << endl;
}
