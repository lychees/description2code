#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s[]={ " ","Rohit", "Dhawan", "Kohli", "Yuvraj", "Raina" , "Dhoni", "Sir Jadeja"};
  		int index=1,k,size=1,flag=1;
  		cin>>k;
  		if(k<=7)
  		{
  			cout<<s[k]<<endl;
  			continue;
  		}
  		while(index<=k)
  		{
  			if(index+size>k )
  			{
  				if(index+1<=k)
  				{
  					flag=(flag%7)+1;
  				}
  				break;
  			}
  			index=index+size;
  			flag=(flag%7) +1;
  			if(flag==7)
  			{
  				size=size*2;
  				
  			}
  		//	cout<<"Flag is  "<<flag<<"    Index is "<<index<<endl;
  		}
  		cout<<s[flag]<<endl;
    }
    return 0;
}
