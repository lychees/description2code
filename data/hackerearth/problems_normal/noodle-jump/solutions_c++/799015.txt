#include <bits/stdc++.h>
#define LL long long
using namespace std;

int main () {
   LL miCoordinateCount = 0;
   LL miJumpMax = 0;
   LL miReachedCoordinate = 0;
   cin >> miCoordinateCount >> miJumpMax ;

   LL miCoor[miCoordinateCount];

   //Store all coordinates to an integer array
   for(LL i = 0; i < miCoordinateCount ; i++){ cin >> miCoor[i];}

   //sort the fucking array
   sort(miCoor, miCoor + miCoordinateCount);
   miReachedCoordinate = miCoor[0];
   
   //Jump to each coordinate
   for(LL i = 0; i < miCoordinateCount ; i++){
      if(miCoor[i] - miCoor[i - 1] <= miJumpMax){
         miReachedCoordinate = miCoor[i];
      }else{
         break;
      }
   }
   
   cout << miReachedCoordinate << endl;
   return 0;
}
