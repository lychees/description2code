int arr[100005];

//stack<pair<int,int> >s;
void dfs(int r)
{
    arr[r]=c++;
    for(int i=0;i<v[r].size();i++)
    {
        brr[v[r][i]]=c1++;
        dfs(v[r][i]);

    }
    brr[r]=c++;
}
