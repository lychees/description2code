template<class T> inline T gcd(T x, T y) { if (!y) return x; return gcd(y, x%y); }
template<class T> inline void read(T&x){bool fu=0;char c;for(c=getchar();c<=32;c=getchar());if(c=='-')fu=1,c=getchar();for(x=0;c>32;c=getchar())x=x*10+c-'0';if(fu)x=-x;};

const long MOD = 1000000007;
const int MAX = 1000005;
ll tree[MAX], a[MAX], tree1[MAX];
int n;

ll read(ll* tree, int idx)
{
    ll sum = 0;
    while(idx > 0)
    {
        sum += tree[idx];
        idx -= (idx & -idx);
    }
    return sum;
}


void update(ll* tree, int idx, ll val)
{
    while(idx <= n)
    {
        tree[idx] += val;
        idx += (idx & -idx);
    }
}

int main()
{
    //freopen("input.txt", "r", stdin);
    int l, r, q;
    read(n, q);
    for(int i = 1;i <= n;++i)
    {
        read(a[i]);
        update(tree, i, a[i]);
        update(tree1, i, -a[i]);
    }
    ll ans1, ans2;
    for(int i = 0;i < q;++i)
    {
        read(l, r);
        ans1 = read(tree, r) - read(tree, l-1);
        ans2 = a[r] + read(tree1, r-1) - read(tree1, l-1);
        printf("%lld %lld\n", ans1, ans2);
    }
    return 0;
}
