#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define sd(t) scanf("%d",&t)
#define sl(t) scanf("%lld",&t);
#define pd(t) printf("%d",t)
#define pl(t) printf("%lld",t)
#define br printf("\n")
#define sp printf(" ")
#define fi(i,x,y) for(long long i=x;i<y;i++)
#define fd(i,x,y) for(long long i=x;i>=y;i--)
ll t,q,x;
ll a[100010];
ll binarysearch(ll a[],ll x,ll u,ll l)
{
	ll low=l,high=u;
	while(low<=high)
	{
		if(a[u]<x)
		return -1;
		ll mid=(low+high)/2;
		if(a[mid]>=x&&a[mid-1]<x)
		return mid;
		else if(a[mid]>x)
		high= mid-1;
		else if(a[mid]<x)
		low=mid+1;
	}
}
int main()
{
	sl(t);
	a[0]=0;
	fi(i,1,t+1)
	{
		sl(a[i]);
		a[i]+=a[i-1];
	}
	sl(q);
	while(q--)
	{
		ll u=t,l=1;
		sl(x);
		pl(binarysearch(a,x,u,l));
		br;
	}
}
