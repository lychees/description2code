#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main(){
	int n,t;
	vector<int> a;
	
	cin>>n;
	for(int i=0; i<n; i++){
		cin>>t;
		a.push_back(t);
	}
	sort(a.begin(), a.begin() + n);
	
	for(int i=0; i<a.size(); i++,n--){
		cout<<n<<endl;
		
		while(a[i]==a[i+1] && i<a.size()){
			n--;
			i++;
		}
		
	}
	return 0;	
}
