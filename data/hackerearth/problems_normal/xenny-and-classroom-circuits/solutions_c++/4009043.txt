#include <iostream>

using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        int n;
        cin >> n;
        n = 2 * n;
        string str;
        cin >> str;
        int res = -1;

        // line starts with B
        int sol = 0;
        for (int i = 0; i < n; i++)
        {
            if (i % 2 == 0)
            {
                if (str[i] != 'B')
                {
                    sol++;
                }
            }
        }
        res = sol;


        // line starts with G
        sol = 0;
        for (int i = 0; i < n; i++)
        {
            if (i % 2 == 1)
            {
                if (str[i] != 'B')
                {
                    sol++;
                }
            }
        }

        if (sol < res)
            res = sol;
        cout << res << endl;

    }


    return 0;
}
