from collections import Counter
 
for i in range(int(raw_input())):
	s = raw_input()
	count = Counter(s)
	if len(count)!=26:
		print "No"
	else:
		val = count.values()
		if len(set(val)) == 1:
			print "Yes"
		else:
			print "No"
