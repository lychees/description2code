#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,i,j,n,x,k,s,maxx,minn;
    cin>>t;
    while(t--)
    {maxx=0;
    minn=1000000000;
    	cin>>n>>k;
    	int a[n];
    	for(i=0;i<n;i++)
    	cin>>a[i];
    
    	sort(a,a+n);
    	x=n/2;
    	j=n-1;
    	for(i=0;i<x;i++)
    	{
    		s=a[i]+a[j];
    		j--;
    		if(s>maxx)
    		maxx=s;
    		if(s<minn)
    		minn=s;
    	}
    	x=maxx-minn;
    	cout<<x<<"\n";

    	 else

    }

}
