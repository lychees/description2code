#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,a,sum=0,i;
    	cin>>n;
    	for(i=0;i<n;i++)
    	{cin>>a;
    	 sum+=a;
    	}
    	if(sum%n==0)
    		cout<<"YES\n"<<sum/n<<' '<<sum%n<<'\n';
    	else
    		cout<<"NO\n"<<sum/n<<' '<<sum%n<<'\n';
    	
    }
    return 0;
}
