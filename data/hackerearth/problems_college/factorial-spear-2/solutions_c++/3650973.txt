#include<stdio.h>
int prm[10000001];
int pm[10000001];
int prime()
{
	long long int i,j;
	for(i=2;i<=10000000;i++)
	{
		if(prm[i])
		continue;
		for(j=i;j*i<=10000000;j++)
		if(!prm[j*i])
		prm[j*i]=i;
	}
	return 0;
}
int main()
{
	prime();
	pm[2]=1;
	for(int i=3;i<=10000000;i++)
	{
		if(!prm[i])
			pm[i]=pm[i-1]+1;
		else
		{
			int count=0;
			int tem=i;
			while(tem&&prm[tem])
			{
				tem=tem/prm[tem];
				count++;
			}
			if(tem)
			count++;
			pm[i]=pm[i-1]+count;
		}
	}
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int a;
		scanf("%d",&a);
		printf("%d\n",pm[a]);
	}
	return 0;
}
