#include <iostream>
    using namespace std;
    long long int dp[10005];
    void solve(long long int a[],int n,int k)
    {
    	long long int ans=0;
    	dp[n-1]=a[n-1];
    	for(int i=n-2;i>=0;i--)
    	{
    		dp[i]=a[i];
			if(i+k+1<n && dp[i+k+1]>0)dp[i]+=dp[i+k+1];
    		dp[i]=max(dp[i+1],dp[i]);
    	}
    	if(dp[0]<0)cout<<0<<endl;
    	else cout<<dp[0]<<endl;
    }
    int main()
    {
        int t;
        cin>>t;
        long long int a[10005];
        while(t>0)
        {
        	t--;
        	int n,k;
        	cin>>n>>k;
        	for(int i=0;i<n;i++)cin>>a[i];
        	solve(a,n,k);
        }
        return 0;
    }
