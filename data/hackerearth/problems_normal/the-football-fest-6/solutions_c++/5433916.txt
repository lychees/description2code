#include "bits/stdc++.h"
using namespace std;
typedef long long ll;
int main()
{
    //cout << "Hello World!" << endl;
    ll t,n,id,p,temp;
    char c;
    cin>>t;
   // cout<<t<<" ";
    while(t--)
    {
    	stack<ll> s;
    	cin>>n>>id;
    	s.push(id);
    	while(n--)
    	{
    	   cin>>c;
    	   if(c=='P')
    	   {
    	   	cin>>p;
    	   	s.push(p);
    	   }
    	   else
    	   {
    	   	 temp=s.top();
    	   	 s.pop();
    	   	 ll temp1=s.top();
    	   	 s.pop();
    	   	 s.push(temp);
    	   	 s.push(temp1);
    	   }
    	}
    	cout<<"Player "<<s.top()<<"\n";
    }
    return 0;
}
