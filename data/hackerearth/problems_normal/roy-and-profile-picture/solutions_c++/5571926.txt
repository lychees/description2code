#include <iostream>
using namespace std;

int main()
{
    int l,w,h,n,i;
    cin>>l;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>w;
    	cin>>h;
    	if(w<l || h<l)
    	cout<<"UPLOAD ANOTHER\n";
    	else
    	{
    	if(w==h)
    	cout<<"ACCEPTED\n";
    	else
       	cout<<"CROP IT\n";
    	}
    }
    return 0;
}
