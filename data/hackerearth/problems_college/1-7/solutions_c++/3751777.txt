#include <iostream>
#include <map>
#include <cstring>
using namespace std;
int Frequency(string str, char ch)
{
	int l = str.length();
	int c = 0;
	for(int i=0; i<l; i++)
		if(str[i] == ch)
			c++;
	return c;
}
int main()
{
	int T;
	cin >> T;
	while(T--){
		string str;
		cin >> str;
		char ch;
		cin >> ch;
		cout << Frequency(str, ch) << endl;
	}
	return 0;
}
