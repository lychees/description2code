N, Q = [int(_) for _ in raw_input().split()]
maze = raw_input()
for i in range(Q):
	s, e = [int(_) for _ in raw_input().split()]
	s, e = max(s, e), min(s, e)
	if maze[s-1] == '1' or maze[e-1] == '1':
		print 'No'
	else:
		ans = True
		while s != e:
			if s < e:
				s, e = e, s
			else:
				s /= 2
				if maze[s-1] == '1':
					ans = False
					break
		print 'Yes' if ans else 'No'
