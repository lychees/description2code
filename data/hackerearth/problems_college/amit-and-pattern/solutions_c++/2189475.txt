#include<bits/stdc++.h>
using namespace std;
# define ll long long
int main()
{
    ll n,m,i,inp,t,j,ans[222][222],pos[222];
    cin>>n>>m;
    for(i=1;i<=222;i++)
    pos[i]=0;
    for(i=1;i<=n;i++)
        for(j=1;j<=m;j++)
        {
            cin>>inp;
            pos[i+j-1]++;
            ans[i+j-1][pos[i+j-1]]=inp;
        }
    for(i=1;i<=n+m-1;i++)
    {
        for(j=1;j<=pos[i];j++)
            cout<<ans[i][j]<<" ";
        cout<<endl;
    }
    return 0;
}
