#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll dp[203][203];
ll ways(int n,int k)
{
	if(k>n)
		return 0;
	if(dp[n][k]>=0)return dp[n][k];
	ll ans=1;
	for(int i=k;i<=n/2;i++)
	{
		ans+=ways(n-i,i);
	}
	return dp[n][k]=ans;
}
int main()
{
	int t;	
	scanf("%d",&t);
	for(int i=0;i<201;i++)
 	       for(int j=0;j<201;j++)
        	       dp[i][j]=-1;
	while(t--)
	{
		int n,k;
		scanf("%d%d",&n,&k);
		printf("%lld\n",ways(n,k));
	}
	return 0;
}
