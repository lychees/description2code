void generateFlipped(string& s, string& sr)
{
	vector<char> vs(s.begin(), s.end());
	list<char> vsl, vsr;
	       
	int count = vs.size();
    int lim = count / 2;
    
    for(int i=0, j=count - 1; i < j ; i++, j--)
   	{
   		vsl.push_back(vs[j]);
   		vsr.push_front(vs[i]);
   	}
   	
    if(count & 1)
          	vsl.push_back(vs[lim]);
    
	for(auto it = vsl.begin(); it != vsl.end(); ++it)
		sr+=*it;
	for(auto it = vsr.begin(); it != vsr.end(); ++it)
    	sr+=*it;
    	
    return;
}
