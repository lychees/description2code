#include<iostream>
#include<vector>
#include<cstring>
using namespace std;

int main(){
	
	int n;
	cin>>n;
	vector<string>out;
	string str,temp="";
	
	for(int i=0;i<n;i++){		
		cin>>str;
		
		for(int j=0;j<str.size();j++){
if(str[j]=='A' ||str[j]=='E'||str[j]=='I'||str[j]=='O'||str[j]=='U' ||str[j]=='a' ||str[j]=='e'||str[j]=='i'||str[j]=='o'||str[j]=='u')
		temp+=str[j];
}

	if(temp==""){
		out.push_back("No");
	}
	else{
	out.push_back(temp);
	temp="";
}
}

for(string i:out)
cout<<i<<endl;
}
