#include <iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main()
{
 long int n,k,i,j,count=0,num,max=0;
 vector<int>v;
 cin>>k;
 long int a[k];
 for(i=0;i<k;i++)
 {
 	cin>>a[i];
 	v.push_back(a[i]);
 }
sort(v.begin(),v.end());
reverse(v.begin(),v.end());
max=0;
for(int i = 0; i < k ; i++)
        {
		
		count++;
		num = v[i] + count + 1;
		if(num > max)
		{
			max = num;
		}
       }
  cout<<max<<endl;
        
  return 0;
}
