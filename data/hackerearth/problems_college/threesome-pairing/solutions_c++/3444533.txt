#include <iostream>
using namespace std;

int main()
{
	long int t,i,j,count,a[1000000],n;
    cin>>t;
    for(int k=0;k<t;k++)
    {
    	count=0;
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	for(i=0;i<n-1;i++)
    	{
    		for(j=i+1;j<n;j++)
    		{
    			if((a[i]+a[j])%3==0)
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
