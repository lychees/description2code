#include<iostream>
#include<stdio.h>

using namespace std;
int main()
{
char s[6];
double h,m,hr,mi,k;
cin>>s;
h=(s[0]-48)*10+(s[1]-48);
m=10*(s[3]-48)+(s[4]-48);
hr=(12-h)*30;
mi=(60-m)*6;
hr=hr-m/2;
k=hr-mi;
if((k)<0)
    k=-1*(k);
    if((k)>180)
        printf("%6f",360-k);
else
printf("%6f",k);
return 0;
}
