#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int r,xc,yc;
		cin>>r>>xc>>yc;
		int x[4],y[4];
		for(int i=0;i<4;i++)
		cin>>x[i]>>y[i];
		int flag=0;
		for(int i=0;i<4;i++)
		{
			int ans1=pow(x[i]-xc,2)+pow(y[i]-yc,2);
			int ans2=r*r;
			if(ans1<=ans2)
			flag=1;
			else
			{
				flag=0;
				break;
			}
		}
		if(flag==1)
		cout<<"Yes"<<endl;
		else
		cout<<"No"<<endl;
	}
	return 0;
}
