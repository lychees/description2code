#include <bits/stdc++.h>
#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int n;
    scanf("%d", &n);
    int p1 = 0, p2 = n - 1;
    int sum1 = 0, sum2 = 0;
    int a[n][n];
    for(int i = 0; i < n; i++) {
    	for(int j = 0; j < n; j++) {
    		scanf("%d", &a[i][j]);
    	}
    	sum1 += a[i][p1];
    	sum2 += a[i][p2];
    	p1++;
    	p2--;
    }
    int ans = abs(sum1 - sum2);
    printf("%d", ans);
    return 0;
}
