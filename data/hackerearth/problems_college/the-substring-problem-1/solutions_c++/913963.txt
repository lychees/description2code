#include <iostream>
#include <cstdio>
using namespace std;

int arr[26];

int main() {
    int t,n,i,max,index;
    char str[102],ans;
    scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		while(n--)
		{
			scanf("%s",str);
			bool map[26];
            for(i=0;i<26;i++)
            map[i]=false;
			for(i=0;str[i];i++)
			{
                index=str[i]-'a';
                if(!map[index])
                {
                	map[index]=true;
                	arr[index]++;
                }
			}
		}
		max=0;
		for(i=0;i<26;i++)
		{
			if(arr[i]>max)
			{
                max=arr[i];
				ans=i+'a';
			}
			arr[i]=0;
		}
        
		printf("%c\n",ans);
	}
	return 0;
}
