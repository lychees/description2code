#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

string str;

const int max_n = 1e5 + 10;

ll cnt[max_n][26];

int main()
{
	ll t, i, j, n, q, odd, diff, l, r;

	bool flag;

	scanf("%lld", &n);

	cin>>str;

	for(i=0;i<n;++i)
	{
		if(i>0)
		{
			for(j=0;j<26;++j)
				cnt[i][j] = cnt[i-1][j];
		}

		++cnt[i][str[i] - 'a'];
	}

	scanf("%lld", &q);

	while(q--)
	{
		flag = true;
		odd = 0;

		scanf("%lld %lld", &l, &r);
		--r, --l;

		if((r-l+1)<=1)
			flag = false;
		else
		{
			for(j=0;j<26;++j)
			{
				diff = cnt[r][j] - (l==0?0:cnt[l-1][j]);
				if(diff&1)
					++odd;
			}

			if(odd>2)
			flag = false;
		}

		if(flag)
		{
			printf("YES\n");
		}
		else
			printf("NO\n");
	}

	return 0;
}
