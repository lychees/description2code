#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int main()
{
	long long t;
	cin>>t;
	while(t--){
		long long r,c;
		cin>>r>>c;
		long long sume=0,sumo=0;
		if(c>=r){
			if(r%2==0){
				long long n=r/2;
				sume+=(n)*(n+1);
				sumo+=(n)*(n);
			}
			else if(r%2!=0){
				long long n=r/2;
				
				sume+=((r/2)*(1+r))/2;;
				n++;
				sumo+=(((r+1)/2)*(1+r))/2;
			}
		}
		else if(c<r){
			if(c%2)
			{
				sumo=(((c+1)/2)*(1+c))/2;
				sume=((c/2)*(1+c))/2;
				
				if((r-c)%2)
				{
					sume+=(((r-c+1)/2)*c);
					sumo+=(((r-c)/2)*c);
				}
				else
				{
					sume+=(((r-c)/2)*c);
					sumo+=(((r-c)/2)*c);
				}
			}
			else
			{
				sumo=((c/2)*(c))/2;
				sume=((c/2)*(2+c))/2;
				
				if((r-c)%2)
				{
					sume+=(((r-c)/2)*c);
					sumo+=(((r-c+1)/2)*c);
				}
				else
				{
					sume+=(((r-c)/2)*c);
					sumo+=(((r-c)/2)*c);
				}
			}
		}
		cout<<sume<<" "<<sumo<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
