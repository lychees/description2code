#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int ans = 0,i=0;
    	if(n==1||n==2)
    		i=1;
    	else
    	{
    		while(ans<=n)
    		{
    			i++;
    			ans = i * (i+1) / 2;
    		}
    		i--;
    	}
    	cout<<i<<endl;
    }
    return 0;
}
