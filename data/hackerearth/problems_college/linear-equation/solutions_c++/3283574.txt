#include <iostream>
using namespace std;
int main(){
    long int t;
    cin>>t;
    long int n,i;
    while(t--){
        cin>>n;
        long long int a[n],sum=0;
        for(i=0;i<n;i++){
            cin>>a[i];
            sum+=a[i];
        }
        sum=sum/(n-1);
        long long int x[n];
        for(i=0;i<n;i++){
            x[i]=sum-a[i];
            cout<<x[i]<<" ";
        }
        cout<<endl;
    }
    return 0;
}
