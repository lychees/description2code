#include<iostream>
using namespace std;

int main()
{
    int a[100][100],s1=0,s2=0,diff,i,j,n;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	for(j=0;j<n;j++)
    	cin>>a[i][j];
    }
    for(i=0;i<n;i++)
    {
    	for(j=0;j<n;j++)
    	{
    	if(i==j)
    	s1+=a[i][j];
    	}
    	
    }
    for(i=0,j=n-1;i<n,j>=0;i++,j--)
    {
    s2+=a[i][j];
    }
     diff=s1-s2;
     if(diff>0)
     cout<<diff;
     else
     cout<<-(s1-s2);
     return 0;
}
