#include <iostream>
#include<stdio.h>
#include <algorithm>
using namespace std;

int main()
{
   int t,n;
   long int result,mod = 1000000007;
   int a[1001];
   long int ans;
   cin>>t;
   
   while(t--) {
   	cin>>n;
   	ans = 1,result =0;
   	for(int i=0;i<n;i++) {
   	  cin>>a[i];
   	}
   	
   	std:sort(a,a+n);
   	int count = 0,val;
   	for(int j=0;j<n;j++) {
   	  val = a[j] - count;
   	  if (val <=0) {
   	  	result = -1;
   	  	break;
   	  } else {
   	  	ans = ((ans%mod) * (val%mod))%mod;
   	  	count++;
   	  }  
   	   	
   	}
   	
   	if (result != 0) {
   		cout << 0<<"\n";
   	} else {
   	cout << ans%mod<<"\n";
   	
   	
   	
   }
   
   
   
   
}

}
