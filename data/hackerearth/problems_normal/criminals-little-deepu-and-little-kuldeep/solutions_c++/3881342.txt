#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin>>t;
    while(t--) {
    	int n;
    	cin>>n;
    	map<int,int> m;
    	vector<int> v(n);
    	int val = 0;
    	for(int i=0;i<n;i++) {
    		cin>>v[i];
    		m[v[i]] += 1;
    		val = max(val,m[v[i]]);
    	}
    	cout<<val<<endl;
    }
    return 0;
}
