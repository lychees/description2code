#include<iostream>
#include<bits/stdc++.h>
using namespace std;
long long int count_ways(int n)
{
    long long int dp[1005];
    memset(dp,0,sizeof(dp));
    dp[0]=0;
    dp[1]=1;
    dp[2]=2;
    for(int i=3;i<=n;i++)
    {
    	dp[i]=dp[i-1]+dp[i-2];
    }
    return dp[n];

}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        cout<<count_ways(n)<<endl;
        //cout<<n<<endl;
    }
    return 0;
}
