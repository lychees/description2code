#include <bits/stdc++.h>
#define ll long long
using namespace std;
int h[26],l;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int test;
	cin>>test;
	while(test--)
	{
		int n;
		string s;
		cin>>n>>s;
		memset(h,0,sizeof(h));
		l=s.length();
		int i,j,mc;
		for(i=0;i<l;i++)
		{
			h[s.at(i)-'a']++;
		}
		int ans=0,lans=0,sc=INT_MIN;
		for(i=0;i<n;i++)
		{
			mc=0;
			cin>>s;
			l=s.length();
			for(j=0;j<l;j++)
			mc+=h[s.at(j)-'a'];
			if(mc>sc)
			{
				sc=mc;
				ans=i+1;
				lans=l;
			}
			else if(mc==sc && l<lans)
			{
				ans=i+1;
				lans=l;
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
