def gcd(a, b):
    if b == 0:
        return a
    return gcd(b, a % b)


T = int(raw_input())
for t in range(T):
    N = int(raw_input())
    timeint = [int(i) for i in raw_input().split()]
    i = 1
    lcm = timeint[0]
    while i < N:
        lcm = (lcm*timeint[i])/gcd(lcm, timeint[i])
        i = i+1
    print lcm % 1000000007
