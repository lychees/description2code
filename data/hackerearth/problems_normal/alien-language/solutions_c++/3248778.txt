#include <iostream>
#include <cstring>
using namespace std;

int main()
{	int t;
	cin>>t;
	while(t--)
	{	string s1;
		string s2;
		cin>>s1;
		cin>>s2;
		int a[26]={0};
		int b[26]={0};
		for(int i=0; s1[i]!='\0'; i++)
		{
			a[s1[i]-'a']=1;
		}
		for(int i=0; s2[i]!='\0'; i++)
		{
			b[s2[i]-'a']=1;
		}
		int f=0;
		for(int i=0; i<26; i++)
		{
			if(a[i] && b[i])
			{
				f=1;
			}
		}
		if(f)	
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
    return 0;
}
