#generating a list of fibbonacci numbers
fib=[1,1,2]
for i in xrange(3,1001):
    fib.append(fib[i-1]+fib[i-2])
    
    
#print fib[2]
#precomputing all the values and storing them in memo


memo=[0,1,4]

for i in xrange(2,1001):
    ans=0
    for j in xrange(i+1):
        if(fib[i]%fib[j]==0):
        
            ans+=1
    memo.append(memo[i]+ans)
    
    
    

#directly printing answer corresponding to query using precomputed values
for _ in xrange(input()):
    x=int(input())
    print memo[x]
