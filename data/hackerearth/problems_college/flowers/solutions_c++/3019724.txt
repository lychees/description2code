#include<bits/stdc++.h>
using namespace std;
# define ll long long
vector <int> v;
int main()
{
    int n,k,x,turn,pass;
    ll ans;
    ans=0;
    cin>>n>>k;
    pass=k;
    turn=1;
    while(n--)
    {
        cin>>x;
        v.push_back(x);
    }
    vector <int>::iterator it;
    sort(v.begin(),v.end());
    for(it=v.end()-1;it!=v.begin()-1;--it)
    {
        if(!pass)
            pass=k,turn++;
        ans+=turn*(*it);
        pass--;
    }
    cout<<ans;
}
