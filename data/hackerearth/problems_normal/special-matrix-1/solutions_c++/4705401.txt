#include <iostream>
using namespace std;

int main()
{
	int t,i,n,j,k,m,r,row,column,v,frow,fcolumn,count;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>n;
    	char a[n][n];
    	for(j=0;j<n;j++)
    	{
    		for(k=0;k<n;k++)
    		{
    			cin>>a[j][k];
    		}
    	}
    	for(m=0;m<n;m++)
    	{
    		for(r=0;r<n;r++)
    		{
    			if(a[m][r]=='*')
    			{
    				row=m;
    				column=r;
    			}
    		}
    	}
    
    	v=n/2;
    	frow=abs(v-row);
    	fcolumn=abs(v-column);
    	count=frow+fcolumn;
    	cout<<count;
    	cout<<endl;
    	
    }
    return 0;
}
