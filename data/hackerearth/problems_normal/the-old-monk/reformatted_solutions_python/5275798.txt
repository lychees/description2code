import bisect
t = int(raw_input())


while t > 0:
    t -= 1
    ans = 0
    n = int(raw_input())
    a = map(int, raw_input().split())
    b = map(int, raw_input().split())
    b = b[::-1]
    for i in range(n):
        ans = max(n-1 - bisect.bisect_left(b, a[i]) - i, ans)
    print ans
