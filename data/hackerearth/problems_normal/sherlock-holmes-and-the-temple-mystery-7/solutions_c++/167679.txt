#include <iostream>
#include<cstdlib>
using namespace std;
bool findpart(int *a,int n,int sum){
	bool p[sum/2+1][n+1];
	for(int i=0;i<n+1;i++)p[0][i]=true;
	for(int x=0;x<sum/2+1;x++)p[x][0]=false;
	for(int i=1;i<=sum/2;i++)
		for(int j=1;j<=n;j++)
			p[i][j]=(p[i][j-1] || p[i-a[j-1]][j-1]);
	return p[sum/2][n];
}

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	int a[n],sum=0;
    	for(int i=0;i<n;i++){
    		cin>>a[i];
    		sum+=a[i];
    	}
    	if(sum%2!=0){
    		cout<<"NO\n";
    		continue;
    	}
    	else{
    		if(findpart(a,n,sum)){
    			cout<<"YES\n";		
    		}
    		else cout<<"NO\n";
    	}
    }
    return 0;
}
