m1, m2 = list(map(int, raw_input().split()))

matrix_lt = [list(map(int, raw_input().split())) for line in range(m1)]

row_lt = []
col_lt = []

for lt in matrix_lt:
    sum = 0
    for i in lt:
        sum = sum + i
    row_lt.append(sum)

for lt in range(len(matrix_lt[0])):
    sum = 0
    for j in matrix_lt:
        sum = sum + j[lt]
    col_lt.append(sum)

print max([i ^ j for i in row_lt for j in col_lt])
