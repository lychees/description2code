#include <bits/stdc++.h>
using namespace std;
int a[20000006];
int main()
{
    //printf("Hello World!\n");
    int t,n;
    scanf("%d",&t);
    a[1] = 0;
    for(n=2;n<=20000000;n++)
    {
    	int mi = a[n-1];
    	if(n%2==0)mi = min(mi,a[n/2]);
    	if(n%3==0)mi = min(mi,a[n/3]);
    	a[n] = 1+mi;
    }
    while(t--)
    {
    	scanf("%d",&n);
    
    	printf("%d\n",a[n]);
    }
    return 0;
}
