#include <bits/stdc++.h>
using namespace std;

int sieve[100005], phi[100005];

int gcd(int a, int b)
{
	if(b==0)
		return a;
	
	return gcd(b, a%b);
}

int main()
{
    sieve[1] = 1;
    
    for(int i=1; i<=100000; i++)
    	phi[i] = i;
    
    for(int i=2; i<=100000; i++)
    {
    	if(sieve[i]==0)
    	{
    		for(int j=1; i*j<=100000; j++)
    		{
    			sieve[i*j] = 1;
    			phi[i*j] = phi[i*j]/i*(i-1);
    		}
    	}
    }
    
    int T;
    scanf("%d", &T);
    
    while(T--)
    {
    	int A, B;
    	scanf("%d %d", &A, &B);
    	
    	long long ans = 0;
    	
    	int x = gcd(A, B);
    	
    	for(int i=1; i<=sqrt(x); i++)
    	{
    		if(x%i==0)
    		{
    			ans+= phi[i];
    			
    			if(i*i!=x)
    				ans+= phi[x/i];
    		}
    	}
    	
    	printf("%lld\n", ans);
    }
    
    return 0;
}
