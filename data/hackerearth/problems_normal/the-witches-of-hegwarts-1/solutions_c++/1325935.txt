#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <set>
#include <utility>

using namespace std;

vector<int> v;
set<int> s;

int main() {
    int t;
    scanf("%d",&t);

    while(t--) {
        int n;
        scanf("%d",&n);
        bool com = false;
        v.clear();
        s.clear();
        if(n==1)
            printf("0\n");
        else {
            int i=0, j=1;
            v.push_back(n);
            int ans = 0;
            while(true) {
                if(com)
                    break;
                ++ans;
                for(int k=i;k<j;k++) {
                    int x = -1, y = -1;
                    if(!(v[k]%3)){
                        x = v[k]/3;
                        if(x==1)
                            com = true;
                        else {
                            if(s.find(x)==s.end()) {
                                v.push_back(x);
                                s.insert(x);
                            }
                        }
                    }
                    if(!(v[k]&0x01)){
                        y = v[k]/2;
                        if(y==1)
                            com = true;
                        else if(s.find(y)==s.end()) {
                                v.push_back(y);
                                s.insert(y);
                        }
                    }
                    int z = v[k]-1;
                    if(z==1)
                        com = true;
                    else if(s.find(z)==s.end()) {
                        v.push_back(z);
                        s.insert(z);
                    }

                    if(com)
                        break;
                }

                i = j;
                j = v.size();
            }

            printf("%d\n",ans);
        }

    }

    return 0;
}
