#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int main()
{
    int t,n,nb;
    vector<long int> v;
    long int p,s;
    cin>>t;
    while(t--)
    {
        cin>>n>>s;
        nb=n;
        while(n--)
        {
            cin>>p;
            v.push_back(p);
        }
        sort(v.begin(),v.end());
        if(count(v.begin(),v.end(),s)==0)
            cout<<"NO 0\n";
        else
        {
            for(int i=0;i<v.size();++i)
            {
                if(v.at(i)==s)
                {   cout<<"YES"<<" "<<nb-(i+1)<<"\n";
                    break;
                }
            }

        }
        v.clear();
    }
}
