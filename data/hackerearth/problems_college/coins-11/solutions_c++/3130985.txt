using namespace std;

bool isPrime(int n){
	if(n==2)
	return true;
	if(n%2==0 || n<2)
	return false;
	for(int i=3;i*i<=n;i+=2){
		if(n%i==0)
		return false;
	}
	return true;
}

int main(){
	int t;
	S(t);
	while(t--){
		int n;
		S(n);
		if(n==3)
		printf("3\n");
		else if(isPrime(n))
		printf("2\n");
		else
		printf("1\n");
	}
	return 0;
}
