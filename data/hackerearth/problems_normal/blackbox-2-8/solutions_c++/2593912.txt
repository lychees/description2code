#include<bits/stdc++.h>
using namespace std;
long long int s[1001];
long long fact(long long n,long long N[],long long v)
{
        long long k,j,t,u;
        t=0;
        for(j=0;j<v;j++)
			{
				u=N[j]*n+t;
				N[j]=u%10;
				t=u/10;
			}
			while(t>0)
			{
				N[v]=t%10;
				t=t/10;
				v++;
			}
        return v;
}
long long f(long long n,long long N[],long long v)
{
        v=fact(n,N,v);
        long long sum=0,p;
		for(p=v-1;p>=0;p--)
		sum+=N[p];
		s[n]=sum;
		return v;
 
}
 
int main()
{
	long long i,t,number,N[3000],v;
	 N[0]=1; s[0]=1;	 v=1;
	for(i=1;i<=1000;i++)
    v=f(i,N,v);
 
	cin>>t;
	while(t--)
    {
    	cin>>number;
    	cout<<s[number]<<endl;
    }
    return 0;
}
