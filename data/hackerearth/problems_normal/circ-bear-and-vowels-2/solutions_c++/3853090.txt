#include <bits/stdc++.h>

using namespace std;

int main()
{
    int T;
    map<char,int> countbag;
	string s;
	cin>>T;
	while(T--)
	{
		countbag['v']=0;
		countbag['c']=0;
		cin>>s;
		int flag=0;
		int count=0;
		for(int i=0;i<s.length();i++)
		{
			if(s[i]=='a' || s[i]=='e' || s[i]=='i' || s[i]=='o' || s[i]=='u' ||s[i]=='y')
				{
					countbag['v']++;
					count=0;
				}
			else
				{
					countbag['c']++;
					count++;
				}
			if(count == 3)
			{
				flag=1;
				break;
			}
			
		}
		if(countbag['c']>countbag['v'])
			flag=1;
		
		if(flag)
			cout<<"hard"<<"\n";
		else
			cout<<"easy"<<"\n";
	}
    return 0;
}
