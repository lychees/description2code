int main()
{
    int flag=0,n;

    while(n--)
    {
    	if(flag==0)
    	{

    		flag=1;
    	}
    	string str1,str2,str3="";

		sort(str2.begin(),str2.end());
		int l1=str1.length();
		int l2=str2.length();
		int j=0;
		for(int i=0;i<l1;i++)
		{
			if(j<l2)
			{
				int a = str1[i]-'a';
				int b = str2[j]-'a';
				//cout<<str1[i]<<" "<<a<<" "<<str2[j]<<" "<<b<<endl;
				if(a<=b)
				{
					str3+=str1[i];
				}

					str3+=str2[j];

				}
			}
			else
			{
				str3+=str1[i];
			}
		}
		cout<<str3<<endl;
    }

}
