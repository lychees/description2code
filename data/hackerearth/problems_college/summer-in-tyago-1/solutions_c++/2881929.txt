#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define MOD 1000000007LL
#define MAXN 100010
ll A[MAXN];

int main()
{
	int n;
	cin >> n;
	for(int i=0;i<n;++i)	
		cin >> A[i];
	reverse(A,A+n);
	for(int i=1;i<n;++i)
	{
		int j = i;
		while(j>0 and A[j]>A[j-1])
		{
			--A[j];
			++A[j-1];
			swap(A[j],A[j-1]);
			--j;
		}
	}
	reverse(A,A+n);
	bool flag = true;
	for(int i=0;i<n-1;++i)
		if(A[i]>A[i+1]) 
			flag = false;
	if(flag)
	for(int i=0;i<n;++i)
		cout << A[i] << " ";
	else
		cout << ":(";
	return 0;
}
