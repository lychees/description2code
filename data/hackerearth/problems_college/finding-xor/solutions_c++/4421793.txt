#include<bits/stdc++.h>
using namespace std;
char  s1[100*100+9],s2[100*100+9];
int main(){
    int tc;scanf("%d",&tc);
    while(tc--){
        scanf("%s",s1);
        scanf("%s",s2);
        int len=strlen(s1);
        for(int i=0;i<len;++i)
        {
            s1[i]=((s1[i]-'0')^(s2[i]-'0'))+'0';
        }
        printf("%s \n",s1);
    }
    return 0;
}
