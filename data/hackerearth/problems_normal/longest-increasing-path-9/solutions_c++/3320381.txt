#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;cin>>t;
	while(t--){
		int n,m;cin>>n>>m;
		int A[n][m],dp[n][m];
		for(int i=0;i<n;i++){
			for(int j=0;j<m;j++){
				cin>>A[i][j];
			}
		}
		memset(dp,0,sizeof(dp));
		
		dp[0][0]=1;int ans=0;
		for(int i=0;i<n;i++){
			for(int j=0;j<m;j++){
				if(i==0&&j==0){
					ans=max(ans,dp[i][j]);
					continue;
				}
				if(i>0&&dp[i-1][j]&&A[i-1][j]<A[i][j])dp[i][j]=max(dp[i][j],1+dp[i-1][j]);
				if(j>0&&dp[i][j-1]&&A[i][j-1]<A[i][j])dp[i][j]=max(dp[i][j],1+dp[i][j-1]);
				ans=max(ans,dp[i][j]);
			}
		}
		cout<<ans<<endl;
	}
    return 0;
}
