#include<iostream>
#include<vector>
#include<bits/stdc++.h>
using namespace std;
long long a,n,k;
vector<long long > g[1000000];long long dp[1000000];
void find(long long v,long long &ans)
{
	long long ct=1;
	for(int i=0;i<g[v].size();i++)
	{
		find(g[v][i],ans);
		ct+=dp[g[v][i]];
	}
	if(ct>=k)
	{
		dp[v]=0;
		ans+=1;
	}
	else
		dp[v]=ct;
	
}
main()
{
	
	cin>>k>>n;
	memset(dp,0,sizeof dp);
	for(int i=2;i<=n;i++)
	{
		cin>>a;
		g[a].push_back(i);
	}
	long long ans=0;
	find(1,ans);
	cout<<ans<<endl;
	

}
