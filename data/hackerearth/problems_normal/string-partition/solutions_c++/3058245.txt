#include <bits/stdc++.h>
using namespace std;
typedef long long int LL;
LL dp[110][110];
string s;
int l;
LL z;
LL fun(int idx,int k)
{
	if(idx==l)
	{
		if(k==0)
		return 0;
		else 
		return -1e18;
	}
	if(dp[idx][k]!=-1)
	return dp[idx][k];
	LL ret1=0;
	LL ret=-1e18;
	for(int i=idx;i<l;i++)
	{
		ret1=ret1*10+s[i]-'0';
		if(ret1>z)
		break;
		ret=max(ret,ret1+fun(i+1,k-1));
	}
	dp[idx][k]=ret;
	return dp[idx][k];
}
int main()
{
	int t,k;
	scanf("%d",&t);
	while(t--)
	{
		memset(dp,-1,sizeof dp);
	scanf("%d%lld",&k,&z);
	cin>>s;
	l=s.length();
	LL ans=fun(0,k);
	if(ans<0)
	printf("-1\n");
	else
	printf("%lld\n",ans);
	}
	return 0;
}
