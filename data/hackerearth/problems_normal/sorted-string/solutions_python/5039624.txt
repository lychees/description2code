n = int(raw_input().strip())
for i in range(0, n):
        str1 = raw_input().strip()
        L = []
        for index, each in enumerate(str1):
                count = str1.count(each)
                L.append((ord(each), count))
        L.sort(key = lambda x:x[::-1])
        print ''.join([chr(each[0]) for each in L])
