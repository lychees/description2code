def primes(n):
    primfac = []
    d = 2
    while d*d <= n:
        while (n % d) == 0:
            primfac.append(d)  # supposing you want multiple factors repeated
            n //= d
        d += 1
    if n > 1:
        primfac.append(n)
    return primfac


no_of_test_cases = int(raw_input())

y = 1


while y <= no_of_test_cases:

    prime_factors = primes(int(raw_input()))

    dictonary = {}

    for n in prime_factors:
        if n in dictonary:
            dictonary[n] += 1
        else:
            dictonary[n] = 1

    ans = 1
    for x in dictonary.values():
        ans = ans * (x + 1)

    print ans - 1

    y += 1
