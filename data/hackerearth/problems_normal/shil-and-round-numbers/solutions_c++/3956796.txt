ll i,n,q,num,fi,la,a[200021],ro[200021],choice,l,r;
void update(ll pos,ll val)
{
    for(;pos<=n;pos+=pos & (-pos)) ro[pos]+=val;
}
ll query(ll pos)
{
    ll sum=0;
    for(;pos>0;pos-=pos & (-pos)) sum+=ro[pos];
    ret sum;
}
int main()
{
    sc("%lld%lld",&n,&q);
    for(i=1;i<=n;i++)
    {
        sc("%lld",&a[i]);
        if(a[i]<=0) ct;
        num=a[i];
        fi=la=num%10;
        whi(num)
        {
            fi=num%10;
            num/=10;
        }
        if(fi==la) update(i,1);
    }
    whi(q--)
    {
        sc("%lld",&choice);
        if(choice==1)
        {
            sc("%lld%lld",&l,&r);
            pr("%lld\n",query(r)-query(l-1));
        }
        else
        {
            sc("%lld%lld",&l,&r);
            if(a[l]>0)
            {
                num=a[l];
                fi=la=num%10;
                whi(num)
                {
                    fi=num%10;
                    num/=10;
                }
                if(fi==la) update(l,-1);
            }
            if(r>0)
            {
                num=r;
                fi=la=num%10;
                whi(num)
                {
                    fi=num%10;
                    num/=10;
                }
                if(fi==la) update(l,1);
            }
            a[l]=r;
        }
    }
    ret 0;
}
