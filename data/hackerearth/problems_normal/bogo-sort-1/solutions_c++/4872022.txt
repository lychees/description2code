#include<iostream>
#include<vector>
using namespace std;
int main(){
	int n;
	cin>>n;
	vector<int> arr(n);
	for(int i=0;i<n;i++){
		cin>>arr[i];
	}
	int mi = arr[0],mx = arr[0];
	for(int i=0;i<n;i++){
		if(arr[i]<mi){
			mi = arr[i];
		}
		if(arr[i]>mx){
			mx = arr[i];
		}
	}
	int k = mx - mi + 1;
	vector<int> aux(k);
	for(int i=0;i<k;i++){
		aux[i] = 0;
	}
	for(int i=0;i<n;i++){
		aux[arr[i] - mi]++;
	}
	for(int i=0;i<k;i++){
		if(aux[i] != 0){
			cout<<i+mi<<endl;
		}
	}
	
	
	return 0;
}
