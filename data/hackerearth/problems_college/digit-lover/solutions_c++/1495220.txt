#include<cstdio>
using namespace std;
int main()
{
	int num,sum,n,t;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		num=n;
	  while(1)
	  {
	  	sum=0;
	  	if(num%10==num)
	  	break;
	  	while(num!=0)
	  	{
	  		sum+=num%10;
	  		num/=10;
	  	}
	  	num=sum;
	  }	
	  printf("%d\n",num);
	}
	return 0;
}
