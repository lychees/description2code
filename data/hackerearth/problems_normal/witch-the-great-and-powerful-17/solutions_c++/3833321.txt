#include <iostream>
#include <cmath>
#include <cstdio>
#include <cstdlib>
using namespace std;

int main()
{
    int T;
    cin>>T;
    
    while(T--){
    	unsigned long long p,q,r,cnt;
    	cin>>p>>q>>r;
    	cnt=0;
    	if((q-p != 1) || (r-q != 1)){
    		if((q-p)>(r-q)){
    			cnt = q-p-1;
    		}else if((q-p)<(r-q)){
    			cnt = r-q-1;
    		}else{
    			cnt = r-q-1;
    		}
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
