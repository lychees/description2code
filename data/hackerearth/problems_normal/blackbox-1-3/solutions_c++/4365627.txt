
#include<iostream>
using namespace std;
bool coprime(size_t a, size_t b)
{
	while (b != 0)
	{
		size_t x = a%b;
		a = b;
		b = x;
	}
	if (a == 1)
		return 1;
	return 0;
}
int main()
{
	size_t n, a, b, c;
	cin >> n;
	for (size_t i = 0; i < n; i++)
	{
		cin >> a >> b >> c;
		if (a < b)
			swap(a, b);
		if (a < c)
			swap(a, c);
		if (b < c)
			swap(b, c);
		if (coprime(a, c) && coprime(b, c) && coprime(a, b))
		{
			if ((a*a) == (b*b + c*c))
				cout << "YES" << endl;
			else
				cout << "NO" << endl;
		}
		else
			cout << "NO" << endl;
	}
    return 0;
}
