using namespace std;

int main() {
	int n;
	cin >> n;

	set<int> s;
	vector<vector<int> > freq(n,vector<int>(501));
	for (int i = 0; i < n; i++) {
		int m;
		cin >> m;
		for (int j = 0; j < m; j++) {
			int c; cin >> c;
			s.insert(c);
			freq[i][c]++;
		}
	}
	LL a = accumulate(all(s), 0);

	for (int i = 0; i <= 500; i++) {
		int flag = 0;
		for (int j = 0; j < n; j++) {
			if (freq[j][i]==0) {
				flag = 1;
				break;
			}
		}
		if (flag == 0) {
			s.erase(s.find(i));
		}
	}
	LL b = accumulate(all(s), 0);

	cout << b - a << endl;
	return 0;
}
