#include <iostream>
using namespace std;

bool visited[20];
int love[20];
int decr[20];
int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;++i)
    {
    	cin>>love[i];
    }
    for(int i=0;i<n;++i)
    {
    	cin>>decr[i];
    }
    long long ans=0;
    for(int step=0;step<n;++step)
    {
    	for(int i=0;i<n;++i)
    		ans+=love[i];
    	
    	int m=0,p=0;
    	for(int i=0;i<n;++i)
    	{
    		if(visited[i]) continue;
    		if(i*decr[i]>m)
    		{
    			m=i*decr[i];
    			p=i;
    		}
    	}
    	visited[p]=1;
    	for(int i=0;i<n;++i)
    	{
    		if(visited[i]) continue;
    		love[i]-=(i*decr[i]);
    	}
    }
    cout<<ans<<endl;
    return 0;
}
