#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
	int t;
	long long int n;
	long long int j;
	double p=0;
	double ans;
	cin >> t;
	for(int i=0; i<t; i++)
	{
		ans = 0;
		cin >> n;
		j = n;
		while(j > 0)
		{
			cin >> p;
			ans = ans + (p/(n*100));
			j--;
		}
		ans = ans*100;
		cout <<fixed<<setprecision(8)<< ans << endl;
	}
    
    return 0;
}
