#include <iostream>
#include <numeric>
#include <cstdio>
long long a[1000005];
long long b[1000005];
using namespace std;

int main()
{
    long long int n;
    scanf("%lld", &n);
    for (int i = 0; i < n; i++) {
    	scanf("%lld", &a[i]);
    	b[i] = 0;
    }
    b[n] = 0;
    long long int m;
    scanf("%lld", &m);
    while (m--) {
    	long long int l, r, c;
    	scanf("%lld%lld%lld", &l, &r, &c);
    	l--;
    	r--;
    	b[l] += c;
    	b[r + 1] -= c;
    	
    }
    printf("%lld\n", a[0] + b[0]);
    
    for (int i = 1; i < n; i++) {
    	b[i] += b[i - 1];
    	a[i] += b[i];
    	printf("%lld\n", a[i]);
    }
    return 0;
}
