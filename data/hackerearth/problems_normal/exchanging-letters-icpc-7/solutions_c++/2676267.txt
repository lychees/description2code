vector<int> arr;;
LL dp[2010][5010];

LL solve(int idx , int left)
{
	if(left < 0) return 0;
	if(left == 0) return 1;
	else if(idx == sz(arr)) return 0;
	else if(dp[idx][left] != -1) return dp[idx][left];
	LL ret = solve(idx + 1 , left);
	ret += solve(idx + 1 , left - 2);
	ret %= MOD;
	return (dp[idx][left] = ret);
}

int main()
{
	inp_s;
	FOR(i,0,2010) FOR(j,0,5010) dp[i][j] = -1;
	int m;
	cin >> n >> m;
	int sum;
	cin >> sum;
	arr.resize(m);
	FOR(i,0,m) cin >> arr[i];
	FOR(i,0,m) arr[i] -= 1;
	FOR(i,0,m) sum -= arr[i];
	if(sum < 0 or sum > 2 * m)
		cout << 0 << endl;
	else
	{
		sort(all(arr));
		cout << solve(0 , sum) << endl;
	}
	return 0;
}
