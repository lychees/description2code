#include <bits/stdc++.h>
using namespace std;
#define ll long long 
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    string x;
    ll len;
    while(t--)
    {
    	cin>>x;
    	len=x.length();
    	map<char,int> m;
    	for(int i=0;i<len;i++)
    	{
    		m.insert(make_pair(x[i],0));
    	}
    	len=m.size();
    	ll ans=len*(len+1);
    	ans/=2;
    	printf("%lld\n",ans);
    }
    
    return 0;
}
