#include <iostream>
using namespace std;

int main()
{
    int a[2],arr[10000];
    int t,i,l,c=0,n,k,flag=0;
    cin>>t;
    for(i=1;i<=t;i++)
    {
    	cin>>n>>k;
    	for(a[0]=1;a[0]<=9;a[0]++)
    	{
    		for(a[1]=0;a[1]<=9;a[1]++)
    		{
    			if(a[0]!=a[1])
    			  c++;
    			if(c==k)
    			{
    				flag=1;
    				for(l=0;l<n;l++)
    				{
    					if(l%2==0)
    					  arr[l]=a[0];
    					else
    					  arr[l]=a[1];
    				}
    				for(l=0;l<n;l++)
    				  cout<<arr[l];
    				cout<<endl;
    				break;
    			}
    		}
    		if(flag==1)
    		  break;
    	}
    	if(c!=k)
    	  cout<<"-1"<<endl;
    	flag=0;
    	c=0;
    }
    return 0;
}
