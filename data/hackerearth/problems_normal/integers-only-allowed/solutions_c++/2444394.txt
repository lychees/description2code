#define _CRT_SECURE_NO_WARNINGS

#include <iostream>
#include <algorithm>
#include <stdio.h>
#include <cmath>
#include <math.h>
#include <set>
#include <vector>
#define MAX 100010
using namespace std;
#define ll long long
#define ld long double

ll x[MAX], y[MAX], n;

ld plos(){
x[0] = x[n]; y[0] = y[n];
ld s = 0;
for (int i = 0; i<n; i++) s += (y[i] - y[i + 1])*(x[i] + x[i + 1]);
return (abs(s) / 2);
}

ll gcd(ll a, ll b){
if (a == 0 || b==0) return (a + b); else
if (a>b) return (gcd(a % b, b)); else return (gcd(a, b % a));
}

ll cntsegment(ll x, ll y, ll x2, ll y2){
return ((gcd(abs(x - x2), abs(y - y2))) + 1);
}

ll sidepolygon(){
ll s = 0;
x[n + 1] = x[1]; y[n + 1] = y[1];
for (int i = 1; i<n + 1; i++)
{
    s += cntsegment(x[i], y[i], x[i + 1], y[i + 1]);
    s--;
}
return s;
}

int main(){
cin >> n;
for (int i = 1; i<n + 1; i++) cin >> x[i] >> y[i];
cout << (ll) (plos()-( (ld) sidepolygon()/2 ) + 1) << endl;

return 0;
}
