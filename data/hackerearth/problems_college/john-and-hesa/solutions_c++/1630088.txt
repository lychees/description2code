#include<iostream>
#include<stdio.h>

using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int x,p,count=0;
        scanf("%d%d",&x,&p);
        if(x%p!=0)
            printf("0 %d\n",x);
        else
        {
            while(x%p==0)
            {
                count++;
                x/=p;
            }
            if(x==1)
                printf("%d 0\n",count);
            else
                printf("%d %d\n",count,x);
        }
    }
    return 0;
}
