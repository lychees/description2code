#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;

int main()
{
    int t,i,c[100],j,r=0,temp;char a[100],b[100][100];
    scanf("%d\n",&t);
    memset(c,0,sizeof(c));
    for(i=0;i<t;i++){
    	scanf("%s",a);
    	for(j=0;j<r;j++)
    	{
    		if(strcmp(a,b[j])==0){
    			
    			c[j]++;
    			break;}
    	}
    	if(j==r)
    	{	strcpy(b[r],a);
    		c[r++]=1;
    	}
    }
    for(i=0;i<r-1;i++){
    	for(j=0;j<r-i-1;j++){
    		if(c[j]<c[j+1]){
    			strcpy(a,b[j]);
    			strcpy(b[j],b[j+1]);
    			strcpy(b[j+1],a);
    			temp=c[j];
    			c[j]=c[j+1];
    			c[j+1]=temp;
    		}
    	}
    }
    for(i=0;i<r;i++)
    	printf("%s %d\n",b[i],c[i]);
    return 0;
}
