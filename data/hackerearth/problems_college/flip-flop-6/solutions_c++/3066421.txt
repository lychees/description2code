#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<iostream>
using namespace std;
char arr[10000001];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		cin>>arr;
		int count=0;
		int prod=0;
		int l=strlen(arr);
	
		for(int i=1;i<l;i++)
		{
			if(arr[i]==arr[i-1])
			count++;
			else
			{
			prod=prod+count;
				count=0;
			}
				
		}
		prod=prod+count;
		cout<<prod<<endl;
	}
}
