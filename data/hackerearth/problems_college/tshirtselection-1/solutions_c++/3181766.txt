#include <iostream>
#include <map>
#include <algorithm>
using namespace std;
map<int ,int> last_pos;
int main()
{
    int T;
    for (cin >> T; T; T--) {
    	last_pos.clear();
    	int N; cin >> N; int a;
    	int maxi = 0, last = 0, ans =0;
    	for (int i = 0; i < N; i++) {
    		cin >> a;
    		if (last_pos.count(a)) {
    			int len = i - last;
    			if (maxi < len) {
    				maxi = len;
    				ans = last;
    			}
    			last = max(last, last_pos[a] + 1);
    		}
    		last_pos[a] = i;
    		//cout << a << " " << last << " " << maxi << " " << ans <<"\n";
    	}
    	int len = N - last;
    	if (maxi < len) { maxi = len; ans = last;}
    	//cout << maxi <<"\n";
    	cout << ans + 1 << " " << ans + maxi <<"\n";
    }
    return 0;
}
