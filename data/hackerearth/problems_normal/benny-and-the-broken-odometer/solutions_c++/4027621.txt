#include <bits/stdc++.h>
using namespace std;

#define ll long long int

ll func(ll n)
{
    ll k,z,y,x,si,cur,fact,i;
    vector <ll> v;
    k = 0;
    z = n;
    while (z) {
        k++;
        x = z % 10;
        v.push_back(x);
        z = z / 10;
    }
    cur = 0;
    fact = 1;
    si = v.size();

    for (i = 0; i < si-1; i++) {
        y = 9 * cur + fact;
        fact = fact * 10;
        cur = y;
    }
    if (v[si-1] > 3) {
        y = max(v[si-1] - 1, 0LL);
        y = y * cur + fact;
        cur = y;
    } else {
        y = max(v[si-1], 0LL);
        y = y * cur;
        cur = y;
    }

    z = cur;
    return z;
}

int main()
{
    ll t,n,z,fact,i,ans,temp;

    scanf("%lld", &t);
    while (t--) {
        scanf("%lld", &n);

        temp = n;
        z = n;
        fact = 1;
        while ((z / 10)) {
            fact = fact*10;
            z = z / 10;
        }
        ans = 0LL;
        while (n) {
            ans = ans + func(n);
            //cout <<func(n)<<endl;
            n = n % fact;
            while (fact > n)
                fact = fact / 10;
        }
        ans = (temp - ans);
        assert(ans >= 0);
        printf("%lld\n", ans);
    }
    return 0;
}
