#include <iostream>
using namespace std;
int main() {
    int t;
    scanf("%d",&t);
    int data[34]={0,135739,16648,4717,2017,997,493,325,196,154,109,79,52,34,34,19,19,10,10,10,10,4,4,4,4,4,1,1,1,1,1,1,1,1};
    while (t--) {
        unsigned long long int n,k,diff=0;
        int ans=0;
        scanf("%llu %llu",&n,&k);
        if(n>k)
            diff=n-k;
        else
            diff=k-n;
            for (int k=1; k<=33; k++)
                    if (diff%k==0)
                        ans+=data[k];
        printf("%0.6f\n",(double)ans/(161700));
    }
    return 0;
}
