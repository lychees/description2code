#include <iostream>
using namespace std;

int main()
{
    int n,k;
    cin>>n>>k;
    int arr[n];
    
    for(int i=0;i<n;i++){
    	cin>>arr[i];
    }
    for(int i=0;i<n;i++){
    	if(arr[i]!=n-i){
    		for(int j=i;j<n;j++){
    			if(arr[j]==n-i){
    				int temp = arr[j];
    				arr[j]=arr[i];
    				arr[i]=temp;
    				k--;
    			}
    		}
    	}
    	if(k<=0) break;
    }
    for(int i=0;i<n;i++){
    	cout<<arr[i]<<" ";
    }
    return 0;
}
