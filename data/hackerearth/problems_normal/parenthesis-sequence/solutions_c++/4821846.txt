#include <iostream>
#include <stack>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int z=0;z<t;z++)
    {
    	char a[1000];
    	cin>>a;
    	stack <int> s;
    	int c=0,d=0,i=0;
    	while(a[i]!='\0')
    	{
    		if(a[i]=='(')
    		{
    			s.push(1);
    			
    		}
    		else
    		{
    			if(s.empty())
    			{
    				c++;
    				
    			}
    			else
    			{
    				s.pop();
    			}
    		}
    		i++;
    		
    	}
    	while(!s.empty())
    	{
    		d++;
    		s.pop();
    	}
    	while(c>0)
    	{
    		cout<<'(';
    		c--;
    	}
    	cout<<a;
    	while(d>0)
    	{
    		cout<<')';
    		d--;
    	}
    
    cout<<endl;
    }
}
