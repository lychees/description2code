#include <iostream>
#include<bits/stdc++.h>
using namespace std;
unsigned long long int fastpow(unsigned long long int w,unsigned long long int e,unsigned long long int MOD)
{
    unsigned long long int a,b,ans=1;
    a=w;
    b=e;
    while(b>0)
    {
        if(b%2==1)
        {
           ans=((ans%MOD)*(a%MOD))%MOD;
        }
        a=((a%MOD)*(a%MOD))%MOD;
        b=b/2;
    }
    return ans;
}
int main()
{
    unsigned long long int t,a,p,i,j,k,l;
    cin>>t;
    while(t--){
    	cin>>a>>p;
    	k=(p-1)/2;
    	l=fastpow(a,k,p);
    	l=l%p;
    	if(l==1){
    		cout<<"YES\n";
    	}
    	else{
    		cout<<"NO\n";
    	}
    }
    return 0;
}
