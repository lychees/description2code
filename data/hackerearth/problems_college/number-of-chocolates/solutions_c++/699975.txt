#include <cstdlib>
#include <cstdio>
#include <iostream>

using namespace std;

int main(int argc, char** argv) {

    int n,q;
    long long a[1000001]={0};
    cin>>n>>q;
    
    scanf("%lld",&a[0]);
    for(int i=1;i<n;i++){
        scanf("%lld",&a[i]);
        a[i]=a[i]+a[i-1];
    }
    int g1,p1,g2,p2,l,r;
    for(int i=0;i<q;i++)
    {
        scanf("%d%d",&g1,&p1);
        scanf("%d%d",&g2,&p2);
   
    if(g1<p1)l=g1;
    else l=p1;
    
    if(g2<p2)r=p2;
    else r=g2;
        
    long long total=0;
    if(l==0)total=a[r];
    else total = a[r]-a[l-1];
    
    if(l==g1 && g2<p1) total=total-(a[p1-1]-a[g2]);
    if(l==p1 && p2<g1) total=total-(a[g1-1]-a[p2]);
    
    printf("%lld\n",total);
    }
    return 0;
}
