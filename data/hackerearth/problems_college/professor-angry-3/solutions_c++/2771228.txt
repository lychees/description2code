#include <iostream>
using namespace std;

int main()
{
    long long int t;
   cin>>t;
while(t--)
{
    long long int n,k;
    cin>>n>>k;
    long long int a[n],i;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    long long int count=0;
    for(i=0;i<n;i++)
    {
    	if(a[i]<=0)
    	count++;
    	if(count==k)
    	break;
    }
    if(count<k)
    cout<<"YES"<<endl;
    else
    cout<<"NO"<<endl;
}
    return 0;
}
