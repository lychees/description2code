#include <bits/stdc++.h>
using namespace std;
int srch(int a[],int l,int h,int x){
	int m;
	m=l+h;
	m/=2;
	if(a[m]==x)
		return 1;
	else if(l>=h)
		return 0;
	else if(a[m]<x)
		return srch(a,m+1,h,x);
	else if(a[m]>x)
		return srch(a,l,m-1,x);
	
}
int main()
{
    //cout << "Hello World!" << endl;
    int t,n,i,x;
    cin>>n;
    cin>>t;
    int a[n];
    for(i=0;i<n;i++)
    	cin>>a[i];
    sort(a,a+n);
    while(t--){
    	int flag=0;
    	cin>>x;
    	flag=srch(a,0,n-1,x);
    	if(flag)
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    return 0;
}
