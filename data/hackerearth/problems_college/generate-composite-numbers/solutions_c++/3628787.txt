#include<bits/stdc++.h>
#define max 10007
using namespace std;
int prime[max];
int main()
{
	int i,j;
	for(i=1;i<max;i++)
	{
		prime[i] = 0;
	}
	prime[0]=0;
	prime[1]=0;
	for(i=2;i<max;i++)
	{
		if(prime[i] == 0)
		{
			prime[i] = 0;
			for(j=i;i*j<max;j++)
			{
				prime[i*j] = 1;
				
			}
		}
	}
	int n;
	cin>>n;
	for(i=0;i<=n;i++)
	{
		if(prime[i]==1)
		{
			cout<<i<<" ";
		}
	}
}
