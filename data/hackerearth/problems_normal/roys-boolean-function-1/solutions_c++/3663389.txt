#include <iostream>

using namespace std;


bool isprime(long int a)
{
	
	long int x;
	if(a==1)return false;
	if(a%2==0 && a!=2) return false;
	
	
	for(x=3;x*x<=a;x+=2)
	if(a%x==0) return false;
	
	return true;
}


long int isrel(long int a,long int b)
{
	if(b==0) return a;
	else return isrel(b,a%b);
	
}

int main() {

 int t;
 

cin>>t;


 while(t--)
 {
 	
 	long int n,i,ans=1;
     
     cin>>n;
     
     for(i=2;i<n;i++)
     {
         if(isrel(i,n)==1) ans++;
     }
     
     
     
    // cout<<ans;
     if(isprime(ans)) cout<<"TRUE";
     else cout<<"FALSE";
     
  cout<<"\n";
}	return 0;
}
