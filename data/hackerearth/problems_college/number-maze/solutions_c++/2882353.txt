#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define MOD 1000000007LL
#define MAXN 100010
ll A[MAXN];

int main()
{
	ll n,m;
	cin >> n >> m;
	bool start = false;
	for(ll i=n+1;i<=m-1;++i)
	{
		bool ls = true;
		if(i%5==0)
			ls = false;
		if(i%3==0)
			ls = false;
		if(i%15==0)
			ls = true;
		if(ls)
		{
			if(start)
				cout << ",";
			start = true;
			cout << i;
		}
	}
	return 0;
}
