import copy

def rev(n):
	n_list = list(str(n))
	n_rev_list = copy.deepcopy(n_list)
	n_rev_list.reverse()
	n_rev = ''.join(n_rev_list)
	return int(n_rev)

t = int(input())
for i in range(t):
	n = int(input())
	n_rev = rev(n)
	if n == n_rev:
		print('No')
	elif n**2 == rev(n_rev**2):
		print(n_rev)
	else:
		print('No')
