#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<int,int> ii;
typedef vector<int> vi;
typedef vector<ii> vii;

const int  MOD=1000000007;
const int  INF= int(1e9);
const int  MAX=50005;
int a[MAX];
int main()
{
	ios_base::sync_with_stdio(false);
    int testCases;
    cin>>testCases;
    while(testCases--) {
        int c,n;
        cin>>a[0]>>a[1]>>c>>n;
        if(n==1) {
            if(a[0]==0) {
                cout<<"YES\n";
            } else {
                cout<<"NO\n";
            }
            continue;
        }
        int num=a[0]*10+a[1]; 
        for(int i=2;i<n;i++) {
            a[i] = ( a[i - 1]*c + a[i - 2] ) % 10;
            num*=10;
            num+=a[i];
            num%=41;
        }
        if(num % 41 == 0) {
            cout<<"YES\n";
        } else {
            cout<<"NO\n";
        }

    }
	return 0;

}
