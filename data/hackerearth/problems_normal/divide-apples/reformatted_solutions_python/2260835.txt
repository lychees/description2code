n = input()
l = map(int, raw_input().split(" "))
s = sum(l)/n
neg = [0]*n
for i in range(n):
    neg[i] = l[i]-s
# print neg
p = 0
cum = [0]*n
for i in range(n):
    cum[i] = p
    p += neg[i]
cum.sort()
# print cum
med = 0
if n % 2 == 0:
    med = (cum[(n-1)/2]+cum[n/2])/2
else:
    med = cum[n/2]
ans = 0
# print med
for i in range(n):
    ans += abs(cum[i]-med)
print ans
