#include <bits/stdc++.h>
#define ll long long
using namespace std;
int h[1010];
int main()
{
ios_base::sync_with_stdio(false);
cin.tie(NULL);
int test;
cin>>test;
while(test--)
{
	int n,m;//number of nodes & relationships
	cin>>n>>m;
	memset(h,0,sizeof(h));
	list<int> l[n];
	int i,x,y;
	for(i=0;i<m;i++)
	{
		cin>>x>>y;
		x--;y--;//Converting to 0-based index
		l[x].push_back(y); //y is dependent on x or in other words x is pre-requisite of y
		h[y]++; //indegree of y
	}
	int ans=0;
	int flag=1;
	while(flag)
	{
		flag=0;
		for(i=0;i<n;i++)
		{
			if(h[i]==0)
			{
				flag=1;
				ans++;
				list<int>::iterator it;
				for(it=l[i].begin();it!=l[i].end();it++)
				{
					h[*it]--; //indegree reduces by 1
				}
				h[i]=-1;//Done
			}
		}
	}
	cout<<ans<<endl;
}
return 0;
}
