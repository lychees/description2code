void bfs(string s){
    my[s]=1;
    queue<string> q;
    q.push(s);
    while(!q.empty()){
        
        string tmp=q.front();
        //cout<<tmp<<endl;
        //cout<<tmp<<endl;
        q.pop();
        string s1=tmp;
        rep(i,n)s1[i]=tmp[(i-h+n)%n];
        if(my.find(s1)==my.end()){my[s1]=1;q.push(s1);ans=min(ans,s1);}
        rep(i,n){
            if(i%2==1)tmp[i]=(tmp[i]-'0'+a)%10+'0';
        }
        if(my.find(tmp)==my.end()){my[tmp]=1;q.push(tmp);ans=min(ans,tmp);}
        
        
    }
}
