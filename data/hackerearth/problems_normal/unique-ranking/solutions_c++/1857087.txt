#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n,m;
    cin>>n>>m;
    int M[n][m];
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<m;j++)
    	cin>>M[i][j];
    }
    int t;
    cin>>t;
    while(t--)
    {
    	int q;
    	cin>>q;
    	int Q[q];
    	for(int i=0;i<q;i++)
    	cin>>Q[i];
    	int ctr=q-1;
    	int a[n];
		for(int j=0;j<n;j++)
		a[j]=1;
		int min;
    	int chk1=0;
    	for(int i=0;i<q;i++)
    	{
    		int pr=Q[i];
    		min=101;
    		for(int j=0;j<n;j++)
    		{
    			if(min>M[j][pr-1] && a[j]>0)
    			{
    				min=M[j][pr-1];
    				chk1=j+1;
    				
    			}
    		}
    		
    		int ctrr=0;
    		int chk2=0;
    		for(int j=0;j<n;j++)
    		{
    			if(min==M[j][pr-1] && ctrr==0 && a[j]>0)
    			{
    				ctrr++;
    				a[j]=1;
    			}
    			else if(min==M[j][pr-1] && ctrr>0 && a[j]>0)
    			{
    				if(ctr>i)
    				{
    					a[j]=1;
    					chk1=0;
    				}
    			}
    			else if(min<M[j][pr-1])
    			a[j]=0;
    			
    		}
    		if(chk1!=0)
    		break;
    		
    	}
    	if(chk1!=0)
		{
			
			cout<<chk1<<endl;
			//break;
		}
    }
    return 0;
}
