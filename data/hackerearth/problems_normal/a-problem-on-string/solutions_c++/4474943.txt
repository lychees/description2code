#include <iostream>
using namespace std;
long cnt[1000];
int main()
{
	long long i,n,t;
	char a,b;
	string str;
    cin >> n;
    cin >> str;
    for(i = 0;i < n;i++){
    	cnt[str[i] - 'a']++;
    }
//    for(i = 0;i < 26;i++){
  //  	cout << cnt[i] << " ";
    //}
    cin >> t;
    while(t--){
    	cin >> a >> b;
    	if(a == b){
    		cout << (cnt[a - 'a']*(cnt[a - 'a'] - 1))/2 << endl;
    		continue;
    	}
    	cout << cnt[a - 'a']*cnt[b - 'a'] << endl;;
    }
    return 0;
}
