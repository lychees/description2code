#include<bits/stdc++.h>
using namespace std;
typedef unsigned long long int des;
bool vali(des a,des b,des c)
{	des longest=a;
	if(b>longest)longest=b;
	if(c>longest)longest=c;
	des sumof=a+b+c;
	return (longest<(sumof)-longest);
}
int main()
{	int t;
	scanf("%d",&t);
	des a,b,c;
	while(t--)
	{	scanf("%llu%llu%llu",&a,&b,&c);
		if(vali(a,b,c))
		{	if(a==b&&b==c)
			{	printf("YES. It is a equilateral triangle.\n");
			}
			else	if(a==b||b==c||a==c)
				{	printf("YES. It is a isosceles triangle.\n");
				}
				else	printf("YES. It is a scalene triangle.\n");
		
		}
		else
		{	printf("NO\n");
		}
	}
	return 0;
}
