#include<bits/stdc++.h>
#define MOD 1000000007
#define ll long long
#define pb push_back
#define mk make_pair
using namespace std;
vector<int> graph[100005];
map<int,int> mp;
int n,m;
int gb[100005];
int dfs(int v,bool visited[],int col)
{
	int inv=0;
	visited[v]=true;
	if(gb[v]==0&&col==1)
		inv+=1;
	else if(gb[v]==1&&col==0)
		inv+=1;
	for(int i=0;i<graph[v].size();i++)
		if(!visited[graph[v][i]])
			inv+=dfs(graph[v][i],visited,1-col);
	return inv;
}
int fun(int v,int col)
{
	bool visited[100005];
	memset(visited,false,sizeof(visited));
	return dfs(v,visited,col);
}
int main()
{
	int ans;
	cin>>n>>m;
	for(int i=1;i<=n;i++)
	{
		char ch;
		cin>>ch;
		if(ch=='B')
			gb[i]=1;
		else
			gb[i]=0;
	}
	while(m--)
	{
		int u,v;
		cin>>u>>v;
		graph[u].push_back(v);
		graph[v].push_back(u);
	}
	ans=min(fun(1,0),fun(1,1));
	cout<<ans<<endl;
	return 0;
}
