#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int t,p=1;
	cin>>t;
	while(p!=t+1)
	{
		int d,i,n;
		double m,y;
		cin>>d>>i>>m>>n;
		int a[n],j,z=0,cost=0,x;
		for(j=0;j<n;j++)
			cin>>a[j];
		for(j=0;j<n-1;j++)
		{
			if(a[j+1]-a[j]>m || a[j]-a[j+1]>m)
			{
				if(a[j+1]>a[j])
				{
					x=a[j+1]-m;
					cost=x-a[j];
					if(cost>d && i>d)
						cost=d;
					else if(d>i){
						y=(a[j+1]-a[j]-m)/m;
						y=ceil(y);
						if(y*i<d && y*i<cost)
							cost=y*i;
					}
				}
				else{
					x=a[j]-m;
					cost=x-a[j+1];
					if(cost>d && i>d)
						cost=d;
					else{
						y=(a[j]-a[j+1]-m)/m;
						y=ceil(y);
						if(y*i<d && y*i<cost)
							cost=y*i;
					}
				}
			}
			//cout<<"cost="<<cost<<endl;
			z=z+cost;
			cost=0;
		}
		cout<<"Case #"<<p<<":"<<" "<<z<<endl;
		p++;
	}
	return 0;
}
