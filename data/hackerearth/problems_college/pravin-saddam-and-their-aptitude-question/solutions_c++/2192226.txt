#include <iostream>
using namespace std;
     
long long fact(long long n){
    if(n==0)
       return 1;
    if(n==1)
       return 1;
    else
       return n*fact(n-1);
}   
int main(){
    int t; cin>>t;
    while(t--){
    	int a,b;
    	long long sum=0,i,k;
    	cin>>a>>b;
    	for(i=a;i<=b;i++){
    		sum+=fact(i);
    	}
    	k=sum%10;
    	cout<<k<<endl;
    }
    return 0;
}
