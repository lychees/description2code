#include<bits/stdc++.h>
using namespace std;
main()
{
  int n;
  cin>>n;
  long long int max1=-1;
  long long int max2=-1;
  int ar[1000002];
  for(int i=0;i<n;i++)
    scanf("%d",&ar[i]);
  long long int diff;
  for(int i=1;i<n;i++)
  {
    if(ar[i]>=ar[i-1])
    diff=ar[i]-ar[i-1];
    else
      diff=ar[i-1]-ar[i];
    if(diff>max1)
      max1=diff;
  }
  for(int i=0;i<n;i++)
    scanf("%d",&ar[i]);
  for(int i=1;i<n;i++)
  {
    if(ar[i]>=ar[i-1])
    diff=ar[i]-ar[i-1];
    else
      diff=ar[i-1]-ar[i];
    if(diff>max2)
      max2=diff;
  }

  if(max1>max2)
    printf("Dom\n%d",max1);
  else if(max2>max1)
    printf("Brian\n%d",max2);
  else
    printf("Tie\n%d",max2);
}
