s=raw_input()
i=0
p=len(s)
q=input()
while i<q:
	a,b=map(int,raw_input().split())
	if a!=0:a=a%p
	else:a=p
	if b!=0:b=b%p
	else:b=p
	if s[a-1]==s[b-1]:print "Yes"
	else:print "No"
	i=i+1
