#include <bits/stdc++.h>
using namespace std;
typedef long long int i64;
int main()
{
   i64 NumStaws,test,length,width;   
   scanf("%lld",&test);
   while(test--)
   {
   	scanf("%lld",&NumStaws);
   	if(NumStaws%2!=0)
   	NumStaws--;
   	length=floor(NumStaws/4);
   	width=(NumStaws%4);
   	width=width>>1;
   	width+=length;
   	printf("%lld\n",length*width);
   }
    return 0;
}
