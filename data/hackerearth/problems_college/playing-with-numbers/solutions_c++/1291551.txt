#include<bits/stdc++.h>
using namespace std;
long long f(long long n)
{
  long long count = 0;
  while(n)
  {
    count += n & 1;
    n >>= 1;
  }
  return count;
}
 

int main()
{
	ios_base::sync_with_stdio(0);
    int t,flag=1;
   scanf("%d",&t);
    while(t--){
        long long n;
        scanf("%lld",&n);
        if(f(n)%2){
            if(flag==1)
            printf("Sherlock\n");
            else 
                printf("Watson\n");
            
        }
        else {
            if(flag==1){
             printf("Watson\n");
                flag=2;
            }
            else {
                printf("Sherlock\n");
                flag=1;
                
            }
            
        }
        
    }
    return 0;
}
