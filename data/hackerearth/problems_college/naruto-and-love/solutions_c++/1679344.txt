#define INF 2000000000 // 2 billion

int n,m;
int nar[150009],ram[150009];

int modu(int i)
{
    if( i < 0 )
        return -1 * i;
    return i;
}

bool check(int ch)
{
    int i = 0;
    int j = 0;
    while( i < n && j < m )
    {
        if( modu(nar[i] - ram[j]) <= ch )
        {
            i++;
            j++;
        }
        else
            j++;
    }
    if( i == n )
        return true;
    return false;
}

int main()
{
    ios_base::sync_with_stdio(false);cin.tie(0);
    //freopen("input.txt","r",stdin);
    //freopen("output.txt","r",stdout);
    int t;s(t);
    while( t-- )
    {
        s(n);s(m);
        REP(i,0,n - 1)
            s(nar[i]);
        REP(i,0,m - 1)
            s(ram[i]);
        sort(nar,nar + n);
        sort(ram,ram + m);
        ll low = 0,high = INF;
        while( low < high )
        {
            ll mid = ( low + high ) / 2;
            if( check(mid) == true )
                high = mid;
            else
                low = mid + 1;
        }
        printf("%lld\n",low);
    }
    return 0;
}
