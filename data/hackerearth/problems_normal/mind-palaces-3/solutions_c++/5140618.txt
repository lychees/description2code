#include <bits/stdc++.h>
using namespace std;
int a[1001][1001];
void foo(int key , int r , int c)
{
	long int i=0,j=c-1;
        while(i<r and j>=0)
        {
            if(a[i][j]==key)
            {
                cout<<i<<" "<<j<<endl;
                return ;
            }
            else if(a[i][j]>key)
            {
                j=j-1;
            }
            else if(a[i][j]<key)
            {
                i=i+1;
            }
        }
       cout<<-1<<" "<<-1<<endl;
       return ;
}
int main()
{
    int r,c,key;
    cin>>r>>c;
    assert(2<=r && r<=1000);
    assert(2<=c && c<=1000);
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            scanf("%d",&a[i][j]);
            assert(-1000000000 <= a[i][j] && a[i][j] <= 1000000000);
        }
    }
    int q; cin>>q;
    assert(2 <= q && q <= 1000);
    while(q--)
    {
        scanf("%d",&key);
        assert(-1000000000 <= key && key <= 1000000000);

        foo(key,r,c);
    }
    return 0;
}
