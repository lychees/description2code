#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;

int main()
{
	int test;
	cin>>test;
	while(test--){
	    int n,x[1005],y[1005],i;
	    cin>>n;
	    for(i=0;i<n;i++){
	    	scanf("%d%d",&x[i],&y[i]);
	    }
	    sort(x,x+n);
	    sort(y,y+n);
	    int h = x[n-1] - x[0];
	    int w = y[n-1] - y[0];
	    int ans = 0;
	    if(h!=0&&w!=0)
            ans = h*w;
        else if(h==0){
                ans=ans+y[n-1]-y[0];
        }
        else{
                ans=ans+x[n-1]-x[0];
        }
        cout<<ans<<"\n";
	}
}
