#include<bits/stdc++.h>
#include<unistd.h>
using namespace std;
int a,b,c;
void divide(long int x,int k)
{

  if(k==c)
  {
    cout<<endl;
    return;
  }
  cout<<(x/b);
  divide((x%b)*10,++k);
}

void init()
{
  scanf("%d %d %d",&a,&b,&c);
  if(c==0)
  {
    cout<<a/b<<endl;
    return;
  }
  cout<<(a/b)<<".";
  divide((a%b)*10,0);
}

int main()
{
  int t;
  cin>>t;
  while(t--)
  init();
  return 0;
}
