#include <iostream>

using namespace std;

long long MAX = 1000000007;


template<class T>
void egcd(T a, T b, T &x, T &y, T &d) {
	if(b == 0) {
		x = 1;
		y = 0;
		d = a;
		return;
	}
	egcd(b, a%b, x, y, d);
	T temp = y;
	T temp2 = x - y*a/b;
	x = temp;
	y = temp2;
}

long long power(long long a, long long p) {
	if(p == 0) return 1;
	if(p == 1) return a;
	if(p%2) {
		return ((power(a, p/2) * power(a, p/2))%MAX * a)%MAX;
	}
	return (power(a, p/2) * power(a, p/2))%MAX;
}


int main(void) {
	int t;
	cin >> t;
	while(t--) {
		long long n;
		cin >> n;
		long long ans = 0;
		if(n < 3) {
			cout << 0 << endl;
		}
		else {
			ans = power(3, n);
			//exactly one char missing
			long long onemiss = 3*power(2, n);
			long long twomiss = 3;
			ans = (ans - onemiss + twomiss)%MAX;
			if(ans < 0)
				ans += MAX;
			cout << ans << endl;
		}
	}
	return 0;
}
