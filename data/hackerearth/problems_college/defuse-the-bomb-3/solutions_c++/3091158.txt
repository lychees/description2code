#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
int main()
{
    char a[20],c;
    long long int m,n,i,j,k,s,t,l;
    slld(t);
    while(t--)
    {
        ss(a);
        //sd(m);
        l=strlen(a);
        slld(n);
        i=0;
        j=n;
        while(j!=0)
        {
            j/=10;
            i++;
        }
        for(j=0;j<=l-i;j++)
        {
            m=0;
            for(k=0;k<i;k++)
            {
                m=m*10+int(a[j+k]-'0');
            }
            printf("%lld\n",m*n);
        }
    }
    return 0;
}
