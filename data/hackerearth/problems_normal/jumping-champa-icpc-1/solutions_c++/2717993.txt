#include<iostream>
#include<vector>
#include<climits>

using namespace std;

int main()
{
	int tc, i, j, q, num;
	long int min, max, temp, sum;
	cin >> tc;
	for(i=0;i<tc;i++)
	{
		min = INT_MAX;
		max = INT_MIN;
		sum = 0;
		cin >> num;
		cin >> q;
		for(j=0;j<num;j++)
		{
			cin >>  temp;
			if(temp < min)
				min = temp;
			if(temp > max)
				max = temp;
		}
		sum = max-min;
		cout << q*sum << endl;
	}
	return 0;
}
