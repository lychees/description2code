#include <iostream>
using namespace std;
int top;

void push(int stack[],int x,int n){
	if(top == n)
		return;
	else{
		top++;
		stack[top] = x;
	}
}

bool isEmpty(){
	if (top == -1)
		return true;
	else
		return false;
}

void pop(int stack[]){
	if (isEmpty())
		return;
	else
		top = top -1;
}

int Top(int stack[]){
	if(isEmpty())
		return 0;
	else
		return stack[top];
}

int main(){
	int T, i, N, A[100], b, p, j, k, c;
	int ID[100000];
	char D;
	cin >> T;
	for (i = 0; i < T; i++){
		top = -1;
		cin >> N;
		cin >> c;
		push(ID, c, N);
		b = 0;
		for (j = 0; j < N; j++){
			cin >> D;
			if(D == 'P'){
				b = Top(ID);
				cin >> c;
				push(ID, c, N);
			}
			else{
				c = Top(ID);
				push(ID, b, N);
				b = c;
			}
		}
		A[i] = ID[top];
	}
	for(i = 0; i < T; i++){
		cout << "Player " << A[i] << endl;
	}
	return 0;
}
