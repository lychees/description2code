#include <iostream>
#include <vector>
#include <stdio.h>
using namespace std;


int main(){

long long int t,r,g,b,i,j;
cin>>t>>r>>g>>b;
vector<long long int> time(t,0);
vector<long long int> arr(9,0);
for(i=r;i<t;i=i+(2*r)){
for(j=0;j<r;j++){
if(i+j<t){
time[i+j]=1;
}
}
}
for(i=g;i<t;i=i+(2*g)){
    for(j=0;j<g && ((i+j)<t);j++){
    if(time[i+j]==1) time[i+j]=4;
    else{
            time[i+j]=2;
    }
    }
}

for(i=b;i<t ;i=i+(2*b)){
    for(j=0;j<b && ((i+j)<t);j++){
    if(time[i+j]==1) time[i+j]=6;
    else if(time[i+j]==2) time[i+j]=5;
    else if(time[i+j]==4) time[i+j]=7;
    else{
        time[i+j]=3;
    }
    }
}

for(i=0;i<t;i++){
if(time[i]==1) arr[1]++;
else if(time[i]==2) arr[2]++;
else if(time[i]==3) arr[3]++;
else if(time[i]==4) arr[4]++;
else if(time[i]==5) arr[5]++;
else if(time[i]==6) arr[6]++;
else if(time[i]==7) arr[7]++;
else if(time[i]==0) arr[8]++;
}

for(i=1;i<9;i++){
    printf("%d ",arr[i]);
}
printf("\n");
return 0;
}
