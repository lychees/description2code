#include <iostream>
using namespace std;

int main()
{
	int no_of_test_cases;
	cin >> no_of_test_cases;
	int result[10000]={0};

	int i, j, no_of_integers, number;
	for (i = 0; i < no_of_test_cases; i++)
	{
		cin >> no_of_integers;
		
		for (j = 0; j < no_of_integers; j++)
		{
			cin >> number;
			if (number % 2 == 0)
			{
				result[i] = result[i] + 1;
			}
		}
		
	}
	
	for (i = 0; i < no_of_test_cases; i++)
		cout << result[i] << endl;
}
