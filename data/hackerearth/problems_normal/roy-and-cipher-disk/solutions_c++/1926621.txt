#include <iostream>
#include <string.h>

int main()
{
    int t,i,x1,x2,x;
    char ch[101],ch1='s';

    //cout<<ch1-'a';
    while(t--)
    {
    	cin>>ch;

    	else if (strlen(ch)==1 && (ch[0]-'a'>13))cout<<(ch[0]-'a'-26);

    	{	if(x>13)cout<<x-26<<" ";

	    	{

	   		 }
