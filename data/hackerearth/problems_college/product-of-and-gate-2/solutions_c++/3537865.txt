#include<bits/stdc++.h>
using namespace std;
int main()
{

	int tc;
	cin>>tc;
	unsigned long long int ans;
	//cout<<ans<<endl;
	//
	while(tc--)
	{
		unsigned  long long int a,b;
		cin>>a>>b;
		ans=1ULL<<63;
		ans--;
		for(unsigned int i=a;i<=b && ans!=0;++i)
			ans&=i;
		cout<<ans<<endl;
	
	}
	return 0;
}
