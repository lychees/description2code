#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	int n;
    	scanf("%d",&n);
    	long long arr[n],sum=0;
    	int i;
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&arr[i]);
    		sum+=arr[i];
    	}
    	long long lsum=0,x=0,rsum=sum,min;
    	int fs=0;
    	for(i=0;i<n;i++){
    		x=abs(rsum-lsum);
    		if(fs==0){
    			min=x;
    			fs=1;
    		}
    		else{
    			if(x<min){
    				min=x;
    			}
    		}
    		lsum+=arr[i];
    		rsum=rsum-arr[i];
    	}
    	printf("%lld\n", min);
    }
    return 0;
}
