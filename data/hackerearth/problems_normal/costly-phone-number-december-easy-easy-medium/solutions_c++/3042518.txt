#include <bits/stdc++.h>
using namespace std;

int main() {
	int t, len;
	string s;
	int a[12];
	cin>>t;
	while(t--) {
		for(int i = 0; i <= 9; i++) {
			cin>>a[i];
		}
		cin>>len;
		cin>>s;
		bool ok = 1;
		while (ok) {
            ok = false;
            for (int i = 0; i < 10; ++i) {
                for (int j = 0; j < 10; ++j) {
                    if (a[(i + j) % 10] > a[i] + a[j]) {
                        a[(i + j) % 10] = a[i] + a[j];
                        ok = true;
                    }
                }
            }
        }
		long long ans = 0;
		for(int i = 0; i < len; i++) {
			ans  += a[s[i] - '0'];
		}
		cout<<ans<<"\n";
	}
	return 0;
}
