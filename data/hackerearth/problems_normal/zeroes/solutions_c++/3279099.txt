#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair

int main()
{
	int i, j, k, t, n, q, flag;
	vector <pair <int ,int> > v[100001];
	bool visit[100001];
	int prime[100001] = {0};

	memset (visit, false, sizeof(visit));

	for (i = 2; i <= 1e5; i++) {
		if (visit[i] == true) {
			continue;
		}
		for (j = i; j <= 1e5; j+=i) {
			visit[j] = true;
			k = j;
			t = 0;
			while (k%i==0) {
				t++;
				k/=i;
			}
			v[j].pb(mp(i,t));
		}
	}
	cin>>n;	
	for (i = 0; i < n; i++) {
		cin>>j;
		for (k = 0; k < v[j].size(); k++) {
			prime[v[j][k].first]+=v[j][k].second;
		}
	}
/*	for (i = 0; i < 10; i++) {
		cout<<prime[i]<<" ";
	}
	cout<<"\n";
*/	cin>>q;

	while (q--) {
		cin>>k;
		t = INT_MAX;
		for (j = 0; j < v[k].size(); j++) {
				t = min (t, prime[v[k][j].first]/v[k][j].second);
		}
		cout<<t<<endl;
	}
	
	return 0;
}
