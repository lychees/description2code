#include <iostream>
using namespace std;
int a[10000];
int p;
void zeros(){
	for(int i=0;i<10000;i++){
		a[i]=0;
	}
	p=0;
}
void bin(int n){
	while(n!=0){
		a[p] = n%2;
		p++;
		n= n /2;
	}
}
bool check(){
	int i,flag=0;
	for(i=0;i<(p-1);i++){
		//cout<<a[i]<<"\t"<<a[i+1]<<endl;
		if(a[i] == 1 && a[i+1] == 1){
			flag=1; break;
		}
	}
	if(flag==1) return false;
	else return true;
}
int main()
{
    int  test;
    cin>>test;
    while(test--){
    	zeros();
    	int n,i,k,flag=0;
    	cin>>n;
    	while(1){
    		bin(n);
    		//cout<<p<<endl;
    		k = check();
    		if(k==1){
    			cout<<n<<endl;
    			break;
    		}
    		else{
    			n++;
    			zeros();
    		}
    	}
    }
    return 0;
}
