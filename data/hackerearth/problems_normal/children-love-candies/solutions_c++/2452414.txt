#include <iostream>
using namespace std;

int main()
{
    int T,Q;
	long long int X,N,CANDY;
   cin>>Q;
   for(int i=0;i<Q;i++)
   {
   cin>>N>>T;
   CANDY=N;
for(int i=0;i<T;i++){
        N=CANDY;
        X=CANDY;
   if(X%2==0)
   {
      long long int t=N-(X/2);
   	CANDY=t-(t/4);
   }
   else{
       long long int p=(N-(X+1)/2);
   	CANDY=p-(p/4);
       }
   }
   cout<<CANDY<<endl;
   }



    return 0;
}
