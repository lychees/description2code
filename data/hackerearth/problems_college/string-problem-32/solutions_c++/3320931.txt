#include <iostream>
#include <cstring>
using namespace std;
int main() {
    int t,l,flag,sum;
    char ch;
    scanf("%d",&t);
    char A[100005];
    for(int q=1;q<=t;q++) {
        scanf("%s",A);
        l=strlen(A);
        sum=-1;
        ch=A[l-1];
        flag=0;
        for (int i=0; i<l-1; i++) {
            flag=0;
            
            if (A[i]==ch) {
                flag=0;
                int temp=i;
                int temp2=l-1;
                while (temp>=0) {
                    if (A[temp]!=A[temp2]) {
                        flag=1;
                        break;
                    }
                    temp--;
                    temp2--;
                }
                if (flag==0) {
                    sum=i+1;
                }
            }
        }
        printf("Case %d: %d\n",q,sum);
    }
    return 0;
}
