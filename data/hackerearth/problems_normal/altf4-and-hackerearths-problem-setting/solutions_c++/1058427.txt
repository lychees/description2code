#define F(i, n) for(i = 0;i < n; ++i)
#define FI(i, st, ft) for(i = st;i <= ft; ++i)
#define pb(a, b) a.push_back(b)

#define getcx getchar_unlocked
inline void inp( li &n )//fast input function
{
n=0;
li ch=getcx();li sign=1;
while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}
while( ch >= '0' && ch <= '9' )
n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
n=n*sign;
}

li a[2015], b[2015];
li dp[2015];
li mp[2015];
li v[2015];

int main()
{
	//freopen("input.txt", "r", stdin);
	
	li t;
	scanf("%lld", &t);
	while(t--){
		
		memset(dp, 0, sizeof dp);
		memset(mp, 0, sizeof mp);
		memset(v, 0, sizeof v);
		
		li i, j, n, k, mx = 0;
		inp(n); inp(k);
		F(i, n) inp(a[i]);
		F(i, n) inp(b[i]);
		F(i, n){
			mp[a[i]] += b[i];
			++v[a[i]];
			mx = max(a[i], mx);
		}
		
		FI(i, 1, 2005){
			li cur = 0;
			dp[i] = inf;
			if(v[i] == 0) continue;
			for(j = i; j >= 1; --j){
				cur += mp[j];
				dp[i] = min(dp[i], (cur + k) * i + dp[j - 1]);
			}
		}
		//FI(i, 1, 55) 
		cout << dp[mx] << "\n";
	}		

	return 0;
}
