#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[999999],sum[999999]={0};
    for(int i=0;i<n;i++){
    	cin>>a[i];
    	if(i==0)
    	{
    		sum[i]=a[i];
    	}
    	else
    	sum[i]=sum[i-1]+a[i];
    }
    int max=sum[0];
    for(int i=0;i<n;i++)
    {
    	for(int j=i+1;j<n;j++)
    	{
    		if(max<sum[j]-sum[i])
    			max=sum[j]-sum[i];
    	}
    	
    }
    cout<<max;
}
