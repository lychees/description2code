#include <iostream>
#include <string>

using namespace std;
int main(int argc, char const *argv[])
{
	int T , Q , query , amount,count;
	cin>>T;
	int *transactions = new int[T];

	for (int t=0; t<T ; t++)
	{
		cin>>transactions[t];
	}
	cin>>Q;

	for(int q=0; q<Q; q++)
	{
		cin>>query;
		amount = 0;
		count = -2;
		for (int t=0; t< T; t++)
		{
			amount += transactions[t];
			if (query <= amount)
			{
				count = t;
				break;
			}
		}

		cout<<(count+1)<<endl;
	}
	return 0;
}
