#include <iostream>
#include <bits/stdc++.h>
#define MOD 1000000007
#define ll long long
using namespace std;
ll fact[1000005];
ll power(ll x,ll y,ll m){
	ll res;
	res=1;
	x=x%m;
	while(y){
	if(y&1)
		res=(res*x)%m;
	y=y/2;
	x=(x*x)%m;
	}
	return res%m;
}
void get(){
	fact[0]=1;
	fact[1]=1;
	for(int i=2;i<100005;i++){
		fact[i]=(i*fact[i-1])%MOD;
	}	
}
ll comb(ll n,ll r){
	ll res;
	res=fact[n]%MOD;
	res=(res*power(fact[r],MOD-2,MOD))%MOD;
	res=(res*power(fact[n-r],MOD-2,MOD))%MOD;
	return res%MOD;
}
int main()
{
    //cout << "Hello World!" << endl;
    ll t,k,m,ans,s,x;
    cin>>t;
    get();
    while(t--){
    	s=0;
    	cin>>k>>m;
    	for(int i=0;i<k;i++){
    		cin>>x;
    		s+=x;
    	}
    	if(s>m)
    	cout<<"0\n";
    	else
    	{
    		s=m-s;
    		k-=1;
    		ans=comb(s+k,k)%MOD;
    		/*cout<<s<<endl;
    		cout<<k<<endl;
    		cout<<fact[s]<<endl;
    		cout<<fact[k]<<endl;*/
    		cout<<ans<<endl;
    	}
    }
    return 0;
}
