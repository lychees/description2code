#include<iostream>
#include<cstdio>
#include<algorithm>

using namespace std;
main()
{
	long long tc;
	long long n,i,c,k;
	scanf("%d",&tc);
	while(tc--)
	{
		c=0;
		scanf("%lld",&n);
		long long a[n];
		
		for(i=0;i<n;i++){
		
		scanf("%lld",&a[i]);
		}
		
		
		for(i=0;i<n;i++)
		{
			for(k=i;k<n;k++)
			{
			if(a[i]==a[k])
				c++;
			}
			
			
		}
		printf("%lld\n",c);
	}
}
