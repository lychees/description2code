#include <iostream>
#include <map>
using namespace std;
map<string, int> mp;
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	string a;
    	cin>>a;
    	if(mp.find(a)==mp.end())
    	mp[a]=1;
    	else
    	mp[a]++;
    }
    map<string, int>::iterator it=mp.begin();
    for(;it!=mp.end();it++)
    {
    	cout<<it->first<<" "<<it->second<<endl;
    }
    return 0;
}
