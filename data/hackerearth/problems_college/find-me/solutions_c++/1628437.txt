#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--)
   {
   	long n;
   	cin>>n;
   	long long p;
   	cin>>p;
   	long long a[n];
   	long long count=0;
   	int f=0;
   	for(long long i=0;i<n;i++ )
   	{
   		cin>>a[i];
   		if(a[i]==p)
   		f=1;
   		else if(a[i]>p)
   		count++;
   	}
   	if(f==1)
   	cout<<"YES "<<count<<endl;
   	else
   	cout<<"NO 0"<<endl;
   }
    return 0;
}
