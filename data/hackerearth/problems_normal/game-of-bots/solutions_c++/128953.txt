#include <cstdio>
#include <iostream>
#include <string.h>

using namespace std;

int main()
{
    bool flag=true;
    char arr[1000005], arr1[1000005];
    long long int t, len, i=0, j=0, a1=0, a2=0, b1=0, b2=0;
    cin>>t;
    while (t--)
    {
        cin>>arr;
        cin>>arr1;
        len=strlen(arr);
        while (1)
        {
            while (arr[i]=='#')
            {
                if (i>=len)
                {
                    break;
                }
                i++;
            }
            if (arr[i]=='A')
            {
                a1++;
            }
            else if(arr[i]=='B')
            {
                b1++;
            }
            while (arr1[j]=='#')
            {
                if (j>=len)
                {
                    break;
                }
                j++;
            }
            if (arr1[j]=='A')
            {
                a2++;
            }
            else if(arr1[j]=='B')
            {
                b2++;
            }
            if ((arr1[j]!=arr[i]))
            {
                cout<<"No\n";
                flag=false;
                break;
            }
            else if(arr[i]=='A' && arr1[j]=='A' && (j>i))
                {
                    cout<<"No\n";
                    flag=false;
                    break;
                }
            else if(arr[i]=='B' && arr1[j]=='B' && (j<i))
            {
                cout<<"No\n";
                flag=false;
                break;
            }
            else
            {
                if (a1!=a2 || b1!=b2)
                {
                    cout<<"No\n";
                    flag=false;
                    break;
                }
            }
            i++;
            j++;
            if (i>=len || j>=len)
            {
                break;
            }
        }
        if (flag)
        {
            cout<<"Yes\n";
        }
        i=j=a1=a2=b1=b2=0;
        flag=true;
    }
    return 0;
}
