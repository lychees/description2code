#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
 	string str[t];
 	for(int i=0;i<t;i++)
 	cin>>str[i];
 	map<string,int> m;
 	map<string,int>:: iterator it;
 	for(int i=0;i<t;i++)
 	{
 	for(it=m.begin();it!=m.end();it++)
 	{
 		if((*it).first==str[i])
 		{
 		(*it).second++;
 		break;
 		}
 	}
 	if(it==m.end())
 	m[str[i]]=1;
 	}
 	for(it=m.begin();it!=m.end();it++)
 	cout<<(*it).first<<" "<<(*it).second<<endl;
 	
}
