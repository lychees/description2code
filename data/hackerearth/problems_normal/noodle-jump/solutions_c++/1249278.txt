#include <iostream>
#include <algorithm>

using namespace std;
typedef long long ll;
ll dists[1000001];

int main()
{
    int N;
    cin >> N;
    int K;
    cin >> K;
    dists[0] = 0;
    for(int i = 1; i < N; i++)
    {
    	cin >> dists[i];
    }
    sort(dists, dists + N);
    for(int i = 0; i < N; i++)
    {
    	if(dists[i] - dists[i-1] > K)
    	{
    		cout << dists[i-1];
    		return 0;
    	}
    }
    cout << dists[N-1];
}
