#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    static long long int x,y,z,prevy,prevz,maxy,maxz,i;
    cin >> x;
    while(i<x)
    {
    	cin >> y;
    	maxy=max(maxy,abs(prevy-y));
    	prevy=y;
    	i++;
    }
    while(i--)
    {
    	cin >> z;
    	maxz=max(maxz,abs(prevz-z));
    	prevz=z;
    }
    if(maxy==maxz)
     cout << "Tie\n" << maxy;
    else if(maxy>maxz)
     cout << "Dom\n" << maxy;
    else
     cout << "Brian\n" << maxz;
    return 0;
}
