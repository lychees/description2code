t = int(raw_input())

while t:

    k, n = map(int, raw_input().split())

    nodes = 1 + k * (k ** n - 1) / (k - 1)

    m = 0

    while (nodes):
        m += nodes % 10
        nodes /= 10

    print m

    t -= 1
