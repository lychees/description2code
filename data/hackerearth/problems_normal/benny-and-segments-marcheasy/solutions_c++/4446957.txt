int main(){
    //ioS
    ll i,j,k,n,L,t;
    cin>>t;
    while(t--)
    {
        cin>>n>>L;
        pii a[n];
        For(i,0,n) cin>>a[i].ff>>a[i].ss;
        sort(a,a+n);
        bool ans=false;
        For(i,0,n)
        {
            int left=a[i].ff, right=a[i].ss, maxRight=a[i].ff+L;
            if(right==maxRight) ans=true;
            For(j,i+1,n)
            {
                if(!(a[j].ff>=a[i].ff&&a[j].ss<=maxRight)) continue;
                if(a[j].ff>right) continue;
                right=max(right,a[j].ss);
                if(right==maxRight) ans=true;
            }
        }
        if(ans) cout<<"Yes\n";
        else cout<<"No\n";
    }
    return 0;
}
