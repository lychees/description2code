#include<iostream>
#include<cstring>
using namespace std;
int main(){
	int t,i,count;
	cin>>t;
	string a,b;
	int c[26];
	while(t--){
		cin>>a>>b;
		memset(c,0,sizeof(c));
		for(i=0;i<a.length();i++)
		++c[(int)a[i]-97];
		for(i=0,count=0;i<b.length();i++){
			if(c[(int)b[i]-97]>0){
				count=10;
				break;	
			}
		}
		if(count==10)
		cout<<"Yes";
		else
		cout<<"No";
		cout<<endl;
	}
	return 0;
}
