#include<iostream>
#include<stack>

int main(int argc, char const *argv[]) {

  int N, value;
  std::stack<int> S;

  std::cin >> N;

  for (int i = 0; i < N; i++) {

    std::cin >> value;

    if (S.empty() && value == 0) {
      continue;
    }

    if (!S.empty() && value == 0) {
      S.pop();
    } else {
      S.push(value);
    }

  }
  
  value = 0;
  
  while (!S.empty()) {

    value += S.top();
    S.pop();
  }

  std::cout << value << std::endl;
  return 0;
}
