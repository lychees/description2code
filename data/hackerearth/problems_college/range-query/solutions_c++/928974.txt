#include <bits/stdc++.h>
using namespace std;

map<int,vector<int>> mymap;
int main()
{
	int n;
	cin >> n;
	int num;
	for (int i = 0; i < n; i++){
		cin >> num;
		mymap[num].push_back(i);
	}
	int t;
	cin >> t;
	int x,y,val;
	while (t > 0){
		cin >> x >> y >> val;
		if (mymap.find(val) == mymap.end())
			cout << 0 << endl;
		else{
			vector<int> v = mymap[val];
			int count = 0;
			for (int i = 0; i < v.size(); i++){
				if (v[i] <= y && v[i] >= x)
					count++;
				if (v[i] > y)
					break;
			}
			cout << count << endl;
		}
		t--;
	}
    return 0;
}
