#include <bits/stdc++.h>
#define endl '\n'

using namespace std;

int n,a[1<<20];
vector <int> ans;
bool safe[1<<20];

int main() {
  //ios_base::sync_with_stdio(false);
  //cin.tie(NULL);
  //freopen("test.txt","r",stdin);
  int i,max_val;
  
  scanf("%d", &n);
  for(i=1;i<=n;i++) {
    scanf("%d", &a[i]);
  } 

  max_val=0;
  for(i=1;i<=n;i++) {
    if(a[i]>=max_val) safe[i]=true;
    max_val=max(a[i],max_val);
  }

  max_val=0;
  for(i=n;i>=1;i--) {
    if(a[i]>=max_val) safe[i]=true;
    max_val=max(a[i],max_val);
  }

  for(i=1;i<=n;i++) if(safe[i]) ans.push_back(i);

  for(i=0;i<ans.size();i++) {
    if(i>0) printf(" ");
    printf("%d", ans[i]);
  }
  printf("\n");
  
  return 0;
}
