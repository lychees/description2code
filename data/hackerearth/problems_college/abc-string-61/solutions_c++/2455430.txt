#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
long long mod1(long long a,long long b)
{
	long long r=1;
	while(b) 
	{
		if(b%2)
		{	
			r=(r*a)%mod;
		}
		a=(a*a)%mod;
		b=b/2;
	}
	return r;
}
int main()
{
    int test;
    cin>>test;
    while(test--)
    {
    	long long int n;
    	cin >> n;
    	long long int ans = (3 * (mod1(3 , n-1) - mod1(2 , n) + 1))%mod;
    	while(ans < 0)
    		ans += mod;
    	cout << ans << endl;
    }
    return 0;
}
