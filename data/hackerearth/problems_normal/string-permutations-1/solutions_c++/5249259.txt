#include<bits/stdc++.h>
using namespace std;
vector<pair<string,int> > a[26]={};
pair<int,int> p1;
int main()
{
    int t,i,j,k;
    string s;
    cin>>t;
    for(k=0;k<t;k++)
    {
        cin>>s;
        int index=s[0]-'a',flag=0;
        //cout<<a[index].size()<<" ";
        for(i=0;i<a[index].size();i++)
            if(a[index][i].first==s)
               {
                   flag=1;
                   a[index][i].second++;
               }
        if(i==a[index].size()&&flag==0)
            a[index].push_back(make_pair(s,1));
    }
    p1.first=p1.second=0;
    int maxi=-1;
    for(i=0;i<26;i++)
    {
        for(j=0;j<a[i].size();j++)
        {
            //cout<<a[i][j].first<<a[i][j].second<<endl;
            if(a[i][j].second>maxi)
                {
                    p1=make_pair(a[i][j].second,1);
                    maxi=a[i][j].second;
                }
            else if(a[i][j].second==maxi)
                {
                    p1.second++;
                    //cout<<"yo";
                }
        }
    }
    //cout<<p1.first<<p1.second<<endl;
    cout<<p1.first*p1.second<<endl;
}
