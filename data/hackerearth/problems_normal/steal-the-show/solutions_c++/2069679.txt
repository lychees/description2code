#include<bits/stdc++.h>
using namespace std;
int main()
{
	int x,y,z,xxx; cin>>x>>y>>z;
	int n; cin>>n;
	vector<int>a,b;
	for(int i=0;i<n;i++)
	{
		cin>>xxx;
		a.push_back(xxx);
	}
	sort(a.begin(),a.end());
	b.resize(n+3,0);
	b[0] = min(y,z);
	b[n+2] = max(y,z);
	b[(n+3)/2] = x;
	int j=0;
	for(int i=0;i<n;)
	{
		if(b[j]==0)
		{
			b[j] = a[i];
			j++;
			i++;
		}
		else j++;
	}
	//for(int i=0;i<n+3;i++)cout<<b[i]<<' ';
	//cout<<endl;
	int sum = 0;
	for(int i=1;i<n+3;i++)sum += abs(b[i]-b[i-1]);
	cout<<sum<<endl;
}
