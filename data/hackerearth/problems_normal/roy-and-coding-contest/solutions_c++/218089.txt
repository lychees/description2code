#include <iostream>
using namespace std;

int main()
{
	int t;
	long long n,m;
	cin >> t;
	while ( t-- ) {
		cin >> n >> m;
		if ( n == 1 ) {
			cout << "0" << endl;
			continue;
		}
		if ( m == 1 ) {
			cout << n << endl;
			continue;
		}
		long long comp=1,pen=1;
		long long ans = 1;
		for ( ;; ) {
			if ( comp+pen >= n ) {
				cout << ans+1 << endl;
				break;
			}
			else {
				long long tmp = comp;
				comp += pen;
				ans++;
				if ( pen+tmp >= m ) {
					ans += (n-comp)/m + ((n-comp)%m != 0);
					cout << ans << endl;
					break;
				}
				pen += tmp;
			}
		}
	}
    return 0;
}
