#include <iostream>
#include <string>
using namespace std;

int main()
{
    string st;
    int t;
    cin >> t;
    char ch[30];
    for(int u=1, ll=90;ll>=65;u++,ll--)
    ch[u]=(char)ll;
    for(int k=1;k<=t;k++)
    {
    cin>>st;
    cout<<"\n";
    int cnt[25],c=0,i,p=0,j,l=st.length();
    for(i=90;i>=65;i--)
    {
        cnt[c]=0;
        for(j=0;j<l;j++)
    	{
    		st[j]=toupper(st[j]);
    		if((int)st[j]==i)
    		{
    			cnt[c]++;
    		}
    	}
    	if(cnt[c]!=0)
    	{
    		if((char)i==ch[cnt[c]])
            continue;
            else
            {
                p=1;
                break;
            }
    	}
    	c++;
    }
    if(p==0)
    {
    	cout<<"Yes";
    }
    else
    cout<<"No";
    p=0;
    }
}
