#include <iostream>
#include<map>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n,m;
    	cin>>n>>m;
    	map<long long int,long long int> mymap;
    	long long int a[n+m];
    	for(int i=0;i<n+m;i++)
    		cin>>a[i];
    	for(int i=0;i<n;i++)
    		mymap[a[i]]++;
    	for(int i=n;i<n+m;i++)
    	{
    		if(mymap.find(a[i])!=mymap.end())
    			cout<<"YES"<<endl;
    		else
    		{   mymap[a[i]]++;
    			cout<<"NO"<<endl;
    		}
    	}
    }
    return 0;
}
