n=int(raw_input())
s=""
for i in range(0,n):
	s=raw_input().split(" ")
	a=s[0]
	b=s[1]
	cnt=0
	for j in range(0,len(a)):
		if((int(a[j])+int(b[j]))>9):
			cnt=cnt+1
	if(cnt==0):
		print'No carry operation'
	elif(cnt==1):
		print cnt,'carry operation'
	else:
		print cnt,'carry operations'
