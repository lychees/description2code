#include <iostream>
using namespace std;

int main()
{
    int t;
    long long r,k,i,j,x,count,sum;
    cin>>t;
    while(t--)
    {
    	sum=0;
    	count=0;
    	long long a[100005];
    	cin>>r>>k;
    	for(i=1;i<=r;i++)
    	{
    		count=0;
    		j=0;
    		x=i;
    		while(x)
    		{
    			a[j]=x%2;
    			x/=2;
    			if(j-2>=0 && a[j]==1)
    			{
    				if(a[j-1]==0 && a[j-2]==1)
    				{
    					count++;
    				}
    			}
    			j++;
    		}
    		if(count>=k)
    		{
    			sum++;
    		}
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
