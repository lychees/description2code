#include <iostream>
using namespace std;

int main()
{
    int test; long a,b,num,num2,rev,divi,rem;
    cin>>test;
    while(test--)
    { int count=0;
    	cin>>a; cin>>b;
    	for(num=a;num<=b;num++)
    	{ num2=num; rev=0;
    	do
    	{
    		rem=num2%10;
    		rev=(rev*10)+rem;
    		divi=num2/10;
    		num2=divi;
    	}while(divi!=0);
    		if(rev==num)
    		count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
