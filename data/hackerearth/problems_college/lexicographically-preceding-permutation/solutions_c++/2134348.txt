#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

bool check(int arr[],int n)
{
    int foo[n];
    for(int i = 0; i < n; i++)
        foo[i] = arr[i];
    sort(foo,foo+n);

    for(int i = 0; i < n; i++)
        if(foo[i] != arr[i])
            return false;
    return true;
}
