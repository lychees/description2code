#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
int dp[1000010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	dp[0]=1;//Empty String 
	dp[1]=1;// '0'
	int i;
	for(i=2;i<1000001;i++)
	{
		dp[i]=(dp[i-1]+dp[i-2])%mod; //As we can concatenate '0' to dp[i-1] and concatenate '11' to dp[i-2]
	}
	int test;
	cin>>test;
	while(test--)
	{
		int n;
		cin>>n;
		cout<<dp[n]<<endl;
	}
	return 0;
}
