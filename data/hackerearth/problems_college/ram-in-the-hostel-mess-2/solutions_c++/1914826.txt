#include <iostream>

using namespace std;

long dp (long n) {
	if (n == 1) return 2;
	if (n == 2) return 3;
	return dp(n-1) + dp(n - 2);
}

int main () {
	int t , _ ;
	cin >> t;

	for (_ = 0; _ < t; _++) {
		long __ , ans = 0;
		cin >> __;
		ans = dp(__);
		ans = ans * ans;
		cout << ans << endl;
	}

	return 0;
}
