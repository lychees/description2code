#include <iostream>
using namespace std;
 
bool palindrome(int x)
{
	int r = x;
	int p = 0;
	while(x)
	{
		p = p*10 + x%10;
		x = x/10;
	}
	if(r == p)
	return true;
	else
	return false;
}
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int a,b;
    	cin>>a>>b;
    	int i;
    	int count = 0;
    	for(i = a; i <= b; i++)
    	{
    		if(palindrome(i))
    		count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
