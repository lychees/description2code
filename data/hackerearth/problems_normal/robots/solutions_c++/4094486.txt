#include <iostream>
#include <algorithm>
using namespace std;
long factorial(int n)
{
   int c;
   long result = 1;

   for (c = 1; c <= n; c++)
         result = result*c;

   return result;
}
int main()
{
    int t;
    int n;
    int max[1000];
    long long int modulo =1000000007;

    cin>>t;
    while(t--){
    cin>>n;
    for(int i=0;i<n;i++)
    		cin>>max[i];
   sort (max,max+n);
   long long int sum = 1;
   for(int i=0;i<n;i++)
   {
   	 //long long int result = (factorial(max[i]-i)/(factorial(1)*factorial(n-(i+1))));
    if(max[i]-i == 0)
    {
      sum =0;
      break;
    }
   	 sum=sum*(max[i]-i);
   	 sum=sum%modulo;
   	// cout<<max[i];



   }

   cout<<sum<<"\n";
    }
}
