#include <iostream>
using namespace std;

int main()
{
    int n,lsum=0,rsum=0;
    cin>>n;
    int a[n][n];
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<n;j++)
    	{
    		cin>>a[i][j];
    	}
    }
    for(int i=0; i<n;i++)
    {
    	lsum=lsum+a[i][i];
    	rsum=rsum+a[i][n-i-1];
    }
    if(lsum>rsum)
    cout<<(lsum-rsum);
    else
    cout<<(rsum-lsum);
    return 0;
}
