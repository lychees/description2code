m = 1000003
fact = []

fact.append(1)
for i in range(1, 1000003):
    fact.append((fact[i-1]*i) % m)

t = input()
while (t):
    t -= 1

    user_input = raw_input().split()
    n = int(user_input[0])
    x = int(user_input[1])

    x %= m

    if (n >= m):
        ans = 0
    else:
        ans = (x*fact[n]) % m

    print ans
