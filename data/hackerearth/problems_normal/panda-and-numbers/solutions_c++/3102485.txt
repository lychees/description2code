inline ll scan(){
    char c = getchar_unlocked();
    register ll x = 0;
    bool neg=false;
    if(c=='-'){
        neg=true;
    }
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    if(neg==false) return x;
    else if(neg==true) return -1*x;
}

bool pandatic[1123456];
void checkpandatic(ll n){
	ll ar[10],p,q,x,y,i,j,k,l,m=0;
	p=n;
	while(p){
		ar[m++]=p%10;p/=10;
	}
	for(i=0;i<m;i++){
		if((n-(ar[i]*ar[i]))>0){
			pandatic[n]|=pandatic[n-(ar[i]*ar[i])];
		}
	}
}

int main(){
	reset(pandatic,false);
	register ll i,j,k,t;
	for(i=1;;i++){
		k=pow(i,i);
		if(k>1000000){break;}
		pandatic[k]=true;
	}
	for(i=1;i<=1000000;i++){
		if(pandatic[i]==0){
			checkpandatic(i);
		}
	}
	t=scan();
	while(t--){
		register ll n;
		n=scan();
		pandatic[n]==true?printf("Yes\n"):printf("No\n");
	}
	return 0;
}
