struct point
{
	double x;
	double y;
};
bool cmp(pair<point,point> a, pair<point,point> b)
{
	int u = max(a.ff.y,a.ss.y), v = max(b.ff.y, b.ss.y);
	return u > v;
}
pair<point,point> p[1000];
int main()
{
	int i,j,k,n,X,Y=INF;
	cin >> X >> n;
	fr(i,0,n)
	{
		cin >> p[i].ff.x >> p[i].ff.y >> p[i].ss.x >> p[i].ss.y;
	}
	sort(p, p + n, cmp);
	fr(i,0,n)
	{
		if(X > p[i].ss.x || X < p[i].ff.x) continue;
		double ang1 = atan2(p[i].ss.y-p[i].ff.y,p[i].ss.x-p[i].ff.x);
		double ang2 = atan2(Y - p[i].ff.y, X - p[i].ff.x);
		if(ang2 >= ang1)
		{
			if(ang1 > 0)
				{if(X != p[i].ff.x) X = p[i].ff.x, Y = min(p[i].ff.y, p[i].ss.y);}
			else
				{if(X != p[i].ss.x) X = p[i].ss.x, Y = min(p[i].ff.y, p[i].ss.y);}
		}
	}
	cout << (int)(X+0.5);
	return 0;
}
