#include<bits/stdc++.h>
using namespace std;
int gcd(int a, int b)
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}
bool isprime(int n)
{
    if(n==1)
        return 0;
    for(int i=2;i*i<=n;i++)
    {
        if(n%i==0)
            return 0;
    }
    return 1;
}
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        int n;
        cin >> n;
        int cnt=0;
        for(int i=1;i<=n;i++)
        {
            if(gcd(i,n)==1)
                cnt++;
        }
        if(isprime(cnt))
            cout << "TRUE" << endl;
        else
            cout << "FALSE" << endl;
    }
    return 0;
}
