#include <iostream>
#include<math.h>
using namespace std;
bool prim(int n)
{
	int k=sqrt(n);
	for(int i=2;i<=k;i++)
	{
		if(n%i==0)return false;
	}
	return true;
}
int main()
{
    int n,i,j;
    cin>>n;
    bool prime[n+1];
    for(int i=0;i<=n;i++)
    prime[i]=false;
    int m=sqrt(n);
    for(int i=2;i<=m;i++)
    {   j=2;
    	if(prim[i])
    	{   
    		while(i*j<=n){prime[i*j]=true;j++;}
    	}
    }
    for(i=2;i<=n;i++)
    {
    	if(prime[i]==true)cout<<i<<"\n";
    }
    return 0;
}
