#include<bits/stdc++.h>
using namespace std;
#define as(n,a,b) assert(a<=n && n<=b);
int readint()
{
    int cc = getc(stdin);
    for (;cc < '0' || cc > '9';)
        cc = getc(stdin);
    int ret = 0;
    for (;cc >= '0' && cc <= '9';)
    {
        ret = ret * 10 + cc - '0';
        cc = getc(stdin);
    }
    return ret;
}
int main()
{
    int t = readint();
    as(t, 1, 100);
    while(t--)
    {
        int a = readint();int b = readint();int n = readint();
        as(a, 1, 1000000000);as(b, 1, 1000000000);as(n, 1, 10000);
        int *ar = (int *)malloc(n*sizeof(int));
        for(int i = 0; i < n;++i)
        {
            *(ar + i) = readint(); as(*(ar + i), 1, 100000);
        }
 
        sort(ar, ar + n);
        int i = 0, cnt1 = 0, cnt2 = 0;
        while(a >= *(ar + i) && i < n)
        {
            a -= *(ar + i);++cnt1;
            ++i;
        }
        i = 0;
        while(b >= *(ar + i) && i < n)
        {
            b -= *(ar + i);++cnt2;
            ++i;
        }
        if(cnt1 < cnt2)
            printf("Jhool Won\n");
        else if(cnt2 < cnt1)
            printf("Chef Won\n");
        else
            printf("Tie\n");
    }
    return 0;
}
