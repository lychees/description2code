#include<bits/stdc++.h>
using namespace std;
int main()
{
          int t;
          scanf("%d",&t);
          while(t--)
          {
          char str1[100000],str2[100000];
          int i,l,c=0,j;
          scanf("%s %s",str1,str2);
          l=strlen(str1);
          for(i=0;i<l;i++)
          {
                    j=(str1[i]-'0')+(str2[i]-'0');
                    if(j>9)
                              c++;
          }
          if(c==0)
                    printf("No carry operation\n");
          else if(c==1)
                    printf("1 carry operation\n");
          else
                    printf("%d carry operations\n",c);
          }
          return 0;
}
