#include<string>
#include<iostream>

using namespace std;

bool is_palindrome(string s){
	int i=0,j=s.size()-1;
	while(i<j){
		if(s[i]!=s[j]) return false;
		++i,--j;}
		
	return true;}
	
int count_palin_sub(string s){
	if(s.empty()) return 0;
	int n=s.size();
	int count=0;
	for(int i=0;i<n;++i){
		for(int len=1;len<=n-i;++len){
			string tmp=s.substr(i,len);
			if(is_palindrome(tmp))++count;}
	}
	
	return count;}

int main()
{    string input;
    cin>>input;
    int res=count_palin_sub(input);
    cout <<res<< endl;
    return 0;
}
