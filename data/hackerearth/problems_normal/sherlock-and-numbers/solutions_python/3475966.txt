Tc = int(input())
while Tc:
	N,K,P = [long(i) for i in raw_input().split()]
	Karr = [long(i) for i in raw_input().split()]
	rankP = P #by default, P is the Pth smallest number
	for x in Karr:
		if x <= rankP:
			rankP+=1 #top of P affects rankP,lower change does nothing
	if rankP>N:
		print '-1'
	else:
		print rankP #rankP+1
	Tc -= 1
