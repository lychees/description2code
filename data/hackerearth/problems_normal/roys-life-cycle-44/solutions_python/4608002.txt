def calculate():
	n=int(raw_input())
	# arr=[]
	mgc=gc=mrowC=rowC=0
	for i in range(n):
		arr=[x for x in raw_input()]
		rowC=0
		for j in range(len(arr)):
			if arr[j]=='C':
				rowC=rowC+1
			else:
				mrowC=rowC if rowC>mrowC else mrowC
				mgc=gc+rowC if gc+rowC>mgc else mgc
				gc=rowC=0	
		gc=gc+rowC
		mrowC=rowC if rowC>mrowC else mrowC
	mgc=gc if gc>mgc else mgc
	if mgc<mrowC:
		mgc=mrowC
	print mrowC,mgc
	
calculate()
