#include <iostream>
using namespace std;

#define MAX 11

int N, M, s, e, answer;
int G[MAX][MAX];
int result[MAX];
bool selected[MAX];

void luckyPermutation(int index)
{
	if(index >= N)
	{
		bool lucky = true;
		for(int i = 1; i < N; i++)
			if(G[result[i]][result[i - 1]] == 0)
			{
				lucky = false;
				break;
			}
		if(lucky == true)
			answer++;
	}
	else
	{
		for(int i = 1; i <= N; i++)
			if(selected[i] == false)
			{
				selected[i] = true;
				result[index] = i;
				luckyPermutation(index + 1);
				selected[i] = false;
			}
	}
}

int main()
{
 //   freopen("Micro and Permutations.txt", "r", stdin);
	// setbuf(stdout, NULL);
    
	cin >> N >> M;
	for(int i = 0; i < M; i++)
	{
		cin >> s >> e;
		G[s][e] = 1;
		G[e][s] = 1;
	}

	luckyPermutation(0);
	cout << answer << endl;
    return 0;
}
