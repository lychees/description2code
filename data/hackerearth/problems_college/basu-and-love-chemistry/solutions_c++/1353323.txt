#include<iostream>
#include<cstring>
using namespace std;
int main(){
    char n[200001];
    cin>>n;
    long long ans=n[0]-'0',question=n[0]-'0',mod=1000000007;
    for(int i=0;i<strlen(n)-1;i++){
        question = question*10 + (i+2)*(n[i+1] - '0');
        question = question%mod;
        ans += question;
        ans = ans%mod;
    }
        cout<<ans;
}
