#include <iostream>

using namespace std;

int main(){
int t;
long long n;
cin>>t;
while(t--){
	string st;
	cin>>st;
	cin>>n;
	int cnt=0;
	for(int i=0;i<st.size();i++){
		if(st[i] == '1'){
			cnt=cnt+1;
		}
	}
	long long total =(n *(n+1))/2;
	if((total%2)==0){
		cout<<cnt <<"\n";
	}
	else
	cout<<(st.size()-cnt)<<"\n";
}
}
