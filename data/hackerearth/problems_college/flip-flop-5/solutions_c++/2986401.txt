#include <iostream>
#include<string.h>
using namespace std;

int delete_String(char *s){
	int i=0;
	int count=0;
	
	while(s[i]!='\0'){
		if(s[i]==s[i+1])
		 {
	     	  count++;
         
         }
         else if(s[i]==s[i+2])
              count=count;
          i++;
	}
	return count;
}
	
int main()
{
    int  N;
    char s[100000];
    cin>>N;
    for(int i=0;i<N;i++){
      cin>>s;
      cout<<delete_String(s)<<"\n";
    }
    return 0;
    
}
