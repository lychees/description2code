n = int(raw_input())
arr = [int(w) for w in raw_input().split()]
assert len(arr) == n

top = sorted(arr[:3])

prod = 1
for x in top:
    prod *= x

for i in range(len(top)-1):
    print -1
print prod

for x in arr[len(top):]:
    if x > top[0]:
        prod = (prod / top[0]) * x
        top[0] = x
        for i in xrange(1, len(top)):
            if top[i-1] <= top[i]:
                break
            top[i-1] = top[i]
            top[i] = x
    print prod
