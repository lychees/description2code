#include <iostream>
#include<string>
#include<cstring>
using namespace std;

int main()
{
    int n;
    cin>>n;
    while(n--)
    {
    	string str;
    	cin>>str;
    	bool arr[26];
    	memset(arr,false,26);
    	for(int i=0;i<str.length();i++)
    	{
    		arr[str[i]-'a']=true;
    	}
    	int flag=0;
    	for(int i=0;i<26;i++)
    	{
    		if(arr[i]==false){
    			flag=1;
    			break;
    		}
    	}
    	if(flag==1)cout<<"NO"<<endl;
    	else cout<<"YES"<<endl;
    }
    return 0;
}
