#include <bits/stdc++.h>
using namespace std;

#define MAX 1000000

vector <int> primes;
int a[MAX + 1];

void sieve()
{
	for(int i = 2; i*i <= MAX; i++) {
		if(!a[i]) {
			primes.push_back(i);
			for(int j = i*i; j <= MAX; j += i)
				a[j] = 1;
		}
	}
}

int main()
{
	ios::sync_with_stdio(false);
	sieve();
	vector <int> :: iterator it;
	int n;
	cin >> n;
	while(n--) {
		int x, flag = 0;
		cin >> x;
		for(it = primes.begin(); it != primes.end() && *it < x; it++) {
			if(!a[*it] && !a[x - *it]) {
				cout << "Deepa" << endl;
				flag = 1;
				break;
			}
		}
		if(!flag)
			cout << "Arjit" << endl;
	}
    return 0;
}
