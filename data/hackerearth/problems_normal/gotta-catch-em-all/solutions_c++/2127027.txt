#include <cstdio>
#include <vector>
#include <algorithm>

int main() {
    int k;
    scanf("%d", &k);
    std::vector<int> a;
    int x;
    while (k--) {
    	scanf("%d", &x);
    	a.push_back(x);
    }
    sort(a.begin(), a.end());
    int d, td = a.back()+1, j = 0;
    d = td;
    for (std::vector<int>::reverse_iterator i = a.rbegin() + 1; 
            i != a.rend(); ++i) {
        ++j;
        d -= j;
        if ((*i)+1 > d) {
            td += ((*i)+1-d);
        }
        d = td;
    }
    printf("%d\n", td+1);
    return 0;
}
