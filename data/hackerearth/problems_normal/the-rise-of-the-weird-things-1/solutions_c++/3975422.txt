#include<iostream>
using namespace std;
int main()
{
	int n,i,j,k,sum=0;
	cin>>n;
	int *a=new int[n];
	for(i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	for(j=2;j<=n;j++)
	{
		k=a[j];
		i=j-1;
		while(i>0&&a[i]>k)
		{
			a[i+1]=a[i];
			i=i-1;
		}
		a[i+1]=k;
	}
	for(i=1;i<=n;i++)
	{
		if(a[i]%2==0)
		{
			cout<<a[i]<<" ";
			sum=sum+a[i];
		}
	}
	cout<<sum<<" ";
	sum=0;
	for(i=1;i<=n;i++)
	{
		if(a[i]%2!=0)
		{
			cout<<a[i]<<" ";
			sum=sum+a[i];
		}
	}
	cout<<sum<<endl;
}
