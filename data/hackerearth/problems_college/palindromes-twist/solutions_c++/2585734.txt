#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,t,ca,cb,n,j,flag,ans;
    char s[10001];
    scanf("%d",&t);
    while(t--)
    {
    	flag=1;
    	ans=0;
    	cin>>s;
    	n=strlen(s);
    	cin>>ca>>cb;
    	for(i=0,j=n-1;i<j;i++,j--)
    	{
    		if((s[i]!=s[j])||((s[i]=='/')&&(s[j]=='/')))
    		{
    			if((s[i]!='/')&&(s[j]!='/'))
    			{
    				flag=0;
    				break;
    			}
    			if((s[i]=='/')&&(s[j]=='/'))
    			{
    				if(ca<cb)
    					ans+=2*ca;
    				else
    					ans+=2*cb;
    			}
    			else if((s[i]=='/'))
    			{
    				if(s[j]=='a')
    					ans+=ca;
    				else
    					ans+=cb;
    			}
    			else
    			{
    				if(s[i]=='a')
    					ans+=ca;
    				else
    					ans+=cb;
    			}
    		}
    	}
    	if(flag)
    		printf("%d\n",ans);
    	else
    		printf("-1\n");
    }
    return 0;
}
