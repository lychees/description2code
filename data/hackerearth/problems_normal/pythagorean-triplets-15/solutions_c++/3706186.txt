bool isInt(float num)
{
	return num == int(num);
}

float howManySolution(float m, float n)
{
	float count = 0;
    for(float a = m; a <= n; a++)
   	{
   		int limitB = sqrt(n*n - a*a);
   		for(float b = a+1; b <= limitB; b++)
   		{
   			float tem = sqrt(a*a + b*b);
   			if(isInt(tem) && tem <= n)
   			{    				
   				count++;
    			cout << a << "_" << b << "_" << tem << endl;
    		}
   		}
   	}
   	return count;
}

int howManyWay(float c)
{
	int res = 0;
	for(float i=1; i< c; i++)
	{
		float tem = sqrt(c*c - i*i);
		if(isInt(tem)) res++;
	}
	return res/2;
}

bool nguyenToCungNhau(int m, int n)
{
    while(m != n)
    {
        if(m > n) m = m - n;
        else n = n - m;
    }
    return m == 1;
}

bool bo3PytagoNguyenTo(int m, int n)
{
    bool con1 = nguyenToCungNhau(m, n);
    bool con2 = (m%2 == 0 && n%2 != 0) || (m%2 != 0 && n%2 == 0);
    return con1 && con2;
}


int pytagoTriagle(int num)
{
    int tem = sqrt(num);
    int res = 0;
    for(int n=1; n<=tem; n++)
    {
        for(int m=n+1; m<=tem; m++)
        {
            if(bo3PytagoNguyenTo(m, n))
            {
                int k=1;
                while(k*(m*m + n*n) <= num)
                {
                    res++;
                    k++;
                }
            }
        }
    }
    return res;
}
