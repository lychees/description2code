#include <stdio.h>
using namespace std;
int dp[1030][1030];
int rec(int low ,int high){
	if(low>high)
	   return 0;
	if(low==high)
		return low;
	int mid = (low+high)/2;
	int left = rec(low,mid);
	int right = rec(mid+1,high);
	if(left==0)
	 return right;
	else if(right==0)
	return left;
	else{
		if(dp[right][left]==0)
		return left;
		else
		return right;
	}

}
int main()
{
	int i ,j,n;
    scanf("%d",&n);
    for(i=2;i<=(1<<n);i++){
    	for(j=1;j<i;j++){
    		scanf("%d",&dp[i][j]);
    	}
    }
    printf("%d",rec(1,1<<n));
    return 0;
}
