#include<iostream>
#include<string>
using namespace std;
int main()
{
	  int t;
	  cin >> t;
	  while(t--)
	  {
  		string s,x,xx;
  		cin >> s;
  		int  i ,j ;
  		
  		for(i = 0 ; i < s.size() ; ++i )
  		{
		  	 if(s[i] >= '0' && s[i] <= '9')
			    { 
			    int tmp = (s[i]-'0');
			    xx.clear();
			       for(j = 1 ; j <= tmp ; ++j )
			       xx = xx + x;
			       x = xx;
    			} 
    			else
    		x+=(char)s[i];
    			
	    }
		  cout << x << endl;
  	}
  	   return 0;
}
