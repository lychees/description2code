tc = input()
for _ in range(tc):
    n, m = map(eval, raw_input().split())
    if (m == 0):
        print n
    elif (n == 0 or n == 2):
        if (m % 2 == 0):
            print 0
        else:
            print 1
    else:
        if (m % 2 == 0):
            print 1
        else:
            print 0
