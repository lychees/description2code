#include <iostream>
#include <cstring>
#include <string>
#include <vector>
#define ull unsigned long long 
using namespace std;
 
int main()
{
	ull cnt;
	ull suml,sumr;
	int l,r,j;
	vector<int > v;
	int h_r[1000001];
	int h_k[1000001];
	int cr;
	int ck;
	string s;
	int t;
	
	cin >> t;
	for (int k = 0; k < t; k++) {
		cin >>l >> r;
		cin >> s;
	//	cin >> l >> r;
		for (int i = 0; i <= 1000000; i++) {
			h_r[i] = 0;
			h_k[i] = 0;
		}
		if (s[0] == 'R') {
			h_r[1] = 1;
			cr = 1;
			h_k[0] = 1;
			ck = 0;
		}
		else {
			h_k[1] = 1;
			ck = 1;
			h_r[0] = 1;
			cr = 0;
		}
		for (int i = 1; i < s.length(); i++) {
			if (s[i] == 'R') 
				cr++;
			else
				ck++;
			/*ar[i] = cr;
			ak[i] = ck;*/
			h_r[cr]++;
			h_k[ck]++;
		}
//		cout << cr << endl;
		/*for (int i = 0; i < s.length(); i++) {
			h_r[ar[i]]++;
			h_k[ak[i]]++;
		}*/
		int te;
	//	cin >> l >> r;
		suml = 0;
		sumr = 0;
		cnt = 0;
		te = l;
//		cout << h_r[te] << endl;
	//	cout << h_r[te-1] << endl;
 
		if(h_r[l] != 0 && te != 0) {
			while (h_r[te] != 0 && te <= s.length()) {
				suml = suml + h_r[te]*h_r[te - l];
				te++;
			//	cout << suml << endl;
			}
			suml = suml + h_r[l];
		}
		else if(te == 0) {
			for (int i = 0; i < s.length(); i++) {
				if (s[i] == 'K') {
					v.push_back(i);
				}
			}
			if (v.size() != 0) {
			for (int i = 0; i < v.size()-1; i++) {
				if (v[i+1] - v[i] == 1) {
					cnt++;
				}
				else {
					suml = suml + (((cnt+1)*(cnt+2))/2);
					cnt = 0;
				}
			}
			}
			if (v.size() == 0)
				suml = 0;
			else {
				suml = suml + (((cnt+1)*(cnt+2))/2);
			}
		}
		v.clear();
		cnt = 0;
	//	cout << h_k[te - 1] << endl;
		 te = r;
	//	cout << h_k[0] << endl;
		if (h_k[r] != 0 && te != 0) {
			while (h_k[te] != 0 && te <= s.length()) {
				sumr = sumr + h_k[te]*h_k[te - r];
				te++;
			}
			sumr = sumr + h_k[r];
		}
		else if(te == 0){
			for (int i = 0; i < s.length(); i++) {
				if (s[i] == 'R') {
					v.push_back(i);
				}
			}
		//	cout << v.size() << endl;
			if (v.size() != 0) {
			for (int i = 0; i < v.size()-1; i++) {
				if (v[i+1] - v[i] == 1) {
					cnt++;
				}
				else {
					sumr = sumr + (((cnt+1)*(cnt+2))/2);
					cnt = 0;
				}
			}
			}
			if (v.size() == 0)
				sumr = 0;
			else {
				sumr = sumr + (((cnt+1)*(cnt+2))/2);
			}
		}
		v.clear();
		cout << suml << " " << sumr << endl;
	}
}
