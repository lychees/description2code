template<typename T> T gcd(T a, T b) {if(!b) return a;return gcd(b, a % b);}
template<typename T> T lcm(T a, T b) {return a * b / gcd(a, b);}
ll a[200002];
ll fen[200002];
int n,q;
bool roundnum(ll num){
	if(num<0) return false;
	int firstdig=0,lastdig=num%10;
	while(num>0){
        firstdig=num%10;
		num/=10;
	}
	if(firstdig==lastdig) return true;
	else return false;
}
void update(int p,int x){
	for(;p<=n;p+=(p & (-p))){
		fen[p]+=x;
	}
}
ll query(int p){
	ll sum=0;
	for(;p>0;p-=(p & (-p)))
	   sum+=fen[p];
	return sum;
}
int main(){
  //ios_base::sync_with_stdio(false); cin.tie(NULL);

  cin>>n>>q;
  For(i,n,1){
    cin>>a[i];
    //cout<<a[i]<<endl;
    if(roundnum(a[i])){
  		update(i,1);
    }
  }
    while(q--){
    	int ch;
    	cin>>ch;
    	if(ch==1){
    		int l,r;
    		cin>>l>>r;
    		int ter=query(r)-query(l-1);
    		cout<<ter<<endl;
    	}
    	else{
    		int pos; ll x;
    		cin>>pos>>x;
    	    if(roundnum(a[pos])==true){
    			update(pos,-1);
    		}
    		a[pos]=x;
    		if(roundnum(x)==true){
    			update(pos,1);
    		}
    	}
    }
   
return 0;
}
