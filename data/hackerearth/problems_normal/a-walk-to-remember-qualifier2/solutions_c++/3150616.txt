using namespace std;
typedef unsigned long long int ull;
typedef long long int ll;
typedef vector<vector<ll> > mat;
#define N 100005

int cycle;
int fin[2*N];
bool visit[2*N],ans[2*N];
int t;
mat P,Q;

void dfs1(int ver){
	if(visit[ver]){
		return;
	}
	visit[ver] = true;
	int sz = P[ver].size();
	rep(i,sz){
		dfs1(P[ver][i]);
	}
	fin[++t] = ver;
}

void dfs2(int ver){
	if(visit[ver]){
		return;
	}
	++cycle;
	visit[ver] = ans[ver] = 1;
	int sz = Q[ver].size();
	rep(i,sz){
		dfs2(Q[ver][i]);
	}
}

int main(){
	boost;
	int n,m;
	cin >> n >> m;	
	P = Q = mat(n+1);
	while(m--){
		int u1,u2;
		cin >> u1 >> u2;
		P[u1].pb(u2);
		Q[u2].pb(u1);
	}
	forp(i,1,n){
		if(!visit[i]){
			dfs1(i);
		}
	}
	
	mem(visit, 0);
	
	forn(i,n,1){
		if(!visit[fin[i]]){
			cycle = 0;
			dfs2(fin[i]);
			if(cycle == 1){
				ans[fin[i]] = 0;
			}
		}
	}
	forp(i,1,n){
		cout << ans[i] << " ";
	}
	return 0;
}
