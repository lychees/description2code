def power(a,b,M):
        
        if b==0: return 1
        if b==1: return a
        halfn=power(a,b/2,M)
        if(b%2==0):
        		return ( halfn * halfn ) % M;
        else:
            	return ( ( ( halfn * halfn ) % M ) * a ) % M;
