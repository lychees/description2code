#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long

int dp[1000001][11];


int main()
{
   int t;
   cin>>t;

   for(int i=0;i<10;i+=1)
        dp[1][i]=1;

    //for(int i=1;i<n;i+=1){
        //dp[i][0]=1
    //}

   for(int i=2;i<1000001;i+=1){
       for(int j=0;j<=9;j+=1){
       		dp[i][j]+=dp[i-1][j+1];
           if(j!=0)
           	dp[i][j]+=dp[i-1][j-1];
           dp[i][j]%=mod;
       }
   }


   while(t--){
       int n;
       cin>>n;
       if(n==1) {
           cout<<"10"<<endl;
           continue;
       }
       ll ans=0;
       for(int i=1;i<=9;i+=1){
           ans+=dp[n][i];
           ans%=mod;
       }
       cout<<ans<<endl;
   }

   return 0;
}
