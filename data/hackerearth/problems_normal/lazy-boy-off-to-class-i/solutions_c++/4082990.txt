#include <iostream>
#include<vector>
#include<algorithm>
#include<math.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;cin>>t;
    while(t--)
    {
    	vector<int> v;int nc=0;
    	for(int i=0;i<7;i++)
    	{
    		int n;
    		cin>>n;
    		nc=nc+n;
    		int x,y;
    		for(int j=0;j<n;j++)
    		{
    			cin>>x>>y;
    			v.push_back(y-x);
    		}
    	}
    	sort(v.begin(),v.end());
    	int x=(int)ceil(0.75*nc);
    	int max=24*60*60*7;
    	for(int i=0;i<x;i++)
    	{
    		max=max-v[i];
    	}
    	cout<<max<<"\n";
    }
    return 0;
}
