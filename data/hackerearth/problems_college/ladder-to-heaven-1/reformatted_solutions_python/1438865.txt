for _ in xrange(input()):
    N = input()

    sumBlue = 2
    a = 1
    b = 2
    c = a + b

    if c >= N:
        print sumBlue
    else:
        while c <= N:
            a = b
            b = c
            c = a + b
            if c % 2 == 0 and c <= N:
                sumBlue += c
    print sumBlue
