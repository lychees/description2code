#include <iostream>
#include<string.h>
#include<stdio.h>
#include<stdlib.h>
#define ll long long
using namespace std;

int main() 
{
	int t;
    cin>>t;
   while(t--)
  {
    	char s1[35];
      scanf("%s",s1);
    	int l=strlen(s1);
 ll a[l];
	for(int i=0;i<l;i++)
{
	a[i]=s1[i]-'0';
}
    for(int j=l-2;j>=1;j=j-2)
    {
     
      if(s1[j]=='+')
         a[j-1]=a[j+1]+(a[j-1]);
      else if(s1[j]=='-')
         a[j-1]=(a[j+1])-a[j-1];
      else if(s1[j]=='*')
         a[j-1]=a[j+1]*(a[j-1]);

     
    }
    printf("%lld\n",a[0]);
   }     

	return 0;}
