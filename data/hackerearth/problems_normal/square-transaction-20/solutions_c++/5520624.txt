#include <stdio.h>
int a[100000],sum[100000],T,q;
 
int binarySearch(int l,int r,int element)
{
	if(l<=r)
	{
		int mid = (l+r)/2;
		if(sum[mid]>=element && sum[mid-1]<element)
			return mid;
		if(sum[mid]>element)
			binarySearch(l,mid-1,element);
		if(sum[mid]<element)
			binarySearch(mid+1,r,element);
	}
	else
		return -1;
}
 
int main()
{
	int t_sum = 0;
	scanf("%d",&T);
	for(int i=0;i<T;i++)
	{
		scanf("%d",&a[i]);
		t_sum += a[i];
		sum[i] = t_sum;
	}
	scanf("%d",&q);
	for(int i=0;i<q;i++)
	{
		int x;
		scanf("%d",&x);
		int ans = binarySearch(0,T-1,x);
		if(ans!=-1)
			printf("%d\n",ans+1);
		else
			printf("-1\n");
	}
    return 0;
}
