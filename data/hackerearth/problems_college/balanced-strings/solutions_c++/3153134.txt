#include<iostream>
main()
{
	int tc,j;
	std::cin>>tc;
	for(j=0;j<tc;j++)
	{
		std::string s;
		int i;
		std::cin>>s;
		int hash[200]={0};
		for(i=0;i<s.length();i++)
		{
		
			hash[s[i]]++;
		}
		for(i=97;i<122;i++)
		{
			if(hash[i]!=hash[i+1])
			break;
		}
		if(i==122)
		std::cout<<"Yes"<<std::endl;
		else
		std::cout<<"No"<<std::endl;
	}
	return 0;
}
