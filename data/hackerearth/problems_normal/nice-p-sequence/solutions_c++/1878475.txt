#include <iostream>
using namespace std;

bool gcdEx(long long a, long long b, long long &x, long long &y) {
	long long x1 = 1, y1 = 0;
	long long x2 = 0, y2 = 1;
	while(b) {
		long long v = a / b;
		x1 -= v * x2;
		y1 -= v * y2;
		a -= v * b;
		swap(a,b);
		swap(x1,x2);
		swap(y1,y2);		
	}
	x = x1;
	y = y1;
	return a == 1;
}

			if (!gcdEx(a,p,x,y)) {

			x = (x % p + p) % p;
