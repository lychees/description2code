#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t,n,i,j,k,p[50];
    char a[50][50];
    cin>>t;
    for(i=0;i<t;i++)
    {	cin>>n;
    	for(j=0;j<n;j++)
    		for(k=0;k<n;k++)
    		{	cin>>a[j][k];
    			if(a[j][k]=='*')
    			p[i]=fabs((n/2)-k)+fabs((n/2)-j);
    		}
    }
    for(i=0;i<t;i++)
    	cout<<p[i]<<endl;
    return 0;
}
