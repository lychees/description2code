def newswap(y, m):
    sw = y[m]
    for x in range(m+1, len(y)):
        y[x-1] = y[x]
    y[len(y)-1] = sw


norl = True
num = 0
var = 0
inr = int(raw_input())
norm = raw_input().split()
idl = raw_input().split()
while (norl):
    if (norm == idl):
        norl = False
    elif (norm[var] != idl[var]):
        newswap(norm, var)
        num = num+1
    elif (idl[var] == norm[var]):
        var = var+1
        num = num+1
num = num+len(norm)-var
