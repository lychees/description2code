def isprime(n):
    x=int(pow(n,0.5))
    flag=0
    for i in range(2,x+1):
        if n%i==0:
            flag=1
            return 0
    return 1
a=[0]*505
a[0]=0
a[1]=0
a[2]=1
cnt=1
for i in range(3,505):
    if isprime(i):
        cnt=cnt*2
        a[i]=a[i-1]+cnt
    else:
        a[i]=a[i-1]+cnt
n=input()
print a[n]
