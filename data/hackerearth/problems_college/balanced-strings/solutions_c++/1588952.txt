#include <iostream>
#include <string>
#include <string.h>

using std::cout;
using std::cin;
using std::string;

int main()
{
	int T;
	string s;
	cin>>T;
	while(T)
	{
		cin>>s;
		bool isequal;
		long long int search;
		int n=s.length();
		int alpha[26];
		memset(alpha,0,sizeof(alpha));
		for(int i=0;i<n;i++)
		{
			   alpha[s[i]-'a']++;
		}
		for(int res=0;res<26;res++)
		{
			   if(res==0)
			   {
			   	  search=alpha[res];
			   }
			   else
			   {
			   	   if(search==alpha[res])
			   	   {
			   	   	   isequal=true;
			   	   }
			   	   else
			   	   {
			   	   	   isequal=false;
			   	   	   break;
			   	   }
			   }
		}
		if(isequal==true)
		{
			cout<<"Yes"<<"\n";
		}
		else
		{
			cout<<"No"<<"\n";
		}
	 T--;
		  
	}
}
