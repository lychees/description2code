#include <iostream>
using namespace std;

int bacteria(int N)
{
	if(N==0)
		return 1;
	if(N==1)
		return 1;
	if(N==2)
		return 2;
	if(N==3)
		return 2;

	int zero,one,two;
	zero=1;
	one=1;
	two=0;
	for(int i=4;i<=N;i++)
	{
		int temp=one+two;
		two=one;
		one=zero;
		zero=temp;
	}
	return zero+one+two;
}
int main()
{
	int N;
	cin>>N;
	cout<<bacteria(N)<<endl;
	return 0;
}
