#include<iostream>
#include <string>
#include <vector>

using namespace std;

int dp[21][3] = {0};
int **cost;
int n;
int findOptimum(int house, int prevColor){
    if(house == n){
        return 0;
    }
    if(dp[house][prevColor] != 0){
        return dp[house][prevColor];
    }

    int ans = 1000000;
    for(int color = 0 ; color < 3;color++){ // for each color
        if(color!= prevColor){
            ans = min(ans,findOptimum(house+1,color)+cost[house][color]);
        }
    }
    return ans;
}

int main()
{
    int t;
    cin>>t;
    while(t--){
        cin>>n;
        cost = new int*[n];
        for(int i = 0 ;i<n;i++){
            cost[i] = new int[3];
            cin>>cost[i][0]>>cost[i][1]>>cost[i][2];
        }
        cout<<findOptimum(0,5)<<endl; // 5 = random color
    }

}
