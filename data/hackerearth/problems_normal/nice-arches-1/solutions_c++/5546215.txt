stack<char>st;

int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int n, ans = 0;
	string s;
	cin>>n;
	while(n--){
		cin>>s;
		for(int i=0; i<s.sz; i++){
			if(st.sz && st.top()==s[i])
				st.pop();
			else
				st.push(s[i]);
		}
		if(st.sz){
			while(st.sz)
				st.pop();
		}
		else
			ans++;
	}
	cout<<ans<<endl;


return 0;
}
