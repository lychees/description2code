#include<bits/stdc++.h>
using namespace std;
#define ll long long int
unordered_map<ll,ll>mp;
ll a[1000006],b[1000006],bit[1000006],n;
ll qry(ll idx)
{
     ll c=0;
    while(idx>0)
    {
        c+=bit[idx];
        idx-=(idx&(-idx));
    }
    return c;
}
void update(ll idx,ll val)
{
    while(idx<1000005)
    {
        bit[idx]+=val;
        idx+=(idx&(-idx));
    }
}
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(false);
    ll i,j;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>a[i];
        b[i]=a[i];
    }
    j=1;
    sort(b,b+n);
    mp[b[0]]=j;
    for(i=1;i<n;i++)
    {
        if(b[i]==b[i-1])
        continue;
        else
        {
            j++;
            mp[b[i]]=j;
        }
    }
    for(i=0;i<n;i++)
    {
        j=mp[a[i]];
        //cout<<j<<"   ";
        cout<<qry(j-1)<<" ";
        update(j,1);
    }
    return 0;
}
