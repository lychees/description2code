#include <iostream>
using namespace std;

int main()
{
    int t,n,k,count,i,a;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	count =0;
    	for(i=1;i<=n;i++)
    	{
    		cin>>a;
    		if(a<=0)
    			count++;
    	}
    	if(count>=k)
    		cout<<"NO\n";
    	else
    		cout<<"YES\n";
    }
    return 0;
}
