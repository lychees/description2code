#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t,i;
    cin>>t;
    while(t--){
    	string s;
    	int c[26]={0};
    	cin>>s;
    	for(i=0;i<s.size();i++){
    		c[s[i]-'a']++;
    	}
    	
    	int max = c[0];
    	for(i=1;i<26;i++){
    		if(max<c[i]){
    			max = c[i];
    		}
    	}
    	cout<<max<<endl;
    }
    return 0;
}
