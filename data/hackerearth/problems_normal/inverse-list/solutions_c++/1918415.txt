#include<bits/stdc++.h>
using namespace std;
int main()
{
    vector<int> v1;
    int t,n,i,j,x,y,v2[100000],flag;
    cin>>t;
    while(t--)
    {
        cin>>n;
        flag=0;
        for(i=0;i<n;i++)
        {
            cin>>x;
            v1.push_back(x);
        }
        for(i=0;i<n;i++)
        {
            x=v1.at(i);
            y=v1.at(x-1);
            v2[y-1]=x;
        }
        for(i=0;i<n;i++)
        {
            if(v1.at(i)!=v2[i])
            {
                flag=1;
                break;
            }
        }
        if(flag==1)
            cout<<"not inverse\n";
        else
            cout<<"inverse\n";

        v1.clear();

    }
}
