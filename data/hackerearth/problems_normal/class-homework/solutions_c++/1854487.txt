#include<cstring>
using namespace std;
#define S(t) scanf("%d",&t)
#define S2(t,q) scanf("%d%d",&t,&q)
#define P(a) printf("%d\n",a)
#define For(i,a,b) for(i=a;i<b;i++)
#define Fore(i,a,b) for(i=a;i<=b;i++)
#define pb push_back
#define all(v) v.begin,v.end
#define mp make_pair
#define Fill(v,a) memset(v,a,sizeof(v))
typedef long long int ll;
typedef pair<int ,int >pi;
int main()
{
	int i,j,q,t;
	S(t);
	while(t--)
	{
		char a[25];
		scanf("%s",a);
		ll fromto[25][25];
		int n=strlen(a);
		if(n<4)
		{
			printf("unlucky\n");
			continue;
		}
		Fill(fromto,-1);
		For(i,0,n)
		{
			ll clc=0;
			if(a[i]=='0')
			{
			fromto[i][i]=0;
			continue;
			}
			For(j,i,n)
			{
			  clc=clc*10+(a[j]-'0');
			  if(clc>1000000000000)
			  break;
			  fromto[i][j]=clc;
			}
		}
		ll dp[5][21];
		
		Fill(dp,-1);
	for(i=0;i<n&&fromto[0][i]!=-1;i++)
		dp[1][i]=fromto[0][i];
		for(i=2;i<=4;i++)
		{
			For(j,0,n)
			For(q,j+1,n)
			{
				if(fromto[j+1][q]!=-1&&dp[i-1][j]!=-1)
				dp[i][q]=max(dp[i][q],fromto[j+1][q]+dp[i-1][j]);
			}
		}
		ll ans=dp[4][n-1];
		if(ans==-1)
		printf("unlucky\n");
		else
		printf("%lld\n",ans);
	}
	return 0;
}
