#include <iostream>
#include <bits/stdc++.h>
using namespace std;
bool comp(const pair<int, int>& a,const pair<int, int>& b)
{
	if(a.first<b.first) return true;
	else if(a.first>b.first) return false;
	
	return a.second<b.second;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int m,n,g,h;
    	cin>>m>>n;
    	vector< pair<int, int> > v,v1;
    	for(int i=0;i<m;i++)
    	{
    		int a,b;
    		cin>>a>>b;
    		v.push_back(make_pair(a,b));
    	}
    	
    	for(int i=0;i<n;i++)
    	{
    		int a,b;
    		cin>>a>>b;
    		v1.push_back(make_pair(a,b));
    	}
    	
    	cin>>g>>h;
    	
    	sort(v.begin(),v.end());
    	sort(v1.begin(),v1.end());
    	
    	int c1=0,c2=0;
    	unordered_map<int, int> ma;
    	
    	
    	for(int i=0;i<n;i++)
    	{
    		if(i==0)
    		{
    			ma[v1[0].first]=0;
    		}
    		else if(v1[0].first!=v1[i].first && ma[v1[i].first]==0)
    		{
    			ma[v1[i].first]=i;
    		}
    	}
    	
    	for(int i=0;i<m;i++)
    	{
    			int a1=v[i].first,b1=v[i].second;
    			int pos=ma[a1];
    			for(int j=ma[a1];j<n;j++)
    			{
    				int a2=v1[j].first;
    				int b2=v1[j].second;
    				
    				if(a1==a2)
    				{
    					if(b2!=-9999)
    					pos=j;
    					
    					if(b1==b2)
    					{
    						c1++;
    						c2++;
    						v1[j].second=-9999;
    						break;
    					}
    					
    					if(b1>b2)
    					break;
    				}
    				else
    				{
    					break;
    				}
    			}
    			if(v1[pos].first==a1 && v1[pos].second!=-9999)
    			{
    				c1++;
    				v1[pos].second=-9999;
    			}
    			
    			if(c1>=g && c2>=h)
    			break;
    	}
    	if(c1>=g && c2>=h)
    	cout<<"Great"<<endl;
    	else if(c1>=g)
    	cout<<"Good"<<endl;
    	else
    	cout<<":("<<endl;
    	
    	//cout<<c1<<" "<<c2<<endl;
    }
    return 0;
}
