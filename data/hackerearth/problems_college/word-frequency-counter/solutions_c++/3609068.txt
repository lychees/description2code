#include<bits/stdc++.h>
using namespace std;
long  n,i,j,m,x;
string s,p;
map<string,long>mp,ck;
int main()
{
   getline(cin,s);
   for(i=0;s[i];i++)
   {
       if(s[i]!=' '){
            if(s[i]>'Z')
            s[i]-=32;
        p+=s[i];
       }
       else
       {
           mp[p]++;
           p.clear();
       }
   }
   mp[p]++;
   p.clear();
   for(i=0;s[i];i++)
   {
       if(s[i]!=' ')
        p+=s[i];
       else if(ck[p]==0)
       {
           cout<<p<<" "<<mp[p]<<endl;
           ck[p]=1;
           p.clear();
       }
       else
        p.clear();
   }
   if(ck[p]==0)
    cout<<p<<" "<<mp[p]<<endl;
}
