def modBinSearch(array, start, end, value):
    if start <= end:
        # print 'enters'
        middle = (start+end)/2
        if array[middle] > value:
            return modBinSearch(array, middle+1, end, value)
        elif array[middle] < value:
            return modBinSearch(array, start, middle-1, value)
        else:
            while (middle == 0 or array[middle-1] == value):
                # print 'happens'
                middle -= 1

            return max(middle, 0)

    else:
        return -1
