//GCD Game
#include<bits/stdc++.h>
#define MAX 1005 

using namespace std;

int dp[2][MAX][MAX];
/*
a-->0
and b-->1
*/
int solve(int turn,int a,int b){
	
	if(a==1&&b==1) return 2;
	if(a==1) return 1;
	else if(b==1) return 0;
	
	if(dp[turn][a][b]!=-1) return dp[turn][a][b];
	
	int gcd=__gcd(a,b),op1=!turn,op2=!turn,winner=!turn;
	
	if(gcd>1){
		
		if(turn==1) 
		  op1=solve(!turn,a-1,b);
		  
		else op1=solve(!turn,a,b-1); 
		
		if(turn==1)
		op2=solve(!turn,a/gcd,b);
		
		else op2=solve(!turn,a,b/gcd);
		
		if(op1==turn||op2==turn)
		  winner=turn;
		
		if(winner!=turn&&(op1==2||op2==2))
		  winner=2;
		
	}
	
	else{
		
		if(turn==1) 
		  winner=solve(!turn,a-1,b);
		  
		else winner=solve(!turn,a,b-1);  
		
	}
	
	return dp[turn][a][b]=winner;

}

int main(void){
	
	memset(dp,-1,sizeof dp);
	solve(0,MAX-1,MAX-1);
	
	int testCases,a,b;
	
	scanf("%d",&testCases);	
	while(testCases--){
		
		scanf("%d%d",&a,&b);
		
		int ans=solve(0,a,b);
		
		if(ans==0)
		  printf("Arjit\n");
		  
		else if(ans==1) printf("Chandu Don\n");	
		
		else printf("Draw\n");
		
	}

	return 0;
	
}
