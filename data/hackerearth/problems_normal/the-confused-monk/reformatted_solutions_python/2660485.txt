def gcd(u, v):
    if u == 0:
        return v
    return gcd(v % u, u)


def gcds(k):
    d = 0
    for i in k:
        d = gcd(d, i)
    return d


def main():
    n = int(raw_input())
    a = map(int, raw_input().split())
    g = gcds(a)
    f = reduce(lambda x, y: x * y, a)
    print(pow(f, g, 10 ** 9 + 7))


if __name__ == '__main__':
    main()
