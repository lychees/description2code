#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;
int gcd( int a, int b)
{
	int res ;
	while( a!=0 && b != 0)
		{
			res = a%b;
			a = b;
			b = res;
		}
	if( a ==0)
		return b;
	else 
		return a;
		
}
int main()
{

int T, N,i,j,temp,k, result;
vector<int> group;
cin>>T;

	for( i=0; i<T; i++)
		{
			cin>>N;
			for( j=0; j<N; j++)
			{
				cin>>temp;
				group.push_back(temp);
			}
			result = group[0];
			for( j=1; j<N; j++)
			{
				result = result > group[j] ?gcd(result , group[j]):gcd( group[j], result);
			}
			cout<<result<<"\n";
			group.clear();
		}
return 0;
}
