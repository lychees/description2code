using namespace std;
#define MAXI 100010
long long MOD = 1000000007;
vector<long long> ins;
int main(){
	int T;
	for (cin >> T; T; T--) {
		long long N; cin >> N;
		ins.clear();
		for (int i = 0; i < N; i++) {
			long long a; cin >> a;
			ins.push_back(a);
		}
		sort(ins.begin(), ins.end());
		long long A = 0, B = 0;
		for (long long i = 0, j = N; i < N; i++, j--) {
			A += (j*ins[j-1])%MOD;
			B += (j*ins[i])%MOD;
			A %= MOD;
			B %= MOD;
		}
		//cout << A << " "<< B <<"\n";
		long long ans = ((A - B + MOD)*ins[N-1])%MOD;
		cout << ans <<"\n";
	}
	return 0;
}
