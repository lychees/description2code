#include <iostream>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while (t--) {
    	int n;
    	scanf("%d",&n);
    	int e = 0;
    	int o = 0;
    	int a[n];
    	for (int i = 0; i < n; i++) {
    		scanf("%d",&a[i]);
    	}
    	if (a[n-1] % 2 == 0)
    		e++;
    	else
    		o++;
    	long long int ans = 0;
    	for (int i = n-2; i >= 0; i--) {
    		if (a[i] % 2 == 0) {
    			ans += o;
    			e++;
    		}
    		else {
    			ans += e;
    			o++;
    		}
    	}
    	printf("%lld\n",ans);
    }
    return 0;
}
