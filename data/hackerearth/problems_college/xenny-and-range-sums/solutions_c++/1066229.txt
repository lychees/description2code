
#include <stdio.h>

int main()
{
	long long int l,r,i,n,s,q,num;
    scanf("%lld%lld",&n,&q);
    long long int arr[n+1];
    arr[0]=0;
    s=0;
    for(i=1;i<=n;i++)
    {
    	scanf("%lld",&num);
    	s=s+num;
    	arr[i]=s;
    }
    while(q--)
    {
    	scanf("%lld%lld",&l,&r);
    	printf("%lld ",arr[r]-arr[l-1]);
    	printf("%lld\n",arr[r]-arr[r-1]-arr[r-1]+arr[l-1]);

    }
}
