n = int(raw_input())
a = raw_input()
a = map(int, a.split(' '))
c = 0
ma = 0
for i in range(0, len(a)-1):
    if abs(a[i]-a[i+1]) == 1:
        c += 1
        if c > ma:
            ma = c
    else:
        c = 0
