#include <iostream>
using namespace std;

int main()
{
    long n,i,j,t;
    cin>>n;
    long a[n];
    for(i=0;i<n;i++)
    	cin>>a[i];
    for(i=0;i<n;i++)
    	for(j=i;j<n;j++)
    		if(a[j]<a[i])
    		{
    			t=a[j];
    			a[j]=a[i];
    			a[i]=t;
    		}
    for(i=0;i<n;i++)
    	cout<<a[i]<<endl;
    return 0;
}
