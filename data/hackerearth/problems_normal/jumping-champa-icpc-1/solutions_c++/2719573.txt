#include <bits/stdc++.h>
using namespace std;

int main() {
	long long int t,n,i,j,q,sum,h;
	cin>>t;
	while(t--){
		sum=0;
		cin>>n>>q;
		vector <long long int> a;
		for(i=0;i<n;i++)
		{
			cin>>j;
			a.push_back(j);
		}
		sort(a.begin(),a.end());
		for(i=0;i<n-1;i++)
		{
			sum+=a[i+1]-a[i];
		}
		h=q*sum;
	cout<<h<<"\n";	
	}
}
