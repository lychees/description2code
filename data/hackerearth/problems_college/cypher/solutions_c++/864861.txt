#include<cstdio>
#include<iostream>

using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    cin.ignore();
    while(t--)
    {
        string s;
        getline(cin,s);
        int i,len,k=0,j;
        len=s.length();
        for(i=0;i<len;i++)
        {
            if(s[i]+k>122)
            {
                j=k;
                j=s[i]+k-122;
                printf("%c",(char)(96+j));
                k++;
                continue;
            }
            printf("%c",(char)(s[i]+k));
            k++;
        }
        printf("\n");
    }
    return 0;
}
