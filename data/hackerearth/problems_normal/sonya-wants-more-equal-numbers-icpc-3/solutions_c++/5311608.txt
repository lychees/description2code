#include <iostream>
#include <vector>
using namespace std;
int dp[105][100001];
int main()
{
    vector <int> ans;
    int n,ptr;
    int a[105],sum[105];
    cin>>n;
    sum[0]=0;
    for(int i=1;i<=n;i++)
    {
        cin>>a[i];
        sum[i]=sum[i-1]+a[i];
    }
    for(int i=1;i<=100000;i++)
    {
        dp[n+1][i]=n+1;
    }
    for(int s=1;s<=100000;s++)
    {
        ptr=n;
        for(int i=n;i>=0;i--)
        {
            dp[i][s]=dp[i+1][s];
            while(sum[ptr]-sum[i-1]>s)
                ptr--;
            if(sum[ptr]-sum[i-1]==s)
                dp[i][s]=ptr;
        }
    }
    int cnt,maxcnt=0;
    for(int s=1;s<=100000;s++)
    {
        cnt=0;
        for(int i=1;i<=n;i++)
        {
            if(dp[i][s]!=n+1)
                cnt++;
            i=dp[i][s];
        }
        if(cnt>maxcnt)
        {
            ans.clear();
            maxcnt=cnt;
            ans.push_back(s);
        }
        else if(cnt==maxcnt)
        {
            ans.push_back(s);
        }
    }
    cout<<maxcnt<<endl;
    for(int i=0;i<ans.size();i++)
    {
        cout<<ans.at(i)<<" ";
    }
    cout<<endl;
    return 0;
}
