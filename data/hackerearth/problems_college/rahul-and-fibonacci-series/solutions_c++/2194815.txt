#include <iostream>
#include<stdio.h>
#include<string.h>
#define m 1000000007
using namespace std;
unsigned long long fib(long long n){
	if(n<=2)return 1;
	unsigned long long mat[2][2],res[2][2],temp[2][2],temp2[2][2];
	res[0][0]=1;res[0][1]=0;res[1][0]=0;res[1][1]=1;
	temp[0][0]=1;mat[0][0]=0;temp[0][1]=0;mat[0][1]=1;
	temp[1][0]=0;mat[1][0]=1;temp[1][1]=1;mat[1][1]=1;
	int i,j,k;
	//printf("hello   \n");
	while(n){
		if(n&1){memset(temp,0,sizeof temp);
			for(i=0;i<2;i++)
			for(j=0;j<2;j++)
			for(k=0;k<2;k++){
				temp[i][j]+=(mat[i][k]*res[k][j])%m;
			}
				//for(i=0;i<2;i++)
				//for(j=0;j<2;j++)
				res[i][j]=temp[i][j];
		}memset(temp2,0,sizeof temp2);
		for(i=0;i<2;i++)
		for(j=0;j<2;j++)
		for(k=0;k<2;k++){
			temp2[i][j]+=(mat[i][k]*mat[k][j])%m;
			}
		for(i=0;i<2;i++)
		for(j=0;j<2;j++){
			res[i][j]=temp[i][j];
			mat[i][j]=temp2[i][j];
			}
		n=n/2;
		
	}
	return res[0][1]%m;
}
unsigned long long nonfib(int n){
	int a=1,b=2,c=3;
	while(n>0){
		a=b;  b=c;  c=a+b;
		n-=c-b-1;
	}
	n+=c-b-1;
	return b+n;
}
unsigned long long power(long long x,long long y){
unsigned long long res=1;
	while(y){
		if(y&1)res=((res%m)*(x%m))%m;
		x=((x%m)*(x%m))%m;
		y/=2;
	}
	return res;
}
	

int main()
{
	int n;scanf("%lld",&n);
	while(n--){//printf("hello\n");
		long long  num;scanf("%lld",&num);
		long long x=fib(num);
		long long y=nonfib(num);
	//	printf("%d  %lld %d \n",x,y,m);
		printf("%llu\n",power(y,x));
	}
    return 0;
}
