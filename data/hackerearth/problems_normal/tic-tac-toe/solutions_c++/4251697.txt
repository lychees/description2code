#include <iostream>
using namespace std;

#define M 1000000007

long long modexp(long long x){
	int y = M-2;
	long long ans = 1;
	while(y){
		if(y&1){
			ans = (ans*x)%M;
		}
		x = (x*x)%M;
		y >>= 1;
	}
	return ans;
}

int main()
{
    long long t,n;
    cin >> t;
    while(t--){
    	cin >> n;
    	long long maxx,minx;
    	maxx = ((n-1)*(n))%M;
    	maxx = (maxx*modexp(6))%M;
    	maxx = (maxx*(2*n-1))%M;
    	minx = (n-1)/2;
    	minx = (minx*minx)%M;
    	minx = (n*minx)%M;
    	cout << minx << " " << maxx << endl;
    }
    return 0;
}
