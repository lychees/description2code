#include <iostream>
using namespace std;
#include<algorithm>
int main()
{
    int k;
    cin>>k;
    int ar[k];
    for(int i=0;i<k;i++)
    cin>>ar[i];
    
    sort(ar,ar+k);
   
    int mx=0;
    int cnt=0;
    for(int j=k-1;j>=0;j--){
    	cnt++;
    	if(cnt+ar[j]>mx)
    	mx=cnt+ar[j];
   }
   cout<<mx+1<<endl;
    
    return 0;
}
