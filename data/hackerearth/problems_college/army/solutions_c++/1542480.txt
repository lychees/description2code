#include <iostream>
using namespace std;

int dp[1000007];

void pre() {
	int n;
	for(int i=1;i<=1000002;i++) {
		n=i;
		int val=0;
		int rem;
		while(n) {
			rem = n%10;
			int is_already_set = (val>>rem)&1;
			if(is_already_set) n/=10;
			else val+=(1<<rem), n/=10;
		}
		dp[i]=val;
	}
} 

int main() {
	pre();
	int n;cin>>n;
	int q;cin>>q;
	while(q--) {
		int m;
		cin>>m;
		int mask=0;
		int x;
		for(int i=0;i<m;i++) {
			cin>>x;
			mask += (1<<x);
		}
		//cout<<"mask = "<<mask<<"\n";
		int ret=0;
		for(int i=1;i<=n;i++) {
			//cout<<"dp["<<i<<"] = "<<dp[i]<<" - ";
			if(dp[i]&mask) {
				//cout<<"ok\n";
				ret++;
			}
			//else cout<<"##\n";
		}
		cout<<ret<<"\n";
	}
}
