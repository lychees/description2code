//-------------
vector<pair<double,int> >dack[10010];
struct lolwa
{
	int x1,y1,x2,y2;
};
lolwa lnes[10010];
int main()
{
	int x,k;
	int i,j;
	sc2(x,k);
	for (i=0;i<k;i++)
	{
		int x1,y1,x2,y2;
		sc2(x1,y1);sc2(x2,y2);
		lnes[i].x1=x1;lnes[i].y1=y1;lnes[i].x2=x2;lnes[i].y2=y2;
		for (j=x1;j<=x2;j++)
		{
			double h=y1+(j-x1)*((long double)y2-y1)/1.0/(x2-x1);
			dack[j].pb(mp(h,i));
		}

	}
	for (i=0;i<=10000;i++)
	{
		dack[i].pb(mp(0.0,0));
		//if (dack[i].size()>0)
			sort(dack[i].begin(),dack[i].end());
	}
	double yc=INT_MAX;
	while (yc>0)
	{
		pair<double,int> lb=*(lower_bound(dack[x].begin(),dack[x].end(),mp(yc,0))-1);
		//printf("got lower bound segment no. %d and current y coordinate :%lf\n",lb.se,yc);
		if (abs(lb.fi)<0.00000001)
			break;

		if (lnes[lb.se].y1<lnes[lb.se].y2)
		{	x=lnes[lb.se].x1;
			yc=lnes[lb.se].y1;
		}
		else 
		{	x=lnes[lb.se].x2;
			yc=lnes[lb.se].y2;
		}

	}
	printf("%d\n",x);

	return 0;

}
