#include <bits/stdc++.h>

using namespace std;

int static const N = 16000000;

typedef long long ll;

bool p[N];
ll a[1000010] , res[1010];

int main() {
	a[0] = 1;
	int mod =1;
	for(int i=0;i<9;i++)
		mod*=10;
	mod += 7;
	int br = 0;
	for(int i=2;i<N;i++)
	{
		if(p[i] == false)
		{
			br++;
			if(br < 1000010)
			{
				a[br] = a[br-1];
				a[br] = (a[br]*i)%mod;
			}
			for(int j=2*i;j<N;j+=i)
				p[j] = 1;
		}
	}
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		int n;
		cin>>n;
		res[i] = a[n];
	}
	for(int i=0;i<t;i++)
		cout<<res[i]<<endl;
	
	return 0;
}
