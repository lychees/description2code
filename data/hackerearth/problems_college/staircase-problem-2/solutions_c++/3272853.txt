#include <iostream>
#include <cmath>
#include <cstdio>
using namespace std;

int t,n;
long long arr[100005];

int main() {
	// your code goes here
	
	scanf("%d",&t);
	arr[1] = 1;
	long long max = 2;
	long long maxsum = 2;
	arr[2] = 1;
	arr[3] = 2;
	
	for(int i=4;i<=100000;i++){
		if(i-maxsum>max+1){
			arr[i] = arr[i-1]+1;
			max++;
			maxsum += max;
		} else
			arr[i] = arr[i-1];
		
		//cout <<i<<" "<< arr[i] << endl;
	}
	
	while(t--){
		scanf("%d",&n);
		printf("%llu\n",arr[n]);
		
	}
	
	return 0;
}
