#include <iostream>
using namespace std;

int main()
{
    int t,n,j,i,start_pos,end_pos,flag;
    long a[10000];
    cin>>t;
    for(i=0;i<t;i++)
    {
    	flag = 0;
    	start_pos = end_pos = -1;
    	cin>>n;
    	for(j=0;j<n;j++)
    		cin>>a[j];
    	for(j=1;j<n;j++)
    	{
    		if(a[j]<a[j-1])
    		{
    			start_pos = j-1;
    			while(j<n && a[j]<a[j-1])
    				j++;
    			end_pos = j-1;
    			if(start_pos!=0)
    			{
	    			if(a[start_pos-1]<a[end_pos])
	    			{
	    				if(j<n)
	    				{
	    					if(a[start_pos]<a[j])
	    					flag++;
	    					else
	    					flag+=2;
	    				}
	    				else
	    				flag++;
	    			}
	    			else
	    			flag+=2;
    			}
    			else
    			{
    				if(j<n)
	    				{
	    					if(a[start_pos]<a[j])
	    					flag++;
	    					else
	    					flag+=2;
	    				}
	    				else
	    				flag++;
    			}
    		}
    	}
    	if(flag==1)
    		cout<<start_pos+1<<" "<<end_pos+1<<"\n";
    	else
    		cout<<"-1 -1\n";
    }
    return 0;
}
