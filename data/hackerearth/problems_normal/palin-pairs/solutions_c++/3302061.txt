#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long n,count=0;
	string str,str1;
	map<string,int>m;
	cin>>n;
	while(n--)
	{
		cin>>str;
		str1=str;
		reverse(str.begin(),str.end());
		if(m[str]>0)
		{
			count=count+m[str];
			m[str1]++;
		}
		else
		m[str1]++;
	}
	cout<<count;
}
