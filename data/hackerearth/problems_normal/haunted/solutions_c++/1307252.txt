#include <stdio.h>
#include <string.h>

#include <iostream>
#include <algorithm>

using namespace std;


static int N, M;
static int a[100 * 1000];
static int b[100 * 1000];
static int c[100 * 1000];
static int dp[100 * 1000];

int main()
{
	scanf("%d %d", &N, &M);

	for (int i = 0; i < N; i++)
	{
		scanf("%d", &a[i]);
	}

	memcpy(b, a, sizeof(a));

	sort(b,b + N);

	c[0] = b[0];
	int i = 1, j = 0;

	while (i < N)
	{
		if (b[i] != c[j])
		{
			j++;
			c[j] = b[i];
		}
		i++;
	}


	int title = 0, age = 0;
	for (int i = 0; i < N; i++)
	{
		int* temp = lower_bound(c, c + j, a[i]);

		dp[temp - c]++;

		if ( dp[temp - c] > title || (dp[temp - c] == title && c[temp - c] > age) )
		{
			title = dp[temp - c];
			age = c[temp - c];
		}

		printf("%d %d\n",age, title);
	}

}
