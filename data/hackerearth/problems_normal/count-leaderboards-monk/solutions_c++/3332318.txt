#include <bits/stdc++.h>
using namespace std;
long long c[1001][1001];
long long invfact[1001];
long long mod = 1000000007;
long long fact[1001];


int main()
{
    fact[0] = 1;

    for(int i = 1;i<=1000;i++)
        {
            fact[i] = i*fact[i-1];
            fact[i]%=mod;
        }
    long long n,k;
    cin>>n>>k;

    for(long long i = 0;i<=1000;i++)
    {
        c[i][0] = 1;
    }
    for(long long i = 1;i<= 1000;i++)
    {
        for(long long j = 1;j<=i;j++)
        {
            c[i][j] = (c[i-1][j]+c[i-1][j-1])%mod;
        }
    }
    long long msum = 0;
    long long l[1001] = {0};
    l[0] = 1;

    for(long long j = 1;j<=n;j++)
    {
        for(int i = 0;i<=j-1;i++)
        {
            l[j] +=((c[j-1][i])%mod*(fact[i+1])%mod*(l[j-1-i]%mod))%mod;
            l[j]%=mod;
        }
    }
    msum = (l[n]*c[n][k])%mod;

    cout<<msum<<endl;
}
