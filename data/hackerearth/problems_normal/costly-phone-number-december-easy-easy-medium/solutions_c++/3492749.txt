#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(false);
    int t;
    cin>>t;
    while(t--)
    {
     int arr[10];
    for(int i=0;i<10;i++)
    {
        cin>>arr[i];

    }
    int len;
    cin>>len;
    int brr[len];
    for(int i=0;i<len;i++)
    {
        char c;
        cin>>c;
        brr[i]=c-48;

    }
    bool flag=true;
    while(flag)
    {
        flag=false;
        for(int a=0;a<=9;a++)
        {
            for(int b=a;b<=9;b++)
            {
                int d=(a+b)%10;
                if(arr[d]>arr[a]+arr[b])
                {
                    flag=true;
                    arr[d]=arr[a]+arr[b];
                }
            }
        }
    }
    int sum=0;
    for(int i=0;i<len;i++)
    {
        sum=sum+arr[brr[i]];
    }
    cout<<sum<<"\n";
    }
    return 0;
}
