#include <iostream>
#include <string>

using namespace std;

int main()
{
	int N;
	long long int A[101], B[101];

	cin >> N;

	for (int i = 0; i < N; i++)
		cin >> B[i];

	long long int S = 0;

	for (int i = 0; i < N; i++)
	{
		A[i] = (i+1) * B[i] - S;
		S += A[i];
	}

	for (int i = 0; i < N; i++)
		cout << A[i] << " ";

	cout << endl;

	return 0;
}
