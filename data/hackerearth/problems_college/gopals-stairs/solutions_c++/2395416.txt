#include<iostream>
#define ll long long
#define mod 1000000007
using namespace std;
ll dp[100000];
ll solve(ll n)
{
	
	
	if(dp[n]>0)
	return dp[n];
    
    for(int i=4;i<=n;i++)
  {  dp[i]=(dp[i-1]+dp[i-2]+dp[i-3]+mod)%mod;
}
    return dp[n];
}
int main()
{
	cin.sync_with_stdio(false);
	cout.sync_with_stdio(false);
    int t,j;
    cin>>t;
    dp[0]=0;
    dp[1]=1;
    dp[2]=2;
    dp[3]=4;
    while(t--)
    {
        cin>>j;
        cout<<solve(j)<<endl;
    }
    return 0;
}
