#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int main()
{
	int n;
	cin>>n;
	vector<bool> test(n+1,true);
	for(int i=2;i<n+1;i++)
	{
		if(test[i]==false)continue;
		for(int j=2;j*i<n+1;j++)test[j*i]=false;
	}
	vector<int> before(n+1,0);
	int count=0;
	for(int i=2;i<n+1;i++)
	{
		if(test[i])
		{
			count++;
		}
		before[i]=count;
	}
	//cout<<count<<endl;
	long long sum=0;
	for(int i=2;i<=n;i++)
	{
		long long add=pow(2,before[i]-1);
		sum+=add;
	}
	cout<<sum<<endl;
	return 0;
}
