#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int p1=0, p2=0, p3=0;
        while(n--)
        {
            int c1, c2, c3;
            int m1, m2, m3;
            cin>>m1>>m2>>m3;
            c1=min(p2,p3)+m1;
            c2=min(p1,p3)+m2;
            c3=min(p1,p2)+m3;
            p1=c1;p2=c2;p3=c3;
        }
        cout<<min(p1,min(p2,p3))<<endl;
    }
    return 0;
}
