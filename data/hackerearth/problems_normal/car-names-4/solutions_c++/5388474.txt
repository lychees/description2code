#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		string s;
		cin >> s;
		int n = s.size();
		int i, hash[26] = {0}, flag = 0;
		int alicia = 0;
		for(i = 0; i < n; i++)
		{
			hash[s[i] - 97]++;
		}
		int rick[3] = {0}, k = 0;
		for(i = 0; i < 26; i++)
		{
			if(hash[i] != 0)
			{
				alicia++;
			}
		}
		
		if(alicia == 3)
		{
			for(i = 0; i < 26; i++)
			{
				if(hash[i] != 0)
				{
					rick[k] = hash[i];
					k++;
				}
			}
			if(rick[0] == rick[1] && rick[0] == rick[2])
			{
				flag = 1;
			}
			if(flag == 1 && rick[0] > 1)
			{
					if(s[0]  != s[1])
					{
						flag = 0;
					}
			}
		}
		if(flag == 0)
		{
			cout << "Not OK\n";
		}
		if(flag == 1)
		{
			cout << "OK\n";
		}
	}
}
