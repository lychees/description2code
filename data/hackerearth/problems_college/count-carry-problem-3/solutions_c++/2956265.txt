#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n=0;
    scanf("%d",&n);  long long a=0,b=0,x=0,count=0,add=0;
    for(int i=0;i<n;i++)
    {  x=0;  count=0; add=0;
    	scanf("%lld%lld",&a,&b); 
    	while((a!=0)|| (b!=0))
    	{
    	    add=(a%10)+(b%10)+x;
    		x=add/10;
    		a/=10; 
    		b/=10;
    		if(x!=0) count++;
    	}
    	if(count>1)
    	printf("%lld %s\n",count,"carry operations");
    	else if(count==0)
    	printf("No carry operation\n");
    	else if(count==1) printf("1 carry operation\n");
    }
    return 0;
}
