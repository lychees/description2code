#include <bits/stdc++.h>

using namespace std;

int main()
{
    
    int test,n;
    cin>>test;
    int a[105];
    a[0] = 0;
    while(test--)
     {
     	cin>>n;
     	for(int i=1;i<=n;i++)
     	 cin>>a[i];
     	 
     	unordered_set<int> s;
     	unordered_set<int>::iterator it;
     	s.insert(0);
     	
     	vector<int> vec;
     	
     	for(int i=1;i<=n;i++)
     	 {
     	 	int val = a[i];
     	 	vec.clear();
     	 	for(it=s.begin();it!=s.end();it++)
     	 	 vec.push_back((*it) + val);
     	 	
     	 	for(int i=0;i<vec.size();i++)
     	 	  s.insert(vec[i]);
     	 }
     	 
     	 cout<<s.size()<<"\n";
     	
     }
    
    return 0;
}
