#include<cstdio>
#include<iostream>
#define MOD 1000000007
using namespace std;
int main(void)
{
	int fib[1000008];
	int ans[1000008];//ans[i]=fib[1]+fib[2]+....+fib[i];
	fib[0]=1;
	fib[1]=2;//if length of string is 1, then there are 2 problems he can like-> H or E.
	for(int i=2;i<1000008;i++)
		fib[i]=((fib[i-1]+fib[i-2])%MOD);
	ans[0]=0;
	ans[1]=fib[1];
	for(int i=2;i<1000008;i++)
		ans[i]=((fib[i]+ans[i-1])%MOD);

	int testcases;
	scanf("%d",&testcases);
	while(testcases--)
	{
		int n;
		scanf("%d",&n);
		printf("%d\n",ans[n]);
	}
	return 0;
}
