#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--){
    	int x;
    	cin>>x;
    	if(x%3 == 0)
    	{
    		for(int i = 0 ; i < x ; i++)
    			cout<<"5";
    	}
    	else if(x%3 == 1)
    	{
    		if(x == 1 || x == 4)
    		{
    			cout<<"-1"<<endl;
    			continue;
    		}
    		for(int i = 0 ; i < x-10; i++)
    			cout<<"5";
    		for(int i = 0; i < 10; i++)
    			cout<<"3";
    	}
    	else
    	{
    		if(x == 2)
    		{
    			cout<<"-1"<<endl;
    			continue;
    		}
    		for(int i = 0 ; i < x-5; i++)
    			cout<<"5";
    		for(int i = 0 ; i < 5; i++)
    			cout<<"3";
    	}
    	cout<<endl;
    }
}
