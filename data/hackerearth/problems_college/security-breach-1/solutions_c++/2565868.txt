#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string t;
    	cin>>t;
    	for(int i=0;i<t.length();i++)
    	{
    		if(t[i]=='2')
    			cout<<"cde";
    		else if(t[i]=='6')
    			cout<<"b3";
    		else if(t[i]=='7')
    			cout<<"acf";
    		else if(t[i]=='9')
    			cout<<"c6a";
    		else
    			continue;
    		
    	}
    	cout<<endl;
    }
    return 0;
}
