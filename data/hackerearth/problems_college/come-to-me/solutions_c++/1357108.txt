#include <iostream>
#include <vector>
#include <string>
using namespace std;
typedef vector<int> vi;
#define pb push_back
#define forn(i, n) for (int i = 0; i < n; ++i)

vi cnt;
bool all_diff()
{
    forn(i, cnt.size())
        if (cnt[i] > 1)
            return false;
    return true;
}

int solve()
{
    string s; cin >> s;
    cnt.assign(26, 0);
    int l = 0;
    int res = 0;
    forn(r, s.size())
    {
        cnt[s[r] - 'A']++;
        while (!all_diff())
            cnt[s[l] - 'A']--, ++l;
        res = max(res, r - l + 1);
    }
    return res;
}
 
int main()
{
    int t; cin >> t;
    while (t --> 0)
        cout << solve() << endl;
 
    return 0;
}
