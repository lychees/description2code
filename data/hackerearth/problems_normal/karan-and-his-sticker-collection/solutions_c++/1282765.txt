#include <iostream>
using namespace std;

typedef unsigned long long ull_t;

const ull_t check = (3*1e8);
int main()
{
    long T, c = 0;
    ull_t x;
    
    ios_base::sync_with_stdio(false);

	cin>>T;
    while(T--)
    {
    	cin>>x;
    	if (x < check) ++c;
    }
    cout<<c<<endl;
    return 0;
}
