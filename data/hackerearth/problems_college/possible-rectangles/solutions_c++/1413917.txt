#include <bits/stdc++.h>
#define lli unsigned long long

using namespace std;

int main()
{
	int t;
	lli m,n,a;
	scanf("%d", &t);
	while ( t-- ) {
		scanf("%llu%llu%llu", &m, &n, &a);
		m--, n--;
		lli ans = 0;
		lli tt = (lli)sqrt(a);
		for ( lli i = 1; i <= tt; i++ ) {
			if ( a%i == 0 ) {
				int fs = i;
				int sc = a/i;
				if ( fs <= m && sc <= n ) {
					ans += (lli)(m-fs+1)*(lli)(n-sc+1);
				}
				if ( fs != sc ) {
					if ( fs <= n && sc <= m ) {
						ans += (lli)(n-fs+1)*(lli)(m-sc+1);
					}
				}
			}
		}
		printf("%llu\n", ans);
	}
	return 0;
}
