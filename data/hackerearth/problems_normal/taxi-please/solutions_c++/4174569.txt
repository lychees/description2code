using namespace std;

typedef pair<int,int> ii;
typedef pair<int,ii> iii;
typedef long long ll;
typedef vector<int> vi;
typedef vector<ll> vl;
typedef vector<ii> vii;
typedef vector<iii> viii;

int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	
	int n,m;
	cin>>n>>m;
	
	viii startend;
	int count = 0;
	for(int i=0; i<n; ++i){
		int s,e;
		cin>>s>>e;
		e+=(s);
		startend.push_back({count++,{s,e}});
	}
	
	sort(startend.begin(), startend.end(), [](iii a, iii b){
		return a.second.first < b.second.first;
	});
	
	//for(auto &v : startend){
	//	cout<<v.first<<":"<<v.second.first<<":"<<v.second.second<<" ";
   	//}
	
	vi busy(m,0);
	vi ans(n,0);
	
	for(int i=0; i<n; ++i){
		bool found = false;
		for(int t = 0; t<m; ++t){
			if(startend[i].second.first >= busy[t]){
				busy[t] = startend[i].second.second;
				found = true;
				//cout<<(t+1)<<" ";
				ans[startend[i].first] = (t+1);
				break;
			}
		}
		if(!found){
				//cout<<-1<<" ";
					ans[startend[i].first] = (-1);
		}
			
	}
	
	for(auto &v : ans){cout <<v<<" ";}
	
	return 0;
}
