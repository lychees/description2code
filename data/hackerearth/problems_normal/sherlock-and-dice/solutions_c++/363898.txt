#include <iostream>
#include <vector>
#include <algorithm>
#include <ctime>
double ct=0;
double dp[10005][55];



using namespace std;

int main(int argc, const char * argv[])
{
    int t;
    cin>>t;
    while (t--) {
        int m,n,k;
        cin>>m>>n>>k;
        ct=0;
        if(k>n*m) {
        	cout<<"0.000 0"<<endl;
        	continue;
        }
        for (int i=0; i<=k; i++) {
            for (int j=0; j<=m; j++) {
                dp[i][j]=0;
            }
        }
        for (int i=1; i<=n; i++) {
            dp[i][1]=1;
        }
        for (int i=2; i<=k; i++) {
            for (int j=2; j<=m; j++) {
                for (int l=1; l<=n&&l<i; l++) {
                    if (i-l>0&&j-1>0) {
                        dp[i][j]+=dp[i-l][j-1];
                        
                    }
                }
            }
        }

        ct=dp[k][m];
        for (int i=0; i<m; i++) {
            ct/=n;
        }
        int exp=0;
        while (ct<1&&ct>0) {
            ct*=10;
            exp++;
        }
        printf("%0.3f %d\n",ct,exp);
    }
    return 0;
}
