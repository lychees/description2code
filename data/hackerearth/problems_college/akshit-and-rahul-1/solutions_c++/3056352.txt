double r1 = (sqrt(5.0)+1)/2.0;
double r2 = (sqrt(5.0)-1)/2.0;

int main()
{
	int i, j;

	int cases;

	si(cases);

	int cs = 1;

	wl(cases)
	{
		int a1, a2, b1, b2;

		cin>>a1>>a2>>b1>>b2;

		ll ans = 0;

		fl(i,a1,a2+1)
		{
			int l = ceil(r2 * i);
			int u = floor(r1 * i);

			if(l > b2 || u<b1)
			{
				ans += (b2 - b1 + 1);
			}
			else
			{
				if(u > b2)
					u = b2;
				if(l < b1)
					l = b1;
				ans += (b2 - u + l - b1);
			}

		}

		cout<<"Case #"<<cs<<": ";
		cout<<ans; nline;
		cs ++;
	}
	return 0;
}
