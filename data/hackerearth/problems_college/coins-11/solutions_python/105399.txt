
import math
def isPrime(val):
    if(val<=1):
        return 0
    if(val==2):
        return 1
    if(val%2==0):
        return 0
    m=math.sqrt(val)
    m=(int)(m+2)
    for i in range(3,m,2):
        if(val%i==0):
            return 0
    return 1

def calc(val):
    if(val==3):
        return 3
    elif(isPrime(val)):
        return 2
    else:
        return 1
