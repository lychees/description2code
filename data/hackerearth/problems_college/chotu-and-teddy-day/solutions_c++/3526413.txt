#include <iostream>
using namespace std;

int main()
{
    long  long  t;
    cin>>t;
    while(t--)
    {
      long long  n,k,sum=0;
      cin>>n>>k;
      for(int i=1;i<=n;i++)
      {
      	sum+=i;
      }
      if(sum>=k)
      {
      	cout<<"YES"<<endl;
      }else{
      		cout<<"NO"<<endl;
      
      }
    	
    }

    return 0;
}
