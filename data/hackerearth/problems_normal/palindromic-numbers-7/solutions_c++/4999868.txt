#include <bits/stdc++.h>
using namespace std;

bool isPalindrome(long long x){
	long long y = 0, temp = x;
	while (x!=0){
		y = y*10 + x%10;
		x/=10;
	}
	return temp==y?true:false;
	
}

int main()
{
    int t;
    cin>>t;
    while (t--){
    	long long a, b;
    	cin>>a>>b;
    	int cnt=0;
    	for (int i=a; i<=b; i++){
    		if (isPalindrome(i))
    			cnt++;
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
