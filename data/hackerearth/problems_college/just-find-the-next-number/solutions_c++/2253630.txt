#include<iostream>
#include<stdio.h>
using namespace std;
#define ll long long
ll arr[100001];
void pre()
{
	ll i,k=0;
	for(i=1;;i++)
	{
		if(!(i&(i>>1)))
		{
			arr[k++]=i;
			if(k==100000)
				return;
		}
	}
}
ll _b(ll l,ll h,ll k)
{
	ll mid;
	if(l<=h)
	{
		mid=(l+h)/2;
		if(arr[mid]==k)
			return arr[mid];
		if(mid==0||(arr[mid+1]>k&&k>arr[mid]))
			return arr[mid+1];
		if(arr[mid]>k)
			return _b(l,mid-1,k);
		return _b(mid+1,h,k);
	}
}
int main()
{
	pre();
	ll i,t,n;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		cout<<_b(0,100000,n)<<endl;
	}
}
