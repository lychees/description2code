#include <iostream>
using namespace std;

double arr[500005];

double percent(double p, double t)
{
	double tmp = 100.0;
//	cout << p <<" "<<t << endl;
	return (p/tmp * t);
}
int main()
{
	
	int t ; cin>>t ;
	while(t--)
	{
		int n ; cin >> n ;
		for(int i=0 ; i<n ; i++) cin >> arr[i];
		
		double tot = 100.0;
		double amt = tot / (double) n;
		
		double res = 0/0 ;
		for(int i=0 ; i<n ; i++)
		{
			res += percent(arr[i] , amt);
			
	//	printf("%.8f\n",res);
		
		}
		
		printf("%.8f\n",res);
	}
    
    return 0;
}
