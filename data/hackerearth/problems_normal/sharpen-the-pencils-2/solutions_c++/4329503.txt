#include<iostream>
using namespace std;
int main()
{


long long int  t;

cin>>t;
for(long long int i=0;i<t;i++)
{
long long int n;
cin>>n;
long long int arr[n];
for(long long int j=0;j<n;j++)
cin>>arr[j];

long long int mad[n];
long long int ria[n];
mad[0]=0;
for(long long int j=1;j<n;j++)
{
mad[j]=arr[j-1]+mad[j-1];
}
ria[n-1]=0;
for(long long int j=n-2;j>=0;j--)
{
ria[j]=arr[j+1]*2+ria[j+1];

}
long long int ans=0;
for(long long int j=0;j<n;j++)
{
if(mad[j]==min(ria[j],mad[j]))
ans++;
else
break;
}
cout<<ans<<" "<<n-ans<<endl;
}


return 0;
}
