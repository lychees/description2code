#include <iostream>

using namespace std;
int a[20];

bool subsetsum(int *a, int n, int sum)
{
	if(sum==0)
		return true;
	if(sum!=0 && n==0)
		return false;

	return subsetsum(a, n-1, sum) || subsetsum(a,n-1,sum-a[n-1]);
}

int main()
{
	int t;
	int n;
	cin>>t;
	int sum;
	while(t--)
	{
		cin>>n;
		for(int i = 0;i<n;i++)
		{
			cin>>a[i];
		}
		cin>>sum;
		if(sum == 0)
		{
			cout<<"YES\n";
		}
		else
		{
			if(subsetsum(a,n,sum))
				cout<<"YES\n";
			else
				cout<<"NO\n";
		}
	}
	return 0;
}
