#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;

int main() {
	int n;
	scanf("%d",&n);
	int a[n+1][n+1];
	int dp[n+1];
	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= n; j++) {
			scanf("%d",&a[i][j]);
		}
	}
	for (int i = 1; i <= n-1; i++) {
		memset(dp,0,sizeof(dp));
		for (int j = 1; j <= n; j++) {
			int temp = a[i][j]+a[i+1][j];
			int k = j;
			while (k) {
				for (int r = k; r <= n; r = r*2) {
					temp = a[i][j]+a[i+1][r];
					if (temp > dp[r])
						dp[r] = temp;
				}
				k = k/2;
			}
		}
		for (int b = 1; b <= n; b++) {
			//cout << dp[b] << " ";
			a[i+1][b] = dp[b];
		}	
	}	
	int ans = 0;
	for (int i = 1; i <= n; i++) {
		if (a[n][i] > ans)
			ans = a[n][i];
	}
	printf("%d\n",ans);
	return 0;
}
