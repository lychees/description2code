#include <bits/stdc++.h>
#define ll long long int
#define pb push_back
#define mp make_pair
#define s(a) scanf("%lld",&a)
#define f first
#define sc second
#define sd(a) scanf("%lf",&a)

using namespace std;

int main()
{
    ll t,i,j,k,w,l;
    ll x,y,hour,mn,ww1,ww2;
    s(t);
    while(t--) {
        s(x);s(y);
        if(x>=12) x-=12;
        hour=30*x;
        mn=6*y;
        hour=hour+y/2;
        ww1=hour-mn+360;
        while(ww1 >= 360) ww1-=360;
        ww2=mn-hour+360;
        while(ww2 >= 360) ww2-=360;
        ll ans=min(ww1,ww2);

        printf("%lld\n",ans);
    }
    return 0;
}
