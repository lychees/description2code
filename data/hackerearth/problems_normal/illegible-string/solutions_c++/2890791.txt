#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

int main()
{
	int n;
	int c = 0;
	int v_pair = 0;
	cin >> n;

	char arr[n+1];
	char arr1[n+1];
	cin >> arr;
	for (int i = 0; arr[i] != '\0'; i++) {
		if (arr[i] != 'w') {
			arr1[c] = arr[i];
			c++;
		}
	}
	arr1[c] = '\0';
	
	for (int i = 0; i < c; i++) {
		if(arr1[i] == 'v'&& arr1[i+1] == 'v' && i+1 < c) {
			v_pair++;
			i++;
		}
	}
	cout << n-v_pair << " " << (2*n)-c << endl;

	return 0;
}
