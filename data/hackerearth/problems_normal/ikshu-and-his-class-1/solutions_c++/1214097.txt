using namespace std;
vector<int> v[100001];
long long int ans = 0;
long long int countt = 0;
bool use[100001];
#define MOD 1000000007
void dfs(int b){
	use[b] = 1;
	countt++;
	for (int i = 0; i < v[b].size(); i++){
		if (use[v[b][i]] == false){
			dfs(v[b][i]);
		}
	}
}
int main(){
	ans = 1;
	int n;
	int k;
	scanf("%d%d", &n, &k);
	for (int i = 0; i < k; i++){
		int kari, kari1;
		scanf("%d%d", &kari, &kari1);
		v[kari].push_back(kari1);
		v[kari1].push_back(kari);
	}
	for (int i = 0; i < n; i++){
		if (use[i] == false){
			countt = 0;
			dfs(i);
			long long int tmp = 1;
			while (countt){
				tmp *= countt;
				tmp %= MOD;
				countt--;
			}
			ans *= tmp;
			ans %= MOD;
		}
	}
	printf("%lld\n",ans);
	return 0;
}
