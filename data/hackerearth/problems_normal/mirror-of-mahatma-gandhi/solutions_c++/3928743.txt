#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string n;
    	cin>>n;
    	int l = n.length(),flag = 0,c=0;
    	for(int i=0;i<l;i++)
    		if(n[i] != '1' && n[i] != '0' && n[i] != '8')
    			flag = 1;
    	if(flag == 1)
   			cout<<"NO\n";
   		else{
   			if(l%2 == 0){
	   			for(int i=0,j=l-1;i<l/2&&j>=l/2;i++,j--){
	   				if(n[i] == n[j])
	   					c++;
	   			}
	   			if(c == l/2)
	   				cout<<"YES\n";
	   			else 
	   				cout<<"NO\n";
   			}
   			else{
   				for(int i=0,j=l-1;i<l/2&&j>l/2;i++,j--){
	   				if(n[i] == n[j])
	   					c++;
	   			}
	   			if(c == l/2)
	   				cout<<"YES\n";
	   			else 
	   				cout<<"NO\n";
   			}
   		}
    }
    return 0;
}
