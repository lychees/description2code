#include <vector>
#include <map>
#include <iostream>
#include <fstream>
#include <algorithm>
#include <string>
using namespace std;

main()
{
   //ifstream infile("input.txt");
   int ctr1;
   string str1,str2;
   cin>>ctr1;
   cin.ignore();
   while(ctr1--)
   {
   	  getline(cin,str1);
   	  getline(cin,str2);   
	  sort(str1.begin(),str1.end());
	  sort(str2.begin(),str2.end());
	  //cout<<str1<<' '<<str2;cin.get();
	  int d=0;
   	  for(int c=0;c<str1.length();c++)
   	  {
   	  	 while(d<str2.length())
   	  	 {
   	  	 	if(str1[c]==str2[d])			
			  {
			    str1[c]=str2[d]=' ';
				break;	
			  }
			else if(str1[c]>str2[d])
			  d++;
			else break;
   	  	 }		       	  	
   	  }   	  
   	  char *ch=remove(&str1[0],&str1[str1.length()],' '); //remove spaces   	  
	  str1.resize(ch-&str1[0]);
   	  
   	  ch=remove(&str2[0],&str2[str2.length()],' '); //remove spaces
   	  str2.resize(ch-&str2[0]);
   	     	     
      //cout<<str1<<' '<<str2;cin.get();					 	     	   	  
   	  if(!str1.length() && str2.length()) cout<<"You lose some.";
   	  else if(str1.length() && !str2.length()) cout<<"You win some.";
   	  else cout<<"You draw some.";
		cout<<endl;
   }   		
}
