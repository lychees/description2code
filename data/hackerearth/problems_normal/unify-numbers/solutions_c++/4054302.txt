#include <iostream>
#include<stdlib.h>

using namespace std;

int compare(const void* a,const void* b)
{
	return(*(int*)a-*(int*)b);
}

int min(int a,int b)
{
	return (a>b?b:a);
}

void fun(int a[],int n)
{
	qsort(a,n,sizeof(int),compare);
	
	int element;
	if(n%2==0)
	{
		element=(a[n/2]+a[n/2-1])/2;
	}
	else
	{
		element=a[n/2];
	}
	long int mindistance=0;
	long int distance=0;
	long int maxdistance=0;
	for(int i=0;i<n;i++)
	{
		distance+=abs(element-a[i]);
		maxdistance+=abs(element-a[i]+1);
		mindistance+=abs(element-a[i]-1);
	}
	cout<<min(distance,min(maxdistance,mindistance))<<endl;
	
}
int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	int n;
    	cin>>n;
    	int arr[n];
    	for(int j=0;j<n;j++)
    	cin>>arr[j];
    	fun(arr,n);
    	
    }
    return 0;
}
