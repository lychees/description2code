#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long int i,j,t,k,n,p,a[100000],count,present;

    scanf("%lld", &t);

    for(i = 0; i < t; i++){
            count = 0;
            present = 0;
        scanf("%lld", &n);
        scanf("%lld", &p);

        for(j = 0; j < n; j++){
            scanf("%lld", &a[j]);

            if(a[j] > p){
                count++;
            }
            if(a[j] == p){
                present = 1;
            }

        }

        if(present == 1){
                printf("YES ");
            printf("%lld\n", count);
        }
        else {
            printf("NO 0\n");
        }

    }
    return 0;
}
