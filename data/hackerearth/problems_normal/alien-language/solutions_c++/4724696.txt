#include<iostream>
#include<string>
using namespace std;

int main(){
	int t;
	string text,pat;
	cin>>t;
	while(t--){
		cin>>text>>pat;
		int find = text.find(pat);
		if(find!=-1){
			cout<<"YES \n";
		}
		else{
			int count =0;
			for(int i =0;i<pat.size();i++){
				int find = text.find(pat[i]);
				if(find!=-1){
					count++;
				}
			}
			if(count!=0){
				cout<<"YES \n";
			}
			else{
				cout<<"NO \n";
			}
			
		}
	}
}
