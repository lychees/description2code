#include <iostream>
using namespace std;

int main()
{
    int z=0, N=0, CT=0, T=0;
    cin>>z;
    for(int i=0;i<z;i++)
    {
    	cin>>N>>CT>>T;
    	if(CT-T==1||CT-T==0)
    	{
    		cout<<"Chotu";
    		cout<<endl;
    	}
    	else if(T-CT==1||T-CT==0)
    	{
    		cout<<"Chotu";
    		cout<<endl;
    	}
    	else
    	{
    		cout<<"Motu";
    		cout<<endl;
    	}
    	
    }
    return 0;
}
