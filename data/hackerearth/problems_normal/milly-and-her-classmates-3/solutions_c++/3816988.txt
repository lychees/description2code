#include<iostream>
int T, N;
long long int Input[10005];
int main() {
	std::cin >> T;
	int order,found,start,end;
	while (T--) {
		std::cin >> N;
		for (int i = 0; i < N; ++i) {
			std::cin >> Input[i];
		}
		found = 0;
		start = -1;
		end = -1;
		order = 1;//1:asc 0:desc
		for (int i = 0; i < N - 1; ++i) {
			if (order == 1){
				if (Input[i] > Input[i + 1]) {
					order = 0;
					found = 1;
					int j = i;
					while (j > 0 && Input[j] == Input[j - 1]) {
						j--;
					}
					start = j + 1;
				}
		}
			else if(order == 0){
				if (Input[i] < Input[i + 1]) {
					order = 2;
					end = i+1;
					
				}
			}
			else {
				if (Input[i] > Input[i + 1]) {
					found = 0;
					break;
				}
			}
		}
		if (found && end == -1)end = N;
		//std::cout << (start - 2 >= 0 && end - 1 >= 0 && Input[start - 2] > Input[end - 1])<<"\n";
		//std::cout << start << " " << end << " " << Input[start - 2] << " " << Input[end - 1] << "\n";

		if (start - 2 >= 0 && end - 1 >= 0 && Input[start - 2] > Input[end - 1]) {
		//	found = 0;
			
			//std::cout << "Yes";
		}
		if (found) {
			
				std::cout << start << " " << end << "\n";
			
			//std::cout << start << " " << (end == -1) ? N : end;std::cout << std::endl;
		}
		else {
			std::cout << "-1 -1\n";
		}
	}
}
