#include <iostream>
using namespace std;

int main()
{
	int T;
	cin >> T;
	string* str1 = new string[T];
	for(int i = 0; i < T; ++i)
	{
		cin >> str1[i];
		//cout << str1[i] << endl;
	}
	int N;
	cin >> N;
	bool started = false;
	for(int i = 0; i < N; ++i)
	{
		string s;
		cin >> s;
		//cout << s << endl;
		//getline(cin,s,' ');
		int k = 0;
		for(; k < T; ++k)
		{
			if (str1[k] == s)
			{
				break;
			}
		}
		if (k == T)
		{
			if (started == true)
			{
				cout <<".";
			}
			cout << char(s[0]-'a' +'A');
			started = true;
		}
	}
    //cout << "Hello World!" << endl;
    return 0;
}
