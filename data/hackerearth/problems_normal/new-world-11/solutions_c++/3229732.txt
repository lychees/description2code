#include <iostream>
typedef long long int ll;
#define MAX 100005
using namespace std;
ll a[5*MAX],n,k;

ll Calc(ll m)
{
	ll count,lp,j;
	ll curr=0;
	lp=a[0];
	
	for(j=0;j<n;j++)
	{
		
		while(curr!=n&&a[curr]-lp<=m)
		curr++;
		
		count++;
		
		if(curr==n)
	    break;
		
		lp=a[curr-1];
		
		
	}

if(curr!=n)
return 0;

if(count<=k)
return 1;

return 0;

}

ll bins()
{
	ll m,low=0,l=0,d;
	
	for(ll i=1;i<n;i++)
	l=max(l,a[i]-a[i-1]);
	
	ll high=a[n-1]-a[0];
	
	while(low<=high)
	{
		m=(low+high)/2;

		if(Calc(m))
		{
			high=m-1;
			d=m;
		}
		
		else
		low=m+1;
	}
	
	return d;
	
}

	while(t--)
	{
	
	cin>>n>>k;
	
	for(i=0;i<n;i++)
	cin>>a[i];
	
    if(k==1)
    cout<<a[n-1]-a[0]<<endl;
    	
    else
    {
	cout<<bins()<<endl;
    }
    
	}
	return 0;
}
