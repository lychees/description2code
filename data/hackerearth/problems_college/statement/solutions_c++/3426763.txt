#include <bits/stdc++.h>
#define upperlimit 1000010
using namespace std;
int arr[upperlimit+1];
int dp[upperlimit+1];
int cnt[upperlimit+1];
int main()
{
    int t,n,i,j,answer;
    cin>>t;
    while(t--){
    	cin>>n;
    	for(i=0;i<=upperlimit;i++)cnt[i]=0;
    	for(i=0;i<n;i++)scanf("%d",&arr[i]),cnt[arr[i]]++;
    	for(j=upperlimit;j>0;j--){
    		int mx=0;
    		for(i=2*j;i<=upperlimit;i+=j)mx=max(dp[i],mx);
    		dp[j]=cnt[j]+mx;
    	}
    	answer=0;
    	for(i=0;i<=upperlimit;i++)answer=max(answer,dp[i]);
    	cout<<answer<<endl;
    	//for(i=0;i<=upperlimit;i++)cout<<dp[i]<<" ";
    }
    return 0;
}
