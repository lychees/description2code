#include<bits/stdc++.h>
using namespace std;
 
int main()
{
    int t,z,m,n,ans=1000000000;
    vector<int> v,factor(100005);
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&z);
    	for(int i=2;i*i<=z;i++)
    		while(z%i==0){factor[i]++;z=z/i;}
    			if(z>1)factor[z]++;
    }
    	
    	scanf("%d",&m);
    	while(m--)
    	{
    		ans=1000000000;
    		int count=0;
    		scanf("%d",&n);
    		for(int i=2;i*i<=n;i++)
    		{
    			if(n%i==0)
    			{
    				count=0;
    				while(n%i==0)
    				{
    					count++;
    					n=n/i;
    				}
    				ans=min(ans,factor[i]/count);
    			}
    		}
    		if(n>1)ans=min(ans,factor[n]);
    			printf("%d\n",ans);
    	}
    return 0;
}
