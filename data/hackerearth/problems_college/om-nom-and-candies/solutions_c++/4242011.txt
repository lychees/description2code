#include<bits/stdc++.h>
#define ll long long

using namespace std;
int main(){
	ll n,j=0;
	cin>>n;
	ll a[n+1],fac[n+1],res[n+1];
	a[0]=0,fac[0]=0,res[0]=0;
	for(int i=1;i<=n;i++){
		cin>>a[i];
		fac[i]=0;
	}
	
	for(int k=1;k<=n;k++){
		ll j=0;
		
		if(k==1){
			res[k]=a[k];
		}
		else{
			ll i=2;
			while(i <= sqrt(k))
		    {
		        if(k%i==0) {
		            fac[++j]=i;
		            if (i != (k / i)) {
		                fac[++j]=k/i;
		            }
		        } 
		
		        i++;
		    }
			fac[0]=1;		
			fac[++j]=k;
			sort(fac,fac+j+1);
			
			ll max=res[1];
			for(int i=1;i<j;i++){
				if(res[fac[i]]>max)
					max=res[fac[i]];
			}
			res[k]=a[k]+max;
		}
		
	}
	for(int k=1;k<=n;k++){
		cout<<res[k]<<" ";
	}	
	return 0;
}
