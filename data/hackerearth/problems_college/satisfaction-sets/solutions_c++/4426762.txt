#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	int N, M, temp;
	cin>>N;
	bool *visited = new bool[501];
	int *hash = new int[501];
	memset(hash, 0, 501);
	for(int i = 0; i < N; i++) {
		memset(visited, false, 501);
		cin>>M;
		for(int j = 0; j < M; j++) {
			cin>>temp;
			if(!visited[temp]) {
				hash[temp]++;
			}
			visited[temp] = true;
		}
	}
	long int ans = 0;
	for(int i = 0; i < 501; i++) {
		// cout<<hash[i]<<endl;
		if(hash[i] == N) {
			ans += i;
		}
		ans %= 1000000000;
	}
	cout<<0 - (ans % 1000000000)<<endl;
    return 0;
}
