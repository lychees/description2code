n = int(raw_input())
for j in range(n):
    s = raw_input()
    d = list(s)
    m = list(s)
    a = raw_input()
    b = "".join(list(reversed(a)))
    ways = 0
    for i in range(len(s)+1):
        d.insert(i, a)
        m.insert(i, b)
        if "".join(d) == "".join(list(reversed(m))):
            ways += 1
        d = list(s)
        m = list(s)
    print ways
