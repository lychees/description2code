#include <iostream>
using namespace std;

int main()
{
    int i,n,k;
    cin>>n;
    while(n--)
    {
        int pos[1000001]={0};
        int neg[1000001]={0};

        cin>>k;
        int a[k],x;
        for(i=0;i<k;i++)
        {
            cin>>x;
            if(x>0)
            {
                pos[x]++;
            }
            else{
                neg[-x]++;
            }
        }

        long long int ans=0;

        for(long long int j=0;j<1000001;j++)
        {
            if(pos[j]>0)
            {
                ans+= (pos[j]*(pos[j]+1))/2;
            }
            if(neg[j]>0)
            {
                ans+= (neg[j]*(neg[j]+1))/2;
            }
        }
        cout<<ans<<endl;


    }
    return 0;
}
