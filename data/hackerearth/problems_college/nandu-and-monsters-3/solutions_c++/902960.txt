#include <iostream>
#include <cstdio>
using namespace std;
int main(int argc, char const *argv[])
{
	int n, q, i, j, x, y;
	char c[100005];
	bool p;
	cin >> n >> q;
	scanf("%s", c);
	while (q != 0) {
		cin >> i >> j;
		if (i < j) {
			x = i;
			y = j;
		} else {
			x = j;
			y = i;
		}
		if (x == y) {
			if (c[x - 1] == '1') cout << "No";
			else cout << "Yes";
		} else {
			p = true;
			while (x != y) {
				if (c[y - 1] == '1' || c[x - 1] == '1') p = false;
				if (y < x) x /= 2;
				else y /= 2;
			}
			if (p) cout << "Yes"; else cout << "No";
		}
		cout << endl;
		q--;
	}
	return 0;
}
