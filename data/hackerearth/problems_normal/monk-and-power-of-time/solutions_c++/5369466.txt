#include <iostream>
#include<queue>
using namespace std;

int main()
{
	int n,num,a[100],t=0,job,exe=0;
	cin>>n;
	
	queue<int>q;
	
	for(int i=0;i<n;i++)
	{
		cin>>num;
		q.push(num);
	}
	
	for(int i=0;i<n;i++)
	cin>>a[i];
	
	while(!q.empty())
	{
		job=q.front();
		if(job==a[exe])
		{
		 q.pop();
		 exe++;
		 t++;
		}
		else
		{
			q.pop();
			q.push(job);
			t++;
		}
	}
	cout<<t;
	return 0;
	
}
