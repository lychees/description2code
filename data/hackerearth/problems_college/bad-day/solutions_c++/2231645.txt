#include <iostream>
using namespace std;
#define ll long long
int f[1000010];
void precomp()
{
	f[1]=1;
	f[2]=2;
	for(ll i=3;i<1000010;i++)
	{
		f[i]=f[i-1]+f[i-2];
	}
}
int main()
{
    ll t;
    cin>>t;
    precomp();
    while(t--)
    {
    	ll n;
    	cin>>n;
    	cout<<f[n]<<endl;
    }
    return 0;
}
