	int i, j, k, n, cases, t;
	vector<int> vi;
	
	cin>>cases;
	int sum = 0;
	while(cases--){
		cin>>t;
		vi.push_back(t);
		sum+= t;
	}
	
	while(sum>0){
		
		int mn = sum;
		for(i=0;i<vi.size();i++)
			if(mn > vi[i] && vi[i]>0)
				mn = vi[i];
		
		//cout<<sum<<" "<<mn<<" ";
		
		int t = 0;
		for(i=0;i<vi.size();i++)
			if(vi[i]>0)
				t++;
		
		if(t)cout<<t<<endl;
		
		for(i=0;i<vi.size();i++)
			vi[i] -= mn;
			
		sum -= mn;		
	}
	
	

return 0;
}
