inline void sc(int &x)
{
	register int c = gc();

    for(; ((c<48 || c>57) && c != '-');c=gc());
	if(c=='-')
	{
		neg = 1;
		c = gc();
	}
	for(; c>47 && c<58 ; c = gc())
		x = (x<<1) + (x<<3) + c - 48;
	if(neg)
    	x = -x;
}
