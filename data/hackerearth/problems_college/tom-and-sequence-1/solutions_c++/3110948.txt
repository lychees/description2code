#include <iostream>
using namespace std;
#define ll long long
#define mod 1000000007
#define N 10000001
int arr[N]={0};
int main() {
	int i,j;
   for(i=2;i<=N;i++)
   {   if(arr[i]==0)
    {  int g;j=2;g=j*i;
    	while(g<=N)
    	{
    		arr[g]=1;j++;g=j*i;
    	}
    }
   }
     int t;
     cin>>t;
     while(t--)
     {  ll ans=1;
     	int k;
     	cin>>k;
     	for(i=2;i<=N;i++)
     	{
     		if(arr[i]==0)
     		{  k--;
     		 ans=(ans*(ll)i)%mod;
     		}
     		if(k==0)
     		break;
     	}
     	
     	cout<<ans<<endl;
     }
   
	return 0;
}
