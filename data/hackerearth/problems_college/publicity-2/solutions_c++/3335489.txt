#include <bits/stdc++.h>

using namespace std;
bool visited[100005];
vector <int> adj[100005];

void dfs(int s) {
    visited[s] = true;
    for(int i = 0;i < adj[s].size();++i)    {
     if(visited[adj[s][i]] == false)
         dfs(adj[s][i]);
    }
}

int main() {
    int tc;
    cin >> tc;

    while(tc--){
    int n, e, x, y, count = 0;
    cin >> n;
    cin >> e;

    for(int i = 0;i < e;++i) {
     cin >> x >> y;
     adj[x].push_back(y);
     adj[y].push_back(x);
    }

    for(int i = 0;i < n;++i)
        visited[i] = false;

    for(int i = 0;i < n;++i) {
     if(visited[i] == false)     {
         dfs(i);
         count++;
     }
    }
    cout << count << endl;
    for (int i = 0;i < n;i++)
        adj[i].clear();

    }
    return 0;
}
