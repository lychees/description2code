#include <iostream>
using namespace std;

int main()
{
    
int test,n,count,arrive[1000],k;
cin>>test;
while(test--)
{
count=0;
cin>>n;
cin>>k;
for(int i=1;i<=n;i++)
cin>>arrive[i];
for(int i=1;i<=n;i++)
{
if( arrive[i]<=0)
count++;
}
(count>=k )? cout<<"NO":cout<<"YES";
cout<<endl;
}
 
return 0;
}
