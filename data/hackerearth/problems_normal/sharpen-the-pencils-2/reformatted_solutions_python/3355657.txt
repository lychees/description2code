T = input()
for _ in xrange(T):
    N = raw_input()
    pencils = map(int, raw_input().strip().split())
    mPCount = 0
    rPCount = 0
    mPencil = 0
    rPencil = 0
    while len(pencils) > 0:
        if mPencil <= rPencil:
            mPencil += pencils.pop(0)
            mPCount += 1
        else:
            rPencil += pencils.pop() * 2
            rPCount += 1

    print mPCount, rPCount
