#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int n,ans[26],temp[26],i,n1,count=0;
    scanf("%d",&n);
    char c[101];
    for(i=0;i<26;i++)
    {
    	ans[i]=0;
    	
    }
    n1=n;
    while(n1--)
    {
    	 for(i=0;i<26;i++)
    {
    	temp[i]=0;
    	
    }
    scanf("%s",c);
    for(i=0;c[i]!='\0';i++)
    {
    	temp[c[i]-'a']=1;
    }
    for(i=0;i<26;i++)
    {
    	ans[i]+=temp[i];
    	
    }
    	
    }
    for(i=0;i<26;i++)
    {
    	if(ans[i]==n)
    	{
    	count++;
    	}
    	
    }
    printf("%d",count);
    
    return 0;
}
