#include<iostream>
#include<stdlib.h>
#include<stdio.h>
using namespace std;
#define MAX 10000001
int a[MAX];//,b[MAX];
int main()
{
	int t;
	int n,i,j,m;
	for ( i = 2 ; i < MAX ; i++ )
	{	if ( a[i] == 0 )
		{
			for ( j = i; j < MAX ; j = j + i )
			{	m = j;
				while ( m % i == 0 )
				{	a[j]++;
					m = m / i;
				}
			}
		}
	}
	
	//b[1]=0;
	for ( i = 2 ; i < MAX ; i++ )
		a[i]+=a[i-1];
		
	cin>>t;
	while(t--) {
		scanf("%d",&n);
		printf("%d\n",a[n]);
	}
	return 0;
}
