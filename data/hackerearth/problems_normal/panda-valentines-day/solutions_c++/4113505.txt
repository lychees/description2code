ll arr[10000005];
ll res[1000005];

void pre1()
{
    ll i,j,tmp;
    for(i=2;i<=1000005;i++)
    {
        if(arr[i]==0)
        {
            //primes[ctr++]=i;
            for(j=i;j<=1000005;j+=i)
            {
				
				tmp = j;
				while(tmp%i==0)// agar j ko wo prime number divide krra hia
				{
					arr[j]++;//itni bar divide krra hai
					tmp/=i;
				}
			}
        }
    }
}


int main()
{
    pre1();
    ll tc,n,i;
    res[2]=1;
    res[3]=2;
    for(i=4;i<1000005;++i)
    	res[i]=res[i-1]+arr[i];
    sll(tc);
    while(tc--)
    {
        sll(n);
        pll(res[n]);
    }
}
