#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
const int MAX_COST = 1000;
const int MAX_TEST = 1000;
const int MAX_LEN = 1000;
int main() {
    //freopen("input.in", "r", stdin);
    //freopen("output.out", "w", stdout);
    int T, cost[10], ans, n;
    string s;
    bool ok;
    scanf("%d", &T);
    assert(T <= MAX_TEST && T);
    while (T--) {
        ok = true;
        ans = 0;
        for (int i = 0; i < 10; ++i) {
            scanf("%d", cost + i);
            assert(cost[i] <= MAX_COST);
        }
        while (ok) {
            ok = false;
            for (int i = 0; i < 10; ++i) {
                for (int j = 0; j < 10; ++j) {
                    if (cost[(i + j) % 10] > cost[i] + cost[j]) {
                        cost[(i + j) % 10] = cost[i] + cost[j];
                        ok = true;
                    }
                }
            }
        }
        scanf("%d", &n);
        cin >> s;
        assert(s.size() == n && n <= MAX_LEN && n);
        for (int i = 0; i < n; ++i) {
            assert(s[i] >= '0' && s[i] <= '9');
            ans += cost[s[i] - '0'];
        }
        printf("%d\n", ans);
    }
    return 0;
}
