#include <iostream>
using namespace std;

int main()
{
	int array[1000000];
	int N,c,Q,x,i,count=1,k;
    cin>>N;
    for(i=1;i<=N;i++){
    	cin>>c;
    	k =0;
    	while(k<c){
    		array[count]=i;
    		count++;
    		k++;
    	}
    }
    cin>>Q;
    for(i=0;i<Q;i++){
    	cin>>x;
    	cout<<array[x]<<endl;
    }
    return 0;
}
