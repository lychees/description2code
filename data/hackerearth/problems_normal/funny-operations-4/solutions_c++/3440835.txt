#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin >> t;
    while(t--){
    	int n;
    	cin >> n;
    	long long int max1=0,len=0,local1=0,local2=(1<<(63))-1,count1=0;
    	vector<long long int> arr(n);
    	for(int i=0;i<n;i++) cin >> arr[i];
    	for(int i=1;i<(1<<n);i++){
    		local1 = 0;
    		local2 = (1<<(63))-1;
    		count1=0;
    		for(int j=0;j<n;j++){
    			if(i&(1<<j)){
    				local1 = local1^arr[j];
    				local2 = local2&arr[j];
    				count1++;
    			}
    		}
    	//	local1 = local1|local2;
    		if((local1|local2)>max1){
    			max1 =local1|local2;
    			len = count1;
    		}
    		else if((local1|local2)==max1){
    			if(len<count1){
    			max1 = (local1|local2);
    			len = count1;
    			}
    		}
    	}
    	if(max1>=0)
    	cout << max1 << " "<<len <<endl;
    	else cout << 0<<" "<<0 << endl;
    }
    return 0;
}
