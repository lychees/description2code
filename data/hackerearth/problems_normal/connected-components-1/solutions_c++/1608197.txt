#include <cstdio>

using namespace std;

typedef long long ll;

const int N = 1000010, mod = 1000000007;
int q;
ll f[N], r[N];

ll calc(ll a, int c)
{
    if (!c) return 1;
    ll x = calc(a, c >> 1);
    x = x * x % mod;
    if (c & 1) x = x * a % mod;
    return x;
}

ll bc(int a, int b)
{
    if (a < b) return 0;
    return f[a] * r[b] % mod * r[a - b] % mod;
}

int main()
{
    f[0] = 1;
    for (int i = 1; i < N; i++) f[i] = f[i - 1] * i % mod;
    r[N - 1] = calc(f[N - 1], mod - 2);
    for (int i = N - 2; i >= 0; i--) r[i] = r[i + 1] * (i + 1) % mod;
    scanf("%d", &q);
    while (q--)
    {
        int n, m, c;
        scanf("%d %d %d", &n, &m, &c);
        if (!c)
        {
            if (m == n) printf("1\n"); else
            printf("0\n");
        } else
        printf("%d\n", bc(n - m - 1, c - 1) * bc(m + 1, c) % mod);
    }
    return 0;
}
