#include <iostream>
using namespace std;
int main()
{
    int tc,c,i,j;
    cin>>tc;
    for(i=0;i<tc;i++)
    {
    	
    	cout<<("\n");
    	cin>>c;
    	if(c%5==0)
    	{
    		for(j=0;j<c;j++)
    		{
    			cout<<"3";
    		}
    	}
    	else if(c==1||c==2||c==4||c==7)
    	{
    		cout<<"-1";
    	}
    	else if(c%3==0)
    	{
    		for(j=0;j<c;j++)
    		{
    			cout<<"5";
    		
    		}
    		
    	}
    	else if(c%3==1)
    	{
    		for(j=0;j<c-10;j++)
    		{
    			cout<<"5";
    			
    		}
    		cout<<"3333333333";
    	}
    	else if(c%3==2)
    	{
    		for(j=0;j<c-5;j++)
    		{
    			cout<<"5";
    		}
    		cout<<"33333";
    	}
    	
    }

return 0;
}
