#include<iostream>
using namespace std;
int main () {
    long long t , n,i , pr=1 , h , a[10003];
    scanf ( "%lld" , &t );
    while ( t-- ) {
        pr = 1;
        scanf ( "%lld %lld" , &n , &h ) ;
        for ( i = 0 ; i < n ; i++ ) {
            scanf ( "%lld" , &a[i] ) ;
            pr = ( pr*a[i] )%h;
        }
        if ( pr == 0 )
            printf ( "YES\n");
        else
            printf ( "NO\n");
    }
    return 0;
}
