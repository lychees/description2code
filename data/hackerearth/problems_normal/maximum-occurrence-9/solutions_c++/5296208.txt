#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    char s[1000];
    gets(s);
    int count[256]={0},max=0;
    char str;
    for(int i=0;i<strlen(s);i++)
    count[s[i]]++;
    for(int i=0;i<256;i++)
    {
    	if(count[i]>max)
    	{
    		max=count[i];
    		str=char(i);
    	}
    }
    cout<<str<<" "<<max<<"\n";
    return 0;
}
