#include <iostream>
using namespace std;
long int arr[100002];

int main()
{
    long int N, Q;
    cin>>N>>Q;
    for(long int i=0; i<N; i++)
    {
    	cin>>arr[i];
    }
    
    bool type;
    bool found;
    long int a, b;
    for(long int i=0; i<Q; i++)
    {
    	cin>>type;
    	if(type==0)
    	{
    		cin>>a>>b;
    		arr[a-1] = b;
    	}
    	else
    	{
    		found = false;
    		cin>>a;
    		for(long int i=0; i<N; i++)
    		{
    			if(arr[i]>=a)
    			{
    				found = true;
    				cout<<i+1<<endl;
    				break;
    			}
    		}
    		if(found == false)
    		{
    			cout<<"-1"<<endl;
    		}
    	}
    }
    return 0;
}
