def discount(n):
    if n <= 10:
        return 0.75
    elif n <= 20:
        return 0.70
    else:
        return 0.80


a = input()
arr = raw_input().strip().split()
b, c = map(int, raw_input().strip().split())
area = b*c
ans = 0
for i in range(a):
    rate = (sum(map(int, arr[i])))*(int(arr[i]))
    ans += rate*area*discount(int(arr[i]))
if ans-int(ans) > 0:
    print int(ans)-1
else:
    print int(ans)
