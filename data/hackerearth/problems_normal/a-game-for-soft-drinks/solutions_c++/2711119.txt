#include<bits/stdc++.h>
using namespace std ;
bool store[101][101][101] ;
void pro(){
	for(int i=0 ; i<101 ; ++i)
	for(int j=0 ; j<101 ; ++j)
	for(int k=0 ; k<101 ; ++k){
		if( (i>0 && !store[i-1][j][k]) ||  (j>0 && !store[i][j-1][k]) || (k>0 && !store[i][j][k-1]) || (i>0 && j>0 && !store[i-1][j-1][k]) ||
		(i>0 && k>0 && !store[i-1][j][k-1]) || (k>0 && j>0 && !store[i][j-1][k-1]) || (i>0 && j>0 && k>0 && !store[i-1][j-1][k-1]) )
		store[i][j][k] = true ;
		
	}
}
int main(){
	ios_base::sync_with_stdio(false) ;
	memset(store,false,sizeof store) ;
	int t ;
	pro() ;
	cin >> t ;
	int a,b,c ;
	while(t--){
		
		cin >> a >> b >> c ;
	    if(store[a][b][c])
	    cout << "Ashima\n" ;
	    else
	    cout << "Aishwarya\n" ;
	}
	return 0 ;
}
