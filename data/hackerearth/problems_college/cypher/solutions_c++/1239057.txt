#include<iostream>
#include<string>

using namespace std;

int main()
{
	long T;
	cin>>T;
	string s;
	int shift, i, j, zeroshift;
	
	char dict[26];
	for(i=0; i < 26; i++)
		dict[i] = char('a' + i);
	
	while(T--)
	{
	 	cin>>s;
	 	shift = 0;
	 	for (auto it = s.begin(); it != s.end(); ++it)
	 	{
	 			zeroshift = (shift++ % 26);
	 			i = (*it - 'a') + zeroshift; i %= 26;
	 			cout<<dict[i];
	 	}
	 	cout<<endl;
	}
	return 0;
}
