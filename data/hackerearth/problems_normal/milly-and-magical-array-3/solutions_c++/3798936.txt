#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int t,n,x,ans,k,t1;
    cin >> t ;
    while(t--)
    {
    	cin >> n >> x ;
    	long long int ar[n],i;
    	for(i=0;i<n;i++)
    	cin >> ar[i] ;
    	//sort(ar,ar+n);
    	ans=0;
    	for(i=1;i<n;i++)
    	{
    		if(ar[i-1]<ar[i])
    		continue;
    		else
    		{
    			if(ar[i-1]==ar[i])
    			{
    				ar[i]+=x;
    				ans++;
    			}
    			else
    			{
    				k=ar[i-1]-ar[i];
    				t1=k/x;
    				if(k%x!=0)
    				t1++;
    				ar[i]+=t1*x;
    				ans+=t1;
    			}
    		}
    	}
    	cout << ans << endl ;
    }
    return 0;
}
