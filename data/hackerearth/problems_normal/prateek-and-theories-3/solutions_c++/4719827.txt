#include <bits/stdc++.h>
#define ll long int
using namespace std;

int main()
{
    int t;
    cin>>t;
    int n;
    while(t--)
    {
    	cin>>n;
    	int t1[n],t2[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>t1[i]>>t2[i];
    	}
		map<int,int> prefixSum;
    	for(int i=0;i<n;i++)
    	{
    		prefixSum[t1[i]]++;
    		prefixSum[t2[i]]--;
    	}
    	int m=0,ans=0;
    	for(map<int,int>::iterator i=prefixSum.begin();i!=prefixSum.end();i++)
    	{
    		m+=i->second;
    		ans=max(m,ans);
    	}
    	cout<<ans<<endl;
    	
    }
    return 0;
}
