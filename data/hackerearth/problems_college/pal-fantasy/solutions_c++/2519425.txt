#include<iostream>
#include<string>
using namespace std;
int length(string s)
{
    int i=0;
    while(s[i]!='\0'){
        i++;
    }
    return i;
}
int main()
{
    int test;
    cin>>test;
    while(test--){
        string s;
        cin>>s;
        int l=length(s);
        int i=0,ans=0;
        while(i<l && ans==0){
            if(s[i]==s[l-1]){
                    //cout<<i<<"--"<<l-1<<"--";
                    int a=i,b=l-1;
                while(s[a]==s[b] && a<=b){
                    //cout<<a<<"sss"<<b<<"kkk";
                    a++;b--;
                }
                if(a>=b){
                    //cout<<"G";
                        ans++;
                    //break;
                }
            }
            i++;
        }
        cout<<l + i-1<<endl;

    }
}
