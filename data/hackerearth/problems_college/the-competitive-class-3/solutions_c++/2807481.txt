int main()
{
    int n ;
    scanf("%d",&n) ; 
    
    vector <int> val(n,0) ;
    
    for(int i = 0 ; i < n ; i ++)
    {
    	scanf("%d",&val[i]) ; 
    }
    
    vector<int> temp = val ;
    
    
    sort(val.begin(),val.end()) ; 
    
    vector<int> :: iterator it ;
    for(int i = 0 ; i < n ; i++)
    {
    	it = upper_bound(val.begin(),val.end(),temp[i]) ; 
    	
    	cout << distance(it,val.end()) + 1 << " "; 
    }
    
    return 0;
}
