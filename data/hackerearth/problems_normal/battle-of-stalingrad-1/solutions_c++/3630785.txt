#include <bits/stdc++.h>
using namespace std;

long long pos(long long a) {
	if(a < 0) {
		return -a ; 
	} else {
		return a ; 
	}
}

long long det(long long d, long long e, long long f, long long g, long long h, long long i, long long a=1, long long b=1, long long c=1) {
	
	return pos(a*e*i-a*h*f+b*f*g-b*d*i+c*d*h-c*e*g) ; 
	
}

int main()
{
    long long T, x, x1, x2, x3, y, y1, y2, y3 ;
    cin >> T ; 
    while(T--) {
    	cin >> x1 >> y1 >> x2 >> y2 >> x3 >> y3 >> x >> y ;
    	if(det(x1,x2,x3,y1,y2,y3) == det(x,x2,x3,y,y2,y3) + det(x1,x,x3,y1,y,y3) + det(x1,x2,x,y1,y2,y)) {
    		cout << "INSIDE\n" ; 
    	} else {
    		cout << "OUTSIDE\n" ; 
    	}
    }
    return 0;
}
