#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n, k;
    	int a[100005];
    	scanf("%d",&n);
    	scanf("%d",&k);
    	for(int i=0;i<n;i++)
    	{
    		scanf("%d",&a[i]);
    	}
    	sort(a,a+n);
    	unsigned long long int sum = 0;
    	int i = n-1;
    	while(k--)
    	{
    		sum = sum + a[i];
    		i--;
    		if(i<0)
    			break;
    	}
    	printf("%llu\n",sum);
    }
    return 0;
}
