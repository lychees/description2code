typedef long long LL;
using namespace std;
const int MOD = 1e9+7;
const int SIZE = 1e5+10;
#define LOCALTEST 0 // change this to 1 to perform localtest on "in.txt"
int n;
LL solve(int x) {
    LL ans=0;
    int prev=x;
    int lo=1,hi=x-1;
    if (hi%2==0) hi--;
    int uselo=1;
    while (lo<=hi) {
        if (uselo) {
            ans+=abs(lo-prev);
            prev=lo;
            lo+=2;
        } else {
            ans+=abs(hi-prev);
            prev=hi;
            hi-=2;
        }
        uselo^=1;
    }
    prev=x;
    lo=2,hi=x-1;
    if (hi%2) hi--;
    uselo=1;
    while (lo<=hi) {
        if (uselo) {
            ans+=abs(lo-prev);
            prev=lo;
            lo+=2;
        } else {
            ans+=abs(hi-prev);
            prev=hi;
            hi-=2;
        }
        uselo^=1;
    }
    return ans;
}
int main(){
    if (LOCALTEST) {
        freopen("in.txt", "r", stdin);
        freopen("out.txt", "w", stdout);
    }
    CASET {
        RI(n);
        printf("%lld\n",solve(n));
    }
    return 0;
}
