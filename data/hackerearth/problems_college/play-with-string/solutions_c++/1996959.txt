#include <bits/stdc++.h>
using namespace std;
#define ll long long 
int main()
{
	ios::sync_with_stdio(false);
	ll n;
	cin>>n;
	while (n--) {
		string s;
		cin>>s;
		string r;
		for (ll i=s.length()-1;i>=0;i--) {
			r+=s[i];
		}
		bool flag=false;
		for (ll i=1;i<s.length();i++) {
			if (abs(s[i]-s[i-1]) != abs(r[i]-r[i-1])) {
				flag =true;
				break;
			}
		}
		if (flag) cout<<"Not Funny\n";
		else cout<<"Funny\n";
	}
	return 0;
}
