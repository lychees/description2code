#include<bits/stdc++.h>
using namespace std;
int main()
 {int t,n,i,x;
 cin>>t;
 while(t--)
   {cin>>n;
   vector<pair<int,int> > a;
    for(i=0;i<n;i++)
      {cin>>x;
      a.push_back(make_pair(x,i));
      }
      sort(a.begin(),a.end());
      int ans=0;
      for(i=0;i<n;i++)
      {ans=max(ans,abs(i-a[i].second));
      }
      cout<<ans<<endl;
   }
 	
 }
