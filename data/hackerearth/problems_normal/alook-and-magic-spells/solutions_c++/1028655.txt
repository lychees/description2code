using namespace std;

const ll mod=1000000007;

struct matrix {
	int m,n;
	ll a[5][5];
};

matrix c,v,tg,mtdv;

matrix mul(matrix a,matrix b) {
	int i,j,k;matrix c;
	c.m=a.m;c.n=b.n;
	for(i=1;i<=c.m;++i) {
		for(j=1;j<=c.n;++j) {
			c.a[i][j]=0;
			for(k=1;k<=a.n;++k) {
				c.a[i][j]=(c.a[i][j]+(a.a[i][k]*b.a[k][j])%mod)%mod;
			}
		}
	}
	return c;
}

matrix power(matrix *a,int n) {
	if (!n) return mtdv;
	tg=power(a,n>>1);
	tg=mul(tg,tg);
	if (n&1) tg=mul(tg,*a);
	return tg;
}
