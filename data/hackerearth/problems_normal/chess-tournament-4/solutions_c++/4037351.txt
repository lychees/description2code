#include <iostream>
#include<vector>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    int n;
    cin>>n;
    int p=1<<n;
    //cout<<p<<endl;
    int a[1025][1025];
    for(int i=2;i<=p;i++)
    {
    	for(int j=1;j<=i-1;j++)
    	{
    		scanf("%i",&a[i][j]);
    	}
    }
    
    vector<int> s;
    for(int i=1;i<=p;i++)
    s.push_back(i);
    while(p!=1)
    {
    /*	for(int i=0;i<p;i++)
        printf("%i ",s[i]);
        printf("\n");*/
      	vector<int> temp;
      	for(int i=0;i<p;i=i+2)
      	{
      		if(a[s[i+1]][s[i]]==1)
      		{
      			temp.push_back(s[i+1]);
      		}
      		else
      		temp.push_back(s[i]);
      	}
      	s.clear();
      	for(int i=0;i<temp.size();i++)
      	 s.push_back(temp[i]);
      p=s.size();	 
      
    }
    printf("%i",s[0]);
    return 0;
}
