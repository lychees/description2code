
t = int(raw_input())

for i in range(0, t):
    n = raw_input()
    inp = raw_input()
    inp = inp.split('-1')

    count = []
    for st in inp:
        count.append(st.count('1'))

    max = 0

    for j in range(0, len(count)-1):
        if count[j] + count[j+1] > max:
            max = count[j] + count[j+1]

    if max == 0:
        print '0/1'
    else:
        print str(max)+'/1'
