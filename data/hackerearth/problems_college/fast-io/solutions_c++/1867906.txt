#include<bits/stdc++.h>
using namespace std;

int main()
{
    long int n, ctr=0;
    string s;
    cin>>n;
    while(n--)
    {
        cin>>s;
        if(s.length() < 9)
        {
            ctr++;
            continue;
        }
        else if(s.length() == 9)
        {
            if(s[0] == '1' || s[0] == '2')
            {
                ctr++;
                continue;
            }
            else if(s[0] == '3')
            {
                int flag = 0;
                for(int i=1; i<9; i++)
                {
                    if(s[i] != '0')
                    {
                        flag = 1;
                        break;
                    }
                }
                if(flag == 0)
                {
                    ctr++;
                    continue;
                }
            }
            else
                continue;
        }
    }
    cout<<ctr<<endl;
}
