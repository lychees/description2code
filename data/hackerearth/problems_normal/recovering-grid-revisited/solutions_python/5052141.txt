from sys import stdin
import math
hi = 10**9 + 7
fa = [1]
for i in xrange(1,3000000):
	fa.append((fa[i-1]*i)%hi)
def ncr(n,r):
	num = fa[n]
	den = (fa[r] * fa[n-r])%hi
	den = pow(den, hi-2, hi)
	return (num*den)%hi
t = int(stdin.readline())
for _ in xrange(t):
	n,m = map(int,stdin.readline().split())
	total = m + n*n -2
	ans =  ncr(total,m)
	if n==1:
		ans = 0
	print ans
