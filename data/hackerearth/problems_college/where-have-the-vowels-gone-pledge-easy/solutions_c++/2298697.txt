#include <iostream>
using namespace std;

int main()
{
	int n,i=0;
	scanf("%d",&n);
	char text[n];
	scanf("%s",text);
	bool mark[5]={false};
	while(text[i]!='\0')
	{
		switch(text[i]-'a'+1)
		{
			case 1:
			case 5:
			case 9:
			case 15:
			case 21:
					if(mark[(text[i]-'a'+2)/5]==false)
						mark[(text[i]-'a'+2)/5]=true;
		}
		i++;
	}
	int count=0;
	for(i=0;i<=4;i++)
	{
		if(mark[i]==true)
		 count++;
	}
	if(count==5)
		printf("YES");
	else
		printf("NO");
	return 0;
}
