void swap (char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}	
void  permute(char *ch1,int i,int j)
{
	if(i==j)
	cout<<ch1<<endl;
	for(int u=i;u<=j;u++)
	{    
		swap((ch1+i),(ch1+u));
		permute(ch1,i+1,j);
		swap(ch1+i,ch1+u);
	}
	
}
