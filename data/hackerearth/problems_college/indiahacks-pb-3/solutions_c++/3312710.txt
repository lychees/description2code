#include<bits/stdc++.h>
using namespace std;
int a[100000],dp[100000];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,k;
        cin>>n>>k;
        for(int i=1;i<=n;i++)
        {
            cin>>a[i];
        }
        memset(dp,0,sizeof dp);
        for(int i=1;i<=k;i++)
        {
            dp[i]=max(dp[i-1],a[i]);
        }
        for(int i=k+1;i<=n;i++)
        {
            dp[i]=max(dp[i-k-1]+a[i],dp[i-1]);
        }
        cout<<dp[n]<<endl;
    }
    return 0;
}
