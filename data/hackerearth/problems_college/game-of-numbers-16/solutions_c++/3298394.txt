#include <iostream>
#include<vector>
#include<set>
using namespace std;

int main(){
   int t;cin>>t;
   while(t--){
    long long n;cin>>n;long long count=0;
    for(long long i=1;i<n;i++){
        for(long long j=i+1;j<=n;j++){
            if(((i*j)%(i+j))==0){
                count++;
            }
        }
    }
    cout<<count<<endl;
   }
    return 0;
}
