using namespace std;
long long tab[100005][45];
int main()
{
    int t,i,mem;
    long long ct=0,keb,n,m;
    scanf("%d",&t);
    while(t--)
    {
        ct=0;
        scanf("%lld%lld",&n,&m);
        if(m==1)
            printf("%lld\n",n);
        else if(m>=10000001)
            printf("%lld\n",n/m+n%m);
        else
        {
            for(i=1;;i++)
            {
                if((long long)pow(m,i)>n)
                    break;
            }
            mem=i-1;
            for(i=mem;i>=0;i--)
            {
                if(n>=(long long)pow(m,i))
                {
                    ct+=n/(long long)pow(m,i);
                    n%=(long long)pow(m,i);
                }
                if(n==0)
                    break;
            }
            printf("%lld\n",ct);
        }
    }
}
