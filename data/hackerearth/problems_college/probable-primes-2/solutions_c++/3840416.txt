bool prime[1000001];
void seive()
{
	int i,j;
	memset(prime,true,1000001);
	prime[0]=false;
	prime[1]=false;
	for(i=0;i*i<=1000000;i++)
	{
		if(prime[i])
		{
			for(j=2*i;j<=1000000;j+=i)
			{
				prime[j]=false;
			}
		}
	}

}
