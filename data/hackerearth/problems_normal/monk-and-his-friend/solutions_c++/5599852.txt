#include <iostream>
using namespace std;
int main()
{
    int t;
    unsigned long long int m,p;
    cin>>t;
    while(t--)
    {
    	cin>>p>>m;
    	int c=0;
    	if(p!=m)
    	{
    	while(p>0||m>0)
    	{
    		int d1=p%2;
    		int d2=m%2;
    		if(d1!=d2)c++;
    		m/=2;
    		p/=2;
    	}
    	cout<<c<<endl;
    	}
    	else cout<<"0"<<endl;
    }
    return 0;
}
