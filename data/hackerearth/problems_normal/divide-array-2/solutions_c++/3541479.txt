#include <iostream>
using namespace std;

int main()
{
    int N;
    cin>>N;
    
    int64_t arr[N];
    for(int i=0;i<N;i++){
    	cin>>arr[i];
    }
    
    int64_t a=1;
    
    int Q;
    cin>>Q;
    for(int i=0;i<Q;i++){
    	int D;
    	cin>>D;
    	
    	if(a*D<0){
    		a=1;
    		for(int i=0;i<N;i++){
    		arr[i]/=a;
    		}
    	}
    	a=a*D;
    	
    }
    
    for(int i=0;i<N;i++){
    	cout<<arr[i]/a<<" ";
    }
    
    
    return 0;
}
