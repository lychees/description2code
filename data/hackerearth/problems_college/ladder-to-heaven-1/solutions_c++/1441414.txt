#include <bits/stdc++.h>
#define ll long long 
#define LARGE 10000000000000000ll
using namespace std;

void multiply(ll F[2][2], ll M[2][2])
{
  	ll x =  F[0][0]*M[0][0] + F[0][1]*M[1][0];
  	ll y =  F[0][0]*M[0][1] + F[0][1]*M[1][1];
  	ll z =  F[1][0]*M[0][0] + F[1][1]*M[1][0];
  	ll w =  F[1][0]*M[0][1] + F[1][1]*M[1][1];
	 
  	F[0][0] = x;
  	F[0][1] = y;
  	F[1][0] = z;
  	F[1][1] = w;
}
 
void power(ll F[2][2], ll n)
{
  	if( n == 0 || n == 1)
      	return;
  	ll M[2][2] = {{1,1},{1,0}};
	 
	power(F, n/2);
  	multiply(F, F);
	 
  	if (n%2)
    	multiply(F, M);
}
 
/* function that returns nth Fibonacci number */
ll fib(ll n)
{
  	ll F[2][2] = {{1,1},{1,0}};
  	if (n == 0)
	    return 0;
  	power(F, n-1);
  	return F[0][0];
}
 
int main()
{
	ll t;
	cin >> t;
	while(t--)
	{
		ll n;
		cin >> n;
		ll sum;
		sum = 0;
		for(int i = 2;i <= INT_MAX; i++)
		{
			if(fib(i) > n) break;
			else if(fib(i)%2 == 0)
			{
				ll temp = fib(i);
				sum += temp;
			}
		}
		cout << sum << endl;
	}
  	return 0;
}
