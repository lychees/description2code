#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   scanf("%d",&t);
   while(t--)
   {
   	int n,k;
   	scanf("%d %d",&n,&k);
   	ll dp[n+5][n+5];
   	for(int i=0;i<=n;i++)
   	{
   		dp[i][0]=0;
   	}
   	for(int i=0;i<=n;i++)
   	{dp[0][i]=1;
   		
   	}
   	for(int i=1;i<=n;i++)
   	{
   		for(int j=1;j<=n;j++)
   		{
   			if(i<k || j<k)
   			dp[i][j]=0;
   			else
   			{dp[i][j]=dp[i][j-1];
   			if(i>=j)
   			dp[i][j]+=dp[i-j][j];}
   			
   		//	printf("%d ",dp[i][j]);
   		}
   	//	printf("\n");
   	}
   	printf("%lld\n",dp[n][n]);
   }
    return 0;
}
