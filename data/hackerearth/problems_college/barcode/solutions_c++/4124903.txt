#include<bits/stdc++.h>
using namespace std;

#define PB push_back
#define MP make_pair
#define F first
#define S second
//#define LET(it,a) __typeof(a) it(a)
#define LET(it,container) __typeof(container.begin()) it(container.begin())
#define ITER(it,container) for(typeof(container.begin()) it=container.begin();it!=container.end();it++)
#define LL long long
const int MOD = 1e9 + 7;
const int INF = INT_MAX;

int n,m,x,y;
char inp[2003][2003];
int col[2003];
int pre[2003];
int dp[2003][2];

int func(int pos,int ty)
{
	if(pos >= m) return 0;
	int &ret = dp[pos][ty];
	if(ret != -1) return ret;
	ret = INF;	
	ty = (ty + 1)%2;
	for(int i=x;i<=y;i++)
	{
		int diff = pre[pos+i] - pre[pos];
		if(ty == 1) diff = i*n - diff;
		ret = min( diff + func(pos+i,ty), ret);	
	}
	return ret;
}

int main()
{
	cin >> n >> m >> x >> y;
	for(int i=0;i<n;i++)
		scanf("%s",inp[i]);

	for(int j=0;j<m;j++)
	{
		for(int i=0;i<n;i++)
		{
			if(inp[i][j] == '.')
				col[j]++;
		}
	}
	pre[0] = 0;
	for(int j=1;j<=max(2000,m);j++)
		pre[j] = pre[j-1] + col[j-1];
	// 0 - '.' 1 - '#'
	memset(dp,-1,sizeof(dp));
	cout << min(func(0,0),func(0,1)) << endl;
	
	return 0;
}
