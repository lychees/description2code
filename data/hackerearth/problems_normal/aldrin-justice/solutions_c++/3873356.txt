#include <iostream>
using namespace std;

int main()
{
    int T,i,bt1,bt2,mt1,mt2;
    int temp;
    cin >> T;
    
    for(i=0;i<T;i++){
    	scanf("%d %d %d %d",&bt1,&bt2,&mt1,&mt2);
    	if(bt1 > bt2){
    		temp = bt1;
    		bt1 = bt2;
    		bt2 = temp;
    	}
    	if(mt1 > mt2){
    		temp = mt1;
    		mt1 = mt2;
    		mt2 = temp;
    	}
    	
    	if(bt1 <= mt1){
    		if(bt2 == mt1){
    			cout << "Point" << endl;
    		}
    		else if (bt2 < mt1){
    			cout << "Line" << endl;
    		}
    		else if(bt2 > mt1){
    			if(mt1 == mt2){
    				cout << "Point" << endl;
    			}
    			else{
    				cout << "Nothing" << endl;	
    			}
    			
    		}
    	}
    	else if(bt1 > mt1){
    		if(mt2 == bt1){
    			cout << "Point" << endl;
    		}
    		else if (mt2 < bt1){
    			cout << "Line" << endl;
    		}
    		else if(mt2 > bt1){
    			if(bt1 == bt2){
    				cout << "Point" << endl;
    			}
    			else{
    				cout << "Nothing" << endl;	
    			}
    		}
    	}
    }
    return 0;
}
