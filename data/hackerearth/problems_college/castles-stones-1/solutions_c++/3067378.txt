#include<iostream>
#include<algorithm>
using namespace std;
int main(){
int t,n,m,i,j,c;
cin>>t;
while(t--){
c=0;
cin>>m;
cin>>n;
int a[n];
for(i=0;i<n;i++)
cin>>a[i];
sort(a,a+n);
for(i=0,j=n-1;i<=j;){
if(a[j]>m)
j--;
else if(a[i]+a[j]>m)
{
c++;
j--;
}
else{
c++;
i++;
j--;
}
}
cout<<c<<endl;
}
return 0;}
