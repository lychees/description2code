#include <iostream>
#include <string>
#include <vector>

using namespace std;

int main(){
    int n;
    cin >> n;
    string s;
    cin >> s;
    
    vector<char> vec = {'a', 'e', 'i', 'o', 'u'};
    
    int flag = 0;
    
    for (int i=0; i<vec.size(); i++){
    	for(int j=0; j<s.size(); j++){
    		if(vec[i] == tolower(s[j])){
    			flag=1;
    			break;
    		}
    		else
    			flag=0;
    			
    	}
    	if (flag == 0)
    		break;
    }
    
    if(flag == 1)
    	cout << "YES" << endl;
    else
    	cout << "NO" << endl;
    	
    return 0;
}
