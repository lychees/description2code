#include<cstdio>
#include<iostream>
#include<cstring>
#include<sstream>
#include<stdlib.h>
#include<algorithm>
using namespace std;

int main()
{
	int arr[10];
	
    long long n,temp=0;
    long long mod=1000000007;
    cin>>n;
    if(n==0)
    	cout<<"10\n";
    else if(n<10)
    	cout<<n<<endl;
    else
    {
    	
    	for(int i=0;i<=mod;i++)
    	{
    		
    		memset(arr,0,sizeof(arr));
    		temp=n+i*mod;
    		for(int i=9;i>1;i--)
    		{
    			while(temp%i==0)
    			{
    				temp/=i;	
    				arr[i]++;
    			}
    		}
    		if(temp==1)
    			break;
    	}
    
    for(int i=2;i<=9;i++)
    	for(int j=1;j<=arr[i];j++)
    		cout<<i;
    cout<<endl;
    }
    return 0;
}
