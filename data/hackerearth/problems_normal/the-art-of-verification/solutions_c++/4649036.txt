    #include <bits/stdc++.h>
    using namespace std;
    int main()
    {
    int pos1,pos2;
    string S,username,pwd,profile,role,key;
    cin>>S;
    pos1=S.find("username=");
    pos2=S.find("&pwd");
    username=S.substr(pos1+9,(pos2-(pos1+9)));
    pos1=S.find("pwd=");
    pos2=S.find("&profile",++pos2);
    pwd=S.substr(pos1+4,(pos2-(pos1+4)));
    pos1=S.find("profile=");
    pos2=S.find("&role",++pos2);
    profile=S.substr(pos1+8,(pos2-(pos1+8)));
    pos1=S.find("role=");
    pos2=S.find("&key",++pos2);
    role=S.substr(pos1+5,(pos2-(pos1+5)));
    pos1=S.find("key=");
    pos2=S.find("&",++pos2);
    key=S.substr(pos1+4,(pos2-(pos1+4)));
    cout<<"username: "<<username<<endl;
    cout<<"pwd: "<<pwd<<endl;
    cout<<"profile: "<<profile<<endl;
    cout<<"role: "<<role<<endl;
    cout<<"key: "<<key<<endl;
    return 0;
    }
