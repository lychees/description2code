N = int(raw_input())
line = raw_input()

val = [0]*26
for i in range(N):
    val[ord(line[i])-ord('a')] += 1

Q = int(raw_input())
for q in range(Q):
    x, y = raw_input().split()
    if x == y:
        print(val[ord(x)-ord('a')] * (val[ord(x)-ord('a')]-1))/2
    else:
        print val[ord(x)-ord('a')]*val[ord(y)-ord('a')]
