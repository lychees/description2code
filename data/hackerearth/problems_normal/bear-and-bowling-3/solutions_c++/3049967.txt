#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define mod_val 1000000007
ll pow_val(ll base,ll num)
{
    if(num==0)
        return 1;
    else if(num==1)
        return base;
    else
    {
        ll ans=pow_val(base,num/2)%mod_val;
        if(num%2!=0)
            return ((ans%mod_val)*(ans%mod_val)*(base))%mod_val;
        return ((ans%mod_val)*(ans%mod_val))%mod_val;
    }
}
int main()
{
    ll total=0,a,b,num,test;
    vector<ll> arr(200001);
    scanf("%lld",&num);
    for(ll i=1;i<=num;i++)
        scanf("%lld",&arr[i]);
    ll first_fact=pow_val(2,num-1);
    ll second_fact=pow_val(2,num-2);
    for(ll i=1;i<=num;i++)
    {
        total=total+((arr[i]%mod_val)*(first_fact%mod_val))%mod_val;
        if(total>mod_val)
            total%=mod_val;
        first_fact+=second_fact;
        first_fact%=mod_val;
    }
    printf("%lld\n",total);
}
