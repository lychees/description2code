#define LOGN 20

class point{
    public:
        LL x,y;
        point(){x=0;y=0;}
        point(LL _x,LL _y) : x(_x),y(_y){}
        ~point(){}
};

class vect{
    public:
        LL x,y;
        vect(){x=0;y=0;}
        vect(point a,point b){x = b.x-a.x; y = b.y-a.y;}
        ~vect(){}
};

LL cross(vect a, vect b)
{
    return (a.x*b.y - b.x*a.y);
}
int main()
{

    int n;
    si(n);
    vector<point> arr;
    for(int i=0;i<n;i++)
    {
        LL a,b;
        sll(a);sll(b);
        arr.PB(point(a,b));
    }
    int ans=0;
    
    for(int i=0;i<n-1;i++)
    {
	    vect curr(arr[i],arr[i+1]);
	    int k=i+1;
	    ans++;
	    for(int j=i+2;j<n;j++)
	    {
		    vect b(arr[k],arr[j]);
		    if(cross(curr,b)>=0)
		    {
			    ans++;
			    curr = vect(arr[i],arr[j]);
			    k = j;
		    }
		    else
			    continue;
	    }
    }
    cout<<ans<<endl;
    return 0;
}
