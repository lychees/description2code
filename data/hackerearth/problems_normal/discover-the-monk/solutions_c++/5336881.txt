#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n,q;
    cin>>n>>q;
    int i;
    int arr[n];
    for(i=0;i<n;i++)
    {
    	cin>>arr[i];
    }
    sort(arr,arr+n);
    int temp;
    while(q--)
    {
    	cin>>temp;
    	if(binary_search(arr,arr+n,temp))
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
    
    return 0;
}
