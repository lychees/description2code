#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    long long a0[100001],a1[100001],n,t;
    scanf("%lld\n",&t);
    a0[1]=1;a0[2]=1;a1[1]=0;a1[2]=1;
    while(t--)
    {
    	scanf("%lld\n",&n);
    	for(int i=3;i<=n;i++)
    	{
    		a0[i]=a0[i-1]+a0[i-2];
    		a1[i]=a1[i-1]+a1[i-2];
    	}
    	cout<<a1[n]<<" "<<a0[n]<<endl;
    }
    return 0;
}
