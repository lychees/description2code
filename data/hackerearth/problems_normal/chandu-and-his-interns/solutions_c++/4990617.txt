#include<bits/stdc++.h>
using namespace std;
main()
{
ios_base::sync_with_stdio(0);cin.tie(0);
long long int t,n,i,count;
cin>>t;
while(t--)
{
count=0;
cin>>n;
i=2;
while(i<=sqrt(n))
{
if(n%i==0)
{
count++;
if((n/i)!=i)
count++;
if(count>1)
goto ll;
}
i++;
}
if(count<2)
cout<<"NO"<<endl;
else
ll:	cout<<"YES"<<endl;
}
}
