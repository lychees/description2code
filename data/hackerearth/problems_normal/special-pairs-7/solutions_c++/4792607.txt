#include <iostream> 
#include <map>
#include <set>

#include <deque>
#include <algorithm>
#include <string.h>

#define fi first
#define se second
#define pb push_back
#define mp make_pair
#define endl '\n'



using namespace std;

typedef long long ll;
typedef vector<int> vi;
typedef pair<int, int> ii;

const int full = (1 << 20) - 1;

int pre[1 << 20];
int cnt[1 << 20];
vi v;

int main() {
    ios_base::sync_with_stdio(0);
    int T;
    cin >> T;
    while(T--) {
        v.clear();
        int n;
        cin >> n;

        ll ans = 0;
        memset(cnt, 0, sizeof cnt);
        for(int i = 0; i < n; i++) {
            int aux;
            cin >> aux; 
            v.pb(aux);
            cnt[aux]++;
        }

        for(int i = 0; i < (1 << 20); i++) {
            pre[i]=cnt[i];
        }


        for(int i = 0; i < 20; i++) {
            for(int j = 0; j < (1 << 20); j++) {
                if(j & (1 << i)) {
                    pre[j]+=pre[j ^ (1 << i)];
                }
            }
        }

        for(int i = 0; i < n; i++) {
            if(v[i] == 0) {
                ans += pre[v[i]^full];
            }else {
                ans += pre[v[i]^full];
            }
        }

        cout << ans << endl;
    }
    return 0;
}
