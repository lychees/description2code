#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair

#define MAG 10000001
int pr[MAG+31];
int phi[MAG+31];
void sieve()
{
    ll i,j;
    for(i = 1LL; i <= 10000000LL; i++) {
       phi[i] = i;
    }
    for(i = 2LL; i <= 10000000LL; i++) {
        if(phi[i] == i) {
            for(j = 2; j*i <= 10000000LL; j++) {
                phi[j*i] = (i-1)*phi[j*i]/i;   
            }
            phi[i] = i-1;
            //x.push_back(i);
        }
    }
}


long long S[MAG+31];
int tests,n;
 
int main(){
	sieve();
	for (int i=1;i<=MAG;i++)
		S[i]=S[i-1]+phi[i];
	
	cin>>tests;
	for (;tests;--tests)
	{
		cin>>n;
		
		cout<<S[n]<<endl;
	}
	
	return 0;
}
