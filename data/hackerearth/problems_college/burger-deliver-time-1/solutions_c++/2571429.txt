#include<iostream>
#include<algorithm>
using namespace std;
int main()
{ int a[1000],b[1000],n,i,j,ti,di;
 cin>>n;
 for(i=0;i<n;i++)
 {cin>>ti>>di;
  a[i]=b[i]=ti+di;
 }
 sort(a,a+n);
 for(i=0;i<n;i++)
 {for(j=0;j<n;j++)
 {if(a[i]==b[j])
     {cout<<j+1<<" ";
     b[j]=0;
     break;}
 }
 }
 return 0;
}
