#define MOD 1000000007
LL arr[MAX];
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    int t;
    LL minn,d,ans,n,g;
    cin>>t;
    while(t--)
    {
        cin>>n;
        minn = INT_MAX;
        for(int i=0;i<n;i++){
            cin>>arr[i];
            minn = min(minn,arr[i]);
        }
        sort(arr,arr+n);
        g = arr[0];
        for(int i=1;i<n;i++){
            g = __gcd(g,arr[i]);
            d = arr[i] - arr[i-1];
            if(d != 0)
                minn = min(minn,d);
        }
        ans = g*n;
        cout<<ans<<endl;
    }
    return 0; 
}
