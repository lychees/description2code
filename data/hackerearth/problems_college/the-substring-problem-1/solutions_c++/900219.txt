#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int l,i,n,t,flag[26],temp[26];
    char str[105],ans;
    
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<26;i++)
    	flag[i]=0;
    	while(n--)
    	{
    		cin>>str;
    		l=strlen(str);
    		for(i=0;i<26;i++)
    		temp[i]=0;
    		for(i=0;i<l;i++)
    		{
    		if(temp[str[i]-97]==0)
    		{
    		flag[str[i]-97]++;
    		temp[str[i]-97]=1;
    		}
    		}
    	}
    	ans='a';
    	for(i=1;i<26;i++)
    	if(flag[i]>flag[ans-97])
    	ans=i+'a';
    	cout<<ans<<endl;
    	
    }
    return 0;
}
