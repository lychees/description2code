#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t, n;
	scanf("%d", &t);
	while (t--) {
		scanf("%d", &n);
		if (n == 1) {
			printf("Pappu Bhujia\n");
			continue;
		}

		vector<int> v;
		int srt = (int)sqrt(n);
		for (int i = 2; i <= srt; i++)	{
			int cnt = 0;
			while (n % i == 0) {
				n /= i;
				cnt++;
			}
			if (cnt != 0)
				v.push_back(cnt);
		}

		if (n > 1)
			v.push_back(1);

		int ans = v[0];
		for (int i = 1; i < v.size(); i++)
			ans ^= v[i];

		if (ans == 0)
			printf("Pappu Bhujia\n");
		else
			printf("Printf{}\n");
	}
}
