#include <iostream>
#include<algorithm>
#include<climits>
using namespace std;

int main()
{
    int n,k;
    cin>>n>>k;
    int arr[n];
    for(int i=0; i<n; i++)
    cin>>arr[i];
    
    sort(arr,arr+n);
    reverse(arr,arr+n);
    int diff, min = INT_MAX;
    
    for(int i=0; i<=n-k; i++)
    {
    	int diff = arr[i] - arr[i+k-1];
    	if(diff<min)
    	min = diff;
    }
    
    cout<<min<<"\n";
    return 0;
}
