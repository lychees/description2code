#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll cmp(ll a,ll b)
{
	return a>b;
}
int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		ll n;
		cin>>n;
		ll a[n],i;
		for(i=0;i<n;i++)
			cin>>a[i];
		int c=0;
		for(int j=0;j<n;j++)
		{
			sort(a,a+n,cmp);
			for(i=0;i<=c;i++)
			{
				a[i]--;
				if(a[i]<0)
					break;
			}
			if(a[i]<0)
				break;
			c++;
		}
		cout<<c<<'\n';
	}
	
	return 0;
}
