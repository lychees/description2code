#include<bits/stdc++.h>

using namespace std;

typedef struct pair<int,int> pii;

bool cmp(pii x,pii y)
{
	if(x.first==y.first)
		return x.second<y.second;
	return x.first<y.first;
}

int main()
{
	int n,a,b;
	cin>>n;
	pii arr[n+1];
	for(int i=0;i<n;i++)
	{
		cin>>a>>b;
		arr[i].first=a+b;
		arr[i].second=i;
	}
	sort(arr,arr+n,cmp);
	for(int i=0;i<n;i++)
	{
		cout<<arr[i].second+1<<" ";
	}
	cout<<endl;
	return 0;
}
