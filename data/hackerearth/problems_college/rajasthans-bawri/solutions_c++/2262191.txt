#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    int px,py,bx,by;
    while(t--){
     scanf("%d%d%d%d",&px,&py,&bx,&by);	
     if(((px>=bx)&&(py>=by))||((px>=by)&&(py>=bx))){
    printf("Possible\n");
    }
    else{
     printf("Not Possible\n");	
    }
    }
    return 0;
}
