from math import sqrt
T=int(input())
for i in range(T):
	x1,y1,x2,y2=[int(x) for x in raw_input().split()]
	r=int(input());
	dx=x2-x1
	dy=y2-y1
	temp = (dx*dx)+(dy*dy)
	dr=sqrt(temp)
	D=x1*y2-x2*y1
	dis=(r*r)*(dr*dr)-(D*D)
	if dis>0:
		print 'REPLANNING'
	elif dis <0:
		print 'SAFE'
	else:
		print 'JUST MISSED'
