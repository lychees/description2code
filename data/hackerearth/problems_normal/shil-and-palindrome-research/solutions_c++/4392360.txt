#include<bits/stdc++.h>
using namespace std;

int read(int fenwick[],int idx)
{
    int sum=0;
    while(idx>0)
    {
        sum=sum+fenwick[idx];
        idx=idx-(idx & -idx);
    }
    return sum;
}
void add(int fenwick[],int idx,int value,int n)
{
    while(idx<=n)
    {
        fenwick[idx]=fenwick[idx]+value;
        idx=idx+(idx & -idx);
    }
}
int read_range(int fenwick[],int L,int R)
{
    return (read(fenwick,R)-read(fenwick,L-1));
}


int main()
{
    int n,q;
    scanf("%d%d",&n,&q);
    char str[n+5];
    int fenwick[26][n+5];
    for(int i=0;i<26;i++)
    {
        for(int j=0;j<=n;j++)
            fenwick[i][j]=0;
    }
    scanf("%s",str);
    int i=0;
    while(str[i]!='\0')
    {
        add(fenwick[str[i]-'a'],i+1,1,n);
        i++;
    }
    int typ;
    char ch,sp;
    int l,r;
    while(q--)
    {
        scanf("%d",&typ);
        if(typ==1)
        {
            scanf("%d%c%c",&l,&sp,&ch);
            add(fenwick[str[l-1]-'a'],l,-1,n);
            add(fenwick[ch-'a'],l,1,n);
            str[l-1]=ch;
        }
        else
        {
            scanf("%d%d",&l,&r);
            int flag=0;
            for(i=0;i<26;i++)
            {
                if((read_range(fenwick[i],l,r))%2==1)
                {
                    if(flag==0)
                        flag=1;
                    else
                    {
                        printf("no\n");
                        break;
                    }
                }
            }
            if(i==26)
                printf("yes\n");
        }
    }

}
