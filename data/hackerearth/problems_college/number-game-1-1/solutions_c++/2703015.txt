#include <iostream>
using namespace std;

int main()
{
    long long int t,n,cnt;
    cin>>t;
    while(t!=0)
    {
    	cnt=0;
    	cin>>n;
    	while(n!=1)
    	{
    	if(n%3==0)
    	{
    		n=n/3;
    	}
        else if(n%2==0)
    	{
    		n=n/2;
    	}
    	else
    	{
    		n=n-1;
    	}
    	cnt++;
    	}
    	cout<<cnt<<"\n";
    	t--;
    }
    return 0;
}
