#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdio>

using namespace std;

int main()
{
	int t,p,n,count;
//	vector<int>a;
    scanf("%d",&t);
    while(t--){
    	count=0;
    	scanf("%d",&n);
    	vector<int>a;
    	for(int i=0;i<n;++i)
    	{
    		scanf("%d",&p);
    		a.push_back(p);
    	}
    	  for(int j=0;j<n;++j){
        	   	   	   for(int k=j;k<n;++k){
        	   	   	   	 if(a[j]==a[k])
        	   	   	   	 	++count;
        	   	   	   }
        	   	   }
        	   	   cout<<count<<endl;
    }
    return 0;
}
