#include<bits/stdc++.h>
using namespace std;
#define lx long long
lx pw[1<<20];
lx s[1<<20];
unordered_set <lx> sset;
string st;
int check(int span)
{
    sset.clear();
    for (int i=0;i+span<=st.size();i++)
    {
        lx q = s[i+span]-s[i];
        q *= pw[1000000-i];
        if (sset.find(q)!=sset.end())
            return true;
        sset.insert(q);
    }
    return false;
}
int main()
{
    ios::sync_with_stdio(0);
    //freopen("input.txt","r",stdin);
    cin>>st;
    pw[0]=1;
    for (int i=1;i<=1000000;i++)
        pw[i]=pw[i-1]*173;
    for (int i=1;i<=st.size();i++)
        s[i]=s[i-1]+st[i-1]*pw[i-1];
    int l,r;
    l=0;
    r=st.size();
    while(l<r)
    {
        int mid = l+r+1;
        mid/=2;
        if (check(mid))
            l=mid;
        else
            r=mid-1;
    }
    cout<<l<<endl;
    return 0;
}
