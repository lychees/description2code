#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    int T,i,flag=0;
    string str;
    cin>>T;
    while(T--)
    {
        cin>>str;
        for(i=0;i<=str.size()-1;i++)
        {
            if(str[i]==str[i+1])
            {
                flag=0;
                break;
            }
            else
                flag=1;
        }
        if(flag==1)
            cout<<"KISS"<<endl;
        else
            cout<<"SLAP"<<endl;
    }
}
