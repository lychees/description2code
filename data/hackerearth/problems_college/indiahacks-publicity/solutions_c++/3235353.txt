#include<bits/stdc++.h>
using namespace std;

bool visited[100005];
vector<int>v[100005];

void relator(int i){                    //function which cancels all the friends related to the person
	visited[i]=1;
	for(int j=0;j<v[i].size();j++){
		if(!visited[v[i][j]])
			relator(v[i][j]);
	}
}



int main(){
	int t,a,b;
	scanf("%d",&t);
	while(t--){
		int n,m;
		scanf("%d%d",&n,&m);
		memset(v,0,sizeof(v));
        memset(visited,0,sizeof(visited));
		while(m--){
			scanf("%d%d",&a,&b);
			v[a].push_back(b);
			v[b].push_back(a);
		}
		
		
        int count=0;
		for(int i=0;i<n;i++){
			if(!visited[i]){
				count++;
				relator(i);
			}
		}
		printf("%d\n",count);
	}
}
