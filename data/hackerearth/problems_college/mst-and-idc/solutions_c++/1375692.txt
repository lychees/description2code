#include<iostream>
using namespace std;
#include<algorithm>
int main()
{
	int t,n;
	long long int s;
	cin>>t;
	while(t--)
	{s=0;
		cin>>n;
		long long int a[n];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a,a+n);
		for(int i=0;i<n-1;i++)
		{
			s=s+(a[i+1]-a[i]);
		}
		cout<<2*s<<endl;
		
	}
	
	return 0;
}
