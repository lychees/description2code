#include <iostream>
using namespace std;
 
int main()
{
    int l,mid,right,left,t,n,i,y;
    long num,nm,a[100001]={0};
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	l=0;l=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>num;
    		nm=num-i;
    		if(nm>0)
    		{
    			left=0;right=l-1;y=-1;
    			while(left<=right)
    			{
    				mid=(left+right)>>1;
    				if(a[mid]<=nm)
    				 left=mid+1;
    				else
    				{
    					y=mid;
    					right=mid-1;
    				}
    			}
    			a[(y>=0) ? (y) : (l++)]=nm;
    		}	
    	}
    	cout<<n-l<<endl;
    }
    return 0;
}
