#include<iostream>
#include<algorithm>
#include<string.h>
using namespace std;

long long a[100005];

int main()
{
    long long i,j,n,m,d,ans=0,k,x;
    bool isPossible=true;
    cin>>n>>k>>d;
    a[0]=0;
    for(i=1;i<=n;i++) {
        cin>>a[i];
        a[i]+=a[i-1];
    }
    for(i=1;i<=n-k+1 && isPossible;i++) {
        long long temp=a[i+k-1]-a[i-1];
        if(temp<d) {
            for(j=i+k-1;j>=i;j--) {
                if(a[j]-a[j-1]>0) {
                    ans++;
                    for(x=j+k-1;x>=j;x--) {
                        if(a[x]-a[x-1]>0) {
                            break;
                        }
                    }
                    break;
                }
            }
            if(j<i) isPossible=false;
            i=x;
        }
    }
    if(isPossible) cout<<ans;
    else cout<<"-1";
    return 0;
}
