#include<iostream>
#include<math.h>
using namespace std;
#define k 1000000

long long a[k+5],g[k+5];

int main()
{
for(long long i=2;i<=1000;i++)
{
 for(long long j=2;i*j<=k;j++)
 {
    long long x=i*j;
      if(i<sqrt(x))
      {
          g[x]+=(i+j);
      }
      else if(i==sqrt(x))
      {
           g[x]+=i;
           

      }
      if(a[x]==0)
           {
               a[x]=i;
           }

 }
}


     for(long long i=2;i<=k;i++)
     {
             if(a[i]==0)
             a[i]=i;
     }
     for(long long i=2;i<=k;i++)
     a[i]=a[i-1]+a[i];


     for(long long i=1;i<=k;i++)
     {
           g[i]+=1;
     }


     g[1]=0;
     for(long long i=2;i<=k;i++)
     g[i]=(g[i-1]+g[i]);


    int t;
    cin>>t;
    while(t--)
    {
              int n;
              cin>>n;
              cout<<(a[n]+g[n])%n<<endl;
    }
}
