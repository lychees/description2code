#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;


int main() {
    int t;
    cin>>t;
    int k;
    string s;
    while(t--){
        cin>>k;
        cin>>s;
        int cnt = 0;
        for(int i=0,sum=0,j=0;i<s.length();i++){
            for(;j<s.length() && sum<k;j++){
                sum += s[j]-'a'+1;
                //cout << sum << endl;
                if (sum==k) cnt++;
                if (sum>=k){
                    sum -= s[j]-'a'+1;
                    break;
                }
            }
            
            sum -= s[i]-'a'+1;
        }
        cout << cnt << "\n";
    }
    return 0;
}
