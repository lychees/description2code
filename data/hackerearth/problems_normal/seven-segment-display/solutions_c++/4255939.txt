#include<bits/stdc++.h>
using namespace std;
int main()
{
   char a[100005];
   int i,j,k,test,ans;
   cin>>test;
   while(test--)
   {
   	ans=0;
   	cin>>a;
   	k=strlen(a);
   	for(i=0;i<k;i++)
   	{
   		if(a[i]=='1')
   		ans=ans+2;
   		else if(a[i]==48||a[i]==54||a[i]==57)
   		ans+=6;
   		else if(a[i]==50||a[i]==51||a[i]==53)
   		ans+=5;
   		else if(a[i]==56)
   		ans+=7;
   		else if(a[i]==55)
   		ans+=3;
   		if(a[i]==52)
   		ans+=4;
   	}
   	cout<<ans<<endl;
   }
   return 0;
}
