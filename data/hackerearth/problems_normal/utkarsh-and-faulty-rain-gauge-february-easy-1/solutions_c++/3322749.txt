#include<bits/stdc++.h>
using namespace std;

const int N=1010;
typedef long long ll;


int main() {
  //freopen("a.cpp","r",stdin);
  double sq[10010];
  sq[1]=1;
  for(int i=2;i<10010;i++) {
    sq[i]=i*i;
    sq[i]+=sq[i-1];
  }
  int n;
  cin>>n;
  double a[n],p[n];
  for(int i=0;i<n;i++) {
    cin>>a[i]>>p[i];
  }
  double res[n];
  memset(res,0,sizeof(res));
  for(int i=0;i<n;i++) {
    for(int j=1;j<=a[i];j++) {
      res[i]+=((double)j/(int)p[i])*((j*j)/sq[(int)a[i]]);
    }
  }
  double ans=res[0];
  for(int i=1;i<n;i++) {
    ans*=res[i];
  }
  printf("%.4lf\012",ans);
  return 0;
}
