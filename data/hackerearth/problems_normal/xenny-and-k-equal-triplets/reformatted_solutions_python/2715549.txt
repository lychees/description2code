from fractions import gcd
t = int(raw_input())
while t > 0:
    n, k = (raw_input().split())
    n, k = long(n), long(k)
    nums = raw_input().split()
    nums = [long(x) for x in nums]
    c = nums.count(k)
    numerator = 0
    if (c >= 3):
        numerator = c*(c-1)*(c-2)
    denominator = n*(n-1)*(n-2)
    g = gcd(numerator, denominator)
    print str(numerator/g)+"/"+str(denominator/g)
    t = t-1
