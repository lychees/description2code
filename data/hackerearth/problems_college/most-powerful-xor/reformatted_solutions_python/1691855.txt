
t = int(raw_input())
while t > 0:
    t -= 1
    line = raw_input().split(' ')
    a = int(line[0])
    b = int(line[1])
    ax = 0
    for i in range(a, b+1):
        for j in range(a, b+1):
            # print i
            # print j
            temp = i ^ j
            if ax < temp:
                ax = temp
    print ax
