#include<bits/stdc++.h>
using namespace std;
#define ll long long

ll solve(map<ll,ll> &mp,ll n)
{
	if(mp.find(n)!=mp.end())
	return mp[n];
	else{
	ll res=solve(mp,n/2)+solve(mp,n/3)+solve(mp,n/4);
	if(res>=n)
	mp[n]=res;
	else
	mp[n]=n;
	return mp[n];}
}


int main()
{
   ll n;
   map<ll,ll>mp;
   mp.insert(make_pair(0,0));
   mp.insert(make_pair(1,1));
   mp.insert(make_pair(2,2));
   mp.insert(make_pair(3,3));
   while(scanf("%lld",&n)>0)
   {
   	  
      ll ans=0;
      ans=solve(mp,n);
      printf("%lld",ans);
      cout<<endl;
     // mp.clear();
      

   }
   return 0;
}
