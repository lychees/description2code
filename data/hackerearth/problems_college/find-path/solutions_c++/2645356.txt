#include<bits/stdc++.h>

using namespace std;
 
long long gcd(long long a,long long b){
	while(b){
		long long temp = b;
		b=a%b;
		a=temp;
	}
	return a;
}
 
 
int main(){
	int t;
	cin>>t;
	while(t--){
		long long n;
		cin>>n;
		long long a[n];
		for(int i = 0; i < n;i++){
			cin>>a[i];
		}
		vector<long long> v(10000001,2000000001);
		for(int i = 0;i<n;i++){
			if(a[i]>i){
				
				vector<long long>::iterator it = upper_bound(v.begin(),v.end(),a[i]-i);
				*it = a[i]-i;
			}
		}
		cout<<n-(find(v.begin(),v.end(),2000000001)-v.begin())<<endl;
	}
	
	
}
