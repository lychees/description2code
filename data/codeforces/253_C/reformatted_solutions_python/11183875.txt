import sys
sys.stdin = open("input.txt")
sys.stdout = open("output.txt", 'w')
n = input()


def I():
    return map(int, raw_input().split())


a = I()
r, c, R, C = I()


def m(x, y):
    return min(a[min(x, y)-1:max(x, y)])+1


print min(abs(r-i-1)+abs(R-i-1)+abs(C-min(c, m(i+1, r), m(i+1, R))) for i in range(n))
