#include <bits/stdc++.h>

using namespace std;

int main() {

  int N;
  cin >> N;
  map<char, int> count;
  string s;
  while(N--) {
    cin >> s;
    count[s[0]]++;
  }

  int res = 0;
  for (auto k : count) {
    res += (k.second/10);
    if (k.second%10 != 0) res++;
  }

  cout << res << endl;

  return 0;
}
