long long calc(long long n) {
    long long result = 0;
    int v = 1;
    long long cnt = 0;
    for (;;) {
        long long tt = cnt + v;
        if (tt > n) break;
        if (v & 1) {
            result += v;
        }
        cnt = tt;
        ++v;
    }
    long long k = n - cnt;
    if (k & 1) {
        result += v;
    }
    return result;
}

        long long l, r;

    }
}
