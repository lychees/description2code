n = int(raw_input())
k = map(int, raw_input().split())
a = []
b = []
suma = 0
sumb = 0
for i in k:
    if i % 2 == 0:
        a.append(i)
    else:
        b.append(i)
a = sorted(a)
b = sorted(b)
suma = sum(a)
sumb = sum(b)
a = map(str, a)
b = map(str, b)
print ' '.join(a) + ' '+str(suma)+' '+' '.join(b)+' '+str(sumb)
