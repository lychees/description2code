#define FOR(i,a,b) for(int i=(a);i<=(b);i++)
#define FORD(i,a,b) for(int i=(a);i>=(b);i--)
#define REP(i,b) for(int i=0;i<(b);i++)

using namespace std;

#define MAXN 1111111
int Min[MAXN];
int Blocked[MAXN];

int main(int argc, char *argv[]) {
  int N, K, M;
  scanf("%d%d%d", &N, &K, &M);
  REP(i, M) {
    int B; // TODO: B = 1 ????
    scanf("%d", &B);
    if (B < MAXN) Blocked[B] = true;
  }
  printf("1");
  int cur = 1;
  REP(i, N - 1) {
    while (true) {
      cur++;
      Min[cur] = K + 1;
      for (int i = 2; i * i <= cur; ++i) if (cur % i == 0) {
        Min[cur] = min(Min[cur], Min[i] + Min[cur / i]);
      }
      if (Min[cur] > K && !Blocked[cur]) break;
    }
    printf(" %d", cur);
    Min[cur] = 1;
  }
  printf("\n");
  return 0;
}
