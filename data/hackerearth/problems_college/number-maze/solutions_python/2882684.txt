import sys
result=list()
tlist=[ int(x) for x in raw_input().split(' ')]
for i in range (tlist[0]+1,tlist[1]):
    if i%3==0 and i%5==0:
        result.append(i)
    elif i%3>0 and i%5>0:
        result.append(i)

for i in range (0,len(result)):
    if i!=len(result)-1:
        sys.stdout.write(str(result[i])+",")
    else:
        sys.stdout.write(str(result[i]))
