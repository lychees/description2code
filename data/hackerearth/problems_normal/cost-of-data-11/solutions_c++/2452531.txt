			struct node
			{
				node* child[26];
				node()
				{for(int i=0;i<26;i++)child[i]=0;}
			};
			int total_nodes=1;
			node* root=new node();
			void insert(string &s)
			{
				node * curr=root;
				for(int i=0;i<s.length();i++)
				{
					int CC=s[i]-'a';
					if(curr->child[CC]==0)
					{
						total_nodes++;
						curr->child[CC]=new node();
					}
					curr=curr->child[CC];
				}
			}
			int main()
			{
				fast;
				int n;
				cin>>n;
				for(int i=0;i<n;i++)
				{
					string s;
					cin>>s;
					insert(s);
				}
				cout<<total_nodes<<endl;
			   return 0;
			}
