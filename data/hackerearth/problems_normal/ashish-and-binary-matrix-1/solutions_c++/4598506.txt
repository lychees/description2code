#include <iostream>
#include <set>

#define MAX 1005
#define MOD 1000000007

using namespace std;

unsigned int hashs(string s)
{
	unsigned int val = 0;
	unsigned int mul = 1;
	
	for (int i = 0; i < s.length(); ++i)
	{
		val += mul * (s[i] - '0');
		mul *= 2;
	}
	
	return val;
}

int main()
{
    int t;
    int n, m;
    string s[MAX];
    unsigned int h[MAX];
    set<unsigned int> st;
    bool ans;
    
    cin >> t;
    while (t--)
    {
    	cin >> n >> m;
    	for (int i = 0; i < n; ++i)
    	{
    		cin >> s[i];
    		h[i] = hashs(s[i]);
    	}
    	
    	int mul = 1;
    	ans = false;
    	for (int c = 0; c < m; ++c)
    	{
    		st.clear();
    		for (int i = 0; i < n; ++i)
    		{
    			st.insert(h[i] - (s[i][c] - '0') * mul);
    		}
    		if (st.size() == n)
    		{
    			ans = true;
    			break;
    		}
    		if (ans)
    			break;
    	}
    	
    	if (ans)
    		cout << "Yes" << endl;
    	else
    		cout << "No" << endl;
    }
    
    return 0;
}
