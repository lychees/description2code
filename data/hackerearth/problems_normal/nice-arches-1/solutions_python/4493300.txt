n=int(raw_input())

	
count=0
for i in range(0,n):
	st=raw_input()
	l=[]
	for j in st:
		if(len(l)==0):
			l.append(j)
			continue
		top=l.pop()
		if(j == top):
			continue
		elif(j!=top):
			l.append(top)
			l.append(j)
			
	if(len(l)==0):
		count=count+1
		
print count
