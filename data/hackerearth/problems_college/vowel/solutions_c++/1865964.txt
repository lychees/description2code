#include<bits/stdc++.h>
using namespace std;

bool isvowel(char c) {
	if(c == 'A' || c == 'E' || c == 'I' || c == 'O' || c == 'U' || c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u')
		return true;
	else
		return false;
}

int main() {
	int t;
	cin >> t;

	while(t--) {
		string s;
		bool hasvowel = false;
		cin >> s;

		for(int i = 0 ; i < s.size() ; i++) {
			if(isvowel(s[i])){
				printf("%c",s[i]);
				hasvowel = true;
			}
		}
		if(!hasvowel){
			printf("No");
		}
		printf("\n");
	}
	return 0;
}
