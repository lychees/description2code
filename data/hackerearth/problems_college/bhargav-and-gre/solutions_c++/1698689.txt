#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int len=s.length();
    int n;
    cin>>n;
    string a[n];
    int count=0;
    while(n--)
    {
    	string temp;
    	cin>>temp;
    	if(temp.length()==len)
    	{
    		a[count]=temp;
    	}
    	count++;
    	
    }
    int syn=0,ant=0;
    int com=s[0];
    for(int i=0;i<count;i++)
    {
    	int j=a[i][0];
    	if(com==j)
    	syn++;
    	else if(com==(j-3))
    	ant++;
    }
    cout<<syn<<" "<<ant<<"\n";
    return 0;
}
