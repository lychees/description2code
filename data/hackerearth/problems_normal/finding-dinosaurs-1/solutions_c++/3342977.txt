#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,k,l,q,i,j,d;
    scanf("%d%d%d%d",&n,&k,&l,&q);
    map<string,string> map;
    for(i=0;i<n;i++){
    	string str;
    	cin>>str;
    	ostringstream oss("");
    	for(j=0;j<k;j++){
    		
    		scanf("%d",&d);
    		
    		oss<<d;
    		oss<<'@';
    		
    	}
    	map[oss.str()]=str;
    }
    for(i=0;i<q;i++){
    	ostringstream ss("");
    	for(j=0;j<k;j++){
    		scanf("%d",&d);
    		ss<<d;
    		ss<<'@';
    	}
    	string f=ss.str();
    	int flag=0;
    	if(map.find(f)!=map.end()){
    		cout<<map[f]<<endl;
    		flag=1;
    	}
    	if(flag!=1){
    	printf("You cant fool me :P\n");
    	}
    }
    return 0;
}
