#include <iostream>
using namespace std;

int main()
{
    int n,i,j,k;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	if(i==0)
    	{
    		if(n==1)
    		cout<<i<<" 0 0"<<"\n";
    		else
    		cout<<i<<" 0 1 \n";
    	}
    	else
    	{
    		j=i*2;
    		j=j%n;
    		cout<<i<<" "<<j<<" ";
    		j=i*2+1;
    		j%=n;
    		cout<<j<<"\n";
    	}
    }
}
