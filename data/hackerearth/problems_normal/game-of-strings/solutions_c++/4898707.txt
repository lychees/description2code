#include<bits/stdc++.h>
using namespace std;
int main()
{
	int m,n,i,j;
	char str1[5001],str2[5001];
    cin>>str1;
    cin>>str2;
    m=strlen(str1);
    n=strlen(str2);
    int dp[m+1][n+1];
    memset(dp,0,sizeof(dp));
    for(i=1;i<=m;i++)
    {
        for(j=1;j<=n;j++)
        {
            if(str1[i-1]==str2[j-1]&&str1[i-1]!='a'&&str1[i-1]!='e'&&str1[i-1]!='i'&&str1[i-1]!='o'&&str1[i-1]!='u')
            {
                dp[i][j]=dp[i-1][j-1]+1;
            }
            else
            {
                dp[i][j]=max(dp[i-1][j],dp[i][j-1]);
            }
        }
    }    
    cout<<dp[m][n]<<endl;
}
