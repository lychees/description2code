#include <iostream>
#include <math.h>
using namespace std;

long func(long a)
{
	double res = sqrt((2*a)+0.25);
	res-=0.5;
	long result = (long)ceil(res);
	return result;
}

long funca(long k)
{
	return ((k*(k+1))/2);
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long n;
    	cin>>n;
    	long a = func(n);
    	while(funca(a)<n)
    		a++;
    	cout<<a<<endl;
    }
    return 0;
}
