#include<bits/stdc++.h>
using namespace std;
int main()
{
	int a,n,i,max,p,cnt[26],checkp[26];
	char arr[101];
	cin>>a;
	while(a--)
	{
		cin>>n;
		max=0;p=0;
		memset(cnt,0,sizeof(cnt));
		while(n--)
		{
			cin>>arr;
			memset(checkp,0,sizeof(checkp));
			for(i=0;arr[i]!=NULL;++i)
			{
				if(!checkp[(int)arr[i]-97])
				{
					++cnt[(int)arr[i]-97];
					checkp[(int)arr[i]-97]=1;
				}	
			}
		}
		for(i=0;i<26;++i)
		{
			if(cnt[i]>max)
			{
				p=i;
				max=cnt[i];
			}
		}
		cout<<(char)(p+97)<<endl;
	}
	return 0;
}
