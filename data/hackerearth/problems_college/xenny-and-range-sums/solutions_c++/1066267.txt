#include <iostream>
#include <algorithm>
#include <cmath>
#include <climits>
#include <vector>
#include <utility>
using namespace std;    
int main()
{
	int n,q;
	cin>>n>>q;
	int a[n];
	int arr[n];
	cin>>a[0];
	arr[0]=a[0];
	for(int i=1;i<n;i++)
	{
		cin>>a[i];
		arr[i]=arr[i-1]+a[i];
	}
    while(q--)
    {
            int P,Q;
            cin>>P>>Q;
            P--;
            P--;
            Q--;
            long long int ans;
            if(P<0)
            	cout<<arr[Q]<<" "<<a[Q]-arr[Q-1]<<endl;
            else
            {
            	long long int ans;
            	ans=arr[Q-1]-arr[P];
            	cout<<ans+a[Q]<<" "<<a[Q]-ans<<endl;
            }

        
            
    }
}
