#include<iostream>
long long ssq(int n){
	long long s=0;
	while(n){
		s+=(n%10)*(n%10);
		n/=10;
	}
	return s;
}
using namespace std;
int main(){
	int t,s;
	cin>>t;
	s=t;
	long long n,z;
	while(t--){
		cin>>n;
		
	     z=ssq(n);
	     while(z>=10){
	     	z=ssq(z);
	     }
	     
	    if(z==1){
	    	cout<<"Case #"<<s-t<<":YES"<<endl;
	    }
	    else
	    cout<<"Case #"<<s-t<<":NO"<<endl;
	}
	return 0;
}
