#include <bits/stdc++.h>
using namespace std;
int arr[51];
int n,m;
int dd(int ind,int sm){
	
	if(sm>n)
		return 0;
	if(sm==n)
		return 1;
	if(ind>=m)
		return 0;
	return dd(ind,sm+arr[ind])+dd(ind+1,sm);
}
int main()
{
//    cout << "Hello World!" << endl;
	//int n,m;
	cin>>n>>m;
	for(int i=0;i<m;i++){
		cin>>arr[i];
	}
	sort(arr,arr+m);
	 long long  table[n+1];
 
    
    memset(table, 0, sizeof(table));
     table[0] = 1;
    for(int i=0; i<m; i++)
        for(int j=arr[i]; j<=n; j++)
            table[j] += table[j-arr[i]];
 
    cout<<table[n];
    return 0;
}
