#include <bits/stdc++.h>
#define ll long long int
using namespace std;
ll fact(int n)
{
	if(n==1)
		return 1;
	else
		return n*fact(n-1);
}
int main()
{
	ll t,i,a,b,sum;
	
	scanf("%lld",&t);
	while(t--)
	{
		sum=0;
		scanf("%lld %lld",&a,&b);
		for(i=a;i<=b;i++)
		{
			sum=(sum+fact(i)%10)%10;
		}
		printf("%d\n",sum%10);
	}
	return 0;
}
