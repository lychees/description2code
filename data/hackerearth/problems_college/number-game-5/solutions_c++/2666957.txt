#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	ios::sync_with_stdio(false);
	vector<long long> pre;
		long long sum = 0;
		long long i=1,j=2;
		while(sum <= 100000000000000000ll){
			pre.push_back(sum);
			sum += i * j*j;
			i++;
			j++;
		}
		int t;
		cin >> t;
		std::vector<long long>::iterator low;
	
		while(t-->0){
			long long n;
			cin >> n;
			if(n<(long long)4){
				cout << 0 << endl << "Draw" << endl;
				continue;
			}
			if(n==(long long)4){
				cout << 1 << endl << "Dhruv" << endl;
				continue;
			}
			low=lower_bound (pre.begin(), pre.end(), n);
			if(*low!=n)low--;
			printf("%d\n",(low- pre.begin()));
			printf("%s\n",(((low- pre.begin())%2)?"Dhruv":"Ketan"));
		}
	return 0;
}
