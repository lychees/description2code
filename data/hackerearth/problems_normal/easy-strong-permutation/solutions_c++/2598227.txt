#include <bits/stdc++.h>
using namespace std;
int main()
{
    long int n;
    cin>>n;
    long int a[n],b[n];
    for(int i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    int j=0;
    for(int i=0;i<n;i++)
    {
    	if(i%2==0){
    	b[i]=a[j];
    	j++;}
    }
    for(int i=0;i<n;i++)
    {
    	if(i%2==1){
    	b[i]=a[j];
    	j++;}
    }
    long long int x=abs(b[0]-b[n-1]);
    for(int i=0;i<n-1;i++)
    {
    	x=(x+abs(b[i]-b[i+1]));
    }
    cout<<x;
    return 0;
}
