
#include <bits/stdc++.h>
using namespace std;
 
 
int main()
{
    int n;
    cin>>n;
    string a;
    bool visit[26];
    int abc[26];
    int count=0;
    memset(abc,0,sizeof(abc));
    for(int j=0;j<n;j++)
    {
    	cin>>a;
    	memset(visit,false,sizeof(visit));
    	for(int i=0;i<a.length();i++)
    	{
    		if(visit[a[i]-'a']==false)
    		{
    			abc[a[i]-'a']++;
    			visit[a[i]-'a']=true;
    		}
    	}
    }
    for(int i=0;i<26;i++)
    {
    	if(abc[i]!=0)
    	{
    		if(abc[i]==n)
    			count++;
    	}
    }
    cout<<count<<endl;
}
