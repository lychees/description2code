#include <iostream>
#include <cstring>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a,b;
    	cin>>a;
    	cin>>b;
    	for(int i=0;i<a.size();i++)
    	{
    		if(a[i]==b[i])
    	    cout<<"0";
    		else
    		cout<<"1";
    	}
    	cout<<endl;
    }
    return 0;
}
