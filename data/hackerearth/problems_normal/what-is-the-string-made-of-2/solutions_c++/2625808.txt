#include <iostream>
#include <string>
#include <map>

using namespace std;

map<char,int> Segment_Map;

void init_Map()
{
	Segment_Map['0'] = 6;
	Segment_Map['1'] = 2;
	Segment_Map['2'] = 5;
	Segment_Map['3'] = 5;
	Segment_Map['4'] = 4;
	Segment_Map['5'] = 5;
	Segment_Map['6'] = 6;
	Segment_Map['7'] = 3;
	Segment_Map['8'] = 7;
	Segment_Map['9'] = 6;
}

int calc_Segments(string value)
{
	int itr = 0;
	int segment_count = 0;
	while(value[itr] != '\0')
	{
		segment_count += Segment_Map[value[itr]];
		itr++;
	}
	
	return segment_count;
}

int main()
{
	string Number;
	
	init_Map();
	cin >> Number;
	cout << calc_Segments(Number);
    return 0;
}
