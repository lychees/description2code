#include <iostream>
#include <string.h>
using namespace std;

int main()
{
    short int n,p,ans,l;
    unsigned long b;
    char a[10001];
    cin >> n;
    while(n--)
    {
    	cin >> a >> b;
    	l = strlen(a);
    	p = a[l-1] - '0';
    	ans = 1;
    	for (;b>0;b--)
    		ans = (ans*p)%10;
    	cout << ans << endl;
    }
    return 0;
}
