#include <iostream>
using namespace std;
int a=3;

int main()
{void check(int);
    int t,x,n;
    cin>>t;
    for(n=1;n<=t;n++)
    {
    cin>>x;
    check(x);
    if(a==1)
    cout<<"Case #"<<n<<":YES"<<endl;
    else
    cout<<"Case #"<<n<<":NO"<<endl;
    a=3;
    }
    
    return 0;
}

void check(int n)
{
	if(n/10==0)
	{
		if(n==1)
		a=1;
		}
		
	else
	{
	int r,s=0;
	
	while(n>0)
	{
		r=n%10;
		s+=r*r;
		n/=10;
	}
	check(s);
}

}
