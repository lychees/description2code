#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int i;
       int h[26]={0};
    	for(i=0;i<s.size();i++)
    	{
    		h[s[i]-97]++;
    	}
    	int max=0;
    for(i=0;i<26;i++)
    {
    	if(h[i]>max)
    	max=h[i];
    }
    cout<<max<<endl;
    }
    return 0;
}
