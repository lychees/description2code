#include <iostream>
#include<cstring>
using namespace std;

int main()
{

 int i,j,t,n;
 
 cin>>t;
 while(t--)
 {
 	cin>>n;
 	int count=0;
 	int a[n+1],b[n+1];
 	
 	for(i=0;i<n;i++)
 	{
 		cin>>a[i];
 	}
 	
 	for(j=1;j<=n;j++)
 	{
 	  int d=a[j-1];
 	  b[d-1]=j;		
 	}


   for(i=0;i<n;i++)
   {
   	if(a[i]==b[i])
   	{
   	 count++;
   	}
   }
 
 if(count==n)
 {
 	cout<<"inverse"<<endl;
 }
 else
 	{
 		cout<<"not inverse"<<endl;
 	}
 	
 	
 	
 } 	
 	
 	
 	
 }
