#include<iostream>
#include<string.h>
using namespace std;
void encoder(char str[],char ans[])
{	int i=0,j=0,l=strlen(str),key1=0,key2=0,k=0;
	if(l==0)
	return ;
	else if(l==1)
	{
		ans[strlen(ans)]=str[l-1];
		return ;
	}
	else if(l%2==1)
	{
		ans[strlen(ans)]=str[l/2];
		for(i=l/2+1;i<l;i++)
		{
			str[i-1]=str[i];
		}
		str[l-1]='\0';
		encoder(str,ans);
	}
	else
	{
		key1=str[l/2-1],key2=str[l/2];
		if(key1<key2)
		{
			ans[strlen(ans)]=str[l/2-1];
			for(i=l/2;i<l;i++)
		{
			str[i-1]=str[i];
		}
		str[l-1]='\0';
		encoder(str,ans);
		}
		else
		{
			ans[strlen(ans)]=str[l/2];
			for(i=l/2+1;i<l;i++)
		{
			str[i-1]=str[i];
		}
		str[l-1]='\0';
		encoder(str,ans);
		}
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int l=0,i=0,j=0;
		char str[100],ans[100];
		memset(ans,'\0',100);
		cin>>str;
		l=strlen(str);
		encoder(str,ans);
		cout<<ans<<endl;
		
	}
	return 0;
	
}
