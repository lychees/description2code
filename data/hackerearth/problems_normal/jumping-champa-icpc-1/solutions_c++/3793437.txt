#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int t,n,q,i;
	unsigned long long int d,x;
	cin>>t;
	while(t--){
		cin>>n>>q;
		int a[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		if(n==1)
		cout<<0;
		else{
			sort(a,a+n);
			d=0;
			for(i=0;i<n-1;i++){
				x=(a[i+1]-a[i]);
				d=d+x*q;
			}
			cout<<d;
		}
		cout<<endl;
	}
	return 0;
}
