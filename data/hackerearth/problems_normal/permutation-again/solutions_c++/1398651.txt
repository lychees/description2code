#include<iostream>
#include<cmath>
using namespace std;

long long ans;
int arr[100002];

int main()
{
	int t,n,i,mid,vacant,left,right,min,max,temp,k;
	cin>>t;
	
	while(t--)
	{
		cin>>n;
		if(n<=2)
		 ans=1;
		else
		 {
   		 	ans=0;
		 	 if(n%2==0)
		 	    mid=n/2-1;
		 	  else mid=n/2;
		 	  
		 	  min=1;
		 	  max=n;
		 	  vacant=n-2;
		 	  
		 	  arr[mid]=max;
		 	  arr[mid+1]=min;
		 	  
		 	  max--;
		 	  min++;
		 	  k=1;
		 	  
		 	  left=mid-1;
		 	  right=mid+2;
		 	  
		 	  while(left<n&&right<n)
		 	  {
		 	  	 arr[right]=max;
		 	  	 arr[left]=min;
		 	  	 
		 	  	 max--;
		 	  	 min++;
		 	  	 
		 	  	 if(k%2==0)
		 	  	 {
		 	  	  temp=right;
		 	  	  right=left;
		 	  	  left=temp;
		 	  	  
		 	  	  right++;
		 	  	  left--;
		 	  	 }
		 	  	 else
		 	  	  {
		 	  	  	 right++;
		 	  	  	 left--;
		 	  	  	 
		 	  	  	 temp=right;
		 	  	  	 right=left;
		 	  	  	 left=temp;
		 	  	  }
		 	  	 
		 	  	 vacant=vacant-2;
		 	  	 k++;
		 	  	
		 	  }
		 	  
		 	 while(vacant)
		 	  {
		 	  	if(right==0)
		 	  	{
		 	  	 arr[right]=max;
		 	  	 max--;
		 	  	 right--;
		 	  	 vacant--;
		 	  }
		 	  else 
		 	   {
		 	   	  arr[left]=min;
		 	   	  min++;
		 	   	  left--;
		 	   	  vacant--;
		 	   }
		 	 }
		 	 for(i=0;i<n-1;i++)
		 	  ans+=abs(arr[i]-arr[i+1]);
		 }
		 cout<<ans<<"\n";
	}
return 0;
}
