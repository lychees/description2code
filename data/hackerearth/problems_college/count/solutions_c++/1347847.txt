#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	int t,r,c,i,j,even_sum,odd_sum;
	cin>>t;
	while(t--){
		even_sum=0;
		odd_sum=0;
		cin>>r>>c;
		for(i=1;i<=r;i++){
			if(i%2==0 && i<c){
				even_sum+=i;
			}else if(i%2==0 && i>=c){
				even_sum+=c;
			}else if(i%2!=0 && i<c){
				odd_sum+=i;
			}else if(i%2!=0 && i>=c){
				odd_sum+=c;
			}
		}
		cout<<even_sum<<" "<<odd_sum<<endl;
	}
	return 0;
}
