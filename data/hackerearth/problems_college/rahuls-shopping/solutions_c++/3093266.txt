#include<iostream>
#include<cmath>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    int n,dim1,dim2,sum=0,c=0,k;
    float d;
    cin>>n;
    int a[n];

    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    cin>>dim1>>dim2;
    for(int i=0;i<n;i++)
    {
        k=a[i];
        if(a[i]<=10)
        d=0.75;
    if(a[i]>10 && a[i]<=20)
        d=0.7;
    if(a[i]>20 && a[i]<=50)
        d=0.8;
         while(k>0)
        {
            sum+=k%10;


            k=k/10;
        }

       c=c+dim1*dim2*a[i]*sum*d;
        sum=0;

    }
cout<<c<<endl;
}
