def total_amount_pay(N,K,M):
	
	totalsum=K*pow((N*(N+1)/2),2)
	totalAmount = totalsum-M
	if totalAmount<0:
		return 0
	return(totalAmount)

testcase = int(raw_input())
for i in range(0,testcase):
	query = raw_input()
	q = map(int,query.split())
	print total_amount_pay(q[0],q[1],q[2])
