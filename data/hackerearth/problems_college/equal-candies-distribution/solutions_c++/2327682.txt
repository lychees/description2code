#include "bits/stdc++.h"
using namespace std;
const int N = 1e4 + 4;
int t;
int n;
int arr[N];
int dp[N];
int mn;
int ans;
void pre(){
	dp[0] = 0;
	dp[1] = 1;
	dp[2] = 1;
	dp[3] = 2;
	dp[4] = 2;
	dp[5] = 1;
	for(int i = 6 ; i < N ; ++i){
		dp[i] = 1 + min(dp[i - 1] , min(dp[i - 2] , dp[i - 5]));
	}
}
int main(){
	pre();
	scanf("%d" , &t);
	while(t--){
		scanf("%d" , &n);
		ans = 999999999;
		mn = 999999999;
		for(int i = 1 ; i <= n ; ++i){
			scanf("%d" , arr + i);
			mn = min(mn , arr[i]);
		}
		for(int i = mn ; i <= mn ; ++i){
			int temp = 0;
			for(int j = 1 ; j <= n ; ++j){
				temp += dp[arr[j] - i];
			}
			ans = min(ans , temp);
		}
		printf("%d\n" , ans);
	}
}
