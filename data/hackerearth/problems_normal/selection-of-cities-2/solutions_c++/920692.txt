#include <iostream>
#include <stdio.h>
#define DIVI 1000000007

typedef long long int Lint;

Lint Raise( Lint a, Lint b ) {
	
	Lint r, v2, v;

	if( b == 0 )
		return 1;
	else {

		if( b & 0x1 ) {

			r = Raise( a, ( b - 1 ) / 2 );
			v2 = ( ( r % DIVI ) * ( r % DIVI ) ) % DIVI;
			v = ( ( v2 % DIVI ) * ( a % DIVI ) ) % DIVI;
		}
		else {
			r = Raise( a, b / 2 );
			v = ( ( r % DIVI ) * ( r % DIVI ) ) % DIVI;
		}

		return v;
	}
}
int main() {
	
	Lint t, n, v, val;

	scanf("%lld", &t );

	while( t-- ) {

		scanf("%lld", &n );
		val = Raise( 2, n );
		v = ( val + 1000000006 ) % DIVI;
		printf("%lld\n", v );
	}

	return 0;
}
