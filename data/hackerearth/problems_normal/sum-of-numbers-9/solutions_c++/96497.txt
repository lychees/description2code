#include <iostream>
#include<string>
#include<vector>
using namespace std;


string process_sum(vector<long int> a, int n, long int sum)
{
	if(sum==0)
	  return "YES";
	else if(sum!=0 && n==0)
	  return "NO";
	else
	   {
	   	if(process_sum(a,n-1,sum-a[n-1]).compare("YES"))
	   	   return process_sum(a,n-1,sum);
	   	else
	   	   return process_sum(a,n-1,sum-a[n-1]);
	   }
}

int main()
{
    int t=0,n=0;
    long int sum=0;
    vector<long int> a;
    cin>>t;
    for(int j=0;j<t;j++)
    {
    	cin>>n;
    	a.resize(n);
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	cin>>sum;
    	cout<<process_sum(a,n,sum)<<endl;
    	a.clear();
    }
    return 0;
}
