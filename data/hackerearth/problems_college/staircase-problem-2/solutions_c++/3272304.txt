#include <iostream>
using namespace std;

int main()
{
    int t,n,i,ans;
	cin>>t;
	while(t--){
    cin>>n;
    for(i=2;i<=1000;i++)
    {
    	if(n<=((i*(i+1))/2)-1)
    	{
    		ans=i-1;
    		break;
    	}
    }
    if(n!=0) cout<<ans<<endl;
    else cout<<0<<endl;}
    return 0;
}
