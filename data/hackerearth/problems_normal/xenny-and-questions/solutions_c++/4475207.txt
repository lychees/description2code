#include <iostream>
#include<algorithm>
using namespace std;
int a[100005];
int main()
{
	int t,n,tim,i;
	long long sum;
    cin >> t;
    while(t--){
    	cin >> n >> tim;
    	for(i = 0;i < n;i++){
    		scanf("%d",&a[i]);
    	}
    	sort(a,a+n);
    	sum = 0;
    	for(i = n-1;i >= 0;i--){
    		if(tim == 0){
    			break;
    		}
    		sum+= a[i];
    		tim--;
    	}
    	cout << sum << endl;
    }
    return 0;
}
