#include <iostream>

using namespace std;

long long sol[70][70][70];

int main()
{
    long long n,k;
    cin>>n>>k;

    sol[1][1][1]=1; sol[1][0][0]=1;
    for (int i=1;i<=n;++i)
    {
        for (int j=0;j<=i;++j)
        {
            for (int z=0;z<=j;++z)
            {
                sol[i+1][j][0]+=sol[i][j][z];
                if (z+1>j)
                {
                    sol[i+1][j+1][z+1]+=sol[i][j][z];
                }else
                {
                    sol[i+1][j][z+1]+=sol[i][j][z];
                }
            }
        }
    }

    long long v1=0,v2=1LL<<n;

    for (int v=k;v<=n;++v)
    {
        for (int i=0;i<=v;++i) v1+=sol[n][v][i];
    }

    long long a=v1,b=v2;
    while (b!=0)
    {
        long long t=a;
        a=b;
        b=t%b;
    }

    v1=v1/a;
    v2=v2/a;

    cout<<v1<<"/"<<v2;

    return 0;
}
