#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	ll n,q;
    	cin>>n>>q;
    	int a[n];
    	for(int i=0;i<n;i++)
    		cin>>a[i];
    	sort(a,a+n);
    	ll sum=0;
    	for(int i=0;i<n-1;i++)
    	{
    		sum+=a[i+1]-a[i];
    	}
    	ll x=q*sum;
    	cout<<x<<endl;
    }
    return 0;
}
