t=int(input())
while t>0:
    ttl=int(input())
    if ttl<6:
        print '-1'
    else:
        l=[]
        for i in range(int(ttl/8)+1):
            for j in range(int(ttl/6)+1):
                if 8*i+6*j==ttl:
                    l.append(i+j)
        if len(l)==0:
            print('-1')
        else:
            print min(l)
    t=t-1
