#include<iostream>
#include<set>
#include<iterator>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		set<long int> s;
		set<long int>::iterator it;
		long int n,x,count = 0;
		cin>>n;
		for(int k = 0; k < n; k++)
		{
			cin>>x;
			s.insert(x);
			count++;
		}
		long int arr[100000];
		long int j = 0;
		for(it = s.begin(); it != s.end(); it++)
		{
			arr[j] = *it;
			++j;
		}
		long int k;
		if(j-1 == 0)
			cout<<arr[j-1];
		else
		{
		for(k = 0, j = j-1; j >= k; k++, j--)
		{
			if(j == k)
			{
				cout<<arr[j];
				break;
			}
			cout<<arr[j]<<" "<<arr[k]<<" ";
		}
		}
		cout<<endl;
		
	}
	
	return 0;
}
