#include <bits/stdc++.h>
using namespace std;
#define ll long long
vector<int>v;
ll po(ll e)
{
	ll ans=1;
	while(e>0)
	{
		ans*=2;
		e--;
	}
	return ans;
}
int main()
{
    ll n,t;
    cin>>t;
    while(t--)
    {
    cin>>n;
      ll s=n;
       int k=0;
      while(s>0)
      {
      	if(s&1)
      	v.push_back(k);
      	
      	k++;
      	s=s>>1;
      }
      //for(int i=0;i<v.size();i++)
      //cout<<v[i]<<" ";
      //cout<<endl;
      ll lim=v[v.size()-1];
      //cout<<lim<<endl;
      cout<<po(lim+1)-1<<endl;
      v.clear();
}
}
