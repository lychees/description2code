#include <stdio.h>
#include <iostream>
#include <algorithm> 
#include <math.h>
#include <vector>
using namespace std;
int main()
{	int testcase;
	int mod=1000000007;
    cin>>testcase;
    for(int i=0;i<testcase;i++)
    {
    	long long int N;
    	long long int sum=1;
    	cin>>N;
    	long long int arr[N];
    	for(long long int j=0;j<N;j++)
    	cin>>arr[j];
    	sort(arr , arr+N);
    	for(long long int k=0;k<N;k++)
    	{
    	  	sum = (sum%mod)*((arr[k]-k)%mod);
    	}
    	if(sum>0)
    	cout<<sum%mod<<endl;
    	else
    	cout<<0<<endl;
    }
    
    
    return 0;
}
