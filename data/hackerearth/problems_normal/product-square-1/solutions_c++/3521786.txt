#include <iostream>
#include<cstdio>
#include<cmath>
using namespace std;
int cnt[5001];
int main()
{
    double p = 2.0;
    int n;
    cin >> n;
    int s;
    for(int i = 0;i < n;++i){
    	cin >> s;
    	cnt[s]++;
    }
    if(n == 1){
    	printf("0.0");
    	return 0;
    }
    p /= n;
    p /= (n - 1);
    long ans = 0;
    for(int i = 1;i <= 5000;++i){
    	long long d = 1ll*cnt[i]*(cnt[i] - 1) / 2;
    	ans += d;
    }
    for(int j = 1;j <= 5000;++j){
		for(int k = j + 1;k <= 5000;++k){
			int root = sqrt(k*j);
			if(root*root == k*j && (cnt[k] > 0 && cnt[j] > 0)){
				ans += (1L*cnt[k]*cnt[j]);
			}
		}
	}
    p *= ans;
    printf("%.6f",p);
    return 0;
}
