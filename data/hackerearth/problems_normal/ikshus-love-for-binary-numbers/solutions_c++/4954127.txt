#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int n, k;
unsigned long long arr[61][61];

int main()
{
	scanf("%d %d", &n, &k);

	int i, j;
	for (i = 1; i <= 60; i++)
		arr[i][i] = 1;

	for (i = 2; i <= 60; i++)
		arr[i][1] = 2 * arr[i - 1][1] + 1;

	for (i = 1; i <= 60; i++)
	for (j = 2; j <= i - 1; j++)
		arr[i][j] = 2 * arr[i - 1][j] + (1LL << i - j - 1) - arr[i - j - 1][j];

	unsigned long long numerator = arr[n][k];
	unsigned long long denominator = 1LL << n;

	while (numerator % 2 == 0)
	{
		numerator = numerator / 2;
		denominator = denominator / 2;
	}
	printf("%llu/%llu", numerator, denominator);
	return 0;
}
