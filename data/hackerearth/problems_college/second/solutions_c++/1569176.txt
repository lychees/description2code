#include <bits/stdc++.h>
#define si(n) scanf("%d",&n);
#define pi(n) printf("%d\n",n);
#define pl(n) printf("%lld\n",n);
#define sl(n) scanf("%lld",&n);
#define pd(n) printf("%lf\n",n);
#define ss(s) scanf("%s",s);
#define ps(s) printf("%s\n",s);
#define ll long long
#define mod 1000000007
#define pb push_back
#define maxn 100005
int arr[maxn];
using namespace std;
int main()
{
    int n,m;
    si(n);si(m);
    for(int i=0;i<n;i++)
        si(arr[i]);
    sort(arr,arr+n);
    int val=m;
    int ans=0;
    for(int i=0;i<n;i++)
    {
        if(arr[i]==val)
        {
            ans=val;
            val+=m;
        }
    }
    pi(ans);
}
