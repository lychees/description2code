#include <iostream>
#include <math.h>
using namespace std;

int gcd (int a, int b) {
    if(a == b)
        return a;
    if(a>b)
        return gcd(a-b, b);
    else
        return gcd(a, b-a);
}

int main() {
    int T;
    cin >> T;
    while(T--) {
        int N,  c=0, a, b, cnt=0;
        long long int n;
        cin >> N;
        long long int m = 2;
        while(m<sqrt(N)) {
            for(n=1; n<m; n++) {
                //cout << "m,n="<<m<<","<<n<<endl;
                a = m*m - n*n;
                b = 2*m*n;
                c = m*m + n*n;
                if(c>N)
                    break;

               // int temp =1;
               // while(c*temp <= N) {
                 //   cnt++;
                  //  temp++;
               // }
               if((gcd(m,n)==1) && ((m-n)%2!=0)) {
                    //cout << "m=" << m << "n=" <<n <<endl;
                //cout << "triplets =" << a <<"," << b <<"," <<c << endl;
                    cnt += N/c;
               }
            }
            m++;
            //cout << "value of c, a,b ,m,n is = " << c<<","<<a<<","<<b<<","<<m<<","<< n << endl;
        }
        cout << cnt << endl;
    }
}
