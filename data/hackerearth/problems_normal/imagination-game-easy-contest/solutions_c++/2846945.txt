template < typename T > T abs(T x)
{
    return x > 0 ? x : -x;
}

template < typename T > T sqr(T x)
{
    return x * x;
}

int a[15];
int x[15];

int main()
{
	int t;
	
	cin >> t;
	
	while (t--) {
		int n;
		
		cin >> n;
		
		for (int i = 0; i < n; i++)
			cin >> a[i];
			
		for (int i = 0; i < n; i++)
			cin >> x[i];
			
		int res = 0;
			
		for (int i = 0; i < n; i++) {
			res ^= (x[i] - 1);
			res ^= (a[i] - x[i]);
		}	
		
		if (res == 0) {
			puts("Second");
		}
		else {
			puts("First");
		}
	}	
	
	return 0;
}
