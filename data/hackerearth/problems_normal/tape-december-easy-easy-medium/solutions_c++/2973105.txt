#include <iostream>
using namespace std;

int main()
{
   int n,k,d,cnt=0,sum=0,x[100005],sub[100005];
   cin>>n>>k>>d;
   for(int i=0;i<n;i++)
   cin>>x[i];
   for(int i=0;i<k;i++) sum+=x[i];
   sub[0]=sum;
   for(int i=k;i<n;i++)
   {
   	sum=sum-x[i-k]+x[i];
    sub[i-k+1]=sum;
   }
   for(int i=0;i<=n-k;)
   {
   	if(sub[i]==0) { cout<<-1<<endl; return 0; }
   	if(sub[i]<d) { 
   		i=i+k-1;
   		while(i<n&&x[i]==0) i--;
   		i=i+k;
   		cnt++; }
    else i++;
   }
   cout<<cnt<<endl;
}
