#include<bits/stdc++.h>
using namespace std;
long long arr[110];
long long ans[110][100010];
long long power(long long a,long long b, long long c){
if(b==0)
return 1;
if(b==1)
return a;
long long ans = power(a,b/2,c);
if(b%2==0)
return (ans*ans)%c;
else
return (((ans*ans)%c)*a)%c;
}
long long mod = 1000000007;
int main()
{
    long long n,k,sum=0;
    cin>>n>>k;
    for(int i = 1; i <=n;i++)
    {
        cin>>arr[i];
        sum+=arr[i];
    }
    if(sum < 2*k)
    {
        printf("0\n");
    }
    else
    {

        ans[0][0] = 1;
        for(int i = 1;i<=n;i++)
        {
            for(int j = 0;j<=k;j++)
            {
                    if(ans[i-1][j]> 0 )
                    {
                        if(j+arr[i] <=k)
                        ans[i][j+arr[i]] = (ans[i][j+arr[i]] +  ans[i-1][j])%mod;
                        ans[i][j] = ( ans[i][j] + ans[i-1][j])%mod;
                    }
            }
        }
        long long temp = 0;
        for(int i = 0;i<k;i++)
        temp = (temp + ans[n][i])%mod;
        //cout<<temp<<endl;
        temp = power(2,n,mod) - 2*temp;
        while(temp < 0)
        temp +=mod;
        temp%=mod;
        cout<<temp<<endl;
    }


}
