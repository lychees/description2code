#include <cstdio>
#include <iostream>
#include <cstring>
#include <cmath>

using namespace std;

int main()
{
    int n, len;
    char arr[105], ch;
    cin>>n;
    while (n--)
    {
        cin>>arr;
        len=strlen(arr);
        ch='a';
        for (int i=0; i<len; i++)
        {
            if (arr[i]>=ch)
            {
                if (arr[i]-ch<=13)
                {
                    cout<<arr[i]-ch<<" ";
                }
                else
                {
                    cout<<((arr[i]-ch)-26)<<" ";
                }
            }
            else
            {
                if (ch-arr[i]<13)
                {
                    cout<<-(ch-arr[i])<<" ";
                }
                else
                {
                    cout<<(26-(ch-arr[i]))<<" ";
                }
            }
            ch=arr[i];
        }
        cout<<endl;
    }
    return 0;
}
