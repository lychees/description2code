if __name__ == "__main__":

    T = int(raw_input())

    for t in xrange(T):

        W = raw_input().strip()
        R = sorted(raw_input().strip())

        X = []

        r = 0

        for w in W:
            if r < len(R):
                if R[r] < w:
                    X.append(R[r])
                    r += 1
                else:
                    X.append(w)
            else:
                X.append(w)

        print "".join(X)
