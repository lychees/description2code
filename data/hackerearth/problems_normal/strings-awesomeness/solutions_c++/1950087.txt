# include<bits/stdc++.h>
using namespace std;

void preprocess(char S[],int a[],int len)
{

	int counter[26]={0};
	if(len>11)
	{
		
		for(int i=0;i<11;i++)
		{
			counter[S[i]-'a']++;
			//cout<<S[i]<<endl;
		}
		if(counter[0]==2 && counter[2]==1 && counter[4]==2 && counter[10]==1 && counter[7]		   	==2 && counter[17]==2 && counter[19]==1)
		{
			a[10]=1;
		//	cout<<"check";
		}
	
		for(int i=11;i<len;i++)
		{
			counter[S[i-11]-'a']--;
			counter[S[i]-'a']++;
			if(counter[0]==2 && counter[2]==1 && counter[4]==2 && counter[10]==1 && counter[7]==2 && counter[17]==2 && counter[19]==1)
			{
				a[i]=a[i-1]+1;
			}
			else
				a[i]=a[i-1];
		}
	}
	
}


int main()
{
	char S[1000006];
	int q,l,r,len;
	int a[1000006]={0};
	memset(a,0,sizeof(a));
	scanf("%s",S);
	len=strlen(S);
	
	for(int i=0;i<len;i++)
		S[i]=tolower(S[i]);
	
	preprocess(S,a,len);
	scanf("%d",&q);
	while(q--)
	{
		scanf("%d %d",&l,&r);
		l--;r--;
		if(r-l+1>=11)
		{
		printf("%d\n",a[r]-a[l+9]);
		}
		else
		printf("0\n");
		
	}
	
	
}
