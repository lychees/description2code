#include<iostream>
using namespace std;

int max(int a,int b,int c)
{
       if(a>=b && a>=c)
             return(a);
       if(b>=a && b>=c)
             return(b);
       return(c);            
}

int dp[2048][2048];

int pow[3];

int calc(int f,int g, int last)
{
      if(f<0 || g<0)
            return(-1*pow[last]);
     if(dp[f][g]==-1)
     {
          dp[f][g]=max(pow[0]+calc(f-4,g,0),pow[1]+calc(f-5,g-2,1),pow[2]+calc(f-2,g-4,2));
     }
     return(dp[f][g]);       
}
int main()
{
    int t,init,i,j;
    cin>>t;
    int f,g;
    while(t--)
    {
         cin>>f>>g;
         f=f/25;
         g=g/25;
         for(i=0;i<=f;i++)
         {
             for(j=0;j<=g;j++)
                dp[i][j]=-1;
         }
         cin>>pow[0]>>pow[1]>>pow[2];
         int x=calc(f,g,0);
         cin>>init;
         if(init>=x)
              cout<<"-1\n";
         else
         {
              x=x-init;
              cout<<x<<"\n";
         }           
    }
    return(0);
}
