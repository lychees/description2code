#include <bits/stdc++.h>
using namespace std;
#define scan(x) scanf("%d",&x)
#define print(x) printf("%d",x)
#define MOD 1000000007
#define scanl(x) scanf("%lld",&x)
#define printl(x) printf("%lld",x)
long long int maxi,maxcount;

void dfs(int i,map<int,vector<int> > &m,vector<int> &know,vector<int> &vis)
{
	vis[i]=1;
	if(know[i-1]>maxi)
	{
		maxcount=1;
		maxi=know[i-1];
	}
	else if(know[i-1]==maxi)
	{
		maxcount++;
	}
	for(int j=0;j<m[i].size();j++)
	{
		if(!vis[m[i][j]])
		{
			dfs(m[i][j],m,know,vis);
		}
	}
}
 
 
int main()
{
	int n,p;
	scan(n);
	scan(p);
	vector<int> know(n);
	for(int i=0;i<n;i++)
	{
		scan(know[i]);
	}
	map<int,vector<int> > m;
	int t1,t2;
	for(int i=0;i<p;i++)
	{
		scan(t1);
		scan(t2);
		m[t1].push_back(t2);
		m[t2].push_back(t1);
	}
	long long int ans=1;
	vector<int> vis(n+1);
	for(int i=1;i<=n;i++)
	{
		if(!vis[i])
		{
			maxcount=0;
			maxi=-1;
			dfs(i,m,know,vis);
			ans=(ans*maxcount)%MOD;
		}
	}
	cout<<ans;
    return 0;
}
