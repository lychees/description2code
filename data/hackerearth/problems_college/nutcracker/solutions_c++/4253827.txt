#include<bits/stdc++.h>
using namespace std ;
int a[10002] ;
int main () {
    int i , t ;
    a[1] = a[3] = a[4] = 1 ;
    a[2] = 0 ;
    for ( i = 5 ; i <= 10001 ; i++ ) {
        if ( a[i-1] == 0 || a[i-4] == 0 )
            a[i] = 1 ;
        else
            a[i] = 0 ;
    }
    scanf ( "%d" , &t ) ;
    while ( t-- ) {
        int n ;
        scanf ( "%d" , &n ) ;
        if ( a[n] == 1 )
            cout << "Yes\n" ;
        else
            cout << "No\n" ;
    }
    return 0 ;
}
