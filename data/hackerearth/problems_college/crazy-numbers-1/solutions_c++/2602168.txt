#include <iostream>
#include <cstring>
#include <vector>
using namespace std;

int main()
{
	int mod = 1000000007;
	int t,n;
	cin>>t;
	while(t--) {
		cin>>n;
		if(n==0) {
			cout<<"0\n";
			continue;
		}
		if(n==1) {
			cout<<"10\n";
			continue;
		}
		
		vector<int> v(10,1);
		
		for(int i=2;i<=n;i++) {
			vector<int> temp(10);			
			for(int i=0;i<v.size();i++) {
				if(i==0) {
					temp[i] = v[i+1];
				}
				else if(i==9) {
					temp[i] = v[i-1];
				}
				else
					temp[i] = (v[i+1] + v[i-1])%mod;
			}
			v = temp;
		}
		
		int sum = 0;
		for(int i =1;i<v.size();i++)
		sum = (sum+v[i])%mod;
		
		cout<<sum<<"\n";
	}
    
    return 0;
}
