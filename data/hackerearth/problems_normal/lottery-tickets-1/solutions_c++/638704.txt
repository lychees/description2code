bool comparator(const pair<int,int> &l,const pair<int,int> &r){
    return l.first<r.first;
}
int main(){
    int N;
    int T;
    cin>>T;
    while(T--){
    cin>>N;
    unsigned int st[51];
    vector< pair<int,int> >et;
    int wp[51],p[51];
    int t;   
    float result[51];
    et.push_back(make_pair(0,0));
    for(int i=1;i<=N;i++){
        cin>>st[i]>>t>>wp[i];
        et.push_back(make_pair(t,i));
    }
    sort(et.begin()+1,et.end(),comparator);
    p[0]=0;
    //for(int i=0;i<N;i++)
      //  cout<<et[i].first<<" "<<et[i].second<<endl;
    for(int i=1;i<=N;i++){
        int j=i-1;
        for(;j>=0;j--)
           if(st[et[i].second]>=et[j].first){
               p[i]=j;
               break;
           }
        if(j<0)
            p[i]=0;
                
    }
   // for(int i=0;i<=N;i++)
     //   cout<<p[i]<<endl;
    float temp=0.00;
    result[0]=0.00;
    for(int i=1;i<=N;i++){
        temp=(float)(wp[et[i].second])/(float)100;
        result[i]=temp+result[p[i]]>result[i-1]?temp+result[p[i]]:result[i-1];
    }
    cout<<fixed;
    cout<<setprecision(2)<<result[N]<<endl;        
    }   
    return 0;
    }
