#include<iostream>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int count=0;
		int solution[n][2];
		for(int i=0;i<n;i++)
		{
			int x,y;
			cin>>x>>y;
			solution[x-1][0]=y;
		}
		int max1=-100;
		
		for(int i=0;i<n;i++)
		{
		int max=0;
		for(int j=0;j<i;j++)
		{
			if(solution[j][0] < solution[i][0])  
			{
				if(solution[j][1] > max)
				  max = solution[j][1];
			}
			
		}
		solution[i][1] = max +1;
		if(solution[i][1]> max1) 
		max1 = solution[i][1];
		
	//	printf("%d\n",c);
	}
	cout<<max1<<"\n";
	}
	
}
