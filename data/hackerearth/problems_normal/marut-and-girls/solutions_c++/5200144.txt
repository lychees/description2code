#include <iostream>
#include <string>
#include <cstring>
#include <sstream>
#include <map>
bool stateQual[10005];
using namespace std;

int main()
{
    int m;
    cin >> m;
    memset(stateQual, false, sizeof stateQual);
    // int qual[m];
    for(int i = 0; i < m; i++)
    {
    	int val;
    	cin >> val;
    	stateQual[val] = true;
    	// cout << qual[i];
    }
    // cout << endl;
    int t;
    cin >> t;
    cin.ignore();
    int count, girlCount = 0;
    for(int i = 0; i < t; i++)
    {
    	count = 0;
    	
    	int val; char c;
    	
    	// string line;
    	// getline(cin, line);
    	// istringstream iss(line);
    	while(scanf("%d%c", &val, &c) > 0)
    	// while(iss >> val)
    	{
    		// cout << val;
    		
    		if(stateQual[val] == true)
    			count++;
    		// cout << count << " ";
    		if(c == '\n')
    			break;
    	}
    	if(count == m)
    		girlCount++;
    	// cout << endl;
    }
    
    cout << girlCount << endl;
    
    
}
