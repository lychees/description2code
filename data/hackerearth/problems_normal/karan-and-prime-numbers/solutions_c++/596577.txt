#include <iostream>
using namespace std;
int isPrime(int n)
{
    int i;
    if(n == 2) return 1;
    if(n%2 == 0 || n==1) return 0;
    for(i = 3; i*i<=n; i+=2) {
        if(n%i == 0) return 0;
    }
return 1;
}
int main()
{
    int a,b, test,sum=0;
    cin>>test;
    for (int i=0;i<test;i++)
    {
    	cin>>a>>b;
   	for (int j=a;j<=b;j++)
    	{
    		int y=isPrime(j);
    		if(y)
                sum+=j;
    	}
    	cout<<sum<<endl;
    	sum=0;
    }
}
