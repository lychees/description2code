template <class T>T Min2(T a,T b){return a<b?a:b;}
template <class T>T Max3(T a,T b,T c){return Max2(Max2(a,b),c);}

int ans=0;
void func(int *a,int l,int d,int i){
	if(i==l-1)
		ans++;
	else if(i<l-1){
		for(int j=i+1;j<l;j++){
			if(a[j]-a[i]>=a[i]-a[d]){
				func(a,l,i,j);
		}
	}
}
}
