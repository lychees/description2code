#include <iostream>
#include <cmath>
#include <ctime>
#include <vector>
#include <memory.h>

#define Max 10000000

using namespace std;

bool isPrime[Max];
bool u[Max];
vector <int> v;

void getPrime() {
	isPrime[1] = 1;
	u[0] = 1;
	u[1] = 1;
	u[2] = 1;
	isPrime[2] = 1;
	for(int i=4; i<Max; i+=2) u[i] = 1;
	v.push_back(2);
	for(int i=3; i<Max; i+=2) {
		if(u[i] == 0) {
			u[i] = 1;
			isPrime[i] = 1;
			v.push_back(i);
			for(int j=i+i; j<Max; j+=i) u[j] = 1;
		}
	}
}

bool check(long long x) {
	if(x < Max) return isPrime[x];
	for(int i=0; i<v.size(); i++) {
		if(v[i] > sqrt(x)) break;
		else if(x % v[i] == 0) return 0;
	}
	return 1;
}

int main() {
	memset(u, 0, sizeof u);
	memset(isPrime, 0, sizeof isPrime);
	getPrime();
	int tt;
	cin >> tt;
	while(tt--) {
		long long x;
		cin >> x;
		if(x < Max && isPrime[x] == 1) cout << "Yes" << endl;
		else {
			if(check(x) == 1) cout << "Yes" << endl;
			else {
				for(int i=0; i<v.size(); i++) {
					if(x % v[i] == 0) {
						x = x / v[i];
						if(x % v[i] == 0) {
							cout << "No" << endl;
							break;
						}
						else if(check(x) == 1) {
							cout << "Yes" << endl;
							break;
						}
					}
				}
			}
		}
	}
	return 0;
}
