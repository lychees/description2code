#include <iostream>
using namespace std;

int cnt[10010];

int main()
{
    int n;
    cin>>n;
    
    for(int i=0; i<10010; i++)
    	cnt[i] = 0;
    
    
    for(int i=0; i<n; i++) {
    	int num;
    	cin>>num;
    	cnt[num]++;
    	//cout<<num<<" "<<cnt[num]<<endl;
    }
    
    int m;
    cin>>m;
    
    for(int i=0 ;i<m; i++) {
    	int num;
    	cin>>num;
    	cnt[num]--;
    	//cout<<num<<" "<<cnt[num]<<endl;
    }
    
    for(int i=0; i<10010; i++) if(cnt[i]<0) cout<<i<<" ";
    	
    return 0;
}
