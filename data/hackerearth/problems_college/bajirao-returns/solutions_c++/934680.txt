#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int main() {
	int n;
	scanf("%d",&n);
	int a[n+1][n+1];
	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= n; j++) {
			scanf("%d",&a[i][j]);
		}
	}
	for (int i = n-1; i  > 0; i--) {
		for (int j = 1; j <= n; j++) {
			int temp, ans = 0;
			int k = j;
			while (k) {
				for (int r = k; r <= n; r = r*2) {
					temp = a[i][j]+a[i+1][r];
					if (temp > ans)
						ans = temp;
				}
				k = k/2;
			}
			a[i][j] = ans;
		}
	}
	int ans = 0;
	for (int i = 1; i <= n; i++) {
		if (a[1][i] > ans)
			ans = a[1][i];
	}
	printf("%d\n",ans);
	return 0;
}
