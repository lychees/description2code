#include <iostream>
#include <string>

using namespace std;

void ruby (int one, int two, int three, int four) {
	int result;
	
	if (one >= two)
		result = two;
	else
		result = one;
	
	if (three >= four)
		one = four;
	else
		one = three;
	
	if (result >= one)
		cout << one << endl;
	else
		cout << result << endl;
}

int main () {
	int size;
	cin >> size;
	
	while (size--) {
		string gemz;
		cin >> gemz;
		
		int count1 = 0, count2 = 0, count3 = 0, count4 = 0;
	
		for (int i = 0; i < gemz.length(); i++) {
			if (gemz[i] == 'r')
				count1++;
			else if (gemz[i] == 'u')
				count2++;
			else if (gemz[i] == 'b')
				count3++;
			else if (gemz[i] == 'y')
				count4++;
		}
		ruby (count1, count2, count3, count4);	
	}

	return 0;
}
