using namespace std;
typedef long long int int64;
#define MOD 1000000007

struct intervals
{
	int time;
	int startend;
};

bool func(intervals i1, intervals i2)
{
	if (i1.time < i2.time) return true;
	else if(i1.time==i2.time)
	{
		return i1.startend>i2.startend;
	}
	return false;
}

int main()
{

	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);

	//code here
	int t,n;
	cin >> t;
	intervals arr[200002];
	int temp;
	int count;int max;
	while (t--)
	{
		cin >> n;
		count=0;
		max=-1;
		int i;
		for (i = 0; i < n; i++)
		{
			cin >> arr[i].time;
			arr[i].startend = 0;
		}
	
		for (int j = 0; j < n; j++)
		{
			cin >> temp;
			arr[i].time = arr[j].time + temp;
			arr[i].startend = 1;
				i++;
		}

		sort(arr, arr + i,func);

		for (int j = 0; j < i; j++)
		{
			if (arr[j].startend ==0) {count++;if(max<count) max=count;}
			else count--;
		}

		cout << max << endl;
	}



	//debug
