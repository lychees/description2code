#include<iostream>
#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
ll gcd(ll a,ll b)
{
	if(b==0)return a;
	else return gcd(b,a%b);
}
int main()
{
	ll t,a,b,c;
	cin>>t;
	while(t--)
	{
		cin>>a>>b>>c;
		ll g=gcd(gcd(a,b),c);
		a=(a/g)%mod;
		b=(b/g)%mod;
		c=(c/g)%mod;
		ll mul=a;
		mul=(mul*b)%mod;
		mul=(mul*c)%mod;
		cout<<g<<" "<<mul<<endl;
	}
}
