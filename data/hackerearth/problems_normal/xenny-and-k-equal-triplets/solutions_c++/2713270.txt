#include <iostream>
#include <cstdio>
using namespace std;

long long gcd(long long a, long long b)
{
	 return b == 0 ? a : gcd(b, a % b);
}

int main()
{
	long long gcd(long long int, long long int);
    long long int t, n, k, totk=0;
    scanf("%lld", &t);
    for(int i=0; i<t; i++)
    {
    	totk = 0;
    	scanf("%lld%lld", &n, &k);
    	int arr[n];
    	for(int i=0; i<n; i++)
    	{
    		scanf("%lld", &arr[i]);
    		if(k == arr[i])
    			totk++;
    	}
    	
    	long long num = (totk*(totk-1)*(totk-2));
    	long long den = (n*(n-1)*(n-2));
    	long long die = gcd(num, den);
    	num = num/die;
    	den = den/die;
    	printf("%lld/%lld\n", num, den);
    }
    return 0;
}
