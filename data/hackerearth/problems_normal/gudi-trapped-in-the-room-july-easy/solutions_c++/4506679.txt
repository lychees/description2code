#include <iostream>
#include <cstring>
#include<vector>
#include <algorithm>  
#include <stack>
using namespace std;

int main()
{
	int t,i,j,k,h,a,min;
	string s,smin;
	int vis[999999];
	
	cin >>t;
	for(i=0;i<t;i++)
	{
		min=999999;
		cin>>s;
		cin>>a>>h;
		stack <string>stc;
		stc.push(s);
		memset(vis,0,sizeof(vis));
		while(!stc.empty())
		{
			string str=stc.top();
			string str0=str;
			stc.pop();
			for(j=0;j<str.size();j++)
			{
				if((j+1)%2==0)
				str[j]=((str[j]-'0'+a)%10)+'0';
			}
			int p=stoi(str);
			if(vis[p]==0)
			{
				vis[p]=1;
				stc.push(str);
				if(p<min)
				{
					min=p;
					smin=str;
				}
			}
			for(k=0;k<h;k++)
			{
				char temp=str0[str0.size()-1];
				for(j=str0.size()-1;j>0;j--)
				{
					str0[j]=str0[j-1];				
				}
				str0[0]=temp;
			}
			
				
				p=stoi(str0);
			if(vis[p]==0)
			{
				vis[p]=1;
				stc.push(str0);
				if(p<min)
				{
					min=p;
					smin=str0;
				}
			}
		}
		
	cout<<smin<<endl;	
	}
    return 0;
}
