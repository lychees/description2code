#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
using namespace std;

int main() {
    int t;
    cin>>t;
    int n,a,b,c,d,mv;
    int kr,pr;
    while(t--){
        cin>>n>>a>>b>>c>>d>>mv;
        int pr,kr;
        pr = n-a;
        kr = max(n-c,abs(b-d));
        if(mv==0){
            if(kr-pr>=1){
                cout << "White Wins"<<endl;
            }
            else {
                cout<<"Draw"<<endl;
            }
        } else if (mv==1){
            if(kr-pr>=2){
                cout << "White Wins"<<endl;
            }
            else {
                cout<<"Draw"<<endl;
            }
        }
    }
    return 0;
}
