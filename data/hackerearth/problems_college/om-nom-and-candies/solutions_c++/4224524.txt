#include <stdio.h>
#include <iostream>
#include <stdlib.h>
#include <algorithm>

using namespace std;

long long int max(long long int a, long long int b)
{
	if (a > b)
	{
		return a;
	}
	else
		return b;
}

int main()
{
	int n, *a ;
	long long int *final;
	cin >> n ;
	a = (int*) malloc((n+1)*sizeof(int));
	final = (long long int*) malloc((n+1)*sizeof(long long int));
	final[0] = 0;
	a[0] = 0;
	for(int i=1; i<=n; i++)
		{ cin >> a[i];
		  final[i] = 0;
		}
	final[1] = a[1];
	for(int j=2; j<=n; j++)
	{
		final[j] = max(final[j], final[1] + a[j]);
		for(int i=j*2; i<=n; i+=j)
		{
			final[i] = max(final[i], final[j] + a[i]);
		}
	}
	for(int i=1; i<=n; i++)
	{
		cout << final[i] << " " ;
	}
	cout << endl;

}
