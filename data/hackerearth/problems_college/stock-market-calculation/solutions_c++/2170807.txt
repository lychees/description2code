#include<iostream>
#include<algorithm>
#include<cmath>
#include<limits.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long n,a[100005],i,x=0,y=0,z=0,ans=0,maxm=INT_MIN,m=INT_MIN,pos=0;
        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>a[i];
        }
        x=0;
        while(1){
        for(i=x;i<n;i++)
        {
            if(a[i]>maxm){maxm=a[i];pos=i;}
        }
        sort(a+x,a+pos+1);
        for(i=x;i<pos;i++)
        {
            if((maxm*(z+1))-(y+a[i])>m){m=(maxm*(z+1))-(y+a[i]);z++;y+=a[i];}
            else break;
        }
        ans+=((maxm*z)-y);
        if(pos==n-1)break;
        x=pos+1;maxm=INT_MIN;m=INT_MIN;y=0;z=0;
    }
    cout<<ans<<endl;
}
return 0;
}
