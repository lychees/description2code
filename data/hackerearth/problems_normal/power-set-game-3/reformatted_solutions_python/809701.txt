MOD = 1000000007


def power(a, b):
    if b == 0:
        return 1
    if b == 1:
        return a
    p = power(a, b/2) % MOD
    if b & 1:
        return (((p*p) % MOD)*a) % MOD
    else:
        return (p*p) % MOD


t = int(raw_input())
while t:
    t -= 1
    n = int(raw_input())
    print(power(2, 2*n)-power(2, n)) % MOD
