#include <iostream>
#include<list>
using namespace std;
list<int> i;
int n,t,k,a,flag;
int main()
{
    scanf("%d",&n);
    while(n--)
    {scanf("%d%d",&t,&k);
    i.erase(i.begin(),i.end());
    for(int l=0;l<t;l++)
   { scanf("%d",&a);
    i.push_back(a);
   }
 
   list<int>::iterator it,ij;
   while(k>0)
   {flag=0;
     
   	for(it=i.begin();it!=i.end();)
   			{ij=it;
   			 it++;
   			 
   			 
   			 if(*ij<*it)
   			 {i.erase(ij);
   			  flag=1;
              
                 k--;
                 it--;
                 it--;
                 if(k<=0)
                 break;
   			  
   			 }
   			}
   			if(!flag)
   		{   k--;
   			i.erase(i.begin());
   		}
   	
   }
   for(it=i.begin();it!=i.end();it++)
   printf("%d ",*it);
   printf("\n");
    }
   
    return 0;
}
