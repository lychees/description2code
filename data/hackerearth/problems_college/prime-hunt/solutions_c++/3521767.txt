#include<bits/stdc++.h>
using namespace std;
#define MAX 10000005
int prime[MAX];
bool arr[MAX];
int main()
{
        int i,j,root;
        for(i=0;i<MAX;i+=2)
                arr[i]=0;
        for(i=3;i<MAX;i+=2)
                arr[i]=1;
        root=sqrt(MAX);
        for(i=3;i<=root;i+=2)
        {
                if(arr[i])
                {
                        for(j=i*i;j<MAX;j+=(i<<1))
                                arr[j]=0;
                }
        }
        prime[0]=0;
        prime[1]=0;
		arr[2] =1;
        j=2;
        for(i=2;i<MAX;i++)
        {
        	//cout<<arr[i]<<" ";
                if(arr[i]){
                        prime[i]= prime[i-1]+1;     
                        
                }       
                else
                	prime[i] = prime[i-1];
	
	//		printf("%d *",arr[i]);
        }
        int t,n;
        scanf("%d",&t);
        while(t--)
        {
        	int n,p;
                scanf("%d %d",&n,&p);
                printf("%d\n",prime[p]-prime[n-1]);
        }
        return 0;
}
