#include <iostream>
#define ll long long int
using namespace std;
const ll mod = 1e9  + 7;
ll fact(ll a, ll b){
	ll res = 1LL;
	for(ll i = a; i <= b; i++){
		res *= i;
		res %= mod;
	}
	return res;
}
int main()
{
	int t;
    ll n,m;
    cin >> t;
    while(t--){
    	cin >> m >> n;
    	if(n > m + 1){
    		cout << "-1\n";
    		continue;
    	}
    	ll a = fact(1, m), b = fact(m + 2 - n, m + 1);
    	cout << (a * b) % mod << endl;
    	
    }
    return 0;
}
