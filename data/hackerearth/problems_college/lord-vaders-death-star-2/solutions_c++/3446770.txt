#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,x;
    cin >> n;
    map<int,int> hmap;
    for(int i=1;i<=n;i++)
    {
    	cin >> x;
    	if(hmap.find(x) == hmap.end())
    	{
    		hmap[x]=1;
    	}
    	else
    		hmap[x]++;
    }
    int maxi = -1;
    for (std::map<int,int>::iterator it=hmap.begin(); it!=hmap.end(); ++it)
    {
    	if(it->second == 1)
    	{
    		maxi = max(maxi,it->first);
    	}
    }
    cout << maxi <<endl;

    return 0;
}
