using namespace std;
int n;
map<string, vector<int> > mp;
int main(){
	scanf("%d", &n);
	string kari;
	for (int i = 0; i < n; i++){
		cin >> kari;
		string m=kari;
		for(int i=0;i<kari.size();i++){
			kari[i]=tolower(kari[i]);
		}
		if(kari!=m){
			return 1;
		}
		mp[kari].push_back(i);
	}
	int q;
	scanf("%d", &q);
	if(q==0){
		return 1;
	}
	int cc=q;
	while (q--){
		cc--;
		int l, r;
		scanf("%d%d", &l, &r);
		l--;
		r--;
		cin >> kari;
		if (mp.count(kari)){
			int ind = lower_bound(mp[kari].begin(), mp[kari].end(), l) - mp[kari].begin();
			int indd = upper_bound(mp[kari].begin(), mp[kari].end(), r) - mp[kari].begin();
			indd -= ind;
			printf("%d", indd);
		}
		else{
			printf("%d",0);
		}
		if(cc){
			puts("");
		}
	}
	return 0;
}
