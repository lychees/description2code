#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    
        int n;
        cin >> n;
        int i,a[n],s[n];
        int sum = 0;
        for(i=0;i<n;i++)
            {
            cin >> a[i];
            if(a[i]>0)
                {
            sum = sum + a[i];
            }
        }
        if(sum==0)
            {
            sum = a[0];
            for(i=1;i<n;i++)
                {
                   if(a[i]>sum)
                       {
                       sum = a[i];
                   }
            }
        }
        s[0]=a[0];
        int max_sum = s[0];
        for(i=1;i<n;i++)
            {
              if(s[i-1]+a[i]>a[i])
                  {
                  s[i] = s[i-1]+a[i];
              }
            else
                {
                  s[i]=a[i];
            }
            if(s[i]>max_sum)
                {
                max_sum = s[i];
            }
        }
        cout << max_sum <<endl;
    
    return 0;
}
