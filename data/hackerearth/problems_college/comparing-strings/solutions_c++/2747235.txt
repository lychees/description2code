#include<bits/stdc++.h>
using namespace std;
main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char a[100000],b[100000];
		cin>>a>>b;
		int l1=strlen(a);
		int l2=strlen(b);
		sort(a,a+l1);
		sort(b,b+l2);
		int flag=0;
		for(int i=0;i<l1;i++)
		{
			for(int j=0;j<l2;j++)
			{
				if(a[i]==b[j])
				{
					flag=1;
					break;
				}
			}
			if(flag==1)
			break;
		}
		if(flag==1)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
	}
}
