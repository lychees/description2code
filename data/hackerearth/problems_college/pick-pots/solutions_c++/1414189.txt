#include <iostream>
#include <cstdio>
#include <math.h>

using namespace std;

#define MOD 1000000007
 

int f[1000005];
int main()
{
	
	int t,n,i;
	scanf("%d",&t);
	f[0]=1;
	f[1]=1;
	for(i=2;i<=1000005;i++)
	{
		f[i] = ((f[i-1]%MOD) + (f[i-2]%MOD))%MOD;
	}
	while(t--)
	{
		scanf("%d",&n);
		printf("%d\n",f[n+1]-1);
	}
	return 0;
	
}
