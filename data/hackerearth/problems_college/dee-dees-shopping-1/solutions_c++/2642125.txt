 #include <iostream>

using namespace std;

int main()
{
        int ts,p,k,r;
        cin>>ts;
        while(ts--){
                cin>>p;
                k=p%8;
                r=0;
                while(k!=0&&p>1){
                        r++;
                        p=p-6;
                        k=p%8;
                }
                r=r+p/8;
                if(p%8!=0)
                        cout<<"-1"<<endl;
                        else
                                cout<<r<<endl;
        }
        return 0;
}
