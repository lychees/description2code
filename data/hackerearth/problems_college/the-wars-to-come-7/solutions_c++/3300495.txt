#include <iostream>
using namespace std;

int main()
{
    int n,i,count=0,sum=0,j,temp;
    cin>>n;
    int a[n];
    int x;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    	
    	
    	
    }cin>>x;
    for(i=0;i<n-1;i++)
    {
    	for(j=0;j<n-i-1;j++)
    	{
    		if(a[j]<a[j+1])
    		{
    			temp=a[j];
    			a[j]=a[j+1];
    			a[j+1]=temp;
    			
    		}
    		
    		
    	}
    	
    	
    }
    for(i=0;i<n;i++)
    {
    	sum=sum+a[i];
    	count++;
    	if(sum>=x)
    	break;
    	
    	
    }
    cout<<count;
    
    
    return 0;
}
