#include<iostream>
#include<vector>
#include<queue>
#include<algorithm>
#include<math.h>
using namespace std;
long long int d=0;
long long int fun(long long int n)
{
	if(n<1)
	return d;
	if(n<10)
	{
		d+=n;
		return d;
	}
	else 
	{
		long long int n2=n,c=0;
		while(n2)
		{
			n2=n2/10;
			c++;
		}
		int k=c-1,s=0;
		s=pow(10,k);
		s--;
		n2=n-s;
		n=s;
		d+=c*n2;
		fun(n);
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		d=0;
		long long int n,ans;
		cin>>n;
		ans=fun(n);
		cout<<ans<<endl;
	}
	return 0;
}
