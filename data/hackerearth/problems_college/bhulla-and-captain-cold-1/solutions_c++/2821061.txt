#include <bits/stdc++.h>
#define P 3
using namespace std;



int main() {
  int T,ans;
  cin>>T;
  for(int i=0;i<T;i++){
  	int N;
  	cin>>N;
  	vector<vector<int>> ss(N,vector<int>(P));
  	for(int i=0;i<N;i++){
  		for(int j=0;j<P;j++){
  			cin>>ss[i][j];
  		}
  	}
  	vector<vector<int>> dp(N,vector<int>(P));
  	for(int i=0;i<P;i++){
  		dp[0][i]=ss[0][i];
  	}
  	for(int i=1;i<N;i++){
  	  dp[i][0] = min(dp[i-1][1] , dp[i-1][2]) + ss[i][0]; 
      dp[i][1] = min(dp[i-1][0] , dp[i-1][2]) + ss[i][1]; 
      dp[i][2] = min(dp[i-1][0] , dp[i-1][1]) + ss[i][2];       
    }   	
    ans = min(min(dp[N-1][0], dp[N-1][1]) , dp[N-1][2]);
    cout<<ans<<"\n";
  }
}
