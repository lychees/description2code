#include <iostream>
#include <string>
using namespace std;

int main()
{
   	int t;
   	cin>>t;
   	while(t--)
   	{
   		string s;
   		cin>>s;
   		int arr[26]={0};
   		int i=0;
   		while(s[i]!='\0')
   		{
   			if(arr[s[i]-'a']==1)--arr[s[i]-'a'];
   			else
   				++arr[s[i]-'a'];
   			++i;		
   		}
   		int sm =0;
   		for(int j=0;j<26;++j)sm+=arr[j];
   		if(sm>0)
   			cout<<-1<<endl;
   		else
   			cout<<1<<endl;
   	}
    return 0;
}
