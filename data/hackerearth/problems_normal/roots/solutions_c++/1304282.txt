#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int i,j,k,n,a,b,c,ans;
    scanf("%lld",&n);
    long long int count=0;
    while(n--)
    {
        scanf("%lld%lld%lld",&a,&b,&c);
        ans=b*b-4*a*c;
        if(ans>=0)
            count++;
    }
    printf("%lld\n",count);
    return 0;

}
