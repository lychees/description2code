#include <stdio.h>
#include <math.h>
#include <iostream>
using namespace std;

int main()
{
	// your code goes here
	int n,a[100001],i,temp,j;
	long long int dp[100001];
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	dp[0]=a[0];
	dp[1]=a[1]+dp[0];
	dp[2]=a[2]+dp[0];
	for(i=3;i<n;i++)
	{
		dp[i]=a[i]+dp[0];
		temp=i+1;
		for(j=2;j<sqrt(temp);j++)
		{
			if(temp%j==0)
			{
				if((dp[j-1]+a[i]>dp[i]))
					dp[i]=dp[j-1]+a[i];
				if(dp[temp/j-1]+a[i]>dp[i])
					dp[i]=dp[temp/j-1]+a[i];
			}
		} 
		if(j*j==temp)
			if((dp[j-1]+a[i]>dp[i]))
					dp[i]=dp[j-1]+a[i];
	}
	for(i=0;i<n;i++)
		printf("%lld ",dp[i]);
	return 0;
}
