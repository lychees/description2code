#include<bits/stdc++.h>
using namespace std;
const int N=1e6+8;
int arr[N];
int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        memset(arr,0,sizeof(arr));
        int lb,ub;
        scanf("%d%d",&lb,&ub);
        for(int i=2;i*i<=ub;i++) {
            int gap=i*i;
            int base=(gap-lb%gap)%gap+lb;
            for(int j=base;j<=ub;j+=gap) {
                arr[j-lb]=1;
                //cout<<j<<endl;
            }
        }
        int ans=0;
        for(int i=0;i<=ub-lb;i++) {
            if(arr[i]==0) ans++;
        }
        printf("%d\012",ans);
    }
    return 0;
}
