#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int array[n];
    for(int i=0;i<n;i++){
    	cin>>array[i];
    }
    int cs=array[0];
    int mx=array[0];
    for(int i=1;i<n;i++){
    	cs=max(array[i],cs+array[i]);
    	mx=max(cs,mx);
    }
    if(mx<=0){
    	cout<<0<<endl;
    }else
    	cout<<mx<<endl;
    return 0;
}
