#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	string s;
    	cin>>s;
    	bool found=false;
    	for(unsigned int i = 0;i<s.size();i++)
    	{
    		if ( s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] =='o' || s[i] == 'u'
    			|| s[i] == 'A' || s[i] == 'E' || s[i] == 'I' || s[i] == 'O' || s[i] == 'U')
    			{
    				found = true;
    				cout<<s[i];
    			}
    	}
    	if ( found )
    		cout<<"\n";
    	else
    		cout<<"No\n";
    }
    return 0;
}
