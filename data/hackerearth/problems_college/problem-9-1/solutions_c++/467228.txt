#include <iostream>
using namespace std;
#include<string.h>
int main()
{
    int n,q,t,p,b,i,count;
    cin>>n;
    int a[n];
    memset(a,0,sizeof(a));
    cin>>q;
    while(q>0)
    {cin>>t>>p>>b;count=0;
    if(t==0)
    {for(i=p;i<=b;i++)
    {if(a[i]==0)
       a[i]=1;
       else
       a[i]=0;
    }
    }
    
    if(t==1)
    {for(i=p;i<=b;i++)
      if(a[i]==1)
      count++;
      cout<<count<<"\n";
    }
    q--;
    }
    return 0;
}
