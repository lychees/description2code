#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
	long long  t;
	cin>>t;
	while(t--)
	{
		long long n;
		cin>>n;
		long long   c=0;
		while(n!=1)
		{
			if(n%3 == 0)
			{
				n = n/3;
				c++;
			}
			else
			if(n%2 == 0)
			{
				n = n/2;
				c++;
			}
			else
			{
				n = n -1;
				c++;
			}
		}
		cout<<c<<"\n";
	}
}
