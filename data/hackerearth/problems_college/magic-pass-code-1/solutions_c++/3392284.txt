#include <iostream>
using namespace std;

int main()
{
  int highest = 0;
  int prevSum = 0;
  int num;
  int sum, dig;
  while(cin >> num)
  {
    sum = 0;
    int n = num;
    
    while(num / 10 >= 1)
    {
      dig = num % 10;
      sum += dig;
      num = (num - dig) / 10;
    }
    
    if(sum >= prevSum)
    {
      highest = n;
      prevSum = sum;
    }
  }
  
  cout << highest;
  return 0;
}
