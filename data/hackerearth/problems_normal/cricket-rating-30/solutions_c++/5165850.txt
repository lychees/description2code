#include <bits/stdc++.h>
using namespace std;

int main() {
	int n, r, s, max = 0;
	vector<int> v, vt;
	vector<int>::iterator it, i;
	cin >> n;
	while(n--){
		cin >> r;
		v.push_back(r);
	}
	for(it=v.begin();it!=v.end();it++) {
		if(*it >= 0) {
			s = 0;
			while(it!=v.end() && *it>=0) {
				s+=*it;
				it++;
			}
			it--;
			vt.push_back(s);
		}
		else {
			s = 0;
			while(it!=v.end() && *it<0) {
				s+=*it;
				it++;
			}
			it--;
			vt.push_back(s);
		}
	}
	
	for(it=vt.begin();it!=vt.end();it++) { 
		s=0;
			if (*it >= 0) {
				i = it;
				while(i!=vt.end()) {
					s += *i;
					if (s > max) {
						max = s;
					}
					i++;
				}
			}
	}
	cout << max <<  endl;

	return 0;
}
