#include<stdio.h>
using namespace std;

int gcd(int a,int b)
{
  return (!b?a:gcd(b,a%b)) ;
}

int  main()
{
 int n , t , a[1001] , g ;

 scanf("%d",&t);
 while(t--)
 {  
     scanf("%d",&n);
     for(int i=1 ; i<=n ; i++) scanf("%d",&a[i]);
    
     g=a[1] ;  
     for(int i=1 ; i<=n ; i++) g=gcd(g,a[i]) ;  

     printf(g!=1?"NO\n":"YES\n");
 }
return 0;
}
