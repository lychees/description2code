#include <iostream>
#include <cstring>
using namespace std;
int n;
string s,team;
int v[300];
int main(){
	int cases;
	cin>>cases;
	while(cases--){
		cin>>n>>s;
		memset(v,0,sizeof v);
		for(char u:s){
			v[u]++;
		}
		int winner=-1,maxi=0,szmaxi=123;
		for(int i=0;i<n;i++){
			cin>>team;
			int op=0;
			for(char u:team){
				op+=v[u];
			}
			if(maxi<op){
				maxi=op,winner=i+1,szmaxi=team.size();
			}
			if(maxi==op&&szmaxi>team.size()){
				winner=i+1,szmaxi=team.size();
			}
		}
		cout<<winner<<"\n";
	}
}
