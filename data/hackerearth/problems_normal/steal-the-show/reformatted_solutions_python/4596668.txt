ceo, coo, cto = [int(x) for x in raw_input().split()]
n = int(raw_input())
arr = [int(x) for x in raw_input().split()]
arr.sort(reverse=True)
res = [0 for i in range(n+3)]
l = len(res)
res[0] = min(coo, cto)
res[l-1] = max(coo, cto)
res[l/2] = ceo
for i in range(l):
    if res[i] == 0:
        res[i] = arr[-1]
        arr.pop()
ans = 0
for i in range(1, l):
    ans += abs(res[i] - res[i-1])
print ans
