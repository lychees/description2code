#include <iostream>
using namespace std;

int main()
{
    int t,i,a[3],x,y,z,r=1,j;
    cin>>t;
    for(i=1;i<=t;i++)
    {
    	for(j=0;j<3;j++)
    	  cin>>a[j];
    	x=a[0];
    	y=a[1];
    	z=a[2];
    	if(x*x+y*y!=z*z&&y*y+z*z!=x*x&&z*z+x*x!=y*y)
    	  cout<<"NO"<<endl;
    	
    	else
    	{
    		while(r!=0)
    		{
    			r=y%x;
    			y=x;
    			x=r;
    		}
    		if(y==1)
    		{
    		  cout<<"YES"<<endl;
    		  r=1;
    		}
    		else
    		{
    			while(r!=0)
    		    {
    		    	r=a[2]%a[1];
    		    	a[2]=a[1];
    		    	a[1]=r;
    	    	}
    	    	if(a[2]==1)
    	    	  cout<<"YES"<<endl;
    	    	else
    	    	  cout<<"NO"<<endl;
    		}
    	}
    	r=1;
    }
    return 0;
}
