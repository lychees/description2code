#include<bits/stdc++.h>

using namespace std;
#define ll long long int 
#define pp pair<ll,ll>
bool comp(pp a,pp b)
{
	if(a.second==b.second)
	return a.first<b.first;
	else
	return a.second<b.second;
	
}
ll m1[100005],m2[100005];
pp kl[100005];
int main()
{
	ll n,i,m,q,c,p;
	
	cin>>n>>c;
	q=0;
	kl[0].first=0;
	kl[0].second=0;
	for(i=1;i<=n;i++)
	{
		cin>>p;
		m1[i]=q;
		
		
		if(p>c)
		q+=c;
		else
		q+=p;
		if(p%c==0)
		{
			p/=c;
			p--;
		}
		else
		{
			p/=c;
		}
		
		kl[i].first=i;
		kl[i].second=p;
		
	}
	sort(kl,kl+n+1,comp);
	
	for(i=1;i<=n;i++)
	{
		m2[kl[i].first]=i;
	//	cout<<i<<"  "<<kl[i].first<<endl;
	}
	cin>>p;
	for(i=1;i<=p;i++)
	{
		cin>>q>>m;
		if(q==1)
		cout<<m1[m]<<endl;
		else
		cout<<m2[m]<<endl;
	}
	
}
