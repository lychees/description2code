#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n;
    cin >> n;
    
    string *S=new string[n];
    
    for(int i=0;i<n;i++)
    {
    	cin >> S[i];
    }
    
    sort(S,S+n);
    
    int c=1;
    for(int i=0;i<n-1;i++)
    {
    	if(S[i]!=S[i+1])
    	{
    		cout << S[i] << " " << c << endl;
    		c=1;
    	}
    	else
    	{
    		c++;
    	}	
    }
    cout << S[n-1] << " " << c << endl;
}
