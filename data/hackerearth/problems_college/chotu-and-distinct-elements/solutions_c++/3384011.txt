#include <iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
long int a[1000000],b[1000000];
int main()
{
    long int t,n,j,k,i,count;
    scanf("%ld",&t);
    while(t--) {
    		
    		k=0;
    		scanf("%ld",&n);
    		for(long int i = 0;i<n;i++)
    		{
    			scanf("%ld",&a[i]);
    		   b[i] = -1;
    		}
    		sort(a,a+n);
    		for(long int i = 0;i< n;i++)
    		{
    			j=0;
    			while(a[i] != b[j] && j <=k )
    			{
    				j++;
    			}
    			if(j == k+1)
    			 {
    			 	b[k] = a[i];
    			     k++; 
    			 }
    		}
    
		   	count = 0;
    		i = 0;
    		while(i<k)
    		{
    			if(count %2 == 0)
    			{
    				printf("%ld ", b[k-1]);
    				k--;
    				count++;
    			}
    			else
    			{
    				printf("%ld ", b[i]);
    				i++;
    				count++;
    			}
    		}
    		
    		printf("\n");
    }
    return 0;
}
