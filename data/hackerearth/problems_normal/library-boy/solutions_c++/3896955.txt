#include<bits/stdc++.h>
using namespace std;
int  main(int argc, char const *argv[])
{
	int n,i;
	cin>>n;
	string str[2000];
	int a[26];
	memset(a,0,sizeof(a));
	for(i=0;i<n;i++){
		cin>>str[i];
		a[str[i][0]-97]++;
	}
	int res=0,temp_val,ten_power,temp;
	for (i = 0; i < 26; ++i)
	{
		temp=a[i];
		//cout<<temp<<" ";
		if(temp<=10 && temp!=0)
			res++;
		else{
			temp_val=temp/10;
			ten_power=temp_val*10;
			if(a[i]>ten_power)
				res+=temp_val+1;
			else
				res+=temp_val;
		}
		//cout<<res<<endl;

	}
	cout<<res<<endl;
	return 0;
}
