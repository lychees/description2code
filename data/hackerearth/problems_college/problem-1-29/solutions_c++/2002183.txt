template<class T>
inline void readInt( T &n )//fast input function
{
   n=0;
   T ch=getcx();int sign=1;
   while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}

   while(  ch >= '0' && ch <= '9' )
           n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
   n=n*sign;
}

int main() {
    //freopen("test.in","r",stdin);

    int T, n, a[55][55], b[55][55];

    cin >> T;
    while(T--) {
        cin >> n;
        rep(i, n) rep(j, n) cin >> a[i][j];
        rep(i, n) rep(j, n) b[i][j] = 0;

        queue<PII> pq;
        bool ok = false;

        pq.push(MP(0, 0));
        while(!pq.empty()) {
            PII top = pq.front();
            pq.pop();

            if(top.X == n - 1 && top.Y == n - 1) { ok = true; break; }
            if(b[top.X][top.Y]) continue;
            b[top.X][top.Y] = 1;

            if(top.X + 1 < n && a[top.X + 1][top.Y]) pq.push(MP(top.X + 1, top.Y));
            if(top.Y + 1 < n && a[top.X][top.Y + 1]) pq.push(MP(top.X, top.Y + 1));
        }

        if(ok) pn("POSSIBLE");
        else pn("NOT POSSIBLE");
    }



    return 0;
}
