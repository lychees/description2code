#include <iostream>
using namespace std;

int main()
{
    int t;
    long long int a,b,temp;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	if(b > a)
    	{
    		while(b % a != 0)
    		{
    			temp = b;
    			b = a;
    			a = temp % a;
    		}
    	}
    	else
    	{
    		while(a % b != 0)
    		{
    			temp = a;
    			a = b;
    			b = temp % b;
    		}    			
    	}
    	if(a == 1 || b == 1)
    	{
    		cout<<"Is a Co-Prime"<<endl;
    	}
    	else
    	{
    		cout<<"Not a Co-Prime"<<endl;
    	}
    }
    return 0;
}
