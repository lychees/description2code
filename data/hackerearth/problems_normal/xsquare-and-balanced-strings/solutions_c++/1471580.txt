#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char arr[100001];
    	cin>>arr;
    	
    	int alpha[26]={0};
    	
    	for(int i=0;arr[i]!='\0';i++)
    	{
    		alpha[arr[i]-97]++;
    	}
    	int flag=0;
    	for(int i=0;i<26;i++)
    	{
    		if(alpha[i]%2==0)
    		{
    			continue;
    		}
    		else
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==1)
    	cout<<"-1\n";
    	else
    	cout<<"1\n";
    }
    return 0;
}
