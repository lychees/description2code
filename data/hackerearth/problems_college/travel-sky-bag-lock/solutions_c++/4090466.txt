#include<bits/stdc++.h>

using namespace std;

vector< long > prime;
vector< long > collection;

void init(){

    for(long int i=2; i<=10; i++)
        collection.push_back(i);

    for(long int i=2; i<=10; i++)
        if(collection[i] != 0){
            for(long int j=i*2; j<=10; j = j+i)
                collection[j] = 0;
        }

    for(long int j=2; j<collection.size(); j++)
        if(collection[j] != 0)
            prime.push_back(j);
}

int main(){

    init();

  //  FILE *in = freopen("Input4.txt","r", stdin);
   // FILE *out = freopen("Output4.txt","w", stdout);

    long T;
    cin>>T;
    while(T--){
        long N;
        cin>>N;

        string S;
        cin>>S;
        long A[N];
        for(int i=0; i<S.length(); i++)
            A[i] = S[i] - 48;

        int count = 0;
        for(int i=0; i<N; i++){
            int back=0, front=prime[0];
            for(int j=1; j<prime.size(); j++){
                if(A[i] >= prime[j-1]){
                    back = front;
                    front = prime[j];
                }
            }

         //   cout<<"A[i] = "<<A[i]<<" , ("<<back<<" ,"<<front<<" )"<<endl;

            int mi = min(abs((A[i] - back)), abs((A[i] - front)));
            count += mi;
        }

        cout<<count<<endl;
    }

    return 0;
}
