#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m1,m2;
		cin>>n>>m1>>m2;
		int a[n+2];
		for(int i=0;i<m1;i++)
		{
			a[i] = 0;
		}
		a[m1] = 1;
		if(m1>n && m2>n)
		{
			a[n] = 0;
		}
		else if(m2>n && m1<=n)
		{
			for(int i = m1+1;i<=n;i++)
			{
				if(a[i-m1] == 0)
				{
					a[i] = 1;
				}
				else
				{
					a[i] = 0;
				}
			}
		}
		else
		{
			for(int i = m1+1;i<=n;i++)
			{
				if(a[i-m1] == 0 || a[i-m2] == 0)
				{
					a[i] = 1;
				}
				else
				{
					a[i] = 0;
				}
			}
		}
		cout<<a[n]<<"\n";
	}
}
