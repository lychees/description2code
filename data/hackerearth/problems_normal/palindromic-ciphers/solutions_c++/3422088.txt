#define MOD 1000000007
string s;
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int T;
	int i,n;
	long long re;
	cin>>T;
	while(T--)
	{
		cin>>s;
		n=s.size();
		for(i=0;i<n/2;i++)
		{
			if(s[i]!=s[n-i-1])
				break;
		}
		if(i<n/2)
		{
			re=1;
			for(i=0;i<n;i++)
				re*=s[i]-'a'+1;
			cout<<re<<endl;
		}
		else
			cout<<"Palindrome"<<endl;
	}
//	system("pause");
	return 0;
}
