#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    scanf("%d", &t);
    while(t--){
    	int flag=0;
    	int n;
    	scanf("%d", &n);
    	int a[n];
    	int sum=0;
    	for(int i=0; i<n; i++){
    	  scanf("%d", &a[i]);
    	  sum+=a[i];
    	}
    	int left=0;
    	for(int i=0;i<n;i++){
    		sum=sum-a[i];
    		if(sum==left) flag=1; 
    		

    	   else
    	    left=left+a[i];
    		
    	}
    	if(flag==1)
    	 printf("YES\n");
    	 else
    	printf("NO\n");
    	
    	
    }
    return 0;
}
