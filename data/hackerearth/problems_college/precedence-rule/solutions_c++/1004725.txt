#include <iostream>
#include<string.h>
using namespace std;
long long fun(long long a, long long b, char c)
{
	if(c=='+')return a+b;
	if(c=='*')return a*b;
	if(c=='-')return b-a;
}
int main() {
	// your code goes here
	int t, i, l;
	cin>>t;
	while(t--)
	{
		char s[32];long long ans=0;
		cin>>s;
		l=strlen(s);
		if(l==1)ans=s[0]-'0';else
		ans=fun(s[l-3]-'0', s[l-1]-'0', s[l-2]);
		for(i=l-3; i>=2; i-=2)
		{
			ans=fun(s[i-2]-'0',ans, s[i-1]);
		}
		cout<<ans<<endl;
	}
	return 0;
}
