#include <bits/stdc++.h>
using namespace std;
int  main(){
	int t,a,b,c,mini,maxi;

	scanf("%d",&t);
	while(t--){
		maxi=0;
		scanf("%d %d %d",&a,&b,&c);
		int arr[3];		
		if(a==b){
			arr[0]=(a+b);
		}	
		else{
			mini=(a<b)?a:b;
			arr[0]=2*mini+1;
		}
		maxi=arr[0];		
		if(b==c){
			arr[1]=(b+c);
		}	
		else{
			mini=(b<c)?b:c;
			arr[1]=2*mini+1;
		}		
		if(arr[1]>maxi)
			maxi=arr[1];		
		if(a==c){
			arr[2]=(a+c);
		}	
		else{
			mini=(a<c)?a:c;
			arr[2]=2*mini+1;		
		}		
		if(arr[2]>maxi)
			maxi=arr[2];		
		printf("%d\n",maxi);
	}
	return 0;
}
