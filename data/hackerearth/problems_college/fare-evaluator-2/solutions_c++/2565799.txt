#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int d,wt,fare=0;
    	cin>>d>>wt;
    	if((d>0)&&(d<=60))
    	{
    		if(d<=2)
    			fare=50+wt*2;
    		else
    			fare=((d-2)*12)+50+(wt*2);
    		
    		cout<<fare<<endl;
    	}
    	else
    		cout<<"NA"<<endl;
    }
    return 0;
}
