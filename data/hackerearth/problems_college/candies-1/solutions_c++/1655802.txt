#include <iostream>
using namespace std;
#include<algorithm>
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	long long int a[1001],sum=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		sum=sum+a[i];
    	}
    	int temp=0;
    	long long int s=sum/n;
    	if(sum%n==0)
    	temp=1;
    	if(temp==1)
    	cout<<"YES"<<"\n";
    	else
    	cout<<"NO"<<"\n";
    	cout<<(s)<<" "<<sum%n<<"\n";
    }
}
