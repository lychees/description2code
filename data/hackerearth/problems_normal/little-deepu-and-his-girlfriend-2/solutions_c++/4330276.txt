#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,m,n;
    cin>>t;
    while(t--){
    	cin>>m>>n;
    	int arr[n];
    	for(int i=0;i<n;++i)
    		cin>>arr[i];
    	bool dp[m+1];
    	sort(arr,arr+n);
    	memset(dp,false,sizeof(dp));
    	for(int i=0;i<=m;++i){
    		for(int j=0;j<n;++j){
    			if(i-arr[j]<0)
    				break;
    			if(!dp[i-arr[j]]){
    				dp[i]=true;
    				break;
    			}
    		}
    	}
    	if(dp[m])
    		cout<<"Little Deepu\n";
    	else
    		cout<<"Kate\n";
    }
    
    return 0;
}
