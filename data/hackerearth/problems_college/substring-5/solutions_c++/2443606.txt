#include <iostream>
#include<string>
using namespace std;

int main()
{
	string s;
	cin>>s;
	long long ans=0;
	int i=0,l=s.length();
	char curr;
	while(i<l){
		int count=0;
		curr=s[i];
		while(s[i]==curr&&i<l)
		count++,i++;
		ans+=(count*(count+1))/2;
		
	}
	cout<<ans<<endl;
    //cout << "Hello World!" << endl;
    return 0;
}
