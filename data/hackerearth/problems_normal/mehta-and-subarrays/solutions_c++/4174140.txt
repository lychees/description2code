#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef long l;
typedef unsigned long long ull;
typedef pair<int,int> pii;
typedef pair<int,pair<int,int> > tii;
typedef vector<pair<int,int> > vec_pii;
typedef vector<vector<int> > vec_vec;
#define mp make_pair
#define setbits(n) __builtin_popcount((n))
#define pb push_back
#define mod 1000000007
#define gcd(a,b) __gcd(a,b)
#define sf(n) scanf("%lld",&(n))
#define pf(n) printf("%lld\n",(n))
#define min3(a,b,c) (min((c),min((a),(b))))
#define max3(a,b,c) (max((a),max((b),(c))))
#define newline "\n"

int main()
{
    int n;
    cin>>n;
    int arr[n+1];
    vector<pair<ll,int> > cum;
    cum.resize(n+1);
    cum[0] = mp(0,0);
    for(int i=1;i<=n;++i)
    {
        cin>>arr[i];
        cum[i].first= cum[i-1].first+arr[i];
        cum[i].second = i;
    }
    sort(cum.begin(),cum.end());
    int minimum = cum[0].second;
    int count =0,ans = 0;

    for(int i=1;i<=n;++i)
    {
        if(cum[i].second>minimum)
        {
            // cout<<cum[i].second<<" "<<minimum<<" "<<cum[i].second-minimum<<" ";
            if(ans==cum[i].second-minimum)
                count++;
            else if(ans<cum[i].second-minimum)
            {
                ans = cum[i].second-minimum;
                count =1;
            }
            // cout<<count<<" "<<ans<<endl;
        }
        minimum = min(minimum,cum[i].second);
    }
    if(ans==0)
        cout<<"-1"<<newline;
    else
        cout<<ans<<" "<<count<<newline;
    
}
