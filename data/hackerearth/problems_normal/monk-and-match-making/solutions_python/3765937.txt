s = raw_input()
n = (int)(raw_input())
for i in xrange(n):
	l1, r1, l2 ,r2 = map(int, raw_input().split(" "))
	x = s[l1-1:r1]
	y = s[l2-1:r2]
	if x == y :
		print("Yes")
	else:
		print("No")
