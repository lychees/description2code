#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
main()
{
	int t,n,a[21],b,ans;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(int i=0;i<n;++i)
		cin>>a[i];
		b=INT_MAX;
		sort(a,a+n);
		for(int i=0;i<n;++i)
		{
			ans=0;
			ans+=(a[i]-a[0])+(a[n-1]-a[0]);
			b=min(ans,b);
		}
		cout<<b*2<<endl;
	}
}
