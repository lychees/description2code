#include <iostream>
#include <cmath>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);

    int t;
    cin >> t;
    while(t--) {
        int n, b, g;
        cin >> n >> b >> g;
        cout << (abs(b - g) > 1 ? "Little Jhool wins!" : "The teacher wins!") << "\n";
    }

    return 0;
}
