#include <iostream>
#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll modular_pow(ll base, ll exponent, int modulus)
{
    ll result = 1;
    while (exponent > 0)
    {
        if (exponent % 2 == 1)
            result = (result * base) % modulus;
        exponent = exponent >> 1;
        base = (base * base) % modulus;
    }
    return result;
}
int main()
{
    int t,ff;
    cin>>t;
    ll x, y;
    int mod;
    while(t--)
    {
        ff=0;
        cin>>x;
        cin>>y;
        cin>>mod;
        int res=modular_pow(x, y , pow(10,mod));
        while(res>0)
        {
            ff+=res%10;
            res/=10;
        }
        printf("%d\n",ff);
    }
    return 0;
}
