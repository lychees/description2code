#include <bits/stdc++.h>
using namespace std;
#define ll long long

long long ar[100];

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	int t; cin >> t;
	for (long long i = 1; i < 46; ++i) {
		ar[i] = ar[i - 1] + i * (1LL << i);
//		cout << ar[i] << endl;
	}
	while (t--) {
		ll n; cin >> n;
		ll x = 1;
		while (n > ar[x]) {
			++x;
		}
		n -= ar[x-1];
		// cout << n << endl;	
		ll fl = n/x;
		ll ind = n % x;
		if (ind) fl++;
		ll goup = (x - ind)%x;
		// cout << x << " " << fl << " " << goup << endl;
		while (goup--) {
			if (fl % 2) {
				fl++;
			}
			fl /= 2;
		}
		cout << (fl % 2 ? "Hacker\n" : "Earth\n");
	}

	return 0;
}
