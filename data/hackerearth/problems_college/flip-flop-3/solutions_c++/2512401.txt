#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	int max=0,a[100005]={0};
    	cin>>str;
    	int l=str.length(),ctr=1,px=-1,py=-1;
    	if(str[0]=='X') px=0;
    	else py=0;
    	a[0]=1;
    	for(int i=1;i<l;i++){
    		if(str[i]=='X'){
    			if(py!=-1) ctr=a[py]+1;
    			else ctr=1;
    			px=i;
    			a[i]=ctr;
    		}
    		else{
    			if(px!=-1) ctr=a[px]+1;
    			else ctr=1;
    			py=i;
    			a[i]=ctr;
    		}
    	}
    	for(int i=0;i<l;i++)
    	if(a[i]>max) max=a[i];
    	cout<<l-max<<endl;
    }
    return 0;
}
