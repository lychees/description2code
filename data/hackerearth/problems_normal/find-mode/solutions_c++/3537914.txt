#include <algorithm>
#include <cstdio>
#include <cstdlib>

using namespace std;

void solve() {
	int N;
	scanf("%d", &N);
	vector<int> v(N);
	for (int i = 0; i < N; ++i) {
		scanf("%d", &v[i]);
	}
	sort(v.rbegin(), v.rend());
	int prev = v[0] + 1;
	int count = 0;
	int max_count = 0;
	vector<int> results;
	for (const int val : v) {
		if (prev == val) ++count;
		else {
			prev = val;
			count = 1;
		}
		if (count > max_count) {
			max_count = count;
			results.clear();
			results.push_back(val);
		}
		else if (count == max_count) results.push_back(val);
	}
	for (const int val : results) {
		printf("%d ", val);
	}
	printf("\n");
}
	
	

int main() {
	int T;
	scanf("%d", &T);
	while(T--) {
		solve();
	}
	return 0;
}
