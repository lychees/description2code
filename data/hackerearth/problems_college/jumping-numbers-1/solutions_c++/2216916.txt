#include<cstdio>
const long long int m = 4294967143LL;
long long int z, x, i, j, start, wynik, w[265][11];

int main()
{
  scanf("%lld", &z);
  
  while(z--)
  {
    scanf("%lld", &x);
    wynik = 0;
    for(start = 1; start < 11; start++)
    {
      for(i = 0; i < 11; i++)
      {
	w[0][i] = 0;
      }
      w[0][start] = 1;
      
      for(j = 1; j < x; j++)
      {
	for(i = 0; i < 11; i++)
	{
	  w[j][i] = 0;
	  if (i > 0){w[j][i] += w[j - 1][i - 1];}
	  if (i < 10){w[j][i] += w[j - 1][i + 1];}
	  w[j][i] %= m;
	}
      }
      if (start > 0)
      {
// 	printf("dla poczatku %lld dodaj %lld\n", start, w[x - 1][start - 1]);
	wynik += w[x - 1][start - 1];
      }
      if (start < 10)
      {
// 	printf("dla poczatku %lld dodaj %lld\n", start, w[x - 1][start + 1]);
	wynik += w[x - 1][start + 1];
      }
      wynik %= m;
    }
    printf("%lld\n", wynik);
  }
  
return 0;
}
