#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t,i,count;
    char str[1000];
    char ch;
    cin>>t;
    while(t--)
    {
       cin>>str;
       count=0;
       for(i=0;i<strlen(str);i++)
       {
         ch=str[i];
         if(ch=='a' || ch=='A' || ch=='e' || ch=='E' || ch=='i' || ch=='I' || ch=='o' || ch=='O' || ch=='u' || ch=='U')
       {
       count=count+1;
       cout<<ch;
       }
      }
    if(count==0)
    {cout<<"No"<<endl;}
    else if(count>0)
    {cout<<endl;}
    }
    return 0;
}
