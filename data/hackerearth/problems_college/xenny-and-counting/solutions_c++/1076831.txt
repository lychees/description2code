#include<stdlib.h>
#include<iostream>

using namespace std;

unsigned long long int C[100][100],j,i;

void init_comb()
{
    for (i = 0; i <= 50 ; i++)
    {
        for (j = 0; j <= i; j++)
        {
            if (j == 0 || j == i) 
                C[i][j] = 1;
            else
                C[i][j] = C[i-1][j-1] + C[i-1][j];
        }
    }
}


int main()
{
	unsigned long long int n,k,t;
	init_comb();
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		cout<<C[n+k-1][n-1]<<endl;
	}
	return 0;
}
