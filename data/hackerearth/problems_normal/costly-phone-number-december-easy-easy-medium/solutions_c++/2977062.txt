#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int arr[10];
		for(int i=0;i<10;i++)
			cin>>arr[i];
		
		int l;
		char str[10000];
		cin>>l>>str;
		
		int flag=true;
		while(flag)
		{
			flag=false;
			for(int i=0;i<10;i++)
			{
				for(int j=0;j<10;j++)
				{
					if(arr[(i+j)%10]>arr[i]+arr[j])
					{
						arr[(i+j)%10]=arr[i]+arr[j];
						flag=true;
					}
				}
			}
		}
		int sum=0;
		for(int i=0;str[i]!='\0';i++)
		{
			int x=str[i]-'0';
			sum+=arr[x];	
		}
		cout<<sum<<endl;
	}
}
