from collections import Counter
for j in range(int(raw_input())):
    a = []
    n = int(raw_input())
    k = map(int, raw_input().split())
    for i in xrange(n):
        a.append(k[i] % 10)
    data = Counter(a)
    b = data.most_common()
    count = 0
    for m in range(len(b)):
        count += b[m][1]-1
    print count
