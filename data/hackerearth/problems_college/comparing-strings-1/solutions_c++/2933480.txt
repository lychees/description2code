#include<iostream>
#include<algorithm>
#include<string.h>
using namespace std;
char a[1000001],b[1000001];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        cin>>a>>b;
        char* z;
        int f=0;
        for(int i=0;i<strlen(a);i++)
        {
            z=find(b,b+strlen(b),a[i]);
            if(z!=b+strlen(b))
            {
                f=1;break;
            }
        }
        if(f==1)
            cout<<"YES\n";
        else
            cout<<"NO\n";
    }
    return 0;
}
