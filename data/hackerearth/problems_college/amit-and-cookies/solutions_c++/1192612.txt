#include<iostream>
#include<cstdlib>
using namespace std;

int compare(const void *a,const void *b){
	return (*(int *)a - *(int *)b);
}

int main(){
	int n,i;
	long int k;

	cin>>n>>k;
	long int a[n];

	for(i=0;i<n;i++)
		cin>>a[i];

	qsort(a,n,sizeof(long int),compare);
	int sum = 0;
	i=0;
	while(1>0){
		sum = sum + a[i];
		if(sum>k){
			cout<<i<<endl;
			break; 
		}
		i++;
	}
return 0;
}
