using namespace std;

typedef pair<ll,ll>pii;

typedef vector<ll> vii;

ll n,x,y,t,ans;
ll a[100007],b[100007];

int main(){
    //fr("toy.txt");
    //fw("out.txt");

    cin >> t;
    while(t--){
        cin >> n >> x >> y;
        for(ll i=0;i<n;i++) cin >> a[i] >> b[i];
        sort(a,a+n);
        sort(b,b+n);

        ans = 0;
        for(ll i=0;i<n;i++){
            ll step = abs(b[i]-a[i]);
            if((b[i]-a[i])>0) ans+=(step*x);
            else if((b[i]-a[i])<0) ans+=(step*y);
        }
        cout << ans << endl;
    }

    return 0;
}
