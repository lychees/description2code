#include <iostream>
#include<algorithm>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--){
	string s;
	int a[4] = {0};
	cin>>s;
	for(int j = 0; j < s.length(); j++){
		if((s[j] - 'r') == 0){
			a[0]++;
		}
		else if((s[j] - 'u') == 0){
			a[1]++;
		}
		else if((s[j] - 'b') == 0){
			a[2]++;
		}
		else if((s[j] - 'y') == 0){
			a[3]++;
		}
	}
	sort(a, a + 4);
	cout<<a[0]<<endl;
	}
	return 0;
}
