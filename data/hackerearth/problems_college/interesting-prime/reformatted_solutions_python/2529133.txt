ntc = input()
tc = 0
factors = set()
while tc < ntc:
    num = input()
    div = 1
    while div*div <= num:
        temp = num % div
        temp2 = num/div
        if (temp == 0):
            factors.add(temp2)
            factors.add(div)
        div = div+1
    if num == 1:
        print 1
    else:
        print len(factors)-1
    tc = tc+1
    factors.clear()
