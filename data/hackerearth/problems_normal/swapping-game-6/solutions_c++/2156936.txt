#include<bits/stdc++.h>
using namespace std;
string swap(string s){
	string s2=s;
	int len=s.size();
	int i;
	for(i=0;i<=len/2;i++){
		s2[i<<1]=s[i];
		s2[(i<<1)+1]=s[len-i-1];
	}
	return s2;
}
int main(){
	long long k,i;
