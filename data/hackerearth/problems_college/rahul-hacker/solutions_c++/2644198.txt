#include <iostream>
using namespace std;
int reverse(long long int n)
{
	long long int sum=0,rem=0;
	
	while(n!=0)
	{
	 rem=n%10;
	 sum=sum*10+rem;
	 n=n/10;
	}
	return sum;
}

int main()
{
    //cout << "Hello World!" << endl;
    long long int n,t,a,b,a1,b1,res=0;
    cin>>t;
    while(t--)
    {
     cin>>a;
     cin>>b;
     a1=reverse(a);
     b1=reverse(b);
     res=reverse(a1+b1);
     cout<<res<<endl;
    }
    
    return 0;
}
