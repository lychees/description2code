#include <iostream>
#include <cstring>

using namespace std;

int main(void) {
	int t;
	cin >> t;
	while(t--) {
		int n;
		cin >> n;
		int a[n];
		for(int i = 0; i < n; i++)
			cin >> a[i];
		int dp[n];
		memset(dp, 0, sizeof dp);
		dp[0] = a[0];
		dp[1] = a[1];
		for(int i = 2; i < n; i++) {
			dp[i] = max(dp[i-1], a[i]+dp[i-2]);
		}
		cout << dp[n-1] << endl;
	}
	return 0;
}
