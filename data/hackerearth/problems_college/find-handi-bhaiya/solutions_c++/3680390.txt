#include <iostream>
using namespace std;
int main()
{
   long int T,a,b,c;
   cin>>T;
   for(int i=0;i<T;i++)
   {
   	int flag=0;
   	cin>>a>>b>>c;
   	long int sum1,sum2,sum3;
   	sum1=a+b;
   	sum2=b+c;
   	sum3=a+c;
   	if(sum1>c)
   	{
   		if(sum2>a)
   		if(sum3>b)
   	 flag=1;
   		
   	}
   	if(flag==1)
   	{
   		if(a==b&&b==c)
   		cout<<"YES. It is a equilateral triangle.\n";
   		else if(a==b||b==c||a==c)
   		cout<<"YES. It is a isosceles triangle.\n";
   		else cout<<"YES. It is a scalene triangle.\n";
   		
   	}
   	else cout<<"NO\n";
   }
    return 0;
}
