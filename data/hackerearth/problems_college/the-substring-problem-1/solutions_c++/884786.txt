#include <bits/stdc++.h>
using namespace std;
 
int main() 
{
    int t;
    cin>>t;
    while(t--) {
        int n,i,j;
        cin>>n;
        int a[n+1][27];
        for(i = 0; i < n; i++)
        for(j = 0; j < 26; j++) {
            a[i][j] = 0;
        }
        string str[n];
        for(i = 0; i < n; i++) {
            cin>>str[i];
            int l = str[i].size();
            for(j = 0; j<l; j++) {
                if(!a[i][str[i][j]-97])
                a[i][str[i][j]-97]++;
            }
        }
        int maxi = -1;
        int s = 0;
        int c=27;
        /*for(i = 0; i < n; i++) {
        	for(j = 0; j < 26; j++) {
        		cout<<a[i][j]<<" ";
        	}
        	cout<<endl;
        }*/
        for(j = 0; j < 26; j++) {
            s = 0;
            for(i = 0; i < n; i++) {
                s += a[i][j];
            }
            if(s == maxi) {
                if(j < c)
                c = j; 
            }
            if(s > maxi) {
            	maxi = s;
            	c = j;
            }
        }
        //cout<<str[0]<<" "<<str[1];
        printf("%c\n",char(c+97));
    }
	return 0;
}
