bool ispalin(string str)
{
    int n = str.length(),i,j;
    for(i=0,j=n-1;i<=n/2;++i,j--)
    {
        if(str[i]!=str[j])
            return false;
    }
    return true;
}

bool isprime(int n)
{
    int val = n,i;
    for(i=2;i<=n+1;++i)
    {
        if(val%i==0)
        {
            return false;
        }
    }
    return true;
}
