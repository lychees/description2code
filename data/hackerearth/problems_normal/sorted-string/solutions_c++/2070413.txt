#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    char a[100],s[100];
    int f[26],l,i,index,top,t,min,tc;
    cin>>tc;
    while(tc--)
    {
    	cin>>a;
    	l=strlen(a);
    	top=-1;
    	for(i=0;i<26;i++)
    	f[i]=0;
    	for(i=0;i<l;i++)
    	{
    		index=int(a[i]-'a');
    		f[index]++;
    	}
    	min=0;
    	for(i=0;i<26;i++)
    	{
    		if(f[i]>min)
    		min=f[i];
    	}
    	t=1;
    	while(t<=min)
    	{
    		for(i=0;i<26;i++)
    		{
    			if(f[i]==t)
    			{
    				for(int j=1;j<=t;j++)
    				{
    					top++;
    					s[top]=char(i+'a');
    				}
    			}
    		}
    		t++;
    	}
    	for(i=0;i<=top;i++)
    	cout<<s[i];
    	cout<<"\n";
    }
    return 0;
}
