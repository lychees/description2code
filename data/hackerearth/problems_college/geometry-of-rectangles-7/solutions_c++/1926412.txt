#include <iostream>
using namespace std;

int main()
{
    int T,x1,y1,x2,y2,x3,y3,x4,y4;
    cin>>T;
    while(T--){
    	cin>>x1>>y1>>x2>>y2;
    	cin>>x3>>y3>>x4>>y4;
    	if ( (x1 == x4 && y1 == y4) || (x2 == x3 && y2 == y3) || (x1 == x4 && y3 == y2) || (x2 == x3 && y1 == y4) ){
    		cout << "Point\n";
    	}
    	else if ( x3 > x2 || x4 < x1 || y3 > y2 || y4 < y1 ){
    		cout << "Apart\n";
    	}
    	else if ( x3 == x2 || x1 == x4 || y3 == y2 || y4 ==y1 ) {
    		cout << "Line\n";
    	}
    	else{
    		cout << "Rectangle\n";
    	}
    }
    return 0;
}
