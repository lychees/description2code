#include<bits/stdc++.h>
using namespace std;
//int count[100000];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i,j;
		cin>>n;
		int num;
		int count[1000002]={0};
		vector<int>vec;
		
		for(i=0;i<n;i++)
		{
		cin>>num;
		count[num]++;
		if(count[num]==1)
		{
			vec.push_back(num);
		}	
		}
		
		sort(vec.begin(),vec.end());
		int p = vec.size();
		if(vec.size()%2==0)
		{
			i=0 ,j=p-1;
		while(i<j)
		{
			cout<<vec[j]<<" "<<vec[i]<<" ";
			j--;
			i++;
		}
		cout<<"\n";
		}
		else
		{
			i=0,j=p-1;
				while(i<j)
		{
			cout<<vec[j]<<" "<<vec[i]<<" ";
			j--;
			i++;
		}
		if(i==j)
		{
			cout<<vec[i]<<"\n";
		}
		}
	}
}
