#include<bits/stdc++.h>
using namespace std;
int a[100001];
int s[100001];
int root(int a[],int i)
{
	while(i!=a[i])
	i=a[i];
	return i;
}
bool find(int a[],int i,int j)
{
	int x=root(a,i);
	int y=root(a,j);
	if(x==y)
	{
		return true;
	}
	else 
	return false;
}
void uni(int a[],int s[],int i,int j,int m)
{
	if(!find(a,i,j))
	{
		int x=root(a,i);
		int y=root(a,j);
		if(s[x]>s[y])
		{
			a[y]=x;
			s[x]+=s[y];
		}
		else
		{
			a[x]=y;
			s[y]+=s[x];
		}
	}
}
int main()
{
		int n,w;
		cin>>n>>w;
		for(int i=1;i<=n;i++)
		{
			a[i]=i;
			s[i]=1;
		}
		int wi,wj;
		for(int i=0;i<w;i++)
		{
			cin>>wi>>wj;
			uni(a,s,wi,wj,n);
		}
		set <int> s;
		for(int i=1;i<=n;i++)
		{
			s.insert(root(a,a[i]));
		}
		cout<<s.size()<<endl;
		return 0;
}
