#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int x[n], y[n],i;
    for(i=0;i<n;i++)
    {
    	cin>>x[i];
    	cin>>y[i];
    }
    int j,home,away;
    for(i=0;i<n;i++)
    {
    	home=n-1, away=n-1;
    	for(j=0;j<n;j++)
    	{
    		if(j!=i)
    			if(y[i]==x[j])
    				{ home++; away--;}
    		
    	}
    	cout<<home<<" "<<away<<"\n";
    }
    return 0;
}
