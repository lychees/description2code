#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    for(int i=1;i<=T;i++) {
    	int n;
    	cin>>n;
    	int a[n];
    	for(int j =0;j<n;j++) {
    		cin>>a[j];
    	}
    	int ans =0;
    	for(int j=0;j<n;j++) {
    		
    		for(int k=j+1;k<n;k++) {
    			if(a[j] != a[k] && (a[j] +a[k])%2 ==0) ans++;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
