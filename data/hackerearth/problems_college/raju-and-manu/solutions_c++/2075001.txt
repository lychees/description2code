#include <iostream>
using namespace std;

int sum(int n){
int count=0;
while(n){
	if(n & 1){
		count++;
	}
	n=n>>1;
}
return count;
}

int main()
{
    int t;
    cin>>t;
    int n;
    while(t--){
    	cin>>n;
    	int max=0,no,y;
    	for(int i=0;i<=n;i++){
    		if(i%2==0){
    			continue;
    		}
    		y=sum(i);
    		if(max<=y){
    			max=y;
    			no=i;
    		}
    		
    	}
    	cout<<no<<"\n";
    }
    return 0;
}
