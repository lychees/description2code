#include <cstdio>
using namespace std;
 
int main()
{
	int a,n,i,x[5000],j,count=0,max,k=0;
    scanf("%d",&a);
    scanf("%d",&n);
    for(i=0;i<n;++i)
    scanf("%d",&x[i]);
    for(i=0;i<n;++i)
    {
    	if(a>x[i])
    	max=x[i];
    	else max=a;
    	for(j=2;j<=max;++j)
    	{
    		if((x[i]%j==0 && a%j==0) || a%x[i]==0)
    		{
    			++count;
    			break;
    		}
    	}
    }
    printf("%d",count);
    return 0;
}
