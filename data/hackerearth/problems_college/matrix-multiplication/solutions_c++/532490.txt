#include <iostream>

using namespace std;

int main()
{
        int test_cases;
        cin>>test_cases;
        while(test_cases > 0)
        {
                int X1,X2,Y1,Y2, count = 0;
                cin>>X1>>Y1>>X2>>Y2;
                
                if(X1 == Y2) count++;
                if(X2 == Y1) count++;
                
                cout<<count<<endl;
                test_cases --;
        }
        return 0;
}
