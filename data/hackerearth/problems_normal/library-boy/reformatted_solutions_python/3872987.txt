n = int(raw_input())
arr = []*n

for i in xrange(n):
    arr.append(raw_input())

count = [0]*26

for i in xrange(0, n):
    count[ord(arr[i][0])-ord('a')] += 1

ans = 0
for i in xrange(0, 26):
    if count[i] % 10 == 0:
        ans += count[i]/10
    else:
        ans += count[i]/10 + 1

print ans
