#include <iostream>
using namespace std;

int main()
{
    long long int i,a[100000],sum,leftsum,ans,t,n;
    cin>>t;
    while(t--)
    {
    	
    	cin>>n;leftsum=0;sum=0,ans=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		sum+=a[i];
    	}
    	for(i=0;i<n;i++)
    	{
    		sum=sum-a[i];
    		if(leftsum==sum)
    		{
    			ans=1;
    			break;
    		}
    		leftsum+=a[i];
    	}
    	if(ans!=0)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
    return 0;
}
