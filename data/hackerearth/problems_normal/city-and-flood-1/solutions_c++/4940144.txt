#include <iostream>
#include <vector>
#define N 100001

using namespace std;

int parent[N];
int cnt;

int x, y, n, m;

int root(int i)
{
    while(parent[ i ] != i)       
    {
    	parent[ i ] = parent[parent[ i ]];
     	i = parent[ i ];
    }
    return i;
}

bool join(int A ,int B)
{
    int root_A = root(A);       
    int root_B = root(B); 
    if (root_A == root_B)
    	return false;
    parent[ root_A ] = root_B ;
    return true;
}

int main()
{
    //freopen("input.txt", "rt", stdin);
	scanf("%d%d", &n, &m);

    for (int  i = 0; i < n; i++)
    {
        parent[i] = i;
    }

    for (int i = 0; i < m; i++)
    {
        vector<int> res;
        scanf("%d%d", &x, &y);
        x--; y--;
        join(x,y);

        
    }
    for (int i = 0; i < n;i++)
        if (parent[i] == i)
            cnt++;
    printf("%d", cnt);
    return 0;
}
