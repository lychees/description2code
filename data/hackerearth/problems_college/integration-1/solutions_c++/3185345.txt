#include <iostream>
using namespace std;

int main()
{
    long long int t,j;
    cin>>t;
    for(j=1;j<=t;j++)
    {
    	long long int N,P,U,R,S,i;
    	cin>>N>>P>>U>>R>>S;
    	for(i=0;i<N;i++)
    	{
    		if(P+R<=U)
    		{
    			P=P+R;
    		}
    		else 
    		{
    			if(P-S>=0)
    			P=P-S;
    		}
    	}
    	cout<<"Case #"<<j<<": "<<P<<endl;
    	
    }
	return 0;
}
