#include <bits/stdc++.h>

using namespace std;

int n,t,a[20];
int main()
{
    int i,j,k,cost;
    string str;
    bool reduced;
    cin>>t;

    while(t--)
    {
        for(i=0;i<=9;i++)
        cin>>a[i];

        while(1)
        {
            reduced=false;
            for(i=0;i<=9;i++)
            {
                for(j=i;j<=9;j++)
                {
                    if(a[i]+a[j]<a[(i+j)%10])
                    {
                        a[(i+j)%10] = a[i]+a[j];
                        reduced = true;
                    }
                }
            }
            if(reduced!=true)
                break;
        }
        cin>>k;
        cin>>str;
        cost=0;
        for(i=0;i<str.length();i++)
        {
        	//cout<<"str"<<str[i];
            cost+=a[str[i]-'0'];
            //cout<<cost<<endl;
        }

        cout<<cost<<endl;
    }
}
