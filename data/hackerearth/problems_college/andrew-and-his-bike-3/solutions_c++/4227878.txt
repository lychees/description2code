ll expo(ll a,ll b,ll c = mod){
    if(!b)return 1;
    ll temp = expo(a,b/2,c);
    temp = (temp*temp)%c;
    if(b&1)temp = (temp*a)%c;
    return temp; 
}
vi G[N];
int su[N];
void dfs(int node,int prev){
	su[node] = 1;
	for(auto &it : G[node]){
		if(it != prev){
			dfs(it,node);
			su[node] += su[it];
		}
	}
}
