#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
	ll n, i, s, d;
    cin>>n;
    ll a[n];
    for(i=0;i<n;i++)
    	cin>>a[i];
    ll b[n];
    b[0]=a[0];
    s=a[0];
    for(i=1;i<n;i++)
    {
    	d=(i+1)*a[i];
    	d=d-s;
    	b[i]=d;
    	s=s+b[i];
    }
    for(i=0;i<n;i++)
    	cout<<b[i]<<' ';
    return 0;
}
