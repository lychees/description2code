#include<iostream>
#include <map>

using namespace std;

map <long long, long long> mapkey;

long long root(long long n)
{
	if(mapkey.find(n)==mapkey.end()){
		long long sum=0;
		if(n==0)
			return 1;
		if(n<0)
			return 0;
		else{
			sum=root(n-1)+root(n-2)+root(n-3);
			mapkey[n]=sum%(1000000007);
		}
		return sum%(1000000007);
	}
	else
		return mapkey[n]%(1000000007);
		
		
}

int main()
{
	mapkey.clear();
	long long n,t;
	cin>>t;
	while(t--)
	{
		cin>>n;		
		cout<<root(n)<<endl;
	}
	return 0;
}
