#include <iostream>
#include<math.h>
using namespace std;

int main()
{
   long long int t,n,x,y,z;
   double temp;
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	x=sqrt(n);   	
   	y=(x+1)*(x+1);
   	z=(x)*(x);   	
   	if(z==n)
   	cout<<"YES"<<endl;
   	else
   	{   		   	
   		if(abs(y-n)>=abs(z-n))
   		{
   			cout<<(z-n)<<endl;
   		}
   		else
   		{
   			cout<<"+"<<(y-n)<<endl;
   		}
   	}
   }
    return 0;
}
