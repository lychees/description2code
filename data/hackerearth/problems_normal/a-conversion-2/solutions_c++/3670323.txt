#include <iostream>
#include<vector>
#include<queue>
#include<set>
#include<cmath>
using namespace std;

int main()
{
	int t;
	cin>>t;
	for(int i = 0; i<t; i++)
	{
		vector<string> t1;
		set<string> s;
    	string x, y;
    	cin>>x>>y;
		int ans = 1000000000;
		int n = 0;
		t1.push_back(x);
		s.insert(x);
    	while(!t1.empty())
    	{
    		vector<string> t2;
    		while(!t1.empty())
    		{
			int temp = n;
    		string current = t1.back();
    		t1.pop_back();
    		for(int j = 0; j<current.size(); j++)
    			temp += abs(current[j] - y[j]);
    		if(temp < ans)
    			ans = temp;
    		for(int k = 0; k<current.size(); k++)
    		{
    			for(int q = k+1; q<current.size(); q++)
    				{
    					swap(current[k], current[q]);
    					if(!s.count(current))
    					{
    						t2.push_back(current);
    						s.insert(current);
    					}
    					swap(current[k], current[q]);
    				}
    		}
    		}
    		t1 = t2;
    		n++;
    	}
    	//cout<<x<<" "<<y<<endl;
    	if(i != t-1)
    		cout<<ans<<endl;
    	else
    		cout<<ans;
	}
    return 0;
}
