#include<bits/stdc++.h>
using namespace std;
unsigned long long int fib[1001] = {0};
void init() {
	fib[0] = 1;
	fib[1] = 1;
	fib[2] = 2;
	fib[3] = 3;
	fib[4] = 5;
	for(int i = 5; i <= 1000; i++) {
		fib[i] = fib[i-1] + fib[i-2];
	}
}
int main() {
	init();
	int tc,n;
	cin>>tc;
	while(tc--) {
		cin>>n;
		cout<<fib[n]<<endl;
	}
	return 0;
}
