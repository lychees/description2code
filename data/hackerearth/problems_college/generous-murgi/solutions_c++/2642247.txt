#include<iostream>
using namespace std;

int main()
{
  int i,t,count;
  long long int d,j,x,y;
  cin>>t;
  for(i=0;i<t;i++)
  {
    cin>>x>>y;
    d=x-y;count=0;
    if(x==y)
    {
      count=-1;
    }
    for(j=1;(j*j)<=d;j++)
    {
        if(j>y)
        {
            if(d%j==0)
            {
                if((j*j)==d)
                    count++;
                else
                    count+=2;
            }
        }
        else
        {
            if(d%j==0&&(d/j)>y)
                count++;
        }
    }
      cout<<count<<endl;
  }


  return 0;
}
