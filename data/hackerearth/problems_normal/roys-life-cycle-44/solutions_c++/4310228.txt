#include <bits/stdc++.h>
using namespace std;

int solve(char a[],int flag)
{
	int count=0;
   	int min=0,k;
   	
   	if(flag==0)
   	k=1440;
   	else
   	k=530000;
   	
	for(int i=0;i<k;i++)
   	{
   		if(a[i]=='C')
   			while(a[i]=='C')
   			count++,i++;
   		if(count>min)
   	     	min=count; 
   	    count=0;
   		
   	}
   	return min;
}


int main()
{
   int t;
   cin>>t;
   int max=0;
   int ans=0;
   char b[530000]={0};
   int flag;
   while(t--)
   {
   	char a[1441];
   	cin>>a;
   	
   	int min=solve(a,0);
    if(min>max)
   	max=min;
   	
   	
   	strcat(b,a);
   	if(t==0)
   	{ flag=1;
   		ans=solve(b,flag);
   	}
   	
    }
   
   if(ans>max)
   cout<<max<<" "<<ans<<endl;
   else
   cout<<max<<" "<<max<<endl;
   
  return 0;
}
