#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int i,f=0,diff=0;
    	string s,ss="";
    	cin>>s;
    	//cout<<s;
    	for(i=s.length()-1;i>=0;i--)
    	{
    		
    		if(f==0 && i == s.length()-4)
    		{
    			s.insert(s.begin()+i+1,',');
    			i++;
    			f=1;
    		}
    		else if( f==1 && diff==2)
    		{
    				s.insert(s.begin()+i+1,',');
    				diff=0;
    				i++;
    		}
    		else
    		{
    			if(f!=0)
    			diff++;
    		}
    	
    	}
    	cout<<s<<endl;
    
    }
}
