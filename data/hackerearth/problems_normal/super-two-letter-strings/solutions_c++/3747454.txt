#include <bits/stdc++.h>
using namespace std;
int t , n , p , i , j;

int main()
{
    scanf("%d" , &t);
    
    while(t--)
    {
    	scanf("%d %d" , &n , &p);
    	long long int dp[10005] = {0};
    	
    	dp[0] = 1;
    	
    	for(i = 1 ; i <= n ; i++)
    	{
    		for(j = i-1 ; j>= (i-p) && (j>=0) ; j--)
    		{
    			dp[i] = (dp[i] + dp[j])%1000000007;
    		}
    	}
    	
    	printf("%lld\n" , dp[n]);
    }
    
    return 0;
}
