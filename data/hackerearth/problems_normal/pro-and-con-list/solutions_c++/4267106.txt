#include<bits/stdc++.h>
using namespace std;
void sol()
{
    int i,n;
    long long t=0;
    cin>>n;
    int a[n+1],b[n+1];
    for(i=0;i<n;++i)
    {
        cin>>a[i]>>b[i];
        t+=b[i];
        a[i]+=b[i];
    }
    sort(a,a+n);
    cout<<((long long)(a[n-1]+a[n-2])-t)<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--) sol();
    return 0;
}
