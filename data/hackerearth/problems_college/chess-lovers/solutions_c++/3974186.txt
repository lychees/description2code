#include <iostream>
using namespace std;
int main()
{
	int t,x1,y1,x2,y2,pos1,pos2;
    cin>>t;
    while(t--){
    	cin>>x1>>y1>>x2>>y2;
    	if(x1%2 == y1%2){
    		pos1 = 0;
    	}
    	else{
    		pos1 = 1;
    	}
    	if(x2%2 == y2%2){
    		pos2 = 0;
    	}
    	else{
    		pos2 = 1;
    	}
    	if(pos1 != pos2){
    		cout<<-1<<endl;
    		continue;
    	}
    	if(abs(x1 - x2) == abs(y1 - y2)){
    		cout<<0<<endl;
    		continue;
    	}
    	else{
    		cout<<1<<endl;
    	}
    }
}
