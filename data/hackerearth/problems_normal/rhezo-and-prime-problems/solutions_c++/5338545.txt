#include <stdio.h>
#include <iostream>

using namespace std;

typedef long long int ll;
int p[5001];
ll dp[5001];
int main() {
	ll n;
	scanf("%lld", &n);
	ll i, j, max = 5000;
	for(i = 2; i * i <= max; i++) {
		if(p[i] == 0) {
			for(j = i * i; j <= max; j+=i) {
				p[j] = 1;
			}
		}
	}
	p[0] = p[1] = 1;
	ll a[n + 1];
	ll sum[n + 1];
	sum[0] = 0;
	for(i = 1; i <= n; i++) {
		scanf("%lld", &a[i]);
		sum[i] = a[i] + sum[i - 1];
	}
	dp[0] = dp[1] = 0;
	for(i = 2; i <= n; i++) {
		if(p[i] == 0) {
			dp[i] = sum[i];
		}
		else {
			dp[i] = dp[i - 1];
			ll s = 0;
			for(j = i - 1; j >= 2; j--) {
				if(p[i + 1 - j] == 0) {
					s = dp[j - 2] + sum[i] - sum[j - 1]; 
				}
				if(s > dp[i]) {
					dp[i] = s;
				}  
			}
		}
	}
	printf("%lld\n", dp[n]);
	return 0;
}
