#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;
typedef int w;
typedef long long l;
w n;
w x;
w f[1000][1000];

void solve(){
     memset(f,0,sizeof f);
    cin>>n>>x;
    for(int i=1;i<=n;++i){
        w y; cin>>y;
        y-=x;
        for(int j=0;j<=n;++j){
            f[i][j]=f[i-1][j];
            if(j>=i) f[i][j]=max(f[i][j],f[i][j-i]+y);
        }
    }
    cout<<f[n][n]<<endl;
}

int main(){
    ios_base::sync_with_stdio(false);
   //freopen("in","r",stdin);
    w T;
    cin>>T;
    while(T--) solve();
}
