#define gc getchar_unlocked


	int ap[4]={2,3,5,7};
	
	
void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc())   
        {x = (x<<1) + (x<<3) + c - 48;}
}
	
	bool isprime(long long x){
	for(int i=0;i<4;i++)
	if(x==ap[i]) return true;
	return false;
		
	}

int main()
{
   int n;
   scanint(n);
   unsigned long long a[n];
   for(int i=0;i<n;i++) scanf("%llu",&a[i]);
    
    if(n==1||n==2){printf("0"); return 0;}
    
    bool p[8][n];
    
    for(int i=0;i<8;i++)
    for(int j=0;j<n;j++)
    p[i][j]=false;
    
    
    unsigned long long ans=0,count=0,c=0,d=0,temp=0;  
    
    for(int i=0;i<n;i++){
    	temp=a[i];
    	while(temp>0){
    		d=temp%10;
    		if(isprime(d)) p[d][i]=true;
    	temp=temp/10;
		}
}
  	 
  	
    	for(int i=0;i<4;i++)
    {
    	count=0;    
    	for(int j=0;j<n;j++)
    	if(p[ap[i]][j]==true) count++;
    	
		if(count>3)
    	ans+=count*(count-1)*(count-2)/6;
   	
	}
 
 //Remove Duplicates:
 
 for(int i=0;i<3;i++)
 for(int j=i+1;j<4;j++)
 { c=0;
 	for(int k=0;k<n;k++)
 if(p[ap[i]][k]&&p[ap[j]][k]) c++;
 ans-=c*(c-1)*(c-2)/6;
 }
 

 
 
 for(int t=0;t<4;t++)
 for(int i=t+1;i<4;i++)
 for(int j=i+1;j<4;j++)
  { 
  	c=0;
  	for(int k=0;k<n;k++)
 
 if(p[(ap[i])][k]&&p[(ap[j])][k]&&p[(ap[t])][k]) c++;
 ans+=c*(c-1)*(c-2)/6;
 }
 
 c=0;
for(int k=0;k<n;k++)
if(p[2][k]&&p[3][k]&&p[5][k]&&p[7][k]) c++;
 ans-=c*(c-1)*(c-2)/6;

    return 0;

}
