  #include<iostream>
  #include<string>
  #include<assert.h>
  using namespace std;
  #define ll long long
  ll fib[99];
  int main(){
  	
  	int t,i;
    scanf("%d",&t);
    assert(1 <= t && t <= 20);
    fib[0] = 0;
    fib[1] = 1;
    for(i=2;i<=90;i++)
        fib[i] = fib[i-1] + fib[i-2];
    while(t--)
    {
        int m;
        scanf("%d",&m);
        assert(1 <= m && m <= 90);
        printf("%lld %lld\n",fib[m-1],fib[m]);
    }
  	return 0;
  }
