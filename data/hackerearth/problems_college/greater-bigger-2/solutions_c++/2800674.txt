#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void swap(char *a,char *b)
{
	char tmp;
	tmp=*a;
	*a=*b;
	*b=tmp;
}
int main()
{
	int t;
	//string s;
	char s[101];
	cin >> t;
	while(t--)
	{
		//cin >> s;
		scanf("%s",s);
		int l=strlen(s);
		int i;
		for(i=l-1; i > 0; i--)
		{
			if(s[i-1] < s[i])
			{
				break;
			}
		}
		if(i==0)
		{
		cout <<"no answer\n";
		continue;
		}
			int smallest=i;
			int small=s[i-1];
			for(int j=i+1; j < l; j++)
			{
				if(s[j] > small && s[j] < s[smallest])
				smallest=j;
			}
			swap(&s[i-1],&s[smallest]);
			sort(s+i,s+l);
			cout << s <<"\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
