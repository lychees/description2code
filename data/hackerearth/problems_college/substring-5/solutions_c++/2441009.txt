#include <iostream>
#include<string>
using namespace std;

int main()
{
	string s;
	cin>>s;
	int l;
	l=s.length();
	int i,cont,count;
	cont=1;
	count=l;
	for(i=1;i<l;i++)
	{
		if(s[i]==s[i-1])
		cont++;
		else
		{
			count+=(cont*(cont-1))/2;
			cont=1;
		}
	}
	count+=(cont*(cont-1))/2;
	cout<<count<<"\n";
    return 0;
}
