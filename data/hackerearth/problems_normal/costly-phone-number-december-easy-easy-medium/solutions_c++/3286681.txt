#include <bits/stdc++.h>
using namespace std;
int cost[20];
int main()
{
	int t,l;
	string s;
	scanf("%d",&t);
	while(t--)
	{
		for(int i=0;i<=9;i++)
		scanf("%d",&cost[i]);
		scanf("%d",&l);
		cin>>s;
		bool ans=true;
		while(ans)
		{
			ans=false;
			for(int i=0;i<=9;i++)
			{
				for(int j=0;j<=9;j++)
				{
					if(cost[(i+j)%10]>cost[i]+cost[j])
					{
						cost[(i+j)%10]=cost[i]+cost[j];
						ans=true;
					}
				}
			}
		}
		int res=0;
		for(int i=0;i<l;i++)
		{
			res=res+cost[s[i]-'0'];
		}
		cout<<res<<"\n";
	}
	return 0;
	
}
