#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    string s;
    long long K;
    cin>>K;
    cin>>s;
    long long count=0,K1=0;
    bool visit[26];
    for(long long i=0;i<s.size();i++){
    	memset(visit,false,sizeof(visit));
    	K1=0;
    	for(long long j=i;j<s.size();j++){
    		if(visit[s[j]-'a']==false){
    			visit[s[j]-'a']=true;
    			K1++;
    			if(K1==K){count++;}
    		}
    		else if(visit[s[j]-'a']==true && K1==K){
    			count++;
    		}
    		else if(K1==K){
    			count++;
    		}
    	}
    }
    cout<<endl<<endl<<count;
    return 0;
}
