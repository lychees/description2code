#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,a,*p;
    cin>>n;
    while(n!=0)
    {
    	cin>>a;
    	p=new int[a];
    
    	for(int i=0;i<a;i++)
    	cin>>p[i];
    	int z=a;
    	for(int j=0;j<a;j++)
    	{
    		for(int k=0;k<j;k++)
    		{
    			
    			if(p[j]==p[k])
    			{z--;break;}
    		}
    	}
    
    	cout<<z<<"\n";
    	n--;
    }
    return 0;
}
