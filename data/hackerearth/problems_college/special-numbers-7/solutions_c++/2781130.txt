int isLucky(ll num)
{
	ll count=0;
	if(num%2==0)
	{
	count++;
	while(num%2==0)
	num/=2;
	}
	for (ll i = 3; i <= sqrt(num); i = i+2)
    {
        if(num%i==0)
        {
        count++;
        while (num%i == 0)
        {
            num = num/i;
            
        }
    }
    }
    if (num > 2)
        count++;
    if(count>=3)
    return 1;
    else
    return 0;
}
