

#include<bits/stdc++.h>
using namespace std;

int main(void){
	
	int a,b,n;
	cin>>a>>b>>n;
	
	if( (a-b) > n ){
		cout<<1; return 0;
	}
	
	int tmp = 0,flag = 1,cnt = 1;

	while( tmp < n ){
		if( flag == 1 ){
			tmp+=a;
			if( tmp > n || tmp == n ){
				cout<<cnt; return 0;
			}
			flag = 0;
		}
		if( flag == 0 ){
			tmp-=b;
			flag = 1;
			cnt++;
		}		
	}
	cout<<"o/p : "<<cnt;
	return 0;
}
