ll ar[MAX];
void init()
{
    for(int i=1;i<MAX;i++)
    {
        int a=0,n=i;
        while(n%5==0)
        {
            a++;
            n/=5;
        }
        ar[i]=ar[i-1]+a;
    }
    for(int i=1;i<MAX;i++)
        ar[i]+=ar[i-1];
}
