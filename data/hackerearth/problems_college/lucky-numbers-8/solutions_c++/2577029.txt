#include <iostream>
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    vector<ll> v;
    int t;
    cin>>t;
    ll n;
    while(t--){
    	cin>>n;
    	v.clear();
    	ll x;
    	for(ll i=0;i<n;i++){
    		cin>>x;
    		v.push_back(x);
    	}
    
    	ll max=0,cnt=0;
    	for(int it=n-1;it>=0;it--){
    		if(v[it]>max){ cnt++; max=v[it]; }
    		//cout<<"--"<<max<<endl;
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
