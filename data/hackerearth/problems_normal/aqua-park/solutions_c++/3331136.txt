#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000000007
#define bitcount __builtin_popcountll
#define ll long long
struct segment
{
    int x1;
    int y1;
    int x2;
    int y2;
};
segment a[105];
int main()
{
	int i,j,k;
	int n,x,y,ydone,pos;
	y=10001;
	sd(x);
	sd(n);
	for(i=0;i<n;i++)
    {
        sd(a[i].x1);
        sd(a[i].y1);
        sd(a[i].x2);
        sd(a[i].y2);
        if(a[i].y2<a[i].y1)
        {
            swap(a[i].y2,a[i].y1);
            swap(a[i].x2,a[i].x1);
        }
        /*if(i==20)
        {
            printf("\n%d %d %d %d\n",a[i].x1,a[i].y1,a[i].x2,a[i].y2);
        }*/
    }
   // printf("\n%d %d %d %d\n",a[20].x1,a[20].y1,a[20].x2,a[20].y2);
    while(1)
    {
        ydone=0;
        pos=-1;
        for(i=0;i<n;i++)
        {
            if(x>=min(a[i].x1,a[i].x2)&&x<=max(a[i].x2,a[i].x1)&&y>=a[i].y2&&x!=a[i].x1)
            {
                if(a[i].y2>ydone)
                {
                    ydone=a[i].y2;
                    pos=i;
                }
            }
        }
        if(pos==-1)
            break;
        else
        {
            //printf("%d:",pos);
            //printf("%d %d ",x,y);
            x=a[pos].x1;
            y=a[pos].y1;
            //printf("%d %d\n",x,y);
        }
    }
    printf("%d\n",x);
	return 0;
}
