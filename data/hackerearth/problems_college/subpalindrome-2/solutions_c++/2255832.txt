#include <bits/stdc++.h>

using namespace std;

long long int calc(long long int dp[][100], string &s, int l, int r)
{
    if (l > r)
        return 0;
    if (dp[l][r])
        return dp[l][r];
    if (l == r)
        return dp[l][r] = 1;
    int i = l, j = r;
    dp[i][j] = calc(dp, s, l, r-1) + calc(dp, s, l+1, r) + 1;
    if (s[i] != s[j])
        dp[i][j] -= (calc(dp, s, l+1, r-1) + 1);
    return dp[i][j];

}

int main()
{
    int t;
    cin >> t;
    string s;
    while (t--) {
        cin >> s;
        long long int dp[100][100];
        memset(dp, 0, sizeof(dp));
        cout << calc(dp, s, 0, s.size()-1) << endl;
    }
    return 0;
}
