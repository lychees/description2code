#include <stdio.h>
long long dp[205][205]={0};
 
long long fun(int k, int n)
{
	if(k > n)
		return 0;
	else if(k == n)
		return 1;
	else if(dp[k][n])
		return dp[k][n];
	dp[k][n] = fun(k, n-k) + fun(k+1, n);
	return dp[k][n];
}
int main()
{
     int i, j;
   
    int T;
    scanf("%d", &T);
    while(T--)
    {
    	int N;
    	scanf("%d", &N);
    	printf("%lld\n", fun(2, N));
    }
    return 0;
}
