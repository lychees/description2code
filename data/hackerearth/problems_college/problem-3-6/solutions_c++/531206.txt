#include <iostream>
using namespace std;

int main()
{
    int n,x=0,y=0,i=0;
    cin>>n;
    int home[n],away[n];
    while(i<n)
    {
      cin>>home[i];
      cin>>away[i];
      i++;
    }
    
    i=0;
    int j=0;
    while(i<n)
    {  x=n-1;
       y=0;
       j=0;
    	while(j<n)
       {
    	if(away[i]==home[j])
    	x++;
    	else
    	y++;
        j++;
       }
       y--;
       cout<<x<<" "<<y;
       cout<<endl;
       i++;
    }
    
    return 0;
}
