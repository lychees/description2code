#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
     ll n,i,a,b,j;
     double m,s;
	 cin>>n;
	 vector< pair < ll,ll > > v;
	 for(i=0;i<n;i++)
	 {
	     cin>>a>>b;
		 v.push_back(pair<ll, ll> (a,b));
	 }
	 ll cnt=0,prev;
	 for(i=0;i<n;i++)
	 {
	     if(i+1<n)
	     cnt++;
	     m=(double)(v[i+1].second-v[i].second)/(v[i+1].first-v[i].first);
	     //cout<<i+1<<" "<<i+2<<endl;
	     for(j=i+2;j<n;j++)
		 {
		     
			     s=(double)(v[j].second-v[i].second)/(v[j].first-v[i].first);
			     if(m<=s)
				 {
				     cnt++;
					 m=s;
					 //cout<<i+1<<" "<<j+1<<endl;
				 }
		 }
		 
	 }
	 cout<<cnt;
	 return 0;
}
