#include <iostream>
#include <vector>
#include <climits>
#include <utility>
#include <set>
using namespace std;
set<pair<pair<int,int>,int> > x;
set<pair<pair<int,int>,int> > :: iterator it;
int main()
{
    int t; cin>>t;
    while(t--){
		x.clear();
    	int n,m,a,b,c,d[101],flag=0;
    	cin>>n>>m;
    	for(int i=0;i<=n;i++) d[i]=INT_MAX;
    	for(int i=0;i<m;i++){
    		cin>>a>>b>>c;
    		x.insert(make_pair(make_pair(a,b),-c));
    	}
    	d[1]=0;
    	for(int i=1;i<n;i++){
    		for(it=x.begin();it!=x.end();it++){
    		int u = (*it).first.first;
            int v = (*it).first.second;
            int weight = (*it).second;
            if (d[u] != INT_MAX && d[u] + weight < d[v])
                d[v] = d[u] + weight;
    		}
    	}
    	
    	for(it=x.begin();it!=x.end();it++){
    		int u = (*it).first.first;
            int v = (*it).first.second;
            int weight = (*it).second;
            if (d[u] != INT_MAX && d[u] + weight < d[v])
                flag=1;
    	}
    	if(flag) cout<<"Yes"<<endl;
    	else cout<<"No"<<endl;
    }
    return 0;
}
