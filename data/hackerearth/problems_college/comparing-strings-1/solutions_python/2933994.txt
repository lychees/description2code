def compare(s1,s2):
    flag = False
    for i in s1 :
        for j in s2 :
            if j==i:
                flag = True
                break
    if not flag :
        print("NO")
    else :
        print("YES")

tc= int(input())

while tc:
    s1 = raw_input()
    s2 = raw_input()
    if len(s1)<len(s2):
        compare(s1,s2)
    else :
        compare(s2,s1)
    tc -= 1
