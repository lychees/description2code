#include <iostream>
using namespace std;

int main()
{
    int n,x;
    cin>>n>>x;
    int q,solve=0,flag=0;
    for(int i=0;i<n;i++)
    {
    	cin>>q;
    	if(flag==0)
    	{
    		if(q>x)
    		flag++;
    		else
    		solve++;
    	}
    	else if(flag==1)
    	{
    		if(q<=x)
    		solve++;
    		else 
    		flag++;
    	}
    	else
    		continue;
    }
    cout<<solve<<endl;
    return 0;
}
