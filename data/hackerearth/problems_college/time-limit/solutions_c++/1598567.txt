LLI min(LLI A[],LLI n){
	LLI m = A[0];
	FOR(i,0,n){
		if(A[i]<m)m = A[i];
	}
	return m;
}


LLI max(LLI A[],LLI n){
	LLI m = A[0];
	FOR(i,0,n){
		if(A[i]>m)m = A[i];
	}
	return m;
}

LLI l,u,n,m,A[MAX],B[MAX];
