#include <iostream>
using namespace std;

int squares(int a, int b)
{
	if(a==0)
		return b;
	if(a/10==1)
		return (squares((1+(a%10)*(a%10)+b*b)/10, (1+(a%10)*(a%10)+b*b)%10));
	else
		return (squares((a*a+b*b)/10, (a*a+b*b)%10));
}	
int main()
{
	int t;
	cin>>t;
	int a[t], i;
	for(i=0;i<t;i++)
		cin>>a[i];
	for(i=0;i<t;i++)
	{
		if(squares((a[i]/10), (a[i]%10))==1)
			cout<<"Case #"<<i+1<<":YES"<<"\n";
		else
			cout<<"Case #"<<i+1<<":NO"<<"\n";
	}		
    return 0;
}
