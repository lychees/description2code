from collections import Counter
n = input()
array = [int(i) for i in raw_input().split()]
prod = 1
for val in array:
    prod = prod*val


def primes(n):
    primfac = []
    d = 2
    while d*d <= n:
        while (n % d) == 0:
            primfac.append(d)  # supposing you want multiple factors repeated
            n //= d
        d += 1
    if n > 1:
        primfac.append(n)
    return primfac


temp = Counter(primes(prod))
value = 1

for val in temp.values():
    value = value*(val+1)

print value % (10**9+7)
