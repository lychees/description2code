#include<cstdio>
#include<iostream>


using namespace std;

bool sunday[3001][12][32];

int days[]={31,28,31,30,31,30,31,31,30,31,30,31};
int leap_days[]={31,29,31,30,31,30,31,31,30,31,30,31};


void precompute(){
	sunday[0][0][2]=true;
	int last = 2;
	bool leap = false;
	for(int i =0; i <= 3000;i++){
		leap = false;
		if((i+2000)%4==0){
			if((i+2000)%100==0 and (i+2000)%400!=0)
				leap=false;
			else leap = true;
		}
		
		for(int j = 0; j < 12; j++){
			for(int k =last ; k<=(leap?leap_days[j]:days[j]);k+=7){
				sunday[i][j][k]=true;
				if(k+7 > (leap?leap_days[j]:days[j])) last = k+7-(leap?leap_days[j]:days[j]);
				//cout << i  << ' '<< j << ' '<< k << endl;
			}
		}
	}
}

int main(){
	precompute();
	int t,m,y;
	scanf("%d",&t);
	while(t--){
		scanf("%d%d",&m,&y);
		for(int i =1;i<=31;i++){
			if(sunday[y-2000][m-1][i])
				printf("%d ",i);
		}
		printf("\n");
	}
}
