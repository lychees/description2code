/*input
Wish you were here
*/

#include <bits/stdc++.h>
using namespace std;

int main()
{
	string str;
	getline(cin, str);
	bool isSpace = false;
	str[0] = toupper(str[0]);
	for (int i = 1; i < str.length(); i++) {
		if (str[i] == ' ') {
			isSpace = true;
		}
		if (isSpace && str[i] != ' ') {
			str[i] = toupper(str[i]);
			isSpace = false;
		}
	}
	cout << str << endl;
    return 0;
}
