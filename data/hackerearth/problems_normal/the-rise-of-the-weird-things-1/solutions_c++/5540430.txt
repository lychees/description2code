#include <iostream>
using namespace std;

int main()
{
    int N ;
    cin >> N ;
    int a[1000] ;
    int ct[1001] ;
    for(int i=0 ; i<N ; i++)
    	cin >> a[i] ;
    for(int i=0 ; i<1001 ; i++)
        ct[i]=0 ;
    for(int i=0 ; i<N ; i++)
       ct[a[i]]++ ;
    int sum=0 ;
    for(int i=2 ; i<=1000 ; i+=2){
    	sum+=ct[i]*i ;
    	for(int j=0 ; j<ct[i] ; j++)
    		cout << i << " " ;
    }
    cout << sum << " " ;
    sum=0 ;
    for(int i=1 ; i<=1000 ; i+=2){
    	sum+=ct[i]*i ;
    	for(int j=0 ; j<ct[i] ; j++)
    		cout << i << " " ;
    }
    cout << sum << " " ;
}
