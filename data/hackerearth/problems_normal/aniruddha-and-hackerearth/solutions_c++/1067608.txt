using namespace std;
#define MAX 1000001
#define MOD 1000000007
long long int dp[MAX+1][2];  //h e
long long int sum[MAX+1];
int main(){
	int t;
	scanf("%d", &t);
	dp[0][1] = 1;
	dp[0][0] = 1;
	sum[0] = 2;
	for (int i = 0; i < MAX; i++){
		dp[i + 1][0] = dp[i][0] + dp[i][1];
		dp[i + 1][0] %= MOD;
		dp[i + 1][1] = dp[i][0];
		dp[i + 1][1] %= MOD;
		sum[i + 1] += dp[i + 1][0] + dp[i + 1][1]+sum[i];
		sum[i + 1] %= MOD;
	}
	while (t--){
		int n;
		scanf("%d", &n);
		printf("%lld\n", sum[n-1]);
	}
	return 0;
}
