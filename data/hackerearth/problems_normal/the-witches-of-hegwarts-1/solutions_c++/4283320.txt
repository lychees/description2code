#include<bits/stdc++.h>
using namespace std;
#define ll long long
set<ll > s;
void bfs(ll num)
{
    int cnt=0;
    //q.erase();
    queue<ll > q;
     s.erase(s.begin(),s.end());
    q.push(num);
    s.insert(num);
    q.push(-1);
    while(!q.empty())
    {
        ll int x=q.front();
        if(x==1)
            break;
        q.pop();
        if(x==-1)
        {
            if(q.size()==0)
                break;
            cnt++;
            q.push(-1);
        }
        else
        {
            s.insert(x);
            if(x-1>=1&&(s.find(x-1)==s.end()))
            {
                q.push(x-1);
                s.insert(x-1);
            }
               if(x%2==0&&(s.find(x/2)==s.end()))
               {
                q.push(x/2);
                s.insert(x/2);
               }
               if(x%3==0&&(s.find(x/3)==s.end()))
               {
                q.push(x/3);
                s.insert(x/3);
               }
        }
    }
    cout<<cnt<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        ll n;
        cin>>n;
        bfs(n);
    }


}
