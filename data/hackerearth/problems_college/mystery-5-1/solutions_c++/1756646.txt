#include <iostream>
#include <string>
using namespace std;

int main()
{
	long t;
	cin>>t;
	string s;
	while(t--){
		cin>>s;
		int a[26]={0};
		for(int i=0;i<s.length();i++){
			a[s[i]-'a']+=1;
		}
		int ans=0;
		for(int i=0;i<26;i++){
			ans=max(ans,a[i]);
		}
		
		cout<<ans<<endl;
	}
	return 0;
}
