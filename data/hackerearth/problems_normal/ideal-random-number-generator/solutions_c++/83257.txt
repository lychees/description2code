#include <iostream>
#include <algorithm>
#include <string>
#include <sstream>
using namespace std;
int main() {
	long long a,b,c;
	cin>>a>>b>>c;
	long long l=min(b,c-a);
	if(l<0)
	l=0;
	long long u=min(b,c);
	//cout<<l<<" "<<u<<endl;
	long long num=c*(u-l)*2;
	num=num-((u*u)-(l*l))+(2*l*a);
	long long den=2*a*b;
	long long g=__gcd(num,den);
	num=num/g;
	den=den/g;
	cout<<num<<"/"<<den<<endl;
	return 0;
}
