#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	char a[100001];
	scanf("%s",a);
	int k=0,d=0,c[26]={0};
	int x=strlen(a);
	for(int i=0;i<x;i++) c[a[i]-97]++;   
	if(!(x%2)) {
	    for(int i=0;i<26;i++) if((c[i]%2)!=0) {k=1;break;}
	    if(k==1) cout<<"NO"; else cout<<"YES";
	} else {
	    for(int i=0;i<26;i++) if(c[i]%2) d++;
	    if(d==1) cout<<"YES"; else cout<<"NO\n";
	}
	return 0;
}
