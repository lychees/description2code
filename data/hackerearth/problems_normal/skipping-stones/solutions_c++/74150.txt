#include <iostream>
#include <iomanip>
using namespace std;

int sol(){
	int n, L, D;
	cin >> n >> L >> D;
	
	double * p = new double[n+2];
	double * r = new double[n+2];
	int *l = new int[n+2];
	
	l[0] = 0; l[n+1] = D;
	p[0] = 1; p[n+1] = 1;
	r[0] = 1;
	
	for(int i=1;i<=n;i++){
		cin >> p[i];
	}

	for(int i=1;i<=n;i++){
		cin >> l[i];
	}
	
	for(int i=1; i<=n+1; i++){
		double maxp = 0.0;
		for(int k=i-1; k>=0 && ((l[i]-l[k])<=L) ; k--){
			maxp = max( maxp, r[k] );
		}
		r[i] = maxp*p[i]; 
	}
	std::cout.setf( std::ios::fixed, std:: ios::floatfield );
	cout << setprecision(6) << r[n+1] << endl; 
	return 0;
}


int main()
{
    sol();
    return 0;
}
