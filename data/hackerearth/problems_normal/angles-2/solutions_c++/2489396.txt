#include <cstdio>
#include <cstring>

const int MAXN = 365;
const int MOD = 360;
int n, k, q;
int dp[ MAXN ];
int ar[ MAXN ];
int angles[ 10 ];

int main( void )
{
	scanf( "%d%d", &n, &q );
	for( int i = 0; i < n; ++i )
	{
		scanf( "%d", angles+i );
		angles[i] %= MOD;
	}
	dp[0] = 1;
	for( int i = 0; i < n; ++i )
	{
		memset( ar, 0, sizeof( ar ) ); 
		for( int sad = 0; !ar[ sad ]; ( sad += angles[i] ) %= MOD )
			ar[ sad ] = 1;
		for( int j = 0; j < MOD; ++j )
		{
			if( dp[j] == 0 ) continue;
			for( int k = 0; k < MOD; ++k )
			{
				dp[ ( j + k ) % MOD ]       |= ar[ k ];
				dp[ ( j - k + MOD ) % MOD ] |= ar[ k ];
			}
		}
	}
	for( int i = 0; i < q; ++i )
	{
		scanf( "%d", &k );
		k %= MOD;
		puts( dp[ k ] ? "YES" : "NO" );
	}
	// for( int i = 0; i < MOD; ++i )
	//   printf( "%d", dp[i] );
	return 0;
}
