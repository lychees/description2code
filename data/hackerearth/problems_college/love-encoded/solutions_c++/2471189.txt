#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T,i,k;
    cin>>T;
    cin.ignore();
    for(k=1;k<=T;k++)
    {
        int x,a[1000];
        cin>>x;
        cin.ignore();
        string s;
        getline(cin,s);
        int n=s.length();
        for(i=0;i<n;i++)
            {
                int ans=int(s[i])-x;
                if(ans<65)
                {
                    ans=91+int(s[i])-65-x;
                    a[i]=ans;
                }
                else
                    a[i]=ans;

                cout<<char(a[i]);
            }
            cout<<endl;
    }
}
