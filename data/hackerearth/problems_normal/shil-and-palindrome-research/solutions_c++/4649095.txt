int BIT[100005][26]={0};
int l,q;

void update(int st,int bt,int val)
{
	for(;st<=l;st+=(st&-st))
	BIT[st][bt]+=val;
	
}

int query(int ed,int bt)
{
	int sum=0;
	
	for(;ed>0;ed-=(ed&-ed))
	sum=sum+BIT[ed][bt];
	
	return sum;
	
}
