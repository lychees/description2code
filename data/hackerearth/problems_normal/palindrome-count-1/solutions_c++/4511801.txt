#include<bits/stdc++.h>
using namespace std;
int dp[51][51];
int main(){
	string str;
	int i,j;
	cin>>str;
	int len=str.length();
	/*for(i=0;i<len;i++){
		cout<<str[i]<<" ";
	}
	for(i=0;i<len;i++){
		cout<<i<<" ";
	}*/
	for(i=0;i<len;i++){
		for(j=0;j<len;j++){
			dp[i][j]=1;
		}
	}
	int iteration=1;
	i=0;
	j=iteration;
	while(1){
		if(j==len){
			iteration++;
			if(iteration==len){
				break;
			}
			i=0;
			j=iteration;
		}
		if(str[i]==str[j]){
			if(dp[i+1][j-1]==1){
				dp[i][j]=1;
			}
			else{
				dp[i][j]=0;
			}
		}
		else{
			dp[i][j]=0;
		}
		i++;
		j++;
	}
	int count=0;
	for(i=0;i<len;i++){
		for(j=0;j<len;j++){
			if(dp[i][j]==1){
				count++;
			}
		}
	}
	cout<<count-(len*(len-1)/2);
	return 0;
}
