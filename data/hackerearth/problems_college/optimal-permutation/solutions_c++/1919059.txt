#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long int t, n, x, y, ans;
    cin >> t;
    while (t--) {
        cin >> n;
        y = n/2;
        y += (n & 1);
        x = n-y;
        if (n != 1)
            n -= (n & 1);
        ans = (x*(x-1) + y*(y-1) + n-1);
        printf("%lld\n", ans);
    }
    return 0;
}
