#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,q;
	cin>>n>>q;
	string s;
	cin>>s;
	
	int hash[n+1][26];
	for(int i=0;i<26;i++)
		hash[0][i]=0;
	
	for(int i=0;i<n;i++)
	{
		int d=s[i]-'a';
		for(int j=0;j<26;j++)
			hash[i+1][j]=hash[i][j];
		hash[i+1][d]++;
	}
	while(q--)
	{
		int l,r,k;
		cin>>l>>r>>k;
		
		if(k>(r-l+1))
			cout<<"Out of range"<<endl;
		else
		{
			for(int j=0;j<26;j++)
			{
				k=k-(hash[r][j]-hash[l-1][j]);
				if(k<=0)
				{
					char ans=j+'a';
					cout<<ans<<endl;
					break;
				}
			}
		}
	}
	return 0;
}
