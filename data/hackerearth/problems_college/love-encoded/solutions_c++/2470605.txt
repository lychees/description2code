#include <bits/stdc++.h>

#define ll long long
#define pb push_back
#define mp make_pair
#define ld double

using namespace std;

char s[60];
int k, n;

int main() {
#ifdef LOCAL
	freopen("input.txt", "r", stdin);
#endif
	int t;
	scanf("%d", &t);
	while (t--) {
		scanf("%d%s", &k, s);
		n = strlen(s);
		k %= 26;
		for (int i = 0; i < n; ++i) {
			s[i] = s[i] - k;
			if (s[i] < 'A')
				s[i] += 26;
		}
		printf("%s\n", s);
	}
#ifdef LOCAL
	printf("%.3lf s\n", (double) clock() / (double) CLOCKS_PER_SEC);
#endif
}
