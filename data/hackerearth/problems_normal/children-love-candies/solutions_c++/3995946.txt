#include <bits/stdc++.h>
 
using namespace std;
 
int main()
{
    int q;
	long long int n,t,x;
	cin>>q;
	while(q>0)
	{
		
		cin>>n>>t;
		x=n;
		while(t>0)
		{
			
			if((x%2)==0)
			{
				x=x/2;
			}
			else
			{
				x=x-((x+1)/2);
			}
			x=x-(x/4);
			t--;
		}
		cout<<x<<endl;
		q--;
			
	}
    return 0;
}
