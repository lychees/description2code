#include<bits/stdc++.h>
using namespace std;
long long int pos[1234567];
long long int dish[1234567];
long long int n,m;
bool check(long long int v){
	long long int i=0,j=0,c=0;
	while(i!=n&&j!=m){
		if(abs(pos[i]-dish[j])<=v){
			i++;
			j++;
			c++;
		}
		else
			j++;
	}
	if(c==n)
	return true;
	else
		return false;
}
int main(){
	int t;
	cin>>t;
	while(t--){
	long long int i;
	cin>>n>>m;
	for(i=0;i<n;i++)
		cin>>pos[i];
	for(i=0;i<m;i++)
		cin>>dish[i];
		sort(pos,pos+n);
		sort(dish,dish+m);
	long long int h,l,mid;
	h=2000000001;
	l=0;
	while(l<h){
		mid=l+(h-l)/2;
		if(check(mid))
			h=mid;
		else
			l=mid+1;
	}
	cout<<l<<endl;
	}
return 0;
}
