#include <algorithm>
#include <cstdio>
#include <iostream>
using namespace std;
int main() {
  int n;
  double w, x, ans, cup[200000];
  scanf("%d %lf", &n, &w);
  for (int i = 0; i < 2 * n; i++)
    scanf("%lf", &cup[i]);
  sort(cup, cup + 2 * n);
  x = min(cup[0], min(cup[n] / 2.0, w / n / 3.0));
  printf("%lf", x * 3.0 * n);
}
