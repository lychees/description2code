#include<bits/stdc++.h>
#include<stdio.h>
using namespace std;
#define MAX 10000007
int pf[MAX];
void fun()
{
	int i,j,temp;
	for(i=0;i<MAX;i++)
	pf[i]=0;
	for(i=2;i<MAX;i++){
		if(pf[i]==0){
			for(j=i;j<MAX;j+=i){
				temp=j;
				while(temp%i==0){
					temp/=i;
					pf[j]++;
				}
			
			}
		}
	}
}
void f()
{
	int i;
	for(i=1;i<MAX;i++)
	pf[i]=pf[i]+pf[i-1];
}
int main()
{
	int t;
	fun();
	f();
	scanf("%d",&t);
	int n,ans,i;
	while(t--){
		scanf("%d",&n);
		ans=pf[n];
		printf("%d\n",ans);
	}
	return 0;
}
