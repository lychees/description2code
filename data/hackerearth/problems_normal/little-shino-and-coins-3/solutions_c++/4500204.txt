#include<bits/stdc++.h>
#define sf(x) scanf("%lld",&x)
#define sd(x) scanf("%lf",&x)
#define ll long long int
#define mod 2147483647
#define infi LLONG_MAX
#define ninfi LLONG_MIN
using namespace std;
char a[10000];
bool visited[200];
int main()
{
	ll k,cunt,ans;
	cin>>k;
	cin>>a;
	for(ll i=0;a[i]!='\0';i++)
	{
		memset(visited,false,sizeof(visited));
		cunt=0;
		for(ll j=i;a[j]!='\0';j++)
		{
			if(visited[(ll)a[j]]==false)
			{
				cunt++;
				visited[(ll)a[j]]=true;
			}
			if(cunt==k)
			{
				ans++;
			}
			else if(cunt>k)
			{
				break;
			}
		}
	}
	cout<<ans;
}
