#include <iostream>
#include <iomanip>
using namespace std;
typedef int ll;
ll prime[1000002],sum[1000002];
void mult(ll num)
{
	for(ll i=2*num;i<1000002;i+=num)
		prime[i]=0;
}
void sieve()
{
	for(ll i=0;i<1000002;i++)
	{
		sum[i]=0;
		prime[i]=1;
	}
	prime[0]=prime[1]=0;
	for(ll i=2;i<1000002;i++)
	{
		if(prime[i])
			mult(i);
	}
}
int main()
{
	int tc;
	sieve();
	for(ll i=1;i<1000002;i++)
	{
		sum[i]=prime[i];
		sum[i]+=sum[i-1];
	}
	//cout<<sum[46]-sum[10];
	cin>>tc;
	while(tc--)
	{
		ll a,b,tP;
		cin>>a>>b;
		tP=sum[b]-sum[a];
		if(prime[a]) tP++;
		float ans=tP/((b-a)+1.0);
		if(ans==0) cout<<"0\n";
		else
		{
			cout<<fixed;
			cout<<setprecision(6)<<ans<<endl;
		}
	}
   // cout << "Hello World!" << endl;
    return 0;
}
