#include <iostream>
#include <map>
using namespace std;
long long count[100005];
int arr[100005];
std::map<int, long long> mp;
#define min(x,y) (x<y) ? (x):(y)
#define MAX 100000


int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(false);
	for(int i=4;i<=MAX;i+=2)
		arr[i]=2;
	for(int k=3;k*k<=MAX;k+=2)
	{
		if(!arr[k])
		{
			for(int j=k*k;j<=MAX;j+=(2*k))
			{
				if(!arr[j])
				{
					arr[j]=k;
				}
			}
		}
	}
	int n;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=0;i<n;i++)
	{
		int temp=a[i];
		if(temp!=1)
		{
			while(arr[temp])
			{
				count[arr[temp]]++;
				temp=temp/arr[temp];
			}
			count[temp]++;
		}
	}
	int q;
	cin>>q;
	while(q--)
	{
		mp.clear();
		int base;
		cin>>base;
		int temp=base;
		while(arr[temp])
		{
			mp[arr[temp]]++;
			temp=temp/arr[temp];
		}
		mp[temp]++;
		std::map<int, long long> ::iterator it;
		long long mn=1000000000000000000;
		for(it=mp.begin();it!=mp.end();it++)
		{
			int cnt=it->second;
			mn=min(mn,(count[(it->first)]/cnt));
		}
		cout<<mn<<"\n";
	}
	return 0;
}
