t=input()
for tt in range(t):
	cry=0
	a,b=raw_input().split()
	m=max(len(a),len(b))+1
	a='0'*(m-len(a))+a
	b='0'*(m-len(b))+b
	cr=0
	for mm in range(m):
		ad,bd=map(int,[a[mm],b[mm]])
		su=ad+bd+cr
		if su>9:
			cry+=1
			cr=su-9
	if cry==0:
		print "No carry operation"
	elif cry==1:
		print "1 carry operation"		
	else:
		print cry,"carry operations"
