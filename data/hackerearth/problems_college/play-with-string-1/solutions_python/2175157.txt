

y=input()
for i in range(0,y):
    var=1
    x=raw_input()
    length=len(x)
    y=x[::-1]
    for loop in range(1,length):
        
        
        if(abs(ord(x[loop]) - ord(x[loop-1]))==abs(ord(y[loop]) - ord(y[loop-1]))):
            var=var+1
           
    if(var==length):
        print 'Funny'
    else:
        print 'Not Funny'
