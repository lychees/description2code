#include<iostream>

using namespace std;
int arr[1601][1601][3];

int max(int a, int b)
{
    return(a>b)?a:b;
}
int min(int a, int b)
{
    return(a>b)?b:a;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int f,g,s,p,c,b;
        cin>>f>>g>>s>>p>>c>>b;
        int maxn=0;
        for(int i=0;100*i<=f;i++)
        {
            for(int j=0;125*j+100*i<=f&&50*j<=g;j++)
            {
                int h=min((f-100*i-125*j)/50,(g-50*j)/100);
                int z=s*i+p*j+c*h;
                maxn=max(maxn,z);
                
            }
        }
         if(b<maxn)
    {
    cout<<maxn-b<<endl;}
    else cout<<-1<<endl;
    }
    return 0;
}
