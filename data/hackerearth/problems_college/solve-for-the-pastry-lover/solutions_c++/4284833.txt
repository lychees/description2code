#include <bits/stdc++.h>
using namespace std;
vector<long>v[11];
//char str[100002];
int main()
{
	string str;
	long t,n,i,min;
	vector<long> :: iterator it,prev;
	cin >> t;
	while(t--){
		for(i = 0;i <= 10;i++){
			v[i].clear();
		}
		cin >> n;
		cin >> str;
		for(i = 0;i < n;i++){
			v[str[i] - '0'].push_back(i);
		}
		min = 100000000;
		for(i = 1;i < 10;i++){
			if(v[i].empty()) continue;
			it = v[i].begin();
			prev = it;
			it++;
			for(;it != v[i].end();it++){
				if((*it - *prev) < min){
					min = (*it - *prev);
				}
				prev = it;
			}
		}
		if(min != 100000000){
			cout << min << endl;
		}
		else{
			cout << "-1\n";
		}
	}
    //cout << "Hello World!" << endl;
    return 0;
}
