def shift(arr):
    arr.append(arr[0])
    return arr[1:]


def execute(arr):
    return arr[1:]


def run(a, b):
    steps = 0
    while len(a) > 0:
        if a[0] != b[0]:
            a = shift(a)
            steps += 1
            continue
        else:
            a = execute(a)
            b = execute(b)
            steps += 1

    print steps
