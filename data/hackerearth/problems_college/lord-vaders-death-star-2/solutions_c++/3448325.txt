#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    long long int a[n+1];
    for(int i =0; i < n; i++){
    	cin>>a[i];
    }
    sort(a,a+n,greater<int>());
    if(n == 1)
    {
    	cout<<a[0]<<endl;
    	return 0;
    }
   
    if(n == 2){
    	if(a[0] == a[1]){
    		cout<<"-1"<<endl;
    	}
    	else{
    		cout<<a[0]<<endl;
    	}
    	return 0;
    }
    if(a[0] != a[1]){
    	cout<<a[0]<<endl;
    	return 0;
    }
    for(int i = 1; i < n; i++){
    	if(abs(a[i]-a[i-1]) != 0){
    		if(i+1 < n){
    			if(abs(a[i]-a[i+1])!=0){
    				cout<<a[i]<<endl;
    				return 0;
    			}
    		}
    		else{
    				cout<<a[i]<<endl;
    				return 0;
    		}
    		
    	}
    
    }
    cout<<"-1"<<endl;
}
