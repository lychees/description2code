#include <iostream>
using namespace std;

int main()
{
    int n,q;
    cin>>n>>q;
    long long  a[n],b[n],suma[n],sumb[n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    	suma[i]=0;
    	sumb[i]=0;
    }
    for(int i=0;i<n;i++)
    {
    	cin>>b[i];
    }
    suma[0]=a[0];
    sumb[0]=b[0];
    for(int i=1;i<n;i++)
    {
    	if(i%2==0)
    		suma[i]=a[i]+suma[i-1];
    	else
    		suma[i]=b[i]+suma[i-1];
    }
    for(int i=1;i<n;i++)
    {
    	if(i%2!=0)
    		sumb[i]=a[i]+sumb[i-1];
    	else
    		sumb[i]=b[i]+sumb[i-1];
    }

    
    int x,y,z;
    for(int i=0;i<q;i++)
    {
    	cin>>x>>y>>z;
    	--y;
    	--z;
    	if(x==1 && y%2==0)
    	{
    		if(y-1<0)
    		cout<<suma[z]<<endl;
    		else
    		cout<<suma[z]-suma[y-1]<<endl;
    	}
    	else if(x==1 && y%2!=0)
    	{
    		if(y-1<0)
    		cout<<sumb[z]<<endl;
    		else
    		cout<<sumb[z]-sumb[y-1]<<endl;
    	}
    	else if(x==2 && y%2==0)
    	{
    		if(y-1<0)
    		cout<<sumb[z]<<endl;
    		else
    		cout<<sumb[z]-sumb[y-1]<<endl;
    	}
    	else
    	{
    		if(y-1<0)
    		cout<<suma[z]<<endl;
    		else
    		cout<<suma[z]-suma[y-1]<<endl;    		
    	}
    	
    }
    
    return 0;
}
