#include <bits/stdc++.h>
using namespace std;
int ar[10];

long long basechange(long long n)
{
	long long value=0,p =1;
	
	if(n==0)
		return 0;
		
	while(n)
	{
		long long x=n%10;
		if(x==0)
			n-=10;

		value+=((long long)ar[x]*p);
		p*=9;
		n/=10;
	}
	
	return value;
}

int main()
{
    long long t;
    cin >> t;
    ar[0]=9;ar[1]=1;ar[2]=2;
    ar[3]=2;ar[4]=3;
    ar[5]=4;ar[6]=5;ar[7]=6;
    ar[8]=7;ar[9]=8;
    
    while(t--)
    {
    	long long n;
    	cin >> n;
    	cout << basechange(n) << endl;
    }
}
