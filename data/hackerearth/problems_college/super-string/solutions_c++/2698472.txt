#include <bits/stdc++.h>

using namespace std;

int main()
{
   	int n;
   	cin >> n;
   	
   	while (n--)
   	{
   		char ch[27];
   		int in[27];
   		char str[1000];
   		int cou[27];
   		int flag = 0;
  	
  		int k = 90;
    	for (int i = 1; i <= 26; i++)
    	{		
  			char c = k;
    		ch[i] = c;
    		in[i] = i;
    		k--;
    	}	
    
    	for (int i = 1; i <= 26; i++)
    		cou[i] = 0;
    	
		cin >> str;

		for (int i = 0; i < strlen(str); i++)
		{
			for (int j = 1; j <=26; j++)
			{
//				cout << str[i] << " " << ch[j] << endl;
				if (str[i] == ch[j])
				{
					cou[j]++;
					break;
				}
			}
		}
		
/*		for (int i = 1; i < 26; i++)
		{
			cout << ch[i] << " " << cou[i] << endl; 
		}
*/		
		for (int i = 1; i <= 26; i++) 
		{	
			if (cou[i] != 0)
			{
				if (cou[i] != in[i])
				{
					cout << "No" << endl;
					flag = 1;
					break;
				}
			}
		}
		if (flag != 1)
			cout << "Yes" << endl;
   	}
    return 0;
}
