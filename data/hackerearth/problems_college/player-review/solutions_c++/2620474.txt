#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,a[1000000],max_so=0,max_end=0,i;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    for(i=0;i<n;i++)
    {
    max_end+=a[i];
    if(max_end<0)
    max_end=0;
    else if(max_so<max_end)
    {
    	max_so=max_end;
    }
    }
    cout<<max_so<<endl;
    return 0;
}
