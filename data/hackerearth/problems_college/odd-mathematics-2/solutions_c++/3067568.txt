#include <iostream>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int x,sum=0;
    	cin>>x;
    	while(x)
    	{
    		long long int n=(x+1)/2;
    		n=n*n;
    		sum=sum+n;
    		x/=2;
    		
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
