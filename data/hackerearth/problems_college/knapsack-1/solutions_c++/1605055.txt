#include <iostream>
using namespace std;
    
int max(int a,int b,int c,int d)
{
	if(a>=b && a>=c && a>=d)
	return a;
	if(b>=a && b>=c && b>=d)
	return b;
	if(c>=a && c>=b && c>=d)return c;
    return d;
}
int main()
{
	
    int t;
    cin>>t;
    while(t--)
    {


	int n,k;
	cin>>n>>k;
	int arr[n];
	int ans[k+1];
	ans[0]=0;
	for(int i=0;i<n;i++)
	cin>>arr[i];
	int m;
	for(int i=1;i<=k;i++)
	{
		m=ans[i-1];
		for(int j=0;j<n;j++)
		{
			if(arr[j]<=i)
			{
				m=max(((arr[j]+ans[i-1])<=i?arr[j]+ans[i-1]:-1),m,arr[j],arr[j]*(i/arr[j])<=i?arr[j]*(i/arr[j]):-1);
				
			}
			
		}
		ans[i]=m;
	}
		//for(int i=0;i<=k;i++)
		cout<<ans[k]<<endl;
	
    }
    return 0;
}
