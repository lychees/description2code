#include <iostream>
using namespace std;

int main()
{
	int t;
    cin>>t;
    long int a,b,x,y,c,s;
    while(t--){
    	cin>>a;
    	cin>>b;
    	cin>>c;
    	
    	s=c;
    	x=s;
    	y=s+1;
    	int state=0;
    	while(x>1 && x<=y){
    		y=s-1;
    		if(state==0){
    		if(s%b==0) x=s/b;
    		else x=s/b+1;
    		}
    		else{
    			
    		if(s%a==0) x=s/a;
    		else x=s/a+1;	
    		}
    		s=x;
    		state=(state+1)%2;
    	}
    	
    	if(state==0) cout<<"Bob"<<endl;
    	else cout<<"Adam"<<endl;
    }
    return 0;
}
