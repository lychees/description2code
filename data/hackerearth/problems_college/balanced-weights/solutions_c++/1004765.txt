#include <iostream>
#include<cstdio>

using namespace std;
#define ll long long

ll gcd(ll a,ll b){
	if(b==0)
		return a;
	return gcd(b,a%b);
}


int main()
{
    int t,n;
    ll ans,a;
    scanf("%d",&t);
    while(t--){
    	scanf("%d",&n);
    	scanf("%lld",&ans);
    	for(int i=1;i<n;i++){
    		scanf("%lld",&a);
    		ans = gcd(ans,a);
    	}
    	printf("%lld\n",ans*n);
    }
    return 0;
}
