#include <bits/stdc++.h>
using namespace std;
const int MAXN = 200005; 
int N; 
bitset <MAXN> ans, L, LA;
int arr [MAXN]; 
int main(){
	ios_base::sync_with_stdio(0); 
	cin >> N; for (int g=1; g<=N; g++) cin >> arr[g]; 
	sort (arr+1, arr+N+1); 
	for (int g=N-1; g>=1; g--){
		if (arr[g] == arr[g+1]) continue; 
		L = (LA << (arr[g+1] - arr[g])); 
		L[arr[g+1]-arr[g]] = 1;
		LA = L; 
		ans|=L;
	}
	int xx = 0; 
	for (int g=1; g<=N-1; g++) xx = max(xx, (int)(arr[g] == arr[g+1])); 
	cout << ans.count() + xx; 
	return 0; 
}
