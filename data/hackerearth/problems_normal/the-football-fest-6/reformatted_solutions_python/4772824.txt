T = int(raw_input().strip())
for t in range(T):
    num_passes, current_player = map(int, raw_input().strip().split(" "))
    back_to = current_player
    for p in range(num_passes):
        cur_pass = raw_input().strip().split(" ")
        if len(cur_pass) == 2:
            back_to = current_player
            current_player = int(cur_pass[1])
        else:
            temp = current_player
            current_player = back_to
            back_to = temp
    print "Player", current_player
