#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int T,i,len;
char S[105];
char TE[27]="qwertyuiopasdfghjklzxcvbnm";
//char T[26]="QWERTYUIOPASDFGHJKLZXCVBNM";
main()
{
	cin>>T;
	cin.ignore();
	while(T--)
	{
		
		gets(S);
		len=strlen(S);
		for(i=0;i<len;i++)
		{
			if(S[i]==' ')
			{
				cout<<" ";
			}
			else
			cout<<TE[S[i]-'a'];
		}
		cout<<"\n";
	}	
}
