#include <bits/stdc++.h>

using namespace std;

int n;
char a[1<<17];
int cnt[2048];
long long ans;
int parity[16];

int main() {
	int i,j,current_mask;
	
	scanf("%d", &n);
	scanf("%s", a+1);
	
	++cnt[0];
	for(i=1;i<=n;i++) {
		parity[a[i]-'0']^=1;
		current_mask=0;
		for(j=0;j<10;j++) if(parity[j]) current_mask|=(1<<j);
		ans+=cnt[current_mask];
		for(j=0;j<10;j++) ans+=cnt[current_mask^(1<<j)];
		++cnt[current_mask];
	}
	
	printf("%lld\n", ans);
	
	return 0;
}
