#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int t[n],d[n],a[n],b[n],c[n];
    for(int i=0;i<n;i++)
    {
    	cin>>t[i]>>d[i];
    	a[i]=t[i]+d[i];
    	b[i]=a[i];
    }
    sort(b,b+n);
    for(int j=0;j<n;j++)
    {
    	for(int k=0;k<n;k++)
    	{
    		if(b[j]==a[k])
    		{
    			c[j]=k+1;
    			a[k]=-1;
    			break;
    		}
    	}
    }
    for(int l=0;l<n;l++) cout<<c[l]<<" ";
    return 0;
}
