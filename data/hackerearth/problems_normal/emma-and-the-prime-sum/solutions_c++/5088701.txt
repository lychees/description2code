  for (i=2; i <1000; i++)

      for (j= 2*i; j<1000001; j= j+i)
      {
         primes[j]=0;
      }
     }
  }

   primes[1]=0;
   ll a=0;
   for (i = 0; i <1000001; i++)
   {
      a= a+ primes[i];
      psum[i]=a;
   }
} 
   
int main()
