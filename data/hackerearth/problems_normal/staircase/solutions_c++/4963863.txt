#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int inp;
    cin>>inp;
    if(inp<0){
    	cout<<"ERROR"<<endl;
    	return 0;
    }
    long long int jack[inp],jill[inp];
    jack[0] = 1;
    jack[1] = 2;
    jack[2] = 4;
    jill[0] = 1;
    jill[1] = 2;
    jill[2] = 3;
    for(int i=3;i<inp;i++){
    	jack[i] = jack[i-1] + jack[i-2] + jack[i-3];
    	jill[i] = jill[i-1] + jill[i-2];
    }
    cout<<"Jack-"<<jack[inp-1]<<endl;
    cout<<"Jill-"<<jill[inp-1]<<endl;
    return 0;
}
