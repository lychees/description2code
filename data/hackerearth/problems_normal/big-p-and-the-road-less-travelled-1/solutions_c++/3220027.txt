vector<int> adj[N];
bool visited[N];
int n;
ll dp[N];
ll dfs(int s)
{
	ll ans=0;
	if(s==n)
	 return 1;
	else if(visited[s])
	  return dp[s];
	visited[s]=true;  
	for(int i=0;i<adj[s].size();i++)
	{
		ans+=dfs(adj[s][i]);
	}
	dp[s]=ans;
	return dp[s]; 
}
