int main()
{
    int t;
    cin >> t;
    while (t--)
    {
        int n;
        cin >> n;
        if (n == 1)
        {
            cout << 1 << endl;
            continue;
        }
        int ans;
        int cur = 0, sum = 0;
        vector<int> b6e5;
        for (int i = 1;; i++)
        {
            int x = (i + 1) * i / 2 + ((i * (i - 1)) / 2);
            if (x > n)
                break;
            b6e5.pb(i);
            ans = i + (i - 1);
            cur = x;
        }
        int x = n - cur;
        while (x)
        {
            while (b6e5.back() > n)
                b6e5.pop_back();
            int temp = x / b6e5.back();
            ans += temp;
            x -= temp * b6e5.back();
            b6e5.pop_back();
        }
        cout << ans << endl;
    }
}
