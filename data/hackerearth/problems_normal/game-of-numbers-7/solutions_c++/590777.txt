using namespace std;
#include<iostream>
#include<math.h>
int main()
{
	int t,x,y;
	cin>>t;
	while(t--)
	{
		cin>>x>>y;
		if(x > y)
		{
			int temp = x;
			x = y;
			y = temp;
		}
		int size=y-x+1;
		int a[1000001]={0},sqr=0,ans=0;
		for(int i=2;i*i<=y;i++)
		{
			sqr = i*i;
			int j = x/sqr;
			if(x%sqr != 0)
				j++;
			while(sqr*j <= y)
			{
				a[sqr*j-x] = 1;
				j++;
			}
		}
		
		for(int i=0;i<=y-x;i++)
			if(a[i] == 0)
				ans++;
		cout<<ans<<endl;
	}
	return 0;	
}
