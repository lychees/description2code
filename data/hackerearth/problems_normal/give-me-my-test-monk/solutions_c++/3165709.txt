#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
typedef long long ll;
ll derange[101];
ll c[101][101];
void prec(int n)
{
	int i,j;
	derange[0]=1;
	derange[1]=0;
	for(i=2;i<=n;i++)
	{
		derange[i]=((i-1)*(derange[i-1]+derange[i-2]))%mod;
	}
	for(i=0;i<=n;i++)
	{
		for(j=0;j<=i;j++)
		{
			if(i==j || j==0)
			c[i][j]=1;
			else
			c[i][j]=(c[i-1][j] + c[i-1][j-1])%mod;
		}
	}
}
ll pw( ll a, ll b, ll m )
{
	ll r =1;
	while(b)
	{
		if(b&1)
		r=(r*a)%m;
		b>>=1;
		a=(a*a)%m;
	}
	return r;
}
int main()
{
	int n ,i;
	cin >> n;
	int l,r;
	cin >> l >> r;
	prec(n);
	ll p=0,q=0;
	for(i=0;i<=n;i++)
	{
		ll ways=(derange[i]*c[n][i])%mod;
		if(l<=i && i<=r)
		{
			p+=ways;
			p%=mod;
			
		}
		q+=ways;
		q%=mod;
	}
	ll ans=(p*pw(q,mod-2,mod))%mod;
	cout << ans << endl;
	return 0;
	
}
