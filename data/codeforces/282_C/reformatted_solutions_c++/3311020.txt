#include <algorithm>
#include <iostream>
#include <string>

using namespace std;

int main() {
  string a, b;

  cin >> a >> b;
  cout << (a.length() != b.length() ||
                   a != b && (min(a, b) == string(a.length(), '0'))
               ? "NO"
               : "YES")
       << endl;

  return 0;
}
