#include <iostream>
using namespace std;

int foo(int size,int* a, int key){
	int sum=0;
	for(int i=0;i<size;i++){
		sum+=a[i];
		if(sum>=key)
			return (i+1);
	}
	return -1;
}

int main()
{
    // sum upto that transaction is daily limit
    int t,q,target;
    cin >> t;
    int a[t];
    for(int j=0;j<t;j++){
    	cin >> a[j];
    }
    cin >> q;
    for(int i=0;i<q;i++){
    	cin >> target;
    	cout << foo(t,a,target) << endl;
    }
    return 0;
}
