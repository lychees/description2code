#include <iostream>
#include <queue>
#include <iostream>
using namespace std;
 
 
int main() {
	/* Enter your code here. Read input from STDIN. Print output to STDOUT */
	int M, N; cin >> M >> N;
	priority_queue<int> q;
	for (int i = 0; i < M; i++)
	{
		int tmp; cin >> tmp;
		q.push(tmp);
	}
	int total = 0;
	while (N--)
	{
		int tmp = q.top();
		total += tmp;
		q.pop();
		q.push(tmp - 1);
	}
	cout << total << endl;
	return 0;
}
