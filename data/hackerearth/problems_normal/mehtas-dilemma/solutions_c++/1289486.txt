ull mod =1000000007;
typedef vector< vector<ull> > mat;
mat M0[100];

mat prod(mat A, mat B) {
	mat C;
	C.resize(16,vector<ull>(16,0));
	for(int i =0; i < 16; i++) for(int j =0; j < 16; j++) 
		for(int k =0; k < 16; k++) C[i][j] =(C[i][j]+A[i][k]*B[k][j])%mod;
	return C;}

mat sum(mat A, mat B) {
	mat C;
	C.resize(16,vector<ull>(16,0));
	for(int i =0; i < 16; i++) for(int j =0; j < 16; j++) 
		C[i][j] =(A[i][j]+B[i][j])%mod;
	return C;}

ull pw(ull a, long long e) {
	if(e <= 0) return 1;
	ull x =pw(a,e/2);
	x =(x*x)%mod;
	if(e%2) x =(x*a)%mod;
	return x;}

mat pwM(long long e) {
	if(e <= 0) return M0[0];
	mat B =pwM(e/2);
	mat C =prod(B,B);
	if(e%2 == 0) return C;
	return prod(C,M0[1]);}

mat sumM(long long e) {
	if(e <= 0) return M0[0];
	if(e%2 == 0) return sum(M0[0],prod(M0[1],sumM(e-1)));
	mat A =sum(M0[0],pwM(e/2+1));
	mat B =sumM(e/2);
	return prod(A,B);}
