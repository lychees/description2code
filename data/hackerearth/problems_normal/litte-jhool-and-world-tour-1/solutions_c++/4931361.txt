#include <iostream>
#include <utility>
#include <algorithm>
#include <set>
using namespace std;
typedef pair <int,int> p;
#define mp(l,r) make_pair((l),(r)) 
#define pb(x) push_back((x))
#define endl "\n"
int main()
    {
        int t;
        cin>>t;
    	while (t--)
        {
            int m, n;
            cin>>m>>n;
    	    vector <p> rng;
            for(int i=0;i<n;i++)
            {
                int l, r;
                cin>>l>>r;
                if(l<=r)
                {
                    rng.pb(mp(l,r));
                    rng.pb(mp(l+m,r+m));
                }
                else
                    rng.pb(mp(l,r+m));
            }
            sort(rng.begin(),rng.end());
    		int T=0;
            int i=0;
           // set < p > s;
           	  set<int> s;
            bool flag = true;
            while(true)
            {
                if(s.empty())
                {
                    if(i==rng.size())
                        break;
                    else
                        T = rng[i].first;
                }
    			while(i<rng.size() && rng[i].first==T)
                {
                   // s.insert(mp(rng[i].second,i));
                    s.insert(i);
                    i++;
                }
    			//int ind = s.begin()->second;
    			int ind=*s.begin();
                s.erase(s.begin());
    			if(!(T>=rng[ind].first && T<=rng[ind].second))
                {
                   flag = false;
                    break;
                }
                T++;
            }
            if(flag)
            	 cout<<"YES"<<endl;
            else 
            	 cout<<"NO"<<endl;
        }
    }
