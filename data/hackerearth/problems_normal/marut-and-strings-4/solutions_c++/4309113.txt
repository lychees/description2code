#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;cin>>t;
    if(!(1<=t&&t<=10))		{cout<<"Invalid Test"<<endl;return 0;}
    while(t--){
    	string s;cin>>s;
    	if(!(1<=s.length()&&s.length()<=100))	{
    		cout<<"Invalid Input"<<endl;
    		continue;
    	}
    	int i=0,l=0,u=0;
    	for(i=0;s[i];i++){
    		if(islower(s[i]))	l++;
    		if(isupper(s[i]))	u++;
	   	}
	   	if(l==0&&u==0)	cout<<"Invalid Input"<<endl;
	   	else			cout<<min(l,u)<<endl;
    }
    return 0;
}
