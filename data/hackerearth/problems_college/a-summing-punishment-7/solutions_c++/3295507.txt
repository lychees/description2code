#include <bits/stdc++.h>
using namespace std ;
int main()
{
  int n ;
  cin  >> n ;
  int i ;
  int a[n] ;
  for ( i = 0; i< n ; i++)
  {
   cin >> a[i] ;
  }
  int q ;
  cin >> q ;
  while ( q--)
  {
  int l ;
  int r ;
  cin >> l ;
  cin >> r ;
    int sum = 0;
    for ( i = l-1 ; i<=r-1 ; i++)
    {
     sum = sum + a[i] ;
    }
    cout << sum <<endl;

  }
  return 0 ;

}
