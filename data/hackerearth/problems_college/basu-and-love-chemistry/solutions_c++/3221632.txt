#include <iostream>

#define M 1000000007
using namespace std;

int main()
{
    string s;
    cin>>s;
    long long int sum=0,sum1=0,i=0,j;
    while(s[i]!='\0')
    {
    j=s[i];
    j-=48;
    sum=sum*10+(i+1)*j;
    sum+=sum1;
    sum1+=(i+1)*j;
    sum%=M;
    sum1%=M;
    
    i++;
    }
    cout<<sum<<endl;
    return 0;
}
