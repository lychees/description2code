#include <iostream>
#include <cstdio>
using namespace std;

int day_month[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

int dp[100005][15][35];

void count()
{
    int count = 0;
    for (int i = 1; i <= 100000; i++) {
        int months = 12;
        for (int j = 1; j <= months; j++) {
            int days = day_month[j];
            if ((i % 4 == 0) && (j == 2)) days++;
            for (int k = 1; k <= days; k++) {
                if ((j + 1 == i % 100) && (k + 1 == j)) {
                    count++;
                }
                dp[i][j][k] = count;
            }
        }
    }
}

int main()
{
    int t;
    scanf("%d", &t);
    count();
    while (t--) {
        int d1, m1, y1;
        int d2, m2, y2;
        scanf("%d:%d:%d", &d1, &m1, &y1);
        scanf("%d:%d:%d", &d2, &m2, &y2);
        int count2 = dp[y2][m2][d2];
        int count1 = dp[y1][m1][d1];
        if (((d1 + 1) == m1) && ((m1 + 1) == y1 % 100)) count1 -= 1;
        printf("%d\n", count2 - count1);
    }
}
