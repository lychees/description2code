#include <iostream>
#include <vector>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    long long int p1,p2,t,a,b,result;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	int rsum[]={0,1,2,6,6,3};
    	result=0;
    	if(a>=6)
    		result=9*(b-a+1);
    	else if(b>=6)
    	{
    		for(p1=a;p1<=5;p1++)
    			result+=rsum[p1];
    		result+=9*(b-6+1);
    	}
    	else
    	{
    		for(p1=a;p1<=b;p1++)
    			result+=rsum[p1];
    	}
    	cout<<result<<endl;
    }
    return 0;
}
