#include <iostream>
#include <map>
using namespace std;

bool is_anagram(string a,string b)
{
	int arr1[26]={0};
	int arr2[26]={0};
	if(a.length()!=b.length()) return false;
	for(int i=0;i<a.length();i++) arr1[a[i]-'a']++;
	for(int i=0;i<a.length();i++) arr2[b[i]-'a']++;
	for(int i=0;i<26;i++) {if(arr1[i]!=arr2[i]) return false;}
	return true;
}


int main()
{
    int n;
    cin>>n;
    map<string,int> ans;
    while(n--)
    {
    	string curr;
    	cin>>curr;
    	bool any = true;
    	for( const auto &p : ans)
    	{
    		if(is_anagram(p.first,curr)) {ans[p.first]++;;any=false;break;}
    	}
    	if(any) ans[curr]=1;
    }
    int max = 0;
    for( const auto &p : ans)
    {
    	if(p.second > max) max = p.second;
    }
    cout<<max<<endl;
}
