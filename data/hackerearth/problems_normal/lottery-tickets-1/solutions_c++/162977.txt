#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<iostream>
#include<vector>


using namespace std;

struct temp {
	int st;
	int et;
	int wp;
};

struct temp s[100];
double a[100];

bool cmp(struct temp a, struct temp b){
	if (a.et==b.et) {
		return (a.st<b.st);
	}
	return a.et<b.et;
}


int main(){
	int t;
	int n;
	
	scanf("%d", &t);
	
	while (t--) {
		scanf("%d", &n);
		
		for (int i=0; i<n; i++) {
			scanf("%d%d%d", &s[i].st, &s[i].et, &s[i].wp);
		}
		
		sort(s, s+n, cmp);
		
		
		double ans= ((double)s[0].wp/(double)100.0);
		a[0]=ans;
		for (int i=1; i<n; i++) {
			int j=i-1;
			double maxi=a[j];
			while (j>=0) {
				if (s[j].et<=s[i].st) {
					break;
				}
				maxi=max(maxi, a[j]);
				j--;
			}
			if (j<0) {
				a[i]=((double)s[i].wp/(double)100.0);
			}
			else {
				a[i]=a[j]+((double)s[i].wp/(double)100.0);
			}
			a[i]=max(maxi, a[i]);

		}
		
		printf("%0.2lf\n", a[n-1]);
		
	}
	
	
	return 0;
}
