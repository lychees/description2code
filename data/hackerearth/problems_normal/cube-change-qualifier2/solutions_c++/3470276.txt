#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long i,j,k,m,n,t;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	k=n-2;
    	m=(n-k)*(n*n+k*k+n*k);
    	if(n==1) cout<<"1\n";
    	else if(n==2) cout<<"7\n";
    	else cout<<m<<endl;
    }
    return 0;
}
