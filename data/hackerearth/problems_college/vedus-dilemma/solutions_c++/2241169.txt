#include<stdio.h>
#include<algorithm>
#include<iostream>
#include<stdlib.h>
using namespace std;
long int a[1000104];
bool prime[1000006];
long long int arr[1000007],temp[1000006];
void sieve()
{
	int i,j,p=0;
	prime[0]=1;
	prime[1]=1;
	for(i=2; i*i<1000006; i++)
	{
		if(prime[i]==0)
		{
			arr[p++]=i;
			j=2;
			while(i*j<1000006)
			{
				prime[i*j]=1;
				++j;
			}
			
		}
	}
	for(j=i; j<1000006; j++)
	{
		if(prime[j]==0)
		arr[p++]=j;
	}
}
struct amit
{
	long long int value,diff;
}answer[1000006];
bool comp(struct amit a,struct amit b)
{
	return (a.diff<b.diff);
}
int main()
{
	sieve();
	long long int t;
	scanf("%lld",&t);
	while(t--)
	{
		long int n,q=0,w=0,p=0,z=0,x=0;
		scanf("%ld",&n);
		long int i,j;
		for(i=0; i<n; i++)
		{
			scanf("%ld",&a[i]);
		}
		scanf("%ld %ld",&p,&q);
		long int save[1006];
		j=0;
		for(; arr[j]<=p; j++)
		{
			//printf("%lld is prime ",arr[j]);
			if(p%arr[j]==0)
			{
			//	printf("%ld is factor ",arr[j]);
				save[w++]=arr[j];
				//p/=arr[j];
			}
			//w++;
		}
		
		
		for(i=0; i<n; i++)
		{
			for(j=0; j<w; j++)
			{
				if(a[i]%save[j]==0)
				{
			//		printf("%ld is divided ",a[i]);
					answer[z].value=a[i];
					answer[z].diff=abs(p-a[i]);
					z++;
					break;
				}
			}
		}
		if(z<q)
		{
			printf("-1\n");
			continue;
		}
		sort(answer,answer+z,comp);
		for(i=q-1; i>=0; i--)
		{
			//printf("%lld ",)
			temp[x++]=answer[i].value;
		}
		sort(temp,temp+x);
		for(i=0; i<q; i++)
		{
			printf("%lld ",temp[i]);
		}
		printf("\n");
	}
	return 0;
}
