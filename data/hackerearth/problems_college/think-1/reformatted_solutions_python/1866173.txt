def isSuchNumber(N):
    sqrN = N * N
    revN = int(str(N)[::-1])
    if N == revN:
        return "No"
    sqrRevN = revN * revN
    revSqrRevN = int(str(sqrRevN)[::-1])
    if sqrN == revSqrRevN:
        return revN
    else:
        return "No"


for i in range(input()):
    print isSuchNumber(input())
