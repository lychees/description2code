#include <bits/stdc++.h>
using namespace std;

int n,t,a[10000],i,ans=0,x;

map<int,int> m;

int main()
{
    cin>>n;
    for(i=0;i<n;i++){
    	cin>>x;
    	m[x]++;
    }
    ans+=m[4];
    ans+=m[2]/2;
    if(m[2]&1){
    	ans++;
    	if(m[1]>1)
    		m[1]-=2;
    	else if(m[1]==1)
    		m[1]-=1;
    }
    ans+=m[3];
    m[1]-=min(m[1],m[3]);
    ans+=ceil(m[1]/4.0);
    cout<<ans<<endl;
    return 0;
}
