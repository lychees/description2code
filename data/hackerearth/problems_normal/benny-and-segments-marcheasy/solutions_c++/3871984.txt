#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin>>t;
    while(t--) {
    	int n,l;
    	cin>>n>>l;
    	vector< pair<int,int> > V(n);
    	for(int i=0;i<n;i++) {
    		cin>>V[i].first>>V[i].second;
    	}
    	bool flag = false;
    	sort(V.begin(),V.end());
    	for(int i=0;i<n;i++) {
    		int target = V[i].first + l;
    		int current = V[i].second;
    		for(int j=i+1;j<n;j++) {
    			if(V[j].first <= target and V[j].first > V[i].first) {
    				if(V[j].second == target)
    					current = max(current,V[j].second);
    			}
    		}
    		if(current == target) {
    			flag = true;
    			break;
    		} 
    	}
    	cout<<(flag ? "Yes" : "No")<<endl;
    }
    return 0;
}
