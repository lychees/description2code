t = input()
while t:
    a = map(str, raw_input("").split())
    base = int(a[0])
    stri = a[1]
    q = 1
    ans = 0
    while len(stri) > 0:
        l = stri[-1]
        if ord(l) < 59:
            l = ord(l)-48
        else:
            l = ord(l)-87
        ans += q*l
        q *= base
        stri = stri[:-1]
    x = str(ans)
    print sum(int(i) for i in x)
    t -= 1
