#include <iostream>
using namespace std;
#define ll long long int


int main()
{
	ll t; cin>>t;
	while(t--){
		ll n,count=0,s=0,r; cin>>n;
		while(n){
	        r=n%10; s+=r;
	        n=n/10;
		}
		
		for(int i=2;i<=s/2;i++){
			if(s%i==0){
                count++;
                break;
            }
        }
        if(count==0 && s!= 1)
		  cout<<"YES"<<"\n";
		else 
		  cout<<"NO"<<"\n";
	} 
    //cout << "Hello World!" << endl;
    return 0;
}
