#include<string>
#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s;
        int c=0;
        cin>>s;
        for(int i=0;i<s.length()-1;i++)
        {
            if(s[i]=='X' && s[i+1]!='Y')
                c++;
            else if(s[i]=='Y' && s[i+1]!='X')
                c++;
        }
        cout<<c<<endl;
    }
    return 0;
}
