#include<iostream>
using namespace std;
int digit(int n)
{
    int sum=0;
    while(n)
    {
        sum+=(n%10);
        n/=10;
    }
    return sum;
}
int main()
{
    int t,n,k;
    cin>>t;
    while(t--)
    {
        cin>>n;
        while(n>=10)
        {
            k=digit(n);
            n=k;
        }
        cout<<k<<"\n";
    }
    return 0;
}
