#include <iostream>
using namespace std;

int rev(int num)
{
	int r = 0;
	while (num != 0)
	{
		short rem = num % 10;
		r = r * 10 + rem;
		num /= 10;
	}
	return r;
}

int main()
{
    short t;
    cin >> t;
    while (t--)
    {
    	int a, b;
    	cin >> a >> b;
    	cout << rev(rev(a) + rev(b)) << "\n";
    }
    
    return 0;
}
