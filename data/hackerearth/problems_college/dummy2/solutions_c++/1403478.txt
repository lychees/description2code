ll modulo(ll a,ll b) {
	ll c = 1e9 + 7;
    long long x=1,y=a;
    while(b > 0){
        if(b%2 == 1){
            x=(x*y)%c;
        }
        y = (y*y)%c;
        b /= 2;
    }
    return x%c;
}

int main() {
	ll n;
	S(n);
	while(n--) {
		string s,s1,ans;
		ll i,j;
		cin >> s;
		ll len = s.length();
		for(i = 0;i < len;i++) {
			if(s[i] >= '0' && s[i] <= '9') {
				ll num = s[i] - '0';
				for(j = 0;j < num;j++) {
					ans = ans + s1;
				}
			//	cout << ans << " " << s1 << "\n";
				s1 = ans;
				ans = "";
			} 
			else {
				s1 = s1 + s[i];
			}
		//	cout <<s1 << "\n";
		}
		cout << s1 << "\n";
	}

}
