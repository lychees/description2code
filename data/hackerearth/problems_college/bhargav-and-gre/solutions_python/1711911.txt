ref_word = str(raw_input())
no_of_inp = int(raw_input())
synonym=0;
antonym=0;
for x in range(0,no_of_inp):
	word = str(raw_input())
	if (word[0]==ref_word[0] and len(word)==len(ref_word)):
		synonym= synonym+1;
	elif((ord(word[0])-3)==(ord(ref_word[0])) and len(word)==len(ref_word)):
		antonym=antonym+1;
print(str(synonym)+" "+str(antonym));
