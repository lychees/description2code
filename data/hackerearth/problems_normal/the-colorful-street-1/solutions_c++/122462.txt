#include <iostream>
using namespace std;

int test,mat[200][200],x,y,z,i,n;

int min(int x,int y)
{
	if(x<y)
		return x;
	return y;
}	

int main()
{
    
    cin>>test;
    while(test--)
    {
    	cin>>n;
    		
    	if(n)
    	{
			cin>>x>>y>>z;
			mat[0][0]=x;
			mat[0][1]=y;
			mat[0][2]=z;
			
			for(i=1;i<n;i++)
			{
				cin>>x>>y>>z;
				
				mat[i][0]=min(mat[i-1][1],mat[i-1][2]) + x;
				mat[i][1]=min(mat[i-1][0],mat[i-1][2]) + y;
				mat[i][2]=min(mat[i-1][0],mat[i-1][1]) + z;
				
			}
			
			cout<<min(mat[n-1][0],min(mat[n-1][1],mat[n-1][2]))<<endl;
    	}
    	
    }
    
    return 0;
}
