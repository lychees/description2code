t = input();
while(t > 0):
	n, b = map(int, raw_input().split()); cp = n; c1 = c2 = 0;
	if(n % b == 0):	print n;
	else:
		n -= 1; cp += 1;
		while(n >= 0 and n % b != 0):	c1 += 1; n -= 1;
		while(cp > n and cp % b != 0):	c2 += 1; cp += 1;
		if(c1 < c2):	print n;
		elif(c2 < c1):	print cp;
		else:	print max(n, cp);
	t -= 1;
