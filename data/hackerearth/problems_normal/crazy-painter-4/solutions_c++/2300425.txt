
#include<stdio.h>
#include<iostream>
#include<math.h>
using namespace std;
int main()
{
	long n,T,res,tot,r,v;
    long A[27];
    scanf("%ld",&T);
    while(T--){
        scanf("%ld",&n);
        for(int i=1;i<27;i++){
            scanf("%ld",&A[i]);
        }
        tot=n*12;
        n=tot/26;
        r=tot%26;
        res=0;
        for(int i=1;i<27;i++){
            v=n;
            if(i<=r){
                v=v+1;
            }
            res=res+(v*(A[i]+A[i]+(v-1))/2);
        }
        printf("%ld\n",res);
    }
    return 0;
}
