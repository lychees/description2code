#include <bits/stdc++.h>

using namespace std;
int main()
{
    long t;
    cin>>t;
    while(t--)
    {
    	long long n;
    	cin>>n;
    	long long s=0;
    	for(long i=1;(i*i)<=n;i++)
    	{
    		if((n%i)==0)
    		{
    			if(i%4!=0)
    				s+=i;
    			if((n/i)!=i && (n/i)%4!=0)
    				s+=(n/i);
    		}
    	}
    	cout<<8*s<<endl;
    }
    return 0;
}
