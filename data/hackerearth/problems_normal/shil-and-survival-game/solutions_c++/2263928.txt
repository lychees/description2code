#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long n;
	scanf("%lld",&n);
	long long a[n];
	long long i;
	for(i=0;i<n;i++)
	scanf("%lld",&a[i]);
	long long maxi[n+1];
	maxi[0]=0;
	maxi[1]=0;
	long long b[n];
	int k=0;
	for(i=2;i<=n;i++)
	{
		maxi[i]=max(maxi[i-1],a[i-2]);
		//printf("%lld ",maxi[i]);
	}
//	printf("\n");
	long long maxj[n+1];
	maxj[n]=0;
	for(i=n-1;i>=1;i--)
	{
		maxj[i]=max(maxj[i+1],a[i]);
//		printf("%lld ",maxj[i]);
	}
//	printf("\n");
	for(i=1;i<=n;i++)
	{
		long long x,y;
		x=max(a[i-1],maxi[i]);
		y=max(a[i-1],maxj[i]);
		if(x==a[i-1]||y==a[i-1])
		b[k++]=i;
	}
	sort(b,b+k);
	for(i=0;i<k;i++)
	printf("%lld ",b[i]);
	return 0;
}
