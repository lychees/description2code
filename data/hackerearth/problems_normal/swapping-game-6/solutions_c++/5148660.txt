#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main()
{
    int k;
    string s;
    cin >> k >> s;
    string scpy = s;
    string temp1, temp2;
    int len = s.length();
    long cycle = 0;
    while (k > 0)
    {
        for (int i = 0; i < len; i += 2)
        {
            temp1 += s[i];
        }
        for (int i = 1; i < len; i += 2)
        {
            temp2 += s[i];
        }
        reverse(temp2.begin(), temp2.end());
        s = temp1 + temp2;
        if (s == scpy)
        {
            cycle++;
            k -= cycle * long (k / cycle);
        }
        k--;
        cycle++;
        temp1 = "";
        temp2 = "";
    }
    cout << s << endl;

    return 0;
}
