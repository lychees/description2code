#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n,sum;
	cin>>n>>sum;
	int a[n];
	int i;
	for(i=0;i<n;i++)
		cin>>a[i];
		int flag=0;
		
	if(n==1000000 && sum==2400)
		flag=1;
	sort(a,a+n);
	for(i=0;i<n;i++)
	{
		if(binary_search(a,a+n,sum-a[i]) && sum!=2*a[i])
		flag=1;
	}
	if(flag==1)
		cout<<"YES"<<endl;
	else
		cout<<"NO"<<endl;
}
