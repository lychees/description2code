#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ll t,i,n;
    vector<ll> dp(1001);
    dp[0]=0;
    dp[1]=1;
    dp[2]=2;
    for(i=3;i<=1000;i++)
    	dp[i]=dp[i-1]+dp[i-2];
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld",&n);
    	printf("%lld\n",dp[n]);
    }
    return 0;
}
