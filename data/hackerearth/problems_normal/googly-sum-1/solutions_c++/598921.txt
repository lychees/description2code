#include <iostream>
using namespace std;

const int MAXN = 5000;
const int A = 200000;

int n, a[MAXN], e[2*A+1];

//this is for search
inline int q(int b)
	{
		return e[b+A];
	}
//this is for update
inline void u(int b)
	{
		e[b+A] = 1;
	}

int main()
{
    scanf("%d", &n);
    for (int i = 0; i < n; ++i)
    	{
    		scanf("%d", a+i);
    	}
    int sol = 0;
    for (int i = 0, flag = 0; i < n; ++i, flag = 0)
    {
    	for(int j = 0;j < i && !flag; ++j) flag = q(a[i]-a[j]);
    	for(int j = 0; j <= i; ++j) u(a[i]+a[j]);
    	sol += flag;
    }
    
    printf("%d\n", sol);
    return 0;
}
