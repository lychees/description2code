T = int(raw_input())
while T:
    T = T-1
    a, b = map(int, raw_input().split())
    count = 0
    while a != b:
        if a == 1:
            count = count+b-1
            break
        elif b == 1:
            count = count+a-1
            break
        elif a > b:
            count = count+a/b
            a = a % b
        else:
            count = count+b/a
            b = b % a
    print count
