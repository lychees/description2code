#include <bits/stdc++.h>

using namespace std;

int main()
{
	long long int t, n, i, j, ans, a[200005];
	
	cin>>t;
	
	while(t--) {
		cin>>n;
		
		for (i = 0; i < n; i++) {
			cin>>a[i];
		}
		
		sort(a, a + n, std::greater<int>());
		ans = 0;
		if (n % 2 == 0) {
			for (i = 0; i < n; i = i + 2) {
				ans = ans + a[i] * a[i + 1];
			}
		} else {
			for (i = 0; i < n - 1; i = i + 2) {
				ans = ans + a[i] * a[i + 1];
			}
			ans = ans + a[n - 1];
		}
		cout<<ans<<endl;
	}
    return 0;
}
