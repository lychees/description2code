#include<cstdio>
#include<cstdlib>
int main()
{
	int n,MAX;
	bool *arr;
	scanf("%d",&n);
	MAX=n+3;
	
long long int i,j,*dp;

arr=(bool *)calloc(MAX,sizeof(bool));
dp=(long long *)calloc(MAX,sizeof(long long));

for(i=4;i<MAX;i+=2)
arr[i]=true ;

for(i=3;i<MAX;i+=2)
for(j=i;j*i<MAX;j+=2)
arr[i*j]=true;

dp[3]=2;
for(i=4;i<MAX;++i) 
{
                  dp[i]=dp[i-1];
                  if(!arr[i-1])
                  dp[i]+=i-1;
                  }
	printf("%lld\n",dp[n]);
	free(arr);
	free(dp);
	return 0;
}
