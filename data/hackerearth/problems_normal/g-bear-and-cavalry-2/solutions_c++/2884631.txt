#include <iostream>
#include <algorithm>

using namespace std;

int t, n, w[500], h[500], str, curr;

int main(){
    cin>>t;
    for(int z=0; z<t; ++z){
        cin>>n;
        str = 0; curr =0;
        for(int i=0; i<n; ++i)
            cin>>w[i];
        for(int i=0; i<n; ++i)
            cin>>h[i];
        sort(w+1, w+n);
        sort(h,h+n);
        curr = w[0]*h[n-1];
        for(int i=1; i<n-1; ++i)
            str = max(str, w[i]*h[n-1-i]);
        if(str>=curr)
            cout<<"NO"<<endl;
        else cout<<"YES"<<endl;

    }
    return 0;
}
