#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    ios::sync_with_stdio(false);cin.tie(0);cout.tie(0);
    int t,n,a[500005],b[500005];
    cin>>t;
    while(t--){
    	cin>>n;
    	for(int i=0;i<2*n;i++){
    		cin>>a[i];
    	}
    	sort(a,a+(2*n));
    	/*for(int i=0;i<2*n;i++){
    		cout<<a[i]<<endl;
    	}*/
    	for(int i=0;i<n;i++){
    		b[i]=a[i]+a[2*n-i-1];
    	}
    	sort(b,b+n);
    	cout<<b[n-1]<<"\n";
    }
    return 0;
}
