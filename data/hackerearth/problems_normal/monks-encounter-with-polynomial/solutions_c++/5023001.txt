#include <iostream>
#include <vector>
#include <cstdlib>
#include <algorithm>

using namespace std;
using std::vector;

int main()
{
    int t=0,i,A,B,C,K;
    cin>>t;
    while(t--) {
		cin>>A;
		cin>>B;
		cin>>C;
		cin>>K;
		for(i=0;i<=K;i++){
			if(((A*i*i)+(B*i)+C)>=K) {
				cout<<i<<endl;
				break;
			}
		}
    	
    }
    return 0;
}
