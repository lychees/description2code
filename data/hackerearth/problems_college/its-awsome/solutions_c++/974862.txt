#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long n;
    	cin>>n;
    	int s=0;
    	while(n)
    	{
    		if(n%10==4||n%10==7)
    		s++;
    		n=n/10;
    	}
    	int f=0;
    	while(s)
    	{
    		if(s%10==4||s%10==7)
    		f=1;
    		else
    		{
    			f=0;
    			break;
    		}
    		s=s/10;
    	}
    	if(f)
    	cout<<"YES\n";
    	else
    	cout<<"NO\n";
    }
}
