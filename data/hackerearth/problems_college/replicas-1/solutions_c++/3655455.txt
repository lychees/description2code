using namespace std;


#define LL long long int
#define sc1(x) scanf("%d", &x)
#define sc2(x, y) scanf("%d%d", &x, &y)
#define sc3(x, y, z) scanf("%d%d%d", &x, &y, &z)
#define pr1(x) printf("%lld\n", x)
#define FOR(i, a, n) for(int i=a;i<n;i++)
#define pp pair<int, int>

#define MAX 100001
#define MOD 1000000007LL

int mat[MAX], divi[MAX];

bool prime[MAX];

void setPrimes(int n) {
    prime[0] = prime[1] = 1;
    for(int i=2;i<n;i++) {
        if(!prime[i]) {
            divi[i] = i;
            for(LL j=i+i;j<n;j+=i) {
                prime[j] = true;
                divi[j] = i;
            }
        }
    }
}


void func(int n) {
	if(!prime[n]) {
		mat[n] = max(mat[n], 1);
	} else {
		int cnt = 0, prev = divi[n];
		while(n > 1) {
			if(prev == divi[n]) {
				cnt++;
			} else {
				mat[prev] = max(mat[prev], cnt);
				cnt = 1;
			}
			prev = divi[n];
			n = n / divi[n];
		}
		mat[prev] = max(mat[prev], cnt);
	}	
}


int main() {

	int t, n, x;

	setPrimes(MAX);


	sc1(t);
	while(t--) {
		sc1(n);

		memset(mat, 0, MAX);
		FOR(i, 0, n) {
			sc1(x);
			func(x);
		}


		LL result = 1;
		FOR(i, 0, MAX) {
			FOR(j, 0, mat[i]) {
				result = (result * i) % MOD;
			}
		}
		pr1(result%MOD);

	}
	return 0;
}
