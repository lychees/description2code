#include<bits/stdc++.h>
using namespace std;
#define ll				long long int
#define MOD				1000000007
#define si(a)			scanf("%d", &a)
#define sl(a)			scanf("%lld", &a)
#define pi(a)			printf("%d", a)
#define pl(a)			printf("%lld", a)
#define pn 				printf("\n")
ll pow_mod(ll a, ll b) {
	ll res = 1;
	while(b) {
		if(b & 1)
			res = (res * a) % MOD;
		a = (a * a) % MOD;
		b >>= 1;
	}
	return res;
}
int main() {
	int t;
	cin >> t;
	while(t--) {
		ll x, y, z;
		cin >> x >> y >> z;
		ll res = 0;
		res = max(res, x / 3 + y / 3 + z / 3);
		if(x >= 1 && y >= 1 && z >= 1) {
			res = max(res, 1 + (x - 1) / 3 + (y - 1) / 3 + (z - 1) / 3);
		}
		if(x >= 2 && y >= 2 && z >= 2) {
			res = max(res, 2 + (x - 2) / 3 + (y - 2) / 3 + (z - 2) / 3);
		}
		cout << res << endl;
	}
	return 0;
}
