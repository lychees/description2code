from heapq import heappush, heappop


def find_three_biggest(numbers):
    h = []
    for i in range(len(numbers)):
        heappush(h, -numbers[i])
        if i < 2:
            yield -1
        else:
            a = heappop(h)
            b = heappop(h)
            c = heappop(h)
            yield -1 * a * b * c
            heappush(h, a)
            heappush(h, b)
            heappush(h, c)


def solve(numbers):
    for v in find_three_biggest(numbers):
        print v


def execute():
    n = int(raw_input())
    numbers = map(int, raw_input().split())
    solve(numbers)


if __name__ == '__main__':
    execute()
