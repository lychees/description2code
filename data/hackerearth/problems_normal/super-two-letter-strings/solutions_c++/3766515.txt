#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define maxi 10007
long long dp[maxi][12];
long long solve(long long i,long long p,long long cont,long long n)
{
	long long val;
if(dp[i][cont]!=-1)
    return dp[i][cont];
    if(cont>=p)
		return 0;
    if(i==n)
		return 1;
		
val=(((solve(i+1,p,cont+1,n)%mod+solve(i+1,p,0,n))%mod)%mod);
dp[i][cont]=val;
return val;


}


int main()
{
 long long t;
 cin>>t;

 while(t--)
 {
     memset(dp,-1,sizeof(dp));
  long long n,p;
  cin>>n>>p;
  long long cont=0;
  cout<<solve(1,p,cont,n)<<endl;
 }
}
