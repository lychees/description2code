#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int T,N,i,a,b;
    cin>>T;
    while(T--)
    {
    	cin>>N;
    	int arr[N];
    	for(i=0;i<N;i++)
    	{
    		cin>>arr[i];
    	}
    	sort(arr,arr+N);
    	cin>>a>>b;
    	cout<<arr[b-1]+arr[N-a]<<endl;
    }
    return 0;
}
