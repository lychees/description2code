    #include <stdio.h>
    using namespace std;
    int gcd (int a , int b)
    {
    	if ( a == b)
    		return a;
    	if (b == 0)
    		return a;
    	return gcd(b , a%b);
    }
    int main()
    {
        int t;
        scanf("%d",&t);
        for(int i = 0 ; i < t ; i++)
        {
        	int n;

        	int small = 0 , large = 0;

        		}
        		else
        		{
        			small = rem * placeValue + small;
        		}
        		
        		rem = rem + carry;
        		if (rem % 2 != 0)
        		{
        			if (rem > 0 && rem < 8)
        			{
        				large = (rem + 1) * placeValue ;
        				carry = 0;
        			}
        			else
        			{
        				large = 20 * placeValue;
        				carry = 1;
        			}
        		}

        		{
        			large = rem * placeValue;

        			large = rem * placeValue + large;

        	}
