#include <iostream>
using namespace std;
long int n,t,i,a[51],h[100000];
int main()
{
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=0;i<100000;i++)
			h[i]=0;
		for(i=0;i<51;i++)
		{
			a[i]=0;
		}
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			h[a[i]]+=1;
		}
		for(i=0;i<n;i++)
		{
			if(h[a[i]]%2==1)
			{
				cout<<a[i]<<endl;
				break;
			}
		}
	}
}
