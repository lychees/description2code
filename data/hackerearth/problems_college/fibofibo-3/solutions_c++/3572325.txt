#include<iostream>
using namespace std;
int main()
{
    long int t,a[42],b[42],i;
    a[0]=1;b[0]=0;
    a[1]=0;b[1]=1;
    for(i=2;i<=
    42;i++)
    {
        a[i]=a[i-1]+a[i-2];
        b[i]=b[i-1]+b[i-2];
       // cout<<a[i]<<" "<<b[i]<<endl;
    }

    cin>>t;

    while(t--)
    {
        long n;
        cin>>n;
        cout<<a[n]<<" "<<b[n]<<endl;
    }
}
