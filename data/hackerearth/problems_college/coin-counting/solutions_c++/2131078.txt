#include <iostream>
using namespace std;

int count(int *a,int n,int sum)
{

	if (sum==0)
	return 1;
	else if (sum<0)
	return 0;
	else if (n<=0 && sum>=1)
	return 0;
	else
	return (count(a,n-1,sum)+count(a,n,sum-a[n-1]));
}	
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int n;
   	cin>>n;
   	int arr[]={1,3,5};
   	int sum=n;
   	cout<<count(arr,3,sum)<<endl;
   }
    return 0;
}
