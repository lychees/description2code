#include <bits/stdc++.h>
using namespace std;

int main()
{
int t;
scanf("%d", &t);
while(t--) {
int n;
scanf("%d", &n);
int a[10005];
memset(a, 0, sizeof(a));
for(int i = 0; i < n; i++) {
int temp;
scanf("%d", &temp);
a[temp] = 1;
}
int ans = 0;
for(int i = 0; i < 10005; i++) {
if(a[i] == 1) {
ans++;
}
}
printf("%d\n", ans);
}
return 0;
}
