#include<bits/stdc++.h>
#include<algorithm>
using namespace std;
#define fr(i,a,b) for(int i=a;i<b;i++)
#define frn(i,a,b) for(int i=a;i>b;i--)
#define si(x) scanf("%d",&x)
#define sll(x)  scanf("%lld",&x)
#define pln printf("\n")
#define pi(x) printf("%d",x)
#define pll(x) printf("%lld",x)
typedef long long ll;
ll MOD=1000000007;

/*
   a+xd=n*m.
  (a+xd)%n=0.
   -a%n=(xd)%n;
    (p-a%n)d^-1=x


*/

ll pw(ll a,ll b){
  ll r=1;
  while(b>0){
     if(b&1){
       r=(r*a)%MOD;
        }
     b=b>>1;
     a=(a*a)%MOD;
  }
return r;
}


int main(){
  int t;
  si(t);
  while(t--){
    ll a,d,p,x,s,ans=-1;
    sll(a),sll(d);
    sll(p);
    MOD=p;
    a%=p;
    d%=p;

    if(a==0){
        ans=0;
    }
    else if(a!=0&&d==0){
        ans=-1;
    }
    else{
       ll invd=pw(d,p-2);
          ans=((p-a)*invd)%p;
    }
    pll(ans);pln;

  }

return 0;
}
