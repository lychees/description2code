#include <iostream>
#include<string.h>
using namespace std;

int main()
{   int n;
    cin>>n;
    while(n--)
    {   long long x,a,b,i,j,k=0;
        cin>>x;
        for(i=0;i<=x/5;++i)
        {   for(j=0;j<=x/3;++j)
                if(5*i+3*j==x)
                {   k=1;
                    break;
                }
            if(k==1)
                break;
        }
        if(k==0)
        {    cout<<"-1\n";
            continue;
        }
        while(j--)
            cout<<"555";
        while(i--)
            cout<<"33333";
        cout<<"\n";
    }
    return 0;
}
