#include <iostream>
#include <cmath>
using namespace std;
int main()
{
    int T;
    cin>>T;
    for(int j=1;j<=T;j++)
    {
    	long long int k;
        long long int x,i;
        cin>>x>>k;
        long long int count=0;
        for(i=1;i<90000000000000;i++)
        {
            if((x+i)==(x^i))
            {
            	count++;	
            }
            if(count==k)
            {
            	cout<<"Case #"<<j<<": "<<i<<endl;
                break;
            }
            
            
        }
       
    }
}
