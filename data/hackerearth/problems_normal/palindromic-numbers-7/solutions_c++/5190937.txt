#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    int visit[100007];
    visit[-1]=0;
    visit[0]=1;
    for(int k=1;k<=100004;k++)
    {
        int n=k;
        int reverse=0;
        while(n != 0) {
            int remainder = n%10;
            reverse = reverse*10 + remainder;
            n/=10;
        }
        if(reverse==k)
            visit[k]=visit[k-1]+1;
        else
            visit[k]=visit[k-1];
    }
    while(t--)
    {
        int a,b,ans;
        cin>>a>>b;
        if(a!=0)
         ans=visit[b]-visit[a-1];
        else
        ans=visit[b];

        cout<<ans<<endl;
    }
    return 0;
}
