#include<bits/stdc++.h>
using namespace std;
int arr[1000002];
void seive(){
	for(int i=0; i<1000002; i++) arr[i]=-1;
	arr[0]=0;
	arr[1]=0;
	for(int i=2; i<1000002; i++){
		if(arr[i]==-1){
			arr[i]=0;
			for(int j=2; i*j<1000002; j++){
				if(arr[i*j]==-1) arr[i*j]=i;
			}
		}
	}
}
int main(){
	seive();
//	for(int i=1; i<100; i++) cout<<arr[i]<<endl;
	for(int i=1; i<1000002; i++) arr[i]=arr[i]+arr[i-1];
	int t;
	cin>>t;
	while(t--){
		int l,r; cin>>l>>r;
		cout<<arr[r]-arr[l-1]<<endl;
	}
}
