#include <bits/stdc++.h>

using namespace std;

long long powsum(long long n)
{
		long long sum=0;
		while(n)
		{
			int temp;
			temp=n%10;
			n/=10;
			sum+=temp*temp;
		}
		return sum;
}

int main()
{
	int t;
	cin>>t;
	int i=1;
	while(i<=t)
	{
		map<long long,bool> area;
		long long n;
		cin>>n;
		if(n==1)
		{
			cout<<"Case #"<<i<<":YES\n";
			i++;
			continue;
		}
		area[n]=true;
		bool flag=true;
		long long j=0;
		do
		{
			long long p=powsum(n);
			if(area[p]==true)
			{
				cout<<"Case #"<<i<<":NO\n";
				break;
			}
			else if(p==1)
			flag=false;
			else
			{
				area[p]==true;
				n=p;
			}
			j++;	
		}while(j<100000);
		if(!flag)
		cout<<"Case #"<<i<<":YES\n";
		else
		cout<<"Case #"<<i<<":NO\n";
	i++;
	}
	return 0;
}
