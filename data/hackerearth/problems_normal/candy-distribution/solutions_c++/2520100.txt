#include<bits/stdc++.h>
using namespace std;
 
const long long int mod = 1000000007;
unordered_map<long long int, long long int> dp;
 
long long int power(long long int a, long long int n)
{
    if(n == 0)
        return 1;
    if(n == 1)
        return a;
    long long int temp = power(a, n/2);
    temp = temp * temp;
    if(temp > mod)
        temp %= mod;
    if(n & 1){
        return (a * temp) % mod;
    }
    return temp;
}
 
long long int fun(long long int n)
{
    if(n == 1 || n == 2)
        return 1;
    if(dp[n])
        return dp[n];
    n--;
    long long int val = 1;
    for(long long int i=2; i*i<=n; i++){
        if(n % i == 0){
            val += power(fun(i), n/i);
            if(val > mod)
                val %= mod;
            if(i != n/i){
                val += power(fun(n/i), i);
                if(val > mod)
                    val %= mod;
            }
        }
    }
    dp[n+1] = val;
    return dp[n+1];
}
 
int main()
{
    long long int t;
    cin>>t;

    cout<<fun(t)<<"\n";
}
