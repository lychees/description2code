    p = list(raw_input())
    b = sorted(raw_input())
    n = 0
    for i in range(len(b)):
        while n < len(p):
            if p[n] > b[i]:
                p[n] = b[i]
                n += 1
                break
            else:
                n += 1
    print ''.join(p)
