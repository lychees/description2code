{
	//ios_base::sync_with_stdio(false);

	int n,i,x,ans=0;
	cin>>n;
	vi a;
	f(i,n) {cin>>x;a.pb(x);}
	while(a.size()>2)
	{
		int maxx=0,indx;
		for(i=1;i<a.size()-1;i++)
		{
			if(a[i-1]*a[i+1]>maxx) {indx=i;maxx=a[i-1]*a[i+1];}
