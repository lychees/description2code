#include<bits/stdc++.h>

#define M 1000000007
#define MX 2000005
#define PI 3.141593
using namespace std;
typedef pair<ll,ll> ii;
ll factorize(ll n)
{
    ll p=0;
    while(n%2==0)
    {
        n/=2;
        p++;
    }
    p++;
    for(ll i=3;i*i<=n;i+=2)
    {
        ll x=1;
        while(n%i==0)
        {
            n/=i;
            x++;
        }
        p*=x;
    }
    p--;
    return p;
}
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
        ll n,ans=0,add=0,prev,fl=0;
        cin>>n;
        for(ll i=1;i*i<=n;i++)
        {
            if(i*i!=n||i*i==n)
            {
                fl=i;
                if(i*i==n)
                    fl=-1;
                ll x=i,y;
                y=n/x;
                x=(x*y)%M;
                ans+=x;
                if(i>1)
                {
                    add=y*(y+1);
                    add/=2;
                    add=((prev*(prev+1))/2)-add;
                    //add%=M;
                    add*=(i-1);
                    ans+=add;
                    ans%=M;
                }
                prev=y;
            }

            //else ans+=n;
            ans%=M;
        }
        if(fl>0)
            {
                add=prev*(prev+1);
                add/=2;
                add=(add-(fl*(fl+1))/2)%M;
                ans+=add*fl;
                ans%=M;
            }
        ans%=M;
        cout<<ans%M<<"\n";
    }
}
