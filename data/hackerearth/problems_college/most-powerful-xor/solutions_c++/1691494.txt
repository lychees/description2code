#include <iostream>
using namespace std;

int main()
{
   int test;
   cin>>test;
   
   while(test>0)
   {
   		int a,b;
   		int max=0;
   		cin>>a;
   		cin>>b;
   		b=b+1;
   		
   		for(int j=a;j<b;j++)
			{
				for(int k=j;k<b;k++)
				{
					if(max<(j^k))
					{
						max=(j^k);
					}
				}
			}
			
			cout<<max<<"\n";
			test--;
   }
    return 0;
}
