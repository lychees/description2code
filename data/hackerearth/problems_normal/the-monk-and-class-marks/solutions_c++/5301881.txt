#include <iostream>
#include <vector>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    
	int n;
	scanf("%d",&n);
	assert(n>0 and n<100001);
	int y; string x;
	vector < pair <int, string> > a;	
	for (int i=0; i<n; i++) {
		cin >> x >> y;
		assert(y>0 and y<101);
		assert(x.size()>0 and x.size()<101);
		a.push_back(make_pair(-y,x));
	}
	
	sort(a.begin(), a.end());
    for(int i = 0; i < a.size(); i++) {
    	cout << a[i].second << " " << -1*a[i].first << endl;
	}
	
    return 0;
}
