t = input()

while t:
    N, K = map(int, raw_input().split(" "))

    ar = map(int, raw_input().split(" "))

    count = 0
    flag = 0
    for x in ar:
        if x <= 0:
            count += 1
        if count >= K:
            print "NO"
            flag = 1
            break

    if not flag:
        print "YES"

    t -= 1
