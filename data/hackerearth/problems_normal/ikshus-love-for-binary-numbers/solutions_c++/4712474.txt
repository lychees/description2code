#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll dp[64][64];

ll rec(int n,int cur,int k){
	ll &ans = dp[n][cur];
	if(ans != -1)
		return ans;

	ans = 0;
	if(n <= 0){
		ans = 1;
		return 1LL;
	}
	if(cur == k-1)
		ans = rec(n-1,0,k);
	else{
		ans += rec(n-1,cur+1,k);
		ans += rec(n-1,0,k);
	}
	return ans;
}
int main(int argc, char const *argv[])
{
	int n,k;
	scanf("%d%d",&n,&k);
	memset(dp,-1,sizeof dp);
	//cout<<dp[0][0]<<endl;

	ll ans = rec(n,0,k);
	//cout<<ans<<endl;
	ll den = (1LL<<n);
	ll num = den - ans;
	//printf("%lld/%lld\n",num,den);

	int gcd = __gcd(num,den);
	while(gcd > 1){
		//while(num%gcd == 0 && den%gcd == 0)
			num /= gcd,den /= gcd;
		gcd = __gcd(num,den);
	}
	printf("%lld/%lld\n",num,den);
	return 0;
}
