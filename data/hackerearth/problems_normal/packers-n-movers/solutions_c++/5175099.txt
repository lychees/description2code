#include<bits/stdc++.h>
using namespace std;
int dp[18][255],w[255],wx[255];
int rec(int m,int n)
{
    if(m==1)
    {
        int l=0;
        l=wx[n];
        dp[m][n]=l;
        return l;
    }
    if(m>=n)
    {
        int l=INT_MIN;
        for(int i=1;i<=n;i++)
            l=max(l,w[i]);
        return dp[m][n]=l;
    }
    if(dp[m][n]!=INT_MAX)
        return dp[m][n];
    for(int k=1;k<=n;k++)
        dp[m][n]=min(dp[m][n],max(dp[m-1][k]=rec(m-1,k),wx[n]-wx[k]));//,cout<<dp[m][n]<<"\n";
    return dp[m][n];
}
int main()
{
    int i,j,k,l,m,n;
    cin>>m>>n;
    for(i=1;i<=n;i++)
        cin>>w[i];
    for(i=0;i<=m;i++)
        for(j=0;j<=n;j++)
            dp[i][j]=INT_MAX;
    for(i=1;i<=n;i++)
        wx[i]=wx[i-1]+w[i];
    cout<<rec(m,n)<<endl;
}
