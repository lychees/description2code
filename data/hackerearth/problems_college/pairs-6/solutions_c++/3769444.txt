#include<iostream>
#include<algorithm>
#include<vector>

using namespace std;

const int MAXN = 100 + 3;

int n, x, y, d[2][2][MAXN][MAXN];

int main(){
	ios::sync_with_stdio(false);
	cin.tie(0);
	for (int f = 0; f < MAXN; f++)
		for (int g = 0; g < MAXN; g++)
			for (int a = 0; a < 2; a++)
				for (int b = 0; b < 2; b++){
					if (f == 0 && g == 0)
						continue;
					if (b == 0 && f == 0){
						if (a == 0)
							d[a][b][f][g] = 1e8;
						else
							d[a][b][f][g] = -1e8;
						continue;
					}
					if (b == 1 && g == 0){
						if (a == 0)
							d[a][b][f][g] = 1e8;
						else
							d[a][b][f][g] = -1e8;
						continue;
					}
					if (f + g < 2)	continue;

					if (a == 0){
						if (b == 0)
							d[a][b][f][g] = max(d[1][0][f - 1][g], d[1][1][f - 1][g] + 1);
						else
							d[a][b][f][g] = max(d[1][1][f][g-1], d[1][0][f][g-1] + 1);
					}
					else{
						if (b == 0)
							d[a][b][f][g] = min(d[0][0][f - 1][g], d[0][1][f - 1][g] + 1);
						else
							d[a][b][f][g] = min(d[0][1][f][g - 1], d[0][0][f][g - 1] + 1);
					}
				}

	int te;	cin >> te;
	while (te--){
		cin >> n >> x >> y;
		if (min(d[0][0][x][y], d[0][1][x][y]) * 2 >= n - 1)
			cout << "Chotu\n";
		else
			cout << "Motu\n";
	}
	return 0;
}
