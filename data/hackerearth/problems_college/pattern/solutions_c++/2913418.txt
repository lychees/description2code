#include<bits/stdc++.h>
#define ll long long int
#define inp(x) scanf("%lld",&x)
using namespace std;
int main(){
	ll n,x,y,m;
	inp(n);
	for(ll i=1;i<=n;++i){
		cout<<"*";
	}cout<<endl;
	x = 2;y = n-1;
	for(ll i=2;i<=n-1;++i){
		for(ll j=1;j<=n;++j){
			if(x==i&&y==j){
				cout<<"*";
				++x;--y;
				j = n + 1;
				cout<<endl;
			}else{
				cout<<" ";
			}
		}
	}
	for(ll i=1;i<=n;++i){
		cout<<"*";
	}cout<<endl;
	return 0;
}
