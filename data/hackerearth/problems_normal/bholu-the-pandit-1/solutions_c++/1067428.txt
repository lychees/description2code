#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}
int main(){
    int t=scan();
    while(t--){
        int a=scan(),b=scan(),c=scan();
        int g = __gcd(a,b);
        g=__gcd(g,c);
        unsigned long long aa = a/g,bb=b/g,cc=c/g;
        unsigned long long ans = aa * bb;
        ans %= mod;
        ans *= cc;
        ans %= mod;
        printf("%d %llu\n",g,ans);
    }
}
