#include <iostream>
using namespace std;

int main()
{
	long long int t,n,x,temp,i;
    cin>>t;
    for(i=1;i<=t;i++)
    {
    	cin>>n;
    
    	while(n>9)
    	{
    		temp=0;
    		while(n>0)
    		{
    			x=n%10;
    	     temp+=x*x;
    	      n=n/10;
    		}
    		
    		n=temp;
    	}
    	if(n==1||n==7||n==10)
        cout<<"Case #"<<i<<":YES"<<endl;
    	else
    	cout<<"Case #"<<i<<":NO"<<endl;
    }
    return 0;
}
