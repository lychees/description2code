
    #include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ll n,m,i,j,ans,num,diff;
    scanf("%lld%lld",&n,&m);
    vector<ll> v(n);
    for(i=0;i<n;i++)
    {
    	scanf("%lld",&(v[i]));
    }
    sort(v.begin(),v.end());
    while(m--)
    {
    	diff=1000000000;
    	scanf("%d",&num);
    	for(i=0,j=n-1;i<j;)
    	{
    		
    		if(v[i]+v[j]==num)
    		{
    			ans=num;
    			break;
    		}
    		else if(v[i]+v[j]>num)
    		{
    			if(v[i]+v[j]-num<diff)
    			{
    				diff=v[i]+v[j]-num;
    				ans=v[i]+v[j];
    			}
    			j--;
    		}
    		else
    		{
    				if(num-v[i]-v[j]<diff)
    					{
    				diff=v[i]+v[j]-num;
    				diff=-diff;
    				ans=v[i]+v[j];
    			}
    			i++;
    		}
    	}
    	printf("%d\n",ans);
    }
    return 0;
}
