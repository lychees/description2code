for i in range(0, int(raw_input())):
    p, a, b, x, y = map(int, raw_input().split())
    j = (x+y-a-b)
    k = (x-y-a+b)
    if j % 2 != 0 and k % 2 != 0:
        print "Terrorists Win !"
    else:
        d = 0
        if j != 0:
            d += 1
        if k != 0:
            d += 1
        if d < p:
            print "Counter-Terrorists Win !"
        else:
            print "Terrorists Win !"
