#include<bits/stdc++.h>
using namespace std;

int main()
 {
 	
 	int n,x;
 	cin>>n;
 	int flag=1;
 	n--;
 	cin>>x;
 	while(x%2==0) x/=2;
    int prev=x;
 	while(n--)
 	 {
 	 	cin>>x;
 	 	while(x%2==0) x/=2;
 	 	if(x!=prev)
 	 	   flag=0;
 	 }
 	if(flag)
 	  cout<<"YES"<<endl;
 	else
 	  cout<<"NO"<<endl;
 }
