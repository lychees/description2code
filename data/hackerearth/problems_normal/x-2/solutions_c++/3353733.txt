#include <bits/stdc++.h>

typedef long long int ll;
#define mod 1000000007
#define S(x) scanf("%lld",&x)
using namespace std;

ll powr (ll a, ll b)
{
    if (b == 0)
        return 1;
    ll x = powr(a, b/2);
    if (b % 2 == 0)
        return (x*x)%mod;
    else
        return (((x*x)%mod)*a)%mod;
}

ll ans[1000011];

int main()
{
    ll t,n,tsum,i,j,x,sum;
    S(t);
    while(t--) {
        S(n);
        S(tsum);
        memset(ans,0,sizeof(ans));
        vector <ll> v;
        for(i = 0; i < n ; i++) {
            S(x);
            v.push_back(x);
        }
        i = 0;
        j = 0;
        sum = 0;
        while(j < n || i < n) {
            if(j < n && sum + v[j] <= tsum) {
                sum += v[j];
                j++;
            } else {
                sum -= v[i];
                if (sum <= tsum) {
                    ans[1]++;
                    ans[j-i+1]--;
                }
                i++;
            }
        }
        sum = 0;
        for(i = 1; i <= n; i++) {
            sum += ans[i];
            printf("%lld ",sum);
        }
        printf("\n");
    }
    return 0;
}
