#include <bits/stdc++.h>
using namespace std;
char s[150];
bool solve(int l,int r){
  while(l<r){
  	if(s[l]!=s[r]) return false;
  	l++;
  	r--;
  }
  return true;
}
int main()
{
	int t;
    scanf("%d",&t);
    while(t--){
    scanf("%s",s);
    int len=strlen(s);
    //reverse(s,s+len);
    int maxLen=0,i;
    int l,r;
    for(i=0;i<len;i++){
   		r=len-1,l=i;
   	    bool flag = solve(l,r);
   	    if(flag)
   	    maxLen = max(maxLen,r-l+1);
    }
  // maxlen=maxlen*2-1;
   printf("%d\n",len+len-maxLen);
    
    }
    return 0;
}
