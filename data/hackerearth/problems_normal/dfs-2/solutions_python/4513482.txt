def startIndex(p,l,h,k):
	res=h+1
	while(l<=h):
		mid=(l+h)>>1
		if p[mid]>=k:
			res=mid
			h=mid-1
		else:
			l=mid+1
	return res

def endIndex(p,l,h,k):
	res=-1
	while l<=h:
		mid=(l+h)>>1
		if p[mid]<=k:
			res=mid
			l=mid+1
		else:
			h=mid-1
	return res


n=int(raw_input())
p=map(int,raw_input().split())
p.sort(reverse=False)
q=int(raw_input())
for i in range(q):
	l,r=map(int,raw_input().split()) 
	s=startIndex(p,0,n-1,l)
	e=endIndex(p,0,n-1,r)
	ans= e-s+1 if s<=e else 0
	print ans
