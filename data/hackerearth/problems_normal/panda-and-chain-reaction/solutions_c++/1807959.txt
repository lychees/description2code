#include <iostream>
#include <cstdio>
#define gc getchar_unlocked

using namespace std;
#define mod 1000003

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

long long fact[mod], i;

int main() {
	int t;
	long long n, x;
	fact[0] = 1;
	for(i = 1; i < mod; i++) {
		fact[i] = (fact[i-1] * i) % mod;
	}
	scanint(t);
	while(t--) {
		cin>>n>>x;
		if(n >= mod)
			cout<<0<<endl;
		else
			cout<<(x * fact[n]) % mod<<endl;
	}
	return 0;
}
