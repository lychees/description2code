#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	long n, id, top=-1,last;
   	cin>>n>>id;
   	long arr[n];
   	arr[++top]=id;
   	last=id;
   	while(n--)
   	{
   		char c;
   		cin>>c;
   		if(c=='P')
   		{
   			long k;
   			cin>>k;
   			last=arr[top];
   			arr[++top]=k;
   		}
   		else if(c=='B')
   		{
   			long last1=arr[top];
   			arr[top-1]=last1;
   			arr[top]=last;
   			last=last1;
   		
   		}
   	}
   	
   	cout<<"Player "<<arr[top]<<endl;
   }
   
   
   
    return 0;
}
