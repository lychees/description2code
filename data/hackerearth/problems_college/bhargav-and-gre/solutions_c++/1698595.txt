#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int i,n,syn=0,ant=0,len;
    char a[100],b[100];
    cin>>a;
    len=strlen(a);
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>b;
    	if(strlen(b)==len)
    	{
    		if(a[0]==b[0])
    			syn++;
    		if(b[0]==a[0]+3)
    			ant++;
    	}
    	else
    		continue;
    }
    cout<<syn<<" "<<ant;
    return 0;
}
