#include<iostream>
using namespace std;

int main()
{
	int n,temp,count;
	cin>>n;
	int a[n],b[n],rank[n],nrank[n];
	
	for(int i=0;i<n;i++)
	nrank[i]==0;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		b[i]=a[i];
	}
	for(int i=0;i<n;i++)
	for(int j=i+1;j<n;j++)
	{
		if(b[i]<b[j])
		{
		temp=b[i];
		b[i]=b[j];
		b[j]=temp;
		}
	}
	rank[0]=1;
	for(int i=0;i<n;i++)
	{
		for(int j=i+1;j<n;j++)
			{
			    if(b[j]==b[i])
				{
					rank[j]=rank[i];
				}
				else
				{
					rank[j]=j+1;
				}
			}
		
	}
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
			{
				if(a[i]==b[j])
				{
				nrank[i]=rank[j];
				break;
				}
				else
				continue;
			}
	}
	for(int i=0;i<n;i++)
	{
		cout<<nrank[i]<<" ";
		
	}
	return 0;
}
