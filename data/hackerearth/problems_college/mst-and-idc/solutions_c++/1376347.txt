#include <iostream>
using namespace std;

#include <stdio.h>

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   int n;
   cin>>n;
   long long int max=-3;
   long long int min=10000000;
   long long int ans=0;
   for(int x=0;x<n;x++)
   {
   	long int h;
   	cin>>h;
   	ans+=h;
   	if(h<min)
   	min=h;
   	if(h>max)
   	max=h;
   }
   cout<<((ans/n)-min)*2+(max-(ans/n))*2<<endl;
   }
return 0;
}
