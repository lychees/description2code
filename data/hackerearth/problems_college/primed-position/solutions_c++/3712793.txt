#include <vector>
#include <iostream>
#include <stdio.h>
#define MOD 1000000007

using namespace std;
#define SIZE 1500000

int main(){
	int a[SIZE] = {0};
	a[0]=a[1]=1;
	for(int i=2; i<SIZE; i++){
		if(a[i]==0){
			int j=2;
			while(i*j<SIZE){
				a[i*j]=1;
				j++;
			}
		}
	}

	std::vector<int> v;
	for(int i=0;i<SIZE;i++){
		if(a[i]==0){
			v.push_back(i);
		}
	}

	//for(int i=0; i<v.size(); i++)
	//	cout << i+1 << "|" << v[i] << endl;

	std::vector<int> v2;
	for(int i=0; v[i]<v.size(); i++){
		int index = v[i];
		v2.push_back(v[index-1]);
		if(v[index] == 0)
			break;
	}

	//cout << "[" << endl;
	//for(int i=0; i<v2.size(); i++)
	//	cout << v2[i] << ",";

	int t;
	scanf("%d", &t);
	while(t--){
		int a,b;
		scanf("%d%d", &a, &b);
		a = v2[a-1]%1000000007;
		b = v2[b-1]%1000000007;
		long long ans = ((long long)a*b ) % MOD;
		printf("%lld\n", ans);
	}
}
