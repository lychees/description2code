#include <iostream>
#include <cstring>

using namespace std;

int main()
{
    int t;
    cin>>t;
    int n;
    int a[100005];
    int b[100005];
    bool flag;
    
    while(t--){
    	memset(a,0,sizeof(a));
    	cin>>n;
    	for(int i=1;i<=n;i++){
    		cin>>a[i];
    		b[a[i]] = i;
    	}
    	
    	
    	for(int i=1;i<=n;i++){
    		if(a[i]==b[i]){
    			flag= true;
    		}else{
    			flag = false;
				break;
			}
    	}
    	if(flag ==false)
    		cout<<"not inverse\n";
    	else cout<<"inverse\n";
    }
    return 0;
}
