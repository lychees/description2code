#include <bits/stdc++.h>
#define PB push_back
using namespace std;

int level[30], in[30], vis[30];
vector<int > start, v[30];
vector<char > r[30];
queue<int > q;

void BFS(){
    int temp;
   int i;
    for(i = 0; i < start.size(); i++){
        q.push(start[i]);
        level[start[i]] = 1;
    }
    while(!q.empty()){
        temp = q.front();
        q.pop();
        for(int i=0;i<v[temp].size();i++){
            if(level[v[temp][i]] < level[temp] + 1){
                q.push(v[temp][i]);
                level[v[temp][i]] = level[temp] + 1;
            }
        }
    }
}

int main(){
    //freopen("input.txt","r",stdin);
    int n, i, j, g;
    string s[110];
    cin>>n;

    for(i=0;i<30;i++){
        in[i] = 0;
        vis[i] = -1;
    }

    for(i=1;i<=n;i++){
        cin>>s[i];
        for(j=0;j<s[i].length();j++){
            vis[s[i][j] - 'a'] = 1;
        }
    }





    for(i=2;i<=n;i++){

        j = 0;
        g = 0;
        while(s[i][j] == s[i - 1][j]){
            j++;
            if(j == s[i].length() || j == s[i - 1].length()){
                g = 1;
                break;
            }

        }
        if(g == 1)
        continue;
        v[s[i - 1][j] - 'a'].PB(s[i][j] - 'a');
        vis[s[i][j] - 'a'] = 1;
        vis[s[i - 1][j] - 'a'] = 1;

        in[s[i][j] - 'a']++;
    }

//    cout<<"here"<<" "<<vis[5]<<" "<<vis[6]<<endl;
    for(i=0;i<26;i++){
        if(in[i] == 0 && vis[i] == 1){
            start.PB(i);
        }
    }

    for(i=0;i<26;i++){
        level[i] = -1;
    }
    BFS();

/*
    for(i=0;i<26;i++){
        if(level[i] == -1)
        continue;
        cout<<i<<" "<<level[i]<<endl;
    }
    */
//    return 0;
    for(i=0;i<26;i++){
        if(level[i] == -1)
        continue;
        r[level[i]].PB((char)(i + 97));
    }

    for(i=1;i<=30;i++){
        sort(r[i].begin(), r[i].end());
        for(j=0;j<r[i].size();j++){
            cout<<r[i][j];
        }
        if(r[i].size() != 0)
        cout<<endl;
    }

}
