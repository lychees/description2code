#include <iostream>
#include <vector>
#include <algorithm>
#define lli long long int
#define pb push_back
using namespace std;

int main()
{
    int t,n,sayan,raghu,i;
    lli a,b,cal;
    vector<lli> dishes;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>n;
    	for(i=0;i<n;++i)
    	{
    		cin>>cal;
    		dishes.pb(cal);
    	}
    	sort(dishes.begin(),dishes.end());
    	cal=0;raghu=sayan=0;
    	for(i=0;i<n;++i)
    	{
    		cal+=dishes[i];
    		if(cal<=a)
    			raghu++;
    		if(cal<=b)
    			sayan++;    		
    	}
    	if(raghu>sayan)
    		cout<<"Raghu Won"<<endl;
    	else 
    		if(raghu<sayan)
    			cout<<"Sayan Won"<<endl;
    		else
    			cout<<"Tie"<<endl;
    	dishes.clear();		    	
    }
    return 0;
}
