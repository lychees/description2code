#include "bits/stdc++.h"
using namespace std;
typedef long long int ll;
int main()
{
    ll t,n,i;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	ll a[n];
    	for(i=0;i<n;i++) cin>>a[i];
    	for(i=0;i<n;i++)
    	if(a[i]>a[i+1])
    	{cout<<"NO\n";goto out;}
    	cout<<"YES\n";
    	out:;
    }
    return 0;
}
