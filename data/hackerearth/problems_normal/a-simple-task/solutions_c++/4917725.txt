using namespace std;
typedef long long int ll;
typedef unsigned long long int llu;

ll n;
ll arr[107];
set<ll>ans;

void check(int kval){
    ll rem = arr[0]%kval;
    for(ll i=1; i<n; i++){
        if((arr[i]%kval)!=rem){
            return;
        }
    }
    if(kval!=1) ans.insert(kval);
}

int main(){
    //fr("rem.txt");
    //fw("out.txt");

    cin >> n;
    for(ll i=0;i<n;i++) cin >> arr[i];

    //sort(arr,arr+n);

    ll minD = abs(arr[1]-arr[0]);
    for(ll k=1;k<=sqrt(minD);k++){
        if((minD%k)!=0) continue;

        check(k);
        check(minD/k);
    }

    for(set<ll>::iterator it=ans.begin();it!=ans.end();it++){
        if(it!=ans.begin()) cout << ' ';
        if((*it)!=1) cout << (*it);
    }
    cout << endl;


    return 0;
}
