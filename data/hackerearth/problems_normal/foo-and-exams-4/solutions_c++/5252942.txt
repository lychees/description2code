#include<iostream>
#include<math.h>
using namespace std;
long long bSearch(long long,long long,long long,long long,long long);
 
int main()
{
	int r;cin>>r;
	for(int y=0;y<r;y++)
	{
		long long a,b,c,d,k;
		cin>>a>>b>>c>>d>>k;
		cout<<bSearch(a,b,c,d,k)<<endl;
	}
	return 0;
}
 
long long bSearch(long long a,long long b,long long c,long long d,long long k)
{
	long long t;
	int low=0,high=cbrt(k);
	while(low<high)
	{
		t=low+(high-low)/2;
		if(t == low) t++;
		
		long long f=t*((a*t+b)*t+c)+d;
		if(f>k) high=t-1;
		if(f<k) low=t;
		
	}
	return low;
}
