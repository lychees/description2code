#include<bits/stdc++.h>
#define _ ios_base::sync_with_stdio(false);cin.tie(0);
using namespace std;

int minv(int x)
{
    vector<int> v;
    while(x>0)
    {
        int temp=x%10;
        if(temp == 6)
            temp=5;
       v.push_back(temp);
       x/=10;
    }
    int sum=0;

    for(int i= v.size()-1;i>=0;i--)
        sum=sum*10+v[i];
    return sum;
}

int maxv(int x)
{
    vector<int> v;
    while(x>0)
    {
        int temp=x%10;
        if(temp == 5)
            temp=6;
       v.push_back(temp);
       x/=10;
    }
    int sum=0;

    for(int i= v.size()-1;i>=0;i--)
        sum=sum*10+v[i];
    return sum;
}

int main()
{
    int a,b;
    cin>>a>>b;
    cout<<minv(a)+minv(b)<<" "<<maxv(a)+maxv(b)<<'\n';
    return 0;
}
