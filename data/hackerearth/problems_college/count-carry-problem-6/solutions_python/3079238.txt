t=input()
while t>0:
    count=0
    a,b=map(int,raw_input().split())
    c=0
    while a!=0 and b!=0:
        x=a%10
        y=b%10
        z=x+y+c
        if z>9:
            c=z-9
            count+=1
        a=a/10
        b=b/10
    if count==0:
        print "No carry operation"
    else:
        if count==1:
            print count,"carry operation"
        else:
            print count,"carry operations" 
    t-=1
