t=int(raw_input())
while t>0:
	index,n=raw_input().split()
	index=int(index)
	n=int(n)
	if index==2 and n==0:
		print("2")
	elif index==1 and n==0:
		print("1")
	elif index==0 and n==0:
		print("0")		
	elif index==0 and n%2==0:
		print("0")
	elif index==0 and n%2==1:
		print("1")
	elif index==1 and n%2==0:
		print("1")
	elif index==1 and n%2==1:
		print("0")
	elif index==2 and n%2==0:
		print("0")
	elif index==2 and n%2==1:
		print("1")	
	
	t=t-1
