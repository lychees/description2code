#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++) cin>>arr[i];
    sort(arr,arr+n);
 //   for(int i=0;i<n;i++) cout<<arr[i]<<" ";
 //   cout<<"\n";
    if(n==1) cout<<"0\n";
    else
    {
   	int arr2[n];
    if(n%2==0)
    {
    	int j=0;
    	for(int i=0;i<n/2;i++)
    	{
    		arr2[j]=arr[i];
    		arr2[j+1]=arr[n-1-i];
    		j+=2;
    	}
    }
    if(n%2==1)
    {
    	int j=0;
    	for(int i=0;i<=n/2;i++)
    	{
    		if(i!=n/2)
    		{
    			arr2[j]=arr[i];
    			arr2[j+1]=arr[n-1-i];
    			j+=2;
    		}
    		else arr2[j]=arr[n/2];
    	}
    }
    long long int ans=abs(arr2[0]-arr2[n-1]);
  //  cout<<arr[0]<<" ";
    for(int i=1;i<n;i++)
    {
    //	cout<<arr2[i]<<" ";
    	ans+=abs(arr2[i-1]-arr2[i]);
    }
   // cout<<"\n";
    cout<<ans;
    }
    return 0;
}
