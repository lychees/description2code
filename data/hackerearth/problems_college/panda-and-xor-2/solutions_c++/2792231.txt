#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    long long int a,b,s,count;
    while(t--)
    {
    	cin>>a>>b;
      s=a^b;
      count=0;
     while(s>0)
     {
     	count += s & 1;
     	s=s/2; 
     }
      if(count%2==0)
       cout<<"YES"<<endl;
      else
       cout<<"NO"<<endl;
    }
    return 0;
}
