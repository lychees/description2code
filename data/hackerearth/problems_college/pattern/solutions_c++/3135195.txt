#include <iostream>

using namespace std;

int main()
{
    int n;
    cin>>n;
    for(int i=1;i<=n;i++)
    {
    	cout<<"*";
    }
    cout<<endl;
    for(int i=1;i<=n-2;i++)
    {
    	for(int j=1;j<=n-i;j++)
    	{
    		if(j!=n-i)
    		{
    			cout<<" ";
    		}
    		else
    		{
    			cout<<"*";
    		}
    	}
    	cout<<endl;
    }
    for(int i=1;i<=n;i++)
    {
    	cout<<"*";
    }
    return 0;
}
