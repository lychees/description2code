#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    vector<long int>vec;
    while(t--)
    {
    	int temp;
    	cin>>temp;
    	vec.push_back(temp);
    	sort(vec.begin(),vec.end());
    }
    	int a=0,b=0;
    	for(int i=1;i<vec.size();i++)
    	{
    		if(vec[i]!=vec[i-1])
    		{
    			if(i%2==1)
    			{
    				a++;
    			}
    			else
    			{
    				b++;
    			}
    		}
    	}
    	if(a==b)
    	cout<<"Bob"<<endl;
    	else if(a>b)
    	cout<<"Alice"<<endl;
    	else 
    	cout<<"Bob"<<endl;
  
}
