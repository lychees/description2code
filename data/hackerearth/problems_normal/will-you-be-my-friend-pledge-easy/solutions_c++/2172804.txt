#include <bits/stdc++.h>

using namespace std;

int main()
{
	int x, n, i, j, ans;
	scanf("%d",&x);
	scanf("%d",&n);
	int a[n];
	for (i = 0; i < n; i++) {
		scanf("%d",&a[i]);
	}
	ans = 0;
	for (i = 0; i < n; i++) {
		if (__gcd(x, a[i]) != 1) {
			ans++;
		}
	}
	printf("%d\n",ans);
	
	return 0;
}
