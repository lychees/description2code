import fractions
from math import sqrt; 
from itertools import count, islice

def isPrime(n):
    return n > 1 and all(n%i for i in islice(count(2), int(sqrt(n)-1)))

def phi(n):
    amount = 0

    for k in range(1, n + 1):
        if fractions.gcd(n, k) == 1:
            amount += 1

    if isPrime(amount):
        print "TRUE"
    else:
        print "FALSE"


u = raw_input()
data = []
for x in range(int(u)):
	ri = raw_input()
	data.append(int(ri))

for n in data:
	phi(n)
