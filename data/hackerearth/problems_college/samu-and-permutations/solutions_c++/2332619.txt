ll madar(ll fuck){

	while(fuck!=0){

		fuck>>=1;
	}

}
ll recurs(ll ind, ll masking, ll pich)
{
	ll temp = 0;
	
	if (ind == n) {
		return 0;
	}
	
	
	if (dp[masking][pich] != -1) {
		return dp[masking][pich];
	}
	
	if (ind != 0) {
		
		for (ll i = 0; i < n; i++) {
			
			ll ai = 1 << i;
			ai = masking & ai;
			
			if (ai == 0) {
				
				ll ct = madar(masking);
				ct += 1;
				
				ll tempo1 = masking;
				ll tempo2 = 1 << i;
				
				tempo1 = tempo1 | tempo2;
				
				//cout << ct << endl;
				
				ll an = abs(arr[pich] - arr[i]) * ct + recurs(ind + 1, tempo1, i);
				//cout << an << endl;
				
				temp = max(temp, an);
				
			}
			
		}
		
		
		
	}
	
	else {
		
		for (ll i = 0; i < n; i++) {
			ll tempo1 = masking;
			ll tempo2 = 1 << i;
			
			tempo1 = tempo1 | tempo2;
			//cout << "aya " << tempo1 << endl;
			temp = max(temp, recurs(ind + 1, tempo1, i));	
		}
		
	}
	
	dp[masking][pich] = temp;
	return dp[masking][pich];
}
