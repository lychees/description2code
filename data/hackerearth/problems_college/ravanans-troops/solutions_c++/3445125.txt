#include<bits/stdc++.h>

using namespace std;

int test(int n) {
    while(n%4 == 0) n/=4;
    if((n-7)%8 == 0) return 1;
    return 0;
}

vector <int> v;
int mark[1000000];

void sieves(int n) {
    int i,j;
    for(i = 2; i <= n; i++) {
        if(mark[i] == 0) {
            if(i >= 3 && (i-3)%4 == 0)
                v.push_back(i);
            for(j = i+i; j < n; j+=i) {
                mark[j] = 1;
            }
        }
    }
}

int test2(int n) {
    for(int i = 0; i < v.size() ; i++) {
        int cnt = 0;
        while(n%v[i] == 0) {
            n /= v[i];
            cnt++;
        }
        if(cnt % 2 != 0) return 0;
    }
    return 1;
}

int main() {
    sieves(1000000);
    long long int t,i,j,n;
    cin>>t;
    while(t--) {
        cin>>n;
        int x = sqrt(n);
        if(n == 1) {
            cout<<"1\n";
            continue;
        }
        if(n == 2) {
            cout<<"2\n";
            continue;
        }
        if(x*x == n) {
            cout<<"1\n";
            continue;
        } else if(test2(n) == 1) {
            cout<<"2\n";
            continue;
        } else {
            if(test(n) == 1) {
                cout<<"4\n";
            } else cout<<"3\n";
        }
    }
    return 0;
}
