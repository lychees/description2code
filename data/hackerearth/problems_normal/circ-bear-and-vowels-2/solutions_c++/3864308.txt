#include<bits/stdc++.h>
using namespace std;
bool consont(char let)
{
	if(let!='a' && let!='e' && let!='i' && let!='o' && let!='u' && let!='y')
		return true;
	return false;
}
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		string word;
		cin>>word;
		int cons_cnt=0;
		int vowel_cnt=0;
		for(int i=0;i<word.size();++i)
		{
			char let=word[i];
			if(let=='a' || let=='e' || let=='i' || let=='o' || let=='u' || let=='y')
				vowel_cnt++;
		}
		cons_cnt=word.size()-vowel_cnt;
		bool flag=false;
		for(int i=1;i<word.size()-1;++i)
		{
			
			if(consont(word[i-1]) && consont(word[i]) && consont(word[i+1]))
			{

				flag=true;
				break;
			}
		}
		if(flag || (cons_cnt>vowel_cnt))
			puts("hard");
		else puts("easy");
	}
	return 0;
}
