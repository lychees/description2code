#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	int k,i,t,n,w,a[1002],turns=0,left,mid,right;
	bool vis[1002];
	scanf("%d",&t);
	while(t--)
	{
		turns=0;
		scanf("%d%d",&w,&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			vis[i]=0;
		}
		sort(a,a+n);
		a[n]=9999999;
		for(i=0;i<n;i++)
		{
			right=n;left=0;
			while(right>left)
			{
				mid=left+(right-left)/2;
				if(a[mid]<w-a[i])
					left=mid+1;
				else
					right=mid;
			}
			k=left;
			while( a[k]>(w-a[i]) || (vis[k]))
			{
				k--;
			}
			if(k>i && (a[k]+a[i]<=w))
			{
				vis[i]=1;
				vis[k]=1;
				turns++;
			}
		}
		for(i=0;i<n;i++)
		{
			if(!vis[i] && a[i]<=w)
				turns++;
		}
		printf("%d\n",turns);
	}
	return 0;
}
