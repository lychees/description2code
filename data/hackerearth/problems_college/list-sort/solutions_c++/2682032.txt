#include<bits/stdc++.h>
using namespace std;

struct name{
    string f;
    int imp;
    string l;
};
vector<name> v;
int n; string s;

char c(int n){ return n+'a';}
bool cmp(name a,name b){ if(a.l!=b.l) return a.l<b.l;else{return a.imp>b.imp;}}

void solve(){
    v.clear(); cin>>n;
    getline(cin,s);
    for(int i=1;i<=n;++i){
        getline(cin,s);
        name tmp;
        string s1="";
        for(int j=0;j<s.size();++j) if(s[j]==' ') s1=""; else s1+=c(tolower(s[j]));
        tmp.l=s1;
        tmp.imp=i;
        tmp.f=s;
        v.push_back(tmp);
    }
    sort(v.begin(),v.end(),cmp);
    for(int i=0;i<v.size();++i){
        cout<<v[i].f<<endl;
    }
}

int main(){
    int t; cin>>t;
    //getline(cin,s);
    while(t--){
    	solve();
    } 
    return 0;
}
