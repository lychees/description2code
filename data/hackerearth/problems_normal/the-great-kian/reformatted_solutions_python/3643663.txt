

n = int(raw_input())
line = raw_input()
a = line.split(" ")
n1 = 0
n2 = 0
n3 = 0
n1turn = True
n2turn = False
n3turn = False
for i in range(n):
    if n1turn:
        n1 += int(a[i])
        n2turn = True
        n1turn = False
    elif n2turn:
        n2 += int(a[i])
        n3turn = True
        n2turn = False
    elif n3turn:
        n3 += int(a[i])
        n1turn = True
        n3turn = False

print str(n1)+" "+str(n2)+" "+str(n3)
