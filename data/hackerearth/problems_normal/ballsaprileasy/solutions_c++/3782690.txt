#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

long long int max(long long int x, long long int y) {
	return (x>y?x:y);
}

int main()
{
    long long int n;
    cin>>n;
    long long int a[n];
    for(int i=0;i<n;i++) {
    	cin>>a[i];
    }
    sort(a, a+n);
    long long int sum=0;
    long long int l = 1;
    
    for(long long int i=0;i<n;i++) {
    //	long long int sq = layer\*l;
    	
    	while(l*l < a[i]) {
    		sum+=l*l;
    		l++;
    	}
    	sum += (l*l - a[i]);
    	l++;
    }
    cout<<sum;
    return 0;
}
