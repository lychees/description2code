#include <iostream>
using namespace std;

int main()
{
    int T;
    cin >> T;
    while(T--)
    {
    	long int n,count=0;
    	cin >> n;
    	while(n!=1)
    	{
    		if(n%3==0)
    		{
    			n/=3;
    			count++;
    		}
    		else if(n%2==0)
    		{
    			n/=2;
    			count++;
    		}
    		else
    		{
    			n-=1;
    			count++;
    		}
    	}
    	cout << count << endl;
    }
    return 0;
}
