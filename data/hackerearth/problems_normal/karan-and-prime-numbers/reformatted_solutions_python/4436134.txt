def mysieve(n):
    temp = list()
    temp.append(2)
    for i in xrange(3, n, 2):
        if prime[i]:
            temp.append(i)
        if i*i <= n:
            for j in xrange(i*i, n, i):
                prime[j] = 0
    return temp


n = 1000001
prime = [1]*n
final = mysieve(n)


for _ in xrange(input()):
    a, b = map(int, raw_input().split())
    ans = 0
    for val in final:
        if val >= a and val <= b:
            ans += val
    print ans
