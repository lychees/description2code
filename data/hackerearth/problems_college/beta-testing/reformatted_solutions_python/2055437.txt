names = dict()
n = int(raw_input())
i = 0
for name in raw_input().split():
    if name not in names:
        names[name] = [1, i]
    else:
        names[name] = [names[name][0]+1, i]
    i += 1

tmp = sorted(names.items(), key=lambda x: x[1][1])
tmp = sorted(tmp, key=lambda x: x[1][0], reverse=True)
for i in tmp:
    print i[0], i[1][0]
