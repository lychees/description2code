#include <iostream>
using namespace std;

int main()
{
    long long int t,n,p1,p2,h;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	h=0;
    	if(n==1)
    	{
    		cout<<0<<" "<<1<<"\n";
    		continue;
    	}
    	p1=1;
    	p2=n;
    	while(p2)
    	{
    		
    		p1*=2;
    	    p2/=2;
    	}
    	p2=n;
    	while(p2>1)
    	{
    		if(p2%2!=0)
    		{
    			h=1;
    			break;
    		}
    		p2/=2;
    	}
    	if(!h)
    		p1/=2;
    		cout<<p1-1<<" ";
    		p1/=2;
    		cout<<(n-p1)*2<<"\n";
    }
    return 0;
}
