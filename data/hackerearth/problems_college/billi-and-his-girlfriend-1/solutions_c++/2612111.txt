#include<bits/stdc++.h>
using namespace std;

long long int square(long long int x, long long n ,long long int max){
    if (x==1) return 1;
    if (n==1)  return  x ;
    else if (n%2==0) return square((x*x)%max,  n/2,max)%max;
    else if(n%2==1)return (x*square((x*x)%max, (n-1)/2,max)%max)%max;
}
    
int main(){
	long long int t;cin>>t;
	while(t--){
		long long int a,b,sum=0;
		int k;
		cin>>a>>b>>k;
		
		long long int ans = square(a,b,pow(10,k));
        while(k--){
          	sum=sum+ans%10;
          	ans/=10;
		}
		
	    cout<<sum<<endl;
	}
}
