#include<bits/stdc++.h>
#include<stdio.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--){
		int  a,b,c,r,q;
		char s[100000]={'\0'};
		scanf("%d%d%d",&a,&b,&c);
		q = a/b;
		r = a%b;
		if(c!=0){
		for(int i=0;i<c;i++){
			int x;
			r = r*10;
			x = r/b;
			r = r%b;
			s[i] = x + '0';
		}
		cout<<q<<"."<<s<<endl;
		}
		else
		cout<<a/b<<endl;
		//cout<<setprecision(c)<<a<<endl;
	}
    return 0;
}
