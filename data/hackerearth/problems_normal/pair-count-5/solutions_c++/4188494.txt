#include <bits/stdc++.h>
#define lli long long
#define MAX 1000005
using namespace std;
int A[MAX];
vector <int> v[MAX];
int cnt[MAX];
void pre()
{
	for ( int i = 1; i <= 1000; i++ ) {
		for ( int j = i; j <= 1000000; j += i ) {
			v[j].push_back(i);
			int nxt = j/i;
			if ( j/i != i && j/i > 1000 ) v[j].push_back(j/i);
		}
	}
	return;
}
//Code for FAST INPUT :)
template <typename T>
inline void fi(T *a)
{
    register char c=0;
    while (c<33) c=getchar_unlocked();
    *a=0;
    int tmp = 0;
    while (c>33)
    {
        if ( c == 45 ) tmp = 1;
        else *a=*a*10+c-'0';
        c=getchar_unlocked();
    }
    if ( tmp == 1 ) *a = 0-(*a);
}
int main()
{
	pre();
	int n,d,zero = 0;
	lli ans = 0;
	fi(&n), fi(&d);
	assert(n >= 1 && n <= 1000000);
	assert(d >= 1 && d <= 1000000);
	
	for ( int i = 0; i < n; i++ ) {
		fi(&A[i]);
		assert(A[i] >= 1 && A[i] <= 1000000000);
		if ( A[i] >= d ) A[i] %= d;
	}
	
	
	for ( int i = 0; i < n; i++ ) {
		if ( A[i] == 0 ) ans += i;
		else {
			int gcd_val = __gcd(d,A[i]);
			ans += zero;
			ans += cnt[d/gcd_val];
		}
		for ( int j = 0; j < (int)v[A[i]].size(); j++ ) cnt[v[A[i]][j]]++;
		zero += (A[i] == 0);
	}
	printf("%lld\n", ans);
	
	return 0;
}
