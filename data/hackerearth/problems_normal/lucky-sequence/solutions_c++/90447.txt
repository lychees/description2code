#include<stdio.h>
#include<string.h>
long long int pow(int n)
{
	long long int p=1,i;
	for(i=0;i<n;i++)
		p=p*2;
	return p;
}
int main()
{
    int t;
    long long int num,temp,n,i,j,ans,k,rem;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld",&num);
        temp=0,n=0;
        while(temp<num)
        {
            n=n+1;
            temp=pow(n+1)-2;
            //printf("%lld ",temp);
        }
        temp=pow(n)-2;
        //printf("%lld %lld\n",n,temp);
        ans=num-temp-1;
        char str[100000];
        int k=-1;
        while(ans>0)
        {
            rem=ans%2;
            k=k+1;
            str[k]=rem+'0';
            ans=ans/2;
        }
        j=n-k-1;
        for(i=1;i<=j;i++)
            printf("4");
        for(i=k;i>=0;i--){
            if(str[i]=='0')
                printf("4");
            else
                printf("7");
        }
        printf("\n");
    }
    return 0;
}
