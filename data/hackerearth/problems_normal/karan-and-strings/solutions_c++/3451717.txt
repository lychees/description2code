#include <iostream>
#include<cstring>
#include<cstdio>
using namespace std;

int main()
{
	int t;
	cin>>t;
	string a;
	char temp;
	int l;
	while(t--)
    {
        cin>>a;
        temp=a[0];
        l=a.length();
        for(int i=0;i<l;i++)
        {
            if(temp!=a[i])
            {
                cout<<temp;
                temp=a[i];
            }
        }
        cout<<temp<<"\n";
	}
    return 0;
}
