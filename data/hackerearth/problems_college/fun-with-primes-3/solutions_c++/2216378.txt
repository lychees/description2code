#include "iostream"
using namespace std;
long long n,w;
string a;
int main()
{
	ios_base::sync_with_stdio(0);
	cin>>n;
	
	while (n--)
	{
		cin>>a;
		w=0;
		for (int i=0; i<a.size(); i++) w+=(a[i]-48);
		bool ok=true;
		for (int i=2; i<w; i++)
		{
			if (w%i==0)
			{
				ok=false;
				break;
			}
		}
		if (w!=1 && ok) cout<<"YES\n";
		else cout<<"NO\n";
	}

	return 0;
}
