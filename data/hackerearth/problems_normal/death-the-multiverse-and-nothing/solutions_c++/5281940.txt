#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int nSize, temp;
    cin >> nSize;
    
    vector <int> arr;
    
    for (int idx = 0; idx < nSize; idx++) {
    	cin >> temp;
    	arr.push_back(temp);
    }
    
    for (int idx = 0; idx < nSize; idx++) {
    	cout << arr[idx] - 1 << endl;
    }
    
    return 0;
}
