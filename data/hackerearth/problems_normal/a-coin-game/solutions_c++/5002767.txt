const int MAXN = 1111111;
int loose[MAXN];

void calc() {
	for (int i = 0, d = 0; i < MAXN - 4; i ++) {
		if(loose[i]) continue;
		loose[i] = i + d;
		if(i + d < MAXN - 4) loose[i + d] = i;
		d ++;
	}
}

int main() {
	ios_base::sync_with_stdio(false);
	
	calc();
	int t; cin >> t;
	while (t --) {
		int a,b; cin >> a >> b;
		if(loose[a] == b) puts("Don't Play");
		else puts("Play");
	}
	
	return 0;
}
