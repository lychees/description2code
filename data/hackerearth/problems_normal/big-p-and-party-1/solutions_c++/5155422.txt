#include <iostream>
using namespace std;

int a[1005][1005];
int A,B;

int main() {
	// your code goes here
	cin>>A>>B;
	
	int src,dest;
	
	for(int i=0;i<A;i++){
		for(int j=0;j<A;j++){
			if(i==j)
				a[i][j]=0;
			else
				a[i][j]=99999;
		}
	}
	
	for(int i=0;i<B;i++){
		cin>>src>>dest;
		a[src][dest]=1;
		a[dest][src]=1;
	}

	for(int k=0;k<A;k++){
		for(int i=0;i<A;i++){
			for(int j=0;j<A;j++){
				if(a[i][j] > a[i][k]+a[k][j])
					a[i][j]=a[i][k]+a[k][j];
			}
		}
	}	
	
	for(int i=1;i<A;i++){
		if(a[0][i]>=99999)
			cout<<"INF"<<endl;
		else
			cout<<a[0][i]<<endl;
	}
	
	return 0;
}
