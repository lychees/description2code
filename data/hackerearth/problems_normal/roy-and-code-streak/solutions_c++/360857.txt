#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
#include<algorithm>

using namespace std;

int main()
{
	int t, n, qid, sol, gmax, max;
	cin>>t;
	while(t--)
	{
		int done[1000010] = {0};
		max = 0;
		gmax = 0;
		scanf("%d", &n);
		while(n--)
		{
			scanf("%d%d", &qid, &sol);
			if(!sol)
			{
				if(max > gmax)
					gmax = max;
				max = 0;
			}
			else if(!done[qid])
			{
				done[qid] = 1;
				max++;
			}
		}
		if(max > gmax)
			gmax = max;
		printf("%d\n", gmax);
	}
	return 0;
}
