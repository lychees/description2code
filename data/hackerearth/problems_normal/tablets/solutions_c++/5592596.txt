#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	scanf("%d",&n);
	long long tab[n],score[n],sum = 0;
	int i;
	for(i=0;i<n;i++) {
		scanf("%lld",&score[i]);
		if(i==0) {
			tab[i] = 1;
		}
		else {
			if(score[i] > score[i-1])
				tab[i] = tab[i-1] + 1;
			else
				tab[i] = 1;
		}
	}
	for(i=n-2;i>=0;i--) {
		if(score[i] > score[i+1]) {
			if(tab[i] <= tab[i+1])
				tab[i] = tab[i+1] + 1;
		}
	}
	for(i=0;i<n;i++)
		sum += tab[i];
	printf("%lld",sum);	
}
