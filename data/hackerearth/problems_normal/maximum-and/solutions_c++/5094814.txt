#include<bits/stdc++.h>
typedef long long int ll;
using namespace std;
int main()
{
    ll t,a,b,ans;
    cin>>t;
    while(t--)
    {
        cin>>a>>b;
        if(b-a==1)
            ans=a&b;
        else
        {
            if(b%2==0)
                ans=b-2;
            else
                ans=b-1;
        }
        cout<<ans<<endl;
    }
}
