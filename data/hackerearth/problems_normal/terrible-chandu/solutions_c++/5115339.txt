#include <iostream>
using namespace std;

int main()
{
	int T;
	cin >> T;
	while(T--){
		string s;
		string r;
		cin >> s;
		int size = s.size();
		for (int i = 0; i < size; ++i)
		{
			r[size - 1 -i] = s[i];
			// cout << r[size - 1 - i] << endl;
		}
		r[size] = '\0';
		for (int i = 0; i < size; ++i)
		{
			cout << r[i];
		}
		cout << endl;
		

	}
    return 0;
}
