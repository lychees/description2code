#include<bits/stdc++.h>
using namespace std;
#define ll long long
const long long NEG = (long long) 1e16 + 5;
int n,k;
vector <int> v;
long long  dp [10005];


ll solve (int idx) {
    if (idx >= n) return 0;
    if (dp [idx] != -1) return dp [idx];
    ll tmp = -NEG;
    tmp= max (tmp, solve (idx + 1));
    tmp = max (tmp, v [idx] + solve (idx + k + 1));
    return dp [idx] = tmp;
}

int main () {

    int t,x;
   cin>>t;
    while (t--) {
        scanf ("%d %d", &n, &k);
        memset (dp, -1, sizeof (dp));
        v.clear ();
        for (int i = 0; i < n; i++) {
          
            scanf ("%d", &x);
            v.push_back (x);
        }
        long long ans = solve (0);
        printf ("%lld\n", ans);
    }

    return 0;
}
