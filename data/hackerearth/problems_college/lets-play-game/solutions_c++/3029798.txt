#include<iostream>

using namespace std;

int main()
{
	long long int n,k,t,i,temp;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		long long int ai[n],bi[n];
		for(i=0;i<n;i++)
			cin>>ai[i];
		/*for(i=0;i<n;i++)
			//cin>>ai[i];*/
		for(i=0;i<n;i++)
			cin>>bi[i];
		/*for(i=0;i<n;i++)
			cin>>bi[i];*/
		long long int max=0;
		for(i=0;i<n;i++)
		{
			temp=(k/ai[i])*bi[i];
			if(temp>max)
				max=temp;
		}
		cout<<max<<"\n";
	}
	return 0;
}
