#include <bits/stdc++.h> 

using namespace std ;

int a[10]={0};


int main()
{
	int t;
	cin >> t; 
	int temp;
	int n,count =0;
	int k;

	while (t--)
	{
		for(k=0;k<10;k++)
			a[k]=0;

		cin >> n;

		while ( n-- )
		{
			cin >> temp;
			temp=temp%10;

		

			if(a[temp]==1)
			{
				count ++ ;	
			}
			else
				a[temp]=1;

		}


		cout << count << endl ;
		count =0;




	}
	return 0;
}
