#include <bits/stdc++.h>
using namespace std;
int nxt[100005][26];
int temp[26];
int main()
{
    string s,t;
    cin>>s;
    for(int i=0;i<26;++i)
    	temp[i]=INT_MAX;
    for(int i=s.length();i>0;--i){
    	for(int j=0;j<26;++j)
    		nxt[i][j]=temp[j];
    	temp[s[i-1]-'a']=i-1;
    }
    for(int j=0;j<26;++j)
    	nxt[0][j]=temp[j];
    int n,p;
    cin>>n;
    bool flag;
    for(int i=0;i<n;++i){
    	cin>>t;
    	flag=true;
    	p=0;
    	for(int i=0;i<t.length();++i){
    		p=nxt[p][t[i]-'a'];
    		//cout<<p<<" ";
    		if(p==INT_MAX && i<t.length()){
    			flag=false;
    			break;
    		}
    	}
    	if(flag)
    		cout<<"Yes\n";
    	else
    		cout<<"No\n";
    }
    return 0;
}
