def function(var1, var2, var3):
    i = var3-1
    j = var3-1
    m = 0
    while (i >= 0 and j >= 0):
        while (i >= 0 and j >= 0 and var2[j] >= var1[i]):
            m = max(m, j-i)
            i = i-1
        j = j-1
        while (i >= 0 and j >= 0 and var2[j] < var1[i]):
            j = j-1
    return m


t = input()
for i in range(t):
    N = input()
    l1 = map(int, raw_input().split(" "))
    l2 = map(int, raw_input().split(" "))
    answer = function(l1, l2, N)
    print answer
