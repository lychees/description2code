#include <iostream>
#include <vector>
#include <math.h>
#include <iomanip>
# define M_PI 3.14159265358979323846
using namespace std;

double volume(double l,double b, double h){
    return (M_PI * h * (l * l + l * b + b * b) / 3);
}

double hcalc(double height, long long count){
    double base = count * 0.001;
    return sqrt(height * height - base * base);
}

int main(){
    double base_radius;
    cin >> base_radius;
    double top_radius = base_radius;
    double initital_height;
    cin >> initital_height;
    double h = initital_height;
    long long k = 0;

    double v = volume(base_radius, top_radius, h);

    top_radius += 0.001;
    ++k;
    h = hcalc(initital_height, k);
    while(volume(base_radius, top_radius, h) > v){
        v = volume(base_radius, top_radius, h);
        top_radius += 0.001;
        ++k;
        h = hcalc(initital_height, k);

    }
    cout << fixed << setprecision(2) << floor(v*100+0.5)/100;
    cin >> base_radius;
}
