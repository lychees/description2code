def solve(s1, s2, k):
	count = 0
	for i in xrange(len(s2)):
		if s1[i] != s2[i]:
			count += 1
	
	if count > k:
		return False
	if count == k or (k-count) % 2 == 0:
		return True
	return False

T = input()
for i in xrange(T):
	s1, s2, N = raw_input().strip().split(" ")
	N = int(N)
	if solve(s1, s2, N):
		print "Yes"
	else:
		print "No"
