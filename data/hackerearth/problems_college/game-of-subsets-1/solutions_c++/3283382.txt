#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define mo 1000000007
long long int ipow(long long int base,long long int exp)
{
    long long int result = 1;
    while (exp)
    {
        if (exp & 1)
            result =(result* base)%mo;
        exp >>= 1;
        base = (base*base)%mo;
    }

    return result;
}

{
    long long int t,n,i,s;
