#include<cstdio>
#include<cstring>
#include<iostream>
#define N 310
using namespace std;

inline int Read()
{
	register int c=getchar();
	int x=0;
	for(;(c<48 || c>57);c=getchar());
	for(;c>47 && c<58;c=getchar())
		x=(x<<1)+(x<<3)+c-48;
	return x;
}

string app(string a)
{
	string ans="";
	int len=a.length();
	for(int i=0;i<310-len;i++)
		ans+="0";
	ans+=a;
	return ans;
}

string add(string s1,string s2)
{
	string ans="";
	for(int i=0;i<N;i++)
		ans+="0";
	int carry=0;
	for(int i=N-1;i>=1;i--)
	{
		int tmp=s1[i]-'0'+s2[i]-'0'+carry;
		int rem=tmp%10;
		carry=tmp/10;
		ans[i]=rem+'0';
	}
	return ans;
}

int main()
{
	int t=Read();
	t--;
	string s1,s2,tmp;
	cin>>tmp;
	s1=app(tmp);
	while(t--)
	{
		cin>>tmp;
		s2=app(tmp);
		s1=add(s1,s2);
	}
	int s=0;
	for(int i=0;i<s1.length();i++)
		s+=s1[i]-'0';
	while(s>9)
	{
		t=0;
		while(s)
		{
			t+=s%10;
			s/=10;
		}
		s=t;
	}
	cout<<s;
	return 0;
}
