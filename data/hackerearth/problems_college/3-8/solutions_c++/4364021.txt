#include <iostream>
#include <bits/stdc++.h>
using namespace std;
map <char,int> mp;
int main()
{
    int t,l,i;
    char ch='A';
    string s,ls,ans;
    for(i=1;i<=26;i++)
    {
    	mp[ch]=i;
    	//cout<<ch<<" "<<mp[ch]<<endl;
    	ch++;
    }
	cin>>t;
	while(t--)
	{
		cin>>s;
		l=s.size();
		ls="";
		ans="";
		ls+=s[0];
		for(i=1;i<l;i++)
		{
			if(mp[ls[0]]<=mp[s[i]])
			{
				ans=ls;
				ls="";
				ls+=s[i];
				ls+=ans;
			}
			else
			{
				ls+=s[i];
				//ans="";
				//cout<<ls<<endl;
			}
		}
		cout<<ls<<endl;
	}
    return 0;
}
