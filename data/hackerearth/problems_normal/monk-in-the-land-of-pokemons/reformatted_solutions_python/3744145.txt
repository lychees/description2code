T = int(raw_input())
for i in range(T):
    N = int(raw_input())
    A = {}
    count = 0
    for j in range(N):
        a, b = raw_input().split()
        a = int(a)
        b = int(b)
        if a not in A:
            A[a] = 1
        else:
            A[a] += 1

        # print j, ' : ',A,' ',a,' ',b

        if b in A and A[b] > 0:
            A[b] -= 1
        else:
            count += 1
    print count
