allens = [1,3,7,15]
for x in range(4, 51):
    allens.append(allens[x-1] * 2 + 1)


for _ in range(input()):
    n, k = map(int, raw_input().split())
    if k > allens[n-1]:
        print "-1"
    else:
        while k%2 == 0:
            k /= 2
        if (k-3)%4 == 0:
            print "C"
        else:
            print "G"
