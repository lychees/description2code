#include "bits/stdc++.h"
using namespace std;
long long abs(long long a,long long b)
{
	return a-b > 0 ? a-b : b-a;
}
long long func(long long *arr,long long n)
{
	long long ans=0;
	for(long long i=0;i<n-1;i++)
	{
		ans=ans+abs(*(arr+i),*(arr+i+1));
	}
	ans=ans+abs(*(arr+0),*(arr+n-1));
	return ans;
}
int main()
{
	long long n;
	cin >> n;
	
	long long arr[n],brr[n];
	for(long long i=0;i<n;i++)
		cin >> arr[i];
	
	long long brrC=0;
	sort(arr,arr+n);
	long long i=0,j=n-1;
	while(brrC<n)
	{
		if(brrC%2==0)
		{
			brr[brrC++]=arr[i];
			i++;
		}
		else
		{
			brr[brrC++]=arr[j];
			j--;
		}
	}
	long long ans1=func(brr,n);
	brrC=0;
	i=0,j=n-1;
	while(brrC<n)
	{
		if(brrC%2==0)
		{
			brr[brrC++]=arr[j];
			j--;
		}
		else
		{
			brr[brrC++]=arr[i];
			i++;
		}
	}
	cout << max(ans1,func(brr,n)) << endl;
	return 0;
}
