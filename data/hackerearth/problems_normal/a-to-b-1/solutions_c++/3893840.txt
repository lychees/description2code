#include <iostream>
#include <stdio.h>
#define MOD 1000000007LL
typedef long long ll;
using namespace std;

ll power(ll a, ll b)
{
	ll base = a%MOD, power = 1ll, c = 1ll;
	while(power<=b)
	{
		if(power&b)
			c = (c * base) % MOD;
		base = (base * base) % MOD;
		power<<=1;
	}
	return c;
}

int main()
{
    int n, q, id, qtype, zero_cnt = 0;
    scanf("%d", &n);
    ll *A = new ll[n];
    ll B = 1ll, v;
    for(int i=0; i<n; i++)
	{
		scanf("%lld", &A[i]);
		if(A[i]==0)
			zero_cnt++;
		else
			B = (B * A[i]) % MOD;
	}
	scanf("%d", &q);
	while(q--)
	{
		scanf("%d %d", &qtype, &id);
		if(qtype)
		{
			int zero = A[id-1]==0?1:0;
			if(zero_cnt - zero > 0)
				printf("0\n");
			else
				printf("%lld\n", (B * power((A[id-1]==0)?1:A[id-1], MOD-2)) % MOD);
		}else{
			scanf("%lld", &v);
			if(v==0 && A[id-1]>0)
				zero_cnt++;
			else if(v>0 && A[id-1]==0)
				zero_cnt--;
			if(A[id-1])
				B = (B * power(A[id-1], MOD-2)) % MOD;
			A[id-1] = v % MOD;
			if(v)
				B = (B * v) % MOD;
		}
	}
    return 0;
}
