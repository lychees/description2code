t = input()
for test_case in xrange(t):
    n, m, a, b, c = map(int, raw_input().split())
    if n != m:
        if n > m:
            remain = n-m
            print max((c*(m+m)), ((a+b)*m)) + remain*a
        else:
            remain = m-n
            print max((c*(n+n)), ((a+b)*n)) + remain*b
    else:
        print max((c*(n+m)), ((a*n)+(b*m)))
