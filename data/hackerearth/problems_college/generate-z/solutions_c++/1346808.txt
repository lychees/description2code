#include<iostream>
using namespace std;
int main()
{
    int t,n,i,j;
    cin>>n;
    for(i=0;i<n;i++)
        cout<<"*";
    cout<<endl;
    j=n-2;
    for(i=0;i<n-2;i++)
    {
        for(t=0;t<j;t++)cout<<" ";
        cout<<"*\n";
        j--;
    }
    for(i=0;i<n;i++)
        cout<<"*";
    cout<<endl;
}
