#include <bits/stdc++.h>

using namespace std;

int main()
{
	int i,j=0,n,t,k;
	
	string s,r;
	
	cin >>t;
	
	while(t--)
	{
		cin >> s;
		int temp =-1;
		
		n = s.length();
		
		for(i =0; i<n-1; i++){
			if(s[i] < s[i+1])
				temp =i;
		}
	//	cout <<"temp = "<< temp;
		
		if(temp == -1){
			cout << "no answer" <<endl;
			continue;
		}
		
		
		for( k =temp+1; k<n-1; k++){
			if(s[k] > s[temp])
				i = k;
		}
		
		swap(s[temp], s[i]);
		
		i =1;
		j =1;
		
		while((temp+i) < (n-j))
		{
			swap(s[temp+i],s[n-j]);
			i++;
			j++;
		}
		
		
		
		cout<< s <<endl;
	}
	return 0;
}
