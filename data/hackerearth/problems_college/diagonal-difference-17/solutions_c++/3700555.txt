#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int A[n][n],sum=0,sum1=0;
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			cin>>A[i][j];
			if(i==j)
				sum1=sum1+A[i][j];
			if(i+j==n-1)
				sum=sum+A[i][j];
		}
		cout<<endl;
	}
	int diff=sum1-sum;
	cout<<endl;
	if(diff>=0)
    cout <<diff;
    else
    cout<<-diff;
    return 0;
}
