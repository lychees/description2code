def factors(n):    
    return set(reduce(list.__add__, 
                ([i, n//i] for i in range(1, int(n**0.5) + 1) if n % i == 0)))

a,j=[3],3
for i in xrange(1,1000):
    a.append(a[i-1]+j)
    j+=1

    
for _ in xrange(input()):
    x=input()
    for k in a:
        if(len(factors(k))>x):
            print k
            break
