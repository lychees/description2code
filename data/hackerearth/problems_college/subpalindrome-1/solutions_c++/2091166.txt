#include<bits/stdc++.h>
using namespace std;
# define ll long long
# define start cin.sync_with_stdio(false);
# define done return 0;
# define MIN -10000000
# define MAX 10000000
int main()
{
    start
    int t,i,a[55][55],p,cnt,j;
    char s[55];
    cin>>t;
    while(t--)
    {
        for(i=0;i<55;i++)
        for(j=0;j<55;j++)
        a[i][j]=0;
        cin>>s;
        p=strlen(s);
        for(i=p-1;i>=0;i--)
        {
            cnt=1;
            a[i][i]=1;
            if(i+1!=p){
            if(s[i]==s[i+1])
            {
                cnt++;
                a[i][i+1]=3;// in case aab-->a,a,aa
            }
            else
            a[i][i+1]=2;
            }
            for(j=i+2;j<p;j++)
            {
                if(s[i]==s[j])
                {
                    cnt++;
                    cnt+=a[i+1][j-1];
                }
                a[i][j]=a[i+1][j]+cnt;
            }
        }
            cout<<a[0][p-1]<<endl;
}
    done
}
