#include<iostream>
using namespace std;
int main()
{
	long a,b,minsum,maxsum,i=1,moda,modb,newa=0,newb=0;
	cin>>a>>b;
	while(a!=0)
	{
	    moda=a%10;
	    a/=10;
	    if(moda==5)
	    	moda=6;
		newa+=moda*i;
		i*=10;
	}
	i=1;
	while(b!=0)
	{
	    modb=b%10;
	    b/=10;
	    if(modb==5)
	    	modb=6;
		newb+=modb*i;
		i*=10;
	}
	maxsum=newa+newb;
	i=1;
	a=newa,b=newb;
	newa=0,newb=0;
	
	while(a!=0)
	{
	    moda=a%10;
	    a/=10;
	    if(moda==6)
	    	moda=5;
		newa+=moda*i;
		i*=10;
	}
	i=1;
	while(b!=0)
	{
	    modb=b%10;
	    b/=10;
	    if(modb==6)
	    	modb=5;
		newb+=modb*i;
		i*=10;
	}
	minsum=newa+newb;
	cout<<minsum<<" "<<maxsum;
	return 0;

}
