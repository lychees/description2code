#include <bits/stdc++.h>
using namespace std;
const int MAX = 2e5 + 5;
int arr[MAX], n, m;
int main(){
	cin >> n >> m;
	for(int i = 1; i <= n + n; i++) cin >> arr[i];
	sort(arr + 1, arr + 1 + n + n);
	int items = 0, sum = 0;
	for(int i = 1; i <= n + n; i++){
		sum += arr[i];
		if(sum <= m) items++;
	}
	items = min(items, n);
	cout << items << '\n';
}
