#include <cstdlib>
#include <iostream>
#include <string.h>
#include <stdio.h>
using namespace std;
char s[200];
int n,i,sum,a[200],maxx,imax;
int main(int argc, char *argv[])
{   gets(s);
    n=strlen(s);
    
    for(i=0;i<n;i++)
    {   sum=sum+(int)s[i];
        a[(int)s[i]]++;   
    }  
    sum=sum/n;
    if(sum%2==1)
    {   maxx=a[0];
        for(i=0;i<199;i++)
           if( a[i]>maxx)
           {   maxx=a[i];
               imax=i;
           }
        
        if(s[imax]!=' ')
           cout<<(char)imax<<endl;
        else
           cout<<" "<<endl;        
    }  
    else
       for(i=n-1;i>=0;i--)
           if(s[i]!=' ')
              cout<<s[i];
           else
              cout<<' ';
     cout<<endl;  
   
    return EXIT_SUCCESS;
}
