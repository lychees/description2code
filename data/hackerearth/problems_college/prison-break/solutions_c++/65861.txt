#include <iostream>
#include<iomanip>
#include<cmath>
using namespace std;

int main()
{
    double i,j,T,h,d,v,u,t1,d1,s;
    cin>>T;
    for(i=0;i<T;i++)
    {
    	cin>>h>>d>>v;
    	s=d*d-h*h;
    	d1=pow(s,0.5);
    	t1=(d-d1)/v;
    	u=v*d1/h;
    	cout<<fixed<<setprecision(9)<<u<<endl;
    }
    return 0;
}
