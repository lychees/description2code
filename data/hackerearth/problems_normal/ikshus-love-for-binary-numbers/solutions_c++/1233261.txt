ll arr[100][100];
ll gcd(ll a,ll b)
{
	ll c;
	while(a!=0)
	{
		c=a;
		a=b%a;
		b=c;
	}
	return b;
}
ll pow2(ll n)
{
	ll calc=1,i;
	rep(i,n)
	{
		calc*=2;
	}
	return calc;
}

			sum+=arr[i-1][j];
			arr[i][j]+=arr[i-1][j-1];
		}
		arr[i][0]=sum;
	}
