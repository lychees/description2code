#include <iostream>
#include <bits/stdc++.h>
using namespace std;
long long int a[100005],n;

int ok(long long int M){
   long long int z = a[0]%M;
   for(int i=1;i<n;++i ){
      if(a[i]%M!=z)
      return 0;
   }
   return 1;
}
int main(){
	cin>>n;
   	vector<long long int>v;
	for(int i=0;i<n;++i)
	cin>>a[i];
    long long int p = a[0]-a[1];
    if(p<0)p=-p;
    for(int d=1; d*d<=p; ++d){
    	if(p%d!= 0)continue;
	    long long int M1=d,M2=p/d;
	    if(ok(M1) && M1!=1)
	    v.push_back(M1);
	    if(ok(M2) && M2!=M1)
	    v.push_back(M2);
    }
    //cout<<v.size()<<endl;
    sort(v.begin(),v.end());
    for(int i=0;i<v.size();i++)
    printf("%lld ",v[i]);
    printf("\n");
    return 0;
}
