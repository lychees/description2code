#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;


int main()
{
	int N;
	cin>>N;
	vector<int> v;
	int tmp=N,val,i;

	
	while(tmp--)
	{
		  cin>>val;
		  v.push_back(val);
	}
	int max,sum=0;
	while(v.size()!=2)
	{
		max=1;
		for(i=2;i<=v.size()-2;i++)
		{
			if(v[i-1]*v[i+1] > v[max-1]*v[max+1])
			    max=i;
		}
		sum+=v[max-1]*v[max+1];
		v.erase(v.begin()+max);

	}
	cout<<sum;

	return 0;
	
}
