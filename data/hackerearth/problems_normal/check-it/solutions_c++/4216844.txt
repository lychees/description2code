#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long int n,m,i,a,b,c=0;
		cin>>n>>m;
		int arr[n-1];
		vector<pair<pair<int,int> ,int> > v;
		vector<int> g;
		for(i=1;i<=m;i++)
		{
			cin>>a>>b;
			v.push_back(make_pair(make_pair(a,b),i));
		}
		for(i=0;i<n-1;i++)
		cin>>arr[i];
		for(i=0;i<n-1;i++)
		{
			for(int j=0;j<m;j++)
			{
				if(v[j].second==arr[i])
				{
					g.push_back(v[j].first.first);
					g.push_back(v[j].first.second);
					break;
				}
			}
		}
		i=0;
		sort(g.begin(),g.end());
		while(i<g.size()-1)
		{
			if(g[i]!=g[i+1])
			c++;
			i++;
		}
		if(c+1==n)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
		g.clear();
		v.clear();
	}
}
