#include<bits/stdc++.h> 
using namespace std ; 
int main()
{
	int n ; 
	cin >> n ; 
	string a[100000] ; 
	 for ( int i =  0; i< n ; i++)
	 {
	 	cin >> a[i] ; 
	 }
	int y ; 
	cin >> y ; 
	string  ab[y] ; 
	char  abc[100000] ;
	int o = 0 ;  
	for ( int i = 0 ; i< y ; i++)
	{
		cin >> ab[i] ; 
	}
	
	for (int  i = 0 ; i< y ; i++)
	{
		int pq = 0 ; 
		for (int j = 0 ; j< n ; j++ )
		{
			if ( ab[i] == a[j])
			{
			//	cout << ab[i] <<" " << a[j] <<endl ;
				pq= 1 ;
				break ;  
			}
		}
		if ( pq == 0)
		{
				
			  string ty = ab[i] ; 
			  char d = ty[0]-32 ; 
			  abc[o] = d  ; 
			  o++ ; 
			 abc[o] = '.' ; 
			 o++ ; 
		}
		
	}
	for ( int i = 0 ; i< o-1 ; i++)
	{
		cout << abc[i] ;
	}
	cout << endl; 
	return 0 ; 
	
}
