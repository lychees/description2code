#define uniq(v)                     sort(all(v));v.erase(unique(all(v)),v.end())
#define clr(a)                      memset(a,0,sizeof a)
#define MAX 1000010

ll a[MAX];
ll dp[MAX];
int main(){
	int t,n;
	ios::sync_with_stdio(0);
	cin>>t;
	while(t--){
		cin>>n;
		for(int i=0;i<n;i++)
			cin>>a[i];
		dp[n] = 0; dp[n+1]=0;
		ll sum = 0;
		for(int i=n-1;i>=0;i--){
			sum += a[i];
			dp[i] = sum - min(dp[i+1],dp[i+2]);
		}
		if((sum%2 == 0) and (dp[0] == sum/2))
			cout<<"DRAW"<<endl;
		else if((sum-dp[0])<dp[0])
			cout<<"LOSE"<<endl;
		else
			cout<<"WIN"<<endl;
	}
	return 0;
}
