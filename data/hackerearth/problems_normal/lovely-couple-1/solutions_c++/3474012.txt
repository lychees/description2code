#include <iostream>
#include<vector>
using namespace std;

int gcd(int a,int b){
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
int main()
{
	
		vector <long long int> ar(5000001);
	for(long long  int i=2;i<5000001;i++)
	ar[i]=i;
	for(long long int i=2;i<=5000000;i++)
	{
	
		for(long long int j=i*i;j<=5000000;j+=i)
		ar[j]=0;
	   	
	}

long long 	int t,a,b,lcm,c;
	cin>>t;
	while(t--){c=0;
		cin>>a>>b;
		if(a>b)
	lcm=a*b/gcd(a,b);
	else
	lcm=a*b/gcd(b,a);
//	for(int i=0;i<100;i++)
//	cout<<ar[i]<<endl;
	int i=2;
	if(ar[lcm]==0){
			while(lcm>1){
		if(ar[i]==i){
		
			if(lcm%ar[i]==0){
					c++;
				while(lcm%ar[i]==0){
					
					lcm/=ar[i];
				}
			}
		}
		i++;
	}
		
	}
	else
	c=1;
    
	if(ar[c]==c && c>1)
	cout<<"Yes"<<endl;
	else
	cout<<"No"<<endl;
	}
	
    return 0;
}
