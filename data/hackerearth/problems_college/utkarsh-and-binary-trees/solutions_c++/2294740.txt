#include <iostream>
using namespace std;
long long MOD = 1000000007;
long long dp[510][510];

long long solve(int N, int H)
{    
    for(int i=0; i<=H; i++)
        dp[0][i] = 1;
    for(int i=1; i<=N; i++)
        for(int j=1; j<=H; j++)
            for(int k=1; k<=i; k++)
                dp[i][j] = (dp[i][j] + dp[k-1][j-1]*dp[i-k][j-1])%MOD;

    return dp[N-1][H-1];
}
int main()
{
    int N,H;
    cin>>N>>H;
    long long ans = 0;
    solve(N,H);
    for(int k=1;k<=N;k++)
        ans = (ans + ((k*dp[k-1][H-1])%MOD*dp[N-k][H-1])%MOD)%MOD;
    cout<<ans<<endl;
    return 0;
}
