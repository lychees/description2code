#include <iostream>
using namespace std;

int main()
{
    int t;
    string a,b;
    cin>>t;
    while(t--){
    	cin>>a>>b;
    	int ans=0;
    	if(a.length()>b.length()){
    		for(int i=0;i<b.length();i++){
    			int x=a[i]-'0';
    			int y=b[i]-'0';
    			if(x+y>9){
    				ans++;
    			}
    		}
    	}else{
    		for(int i=0;i<a.length();i++){
    			int x=a[i]-'0';
    			int y=b[i]-'0';
    			if(x+y>9){
    				ans++;
    			}
    		}
    	}
    	if(ans==0){
    		cout<<"No carry operation"<<endl;
    	}else{
    		if(ans==1){
    			cout<<"1 carry operation"<<endl;
    		}else{
    			cout<<ans<<" carry operations"<<endl;
    		}
    	}
    }
    return 0;
}
