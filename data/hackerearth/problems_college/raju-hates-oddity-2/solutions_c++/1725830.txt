int Combin(int iN, int iR)
{
    if (iR < 0 || iR > iN)
    {
        return 0;
    }
    int iComb = 1;
    int i = 0;
    while (i < iR) {
        ++i;
        iComb *= iN - i + 1;
        iComb /= i;
    }
    return iComb;
}
