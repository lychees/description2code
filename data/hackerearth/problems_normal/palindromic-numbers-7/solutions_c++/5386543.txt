#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
    int a,b,t,count=0,p,q=0;
    scanf("%d",&t);
    for(int k=0;k<t;k++)
    {
    	int a,b;
    	scanf("%d%d",&a,&b);
    	for(int j=a;j<=b;j++)
    	{
    		for(int i=j;i>0;i=i/10)
    		{
    			p=i%10;
    			q=q*10+p;
    		}
    		if(q==j)
    		  count++;
    		q=0;
    	}
    	printf("%d\n",count);
    	count=0;
    }
    return 0;
}
