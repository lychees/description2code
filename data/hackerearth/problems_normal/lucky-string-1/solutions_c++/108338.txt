typedef long long LLD;
using namespace std;
LLD twop[61];

inline string getnth(LLD n)
{
    string ans = "";
    while(n != 0)
    {
	LLD x = n%2;
	LLD p = n/2;
	if(x == 0)
	{
	    ans = '5' + ans;
	    x = 2;
	    p --;
	}
	else
	    ans = '4' + ans;
	n = p;
    }
    return ans;
}
int main()
{
    string con[6] = {"", "", "", "", "Hacker", "Earth"};
    twop[0] = 1LL;
    for(int i = 1; i <= 62; i ++)
	twop[i] = twop[i-1]*2LL;

    int t;
    scanf("%d", &t);
    while(t--)
    {
	LLD n;
	scanf("%lld", &n);
	LLD totald = 0;
	LLD d = 1;
	LLD totaln = 0;
	while(totald + d*twop[d] < n)
	{
	    totald += d*twop[d];
	    totaln += twop[d];
	    d ++;
	}
	n -= totald;
	LLD nth = n/d + 1;
	LLD rem = n%d;
	if(rem == 0) 
	{
	    rem = d;
	    nth --;
	}
	nth += totaln;
	string num = getnth(nth);
	if(num[rem-1] == '4')
	    printf("Hacker\n");
	else
	    printf("Earth\n");
    }
    return 0;
}
