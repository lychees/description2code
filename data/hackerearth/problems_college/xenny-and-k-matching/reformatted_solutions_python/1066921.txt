s = map(int, raw_input().split())
n, k = s
N = map(int, raw_input().split())
d = {}
for i in xrange(0, n):
    if (i+k) < n:
        if (N[i]+N[i+k]) % k == 0 or abs(N[i]-N[i+k]) % k == 0:
            if (i, i+k) not in d and (i+k, i) not in d:
                d[(i, i+k)] = 1
    else:
        if (i-k) >= 0:
            if (N[i]+N[i-k]) % k == 0 or abs(N[i]-N[i-k]) % k == 0:
                if (i, i-k) not in d and (i-k, i) not in d:
                    d[(i, i-k)] = 1
print len(d)
