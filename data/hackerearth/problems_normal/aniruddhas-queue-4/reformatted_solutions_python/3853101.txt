for _ in xrange(int(raw_input())):
    raw_input()
    inp = raw_input().split(" ")
    N = []
    for ii in inp:
        if ii != '':
            N.append(int(ii))

    m = int(raw_input())
    s = sum(N)

    if s >= m:
        i = 0
        while m > 0:
            m -= N[i]
            i += 1

        print i
    else:
        years = m / s - 1
        m -= years * s

        i = 0
        while m > 0:
            m -= N[i % len(N)]
            i += 1

        print i % len(N)
