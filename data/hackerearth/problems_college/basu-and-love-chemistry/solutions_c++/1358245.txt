#include <algorithm>
#include <iostream>
#include <string>

const int MOD = 1e9 + 7;

int main() {
    std::cin.tie(NULL);
    std::ios::sync_with_stdio(false);
    std::string s;
    std::cin >> s;
    int ans = 0, last = 0;
    for (int i = 0; i < int(s.size()); ++i) {
        last = (1LL * last * 10 + (s[i] - '0') * (i + 1)) % MOD;
        ans = (ans + last) % MOD;
    }
    std::cout << ans;
}
