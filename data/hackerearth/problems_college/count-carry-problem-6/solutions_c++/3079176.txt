#include<bits/stdc++.h>
using namespace std;
#define lli long long int
int main()
{
    lli n,x,y,counter;
    int a[100000],b[100000];
    cin>>n;
    while(n--)
    {
        for(lli i=0;i<100000;i++)
        {
            a[i]=0;
            b[i]=0;
        }
        counter=0;
        cin>>x>>y;
        lli i=0;
        while(x)
        {
            a[i++]=x%10;
            x/=10;
        }
        lli z=0;
        while(y)
        {
            b[z++]=y%10;
            y/=10;
        }
        lli m=max(i,z);
        lli carry=0;

        lli sum=0;
        for(lli i=0;i<m;i++)
        {
            sum =a[i]+b[i]+carry;
            if(sum>9)
                counter++;
            carry=(a[i]+b[i]+carry)/10;
            //cout<<sum<<endl;
        }
        if(counter==0)
            cout<<"No carry operation";
        else if(counter==1)
            cout<<"1 carry operation";
        else
            cout<<counter<<" carry operations";
        cout<<"\n";
    }

    return 0;
}
