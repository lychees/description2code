#include <iostream>
#include <vector>
using namespace std;
vector<int> v1,v2,v3;
int main()
{
    int n,p1,p2,result_cur,result_final;
    cin>>n;
    v1.resize(n+1,-1e9);
    v2.resize(n+1,0);
    for(p1=1;p1<n+1;p1++)
    	cin>>v1[p1];
    for(p1=1;p1<=n;p1++)
    {
    	result_cur=0;
    	for(p2=0;p2<p1;p2++)
    	{
    		if(v1[p2]<v1[p1])
    			result_cur=max(result_cur,v2[p2]+1);
    	}
    	v2[p1]=result_cur;
    }
    result_final=0;
    for(p1=1;p1<=n;p1++)
    	result_final=max(result_final,v2[p1]);
    while(result_final>0)
    {
    	v3.push_back(result_final&1);
    	result_final>>=1;
    }
    for(p1=v3.size()-1;p1>=0;p1--)
    	cout<<v3[p1];
    return 0;
}
