#include "bits/stdc++.h"
using namespace std;
const int N = 1e3 + 3;
int n;
int x[N];
int y[N];
int ans = 0;
int main(){
	cin >> n;
	for(int i = 1 ; i <= n ; ++i){
		cin >> x[i] >> y[i];
	}
	for(int i = 1 ; i < n ; ++i){
		++ans;
		double slp = y[i + 1] - y[i];
		slp /= x[i + 1] - x[i];
		for(int j = i + 2 ; j <= n ; ++j){
			if(y[i] + (slp * (x[j] - x[i])) <= y[j]){
				++ans;
			}
			double temp = y[j] - y[i];
			temp /= x[j] - x[i];
			slp = max(slp , temp);
		} 
	}
	cout << ans;
}
