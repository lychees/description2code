#include<iostream>
using namespace std;
int gcd(int n, int m);
int main()
{
    int t,n,m,count,k,flag;
    string s;
        cin>>t;
    while(t--)
    {
        cin>>n>>m;
        cin>>s;
        if(s[0]=='C')flag=1;
        else flag=2;
        count=0;
        if(n<m)swap(n,m);
        while(n>0 && m>0)
        {
        	//cout<<n<<"-"<<m<<"\n";
            count++;
            k=gcd(n,m);
            n=n-k;
            m=m-k;
            }
            //cout<<count<<"-"<<n<<"-"<<m<<"\n";
            if(count%2==1)
            {
            if(s[0]=='C' )cout<<"Chandu";
            else cout<<"Daspal";
            }
            
            else
            {
            if(s[0]=='C')cout<<"Daspal";
            else cout<<"Chandu";
            }
            cout<<"\n";
            
    }
    return(0);
}
int gcd(int n, int m)
{
    int i;
    for(i=n;i>=1;i--)
    {
        if(m%i==0 && n%i==0)return(i);
    }
}
