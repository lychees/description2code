s = raw_input()
a = [[] for i in range(26)]
l = len(s)
for i in range(l):
    a[ord(s[i])-ord('a')].append(i)
ct = [0]*l
g = [0]*l
n = int(raw_input())
for i in range(n):
    p, c = raw_input().split()
    p = int(p)
    g[a[ord(c)-ord('a')][p-1]] = 1
    a[ord(c)-ord('a')].pop(p-1)
t = ''
for i in range(l):
    if not g[i]:
        t += s[i]
print t
