#include <iostream>
#include <cstdio>

using namespace std;

#define LL long long int 

int main()
{
    LL a, b;
    int i, j;
    int n;
    scanf("%d", &n);
    scanf("%d", &j);
    a = j;
    b = j;
    for(i = 1; i < n; i++){
    	scanf("%d", &j);
    	if(j > a){
    		a = j;
    	}
    	if(j < b){
    		b = j;
    	}
    }
    cout<<a*b<<endl;
    return 0;
}
