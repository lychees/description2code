#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

#define pii pair<int,int>
#define pll pair<ll,ll>
#define PB push_back
#define MP make_pair
#define N 100001

int dp[N];

int main(){
	int n, i, j, k;

	scanf("%d", &n);

	for(i = 1; i <= n; i++){
		scanf("%d", &j);
		dp[i] += j;

		for(j = 2 * i; j <= n; j += i)
			dp[j] = max(dp[j], dp[i]);

		printf("%d ", dp[i]);
	}

	printf("\n");

	return 0;
}
