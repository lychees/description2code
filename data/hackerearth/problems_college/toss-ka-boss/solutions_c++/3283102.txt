#include <iostream>
#include<string.h>
using namespace std;

int main() {
	// your code goes here
	int n;
	long long int i,p=0,s=0;
	char a[1000000];
	cin>>n;
	while(n--)
	{ s=p=0;
		cin>>a;
		for(i=0;i<=strlen(a);i++)
		{
			if(a[i]=='H') s+=++p;
			else p=0;
		}
		cout<<s<<endl;
	}
	return 0;
}
