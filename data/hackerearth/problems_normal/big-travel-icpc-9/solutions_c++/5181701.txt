#include <bits/stdc++.h>

using namespace std;
#define MOD 1000000007
int main()
{
 long long int n,i;
 cin>>n;
 long long int ans,x,y;
 vector<long long int>xi;
 vector<long long int>yi;
 for(i=0;i<n;i++)
 {
     cin>>x>>y;
     xi.push_back(x);
     yi.push_back(y);
 }
 sort(xi.begin(),xi.end());
 sort(yi.begin(),yi.end());
ans=0;
 for(i=0;i<n;i++)
 {
     ans=(ans+((2*i-n+1)*xi[i])+MOD)%MOD;
     //ans=(ans-((n-i-1)*xi[i])+MOD)%MOD;
   ans=(ans+((2*i-n+1)*yi[i])+MOD)%MOD;
     //ans=(ans-((n-i-1)*yi[i])+MOD)%MOD;
 }
 if(ans<0)
 {
     ans=(ans+MOD)%MOD;
 }
 cout<<ans;
    return 0;
}
