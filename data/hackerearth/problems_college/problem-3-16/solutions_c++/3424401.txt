#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,k,a;
    cin>>t;
    while(t--){
    	cin>>n>>k;
    	map<int,int> m;
    	set<int>s;
    	int ma=-99999;
    	int toal;
    	for(int i=0;i<n;i++){
    		cin>>a;
    		s.insert(a);
    		m[a]++;
    		if(m[a]>ma){
    			ma=m[a];
    			toal=a;
    		}
    	}
    	s.erase(toal);
    	int ans=0;
    	for(set<int>::iterator i=s.begin();i!=s.end();i++){
    		ans+=m[*i];
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
