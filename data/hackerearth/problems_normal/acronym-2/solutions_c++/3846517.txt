#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,m;
    string x;
    cin>>n;
    map<string,int>mp;
    for(int i=0;i<n;++i) 
    {
    	cin>>x;
    	mp[x]=1;
    }
    cin>>m;
    vector<char>ans;
    for(int i=0;i<m;++i)
    {
    	cin>>x;
    	if(mp.count(x)>0) continue;
    	ans.push_back((char)(toupper(x[0])-65+'A'));
    }
    
    for(int i=0;i<ans.size()-1;++i)
    cout<<ans[i]<<".";
    cout<<ans[ans.size()-1];
    
    return 0;
}
