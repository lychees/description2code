#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
    int n,r,i,A[1000],mean,sum,ch1,ch2,mini,joints;
	
	scanf("%d %d",&n,&r);
	sum=0;
	for(i=0;i<n;i++)
	{
		scanf("%d",&A[i]);
		sum+=A[i];
	}
	
	mean = sum/n;
	
	if(sum!=(mean*n))
	{
		printf("-1");
		return 0;
	}
	sort(A,A+n);
	joints=0;
	while(A[0]<mean)
	{
		ch1 = mean - A[0];
		ch2 = A[n-1] - mean;
		mini = min(ch1,ch2);
		A[0] += mini;
		A[n-1] -= mini;
		joints += r;
		sort(A,A+n);
	}
	printf("%d",joints);
	return 0;
}
