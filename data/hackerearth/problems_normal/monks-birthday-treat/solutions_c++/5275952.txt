#include<bits/stdc++.h>
using namespace std;
vector <int> node[1001];
bool visited[1001];
int n,d,a,b,cnt,sum=INT_MAX;
void dfs(int start)
{
    visited[start]=true;

    for(int i=0;i<node[start].size();i++)
    {
        if(visited[node[start][i]]==false)
            {
                cnt++;
                dfs(node[start][i]);
            }
    }

}
int main()
{
    cin>>n>>d;
    //int sum;
    for(int i=0;i<d;i++)
    {
        cin>>a>>b;
        node[a].push_back(b);
    }
    for(int i=1;i<=n;i++)
    {
        for(int j=0;j<1001;j++)
            visited[j]=false;
        cnt=1;
        dfs(i);
       sum=min(cnt,sum);
    }
    cout<<sum;
    return 0;
}
