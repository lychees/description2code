#include <iostream>
#include<math.h>
using namespace std;

int main()
{ int t, n, k, i, p, s;
  
  cin >> t;
  
  for(i=1;i<=t;i++)
  { cin >> n;
    
    cin >> k;
    s=0;
    p=(pow(2,n));
    if(n>k)
      cout << "NO" << "\n";
    else if(n<=k)
      { s=k-p+1;
        if(s==0 || (s%p==0))
          cout << "YES" << "\n";
        else 
          cout << "NO" << "\n";
      }
      
        
        
    
    
  }
    return 0;
}
