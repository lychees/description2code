#include<bits/stdc++.h>
bool arr[1000000];
using namespace std;
void initialize()
{
    long int i,j;
    arr[0]=true;
    arr[1]=true;
   for(i=2;i<1000000;i++)
    {
       for(j=2;j*i<1000000;j++)
        {
            if(arr[i*j]==0)
             arr[i*j]=true;
        }
    }
 
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int M,N;
        int i,j;
        initialize();
         cin>>M>>N;
         int p=0;
         for(i=M;i<=N;i++)
         {
             if(arr[i]==0)
                p=p+i;
         }
         cout<<p<<endl;
 
    }
      return 0;
}
