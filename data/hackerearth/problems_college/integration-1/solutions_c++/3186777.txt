#include <iostream>
using namespace std;

int main()
{
	int i,t,n,p,u,r,s,x;
	scanf("%d",&t);
	for(i=1;i<=t;i++){
		scanf("%d%d%d%d%d",&n,&p,&u,&r,&s);
		while(n && p>=0){
		
			if(p+r<=u){
				x=(u-p)/r;
				if(n>=x)
				{
					p=u-((u-p)%r);
						n-=x;
				}
				else
				{
					p=p+n*r;
					n=0;
				}
			}
			else if(p-s>=0){
				x=(p-(u-r))%s==0?(p-(u-r))/s:((p-(u-r))/s)+1;
				if(n>=x)
				{
					p=p-(x*s);
					n-=x;
				}
				else
				{
					p=p-(n*s);
					n=0;
				}
			}
			else 
			break;
			//cout<<p<<" "<<n<<endl;
		}
		printf("Case #%d: %d\n",i,p);
	}
    return 0;
}
