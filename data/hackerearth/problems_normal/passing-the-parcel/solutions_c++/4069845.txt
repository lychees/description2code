#include <iostream>
using namespace std;

int main()
{
    static long long int num,i,count,c[1050],j;
    string s;
    cin >> num;
    cin >> s; 
    while(count<num-1)
    {
    	j++;
    	if(j>num)
    		j=1;
    	while(c[j]==1)
    	{
    		j++;
    		if(j>num)
    		j=1;
    	}
    	if(s[i]=='b')
    	{
    		count++;
    		while(c[j]==1)
    		{
    			j++;
    			if(j>num)
    			j=1;
    		}
    		c[j]++;
    	}

    	i++;
    	if(i==s.length())
    	i=0;
    }
    j=1;
    while(c[j]==1)
    	{
    		j++;
    		if(j>num)
    		j=1;
    	}
    cout <<j;
    return 0;
}
