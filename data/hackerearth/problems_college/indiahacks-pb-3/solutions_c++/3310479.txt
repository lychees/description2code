#include <iostream>
 
using namespace std;
 
int main()
{
    long int a[10005],sum[10005],i,j,k,t,n;
cin >> t;
for (j=0;j<t;j++)
{
 
    cin >> n;
    cin >>k;
    for (i=1;i<=n;i++)
   {
 
    cin>> a[i];
    sum[i]=0;}
a[0]=0;
 
for (i=1;i<=k;i++)
   {
if (sum[i-1]>a[i])
sum[i]=sum[i-1];
  else 
  sum[i]=a[i];
   }
 
    for (i=k+1;i<=n;i++)
   
    {
    	sum[i]=max(sum[i-k-
 
                1]+a[i],sum[i-1]);
 
    }
    cout<< sum[n]<<"\n";
}
return 0;
}
