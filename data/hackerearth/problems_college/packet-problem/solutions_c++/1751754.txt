#include <bits/stdc++.h>
using namespace std;

int main()
{
   int n,m;
   long long int one[1000];
   long long int zero[1000];
   one[1]=0;
   zero[1]=1;
   for(int i=2;i<=100;i++)
   {
   	one[i]=zero[i-1];
   	zero[i]=one[i-1]+zero[i-1];
   }
   scanf("%d",&n);
   while(n--)
   {
   	scanf("%d",&m);
   	cout<<one[m]<<" "<<zero[m]<<endl;
   }
    return 0;
}
