#include <iostream>
#include<cmath>
#include <vector>
#include<algorithm>
using namespace std;





int main()
{
	int a[89];
	vector<int>vec;
	for(int i=0;i<89;i++)
    	{
    		a[i] = pow(i+1,3);
    	}


    for(int i=0;i<89;i++)
    	{
    		for(int j=i+1;j<89;j++)
    		{
    			vec.push_back(a[i]+a[j]);
    		}
    	}
    sort(vec.begin(),vec.end());

    vector <int> vsum;
    for(int i=0;i<vec.size();i++)
    {
    	if(vec[i]==vec[i+1] )
    	{
    		vsum.push_back(vec[i]);
    	}
    }
    unique(vsum.begin(),vsum.end());
    int t;
    cin>>t;
    int n;

    while(t--)
    {
    	cin>>n;
    	bool flag = false;
    	int ind;
    	for(int i = vsum.size()-1;i>=0;i--)
    	{
    		if(vsum[i]<=n)
    		{
    			flag=true;
    			ind = i;
    			break;
    		}
    	}
    	if(flag)
    	cout<<vsum[ind]<<endl;
    	else
    	cout<<-1<<endl;

    }

    return 0;
}
