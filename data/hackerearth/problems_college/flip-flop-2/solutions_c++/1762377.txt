#include <iostream>
#include <string>
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <algorithm>
#include <vector>
#include <stack>

using namespace std;
int main()
{

 int t;
 scanf("%d",&t);
 while(t--)
{

	string str;
	cin>>str;
	int count=0;
	for (int i=0; i<str.length();i++ ) {
			if(str[i]=='X')
			{
				if(str[i-1]=='Y' || i==0 )
					i++;

				while(str[i]!='Y' && i<str.length())
					{
						count++;
						i++;
					}
			}
			else if(str[i]=='Y')
			{
					if(str[i-1]=='X' || i==0)
					i++;

				while(str[i]!='X' && i<str.length())
					{
						count++;
						i++;
					}
					
			}			
		}
		if(count==str.length())
		printf("%d\n",count-1);
		else
		printf("%d\n",count );
		
	
}
 return 0;

}
