#include <bits/stdc++.h>
#define ll long long int
using namespace std;

ll a[100005][27];

int main ()
{
	ll n, q;
	string s;
	cin >> n;
	cin >> s;
	ll i,j;
	for (i = 0; i < 100005; i++) {
		for (j = 0; j < 27; j++) {
			a[i][j] = 0;
		}
	}
	
	for (i= 1; i <=n; i++) {
		for(j=0;j<27;j++){
			a[i][j]=a[i-1][j];
		}
		a[i][s[i-1]-'a']++;
	}
	ll l,r;
	ll ch;
	cin >> q;
	while(q--){
		cin>>l;
		cin>>r;
		if(l==r){
			cout << "NO"<<endl;
			continue;
		}
		ll count=0;
		for(i=0;i<27;i++){
			ch=a[r][i]-a[l-1][i];
		//	cout <<ch<<"  ";
			if(ch%2!=0){
				count++;
			}
		}
	////	cout <<endl;
		if(count<=2){
			cout << "YES"<<endl;
		}else {
			cout << "NO"<<endl;
		}
	}
	return 0;
}
