#include <bits/stdc++.h>
using namespace std;

int main() {
	long long int t,i,j,k,l,m,n,o,p;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		long long int a[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		p=0;
		for(i=1;i<n;i++)
		{
			if(a[i-1]>=a[i])
			{
				j=a[i-1]-a[i];
				j=j/k;
				if(j*k + a[i] <=a[i-1])
				j++;
				
				p+=j;
				
				a[i]=a[i]+(j*k);
				
			}
			
			//cout<<a[i]<<" ";
			
		}
		cout<<p<<"\n";
	}
}
