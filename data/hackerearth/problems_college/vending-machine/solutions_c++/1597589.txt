#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
     char s1[10]= "Sheldon", s2[10]="Leonard", s3[10]="Penny", s4[10]="Rajesh", s5[10]="Howard";
     long long int k1=5,k2=6,k3=7,k4=8,k5=9,n,i,x=1,y=2,z=3,u=4,v=5;
     cin >> n;
     if(n<=5)
     {
      if(n==1)
           printf("%s\n",s1);
      else if(n==2)
           printf("%s\n",s2);
      else if(n==3)
           printf("%s\n",s3);
      else if(n==4)
           printf("%s\n",s4);
      else
           printf("%s\n",s5);
      return 0;
     }
     long long int p=2;
     for(i=1;i<=n;i++)
     {
      if(n>=k1+x && n<x+k1+p)
      {
           printf("%s\n",s1);
           return 0;
      }
      x=x+k1;
      k1=k1*2;
      p=p*2;
      if(k1>n)
           break;
     }
     p=2;
     for(i=1;i<=n;i++)
     {
      if(n>=k2+y && n<y+k2+p)
      {
           printf("%s\n",s2);
           return 0;
      }
      y=y+k2;
      k2=k2*2;
      p=2*p;
      if(k2>n)
           break;
     }
     p=2;
     for(i=1;i<=n;i++)
     {
      if(n>=k3+z && n<z+k3+p)
      {
           printf("%s\n",s3);
           return 0;
      }
      z=z+k3;
      k3=k3*2;
      p=p*2;
      if(k3>n)
           break;
     }
     p=2;
     for(i=1;i<=n;i++)
     {
      if(n>=k4+u && n<u+k4+p)
      {
           printf("%s\n",s4);
           return 0;
      }
      u=u+k4;
      k4=k4*2;
      p=p*2;
      if(k4>n)
           break;
     }
     p=2;
     for(i=1;i<=n;i++)
     {
      if(n>=k5+v && n<v+k5+p)
      {
           printf("%s\n",s5);
           return 0;
      }
      v=v+k5;
      k5=k5*2;
      p=p*2;
      if(k5>n)
           break;
     }
     return 0;
}
