import math

sqrs = []
for i in xrange(2, 31623):
    sqrs.append(i*i)

t = int(raw_input())
while t > 0:
    x, y = map(int, raw_input().split())
    hsh = [1]*(abs(y-x)+1)
    end = abs(y-x)+1
    for i in xrange(31621):
        start = sqrs[i]-(x % sqrs[i])
        if start == sqrs[i]:
            hsh[0] = 0
        for j in xrange(start, end, sqrs[i]):
            hsh[j] = 0
    cnt = 0
    for i in xrange(end):
        if hsh[i] == 1:
            cnt += 1
    print cnt
    t -= 1
