#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;

   while(t--)
   {
       long long int d,sum=0;
       cin>>d;
      int i=0;
     while(sum<d)
      {    i++;
          sum+=i;
      }
      if((sum-i)==d)
      i--;
      if(i%3==0)
        cout<<"SL";
      else if(i%3==1)
         cout<<"LB";
      else cout<<"BS";
      cout<<endl;
   }
       return 0;
}
