#include <bits/stdc++.h>
using namespace std;

int main()
{
    int q,r,k,cnt,num,x;
    scanf("%d",&q);
    while(q--)
    {
    	num=0;
    	scanf("%d %d",&r,&k);
    	for(int i=1;i<=r;i++)
 		{
 			x=i;
 			cnt=0;
 			while(1)
 			{
    			if((x%8)==5)
    				cnt++;
 				if(cnt>=k)
 				{
 					num++;
 					break;
 				}
 				x>>=1;
 				if(x==0)
 					break;
 			}
 		}
 		printf("%d\n",num);
    }
    return 0;
}
