#include <iostream>
#include <string.h>
#include <stdio.h>
using namespace std;
void find_pos(int arr[],int n,int temp)
{
    int low=1,up=n,mid;
    while(low<=up)
    {
        mid=low+(up-low)/2;
        if(temp>arr[mid])
            low=mid+1;
        else up=mid-1;
    }
    arr[low]=temp;
}

void sol()
{
    int i,n,l;
    cin>>n;
    int arr[n+1];
    int result[n+1];
    for(i=0;i<n;++i) cin>>arr[i];
    int top=0;
    result[top]=-1;
    for(i=0;i<n;++i)
    {
        if(arr[i]>result[top])
            result[++top]=arr[i];
        else
            find_pos(result,top,arr[i]);
    }
    cin>>l;
   if(l>top) cout<<"-1"<<endl;
   else cout<<result[l]<<endl;

}
int main()
{
 int t;
 cin>>t;
 while(t--) sol();
    return 0;
}
