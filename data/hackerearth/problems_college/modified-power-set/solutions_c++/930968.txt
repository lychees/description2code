#include <iostream>
using namespace std;

int main()
{
	long long int t,l,i,ans,flag[200];
	char str[1000009];
    cin>>t;
    while(t--)
    {
    	for(i=0;i<200;i++)
    	flag[i]=0;
    	cin>>str;
    	l=0;
    	i=0;
    	while(str[i]!=NULL)
    	 {
    	 if(flag[str[i]]==0)
    	    {
    	     flag[str[i]]=1;
    	      l++;
    	    }
    	 i++;
    	}
    	ans=(l*(l+1))/2;
    	cout<<ans<<endl;
    }
    return 0;
}
