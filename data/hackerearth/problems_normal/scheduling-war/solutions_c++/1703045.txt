#include<iostream>
using namespace std;

struct batch{
	int size;
	int val;
	int *arr;
};

int main(){
	int systems;
	cin>>systems;
	int *sys_pro = new int[systems];
	int sum = 0;
	for(int i=0;i<systems;i++) {cin>>sys_pro[i];sum+=sys_pro[i];}

	int batches;
	cin>>batches;
	int process;
	batch *batch_arr = new batch[batches];
	for(int i=0;i<batches;i++) {
		cin>>process;
		batch_arr[i].size = process;
		batch_arr[i].arr = new int[process];
		for(int j=0;j<process;j++) cin>>batch_arr[i].arr[j];
		cin>>batch_arr[i].val;
	}

	int allrun = 0;
	for(int i=0;i<batches;i++) allrun+= batch_arr[i].val;
	int allgap = (allrun - sum);
	int previous;
	int current;
	int gapsum = 0;
	for(int i=0;i<batches;i++){
		previous = current = 0;
		for(int j=0;j<batch_arr[i].size;j++){
			previous += sys_pro[batch_arr[i].arr[j]-1];
		}
		if(batch_arr[i].val>previous) gapsum +=(batch_arr[i].val - previous);
	}
	cout<<max(gapsum,allgap);
}
