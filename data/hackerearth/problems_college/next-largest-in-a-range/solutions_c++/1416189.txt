#include <bits/stdc++.h>
using namespace std;
int main()
{

	std::ios_base::sync_with_stdio(false);
	int i,j,k,l,m,n,o;
	cin>>i>>j>>k>>l;
	vector<int> arr;
	for(m=0;m<i;m++)
	{
		cin>>n;
		if(m>=j&&m<=k)
			arr.push_back(n);
	}
	sort(arr.begin(), arr.end());
	for(m=0;m<l;m++)
	{
		cin>>n;
		o=upper_bound(arr.begin(), arr.end(),n)-arr.begin();
		o+=j;
		if(o>k)
			cout<<"-1\n";
		else
			cout<<arr[o-j]<<endl;;
	}
	return 0;
}
