#include<bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	string s;
	map<string,vector<int> > maps;
	for(int i=1;i<=n;i++){
		cin>>s;
		for(int j=0;j<s.length();j++){
			string s1="";
			for(int k=j;k<s.length();k++){
				s1.push_back(s[k]);
				maps[s1].push_back(i);
			}
		}
	}
	map<string,vector<int> >::iterator it;
	vector<int>::iterator it1;
	for(it=maps.begin();it!=maps.end();it++){
		it1=unique((it->second).begin(),(it->second).end());
		(it->second).resize(distance((it->second).begin(),it1));
	}
	int q;
	cin>>q;
	while(q--){
		int l,r;
		string str;
		scanf("%d%d",&l,&r);
		cin>>str;
		if(str.empty()){
			cout<<r-l+1<<endl;
		}else{
			int i1=upper_bound(maps[str].begin(),maps[str].end(),r)-maps[str].begin();
			int i2=lower_bound(maps[str].begin(),maps[str].end(),l)-maps[str].begin();
			cout<<i1-i2<<endl;
		}
	}
	return 0;
}
