#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    std::ios::sync_with_stdio(false);
    cin>>t;
    while(t--)
    {
        long i,n,m;
        string s,s1,s2,s3,p;
        cin>>s>>p;
        n=s.size();
        m=p.size();
        if(2*n!=m)
        {
            cout<<"Impossible\n";
            continue;
        }
        i=p.find(s);
        if(i!=0)
            s1=p.substr(0,i);
        else
            s1="";
        i=i+s.size();
        if(i<p.size())
            s2=p.substr(i);
        else
            s2="";
        s1=s1.append(s);
        s1=s1.append(s2);
        if(s1==p)
            cout<<"Possible\n";
        else
            cout<<"Impossible\n";
    }
    return 0;
}
