import itertools as it


def prime_factors(n):
    i = 2
    factors = []
    while i * i <= n:
        if n % i:
            i += 1
        else:
            n //= i
            factors.append(i)
    if n > 1:
        factors.append(n)
    return factors


def combinations(number):
    factors = prime_factors(number)
    combinations = 0
    for k in range(1, len(factors) + 1):
        combinations += len(set(it.combinations(factors, k)))
    return combinations


if __name__ == '__main__':
    total = int(input(""))
    for _ in range(total):
        print(combinations(int(input(""))))
