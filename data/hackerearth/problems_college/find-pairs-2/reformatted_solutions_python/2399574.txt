# print 'Hello World!'
t = int(raw_input())
for i in range(t):
    n = int(raw_input())
    l = [int(x) for x in raw_input().split()]
    count = 0
    for j in range(n):
        for k in range(j, n):
            if l[j] == l[k]:
                count += 1
    print count
