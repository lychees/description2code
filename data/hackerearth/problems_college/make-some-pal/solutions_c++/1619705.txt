#include <iostream>
#include <algorithm>
using namespace std;
int solve(string s) {
	int p[2000011], n = s.size();
	
	string current = s + '$';

	for (int i = 0; i < n; i++) {
		current += s[n - 1 - i];
	}

	p[0] = 0;
	for (int i = 1; i < 2 * n + 1; i++) {
		int j = p[i - 1];
		while (j > 0 && current[j] != current[i])
			j = p[j - 1];
		j += current[i] == current[j];
		p[i] = j;
	}
	return 2*n - p[2 * n];
}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s;
    	cin>>s;
    	reverse(s.begin(),s.end());
    	cout << solve(s) << "\n";
    }
    return 0;
}
