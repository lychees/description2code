#include <iostream>
#include<set>
using namespace std;



int main()
{
	
    string a;
    int t,n;
    cin >> t;
    while(t--){
    	set <string> s;
    	cin >>n;
    	while(n--){
    	cin >> a;
    	s.insert(a);
    	}
    	set <string>::iterator it;
    	for(it = s.begin();it != s.end();++it)
        	cout << *it <<  endl;
    
    }
    return 0;
}
