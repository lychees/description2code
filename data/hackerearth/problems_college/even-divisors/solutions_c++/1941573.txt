#include<iostream>
#include<stdio.h>
#include<bits/stdc++.h>
int main(void)
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long int n;
		scanf("%ld",&n);
		if(n&1)
		{
			printf("0\n");
			continue;
		}
		if(n==2)
		{
			printf("1\n");
			continue;
		}
		if(n==4)
		{
			printf("2\n");
			continue;
		}
		long int ct=2; //the number and 2 so count as 2.
		if(((n/2)&1)==0)  // check whether number divided by 2 is even 
		ct++; 
		for(long int i=3;i<=sqrt(n);i++)
		if(n%i==0)
		{
			if((i&1)==0)
			ct++;
			if(i!=(n/i))
			if(((n/i)&1)==0)
			ct++;
		}
		printf("%ld\n",ct);
	}
	return 0;
}
