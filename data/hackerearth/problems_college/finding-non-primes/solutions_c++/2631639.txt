#include <iostream>
#define MAX 10001
using namespace std;

int main()
{
    int p[MAX],i,j,n;
    p[0]=0;
    p[1]=0;
    for(i=2;i<=MAX;i++){
    	p[i]=1;
    }
    
    for(i=2;i<=MAX;i++){
    	if(p[i]){
    		for(j=i*i;j<=MAX;j+=i){
    			p[j]=0;
    		}
    	}
    }
    
    cin>>n;
    for(i=4;i<=n;i++){
    	if(p[i]==0){
    		cout<<i<<endl;
    	}
    }
    return 0;
}
