int f[MAX];
set<int> v[MAX];

void init() {
	fill(f, -1);
	f[1] = 0;
	for (int i = 1; i < MAX; i++) {
		f[i] = v[i].size();
		int id = -1;
		for (set<int>::iterator it = v[i].begin(); it != v[i].end(); it++) {
			if (*it != ++id) {
				f[i] = id;
				break;
			}
		}
		v[i].clear();
		for (int j = i + i; j < MAX; j += i) {
			v[j].insert(f[i]);
		}
	}
}
