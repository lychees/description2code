#include <bits/stdc++.h>


using namespace std;
const long long N=1<<17 , md = 4294967143;
long long a[264][11][11],ans[264],t,x;
int main()
{
  //  freopen("in","r",stdin);
    a[2][1][0]=1;
    a[2][0][1]=1;
    a[2][1][2]=1;
    a[2][2][1]=1;
    a[2][2][3]=1;
    a[2][3][2]=1;
    a[2][3][4]=1;
    a[2][4][3]=1;
    a[2][4][5]=1;
    a[2][5][4]=1;
    a[2][5][6]=1;
    a[2][6][5]=1;
    a[2][6][7]=1;
    a[2][7][6]=1;
    a[2][7][8]=1;
    a[2][8][7]=1;
    a[2][8][9]=1;
    a[2][9][8]=1;
    a[2][9][10]=1;
    a[2][10][9]=1;
    ans[2]=19;
    for(int g=4;g<264;g+=2)
    {
        for (int i=0;i<=10;i++)
            for (int j=0;j<=10;j++)
                for (int start=0;start<=10;start++)
                    for (int last=0;last<=10;last++)
                        if (abs(last-j)==1 && abs(start-i)==1)
                        {
                            a[g][i][j]=(a[g-2][start][last]+a[g][i][j])%md;
                        }
        for (int i=1;i<=10;i++)
            for (int j=0;j<=10;j++)
                if (abs(i-j)==1)
                ans[g]=(ans[g]+a[g][i][j])%md;

    }
    //2 1 2 1 0 1
    cin>>t;
    while (t--)
    {
        cin>>x;
        cout<<ans[x]<<endl;
    }

    return 0;
}
