int palindrome(char* string) {
    int len = strlen(string); // handle empty string and string of length 1:
    if (len == 0) return 0;
    if (len == 1) return 1;
    char *ptr1 = string;
    char *ptr2 = string + len - 1;
    while(ptr2 >= ptr1) {
            if (!isalpha(*ptr2))
            {
                ptr2--;
            continue;
            }
            if (!isalpha(*ptr1))
                {
                    ptr1++;
                continue;
                }
                if( tolower(*ptr1) != tolower(*ptr2))
                    {
                        return 0;
                        }
                        ptr1++; ptr2--;
                        } return 1;
                        }
                        int findMinAppend(char str[])
                        {
                            if (palindrome(str))
                                return 0;
                                else

                                        str++;
                                        return 1 + findMinAppend(str);
