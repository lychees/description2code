#include <iostream>
using namespace std;
#define ll long long int
ll min(ll a,ll b)
{
	return (a<b)?a:b;
}
int main()
{
    //cout << "Hello World!" << endl;
    ll t,x,y,a,b,c,count,i;
    cin>>t;
    ll arr[5];
    while(t--)
    {
    	cin>>x>>y>>a>>b>>c;
    
    arr[0]=x*a+y*b;
    arr[1]=x*min(a,b+c)+y*b;
    arr[2]=x*a+y*min(b,a+c);
    //arr[3]=x*min(a,b+c)+y*min(b,a+c);
    count=arr[0];
    for(i=0;i<3;i++)
    {
    	count=min(count,arr[i]);
    }
    	
    	
    	cout<<count<<endl;
    }
    return 0;
}
