num = raw_input()
num = int(num)
ans = -100

array = raw_input()

st = array.split(" ")

for i in range(0, len(st)):
    st[i] = int(st[i])

st.sort()

div = [0]*num
for i in range(0, num):
    div[i] = 1
    for j in range(0, i):
        if st[i] % st[j] == 0:
            div[i] = max(div[i], div[j]+1)

for i in range(0, num):
    if div[i] > ans:
        ans = div[i]

if (ans >= 2):
    print ans
else:
    print "-1"
