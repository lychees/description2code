#include <iostream>
using namespace std;

int main()
{
 	int t;
 	cin >> t;
 	while (t--) {
 		int n;
 		cin >> n;
 		int c = 0;
 		while (n != 0) {
 			if (n & 1) {
 				c++;
 			}
 			n >>= 1;
 		}
 		if (c % 2) {
 			cout << "odd" << endl;
 		} else {
 			cout << "even" << endl;
 		}
 	}
    return 0;
}
