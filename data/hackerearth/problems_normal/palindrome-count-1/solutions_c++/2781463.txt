#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int len=s.length(),count=0,i,j;
    vector<vector<int> > res(len,vector<int>(len));
    for(i=0;i<len;++i){
    	res[i][i]=true;
    	count++;
    }
    for(i=0;i<len-1;++i){
    	if(s[i]==s[i+1]){
    		count++;
    		res[i][i+1]=true;
    	}
    }
    for(i=2;i<len;++i){
    	for(j=0;j<len-i;++j){
    		if(s[j]==s[i+j] && res[j+1][i+j-1]){
    			res[j][i+j]=true;
    			count++;
    		}
    	}
    }
    cout<<count;
    return 0;
}
