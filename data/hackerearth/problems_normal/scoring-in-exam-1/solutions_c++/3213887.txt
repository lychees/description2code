#include <bits/stdc++.h>
using namespace std;

int main()
{

	long long n,q;
	cin>>n>>q;
	long long time[n],score[n];

	for(int i=0;i<n;i++)
	{
		cin>>time[i];
	}

	for(int i=0;i<n;i++)
	{
		cin>>score[i];
	}

	vector < pair<long long, long long > > v;
	for(int i=0;i<n;i++)
	{
		v.push_back(make_pair(score[i],time[i]));
	}
	sort(v.begin(), v.end());
	reverse(v.begin(),v.end());

	//cout<<v[0].first<<" "<<v[0].second<<endl;

	long long prefix[n];
	prefix[0] = v[0].second;
	for(int i=1;i<n;i++)
	{
		prefix[i] = prefix[i-1] + v[i].second ;

	}

	long long k;
	
	while(q--)
	{
		cin>>k;
		cout<<prefix[k-1]<<endl;
	}

}
