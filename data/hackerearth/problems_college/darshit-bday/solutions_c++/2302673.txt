#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int b,w,x,y,z;
    	cin>>b>>w>>x>>y>>z;
    		
    		if(x>(y+z))
    		{
    			x=y+z;
    		}
    		if(y>(x+z))
    	{
    		y=x+z;
    	}
    	
    	long int sum=(b*x)+(w*y);
    	cout<<sum<<"\n";
    }
}
