#include<iostream>
#include<map>
#include<algorithm>
using namespace std;
int main()
{
    int n;
    cin >> n;
    string s[100000];
    map<string,int> m;
    int si = 0;
    for(int i = 0; i < n; i++)
    {
        string c;
        cin >> c;
        if(m[c] > 0) m[c]++;
        else{
            m[c]++;     s[i] = c;  si++;
        }
    }
    int mi = 0;
    int sum = 0;
    for(int i = 0; i < si; i++)
    {
        mi = max(mi,m[s[i]]);
    }
    for(int i = 0; i < si; i++)
    {
        if(m[s[i]] == mi) sum += mi;
    }
    cout << sum << endl;
    return 0;
}
