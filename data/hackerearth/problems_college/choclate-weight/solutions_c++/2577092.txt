#include <iostream>
#include <algorithm>
#include <cstdlib>
using namespace std;

int bs(int a[],int x,int l,int h){
	
	if(l<=h){
		int mid = (l+h)/2;
		if(x==a[mid]){
			return 1;
		}else if(x>a[mid]){
			return bs(a,x,mid+1,h);
		}else{
			return bs(a,x,l,mid-1);
		}
	}
	return 0;
}

int main()
{
    int t,n,i,j,w;
    int *a;
    cin>>n;
    a = (int *)malloc(sizeof(int)*n);
    for(i=0;i<n;i++){
    	cin>>a[i];
    }
    sort(a,a+n);
    cin>>t;
    while(t--){
    	int f=0;
    	cin>>w;
    	f = bs(a,w,0,n-1);
    	if(f==0){
    		for(i=0;i<n-1;i++){
    			for(j=i+1;j<n;j++){
    				if(w==a[i]+a[j]){
    					f=1;
    					break;
    				}
    			}
    		}
    	}
    	
    	if(f==1){
    		cout<<"I Like You"<<endl;
    	}else{
    		cout<<"I Hate You"<<endl;
    	}
    }
    return 0;
}
