#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <map>
 
using namespace std;

int main()  
{      

    
    map <int, int> scratch;

    int length;
    cin >> length;
    
    for(int i = 0; i < length; i++){

	int item;
	cin >> item;
	
	if(scratch.count(item) == 0){

	    scratch[item] = 1;
	}
	else {

	    scratch[item]++; 
	}
    }
    
    cin >> length;
    
    for(int i = 0; i < length; i++){

	int item;
	cin >> item;
	
	if(scratch.count(item) == 0){

	    scratch[item] = -1;
	}
	else {

	    scratch[item]--; 
	}
    }

    map<int, int>::const_iterator itr;
     
    for(itr = scratch.begin(); itr != scratch.end(); itr++) {

	if(itr->second < 0){
	    
	    cout << itr->first << " ";
	}
    }
    
    return 0;

}
