#include <algorithm>
#include <vector>
#include <iostream>

using namespace std;

typedef vector<bool> vbl;
typedef vector<vbl> vvbl;

void toggle(vvbl &mat, int x, int y)
{
	for(int i = 0; i < x; ++i)
		for(int j = 0; j < y; ++j)
			mat[i][j] = !mat[i][j];
	return;
}

int main()
{
	int t, n , m;
	cin >> t;
	while(t--)
	{
		cin >> n >> m;
		char ch;
		int moves = 0;
		vvbl mat(n, vbl(m, false));
		for(int i = 0; i < n; ++i)
			for(int j = 0; j < m; ++j)
			{
				cin >> ch;
				if(ch == '1') mat[i][j] = true; 
			}
		for(int i = n-1; i >= 0; --i)
		{
			for(int j = m-1; j >=0; --j)
			{
				if(!mat[i][j])
				{
					++moves;
					toggle(mat, i+1, j+1);
				}
			}
		}
		cout << moves << endl;
	}
	return 0;
}
