#include <iostream>
#include <set>

using namespace std;

typedef long long int Lint;

int main()
{
	ios_base::sync_with_stdio( false );
	Lint t, n, i, val;
	
	cin >> t;
	
	while( t-- ) {
		
		cin >> n;
		set<Lint> A;
		
		for( i = 0;i < n;i++ ) {
			cin >> val;
			A.insert( val );
		}
		
		cout << A.size() << "\n";
	}

	return 0;
}
