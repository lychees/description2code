if __name__ == '__main__':
    n = input()
    a = []
    new_z = {}
    for i in xrange(n):
        ids, z, p, l, c, s = map(int, raw_input().split())
        a.append(((p*50 + l*5 + c*10 + s*20 - z), ids))
        new_z[ids] = p*50 + l*5 + c*10 + s*20
    a.sort(reverse=True)
    for i in xrange(5):
        print '{0} {1}'.format(a[i][1], new_z[a[i][1]])
