#include <iostream>
using namespace std;

int main()
{
   long long int t;
   cin >> t;
    for(long long int z=0; z<t; z++){
    	long long int n,q;

    	cin >> n;
    	cin >> q;

    	long long int x=n/(q-1);
    	long long int r=n%(q-1);
  long long int ans= (n-x-r)*x*(q-1)+2*r*(n-x-r)+r*(r-1);
  ans/=2;
  ans=n*(n-1)/2 - ans;
  std::cout << ans << std::endl;






    }
    return 0;
}
