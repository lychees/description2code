#include<bits/stdc++.h>
using namespace std;
int dp[2009][2009];
char a[2009],b[2009];
#define min(a,b,c) min(min(a,b),c)
int main(){
        scanf("%s",a+1);
        scanf("%s",b+1);
        int m=strlen(a+1);
        int n=strlen(b+1);
        for(int i=1;i<=2008;++i)dp[0][i]=i;
        for(int i=1;i<=2008;++i)dp[i][0]=i;
        for(int i=1;i<=m;++i)
            for(int j=1;j<=n;++j){
                dp[i][j]=min(dp[i-1][j]+1,dp[i][j-1]+1,dp[i-1][j-1]+
                        ((a[i]==b[j])?0:1));
            }
        printf("%d\n",dp[m][n]);
    return 0;
}
