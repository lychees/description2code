#include<iostream>
#include<stdio.h>

using namespace std;

long long int ar[20005]={0};
int arr[5]={16,9,26,26,1};
int main()
{
    long long int n,q,i;
    scanf("%lld %lld",&n,&q);
    i=1;
    ar[0]=0;
    while(i<=n)
    {
        scanf("%lld",&ar[i]);
        ar[i]+=ar[i-1];
        i++;
    }
    while(q--)
    {
        int t,l,r;
        scanf("%d %d %d",&t,&l,&r);
        if(t==1)
        {
            cout<<ar[r]-ar[l-1]<<"/"<<r-l+1<<endl;
        }
        else
        {
            long long int s=0;
            i=0;
            while(l<=r)
            {
                s+=arr[i];
                ar[l]+=s;
                i=(i+1)%5;
                l++;
            }
            while(l<=n)
                ar[l++]+=s;
        }
    }
    return 0;
}
