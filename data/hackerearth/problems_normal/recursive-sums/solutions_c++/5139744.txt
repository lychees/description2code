#include <algorithm>
#include <iostream>
#include <string>
#include <vector>
using namespace std;

int recursive(unsigned long long int n)
{
	unsigned long long int t = n;
	unsigned long long int sum=0, remainder;
    while (t != 0)
    {
       remainder = t % 10;
       sum       = sum + remainder;
       t         = t / 10;
    }
    //cout << sum << endl;
    if(sum>9){recursive(sum);}
    else{return sum;}
}

int main()
{
    unsigned long long int n, m, t, d, l;
    string s;
    cin >> t;
    for(int tt=0; tt<t; tt++)
    {
    	cin >> n;
    	m = 0;
    	for(int i=0; i<n; i++)
    	{
    		cin >> d >> l;
    		d=recursive(d);
    		m = m + (d*l);
    	}
    	m=recursive (m);
    	cout << m <<endl;
    }
    return 0;
}
