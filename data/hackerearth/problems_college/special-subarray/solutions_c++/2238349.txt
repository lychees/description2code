#include <bits/stdc++.h>
using namespace std;

int a[123456];
int dp1[123456]{}, dp2[123456]{};

int main(void) {
	ios_base::sync_with_stdio(false);
	int n, m, x, y;
	cin>>n>>m;
	for(int i=0; i<n; i++)
		cin>>a[i];
	for(int i=1; i<n; i++) {
		dp1[i]=(a[i]>=a[i-1])?dp1[i-1]:i;
		dp2[i]=(a[i]<=a[i-1])?dp2[i-1]:i;
	}
	while(m--) {
		cin>>x>>y;
		x-=1, y-=1;
		cout<<((dp1[dp2[y]]<=x)?"Yes\n":"No\n");
	}
	return (0);
}
