#include <iostream>
#include <algorithm>
#include <vector>

int main(void) {
    int N = 0, M = 0, count = 0;
    std::cin >> N >> M;
    std::vector<int> vec(2 * N);
    for(int i = 0; i < N; ++i)
    	scanf("%d", &vec[i]);
    for(int i = N; i < 2 * N; ++i)
    	scanf("%d", &vec[i]);
    std::sort(vec.begin(), vec.end());
    for(int i = 0; i < 2 * N; ++i) {
    	if (M < vec[i])
    		break;
    	M -= vec[i];
    	++count;
    	if (count == N)
    		break;
    }
    std::cout << count << std::endl;
    return 0;
}
