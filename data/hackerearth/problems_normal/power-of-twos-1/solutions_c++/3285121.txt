const int lim = 1000000; 
int dp[N];
void f()
{
	int k = (int)sqrt(lim);
	for(int i=1;i<=k;i++)
	{
		for(int j=i;j<=lim;j+=i)
		{
			dp[j]++;
			if(j/i > k && j/i != i)dp[j]++;
		}
	}
	dp[1] = 0;
	for(int i=2;i<=lim;i++)dp[i] = dp[i-1] + (dp[i] - 1); // 1case should be excluded  
}
int main()
{
	f();
	int t,n;
	si(t);
	while(t--)
	{
		si(n);
		printf("%lld\n",dp[n]);
	}
	return 0;
}
