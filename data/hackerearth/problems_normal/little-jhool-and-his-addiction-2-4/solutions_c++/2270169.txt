#include <bits/stdc++.h>

using namespace std;

int A[1000006];

int main()
{
	int T;
	scanf("%d", &T);
	
	while(T--)
	{
		int N, K;
		scanf("%d %d", &N, &K);
		
		for(int i=0; i<N; i++)
			scanf("%d", &A[i]);
		
		sort(A, A+N);
		
		int maxx = 0, minn = 2000006;
		
		for(int i=0; i<N/2; i++)
		{
			maxx = max(maxx, A[i]+A[N-i-1]);
			minn = min(minn, A[i]+A[N-i-1]);
		}
		
		printf("%d\n", maxx-minn);
		
		if(maxx-minn==K)
			printf("Lucky chap!\n");
		else if(maxx-minn<K)
			printf("Chick magnet Jhool!\n");
		else
			printf("No more girlfriends!\n");
	}
	
	return 0;
}
