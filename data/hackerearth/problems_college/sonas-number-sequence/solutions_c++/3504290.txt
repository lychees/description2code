#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
	int n;
	cin>>n;
	ll arr[n];
	memset(arr,0,sizeof(arr));
	for(int i=0;i<n;i++)scanf("%lld",&arr[i]);
	ll b[n];
	memset(b,0,sizeof(b));
	b[0]=arr[0];
	cout<<b[0]<<" ";
	for(int i=1;i<n;i++){
		b[i]=(i+1)*arr[i]-i*arr[i-1];
		cout<<b[i]<<" ";
	}
	cout<<endl;
	return 0;
}
