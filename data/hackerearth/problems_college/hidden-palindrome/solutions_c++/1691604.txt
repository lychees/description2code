ll gcd(ll a, ll b)
{
	return b == 0 ? a : a > b ? gcd(b, a%b) : gcd(a, b%a);
}
bool myfunc(struct node a,struct node b)
{
	if(a.x == b.x) {
		return a.y < b.y;
	} else {
		return a.x < b.x;
	}
}
ll po(ll a, ll b, ll mod)
{
	if(b == 0) {
		return 1;
	}
	ll t= po(a, b/2 , mod) % mod;
	if(b % 2== 0) {
		return (t*t) %mod;
	} else {
		ll ans = (((t*t) %mod) *a) %mod;
		return ans;
	}
}
int main()
{
	int t;
	S(t);
	while(t--) {
		string s;
		cin >> s;
		int i;
		int j;
		int k;
		int ans = 0;
		int n = s.length();
		for(i = 0;i < n; i++) {
			for(j = i; j < n; j++) {
				string temp = "";
				for(k = i; k <= j; k++) {
					temp = temp + s[k];
				}
				int l = temp.length();
				int fg = 0;
				for(k = 0; k < l/2; k++) {
					if(temp[k] != temp[l-k-1]) {
						fg = 1;
						break;
					}
				}
				if(fg == 0) {
					ans = max(ans, l);
				}
			}
		}
		cout << ans << endl;

	}

}
