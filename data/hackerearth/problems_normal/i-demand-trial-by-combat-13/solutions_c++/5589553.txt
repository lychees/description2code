#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int n,m,f,s;
    
 
    while(t--)
    {
    	cin>>n>>m;
    	vector< vector<int> >lan(2,vector<int>(n));
    	for(int i=0;i<n;i++)
    	{
    		cin>>lan[0][i];
    	}
    	int flag=1;
    	f=0;
    	s=1;
    	while(m-- && flag==1)
    	{
    		flag=0;
    		for(int i=0;i<n;i++)
    		{
    			if(lan[f][i]==1)
    			flag=1;
    		}
    		
    		if(lan[f][1]==1)
    		lan[s][0]=1;
    		else
    		lan[s][0]=0;
    		if(lan[f][n-2]==1)
    		lan[s][n-1]=1;
    		else
    		lan[s][n-1]=0;
    		for(int i=1;i<n-1;i++)
    		{
    			if(lan[f][i-1]==1 && lan[f][i+1]==1)
    			lan[s][i]=1;
    			else
    			lan[s][i]=0;
    		}
    		if(f==1)
    		{
    			f=0;
    			s=1;
    		}
    		else
    		{
    			f=1;
    			s=0;
    		}
    		
    	}
    	for(int j=0;j<n;j++)
    		cout<<lan[f][j]<<" ";
    		cout<<endl;
    	
    	
    }
    return 0;
}
