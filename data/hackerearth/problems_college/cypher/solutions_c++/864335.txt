#include <iostream>
#include<string>
using namespace std;

int main()
{
    long int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	for(int i=0;i<s.size();i++)
          if(s[i]+i>122)
    		cout<<char('a'+((s[i]+i)-123));
          else
            cout<<char(s[i]+i);
    	cout<<endl;
    }
    return 0;
}
