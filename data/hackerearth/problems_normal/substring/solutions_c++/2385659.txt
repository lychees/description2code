#include <bits/stdc++.h>
using namespace std;
typedef unsigned int uint;

uint sum(uint n)
{
	return n*(n+1)*(0.5);
}
int main()
{
	 map<char,bool> x;
	 x['a']=0; x['b']=0; x['c']=0;
	 x['A']=0; x['B']=0; x['C']=0;
	 
     uint t;
     cin>>t;
     ++t;
     
     while(--t)
     {
     	string s;
     	cin>>s;
        uint ans=0,len=s.size();
     	
     	for(uint i=0;i<len;++i)
     	{
     		uint j=0,k=i;
     		while(x.count(s[k++]))
     		 {
     		  ++j;
     	     }
     	     i+=j;
     	     ans+=sum(j);
     	}
     	cout<<ans<<endl;
     	
     }
    return 0;
}
