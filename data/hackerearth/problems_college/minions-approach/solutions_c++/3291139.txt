#include <iostream>
using namespace std;

bool prime[100010];

void sieve() {
	for(int i=0; i<100010; i++) prime[i] = true;
	prime[0] = false;
	prime[1] = false;
	
	for(int i=2; i<100010; i++) {
		if(!prime[i]) continue;
		for(int j=2; i*j<100010; j++)
			prime[i*j] = false;
	}
}

int gcd(int a,int b) {
	while(b!=0) {
		int tmp = b;
		b = a%b;
		a = tmp;
	}
	return a;
}

int main()
{
    sieve();
    int test;
    cin>>test;
    //cout<<"Test "<<test<<endl;
   	while(test--) {
   		int n,cnt=0;
   		cin>>n;
   		//cout<<"N "<<n<<endl;
   		
   		for(int i=2; i<=n; i++)
   			if(gcd(i,n)==1)
   				cnt++;
   		cnt++;
   		//cout<<cnt<<endl;
   		if(prime[cnt]) 
   			cout<<"Correct"<<endl;
   		else 
   			cout<<"Incorrect"<<endl;
   		
   	}
    return 0;
}
