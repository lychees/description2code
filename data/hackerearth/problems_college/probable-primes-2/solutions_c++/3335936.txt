int ary[1000020], t, l, r, newary[1000020];
float ans;

void esieve(int num)
{
    for(int i = 2; i < num; i++)
        if(ary[i] == 1)
        {
            int ii = 2;
            while(i*ii <= num)
                ary[i*ii] = 0, ++ii;
        }
}

int main(){
//freopen("input.txt", "r", stdin);

ary[0] = 0, ary[1] = 0;
for(int i = 2; i <= 1000010; i++)
    ary[i] = 1;
esieve(1000007);
cout<<fixed;
for(int i = 2; i <= 1000003; i++)
    newary[i] += ary[i]+newary[i-1];
cin>>t;
while(t--)
{
    cin>>l>>r;
    if(l == r)
    {
        if(ary[l] == 1)
            ans = 1;
        else
            ans = 0;
    }
    else
        ans = (newary[r]-newary[l-1])/float(r-l+1);
    cout<<setprecision(6)<<ans<<endl;
}

return 0;}
