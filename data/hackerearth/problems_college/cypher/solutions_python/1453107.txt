t=input()
if t>=1 and t<=10000:
  for i in range(0,t):
    s=raw_input()
    if len(s)>=1 and len(s)<=100:
      wd=""
      for i in range(0,len(s)):
        if ord(s[i])+i>122:
          wd=wd+chr(ord(s[i])+i-122+96)
        else:  
          wd=wd+chr(ord(s[i])+i)
      print(wd)
