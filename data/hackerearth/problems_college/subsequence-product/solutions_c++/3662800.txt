#include <bits/stdc++.h>
using namespace std;
int a[10000];
int main()
{
    int t, n, h, i, te, j;
    scanf("%d", &t);
    while (t--) {
        scanf("%d%d", &n, &h);
        vector<pair<int, int> > v;
        for (i = 0; i < n; i++)
            scanf("%d", &a[i]);
        te = h;
        int k = -1;
        for (i = 2; i * i <= h; i++) {
            if (te % i == 0) {
                v.push_back(make_pair(i, 0));
                k++;
            }
            while (te % i == 0) {
                (v[k].second)++;
                te /= i;
            }
        }
        if(te>1)
            v.push_back(make_pair(te,1));
        for (i = 0; i < n; i++) {
            for (j = 0; j < v.size(); j++) {
                while (a[i] % v[j].first == 0) {
                    (v[j].second)--;
                    if (v[j].second <= 0)
                        break;
                    a[i] /= v[j].first;
                }
            }
        }

        int flag = 0;
        for (i = 0; i < v.size(); i++) {
            //printf("%d %d\n", v[i].first, v[i].second);
            if (v[i].second > 0) {
                flag = 1;
                break;
            }
        }
        if (flag)
            printf("NO\n");
        else
            printf("YES\n");
    }
    return 0;
}
