#include <iostream>
using namespace std;
#include <bits/stdc++.h>

typedef long long ll;
ll MIN = -1000000000000005;

double distance(ll x0, ll y0, ll x1, ll y1)
{
	double d = sqrt(  (x1-x0)*(x1-x0)  + (y1-y0)*(y1-y0)  );
	return d;
}



int main()
{
    ll n;
    cin>>n;
    ll i,j,k,l;
    ll x[n+1], y[n+1], f[n+1];
    double dp[n+1];
    for(i=0;i<n;i++)dp[i]=MIN;
    dp[0]=0;
    for(i=0;i<n;i++)
    {
    	scanf("%lld%lld%lld", &x[i], &y[i], &f[i]);
    }
    for(i=0;i<=n-1;i++)
    {	
    	dp[i] += f[i];
    	for(j=i+1;j<=n-1;j++)
    	{
    		double dis = distance(x[i],y[i],x[j],y[j]); 
    		dp[j] = max(dp[j], dp[i] - dis);
    	}
    }
    
    cout.precision(6);
 
	cout<<fixed<<dp[n-1]<<endl;
    return 0;
}
