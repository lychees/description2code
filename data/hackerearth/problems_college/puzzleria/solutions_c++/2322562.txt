#include<iostream>
#define ll long long
using namespace std;
ll fact(ll n)
{
    if(n==0||n==1)
        return 1;
    return n*fact(n-1);
}
ll com(ll n,ll r)
{
    ll t=n-r;
    ll k=1;
    while(n>t)
    {
        k=k*n;
        n--;
    }
    return k/fact(r);
}
int main()
{
    int t,a,b,c,d;
    cin>>t;
    while(t--)
    {
        cin>>a>>b;
        cout<<com(7,a)*com(5,b)*fact(a+b)<<endl;
    }
    return 0;
}
