#include <iostream>
#include <cstdio>
using namespace std;

double ncr[1001][1001];

void init() {
    ncr[0][0] = 1;
    for(int i=1;i<1001;++i) {
        ncr[i][0] = ncr[i][i] = 1.0;
        for(int j=1;j<i;++j) ncr[i][j] = ncr[i-1][j-1]+ncr[i-1][j];
    }
}

int main() {
    init();
    int t, s, n, m, k;
    cin >> t;
    while(t--) {
        cin >> s >> n >> m >> k;
        double p, q, p1;
        if ( (s-m)<=(n-k) ) {
            printf("1.000000\n");
            continue;
        }
        q = ncr[s][n];
        int l=min(m,n);
        p = 0.0;
        for(int f=k;f<=l;++f) p += ncr[m][f]*ncr[s-m][n-f];
        printf("%.6f\n",p/q);
    }
    return 0;
}
