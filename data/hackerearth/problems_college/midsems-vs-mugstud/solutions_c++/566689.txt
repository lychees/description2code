#include <bits/stdc++.h>
using namespace std;

int main() {
	int t,n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int B[n][n][n],A[n][n][n];
		for(int i=0;i<n;++i)
			for(int j=0;j<n;++j)
				for(int k=0;k<n;++k)
				{
					scanf("%d",&B[i][j][k]);
					A[i][j][k] = B[i][j][k];
				}
		for(int i=n-1;i>=0;--i)
		{
			for(int j=1;j<n;++j)	
				A[i][j][0] -= B[i][j-1][0];
			for(int j=1;j<n;++j)
				A[i][0][j] -= B[i][0][j-1];
			for(int j=1;j<n;++j)
				for(int k=1;k<n;++k)
					A[i][j][k] = A[i][j][k] - B[i][j-1][k] - B[i][j][k-1] + B[i][j-1][k-1];
		}
		for(int i=n-1;i>=1;--i)
			for(int j=0;j<n;++j)
				for(int k=0;k<n;++k)
					A[i][j][k] -= A[i-1][j][k];
		for(int i=0;i<n;++i)
			for(int j=0;j<n;++j)
			{
				for(int k=0;k<n;++k)
					printf("%d ",A[i][j][k]);
				printf("\n");
			}
	}
	return 0;
}
