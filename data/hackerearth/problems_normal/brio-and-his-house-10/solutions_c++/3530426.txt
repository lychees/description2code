#include<stdio.h>
#include<math.h>
const double pi=3.1415;
int main()
{
	int tc;
	scanf("%d",&tc);
	while(tc--)
	{
		double a,b,c;
		scanf("%lf %lf %lf",&a,&b,&c);
		double s;
		s=(a+b+c)/2.0;
		//printf("s is %lf\n",s);
		double area=sqrt((s)*(s-a)*(s-b)*(s-c));
		double R=(a*b*c)/(4.0*area);
		double circ_area=(pi)*(R)*(R);
		printf("%.4lf\n",circ_area);

	}
	return 0;
}
