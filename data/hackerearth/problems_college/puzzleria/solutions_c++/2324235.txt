#include<bits/stdc++.h>
using namespace std;
int main(){
long long int t,x,y,a,b,c,d,e,f,g,i;
cin>>t;
while(t--){
    cin>>x>>y;d=1;
    if(x==0||x==7)
        a=1;
    else if(x==1||x==6)
        a=7;
    else if(x==2||x==5)
        a=21;
    else if(x==3||x==4)
        a=35;
    if(y==0||y==5)
        b=1;
    else if(y==1||y==4)
        b=5;
    else if(y==2||y==3)
        b=10;
        c=x+y;
    for(i=2;i<=c;i++)
        d=d*i;
    d=d*a*b;
    cout<<d<<'\n';
}
return 0;
}
