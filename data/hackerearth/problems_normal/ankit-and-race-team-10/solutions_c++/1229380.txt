#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

const int N=100010;
const int md=1e9+7;

ll fact[N],iFact[N];
ll a[N];

ll powmod(ll a,int b) {
    if(b==0) {
        return 1ll;
    }
    ll p=powmod(a,b/2);
    p=p*p%md;
    if(b&1) {
        p=p*a%md;
    }
    return p;
}
void preprocess() {
    fact[0]=1;
    for(int i=1;i<N;i++) {
        fact[i]=fact[i-1]*i%md;
    }
    iFact[N-1]=powmod(fact[N-1],md-2);
    for(int i=N-2;i>=0;i--) {
        iFact[i]=iFact[i+1]*(i+1)%md;
    }
    return;
}
int main() {
    preprocess();
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        ll n,k;
        scanf("%lld%lld",&n,&k);
        for(int i=0;i<n;i++) {
            scanf("%lld",&a[i]);
        }
        ll ans=0;
        sort(a,a+n);
        for(int i=0;i<n-k+1;i++) {
            ll t1;
            t1=fact[n-i-1];
            t1=t1*iFact[k-1]%md;
            t1=t1*iFact[n-i-k]%md;
            t1=t1*a[i]%md;
            ans+=t1;
            if(ans>=md) {
                ans-=md;
            }
        }
        printf("%lld\012",ans);
    }
    return 0;
}
