#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define sc(a) scanf("%d",&a)
#define scl(a) scanf("%lld",&a)

int main()
{
    int t,i,j,k;
    ll n,a,b;
    sc(t);
    while(t--)
    {
    	scl(n);
    	scl(a);
    	scl(b);
    	if(a==0 || b==0)
    	{
    		printf("-1\n");
    		continue;
    	}
    	ll ans=(n/a)+(n/b)-(n/((a*b)/__gcd(a,b)));
    	printf("%lld\n",ans);
    }
    return 0;
}
