#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <iostream>
#include <math.h>
using namespace std;
long long int a[1000009],a1[1000009];

int main ()
{
	int t;
	scanf("%d",&t);
	while(t--) {
		long long int n,k,i,j,count = 0,max,max_element =0;
		scanf("%lld%lld",&n,&k);
		for(i = 1 ; i <= n;i++) {
			scanf("%lld",&a[i]);
			
		}
		for(i = 1; i<= k;i++) {
			scanf("%lld",&a1[i]);
		
		}
		
		count = 0;
		
		max = -1;
	
		long long int p  = 0;
		a[0] = -1;
		for( i = 1;i <= n ; i++) {
			if(a[i] <= a[i-1] ) {
				continue;
			}
			
			
			for(j =0;j < k;j++) {
				
				if(a[i] >= a1[j]) {
					count++;
				}
				else {
					break;
				}
			}
		
			if(count > max) {
				
				max = count;
				
				max_element = i;
				if(max==k) {
					break;
				}
			}
			count = 0;
			

		}
		printf("%lld\n",max_element-1);
		
	}

return 0;
}
