#include <iostream>
using namespace std;
#include<vector>
#include<bits/stdc++.h>
#define ll long long
#define ull unsigned long long
#define pb push_back
int main()
{
    
    ll i,j,count=0,test,N,num;
    cin>>test;
    while(test--)
    {
    	cin>>N;
    	vector<ll> vec;
    	for(i=0;i<N;i++)
    		{
    			cin>>num;
    		//	cout<<num<<" ";
    			vec.pb(num);
    		//	cout<<vec[i]<<endl;
    		}
    	for(i=0;i<N;i++)
    	{
    		for(j=i;j<N;j++)
    		{
    			if(j>=i && vec[i]==vec[j]){count++;}
    		//	cout<<i<<" "<<j<<endl;;
    		}
    	}
    	cout<<count<<endl;
    	count=0;
    }
    return 0;
}
