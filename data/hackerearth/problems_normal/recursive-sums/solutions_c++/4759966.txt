#include <iostream>
using namespace std;
#include <string>
#include <algorithm>

long long int sumit(long long int a){
	if(a < 10) return a;
	else{
		return (sumit(a%10) + sumit(a/10));
	}
} 

 
int main()
{
	int t;
	cin >> t;
	while(t--){

		int m;
		cin >> m;
		long long int finalsum = 0;

		 for(int i = 0; i < m ;i++){
		 	long long int q,w;
		 	cin >>q>>w;
		 	long long int temp = sumit(q*w);
		 	finalsum = sumit(finalsum+temp);



		 }
		 finalsum = sumit(finalsum);
		 cout << finalsum << endl;
	}

    return 0;
}
