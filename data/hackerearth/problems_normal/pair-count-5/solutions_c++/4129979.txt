#define MAX 1000003
#include <bits/stdc++.h>

using namespace std;
int main()
{
	long n,d,a[MAX];
	cin>>n>>d;
	long freq[d];
	for(long i=0;i<d;i++)
		freq[i]=0;
	for(long i=0;i<n;i++)
	{
		cin>>a[i];
		a[i]%=d;
		freq[a[i]]++;
		a[i]=__gcd(a[i],d);
		a[i]=d/a[i];
		
	}
	long long ans=0;
	for(long i=0;i<n;i++)
	{
		if(a[i]==1)
			ans--;
		for(long j=0;j<d;j+=a[i])
		{
			ans+=freq[j];
		}
	}
	cout<<(long long)ans/2<<endl;
	return 0;
}
