#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
        int n;
        cin >> n;
        char a[n][n];
        string s;
        for(int i = 0; i < n; i++) {
            cin >> s;
            for(int j = 0; j < n; j++) {
                a[i][j] = s[j];
            }
        }
        bool vertical = true;
        bool horizontal = true;

        for(int i = 0; i < n; i++) {
            for(int j = 0; j < n; j++) {
                if(a[i][j] != a[n - i - 1][j]) {
                    horizontal = false;
                }
                if(a[i][j] != a[i][n - j - 1]) {
                    vertical = false;
                }
            }
        }

        if(horizontal && vertical) {
            cout << "BOTH" << endl;
        }
        else if(!horizontal && !vertical) {
            cout << "NO" << endl;
        }
        else {
            (horizontal) ? cout << "HORIZONTAL" << endl : cout << "VERTICAL" << endl;
        }
    }
    return 0;
}
