#include <iostream>
using namespace std;

int binsearch(int sum[],int st, int end, int ind);
int main()
{
    int n;
    int c[100000];
    cin >> n;
    for(int i=0;i<n;i++)
    cin >> c[i];
    int sum[100000];
    sum[0]=c[0];
    for(int i=1;i<n;i++)
    sum[i] = sum[i-1]+c[i];
    int q;
    cin >> q;
    while(q--)
    {
    	int ind;
    	cin >> ind;
    	cout<<binsearch(sum,0,n,ind)+1<<endl;
    }
    return 0;
}

int binsearch(int sum[],int st, int end, int ind)
{
	if(st <= end)
	{
		int mid = (st + end)/2;
		if (sum[mid] == ind || (ind>sum[mid-1]&&ind<sum[mid]))
		return mid;
		else if(ind<sum[mid])
		binsearch(sum,st,mid-1,ind);
		else
		binsearch(sum,mid+1,end,ind);
	}
}
