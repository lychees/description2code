
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
#include <set>
#include <map>
#include <string>
#include <vector>
#include <queue>
#include <algorithm>
#include <iostream>
using namespace std;

const int N = 105;
int a[N], q[N], sz = 0;

int main() {
	int t, n;
	cin >> t;
	for(; t--; ) {
		cin >> n;
		sz = 0;
		for(int i = 0; i < n; ++i)
			cin >> a[i];
		for(int i = 0; i < n; ++i) {
			int x;
			cin >> x;
			if(x > 1) q[sz++] = x-1;
			if(a[i] > x) q[sz++] = a[i] - x;
		}

		int ans = 0;
		for(int i = 0; i < sz; ++i)
			ans ^= q[i];

		cout << (ans ? "First" : "Second") << endl;
	}

    return 0;
}
