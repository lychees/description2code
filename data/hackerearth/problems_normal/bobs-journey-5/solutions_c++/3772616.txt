#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	
	while(t--) {
		int n;
		cin>>n;
		string str;
		int a[26]={0};
		for( int i=0;i<n;i++) {
			cin>>str;
			a[str[0]-'a']++;
		}
		int f=0;
		for( int i=0;i<26;i++) {
			if(a[i]>1) {
				f=1;
				break;
			}
		}
		if(f)
		cout<<"NO"<<endl;
		else
		cout<<"YES"<<endl;
		
	}
    return 0;
}
