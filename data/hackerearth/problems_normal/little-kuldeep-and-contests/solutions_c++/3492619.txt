#include<bits/stdc++.h>
using namespace std;
int main() {
	int t;
	cin >> t;
	pair<int, int> p[t];
	for(int j = 0; j < t; j++){
		char s[15];
		scanf("%s",s);
		char *ptr =  strtok(s, "-:");
		int a[4], i = 0;
		
		while(ptr){
			string b = (string)ptr;
			ptr = strtok (NULL, "-:");
		
			a[i++] = stoi(b);
		}
		p[j].first = a[0]*60 + a[1];
		p[j].second = a[2]*60 + a[3];
		//cout << p[j].first << " " << p[j].second << endl;
	}
	sort(p, p+t);
	int f = 0;
	for(int i = 0; i < t-1; i++) {
		if(p[i].second > p[i+1].first) {
			f = 1;
			break;
		}
	}
	if(f) {
		cout <<"Will need a moderator!\n";
	}else cout <<"Who needs a moderator?\n";
}
