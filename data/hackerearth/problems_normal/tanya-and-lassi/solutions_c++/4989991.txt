#include <bits/stdc++.h>
using namespace std;
long long int max_lassi(long long int arr[],long long int n)
{
	long long int dp[100000],max,temp;
	dp[0]=0;
	for(long long int l=1;l<=n;l++)
	{
		max=LLONG_MIN;
		for(long long int i=1;i<=l;i++)
		{
			temp=arr[i-1]+dp[l-i];
			if(temp>max)
			max=temp;
		}
		dp[l]=max;
	}
	return dp[n];
}
int main()
{
    long long int t,l,arr[100000],temp,i;
    cin>>t;
    while(t--)
    {
    	cin>>l;
    	for(i=0;i<l;i++)
    	cin>>arr[i];
    	temp=max_lassi(arr,l);
    	cout<<temp<<"\n";
    }
    return 0;
}
