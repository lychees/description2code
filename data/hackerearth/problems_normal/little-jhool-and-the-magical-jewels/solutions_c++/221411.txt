#include <stdio.h>
#include <string>
#include <iostream>
#include <algorithm>
#define N 101
#define ll  long long int

using namespace std;


int noOfRubies(char *str)
{
    int i=0;
    //printf("Hello\n");
    int charMap[30]={0};
    while(str[i]!='\0')
    {
        int t=str[i]-97;
        charMap[t]++;
        i++;
        //printf("%c",str[i++]);
    }
   // for(i=0;i<26;i++)
   // printf("%d ",charMap[i]);
    //1,17,20,24
    return min(charMap[1],min(charMap[17],min(charMap[20],charMap[24])));
}

int main()
{
    int t;
    char *str=(char*)malloc(sizeof(char)*N);
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s",str);
        printf("%d\n",noOfRubies(str));
    }
    return 0;
}
