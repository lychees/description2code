/*

         Do not go gentle into that good night,
         Old age should burn and rave at close of day;
         Rage, rage against the dying of the light.
         Though wise men at their end know dark is right,
         Because their words had forked no lightning they
         Do not go gentle into that good night.


*/
#include<bits/stdc++.h>
using namespace std;


int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		long long int a[n];
		int occupied[n+1];
		for(int i=0;i<n;i++)cin>>a[i],occupied[i]=0;
		int l=0,r=n-1;
		long long int timeL=0,timeR=0,cntL=0,cntR=0;
		while(l<=r && (!occupied[l] && (!occupied[r]))){
			if(timeL<=2*timeR){
				timeL+=(a[l]);
				occupied[l]=1;
				l++;
				cntL++;
			}
			else {
				timeR+=a[r];
				occupied[r]=1;
				r--;
				cntR++;
			}
		}
		cout<<cntL<<" "<<cntR<<endl;
	}
	return 0;
}
