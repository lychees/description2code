#include <iostream>
#include <cstring>
using namespace std;

long long C[1024][1024];

void binomialCoeff(int N, int k)
{
    memset(C, 0, sizeof(C));
 
 	for(int r=1;r<=N;r++){
    C[r][0] = 1;  // nC0 is 1
 
    for (int i = 1; i <= r; i++)
    {
        // Compute next row of pascal triangle using
        // the previous row
        for (int j = min(i, k); j > 0; j--)
            C[r][j] = (C[r][j] + C[r][j-1])%1000000007;
    }
 	}
}

int main()
{
	binomialCoeff(1002,1002);
	int n,t,k;
	cin>>t;
	while(t--){
		cin>>n>>k;
		long long sum=0;
		for(int i=0;i<=k;i++){
			sum=(sum+C[n][i])%1000000007;
		}
		cout<<sum<<endl;
	}
    return 0;
}
