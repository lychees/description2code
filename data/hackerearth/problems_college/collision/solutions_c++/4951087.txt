#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int t;
	cin>>t;
	for(int j=0;j<t;j++)
	{
	int n;
    cin>>n;
    vector<int> v;
    for(int i=0;i<n;i++)
    {
    	int num;
    	cin>>num;
		v.push_back(num);
    }
    
    vector<int> hash(10,-1);
    for(int i=0;i<n;i++)
    {
    	hash[v[i]%10]++;
    }
    int sum=0;
    for(int i=0;i<10;i++)
    {
    	if(hash[i]>0)
    	sum+=hash[i];
    }
    cout<<sum<<endl;
	}
    return 0;
}
