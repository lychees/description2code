#include<bits/stdc++.h>
using namespace std;
int freq[26]={0};
int main()
{
    ios::sync_with_stdio(false);
    string str;
    int n,i,c;
    cin>>n;
    while(n--)
    {
        cin>>str;
        freq[str[0]-'a']++;
    }
    c=0;
    for(i=0;i<26;i++)
    {
        if(freq[i]!=0){
            if(freq[i]%10!=0)
                c+=((freq[i]/10)+1);
            else if(freq[i]%10==0)
                c+=freq[i]/10;
        }
    }
    cout<<c;
    return 0;
}
