#include<iostream>
#include<cstdio>
using namespace std;
int inline gcd(int a, int b)
{
	if(b==0) return a;
	return gcd(b,a%b);
}
int main()
{
	int t,i,j,k;
	int gcdval[34];
	for(i=1;i<=33;i++) gcdval[i]=0;
	for(i=1;i<=98;i++)
	{
		for(j=i+1;j<=99;j++)
		{
			for(k=j+1;k<=100;k++)
			{
				gcdval[gcd(gcd(i,j),k)]++;
			}
		}
	}
	//for(i=1;i<=100;i++) cout<<i<<' '<<gcdval[i]<<' ';cout<<endl;
	cin>>t;
	while(t--)
	{
		unsigned long long x,z;
		scanf("%lld %lld",&x,&z);
		//cout<<x<<' '<<z<<endl;
		if(z>=x) x=z-x;
		else x=x-z;
		/*if(x==0) cout<<"1.000000\n"<<endl;
		else*/
		{
			int num=0,den=0;
			for(i=1;i<=33;i++)
			{
				if(x%i==0) num+=gcdval[i];
				den+=gcdval[i];
			}
			double answer;
			answer=num;
			answer/=den;
			//cout<<den<<endl;
			printf("%0.6f\n",answer);
		}
	}
}
