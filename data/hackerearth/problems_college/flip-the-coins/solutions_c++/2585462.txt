#include <iostream>
using namespace std;

int main()
{
    int n,q,x,y,z,i;
    cin>>n;
    int *a = (int *)malloc(sizeof(int)*n);
    for(i=0;i<n;i++){
    	a[i] = 0;
    }
    
    cin>>q;
    while(q--){
    	cin>>x>>y>>z;
    	if(x==0){
    		for(i=y;i<=z;i++){
    			if(a[i]==0){
    				a[i] = 1;
    			}
    			else if(a[i]==1){
    				a[i] = 0;
    			}
    		}
    	}else{
    		int c=0;
    		for(i=y;i<=z;i++){
    			if(a[i]==1){
    				c++;
    			}
    		}
    		cout<<c<<endl;
    	}
    }
    
    return 0;
}
