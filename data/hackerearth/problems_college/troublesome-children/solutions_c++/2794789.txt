#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll ans=0;

int main(){
ios::sync_with_stdio(false);

ll i,j,k,n;

cin>>k;
int arr[k],pos[k];
for(i=0;i<k;i++)
    cin>>arr[i];
memset(pos,0,sizeof(pos));
int cp=0;
int p=0;
while(cp<k)
{
    int val=arr[cp];
    while(val)
    {

        if(pos[p]==0)
        {
            if(val==1)
            {
                pos[p]=1;
                cout<<p+1<<"\n";
            }
            p++;

            val--;
        }
        else
            p++;

      if(p==k)
        p=0;


    }
    cp++;


}



return 0;
}
