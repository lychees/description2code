#include<iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    int l,r;
    while(t--)
    {
        int b1,b2,m1,m2;
        cin>>b1>>b2>>m1>>m2;
        r= max( min(b1,b2),min(m1,m2 ));
        l = min ( max(b1,b2),max(m1,m2 ));

        if((l>r)) cout<<"Nothing"<<endl;
        else if(l==r) cout<<"Point"<<endl;
        else cout<<"Line"<<endl;
    }
    return 0;
}
