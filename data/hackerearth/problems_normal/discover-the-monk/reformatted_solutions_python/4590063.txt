def is_present(nums, x):
    i = 0
    j = len(nums) - 1

    while i <= j:
        mid = (i + j) / 2
        if nums[mid] == x:
            return 'YES'
        elif nums[mid] < x:
            i = mid + 1
        else:
            j = mid - 1

    return 'NO'


N, Q = map(int, raw_input().split())
nums = sorted(map(int, raw_input().split()))

for i in range(Q):
    x = input()
    print is_present(nums, x)
