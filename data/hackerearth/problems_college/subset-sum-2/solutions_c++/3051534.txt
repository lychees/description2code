#include <iostream>
#define ll long long
#define mod 1000000007
using namespace std;
ll power(int a, int b)
{
	ll arr[100000], p=0;
	while(b!=0)
	{
		arr[p++]=b&1;
		b=b>>1;
	}
	ll ans=1, i;
	for(i=p-1;i>=0;i--)
	{
		if(arr[i]==1)
			ans=(ans*ans*a)%mod;
		else
			ans=(ans*ans)%mod;
	}
	return ans;
}
int main()
{
	ll t, n, ans, p;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	ans=((n*(n+1))/2)%mod;
    	p=power(2,n-1);
    	ans=(ans*p)%mod;
    	cout<<ans<<endl;
    }
    return 0;
}
