#include <iostream>
#include <math.h>

using namespace std;

int main()
{
    double number_of_banana,lemurs;
            int best_steps = 0;
    int count_steps = 0;
    cin >> number_of_banana >> lemurs;
    if(lemurs>number_of_banana)
    {
        //kill the extra moneys or increase the banana
        count_steps = abs(int(lemurs-number_of_banana));
        if(count_steps > number_of_banana)
            count_steps = number_of_banana;
        best_steps = count_steps;

    }
    else{
        int h1,h2;
        int upper,lower;

        for (int k=1;k<=number_of_banana;k++)
        {

            count_steps = abs(int(lemurs - k));
            h1 = int(number_of_banana/k)*k;
            h2 = h1 + k;
            h1 = number_of_banana - h1;
            h2 = h2 - number_of_banana;
        if(h1<h2)
        {
            count_steps = count_steps+h1;
        }
        else{
            count_steps = count_steps+h2;
        }
        if(k==1)
            best_steps = count_steps;
        else if(best_steps > count_steps)
            best_steps = count_steps;
        }

        }
    cout << best_steps;
    return 0;
}
