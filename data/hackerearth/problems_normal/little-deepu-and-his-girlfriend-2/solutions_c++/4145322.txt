#include<iostream>
#include<vector>
using namespace std;

void printVector(vector<int> table)
{
	for(int i=0;i<table.size();i++)
		cout<<i<<" "<<table[i]<<endl;
	cout<<endl;
}
int main()
{
	int NTest;
	cin>>NTest;
	
	for(int i=0;i<NTest;i++)
	{
		int NIteam;
		cin>>NIteam;
		NIteam=NIteam+1;
		int N;
		cin>>N;
		
		vector<int> list(N);
		for(int i=0;i<N;i++)
			cin>>list[i];
		
		vector<int> table(NIteam,0);
		
		for(int i=0;i<list.size();i++)
			table[list[i]]=1;
		
		
		for(int i=0;i<NIteam;i++)
		{
			if(table[i]==1) continue;	//is the list numbers
			
			for(int k=0;k<list.size();k++)
			{
				
				if(i-list[k] >= 0)
				{
					if(table[i-list[k]]==0)
					{
						table[i]=1;
						break;
					}
						
				}
			}
			
		}
		
		//printVector(table);
		int ans=table[NIteam-1];
		
	//	cout<<NIteam-1<<" "<<table[NIteam-1]<<endl;
		
		if(ans==1) cout<<"Little Deepu\n";
		else cout<<"Kate\n";
		
	}
}
