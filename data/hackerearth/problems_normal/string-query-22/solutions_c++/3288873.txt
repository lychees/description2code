typedef
tree<

set_t;

set_t pos[200];
string ss, s;
int q, x;
int ok[me];
int main()
{
    ios_base:: sync_with_stdio(false);
    cin.tie(false);

    cin >> ss;
    for(int i = 0 ; i < ss.size() ; i ++)
        pos[ ss[i] ].insert(i);

    cin >> q;
    while(q --){
        cin >> x >> s;
        char gg = s[0];
        auto i = pos[gg].find_by_order(x - 1); pos[gg].erase(i);
    }

    for(int i = 'a' ; i <= 'z' ; i ++)
        for(int j : pos[i])
            ok[j] = 1;

    for(int i = 0 ; i < ss.size() ; i ++)
        if(ok[i])
            cout << ss[i];
    cout << endl;

}
