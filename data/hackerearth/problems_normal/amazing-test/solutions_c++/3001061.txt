#include <bits/stdc++.h>
using namespace std;
int dp[105][100005], A[105];
int main()
{
	int t;
    cin>>t;
    while(t--){
    	int x,n;
    	cin>>n>>x;
    	int s=0;
    	for(int i=1; i<=n; i++){
    		cin>>A[i];
    		s+=A[i];
    	}
    	for(int i=0; i<=n; i++){
    		for(int j=0; j<=s; j++)
    			dp[i][j]=0;
    	}
    	dp[0][0]=1;
    	for(int i=1; i<=n; i++){
    		for(int j=0; j<=s; j++){
    			dp[i][j]=dp[i-1][j];
    			if(j-A[i]>=0)
    				dp[i][j]=dp[i][j]|dp[i-1][j-A[i]];
    		}
    	}
    	if(s/2>x){
    		cout<<"NO\n";
    	}
    	else{
    		int f=0;
    		for(int i=s/2; i<=x; i++){
    			if(dp[n][i]){
    				cout<<"YES\n";
    				f=1;
    				break;
    			}
    		}
    		if(!f) cout<<"NO\n";
    	}
    }
    return 0;
}
