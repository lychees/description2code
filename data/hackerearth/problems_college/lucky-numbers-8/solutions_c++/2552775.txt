#include <iostream>
using namespace std;
int main()
{
    int a[100009],i,t,n,m;
    cin>>t;
    while(t--)
    {
        int c=1;
        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>a[i];
        }
        m=a[n-1];
        for(i=n-1;i>=0;i--)
        {
            if(m<a[i])
            {
                c++;
                m=a[i];
            }
        }
        cout<<c<<endl;
    }
    return 0;
}
