for i in range(input()):
    a, k = map(int, raw_input().split())
    cnt = 0
    start= 1
    while True:
        if (a & start) << 1 > 0:
            start += 1
            continue
        else:
            cnt += 1
            start += 1
            if cnt == k:
                break
    print "Case #%d: %d" % (i+1, start-1)
