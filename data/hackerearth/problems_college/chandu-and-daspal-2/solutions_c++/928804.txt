long long int gcd(long long int a,long long int b)
{
	long long int c;
	while(a!=0)
	{
		c = a;
		a = b%a;
		b = c;
	}
	return b;
}
int decide_winner(int x, int y, int calc)
{
	if(x==0 || y==0)
		return calc;
	ll val=gcd(x,y);
	x-=val;
	y-=val;
	decide_winner(x,y,1-calc);
}
