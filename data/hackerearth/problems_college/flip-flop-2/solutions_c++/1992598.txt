#include <iostream>
using namespace std;
int main()
{
    int t,test=1;
    cin>>t;
    while(test<=t)
    {
    	int i,count=0;
    string str;
    cin>>str;
    for(i=0;i<str.size()-1;i++)
    {
    	if(str[i]==str[i+1])
    	count++;
    }
    cout<<count<<endl;
    test++;
    }
    return 0;
}
