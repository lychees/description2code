#include<iostream>
#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;
int count_one (int n)
    {   int count=0;
        while( n )
        {
        n = n&(n-1);
           count++;
        }
        return count;}

int main()
{ int tc;
  cin>>tc;
  while(tc--)
  { int n;
    cin>>n;
    vector <pair<int,int> > a(n);
    int i;
    for(i=0;i<n;i++)
    { int val;
      cin>>val;
      a[i].second=val;
      a[i].first=count_one(val);
    }
   sort(a.begin(),a.end());
   cout<<a[0].first<<"\n";
  }
return 0;
}
