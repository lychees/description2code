#include <iostream>
#include<stdio.h>
#include<cmath>
using namespace std;

class digit
{

	public:
	int num;
	int cycle[5];
	 void createcycle(int n)
	  {
	  	num=n;
	  	for(int i=1;i<=4;i++)
	  	  cycle[i]=((int)pow(num,i)%10);
        cycle[0]=cycle[4];
	  }
};

int main()
{
	digit d [10];
	for(int i=0;i<10;i++)
      d[i].createcycle(i);
	  long long t,b,e;
	  int lst,c;
	  scanf("%lld",&t);
	  while(t)
	  {
	  	scanf("%lld",&b);
	  	lst=b%10;
	  	scanf("%lld",&e);
	  	c=e%4;
		printf("%d \n",d[lst].cycle[c]);
	  	t--;

	  }
	 return 0;

}
