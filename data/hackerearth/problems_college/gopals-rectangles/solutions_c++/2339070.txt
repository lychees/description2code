#include <iostream>
using namespace std;
#define ll long long 

int main()
{
	ll t;
	cin >> t;
	while(t--){
		ll n, m;
		cin >> n >> m;
		if(n < m){
			ll t = m;
			m = n;
			n = t;
		}
		
		// 4 6
		cout << m*(m+1)*(n)*(n+1)/4<< endl;
		n = m = 0;
	}
// cout << "Hello World!" << endl;
    
    return 0;
}
