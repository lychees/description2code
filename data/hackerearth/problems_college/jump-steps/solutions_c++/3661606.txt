#include <iostream>
#include <bits/stdc++.h>
#include <math.h>
using namespace std;
long long int bin(long long int a,long long int b)
{
	if(b==0)
	{
		return 1;
	}
	long long int x=bin(a,b/2);
	if(b%2==0)
	{
		return x*x;
	}
	else
	{
		return x*x*a;
	}
}
int main()
{
    int t;
    long long int n,cnt,i,j,max;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cnt=1;
    	max=log2(n);
    	i=bin(2,max);
    	j=i;
    	while(i<n)
    	{
    		j/=2;
    		while((i+j)>n)
    		{
    			j/=2;
    		}
    		i=i+j;
    		cnt++;
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
