int a[55][55];
int visited[51][51]={0},n;
bool flag=0;
void dfs(int r,int c)
{
	if(r==n-1&&c==n-1)
	{
		flag=1;
		return ;
	}
	visited[r][c]=1;
	if(visited[r][c+1]==0&&a[r][c+1]==1)
	{
		dfs(r,c+1);
	}
	if(visited[r+1][c]==0&&a[r+1][c]==1)
	{
		dfs(r+1,c);
	}

}
int main(int argc, char const *argv[])
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		flag=0;
		ini(visited,0);
	int i,j;
	scanf("%d",&n);
	for(i=0;i<n;++i)
	{
		for(j=0;j<n;++j)
		{
			scanf("%d",&a[i][j]);
		}
	}
	dfs(0,0);
	if(flag)
	{
		cout<<"POSSIBLE\n";
	}
	else
		cout<<"NOT POSSIBLE\n";
	}
	return 0;
}
