#include <bits/stdc++.h>
using namespace std;
const int N =1048576+5;
int a[N];
int dp[N][21];
int freq[N];
int solve(int x) {
    int ret = 0;
    for(int i=0;i<20;i++) {
        if(a[i]&(1<<i)) ret = ret;
        else ret += (1<<i);
    }
    return ret;
}
int main() {
    int t;
    scanf("%d",&t);
    while(t--) {
        memset(freq,0,sizeof freq);
        int n;
        scanf("%d",&n);
        for(int i=0;i<n;i++) {
        scanf("%d",&a[i]);
        freq[a[i]]++;
        }

        long long ans = 0;

        for(int mask = 0; mask<(1<<20);mask++) {

            dp[mask][0] = freq[mask];
            if(mask&1) dp[mask][0] +=freq[mask^1];
            for(int p = 1;p<=20;p++) {
                dp[mask][p] = dp[mask][p-1];
                if(mask&(1<<p)) dp[mask][p] += dp[mask^(1<<p)][p-1];
            }
        }
        for(int i=0;i<n;i++) {
            ans += dp[(1<<20)-1-a[i]][20];
        }
        printf("%lld\n",ans);
    }
}
