#include<iostream>
using namespace std;
  long long  ar[1000001];
void tot()
{
    ar[1] = 0;
     for(int i = 2 ; i <= 1000000; ++i ){

          int ans = i;
          int n = i;
           if(n%2==0)
            {
                ans = ans - ans/2;
            while(n%2==0)
                n=n/2;
           }
           for(int j = 3 ; j*j <= n ; j=j+2)
           {
                 if(n%j==0)
                 {
                       ans = ans - ans/j;
                    while(n%j==0)
                        n=n/j;
                 }
           }
             if(n>1)
                 ans = ans - ans/n;
             ar[i] = ans;
        ar[i]+=ar[i-1];

     }



    }
int main()
{
      tot();
       int t;
       cin >> t;
       while(t--){
            long long n;
            cin >> n;
            long long val = n*(n-1)/2;
               cout << (val - ar[n] ) << endl;


       }
}
