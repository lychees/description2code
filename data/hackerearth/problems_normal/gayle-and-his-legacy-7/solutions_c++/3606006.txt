#include <bits/stdc++.h>
using namespace std;
#define ll long long 
//bool greater(int i,int j){return i>j;}
int main() {
    int t;
    cin>>t;
    while(t--){
    	ll int n,m;
    	cin>>n>>m;
    	pair<ll int,ll int>pairedGainAndPower[n];
    	for(int i =0;i<n;i++) {
    		ll int p,g;
    		cin>>g>>p;
    		pairedGainAndPower[i].first = -p;
    		pairedGainAndPower[i].second = g;
    	}
    	sort(pairedGainAndPower,pairedGainAndPower+n ,greater<pair<ll int,ll int> >() );
    	//int totalPower = m;
    	bool flag = 0;
    	for(int i = 0;i<n;i++) {
    		
    		ll int p = -pairedGainAndPower[i].first;
    		ll int g = pairedGainAndPower[i].second;
    		if(m > p) {
    			m += g;
    		}else{
    			cout<<"NO"<<endl;
    			flag = 1;
    			break;
    		}
    	}
    	if(!flag){
    		cout<<"YES"<<endl;
    	}
    }
    return 0;
}
