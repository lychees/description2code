#include<bits/stdc++.h>
using namespace std;
string str;
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	cin>>str;
    	int i;
    	int num[10];
    	memset(num,-1,sizeof(num));
    	int mcnt = INT_MAX;
    	int cnt = INT_MAX;
    	int tmp;
    	for(i=0;i<str.size();++i)
    	{
    		tmp = str[i]-'0';
    		if(num[tmp] != -1) cnt = i-num[tmp];
    		mcnt = min(mcnt,cnt);
    		num[tmp] = i;
    	}
    	if(mcnt != INT_MAX) cout<<mcnt<<endl;
    	else cout<<-1<<endl;
    }
    return 0;
}
