#include <bits/stdc++.h>
using namespace std;
#define INF 10000000
int dp[115];

int rec(int n)
{
	if(n==0)
		return 0;
	if(n<0)
		return INF;
	if(dp[n]!=-1)
		return dp[n];
	dp[n] = 1+min(rec(n-12),rec(n-10));
	return dp[n];
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	memset(dp,-1,sizeof(dp));
    	int ans = rec(n);
    	if(ans >= INF)
    		cout<<"-1\n";
    	else
    		cout<<ans<<endl;
    }
    return 0;
}
