#include <iostream>
using namespace std;

int main()
{
	int T;
	bool cont;
	int n;
	int x, y;
	cin>>T;
	while(T--)
	{
		cin>>n;
		if(n<3)
			cout<<-1<<endl;
		else {
		x=n/3;
		x=x*3;
		cont=true;
		while(cont && x>=0)
		{
			y=n-x;
			if((y%5)==0)
				cont=false;
			else
				x=x-3;
		}
		while(x)
		{
			cout<<5;
			x--;
		}
		while(y)
		{
			cout<<3;
			y--;
		}
		cout<<endl;
		}
	}
	
	return 0;
}
